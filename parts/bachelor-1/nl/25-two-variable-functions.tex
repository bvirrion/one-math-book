\chapter{Functies van Twee Variabelen}\label{ch:b1:multivar}

Het jaar eindigt met een eerste wandeling in hogere dimensie: functies
$f(x, y)$ van twee reële variabelen. Alles veralgemeent --- limieten,
\omterm{def:b1:continuity:continuous}{continuïteit}, \omterm{def:b1:derivative:def}{afgeleiden}, extrema --- maar elk begrip krijgt een
wending: limieten kunnen tegelijk langs elke richting benaderd worden,
\omterm{def:b1:derivative:def}{afgeleiden} splitsen in \omterm{def:b1:multivar:partial}{partiële afgeleiden}, en de \omterm{def:b1:multivar:partial}{gradiënt} wijst de weg
bergop. De volledige theorie (differentialen, algemeen $\R^n$,
deelvariëteiten) hoort thuis in het tweede jaar; hier leggen we de
woordenschat en de eerste eerlijke stellingen vast.

\section{Het vlak $\R^2$; continuïteit}

\begin{definition}\label{def:b1:multivar:topology}
Op $\R^2$ gebruiken we de euclidische norm $\norm{(x,y)} = \sqrt{x^2 +
y^2}$ (\cref{ch:b1:euclid}). \omterm{def:b1:topology:open}{Open} ballen, \omterm{def:b1:topology:open}{omgevingen}, \emph{\omterm{def:b1:topology:open}{open}
delen} van $\R^2$ worden exact gedefinieerd zoals in
\cref{ch:b1:topology}, met ballen in plaats van intervallen. Een functie
$f \colon U \to \R$ ($U \subseteq \R^2$ \omterm{def:b1:topology:open}{open}) is \emph{\omterm{def:b1:continuity:continuous}{continu}
in} $a \in U$ wanneer
\[
\forall\varepsilon > 0,\ \exists\delta > 0, \quad
\norm{X - a} \leq \delta \implies \abs{f(X) - f(a)} \leq
\varepsilon,
\]
met dezelfde sequentiële karakterisering als in één variabele.
Sommen, producten, quotiënten en samenstellingen met continue
functies van één variabele behouden de \omterm{def:b1:continuity:continuous}{continuïteit}; de coördinaat-\-
\omterm{def:b1:logic:map}{afbeeldingen} zijn \omterm{def:b1:continuity:continuous}{continu}, dus ook de \omterm{def:b1:poly:def}{veeltermen} in $(x,y)$.
\end{definition}

\begin{example}[De poolgrens, de zuivere manier om een limiet te bewijzen]
\label{ex:b1:multivar:polarbound}
Toon aan dat $f(x, y) = \dfrac{x^2y^2}{x^2 + y^2}$ (met $f(0,0) =
0$) \omterm{def:b1:continuity:continuous}{continu} is in de oorsprong. In poolcoördinaten $x =
\rho\cos\theta$, $y = \rho\sin\theta$:
\[
\abs{f} = \frac{\rho^4\cos^2\theta\sin^2\theta}{\rho^2}
= \rho^2\,(\cos\theta\sin\theta)^2 \leq \frac{\rho^2}{4}
\xrightarrow[\rho \to 0]{} 0 ,
\]
een grens \emph{onafhankelijk van $\theta$}: welke richting de
benadering ook heeft, de waarden worden naar $0$ geklemd. Die
uniformiteit in $\theta$ is de hele clou --- een grens als $\abs g =
\abs{\cos\theta\sin\theta}$ (geen $\rho$ over) bewijst niets,
en die $g$ is inderdaad de discontinue radiale val van het
volgende voorbeeld.
\end{example}

\begin{example}[De radiale val]\label{ex:b1:multivar:trap}
Zij $f(x, y) = \dfrac{xy}{x^2 + y^2}$ voor $(x,y) \neq (0,0)$, $f(0,
0) = 0$. Langs elke as is $f = 0 \to 0$; maar langs de diagonaal $y
= x$ is $f(x, x) = \frac12 \not\to 0$. \emph{Geen limiet in de
oorsprong}: benaderen langs elke rechte, en zelfs telkens dezelfde
limiet vinden, volstaat niet (hier verschillen de limieten langs de
rechten; ergere voorbeelden stemmen overeen langs alle rechten en
falen toch langs een parabool,
\cref{exo:b1:multivar:3}). \omterm{def:b1:continuity:continuous}{Continuïteit} in elke variabele afzonderlijk
impliceert geen \omterm{def:b1:continuity:continuous}{continuïteit}.
\end{example}

\section{Partiële afgeleiden}

\begin{definition}\label{def:b1:multivar:partial}
De \emph{partiële afgeleiden}\index{partiële afgeleide} van $f$ in
$(a, b)$ zijn de \omterm{def:b1:derivative:def}{afgeleiden} in één variabele langs de assen:
\[
\frac{\partial f}{\partial x}(a,b)
= \lim_{h \to 0} \frac{f(a + h, b) - f(a,b)}{h},
\qquad
\frac{\partial f}{\partial y}(a,b)
= \lim_{k \to 0} \frac{f(a, b + k) - f(a,b)}{k}.
\]
$f$ is van klasse $C^1$ op $U$ wanneer beide bestaan en \omterm{def:b1:continuity:continuous}{continu} zijn op
$U$. De \emph{gradiënt}\index{gradiënt} is $\nabla f(a,b) =
\bigl(\frac{\partial f}{\partial x},\, \frac{\partial f}{\partial
y}\bigr)(a,b)$.
\end{definition}

\begin{theorem}[$C^1$ impliceert een raakvlak]\label{thm:b1:multivar:c1}
Zij $f$ van klasse $C^1$ op $U$ en $(a,b) \in U$. Dan geldt, als $(h, k) \to
(0,0)$:
\[
f(a + h, b + k) = f(a, b)
+ h\,\frac{\partial f}{\partial x}(a,b)
+ k\,\frac{\partial f}{\partial y}(a,b)
+ o\bigl(\norm{(h,k)}\bigr) .
\]
In het bijzonder is $f$ \omterm{def:b1:continuity:continuous}{continu}, en de grafiek $z = f(x,y)$ heeft in
elk punt het raakvlak dat rechtstreeks uit de formule af te lezen is.
\end{theorem}

\begin{example}[Lineaire benadering in actie]
\label{ex:b1:multivar:linapprox}
Schat $f(1.02,\ 0.99)$ voor $f(x, y) = x^3y^2$. In $(1, 1)$:
$f = 1$, $\frac{\partial f}{\partial x} = 3x^2y^2 = 3$,
$\frac{\partial f}{\partial y} = 2x^3y = 2$, dus geeft
\cref{thm:b1:multivar:c1}
\[
f(1.02,\ 0.99) \approx 1 + 3\,(0.02) + 2\,(-0.01) = 1.04 ,
\]
tegenover de echte waarde $1.02^3 \times 0.99^2 = 1.04006\dots$ ---
de fout is van tweede orde in de aangroeiingen, zoals de
$o(\norm{(h,k)})$ belooft. Het raakvlak aan de grafiek in
$(1, 1, 1)$ is $z = 1 + 3(x - 1) + 2(y - 1)$, de vergelijking die
impliciet in de schatting zit.
\end{example}

\begin{proof}
Beweeg één coördinaat per keer:
\[
f(a+h, b+k) - f(a,b)
= \bigl[f(a+h, b+k) - f(a, b+k)\bigr] + \bigl[f(a, b+k) -
f(a,b)\bigr].
\]
Volgens de middelwaardestelling in één variabele
(\cref{thm:b1:derivative:mvt}) is de eerste term $h\,
\frac{\partial f}{\partial x}(a + \theta h,\, b + k)$ voor zekere
$\theta \in \intoo{0}{1}$, en de tweede is $k\,\frac{\partial
f}{\partial y}(a,\, b + \theta' k)$. De \omterm{def:b1:continuity:continuous}{continuïteit} van de \omterm{def:b1:multivar:partial}{partiële
afgeleiden} in $(a,b)$ laat ons elk schrijven als (waarde in $(a,b)$) $+$
(fout $\to 0$); de totale fout is $h\,\varepsilon_1 +
k\,\varepsilon_2 = o(\norm{(h,k)})$ aangezien $\abs h, \abs k \leq
\norm{(h,k)}$.
\end{proof}

\begin{theorem}[Kettingregel]\label{thm:b1:multivar:chain}
Zij $f$ van klasse $C^1$ op $U$ en $t \mapsto (x(t), y(t))$ van klasse $C^1$ van
een \omterm{prop:b1:reals:intervals}{interval} naar $U$. Dan is $g(t) = f\bigl(x(t), y(t)\bigr)$ van
klasse $C^1$, met\index{kettingregel!twee variabelen}
\[
g'(t) = x'(t)\,\frac{\partial f}{\partial x}\bigl(x(t),y(t)\bigr)
+ y'(t)\,\frac{\partial f}{\partial y}\bigl(x(t),y(t)\bigr)
= \bigl\langle \nabla f,\ (x', y')\bigr\rangle .
\]
\end{theorem}

\begin{proof}
Pas \cref{thm:b1:multivar:c1} toe in $(x(t), y(t))$ met $(h, k) =
(x(t+s) - x(t),\, y(t+s) - y(t))$: als $s \to 0$ geeft de
\omterm{def:b1:derivative:def}{afleidbaarheid} in één variabele $h = s\,x'(t) + o(s)$ en $k = s\,y'(t)
+ o(s)$, zodat $\norm{(h, k)} = O(s)$ en
\[
g(t+s) - g(t)
= h\,\frac{\partial f}{\partial x} + k\,\frac{\partial
f}{\partial y} + o\bigl(\norm{(h,k)}\bigr)
= s\,\Bigl(x'\,\frac{\partial f}{\partial x} +
y'\,\frac{\partial f}{\partial y}\Bigr) + o(s) ,
\]
waarbij de laatste fout zowel de beide $o(s)$'s van $h$ en $k$
opneemt (vermenigvuldigd met de vaste waarden van de \omterm{def:b1:multivar:partial}{partiële
afgeleiden}) als de $o(O(s))$ van de raakvlakschatting. Deel door $s$ en
laat $s \to 0$. De \omterm{def:b1:continuity:continuous}{continuïteit} van $g'$ volgt uit die van alle
ingrediënten.
\end{proof}

\begin{example}[De kettingregel, op beide manieren geverifieerd]
\label{ex:b1:multivar:chainrun}
Zij $f(x, y) = x^2 y$ en $g(t) = f(t, t^2)$. Rechtstreeks: $g(t) =
t^2\cdot t^2 = t^4$, dus $g'(t) = 4t^3$. Via de kettingregel:
$\frac{\partial f}{\partial x} = 2xy$ en $\frac{\partial
f}{\partial y} = x^2$, geëvalueerd langs de kromme $(t, t^2)$:
\[
g'(t) = 1\cdot 2t\cdot t^2 + 2t\cdot t^2 = 2t^3 + 2t^3 = 4t^3 .
\]
De twee berekeningen stemmen overeen, en de splitsing is betekenisvol: $2t^3$
van de groei komt van het \emph{naar rechts} bewegen doorheen de
$x$-helling, $2t^3$ van het \emph{naar boven} bewegen doorheen de
$y$-helling. Op krommen waar geen gesloten vorm voor $g$ bestaat, blijft alleen
de tweede berekening overeind --- dat is de clou van de stelling.
\end{example}

\begin{remark}[De gradiënt lezen]
Langs een eenheidsrichting $u$ geeft de kettingregel op $t \mapsto f(a + tu)$
de \emph{richtingsafgeleide} $\langle \nabla f(a),
u\rangle$: maximaal wanneer $u$ langs $\nabla f(a)$ wijst (volgens
Cauchy--Schwarz, \cref{thm:b1:euclid:cs}). De \omterm{def:b1:multivar:partial}{gradiënt} is de
richting van de steilste stijging, en hij staat orthogonaal op de
niveaukrommen $\{f = c\}$ (leid $f$ af langs een kromme getekend in een
niveauverzameling: de kettingregel geeft $\langle\nabla f,\,
\text{raaklijn}\rangle = 0$).
\end{remark}

\begin{example}[Niveaukrommen en gradiënten, op één functie]
\label{ex:b1:multivar:levelcurves}
Neem $f(x, y) = x^2 - y^2$. De niveauverzamelingen: $\{f = c\}$ is een
hyperbool die links-rechts opent voor $c > 0$, boven-onder voor $c < 0$,
en het gekruiste paar rechten $y = \pm x$ voor $c = 0$ --- de
hoogtekaart van een bergpas, met het \omterm{met:b1:multivar:monge}{zadelpunt} in de oorsprong
waar de twee nulniveaurechten kruisen. \omterm{def:b1:multivar:partial}{Gradiënt}: $\nabla
f = (2x, -2y)$. In het punt $(2, 1)$ (op het niveau $c = 3$):
$\nabla f = (4, -2)$, terwijl de raakvector van de niveau-\-
kromme, nabij dat punt geparametriseerd door $\bigl(t,
\sqrt{t^2 - 3}\bigr)$, gelijk is aan $\bigl(1, \frac{t}{\sqrt{t^2 -
3}}\bigr) = (1, 2)$ in $t = 2$ --- en inderdaad
\[
\langle (4, -2),\ (1, 2)\rangle = 4 - 4 = 0 :
\]
\omterm{def:b1:multivar:partial}{gradiënt} loodrecht op de hoogtelijn, wijzend naar hogere
waarden van $f$ (hier: weg van de $y$-as). Nog twee
lezingen: de \omterm{def:b1:multivar:partial}{gradiënt} verdwijnt precies in het \omterm{met:b1:multivar:monge}{zadelpunt}, waar
de hoogtekaart samenknijpt; en de \emph{raaklijn} aan de
niveaukromme in $(2,1)$ is $4(x - 2) - 2(y - 1) = 0$, d.w.z.\ $2x -
y = 3$ --- de vergelijking ``$\langle \nabla f,\ M - M_0\rangle =
0$'' die de ellipsraaklijn van
\cref{exo:b1:curves:11} veralgemeent.
\end{example}

\begin{theorem}[Schwarz]\label{thm:b1:multivar:schwarz}
Als $f$ van klasse $C^2$ is (\omterm{def:b1:multivar:partial}{partiële afgeleiden} van \omterm{def:b1:multivar:partial}{partiële afgeleiden}
bestaan en zijn \omterm{def:b1:continuity:continuous}{continu}), dan geldt\index{stelling van Schwarz}
\[
\frac{\partial^2 f}{\partial x\,\partial y}
= \frac{\partial^2 f}{\partial y\,\partial x} .
\]
\end{theorem}
\admitted

\section{Lokale extrema}

\begin{method}[Extremumstudies, geordend]
\label{met:b1:multivar:extremamethod}
\begin{enumerate}
  \item \emph{Los $\nabla f = 0$ volledig op.} Ontbind elke
        \omterm{def:b1:multivar:partial}{partiële afgeleide} waar mogelijk (producten van lineaire factoren
        splitsen het stelsel in doorzichtige gevallen, zoals in
        \cref{ex:b1:multivar:fourpoints} hieronder); een vergeten
        geval is een vergeten \omterm{thm:b1:multivar:critical}{kritiek punt}.
  \item \emph{Classificeer elk punt} met de Monge-gegevens $r, s,
        t$ --- telkens \emph{in elk punt} herberekend, nooit één keer voor
        alles.
  \item \emph{Als $rt - s^2 = 0$}, onderzoek $f$ rechtstreeks langs
        goedgekozen krommen door het punt (eerst rechten, dan
        parabolen), op zoek naar ofwel twee tekens (geen extremum)
        ofwel een vast teken met een argument dat alle
        richtingen dekt.
  \item \emph{Stap terug voor het globale plaatje}: controleer het
        gedrag in het oneindige (een lokaal minimum hoeft geen globaal
        te zijn), en als het domein niet \omterm{def:b1:topology:open}{open} is, behandel dan zijn \omterm{def:b1:topology:closure}{rand}
        apart (\cref{exo:b1:multivar:12}) --- de stelling over
        kritieke punten ziet alleen \omterm{def:b1:topology:closure}{inwendige} punten.
\end{enumerate}
\end{method}

\begin{theorem}[Kritieke punten]\label{thm:b1:multivar:critical}
Als $f$ ($C^1$ op de \omterm{def:b1:topology:open}{open verzameling} $U$) een lokaal extremum heeft in $(a,b)
\in U$, dan is $\nabla f(a,b) = (0,0)$: het punt is
\emph{kritiek}\index{kritiek punt}.
\end{theorem}

\begin{proof}
De functies in één variabele $x \mapsto f(x, b)$ en $y \mapsto f(a,
y)$ hebben \omterm{def:b1:topology:closure}{inwendige} lokale extrema in $a$, resp.\ $b$:
\cref{prop:b1:derivative:fermat} maakt beide \omterm{def:b1:multivar:partial}{partiële afgeleiden} nul.
\end{proof}

\begin{method}[Test van tweede orde (Monge-notatie)]\label{met:b1:multivar:monge}
In een \omterm{thm:b1:multivar:critical}{kritiek punt} van een $C^2$-functie stel je
\[
r = \frac{\partial^2 f}{\partial x^2},
\qquad
s = \frac{\partial^2 f}{\partial x \partial y},
\qquad
t = \frac{\partial^2 f}{\partial y^2}
\qquad (\text{waarden in het punt}).
\]
\begin{itemize}
  \item Als $rt - s^2 > 0$: lokaal extremum --- minimum voor $r > 0$,
        maximum voor $r < 0$;
  \item als $rt - s^2 < 0$: geen extremum (een \emph{zadel\-
        punt}\index{zadelpunt});
  \item als $rt - s^2 = 0$: de test zwijgt; onderzoek rechtstreeks.
\end{itemize}
(Rechtvaardiging --- een Taylor--Young-ontwikkeling van orde $2$ en de
tekenstudie van de kwadratische vorm $r h^2 + 2shk + tk^2$ --- wordt
uitgevoerd in het tweede jaar; de test wordt hier als werkinstrument
gebruikt.)
\end{method}

\begin{example}\label{ex:b1:multivar:extrema}
$f(x,y) = x^3 + y^3 - 3xy$. Kritieke punten: $\nabla f = (3x^2 -
3y,\; 3y^2 - 3x) = 0$ geeft $y = x^2$ en $x = y^2$, dus $x = x^4$:
$x \in \{0, 1\}$: punten $(0,0)$ en $(1,1)$.

Tweede \omterm{def:b1:derivative:def}{afgeleiden}: $r = 6x$, $s = -3$, $t = 6y$. In $(0,0)$: $rt -
s^2 = -9 < 0$: zadel. In $(1,1)$: $rt - s^2 = 36 - 9 > 0$, $r = 6
> 0$: lokaal minimum, $f(1,1) = -1$. (Niet globaal: $f(x, 0) = x^3 \to
-\infty$.)
\end{example}

\begin{example}[Een studie in vier punten, volledig]
\label{ex:b1:multivar:fourpoints}
$f(x, y) = xy\,(3 - x - y) = 3xy - x^2y - xy^2$. \omterm{def:b1:multivar:partial}{Gradiënt}:
\[
\frac{\partial f}{\partial x} = y\,(3 - 2x - y),
\qquad
\frac{\partial f}{\partial y} = x\,(3 - x - 2y).
\]
Kritieke punten: als $y = 0$, geeft de tweede vergelijking $x \in
\{0, 3\}$; als $x = 0$, geeft de eerste $y \in \{0, 3\}$; als $xy
\neq 0$, los $2x + y = 3$, $x + 2y = 3$ op: $x = y = 1$. Vier
punten: $(0,0)$, $(3,0)$, $(0,3)$, $(1,1)$. Tweede \omterm{def:b1:derivative:def}{afgeleiden}:
$r = -2y$, $s = 3 - 2x - 2y$, $t = -2x$.
\begin{itemize}
  \item $(1,1)$: $r = -2$, $s = -1$, $t = -2$: $rt - s^2 = 3 >
        0$, $r < 0$: lokaal maximum, $f(1,1) = 1$.
  \item $(0,0)$: $r = t = 0$, $s = 3$: $rt - s^2 = -9 < 0$:
        zadel; evenzo $(3, 0)$ ($s = -3$) en $(0, 3)$: drie
        zadels.
\end{itemize}
Het maximum is slechts lokaal: $f(-T, -T) = T^2(3 + 2T) \to
+\infty$. Symmetriecontrole: $f(x, y) = f(y, x)$, en inderdaad zijn de
kritieke \omterm{def:b1:logic:sets}{verzameling} en de classificatie symmetrisch in $x
\leftrightarrow y$. Interpretatie: onder de rechthoeken-met-speling
$x, y \geq 0$, $x + y \leq 3$, is het product $xy(3 - x - y)$ van
de drie ``delen'' van $3$ het grootst wanneer de delen gelijk zijn
--- een tweevariabele-schaduw van het RM--MK-principe.
\end{example}

\begin{remark}[Veelvoorkomende valkuilen]
\label{rem:b1:multivar:pitfalls}
\emph{\omterm{def:b1:multivar:partial}{Partiële afgeleiden} kunnen bestaan in een punt van
discontinuïteit}: de radiale val $g(x,y) = \frac{xy}{x^2+y^2}$ van
\cref{ex:b1:multivar:trap} heeft $\frac{\partial g}{\partial
x}(0,0) = \frac{\partial g}{\partial y}(0,0) = 0$ (beide as-\-
beperkingen zijn identiek nul), en toch heeft $g$ geen limiet in de
oorsprong --- \omterm{def:b1:multivar:partial}{partiële afgeleiden} peilen slechts twee richtingen, \omterm{def:b1:continuity:continuous}{continuïteit}
heeft ze alle nodig; alleen de $C^1$-hypothese herstelt de orde
(\cref{thm:b1:multivar:c1}).
\emph{Limieten langs rechten volstaan nooit}: de functie van
\cref{exo:b1:multivar:3} heeft limiet $0$ langs elke rechte en toch geen
limiet --- probeer altijd parabolen (of poolgrenzen uniform geldig in
$\theta$).
\emph{\omterm{thm:b1:multivar:critical}{Kritiek} is nodig, niet voldoende}: zadels zijn er in overvloed
(drie van de vier punten in \cref{ex:b1:multivar:fourpoints}); en
de stelling geldt alleen op \emph{\omterm{def:b1:topology:open}{open}} \omterm{def:b1:logic:sets}{verzamelingen} --- op een \omterm{def:b1:topology:closed}{gesloten} schijf
kunnen de extrema op de \omterm{def:b1:topology:closure}{rand} liggen met een \omterm{def:b1:multivar:partial}{gradiënt} die niet nul is
(\cref{exo:b1:multivar:12}).
\emph{Het stille geval $rt - s^2 = 0$ is echt stil}: $x^4
+ y^4$ (minimum) en $x^3 + y^3$ (geen van beide) hebben allebei $r = s = t
= 0$ in de oorsprong; alleen een rechtstreekse tekenstudie beslist
(\cref{exo:b1:multivar:6}, functie $h$).
\emph{De \omterm{def:b1:multivar:partial}{gradiënt} staat orthogonaal op de niveaukrommen, niet er\-
langs}: om een hoogtelijn te volgen, beweeg loodrecht op $\nabla
f$; om het snelst te klimmen, beweeg er langs --- de twee door
elkaar halen keert de geometrie van elke hoogtekaart om.
\end{remark}

\begin{remark}[Waarheen twee variabelen leiden]
\label{rem:b1:multivar:whereused}
Dit hoofdstuk is een deur. De \omterm{def:b1:multivar:partial}{gradiënt} en de kettingregel breiden zich
letterlijk uit naar $n$ variabelen in het volume van Jaar~2, waar de
$o(\norm{(h,k)})$ van \cref{thm:b1:multivar:c1} de
\emph{differentiaal} wordt en de Monge-test volledig bewezen wordt via de
Taylor-formule van orde twee en kwadratische vormen. Het speciale geval
dat \emph{dit} jaar afgehandeld kan worden --- kwadratische functies,
waarvoor de ontwikkeling van tweede orde exact is --- is het onderwerp
van het weekendprobleem, en het blijkt het geval te zijn dat het
gegevensfitten van de hele wereld aandrijft: kleinste-kwadraten-regressie.
Extrema onder voorwaarden (\cref{exo:b1:multivar:5} was een voorproefje)
worden Lagrange-multiplicatoren in Jaar~2; harmonische functies
(\cref{exo:b1:multivar:7}) keren terug in de complexe analyse van het
volume van Jaar~3.
\end{remark}

\begin{remark}[Perspectieven binnen Boek 3: het jaar, afgesloten]
\label{rem:b1:multivar:perspectives}
Dit hoofdstuk is waar de twee helften van het volume elkaar de hand
schudden. De analysehelft leverde haar werktuigen één \omterm{def:b1:derivative:def}{afgeleide} per keer: de
middelwaardestelling drijft \cref{thm:b1:multivar:c1} aan, Taylor-\-
ontwikkelingen drijven de extremumtests aan, en de $\varepsilon$'s van
\cref{ch:b1:topology} kwamen terug met ballen in plaats van
intervallen. De algebrahelft leverde de geometrie: de \omterm{def:b1:multivar:partial}{gradiënt}
wordt gelezen via het \omterm{def:b1:euclid:def}{inwendig product} van \cref{ch:b1:euclid}
(Cauchy--Schwarz maakt hem tot de steilste richting), de Monge-\-
gegevens $(r, s, t)$ zijn een symmetrische matrix van
\cref{ch:b1:matrices} met de determinanttest van
\cref{ch:b1:det}, en het weekendprobleem voert \omterm{thm:b1:euclid:projection}{orthogonale
projectie} op gegevensvectoren uit. Zelfs de krommen van
\cref{ch:b1:curves} keren terug als niveauverzamelingen. Een lezer die
kan reconstrueren waarom elk van deze vijf overdrachten werkt, heeft in
feite het hele jaar herhaald --- wat het echte doel is
van dit slothoofdstuk.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b1:multivar:1}
Bereken de \omterm{def:b1:multivar:partial}{partiële afgeleiden}:
$f(x,y) = x^2 y + \eu^{xy}$; $\;g(x,y) = \ln(x^2 + y^2)$ (op
$\R^2\setminus\{0\}$); $\;h(x,y) = \arctan\frac yx$ (op $x > 0$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:multivar:2}
Bestudeer de \omterm{def:b1:continuity:continuous}{continuïteit} in $(0,0)$ (met waarde $0$ daar) van:
\[
f(x,y) = \frac{x^2 y}{x^2 + y^2},
\qquad
g(x,y) = \frac{xy}{x^2 + y^2},
\qquad
h(x,y) = \frac{x^3 + y^3}{x^2 + y^2}.
\]
\emph{(Poolcoördinaten $x = \rho\cos\theta$, $y = \rho\sin\theta$
helpen: begrens waar mogelijk door een functie van $\rho$ alleen.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:3}
Zij $f(x,y) = \dfrac{x^2 y}{x^4 + y^2}$ ($f(0,0) = 0$). Bewijs dat
$f$ limiet $0$ heeft in de oorsprong \emph{langs elke rechte lijn},
maar dat $f\bigl(x, x^2\bigr) = \frac12$: $f$ is niet \omterm{def:b1:continuity:continuous}{continu} in
$(0,0)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:multivar:4}
Verifieer de stelling van Schwarz met de hand op $f(x, y) = x^3 y^2 + \sin(xy)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:5}
Zij $f$ van klasse $C^1$ op $\R^2$ en $g(t) = f(\cos t, \sin t)$. Druk
$g'(t)$ uit via de kettingregel. Leid af dat $f$ beperkt tot de
eenheidscirkel extrema bereikt in punten waar $\nabla f$ evenwijdig is met
de straalvector.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:6}
Zoek en classificeer de kritieke punten van:
\[
f(x, y) = x^2 + xy + y^2 - 3x,
\qquad
g(x, y) = x^2 - y^2 + 4y,
\qquad
h(x, y) = x^4 + y^4 - 2(x - y)^2 .
\]
\emph{(Voor $h$ zwijgt de determinanttest in de oorsprong:
onderzoek $h(x, x)$ en $h(x, -x)$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:7}
Een functie $f$ is \emph{harmonisch} wanneer $\frac{\partial^2 f}{\partial
x^2} + \frac{\partial^2 f}{\partial y^2} = 0$. Ga na dat $x^2 -
y^2$, $xy$, $\eu^x\cos y$ en $\ln(x^2 + y^2)$ (buiten de oorsprong)
harmonisch zijn.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:multivar:8}
Zoek het punt van het vlak $\{x + 2y - z = 4\} \subseteq \R^3$
dat het dichtst bij de oorsprong ligt, op twee manieren: via \omterm{thm:b1:euclid:projection}{orthogonale projectie}
(\cref{ch:b1:euclid}), en door het minimaliseren van de tweevariabele-functie
$f(x, y) = x^2 + y^2 + (x + 2y - 4)^2$ verkregen door $z$ te elimineren.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:multivar:9}
(Laplaciaan in poolcoördinaten, eerste contact) Zij $f$ van klasse $C^2$ op
$\R^2 \setminus \{0\}$ en \emph{radiaal}: $f(x, y) =
\varphi\bigl(\sqrt{x^2+y^2}\bigr)$ met $\varphi$ van klasse $C^2$ op
$\intoo{0}{+\infty}$. Bewijs dat
\[
\frac{\partial^2 f}{\partial x^2} + \frac{\partial^2 f}{\partial
y^2} = \varphi''(\rho) + \frac{\varphi'(\rho)}{\rho},
\qquad \rho = \sqrt{x^2 + y^2},
\]
en zoek alle radiale harmonische functies op het geperforeerde vlak.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:10}
Geef de vergelijking van het raakvlak aan de grafiek $z = xy$ in
het punt $(1, 1, 1)$. Zoek dan alle punten van de grafiek van
$f(x,y) = x^3 + y^3 - 3xy$ waar het raakvlak
\emph{horizontaal} is, en breng het antwoord in verband met
\cref{ex:b1:multivar:extrema}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:11}
Zij $f$ van klasse $C^1$ op $\R^2$.
\begin{enumerate}
  \item Als $\frac{\partial f}{\partial x} = 0$ overal, bewijs
        dan dat $f(x, y)$ alleen van $y$ afhangt.
  \item Zoek alle $C^1$-oplossingen van de vergelijking
        $\frac{\partial f}{\partial x} = \frac{\partial
        f}{\partial y}$ op $\R^2$. \emph{(Stel $g(u, v) =
        f\bigl(\frac{u+v}2, \frac{u-v}2\bigr)$ en bereken
        $\frac{\partial g}{\partial v}$ via de kettingregel.)}
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:multivar:12}
Zoek het globale maximum en minimum van $f(x, y) = xy$ op de
\omterm{def:b1:topology:closed}{gesloten} schijf $x^2 + y^2 \leq 1$. \emph{(Neem de tweevariabele-\-
extremumwaardenstelling aan: een continue functie op de \omterm{def:b1:topology:closed}{gesloten} schijf
bereikt haar grenzen --- bewezen in het volume van Jaar~2. Behandel de
\omterm{def:b1:topology:open}{open} schijf via kritieke punten en de randcirkel via de
parametrisering van \cref{exo:b1:multivar:5}.)}
\end{exercise}

\section{Probleem: kleinste kwadraten en de regressierechte}

\begin{problem}\label{pb:b1:multivar:1}
Gegeven $n$ gegevenspunten, welke rechte passeert het ``dichtst'' bij
al deze punten? Legendre en Gauss antwoordden: de rechte die de
som van de \emph{gekwadrateerde} verticale fouten minimaliseert --- omdat
die minimalisering exact oplosbaar is met lineaire algebra. Dit probleem
bewijst eerst de Monge-test van \cref{met:b1:multivar:monge}
eerlijk voor kwadratische functies (het ene geval waar de
tweede-orde-ontwikkeling exact is), en bouwt dan de normaal\-
vergelijkingen, de regressierechte en de correlatiecoëfficiënt
bovenop de euclidische geometrie van
\cref{ch:b1:euclid}.\index{kleinste kwadraten}\index{regressierechte}
\index{normaalvergelijkingen}\index{correlatie}

\medskip
\noindent\textbf{Deel I --- Kwadratische functies: de Monge-test,
bewezen.} Fixeer reële getallen $r, s, t$ en zij $q(h, k) = r h^2 + 2s hk +
t k^2$.
\begin{enumerate}
  \item Onderstel $r \neq 0$. Leid de kwadraatafsplitsingsvorm af
        \[
        q(h, k) = r\Bigl(h + \frac{s}{r}k\Bigr)^{\!2}
        + \frac{rt - s^2}{r}\,k^2 ,
        \]
        en leid af: als $rt - s^2 > 0$, heeft $q$ het strikte teken
        van $r$ buiten de oorsprong; als $rt - s^2 < 0$, neemt $q$ beide
        tekens aan.
  \item Behandel de resterende gevallen: $r = 0$, $t \neq 0$
        (symmetrisch); en $r = t = 0$ ($q = 2shk$). Besluit:
        $q$ neemt beide tekens aan desda $rt - s^2 < 0$, en $q$
        verdwijnt alleen in de oorsprong desda $rt - s^2 > 0$.
  \item Zij nu $f(x, y) = \frac12\bigl(r x^2 + 2s xy + t
        y^2\bigr) + \beta x + \gamma y + \delta$ een kwadratische
        functie met een \omterm{thm:b1:multivar:critical}{kritiek punt} $X_0 = (a, b)$. Bewijs de
        \emph{exacte} ontwikkeling
        \[
        f(X_0 + (h,k)) = f(X_0) + \tfrac12\,q(h, k)
        \qquad (\text{geen restterm}),
        \]
        en leid de Monge-classificatie voor kwadratische
        functies af: strikt globaal minimum als $rt - s^2 > 0$, $r >
        0$; strikt globaal maximum als $rt - s^2 > 0$, $r < 0$;
        zadel als $rt - s^2 < 0$.
  \item Onderstel $rt - s^2 > 0$ en $r > 0$ (dus ook $t > 0$).
        Bewijs de expliciete coërciviteitsgrens
        \[
        q(h, k) \geq c\,(h^2 + k^2),
        \qquad
        c = \frac{rt - s^2}{r + t} > 0 .
        \]
        \emph{(Toon aan dat de kwadratische vorm $q - c(h^2 + k^2)$,
        met coëfficiënten $r - c$, $s$, $t - c$, nog steeds
        niet-negatieve discriminanttestgegevens heeft.)}
  \item Leid af dat een kwadratische functie met positief definiet
        kwadratisch deel naar $+\infty$ neigt als $\norm{(x,y)} \to
        \infty$, en dus een \emph{unieke globale}
        minimaliseerder heeft: haar kritieke punt. Contrasteer met
        \cref{ex:b1:multivar:extrema}, waar een lokaal minimum van
        een niet-kwadratische functie niet globaal was.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De normaalvergelijkingen.} Zij $C_1,
C_2, b \in \R^n$ (canoniek \omterm{def:b1:euclid:def}{inwendig product}), en
\[
f(u, v) = \norm{u\,C_1 + v\,C_2 - b}^2 .
\]
\begin{enumerate}[resume]
  \item Werk $f$ uit en toon aan dat het een kwadratische functie is van $(u,
        v)$ met kwadratisch-deel-coëfficiënten $r = 2\norm{C_1}^2$,
        $s = 2\langle C_1, C_2\rangle$, $t = 2\norm{C_2}^2$.
  \item Toon aan dat $rt - s^2 > 0$ desda $(C_1, C_2)$
        \omterm{def:b1:vspaces:free}{vrij} is \emph{(het gelijkheidsgeval van Cauchy--Schwarz,
        \cref{thm:b1:euclid:cs})}; onderstel dit voortaan.
  \item Toon aan dat de kritieke vergelijkingen $\nabla f = 0$ de
        \emph{normaalvergelijkingen} zijn
        \[
        \begin{pmatrix}
        \norm{C_1}^2 & \langle C_1, C_2\rangle\\
        \langle C_1, C_2\rangle & \norm{C_2}^2
        \end{pmatrix}
        \begin{pmatrix} u\\ v\end{pmatrix}
        =
        \begin{pmatrix} \langle C_1, b\rangle\\ \langle C_2,
        b\rangle\end{pmatrix}
        \]
        --- de Gram-matrix van \cref{exo:b1:euclid:11} links ---
        en dat ze precies zeggen: $b - (uC_1 + vC_2)
        \perp C_1, C_2$. Besluit met Deel I en
        \cref{thm:b1:euclid:projection}: de unieke minimaliseerder
        geeft $p = uC_1 + vC_2 = $ de \omterm{thm:b1:euclid:projection}{orthogonale projectie} van
        $b$ op $\operatorname{Vect}(C_1, C_2)$.
  \item Toon aan dat de minimale waarde $\norm{b}^2 - \langle p,
        b\rangle = \norm{b - p}^2$ is, en teken het Pythagoras\-
        \omterm{def:b1:linmaps:kerim}{beeld}.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De regressierechte.} Gegevenspunten
$(x_1, y_1), \dots, (x_n, y_n)$, niet alle $x_i$ gelijk. Minimaliseer
\[
E(\alpha, \beta) = \sum_{i=1}^{n}\bigl(y_i - \alpha - \beta
x_i\bigr)^2 .
\]
Schrijf $\bar x = \frac1n\sum x_i$, $\bar y = \frac1n\sum y_i$,
$v_x = \frac1n\sum x_i^2 - \bar x^2$, $v_y = \frac1n\sum y_i^2 -
\bar y^2$, $c_{xy} = \frac1n\sum x_i y_i - \bar x\bar y$.
\begin{enumerate}[resume]
  \item Herken Deel II met $C_1 = (1, \dots, 1)$, $C_2 =
        (x_1, \dots, x_n)$, $b = (y_1, \dots, y_n)$, en schrijf
        de normaalvergelijkingen
        \[
        n\alpha + \Bigl(\sum x_i\Bigr)\beta = \sum y_i,
        \qquad
        \Bigl(\sum x_i\Bigr)\alpha + \Bigl(\sum
        x_i^2\Bigr)\beta = \sum x_i y_i .
        \]
  \item Los ze op:
        \[
        \beta = \frac{c_{xy}}{v_x},
        \qquad
        \alpha = \bar y - \beta\,\bar x ,
        \]
        en merk op dat de \emph{regressierechte} $y = \alpha
        + \beta x$ door het gemiddelde punt $(\bar x, \bar
        y)$ gaat.
  \item Controleer dat $v_x > 0$ precies wanneer de $x_i$ niet alle
        gelijk zijn, en breng dit in overeenstemming met vraag 7.
  \item Bewijs dat de minimale fout gelijk is aan
        \[
        E_{\min} = n\Bigl(v_y - \frac{c_{xy}^2}{v_x}\Bigr)
        = n\,v_y\,(1 - \rho^2),
        \qquad
        \rho = \frac{c_{xy}}{\sqrt{v_x v_y}}
        \quad (v_y \neq 0).
        \]
        Leid $\abs\rho \leq 1$ af, met $\abs\rho = 1$ desda
        de gegevens perfect uitgelijnd zijn.
  \item Uitgewerkt voorbeeld: voor de gegevens $(0,0)$, $(1,1)$, $(2,3)$,
        $(3,4)$, bereken $\bar x, \bar y, v_x, c_{xy}$, de
        regressierechte, de vier residu's en $E_{\min}$.
  \item Bewijs dat de residu's $\varepsilon_i = y_i - \alpha -
        \beta x_i$ van de optimale rechte voldoen aan $\sum_i
        \varepsilon_i = 0$ en $\sum_i x_i\varepsilon_i = 0$,
        en interpreteer beide via orthogonaliteit.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Variaties en toepassingen.}
\begin{enumerate}[resume]
  \item (Beste constante) Toon aan dat de constante $\alpha$
        die $\sum_i (y_i - \alpha)^2$ minimaliseert het gemiddelde $\bar
        y$ is, en dat de minimale waarde $n\,v_y$ is: de variantie
        meet het falen van de gegevens om constant te zijn.
  \item (Rechte door de oorsprong) Toon aan dat de richtingscoëfficiënt die
        $\sum_i (y_i - \beta x_i)^2$ minimaliseert $\beta_0 = \frac{\sum
        x_iy_i}{\sum x_i^2}$ is, en geef een voorwaarde op de gegevens
        opdat $\beta_0$ samenvalt met de richtingscoëfficiënt
        $c_{xy}/v_x$ van vraag 11.
  \item (Afstand tot een rechte, opnieuw) Voor een punt $M$ en de rechte
        $D$ door $P_0$ met richting de eenheidsvector $w$,
        minimaliseer $g(\tau) = \norm{P_0 + \tau w - M}^2$ en
        leid af $d(M, D)^2 = \norm{M - P_0}^2 - \langle M - P_0,
        w\rangle^2$; herwin de formule $d = \frac{\abs{a x_0 +
        b y_0 + c}}{\sqrt{a^2 + b^2}}$ voor de rechte $ax + by + c
        = 0$ in het vlak.
  \item (Variantieanalyse) Bewijs de decompositie $v_y =
        \beta^2 v_x + \frac{E_{\min}}n$: de variantie van de
        $y_i$ splitst in het deel verklaard door de rechte plus
        de residuele variantie.
  \item (Parabolische fit) Om $y = a + bx + cx^2$ te fitten, toon aan dat
        de normaalvergelijkingen het $3 \times 3$-stelsel zijn met
        de momentmatrix
        \[
        \begin{pmatrix}
        n & \sum x_i & \sum x_i^2\\
        \sum x_i & \sum x_i^2 & \sum x_i^3\\
        \sum x_i^2 & \sum x_i^3 & \sum x_i^4
        \end{pmatrix},
        \]
        een Gram-matrix die inverteerbaar is zodra drie van
        de $x_i$ verschillend zijn \emph{(vrijheid van $(1, X, X^2)$
        bemonsterd in de gegevens, \cref{exo:b1:euclid:11}; vergelijk
        de momentmatrices van het weekendprobleem van
        \cref{ch:b1:det})}.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Robuustheid, en synthese.}
\begin{enumerate}[resume]
  \item Classificeer de kritieke punten van de drie kwadratische
        functies
        \[
        x^2 + xy + y^2, \qquad x^2 + 3xy + y^2, \qquad
        x^2 + 2xy + y^2 ,
        \]
        via Deel I, en behandel het gedegenereerde derde geval met de hand
        (waar wordt het minimum bereikt?).
  \item (Uitschieterexperiment) Voeg het punt $(10, 0)$ toe aan de
        gegevens van vraag 14 en herbereken de richtingscoëfficiënt $\beta$.
        Wat gebeurde er, en waarom is de \emph{gekwadrateerde} fout zo
        gevoelig voor één ver punt?
  \item (Gewogen \omterm{pb:b1:multivar:1}{kleinste kwadraten}) Gegeven gewichten $w_i > 0$,
        minimaliseer $\sum_i w_i(y_i - \alpha - \beta x_i)^2$: toon
        aan dat de oplossing gegeven wordt door dezelfde formules als vraag
        11 met gewogen gemiddelden, variantie en covariantie
        (definieer ze).
  \item Toon aan dat voor de optimale rechte $E_{\min} = 0$
        dwingt dat alle punten er exact op liggen, en verbind met de
        gelijkheidsgeval $\abs\rho = 1$ van vraag 13; test op de
        uitgelijnde gegevens $(1,1), (2,3), (3,5)$.
  \item Synthese, in vier zinnen: waarom kwadratische functies
        de ene klasse zijn waar de tweede-orde-test van dit hoofdstuk
        geen aangenomen stelling nodig heeft; hoe de normaalvergelijkingen
        de analytische minimalisering identificeren met de \omterm{thm:b1:euclid:projection}{orthogonale
        projectie} van \cref{ch:b1:euclid}; wat de correlatie\-
        coëfficiënt meet en welke ongelijkheid hem begrenst;
        en welke stukken van Hoofdstukken 18--23 (bases, Gram- en
        momentmatrices, \omterm{def:b1:linmaps:projection}{projecties}) terug opdoken. Benoem de
        methode en de vergelijkingen bestudeerd in Delen II--III.
\end{enumerate}
\end{problem}
