\chapter{Getaltheorie in de gehele getallen}\label{ch:b1:arith}

Getaltheorie --- de studie van \omterm{def:b1:arith:divides}{deelbaarheid} in $\Z$ --- werd begonnen in
het Middelbaar-onderwijsvolume. Dit hoofdstuk bouwt haar volledig opnieuw
op vanaf de euclidische deling, met volledige bewijzen: \omterm{thm:b1:arith:gcd}{grootste gemene
deler} en het \omterm{met:b1:arith:euclid}{algoritme van Euclides}, de identiteit van BĂ©zout en het lemma
van Gauss, \omterm{thm:b1:arith:fta}{priemontbinding}, en de rekenkunde van \omterm{def:b1:arith:congruence}{congruenties} tot en met
de kleine stelling van Fermat. Naast haar eigen bekoring is dit materiaal
het model dat \cref{ch:b1:poly} nabootst voor veeltermen.

\section{Deelbaarheid en euclidische deling}

\begin{definition}[Deelbaarheid]\label{def:b1:arith:divides}
Voor $a, b \in \Z$ \emph{deelt} $b$ het getal $a$ (genoteerd $b \mid
a$)\index{deelbaarheid} wanneer $a = bq$ voor een zekere $q \in \Z$.
Basisgevolgen: als $b \mid a$ en $b \mid a'$ dan $b \mid (ua + va')$
voor alle $u, v \in \Z$; als $b \mid a$ en $a \neq 0$ dan $\abs b \leq
\abs a$; en $a \mid b$ samen met $b \mid a$ dwingen $b = \pm a$ af.
\end{definition}

\begin{theorem}[Euclidische deling]\label{thm:b1:arith:division}
Voor alle $a \in \Z$ en $b \in \N^*$ bestaat er precies Ă©Ă©n paar $(q, r)
\in \Z \times \N$ met
\[
a = bq + r, \qquad 0 \leq r < b .
\]
\end{theorem}

\begin{proof}
\emph{Bestaan.} De \omterm{def:b1:logic:sets}{verzameling} $A = \{a - bk : k \in \Z\} \cap \N$ is een
niet-lege deelverzameling van $\N$ (neem $k = -\abs a$: $a + b\abs a \geq a +
\abs a \geq 0$). Zij $r = a - bq$ haar kleinste element. Als $r \geq b$,
dan zou $r - b = a - b(q+1)$ een kleiner element van $A$ zijn:
tegenspraak. Dus $0 \leq r < b$.

\emph{Uniciteit.} Als $bq + r = bq' + r'$ met $0 \leq r, r' < b$,
dan $b(q - q') = r' - r$ en $\abs{r' - r} < b$: het veelvoud van $b$
in het linkerlid moet $0$ zijn, dus $q = q'$ en $r = r'$.
\end{proof}

\begin{example}[Positionele getalnotatie door herhaalde deling]\label{ex:b1:arith:base7}
Schrijf $2026$ in grondtal $7$. Deel herhaaldelijk door $7$, en houd de
resten bij:
\[
2026 = 7 \times 289 + 3, \quad
289 = 7 \times 41 + 2, \quad
41 = 7 \times 5 + 6, \quad
5 = 7 \times 0 + 5 .
\]
Lees de resten van laatste naar eerste: $2026 =
(5\,6\,2\,3)_7$. Controle: $5 \times 343 + 6 \times 49 + 2 \times 7
+ 3 = 1715 + 294 + 14 + 3 = 2026$. De uniciteit van de euclidische
deling is precies wat elk cijfer \emph{afdwingt}: bij elke
stap is de rest het enige geheel getal in $\intint06$ congruent
met de huidige waarde mod $7$, dus de schrijfwijze in grondtal $7$ is uniek ---
het feit dat stilzwijgend gebruikt wordt telkens wanneer het weekendprobleem
``de cijfers van $n$ in grondtal $p$'' manipuleert.
\end{example}

\section{Grootste gemene deler}

\begin{theorem}[Deelgroepen van $\Z$; bestaan van de ggd]\label{thm:b1:arith:gcd}
\begin{enumerate}
  \item Elke deelgroep van $(\Z, +)$ is van de vorm $n\Z = \{nk : k
        \in \Z\}$ voor een unieke $n \in \N$.
  \item Voor $a, b \in \Z$ niet beide nul is de \omterm{def:b1:logic:sets}{verzameling} $a\Z + b\Z = \{au +
        bv : u, v \in \Z\}$ een deelgroep van $\Z$, en dus gelijk aan
        $d\,\Z$ voor een unieke $d \in \N^*$. Deze $d$ is de
        \emph{grootste gemene deler}\index{grootste gemene deler}
        $\gcd(a, b)$: ze \omterm{def:b1:arith:divides}{deelt} $a$ en $b$, en elke gemene
        deler van $a$ en $b$ \omterm{def:b1:arith:divides}{deelt} $d$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Zij $H \subseteq \Z$ een deelgroep (niet-leeg, stabiel onder
aftrekking; de formele definitie staat in \cref{ch:b1:structures},
en enkel deze twee eigenschappen worden gebruikt). Als $H = \{0\}$, neem $n = 0$.
Anders bevat $H$ een niet-nul element en zijn tegengestelde, dus een
kleinste strikt positief element $n$. Dan $n\Z \subseteq H$. Voor $x
\in H$, schrijf $x = nq + r$ met $0 \leq r < n$
(\cref{thm:b1:arith:division}); $r = x - nq \in H$, en de minimaliteit van
$n$ dwingt $r = 0$ af: $x \in n\Z$. Uniciteit: $n$ is het kleinste positieve
element van $n\Z$.

(2) $a\Z + b\Z$ bevat $0$ en is stabiel onder aftrekking, dus het is
$d\Z$ met $d \geq 1$ (het bevat $a$ of $b$ niet-nul). Aangezien $a, b
\in d\Z$, \omterm{def:b1:arith:divides}{deelt} $d$ beide. En als $c$ zowel $a$ als $b$ \omterm{def:b1:arith:divides}{deelt}, dan \omterm{def:b1:arith:divides}{deelt} $c$
elke $au + bv$ --- in het bijzonder $c \mid d$, aangezien $d \in a\Z
+ b\Z$. Dit is de aangekondigde eigenschap (en ze impliceert $\abs c \leq
d$, zodat $d$ de naam \emph{grootste} gemene deler verdient).
\end{proof}

\begin{corollary}[Identiteit van BĂ©zout]\label{cor:b1:arith:bezout}
Voor $a, b$ niet beide nul bestaan er $u, v \in \Z$ met
\[
au + bv = \gcd(a, b) .
\]
In het bijzonder ($\gcd(a,b) = 1$, het \emph{relatief priem}\index{relatief priem}
geval): $a$ en $b$ zijn \omterm{cor:b1:arith:bezout}{relatief priem} als en slechts als $au + bv = 1$ een
oplossing heeft.
\end{corollary}

\begin{proof}
$\gcd(a,b) = d \in d\Z = a\Z + b\Z$. Voor de equivalentie: als
$\gcd(a,b) = 1$, levert BĂ©zout de oplossing; omgekeerd dwingt $au + bv =
1$ elke gemene deler van $a, b$ af om $1$ te delen.
\end{proof}

\begin{method}[Algoritme van Euclides, uitgebreid]\label{met:b1:arith:euclid}
Om $\gcd(a, b)$ te berekenen ($a > b > 0$): deel $a = bq + r$; dan is
$\gcd(a, b) = \gcd(b, r)$ (de gemene delers van $(a,b)$ en van $(b,r)$
vallen samen, aangezien $r = a - bq$); herhaal totdat de rest $0$ is; de
laatste niet-nul rest is de ggd\index{algoritme van Euclides}. Door de
delingen achterwaarts te doorlopen (of de coĂ«fficiĂ«nten bij te houden op de weg
naar beneden) verkrijg je een BĂ©zout-paar $(u, v)$.
\end{method}

\begin{example}\label{ex:b1:arith:euclid}
$\gcd(120, 23)$: $120 = 5 \times 23 + 5$; $23 = 4 \times 5 + 3$; $5 =
1\times 3 + 2$; $3 = 1 \times 2 + 1$; $2 = 2 \times 1 + 0$. Dus
$\gcd = 1$. Achterwaarts:
\begin{align*}
1 &= 3 - 2 = 3 - (5 - 3) = 2\times 3 - 5 = 2(23 - 4\times 5) - 5 \\
&= 2 \times 23 - 9 \times 5 = 2\times 23 - 9(120 - 5\times 23)
= 47 \times 23 - 9 \times 120 .
\end{align*}
Controle: $47 \times 23 = 1081$, $9 \times 120 = 1080$.
\end{example}

\begin{theorem}[Lemma van Gauss en gevolgen]\label{thm:b1:arith:gauss}
Zij $a, b, c \in \Z$.
\begin{enumerate}
  \item (Lemma van Gauss\index{lemma van Gauss}) Als $a \mid bc$ en
        $\gcd(a, b) = 1$, dan $a \mid c$.
  \item Als $a \mid c$, $b \mid c$ en $\gcd(a,b) = 1$, dan $ab \mid
        c$.
  \item Als $\gcd(a, b) = \gcd(a, c) = 1$, dan $\gcd(a, bc) = 1$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) BĂ©zout: $au + bv = 1$. Vermenigvuldig met $c$: $acu + bcv = c$. Beide
termen zijn \omterm{def:b1:arith:divides}{deelbaar} door $a$ (de tweede omdat $a \mid bc$), dus $a
\mid c$.

(2) Schrijf $c = aq$; uit $b \mid aq$ en $\gcd(a, b) = 1$ geeft punt (1)
$b \mid q$, dus $ab \mid aq = c$.

(3) $au + bv = 1$ en $au' + cv' = 1$. Vermenigvuldig de twee
betrekkingen:
\[
1 = (au + bv)(au' + cv')
= a\,\bigl(auu' + ucv' + u'bv\bigr) + bc\,(vv') ,
\]
een BĂ©zout-betrekking tussen $a$ en $bc$: volgens
\cref{cor:b1:arith:bezout} is $\gcd(a, bc) = 1$.
\end{proof}

\begin{example}[Een lineaire diophantische vergelijking oplossen]\label{ex:b1:arith:diophantine}
Vind alle $(x, y) \in \Z^2$ met $6x + 10y = 4$. Eerst de
\emph{bestaanstoets}: $\gcd(6, 10) = 2$ \omterm{def:b1:arith:divides}{deelt} $4$, dus er bestaan
oplossingen (als de ggd het rechterlid niet zou delen, zou het linkerlid
altijd een veelvoud ervan zijn en zouden er geen zijn). Deel
door: $3x + 5y = 2$. Een particuliere oplossing is zichtbaar: $(x_0,
y_0) = (-1, 1)$. Voor de algemene, trek af: $3(x + 1) = -5(y -
1)$, dus $3 \mid 5(y-1)$, en het lemma van Gauss ($\gcd(3,5) = 1$) geeft
$3 \mid y - 1$: $y = 1 - 3k$, en dan $x = -1 + 5k$. Omgekeerd werkt elk
zulk paar:
\[
(x, y) = (-1 + 5k,\ 1 - 3k), \qquad k \in \Z .
\]
Het patroon is algemeen: Ă©Ă©n particuliere oplossing plus de gehele
veelvouden van $\bigl(\frac b{\gcd}, -\frac a{\gcd}\bigr)$ --- dezelfde
``particulier plus homogeen''-structuur als in
\cref{ch:b1:diffeq}, met het lemma van Gauss in de rol van de uniciteit.
\end{example}

\begin{definition}[Kleinste gemene veelvoud]\label{def:b1:arith:lcm}
$\operatorname{lcm}(a, b)$\index{kleinste gemene veelvoud} is de
voortbrenger in $\N$ van de deelgroep $a\Z \cap b\Z$: het is een gemeen
veelvoud van $a$ en $b$ dat elk gemeen veelvoud \omterm{def:b1:arith:divides}{deelt}, en voor
$a, b \in \N^*$,
\[
\gcd(a,b) \times \operatorname{lcm}(a,b) = ab
\qquad (\text{bewijs in } \cref{exo:b1:arith:5}).
\]
\end{definition}

\begin{example}[Uitlijningsproblemen zijn kgv-problemen]\label{ex:b1:arith:gears}
Twee in elkaar grijpende tandwielen hebben $84$ en $36$ tanden.
Na hoeveel tanden van gemeenschappelijke beweging keren ze samen terug naar
hun beginpositie? De configuratie herhaalt zich wanneer het aantal verstreken
tanden een gemeen veelvoud is van $84$ en $36$; de eerste keer is
\[
\operatorname{lcm}(84, 36) = \frac{84 \times 36}{\gcd(84, 36)}
= \frac{3024}{12} = 252
\]
tanden --- dat wil zeggen $3$ omwentelingen van het grote tandwiel en $7$ van het
kleine ($252/84$ en $252/36$). Merk de praktische weg op:
\emph{bereken eerst de ggd} (Euclides: $84 = 2\times36 + 12$, $36
= 3\times12$), en deel dan --- bouw het kgv nooit op door
veelvouden op te sommen. Elke periodiek-samenvallingsvraag (tandwielen, planetaire
uitlijningen, samenvallende repeterende decimalen) herleidt zich tot deze ene
berekening.
\end{example}

\section{Priemgetallen}

\begin{definition}\label{def:b1:arith:prime}
Een geheel getal $p \geq 2$ is \emph{priem}\index{priemgetal} wanneer zijn
enige positieve delers $1$ en $p$ zijn. Voor $p$ priem en $a \in \Z$:
ofwel $p \mid a$, ofwel $\gcd(p, a) = 1$. Bijgevolg
(\cref{thm:b1:arith:gauss}) geldt \emph{Euclides' lemma}: als $p \mid
ab$ dan $p \mid a$ of $p \mid b$.
\end{definition}

\begin{remark}[Priemtoets door proefdeling]\label{rem:b1:arith:trialdivision}
Als $n = ab$ met $2 \leq a \leq b$, dan $a^2 \leq ab = n$, dus $a
\leq \sqrt n$: een samengesteld getal $n$ heeft altijd een priemdeler $\leq
\sqrt n$. Om te toetsen of $n$ \omterm{def:b1:arith:prime}{priem} is volstaat het dus de
priemgetallen tot $\sqrt n$ te proberen. Voor $n = 271$: $\sqrt{271} < 17$, en
$271$ is door geen van $2, 3, 5, 7, 11, 13$ \omterm{def:b1:arith:divides}{deelbaar} (oneven, cijfersom
$10$, eindigt niet op $0$ of $5$, $271 = 7\cdot38 + 5 =
11\cdot24 + 7 = 13\cdot20 + 11$): \omterm{def:b1:arith:prime}{priem}, na zes delingen
in plaats van tweehonderd. De $\sqrt n$-barriĂ¨re is een echte
drempel: die efficiĂ«nt overschrijden voor honderdcijferige getallen
vereist de moderne priemtoetsen die zijn voortgekomen uit
\cref{thm:b1:arith:fermat}.
\end{remark}

\begin{theorem}[Euclides]\label{thm:b1:arith:euclidprimes}
Er zijn oneindig veel priemgetallen.
\end{theorem}

\begin{proof}
Elk geheel getal $n \geq 2$ heeft een priemdeler: zijn kleinste deler
$\geq 2$ is \omterm{def:b1:arith:prime}{priem} (een echte ontbinding ervan zou een
kleinere deler van $n$ opleveren). Veronderstel nu dat $p_1, \dots, p_k$ alle
priemgetallen waren, en zij $N = p_1 p_2 \cdots p_k + 1 \geq 2$. Een zeker \omterm{def:b1:arith:prime}{priemgetal} $p_i$
\omterm{def:b1:arith:divides}{deelt} $N$; maar $p_i$ \omterm{def:b1:arith:divides}{deelt} ook $N - 1 = p_1\cdots p_k$, dus $p_i
\mid 1$ --- absurd.
\end{proof}

\begin{theorem}[Hoofdstelling van de getaltheorie]\label{thm:b1:arith:fta}
Elk geheel getal $n \geq 2$ is een product van priemgetallen, en de ontbinding
\[
n = p_1^{\alpha_1} p_2^{\alpha_2} \cdots p_k^{\alpha_k}
\qquad (p_1 < p_2 < \dots < p_k \text{ priem},\ \alpha_i \in \N^*)
\]
is uniek.\index{priemontbinding}
\end{theorem}

\begin{proof}
\emph{Bestaan} door sterke inductie (\cref{thm:b1:logic:induction}):
$n = 2$ is \omterm{def:b1:arith:prime}{priem}; voor $n > 2$ is ofwel $n$ \omterm{def:b1:arith:prime}{priem}, ofwel $n = ab$ met
$2 \leq a, b < n$, en de inductiehypothese ontbindt $a$ en $b$.

\emph{Uniciteit.} Veronderstel $p_1 \cdots p_r = q_1 \cdots q_s$
(priemgetallen opgesomd met herhaling, zeg $r \leq s$), en pas inductie toe op
$r$. Als $r = 0$ is het linkerlid $1$, wat $s = 0$ afdwingt (een niet-leeg
product van priemgetallen is groter dan $1$). Voor $r \geq 1$: het \omterm{def:b1:arith:prime}{priemgetal} $p_1$
\omterm{def:b1:arith:divides}{deelt} $q_1(q_2\cdots q_s)$, dus volgens Euclides' lemma ofwel $p_1
\mid q_1$ ofwel $p_1 \mid q_2\cdots q_s$; door herhaling \omterm{def:b1:arith:divides}{deelt} $p_1$
een zekere $q_j$. Maar $q_j$ is \omterm{def:b1:arith:prime}{priem} en $p_1 \geq 2$: noodzakelijk $p_1
= q_j$. Schrap deze gemene factor (geoorloofd: $\Z$ is een
integriteitsdomein) om
\[
p_2 \cdots p_r = q_1 \cdots \widehat{q_j} \cdots q_s
\]
te verkrijgen (de hoed markeert weglating), een gelijkheid van kortere producten; de
inductiehypothese zegt dat de twee lijsten $p_2, \dots, p_r$ en
$q_1, \dots, \widehat{q_j}, \dots, q_s$ op volgorde na samenvallen,
en dus deden de oorspronkelijke dat ook. De exponentvorm groepeert gelijke
priemgetallen.
\end{proof}

\begin{proposition}[Valuaties]\label{prop:b1:arith:valuation}
Voor $p$ \omterm{def:b1:arith:prime}{priem} en $n \in \N^*$, schrijf $v_p(n)$\index{p-adische
valuatie@$p$-adische valuatie} voor de exponent van $p$ in de
ontbinding van $n$ (met $v_p(n) = 0$ als $p \nmid n$). Dan
\[
v_p(mn) = v_p(m) + v_p(n),
\qquad
m \mid n \iff \forall p,\ v_p(m) \leq v_p(n),
\]
\[
v_p\bigl(\gcd(m,n)\bigr) = \min\bigl(v_p(m), v_p(n)\bigr),
\qquad
v_p\bigl(\operatorname{lcm}(m,n)\bigr) = \max\bigl(v_p(m),
v_p(n)\bigr).
\]
\end{proposition}

\begin{proof}
De eerste identiteit geldt omdat de ontbindingen vermenigvuldigen en de
ontbinding van $mn$ uniek is. Als $m \mid n$, schrijf $n = mq$ en
pas ze toe. Omgekeerd, als alle $v_p(m) \leq v_p(n)$, voldoet het geheel getal $q =
\prod_p p^{\,v_p(n) - v_p(m)}$ aan $mq = n$. De ggd-formule:
het geheel getal $d = \prod p^{\min}$ \omterm{def:b1:arith:divides}{deelt} beide volgens het criterium, en
elke gemene deler $c$ heeft $v_p(c) \leq \min$ voor alle $p$, dus $c
\mid d$; dezelfde redenering voor het kgv met $\max$.
\end{proof}

\begin{example}[Kwadraten en derdemachten via valuaties]\label{ex:b1:arith:squarecube}
Een geheel getal $n \geq 1$ is een volkomen kwadraat als en slechts als elke
$v_p(n)$ even is (als $n = m^2$, dan $v_p(n) = 2v_p(m)$;
omgekeerd halveer elke exponent). Evenzo voor derdemachten met
veelvouden van $3$. Zo is $21168 = 2^4 \times 3^3 \times 7^2$ geen
kwadraat ($v_3 = 3$ is oneven) en geen derdemacht ($v_2 = 4$); het
kleinste positieve geheel getal $m$ zodanig dat $21168\,m$ \emph{wel} een
derdemacht is, wordt gevonden door elke exponent aan te vullen tot het volgende veelvoud van
$3$:
\[
m = 2^{6-4} \times 3^{3-3} \times 7^{3-2} = 2^2 \times 7 = 28,
\qquad
21168 \times 28 = 2^6\,3^3\,7^3 = (2^2 \times 3 \times 7)^3
= 84^3 .
\]
Het inzicht: multiplicatieve vragen (kwadraten, derdemachten, delers,
ggd, kgv) worden \emph{coĂ¶rdinaatsgewijze} vragen over de exponent\-vectoren
$(v_2, v_3, v_5, \dots)$ --- unieke ontbinding is de
\omterm{def:b1:logic:statement}{uitspraak} dat deze coĂ¶rdinaten bestaan en goed gedefinieerd zijn.
\end{example}

\section{Congruenties}

\begin{definition}\label{def:b1:arith:congruence}
Voor $n \in \N^*$: $a \equiv b \pmod n$\index{congruentie} wanneer $n \mid
a - b$. Dit is een \omterm{def:b1:logic:equiv}{equivalentierelatie} verenigbaar met optelling en
vermenigvuldiging: als $a \equiv b$ en $a' \equiv b'$ (mod $n$), dan
$a + a' \equiv b + b'$, $aa' \equiv bb'$, en $a^k \equiv b^k$ voor
$k \in \N$.
\end{definition}

\begin{example}[De negenproef]\label{ex:b1:arith:castingnines}
Verenigbaarheid met $+$ en $\times$ is een controlemiddel zo oud
als de handel. Aangezien $10 \equiv 1 \pmod 9$, is elk geheel getal
congruent mod $9$ met zijn cijfersom (bewezen als
\cref{exo:b1:arith:2}). Om de bewering $1234 \times 567 =
699\,678$ te controleren: cijfersommen geven $1234 \equiv 1$ en $567 \equiv 18
\equiv 0 \pmod 9$, dus het product moet $\equiv 1 \times 0 =
0$ zijn; en inderdaad $6 + 9 + 9 + 6 + 7 + 8 = 45 \equiv 0$. De controle
slaagt (en het product is in feite correct). Had iemand
$699\,478$ gerapporteerd, dan zou de cijfersom $43 \equiv 7 \not\equiv 0$ hem
onmiddellijk ontmaskeren. De toets is eenzijdig --- hij vangt een
fout op tenzij de fout zelf een veelvoud van $9$ is --- wat
precies de pseudopriemles is van
\cref{ex:b1:arith:pseudoprime} in het klein: congruentiecontroles
weerleggen, ze certificeren niet.
\end{example}

\begin{proposition}[Inverteerbaarheid mod $n$]\label{prop:b1:arith:invmod}
$a$ is \emph{inverteerbaar mod $n$} (d.w.z.\ $ab \equiv 1 \pmod n$ voor een zekere
$b$) als en slechts als $\gcd(a, n) = 1$. De inverse is dan uniek mod
$n$ en wordt berekend met het uitgebreide \omterm{met:b1:arith:euclid}{algoritme van Euclides}.
\end{proposition}

\begin{proof}
$ab \equiv 1 \pmod n$ betekent $ab + nk = 1$ voor een zekere $k$: een BĂ©zout-
betrekking, die bestaat asa $\gcd(a,n) = 1$
(\cref{cor:b1:arith:bezout}). Uniciteit: als $ab \equiv ab' \equiv 1$,
dan $b \equiv b(ab') = (ab)b' \equiv b' \pmod n$.
\end{proof}

\begin{example}[$7$ inverteren modulo $26$]\label{ex:b1:arith:inv7mod26}
Aangezien $\gcd(7, 26) = 1$ is de klasse van $7$ inverteerbaar mod $26$.
Uitgebreide Euclides:
\[
26 = 3 \times 7 + 5, \qquad
7 = 1 \times 5 + 2, \qquad
5 = 2 \times 2 + 1 ,
\]
en dan achterwaarts:
\[
1 = 5 - 2 \times 2 = 5 - 2(7 - 5) = 3 \times 5 - 2 \times 7
= 3(26 - 3 \times 7) - 2 \times 7 = 3 \times 26 - 11 \times 7 .
\]
Dus $7 \times (-11) \equiv 1 \pmod{26}$, d.w.z.\ $7^{-1} \equiv
-11 \equiv 15 \pmod{26}$; controle: $7 \times 15 = 105 = 4 \times 26
+ 1$. Met de inverse in de hand wordt elke \omterm{def:b1:arith:congruence}{congruentie} $7x \equiv c
\pmod{26}$ opgelost in Ă©Ă©n vermenigvuldiging: $x \equiv 15c$. Deze
mechanische inversie is het werkpaard van de modulaire rekenkunde ---
en van de publieke-sleutelprotocollen vermeld in
\cref{rem:b1:arith:whereused}, waar de moduli honderden
cijfers hebben maar het algoritme precies dit is.
\end{example}

\begin{example}[Wanneer de coĂ«fficiĂ«nt niet inverteerbaar is]\label{ex:b1:arith:noninvertible}
Los $12x \equiv 8 \pmod{20}$ op. Hier is $\gcd(12, 20) = 4$, dus $12$
is niet inverteerbaar mod $20$ --- maar de vergelijking is nog steeds
oplosbaar. De \omterm{def:b1:arith:congruence}{congruentie} zegt $20 \mid 12x - 8$; door de
hele betrekking door $4$ te delen (deler van alle drie de ingrediĂ«nten), is ze
equivalent met $5 \mid 3x - 2$, d.w.z.\
\[
3x \equiv 2 \pmod 5 .
\]
Nu is $\gcd(3, 5) = 1$ en $3^{-1} \equiv 2 \pmod 5$ ($3 \times 2 =
6 \equiv 1$), dus $x \equiv 4 \pmod 5$: de oplossingen zijn $x
\equiv 4, 9, 14, 19 \pmod{20}$ --- \emph{vier} klassen mod $20$,
overeenstemmend met de ggd. (Was het rechterlid niet \omterm{def:b1:arith:divides}{deelbaar} door $4$,
zeg $12x \equiv 6 \pmod{20}$, dan zou er helemaal geen oplossing zijn:
het linkerlid is altijd $\equiv 0 \pmod 4$.) Algemene vorm: $ax
\equiv b \pmod n$ is oplosbaar asa $\gcd(a, n) \mid b$, en heeft dan
precies $\gcd(a, n)$ oplossingsklassen --- deel alles
door de ggd en inverteer.
\end{example}

\begin{theorem}[Kleine stelling van Fermat]\label{thm:b1:arith:fermat}
Zij $p$ \omterm{def:b1:arith:prime}{priem}. Voor elke $a \in \Z$:
\[
a^p \equiv a \pmod p,
\]
en als $p \nmid a$, dan $a^{p-1} \equiv 1 \pmod
p$.\index{kleine stelling van Fermat}
\end{theorem}

\begin{proof}
Eerst, voor $1 \leq k \leq p - 1$, is de \omterm{def:b1:counting:objects}{binomiaalcoĂ«fficiĂ«nt} $\binom pk
= \frac{p!}{k!(p-k)!}$ \omterm{def:b1:arith:divides}{deelbaar} door $p$: inderdaad $k!\,(p-k)!\,
\binom pk = p!$ en $p$ \omterm{def:b1:arith:divides}{deelt} $p!$ maar is \omterm{cor:b1:arith:bezout}{relatief priem} met $k!(p-k)!$
(alle factoren zijn $< p$), dus het lemma van Gauss geeft $p \mid \binom pk$.

Bewijs nu $a^p \equiv a$ voor $a \in \N$ door inductie. Waar voor $a =
0$. Als $a^p \equiv a$, dan volgens het binomium van Newton
\[
(a+1)^p = \sum_{k=0}^{p} \binom pk a^k
\equiv a^p + 1 \equiv a + 1 \pmod p,
\]
waarbij alle middelste termen verdwijnen mod $p$. Voor $a < 0$, pas het resultaat toe op
$-a$ en scheid $p = 2$ (waar $x \equiv -x$) van oneven $p$ (waar
$(-a)^p = -a^p$). Ten slotte, als $p \nmid a$, vermenigvuldig $a^p \equiv a$ met
een inverse van $a$ mod $p$ (\cref{prop:b1:arith:invmod}).
\end{proof}

\begin{example}[Fermats omkering faalt: $341$]\label{ex:b1:arith:pseudoprime}
De kleine stelling van Fermat geeft een goedkope \emph{samengesteldheidstoets}:
als $a^{n-1} \not\equiv 1 \pmod n$ voor een zekere $a$ \omterm{cor:b1:arith:bezout}{relatief priem} met $n$,
dan is $n$ niet \omterm{def:b1:arith:prime}{priem}. Zou de toets ook priemheid kunnen certificeren? Nee:
neem $n = 341 = 11 \times 31$, samengesteld, en $a = 2$. Aangezien
$2^{10} = 1024 = 3 \times 341 + 1$,
\[
2^{10} \equiv 1 \pmod{341}
\qquad\Longrightarrow\qquad
2^{340} = \bigl(2^{10}\bigr)^{34} \equiv 1 \pmod{341} :
\]
het samengestelde getal $341$ doorstaat Fermats toets voor het grondtal $2$ (het is
het kleinste zulke \emph{pseudopriemgetal}). Het grondtal $3$ ontmaskert het
($3^{340} \not\equiv 1$), en praktische priemtoetsing
draait de toets daarom op verscheidene grondtallen, plus verfijningen ---
de industriĂ«le versies van dit idee zijn wat de grote
priemgetallen van \cref{rem:b1:arith:whereused} certificeert. Moraal: een implicatie
en haar omkering leiden een gescheiden leven
(\cref{rem:b1:logic:pitfalls}), zelfs voor stellingen.
\end{example}

\begin{example}[Praktische congruentieberekeningen]\label{ex:b1:arith:congruences}
Wat is de rest van $7^{2026}$ mod $11$? Volgens Fermat, $7^{10}
\equiv 1 \pmod{11}$. Aangezien $2026 = 10 \times 202 + 6$:
\[
7^{2026} \equiv 7^6 = (7^2)^3 = 49^3 \equiv 5^3 = 125 \equiv 4
\pmod{11}.
\]
De rest is $4$. De strategie: herleid de exponent modulo de
orde geleverd door Fermat, en herleid dan tussentijdse machten bij elke
stap.
\end{example}

\begin{remark}[Veelvoorkomende valkuilen in de getaltheorie]\label{rem:b1:arith:pitfalls}
\begin{enumerate}
  \item \emph{Een \omterm{def:b1:arith:congruence}{congruentie} delen.} Uit $ac \equiv bc \pmod n$
        mag men \emph{niet} besluiten $a \equiv b$ tenzij $\gcd(c,
        n) = 1$: $6 \equiv 2 \pmod 4$ maar $3 \not\equiv 1 \pmod
        4$. De juiste algemene regel \omterm{def:b1:arith:divides}{deelt} ook de \omterm{def:b1:complex:field}{modulus}: $ac
        \equiv bc \pmod n \iff a \equiv b \pmod{n/\gcd(c,n)}$.
  \item \emph{Euclides' lemma verkeerd gebruiken.} $a \mid bc$ impliceert $a
        \mid b$ of $a \mid c$ enkel voor $a$ \emph{\omterm{def:b1:arith:prime}{priem}} (of
        \omterm{cor:b1:arith:bezout}{relatief priem} met Ă©Ă©n factor): $6 \mid 4 \times 9$ maar $6$
        \omterm{def:b1:arith:divides}{deelt} geen van beide factoren.
  \item \emph{\omterm{cor:b1:arith:bezout}{Relatief priem} is een relatie, geen eigenschap.} ``$8$ en
        $9$ zijn \omterm{cor:b1:arith:bezout}{relatief priem}'' is waar hoewel geen van beide \omterm{def:b1:arith:prime}{priem} is;
        ``paarsgewijs \omterm{cor:b1:arith:bezout}{relatief priem}'' is sterker dan ``globaal
        \omterm{cor:b1:arith:bezout}{relatief priem}'' ($\gcd(6, 10, 15) = 1$ maar geen enkel paar is
        \omterm{cor:b1:arith:bezout}{relatief priem}).
  \item \emph{Exponenten leven niet mod $n$.} In $a^k \bmod n$
        mag de exponent enkel herleid worden modulo de \emph{orde}
        van $a$ (bijvoorbeeld $p - 1$ wanneer Fermat van toepassing is), nooit
        modulo $n$: $2^{10} \bmod 11$ is $1$, niet $2^{10 \bmod
        11} = 2^{10}$ --- de herleiding die werkt is die welke
        \cref{ex:b1:arith:congruences} uitvoert.
\end{enumerate}
\end{remark}

\begin{remark}[Waar dit hoofdstuk gebruikt wordt]\label{rem:b1:arith:whereused}
Dit hoofdstuk is evenzeer een sjabloon als een gereedschapskist. De hele keten
--- euclidische deling, ggd, BĂ©zout, Gauss, unieke ontbinding
--- wordt letterlijk herhaald voor veeltermen in \cref{ch:b1:poly},
waar ``graad'' de rol van absolute waarde speelt; de
twee hoofdstukken naast elkaar vergelijken is de beste manier om beide te begrijpen. De
congruentierekening wordt de ring $\Z/n\Z$ in
\cref{ch:b1:structures}, wier inverteerbare elementen
(\cref{prop:b1:arith:invmod}) het eerste niet-triviale voorbeeld vormen van
een groep van eenheden. Valuaties keren terug in het weekendprobleem hieronder
(de formule van Legendre) en voeden de irrationaliteitsbewijzen van
\cref{ch:b1:reals}. Buiten dit volume is BĂ©zout-inversie mod $n$
de motor van de publieke-sleutelcryptografie, en de kleine stelling van Fermat
is de grootvader van de priemtoetsen die de
grote priemgetallen certificeren die daar gebruikt worden.
\end{remark}

\begin{remark}[Intermezzo: $\Z$ als sjabloon]\label{rem:b1:arith:interlude}
Doe een stap terug van de individuele stellingen en observeer de
architectuur van het hoofdstuk: Ă©Ă©n gereedschap (euclidische deling)
bracht een classificatie voort (deelgroepen $n\Z$), die een
bestaansstelling voortbracht (ggd, BĂ©zout), die een \omterm{def:b1:arith:divides}{deelbaarheids}\-rekening
voortbracht (Gauss), die unieke ontbinding voortbracht --- elke
verdieping rust enkel op de verdieping eronder. Hetzelfde gebouw zal nog
tweemaal opgetrokken worden in dit volume met verschillende benedenverdiepingen:
in \cref{ch:b1:poly}, waar deling door graad de deling
door grootte vervangt en alles daarboven zich \emph{letterlijk} herhaalt; en, in
het klein, binnen elke $\Z/n\Z$ van \cref{ch:b1:structures},
waar inverteerbaarheidsvragen (het \cref{prop:b1:arith:invmod} van dit hoofdstuk)
structurele uitspraken over ringen en lichamen worden.
Een argument herkennen als ``het $\Z$-argument,
overgeplant'' is de snelste manier om die hoofdstukken te leren --- en
de eerste voorproef van de kerngewoonte van de algebra: stellingen bewijzen over
\emph{axioma's} in plaats van over objecten.
\end{remark}

\section{Probleem: de formule van Legendre en de overdrachten van Kummer}

\begin{problem}\label{pb:b1:arith:1}
Op hoeveel nullen eindigt de decimale schrijfwijze van $1000!$ --- en, dieper,
wat is de exacte macht van een \omterm{def:b1:arith:prime}{priemgetal} $p$ die $n!$ \omterm{def:b1:arith:divides}{deelt}, of die een
\omterm{def:b1:counting:objects}{binomiaalcoĂ«fficiĂ«nt} \omterm{def:b1:arith:divides}{deelt}? De volledige antwoorden zijn twee juwelen van de
elementaire getaltheorie: \emph{de formule van Legendre}
\index{formule van Legendre} $v_p(n!) = \sum_{k\geq1}
\lfloor n/p^k \rfloor$, met haar digitale gedaante $v_p(n!) = \frac{n
- s_p(n)}{p-1}$, en \emph{de stelling van Kummer}\index{stelling van
Kummer}: $v_p\binom{m+n}m$ telt de \emph{overdrachten} bij het optellen van
$m$ en $n$ in grondtal $p$. Dit probleem bewijst beide, controleert ze
numeriek tegen elkaar, en oogst de klassieke
gevolgen --- eindnullen, de pariteit van de driehoek van Pascal,
en een eerste afschatting in de richting van de priemgetalstelling.
Overal is $p$ een \omterm{def:b1:arith:prime}{priemgetal}, $\floor{x}$ het gehele deel, en
$s_p(n)$ duidt de som aan van de cijfers van $n$ geschreven in grondtal $p$.

\medskip
\noindent\textbf{Deel I --- Gehele delen, valuaties en de formule van
Legendre.}
\begin{enumerate}
  \item Opwarmer: bereken $10!$ en lees zijn aantal eindnullen
        af; bereken $v_2(10!)$ en $v_5(10!)$ rechtstreeks uit de
        ontbinding van elke factor $1, 2, \dots, 10$.
  \item Bewijs dat voor $x \in \R$ en $n \in \N^*$,
        $\bigl\lfloor \lfloor x \rfloor / n \bigr\rfloor =
        \lfloor x/n \rfloor$.
  \item Bewijs dat $v_p(a + b) \geq \min\bigl(v_p(a),
        v_p(b)\bigr)$ voor alle $a, b \in \N^*$, met gelijkheid
        telkens wanneer $v_p(a) \neq v_p(b)$.
  \item Toon dat het aantal veelvouden van $m$ in $\intint1n$ gelijk is aan
        $\lfloor n/m \rfloor$.
  \item Bewijs \emph{de formule van Legendre}: voor elke $n \in \N^*$,
        \[
        v_p(n!) = \sum_{k=1}^{\infty}
        \Bigl\lfloor \frac{n}{p^k} \Bigr\rfloor
        \]
        (een eindige som: de termen verdwijnen zodra $p^k > n$).
        \emph{Tel, voor elke $k$, de factoren van $\intint1n$
        die \omterm{def:b1:arith:divides}{deelbaar} zijn door $p^k$: elke draagt precies Ă©Ă©n eenheid bij per
        niveau dat ze bereikt.}
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De digitale vorm en de
eindnullen.}
\begin{enumerate}[resume]
  \item Bereken $v_5(1000!)$ en $v_2(1000!)$, en besluit: op hoeveel
        nullen eindigt $1000!$?
  \item Bewijs de digitale vorm van de formule van Legendre: schrijf $n =
        \sum_i a_i p^i$ in grondtal $p$,
        \[
        v_p(n!) = \frac{n - s_p(n)}{p - 1} .
        \]
  \item Twee gevolgen voor $p = 2$: toon dat $2^n$ nooit
        $n!$ \omterm{def:b1:arith:divides}{deelt}, en dat $2^{n-1}$ $n!$ precies dan \omterm{def:b1:arith:divides}{deelt} wanneer
        $n$ een macht van $2$ is.
  \item Schat het tekort af: toon $\frac n{p-1} - \log_p(n) - 1 \leq
        v_p(n!) < \frac n{p-1}$, zodat $\frac{v_p(n!)}{n} \to
        \frac1{p-1}$: op de lange duur komt een aandeel
        $\frac1{p-1}$ van Ă©Ă©n factor $p$ per eenheid bij.
  \item Zij $Z(n) = v_5(n!)$ het aantal eindnullen van
        $n!$. Toon $Z(n) - Z(n-1) = v_5(n)$, leid af dat $Z$
        de waarde $5$ volledig overslaat (bereken $Z(24)$ en $Z(25)$), en
        bewijs dat geen enkele faculteit eindigt op precies vijf nullen.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De stelling van Kummer.}
\begin{enumerate}[resume]
  \item Bewijs dat $\lfloor x + y \rfloor - \lfloor x \rfloor -
        \lfloor y \rfloor \in \{0, 1\}$ voor alle $x, y \in \R$, en
        leid uit de formule van Legendre af dat
        \[
        v_p\binom{m+n}m
        = \sum_{k\geq1}\Bigl(
        \Bigl\lfloor\frac{m+n}{p^k}\Bigr\rfloor
        - \Bigl\lfloor\frac{m}{p^k}\Bigr\rfloor
        - \Bigl\lfloor\frac{n}{p^k}\Bigr\rfloor\Bigr),
        \]
        een som van termen elk gelijk aan $0$ of $1$.
  \item Bewijs \emph{de stelling van Kummer}: de $k$-de term van die som
        is gelijk aan $1$ precies wanneer de optelling van $m$ en $n$ in
        grondtal $p$ een overdracht naar positie $k$ voortbrengt; dus is
        $v_p\binom{m+n}m$ het totale aantal overdrachten.
        \emph{(Schrijf $m = p^km_1 + m_0$ en $n = p^kn_1 + n_0$
        met $0 \leq m_0, n_0 < p^k$ en inspecteer $\lfloor (m_0 +
        n_0)/p^k \rfloor$.)}
  \item Leid af dat voor $0 < j < p^k$:
        \[
        v_p\binom{p^k}{j} = k - v_p(j) ,
        \]
        door de overdrachten te tellen in de optelling $j + (p^k - j)$.
        (In het bijzonder $p \mid \binom p j$ voor $0 < j < p$: de
        sleutelstap van \cref{thm:b1:arith:fermat}, herwonnen.)
  \item Bewijs dat $v_2\binom{2n}n = s_2(n)$. Leid af dat de
        centrale \omterm{def:b1:counting:objects}{binomiaalcoĂ«fficiĂ«nt} altijd even is, en dat
        $\binom{2n}n \equiv 2 \pmod 4$ precies wanneer $n$ een macht
        van $2$ is.
  \item Toon, met de identiteit van Vandermonde
        (\cref{exo:b1:counting:7}) en vraag 13, dat
        $\binom{2p}p \equiv 2 \pmod p$ voor elk \omterm{def:b1:arith:prime}{priemgetal} $p$.
  \item Bereken $v_3\binom{1000}{500}$ tweemaal: eenmaal via Kummer
        (schrijf $500$ in grondtal $3$ en tel de overdrachten in $500 +
        500$), eenmaal via de digitale vorm van Legendre (bereken $s_3(500)$
        en $s_3(1000)$); controleer dat beide dezelfde waarde geven.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De pariteit van de driehoek van Pascal, en
een priemdichtheidsafschatting.}
\begin{enumerate}[resume]
  \item Bewijs het cijfercriterium: $\binom nk$ is \emph{oneven} als en
        slechts als elk binair cijfer van $k$ ten hoogste het
        overeenkomstige cijfer van $n$ is. Formuleer en bewijs het analoge
        criterium voor $p \nmid \binom nk$ in grondtal $p$.
  \item Leid af dat rij $n$ van de driehoek van Pascal precies
        $2^{s_2(n)}$ oneven elementen bevat; verifieer op rijen $4$ en $5$.
  \item Leid af dat alle inwendige elementen $\binom nk$ ($0 < k < n$)
        even zijn als en slechts als $n$ een macht van $2$ is.
  \item Bewijs dat elke priemmacht die $\binom{m+n}m$ \omterm{def:b1:arith:divides}{deelt} ten
        hoogste $m + n$ is: als $p^a \mid \binom{m+n}m$ dan $p^a \leq
        m + n$. \emph{(Hoeveel niet-nul termen kan de som van
        vraag 11 hebben?)}
  \item Leid af dat $\binom{2n}n$ het getal
        $\operatorname{lcm}(1, 2, \dots, 2n)$ \omterm{def:b1:arith:divides}{deelt}, en combineer met
        de ondergrens $\binom{2n}n \geq \frac{4^n}{2n+1}$
        (die je zult bewijzen: het centrale element is het grootste
        van de $2n + 1$ elementen van rij $2n$) om
        \[
        \operatorname{lcm}(1, \dots, 2n) \geq \frac{4^n}{2n+1} :
        \]
        te verkrijgen: de gemene veelvouden van de eerste gehele getallen groeien
        \emph{exponentieel} --- een eerste kwantitatieve glimp van
        de overvloed aan priemgetallen.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Synthese.}
\begin{enumerate}[resume]
  \item Vind de kleinste $n$ zodanig dat $n!$ eindigt op ten minste
        $2026$ nullen. \emph{(Schat $Z(n) \approx n/4$, en
        stel dan bij met behulp van de exacte formule.)}
  \item Een laatste kruiscontrole: toon dat $7$ het getal
        $\binom{100}{50}$ \emph{niet} \omterm{def:b1:arith:divides}{deelt}, eerst door $50$ in grondtal $7$ te
        schrijven en te controleren dat de optelling $50 + 50$ overdrachtvrij is,
        en dan door $v_7(100!)$ en $v_7(50!)$ met de formule van Legendre
        te berekenen.
  \item Waar precies gebruikte het probleem: (i) unieke
        ontbinding; (ii) de euclidische-delingsontbinding
        $n = p^k n_1 + n_0$; (iii) een telargument uit

  \item Synthese, in een korte paragraaf: de formule van Legendre zet

        ligt ver buiten dit volume; het polynomiale
