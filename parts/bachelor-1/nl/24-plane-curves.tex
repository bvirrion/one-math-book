\chapter{Vlakke krommen}\label{ch:b1:curves}

De analyse van \cref{ch:b1:derivative,ch:b1:taylor} was gebouwd voor
functies $y = f(x)$; de meeste krommen uit de meetkunde en de mechanica
--- banen, cirkels die over cirkels rollen, omlopen --- weigeren die
vorm en verschijnen in plaats daarvan als \emph{geparametriseerde}
krommen $t \mapsto (x(t), y(t))$ of als \emph{\omterm{def:b1:curves:polar}{poolkrommen}} $r =
r(\theta)$. Dit klassieke hoofdstuk bestudeert beide, en sluit af met de
\omterm{def:b1:curves:conics}{kegelsneden}.

\section{Geparametriseerde krommen}

\begin{definition}\label{def:b1:curves:def}
Een \emph{geparametriseerde kromme}\index{geparametriseerde kromme} is
een \omterm{def:b1:logic:map}{afbeelding} $f \colon I \to \R^2$, $t \mapsto M(t) = (x(t), y(t))$,
met $x, y$ van klasse $C^1$ (ten minste) op het \omterm{prop:b1:reals:intervals}{interval} $I$. De
\emph{snelheidsvector} is $f'(t) = (x'(t), y'(t))$; het punt $M(t_0)$ is
\emph{regulier} wanneer $f'(t_0) \neq (0,0)$, en de \emph{raaklijn}
daar is de rechte door $M(t_0)$ gericht door $f'(t_0)$.
\end{definition}

\begin{proof}[Waarom $f'$ de raaklijn richt]
Volgens Taylor--Young per component geldt $M(t) = M(t_0) + (t - t_0)
f'(t_0) + o(t - t_0)$: de koorderichting $\frac{M(t) - M(t_0)}{t - t_0}$
streeft naar $f'(t_0)$. De regulariteitshypothese is wat de limiet tot
een \emph{richting} maakt: als $f'(t_0) = 0$, dan valt de uitdrukking
terug op $M(t) = M(t_0) + o(t - t_0)$ en zegt ze niets over hoe de
kromme het punt verlaat --- vandaar de aparte behandeling van singuliere
punten in de methode hieronder, waar de eerste \emph{niet-verdwijnende}
\omterm{def:b1:derivative:def}{afgeleide} de rol van $f'$ overneemt. Merk ook op dat de raaklijn een
meetkundig object is: elke herparametrisatie wijzigt $f'$ met een
niet-nulle scalaire factor en laat de rechte onveranderd.
\end{proof}

\begin{method}[Een geparametriseerde kromme bestuderen]\label{met:b1:curves:study}
\begin{enumerate}
  \item \emph{Beperk het domein} met behulp van symmetrieën: verbanden
        tussen $M(-t)$, $M(t + T)$, \dots\ en $M(t)$ (spiegelingen,
        translaties, periodiciteit), en teken dan enkel het beperkte
        deel.
  \item \emph{Variaties}: tabelleer de tekens van $x'$ en $y'$ samen;
        de kromme beweegt rechts/links naargelang $x'$, omhoog/omlaag
        naargelang $y'$.
  \item \emph{Opmerkelijke punten}: horizontale raaklijn ($y' = 0 \neq
        x'$), verticale raaklijn ($x' = 0 \neq y'$); in een
        \emph{singulier} punt ($f' = 0$) onderzoek je hogere \omterm{def:b1:derivative:def}{afgeleiden}
        voor de raakrichting.
  \item \emph{Asymptotisch gedrag} aan de uiteinden van $I$, en schets
        vervolgens.
\end{enumerate}
\end{method}

\begin{example}[Asymptotische takken, uitgewerkt]
\label{ex:b1:curves:asymptotes}
$x(t) = t$, $y(t) = t + \dfrac1t$ op $\intoo{0}{+\infty}$. Als $t
\to 0^{+}$: $x \to 0$ terwijl $y \to +\infty$ --- de kromme klimt
langs de \emph{verticale asymptoot} $x = 0$ (eindige limiet voor
één coördinaat, oneindig voor de andere). Als $t \to +\infty$:
beide \omterm{prop:b1:vspaces:coordinates}{coördinaten} lopen op naar oneindig, dus test een rechte: het
verschil
\[
y(t) - x(t) = \frac1t \longrightarrow 0^{+}
\]
onthult de \emph{scheve asymptoot} $y = x$, van bovenaf benaderd.
Daartussen verdwijnt $y' = 1 - \frac1{t^2}$ bij $t = 1$:
het punt $(1, 2)$ is het laagste punt van de tak, en $x' = 1 >
0$ overal, dus de kromme gaat steeds naar rechts. Eén boog,
twee asymptoten, één minimum: een volledig \omterm{def:b1:linmaps:kerim}{beeld} uit drie
berekeningen --- met het algemene recept eronder zichtbaar:
wanneer $x, y \to \infty$ samen, onderzoek dan $y/x$ voor een
kandidaat-richtingscoëfficiënt (hier $\to 1$), en vervolgens $y -
(\text{richtingscoëfficiënt})\,x$ voor het snijpunt en de kant van
de benadering.
\end{example}

\begin{example}[De astroïde]\label{ex:b1:curves:astroid}
$x(t) = \cos^3 t$, $y(t) = \sin^3 t$.\index{astroïde} Symmetrieën:
$M(t + 2\pi) = M(t)$ (bestudeer één periode); $M(-t)$ is de spiegeling
van $M(t)$ in de $x$-as; $M(\pi - t)$ in de $y$-as; $M(\frac\pi2 -
t)$ in de diagonaal $y = x$: het volstaat $t \in
\intcc{0}{\frac\pi4}$ te bestuderen en te ontvouwen.

Snelheid: $f'(t) = 3\sin t\cos t\,(-\cos t, \sin t)$. Op
$\intoo{0}{\frac\pi2}$ zijn alle punten regulier met raaklijn gericht
door $(-\cos t, \sin t)$; bij $t = 0$ (het punt $(1,0)$) verdwijnt de
snelheid: een \emph{keerpunt}\index{keerpunt}, waar de kromme omkeert langs
de raakrichting $(-1, 0)\cdot(\pm)$ --- door symmetrie liggen de vier
\omterm{ex:b1:curves:astroid}{keerpunten} bij $(\pm1, 0), (0, \pm1)$.
\end{example}

\begin{example}[Een lemniscaatvorm, volledig bestudeerd]
\label{ex:b1:curves:lissajous}
$x(t) = \sin t$, $y(t) = \sin 2t$ (een \emph{Lissajous-kromme}).
\emph{Symmetrieën}: $M(t + \pi) = (-x(t), y(t))$ (spiegeling in
de $y$-as), $M(-t) = (-x(t), -y(t))$ (centrale symmetrie),
$M(\pi - t) = (x(t), -y(t))$ (spiegeling in de $x$-as): het
volstaat $t \in \intcc{0}{\frac\pi2}$ te bestuderen en te ontvouwen.
\emph{Variaties}: $x' = \cos t \geq 0$ overal, terwijl $y' =
2\cos 2t$ positief is vóór $t = \frac\pi4$ en negatief erna:
de boog klimt naar rechts tot de top
$\bigl(\frac{\sqrt2}2,\ 1\bigr)$ bij $t = \frac\pi4$ (horizontale
raaklijn), en daalt dan naar rechts tot $(1, 0)$ bij $t =
\frac\pi2$, waar $x' = 0 \neq y'$: verticale raaklijn.
\emph{Dubbelpunt}: $M(0) = M(\pi) = (0,0)$, met twee verschillende
snelheden
\[
f'(0) = (1,\ 2), \qquad f'(\pi) = (-1,\ 2) :
\]
twee \emph{reguliere} takken die elkaar in de oorsprong kruisen onder
verschillende hoeken --- een dubbelpunt, geen singulier punt: elke
doorgang is volkomen glad, de twee doorgangen delen enkel hun ligging.
De hele kromme is de lemniscaatvorm hieronder.
\end{example}

\section{Poolkrommen}

\begin{definition}\label{def:b1:curves:polar}
Een \emph{poolkromme}\index{poolkromme} wordt gegeven door $r =
r(\theta)$: het punt met parameter $\theta$ is
\[
M(\theta) = r(\theta)\,\vec u(\theta),
\qquad \vec u(\theta) = (\cos\theta, \sin\theta).
\]
Met $\vec v(\theta) = (-\sin\theta, \cos\theta) = \vec
u\,'(\theta)$ is de snelheid
\[
M'(\theta) = r'(\theta)\, \vec u(\theta) + r(\theta)\, \vec
v(\theta) .
\]
Gevolgen: waar $r \neq 0$ is het punt regulier, en de raaklijn maakt
met de straal de hoek $V$ gegeven door $\tan V =
\frac{r}{r'}$ (hoek tussen $M'$ en $\vec u$); waar $r(\theta_0)
= 0$, gaat de kromme door de oorsprong \emph{met als raaklijn de straal
$\theta = \theta_0$} (richting $\vec u(\theta_0)$, af te lezen uit
$M'(\theta_0) = r'(\theta_0)\vec u(\theta_0)$, of uit de
limietkoorde: de koorde van $O$ naar $M(\theta)$ wordt gedragen door
$\vec u(\theta)$ zelf, die naar $\vec u(\theta_0)$ streeft --- dus de
regel geldt zelfs wanneer $r'(\theta_0) = 0$ en het punt
singulier is, en dat is waarom \omterm{ex:b1:curves:astroid}{pool-keerpunten} in de oorsprong, zoals
dat van de cardioïde, hun raaklijn gratis krijgen).
\end{definition}

\begin{example}[De cardioïde]\label{ex:b1:curves:cardioid}
$r(\theta) = 1 + \cos\theta$.\index{cardioïde} Symmetrie: $r(-\theta)
= r(\theta)$: spiegeling in de $x$-as; bestudeer $\theta \in
\intcc{0}{\pi}$. $r$ daalt van $2$ naar $0$; bij $\theta = \pi$,
$r = 0$: de kromme bereikt de oorsprong rakend aan de straal
$\theta = \pi$ (de $x$-as), en vormt daar een \omterm{ex:b1:curves:astroid}{keerpunt} --- de punt van
het hart. Raaklijn bij $\theta = 0$: $r' = 0$, dus $\tan V = \infty$:
loodrecht op de as.

De hoek $V$ verdient nog één lezing. Bij $\theta =
\frac\pi2$: $r = 1$ en $r' = -1$, dus $\tan V = \frac{r}{r'} =
-1$: de raaklijn maakt driekwart van een rechte hoek met de
uitgaande straal --- de kromme buigt al terug naar haar
\omterm{ex:b1:curves:astroid}{keerpunt}. De formule $\tan V = r/r'$ levert raakrichtingen
langs de hele kromme zonder ook maar enige berekening van
$M'(\theta)$: het is het pool-analogon van het aflezen van een
richtingscoëfficiënt.
\end{example}

\begin{example}[Van pool- naar cartesiaans: een verborgen cirkel]
\label{ex:b1:curves:polarcircle}
Wat is de \omterm{def:b1:curves:polar}{poolkromme} $r = 2\cos\theta$? Vermenigvuldig met $r$:
$r^2 = 2r\cos\theta$, d.w.z.\ $x^2 + y^2 = 2x$, d.w.z.
\[
(x - 1)^2 + y^2 = 1 :
\]
de cirkel met middelpunt $(1, 0)$ en straal $1$, door de
oorsprong gaand. De boekhouding is van belang: terwijl $\theta$ over
$\intoc{-\frac\pi2}{\frac\pi2}$ loopt, wordt de hele cirkel
precies één keer doorlopen ($r$ verdwijnt aan beide uiteinden), en bij
$\theta = \pm\frac\pi2$ geeft de regel ``raaklijn in de oorsprong langs
de straal $\theta = \theta_0$'' daar een \emph{verticale} raaklijn ---
in overeenstemming met de meetkunde, want de verticale as is inderdaad
rakend aan deze cirkel in $O$. Voor $\theta$ buiten dat bereik doorloopt
$r < 0$ dezelfde cirkel opnieuw: een herinnering dat een \omterm{def:b1:curves:polar}{poolkromme} een
\emph{geparametriseerd} object is, dat over zichzelf mag heenlopen.
\end{example}

\section{Booglengte}

\begin{definition}[Lengte van een boog]\label{def:b1:curves:arclength}
De \emph{lengte}\index{booglengte} van een $C^1$-boog $f \colon
\intcc{a}{b} \to \R^2$ is de \omterm{thm:b1:integration:def}{integraal} van de snelheid:
\[
L = \int_a^b \norm{f'(t)}\,\dd t
= \int_a^b \sqrt{x'(t)^2 + y'(t)^2}\;\dd t .
\]
(Motivatie: op een klein \omterm{prop:b1:reals:intervals}{interval} geldt $M(t + h) \approx M(t) + h
f'(t)$, dus de boog ligt \omterm{def:b1:topology:dense}{dicht} bij een veelhoek waarvan de
segmentlengten optellen tot een \omterm{thm:b1:integration:riemann}{Riemannsom} van $\norm{f'}$,
\cref{thm:b1:integration:riemann}.) Voor een \omterm{def:b1:curves:polar}{poolkromme} $r =
r(\theta)$ heeft de snelheid $r'\vec u + r\vec v$ orthogonale
componenten, dus
\[
L = \int_{\theta_1}^{\theta_2}
\sqrt{r'(\theta)^2 + r(\theta)^2}\;\dd\theta .
\]
De lengte hangt niet af van de gekozen (monotone, $C^1$)
parametrisatie: door $t = \varphi(s)$ te substitueren in de
\omterm{thm:b1:integration:def}{integraal} (\cref{thm:b1:integration:parts}) wordt $f'$
vermenigvuldigd met $\varphi'$ en $\dd t$ met $\varphi'^{-1}$.
\end{definition}

\begin{example}[Controle: de cirkel]\label{ex:b1:curves:circlelength}
Voor $f(t) = (R\cos t, R\sin t)$ op $\intcc{0}{2\pi}$: $\norm{f'} =
R$, dus $L = 2\pi R$ --- de definitie geeft de omtrek terug.
Herparametrisatietest: de \omterm{def:b1:logic:map}{afbeelding} $g(t) = (R\cos 2t, R\sin 2t)$ op
$\intcc{0}{\pi}$ tekent \emph{dezelfde} cirkel met dubbele snelheid
$\norm{g'} = 2R$, en
\[
\int_0^{\pi} 2R\,\dd t = 2\pi R
\]
opnieuw: de helft van de tijd, dubbele snelheid, dezelfde \omterm{def:b1:curves:arclength}{lengte} ---
de in de definitie beloofde invariantie, één keer op getallen bekeken.
($g$ over heel $\intcc{0}{2\pi}$ laten lopen zou $4\pi R$ geven: een
kromme die tweemaal wordt doorlopen is tweemaal zo lang \emph{als een
reis}; \omterm{def:b1:curves:arclength}{lengte} meet het geparametriseerde pad, en een eerlijke
boekhouding van het \omterm{prop:b1:reals:intervals}{interval} is deel van de berekening.)
Voor de astroïde $(\cos^3t, \sin^3t)$: $\norm{f'} = 3\abs{\sin
t\cos t} = \tfrac32\abs{\sin 2t}$, en door symmetrie $L = 4\int_0^{\pi/2}
\tfrac32\sin 2t\,\dd t = 6$: een kromme getekend binnen de
eenheidscirkel, met \omterm{def:b1:curves:arclength}{lengte} $6 < 2\pi$. Het weekendprobleem meet de meest
beroemde boog van allemaal.
\end{example}

\section{Kegelsneden}

\begin{definition}[Brandpunt--richtlijn-definitie]\label{def:b1:curves:conics}
Kies een punt $F$ (\emph{brandpunt}), een rechte $D$ niet door $F$
(\emph{richtlijn}) en $e > 0$ (\emph{excentriciteit}). De
\emph{kegelsnede}\index{kegelsnede} met deze gegevens is
\[
\mathcal{C} = \{M : d(M, F) = e\; d(M, D)\}:
\]
een \emph{ellips} voor $e < 1$, een \emph{parabool} voor $e = 1$, een
\emph{hyperbool} voor $e > 1$. (De cirkel verschijnt als een ontaarde
limiet $e \to 0$.)
\end{definition}

\begin{example}[De definitie, gecontroleerd op een parabool]
\label{ex:b1:curves:parabolacheck}
Neem de parabool $y^2 = 4x$, d.w.z.\ $2p = 4$: brandpunt $F = (1, 0)$
en richtlijn $D : x = -1$ (de herleide vorm
\cref{thm:b1:curves:reduced} plaatst ze op $\pm\frac p2$). In het
punt $M = (1, 2)$ van de kromme:
\[
MF = \sqrt{(1-1)^2 + 2^2} = 2,
\qquad
d(M, D) = 1 - (-1) = 2 :
\]
gelijk, zoals $e = 1$ eist. In $M' = (4, 4)$: $MF' = \sqrt{9 +
16} = 5$ en $d(M', D) = 5$ opnieuw. De brandpunt--richtlijn-definitie
is geen abstractie: het is een paar afstanden dat men op elk punt kan
meten, en de algebra van de herleide vergelijkingen is niets anders dan
deze meting eens en voor altijd uitgevoerd.
\end{example}

\begin{theorem}[Herleide vergelijkingen]\label{thm:b1:curves:reduced}
In een goed gekozen \omterm{def:b1:euclid:orthogonal}{orthonormaal} assenstelsel:
\[
\text{ellips: } \frac{x^2}{a^2} + \frac{y^2}{b^2} = 1
\quad (a \geq b > 0),
\qquad
\text{hyperbool: } \frac{x^2}{a^2} - \frac{y^2}{b^2} = 1,
\qquad
\text{parabool: } y^2 = 2px,
\]
met, voor de ellips: brandpunten in $(\pm c, 0)$, $c = \sqrt{a^2 - b^2}$,
$e = \frac ca$, en de bifocale karakterisering $MF + MF' = 2a$;
voor de hyperbool: $c = \sqrt{a^2 + b^2}$, $e = \frac ca$,
$\abs{MF - MF'} = 2a$, asymptoten $y = \pm\frac ba x$.
\end{theorem}

\begin{proof}
Neem het brandpunt in de oorsprong en de richtlijn verticaal, $x = -h$
($h > 0$): de voorwaarde $MF^2 = e^2 d(M, D)^2$ luidt $x^2 + y^2 =
e^2 (x + h)^2$. Voor $e = 1$: $y^2 = 2hx + h^2$, een parabool na de
verschuiving $x \mapsto x - \frac h2$ (dus $p = h$). Voor $e \neq 1$:
door de kwadraatsvervollediging in $x$,
\[
(1 - e^2)\Bigl(x - \frac{e^2 h}{1 - e^2}\Bigr)^{\!2} + y^2
= e^2h^2 + \frac{e^4 h^2}{1 - e^2}
= \frac{e^2 h^2}{1 - e^2} .
\]
Stel $X = x - \frac{e^2h}{1-e^2}$ (een verschuiving van assenstelsel).
Als $e < 1$, deel dan door het positieve rechterlid:
$\frac{X^2}{a^2} + \frac{y^2}{b^2} = 1$ met $a = \frac{eh}{1 -
e^2}$, $b = \frac{eh}{\sqrt{1-e^2}}$; als $e > 1$, dan wisselen beide
leden van de uitdrukking gepast van teken en dezelfde deling geeft
$\frac{X^2}{a^2} - \frac{y^2}{b^2} = 1$ met $a = \frac{eh}{e^2 -
1}$, $b = \frac{eh}{\sqrt{e^2 - 1}}$. De vermelde waarden van $c$
volgen ($c^2 = a^2 - b^2$ of $a^2 + b^2$ geeft $c = ea$ in beide
gevallen, waardoor het brandpunt correct wordt geplaatst), en de
bifocale eigenschappen zijn directe verificaties op de herleide
vergelijkingen. In detail voor de ellips, met $F' = (c, 0)$ en $M = (X,
y)$ op de kromme:
\[
MF'^2 = (X - c)^2 + y^2
= X^2 - 2cX + c^2 + b^2\Bigl(1 - \frac{X^2}{a^2}\Bigr)
= \frac{c^2}{a^2}X^2 - 2cX + a^2
= (a - eX)^2 ,
\]
met behulp van $c^2 + b^2 = a^2$ en $e = \frac ca$; aangezien $\abs X
\leq a$ en $e < 1$, is $a - eX > 0$, dus $MF' = a - eX$ zonder ooit
een vierkantswortel te trekken. De spiegelberekening geeft $MF = a +
eX$, waaruit $MF + MF' = 2a$, constant. Voor de hyperbool levert
dezelfde algebra $MF = \abs{a + eX}$ en $MF' = \abs{eX - a}$,
met verschil $\pm2a$ naargelang de tak.
\end{proof}

\begin{example}\label{ex:b1:curves:conicexample}
$\frac{x^2}{25} + \frac{y^2}{9} = 1$: ellips, $a = 5$, $b = 3$, $c
= 4$: brandpunten $(\pm4, 0)$, excentriciteit $\frac45$. Haar
parametrisatie: $(5\cos t, 3\sin t)$ --- een anisotroop uitgerekte
cirkel; de som van de afstanden tot de brandpunten van elk van haar
punten is $10$.
\end{example}

\begin{example}[Een omloop aflezen uit haar poolvergelijking]
\label{ex:b1:curves:orbit}
De poolkegelsnede $r = \dfrac{1}{1 + \frac12\cos\theta}$
(\cref{exo:b1:curves:7} met $p = 1$, $e = \frac12$) is een
ellips met een \emph{brandpunt in de oorsprong} --- de meetkunde van een
planeetbaan met de zon in $O$. Haal alles uit
$p$ en $e$: uit de herleiding van
\cref{thm:b1:curves:reduced}, $a = \dfrac{p}{1 - e^2} =
\dfrac{1}{3/4} = \dfrac43$ en $c = ea = \dfrac23$. De twee
apsiden controleren het zonder enige theorie:
\[
r(0) = \frac{1}{3/2} = \frac23 = a - c
\quad (\text{perihelium}),
\qquad
r(\pi) = \frac{1}{1/2} = 2 = a + c
\quad (\text{aphelium}),
\]
en hun som $\frac23 + 2 = \frac83 = 2a$ herstelt de grote
as. De poolvorm is de natuurlijke telkens wanneer een brandpunt
fysisch onderscheiden is; de herleide cartesiaanse vorm telkens wanneer
de symmetrieassen dat zijn. Omzetten tussen beide is precies
wat de kwadraatsvervolledigingsberekening van de stelling doet.

Excentriciteit ten slotte als draaiknop: houd $p = 1$ en verdraai $e$ in
$r = \frac{1}{1 + e\cos\theta}$. Bij $e = 0$: de cirkel $r = 1$. Bij
$e = \frac12$: de zojuist bestudeerde ellips, waarbij $r$ oscilleert
tussen $\frac23$ en $2$. Bij $e = 1$: $r(\theta) \to \infty$ als
$\theta \to \pi$ --- de kromme sluit niet langer: een parabool, met haar
verste punt naar oneindig geduwd. Bij $e = 2$: de noemer
verdwijnt bij $\cos\theta = -\frac12$, en enkel $\theta \in
\intoo{-\frac{2\pi}3}{\frac{2\pi}3}$ overleeft: één tak van een
hyperbool, ontsnappend langs twee asymptotische richtingen. Eén
formule, de hele familie \omterm{def:b1:curves:conics}{kegelsneden}, en de overgangspunten $e =
1$ zichtbaar als het moment waarop de noemer voor het eerst nul bereikt.
\end{example}

\begin{remark}[Veelvoorkomende valkuilen]
\label{rem:b1:curves:pitfalls}
\emph{Dubbelpunten zijn geen singuliere punten}: bij een
zelfkruising is elke tak regulier; ``singulier'' verwijst naar
$f'(t_0) = 0$ voor één waarde van de parameter
(\cref{ex:b1:curves:lissajous} tegenover de \omterm{ex:b1:curves:astroid}{keerpunten} van de
astroïde). \emph{Verticale raaklijn versus \omterm{ex:b1:curves:astroid}{keerpunt}}: $x'(t_0) = 0
\neq y'(t_0)$ is een regulier punt met verticale raaklijn; enkel
$x' = y' = 0$ vereist de hogere-orde-ontwikkeling.
\emph{Negatieve stralen worden afgezet op de tegenoverliggende straal}:
voor $r(\theta) < 0$ is het punt $-\abs{r}\,\vec u(\theta)$, onder hoek
$\theta + \pi$ (\cref{exo:b1:curves:5}); dit vergeten kost binnenlussen
of verdubbelt krommen.
\emph{\omterm{def:b1:curves:arclength}{Booglengte} integreert $\norm{f'}$, niet $f'$}: splits de
\omterm{thm:b1:integration:def}{integraal} bij de nulpunten van de snelheid --- voor de astroïde geeft
het integreren van $\frac32\sin 2t$ over een volledige periode zonder
absolute waarden $0$, niet $6$.
\emph{\omterm{def:b1:curves:conics}{Kegelsneden} moeten herleid worden voordat ze gelezen worden}: in
$9x^2 + 25y^2 - 36x - 50y - 164 = 0$ is noch de assen, noch het
middelpunt, noch de excentriciteit zichtbaar totdat de kwadraten zijn
vervolledigd (\cref{exo:b1:curves:6}); en een verdwijnend kwadratisch
deel op één variabele betekent parabool, geen ``ontaarde ellips''.
\end{remark}

\begin{remark}[Waar deze krommen naartoe gaan]
\label{rem:b1:curves:whereused}
\omterm{def:b1:curves:def}{Geparametriseerde krommen} zijn de taal van de mechanica: banen
zijn krommen, snelheidsvectoren zijn raakvectoren, en de
\omterm{def:b1:curves:arclength}{booglengte} van \cref{def:b1:curves:arclength} is de afgelegde afstand.
De \omterm{def:b1:curves:conics}{kegelsneden} duiken op overal waar een omgekeerd-kwadratische wet
werkt --- planeetbanen zijn ellipsen met de zon in een brandpunt. In
\cref{ch:b1:multivar} worden krommen de niveaukrommen van functies
van twee variabelen, en de raaklijn van dit hoofdstuk ontmoet de
gradiënt van het volgende. Het Jaar~2-volume voegt kromming toe en de
lokale canonieke vorm van een kromme; het weekendprobleem hieronder
haalt, met enkel het gereedschap van dit jaar, al alles eruit wat de
zeventiende eeuw over haar meest gevierde kromme wist.
\end{remark}

\begin{remark}[Perspectieven binnen Boek 3]
\label{rem:b1:curves:perspectives}
Dit hoofdstuk verbruikt de hele eerste helft van het volume en
voedt het laatste hoofdstuk. \emph{Verbruikt}: raakvectoren zijn
\omterm{def:b1:derivative:def}{afgeleiden} (\cref{ch:b1:derivative}), \omterm{def:b1:curves:arclength}{booglengte} en oppervlakten zijn
integralen (\cref{ch:b1:integration}), \omterm{ex:b1:curves:astroid}{keerpunten} worden geregeld door
Taylor-ontwikkelingen (\cref{ch:b1:taylor}), de tautochroon is een
lineaire differentiaalvergelijking (\cref{ch:b1:diffeq}), en elke
afstand en hoek is euclidisch (\cref{ch:b1:euclid}).
\emph{Gevoed}: in \cref{ch:b1:multivar} wordt een kromme $t \mapsto
(x(t), y(t))$ getekend binnen een niveaukromme $\{f = c\}$
gedifferentieerd door de kettingregel, en de resulterende identiteit
$\langle \nabla f,\ f'(t)\rangle = 0$ verbindt de raakvectoren van dit
hoofdstuk met de gradiënten van het volgende --- de raaklijn van de
kromme en de normaalrichting van het oppervlak zijn dezelfde berekening
gezien vanaf beide oevers.
\end{remark}

\begin{exercise}[$\star\star\star$]\label{exo:b1:curves:9}
De \emph{lemniscaat van Bernoulli} is de \omterm{def:b1:curves:polar}{poolkromme} $r^2 = \cos

  \item Geef haar domein, symmetrieën, raaklijnen in de oorsprong, en

  \item Bewijs dat ze de meetkundige plaats is van de punten $M$ met $MF

\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:10}
Bereken de totale \omterm{def:b1:curves:arclength}{lengte} van de cardioïde $r = 1 + \cos\theta$
met de poolformule van \cref{def:b1:curves:arclength}
\emph{(de identiteit $1 + \cos\theta = 2\cos^2\frac\theta2$ maakt
de vierkantswortel tot $2\abs{\cos\frac\theta2}$)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:11}
Toon aan dat de raaklijn aan de ellips $(a\cos t, b\sin t)$ in het
punt met parameter $t$ als vergelijking
\[
\frac{x\cos t}{a} + \frac{y\sin t}{b} = 1 ,
\]
heeft, en leid de raaklijn aan $\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1$
af in een punt $(x_0, y_0)$ van de ellips: $\frac{x\,x_0}{a^2} +
\frac{y\,y_0}{b^2} = 1$ (de ``splitsing van de kwadraten''-regel).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:curves:12}
De \emph{logaritmische spiraal} is de \omterm{def:b1:curves:polar}{poolkromme} $r =
\eu^{k\theta}$ ($k > 0$ vast, $\theta \in \R$).
\begin{enumerate}
  \item Toon aan dat de hoek $V$ tussen de straal en de
        raaklijn \emph{constant} is ($\tan V = \frac1k$) --- de
        spiraal kruist elke straal onder dezelfde hoek.
  \item Toon aan dat het roteren van de spiraal over een hoek $c$ ze
        afbeeldt op haar schaling met de factor $\eu^{-kc}$: elke
        rotatie van de spiraal is een vergroting ervan
        (zelfgelijkvormigheid).
  \item Bereken de \omterm{def:b1:curves:arclength}{lengte} van de boog $\theta \in
        \intoc{-\infty}{\theta_0}$ (als een limiet van \omterm{def:b1:curves:arclength}{lengten} op
        $\intcc{A}{\theta_0}$, $A \to -\infty$) en merk op dat
        ze eindig is: een kromme die oneindig vaak rond de oorsprong
        spiraalt, met eindige \omterm{def:b1:curves:arclength}{lengte}.
\end{enumerate}
\end{exercise}

\section{Probleem: de cycloïde, koningin der krommen}

\begin{problem}\label{pb:b1:curves:1}
Een wiel met straal $R$ rolt zonder slippen langs de $x$-as;
het punt van de \omterm{def:b1:topology:closure}{rand} dat aanvankelijk in de oorsprong ligt beschrijft de
\emph{cycloïde}. De zeventiende eeuw streed over deze kromme
--- Galilei woog papieren uitknipsels ervan, Wren mat ze,
Roberval berekende haar oppervlakte, Huygens bouwde er klokken op --- en
elk van hun resultaten ligt binnen bereik van dit hoofdstuk. We
bewijzen de vier klassiekers: de raaklijnconstructie, Wrens \omterm{def:b1:curves:arclength}{lengte}
$8R$, de oppervlakte $3\pi R^2$, en Huygens' tautochroon
eigenschap.\index{cycloïde}\index{tautochroon}

\medskip
\noindent\textbf{Deel I --- Het rollen en de raaklijn.}
\begin{enumerate}
  \item Nadat het wiel over een hoek $t$ is gedraaid, ligt zijn
        middelpunt in $\Omega(t) = (Rt, R)$ (rollen zonder slippen:
        contactafstand $=$ gerolde boog). Toon aan dat het gemarkeerde
        punt zich bevindt in
        \[
        M(t) = \bigl(R(t - \sin t),\; R(1 - \cos t)\bigr).
        \]
  \item Bereken $f'(t)$ en toon aan dat $\norm{f'(t)} =
        2R\,\abs{\sin\frac t2}$; lokaliseer de singuliere punten
        (\omterm{ex:b1:curves:astroid}{keerpunten} --- vgl.\ \cref{exo:b1:curves:2}) en de top van
        elke boog.
  \item Zij $C(t) = (Rt, 0)$ het contactpunt en $T(t) =
        (Rt, 2R)$ de top van het wiel. Bewijs dat voor $0 < t <
        2\pi$ de vector $M - C$ \emph{normaal} is aan de kromme in
        $M(t)$ en de vector $T - M$ \emph{rakend}: om de
        raaklijn aan een cycloïde te tekenen, verbind je het punt met de
        top van zijn rollende cirkel. \emph{(Factoriseer alles door
        $\sin\frac t2$ en $\cos\frac t2$.)}
  \item Interpreteer vraag 3 kinematisch: het contactpunt is
        het \emph{ogenblikkelijke rotatiecentrum}, en de
        snelheid van $M$ is gelijk aan zijn afstand tot $C$ (voor
        eenheidshoeksnelheid). Verifieer $\norm{M - C} =
        2R\abs{\sin\frac t2}$.
  \item Bewijs de hoogte--snelheid-relatie
        \[
        \norm{f'(t)}^2 = 2R\,y(t) :
        \]
        op een cycloïde doorlopen met eenheidshoeksnelheid is de
        snelheid in elk punt precies de vrijeval-snelheid voor een
        val gelijk aan de huidige hoogte. (Bewaar dit voor
        Deel IV.)
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Wrens stelling: de boog heeft \omterm{def:b1:curves:arclength}{lengte}
$8R$.}
\begin{enumerate}[resume]
  \item Bereken met \cref{def:b1:curves:arclength} de \omterm{def:b1:curves:arclength}{lengte}
        van één boog:
        \[
        L = \int_0^{2\pi} 2R\sin\frac t2\,\dd t = 8R
        \]
        (Wrens stelling, 1658). Vier wieldiameters, en nergens een
        $\pi$.
  \item Bereken de \omterm{def:b1:curves:arclength}{booglengte} vanaf het \omterm{ex:b1:curves:astroid}{keerpunt}: $s(t) =
        4R\bigl(1 - \cos\frac t2\bigr)$, en controleer $s(2\pi) =
        8R$.
  \item Meet nu de boog vanaf de \emph{top} $t = \pi$:
        $\sigma(t) = \abs{4R\cos\frac t2}$. Bewijs de intrinsieke
        relatie
        \[
        \sigma^2 = 8R\,\bigl(2R - y\bigr) :
        \]
        de gekwadrateerde boogafstand vanaf de top is evenredig met
        de hoogtedaling onder de top.
  \item Controles: herleid uit vraag 8 dat de halve boog
        van top tot \omterm{ex:b1:curves:astroid}{keerpunt} \omterm{def:b1:curves:arclength}{lengte} $4R$ heeft, en vergelijk met de
        astroïdeberekening van \cref{ex:b1:curves:circlelength}
        --- beide krommen hebben algebraïsche, $\pi$-vrije \omterm{def:b1:curves:arclength}{lengten};
        leg uit wat dit mogelijk maakt hoewel beide zijn
        opgebouwd uit cirkels. \emph{(Kijk naar de vorm van
        $\norm{f'}$.)}
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Robervals oppervlakte: $3\pi R^2$.}
\begin{enumerate}[resume]
  \item Verantwoord dat de oppervlakte tussen één boog en de grond
        $A = \int_0^{2\pi} y(t)\,x'(t)\,\dd t$ is \emph{(de
        substitutie $x = x(t)$ in $\int y\,\dd x$,
        \cref{thm:b1:integration:parts}; $x$ is stijgend)}.
  \item Bereken
        \[
        A = R^2\int_0^{2\pi}(1 - \cos t)^2\,\dd t = 3\pi R^2 :
        \]
        precies \emph{drie} keer de oppervlakte van het wiel ---
        de verhouding die Galilei door wegen had geraden.
  \item Dezelfde methode voor de astroïde $(\cos^3t, \sin^3t)$: toon
        aan dat de ingesloten oppervlakte $\frac{3\pi}8$ is
        \emph{(lineariseer $\sin^4 t\cos^2 t$; enkel de constante
        term overleeft over een volledige periode)}.
  \item Controleer de formule van vraag 10 op de bovenste
        eenheidshalvecirkel $(\cos t, \sin t)$, $t$ van $\pi$ naar $0$:
        geeft ze $\frac\pi2$ terug?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Huygens' tautochroon.} Keer de
boog om: een wrijvingsloze kraal glijdt, onder de zwaartekracht $g$,
binnen de cycloïdale schaal
\[
x(t) = R(t + \sin t), \qquad y(t) = R(1 - \cos t)
\qquad (t \in \intcc{-\pi}{\pi}),
\]
waarvan het laagste punt de oorsprong is ($y$ opwaarts gemeten).
\begin{enumerate}[resume]
  \item Bereken de snelheid $\norm{f'(t)} = 2R\cos\frac t2$, de
        \omterm{def:b1:curves:arclength}{booglengte} vanaf de bodem $s(t) = 4R\sin\frac t2$, en
        bewijs de sleutelidentiteit
        \[
        y = \frac{s^2}{8R} .
        \]
  \item De kraal losgelaten in rust vanuit het punt met parameter
        $t_0 > 0$ voldoet aan behoud van energie: als $s(\tau)$ zijn
        boogpositie op tijdstip $\tau$ aangeeft, dan $\frac12\bigl(
        \frac{\dd s}{\dd\tau}\bigr)^2 + g\,y = g\,y_0$. Herschrijf
        dit, met vraag 14, als
        \[
        \Bigl(\frac{\dd s}{\dd\tau}\Bigr)^{\!2}
        = \frac{g}{4R}\,\bigl(s_0^2 - s^2\bigr),
        \qquad s_0 = s(t_0).
        \]
  \item Differentieer naar $\tau$ en verkrijg de
        harmonische oscillator
        \[
        \frac{\dd^2 s}{\dd\tau^2} = -\frac{g}{4R}\,s ;
        \]
        los ze op met \cref{thm:b1:diffeq:homogeneous2} en de
        beginvoorwaarden: $s(\tau) = s_0\cos(\omega\tau)$,
        $\omega = \sqrt{g/4R}$.
  \item Leid de \emph{tautochroon eigenschap} af (Huygens, 1659):
        de tijd om de bodem te bereiken,
        \[
        T_{\downarrow} = \frac{\pi}{2\omega}
        = \pi\sqrt{\frac Rg}\,,
        \]
        hangt niet af van het losgelaten punt --- kralen die samen
        worden losgelaten vanuit twee willekeurige hoogten van de schaal
        komen samen aan.
  \item Verifieer door substitutie dat $s(\tau) =
        s_0\cos\omega\tau$ de eerste-orde-energievergelijking van
        vraag 15 exact voldoet (niet enkel de
        gedifferentieerde), en leg in één zin uit waarom een
        cirkelvormige slinger slechts \emph{bij benadering}
        isochroon is terwijl de cycloïde het exact is.
  \item Numeriek: welke straal $R$ maakt de daaltijd
        precies één seconde ($g = 9{,}81$)? Merk op hoe \omterm{def:b1:topology:dense}{dicht} het
        antwoord bij één meter ligt, en hoe de volledige
        oscillatieperiode $2\pi\sqrt{4R/g}$ zich verhoudt tot de
        kleinehoek-slingerformule $2\pi\sqrt{\ell/g}$ voor $\ell = 4R$.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Dividenden, en synthese.}
\begin{enumerate}[resume]
  \item Pas de raaklijnregel van vraag 3 toe bij $t =
        \frac\pi2$ (neem $R = 1$): bereken $M$, $T$, de
        richting van $MT$, en controleer ze tegen
        $f'(\frac\pi2)$.
  \item (Trochoïden) Markeer in plaats daarvan een punt op afstand $d$ van
        het middelpunt ($d \neq R$): de kromme is $x = Rt -
        d\sin t$, $y = R - d\cos t$. Toon aan dat voor $d < R$ de
        kromme overal regulier is en de grafiek is van geen
        singulier gedrag ($x' > 0$: ze gaat vooruit), terwijl voor $d
        > R$ de abscis $x'$ van teken wisselt en de kromme
        lussen maakt --- het geflensde spoorwegwiel waarvan de
        randpunten \emph{achteruit} reizen.
  \item (Brachistochroon-voorproefje) Vergelijk van het \omterm{ex:b1:curves:astroid}{keerpunt} $(\pi
        R, 2R)$ van de schaal tot de bodem de cycloïdale daaltijd
        $\pi\sqrt{R/g}$ met de tijd langs de rechte glijbaan
        die dezelfde punten verbindt \emph{(constante versnelling
        $g\sin\alpha$ langs de koorde)}: toon aan dat de koorde
        $\sqrt{\pi^2 + 4}\,\sqrt{R/g} \approx 3{,}72\sqrt{R/g}$ neemt.
        De kromme verslaat de rechte --- ze is in feite de snelste
        van alle krommen, een resultaat van de variatierekening.
  \item Toon aan dat op de boog ($0 < t < 2\pi$),
        \[
        \frac{\dd y}{\dd x} = \cot\frac t2,
        \qquad
        \frac{\dd^2 y}{\dd x^2} =
        -\frac{1}{4R\sin^4\frac t2} < 0 :
        \]
        de boog is concaaf, met verticale raaklijnen precies in de
        \omterm{ex:b1:curves:astroid}{keerpunten}.
  \item Herwin de verticale keerpuntraaklijn van
        \cref{exo:b1:curves:2} meetkundig: bereken de limiet
        richting van de koorde $MT$ van vraag 3 als $t \to
        0^{+}$, zonder enige ontwikkeling.
  \item Synthese, in vier zinnen: welke drie benoemde stellingen
        dit probleem bewees (met hun getallen $8R$, $3\pi
        R^2$, $\pi\sqrt{R/g}$ en hun auteurs); welk enkel
        rekenkundig hulpmiddel ($\sin\frac t2$, $\cos\frac t2$
        factorisaties) heel Deel I, II en IV aandreef; hoe
        de intrinsieke relatie $y = s^2/8R$ meetkunde omzette
        in een lineaire differentiaalvergelijking; en op welk hoofdstuk
        van dit boek elk Deel steunde.
\end{enumerate}
\end{problem}
