\chapter{Topologie van de reële rechte}\label{ch:b1:topology}

Limieten blijven verwijzen naar dezelfde meetkundige woordenschat:
punten ``\omterm{def:b1:topology:dense}{dicht} bij'' een \omterm{def:b1:logic:sets}{verzameling}, \omterm{def:b1:logic:sets}{verzamelingen} ``zonder
randlekken'', intervallen waaruit rijen niet kunnen ontsnappen. Dit
hoofdstuk legt die woordenschat vast --- \omterm{def:b1:topology:open}{open} en \omterm{def:b1:topology:closed}{gesloten
verzamelingen}, \omterm{def:b1:topology:closure}{inwendige} en \omterm{def:b1:topology:closure}{afsluiting}, \omterm{def:b1:topology:dense}{dichtheid} --- op de reële
rechte, en bewijst de compactheid van segmenten in haar sequentiële
vorm. Dezelfde begrippen, in genormeerde vectorruimten, zijn
tweedejaarsstof; op $\R$ liggen ze binnen bereik en zijn ze meteen
nuttig voor \cref{ch:b1:continuity}.

\section{Open verzamelingen, gesloten verzamelingen}

\begin{definition}[Omgeving, open verzameling]\label{def:b1:topology:open}
Een \omterm{def:b1:logic:sets}{verzameling} $V \subseteq \R$ is een \emph{omgeving}\index{omgeving}
van $x \in \R$ wanneer ze een \omterm{prop:b1:reals:intervals}{interval} $\intoo{x - r}{x + r}$ bevat
voor een zekere $r > 0$. Een \omterm{def:b1:logic:sets}{verzameling} $U \subseteq \R$ is
\emph{open}\index{open verzameling} wanneer ze een omgeving is van
elk van haar punten:
\[
\forall x \in U,\ \exists r > 0, \quad \intoo{x - r}{x + r}
\subseteq U .
\]
\end{definition}

\begin{example}\label{ex:b1:topology:open}
\omterm{def:b1:topology:open}{Open} intervallen zijn \omterm{def:b1:topology:open}{open}: voor $x \in \intoo{a}{b}$ neem je $r =
\min(x - a,\, b - x) > 0$. Halve rechten $\intoo{a}{+\infty}$ zijn
\omterm{def:b1:topology:open}{open}; $\R$ en $\emptyset$ zijn \omterm{def:b1:topology:open}{open} (de laatste vacuüm).
$\intcc{0}{1}$ is \emph{niet} \omterm{def:b1:topology:open}{open}: geen enkel \omterm{prop:b1:reals:intervals}{interval} rond $0$
blijft binnen.
\end{example}

\begin{proposition}[Stabiliteit van open verzamelingen]\label{prop:b1:topology:openstable}
Elke unie van \omterm{def:b1:topology:open}{open verzamelingen} is \omterm{def:b1:topology:open}{open}; een \emph{eindige}
doorsnede van \omterm{def:b1:topology:open}{open verzamelingen} is \omterm{def:b1:topology:open}{open}. Oneindige doorsneden kunnen
mislukken: $\bigcap_{n \geq 1} \intoo{-\frac 1n}{\frac 1n} = \{0\}$,
niet \omterm{def:b1:topology:open}{open}.
\end{proposition}

\begin{proof}
Unie: als $x \in \bigcup_i U_i$, dan $x \in U_{i_0}$ voor een zekere
$i_0$, en het \omterm{prop:b1:reals:intervals}{interval} geleverd door $U_{i_0}$ ligt binnen de unie.
Eindige doorsnede: als $x \in U_1 \cap \dots \cap U_k$, neem $r =
\min(r_1, \dots, r_k) > 0$ van de stralen geleverd door elke $U_j$.
Voor het tegenvoorbeeld: elk \omterm{prop:b1:reals:intervals}{interval} rond $0$ bevat een zekere
$\frac{1}{n}$ (Archimedes), en verlaat dus de doorsnede.
\end{proof}

\begin{example}[Openheid certificeren met expliciete stralen]\label{ex:b1:topology:radii}
Is $U = \{x \in \R : x^2 > 2\}$ \omterm{def:b1:topology:open}{open}? Ja, en het certificaat kan
opgeschreven worden: $U = \intoo{-\infty}{-\sqrt2} \cup
\intoo{\sqrt2}{+\infty}$, een unie van twee \omterm{def:b1:topology:open}{open} halve rechten, \omterm{def:b1:topology:open}{open}
volgens \cref{prop:b1:topology:openstable}. Alternatief, redeneer
punt per punt: voor $x \in U$ met $x > \sqrt 2$, neem $r = x - \sqrt2
> 0$: elke $y \in \intoo{x - r}{x + r}$ voldoet aan $y > \sqrt 2$,
dus $y^2 > 2$; symmetrisch aan de linkerkant. Beide stijlen zijn van
belang --- de structurele (bouw op uit bekende \omterm{def:b1:topology:open}{open verzamelingen} via
unies en eindige doorsneden) schaalt beter, de $\varepsilon$-stijl
werkt wanneer er geen structuur zichtbaar is; en
\cref{ch:b1:continuity} zal een derde toevoegen, de krachtigste: $U$
is het \omterm{def:b1:logic:map}{origineel} van de \omterm{def:b1:topology:open}{open} $\intoo{2}{+\infty}$ onder de continue
$x \mapsto x^2$.
\end{example}

\begin{definition}[Gesloten verzameling]\label{def:b1:topology:closed}
Een \omterm{def:b1:logic:sets}{verzameling} $F \subseteq \R$ is \emph{gesloten}\index{gesloten
verzameling} wanneer haar complement $\R \setminus F$ \omterm{def:b1:topology:open}{open} is. Volgens
de wetten van de Morgan en \cref{prop:b1:topology:openstable}: elke
doorsnede van gesloten \omterm{def:b1:logic:sets}{verzamelingen} is gesloten, eindige unies van
gesloten \omterm{def:b1:logic:sets}{verzamelingen} zijn gesloten.
\end{definition}

\begin{theorem}[Sequentiële karakterisering van gesloten verzamelingen]\label{thm:b1:topology:seqclosed}
$F$ is \omterm{def:b1:topology:closed}{gesloten} als en slechts als: voor elke rij $(u_n)$ van punten
van $F$ die convergeert naar een zekere $\ell \in \R$, de limiet
$\ell$ tot $F$ behoort. (``\omterm{def:b1:topology:closed}{Gesloten}'' $=$ ``stabiel onder
limieten''.)
\end{theorem}

\begin{proof}
($\Rightarrow$) Zij $F$ \omterm{def:b1:topology:closed}{gesloten}, $u_n \in F$, $u_n \to \ell$, en
veronderstel $\ell \notin F$. Het complement is \omterm{def:b1:topology:open}{open}: een zekere
$\intoo{\ell - r}{\ell + r}$ vermijdt $F$. Maar convergentie plaatst
$u_n$ in dat \omterm{prop:b1:reals:intervals}{interval} voor grote $n$: tegenspraak met $u_n \in F$.

($\Leftarrow$) Veronderstel dat $F$ niet \omterm{def:b1:topology:closed}{gesloten} is: het complement
is niet \omterm{def:b1:topology:open}{open}, dus een zekere $x \notin F$ heeft geen \omterm{prop:b1:reals:intervals}{interval}
$\intoo{x - r}{x + r}$ binnen het complement; door $r = \frac{1}{n+1}$
te nemen, kies $u_n \in F$ met $\abs{u_n - x} < \frac{1}{n+1}$. Dan
$u_n \in F$, $u_n \to x \notin F$: de sequentiële eigenschap
mislukt.
\end{proof}

\begin{example}[De sequentiële test, beide richtingen]\label{ex:b1:topology:seqtest}
\emph{\omterm{def:b1:topology:closed}{Gesloten}:} $F = \Z \cup \bigl\{n + \frac1n : n \geq 2\bigr\}$.
Zij $u_k \in F$ met $u_k \to \ell$. Het venster $\intcc{\ell -
1}{\ell + 1}$ bevat slechts eindig veel punten van $F$ (eindig veel
gehele getallen, eindig veel $n + \frac1n$), en vanaf een zekere rang
liggen alle $u_k$ erin: de rij neemt dan eindig veel waarden aan, en,
convergerend, is uiteindelijk constant (zoals in
\cref{exo:b1:topology:3}): $\ell \in F$. \omterm{def:b1:topology:closed}{Gesloten} --- ook al bevat
$F$ paren punten op afstand $\frac1n$, willekeurig dichtbij.

\emph{Niet \omterm{def:b1:topology:closed}{gesloten}:} $G = \bigl\{\frac1m + \frac1n : m, n \in
\N^*\bigr\}$. De rij $\frac1n + \frac1n \in G$ streeft naar $0$, en $0
\notin G$ (een som van twee positieve termen): de sequentiële test
mislukt, $G$ is niet \omterm{def:b1:topology:closed}{gesloten}. Interessant is dat elke $\frac1m$
\emph{wel} tot $\overline G \cap G$ behoort: inderdaad $\frac1m =
\frac{1}{m+1} + \frac{1}{m(m+1)} \in G$. Het afsluitende inzicht: om
geslotenheid te bewijzen, beheers \emph{alle} convergente rijen
tegelijk (meestal via een lokale eindigheid of een argument met
\omterm{def:b1:topology:closed}{gesloten} formule); om ze te weerleggen, volstaat één goedgekozen
ontsnappende rij --- de asymmetrie maakt de negatieve richting de
gemakkelijke, en de tegenvoorbeelden van dit hoofdstuk hebben allemaal
deze eenregelige vorm.
\end{example}

\begin{example}\label{ex:b1:topology:closed}
Segmenten $\intcc{a}{b}$, halve rechten $\intco{a}{+\infty}$, \omterm{def:b1:counting:card}{eindige
verzamelingen}, $\Z$ (een convergente rij van gehele getallen is
uiteindelijk constant) zijn \omterm{def:b1:topology:closed}{gesloten}. $\intoc{0}{1}$ is noch \omterm{def:b1:topology:open}{open}
(faalt in $1$) noch \omterm{def:b1:topology:closed}{gesloten} ($\frac 1n \to 0 \notin$ de
\omterm{def:b1:logic:sets}{verzameling}): de meeste \omterm{def:b1:logic:sets}{verzamelingen} zijn geen van beide. $\R$ en
$\emptyset$ zijn zowel \omterm{def:b1:topology:open}{open} als \omterm{def:b1:topology:closed}{gesloten} --- en zij zijn de enige
zulke deelverzamelingen van $\R$ (\cref{exo:b1:topology:9}).
\end{example}

\begin{example}[Een open verzameling samengesteld uit oneindig veel stukken]\label{ex:b1:topology:rminusz}
$\R \setminus \Z = \bigcup_{n \in \Z} \intoo{n}{n+1}$: een oneindige
unie van \omterm{def:b1:topology:open}{open} intervallen, \omterm{def:b1:topology:open}{open} volgens
\cref{prop:b1:topology:openstable} --- dus $\Z$ is \omterm{def:b1:topology:closed}{gesloten} zonder dat
er een sequentieel argument nodig is. Merk de arbeidsverdeling in de
stabiliteitsregels op: \emph{unies} van \omterm{def:b1:topology:open}{open verzamelingen} mogen
willekeurig zijn (elk punt heeft enkel zijn eigen certificaat nodig,
geleverd door de ene \omterm{def:b1:logic:sets}{verzameling} die het bevat), terwijl
\emph{doorsneden} eindig moeten blijven (certificaten moeten
doorsneden worden, en oneindig veel stralen kunnen tot niets
krimpen). \Cref{exo:b1:topology:10} zal aantonen dat dit voorbeeld de
algemene vorm is: elke \omterm{def:b1:topology:open}{open} deelverzameling van $\R$ is een aftelbare
disjuncte unie van \omterm{def:b1:topology:open}{open} intervallen.
\end{example}

\section{Inwendige, afsluiting, dichtheid}

\begin{definition}[Inwendige, afsluiting, rand]\label{def:b1:topology:closure}
Zij $A \subseteq \R$.
\begin{itemize}
  \item Een punt $x$ is \emph{inwendig} aan $A$ wanneer $A$ een
        \omterm{def:b1:topology:open}{omgeving} van $x$ is; het \emph{inwendige}
        $\mathring{A}$\index{inwendige} is de \omterm{def:b1:logic:sets}{verzameling} van de
        inwendige punten.
  \item Een punt $x$ is \emph{aanklevend} aan $A$ wanneer elke
        \omterm{def:b1:topology:open}{omgeving} van $x$ $A$ ontmoet; de \emph{afsluiting}
        $\overline{A}$\index{afsluiting} is de \omterm{def:b1:logic:sets}{verzameling} van de
        aanklevende punten.
  \item De \emph{rand} is $\partial A = \overline A \setminus
        \mathring A$.\index{rand}
\end{itemize}
Dan is $\mathring A \subseteq A \subseteq \overline A$.
\end{definition}

\begin{proposition}[Hoofdeigenschappen]\label{prop:b1:topology:closureprops}
\begin{enumerate}
  \item $\mathring A$ is de grootste \omterm{def:b1:topology:open}{open verzameling} bevat in $A$;
        $A$ is \omterm{def:b1:topology:open}{open} desda $A = \mathring A$.
  \item $\overline A$ is de kleinste \omterm{def:b1:topology:closed}{gesloten verzameling} die $A$
        bevat; $A$ is \omterm{def:b1:topology:closed}{gesloten} desda $A = \overline A$.
  \item (Sequentiële karakterisering van aankleving) $x \in \overline
        A$ als en slechts als $x$ de limiet is van een rij van punten
        van $A$.
  \item Complementering wisselt de begrippen om: $\R \setminus
        \overline A = \bigl(\R \setminus A\bigr)^{\!\circ}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(4) $x \notin \overline A$ $\iff$ een zekere \omterm{def:b1:topology:open}{omgeving} van $x$ vermijdt
$A$ $\iff$ een zeker \omterm{prop:b1:reals:intervals}{interval} rond $x$ ligt in $\R \setminus A$ $\iff$
$x$ is inwendig aan $\R \setminus A$.

(1) $\mathring A$ is \omterm{def:b1:topology:open}{open}: als $x \in \mathring A$, dan een zekere
$\intoo{x-r}{x+r} \subseteq A$; elk punt $y$ van dat \omterm{prop:b1:reals:intervals}{interval} heeft
een kleiner \omterm{prop:b1:reals:intervals}{interval} eromheen dat erin ligt, dus in $A$: het hele
\omterm{prop:b1:reals:intervals}{interval} ligt in $\mathring A$. Elke \omterm{def:b1:topology:open}{open} $U \subseteq A$ bestaat uit
\omterm{def:b1:topology:closure}{inwendige} punten van $A$, dus $U \subseteq \mathring A$: grootste. De
karakterisering van openheid volgt.

(2) In detail. Volgens (4) is $\R \setminus \overline A$ het \omterm{def:b1:topology:closure}{inwendige}
van $\R \setminus A$, een \omterm{def:b1:topology:open}{open verzameling} volgens (1): dus $\overline
A$ is \omterm{def:b1:topology:closed}{gesloten}, en het bevat $A$. Minimaliteit: zij $F \supseteq A$
\omterm{def:b1:topology:closed}{gesloten}. Dan is $\R \setminus F$ \omterm{def:b1:topology:open}{open} en bevat in $\R \setminus A$,
dus volgens de maximaliteit in (1),
\[
\R \setminus F \subseteq \bigl(\R \setminus A\bigr)^{\!\circ}
= \R \setminus \overline A ,
\]
en door opnieuw de complementen te nemen: $\overline A \subseteq F$.
Dus $\overline A$ is de kleinste \omterm{def:b1:topology:closed}{gesloten} superverzameling.
Karakterisering: als $A = \overline A$ dan is $A$ \omterm{def:b1:topology:closed}{gesloten} (net
aangetoond); als $A$ \omterm{def:b1:topology:closed}{gesloten} is, is het zelf een \omterm{def:b1:topology:closed}{gesloten}
superverzameling van $A$, dus minimaliteit dwingt $\overline A
\subseteq A$ af, en gelijkheid.

(3) Als $u_n \in A$, $u_n \to x$: elke \omterm{def:b1:topology:open}{omgeving} van $x$ bevat een
zekere $u_n \in A$, dus $x \in \overline A$. Omgekeerd, als $x \in
\overline A$: elk \omterm{prop:b1:reals:intervals}{interval} $\intoo{x - \frac{1}{n+1}}{x +
\frac{1}{n+1}}$ ontmoet $A$ in een zekere $u_n$, en $u_n \to x$.
\end{proof}

\begin{example}\label{ex:b1:topology:closureexamples}
$\overline{\intoo{0}{1}} = \intcc{0}{1}$;
$\mathring{\intcc{0}{1}} = \intoo{0}{1}$; $\partial\intoo{0}{1} =
\{0, 1\}$. Voor $A = \{\frac 1n : n \in \N^*\}$: $\overline A = A
\cup \{0\}$, $\mathring A = \emptyset$, $\partial A = A \cup \{0\}$.
Voor $\Q$: door \omterm{def:b1:topology:dense}{dichtheid} (\cref{thm:b1:reals:density}) is elk reëel
getal aanklevend aan $\Q$, dus $\overline{\Q} = \R$ terwijl
$\mathring{\Q} = \emptyset$ (elk \omterm{prop:b1:reals:intervals}{interval} bevat irrationale getallen):
de \omterm{def:b1:topology:closure}{rand} van $\Q$ is heel $\R$.
\end{example}

\begin{example}[Een volledige anatomie]\label{ex:b1:topology:anatomy}
Zij $A = \intoc{0}{1} \,\cup\, \bigl(\Q \cap \intoo{2}{3}\bigr)
\,\cup\, \{4\}$. We berekenen de drie \omterm{def:b1:logic:sets}{verzamelingen} van
\cref{def:b1:topology:closure}, stuk voor stuk.

\emph{\omterm{def:b1:topology:closure}{Inwendige}.} Een punt van $\intoo{0}{1}$ heeft een heel \omterm{prop:b1:reals:intervals}{interval}
binnen $A$: inwendig. Het punt $1$: elk \omterm{prop:b1:reals:intervals}{interval} eromheen lekt rechts
van $1$, waar $A$ niets heeft tot $2$: niet inwendig. Geen enkel punt
van $\Q \cap \intoo{2}{3}$ is inwendig (elk \omterm{prop:b1:reals:intervals}{interval} bevat irrationale
getallen, \cref{thm:b1:reals:density}); evenmin de geïsoleerde $4$.
Dus $\mathring A = \intoo{0}{1}$.

\emph{\omterm{def:b1:topology:closure}{Afsluiting}.} Limieten van punten van $A$: heel $\intcc{0}{1}$
($0 = \lim \frac1n$ met $\frac 1n \in A$); heel $\intcc{2}{3}$ (elk
reëel getal daar is een limiet van rationale getallen van het
\omterm{prop:b1:reals:intervals}{interval}, opnieuw \omterm{def:b1:topology:dense}{dichtheid}); en $4$. Niets anders: een punt buiten
$\intcc{0}{1} \cup \intcc{2}{3} \cup \{4\}$ heeft positieve afstand
tot die \omterm{def:b1:topology:closed}{gesloten verzameling}. Dus $\overline A = \intcc{0}{1} \cup
\intcc{2}{3} \cup \{4\}$.

\emph{\omterm{def:b1:topology:closure}{Rand}.} $\partial A = \overline A \setminus \mathring A
= \{0, 1\} \cup \intcc{2}{3} \cup \{4\}$.

Het afsluitende inzicht: de drie bewerkingen werken \emph{lokaal} ---
elk stuk van $A$ draagt bij volgens zijn eigen aard (een vol \omterm{prop:b1:reals:intervals}{interval}
behoudt zijn binnenkant, een dicht-maar-poreus stuk wordt volledig
\omterm{def:b1:topology:closure}{rand}, een geïsoleerd punt is pure \omterm{def:b1:topology:closure}{rand}), en een tweeregelige tekening
van $A$ voorspelt elk antwoord voordat er enig bewijs geschreven is.
\end{example}

\begin{remark}[Veelvoorkomende valkuilen in puntverzamelingsredeneringen]
(i) \emph{``Niet \omterm{def:b1:topology:open}{open}'' betekent niet ``\omterm{def:b1:topology:closed}{gesloten}''}: de meeste
\omterm{def:b1:logic:sets}{verzamelingen} zijn geen van beide ($\intoc{0}{1}$), en twee
\omterm{def:b1:logic:sets}{verzamelingen} zijn beide ($\emptyset$, $\R$) --- \omterm{def:b1:topology:open}{open} en \omterm{def:b1:topology:closed}{gesloten} zijn
geen tegengestelden maar duaal via complementering. (ii)
\emph{\omterm{def:b1:topology:closure}{Inwendige} en \omterm{def:b1:topology:closure}{afsluiting} commuteren niet}: voor $A = \Q$,
\[
\overline{\mathring A} = \overline\emptyset = \emptyset
\qquad\text{terwijl}\qquad
\bigl(\,\overline A\,\bigr)^{\!\circ} = \mathring \R = \R :
\]
de twee geïtereerde operatoren verschillen zo veel als \omterm{def:b1:logic:sets}{verzamelingen}
maar kunnen. (iii) \emph{Oneindige unies van \omterm{def:b1:topology:closed}{gesloten verzamelingen}
kunnen niet \omterm{def:b1:topology:closed}{gesloten} zijn}: $\bigcup_{n\geq1} \intcc{\frac1n}{1} =
\intoc{0}{1}$ --- de spiegel van het doorsnede-tegenvoorbeeld van
\cref{prop:b1:topology:openstable}. (iv) \emph{\omterm{def:b1:topology:dense}{Dicht} betekent niet
groot}: $\Q$ is \omterm{def:b1:topology:dense}{dicht}, aftelbaar, met leeg \omterm{def:b1:topology:closure}{inwendige}, en zijn
complement is ook \omterm{def:b1:topology:dense}{dicht}; \omterm{def:b1:topology:dense}{dichtheid} zegt ``willekeurig \omterm{def:b1:topology:dense}{dicht} bij
alles'', niet ``bijna alles'' --- de \omterm{pb:b1:topology:1}{Cantorverzameling} van het
weekendvraagstuk (\cref{pb:b1:topology:1}) maakt het tegengestelde
punt, een topologisch kleine \omterm{def:b1:logic:sets}{verzameling} die overaftelbaar groot is.
\end{remark}

\begin{example}[Een afsluiting exact berekend]\label{ex:b1:topology:closurecomputed}
Zij $G = \bigl\{\frac1m + \frac1n : m, n \in \N^*\bigr\}$ (uit
\cref{ex:b1:topology:seqtest}). Bewering:
\[
\overline G = G \,\cup\, \Bigl\{\frac1m : m \in \N^*\Bigr\}
\,\cup\, \{0\} .
\]
($\supseteq$) $\frac1m = \lim_n \bigl(\frac1m + \frac1n\bigr)$ en $0 =
\lim_n \frac2n$: aanklevend volgens de sequentiële karakterisering.
($\subseteq$) Zij $x = \lim_k \bigl( \frac{1}{m_k} +
\frac{1}{n_k}\bigr)$; orden elk paar zo dat $m_k \leq n_k$. Als
$(m_k)$ onbegrensd is, heeft een \omterm{def:b1:seq:subsequence}{deelrij} $m_k \to \infty$, dus $n_k
\to \infty$ ook en $x = 0$. Anders neemt $(m_k)$ eindig veel waarden
aan, een ervan, zeg $m$, oneindig vaak; langs die \omterm{def:b1:seq:subsequence}{deelrij}
$\frac{1}{n_k} \to x - \frac1m$: als $(n_k)$ begrensd is, neemt ze een
zekere waarde $n$ oneindig vaak aan en $x = \frac1m + \frac1n \in G$;
zo niet, $x = \frac1m$. Elk geval landt in de aangekondigde
\omterm{def:b1:logic:sets}{verzameling}. Het afsluitende inzicht: een \omterm{def:b1:topology:closure}{afsluiting} berekenen is een
gevalsanalyse in compactheidsstijl op indices --- begrensde index
betekent eindig veel waarden (duiventil), onbegrensde index betekent
dat een limiet ontsnapt --- en het antwoord toont de typische
tweelaagse structuur van limietpunten: de \omterm{def:b1:logic:sets}{verzameling}, haar
eerstegeneratielimieten, en hun limiet $0$.
\end{example}

\begin{definition}[Dichtheid, topologische vorm]\label{def:b1:topology:dense}
$A$ is \emph{dicht}\index{dichtheid} in $\R$ wanneer $\overline A =
\R$ --- equivalent, elk niet-leeg \omterm{def:b1:topology:open}{open} \omterm{prop:b1:reals:intervals}{interval} ontmoet $A$;
equivalent (volgens \cref{prop:b1:topology:closureprops}~(3)) is elk
reëel getal een limiet van elementen van $A$. Voorbeelden: $\Q$, $\R
\setminus \Q$, de dyadische getallen (\cref{exo:b1:reals:8}), dichte
\omterm{def:b1:structures:subgroup}{deelgroepen} (\cref{exo:b1:reals:9}).
\end{definition}

\begin{example}[Dichtheid is relatief]\label{ex:b1:topology:relativedensity}
``\omterm{def:b1:topology:dense}{Dicht}'' zoals hier gedefinieerd betekent \omterm{def:b1:topology:dense}{dicht} \emph{in $\R$}; een
\omterm{def:b1:logic:sets}{verzameling} kan in plaats daarvan \omterm{def:b1:topology:dense}{dicht} zijn in slechts een deel van
de rechte. De dyadische getallen van $\intcc{0}{1}$, d.w.z.\ $D \cap
\intcc{0}{1}$ (\cref{exo:b1:reals:8}), ontmoeten elk \omterm{def:b1:topology:open}{open} \omterm{prop:b1:reals:intervals}{interval}
opgenomen in $\intcc{0}{1}$ maar missen natuurlijk $\intoo{2}{3}$
volledig: zij zijn \omterm{def:b1:topology:dense}{dicht} \emph{in} $\intcc{0}{1}$, wat betekent
$\overline{D \cap \intcc{0}{1}} = \intcc{0}{1}$. De algemene
uitdrukking ``$A$ is \omterm{def:b1:topology:dense}{dicht} in $B$'' is een afkorting voor $B
\subseteq \overline A$ --- noem altijd de omvattende \omterm{def:b1:logic:sets}{verzameling},
aangezien de eindpunten van het weekendvraagstuk \omterm{def:b1:topology:dense}{dicht} zijn \emph{in
de \omterm{pb:b1:topology:1}{Cantorverzameling}} terwijl ze nergens \omterm{def:b1:topology:dense}{dicht} zijn \emph{in $\R$}:
dezelfde \omterm{def:b1:logic:sets}{verzameling}, twee waarheidsgetrouwe en tegengesteld klinkende
beschrijvingen.
\end{example}

\begin{example}[Dichtheid hanteren]\label{ex:b1:topology:densitytricks}
Drie snelle bewegingen die voortdurend terugkeren. \emph{Vergroten}:
als $A$ \omterm{def:b1:topology:dense}{dicht} is en $A \subseteq B$, dan is $B$ \omterm{def:b1:topology:dense}{dicht} (elk \omterm{prop:b1:reals:intervals}{interval}
ontmoet reeds $A$). \emph{Transporteren}: als $A$ \omterm{def:b1:topology:dense}{dicht} is, dan is
$\lambda A + \mu$ dat ook voor $\lambda \neq 0$ --- een \omterm{prop:b1:reals:intervals}{interval} $I$
ontmoet $\lambda A + \mu$ desda het \omterm{prop:b1:reals:intervals}{interval} $\frac{I - \mu}{\lambda}$
$A$ ontmoet; dus de oneven veelvouden van $10^{-9}$, bijvoorbeeld,
zijn \omterm{def:b1:topology:dense}{dicht}. \emph{Doorsnijden mislukt}: twee dichte \omterm{def:b1:logic:sets}{verzamelingen}
kunnen elkaar volledig missen ($\Q$ en $\R \setminus \Q$): \omterm{def:b1:topology:dense}{dichtheid}
overleeft unies en affiene \omterm{def:b1:logic:map}{afbeeldingen}, nooit doorsneden.
\end{example}

\section{Compactheid van segmenten}

\begin{theorem}[Segmenten zijn sequentieel compact]\label{thm:b1:topology:compact}
Zij $a \leq b$. Elke rij van punten van $\intcc{a}{b}$ heeft een
\omterm{def:b1:seq:subsequence}{deelrij} die convergeert naar een punt \emph{van}
$\intcc{a}{b}$.\index{compactheid}

Algemener zijn de deelverzamelingen van $\R$ met deze eigenschap (elke
rij heeft een \omterm{def:b1:seq:subsequence}{deelrij} die in de \omterm{def:b1:logic:sets}{verzameling} convergeert) precies de
\emph{\omterm{def:b1:topology:closed}{gesloten} en begrensde} \omterm{def:b1:logic:sets}{verzamelingen}.
\end{theorem}

\begin{proof}
Een rij in $\intcc{a}{b}$ is begrensd, dus Bolzano--Weierstrass
(\cref{thm:b1:seq:bw}) extraheert een convergente \omterm{def:b1:seq:subsequence}{deelrij}; haar
limiet blijft in $\intcc{a}{b}$ omdat segmenten \omterm{def:b1:topology:closed}{gesloten} zijn
(\cref{thm:b1:topology:seqclosed}).

Algemeen geval. \emph{(\omterm{def:b1:topology:closed}{Gesloten} begrensd $\Rightarrow$ compact)}: zij
$F$ \omterm{def:b1:topology:closed}{gesloten} en begrensd, en $(u_n)$ een rij in $F$. De begrensdheid
van $F$ begrenst de rij, dus Bolzano--Weierstrass extraheert
$u_{\varphi(n)} \to \ell$; en $\ell \in F$ omdat $F$ \omterm{def:b1:topology:closed}{gesloten} is en de
\omterm{def:b1:seq:subsequence}{deelrij} een convergente rij van punten van $F$ is
(\cref{thm:b1:topology:seqclosed}): de twee hypothesen worden elk één
keer verbruikt, begrensdheid voor het bestaan van de limiet,
geslotenheid voor het lidmaatschap ervan. \emph{(Compact $\Rightarrow$
\omterm{def:b1:topology:closed}{gesloten} en begrensd)}: als $F$ onbegrensd is, kies $u_n \in F$ met
$\abs{u_n} \geq n$; elke \omterm{def:b1:seq:subsequence}{deelrij} is onbegrensd, dus divergent
(\cref{prop:b1:seq:first}): helemaal geen convergente \omterm{def:b1:seq:subsequence}{deelrij}. Als $F$
niet \omterm{def:b1:topology:closed}{gesloten} is, neem $u_n \in F$ met $u_n \to \ell \notin F$
(\cref{thm:b1:topology:seqclosed}): elke \omterm{def:b1:seq:subsequence}{deelrij} convergeert naar
$\ell \notin F$, dus geen enkele \omterm{def:b1:seq:subsequence}{deelrij} convergeert \emph{in} $F$.
\end{proof}

\begin{example}[Geneste compacte verzamelingen]\label{ex:b1:topology:nested}
Een eerste oefening voor de stelling. Zij $K_0 \supseteq K_1
\supseteq K_2 \supseteq \dots$ niet-lege compacte (\omterm{def:b1:topology:closed}{gesloten} begrensde)
deelverzamelingen van $\R$. Dan $\bigcap_n K_n \neq \emptyset$.
Inderdaad, kies $x_n \in K_n$ voor elke $n$: de rij leeft in de
compacte $K_0$, dus een \omterm{def:b1:seq:subsequence}{deelrij} $x_{\varphi(n)}$ convergeert naar een
zekere $x$ (\cref{thm:b1:topology:compact}). Voor elke vaste $m$
liggen de termen $x_{\varphi(n)}$ met $\varphi(n) \geq m$ allemaal in
de \omterm{def:b1:topology:closed}{gesloten verzameling} $K_m$, dus de limiet $x$ ligt in $K_m$
(\cref{thm:b1:topology:seqclosed}); aangezien $m$ willekeurig was, $x
\in \bigcap_n K_n$. Een nuttige metgezel: als een \omterm{def:b1:topology:open}{open verzameling} $U$
$\bigcap_n K_n$ bevat, dan $U \supseteq K_n$ voor een zekere $n$ ---
pas hetzelfde argument toe op punten $x_n \in K_n \setminus U$; de
limiet $x$ zou in $\bigcap K_n \subseteq U$ liggen, maar $U$ \omterm{def:b1:topology:open}{open}
dwingt $x_{\varphi(n)} \in U$ af uiteindelijk, een tegenspraak. Beide
uitspraken mislukken zonder compactheid: $\bigcap_n
\intoo{0}{\frac 1n} = \emptyset$ en $\bigcap_n \intco{n}{+\infty} =
\emptyset$. Het afsluitende inzicht: compactheid zet een oneindige
keten van niet-leegheidsbeweringen om in één enkel limietpunt --- het
is het gereedschap dat de overgang naar de oneindige doorsnede
overleeft, en het weekendvraagstuk (\cref{pb:b1:topology:1}) zal er
tweemaal op steunen.
\end{example}

\begin{example}[Randen onder unies]\label{ex:b1:topology:boundaryunion}
Altijd $\partial(A \cup B) \subseteq \partial A \cup \partial B$: een
punt van $\partial(A \cup B)$ heeft elke \omterm{def:b1:topology:open}{omgeving} die $A \cup B$
ontmoet (dus $A$ of $B$, oneindig vaak een van beide) en die het
complement van $A \cup B$ ontmoet, dat in beide complementen ligt ---
een korte controle plaatst het punt dan in $\partial A$ of $\partial
B$. De inclusie kan spectaculair strikt zijn: met $A = \Q$ en $B =
\R\setminus\Q$,
\[
\partial(A \cup B) = \partial \R = \emptyset ,
\qquad
\partial A \cup \partial B = \R \cup \R = \R :
\]
twee gerafelde \omterm{def:b1:logic:sets}{verzamelingen} kunnen samen tot één naadloze lijmen,
waarbij hun \omterm{def:b1:topology:closure}{randen} elkaar vernietigen. Het afsluitende inzicht:
\omterm{def:b1:topology:closure}{inwendigen} en \omterm{def:b1:topology:closure}{afsluitingen} gedragen zich \emph{monotoon} onder unies
en doorsneden, maar \omterm{def:b1:topology:closure}{randen} niet --- behandel $\partial$ als een
afgeleide grootheid ($\overline A \setminus \mathring A$), nooit als
een operator met een eigen algebra.
\end{example}

\begin{remark}[Perspectieven binnen dit boekdeel]
De hier opgebouwde woordenschat wordt in dit boek nog tweemaal
verbruikt. In \cref{ch:b1:continuity} is elke stelling een
\omterm{def:b1:logic:statement}{topologie-uitspraak} in vermomming: de tussenwaardestelling zegt dat
continue \omterm{def:b1:logic:map}{afbeeldingen} de intervaleigenschap behouden, de
extremewaardestelling zegt dat ze compactheid behouden --- en de
bewijzen roepen \cref{thm:b1:topology:seqclosed,thm:b1:topology:compact}
bij naam op. In \cref{ch:b1:multivar} worden dezelfde definities
herlezen in $\R^2$ met schijven in plaats van intervallen: \omterm{def:b1:topology:open}{open
verzamelingen}, \omterm{def:b1:topology:closure}{afsluitingen} en compactheid dragen woord voor woord
over, en de extremewaardestelling in twee veranderlijken rijdt opnieuw
op Bolzano--Weierstrass (extraheer op elke coördinaat). Het ene begrip
dat \emph{niet} pijnloos veralgemeent is het \omterm{prop:b1:reals:intervals}{interval} zelf --- in het
vlak vervangt samenhang de convexiteit, een verhaal dat begint met
\cref{exo:b1:topology:9}'s ``enkel $\emptyset$ en $\R$ zijn \omterm{def:b1:topology:open}{open} en
\omterm{def:b1:topology:closed}{gesloten}''.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b1:topology:1}
Zeg voor elke \omterm{def:b1:logic:sets}{verzameling} of ze \omterm{def:b1:topology:open}{open}, \omterm{def:b1:topology:closed}{gesloten}, beide of geen van
beide is (met verantwoording): $\intoo{0}{1} \cup \intoo{2}{3}$;
$\;\intco{0}{1}$; $\;\{0\} \cup \intcc{1}{2}$; $\;\R \setminus \Z$;
$\;\Q \cap \intoo{0}{1}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:topology:2}
Bepaal $\mathring A$, $\overline A$ en $\partial A$ voor: $A =
\intoc{0}{1} \cup \{2\}$; $\;A = \R \setminus \Q$; $\;A =
\bigl\{\frac{(-1)^n n}{n+1} : n \in \N\bigr\}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:topology:3}
Bewijs dat een \omterm{def:b1:counting:card}{eindige verzameling} \omterm{def:b1:topology:closed}{gesloten} is, eerst via
complementen, dan via de sequentiële karakterisering.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:topology:4}
Bewijs dat voor alle $A, B \subseteq \R$: $\overline{A \cup B} =
\overline A \cup \overline B$. Toon met een voorbeeld dat
$\overline{A \cap B}$ kan verschillen van $\overline A \cap \overline
B$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:topology:5}
Zij $u_n \to \ell$ in $\R$. Bewijs dat de \omterm{def:b1:logic:sets}{verzameling} $\{u_n : n \in
\N\} \cup \{\ell\}$ \omterm{def:b1:topology:closed}{gesloten} is (dus compact als we eraan toevoegen
dat ze begrensd is --- wat ze is).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:topology:6}
Zij $U$ \omterm{def:b1:topology:open}{open} en $A$ willekeurig. Bewijs dat $U + A = \{u + a\}$ \omterm{def:b1:topology:open}{open}
is. Leid af dat de som van een \omterm{def:b1:topology:open}{open verzameling} en een willekeurige
\omterm{def:b1:logic:sets}{verzameling} \omterm{def:b1:topology:open}{open} is, en contrasteer: geef twee \emph{\omterm{def:b1:topology:closed}{gesloten}}
\omterm{def:b1:logic:sets}{verzamelingen} waarvan de som niet \omterm{def:b1:topology:closed}{gesloten} is. \emph{(Probeer $\Z$ en
$\sqrt 2\,\Z$, met \cref{exo:b1:reals:9}.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:topology:7}
Een punt $x \in A$ is \emph{geïsoleerd} in $A$ wanneer een zekere
\omterm{def:b1:topology:open}{omgeving} van $x$ $A$ enkel in $x$ ontmoet. Bewijs dat elk punt van
$\Z$ geïsoleerd is in $\Z$, dat $A = \{\frac1n\}$ al zijn punten
geïsoleerd heeft en toch $\overline A \neq A$, en dat een \omterm{def:b1:logic:sets}{verzameling}
waarvan alle punten geïsoleerd zijn een leeg \omterm{def:b1:topology:closure}{inwendige} heeft.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:topology:8}
Bewijs dat de \omterm{def:b1:topology:closure}{afsluiting} van een begrensde \omterm{def:b1:logic:sets}{verzameling} begrensd is, en
dat $\sup \overline A = \sup A$ voor $A$ niet-leeg en naar boven
begrensd. Leid af dat $\sup A \in \overline A$: het \omterm{def:b1:reals:bounds}{supremum} is altijd
aanklevend.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:topology:9}
Bewijs dat de enige deelverzamelingen van $\R$ die zowel \omterm{def:b1:topology:open}{open} als
\omterm{def:b1:topology:closed}{gesloten} zijn $\emptyset$ en $\R$ zijn. \emph{Hint: veronderstel dat
$A$ \omterm{def:b1:topology:open}{open}, \omterm{def:b1:topology:closed}{gesloten} is, met $A \neq \emptyset$ en $\R \setminus A \neq
\emptyset$; kies $a \in A$, $b \notin A$, zeg $a < b$, en beschouw $s
= \sup\,(A \cap \intcc{a}{b})$; beslis of $s$ tot $A$ of tot zijn
complement kan behoren.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:topology:10}
(Structuur van \omterm{def:b1:topology:open}{open verzamelingen}) Zij $U \subseteq \R$ \omterm{def:b1:topology:open}{open},
niet-leeg. Voor $x \in U$, zij $I_x$ de unie van alle \omterm{def:b1:topology:open}{open} intervallen
die $x$ bevatten en bevat zijn in $U$. Bewijs dat $I_x$ een \omterm{def:b1:topology:open}{open}
\omterm{prop:b1:reals:intervals}{interval} is, dat twee \omterm{def:b1:logic:sets}{verzamelingen} $I_x$, $I_y$ gelijk of disjunct
zijn, en dat $U$ een unie is van \emph{aftelbaar veel} paarsgewijs
disjuncte \omterm{def:b1:topology:open}{open} intervallen \emph{(kies een rationaal getal in elk)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:topology:11}
Een punt $x \in \R$ is een \emph{ophopingspunt} van $A$ wanneer elke
\omterm{def:b1:topology:open}{omgeving} van $x$ $A \setminus \{x\}$ ontmoet; hun \omterm{def:b1:logic:sets}{verzameling} is de
\emph{afgeleide \omterm{def:b1:logic:sets}{verzameling}} $A'$. Bewijs dat $\overline A = A \cup
A'$, en dat $A$ \omterm{def:b1:topology:closed}{gesloten} is als en slechts als $A' \subseteq A$.
Bepaal $A'$ voor $A = \{\frac 1n : n \in \N^*\}$, voor $A = \Z$, en
voor $A = \Q$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:topology:12}
Voor $A \subseteq \R$ niet-leeg, definieer $d_A(x) = \inf\{\abs{x - a}
: a \in A\}$. Bewijs:
\begin{enumerate}
  \item $\abs{d_A(x) - d_A(y)} \leq \abs{x - y}$ voor alle $x, y$
        ($d_A$ is $1$-Lipschitz);
  \item $d_A(x) = 0$ als en slechts als $x \in \overline A$; in het
        bijzonder, als $F$ \omterm{def:b1:topology:closed}{gesloten} is en $x \notin F$, dan $d_F(x) >
        0$;
  \item voor elke $\varepsilon > 0$ is de \omterm{def:b1:logic:sets}{verzameling} $V_\varepsilon =
        \{x : d_F(x) < \varepsilon\}$ \omterm{def:b1:topology:open}{open}, bevat ze $F$, en
        $\bigcap_{\varepsilon > 0} V_\varepsilon = F$ voor $F$
        \omterm{def:b1:topology:closed}{gesloten}: elke \omterm{def:b1:topology:closed}{gesloten verzameling} is een aftelbare doorsnede
        van \omterm{def:b1:topology:open}{open verzamelingen}.
\end{enumerate}
\end{exercise}

\section{Vraagstuk: De Cantorverzameling, klein en enorm tegelijk}

\begin{problem}[{Weekendvraagstuk --- de Cantor-middelste-derden-verzameling:
lengte nul, overaftelbaar, perfect, en $C + C = \intcc{0}{2}$}]
\label{pb:b1:topology:1}
Verwijder uit $\intcc{0}{1}$ zijn \omterm{def:b1:topology:open}{open} middelste derde, dan de
middelste derde van elk overblijvend segment, en herhaal eeuwig: wat
overleeft is de \emph{\omterm{pb:b1:topology:1}{Cantorverzameling}} $C$, de fundamentele
tegenvoorbeeldenfabriek van de analyse. Dit vraagstuk construeert
haar, leest haar via de grondtal-$3$-machinerie van
\cref{pb:b1:reals:1}, en vestigt haar paradoxale portret: totale
lengte nul, en toch overaftelbaar; leeg \omterm{def:b1:topology:closure}{inwendige}, en toch geen enkel
geïsoleerd punt; volledig onsamenhangend, en toch vult $C + C$ het
hele segment $\intcc{0}{2}$.\index{Cantorverzameling} Formeel: $C_0 =
\intcc{0}{1}$, en $C_{n+1}$ wordt uit $C_n$ verkregen door de \omterm{def:b1:topology:open}{open}
middelste derde van elk segment van $C_n$ te verwijderen; ten slotte
$C = \bigcap_{n \geq 0} C_n$. Overal is een \emph{ternaire code} van
$x \in \intcc{0}{1}$ elke cijferrij $(d_k)_{k\geq1}$ met $d_k \in \{0,
1, 2\}$ waarvan de waarde $\sup_n \sum_{k=1}^n d_k 3^{-k}$ gelijk is
aan $x$ --- oneigenlijke codes (uiteindelijk $2$) zijn toegestaan;
volgens \cref{pb:b1:reals:1} (vragen 9--11) heeft elke $x \in
\intcc{0}{1}$ één of twee codes, twee precies wanneer $x = m/3^N \in
\intoo{0}{1}$.

\medskip
\noindent\textbf{Deel I --- De constructie.}
\begin{enumerate}
  \item Beschrijf $C_1$ en $C_2$ expliciet als unies van segmenten,
        en bewijs door inductie: $C_n$ is een disjuncte unie van $2^n$
        \omterm{def:b1:topology:closed}{gesloten} segmenten, elk van lengte $3^{-n}$.
  \item Toon dat $C$ \omterm{def:b1:topology:closed}{gesloten}, begrensd --- dus compact
        (\cref{thm:b1:topology:compact}) --- niet-leeg is, en dat elk
        eindpunt van elk segment van elke $C_n$ tot $C$ behoort.
  \item De totale lengte van $C_n$ is $\bigl(\frac23\bigr)^n$. Leid af
        dat voor elke $\varepsilon > 0$ de \omterm{def:b1:logic:sets}{verzameling} $C$ kan overdekt
        worden door eindig veel segmenten van totale lengte $\leq
        \varepsilon$: de \omterm{pb:b1:topology:1}{Cantorverzameling} heeft \emph{lengte nul}.
  \item Toon dat een \omterm{prop:b1:reals:intervals}{interval} bevat in $C$ lengte $\leq 3^{-n}$ heeft
        voor elke $n$, dus een singleton of leeg is: $\mathring C =
        \emptyset$. \omterm{def:b1:topology:closed}{Gesloten} zijnd met leeg \omterm{def:b1:topology:closure}{inwendige}, is $C$
        \emph{nergens \omterm{def:b1:topology:dense}{dicht}}.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De ternaire code.}
\begin{enumerate}[resume]
  \item Bewijs de zelfgelijkvormigheidsrecursie
        \[
        C_{n+1} = \tfrac13 C_n \,\cup\,
        \bigl(\tfrac23 + \tfrac13 C_n\bigr),
        \qquad\text{dus}\qquad
        C = \tfrac13 C \,\cup\, \bigl(\tfrac23 + \tfrac13
        C\bigr),
        \]
        waarbij de twee stukken disjunct zijn: $C$ is twee kopieën van
        zichzelf op schaal $\frac13$.
  \item Bewijs door inductie op $n$: $x \in C_n$ als en slechts als
        $x$ een ternaire code heeft waarvan de eerste $n$ cijfers in
        $\{0, 2\}$ liggen. Leid af, gebruikmakend van het feit dat $x$
        ten hoogste twee codes heeft: $x \in C$ als en slechts als $x$
        een code heeft met \emph{geen enkel cijfer gelijk aan $1$}
        (een \emph{$1$-vrije} code).
  \item Codes in actie: geef $1$-vrije codes voor $0$, $1$, $\frac13$,
        $\frac23$; toon $\frac14 = (0.\overline{02})_3$ en $\frac34 =
        (0.\overline{20})_3$, zodat beide tot $C$ behoren; en
        controleer dat $\frac14$ \emph{geen} eindpunt is van enige
        $C_n$ (eindpunten hebben de vorm $m/3^n$).
  \item Toon dat elke $x \in C$ \emph{precies één} $1$-vrije code
        heeft \emph{(wanneer $x$ twee codes heeft, bewijs dat precies
        één van het paar het cijfer $1$ bevat)}. Besluit: de
        waardeafbeelding is een bijectie van $\{0,2\}$-rijen op $C$.
  \item (Diagonaal) Zij $k \mapsto x_k$ een willekeurige \omterm{def:b1:logic:map}{afbeelding}
        $\N^* \to C$. Bouw een $\{0, 2\}$-rij die op index $k$
        verschilt van de code van $x_k$, en besluit dat $C$
        overaftelbaar is --- terwijl, in tegenstelling, vraag 3 zegt
        dat ze metrisch verwaarloosbaar is.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Topologisch portret.}
\begin{enumerate}[resume]
  \item Verzamel het dossier tot nu toe: $C$ is compact,
        overaftelbaar, van lengte nul, nergens \omterm{def:b1:topology:dense}{dicht}. Welke enkele
        inclusie $C \subseteq C_n$ draagt elke eigenschap?
  \item ($C$ is perfect) Zij $x \in C$ met $1$-vrije code $(d_k)$. Het
        omkeren van het cijfer $d_n$ ($0 \leftrightarrow 2$) levert
        $x_n \in C$ met $\abs{x_n - x} = 2 \cdot 3^{-n}$. Besluit dat
        $C$ geen enkel geïsoleerd punt heeft: elk punt van $C$ is een
        limiet van \emph{andere} punten van $C$.
  \item Toon dat de eindpunten van vraag 2 een aftelbare dichte
        deelverzameling van $C$ vormen \emph{(kap de code af na $n$
        cijfers en ga verder met $0$'s; aftelbaarheid zoals in
        \cref{exo:b1:topology:10})}. Besluit: het typische punt van
        $C$ --- zoals $\frac14$ --- is \emph{geen} eindpunt:
        eindpunten zijn een aftelbaar skelet binnen een overaftelbaar
        \omterm{def:b1:structures:field}{lichaam}.
  \item (Volledig onsamenhangend) Zij $x < y$ in $C$. Kies $n$ met
        $3^{-n} < y - x$ en produceer een punt $z \in \intoo{x}{y}$
        met $z \notin C$. Besluit dat de enige niet-lege intervallen
        bevat in $C$ singletons zijn.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Rekenkunde van $C$.}
\begin{enumerate}[resume]
  \item Toon $1 - C = C$ \emph{(wat doet $x \mapsto 1 - x$ met een
        $1$-vrije code? denk aan $1 = (0.\overline{2})_3$)}.
  \item (Optelling van codes) Toon dat als $x$, $x'$ codes $(a_k)$,
        $(b_k)$ hebben, dan $x + x' = \lim_n\,(t_n + t'_n)$ waarbij
        $t_n, t'_n$ de partiële sommen zijn. Leid af: elke $y \in
        \intcc{0}{1}$ is het \emph{middelpunt} van twee punten van $C$
        --- gegeven een code $(e_k)$ van $y$, kies cijfers $a_k, b_k
        \in \{0, 2\}$ met $\frac{a_k + b_k}{2} = e_k$.
  \item Besluit $C + C = \intcc{0}{2}$ en, met vraag 14, $C - C =
        \intcc{-1}{1}$. Concreet geval: schrijf $1$ als een som van de
        twee niet-eindpunten gevonden in vraag 7.
  \item Reflecteer: een \omterm{def:b1:logic:sets}{verzameling} van lengte nul waarvan de
        verschilverzameling $\intcc{-1}{1}$ vult. Waarom is er geen
        tegenspraak tussen ``$C$ is metrisch verwaarloosbaar'' en ``$C
        + C$ heeft volle lengte''? (Eén zin; denk na over wat lengte
        controleert en wat niet.)
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Leden, rationaal en irrationaal.}
\begin{enumerate}[resume]
  \item Combineer vraag 8 met het periodiciteitscriterium van
        \cref{pb:b1:reals:1}: een punt van $C$ is rationaal als en
        slechts als zijn $1$-vrije code uiteindelijk periodiek is. Voer
        de staartdeling in grondtal $3$ uit om te controleren dat
        $\frac1{13} = (0.\overline{002})_3 \in C$.
  \item Produceer een expliciet \emph{irrationaal} lid van $C$: de
        waarde van de code met $d_k = 2$ op de driehoekige posities $k
        = \frac{j(j+1)}{2}$ en $d_k = 0$ elders. Verantwoord
        irrationaliteit via het groeiende-gaten-argument van
        \cref{pb:b1:reals:1} (vraag 20).
  \item (Op een vol segment) Beschouw $h$ dat het punt van $C$ met
        $1$-vrije code $(d_k)$ afbeeldt op de waarde van de
        \emph{binaire} rij $\bigl(\frac{d_k}2\bigr)$, d.w.z.\ $h(x) =
        \sup_n \sum_{k=1}^n \frac{d_k}{2}\,2^{-k}$. Toon dat $h$ $C$
        \emph{op} $\intcc{0}{1}$ afbeeldt. Dus de verwaarloosbare $C$
        surjecteert op een segment van volle lengte --- een tweede
        bewijs dat $C$ overaftelbaar is.
  \item (Zelfgelijkvormige lengte) Veronderstel dat een zeker
        lengtebegrip $L$ gedefinieerd was voor $C$ en zijn gekrompen
        kopieën, dat schaling ($L(\lambda A) = \lambda L(A)$),
        translatie-invariantie en additiviteit over de disjuncte
        ontbinding van vraag 5 respecteert. Toon dat dan $L(C) =
        \frac23\,L(C)$, wat $L(C) = 0$ afdwingt:
        zelfgelijkvormigheid alleen veroordeelt $C$ reeds tot lengte
        nul.
\end{enumerate}

\medskip
\noindent\textbf{Deel VI --- Een dikke neef, en de moraal.}
\begin{enumerate}[resume]
  \item (Dikke \omterm{pb:b1:topology:1}{Cantorverzameling}) Herhaal de constructie, maar
        verwijder in stadium $n$ ($n = 0, 1, 2, \dots$) uit elk van de
        $2^n$ huidige segmenten enkel een centraal \omterm{def:b1:topology:open}{open} \omterm{prop:b1:reals:intervals}{interval} van
        lengte $4^{-(n+1)}$. Toon dat de segmentlengten $l_n$ voldoen
        aan $l_{n+1} = \frac{l_n - 4^{-(n+1)}}{2}$, $l_n =
        \frac{2^n + 1}{2\cdot 4^n} > 0$, dat de resulterende $K =
        \bigcap K_n$ compact is met leeg \omterm{def:b1:topology:closure}{inwendige}, en dat de totale
        verwijderde lengte $\sum_{n\geq0} 2^n 4^{-(n+1)} = \frac12$
        is. Toegevend de (intuïtieve, jaar 3) additiviteit van lengte
        voor eindige unies van intervallen, en gebruikmakend van beide
        uitspraken van \cref{ex:b1:topology:nested}, toon dat elke
        eindige familie van \omterm{def:b1:topology:open}{open} intervallen die $K$ overdekt totale
        lengte $\geq \frac12$ heeft: $K$ is nergens \omterm{def:b1:topology:dense}{dicht} maar
        \emph{niet} verwaarloosbaar. Kleinheid heeft verschillende
        niet-equivalente betekenissen.
  \item (Afstanden) Toon dat voor een niet-lege \omterm{def:b1:topology:closed}{gesloten} $F \subseteq
        \R$ en $x \in \R$ het \omterm{def:b1:reals:bounds}{infimum} $d(x, F)$ \emph{bereikt} wordt
        \emph{(minimaliserende rij plus Bolzano--Weierstrass)}. Bereken
        dan
        \[
        \max_{y \in \intcc{0}{1}} d(y, C) = \frac16 ,
        \]
        bereikt precies in het centrum $y = \frac12$ \emph{(een punt
        van een gat gecreëerd in stadium $n$ ligt binnen
        $\frac{3^{-n}}{2}$ van de eindpunten van het gat, die in $C$
        liggen)}.
  \item (Elk punt een deelrijlimiet) Gebruikmakend van vragen 12 en 2,
        produceer een enkele rij in $C$ waarvan de \omterm{def:b1:logic:sets}{verzameling} van
        deelrijlimieten \emph{heel} $C$ is. (Vergelijk: voor een
        convergente rij is die \omterm{def:b1:logic:sets}{verzameling} één punt --- $C$
        verwezenlijkt het tegengestelde uiterste onder compacte
        \omterm{def:b1:logic:sets}{verzamelingen}.)
  \item Synthese, telkens één zin: (i) welke stellingen van dit
        hoofdstuk heeft de constructie werkelijk verbruikt (stabiliteit
        van \omterm{def:b1:topology:closed}{gesloten verzamelingen}, compactheid, sequentiële
        karakteriseringen)? (ii) noem de vier paradoxale koppelingen
        van het portret (lengte nul/overaftelbaar, \omterm{def:b1:topology:closed}{gesloten}/leeg
        \omterm{def:b1:topology:closure}{inwendige}, perfect/volledig onsamenhangend, verwaarloosbaar/$C+C$
        vol); (iii) waar duikt $C$ later weer op (de duiveltrap
        gebouwd op $h$ in de theorie van de continuïteit, en de
        maattheorie van het boekdeel van jaar 3, waar $C$ ``aftelbaar''
        van ``verwaarloosbaar'' scheidt)?
\end{enumerate}
\end{problem}
