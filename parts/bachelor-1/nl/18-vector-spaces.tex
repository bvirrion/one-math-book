\chapter{Vectorruimten}\label{ch:b1:vspaces}

Hier begint de lineaire algebra: de axioma's van \omterm{def:b1:vspaces:def}{vectorruimten}
isoleren wat $\R^2$, $\R^3$, veeltermruimten en functieruimten
gemeen hebben --- men kan optellen en schalen. Twee hoofdstukken
bouwen de theorie op (\cref{ch:b1:findim} voegt de dimensie toe);
de taal die ze vastleggen --- \omterm{def:b1:vspaces:span}{opspansel}, \omterm{def:b1:vspaces:free}{vrije familie}, \omterm{def:b1:vspaces:free}{basis},
\omterm{def:b1:vspaces:sum}{directe som} --- is het dagelijks brood van elk hoofdstuk daarna.
Overal duidt $K$ op $\R$ of $\C$ (\emph{scalairen}).

\section{Definitie en voorbeelden}

\begin{definition}[Vectorruimte]\label{def:b1:vspaces:def}
Een \emph{$K$-vectorruimte}\index{vectorruimte} is een \omterm{def:b1:logic:sets}{verzameling}
$E$ met een optelling die van $(E, +)$ een \omterm{def:b1:structures:group}{abelse groep} maakt (nul
geschreven $0_E$ of $0$), en een scalaire vermenigvuldiging $K
\times E \to E$ zodat, voor alle $\lambda, \mu \in K$ en $x, y \in
E$:
\[
\lambda(x + y) = \lambda x + \lambda y,\quad
(\lambda + \mu) x = \lambda x + \mu x,\quad
\lambda(\mu x) = (\lambda\mu) x,\quad
1\,x = x .
\]
Gevolgen: $0\,x = 0_E$, $\lambda\,0_E = 0_E$, $(-1)x = -x$, en
$\lambda x = 0_E \implies \lambda = 0$ of $x = 0_E$
(vermenigvuldig met $\lambda^{-1}$).
\end{definition}

\begin{proof}[Bewijs van de gevolgen]
Voor $0\,x = 0_E$: uit $(0 + 0)x = 0x + 0x$ en $(0+0)x = 0x$,
schrap $0x$ in de \omterm{def:b1:structures:group}{groep} $(E, +)$. Voor $\lambda\,0_E$: dezelfde
truc op $\lambda(0_E + 0_E)$. Voor $(-1)x$: tel $x$ op,
\[
x + (-1)x = 1\,x + (-1)x = \bigl(1 + (-1)\bigr)x = 0\,x = 0_E ,
\]
dus $(-1)x$ is het additieve invers van $x$. Ten slotte, als
$\lambda x = 0_E$ met $\lambda \neq 0$: vermenigvuldig met
$\lambda^{-1}$ (de scalairen vormen een \omterm{def:b1:structures:field}{lichaam}) en gebruik de
twee axioma's $\lambda^{-1}(\lambda x) = (\lambda^{-1}\lambda)x =
1x = x$ samen met $\lambda^{-1}0_E = 0_E$: $x = 0_E$. Hoe klein ze
ook zijn, deze vier regels worden stilzwijgend gebruikt op elke
bladzijde die volgt --- en de laatste is precies waar lichamen
nodig zijn: over de scalairen $\Z$ zou de ``ruimte'' $\Z/2\Z$ hem
schenden met $2\,x = 0$.
\end{proof}

\begin{example}\label{ex:b1:vspaces:examples}
$K^n$ (coördinaatsgewijze bewerkingen); de \omterm{def:b1:poly:def}{veeltermen} $K[X]$; de
functies $\mathcal{F}(A, K)$ van een willekeurige \omterm{def:b1:logic:sets}{verzameling} $A$
naar $K$ (puntsgewijze bewerkingen) --- die continue functies,
rijen $\mathcal{F}(\N, \R)$, enz. bevatten; $\C$ als
$\R$-vectorruimte. In elk geval worden de axioma's overgeërfd van
die van $K$.
\end{example}

\begin{definition}[Deelruimte]\label{def:b1:vspaces:subspace}
$F \subseteq E$ is een \emph{deelruimte}\index{deelruimte} wanneer
$0_E \in F$ en $F$ stabiel is onder optelling en scalaire
vermenigvuldiging --- equivalent:
\[
F \neq \emptyset
\qquad\text{en}\qquad
\forall x, y \in F,\ \forall \lambda \in K,\quad
x + \lambda y \in F .
\]
Een deelruimte is zelf een \omterm{def:b1:vspaces:def}{vectorruimte}. Elke doorsnede van
deelruimten is een deelruimte; een vereniging bijna nooit
(zelfde bewijs als \cref{exo:b1:structures:6}).
\end{definition}

\begin{example}\label{ex:b1:vspaces:subspaces}
In $\mathcal{F}(\R, \R)$: de continue functies, de afleidbare, de
\omterm{def:b1:poly:def}{veeltermen} van graad $\leq n$ (geschreven $K_n[X]$ binnen
$K[X]$), de oplossingen van een homogene lineaire
differentiaalvergelijking (\cref{thm:b1:diffeq:homogeneous2} zei
precies dat). Niet-voorbeelden: $\{f : f(0) = 1\}$ (geen nul),
graad precies $n$ (niet stabiel onder optelling).
\end{example}

\begin{example}[Deelruimte of niet: vier uitspraken, beargumenteerd]
\label{ex:b1:vspaces:subspaceornot}
In de ruimte van reële rijen:
\begin{itemize}
  \item $\{u : u \text{ begrensd}\}$ \emph{is} een \omterm{def:b1:vspaces:subspace}{deelruimte}: $0$
        is begrensd, en als $\abs{u_n} \leq M$, $\abs{v_n} \leq
        M'$, dan $\abs{u_n + \lambda v_n} \leq M + \abs\lambda M'$.
  \item $\{u : u_n \to 1\}$ is het \emph{niet}: de nulrij
        ontbreekt (en de som van twee leden gaat naar $2$).
  \item $\{u : u \text{ monotoon}\}$ is het \emph{niet}: $u_n = n$
        en $v_n = -n + (-1)^n$ zijn monotoon, hun som $(-1)^n$ is
        het niet; stabiliteit onder optelling is het axioma dat
        faalt, ook al bevat de \omterm{def:b1:logic:sets}{verzameling} $0$ en alle scalaire
        veelvouden van haar leden.
  \item $\{u : u_{n+1} = u_n^2\}$ is het \emph{niet}: ze bevat $0$
        maar $2u$ ontsnapt zodra $u$ een niet-nul lid is
        ($2u_{n+1} \neq (2u_n)^2$ in het algemeen) --- kwadrateren
        is de niet-lineariteit.
\end{itemize}
De werkvolgorde is altijd dezelfde: test eerst $0$ (goedkoopst),
dan stabiliteit --- en om te weerleggen verslaat één expliciet
tegenvoorbeeldpaar elke hoeveelheid twijfel.
\end{example}

\section{Opspansel, sommen, directe sommen}

\begin{definition}[Lineaire combinaties, opspansel]\label{def:b1:vspaces:span}
Een \emph{lineaire combinatie} van de familie $(x_1, \dots, x_p)$
van vectoren van $E$ is een willekeurige $\lambda_1 x_1 + \dots +
\lambda_p x_p$ ($\lambda_i \in K$). De \omterm{def:b1:logic:sets}{verzameling} van al deze is
het \emph{opspansel}\index{opspansel} $\operatorname{Vect}(x_1,
\dots, x_p)$: het is een \omterm{def:b1:vspaces:subspace}{deelruimte}, de kleinste die de familie
bevat.
\end{definition}

\begin{proof}[Bewijs van de twee beweringen]
Stabiliteit: een som van twee lineaire combinaties
$\sum\lambda_i x_i + \sum\mu_i x_i = \sum(\lambda_i + \mu_i)x_i$
is er weer een, en zo ook een scalair veelvoud
$\mu\sum\lambda_i x_i = \sum(\mu\lambda_i) x_i$; de nulcombinatie
toont dat $0$ erbij hoort: het \omterm{def:b1:vspaces:span}{opspansel} is een \omterm{def:b1:vspaces:subspace}{deelruimte}.
Minimaliteit: zij $H$ een willekeurige \omterm{def:b1:vspaces:subspace}{deelruimte} die $x_1,
\dots, x_p$ bevat. Door stabiliteit onder scalaire
vermenigvuldiging ligt elke $\lambda_i x_i \in H$, en door
stabiliteit onder optelling ligt hun som in $H$: elke lineaire
combinatie hoort bij $H$, d.w.z.\ $\operatorname{Vect}(x_1,
\dots, x_p) \subseteq H$. Dus het \omterm{def:b1:vspaces:span}{opspansel} is bevat in elke
\omterm{def:b1:vspaces:subspace}{deelruimte} die de familie bevat: het is de kleinste.
\end{proof}

\begin{definition}[Som, directe som]\label{def:b1:vspaces:sum}
Voor \omterm{def:b1:vspaces:subspace}{deelruimten} $F, G$ van $E$:
\[
F + G = \{\,u + v : u \in F,\ v \in G\,\}
\]
is een \omterm{def:b1:vspaces:subspace}{deelruimte} (de kleinste die $F \cup G$ bevat). De som is
\emph{direct}\index{directe som}, geschreven $F \oplus G$,
wanneer elk element van $F + G$ zich \emph{uniek} ontbindt als $u
+ v$; equivalent (zie hieronder) wanneer $F \cap G = \{0\}$.
Wanneer $E = F \oplus G$, zijn de \omterm{def:b1:vspaces:subspace}{deelruimten}
\emph{complementair}\index{complementaire deelruimten} in $E$.
\end{definition}

\begin{example}[Een som van twee rechten]
\label{ex:b1:vspaces:sumoflines}
In $\R^3$, zij $F = \operatorname{Vect}\bigl((1,0,1)\bigr)$ en $G
= \operatorname{Vect}\bigl((0,1,1)\bigr)$. Hun som is
\[
F + G = \{\,a(1,0,1) + b(0,1,1)\,\}
= \{(a,\ b,\ a + b)\} = \{(x, y, z) : z = x + y\},
\]
het vlak door de oorsprong dat beide rechten bevat. Het is strikt
groter dan de vereniging $F \cup G$ (het loutere kruis van de twee
rechten): de vector $(1, 1, 2) = (1,0,1) + (0,1,1)$ ligt in de som
maar op geen van beide rechten. En $F \cap G = \{0\}$ (een
gemeenschappelijke vector vereist $a(1,0,1) = b(0,1,1)$, waarvan de
eerste twee \omterm{prop:b1:vspaces:coordinates}{coördinaten} $a = b = 0$ afdwingen): de som is direct,
en $F \oplus G$ is precies dat vlak.
\end{example}

\begin{proposition}\label{prop:b1:vspaces:directsum}
$F + G$ is direct als en slechts als $F \cap G = \{0\}$.
\end{proposition}

\begin{proof}
Als een $w \neq 0$ in $F \cap G$ ligt: $w = w + 0 = 0 + w$ zijn twee
ontbindingen van $w$. Omgekeerd, als $u + v = u' + v'$ met $u, u'
\in F$, $v, v' \in G$, dan hoort $u - u' = v' - v$ bij $F \cap G =
\{0\}$: ontbindingen zijn uniek.
\end{proof}

\begin{method}[Bewijzen dat $E = F \oplus G$]
\label{met:b1:vspaces:directsum}
Twee dingen om na te gaan, elk met zijn standaard openingszet.
\begin{enumerate}
  \item \emph{Triviale doorsnede.} Neem $x \in F \cap G$, schrijf
        beide lidmaatschapsvoorwaarden uit, en pers $x = 0$
        eruit. (Redeneer nooit via een tekening: vgl.\ de
        valkuilen hieronder.)
  \item \emph{Som is alles.} Neem een willekeurige $x \in E$ en
        \emph{produceer} de ontbinding $x = f + g$ --- ofwel door
        $f$ te raden uit het doel ($f$ moet voldoen aan de
        definiërende eigenschap van $F$, die meestal zijn formule
        dicteert) ofwel door het lineair stelsel op te lossen dat
        $x$ uitdrukt tegenover voortbrengers van $F$ en $G$.
\end{enumerate}
Wanneer de ontbindingsformule geraden is, is uniciteit automatisch
uit stap 1; wanneer alleen het bestaan onduidelijk is, is stap 2
waar het werk zit. De twee voorbeelden hieronder laten de methode
lopen: voor even/oneven functies wordt de formule voor $f$
afgedwongen door de gewenste identiteit te evalueren in $x$ en
$-x$; voor \omterm{def:b1:poly:def}{veeltermen} die in een punt verdwijnen, door te
evalueren in $a$.
\end{method}

\begin{example}\label{ex:b1:vspaces:evenodd}
In $\mathcal{F}(\R, \R)$ zijn de even functies $\mathcal{P}$ en de
oneven functies $\mathcal{I}$ \omterm{def:b1:vspaces:sum}{complementair}: elke $f$ schrijft
zich als
\[
f(x) = \underbrace{\frac{f(x) + f(-x)}{2}}_{\text{even}}
+ \underbrace{\frac{f(x) - f(-x)}{2}}_{\text{oneven}},
\]
en een functie die zowel even als oneven is, is nul. (Toegepast
op $\exp$ is dit het paar $(\cosh, \sinh)$ van
\cref{ch:b1:functions}.)
\end{example}

\begin{example}[{Een complementair paar in $K_n[X]$}]
\label{ex:b1:vspaces:hyperplane}
Kies $a \in K$ en stel $F = \{P \in K_n[X] : P(a) = 0\}$, $G =
\operatorname{Vect}(1)$ (de constanten). Dan $K_n[X] = F \oplus G$.
Inderdaad bestaat $F \cap G$ uit de constanten die in $a$
verdwijnen, d.w.z.\ $\{0\}$; en elke $P$ ontbindt zich als
\[
P = \underbrace{\bigl(P - P(a)\bigr)}_{\in F}
+ \underbrace{P(a)}_{\in G} .
\]
De ontbinding is het onthouden waard: de waarde in een punt
aftrekken is de standaardmanier om te projecteren op ``functies
die in $a$ verdwijnen''. Merk op dat $F$ een grote \omterm{def:b1:vspaces:subspace}{deelruimte} is en
$G$ een kleine; een \omterm{def:b1:vspaces:sum}{complementair} paar hoeft in geen enkele zin
gebalanceerd te zijn.
\end{example}

\begin{example}[Een complementaire deelruimte is nooit uniek]
\label{ex:b1:vspaces:manysupplements}
In $\R^2$, zij $F = \operatorname{Vect}\bigl((1,0)\bigr)$ (de
$x$-as). Zowel $G = \operatorname{Vect}\bigl((0,1)\bigr)$ als $G' =
\operatorname{Vect}\bigl((1,1)\bigr)$ zijn \omterm{def:b1:vspaces:sum}{complementair} aan $F$:
elk snijdt $F$ enkel in $0$, en elk paar telt op tot $\R^2$. De
ontbindingen van eenzelfde vector verschillen:
\[
(2,\ 1.5) = \underbrace{(2, 0)}_{\in F} +
\underbrace{(0, 1.5)}_{\in G}
= \underbrace{(0.5,\ 0)}_{\in F} +
\underbrace{(1.5,\ 1.5)}_{\in G'} .
\]
In feite is \emph{elke} rechte behalve $F$ zelf een complementaire
van $F$ in $\R^2$: \omterm{def:b1:vspaces:sum}{complementairen} zijn er in overvloed, en
spreken van ``de'' complementaire is zinloos totdat een extra
structuur (een inwendig product, \cref{ch:b1:euclid}) er één
uitkiest.
\end{example}

\section{Vrije families, voortbrengende families, bases}

\begin{definition}\label{def:b1:vspaces:free}
Een familie $(x_1, \dots, x_p)$ van vectoren van $E$ is:
\begin{itemize}
  \item \emph{voortbrengend} (van $E$) wanneer
        $\operatorname{Vect}(x_1,\dots,x_p) = E$;
  \item \emph{vrij}\index{vrije familie} (haar vectoren
        \emph{lineair onafhankelijk}) wanneer
        \[
        \lambda_1 x_1 + \dots + \lambda_p x_p = 0
        \implies \lambda_1 = \dots = \lambda_p = 0 ;
        \]
        anders \emph{gebonden};
  \item een \emph{basis}\index{basis} wanneer ze vrij en
        voortbrengend is.
\end{itemize}
\end{definition}

\begin{proposition}[Coördinaten]\label{prop:b1:vspaces:coordinates}
$(e_1, \dots, e_n)$ is een \omterm{def:b1:vspaces:free}{basis} van $E$ als en slechts als elke $x
\in E$ \emph{uniek} een combinatie $x = \lambda_1 e_1 + \dots +
\lambda_n e_n$ is; de scalairen $\lambda_i$ zijn de
\emph{coördinaten}\index{coördinaten} van $x$ in de \omterm{def:b1:vspaces:free}{basis}.
\end{proposition}

\begin{proof}
Voortbrengend $=$ bestaan van de ontbinding. Uniciteit $=$
vrijheid: twee ontbindingen van dezelfde $x$ verschillen door een
combinatie gelijk aan $0$; vrijheid dwingt al haar coëfficiënten
--- de verschillen van \omterm{prop:b1:vspaces:coordinates}{coördinaten} --- af tot nul. Omgekeerd geeft
een niet-triviale nulcombinatie de twee ontbindingen $0 = \sum
\lambda_i e_i = \sum 0\,e_i$.
\end{proof}

\begin{example}\label{ex:b1:vspaces:bases}
De \emph{canonieke \omterm{def:b1:vspaces:free}{basis}} van $K^n$: $e_i = (0, \dots, 1, \dots,
0)$ ($1$ op plaats $i$). De monomen $(1, X, X^2, \dots, X^n)$: een
\omterm{def:b1:vspaces:free}{basis} van $K_n[X]$ (vrijheid: een nulcombinatie is de nulveelterm,
dus alle coëfficiënten verdwijnen, \cref{def:b1:poly:def}). In $\C$
over $\R$: de \omterm{def:b1:vspaces:free}{basis} $(1, \iu)$.
\end{example}

\begin{remark}[Coördinaten zijn teamwerk]
\label{rem:b1:vspaces:teameffort}
De eerste coördinaat van $x$ in een \omterm{def:b1:vspaces:free}{basis} $(e_1, \dots, e_n)$
hangt af van \emph{alle} basisvectoren, niet alleen $e_1$. In
$\R^2$: de vector $(3, 1)$ heeft eerste coördinaat $3$ in de
canonieke \omterm{def:b1:vspaces:free}{basis}, maar eerste coördinaat $2$ in de \omterm{def:b1:vspaces:free}{basis}
$\bigl((1,0), (1,1)\bigr)$ --- los $(3,1) = a(1,0) + b(1,1)$ op:
$b = 1$, $a = 2$. Het veranderen van één basisvector herschikt
\emph{elke} coördinaat; \cref{ch:b1:matrices} zal deze
herschikking verpakken in de basisveranderingsmatrix.
\end{remark}

\begin{example}[Een kandidaat-basis testen, van begin tot eind]
\label{ex:b1:vspaces:testbasis}
Is $\mathcal{F} = (1 + X,\ 1 + X^2,\ X + X^2)$ een \omterm{def:b1:vspaces:free}{basis} van
$\R_2[X]$? Schrijf $u_1, u_2, u_3$ voor de drie \omterm{def:b1:poly:def}{veeltermen}.
\emph{Vrijheid}: een nulcombinatie $a\,u_1 + b\,u_2 + c\,u_3 = 0$
geeft, coëfficiënt per coëfficiënt,
\[
a + b = 0, \qquad a + c = 0, \qquad b + c = 0 ;
\]
door de eerste twee af te trekken, $b = c$, dan geeft de derde
$2b = 0$: $a = b = c = 0$, \omterm{def:b1:vspaces:free}{vrij}. \emph{Voortbrengend}: in plaats
van drie stelsels op te lossen, merk de symmetrische combinatie op
\[
u_1 + u_2 - u_3 = (1 + X) + (1 + X^2) - (X + X^2) = 2 ,
\]
dus $1 = \frac12(u_1 + u_2 - u_3)$; dan
\[
X = u_1 - 1 = \tfrac12\bigl(u_1 - u_2 + u_3\bigr),
\qquad
X^2 = u_2 - 1 = \tfrac12\bigl(-u_1 + u_2 + u_3\bigr).
\]
De monomen liggen in het \omterm{def:b1:vspaces:span}{opspansel}, dus alles ligt erin:
$\mathcal{F}$ is een \omterm{def:b1:vspaces:free}{basis}. Als bonus geeft het samenvoegen van de
drie formules de \omterm{prop:b1:vspaces:coordinates}{coördinaten} van elke $P = \alpha + \beta X +
\gamma X^2$:
\[
P = \frac{\alpha + \beta - \gamma}{2}\,u_1
+ \frac{\alpha - \beta + \gamma}{2}\,u_2
+ \frac{-\alpha + \beta + \gamma}{2}\,u_3 .
\]
(Controle met $P = X$: \omterm{prop:b1:vspaces:coordinates}{coördinaten} $\bigl(\frac12, -\frac12,
\frac12\bigr)$, zoals hierboven gevonden.) Twee lessen: symmetrie
in de familie verbergt meestal een verkortende combinatie; en
zodra de dimensie beschikbaar is (\cref{ch:b1:findim}) komt de hele
voortbrengende helft van dit werk gratis --- drie vrije vectoren
van een $3$-dimensionale ruimte vormen altijd een \omterm{def:b1:vspaces:free}{basis}.
\end{example}

\begin{proposition}[Nuttige vrijheidscriteria]\label{prop:b1:vspaces:freecriteria}
\begin{enumerate}
  \item Een familie van \emph{niet-nul \omterm{def:b1:poly:def}{veeltermen} met paarsgewijs
        verschillende graden} is \omterm{def:b1:vspaces:free}{vrij}.
  \item Een vector toevoegen aan een \omterm{def:b1:vspaces:free}{vrije familie} houdt haar \omterm{def:b1:vspaces:free}{vrij}
        als en slechts als de vector buiten het \omterm{def:b1:vspaces:span}{opspansel} van de
        familie ligt.
  \item Elke deelfamilie van een \omterm{def:b1:vspaces:free}{vrije familie} is \omterm{def:b1:vspaces:free}{vrij}; elke
        familie die een voortbrengende familie bevat, is
        voortbrengend.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) Bekijk in een nulcombinatie de hoogste aanwezige graad: zijn
coëfficiënt moet verdwijnen (niets heft die graad op), en cascadeer
naar beneden.

(2) Als $x \in \operatorname{Vect}(x_1, \dots, x_p)$, dan is de
relatie $x - \sum\lambda_i x_i = 0$ niet-triviaal. Omgekeerd moet
een niet-triviale nulcombinatie van $(x_1, \dots, x_p, x)$ $x$
bevatten met een niet-nul coëfficiënt (anders spreekt ze de
vrijheid van de kleine familie tegen), en $x$ oplossen legt het in
het \omterm{def:b1:vspaces:span}{opspansel}.

(3) Deelfamilie: een nulcombinatie van de deelfamilie is er een van
de hele familie met de ontbrekende coëfficiënten op $0$ gezet;
vrijheid van de grote familie doodt ze allemaal. Superfamilie: elke
vector van $E$ is al een combinatie van het voortbrengende deel;
geef de extra vectoren de coëfficiënt $0$.
\end{proof}

\begin{example}[Het trapprincipe]
\label{ex:b1:vspaces:staircase}
Zij $P_0, P_1, \dots, P_n \in K_n[X]$ met $\deg P_k = k$ voor elke
$k$ (een ``trap'' van graden). Dan is $(P_0, \dots, P_n)$ een \omterm{def:b1:vspaces:free}{basis}
van $K_n[X]$. Vrijheid is
\cref{prop:b1:vspaces:freecriteria}~(1). Voor de voortbrengende
eigenschap, redeneer via eindige afdaling op de graad: zij $Q \in
K_n[X]$, $Q \neq 0$, van graad $d$, met leidende coëfficiënt $a$,
en zij $b \neq 0$ de leidende coëfficiënt van $P_d$. Dan heeft $Q -
\frac ab P_d$ graad $< d$ (de topterme heffen op); door $Q$ te
vervangen door dit verschil en te itereren bereikt men na hoogstens
$n + 1$ stappen de nulveelterm, en het afwikkelen van de
aftrekkingen drukt $Q$ uit als een combinatie van de $P_k$. Twee
trappen die we al tegenkwamen: de verschoven machten
$\bigl((X-a)^k\bigr)_{0 \leq k \leq n}$ (\cref{exo:b1:vspaces:4}),
en de Newton-producten $\bigl((X - x_0)(X - x_1)\cdots(X -
x_{k-1})\bigr)_{0 \leq k \leq n}$, aan het werk gezet in het
weekendprobleem.
\end{example}

\begin{example}[Vrijheid in functieruimten]\label{ex:b1:vspaces:functionfree}
In $\mathcal{F}(\R,\R)$ is de familie $(\eu^{a_1 x}, \dots,
\eu^{a_p x})$ met $a_1 < \dots < a_p$ \omterm{def:b1:vspaces:free}{vrij}: deel een nulcombinatie
door $\eu^{a_p x}$ en laat $x \to +\infty$; de laatste coëfficiënt
sterft, en men cascadeert naar beneden (\cref{exo:b1:vspaces:8}
werkt dit en varianten uit). Vrijheid van functies wordt bewezen
door te \emph{evalueren}: in welgekozen punten, in het oneindige,
of na afleiden.
\end{example}

\begin{example}[Een verborgen relatie krimpt een opspansel]
\label{ex:b1:vspaces:hiddenrelation}
In $\mathcal{F}(\R, \R)$, wat is
$\operatorname{Vect}(1,\ \cos^2,\ \sin^2)$? De identiteit $\cos^2 +
\sin^2 = 1$ is een niet-triviale nulcombinatie
\[
1\cdot\mathbf{1} + (-1)\cos^2 + (-1)\sin^2 = 0 :
\]
de familie is gebonden, en het \omterm{def:b1:vspaces:span}{opspansel} wordt al voortgebracht
door $(1, \cos^2)$ alleen ($\sin^2 = 1 - \cos^2$). Die kleinere
familie is \omterm{def:b1:vspaces:free}{vrij}: $a + b\cos^2 x = 0$ voor alle $x$ geeft, in $x =
0$ en $x = \frac\pi2$: $a + b = 0$ en $a = 0$. Dus het \omterm{def:b1:vspaces:span}{opspansel} is
een \emph{vlak} binnen de functieruimte --- en het bevat ook $\cos
2x = 2\cos^2 x - 1$: lineair ogende families van trigonometrische
functies storten routineus in onder identiteiten, en daarom moet
vrijheid \emph{bewezen} worden, nooit verondersteld uit de lengte
van de lijst.
\end{example}

\begin{remark}[Veelvoorkomende valkuilen]
\label{rem:b1:vspaces:pitfalls}
Vier klassieke valstrikken.
\emph{Paarsgewijs is niet genoeg}: in $\R^2$ zijn de vectoren
$(1,0)$, $(0,1)$, $(1,1)$ paarsgewijs niet-proportioneel, en toch
gebonden --- vrijheid is een eigenschap van de \emph{hele} familie,
getest door één globale combinatie, nooit twee aan twee.
\emph{De nulvector vergiftigt alles}: elke familie die $0$ bevat is
gebonden ($1\cdot 0 = 0$ is een niet-triviale relatie), hoe
onschuldig de andere vectoren ook zijn.
\emph{Vereniging is geen som}: $F \cup G$ is bijna nooit een
\omterm{def:b1:vspaces:subspace}{deelruimte} (\cref{def:b1:vspaces:subspace}); de kleinste \omterm{def:b1:vspaces:subspace}{deelruimte}
die beide bevat is $F + G$, meestal veel groter dan de vereniging
--- in $\R^2$ hebben twee verschillende rechten een kruis als
vereniging en het hele vlak als som.
\emph{Direct vereist triviale doorsnede, geen disjunctheid}: twee
\omterm{def:b1:vspaces:subspace}{deelruimten} zijn nooit disjunct (beide bevatten $0$); de juiste
voorwaarde is $F \cap G = \{0\}$, en die moet \emph{bewezen}
worden, niet afgelezen van een tekening --- vgl.\
\cref{ex:b1:vspaces:manysupplements}, waar veel verschillende $G$
werken.
\emph{Vrijheid hangt af van de scalairen}: het paar $(1, \iu)$ is
\omterm{def:b1:vspaces:free}{vrij} in $\C$ gezien als $\R$-vectorruimte, maar gebonden in $\C$
gezien als $\C$-vectorruimte ($\iu\cdot 1 + (-1)\cdot\iu = 0$).
Weet altijd welk \omterm{def:b1:structures:field}{lichaam} werkzaam is voordat je een familie \omterm{def:b1:vspaces:free}{vrij}
verklaart --- het weekendprobleem van \cref{ch:b1:findim} zet
precies deze gevoeligheid om in irrationaliteitsbewijzen.
\end{remark}

\begin{remark}[Waar deze taal naartoe gaat]
\label{rem:b1:vspaces:whereused}
Alles na dit hoofdstuk spreekt de taal die hier wordt vastgelegd.
\cref{ch:b1:findim} telt basisvectoren en zet ``\omterm{def:b1:vspaces:free}{vrij}'' en
``voortbrengend'' om in ongelijkheden over één enkel geheel getal,
de dimensie. \cref{ch:b1:linmaps} bestudeert de \omterm{def:b1:logic:map}{afbeeldingen} die
compatibel zijn met de twee bewerkingen; directe sommen worden daar
projectoren. \cref{ch:b1:matrices} codeert vectoren door hun
\omterm{prop:b1:vspaces:coordinates}{coördinaten} in een \omterm{def:b1:vspaces:free}{basis} --- \cref{prop:b1:vspaces:coordinates} is
het licentie voor die codering --- en \cref{ch:b1:euclid} voegt
lengtes en hoeken toe bovenop de lineaire structuur. In het
Jaar~2-volume lopen dezelfde axioma's, woordelijk, over willekeurige
lichamen en in oneindige dimensie; niets in dit hoofdstuk gebruikte
ergens eindigheid.
\end{remark}

\begin{remark}[Drie draden om doorheen Boek 3 te volgen]
\label{rem:b1:vspaces:threads}
Volg drie specifieke ideeën van dit hoofdstuk in hun groei.
\emph{Het trapprincipe} (\cref{ex:b1:vspaces:staircase}) verschijnt
opnieuw als de \omterm{def:b1:vspaces:free}{Newton-basis} in het weekendprobleem van dit
hoofdstuk, als de binomiale \omterm{def:b1:vspaces:free}{basis} $(B_k)$ daar, en als de
veelterm-alternanttruc in het weekendprobleem van
\cref{ch:b1:det}: één lemma, drie determinantvrije dividenden.
\emph{Evaluatie als vrijheidstest}
(\cref{ex:b1:vspaces:functionfree}) wordt het interpolatie-\linebreak
isomorfisme van \cref{ch:b1:linmaps}, dan het Vandermonde-criterium
van \cref{ch:b1:det}, dan de Gram-test van \cref{ch:b1:euclid}:
dezelfde reflex, drie keer aangescherpt. \emph{Directe sommen}
(\cref{def:b1:vspaces:sum}) worden projectoren in
\cref{ch:b1:linmaps}, orthogonale splitsingen $E = F \oplus
F^\perp$ in \cref{ch:b1:euclid}, en de verklaarde-plus-residuele
ontbinding van kleinste kwadraten in het weekendprobleem van
\cref{ch:b1:multivar}. Heel weinig van dit boek is, ten diepste,
niet één van deze drie ideeën in nieuwe kleren.
\end{remark}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:vspaces:7}
Zij $F, G, H$ \omterm{def:b1:vspaces:subspace}{deelruimten} van $E$. Bewijs dat

F \cap (G + (F \cap H)) = (F \cap G) + (F \cap H),
\]
en toon met een voorbeeld in $\R^2$ dat de onbeperkte

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:vspaces:8}

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:vspaces:9}

Geef een tegenvoorbeeld zonder de hypothese $G \subseteq H$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:vspaces:10}
In $\R[X]$, zij $\mathcal P$ de \omterm{def:b1:logic:sets}{verzameling} van \emph{even}

\oplus \mathcal I$, en toon dat $\mathcal P =
\operatorname{Vect}(1, X^2, X^4, \dots)$, d.w.z.\ dat de even

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:vspaces:11}

+ x_3,\ x_3 + x_4,\ x_4 + x_1)$ \omterm{def:b1:vspaces:free}{vrij} wanneer $(x_1, x_2, x_3, x_4)$
het is?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:vspaces:12}
Zij $E$ een \omterm{def:b1:vspaces:def}{vectorruimte} over $\R$ (of $\C$) en $F_1, \dots, F_k$
\emph{eigenlijke} \omterm{def:b1:vspaces:subspace}{deelruimten} van $E$ (elke $F_i \neq E$).
\begin{enumerate}
  \item Behandel het geval $k = 2$ rechtstreeks: als $F_1
        \not\subseteq F_2$ en $F_2 \not\subseteq F_1$, kies $x \in
        F_1 \setminus F_2$ en $y \in F_2 \setminus F_1$ en lokaliseer
        $x + y$.
  \item Bewijs in het algemeen dat $E \neq F_1 \cup \dots \cup

        een eindige vereniging van eigenlijke \omterm{def:b1:vspaces:subspace}{deelruimten}.
        \emph{(Neem $k$ minimaal, kies $x \in F_1$ buiten de andere
        $F_i$, kies $y \notin F_1$, en volg de rechte $t \mapsto y
        + tx$.)}
\end{enumerate}
\end{exercise}

\section{Probleem: interpolatie, drie bases voor één ruimte}

\begin{problem}\label{pb:b1:vspaces:1}
Fixeer $n + 1$ \emph{verschillende} punten $x_0, x_1, \dots, x_n$
van $\R$. Dit probleem herbekijkt \omterm{thm:b1:poly:lagrange}{Lagrange-interpolatie}
(\cref{thm:b1:poly:lagrange}) met de ogen van dit hoofdstuk: de
ruimte $\R_n[X]$ draagt drie natuurlijke bases --- die van
Lagrange, die van Newton, en (voor gelijkmatig verspreide punten)
de binomiale \omterm{def:b1:vspaces:free}{basis} --- en elke \omterm{def:b1:vspaces:free}{basis} maakt één vraag gemakkelijk.
De weg eindigt bij een echte rekenkundige stelling: Pólya's
karakterisatie van de \omterm{def:b1:poly:def}{veeltermen} die $\Z$ in $\Z$
afbeelden.\index{Lagrange-interpolatie}
\index{gedeelde differenties}\index{geheelwaardige veelterm}

\medskip
\noindent\textbf{Deel I --- De \omterm{def:b1:vspaces:free}{Lagrange-basis}.} Voor $0 \leq i
\leq n$ stel
\[
L_i \;=\; \prod_{j \neq i} \frac{X - x_j}{x_i - x_j} \;\in\;
\R_n[X].
\]
\begin{enumerate}
  \item Ga na dat $\deg L_i = n$ en dat $L_i(x_j) = 1$ als $j =
        i$, en $0$ als $j \neq i$.
  \item Bewijs dat de familie $(L_0, \dots, L_n)$ \omterm{def:b1:vspaces:free}{vrij} is.
  \item Bewijs dat voor elke $P \in \R_n[X]$,
        \[
        P \;=\; \sum_{i=0}^{n} P(x_i)\, L_i ,
        \]
        en besluit dat $(L_0, \dots, L_n)$ een \omterm{def:b1:vspaces:free}{basis} van $\R_n[X]$
        is. \emph{(Beschouw het verschil van de twee leden en tel
        zijn wortels, \cref{cor:b1:poly:nroots}.)}
  \item Leid de interpolatiestelling af: voor willekeurige waarden
        $y_0, \dots, y_n \in \R$ bestaat er een \emph{unieke} $P
        \in \R_n[X]$ met $P(x_i) = y_i$ voor alle $i$. Wat zijn, in
        de \omterm{def:b1:vspaces:free}{Lagrange-basis}, de \omterm{prop:b1:vspaces:coordinates}{coördinaten} van een \omterm{def:b1:poly:def}{veelterm} $P$?
  \item Bewijs de identiteiten
        \[
        \sum_{i=0}^{n} L_i = 1
        \qquad\text{en, voor } 0 \leq k \leq n,\qquad
        \sum_{i=0}^{n} x_i^{k}\, L_i = X^{k} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De \omterm{def:b1:vspaces:free}{Newton-basis} en \omterm{pb:b1:vspaces:1}{gedeelde
differenties}.} Stel $N_0 = 1$ en $N_k = (X - x_0)(X - x_1) \cdots
(X - x_{k-1})$ voor $1 \leq k \leq n$. Voor een functie $f$
gedefinieerd in de knopen, definieer de \emph{\omterm{pb:b1:vspaces:1}{gedeelde
differenties}} door $f[x_i] = f(x_i)$ en
\[
f[x_i, \dots, x_{i+k}] \;=\;
\frac{f[x_{i+1}, \dots, x_{i+k}] - f[x_i, \dots, x_{i+k-1}]}
{x_{i+k} - x_i} .
\]
\begin{enumerate}[resume]
  \item Bewijs dat $(N_0, N_1, \dots, N_n)$ een \omterm{def:b1:vspaces:free}{basis} van
        $\R_n[X]$ is.
  \item Bereken $f[x_0, x_1]$ en $f[x_0, x_1, x_2]$ in termen van
        de waarden van $f$, en bereken dan alle \omterm{pb:b1:vspaces:1}{gedeelde
        differenties} van $f(x) = x^2$ in drie willekeurige knopen.
  \item (Lemma van Aitken) Zij $R$ interpolant van $f$ in $x_0,
        \dots, x_{n-1}$ en $Q$ interpolant van $f$ in $x_1, \dots,
        x_n$, beide van graad $\leq n - 1$. Bewijs dat
        \[
        S \;=\; \frac{(X - x_0)\,Q - (X - x_n)\,R}{x_n - x_0}
        \]
        $f$ interpoleert in $x_0, x_1, \dots, x_n$.
  \item Leid, door inductie op het aantal knopen, af dat de
        coëfficiënt van $X^{k}$ in de interpolant van $f$ in $x_0,
        \dots, x_k$ precies $f[x_0, \dots, x_k]$ is.
  \item Bewijs \emph{Newtons interpolatieformule}: de interpolant
        van $f$ in $x_0, \dots, x_n$ is
        \[
        P \;=\; \sum_{k=0}^{n} f[x_0, \dots, x_k]\, N_k ,
        \]
        en leid de \omterm{def:b1:topology:closed}{gesloten} formule af
        \[
        f[x_0, \dots, x_k] \;=\; \sum_{i=0}^{k}
        \frac{f(x_i)}{\prod_{j \neq i,\, j \leq k} (x_i - x_j)} ,
        \]
        die toont dat $f[x_0, \dots, x_k]$ niet afhangt van de
        ordening van de knopen.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Gelijkmatig verspreide knopen: de
verschiloperator.} Vanaf nu zijn de knopen $0, 1, 2, \dots$ en, voor
een \omterm{def:b1:poly:def}{veelterm} $P$, stellen we
\[
\Delta P(X) = P(X + 1) - P(X),
\qquad
B_k = \frac{X(X-1)\cdots(X-k+1)}{k!} \quad (B_0 = 1).
\]
\begin{enumerate}[resume]
  \item Toon dat als $\deg P = m \geq 1$ met leidende coëfficiënt
        $a$, dan $\deg \Delta P = m - 1$ met leidende coëfficiënt
        $m\,a$, en dat $\Delta$ constanten doodt.
  \item Toon dat $(B_0, B_1, \dots, B_n)$ een \omterm{def:b1:vspaces:free}{basis} van $\R_n[X]$
        is en dat $\Delta B_k = B_{k-1}$ voor $k \geq 1$.
  \item (Newtons voorwaartse-verschilformule) Bewijs dat elke $P
        \in \R_n[X]$ voldoet aan
        \[
        P \;=\; \sum_{k=0}^{n} \bigl(\Delta^{k} P\bigr)(0)\, B_k .
        \]
  \item Bewijs dat voor elke $k \geq 0$,
        \[
        \bigl(\Delta^{k} P\bigr)(0) \;=\;
        \sum_{j=0}^{k} (-1)^{k-j} \binom{k}{j} P(j) .
        \]
  \item Toon dat als $\deg P = n$ met leidende coëfficiënt $a_n$,
        dan $\Delta^{n} P$ de constante $n!\,a_n$ is en
        $\Delta^{n+1} P = 0$.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- \omterm{pb:b1:vspaces:1}{Geheelwaardige veeltermen}.} Een
\omterm{def:b1:poly:def}{veelterm} $P \in \R[X]$ is \emph{geheelwaardig} wanneer $P(m) \in
\Z$ voor elke $m \in \Z$.
\begin{enumerate}[resume]
  \item Bewijs dat elke $B_k$ geheelwaardig is. \emph{(Behandel $m
        \geq k$, $0 \leq m < k$ en $m < 0$ apart; voor $m = -q <
        0$, toon $B_k(-q) = (-1)^k \binom{q + k - 1}{k}$.)}
  \item Bewijs \emph{Pólya's karakterisatie}: $P \in \R_n[X]$ is
        geheelwaardig als en slechts als zijn \omterm{prop:b1:vspaces:coordinates}{coördinaten} in de
        \omterm{def:b1:vspaces:free}{basis} $(B_0, \dots, B_n)$ gehele getallen zijn.
  \item Leid af: als $P \in \R_n[X]$ gehele waarden aanneemt in $n
        + 1$ \emph{opeenvolgende} gehele getallen $a, a+1, \dots,
        a+n$, dan is $P$ geheelwaardig. \emph{(Verschuiving: pas de
        studie toe op $Q(X) = P(X + a)$.)}
  \item Leid uit vraag 16 af dat een product van $k$ opeenvolgende
        gehele getallen altijd \omterm{def:b1:arith:divides}{deelbaar} is door $k!$.
  \item Zij $P = \dfrac{X(X+1)(2X+1)}{6}$. Bereken zijn
        Newton-tabel in $0, 1, 2, 3$, schrijf $P$ in de \omterm{def:b1:vspaces:free}{basis}
        $(B_k)$, en besluit dat $P$ geheelwaardig is hoewel geen
        van zijn monomiale coëfficiënten een geheel getal is.
        Verifieer $\Delta P = (X+1)^2$ en leid af $P(m) = 1^2 + 2^2
        + \dots + m^2$ voor $m \in \N$.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Dividenden.}
\begin{enumerate}[resume]
  \item Zij $P \in \R_n[X]$ interpolant van de waarden $2^i$ in $i
        = 0, 1, \dots, n$. Toon dat $P = B_0 + B_1 + \dots + B_n$
        en dat $P(n + 1) = 2^{n+1} - 1$: het ``verdubbelingspatroon''
        breekt altijd bij het allervolgende punt.
  \item (Discrete primitieve) Bewijs dat voor alle gehele getallen
        $m \geq 1$ en $k \geq 0$,
        \[
        \sum_{j=0}^{m-1} B_k(j) \;=\; B_{k+1}(m),
        \]
        d.w.z.\ de hockeystick-identiteit $\sum_{j=k}^{m-1}
        \binom{j}{k} = \binom{m}{k+1}$.
  \item Ontwikkel $X^2$ en $X^3$ in de \omterm{def:b1:vspaces:free}{basis} $(B_k)$ en leid
        \omterm{def:b1:topology:closed}{gesloten} formules af voor $\sum_{j=0}^{m-1} j^2$ en
        $\sum_{j=0}^{m-1} j^3$; herwin Nicomachus' identiteit $1^3
        + \dots + m^3 = (1 + \dots + m)^2$.
  \item Neem $n = 2$ en knopen $0, 1, 2$. Schrijf de \omterm{prop:b1:vspaces:coordinates}{coördinaten}
        van $X^2$ in de drie bases van dit probleem: de monomiale
        \omterm{def:b1:vspaces:free}{basis}, de \omterm{def:b1:vspaces:free}{Lagrange-basis}, de \omterm{def:b1:vspaces:free}{Newton-basis}. Controleer de
        drie antwoorden tegen vragen 4 en 9.
  \item Synthese. In vier zinnen: welk vectorruimtebegrip vraag 4
        automatisch maakt; waarom de \omterm{def:b1:vspaces:free}{Newton-basis} \omterm{prop:b1:vspaces:coordinates}{coördinaten}
        \emph{recursief} berekent terwijl de \omterm{def:b1:vspaces:free}{Lagrange-basis} ze
        \emph{onmiddellijk} afleest; welk vrijheidscriterium beide
        bases delen; en in welke precieze zin Pólya's stelling zegt
        dat de geheeltalligheid van een \omterm{def:b1:poly:def}{veelterm} een eigenschap is
        van zijn \omterm{prop:b1:vspaces:coordinates}{coördinaten} \emph{in de juiste \omterm{def:b1:vspaces:free}{basis}}.
\end{enumerate}
\end{problem}
