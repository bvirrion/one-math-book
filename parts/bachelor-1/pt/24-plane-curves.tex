\chapter{Curvas planas}\label{ch:b1:curves}

O cálculo de \cref{ch:b1:derivative,ch:b1:taylor} foi construído para
funções $y = f(x)$; a maioria das curvas de geometria e mecânica ---
trajetórias, círculos rolados em círculos, órbitas --- recusar essa forma
e venha em vez disso como \emph{parametrizado} curvas $ t \mapsto (x(t),
y(t))$ ou como \emph{polar} curvas $ r = r(\theta)$. Este clássico
o capítulo estuda ambos e termina com o cônicas.

\section{Curvas parametrizadas}

\begin{definition}\label{def:b1:curves:def}
UM \emph{curva parametrizada}\index{curva parametrizada} é um \omterm{def:b1:logic:map}{mapa} $ f
\colon I \to \R^2$, $ t \mapsto M(t) = (x(t), y(t))$, com $ x, y $ de
aula $ C^1$ (pelo menos) no \omterm{prop:b1:reals:intervals}{intervalo} $ I $. O \emph{velocidade
vetor} é $ f'(t) = (x'(t), y'(t))$; o ponto $ M(t_0)$ é
\emph{regular} quando $ f'(t_0) \neq (0,0)$, e o \emph{tangente}
há a linha através $ M(t_0)$ dirigido por $ f'(t_0)$.
\end{definition}

\begin{proof}[Por que $ f'$ direciona a tangente]
Por Taylor--Young em termos de componentes, $ M(t) = M(t_0) + (t - t_0) f'(t_0) +
o(t - t_0)$: a direção do acorde $\frac{M(t) - M(t_0)}{t - t_0}$
tende a $ f'(t_0)$. A hipótese de regularidade é o que faz com que a
limitar um \emph{direção}: se $ f'(t_0) = 0$, a tela entra em colapso
para $ M(t) = M(t_0) + o(t - t_0)$ e não diz nada sobre como
curva deixa o ponto --- de onde o tratamento separado de
pontos singulares no método abaixo, onde o primeiro
\emph{que não desaparece} derivado assume o papel de $ f'$.
Observe também que a linha tangente é um objeto geométrico: qualquer
mudanças de reparametrização $ f'$ por um fator escalar diferente de zero e
deixa a linha inalterada.
\end{proof}

\begin{method}[Estudando uma curva parametrizada]\label{met:b1:curves:study}
\begin{enumerate}
  \item \emph{Reduza o domínio} usando simetrias: relações entre
        $ M(-t)$, $ M(t + T)$, \dots\ e $ M(t)$ (reflexões,
        traduções, periodicidade), depois desenhe apenas a parte reduzida.
  \item \emph{Variações}: tabular sinais de $ x'$ e $ y'$ conjuntamente;
        a curva se move para a direita/esquerda conforme $ x'$, para cima/para baixo como $ y'$.
  \item \emph{Pontos notáveis}: tangente horizontal ($ y' = 0 \neq
        x'$), tangente vertical ($ x' = 0 \neq y'$); em um
        \emph{singular} apontar ($ f' = 0$), examine superior derivados
        para a direção tangente.
  \item \emph{Comportamento assintótico} nos finais de $ I $ e, em seguida, esboce.
\end{enumerate}
\end{method}

\begin{example}[Ramos assintóticos, trabalhados]
\label{ex:b1:curves:asymptotes}
$ x(t) = t $, $ y(t) = t + \dfrac1t $ sobre $\intoo{0}{+\infty}$. Como $ t
\to 0^{+}$: $ x \to 0$ enquanto $ y \to +\infty $ --- a curva sobe
ao longo do \emph{assíntota vertical} $ x = 0$ (\omterm{def:b1:topology:closure}{limite} finito para
uma coordenada, infinita para a outra). Como $ t \to +\infty $:
ambos \omterm{prop:b1:vspaces:coordinates}{coordenadas} explodir, então teste uma linha: a diferença
\[
y(t) - x(t) = \frac1t \longrightarrow 0^{+}
\]
exibe o \emph{assíntota oblíqua} $ y = x $, aproximou-se de
acima. No meio, $ y' = 1 - \frac1{t^2}$ desaparece em $ t = 1$:
o ponto $(1, 2)$ é o ponto mais baixo do galho, e $ x' = 1 >
0$ o tempo todo, então a curva sempre avança para a direita. Um arco,
duas assíntotas, um mínimo: uma \omterm{def:b1:linmaps:kerim}{imagem} completa de três
cálculos --- com a receita geral visível abaixo:
quando $ x, y \to \infty $ juntos, examinar $ y/x $ para um candidato
declive (aqui $\to 1$), então $ y - (\text{slope})\,x $ para o
interceptação e o lado de aproximação.
\end{example}

\begin{example}[O astroide]\label{ex:b1:curves:astroid}
$ x(t) = \cos^3 t $, $ y(t) = \sin^3 t $.\index{astroide} Simetrias:
$ M(t + 2\pi) = M(t)$ (estudar um \omterm{def:b1:vspaces:span}{período}); $ M(-t)$ é o reflexo de
$ M(t)$ no $ x $-eixo; $ M(\pi - t)$ no $ y $-eixo; $ M(\frac\pi2 -
t)$ na diagonal $ y = x $: basta estudar $ t \in
\intcc{0}{\frac\pi4}$ e desdobrar.

Velocidade: $ f'(t) = 3\sin t\cos t\,(-\cos t, \sin t)$. Sobre
$\intoo{0}{\frac\pi2}$ todos os pontos são regulares com tangente direcionada
por $(-\cos t, \sin t)$; no $ t = 0$ (o ponto $(1,0)$) a velocidade
desaparece: um \emph{cúspide}\index{cúspide}, onde a curva inverte ao longo
a direção tangente $(-1, 0)\cdot(\pm)$ --- por simetria os quatro
\omterm{ex:b1:curves:astroid}{cúspides} sente-se em $(\pm1, 0), (0, \pm1)$.
\end{example}

\begin{example}[Uma figura oito, estudada na íntegra]
\label{ex:b1:curves:lissajous}
$ x(t) = \sin t $, $ y(t) = \sin 2t $ (um \emph{Curva de Lissajous}).
\emph{Simetrias}: $ M(t + \pi) = (-x(t), y(t))$ (reflexão em
o $ y $-eixo), $ M(-t) = (-x(t), -y(t))$ (simetria central),
$ M(\pi - t) = (x(t), -y(t))$ (reflexo no $ x $-eixo): é
basta estudar $ t \in \intcc{0}{\frac\pi2}$ e desdobrar.
\emph{Variações}: $ x' = \cos t \geq 0$ por toda parte, enquanto $ y' =
2\cos 2t $ é positivo antes $ t = \frac\pi4$ e negativo depois:
o arco sobe para a direita até o cume
$\bigl(\frac{\sqrt2}2,\ 1\bigr)$ no $ t = \frac\pi4$ (horizontal
tangente), depois desce para a direita até $(1, 0)$ no $ t =
\frac\pi2$, onde $ x' = 0 \neq y'$: tangente vertical.
\emph{Ponto duplo}: $ M(0) = M(\pi) = (0,0)$, com dois diferentes
velocidades
\[
f'(0) = (1,\ 2), \qquad f'(\pi) = (-1,\ 2) :
\]
dois \emph{regular} ramos que se cruzam na origem em pontos distintos
ângulos --- um ponto duplo, não um ponto singular: cada passagem é
perfeitamente lisa, as duas passagens apenas compartilham sua localização.
A curva inteira é o oito abaixo.
\end{example}

\section{Curvas polares}

\begin{definition}\label{def:b1:curves:polar}
UM \emph{curva polar}\index{curva polar} é dado por $ r = r(\theta)$:
o ponto do parâmetro $\theta $ é
\[
M(\theta) = r(\theta)\,\vec u(\theta),
\qquad \vec u(\theta) = (\cos\theta, \sin\theta).
\]
Com $\vec v(\theta) = (-\sin\theta, \cos\theta) = \vec
u\,'(\theta)$, a velocidade é
\[
M'(\theta) = r'(\theta)\, \vec u(\theta) + r(\theta)\, \vec
v(\theta) .
\]
Consequências: onde $ r \neq 0$, o ponto é regular e o
tangente faz com o raio o ângulo $ V $ dado por $\tan V =
\frac{r}{r'}$ (ângulo entre $ M'$ e $\vec u $); onde $ r(\theta_0)
= 0$, a curva passa pela origem \emph{com tangente o raio
$\theta = \theta_0$} (direção $\vec u(\theta_0)$, leia
$ M'(\theta_0) = r'(\theta_0)\vec u(\theta_0)$, ou do \omterm{def:b1:topology:closure}{limite}
acorde: o acorde de $ O $ para $ M(\theta)$ é transportado por $\vec
u(\theta)$ em si, o que tende a $\vec u(\theta_0)$ --- então o
regra é válida mesmo quando $ r'(\theta_0) = 0$ e o ponto é
singular, e é por isso que polar \omterm{ex:b1:curves:astroid}{cúspides} na origem, como o
cardióides, obtenha sua tangente gratuitamente).
\end{definition}

\begin{example}[O cardióide]\label{ex:b1:curves:cardioid}
$ r(\theta) = 1 + \cos\theta $.\index{cardióide} Simetria: $ r(-\theta)
= r(\theta)$: reflexão no $ x $-eixo; estudar $\theta \in
\intcc{0}{\pi}$. $ r $ diminui de $2$ para $0$; no $\theta = \pi $,
$ r = 0$: a curva atinge a origem tangencialmente ao raio
$\theta = \pi $ (o $ x $-eixo), formando um \omterm{ex:b1:curves:astroid}{cúspide} lá --- o coração
ponto. Tangente em $\theta = 0$: $ r' = 0$, então $\tan V = \infty $:
perpendicular ao eixo.

O ângulo $ V $ merece mais uma leitura. No $\theta =
\frac\pi2$: $ r = 1$ e $ r' = -1$, então $\tan V = \frac{r}{r'} =
-1$: a tangente forma três quartos de um ângulo reto com o
raio de saída --- a curva já está se curvando em direção ao seu
\omterm{ex:b1:curves:astroid}{cúspide}. A fórmula $\tan V = r/r'$ fornece direções tangentes
ao longo de toda a curva sem cálculo de $ M'(\theta)$
seja qual for: é o análogo polar da leitura de uma inclinação.
\end{example}

\begin{example}[Polar para cartesiano: um círculo oculto]
\label{ex:b1:curves:polarcircle}
Qual é o \omterm{def:b1:curves:polar}{curva polar} $ r = 2\cos\theta $? Multiplicar por $ r $:
$ r^2 = 2r\cos\theta $, ou seja\ $ x^2 + y^2 = 2x $, ou seja
\[
(x - 1)^2 + y^2 = 1 :
\]
o círculo do centro $(1, 0)$ e raio $1$, passando por
a origem. A contabilidade é importante: como $\theta $ atropela
$\intoc{-\frac\pi2}{\frac\pi2}$ todo o círculo é varrido
exatamente uma vez ($ r $ desaparece em ambas as extremidades), e em $\theta =
\pm\frac\pi2$ a regra ``tangente na origem ao longo do raio
$\theta = \theta_0$'' dá um \emph{vertical} tangente ali ---
correspondendo à geometria, uma vez que o eixo vertical é de fato tangente
para este círculo em $ O $. Para $\theta $ além dessa faixa, $ r < 0$
refaz o mesmo círculo: um lembrete de que um \omterm{def:b1:curves:polar}{curva polar} é um
\emph{parametrizado} objeto, permitido passar por cima de si mesmo.
\end{example}

\section{Comprimento do arco}

\begin{definition}[Comprimento de um arco]\label{def:b1:curves:arclength}
O \emph{comprimento}\index{comprimento} de um $ C^1$ arco $ f \colon
\intcc{a}{b} \to \R^2$ é o integrante da velocidade:
\[
L = \int_a^b \norm{f'(t)}\,\dd t
= \int_a^b \sqrt{x'(t)^2 + y'(t)^2}\;\dd t .
\]
(Motivação: em um pequeno \omterm{prop:b1:reals:intervals}{intervalo}, $ M(t + h) \approx M(t) + h
f'(t)$, então o arco está próximo de um polígono cujos comprimentos de segmento somam
para uma soma de Riemann de $\norm{f'}$,
\cref{thm:b1:integration:riemann}.) Por um \omterm{def:b1:curves:polar}{curva polar} $ r =
r(\theta)$, a velocidade $ r'\vec u + r\vec v $ tem ortogonal
componentes, então
\[
L = \int_{\theta_1}^{\theta_2}
\sqrt{r'(\theta)^2 + r(\theta)^2}\;\dd\theta .
\]
O comprimento não depende do (monótono, $ C^1$)
parametrização escolhida: substituindo $ t = \varphi(s)$ no
integrante (\cref{thm:b1:integration:parts}) multiplica $ f'$ por
$\varphi'$ e $\dd t $ por $\varphi'^{-1}$.
\end{definition}

\begin{example}[Verificação de sanidade: o círculo]\label{ex:b1:curves:circlelength}
Para $ f(t) = (R\cos t, R\sin t)$ sobre $\intcc{0}{2\pi}$: $\norm{f'} =
R $, então $ L = 2\pi R $ --- a definição retorna a circunferência.
Teste de reparametrização: o \omterm{def:b1:logic:map}{mapa} $ g(t) = (R\cos 2t, R\sin 2t)$ sobre
$\intcc{0}{\pi}$ desenha o \emph{mesmo} círculo em velocidade dobrada
$\norm{g'} = 2R $, e
\[
\int_0^{\pi} 2R\,\dd t = 2\pi R
\]
novamente: metade do tempo, duas vezes a velocidade, o mesmo \omterm{def:b1:curves:arclength}{comprimento} --- o
invariância prometida na definição, observada uma vez em números.
(Correndo $ g $ em todos $\intcc{0}{2\pi}$ daria $4\pi R $: um
curva percorrida duas vezes tem o dobro do \omterm{def:b1:curves:arclength}{comprimento} \emph{como uma viagem};
\omterm{def:b1:curves:arclength}{comprimento} mede o caminho parametrizado e a contabilidade honesta de
o \omterm{prop:b1:reals:intervals}{intervalo} faz parte do cálculo.)
Para o astroide $(\cos^3t, \sin^3t)$: $\norm{f'} = 3\abs{\sin
t\cos t} = \tfrac32\abs{\sin 2t}$, e por simetria $ L = 4\int_0^{\pi/2}
\tfrac32\sin 2t\,\dd t = 6$: uma curva desenhada dentro do círculo unitário,
de \omterm{def:b1:curves:arclength}{comprimento} $6 < 2\pi $. O problema do fim de semana mede o mais famoso
arco de todos.
\end{example}

\section{Cônicas}

\begin{definition}[Foco - definição de diretriz]\label{def:b1:curves:conics}
Fixar um ponto $ F $ (\emph{foco}), uma linha $ D $ não através $ F $
(\emph{diretriz}) e $ e > 0$ (\emph{excentricidade}). O
\emph{cônico}\index{cônico} desses dados é
\[
\mathcal{C} = \{M : d(M, F) = e\; d(M, D)\}:
\]
um \emph{elipse} para $ e < 1$, um \emph{parábola} para $ e = 1$, um
\emph{hipérbole} para $ e > 1$. (O círculo aparece como um degenerado
\omterm{def:b1:topology:closure}{limite} $ e \to 0$.)
\end{definition}

\begin{example}[A definição, verificada em uma parábola]
\label{ex:b1:curves:parabolacheck}
Pegue a parábola $ y^2 = 4x $, ou seja\ $2p = 4$: foco $ F = (1, 0)$
e diretriz $ D : x = -1$ (a forma reduzida
\cref{thm:b1:curves:reduced} coloca-os em $\pm\frac p2$). No
ponto $ M = (1, 2)$ da curva:
\[
MF = \sqrt{(1-1)^2 + 2^2} = 2,
\qquad
d(M, D) = 1 - (-1) = 2 :
\]
igual, como $ e = 1$ demandas. No $ M' = (4, 4)$: $ MF' = \sqrt{9 +
16} = 5$ e $ d(M', D) = 5$ de novo. O foco - diretriz
definição não é uma abstração: é um par de distâncias uma
pode medir em qualquer ponto, e a álgebra do reduzido
equações nada mais é do que essa medição feita de uma vez por todas
tudo.
\end{example}

\begin{theorem}[Equações reduzidas]\label{thm:b1:curves:reduced}
Em um bem escolhido \omterm{def:b1:euclid:orthogonal}{ortonormal} quadro:
\[
\text{ellipse: } \frac{x^2}{a^2} + \frac{y^2}{b^2} = 1
\quad (a \geq b > 0),
\qquad
\text{hyperbola: } \frac{x^2}{a^2} - \frac{y^2}{b^2} = 1,
\qquad
\text{parabola: } y^2 = 2px,
\]
com, para a elipse: focos em $(\pm c, 0)$, $ c = \sqrt{a^2 - b^2}$,
$ e = \frac ca $, e a caracterização bifocal $ MF + MF' = 2a $;
para a hipérbole: $ c = \sqrt{a^2 + b^2}$, $ e = \frac ca $,
$\abs{MF - MF'} = 2a $, assíntotas $ y = \pm\frac ba x $.
\end{theorem}

\begin{proof}
Tome o foco na origem e a diretriz vertical, $ x = -h $
($ h > 0$): a condição $ MF^2 = e^2 d(M, D)^2$ lê $ x^2 + y^2 =
e^2 (x + h)^2$. Para $ e = 1$: $ y^2 = 2hx + h^2$, uma parábola após
mudança $ x \mapsto x - \frac h2$ (então $ p = h $). Para $ e \neq 1$:
completando o quadrado em $ x $,
\[
(1 - e^2)\Bigl(x - \frac{e^2 h}{1 - e^2}\Bigr)^{\!2} + y^2
= e^2h^2 + \frac{e^4 h^2}{1 - e^2}
= \frac{e^2 h^2}{1 - e^2} .
\]
\omterm{def:b1:logic:sets}{Definir} $ X = x - \frac{e^2h}{1-e^2}$ (uma mudança de quadro). Quando $ e < 1$,
divida pelo lado positivo direito:
$\frac{X^2}{a^2} + \frac{y^2}{b^2} = 1$ com $ a = \frac{eh}{1 -
e^2}$, $ b = \frac{eh}{\sqrt{1-e^2}}$; quando $ e > 1$, ambos os lados
o sinal de inversão do display apropriadamente e a mesma \omterm{thm:b1:logic:partition}{divisão} dá
$\frac{X^2}{a^2} - \frac{y^2}{b^2} = 1$ com $ a = \frac{eh}{e^2 -
1}$, $ b = \frac{eh}{\sqrt{e^2 - 1}}$. Os valores declarados de $ c $
seguir ($ c^2 = a^2 - b^2$ ou $ a^2 + b^2$ dá $ c = ea $ em ambos
casos, colocando o foco corretamente), e as propriedades bifocais são
verificações diretas nas equações reduzidas. Em detalhes para o
elipse, com $ F' = (c, 0)$ e $ M = (X, y)$ na curva:
\[
MF'^2 = (X - c)^2 + y^2
= X^2 - 2cX + c^2 + b^2\Bigl(1 - \frac{X^2}{a^2}\Bigr)
= \frac{c^2}{a^2}X^2 - 2cX + a^2
= (a - eX)^2 ,
\]
usando $ c^2 + b^2 = a^2$ e $ e = \frac ca $; desde $\abs X \leq
a $ e $ e < 1$, $ a - eX > 0$, então $ MF' = a - eX $ sem quadrado
raiz já extraída. O cálculo do espelho fornece $ MF = a + eX $,
de onde $ MF + MF' = 2a $, constante. Para a hipérbole o mesmo
rendimentos de álgebra $ MF = \abs{a + eX}$ e $ MF' = \abs{eX - a}$,
com diferença $\pm2a $ de acordo com o ramo.
\end{proof}

\begin{example}\label{ex:b1:curves:conicexample}
$\frac{x^2}{25} + \frac{y^2}{9} = 1$: elipse, $ a = 5$, $ b = 3$, $ c
= 4$: focos $(\pm4, 0)$, excentricidade $\frac45$. Sua parametrização:
$(5\cos t, 3\sin t)$ --- um círculo esticado anisotropicamente; o
a soma das distâncias aos focos de qualquer um de seus pontos é $10$.
\end{example}

\begin{example}[Lendo uma órbita a partir de sua equação polar]
\label{ex:b1:curves:orbit}
O polar \omterm{def:b1:curves:conics}{cônico} $ r = \dfrac{1}{1 + \frac12\cos\theta}$
(\cref{exo:b1:curves:7} com $ p = 1$, $ e = \frac12$) é um
elipse com um \emph{foco na origem} --- a geometria de um
órbita planetária com o Sol em $ O $. Extraia tudo de
$ p $ e $ e $: da redução de
\cref{thm:b1:curves:reduced}, $ a = \dfrac{p}{1 - e^2} =
\dfrac{1}{3/4} = \dfrac43$ e $ c = ea = \dfrac23$. Os dois
apsides verifique sem nenhuma teoria:
\[
r(0) = \frac{1}{3/2} = \frac23 = a - c
\quad (\text{perihelion}),
\qquad
r(\pi) = \frac{1}{1/2} = 2 = a + c
\quad (\text{aphelion}),
\]
e sua soma $\frac23 + 2 = \frac83 = 2a $ recupera o principal
eixo. A forma polar é a natural sempre que um foco é
fisicamente distinto; a forma cartesiana reduzida, sempre que
os eixos de simetria são. Converter entre os dois é exatamente
o que o cálculo do quadrado completo do teorema faz.

A excentricidade como mostrador, finalmente: mantenha $ p = 1$ e vire $ e $ em $ r
= \frac{1}{1 + e\cos\theta}$. No $ e = 0$: o círculo $ r = 1$. No
$ e = \frac12$: a elipse que acabamos de estudar, $ r $ oscilando entre
$\frac23$ e $2$. No $ e = 1$: $ r(\theta) \to \infty $ como $\theta
\to \pi $ --- a curva não fecha mais: uma parábola, é
ponto mais distante empurrado para o infinito. No $ e = 2$: o denominador
desaparece em $\cos\theta = -\frac12$, e apenas $\theta \in
\intoo{-\frac{2\pi}3}{\frac{2\pi}3}$ sobrevive: um ramo de uma
hipérbole, escapando ao longo de duas direções assintóticas. Um
fórmula, toda \omterm{def:b1:curves:conics}{cônico} família e os pontos de transição $ e =
1$ visível no momento em que o denominador chega a zero pela primeira vez.
\end{example}

\begin{remark}[Armadilhas comuns]
\label{rem:b1:curves:pitfalls}
\emph{Pontos duplos não são pontos singulares}: em um
autocruzamento, cada ramo é regular; ``singular'' refere-se a
$ f'(t_0) = 0$ para um valor do parâmetro
(\cref{ex:b1:curves:lissajous} contra o astroide \omterm{ex:b1:curves:astroid}{cúspides}).
\emph{Tangente vertical versus \omterm{ex:b1:curves:astroid}{cúspide}}: $ x'(t_0) = 0 \neq y'(t_0)$
é um ponto regular com tangente vertical; apenas $ x' = y' = 0$
exige a expansão de ordem superior.
\emph{Gráfico de raios negativos no raio oposto}: para $ r(\theta) <
0$ o ponto é $-\abs{r}\,\vec u(\theta)$, em ângulo $\theta +
\pi $ (\cref{exo:b1:curves:5}); esquecer isso perde loops internos
ou curvas duplas.
\emph{\omterm{def:b1:curves:arclength}{Comprimento} do arco integra $\norm{f'}$, não $ f'$}: dividir o
integrante nos zeros da velocidade --- para o astroide,
integrando $\frac32\sin 2t $ durante um \omterm{def:b1:vspaces:span}{período} completo sem absoluta
valores dá $0$, não $6$.
\emph{Cônicas deve ser reduzido antes de ser lido}: em $9x^2 +
25y^2 - 36x - 50y - 164 = 0$, nem os eixos, nem o centro,
nem a excentricidade é visível até que os quadrados sejam concluídos
(\cref{exo:b1:curves:6}); e uma parte quadrática desaparecendo em um
variável significa parábola, não ``elipse degenerada''.
\end{remark}

\begin{remark}[Para onde vão essas curvas]
\label{rem:b1:curves:whereused}
Curvas parametrizadas são a linguagem da mecânica: trajetórias
são curvas, os vetores velocidade são vetores tangentes e o arco
\omterm{def:b1:curves:arclength}{comprimento} de \cref{def:b1:curves:arclength} é a distância percorrida. O
cônicas reaparecem onde quer que uma lei do inverso do quadrado atue --- planetária
órbitas são elipses com o Sol em foco. Em
\cref{ch:b1:multivar}, as curvas se tornam o nível conjuntos de funções
de duas variáveis, e a tangente deste capítulo atende ao
gradiente do próximo. O volume do Ano ~ 2 adiciona curvatura e o
forma canônica local de uma curva; o problema do fim de semana abaixo
já extrai, apenas com as ferramentas deste ano, tudo o que
século XVII conhecia sua curva mais célebre.
\end{remark}

\begin{remark}[Perspectivas dentro do Livro 3]
\label{rem:b1:curves:perspectives}
Este capítulo consome toda a primeira metade do volume e
alimenta o último capítulo. \emph{Consumido}: vetores tangentes são
derivados (\cref{ch:b1:derivative}), \omterm{def:b1:curves:arclength}{comprimento} do arco e as áreas são
integrais (\cref{ch:b1:integration}), \omterm{ex:b1:curves:astroid}{cúspides} são resolvidos por
Expansões de Taylor (\cref{ch:b1:taylor}), a tautócrona é uma
\omterm{def:b1:linmaps:def}{linear} equação diferencial (\cref{ch:b1:diffeq}) e cada
distância e ângulo são euclidianos (\cref{ch:b1:euclid}).
\emph{Fed}: em \cref{ch:b1:multivar}, uma curva $ t \mapsto (x(t),
y(t))$ desenhado dentro de um nível \omterm{def:b1:logic:sets}{definir} $\{f = c\}$ é diferenciado por
a regra da cadeia e a identidade resultante $\langle \nabla f,\
f'(t)\rangle = 0$ casa os vetores tangentes deste capítulo com o
gradientes do próximo --- a linha tangente da curva e o
direção normal da superfície são os mesmos cálculos vistos
de ambos os bancos.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b1:curves:1}
Estude e esboce a curva $ x(t) = t^2$, $ y(t) = t^3$ (simetrias,
variações, comportamento no ponto singular $ t = 0$).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:curves:2}
Para a ciclóide $ x(t) = t - \sin t $, $ y(t) = 1 - \cos t $ (o
trajetória de um ponto de uma roda rolante de raio $1$): identificar
a simetria de translação de \omterm{def:b1:vspaces:span}{período}, os pontos singulares e o
direção tangente em $ t = 0$ \emph{(expandir $ x $ e $ y $ para o primeiro
pedidos diferentes de zero)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:curves:3}
Dê a linha tangente à curva $(\cos^3 t, \sin^3 t)$ no $ t =
\frac\pi4$, e verifique se o segmento desta tangente cortado pelo
eixos tem \omterm{def:b1:curves:arclength}{comprimento} $1$ --- uma propriedade famosa do astroide (verdade em
cada ponto regular).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:curves:4}
Esboce o curvas polares $ r = \cos 2\theta $ (rosa de quatro pétalas) e
$ r = \frac{1}{\cos\theta}$ sobre $\intoo{-\frac\pi2}{\frac\pi2}$
(reconhecer uma linha).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:curves:5}
Estude o \omterm{def:b1:curves:polar}{curva polar} $ r = 1 + 2\cos\theta $ (uma limaçativado): domínio
onde $ r \geq 0$ contra $ r < 0$ (pontos plotados com raio negativo
sentar no raio oposto), passagem pela origem, loop interno,
esboço.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:curves:6}
Identifique o \omterm{def:b1:curves:conics}{cônico} $9x^2 + 25y^2 - 36x - 50y - 164 = 0$: reduzir em
completando quadrados, dê centro, semieixos, focos, excentricidade.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:curves:7}
Prove que a equação polar de a \omterm{def:b1:curves:conics}{cônico} com foco na origem
é
\[
r = \frac{p}{1 + e\cos\theta}
\]
(diretriz vertical à distância $\frac pe $ direita do foco).
Quais valores de $\theta $ são permitidos quando $ e > 1$?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:curves:8}
Da propriedade bifocal $ MF + MF' = 2a $, deduza o ``jardineiro
construção'' da elipse, e provar que a tangente em $ M $
faz ângulos iguais com $ MF $ e $ MF'$ \emph{(propriedade de reflexão;
diferenciar $\norm{M(t) - F} + \norm{M(t) - F'} = 2a $ e
interpretar a soma nula dos produtos de vetores unitários)}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:curves:9}
O \emph{lemniscata de Bernoulli} é o \omterm{def:b1:curves:polar}{curva polar} $ r^2 = \cos
2\theta $ (pegar $ r = \sqrt{\cos2\theta}$ onde definido).
\begin{enumerate}
  \item Dê seu domínio, simetrias, tangentes na origem e
        esboce-o.
  \item Prove que é o lugar geométrico dos pontos $ M $ com $ MF \cdot MF'
        = \frac12$ onde $ F, F' = \bigl(\pm\frac{1}{\sqrt2},
        0\bigr)$. \emph{(Calcular $ MF^2\,MF'^2$ em polar
        \omterm{prop:b1:vspaces:coordinates}{coordenadas}.)}
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:curves:10}
Calcule o total \omterm{def:b1:curves:arclength}{comprimento} do cardióide $ r = 1 + \cos\theta $
usando a fórmula polar de \cref{def:b1:curves:arclength}
\emph{(a identidade $1 + \cos\theta = 2\cos^2\frac\theta2$ voltas
a raiz quadrada em $2\abs{\cos\frac\theta2}$)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:curves:11}
Mostre que a tangente à elipse $(a\cos t, b\sin t)$ no
ponto de parâmetro $ t $ tem equação
\[
\frac{x\cos t}{a} + \frac{y\sin t}{b} = 1 ,
\]
e deduza a tangente a $\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1$
em um ponto $(x_0, y_0)$ da elipse: $\frac{x\,x_0}{a^2} +
\frac{y\,y_0}{b^2} = 1$ (a regra da ``\omterm{thm:b1:logic:partition}{divisão} dos quadrados'').
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:curves:12}
O \emph{espiral logarítmica} é o \omterm{def:b1:curves:polar}{curva polar} $ r =
\eu^{k\theta}$ ($ k > 0$ fixo, $\theta \in \R $).
\begin{enumerate}
  \item Mostre que o ângulo $ V $ entre o raio e o
        tangente é \emph{constante} ($\tan V = \frac1k $) --- o
        espiral cruza todos os raios no mesmo ângulo.
  \item Mostre que girando a espiral por um ângulo $ c $ \omterm{def:b1:logic:map}{mapas} isso
        em sua escala pelo fator $\eu^{-kc}$: cada
        a rotação da espiral é uma ampliação dela
        (auto-similaridade).
  \item Calcule o \omterm{def:b1:curves:arclength}{comprimento} do arco $\theta \in
        \intoc{-\infty}{\theta_0}$ (como um \omterm{def:b1:topology:closure}{limite} de \omterm{def:b1:curves:arclength}{comprimentos} sobre
        $\intcc{A}{\theta_0}$, $ A \to -\infty $) e observe que
        é finito: uma curva espiralando infinitamente muitas vezes
        em torno da origem, de finito \omterm{def:b1:curves:arclength}{comprimento}.
\end{enumerate}
\end{exercise}

\section{Problema: a ciclóide, rainha das curvas}

\begin{problem}\label{pb:b1:curves:1}
Uma roda de raio $ R $ rola sem escorregar ao longo do $ x $-eixo;
o ponto do aro inicialmente na origem vestígios o
\emph{ciclóide}. O século XVII lutou por esta curva
--- Galileu pesou recortes de papel, Wren mediu,
Roberval calculou sua área, Huygens construiu relógios nela --- e
cada um de seus resultados está ao alcance deste capítulo. Nós
provar os quatro clássicos: a construção tangente, Wren's \omterm{def:b1:curves:arclength}{comprimento}
$8R $, a área $3\pi R^2$ e a tautócrona de Huygens
propriedade.\index{ciclóide}\index{tautócrona}

\medskip
\noindent\textbf{Part I --- Rolling and the tangent.}
\begin{enumerate}
  \item Depois que a roda girou um ângulo $ t $, seu centro
        senta-se em $\Omega(t) = (Rt, R)$ (rolando sem escorregar:
        distância de contato $=$ arco enrolado). Mostre que o marcado
        ponto está em
        \[
        M(t) = \bigl(R(t - \sin t),\; R(1 - \cos t)\bigr).
        \]
  \item Calcular $ f'(t)$ e mostrar $\norm{f'(t)} =
        2R\,\abs{\sin\frac t2}$; localizar os pontos singulares
        (\omterm{ex:b1:curves:astroid}{cúspides} ---cf.\ \cref{exo:b1:curves:2}) e o topo de
        cada arco.
  \item Deixar $ C(t) = (Rt, 0)$ ser o ponto de contato e $ T(t) =
        (Rt, 2R)$ o topo da roda. Prove isso para $0 < t <
        2\pi $ o vetor $ M - C $ é \emph{normal} para a curva em
        $ M(t)$ e o vetor $ T - M $ é \emph{tangente}: desenhar
        a tangente a uma ciclóide, junte o ponto ao topo de
        seu círculo rolante. \emph{(Fatore tudo através
        $\sin\frac t2$ e $\cos\frac t2$.)}
  \item Interprete a questão 3 cinematicamente: o ponto de contato é
        o \emph{centro de rotação instantâneo}, e o
        velocidade de $ M $ é igual à sua distância até $ C $ (para unidade angular
        velocidade). Verificar $\norm{M - C} = 2R\abs{\sin\frac t2}$.
  \item Prove a relação altura-velocidade
        \[
        \norm{f'(t)}^2 = 2R\,y(t) :
        \]
        em uma ciclóide atravessada com velocidade angular unitária, o
        velocidade em cada ponto é exatamente a velocidade de queda \omterm{def:b1:vspaces:free}{livre} para um
        queda igual à altura atual. (Guarde isso por
        Parte IV.)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Wren's theorem: the arch has \omterm{def:b1:curves:arclength}{comprimento}
$8R $.}
\begin{enumerate}[resume]
  \item Usando \cref{def:b1:curves:arclength}, calcule o \omterm{def:b1:curves:arclength}{comprimento}
        de um arco:
        \[
        L = \int_0^{2\pi} 2R\sin\frac t2\,\dd t = 8R
        \]
        (Teorema de Wren, 1658). Quatro diâmetros de roda e nenhum
        $\pi $ em qualquer lugar.
  \item Calcule o \omterm{def:b1:curves:arclength}{comprimento} do arco do \omterm{ex:b1:curves:astroid}{cúspide}: $ s(t) =
        4R\bigl(1 - \cos\frac t2\bigr)$ e verifique $ s(2\pi) =
        8R $.
  \item Agora meça o arco a partir do \emph{ápice} $ t = \pi $:
        $\sigma(t) = \abs{4R\cos\frac t2}$. Prove o intrínseco
        relação
        \[
        \sigma^2 = 8R\,\bigl(2R - y\bigr) :
        \]
        a distância quadrada do arco do topo é proporcional a
        a altura cai abaixo do topo.
  \item Verificações de sanidade: recupere da questão 8 que o meio arco
        do ápice ao \omterm{ex:b1:curves:astroid}{cúspide} tem \omterm{def:b1:curves:arclength}{comprimento} $4R $, e compare com o
        cálculo astroide de \cref{ex:b1:curves:circlelength}
        --- ambas as curvas têm algébrica, $\pi $-livre \omterm{def:b1:curves:arclength}{comprimentos};
        explicar o que torna isso possível, embora ambos sejam
        construído a partir de círculos. \emph{(Veja a forma de
        $\norm{f'}$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Roberval's area: $3\pi R^2$.}
\begin{enumerate}[resume]
  \item Justifique que a área entre um arco e o solo é
        $ A = \int_0^{2\pi} y(t)\,x'(t)\,\dd t $ \emph{(o
        substituição $ x = x(t)$ em $\int y\,\dd x $,
        \cref{thm:b1:integration:parts}; $ x $ está aumentando)}.
  \item Calcular
        \[
        A = R^2\int_0^{2\pi}(1 - \cos t)^2\,\dd t = 3\pi R^2 :
        \]
        exatamente \emph{três} vezes a área da roda ---
        a proporção que Galileu adivinhou através da pesagem.
  \item O mesmo método para o astroide $(\cos^3t, \sin^3t)$: mostrar
        que a área fechada é $\frac{3\pi}8$
        \emph{(linearizar $\sin^4 t\cos^2 t $; apenas a constante
        prazo sobrevive por um \omterm{def:b1:vspaces:span}{período} completo)}.
  \item Sanidade, verifique a fórmula da questão 10 na unidade superior
        semicírculo $(\cos t, \sin t)$, $ t $ de $\pi $ para $0$:
        isso retorna $\frac\pi2$?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Huygens' tautochrone.} Vire o
arco: um cordão desliza sem atrito, sob a gravidade $ g $, dentro do
tigela cicloidal
\[
x(t) = R(t + \sin t), \qquad y(t) = R(1 - \cos t)
\qquad (t \in \intcc{-\pi}{\pi}),
\]
cujo ponto mais baixo é a origem ($ y $ medido para cima).
\begin{enumerate}[resume]
  \item Calcule a velocidade $\norm{f'(t)} = 2R\cos\frac t2$, o
        \omterm{def:b1:curves:arclength}{comprimento} do arco de baixo $ s(t) = 4R\sin\frac t2$ e
        provar a identidade chave
        \[
        y = \frac{s^2}{8R} .
        \]
  \item O cordão liberado em repouso a partir do ponto do parâmetro
        $ t_0 > 0$ obedece à conservação de energia: se $ s(\tau)$ denota
        sua posição do arco no momento $\tau $, então $\frac12\bigl(
        \frac{\dd s}{\dd\tau}\bigr)^2 + g\,y = g\,y_0$. Reescrever
        isso, usando a questão 14, como
        \[
        \Bigl(\frac{\dd s}{\dd\tau}\Bigr)^{\!2}
        = \frac{g}{4R}\,\bigl(s_0^2 - s^2\bigr),
        \qquad s_0 = s(t_0).
        \]
  \item Diferencie em relação a $\tau $ e obter o
        oscilador harmônico
        \[
        \frac{\dd^2 s}{\dd\tau^2} = -\frac{g}{4R}\,s ;
        \]
        resolva isso com \cref{thm:b1:diffeq:homogeneous2} e o
        condições iniciais: $ s(\tau) = s_0\cos(\omega\tau)$,
        $\omega = \sqrt{g/4R}$.
  \item Deduza o \emph{propriedade tautócrona} (Huygens, 1659):
        a hora de chegar ao fundo,
        \[
        T_{\downarrow} = \frac{\pi}{2\omega}
        = \pi\sqrt{\frac Rg}\,,
        \]
        não depende do ponto de liberação --- contas liberadas
        juntos de quaisquer duas alturas da tigela chegam
        juntos.
  \item Verifique por substituição que $ s(\tau) =
        s_0\cos\omega\tau $ satisfaz a energia de primeira ordem
        equação da questão 15 exatamente (não apenas a
        diferenciado) e explique em uma frase por que um
        pêndulo circular é apenas \emph{aproximadamente}
        isócrona enquanto a ciclóide é exatamente assim.
  \item Numericamente: qual raio $ R $ faz a hora da descida
        exatamente um segundo ($ g = 9.81$)? Observe quão perto o
        a resposta é um metro, e como a oscilação completa
        \omterm{def:b1:vspaces:span}{período} $2\pi\sqrt{4R/g}$ compara com o pequeno ângulo
        fórmula do pêndulo $2\pi\sqrt{\ell/g}$ para $\ell = 4R $.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends, and synthesis.}
\begin{enumerate}[resume]
  \item Aplique a regra da tangente da questão 3 em $ t =
        \frac\pi2$ (pegar $ R = 1$): calcular $ M $, $ T $, o
        direção de $ MT $, e compará-lo
        $ f'(\frac\pi2)$.
  \item (Trochoids) Marque em vez disso um ponto à distância $ d $ de
        o centro ($ d \neq R $): a curva é $ x = Rt -
        d\sin t $, $ y = R - d\cos t $. Mostre isso para $ d < R $ o
        curva é regular em todos os lugares e é o gráfico de não
        comportamento singular ($ x' > 0$: avança), enquanto para $ d
        > R $ a abscissa $ x'$ muda de sinal e a curva faz
        loops --- a roda ferroviária flangeada cujo aro aponta
        viajar \emph{para trás}.
  \item (teaser da Braquistócrona) Do \omterm{ex:b1:curves:astroid}{cúspide} $(\pi R, 2R)$ de
        a tigela até o fundo, compare o tempo de descida da ciclóide
        $\pi\sqrt{R/g}$ com o tempo ao longo da rampa reta
        juntando os mesmos pontos \emph{(aceleração constante
        $ g\sin\alpha $ ao longo do acorde)}: mostra a duração do acorde
        $\sqrt{\pi^2 + 4}\,\sqrt{R/g} \approx 3.72\sqrt{R/g}$.
        A curva supera a linha --- é, na verdade, a mais rápida
        de todas as curvas, resultado do cálculo das variações.
  \item Mostre isso no arco ($0 < t < 2\pi $),
        \[
        \frac{\dd y}{\dd x} = \cot\frac t2,
        \qquad
        \frac{\dd^2 y}{\dd x^2} =
        -\frac{1}{4R\sin^4\frac t2} < 0 :
        \]
        o arco é côncavo, com tangentes verticais exatamente
        o \omterm{ex:b1:curves:astroid}{cúspides}.
  \item Recuperar a vertical \omterm{ex:b1:curves:astroid}{cúspide} tangente de
        \cref{exo:b1:curves:2} geometricamente: calcule o \omterm{def:b1:topology:closure}{limite}
        direção do acorde $ MT $ da questão 3 como $ t \to
        0^{+}$, sem nenhuma expansão.
  \item Síntese, em quatro frases: quais são os três teoremas nomeados
        este problema provou (com seus números $8R $, $3\pi
        R^2$, $\pi\sqrt{R/g}$ e seus autores); qual solteiro
        dispositivo computacional ($\sin\frac t2$, $\cos\frac t2$
        fatorações) alimentaram todas as Partes I, II e IV; como
        a relação intrínseca $ y = s^2/8R$ geometria convertida
        em um \omterm{def:b1:linmaps:def}{linear} equação diferencial; e qual capítulo
        deste livro cada parte se apoiou.
\end{enumerate}
\end{problem}
