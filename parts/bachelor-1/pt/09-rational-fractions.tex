\chapter{Frações Racionais}\label{ch:b1:fractions}

UM \omterm{def:b1:fractions:field}{fração racional} é um quociente de \omterm{def:b1:poly:def}{polinômios}. O teorema central
deste breve capítulo --- fração parcial
decomposição\index{decomposição em fração parcial} --- quebra qualquer
tal quociente em uma soma de tijolos elementares $\frac{c}{(X - a)^k}$.
Além do seu interesse \omterm{pb:b1:logic:1}{algébrico}, é a máquina padrão para
integrando funções racionais (\cref{ch:b1:integration}) e para
somando certas séries (\cref{ch:b1:series}).

\section{O campo $ K(X)$}

\begin{definition}\label{def:b1:fractions:field}
UM \emph{fração racional}\index{fração racional} sobre $ K $ ($= \R $
ou $\C $) é um quociente $ F = \frac{A}{B}$ com $ A, B \in K[X]$, $ B \neq
0$; dois quocientes $\frac AB $ e $\frac{A'}{B'}$ são identificados quando
$ AB' = A'B $. Cada fração tem um \emph{forma reduzida} com
$\gcd(A, B) = 1$, exclusivo até constantes. Com as operações naturais,
o \omterm{def:b1:logic:sets}{definir} $ K(X)$ de frações racionais é um \omterm{def:b1:structures:field}{campo}.

O \emph{postagens}\index{postagens} de $ F $ (em forma reduzida) são as raízes de
$ B $; o \emph{ordem} de um poste é o seu \omterm{def:b1:poly:derivative}{multiplicidade} como raiz de $ B $.
O \emph{grau} de $ F $ é $\deg F = \deg A - \deg B \in \Z \cup
\{-\infty\}$.
\end{definition}

\begin{example}[Lendo postes, ordens e graus]\label{ex:b1:fractions:vocab}
Deixar $ F = \dfrac{X^3 - X}{X^4 - 2X^3 + X^2}$. Fatore ambas as camadas:
numerador $ X(X-1)(X+1)$, denominador $ X^2(X-1)^2$; cancelar o
fator comum $ X(X-1)$:
\[
F = \frac{X + 1}{X(X - 1)} \quad\text{(reduced form).}
\]
Poloneses: $0$ e $1$, ambos \emph{simples} --- os pedidos são lidos
o denominador reduzido, então as raízes duplas aparentes do
original são irrelevantes. Grau: $\deg F = 1 - 2 = -1$,
visível assintoticamente ($ xF(x) \to 1$ como $ x \to \infty $). O
grau se comporta como o polinomial grau ($\deg FG = \deg F +
\deg G $, $\deg(F + G) \leq \max $), uma regra de escrituração usada
constantemente no coeficiente caça abaixo: cada ``limite de
$ xF(x)$''O argumento é uma contagem de diplomas disfarçada.
\end{example}

\begin{proposition}[Parte inteira]\label{prop:b1:fractions:integerpart}
Todo $ F = \frac AB \in K(X)$ é exclusivamente $ F = E + \frac RB $ com $ E
\in K[X]$ (o \emph{parte inteira}, ou polinomial parte, de $ F $) e
$\deg R < \deg B $. Um tem $ E \neq 0$ se $\deg F \geq 0$.
\end{proposition}

\begin{proof}
\omterm{thm:b1:logic:partition}{Divisão} euclidiana $ A = BE + R $ (\cref{thm:b1:poly:division}), dividido
por $ B $. Singularidade: se $ E + \frac RB = E' + \frac{R'}{B}$, então $(E -
E')B = R' - R $ com $\deg(R' - R) < \deg B $, forçando $ E = E'$ então $ R
= R'$.
\end{proof}

\begin{example}[Reduza primeiro e depois divida]\label{ex:b1:fractions:integerpart}
Encontre a parte inteira de $ F = \dfrac{X^3 + 1}{X^2 - 1}$. Dividindo
cegamente: $ X^3 + 1 = (X^2 - 1)X + (X + 1)$, então $ F = X +
\frac{X + 1}{X^2 - 1}$. Mas a fração não foi reduzida: $ X^3 + 1
= (X + 1)(X^2 - X + 1)$ e $ X^2 - 1 = (X+1)(X-1)$ compartilhe o fator
$ X + 1$, e
\[
F = \frac{X^2 - X + 1}{X - 1} = X + \frac{1}{X - 1} :
\]
a mesma parte inteira $ X $, mas a parte fracionária se reduz a um
tijolo único, e o ``pólo'' no $-1$ nunca foi um pólo de forma alguma.
Sempre reduza para os termos mais baixos antes de caçar postes: o postes de
$ F $ são as raízes do \emph{reduzido} denominador. (O número inteiro
parte é insensível à simplificação, pois a singularidade em
\cref{prop:b1:fractions:integerpart} garantias.)
\end{example}

\section{Decomposição de frações parciais sobre $\C $}

\begin{theorem}[Decomposição terminada $\C $]\label{thm:b1:fractions:complex}
Deixar $ F = \frac AB \in \C(X)$ estar em forma reduzida, com $ B = c\,(X -
a_1)^{m_1} \cdots (X - a_r)^{m_r}$. Então $ F $ é, de uma forma única,
\[
F = E + \sum_{i=1}^{r} \sum_{k=1}^{m_i}
\frac{c_{i,k}}{(X - a_i)^k},
\qquad E \in \C[X],\ c_{i,k} \in \C ,
\]
$ E $ sendo a parte inteira de $ F $.
\end{theorem}

\begin{proof}
Por \cref{prop:b1:fractions:integerpart} podemos assumir $\deg A < \deg
B $ e prove a decomposição da soma com $ E = 0$.

\emph{Dividindo o postes.} Escrever $ B = (X - a_1)^{m_1} B_1$ com
$ B_1(a_1) \neq 0$. O \omterm{def:b1:poly:def}{polinômios} $(X-a_1)^{m_1}$ e $ B_1$ são
\omterm{cor:b1:arith:bezout}{coprime} (sem raiz comum), então por Bézout em $\C[X]$ (veja a observação em
\cref{ch:b1:poly}) há $ U_0, V_0$ com $ U_0 (X-a_1)^{m_1} +
V_0 B_1 = 1$; multiplicando por $ A $ e configuração $ U = AU_0$, $ V =
AV_0$, então dividindo por $ B $:
\[
\frac AB = \frac{V}{(X - a_1)^{m_1}} + \frac{U}{B_1} .
\]
As condições de graduação podem ser aplicadas: dividir $ V $ por
$(X-a_1)^{m_1}$, dizer $ V = (X-a_1)^{m_1}Q + V_1$ com $\deg V_1 <
m_1$ e absorva o quociente no segundo termo ($ U_1 = U +
QB_1$):
\[
\frac AB = \frac{V_1}{(X - a_1)^{m_1}} + \frac{U_1}{B_1},
\qquad \deg V_1 < m_1 ;
\]
comparando graus ($\deg A < \deg B $ e $\deg V_1 < m_1$) forças
$\deg U_1 < \deg B_1$ também. Iterando $\frac{U_1}{B_1}$,
pólo depois pólo, reduz tudo ao unipolar caso abaixo.

\emph{Um pólo.} Para $\frac{A}{(X-a)^m}$ com $\deg A < m $: expandir $ A $
em poderes de $(X - a)$, $ A = \sum_{j=0}^{m-1} \alpha_j (X - a)^j $
(Expansão de Taylor de um polinomial, como na prova de
\cref{prop:b1:poly:multiplicity}); dividindo dá exatamente os tijolos
$\frac{\alpha_j}{(X-a)^{m-j}}$. Concretamente, para $\frac{X^2 +
1}{(X+2)^3}$: substituindo $ X = Y - 2$,
\[
X^2 + 1 = (Y - 2)^2 + 1 = Y^2 - 4Y + 5 ,
\qquad\text{so}\qquad
\frac{X^2+1}{(X+2)^3} = \frac1{Y} - \frac4{Y^2} + \frac5{Y^3}
\]
com $ Y = X + 2$: os três tijolos aparecem por mera \omterm{thm:b1:logic:partition}{divisão} do
expansão deslocada --- a rota mais rápida sempre que um único
de alta ordem pólo está envolvido, e o recomendado para
\cref{exo:b1:fractions:3}.

\emph{Singularidade.} Suponha que duas decomposições coincidam; seu
diferença é uma identidade $0 = \sum_{i,k} \frac{d_{i,k}}{(X -
a_i)^k}$. Multiplique por $(X - a_1)^{m_1}$: cada termo
adquire um fator que desaparece em $ a_1$ \emph{exceto} aquele com
$ i = 1$, $ k = m_1$, cujo coeficiente se torna $ d_{1,m_1}$ mais
termos que carregam pelo menos um fator $(X - a_1)$. Avaliando em
$ a_1$ (legítimo: após a multiplicação, não pólo permanece em
$ a_1$) dá $ d_{1,m_1} = 0$. Sem o coeficiente superior,
repita com $(X - a_1)^{m_1 - 1}$, e assim por diante para baixo até $ k = 1$;
então passe para o próximo pólo. Todos $ d_{i,k}$ desaparecer: o
decomposição é única.
\end{proof}

\begin{method}[Calculando os coeficientes]\label{met:b1:fractions:compute}
Na prática, evite Bézout; combinar:
\begin{enumerate}
  \item \emph{encobrimento para o poder mais alto}: o coeficiente de
        $\frac{1}{(X-a)^m}$ (com $ m $ a ordem do pólo $ a $) é
        \[
        c_{a,m} = \Bigl[\,(X - a)^m F\,\Bigr]_{X = a};
        \]
        por um \emph{simples} pólo de $ F = \frac AB $, isso é
        $\frac{A(a)}{B'(a)}$;
  \item \emph{avaliações} em pontos convenientes e \emph{limites} de
        $ xF(x)$ como $ x \to \infty $ para coletar relações lineares para o
        coeficientes restantes;
  \item \emph{paridade} ou simetrias de conjugação, quando presentes, para cortar
        o trabalho pela metade.
\end{enumerate}
\end{method}

\begin{remark}[Armadilhas comuns com frações parciais]\label{rem:b1:fractions:pitfalls}
\begin{enumerate}
  \item \emph{Ignorando a parte inteira.} A decomposição do tijolo
        aplica-se a frações de grau $< 0$; quando $\deg A \geq
        \deg B $, divida primeiro
        (\cref{prop:b1:fractions:integerpart}), ou o
        a caça aos coeficientes produzirá contradições.
  \item \emph{Encobrimento além de seu escopo.} Multiplicando por
        $(X - a)^k $ e avaliando em $ a $ produz o coeficiente
        apenas para $ k = m $, o \emph{completo} ordem do pólo; o
        coeficientes de ordem inferior requerem outras relações (limites,
        avaliações) --- veja \cref{ex:b1:fractions:multiple}.
  \item \emph{Esquecendo de reduzir.} Poloneses são lidos no
        \emph{reduzido} forma; um fator comum entre o numerador
        e denominador cria fantasma postes
        (\cref{ex:b1:fractions:integerpart}).
  \item \emph{Formas de tijolo erradas $\R $.} Acima de um irredutível
        quadrático os numeradores são \emph{afim} ($\alpha X +
        \beta $), não constante; escrita $\frac{c}{X^2 + 1}$ sozinho
        perde soluções --- as formas corretas são ditadas por
        \cref{thm:b1:fractions:real}, nunca improvisado.
\end{enumerate}
\end{remark}

\begin{proof}[Prova da fórmula de pólo simples]
Perto de um simples pólo $ a $: $ B = (X - a) Q $ com $ Q(a) \neq 0$, e $ B' =
Q + (X - a) Q'$, então $ B'(a) = Q(a)$. O valor de cobertura é
$\frac{A(a)}{Q(a)} = \frac{A(a)}{B'(a)}$.
\end{proof}

\begin{example}\label{ex:b1:fractions:simple}
Decompor $ F = \dfrac{1}{X(X-1)(X-2)}$. Três simples postes; encobrimento
em cada um:
\[
c_0 = \frac{1}{(0-1)(0-2)} = \frac12,
\quad
c_1 = \frac{1}{1 \times (1 - 2)} = -1,
\quad
c_2 = \frac{1}{2 \times 1} = \frac12,
\]
então $ F = \dfrac{1/2}{X} - \dfrac{1}{X-1} + \dfrac{1/2}{X-2}$.
\emph{Verifique em $ X = 3$:} diretamente, $ F(3) = \frac{1}{3\cdot2\cdot1} =
\frac16$; da decomposição, $\frac{1/2}{3} - \frac{1}{2} +
\frac{1/2}{1} = \frac16 - \frac12 + \frac12 = \frac16$.
\end{example}

\begin{example}[Pólo múltiplo]\label{ex:b1:fractions:multiple}
Decompor $ F = \dfrac{X}{(X-1)^2 (X+1)}$. Forma:
$\frac{a}{(X-1)^2} + \frac{b}{X-1} + \frac{c}{X+1}$.
\begin{itemize}
  \item Encobrimento em dobro pólo: $ a = \bigl[\frac{X}{X+1}\bigr]_{X=1}
        = \frac12$.
  \item Encobrimento em $-1$: $ c = \bigl[\frac{X}{(X-1)^2}\bigr]_{X=-1} =
        -\frac14$.
  \item Limite de $ xF(x)$ no $\infty $: $0 = b + c $, então $ b = \frac14$.
\end{itemize}
\[
F = \frac{1/2}{(X-1)^2} + \frac{1/4}{X-1} - \frac{1/4}{X+1} .
\]
\emph{Verifique em $ X = 0$:} $ F(0) = 0$ e $\frac12 - \frac14 - \frac14
= 0$.
\end{example}

\section{Decomposição terminada $\R $}

\begin{theorem}[Decomposição terminada $\R $]\label{thm:b1:fractions:real}
Deixar $ F \in \R(X)$ estar na forma reduzida com denominador
\[
B = c \prod_i (X - a_i)^{m_i} \prod_j (X^2 + p_j X + q_j)^{n_j}
\qquad (p_j^2 - 4q_j < 0).
\]
Então $ F $ decompõe-se exclusivamente como sua parte inteira mais termos
\[
\frac{c_{i,k}}{(X - a_i)^k} \quad (1 \leq k \leq m_i),
\qquad
\frac{\alpha_{j,l}\, X + \beta_{j,l}}{(X^2 + p_j X + q_j)^{l}}
\quad (1 \leq l \leq n_j),
\]
com coeficientes reais.
\end{theorem}

\begin{proof}
Decompor $\C $ (\cref{thm:b1:fractions:complex}). Desde $ F $ é
real, o coeficiente sobre o pólo $\conj a $ (em cada pedido) é o
\omterm{def:b1:complex:field}{conjugado} do coeficiente sobre $ a $ (aplicar conjugação ao
decomposição e invocar exclusividade). Agrupe cada par \omterm{def:b1:complex:field}{conjugado}:
\[
\frac{c}{(X - z)^l} + \frac{\conj c}{(X - \conj z)^l}
= \frac{c\,(X - \conj z)^l + \conj c\,(X - z)^l}{\bigl(X^2 - 2\Re(z)X
+ \abs z^2\bigr)^{l}},
\]
cujo numerador é seu próprio \omterm{def:b1:complex:field}{conjugado}, portanto real, de grau $\leq l $;
dividir múltiplos da quadrática real reduz-la ao grau
$\leq 1$ em cada nível $ l $ (uma pequena indução descendente). Real postes
mantêm seus coeficientes reais (a conjugação os corrige). Singularidade
segue da singularidade sobre $\C $.
\end{proof}

\begin{example}[Observando o emparelhamento conjugado]\label{ex:b1:fractions:pairing}
O mecanismo da prova, no menor caso: mais $\C $, o
postes de $\frac1{X^2+1}$ são $\pm\iu $, com coeficientes de encobrimento
$\frac1{2\iu}$ no $\iu $ e $\frac1{-2\iu}$ no $-\iu $ ---
\omterm{def:b1:complex:field}{conjugados} um do outro, como prevê o teorema:
\[
\frac{1}{X^2 + 1}
= \frac{1/(2\iu)}{X - \iu} - \frac{1/(2\iu)}{X + \iu} .
\]
Recombinando sobre o denominador comum:
\[
\frac{1}{2\iu}\cdot\frac{(X + \iu) - (X - \iu)}{X^2 + 1}
= \frac{1}{2\iu}\cdot\frac{2\iu}{X^2 + 1} = \frac{1}{X^2+1} :
\]
as partes imaginárias se cancelam e o tijolo real reaparece intacto.
Para integrandos reais geralmente nunca se sai $\R $ --- mas quando
avaliando \emph{somas} em pontos complexos (como o problema do fim de semana
faz com raízes da unidade), os tijolos complexos são os naturais
moeda, e este emparelhamento é a taxa de câmbio entre os dois
decomposições.
\end{example}

\begin{example}\label{ex:b1:fractions:real}
Decompor $ F = \dfrac{4}{(X^2+1)(X-1)^2}$ sobre $\R $. Forma:
$\frac{aX + b}{X^2 + 1} + \frac{c}{(X-1)^2} + \frac{d}{X - 1}$.
Encobrimento em dobro pólo: $ c = \bigl[\frac{4}{X^2+1}\bigr]_{X=1} =
2$. Encobrimento no complexo pólo $\iu $ (numerador acima $ X^2 + 1$
avaliado através da decomposição complexa, ou diretamente): multiplique por
$ X^2 + 1$ e \omterm{def:b1:logic:sets}{definir} $ X = \iu $:
\[
a\iu + b = \frac{4}{(\iu - 1)^2} = \frac{4}{-2\iu} = 2\iu ,
\]
então $ a = 2$, $ b = 0$. Limite de $ xF(x)$ no infinito: $0 = a + d $, então $ d
= -2$. Por isso
\[
F = \frac{2X}{X^2+1} + \frac{2}{(X-1)^2} - \frac{2}{X-1} .
\]
\emph{Verifique em $ X = 0$:} $ F(0) = 4$ e $0 + 2 + 2 = 4$.
\end{example}

\begin{example}[Um denominador cúbico, do início ao fim]\label{ex:b1:fractions:cubic}
Decompor $ F = \dfrac{1}{X^3 + 1}$ sobre $\R $. Fatore primeiro: $ X^3 +
1 = (X + 1)(X^2 - X + 1)$, o quadrático tendo discriminante $-3 <
0$. Forma: $\frac{a}{X+1} + \frac{bX + c}{X^2 - X + 1}$. Encobrimento
no simples pólo $-1$: $ a = \bigl[\frac1{X^2 - X +
1}\bigr]_{X=-1} = \frac13$. Limite de $ xF(x)$ no infinito: $0 = a +
b $, então $ b = -\frac13$. Avaliação em $ X = 0$: $1 = a + c $, então $ c =
\frac23$. Por isso
\[
\frac{1}{X^3+1} = \frac13\Bigl(\frac{1}{X+1}
+ \frac{-X + 2}{X^2 - X + 1}\Bigr) ,
\]
confirmado em $ X = 1$: lado esquerdo $\frac12$, lado direito
$\frac13\bigl(\frac12 + 1\bigr) = \frac12$. Observe a economia:
três incógnitas, três fatos lineares baratos (um encobrimento, um
limite, uma avaliação), sem expansão de nada --- o fluxo de trabalho
de \cref{met:b1:fractions:compute} em sua forma pura.
\end{example}

\begin{remark}[Para que serve]
Uma vez decomposta, uma função racional integra termo por termo: tijolos
$\frac{1}{(x-a)^k}$ têm primitivos elementares e tijolos
$\frac{\alpha x + \beta}{(x^2 + px + q)^l}$ reduzir para $\ln $ e
$\arctan $ (\cref{ch:b1:integration}). Somas telescópicas são os outros
aplicação padrão (\cref{exo:b1:fractions:8}).
\end{remark}

\begin{remark}[Onde este capítulo é usado]\label{rem:b1:fractions:whereused}
As frações parciais são acima de tudo uma \emph{pré-processamento} passo: o
capítulo de integração (\cref{ch:b1:integration}) alimenta todo racional
integrando através \cref{thm:b1:fractions:real} antes de integrar,
e o capítulo da série (\cref{ch:b1:series}) telescópios racionais
termos exatamente como em \cref{exo:b1:fractions:8} e no fim de semana
problema abaixo --- o que leva a técnica até o fim
$\sum 1/k^2 = \pi^2/6$. A derivada logarítmica $ P'/P = \sum
m_i/(X - a_i)$ (\cref{exo:b1:poly:11}) reaparece sempre que root
locais são estudados. Além deste volume, a decomposição de
$1/\chi(X)$ para uma característica polinomial $\chi $ subjacente ao
cálculo de potências matriciais e de transformadas de Laplace no ano
2º volume: os tijolos $\frac{c}{(X - a)^k}$ são a sombra algébrica
das soluções $ t^{k-1}\eu^{at}$ conheci em \cref{ch:b1:diffeq}.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b1:fractions:1}
Decompor $\R $: $\dfrac{1}{X^2 - 1}$;
$\;\dfrac{X}{X^2 - 3X + 2}$; $\;\dfrac{X^2 + 1}{X(X-1)}$ \emph{(mente
a parte inteira)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:fractions:2}
Decompor $\R $: $\dfrac{1}{X(X^2 + 1)}$ e
$\dfrac{X^3}{X^2 + X + 1}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:fractions:3}
Decompor $\dfrac{1}{X^2(X - 1)}$ e $\dfrac{X + 1}{(X - 1)^3}$
\emph{(para o segundo, substitua $ Y = X - 1$)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:fractions:4}
Decompor $\C $, então acabou $\R $: $\dfrac{1}{X^4 - 1}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:fractions:5}
Decompor $\R $: $\dfrac{X^2}{(X^2 + 1)^2}$, e deduzir um
primitivo de $ x \mapsto \dfrac{x^2}{(x^2+1)^2}$ dado
$\int \frac{\dd x}{(x^2+1)^2} = \frac12\bigl(\arctan x +
\frac{x}{x^2+1}\bigr) + C $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:fractions:6}
Para $ n \in \N^*$, decompor $ F_n = \dfrac{n!}{X(X+1)\cdots(X+n)}$
\emph{(simples postes no $0, -1, \dots, -n $; use a fórmula de encobrimento
e reconhecer coeficientes binomiais)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:fractions:7}
Usando a identidade de \cref{exo:b1:poly:11} para $ P = X^n - 1$,
provar isso
\[
\sum_{k=0}^{n-1} \frac{1}{X - \omega^k} = \frac{n X^{n-1}}{X^n - 1},
\qquad \omega = \eu^{2\iu\pi/n},
\]
e avaliar ambos os lados em $ X = 2$ para $ n = 4$ como um cheque.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:fractions:8}
Decompor $\dfrac{1}{k(k+1)(k+2)}$ e calcular
\[
S_n = \sum_{k=1}^{n} \frac{1}{k(k+1)(k+2)},
\qquad\text{then}\qquad
\lim_{n \to \infty} S_n .
\]
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:fractions:9}
Deixar $ P \in \R[X]$ ser \omterm{def:b1:poly:def}{Mônica} de grau $ n $ com $ n $ raízes reais distintas
$ x_1 < \dots < x_n $. Prove isso
\[
\sum_{i=1}^{n} \frac{1}{P'(x_i)} = 0 \quad (n \geq 2),
\qquad
\sum_{i=1}^{n} \frac{x_i^{\,n-1}}{P'(x_i)} = 1 .
\]
\emph{Dica: decompor $\frac{X^m}{P}$ para $ m \leq n - 1$ e olhe para
o coeficiente decai no infinito --- ou use \omterm{thm:b1:poly:lagrange}{Interpolação de Lagrange}
(\cref{thm:b1:poly:lagrange}) do monômio $ X^m $ nos nós
$ x_i $.}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:fractions:10}
Decompor $\dfrac{1}{X(X+1)^2}$ sobre $\R $. Admitindo o valor
$\sum_{k \geq 1} \frac1{k^2} = \frac{\pi^2}6$ (provado neste
problema de fim de semana do capítulo), deduza
\[
\sum_{n=1}^{\infty} \frac{1}{n(n+1)^2} = 2 - \frac{\pi^2}{6} .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:fractions:11}
Decompor $\R $: $\dfrac{1}{(X^2+1)(X^2+4)}$, então
$\dfrac{X^2}{(X^2+1)(X^2+4)}$. \emph{Dica: ambos os denominadores são
\omterm{def:b1:poly:def}{polinômios} em $ X^2$: decompor $\frac1{(Y+1)(Y+4)}$ primeiro.}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:fractions:12}
(Equações seculares) Deixe $ F = \sum_{i=1}^{r} \frac{c_i}{X - p_i}$
com $ p_1 < p_2 < \dots < p_r $ real e tudo $ c_i > 0$.
\begin{enumerate}
  \item Mostre isso $ F $ é estritamente decrescente em cada intervalo de
        seu domínio, e dê seus limites em $\pm\infty $ e em ambos
        lados de cada pólo.
  \item Deduza isso para cada $\lambda > 0$, a equação $ F(x) =
        \lambda $ tem exatamente $ r $ soluções reais, uma em cada
        intervalo $\intoo{p_i}{p_{i+1}}$ e um além $ p_r $.
        (Tais equações governam perturbações de autovalores; o
        o teorema do valor intermediário é usado no nível do ensino médio
        aqui e comprovado em \cref{ch:b1:continuity}.)
\end{enumerate}
\end{exercise}

\section{Problema: frações parciais como motor}

\begin{problem}\label{pb:b1:fractions:1}
A decomposição de frações parciais parece uma contabilidade; este problema
mostra que é um motor. Alimentado com a fração
$\frac1{X(X+1)\cdots(X+k)}$, ele telescopia famílias inteiras de somas
de forma fechada; alimentado com $\frac1{X^n - 1}$, produz
identidades trigonométricas como
\[
\sum_{k=1}^{n-1}\frac{1}{\sin^2\frac{k\pi}{n}} = \frac{n^2-1}{3} ;
\]
e, dando um passo adiante, essa identidade espreme um dos
fórmulas mais célebres da matemática, a de Euler
\[
\sum_{k=1}^{\infty}\frac1{k^2} = \frac{\pi^2}{6}
\]
--- aqui obtido com nada além da álgebra deste capítulo e
trigonometria do ensino médio.\index{telescópico}\index{Problema de Basileia}
Por todo, $\omega = \eu^{2\iu\pi/n}$; limites de sequências são
usado no ensino médio (\cref{ch:b1:seq} os formaliza).

\medskip
\noindent\textbf{Part I --- The telescope.}
\begin{enumerate}
  \item Decompor $\frac1{X(X+1)}$ e calcular $\sum_{n=1}^{N}
        \frac1{n(n+1)}$ exatamente; concluir que a soma tende a
        $1$.
  \item O mesmo para $\frac1{X(X+2)}$: mostrar $\sum_{n=1}^{N}
        \frac1{n(n+2)} = \frac12\bigl(\frac32 - \frac1{N+1} -
        \frac1{N+2}\bigr) \to \frac34$. (Com uma lacuna, \emph{dois}
        os termos de fronteira sobrevivem em cada extremidade.)
  \item Formalize o mecanismo: se $ F(X) = G(X) - G(X+1)$ para
        algum racional $ G $ sem postes em $\intco1{+\infty}$,
        então $\sum_{n=1}^N F(n) = G(1) - G(N+1)$. Recuperar o
        valor $\frac14$ de \cref{exo:b1:fractions:8} exibindo
        a testemunha $ G $ para $ F = \frac1{X(X+1)(X+2)}$.
  \item Prove o telescópio fatorial geral: para $ k \geq 1$,
        \[
        \frac{1}{X(X+1)\cdots(X+k)}
        = \frac1k\biggl(\frac{1}{X(X+1)\cdots(X+k-1)}
        - \frac{1}{(X+1)\cdots(X+k)}\biggr),
        \]
        e deduzir
        \[
        \sum_{n=1}^{\infty}\frac{1}{n(n+1)\cdots(n+k)}
        = \frac{1}{k \cdot k!} .
        \]
        Verifique o caso $ k = 2$ contra a questão 3.
  \item Avalie a decomposição de \cref{exo:b1:fractions:6}
        em pontos bem escolhidos para provar
        \[
        \sum_{j=0}^{n}(-1)^j\binom nj\,\frac1{j+1}
        = \frac1{n+1},
        \qquad
        \sum_{j=0}^{n}(-1)^j\binom nj\,\frac1{j+2}
        = \frac1{(n+1)(n+2)} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The fraction $1/(X^n - 1)$.}
\begin{enumerate}[resume]
  \item Mostre pela fórmula de encobrimento que
        \[
        \frac{1}{X^n - 1}
        = \frac1n\sum_{k=0}^{n-1}\frac{\omega^k}{X - \omega^k} .
        \]
  \item Duas verificações de sanidade: verifique a fórmula diretamente para $ n =
        2$, e mostre que a soma dos $ n $ coeficientes
        desaparece por $ n \geq 2$ --- explique por que isso deve ser feito (considere
        $ xF(x)$ como $ x \to \infty $).
  \item Re-derivar por encobrimento a identidade de
        \cref{exo:b1:fractions:7}:
        $\frac{nX^{n-1}}{X^n-1} = \sum_k \frac1{X - \omega^k}$.
  \item \omterm{def:b1:complex:field}{Conjugado} de \omterm{def:b1:structures:group}{grupo} postes para provar a decomposição real:
        com $\theta_k = \frac{2k\pi}n $,
        \[
        \frac{\omega^k}{X - \omega^k} +
        \frac{\omega^{n-k}}{X - \omega^{n-k}}
        = \frac{2\cos\theta_k\,X - 2}
        {X^2 - 2\cos\theta_k\,X + 1} ,
        \]
        e anote a decomposição real completa de
        $\frac1{X^n-1}$ (distinguir $ n $ estranho e $ n $ até).
  \item Especialize-se em $ n = 4$ e verificar contra
        \cref{exo:b1:fractions:4}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Trigonometric sums, and Euler's
$\pi^2/6$.} Deixar $ P = 1 + X + \dots + X^{n-1}$, cujas raízes são
$\omega, \omega^2, \dots, \omega^{n-1}$ (tudo simples).
\begin{enumerate}[resume]
  \item Usando $\frac{P'}P = \sum_{k=1}^{n-1}\frac1{X - \omega^k}$
        (\cref{exo:b1:poly:11}), provar
        \[
        \sum_{k=1}^{n-1}\frac{1}{1 - \omega^k} = \frac{n-1}2 .
        \]
  \item Provar $\dfrac1{1 - \eu^{\iu\theta}} = \dfrac12 +
        \dfrac\iu2\,\frac{\cos(\theta/2)}{\sin(\theta/2)}$ para
        $\theta \notin 2\pi\Z $ (fatoração de meio ângulo,
        \cref{met:b1:complex:trig}) e deduza da questão 11
        isso $\sum_{k=1}^{n-1}
        \frac{\cos(k\pi/n)}{\sin(k\pi/n)} = 0$ --- também visível
        pela simetria $ k \leftrightarrow n - k $.
  \item Diferenciando a identidade da questão 11 (ou seja, usando
        $\bigl(\frac{P'}P\bigr)' = \frac{P''}P -
        \bigl(\frac{P'}P\bigr)^2$ avaliado em $ X = 1$), provar
        \[
        \sum_{k=1}^{n-1}\frac{1}{(1 - \omega^k)^2}
        = \frac{(n-1)(5-n)}{12} .
        \]
  \item Escrita $\cot t = \frac{\cos t}{\sin t}$, deduza de
        questões 12--13 as duas formas fechadas
        \[
        \sum_{k=1}^{n-1}\cot^2\frac{k\pi}n = \frac{(n-1)(n-2)}3,
        \qquad
        \sum_{k=1}^{n-1}\frac1{\sin^2\frac{k\pi}n}
        = \frac{n^2-1}3 .
        \]
  \item Verifique ambas as fórmulas manualmente para $ n = 3$ e $ n = 4$.
  \item Prove as desigualdades $\cot t < \frac1t < \frac1{\sin t}$
        para $ t \in \intoo0{\frac\pi2}$ (de $\sin t < t < \tan
        t $) e deduza, por $ n = 2m + 1$ e $1 \leq k \leq m $:
        \[
        \cot^2\frac{k\pi}n \;<\; \frac{n^2}{k^2\pi^2}
        \;<\; \frac1{\sin^2\frac{k\pi}n} .
        \]
  \item Some essas desigualdades para $ k = 1, \dots, m $ (usando o
        simetria $ k \leftrightarrow n - k $ reduzir pela metade as fórmulas
        da questão 14) e aperte:
        \[
        \sum_{k=1}^{\infty}\frac1{k^2} = \frac{\pi^2}6 .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Higher bricks.}
\begin{enumerate}[resume]
  \item Ao elevar ao quadrado a decomposição de $\frac1{X^2-1}$ e
        re-decompondo o termo cruzado, prove
        \[
        \frac1{(X^2-1)^2}
        = \frac14\Bigl(\frac1{(X-1)^2} + \frac1{(X+1)^2}\Bigr)
        - \frac14\Bigl(\frac1{X-1} - \frac1{X+1}\Bigr),
        \]
        e confira em $ X = 0$.
  \item Combine a questão 18, o telescópio e o valor de Euler
        (questão 17) para provar
        \[
        \sum_{n=2}^{\infty}\frac1{(n^2-1)^2}
        = \frac{\pi^2}{12} - \frac{11}{16} ,
        \]
        e confirme o valor numericamente com três casas decimais.
  \item Diferencie a identidade da questão 8 para obter um
        formulário fechado para $\sum_{k=0}^{n-1}\frac1{(X -
        \omega^k)^2}$ e confira em $ X = 2$, $ n = 2$.
  \item Prove isso para $ k \geq 2$,
        \[
        \sum_{n=k}^{\infty}\frac{1}{\binom nk} = \frac{k}{k-1} .
        \]
        \emph{(Reduza para a questão 4 escrevendo $1/\binom nk $ com
        fatoriais.)}
  \item Para $ k = 3$, forneça a soma parcial exata
        $\sum_{n=3}^{N}\frac1{\binom n3}$ e seu limite.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Synthesis.}
\begin{enumerate}[resume]
  \item Como cálculo final, escreva o valor real completo
        decomposição de $\dfrac1{X^6 - 1}$ e confira em $ X =
        0$.
  \item Onde exatamente o problema usou: (i) a singularidade de
        a decomposição; (ii) o raízes da unidade de
        \cref{ch:b1:complex}; (iii) a derivada logarítmica
        de \cref{exo:b1:poly:11}? Uma frase cada.
  \item Síntese, num pequeno parágrafo: uma identidade algébrica
        --- quebrando uma fração em tijolos --- gerado exato
        somas, identidades trigonométricas e $\pi^2/6$. Comente
        a \omterm{thm:b1:logic:partition}{divisão} do trabalho entre álgebra (exato
        decomposições, válidas em todos os lugares) e análise (limites,
        apertando) e aponte para onde cada fio está
        industrializado: telescopia e comparação em
        \cref{ch:b1:series}, integração dos tijolos em
        \cref{ch:b1:integration}.
\end{enumerate}
\end{problem}
