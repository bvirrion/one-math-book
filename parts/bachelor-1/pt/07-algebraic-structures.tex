\section{Leis de composição}

\begin{definition}\label{def:b1:structures:law}
UM \emph{lei da composição}\index{lei da composição} em um \omterm{def:b1:logic:sets}{definir} $E $ é
um \omterm{def:b1:logic:map}{mapa} $ E \times E \to E $, escrito $(x, y) \mapsto x * y $. Isso é
\emph{associativo} quando $(x*y)*z = x*(y*z)$ sempre, \emph{comutativo}
quando $ x * y = y * x $ sempre. Um elemento $ e $ é um \emph{identidade} quando
$ e * x = x * e = x $ para todos $ x $; então $ x'$ é um \emph{inverso} de $ x $
quando $ x * x' = x' * x = e $.
\end{definition}

\begin{proposition}[Singularidade]\label{prop:b1:structures:unique}
Uma lei tem no máximo uma identidade; por uma lei associativa com identidade,
cada elemento tem no máximo um inverso.
\end{proposition}

\begin{proof}
Se $ e $ e $ e'$ são identidades: $ e = e * e' = e'$. Se $ x'$ e $ x''$
invertido $ x $: $ x' = x' * e = x' * (x * x'') = (x' * x) * x'' = e * x''
= x''$.
\end{proof}

\section{Grupos}

\begin{definition}[Grupo]\label{def:b1:structures:group}
UM \emph{grupo}\index{grupo} $(G, *)$ é um \omterm{def:b1:logic:sets}{definir} com uma lei associativa
admitindo uma identidade e em que todo elemento tem um inverso. O
grupo é \emph{abeliano}\index{abeliano} quando a lei é
comutativo.
\end{definition}

\begin{example}\label{ex:b1:structures:groups}
$(\Z, +)$, $(\Q, +)$, $(\R, +)$, $(\C, +)$; $(\Q^*, \times)$,
$(\R^*, \times)$, $(\C^*, \times)$, $(\mathbb{U}_n, \times)$ (raízes de
unidade, \cref{def:b1:complex:unity}); o \omterm{def:b1:logic:sets}{definir} $\mathfrak{S}(E)$ de
bijeções de um \omterm{def:b1:logic:sets}{definir} $ E $ sobre si mesmo, sob composição --- o
\emph{grupo simétrico}\index{grupo simétrico} de $ E $, \omterm{def:b1:structures:group}{não-abeliano} assim que $\abs E \geq 3$. Não \omterm{def:b1:structures:group}{grupos}: $(\N, +)$ (sem inversos),
$(\Z, \times)$ (apenas $\pm 1$ invertível).
\end{example}

\begin{proposition}[Regras de computação]\label{prop:b1:structures:rules}
Em um \omterm{def:b1:structures:group}{grupo} $ G $ (escrito multiplicativamente, identidade $ e $):
\begin{enumerate}
  \item cancelamento: $ ax = ay \implies x = y $ e $ xa = ya \implies
        x = y $;
  \item $(ab)^{-1} = b^{-1} a^{-1}$ e $(a^{-1})^{-1} = a $;
  \item para $ a, b \in G $, cada equação $ ax = b $ e $ xa = b $ tem um
        solução única ($ x = a^{-1}b $, resp.\ $ x = b a^{-1}$).
\end{enumerate}
\end{proposition}

\begin{proof}
(1) Multiplique por $ a^{-1}$ no lado apropriado, usando
associatividade. (2) $(b^{-1}a^{-1})(ab) = b^{-1}(a^{-1}a)b = b^{-1}b =
e $ e simetricamente; conclui-se a singularidade do inverso; o segundo
ponto é \cref{prop:b1:structures:unique} aplicado a $ a^{-1}$. (3)
Substitua e use (1) para exclusividade.
\end{proof}

\begin{example}[As simetrias de um retângulo]\label{ex:b1:structures:klein}
Um retângulo (não quadrado) admite exatamente quatro isometrias em
em si: a identidade $ e $, a reflexão do eixo horizontal $ h $, o
reflexão do eixo vertical $ v $, e a meia volta $ r $ sobre o
centro. A composição torna este quatro elementos \omterm{def:b1:logic:sets}{definir} um \omterm{def:b1:structures:group}{grupo}: cada
elemento é seu próprio inverso ($ h^2 = v^2 = r^2 = e $) e o
produto de quaisquer dois elementos distintos de não identidade é o terceiro
($ hv = vh = r $: refletindo em ambos os eixos é a meia volta). O
tabela completa é simétrica, então o \omterm{def:b1:structures:group}{grupo} é \omterm{def:b1:structures:group}{abeliano} --- ainda é
\emph{não} o mesmo \omterm{def:b1:structures:group}{grupo} como as rotações $\mathbb U_4$ de
\cref{ex:b1:structures:order}: lá, $\iu $ tem \omterm{def:b1:structures:order}{ordem} $4$, enquanto
aqui cada elemento tem \omterm{def:b1:structures:order}{ordem} $\leq 2$. Dois \omterm{def:b1:structures:group}{grupos} do mesmo
o tamanho pode, portanto, ter estruturas de multiplicação genuinamente diferentes
--- a figura abaixo exibe as duas tabelas lado a lado. Isto
quatro elementos \omterm{def:b1:structures:group}{grupo} retorna como $\{\pm1\} \times \{\pm1\}$, e
\cref{exo:b1:structures:7} explica por que qualquer \omterm{def:b1:structures:group}{grupo} com tudo
quadrados triviais devem, como este, ser \omterm{def:b1:structures:group}{abeliano}.
\end{example}

\begin{definition}[Subgrupo]\label{def:b1:structures:subgroup}
Um subconjunto $ H $ de um \omterm{def:b1:structures:group}{grupo} $ G $ é um \emph{subgrupo}\index{subgrupo}
(escrito $ H \leq G $) quando contém $ e $, é estável perante a lei
e sob inversão. Então $ H $ é em si um \omterm{def:b1:structures:group}{grupo}.

\emph{Critério:} um não vazio $ H \subseteq G $ é um subgrupo se e somente
se
\[
\forall x, y \in H, \quad x y^{-1} \in H .
\]
\end{definition}

\begin{proof}[Prova do critério]
Um \omterm{def:b1:structures:subgroup}{subgrupo} obviamente o satisfaz. Por outro lado, deixe $ H \neq \emptyset $
satisfazê-lo e escolher $ x_0 \in H $. Então $ e = x_0 x_0^{-1} \in H $; para
$ y \in H $, $ y^{-1} = e\,y^{-1} \in H $; e para $ x, y \in H $, $ xy =
x (y^{-1})^{-1} \in H $.
\end{proof}

\begin{example}\label{ex:b1:structures:subgroups}
$\mathbb{U}_n \leq (\C^*, \times)$: não vazio, e para $ z, w \in
\mathbb{U}_n $, $(zw^{-1})^n = z^n (w^n)^{-1} = 1$. O \omterm{def:b1:structures:subgroup}{subgrupos} de
$(\Z, +)$ são exatamente os $ n\Z $ (comprovado em
\cref{thm:b1:arith:gcd}). Uma intersecção de \omterm{def:b1:structures:subgroup}{subgrupos} é sempre um
\omterm{def:b1:structures:subgroup}{subgrupo}, mas um sindicato quase nunca é
(\cref{exo:b1:structures:6}).
\end{example}

\begin{remark}[Armadilhas comuns com estruturas]\label{rem:b1:structures:pitfalls}
\begin{enumerate}
  \item \emph{A estabilidade perante a lei não é suficiente.} $\N $ é
        estável sob adição dentro $\Z $, contém $0$, ainda é
        não \omterm{def:b1:structures:subgroup}{subgrupo}: faltam inversos. O critério
        $ xy^{-1} \in H $ testa tudo de uma vez --- mas apenas
        depois de verificar $ H \neq \emptyset $.
  \item \emph{Reflexos \omterm{def:b1:structures:group}{não-abelianos}.} Em um geral \omterm{def:b1:structures:group}{grupo}, $(ab)^2 =
        abab $, que é $ a^2b^2$ somente quando $ a $ e $ b $ trajeto;
        da mesma maneira $(ab)^{-1} = b^{-1}a^{-1}$, \omterm{def:b1:structures:order}{ordem} invertido.
        Toda identidade importada da álgebra escolar deve ser
        re-derivado dos axiomas ou sinalizado como comutativo.
  \item \emph{\omterm{def:b1:structures:morphism}{Núcleo} versus imagem.} $\ker f $ mora no
        \emph{fonte}, $\operatorname{im} f $ no alvo;
        ``$ f $ \omterm{def:b1:logic:inj}{injetivo} se $\ker f $ trivial''
        (\cref{prop:b1:structures:kernel}) não tem análogo com
        a imagem ($\operatorname{im} f = G'$ é sobrejetividade).
  \item \emph{Anéis não são \omterm{def:b1:structures:group}{grupos} para $\times $.} Em um \omterm{def:b1:structures:ring}{anel}, a maioria
        elementos não precisam ser invertíveis, e o cancelamento por $ a $
        requer $ a $ ser uma unidade ou o \omterm{def:b1:structures:ring}{anel} ser um integrante
        domínio: em $\Z/12\Z $, $\overline3\,\overline2 =
        \overline3\,\overline6$ ainda $\overline2 \neq \overline6$
        (\cref{ex:b1:structures:zncomputation}).
\end{enumerate}
\end{remark}

\begin{definition}[Morfismo de grupo]\label{def:b1:structures:morphism}
Deixar $(G, *)$ e $(G', \star)$ ser \omterm{def:b1:structures:group}{grupos}. UM \omterm{def:b1:logic:map}{mapa} $ f \colon G \to G'$
é um \emph{morfismo}\index{morfismo} quando
\[
\forall x, y \in G, \qquad f(x * y) = f(x) \star f(y).
\]
Então $ f(e_G) = e_{G'}$ e $ f(x^{-1}) = f(x)^{-1}$. O
\emph{núcleo}\index{núcleo} e \emph{imagem} de $ f $ são
\[
\ker f = f^{-1}(\{e_{G'}\}) \leq G,
\qquad
\operatorname{im} f = f(G) \leq G' .
\]
UM \omterm{def:b1:logic:inj}{bijetivo} morfismo é um \emph{isomorfismo}; é inverso \omterm{def:b1:logic:map}{mapa} é então
automaticamente um morfismo.
\end{definition}

\begin{proof}[Prova das afirmações]
$ f(e) = f(e * e) = f(e)\star f(e)$ e cancelando $ f(e)$ dá
$ e_{G'} = f(e)$. Então $ f(x)\star f(x^{-1}) = f(x x^{-1}) = e_{G'}$
identifica $ f(x^{-1})$ como o inverso. \omterm{def:b1:structures:morphism}{Núcleo}: $ e \in \ker f $; se
$ x, y \in \ker f $, $ f(xy^{-1}) = f(x)f(y)^{-1} = e $; o critério
se aplica. Imagem: mesmo critério com $ f(x)f(y)^{-1} = f(xy^{-1})$.
Inverso de um isomorfismo: para $ u, v \in G'$, escrever $ u = f(x)$, $ v =
f(y)$; então $ f^{-1}(u \star v) = f^{-1}(f(xy)) = xy =
f^{-1}(u) f^{-1}(v)$.
\end{proof}

\begin{proposition}[Injetividade via kernel]\label{prop:b1:structures:kernel}
UM \omterm{def:b1:structures:group}{grupo} \omterm{def:b1:structures:morphism}{morfismo} $ f $ é \omterm{def:b1:logic:inj}{injetivo} se e somente se $\ker f =
\{e\}$.
\end{proposition}

\begin{proof}
Se $ f $ é \omterm{def:b1:logic:inj}{injetivo}, $\ker f $ só pode conter aquele \omterm{def:b1:logic:map}{pré-imagem} de
$ e_{G'}$, que é $ e $. Por outro lado, se $\ker f = \{e\}$ e $ f(x) =
f(y)$, então $ f(xy^{-1}) = f(x) f(y)^{-1} = e_{G'}$, então $ xy^{-1} = e $,
ou seja\ $ x = y $.
\end{proof}

\begin{example}\label{ex:b1:structures:expmorphism}
$\exp \colon (\R, +) \to (\R_+^*, \times)$ é um \omterm{def:b1:structures:morphism}{morfismo}
($\eu^{x+y} = \eu^x \eu^y $), \omterm{def:b1:logic:inj}{bijetivo}
(\cref{prop:b1:functions:expln}): o aditivo e o multiplicativo
estruturas são isomórficas --- a razão de ser histórica de
logaritmos. Outro \omterm{def:b1:structures:morphism}{morfismo}: $\theta \mapsto \eu^{\iu\theta}$ de
$(\R, +)$ no círculo unitário $(\mathbb{U}, \times)$, com \omterm{def:b1:structures:morphism}{núcleo}
$2\pi\Z $.
\end{example}

\begin{example}[O morfismo de sinal]\label{ex:b1:structures:signmorphism}
O \omterm{def:b1:logic:map}{mapa} $ s \colon (\R^*, \times) \to (\{\pm1\}, \times)$ enviando
$ x $ ao seu signo é um \omterm{def:b1:structures:morphism}{morfismo}: o sinal de um produto é o
produto dos signos. Isso é \omterm{def:b1:structures:morphism}{núcleo} é $\intoo0{+\infty}$ (um
\omterm{def:b1:structures:subgroup}{subgrupo}, como \cref{def:b1:structures:morphism} promessas), seu
imagem de tudo $\{\pm1\}$: \omterm{def:b1:logic:inj}{sobrejetivo}, massivamente não \omterm{def:b1:logic:inj}{injetivo}.
Duas lições gerais em miniatura. Primeiro, um \omterm{def:b1:structures:morphism}{morfismo} pode esmagar
informações: $ s $ não se lembra de nada $ x $ mas um pouco, e isso
é a sua virtude --- argumentos de sinal são exatamente os cálculos
esse fator através $ s $. Segundo, \omterm{def:b1:structures:morphism}{morfismos} para $\{\pm1\}$ são os
``invariantes'' mais simples: a assinatura de permutações, construído em
o problema do fim de semana deste capítulo, é o mesmo fenômeno no
\omterm{def:b1:structures:group}{grupo} $\mathfrak S_n $, e os argumentos de paridade que alimentam todos
descer através de tal dois valores \omterm{def:b1:structures:morphism}{morfismo}.
\end{example}

\begin{definition}[Poderes, ordem de um elemento]\label{def:b1:structures:order}
Em um \omterm{def:b1:structures:group}{grupo} $ G $ (notação multiplicativa), conjunto $ x^0 = e $, $ x^{k+1} =
x^k x $ e $ x^{-k} = (x^k)^{-1}$ para $ k \in \N $; então $ x^{k+l} = x^k
x^l $ para todos $ k, l \in \Z $, então $ k \mapsto x^k $ é um \omterm{def:b1:structures:morphism}{morfismo} $(\Z,
+) \to G $ cuja imagem $\langle x \rangle = \{x^k : k \in \Z\}$ é um
\omterm{def:b1:structures:subgroup}{subgrupo}, o \omterm{def:b1:structures:subgroup}{subgrupo} \emph{gerado} por $ x $. O
\emph{ordem}\index{ordem} de $ x $ é o mínimo $ m \geq
1$ com $ x^m = e $ se existir (então $\langle x\rangle = \{e, x,
\dots, x^{m-1}\}$ tem exatamente $ m $ elementos, e $ x^k = e \iff m
\mid k $), e $\infty $ de outra forma.
\end{definition}

\begin{example}\label{ex:b1:structures:order}
Em $(\C^*, \times)$: $\iu $ tem \omterm{def:b1:structures:order}{ordem} $4$, com $\langle \iu \rangle
= \{1, \iu, -1, -\iu\} = \mathbb{U}_4$; mais geralmente $\omega =
\eu^{2\iu\pi/n}$ tem \omterm{def:b1:structures:order}{ordem} $ n $ e $\langle\omega\rangle =
\mathbb{U}_n $. Em $(\Z, +)$, todo $ x \neq 0$ tem infinito \omterm{def:b1:structures:order}{ordem}.
Por que as afirmações da definição são válidas: se $ x $ tem \omterm{def:b1:structures:order}{ordem} $ m $, dividir
qualquer $ k $ por $ m $ ($ k = mq + r $, $0 \leq r < m $,
\cref{thm:b1:arith:division}): $ x^k = (x^m)^q x^r = x^r $, então o
ciclo de potências com período $ m $, os elementos listados são pares
distinto pela minimalidade de $ m $, e $ x^k = e $ forças $ r = 0$. Pedidos
de permutações são calculados no problema de fim de semana abaixo.
\end{example}

\begin{example}[Pedidos dentro $\mathbb U_{12}$]\label{ex:b1:structures:orderU12}
Qual é o \omterm{def:b1:structures:order}{ordem} de $\omega^k $ em $\mathbb U_n $, para $\omega =
\eu^{2\iu\pi/n}$? Um tem $(\omega^k)^m = 1$ se $ n \mid km $ e
escrevendo $ d = \gcd(n, k)$, $ n = dn'$, $ k = dk'$ com $\gcd(n', k')
= 1$: $ n \mid km \iff n' \mid k'm \iff n' \mid m $ (Gauss
lema, \cref{thm:b1:arith:gauss}). O menos tal $ m \geq 1$ é
$ n' = \frac{n}{\gcd(n,k)}$. Em $\mathbb U_{12}$ por exemplo,
$\omega^8$ tem \omterm{def:b1:structures:order}{ordem} $\frac{12}{\gcd(12,8)} = 3$ (de fato
$\omega^8 = \eu^{4\iu\pi/3} \in \mathbb U_3$), enquanto $\omega^5$
tem \omterm{def:b1:structures:order}{ordem} $12$: gera o todo \omterm{def:b1:structures:group}{grupo}, embora não seja
o gerador ``padrão''. Contando os geradores --- o $ k $
com $\gcd(k, n) = 1$ --- recupera o \omterm{cor:b1:arith:bezout}{coprime} conta de
\cref{ex:b1:counting:coprime}: \omterm{def:b1:structures:group}{grupo} teoria e contagem se encontram.
\end{example}

\section{Anéis e campos}

\begin{definition}[Anel]\label{def:b1:structures:ring}
UM \emph{anel}\index{anel} $(A, +, \times)$ é um \omterm{def:b1:logic:sets}{definir} com duas leis como
isso: $(A, +)$ é um \omterm{def:b1:structures:group}{grupo} \omterm{def:b1:structures:group}{abeliano} (identidade $0$); $\times $ é
associativo com uma identidade $1$; e $\times $ distribui $+$ ligado
ambos os lados. O anel é \emph{comutativo} quando $\times $ é. Um
elemento $ a $ é \emph{invertível} (um \emph{unidade}) quando $ ab = ba = 1$
para alguns $ b $; as unidades formam um \omterm{def:b1:structures:group}{grupo} $(A^\times, \times)$.
\end{definition}

\begin{proof}[Prova de que as unidades formam um grupo]
Estabilidade: se $ a, a'$ são unidades com inversos $ b, b'$, então
\[
(aa')(b'b) = a(a'b')b = a\,1\,b = ab = 1,
\qquad (b'b)(aa') = 1
\]
simetricamente, então $ aa'$ é uma unidade. O elemento $1$ é uma unidade (seu
próprio inverso), a associatividade é herdada de $ A $, e o
inverso $ b $ de uma unidade $ a $ é em si uma unidade (com inverso $ a $).
Então $(A^\times, \times)$ satisfaz a todos \omterm{def:b1:structures:group}{grupo} axiomas. Todo \omterm{def:b1:structures:group}{grupo}
neste livro que não é construído a partir permutações surge desta forma:
$\Q^* = \Q^\times $, $\R^*$, $\C^*$, as unidades de $\Z/n\Z $ abaixo,
e mais tarde as matrizes invertíveis (\cref{ch:b1:matrices}).
\end{proof}

\begin{example}\label{ex:b1:structures:rings}
$\Z, \Q, \R, \C $ são comutativos anéis; $\Z^\times = \{1, -1\}$,
$\Q^\times = \Q^*$. Mais tarde: polinômio anéis $ K[X]$
(\cref{ch:b1:poly}), matriz anéis (não comutativo,
\cref{ch:b1:matrices}), e $\Z/n\Z $ abaixo. Em cada \omterm{def:b1:structures:ring}{anel}, $0 \times
a = 0$ (da distributividade: $0a = (0+0)a = 0a + 0a $), e
$(-1)a = -a $.
\end{example}

\begin{example}[Idempotentes: novos fenômenos em novos anéis]\label{ex:b1:structures:idempotents}
Em $\Z $, a equação $ x^2 = x $, ou seja\ $ x(x - 1) = 0$, tem apenas
as soluções $0$ e $1$. Em $\Z/6\Z $, testando todas as classes:
$\overline0^2 = \overline0$, $\overline1^2 = \overline1$,
$\overline3^2 = \overline9 = \overline3$ e $\overline4^2 =
\overline{16} = \overline4$ --- \emph{quatro} idempotentes. Os dois
os exóticos vêm de divisores zero: $\overline3\,(\overline3 -
\overline1) = \overline3 \times \overline2 = \overline6 =
\overline0$ sem nenhum fator zero. Tais cálculos calibram
instintos: fatos familiares sobre equações sobrevivem em
domínios integrais e \omterm{def:b1:structures:field}{campos}, mas um geral \omterm{def:b1:structures:ring}{anel} pode e faz
se comportar de maneira diferente --- veja também o booleano anéis de
\cref{exo:b1:structures:10}, onde \emph{todo} elemento é
idempotente.
\end{example}

\begin{proposition}[Teorema binomial em um anel comutativo]\label{prop:b1:structures:binomial}
Se $ a, b $ são elementos de um comutativo \omterm{def:b1:structures:ring}{anel} (de forma mais geral, se
$ ab = ba $), então para $ n \in \N $:
\[
(a+b)^n = \sum_{k=0}^n \binom nk a^k b^{n-k},
\qquad
a^n - b^n = (a - b) \sum_{k=0}^{n-1} a^k b^{\,n-1-k} .
\]
\end{proposition}

\begin{proof}
As provas de \cref{thm:b1:counting:binomial} e da geometria
identidade usa apenas associatividade, comutatividade dos dois elementos,
e distributividade --- eles se aplicam literalmente.
\end{proof}

\begin{example}[O teorema binomial em um anel desconhecido]\label{ex:b1:structures:binomialring}
Duas recompensas rápidas da generalidade. Em $\Z/p\Z $ ($ p $ \omterm{def:b1:arith:prime}{melhor}), o
meio coeficientes binomiais desaparecer
(\cref{thm:b1:arith:fermat}primeiro passo), então o teorema
desmorona para o \emph{sonho de calouro}
\[
(a + b)^p = a^p + b^p \qquad \text{in } \Z/p\Z ,
\]
uma identidade genuína lá, por mais criminosa que pareça $\R $.
E em qualquer comutativo \omterm{def:b1:structures:ring}{anel} contendo um elemento $\varepsilon $
com $\varepsilon^2 = 0$, o teorema trunca:
$(a + \varepsilon)^n = a^n + n\,a^{n-1}\varepsilon $, tudo mais alto
termos carregando um fator $\varepsilon^2 = 0$. O coeficiente
$ n\,a^{n-1}$ de $\varepsilon $ é a derivada de $ x^n $ ---
não por acaso, e uma primeira dica de que derivadas são álgebra como
tanto quanto análise (compare a derivada formal de
\cref{ch:b1:poly}).
\end{example}

\begin{definition}[Domínio integral, campo]\label{def:b1:structures:field}
Um comutativo \omterm{def:b1:structures:ring}{anel} $ A \neq \{0\}$ é um \emph{membro
domínio domínio}\index{membro
domínio domínio} quando não tem divisores de zero: $ ab = 0
\implies a = 0$ ou $ b = 0$. É um \emph{campo}\index{campo} quando
todo elemento diferente de zero é invertível. Todo campo é uma integral
domínio ($ ab = 0$ e $ a \neq 0$ dar $ b = a^{-1}ab = 0$).
\end{definition}

\begin{example}
$\Q $, $\R $, $\C $ são \omterm{def:b1:structures:field}{campos}; $\Z $ é um domínio integral mas não um
\omterm{def:b1:structures:field}{campo}. Em um domínio integral, o cancelamento vale por $\times $: $ ab =
ac $ e $ a \neq 0$ implicar $ b = c $.
\end{example}

\section{O anel $\Z/n\Z $}

\begin{definition}\label{def:b1:structures:zn}
Consertar $ n \in \N^*$. O \omterm{def:b1:arith:congruence}{congruência} mod de aulas $ n $
(\cref{ex:b1:logic:congruence}) formam um \omterm{def:b1:logic:sets}{definir}
$\Z/n\Z $\index{Z/nZ@$\Z/n\Z $} de $ n $ elementos, escritos $\overline 0,
\overline 1, \dots, \overline{n-1}$. As operações
\[
\overline a + \overline b = \overline{a + b},
\qquad
\overline a \times \overline b = \overline{ab}
\]
são bem definidos --- as classes dos resultados não dependem do
representantes, justamente porque \omterm{def:b1:arith:congruence}{congruência} é compatível com $+$
e $\times $ (\cref{def:b1:arith:congruence}) --- e faça $\Z/n\Z $ um
comutativo \omterm{def:b1:structures:ring}{anel}.
\end{definition}

\begin{theorem}[Unidades de $\Z/n\Z $; os campos $\Z/p\Z $]\label{thm:b1:structures:zp}
\begin{enumerate}
  \item $\overline a $ é invertível em $\Z/n\Z $ se e somente se
        $\gcd(a, n) = 1$.
  \item $\Z/n\Z $ é um \omterm{def:b1:structures:field}{campo} se e somente se $ n $ é \omterm{def:b1:arith:prime}{melhor}.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) é \cref{prop:b1:arith:invmod} reescrito com aulas.

(2) Se $ n = p $ é \omterm{def:b1:arith:prime}{melhor}, todo $\overline a \neq \overline 0$ tem $ p
\nmid a $, então $\gcd(a, p) = 1$: invertível por (1) --- a \omterm{def:b1:structures:field}{campo}. Se $ n =
ab $ com $1 < a, b < n $, então $\overline a\, \overline b = \overline
n = \overline 0$ com $\overline a, \overline b \neq \overline 0$:
zero divisores, então nem mesmo um domínio integral; e $ n = 1$ dá o
zero \omterm{def:b1:structures:ring}{anel}, excluído.
\end{proof}

\begin{example}[Quantas raízes quadradas de $1$?]\label{ex:b1:structures:sqrtone}
Resolver $ x^2 = \overline 1$ em $\Z/8\Z $ e em $\Z/7\Z $. Testando o
mod de oito classes $8$: $1^2 = 1$, $3^2 = 9 \equiv 1$, $5^2 = 25
\equiv 1$, $7^2 = 49 \equiv 1$ --- \emph{quatro} soluções
$\{\overline1, \overline3, \overline5, \overline7\}$, embora
o polinômio $ X^2 - 1$ tem diploma $2$. No \omterm{def:b1:structures:field}{campo} $\Z/7\Z $,
em contraste, $ x^2 = \overline1$ significa $(x - \overline1)(x +
\overline1) = \overline0$, e um \omterm{def:b1:structures:field}{campo} não tem divisores zero: $ x =
\pm\overline1$, apenas duas soluções. O mod de falha $8$ é
rastreável: $(3-1)(3+1) = 2 \times 4 = 8 \equiv 0$ sem nenhum dos dois
desaparecimento do fator. Moral: a regra familiar “um grau-$ d $
equação tem no máximo $ d $ raízes'' é um teorema sobre
\emph{domínios integrais} (\cref{cor:b1:poly:nroots} prova que acabou
\omterm{def:b1:structures:field}{campos}); em anéis com divisores zero ele falha silenciosamente --- o que
é exatamente por isso que a prova de emparelhamento do teorema de Wilson
(\cref{exo:b1:arith:11}) necessário $ p $ \omterm{def:b1:arith:prime}{melhor}.
\end{example}

\begin{example}[Computação em $\Z/n\Z $]\label{ex:b1:structures:zncomputation}
Em $\Z/12\Z $: as unidades são $\overline 1, \overline 5, \overline 7,
\overline{11}$ (as aulas \omterm{cor:b1:arith:bezout}{coprime} para $12$), e cada um é seu
inverso ($5^2 = 25 \equiv 1$, $7^2 = 49 \equiv 1$, $11^2 = 121 \equiv
1$). A equação $\overline 3\, x = \overline 6$ tem \emph{três}
soluções ($ x \in \{\overline 2, \overline 6, \overline{10}\}$):
sem invertibilidade, sem cancelamento. Em $\Z/11\Z $ em contraste,
cada equação $\overline a x = \overline b $ com $\overline a \neq
\overline 0$ tem exatamente uma solução.
\end{example}

\begin{example}[Axiomas de grupo como licença de resolução]\label{ex:b1:structures:solving}
No \omterm{def:b1:structures:group}{grupo} $\bigl((\Z/7\Z)^*, \times\bigr)$, resolver $\overline
3\,x = \overline 5$. Por \cref{prop:b1:structures:rules}~(3) o
a solução existe, é única e é igual $\overline3^{-1}\,
\overline5$; desde $\overline3 \times \overline5 = \overline{15}
= \overline1$, o inverso de $\overline 3$ é $\overline 5$, então
\[
x = \overline5 \times \overline5 = \overline{25} = \overline4,
\qquad\text{check: } \overline3 \times \overline4 =
\overline{12} = \overline5 .
\]
A questão é menos a resposta do que a garantia: num \omterm{def:b1:structures:group}{grupo},
cada uma dessas equações pode ser resolvida exclusivamente \emph{antes} qualquer
computação, então um procedimento de resolução nunca pode se deparar com ``não
solução'' ou ``várias''. Comparar $\overline3\,x = \overline6$
em $\Z/12\Z $ acima, onde a garantia falha --- saber qual
A estrutura em que se está é saber o que se pode considerar garantido.
\end{example}

\begin{example}[Produtos diretos]\label{ex:b1:structures:product}
Se $ G $ e $ H $ são \omterm{def:b1:structures:group}{grupos}, o produto \omterm{def:b1:logic:sets}{definir} $ G \times H $ com o
lei componente $(g, h)(g', h') = (gg', hh')$ é um \omterm{def:b1:structures:group}{grupo}:
axiomas são verificados coordenada por coordenada, com identidade
$(e_G, e_H)$ e inverso $(g^{-1}, h^{-1})$. Pedidos combinar por
o lcm: $(g, h)^m = (g^m, h^m)$ é a identidade se o \omterm{def:b1:structures:order}{ordem}
de $ g $ e o \omterm{def:b1:structures:order}{ordem} de $ h $ ambos dividem $ m $. Assim em $\Z/2\Z
\times \Z/2\Z $ (aditivo) todo elemento diferente de zero tem \omterm{def:b1:structures:order}{ordem} $2$
--- este é exatamente o retângulo \omterm{def:b1:structures:group}{grupo} de
\cref{ex:b1:structures:klein} em coordenadas --- enquanto $\Z/4\Z $
tem um elemento de \omterm{def:b1:structures:order}{ordem} $4$: uma segunda prova sem cálculo
que os dois \omterm{def:b1:structures:group}{grupos} de tamanho $4$ não são isomórficos (um
isomorfismo preserva pedidos). Os produtos são a maneira mais fácil de
fabricar novo \omterm{def:b1:structures:group}{grupos} de antigamente, e o avião $\R^2 = \R \times
\R $ de \cref{ch:b1:vspaces} é o mais importante da construção
instância.
\end{example}

\begin{remark}[Fermat, estruturalmente]
No \omterm{def:b1:structures:field}{campo} $\Z/p\Z $, as classes diferentes de zero formam um multiplicativo \omterm{def:b1:structures:group}{grupo}
com $ p - 1$ elementos e o pequeno teorema de Fermat
(\cref{thm:b1:arith:fermat}) diz: cada elemento $ x $ disso \omterm{def:b1:structures:group}{grupo}
satisfaz $ x^{p-1} = \overline 1$. Este é um exemplo de uma situação geral
fato sobre finito \omterm{def:b1:structures:group}{grupos} (teorema de Lagrange), provado no segundo
ano; a prova de emparelhamento do teorema de Wilson
(\cref{exo:b1:arith:11}) já tinha esse sabor de teoria de \omterm{def:b1:structures:group}{grupo}.
\end{remark}

\begin{remark}[Interlúdio: o que a abstração compra]\label{rem:b1:structures:interlude}
É justo perguntar o que foi ganho provando, digamos,
\cref{prop:b1:structures:unique} por uma lei abstrata em vez de
para números. A resposta é alavancagem. Esse argumento de duas linhas agora
cobre, de uma só vez: inversas de funções sob composição
(\cref{thm:b1:logic:inverse}, cuja prova de unicidade ele repete
palavra por palavra), mod inverso $ n $
(\cref{prop:b1:arith:invmod}), inversos de reais diferentes de zero, de
unidades em qualquer \omterm{def:b1:structures:ring}{anel}, e --- invisível --- do invertível
matrizes de \cref{ch:b1:matrices}, onde a singularidade de $ A^{-1}$
não precisará de uma única linha de prova. A mesma economia vale para
\cref{prop:b1:structures:kernel} (um critério de injetividade,
reutilizado para linear \omterm{def:b1:logic:map}{mapas} em \cref{ch:b1:linmaps}) e para o
\omterm{def:b1:structures:subgroup}{subgrupo} critério. Abstração aqui não é generalidade por sua
próprio bem: é a recusa em provar o mesmo lema cinco vezes
sob cinco nomes. O preço --- acompanhar quais axiomas
cada \omterm{def:b1:logic:statement}{declaração} realmente usado --- é exatamente o que os exercícios de
este capítulo treina.
\end{remark}

\begin{remark}[Onde este capítulo é usado]\label{rem:b1:structures:whereused}
O vocabulário deste capítulo é a gramática do resto do
volume. Anéis e \omterm{def:b1:structures:field}{campos} organizar \cref{ch:b1:poly} ($ K[X]$ é um
\omterm{def:b1:structures:ring}{anel} imitando $\Z $) e \cref{ch:b1:fractions} ($ K(X)$ é o seu
\omterm{def:b1:structures:field}{campo} de frações); espaços vetoriais (\cref{ch:b1:vspaces}) são
\omterm{def:b1:structures:group}{grupos} \omterm{def:b1:structures:group}{abelianos} com um \omterm{def:b1:structures:field}{campo} agindo sobre eles; matrizes
(\cref{ch:b1:matrices}) formam o primeiro volume sério
não comutativo \omterm{def:b1:structures:ring}{anel}, e seus elementos invertíveis a \omterm{def:b1:structures:group}{grupo} cujo
estudo é a própria álgebra linear. \omterm{def:b1:structures:morphism}{Morfismos} e grãos retornar como
linear \omterm{def:b1:logic:map}{mapas} e espaços nulos em \cref{ch:b1:linmaps} ---
\cref{prop:b1:structures:kernel} \emph{é} a injetividade
critério desse capítulo, provado de uma vez por todas aqui. O
\omterm{ex:b1:structures:groups}{grupo simétrico}, estrela do problema do fim de semana abaixo, fornece o
assinatura na qual os determinantes são construídos \cref{ch:b1:det}.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b1:structures:1}
Sobre $ E = \R \setminus \{1\}$, \omterm{def:b1:logic:sets}{definir} $ x * y = x + y - xy $. Prove isso
$(E, *)$ é um \omterm{def:b1:structures:group}{grupo} \omterm{def:b1:structures:group}{abeliano}. \emph{(Identificar a identidade e o
inverso de $ x $; verifique a estabilidade: por que é $ x * y \neq 1$?)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:structures:2}
Quais dos seguintes são \omterm{def:b1:structures:group}{grupos}?
\begin{enumerate}
  \item $(\intoo{0}{+\infty}, \times)$;
  \item $(\{-1, 0, 1\}, +)$;
  \item $(\Q^*, \times)$;
  \item o \omterm{def:b1:logic:sets}{definir} de inteiros ímpares sob adição.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:structures:3}
Escreva a tabela de composição do \omterm{ex:b1:structures:groups}{grupo simétrico} $\mathfrak{S}_3$
de $\{1,2,3\}$ (seis bijeções: identidade, três transposições, duas
$3$-ciclos) e exibem dois elementos que não comutam.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:structures:4}
Prove isso $ H = \{z \in \C^* : \abs z = 1\}$ é um \omterm{def:b1:structures:subgroup}{subgrupo} de $(\C^*,
\times)$, e isso $\R_+^*$ é outro; é $ H \cup \R_+^*$ um
\omterm{def:b1:structures:subgroup}{subgrupo}?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:structures:5}
Deixar $ f \colon (\R, +) \to (\C^*, \times)$, $\theta \mapsto
\eu^{\iu\theta}$. Prove isso $ f $ é um \omterm{def:b1:structures:morphism}{morfismo}, calcular $\ker f $ e
$\operatorname{im} f $, e deduza de
\cref{prop:b1:structures:kernel} que $ f $ não é \omterm{def:b1:logic:inj}{injetivo}. Restringir
o domínio para fazer isso \omterm{def:b1:logic:inj}{injetivo} em um intervalo o maior possível.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:structures:6}
Deixar $ H, K $ ser \omterm{def:b1:structures:subgroup}{subgrupos} de $ G $. Prove isso $ H \cap K $ é um \omterm{def:b1:structures:subgroup}{subgrupo},
e isso $ H \cup K $ é um \omterm{def:b1:structures:subgroup}{subgrupo} \emph{apenas} quando $ H \subseteq K $ ou
$ K \subseteq H $. \emph{(Se $ h \in H \setminus K $ e $ k \in K
\setminus H $, onde pode $ hk $ ao vivo?)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:structures:7}
UM \omterm{def:b1:structures:group}{grupo} $ G $ satisfaz $ x^2 = e $ para todos $ x \in G $. Prove isso $ G $ é
\omterm{def:b1:structures:group}{abeliano}. \emph{(Expandir $(xy)^2$.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:structures:8}
Em $\Z/18\Z $: liste as unidades e encontre o inverso de $\overline 5$;
resolver $\overline 5\, x = \overline 7$; resolver $\overline 6\, x =
\overline 3$ e $\overline 6\, x = \overline{12}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:structures:9}
Prove que o \omterm{def:b1:logic:sets}{definir} $\Z[\sqrt 2] = \{a + b\sqrt 2 : a, b \in \Z\}$ é
um \omterm{def:b1:structures:ring}{anel} (um subanel de $\R $) e que $1 + \sqrt 2$ é uma unidade disso
com infinitos poderes distintos --- então $\Z[\sqrt 2]^\times $ é
infinito, ao contrário $\Z^\times $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:structures:10}
(Booleano anéis) Deixar $ A $ seja um \omterm{def:b1:structures:ring}{anel} em que $ x^2 = x $ para cada $ x $.
Prove isso $ x + x = 0$ para todos $ x $, e isso $ A $ é comutativo.
\emph{(Expandir $(x+x)^2$ e $(x+y)^2$.)} Dê um exemplo de tal
\omterm{def:b1:structures:ring}{anel} com $\mathcal{P}(E)$, tomando a diferença simétrica como
adição e interseção como multiplicação.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:structures:11}
Deixar $ G $ seja um \omterm{def:b1:structures:group}{grupo} em que, para alguns fixos $ n \geq 1$, $(xy)^n =
x^n y^n $, $(xy)^{n+1} = x^{n+1}y^{n+1}$ e $(xy)^{n+2} =
x^{n+2}y^{n+2}$ para todos $ x, y $. Prove isso $ G $ é \omterm{def:b1:structures:group}{abeliano}.
\emph{(Das três identidades, deduza primeiro $ y^n x = x y^n $, então
$ y^{n+1} x = x y^{n+1}$ e conclua.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:structures:12}
\begin{enumerate}
  \item Determinar tudo \omterm{def:b1:structures:group}{grupo} \omterm{def:b1:structures:morphism}{morfismos} de $(\Z, +)$ para $(\Z, +)$.
  \item Prove que o único \omterm{def:b1:structures:group}{grupo} \omterm{def:b1:structures:morphism}{morfismo} de $(\Q, +)$ para
        $(\Z, +)$ é o zero \omterm{def:b1:structures:morphism}{morfismo}. \emph{(Para $ x \in \Q $ e
        $ n \in \N^*$, comparar $ f(x)$ e $ n\,f(x/n)$.)}
\end{enumerate}
\end{exercise}

\section{Problema: O grupo simétrico e o quebra-cabeça de 8}

\begin{problem}\label{pb:b1:structures:1}
O \omterm{def:b1:structures:group}{grupo} $\mathfrak S_n $ de permutações de $\intint1n $ é o
mais antigo \omterm{def:b1:structures:group}{grupo} em matemática e ainda o mais instrutivo. Isto
problema constrói sua teoria de estrutura do zero --- ciclos,
geração por transposições, o \emph{assinatura} \omterm{def:b1:structures:morphism}{morfismo}
$\varepsilon \colon \mathfrak S_n \to \{\pm1\}$ (cuja existência é
genuinamente não trivial), e o \omterm{pb:b1:structures:1}{grupo alternado} $\mathfrak A_n $
gerado por $3$-cycles --- então ganha dinheiro com um clássico
quebra-cabeça: no $3 \times 3$ jogo de peças deslizantes, sem sequência de movimentos
pode trocar duas peças e deixar todo o resto no lugar.
\index{grupo simétrico}\index{assinatura}\index{grupo alternado}
Permutações agir $\intint1n $; produtos $\sigma\tau $ significa
``aplicar $\tau $ primeiro''; $[\,v_1, \dots, v_n]$ denota o
\omterm{def:b1:counting:objects}{permutação} enviando $ i $ para $ v_i $.

\medskip
\noindent\textbf{Part I --- Cycles and transpositions.}
\begin{enumerate}
  \item Justificar $\abs{\mathfrak S_n} = n!$
        (\cref{thm:b1:counting:counts}). Em $\mathfrak S_3$,
        calcular ambos os produtos de $\sigma = [2, 3, 1]$ e $\tau =
        [1, 3, 2]$, e concluir que $\mathfrak S_3$ não é
        \omterm{def:b1:structures:group}{abeliano}.
  \item UM \emph{$ k $-ciclo} $(a_1\ a_2\ \dots\ a_k)$ ($ k \geq 2$,
        o $ a_i $ distinto aos pares) envia $ a_1 \mapsto a_2
        \mapsto \dots \mapsto a_k \mapsto a_1$ e correções
        todo o resto; isso é \emph{apoiar} é $\{a_1, \dots,
        a_k\}$. Prove que dois ciclos com apoios disjuntos
        deslocamento.
  \item Prove que cada $\sigma \in \mathfrak S_n $ é um produto
        de ciclos com suportes disjuntos aos pares, e que este
        decomposição é única até o \omterm{def:b1:structures:order}{ordem} dos fatores.
        \emph{(Considere, para cada $ i $, a sequência $ i, \sigma(i),
        \sigma^2(i), \dots $: deve retornar para $ i $; o resultante
        \emph{orbits} partição $\intint1n $, e $\sigma $ atua em
        cada um como um ciclo.)}
  \item Decompor $\sigma = [4, 1, 5, 2, 3, 7, 8, 6] \in
        \mathfrak S_8$ em ciclos disjuntos. Definindo o
        \emph{\omterm{def:b1:structures:order}{ordem}} de $\sigma $ como em
        \cref{def:b1:structures:order}, prove que \omterm{def:b1:structures:order}{ordem} de um
        o produto de ciclos disjuntos é o lcm de seus comprimentos,
        e calcular o \omterm{def:b1:structures:order}{ordem} disso $\sigma $.
  \item Prove a identidade telescópica
        \[
        (a_1\ a_2\ \dots\ a_k)
        = (a_1\ a_k)(a_1\ a_{k-1})\cdots(a_1\ a_2) ,
        \]
        e concluir que cada \omterm{def:b1:counting:objects}{permutação} é um produto de
        transposições. Escreva o $\sigma $ da questão 4 como tal
        produto.
  \item Mostre ainda que \emph{adjacente} transposições
        $(i\ \ i{+}1)$ basta: para $ a < b $,
        \[
        (a\ b) = (a\ \ a{+}1)(a{+}1\ \ a{+}2)\cdots(b{-}1\ \ b)
        \cdots(a{+}1\ \ a{+}2)(a\ \ a{+}1),
        \]
        um produto de $2(b - a) - 1$ transposições adjacentes ---
        um \emph{chance} número (esta paridade importará duas vezes
        abaixo).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The signature exists.} Para $\sigma
\in \mathfrak S_n $, deixar
\[
N(\sigma) = \#\bigl\{(i, j) : i < j,\ \sigma(i) >
\sigma(j)\bigr\}
\]
seja o seu número de \emph{inversões}e \omterm{def:b1:logic:sets}{definir} $\varepsilon(\sigma) =
(-1)^{N(\sigma)}$.
\begin{enumerate}[resume]
  \item Calcular $ N $ e $\varepsilon $ pela identidade, por um
        transposição $(i\ \ i{+}1)$, e para $[2, 3, 1]$.
  \item Prove que para cada $\sigma $ e cada adjacente
        transposição $\tau = (i\ \ i{+}1)$:
        $ N(\sigma\tau) = N(\sigma) \pm 1$. \emph{(Compondo com
        $\tau $ à direita troca os valores em posições $ i $ e
        $ i + 1$; exatamente um par muda sua inversão
        estado.)}
  \item Deduza, usando a questão 6, que para \emph{qualquer}
        transposição $\tau $, $\varepsilon(\sigma\tau) =
        -\varepsilon(\sigma)$; concluir que se $\sigma $ é um
        produto de $ p $ transposições, então $\varepsilon(\sigma)
        = (-1)^p $ --- em particular a paridade de $ p $ depende apenas
        ligado $\sigma $, não na fatoração escolhida --- e isso
        $\varepsilon \colon \mathfrak S_n \to \{\pm 1\}$ é um
        \omterm{def:b1:structures:group}{grupo} \omterm{def:b1:structures:morphism}{morfismo}.
  \item Mostre que um $ k $-cycle tem assinatura $(-1)^{k-1}$ e
        que em geral $\varepsilon(\sigma) = (-1)^{n -
        c(\sigma)}$, onde $ c(\sigma)$ é o número de órbitas de
        $\sigma $ (pontos fixos incluídos).
  \item O \emph{\omterm{pb:b1:structures:1}{grupo alternado}} é $\mathfrak A_n =
        \ker\varepsilon $. Justifique que é um \omterm{def:b1:structures:subgroup}{subgrupo}, e provar
        $\abs{\mathfrak A_n} = \frac{n!}2$ para $ n \geq 2$.
        \emph{(Corrigir uma transposição $\tau_0$ e considere $\sigma
        \mapsto \sigma\tau_0$.)}
  \item Verificação de consistência ativada $\sigma = [4, 1, 5, 2, 3, 7, 8, 6]$:
        calcular $\varepsilon(\sigma)$ três maneiras --- contando
        inversões, do tipo de ciclo via questão 10, e de
        sua contagem de transposições na questão 5.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- $\mathfrak A_n $ is generated by
$3$-cycles.}
\begin{enumerate}[resume]
  \item Deixar $ a, b, c, d $ ser pares distintos. Verifique os dois
        identidades
        \[
        (a\ b)(a\ c) = (a\ c\ b),
        \qquad
        (a\ b)(c\ d) = (a\ c\ b)(a\ c\ d) .
        \]
  \item Prove isso para $ n \geq 3$, cada elemento de $\mathfrak
        A_n $ é um produto de $3$-ciclos. \emph{(Um par
        \omterm{def:b1:counting:objects}{permutação} é um produto de um número par de
        transposições; absorva-os dois de cada vez.)}
  \item Escrever $(1\ 2)(3\ 4)$ e o $5$-ciclo $(1\ 2\ 3\ 4\ 5)$
        explicitamente como produtos de $3$-ciclos.
  \item Prove a fórmula de conjugação: para cada $\sigma \in
        \mathfrak S_n $,
        \[
        \sigma\,(a_1\ \dots\ a_k)\,\sigma^{-1}
        = \bigl(\sigma(a_1)\ \dots\ \sigma(a_k)\bigr) .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The 8-puzzle.} Azulejos $1, \dots, 8$
deslize em um $3 \times 3$ quadro com uma célula vazia; um \emph{mover}
desliza um bloco adjacente à célula vazia para dentro dele. Numere as células
$1, \dots, 9$ (linha por linha; a posição resolvida tem bloco $ i $ em
célula $ i $ e a célula vazia na célula $9$). Trate a célula vazia como
uma nona peça, então uma posição é uma \omterm{def:b1:counting:objects}{permutação} $\sigma \in \mathfrak
S_9$ (telha $\sigma(i)$ senta na cela $ i $).
\begin{enumerate}[resume]
  \item Mostre que um movimento substitui $\sigma $ por $\sigma \circ \tau $
        onde $\tau $ é a transposição das duas células
        envolvido; deduzir que cada movimento vira
        $\varepsilon(\sigma)$.
  \item Deixar $ d(\sigma)$ será a distância do táxi (linhas mais
        colunas) entre a célula atual da célula vazia e sua
        célula doméstica $9$. Mostre que cada movimento muda $ d $ por $\pm1$,
        então cada movimento também vira $(-1)^{d(\sigma)}$. Conclua isso
        \[
        I(\sigma) = \varepsilon(\sigma)\cdot(-1)^{d(\sigma)}
        \]
        é \emph{invariante} sob cada movimento.
  \item Prove a impossibilidade clássica do quebra-cabeça: a posição
        que troca peças $7$ e $8$ e deixa todo o resto
        (incluindo a célula vazia) no local não pode ser alcançado
        da posição resolvida.
  \item Admitimos o inverso (sua prova é instrutiva, mas
        indução prolongada): cada posição com $ I = +1$ é
        alcançável. Deduza que exatamente metade do $8!$ posições
        com a cela vazia em casa são solucionáveis, ou seja,\
        $\frac{8!}2 = 20\,160$.
  \item Deduza da questão 20 que o bloco alcançável
        arranjos com a cela vazia em casa formam exatamente o mesmo
        \omterm{def:b1:structures:subgroup}{subgrupo} $\mathfrak A_8 \leq \mathfrak S_8$.
  \item Aplicações do invariante: pode-se chegar (a) ao
        posição onde as peças $1, 2, 3$ são ciclicamente permutados
        e todo o resto, inclusive a cela vazia, está em casa?
        (b) a posição onde o ladrilho $5$ e a célula vazia tem
        trocados de lugar e todas as outras peças estão em casa? Justificar
        ambas as respostas com $ I $.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Synthesis.}
\begin{enumerate}[resume]
  \item Prove isso para $ n \geq 3$ a única \omterm{def:b1:structures:group}{grupo} \omterm{def:b1:structures:morphism}{morfismos} $ f
        \colon \mathfrak S_n \to \{\pm 1\}$ são a constante
        \omterm{def:b1:structures:morphism}{morfismo} e $\varepsilon $. \emph{(Usando a pergunta 16 e
        comutatividade de $\{\pm1\}$, mostrar $ f $ leva o mesmo
        valor em todas as transposições.)}
  \item Onde exatamente o problema usou: (i) o \omterm{def:b1:structures:morphism}{morfismo}
        conceito e \cref{prop:b1:structures:kernel}; (ii) o
        princípios de contagem de \cref{ch:b1:counting}; (iii) o
        questão de bem-definição que as questões 8 a 9 resolvem? Um
        sentença cada.
  \item Síntese, num pequeno parágrafo: uma função de paridade,
        mostrou-se bem definido uma vez, organiza simultaneamente o
        estrutura interna de $\mathfrak S_n $ (o \omterm{def:b1:structures:subgroup}{subgrupo}
        $\mathfrak A_n $), decide um quebra-cabeça físico e ---
        através da fórmula $\det = \sum_\sigma
        \varepsilon(\sigma)\cdots$ --- definirá determinantes
        em \cref{ch:b1:det}. Comente sobre o padrão recorrente:
        invariantes transformam ``tentar todas as sequências de movimentos'' em uma
        computação.
\end{enumerate}
\end{problem}
