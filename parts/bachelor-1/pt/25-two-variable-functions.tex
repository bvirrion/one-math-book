\chapter{FunĂ§Ăľes de duas variĂĄveis}\label{ch:b1:multivar}

O ano termina com uma primeira caminhada para uma dimensĂŁo superior: funĂ§Ăľes
$f(x, y)$ de duas variĂĄveis reais. Tudo generaliza --- \omterm{def:b1:topology:closure}{limites},
continuidade, derivados, extremos --- mas cada noĂ§ĂŁo ganha uma reviravolta:
\omterm{def:b1:topology:closure}{limites} podem ser alcanĂ§ados em todas as direĂ§Ăľes ao mesmo tempo, derivados

o segundo ano; aqui definimos o vocabulĂĄrio e o primeiro honesto

\section{O aviĂŁo $\R^2$; continuidade}

\begin{definition}\label{def:b1:multivar:topology}
Sobre $\R^2$, use a norma euclidiana $\norm{(x,y)} = \sqrt{x^2 + y^2}$
(\cref{ch:b1:euclid}). \omterm{def:b1:topology:open}{Abrir} bolas, bairros, \emph{\omterm{def:b1:topology:open}{abrir}
subconjuntos} de $\R^2$ sĂŁo definidos exatamente como em
\cref{ch:b1:topology}, com bolas no lugar de \omterm{prop:b1:reals:intervals}{intervalos}. Uma funĂ§ĂŁo
$ f \colon U \to \R $ ($ U \subseteq \R^2$ \omterm{def:b1:topology:open}{abrir}) ĂŠ \emph{contĂ­nuo
no} $ a \in U $ quando
\[
\forall\varepsilon > 0,\ \exists\delta > 0, \quad
\norm{X - a} \leq \delta \implies \abs{f(X) - f(a)} \leq
\varepsilon,
\]
com a mesma caracterizaĂ§ĂŁo sequencial de uma variĂĄvel.
Somas, produtos, quocientes e composiĂ§Ăľes com contĂ­nuo
funĂ§Ăľes de uma variĂĄvel preservam continuidade; a coordenada \omterm{def:b1:logic:map}{mapas} sĂŁo
contĂ­nuo, portanto, tambĂŠm sĂŁo \omterm{def:b1:poly:def}{polinĂ´mios} em $(x,y)$.
\end{definition}

\begin{example}[O limite polar, a maneira limpa de provar um limite]
\label{ex:b1:multivar:polarbound}
Mostre isso $ f(x, y) = \dfrac{x^2y^2}{x^2 + y^2}$ (com $ f(0,0) =
0$) ĂŠ contĂ­nuo na origem. Em polar \omterm{prop:b1:vspaces:coordinates}{coordenadas} $ x =
\rho\cos\theta $, $ y = \rho\sin\theta $:
\[
\abs{f} = \frac{\rho^4\cos^2\theta\sin^2\theta}{\rho^2}
= \rho^2\,(\cos\theta\sin\theta)^2 \leq \frac{\rho^2}{4}
\xrightarrow[\rho \to 0]{} 0 ,
\]
um \omterm{def:b1:topology:closure}{limite} \emph{independente de $\theta $}: qualquer que seja a direĂ§ĂŁo
abordagem, os valores sĂŁo comprimidos para $0$. Essa uniformidade em
$\theta $ ĂŠ o ponto principal --- um \omterm{def:b1:topology:closure}{limite} como $\abs g =
\abs{\cos\theta\sin\theta}$ (nĂŁo $\rho $ esquerda) nĂŁo prova nada,
e de fato isso $ g $ ĂŠ a armadilha radial descontĂ­nua do
prĂłximo exemplo.
\end{example}

\begin{example}[A armadilha radial]\label{ex:b1:multivar:trap}
Deixar $ f(x, y) = \dfrac{xy}{x^2 + y^2}$ para $(x,y) \neq (0,0)$, $ f(0,
0) = 0$. Ao longo de cada eixo, $ f = 0 \to 0$; mas ao longo da diagonal $ y
= x $, $ f(x, x) = \frac12 \not\to 0$. \emph{Sem \omterm{def:b1:topology:closure}{limite} na origem}:
aproximando-se ao longo de cada linha, e atĂŠ mesmo encontrando o mesmo \omterm{def:b1:topology:closure}{limite} ao longo
cada um, nĂŁo ĂŠ suficiente (aqui os \omterm{def:b1:topology:closure}{limites} da linha discordam; exemplos piores
concordam em todas as linhas, mas falham em uma parĂĄbola,
\cref{exo:b1:multivar:3}). Continuidade em cada variĂĄvel separadamente
nĂŁo implica continuidade.
\end{example}

\section{Derivadas parciais}

\begin{definition}\label{def:b1:multivar:partial}
O \emph{derivadas parciais}\index{derivadas parciais} de $ f $ no
$(a, b)$ sĂŁo a Ăşnica variĂĄvel derivados ao longo dos eixos:
\[
\frac{\partial f}{\partial x}(a,b)
= \lim_{h \to 0} \frac{f(a + h, b) - f(a,b)}{h},
\qquad
\frac{\partial f}{\partial y}(a,b)
= \lim_{k \to 0} \frac{f(a, b + k) - f(a,b)}{k}.
\]
$ f $ ĂŠ de classe $ C^1$ sobre $ U $ quando ambos existem e sĂŁo contĂ­nuo sobre
$ U $. O \emph{gradiente}\index{gradiente} ĂŠ $\nabla f(a,b) =
\bigl(\frac{\partial f}{\partial x},\, \frac{\partial f}{\partial
y}\bigr)(a,b)$.
\end{definition}

\begin{theorem}[$ C^1$ implica um plano tangente]\label{thm:b1:multivar:c1}
Deixar $ f $ ser $ C^1$ sobre $ U $ e $(a,b) \in U $. EntĂŁo, como $(h, k) \to
(0,0)$:
\[
f(a + h, b + k) = f(a, b)
+ h\,\frac{\partial f}{\partial x}(a,b)
+ k\,\frac{\partial f}{\partial y}(a,b)
+ o\bigl(\norm{(h,k)}\bigr) .
\]
Em particular $ f $ ĂŠ contĂ­nuo, e o grĂĄfico $ z = f(x,y)$ tem em
cada ponto do plano tangente lĂŞ a fĂłrmula.
\end{theorem}

\begin{example}[AproximaĂ§ĂŁo linear em aĂ§ĂŁo]
\label{ex:b1:multivar:linapprox}
Estimativa $ f(1.02,\ 0.99)$ para $ f(x, y) = x^3y^2$. No $(1, 1)$:
$ f = 1$, $\frac{\partial f}{\partial x} = 3x^2y^2 = 3$,
$\frac{\partial f}{\partial y} = 2x^3y = 2$, entĂŁo
\cref{thm:b1:multivar:c1} dĂĄ
\[
f(1.02,\ 0.99) \approx 1 + 3\,(0.02) + 2\,(-0.01) = 1.04 ,
\]
contra o verdadeiro valor $1.02^3 \times 0.99^2 = 1.04006\dots $ ---
o erro ĂŠ de segunda ordem nos incrementos, pois o
$ o(\norm{(h,k)})$ promessas. O plano tangente ao grĂĄfico em
$(1, 1, 1)$ ĂŠ $ z = 1 + 3(x - 1) + 2(y - 1)$, a equaĂ§ĂŁo
implĂ­cita na estimativa.
\end{example}

\begin{proof}
Mova uma coordenada de cada vez:
\[
f(a+h, b+k) - f(a,b)
= \bigl[f(a+h, b+k) - f(a, b+k)\bigr] + \bigl[f(a, b+k) -
f(a,b)\bigr].
\]
Pelo teorema do valor mĂŠdio de uma variĂĄvel
(\cref{thm:b1:derivative:mvt}), o primeiro colchete ĂŠ $ h\,
\frac{\partial f}{\partial x}(a + \theta h,\, b + k)$ para alguns
$\theta \in \intoo{0}{1}$, e o segundo ĂŠ $ k\,\frac{\partial
f}{\partial y}(a,\, b + \theta' k)$. Continuidade das parciais em
$(a,b)$ vamos escrever cada um como (valor em $(a,b)$) $+$
(erro $\to 0$); o erro total ĂŠ $ h\,\varepsilon_1 +
k\,\varepsilon_2 = o(\norm{(h,k)})$ desde $\abs h, \abs k \leq
\norm{(h,k)}$.
\end{proof}

\begin{theorem}[Regra da cadeia]\label{thm:b1:multivar:chain}
Deixar $ f $ ser $ C^1$ sobre $ U $ e $ t \mapsto (x(t), y(t))$ ser $ C^1$ de
um \omterm{prop:b1:reals:intervals}{intervalo} em $ U $. EntĂŁo $ g(t) = f\bigl(x(t), y(t)\bigr)$ ĂŠ
$ C^1$, com\index{regra da cadeia!duas variĂĄveis}
\[
g'(t) = x'(t)\,\frac{\partial f}{\partial x}\bigl(x(t),y(t)\bigr)
+ y'(t)\,\frac{\partial f}{\partial y}\bigl(x(t),y(t)\bigr)
= \bigl\langle \nabla f,\ (x', y')\bigr\rangle .
\]
\end{theorem}

\begin{proof}
Aplicar \cref{thm:b1:multivar:c1} no $(x(t), y(t))$ com $(h, k) =
(x(t+s) - x(t),\, y(t+s) - y(t))$: como $ s \to 0$, uma variĂĄvel
diferenciabilidade dĂĄ $ h = s\,x'(t) + o(s)$ e $ k = s\,y'(t)
+ o(s)$, entĂŁo $\norm{(h, k)} = O(s)$ e
\[
g(t+s) - g(t)
= h\,\frac{\partial f}{\partial x} + k\,\frac{\partial
f}{\partial y} + o\bigl(\norm{(h,k)}\bigr)
= s\,\Bigl(x'\,\frac{\partial f}{\partial x} +
y'\,\frac{\partial f}{\partial y}\Bigr) + o(s) ,
\]
o erro final absorvendo tanto o $ o(s)$ de $ h $ e $ k $
(multiplicado pelos valores fixos das parciais) e o
$ o(O(s))$ da estimativa do plano tangente. Dividir por $ s $ e deixe
$ s \to 0$. Continuidade de $ g'$ segue de tudo
ingredientes.
\end{proof}

\begin{example}[A regra da cadeia, verificada em ambos os sentidos]
\label{ex:b1:multivar:chainrun}
Deixar $ f(x, y) = x^2 y $ e $ g(t) = f(t, t^2)$. Diretamente: $ g(t) =
t^2\cdot t^2 = t^4$, entĂŁo $ g'(t) = 4t^3$. Pela regra da cadeia:
$\frac{\partial f}{\partial x} = 2xy $ e $\frac{\partial
f}{\partial y} = x^2$, avaliado ao longo da curva $(t, t^2)$:
\[
g'(t) = 1\cdot 2t\cdot t^2 + 2t\cdot t^2 = 2t^3 + 2t^3 = 4t^3 .
\]
Os dois cĂĄlculos concordam e a \omterm{thm:b1:logic:partition}{divisĂŁo} ĂŠ significativa: $2t^3$
do crescimento vem da movimentaĂ§ĂŁo \emph{para a direita} atravĂŠs do
$ x $-declive, $2t^3$ de se mover \emph{para cima} atravĂŠs do
$ y $-declive. Em curvas onde nĂŁo hĂĄ forma fechada para $ g $ existe, apenas
o segundo cĂĄlculo sobrevive â esse ĂŠ o ponto do teorema.
\end{example}

\begin{remark}[Lendo o gradiente]
Ao longo de uma direĂ§ĂŁo de unidade $ u $, a regra da cadeia em $ t \mapsto f(a + tu)$
dĂĄ o \emph{direcional derivado} $\langle \nabla f(a),
u\rangle $: mĂĄximo quando $ u $ aponta ao longo $\nabla f(a)$ (por
Cauchy-Schwarz, \cref{thm:b1:euclid:cs}). O \omterm{def:b1:multivar:partial}{gradiente} ĂŠ o
direĂ§ĂŁo de subida mais Ă­ngreme, e ĂŠ ortogonal para o nĂ­vel
curvas $\{f = c\}$ (diferenciar $ f $ ao longo de uma curva desenhada em
nĂ­vel \omterm{def:b1:logic:sets}{definir}: a regra da cadeia dĂĄ $\langle\nabla f,\,
\text{tangent}\rangle = 0$).
\end{remark}

\begin{example}[Curvas de nĂ­vel e gradientes, em uma funĂ§ĂŁo]
\label{ex:b1:multivar:levelcurves}
Pegar $ f(x, y) = x^2 - y^2$. Seu nĂ­vel conjuntos: $\{f = c\}$ ĂŠ um
hipĂŠrbole abrindo da esquerda para a direita para $ c > 0$, de cima para baixo para $ c < 0$,
e o par de linhas cruzadas $ y = \pm x $ para $ c = 0$ --- o
contorno \omterm{def:b1:logic:map}{mapa} de uma passagem de montanha, com ponto de sela no
origem onde as duas linhas de nĂ­vel zero se cruzam. \omterm{def:b1:multivar:partial}{Gradiente}: $\nabla
f = (2x, -2y)$. No ponto $(2, 1)$ (no nĂ­vel $ c = 3$):
$\nabla f = (4, -2)$, enquanto o vetor tangente do nĂ­vel
curva, parametrizada perto desse ponto por $\bigl(t,
\sqrt{t^2 - 3}\bigr)$, ĂŠ $\bigl(1, \frac{t}{\sqrt{t^2 -
3}}\bigr) = (1, 2)$ no $ t = 2$ --- e de fato
\[
\langle (4, -2),\ (1, 2)\rangle = 4 - 4 = 0 :
\]
\omterm{def:b1:multivar:partial}{gradiente} perpendicular ao contorno, apontando para cima
valores de $ f $ (aqui: longe do $ y $-eixo). Mais dois
leituras: o \omterm{def:b1:multivar:partial}{gradiente} desaparece exatamente na sela, onde
o contorno \omterm{def:b1:logic:map}{mapa} beliscĂľes; e a tangente \emph{linha} para o
curva de nĂ­vel em $(2,1)$ ĂŠ $4(x - 2) - 2(y - 1) = 0$, ou seja\ $2x -
y = 3$ --- a equaĂ§ĂŁo ``$\langle \nabla f,\ M - M_0\rangle =
0$'' que generaliza a tangente da elipse de
\cref{exo:b1:curves:11}.
\end{example}

\begin{theorem}[Preto]\label{thm:b1:multivar:schwarz}
Se $ f $ ĂŠ de classe $ C^2$ (parciais de parciais existem e sĂŁo
contĂ­nuo), entĂŁo\index{Teorema de Schwarz}
\[
\frac{\partial^2 f}{\partial x\,\partial y}
= \frac{\partial^2 f}{\partial y\,\partial x} .
\]
\end{theorem}
\admitted

\section{Extremos locais}

\begin{method}[Estudos extremos, organizados]
\label{met:b1:multivar:extremamethod}
\begin{enumerate}
  \item \emph{Resolver $\nabla f = 0$ completamente.} Fatore cada
        parcial sempre que possĂ­vel (produtos de \omterm{def:b1:linmaps:def}{linear} fatores
        dividir o sistema em casos transparentes, como em
        \cref{ex:b1:multivar:fourpoints} abaixo); um esquecido
        caso ĂŠ um esquecido ponto \omterm{thm:b1:multivar:critical}{crĂ­tico}.
  \item \emph{Classifique cada ponto} com os dados de Monge $ r, s,
        t $ --- recalculado \emph{em cada ponto}, nunca uma vez por
        tudo.
  \item \emph{Se $ rt - s^2 = 0$}, examinar $ f $ diretamente ao longo
        curvas bem escolhidas atravĂŠs do ponto (linhas primeiro, depois
        parĂĄbolas), caĂ§ando dois sinais (sem extremo)
        ou para uma placa trancada com um argumento cobrindo todos
        direĂ§Ăľes.
  \item \emph{DĂŞ um passo para trĂĄs para ver a \omterm{def:b1:linmaps:kerim}{imagem} global}: verifique o
        comportamento no infinito (um mĂ­nimo local pode nĂŁo ser global
        um), e se o domĂ­nio nĂŁo for \omterm{def:b1:topology:open}{abrir}, trate seu \omterm{def:b1:topology:closure}{limite}
        separadamente (\cref{exo:b1:multivar:12}) --- o \omterm{thm:b1:multivar:critical}{crĂ­tico}
        ponto teorema vĂŞ \omterm{def:b1:topology:closure}{interior} apenas pontos.
\end{enumerate}
\end{method}

\begin{theorem}[Pontos crĂ­ticos]\label{thm:b1:multivar:critical}
Se $ f $ ($ C^1$ no conjunto aberto $ U $) tem um extremo local em $(a,b)
\in U $, entĂŁo $\nabla f(a,b) = (0,0)$: o ponto ĂŠ
\emph{crĂ­tico}\index{crĂ­tico}.
\end{theorem}

\begin{proof}
As funĂ§Ăľes de uma variĂĄvel $ x \mapsto f(x, b)$ e $ y \mapsto f(a,
y)$ ter \omterm{def:b1:topology:closure}{interior} extremos locais em $ a $, resp.\ $ b $:
\cref{prop:b1:derivative:fermat} mata ambos os parciais.
\end{proof}

\begin{method}[Teste de segunda ordem (notaĂ§ĂŁo Monge)]\label{met:b1:multivar:monge}
Em um ponto \omterm{thm:b1:multivar:critical}{crĂ­tico} de um $ C^2$ funĂ§ĂŁo, conjunto
\[
r = \frac{\partial^2 f}{\partial x^2},
\qquad
s = \frac{\partial^2 f}{\partial x \partial y},
\qquad
t = \frac{\partial^2 f}{\partial y^2}
\qquad (\text{values at the point}).
\]
\begin{itemize}
  \item Se $ rt - s^2 > 0$: extremo local --- mĂ­nimo para $ r > 0$,
        mĂĄximo para $ r < 0$;
  \item se $ rt - s^2 < 0$: sem extremo (um \emph{sela
        ponto}\index{sela
        ponto});
  \item se $ rt - s^2 = 0$: o teste ĂŠ silencioso; examinar diretamente.
\end{itemize}
(Justificativa --- uma expansĂŁo Taylor--Young no pedido $2$ e o
estudo de sinal da forma quadrĂĄtica $ r h^2 + 2shk + tk^2$ --- ĂŠ
realizado no segundo ano; o teste ĂŠ usado aqui como um trabalho
ferramenta.)
\end{method}

\begin{example}\label{ex:b1:multivar:extrema}
$ f(x,y) = x^3 + y^3 - 3xy $. Pontos \omterm{thm:b1:multivar:critical}{crĂ­ticos}: $\nabla f = (3x^2 -
3y,\; 3y^2 - 3x) = 0$ dĂĄ $ y = x^2$ e $ x = y^2$, entĂŁo $ x = x^4$:
$ x \in \{0, 1\}$: pontos $(0,0)$ e $(1,1)$.

Segundo derivados: $ r = 6x $, $ s = -3$, $ t = 6y $. No $(0,0)$: $ rt -
s^2 = -9 < 0$: sela. No $(1,1)$: $ rt - s^2 = 36 - 9 > 0$, $ r = 6
> 0$: mĂ­nimo local, $ f(1,1) = -1$. (NĂŁo global: $ f(x, 0) = x^3 \to
-\infty $.)
\end{example}

\begin{example}[Um estudo de quatro pontos, na Ă­ntegra]
\label{ex:b1:multivar:fourpoints}

\[

\qquad

$ r = -2y $, $ s = 3 - 2x - 2y $, $ t = -2x $.
\begin{itemize}
  \item $(1,1)$: $ r = -2$, $ s = -1$, $ t = -2$: $ rt - s^2 = 3 >

+\infty $. VerificaĂ§ĂŁo de simetria: $ f(x, y) = f(y, x)$, e de fato o
\omterm{thm:b1:multivar:critical}{crĂ­tico} \omterm{def:b1:logic:sets}{definir} e a classificaĂ§ĂŁo ĂŠ simĂŠtrica em $ x
\leftrightarrow y $. InterpretaĂ§ĂŁo: entre retĂ˘ngulos com folga

\end{example}

\begin{remark}[Armadilhas comuns]
\label{rem:b1:multivar:pitfalls}
\emph{\omterm{def:b1:multivar:partial}{Derivadas parciais} pode existir em um ponto de
descontinuidade}: a armadilha radial $ g(x,y) = \frac{xy}{x^2+y^2}$ de
\cref{ex:b1:multivar:trap} tem $\frac{\partial g}{\partial
x}(0,0) = \frac{\partial g}{\partial y}(0,0) = 0$ (ambos os eixos
restriĂ§Ăľes desaparecem de forma idĂŞntica), ainda $ g $ nĂŁo tem \omterm{def:b1:topology:closure}{limite} no
origem --- parciais sondam apenas duas direĂ§Ăľes, continuidade necessidades
todos eles; apenas o $ C^1$ hipĂłtese restaura a ordem
(\cref{thm:b1:multivar:c1}).
\emph{\omterm{def:b1:topology:closure}{Limites} de linha nunca sĂŁo suficientes}: \cref{exo:b1:multivar:3}de
funĂ§ĂŁo tem \omterm{def:b1:topology:closure}{limite} $0$ ao longo de todas as linhas e ainda sem \omterm{def:b1:topology:closure}{limite} ---
sempre tente parĂĄbolas (ou \omterm{def:b1:topology:closure}{limites} polares vĂĄlidos uniformemente em
$\theta $).
\emph{\omterm{thm:b1:multivar:critical}{CrĂ­tico} ĂŠ necessĂĄrio, nĂŁo suficiente}: selas abundam
(trĂŞs dos quatro pontos em \cref{ex:b1:multivar:fourpoints}); e
o teorema se mantĂŠm \emph{\omterm{def:b1:topology:open}{abrir}} conjuntos apenas --- em um \omterm{def:b1:topology:closed}{fechado} disco,
extremos podem ficar no \omterm{def:b1:topology:closure}{limite} com diferente de zero \omterm{def:b1:multivar:partial}{gradiente}

\section{Problema: mĂ­nimos quadrados e a linha de regressĂŁo}

\begin{problem}\label{pb:b1:multivar:1}
Dado $ n $ pontos de dados, cuja linha reta passa âmais prĂłximaâ de
todos eles? Legendre e Gauss responderam: a linha que minimiza o
soma do \emph{ao quadrado} erros verticais --- porque isso
a minimizaĂ§ĂŁo ĂŠ exatamente solucionĂĄvel pela ĂĄlgebra \omterm{def:b1:linmaps:def}{linear}. Este problema
prova pela primeira vez o teste de Monge de \cref{met:b1:multivar:monge}
honestamente para funĂ§Ăľes quadrĂĄticas (o Ăşnico caso em que o
expansĂŁo de segunda ordem ĂŠ exata), entĂŁo constrĂłi o normais
equaĂ§Ăľes, o \omterm{pb:b1:multivar:1}{linha de regressĂŁo}, e o coeficiente de correlaĂ§ĂŁo
no topo da geometria euclidiana de
\cref{ch:b1:euclid}.\index{mĂ­nimos quadrados}\index{linha de regressĂŁo}
\index{equaĂ§Ăľes normais}\index{correlaĂ§ĂŁo}

\medskip
\noindent\textbf{Part I --- Quadratic functions: the Monge test,
proved.} Corrigir reais $ r, s, t $ e deixe $ q(h, k) = r h^2 + 2s hk +
t k^2$.
\begin{enumerate}
  \item Suponha $ r \neq 0$. EstabeleĂ§a o formulĂĄrio quadrado preenchido
        \[
        q(h, k) = r\Bigl(h + \frac{s}{r}k\Bigr)^{\!2}
        + \frac{rt - s^2}{r}\,k^2 ,
        \]
        e deduza: se $ rt - s^2 > 0$, $ q $ tem o sinal estrito
        de $ r $ fora da origem; se $ rt - s^2 < 0$, $ q $ leva os dois
        sinais.
  \item Resolva os casos restantes: $ r = 0$, $ t \neq 0$
        (simĂŠtrico); e $ r = t = 0$ ($ q = 2shk $). Concluir:
        $ q $ assume ambos os sinais se $ rt - s^2 < 0$, e $ q $
        desaparece apenas na origem se $ rt - s^2 > 0$.
  \item Deixe agora $ f(x, y) = \frac12\bigl(r x^2 + 2s xy + t
        y^2\bigr) + \beta x + \gamma y + \delta $ ser um quadrĂĄtico
        funcionar com um ponto \omterm{thm:b1:multivar:critical}{crĂ­tico} $ X_0 = (a, b)$. Prove o
        \emph{exato} expansĂŁo
        \[
        f(X_0 + (h,k)) = f(X_0) + \tfrac12\,q(h, k)
        \qquad (\text{no remainder}),
        \]
        e deduzir a classificaĂ§ĂŁo de Monge para quadrĂĄtica
        funĂ§Ăľes: mĂ­nimo global estrito se $ rt - s^2 > 0$, $ r >
        0$; mĂĄximo global estrito se $ rt - s^2 > 0$, $ r < 0$;
        sela se $ rt - s^2 < 0$.
  \item Suponha $ rt - s^2 > 0$ e $ r > 0$ (assim tambĂŠm $ t > 0$).
        Prove o \omterm{def:b1:topology:closure}{limite} de coercividade explĂ­cito
        \[
        q(h, k) \geq c\,(h^2 + k^2),
        \qquad
        c = \frac{rt - s^2}{r + t} > 0 .
        \]
        \emph{(Mostre que a forma quadrĂĄtica $ q - c(h^2 + k^2)$,
        com coeficientes $ r - c $, $ s $, $ t - c $, ainda tem
        dados de teste discriminante nĂŁo negativo.)}
  \item Deduza que uma funĂ§ĂŁo quadrĂĄtica com definido positivo
        parte quadrĂĄtica tende a $+\infty $ como $\norm{(x,y)} \to
        \infty $, e portanto tem um \emph{Ăşnico global}
        minimizador: seu ponto \omterm{thm:b1:multivar:critical}{crĂ­tico}. Compare com
        \cref{ex:b1:multivar:extrema}, onde um mĂ­nimo local de
        uma funĂ§ĂŁo nĂŁo quadrĂĄtica nĂŁo era global.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The \omterm{pb:b1:multivar:1}{equaĂ§Ăľes normais}.} Deixar $ C_1,
C_2, b \in \R^n $ (canĂ´nico \omterm{def:b1:euclid:def}{produto interno}), e
\[

\]
\begin{enumerate}[resume]
  \item Expandir $ f $ e mostre que ĂŠ uma funĂ§ĂŁo quadrĂĄtica de $(u,
        v)$ com coeficientes de parte quadrĂĄtica $ r = 2\norm{C_1}^2$,
        $ s = 2\langle C_1, C_2\rangle $, $ t = 2\norm{C_2}^2$.
  \item Mostre isso $ rt - s^2 > 0$ se e somente se $(C_1, C_2)$ ĂŠ
        \omterm{def:b1:vspaces:free}{livre} \emph{(o caso da igualdade de Cauchy-Schwarz,
        \cref{thm:b1:euclid:cs})}; assuma isso de agora em diante.
  \item Mostre que as equaĂ§Ăľes crĂ­ticas $\nabla f = 0$ sĂŁo os
        \emph{\omterm{pb:b1:multivar:1}{equaĂ§Ăľes normais}}
        \[
        \begin{pmatrix}
        \norm{C_1}^2 & \langle C_1, C_2\rangle\\
        \langle C_1, C_2\rangle & \norm{C_2}^2
        \end{pmatrix}
        \begin{pmatrix} u\\ v\end{pmatrix}
        =
        \begin{pmatrix} \langle C_1, b\rangle\\ \langle C_2,
        b\rangle\end{pmatrix}
        \]
        --- a matriz Gram de \cref{exo:b1:euclid:11} no
        esquerda --- e que eles dizem exatamente: $ b - (uC_1 + vC_2)
        \perp C_1, C_2$. Conclua com a Parte I e
        \cref{thm:b1:euclid:projection}: o minimizador exclusivo
        dĂĄ $ p = uC_1 + vC_2 = $ o projeĂ§ĂŁo ortogonal de
        $ b $ para $\operatorname{Vect}(C_1, C_2)$.
  \item Mostre que o valor mĂ­nimo ĂŠ $\norm{b}^2 - \langle p,
        b\rangle = \norm{b - p}^2$, e desenhe o pitagĂłrico
        foto.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The \omterm{pb:b1:multivar:1}{linha de regressĂŁo}.} Pontos de dados
$(x_1, y_1), \dots, (x_n, y_n)$, nem todos $ x_i $ igual. Minimizar
\[
E(\alpha, \beta) = \sum_{i=1}^{n}\bigl(y_i - \alpha - \beta
x_i\bigr)^2 .
\]
Escrever $\bar x = \frac1n\sum x_i $, $\bar y = \frac1n\sum y_i $,
$ v_x = \frac1n\sum x_i^2 - \bar x^2$, $ v_y = \frac1n\sum y_i^2 -
\bar y^2$, $ c_{xy} = \frac1n\sum x_i y_i - \bar x\bar y $.
\begin{enumerate}[resume]
  \item Reconhecer a Parte II com $ C_1 = (1, \dots, 1)$, $ C_2 =
        (x_1, \dots, x_n)$, $ b = (y_1, \dots, y_n)$ e escreva
        o \omterm{pb:b1:multivar:1}{equaĂ§Ăľes normais}
        \[
        n\alpha + \Bigl(\sum x_i\Bigr)\beta = \sum y_i,
        \qquad
        \Bigl(\sum x_i\Bigr)\alpha + \Bigl(\sum
        x_i^2\Bigr)\beta = \sum x_i y_i .
        \]
  \item Resolva-os:
        \[
        \beta = \frac{c_{xy}}{v_x},
        \qquad
        \alpha = \bar y - \beta\,\bar x ,
        \]
        e observe que o \emph{\omterm{pb:b1:multivar:1}{linha de regressĂŁo}} $ y = \alpha
        + \beta x $ passa pelo ponto mĂŠdio $(\bar x, \bar
        y)$.
  \item Verificar $ v_x > 0$ exatamente quando o $ x_i $ nĂŁo sĂŁo todos
        igual e combine isso com a pergunta 7.
  \item Prove que o erro mĂ­nimo ĂŠ
        \[
        E_{\min} = n\Bigl(v_y - \frac{c_{xy}^2}{v_x}\Bigr)
        = n\,v_y\,(1 - \rho^2),
        \qquad
        \rho = \frac{c_{xy}}{\sqrt{v_x v_y}}
        \quad (v_y \neq 0).
        \]
        Deduzir $\abs\rho \leq 1$, com $\abs\rho = 1$ se e
        somente se os dados estiverem perfeitamente alinhados.
  \item Exemplo resolvido: para os dados $(0,0)$, $(1,1)$, $(2,3)$,
        $(3,4)$, calcular $\bar x, \bar y, v_x, c_{xy}$, o
        \omterm{pb:b1:multivar:1}{linha de regressĂŁo}, os quatro resĂ­duos e $ E_{\min}$.
  \item Prove que os resĂ­duos $\varepsilon_i = y_i - \alpha -
        \beta x_i $ da linha Ăłtima satisfaz $\sum_i
        \varepsilon_i = 0$ e $\sum_i x_i\varepsilon_i = 0$,
        e interpretar ambos via ortogonalidade.
\end{enumerate}

  \item (Ajuste parabĂłlico) Para ajustar $ y = a + bx + cx^2$, mostre que
        o \omterm{pb:b1:multivar:1}{equaĂ§Ăľes normais} sĂŁo os $3 \times 3$ sistema com

        \[
