\chapter{Limites e Continuidade}\label{ch:b1:continuity}

O teorema do valor intermediário e o teorema do valor extremo foram
usado no ensino médio na fé visual. Com sequências
(\cref{ch:b1:seq}) e a topologia de $\R $ (\cref{ch:b1:topology})
em mãos, este capítulo os prova --- e completa a teoria com
o teorema da bijeção monótona (que legitima $\arcsin $,
$\operatorname{arcosh}$ e seus parentes de \cref{ch:b1:functions})
e o teorema de Heine sobre continuidade uniforme.

Por todo, $ I $ é um \omterm{prop:b1:reals:intervals}{intervalo} e $ f \colon I \to \R $; ``$ x_0 \in
\overline I $'' permite \omterm{def:b1:topology:closure}{limites} nos pontos finais.

\section{Limites de funções}

\begin{definition}[Limite em um ponto]\label{def:b1:continuity:limit}
Deixar $ x_0 \in \overline{I}$ e $\ell \in \R $. Então $ f(x) \to \ell $ como
$ x \to x_0$ quando\index{limite!de uma função}
\[
\forall \varepsilon > 0,\ \exists \delta > 0,\ \forall x \in I,
\qquad \abs{x - x_0} \leq \delta \implies \abs{f(x) - \ell} \leq
\varepsilon .
\]
\omterm{def:b1:topology:closure}{Limites} em $\pm\infty $ e \omterm{def:b1:topology:closure}{limites} infinitos são definidos pelo mesmo
padrão ($\abs{x - x_0} \leq \delta $ torna-se $ x \geq M $; $\abs{f(x) -
\ell} \leq \varepsilon $ torna-se $ f(x) \geq M'$). \omterm{def:b1:topology:closure}{Limites} unilaterais
restringir $ x $ para $ x > x_0$ (escrito $ x \to x_0^+$) ou $ x < x_0$. O
\omterm{def:b1:topology:closure}{limite} é único quando existe (mesma prova que
\cref{prop:b1:seq:first}).
\end{definition}

\begin{example}[Um limite através do aperto]\label{ex:b1:continuity:floorlimit}
Calcular $\lim_{x \to 0} x\,\bigl\lfloor \frac1x \bigr\rfloor $.
O \omterm{thm:b1:reals:floor}{chão} colchetes $\frac1x - 1 < \lfloor \frac1x \rfloor \leq
\frac1x $ dá, depois de multiplicar por $ x $ (cuidado com o sinal!):
\[
1 - x < x\Bigl\lfloor \frac1x \Bigr\rfloor \leq 1 \quad (x > 0),
\qquad
1 \leq x\Bigl\lfloor \frac1x \Bigr\rfloor < 1 - x \quad (x < 0),
\]
e ambos os apertos unilaterais próximos $1$: o \omterm{def:b1:topology:closure}{limite} é $1$. Nota
o que aconteceu: $\lfloor \frac1x\rfloor $ sozinho tem saltos selvagens perto
$0$, mas o fator $ x $ doma cada salto ($ x $ vezes que um salto unitário é
pequeno), e apenas o escalonamento sobrevive. A visão final:
\omterm{def:b1:topology:closure}{limites} de produtos de um fator pequeno com uma oscilação limitada
fator são problemas de compressão, nunca problemas de teorema de operação ---
o teorema das operações precisa \emph{ambos} fatores convergirem.
\end{example}

\begin{theorem}[Caracterização sequencial]\label{thm:b1:continuity:seqchar}
$ f(x) \to \ell $ como $ x \to x_0$ se e somente se: para \emph{todo}
sequência $(u_n)$ de pontos de $ I $ com $ u_n \to x_0$, um tem
$ f(u_n) \to \ell $.
\end{theorem}

\begin{proof}
($\Rightarrow $) Deixar $ u_n \to x_0$ e $\varepsilon > 0$. Pegar
$\delta $ da definição, então $ N $ com $\abs{u_n - x_0} \leq
\delta $ para $ n \geq N $: além $ N $, $\abs{f(u_n) - \ell} \leq
\varepsilon $.

($\Leftarrow $) Contrapositivo. Se $ f \not\to \ell $: alguns
$\varepsilon_0 > 0$ derrota todos $\delta $; escolhendo $\delta =
\frac{1}{n+1}$ produz $ u_n \in I $ com $\abs{u_n - x_0} \leq
\frac{1}{n+1}$ e $\abs{f(u_n) - \ell} > \varepsilon_0$. Então $ u_n
\to x_0$ mas $ f(u_n) \not\to \ell $.
\end{proof}

\begin{corollary}[Operações, composição, ordem]\label{cor:b1:continuity:operations}
Somas, produtos, quocientes (\omterm{def:b1:topology:closure}{limite} diferente de zero no andar de baixo) dos \omterm{def:b1:topology:closure}{limites} se comportam
quanto às sequências; se $ f \to \ell $ no $ x_0$ e $ g \to m $ no $\ell $,
e $ g $ é definido em torno $\ell $ com $ g(\ell) = m $ ou $ f \neq \ell $
aproximar $ x_0$, então $ g \circ f \to m $ no $ x_0$; \omterm{def:b1:topology:closure}{limites} preservam ampla
desigualdades, e o teorema da compressão é válido.
\end{corollary}

\begin{proof}
Cada \omterm{def:b1:logic:statement}{declaração} transferências através
\cref{thm:b1:continuity:seqchar} ao seu análogo de sequência
(\cref{thm:b1:seq:operations,thm:b1:seq:order}). Para composição,
vale a pena escrever o encadeamento direto uma vez: dado $\varepsilon >
0$, o \omterm{def:b1:topology:closure}{limite} de $ g $ no $\ell $ fornece $\eta > 0$ com
\[
\abs{y - \ell} \leq \eta \implies \abs{g(y) - m} \leq
\varepsilon \quad (y \text{ in the domain of } g),
\]
onde o caso $ y = \ell $ está coberto porque $ g(\ell) = m $; então
o \omterm{def:b1:topology:closure}{limite} de $ f $ no $ x_0$ fornece $\delta > 0$ com $\abs{x -
x_0} \leq \delta \implies \abs{f(x) - \ell} \leq \eta $; encadeamento,
$\abs{x - x_0} \leq \delta $ dá $\abs{g(f(x)) - m} \leq
\varepsilon $. Na hipótese alternativa ($ f \neq \ell $ aproximar
$ x_0$), o valor $ y = \ell $ nunca é alimentado para $ g $ e seu valor
aí é irrelevante.
\end{proof}

\begin{example}[Por que existe a cláusula de composição]\label{ex:b1:continuity:compositiontrap}
Deixar $ g(y) = 0$ para $ y \neq 0$ e $ g(0) = 1$, e deixe $ f $ ser
de forma idêntica $0$. Então $ f(x) \to 0$ como $ x \to 0$, e $ g(y) \to 0$
como $ y \to 0$; ainda $ g(f(x)) = g(0) = 1$ para cada $ x $, então $ g
\circ f \to 1 \neq 0$. A função interna fica \emph{exatamente em}
o valor proibido $\ell = 0$ para sempre, e $ g $ \omterm{def:b1:topology:closure}{limite} de $0$
ignora o que $ g $ faz em $0$. A disposição de
\cref{cor:b1:continuity:operations} --- qualquer $ g(\ell) = m $
(ou seja,\ $ g $ contínuo no $\ell $), ou $ f \neq \ell $ aproximar $ x_0$ ---
é precisamente o que exclui isso. A visão final: em
pratique um compõe \emph{contínuo} funções e o
a condição é gratuita; morde apenas quando os \omterm{def:b1:topology:closure}{limites} são levados junto
perfurado bairros, razão pela qual a definição de
$\lim_{x \to x_0}$ usado neste livro inclui o ponto em que
está no domínio.
\end{example}

\section{Continuidade}

\begin{definition}\label{def:b1:continuity:continuous}
$ f $ é \emph{contínuo em $ x_0 \in I $}\index{contínuo em $ x_0 \in I $} quando $ f(x)
\to f(x_0)$ como $ x \to x_0$; \emph{contínuo ligado $ I $} quando é
contínua em todos os pontos. Por
\cref{thm:b1:continuity:seqchar}: $ f $ é contínuo em $ x_0$ se
$ f(u_n) \to f(x_0)$ para cada sequência $ u_n \to x_0$ em $ I $.
\end{definition}

\begin{example}[Uma taxonomia de descontinuidades]\label{ex:b1:continuity:taxonomy}
Três maneiras de falhar em um determinado momento, aumentando a gravidade.
\emph{Removível}: $ f(x) = \frac{\sin x}{x}$ sobre $\R^*$ tem \omterm{def:b1:topology:closure}{limite}
$1$ no $0$; definindo $ f(0) = 1$ repara-o --- a descontinuidade
era um buraco, não uma característica. \emph{Pular}: $\lfloor x \rfloor $ em
um número inteiro tem \omterm{def:b1:topology:closure}{limites} unilaterais distintos ($ n - 1$ e $ n $); não
a escolha do valor pode reconciliá-los, mas ambos os \omterm{def:b1:topology:closure}{meios-limites} existem.
\emph{Essencial}: $\sin\frac1x $ no $0$ não tem \omterm{def:b1:topology:closure}{limite} unilateral em
tudo (\cref{exo:b1:continuity:1}) --- oscilação sem
liquidação. Funções monótonas só podem produzir o tipo intermediário
(seus \omterm{def:b1:topology:closure}{limites} unilaterais sempre existem, sendo suprema e ínfima),
é por isso que sua descontinuidade conjuntos são no máximo contáveis ---
um racional por salto. Derivadas, pelo teorema de Darboux
(\cref{exo:b1:derivative:10}), só pode produzir o último tipo:
uma função com descontinuidade de salto nunca é a derivada de
qualquer coisa.
\end{example}

\begin{proposition}\label{prop:b1:continuity:algebra}
Somas, produtos, quocientes (quando definidos) e composições de
contínuo funções são contínuo. \omterm{def:b1:poly:def}{Polinômios}, frações racionais
(fora deles postes), $\abs{\,\cdot\,}$, $\exp $, $\ln $, o
trigonométrico e funções hiperbólicas e seus inversos
(\cref{ch:b1:functions}) são contínuo em seus domínios.
\end{proposition}

\begin{proof}
Operações: \cref{cor:b1:continuity:operations}. Constantes e
a identidade é contínuo diretamente da definição
($\delta = \varepsilon $ serve a identidade, qualquer $\delta $ o
constantes); produtos de contínuo funções sendo contínuo,
cada monômio $ a_k x^k $ segue por indução em $ k $ e somas
terminar o \omterm{def:b1:poly:def}{polinômios}; um \omterm{def:b1:fractions:field}{fração racional} é um quociente de dois
\omterm{def:b1:poly:def}{polinômios}, contínuo onde quer que o denominador não
desaparecer. $\abs{\,\cdot\,}$: desigualdade triangular reversa,
$\bigl|\abs{f(x)} - \abs{f(x_0)}\bigr| \leq \abs{f(x) -
f(x_0)}$, então o mesmo $\delta $ funciona. Para o clássico
funções que concedemos continuidade aqui; diferenciabilidade (comprovada em
\cref{ch:b1:derivative}) é mais forte.
\end{proof}

\begin{example}[Máximo e mínimo de funções contínuas]\label{ex:b1:continuity:maxmin}
Se $ f, g $ são contínuo, assim são $\max(f, g)$ e $\min(f, g)$:
nenhuma análise de caso necessária, graças às identidades
\[
\max(f, g) = \frac{f + g + \abs{f - g}}{2},
\qquad
\min(f, g) = \frac{f + g - \abs{f - g}}{2},
\]
e o continuidade de somas e de $\abs{\,\cdot\,}$
(\cref{prop:b1:continuity:algebra}). Em particular $ f^+ =
\max(f, 0)$ e $ f^- = \max(-f, 0)$ são contínuo com $ f =
f^+ - f^-$: a \omterm{thm:b1:logic:partition}{divisão} de sinal usada para séries
(\cref{ch:b1:series}) e, em grande escala, na integração
teoria do volume do Ano 3, não custa nada em regularidade.
\end{example}

\begin{theorem}[Teorema do valor intermediário]\label{thm:b1:continuity:ivt}
Deixar $ f $ ser contínuo sobre $\intcc{a}{b}$ com $ f(a) \leq 0 \leq
f(b)$. Então $ f(c) = 0$ para alguns $ c \in \intcc{a}{b}$.
Consequentemente, um contínuo funcionar em um \omterm{prop:b1:reals:intervals}{intervalo} leva todos os valores
entre quaisquer dois de seus valores: $ f(I)$ é um
\omterm{prop:b1:reals:intervals}{intervalo}.\index{teorema do valor intermediário}
\end{theorem}

\begin{proof}
Dicotomia. \omterm{def:b1:logic:sets}{Definir} $ a_0 = a $, $ b_0 = b $. Dado $\intcc{a_n}{b_n}$ com
$ f(a_n) \leq 0 \leq f(b_n)$, deixar $ m $ seja o ponto médio: se $ f(m) \leq
0$ manter $\intcc{m}{b_n}$, caso contrário, mantenha $\intcc{a_n}{m}$; o sinal
as condições persistem. As sequências $(a_n), (b_n)$ são adjacentes ($ b_n -
a_n = \frac{b-a}{2^n}$), com \omterm{def:b1:topology:closure}{limite} comum $ c $
(\cref{thm:b1:seq:adjacent}). Por continuidade e
\cref{thm:b1:seq:order}: $ f(c) = \lim f(a_n) \leq 0$ e $ f(c) =
\lim f(b_n) \geq 0$, então $ f(c) = 0$.

Para a consequência: valores dados $ f(u) < v < f(w)$, aplique o acima
para $ x \mapsto f(x) - v $ no segmento com extremidades $ u $ e $ w $;
por isso $ f(I)$ é convexo, ou seja,\ um \omterm{prop:b1:reals:intervals}{intervalo}
(\cref{prop:b1:reals:intervals}).
\end{proof}

\begin{example}[Uma equação, o protocolo completo]\label{ex:b1:continuity:protocol}
Resolver $\eu^x = 3 - x $ sobre $\R $: existência, singularidade,
localização. \omterm{def:b1:logic:sets}{Definir} $ g(x) = \eu^x + x - 3$, contínuo. Localização e
existência: $ g(0) = -2 < 0$ e $ g(1) = \eu - 2 > 0$, então o
teorema do valor intermediário planta uma solução em
$\intoo{0}{1}$. Singularidade: $ g $ é uma soma do estritamente
aumentando $\eu^x $ e $ x - 3$, portanto, aumentando estritamente em
$\R $; uma função estritamente monótona assume cada valor no máximo
uma vez, então a solução é única em todos $\R $ (não apenas no
\omterm{prop:b1:reals:intervals}{intervalo} sondado). O protocolo --- reorganizar para $ g = 0$, sinal
mudança pela existência, monotonicidade pela singularidade --- estabelece-se
a maioria das perguntas "quantas soluções" em três linhas, e
\cref{ch:b1:derivative}tabelas de variação
estendê-lo para não monótono $ g $ cortando $\R $ em monótono
filiais.
\end{example}

\begin{example}[Dicotomia como algoritmo]\label{ex:b1:continuity:dichotomy}
A prova de \cref{thm:b1:continuity:ivt} computa. Pegar $ f(x) =
x^3 + x - 1$: $ f(0) = -1 < 0 < 1 = f(1)$, então uma raiz está em
$\intoo{0}{1}$. Redução pela metade:
\[
f(0.5) = -0.375 < 0, \qquad
f(0.75) = 0.171875 > 0, \qquad
f(0.625) = -0.130859375 < 0 ,
\]
então a raiz é sucessivamente presa em $\intoo{0.5}{1}$, então
$\intoo{0.5}{0.75}$, então $\intoo{0.625}{0.75}$ (valor verdadeiro:
$ c \approx 0.6823$). Depois $ n $ passos o erro é no máximo
$\frac{b - a}{2^n}$: dez passos dão três casas decimais, vinte dão
seis. A conclusão final: o teorema do valor intermediário não é
apenas uma existência \omterm{def:b1:logic:statement}{declaração} --- sua prova de dicotomia é uma
algoritmo de localização de raiz garantido, embora lento, contra o qual o
método de Newton rápido, mas local, de \cref{ch:b1:derivative} deveria
ser medido.
\end{example}

\begin{theorem}[Teorema do valor extremo]\label{thm:b1:continuity:evt}
UM contínuo função em um segmento $\intcc{a}{b}$ é limitado e
atinge seus \omterm{def:b1:topology:closure}{limites}: há $ c, d \in \intcc{a}{b}$ com
\[
f(c) = \inf_{\intcc{a}{b}} f,
\qquad
f(d) = \sup_{\intcc{a}{b}} f .
\]
Combinado com \cref{thm:b1:continuity:ivt}: \emph{o contínuo
imagem de um segmento é um segmento} $\intcc{f(c)}{f(d)}$.
\index{teorema do valor extremo}
\end{theorem}

\begin{proof}
\emph{Limitada acima:} caso contrário, escolha $ u_n $ com $ f(u_n) \geq n $. Por
compacidade do segmento (\cref{thm:b1:topology:compact}), um
\omterm{def:b1:seq:subsequence}{subsequência} $ u_{\varphi(n)} \to x \in \intcc{a}{b}$; continuidade dá
$ f(u_{\varphi(n)}) \to f(x)$, mas $ f(u_{\varphi(n)}) \geq \varphi(n)
\to +\infty $: contradição.

\emph{Sup alcançado:} deixar $ M = \sup f $ e escolha $ v_n $ com $ f(v_n) >
M - \frac{1}{n+1}$ (\cref{prop:b1:reals:epsilon}). Extrair
$ v_{\varphi(n)} \to d \in \intcc{a}{b}$: então $ f(d) = \lim
f(v_{\varphi(n)}) = M $ pelo aperto. O \omterm{def:b1:reals:bounds}{ínfimo} é tratado por
$-f $.
\end{proof}

\begin{example}[Mínimo positivo em um segmento]\label{ex:b1:continuity:posmin}
Deixar $ f $ ser contínuo sobre $\intcc{0}{1}$ com $ f(x) > 0$ para
cada $ x $. Então $\inf f > 0$: pelo teorema do valor extremo o
\omterm{def:b1:reals:bounds}{ínfimo} é um \emph{valor} $ f(c)$, e $ f(c) > 0$ por hipótese.
Então um contínuo função positiva em um segmento é limitada
de $0$ --- um argumento de duas linhas usado uma dúzia de vezes no
próximos capítulos (denominadores sob controle, função passo
enquadramentos, \omterm{def:b1:topology:closure}{limites} de erro). Em um não compacto \omterm{prop:b1:reals:intervals}{intervalo} isso falha
espetacularmente: $ f(x) = x $ sobre $\intoc{0}{1}$ é contínuo e
positivo com $\inf f = 0$, não alcançado. A visão final:
``positivo'' é atualizado para ``uniformemente positivo'' exatamente quando o
o domínio é compacto; cada hipótese do teorema do valor extremo
é resistente.
\end{example}

\begin{remark}[Armadilhas comuns em torno dos três teoremas]
(eu) \emph{Contínuo imagens}: apenas \emph{segmentos} são robustos.
O contínuo imagem de um \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalo} não precisa ser \omterm{def:b1:topology:open}{abrir}
($ x \mapsto x^2$ \omterm{def:b1:logic:map}{mapas} $\intoo{-1}{1}$ para $\intco{0}{1}$), o
imagem de um conjunto \omterm{def:b1:topology:closed}{fechado} não precisa ser \omterm{def:b1:topology:closed}{fechado} ($\arctan $ \omterm{def:b1:logic:map}{mapas} o
\omterm{def:b1:topology:closed}{fechado} $\R $ para o \omterm{def:b1:topology:open}{abrir}
$\intoo{-\frac\pi2}{\frac\pi2}$); mas a imagem de um segmento é
um segmento (\cref{thm:b1:continuity:evt}). (ii) \emph{O
teorema do valor intermediário precisa de um \omterm{prop:b1:reals:intervals}{intervalo}}: a função $ x
\mapsto \frac1x $, contínuo sobre $\intco{-1}{0} \cup
\intoc{0}{1}$, assume os valores $-1$ e $1$ ainda assim nunca
desaparece --- seu domínio é composto por duas peças disjuntas e o valor
$0$ cai na lacuna; sempre nomeie o \omterm{prop:b1:reals:intervals}{intervalo} em que o
teorema é aplicado. (iii) \emph{Continuidade uniforme é uma propriedade do
par (função, \omterm{def:b1:logic:sets}{definir})}: $ x^2$ é uniformemente contínuo em cada
segmento ainda não ativado $\R $
(\cref{ex:b1:continuity:notuniform}) --- as palavras ``uniformemente
contínuo'' sem um domínio não têm sentido. (4)
\emph{Continuidade do inverso não é formal}: ele segura
\omterm{prop:b1:reals:intervals}{intervalos} via monotonicidade
(\cref{thm:b1:continuity:bijection}), mas um contínuo bijeção
entre sindicatos de \omterm{prop:b1:reals:intervals}{intervalos} pode ter um inverso descontínuo ---
a observação após esse teorema existe porque os alunos o citam
sem o \omterm{prop:b1:reals:intervals}{intervalo} hipótese.
\end{remark}

\section{Funções monótonas e funções inversas}

\begin{theorem}[Teorema da bijeção monótona]\label{thm:b1:continuity:bijection}
Deixar $ f $ ser contínuo e estritamente monótono em um \omterm{prop:b1:reals:intervals}{intervalo} $ I $. Então
\begin{enumerate}
  \item $ f $ é uma bijeção de $ I $ para o \omterm{prop:b1:reals:intervals}{intervalo} $ J = f(I)$;
  \item o inverso $ f^{-1} \colon J \to I $ é estritamente monotônico (em
        na mesma direção) e \emph{contínuo}.
\end{enumerate}
\end{theorem}

\begin{proof}
Dizer $ f $ está aumentando estritamente. (1) A injetividade é imediata de
monotonicidade estrita; sobrejetividade em $ f(I)$ é trivial e $ f(I)$
é um \omterm{prop:b1:reals:intervals}{intervalo} por \cref{thm:b1:continuity:ivt}.

(2) $ f^{-1}$ está aumentando estritamente: se $ y < y'$ em $ J $ mas $ f^{-1}(y)
\geq f^{-1}(y')$, aplicando o aumento $ f $ dá $ y \geq y'$,
absurdo. Continuidade de $ f^{-1}$ no $ y_0 = f(x_0) \in J $: deixar
$\varepsilon > 0$. Suponha primeiro $ x_0$ \omterm{def:b1:topology:closure}{interior} para $ I $ e
encolher $\varepsilon $ para que $ x_0 \pm \varepsilon \in I $: seu
imagens satisfazem $ f(x_0 - \varepsilon) < y_0 < f(x_0 +
\varepsilon)$. Pegar
$\delta = \min\bigl(y_0 - f(x_0 - \varepsilon),\, f(x_0 +
\varepsilon) - y_0\bigr) > 0$: para $\abs{y - y_0} \leq \delta $,
monotonicidade de $ f^{-1}$ aperta $ f^{-1}(y)$ entre $ x_0 -
\varepsilon $ e $ x_0 + \varepsilon $. Se $ x_0$ é, digamos, a esquerda
ponto final de $ I $, apenas $ x_0 + \varepsilon $ está disponível: então
$ y_0 = \min J $ (monotonicidade), cada $ y \in J $ com $ y - y_0
\leq f(x_0 + \varepsilon) - y_0$ satisfaz $ x_0 \leq f^{-1}(y)
\leq x_0 + \varepsilon $, e a estimativa unilateral é exatamente
continuidade em um ponto final; o ponto final direito é simétrico.
(Observação: continuidade de $ f^{-1}$
era \emph{não} deduzido daquele de $ f $ por simetria --- é
monotonicidade em um \omterm{prop:b1:reals:intervals}{intervalo} isso faz o trabalho.)
\end{proof}

\begin{remark}
Este teorema é o que \cref{def:b1:functions:arc} e
\cref{prop:b1:functions:invhyp} usado silenciosamente: $\arcsin $, $\arccos $,
$\arctan $, $\operatorname{arsinh}$, \dots\ são contínuo. Um
complemento (\cref{exo:b1:continuity:10}): um contínuo
\emph{\omterm{def:b1:logic:inj}{injetivo}} funcionar em um \omterm{prop:b1:reals:intervals}{intervalo} é automaticamente estritamente
monotônico, então a hipótese da monotonicidade não custa nada.
\end{remark}

\begin{example}[Raízes de todas as ordens]\label{ex:b1:continuity:nthroot}
Para $ n \in \N^*$, a função $ f(x) = x^n $ é contínuo e
aumentando estritamente em $\intco{0}{+\infty}$, com $ f(0) = 0$ e
$ f(x) \to +\infty $: sua imagem é $\intco{0}{+\infty}$
(\cref{thm:b1:continuity:ivt} para o \omterm{prop:b1:reals:intervals}{intervalo} estrutura). O
o teorema da bijeção monótona então fornece, de uma só vez, um
aumentando estritamente \emph{contínuo} inverso
\[
x \mapsto x^{1/n} \colon \intco{0}{+\infty} \to
\intco{0}{+\infty} :
\]
existência, singularidade e continuidade de $ n $-ésimas raízes, sem
computação. Compare com \cref{exo:b1:reals:12}, que construiu
$\sqrt y $ à mão do \omterm{def:b1:reals:bounds}{supremo}: um capítulo da teoria tem
comprimiu aquela página de trabalho em duas linhas, e as mesmas duas
linhas legitimadas $\arcsin $, $\arctan $ e
$\operatorname{arsinh}$ diante deles. O insight final: um bom
teorema é trabalho armazenado.
\end{example}

\section{Continuidade uniforme}

\begin{definition}\label{def:b1:continuity:uniform}
$ f \colon I \to \R $ é \emph{uniformemente contínua}\index{uniformemente contínua} quando
\[
\forall \varepsilon > 0,\ \exists \delta > 0,\ \forall x, y \in I,
\qquad \abs{x - y} \leq \delta \implies \abs{f(x) - f(y)} \leq
\varepsilon .
\]
O ponto: $\delta $ depende apenas de $\varepsilon $, não no
localização em $ I $. Uniforme continuidade implica continuidade; um
\emph{Lipschitz} função ($\abs{f(x) - f(y)} \leq k \abs{x - y}$)
é uniformemente contínuo ($\delta = \varepsilon/k $).
\end{definition}

\begin{example}\label{ex:b1:continuity:notuniform}
$ x \mapsto x^2$ é contínuo sobre $\R $ mas \emph{não} uniformemente
contínuo: $\abs{(n + \frac 1n)^2 - n^2} = 2 + \frac{1}{n^2} \geq 2$
embora os argumentos estejam à distância $\frac 1n \to 0$. Em qualquer
limitado \omterm{prop:b1:reals:intervals}{intervalo} é Lipschitz, portanto uniformemente contínuo ---
consistente com o teorema de Heine abaixo.
\end{example}

\begin{example}[Módulos uniformes, explicitamente]\label{ex:b1:continuity:moduli}
Em segmento, Heine garante uniforme $\delta $; muitas vezes pode-se
também \emph{calcular} isto. Para $ f(x) = x^2$ sobre $\intcc{0}{10}$:
\[
\abs{x^2 - y^2} = \abs{x + y}\,\abs{x - y} \leq 20\,\abs{x - y},
\]
então $\delta = \frac{\varepsilon}{20}$ funciona uniformemente (um
Lipschitz \omterm{def:b1:complex:field}{módulo}, linear em $\varepsilon $). Para $\sqrt x $ sobre
$\intcc{0}{1}$: $\abs{\sqrt x - \sqrt y} \leq \sqrt{\abs{x -
y}}$ (\cref{exo:b1:continuity:9}), então $\delta = \varepsilon^2$
funciona --- uniforme, mas \emph{não} linear: próximo $0$ a praça
a raiz é íngreme e o preço aparece no expoente de
$\varepsilon $, não em uma falha de uniformidade. O fechamento
visão: continuidade uniforme é um espectro, não um sim/não ---
a função $\delta(\varepsilon)$, chamado de \omterm{def:b1:complex:field}{módulo}, medidas
quão cara é a uniformidade, e Lipschitz é simplesmente seu
\omterm{def:b1:arith:prime}{melhor} nota.
\end{example}

\begin{theorem}[Heine]\label{thm:b1:continuity:heine}
UM contínuo funcionar em um \emph{segmento} $\intcc{a}{b}$ é uniformemente
contínuo.\index{Teorema de Heine}
\end{theorem}

\begin{proof}
Por contradição: suponha que algum $\varepsilon_0 > 0$ derrota todos
$\delta $. Com $\delta = \frac{1}{n+1}$, escolha $ x_n, y_n \in
\intcc{a}{b}$ com $\abs{x_n - y_n} \leq \frac{1}{n+1}$ e
$\abs{f(x_n) - f(y_n)} > \varepsilon_0$. Compacidade
(\cref{thm:b1:topology:compact}) extratos $ x_{\varphi(n)} \to c \in
\intcc{a}{b}$; então $ y_{\varphi(n)} \to c $ também (aperte $\abs{x -
y}$). Continuidade no $ c $ dá $ f(x_{\varphi(n)}) \to f(c)$ e
$ f(y_{\varphi(n)}) \to f(c)$, então $\abs{f(x_{\varphi(n)}) -
f(y_{\varphi(n)})} \to 0 < \varepsilon_0$: contradição.
\end{proof}

\begin{example}[Limitado, contínuo, mas não uniformemente]\label{ex:b1:continuity:sinx2}
A função $ f(x) = \sin(x^2)$ é contínuo e delimitado
$\R $, mas não uniformemente contínuo. Pegar
\[
x_n = \sqrt{2\pi n}, \qquad y_n = \sqrt{2\pi n + \tfrac\pi2}:
\qquad
y_n - x_n = \frac{\pi/2}{\sqrt{2\pi n + \frac\pi2} +
\sqrt{2\pi n}} \longrightarrow 0 ,
\]
ainda $ f(y_n) - f(x_n) = \sin\bigl(2\pi n + \frac\pi2\bigr) -
\sin(2\pi n) = 1 - 0 = 1$ para cada $ n $: nenhum único $\delta $ pode
servir $\varepsilon = \frac12$ em todos os lugares. Geometricamente, o
oscilações de $\sin(x^2)$ \emph{acelerar}: o gráfico
completa uma onda completa em janelas cada vez mais curtas, então o
escala horizontal um dado $\varepsilon $ requer redução para zero
como $ x $ cresce. O insight final: a limitação não compra
uniformidade (este exemplo), e a ilimitação não impede
isso ($\sqrt x $, \cref{exo:b1:continuity:9}); o que decide é o
\emph{\omterm{def:b1:complex:field}{módulo} de oscilação}, e o teorema de Heine diz compacto
domínios o disciplinam automaticamente.
\end{example}

\begin{example}[O experimento da calculadora, explicado]\label{ex:b1:continuity:cositeration}
Digite qualquer número em uma calculadora e pressione $\cos $ repetidamente:
o display trava em $0.7390851\dots $ Por que? Depois de pressionar o
o valor está em $\intcc{-1}{1}$, depois das duas $\intcc{\cos 1}{1}
\subseteq \intcc{0.54}{1}$, um estábulo \omterm{prop:b1:reals:intervals}{intervalo} para $\cos $. Nele,
$\abs{\cos a - \cos b} \leq \sin(1)\,\abs{a - b}$ com $\sin 1 =
0.841\dots < 1$ (o \omterm{def:b1:topology:closure}{limite} produto-soma de
\cref{pb:b1:seq:1}, ou a desigualdade do valor médio de
\cref{ch:b1:derivative}): os contratos de iteração, então pelo
etapa de controle de erros de \cref{met:b1:seq:recurrent},
\[
\abs{u_n - c} \leq (0.842)^{\,n-2}\,\abs{u_2 - c}
\longrightarrow 0 ,
\]
onde $ c $ é o único ponto fixo $\cos c = c $
(\cref{exo:b1:continuity:6}). Sobre $40$ prensas compram três
decimais ($0.842^{40} \approx 10^{-3}$) --- uma taxa geométrica,
mais suave que a dicotomia $2^{-n}$ por etapa, mas cada prensa custa
um toque de tecla enquanto cada etapa da dicotomia custa um sinal completo
avaliação. O insight final: a imagem de ponto fixo de
\cref{ch:b1:seq} e os teoremas de existência deste capítulo são
duas metades de uma história --- IVT descobre $ c $, a contração atinge
isso.
\end{example}

\begin{remark}[Onde esses teoremas funcionam a seguir]
Cada um dos três pilares deste capítulo fortalece um capítulo posterior. O
teorema do valor intermediário alimenta toda existência de soluções
argumento e teorema da bijeção monótona; o valor extremo
teorema transforma problemas de otimização em teoremas (Rolle e o
teorema do valor médio em \cref{ch:b1:derivative} começar exatamente
lá); O teorema de Heine é a razão contínuo funções ativadas
segmentos podem ser integrados em \cref{ch:b1:integration} --- o
uniforme $\delta $ é o que faz convergir as somas de Riemann. No ano
2 volume, o mesmo trio reaparece em espaços vetoriais normados, com
compactação fazendo o trabalho que os segmentos fazem aqui.
\end{remark}

\begin{remark}[Perspectivas dentro deste volume]
Continuidade está prestes a ser superado, mas nunca se aposentou.
\Cref{ch:b1:derivative} fortalece-o para a diferenciabilidade e
retorna o favor (diferenciável implica contínuo);
\cref{ch:b1:integration} repousa sobre ele duas vezes, através de Heine por
a construção e através do teorema fundamental, cujo
objeto central $ x \mapsto \int_a^x f $ atualiza um meramente
contínuo $ f $ para um $ C^1$ primitivo. Em
\cref{ch:b1:multivar}, continuidade em duas variáveis contém uma armadilha
vale a pena visualizar: a função $\frac{xy}{x^2 + y^2}$ (estendido
por $0$) é contínuo em $ x $ para cada fixo $ y $ e em $ y $ para
cada um fixo $ x $, ainda não contínuo na origem --- ao longo do
diagonal $ x = y $ é constantemente $\frac12$. Separar continuidade
é estritamente mais fraco do que continuidade: o sequencial
caracterização sobrevive à mudança para $\R^2$, mas as sequências devem
ser autorizado a aproximar-se de \emph{todo} direção, não só
ao longo dos eixos.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b1:continuity:1}
Usando a caracterização sequencial, prove que $ x \mapsto
\sin\frac 1x $ não tem \omterm{def:b1:topology:closure}{limite} em $0^+$ \emph{(exibir duas sequências)}.
Faz $ x \mapsto x \sin\frac 1x $ tem um?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:continuity:2}
Estude o continuidade sobre $\R $ de $ f(x) = \lfloor x \rfloor $ e de
$ g(x) = x - \lfloor x \rfloor $, e de $ h(x) = \lfloor x \rfloor +
(x - \lfloor x\rfloor)^2$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:continuity:3}
Prove que a equação $ x^5 - 3x + 1 = 0$ tem pelo menos três reais
soluções \emph{(avaliar em pontos bem escolhidos e aplicar
\cref{thm:b1:continuity:ivt} em três segmentos disjuntos)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:continuity:4}
Prove que cada polinomial de grau ímpar tem raiz real.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:continuity:5}
(Ponto fixo) Deixe $ f \colon \intcc{0}{1} \to \intcc{0}{1}$ ser
contínuo. Prove isso $ f $ tem um ponto fixo: $ f(c) = c $ para alguns
$ c $. Ilustre que nem continuidade nem o segmento pode ser
caiu.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:continuity:6}
Prove que a equação $\cos x = x $ tem exatamente uma solução real,
e que está em $\intoo{0}{1}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:continuity:7}
Deixar $ f \colon \R \to \R $ ser contínuo com $ f(x) \to +\infty $ como $ x
\to \pm\infty $. Prove isso $ f $ atinge um mínimo global em $\R $.
\emph{(Reduza para um segmento contendo um subnível \omterm{def:b1:logic:sets}{definir}.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:continuity:8}
Deixar $ f \colon \R \to \R $ ser contínuo e periódico (de período $ T >
0$). Prove isso $ f $ é limitado e atinge seus \omterm{def:b1:topology:closure}{limites}, e que não há
existe $ c $ com $ f(c + \frac T2) = f(c)$. \emph{(Para o segundo
apontar, estudar $ g(x) = f(x + \frac T2) - f(x)$ durante um período.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:continuity:9}
Prove isso $ x \mapsto \sqrt x $ é uniformemente contínuo sobre
$\intco{0}{+\infty}$, embora não seja Lipschitz perto $0$.
\emph{(Prove e use $\abs{\sqrt x - \sqrt y} \leq \sqrt{\abs{x -
y}}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:continuity:10}
Deixar $ f $ ser contínuo e \omterm{def:b1:logic:inj}{injetivo} em um \omterm{prop:b1:reals:intervals}{intervalo} $ I $. Prove isso
$ f $ é estritamente monotônico. \emph{Dica: se não, existem $ a < b < c $
com, digamos, $ f(b) > f(a)$ e $ f(b) > f(c)$; aplique o intermediário
teorema do valor para um valor entre $\max(f(a), f(c))$ e $ f(b)$ ligado
ambos os lados de $ b $.}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:continuity:11}
(\omterm{pb:b1:continuity:1}{Equação funcional} de Cauchy, contínuo caso) Deixe $ f \colon \R \to
\R $ ser contínuo com $ f(x + y) = f(x) + f(y)$ para todos $ x, y $.
Prove isso $ f(x) = f(1)\,x $ para todos $ x $: primeiro em $\N $, $\Z $, $\Q $
(somente por aditividade), então em $\R $ por continuidade e densidade
(\cref{thm:b1:reals:density}).
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:continuity:12}
Deixar $ f \colon \intco{0}{+\infty} \to \R $ ser contínuo com
$ f(x) \to \ell \in \R $ como $ x \to +\infty $. Prove isso $ f $ é
\emph{uniformemente} contínuo sobre $\intco{0}{+\infty}$. \emph{(Corte em um
grande $ A $: Heine em $\intcc{0}{A+1}$, o \omterm{def:b1:topology:closure}{limite} além $ A $; faça o
dois regimes se sobrepõem.)}
\end{exercise}

\section{Problema: equação funcional de Cauchy e suas irmãs}

\begin{problem}[{Problema de fim de semana --- $ f(x + y) = f(x) + f(y)$:
a regularidade força a linearidade e o retrato dos monstros}]
\label{pb:b1:continuity:1}
Quais funções satisfazem $ f(x + y) = f(x) + f(y)$ para tudo de verdade $ x,
y $? Cauchy levantou a questão em 1821; a resposta é um paradigma.
\Cref{exo:b1:continuity:11} mostra que tal \emph{aditivo}
a função é linear em $\Q $ e tão cheio continuidade forças $ f(x)
= cx $. Este problema aguça dramaticamente a hipótese ---
continuidade em um \emph{solteiro} ponto, ou monotonicidade, ou mera
limitação em um pequeno \omterm{prop:b1:reals:intervals}{intervalo}, cada um é suficiente --- então pinta
o retrato de uma solução hipotética não linear (seu gráfico
preenche o plano), resolve as equações irmãs que caracterizam
$\eu^{cx}$, $ c\ln x $, $ x^c $ e $ cx^2$, e fecha com Jensen
equação e o teorema \emph{ponto médio-convexo $+$ contínuo
$\implies $ convexo}.\index{Funcional de Cauchy
equação}\index{equação funcional} Por todo, \emph{aditivo}
significa: $ f(x + y) = f(x) + f(y)$ para todos $ x, y \in \R $.

\medskip
\noindent\textbf{Part I --- $\Q $-linearity, and one point of
continuidade.}
\begin{enumerate}
  \item Deixar $ f $ ser aditivo. De \cref{exo:b1:continuity:11},
        $ f(r) = f(1)\,r $ para racional $ r $. Prove o \omterm{def:b1:arith:prime}{melhor}
        \omterm{def:b1:logic:statement}{declaração} usado abaixo: para \emph{todo} $ x \in \R $ e $ r
        \in \Q $, $ f(rx) = r\,f(x)$ ($ f $ é
        $\Q $-\emph{linear}).
  \item Suponha que o aditivo $ f $ é contínuo em um único
        ponto $ x_0$. Mostre isso $ f $ é contínuo em todos os lugares
        \emph{(calcular $ f(x + h) - f(x)$ em termos de $ f(x_0 + h)
        - f(x_0)$)}, por isso $ f(x) = f(1)\,x $.
  \item Mostre que uma função aditiva é determinada por seu
        restrição a qualquer \omterm{def:b1:topology:dense}{denso} \omterm{def:b1:structures:subgroup}{subgrupo}: se dois aditivos
        funções concordam $\Z + \sqrt2\,\Z $ e ambos são
        contínuo, eles são iguais --- enquanto sem continuidade,
        prescrever $ f(1) = 0$ e $ f(\sqrt 2) = 1$ é consistente
        com $\Q $-linearidade no \omterm{def:b1:structures:subgroup}{subgrupo}. Calcular $ f(m +
        n\sqrt2)$ para esta prescrição.
  \item Deixar $ f $ ser aditivo e limitado acima por $ M $ em alguns
        \omterm{prop:b1:reals:intervals}{intervalo} $\intcc{a}{b}$ com $ a < b $. Mostre isso $ f $ é
        limitado acima em $\intcc{0}{\ell}$, $\ell = b - a $
        \emph{(traduzir por $ a $)}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The regularity ladder.}
\begin{enumerate}[resume]
  \item Pergunta de continuação 4: usando $ f(\ell - t) + f(t) =
        f(\ell)$, mostre que $ f $ também é limitado \emph{abaixo} sobre
        $\intcc{0}{\ell}$: $\abs f \leq C $ lá.
  \item Mostre isso $\abs{f(t)} \leq \frac{C}{n}$ para $ t \in
        \intcc{0}{\ell/n}$, e deduza que $ f $ é contínuo no
        $0$ \emph{(estranheza lida com o lado esquerdo)}, portanto
        em todos os lugares (questão 2): uma função aditiva limitada por
        um \omterm{prop:b1:reals:intervals}{intervalo} é linear.
  \item Deduza o caso monótono: uma função aditiva
        não decrescente em alguns $\intcc{a}{b}$ ($ a < b $) é $ f(x)
        = cx $ com $ c \geq 0$.
  \item Monte o \emph{escada de regularidade}: para um aditivo
        $ f $, os seguintes são equivalentes --- (a) $ f(x) = cx $;
        (b) $ f $ contínuo; (c) $ f $ contínuo em um ponto; (d)
        $ f $ monótono em alguns não degenerados \omterm{prop:b1:reals:intervals}{intervalo}; (e) $ f $
        limitado por algum não degenerado \omterm{prop:b1:reals:intervals}{intervalo}. Organize o
        implicações de modo que cada uma é trivial ou já
        comprovado.
  \item Verifique se as perguntas 4 a 6 consumiram apenas um \omterm{def:b1:topology:closure}{limite}
        \emph{acima}: uma função aditiva limitada acima em um
        não degenerado \omterm{prop:b1:reals:intervals}{intervalo} já é linear. Deduza o
        espelho \omterm{def:b1:logic:statement}{declaração} para um \omterm{def:b1:topology:closure}{limite} abaixo e registre o
        forma mais forte do degrau da escada (e) assim obtida.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Portrait of a monster.} Suponha
agora $ f $ é aditivo, mas \emph{não} linear.
\begin{enumerate}[resume]
  \item Mostre que existem reais diferentes de zero $ u, v $ com
        $\dfrac{f(u)}{u} \neq \dfrac{f(v)}{v}$, e que o
        vetores $(u, f(u))$ e $(v, f(v))$ abranger o avião (seu
        determinante $ u f(v) - v f(u)$ é diferente de zero).
  \item Mostre que o gráfico de $ f $ contém todos os pontos
        \[
        r\,(u, f(u)) + s\,(v, f(v)), \qquad r, s \in \Q ,
        \]
        e deduza que o gráfico é \emph{\omterm{def:b1:topology:dense}{denso} em $\R^2$}: para
        cada ponto $(x_0, y_0)$ do avião e cada
        $\varepsilon > 0$, alguns $(x, f(x))$ está dentro
        $\varepsilon $ disso \emph{(resolver o $2\times2$ verdadeiro
        sistema, então aproxime os coeficientes reais por
        racionais)}.
  \item Deduza da questão 11 o retrato completo: uma situação não linear
        função aditiva é ilimitada em todos os não degenerados
        \omterm{prop:b1:reals:intervals}{intervalo}, descontínuo em todos os pontos, monótono em nenhum
        \omterm{prop:b1:reals:intervals}{intervalo}, e sua imagem de qualquer \omterm{prop:b1:reals:intervals}{intervalo} é \omterm{def:b1:topology:dense}{denso} em
        $\R $. Reconcilie com a questão 8.
  \item Monstros existem --- em um \omterm{def:b1:topology:dense}{denso} \omterm{def:b1:structures:subgroup}{subgrupo}, construtivamente:
        ligado $ G = \Z + \sqrt2\,\Z $ \omterm{def:b1:logic:sets}{definir} $ f(m + n\sqrt2) = n $.
        Mostrar $ f $ é bem definido e aditivo em $ G $, e isso
        $ f $ é ilimitado em $ G \cap \intoo{0}{\varepsilon}$ para
        cada $\varepsilon > 0$ \emph{(para fixo $ N $, apenas
        finitos $ g = m + n\sqrt2 \in \intoo{0}{1}$ ter
        $\abs n \leq N $; ainda $ G \cap \intoo{0}{\varepsilon}$ é
        infinito)}. Explique em um parágrafo por que estender tal
        um $ f $ para todos $\R $ requer uma base de $\R $ como um
        $\Q $-espaço vetorial (um \emph{Base de Hamel}), cujo
        a existência é uma questão de axioma de escolha além deste
        volume.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The sister equations.} Todos
funções aqui são contínuo.
\begin{enumerate}[resume]
  \item Deixar $ f \colon \R \to \R $ ser contínuo, não de forma idêntica
        $0$, com $ f(x + y) = f(x)f(y)$. Mostrar $ f(x) =
        f\bigl(\frac x2\bigr)^2 \geq 0$, então $ f > 0$ em todos os lugares,
        então $ f(x) = \eu^{cx}$ para alguns $ c $: as exponenciais
        são exatamente os contínuo \omterm{def:b1:structures:morphism}{morfismos} de $(\R, +)$ para
        $(\R^*, \times)$.
  \item Deixar $ f \colon \intoo{0}{+\infty} \to \R $ ser contínuo
        com $ f(xy) = f(x) + f(y)$. Mostrar $ f(x) = c\ln x $
        \emph{(transporte por $\exp $)}.
  \item Deixar $ f \colon \intoo{0}{+\infty} \to \intoo{0}{+\infty}$
        ser contínuo com $ f(xy) = f(x)f(y)$. Mostrar $ f(x) = x^c $.
  \item Encontre tudo contínuo $ f \colon \R \to \R $ com
        \[
        f(x + y) = f(x) + f(y) + f(x)f(y) .
        \]
        \emph{(Estudar $ h = 1 + f $; tratar o caso degenerado
        separadamente.)}
  \item (Paralelogramo) Encontre tudo contínuo $ f \colon \R \to \R $
        com $ f(x + y) + f(x - y) = 2f(x) + 2f(y)$: mostrar $ f $ é
        mesmo, $ f(0) = 0$, $ f(nx) = n^2 f(x)$ por indução, então
        $ f(x) = f(1)\,x^2$. (Esta equação é a impressão digital de
        formas quadráticas --- a lei do paralelogramo que detecta,
        no volume do Ano 2, cujas normas vêm de um ambiente interno
        produto.)
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Jensen and midpoint convexity.}
\begin{enumerate}[resume]
  \item (Equação de Jensen) Deixe $ f \colon \R \to \R $ ser
        contínuo com $ f\bigl(\frac{x+y}{2}\bigr) = \frac{f(x)
        + f(y)}{2}$. Mostre isso $ g = f - f(0)$ satisfaz
        $ g\bigl(\frac x2\bigr) = \frac{g(x)}{2}$, deduza que
        $ g $ é aditivo e conclui $ f(x) = cx + d $.
  \item Suponha agora apenas o \emph{desigualdade}: $ f $ contínuo
        com
        \[
        f\Bigl(\frac{x + y}{2}\Bigr) \leq \frac{f(x) +
        f(y)}{2} \qquad (x, y \in \R).
        \]
        Prove por indução em $ n $ que para todos os pesos diádicos
        $\lambda = \frac{k}{2^n} \in \intcc{0}{1}$:
        \[
        f\bigl(\lambda x + (1 - \lambda)y\bigr) \leq \lambda
        f(x) + (1 - \lambda) f(y) .
        \]
  \item Estender por continuidade e densidade das diádicas
        (\cref{exo:b1:reals:8}) para cada $\lambda \in
        \intcc{0}{1}$: um contínuo função ponto médio-convexo
        satisfaz a desigualdade de convexidade completa (a noção
        estudado sistematicamente em \cref{ch:b1:derivative}).
  \item Mostrar continuidade não pode ser eliminado: um aditivo não linear
        $ f $ satisfaz o \emph{igualdade de ponto médio} da pergunta
        19 ainda não há desigualdade de convexidade em qualquer \omterm{prop:b1:reals:intervals}{intervalo} (pergunta
        12). Moral: a convexidade do ponto médio é um estágio contável
        propriedade (díádica), convexidade é contínua;
        continuidade é a ponte --- exatamente como nas Partes I--II.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Last variations and synthesis.}
\begin{enumerate}[resume]
  \item Encontre tudo contínuo $ f \colon \R \to \R $ com $ f(x + y)
        = f(x) + f(y) + xy $ \emph{(subtraia o particular
        solução $\frac{x^2}{2}$)}.
  \item Prove: se $ f \colon \R \to \R $ é contínuo e
        aditivo apenas em um \emph{\omterm{def:b1:topology:dense}{denso} \omterm{def:b1:structures:subgroup}{subgrupo}} $ G $ (ou seja,\
        $ f(g + g') = f(g) + f(g')$ para $ g, g' \in G $), então $ f $
        é aditivo em $\R $. Mais genericamente, dois contínuo
        funções que concordam com um \omterm{def:b1:topology:dense}{denso} subconjunto de $\R $ são
        igual.
  \item Síntese, uma frase cada: (i) indicar a regularidade
        escada da questão 8 de memória; (ii) explicar por que
        ``gráfico \omterm{def:b1:topology:dense}{denso} no avião'' é a imagem mental correta
        pela falta de regularidade; (iii) listar os cinco
        funções clássicas caracterizadas na Parte IV-V e na
        único método que pegou todos eles; (iv) nomeie os dois
        lugares onde densidade de $\Q $ (ou das diádicas) em $\R$
        levou o argumento, e o lugar onde ele não poderia
        (questão 13).
\end{enumerate}
\end{problem}
