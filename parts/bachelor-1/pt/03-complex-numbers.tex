\chapter{Números Complexos}\label{ch:b1:complex}

Números complexos foram encontrados no volume do Ensino Médio como uma solução computacional
dispositivo para equações quadráticas. Este capítulo os trata como um elemento central
objeto: a forma exponencial e suas consequências (de Moivre, $n $-º
raízes, raízes da unidade), a tradução sistemática entre $\C $ e
geometria plana e o uso de $\eu^{\iu\theta}$ como uma máquina para
provar identidades trigonométricas.

\section{O campo $\C $, módulo e conjugado}

\begin{definition}[O campo complexo]\label{def:b1:complex:field}
$\C = \{a + \iu b : a, b \in \R\}$ com a adição habitual e o
multiplicação determinada por $\iu^2 = -1$. Todo diferente de zero $ z = a + \iu
b $ tem um inverso: $ z^{-1} = \frac{a - \iu b}{a^2 + b^2}$ --- no
linguagem de \cref{ch:b1:structures}, $\C $ é um campo. Um escreve $ a =
\Re(z)$, $ b = \Im(z)$, $\conj{z} = a - \iu b $ (o
\emph{conjugado}\index{conjugado}) e $\abs{z} = \sqrt{a^2 + b^2}$
(o \emph{módulo}\index{módulo}).
\end{definition}

\begin{proposition}[Regras para conjugado e módulo]\label{prop:b1:complex:rules}
Para $ z, w \in \C $:
\begin{enumerate}
  \item $\conj{z + w} = \conj z + \conj w $, $\conj{zw} = \conj z\,
        \conj w $, $\conj{\conj z} = z $;
  \item $ z \conj z = \abs z^2$; $\;\Re(z) = \frac{z + \conj z}{2}$,
        $\Im(z) = \frac{z - \conj z}{2\iu}$;
  \item $\abs{zw} = \abs z\, \abs w $, e $\abs{z^{-1}} =
        \abs{z}^{-1}$ para $ z \neq 0$;
  \item (desigualdade triangular\index{desigualdade triangular}) $\abs{z + w}
        \leq \abs z + \abs w $, com igualdade se e somente se $ z $ e
        $ w $ mentir em um raio comum de $0$ ($ w = \lambda z $ ou
        $ z = \lambda w $ com $\lambda \geq 0$);
  \item (desigualdade triangular reversa) $\bigl|\abs z - \abs w\bigr|
        \leq \abs{z - w}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) e (2) são cálculos diretos em partes reais e imaginárias. (3):
$\abs{zw}^2 = zw\,\conj{zw} = z\conj z\, w \conj w = \abs z^2 \abs
w^2$, então tire raízes quadradas; aplicar para $ z \cdot z^{-1} = 1$ para o
inverso.

(4) Ambos os lados são não negativos, então compare os quadrados:
\[
\abs{z+w}^2 = (z+w)(\conj z + \conj w)
= \abs z^2 + \abs w^2 + 2\,\Re(z \conj w),
\]
e $\Re(z\conj w) \leq \abs{z \conj w} = \abs z \abs w $ dá
$\abs{z+w}^2 \leq (\abs z + \abs w)^2$. Forças de igualdade $\Re(z \conj
w) = \abs{z \conj w}$, ou seja\ $ z \conj w \in \R_+$; se $ w \neq 0$ isso
dá $ z = \frac{z\conj w}{\abs w^2}\, w = \lambda w $ com $\lambda
\geq 0$ (e o caso $ w = 0$ é trivial).

(5) $\abs z = \abs{(z - w) + w} \leq \abs{z-w} + \abs w $ dá $\abs z
- \abs w \leq \abs{z - w}$; trocar $ z $ e $ w $ para o outro sinal.
\end{proof}

\begin{example}[Um treino completo de módulo e argumento]\label{ex:b1:complex:workout}
Colocar $ w = \dfrac{3 + 4\iu}{1 - 2\iu}$ na forma algébrica e calcule
é \omterm{def:b1:complex:field}{módulo} duas vezes. Multiplicando por \omterm{def:b1:complex:field}{conjugado} do
denominador:
\[
w = \frac{(3 + 4\iu)(1 + 2\iu)}{(1 - 2\iu)(1 + 2\iu)}
= \frac{3 + 6\iu + 4\iu - 8}{1 + 4}
= \frac{-5 + 10\iu}{5} = -1 + 2\iu .
\]
Diretamente: $\abs w = \sqrt{1 + 4} = \sqrt5$. Através da regra do quociente
(\cref{prop:b1:complex:rules}~(3)): $\abs w = \frac{\abs{3 +
4\iu}}{\abs{1 - 2\iu}} = \frac5{\sqrt5} = \sqrt5$ --- mesma resposta,
nenhuma forma algébrica é necessária. A lição generaliza: \omterm{def:b1:complex:field}{módulos} e
argumentos viajam bem através de produtos e quocientes, reais e
partes imaginárias viajam bem através de somas. Escolha o
representação que corresponda às operações em questão e converta
somente quando forçado.
\end{example}

\begin{example}[Equações envolvendo o conjugado]\label{ex:b1:complex:conjeq}
Resolva em $\C $: $\;z + 2\conj z = 6 + 2\iu $. Uma mistura de equações $ z $
e $\conj z $ é \emph{não} polinômio em $ z $; o movimento confiável
é dividir em coordenadas reais. Com $ z = x + \iu y $:
\[
z + 2\conj z = 3x - \iu y ,
\]
então a equação é lida $3x = 6$ e $-y = 2$: a solução única
é $ z = 2 - 2\iu $. (Verificar: $(2 - 2\iu) + 2(2 + 2\iu) = 6 +
2\iu $.) Alternativamente, \omterm{def:b1:complex:field}{conjugado} toda a equação para obter
$\conj z + 2z = 6 - 2\iu $ e resolva o sistema linear no
incógnitas $ z, \conj z $ --- mesma resposta e um truque útil quando
os coeficientes são complexos. Equações em $ z $ e $\conj z $ são
realmente sistemas de duas equações reais; esperando ``grau $1$, um
solução'' é segura aqui, mas $ z\conj z = -1$ (sem solução) mostra
a intuição polinomial falha assim que os produtos aparecem.
\end{example}

\section{Forma exponencial}

\begin{definition}[Exponencial complexo de um argumento imaginário]\label{def:b1:complex:expi}
Para $\theta \in \R $ um define
\[
\eu^{\iu\theta} = \cos\theta + \iu \sin\theta .
\]
Todo $ z \neq 0$ pode ser escrito $ z = r\,\eu^{\iu\theta}$ com $ r =
\abs z > 0$; $\theta $ é um \emph{argumento}\index{argumento} de $ z $,
determinado até somar um múltiplo de $2\pi $. O valor em
$\intoc{-\pi}{\pi}$ é o \emph{argumento principal}, escrito
$\arg z $.
\end{definition}

\begin{example}[Primeiros valores e uma identidade famosa]\label{ex:b1:complex:eulerid}
Lendo a definição nos ângulos cardeais:
\[
\eu^{\iu\pi/2} = \iu, \qquad
\eu^{\iu\pi} = -1, \qquad
\eu^{2\iu\pi} = 1, \qquad
\eu^{\iu\pi/4} = \frac{\sqrt2}2\,(1 + \iu) .
\]
O segundo, reorganizado como $\eu^{\iu\pi} + 1 = 0$, é de Euler
célebre ligação de identidade $\eu $, $\iu $, $\pi $, $1$ e $0$; em
nesta fase do livro, é um desdobramento de definição, em vez de um
teorema e seu conteúdo real --- por que o exponencial analítico
função de \cref{ch:b1:functions}, estendido ao imaginário
argumentos, merece o mesmo nome --- é resolvido pelo poder
série de \cref{ch:b1:series}. Enquanto isso, a tela acima é
vale a pena memorizar como uma tabela de conversão: ela é usada silenciosamente a cada
vez que um argumento é lido em uma imagem.
\end{example}

\begin{theorem}[A equação funcional]\label{thm:b1:complex:funceq}
Para todos $\theta, \varphi \in \R $:
\[
\eu^{\iu\theta}\, \eu^{\iu\varphi} = \eu^{\iu(\theta + \varphi)},
\qquad
\abs{\eu^{\iu\theta}} = 1,
\qquad
\conj{\eu^{\iu\theta}} = \eu^{-\iu\theta} = (\eu^{\iu\theta})^{-1}.
\]
Consequentemente $\abs{zw} = \abs z \abs w $ carrega os argumentos também:
$\arg(zw) \equiv \arg z + \arg w \pmod{2\pi}$.
\end{theorem}

\begin{proof}
Expanda o produto e use as fórmulas de adição:
\[
(\cos\theta + \iu\sin\theta)(\cos\varphi + \iu\sin\varphi)
= (\cos\theta\cos\varphi - \sin\theta\sin\varphi)
+ \iu\,(\sin\theta\cos\varphi + \cos\theta\sin\varphi),
\]
o que é $\cos(\theta+\varphi) + \iu\sin(\theta+\varphi)$. O \omterm{def:b1:complex:field}{módulo}
é $\sqrt{\cos^2\theta + \sin^2\theta} = 1$, e o \omterm{def:b1:complex:field}{conjugado} fórmula
é a paridade do cosseno e do seno; ele inverte $\eu^{\iu\theta}$ desde
$\eu^{\iu\theta}\eu^{-\iu\theta} = \eu^0 = 1$.
\end{proof}

\begin{corollary}[fórmula de Moivre]\label{cor:b1:complex:demoivre}
Para $\theta \in \R $ e $ n \in \Z $:
$\;(\cos\theta + \iu\sin\theta)^n = \cos(n\theta) +
\iu\sin(n\theta)$.\index{fórmula de Moivre}
\end{corollary}

\begin{proof}
Para $ n \geq 0$, induzir. O caso $ n = 0$ lê $1 = 1$. Supondo
a fórmula para $ n $, a equação funcional
(\cref{thm:b1:complex:funceq}) dá
\[
\bigl(\eu^{\iu\theta}\bigr)^{n+1}
= \bigl(\eu^{\iu\theta}\bigr)^{n}\,\eu^{\iu\theta}
= \eu^{\iu n\theta}\,\eu^{\iu\theta}
= \eu^{\iu(n+1)\theta} ,
\]
qual é a fórmula na classificação $ n + 1$. Para $ n < 0$, escrever $ n = -m $
com $ m > 0$: desde $\eu^{\iu m\theta}$ tem inverso
$\eu^{-\iu m\theta}$ (\cref{thm:b1:complex:funceq} de novo),
\[
\bigl(\eu^{\iu\theta}\bigr)^{-m}
= \bigl(\eu^{\iu m\theta}\bigr)^{-1}
= \eu^{\iu(-m)\theta} . \qedhere
\]
\end{proof}

\begin{example}[Expandindo $\cos 3\theta $ por de Moivre]\label{ex:b1:complex:cos3}
Escrever $ c = \cos\theta $, $ s = \sin\theta $. De Moivre e o binômio
teorema dar
\[
\cos 3\theta + \iu \sin 3\theta = (c + \iu s)^3
= c^3 - 3cs^2 + \iu\,(3c^2 s - s^3),
\]
e identificando as partes reais e, em seguida, substituindo $ s^2 = 1 - c^2$:
\[
\cos 3\theta = c^3 - 3c(1 - c^2) = 4\cos^3\theta - 3\cos\theta .
\]
A parte imaginária rende $\sin 3\theta = 3\sin\theta -
4\sin^3\theta $ de graça: uma identidade complexa sempre carrega
\emph{dois} reais. Lido de trás para frente, a identidade em caixa é o
chave para a equação clássica da trissecção: construir
$\cos(\theta/3)$ de $\cos\theta $ significa resolver o cúbico $4x^3 -
3x = \cos\theta $, que é onde a álgebra de
\cref{ch:b1:poly} assume.
\end{example}

\begin{proposition}[Fórmulas de Euler]\label{prop:b1:complex:euler}
\[
\cos\theta = \frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2},
\qquad
\sin\theta = \frac{\eu^{\iu\theta} - \eu^{-\iu\theta}}{2\iu}.
\]
\end{proposition}

\begin{proof}
Adicionar, respectivamente subtrair, $\eu^{\iu\theta} = \cos\theta +
\iu\sin\theta $ e $\eu^{-\iu\theta} = \cos\theta - \iu\sin\theta $.
\end{proof}

\begin{method}[Trigonometria via exponenciais]\label{met:b1:complex:trig}
\begin{enumerate}
  \item \emph{Linearizar} $\cos^p\theta \sin^q\theta $ (transformar poderes em
        uma soma de $\cos k\theta $, $\sin k\theta $): substitua Euler
        fórmulas, expanda com o teorema binomial
        (\cref{thm:b1:counting:binomial}), reagrupar \omterm{def:b1:complex:field}{conjugado} termos.
  \item \emph{Expandir} $\cos n\theta $ como um polinômio em $\cos\theta $:
        escrever $\cos n\theta = \Re\bigl((\cos\theta +
        \iu\sin\theta)^n\bigr)$, expandir e converter até potências de
        $\sin $ através de $\sin^2 = 1 - \cos^2$.
  \item \emph{Soma de séries trigonométricas} como $\sum_k \cos k\theta $:
        reconhecer a parte real de uma soma geométrica $\sum_k
        (\eu^{\iu\theta})^k $.
  \item O \emph{fatoração de meio ângulo}: para todos $ p, q $,
        \[
        \eu^{\iu p} + \eu^{\iu q}
        = 2 \cos\tfrac{p - q}{2}\; \eu^{\iu \frac{p+q}{2}},
        \qquad
        \eu^{\iu p} - \eu^{\iu q}
        = 2\iu \sin\tfrac{p - q}{2}\; \eu^{\iu \frac{p+q}{2}} .
        \]
\end{enumerate}
\end{method}

\begin{example}[Linearização]\label{ex:b1:complex:linearize}
\[
\cos^3\theta
= \Bigl(\frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2}\Bigr)^{\!3}
= \frac{\eu^{3\iu\theta} + 3\eu^{\iu\theta} + 3\eu^{-\iu\theta} +
\eu^{-3\iu\theta}}{8}
= \frac{\cos 3\theta + 3\cos\theta}{4}.
\]
Esta forma integra imediatamente --- a razão pela qual a linearização
assuntos em \cref{ch:b1:integration}. Um produto misto funciona
da mesma forma, apenas com as duas fórmulas de Euler ao mesmo tempo:
\[
\sin^2\theta\cos^2\theta
= \Bigl(\frac{\sin2\theta}2\Bigr)^{\!2}
= \frac{1}{4}\cdot
\Bigl(\frac{\eu^{2\iu\theta} - \eu^{-2\iu\theta}}{2\iu}
\Bigr)^{\!2}
= \frac{2 - \eu^{4\iu\theta} - \eu^{-4\iu\theta}}{16}
= \frac{1 - \cos4\theta}{8} ,
\]
onde o atalho de ângulo duplo na primeira etapa salvou um
expansão binomial --- sempre vale a pena procurar antes
mecanização.
\end{example}

\begin{example}[Uma soma trigonométrica binomial]\label{ex:b1:complex:binomsum}
Para $ n \in \N $ e $\theta \in \R $, avaliar $ S =
\sum_{k=0}^{n}\binom nk \cos k\theta $. Reconheça a parte real de
uma expansão binomial:
\[
S = \Re\sum_{k=0}^n \binom nk \bigl(\eu^{\iu\theta}\bigr)^k
= \Re\bigl(1 + \eu^{\iu\theta}\bigr)^n ,
\]
então fatore o meio ângulo (\cref{met:b1:complex:trig}~(4)): $1 +
\eu^{\iu\theta} = 2\cos\frac\theta2\,\eu^{\iu\theta/2}$, então
\[
S = \Re\Bigl(2^n\cos^n\frac\theta2\;\eu^{\iu n\theta/2}\Bigr)
= 2^n \cos^n\frac\theta2\,\cos\frac{n\theta}2 .
\]
A parte imaginária rende $\sum_k\binom nk\sin k\theta =
2^n\cos^n\frac\theta2\sin\frac{n\theta}2$ de graça. Verificações de sanidade:
$\theta = 0$ recupera $\sum\binom nk = 2^n $, e $\theta = \pi $
dá $ S = 0$ para $ n \geq 1$ (cada fator $\cos\frac\pi2$
desaparece), ou seja,\ a soma das linhas alternadas de
\cref{ex:b1:counting:binomial}. O método --- ``veja o cosseno
soma como a sombra de uma potência complexa e então fatora meios-ângulos''
--- é exatamente o de \cref{exo:b1:complex:6}, com o binômio
teorema substituindo a série geométrica.
\end{example}

\section{Raízes de números complexos}

\begin{theorem}[$ n $-ésimas raízes]\label{thm:b1:complex:roots}
Deixar $ a = r\,\eu^{\iu\alpha} \neq 0$ e $ n \in \N^*$. A equação
$ z^n = a $ tem exatamente $ n $ soluções:
\[
z_k = r^{1/n}\, \eu^{\iu\left(\frac{\alpha}{n} +
\frac{2k\pi}{n}\right)},
\qquad k = 0, 1, \dots, n - 1 .
\]
\end{theorem}

\begin{proof}
Escrever $ z = \rho\,\eu^{\iu\theta}$ ($\rho > 0$). Então $ z^n =
\rho^n \eu^{\iu n\theta} = r \eu^{\iu\alpha}$ se e somente se $\rho^n
= r $ (\omterm{def:b1:complex:field}{módulos}) e $ n\theta \equiv \alpha \pmod{2\pi}$ (argumentos),
ou seja\ $\rho = r^{1/n}$ e $\theta = \frac{\alpha}{n} +
\frac{2k\pi}{n}$ para alguns $ k \in \Z $. Resta ver quando dois
inteiros $ k, k'$ produz o mesmo número: isso acontece exatamente quando
os ângulos diferem por um múltiplo de $2\pi $,
\[
\frac{2k\pi}n - \frac{2k'\pi}n \in 2\pi\Z
\iff \frac{k - k'}n \in \Z
\iff n \mid k - k' .
\]
Pela \omterm{thm:b1:logic:partition}{divisão} euclidiana, cada $ k \in \Z $ é mod congruente $ n $ para
exatamente um elemento de $\{0, 1, \dots, n-1\}$, então esse intervalo lista
cada solução uma vez e a contagem é exatamente $ n $. (O mesmo
argumento, corra para dentro $\mathbb U_n $, mostra que as raízes formam um regular
$ n $-gon: valores consecutivos de $ k $ girar pelo ângulo fixo
$\frac{2\pi}n $.)
\end{proof}

\begin{example}[Raízes cúbicas de $-27$]\label{ex:b1:complex:cuberoots}
Resolver $ z^3 = -27$. Forma exponencial do lado direito: $-27 =
27\,\eu^{\iu\pi}$, então as três raízes são
\[
z_k = 3\,\eu^{\iu(\frac\pi3 + \frac{2k\pi}3)}, \quad k = 0, 1, 2 :
\qquad
z_0 = 3\eu^{\iu\pi/3} = \frac32 + \frac{3\sqrt3}2\,\iu,
\quad
z_1 = -3,
\quad
z_2 = \conj{z_0} .
\]
Dois cheques. Primeiro, a verdadeira raiz $-3$ é o óbvio, e o
outros dois são suas rotações por $\pm\frac{2\pi}3$ --- equivalentemente
$-3j $ e $-3j^2$. Em segundo lugar, a álgebra confirma: $ z^3 + 27 = (z +
3)(z^2 - 3z + 9)$, e a quadrática tem discriminante $9 - 36 =
-27 < 0$ com raízes $\frac{3 \pm 3\iu\sqrt3}2 = z_0, \conj{z_0}$.
O insight: para lados direitos reais, as raízes não reais sempre
entre \omterm{def:b1:complex:field}{conjugado} pares, então uma imagem da solução \omterm{def:b1:logic:sets}{definir} é
simétrico em relação ao eixo real --- uma prévia do real
teorema da fatoração de \cref{ch:b1:poly}.
\end{example}

\begin{example}[Um lado direito não real]\label{ex:b1:complex:fourthroots}
Resolver $ z^4 = -8 + 8\iu\sqrt3$. Forma exponencial do lado direito:
\omterm{def:b1:complex:field}{módulo} $\sqrt{64 + 192} = 16$, argumento $\theta $ com
$\cos\theta = -\frac12$, $\sin\theta = \frac{\sqrt3}2$, ou seja\
$\theta = \frac{2\pi}3$. As quatro raízes são
\[
z_k = 2\,\eu^{\iu(\frac\pi6 + \frac{k\pi}2)}, \quad k = 0, 1, 2,
3 :
\qquad
z_0 = \sqrt3 + \iu,\quad z_1 = \iu z_0 = -1 + \iu\sqrt3,
\]
\[
z_2 = -z_0 = -\sqrt3 - \iu,\qquad z_3 = -\iu z_0 = 1 - \iu\sqrt3 .
\]
(Verificar: $ z_0^2 = 2 + 2\iu\sqrt3$, então $ z_0^4 = (2 +
2\iu\sqrt3)^2 = 4 - 12 + 8\iu\sqrt3 = -8 + 8\iu\sqrt3$.) Uma vez
\emph{um} raiz é encontrada, as outras três se libertam: elas são a sua
rotações sucessivas por $\frac\pi2$, ou seja, seus produtos com o
quarto raízes da unidade --- a estrutura geral por trás
\cref{thm:b1:complex:roots}, vale a pena explorar antes de recalcular
cada raiz do zero. Não \omterm{def:b1:complex:field}{conjugado} simetria desta vez: o
lado direito não é real.
\end{example}

\begin{definition}[Raízes da unidade]\label{def:b1:complex:unity}
O \emph{$ n $-as raízes da unidade}\index{$ n $-as raízes da unidade} são os
soluções de $ z^n = 1$:
\[
\mathbb{U}_n = \bigl\{\, \omega^k : k = 0, \dots, n-1 \,\bigr\},
\qquad \omega = \eu^{2\iu\pi/n}.
\]
Eles formam um grupo sob multiplicação (\cref{ch:b1:structures}) e
sente-se nos vértices de um regular $ n $-gon inscrito no círculo unitário.
\end{definition}

\begin{proposition}[Soma das raízes da unidade]\label{prop:b1:complex:sumroots}
Para $ n \geq 2$, o $ n $-º raízes da unidade soma a zero:
$\sum_{k=0}^{n-1} \omega^k = 0$.
\end{proposition}

\begin{proof}
Soma geométrica com proporção $\omega \neq 1$:
$\sum_{k=0}^{n-1} \omega^k = \frac{\omega^n - 1}{\omega - 1} = 0$
desde $\omega^n = 1$.
\end{proof}

\begin{example}[Lendo partes reais e imaginárias]\label{ex:b1:complex:sumrootsparts}
\omterm{thm:b1:logic:partition}{Divisão} $\sum_{k=0}^{n-1}\omega^k = 0$ em real e imaginário
parts produz duas identidades trigonométricas gratuitamente:
\[
\sum_{k=0}^{n-1}\cos\frac{2k\pi}n = 0,
\qquad
\sum_{k=0}^{n-1}\sin\frac{2k\pi}n = 0
\qquad (n \geq 2).
\]
Geometricamente: o centróide de um regular $ n $-gon inscrito no
o círculo unitário é seu centro --- os vértices se equilibram exatamente.
Para $ n = 5$ a primeira identidade dá $1 + 2\cos\frac{2\pi}5 +
2\cos\frac{4\pi}5 = 0$ (emparelhamento $ k $ com $ n - k $), o início
ponto de \cref{exo:b1:complex:8}cálculo de
$\cos\frac{2\pi}5$.
\end{example}

\begin{example}[Raízes quadradas em forma algébrica]\label{ex:b1:complex:sqrt}
Para resolver $ z^2 = 3 + 4\iu $ sem trigonometria, defina $ z = x + \iu y $:
\[
x^2 - y^2 = 3, \qquad 2xy = 4, \qquad x^2 + y^2 = \abs{3 + 4\iu} = 5 .
\]
Adicionando o primeiro e o último: $ x^2 = 4$, então $ x = \pm 2$, então $ y = 2/x =
\pm 1$ com o \emph{mesmo} emparelhamento de sinal ($ xy = 2 > 0$): $ z = \pm(2 +
\iu)$. Combinado com a fórmula usual, isso resolve todos os quadráticos
equação com coeficientes complexos
(\cref{exo:b1:complex:7}).
\end{example}

\section{Números complexos e geometria plana}

\begin{proposition}[Dicionário geométrico]\label{prop:b1:complex:geometry}
Identifique o ponto $ M(x, y)$ do avião com seu \emph{afixo} $ z = x
+ \iu y $\index{afixo}. Para pontos distintos $ A, B, C $ de afixos $ a, b,
c $:
\begin{enumerate}
  \item $\abs{b - a}$ é a distância $ AB $;
  \item $\arg\dfrac{c - a}{b - a}$ é o ângulo entre os vetores
        $\vect{AB}$ e $\vect{AC}$ (modelo $2\pi $);
  \item $ A, B, C $ estão alinhados se e somente se $\dfrac{c - a}{b - a} \in
        \R $; as linhas $ AB $ e $ AC $ são perpendiculares se e somente se
        $\dfrac{c - a}{b - a} \in \iu\R $.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) é a definição do \omterm{def:b1:complex:field}{módulo} aplicado a $ b - a $, o afixo de
$\vect{AB}$. (2): escrever $ b - a = r\eu^{\iu\theta}$, $ c - a =
s\eu^{\iu\varphi}$; então $\frac{c-a}{b-a} = \frac sr
\eu^{\iu(\varphi - \theta)}$ tem argumento $\varphi - \theta $, o
ângulo de $\vect{AB}$ para $\vect{AC}$. (3): alinhamento significa ângulo $0$
ou $\pi $, ou seja,\ argumento em $\pi\Z $, ou seja,\ o quociente é real;
perpendicularidade significa ângulo $\pm\frac\pi2$, ou seja,\ o quociente é
puramente imaginário. (O quociente é diferente de zero, pois $ C \neq A $.)
\end{proof}

\begin{remark}[Interlúdio: $\C $ é o plano com uma multiplicação]\label{rem:b1:complex:interlude}
Vale a pena fazer uma pausa sobre o que torna este capítulo possível:
o avião $\R^2$ carrega acréscimos em todas as direções, mas não
Multiplicação dada por Deus --- e $\C $ \emph{é} o avião
equipado com um, em que multiplicar por um número fixo
gira e escala. Esta estrutura será ordenhada mais três
vezes neste volume. Em \cref{ch:b1:matrices}, multiplicação por
$ a + \iu b $ reaparece como o $2 \times 2$ matriz com linhas $(a,
-b)$ e $(b, a)$: aritmética complexa é a primeira, inteiramente
família concreta de produtos matriciais. Em \cref{ch:b1:euclid}, o
fórmula $\Re(\conj z\,w)$ acaba sendo o produto escalar, e
$\abs z $ a norma euclidiana: a desigualdade triangular provada aqui
é o modelo para a história de Cauchy-Schwarz ali. E em
\cref{ch:b1:curves}, um ponto móvel é melhor escrito $ t \mapsto
z(t)$, de modo que a velocidade e a aceleração se tornam de valor complexo
derivadas --- movimento circular, por exemplo, é simplesmente $ z(t) =
R\,\eu^{\iu\omega t}$. Uma boa multiplicação, quatro capítulos de
dividendos.
\end{remark}

\begin{proposition}[Mapas $ z \mapsto az + b $]\label{prop:b1:complex:similitude}
Deixar $ a \in \C^*$, $ b \in \C $. A transformação $ f(z) = az + b $ de
o avião:
\begin{itemize}
  \item é uma tradução se $ a = 1$;
  \item caso contrário, tem um ponto fixo único $\zeta = \frac{b}{1-a}$,
        e $ f(z) - \zeta = a\,(z - \zeta)$: $ f $ é a rotação de
        centro $\zeta $ e ângulo $\arg a $, composto com a escala
        do centro $\zeta $ e proporção $\abs a $.
\end{itemize}
Em particular $ z \mapsto \eu^{\iu\theta} z $ é a rotação do ângulo
$\theta $ sobre a origem e $\conj z $ é o reflexo no real
eixo.
\end{proposition}

\begin{proof}
Se $ a = 1$, $ f(z) = z + b $ traduz pelo vetor de afixo $ b $. Se
$ a \neq 1$, a equação de ponto fixo $ z = az + b $ tem o único
solução $\zeta = \frac{b}{1-a}$, e então $ f(z) - \zeta = az + b -
(a\zeta + b) = a(z - \zeta)$. Escrita $ a = \abs a\, \eu^{\iu\arg a}$,
multiplicação por $ a $ dimensiona distâncias para $\zeta $ por $\abs a $ e adiciona
$\arg a $ para ângulos em $\zeta $, que é a composição anunciada.
\end{proof}

\begin{example}[Classificando um mapa $ z \mapsto az+b $]\label{ex:b1:complex:classify}
Pegar $ f(z) = \iu z + 1$. Aqui $ a = \iu \neq 1$: o ponto fixo é
\[
\zeta = \frac{b}{1 - a} = \frac1{1 - \iu}
= \frac{1 + \iu}{2},
\]
e desde então $\abs a = 1$, não há escala alguma: $ f $ é o puro
rotação do centro $\frac{1+\iu}2$ e ângulo $\arg \iu = \frac\pi2$.
Verificar: $ f(\zeta) = \iu\,\frac{1+\iu}2 + 1 = \frac{\iu - 1}2 + 1 =
\frac{1 + \iu}2 = \zeta $, e $ f(0) = 1$, $ f(1) = 1 + \iu $, $ f(1 +
\iu) = \iu\,(1 + \iu) + 1 = \iu $: os quatro pontos $0, 1, 1+\iu, \iu $
do ciclo quadrado unitário em torno de seu centro $\zeta $, um quarto
gire de cada vez --- exatamente que rotação $\frac\pi2$ sobre o
o centro do quadrado deve servir.
\end{example}

\begin{remark}[Armadilhas comuns com módulos e argumentos]\label{rem:b1:complex:pitfalls}
\begin{enumerate}
  \item \emph{Não há desigualdades $\C $.} Escrita $ z \leq w $ para
        números não reais não têm sentido; apenas \omterm{def:b1:complex:field}{módulos}, peças reais
        e partes imaginárias podem ser comparadas.
  \item \emph{$\sqrt{\phantom z}$ está reservado para $\R_+$.} Cada
        número complexo diferente de zero tem \emph{dois} raízes quadradas e
        nenhum deles é privilegiado: escreva ``deixe $\delta $ ser um quadrado
        raiz de $\Delta $'' (calculado como em
        \cref{ex:b1:complex:sqrt}), nunca $\sqrt\Delta $ ---
        a regra $\sqrt{ab} = \sqrt a\sqrt b $ já falha em
        $ a = b = -1$.
  \item \emph{Argumentos \omterm{def:b1:complex:field}{módulo} ao vivo $2\pi $.} De
        $\eu^{\iu\alpha} = \eu^{\iu\beta}$ concluir $\alpha
        \equiv \beta \pmod{2\pi}$, não $\alpha = \beta $;
        esquecendo o $2k\pi $ é como a solução conjuntos de $ z^n = a $
        perder $ n - 1$ dos seus $ n $ elementos.
  \item \emph{$\abs{z + w}$ não é $\abs z + \abs w $.} Igualdade
        na desigualdade triangular é o alinhamento excepcional
        caso (\cref{prop:b1:complex:rules}~(4)); em geral o
        \omterm{def:b1:complex:field}{módulo} de uma soma deve ser estimada, não computada.
\end{enumerate}
\end{remark}

\begin{example}\label{ex:b1:complex:equilateral}
$ A, B, C $ (afixos $ a, b, c $, pares distintos) formam um equilátero
triângulo com os vértices em ordem direta (sentido anti-horário) se e
somente se $\frac{c - a}{b - a} = \eu^{\iu\pi/3}$: a rotação de
centro $ A $ e ângulo $\frac\pi3$ envia $ B $ para $ C $. Ambas as orientações
juntos são capturados pela equação simétrica $ a^2 + b^2 + c^2 = ab
+ bc + ca $ (\cref{exo:b1:complex:10}).
\end{example}

\begin{remark}[Onde este capítulo é usado]\label{rem:b1:complex:whereused}
A forma exponencial é o dispositivo computacional mais reutilizado
do volume. Raízes da unidade tornar-se o exemplo padrão de um
grupo cíclico em \cref{ch:b1:structures} e conduza a fatoração
de $ X^n - 1$ em \cref{ch:b1:poly}; agrupamento \omterm{def:b1:complex:field}{conjugado} raízes lá
produz as fatorações reais usadas por frações parciais em
\cref{ch:b1:fractions}. Linearização (\cref{met:b1:complex:trig})
é a preparação padrão para integrar potências trigonométricas em
\cref{ch:b1:integration}, e as equações características de
\cref{ch:b1:diffeq} têm raízes complexas cujas reais e imaginárias
partes produzem as soluções oscilantes $\eu^{\lambda t}\cos\omega
t $. O dicionário geométrico retorna em roupas de matriz: rotações
e semelhanças tornam-se as matrizes ortogonais de
\cref{ch:b1:matrices,ch:b1:euclid}e curvas parametrizadas em
\cref{ch:b1:curves} geralmente são melhor escritos como \omterm{def:b1:logic:map}{mapas} $ t \mapsto z(t)
\in \C $.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b1:complex:1}
Coloque na forma exponencial: $1 + \iu $; $\;\sqrt 3 - \iu $; $\;-5$;
$\;\dfrac{1 + \iu}{\sqrt 3 - \iu}$. Deduzir
$\cos\frac{5\pi}{12}$ e $\sin\frac{5\pi}{12}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:complex:2}
Calcular $(1 + \iu)^{20}$. Para qual $ n \in \N $ é $(1 + \iu)^n $ um
número real?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:complex:3}
Descreva geometricamente o \omterm{def:b1:logic:sets}{definir} de $ z \in \C $ tal que:
$\abs{z - 2} = \abs{z + \iu}$; $\;\abs{z - 1} = 2$;
$\;\dfrac{z - 1}{z + 1} \in \iu\R $ (para $ z \neq -1$).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:complex:4}
Linearizar $\sin^4\theta $ e expandir $\cos 4\theta $ como um polinômio em
$\cos\theta $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:complex:5}
Resolver $ z^3 = 8\iu $ e coloque as soluções em uma imagem. Resolver
$ z^4 = -4$ e fator $ X^4 + 4$ em dois polinômios quadráticos reais.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:complex:6}
Para $\theta \in \R $ com $\eu^{\iu\theta} \neq 1$ e $ n \in \N $,
calcular
\[
C_n = \sum_{k=0}^{n} \cos k\theta
\qquad\text{and}\qquad
S_n = \sum_{k=0}^{n} \sin k\theta
\]
somando a série geométrica $\sum_k \eu^{\iu k\theta}$ e usando
a fatoração de meio ângulo.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:complex:7}
Resolva em $\C $: $ z^2 - (3 + 4\iu) z + (-1 + 5\iu) = 0$. \emph{(Calcular
o discriminante, extraia suas raízes quadradas como em
\cref{ex:b1:complex:sqrt}.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:complex:8}
Deixar $\omega = \eu^{2\iu\pi/5}$.
\begin{enumerate}
  \item Justificar $1 + \omega + \omega^2 + \omega^3 + \omega^4 = 0$.
  \item \omterm{def:b1:logic:sets}{Definir} $ u = \omega + \omega^4$ e $ v = \omega^2 + \omega^3$.
        Calcular $ u + v $ e $ uv $, e deduza que $ u $ e $ v $ são os
        raízes de $ X^2 + X - 1$.
  \item Conclua isso $\cos\frac{2\pi}{5} = \frac{\sqrt 5 - 1}{4}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:complex:9}
Prove isso para todos $ z, w \in \C $ (identidade do paralelogramo):
\[
\abs{z + w}^2 + \abs{z - w}^2 = 2\abs z^2 + 2\abs w^2 ,
\]
e interpretá-lo geometricamente no paralelogramo com vértices $0,
z, w, z + w $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:complex:10}
Prove que três pontos de afixos distintos entre pares $ a, b, c $ formar um
triângulo equilátero (qualquer orientação) se e somente se
\[
a^2 + b^2 + c^2 = ab + bc + ca .
\]
\emph{Dica: o caso direto é $\frac{c-a}{b-a} = -j^2$ e o
caso indireto $\frac{c-a}{b-a} = -j $, onde $ j = \eu^{2\iu\pi/3}$
satisfaz $ j^2 + j + 1 = 0$; ou fator $ a + jb + j^2c $ e $ a + j^2 b
+ jc $.}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:complex:11}
Para $ n \in \N^*$, avaliar $ P = \prod_{k=1}^{n-1}
\bigl(1 - \omega^k\bigr)$ onde $\omega = \eu^{2\iu\pi/n}$.
\emph{Dica: $ X^n - 1 = \prod_{k=0}^{n-1} (X - \omega^k)$; dividir por
$ X - 1$ e avaliar em $ X = 1$.} Deduzir
$\prod_{k=1}^{n-1} \sin\frac{k\pi}{n} = \dfrac{n}{2^{n-1}}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:complex:12}
Deixar $ n \geq 2$. Resolva a equação $(z + 1)^n = (z - 1)^n $ em $\C $:
mostre que tem exatamente $ n - 1$ soluções, todas puramente imaginárias,
a saber
\[
z_k = -\iu\,\frac{\cos\frac{k\pi}{n}}{\sin\frac{k\pi}{n}},
\qquad k = 1, \dots, n - 1 .
\]
\emph{Dica: $ z = 1$ não é uma solução, então divida e use o raízes de
unidade; em seguida, aplique a fatoração de meio ângulo de
\cref{met:b1:complex:trig}.}
\end{exercise}

\section{Problema: teorema de Napoleão}

\begin{problem}\label{pb:b1:complex:1}
Erga um triângulo equilátero para fora de cada lado de um
\emph{arbitrário} triângulo: os três centros desses triângulos
sempre formam um triângulo equilátero. Este é o teorema de Napoleão
\index{Teorema de Napoleão} --- um \omterm{def:b1:logic:statement}{declaração} sem nenhuma razão visível para
ser verdade, que a álgebra de $ j = \eu^{2\iu\pi/3}$ prova em três
linhas de computação. Este problema constrói o kit de ferramentas completo
(rotações, semelhanças diretas, o $ j $-critério de
\cref{exo:b1:complex:10}), prova o teorema de Napoleão interno e
externo, localiza a caixa degenerada e fecha com uma segunda joia
da mesma escola: teorema de van Schooten sobre triângulos equiláteros
e a desigualdade de Ptolomeu. Ao longo do tempo, os pontos do plano são
identificados com seus afixos.

\medskip
\noindent\textbf{Part I --- The number $ j $ and rotations.}
\begin{enumerate}
  \item Calcule a forma algébrica de $ j $, então $ j^2$, $ j^3$, $1 + j
        + j^2$, $\conj j $ e $ j^{-1}$. Lugar $1$, $ j $, $ j^2$ em um
        esboço do círculo unitário.
  \item Mostre que a rotação do centro $ a $ e ângulo $\theta $ é
        $ r(z) = a + \eu^{\iu\theta}(z - a)$, e que o
        composição de duas rotações, de ângulos $\theta $ e
        $\theta'$, é uma rotação do ângulo $\theta + \theta'$ se
        $\theta + \theta' \notin 2\pi\Z $, e uma tradução
        caso contrário. (Usar \cref{prop:b1:complex:similitude}.)
  \item Deixar $ b \neq c $. Mostre que existem exatamente dois pontos $ p $
        fazendo $(b, c, p)$ equilátero, nomeadamente $ p = b +
        \eu^{\pm\iu\pi/3}(c - b)$. Por um \emph{direto} triângulo
        $(a, b, c)$, verifique no exemplo $ a = 0$, $ b = 1$, $ c =
        \iu $ que a escolha $\eu^{-\iu\pi/3}$ é aquele que está deitado
        o outro lado da linha $ BC $ de $ a $ --- o
        \emph{para fora} ápice.
  \item Da solução de \cref{exo:b1:complex:10}, $(a, b, c)$
        é equilátero direto se $ a + jb + j^2c = 0$. Prove os dois
        identidades de rotação
        \[
        b + jc + j^2a = j^2\,(a + jb + j^2c),
        \qquad
        c + ja + j^2b = j\,(a + jb + j^2c),
        \]
        e deduzir que o critério é invariante sob condições cíclicas
        \omterm{def:b1:counting:objects}{permutação} de $(a, b, c)$.
  \item Mostre que se $ a + jb + j^2c = 0$ e dois dos três pontos
        coincidem, então todos os três coincidem. (Portanto, o critério exatamente
        caracteriza: triângulo equilátero reto, ou um único
        ponto.)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Direct similarities.}
\begin{enumerate}[resume]
  \item Mostre que o \omterm{def:b1:logic:map}{mapas} $ f(z) = az + b $ com $ a \in \C^*$
        (\emph{semelhanças diretas}) são estáveis sob composição
        e inversão: a composição de dois deles, e o
        inverso de qualquer um deles, é novamente desta forma. (No
        linguagem de \cref{ch:b1:structures}: eles formam um grupo.)
  \item Dado $ z_1 \neq z_2$ e $ w_1 \neq w_2$, mostre que há um
        \emph{exclusivo} semelhança direta $ f $ com $ f(z_1) = w_1$ e
        $ f(z_2) = w_2$, e dar $ a $ e $ b $ explicitamente.
  \item Mostre que uma semelhança direta $ f(z) = az + b $ multiplica
        todas as distâncias por $\abs a $ e preserva o \emph{forma}
        $\frac{c - a'}{b' - a'}$ de qualquer triângulo $(a', b', c')$
        \emph{(numerador e denominador são ambos multiplicados por
        $ a $)}. Deduza que dois triângulos são diretamente semelhantes
        exatamente quando suas formas são iguais.
  \item Determine completamente a semelhança direta com $ f(0) = 1$
        e $ f(1) = \iu $: dar $ a $, $ b $, o ponto fixo, a razão
        e o ângulo.
  \item Deixar $\alpha + \beta = 1$. Mostre que o ``ponto ponderado''
        $ g(a', b') = \alpha a' + \beta b'$ viaja com cada
        semelhança direta: $ f(\alpha a' + \beta b') = \alpha f(a')
        + \beta f(b')$. Deduza que centróides, pontos médios e
        Os centros de Napoleão abaixo são todos transportados por semelhanças
        --- a licença algébrica por trás de cada ``sem perda de
        generalidade, coloque o círculo circunscrito no círculo unitário''
        argumento.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Napoleon's theorem.} Deixar $(a, b, c)$
seja um triângulo direto. De cada lado, erga o equilátero externo
triângulo (questão 3) e deixe $ n_a $, $ n_b $, $ n_c $ sejam os centros
(centróides) dos triângulos erguidos em $[b, c]$, $[c, a]$,
$[a, b]$ respectivamente.
\begin{enumerate}[resume]
  \item Mostre isso
        \[
        n_a = \alpha b + \beta c, \qquad
        n_b = \alpha c + \beta a, \qquad
        n_c = \alpha a + \beta b,
        \qquad\text{with}\quad
        \alpha = \frac{3 + \iu\sqrt3}{6},\ \beta = \conj\alpha .
        \]
  \item Verificar $\alpha + \beta = 1$, e deduza que o triângulo
        $(n_a, n_b, n_c)$ tem o \emph{mesmo centróide} como
        $(a, b, c)$.
  \item Usando as identidades da questão 4, mostre que
        \[
        n_a + j\,n_b + j^2 n_c
        = (\alpha j^2 + \beta j)\,(a + jb + j^2c) .
        \]
  \item Calcular $\alpha j + \beta $ e concluo: $\alpha j^2 +
        \beta j = j(\alpha j + \beta) = 0$, então $ n_a + j n_b + j^2
        n_c = 0$ para \emph{todo} triângulo: o Napoleão exterior
        triângulo é equilátero direto (ou um ponto). de Napoleão
        teorema está provado.
  \item Erga os triângulos equiláteros \emph{para dentro} em vez disso
        (escolha $\eu^{+\iu\pi/3}$ na questão 3) e deixe $ m_a, m_b,
        m_c $ sejam seus centros. Mostrar $ m_a = \beta b + \alpha c $ (e
        ciclicamente), então prove $ m_a + j^2 m_b + j\,m_c = 0$: o
        O triângulo interno de Napoleão também é equilátero, com o
        orientação oposta.
  \item Localize a degenerescência: mostre que $ n_a = n_b = n_c $ acontece
        exatamente quando $ a + j^2 b + jc = 0$, ou seja,\ quando $(a, b, c)$
        é um \emph{indireto} triângulo equilátero. \emph{(Dois
        pontos entre $ n_a, n_b, n_c $ coincidem se todos os três coincidirem, por
        questão 5; então use $ n_a + j^2 n_b + j n_c = j(\alpha +
        \beta j)(a + j^2b + jc)$ e verifique $\alpha + \beta j \neq
        0$.)}
  \item Faça todo o cálculo no triângulo $ a = 0$,
        $ b = 1$, $ c = \iu $: dar $ n_a, n_b, n_c $ exatamente, e
        verificar por cálculo direto do lado de três quadrados
        comprimentos que o triângulo é equilátero, de lado quadrado
        $\frac{2 + \sqrt3}{3}$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Ptolemy and van Schooten.}
\begin{enumerate}[resume]
  \item Prove a identidade, válida para todos os complexos $ a, b, c, d $:
        \[
        (a - b)(c - d) + (a - d)(b - c) = (a - c)(b - d) .
        \]
  \item Deduzir \emph{A desigualdade de Ptolomeu}\index{A desigualdade de Ptolomeu}: para quaisquer quatro pontos $ A, B, C, D $,
        \[
        AC \cdot BD \;\leq\; AB \cdot CD + AD \cdot BC ,
        \]
        com igualdade se e somente se $(a-b)(c-d)$ e
        $(a-d)(b-c)$ mentir em um raio comum de $0$.
  \item Mostre isso para $\theta, \varphi \in \R $,
        $\abs{\eu^{\iu\theta} - \eu^{\iu\varphi}} =
        2\,\abs{\sin\frac{\theta - \varphi}2}$.
  \item (teorema de van Schooten\index{teorema de van Schooten}) Deixar
        $(a, b, c) = (1, j, j^2)$ --- na Parte II isso não perde
        generalidade entre triângulos equiláteros diretos --- e deixe
        $ p = \eu^{\iu\theta}$ com $\theta \in
        \intoo{2\pi/3}{4\pi/3}$, um ponto do círculo circunscrito no
        arco $ BC $ não contendo $ A $. Provar
        \[
        PA = PB + PC .
        \]
        \emph{(Expresse as três distâncias com a questão 20 e use
        uma fórmula de soma para produto.)}
  \item Verifique em $ p = -1$ que este é exatamente o caso de igualdade
        da desigualdade de Ptolomeu para a ordem cíclica $ A, B, P, C $:
        calcular $(a - b)(p - c)$ e $(a - c)(b - p)$ e verifique
        que sua proporção é um real positivo.
\end{enumerate}
