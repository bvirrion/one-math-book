\chapter{Equações Diferenciais Lineares}\label{ch:b1:diffeq}

Conhecidas pela primeira vez no volume do Ensino Médio, equações diferenciais são tratadas
aqui com provas completas e em maior generalidade: linear de primeira ordem
equações com coeficientes variáveis (resolvidas completamente pelo
método de variação de constantes) e equações lineares de segunda ordem com
coeficientes constantes, o modelo para oscilações. Ambos os casos exibem
a mesma estrutura: \emph{solução geral $=$ uma solução específica
$+$ solução geral da equação homogênea}.

\section{Equações lineares de primeira ordem}

\begin{definition}\label{def:b1:diffeq:linear1}
Deixar $ I $ seja um intervalo e $ a, b \colon I \to \R $ (ou $\C $) ser
contínuo. A equação
\[
(E)\colon\quad y' + a(x)\,y = b(x),
\]
na função diferenciável desconhecida $ y \colon I \to \R $ (ou $\C $),
é um \emph{diferencial linear de primeira ordem
pesquisa}\index{diferencial linear de primeira ordem
pesquisa}. O
equação $(H)\colon y' + a(x) y = 0$ é o seu \emph{homogêneo}
equação.
\end{definition}

\begin{theorem}[Resolvendo a equação homogênea]\label{thm:b1:diffeq:homogeneous1}
Deixar $ A $ ser um primitivo de $ a $ sobre $ I $ (existe:
\cref{ch:b1:integration}). As soluções de $(H)$ sobre $ I $ são exatamente
as funções
\[
y(x) = \lambda\, \eu^{-A(x)}, \qquad \lambda \in \R \text{ (or } \C).
\]
\end{theorem}

\begin{proof}
Essas funções são soluções: $ y' = -\lambda A' \eu^{-A} = -a y $.
Por outro lado, deixe $ y $ resolver $(H)$ e \omterm{def:b1:logic:sets}{definir} $ z(x) = y(x)\, \eu^{A(x)}$.
Então
\[
z' = y' \eu^{A} + y\, a\, \eu^{A} = (y' + ay)\, \eu^{A} = 0 ,
\]
então $ z $ é constante no intervalo $ I $, dizer $ z = \lambda $: $ y =
\lambda \eu^{-A}$. (Observe a lógica: nenhuma solução está perdida, porque
\emph{todo} solução foi escrita na forma anunciada.)
\end{proof}

\begin{example}[Uma equação homogênea com coeficiente variável]\label{ex:b1:diffeq:varcoeff}
Resolver $ y' + (\cos x)\,y = 0$ sobre $\R $. Um primitivo de $ a(x) =
\cos x $ é $ A(x) = \sin x $, então as soluções são
\[
y(x) = \lambda\,\eu^{-\sin x}, \qquad \lambda \in \R .
\]
Duas leituras. Toda solução é periódica (período $2\pi $) e nunca
desaparece a menos que $\lambda = 0$ --- o sinal de $\lambda $ é o
sinal de $ y $ para sempre, já que uma exponencial não pode cruzar zero. E
a solução através $ y(0) = y_0$ é $ y_0\eu^{-\sin x}$: exatamente
uma curva da família através de cada ponto inicial, o
imagem unidimensional de \cref{thm:b1:diffeq:voc}~(2).
\end{example}

\begin{theorem}[Variação de constantes; Problema de Cauchy]\label{thm:b1:diffeq:voc}
Com a notação acima:
\begin{enumerate}
  \item As soluções de $(E)$ sobre $ I $ são exatamente
        \[
        y(x) = \Bigl(\lambda + \int_{x_0}^{x} b(t)\, \eu^{A(t)} \dd t
        \Bigr)\, \eu^{-A(x)},
        \qquad \lambda \in \R,
        \]
        onde $ x_0 \in I $ é fixo. Equivalentemente: solução geral de
        $(H)$ mais uma solução particular de $(E)$.
  \item Para cada $ x_0 \in I $ e $ y_0$, o \emph{Cauchy
        problema}\index{Cauchy
        problema} ``$(E)$ e $ y(x_0) = y_0$''
        tem exatamente uma solução $ I $.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Seguindo o método nomeado \emph{variação de variação
constantes}\index{variação de variação
constantes}, procure soluções do
formulário $ y = \mu(x)\, \eu^{-A(x)}$ com $\mu $ diferenciável --- não
generalidade é perdida, uma vez que cada função em $ I $ pode ser tão escrito
($\mu = y\,\eu^{A}$). Substituindo,
\[
y' + ay = \mu' \eu^{-A} - \mu a \eu^{-A} + a\mu\eu^{-A}
= \mu'\, \eu^{-A},
\]
então $ y $ resolve $(E)$ se e somente se $\mu'(x) = b(x)\,\eu^{A(x)}$, se
e somente se $\mu(x) = \lambda + \int_{x_0}^x b(t)\eu^{A(t)}\dd t $ para
alguma constante $\lambda $ (duas primitivas do mesmo contínuo
função em um intervalo diferem por uma constante).

(2) Na fórmula, $ y(x_0) = \lambda\,\eu^{-A(x_0)}$: a condição
$ y(x_0) = y_0$ determina $\lambda = y_0 \eu^{A(x_0)}$ exclusivamente.
\end{proof}

\begin{example}\label{ex:b1:diffeq:order1}
Resolver $ y' + \dfrac{y}{x} = x^2$ sobre $ I = \intoo{0}{+\infty}$. Aqui
$ a(x) = \frac 1x $, $ A(x) = \ln x $, $\eu^{-A(x)} = \frac 1x $.
Soluções homogêneas: $\frac{\lambda}{x}$. Variação de constantes:
$\mu'(x) = x^2 \cdot x = x^3$, então $\mu = \frac{x^4}{4} + \lambda $, e
\[
y(x) = \frac{x^3}{4} + \frac{\lambda}{x}, \qquad \lambda \in \R .
\]
Com a condição inicial $ y(1) = 0$: $\lambda = -\frac14$.
\emph{Verificar:} $ y' + \frac yx = \frac{3x^2}{4} - \frac{\lambda}{x^2} +
\frac{x^2}{4} + \frac{\lambda}{x^2} = x^2$.
\end{example}

\begin{example}[Adivinhar é melhor do que integrar]\label{ex:b1:diffeq:guess}
Resolver $ y' + 2x\,y = x $ sobre $\R $. Variação de constantes funciona ($ A =
x^2$, $\mu' = x\,\eu^{x^2}$, $\mu = \frac12\eu^{x^2} + \lambda $),
mas observando que o \emph{constante} $ y_p = \frac12$ resolve o
equação ($0 + 2x\cdot\frac12 = x $) é mais rápido. Com o
soluções homogêneas $\lambda\,\eu^{-x^2}$:
\[
y(x) = \frac12 + \lambda\,\eu^{-x^2}, \qquad \lambda \in \R .
\]
Toda solução converge para $\frac12$ extremamente rápido como $ x \to
\pm\infty $: a solução particular constante é uma
\emph{equilíbrio} que todas as soluções se juntem. O insight: antes
iniciando o método geral, gaste dez segundos procurando por um
solução particular óbvia (constante, monômio, múltiplo do
lado direito); o teorema da estrutura então termina o trabalho.
\end{example}

\begin{remark}[Intervalos são importantes]
O teorema vive em um \emph{intervalo} onde $ a $ e $ b $ são
contínuo. Para $ y' + \frac yx = 0$ sobre $\R^*$, as soluções são
$\frac{\lambda}{x}$ sobre $\intoo{0}{+\infty}$ e
$\frac{\mu}{x}$ sobre $\intoo{-\infty}{0}$ com \emph{independente}
constantes: não há razão para uma única fórmula colar no
singularidade em $0$.
\end{remark}

\begin{example}[Um lado direito complexo, duas respostas reais]\label{ex:b1:diffeq:complexrhs}
Resolver $ y' - y = \cos x $ e $ y' - y = \sin x $ de uma só vez. Trabalho
em $\C $ com o lado direito $\eu^{\iu x}$: tentando $ y_p =
c\,\eu^{\iu x}$ dá $ c(\iu - 1)\eu^{\iu x} = \eu^{\iu x}$, então
\[
c = \frac1{\iu - 1} = \frac{-1 - \iu}2,
\qquad
y_p = -\frac{(1 + \iu)(\cos x + \iu\sin x)}2
= \frac{\sin x - \cos x}2 + \iu\,\frac{-\sin x - \cos x}2 .
\]
Como a equação tem coeficientes reais, reais e imaginários
\omterm{thm:b1:logic:partition}{divisão} de peças: $\frac{\sin x - \cos x}2$ resolve $ y' - y = \cos x $,
e $-\frac{\sin x + \cos x}2$ resolve $ y' - y = \sin x $ (verificar
o primeiro: derivado $\frac{\cos x + \sin x}2$, menos o
função, dá $\cos x $). Uma linha complexa substituiu duas séries de
variação de constantes --- a mesma economia que
\cref{met:b1:diffeq:particular} sistematiza para segunda ordem,
e um dividendo recorrente de \cref{ch:b1:complex}.
\end{example}

\section{Equações lineares de segunda ordem com coeficientes constantes}

\begin{definition}\label{def:b1:diffeq:linear2}
Deixar $ a, b \in \R $ e $ f \colon I \to \R $ contínuo. A equação
\[
(E)\colon\quad y'' + a\,y' + b\,y = f(x)
\]
é um \emph{solução linear de segunda ordem com constante
coeficientes}\index{solução linear de segunda ordem com constante
coeficientes};
$(H)\colon y'' + ay' + by = 0$ é sua equação homogênea, e
$\chi(r) = r^2 + ar + b $ isso é \emph{característica
polinômio}\index{característica
polinômio}.
\end{definition}

\begin{theorem}[Soluções homogêneas]\label{thm:b1:diffeq:homogeneous2}
Deixar $\Delta = a^2 - 4b $ ser o discriminante de $\chi $. O verdadeiro
soluções de $(H)$ sobre $\R $ são:
\begin{enumerate}
  \item se $\Delta > 0$, com $ r_1 \neq r_2$ as duas raízes reais:
        $\;y = \lambda\, \eu^{r_1 x} + \mu\, \eu^{r_2 x}$;
  \item se $\Delta = 0$, com $ r_0$ a raiz dupla:
        $\;y = (\lambda + \mu x)\, \eu^{r_0 x}$;
  \item se $\Delta < 0$, com raízes $\alpha \pm \iu\omega $
        ($\omega > 0$):
        $\;y = \eu^{\alpha x} (\lambda \cos\omega x + \mu
        \sin\omega x)$;
\end{enumerate}
em cada caso com $(\lambda, \mu)$ atropelando $\R^2$.
\end{theorem}

\begin{proof}
Primeiro observe que para $ r \in \C $, $ x \mapsto \eu^{rx}$ resolve $(H)$
se e somente se $\chi(r) = 0$ (substituir: $(r^2 + ar + b)\eu^{rx} =
0$). É por isso que as exponenciais são a primeira estimativa natural:
a diferenciação atua sobre $\eu^{rx}$ como multiplicação pelo número
$ r $, então a equação diferencial se torna a equação numérica
$\chi(r) = 0$ --- todo o problema analítico é comprimido em
encontrar as raízes de uma quadrática.

O passo principal é uma mudança no desconhecido que reduz a ordem. Deixar $ r $ ser
uma raiz (possivelmente complexa) de $\chi $ e escrever $ y = z\, \eu^{rx}$,
que não perde generalidade. Então
\[
y'' + ay' + by
= \bigl(z'' + (2r + a) z' + \chi(r) z\bigr)\eu^{rx}
= \bigl(z'' + (2r + a) z'\bigr)\eu^{rx},
\]
então $(H)$ torna-se o \emph{primeira ordem} equação $ u' + (2r + a) u = 0$
para $ u = z'$.

\emph{Caso $\Delta \neq 0$:} escolher $ r = r_1$; então $2r_1 + a = r_1 -
r_2$ (desde $ r_1 + r_2 = -a $). Por
\cref{thm:b1:diffeq:homogeneous1}, $ z' = c\,\eu^{(r_2 - r_1)x}$ para
alguma constante $ c $; integrando em $\R $, $ z = \mu\, \eu^{(r_2 - r_1)x}
+ \lambda $ com $\mu = \frac{c}{r_2 - r_1}$, e portanto $ y = z\,
\eu^{r_1 x} = \lambda\,\eu^{r_1x} + \mu\,\eu^{r_2x}$.
Quando $\Delta < 0$, as raízes são $\alpha \pm \iu\omega $ e o
soluções complexas são $ y = c_1\eu^{(\alpha+\iu\omega)x} +
c_2\eu^{(\alpha-\iu\omega)x}$ com $ c_1, c_2 \in \C $. Qual de
estes são de valor real? Desde $\conj{\eu^{(\alpha+\iu\omega)x}} =
\eu^{(\alpha-\iu\omega)x}$, o \omterm{def:b1:complex:field}{conjugado} de $ y $ é $\conj{c_2}\,
\eu^{(\alpha+\iu\omega)x} + \conj{c_1}\,\eu^{(\alpha-\iu\omega)x}$,
e $ y = \conj y $ para todos $ x $ forças $ c_2 = \conj{c_1}$ (os dois
exponenciais são linearmente independentes: avalie em dois pontos, ou
comparar em $ x=0$ depois de dividir por $\eu^{\alpha x}$). Escrita $ c_1
= \frac{\lambda - \iu\mu}2$ com $\lambda, \mu $ real:
\[
y = 2\,\Re\Bigl(c_1\,\eu^{\alpha x}(\cos\omega x +
\iu\sin\omega x)\Bigr)
= \eu^{\alpha x}\bigl(\lambda\cos\omega x + \mu\sin\omega
x\bigr),
\]
e, inversamente, toda função desse tipo é uma solução (parte real de um
solução complexa de uma equação real): o espaço de solução real é
conforme anunciado.

\emph{Caso $\Delta = 0$:} $ r = r_0$, $2r_0 + a = 0$, então $ z'' = 0$:
$ z = \lambda + \mu x $ e $ y = (\lambda + \mu x)\eu^{r_0 x}$.
\end{proof}

\begin{example}[Um problema de Cauchy, do início ao fim]\label{ex:b1:diffeq:cauchy2}
Resolver $ y'' - 3y' + 2y = 0$ com $ y(0) = 0$, $ y'(0) = 1$. O
polinômio característico $ r^2 - 3r + 2 = (r - 1)(r - 2)$ tem o
raízes reais $1$ e $2$: solução geral $ y = \lambda\eu^{x} +
\mu\eu^{2x}$. As duas condições fornecem o sistema linear
\[
\lambda + \mu = 0, \qquad \lambda + 2\mu = 1 ,
\]
então $\mu = 1$, $\lambda = -1$:
\[
y(x) = \eu^{2x} - \eu^{x} .
\]
Verificar: $ y(0) = 0$; $ y' = 2\eu^{2x} - \eu^x $ tem $ y'(0) = 1$; e
$ y'' - 3y' + 2y = (4 - 6 + 2)\eu^{2x} + (-1 + 3 - 2)\eu^x = 0$.
Observe o formato da resposta: próximo $-\infty $ o modo lento
$-\eu^x $ domina; aproximar $+\infty $ o modo rápido $\eu^{2x}$ faz.
Lendo soluções como superposições de modos com diferentes decaimentos
ou taxas de crescimento é o hábito lucrativo --- é como o
A \omterm{thm:b1:logic:partition}{divisão} transitória/estado estacionário do problema do fim de semana é organizada.
\end{example}

\begin{theorem}[Estrutura e problema de Cauchy]\label{thm:b1:diffeq:structure2}
\begin{enumerate}
  \item Se $ y_p $ é uma solução particular de $(E)$, as soluções de
        $(E)$ são exatamente $ y_p + y_h $, $ y_h $ correndo sobre o
        soluções de $(H)$.
  \item (Superposição) Se $ y_1$ resolve $ y'' + ay' + by = f_1$ e
        $ y_2$ resolve $ y'' + ay' + by = f_2$, então $ y_1 + y_2$ resolve
        a equação com lado direito $ f_1 + f_2$.
  \item Para todos $ x_0 \in I $ e $(y_0, y_0')$, o Problema de Cauchy
        ``$(E)$, $ y(x_0) = y_0$, $ y'(x_0) = y_0'$'' tem exatamente um
        solução em $ I $. \emph{(Existência concedida a um determinado
        solução; singularidade na íntegra.)}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) $ y $ resolve $(E)$ se $ y - y_p $ resolve $(H)$, por linearidade de $ y
\mapsto y'' + ay' + by $. (2) é a mesma linearidade.

(3) Por (1) é suficiente provar que as constantes $(\lambda,
\mu)$ sempre pode ser ajustado, exclusivamente, a quaisquer dados $(y_0, y_0')$.
Traduzindo a variável, suponha $ x_0 = 0$. No caso (1) de
\cref{thm:b1:diffeq:homogeneous2}, $ y = \lambda\eu^{r_1x} +
\mu\eu^{r_2x}$ dá
\[
y(0) = \lambda + \mu, \qquad y'(0) = r_1\lambda + r_2\mu :
\]
um sistema linear em $(\lambda, \mu)$ cujo determinante é $ r_2 -
r_1 \neq 0$; resolvendo-o explicitamente, $\mu = \frac{y_0' -
r_1y_0}{r_2 - r_1}$ e $\lambda = y_0 - \mu $: exatamente um
solução. No caso (2), $ y(0) = \lambda $ e $ y'(0) = r_0\lambda +
\mu $: o sistema é triangular com determinante $1$, resolvido por
$\lambda = y_0$, $\mu = y_0' - r_0y_0$. No caso (3), $ y(0) =
\lambda $ e $ y'(0) = \alpha\lambda + \omega\mu $: determinante
$\omega \neq 0$, resolvido por $\lambda = y_0$, $\mu = \frac{y_0' -
\alpha y_0}\omega $. Em cada caso o \omterm{def:b1:logic:map}{mapa} $(\lambda, \mu) \mapsto
(y(x_0), y'(x_0))$ é uma bijeção linear --- a linguagem de
\cref{ch:b1:linmaps} irá compactar esta verificação de caso em um
sentença.
\end{proof}

\begin{method}[Solução específica para $ f(x) = P(x)\,\eu^{\gamma x}$]\label{met:b1:diffeq:particular}
Quando o lado direito é $ P(x)\,\eu^{\gamma x}$ com $ P $ um
polinômio e $\gamma \in \R $ (isso cobre polinômios, exponenciais,
e via complexo $\gamma $ ou superposição, $\cos $ e $\sin $): olha
para uma solução particular da forma
\[
y_p(x) = x^{m}\, Q(x)\, \eu^{\gamma x},
\qquad
m = \text{multiplicity of } \gamma \text{ as a root of } \chi
\ (m = 0, 1 \text{ or } 2),
\]
com $ Q $ um polinômio de mesmo grau que $ P $, cujos coeficientes
são encontrados por substituição e identificação. Para $ f = K\cos\omega x $
(ou $\sin $), resolva com o lado direito $ K\eu^{\iu\omega x}$ e pegue
a parte real (resp.\ imaginária).
\end{method}

\begin{example}[Superposição em ação]\label{ex:b1:diffeq:superposition}
Resolver $ y'' - y = \eu^{x} + 4$ sobre $\R $. Homogêneo: $\chi(r) = r^2
- 1$, raízes $\pm1$, então $ y_h = \lambda\eu^x + \mu\eu^{-x}$. Dividir
o lado direito e trate cada peça pela caixa de método.
\emph{Pedaço $\eu^x $:} aqui $\gamma = 1$ é uma raiz simples de
$\chi $, então tente $ y_1 = c\,x\,\eu^x $: então $ y_1'' - y_1 = c(x +
2)\eu^x - cx\eu^x = 2c\,\eu^x $, dando $ c = \frac12$.
\emph{Pedaço $4$:} $\gamma = 0$ não é uma raiz; a constante $ y_2 =
-4$ funciona. Por superposição
(\cref{thm:b1:diffeq:structure2}~(2)):
\[
y = \frac{x\,\eu^x}2 - 4 + \lambda\,\eu^x + \mu\,\eu^{-x},
\qquad (\lambda, \mu) \in \R^2 .
\]
Observe como as duas peças exigiram \emph{diferente} formas ($ m = 1$
contra $ m = 0$): o teste de multiplicidade é aplicado a cada
expoente separadamente, que é o objetivo de dividir o
lado direito antes de adivinhar.
\end{example}

\begin{example}[A regra da multiplicidade em ação]\label{ex:b1:diffeq:multiplicity}
Resolver $ y'' + y' = x $ sobre $\R $. O lado direito é $ P(x)\eu^{0
\cdot x}$ com $ P(x) = x $, e $\gamma = 0$ é um \emph{simples} raiz
de $\chi(r) = r^2 + r = r(r + 1)$: então $ m = 1$, e o correto
acho que é $ y_p = x\,(\alpha x + \beta) = \alpha x^2 + \beta x $, um
grau superior a $ P $. Substituindo:
\[
y_p'' + y_p' = 2\alpha + (2\alpha x + \beta)
= 2\alpha x + (2\alpha + \beta) ,
\]
e identificação com $ x $ dá $\alpha = \frac12$, $\beta = -1$:
$ y_p = \frac{x^2}2 - x $. Solução geral: $ y = \frac{x^2}2 - x +
\lambda + \mu\,\eu^{-x}$. se tivéssemos adivinhado $ y_p = \alpha x + \beta $
(ignorando a multiplicidade), a substituição daria $ y_p'' + y_p'
= \alpha $, uma constante --- sem escolha de $\alpha, \beta $ pode combinar
$ x $, e a falha é estrutural: as constantes já resolvem o
equação homogênea, então eles são invisíveis para o lado esquerdo.
O fator $ x^m $ existe precisamente para sair do homogêneo
espaço de solução.
\end{example}

\begin{remark}[A trigésima segunda apólice de seguro]\label{rem:b1:diffeq:check}
Cada equação resolvida neste capítulo termina com uma substituição
verifique, e isso não é decorativo. Uma equação diferencial
a computação encadeia muitos pequenos passos (uma primitiva, uma regra de produto,
duas constantes), e um erro de sinal único se propaga de forma invisível;
substituir a fórmula final de volta na equação pega
essencialmente todos eles ao custo de uma diferenciação.
Cultive o reflexo em três camadas: verifique o \emph{particular
solução} sozinho (a parte homogênea é cancelada de qualquer maneira), verifique o
\emph{condições iniciais} na solução completa, e quando um
parâmetro estiver presente, verifique um \emph{valor degenerado} (será que
fórmula para geral $\Omega $ reproduza a resposta conhecida em
$\Omega = 0$?). O hábito custa meio minuto; ele converte
``provavelmente certo'' em ``verificado''.
\end{remark}

\begin{remark}[Armadilhas comuns]\label{rem:b1:diffeq:pitfalls}
\begin{enumerate}
  \item \emph{Normalize primeiro.} As fórmulas assumem a equação
        lê $ y' + a(x)y = b(x)$ --- coeficiente $1$ sobre $ y'$.
        Para $ xy' - 2y = x^3$, divida por $ x $ (em um intervalo
        evitando $0$) antes de identificar $ a $ e $ b $, como em
        \cref{exo:b1:diffeq:2}.
  \item \emph{Uma constante por dimensão, fixada no final.} O
        solução geral de primeira ordem carrega uma constante, a
        segunda ordem um dois; condições iniciais são impostas
        o \emph{completo} solução $ y_p + y_h $, nunca ligado $ y_h $
        sozinho --- impondo-os antes de adicionar $ y_p $ é o mais
        erro estrutural frequente.
  \item \emph{Cuidado com a multiplicidade.} Uma suposição de solução particular
        que resolve a equação homogênea é invisível para o
        lado esquerdo; o fator $ x^m $ de
        \cref{met:b1:diffeq:particular} não é opcional
        (\cref{ex:b1:diffeq:multiplicity}).
  \item \emph{Os intervalos fazem parte da resposta.} Soluções ao vivo
        em intervalos onde os coeficientes são contínuos;
        colar através de uma singularidade pode criar constantes espúrias
        (\cref{exo:b1:diffeq:12}) ou destruir a exclusividade. ``Resolver
        ligado $\R^*$'' significa dois problemas independentes.
\end{enumerate}
\end{remark}

\begin{example}[Forçando fora da ressonância]\label{ex:b1:diffeq:offresonance}
Resolver $ y'' + 4y = \sin x $ sobre $\R $. Frequência natural $2$, forçando
frequência $1$: desde $\iu $ é \emph{não} uma raiz de $\chi(r) = r^2
+ 4$, a multiplicidade é $ m = 0$ e uma sinusóide simples é suficiente.
Tentando $ y_p = \alpha\sin x $ (não é necessário cosseno: a equação tem
não $ y'$ prazo, e $\sin $ regenera $\sin $):
\[
y_p'' + 4y_p = -\alpha\sin x + 4\alpha\sin x = 3\alpha\sin x ,
\]
então $\alpha = \frac13$ e a solução geral é
\[
y = \frac{\sin x}3 + \lambda\cos 2x + \mu\sin 2x .
\]
Toda solução permanece limitada: uma superposição de duas oscilações
nas frequências $1$ (forçado) e $2$ (natural). Compare com
o próximo exemplo, onde forçar \emph{no} a frequência natural
muda a própria forma da resposta.
\end{example}

\begin{example}[Uma oscilação forçada]\label{ex:b1:diffeq:oscillation}
Resolver $ y'' + y = \cos x $, $ y(0) = 0$, $ y'(0) = 0$.

\emph{Homogêneo:} $\chi(r) = r^2 + 1$, raízes $\pm\iu $: $ y_h =
\lambda\cos x + \mu \sin x $.

\emph{Especial:} lado direito $\Re(\eu^{\iu x})$ com $\gamma =
\iu $ uma raiz simples de $\chi $: tentar $ z_p = c\, x\, \eu^{\iu x}$
($ c \in \C $). Então $ z_p'' + z_p = c\,(2\iu)\eu^{\iu x}$, que é igual
$\eu^{\iu x}$ para $ c = \frac{1}{2\iu} = -\frac\iu2$. Então $ z_p =
-\frac{\iu}{2} x (\cos x + \iu \sin x)$ e $ y_p = \Re(z_p) =
\frac{x \sin x}{2}$.

\emph{Solução geral:} $ y = \frac{x\sin x}{2} + \lambda\cos x +
\mu\sin x $. Condições: $ y(0) = \lambda = 0$; $ y' = \frac{\sin x +
x\cos x}{2} + \mu\cos x $, então $ y'(0) = \mu = 0$. Responder: $ y =
\frac{x\sin x}{2}$ --- uma oscilação cuja amplitude cresce linearmente:
o \emph{ressonância}\index{ressonância} fenômeno, causado por forçar
o sistema em sua frequência natural.
\end{example}

\begin{remark}[Interlúdio: linearidade é uma geometria]\label{rem:b1:diffeq:interlude}
Reveja a forma de cada solução \omterm{def:b1:logic:sets}{definir} neste capítulo: um
solução especial mais um espaço de soluções homogêneas com um
constante livre (primeira ordem) ou duas (segunda ordem). Os capítulos sobre
álgebra linear (\cref{ch:b1:vspaces,ch:b1:findim,ch:b1:linmaps})
fornecerá o exato
vocabulário: o \omterm{def:b1:logic:map}{mapa} $ L(y) = y'' + ay' + by $ é \emph{linear}, é
soluções homogêneas formam o \emph{núcleo} de $ L $, um vetor
espaço cujo \emph{dimensão} é igual à ordem da equação ---
esse é o conteúdo honesto de ``uma constante por pedido'' --- e
a solução \omterm{def:b1:logic:sets}{definir} de $ L(y) = f $ é um \emph{subespaço afim}, um
tradução do kernel. Até mesmo o Cauchy \omterm{def:b1:logic:map}{mapa} $(\lambda, \mu)
\mapsto (y(x_0), y'(x_0))$ de \cref{thm:b1:diffeq:structure2} é um
bijeção linear entre dois planos, ou seja, uma inversão $2
\times 2$ sistema (\cref{ch:b1:matrices}). Nada neste capítulo
precisará ser refeito --- só precisará ser renomeado e
a renomeação é o melhor aquecimento possível para álgebra linear:
toda definição abstrata já ganhou a vida
aqui.
\end{remark}

\begin{remark}[Onde este capítulo é usado]\label{rem:b1:diffeq:whereused}
O teorema da estrutura --- soluções de $(E)$ formar ``um particular
solução mais as soluções de $(H)$'' --- é a primeira aparição
de um padrão que \cref{ch:b1:vspaces,ch:b1:linmaps} nomeará: o
solução \omterm{def:b1:logic:sets}{definir} de $(H)$ é o \emph{núcleo} da linearidade \omterm{def:b1:logic:map}{mapa} $ y
\mapsto y'' + ay' + by $, e a solução \omterm{def:b1:logic:sets}{definir} de $(E)$ é um
tradução afim disso. O polinômio característico reaparece como
o polinômio característico de uma matriz em \cref{ch:b1:matrices}:
uma equação de segunda ordem é $2 \times 2$ sistema de primeira ordem em
disfarce, um ponto de vista que o volume do Ano 2 sistematiza. O
integrais exigidas por variação de constantes são fornecidos por
\cref{ch:b1:integration}, e o problema do fim de semana abaixo --- o
oscilador amortecido acionado --- é o caso modelo para cada
questão da oscilação nas ciências, dos circuitos à suspensão
pontes.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b1:diffeq:1}
Resolver $\R $: $\;y' + 2y = \eu^{3x}$; então o Problema de Cauchy $ y(0) =
1$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:diffeq:2}
Resolver $\intoo{0}{+\infty}$: $\;x y' - 2y = x^3$ \emph{(coloque o
equação na forma normalizada primeiro)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:diffeq:3}
Resolver $\R $, fornecendo a solução geral real:
$\;y'' - 3y' + 2y = 0$; $\;y'' + 4y' + 4y = 0$; $\;y'' - 2y' + 5y =
0$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:diffeq:4}
Resolver $ y'' - y = x^2$ sobre $\R $, então o Problema de Cauchy $ y(0) = 0$,
$ y'(0) = 1$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:diffeq:5}
Resolver $\intoo{-\frac\pi2}{\frac\pi2}$:
$\;y' + y\tan x = \sin 2x $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:diffeq:6}
Resolver $ y'' - 4y' + 3y = (2x + 1)\,\eu^{x}$ sobre $\R $. \emph{(Cuidado com o
multiplicidade: é $1$ uma raiz do polinômio característico?)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:diffeq:7}
Resolver $ y'' + 4y = \sin 2x + x $ sobre $\R $ \emph{(superposição; tratar
cada lado direito separadamente)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:diffeq:8}
Uma xícara de café em temperatura $ T_0 = 80\,^\circ $ C está sentado em uma sala às
$20\,^\circ $ C. Lei dos estados de resfriamento de Newton $ T' = -k\,(T - 20)$ com
$ k > 0$. Resolva para $ T(t)$, e dado que o café está em
$50\,^\circ $ C depois $10$ minutos, descubra quando chega
$25\,^\circ $ C.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:diffeq:9}
(Oscilador amortecido) Para $\varepsilon \geq 0$, considere
$ y'' + 2\varepsilon y' + y = 0$.
\begin{enumerate}
  \item Resolva para $\varepsilon \in \intco{0}{1}$, $\varepsilon = 1$,
        e $\varepsilon > 1$.
  \item Mostre isso para $\varepsilon > 0$ toda solução tende a $0$
        no $+\infty $, e isso para $\varepsilon = 0$ o diferente de zero
        soluções não.
  \item Para $\varepsilon \in \intoo{0}{1}$, mostre que os zeros de a
        solução diferente de zero são regularmente espaçadas, com lacuna
        $\frac{\pi}{\sqrt{1 - \varepsilon^2}}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:diffeq:10}
Encontre todas as funções $ f \colon \R \to \R $, duas vezes diferenciável, tal
isso
\[
\forall x, y \in \R, \qquad f(x + y) + f(x - y) = 2 f(x) f(y),
\]
com $ f(0) \ne 0$ e $ f $ não constante.
\emph{Dica: consertar $ y $, diferencie duas vezes em relação a $ x $ no $0$;
mostrar $ f(0) = 1$ e $ f'' = c f $ para alguma constante $ c $; então resolva
de acordo com o sinal de $ c $ e verifique quais soluções satisfazem a
equação funcional.}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:diffeq:11}
(Equação de Euler) Resolver $ x^2 y'' - x y' + y = 0$ sobre
$\intoo{0}{+\infty}$. \emph{Dica: \omterm{def:b1:logic:sets}{definir} $ z(t) = y(\eu^t)$, ou seja\
substituto $ x = \eu^t $, e mostre que $ z $ satisfaz uma linear
equação com coeficientes constantes.}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:diffeq:12}
Considere a equação $ x\,y' = 2y $ em toda a linha real, no
função diferenciável desconhecida $ y \colon \R \to \R $.
\begin{enumerate}
  \item Resolver $\intoo{0}{+\infty}$ e assim por diante $\intoo{-\infty}{0}$.
  \item Mostre isso para \emph{qualquer} constantes $ a, b \in \R $, o
        função igual a $ ax^2$ para $ x \geq 0$ e para $ bx^2$ para
        $ x < 0$ é diferenciável em $\R $ e resolve a equação
        em todos os lugares.
  \item Conclua que a solução \omterm{def:b1:logic:sets}{definir} sobre $\R $ é um parâmetro de dois parâmetros
        família, e explique por que isso não contradiz a
        singularidade em \cref{thm:b1:diffeq:voc}.
\end{enumerate}
\end{exercise}

\section{Problema: O oscilador amortecido acionado}

\begin{problem}\label{pb:b1:diffeq:1}
Uma equação governa uma massa em uma mola em um meio viscoso, a
carga em um circuito RLC e um prédio balançando ao vento:
\[
(E_\Omega)\colon\quad
x'' + 2\lambda x' + \omega_0^2\,x = A\cos(\Omega t),
\]
com $\lambda \geq 0$ o amortecimento, $\omega_0 > 0$ o natural
frequência, e $ A > 0$, $\Omega > 0$ a amplitude e a frequência de
o forçamento. Este problema extrai seu comportamento completo: o decaimento
de transientes, o único estado estacionário periódico, o \omterm{ex:b1:diffeq:oscillation}{ressonância}
curva e sua nitidez (o \emph{\omterm{pb:b1:diffeq:1}{fator de qualidade}}), as \omterm{pb:b1:diffeq:1}{batidas} de
o caso não amortecido e o balanço de energia que sustenta o
oscilação.\index{ressonância}\index{oscilador}
Salvo indicação em contrário, $0 < \lambda < \omega_0$ (subamortecido
regime) e escrevemos $\omega_d = \sqrt{\omega_0^2 - \lambda^2}$.

\medskip
\noindent\textbf{Part I --- The free oscillator.} Aqui $ A = 0$.
\begin{enumerate}
  \item Resolva a equação homogênea $(H)$ para $0 < \lambda <
        \omega_0$, e para $\lambda = 0$. (Os regimes $\lambda
        \geq \omega_0$ foram tratados em \cref{exo:b1:diffeq:9};
        citá-los.)
  \item Mostre isso para cada $\lambda > 0$, todas as soluções de $(H)$
        tendem a $0$ no $+\infty $ --- em todos os três regimes.
  \item Defina a energia $\mathcal E(t) = \frac12 x'(t)^2 +
        \frac12\omega_0^2\,x(t)^2$ ao longo de uma solução de $(H)$. Mostrar
        $\mathcal E'(t) = -2\lambda\,x'(t)^2 \leq 0$, e deduza
        (sem resolver nada) que o Problema de Cauchy
        ``$(H)$, $ x(t_0) = x'(t_0) = 0$'' tem apenas o zero
        solução, para cada $\lambda \geq 0$.
  \item Para $0 < \lambda < \omega_0$, escreva a solução diferente de zero como
        $ x(t) = R\,\eu^{-\lambda t}\cos(\omega_d t - \varphi)$ e
        deixe $ T_d = \frac{2\pi}{\omega_d}$ seja o pseudoperíodo.
        Mostre isso $ x(t + T_d) = \eu^{-\lambda T_d}\,x(t)$: cada
        swing é o anterior reduzido pelo fator constante
        $\eu^{-\delta}$, $\delta = \frac{2\pi\lambda}{\omega_d}$
        (o \emph{decremento logarítmico}). Calcular $\delta $ para
        $\omega_0 = 1$, $\lambda = 0.1$.
  \item Defina o \emph{\omterm{pb:b1:diffeq:1}{fator de qualidade}} $ Q =
        \dfrac{\omega_0}{2\lambda}$\index{fator de qualidade}. Mostrar
        que depois do tempo $\frac1\lambda $ (uma amplitude
        $\eu $-folding), o oscilador completou
        $\frac{\omega_d}{2\pi\lambda}$ pseudoperíodos, que para
        amortecimento fraco ($\lambda \ll \omega_0$) é aproximadamente
        $\frac Q\pi $: o \omterm{pb:b1:diffeq:1}{fator de qualidade} conta, até $\pi $, o
        oscilações sobreviveram antes que a amplitude decaísse em
        $\eu $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The steady state.} Agora $ A > 0$ e
$\lambda > 0$.
\begin{enumerate}[resume]
  \item Procure uma solução específica como a parte real do
        $ z\,\eu^{\iu\Omega t}$ com $ z \in \C $
        (\cref{met:b1:diffeq:particular}). Mostre que isso funciona
        com
        \[
        z = \frac{A}{\omega_0^2 - \Omega^2 + 2\iu\lambda\Omega} .
        \]
  \item Deduza o estado estacionário na forma amplitude-fase:
        $ x_p(t) = R(\Omega)\cos\bigl(\Omega t -
        \varphi(\Omega)\bigr)$ com
        \[
        R(\Omega) = \frac{A}{\sqrt{(\omega_0^2 - \Omega^2)^2 +
        4\lambda^2\Omega^2}},
        \qquad
        \tan\varphi = \frac{2\lambda\Omega}{\omega_0^2 -
        \Omega^2},
        \quad \varphi \in \intoo0\pi .
        \]
  \item Interprete os dois regimes extremos: calcule os limites de
        $ R $ e $\varphi $ como $\Omega \to 0^+$ (quase estático
        resposta $ A/\omega_0^2$, fase $0$) e como $\Omega \to
        +\infty $ ($ R \sim A/\Omega^2 \to 0$, fase $\to \pi $: o
        a massa se move na direção oposta a um forçamento muito rápido).
  \item Mostre isso \emph{todo} solução de $(E_\Omega)$ é $ x_p $
        mais uma solução de $(H)$, portanto converge para o constante
        estado $ x_p $ como $ t \to +\infty $, qualquer que seja o inicial
        condições: depois que o transitório morre, o oscilador
        nenhuma memória de como tudo começou.
  \item Mostre isso $ x_p $ é o \emph{apenas} solução periódica de
        $(E_\Omega)$.
  \item Trabalhe um Problema de Cauchy até o fim: para $ x'' + 2x' + 2x =
        \cos t $ com $ x(0) = x'(0) = 0$, mostre que a solução é
        \[
        x(t) = \frac{\cos t + 2\sin t}5
        - \eu^{-t}\,\frac{\cos t + 3\sin t}5 ,
        \]
        e identificar partes transitórias e estáveis.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The \omterm{ex:b1:diffeq:oscillation}{ressonância} curve.} Estudo de
$\Omega \mapsto R(\Omega)$ sobre $\intoo0{+\infty}$.
\begin{enumerate}[resume]
  \item Contexto $ u = \Omega^2$ e $ g(u) = (\omega_0^2 - u)^2 +
        4\lambda^2 u $, mostrar: se $2\lambda^2 < \omega_0^2$, então $ R $
        atinge um máximo estrito no \emph{\omterm{ex:b1:diffeq:oscillation}{ressonância} freqüência}
        $\Omega_r = \sqrt{\omega_0^2 - 2\lambda^2}$, com
        \[
        R_{\max} = R(\Omega_r)
        = \frac{A}{2\lambda\sqrt{\omega_0^2 - \lambda^2}} .
        \]
  \item Mostre isso $\dfrac{R_{\max}}{R(0)} = Q\,\bigl(1 -
        \tfrac{\lambda^2}{\omega_0^2}\bigr)^{-1/2} \geq Q $: no
        \omterm{ex:b1:diffeq:oscillation}{ressonância}, o forçamento é amplificado (essencialmente) pelo
        \omterm{pb:b1:diffeq:1}{fator de qualidade}.
  \item Prove que o \emph{velocidade} amplitude $ V(\Omega) =
        \Omega\,R(\Omega)$ é máximo exatamente em $\Omega =
        \omega_0$ (não em $\Omega_r $), e que a fase existe
        $\varphi(\omega_0) = \frac\pi2$: no $\Omega = \omega_0$
        a velocidade está exatamente em fase com a força.
  \item (Largura de banda) Resolver $ g(u) = 2\,g(u_r)$ exatamente, onde $ u_r =
        \omega_0^2 - 2\lambda^2$, e deduza que os dois
        frequências $\Omega_\pm $ onde $ R = R_{\max}/\sqrt2$
        satisfazer $\Omega_+^2 - \Omega_-^2 =
        4\lambda\sqrt{\omega_0^2 - \lambda^2}$; concluir que para
        amortecimento fraco a largura de banda é $\Omega_+ - \Omega_-
        \approx 2\lambda $, ou seja\ $ Q \approx
        \frac{\omega_0}{\Omega_+ - \Omega_-}$: afiado \omterm{ex:b1:diffeq:oscillation}{ressonância}
        os picos são altos $ Q $ sistemas.
  \item Retrato numérico para $\omega_0 = 1$, $\lambda = 0.05$
        ($ Q = 10$), $ A = 1$: calcular $\Omega_r $, $ R_{\max}$, o
        resposta estática $ R(0)$ e a largura de banda aproximada.
  \item Mostre que se $2\lambda^2 \geq \omega_0^2$, então $ R $ é
        diminuindo estritamente em $\intoo0{+\infty}$: fortemente amortecido
        sistemas não têm \omterm{ex:b1:diffeq:oscillation}{ressonância} pico em tudo.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- No damping: beats and \omterm{ex:b1:diffeq:oscillation}{ressonância}.}
Aqui $\lambda = 0$.
\begin{enumerate}[resume]
  \item Para $\Omega \neq \omega_0$, encontre a solução geral de
        $ x'' + \omega_0^2 x = A\cos(\Omega t)$.
  \item Resolva o Problema de Cauchy $ x(0) = x'(0) = 0$ e transformar
        a resposta no formulário do produto
        \[
        x(t) = \frac{2A}{\omega_0^2 - \Omega^2}\,
        \sin\Bigl(\frac{(\omega_0 - \Omega)t}2\Bigr)
        \sin\Bigl(\frac{(\omega_0 + \Omega)t}2\Bigr) .
        \]
  \item Para $\Omega $ perto de $\omega_0$, leia o produto como um
        oscilação rápida na frequência $\frac{\omega_0 + \Omega}2$
        modulado por um envelope lento na frequência
        $\frac{\abs{\omega_0 - \Omega}}2$: o \emph{batidas}\index{batidas}. Informe o período do envelope e o
        amplitude máxima e observe como ambos explodem conforme $\Omega
        \to \omega_0$.
  \item Consertar $ t $ e deixe $\Omega \to \omega_0$ na questão 19
        fórmula: mostre que o limite é
        \[
        x_\infty(t) = \frac{A\,t\,\sin(\omega_0 t)}{2\omega_0} ,
        \]
        e verifique diretamente isso $ x_\infty $ resolve o ressonante
        equação $ x'' + \omega_0^2 x = A\cos(\omega_0 t)$ com
        $ x(0) = x'(0) = 0$ (comparar
        \cref{ex:b1:diffeq:oscillation}): \omterm{ex:b1:diffeq:oscillation}{ressonância} é o limite
        de \omterm{pb:b1:diffeq:1}{batidas} cada vez mais lentas e cada vez maiores.
  \item Compare os dois destinos de \omterm{ex:b1:diffeq:oscillation}{ressonância}: crescimento linear
        $\frac{At}{2\omega_0}$ sem amortecimento, versus saturação
        em $ R_{\max} \approx Q\,\frac{A}{\omega_0^2}$ com fraco
        amortecimento. Em uma frase: que mecanismo físico converte
        o primeiro no segundo?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Energy balance and synthesis.}
\begin{enumerate}[resume]
  \item No estado estacionário da Parte II, calcule a média ao longo
        um período $\frac{2\pi}\Omega $ de (a) a potência injetada
        pelo forçamento, $ P_{\mathrm{in}}(t) = A\cos(\Omega t)
        \cdot x_p'(t)$, e (b) a potência dissipada pelo
        amortecimento, $ P_{\mathrm{diss}}(t) = 2\lambda\,x_p'(t)^2$.
        Mostrar ambas as médias iguais $\lambda\,R^2\Omega^2$: o
        forçar alimenta exatamente o que o amortecimento queima --- isso
        é por isso que o estado estacionário é estacionário.
  \item Onde exatamente o problema usou: (i) a estrutura
        teorema \cref{thm:b1:diffeq:structure2}; (ii) o complexo
        método exponencial; (iii) um estudo de função de variável real
        no estilo de \cref{ch:b1:functions}? Uma frase
        cada um.
  \item Síntese: descreva o comportamento completo \omterm{def:b1:logic:map}{mapa} de $(E_\Omega)$
        --- livre versus forçado, amortecido versus não amortecido, o papel
        de $ Q $ como o pico de sintonia de dial adimensional único
        altura, largura de banda e vida útil transitória --- e mencione
        onde a história continua: primeira ordem $2 \times 2$
        sistemas (\cref{ch:b1:matrices} e o volume do Ano 2) e
        a decomposição de um forçamento periódico geral em
        sinusóides (série de Fourier, no volume do Ano 3), para
        qual o caso sinusoidal deste problema é o fundamental
        bloco de construção.
\end{enumerate}
\end{problem}
