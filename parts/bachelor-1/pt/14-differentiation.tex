\chapter{Diferenciação}\label{ch:b1:derivative}

Derivados foram computados ao longo do volume do Ensino Médio; o que foi
faltando é a cadeia de teoremas que transforma a computação em
informações sobre funções: teorema de Rolle, teorema do valor médio,
e suas consequências --- critérios de monotonicidade, \omterm{def:b1:topology:closure}{limites} de Lipschitz,
convexidade. Tudo neste capítulo diz respeito a funções definidas em
um \omterm{prop:b1:reals:intervals}{intervalo} $I $.

\section{A derivada}

\begin{definition}\label{def:b1:derivative:def}
$ f \colon I \to \R $ é \emph{diferenciável em $ x_0 \in
I $}\index{diferenciável em $ x_0 \in
I $} quando o quociente de diferença
$\frac{f(x) - f(x_0)}{x - x_0}$ tem um \omterm{def:b1:topology:closure}{limite} (finito) como $ x \to x_0$;
o \omterm{def:b1:topology:closure}{limite} está escrito $ f'(x_0)$. Equivalentemente:
\[
f(x_0 + h) = f(x_0) + f'(x_0)\,h + h\,\varepsilon(h),
\qquad \varepsilon(h) \xrightarrow[h \to 0]{} 0 ,
\]
o gráfico então admitindo a reta tangente $ y = f(x_0) + f'(x_0)(x -
x_0)$. Diferenciabilidade em $ x_0$ implica continuidade no $ x_0$ (leia
o visor). $ f $ é diferenciável em $ I $ quando está em todos os pontos;
$ f $ é de classe $ C^1$ quando além disso $ f'$ é contínuo, e de
aula $ C^k $\index{Função Ck@$ C^k $ função} quando $ f $ pode ser
diferenciado $ k $ vezes com $ f^{(k)}$ contínuo.
\end{definition}

\begin{example}\label{ex:b1:derivative:notc1}
O inverso de ``diferenciável $\Rightarrow $ contínuo'' falha:
$\abs{\,\cdot\,}$ no $0$. Mais surpreendentemente, diferenciável não
implicar $ C^1$: a função $ f(x) = x^2 \sin\frac 1x $ ($ f(0) = 0$) é
diferenciável em todos os lugares, com $ f'(0) = 0$, mas $ f'(x) = 2x
\sin\frac1x - \cos\frac 1x $ não tem \omterm{def:b1:topology:closure}{limite} em $0$
(\cref{exo:b1:derivative:2}).
\end{example}

\begin{example}[Diferenciável em exatamente um ponto]\label{ex:b1:derivative:onepoint}
Deixar $ f(x) = x^2$ para $ x \in \Q $ e $ f(x) = 0$ para $ x \notin
\Q $. No $0$: $\bigl|\frac{f(h) - 0}{h}\bigr| \leq \abs h \to 0$,
então $ f $ é diferenciável no $0$ com $ f'(0) = 0$. Em qualquer $ x_0
\neq 0$, $ f $ não é mesmo contínuo: racional e irracional
sequências convergindo para $ x_0$ enviar $ f $ para $ x_0^2 \neq 0$ e para
$0$ respectivamente (densidade, \cref{thm:b1:reals:density}). Então
a diferenciabilidade é uma questão genuinamente \emph{pontualmente} noção: pode
segure em um ponto de $\R $ e em nenhum outro lugar. A moral para
praticar: declarações como o critério de monotonicidade ou Rolle
exigir o derivado \emph{em um \omterm{prop:b1:reals:intervals}{intervalo}} --- possuindo
$ f'(x_0)$ em pontos isolados, por mais numerosos que sejam, não apoia nenhuma
qualquer conclusão.
\end{example}

\begin{theorem}[Operações]\label{thm:b1:derivative:operations}
Se $ f, g $ são diferenciável no $ x_0$ (e onde as fórmulas fazem
sentido):
\[
(f + g)' = f' + g', \qquad
(fg)' = f'g + fg', \qquad
\Bigl(\frac fg\Bigr)' = \frac{f'g - fg'}{g^2},
\]
e se $ g $ é diferenciável no $ f(x_0)$:
$\;(g \circ f)'(x_0) = g'\bigl(f(x_0)\bigr)\, f'(x_0)$ (corrente
regra\index{regra da cadeia}).
\end{theorem}

\begin{proof}
Soma: imediato. Produto: escrever
\[
f(x)g(x) - f(x_0)g(x_0)
= \bigl(f(x) - f(x_0)\bigr) g(x) + f(x_0)\bigl(g(x) - g(x_0)\bigr),
\]
dividir por $ x - x_0$ e deixe $ x \to x_0$ ($ g $ é contínuo no $ x_0$).
Quociente: tratar $\frac 1g $ através de $\frac{1/g(x) - 1/g(x_0)}{x - x_0} =
\frac{-1}{g(x)g(x_0)}\cdot\frac{g(x) - g(x_0)}{x - x_0}$, então aplique
a regra do produto. Regra da cadeia: com $ y_0 = f(x_0)$, \omterm{def:b1:logic:sets}{definir}
$\theta(y) = \frac{g(y) - g(y_0)}{y - y_0}$ para $ y \neq y_0$ e
$\theta(y_0) = g'(y_0)$: $\theta $ é contínuo no $ y_0$, e para
$ x \neq x_0$,
\[
\frac{g(f(x)) - g(f(x_0))}{x - x_0}
= \theta\bigl(f(x)\bigr)\cdot \frac{f(x) - f(x_0)}{x - x_0}
\longrightarrow g'(y_0)\, f'(x_0),
\]
o primeiro fator por composição de \omterm{def:b1:topology:closure}{limites} (este dispositivo trata do
caso $ f(x) = f(x_0)$ de forma limpa, onde o ingênuo ``multiplicar e dividir
por $ f(x) - f(x_0)$'' quebra).
\end{proof}

\begin{theorem}[Derivada de uma função inversa]\label{thm:b1:derivative:inverse}
Deixar $ f $ ser contínuo e estritamente monótono em $ I $, diferenciável
no $ x_0$ com $ f'(x_0) \neq 0$. Então $ f^{-1}$
(\cref{thm:b1:continuity:bijection}) é diferenciável no $ y_0 =
f(x_0)$, com
\[
(f^{-1})'(y_0) = \frac{1}{f'(x_0)} = \frac{1}{f'\bigl(f^{-1}(y_0)\bigr)} .
\]
Se $ f'(x_0) = 0$, o inverso tem uma tangente vertical em $ y_0$.
\end{theorem}

\begin{proof}
Para $ y \to y_0$, \omterm{def:b1:logic:sets}{definir} $ x = f^{-1}(y)$: continuidade de $ f^{-1}$ dá $ x
\to x_0$, e
\[
\frac{f^{-1}(y) - f^{-1}(y_0)}{y - y_0}
= \frac{x - x_0}{f(x) - f(x_0)}
= \frac{1}{\dfrac{f(x) - f(x_0)}{x - x_0}}
\longrightarrow \frac{1}{f'(x_0)} .
\]
Afirmação tangente vertical: se $ f'(x_0) = 0$, o quociente exibido
é o inverso de uma quantidade que tende a $0$ enquanto mantém
um sinal constante (para $ f $ aumentando estritamente, $\frac{f(x) -
f(x_0)}{x - x_0} > 0$ para todos $ x \neq x_0$): a diferença
quociente de $ f^{-1}$ portanto tende a $+\infty $ (para $-\infty $
para $ f $ diminuindo). O inverso permanece contínuo mas não é
diferenciável no $ y_0$ --- seu gráfico, o reflexo de $ f $ de
na diagonal, fica vertical exatamente onde $ f $ correu
horizontais, como $ x^{1/3}$ no $0$ ilustra contra $ x^3$.
\end{proof}

\begin{example}[Derivadas inversas, duas vezes]\label{ex:b1:derivative:inverseaction}
O teorema recomputa o clássico derivados sem \omterm{def:b1:topology:closure}{limite}
trabalho. Para $\ln = \exp^{-1}$: no $ y = \eu^x $,
\[
(\ln)'(y) = \frac{1}{\exp'(x)} = \frac{1}{\eu^{x}} = \frac1y ,
\]
válido para cada $ y > 0$ desde $\exp' = \exp $ nunca desaparece. Para
$\arctan = \tan^{-1}$: no $ y = \tan x $,
\[
(\arctan)'(y) = \frac{1}{1 + \tan^2 x} = \frac{1}{1 + y^2} ,
\]
usando $\tan' = 1 + \tan^2 > 0$. O insight final: a fórmula
converte conhecimento sobre uma função em conhecimento sobre sua
inverso ao preço de uma substituição --- e o
substituição ($ x = \ln y $, $ x = \arctan y $) é exatamente o
\omterm{def:b1:logic:statement}{declaração} que as duas variáveis vivem em lados opostos do
bijeção.
\end{example}

\section{Rolle e o teorema do valor médio}

\begin{proposition}[Extremo interior]\label{prop:b1:derivative:fermat}
Se $ f $ é diferenciável em um \emph{\omterm{def:b1:topology:closure}{interior}} apontar $ x_0$ de $ I $ e
tem um extremo local ali, então $ f'(x_0) = 0$.
\end{proposition}

\begin{proof}
Diga um máximo local: existe $ r > 0$ com $ f(x) \leq f(x_0)$
para $\abs{x - x_0} \leq r $, e a interioridade garante tanto
lados de $ x_0$ estão disponíveis dentro $ I $. Para $0 < h \leq r $ o
quociente $\frac{f(x_0 + h) - f(x_0)}{h}$ tem numerador $\leq 0$
e denominador $> 0$: isso é $\leq 0$, e seu \omterm{def:b1:topology:closure}{limite} $ f'(x_0)$
herda $\leq 0$ (grandes desigualdades chegam aos \omterm{def:b1:topology:closure}{limites},
\cref{thm:b1:seq:order}); para $-r \leq h < 0$ o quociente é
$\geq 0$, dando $ f'(x_0) \geq 0$. Por isso $ f'(x_0) = 0$. (Em um
ponto final, apenas um sinal está disponível: a conclusão falha aí ---
pense em $ x $ sobre $\intcc{0}{1}$, máximo em $1$ com derivado
$1$.)
\end{proof}

\begin{theorem}[Rolle]\label{thm:b1:derivative:rolle}
Deixar $ f $ ser contínuo sobre $\intcc{a}{b}$, diferenciável sobre
$\intoo{a}{b}$, com $ f(a) = f(b)$. Então $ f'(c) = 0$ para alguns $ c \in
\intoo{a}{b}$.\index{Teorema de Rolle}
\end{theorem}

\begin{proof}
Pelo teorema do valor extremo (\cref{thm:b1:continuity:evt}), $ f $
atinge seu máximo e mínimo em $\intcc{a}{b}$. Se ambos forem
alcançado nos pontos finais, então (desde $ f(a) = f(b)$) máx. $=$ min e $ f $
é constante: qualquer \omterm{def:b1:topology:closure}{interior} $ c $ funciona. Caso contrário, um extremo é
alcançado em um \omterm{def:b1:topology:closure}{interior} apontar $ c $, e
\cref{prop:b1:derivative:fermat} dá $ f'(c) = 0$.
\end{proof}

\begin{theorem}[Teorema do valor médio]\label{thm:b1:derivative:mvt}
Deixar $ f $ ser contínuo sobre $\intcc{a}{b}$, diferenciável sobre
$\intoo{a}{b}$. Existe $ c \in \intoo{a}{b}$ com
\[
f(b) - f(a) = f'(c)\,(b - a) .
\]
\emph{Desigualdade de valor médio:} se além disso $ m \leq f' \leq M $ sobre
$\intoo{a}{b}$, então $ m(b-a) \leq f(b) - f(a) \leq M(b-a)$; em particular $\abs{f'} \leq K $ implica que $ f $ é $ K $-Lipschitz.
\index{teorema do valor médio}
\end{theorem}

\begin{proof}
Aplicar Rolle a $ g(x) = f(x) - \frac{f(b) - f(a)}{b - a}(x - a)$: $ g $
é contínuo sobre $\intcc{a}{b}$, diferenciável dentro, e $ g(a) =
f(a) = g(b)$. No ponto $ c $ onde $ g'(c) = 0$: $ f'(c) =
\frac{f(b)-f(a)}{b-a}$. A desigualdade segue delimitando $ f'(c)$;
o Lipschitz \omterm{def:b1:logic:statement}{declaração} aplica-o a cada par de pontos.
\end{proof}

\begin{example}[O método de Newton é o de Heron]\label{ex:b1:derivative:newton}
Método de Newton para resolver $ f(x) = 0$ substitui a curva por
sua tangente na estimativa atual $ x_n $ e pega a tangente
root como a próxima estimativa:
\[
0 = f(x_n) + f'(x_n)(x_{n+1} - x_n)
\quad\Longrightarrow\quad
x_{n+1} = x_n - \frac{f(x_n)}{f'(x_n)} .
\]
Execute-o $ f(x) = x^2 - 2$:
\[
x_{n+1} = x_n - \frac{x_n^2 - 2}{2x_n}
= \frac{x_n}{2} + \frac{1}{x_n}
= \frac12\Bigl(x_n + \frac{2}{x_n}\Bigr) :
\]
exatamente a iteração de Heron (\cref{ex:b1:seq:heron}), dois
milênios antes. A velocidade quadrática observada lá é agora
explicado pela imagem tangente: perto de uma raiz simples, curva e
tangente diferem por um erro de segunda ordem, então cada passo aproximadamente
eleva ao quadrado o erro --- o geral \omterm{def:b1:logic:statement}{declaração} segue do
\omterm{def:b1:topology:closure}{Limites} de Taylor de \cref{ch:b1:taylor}. A visão final:
onde dicotomia (\cref{ex:b1:continuity:dichotomy}) usa apenas
continuidade e ganha um bit por passo, Newton gasta um
derivado para dobrar o número de dígitos corretos por etapa.
\end{example}

\begin{example}[O teorema do valor médio como estimador]\label{ex:b1:derivative:mvtestimate}
Quão grande é $\sqrt{101}$? Aplique o teorema a $ f(t) = \sqrt t $
sobre $\intcc{100}{101}$: para alguns $ c \in \intoo{100}{101}$,
\[
\sqrt{101} - 10 = \frac{1}{2\sqrt c},
\qquad\text{so}\qquad
\frac{1}{2\sqrt{101}} < \sqrt{101} - 10 < \frac{1}{20} = 0.05 ,
\]
e desde então $\sqrt{101} < 10.05$, o \omterm{def:b1:topology:closure}{limite} esquerdo excede
$\frac{1}{20.1} > 0.0497$: por isso $10.0497 < \sqrt{101} < 10.05$
(valor verdadeiro $10.049875\dots $) --- três decimais corretos de um
derivado avaliação. Da mesma maneira $\abs{\sin a - \sin b} \leq
\abs{a - b}$ (vinculado $\abs{\cos}\leq 1$): as estimativas de Lipschitz
usado desde \cref{ch:b1:seq} são todo esse teorema. O fechamento
insight: o teorema do valor médio é uma fórmula de Taylor de ordem zero
--- negocia um ponto desconhecido $ c $ para uma dura desigualdade, e
\cref{ch:b1:taylor} irá iterar exatamente esse comércio.
\end{example}

\begin{corollary}[Critério de monotonicidade]\label{cor:b1:derivative:monotone}
Deixar $ f $ ser contínuo sobre $ I $, diferenciável no \omterm{def:b1:topology:closure}{interior}.
\begin{enumerate}
  \item $ f' \geq 0$ no \omterm{def:b1:topology:closure}{interior} $\iff $ $ f $ está aumentando; $ f' = 0$
        $\iff $ $ f $ constante.
  \item Se $ f' > 0$ exceto em um número finito de pontos onde ele desaparece,
        $ f $ é \emph{estritamente} aumentando.
\end{enumerate}
\end{corollary}

\begin{proof}
Se $ f' \geq 0$: para $ x < y $ em $ I $, o teorema do valor médio em
$\intcc{x}{y}$ dá $ f(y) - f(x) = f'(c)(y - x) \geq 0$. Por outro lado,
quocientes de diferença de uma função crescente são $\geq 0$, então seu
os \omterm{def:b1:topology:closure}{limites} também o são. O caso constante: aplique o anterior a $ f'$ e
$-f' \geq 0$. Versão estrita: $ f $ está aumentando; igualdade $ f(x) =
f(y)$ para $ x < y $ congelaria $ f $ sobre $\intcc{x}{y}$, forçando $ f' =
0$ lá --- infinitos pontos.
\end{proof}

\begin{example}[Derivadas iguais, funções desiguais]\label{ex:b1:derivative:intervalmatters}
Sobre $\R^* = \intoo{-\infty}{0} \cup \intoo{0}{+\infty}$, ambos
$ f(x) = \ln\abs x $ e $ g(x) = \ln\abs x + \mathbf{1}_{x>0}$
(adicionar $1$ apenas na meia linha direita) satisfaz $ f' = g' =
\frac1x $. Eles não diferem por uma constante: o critério
``$ f' = 0 \implies f $ constante'' é um \emph{\omterm{prop:b1:reals:intervals}{intervalo}} \omterm{def:b1:logic:statement}{declaração}
--- sua prova executa o teorema do valor médio entre dois pontos,
o que exige que todo o segmento que os une esteja no
domínio. Em cada meia linha separadamente, as primitivas de
$\frac1x $ são $\ln\abs x + c $ com uma constante por meia linha,
duas constantes independentes no total. \Cref{ch:b1:integration}
herda estas letras miúdas: ``a'' primitiva de uma função é
bem definido até uma constante \emph{em cada \omterm{prop:b1:reals:intervals}{intervalo} do seu
domínio}, e as tabelas antiderivadas assumem silenciosamente a conectividade.
\end{example}

\begin{example}[Rigor de graça]\label{ex:b1:derivative:strictness}
$ x \mapsto x^3$ é \emph{estritamente} aumentando em $\R $ mesmo
embora seja derivado desaparece em $0$: a cláusula do critério
``$ f' > 0$ exceto em um número finito de pontos'' é exatamente projetado
para esses pontos planos. Em contraste, $ f' \geq 0$ sozinho só dá
aumento no sentido amplo (uma função constante se qualifica), e
um derivado desaparecer em um subintervalo inteiro congela o
funcionar lá. A regra prática: reivindicar estrito
monotonicidade, liste os zeros de $ f'$; finitos (ou mais
geralmente, nenhum em qualquer subintervalo) é inofensivo, um \omterm{prop:b1:reals:intervals}{intervalo} de
eles é fatal.
\end{example}

\begin{example}[Um estudo de variação completo]\label{ex:b1:derivative:variations}
Estudar $ f(x) = x^3 - 3x + 1$ sobre $\R $. Derivado: $ f'(x) =
3(x^2 - 1)$, positivo em $\intoo{-\infty}{-1}$, negativo em
$\intoo{-1}{1}$, positivo em $\intoo{1}{+\infty}$: pelo
critério de monotonicidade, $ f $ aumenta, depois diminui, então
aumenta, com um máximo local $ f(-1) = 3$ e um mínimo local
$ f(1) = -1$. \omterm{def:b1:topology:closure}{Limites}: $\mp\infty $ no $\mp\infty $. Consequências,
leia a tabela de variação com o valor intermediário
teorema em cada ramo monótono: $ f $ desaparece exatamente uma vez em
cada um
\[
\intoo{-\infty}{-1}, \qquad \intoo{-1}{1}, \qquad
\intoo{1}{+\infty}
\]
(os valores nas junções têm sinais opostos: $3 > 0 > -1$),
então a equação $ x^3 - 3x + 1 = 0$ tem exatamente três reais
raízes; numericamente eles se sentam perto $-1.88$, $0.35$, $1.53$. O
conclusão final: uma tabela de variação é um \emph{dispositivo de prova},
não é um esboço --- ramo monótono mais mudança de sinal é igual
exatamente uma raiz, e a tabela enumera os ramos
exaustivamente.
\end{example}

\begin{theorem}[Fórmula de Leibniz]\label{thm:b1:derivative:leibniz}
Se $ f, g $ são $ n $ vezes diferenciável, assim é $ fg $,
e\index{Fórmula de Leibniz}
\[
(fg)^{(n)} = \sum_{k=0}^{n} \binom nk f^{(k)}\, g^{(n-k)} .
\]
\end{theorem}

\begin{proof}
Indução ativada $ n $, exatamente paralelo ao teorema binomial. O
caso $ n = 1$ é a regra do produto. Assumindo a fórmula na classificação
$ n $, diferencie mais uma vez:
\[
(fg)^{(n+1)} = \sum_{k=0}^{n} \binom nk
\Bigl( f^{(k+1)} g^{(n-k)} + f^{(k)} g^{(n-k+1)} \Bigr),
\]
então reindexar a primeira soma com $ j = k + 1$ e recolher o
coeficiente de $ f^{(j)} g^{(n+1-j)}$: isso é $\binom{n}{j-1} +
\binom nj = \binom{n+1}{j}$ pela regra de Pascal
(\cref{prop:b1:counting:identities}), o \omterm{def:b1:topology:closure}{limite} termos $ j = 0$
e $ j = n + 1$ carregando $\binom{n+1}{0} = \binom{n+1}{n+1} =
1$ como deveriam.
\end{proof}

\begin{example}[Leibniz em ação]\label{ex:b1:derivative:leibnizaction}
Calcular $\bigl(x^2 \eu^x\bigr)^{(n)}$ para $ n \geq 2$. Pegar $ f =
x^2$, cujo derivados morrer rapidamente ($ f' = 2x $, $ f'' = 2$,
$ f^{(k)} = 0$ para $ k \geq 3$), e $ g = \eu^x $: apenas três termos
da soma de Leibniz sobreviver,
\[
\bigl(x^2\eu^x\bigr)^{(n)}
= \binom n0 x^2 \eu^x + \binom n1 (2x)\,\eu^x + \binom n2\,
2\,\eu^x
= \eu^x\bigl(x^2 + 2nx + n(n-1)\bigr).
\]
Verificação de sanidade em $ n = 1$: $\eu^x(x^2 + 2x)$, o que é de fato
$(x^2\eu^x)'$. O insight final: use Leibniz quando um fator
é um polinomial --- a soma então tem apenas $\deg + 1$ termos, e
a fórmula é uma forma fechada, não uma identidade abstrata. (Por dois
fatores infinitamente vivos como $\eu^x\sin x $, complexo
exponenciais de \cref{ch:b1:complex} são a \omterm{def:b1:arith:prime}{melhor} ferramenta.)
\end{example}

\section{Convexidade}

\begin{definition}\label{def:b1:derivative:convex}
$ f \colon I \to \R $ é \emph{convexo}\index{convexo} quando
cada acorde está acima do gráfico:
\[
\forall x, y \in I,\ \forall t \in \intcc{0}{1}, \quad
f\bigl(tx + (1-t)y\bigr) \leq t f(x) + (1-t) f(y).
\]
($ f $ é \emph{côncavo} quando $-f $ é convexo.)
\end{definition}

\begin{theorem}[Caracterizações diferenciais]\label{thm:b1:derivative:convexchar}
Deixar $ f $ ser diferenciável sobre $ I $. Os seguintes são equivalentes:
\begin{enumerate}
  \item $ f $ é \omterm{def:b1:derivative:convex}{convexo};
  \item $ f'$ está aumentando em $ I $;
  \item o gráfico está acima de cada tangente: $ f(y) \geq f(x) +
        f'(x)(y - x)$ para todos $ x, y \in I $.
\end{enumerate}
Se $ f $ é duas vezes diferenciável: $ f $ \omterm{def:b1:derivative:convex}{convexo} $\iff f'' \geq 0$.
\end{theorem}

\begin{proof}
(1 $\Rightarrow $ 3) Convexidade escrita como $\frac{f(x + t(y-x)) -
f(x)}{t} \leq f(y) - f(x)$ para $ t \in \intoc{0}{1}$; deixar $ t \to 0^+$:
$ f'(x)(y - x) \leq f(y) - f(x)$.

(3 $\Rightarrow $ 2) Para $ x < y $, as duas desigualdades tangentes em $ x $
e em $ y $ dar $ f'(x)(y-x) \leq f(y) - f(x) \leq f'(y)(y - x)$,
por isso $ f'(x) \leq f'(y)$.

(2 $\Rightarrow $ 1) Correção $ x < y $ e $ t \in \intoo{0}{1}$, e deixe $ z
= tx + (1-t)y \in \intoo{x}{y}$. Pelo teorema do valor médio em
$\intcc{x}{z}$ e $\intcc{z}{y}$: há $ c_1 < z < c_2$ com
\[
\frac{f(z) - f(x)}{z - x} = f'(c_1) \leq f'(c_2)
= \frac{f(y) - f(z)}{y - z} ,
\]
e compensando denominadores ($ z - x = (1-t)(y-x)$, $ y - z = t(y-x)$)
reorganiza exatamente na desigualdade de convexidade.

Duas vezes diferenciável caso: $ f'' \geq 0 \iff f'$ aumentando
(\cref{cor:b1:derivative:monotone}).
\end{proof}

\begin{example}[Desigualdades clássicas de convexidade]\label{ex:b1:derivative:convexineq}
$\exp $ é \omterm{def:b1:derivative:convex}{convexo} ($\exp'' = \exp > 0$): sua tangente em $0$ dá
$\eu^x \geq 1 + x $ para todos $ x $. $\ln $ é côncavo: sua tangente em $1$
dá $\ln x \leq x - 1$; seus acordes dão, por $0 < a \leq b $, o
desigualdade entre médias geométricas e aritméticas: tomando $ t =
\frac12$ na concavidade,
\[
\ln\frac{a + b}{2} \geq \frac{\ln a + \ln b}{2} = \ln\sqrt{ab},
\qquad\text{so}\qquad
\sqrt{ab} \leq \frac{a+b}{2} .
\]
A desigualdade aritmética-geométrica geral é
\cref{exo:b1:derivative:9}.
\end{example}

\begin{example}[Uma desigualdade de convexidade do zero]\label{ex:b1:derivative:xlnx}
A função $ f(t) = t\ln t $ é \omterm{def:b1:derivative:convex}{convexo} sobre $\intoo{0}{+\infty}$:
$ f''(t) = \frac1t > 0$. Sua desigualdade de ponto médio, multiplicada por
$2$, lê-se: para todos $ a, b > 0$,
\[
a\ln a + b\ln b \;\geq\; (a + b)\,\ln\frac{a + b}{2} ,
\]
com igualdade se $ a = b $ (convexidade estrita). Teste drive: $ a =
1$, $ b = 3$ dá $3\ln 3 = 3.296$ contra $4\ln 2 = 2.773$.
Esta desigualdade inócua é o caso de dois pontos da
\emph{entropia} comparação que reaparece com Jensen
desigualdade (\cref{exo:b1:derivative:9}) e no
assintótica teórica da informação do volume do Ano 3. O
conclusão final: para fabricar uma desigualdade, encontre uma função
cujo segundo derivado tem um sinal e escreva qual convexidade
diz --- a caracterização diferencial transforma uma verificação de sinal
em infinitas desigualdades.
\end{example}

\begin{remark}[Armadilhas comuns com derivativos]
(eu) \emph{Um positivo derivado em um ponto não dá
monotonicidade perto dele}: $ f(x) = \frac x2 + x^2\sin\frac1x $ (com
$ f(0) = 0$) tem $ f'(0) = \frac12 > 0$, ainda
\[
f'(x) = \frac12 + 2x\sin\frac1x - \cos\frac1x
\]
é igual $-\frac12$ em cada $ x_n = \frac{1}{2\pi n}$: todo
\omterm{def:b1:topology:open}{vizinhança} de $0$ contém descidas. Necessidades de monotonicidade $ f'
\geq 0$ \emph{em um \omterm{prop:b1:reals:intervals}{intervalo}}
(\cref{cor:b1:derivative:monotone}); apenas o sinal pontual
controla o cruzamento da linha tangente. (ii) \emph{Rolle's
três hipóteses estão todas ativas}: $\abs x $ sobre $\intcc{-1}{1}$
(não \omterm{def:b1:topology:closure}{interior} diferenciabilidade), $ x $ sobre $\intcc{0}{1}$ (não termina
iguais) e $ x - \lfloor x\rfloor $ sobre $\intcc{0}{1}$ (continuidade
falha em $1$) cada um quebra exatamente uma hipótese e o
conclusão. (iii) \emph{Derivados pode ser descontínuo, mas
não arbitrariamente}: $ f'$ pode oscilar
(\cref{ex:b1:derivative:notc1}) mas sempre satisfaz o
propriedade de valor intermediário (Darboux,
\cref{exo:b1:derivative:10}): um derivado nunca pula --- se
você calcula um `` unilateralderivado \omterm{def:b1:topology:closure}{limite}'' com um salto, você
diferenciaram um não diferenciável função. (4) \emph{O
necessidades de fórmula inversa $ f' \neq 0$}: $ x \mapsto x^3$ é um suave
bijeção estritamente crescente cujo inverso $ x^{1/3}$ tem um
tangente vertical em $0$ --- diferenciabilidade do inverso é
perdido exatamente onde $ f'$ desaparece
(\cref{thm:b1:derivative:inverse}).
\end{remark}

\begin{remark}[Onde o teorema do valor médio funciona a seguir]
Quase todos os quantitativos \omterm{def:b1:logic:statement}{declaração} dos próximos capítulos é este
teorema do valor médio do capítulo fantasiado: o teorema fundamental
de cálculo (\cref{ch:b1:integration}) diferencia a área
função e conclui com o critério de monotonicidade; o
Fórmula de Taylor-Lagrange (\cref{ch:b1:taylor}) é o valor médio
teorema iterado $ n $ vezes; a análise de erro do método de Newton
e de iterações de ponto fixo (\cref{exo:b1:derivative:11}) é
a forma Lipschitz; e o problema de fim de semana deste capítulo
(\cref{pb:b1:derivative:1}) transforma o mesmo \omterm{def:b1:topology:closure}{limite} de Lipschitz em
teoria dos números --- uma desigualdade de repulsão entre \omterm{pb:b1:logic:1}{algébrico}
números e racionais, produzindo o primeiro \omterm{pb:b1:derivative:1}{número transcendental}
na história. No volume do Ano 2, o valor médio
\emph{desigualdade} sobrevive em diversas variáveis quando a igualdade
não.
\end{remark}

\begin{example}[Desigualdade de Young da concavidade]\label{ex:b1:derivative:young}
Deixar $ p, q > 1$ com $\frac1p + \frac1q = 1$. Para todos $ a, b > 0$:
\[
ab \;\leq\; \frac{a^p}{p} + \frac{b^q}{q} .
\]
Prova por uma aplicação da concavidade de $\ln $ com pesos
$\frac1p, \frac1q $ (a desigualdade de Jensen de dois pontos, como em
\cref{exo:b1:derivative:9}):
\[
\ln\Bigl(\frac{a^p}{p} + \frac{b^q}{q}\Bigr)
\;\geq\; \frac1p \ln(a^p) + \frac1q \ln(b^q)
= \ln a + \ln b = \ln(ab),
\]
e $\ln $ aumentar converte a desigualdade dos logaritmos em
a reivindicação; igualdade se $ a^p = b^q $ (concavidade estrita). O caso
$ p = q = 2$ é a desigualdade aritmético-geométrica $ ab \leq
\frac{a^2 + b^2}{2}$ disfarçado. O insight final: Young's
a desigualdade é a semente algébrica do Hölder e Minkowski
desigualdades do volume do Ano 2 --- uma concavidade \omterm{def:b1:logic:statement}{declaração}
sobre $\ln $, colhido de acordo com as normas.
\end{example}

\begin{remark}[Perspectivas dentro deste volume]
O derivado adquire três novas vidas antes do volume terminar.
Em \cref{ch:b1:taylor} itera: $ n $ derivados em um ponto
comprimir em um polinomial mais um erro controlado, e o
o teorema do valor médio torna-se o resto de Lagrange. Em
\cref{ch:b1:curves}, a diferenciação torna-se geométrica: para um
curva parametrizada $ t \mapsto (x(t), y(t))$, o par $(x'(t),
y'(t))$ é uma velocidade \emph{vetor}, a tangência se torna
colinearidade e os pontos críticos tornam-se cúspides para classificação. Em
\cref{ch:b1:multivar}, uma variável é congelada por vez: parcial
derivados repita este capítulo duas vezes, e a linha tangente
cresce em um plano tangente. Todos os três capítulos herdam o mesmo
gramática --- aproximação linear local mais um termo de erro ---
falado pela primeira vez aqui.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b1:derivative:1}
Diferenciar (especificando os domínios): $ x^x $;
$\;\ln\bigl(x + \sqrt{x^2+1}\bigr)$; $\;\arctan\frac{1}{x}$;
$\;\sqrt{1 + \eu^{2x}}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:derivative:2}
Completo \cref{ex:b1:derivative:notc1}: provar isso $ f(x) = x^2
\sin\frac1x $, $ f(0) = 0$, é diferenciável no $0$ com $ f'(0) = 0$,
e isso $ f'$ não tem \omterm{def:b1:topology:closure}{limite} em $0$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:derivative:3}
Usando o teorema do valor médio ou as desigualdades tangentes, prove que
para todos $ x > 0$:
\[
\frac{x}{1 + x} < \ln(1 + x) < x .
\]
Deduzir $\lim_{n\to\infty} \bigl(1 + \frac xn\bigr)^n = \eu^x $ para
cada $ x > 0$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:derivative:4}
Deixar $ P $ seja um verdadeiro polinomial com $ k $ raízes reais distintas. Prove isso
$ P'$ tem pelo menos $ k - 1$ raízes reais distintas, entrelaçadas com aquelas
de $ P $. Deduza que se $ P $ tem todas as suas raízes reais, o mesmo acontece $ P'$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:derivative:5}
Deixar $ f $ ser diferenciável sobre $\R $ com $ f' (x)\to \ell $ como $ x \to
+\infty $. Prove isso $\frac{f(x)}{x} \to \ell $ \emph{(valor médio
teorema sobre $\intcc{A}{x}$)}. Faz $ f(x+1) - f(x) \to \ell $ segura também?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:derivative:6}
(Um Rolle discreto) Deixe $ f $ ser $ n $ vezes diferenciável sobre $ I $ e
desaparecer em $ n + 1$ pontos distintos. Prove isso $ f^{(n)}$ desaparece em
pelo menos uma vez. Aplicação: um polinomial de grau $\leq n $ desaparecendo em
$ n+1$ pontos é zero (novamente).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:derivative:7}
Deixar $ f $ ser duas vezes diferenciável sobre $\intcc{a}{b}$ com $ f(a) = f(b)
= 0$ e $ f(x_0) > 0$ para alguns \omterm{def:b1:topology:closure}{interior} $ x_0$. Prove isso $ f''(c) <
0$ para alguns $ c \in \intoo{a}{b}$. \emph{(Dois teoremas do valor médio e
uma comparação de declives.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:derivative:8}
Estude a função $ f(x) = \dfrac{\ln x}{x}$ sobre
$\intoo{0}{+\infty}$: variações, \omterm{def:b1:topology:closure}{limites}, máximo. Deduza isso $ a^b
> b^a $ para todos os reais $\eu \leq a < b $, e resolver o famoso especial
caso: qual $\eu^\pi $, $\pi^\eu $ é maior? Verifique contra o
pares inteiros pequenos $(2,3)$ e $(2,4)$: por que eles se comportam
diferentemente?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:derivative:9}
(Desigualdade aritmética-geométrica) Usando a concavidade de $\ln $ com
pesos gerais (desigualdade de Jensen para $ n $ pontos, a serem comprovados por
indução ligada $ n $), mostre que para reais positivos $ a_1, \dots, a_n $:
\[
\sqrt[n]{a_1 a_2 \cdots a_n} \leq \frac{a_1 + \dots + a_n}{n},
\]
com igualdade se tudo $ a_i $ são iguais.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:derivative:10}
(Darboux: derivados tome valores intermediários) Deixe $ f $ ser
diferenciável sobre $ I $ e $ a < b $ em $ I $ com $ f'(a) < v < f'(b)$.
Ao considerar $ g(x) = f(x) - vx $ e o ponto onde $ g $ atinge
seu mínimo em $\intcc{a}{b}$, prove que $ f'(c) = v $ para alguns $ c
\in \intoo{a}{b}$ --- embora $ f'$ não precisa ser contínuo
(\cref{exo:b1:derivative:2}).
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:derivative:11}
Deixar $ f \colon \R \to \R $ ser diferenciável com $\abs{f'(x)} \leq k
< 1$ para todos $ x $ (um \emph{contração}). Prove isso $ f $ tem exatamente
um ponto fixo $\ell $, e que toda sequência $ u_{n+1} = f(u_n)$
converge para $\ell $ com $\abs{u_n - \ell} \leq k^n\abs{u_0 -
\ell}$. \emph{(Existência: aplique o teorema do valor intermediário para
$ f(x) - x $ em um segmento grande, usando o \omterm{def:b1:topology:closure}{limite} de Lipschitz; ou usar
completude com o critério de Cauchy.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:derivative:12}
(Teorema do valor médio de Cauchy e regra de l'Hospital)
\begin{enumerate}
  \item Deixar $ f, g $ ser contínuo sobre $\intcc{a}{b}$,
        diferenciável sobre $\intoo{a}{b}$, com $ g'$ nunca zero
        lá. Prove isso $ g(b) \neq g(a)$ e que alguns $ c \in
        \intoo{a}{b}$ satisfaz
        \[
        \frac{f(b) - f(a)}{g(b) - g(a)} = \frac{f'(c)}{g'(c)}
        \]
        \emph{(aplique Rolle para $ h = f - \lambda g $ para a direita
        constante $\lambda $)}.
  \item Deduza a regra de l'Hospital no $\frac00$ formar em um
        ponto: se $ f(a) = g(a) = 0$ e
        $\frac{f'(x)}{g'(x)} \to \ell $ como $ x \to a^+$, então
        $\frac{f(x)}{g(x)} \to \ell $.
  \item Mostre que o inverso falha: para $ f(x) = x^2\sin\frac1x $
        ($ f(0) = 0$) e $ g(x) = x $, o quociente
        $\frac{f}{g}$ tem um \omterm{def:b1:topology:closure}{limite} em $0$ mas
        $\frac{f'}{g'}$ não tem nenhum.
\end{enumerate}
\end{exercise}

\section{Problema: a desigualdade de Liouville e a primeira
número transcendental}

\begin{problem}[{Problema de fim de semana --- números \omterm{pb:b1:logic:1}{algébricos} repelir
racionais: $\abs{x - p/q} \geq C/q^d $ e a transcendência de
$\sum 10^{-n!}$}]
\label{pb:b1:derivative:1}
Um número real é \emph{\omterm{pb:b1:logic:1}{algébrico}} quando é raiz de um diferente de zero
polinomial com coeficientes inteiros, e \emph{\omterm{pb:b1:logic:1}{transcendental}}
de outra forma. Em 1844, Liouville produziu o primeiro número de sempre
\emph{comprovado} \omterm{pb:b1:logic:1}{transcendental}, e o motor de sua prova é este
teorema do valor médio do capítulo: um \omterm{pb:b1:derivative:1}{número algébrico} de grau $ d $
não pode ser aproximado por racionais \omterm{def:b1:arith:prime}{melhor} do que $ C/q^d $ --- então um
número aproximado \emph{mais rápido que qualquer poder} não pode ser
\omterm{pb:b1:logic:1}{algébrico}. Este problema constrói a desigualdade, constrói
Número de Liouville $ L = 0.110001000\dots $ (aqueles no fatorial
posições, através da maquinaria de dígitos de \cref{pb:b1:reals:1}),
prova sua transcendência e termina com a prova rival de Cantor e
\omterm{def:b1:topology:closure}{limites} efetivos para $\sqrt2$ e
$2^{1/3}$.\index{número transcendental}\index{Liouville's
desigualdade}\index{número algébrico}

\medskip
\noindent\textbf{Part I --- How well can rationals be
approximated?}
\begin{enumerate}
  \item Mostre que dois racionais distintos $\frac ab \neq \frac pq $
        (escrito com $ b, q \geq 1$) satisfazer $\bigl|\frac ab -
        \frac pq\bigr| \geq \frac{1}{bq}$. Deduza: se $ x =
        \frac ab $ e $0 < \bigl|x - \frac pq\bigr| <
        \frac{1}{bq}$, não existe tal $\frac pq $ existe --- um racional
        repele todos os outros racionais em escala $\frac 1q $.
  \item Prove isso para \emph{todo} racional $\frac pq $ ($ q \geq
        1$): $\bigl|\sqrt2 - \frac pq\bigr| \geq
        \frac{1}{4q^2}$ \emph{(se a distância exceder $1$ isso
        é claro; de outra forma vinculado $\abs{\sqrt2 + p/q} < 4$ e
        use o número inteiro diferente de zero $\abs{p^2 - 2q^2} \geq 1$)}.
  \item Na outra direção: verifique se $(p, q) \mapsto (p +
        2q, p + q)$ conservas $\abs{p^2 - 2q^2} = 1$, gerar
        de $(1,1)$ os pares $(3,2)$, $(7,5)$, $(17,12)$,
        $(41,29)$, $(99,70)$, e mostre que cada um satisfaz
        \[
        \Bigl|\sqrt2 - \frac pq\Bigr| =
        \frac{1}{q^2\,(\sqrt2 + p/q)} < \frac{1}{2q^2} :
        \]
        infinitas aproximações de ordem $2$. Com
        questão 2: o expoente de aproximação de $\sqrt 2$ é
        \emph{exatamente} $2$.
  \item (Dirichlet) Deixe $ x $ ser irracional e $ N \in \N^*$.
        Considere o $ N + 1$ partes fracionárias de $0, x, 2x,
        \dots, Nx $ no $ N $ caixas $\intco{\frac kN}{\frac{k +
        1}{N}}$: pelo princípio do pombo
        (\cref{cor:b1:counting:pigeonhole}), dois caem em uma caixa.
        Deduzir $ q \leq N $ e $ p $ com $\abs{qx - p} <
        \frac 1N $, portanto, um número infinito de racionais com
        $\bigl|x - \frac pq\bigr| < \frac{1}{q^2}$: \emph{todo}
        irracional é aproximado da ordem $2$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Liouville's inequality.} Deixar $ x $ ser
irracional e \omterm{pb:b1:logic:1}{algébrico}.
\begin{enumerate}[resume]
  \item Mostre que entre o inteiro diferente de zero \omterm{def:b1:poly:def}{polinômios}
        desaparecendo em $ x $ há um, digamos $ P $ de grau $ d $,
        com \emph{sem raiz racional}; e verifique $ d \geq 2$
        \emph{(divida um fator $ X - \frac ab $ sobre $\Q $ e
        denominadores claros; grau $1$ faria $ x $
        racional)}.
  \item Mostre que para todo racional $\frac pq $ ($ q \geq 1$):
        $\bigl|P\bigl(\frac pq\bigr)\bigr| \geq \frac{1}{q^d}$
        \emph{($ q^d P(p/q)$ é um número inteiro diferente de zero)}.
  \item Deixar $ M = \max_{\intcc{x-1}{x+1}} \abs{P'}$
        (\cref{thm:b1:continuity:evt}). Usando o valor médio
        teorema entre $ x $ e $\frac pq $, provar
        \emph{A desigualdade de Liouville}: com $ C = \min\bigl(1,
        \frac 1M\bigr) > 0$,
        \[
        \Bigl| x - \frac pq \Bigr| \geq \frac{C}{q^{\,d}}
        \qquad\text{for every rational } \frac pq,\ q \geq 1 .
        \]
  \item Chamar $ x $ um \emph{Número de Liouville} quando para cada $ n \in
        \N $ existe um racional $\frac pq $ com $ q \geq 2$ e
        $0 < \bigl|x - \frac pq\bigr| < q^{-n}$. Prove que um
        O número de Liouville é irracional \emph{(pergunta 1: escolha
        $ n $ com $2^{\,n-1} > b $)}.
  \item Prove o teorema de Liouville: \emph{um número de Liouville é
        \omterm{pb:b1:logic:1}{transcendental}} \emph{(combine as questões 7 e 8: o
        desigualdade $ C < q^{\,d-n}$ falha em grande $ n $)}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The number $ L $.}
\begin{enumerate}[resume]
  \item Deixar $ L $ ser o valor (no sentido de
        \cref{pb:b1:reals:1}) da sequência de dígitos decimais com
        dígito $1$ nas posições $ n!$ ($ n = 1, 2, 3, \dots $)
        e $0$ em outro lugar, ou seja,\ $ L = \sup_k t_k $ com $ t_k =
        \sum_{n=1}^{k} 10^{-n!}$. Escreva o primeiro $25$
        dígitos de $ L $.
  \item Prove o bracketing da cauda, para cada $ k \geq 1$:
        \[
        10^{-(k+1)!} \;\leq\; L - t_k \;\leq\;
        \frac{10}{9}\,10^{-(k+1)!} \;<\; 2\cdot 10^{-(k+1)!}
        \]
        \emph{(vinculado cada soma parcial além $ t_k $ por um finito
        soma geométrica)}.
  \item Escrever $ t_k = \frac{p_k}{q_k}$ com $ q_k = 10^{k!}$.
        Mostrar $0 < L - \frac{p_k}{q_k} < \frac{2}{q_k^{\,k+1}}$,
        e concluir que $ L $ é um número de Liouville no sentido
        da questão 8.
  \item Concluir: $ L $ é \omterm{pb:b1:logic:1}{transcendental} --- o primeiro explícito
        exemplo na história (Liouville, 1844). Verifique
        irracionalidade diretamente: seus dígitos não são eventualmente
        periódico (lacunas crescentes, como em \cref{pb:b1:reals:1},
        questão 20).
  \item Generalizar: substitua cada dígito $1$ por um arbitrário
        dígito diferente de zero $ d_n \in \intint{1}{9}$. Mostre que o valor é
        ainda um número de Liouville, e deduza --- pela diagonal
        argumento de \cref{pb:b1:reals:1} (questão 22) aplicada
        para essas escolhas de dígitos --- que existem incontáveis
        muitos números transcendentais desta forma.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The hierarchy of approximation
orders.} Dizer $ x $ é \emph{próximo ao pedido $\mu $} quando para
alguma constante $ c > 0$ infinitos racionais satisfazem
$\bigl|x - \frac pq\bigr| < \frac{c}{q^{\mu}}$.
\begin{enumerate}[resume]
  \item Monte a hierarquia das Partes I-III: os racionais são
        próximo ao pedido $1$ e não \omterm{def:b1:arith:prime}{melhor}; $\sqrt 2$ para
        ordem $2$ e não \omterm{def:b1:arith:prime}{melhor}; todo irracional pedir em
        menos $2$; um \omterm{pb:b1:derivative:1}{número algébrico} de grau $ d $ sem ordem
        além $ d $; Números Liouville para cada pedido. Justificar
        cada afirmação citando a questão relevante.
  \item Mostre isso $ L + r $ é um número de Liouville para cada
        racional $ r = \frac ab $ \emph{(traduzir o
        aproximantes: os novos denominadores são $ b\,q_k $)}.
        Conclua que Liouville --- portanto \omterm{pb:b1:logic:1}{transcendental} ---
        números são \omterm{def:b1:topology:dense}{denso} em $\R $.
  \item (Cantor, 1874) Prove que o \omterm{def:b1:logic:sets}{definir} de números \omterm{pb:b1:logic:1}{algébricos}
        é contável: existem finitos números inteiros
        \omterm{def:b1:poly:def}{polinômios} com grau mais soma de $\abs{\text{coefficients}}$
        limitado por $ h $, cada um com no máximo $\deg $ raízes; um
        união contável de conjuntos finitos é contável. Já que não
        escapamentos de sequência $\R $ (\cref{pb:b1:reals:1}, pergunta
        22), números transcendentais existe --- na verdade forma um
        incontável \omterm{def:b1:logic:sets}{definir}. Compare as duas provas: o que
        A oferta de Liouville que a de Cantor não pode?
  \item Prove diretamente da questão 2 que $\sqrt 2$ é
        \emph{não} um número de Liouville \emph{(para $ n \geq 3$,
        a desigualdade $ q^{-n} > \frac{1}{4q^2}$ \omterm{def:b1:topology:closure}{limites} $ q $;
        então restam apenas um número finito de candidatos racionais, todos
        a uma distância positiva de $\sqrt2$)}. Generalizar:
        não \omterm{pb:b1:derivative:1}{número algébrico} é Liouville.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Effective constants.}
\begin{enumerate}[resume]
  \item Para o par Pell $(99, 70)$: verificar $99^2 - 2\cdot70^2
        = 1$ e avalie o erro exato
        \[
        \sqrt2 - \frac{99}{70}
        = \frac{-1}{70^2\,\bigl(\sqrt2 + \frac{99}{70}\bigr)},
        \qquad
        \Bigl|\sqrt 2 - \frac{99}{70}\Bigr| \approx 7.2\cdot
        10^{-5} :
        \]
        cinco dígitos corretos de uma fração de três dígitos.
  \item Execute a Parte II em $ x = 2^{1/3}$, $ P = X^3 - 2$: verificar $ P $
        não tem raiz racional, limitado $ M = \max_{\intcc{x-1}{x+1}}
        3t^2 \leq 3\,(1 + 2^{1/3})^2 < 16$, e concluir o
        desigualdade efetiva
        \[
        \Bigl| 2^{1/3} - \frac pq \Bigr| \geq
        \frac{1}{16\,q^3} \qquad \text{for all } \frac pq .
        \]
  \item Resultado: mostre que qualquer aproximação racional $2^{1/3}$
        dentro de $10^{-6}$ deve ter denominador $ q \geq 40$.
  \item Mostre que a base $10$ é irrelevante: o binário
        analógico $\sum_{n\geq1} 2^{-n!}$ (valor do binário
        string com aqueles em posições fatoriais) também é um
        Número de Liouville, portanto \omterm{pb:b1:logic:1}{transcendental}.
\end{enumerate}

\noindent\textbf{Part VI --- Frontiers and synthesis.}
\begin{enumerate}[resume]
  \item Deixar $ x^\dagger $ seja o valor da string decimal com
        exatamente nas posições $3^k $ ($ k \geq 0$). Mostrar
        $ x^\dagger $ é próximo do pedido $3$, e deduza
        da desigualdade de Liouville que $ x^\dagger $ não é nenhum dos dois
        racional nem um irracional quadrático. Explique por que
        método para aí: pedido $3$ é compatível com
        algebricidade de grau $\geq 3$ e fechando essa lacuna
        (qualquer expoente $> 2$ é suficiente, para cada \omterm{pb:b1:logic:1}{algébrico}
        número) é o teorema de Roth, muito além deste volume.
  \item Quantifique Cantor: mostre que o números \omterm{pb:b1:logic:1}{algébricos} de
        grau $\leq d $ dado por \omterm{def:b1:poly:def}{polinômios} com coeficientes
        em $\intint{-H}{H}$ número no máximo $ d\,(2H + 1)^{d+1}$.
        (Essa finitude é o que fez a questão 17 funcionar.)
  \item Síntese, uma frase cada: (i) localizar o único
        ingrediente analítico da prova de Liouville (cujo teorema
        deste capítulo, usado onde); (ii) indicar a tensão
        que o alimenta (forças de integralidade $\abs{P(p/q)} \geq
        q^{-d}$, suavidade proíbe $\abs{P(p/q)} > M\abs{x -
        p/q}$); (iii) contrastar as provas de Liouville e Cantor
        da existência de números transcendentais; (iv) nome
        onde este volume reencontra o tema --- o fim de semana
        problema de \cref{ch:b1:integration} prova $\pi$
        irracional pela mesma integralidade versus pequenez
        squeeze, com integrais no lugar de derivados.
\end{enumerate}
\end{problem}
