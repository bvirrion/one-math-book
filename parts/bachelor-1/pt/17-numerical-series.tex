\chapter{Série Numérica}\label{ch:b1:series}

Somar infinitos números significa tomar o \omterm{def:b1:topology:closure}{limite} da parcial
somas --- nada mais, nada menos. Este capítulo conjuntos até o
definições e testes de convergência utilizáveis no primeiro ano:
comparação e equivalentes para termos positivos, o \omterm{thm:b1:series:ratio}{teste de razão}, o
integrante comparação dando a \omterm{thm:b1:series:riemann}{Série Riemann}, convergência absoluta,
e o \omterm{thm:b1:series:alternating}{série alternada} teorema. A teoria mais refinada (produtos de
\omterm{def:b1:series:def}{série}, somatório por pacotes, \omterm{def:b1:series:def}{série} de funções) pertence ao
segundo ano.

\section{Generalidades}

\begin{definition}\label{def:b1:series:def}
Dada uma sequência $(u_n)$, o \emph{série}\index{série} $\sum u_n $
é a sequência de \emph{somas parciais} $ S_N = \sum_{n=0}^{N} u_n $.
A série \emph{converge} quando $(S_N)$ converge; o \omterm{def:b1:topology:closure}{limite} é o
\emph{soma} $\sum_{n=0}^{\infty} u_n $, e $ R_N = \sum_{n > N} u_n =
S - S_N $ é o \emph{restante}, o que tende a $0$.
\end{definition}

\begin{example}[Série geométrica]\label{ex:b1:series:geometric}
Para $ q \in \C $: $\;S_N = \sum_{n=0}^{N} q^n = \frac{1 -
q^{N+1}}{1-q}$ ($ q \neq 1$). O \omterm{def:b1:series:def}{série} converge se e se $\abs q < 1$
(\cref{exo:b1:seq:3}), com\index{série geométrica}
\[
\sum_{n=0}^{\infty} q^n = \frac{1}{1 - q} .
\]
\end{example}

\begin{example}[Decimais periódicos são séries geométricas]\label{ex:b1:series:periodicdecimal}
Que número é $0.363636\dots $? Sua própria escrita é um \omterm{def:b1:series:def}{série}:
\[
0.\overline{36} = \sum_{k=1}^{\infty} \frac{36}{100^k}
= 36\cdot\frac{1/100}{1 - 1/100} = \frac{36}{99} =
\frac{4}{11} ,
\]
pela soma geométrica com $ q = \frac{1}{100}$. Em geral um
bloquear $ B $ de $ p $ dígitos repetidos para sempre valem
$\frac{B}{10^p - 1}$ --- o mecanismo por trás da periodicidade
critério de \cref{pb:b1:reals:1}, qual a linguagem deste capítulo
finalmente afirma em uma linha: a \omterm{pb:b1:reals:1}{expansão decimal} é um convergente
\omterm{def:b1:series:def}{série}, eventualmente periódico exatamente quando sua soma é racional.
A maquinaria digital do Capítulo 10, construída ali com
\omterm{def:b1:reals:bounds}{supremo}, foi \omterm{def:b1:series:def}{série} teoria viajando incógnita.
\end{example}

\begin{proposition}[Primeiros fatos]\label{prop:b1:series:first}
\begin{enumerate}
  \item Se $\sum u_n $ converge, então $ u_n \to 0$. (O inverso é
        \emph{falso}: o harmônico \omterm{def:b1:series:def}{série}.)
  \item Linearidade: convergente \omterm{def:b1:series:def}{série} adicionar e dimensionar, com o esperado
        somas.
  \item (Telescópico\index{série telescópica}) $\sum (v_{n+1} -
        v_n)$ converge se e se $(v_n)$ converge, com soma $\lim v_n -
        v_0$.
  \item Alterar um número finito de termos não afeta a convergência
        (somente a soma).
\end{enumerate}
\end{proposition}

\begin{proof}
(1) $ u_N = S_N - S_{N-1} \to S - S = 0$. O harmônico \omterm{def:b1:series:def}{série} tem
$ u_n = \frac1n \to 0$ ainda diverge (\cref{exo:b1:seq:5}). (2)
Operações sobre \omterm{def:b1:topology:closure}{limites}. (3) $ S_N = v_{N+1} - v_0$. (4) O parcial
as somas mudam por um valor eventualmente constante.
\end{proof}

\begin{example}[Planejando dígitos com resto geométrico]\label{ex:b1:series:georemainder}
Para $\abs q < 1$ o restante do \omterm{ex:b1:series:geometric}{série geométrica} é
explícito:
\[
R_N = \sum_{n = N+1}^{\infty} q^n = \frac{q^{N+1}}{1 - q} .
\]
Isso converte metas de precisão em contagens de prazos antes de qualquer
computação. Para avaliar $\sum_{n\geq0} \bigl(\frac13\bigr)^n =
\frac32$ dentro de $10^{-10}$: precisar $\frac{(1/3)^{N+1}}{2/3} \leq
10^{-10}$, ou seja\ $3^{N} \geq \frac{3}{2}\cdot 10^{10}$, ou seja\ $ N
\geq 22$ (como $3^{22} \approx 3.1\cdot10^{10}$): vinte e três
termos, conhecidos antecipadamente. Cada estimativa de taxa geométrica do
problemas de fim de semana (o $\frac13$-série para $\ln 2$, Máquina
\omterm{def:b1:functions:arc}{arco tangentes} em \cref{pb:b1:taylor:1}) é este orçamento de duas linhas
em traje profissional.
\end{example}

\begin{example}[Um telescópio mais longo]\label{ex:b1:series:telescope3}
Calcular $\sum_{n\geq1} \frac{1}{n(n+1)(n+2)}$. Parcial
frações (\cref{ch:b1:fractions}):
\[
\frac{1}{n(n+1)(n+2)}
= \frac{1/2}{n} - \frac{1}{n+1} + \frac{1/2}{n+2}
= \frac12\Bigl(\frac{1}{n(n+1)} - \frac{1}{(n+1)(n+2)}\Bigr),
\]
onde a segunda forma --- uma diferença de valores consecutivos de
$ w_n = \frac{1}{n(n+1)}$ --- é o telescópico. Por isso
\[
\sum_{n=1}^{N} \frac{1}{n(n+1)(n+2)}
= \frac12\Bigl(w_1 - w_{N+1}\Bigr)
= \frac12\Bigl(\frac12 - \frac{1}{(N+1)(N+2)}\Bigr)
\longrightarrow \frac14 .
\]
O insight final: frações parciais de três termos raramente
telescópio conforme escrito; reagrupá-los em uma diferença $ w_n -
w_{n+1}$ primeiro --- a recompensa não é apenas a convergência, mas a
soma exata, que nenhum teste de comparação jamais fornece.
\end{example}

\section{Série com termos não negativos}

\begin{theorem}[Somas parciais limitadas]\label{thm:b1:series:positive}
Se $ u_n \geq 0$ para todos $ n $, as somas parciais aumentam, então: $\sum
u_n $ converge $\iff $ suas somas parciais são limitadas acima. Daí o
\emph{teste de comparação}: se $0 \leq u_n \leq v_n $ para todos (grande)
$ n $,
\[
\sum v_n \text{ converges} \implies \sum u_n \text{ converges},
\qquad
\sum u_n \text{ diverges} \implies \sum v_n \text{ diverges}.
\]
E o \emph{teste de equivalentes}: se $ u_n \sim v_n $ com $ v_n \geq
0$, os dois \omterm{def:b1:series:def}{série} têm a mesma natureza.
\end{theorem}

\begin{proof}
Teorema do \omterm{def:b1:topology:closure}{limite} monótono (\cref{thm:b1:seq:monotone}) pela primeira
ponto; comparação de somas parciais para o segundo. Equivalentes: para
grande $ n $, $\frac12 v_n \leq u_n \leq 2 v_n $ (definição de $\sim $
com $\varepsilon = \frac12$), e a comparação se aplica nos dois sentidos.
\end{proof}

\begin{example}[Um equivalente que prova a divergência]\label{ex:b1:series:equivdiverge}
Natureza de $\sum_{n\geq1} n\sin\dfrac{1}{n^2}$? Desde
$\frac{1}{n^2} \to 0$ e $\sin h \sim h $ no $0$:
\[
n\sin\frac{1}{n^2} \;\sim\; n\cdot\frac{1}{n^2} = \frac1n ,
\]
e o teste de equivalentes transfere a divergência do
harmônico \omterm{def:b1:series:def}{série}: divergente --- embora os termos tendam a
$0$. Uma expansão, uma escala, um veredicto; as mesmas duas etapas
padrão (equivalente, então Riemann ou pesquisa geométrica) decide
todos os quatro \omterm{def:b1:series:def}{série} de \cref{exo:b1:series:3}.
\end{example}

\begin{example}[O teste de equivalentes em uma linha]\label{ex:b1:series:equivtest}
Natureza de $\sum_{n \geq 1} \frac{\sqrt{n+1} - \sqrt n}{n}$?
Conjugue o numerador:
\[
\frac{\sqrt{n+1} - \sqrt n}{n}
= \frac{1}{n\,(\sqrt{n+1} + \sqrt n)}
\sim \frac{1}{2\,n^{3/2}} ,
\]
uma escala convergente de Riemann ($\alpha = \frac32 > 1$): o \omterm{def:b1:series:def}{série}
converge. Toda a decisão exigiu um equivalente e uma pesquisa
--- desde que os termos não sejam negativos, o que realmente são. O
visão final: para positivo \omterm{def:b1:series:def}{série}, toda a convergência
teoria é uma \emph{dicionário de escalas} ($ n^{-\alpha}$, $ q^n $,
$\frac{1}{n(\ln n)^\alpha}$) mais a licença para substituir um termo
por um equivalente; o trabalho analítico está nos assintóticos
(\cref{ch:b1:taylor}), nunca no somatório.
\end{example}

\begin{theorem}[Comparação integral; Série Riemann]\label{thm:b1:series:riemann}
Deixar $ f $ ser contínuo, não negativo e \emph{diminuindo} sobre
$\intco{1}{+\infty}$. Então
\[
\int_1^{N+1} f(t)\,\dd t \;\leq\; \sum_{n=1}^{N} f(n) \;\leq\; f(1)
+ \int_1^{N} f(t)\,\dd t ,
\]
então $\sum f(n)$ converge se e se $\bigl(\int_1^x f\bigr)$ é limitado. Em
especial, para $\alpha \in \R $:\index{Série Riemann}
\[
\sum_{n \geq 1} \frac{1}{n^\alpha} \text{ converges}
\iff \alpha > 1,
\]
e $\sum_{n=1}^{N} \frac1n = \ln N + O(1)$.
\end{theorem}

\begin{proof}
Para $ n \leq t \leq n+1$, a monotonicidade dá $ f(n+1) \leq f(t)
\leq f(n)$; integrando sobre $\intcc{n}{n+1}$ (um segmento de
comprimento $1$):
\[
f(n+1) \;\leq\; \int_n^{n+1} f(t)\,\dd t \;\leq\; f(n) .
\]
Somando as desigualdades corretas para $ n = 1, \dots, N-1$ dá
$\int_1^{N} f \leq \sum_{n=1}^{N-1} f(n)$, daí a parte superior
enquadramento depois de adicionar $ f(N) \leq f(1)$; somando os da esquerda para
$ n = 1, \dots, N $ dá $\sum_{n=2}^{N+1} f(n) \leq
\int_1^{N+1} f $, que após a reindexação é o enquadramento inferior.
Convergência: as somas parciais e a integrais $\int_1^x f $
ligam-se entre si dentro da constante $ f(1)$, e ambos são
não decrescente, então um é limitado se o outro for
(\cref{thm:b1:series:positive}). Para $ f(t)
= t^{-\alpha}$ ($\alpha \neq 1$): $\int_1^x t^{-\alpha}\dd t =
\frac{x^{1-\alpha} - 1}{1 - \alpha}$, limitado se $\alpha > 1$; para
$\alpha = 1$ o integrante é $\ln x \to \infty $, e o enquadramento
dá $\ln(N+1) \leq H_N \leq 1 + \ln N $. Para $\alpha \leq 0$ o
termos não tendem a $0$.
\end{proof}

\begin{example}[A pilha harmônica]\label{ex:b1:series:harmonicstack}
Quantos termos deve o harmônico \omterm{def:b1:series:def}{série} acumular para passar $20$?
O enquadramento $\ln(N+1) \leq H_N \leq 1 + \ln N $ responde sem
somatório em tudo: $ H_N \geq 20$ requer $1 + \ln N \geq 20$,
ou seja\ $ N \geq \eu^{19} \approx 1.8\cdot10^{8}$, e é garantido
uma vez $\ln(N + 1) \geq 20$, ou seja\ $ N \approx \eu^{20} \approx
4.9\cdot10^{8}$. (O problema do fim de semana aguça isso para $ N \approx
\eu^{20 - \gamma} \approx 2.7\cdot10^{8}$ através de \omterm{ex:b1:series:oddtelescope}{Constante de Euler}.)
O insight final: o integrante a comparação não se limita
decidir convergência --- isso \emph{localiza} somas parciais com
precisão logarítmica, transformando uma computação sem esperança
(centenas de milhões de termos) em uma estimativa de duas linhas.
\end{example}

\begin{theorem}[Teste de proporção (d'Alembert)]\label{thm:b1:series:ratio}
Deixar $ u_n > 0$ com $\frac{u_{n+1}}{u_n} \to \ell $.\index{teste de razão}
\begin{itemize}
  \item Se $\ell < 1$: $\sum u_n $ converge;
  \item se $\ell > 1$: $ u_n \to +\infty $, divergência;
  \item se $\ell = 1$: nenhuma conclusão ($\sum \frac1n $ diverge, $\sum
        \frac{1}{n^2}$ converge).
\end{itemize}
\end{theorem}

\begin{proof}
Se $\ell < 1$, consertar $ q \in \intoo{\ell}{1}$: além de alguns $ N $,
$ u_{n+1} \leq q\,u_n $, então $ u_n \leq u_N q^{\,n-N}$ por indução:
comparação com um \omterm{ex:b1:series:geometric}{série geométrica}. Se $\ell > 1$: além de alguns $ N $
a sequência $(u_n)$ está aumentando, então não pode tender a $0$
(seu \omterm{def:b1:topology:closure}{limite}, se houver, é $\geq u_N > 0$); por
\cref{prop:b1:series:first}~(1), divergência --- e de fato $ u_n
\geq u_N q^{n-N}$ com $ q > 1$ dá $ u_n \to \infty $.
\end{proof}

\begin{example}\label{ex:b1:series:ratio}
$\sum \frac{x^n}{n!}$ converge para cada $ x > 0$: razão
$\frac{x}{n+1} \to 0$. Sua soma é $\eu^x $: por Taylor--Lagrange
(\cref{thm:b1:taylor:lagrange}) sobre $\intcc{0}{x}$,
\[
\Bigl| \eu^x - \sum_{k=0}^{n} \frac{x^k}{k!} \Bigr|
\leq \eu^{x}\, \frac{x^{n+1}}{(n+1)!} \xrightarrow[n\to\infty]{} 0 ,
\]
o \omterm{def:b1:topology:closure}{limite} tendendo a $0$ porque o fatorial domina
(\cref{exo:b1:integration:9}~(1) usou o mesmo fato). O mesmo
argumento resume o $\sin $, $\cos $, $\sinh $, $\cosh $ \omterm{def:b1:series:def}{série} em todos
$\R $.
\end{example}

\begin{example}[O teste da razão é suficiente, não necessário]\label{ex:b1:series:rationotnecessary}
Deixar $ u_n = 2^{-n}$ até mesmo $ n $ e $ u_n = 2^{-n-2}$ por estranho
$ n $. As relações consecutivas oscilam entre $\frac{1}{8}$ e
$\frac12\cdot4 = 2$, então $\frac{u_{n+1}}{u_n}$ não tem \omterm{def:b1:topology:closure}{limite} e
d'Alembert está mudo --- ainda $ u_n \leq 2^{-n}$ e a comparação
o teste estabelece a convergência instantaneamente. A hipótese do teste (a
proporção \emph{converge}) é uma restrição real: atende aos termos
com uma estrutura multiplicativa dominante (fatoriais, potências),
e falha em qualquer coisa que respire. Quando as proporções se comportam mal,
recuar para a comparação com um envelope geométrico --- que
é tudo o \omterm{thm:b1:series:ratio}{teste de razão} sempre existiu, como mostra sua prova.
\end{example}

\begin{example}[Teste de proporção em batalhas fatoriais]\label{ex:b1:series:ratiobinom}
Natureza de $\sum_{n\geq0} \dfrac{(n!)^2}{(2n)!}$ (recíprocos de
o centro coeficientes binomiais, até o fator $ n + 1$)?
A proporção colapsa os fatoriais:
\[
\frac{u_{n+1}}{u_n}
= \frac{((n+1)!)^2}{(n!)^2}\cdot\frac{(2n)!}{(2n+2)!}
= \frac{(n+1)^2}{(2n+1)(2n+2)}
\longrightarrow \frac14 < 1 :
\]
convergente, com espaço de sobra --- os termos decaem essencialmente
gosto $4^{-n}$, consistente com $\binom{2n}{n} \geq
\frac{4^n}{2n+1}$ de \cref{pb:b1:integration:1}. O fechamento
insight: quocientes de fatoriais são exatamente o que \omterm{thm:b1:series:ratio}{teste de razão}
digests --- todo fatorial é cancelado em uma função racional de
$ n $, cujo \omterm{def:b1:topology:closure}{limite} é lido nos termos iniciais.
\end{example}

\section{Convergência absoluta; série alternada}

\begin{theorem}[Convergência absoluta]\label{thm:b1:series:absolute}
Se $\sum \abs{u_n}$ converge (\emph{absoluto
convergência}\index{absoluto
convergência}), então $\sum u_n $ converge,
e $\bigl|\sum u_n\bigr| \leq \sum \abs{u_n}$. Isso vale de verdade
ou termos complexos.
\end{theorem}

\begin{proof}
As somas parciais satisfazem, por $ M > N $ (critério de Cauchy,
\cref{thm:b1:seq:complete}):
\[
\abs{S_M - S_N} = \Bigl| \sum_{n=N+1}^{M} u_n \Bigr|
\leq \sum_{n=N+1}^{M} \abs{u_n},
\]
que é pequeno para grande $ N $ já que as somas parciais de $\sum\abs{u_n}$
formar um \omterm{def:b1:seq:cauchy}{Sequência de Cauchy}. Então $(S_N)$ é Cauchy, portanto convergente. O
a desigualdade passa ao \omterm{def:b1:topology:closure}{limite} da desigualdade triangular finita.
\end{proof}

\begin{example}[Convergência absoluta, real e complexa]\label{ex:b1:series:absexamples}
$\sum_{n\geq1} \frac{\sin n}{n^2}$: os termos mudam de sinal
erraticamente (na verdade $(\sin n)$ é \omterm{def:b1:topology:dense}{denso} em $\intcc{-1}{1}$,
\cref{exo:b1:seq:12}), e nenhuma estrutura alternada está em
vista. Convergência absoluta resgata tudo de uma vez:
$\bigl|\frac{\sin n}{n^2}\bigr| \leq \frac{1}{n^2}$, um
escala convergente, então o \omterm{def:b1:series:def}{série} converge. O mesmo escudo funciona
acabou $\C $: $\sum_{n\geq1}\frac{\eu^{\iu n}}{n^2}$ converge
porque $\bigl|\frac{\eu^{\iu n}}{n^2}\bigr| = \frac{1}{n^2}$
--- padrões de sinais, mesmo os bidimensionais, são irrelevantes
uma vez que os \omterm{def:b1:complex:field}{módulos} são somáveis. A visão final: \omterm{thm:b1:series:absolute}{absoluto
convergência} é a única ferramenta deste capítulo que nunca pergunta como
os sinais estão organizados; experimente primeiro
(\cref{met:b1:series:decide}), e reserve os delicados testes
para o \omterm{def:b1:series:def}{série} que falham.
\end{example}

\begin{theorem}[Teste de série alternada]\label{thm:b1:series:alternating}
Deixar $(a_n)$ estar diminuindo com $ a_n \to 0$. Então o alternando
\omterm{def:b1:series:def}{série} $\sum (-1)^n a_n $ converge; sua soma está entre quaisquer dois
somas parciais consecutivas, e\index{série alternada}
\[
\abs{R_N} = \Bigl| \sum_{n > N} (-1)^n a_n \Bigr| \leq a_{N+1} .
\]
\end{theorem}

\begin{proof}
As somas parciais pares e ímpares são adjacentes: $ S_{2p+2} - S_{2p} =
a_{2p+2} - a_{2p+1} \leq 0$ (diminuindo), $ S_{2p+1} - S_{2p-1} =
a_{2p} - a_{2p+1} \geq 0$ (aumentando) e $ S_{2p} - S_{2p+1} =
a_{2p+1} \to 0$. Por \cref{thm:b1:seq:adjacent} eles compartilham um \omterm{def:b1:topology:closure}{limite}
$ S $, que os dois \omterm{def:b1:seq:subsequence}{subsequências} critério
(\cref{prop:b1:seq:subsequences}) faz o \omterm{def:b1:topology:closure}{limite} de $(S_N)$;
além disso $ S $ está preso entre somas parciais consecutivas, e
$\abs{S - S_N}$ é no máximo a diferença para o próximo, $ a_{N+1}$.
\end{proof}

\begin{example}[Série harmônica alternada]\label{ex:b1:series:ln2}
$\sum_{n \geq 1} \frac{(-1)^{n-1}}{n}$ converge (teste alternado)
mas não absolutamente (harmônico \omterm{def:b1:series:def}{série}). Sua soma é $\ln 2$: do
identidade geométrica finita $\frac{1}{1+t} = \sum_{k=0}^{n-1} (-t)^k +
\frac{(-t)^n}{1+t}$, integrar sobre $\intcc{0}{1}$:
\[
\ln 2 = \sum_{k=1}^{n} \frac{(-1)^{k-1}}{k}
+ (-1)^n \int_0^1 \frac{t^n}{1+t}\,\dd t ,
\qquad
0 \leq \int_0^1 \frac{t^n}{1+t}\,\dd t \leq \frac{1}{n+1} \to 0 .
\]
A convergência é dolorosamente lenta ($ R_N \approx \frac{1}{N}$) ---
\omterm{thm:b1:series:alternating}{série alternada} convergem por cancelamento, não por pequenez.
\end{example}

\begin{remark}[Armadilhas comuns com séries]
(eu) \emph{O teste de equivalentes precisa de um sinal}: deixar $ v_n =
\frac{(-1)^n}{\sqrt n}$ e $ u_n = v_n + \frac1n $. Então
$\frac{u_n}{v_n} = 1 + \frac{(-1)^n}{\sqrt n} \to 1$, então $ u_n
\sim v_n $; ainda $\sum v_n $ converge (teste alternado) enquanto
$\sum u_n = \sum v_n + \sum \frac1n $ diverge. Equivalência
controla o \emph{tamanho} de termos e para assinatura \omterm{def:b1:series:def}{série} tamanho
não é destino --- o teste é declarado, e verdadeiro, para
(eventualmente) apenas termos não negativos. (ii) \emph{$ u_n \to 0$
não prova nada}: o harmônico \omterm{def:b1:series:def}{série} é o eterno
contraexemplo; a direção inversa
(\cref{prop:b1:series:first}~(1)) é apenas uma divergência rápida
teste. (iii) \emph{\omterm{def:b1:topology:closure}{Limite} de proporção $1$ é silêncio, não
convergência}: ambos $\sum\frac1n $ e $\sum\frac{1}{n^2}$ tem
proporção $\to 1$; mudar para escalas Riemann ou integrante comparação.
(4) \emph{Necessidades alternadas diminuindo}: $\sum
\frac{(-1)^n}{n + (-1)^n}$ parece alternado e é tratado
apenas por expansão (\cref{exo:b1:series:5}); o problema do fim de semana
de \cref{ch:b1:taylor} (pergunta 23 ali) mostra que o teste pode
falhar completamente sem monotonicidade. (v) \emph{Agrupamento e
reordenar não é gratuito}: inserir parênteses é inofensivo para
convergente \omterm{def:b1:series:def}{série} mas pode criar convergência a partir da divergência
($1 - 1 + 1 - \cdots $ agrupados em pares), e a reordenação pode
mudar a própria soma --- o drama encenado neste capítulo
problema de fim de semana (\cref{pb:b1:series:1}).
\end{remark}

\begin{method}[Decidindo a natureza de uma série]\label{met:b1:series:decide}
\begin{enumerate}
  \item Faz $ u_n \to 0$? Se não, divergência, pare.
  \item Termos não negativos: procure um equivalente de $ u_n $ (expansões,
        \cref{ch:b1:taylor}!), compare com Riemann ou geométrico
        escalas; fatoriais e potências exigem o \omterm{thm:b1:series:ratio}{teste de razão};
        diminuindo $ f(n)$ pede integrante comparação.
  \item Os sinais variam: experimente convergência absoluta primeiro; se falhar, o
        teste alternado (verificar \emph{diminuindo} com cuidado); além
        isso, ferramentas do segundo ano.
\end{enumerate}
\end{method}

\begin{example}[Denominadores ímpares, metade do telescópio]\label{ex:b1:series:oddtelescope}
Calcular $\sum_{n \geq 1} \dfrac{1}{4n^2 - 1}$. Parcial
frações: $\frac{1}{(2n-1)(2n+1)} =
\frac12\bigl(\frac{1}{2n-1} - \frac{1}{2n+1}\bigr)$, então
\[
\sum_{n=1}^{N} \frac{1}{4n^2 - 1}
= \frac12\Bigl(1 - \frac{1}{2N+1}\Bigr)
\longrightarrow \frac12 .
\]
Compare com $\sum \frac{1}{n(n+1)} = 1$
(\cref{exo:b1:series:1}): mesmo esqueleto telescópico, mas o
termos consecutivos aqui estão separados por dois nos números ímpares, e
o fator $\frac12$ registra a etapa. A visão final:
telescópica é uma mudança de ponto de vista, não um truque --- sempre
o termo geral é uma diferença $ w_n - w_{n+1}$ de uma sequência
com um \omterm{def:b1:topology:closure}{limite}, a soma é $ w_1 - \lim w $, exatamente
\cref{prop:b1:series:first}~(3).
\end{example}

\begin{remark}[O pipeline de análise, em retrospecto]
Este capítulo é onde converge a análise do volume, e cada
test nomeia seu ancestral. Somas parciais limitadas são o monótono
teorema do \omterm{def:b1:topology:closure}{limite} (\cref{ch:b1:seq}), em si o axioma da completude
de \cref{ch:b1:reals}; convergência absoluta é o Cauchy
critério; o integrante teste é \cref{ch:b1:integration}de
enquadramento de áreas; equivalentes de termos gerais são
\cref{ch:b1:taylor}expansões; e o teorema da alternância é
o lema das \omterm{thm:b1:seq:adjacent}{sequências adjacentes} em sua roupagem de domingo. Leia
ao contrário, o pipeline explica o que cada capítulo foi
\emph{para} --- e os problemas do fim de semana passando por ele
($ b $-dígitos ádicos, Cesàro--Stolz, as máquinas da irracionalidade,
\omterm{ex:b1:series:oddtelescope}{Constante de Euler}) are the same few ideas meeting at higher and
altitude mais elevada. The linear algebra that follows changes
subject, not standards: the habit of exact declarações com
o erro certificado sobrevive à passagem dos \omterm{def:b1:topology:closure}{limites} para as dimensões.
\end{remark}

\begin{remark}[Para onde vai a próxima série]
Este capítulo encerra a análise do volume e abre três
portas. No volume do Ano 2, \omterm{def:b1:series:def}{série} adquirir uma variável ($\sum
a_n x^n $: poder \omterm{def:b1:series:def}{série}, com seu raio de convergência) e
então uma teoria de valor de função (Fourier \omterm{def:b1:series:def}{série}); a dicotomia
convergência absoluta versus condicional, dramatizada no
problema de fim de semana abaixo, torna-se a pedra angular de ambos. Em
probabilidade (volume do ano 3), expectativas de eventos aleatórios discretos
variáveis \emph{são} \omterm{def:b1:series:def}{série}, e convergência absoluta é o que
os torna bem definidos. E o \omterm{thm:b1:series:riemann}{Série Riemann} $\sum n^{-s}$,
empurrado para complexo $ s $, torna-se a função zeta --- o único
mais estudado \omterm{def:b1:series:def}{série} em matemática.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b1:series:1}
Natureza (e soma, quando telescópica) de:
\[
\sum_{n\geq1} \frac{1}{n(n+1)},
\qquad
\sum_{n\geq2} \ln\Bigl(1 - \frac{1}{n^2}\Bigr),
\qquad
\sum_{n\geq0} \frac{3^n + 4^n}{5^n} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:series:2}
Natureza de: $\;\sum \dfrac{n^2}{2^n}$; $\;\sum \dfrac{n!}{n^n}$;
$\;\sum \dfrac{2^n\,n!}{n^n}$; $\;\sum \dfrac{3^n\,n!}{n^n}$.
\emph{(Teste de proporção; lembrar $\bigl(1 + \frac1n\bigr)^n \to \eu $.)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:series:3}
Natureza de: $\;\sum \sin\dfrac{1}{n^2}$; $\;\sum
\Bigl(1 - \cos\dfrac1n\Bigr)$; $\;\sum \dfrac{1}{\sqrt{n(n+1)}}$;
$\;\sum \dfrac{\ln n}{n^2}$ \emph{(comparar com $ n^{-3/2}$)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:series:4}
Prove isso $\sum_{n\geq1} \frac{1}{n^2}$ converge com a soma $\leq 2$,
usando $\frac{1}{n^2} \leq \frac{1}{n(n-1)}$ para $ n \geq 2$ e um
telescópico limitado.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:series:5}
Natureza de $\;\sum \dfrac{(-1)^n}{\sqrt n}$, de $\;\sum
\dfrac{(-1)^n}{n + (-1)^n}$ \emph{(expandir: o teste alternado faz
não se aplica diretamente --- por quê?)}, e de $\;\sum
\sin\bigl(\pi\sqrt{n^2+1}\,\bigr)$ \emph{(reduzir \omterm{def:b1:complex:field}{módulo} $\pi $:
$\sqrt{n^2+1} = n + \frac{1}{2n} + O(n^{-3})$)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:series:6}
Para qual $\alpha > 0$ faz $\sum_{n \geq 2}
\dfrac{1}{n (\ln n)^{\alpha}}$ convergem? \emph{(Integrante comparação;
substituir $ u = \ln t $.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:series:7}
Deixar $ u_n = \dfrac{1}{n} - \ln\Bigl(1 + \dfrac1n\Bigr)$. Prove isso
$0 \leq u_n \leq \dfrac{1}{2n^2}$, que $\sum u_n $ converge, e
deduzir a existência de \omterm{ex:b1:series:oddtelescope}{Constante de Euler}\index{Constante de Euler}:
\[
\gamma = \lim_{N \to \infty} \Bigl( \sum_{n=1}^{N} \frac 1n - \ln N
\Bigr) .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:series:8}
Calcule as somas
\[
\sum_{n=1}^{\infty} \frac{1}{n(n+2)}
\qquad\text{and}\qquad
\sum_{n=0}^{\infty} \frac{n}{2^n} .
\]
\emph{(Para o primeiro: frações parciais. Para o segundo: calcule
$\sum_{n=1}^{N} n x^{n-1}$ de forma fechada e deixe $ N \to \infty $ no
$ x = \frac12$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:series:9}
(Condensação de Cauchy) Deixe $(u_n)$ ser não negativo e decrescente.
Prove isso
\[
\sum_{n \geq 1} u_n \text{ converges}
\iff
\sum_{k \geq 0} 2^k\, u_{2^k} \text{ converges},
\]
comparando pacotes de termos entre potências consecutivas de $2$.
Recupere dele o critério de Riemann e
\cref{exo:b1:series:6}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:series:10}
Usando o integrante identidade de \cref{ex:b1:series:ln2} adaptado para
$\frac{1}{1+t^2}$, prove a fórmula de Leibniz
\[
\frac{\pi}{4} = \sum_{n=0}^{\infty} \frac{(-1)^n}{2n+1}
= 1 - \frac13 + \frac15 - \frac17 + \cdots
\]
com o erro vinculado $\abs{R_N} \leq \frac{1}{2N+3}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:series:11}
Natureza de $\displaystyle\sum_{n \geq 1} \frac{1}{n^{1 + 1/n}}$.
\emph{(Calcule o \omterm{def:b1:topology:closure}{limite} de $ n^{1/n}$ e encontre um equivalente de
o termo geral: o teste de Riemann precisa de um \emph{fixed}
expoente.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:series:12}
Deixar $(u_n)$ ser não negativo e \emph{diminuindo} com $\sum u_n $
convergente. Prove isso $ n\,u_n \to 0$ \emph{(vinculado $ n\,u_{2n}$
por uma fatia $\sum_{k=n+1}^{2n} u_k $ e use o Cauchy
critério)}. Mostre que a recíproca falha e que a
A hipótese de monotonicidade não pode ser removida.
\end{exercise}

\section{Problema: Constante de Euler e a série que muda
sua soma}

\begin{problem}[{Problema de fim de semana --- $ H_n = \ln n + \gamma +
\frac{1}{2n} + O(n^{-2})$ e reorganizando $1 - \frac12 +
\frac13 - \dots $ pára $\frac{\ln 2}{2}$}]
\label{pb:b1:series:1}
Duas histórias compartilham a harmônica \omterm{def:b1:series:def}{série}. Primeiro, o exato
escrituração de sua divergência: $ H_n - \ln n $ converge para \omterm{pb:b1:series:1}{Euler
constante} $\gamma $ (\cref{exo:b1:series:7}), e esse problema
afia o \omterm{def:b1:logic:statement}{declaração} em uma lei bilateral $\frac{1}{2(n+1)}
\leq H_n - \ln n - \gamma \leq \frac{1}{2n}$, certificando $\gamma
= 0.5772\dots $ à mão. Em segundo lugar, o escândalo das condicionalidades
convergência: o harmônico alternado \omterm{def:b1:series:def}{série} somas para $\ln 2$
(\cref{ex:b1:series:ln2}), ainda \emph{nos mesmos termos, num
ordem diferente}, soma para $\frac{\ln 2}{2}$ --- ou para $\ln 2 +
\frac12\ln\frac pq $ para qualquer $ p, q $, ou para qualquer real
(Riemann). As duas histórias são uma só: as somas reorganizadas são
computado \emph{com} o $\gamma $-lei.\index{Euler
constante}\index{rearranjo de série}

\medskip
\noindent\textbf{Part I --- $\gamma $, bracketed.} \omterm{def:b1:logic:sets}{Definir} $ a_n = H_n
- \ln n $ e $ b_n = H_n - \ln(n+1)$.
\begin{enumerate}
  \item Usando $\frac{t}{1+t} \leq \ln(1 + t) \leq t $, mostre que
        $(a_n)$ diminui, $(b_n)$ aumenta e que são
        adjacente; seu \omterm{def:b1:topology:closure}{limite} comum é $\gamma $, com $ b_n \leq
        \gamma \leq a_n $ para cada $ n $.
  \item Primeiro tiro numérico: de $ H_{10} = 2.928968\dots $,
        colchete $\gamma $ entre $ b_{10} = 0.5311$ e $ a_{10} =
        0.6264$. Quão grande é um $ n $ esse colchete bruto precisaria
        para quatro casas decimais?
  \item Mostre a representação exata da cauda $ a_n - \gamma =
        \sum_{k \geq n} w_k $ (\omterm{def:b1:topology:closure}{limite} de somas parciais), onde
        \[
        w_k = a_k - a_{k+1}
        = \ln\Bigl(1 + \frac1k\Bigr) - \frac{1}{k+1}
        = \int_k^{k+1} \frac{(k + 1 - t)}{t\,(k+1)}\,\dd t ,
        \]
        e deduza do integrante formar o \omterm{def:b1:topology:closure}{limite} bilateral
        $\dfrac{1}{2(k+1)^2} \leq w_k \leq \dfrac{1}{2k(k+1)}$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The $\frac{1}{2n}$ law.}
\begin{enumerate}[resume]
  \item Some os \omterm{def:b1:topology:closure}{limites} da questão 3 (ambos os lados telescópicos ou
        compare com telescópios) e conclua a lei:
        \[
        \frac{1}{2(n+1)} \;\leq\; H_n - \ln n - \gamma \;\leq\;
        \frac{1}{2n} \qquad (n \geq 1).
        \]
  \item Deduzir $ H_n = \ln n + \gamma + \frac{1}{2n} +
        O\bigl(\frac{1}{n^2}\bigr)$; precisamente, mostre que
        $\gamma_n = H_n - \ln n - \frac{1}{2n}$ satisfaz
        $-\frac{1}{2n(n+1)} \leq \gamma_n - \gamma \leq 0$.
  \item Certifique quatro casas decimais com $ n = 100$: dado $ H_{100} =
        5.1873775\dots $, calcular $\gamma_{100} = 0.577207\dots $
        e concluir $\gamma = 0.5772 \pm 5\cdot10^{-5}$ (verdadeiro
        valor $0.5772156\dots $).
  \item Dois dividendos da lei, ambos necessários mais tarde: como $ m \to
        \infty $,
        \[
        H_{2m} - H_m = \ln 2 - \frac{1}{4m} +
        O\Bigl(\frac{1}{m^2}\Bigr),
        \qquad
        \sum_{j=1}^{m} \frac{1}{2j-1} = \frac{\ln m}{2} + \ln 2
        + \frac\gamma2 + o(1) ,
        \]
        a segunda via $\sum_{j \leq m} \frac{1}{2j-1} = H_{2m}
        - \frac12 H_m $, e da mesma forma $\sum_{j=1}^{m}
        \frac{1}{2j} = \frac{\ln m}{2} + \frac\gamma2 + o(1)$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The alternating harmonic \omterm{def:b1:series:def}{série},
to second order.}
\begin{enumerate}[resume]
  \item Mostrar (indução ou agrupamento) a identidade
        $\sum_{k=1}^{2m} \frac{(-1)^{k-1}}{k} = H_{2m} - H_m $,
        e deduza a soma $\ln 2$ (de novo) e o exato
        velocidade:
        \[
        \sum_{k=1}^{2m} \frac{(-1)^{k-1}}{k}
        = \ln 2 - \frac{1}{4m} + O\Bigl(\frac{1}{m^2}\Bigr) .
        \]
  \item Deduza o erro assintótico do harmônico alternado
        \omterm{def:b1:series:def}{série} no \emph{qualquer} índice: $ S - S_N \sim
        \frac{(-1)^N}{2N}$ --- duas vezes menor que o pior caso
        vinculado $ a_{N+1} \approx \frac1N $ de
        \cref{thm:b1:series:alternating}.
  \item (Aceleração gratuita) Mostre que as somas médias
        $\tilde S_N = \frac{S_N + S_{N+1}}{2}$ satisfazer $\tilde
        S_N = \ln 2 + O\bigl(\frac{1}{N^2}\bigr)$. Verificar: $ S_{10}
        = 0.64563$, $ S_{11} = 0.73654$, $\tilde S_{10} =
        0.69109$, contra $\ln 2 = 0.69315$: uma média compra
        duas casas decimais.
  \item Explique em duas frases por que nenhum truque desse tipo pode ajudar um
        \emph{positivo} fenômeno de cauda divergente como pergunta
        Colchete de 2: o erro alternado oscila (sinal
        $(-1)^N $), então a média cancela seu termo principal, enquanto
        o $\gamma $- erro de colchete $\frac{1}{2n}$ tem constante
        sinal. (Média $ a_n $ e $ b_n $ \emph{faz} ajuda:
        relacionar $\frac{a_n + b_n}{2}$ para a estimativa do ponto médio
        $ H_n - \ln\bigl(n + \frac12\bigr)$ e mostre que seu erro é
        $ O\bigl(\frac{1}{n^2}\bigr)$.)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Rigidity and its failure.}
\begin{enumerate}[resume]
  \item Mostre que a parte positiva $\sum \frac{1}{2j-1}$ e
        a parte negativa $\sum \frac{1}{2j}$ da alternância
        harmônico \omterm{def:b1:series:def}{série} ambos divergem --- a assinatura de
        \emph{condicional} convergência.
  \item Prove o geral \omterm{def:b1:logic:statement}{declaração} atrás da pergunta 12: se
        $\sum u_n $ converge, mas $\sum \abs{u_n}$ diverge, então
        o \omterm{def:b1:series:def}{série} de partes positivas $\sum u_n^+$ e de
        partes negativas $\sum u_n^-$ ambos divergem \emph{(de
        $ u_n^\pm = \frac{\abs{u_n} \pm u_n}{2}$: se um
        convergiram, o mesmo aconteceria com o outro, portanto $\sum\abs{u_n}$)}.
        Este reservatório inesgotável de emoções positivas e negativas
        massa é o que a receita de Riemann vai gastar.
  \item (Rigidez) Prove: se $\sum u_n $ converge
        \emph{absolutamente} e $\sigma \colon \N \to \N $ é um
        bijeção, então $\sum u_{\sigma(n)}$ converge para o
        mesma soma \emph{(para $ N $ grande o primeiro $ M $ reorganizado
        os termos contêm $ u_0, \dots, u_N $; comparar somas parciais
        pela cauda $\sum_{n > N}\abs{u_n}$)}.
  \item (receita de Riemann) Vamos $ t \in \R $. Descreva o ganancioso
        rearranjo do harmônico alternado \omterm{def:b1:series:def}{série}: pegue
        termos positivos $1, \frac13, \frac15, \dots $ até o
        soma parcial excede primeiro $ t $, então termos negativos
        $-\frac12, -\frac14, \dots $ até que primeiro caia abaixo
        $ t $ e repita. Mostre que cada termo é usado exatamente
        uma vez, que após o primeiro cruzamento das somas parciais
        permanecer dentro do último termo usado de $ t $, e concluir que
        o reorganizado \omterm{def:b1:series:def}{série} converge para $ t $: \emph{qualquer}
        soma prescrita é atingível.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The $(p, q)$ formula.} Corrigir números inteiros
$ p, q \geq 1$. Reorganizar o harmônico alternado \omterm{def:b1:series:def}{série} em
blocos: $ p $ termos positivos (os próximos recíprocos ímpares), então $ q $
termos negativos (os próximos recíprocos pares) e repita.
\begin{enumerate}[resume]
  \item Verifique se este é um rearranjo genuíno (cada termo
        exatamente uma vez), e isso por $(p, q) = (1, 2)$ lê
        \[
        1 - \frac12 - \frac14 + \frac13 - \frac16 - \frac18 +
        \frac15 - \frac1{10} - \frac1{12} + \dots
        \]
  \item (A metade exata) Para $(p, q) = (1, 2)$, comprove o
        bloquear identidade
        \[
        \frac{1}{2k-1} - \frac{1}{4k-2} - \frac{1}{4k}
        = \frac12\Bigl(\frac{1}{2k-1} - \frac{1}{2k}\Bigr),
        \]
        e deduzir o \emph{exato} relação $ T_{3K} = \frac12
        S_{2K}$ entre as somas parciais reorganizadas e o
        originais: a metade da soma é visível em
        cada estágio finito, não apenas no \omterm{def:b1:topology:closure}{limite}.
  \item Mostre que a soma parcial após $ K $ blocos completos
        é igual $\sum_{j=1}^{pK} \frac{1}{2j-1} -
        \sum_{j=1}^{qK} \frac{1}{2j}$, e calcule seu \omterm{def:b1:topology:closure}{limite}
        com a pergunta 7:
        \[
        \ln 2 + \frac12 \ln\frac pq .
        \]
  \item Controle as somas parciais \emph{dentro} um bloco (o
        termos tendem a $0$) e concluímos que $(p,
        q)$-reorganizado \omterm{def:b1:series:def}{série} converge para $\ln 2 + \frac12\ln
        \frac pq $. Em particular $(1, 2)$ dá $\frac{\ln
        2}{2}$: verificar em relação aos primeiros nove termos, $ T_9 =
        0.3083$, rastejando em direção $0.3466$.
  \item Verificações de sanidade e alcance: $(1,1)$ recupera $\ln 2$;
        $(2,1)$ dá $\frac32\ln 2$; quais somas são alcançáveis
        por $(p, q)$-blocks, e como funciona esse menu contável
        comparar com a carta completa de Riemann (questão 14)?
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Epilogue: $\gamma $ at work, and
synthesis.}
\begin{enumerate}[resume]
  \item Identifique a soma do convergente \omterm{def:b1:series:def}{série}
        $\sum_{k\geq1} \bigl(\frac1k - \ln\frac{k+1}{k}\bigr)$
        (\cref{exo:b1:series:7}): mostre que é igual $\gamma $.
  \item Execute a receita de Riemann (pergunta 14) para o alvo $ t =
        1$ e liste os primeiros doze termos produzidos ($1, \frac13,
        -\frac12, \frac15, -\frac14, \frac17, \frac19, -\frac16,
        \frac1{11}, \frac1{13}, -\frac18, \frac1{15}$),
        calculando a soma parcial ($\approx 0.980$) --- assistir
        o algoritmo respira em torno de seu alvo.
  \item Mostre que algum rearranjo do harmônico alternado
        \omterm{def:b1:series:def}{série} diverge para $+\infty $ \emph{(blocos de positivo
        termos longos o suficiente para ganhar $1$ cada vez, usando a pergunta
        12, separados por termos negativos únicos)}.
  \item Afiado \cref{ex:b1:series:harmonicstack} com o
        $\gamma $-law: mostre que o primeiro índice com $ H_N \geq
        20$ satisfaz $ N = \eu^{\,20 - \gamma}\,(1 + o(1))
        \approx 2.7\cdot10^{8}$ --- \omterm{ex:b1:series:oddtelescope}{Constante de Euler} é exatamente
