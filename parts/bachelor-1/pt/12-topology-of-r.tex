\chapter{Topologia da Linha Real}\label{ch:b1:topology}

Os \omterm{def:b1:topology:closure}{limites} continuam se referindo ao mesmo vocabulário geométrico: pontos
``perto de'' um \omterm{def:b1:logic:sets}{definir}, conjuntos ``sem \omterm{def:b1:topology:closure}{limite} vazamentos'', \omterm{prop:b1:reals:intervals}{intervalos} em
quais sequências não podem escapar. Este capítulo corrige esse vocabulário ---
\omterm{def:b1:topology:open}{abrir} e conjuntos \omterm{def:b1:topology:closed}{fechados}, \omterm{def:b1:topology:closure}{interior} e \omterm{def:b1:topology:closure}{encerramento}, densidade --- no real
linha, e comprova a compactação dos segmentos em sua forma sequencial.
As mesmas noções, em espaços vetoriais normados, são material do segundo ano;
sobre $\R $ eles estão ao seu alcance e imediatamente úteis para
\cref{ch:b1:continuity}.

\section{Conjuntos abertos, conjuntos fechados}

\begin{definition}[Bairro, conjunto aberto]\label{def:b1:topology:open}
UM \omterm{def:b1:logic:sets}{definir} $ V \subseteq \R $ é um \emph{vizinhança}\index{vizinhança}
de $ x \in \R $ quando contém um \omterm{prop:b1:reals:intervals}{intervalo} $\intoo{x - r}{x + r}$ para
alguns $ r > 0$. UM \omterm{def:b1:logic:sets}{definir} $ U \subseteq \R $ é \emph{abrir}\index{abrir}
quando é vizinhança de cada um de seus pontos:
\[
\forall x \in U,\ \exists r > 0, \quad \intoo{x - r}{x + r}
\subseteq U .
\]
\end{definition}

\begin{example}\label{ex:b1:topology:open}
\omterm{def:b1:topology:open}{Abrir} \omterm{prop:b1:reals:intervals}{intervalos} são \omterm{def:b1:topology:open}{abrir}: para $ x \in \intoo{a}{b}$, pegar $ r = \min(x -
a,\, b - x) > 0$. Meias linhas $\intoo{a}{+\infty}$ são \omterm{def:b1:topology:open}{abrir}; $\R $ e
$\emptyset $ são \omterm{def:b1:topology:open}{abrir} (este último vagamente). $\intcc{0}{1}$ é
\emph{não} \omterm{def:b1:topology:open}{abrir}: não \omterm{prop:b1:reals:intervals}{intervalo} em volta $0$ fica dentro.
\end{example}

\begin{proposition}[Estabilidade de conjuntos abertos]\label{prop:b1:topology:openstable}
Qualquer união de conjuntos abertos é \omterm{def:b1:topology:open}{abrir}; um \emph{finito} interseção de aberto
conjuntos é \omterm{def:b1:topology:open}{abrir}. Interseções infinitas podem falhar:
$\bigcap_{n \geq 1} \intoo{-\frac 1n}{\frac 1n} = \{0\}$, não \omterm{def:b1:topology:open}{abrir}.
\end{proposition}

\begin{proof}
União: se $ x \in \bigcup_i U_i $, então $ x \in U_{i_0}$ para alguns $ i_0$,
e o \omterm{prop:b1:reals:intervals}{intervalo} fornecido por $ U_{i_0}$ fica dentro do sindicato. Finito
interseção: se $ x \in U_1 \cap \dots \cap U_k $, pegar $ r = \min(r_1,
\dots, r_k) > 0$ dos raios fornecidos por cada $ U_j $. Para o
contra-exemplo: qualquer \omterm{prop:b1:reals:intervals}{intervalo} em volta $0$ contém alguns $\frac{1}{n}$
(Arquimedes), portanto sai do cruzamento.
\end{proof}

\begin{example}[Certificando abertura com raios explícitos]\label{ex:b1:topology:radii}
É $ U = \{x \in \R : x^2 > 2\}$ \omterm{def:b1:topology:open}{abrir}? Sim, e o certificado
pode ser escrito: $ U = \intoo{-\infty}{-\sqrt2} \cup
\intoo{\sqrt2}{+\infty}$, uma união de dois \omterm{def:b1:topology:open}{abrir} meias linhas, \omterm{def:b1:topology:open}{abrir}
por \cref{prop:b1:topology:openstable}. Alternativamente, argumente
ponto a ponto: para $ x \in U $ com $ x > \sqrt 2$, pegar $ r = x -
\sqrt2 > 0$: todo $ y \in \intoo{x - r}{x + r}$ satisfaz $ y >
\sqrt 2$, por isso $ y^2 > 2$; simetricamente à esquerda. Ambos
estilos importam --- o estrutural (construído a partir de aberto
conjuntos por uniões e interseções finitas) escala \omterm{def:b1:arith:prime}{melhor}, o
$\varepsilon $-style funciona quando nenhuma estrutura está visível; e
\cref{ch:b1:continuity} adicionará um terceiro, o mais poderoso:
$ U $ é o \omterm{def:b1:logic:map}{pré-imagem} do \omterm{def:b1:topology:open}{abrir} $\intoo{2}{+\infty}$ sob o
contínuo $ x \mapsto x^2$.
\end{example}

\begin{definition}[Conjunto fechado]\label{def:b1:topology:closed}
UM \omterm{def:b1:logic:sets}{definir} $ F \subseteq \R $ é \emph{fechado}\index{fechado} quando é
complemento $\R \setminus F $ é \omterm{def:b1:topology:open}{abrir}. Por de Morgan e
\cref{prop:b1:topology:openstable}: qualquer interseção de fechado conjuntos
é fechado, uniões finitas de fechado conjuntos estão fechados.
\end{definition}

\begin{theorem}[Caracterização sequencial de conjuntos fechados]\label{thm:b1:topology:seqclosed}
$ F $ é \omterm{def:b1:topology:closed}{fechado} se e somente se: para cada sequência $(u_n)$ de pontos de
$ F $ convergindo para alguns $\ell \in \R $, o \omterm{def:b1:topology:closure}{limite} $\ell $ pertence a
$ F $. (``\omterm{def:b1:topology:closed}{Fechado}'' $=$ ``estável dentro dos \omterm{def:b1:topology:closure}{limites}''.)
\end{theorem}

\begin{proof}
($\Rightarrow $) Deixar $ F $ ser \omterm{def:b1:topology:closed}{fechado}, $ u_n \in F $, $ u_n \to \ell $ e
suponha $\ell \notin F $. O complemento é \omterm{def:b1:topology:open}{abrir}: alguns
$\intoo{\ell - r}{\ell + r}$ evita $ F $. Mas a convergência coloca $ u_n $
nisso \omterm{prop:b1:reals:intervals}{intervalo} para grande $ n $: contradição com $ u_n \in F $.

($\Leftarrow $) Suponha $ F $ não é \omterm{def:b1:topology:closed}{fechado}: o complemento não é \omterm{def:b1:topology:open}{abrir},
então alguns $ x \notin F $ não tem \omterm{prop:b1:reals:intervals}{intervalo} $\intoo{x - r}{x + r}$ dentro
o complemento; tirando $ r = \frac{1}{n+1}$, escolher $ u_n \in F $ com
$\abs{u_n - x} < \frac{1}{n+1}$. Então $ u_n \in F $, $ u_n \to x \notin
F $: a propriedade sequencial falha.
\end{proof}

\begin{example}[O teste sequencial, em ambos os sentidos]\label{ex:b1:topology:seqtest}
\emph{\omterm{def:b1:topology:closed}{Fechado}:} $ F = \Z \cup \bigl\{n + \frac1n : n \geq 2\bigr\}$.
Deixar $ u_k \in F $ com $ u_k \to \ell $. A janela $\intcc{\ell -
1}{\ell + 1}$ contém apenas um número finito de pontos de $ F $ (finitamente
muitos inteiros, finitos $ n + \frac1n $), e além de alguma classificação
tudo $ u_k $ reside nele: a sequência então leva um número finito
valores e, convergindo, é eventualmente constante (como em
\cref{exo:b1:topology:3}): $\ell \in F $. \omterm{def:b1:topology:closed}{Fechado} --- embora
$ F $ contém pares de pontos à distância $\frac1n $, arbitrariamente
perto.

\emph{Não \omterm{def:b1:topology:closed}{fechado}:} $ G = \bigl\{\frac1m + \frac1n : m, n \in
\N^*\bigr\}$. A sequência $\frac1n + \frac1n \in G $ tende a
$0$, e $0 \notin G $ (uma soma de dois termos positivos): o
teste sequencial falha, $ G $ não é \omterm{def:b1:topology:closed}{fechado}. Curiosamente, cada
$\frac1m $ \emph{faz} pertencer a $\overline G \cap G $: de fato
$\frac1m = \frac{1}{m+1} + \frac{1}{m(m+1)} \in G $. O fechamento
insight: para provar fechamento, controle \emph{todos} convergente
sequências de uma só vez (normalmente através de uma finitude local ou de uma
argumento de fórmula fechada); para refutá-lo, um fugitivo bem escolhido
sequência é suficiente --- a assimetria torna a direção negativa
o fácil, e todos os contra-exemplos deste capítulo têm
esta forma de uma linha.
\end{example}

\begin{example}\label{ex:b1:topology:closed}
Segmentos $\intcc{a}{b}$, meias-linhas $\intco{a}{+\infty}$, finito
conjuntos, $\Z $ (uma sequência convergente de inteiros é eventualmente constante)
são \omterm{def:b1:topology:closed}{fechado}. $\intoc{0}{1}$ não é nenhum dos dois \omterm{def:b1:topology:open}{abrir} (falha em $1$) nem \omterm{def:b1:topology:closed}{fechado}
($\frac 1n \to 0 \notin $ o \omterm{def:b1:logic:sets}{definir}): maioria conjuntos não são nenhum dos dois. $\R $ e
$\emptyset $ são ambos \omterm{def:b1:topology:open}{abrir} e \omterm{def:b1:topology:closed}{fechado} --- e eles são os únicos
subconjuntos de $\R $ (\cref{exo:b1:topology:9}).
\end{example}

\begin{example}[Um conjunto aberto montado a partir de um número infinito de peças]\label{ex:b1:topology:rminusz}
$\R \setminus \Z = \bigcup_{n \in \Z} \intoo{n}{n+1}$: um
união infinita de \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalos}, \omterm{def:b1:topology:open}{abrir} por
\cref{prop:b1:topology:openstable} --- então $\Z $ é \omterm{def:b1:topology:closed}{fechado} sem
argumento sequencial necessário. Observe a \omterm{thm:b1:logic:partition}{divisão} do trabalho no
regras de estabilidade: \emph{sindicatos} de conjuntos abertos pode ser arbitrário
(cada ponto necessita apenas de um certificado próprio, fornecido por aquele
\omterm{def:b1:logic:sets}{definir} contendo-o), enquanto \emph{cruzamentos} deve permanecer finito
(os certificados devem ser intersectados, e infinitos raios podem
encolher até nada). \Cref{exo:b1:topology:10} vai mostrar isso
exemplo é a forma geral: cada \omterm{def:b1:topology:open}{abrir} subconjunto de $\R $ é um
união disjunta contável de \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalos}.
\end{example}

\section{Interior, fechamento, densidade}

\begin{definition}[Interior, fechamento, limite]\label{def:b1:topology:closure}
Deixar $ A \subseteq \R $.
\begin{itemize}
  \item Um ponto $ x $ é \emph{interior} para $ A $ quando $ A $ é um
        \omterm{def:b1:topology:open}{vizinhança} de $ x $; o \emph{interior}
        $\mathring{A}$\index{interior} é o \omterm{def:b1:logic:sets}{definir} de pontos interiores.
  \item Um ponto $ x $ é \emph{aderente} para $ A $ quando cada \omterm{def:b1:topology:open}{vizinhança}
        de $ x $ conhece $ A $; o \emph{encerramento}
        $\overline{A}$\index{encerramento} é o \omterm{def:b1:logic:sets}{definir} de pontos aderentes.
  \item O \emph{limite} é $\partial A = \overline A \setminus
        \mathring A $.\index{limite}
\end{itemize}
Então $\mathring A \subseteq A \subseteq \overline A $.
\end{definition}

\begin{proposition}[Propriedades principais]\label{prop:b1:topology:closureprops}
\begin{enumerate}
  \item $\mathring A $ é o maior conjunto aberto contido em $ A $; $ A $
        é \omterm{def:b1:topology:open}{abrir} se $ A = \mathring A $.
  \item $\overline A $ é o menor conjunto \omterm{def:b1:topology:closed}{fechado} contendo $ A $; $ A $
        é \omterm{def:b1:topology:closed}{fechado} se $ A = \overline A $.
  \item (Caracterização sequencial da adesão) $ x \in \overline
        A $ se e somente se $ x $ é o \omterm{def:b1:topology:closure}{limite} de uma sequência de pontos de
        $ A $.
  \item A complementação troca as noções: $\R \setminus
        \overline A = \bigl(\R \setminus A\bigr)^{\!\circ}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(4) $ x \notin \overline A $ $\iff $ alguns \omterm{def:b1:topology:open}{vizinhança} de $ x $ evita $ A $
$\iff $ alguns \omterm{prop:b1:reals:intervals}{intervalo} em volta $ x $ reside em $\R \setminus A $ $\iff $ $ x $
é \omterm{def:b1:topology:closure}{interior} para $\R \setminus A $.

(1) $\mathring A $ é \omterm{def:b1:topology:open}{abrir}: se $ x \in \mathring A $, alguns
$\intoo{x-r}{x+r} \subseteq A $; cada ponto $ y $ disso \omterm{prop:b1:reals:intervals}{intervalo} tem
um menor \omterm{prop:b1:reals:intervals}{intervalo} em torno dele dentro dele, portanto dentro $ A $: o todo
\omterm{prop:b1:reals:intervals}{intervalo} está em $\mathring A $. Qualquer \omterm{def:b1:topology:open}{abrir} $ U \subseteq A $ consiste em
\omterm{def:b1:topology:closure}{interior} pontos de $ A $, então $ U \subseteq \mathring A $: maior. O
segue a caracterização da abertura.

(2) Em detalhes. Por (4), $\R \setminus \overline A $ é o \omterm{def:b1:topology:closure}{interior}
de $\R \setminus A $, um conjunto aberto por (1): então $\overline A $ é
\omterm{def:b1:topology:closed}{fechado}, e contém $ A $. Minimidade: deixe $ F \supseteq A $ ser
\omterm{def:b1:topology:closed}{fechado}. Então $\R \setminus F $ é \omterm{def:b1:topology:open}{abrir} e contido em $\R
\setminus A $, então pela maximidade em (1),
\[
\R \setminus F \subseteq \bigl(\R \setminus A\bigr)^{\!\circ}
= \R \setminus \overline A ,
\]
e tomando complementos novamente: $\overline A \subseteq F $. Por isso
$\overline A $ é o menor \omterm{def:b1:topology:closed}{fechado} superconjunto. Caracterização:
se $ A = \overline A $ então $ A $ é \omterm{def:b1:topology:closed}{fechado} (acabado de mostrar); se $ A $ é
\omterm{def:b1:topology:closed}{fechado}, é em si um \omterm{def:b1:topology:closed}{fechado} superconjunto de $ A $, então minimalismo
forças $\overline A \subseteq A $ e igualdade.

(3) Se $ u_n \in A $, $ u_n \to x $: todo \omterm{def:b1:topology:open}{vizinhança} de $ x $ contém
alguns $ u_n \in A $, então $ x \in \overline A $. Por outro lado, se $ x \in
\overline A $: cada \omterm{prop:b1:reals:intervals}{intervalo} $\intoo{x - \frac{1}{n+1}}{x +
\frac{1}{n+1}}$ conhece $ A $ em algum $ u_n $, e $ u_n \to x $.
\end{proof}

\begin{example}\label{ex:b1:topology:closureexamples}
$\overline{\intoo{0}{1}} = \intcc{0}{1}$;
$\mathring{\intcc{0}{1}} = \intoo{0}{1}$; $\partial\intoo{0}{1} =
\{0, 1\}$. Para $ A = \{\frac 1n : n \in \N^*\}$: $\overline A = A
\cup \{0\}$, $\mathring A = \emptyset $, $\partial A = A \cup \{0\}$.
Para $\Q $: por densidade (\cref{thm:b1:reals:density}) todo real é
aderente a $\Q $, então $\overline{\Q} = \R $ enquanto $\mathring{\Q} =
\emptyset $ (todo \omterm{prop:b1:reals:intervals}{intervalo} contém irracionais): o \omterm{def:b1:topology:closure}{limite} de
$\Q $ é tudo de $\R $.
\end{example}

\begin{example}[Uma anatomia completa]\label{ex:b1:topology:anatomy}
Deixar $ A = \intoc{0}{1} \,\cup\, \bigl(\Q \cap \intoo{2}{3}\bigr)
\,\cup\, \{4\}$. Calculamos os três conjuntos de
\cref{def:b1:topology:closure}, pedaço por pedaço.

\emph{\omterm{def:b1:topology:closure}{Interior}.} Um ponto de $\intoo{0}{1}$ tem um todo \omterm{prop:b1:reals:intervals}{intervalo}
dentro $ A $: \omterm{def:b1:topology:closure}{interior}. O ponto $1$: todo \omterm{prop:b1:reals:intervals}{intervalo} em torno disso
vaza à direita de $1$, onde $ A $ não tem nada até $2$: não
\omterm{def:b1:topology:closure}{interior}. Nenhum ponto de $\Q \cap \intoo{2}{3}$ é \omterm{def:b1:topology:closure}{interior} (todo
\omterm{prop:b1:reals:intervals}{intervalo} contém irracionais, \cref{thm:b1:reals:density});
nem o isolado é $4$. Então $\mathring A = \intoo{0}{1}$.

\emph{\omterm{def:b1:topology:closure}{Encerramento}.} \omterm{def:b1:topology:closure}{Limites} de pontos de $ A $: todos $\intcc{0}{1}$
($0 = \lim \frac1n $ com $\frac 1n \in A $); todos
$\intcc{2}{3}$ (para cada real há um \omterm{def:b1:topology:closure}{limite} de racionais do
\omterm{prop:b1:reals:intervals}{intervalo}, densidade de novo); e $4$. Nada mais: um ponto
fora $\intcc{0}{1} \cup \intcc{2}{3} \cup \{4\}$ tem positivo
distância para isso conjunto \omterm{def:b1:topology:closed}{fechado}. Então $\overline A = \intcc{0}{1} \cup
\intcc{2}{3} \cup \{4\}$.

\emph{\omterm{def:b1:topology:closure}{Limite}.} $\partial A = \overline A \setminus \mathring A
= \{0, 1\} \cup \intcc{2}{3} \cup \{4\}$.

O insight final: o ato das três operações \emph{localmente} ---
cada pedaço de $ A $ contribui de acordo com a sua própria natureza (um
sólido \omterm{prop:b1:reals:intervals}{intervalo} mantém seu \omterm{def:b1:topology:closure}{interior}, uma peça densa, mas porosa, gira
inteiramente em \omterm{def:b1:topology:closure}{limite}, um ponto isolado é puro \omterm{def:b1:topology:closure}{limite}) e
um desenho de duas linhas de $ A $ prevê cada resposta antes de qualquer prova
está escrito.
\end{example}

\begin{remark}[Armadilhas comuns no raciocínio baseado em pontos]
(eu) \emph{``Não \omterm{def:b1:topology:open}{abrir}'' não significa ``\omterm{def:b1:topology:closed}{fechado}''}: maioria conjuntos são
nem ($\intoc{0}{1}$) e dois conjuntos são ambos
($\emptyset $, $\R $) --- \omterm{def:b1:topology:open}{abrir} e \omterm{def:b1:topology:closed}{fechado} não são opostos, mas
duais por meio de complementação. (ii) \emph{\omterm{def:b1:topology:closure}{Interior} e \omterm{def:b1:topology:closure}{encerramento}
não viaje}: para $ A = \Q $,
\[
\overline{\mathring A} = \overline\emptyset = \emptyset
\qquad\text{while}\qquad
\bigl(\,\overline A\,\bigr)^{\!\circ} = \mathring \R = \R :
\]
os dois operadores iterados diferem tanto quanto conjuntos pode. (iii)
\emph{Uniões infinitas de conjuntos \omterm{def:b1:topology:closed}{fechados} pode deixar de ser \omterm{def:b1:topology:closed}{fechado}}:
$\bigcup_{n\geq1} \intcc{\frac1n}{1} = \intoc{0}{1}$ --- o
espelho do contra-exemplo de intersecção de
\cref{prop:b1:topology:openstable}. (4) \emph{\omterm{def:b1:topology:dense}{Denso} não
significa grande}: $\Q $ é \omterm{def:b1:topology:dense}{denso}, contável, com vazio interiore
seu complemento é \omterm{def:b1:topology:dense}{denso} também; densidade diz ``arbitrariamente perto de
tudo'', não ``quase tudo'' --- o problema do fim de semana
\omterm{pb:b1:topology:1}{Conjunto de cantores} (\cref{pb:b1:topology:1}) faz o ponto oposto,
um topologicamente pequeno \omterm{def:b1:logic:sets}{definir} isso é incontávelmente grande.
\end{remark}

\begin{example}[Um fechamento calculado exatamente]\label{ex:b1:topology:closurecomputed}
Deixar $ G = \bigl\{\frac1m + \frac1n : m, n \in \N^*\bigr\}$ (de
\cref{ex:b1:topology:seqtest}). Alegar:
\[
\overline G = G \,\cup\, \Bigl\{\frac1m : m \in \N^*\Bigr\}
\,\cup\, \{0\} .
\]
($\supseteq $) $\frac1m = \lim_n \bigl(\frac1m + \frac1n\bigr)$
e $0 = \lim_n \frac2n $: aderente pelo sequencial
caracterização. ($\subseteq $) Deixar $ x = \lim_k \bigl(
\frac{1}{m_k} + \frac{1}{n_k}\bigr)$; ordene cada par de modo que
$ m_k \leq n_k $. Se $(m_k)$ é ilimitado, um \omterm{def:b1:seq:subsequence}{subsequência} tem $ m_k
\to \infty $, por isso $ n_k \to \infty $ também e $ x = 0$. De outra forma
$(m_k)$ assume um número finito de valores, um deles, digamos $ m $,
infinitamente frequentemente; junto com isso \omterm{def:b1:seq:subsequence}{subsequência} $\frac{1}{n_k} \to x -
\frac1m $: se $(n_k)$ é limitado, leva algum valor $ n $
infinitamente e frequentemente $ x = \frac1m + \frac1n \in G $; se não, $ x =
\frac1m $. Cada caso chega ao anunciado \omterm{def:b1:logic:sets}{definir}. O fechamento
insight: computando um \omterm{def:b1:topology:closure}{encerramento} é um case de estilo compacto
análise de índices --- índice limitado significa um número finito
valores (cabo), índice ilimitado significa que um \omterm{def:b1:topology:closure}{limite} escapa ---
e a resposta exibe a estrutura típica de duas camadas de
pontos \omterm{def:b1:topology:closure}{limite}: o \omterm{def:b1:logic:sets}{definir}, seus \omterm{def:b1:topology:closure}{limites} de primeira geração e seus
\omterm{def:b1:topology:closure}{limite} $0$.
\end{example}

\begin{definition}[Densidade, forma topológica]\label{def:b1:topology:dense}
$ A $ é \emph{denso}\index{denso} em $\R $ quando $\overline A = \R $
--- equivalentemente, todo não vazio \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalo} conhece $ A $;
equivalentemente (por \cref{prop:b1:topology:closureprops}~(3)), cada
real é um \omterm{def:b1:topology:closure}{limite} de elementos de $ A $. Exemplos: $\Q $, $\R \setminus
\Q $, as diádicas (\cref{exo:b1:reals:8}), denso \omterm{def:b1:structures:subgroup}{subgrupos}
(\cref{exo:b1:reals:9}).
\end{definition}

\begin{example}[A densidade é relativa]\label{ex:b1:topology:relativedensity}
``\omterm{def:b1:topology:dense}{Denso}'' conforme definido aqui significa \omterm{def:b1:topology:dense}{denso} \emph{em $\R $}; um \omterm{def:b1:logic:sets}{definir} pode
em vez disso, seja \omterm{def:b1:topology:dense}{denso} apenas em uma parte da linha. As diádicas de
$\intcc{0}{1}$, ou seja\ $ D \cap \intcc{0}{1}$
(\cref{exo:b1:reals:8}), conheça todos \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalo} incluído em
$\intcc{0}{1}$ mas é claro, senhorita $\intoo{2}{3}$ inteiramente: eles
são \omterm{def:b1:topology:dense}{denso} \emph{em} $\intcc{0}{1}$, significado $\overline{D \cap
\intcc{0}{1}} = \intcc{0}{1}$. A frase geral ``$ A $ é \omterm{def:b1:topology:dense}{denso}
em $ B $'' abrevia $ B \subseteq \overline A $ --- sempre nome
o ambiente \omterm{def:b1:logic:sets}{definir}, já que os pontos finais do problema do fim de semana são \omterm{def:b1:topology:dense}{denso}
\emph{no \omterm{pb:b1:topology:1}{Conjunto de cantores}} enquanto não estou em lugar nenhum \omterm{def:b1:topology:dense}{denso} \emph{em
$\R $}: o mesmo \omterm{def:b1:logic:sets}{definir}, dois verdadeiros e de som oposto
descrições.
\end{example}

\begin{example}[Densidade de manuseio]\label{ex:b1:topology:densitytricks}
Três movimentos rápidos que se repetem constantemente. \emph{Ampliando}: se
$ A $ é \omterm{def:b1:topology:dense}{denso} e $ A \subseteq B $, então $ B $ é \omterm{def:b1:topology:dense}{denso} (todo
\omterm{prop:b1:reals:intervals}{intervalo} já conhece $ A $). \emph{Transporte}: se $ A $ é
\omterm{def:b1:topology:dense}{denso}, assim é $\lambda A + \mu $ para $\lambda \neq 0$ --- um
\omterm{prop:b1:reals:intervals}{intervalo} $ I $ conhece $\lambda A + \mu $ se o \omterm{prop:b1:reals:intervals}{intervalo}
$\frac{I - \mu}{\lambda}$ conhece $ A $; portanto, os múltiplos ímpares de
$10^{-9}$, digamos, são \omterm{def:b1:topology:dense}{denso}. \emph{A intersecção falha}: dois \omterm{def:b1:topology:dense}{denso}
conjuntos podem sentir falta um do outro completamente ($\Q $ e $\R \setminus \Q $):
densidade sobrevive a sindicatos e afins \omterm{def:b1:logic:map}{mapas}, nunca interseções.
\end{example}

\section{Compacidade dos segmentos}

\begin{theorem}[Os segmentos são sequencialmente compactos]\label{thm:b1:topology:compact}
Deixar $ a \leq b $. Cada sequência de pontos de $\intcc{a}{b}$ tem um
\omterm{def:b1:seq:subsequence}{subsequência} convergindo para um ponto \emph{de}
$\intcc{a}{b}$.\index{compacidade}

De forma mais geral, os subconjuntos de $\R $ com esta propriedade (cada
sequência tem um \omterm{def:b1:seq:subsequence}{subsequência} convergindo no \omterm{def:b1:logic:sets}{definir}) são exatamente os
\emph{\omterm{def:b1:topology:closed}{fechado} e limitado} conjuntos.
\end{theorem}

\begin{proof}
Uma sequência em $\intcc{a}{b}$ é limitado, então Bolzano--Weierstrass
(\cref{thm:b1:seq:bw}) extrai um convergente \omterm{def:b1:seq:subsequence}{subsequência}; seu \omterm{def:b1:topology:closure}{limite}
permanece em $\intcc{a}{b}$ porque os segmentos são \omterm{def:b1:topology:closed}{fechado}
(\cref{thm:b1:topology:seqclosed}).

Caso geral. \emph{(\omterm{def:b1:topology:closed}{Fechado} limitado $\Rightarrow $ compactar)}: deixar
$ F $ ser \omterm{def:b1:topology:closed}{fechado} e limitado, e $(u_n)$ uma sequência em $ F $.
\omterm{def:b1:topology:closure}{Limite} de $ F $ limita a sequência, então Bolzano--Weierstrass
extratos $ u_{\varphi(n)} \to \ell $; e $\ell \in F $ porque $ F $
é \omterm{def:b1:topology:closed}{fechado} e o \omterm{def:b1:seq:subsequence}{subsequência} é uma sequência convergente de pontos
de $ F $ (\cref{thm:b1:topology:seqclosed}): as duas hipóteses
são consumidos um de cada, \omterm{def:b1:topology:closure}{limite} para a existência do \omterm{def:b1:topology:closure}{limite},
fechamento para seus membros. \emph{(Compactar $\Rightarrow $ \omterm{def:b1:topology:closed}{fechado} e
limitado)}: se $ F $ é ilimitado, escolha $ u_n \in F $ com $\abs{u_n} \geq
n $; todo \omterm{def:b1:seq:subsequence}{subsequência} é ilimitado, portanto divergente
(\cref{prop:b1:seq:first}): não convergente \omterm{def:b1:seq:subsequence}{subsequência} de forma alguma. Se $ F $
não é \omterm{def:b1:topology:closed}{fechado}, pegar $ u_n \in F $ com $ u_n \to \ell \notin F $
(\cref{thm:b1:topology:seqclosed}): todo \omterm{def:b1:seq:subsequence}{subsequência} converge para
$\ell \notin F $, então não \omterm{def:b1:seq:subsequence}{subsequência} converge \emph{em} $ F $.
\end{proof}

\begin{example}[Conjuntos compactos aninhados]\label{ex:b1:topology:nested}
Um primeiro treino para o teorema. Deixar $ K_0 \supseteq K_1 \supseteq
K_2 \supseteq \dots $ ser compacto não vazio (\omterm{def:b1:topology:closed}{fechado} limitados) subconjuntos
de $\R $. Então $\bigcap_n K_n \neq \emptyset $. Na verdade, escolha $ x_n
\in K_n $ para cada $ n $: a sequência vive no compacto $ K_0$,
então um \omterm{def:b1:seq:subsequence}{subsequência} $ x_{\varphi(n)}$ converge para alguns $ x $
(\cref{thm:b1:topology:compact}). Para cada fixo $ m $, os termos
$ x_{\varphi(n)}$ com $\varphi(n) \geq m $ todos mentem no \omterm{def:b1:topology:closed}{fechado}
\omterm{def:b1:logic:sets}{definir} $ K_m $, então o \omterm{def:b1:topology:closure}{limite} $ x $ reside em $ K_m $
(\cref{thm:b1:topology:seqclosed}); como $ m $ foi arbitrário, $ x \in
\bigcap_n K_n $. Um companheiro útil: se um conjunto aberto $ U $ contém
$\bigcap_n K_n $, então $ U \supseteq K_n $ para alguns $ n $ --- aplicar
o mesmo argumento para pontos $ x_n \in K_n \setminus U $; o \omterm{def:b1:topology:closure}{limite}
$ x $ mentiria em $\bigcap K_n \subseteq U $, ainda $ U $ \omterm{def:b1:topology:open}{abrir} forças
$ x_{\varphi(n)} \in U $ eventualmente, uma contradição. Ambos
declarações falhar sem compacidade: $\bigcap_n \intoo{0}{\frac
1n} = \emptyset $ e $\bigcap_n \intco{n}{+\infty} = \emptyset $.
A conclusão final: a compacidade converte uma cadeia infinita de
afirmações de não-vazio em um único ponto \omterm{def:b1:topology:closure}{limite} --- é o
ferramenta que sobrevive à passagem para a interseção infinita, e o
problema de fim de semana (\cref{pb:b1:topology:1}) se apoiará nele duas vezes.
\end{example}

Este é o motor por trás do teorema do valor extremo

\begin{example}[Limites sob sindicatos]\label{ex:b1:topology:boundaryunion}
Sempre $\partial(A \cup B) \subseteq \partial A \cup \partial
B $: um ponto de $\partial(A \cup B)$ tem cada \omterm{def:b1:topology:open}{vizinhança}
reunião $ A \cup B $ (por isso $ A $ ou $ B $, infinitamente frequentemente um dos
eles) e atendendo ao complemento de $ A \cup B $, que reside em
ambos os complementos --- uma breve verificação coloca o ponto em
$\partial A $ ou $\partial B $. A inclusão pode ser espetacularmente
rigoroso: com $ A = \Q $ e $ B = \R\setminus\Q $,
\[
\partial(A \cup B) = \partial \R = \emptyset ,
\qquad
\partial A \cup \partial B = \R \cup \R = \R :
\]
dois esfarrapados conjuntos pode colar em um perfeito, seus \omterm{def:b1:topology:closure}{limites}
aniquilando uns aos outros. A visão final: \omterm{def:b1:topology:closure}{interiores} e
fechamentos comportar-se \emph{monotonicamente} sob sindicatos e
interseções, mas os \omterm{def:b1:topology:closure}{limites} não --- tratam $\partial $ como um
quantidade derivada ($\overline A \setminus \mathring A $), nunca tão
um operador com álgebra própria.
\end{example}

\begin{remark}[Perspectivas dentro deste volume]
O vocabulário aqui construído é consumido mais duas vezes neste livro.
Em \cref{ch:b1:continuity}, todo teorema é uma topologia
\omterm{def:b1:logic:statement}{declaração} disfarçado: o teorema do valor intermediário diz
contínuo \omterm{def:b1:logic:map}{mapas} preservar o \omterm{prop:b1:reals:intervals}{intervalo} propriedade, o extremo
o teorema do valor diz que eles preservam a compactação --- e as provas
ligar \cref{thm:b1:topology:seqclosed,thm:b1:topology:compact}
pelo nome. Em \cref{ch:b1:multivar}, as mesmas definições são
releia em $\R^2$ com discos no lugar de \omterm{prop:b1:reals:intervals}{intervalos}: conjuntos abertos,
fechamentos e compactação são transferidas palavra por palavra, e o
o teorema do valor extremo de duas variáveis novamente segue em frente
Bolzano--Weierstrass (extrato em cada coordenada). Aquele
noção que faz \emph{não} generalizar sem dor é o
\omterm{prop:b1:reals:intervals}{intervalo} em si --- no plano, a conexão substitui
convexidade, uma história que começa com
\cref{exo:b1:topology:9}é ``apenas $\emptyset $ e $\R $ são \omterm{def:b1:topology:open}{abrir}
e \omterm{def:b1:topology:closed}{fechado}''.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b1:topology:1}
Para cada \omterm{def:b1:logic:sets}{definir}, diga se é \omterm{def:b1:topology:open}{abrir}, \omterm{def:b1:topology:closed}{fechado}, ambos ou nenhum (com
justificativa): $\intoo{0}{1} \cup \intoo{2}{3}$; $\;\intco{0}{1}$;
$\;\{0\} \cup \intcc{1}{2}$; $\;\R \setminus \Z $; $\;\Q \cap
\intoo{0}{1}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:topology:2}
Determinar $\mathring A $, $\overline A $ e $\partial A $ para:
$ A = \intoc{0}{1} \cup \{2\}$; $\;A = \R \setminus \Q $;
$\;A = \bigl\{\frac{(-1)^n n}{n+1} : n \in \N\bigr\}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:topology:3}
Prove que um conjunto finito é \omterm{def:b1:topology:closed}{fechado}, primeiro via complementos, depois via
a caracterização sequencial.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:topology:4}
Prove que para qualquer $ A, B \subseteq \R $: $\overline{A \cup B} =
\overline A \cup \overline B $. Mostre pelo exemplo que
$\overline{A \cap B}$ pode diferir de $\overline A \cap \overline
B $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:topology:5}
Deixar $ u_n \to \ell $ em $\R $. Prove que o \omterm{def:b1:logic:sets}{definir} $\{u_n : n \in \N\}
\cup \{\ell\}$ é \omterm{def:b1:topology:closed}{fechado} (portanto compacto se adicionarmos que é
limitado --- o que é).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:topology:6}
Deixar $ U $ ser \omterm{def:b1:topology:open}{abrir} e $ A $ arbitrário. Prove isso $ U + A = \{u + a\}$ é
\omterm{def:b1:topology:open}{abrir}. Deduza que a soma de um conjunto aberto e qualquer \omterm{def:b1:logic:sets}{definir} é abrire
contraste: exposição dois \emph{\omterm{def:b1:topology:closed}{fechado}} conjuntos cuja soma não é \omterm{def:b1:topology:closed}{fechado}.
\emph{(Tentar $\Z $ e $\sqrt 2\,\Z $, com \cref{exo:b1:reals:9}.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:topology:7}
Um ponto $ x \in A $ é \emph{isolado} em $ A $ quando alguns \omterm{def:b1:topology:open}{vizinhança} de
$ x $ conhece $ A $ apenas em $ x $. Prove que todo ponto de $\Z $ está isolado
em $\Z $, que $ A = \{\frac1n\}$ tem todos os seus pontos isolados ainda
$\overline A \neq A $, e que um \omterm{def:b1:logic:sets}{definir} cujos pontos estão todos isolados
está vazio \omterm{def:b1:topology:closure}{interior}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:topology:8}
Prove que o \omterm{def:b1:topology:closure}{encerramento} de um \omterm{def:b1:topology:closure}{limite} \omterm{def:b1:logic:sets}{definir} é limitado, e que
$\sup \overline A = \sup A $ para $ A $ não vazio limitado acima. Deduzir
isso $\sup A \in \overline A $: o \omterm{def:b1:reals:bounds}{supremo} é sempre aderente.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:topology:9}
Prove que os únicos subconjuntos de $\R $ isso são ambos \omterm{def:b1:topology:open}{abrir} e \omterm{def:b1:topology:closed}{fechado} são
$\emptyset $ e $\R $. \emph{Dica: suponha $ A $ é \omterm{def:b1:topology:open}{abrir}, \omterm{def:b1:topology:closed}{fechado}, com
$ A \neq \emptyset $ e $\R \setminus A \neq \emptyset $; escolha $ a \in
A $, $ b \notin A $, dizer $ a < b $, e considere $ s = \sup\,(A \cap
\intcc{a}{b})$; decidir se $ s $ pode pertencer a $ A $ ou para o seu
complemento.}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:topology:10}
(Estrutura de conjuntos abertos) Deixar $ U \subseteq \R $ ser \omterm{def:b1:topology:open}{abrir}, não vazio. Para
$ x \in U $, deixar $ I_x $ seja a união de todos \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalos} contendo
$ x $ e contido em $ U $. Prove isso $ I_x $ é um \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalo}, isso
dois conjuntos $ I_x $, $ I_y $ são iguais ou disjuntos, e que $ U $ é uma união
de \emph{incontáveis} disjunto aos pares \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalos} \emph{(escolher
um racional em cada)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:topology:11}
Um ponto $ x \in \R $ é um \emph{ponto de acumulação} de $ A $ quando
cada \omterm{def:b1:topology:open}{vizinhança} de $ x $ conhece $ A \setminus \{x\}$; deles \omterm{def:b1:logic:sets}{definir} é
o \emph{derivado \omterm{def:b1:logic:sets}{definir}} $ A'$. Prove isso $\overline A = A \cup A'$,
e isso $ A $ é \omterm{def:b1:topology:closed}{fechado} se e somente se $ A' \subseteq A $. Determinar
$ A'$ para $ A = \{\frac 1n : n \in \N^*\}$, para $ A = \Z $, e para
$ A = \Q $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:topology:12}
Para $ A \subseteq \R $ não vazio, \omterm{def:b1:logic:sets}{definir} $ d_A(x) = \inf\{\abs{x - a}
: a \in A\}$. Provar:
\begin{enumerate}
  \item $\abs{d_A(x) - d_A(y)} \leq \abs{x - y}$ para todos $ x, y $
        ($ d_A $ é $1$-Lipschitz);
  \item $ d_A(x) = 0$ se e somente se $ x \in \overline A $; em
        especial, se $ F $ é \omterm{def:b1:topology:closed}{fechado} e $ x \notin F $, então
        $ d_F(x) > 0$;
  \item para cada $\varepsilon > 0$, o \omterm{def:b1:logic:sets}{definir} $ V_\varepsilon =
        \{x : d_F(x) < \varepsilon\}$ é \omterm{def:b1:topology:open}{abrir}, contém $ F $, e
        $\bigcap_{\varepsilon > 0} V_\varepsilon = F $ para $ F $
        \omterm{def:b1:topology:closed}{fechado}: todo conjunto \omterm{def:b1:topology:closed}{fechado} é uma interseção contável de
        conjuntos abertos.
\end{enumerate}
\end{exercise}

\section{Problema: O conjunto Cantor, pequeno e enorme ao mesmo tempo}

\begin{problem}[{Problema de fim de semana --- os terços médios do Cantor
\omterm{def:b1:logic:sets}{definir}: comprimento zero, incontável, perfeito e $ C + C =
\intcc{0}{2}$}]
\label{pb:b1:topology:1}
Remover de $\intcc{0}{1}$ isso é \omterm{def:b1:topology:open}{abrir} terço médio, depois o meio
terço de cada segmento restante e repita para sempre: o que
sobrevive é o \emph{\omterm{pb:b1:topology:1}{Conjunto de cantores}} $ C $, o fundamental
fábrica de análise de contra-exemplos. Este problema o constrói,
lê através da base-$3$ maquinaria de \cref{pb:b1:reals:1},
e estabelece seu retrato paradoxal: comprimento total zero, mas
incontável; vazio \omterm{def:b1:topology:closure}{interior}, ainda sem ponto isolado; totalmente
desconectado, ainda $ C + C $ preenche todo o segmento
$\intcc{0}{2}$.\index{Conjunto de cantores} Formalmente: $ C_0 = \intcc{0}{1}$,
e $ C_{n+1}$ é obtido de $ C_n $ eliminando o \omterm{def:b1:topology:open}{abrir} meio
terço de cada segmento de $ C_n $; finalmente $ C = \bigcap_{n \geq 0}
C_n $. Durante todo o tempo, um \emph{código ternário} de $ x \in \intcc{0}{1}$
é qualquer sequência de dígitos $(d_k)_{k\geq1}$ com $ d_k \in \{0, 1, 2\}$
cujo valor $\sup_n \sum_{k=1}^n d_k 3^{-k}$ é igual $ x $ ---
códigos impróprios (eventualmente $2$) são permitidos; por
\cref{pb:b1:reals:1} (questões 9 a 11), cada $ x \in
\intcc{0}{1}$ tem um ou dois códigos, dois exatamente quando $ x = m/3^N
\in \intoo{0}{1}$.

\medskip
\noindent\textbf{Part I --- The construction.}
\begin{enumerate}
  \item Descrever $ C_1$ e $ C_2$ explicitamente como sindicatos de
        segmentos e prove por indução: $ C_n $ é um disjunto
        união de $2^n $ \omterm{def:b1:topology:closed}{fechado} segmentos, cada um de comprimento $3^{-n}$.
  \item Mostre isso $ C $ é \omterm{def:b1:topology:closed}{fechado}, limitado --- portanto compacto
        (\cref{thm:b1:topology:compact}) --- não vazio, e isso
        cada ponto final de cada segmento de cada $ C_n $ pertence
        para $ C $.
  \item O comprimento total de $ C_n $ é $\bigl(\frac23\bigr)^n $.
        Deduza isso para cada $\varepsilon > 0$, o \omterm{def:b1:logic:sets}{definir} $ C $ pode
        ser coberto por um número finito de segmentos de comprimento total
        $\leq \varepsilon $: o \omterm{pb:b1:topology:1}{Conjunto de cantores} tem \emph{comprimento
        zero}.
  \item Mostre que um \omterm{prop:b1:reals:intervals}{intervalo} contido em $ C $ tem comprimento $\leq
        3^{-n}$ para cada $ n $, portanto, é um singleton ou vazio:
        $\mathring C = \emptyset $. Ser \omterm{def:b1:topology:closed}{fechado} com vazio
        \omterm{def:b1:topology:closure}{interior}, $ C $ é \emph{em lugar nenhum \omterm{def:b1:topology:dense}{denso}}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The ternary code.}
\begin{enumerate}[resume]
  \item Prove a recursão de auto-similaridade
        \[
        C_{n+1} = \tfrac13 C_n \,\cup\,
        \bigl(\tfrac23 + \tfrac13 C_n\bigr),
        \qquad\text{hence}\qquad
        C = \tfrac13 C \,\cup\, \bigl(\tfrac23 + \tfrac13
        C\bigr),
        \]
        as duas peças sendo disjuntas: $ C $ são duas cópias de
        em escala $\frac13$.
  \item Prove por indução em $ n $: $ x \in C_n $ se e somente se
        $ x $ tem um código ternário cujo primeiro $ n $ dígitos estão em
        $\{0, 2\}$. Deduza, usando o fato de que $ x $ tem no máximo
        dois códigos: $ x \in C $ se e somente se $ x $ tem um código com
        \emph{nenhum dígito igual a $1$} (um \emph{$1$-livre} código).
  \item Códigos em ação: dê $1$-códigos gratuitos para $0$, $1$,
        $\frac13$, $\frac23$; mostrar $\frac14 = (0.\overline{02})_3$
        e $\frac34 = (0.\overline{20})_3$, então ambos pertencem a
        $ C $; e verifique isso $\frac14$ é \emph{não} um ponto final
        de qualquer $ C_n $ (os pontos finais têm o formato $ m/3^n $).
  \item Mostre que cada $ x \in C $ tem \emph{exatamente um} $1$-grátis
        código \emph{(quando $ x $ tem dois códigos, prove que exatamente
        um dos pares contém o dígito $1$)}. Conclua: o
        valor \omterm{def:b1:logic:map}{mapa} é uma bijeção de $\{0,2\}$-strings em
        $ C $.
  \item (Diagonal) Deixe $ k \mapsto x_k $ seja qualquer \omterm{def:b1:logic:map}{mapa} $\N^* \to C $.
        Construa um $\{0, 2\}$-string diferente no índice $ k $ de
        o código de $ x_k $, e concluir que $ C $ é incontável
        --- enquanto, por outro lado, a pergunta 3 diz que é
        metricamente desprezível.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Topological portrait.}
\begin{enumerate}[resume]
  \item Monte o registro até agora: $ C $ é compacto, incontável,
        de comprimento zero, em lugar nenhum \omterm{def:b1:topology:dense}{denso}. Qual única contenção
        $ C \subseteq C_n $ carrega cada propriedade?
  \item ($ C $ é perfeito) Deixe $ x \in C $ com $1$-código grátis
        $(d_k)$. Invertendo o dígito $ d_n $ ($0 \leftrightarrow
        2$) produz $ x_n \in C $ com $\abs{x_n - x} = 2 \cdot
        3^{-n}$. Conclua isso $ C $ não tem ponto isolado: cada
        ponto de $ C $ é um \omterm{def:b1:topology:closure}{limite} de \emph{outro} pontos de $ C $.
  \item Mostre que os extremos da questão 2 formam um número contável
        \omterm{def:b1:topology:dense}{denso} subconjunto de $ C $ \emph{(truncar o código depois $ n $
        dígitos e continue com $0$ é; contabilidade como em
        \cref{exo:b1:topology:10})}. Concluir: o ponto típico
        de $ C $ --- como $\frac14$ --- é \emph{não} um ponto final:
        pontos finais são um esqueleto contável dentro de um incontável
        corpo.
  \item (Totalmente desconectado) Vamos $ x < y $ em $ C $. Escolher $ n $
        com $3^{-n} < y - x $ e produzir um ponto $ z \in
        \intoo{x}{y}$ com $ z \notin C $. Conclua que o único
        não vazio \omterm{prop:b1:reals:intervals}{intervalos} contido em $ C $ são solteiros.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Arithmetic of $ C $.}
\begin{enumerate}[resume]
  \item Mostrar $1 - C = C $ \emph{(o que faz $ x \mapsto 1 - x $ fazer
        para um $1$-código grátis? lembrar $1 = (0.\overline{2})_3$)}.
  \item (Adição de códigos) Mostre que se $ x $, $ x'$ tem códigos
        $(a_k)$, $(b_k)$, então $ x + x' = \lim_n\,(t_n + t'_n)$
        onde $ t_n, t'_n $ são as somas parciais. Deduza: cada $ y
        \in \intcc{0}{1}$ é o \emph{ponto médio} de dois pontos
        de $ C $ --- dado um código $(e_k)$ de $ y $, escolha dígitos
        $ a_k, b_k \in \{0, 2\}$ com $\frac{a_k + b_k}{2} =
        e_k $.
  \item Concluir $ C + C = \intcc{0}{2}$ e, com a questão 14,
        $ C - C = \intcc{-1}{1}$. Instância concreta: escrever $1$ como
        uma soma dos dois não-pontos encontrados na questão 7.
  \item Refletir: um \omterm{def:b1:logic:sets}{definir} de comprimento zero cuja diferença \omterm{def:b1:logic:sets}{definir} preenche
        $\intcc{-1}{1}$. Por que não há contradição entre
        ``$ C $ é metricamente insignificante'' e ``$ C + C $ está cheio
        comprimento''? (Uma frase; pense em quanto tempo
        controla e o que não faz.)
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Members, rational and irrational.}
\begin{enumerate}[resume]
  \item Combine a questão 8 com o critério de periodicidade de
        \cref{pb:b1:reals:1}: um ponto de $ C $ é racional se e
        só se for $1$-o código livre é eventualmente periódico. Corre
        a base-$3$ \omterm{thm:b1:logic:partition}{divisão} longa para verificar $\frac1{13} =
        (0.\overline{002})_3 \in C $.
  \item Produza um explicitamente \emph{irracional} membro de $ C $:
        o valor do código com $ d_k = 2$ no triangular
        posições $ k = \frac{j(j+1)}{2}$ e $ d_k = 0$
        em outro lugar. Justifique a irracionalidade pelas lacunas crescentes
        argumento de \cref{pb:b1:reals:1} (questão 20).
  \item (Em um segmento completo) Considere $ h $ mapeando o ponto de
        $ C $ com $1$-código grátis $(d_k)$ ao valor do
        \emph{binário} corda $\bigl(\frac{d_k}2\bigr)$, ou seja\
        $ h(x) = \sup_n \sum_{k=1}^n \frac{d_k}{2}\,2^{-k}$.
        Mostre isso $ h $ \omterm{def:b1:logic:map}{mapas} $ C $ \emph{para} $\intcc{0}{1}$. Então
        o insignificante $ C $ sobrepõe-se a um segmento de cheio
        comprimento --- uma segunda prova de que $ C $ é incontável.
  \item (Comprimento auto-semelhante) Suponha alguma noção de comprimento $ L $
        foram definidos para $ C $ e suas cópias reduzidas, respeitando
        dimensionamento ($ L(\lambda A) = \lambda L(A)$), tradução
        invariância e aditividade sobre a decomposição disjunta
        da questão 5. Mostre isso então $ L(C) = \frac23\,L(C)$,
        forçando $ L(C) = 0$: auto-semelhança sozinha já
        sentenças $ C $ para comprimento zero.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- A fat cousin, and the moral.}
\begin{enumerate}[resume]
  \item (Gordo \omterm{pb:b1:topology:1}{Conjunto de cantores}) Repita a construção, mas na fase
        $ n $ ($ n = 0, 1, 2, \dots $) remova de cada um dos $2^n $
        segmentos atuais uma central \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalo} de comprimento
        $4^{-(n+1)}$ apenas. Mostre que os comprimentos dos segmentos $ l_n $
        obedecer $ l_{n+1} = \frac{l_n - 4^{-(n+1)}}{2}$, $ l_n =
        \frac{2^n + 1}{2\cdot 4^n} > 0$, que o resultado
        $ K = \bigcap K_n $ é compacto com vazio interiore
        que o comprimento total removido é $\sum_{n\geq0} 2^n
        4^{-(n+1)} = \frac12$. Admitindo o (intuitivo, Ano 3)
        aditividade de comprimento para uniões finitas de intervalose
        usando ambos declarações de \cref{ex:b1:topology:nested},
        Mostre que qualquer família finita de \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalos} cobertura
        $ K $ tem comprimento total $\geq \frac12$: $ K $ não está em lugar nenhum
        \omterm{def:b1:topology:dense}{denso} mas \emph{não} insignificante. A pequenez tem vários
        significados desiguais.
  \item (Distâncias) Mostre que para um não vazio \omterm{def:b1:topology:closed}{fechado} $ F \subseteq
        \R $ e $ x \in \R $, o \omterm{def:b1:reals:bounds}{ínfimo} $ d(x, F)$ é
        \emph{alcançado} \emph{(minimizando a sequência mais
        Bolzano-Weierstrass)}. Então calcule
        \[
        \max_{y \in \intcc{0}{1}} d(y, C) = \frac16 ,
        \]
        alcançado exatamente no centro $ y = \frac12$ \emph{(um
        ponto de uma lacuna criada na fase $ n $ está dentro
        $\frac{3^{-n}}{2}$ dos pontos finais da lacuna, que ficam em
        $ C $)}.
  \item (Cada ponto é um \omterm{def:b1:topology:closure}{limite} subseqüente) Usando as questões 12
        e 2, produzem uma única sequência em $ C $ cujo \omterm{def:b1:logic:sets}{definir} de
        \omterm{def:b1:topology:closure}{limites} subseqüentes é \emph{todos} de $ C $. (Comparar:
        para uma sequência convergente que \omterm{def:b1:logic:sets}{definir} é um ponto ---
        $ C $ percebe o extremo oposto entre os compactos.)
  \item Síntese, uma frase cada: (i) quais teoremas deste
        capítulo a construção realmente consumiu (estabilidade
        de conjuntos \omterm{def:b1:topology:closed}{fechados}, compacidade, sequencial
        caracterizações)? (ii) listar os quatro paradoxos
        pares do retrato (comprimento zero/incontável,
        \omterm{def:b1:topology:closed}{fechado}/vazio \omterm{def:b1:topology:closure}{interior}, perfeito/totalmente desconectado,
        insignificante/$ C+C $ completo); (iii) onde $ C $ ressurgir
        mais tarde (a escada do diabo construída sobre $ h $ na teoria
        de continuidade e a teoria da medida do Ano 3
        volume, onde $ C$ separa ``contável'' de
        ``insignificante'')?
\end{enumerate}
\end{problem}
