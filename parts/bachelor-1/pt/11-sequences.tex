\chapter{Sequências}\label{ch:b1:seq}

As sequências foram manipuladas no volume High School com o limite
conceito tomado metade da fé. Aqui a teoria é reconstruída
completude de $\R $ (\cref{ch:b1:reals}): todo teorema clássico
--- convergência monótona, \omterm{thm:b1:seq:adjacent}{sequências adjacentes}, Bolzano-Weierstrass,
o critério de Cauchy --- é uma face desse único axioma. O capítulo
termina com o estudo prático de sequências definidas por $ u_{n+1} =
f(u_n)$.

\section{Convergência}

\begin{definition}[Limite de uma sequência]\label{def:b1:seq:limit}
Uma sequência $(u_n)$ de reais \emph{converge} para $\ell \in \R $
quando\index{limite!de uma sequência}
\[
\forall \varepsilon > 0,\ \exists N \in \N,\ \forall n \geq N,
\qquad \abs{u_n - \ell} \leq \varepsilon .
\]
Um escreve $ u_n \to \ell $ ou $\lim u_n = \ell $. Uma sequência que faz
não converge (para qualquer real) \emph{diverge}. Divergência \emph{para
$+\infty $}: $\forall M,\ \exists N,\ \forall n \geq N,\ u_n \geq M $
(de forma similar $-\infty $).
\end{definition}

\begin{example}[Um $\varepsilon $--$ N $ prova, escrita uma vez]\label{ex:b1:seq:epsilonN}
Alegar: $ u_n = \dfrac{n^2 + 1}{2n^2 - 3} \to \dfrac12$. Primeiro
isole o erro:
\[
\Bigl| u_n - \frac12 \Bigr|
= \Bigl| \frac{2(n^2 + 1) - (2n^2 - 3)}{2(2n^2 - 3)} \Bigr|
= \frac{5}{2\,\abs{2n^2 - 3}}
= \frac{5}{2\,(2n^2 - 3)} \quad (n \geq 2).
\]
Então domine-o com algo simples: por $ n \geq 2$, $2n^2 - 3
\geq n^2$, então o erro é $\leq \frac{5}{2n^2} \leq \frac 5{2n}$.
Dado $\varepsilon > 0$, o \omterm{thm:b1:reals:archimedes}{Propriedade arquimediana} fornece $ N \geq
\max\bigl(2, \frac{5}{2\varepsilon}\bigr)$; para $ n \geq N $ o
erro é $\leq \varepsilon $. Feito. A visão final: uma
$\varepsilon $--$ N $ a prova tem exatamente três movimentos --- calcule o
erro, limitando-o por uma expressão elementar decrescente, resolva para
o limite --- e depois dos teoremas deste capítulo (operações,
squeeze) quase nunca mais se escreve tal prova: os teoremas
empacotar os três movimentos de uma vez por todas.
\end{example}

\begin{example}[Divergência ao infinito, certificada]\label{ex:b1:seq:toinfinity}
Alegar: $ u_n = n^2 - 100n \to +\infty $. Fatore o termo dominante:
$ u_n = n^2\bigl(1 - \frac{100}{n}\bigr) \geq \frac{n^2}{2}$ para
$ n \geq 200$. Dado $ M $, pegar $ N = \max\bigl(200,
\lceil\sqrt{2M}\rceil\bigr)$: para $ n \geq N $, $ u_n \geq
\frac{n^2}{2} \geq M $. Dois hábitos estão em exibição: termo dominante
factoring converte uma competição ($ n^2$ contra $-100n $) em um
escala única vezes um fator que tende a $1$; e o limite
pode ser enorme ($ u_{100} = 0$, a sequência é ainda negativa antes
$ n = 100$) --- divergência para $+\infty $ é um \omterm{def:b1:logic:statement}{declaração} sobre o
cauda, indiferente a qualquer quantidade finita de mau comportamento.
\end{example}

\begin{proposition}[Primeiras propriedades]\label{prop:b1:seq:first}
\begin{enumerate}
  \item O limite, se existir, é único.
  \item Uma sequência convergente é limitada.
  \item Se $ u_n \to \ell $, toda modificação de um número finito de termos
        deixa a convergência e o limite inalterados.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) Se $ u_n \to \ell $ e $ u_n \to \ell'$ com $\ell \neq \ell'$,
pegar $\varepsilon = \frac{\abs{\ell - \ell'}}{3}$: além dos dois
limites, $\abs{\ell - \ell'} \leq \abs{\ell - u_n} + \abs{u_n -
\ell'} \leq 2\varepsilon = \frac23 \abs{\ell - \ell'}$, absurdo.

(2) Com $\varepsilon = 1$: além $ N $, $\abs{u_n} \leq \abs\ell +
1$; os finitos termos anteriores também são limitados, então $\abs{u_n}
\leq \max(\abs{u_0}, \dots, \abs{u_{N-1}}, \abs\ell + 1)$.

(3) Em detalhes: suponha $ v_n = u_n $ para $ n \geq n_0$ e $ u_n \to
\ell $. Dado $\varepsilon > 0$, pegue o limite $ N $ para
$(u_n)$: para $ n \geq \max(N, n_0)$, $\abs{v_n - \ell} = \abs{u_n
- \ell} \leq \varepsilon $. Então $ v_n \to \ell $: a definição
quantifica apenas mais $ n \geq N $, e qualquer prefixo finito pode ser
substituídos à custa do alargamento do limiar. (É por isso
hipóteses ``para todos os grandes $ n $'' suficiente em todos os lugares neste
capítulo.)
\end{proof}

\begin{theorem}[Operações sobre limites]\label{thm:b1:seq:operations}
Se $ u_n \to \ell $ e $ v_n \to m $, então
\[
u_n + v_n \to \ell + m, \qquad u_n v_n \to \ell m, \qquad
\frac{u_n}{v_n} \to \frac{\ell}{m} \ (\text{if } m \neq 0),
\qquad \abs{u_n} \to \abs\ell .
\]
\end{theorem}

\begin{proof}
\emph{Soma:} $\abs{(u_n + v_n) - (\ell + m)} \leq \abs{u_n - \ell} +
\abs{v_n - m} \leq 2\varepsilon $ além do limite maior.
\emph{Produto:} escrever
\[
u_n v_n - \ell m = (u_n - \ell)\,v_n + \ell\,(v_n - m);
\]
$(v_n)$ é limitado por alguns $ B $ (\cref{prop:b1:seq:first}), então o
lado direito é $\leq B\abs{u_n - \ell} + \abs{\ell}\,\abs{v_n - m}$,
arbitrariamente pequeno. \emph{Quociente:} basta tratar $\frac
1{v_n}$. Com $\varepsilon = \frac{\abs m}{2}$: além de alguns $ N_0$,
$\abs{v_n} \geq \frac{\abs m}{2}$, então
\[
\Bigl| \frac{1}{v_n} - \frac 1m \Bigr|
= \frac{\abs{m - v_n}}{\abs{v_n m}}
\leq \frac{2}{m^2}\,\abs{v_n - m} \longrightarrow 0 .
\]
\emph{Valor absoluto:} $\bigl|\abs{u_n} - \abs\ell\bigr| \leq
\abs{u_n - \ell}$ (desigualdade triangular reversa,
\cref{prop:b1:complex:rules}).
\end{proof}

\begin{example}[Operações mais um truque algébrico]\label{ex:b1:seq:conjugatetrick}
Calcular $\lim\,\bigl(\sqrt{n^2 + n} - n\bigr)$. As duas peças
separadamente tendem a $+\infty $: o teorema das operações não diz nada
sobre sua diferença (um \emph{forma indeterminada}). Multiplicar
pelo \omterm{def:b1:complex:field}{conjugado}:
\[
\sqrt{n^2 + n} - n
= \frac{(n^2 + n) - n^2}{\sqrt{n^2 + n} + n}
= \frac{n}{\sqrt{n^2+n} + n}
= \frac{1}{\sqrt{1 + \frac1n} + 1} .
\]
Agora tudo converge: $\sqrt{1 + \frac1n} \to 1$, porque
$0 \leq \sqrt{1 + h} - 1 = \frac{h}{\sqrt{1+h} + 1} \leq h $
(conjugue novamente e depois aperte com $ h = \frac1n $); então o
teorema das operações dá o limite $\frac{1}{1 + 1} =
\frac12$. A conclusão final: o teorema das operações não é uma
calculadora para todos os limites --- formas indeterminadas ($\infty -
\infty $, $\frac00$, $0 \times \infty $, $1^\infty $) deve primeiro ser
\emph{transformado} por álgebra (\omterm{def:b1:complex:field}{conjugados}, fatorando o dominante
prazo) até que cada peça convirja; a máquina sistemática para
casos resistentes é a expansão assintótica de
\cref{ch:b1:taylor}.
\end{example}

\begin{theorem}[Limites e ordem]\label{thm:b1:seq:order}
\begin{enumerate}
  \item Se $ u_n \leq v_n $ para todos os grandes $ n $, e ambos convergem, então
        $\lim u_n \leq \lim v_n $. (Desigualdades estritas \emph{não}
        passar ao limite: $\frac 1n > 0$ mas $\lim = 0$.)
  \item (Teorema da compressão\index{teorema do aperto}) Se $ u_n \leq w_n
        \leq v_n $ para todos os grandes $ n $ e $ u_n, v_n \to \ell $, então
        $ w_n \to \ell $.
  \item Se $ u_n \to \ell > 0$, então $ u_n > \frac\ell2 > 0$ para todos
        grande $ n $.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Suponha $\ell = \lim u_n > m = \lim v_n $; com $\varepsilon =
\frac{\ell - m}{3}$, termos grandes satisfazem $ v_n \leq m + \varepsilon <
\ell - \varepsilon \leq u_n $, contradizendo $ u_n \leq v_n $.

(2) Além dos limites: $\ell - \varepsilon \leq u_n \leq w_n \leq
v_n \leq \ell + \varepsilon $.

(3) é \cref{def:b1:seq:limit} com $\varepsilon = \frac\ell2$.
\end{proof}

\begin{example}[Dois apertos]\label{ex:b1:seq:squeezes}
(eu) $\dfrac{\sin n}{n} \to 0$: de $-\frac1n \leq \frac{\sin
n}{n} \leq \frac1n $, ambas as paredes desabaram $0$ --- não há necessidade
entender o numerador errático. (ii) $(2^n +
3^n)^{1/n} \to 3$: enquadrar o interior,
\[
3^n \leq 2^n + 3^n \leq 2\cdot3^n
\quad\Longrightarrow\quad
3 \leq (2^n + 3^n)^{1/n} \leq 3\cdot 2^{1/n} ,
\]
e $2^{1/n} = \eu^{\frac{\ln 2}{n}} \to 1$ (como para $5^{1/n}$ em
\cref{exo:b1:seq:2}): o aperto entrega $3$. O fechamento
insight: uma soma de exponenciais concorrentes se comporta como seu
\emph{maior} termo --- os menores são absorvidos por um
fator constante inofensivo, que o $ n $-th root então apaga.
\end{example}

\section{Sequências monótonas}

\begin{theorem}[Teorema do limite monótono]\label{thm:b1:seq:monotone}
Uma sequência crescente limitada acima converge, para $\sup\{u_n : n \in
\N\}$; uma sequência crescente não limitada acima diverge para
$+\infty $. (Espelho declarações para diminuir
sequências.)\index{teorema do limite monótono}
\end{theorem}

\begin{proof}
Deixar $ s = \sup\{u_n\}$ (\cref{thm:b1:reals:sup}). Dado $\varepsilon >
0$, o $\varepsilon $-caracterização
(\cref{prop:b1:reals:epsilon}) rendimentos $ N $ com $ u_N > s -
\varepsilon $; pela monotonicidade, $ s - \varepsilon < u_N \leq u_n \leq
s $ para todos $ n \geq N $: convergência para $ s $. Se ilimitado: para cada
$ M $ alguns $ u_N > M $, e a monotonicidade mantém todos os termos posteriores acima $ M $.
\end{proof}

\begin{example}[O teorema monótono como máquina de existência]\label{ex:b1:seq:existencemachine}
Deixar $ u_n = \prod_{k=1}^{n} \bigl(1 + \frac{1}{2^k}\bigr)$. Cada
fator excede $1$, então $(u_n)$ está aumentando. Limitada acima? Pegue
logaritmos e uso $\ln(1 + x) \leq x $
(\cref{ex:b1:derivative:convexineq} antecipa isso; ou o
bruto $1 + x \leq \eu^x $ do volume do Ensino Médio):
\[
\ln u_n = \sum_{k=1}^{n} \ln\Bigl(1 + \frac{1}{2^k}\Bigr)
\leq \sum_{k=1}^{n} \frac{1}{2^k} < 1 ,
\]
então $ u_n < \eu $. Crescente e limitado: $(u_n)$ converge para alguns
$\ell \in \intoc{u_1}{\eu}$ --- um real perfeitamente bem definido
número sem forma fechada à vista ($\ell = 2.384\dots $). O
conclusão final: o teorema do limite monótono é o mais barato
máquina de existência em análise; nomeou $\eu $ em si
(\cref{ex:b1:seq:e} abaixo) e em \cref{ch:b1:series} isso vai
decidir a convergência de cada série positiva por mera
limitação.
\end{example}

\begin{theorem}[Sequências adjacentes]\label{thm:b1:seq:adjacent}
Deixar $(a_n)$ estar aumentando, $(b_n)$ diminuindo, com $ b_n - a_n \to
0$. Então ambos convergem, para um \emph{comum} limite $\ell $, e $ a_n
\leq \ell \leq b_n $ para todos $ n $.\index{sequências adjacentes}
\end{theorem}

\begin{proof}
Primeiro, $ a_n \leq b_n $ para todos $ n $: a sequência $(b_n - a_n)$ é
diminuindo e tende a $0$, então é $\geq 0$ (um termo negativo seria
congele abaixo $0$). Então $(a_n)$ está aumentando e limitado acima por
$ b_0$: converge para alguns $\ell $ (\cref{thm:b1:seq:monotone});
da mesma maneira $(b_n) \to \ell'$; e $\ell' - \ell = \lim (b_n - a_n) =
0$. As desigualdades $ a_n \leq \ell \leq b_n $ seguir de
monotonicidade ($\ell = \sup a_k \geq a_n $, etc.).
\end{proof}

\begin{example}[O número $\eu $]\label{ex:b1:seq:e}
\omterm{def:b1:logic:sets}{Definir} $ a_n = \sum_{k=0}^{n} \frac{1}{k!}$ e $ b_n = a_n + \frac{1}{n
\cdot n!}$ ($ n \geq 1$). Então $(a_n)$ aumenta; e
\[
b_{n+1} - b_n = \frac{1}{(n+1)!} + \frac{1}{(n+1)(n+1)!} -
\frac{1}{n\,n!}
= \frac{n(n+1) + n - (n+1)^2}{n(n+1)(n+1)!}
= \frac{-1}{n(n+1)(n+1)!} < 0 ,
\]
então $(b_n)$ diminui, e $ b_n - a_n \to 0$: adjacente. Seu comum
limite é (por definição aqui) o número $\eu \approx 2.71828$; o
desigualdades $ a_n < \eu < b_n $ são afiados o suficiente para provar $\eu \notin
\Q $ (\cref{exo:b1:seq:9}).
\end{example}

\section{Subsequências e Bolzano--Weierstrass}

\begin{definition}[Subsequência]\label{def:b1:seq:subsequence}
UM \emph{subsequência}\index{subsequência} de $(u_n)$ é uma sequência
$(u_{\varphi(n)})$ onde $\varphi \colon \N \to \N $ é estritamente
aumentando (nota $\varphi(n) \geq n $, por indução).
\end{definition}

\begin{proposition}\label{prop:b1:seq:subsequences}
Se $ u_n \to \ell $ ($\ell \in \R $ ou $\pm\infty $), todo \omterm{def:b1:seq:subsequence}{subsequência}
tende a $\ell $. Consequentemente, uma sequência com dois \omterm{def:b1:seq:subsequence}{subsequências} de
limites diferentes divergem. Por outro lado, se $(u_{2n})$ e
$(u_{2n+1})$ ambos convergem para o \emph{mesmo} $\ell $, então $ u_n \to
\ell $.
\end{proposition}

\begin{proof}
Além do limite $ N $ para $(u_n)$, todos os índices $\varphi(n) \geq n
\geq N $ qualificar (a desigualdade $\varphi(n) \geq n $ é o
indução anotada em \cref{def:b1:seq:subsequence}: $\varphi(0)
\geq 0$, e $\varphi(n+1) > \varphi(n) \geq n $ forças
$\varphi(n+1) \geq n + 1$). Pelo contrário: dado
$\varepsilon $, pegue os dois limites $ N_0$ (iguais) e $ N_1$
(chances); um índice arbitrário $ n \geq \max(2N_0, 2N_1 + 1)$ é
ou mesmo, $ n = 2k $ com $ k \geq N_0$, ou estranho, $ n = 2k+1$ com
$ k \geq N_1$ --- em ambos os casos $\abs{u_n - \ell} \leq
\varepsilon $: cada índice é coberto por um dos dois
\omterm{def:b1:seq:subsequence}{subsequências}, e esse é o ponto principal.
\end{proof}

\begin{example}[Limites subsequentes]\label{ex:b1:seq:sublimits}
Para $ u_n = (-1)^n \frac{n}{n+1}$: o par \omterm{def:b1:seq:subsequence}{subsequência} tende a
$1$, o estranho para $-1$, então a sequência diverge --- mas isso acontece
de forma organizada, agrupando-se em torno dos dois valores $\pm 1$.
Para $ u_n = \cos\frac{2\pi n}{3}$: os três \omterm{def:b1:seq:subsequence}{subsequências} de
índices $3k $, $3k + 1$, $3k + 2$ são constantes, iguais a $1$,
$-\frac12$, $-\frac12$; o \omterm{def:b1:logic:sets}{definir} de limites subseqüentes é $\{1,
-\frac12\}$. A visão final: uma \emph{limitado} sequência
converge exatamente quando tem um único limite subseqüente
(\cref{exo:b1:seq:8}); divergência de uma sequência limitada sempre
significa pelo menos dois clusters, e Bolzano--Weierstrass abaixo
garante que há pelo menos um.
\end{example}

\begin{theorem}[Bolzano-Weierstrass]\label{thm:b1:seq:bw}
Toda sequência limitada de reais tem um convergente
\omterm{def:b1:seq:subsequence}{subsequência}.\index{Teorema de Bolzano-Weierstrass}
\end{theorem}

\begin{proof}
Deixar $ u_n \in \intcc{a}{b}$ para todos $ n $. Crie segmentos aninhados por
dicotomia: conjunto $\intcc{a_0}{b_0} = \intcc{a}{b}$; dado
$\intcc{a_k}{b_k}$ contendo $ u_n $ por infinitos $ n $, um dos
suas duas metades ainda contém $ u_n $ por infinitos $ n $ --- ligue
isso $\intcc{a_{k+1}}{b_{k+1}}$. As sequências $(a_k)$, $(b_k)$ são
adjacente ($ b_k - a_k = \frac{b-a}{2^k} \to 0$), com limite comum
$\ell $ (\cref{thm:b1:seq:adjacent}).

Extrair: escolha $\varphi(0)$ com $ u_{\varphi(0)} \in
\intcc{a_0}{b_0}$, então, indutivamente, $\varphi(k+1) > \varphi(k)$
com $ u_{\varphi(k+1)} \in \intcc{a_{k+1}}{b_{k+1}}$ --- possível
já que esse segmento contém infinitos termos. Então $ a_k \leq
u_{\varphi(k)} \leq b_k $, e o teorema da compressão dá
$ u_{\varphi(k)} \to \ell $.
\end{proof}

\begin{remark}[O que Bolzano-Weierstrass faz e não diz]
Isto \emph{faz} dizem: apenas por limitação, alguns \omterm{def:b1:seq:subsequence}{subsequência}
converge --- existência sem fórmula, como prova da dicotomia
deixa claro (nada nos diz \emph{qual} os índices sobrevivem). Isso
faz \emph{não} digamos que o limite é único: $((-1)^n)$ tem
\omterm{def:b1:seq:subsequence}{subsequências} convergindo para $1$ e para $-1$, e o \omterm{def:b1:logic:sets}{definir} de
limites subseqüentes podem até ser infinitos
(\cref{ex:b1:seq:sublimits}, e todo o Cantor \omterm{def:b1:logic:sets}{definir} em
\cref{pb:b1:topology:1}). Não sobrevive à ilimitação:
$(n)$ não tem convergente \omterm{def:b1:seq:subsequence}{subsequência} em absoluto --- embora se possa
sempre extraia um \omterm{def:b1:seq:subsequence}{subsequência} tendendo a $+\infty $ ou $-\infty $
de qualquer sequência ilimitada (escolha $\varphi(k)$ com
$ u_{\varphi(k)} \geq k $, dizer). Usado corretamente, o teorema é um
\emph{bomba de existência}: aparece no ponto crucial do Cauchy
critério abaixo, do teorema de Heine, e do valor extremo
teorema --- sempre para produzir um ponto que nenhum explícito
ofertas de construção.
\end{remark}

\section{Sequências e completude de Cauchy}

\begin{definition}\label{def:b1:seq:cauchy}
Uma sequência $(u_n)$ é um \emph{Sequência de Cauchy}\index{Sequência de Cauchy}
quando seus termos se tornam arbitrariamente próximos \emph{um para o outro}:
\[
\forall \varepsilon > 0,\ \exists N,\ \forall p, q \geq N,
\qquad \abs{u_p - u_q} \leq \varepsilon .
\]
\end{definition}

\begin{example}[Verificando a propriedade Cauchy manualmente]\label{ex:b1:seq:cauchyhand}
Deixar $ u_n = \sum_{k=0}^{n} \frac{\cos k}{2^k}$ --- sem monotonicidade,
nenhum limite adivinhável. Para $ p > q $:
\[
\abs{u_p - u_q}
= \Bigl| \sum_{k=q+1}^{p} \frac{\cos k}{2^k} \Bigr|
\leq \sum_{k=q+1}^{p} \frac{1}{2^k}
< \frac{1}{2^{q}} ,
\]
pela desigualdade triangular, $\abs{\cos k} \leq 1$ e um finito
soma geométrica. Dado $\varepsilon > 0$, escolher $ N $ com $2^{-N}
\leq \varepsilon $: todas as lacunas além $ N $ são $\leq \varepsilon $,
a sequência é Cauchy, portanto converge --- para um limite ninguém
posso nomear de forma fechada, que é exatamente o ponto. O fechamento
insight: o domínio geométrico dos incrementos é o padrão
maneira de ganhar a propriedade Cauchy, e \cref{ch:b1:series} vontade
engarrafar o argumento como "a convergência absoluta implica
convergência''.
\end{example}

\begin{theorem}[Completude de $\R $]\label{thm:b1:seq:complete}
Uma sequência de reais converge se e somente se for Cauchy
sequência.\index{completeness of R@completeness of $\R $}
\end{theorem}

\begin{proof}
($\Rightarrow $) Se $ u_n \to \ell $: além do limite para
$\frac\varepsilon2$, $\abs{u_p - u_q} \leq \abs{u_p - \ell} +
\abs{\ell - u_q} \leq \varepsilon $.

($\Leftarrow $) Deixar $(u_n)$ seja Cauchy. \emph{É limitado}: com
$\varepsilon = 1$, além $ N $ todos os termos estão dentro $1$ de $ u_N $ e
a cabeça é finita. \emph{Extrair}: por
\cref{thm:b1:seq:bw}, alguns \omterm{def:b1:seq:subsequence}{subsequência} $ u_{\varphi(n)} \to \ell $.
\emph{Concluir}: dado $\varepsilon > 0$, pegar $ N $ (Cauchy, por
$\frac\varepsilon2$) e $ n \geq N $ com $\abs{u_{\varphi(n)} - \ell}
\leq \frac\varepsilon2$ e $\varphi(n) \geq N $; então para cada $ p
\geq N $:
\[
\abs{u_p - \ell} \leq \abs{u_p - u_{\varphi(n)}} +
\abs{u_{\varphi(n)} - \ell} \leq \varepsilon . \qedhere
\]
\end{proof}

\begin{remark}
O valor do critério: certifica a convergência \emph{sem
nomeando o limite}. Ele falha $\Q $ (os truncamentos decimais de
$\sqrt 2$ formar um \omterm{def:b1:seq:cauchy}{Sequência de Cauchy} de racionais sem racional
limite): completude é uma propriedade de $\R $, equivalente ao \omterm{def:b1:reals:bounds}{superior
vinculado} axioma. É também o carro-chefe da convergência de
série (\cref{ch:b1:series}).
\end{remark}

\begin{example}[Uma sequência de Cauchy com limite invisível]\label{ex:b1:seq:basel}
Deixar $ S_n = \sum_{k=1}^{n} \frac{1}{k^2}$. Para $ p > q \geq 1$:
\[
S_p - S_q = \sum_{k=q+1}^{p} \frac{1}{k^2}
\leq \sum_{k=q+1}^{p} \frac{1}{k(k-1)}
= \sum_{k=q+1}^{p} \Bigl(\frac{1}{k-1} - \frac 1k\Bigr)
= \frac 1q - \frac 1p < \frac 1q ,
\]
tão além $ N > \frac1\varepsilon $ todas as lacunas são $\leq \varepsilon $:
$(S_n)$ é Cauchy, portanto converge. Observe o que aconteceu: nós
provou que existe um número real específico sem ter qualquer nome
por isso. (Isso é $\frac{\pi^2}{6}$ --- uma identidade célebre de
Euler, provado no volume do Ano 2; nada neste capítulo poderia
diga-nos isso.) Esta \omterm{thm:b1:logic:partition}{divisão} do trabalho --- existência agora,
identificação mais tarde, se alguma vez --- é todo o critério de Cauchy
ponto, e o motor da teoria das séries em
\cref{ch:b1:series}.
\end{example}

\section{Sequências recorrentes}

\begin{method}[Estudo $ u_{n+1} = f(u_n)$]\label{met:b1:seq:recurrent}
Dado $ f $ e um ponto de partida $ u_0$:\index{sequência recorrente}
\begin{enumerate}
  \item \emph{Estável \omterm{prop:b1:reals:intervals}{intervalo}:} encontre um \omterm{prop:b1:reals:intervals}{intervalo} $ I $ com $ f(I)
        \subseteq I $ contendo $ u_0$: então tudo $ u_n \in I $ (por
        indução).
  \item \emph{Limites de candidatos:} se $ u_n \to \ell \in I $ e $ f $ é
        contínuo em $\ell $ (\cref{ch:b1:continuity}), então $\ell $
        é um \emph{ponto fixo}: $ f(\ell) = \ell $. Resolver $ f(x) = x $.
  \item \emph{Monotonicidade:} se $ f $ está aumentando em $ I $, então
        $(u_n)$ é monotônico (aumentando se $ u_1 \geq u_0$,
        diminuindo caso contrário); combinado com limitação,
        \cref{thm:b1:seq:monotone} conclui. Se $ f $ está diminuindo,
        estude os dois \omterm{def:b1:seq:subsequence}{subsequências} $(u_{2n})$ e $(u_{2n+1})$,
        que são monotônicos para $ f \circ f $.
  \item \emph{Controle de erros:} uma desigualdade $\abs{f(x) - \ell} \leq
        k\abs{x - \ell}$ com $ k < 1$ dá $\abs{u_n - \ell} \leq
        k^n \abs{u_0 - \ell} \to 0$ diretamente.
\end{enumerate}
\end{method}

\begin{example}[Método de Heron]\label{ex:b1:seq:heron}
Deixar $ u_0 = 2$ e $ u_{n+1} = \dfrac12\Bigl(u_n +
\dfrac{2}{u_n}\Bigr)$: o antigo algoritmo para
$\sqrt 2$.\index{Método de Heron}
\begin{itemize}
  \item \emph{Estabilidade:} para $ x > 0$, a média aritmética-geométrica
        desigualdade dá $\frac12(x + \frac2x) \geq \sqrt{x \cdot
        \frac 2x} = \sqrt 2$; então $ I = \intco{\sqrt 2}{+\infty}$ é
        estável e contém $ u_1$ (de fato $ u_1 = \frac32 \geq
        \sqrt 2$).
  \item \emph{Monotonicidade:} para $ x \geq \sqrt 2$, $\;x - f(x) =
        \frac{x^2 - 2}{2x} \geq 0$: a sequência diminui de
        $ u_1$ ligado, e é limitado abaixo por $\sqrt 2$: converge.
  \item \emph{Limite:} os pontos fixos resolvem $ x = \frac12(x +
        \frac2x)$, ou seja\ $ x^2 = 2$: sobre $ I $, $\ell = \sqrt 2$.
  \item \emph{Velocidade:} $ u_{n+1} - \sqrt 2 = \frac{(u_n -
        \sqrt2)^2}{2u_n} \leq \frac{(u_n - \sqrt 2)^2}{2\sqrt 2}$:
        o número de dígitos corretos aproximadamente \emph{dobra} em cada
        passo (convergência quadrática).
\end{itemize}
\end{example}

\begin{remark}[Armadilhas comuns com limites]
Quatro clássicos. (eu) \emph{Pequenos passos não implicam convergência}:
$ u_{n+1} - u_n \to 0$ é muito mais fraco do que a propriedade Cauchy ---
as somas harmônicas $ H_n $ tem passos $\frac{1}{n+1} \to 0$ ainda
divergir para $+\infty $ (\cref{exo:b1:seq:5}); a condição de Cauchy
controles $\abs{u_p - u_q}$ para \emph{todos} pares grandes, não
consecutivos. (ii) \emph{Desigualdades estritas morrem no
limite}: de $ u_n < v_n $ para todos $ n $ só se consegue $\lim u_n
\leq \lim v_n $ (\cref{thm:b1:seq:order}); $\frac1n > 0$ e ainda
$\lim = 0$. (iii) \emph{Limitado não é convergente}: $((-1)^n)$
é limitado e diverge; limitação mais \emph{monotonicidade}
converge, a limitação por si só garante uma convergência
\omterm{def:b1:seq:subsequence}{subsequência} (\cref{thm:b1:seq:bw}). (4) \emph{O ponto fixo
equação vem em segundo lugar, não em primeiro}: para $ u_{n+1} = f(u_n)$,
resolvendo $ f(\ell) = \ell $ identifica o limite \emph{só depois}
a convergência está provada. A recorrência $ u_{n+1} = 2u_n $ tem o
ponto fixo único $\ell = 0$, ainda de $ u_0 = 1$ a sequência
corre para $+\infty $: a equação $\ell = 2\ell $ nunca foi
tem direito a um limite. Ordem das operações, sempre: existência
primeiro (\cref{met:b1:seq:recurrent}, etapas 1 a 3), identificação
segundo.
\end{remark}

\begin{example}[Uma diminuição $ f $: a recorrência dourada]\label{ex:b1:seq:golden}
Deixar $ u_0 = 1$ e $ u_{n+1} = \dfrac{1}{1 + u_n}$. Aqui $ f(x) =
\frac{1}{1+x}$ é \emph{diminuindo}, então a sequência não é
monotônico (alterna em torno do seu limite); a etapa de contração
de \cref{met:b1:seq:recurrent} é a ferramenta certa. Estabilidade: se
$ x \in \intcc{\frac12}{1}$ então $1 + x \in
\intcc{\frac32}{2}$, então $ f(x) \in \intcc{\frac12}{\frac23}
\subseteq \intcc{\frac12}{1}$, e $ u_1 = \frac12$ coloca o
toda a sequência lá. Ponto fixo: $\ell = \frac{1}{1+\ell}$
com $\ell > 0$ dá $\ell^2 + \ell - 1 = 0$, ou seja
\[
\ell = \frac{\sqrt5 - 1}{2} = 0.6180\dots
\]
(o inverso da proporção áurea). Contração: para $ x, y \in
\intcc{\frac12}{1}$,
\[
\abs{f(x) - f(y)} = \frac{\abs{x - y}}{(1+x)(1+y)}
\leq \frac{\abs{x-y}}{(3/2)^2} = \frac49\,\abs{x - y} ,
\]
então $\abs{u_n - \ell} \leq \bigl(\frac49\bigr)^{n-1}\abs{u_1 -
\ell} \to 0$: convergência, com velocidade geométrica, não
monotonicidade necessária. A visão final: métodos monótonos e
métodos de contração dividem o mundo recorrente entre eles ---
aumentando $ f $ dá órbitas monótonas, diminuindo $ f $ dá
órbitas alternadas domesticadas por uma constante de Lipschitz $< 1$ (o
teoria sistemática é \cref{exo:b1:derivative:11}).
\end{example}

\begin{remark}[Perspectivas dentro deste volume]
As sequências são o instrumento de medição do resto do volume
suporta todos os objetos. Em \cref{ch:b1:topology} eles
\emph{caracterizar} fechamento e compacidade; em
\cref{ch:b1:continuity} transportam limites de funções; em
\cref{ch:b1:integration} Somas de Riemann são sequências convergentes
para a integral; \cref{ch:b1:series} \emph{é} a teoria de um
classe especial de sequências, as somas parciais. Até a álgebra
capítulos os consomem: as iterações de uma matriz em
\cref{ch:b1:matrices} formam sequências cujo comportamento
(convergência de $ A^n $) é uma questão de álgebra linear com este
vocabulário do capítulo. Os dois teoremas para levar a todos os lugares:
limite monótono (existência da ordem) e Bolzano--Weierstrass
(existência a partir da limitação) --- entre eles, quase todos
limite neste livro nasce.
\end{remark}

\begin{remark}[Sequências complexas]
Uma sequência $(z_n)$ de números complexos converge para $\ell $ quando
$\abs{z_n - \ell} \to 0$; equivalentemente, quando $\Re(z_n) \to \Re(\ell)$
e $\Im(z_n) \to \Im(\ell)$ (comparar $\abs{z}$ com $\abs{\Re z} +
\abs{\Im z}$). Os teoremas que não envolvem ordem --- operações,
Bolzano--Weierstrass (extrato duas vezes), critério de Cauchy --- transferência
literalmente.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b1:seq:1}
Diretamente de \cref{def:b1:seq:limit}, prove que
$\dfrac{2n+1}{n+3} \to 2$, e isso $(u_n) = ((-1)^n)$ diverge.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:seq:2}
Calcule os limites:
\[
\frac{n^2 - 3n + 1}{2n^2 + 5},
\qquad
\sqrt{n+1} - \sqrt n,
\qquad
\frac{2^n + n^3}{3^n - n^2},
\qquad
\sqrt[n]{5}\ \Bigl(= 5^{1/n}\Bigr).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:seq:3}
Prove a comparação padrão: se $\abs{q} < 1$ então $ q^n \to 0$
\emph{(escrever $\frac{1}{\abs q} = 1 + h $, $ h > 0$ e use o
Desigualdade de Bernoulli $(1+h)^n \geq 1 + nh $, a ser provado por
indução)}. Para que servem os comportamentos $ q = 1$, $ q = -1$, $\abs q >
1$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:seq:4}
Deixar $ u_{n+1} = \frac{u_n + 3}{2}$, $ u_0 = 0$. Encontre o ponto fixo
$\ell $, prove que $ v_n = u_n - \ell $ é geométrica e dá uma
fórmula explícita e o limite de $(u_n)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:seq:5}
(Série harmônica) Deixe $ H_n = \sum_{k=1}^{n} \frac 1k $. Prove isso
$ H_{2n} - H_n \geq \frac12$ para todos $ n \geq 1$, e concluir que
$(H_n)$ é \emph{não} um \omterm{def:b1:seq:cauchy}{Sequência de Cauchy}, portanto diverge (para
$+\infty $, sendo crescente).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:seq:6}
Suponha $(u_{2n})$, $(u_{2n+1})$ e $(u_{3n})$ todos convergem. Provar
isso $(u_n)$ converge. \emph{(Encontrar comum \omterm{def:b1:seq:subsequence}{subsequências} para igualar o
limites.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:seq:7}
Estude a sequência $ u_0 = 0$, $ u_{n+1} = \sqrt{2 + u_n}$:
estabilidade, monotonicidade, limite. Então prove o erro vinculado
$\abs{u_n - 2} \leq \dfrac{2}{3^{\,n}}$ \emph{(mostrar $2 - u_{n+1} =
\dfrac{2 - u_n}{2 + \sqrt{2 + u_n}}$ e ligue o denominador abaixo
por $3$)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:seq:8}
Deixar $(u_n)$ ser limitado, tal que todo convergente \omterm{def:b1:seq:subsequence}{subsequência} de
$(u_n)$ tem o \emph{mesmo} limite $\ell $. Provar $ u_n \to \ell $.
\emph{(Contradição mais Bolzano-Weierstrass.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:seq:9}
Com a notação de \cref{ex:b1:seq:e}, suponha $\eu = \frac pq $
com $ p, q \in \N^*$. Usando $ a_q < \eu < b_q = a_q + \frac{1}{q\,
q!}$, multiplique por $ q!$ e derivar uma contradição entre dois
inteiros. Concluir: $\eu $ é irracional.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:seq:10}
(Cesàro significa) Para uma sequência $(u_n)_{n \geq 1}$, \omterm{def:b1:logic:sets}{definir} $ c_n =
\frac{u_1 + \dots + u_n}{n}$.
\begin{enumerate}
  \item Prove isso $ u_n \to \ell $ implica $ c_n \to \ell $ \emph{(cortar
        a soma em um limite $ N $; amarrou a cabeça por um fixo
        quantidade acima $ n $, a cauda por $\varepsilon $)}.
  \item Mostre por exemplo que a recíproca falha.
  \item Deduza que se $ u_{n+1} - u_n \to \ell $, então $\frac{u_n}{n}
        \to \ell $.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:seq:11}
Deixar $(u_n)$ satisfazer $0 \leq u_{m+n} \leq u_m + u_n $ para todos $ m, n $
(subaditividade). Prove isso $\bigl(\frac{u_n}{n}\bigr)$ converge para
$\inf_{n \geq 1} \frac{u_n}{n}$. \emph{(Para fixo $ m $, escrever $ n = qm
+ r $ e vinculado $\frac{u_n}{n}$ usando $ u_n \leq q\,u_m + u_r $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:seq:12}
Usando a densidade do \omterm{def:b1:structures:subgroup}{subgrupo} $\Z + 2\pi\Z $ de $(\R, +)$
(\cref{exo:b1:reals:9}), prove que a sequência $(\sin n)_{n \in
\N}$ é denso em $\intcc{-1}{1}$ --- em particular diverge.
\end{exercise}

\section{Problema: Cesàro, Stolz e a lenta queda do seno}

\begin{problem}[{Problema de fim de semana --- o Cesàro--Stolz
e os assintóticos $ u_n \sim \sqrt{3/n}$ pára $ u_{n+1} = \sin
u_n $}]
\label{pb:b1:seq:1}
O Cesò teorema de aro-Stolz é a regra discreta de l'Hospital: para
encontrar o limite de um quociente $ a_n/b_n $, basta encontrar o
limite do quociente de \emph{incrementos} $(a_{n+1} -
a_n)/(b_{n+1} - b_n)$. Este problema prova o teorema, colhe
limites clássicos com ele, e então aponta para um alvo famoso:
a sequência $ u_{n+1} = \sin u_n $, que se arrasta para $0$ no
velocidade exatamente computável $ u_n \sim \sqrt{3/n}$.\index{Cesaro--Stolz
theorem@Cesàro--Stolz theorem} Dois fatos do ensino médio
volume são concedidos aqui e re-provados honestamente mais tarde neste
volume: a desigualdade da linha tangente
\[
\tag{G1} \eu^{u} \geq 1 + u \quad (u \in \R),
\]
re-provado pela convexidade em \cref{ch:b1:derivative}, e o seno
colchetes
\[
\tag{G2} x - \frac{x^3}{6} \;\leq\; \sin x \;\leq\; x -
\frac{x^3}{6} + \frac{x^5}{120} \quad (0 \leq x \leq 1), \qquad
\abs{\sin x} \leq \abs{x} \quad (x \in \R),
\]
re-provado pela fórmula de Taylor em \cref{ch:b1:taylor}.

\medskip
\noindent\textbf{Part I --- Sums without closed formulas.}
\begin{enumerate}
  \item Usando $1 + 2 + \dots + n = \frac{n(n+1)}{2}$ e $1^2 +
        \dots + n^2 = \frac{n(n+1)(2n+1)}{6}$, calcular
        $\lim \frac{1 + 2 + \dots + n}{n^2}$ e $\lim \frac{1^2 +
        \dots + n^2}{n^3}$.
  \item Deixar $ T_n = \sum_{k=1}^n \sqrt k $, para o qual não foi fechado
        fórmula existe. Prove o colchetes
        \[
        \frac{1}{2\sqrt 2}\,n^{3/2} \;\leq\; T_n \;\leq\;
        n^{3/2}
        \]
        \emph{(guarde apenas os termos $ k > \frac n2$ para o inferior
        vinculado)}. Então $ T_n $ tem a ordem $ n^{3/2}$ --- mas qual
        constante? Segure a pergunta até a pergunta 8.
  \item (Lema telescópico) Deixe $(b_n)$ ser estritamente crescente
        e suponha que para todos $ k \geq N $,
        \[
        m \;\leq\; \frac{a_{k+1} - a_k}{b_{k+1} - b_k} \;\leq\;
        M .
        \]
        Prove isso $ m \leq \dfrac{a_n - a_N}{b_n - b_N} \leq M $
        para cada $ n > N $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The \omterm{pb:b1:seq:1}{Cesàro--Stolz theorem}.} Deixar
$(b_n)$ ser estritamente crescente com $ b_n \to +\infty $ e
suponha $\dfrac{a_{n+1} - a_n}{b_{n+1} - b_n} \to \ell \in \R $.
\begin{enumerate}[resume]
  \item Consertar $\varepsilon > 0$. Mostre que existe $ N $ tal que
        $\ell - \varepsilon \leq \dfrac{a_n - a_N}{b_n - b_N}
        \leq \ell + \varepsilon $ para todos $ n > N $.
  \item Estabelecer, para $ n > N $, a identidade
        \[
        \frac{a_n}{b_n} - \ell
        = \frac{a_N - \ell\,b_N}{b_n}
        + \Bigl(1 - \frac{b_N}{b_n}\Bigr)
        \Bigl(\frac{a_n - a_N}{b_n - b_N} - \ell\Bigr),
        \]
        e conclua o teorema: $\dfrac{a_n}{b_n} \to \ell $.
  \item Prove o $+\infty $ variante: se $\dfrac{a_{n+1} -
        a_n}{b_{n+1} - b_n} \to +\infty $ (mesmas hipóteses em
        $(b_n)$), então $\dfrac{a_n}{b_n} \to +\infty $.
  \item Pegar $ b_n = n $: recuperar o Cesàro significa teorema de
        \cref{exo:b1:seq:10}. Então mostre a recíproca de
        Cesàro--Stolz falha: para $ a_n = (-1)^n $, $ b_n = n $, o
        quociente $ a_n/b_n $ converge enquanto o quociente de
        incrementos não. Stolz é uma rua de mão única.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- First dividends.}
\begin{enumerate}[resume]
  \item Provar $(1+h)^{3/2} - 1 = \dfrac{3h + 3h^2 +
        h^3}{(1+h)^{3/2} + 1}$ por conjugação, deduzir
        $ n\bigl((1 + \tfrac1n)^{3/2} - 1\bigr) \to \tfrac32$,
        e concluo com Cesàro--Stolz:
        \[
        T_n = \sum_{k=1}^{n} \sqrt k \;\sim\; \tfrac23\,
        n^{3/2} ,
        \]
        resolvendo o suspense da questão 2.
  \item Somente a partir de (G1), derive o escalonamento logarítmico
        \[
        \frac{t}{1 + t} \;\leq\; \ln(1 + t) \;\leq\; t
        \qquad (t > -1)
        \]
        \emph{(aplicar (G1) em $ u = \ln(1+t)$ e em $ u =
        -t/(1+t)$)}.
  \item Mostre isso $ b_n = \ln n $ está aumentando estritamente com
        $\ln n \to +\infty $, e prove com Cesàro--Stolz e
        questão 9 que
        \[
        H_n = \sum_{k=1}^{n} \frac 1k \;\sim\; \ln n .
        \]
        (A estrutura mais fina $ H_n = \ln n + \gamma + o(1)$ é o
        problema de fim de semana de \cref{ch:b1:series}.)
  \item (Das proporções às raízes) Deixe $ u_n > 0 $ com
        $\frac{u_{n+1}}{u_n} \to L > 0$. Usando a questão 9, mostre
        $\ln\frac{u_{n+1}}{u_n} \to \ln L $; aplicar Cesàro para
        concluir $\frac{\ln u_n}{n} \to \ln L $, então, com (G1),
        isso $ u_n^{1/n} \to L $. Aplicação: computar
        $\lim\,\binom{2n}{n}^{1/n}$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The slow fall of the sine.} Deixar
$ u_0 \in \R $ e $ u_{n+1} = \sin u_n $.
\begin{enumerate}[resume]
  \item De (G2), mostre $0 < \sin x < x $ para $0 < x \leq 1$.
        Deduzir: $ u_1 \in \intcc{-1}{1}$; se $ u_1 = 0$ o
        sequência é zero da classificação $1$; e se $ u_1 > 0$ (o
        caso $ u_1 < 0$ sendo simétrico, $\sin $ estranho), então
        $(u_n)_{n \geq 1}$ é estritamente decrescente, positivo e
        converge para $0$ \emph{(identifique o limite através $\ell =
        \sin \ell $, usando $\abs{\sin a - \sin b} \leq \abs{a -
        b}$, em si uma consequência de (G2) e do produto para soma
        fórmula)}.
  \item Suponha de agora em diante $ u_1 \in \intoc{0}{1}$. Mostrar por
        apertando, usando (G2):
        \[
        \frac{\sin u_n}{u_n} \to 1
        \qquad\text{and}\qquad
        \frac{u_n - \sin u_n}{u_n^{3}} \to \frac16 .
        \]
  \item Prove a fatoração
        \[
        w_n := \frac{1}{u_{n+1}^{2}} - \frac{1}{u_n^{2}}
        = \frac{u_n - \sin u_n}{u_n^{3}} \cdot
        \frac{u_n + \sin u_n}{u_n} \cdot
        \Bigl(\frac{u_n}{\sin u_n}\Bigr)^{2},
        \]
        e deduzir $ w_n \to \frac13$.
  \item Conclua com \cref{exo:b1:seq:10} (incrementa a versão)
        isso $\frac{1}{n\,u_n^{2}} \to \frac13$, então, por um
        argumento de conjugação para a raiz quadrada, o título:
        \[
        \sqrt n\;u_n \longrightarrow \sqrt 3 ,
        \qquad\text{i.e.}\qquad
        u_n \sim \sqrt{\frac 3n} .
        \]
  \item Quantifique a lentidão: mostre que eventualmente $\sqrt{2/n}
        \leq u_n \leq 2/\sqrt n $, para que alcance $ u_n \leq
        10^{-2}$ requer mais do que $20\,000$ iterações (cerca de
        $30\,000$, pelos assintóticos). Compare com Garça
        método (\cref{ex:b1:seq:heron}) e explique o
        razão estrutural: no ponto fixo $0$, a inclinação de
        $\sin $ é $1$ (um \emph{neutro} ponto fixo), enquanto
        iterações de redução de erro pela metade precisam de uma inclinação de \omterm{def:b1:complex:field}{módulo} $< 1$.
  \item Mostre isso para \emph{todo} ponto de partida $ u_0 \in \R $,
        qualquer $ u_n = 0$ da classificação $1$ ligado, ou $\abs{u_n} \sim
        \sqrt{3/n}$ --- a queda é universal, apenas o sinal
        lembra $ u_0$.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The general principle.} O seno é
uma instância de uma máquina.
\begin{enumerate}[resume]
  \item Deixar $ u_n > 0$, $ u_n \to 0$, e $\dfrac{u_n -
        u_{n+1}}{u_n^{2}} \to a > 0$. Prove sucessivamente:
        $\frac{u_{n+1}}{u_n} \to 1$; então $\frac{1}{u_{n+1}} -
        \frac{1}{u_n} \to a $; então $ n\,u_n \to \frac1a $.
  \item (Modelo exato) Para $ u_{n+1} = \dfrac{u_n}{1 + u_n}$, $ u_0
        > 0$: mostrar $\frac{1}{u_n}$ é aritmética, resolva exatamente,
        e compare a conclusão da questão 18 com a exata
        fórmula.
  \item Para $ u_{n+1} = u_n \eu^{-u_n}$, $ u_0 > 0$: mostrar $ u_n \to
        0$, use (G1) para apertar $\frac{1 - \eu^{-t}}{t}$ entre
        $\frac{1}{1+t}$ e $1$ para $ t > 0$, e concluir $ u_n
        \sim \frac 1n $.
  \item (Contato cúbico, telescópio quadrado) Deixe $ u_n > 0$, $ u_n
        \to 0$, $\dfrac{u_n - u_{n+1}}{u_n^{3}} \to a > 0$.
        Adapte a fatoração da questão 14 para mostrar
        $\frac{1}{u_{n+1}^2} - \frac{1}{u_n^2} \to 2a $ e
        concluir $ n\,u_n^{2} \to \frac{1}{2a}$. Verifique isso $ a =
        \frac16$ recupera a Parte IV.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Limits of the method, and morals.}
\begin{enumerate}[resume]
  \item Mostre que a hipótese $ b_n \to +\infty $ não pode ser
        caiu: para $ a_n = 2 - 2^{-n}$ e $ b_n = 1 - 2^{-n}$,
        o quociente de incremento tende a $1$ enquanto $\frac{a_n}
        {b_n} \to 2$. Aponte para a linha exata da pergunta 5
        prova que quebra.
  \item (Stolz duas vezes) Prove $\sum_{k=1}^{n} H_k \sim n \ln n $
        \emph{(uma aplicação de Cesàro--Stolz, então questione
        10; vinculado $(n+1)\ln(n+1) - n\ln n $ usando a questão 9)}.
  \item (Meios geométricos) Se $ u_n > 0$ e $ u_n \to \ell > 0$,
        mostrar $(u_1 u_2 \cdots u_n)^{1/n} \to \ell $; se $ u_n \to
        +\infty $, mostrar $(u_1 \cdots u_n)^{1/n} \to +\infty $.
        Deduzir $(n!)^{1/n} \to +\infty $.
  \item Síntese, uma frase cada: (i) onde exatamente
        completude entra neste problema; (ii) em que sentido é
        Cesàro--Stolz uma regra discreta de l'Hospital (seu
        gêmeo diferencial baseia-se no teorema do valor médio de
        \cref{ch:b1:derivative}); (iii) declarar a heurística
        vinculando a ordem de contato de $ f $ em ponto neutro fixo
        apontar para o expoente de decaimento de $ u_{n+1} = f(u_n)$;
        (iv) traçar a constante $3$ de $\sqrt{3/n}$ de volta
        o gasoduto $\frac16 \to \frac13 \to 3$.
\end{enumerate}
\end{problem}
