\chapter{Funções padrão}\label{ch:b1:functions}

A análise é tão útil quanto o estoque de funções que se domina. Para
a coleção herdada do ensino médio --- poderes, exponencial,
logaritmo, funções trigonométricas --- este capítulo adiciona seu inverso
funções ($\arcsin $, $\arccos $, $\arctan $) e a família hiperbólica.
Os derivativos são usados livremente no ensino médio; o
a teoria por trás das funções inversas é concluída em
\cref{ch:b1:continuity,ch:b1:derivative}.

\section{Exponencial, logaritmo, potências}

\begin{proposition}[Recapitulação e caracterização]\label{prop:b1:functions:expln}
$\exp \colon \R \to \intoo{0}{+\infty}$ e $\ln \colon
\intoo{0}{+\infty} \to \R $ são bijeções recíprocas, estritamente
aumentando, com
\[
\exp(x + y) = \exp x \exp y, \qquad \ln(xy) = \ln x + \ln y,
\qquad \exp' = \exp, \qquad \ln'(x) = \frac 1x .
\]
Além disso $\exp $ é o \emph{apenas} função diferenciável $ f \colon
\R \to \R $ com $ f' = f $ e $ f(0) = 1$.
\end{proposition}

\begin{proof}
A recapitulação é material do Ensino Médio. Para exclusividade, deixe $ f' = f $,
$ f(0) = 1$ e \omterm{def:b1:logic:sets}{definir} $ g(x) = f(x)\,\eu^{-x}$. Então $ g' = f'\eu^{-x} -
f\eu^{-x} = 0$, então $ g $ é constante igual a $ g(0) = 1$: $ f = \exp $.
\end{proof}

\begin{definition}[Poderes gerais]\label{def:b1:functions:powers}
Para $ x > 0$ e $\alpha \in \R $:
$\;x^\alpha = \eu^{\alpha \ln x}$\index{função de potência}. Para $ a > 0$,
$ a \neq 1$, o \emph{logaritmo para base $ a $} é $\log_a x =
\frac{\ln x}{\ln a}$, o inverso de $ x \mapsto a^x $.
\end{definition}

\begin{example}[Resolvendo equações exponenciais]\label{ex:b1:functions:expequations}
Resolver $2^x = 5^{\,x-1}$ em $\R $. Ambos os lados são positivos, então tome
logaritmos --- uma etapa reversível:
\[
x\ln2 = (x - 1)\ln5
\iff x(\ln2 - \ln5) = -\ln5
\iff x = \frac{\ln5}{\ln5 - \ln2} = \frac{\ln 5}{\ln\frac52}
\approx 1.756 .
\]
A seguir, resolva $ x^{\sqrt2} = 3$ para $ x > 0$: elevar ao poder
$\frac1{\sqrt2}$ (ou seja, aplique a bijeção recíproca):
$ x = 3^{1/\sqrt2} = \eu^{(\ln 3)/\sqrt2} \approx 2.175$. O
insight: cada equação que mistura poderes se desenrola $\ln $ e
$\exp $, porque a definição $ x^\alpha = \eu^{\alpha\ln x}$
reduz todas as manipulações de potência à aritmética de expoentes ---
mas apenas no domínio $ x > 0$ onde reside essa definição.
\end{example}

\begin{example}[Tempos de duplicação]\label{ex:b1:functions:doubling}
Uma quantidade cresce em $3\%$ por passo: depois $ n $ passos é isso
multiplicado por $(1.03)^n $. Quando dobra? Resolver $(1.03)^n
\geq 2$:
\[
n\ln(1.03) \geq \ln2
\iff n \geq \frac{\ln 2}{\ln 1.03}
= \frac{0.6931}{0.02956} \approx 23.4 ,
\]
então a primeira duplicação ocorre na etapa $24$. (Os financiadores
``regra de $72$'', que estima o tempo de duplicação como
$72$ dividido pela taxa em porcentagem, este cálculo é com
a aproximação $\ln(1 + x) \approx x $, quantificado em
\cref{ch:b1:taylor}.) Exponencial
os processos são melhor fundamentados através de seus logaritmos: em
nessa escala, o crescimento é linear e as questões tornam-se divisões.
\end{example}

\begin{example}[Quanto tempo é $2^{2026}$?]\label{ex:b1:functions:digits}
O número de dígitos decimais de um inteiro $ N \geq 1$ é
$\floor{\log_{10} N} + 1$ (de fato $ N $ tem $ d $ dígitos exatamente quando
$10^{d-1} \leq N < 10^d $, ou seja\ $ d - 1 \leq \log_{10}N < d $). Para
$ N = 2^{2026}$:
\[
\log_{10} 2^{2026} = 2026\,\log_{10}2
= 2026 \times 0.301030 = 609.887 ,
\]
então $2^{2026}$ tem $610$ dígitos. A parte fracionária carrega um
bônus: $10^{0.887} \approx 7.7$, então o número \emph{começa} com
$7$. Uma multiplicação respondeu a uma pergunta sobre um número que ninguém
algum dia escreverá --- logaritmos compactam tamanho multiplicativo
em tamanho aditivo, que é todo o seu ponto histórico
(\cref{ex:b1:structures:expmorphism} torna essa frase precisa).
\end{example}

\begin{proposition}[Regras de energia e comparação de crescimento]\label{prop:b1:functions:powerrules}
Para $ x, y > 0$ e $\alpha, \beta \in \R $:
\[
x^{\alpha+\beta} = x^\alpha x^\beta, \quad
(x^\alpha)^\beta = x^{\alpha\beta}, \quad
(xy)^\alpha = x^\alpha y^\alpha, \quad
(x^\alpha)' = \alpha\, x^{\alpha - 1}.
\]
Escala de crescimento\index{comparação de crescimento} como $ x \to +\infty $, para cada
$\alpha > 0$ e $\beta > 0$:
\[
\frac{(\ln x)^\beta}{x^\alpha} \longrightarrow 0,
\qquad
\frac{x^\alpha}{\eu^{\beta x}} \longrightarrow 0 .
\]
\end{proposition}

\begin{proof}
As identidades transcrevem as de $\exp $ e $\ln $ através do
definição. Em detalhes para o segundo (o menos óbvio): $ x^\alpha
> 0$ e $\ln(x^\alpha) = \alpha\ln x $ (aplicar $\ln $ para o
definição), então
\[
(x^\alpha)^\beta = \eu^{\beta\ln(x^\alpha)}
= \eu^{\beta\alpha\ln x} = x^{\alpha\beta} ;
\]
o primeiro e o terceiro são os mesmos desenrolamentos de duas linhas via
$\exp(u + v) = \exp u\exp v $ e $\ln(xy) = \ln x + \ln y $. O
derivada é a regra da cadeia: $(\eu^{\alpha\ln x})' =
\frac{\alpha}{x} \eu^{\alpha\ln x}$.

Comparações: de $\frac{\ln t}{t} \to 0$ (comprovado no Alto
Volume escolar), substituto $ t = x^{\alpha/\beta}$:
$\frac{\ln x}{x^{\alpha/\beta}} \to 0$, então eleve à potência
$\beta $. Para o segundo, substitua $ x = \ln u $ no primeiro.
\end{proof}

\begin{example}[Dois limites que todo leitor deve possuir]\label{ex:b1:functions:xx}
O que são $\lim_{x \to 0^+} x^x $ e $\lim_{x \to +\infty}
x^{1/x}$? Ambas as potências são definidas através da exponencial, então
resolva o expoente primeiro:
\[
x^x = \eu^{x\ln x}, \qquad
x\ln x = -\frac{\ln(1/x)}{1/x} \longrightarrow 0
\quad (x \to 0^+),
\qquad\text{so } x^x \longrightarrow \eu^0 = 1 ;
\]
e $ x^{1/x} = \eu^{(\ln x)/x} \to \eu^0 = 1$ como $ x \to +\infty $,
diretamente pelo \omterm{prop:b1:functions:powerrules}{comparação de crescimento}. O insight: um indeterminado
poder (de forma $0^0$ ou $\infty^0$) é \emph{sempre} tratado por
reescrevendo $ u^v = \eu^{v\ln u}$ e analisando o produto $ v \ln u $
--- nunca adivinhando a partir da base e do expoente separadamente. O
função $ x \mapsto \frac{\ln x}{x}$ que decidiu ambos os limites é
estudado exaustivamente no problema de fim de semana deste capítulo.
\end{example}

\begin{remark}[Armadilhas comuns com potências e logaritmos]\label{rem:b1:functions:pitfalls}
\begin{enumerate}
  \item \emph{Domínios.} $ x^\alpha $ para irracional $\alpha $ precisa
        $ x > 0$; $(-8)^{1/3}$ é melhor evitar em favor de “o
        raiz cúbica real de $-8$'', porque a regra
        $(x^\alpha)^\beta = x^{\alpha\beta}$ falha silenciosamente
        negativos: $\bigl((-8)^2\bigr)^{1/6} = 2 \neq -2$.
  \item \emph{$\sqrt{x^2} = \abs x $}, não $ x $: esquecendo o
        o valor absoluto é a fonte clássica de perda negativa
        soluções.
  \item \emph{$\ln(xy) = \ln x + \ln y $ requer $ x, y > 0$.} Para
        argumentos negativos, $\ln(xy)$ pode ser definido enquanto o
        lado direito não é.
  \item \emph{Qual função cresce?} Em $ x^\alpha $ o
        \emph{base} varia; em $ a^x $ o \emph{expoente} faz.
        Seus crescimentos são totalmente diferentes
        (\cref{prop:b1:functions:powerrules}) e híbrido
        expressões como $ x^{\ln x}$ ou $ x^{1/x}$ deve ser
        reescrito como $\eu^{(\cdot)}$ antes de qualquer raciocínio ---
        como em \cref{ex:b1:functions:xx}.
\end{enumerate}
\end{remark}

\section{Funções trigonométricas inversas}

\begin{definition}[$\arcsin $, $\arccos $, $\arctan $]\label{def:b1:functions:arc}
As restrições
\[
\sin \colon \intcc{-\tfrac\pi2}{\tfrac\pi2} \to \intcc{-1}{1},
\qquad
\cos \colon \intcc{0}{\pi} \to \intcc{-1}{1},
\qquad
\tan \colon \intoo{-\tfrac\pi2}{\tfrac\pi2} \to \R
\]
são bijeções estritamente monotônicas. Seu inverso \omterm{def:b1:logic:map}{mapas} estão escritos
$\arcsin $\index{arco seno}, $\arccos $\index{arco cosseno} e
$\arctan $\index{arco tangente}. Assim, por exemplo, $ y = \arcsin x $
($ x \in \intcc{-1}{1}$) é \emph{o} ângulo em
$\intcc{-\frac\pi2}{\frac\pi2}$ cujo seno é $ x $.
\end{definition}

\begin{proposition}[Derivados]\label{prop:b1:functions:arcderiv}
No interior de seus domínios:
\[
\arcsin' x = \frac{1}{\sqrt{1 - x^2}},
\qquad
\arccos' x = \frac{-1}{\sqrt{1 - x^2}},
\qquad
\arctan' x = \frac{1}{1 + x^2}.
\]
\end{proposition}

\begin{proof}
Antecipando a regra da função inversa provada em
\cref{ch:b1:derivative}: se $ f $ é uma bijeção diferenciável e
$ f' \neq 0$, então $(f^{-1})'(x) = \frac{1}{f'(f^{-1}(x))}$. Para
$\arcsin $: com $ y = \arcsin x $,
\[
\arcsin' x = \frac{1}{\cos y}
= \frac{1}{\sqrt{1 - \sin^2 y}} = \frac{1}{\sqrt{1 - x^2}},
\]
onde $\cos y = +\sqrt{1 - \sin^2 y}$ porque $ y \in
\intoo{-\frac\pi2}{\frac\pi2}$ forças $\cos y > 0$. De forma similar
$\arccos' x = \frac{1}{-\sin y}$ com $\sin y > 0$ sobre
$\intoo{0}{\pi}$, e $\arctan' x = \frac{1}{1 + \tan^2 y} =
\frac{1}{1 + x^2}$ usando $\tan' = 1 + \tan^2$.
\end{proof}

\begin{example}[Reconhecendo uma constante oculta]\label{ex:b1:functions:hiddenconst}
Estudar $ g(x) = \arctan\dfrac{1 - x}{1 + x}$ sobre $\intoo{-1}
{+\infty}$. Regra da cadeia e um breve cálculo:
\[
g'(x) = \frac{1}{1 + \bigl(\frac{1-x}{1+x}\bigr)^2}\cdot
\frac{-(1+x) - (1-x)}{(1+x)^2}
= \frac{-2}{(1+x)^2 + (1-x)^2}
= \frac{-1}{1 + x^2} ,
\]
desde $(1+x)^2 + (1-x)^2 = 2 + 2x^2$. Então $ g' = -\arctan'$: o
função $ g + \arctan $ tem derivada zero no \emph{intervalo}
$\intoo{-1}{+\infty}$, portanto, é constante lá; seu valor em $ x =
0$ é $\arctan 1 + \arctan 0 = \frac\pi4$. Conclusão:
\[
\arctan\frac{1 - x}{1 + x} = \frac\pi4 - \arctan x
\qquad (x > -1) .
\]
Sobre $\intoo{-\infty}{-1}$ o mesmo cálculo da derivada é válido, mas
a constante é diferente ($-\frac{3\pi}4$: avalie o limite como
$ x \to -\infty $). O insight: ``derivada zero implica
constante'' é um intervalo por intervalo \omterm{def:b1:logic:statement}{declaração} --- exatamente o
sutileza explorada em \cref{exo:b1:functions:6} e
\cref{exo:b1:functions:10}.
\end{example}

\begin{proposition}[Identidades padrão]\label{prop:b1:functions:arcidentities}
\begin{enumerate}
  \item Para $ x \in \intcc{-1}{1}$: $\arcsin x + \arccos x =
        \dfrac{\pi}{2}$.
  \item Para $ x > 0$: $\arctan x + \arctan\dfrac 1x = \dfrac{\pi}{2}$
        (e $-\frac\pi2$ para $ x < 0$).
  \item $\sin(\arccos x) = \cos(\arcsin x) = \sqrt{1 - x^2}$;
        $\;\tan(\arcsin x) = \dfrac{x}{\sqrt{1 - x^2}}$ para
        $\abs x < 1$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) A derivada de $ x \mapsto \arcsin x + \arccos x $ está zero ligado
$\intoo{-1}{1}$ (\cref{prop:b1:functions:arcderiv}), então a função
é constante lá, igual ao seu valor $\frac\pi2$ no $0$; o ponto final
valores $\pm 1$ são verificados diretamente ($\frac\pi2 + 0$ e $-\frac\pi2 +
\pi $).

(2) Mesmo método em $\intoo{0}{+\infty}$: a derivada é
$\frac{1}{1+x^2} + \frac{-1/x^2}{1 + 1/x^2} = \frac{1}{1+x^2} -
\frac{1}{x^2 + 1} = 0$, e em $ x = 1$ a soma é $2\arctan 1 =
\frac\pi2$. Para $ x < 0$, use a estranheza de $\arctan $.

(3) Com $ y = \arccos x \in \intcc{0}{\pi}$: $\sin y \geq 0$, então
$\sin y = \sqrt{1 - \cos^2 y} = \sqrt{1 - x^2}$; da mesma forma para
$\cos(\arcsin x)$, e a fórmula da tangente é o quociente.
\end{proof}

\begin{remark}
$\arcsin(\sin\theta) = \theta $ detém \emph{apenas} para $\theta \in
\intcc{-\frac\pi2}{\frac\pi2}$: por exemplo
$\arcsin(\sin\pi) = 0 \neq \pi $. A composição no outro
direção, $\sin(\arcsin x) = x $, é válido em todos
$\intcc{-1}{1}$.
\end{remark}

\begin{example}[Voltando ao intervalo principal]
\label{ex:b1:functions:wrap}
Calcular
\[
\arctan\Bigl(\tan\frac{3\pi}4\Bigr)
\qquad\text{and}\qquad
\arctan\Bigl(\tan\frac{17\pi}5\Bigr).
\]
A receita: substitua o
ângulo pelo ângulo único de $\intoo{-\frac\pi2}{\frac\pi2}$ com
a mesma tangente, ou seja, subtraia o múltiplo certo de $\pi $
(o período de $\tan $). Primeiro: $\frac{3\pi}4 - \pi =
-\frac\pi4$, então a resposta é $-\frac\pi4$. Segundo: $\frac{17\pi}
5 - 3\pi = \frac{2\pi}5 \in \intoo{-\frac\pi2}{\frac\pi2}$, então
a resposta é $\frac{2\pi}5$. O cálculo é euclidiano
\omterm{thm:b1:logic:partition}{divisão} do ângulo por $\pi $ disfarçado --- e o análogo
receitas para $\arcsin $ (refletir em
$\intcc{-\frac\pi2}{\frac\pi2}$, período $2\pi $) e $\arccos $
(refletir em $\intcc0\pi $) dirigir \cref{exo:b1:functions:1} e
a resposta por partes de \cref{exo:b1:functions:10}.
\end{example}

\begin{example}[Adicionando arcotangentes com segurança]\label{ex:b1:functions:arctansum}
Vamos provar
\[
\arctan\frac12 + \arctan\frac15 + \arctan\frac18 = \frac\pi4 .
\]
Dois ingredientes: a fórmula de adição tangente e --- a etapa
iniciantes esquecem --- a \emph{localização} da soma. Primeiro,
$\tan(u + v) = \frac{\tan u + \tan v}{1 - \tan u\tan v}$ com
$ u = \arctan\frac12$, $ v = \arctan\frac15$ dá
\[
\tan(u + v) = \frac{\frac12 + \frac15}{1 - \frac1{10}}
= \frac{7/10}{9/10} = \frac79,
\qquad\text{then}\qquad
\tan\Bigl(u + v + \arctan\frac18\Bigr)
= \frac{\frac79 + \frac18}{1 - \frac7{72}}
= \frac{65/72}{65/72} = 1 .
\]
Em segundo lugar, cada um dos três ângulos está em $\intoo0{\frac\pi4}$
(seus argumentos são inferiores a $1$), então a soma está em
$\intoo0{\frac{3\pi}4}$; o único ângulo ali com tangente $1$ é
$\frac\pi4$. Sem a localização a conclusão seria
``$\frac\pi4$ até um múltiplo de $\pi $'' --- meia prova. O
a mesma disciplina de duas etapas é executada \cref{exo:b1:functions:8} e
\cref{exo:b1:functions:12}.
\end{example}

\begin{example}[Uma equação arco tangente]\label{ex:b1:functions:arctaneq}
Resolver $\arctan x + \arctan 2x = \dfrac\pi4$. Primeira localização: o
lado esquerdo tem o sinal de $ x $ (ambos os termos têm), então qualquer solução tem
$ x > 0$, e então a soma fica em $\intoo0\pi $. Tome tangentes
(\omterm{def:b1:logic:inj}{injetivo} em nenhum intervalo de comprimento $\pi $, mas combinado com o
localização isso será suficiente): a fórmula de adição fornece
\[
\tan\bigl(\arctan x + \arctan 2x\bigr)
= \frac{3x}{1 - 2x^2} = 1
\iff 2x^2 + 3x - 1 = 0
\iff x = \frac{-3 \pm \sqrt{17}}4 .
\]
A raiz negativa é descartada pela localização; para o
candidato positivo $ x_0 = \frac{\sqrt{17} - 3}4 \approx 0.28$,
a soma está em $\intoo0\pi $ com tangente $1$, e o único
ângulo é $\frac\pi4$ (em $\intoo{\frac\pi2}\pi $ a tangente é
negativo): $ x_0$ é a solução única. Observe o formato do
argumento: tomar tangentes pode criar soluções, nunca perdê-las,
então resolvemos a equação polinomial \emph{então} filtra por
localização --- a mesma disciplina de verificação antecipada da quadratura
uma equação.
\end{example}

\section{Funções hiperbólicas}

\begin{definition}[$\cosh $, $\sinh $, $\tanh $]\label{def:b1:functions:hyperbolic}
Para $ x \in \R $:
\[
\cosh x = \frac{\eu^x + \eu^{-x}}{2},
\qquad
\sinh x = \frac{\eu^x - \eu^{-x}}{2},
\qquad
\tanh x = \frac{\sinh x}{\cosh x}
\]
(\emph{cosseno hiperbólico, seno, tangente}\index{cosseno hiperbólico, seno, tangente}). $\cosh $ é par, $\sinh $ e $\tanh $ são estranhos.
\end{definition}

\begin{proposition}[Propriedades básicas]\label{prop:b1:functions:hyprules}
Para todos $ x, y \in \R $:
\begin{enumerate}
  \item $\cosh^2 x - \sinh^2 x = 1$;
  \item $\cosh' = \sinh $, $\;\sinh' = \cosh $, $\;\tanh' = 1 - \tanh^2
        = \dfrac{1}{\cosh^2}$;
  \item $\cosh(x+y) = \cosh x\cosh y + \sinh x \sinh y $ e
        $\sinh(x+y) = \sinh x \cosh y + \cosh x \sinh y $;
  \item $\sinh $ é uma bijeção estritamente crescente de $\R $ para $\R $;
        $\cosh $ restrito a $\R_+$ é um aumento estritamente
        bijeção em $\intco{1}{+\infty}$; $\tanh $ é estritamente
        aumentando a bijeção de $\R $ para $\intoo{-1}{1}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) $\bigl(\frac{\eu^x + \eu^{-x}}{2}\bigr)^2 - \bigl(\frac{\eu^x -
\eu^{-x}}{2}\bigr)^2 = \frac{(\eu^{2x} + 2 + \eu^{-2x}) - (\eu^{2x} -
2 + \eu^{-2x})}{4} = 1$.

(2) Diferenciar as fórmulas definidoras; para $\tanh $, o quociente
regra dá $\frac{\cosh^2 - \sinh^2}{\cosh^2}$, que é ambos
$1 - \tanh^2$ e $\frac{1}{\cosh^2}$ por (1).

(3) Expanda os lados direitos usando as definições. Em detalhes
para a primeira fórmula:
\[
\cosh x\cosh y + \sinh x\sinh y
= \frac{(\eu^x + \eu^{-x})(\eu^y + \eu^{-y})
+ (\eu^x - \eu^{-x})(\eu^y - \eu^{-y})}{4} ;
\]
dos oito produtos, os quatro “mistos”
($\eu^{x-y}$, $\eu^{y-x}$) cancelar em pares, enquanto
$\eu^{x+y}$ e $\eu^{-(x+y)}$ cada um aparece duas vezes:
o total é $\frac{2\eu^{x+y} + 2\eu^{-(x+y)}}{4} = \cosh(x+y)$.
A fórmula do seno é idêntica aos termos mistos sobreviventes
em vez disso.

(4) $\sinh' = \cosh \geq 1 > 0$, então $\sinh $ está aumentando estritamente,
com limites $\pm\infty $ (termo dominante $\pm\frac{\eu^{\abs
x}}{2}$); a bijeção \omterm{def:b1:logic:statement}{declaração} então segue do intermediário
teorema do valor (usado no ensino médio; tratamento sistemático em
\cref{ch:b1:continuity}). Sobre $\R_+$, $\cosh' = \sinh \geq 0$
desaparecendo apenas em $0$: aumentando estritamente de $\cosh 0 = 1$ para
$+\infty $. E $\tanh' > 0$ com limites $\pm 1$ no $\pm\infty $: em
detalhe,
\[
\tanh x = \frac{\eu^x - \eu^{-x}}{\eu^x + \eu^{-x}}
= \frac{1 - \eu^{-2x}}{1 + \eu^{-2x}}
\xrightarrow[x \to +\infty]{} 1 ,
\]
depois de dividir o numerador e o denominador por $\eu^x $ e estranheza
dá o limite $-1$ no $-\infty $; o estritamente crescente
função $\tanh $ portanto \omterm{def:b1:logic:map}{mapas} $\R $ para $\intoo{-1}1$.
\end{proof}

\begin{remark}[Por que “hiperbólico”?]
O ponto $(\cosh t, \sinh t)$ corre ao longo do galho $ x > 0$ do
hipérbole $ x^2 - y^2 = 1$ (por
\cref{prop:b1:functions:hyprules}~(1)), exatamente como $(\cos t, \sin
t)$ corre ao longo do círculo $ x^2 + y^2 = 1$. Toda identidade circular tem
um irmão hiperbólico, com mudanças de sinal governadas por
$\sinh^2 \leftrightarrow -\sin^2$.
\end{remark}

\begin{example}[A fórmula de adição para $\tanh $]\label{ex:b1:functions:tanhadd}
Dividindo as duas fórmulas de adição de
\cref{prop:b1:functions:hyprules}~(3) por $\cosh x\cosh y $:
\[
\tanh(x + y)
= \frac{\sinh x\cosh y + \cosh x\sinh y}
{\cosh x\cosh y + \sinh x\sinh y}
= \frac{\tanh x + \tanh y}{1 + \tanh x\,\tanh y} ,
\]
o irmão hiperbólico da fórmula de adição tangente --- com um
$+$ onde a trigonometria tem um $-$. Um dividendo: desde $\abs{\tanh}
< 1$, o lado direito é uma regra de “adição de velocidade” que
nunca sai $\intoo{-1}1$: se $ u, v \in \intoo{-1}1$ então
$\frac{u + v}{1 + uv} \in \intoo{-1}1$ também (escrever $ u = \tanh
a $, $ v = \tanh b $, possível por bijetividade, e leia a fórmula
para trás). Verificando isso algebricamente, sem hiperbólico
funções, é um exercício ligeiramente doloroso; parametrização por
$\tanh $ torna uma linha --- a mesma estratégia que circular
funções fornecem para o círculo unitário.
\end{example}

\begin{proposition}[Funções hiperbólicas inversas]\label{prop:b1:functions:invhyp}
As inversas concedidas por \cref{prop:b1:functions:hyprules}~(4) tem
formulários fechados:
\[
\operatorname{arsinh} x = \ln\bigl(x + \sqrt{x^2 + 1}\bigr)
\ (x \in \R),
\qquad
\operatorname{arcosh} x = \ln\bigl(x + \sqrt{x^2 - 1}\bigr)
\ (x \geq 1),
\]
\[
\operatorname{artanh} x = \frac 12 \ln\frac{1 + x}{1 - x}
\ (\abs x < 1),
\]
com derivados $\frac{1}{\sqrt{x^2+1}}$, $\frac{1}{\sqrt{x^2-1}}$
($ x > 1$) e $\frac{1}{1 - x^2}$ respectivamente.
\end{proposition}

\begin{proof}
Para $\operatorname{arsinh}$: resolver $ x = \sinh y = \frac{\eu^y -
\eu^{-y}}{2}$. Contexto $ u = \eu^y > 0$: $ u^2 - 2xu - 1 = 0$, então $ u = x
+ \sqrt{x^2 + 1}$ (a raiz $ x - \sqrt{x^2+1}$ é negativo) e $ y =
\ln(x + \sqrt{x^2+1})$. Os outros dois são cálculos idênticos
($ u^2 - 2xu + 1 = 0$ para $\operatorname{arcosh}$, mantendo a raiz
$\geq 1$; um rearranjo de duas linhas para $\operatorname{artanh}$).
Derivadas: diferencie as expressões logarítmicas, por ex.
\[
\bigl(\ln(x + \sqrt{x^2+1})\bigr)'
= \frac{1 + \frac{x}{\sqrt{x^2+1}}}{x + \sqrt{x^2+1}}
= \frac{1}{\sqrt{x^2 + 1}} . \qedhere
\]
\end{proof}

\begin{example}[Formulários fechados no trabalho]\label{ex:b1:functions:arcoshtwo}
A solução de $\cosh t = 2$ com $ t \geq 0$ é, pelo fechado
forma, $ t = \operatorname{arcosh} 2 = \ln(2 + \sqrt3) \approx
1.317$; a outra solução é $-t $, por uniformidade --- e de fato
$\ln(2 - \sqrt3) = \ln\frac1{2 + \sqrt3} = -\ln(2 + \sqrt3)$: o
duas raízes $ u = \eu^t $ do quadrático $ u^2 - 4u + 1 = 0$ são
recíprocos, como seu produto $1$ (Vieta) exige. Este minúsculo
a computação exibe o padrão geral: equações hiperbólicas
converter para quadrático em $\eu^t $, e a simetria $ t \mapsto -t $
aparece como a simetria $ u \mapsto 1/u $ do quadrático ---
vale a pena lembrar ao resolver \cref{exo:b1:functions:7}.
\end{example}

\begin{method}[Escolhendo a forma primitiva correta]\label{met:b1:functions:primitive}
Os três padrões de derivadas a serem memorizados para integração
(\cref{ch:b1:integration}):
\[
\int \frac{\dd x}{1 + x^2} = \arctan x + C,
\qquad
\int \frac{\dd x}{\sqrt{1 - x^2}} = \arcsin x + C,
\qquad
\int \frac{\dd x}{\sqrt{x^2 + 1}} = \operatorname{arsinh} x + C,
\]
e para uma quadrática geral, reduza a estes completando o quadrado
e redimensionamento.
\end{method}

\begin{remark}[Onde este capítulo é usado]\label{rem:b1:functions:whereused}
Este capítulo é o vocabulário de trabalho de todas as análises que virão.
A escala de crescimento \cref{prop:b1:functions:powerrules} decide
questões de convergência em todo
\cref{ch:b1:seq,ch:b1:series}; as fórmulas derivadas de
\cref{prop:b1:functions:arcderiv} e
\cref{prop:b1:functions:invhyp} são os primitivos com mais frequência
necessário em \cref{ch:b1:integration}, através
\cref{met:b1:functions:primitive}; funções hiperbólicas
parametrizar as soluções da equação $ y'' = y $ em
\cref{ch:b1:diffeq} exatamente como funções circulares parametrizam
aqueles de $ y'' = -y $. A caracterização de $\exp $ por $ f' = f $,
$ f(0) = 1$ (\cref{prop:b1:functions:expln}) é unidimensional
semente da teoria das equações diferenciais lineares, e o
curva catenária $ y = \cosh x $ retornos entre as curvas planas de
\cref{ch:b1:curves}.
\end{remark}

\begin{remark}[Interlúdio: o programa de função inversa]\label{rem:b1:functions:interlude}
Este capítulo executou um programa quatro vezes: restringir uma função
até que se torne uma bijeção estritamente monotônica, nomeie o inverso,
transportar todas as fórmulas através dele. O que foi \emph{usado} em cada
passo --- que uma função contínua estritamente monotônica em um
intervalo é uma bijeção em um intervalo, e que seu inverso é
contínuo, então diferenciável dos pontos críticos --- era
emprestado com crédito do ensino médio. A dívida é paga dentro deste
volume: \cref{ch:b1:continuity} prova a bijeção \omterm{def:b1:logic:statement}{declaração}
(o teorema inverso monótono, baseado no valor intermediário
teorema) e \cref{ch:b1:derivative} prova a regra da derivada
$(f^{-1})' = 1/(f' \circ f^{-1})$ que produzia silenciosamente cada
fórmula de \cref{prop:b1:functions:arcderiv} e
\cref{prop:b1:functions:invhyp}. Lendo esses capítulos com
$\arcsin $ e $\operatorname{arsinh}$ em mente --- como o trabalhado
exemplos que a teoria foi construída para justificar --- é a forma pretendida
em torno da aparente circularidade.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b1:functions:1}
Calcule sem calculadora: $\arcsin\frac{\sqrt 3}{2}$;
$\;\arccos\bigl(-\frac 12\bigr)$; $\;\arctan(-1)$;
$\;\arcsin\bigl(\sin\frac{5\pi}{6}\bigr)$;
$\;\arccos\bigl(\cos\frac{7\pi}{4}\bigr)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:functions:2}
Dê o domínio de definição de $ f(x) = \arcsin(2x - 1)$ e calcular
$ f'$ onde definido. As mesmas perguntas para $ g(x) = \arctan\sqrt{x}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:functions:3}
Prove isso para todos $ x \in \R $: $\cosh x + \sinh x = \eu^x $ e
$(\cosh x + \sinh x)^n = \cosh nx + \sinh nx $ para cada $ n \in \Z $
(uma fórmula hiperbólica de Moivre).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:functions:4}
Simplificar $\cos(2\arcsin x)$ e $\sin(2\arctan x)$ em \omterm{pb:b1:logic:1}{algébrico}
expressões de $ x $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:functions:5}
Encomende, para grandes $ x $, as funções
$ x^{100}$, $\;\eu^{\sqrt x}$, $\;(\ln x)^{1000}$, $\;\eu^{x/100}$,
$\;x^{\ln x}$, do crescimento mais lento para o mais rápido, com justificativas
baseado em \cref{prop:b1:functions:powerrules}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:functions:6}
Estude a função $ f(x) = \arctan\dfrac{2x}{1 - x^2}$ em seu domínio:
calcular $ f'$, compare com $(2\arctan x)'$ e expressar $ f(x)$ em
termos de $\arctan x $ em cada um dos três intervalos do domínio.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:functions:7}
Resolva em $\R $: $\;\cosh x = 2$; então $5\cosh x - 4 \sinh x = 3$
(expresse as soluções com logaritmos). \emph{Dica para o segundo:
escreva tudo com $ u = \eu^x $.}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:functions:8}
Prove a identidade $\arctan\frac{1}{2} + \arctan\frac{1}{3} =
\frac{\pi}{4}$, então a fórmula de Machin\index{Fórmula de Machin}
\[
4\arctan\frac 15 - \arctan\frac{1}{239} = \frac{\pi}{4}.
\]
\emph{Dica: calcule a tangente de ambos os lados usando a adição
fórmula e controle em qual intervalo cada lado se encontra.}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:functions:9}
Prove isso para todos $ x \geq 0$: $\;x - \dfrac{x^3}{6} \leq \sin x \leq
x $ \emph{(estudar as derivadas sucessivas das diferenças)}e
deduzir $\lim_{x \to 0^+} \frac{x - \sin x}{x^3} = \frac 16$ é
plausível --- o próprio limite é estabelecido em
\cref{ch:b1:taylor}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:functions:10}
Para $ x \in \intcc{-1}{1}$, \omterm{def:b1:logic:sets}{definir}
$ h(x) = \arcsin\bigl(2x\sqrt{1 - x^2}\,\bigr) - 2\arcsin x $. Alguém pode
esperar $ h = 0$ da identidade $\sin 2y = 2\sin y\cos y $ --- mas
$ h $ é \emph{não} identicamente zero. Calcular $ h'$ nos intervalos abertos
onde existe, avalie $ h $ em pontos bem escolhidos, e dar o total
descrição constante por partes de $ h $ sobre $\intcc{-1}{1}$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:functions:11}
(Gudermanniano) Deixe $ g(x) = \arctan(\sinh x)$ para $ x \in \R $. Provar
isso $ g $ é uma bijeção ímpar e estritamente crescente de $\R $ para
$\intoo{-\frac\pi2}{\frac\pi2}$, que $ g'(x) = \frac{1}{\cosh x}$,
e isso $\tan g(x) = \sinh x $, $\;\sin g(x) = \tanh x $,
$\;\cos g(x) = \frac{1}{\cosh x}$: a função $ g $ links circulares
e trigonometria hiperbólica sem números complexos.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:functions:12}
Prove isso
\[
\arctan 1 + \arctan 2 + \arctan 3 = \pi .
\]
\emph{Dica: calcular $\arctan 2 + \arctan 3$ primeiro, localizar o
soma como em \cref{ex:b1:functions:arctansum}; cuidado para que o
o denominador da fórmula de adição é negativo aqui.}
\end{exercise}

\section{Problema: A equação $ x^y = y^x $}

\begin{problem}\label{pb:b1:functions:1}
Quais pares de números positivos satisfazem $ x^y = y^x $? Todo mundo sabe
um exemplo de aparência acidental, $2^4 = 4^2 = 16$; esse problema mostra
que nada sobre isso é acidental. Toda a equação é governada
pelas variações da função única
\[
f(t) = \frac{\ln t}{t} \qquad (t > 0),
\]
cujo estudo produz: a solução completa \omterm{def:b1:logic:sets}{definir} (uma diagonal mais um
ramo curvo através $(\eu, \eu)$), uma parametrização racional de
a filial, o fato de que $(2, 4)$ é o seu \emph{apenas} ponto inteiro
e a classificação de todos os seus pontos racionais --- mais, como
dividendos, a comparação de $\eu^\pi $ com $\pi^\eu $ e o
convergência monótona de $\bigl(1 + \frac1n\bigr)^n $ para $\eu $.
\index{função de potência}

\medskip
\noindent\textbf{Part I --- The function $ f(t) = \ln t / t $.}
\begin{enumerate}
  \item Justifique isso $ f $ é diferenciável em $\intoo0{+\infty}$,
        calcular $ f'$, e elabore a tabela de variação: $ f $
        aumenta estritamente em $\intoc0\eu $, diminui estritamente em
        $\intco\eu{+\infty}$, com máximo $ f(\eu) = \frac1\eu $.
  \item Determine os limites de $ f $ no $0^+$ e $+\infty $
        (\cref{prop:b1:functions:powerrules}), o sinal de $ f $
        (negativo em $\intoo01$, zero em $1$, positivo além),
        e esboce o gráfico.
  \item Verifique por cálculo direto que $ f(2) = f(4)$. (Mantenha isso
        igualdade em mente: todo o problema surge dela.)
  \item Deixar $ c \in \R $. Discuta, de acordo com o valor de $ c $, o
        número de soluções de $ f(t) = c $: exatamente um para $ c \leq
        0$; exatamente dois (um em $\intoo1\eu $, um em
        $\intoo\eu{+\infty}$) para $0 < c < \frac1\eu $; exatamente um
        para $ c = \frac1\eu $; nenhum para $ c > \frac1\eu $.
  \item Deduza isso $\eu^x > x^\eu $ para cada $ x > 0$ com $ x \neq
        \eu $ e, em particular, decidir qual dos $\eu^\pi $ e
        $\pi^\eu $ é maior.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The equation and its curve.} Neste
parte $ x, y > 0$.
\begin{enumerate}[resume]
  \item Mostre isso $ x^y = y^x \iff f(x) = f(y)$.
  \item Deduza a estrutura da solução \omterm{def:b1:logic:sets}{definir}: tudo diagonal
        pares $(x, x)$; e o \emph{não trivial} pares ($ x \neq
        y $), que satisfazem: ambas as coordenadas são $> 1$, e um
        reside em $\intoo1\eu $ enquanto o outro está em
        $\intoo\eu{+\infty}$.
  \item Parametrize os pares não triviais: escrevendo $ y = tx $ com $ t
        > 0$, $ t \neq 1$, mostre que $ x^y = y^x $ forças
        \[
        x(t) = t^{\frac1{t-1}},
        \qquad
        y(t) = t\,x(t) = t^{\frac{t}{t-1}},
        \]
        e que, inversamente, cada par é uma solução.
  \item Verifique isso $ t = 2$ dá $(2, 4)$, e prove a simetria
        $ x(1/t) = y(t)$, $ y(1/t) = x(t)$: invertendo o parâmetro
        troca as duas coordenadas.
  \item Determine os limites de $ x(t)$ e $ y(t)$ como $ t \to 1$
        (ambos tendem a $\eu $), como $ t \to +\infty $ ($ x \to 1$, $ y
        \to +\infty $) e como $ t \to 0^+$ ($ x \to +\infty $, $ y \to
        1$). Descreva o ramo resultante: uma curva assintótica para
        as linhas $ x = 1$ e $ y = 1$, cruzando a diagonal em
        $(\eu, \eu)$.
  \item Mostre isso $ t \mapsto x(t)$ está diminuindo estritamente em
        $\intoo1{+\infty}$. \emph{(Estudar $ h(t) = 1 - \frac1t - \ln
        t $, cujo sinal controla a derivada de $\ln x(t) =
        \frac{\ln t}{t - 1}$.)}
  \item Conclua que as soluções não triviais definem uma solução estritamente
        bijeção decrescente $\varphi \colon \intoo1\eu \to
        \intoo\eu{+\infty}$ com $ x^{\varphi(x)} = \varphi(x)^x $,
        e isso $\varphi $ é uma involução do ramo:
        $\varphi(\varphi(x)) = x $ onde quer que ambos os lados sejam definidos.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Integer and rational points.}
\begin{enumerate}[resume]
  \item Prove isso $(2, 4)$ e $(4, 2)$ são os únicos não triviais
        \emph{inteiro} soluções de $ x^y = y^x $.
  \item Para $ n \in \N^*$, aplique a parametrização com $ t = 1 +
        \frac1n $ e mostre isso
        \[
        x_n = \Bigl(1 + \frac1n\Bigr)^{n},
        \qquad
        y_n = \Bigl(1 + \frac1n\Bigr)^{n+1}
        \]
        é algo não trivial \emph{racional} solução para cada $ n $,
        com $(x_1, y_1) = (2, 4)$.
  \item Por outro lado, deixe $(x, y)$ ser uma solução racional não trivial
        com $ y > x $ e escreva $ t = y/x = 1 + \frac rs $ no mais baixo
        termos ($ r, s \in \N^*$). Usando $ x = t^{s/r}$ e admitindo
        a singularidade da fatoração primária (familiar de
        escola; comprovado em \cref{ch:b1:arith}), mostre que $ x $
        forças racionais tanto $ s $ e $ s + r $ ser $ r $-ésimos poderes
        de números inteiros.
  \item Mostre com o teorema binomial que $ a^r - b^r = r $ não tem
        soluções inteiras $ a > b \geq 1$ quando $ r \geq 2$ e
        concluir: as soluções racionais de $ x^y = y^x $ são
        exatamente os pares $(x_n, y_n)$ da questão 14 (e seus
        trocas).
  \item Verifique o caso $ n = 2$ numericamente: calcular $ f(9/4)$ e
        $ f(27/8)$ com quatro casas decimais e verifique se concordam.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Dividends.}
\begin{enumerate}[resume]
  \item Use as questões 7 e 11 para provar, sem mais
        cálculo, que a sequência $ x_n = \bigl(1 +
        \frac1n\bigr)^n $ está aumentando estritamente com $ x_n < \eu $,
        que $ y_n = \bigl(1 + \frac1n\bigr)^{n+1}$ é estritamente
        diminuindo com $ y_n > \eu $, e que ambos convergem para
        $\eu $. (O parâmetro $ t_n = 1 + \frac1n $ diminui para
        $1$.)
  \item Estabeleça a regra geral de comparação para $1 < a < b $:
        se $\eu \leq a $ então $ a^b > b^a $; se $ b \leq \eu $ então
        $ a^b < b^a $; e mostre pelos dois exemplos $(2, 3)$ e
        $(2, 5)$ que no caso misto $ a < \eu < b $ ambos
        os resultados realmente ocorrem.
  \item Suponha que a involução $\varphi $ da questão 12 é
        diferenciável em $\eu $ (isso é). Diferencie o
        identidade $\varphi(\varphi(x)) = x $ no $ x = \eu $ e deduzir
        $\varphi'(\eu) = -1$: o ramo cruza a diagonal em
        ângulo reto com ele.
  \item Encontre o par de soluções exclusivo com $ y = 3x $, em fechado
        formulário e verifique-o numericamente com quatro casas decimais via $ f $.
  \item Entre os números $\sqrt2$, $\sqrt[3]3$, $\sqrt[4]4$,
        $\sqrt[5]5$, determine o maior e identifique os dois
        que são iguais. \emph{(Comparar $ n^{1/n} =
        \eu^{f(n)}$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Synthesis.}
\begin{enumerate}[resume]
  \item Descreva a solução completa \omterm{def:b1:logic:sets}{definir} de $ x^y = y^x $ no
        quarto de plano $ x, y > 0$ --- diagonal mais ramo, seus
        interseção $(\eu, \eu)$, as assíntotas, o número inteiro
        ponto $(2, 4)$, os pontos racionais acumulando em
        $(\eu, \eu)$ --- em uma forma que você possa esboçar de memória.
  \item Onde exatamente o problema usou: (i) o crescimento
        comparações de \cref{prop:b1:functions:powerrules}; (ii)
        o teorema do valor intermediário (através da bijeção
        declarações); (iii) a unicidade admitida do primo
        fatoração? Uma frase cada.
  \item Moral, em um parágrafo curto: uma única tabela de variação
        resolveu uma equação com duas incógnitas, classificou sua
        pontos racionais, e provou a convergência monótona de
        $\bigl(1 + \frac1n\bigr)^n$ --- comente sobre esta economia,
        e nomeie onde cada fio é industrializado posteriormente no
        volume (\cref{ch:b1:seq} para a sequência,
        \cref{ch:b1:derivative} para tabelas de variação,
        \cref{ch:b1:taylor} pela precisão que falta à tabela).
\end{enumerate}
\end{problem}
