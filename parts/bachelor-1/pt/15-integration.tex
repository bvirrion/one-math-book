\chapter{Integração em um segmento}\label{ch:b1:integration}

O integrante do volume do Ensino Médio foi fundamentado em áreas ocupadas
intuitivamente. Este capítulo o constrói: primeiro para funções de etapa,
onde o integrante é uma soma finita, então para contínuo (e
por partes contínuo) funciona por aproximação uniforme --- o
lugar onde o teorema de Heine (\cref{thm:b1:continuity:heine}) ganha
seu sustento. O teorema fundamental do cálculo então conecta o
construção até os primitivos, e \omterm{thm:b1:integration:riemann}{Somas de Riemann} conecte-o a discreto
médias.

Por todo, $a < b $ são reais.

\section{Funções de etapa}

\begin{definition}\label{def:b1:integration:step}
$\varphi \colon \intcc{a}{b} \to \R $ é um \emph{passo
função}\index{passo
função} quando há uma subdivisão $ a = x_0
< x_1 < \dots < x_n = b $ tal que $\varphi $ é constante, igual a
$ c_i $, em cada \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalo} $\intoo{x_{i-1}}{x_i}$ (valores no
nós são irrestritos). Isso é integrante é
\[
\int_a^b \varphi = \sum_{i=1}^{n} c_i\,(x_i - x_{i-1}),
\]
independente da subdivisão escolhida (refinar duas subdivisões por
seu lado comum: cada lado permanece inalterado sob refinamento).
\end{definition}

\begin{proposition}\label{prop:b1:integration:stepprops}
Sobre funções de etapa, o integrante é linear, crescente ($\varphi \leq
\psi \implies \int\varphi \leq \int\psi $), e satisfaz os Chasles
relação $\int_a^b = \int_a^c + \int_c^b $ para $ a < c < b $.
\end{proposition}

\begin{proof}
O motor está \emph{invariância de refinamento}, afirmado no
definição: inserir um nó extra $ t \in \intoo{x_{i-1}}{x_i}$
em uma subdivisão substitui o termo $ c_i(x_i - x_{i-1})$ por
$ c_i(t - x_{i-1}) + c_i(x_i - t)$ --- o mesmo número --- então o
integrante permanece inalterado por qualquer refinamento finito. Agora pegue
$\varphi $ com subdivisão $\sigma $ e $\psi $ com subdivisão
$\sigma'$: no refinamento comum $\sigma \cup \sigma'$ ambos
são funções de etapa com o \emph{mesmo} nós, e em cada peça
$\varphi + \lambda\psi $ é constante igual a $ c_i + \lambda
d_i $: a linearidade se reduz à linearidade de somas finitas. Aumentar:
$ c_i \leq d_i $ em cada peça dá $\sum c_i \Delta_i \leq \sum
d_i \Delta_i $ (comprimentos $\Delta_i \geq 0$). Chasles: insira o
nó $ c $ e divida a soma nele.
\end{proof}

\section{Integral de uma função contínua}

\begin{theorem}[Aproximação uniforme]\label{thm:b1:integration:approx}
Deixar $ f $ ser contínuo sobre $\intcc{a}{b}$. Para cada $\varepsilon > 0$
há funções de etapa $\varphi, \psi $ com
\[
\varphi \leq f \leq \psi
\qquad\text{and}\qquad
\psi - \varphi \leq \varepsilon \text{ on } \intcc{a}{b}.
\]
\end{theorem}

\begin{proof}
Pelo teorema de Heine (\cref{thm:b1:continuity:heine}), $ f $ é uniformemente
contínuo: escolha $\delta $ para $\varepsilon $, e uma subdivisão de
malha $< \delta $ (igualmente espaçados, digamos, com $ n > \frac{b -
a}{\delta}$ peças). Em cada \omterm{def:b1:topology:closed}{fechado} pedaço $\intcc{x_{i-1}}{x_i}$, $ f $
atinge um mínimo $ m_i $ e um máximo $ M_i $
(\cref{thm:b1:continuity:evt}), e $ M_i - m_i \leq \varepsilon $
(os dois pontos extremos estão dentro $\delta $). \omterm{def:b1:logic:sets}{Definir} $\varphi = m_i $
e $\psi = M_i $ sobre $\intoo{x_{i-1}}{x_i}$ (e $\varphi = \psi = f $
nos nós).
\end{proof}

\begin{theorem}[Definição da integral]\label{thm:b1:integration:def}
Deixar $ f $ ser contínuo sobre $\intcc{a}{b}$. Os dois números
\[
I_-(f) = \sup\Bigl\{\int_a^b \varphi : \varphi \text{ step},\
\varphi \leq f\Bigr\},
\qquad
I_+(f) = \inf\Bigl\{\int_a^b \psi : \psi \text{ step},\ \psi \geq
f\Bigr\}
\]
são iguais; seu valor comum é o
\emph{membro}\index{membro} $\int_a^b f $ (também escrito $\int_a^b
f(t)\,\dd t $). Coincide com a noção anterior sobre passo
funções, e se estende por partes contínuo funções por
\omterm{thm:b1:logic:partition}{divisão} $\intcc{a}{b}$ nas descontinuidades (Chasles como
definição aí).
\end{theorem}

\begin{proof}
Ambos conjuntos não estão vazios ($ f $ é limitado) e cada degrau inferior integrante
é $\leq $ cada superior (monotonicidade em funções de etapa): então
$ I_-(f) \leq I_+(f)$. Por \cref{thm:b1:integration:approx}, para cada
$\varepsilon $ há um par com $\int\psi - \int\varphi \leq
\varepsilon(b - a)$: o sup e o inf são comprimidos juntos, $ I_- =
I_+$.
\end{proof}

\begin{example}[Contínuo por partes, sem drama]\label{ex:b1:integration:piecewise}
O função de piso sobre $\intcc{0}{3}$ é um função de passo em
disfarce: dividindo-se em seus saltos,
\[
\int_0^3 \lfloor t \rfloor\,\dd t
= \int_0^1 0 + \int_1^2 1 + \int_2^3 2 = 0 + 1 + 2 = 3 ,
\]
e os valores \emph{no} os pontos de salto $1, 2$ são irrelevantes:
mudar uma função em um número finito de pontos não muda integrante
(o enquadramento funções de etapa não são afetados). Este é o todo
conteúdo do ``por partes contínuo'' extensão: corte no
um número finito de descontinuidades, integre cada contínuo peça,
adicione --- Chasles como definição.
\end{example}

\begin{example}[A definição computa, uma vez]\label{ex:b1:integration:defcompute}
Deixar $ f(x) = x $ sobre $\intcc{0}{1}$ e corte em $ n $ peças iguais.
O \omterm{def:b1:arith:prime}{melhor} funções de etapa constante nas peças são $\varphi =
\frac{k-1}{n}$ e $\psi = \frac kn $ no $ k $-ª peça, com
\[
\int_0^1 \varphi = \sum_{k=1}^{n} \frac{k-1}{n}\cdot\frac1n
= \frac{n-1}{2n},
\qquad
\int_0^1 \psi = \sum_{k=1}^{n} \frac{k}{n}\cdot\frac1n
= \frac{n+1}{2n} .
\]
Cada menor integrante é $\leq I_-(f) \leq I_+(f) \leq $ cada
superior, então $\frac{n-1}{2n} \leq I_-(f) \leq I_+(f) \leq
\frac{n+1}{2n}$ para todos $ n $: ambos apertam $\frac12$, e
$\int_0^1 x\,\dd x = \frac12$ direto da definição. O
visão final: esta é a primeira e última vez que integramos
da definição --- o teorema fundamental abaixo substitui
todos esses cálculos por uma pesquisa antiderivada, que é
todo o ponto econômico deste capítulo.
\end{example}

\begin{theorem}[Propriedades]\label{thm:b1:integration:props}
Para $ f, g $ contínuo (ou em pedaços contínuo) sobre $\intcc{a}{b}$ e
$\lambda \in \R $:
\begin{enumerate}
  \item linearidade: $\int (f + \lambda g) = \int f + \lambda \int g $;
  \item monotonicidade: $ f \leq g \implies \int_a^b f \leq \int_a^b g $;
        e $\bigl|\int_a^b f\bigr| \leq \int_a^b \abs f \leq (b -
        a)\, \sup\abs f $;
  \item Castelos: $\int_a^b = \int_a^c + \int_c^b $ (com o
        convenção $\int_b^a = -\int_a^b $, válido para qualquer pedido de
        os \omterm{def:b1:topology:closure}{limites});
  \item positividade estrita: se $ f $ é contínuo, $ f \geq 0$ e
        $\int_a^b f = 0$, então $ f = 0$ em todos os lugares
        $\intcc{a}{b}$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1)--(3) passar de funções de etapa até o \omterm{def:b1:topology:closure}{limite} através do
definição sup/inf. A linearidade merece os detalhes uma vez: dado
$\varepsilon > 0$, quadro $\varphi_f \leq f \leq \psi_f $ e
$\varphi_g \leq g \leq \psi_g $ com lacunas $\leq \varepsilon $
(\cref{thm:b1:integration:approx}). Para $\lambda \geq 0$,
$\varphi_f + \lambda\varphi_g \leq f + \lambda g \leq \psi_f +
\lambda\psi_g $ é um enquadramento de funções de etapa com lacuna $\leq
(1 + \lambda)\varepsilon $, e seu passo integrais igual $\int
\varphi_f + \lambda\int\varphi_g $ etc.\
(\cref{prop:b1:integration:stepprops}): deixar $\varepsilon \to
0$ aperta $\int(f + \lambda g)$ para $\int f + \lambda\int g $.
Para $\lambda < 0$, multiplicando por $\lambda $ \emph{inverte} o
enquadramento de $ g $ --- o inferior função de passo de $\lambda g $ é
$\lambda \psi_g $ --- e o mesmo aperto acontece com os papéis
trocado. O \omterm{def:b1:topology:closure}{limite} $\abs{\int f} \leq \int \abs f $ vem de
$-\abs f \leq f \leq \abs f $ e monotonicidade.

(4) Contrapositivo: se $ f(x_0) = m > 0$, continuidade fornece um
subintervalo de comprimento $\eta > 0$ em que $ f \geq \frac m2$; o
função de passo valor $\frac m2$ lá e $0$ em outro lugar é $\leq f $,
então $\int f \geq \frac{m\eta}{2} > 0$.
\end{proof}

\begin{example}[Chasles no trabalho: integrais com valores absolutos]\label{ex:b1:integration:chasleswork}
Para integrar um valor absoluto, corte onde o sinal muda.
\[
\int_0^2 \abs{x - 1}\,\dd x
= \int_0^1 (1 - x)\,\dd x + \int_1^2 (x - 1)\,\dd x
= \frac12 + \frac12 = 1 ,
\]
e, cortando $\intcc{0}{2\pi}$ no $\pi $:
\[
\int_0^{2\pi} \abs{\sin t}\,\dd t
= \int_0^{\pi} \sin t\,\dd t - \int_{\pi}^{2\pi} \sin t\,\dd t
= 2 + 2 = 4 ,
\]
enquanto $\int_0^{2\pi} \sin t\,\dd t = 0$: o cancelamento é real,
e é por isso que a estrita positividade \omterm{def:b1:logic:statement}{declaração}
(\cref{thm:b1:integration:props}~(4)) carrega a hipótese $ f
\geq 0$ --- sem ele, um desaparecimento integrante não prova nada
sobre $ f $. A visão final: $\int \abs f $ medidas
\emph{área}, $\int f $ medidas \emph{saldo assinado}; o
desigualdade $\abs{\int f} \leq \int\abs f $ é o registro exato de
o que o cancelamento pode destruir.
\end{example}

\section{O teorema fundamental do cálculo}

\begin{theorem}[Teorema fundamental do cálculo]\label{thm:b1:integration:ftc}
Deixar $ f $ ser contínuo em um \omterm{prop:b1:reals:intervals}{intervalo} $ I $ e $ a \in I $. A função
\[
F(x) = \int_a^x f(t)\, \dd t
\]
é de classe $ C^1$ sobre $ I $, com $ F' = f $: todo contínuo função
em um \omterm{prop:b1:reals:intervals}{intervalo} tem primitivos. Consequentemente, para \emph{qualquer}
primitivo $ G $ de $ f $:\index{teorema fundamental do cálculo}
\[
\int_a^b f(t)\,\dd t = G(b) - G(a) .
\]
\end{theorem}

\begin{proof}
Consertar $ x_0 \in I $ e $\varepsilon > 0$; continuidade no $ x_0$ fornece
$\delta $ com $\abs{f(t) - f(x_0)} \leq \varepsilon $ para $\abs{t -
x_0} \leq \delta $. Para $0 < \abs{h} \leq \delta $ (e $ x_0 + h \in
I $), Chasles dá
\[
\frac{F(x_0 + h) - F(x_0)}{h} - f(x_0)
= \frac 1h \int_{x_0}^{x_0+h} \bigl(f(t) - f(x_0)\bigr)\dd t ,
\]
cujo valor absoluto é $\leq \frac{1}{\abs h}\cdot \abs h\,
\varepsilon = \varepsilon $ (encadernação (2), válido para qualquer ordem do
\omterm{def:b1:topology:closure}{limites}). Então $ F'(x_0) = f(x_0)$; $ F' = f $ é contínuo: $ F $ é
$ C^1$. Se $ G' = f $ também, então $(G - F)' = 0$ no \omterm{prop:b1:reals:intervals}{intervalo}, então $ G =
F + c $ (\cref{cor:b1:derivative:monotone}), e $ G(b) - G(a) = F(b)
- F(a) = \int_a^b f $.
\end{proof}

\begin{example}[Simetria antes da computação]\label{ex:b1:integration:parity}
Em um simétrico \omterm{prop:b1:reals:intervals}{intervalo}, a paridade faz o trabalho: se $ f $ é estranho,
a substituição $ t \mapsto -t $ envia $\int_{-a}^{0} f $ para
$-\int_0^a f $, então
\[
\int_{-a}^{a} f(t)\,\dd t = 0 ;
\qquad\text{if $ f $ is even,}\quad
\int_{-a}^{a} f = 2\int_0^a f .
\]
Por isso $\int_{-1}^{1} \frac{t^3\cos t}{1 + t^4}\,\dd t = 0$ com
nenhum primitivo à vista (o integrando é ímpar), e
$\int_{-\pi}^{\pi} t^2\cos t\,\dd t = 2\int_0^\pi t^2\cos t\,
\dd t $. Verifique a simetria antes de recorrer às técnicas: o
mais rápido integrante é aquele que nunca foi computado.
\end{example}

\begin{example}[Reconhecendo um derivado à vista]\label{ex:b1:integration:halfangle}
Calcular $\displaystyle\int_0^{\pi/2} \frac{\dd x}{1 + \cos x}$.
A identidade do meio ângulo $1 + \cos x = 2\cos^2\frac x2$ vira o
integrando em $\frac{1}{2}\bigl(1 + \tan^2\frac
x2\bigr)$, que é exatamente o derivado de $\tan\frac x2$:
\[
\int_0^{\pi/2} \frac{\dd x}{1 + \cos x}
= \Bigl[\tan\frac x2\Bigr]_0^{\pi/2} = \tan\frac\pi4 = 1 .
\]
Não foi necessária nenhuma maquinaria de substituição – apenas o reflexo da
lendo um integrando como alguém derivado, o fundamental
teorema fazendo o resto. (A ferramenta sistemática por trás de tal
trigonométrico integrais, a substituição $ t = \tan\frac x2$,
pertence ao kit de ferramentas padrão construído a partir de
\cref{thm:b1:integration:parts}~(2).)
\end{example}

\begin{example}[Funções definidas por integrais]\label{ex:b1:integration:definedby}
O teorema fundamental fabrica funções. Deixar
\[
F(x) = \int_0^x \eu^{-t^2}\,\dd t .
\]
Nenhuma combinação de funções clássicas foi derivado
$\eu^{-t^2}$ (um teorema de Liouville, admitido); ainda $ F $ existe,
é $ C^1$ com $ F'(x) = \eu^{-x^2} > 0$, estritamente crescente, ímpar
(substituto $ t \mapsto -t $) e limitado: para $ x \geq 1$,
\[
F(x) - F(1) = \int_1^x \eu^{-t^2}\dd t
\leq \int_1^x \eu^{-t}\dd t \leq \eu^{-1} ,
\]
então $ F \leq F(1) + \eu^{-1} \leq 1 + \eu^{-1}$. (O \omterm{def:b1:topology:closure}{limite} exato,
$\frac{\sqrt\pi}{2}$, é calculado com duplo integrais no
Volume do ano 3.) Regra da cadeia para \omterm{def:b1:topology:closure}{limites} móveis: $\frac{\dd}{\dd
x}\int_x^{x^2} \eu^{-t^2}\dd t = 2x\,\eu^{-x^4} - \eu^{-x^2}$.
O insight final: integração \emph{cria} novas funções
dos antigos, com todas as suas propriedades legíveis a partir do
integrando --- o primitivo que você não pode escrever ainda é um
função que você controla totalmente.
\end{example}

\begin{example}[Estimar sem avaliar]\label{ex:b1:integration:estimate}
O integrais $ R_n = \int_0^1 \frac{t^n}{1 + t}\,\dd t $ não tenho
agradável forma fechada, mas a monotonicidade os fixa com precisão: em
$\intcc{0}{1}$, $\frac12 \leq \frac{1}{1+t} \leq 1$, então
\[
\frac{1}{2(n+1)} = \frac12\int_0^1 t^n\,\dd t
\;\leq\; R_n \;\leq\; \int_0^1 t^n \,\dd t = \frac{1}{n+1} :
\]
a ordem exata de decaimento ($ R_n \sim $ um múltiplo de $\frac1n $,
na verdade $ R_n \sim \frac{1}{2n}$) com duas linhas e sem
antiderivada. Os problemas de fim de semana deste capítulo e do
próxima execução exatamente desses enquadramentos --- o primeiro
instinto antes de um integrante deveria ser \emph{amarrou}, e apenas
então, se necessário, calcule-o.
\end{example}

\begin{example}[Valores médios]\label{ex:b1:integration:average}
O \emph{significar} de um contínuo $ f $ sobre $\intcc{a}{b}$ é
$\frac{1}{b-a}\int_a^b f $. Para o arco do seno:
\[
\frac{1}{\pi}\int_0^\pi \sin t\,\dd t
= \frac{1}{\pi}\bigl[-\cos t\bigr]_0^\pi = \frac{2}{\pi}
\approx 0.637 :
\]
um arco positivo completo em média não $\frac12$ mas para
$\frac2\pi $ --- a curva passa mais tempo alta do que um triângulo
faria. Por \cref{exo:b1:integration:11} (teorema do valor médio para
integrais, $ g = 1$), a média é \emph{valor}: $\sin c =
\frac2\pi $ para alguns $ c \in \intoo{0}{\pi}$. E pelo Riemann
somas of this chapter, the mean is the limit of ordinary
médias de $ n $ amostras --- a ponte entre o discreto
média dos dados e o contínuo média de um sinal, que é como
o integrante entra na física.
\end{example}

\begin{theorem}[Integração por partes; substituição]\label{thm:b1:integration:parts}
\begin{enumerate}
  \item Se $ u, v $ são $ C^1$ sobre $\intcc{a}{b}$:\index{integração por
        peças}
        \[
        \int_a^b u'v = \bigl[uv\bigr]_a^b - \int_a^b uv' .
        \]
  \item Se $\varphi $ é $ C^1$ sobre $\intcc{\alpha}{\beta}$ e $ f $
        contínuo sobre $\varphi(\intcc{\alpha}{\beta})$:
        \index{substituição}
        \[
        \int_{\alpha}^{\beta} f\bigl(\varphi(t)\bigr)\,\varphi'(t)\,
        \dd t = \int_{\varphi(\alpha)}^{\varphi(\beta)} f(x)\, \dd x .
        \]
\end{enumerate}
\end{theorem}

\begin{proof}
(1) $(uv)' = u'v + uv'$; integrar sobre $\intcc{a}{b}$ e aplique o
teorema fundamental para o $ C^1$ função $ uv $.

(2) Deixe $ F $ ser um primitivo de $ f $ na imagem \omterm{prop:b1:reals:intervals}{intervalo}
(\cref{thm:b1:integration:ftc}). Então $(F \circ \varphi)' = (f
\circ \varphi)\,\varphi'$ (regra da cadeia), então ambos os lados são iguais
$ F(\varphi(\beta)) - F(\varphi(\alpha))$.
\end{proof}

\begin{example}\label{ex:b1:integration:parts}
$\int_0^1 t\,\eu^t \dd t = \bigl[t\,\eu^t\bigr]_0^1 - \int_0^1
\eu^t\dd t = \eu - (\eu - 1) = 1$. E com a substituição $ x =
\sin t $ ($ t \in \intcc{0}{\frac\pi2}$):
\[
\int_0^1 \sqrt{1 - x^2}\, \dd x
= \int_0^{\pi/2} \cos t \cdot \cos t \, \dd t
= \int_0^{\pi/2} \frac{1 + \cos 2t}{2}\, \dd t = \frac\pi4 ,
\]
--- um quarto do disco unitário, conforme a geometria exigir. (Linearização
de \cref{met:b1:complex:trig} No trabalho.)
\end{example}

\begin{remark}[Armadilhas comuns no cálculo integral]
(eu) \emph{As substituições devem ser $ C^1$ no geral \omterm{prop:b1:reals:intervals}{intervalo}}:
a mudança $ x = \frac1t $ é ilegal em $0$; aplicado cegamente
para $\int_{-1}^{1}\frac{\dd x}{1 + x^2}$ isso “prova” que
integrante é igual ao seu próprio negativo. Quando uma substituição tem um
singularidade, corte o \omterm{prop:b1:reals:intervals}{intervalo} primeiro (Chasles), substituto em
cada peça, e só então recombinar. (ii) \emph{Logarítmico
primitivos precisam de valores absolutos}: $\int \frac{\dd x}{x - 2} =
\ln\abs{x - 2} + C $ em cada lado de $2$ separadamente --- escrevendo
$\ln(x - 2)$ sobre $\intoo{0}{1}$ está escrevendo o logaritmo de um
número negativo; e a constante $ C $ pode diferir nos dois
lados da singularidade. (iii) \emph{Um desaparecimento integrante faz
não mate a função}: $\int_0^{2\pi}\sin = 0$; positividade de
o integrando é necessário antes de concluir $ f = 0$
(\cref{ex:b1:integration:chasleswork}). (4) \emph{\omterm{thm:b1:integration:riemann}{Somas de Riemann}
deve ser calibrado}: em $\frac{b-a}{n}\sum f\bigl(a +
k\frac{b-a}{n}\bigr)$, o passo para fora e os pontos para dentro
deve corresponder à mesma subdivisão --- o erro frequente é uma soma
$\sum_{k=1}^{n} f\bigl(\frac kn\bigr)$ \emph{sem} o fator
$\frac1n $, que diverge em vez de convergir para $\int_0^1
f $. Lista de verificação antes de invocar \cref{thm:b1:integration:riemann}:
fatorar $\frac1n $, reescreva a soma como $ f $ de $\frac kn $,
nome $ f $ e verifique seu continuidade.
\end{remark}

\begin{example}[Adivinhe, diferencie, ajuste]\label{ex:b1:integration:guessdiff}
O que é $\int_1^x (\ln t)^2\,\dd t $? Adivinhe uma primitiva do
formulário $ t\,P(\ln t)$ com $ P $ polinomial e diferenciar:
\[
\bigl(t\,P(\ln t)\bigr)' = P(\ln t) + P'(\ln t) .
\]
Nós precisamos $ P(u) + P'(u) = u^2$: pegar $ P(u) = u^2 - 2u + 2$
(coeficientes de correspondência para baixo de $ u^2$). Por isso
\[
\int_1^x (\ln t)^2\,\dd t
= \bigl[t\bigl((\ln t)^2 - 2\ln t + 2\bigr)\bigr]_1^x
= x(\ln x)^2 - 2x\ln x + 2x - 2 ,
\]
um resultado alcançado de outra forma por duas integrações por partes. O
visão final: para integrandos da forma
(polinomial em $\ln t $) ou (polinomial vezes $\eu^{\lambda t}$),
o primitivo tem a mesma forma --- diferenciando uma forma
palpite converte integração em álgebra linear em coeficientes,
mais rápido e menos sujeito a erros do que partes iteradas.
\end{example}

\begin{example}[O bumerangue integral]\label{ex:b1:integration:boomerang}
Calcular $ I = \int_0^{\pi/2} \eu^x \cos x\,\dd x $. Integrar por
partes duas vezes, diferenciando o fator trigonométrico de cada vez:
\[
I = \bigl[\eu^x \sin x\bigr]_0^{\pi/2}
- \int_0^{\pi/2} \eu^x \sin x\,\dd x
= \eu^{\pi/2} - J,
\qquad
J = \bigl[-\eu^x\cos x\bigr]_0^{\pi/2}
+ \int_0^{\pi/2} \eu^x\cos x\,\dd x = 1 + I .
\]
O integrante voltou a si mesmo: $ I = \eu^{\pi/2} - 1 - I $,
de onde
\[
I = \frac{\eu^{\pi/2} - 1}{2} .
\]
A conclusão final: quando o integrando é um produto de dois
funções que se reproduzem sob diferenciação
($\eu^{ax}$, $\cos bx $, $\sin bx $), duas integrações por partes
produzir uma equação linear para a incógnita integrante --- resolva
em vez de integrar; equivalentemente, passar por
$\eu^{(a + \iu b)x}$ (\cref{ch:b1:complex}) e pegue real
peças. Ambas as estradas dão a mesma resposta, e verificando se elas
fazer é um teste de sanidade gratuito.
\end{example}

\section{Somas de Riemann}

\begin{theorem}[Somas de Riemann]\label{thm:b1:integration:riemann}
Deixar $ f $ ser contínuo sobre $\intcc{a}{b}$. Então\index{Somas de Riemann}
\[
S_n = \frac{b - a}{n} \sum_{k=0}^{n-1}
f\Bigl(a + k\,\frac{b-a}{n}\Bigr)
\xrightarrow[n \to \infty]{} \int_a^b f(t)\, \dd t ,
\]
e da mesma forma com quaisquer pontos de avaliação dentro dos subintervalos.
\end{theorem}

\begin{proof}
$ S_n $ é o integrante do função de passo $\varphi_n $ igual a
$ f(a + k\frac{b-a}{n})$ no $ k $-ésimo subintervalo. Dado
$\varepsilon > 0$, continuidade uniforme (Heine) fornece $\delta $; para
$ n > \frac{b-a}{\delta}$, cada ponto de um subintervalo está dentro
$\delta $ do seu ponto de avaliação, então $\abs{f - \varphi_n} \leq
\varepsilon $ sobre $\intcc{a}{b}$, de onde
\[
\Bigl| \int_a^b f - S_n \Bigr|
= \Bigl| \int_a^b (f - \varphi_n) \Bigr|
\leq (b-a)\,\varepsilon . \qedhere
\]
\end{proof}

\begin{example}\label{ex:b1:integration:riemannexample}
$\displaystyle\sum_{k=1}^{n} \frac{1}{n + k} = \frac 1n
\sum_{k=1}^{n} \frac{1}{1 + k/n} \xrightarrow[n\to\infty]{}
\int_0^1 \frac{\dd x}{1 + x} = \ln 2$: um \omterm{def:b1:topology:closure}{limite} invisível para
\omterm{def:b1:topology:closure}{limites} elementares, transparentes como uma soma de Riemann.
\end{example}

\begin{example}[Uma segunda soma de Riemann, com calibração]\label{ex:b1:integration:riemann2}
Encontrar $\lim_{n\to\infty} \sum_{k=1}^{n} \dfrac{n}{(n+k)^2}$.
Calibrar:
\[
\sum_{k=1}^{n} \frac{n}{(n+k)^2}
= \frac{1}{n}\sum_{k=1}^{n} \frac{n^2}{(n+k)^2}
= \frac1n \sum_{k=1}^{n} \frac{1}{\bigl(1 + \frac kn\bigr)^2} ,
\]
uma soma de Riemann do contínuo $ f(x) = \frac{1}{(1+x)^2}$ sobre
$\intcc{0}{1}$: o \omterm{def:b1:topology:closure}{limite} é
\[
\int_0^1 \frac{\dd x}{(1 + x)^2}
= \Bigl[-\frac{1}{1+x}\Bigr]_0^1 = \frac12 .
\]
O insight final: toda a arte é a linha intermediária ---
forçar o summand na forma $ f(\frac kn)$ ao custo de
extraindo exatamente um fator $\frac1n $; uma vez que a forma é
certo, o teorema faz a análise e o teorema fundamental
faz aritmética.
\end{example}

\begin{remark}[Onde a integral funciona a seguir]
Cada construção do capítulo tem uma sequência. \omterm{thm:b1:integration:riemann}{Somas de Riemann}
retornar \cref{ch:b1:series} como a ponte entre séries e
integrais (comparação de $\sum \frac{1}{n^\alpha}$ com $\int
\frac{\dd t}{t^\alpha}$); o integrante restante é o mais nítido
forma da fórmula de Taylor em \cref{ch:b1:taylor}; o
$\sup $ A definição baseada em Lebesgue é o protótipo do Lebesgue
integrante do volume do Ano 3, onde as mesmas três propriedades
(linearidade, monotonicidade, um teorema de convergência) são reconstruídos em um
classe muito maior de funções. E o problema do fim de semana abaixo
voltas integração por partes em aritmética: a irracionalidade da
$\pi^2$.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b1:integration:1}
Calcular: $\displaystyle\int_0^1 \frac{\dd x}{x^2 - 4}$
\emph{(frações parciais, \cref{ch:b1:fractions})};
$\displaystyle\int_1^{\eu} \ln t \,\dd t $;
$\displaystyle\int_0^{\pi} t \sin t\, \dd t $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:integration:2}
Calcular $\displaystyle\int_0^{1} \frac{t}{(t^2+1)^2}\,\dd t $
(substituição) e
$\displaystyle\int_0^{\pi/2} \cos^3 t\, \dd t $ (escrever $\cos^3 =
\cos(1 - \sin^2)$).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:integration:3}
Encontre os \omterm{def:b1:topology:closure}{limites}, como \omterm{thm:b1:integration:riemann}{Somas de Riemann}:
\[
u_n = \sum_{k=1}^{n} \frac{n}{n^2 + k^2},
\qquad
v_n = \frac{1}{n}\sqrt[n]{\frac{(2n)!}{n!\,n^n}}
\ \emph{(pegue logaritmos)}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:integration:4}
Deixar $ f $ ser contínuo sobre $\intcc{0}{1}$. Calcular $\lim_{n\to\infty}
\int_0^1 x^n f(x)\,\dd x $. \emph{(Corte $\intcc{0}{1}$ no $1 -
\delta $.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:integration:5}
(Cauchy--Schwarz) Para $ f, g $ contínuo sobre $\intcc{a}{b}$, provar
\[
\Bigl(\int_a^b fg\Bigr)^{\!2} \leq \int_a^b f^2 \cdot \int_a^b g^2 ,
\]
expandindo $\int_a^b (f + \lambda g)^2 \geq 0$ como uma quadrática em
$\lambda $. Quando é uma igualdade?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:integration:6}
Deixar $ f $ ser contínuo sobre $\R $, $ T $-periódico. Prove isso $\int_a^{a+T}
f $ não depende de $ a $, e isso $\frac1x \int_0^x f(t)\,\dd t
\to \frac 1T \int_0^T f $ como $ x \to +\infty $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:integration:7}
Para $ f $ contínuo sobre $\intcc{0}{1}$ com $\int_0^1 f = \frac12$,
provar isso $ f $ tem um ponto fixo em $\intcc{0}{1}$. \emph{(Integrar
$ f(x) - x $ e usar positividade estrita,
\cref{thm:b1:integration:props}~ (4), através de sua contrapositiva
combinado com o teorema do valor intermediário.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:integration:8}
(Wallis integrais) Deixar $ W_n = \int_0^{\pi/2} \sin^n t\,\dd t $.
\begin{enumerate}
  \item Prove a recorrência $ n W_n = (n-1) W_{n-2}$ ($ n \geq 2$) por
        partes e calcular $ W_0, W_1$, então $ W_{2p}$ e $ W_{2p+1}$
        em forma fechada.
  \item Prove isso $(W_n)$ está diminuindo com $\frac{W_{n+1}}{W_n}
        \to 1$; provar que a quantidade $(n+1)\,W_{n+1} W_n $ é
        constante, igual a $\frac\pi2$; e deduza o equivalente
        $ W_n \sim \sqrt{\dfrac{\pi}{2n}}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:integration:9}
(Niven: $\pi $ é irracional) Suponha $\pi = \frac ab $ com $ a, b \in
\N^*$, e \omterm{def:b1:logic:sets}{definir}, para $ n $ a ser escolhido,
\[
P(x) = \frac{x^n (a - bx)^n}{n!},
\qquad
I_n = \int_0^{\pi} P(x)\sin x\, \dd x .
\]
\begin{enumerate}
  \item Prove isso $0 < I_n \leq \pi\,\frac{(\pi a)^n}{n!}$, que é
        $< 1$ para $ n $ grande.
  \item Prove isso $ P $ e todos os seus derivados pegar \emph{inteiro}
        valores em $0$ e em $\pi = \frac ab $. \emph{(Binômio
        expansão: os coeficientes de $ P $ vezes $ k!$ são inteiros
        para $ k \geq n $; e $ P(\pi - x) = P(x)$.)}
  \item \omterm{def:b1:logic:sets}{Definir} $ Q = P - P'' + P^{(4)} - \dots $ (uma soma finita). Verifique
        isso $\bigl(Q'\sin x - Q\cos x\bigr)' = P \sin x $, e deduza
        isso $ I_n = Q(\pi) + Q(0)$ é um número inteiro.
  \item Concluir.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:integration:10}
Deixar $ f $ ser $ C^1$ sobre $\intcc{a}{b}$. Prove o tipo Riemann-Lebesgue
\omterm{def:b1:topology:closure}{limite}
\[
\int_a^b f(t)\sin(\lambda t)\,\dd t \xrightarrow[\lambda \to
+\infty]{} 0
\]
integrando por partes. Então prove novamente para $ f $ apenas
contínuo, por aproximação uniforme com funções de etapa
(\cref{thm:b1:integration:approx}).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:integration:11}
(Teorema do valor médio para integrais) Deixar $ f, g $ ser contínuo sobre
$\intcc{a}{b}$ com $ g \geq 0$. Prove que existe $ c \in
\intcc{a}{b}$ com
\[
\int_a^b f(t)\,g(t)\,\dd t = f(c)\int_a^b g(t)\,\dd t ,
\]
e mostre com um exemplo que a hipótese $ g \geq 0$ não pode ser
caiu.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:integration:12}
(Momentos forçam zeros) Deixe $ f $ ser contínuo sobre $\intcc{a}{b}$
com
\[
\int_a^b f(t)\,t^k\,\dd t = 0 \qquad \text{for } k = 0, 1,
\dots, n .
\]
Prove isso $ f $ desaparece em $ n + 1$ pontos distintos de
$\intoo{a}{b}$. \emph{(Se $ f $ muda de sinal apenas em $ z_1 < \dots
< z_m $ com $ m \leq n $, integrar $ f $ contra $ P(t) = (t - z_1)
\cdots (t - z_m)$ e use positividade estrita.)}
\end{exercise}

\begin{remark}[Perspectivas dentro deste volume]
Três capítulos à frente apoiam-se diretamente neste.
\Cref{ch:b1:taylor} carrega o integrante restante --- o
mais nítida das três fórmulas de Taylor é uma integração por partes
iterado $ n $ vezes. \Cref{ch:b1:series} converte o enquadramento de
somas por integrais no teste decisivo para $\sum n^{-\alpha}$,
e seu problema de fim de semana refina esse enquadramento no modelo de Euler
constante. \Cref{ch:b1:curves} faz o integrante geométrico: o
comprimento de um arco parametrizado é $\int \sqrt{x'(t)^2 +
y'(t)^2}\,\dd t $, um integrante de um contínuo funcionar em um
segmento --- precisamente o objeto construído aqui, nenhuma teoria imprópria
necessário. O fato mais reutilizado será o mais humilde:
$\bigl|\int f\bigr| \leq (b - a)\sup\abs{f}$, a desigualdade
que transforma cada estimativa pontual em um integrante estimativa.
\end{remark}

\section{Problema: A máquina da irracionalidade integral}

\begin{problem}[{Problema de fim de semana --- $\eu $ e $\pi^2$ são
irracional, $\eu $ com seis casas decimais, e $\frac{22}{7} > \pi $ com
prova}]
\label{pb:b1:integration:1}
Um mecanismo alimenta todo esse problema: \emph{uma expressão que
deve ser um número inteiro positivo, mas é provavelmente menor que $1$,
não pode existir}. \Cref{exo:b1:integration:9} (Niven) correu uma vez para
provar $\pi \notin \Q $; aqui a gente industrializa. A máquina
precisa de três partes: um \emph{integralidade} entrada (valores finais
de bem escolhido \omterm{def:b1:poly:def}{polinômios}), um \emph{pequenez} entrada (um fator
$\frac{1}{n!}$ esmagando o integrante) e um \emph{ponte}
(integração por partes) conectando-os. Nós provamos isso $\eu $ é
irracional e calculá-lo com erro certificado, prove a hipótese de Legendre
teorema mais nítido de que $\pi^2$ é irracional e termina com o
mais charmoso integrante em análise: $\int_0^1 \frac{x^4(1 -
x)^4}{1 + x^2}\dd x = \frac{22}{7} - \pi $, quais colchetes $\pi $
à mão.\index{irrationality!of pi squared@of
$\pi^2$}\index{irrationality!of e@of $\eu $}

\medskip
\noindent\textbf{Part I --- Fuel.}
\begin{enumerate}
  \item Prove isso $\dfrac{c^{\,n}}{n!} \to 0$ para cada fixo $ c
        > 0$ \emph{(além $ n \geq 2c $, cada passo pelo menos metade
        o termo)}.
  \item (Beta integrais) Prove, por indução em $ m $ com
        integração por partes:
        \[
        \int_0^1 x^{\,k}\,(1 - x)^{\,m}\,\dd x =
        \frac{k!\,m!}{(k + m + 1)!}
        \qquad (k, m \in \N).
        \]
  \item Deduzir $\displaystyle\int_0^1 \bigl(x(1-x)\bigr)^n \dd x
        = \frac{1}{(2n+1)\binom{2n}{n}}$, e --- comparando com
        o \omterm{def:b1:topology:closure}{limite} $ x(1 - x) \leq \frac14$ --- a estimativa
        $\binom{2n}{n} \geq \dfrac{4^n}{2n+1}$, correspondência
        $\binom{2n}{n}^{1/n} \to 4$ de \cref{pb:b1:seq:1}.
  \item Prove o lema da pequenez usado duas vezes abaixo: para cada
        contínuo $ g > 0$ sobre $\intoo{0}{1}$,
        \[
        0 < \int_0^1 \bigl(x(1-x)\bigr)^n g(x)\,\dd x
        \leq \frac{\sup_{\intcc{0}{1}}\abs g}{4^{\,n}} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part II --- $\eu $: irrationality, then six
decimals.} \omterm{def:b1:logic:sets}{Definir} $ A_n = \displaystyle\int_0^1 x^n \eu^x \dd x $.
\begin{enumerate}[resume]
  \item Calcular $ A_0$ e $ A_1$, comprove a recorrência $ A_n = \eu
        - n A_{n-1}$, e os \omterm{def:b1:topology:closure}{limites} $0 < A_n \leq
        \dfrac{\eu}{n+1}$.
  \item Mostre por indução que $ A_n = \alpha_n + \beta_n \eu $
        com $\alpha_n, \beta_n \in \Z $.
  \item Deduza isso $\eu $ é irracional \emph{(se $\eu = \frac
        pq $, então $ q A_n $ é um número inteiro preso em
        $\intoo{0}{1}$ para $ n $ grande)}. Compare com a prova
        de \cref{exo:b1:seq:9}: mesma piada, combustível diferente.
  \item Prove, por indução e integração por partes, o exato
        fórmula do resto
        \[
        \eu = \sum_{k=0}^{n} \frac{1}{k!} + R_n,
        \qquad
        R_n = \frac{1}{n!}\int_0^1 (1 - t)^n\,\eu^{\,t}\,\dd t,
        \qquad
        \frac{1}{(n+1)!} \leq R_n \leq \frac{\eu}{(n+1)!} .
        \]
  \item Pegar $ n = 9$: vinculado $ R_9$ usando $\eu < 2.75$ (de
        $ b_2 = 2.75$ em \cref{ex:b1:seq:e}), avalie a soma,
        e concluir o bracketing certificado $2.7182818 \leq
        \eu \leq 2.7182823$ --- seis casas decimais, $\eu \approx
        2.718282$, com prova.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Legendre's theorem: $\pi^2$ is
irrational.} Deixar $ f(x) = \dfrac{x^n (1 - x)^n}{n!}$, e suponha
$\pi^2 = \frac ab $ com $ a, b \in \N^*$.
\begin{enumerate}[resume]
  \item Mostrar $ f(1 - x) = f(x)$ e $0 < f \leq
        \dfrac{1}{4^n\,n!}$ sobre $\intoo{0}{1}$.
  \item Mostre isso $ f^{(k)}(0)$ e $ f^{(k)}(1)$ são números inteiros para
        cada $ k \geq 0$ \emph{(expandir $ x^n(1-x)^n $ com número inteiro
        coeficientes; $\frac{k!}{n!} \in \Z $ para $ k \geq n $;
        então use a simetria)}.
  \item \omterm{def:b1:logic:sets}{Definir}
        \[
        G = b^{\,n} \sum_{k=0}^{n} (-1)^k\,
        \pi^{2n - 2k} f^{(2k)} .
        \]
        Mostre isso $ G(0)$ e $ G(1)$ são inteiros \emph{(cada
        $ b^n \pi^{2n-2k} = a^{\,n-k}\,b^{\,k}$)}.
  \item Verifique a telescopia $ G'' + \pi^2 G = b^n \pi^{2n+2} f
        = \pi^2 a^n f $, então
        \[
        \frac{\dd}{\dd x}\Bigl(G'(x)\sin \pi x - \pi\,G(x)\cos
        \pi x\Bigr) = \pi^2 a^n f(x)\sin \pi x .
        \]
  \item Integrar $\intcc{0}{1}$ e concluir
        \[
        \pi\,a^n \int_0^1 f(x)\sin(\pi x)\,\dd x = G(0) + G(1)
        \in \Z ,
        \]
        um \emph{positivo} inteiro limitado por $\dfrac{\pi
        a^n}{4^n\,n!}$.
  \item Conclua com a questão 1 que $\pi^2$ é irracional
        (Legendre, 1794), e que isso fortalece
        \cref{exo:b1:integration:9}: por que a irracionalidade
        de $\pi^2$ implicar que de $\pi $, e não inversamente?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Understanding the machine.}
\begin{enumerate}[resume]
  \item Localize as duas forças opostas (integralidade do ponto final
        dados; pequenez analítica do integrante) e o
        ponte, nas Partes II e III. Em seguida, explique por que o fator
        $\frac{1}{n!}$ em $ f $ é o ponto crucial: se for removido,
        integralidade sobrevive, mas cuja desigualdade morre, e por
        que reivindicou frações $\frac ab $ faz a prova então
        falhar?
  \item Efetividade: suponha que alguém afirme $\pi^2 = \frac ab $
        com $ a \leq 10$. Mostre que a contradição já
        pousa em $ n = 7$: calcular $\pi\,(10/4)^7/7! \approx
        0.38 < 1$. A máquina não apenas refuta; isso
        refuta por um estágio fixo e computável.
  \item A sanidade verifica a ponte incondicionalmente: prove por dois
        integrações por partes que
        \[
        \int_0^1 x(1 - x)\sin(\pi x)\,\dd x = \frac{4}{\pi^3},
        \]
        e conciliar com a pergunta 14 em $ n = 1$ (manter
        $\pi^2$ simbólico: a identidade telescópica diz $\pi^3
        \int_0^1 f_1 \sin \pi x = -(f_1''(0) + f_1''(1)) = 4$).
  \item O que faz $\eu^x $ e $\sin \pi x $ elegível como
        núcleos da máquina? Identifique a propriedade (cada uma satisfaz
        uma equação diferencial linear com constante
        coeficientes, tão repetidos integração por partes ciclos
        de volta ao início) e nomeie a fronteira: a mesma
        máquina, refinada por Hermite e Lindemann, prova $\eu $
        e $\pi $ \emph{\omterm{pb:b1:logic:1}{transcendental}} --- além deste volume.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- $\frac{22}{7}$ versus $\pi $, and the
moral.}
\begin{enumerate}[resume]
  \item Estabeleça o polinomial \omterm{thm:b1:logic:partition}{divisão}
        \[
        \frac{x^4(1-x)^4}{1 + x^2}
        = x^6 - 4x^5 + 5x^4 - 4x^2 + 4 - \frac{4}{1 + x^2},
        \]
        e deduzir a célebre identidade
        \[
        \int_0^1 \frac{x^4 (1-x)^4}{1 + x^2}\,\dd x
        = \frac{22}{7} - \pi .
        \]
  \item O integrando é positivo: conclua $\pi <
        \frac{22}{7}$. Então, saltando $\frac{1}{1+x^2}$ entre
        $\frac12$ e $1$ e usando $\int_0^1 (x(1-x))^4 =
        \frac{1}{630}$ (questão 3), prove
        \[
        \frac{22}{7} - \frac{1}{630} \;\leq\; \pi \;\leq\;
        \frac{22}{7} - \frac{1}{1260} ,
        \]
        ou seja\ $3.14126 \leq \pi \leq 3.14207$: dois corretos
        decimais, à mão.
  \item Generalizar: dividir $ x^{4m}(1-x)^{4m}$ por $1 + x^2$,
        mostre que o resto é a constante $(-4)^m $ \emph{(trabalho
        \omterm{def:b1:complex:field}{módulo} $ x^2 + 1$: $(1-x)^2 \equiv -2x $)}, deduzir
        racionais $ r_m $ com
        \[
        \abs{\pi - r_m} \leq 4^{\,1 - 5m} ,
        \]
        e verifique isso $ m = 1$ reproduz as questões 20--21.
  \item Confronte esses racionais com a teoria da aproximação
        de \cref{pb:b1:derivative:1}: calcular $\abs{\pi -
        \frac{22}{7}} \approx 1.26\cdot10^{-3}$ contra o
        Garantia de Dirichlet $\frac{1}{49}$ e citar
        $\abs{\pi - \frac{355}{113}} \approx 2.7\cdot10^{-7}$
        contra $\frac{1}{113^2} \approx 7.8\cdot10^{-5}$:
        existem aproximações racionais excepcionalmente boas para
        $\pi $ --- consistente, já que $\pi $ não é conhecido por ser
        mal aproximado.
  \item (A armadilha inteira, abstraída) Prove o lema que
        unifica tudo: \emph{se $ x \in \R $ e existem
        inteiros $ a_n, b_n $ com $0 < \abs{a_n + b_n x} \to 0$,
        então $ x $ é irracional.} Liste suas instâncias neste
        problema, em \cref{exo:b1:integration:9}, em
        \cref{exo:b1:seq:9}, e em \cref{pb:b1:derivative:1}.
  \item Síntese, uma frase cada: (i) os três
        partes e onde cada uma vive na caixa de ferramentas deste
        capítulo; (ii) o que integrante contribui para que o
        teorema do valor médio de \cref{pb:b1:derivative:1} poderia
        não; (iii) o inventário dos resultados extraídos (dois
        irracionalidades, uma constante de seis casas decimais, uma
        colchetes de $\pi $, um \omterm{def:b1:topology:closure}{limite} binomial); (iv) o
