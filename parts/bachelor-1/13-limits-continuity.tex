\chapter{Limits and Continuity}\label{ch:b1:continuity}

The intermediate value theorem and the extreme value theorem were
used at High School level on visual faith. With sequences
(\cref{ch:b1:seq}) and the topology of $\R$ (\cref{ch:b1:topology})
in hand, this chapter proves them --- and completes the theory with
the monotone bijection theorem (which legitimizes $\arcsin$,
$\operatorname{arcosh}$ and their kin from \cref{ch:b1:functions})
and Heine's theorem on uniform continuity.

Throughout, $I$ is an interval and $f \colon I \to \R$; ``$x_0 \in
\overline I$'' allows limits at endpoints.

\section{Limits of functions}

\begin{definition}[Limit at a point]\label{def:b1:continuity:limit}
Let $x_0 \in \overline{I}$ and $\ell \in \R$. Then $f(x) \to \ell$ as
$x \to x_0$ when\index{limit!of a function}
\[
\forall \varepsilon > 0,\ \exists \delta > 0,\ \forall x \in I,
\qquad \abs{x - x_0} \leq \delta \implies \abs{f(x) - \ell} \leq
\varepsilon .
\]
Limits at $\pm\infty$ and infinite limits are defined by the same
pattern ($\abs{x - x_0} \leq \delta$ becomes $x \geq M$; $\abs{f(x) -
\ell} \leq \varepsilon$ becomes $f(x) \geq M'$). One-sided limits
restrict $x$ to $x > x_0$ (written $x \to x_0^+$) or $x < x_0$. The
limit is unique when it exists (same proof as
\cref{prop:b1:seq:first}).
\end{definition}

\begin{theorem}[Sequential characterization]\label{thm:b1:continuity:seqchar}
$f(x) \to \ell$ as $x \to x_0$ if and only if: for \emph{every}
sequence $(u_n)$ of points of $I$ with $u_n \to x_0$, one has
$f(u_n) \to \ell$.
\end{theorem}

\begin{proof}
($\Rightarrow$) Let $u_n \to x_0$ and $\varepsilon > 0$. Take
$\delta$ from the definition, then $N$ with $\abs{u_n - x_0} \leq
\delta$ for $n \geq N$: beyond $N$, $\abs{f(u_n) - \ell} \leq
\varepsilon$.

($\Leftarrow$) Contrapositive. If $f \not\to \ell$: some
$\varepsilon_0 > 0$ defeats every $\delta$; choosing $\delta =
\frac{1}{n+1}$ produces $u_n \in I$ with $\abs{u_n - x_0} \leq
\frac{1}{n+1}$ and $\abs{f(u_n) - \ell} > \varepsilon_0$. Then $u_n
\to x_0$ but $f(u_n) \not\to \ell$.
\end{proof}

\begin{corollary}[Operations, composition, order]\label{cor:b1:continuity:operations}
Sums, products, quotients (nonzero limit downstairs) of limits behave
as for sequences; if $f \to \ell$ at $x_0$ and $g \to m$ at $\ell$,
and $g$ is defined around $\ell$ with $g(\ell) = m$ or $f \neq \ell$
near $x_0$, then $g \circ f \to m$ at $x_0$; limits preserve wide
inequalities, and the squeeze theorem holds.
\end{corollary}

\begin{proof}
Each statement transfers through
\cref{thm:b1:continuity:seqchar} to its sequence analogue
(\cref{thm:b1:seq:operations,thm:b1:seq:order}). (Composition can
also be checked directly by chaining the two definitions.)
\end{proof}

\section{Continuity}

\begin{definition}\label{def:b1:continuity:continuous}
$f$ is \emph{continuous at $x_0 \in I$}\index{continuity} when $f(x)
\to f(x_0)$ as $x \to x_0$; \emph{continuous on $I$} when it is
continuous at every point. By
\cref{thm:b1:continuity:seqchar}: $f$ is continuous at $x_0$ iff
$f(u_n) \to f(x_0)$ for every sequence $u_n \to x_0$ in $I$.
\end{definition}

\begin{proposition}\label{prop:b1:continuity:algebra}
Sums, products, quotients (where defined) and compositions of
continuous functions are continuous. Polynomials, rational fractions
(off their poles), $\abs{\,\cdot\,}$, $\exp$, $\ln$, the
trigonometric and hyperbolic functions and their inverses
(\cref{ch:b1:functions}) are continuous on their domains.
\end{proposition}

\begin{proof}
Operations: \cref{cor:b1:continuity:operations}. Constants and the
identity are continuous, hence polynomials and rational fractions.
$\abs{\,\cdot\,}$: reverse triangle inequality. For the classical
functions we grant continuity here; differentiability (proved in
\cref{ch:b1:derivative}) is stronger.
\end{proof}

\begin{theorem}[Intermediate value theorem]\label{thm:b1:continuity:ivt}
Let $f$ be continuous on $\intcc{a}{b}$ with $f(a) \leq 0 \leq
f(b)$. Then $f(c) = 0$ for some $c \in \intcc{a}{b}$.
Consequently, a continuous function on an interval takes every value
between any two of its values: $f(I)$ is an
interval.\index{intermediate value theorem}
\end{theorem}

\begin{proof}
Dichotomy. Set $a_0 = a$, $b_0 = b$. Given $\intcc{a_n}{b_n}$ with
$f(a_n) \leq 0 \leq f(b_n)$, let $m$ be the midpoint: if $f(m) \leq
0$ keep $\intcc{m}{b_n}$, otherwise keep $\intcc{a_n}{m}$; the sign
conditions persist. The sequences $(a_n), (b_n)$ are adjacent ($b_n -
a_n = \frac{b-a}{2^n}$), with common limit $c$
(\cref{thm:b1:seq:adjacent}). By continuity and
\cref{thm:b1:seq:order}: $f(c) = \lim f(a_n) \leq 0$ and $f(c) =
\lim f(b_n) \geq 0$, so $f(c) = 0$.

For the consequence: given values $f(u) < v < f(w)$, apply the above
to $x \mapsto f(x) - v$ on the segment with endpoints $u$ and $w$;
thus $f(I)$ is convex, i.e.\ an interval
(\cref{prop:b1:reals:intervals}).
\end{proof}

\begin{theorem}[Extreme value theorem]\label{thm:b1:continuity:evt}
A continuous function on a segment $\intcc{a}{b}$ is bounded and
attains its bounds: there are $c, d \in \intcc{a}{b}$ with
\[
f(c) = \inf_{\intcc{a}{b}} f,
\qquad
f(d) = \sup_{\intcc{a}{b}} f .
\]
Combined with \cref{thm:b1:continuity:ivt}: \emph{the continuous
image of a segment is a segment} $\intcc{f(c)}{f(d)}$.
\index{extreme value theorem}
\end{theorem}

\begin{proof}
\emph{Bounded above:} otherwise pick $u_n$ with $f(u_n) \geq n$. By
compactness of the segment (\cref{thm:b1:topology:compact}), a
subsequence $u_{\varphi(n)} \to x \in \intcc{a}{b}$; continuity gives
$f(u_{\varphi(n)}) \to f(x)$, but $f(u_{\varphi(n)}) \geq \varphi(n)
\to +\infty$: contradiction.

\emph{Sup attained:} let $M = \sup f$ and choose $v_n$ with $f(v_n) >
M - \frac{1}{n+1}$ (\cref{prop:b1:reals:epsilon}). Extract
$v_{\varphi(n)} \to d \in \intcc{a}{b}$: then $f(d) = \lim
f(v_{\varphi(n)}) = M$ by the squeeze. The infimum is handled by
$-f$.
\end{proof}

\section{Monotone functions and inverse functions}

\begin{theorem}[Monotone bijection theorem]\label{thm:b1:continuity:bijection}
Let $f$ be continuous and strictly monotonic on an interval $I$. Then
\begin{enumerate}
  \item $f$ is a bijection from $I$ onto the interval $J = f(I)$;
  \item the inverse $f^{-1} \colon J \to I$ is strictly monotonic (in
        the same direction) and \emph{continuous}.
\end{enumerate}
\end{theorem}

\begin{proof}
Say $f$ is strictly increasing. (1) Injectivity is immediate from
strict monotonicity; surjectivity onto $f(I)$ is trivial, and $f(I)$
is an interval by \cref{thm:b1:continuity:ivt}.

(2) $f^{-1}$ is strictly increasing: if $y < y'$ in $J$ but $f^{-1}(y)
\geq f^{-1}(y')$, applying the increasing $f$ gives $y \geq y'$,
absurd. Continuity of $f^{-1}$ at $y_0 = f(x_0) \in J$: let
$\varepsilon > 0$. The points $x_0 \pm \varepsilon$ (intersected with
$I$; treat interior points --- endpoints are similar, one-sided) have
images $f(x_0 - \varepsilon) < y_0 < f(x_0 + \varepsilon)$. Take
$\delta = \min\bigl(y_0 - f(x_0 - \varepsilon),\, f(x_0 +
\varepsilon) - y_0\bigr) > 0$: for $\abs{y - y_0} \leq \delta$,
monotonicity of $f^{-1}$ squeezes $f^{-1}(y)$ between $x_0 -
\varepsilon$ and $x_0 + \varepsilon$. (Note: continuity of $f^{-1}$
was \emph{not} deduced from that of $f$ by symmetry --- it is
monotonicity on an interval that does the work.)
\end{proof}

\begin{remark}
This theorem is what \cref{def:b1:functions:arc} and
\cref{prop:b1:functions:invhyp} silently used: $\arcsin$, $\arccos$,
$\arctan$, $\operatorname{arsinh}$, \dots\ are continuous. A
complement (\cref{exo:b1:continuity:10}): a continuous
\emph{injective} function on an interval is automatically strictly
monotonic, so the monotonicity hypothesis costs nothing.
\end{remark}

\section{Uniform continuity}

\begin{definition}\label{def:b1:continuity:uniform}
$f \colon I \to \R$ is \emph{uniformly continuous}\index{uniform
continuity} when
\[
\forall \varepsilon > 0,\ \exists \delta > 0,\ \forall x, y \in I,
\qquad \abs{x - y} \leq \delta \implies \abs{f(x) - f(y)} \leq
\varepsilon .
\]
The point: $\delta$ depends only on $\varepsilon$, not on the
location in $I$. Uniform continuity implies continuity; a
\emph{Lipschitz} function ($\abs{f(x) - f(y)} \leq k \abs{x - y}$)
is uniformly continuous ($\delta = \varepsilon/k$).
\end{definition}

\begin{example}\label{ex:b1:continuity:notuniform}
$x \mapsto x^2$ is continuous on $\R$ but \emph{not} uniformly
continuous: $\abs{(n + \frac 1n)^2 - n^2} = 2 + \frac{1}{n^2} \geq 2$
although the arguments are at distance $\frac 1n \to 0$. On any
bounded interval it is Lipschitz, hence uniformly continuous ---
consistent with Heine's theorem below.
\end{example}

\begin{theorem}[Heine]\label{thm:b1:continuity:heine}
A continuous function on a \emph{segment} $\intcc{a}{b}$ is uniformly
continuous.\index{Heine's theorem}
\end{theorem}

\begin{proof}
By contradiction: suppose some $\varepsilon_0 > 0$ defeats every
$\delta$. With $\delta = \frac{1}{n+1}$, pick $x_n, y_n \in
\intcc{a}{b}$ with $\abs{x_n - y_n} \leq \frac{1}{n+1}$ and
$\abs{f(x_n) - f(y_n)} > \varepsilon_0$. Compactness
(\cref{thm:b1:topology:compact}) extracts $x_{\varphi(n)} \to c \in
\intcc{a}{b}$; then $y_{\varphi(n)} \to c$ too (squeeze on $\abs{x -
y}$). Continuity at $c$ gives $f(x_{\varphi(n)}) \to f(c)$ and
$f(y_{\varphi(n)}) \to f(c)$, so $\abs{f(x_{\varphi(n)}) -
f(y_{\varphi(n)})} \to 0 < \varepsilon_0$: contradiction.
\end{proof}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:continuity:1}
Using the sequential characterization, prove that $x \mapsto
\sin\frac 1x$ has no limit at $0^+$ \emph{(exhibit two sequences)}.
Does $x \mapsto x \sin\frac 1x$ have one?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:continuity:2}
Study the continuity on $\R$ of $f(x) = \lfloor x \rfloor$ and of
$g(x) = x - \lfloor x \rfloor$, and of $h(x) = \lfloor x \rfloor +
(x - \lfloor x\rfloor)^2$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:continuity:3}
Prove that the equation $x^5 - 3x + 1 = 0$ has at least three real
solutions \emph{(evaluate at well-chosen points and apply
\cref{thm:b1:continuity:ivt} on three disjoint segments)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:continuity:4}
Prove that every polynomial of odd degree has a real root.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:continuity:5}
(Fixed point) Let $f \colon \intcc{0}{1} \to \intcc{0}{1}$ be
continuous. Prove that $f$ has a fixed point: $f(c) = c$ for some
$c$. Illustrate that neither continuity nor the segment can be
dropped.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:continuity:6}
Prove that the equation $\cos x = x$ has exactly one real solution,
and that it lies in $\intoo{0}{1}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:continuity:7}
Let $f \colon \R \to \R$ be continuous with $f(x) \to +\infty$ as $x
\to \pm\infty$. Prove that $f$ attains a global minimum on $\R$.
\emph{(Reduce to a segment containing a sublevel set.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:continuity:8}
Let $f \colon \R \to \R$ be continuous and periodic (of period $T >
0$). Prove that $f$ is bounded and attains its bounds, and that there
exists $c$ with $f(c + \frac T2) = f(c)$. \emph{(For the second
point, study $g(x) = f(x + \frac T2) - f(x)$ over one period.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:continuity:9}
Prove that $x \mapsto \sqrt x$ is uniformly continuous on
$\intco{0}{+\infty}$, although it is not Lipschitz near $0$.
\emph{(Prove and use $\abs{\sqrt x - \sqrt y} \leq \sqrt{\abs{x -
y}}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:continuity:10}
Let $f$ be continuous and injective on an interval $I$. Prove that
$f$ is strictly monotonic. \emph{Hint: if not, there are $a < b < c$
with, say, $f(b) > f(a)$ and $f(b) > f(c)$; apply the intermediate
value theorem to a value between $\max(f(a), f(c))$ and $f(b)$ on
both sides of $b$.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:continuity:11}
(Cauchy's functional equation, continuous case) Let $f \colon \R \to
\R$ be continuous with $f(x + y) = f(x) + f(y)$ for all $x, y$.
Prove that $f(x) = f(1)\,x$ for all $x$: first on $\N$, $\Z$, $\Q$
(by additivity alone), then on $\R$ by continuity and density
(\cref{thm:b1:reals:density}).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:continuity:12}
Let $f \colon \intco{0}{+\infty} \to \R$ be continuous with
$f(x) \to \ell \in \R$ as $x \to +\infty$. Prove that $f$ is
\emph{uniformly} continuous on $\intco{0}{+\infty}$. \emph{(Cut at a
large $A$: Heine on $\intcc{0}{A+1}$, the limit beyond $A$; make the
two regimes overlap.)}
\end{exercise}
