\chapter{Real Numbers}\label{ch:b1:reals}

All of analysis rests on one property that distinguishes $\R$ from
$\Q$: every nonempty \omterm{def:b1:logic:sets}{set} bounded from above has a \emph{least} \omterm{def:b1:reals:bounds}{upper
bound}. This chapter states it precisely, derives its first
consequences --- the \omterm{thm:b1:reals:archimedes}{Archimedean property}, the \omterm{thm:b1:reals:floor}{floor function}, the
density of the rationals and of the irrationals --- and \omterm{def:b1:logic:sets}{sets} up the
vocabulary (sup, inf, max, min) used constantly from
\cref{ch:b1:seq} onward.

\section{The upper bound property}

\begin{definition}[Bounds, sup and inf]\label{def:b1:reals:bounds}
Let $A \subseteq \R$ be nonempty. A real $M$ is an \emph{upper
bound}\index{upper bound} of $A$ when $a \leq M$ for all $a \in A$;
$A$ is \emph{bounded above} when it has an upper bound (similarly
below, with lower bounds; \emph{bounded} means both). A
\emph{maximum} of $A$ is an upper bound belonging to $A$.

The \emph{supremum} $\sup A$\index{supremum} is the least upper bound
of $A$, when it exists; the \emph{infimum} $\inf A$\index{infimum} is
the greatest lower bound.
\end{definition}

\begin{theorem}[Completeness axiom of $\R$]\label{thm:b1:reals:sup}
$\R$ is an ordered \omterm{def:b1:structures:field}{field} containing $\Q$ in which \emph{every
nonempty subset bounded above has a \omterm{def:b1:reals:bounds}{supremum}}.\index{completeness of
R@completeness of $\R$}
\end{theorem}
\admitted

\begin{remark}
We take this as the defining axiom of $\R$; constructing a model (by
Dedekind cuts or by Cauchy sequences of rationals) and proving its
uniqueness is honest but long, and is left for further study. Note
that $\Q$ fails the property: $\{x \in \Q : x^2 < 2\}$ is bounded
above but has no least \omterm{def:b1:reals:bounds}{upper bound} \emph{in $\Q$} --- its candidate,
$\sqrt 2$, is missing (\cref{ex:b1:logic:sqrt2}). By passing to
opposites ($\sup(-A) = -\inf A$), every nonempty \omterm{def:b1:logic:sets}{set} bounded below
has an \omterm{def:b1:reals:bounds}{infimum}.
\end{remark}

\begin{proposition}[The $\varepsilon$-characterization]\label{prop:b1:reals:epsilon}
Let $A \neq \emptyset$ be bounded above and $s \in \R$. Then $s =
\sup A$ if and only if
\begin{enumerate}
  \item $s$ is an \omterm{def:b1:reals:bounds}{upper bound}: $\forall a \in A$, $a \leq s$; and
  \item nothing smaller is: $\forall \varepsilon > 0$, $\exists a \in
        A$, $a > s - \varepsilon$.
\end{enumerate}
\end{proposition}

\begin{proof}
If $s = \sup A$: (1) holds by definition, and for (2), $s -
\varepsilon < s$ is not an \omterm{def:b1:reals:bounds}{upper bound}, which is exactly the
existence of $a > s - \varepsilon$. Conversely, (1) says $s$ is an
\omterm{def:b1:reals:bounds}{upper bound}; (2) says no $t < s$ is an \omterm{def:b1:reals:bounds}{upper bound} (take $\varepsilon
= s - t$): $s$ is the least one.
\end{proof}

\begin{example}\label{ex:b1:reals:supexamples}
$\sup \intoo{0}{1} = 1$, not attained (no maximum);
$\sup \intcc{0}{1} = 1 = \max$. For $A = \{1 - \frac 1n : n \in
\N^*\}$: $\sup A = 1$, not attained; $\inf A = \min A = 0$. A maximum,
when it exists, is the \omterm{def:b1:reals:bounds}{supremum}; the whole point of $\sup$ is to have
a substitute when the maximum does not exist.
\end{example}

\begin{example}[Computing suprema in practice]\label{ex:b1:reals:supcompute}
Two full workouts of \cref{prop:b1:reals:epsilon}.

\emph{The \omterm{def:b1:logic:sets}{set} $A = \{x + \frac1x : x > 0\}$.} For every $x > 0$,
$x + \frac1x - 2 = \frac{(\,\sqrt x - 1/\sqrt x\,)^2}{1} \geq 0$,
so $2$ is a lower bound; and $2 = 1 + \frac11 \in A$: therefore
$\inf A = \min A = 2$, attained at $x = 1$. Above, $A$ is
unbounded ($x + \frac1x > x$ can exceed any $M$ by
\cref{thm:b1:reals:archimedes}): $\sup A$ does not exist in $\R$
(it is $+\infty$ in $\overline\R$).

\emph{The \omterm{def:b1:logic:sets}{set} $B = \bigl\{\frac{m}{m + n} : m, n \in
\N^*\bigr\}$.} Every element lies in $\intoo{0}{1}$, so $0$ and
$1$ are bounds. Neither is attained: $\frac{m}{m+n} = 1$ would
force $n = 0$. For the \omterm{def:b1:reals:bounds}{supremum}, freeze $n = 1$ and let $m$ grow:
$\frac{m}{m+1} = 1 - \frac{1}{m+1} > 1 - \varepsilon$ as soon as
$m + 1 > \frac1\varepsilon$ (Archimedes): $\sup B = 1$.
Symmetrically ($m = 1$, $n$ large), $\inf B = 0$. The closing
insight: to pin a \omterm{def:b1:reals:bounds}{supremum}, one well-chosen \emph{one-parameter
path} inside the \omterm{def:b1:logic:sets}{set} suffices --- here the path $n = 1$ --- and
the $\varepsilon$-characterization asks for nothing more.
\end{example}

\begin{example}[The infimum mirror]\label{ex:b1:reals:infmirror}
The \omterm{def:b1:reals:bounds}{infimum} has its own $\varepsilon$-characterization, obtained
from \cref{prop:b1:reals:epsilon} through $\inf A = -\sup(-A)$:
$i = \inf A$ iff $i$ bounds $A$ below and, for every
$\varepsilon > 0$, some $a \in A$ has $a < i + \varepsilon$. A
workout with both bounds at once: let
\[
A = \Bigl\{(-1)^n + \frac1n : n \in \N^*\Bigr\}
= \Bigl\{0,\ \tfrac32,\ -\tfrac23,\ \tfrac54,\ -\tfrac45,\
\dots\Bigr\} .
\]
Even indices give $1 + \frac1n \leq \frac32$, with equality at
$n = 2$: since also the odd-index values are $\leq 0 <
\frac32$, we get $\sup A = \max A = \frac32$. Odd indices give
$-1 + \frac1n > -1$, decreasing toward $-1$: every element of
$A$ is $> -1$, and $-1 + \varepsilon$ is beaten by $-1 +
\frac1n$ for odd $n > \frac1\varepsilon$: $\inf A = -1$, not
attained. One \omterm{def:b1:logic:sets}{set}, all four behaviors on display: a \omterm{def:b1:reals:bounds}{supremum}
that is a maximum, an \omterm{def:b1:reals:bounds}{infimum} that is not a minimum.
\end{example}

\begin{remark}[Common pitfalls with sup and inf]
Four errors account for most lost points. (i) \emph{Confusing
$\sup$ and $\max$}: $\sup A$ need not belong to $A$; write $\max$
only after exhibiting an element of $A$ that is an \omterm{def:b1:reals:bounds}{upper bound}.
(ii) \emph{Passing strict inequalities to the \omterm{def:b1:reals:bounds}{supremum}}: if $a <
b$ for all $a \in A$, one may only conclude $\sup A \leq b$ ---
witness $A = \intoo{0}{1}$, $b = 1$. (iii) \emph{Writing $\sup A$
before checking legality}: the symbol requires $A$ nonempty and
bounded above (\cref{met:b1:reals:supproofs}); $\sup \emptyset$
and $\sup \N$ are undefined in $\R$ (the conventions of
$\overline\R$ are a separate, explicit act). (iv) \emph{Set
operations}: $\sup(A \cup B) = \max(\sup A, \sup B)$ always, but
nothing general holds for $A \cap B$ --- it may be empty, and
even when it is not, $\sup(A \cap B)$ can be far below
$\min(\sup A, \sup B)$: take $A = \{0, 2\}$ and $B = \{0, 3\}$,
where $\sup(A \cap B) = 0$.
\end{remark}

\begin{example}[Finite sets have maxima --- a lemma used silently]\label{ex:b1:reals:finitemax}
Every finite nonempty $F \subseteq \R$ has a maximum (and a
minimum). Induction on the number of elements: a singleton
$\{a\}$ has $\max = a$; if the claim holds for $n$-element \omterm{def:b1:logic:sets}{sets}
and $F$ has $n + 1$ elements, pick any $a \in F$: the \omterm{def:b1:logic:sets}{set} $F
\setminus \{a\}$ has a maximum $m$, and $\max F$ is $m$ if $a
\leq m$, else $a$. No completeness is involved --- this is pure
order plus induction, valid already in $\Q$ --- yet the lemma
deserves one honest \omterm{def:b1:logic:statement}{statement} because the coming proofs invoke
it silently: the \omterm{thm:b1:reals:floor}{floor} construction below (``a \omterm{def:b1:logic:sets}{set} of integers
trapped in a finite range has a greatest element''), every
$\max(\abs{u_0}, \dots, \abs{u_{N-1}}, \dots)$ bound in
\cref{ch:b1:seq}, every ``take the largest of the finitely many
$\delta$s'' in \cref{ch:b1:continuity}. Infinite \omterm{def:b1:logic:sets}{sets} are where
maxima die and suprema take over: this chapter exists for the
infinite case.
\end{example}

\begin{theorem}[Archimedean property]\label{thm:b1:reals:archimedes}
For every $x \in \R$ there is $n \in \N$ with $n > x$. Equivalently:
for all $\varepsilon > 0$ and $y > 0$, some multiple $n\varepsilon$
exceeds $y$.\index{Archimedean property}
\end{theorem}

\begin{proof}
Suppose not: some $x$ is an \omterm{def:b1:reals:bounds}{upper bound} of $\N$. Then $s = \sup \N$
exists (\cref{thm:b1:reals:sup}). By
\cref{prop:b1:reals:epsilon}~(2) with $\varepsilon = 1$, there is $n
\in \N$ with $n > s - 1$; but then $n + 1 \in \N$ and $n + 1 > s$,
contradicting that $s$ is an \omterm{def:b1:reals:bounds}{upper bound}. For the second form,
let $\varepsilon > 0$ and $y > 0$: the first form applied to $x
= \frac{y}{\varepsilon}$ produces $n \in \N$ with $n >
\frac{y}{\varepsilon}$, and multiplying by $\varepsilon > 0$
(which preserves strict inequalities) gives $n\varepsilon > y$.
Conversely, the second form with $\varepsilon = 1$ and $y = x$
recovers the first for $x > 0$, and $n = 1$ handles $x \leq 0$:
the two \omterm{def:b1:logic:statement}{statements} are strictly equivalent.
\end{proof}

\begin{example}[Archimedes at work]\label{ex:b1:reals:archimedeswork}
Three immediate uses, constantly needed later. (i) \emph{No
positive real is below every $\frac1n$}: if $0 < \varepsilon$,
pick $n > \frac1\varepsilon$; then $\frac1n < \varepsilon$. In
other words, $\R$ contains no infinitesimals --- the informal
``$\frac1n$ becomes arbitrarily small'' is exactly this theorem.
(ii) \emph{Explicit thresholds}: how large must $n$ be for
$\frac{1}{n^2} \leq 10^{-6}$? It suffices that $n \geq 10^3$
--- Archimedes guarantees such $n$ exist, and the algebra
locates them. (iii) \emph{Powers beat any bound}: $2^n \geq n +
1$ (induction), so for every $M$ some power of $2$ exceeds $M$:
the geometric growth used for the dyadics in
\cref{exo:b1:reals:8}. The closing insight: the \omterm{thm:b1:reals:archimedes}{Archimedean
property} is the licence behind every phrase of the form ``take
$n$ large enough'' --- from now on we use that phrase freely,
and this example is its one-time justification.
\end{example}

\begin{theorem}[Floor function]\label{thm:b1:reals:floor}
For every $x \in \R$ there is exactly one integer, the
\emph{floor}\index{floor function} $\lfloor x \rfloor$, with
\[
\lfloor x \rfloor \leq x < \lfloor x \rfloor + 1 .
\]
\end{theorem}

\begin{proof}
\emph{Existence.} The \omterm{def:b1:logic:sets}{set} $E = \{k \in \Z : k \leq x\}$ is nonempty:
by \cref{thm:b1:reals:archimedes} there is $m \in \N$ with $m > -x$,
and then $-m < x$, so $-m \in E$. It is bounded above (by any integer
$n > x$, which exists for the same reason), so, being a \omterm{def:b1:logic:sets}{set} of
integers trapped in the finite range $\intint{-m}{n}$, it has a
greatest element $k = \max E$. Then $k \leq x$, and $k + 1 \notin E$
means $x < k + 1$.

\emph{Uniqueness.} If $k$ and $k'$ both satisfy the inequalities,
then $k \leq x < k' + 1$ gives $k \leq k'$, and symmetrically $k'
\leq k$.
\end{proof}

\begin{example}[Floors in practice]\label{ex:b1:reals:floorpractice}
$\lfloor 3.7 \rfloor = 3$, $\lfloor 5 \rfloor = 5$, and $\lfloor
-3.7 \rfloor = -4$: the \omterm{thm:b1:reals:floor}{floor} goes \emph{down}, not toward $0$.
Two consequences of uniqueness in \cref{thm:b1:reals:floor} that
we will use silently. First, for $n \in \Z$,
\[
\lfloor x + n \rfloor = \lfloor x \rfloor + n ,
\]
because $\lfloor x \rfloor + n$ is an integer satisfying the two
defining inequalities for $x + n$ --- and only one integer does.
Second, $\lfloor \, \cdot \, \rfloor$ is nondecreasing: if $x \leq
y$ then $\lfloor x \rfloor \leq x \leq y < \lfloor y \rfloor + 1$,
and an integer $< \lfloor y \rfloor + 1$ is $\leq \lfloor y
\rfloor$. Beware, however, that $\lfloor 2x \rfloor \neq 2\lfloor
x \rfloor$ in general: $x = 0.6$ gives $\lfloor 1.2 \rfloor = 1
\neq 0 = 2\lfloor 0.6 \rfloor$.

What \emph{is} true is a worked identity worth keeping (Hermite's,
in its simplest case): for every real $x$,
\[
\lfloor x \rfloor + \Bigl\lfloor x + \frac12 \Bigr\rfloor
= \lfloor 2x \rfloor .
\]
Write $x = \lfloor x\rfloor + u$ with $u \in \intco{0}{1}$ and
separate two cases. If $u < \frac12$: the left side is $\lfloor
x\rfloor + \lfloor x\rfloor = 2\lfloor x\rfloor$, and $2x =
2\lfloor x\rfloor + 2u$ with $2u \in \intco{0}{1}$, so the right
side is $2\lfloor x\rfloor$ too. If $u \geq \frac12$: the left
side is $\lfloor x\rfloor + (\lfloor x\rfloor + 1)$, and $2u \in
\intco{1}{2}$ makes the right side $2\lfloor x\rfloor + 1$. The
closing insight: $\lfloor x + \frac12\rfloor$ is the
\emph{rounding} of $x$ to the nearest integer, so the identity
says that \omterm{thm:b1:reals:floor}{floor} plus rounding equals \omterm{thm:b1:reals:floor}{floor} of the double --- and
the case split on the fractional part $u$ is the standard
technique behind every \omterm{thm:b1:reals:floor}{floor} identity
(\cref{exo:b1:reals:2,exo:b1:reals:3} run on it too).
\end{example}

\begin{theorem}[Density of $\Q$ and of $\R \setminus \Q$]\label{thm:b1:reals:density}
Between any two reals $x < y$ there lie a rational and an
irrational.\index{density}
\end{theorem}

\begin{proof}
\emph{A rational.} By \cref{thm:b1:reals:archimedes}, pick $n \in
\N^*$ with $n > \frac{1}{y - x}$, so $ny - nx > 1$. Let $m = \lfloor
nx \rfloor + 1$. On one side, $nx < \lfloor nx \rfloor + 1 = m$
(\cref{thm:b1:reals:floor}); on the other, $m = \lfloor nx \rfloor +
1 \leq nx + 1 < ny$. Dividing by $n$: $x < \frac mn < y$.

\emph{An irrational.} Apply the previous point to the pair $x -
\sqrt 2 < y - \sqrt 2$: some rational $q$ lies between them, and then
$q + \sqrt 2 \in \intoo{x}{y}$ is irrational (if $q + \sqrt 2$ were
rational, so would be $\sqrt 2$).
\end{proof}

\begin{example}[Running the proof of density]\label{ex:b1:reals:densityrun}
The proof is an algorithm; let us execute it on $x = 1.414$ and
$y = \sqrt 2$. Since $1.4142^2 = 1.99996164 < 2$, we have $\sqrt 2
> 1.4142$, so $y - x > 0.0002$ and $\frac{1}{y - x} < 5000$: the
choice $n = 5000$ is legitimate. Then $nx = 7070$, so $m = \lfloor
7070 \rfloor + 1 = 7071$, and the rational produced is
\[
\frac{m}{n} = \frac{7071}{5000} = 1.4142,
\qquad
1.414 < 1.4142 < \sqrt 2 .
\]
The closing insight: the proof needs $n$ only slightly larger than
$\frac{1}{y-x}$, and returns the first multiple of $\frac 1n$
beyond $x$. Density is not an abstract miracle --- it is long
division in disguise, a theme developed at length in the weekend
problem (\cref{pb:b1:reals:1}).
\end{example}

\begin{remark}[Where completeness is used next]
\Cref{thm:b1:reals:sup} is the single non-algebraic axiom of this
book, and every existence theorem of analysis is that axiom
wearing different clothes: the monotone convergence theorem
(\cref{ch:b1:seq}), the Bolzano--Weierstrass theorem
(\cref{ch:b1:topology}), the intermediate value and extreme value
theorems (\cref{ch:b1:continuity}), and the very definition of the
integral as a \omterm{def:b1:reals:bounds}{supremum} of lower sums (\cref{ch:b1:integration}).
The Year 3 volume builds measure theory and Hilbert spaces on the
same single axiom. When a proof in the coming chapters produces a
real number out of thin air, look for the hidden \omterm{def:b1:reals:bounds}{supremum}.
\end{remark}

\begin{remark}[Between discreteness and density]
$\Z$ and $\Q$ sit at opposite extremes inside $\R$: around each
integer there is a gap of length $1$ containing no other
(discreteness --- this is what makes the \omterm{thm:b1:reals:floor}{floor} well defined),
while between any two reals there are infinitely many rationals
(density). Remarkably, for \emph{additive \omterm{def:b1:structures:subgroup}{subgroups}} of $\R$
there is nothing in between: \cref{exo:b1:reals:9} proves that
such a \omterm{def:b1:structures:subgroup}{subgroup} is either of the form $\alpha\Z$ (discrete) or
dense --- a dichotomy that powers the density of $\{\sin n\}$ in
\cref{ch:b1:seq} and the constructive monster of
\cref{pb:b1:continuity:1}. General \omterm{def:b1:logic:sets}{sets}, of course, mix the
behaviors freely: $\Z \cup \Q\cap\intcc{0}{1}$ is discrete far
away and dense in the middle.
\end{remark}

\begin{method}[Proving equalities with sup and inf]\label{met:b1:reals:supproofs}
To prove $\sup A = s$: check that $s$ bounds $A$ above, then produce,
for each $\varepsilon > 0$ (or for a sequence $\varepsilon = \frac
1n$), an element of $A$ above $s - \varepsilon$. To compare suprema,
use: $A \subseteq B \implies \sup A \leq \sup B$; and for all $a, b$:
$\sup(A + B) = \sup A + \sup B$, where $A + B = \{a + b\}$
(\cref{exo:b1:reals:5}). Never write $\sup A$ before knowing $A$ is
nonempty and bounded above.
\end{method}

\section{Intervals}

\begin{proposition}[Characterization of intervals]\label{prop:b1:reals:intervals}
A subset $I \subseteq \R$ is an \emph{interval}\index{interval} (one
of the familiar types $\intoo{a}{b}$, $\intcc{a}{b}$, $\intco{a}{b}$,
$\intoc{a}{b}$, half-lines, $\R$, $\emptyset$, singletons) if and
only if it is \emph{convex}:
\[
\forall x, y \in I,\ \forall z \in \R, \quad
x \leq z \leq y \implies z \in I .
\]
\end{proposition}

\begin{proof}
Every listed type is clearly convex. Conversely, let $I$ be convex and
nonempty. Set $a = \inf I$ if $I$ is bounded below, else $a =
-\infty$; likewise $b = \sup I$ or $+\infty$. We claim $\intoo{a}{b}
\subseteq I \subseteq \intcc{a}{b}$ (with obvious conventions at
$\pm\infty$). The second inclusion is the definition of bounds. For
the first, let $z \in \intoo{a}{b}$: since $z > a$, $z$ is not a
lower bound (or $a = -\infty$), so some $x \in I$ has $x < z$;
similarly some $y \in I$ has $y > z$; convexity puts $z \in I$.

It remains to read off the type from the double inclusion
$\intoo{a}{b} \subseteq I \subseteq \intcc{a}{b}$: the \omterm{def:b1:logic:sets}{sets}
squeezed between an open \omterm{prop:b1:reals:intervals}{interval} and its closure differ from
$\intoo{a}{b}$ only by the presence or absence of the (finite)
endpoints. Explicitly: if $a, b \in \R$, the four possibilities
for $(a \in I,\ b \in I)$ give $\intoo{a}{b}$, $\intco{a}{b}$,
$\intoc{a}{b}$, $\intcc{a}{b}$ (including the degenerate cases
$a = b$: singleton if $a \in I$); if $a = -\infty$ and $b \in
\R$, one gets $\intoo{-\infty}{b}$ or $\intoc{-\infty}{b}$;
symmetrically for $a \in \R$, $b = +\infty$; and $a = -\infty$,
$b = +\infty$ gives $I = \R$. Every case is on the list: done.
\end{proof}

\begin{remark}[Why convexity is the right test]
The proposition converts a \emph{geometric} definition (a list
of ten shapes) into a \emph{one-line logical} test, and the test
is what one actually uses: to prove that a \omterm{def:b1:logic:sets}{set} is an \omterm{prop:b1:reals:intervals}{interval},
never chase which of the ten shapes it is --- verify convexity
and let the proposition sort out the type. The intermediate
value theorem of \cref{ch:b1:continuity} will be stated exactly
this way (``the continuous image of an \omterm{prop:b1:reals:intervals}{interval} is an
\omterm{prop:b1:reals:intervals}{interval}''), and its proof produces the convexity, not the
shape.
\end{remark}

\begin{remark}[Extended real line]
It is convenient to adjoin two symbols and work in $\overline\R = \R
\cup \{-\infty, +\infty\}$, with the conventions $\sup A = +\infty$
when $A$ is not bounded above and $\sup \emptyset = -\infty$. Then
\emph{every} subset of $\R$ has a \omterm{def:b1:reals:bounds}{supremum} in $\overline\R$ --- a
notational comfort used freely for limits in \cref{ch:b1:seq}.
\end{remark}

\begin{example}[Computing in $\overline\R$]\label{ex:b1:reals:extendedcompute}
With the conventions in force: $\sup \Z = +\infty$, $\inf \Z =
-\infty$; for $A = \{n + (-1)^n n : n \in \N\} = \{0, 4, 0, 8,
\dots\} \cup \{0\}$, $\sup A = +\infty$ (the even terms $2n$ are
unbounded) and $\inf A = \min A = 0$; and $\sup\emptyset =
-\infty \leq \inf\emptyset = +\infty$ --- the one \omterm{def:b1:logic:sets}{set} whose
\omterm{def:b1:reals:bounds}{supremum} is \emph{smaller} than its \omterm{def:b1:reals:bounds}{infimum}, a reminder that the
conventions are chosen to make $\sup$ increasing and $\inf$
decreasing with respect to inclusion:
\[
A \subseteq B \implies \sup A \leq \sup B
\quad\text{and}\quad \inf A \geq \inf B ,
\]
now valid with no nonemptiness caveat. What the conventions do
\emph{not} provide is arithmetic: $+\infty + (-\infty)$ and $0
\times (+\infty)$ stay undefined, and every algebraic
manipulation of suprema must first check it never forms these.
The extended line is bookkeeping, not a number system.
\end{example}

\begin{example}[The supremum that escaped $\Q$]\label{ex:b1:reals:supescape}
Return to the \omterm{def:b1:logic:sets}{set} of the opening remark, $A = \{x \in \Q : x^2 <
2\}$, and compute its \omterm{def:b1:reals:bounds}{supremum} \emph{in $\R$}. It is nonempty
($1 \in A$) and bounded above by $1.5$ (if $x > 1.5$ then $x^2 >
2.25 > 2$), so $s = \sup A$ exists. We claim $s = \sqrt 2$ (the
real number built in \cref{exo:b1:reals:12}). \omterm{def:b1:reals:bounds}{Upper bound}: every
$a \in A$ satisfies $a < \sqrt2$ --- for $a \leq 0$ this is
clear, and for $a > 0$, $a \geq \sqrt2$ would give $a^2 \geq 2$.
Nothing smaller works: given $t < \sqrt2$, density
(\cref{thm:b1:reals:density}) provides a rational $q$ with
$\max(1, t) < q < \sqrt 2$, and then $q^2 < 2$, so $q \in A$
exceeds $t$. By \cref{prop:b1:reals:epsilon}, $s = \sqrt2
\notin \Q$. The closing insight: the \omterm{def:b1:reals:bounds}{supremum} of a \omterm{def:b1:logic:sets}{set} of
rationals need not be rational --- completeness is precisely the
promise that $\R$, unlike $\Q$, never lets a \omterm{def:b1:reals:bounds}{supremum} escape;
this example is the opening remark of the chapter, now proved
rather than pointed at.
\end{example}

\begin{remark}[Perspectives inside this volume]
The chapter's three tools have distinct careers ahead. The
\omterm{def:b1:reals:bounds}{supremum} runs the analysis half: monotone limits
(\cref{ch:b1:seq}), the integral's very definition
(\cref{ch:b1:integration}), and, in the geometry of
\cref{ch:b1:euclid}, the distance from a point to a subspace ---
an \omterm{def:b1:reals:bounds}{infimum} that orthogonal projection turns into a minimum. The
\omterm{thm:b1:reals:floor}{floor function} returns wherever the discrete meets the
continuous: digit expansions (this chapter's weekend problem),
Dirichlet's pigeonhole approximation
(\cref{pb:b1:derivative:1}), integral comparisons of sums
(\cref{ch:b1:series}). Density arguments upgrade into a method
in \cref{ch:b1:continuity}: an identity of continuous functions
needs checking only on $\Q$ --- half of Cauchy's functional
equation (\cref{pb:b1:continuity:1}) is exactly that move. When
in doubt about where a proof in this volume gets its existence
\omterm{def:b1:logic:statement}{statements}, the answer is nearly always: this chapter.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:reals:1}
Determine (with proofs) sup, inf, max, min --- when they exist --- of:
\[
A = \Bigl\{\frac{1}{n} : n \in \N^*\Bigr\},
\qquad
B = \Bigl\{\frac{(-1)^n n}{n+1} : n \in \N\Bigr\},
\qquad
C = \{x \in \R : x^2 < 3\}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:reals:2}
Prove that for all $x, y \in \R$: $\lfloor x \rfloor + \lfloor y
\rfloor \leq \lfloor x + y \rfloor \leq \lfloor x \rfloor + \lfloor y
\rfloor + 1$, and that both bounds are attained.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:reals:3}
Prove that for every $x \in \R$ and $n \in \N^*$:
$\Bigl\lfloor \frac{\lfloor nx \rfloor}{n} \Bigr\rfloor = \lfloor x
\rfloor$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:reals:4}
Let $A \subseteq B$ be nonempty subsets of $\R$, $B$ bounded. Prove
$\inf B \leq \inf A \leq \sup A \leq \sup B$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:reals:5}
For nonempty bounded $A, B \subseteq \R$, define $A + B = \{a + b : a
\in A,\ b \in B\}$ and $-A = \{-a : a \in A\}$. Prove:
\[
\sup(A + B) = \sup A + \sup B,
\qquad
\sup(-A) = -\inf A .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:reals:6}
Let $f, g \colon E \to \R$ be bounded functions. Prove
\[
\sup_{x \in E}\, \bigl(f(x) + g(x)\bigr) \leq \sup_{x \in E} f(x) +
\sup_{x \in E} g(x),
\]
and give an example where the inequality is strict. Why does this not
contradict \cref{exo:b1:reals:5}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:reals:7}
Prove that $\sqrt 2 + \sqrt 3$ is irrational. \emph{(Square it and
use the irrationality of $\sqrt 6$, to be proved via
\cref{exo:b1:arith:7}.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:reals:8}
Prove that the \omterm{def:b1:logic:sets}{set} $D = \bigl\{\frac{m}{2^n} : m \in \Z,\ n \in
\N\bigr\}$ of dyadic rationals is dense in $\R$: between any two
reals lies a dyadic rational.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:reals:9}
Let $G$ be a \omterm{def:b1:structures:subgroup}{subgroup} of $(\R, +)$ with $G \neq \{0\}$. Set $\alpha =
\inf\,(G \cap \intoo{0}{+\infty})$. Prove:
\begin{enumerate}
  \item if $\alpha > 0$, then $G = \alpha\Z$;
  \item if $\alpha = 0$, then $G$ is dense in $\R$.
\end{enumerate}
Deduce that $\Z + \sqrt 2\,\Z$ is dense in $\R$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:reals:10}
For $A, B$ nonempty \omterm{def:b1:logic:sets}{sets} of positive reals, let $AB = \{ab : a \in
A, b \in B\}$. Prove $\sup(AB) = \sup A \cdot \sup B$ (bounded case),
and show by example that positivity is essential.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:reals:11}
For a nonempty bounded $A \subseteq \R$, define the
\emph{diameter}
\[
\operatorname{diam} A = \sup\,\{\abs{a - a'} : a, a' \in A\} .
\]
Prove that $\operatorname{diam} A = \sup A - \inf A$, and that
$\intcc{\inf A}{\sup A}$ is the smallest closed \omterm{prop:b1:reals:intervals}{interval}
containing $A$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:reals:12}
Let $y > 0$ and $E = \{x \geq 0 : x^2 \leq y\}$. Prove that $E$ is
nonempty and bounded above, and that $s = \sup E$ satisfies $s^2 =
y$ \emph{(rule out $s^2 < y$ and $s^2 > y$ by exhibiting, in each
case, a small $h > 0$ contradicting the definition of the
\omterm{def:b1:reals:bounds}{supremum})}. Deduce that every $y > 0$ has a unique square root
$\sqrt y > 0$ and that $y \mapsto \sqrt y$ is increasing on
$\intoo{0}{+\infty}$.
\end{exercise}

\section{Problem: Digit expansions and the rhythm of the rationals}

\begin{problem}[{Weekend problem --- $b$-adic expansions:
existence, uniqueness, and periodicity characterizes $\Q$}]
\label{pb:b1:reals:1}
Every real in $\intco{0}{1}$ has a digit expansion in every base
$b \geq 2$; the expansion is unique once trailing strings of the
digit $b - 1$ are outlawed; and it is eventually periodic exactly
when the number is rational. This problem proves all three facts
from the completeness axiom alone --- no sequences, no series:
only the \omterm{def:b1:reals:bounds}{supremum}, the \omterm{thm:b1:reals:archimedes}{Archimedean property} and the \omterm{thm:b1:reals:floor}{floor function}
--- and closes with Cantor's diagonal argument in digit form.
Throughout, $b \geq 2$ is a fixed integer (the \emph{base}), a
\emph{digit} is an element of $\intint{0}{b-1}$, and a digit
string $(d_n)_{n \geq 1}$ is \emph{proper} when it is not
eventually equal to $b - 1$ (that is: for every $N$ there is $n >
N$ with $d_n \leq b - 2$).\index{b-adic expansion@$b$-adic
expansion}\index{decimal expansion}

\medskip
\noindent\textbf{Part I --- Digits by hand.} Long division of $p$
by $q$ in base $b$: multiply the current remainder by $b$, divide
by $q$, record the quotient as the next digit, keep the remainder.
\begin{enumerate}
  \item In base $10$, run the algorithm on $\frac 18$ and on
        $\frac 17$, recording at each step the digit \emph{and}
        the remainder. Check that the remainders for $\frac 17$
        cycle through $1, 3, 2, 6, 4, 5$ and that the digits
        $142857$ then repeat forever.
  \item Compute the base-$2$ expansions of $\frac 13$ and of
        $\frac{5}{16}$, and the base-$3$ expansion of $\frac 12$.
        Observe: one number terminates, the two others repeat ---
        and $\frac 12$, so tame in base $10$, repeats forever in
        base $3$.
  \item For $x = \frac pq \in \intco{0}{1}$ in lowest terms, show
        that the digits produced by the algorithm are eventually
        all $0$ if and only if the remainder $b^N p \bmod q$
        vanishes for some $N$, if and only if $q$ \omterm{def:b1:arith:divides}{divides} some
        power $b^N$, if and only if every \omterm{def:b1:arith:prime}{prime} factor of $q$
        \omterm{def:b1:arith:divides}{divides} $b$. Check: $\frac{1}{20}$ terminates in base
        $10$, not in base $3$.
  \item Define the \emph{truncation} $s_n = \lfloor b^n x \rfloor
        / b^n$. For $x = \sqrt 2$ and $b = 10$, compute $s_0,
        \dots, s_4$ by verifying at each step that two
        consecutive squares straddle $2$ (for instance $1.4142^2
        = 1.99996164 < 2 < 2.00024449 = 1.4143^2$), and check
        $s_n \leq \sqrt 2 < s_n + 10^{-n}$ each time.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Existence, from the \omterm{def:b1:reals:bounds}{supremum}.} Fix
$x \in \intco{0}{1}$ and set $A_n = \lfloor b^n x \rfloor$ and
$d_n = A_n - b\,A_{n-1}$ for $n \geq 1$.
\begin{enumerate}[resume]
  \item Show $A_0 = 0$ and $b\,A_{n-1} \leq A_n \leq b\,A_{n-1} +
        b - 1$; conclude that each $d_n$ is a digit.
  \item Show that $s_n := A_n b^{-n}$ satisfies
        \[
        s_n = \sum_{k=1}^{n} d_k\,b^{-k}
        \qquad\text{and}\qquad
        s_n \leq x < s_n + b^{-n} .
        \]
  \item Prove $b^n \geq n + 1$ by induction, then show that
        $(s_n)$ is nondecreasing and that $x = \sup_n s_n$
        \emph{(use \cref{prop:b1:reals:epsilon} and
        \cref{thm:b1:reals:archimedes})}.
  \item Show that the string $(d_n)$ is proper: if $d_k = b - 1$
        for all $k > N$, compute $s_n$ for $n > N$ by a finite
        geometric sum and contradict question 6.
  \item Conversely, let $(e_n)_{n \geq 1}$ be any proper digit
        string and $t_n = \sum_{k=1}^n e_k b^{-k}$. Show that $y
        = \sup_n t_n$ exists, lies in $\intco{0}{1}$, and
        satisfies $t_n \leq y < t_n + b^{-n}$ for every $n$
        \emph{(for the strict inequality, use a digit $e_m \leq b
        - 2$ with $m > n$)}. Deduce $\lfloor b^n y \rfloor = b^n
        t_n$, then that the digits of $y$, in the sense of
        question 5, are exactly the $e_n$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Uniqueness, order, shift.}
\begin{enumerate}[resume]
  \item Assemble questions 5--9 into the \emph{$b$-adic expansion
        theorem}: the \omterm{def:b1:logic:map}{maps} $x \mapsto (d_n)$ and $(e_n) \mapsto
        \sup_n t_n$ are mutually inverse bijections between
        $\intco{0}{1}$ and the \omterm{def:b1:logic:sets}{set} of proper digit strings. In
        particular no two distinct proper strings have the same
        value.
  \item Now allow improper strings. Show that a string with $e_n
        = b - 1$ for all $n > M$ (with $M \geq 0$ minimal) has
        value $t_M + b^{-M}$; conclude that $0.999\dots = 1$ in
        base $10$, and that the reals with two digit
        representations are exactly the $b$-adic fractions
        $m/b^N \in \intoo{0}{1}$ --- every other real has just
        one, even among improper strings.
  \item Prove that the bijection of question 10 is
        order-preserving for the lexicographic order: if the
        proper strings of $x$ and $y$ first differ at index $m$,
        then $x < y$ if and only if $d_m < e_m$.
  \item (Shift lemma) Let $x \in \intco{0}{1}$ have digits
        $(d_n)$. Show that the fractional part of $bx$ has digits
        $(d_{n+1})_{n \geq 1}$ \emph{(compute $\lfloor b^n(bx -
        A_1)\rfloor$ using $\lfloor u - K \rfloor = \lfloor u
        \rfloor - K$ for integer $K$)}, and deduce by induction
        that the fractional part of $b^m x$ has digits
        $(d_{n+m})_{n \geq 1}$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Rationality is periodicity.} Let $x
= \frac pq \in \intco{0}{1}$ be in lowest terms and $r_n = b^n p
\bmod q$ the remainder of the Euclidean division of $b^n p$ by
$q$.
\begin{enumerate}[resume]
  \item Show $A_n = \dfrac{b^n p - r_n}{q}$ and $r_n =
        (b\,r_{n-1}) \bmod q$.
  \item Show $d_n = \Bigl\lfloor \dfrac{b\,r_{n-1}}{q}
        \Bigr\rfloor$: each digit is a function of the previous
        remainder alone. This is exactly the long division of
        Part I.
  \item Apply the pigeonhole principle
        (\cref{cor:b1:counting:pigeonhole}) to $r_0, \dots, r_q$
        and conclude: the expansion of every rational is
        eventually periodic, with preperiod and period at most
        $q$.
  \item Conversely, suppose the digits of $y \in \intco{0}{1}$
        are \emph{purely} periodic: $d_{n+T} = d_n$ for all $n
        \geq 1$. Using the shift lemma and the uniqueness of
        question 10, show that the fractional part of $b^T y$
        equals $y$, and deduce $(b^T - 1)\,y \in \N$: thus $y$ is
        rational with denominator dividing $b^T - 1$. Verify the
        mechanism on $0.(142857)$: $142857 \times 7 = 999999$.
  \item Treat the eventually periodic case by shifting, and state
        the \emph{periodicity criterion}: $x \in \intco{0}{1}$ is
        rational if and only if its proper $b$-adic expansion is
        eventually periodic --- in one base if and only if in
        all.
  \item For $x = \frac 1q$ with $\gcd(q, b) = 1$, show that the
        expansion is purely periodic and that its least period is
        the least $T \geq 1$ with $b^T \equiv 1 \pmod q$ (the
        multiplicative order of $b$ modulo $q$). Check that for
        $q = 7$, $b = 10$ the powers of $10$ modulo $7$ run
        through $3, 2, 6, 4, 5, 1$: order $6$, matching question
        1.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends and the diagonal.}
\begin{enumerate}[resume]
  \item Let $x^*$ be the real of $\intco{0}{1}$ whose base-$10$
        digits are $1$ at the triangular positions $\frac{j(j +
        1)}{2}$ ($j \geq 1$) and $0$ elsewhere: $x^* =
        0.101001000100001\dots$ Show that its digit string is
        proper but not eventually periodic \emph{(a period $T$
        would force ones at gaps at most $T$, but the gaps grow)},
        and conclude that $x^*$ is irrational: a number proved
        irrational by pure rhythm.
  \item Show that for every base $b$ the \omterm{def:b1:logic:sets}{set} $\{m/b^n : m \in \Z,
        n \in \N\}$ is dense in $\R$ (generalizing
        \cref{exo:b1:reals:8}), and that every rational $\frac pq
        \in \intoo{0}{1}$ has a \emph{terminating} expansion in
        base $q$. Moral: terminating is a property of the pair
        (number, base); periodicity --- rationality --- is
        intrinsic.
  \item (Cantor's diagonal) Let $k \mapsto x_k$ be any \omterm{def:b1:logic:map}{map} from
        $\N^*$ to $\intco{0}{1}$. Define the digit string $e_k =
        1$ if the $k$-th digit of $x_k$ differs from $1$, and
        $e_k = 2$ otherwise. Show that $(e_k)$ is proper, that
        its value $y$ lies in $\intco{0}{1}$, and that $y \neq
        x_k$ for every $k$. Conclude: no \omterm{def:b1:logic:map}{map} $\N^* \to
        \intco{0}{1}$ is \omterm{def:b1:logic:inj}{surjective}. (The vocabulary of
        countability, and this theorem's proper home, is
        \cref{ch:b1:topology}.)
  \item Show that if the proper expansions of $x$ and $y$ agree
        up to index $n$ then $\abs{x - y} < b^{-n}$, and disprove
        the converse with $x = 0.1$, $y = 0.0999$ in base $10$:
        closeness of numbers does not force agreement of digits.
        Which reals are to blame?
  \item Run Part IV on $x = \frac{1}{10}$ in base $b = 2$:
        compute remainders and digits until they cycle, and
        conclude $\frac{1}{10} = (0.0\overline{0011})_2$, with
        preperiod $1$ and period $4$. Explain via question 3 why
        no finite binary string will ever equal $\frac{1}{10}$
        --- the reason a computer's floating-point $0.1 + 0.2$ is
        not exactly $0.3$.
  \item Synthesis. In one sentence each: where did the proof use
        (i) completeness, (ii) the \omterm{thm:b1:reals:archimedes}{Archimedean property}, (iii)
        the uniqueness clause of the \omterm{thm:b1:reals:floor}{floor}, (iv) the pigeonhole
        principle? And the moral: $\intco{0}{1}$ is faithfully
        coded by proper digit strings, rationality reads off as
        periodicity --- yet analysis prefers the \omterm{def:b1:reals:bounds}{supremum} to the
        digits. Why? (Think about adding two digit strings.)
\end{enumerate}
\end{problem}
