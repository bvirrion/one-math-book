\chapter{Curvas planas}\label{ch:b1:curves}

El cálculo de \cref{ch:b1:derivative,ch:b1:taylor} fue construido para
funciones $ y = f(x)$; la mayoría de las curvas de geometría y mecánica ---
trayectorias, círculos rodados sobre círculos, órbitas --- rechaza esa forma
y vienen en cambio como \emph{parametrizado} curvas $t \mapsto (x(t),
y(t))$ o como \emph{polar} curvas $r = r(\theta)$. este clasico
El capítulo estudia ambos y cierra con el \omterm{def:b1:curves:conics}{cónicas}.

\section{Curvas parametrizadas}

\begin{definition}\label{def:b1:curves:def}
Un \emph{curva parametrizada}\index{curva parametrizada} es una \omterm{def:b1:logic:map}{aplicación} $f
\colon I \to \R^2$,$t \mapsto M(t) = (x(t), y(t))$, con $x, y $ de
clase $C^1$(al menos) en el \omterm{prop:b1:reals:intervals}{intervalo}$I$. El \emph{velocidad
vector} es $f'(t) = (x'(t), y'(t))$; el punto $M(t_0)$ es
\emph{regular} cuando $f'(t_0) \neq (0,0)$ y \emph{tangente}
ahí está la línea que pasa por $M(t_0)$ dirigida por $f'(t_0)$.
\end{definition}

\begin{proof}[Por qué $f'$ dirige la tangente]
Por Taylor--Young por componentes, $M(t) = M(t_0) + (t - t_0) f'(t_0) +
o(t - t_0)$: la dirección del acorde $\frac{M(t) - M(t_0)}{t - t_0}$
tiende a $f'(t_0)$. La hipótesis de la regularidad es lo que hace que la
limitar a \emph{dirección}: si $f'(t_0) = 0$, la pantalla colapsa
a $M(t) = M(t_0) + o(t - t_0)$ y no dice nada sobre cómo
La curva abandona el punto --- de donde el tratamiento separado de
puntos singulares en el método siguiente, donde el primero
\emph{no desaparecer} \omterm{def:b1:derivative:def}{derivado} asume el papel de $f'$.
Tenga en cuenta también que la recta tangente es un objeto geométrico: cualquier
La reparametrización cambia $f'$ por un factor escalar distinto de cero y
deja la línea sin cambios.
\end{proof}

\begin{method}[Estudiando una curva parametrizada]\label{met:b1:curves:study}
\begin{enumerate}
 \item \emph{Reducir el dominio} usando simetrías: relaciones entre
 $M(-t)$,$M(t + T)$, \dots\ y $M(t)$(reflexiones,
 traducciones, periodicidad), luego dibuje solo la parte reducida.
 \item \emph{Variaciones}: tabular los signos de $x'$ y $ y'$ de forma conjunta;
 la curva se mueve hacia la derecha/izquierda como $x'$, arriba/abajo como $ y'$.
 \item \emph{Puntos destacables}: tangente horizontal ($ y' = 0 \neq
 x'$), tangente vertical ($x' = 0 \neq y'$); en un
 Punto \emph{singular} ($f' = 0$), examinar \omterm{def:b1:derivative:def}{derivados} superior
 para la dirección tangente.
 \item \emph{Comportamiento asintótico} en los extremos de $I$, luego dibuje.
\end{enumerate}
\end{method}

\begin{example}[Ramas asintóticas, trabajadas]
\label{ex:b1:curves:asymptotes}
$x(t) = t$,$ y(t) = t + \dfrac1t$ en $\intoo{0}{+\infty}$. Como $t
\to 0^{+}$:$x \to 0$ mientras que $ y \to +\infty $ --- la curva sube
a lo largo del \emph{asíntota vertical} $x = 0$(límite finito para
una coordenada, infinita para la otra). Como $t \to +\infty$:
ambos \omterm{prop:b1:vspaces:coordinates}{coordenadas} explotan, así que prueba una línea: la diferencia
\[
y(t) - x(t) = \frac1t \longrightarrow 0^{+}
\]
exhibe el \emph{asíntota oblicua} $ y = x$, abordado desde
arriba. Mientras tanto, $ y' = 1 - \frac1{t^2}$ desaparece en $t = 1$:
el punto $(1, 2)$ es el punto bajo de la rama, y $x' = 1 >
0$ en todas partes, por lo que la curva siempre avanza hacia la derecha. un arco,
dos asíntotas, una mínima: una imagen completa de tres
cálculos --- con la receta general visible debajo:
cuando $x, y \to \infty$ estén juntos, examine $ y/x$ en busca de un candidato
pendiente (aquí $\to 1$), luego $ y - (\text{slope})\,x$ para el
interceptación y el lado de aproximación.
\end{example}

\begin{example}[El astroide]\label{ex:b1:curves:astroid}
$x(t) = \cos^3 t$,$ y(t) = \sin^3 t$.\index{astroide} Simetrías:
$M(t + 2\pi) = M(t)$(estudiar un período);$M(-t)$ es el reflejo de
$M(t)$ en el eje $x$;$M(\pi - t)$ en el eje$ y $;$M(\frac\pi2 -
t)$ en la diagonal$ y = x $: basta estudiar $t \in
\intcc{0}{\frac\pi4}$ y desplegar.

Velocidad: $f'(t) = 3\sin t\cos t\,(-\cos t, \sin t)$. encendido
$\intoo{0}{\frac\pi2}$ todos los puntos son regulares con dirección tangente
por $(-\cos t, \sin t)$; en $t = 0$(el punto $(1,0)$) la velocidad
desaparece: a \emph{cúspide}\index{cúspide}, donde la curva se invierte a lo largo
la dirección tangente $(-1, 0)\cdot(\pm)$ --- por simetría los cuatro
\omterm{ex:b1:curves:astroid}{cúspides} siéntate en $(\pm1, 0), (0, \pm1)$.
\end{example}

\begin{example}[Un ocho, estudiado en su totalidad]
\label{ex:b1:curves:lissajous}
$x(t) = \sin t$,$ y(t) = \sin 2t $(un \emph{curva de lissajous}).
\emph{Simetrías}: $M(t + \pi) = (-x(t), y(t))$(reflejo en
el eje $ y $),$M(-t) = (-x(t), -y(t))$(simetría central),
$M(\pi - t) = (x(t), -y(t))$(reflexión en el eje $x$):
basta con estudiar $t \in \intcc{0}{\frac\pi2}$ y desarrollarlo.
\emph{Variaciones}: $x' = \cos t \geq 0$ en todo momento, mientras que $ y' =
2\cos 2t $ es positivo antes de $t = \frac\pi4$ y negativo después:
el arco sube hacia la derecha hasta la cumbre
$\bigl(\frac{\sqrt2}2,\ 1\bigr)$ y $t = \frac\pi4$(horizontal
tangente), luego desciende hacia la derecha hasta $(1, 0)$ en $t =
\frac\pi2$, donde $x' = 0 \neq y'$: tangente vertical.
\emph{punto doble}: $M(0) = M(\pi) = (0,0)$, con dos diferentes
velocidades
\[
f'(0) = (1,\ 2), \qquad f'(\pi) = (-1,\ 2) :
\]
dos ramas \emph{regular} que se cruzan en el origen en puntos distintos
ángulos --- un punto doble, no un punto singular: cada pasaje es
Perfectamente suave, los dos pasajes simplemente comparten su ubicación.
La curva completa es la figura del ocho que aparece a continuación.
\end{example}

\section{Curvas polares}

\begin{definition}\label{def:b1:curves:polar}
A \emph{curva polar}\index{curva polar} viene dado por $r = r(\theta)$:
el punto del parámetro $\theta$ es
\[
M(\theta) = r(\theta)\,\vec u(\theta),
\qquad \vec u(\theta) = (\cos\theta, \sin\theta).
\]
Con $\vec v(\theta) = (-\sin\theta, \cos\theta) = \vec
u\,'(\theta)$, la velocidad es
\[
M'(\theta) = r'(\theta)\, \vec u(\theta) + r(\theta)\, \vec
v(\theta) .
\]
Consecuencias: donde $r \neq 0$, el punto es regular, y el
la tangente forma con el rayo el ángulo $V$ dado por $\tan V =
\frac{r}{r'}$(ángulo entre $M'$ y $\vec u $); donde $r(\theta_0)
= 0$, la curva pasa por el origen \emph{with tangent the ray
$\theta = \theta_0$} (dirección $\vec u(\theta_0)$, lectura
$M'(\theta_0) = r'(\theta_0)\vec u(\theta_0)$, o desde el límite
acorde: el acorde de $O$a$M(\theta)$ lo lleva el propio $\vec
u(\theta)$, que tiende a$\vec u(\theta_0)$ --- por lo que el
La regla se mantiene incluso cuando $r'(\theta_0) = 0$ y el punto es
singular, por lo que el polar \omterm{ex:b1:curves:astroid}{cúspides} en el origen, al igual que el
cardioides, obtén su tangente libre).
\end{definition}

\begin{example}[El cardioide]\label{ex:b1:curves:cardioid}
$r(\theta) = 1 + \cos\theta$.\index{cardioide} Simetría: $r(-\theta)
= r(\theta)$: reflexión en el eje $x $; estudio $\theta \in
\intcc{0}{\pi}$.$r$ disminuye de $2$a$0$; en $\theta = \pi $,
$r = 0$: la curva llega al origen tangencialmente al rayo
$\theta = \pi $(el eje $x$), formando un \omterm{ex:b1:curves:astroid}{cúspide} allí --- el corazón
punto. Tangente en $\theta = 0$:$r' = 0$, entonces $\tan V = \infty $:
perpendicular al eje.

El ángulo $V$ merece una lectura más. En$\theta =
\frac\pi2$:$r = 1$ y $ r' = -1$, entonces $\tan V = \frac{r}{r'} =
-1$: la tangente forma tres cuartos de ángulo recto con la
rayo saliente --- la curva ya se está doblando hacia su
\omterm{ex:b1:curves:astroid}{cúspide}. La fórmula $\tan V = r/r'$ proporciona direcciones tangentes
a lo largo de toda la curva sin cálculo de $M'(\theta)$
en absoluto: es el análogo polar de leer una pendiente.
\end{example}

\begin{example}[Polar a cartesiano: un círculo oculto]
\label{ex:b1:curves:polarcircle}
¿Qué es el \omterm{def:b1:curves:polar}{curva polar} $r = 2\cos\theta$? Multiplicar por $r$:
$r^2 = 2r\cos\theta$, es decir $x^2 + y^2 = 2x$, es decir
\[
(x - 1)^2 + y^2 = 1 :
\]
la circunferencia de centro $(1, 0)$ y radio $1$, que pasa por
el origen. La contabilidad importa: como se desborda $\theta$
$\intoc{-\frac\pi2}{\frac\pi2}$ se barre todo el círculo
exactamente una vez ($r$ desaparece en ambos extremos), y en $\theta =
\pm\frac\pi2$ la regla ``tangente al origen a lo largo del rayo
$\theta = \theta_0$'' da una tangente \emph{vertical} allí ---
coincidiendo con la geometría, ya que el eje vertical es de hecho tangente
a este círculo en $O$. Para $\theta$ más allá de ese rango,$r < 0$
vuelve sobre el mismo círculo: un recordatorio de que un \omterm{def:b1:curves:polar}{curva polar} es un
Objeto \emph{parametrizado}, permitido pasar sobre sí mismo.
\end{example}

\section{Longitud del arco}

\begin{definition}[longitud de un arco]\label{def:b1:curves:arclength}
El \emph{longitud}\index{longitud del arco} de un arco $C^1$ $f \colon
\intcc{a}{b} \to \R^2$ es el \omterm{thm:b1:integration:def}{integral} de la velocidad:
\[
L = \int_a^b \norm{f'(t)}\,\dd t
= \int_a^b \sqrt{x'(t)^2 + y'(t)^2}\;\dd t .
\]
(Motivación: en un pequeño \omterm{prop:b1:reals:intervals}{intervalo}, $M(t + h) \approx M(t) + h
f'(t)$, entonces el arco está cerca de un polígono cuyas longitudes de segmento suman
a una suma de Riemann de $\norm{f'}$,
\cref{thm:b1:integration:riemann}.) Para un \omterm{def:b1:curves:polar}{curva polar} $r =
r(\theta)$, la velocidad $r'\vec u + r\vec v $ tiene \omterm{def:b1:euclid:orthogonal}{ortogonal}
componentes, entonces
\[
L = \int_{\theta_1}^{\theta_2}
\sqrt{r'(\theta)^2 + r(\theta)^2}\;\dd\theta .
\]
La longitud no depende del (monótono, $C^1$)
parametrización elegida: sustituyendo $t = \varphi(s)$ en el
\omterm{thm:b1:integration:def}{integral} (\cref{thm:b1:integration:parts}) multiplica $f'$ por
$\varphi'$ y $\dd t$ por $\varphi'^{-1}$.
\end{definition}

\begin{example}[Comprobación de cordura: el círculo]\label{ex:b1:curves:circlelength}
Para $f(t) = (R\cos t, R\sin t)$ en $\intcc{0}{2\pi}$ se tiene
$\norm{f'} = R$, luego $L = 2\pi R$ --- la definición recupera la
circunferencia.
Prueba de reparametrización: la \omterm{def:b1:logic:map}{aplicación}
$g(t) = (R\cos 2t, R\sin 2t)$ en $\intcc{0}{\pi}$ dibuja el
\emph{mismo} círculo al doble de velocidad $\norm{g'} = 2R$, y
\[
\int_0^{\pi} 2R\,\dd t = 2\pi R
\]
de nuevo: la mitad del tiempo, el doble de velocidad, mismo \omterm{def:b1:curves:arclength}{longitud} --- el
invariancia prometida en la definición, observada una vez en los números.
(Ejecutar $g$ en todo $\intcc{0}{2\pi}$ daría $4\pi R$: a
la curva recorrida dos veces tiene el doble de longitud \emph{como un viaje};
\omterm{def:b1:curves:arclength}{longitud} mide la ruta parametrizada y la contabilidad honesta de
el \omterm{prop:b1:reals:intervals}{intervalo} es parte del cálculo.)
Para el astroide $(\cos^3t, \sin^3t)$:$\norm{f'} = 3\abs{\sin
t\cos t} = \tfrac32\abs{\sin 2t}$, y por simetría $L = 4\int_0^{\pi/2}
\tfrac32\sin 2t\,\dd t = 6$: una curva trazada dentro del círculo unitario,
de \omterm{def:b1:curves:arclength}{longitud} $6 < 2\pi$. El problema del fin de semana mide a los más famosos
arco de todos.
\end{example}

\section{Cónicas}

\begin{definition}[Enfoque - definición de directora]\label{def:b1:curves:conics}
Se fija un punto $F$(\emph{enfocar}), una línea $D$ que no pasa por $F$
(\emph{directora}) y $e > 0$(\emph{excentricidad}). el
\emph{cónico}\index{cónico} de estos datos es
\[
\mathcal{C} = \{M : d(M, F) = e\; d(M, D)\}:
\]
un \emph{elipse} para $e < 1$, un \emph{parábola} para $e = 1$, un
\emph{hipérbola} para $e > 1$. (El círculo aparece como un degenerado
límite $e \to 0$.)
\end{definition}

\begin{example}[La definición, comprobada en una parábola.]
\label{ex:b1:curves:parabolacheck}
Tome la parábola $ y^2 = 4x$, es decir $2p = 4$: enfoque $F = (1, 0)$
y directriz $D : x = -1$(la forma reducida
\cref{thm:b1:curves:reduced} los pone en $\pm\frac p2$). en el
punto $M = (1, 2)$ de la curva:
\[
MF = \sqrt{(1-1)^2 + 2^2} = 2,
\qquad
d(M, D) = 1 - (-1) = 2 :
\]
iguales, como exige $e = 1$. En$M' = (4, 4)$:$MF' = \sqrt{9 +
16} = 5$ y $ d(M', D) = 5$ nuevamente. El foco--directriz
La definición no es una abstracción: es un par de distancias.
puede medir en cualquier punto, y el álgebra de la reducción
ecuaciones no es más que esta medición hecha de una vez por todas
todos.
\end{example}

\begin{theorem}[Ecuaciones reducidas]\label{thm:b1:curves:reduced}
En un marco \omterm{def:b1:euclid:orthogonal}{ortonormal} bien elegido:
\[
\text{ellipse: } \frac{x^2}{a^2} + \frac{y^2}{b^2} = 1
\quad (a \geq b > 0),
\qquad
\text{hyperbola: } \frac{x^2}{a^2} - \frac{y^2}{b^2} = 1,
\qquad
\text{parabola: } y^2 = 2px,
\]
con, para la elipse: focos en $(\pm c, 0)$,$c = \sqrt{a^2 - b^2}$,
$e = \frac ca$, y la caracterización bifocal $MF + MF' = 2a$;
para la hipérbola: $c = \sqrt{a^2 + b^2}$,$e = \frac ca $,
$\abs{MF - MF'} = 2a$, asíntotas$ y = \pm\frac ba x$.
\end{theorem}

\begin{proof}
Tome el foco en el origen y la directriz vertical, $x = -h$
($h > 0$): la condición $MF^2 = e^2 d(M, D)^2$ dice $x^2 + y^2 =
e^2 (x + h)^2$. Para $e = 1$:$ y^2 = 2hx + h^2$, una parábola después de la
cambie $x \mapsto x - \frac h2$(entonces $p = h$). Para $e \neq 1$:
completando el cuadrado en $x$,
\[
(1 - e^2)\Bigl(x - \frac{e^2 h}{1 - e^2}\Bigr)^{\!2} + y^2
= e^2h^2 + \frac{e^4 h^2}{1 - e^2}
= \frac{e^2 h^2}{1 - e^2} .
\]
Establezca $X = x - \frac{e^2h}{1-e^2}$(un cambio de fotograma). Cuando $e < 1$,
dividir por el lado derecho positivo:
$\frac{X^2}{a^2} + \frac{y^2}{b^2} = 1$ con $a = \frac{eh}{1 -
e^2}$,$b = \frac{eh}{\sqrt{1-e^2}}$; cuando $e > 1$, ambos lados de
el signo de giro de pantalla apropiadamente y la misma división da
$\frac{X^2}{a^2} - \frac{y^2}{b^2} = 1$ con $a = \frac{eh}{e^2 -
1}$,$b = \frac{eh}{\sqrt{e^2 - 1}}$. Los valores indicados de $c $
seguir ($c^2 = a^2 - b^2$ o $a^2 + b^2$ da $c = ea $ en ambos
casos, colocando el foco correctamente), y las propiedades bifocales son
verificaciones directas sobre las ecuaciones reducidas. En detalle para el
elipse, con $F' = (c, 0)$ y $M = (X, y)$ en la curva:
\[
MF'^2 = (X - c)^2 + y^2
= X^2 - 2cX + c^2 + b^2\Bigl(1 - \frac{X^2}{a^2}\Bigr)
= \frac{c^2}{a^2}X^2 - 2cX + a^2
= (a - eX)^2 ,
\]
utilizando $c^2 + b^2 = a^2$ y $e = \frac ca $; desde $\abs X \leq
a $ y $e < 1$,$a - eX > 0$, entonces $MF' = a - eX $ sin cuadrado
raíz alguna vez extraída. El cálculo del espejo da $MF = a + eX$,
de donde $MF + MF' = 2a$, constante. Para la hipérbola lo mismo
el álgebra produce $MF = \abs{a + eX}$ y $MF' = \abs{eX - a}$,
con diferencia $\pm2a$ según la sucursal.
\end{proof}

\begin{example}\label{ex:b1:curves:conicexample}
$\frac{x^2}{25} + \frac{y^2}{9} = 1$: elipse,$a = 5$,$b = 3$,$c
= 4$: focos $(\pm4, 0)$, excentricidad $\frac45$. Su parametrización:
$(5\cos t, 3\sin t)$ --- un círculo estirado anisotrópicamente; el
La suma de distancias a los focos de cualquiera de sus puntos es $10$.
\end{example}

\begin{example}[Leyendo una órbita a partir de su ecuación polar]
\label{ex:b1:curves:orbit}
El polar \omterm{def:b1:curves:conics}{cónico} $r = \dfrac{1}{1 + \frac12\cos\theta}$

\begin{remark}[Errores comunes]
\label{rem:b1:curves:pitfalls}
\emph{Los puntos dobles no son puntos singulares.}: en un
autocruzado, cada rama es regular; "singular" se refiere a
$f'(t_0) = 0$ para un valor del parámetro
(\cref{ex:b1:curves:lissajous} vs \omterm{ex:b1:curves:astroid}{cúspides} del astroide).
\emph{Vertical tangent versus \omterm{ex:b1:curves:astroid}{cúspide}}: $x'(t_0) = 0 \neq y'(t_0)$
es un punto regular con tangente vertical; solo $x' = y' = 0$
exige la expansión de orden superior.
\emph{Trazado de radios negativos en el rayo opuesto}: para $r(\theta) <
0$ el punto es $-\abs{r}\,\vec u(\theta)$, en el ángulo $\theta +
\pi $(\cref{exo:b1:curves:5}); olvidar esto pierde bucles internos
o dobla curvas.
\emph{\omterm{def:b1:curves:arclength}{Longitud del arco} integrates $\norm{f'}$, not $f'$}: dividir el
\omterm{thm:b1:integration:def}{integral} en los ceros de la velocidad --- para el astroide,
integrando $\frac32\sin 2t$ durante un período completo sin absoluta
Los valores dan $0$, no $6$.
\emph{\omterm{def:b1:curves:conics}{Cónicas} must be reduced before being read}: en $9x^2 +
25y^2 - 36x - 50y - 164 = 0$, ni los ejes, ni el centro,
ni la excentricidad es visible hasta que se completan los cuadrados
(\cref{exo:b1:curves:6}); y una parte cuadrática fugaz en uno
variable significa parábola, no "elipse degenerada".
\end{remark}

\begin{remark}[Adónde van estas curvas]
\label{rem:b1:curves:whereused}
\omterm{def:b1:curves:def}{Curvas parametrizadas} son el lenguaje de la mecánica: trayectorias
son curvas, los vectores de velocidad son vectores tangentes y \omterm{def:b1:curves:arclength}{arco
longitud} de \cref{def:b1:curves:arclength} es la distancia recorrida. el
\omterm{def:b1:curves:conics}{cónicas} reaparece allí donde actúa una ley del cuadrado inverso --- planetario
las órbitas son elipses con el sol en un foco. en
\cref{ch:b1:multivar}, las curvas pasan a ser el nivel \omterm{def:b1:logic:sets}{conjuntos} de funciones
de dos variables, y la tangente de este capítulo cumple con la
gradiente del siguiente. El volumen Year~2 añade curvatura y el
forma canónica local de una curva; el problema del fin de semana a continuación
ya extrae, sólo con las herramientas de este año, todo lo que
El siglo XVII conoció su curva más famosa.
\end{remark}

\begin{remark}[Perspectivas dentro del Libro 3]
\label{rem:b1:curves:perspectives}
Este capítulo ocupa toda la primera mitad del volumen y
alimenta el último capítulo. \emph{Consumado}: los vectores tangentes son
\omterm{def:b1:derivative:def}{derivados} (\cref{ch:b1:derivative}), \omterm{def:b1:curves:arclength}{longitud del arco} y las áreas son
\omterm{thm:b1:integration:def}{integrales} (\cref{ch:b1:integration}), \omterm{ex:b1:curves:astroid}{cúspides} se liquidan mediante
Expansiones de Taylor (\cref{ch:b1:taylor}), la tautocrona es una
\omterm{def:b1:linmaps:def}{lineal} ecuación diferencial (\cref{ch:b1:diffeq}), y cada
la distancia y el ángulo son euclidianos (\cref{ch:b1:euclid}).
\emph{alimentado}: en \cref{ch:b1:multivar}, una curva $t \mapsto (x(t),
y(t))$ dibujada dentro de un nivel \omterm{def:b1:logic:sets}{conjunto}$\{f = c\}$ se diferencia por
la regla de la cadena, y la identidad resultante $\langle \nabla f,\
f'(t)\rangle = 0$ une los vectores tangentes de este capítulo con la
gradientes del siguiente --- la línea tangente de la curva y la
dirección normal de la superficie son los mismos cálculos vistos
de ambos bancos.
\end{remark}

\section{Problema: la cicloide, reina de las curvas}

\begin{problem}\label{pb:b1:curves:1}
Una rueda de radio $R$ rueda sin deslizarse a lo largo del eje $x$;
la punta de la llanta inicialmente en el origen \omterm{def:b1:matrices:transpose}{rastros} el
\emph{cicloide}. El siglo XVII luchó por esta curva.
--- Galileo pesó recortes de papel, Wren los midió,
Roberval calculó su área, Huygens construyó relojes en ella\dots y
cada uno de sus resultados está al alcance de este capítulo. Nosotros
prueba los cuatro clásicos: la construcción tangente, \omterm{def:b1:curves:arclength}{longitud} de Wren
$8R$, el área $3\pi R^2$ y la tautocrona de Huygens
propiedad.\index{cicloide}\index{tautocrona}

\medskip
\noindent\textbf{Parte I --- Rolling and the tangent.}
\begin{enumerate}
 \item Después de que la rueda haya girado un ángulo $t$, su centro
 se sienta en $\Omega(t) = (Rt, R)$(rodando sin resbalar:
 distancia de contacto $=$ arco laminado). Demuestre que lo marcado
 el punto es en
 \[
 M(t) = \bigl(R(t - \sin t),\; R(1 - \cos t)\bigr).
 \]
 \item Calcula $f'(t)$ y muestra $\norm{f'(t)} =
 2R\,\abs{\sin\frac t2}$; localizar los puntos singulares
 (\omterm{ex:b1:curves:astroid}{cúspides} --- cf.\ \cref{exo:b1:curves:2}) y la parte superior de
 cada arco.
 \item Sea $C(t) = (Rt, 0)$ el punto de contacto y $T(t) =
 (Rt, 2R)$ la parte superior de la rueda. Demuestre que para $0 < t <
 2\pi $ el vector $M - C $ es \emph{normal} con respecto a la curva en
 $M(t)$ y el vector $T - M$ es \emph{tangente}: para dibujar
 la tangente a una cicloide, une el punto a la parte superior de
 su círculo rodante. \emph{(Factor everything through
 $\sin\frac t2$ and $\cos\frac t2$.)}
 \item Interpretar la pregunta 3 cinemáticamente: el punto de contacto es
 el \emph{centro de rotación instantáneo} y el
 La velocidad de $M$ es igual a su distancia a$C$(para unidades angulares).
 velocidad). Verifique $\norm{M - C} = 2R\abs{\sin\frac t2}$.
 \item Demostrar la relación altura-velocidad
 \[
 \norm{f'(t)}^2 = 2R\,y(t) :
 \]
 en una cicloide atravesada con velocidad angular unitaria, la
 La velocidad en cada punto es exactamente la velocidad de caída libre de un
 caída igual a la altura actual. (Guarde esto para
 Parte IV.)
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Wren's theorem: the arch has \omterm{def:b1:curves:arclength}{longitud}
$8R$.}
\begin{enumerate}[resume]
 \item Usando \cref{def:b1:curves:arclength}, calcule el \omterm{def:b1:curves:arclength}{longitud}
 de un arco:
 \[
 L = \int_0^{2\pi} 2R\sin\frac t2\,\dd t = 8R
 \]
 (Teorema de Wren, 1658). Cuatro diámetros de rueda, y no
 $\pi$ en cualquier lugar.
 \item Calcule \omterm{def:b1:curves:arclength}{longitud del arco} a partir de \omterm{ex:b1:curves:astroid}{cúspide}: $s(t) =
 4R\bigl(1 - \cos\frac t2\bigr)$ y verifique $s(2\pi) =
 8R $.
 \item Ahora mida el arco desde \emph{ápex} $t = \pi$:
 $\sigma(t) = \abs{4R\cos\frac t2}$. Demostrar lo intrínseco
 relación
 \[
 \sigma^2 = 8R\,\bigl(2R - y\bigr) :
 \]
 la distancia del arco cuadrado desde la parte superior es proporcional a
 la altura cae por debajo de la parte superior.
 \item Comprobaciones de cordura: recupera de la pregunta 8 que el medio arco
 desde el ápice hasta \omterm{ex:b1:curves:astroid}{cúspide} tiene \omterm{def:b1:curves:arclength}{longitud} $4R$, y comparar con el
 cálculo astroide de \cref{ex:b1:curves:circlelength}

 explicar qué hace esto posible a pesar de que ambos son
 construido a partir de círculos. \emph{(Look at the form of
 $\norm{f'}$.)}
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Roberval's area: $3\pi R^2$.}
\begin{enumerate}[resume]
 \item Justify that the area between one arch and the ground is
 $A = \int_0^{2\pi} y(t)\,x'(t)\,\dd t$\emph{(the

 \item Computar

 que el área cerrada es $\frac{3\pi}8$\emph{(linearize $\sin^4 t\cos^2 t$; only the constant
