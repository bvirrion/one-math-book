\chapter{Diferenciación}\label{ch:b1:derivative}

\omterm{def:b1:derivative:def}{Derivados} se computaron en todo el volumen de High School; que fue
Falta la cadena de teoremas que convierte la computación en
información sobre funciones: teorema de Rolle, teorema del valor medio,

\section{la derivada}

\begin{definition}\label{def:b1:derivative:def}
$f \colon I \to \R$ es \emph{differentiable at $x_0 \in
I $}\index{derivado} cuando el cociente de diferencias
$\frac{f(x) - f(x_0)}{x - x_0}$ tiene un límite (finito) como $x \to x_0$;
el límite está escrito $f'(x_0)$. Equivalentemente:
\[
f(x_0 + h) = f(x_0) + f'(x_0)\,h + h\,\varepsilon(h),
\qquad \varepsilon(h) \xrightarrow[h \to 0]{} 0 ,
\]
la gráfica admite entonces la recta tangente $ y = f(x_0) + f'(x_0)(x -
x_0)$. La diferenciabilidad en $x_0$ implica \omterm{def:b1:continuity:continuous}{continuidad} en $x_0$(leer
la pantalla). $f$ es diferenciable en $I$ cuando está en todos los puntos;
$f$ es de clase $C^1$ cuando además $f'$ es \omterm{def:b1:continuity:continuous}{continuo}, y de
clase $C^k$\index{Función ck@$C^k$ función} cuando $f$ puede ser
diferenciado $k$ veces con $f^{(k)}$\omterm{def:b1:continuity:continuous}{continuo}.
\end{definition}

\begin{example}\label{ex:b1:derivative:notc1}
El inverso de ``\omterm{def:b1:derivative:def}{diferenciable} $\Rightarrow$\omterm{def:b1:continuity:continuous}{continuo}'' falla:
$\abs{\,\cdot\,}$ en $0$. Más sorprendentemente, \omterm{def:b1:derivative:def}{diferenciable} no
implica $C^1$: la función $f(x) = x^2 \sin\frac 1x $($f(0) = 0$) es
\omterm{def:b1:derivative:def}{diferenciable} en todas partes, con $f'(0) = 0$, pero $f'(x) = 2x
\sin\frac1x - \cos\frac 1x $ no tiene límite en $0$
(\cref{exo:b1:derivative:2}).
\end{example}

\begin{example}[Diferenciable exactamente en un punto]\label{ex:b1:derivative:onepoint}
Sea $f(x) = x^2$ para $x \in \Q$ y $f(x) = 0$ para $x \notin
\Q $. En$0$:$\bigl|\frac{f(h) - 0}{h}\bigr| \leq \abs h \to 0$,
entonces $f$ es \omterm{def:b1:derivative:def}{diferenciable} en $0$ con $f'(0) = 0$. En cualquier $x_0
\neq 0$,$f$ ni siquiera es \omterm{def:b1:continuity:continuous}{continuo}: racional e irracional
secuencias que convergen a $x_0$ envían $f$a$x_0^2 \neq 0$ y a
$0$ respectivamente (\omterm{def:b1:topology:dense}{densidad}, \cref{thm:b1:reals:density}). entonces
La diferenciabilidad es una noción genuinamente \emph{puntualmente}: puede
manténgalo en un punto de $\R$ y en ningún otro lugar. La moraleja para
práctica: \omterm{def:b1:logic:statement}{enunciados} como el criterio de monotonicidad o Rolle
requieren el \omterm{def:b1:derivative:def}{derivado} \emph{on an \omterm{prop:b1:reals:intervals}{intervalo}} --- poseer
$f'(x_0)$ en puntos aislados, por muchos que sean, no admite ninguna
conclusión alguna.
\end{example}

\begin{theorem}[Operaciones]\label{thm:b1:derivative:operations}
Si $f, g$ son \omterm{def:b1:derivative:def}{diferenciable} en $x_0$(y donde las fórmulas hacen
sentido):
\[
(f + g)' = f' + g', \qquad
(fg)' = f'g + fg', \qquad
\Bigl(\frac fg\Bigr)' = \frac{f'g - fg'}{g^2},
\]
y si $g$ es \omterm{def:b1:derivative:def}{diferenciable} en $f(x_0)$:
$\;(g \circ f)'(x_0) = g'\bigl(f(x_0)\bigr)\, f'(x_0)$(cadena
regla\index{regla de la cadena}).
\end{theorem}

\begin{proof}
Suma: inmediata. Producto: escribir
\[
f(x)g(x) - f(x_0)g(x_0)
= \bigl(f(x) - f(x_0)\bigr) g(x) + f(x_0)\bigl(g(x) - g(x_0)\bigr),
\]
divida por $x - x_0$ y deje $x \to x_0$($g$ es \omterm{def:b1:continuity:continuous}{continuo} en $x_0$).
Cociente: tratar $\frac 1g$ mediante $\frac{1/g(x) - 1/g(x_0)}{x - x_0} =
\frac{-1}{g(x)g(x_0)}\cdot\frac{g(x) - g(x_0)}{x - x_0}$, luego aplicar
La regla del producto. Regla de la cadena: con $y_0 = f(x_0)$, defina
$\theta(y) = \frac{g(y) - g(y_0)}{y - y_0}$ para $ y \neq y_0$ y
$\theta(y_0) = g'(y_0)$:$\theta$ es \omterm{def:b1:continuity:continuous}{continuo} en $y_0$, y para
$x \neq x_0$,
\[
\frac{g(f(x)) - g(f(x_0))}{x - x_0}
= \theta\bigl(f(x)\bigr)\cdot \frac{f(x) - f(x_0)}{x - x_0}
\longrightarrow g'(y_0)\, f'(x_0),
\]
el primer factor por composición de límites (este dispositivo maneja el
caso $f(x) = f(x_0)$ limpiamente, donde el ingenuo ``multiplica y divide
por $f(x) - f(x_0)$'' descansos).
\end{proof}

\begin{theorem}[Derivada de una función inversa]\label{thm:b1:derivative:inverse}
Sea $f$\omterm{def:b1:continuity:continuous}{continuo} y estrictamente monótono en $I$, \omterm{def:b1:derivative:def}{diferenciable}
en $x_0$ con $f'(x_0) \neq 0$. Entonces $f^{-1}$
(\cref{thm:b1:continuity:bijection}) es \omterm{def:b1:derivative:def}{diferenciable} en $y_0 =
f(x_0)$, con
\[
(f^{-1})'(y_0) = \frac{1}{f'(x_0)} = \frac{1}{f'\bigl(f^{-1}(y_0)\bigr)} .
\]
Si $f'(x_0) = 0$, la inversa tiene una tangente vertical en $y_0$.
\end{theorem}

\begin{proof}
Para $ y \to y_0$, configure $x = f^{-1}(y)$: \omterm{def:b1:continuity:continuous}{continuidad} de $f^{-1}$ da $x
\to x_0$, y
\[
\frac{f^{-1}(y) - f^{-1}(y_0)}{y - y_0}
= \frac{x - x_0}{f(x) - f(x_0)}
= \frac{1}{\dfrac{f(x) - f(x_0)}{x - x_0}}
\longrightarrow \frac{1}{f'(x_0)} .
\]
Reclamación de tangente vertical: si $f'(x_0) = 0$, el cociente mostrado
es el recíproco de una cantidad que tiende a $0$ manteniendo
un signo constante (para $f$ estrictamente creciente,$\frac{f(x) -
f(x_0)}{x - x_0} > 0$ para todos los $x \neq x_0$): la diferencia
el cociente de $f^{-1}$ tiende por lo tanto a$+\infty $(a$-\infty $
para $f$ disminuyendo). Lo inverso sigue siendo \omterm{def:b1:continuity:continuous}{continuo} pero no es
\omterm{def:b1:derivative:def}{diferenciable} en $y_0$--- su gráfica, el reflejo de $f$
a lo largo de la diagonal, se encuentra vertical exactamente donde corrió $f$
horizontal, como se ilustra $x^{1/3}$ en $0$ frente a$x^3$.
\end{proof}

\begin{example}[Derivadas inversas, dos veces]\label{ex:b1:derivative:inverseaction}
El teorema vuelve a calcular el clásico \omterm{def:b1:derivative:def}{derivados} sin límite
trabajo. Para $\ln = \exp^{-1}$: en $ y = \eu^x$,
\[
(\ln)'(y) = \frac{1}{\exp'(x)} = \frac{1}{\eu^{x}} = \frac1y ,
\]
válido para cada $ y > 0$ ya que $\exp' = \exp $ nunca desaparece. Para
$\arctan = \tan^{-1}$: en $ y = \tan x$,
\[
(\arctan)'(y) = \frac{1}{1 + \tan^2 x} = \frac{1}{1 + y^2} ,
\]
utilizando $\tan' = 1 + \tan^2 > 0$. La idea final: la fórmula
convierte el conocimiento sobre una función en conocimiento sobre su
inversa al precio de una sustitución --- y el
sustitución ($x = \ln y $,$x = \arctan y $) es exactamente la
\omterm{def:b1:logic:statement}{enunciado} que las dos variables viven en lados opuestos de la
biyección.
\end{example}

\section{Rolle y el teorema del valor medio}

\begin{proposition}[interior extremo]\label{prop:b1:derivative:fermat}
Si $f$ es \omterm{def:b1:derivative:def}{diferenciable} en un punto \emph{\omterm{def:b1:topology:closure}{interior}}$x_0$ de $I$ y
tiene un extremo local allí, entonces $f'(x_0) = 0$.
\end{proposition}

\begin{proof}
Digamos un máximo local: hay $r > 0$ con $f(x) \leq f(x_0)$
para $\abs{x - x_0} \leq r$, y la interioridad garantiza ambos
Los lados de $x_0$ están disponibles en $I$. Para $0 < h \leq r$ el
el cociente $\frac{f(x_0 + h) - f(x_0)}{h}$ tiene numerador $\leq 0$
y denominador $> 0$: es $\leq 0$, y su límite $f'(x_0)$
hereda $\leq 0$(las grandes desigualdades pasan a límites,
\cref{thm:b1:seq:order}); para $-r \leq h < 0$ el cociente es
$\geq 0$, dando $f'(x_0) \geq 0$. Por lo tanto $f'(x_0) = 0$. (En un
punto final, sólo hay un signo disponible: la conclusión falla allí ---
piense en $x$ en $\intcc{0}{1}$, máximo en $1$ con \omterm{def:b1:derivative:def}{derivado}
$1$.)
\end{proof}

\begin{theorem}[rollo]\label{thm:b1:derivative:rolle}
Sea $f$\omterm{def:b1:continuity:continuous}{continuo} en $\intcc{a}{b}$, \omterm{def:b1:derivative:def}{diferenciable} en
$\intoo{a}{b}$, con $f(a) = f(b)$. Entonces $f'(c) = 0$ para algunos $c \in
\intoo{a}{b}$.\index{teorema de rolle}
\end{theorem}

\begin{proof}
Por el teorema del valor extremo (\cref{thm:b1:continuity:evt}), $f$
alcanza su máximo y mínimo en $\intcc{a}{b}$. si ambos son
alcanzado en los puntos finales, entonces (desde $f(a) = f(b)$) max$=$min y $f$
es constante: cualquier \omterm{def:b1:topology:closure}{interior} $c$ funciona. De lo contrario, se llega a un extremo.
alcanzado en un punto \omterm{def:b1:topology:closure}{interior} $c$, y
\cref{prop:b1:derivative:fermat} da $f'(c) = 0$.
\end{proof}

\begin{theorem}[Teorema del valor medio]\label{thm:b1:derivative:mvt}
Sea $f$\omterm{def:b1:continuity:continuous}{continuo} en $\intcc{a}{b}$, \omterm{def:b1:derivative:def}{diferenciable} en
$\intoo{a}{b}$. Existe $c \in \intoo{a}{b}$ con
\[
f(b) - f(a) = f'(c)\,(b - a) .
\]
\emph{Desigualdad del valor medio:} si además $m \leq f' \leq M$ en
$\intoo{a}{b}$, luego $m(b-a) \leq f(b) - f(a) \leq M(b-a)$; en
En particular, $\abs{f'} \leq K$ implica que $f$ es $K$-Lipschitz.
\index{teorema del valor medio}
\end{theorem}

\begin{proof}
Aplicar Rolle a $g(x) = f(x) - \frac{f(b) - f(a)}{b - a}(x - a)$:$g$
es \omterm{def:b1:continuity:continuous}{continuo} en $\intcc{a}{b}$, \omterm{def:b1:derivative:def}{diferenciable} en el interior y $g(a) =
f(a) = g(b)$. En el punto $c $ donde $g'(c) = 0$:$f'(c) =
\frac{f(b)-f(a)}{b-a}$. La desigualdad sigue acotando $f'(c)$;
el Lipschitz \omterm{def:b1:logic:statement}{enunciado} lo aplica a cada par de puntos.
\end{proof}

\begin{example}[El método de Newton es el de Heron.]\label{ex:b1:derivative:newton}
El método de Newton para resolver $f(x) = 0$ reemplaza la curva por
su tangente en la estimación actual $x_n$ y toma la tangente
root como la siguiente suposición:
\[
0 = f(x_n) + f'(x_n)(x_{n+1} - x_n)
\quad\Longrightarrow\quad
x_{n+1} = x_n - \frac{f(x_n)}{f'(x_n)} .
\]
Ejecútelo en $f(x) = x^2 - 2$:
\[
x_{n+1} = x_n - \frac{x_n^2 - 2}{2x_n}
= \frac{x_n}{2} + \frac{1}{x_n}
= \frac12\Bigl(x_n + \frac{2}{x_n}\Bigr) :
\]
exactamente la iteración de Heron (\cref{ex:b1:seq:heron}), dos
milenios antes. La velocidad cuadrática observada allí ahora es
explicado por la imagen tangente: cerca de una raíz simple, curva y
tangentes difieren por un error de segundo orden, por lo que cada paso aproximadamente
eleva al cuadrado el error --- el \omterm{def:b1:logic:statement}{enunciado} general se sigue del
Límites de Taylor de \cref{ch:b1:taylor}. La idea final:
donde la dicotomía (\cref{ex:b1:continuity:dichotomy}) usa solo
\omterm{def:b1:continuity:continuous}{continuidad} y gana un bit por paso, Newton gasta un
\omterm{def:b1:derivative:def}{derivado} para duplicar el número de dígitos correctos por paso.
\end{example}

\begin{example}[El teorema del valor medio como estimador]\label{ex:b1:derivative:mvtestimate}
¿Qué tamaño tiene $\sqrt{101}$? Aplicar el teorema a$f(t) = \sqrt t$
en $\intcc{100}{101}$: para algunos $c \in \intoo{100}{101}$,
\[
\sqrt{101} - 10 = \frac{1}{2\sqrt c},
\qquad\text{so}\qquad
\frac{1}{2\sqrt{101}} < \sqrt{101} - 10 < \frac{1}{20} = 0.05 ,
\]
y desde $\sqrt{101} < 10.05$, el límite izquierdo excede
$\frac{1}{20.1} > 0.0497$: así $10.0497 < \sqrt{101} < 10.05$
(valor verdadero $10.049875\dots$) --- tres decimales correctos de uno
Evaluación \omterm{def:b1:derivative:def}{derivado}. Asimismo $\abs{\sin a - \sin b} \leq
\abs{a - b}$(cota $\abs{\cos}\leq 1$): las estimaciones de Lipschitz
utilizados desde \cref{ch:b1:seq} son todo este teorema. el cierre
Idea: el teorema del valor medio es una fórmula de Taylor de orden cero
--- cambia un punto desconocido $c$ por una desigualdad dura, y
\cref{ch:b1:taylor} iterará exactamente esta operación.
\end{example}

\begin{corollary}[Criterio de monotonicidad]\label{cor:b1:derivative:monotone}
Sea $f$\omterm{def:b1:continuity:continuous}{continuo} en $I$, \omterm{def:b1:derivative:def}{diferenciable} en \omterm{def:b1:topology:closure}{interior}.
\begin{enumerate}
 \item $f' \geq 0$ en \omterm{def:b1:topology:closure}{interior}$\iff$ $f$ está aumentando;$f' = 0$
 $\iff$ $f$ constante.
 \item Si $f' > 0$ excepto en un número finito de puntos donde desaparece,
 $f$ está aumentando \emph{estrictamente}.
\end{enumerate}
\end{corollary}

\begin{proof}
Si $f' \geq 0$: para $x < y $ en $I$, el teorema del valor medio en
$\intcc{x}{y}$ da $f(y) - f(x) = f'(c)(y - x) \geq 0$. Por el contrario,
Los cocientes en diferencias de una función creciente son $\geq 0$, por lo que sus
los límites también lo son. El caso constante: aplicar lo anterior a $f'$ y
$-f' \geq 0$. Versión estricta:$f$ está aumentando; la igualdad $f(x) =
f(y)$ para $ x < y $congelaría $f $ en $\intcc{x}{y}$, forzando a$f' =
0$ allí --- infinitos puntos.
\end{proof}

\begin{example}[Derivadas iguales, funciones desiguales]\label{ex:b1:derivative:intervalmatters}
En $\R^* = \intoo{-\infty}{0} \cup \intoo{0}{+\infty}$, ambos
$f(x) = \ln\abs x$ y $g(x) = \ln\abs x + \mathbf{1}_{x>0}$
(agregue $1$ solo en la media línea derecha) satisfaga $f' = g' =
\frac1x $. No se diferencian por una constante: el criterio
``Constante $f' = 0 \implies f$'' es una \emph{\omterm{prop:b1:reals:intervals}{intervalo}} \omterm{def:b1:logic:statement}{enunciado}
--- su prueba ejecuta el teorema del valor medio entre dos puntos,
lo que requiere que todo el segmento que los une se encuentre en el
dominio. En cada media línea por separado, las primitivas de
$\frac1x$ son $\ln\abs x + c$ con una constante por media línea,
dos constantes independientes en total. \Cref{ch:b1:integration}
hereda esta letra pequeña: ``la'' primitiva de una función es
bien definido hasta una constante \emph{on each \omterm{prop:b1:reals:intervals}{intervalo} of its
domain}, y las tablas antiderivadas asumen silenciosamente conectividad.
\end{example}

\begin{example}[Riguroso libre]\label{ex:b1:derivative:strictness}
$x \mapsto x^3$ es \emph{estrictamente} aumentando en $\R$ incluso
aunque su \omterm{def:b1:derivative:def}{derivado} desaparece en $0$: la cláusula del criterio
``$f' > 0$ excepto en un número finito de puntos'' está diseñado exactamente
para puntos tan planos. Por el contrario, $f' \geq 0$ por sí solo sólo da
aumento en el sentido amplio (una función constante califica), y
un \omterm{def:b1:derivative:def}{derivado} que desaparece en un subintervalo completo congela el
funcionar allí. La regla práctica: reclamar estrictas
monotonicidad, enumere los ceros de $f'$; finitos (o más
generalmente, ninguno en ningún subintervalo) es inofensivo, un \omterm{prop:b1:reals:intervals}{intervalo} de
ellos es fatal.
\end{example}

\begin{example}[Un estudio de variación completo.]\label{ex:b1:derivative:variations}
Estudio $f(x) = x^3 - 3x + 1$ sobre $\R$. \omterm{def:b1:derivative:def}{Derivado}:$f'(x) =
3(x^2 - 1)$, positivo en $\intoo{-\infty}{-1}$, negativo en
$\intoo{-1}{1}$, positivo en $\intoo{1}{+\infty}$: por el
criterio de monotonicidad, $f$ aumenta, luego disminuye, luego
aumenta, con un máximo local $f(-1) = 3$ y un mínimo local
$f(1) = -1$. Límites:$\mp\infty $ en $\mp\infty $. Consecuencias,
leer la tabla de variaciones con el valor intermedio
teorema en cada rama monótona: $f$ desaparece exactamente una vez en
cada uno de
\[
\intoo{-\infty}{-1}, \qquad \intoo{-1}{1}, \qquad
\intoo{1}{+\infty}
\]
(los valores en los cruces tienen signos opuestos: $3 > 0 > -1$),
entonces la ecuación $x^3 - 3x + 1 = 0$ tiene exactamente tres reales
raíces; numéricamente se encuentran cerca de $-1.88$,$0.35$,$1.53$. el
Información final: una tabla de variación es \emph{dispositivo de prueba},
no es un boceto --- rama monótona más cambio de signo es igual
exactamente una raíz, y la tabla enumera las ramas
exhaustivamente.
\end{example}

\begin{theorem}[Fórmula de Leibniz]\label{thm:b1:derivative:leibniz}
Si $f, g$ es $n$ multiplicado por \omterm{def:b1:derivative:def}{diferenciable}, también lo es $fg $,
y\index{Fórmula de Leibniz}
\[
(fg)^{(n)} = \sum_{k=0}^{n} \binom nk f^{(k)}\, g^{(n-k)} .
\]
\end{theorem}

\begin{proof}
Inducción en $n$, exactamente paralela al teorema del binomio. el
El caso $n = 1$ es la regla del producto. Asumiendo la fórmula en el rango
$n$, diferencia una vez más:
\[
(fg)^{(n+1)} = \sum_{k=0}^{n} \binom nk
\Bigl( f^{(k+1)} g^{(n-k)} + f^{(k)} g^{(n-k+1)} \Bigr),
\]
luego reindexe la primera suma con $j = k + 1$ y recopile el
coeficiente de $f^{(j)} g^{(n+1-j)}$: es $\binom{n}{j-1} +
\binom nj = \binom{n+1}{j}$ según la regla de Pascal
(\cref{prop:b1:counting:identities}), los términos \omterm{def:b1:topology:closure}{límite} $j = 0$
y $j = n + 1$ llevando $\binom{n+1}{0} = \binom{n+1}{n+1} =
1$ como deberían.
\end{proof}

\begin{example}[Leibniz en acción]\label{ex:b1:derivative:leibnizaction}
Calcule $\bigl(x^2 \eu^x\bigr)^{(n)}$ para $n \geq 2$. Tome $f =
x^2$, cuyo \omterm{def:b1:derivative:def}{derivados} muere rápidamente ($f' = 2x$,$f'' = 2$,
$f^{(k)} = 0$ para $k \geq 3$) y $g = \eu^x$: solo tres términos
de la suma de Leibniz sobreviven,
\[
\bigl(x^2\eu^x\bigr)^{(n)}
= \binom n0 x^2 \eu^x + \binom n1 (2x)\,\eu^x + \binom n2\,
2\,\eu^x
= \eu^x\bigl(x^2 + 2nx + n(n-1)\bigr).
\]
Verificación de cordura en $n = 1$:$\eu^x(x^2 + 2x)$, que de hecho es
$(x^2\eu^x)'$. La idea final: utilice Leibniz cuando un factor
es un \omterm{def:b1:poly:def}{polinomio} --- la suma entonces solo tiene términos $\deg + 1$, y
la fórmula es una forma cerrada, no una identidad abstracta. (por dos
factores infinitamente animados como $\eu^x\sin x$, complejos
los exponenciales de \cref{ch:b1:complex} son la mejor herramienta).
\end{example}

\section{Convexidad}

\begin{definition}\label{def:b1:derivative:convex}
$f \colon I \to \R$ es \emph{convexo}\index{función convexa} cuando
cada acorde se encuentra encima del gráfico:
\[
\forall x, y \in I,\ \forall t \in \intcc{0}{1}, \quad
f\bigl(tx + (1-t)y\bigr) \leq t f(x) + (1-t) f(y).
\]
($f$ es \emph{cóncavo} cuando $-f$ es convexo).
\end{definition}

\begin{theorem}[Caracterizaciones diferenciales]\label{thm:b1:derivative:convexchar}
Sea $f$\omterm{def:b1:derivative:def}{diferenciable} en $I$. Los siguientes son equivalentes:
\begin{enumerate}
 \item $f$ es \omterm{def:b1:derivative:convex}{convexo};
 \item $f'$ está aumentando en $I$;
 \item el gráfico se encuentra encima de cada tangente: $f(y) \geq f(x) +
 f'(x)(y - x)$ para todo $x, y \in I $.
\end{enumerate}
Si $f$ es el doble de \omterm{def:b1:derivative:def}{diferenciable}:$f$\omterm{def:b1:derivative:convex}{convexo}$\iff f'' \geq 0$.
\end{theorem}

\begin{proof}
(1 $\Rightarrow $3) Convexidad escrita como $\frac{f(x + t(y-x)) -
f(x)}{t} \leq f(y) - f(x)$ para $ t \in \intoc{0}{1}$; deja $t \to 0^+$:
$f'(x)(y - x) \leq f(y) - f(x)$.

(3 $\Rightarrow $2) Para $x < y $, las dos desigualdades tangentes en $x$
y en $ y $dar $f'(x)(y-x) \leq f(y) - f(x) \leq f'(y)(y - x)$,
por lo tanto $f'(x) \leq f'(y)$.

(2 $\Rightarrow $1) Repare $x < y $ y $t \in \intoo{0}{1}$, y deje que $z
= tx + (1-t)y \in \intoo{x}{y}$. Por el teorema del valor medio de
$\intcc{x}{z}$ y $\intcc{z}{y}$: hay $c_1 < z < c_2$ con
\[
\frac{f(z) - f(x)}{z - x} = f'(c_1) \leq f'(c_2)
= \frac{f(y) - f(z)}{y - z} ,
\]
y compensación de denominadores ($z - x = (1-t)(y-x)$,$ y - z = t(y-x)$)
se reorganiza exactamente en la desigualdad de convexidad.

Dos veces caso \omterm{def:b1:derivative:def}{diferenciable}: $f'' \geq 0 \iff f'$ aumentando
(\cref{cor:b1:derivative:monotone}).
\end{proof}

\begin{example}[Desigualdades clásicas de convexidad]\label{ex:b1:derivative:convexineq}
$\exp$ es \omterm{def:b1:derivative:convex}{convexo} ($\exp'' = \exp > 0$): su tangente en $0$ da
$\eu^x \geq 1 + x$ para todos los $x$.$\ln$ es cóncava: su tangente en $1$
da $\ln x \leq x - 1$; sus acordes dan, para $0 < a \leq b$, el
desigualdad entre medias geométricas y aritméticas: tomando $t =
\frac12$ en concavidad,
\[
\ln\frac{a + b}{2} \geq \frac{\ln a + \ln b}{2} = \ln\sqrt{ab},
\qquad\text{so}\qquad
\sqrt{ab} \leq \frac{a+b}{2} .
\]
La desigualdad aritmética-geométrica general es
\cref{exo:b1:derivative:9}.
\end{example}

\begin{example}[Una desigualdad de convexidad desde cero.]\label{ex:b1:derivative:xlnx}
La función $f(t) = t\ln t$ es \omterm{def:b1:derivative:convex}{convexo} en $\intoo{0}{+\infty}$:
$f''(t) = \frac1t > 0$. Su desigualdad del punto medio, multiplicada por
$2$, dice: para todos $a, b > 0$,
\[
a\ln a + b\ln b \;\geq\; (a + b)\,\ln\frac{a + b}{2} ,
\]
con igualdad sif $a = b $(convexidad estricta). Prueba de manejo:$a =
1$,$b = 3$ dan $3\ln 3 = 3.296$ contra $4\ln 2 = 2.773$.
Esta desigualdad inocua es el caso de dos puntos de la
\emph{entropía} comparación que reaparece con la de Jensen
desigualdad (\cref{exo:b1:derivative:9}) y en el
Asintóticas de la teoría de la información del volumen del año 3. el
Idea final: para fabricar una desigualdad, encuentre una función.
cuyo segundo \omterm{def:b1:derivative:def}{derivado} tiene signo y escribe qué convexidad
dice --- la caracterización diferencial convierte una verificación de signo
en infinitas desigualdades.
\end{example}

\begin{remark}[Errores comunes con los derivados]
(i) \emph{A positive \omterm{def:b1:derivative:def}{derivado} at one point does not give
monotonicity near it}: $f(x) = \frac x2 + x^2\sin\frac1x $(con
$f(0) = 0$) tiene $f'(0) = \frac12 > 0$, todavía
\[
f'(x) = \frac12 + 2x\sin\frac1x - \cos\frac1x
\]
es igual a $-\frac12$ en cada $x_n = \frac{1}{2\pi n}$: cada
\omterm{def:b1:topology:open}{entorno} de $0$ contiene descensos. Necesidades de monotonicidad $f'
\geq 0$\emph{on an \omterm{prop:b1:reals:intervals}{intervalo}}
(\cref{cor:b1:derivative:monotone}); el signo puntual solamente
controla el cruce de la recta tangente. (ii) \emph{Rolle's
Tres hipótesis están todas activas.}: $\abs x$ en $\intcc{-1}{1}$
(sin diferenciabilidad \omterm{def:b1:topology:closure}{interior}), $x$ en $\intcc{0}{1}$(termina no
igual), y $x - \lfloor x\rfloor$ en $\intcc{0}{1}$(\omterm{def:b1:continuity:continuous}{continuidad}
falla en $1$) cada uno rompe exactamente una hipótesis y la
conclusión. (iii) \emph{\omterm{def:b1:derivative:def}{Derivados} may be discontinuous, but
not arbitrarily}: $f'$ puede oscilar
(\cref{ex:b1:derivative:notc1}) pero siempre satisface las
propiedad de valor intermedio (Darboux,
\cref{exo:b1:derivative:10}): un \omterm{def:b1:derivative:def}{derivado} nunca salta --- si
calcula un ``límite \omterm{def:b1:derivative:def}{derivado}'' unilateral con un salto,
han diferenciado una función \omterm{def:b1:derivative:def}{no diferenciable}. (iv) \emph{The
inverse formula needs $f' \neq 0$}:$x \mapsto x^3$ es un suave
biyección estrictamente creciente cuya inversa $x^{1/3}$ tiene una
tangente vertical en $0$ --- la diferenciabilidad de la inversa es
perdido exactamente donde $f'$ desaparece
(\cref{thm:b1:derivative:inverse}).
\end{remark}

\begin{remark}[Dónde funciona a continuación el teorema del valor medio]
Casi todos los \omterm{def:b1:logic:statement}{enunciado} cuantitativos de los siguientes capítulos son este
Teorema del valor medio del capítulo disfrazado: el teorema fundamental
de cálculo (\cref{ch:b1:integration}) diferencia el área
función y concluye con el criterio de monotonicidad; el
La fórmula de Taylor-Lagrange (\cref{ch:b1:taylor}) es el valor medio
teorema iterado $n$ veces; el análisis de errores del método de Newton
y de iteraciones de punto fijo (\cref{exo:b1:derivative:11}) es
la forma Lipschitz; y el problema del fin de semana de este capítulo
(\cref{pb:b1:derivative:1}) convierte el mismo cota de Lipschitz en
teoría de números --- una desigualdad de repulsión entre \omterm{pb:b1:logic:1}{algebraico
numeros} y racionales, produciendo el primer \omterm{pb:b1:logic:1}{numero trascendental}
en la historia. En el volumen del año 2, el valor medio
\emph{desigualdad} sobrevive en varias variables cuando la igualdad
no lo hace.
\end{remark}

\begin{example}[Desigualdad de Young por concavidad]\label{ex:b1:derivative:young}
Sea $p, q > 1$ con $\frac1p + \frac1q = 1$. Para todos los $a, b > 0$:
\[
ab \;\leq\; \frac{a^p}{p} + \frac{b^q}{q} .
\]
Prueba mediante una aplicación de la concavidad de $\ln$ con pesas
$\frac1p, \frac1q $(la desigualdad de Jensen de dos puntos, como en
\cref{exo:b1:derivative:9}):
\[
\ln\Bigl(\frac{a^p}{p} + \frac{b^q}{q}\Bigr)
\;\geq\; \frac1p \ln(a^p) + \frac1q \ln(b^q)
= \ln a + \ln b = \ln(ab),
\]
y $\ln$ al aumentar convierte la desigualdad de logaritmos en
el reclamo; igualdad iff $a^p = b^q $(concavidad estricta). el caso
$p = q = 2$ es la desigualdad aritmético-geométrica$ ab \leq
\frac{a^2 + b^2}{2}$ disfrazada. La idea final: Young
la desigualdad es la semilla algebraica del Hölder y Minkowski
desigualdades del volumen del Año 2 --- una concavidad \omterm{def:b1:logic:statement}{enunciado}
Aproximadamente $\ln$, ensamblado según las normas.
\end{example}

\begin{remark}[Perspectivas dentro de este volumen]
El \omterm{def:b1:derivative:def}{derivado} adquiere tres nuevas vidas antes de que termine el volumen.
En \cref{ch:b1:taylor} itera: $n$\omterm{def:b1:derivative:def}{derivados} en un punto
comprimir en un \omterm{def:b1:poly:def}{polinomio} más un error controlado, y el
El teorema del valor medio se convierte en el resto de Lagrange. en
\cref{ch:b1:curves}, la diferenciación se vuelve geométrica: para un
curva parametrizada $t \mapsto (x(t), y(t))$, el par $(x'(t),
y'(t))$ es una velocidad \emph{vector}, la tangencia se vuelve
colinealidad y los puntos críticos se convierten en cúspides para clasificar. en
\cref{ch:b1:multivar}, se congela una variable a la vez: parcial
\omterm{def:b1:derivative:def}{derivados} repite este capítulo dos veces y la recta tangente
crece hasta convertirse en un plano tangente. Los tres capítulos heredan lo mismo.
gramática --- aproximación lineal local más un término de error ---
hablado por primera vez aquí.
\end{remark}

\section{Problema: la desigualdad de Liouville y la primera
numero trascendental}

\begin{problem}[{Problema de fin de semana --- \omterm{pb:b1:logic:1}{números algebraicos} repelente
racionales: $\abs{x - p/q} \geq C/q^d$, y la trascendencia de
$\sum 10^{-n!}$}]
\label{pb:b1:derivative:1}
Un número real es \emph{algebraico} cuando es raíz de un número distinto de cero.
\omterm{def:b1:poly:def}{polinomio} con coeficientes enteros y \emph{trascendental}
de lo contrario. En 1844, Liouville produjo el primer número de la historia.
\emph{probado} trascendental, y el motor de su prueba es este
Teorema del valor medio del capítulo: un \omterm{pb:b1:logic:1}{número algebraico} de grado $d$
no puede ser aproximado por racionales mejores que $C/q^d$ --- entonces un
El número aproximado \emph{más rápido que cualquier poder} no puede ser
algebraico. Este problema construye la desigualdad, construye
El número de Liouville $L = 0.110001000\dots $(unos en el factorial
posiciones, a través de la maquinaria de dígitos de \cref{pb:b1:reals:1}),
prueba su trascendencia, y termina con la prueba rival de Cantor y
límites efectivos para $\sqrt2$ y
$2^{1/3}$.\index{numero trascendental}\index{Liouville's
desigualdad}\index{número algebraico}

\medskip
\noindent\textbf{Parte I --- How well can rationals be
approximated?}
\begin{enumerate}
 \item Demuestre que dos racionales distintos $\frac ab \neq \frac pq$
 (escrito con $b, q \geq 1$) satisface $\bigl|\frac ab -
 \frac pq\bigr| \geq \frac{1}{bq}$. Deducir: si $x =
 \frac ab $ y $0 < \bigl|x - \frac pq\bigr| <
 \frac{1}{bq}$, no existe tal $\frac pq $ --- un racional
 repele todos los demás racionales a escala $\frac 1q$.
 \item Demuestre que para \emph{cada} racional $\frac pq $($q \geq
 1$):$\bigl|\sqrt2 - \frac pq\bigr| \geq
 \frac{1}{4q^2}$\emph{(if the distance exceeds $1$ this
 is clear; otherwise bound $\abs{\sqrt2 + p/q} < 4$ and
 use the nonzero integer $\abs{p^2 - 2q^2} \geq 1$)}.
 \item En la otra dirección: comprobar que $(p, q) \mapsto (p +
 2q, p + q)$ conserve $\abs{p^2 - 2q^2} = 1$, generar
 de $(1,1)$ los pares $(3,2)$,$(7,5)$,$(17,12)$,
 $(41,29)$,$(99,70)$ y muestran que cada uno satisface
 \[
 \Bigl|\sqrt2 - \frac pq\Bigr| =
 \frac{1}{q^2\,(\sqrt2 + p/q)} < \frac{1}{2q^2} :
 \]
 infinitas aproximaciones de orden $2$. con
 pregunta 2: el exponente de aproximación de $\sqrt 2$ es
 \emph{exactamente}$2$.
 \item (Dirichlet) Sea $x$ irracional y $N \in \N^*$.
 Considere las partes fraccionarias $N + 1$ de $0, x, 2x,
 \dots, Nx $ en las casillas $N $ $\intco{\frac kN}{\frac{k +
 1}{N}}$: según el principio del casillero
 (\cref{cor:b1:counting:pigeonhole}), caen dos en una caja.
 Deduzca $q \leq N$ y $p$ con $\abs{qx - p} <
 \frac 1N$, por lo tanto infinitos racionales con
 $\bigl|x - \frac pq\bigr| < \frac{1}{q^2}$: \emph{cada}
 irracional es aproximado al pedido $2$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Liouville's inequality.} Sea $x$
irracional y algebraico.
\begin{enumerate}[resume]
 \item Mostrar que entre los enteros distintos de cero \omterm{def:b1:poly:def}{polinomios}
 desapareciendo en $x$ hay uno, digamos $P$ de grado $d$,
 con \emph{sin raíz racional}; y verifique $d \geq 2$\emph{(divide out a factor $X - \frac ab$ over $\Q$ and
 clear denominators; degree $1$ would make $x$
 rational)}.
 \item Demuestre que para cada racional $\frac pq $($q \geq 1$):
 $\bigl|P\bigl(\frac pq\bigr)\bigr| \geq \frac{1}{q^d}$\emph{($q^d P(p/q)$ es un número entero distinto de cero)}.
 \item Vamos $M = \max_{\intcc{x-1}{x+1}} \abs{P'}$
 (\cref{thm:b1:continuity:evt}). Usando el valor medio
 teorema entre $x$ y $\frac pq $, demuestre
 \emph{La desigualdad de Liouville}: con $C = \min\bigl(1,
 \frac 1M\bigr) > 0$,
 \[
 \Bigl| x - \frac pq \Bigr| \geq \frac{C}{q^{\,d}}
 \qquad\text{for every rational } \frac pq,\ q \geq 1 .
 \]
 \item Llame a $x$a \emph{número de liouville} cuando por cada $n \in
 \N $existe un $\frac pq $racional con $q \geq 2$ y
 $0 < \bigl|x - \frac pq\bigr| < q^{-n}$. demostrar que un
 El número de Liouville es irracional \emph{(question 1: choose
 $n$ with $2^{\,n-1} > b$)}.
 \item Demuestre el teorema de Liouville: \emph{un número de Liouville es
 trascendental} \emph{(combine questions 7 and 8: the
 inequality $C < q^{\,d-n}$ fails for large $n$)}.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The number $L$.}
\begin{enumerate}[resume]
 \item Sea $L$ el valor (en el sentido de
 \cref{pb:b1:reals:1}) de la cadena de dígitos decimales con
 dígito $1$ en las posiciones $n!$($n = 1, 2, 3, \dots$)
 y $0$ en otro lugar, es decir,$L = \sup_k t_k$ con $t_k =
 \sum_{n=1}^{k} 10^{-n!}$. Escribe el primer $25$.
 dígitos de $L$.
 \item Pruebe el bracketing de cola, para cada $k \geq 1$:
 \[
 10^{-(k+1)!} \;\leq\; L - t_k \;\leq\;
 \frac{10}{9}\,10^{-(k+1)!} \;<\; 2\cdot 10^{-(k+1)!}
 \]
 \emph{(limitado cada suma parcial más allá de $t_k$ por un finito
 suma geométrica)}.
 \item Escriba $t_k = \frac{p_k}{q_k}$ con $q_k = 10^{k!}$.
 Mostrar $0 < L - \frac{p_k}{q_k} < \frac{2}{q_k^{\,k+1}}$,
 y concluir que $L$ es un número de Liouville en el sentido
 de la pregunta 8.
 \item Concluye: $L$ es trascendental --- el primero explícito
 ejemplo en la historia (Liouville, 1844). Verifique su
 irracionalidad directamente: sus dígitos no son finalmente
 periódico (brechas crecientes, como en \cref{pb:b1:reals:1},
 pregunta 20).
 \item Generalizar: reemplazar cada dígito $1$ por un dígito arbitrario
