\chapter{Ecuaciones diferenciales lineales}\label{ch:b1:diffeq}

Conocido por primera vez en el volumen de High School, se tratan las ecuaciones diferenciales.
aquí con pruebas completas y en mayor generalidad: lineal de primer orden
ecuaciones con coeficientes variables (resueltas completamente por el
método de variación de constantes) y ecuaciones lineales de segundo orden con
coeficientes constantes, el modelo de oscilaciones. Ambos casos muestran
la misma estructura: \emph{solución general $=$ una solución particular
$+$ solución general de la ecuación homogénea}.

\section{Ecuaciones lineales de primer orden}

\begin{definition}\label{def:b1:diffeq:linear1}
Sea $I$ un intervalo y $a, b \colon I \to \R$(o $\C$) sea
continuo. la ecuacion
\[
(E)\colon\quad y' + a(x)\,y = b(x),
\]
en la función diferenciable desconocida $ y \colon I \to \R$(o $\C$),
es un \emph{diferencial lineal de primer orden
ecuación}\index{ecuación diferencial!lineal de primer orden}. el
la ecuación $(H)\colon y' + a(x) y = 0$ es su \emph{homogéneo}
ecuación.
\end{definition}

\begin{theorem}[Resolviendo la ecuación homogénea]\label{thm:b1:diffeq:homogeneous1}
Sea $A$ una primitiva de $a$ en $I$(existe:
\cref{ch:b1:integration}). Las soluciones de $(H)$ en $I$ son exactamente
las funciones
\[
y(x) = \lambda\, \eu^{-A(x)}, \qquad \lambda \in \R \text{ (or } \C).
\]
\end{theorem}

\begin{proof}
Estas funciones son soluciones: $ y' = -\lambda A' \eu^{-A} = -a y $.
Por el contrario, deje que $ y $resuelva $(H)$ y configure $z(x) = y(x)\, \eu^{A(x)}$.
entonces
\[
z' = y' \eu^{A} + y\, a\, \eu^{A} = (y' + ay)\, \eu^{A} = 0 ,
\]
entonces $z$ es constante en el intervalo $I$, digamos $z = \lambda $:$ y =
\lambda \eu^{-A}$. (Tenga en cuenta la lógica: ninguna solución se pierde, porque
La solución \emph{cada} se ha escrito en el formato anunciado).
\end{proof}

\begin{example}[Una ecuación homogénea con coeficiente variable.]\label{ex:b1:diffeq:varcoeff}
Resuelva $ y' + (\cos x)\,y = 0$ en $\R$. Una primitiva de $a(x) =
\cos x $ es $A(x) = \sin x $, por lo que las soluciones son
\[
y(x) = \lambda\,\eu^{-\sin x}, \qquad \lambda \in \R .
\]
Dos lecturas. Cada solución es periódica (período $2\pi$) y nunca
desaparece a menos que $\lambda = 0$--- el signo de $\lambda$ sea el
signo de $ y $ para siempre, ya que una exponencial no puede cruzar el cero. y
la solución a través de $ y(0) = y_0$ es $y_0\eu^{-\sin x}$: exactamente
una curva de la familia que pasa por cada punto inicial, la
imagen unidimensional de \cref{thm:b1:diffeq:voc}~(2).
\end{example}

\begin{theorem}[Variación de constantes; Problema de Cauchy]\label{thm:b1:diffeq:voc}
Con la notación anterior:
\begin{enumerate}
 \item Las soluciones de $(E)$ en $I$ son exactamente
 \[
 y(x) = \Bigl(\lambda + \int_{x_0}^{x} b(t)\, \eu^{A(t)} \dd t
 \Bigr)\, \eu^{-A(x)},
 \qquad \lambda \in \R,
 \]
 donde se fija $x_0 \in I$. Equivalentemente: solución general de
 $(H)$ más una solución particular de $(E)$.
 \item Por cada $x_0 \in I$ y $y_0$, el \emph{cauchy
 problema}\index{problema de cauchy} ``$(E)$ y $ y(x_0) = y_0$''
 tiene exactamente una solución en $I$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Siguiendo el método denominado \emph{variación de
constantes}\index{variación de constantes}, busque soluciones del
formulario $ y = \mu(x)\, \eu^{-A(x)}$ con $\mu$ diferenciable --- no
la generalidad se pierde, ya que cada función en $I$ se puede escribir así
($\mu = y\,\eu^{A}$). Sustituyendo,
\[
y' + ay = \mu' \eu^{-A} - \mu a \eu^{-A} + a\mu\eu^{-A}
= \mu'\, \eu^{-A},
\]
entonces $ y $resuelve $(E)$ si y solo si $\mu'(x) = b(x)\,\eu^{A(x)}$, si
y sólo si $\mu(x) = \lambda + \int_{x_0}^x b(t)\eu^{A(t)}\dd t$ para
alguna constante $\lambda $(dos primitivas del mismo continuo
función en un intervalo difieren por una constante).

(2) En la fórmula, $ y(x_0) = \lambda\,\eu^{-A(x_0)}$: la condición
$ y(x_0) = y_0$ determina $\lambda = y_0 \eu^{A(x_0)}$ de forma única.
\end{proof}

\begin{example}\label{ex:b1:diffeq:order1}
Resuelva $ y' + \dfrac{y}{x} = x^2$ en $I = \intoo{0}{+\infty}$. aquí
$a(x) = \frac 1x$,$A(x) = \ln x$,$\eu^{-A(x)} = \frac 1x$.
Soluciones homogéneas: $\frac{\lambda}{x}$. \omterm{thm:b1:diffeq:voc}{Variación de constantes}:
$\mu'(x) = x^2 \cdot x = x^3$, entonces $\mu = \frac{x^4}{4} + \lambda $, y
\[
y(x) = \frac{x^3}{4} + \frac{\lambda}{x}, \qquad \lambda \in \R .
\]
Con la condición inicial $ y(1) = 0$:$\lambda = -\frac14$.
\emph{Controlar:}$ y' + \frac yx = \frac{3x^2}{4} - \frac{\lambda}{x^2} +
\frac{x^2}{4} + \frac{\lambda}{x^2} = x^2$.
\end{example}

\begin{example}[Adivinando tiempos integrando]\label{ex:b1:diffeq:guess}
Resuelva $ y' + 2x\,y = x$ en $\R$. \omterm{thm:b1:diffeq:voc}{Variación de constantes} funciona ($A =
x^2$,$\mu' = x\,\eu^{x^2}$,$\mu = \frac12\eu^{x^2} + \lambda $),
pero observando que el \emph{constante} $y_p = \frac12$ resuelve el
La ecuación ($0 + 2x\cdot\frac12 = x$) es más rápida. con el
soluciones homogéneas $\lambda\,\eu^{-x^2}$:
\[
y(x) = \frac12 + \lambda\,\eu^{-x^2}, \qquad \lambda \in \R .
\]
Cada solución converge a $\frac12$ extremadamente rápido como $x \to
\pm\infty $: la solución particular constante es una
\emph{equilibrio} que todas las soluciones se unan. La idea: antes
Al iniciar el método general, dedica diez segundos a buscar un
solución particular obvia (constante, monomio, múltiplo de la
lado derecho); el teorema de la estructura entonces termina el trabajo.
\end{example}

\begin{remark}[Los intervalos importantes]
El teorema vive en un \emph{intervalo} donde $a$ y $b$ son
continuo. Para $ y' + \frac yx = 0$ en $\R^*$, las soluciones son
$\frac{\lambda}{x}$ en $\intoo{0}{+\infty}$ y
$\frac{\mu}{x}$ en $\intoo{-\infty}{0}$ con \emph{independiente}
constantes: no hay razón para que una única fórmula se pegue a través de la
singularidad en $0$.
\end{remark}

\begin{example}[Un lado derecho complejo, dos respuestas reales]\label{ex:b1:diffeq:complexrhs}
Resuelve $ y' - y = \cos x$ y $ y' - y = \sin x$ de un solo golpe. trabajo
en $\C$ con el lado derecho $\eu^{\iu x}$: probando $y_p =
c\,\eu^{\iu x}$ da $c(\iu - 1)\eu^{\iu x} = \eu^{\iu x}$, entonces
\[
c = \frac1{\iu - 1} = \frac{-1 - \iu}2,
\qquad
y_p = -\frac{(1 + \iu)(\cos x + \iu\sin x)}2
= \frac{\sin x - \cos x}2 + \iu\,\frac{-\sin x - \cos x}2 .
\]
Dado que la ecuación tiene coeficientes reales, reales e imaginarios.
división de partes: $\frac{\sin x - \cos x}2$ resuelve$ y' - y = \cos x$,
y $-\frac{\sin x + \cos x}2$ resuelve$ y' - y = \sin x $(verifique
la primera: derivada $\frac{\cos x + \sin x}2$, menos la
función, da $\cos x$). Una línea compleja reemplazó dos tramos de
\omterm{thm:b1:diffeq:voc}{variación de constantes} --- la misma economía que
\cref{met:b1:diffeq:particular} sistematiza para segundo orden,
y un dividendo recurrente de \cref{ch:b1:complex}.
\end{example}

\section{Ecuaciones lineales de segundo orden con coeficientes constantes.}

\begin{definition}\label{def:b1:diffeq:linear2}
Sea $a, b \in \R$ y $f \colon I \to \R$ continúen. la ecuacion
\[
(E)\colon\quad y'' + a\,y' + b\,y = f(x)
\]
es un \emph{ecuación lineal de segundo orden con constante
coeficientes}\index{ecuación diferencial!lineal de segundo orden};
$(H)\colon y'' + ay' + by = 0$ es su ecuación homogénea, y
$\chi(r) = r^2 + ar + b$ es \emph{característica
polinomio}\index{polinomio característico}.
\end{definition}

\begin{theorem}[Soluciones homogéneas]\label{thm:b1:diffeq:homogeneous2}
Sea $\Delta = a^2 - 4b$ el discriminante de $\chi$. lo real
Las soluciones de $(H)$ en $\R$ son:
\begin{enumerate}
 \item si $\Delta > 0$, con $r_1 \neq r_2$ las dos raíces reales:
 $\;y = \lambda\, \eu^{r_1 x} + \mu\, \eu^{r_2 x}$;
 \item si $\Delta = 0$, con $r_0$ la doble raíz:
 $\;y = (\lambda + \mu x)\, \eu^{r_0 x}$;
 \item si $\Delta < 0$, con raíces $\alpha \pm \iu\omega $
 ($\omega > 0$):
 $\;y = \eu^{\alpha x} (\lambda \cos\omega x + \mu
 \sin\omega x)$;
\end{enumerate}
en cada caso con $(\lambda, \mu)$ ejecutándose sobre $\R^2$.
\end{theorem}

\begin{proof}
Obsérvese primero que, para $r \in \C$, la función
$x \mapsto \eu^{rx}$ resuelve $(H)$ si y solo si $\chi(r) = 0$
(sustituir: $(r^2 + ar + b)\eu^{rx} = 0$). Por eso las
exponenciales son la primera conjetura natural: la diferenciación
actúa sobre $\eu^{rx}$ como multiplicación por el número $r$, y la
ecuación diferencial se reduce a la ecuación numérica
$\chi(r) = 0$ --- todo el problema analítico se comprime en hallar
las raíces de una cuadrática.

El paso clave es un cambio de incógnita que reduce el orden. Sea $r$
una raíz (posiblemente compleja) de $\chi$ y escriba$ y = z\, \eu^{rx}$,
que no pierde generalidad. entonces
\[
y'' + ay' + by
= \bigl(z'' + (2r + a) z' + \chi(r) z\bigr)\eu^{rx}
= \bigl(z'' + (2r + a) z'\bigr)\eu^{rx},
\]
entonces $(H)$ se convierte en la ecuación \emph{primer orden}$u' + (2r + a) u = 0$
para $u = z'$.

\emph{Case $\Delta \neq 0$:} elija $r = r_1$; luego $2r_1 + a = r_1 -
r_2$(desde $r_1 + r_2 = -a $). Por
\cref{thm:b1:diffeq:homogeneous1}, $z' = c\,\eu^{(r_2 - r_1)x}$ para
alguna constante $c$; integrándose en $\R$,$z = \mu\, \eu^{(r_2 - r_1)x}
+ \lambda $ con $\mu = \frac{c}{r_2 - r_1}$ y, por lo tanto,$ y = z\,
\eu^{r_1 x} = \lambda\,\eu^{r_1x} + \mu\,\eu^{r_2x}$.
Cuando $\Delta < 0$, las raíces son $\alpha \pm \iu\omega $ y el
Las soluciones complejas son $ y = c_1\eu^{(\alpha+\iu\omega)x} +
c_2\eu^{(\alpha-\iu\omega)x}$ con $ c_1, c_2 \in \C $. cual de
¿Estos son de valor real? Dado que $\conj{\eu^{(\alpha+\iu\omega)x}} =
\eu^{(\alpha-\iu\omega)x}$, el conjugado de $ y $ es $\conj{c_2}\,
\eu^{(\alpha+\iu\omega)x} + \conj{c_1}\,\eu^{(\alpha-\iu\omega)x}$,
y $ y = \conj y $ para todas las fuerzas $x$ $c_2 = \conj{c_1}$(las dos
los exponenciales son linealmente independientes: evaluar en dos puntos, o
comparar en $x=0$ después de dividir por $\eu^{\alpha x}$). Escribiendo $c_1
= \frac{\lambda - \iu\mu}2$ con $\lambda, \mu $ real:
\[
y = 2\,\Re\Bigl(c_1\,\eu^{\alpha x}(\cos\omega x +
\iu\sin\omega x)\Bigr)
= \eu^{\alpha x}\bigl(\lambda\cos\omega x + \mu\sin\omega
x\bigr),
\]
y a la inversa, cada función de este tipo es una solución (parte real de una
solución compleja de una ecuación real): el espacio de solución real es
como se anunció.

\emph{Case $\Delta = 0$:}$r = r_0$,$2r_0 + a = 0$, entonces $z'' = 0$:
$z = \lambda + \mu x$ y $ y = (\lambda + \mu x)\eu^{r_0 x}$.
\end{proof}

\begin{example}[Un problema de Cauchy, de principio a fin]\label{ex:b1:diffeq:cauchy2}
Resuelva $ y'' - 3y' + 2y = 0$ con $ y(0) = 0$,$ y'(0) = 1$. el
\omterm{def:b1:diffeq:linear2}{polinomio característico} $r^2 - 3r + 2 = (r - 1)(r - 2)$ tiene el
raíces reales $1$ y $2$: solución general$ y = \lambda\eu^{x} +
\mu\eu^{2x}$. Las dos condiciones dan el sistema lineal.
\[
\lambda + \mu = 0, \qquad \lambda + 2\mu = 1 ,
\]
entonces $\mu = 1$,$\lambda = -1$:
\[
y(x) = \eu^{2x} - \eu^{x} .
\]
Verificar: $ y(0) = 0$;$ y' = 2\eu^{2x} - \eu^x$ tiene $ y'(0) = 1$; y
$ y'' - 3y' + 2y = (4 - 6 + 2)\eu^{2x} + (-1 + 3 - 2)\eu^x = 0$.
Observe la forma de la respuesta: cerca de $-\infty$ el modo lento
$-\eu^x$ domina; cerca de $+\infty $lo hace el modo rápido $\eu^{2x}$.
Lectura de soluciones como superposiciones de modos con diferente desintegración.
o tasas de crecimiento es el hábito rentable: así es como
Se organiza la división transitoria/estado estable del problema del fin de semana.
\end{example}

\begin{theorem}[Estructura y problema de Cauchy]\label{thm:b1:diffeq:structure2}
\begin{enumerate}
 \item Si $y_p$ es una solución particular de $(E)$, las soluciones de
 $(E)$ son exactamente $y_p + y_h$,$y_h$ corriendo sobre el
 soluciones de $(H)$.
 \item (Superposición) Si $y_1$ resuelve$ y'' + ay' + by = f_1$ y
 $y_2$ resuelve$ y'' + ay' + by = f_2$, luego $y_1 + y_2$ resuelve
 la ecuación con el lado derecho $f_1 + f_2$.
 \item Para todos los $x_0 \in I$ y $(y_0, y_0')$, el \omterm{thm:b1:diffeq:voc}{problema de cauchy}
 ``$(E)$,$ y(x_0) = y_0$,$ y'(x_0) = y_0'$'' tiene exactamente uno
 solución en $I$. \emph{(La existencia concede un particular
 solución; unicidad en su totalidad.)}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) $ y $resuelve $(E)$ si $ y - y_p$ resuelve $(H)$, por linealidad de $ y
\mapsto y'' + ay' + by $. (2) es la misma linealidad.

(3) Por (1) basta demostrar que las constantes $(\lambda,
\mu)$ siempre se pueden ajustar, de forma única, a cualquier dato $(y_0, y_0')$.
Al traducir la variable, se supone $x_0 = 0$. En el caso (1) de
\cref{thm:b1:diffeq:homogeneous2}, $ y = \lambda\eu^{r_1x} +
\mu\eu^{r_2x}$ da
\[
y(0) = \lambda + \mu, \qquad y'(0) = r_1\lambda + r_2\mu :
\]
un sistema lineal en $(\lambda, \mu)$ cuyo determinante es $r_2 -
r_1 \neq 0$; resolviéndolo explícitamente,$\mu = \frac{y_0' -
r_1y_0}{r_2 - r_1}$ y $\lambda = y_0 - \mu $: exactamente uno
solución. En el caso (2), $ y(0) = \lambda$ y $ y'(0) = r_0\lambda +
\mu $: el sistema es triangular con determinante $1$, resuelto por
$\lambda = y_0$,$\mu = y_0' - r_0y_0$. En el caso (3),$ y(0) =
\lambda $ y $ y'(0) = \alpha\lambda + \omega\mu$: determinante
$\omega \neq 0$, resuelto por $\lambda = y_0$,$\mu = \frac{y_0' -
\alpha y_0}\omega $. En cada caso, \omterm{def:b1:logic:map}{aplicación}$(\lambda, \mu) \mapsto
(y(x_0), y'(x_0))$ es una biyección lineal --- el lenguaje de
\cref{ch:b1:linmaps} comprimirá este registro de caso en uno
frase.
\end{proof}

\begin{method}[Solución particular para $f(x) = P(x)\,\eu^{\gamma x}$]\label{met:b1:diffeq:particular}
Cuando el lado derecho es $P(x)\,\eu^{\gamma x}$ con $P$ a
polinomio y $\gamma \in \R$(esto cubre polinomios, exponenciales,
y vía complejo $\gamma$ o superposición,$\cos$ y $\sin $): mira
para una solución particular de la forma
\[
y_p(x) = x^{m}\, Q(x)\, \eu^{\gamma x},
\qquad
m = \text{multiplicity of } \gamma \text{ as a root of } \chi
\ (m = 0, 1 \text{ or } 2),
\]
siendo $Q$ un polinomio del mismo grado que $P$, cuyos coeficientes
se encuentran por sustitución e identificación. Para $f = K\cos\omega x$
(o $\sin $), resuelve con el lado derecho $K\eu^{\iu\omega x}$ y toma
la parte real (resp.\ imaginaria).
\end{method}

\begin{example}[Superposición en acción]\label{ex:b1:diffeq:superposition}
Resuelva $ y'' - y = \eu^{x} + 4$ en $\R$. Homogéneo:$\chi(r) = r^2
- 1$, raíces $\pm1$, entonces $y_h = \lambda\eu^x + \mu\eu^{-x}$. dividir
el lado derecho y trate cada pieza según el cuadro de método.
\emph{Piece $\eu^x$:} aquí $\gamma = 1$ es una raíz simple de
$\chi$, entonces prueba $y_1 = c\,x\,\eu^x$: luego $y_1'' - y_1 = c(x +
2)\eu^x - cx\eu^x = 2c\,\eu^x $, dando $c = \frac12$.
\emph{Pieza $4$:}$\gamma = 0$ no es una raíz; la constante $y_2 =
-4$ funciona. Por superposición
(\cref{thm:b1:diffeq:structure2}~(2)):
\[
y = \frac{x\,\eu^x}2 - 4 + \lambda\,\eu^x + \mu\,\eu^{-x},
\qquad (\lambda, \mu) \in \R^2 .
\]
Observe cómo las dos piezas exigieron formas \emph{diferente} ($m = 1$
versus $m = 0$): la prueba de multiplicidad se aplica a cada
exponente por separado, que es el objetivo de dividir el
lado derecho antes de adivinar.
\end{example}

\begin{example}[La regla de la multiplicidad en acción.]\label{ex:b1:diffeq:multiplicity}
Resuelva $ y'' + y' = x$ en $\R$. El lado derecho es $P(x)\eu^{0
\cdot x}$ con $ P(x) = x $ y $\gamma = 0$ es una raíz \emph{simple}.
de $\chi(r) = r^2 + r = r(r + 1)$: entonces $m = 1$, y el correcto
supongo que es $y_p = x\,(\alpha x + \beta) = \alpha x^2 + \beta x$, uno
grado superior a $P$. Sustituyendo:
\[
y_p'' + y_p' = 2\alpha + (2\alpha x + \beta)
= 2\alpha x + (2\alpha + \beta) ,
\]
y la identificación con $x$ da $\alpha = \frac12$,$\beta = -1$:
$y_p = \frac{x^2}2 - x$. Solución general:$ y = \frac{x^2}2 - x +
\lambda + \mu\,\eu^{-x}$. ¿Habíamos adivinado $y_p = \alpha x + \beta $?
(ignorando la multiplicidad), la sustitución daría $y_p'' + y_p'
= \alpha $, una constante --- ninguna elección de $\alpha, \beta $ puede igualar
$x$, y la falla es estructural: las constantes ya resuelven el
ecuación homogénea, por lo que son invisibles para el lado izquierdo.
El factor $x^m$ existe precisamente para salir de lo homogéneo
espacio de solución.
\end{example}

\begin{remark}[La trigésima segunda póliza de seguro]\label{rem:b1:diffeq:check}
Cada ecuación resuelta en este capítulo termina con una sustitución.
Compruébelo, y esto no es decorativo. Una ecuación diferencial
El cálculo encadena muchos pequeños pasos (una primitiva, una regla del producto,
dos constantes), y un error de signo único se propaga de forma invisible;
sustituir la fórmula final nuevamente en la ecuación captura
esencialmente todos ellos a costa de una diferenciación.
Cultivar el reflejo en tres capas: verificar el \emph{particular
solución} solo (la parte homogénea se cancela de todos modos), verificar el
\emph{condiciones iniciales} en la solución completa, y cuando un
parámetro está presente, verifique un \emph{valor degenerado} (¿el
fórmula general $\Omega$ reproduce la respuesta conocida en
$\Omega = 0$?). El hábito cuesta medio minuto; se convierte
"probablemente correcto" en "verificado".
\end{remark}

\begin{remark}[Errores comunes]\label{rem:b1:diffeq:pitfalls}
\begin{enumerate}
 \item \emph{Normalizar primero.} Las fórmulas asumen la ecuación
 lee $ y' + a(x)y = b(x)$--- coeficiente $1$ en $ y'$.
 Para $xy' - 2y = x^3$, divida por $x $(en un intervalo
 evitando $0$) antes de identificar $a$ y $b$, como en
 \cref{exo:b1:diffeq:2}.
 \item \emph{Una constante por dimensión, fijada al final.} El
 solución general de primer orden lleva una constante, la
 segundo orden uno dos; Se imponen condiciones iniciales.
 la solución \emph{completo} $y_p + y_h$, nunca en $y_h$
 solo --- imponerlos antes de agregar $y_p$ es lo más
 Error estructural frecuente.
 \item \emph{Cuidado con la multiplicidad.} Una suposición de solución particular
 que resuelve la ecuación homogénea es invisible para el
 lado izquierdo; el factor $x^m$ de
 \cref{met:b1:diffeq:particular} no es opcional
 (\cref{ex:b1:diffeq:multiplicity}).
 \item \emph{Los intervalos son parte de la respuesta.} Soluciones en vivo
 en intervalos donde los coeficientes son continuos;
 pegar a través de una singularidad puede crear constantes espurias
 (\cref{exo:b1:diffeq:12}) o destruir la unicidad. ``Resolver
 en $\R^*$'' significa dos problemas independientes.
\end{enumerate}
\end{remark}

\begin{example}[Forzado fuera de resonancia]\label{ex:b1:diffeq:offresonance}
Resuelva $ y'' + 4y = \sin x$ en $\R$. Frecuencia natural $2$, forzando
frecuencia $1$: dado que $\iu$ es \emph{no} una raíz de $\chi(r) = r^2
+ 4$, la multiplicidad es $m = 0$ y basta con una sinusoide simple.
Probando $y_p = \alpha\sin x $(no se necesita coseno: la ecuación tiene
no hay término $ y'$ y $\sin $regenera $\sin $):
\[
y_p'' + 4y_p = -\alpha\sin x + 4\alpha\sin x = 3\alpha\sin x ,
\]
entonces $\alpha = \frac13$ y la solución general es
\[
y = \frac{\sin x}3 + \lambda\cos 2x + \mu\sin 2x .
\]
Toda solución permanece acotada: una superposición de dos oscilaciones
en las frecuencias $1$(forzada) y $2$(natural). Comparar con
el siguiente ejemplo, donde forzar \emph{at} la frecuencia natural
cambia la forma misma de la respuesta.
\end{example}

\begin{example}[Una oscilación forzada]\label{ex:b1:diffeq:oscillation}
Resuelva $ y'' + y = \cos x$,$ y(0) = 0$,$ y'(0) = 0$.

\emph{Homogéneo:} $\chi(r) = r^2 + 1$, raíces $\pm\iu $:$y_h =
\lambda\cos x + \mu \sin x $.

\emph{Particular:} lado derecho $\Re(\eu^{\iu x})$ con $\gamma =
\iu $ una raíz simple de $\chi$: prueba $z_p = c\, x\, \eu^{\iu x}$
($c \in \C$). Entonces $z_p'' + z_p = c\,(2\iu)\eu^{\iu x}$, que es igual
$\eu^{\iu x}$ para $c = \frac{1}{2\iu} = -\frac\iu2$. Entonces $z_p =
-\frac{\iu}{2} x (\cos x + \iu \sin x)$ y $ y_p = \Re(z_p) =
\frac{x \sin x}{2}$.

\emph{Solución general:}$ y = \frac{x\sin x}{2} + \lambda\cos x +
\mu\sin x $. Condiciones:$ y(0) = \lambda = 0$;$ y' = \frac{\sin x +
x\cos x}{2} + \mu\cos x $, entonces $ y'(0) = \mu = 0$. Respuesta:$ y =
\frac{x\sin x}{2}$ --- una oscilación cuya amplitud crece linealmente:
el fenómeno \emph{resonancia}\index{resonancia}, causado por el forzamiento
el sistema a su frecuencia natural.
\end{example}

\begin{remark}[Interludio: la linealidad es una geometría]\label{rem:b1:diffeq:interlude}
Vuelva a mirar la forma de cada solución \omterm{def:b1:logic:sets}{conjunto} en este capítulo: a
solución especial más un espacio de soluciones homogéneas con una
constante libre (primer orden) o dos (segundo orden). Los capítulos sobre
álgebra lineal (\cref{ch:b1:vspaces,ch:b1:findim,ch:b1:linmaps})
proporcionará el exacto
vocabulario: la \omterm{def:b1:logic:map}{aplicación} $L(y) = y'' + ay' + by$ es \emph{lineal}, es
soluciones homogéneas forman el \emph{núcleo} de $L$, un vector
espacio cuyo \emph{dimensión} es igual al orden de la ecuación ---
ese es el contenido honesto de "una constante por orden" --- y
la solución \omterm{def:b1:logic:sets}{conjunto} de $L(y) = f$ es una \emph{subespacio afín}, una
traducir del núcleo. Incluso el Cauchy \omterm{def:b1:logic:map}{aplicación} $(\lambda, \mu)
\mapsto (y(x_0), y'(x_0))$ de \cref{thm:b1:diffeq:structure2} es un
biyección lineal entre dos planos, es decir, un sistema $2
\times 2$ invertible (\cref{ch:b1:matrices}). Nada en este capitulo
será necesario rehacerlo, solo será necesario cambiarle el nombre y
el cambio de nombre es el mejor calentamiento posible para el álgebra lineal:
cada definición abstracta que existe ya se ha ganado la vida
aquí.
\end{remark}

\begin{remark}[Dónde se utiliza este capítulo]\label{rem:b1:diffeq:whereused}
El teorema de la estructura --- soluciones de $(E)$ forman "un particular
solución más las soluciones de $(H)$'' --- es la primera aparición
de un patrón que \cref{ch:b1:vspaces,ch:b1:linmaps} nombrará: el
La solución \omterm{def:b1:logic:sets}{conjunto} de $(H)$ es la \emph{núcleo} del \omterm{def:b1:logic:map}{aplicación}$ y
\mapsto y'' + ay' + by $lineal, y la solución \omterm{def:b1:logic:sets}{conjunto} de $(E)$ es una
traducción afín del mismo. El \omterm{def:b1:diffeq:linear2}{polinomio característico} reaparece como
el \omterm{def:b1:diffeq:linear2}{polinomio característico} de una matriz en \cref{ch:b1:matrices}:
una ecuación de segundo orden es un sistema de primer orden $2 \times 2$ en
disfraz, un punto de vista que sistematiza el volumen del Año 2. el
integrales demandadas por \omterm{thm:b1:diffeq:voc}{variación de constantes} son suministradas por
\cref{ch:b1:integration}, y el problema del fin de semana a continuación --- el
oscilador amortiguado accionado --- es el caso modelo para cada
La cuestión de la oscilación en las ciencias, desde los circuitos hasta la suspensión.
puentes.
\end{remark}

\section{Problema: el oscilador amortiguado accionado}

\begin{problem}\label{pb:b1:diffeq:1}
Una ecuación gobierna una masa sobre un resorte en un medio viscoso, la
carga en un circuito RLC y un edificio meciéndose con el viento:
\[
(E_\Omega)\colon\quad
x'' + 2\lambda x' + \omega_0^2\,x = A\cos(\Omega t),
\]
con $\lambda \geq 0$ la amortiguación,$\omega_0 > 0$ la natural
frecuencia, y $A > 0$,$\Omega > 0$ la amplitud y frecuencia de
el forzamiento. Este problema extrae su comportamiento completo: el decaimiento
de transitorios, el único estado estable periódico, el \omterm{ex:b1:diffeq:oscillation}{resonancia}
curva y su nitidez (el \emph{\omterm{pb:b1:diffeq:1}{factor de calidad}}), los latidos de
el caso no amortiguado, y el balance energético que sostiene el
oscilación.\index{resonancia}\index{oscilador}
A menos que se indique lo contrario, $0 < \lambda < \omega_0$(subamortiguado
régimen) y escribimos $\omega_d = \sqrt{\omega_0^2 - \lambda^2}$.

\medskip
\noindent\textbf{Parte I --- The free oscillator.} Aquí $A = 0$.
\begin{enumerate}
 \item Resuelva la ecuación homogénea $(H)$ para $0 < \lambda <
 \omega_0$ y para $\lambda = 0$. (Los regímenes $\lambda
 \geq \omega_0$ fueron tratados en \cref{exo:b1:diffeq:9};
 citarlos.)
 \item Muestre que para cada $\lambda > 0$, todas las soluciones de $(H)$
 tienden a $0$ en $+\infty $ --- en los tres regímenes.
 \item Defina la energía $\mathcal E(t) = \frac12 x'(t)^2 +
 \frac12\omega_0^2\,x(t)^2$ junto con una solución de $(H)$. Mostrar
 $\mathcal E'(t) = -2\lambda\,x'(t)^2 \leq 0$, y deducir
 (sin resolver nada) que el \omterm{thm:b1:diffeq:voc}{problema de cauchy}
 ``$(H)$,$x(t_0) = x'(t_0) = 0$'' tiene solo el cero
 solución, para cada $\lambda \geq 0$.
 \item Para $0 < \lambda < \omega_0$, escriba la solución distinta de cero como
 $x(t) = R\,\eu^{-\lambda t}\cos(\omega_d t - \varphi)$ y
 sea $T_d = \frac{2\pi}{\omega_d}$ el pseudoperíodo.
 Demuestre que $x(t + T_d) = \eu^{-\lambda T_d}\,x(t)$: cada uno
 swing es el anterior reducido por el factor constante
 $\eu^{-\delta}$,$\delta = \frac{2\pi\lambda}{\omega_d}$
 (el \emph{decremento logarítmico}). Calcular $\delta$ para
 $\omega_0 = 1$,$\lambda = 0.1$.
 \item Defina \emph{\omterm{pb:b1:diffeq:1}{factor de calidad}} $Q =
 \dfrac{\omega_0}{2\lambda}$\index{factor de calidad}. Mostrar
 que después del tiempo $\frac1\lambda $(una amplitud
 $\eu$-plegado), el oscilador ha completado
 $\frac{\omega_d}{2\pi\lambda}$ pseudoperíodos, que para
 la amortiguación débil ($\lambda \ll \omega_0$) es aproximadamente
 $\frac Q\pi$: el \omterm{pb:b1:diffeq:1}{factor de calidad} cuenta, hasta el $\pi$, el
 las oscilaciones sobrevivieron antes de que la amplitud decayera por
 $\eu$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The steady state.} Ahora $A > 0$ y
$\lambda > 0$.
\begin{enumerate}[resume]
 \item Busque una solución particular como parte real de
 $z\,\eu^{\iu\Omega t}$ con $z \in \C$
 (\cref{met:b1:diffeq:particular}). Demuestre que esto funciona
 con
 \[
 z = \frac{A}{\omega_0^2 - \Omega^2 + 2\iu\lambda\Omega} .
 \]
 \item Deducir el estado estacionario en forma amplitud-fase:
 $x_p(t) = R(\Omega)\cos\bigl(\Omega t -
 \varphi(\Omega)\bigr)$ con
 \[
 R(\Omega) = \frac{A}{\sqrt{(\omega_0^2 - \Omega^2)^2 +
 4\lambda^2\Omega^2}},
 \qquad
 \tan\varphi = \frac{2\lambda\Omega}{\omega_0^2 -
 \Omega^2},
 \quad \varphi \in \intoo0\pi .
 \]
 \item Interpretar los dos regímenes extremos: calcular los límites de
 $R$ y $\varphi$ como $\Omega \to 0^+$(cuasi-estático
 respuesta $A/\omega_0^2$, fase $0$) y como $\Omega \to
 +\infty $($R \sim A/\Omega^2 \to 0$, fase $\to \pi $: la
 la masa se mueve en sentido contrario a un forzamiento demasiado rápido).
 \item Demuestre que \emph{cada} la solución de $(E_\Omega)$ es $x_p$
 más una solución de $(H)$, por lo tanto converge al estado estacionario
 indique $x_p$ como $t \to +\infty $, cualquiera que sea la inicial
 condiciones: después de que el transitorio muere, el oscilador ha
 No recuerdo cómo empezó.
 \item Demuestre que $x_p$ es la solución periódica \emph{solo} de
 $(E_\Omega)$.
 \item Trabajar un \omterm{thm:b1:diffeq:voc}{problema de cauchy} hasta el final: para $x'' + 2x' + 2x =
 \cos t $ con $x(0) = x'(0) = 0$, demostrar que la solución es
 \[
 x(t) = \frac{\cos t + 2\sin t}5
 - \eu^{-t}\,\frac{\cos t + 3\sin t}5 ,
 \]
 e identificar piezas transitorias y estables.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The \omterm{ex:b1:diffeq:oscillation}{resonancia} curve.} Estudio de
$\Omega \mapsto R(\Omega)$ en $\intoo0{+\infty}$.
\begin{enumerate}[resume]
 \item Configurando $u = \Omega^2$ y $g(u) = (\omega_0^2 - u)^2 +
 4\lambda^2 u $, muestra: si $2\lambda^2 < \omega_0^2$, entonces $R $
 alcanza un máximo estricto en el \emph{\omterm{ex:b1:diffeq:oscillation}{resonancia} frequency}
 $\Omega_r = \sqrt{\omega_0^2 - 2\lambda^2}$, con
 \[
 R_{\max} = R(\Omega_r)
 = \frac{A}{2\lambda\sqrt{\omega_0^2 - \lambda^2}} .
 \]
 \item Demuestre que $\dfrac{R_{\max}}{R(0)} = Q\,\bigl(1 -
 \tfrac{\lambda^2}{\omega_0^2}\bigr)^{-1/2} \geq Q$: en
 \omterm{ex:b1:diffeq:oscillation}{resonancia}, el forzamiento es amplificado por (esencialmente) el
 \omterm{pb:b1:diffeq:1}{factor de calidad}.
 \item Demuestre que la amplitud de \emph{velocidad} $V(\Omega) =
 \Omega\,R(\Omega)$ es máxima exactamente en $\Omega =
 \omega_0$(no en $\Omega_r$), y que la fase allí es
 $\varphi(\omega_0) = \frac\pi2$: en $\Omega = \omega_0$
 la velocidad está exactamente en fase con la fuerza.
 \item (Ancho de banda) Resuelva exactamente $g(u) = 2\,g(u_r)$, donde $u_r =
 \omega_0^2 - 2\lambda^2$, y deduzca que los dos
 frecuencias $\Omega_\pm$ donde $R = R_{\max}/\sqrt2$
 satisfacer $\Omega_+^2 - \Omega_-^2 =
 4\lambda\sqrt{\omega_0^2 - \lambda^2}$; concluir que para
 La amortiguación débil del ancho de banda es $\Omega_+ - \Omega_-
 \approx 2\lambda$, es decir, $Q \approx
 \frac{\omega_0}{\Omega_+ - \Omega_-}$: fuerte \omterm{ex:b1:diffeq:oscillation}{resonancia}.
 Los picos son sistemas altos $Q$.
 \item Retrato numérico para $\omega_0 = 1$,$\lambda = 0.05$
 ($Q = 10$),$A = 1$: calcula $\Omega_r$,$R_{\max}$, el
 respuesta estática $R(0)$ y el ancho de banda aproximado.
 \item Demuestre que si $2\lambda^2 \geq \omega_0^2$, entonces $R$ es
 estrictamente decreciente en $\intoo0{+\infty}$: fuertemente amortiguado
 Los sistemas no tienen ningún pico \omterm{ex:b1:diffeq:oscillation}{resonancia}.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- No damping: beats and \omterm{ex:b1:diffeq:oscillation}{resonancia}.}
Aquí $\lambda = 0$.
\begin{enumerate}[resume]
 \item Para $\Omega \neq \omega_0$, encuentre la solución general de
 $x'' + \omega_0^2 x = A\cos(\Omega t)$.
 \item Resuelve el \omterm{thm:b1:diffeq:voc}{problema de cauchy} $x(0) = x'(0) = 0$ y transforma
 la respuesta en forma de producto
 \[
 x(t) = \frac{2A}{\omega_0^2 - \Omega^2}\,
 \sin\Bigl(\frac{(\omega_0 - \Omega)t}2\Bigr)
 \sin\Bigl(\frac{(\omega_0 + \Omega)t}2\Bigr) .
 \]
 \item Para $\Omega$ cerca de $\omega_0$, lea el producto como
 oscilación rápida en la frecuencia $\frac{\omega_0 + \Omega}2$
 modulado por una envolvente lenta en frecuencia
 $\frac{\abs{\omega_0 - \Omega}}2$: el \emph{late}
 \index{late}. Indique el periodo del sobre y el
 amplitud máxima, y observe cómo ambos explotan como $\Omega
 \to \omega_0$.
 \item Reparar $t$ y dejar $\Omega \to \omega_0$ en la pregunta 19
 fórmula: muestra que el límite es
 \[
 x_\infty(t) = \frac{A\,t\,\sin(\omega_0 t)}{2\omega_0} ,
 \]
 y comprobar directamente que $x_\infty$ resuelve el resonante
 ecuación $x'' + \omega_0^2 x = A\cos(\omega_0 t)$ con
 $x(0) = x'(0) = 0$(comparar
 \cref{ex:b1:diffeq:oscillation}): \omterm{ex:b1:diffeq:oscillation}{resonancia} es el límite
 de ritmos cada vez más lentos y más grandes.
 \item Contraste los dos destinos de \omterm{ex:b1:diffeq:oscillation}{resonancia}: crecimiento lineal
 $\frac{At}{2\omega_0}$ sin amortiguamiento, versus saturación
 en $R_{\max} \approx Q\,\frac{A}{\omega_0^2}$ con débil
 amortiguación. En una frase: ¿qué mecanismo físico convierte?
 el primero en el segundo?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Energy balance and synthesis.}
\begin{enumerate}[resume]
 \item En el estado estacionario de la Parte II, calcule el promedio sobre
 un período $\frac{2\pi}\Omega$ de (a) la potencia inyectada
 por el forzado, $P_{\mathrm{in}}(t) = A\cos(\Omega t)
 \cdot x_p'(t)$, y (b) la potencia disipada por el
 amortiguación, $P_{\mathrm{diss}}(t) = 2\lambda\,x_p'(t)^2$.
 Mostrar ambos promedios iguales $\lambda\,R^2\Omega^2$: el
 forzar alimenta exactamente lo que quema la amortiguación --- esto
 Por eso el estado estacionario es estable.
 \item ¿Dónde exactamente se utilizó el problema: (i) la estructura?
 teorema \cref{thm:b1:diffeq:structure2}; (ii) el complejo
 método exponencial; (iii) un estudio de función de variable real
 al estilo de \cref{ch:b1:functions}? una frase
 cada uno.
 \item Síntesis: describe el comportamiento completo \omterm{def:b1:logic:map}{aplicación} de $(E_\Omega)$
 --- libre versus forzado, amortiguado versus no amortiguado, el papel
 de $Q$ como pico de sintonización de dial unidimensional y adimensional
 altura, ancho de banda y vida útil transitoria --- y mencione
 donde continúa la historia: primer orden $2 \times 2$
 sistemas (\cref{ch:b1:matrices} y el volumen del año 2) y
 la descomposición de un forzamiento periódico general en
 sinusoides (serie de Fourier, en el volumen del año 3), para
 cual el caso sinusoidal de este problema es el fundamental
 bloque de construcción.
\end{enumerate}
\end{problem}
