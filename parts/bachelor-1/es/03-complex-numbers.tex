\chapter{Números complejos}\label{ch:b1:complex}

Los números complejos se encontraron en el volumen de High School como método computacional.
Dispositivo para ecuaciones cuadráticas. Este capítulo los trata como un tema central.
objeto: la forma exponencial y sus consecuencias (de Moivre, $n$-th
raíces, \omterm{def:b1:complex:unity}{raíces de la unidad}), la traducción sistemática entre $\C$ y
geometría plana, y el uso de $\eu^{\iu\theta}$ como máquina para
Demostrar identidades trigonométricas.

\section{El campo $\C$, módulo y conjugado}

\begin{definition}[El campo complejo]\label{def:b1:complex:field}
$\C = \{a + \iu b : a, b \in \R\}$ con la adición habitual y el
multiplicación determinada por $\iu^2 = -1$. Cada $z = a + \iu
b $distinto de cero tiene un inverso:$z^{-1} = \frac{a - \iu b}{a^2 + b^2}$ --- en el
idioma de \cref{ch:b1:structures}, $\C$ es un campo. Se escribe $a =
\Re(z)$,$b = \Im(z)$,$\conj{z} = a - \iu b $(el
\emph{conjugado}\index{conjugado}) y $\abs{z} = \sqrt{a^2 + b^2}$
(el \emph{módulo}\index{módulo}).
\end{definition}

\begin{proposition}[Reglas para conjugado y módulo]\label{prop:b1:complex:rules}
Para $z, w \in \C$:
\begin{enumerate}
 \item $\conj{z + w} = \conj z + \conj w$,$\conj{zw} = \conj z\,
 \conj w$,$\conj{\conj z} = z$;
 \item $z \conj z = \abs z^2$;$\;\Re(z) = \frac{z + \conj z}{2}$,
 $\Im(z) = \frac{z - \conj z}{2\iu}$;
 \item $\abs{zw} = \abs z\, \abs w$ y $\abs{z^{-1}} =
 \abs{z}^{-1}$ para $ z \neq 0$;
 \item (desigualdad triangular\index{desigualdad triangular}) $\abs{z + w}
 \leq \abs z + \abs w$, con igualdad si y sólo si $z $ y
 $w$ se encuentran en un rayo común de $0$($w = \lambda z$ o
 $z = \lambda w$ con $\lambda \geq 0$);
 \item (desigualdad del triángulo inverso) $\bigl|\abs z - \abs w\bigr|
 \leq \abs{z - w}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) y (2) son cálculos directos sobre partes reales e imaginarias. (3):
$\abs{zw}^2 = zw\,\conj{zw} = z\conj z\, w \conj w = \abs z^2 \abs
w^2$, luego saca raíces cuadradas; aplicar a$z \cdot z^{-1} = 1$ para el
inversa.

(4) Ambos lados son no negativos, así que compara cuadrados:
\[
\abs{z+w}^2 = (z+w)(\conj z + \conj w)
= \abs z^2 + \abs w^2 + 2\,\Re(z \conj w),
\]
y $\Re(z\conj w) \leq \abs{z \conj w} = \abs z \abs w$ da
$\abs{z+w}^2 \leq (\abs z + \abs w)^2$. La igualdad obliga a$\Re(z \conj
w) = \abs{z \conj w}$, es decir $z \conj w \in \R_+$; si $w \neq 0$ esto
da $z = \frac{z\conj w}{\abs w^2}\, w = \lambda w$ con $\lambda
\geq 0$(y el caso $w = 0$ es trivial).

(5) $\abs z = \abs{(z - w) + w} \leq \abs{z-w} + \abs w$ da $\abs z
- \abs w \leq \abs{z - w}$; intercambie $z $ y $w$ por el otro signo.
\end{proof}

\begin{example}[Un entrenamiento completo de módulo y argumento.]\label{ex:b1:complex:workout}
Ponga $w = \dfrac{3 + 4\iu}{1 - 2\iu}$ en forma algebraica y calcule
es \omterm{def:b1:complex:field}{módulo} dos veces. Multiplicando por el \omterm{def:b1:complex:field}{conjugado} del
denominador:
\[
w = \frac{(3 + 4\iu)(1 + 2\iu)}{(1 - 2\iu)(1 + 2\iu)}
= \frac{3 + 6\iu + 4\iu - 8}{1 + 4}
= \frac{-5 + 10\iu}{5} = -1 + 2\iu .
\]
Directamente: $\abs w = \sqrt{1 + 4} = \sqrt5$. A través de la regla del cociente
(\cref{prop:b1:complex:rules}~(3)): $\abs w = \frac{\abs{3 +
4\iu}}{\abs{1 - 2\iu}} = \frac5{\sqrt5} = \sqrt5$ --- misma respuesta,
no se necesita forma algebraica. La lección generaliza: módulos y
Los argumentos viajan bien a través de productos y cocientes, reales y
Las partes imaginarias viajan bien a través de sumas. Elige el
representación que coincida con las operaciones en cuestión y convertir
sólo cuando se ve obligado.
\end{example}

\begin{example}[Ecuaciones que involucran el conjugado]\label{ex:b1:complex:conjeq}
Resolver en $\C$:$\;z + 2\conj z = 6 + 2\iu $. Una ecuación que mezcla $z$
y $\conj z$ es el polinomio \emph{no} en $z$; el movimiento confiable
es dividir en coordenadas reales. Con $z = x + \iu y $:
\[
z + 2\conj z = 3x - \iu y ,
\]
entonces la ecuación dice $3x = 6$ y $-y = 2$: la solución única
es $z = 2 - 2\iu$. (Compruebe: $(2 - 2\iu) + 2(2 + 2\iu) = 6 +
2\iu $.) Alternativamente, \omterm{def:b1:complex:field}{conjugado} la ecuación completa para obtener
$\conj z + 2z = 6 - 2\iu$ y resuelve el sistema lineal en el
incógnitas $z, \conj z$ --- misma respuesta y un truco útil cuando
los coeficientes son complejos. Las ecuaciones en $z$ y $\conj z$ son
realmente sistemas de dos ecuaciones reales; esperando ``grado $1$, uno
solución'' es segura aquí, pero $z\conj z = -1$(sin solución) muestra
la intuición polinómica falla tan pronto como aparecen los productos.
\end{example}

\section{forma exponencial}

\begin{definition}[Exponencial complejo de un argumento imaginario]\label{def:b1:complex:expi}
Para $\theta \in \R$ se define
\[
\eu^{\iu\theta} = \cos\theta + \iu \sin\theta .
\]
Cada $z \neq 0$ se puede escribir $z = r\,\eu^{\iu\theta}$ con $r =
\abs z > 0$;$\theta$ es un \emph{argumento}\index{argumento} de $z $,
determinado hasta sumar un múltiplo de $2\pi$. El valor en
$\intoc{-\pi}{\pi}$ es el \emph{argumento principal}, escrito
$\arg z$.
\end{definition}

\begin{example}[Primeros valores y una identidad famosa]\label{ex:b1:complex:eulerid}
Leyendo la definición en los ángulos cardinales:
\[
\eu^{\iu\pi/2} = \iu, \qquad
\eu^{\iu\pi} = -1, \qquad
\eu^{2\iu\pi} = 1, \qquad
\eu^{\iu\pi/4} = \frac{\sqrt2}2\,(1 + \iu) .
\]
El segundo, reordenado como $\eu^{\iu\pi} + 1 = 0$, es el de Euler.
célebre identidad que vincula a $\eu$,$\iu$,$\pi$,$1$ y $0$; en
En esta etapa del libro se trata de una definición más que de una
teorema, y su contenido real --- por qué el exponencial analítico
función de \cref{ch:b1:functions}, extendida a imaginario
argumentos, merece el mismo nombre --- se resuelve por el poder
serie de \cref{ch:b1:series}. Mientras tanto, la pantalla de arriba es
vale la pena memorizar como tabla de conversión: se usa silenciosamente cada
momento en que se lee un argumento en una imagen.
\end{example}

\begin{theorem}[La ecuación funcional]\label{thm:b1:complex:funceq}
Para todos los $\theta, \varphi \in \R$:
\[
\eu^{\iu\theta}\, \eu^{\iu\varphi} = \eu^{\iu(\theta + \varphi)},
\qquad
\abs{\eu^{\iu\theta}} = 1,
\qquad
\conj{\eu^{\iu\theta}} = \eu^{-\iu\theta} = (\eu^{\iu\theta})^{-1}.
\]
En consecuencia, $\abs{zw} = \abs z \abs w$ también incluye los argumentos:
$\arg(zw) \equiv \arg z + \arg w \pmod{2\pi}$.
\end{theorem}

\begin{proof}
Expanda el producto y use las fórmulas de suma:
\[
(\cos\theta + \iu\sin\theta)(\cos\varphi + \iu\sin\varphi)
= (\cos\theta\cos\varphi - \sin\theta\sin\varphi)
+ \iu\,(\sin\theta\cos\varphi + \cos\theta\sin\varphi),
\]
que es $\cos(\theta+\varphi) + \iu\sin(\theta+\varphi)$. El \omterm{def:b1:complex:field}{módulo}
es $\sqrt{\cos^2\theta + \sin^2\theta} = 1$ y la fórmula \omterm{def:b1:complex:field}{conjugado}
es la paridad del coseno y el seno; invierte $\eu^{\iu\theta}$ desde
$\eu^{\iu\theta}\eu^{-\iu\theta} = \eu^0 = 1$.
\end{proof}

\begin{corollary}[fórmula de Moivre]\label{cor:b1:complex:demoivre}
Para $\theta \in \R$ y $n \in \Z$:
$\;(\cos\theta + \iu\sin\theta)^n = \cos(n\theta) +
\iu\sin(n\theta)$.\index{La fórmula de Moivre.}
\end{corollary}

\begin{proof}
Para $n \geq 0$, inducción. El caso $n = 0$ dice $1 = 1$. Suponiendo
la fórmula para $n$, la ecuación funcional
(\cref{thm:b1:complex:funceq}) da
\[
\bigl(\eu^{\iu\theta}\bigr)^{n+1}
= \bigl(\eu^{\iu\theta}\bigr)^{n}\,\eu^{\iu\theta}
= \eu^{\iu n\theta}\,\eu^{\iu\theta}
= \eu^{\iu(n+1)\theta} ,
\]
que es la fórmula en el rango $n + 1$. Para $n < 0$, escriba $n = -m$
con $m > 0$: ya que $\eu^{\iu m\theta}$ tiene inversa
$\eu^{-\iu m\theta}$(\cref{thm:b1:complex:funceq} otra vez),
\[
\bigl(\eu^{\iu\theta}\bigr)^{-m}
= \bigl(\eu^{\iu m\theta}\bigr)^{-1}
= \eu^{\iu(-m)\theta} . \qedhere
\]
\end{proof}

\begin{example}[Ampliando $\cos 3\theta$ por de Moivre]\label{ex:b1:complex:cos3}
Escriba $c = \cos\theta$, $s = \sin\theta$. De Moivre y el binomio
teorema dar
\[
\cos 3\theta + \iu \sin 3\theta = (c + \iu s)^3
= c^3 - 3cs^2 + \iu\,(3c^2 s - s^3),
\]
e identificando piezas reales, sustituyendo luego $s^2 = 1 - c^2$:
\[
\cos 3\theta = c^3 - 3c(1 - c^2) = 4\cos^3\theta - 3\cos\theta .
\]
La parte imaginaria produce $\sin 3\theta = 3\sin\theta -
4\sin^3\theta $ libre: una identidad compleja siempre lleva
\emph{dos} reales. Leído al revés, la identidad del recuadro es la
clave para la ecuación de trisección clásica: construir
$\cos(\theta/3)$ de $\cos\theta $ significa resolver el $4x^3 -
3x = \cos\theta $ cúbico, que es de donde surge el álgebra de
\cref{ch:b1:poly} toma el relevo.
\end{example}

\begin{proposition}[Fórmulas de Euler]\label{prop:b1:complex:euler}
\[
\cos\theta = \frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2},
\qquad
\sin\theta = \frac{\eu^{\iu\theta} - \eu^{-\iu\theta}}{2\iu}.
\]
\end{proposition}

\begin{proof}
Sumar, restar respectivamente, $\eu^{\iu\theta} = \cos\theta +
\iu\sin\theta$ y $\eu^{-\iu\theta} = \cos\theta - \iu\sin\theta$.
\end{proof}

\begin{method}[Trigonometría vía exponenciales]\label{met:b1:complex:trig}
\begin{enumerate}
 \item \emph{Linealizar} $\cos^p\theta \sin^q\theta $(convierte los poderes en
 una suma de $\cos k\theta$, $\sin k\theta$): sustituir la de Euler
 fórmulas, expandir con el teorema del binomio
 (\cref{thm:b1:counting:binomial}), reagrupe los términos \omterm{def:b1:complex:field}{conjugado}.
 \item \emph{Expandir} $\cos n\theta$ como polinomio en $\cos\theta$:
 escriba $\cos n\theta = \Re\bigl((\cos\theta +
 \iu\sin\theta)^n\bigr)$, expanda y convierta potencias pares de
 $\sin $vía $\sin^2 = 1 - \cos^2$.
 \item \emph{Suma de series trigonométricas} como $\sum_k \cos k\theta$:
 Reconocer la parte real de una suma geométrica $\sum_k
 (\eu^{\iu\theta})^k$.
 \item El \emph{factorización de medio ángulo}: para todos los $p, q$,
 \[
 \eu^{\iu p} + \eu^{\iu q}
 = 2 \cos\tfrac{p - q}{2}\; \eu^{\iu \frac{p+q}{2}},
 \qquad
 \eu^{\iu p} - \eu^{\iu q}
 = 2\iu \sin\tfrac{p - q}{2}\; \eu^{\iu \frac{p+q}{2}} .
 \]
\end{enumerate}
\end{method}

\begin{example}[Linealización]\label{ex:b1:complex:linearize}
\[
\cos^3\theta
= \Bigl(\frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2}\Bigr)^{\!3}
= \frac{\eu^{3\iu\theta} + 3\eu^{\iu\theta} + 3\eu^{-\iu\theta} +
\eu^{-3\iu\theta}}{8}
= \frac{\cos 3\theta + 3\cos\theta}{4}.
\]
Esta forma se integra inmediatamente --- la razón misma de la linealización
asuntos en \cref{ch:b1:integration}. Un producto mixto funciona
De la misma manera, sólo que con ambas fórmulas de Euler a la vez:
\[
\sin^2\theta\cos^2\theta
= \Bigl(\frac{\sin2\theta}2\Bigr)^{\!2}
= \frac{1}{4}\cdot
\Bigl(\frac{\eu^{2\iu\theta} - \eu^{-2\iu\theta}}{2\iu}
\Bigr)^{\!2}
= \frac{2 - \eu^{4\iu\theta} - \eu^{-4\iu\theta}}{16}
= \frac{1 - \cos4\theta}{8} ,
\]
donde el atajo de doble ángulo en el primer paso salvó uno
expansión binomial --- siempre vale la pena escanear antes
mecanizando.
\end{example}

\begin{example}[Una suma binomial trigonométrica]\label{ex:b1:complex:binomsum}
Para $n \in \N$ y $\theta \in \R$, evalúe $S =
\sum_{k=0}^{n}\binom nk \cos k\theta $. Reconocer la parte real de
una expansión binomial:
\[
S = \Re\sum_{k=0}^n \binom nk \bigl(\eu^{\iu\theta}\bigr)^k
= \Re\bigl(1 + \eu^{\iu\theta}\bigr)^n ,
\]
luego factoriza el medio ángulo (\cref{met:b1:complex:trig}~(4)): $1 +
\eu^{\iu\theta} = 2\cos\frac\theta2\,\eu^{\iu\theta/2}$, entonces
\[
S = \Re\Bigl(2^n\cos^n\frac\theta2\;\eu^{\iu n\theta/2}\Bigr)
= 2^n \cos^n\frac\theta2\,\cos\frac{n\theta}2 .
\]
La parte imaginaria produce $\sum_k\binom nk\sin k\theta =
2^n\cos^n\frac\theta2\sin\frac{n\theta}2$ libre. Controles de cordura:
$\theta = 0$ recupera $\sum\binom nk = 2^n$ y $\theta = \pi $
da $S = 0$ para $n \geq 1$(cada factor $\cos\frac\pi2$
desaparece), es decir, la suma de filas alternas de
\cref{ex:b1:counting:binomial}. El método --- ``ver el coseno
suma como la sombra de una potencia compleja, luego factoriza los semiángulos''
--- es exactamente el de \cref{exo:b1:complex:6}, con el binomio
Teorema que reemplaza la serie geométrica.
\end{example}

\section{Raíces de números complejos}

\begin{theorem}[$n$-ésima raíz]\label{thm:b1:complex:roots}
Sean $a = r\,\eu^{\iu\alpha} \neq 0$ y $n \in \N^*$. la ecuacion
$z^n = a$ tiene exactamente soluciones $n$:
\[
z_k = r^{1/n}\, \eu^{\iu\left(\frac{\alpha}{n} +
\frac{2k\pi}{n}\right)},
\qquad k = 0, 1, \dots, n - 1 .
\]
\end{theorem}

\begin{proof}
Escriba $z = \rho\,\eu^{\iu\theta}$($\rho > 0$). Entonces $z^n =
\rho^n \eu^{\iu n\theta} = r \eu^{\iu\alpha}$ si y solo si $\rho^n
= r $(módulos) y $n\theta \equiv \alpha \pmod{2\pi}$(argumentos),
es decir,\ $\rho = r^{1/n}$ y $\theta = \frac{\alpha}{n} +
\frac{2k\pi}{n}$ para algunos $k \in \Z $. Queda por ver cuando dos
Los números enteros $k, k'$ producen el mismo número: esto sucede exactamente cuando
los ángulos difieren en un múltiplo de $2\pi$,
\[
\frac{2k\pi}n - \frac{2k'\pi}n \in 2\pi\Z
\iff \frac{k - k'}n \in \Z
\iff n \mid k - k' .
\]
Por división euclidiana, cada $k \in \Z$ es mod $n$ congruente con
exactamente un elemento de $\{0, 1, \dots, n-1\}$, por lo que este rango enumera
cada solución una vez y el recuento es exactamente $n$. (lo mismo
argumento, ejecutado dentro de $\mathbb U_n$, muestra las raíces de una forma regular
$n$-gon: valores consecutivos de $k$ giran según el ángulo fijo
$\frac{2\pi}n$.)
\end{proof}

\begin{example}[Raíces cúbicas de $-27$]\label{ex:b1:complex:cuberoots}
Resuelve $z^3 = -27$. Forma exponencial del lado derecho:$-27 =
27\,\eu^{\iu\pi}$, por lo que las tres raíces son
\[
z_k = 3\,\eu^{\iu(\frac\pi3 + \frac{2k\pi}3)}, \quad k = 0, 1, 2 :
\qquad
z_0 = 3\eu^{\iu\pi/3} = \frac32 + \frac{3\sqrt3}2\,\iu,
\quad
z_1 = -3,
\quad
z_2 = \conj{z_0} .
\]
Dos cheques. Primero, la raíz real $-3$ es la obvia, y la
los otros dos son sus rotaciones por $\pm\frac{2\pi}3$ --- equivalentemente
$-3j$ y $-3j^2$. En segundo lugar, el álgebra confirma:$z^3 + 27 = (z +
3)(z^2 - 3z + 9)$, y la cuadrática tiene el discriminante $9 - 36 =
-27 < 0$ con raíces $\frac{3 \pm 3\iu\sqrt3}2 = z_0, \conj{z_0}$.
La idea: para los lados derechos reales, las raíces no reales siempre
vienen en pares \omterm{def:b1:complex:field}{conjugado}, por lo que se muestra una imagen de la solución \omterm{def:b1:logic:sets}{conjunto}.
simétrico con respecto al eje real --- una vista previa del real
Teorema de factorización de \cref{ch:b1:poly}.
\end{example}

\begin{example}[Un lado derecho no real]\label{ex:b1:complex:fourthroots}
Resuelva $z^4 = -8 + 8\iu\sqrt3$. Forma exponencial del lado derecho:
\omterm{def:b1:complex:field}{módulo} $\sqrt{64 + 192} = 16$, argumento $\theta$ con
$\cos\theta = -\frac12$,$\sin\theta = \frac{\sqrt3}2$, es decir $\theta = \frac{2\pi}3$. Las cuatro raíces son
\[
z_k = 2\,\eu^{\iu(\frac\pi6 + \frac{k\pi}2)}, \quad k = 0, 1, 2,
3 :
\qquad
z_0 = \sqrt3 + \iu,\quad z_1 = \iu z_0 = -1 + \iu\sqrt3,
\]
\[
z_2 = -z_0 = -\sqrt3 - \iu,\qquad z_3 = -\iu z_0 = 1 - \iu\sqrt3 .
\]
(Compruebe: $z_0^2 = 2 + 2\iu\sqrt3$, entonces $z_0^4 = (2 +
2\iu\sqrt3)^2 = 4 - 12 + 8\iu\sqrt3 = -8 + 8\iu\sqrt3$.) Una vez
Se encuentra la raíz \emph{uno}, las otras tres vienen libre: son sus
rotaciones sucesivas por $\frac\pi2$, es decir,\ sus productos con el
cuarto \omterm{def:b1:complex:unity}{raíces de la unidad} --- la estructura general detrás
\cref{thm:b1:complex:roots}, vale la pena explotarlo antes de volver a calcularlo
cada raíz desde cero. Esta vez no hay simetría \omterm{def:b1:complex:field}{conjugado}: el
El lado derecho no es real.
\end{example}

\begin{definition}[Raíces de la unidad]\label{def:b1:complex:unity}
Los \emph{$n$-ésimas raíces de la unidad}\index{raíces de la unidad} son los
soluciones de $z^n = 1$:
\[
\mathbb{U}_n = \bigl\{\, \omega^k : k = 0, \dots, n-1 \,\bigr\},
\qquad \omega = \eu^{2\iu\pi/n}.
\]
Forman un grupo bajo multiplicación (\cref{ch:b1:structures}) y
siéntate en los vértices de un góndono $n$ regular inscrito en el círculo unitario.
\end{definition}

\begin{proposition}[Suma de las raíces de la unidad]\label{prop:b1:complex:sumroots}
Para $n \geq 2$, el $n$-ésimo \omterm{def:b1:complex:unity}{raíces de la unidad} suma cero:
$\sum_{k=0}^{n-1} \omega^k = 0$.
\end{proposition}

\begin{proof}
Suma geométrica con relación $\omega \neq 1$:
$\sum_{k=0}^{n-1} \omega^k = \frac{\omega^n - 1}{\omega - 1} = 0$
desde $\omega^n = 1$.
\end{proof}

\begin{example}[Lectura de partes reales e imaginarias]\label{ex:b1:complex:sumrootsparts}
Dividiendo $\sum_{k=0}^{n-1}\omega^k = 0$ en real e imaginario
partes produce dos identidades trigonométricas de forma gratuita:
\[
\sum_{k=0}^{n-1}\cos\frac{2k\pi}n = 0,
\qquad
\sum_{k=0}^{n-1}\sin\frac{2k\pi}n = 0
\qquad (n \geq 2).
\]
Geométricamente: el centroide de un góno regular $n$ inscrito en el
el círculo unitario es su centro --- los vértices se equilibran exactamente.
Para $n = 5$, la primera identidad da $1 + 2\cos\frac{2\pi}5 +
2\cos\frac{4\pi}5 = 0$(emparejando $k $ con $n - k $), el inicio
punto del cálculo de \cref{exo:b1:complex:8} de
$\cos\frac{2\pi}5$.
\end{example}

\begin{example}[Raíces cuadradas en forma algebraica]\label{ex:b1:complex:sqrt}
Para resolver $z^2 = 3 + 4\iu$ sin trigonometría, configure $z = x + \iu y $:
\[
x^2 - y^2 = 3, \qquad 2xy = 4, \qquad x^2 + y^2 = \abs{3 + 4\iu} = 5 .
\]
Agregando el primero y el último: $x^2 = 4$, entonces $x = \pm 2$, luego $ y = 2/x =
\pm 1$ con el emparejamiento de signos \emph{mismo} ($xy = 2 > 0$):$z = \pm(2 +
\iu)$. Combinado con la fórmula habitual, esto resuelve cada cuadrática
ecuación con coeficientes complejos
(\cref{exo:b1:complex:7}).
\end{example}

\section{Números complejos y geometría plana.}

\begin{proposition}[diccionario geométrico]\label{prop:b1:complex:geometry}
Identificar el punto $M(x, y)$ del avión con su \emph{afijo}$z = x
+ \iu y $\index{afijo}. Para puntos distintos $A, B, C$ de afijos $a, b,
c $:
\begin{enumerate}
 \item $\abs{b - a}$ es la distancia $AB$;
 \item $\arg\dfrac{c - a}{b - a}$ es el ángulo entre los vectores
 $\vect{AB}$ y $\vect{AC}$(mod $2\pi $);
 \item $A, B, C$ están alineados si y sólo si $\dfrac{c - a}{b - a} \in
 \R$; las rectas $AB $ y $AC$ son perpendiculares si y sólo si
 $\dfrac{c - a}{b - a} \in \iu\R$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) es la definición de \omterm{def:b1:complex:field}{módulo} aplicada a $b - a$, el afijo de
$\vect{AB}$. (2): escriba $b - a = r\eu^{\iu\theta}$,$c - a =
s\eu^{\iu\varphi}$; entonces $\frac{c-a}{b-a} = \frac sr
\eu^{\iu(\varphi - \theta)}$ tiene el argumento $\varphi - \theta $, el
ángulo de $\vect{AB}$a$\vect{AC}$. (3): alineación significa ángulo $0$
o $\pi$, es decir argumento en $\pi\Z$, es decir el cociente es real;
perpendicularidad significa ángulo $\pm\frac\pi2$, es decir, el cociente es
puramente imaginario. (El cociente es distinto de cero desde $C \neq A$).
\end{proof}

\begin{remark}[Interludio: $\C$ es el plano con una multiplicación]\label{rem:b1:complex:interlude}
Vale la pena detenerse en lo que hace posible este capítulo:
el avión $\R^2$ lleva adiciones en todas direcciones, pero no
Multiplicación dada por Dios --- y $\C$\emph{is} el avión
equipado con uno, en el que multiplicar por un número fijo
gira y escala. Esta estructura será ordeñada tres más.
veces en este volumen. En \cref{ch:b1:matrices}, la multiplicación por
$a + \iu b$ reaparece como la matriz $2 \times 2$ con filas $(a,
-b)$ y $(b, a)$: la aritmética compleja es la primera, completamente
familia concreta de productos matriciales. En \cref{ch:b1:euclid}, el
la fórmula $\Re(\conj z\,w)$ resulta ser el producto escalar, y
$\abs z$ la norma euclidiana: la desigualdad del triángulo demostrada aquí
Éste es el modelo para la historia de Cauchy-Schwarz allí. y en
\cref{ch:b1:curves}, es mejor escribir un punto en movimiento $t \mapsto
z(t)$, de modo que la velocidad y la aceleración adquieran valores complejos
derivadas --- el movimiento circular, por ejemplo, es simplemente $z(t) =
R\,\eu^{\iu\omega t}$. Una buena multiplicación, cuatro capítulos de
dividendos.
\end{remark}

\begin{proposition}[Aplicaciones $z \mapsto az + b$]\label{prop:b1:complex:similitude}
Sea $a \in \C^*$,$b \in \C$. La transformación $f(z) = az + b$ de
el avión:
\begin{itemize}
 \item es una traducción si $a = 1$;
 \item de lo contrario tiene un punto fijo único $\zeta = \frac{b}{1-a}$,
 y $f(z) - \zeta = a\,(z - \zeta)$:$f$ es la rotación de
 centro $\zeta$ y ángulo $\arg a$, compuestos con la escala
 de centro $\zeta$ y relación $\abs a$.
\end{itemize}
En particular $z \mapsto \eu^{\iu\theta} z$ es la rotación del ángulo.
$\theta$ sobre el origen, y $\conj z$ es el reflejo en lo real.
eje.
\end{proposition}

\begin{proof}
Si $a = 1$,$f(z) = z + b$ se traduce por el vector del afijo $b$. si
$a \neq 1$, la ecuación de punto fijo $z = az + b$ tiene la única
solución $\zeta = \frac{b}{1-a}$ y luego $f(z) - \zeta = az + b -
(a\zeta + b) = a(z - \zeta)$. Escribiendo $a = \abs a\, \eu^{\iu\arg a}$,
la multiplicación por $a$ escala las distancias a$\zeta$ por $\abs a$ y suma
$\arg a$a ángulos en $\zeta$, que es la composición anunciada.
\end{proof}

\begin{example}[Clasificando una aplicación $z \mapsto az+b$]\label{ex:b1:complex:classify}
Tome $f(z) = \iu z + 1$. Aquí $a = \iu \neq 1$: el punto fijo es
\[
\zeta = \frac{b}{1 - a} = \frac1{1 - \iu}
= \frac{1 + \iu}{2},
\]
y desde $\abs a = 1$, no hay ningún escalado:$f$ es el puro
rotación del centro $\frac{1+\iu}2$ y ángulo $\arg \iu = \frac\pi2$.
Verifique: $f(\zeta) = \iu\,\frac{1+\iu}2 + 1 = \frac{\iu - 1}2 + 1 =
\frac{1 + \iu}2 = \zeta $ y $f(0) = 1$,$f(1) = 1 + \iu$,$ f(1 +
\iu) = \iu\,(1 + \iu) + 1 = \iu $: los cuatro puntos $0, 1, 1+\iu, \iu $
del ciclo del cuadrado unitario alrededor de su centro $\zeta$, un cuarto
giro a la vez --- exactamente qué rotación por $\frac\pi2$ sobre el
El centro del cuadrado debe ser suficiente.
\end{example}

\begin{remark}[Errores comunes con módulos y argumentos]\label{rem:b1:complex:pitfalls}
\begin{enumerate}
 \item \emph{No inequalities in $\C$.} Escritura $z \leq w$ para
 los números no reales no tienen sentido; solo módulos, partes reales
 y se pueden comparar partes imaginarias.
 \item \emph{$\sqrt{\phantom z}$ is reserved for $\R_+$.} Cada
 El número complejo distinto de cero tiene \emph{dos} raíces cuadradas y
 ninguno de los dos tiene privilegios: escriba ``sea $\delta$ un cuadrado
 raíz de $\Delta$'' (calculada como en
 \cref{ex:b1:complex:sqrt}), nunca $\sqrt\Delta$ ---
 la regla $\sqrt{ab} = \sqrt a\sqrt b$ ya falla en
 $a = b = -1$.
 \item \emph{Arguments live modulo $2\pi$.} Desde
 $\eu^{\iu\alpha} = \eu^{\iu\beta}$ concluye $\alpha
 \equiv \beta \pmod{2\pi}$, no $\alpha = \beta $;
 olvidando el $2k\pi$ es como solución \omterm{def:b1:logic:sets}{conjuntos} de $z^n = a$
 pierden $n - 1$ de sus elementos $n$.
 \item \emph{$\abs{z + w}$ is not $\abs z + \abs w$.} Igualdad
 en el triángulo la desigualdad es la alineada excepcional
 caso (\cref{prop:b1:complex:rules}~(4)); en general el
 \omterm{def:b1:complex:field}{módulo} de una suma debe estimarse, no calcularse.
\end{enumerate}
\end{remark}

\begin{example}\label{ex:b1:complex:equilateral}
$A, B, C$(afijos $a, b, c$, distintos por pares) forman un equilátero
triángulo con los vértices en orden directo (en sentido antihorario) si y
sólo si $\frac{c - a}{b - a} = \eu^{\iu\pi/3}$: la rotación de
el centro $A$ y el ángulo $\frac\pi3$ envía $B$a$C$. Ambas orientaciones
juntos son capturados por la ecuación simétrica $a^2 + b^2 + c^2 = ab
+ bc + ca $(\cref{exo:b1:complex:10}).
\end{example}

\begin{remark}[Dónde se utiliza este capítulo]\label{rem:b1:complex:whereused}
La forma exponencial es el dispositivo computacional más reutilizado
del volumen. \omterm{def:b1:complex:unity}{Raíces de unidad} se convierte en el ejemplo estándar de un
grupo cíclico en \cref{ch:b1:structures} e impulsa la factorización
de $X^n - 1$ en \cref{ch:b1:poly}; agrupando raíces \omterm{def:b1:complex:field}{conjugado} allí
produce las factorizaciones reales utilizadas por fracciones parciales en
\cref{ch:b1:fractions}. Linealización (\cref{met:b1:complex:trig})
es la preparación estándar para integrar potencias trigonométricas en
\cref{ch:b1:integration}, y las ecuaciones características de
\cref{ch:b1:diffeq} tienen raíces complejas cuyas raíces reales e imaginarias
Las piezas producen las soluciones oscilantes $\eu^{\lambda t}\cos\omega
t$. El diccionario geométrico regresa vestido de matrices: rotaciones
y las similitudes se convierten en las matrices ortogonales de
\cref{ch:b1:matrices,ch:b1:euclid} y curvas parametrizadas en
\cref{ch:b1:curves} suele escribirse mejor como \omterm{def:b1:logic:map}{aplicaciones} $t \mapsto z(t)
\in \C $.
\end{remark}

\section{Problema: el teorema de Napoleón}

\begin{problem}\label{pb:b1:complex:1}
Levanta un triángulo equilátero hacia afuera a cada lado de un
\emph{arbitrario} triángulo: los tres centros de esos triángulos
Siempre forman un triángulo equilátero. Este es el teorema de Napoleón
\index{teorema de napoleón} --- un \omterm{def:b1:logic:statement}{enunciado} sin motivo visible para
ser cierto, lo cual el álgebra de $j = \eu^{2\iu\pi/3}$ demuestra en tres
líneas de cálculo. Este problema construye el conjunto de herramientas completo.
(rotaciones, similitudes directas, el criterio $j$ de
\cref{exo:b1:complex:10}), demuestra el teorema de Napoleón interno y
exterior, localiza el caso degenerado y cierra con una segunda joya
de la misma escuela: teorema de van Schooten sobre triángulos equiláteros
y la desigualdad de Ptolomeo. En todo momento, los puntos del avión son
identificados con sus afijos.

\medskip
\noindent\textbf{Parte I --- The number $j$ and rotations.}
\begin{enumerate}
 \item Calcule la forma algebraica de $j$, luego $j^2$,$j^3$,$1 + j
 + j^2$,$\conj j $ y $j^{-1}$. Coloque $1$,$j$,$j^2$ en un
 bosquejo del círculo unitario.
 \item Demuestre que la rotación del centro $a$ y el ángulo $\theta$ es
 $r(z) = a + \eu^{\iu\theta}(z - a)$, y que el
 composición de dos rotaciones, de ángulos $\theta$ y
 $\theta'$, es una rotación del ángulo $\theta + \theta'$ si
 $\theta + \theta' \notin 2\pi\Z$ y una traducción.
 de lo contrario. (Utilice \cref{prop:b1:complex:similitude}).
 \item Vamos $b \neq c$. Demuestre que hay exactamente dos puntos $p$
 haciendo que $(b, c, p)$ sea equilátero, es decir, $p = b +
 \eu^{\pm\iu\pi/3}(c - b)$. Para un triángulo \emph{directo}
 $(a, b, c)$, verifique en el ejemplo $a = 0$,$b = 1$,$c =
 \iu $ que la opción $\eu^{-\iu\pi/3}$ es la que está en
 el otro lado de la línea $BC$ de $a$ --- el
 \emph{exterior} ápice.
 \item De la solución de \cref{exo:b1:complex:10}, $(a, b, c)$
 es equilátero directo si $a + jb + j^2c = 0$. probar los dos
 identidades de rotación
 \[
 b + jc + j^2a = j^2\,(a + jb + j^2c),
 \qquad
 c + ja + j^2b = j\,(a + jb + j^2c),
 \]
 y deducir que el criterio es invariante bajo condiciones cíclicas.
 \omterm{def:b1:counting:objects}{permutación} de $(a, b, c)$.
 \item Demuestre que si $a + jb + j^2c = 0$ y dos de los tres puntos
 coinciden, entonces los tres coinciden. (Entonces el criterio exactamente
 caracteriza: triángulo equilátero directo, o un solo
 punto.)
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Direct similarities.}
\begin{enumerate}[resume]
 \item Mostrar que el \omterm{def:b1:logic:map}{aplicaciones} $f(z) = az + b$ con $a \in \C^*$
 (\emph{similitudes directas}) son estables bajo composición
 e inversión: la composición de dos de ellos, y la
 inverso de cualquiera de ellos, es nuevamente de esta forma. (En el
 idioma de \cref{ch:b1:structures}: forman un grupo.)
 \item Dados $z_1 \neq z_2$ y $w_1 \neq w_2$, muestra que hay un
 \emph{único} similitud directa $f$ con $f(z_1) = w_1$ y
 $f(z_2) = w_2$ y proporcione $a$ y $b$ explícitamente.
 \item Demuestre que una semejanza directa $f(z) = az + b$ multiplica
 todas las distancias por $\abs a$ y conserva el \emph{forma}
 $\frac{c - a'}{b' - a'}$ de cualquier triángulo $(a', b', c')$\emph{(El numerador y el denominador se multiplican por
 $a$)}. Deducir que dos triángulos son directamente similares
 exactamente cuando sus formas son iguales.
 \item Determinar completamente la similitud directa con $f(0) = 1$.
 y $f(1) = \iu$: proporcione $a$,$b$, el punto fijo, la relación
 y el ángulo.
 \item Vamos $\alpha + \beta = 1$. Demuestre que el "punto ponderado"
 $g(a', b') = \alpha a' + \beta b'$ viaja con cada
 similitud directa: $f(\alpha a' + \beta b') = \alpha f(a')
 + \beta f(b')$. Deduzca que los centroides, los puntos medios y la
 Los centros de Napoleón a continuación están todos transportados por similitudes.
 --- la licencia algebraica detrás de cada "sin pérdida de
 En general, coloque el círculo circunstante en el círculo unitario''
 argumento.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Napoleon's theorem.} Deja $(a, b, c)$
ser un triángulo directo. A cada lado levante el equilátero hacia afuera.
triángulo (pregunta 3) y sean $n_a$,$n_b$,$n_c$ los centros
(centroides) de los triángulos erigidos en $[b, c]$,$[c, a]$,
$[a, b]$ respectivamente.
\begin{enumerate}[resume]
 \item Demuestra que
 \[
 n_a = \alpha b + \beta c, \qquad
 n_b = \alpha c + \beta a, \qquad
 n_c = \alpha a + \beta b,
 \qquad\text{with}\quad
 \alpha = \frac{3 + \iu\sqrt3}{6},\ \beta = \conj\alpha .
 \]
 \item Verifique $\alpha + \beta = 1$ y deduzca que el triángulo
 $(n_a, n_b, n_c)$ tiene el \emph{mismo centroide} como
 $(a, b, c)$.
 \item Utilizando las identidades de la pregunta 4, demuestre que
 \[
 n_a + j\,n_b + j^2 n_c
 = (\alpha j^2 + \beta j)\,(a + jb + j^2c) .
 \]
 \item Calcule $\alpha j + \beta$ y concluya: $\alpha j^2 +
 \beta j = j(\alpha j + \beta) = 0$, por lo que $n_a + j n_b + j^2
 n_c = 0$ para el triángulo \emph{cada}: el Napoleón exterior
 el triángulo es equilátero directo (o un punto). Napoleón
 El teorema está demostrado.
 \item En su lugar, erige los triángulos equiláteros \emph{interior}.
 (opción $\eu^{+\iu\pi/3}$ en la pregunta 3) y deje que $m_a, m_b,
 m_c $ sea sus centros. Mostrar $m_a = \beta b + \alpha c $(y
 cíclicamente), luego pruebe $m_a + j^2 m_b + j\,m_c = 0$: el
 El triángulo interior de Napoleón también es equilátero, con el
 orientación opuesta.
 \item Localice la degeneración: muestre que sucede $n_a = n_b = n_c$
 exactamente cuando $a + j^2 b + jc = 0$, es decir cuando $(a, b, c)$
 es un triángulo equilátero \emph{indirecto}. \emph{(Two
 points among $n_a, n_b, n_c$ coincide iff all three do, by

 \item Realizar todo el cálculo en el triángulo $a = 0$,
 $b = 1$,$c = \iu $: proporcione $n_a, n_b, n_c$ exactamente, y

\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Ptolemy and van Schooten.}
\begin{enumerate}[resume]
 \item Acreditar la identidad, válido para todos los complejos $a, b, c, d$:
 \[
 (a - b)(c - d) + (a - d)(b - c) = (a - c)(b - d) .
 \]
 \item Deduzca \emph{La desigualdad de Ptolomeo}\index{Ptolomeo
 desigualdad}: para cuatro puntos cualesquiera $A, B, C, D$,
 \[
 AC \cdot BD \;\leq\; AB \cdot CD + AD \cdot BC ,
 \]
 con igualdad si y sólo si $(a-b)(c-d)$ y
 $(a-d)(b-c)$ se encuentran en un rayo común de $0$.
 \item Muestre que para $\theta, \varphi \in \R$,
 $\abs{\eu^{\iu\theta} - \eu^{\iu\varphi}} =
 2\,\abs{\sin\frac{\theta - \varphi}2}$.
 \item (teorema de van Schooten\index{teorema de van Schooten}) Sea
 $(a, b, c) = (1, j, j^2)$ --- por la Parte II esto no pierde
 generalidad entre triángulos equiláteros directos --- y dejemos
 $p = \eu^{\iu\theta}$ con $\theta \in
 \intoo{2\pi/3}{4\pi/3}$, un punto del círculo circunstante en el
 arco $BC$ que no contiene $A$. probar
 \[
 PA = PB + PC .
 \]
