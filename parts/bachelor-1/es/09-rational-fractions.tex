\chapter{fracciones racionales}\label{ch:b1:fractions}

A \omterm{def:b1:fractions:field}{fracción racional} es un cociente de \omterm{def:b1:poly:def}{polinomios}. El teorema central
de este breve capítulo --- fracción parcial
descomposición\index{descomposición en fracciones parciales} --- rompe cualquier
such quotient into a sum of elementary bricks $\frac{c}{(X - a)^k}$.
Más allá de su interés algebraico, es la máquina estándar para
integrando funciones racionales (\cref{ch:b1:integration}) y para
sumando ciertas series (\cref{ch:b1:series}).

\section{El campo $K(X)$}

\begin{definition}\label{def:b1:fractions:field}
A \emph{fracción racional}\index{fracción racional} sobre $K$($= \R$
o $\C$) es un cociente $F = \frac{A}{B}$ con $A, B \in K[X]$,$B \neq
0$; Se identifican dos cocientes $\frac AB $ y $\frac{A'}{B'}$ cuando
$AB' = A'B$. Cada fracción tiene un \emph{forma reducida} con
$\gcd(A, B) = 1$, único hasta constantes. Con las operaciones naturales,
el \omterm{def:b1:logic:sets}{conjunto} $K(X)$ de fracciones racionales es un \omterm{def:b1:structures:field}{campo}.

Las \emph{polos}\index{polo} de $F$(en forma reducida) son las raíces de
$B$; el \emph{orden} de un polo es su \omterm{def:b1:poly:derivative}{multiplicidad} como raíz de $B$.
El \emph{grado} de $F$ es $\deg F = \deg A - \deg B \in \Z \cup
\{-\infty\}$.
\end{definition}

\begin{example}[Lectura de polos, órdenes y grados]\label{ex:b1:fractions:vocab}
Sea $F = \dfrac{X^3 - X}{X^4 - 2X^3 + X^2}$. Factoriza ambas capas:
numerador $X(X-1)(X+1)$, denominador $X^2(X-1)^2$; cancelar el
factor común $X(X-1)$:
\[
F = \frac{X + 1}{X(X - 1)} \quad\text{(reduced form).}
\]
\omterm{def:b1:fractions:field}{polacos}: $0$ y $1$, ambos \emph{simple} --- las órdenes se siguen leyendo
el denominador reducido, por lo que las aparentes raíces dobles de la
el original es irrelevante. Grado: $\deg F = 1 - 2 = -1$,
visible asintóticamente ($xF(x) \to 1$ como $x \to \infty $). el
El grado se comporta como el grado \omterm{def:b1:poly:def}{polinomio} ($\deg FG = \deg F +
\deg G $,$\deg(F + G) \leq \max $), una regla contable utilizada
constantemente en el coeficiente caza a continuación: cada ``límite de
El argumento $xF(x)$'' es un recuento de grados disfrazado.
\end{example}

\begin{proposition}[Parte entera]\label{prop:b1:fractions:integerpart}
Cada $F = \frac AB \in K(X)$ es exclusivamente $F = E + \frac RB$ con $E
\in K[X]$(la parte \emph{parte entera}, o \omterm{def:b1:poly:def}{polinomio}, de $F $) y
$\deg R < \deg B$. Uno tiene $E \neq 0$ si $\deg F \geq 0$.
\end{proposition}

\begin{proof}
División euclidiana $A = BE + R$(\cref{thm:b1:poly:division}), dividida
por $B$. Unicidad: si $E + \frac RB = E' + \frac{R'}{B}$, entonces $(E -
E')B = R' - R $ con $\deg(R' - R) < \deg B $, forzando a$E = E'$ y luego a$R
= R'$.
\end{proof}

\begin{example}[Reducir primero, luego dividir]\label{ex:b1:fractions:integerpart}
Encuentre la parte entera de $F = \dfrac{X^3 + 1}{X^2 - 1}$. Dividiendo
a ciegas: $X^3 + 1 = (X^2 - 1)X + (X + 1)$, entonces $F = X +
\frac{X + 1}{X^2 - 1}$. Pero la fracción no se redujo:$X^3 + 1
= (X + 1)(X^2 - X + 1)$ y $ X^2 - 1 = (X+1)(X-1)$ comparten el factor
$X + 1$, y
\[
F = \frac{X^2 - X + 1}{X - 1} = X + \frac{1}{X - 1} :
\]
la misma parte entera $X$, pero la parte fraccionaria se colapsa a una
un solo ladrillo, y el ``\omterm{def:b1:fractions:field}{polo}'' en $-1$ nunca fue un \omterm{def:b1:fractions:field}{polo} en absoluto.
Reduzca siempre a los términos más bajos antes de buscar \omterm{def:b1:fractions:field}{polos}: el \omterm{def:b1:fractions:field}{polos} de
$F$ son las raíces del denominador \emph{reducido}. (El número entero
parte es insensible a la simplificación, ya que la unicidad en
\cref{prop:b1:fractions:integerpart} garantías.)
\end{example}

\section{Descomposición en fracciones parciales sobre $\C$}

\begin{theorem}[Descomposición sobre $\C$]\label{thm:b1:fractions:complex}
Sea $F = \frac AB \in \C(X)$ en forma reducida, con $B = c\,(X -
a_1)^{m_1} \cdots (X - a_r)^{m_r}$. Entonces $F $ es, de forma única,
\[
F = E + \sum_{i=1}^{r} \sum_{k=1}^{m_i}
\frac{c_{i,k}}{(X - a_i)^k},
\qquad E \in \C[X],\ c_{i,k} \in \C ,
\]
siendo $E$ la parte entera de $F$.
\end{theorem}

\begin{proof}
Por \cref{prop:b1:fractions:integerpart} podemos suponer $\deg A < \deg
B $ y probar la descomposición de la suma con $E = 0$.

\emph{Splitting the \omterm{def:b1:fractions:field}{polos}.} Escribe $B = (X - a_1)^{m_1} B_1$ con
$B_1(a_1) \neq 0$. Los \omterm{def:b1:poly:def}{polinomios}$(X-a_1)^{m_1}$ y $B_1$ son
\omterm{cor:b1:arith:bezout}{coprimo} (sin raíz común), por lo que Bézout en $\C[X]$(ver el comentario en
\cref{ch:b1:poly}) hay $U_0, V_0$ con $U_0 (X-a_1)^{m_1} +
V_0 B_1 = 1$; multiplicando por $A $ y configurando $U = AU_0$,$ V =
AV_0$, luego dividiendo por $B $:
\[
\frac AB = \frac{V}{(X - a_1)^{m_1}} + \frac{U}{B_1} .
\]
Las condiciones de grado se pueden aplicar: divida $V$ por
$(X-a_1)^{m_1}$, digamos $V = (X-a_1)^{m_1}Q + V_1$ con $\deg V_1 <
m_1$ y absorba el cociente en el segundo término ($U_1 = U +
QB_1$):
\[
\frac AB = \frac{V_1}{(X - a_1)^{m_1}} + \frac{U_1}{B_1},
\qquad \deg V_1 < m_1 ;
\]
comparando grados ($\deg A < \deg B$ y $\deg V_1 < m_1$) fuerzas
$\deg U_1 < \deg B_1$ también. Iterando en $\frac{U_1}{B_1}$,
\omterm{def:b1:fractions:field}{polo} después de \omterm{def:b1:fractions:field}{polo}, reduce todo al caso \omterm{def:b1:fractions:field}{unipolar} siguiente.

\emph{One \omterm{def:b1:fractions:field}{polo}.} Para $\frac{A}{(X-a)^m}$ con $\deg A < m$: expanda $A$
en potencias de $(X - a)$,$A = \sum_{j=0}^{m-1} \alpha_j (X - a)^j$
(Expansión de Taylor de un \omterm{def:b1:poly:def}{polinomio}, como en la prueba de
\cref{prop:b1:poly:multiplicity}); al dividir se obtienen exactamente los ladrillos
$\frac{\alpha_j}{(X-a)^{m-j}}$. En concreto, para $\frac{X^2 +
1}{(X+2)^3}$: sustituyendo $X = Y - 2$,
\[
X^2 + 1 = (Y - 2)^2 + 1 = Y^2 - 4Y + 5 ,
\qquad\text{so}\qquad
\frac{X^2+1}{(X+2)^3} = \frac1{Y} - \frac4{Y^2} + \frac5{Y^3}
\]
con $Y = X + 2$: los tres ladrillos aparecen por mera división del
expansión desplazada --- la ruta más rápida siempre que un solo
\omterm{def:b1:fractions:field}{polo} de alto orden está involucrado, y el recomendado para
\cref{exo:b1:fractions:3}.

\emph{Unicidad.} Supongamos que coinciden dos descomposiciones; su
la diferencia es una identidad $0 = \sum_{i,k} \frac{d_{i,k}}{(X -
a_i)^k}$. Multiplicar por $(X - a_1)^{m_1}$: cada término
adquiere un factor que desaparece en $a_1$\emph{excepto} el que tiene
$i = 1$,$k = m_1$, cuyo coeficiente pasa a ser $d_{1,m_1}$ más
términos que llevan al menos un factor $(X - a_1)$. Evaluando en
$a_1$(legítimo: después de la multiplicación, no queda ningún \omterm{def:b1:fractions:field}{polo} en
$a_1$) da $d_{1,m_1} = 0$. Sin el coeficiente superior,
repita con $(X - a_1)^{m_1 - 1}$, y así sucesivamente hacia abajo hasta $k = 1$;
luego pase al siguiente \omterm{def:b1:fractions:field}{polo}. Todos los $d_{i,k}$ desaparecen: el
La descomposición es única.
\end{proof}

\begin{method}[Cálculo de los coeficientes]\label{met:b1:fractions:compute}
En la práctica, evite Bézout; combinar:
\begin{enumerate}
 \item \emph{encubrimiento para el poder más alto}: el coeficiente de
 $\frac{1}{(X-a)^m}$(con $m$ el orden del \omterm{def:b1:fractions:field}{polo}$a$) es
 \[
 c_{a,m} = \Bigl[\,(X - a)^m F\,\Bigr]_{X = a};
 \]
 para un \emph{simple} \omterm{def:b1:fractions:field}{polo} de $F = \frac AB$, esto es
 $\frac{A(a)}{B'(a)}$;
 \item \emph{evaluaciones} en puntos convenientes y \emph{límites} de
 $xF(x)$ como $x \to \infty $ para recopilar relaciones lineales para el
 coeficientes restantes;
 \item \emph{paridad} o simetrías de conjugación, cuando estén presentes, para cortar
 el trabajo a la mitad.
\end{enumerate}
\end{method}

\begin{remark}[Errores comunes con fracciones parciales]\label{rem:b1:fractions:pitfalls}
\begin{enumerate}
 \item \emph{Saltándose la parte entera.} La descomposición del ladrillo
 aplica a fracciones de grado $< 0$; cuando $\deg A \geq
 \deg B $, divide primero
 (\cref{prop:b1:fractions:integerpart}), o el
 La búsqueda de coeficientes producirá contradicciones.
 \item \emph{Encubrimiento más allá de su alcance.} Multiplicando por
 $(X - a)^k$ y evaluar en $a$ produce el coeficiente
 sólo para $k = m$, el pedido \emph{lleno} del \omterm{def:b1:fractions:field}{polo}; el
 Los coeficientes de orden inferior requieren otras relaciones (límites,
 evaluaciones) --- ver \cref{ex:b1:fractions:multiple}.
 \item \emph{Olvidarse de reducir.} \omterm{def:b1:fractions:field}{polacos} se leen en el
 formulario \emph{reducido}; un factor común entre el numerador
 y el denominador crea el fantasma \omterm{def:b1:fractions:field}{polos}
 (\cref{ex:b1:fractions:integerpart}).
 \item \emph{Wrong brick shapes over $\R$.} Por encima de un irreductible
 cuadrático los numeradores son \emph{afín} ($\alpha X +
 \beta $), no constantes; escribiendo $\frac{c}{X^2 + 1}$ solo
 pierde soluciones --- las formas correctas son dictadas por
 \cref{thm:b1:fractions:real}, nunca improvisado.
\end{enumerate}
\end{remark}

\begin{proof}[Demostración de la fórmula de polo simple]
Cerca de un \omterm{def:b1:fractions:field}{polo} $a$ simple:$B = (X - a) Q$ con $Q(a) \neq 0$ y $B' =
Q + (X - a) Q'$, entonces $B'(a) = Q(a)$. El valor de encubrimiento es
$\frac{A(a)}{Q(a)} = \frac{A(a)}{B'(a)}$.
\end{proof}

\begin{example}\label{ex:b1:fractions:simple}
Descomponer $F = \dfrac{1}{X(X-1)(X-2)}$. Tres simples \omterm{def:b1:fractions:field}{polos}; encubrimiento
en cada uno:
\[
c_0 = \frac{1}{(0-1)(0-2)} = \frac12,
\quad
c_1 = \frac{1}{1 \times (1 - 2)} = -1,
\quad
c_2 = \frac{1}{2 \times 1} = \frac12,
\]
entonces $F = \dfrac{1/2}{X} - \dfrac{1}{X-1} + \dfrac{1/2}{X-2}$.
\emph{Cheque en $X = 3$:} directamente,$F(3) = \frac{1}{3\cdot2\cdot1} =
\frac16$; de la descomposición,$\frac{1/2}{3} - \frac{1}{2} +
\frac{1/2}{1} = \frac16 - \frac12 + \frac12 = \frac16$.
\end{example}

\begin{example}[Polos Múltiples]\label{ex:b1:fractions:multiple}
Descomponer $F = \dfrac{X}{(X-1)^2 (X+1)}$. Forma:
$\frac{a}{(X-1)^2} + \frac{b}{X-1} + \frac{c}{X+1}$.
\begin{itemize}
 \item Encubrimiento en el doble \omterm{def:b1:fractions:field}{polo}: $a = \bigl[\frac{X}{X+1}\bigr]_{X=1}
 = \frac12$.
 \item Encubrimiento en $-1$:$c = \bigl[\frac{X}{(X-1)^2}\bigr]_{X=-1} =
 -\frac14$.
 \item Límite de $xF(x)$ en $\infty$:$0 = b + c$, entonces $b = \frac14$.
\end{itemize}
\[
F = \frac{1/2}{(X-1)^2} + \frac{1/4}{X-1} - \frac{1/4}{X+1} .
\]
\emph{Verificar en $X = 0$:}$F(0) = 0$ y $\frac12 - \frac14 - \frac14
= 0$.
\end{example}

\section{Descomposición sobre $\R$}

\begin{theorem}[Descomposición sobre $\R$]\label{thm:b1:fractions:real}
Sea $F \in \R(X)$ en forma reducida con denominador
\[
B = c \prod_i (X - a_i)^{m_i} \prod_j (X^2 + p_j X + q_j)^{n_j}
\qquad (p_j^2 - 4q_j < 0).
\]
Entonces $F$ se descompone únicamente como su parte entera más términos
\[
\frac{c_{i,k}}{(X - a_i)^k} \quad (1 \leq k \leq m_i),
\qquad
\frac{\alpha_{j,l}\, X + \beta_{j,l}}{(X^2 + p_j X + q_j)^{l}}
\quad (1 \leq l \leq n_j),
\]
con coeficientes reales.
\end{theorem}

\begin{proof}
Descomponer sobre $\C$(\cref{thm:b1:fractions:complex}). Dado que $F$ es
real, el coeficiente sobre \omterm{def:b1:fractions:field}{polo} $\conj a $(en cada orden) es el
conjugado del coeficiente sobre $a $(aplicar conjugación al
descomposición e invocar unicidad). Agrupa cada par conjugado:
\[
\frac{c}{(X - z)^l} + \frac{\conj c}{(X - \conj z)^l}
= \frac{c\,(X - \conj z)^l + \conj c\,(X - z)^l}{\bigl(X^2 - 2\Re(z)X
+ \abs z^2\bigr)^{l}},
\]
cuyo numerador es su propio conjugado, por tanto real, de grado $\leq l$;
dividir múltiplos de la cuadrática real la reduce a grados
$\leq 1$ en cada nivel $l $(una pequeña inducción hacia abajo). Real \omterm{def:b1:fractions:field}{polos}
mantienen sus coeficientes reales (la conjugación los fija). Unicidad
se deriva de la unicidad sobre $\C$.
\end{proof}

\begin{example}[Viendo el emparejamiento conjugado]\label{ex:b1:fractions:pairing}
El mecanismo de la prueba, en el caso más pequeño: sobre $\C$, el
\omterm{def:b1:fractions:field}{polos} de $\frac1{X^2+1}$ son $\pm\iu $, con coeficientes de encubrimiento
$\frac1{2\iu}$ en $\iu$ y $\frac1{-2\iu}$ en $-\iu $ ---
conjugados entre sí, como predice el teorema:
\[
\frac{1}{X^2 + 1}
= \frac{1/(2\iu)}{X - \iu} - \frac{1/(2\iu)}{X + \iu} .
\]
Recombinando sobre el denominador común:
\[
\frac{1}{2\iu}\cdot\frac{(X + \iu) - (X - \iu)}{X^2 + 1}
= \frac{1}{2\iu}\cdot\frac{2\iu}{X^2 + 1} = \frac{1}{X^2+1} :
\]
las partes imaginarias se cancelan y el ladrillo real reaparece intacto.
Para integrandos reales normalmente nunca se sale de $\R$ --- pero cuando
evaluando \emph{sumas} en puntos complejos (como el problema del fin de semana
hace con \omterm{def:b1:complex:unity}{raíces de la unidad}), los ladrillos complejos son los naturales
moneda, y este par es el tipo de cambio entre las dos
descomposiciones.
\end{example}

\begin{example}\label{ex:b1:fractions:real}
Descomponer $F = \dfrac{4}{(X^2+1)(X-1)^2}$ sobre $\R$. Forma:
$\frac{aX + b}{X^2 + 1} + \frac{c}{(X-1)^2} + \frac{d}{X - 1}$.
Encubrimiento en el doble \omterm{def:b1:fractions:field}{polo}: $c = \bigl[\frac{4}{X^2+1}\bigr]_{X=1} =
2$. Encubrimiento en el complejo \omterm{def:b1:fractions:field}{polo}$\iu $(numerador sobre $X^2 + 1$
evaluado mediante la descomposición compleja, o directamente): multiplicar por
$X^2 + 1$ y configure $X = \iu $:
\[
a\iu + b = \frac{4}{(\iu - 1)^2} = \frac{4}{-2\iu} = 2\iu ,
\]
entonces $a = 2$,$b = 0$. Límite de $xF(x)$ al infinito:$0 = a + d$, entonces $d
= -2$. Por lo tanto
\[
F = \frac{2X}{X^2+1} + \frac{2}{(X-1)^2} - \frac{2}{X-1} .
\]
\emph{Verificar en $X = 0$:}$F(0) = 4$ y $0 + 2 + 2 = 4$.
\end{example}

\begin{example}[Un denominador cúbico, de principio a fin.]\label{ex:b1:fractions:cubic}
Descomponer $F = \dfrac{1}{X^3 + 1}$ sobre $\R$. Factorice primero:$X^3 +
1 = (X + 1)(X^2 - X + 1)$, la cuadrática tiene el discriminante$-3 <
0$. Forma:$\frac{a}{X+1} + \frac{bX + c}{X^2 - X + 1}$. encubrimiento
al simple \omterm{def:b1:fractions:field}{polo} $-1$:$a = \bigl[\frac1{X^2 - X +
1}\bigr]_{X=-1} = \frac13$. Límite de $xF(x)$ al infinito:$0 = a +
b $, entonces $b = -\frac13$. Evaluación en $X = 0$:$1 = a + c $, entonces $c =
\frac23$. Por lo tanto
\[
\frac{1}{X^3+1} = \frac13\Bigl(\frac{1}{X+1}
+ \frac{-X + 2}{X^2 - X + 1}\Bigr) ,
\]
confirmado en $X = 1$: lado izquierdo $\frac12$, lado derecho
$\frac13\bigl(\frac12 + 1\bigr) = \frac12$. Tenga en cuenta la economía:
tres incógnitas, tres hechos lineales baratos (un encubrimiento, un
límite, una evaluación), sin expansión de nada --- el flujo de trabajo
de \cref{met:b1:fractions:compute} en su forma pura.
\end{example}

\begin{remark}[Para qué sirve]
Una vez descompuesta, una función racional integra término por término: ladrillos
$\frac{1}{(x-a)^k}$ tiene primitivas elementales y ladrillos.
$\frac{\alpha x + \beta}{(x^2 + px + q)^l}$ reducir a$\ln$ y
$\arctan $(\cref{ch:b1:integration}). Las sumas telescópicas son la otra
Aplicación estándar (\cref{exo:b1:fractions:8}).
\end{remark}

primitiva de $x \mapsto \dfrac{x^2}{(x^2+1)^2}$ dada
$\int \frac{\dd x}{(x^2+1)^2} = \frac12\bigl(\arctan x +
\frac{x}{x^2+1}\bigr) + C $.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:6}
Para $n \in \N^*$, descomponga $F_n = \dfrac{n!}{X(X+1)\cdots(X+n)}$\emph{(simple \omterm{def:b1:fractions:field}{polos} at $0, -1, \dots, -n$; use the cover-up formula
and recognize \omterm{def:b1:counting:objects}{coeficientes binomiales})}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:7}
Usando la identidad de \cref{exo:b1:poly:11} para $P = X^n - 1$,
demostrar que
\[
\sum_{k=0}^{n-1} \frac{1}{X - \omega^k} = \frac{n X^{n-1}}{X^n - 1},
\qquad \omega = \eu^{2\iu\pi/n},
\]
y evalúe ambos lados en $X = 2$ para $n = 4$ como verificación.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:8}
Descomponga $\dfrac{1}{k(k+1)(k+2)}$ y calcule
\[
S_n = \sum_{k=1}^{n} \frac{1}{k(k+1)(k+2)},
\qquad\text{then}\qquad
\lim_{n \to \infty} S_n .
\]
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:fractions:9}
Sea $P \in \R[X]$\omterm{def:b1:poly:def}{mónico} de grado $n$ con raíces reales distintas $n$
$x_1 < \dots < x_n$. demostrar que
\[
\sum_{i=1}^{n} \frac{1}{P'(x_i)} = 0 \quad (n \geq 2),
\qquad
\sum_{i=1}^{n} \frac{x_i^{\,n-1}}{P'(x_i)} = 1 .
\]
\emph{Hint: decompose $\frac{X^m}{P}$ for $m \leq n - 1$ and look at
the coefficient decay at infinity --- or use \omterm{thm:b1:poly:lagrange}{interpolación de Lagrange}
(\cref{thm:b1:poly:lagrange}) of the monomial $X^m$ at the nodes
$x_i$.}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:10}
Descomponer $\dfrac{1}{X(X+1)^2}$ sobre $\R$. Admitiendo el valor
$\sum_{k \geq 1} \frac1{k^2} = \frac{\pi^2}6$(probado en este
problema del fin de semana del capítulo), deducir
\[
\sum_{n=1}^{\infty} \frac{1}{n(n+1)^2} = 2 - \frac{\pi^2}{6} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:11}
Descomponer sobre $\R$:$\dfrac{1}{(X^2+1)(X^2+4)}$, luego
$\dfrac{X^2}{(X^2+1)(X^2+4)}$. \emph{Hint: both denominators are
\omterm{def:b1:poly:def}{polinomios} in $X^2$: decompose $\frac1{(Y+1)(Y+4)}$ first.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:fractions:12}
(Ecuaciones seculares) Sea $F = \sum_{i=1}^{r} \frac{c_i}{X - p_i}$
con $p_1 < p_2 < \dots < p_r$ real y todo $c_i > 0$.
\begin{enumerate}
 \item Demuestre que $F$ es estrictamente decreciente en cada intervalo de
 su dominio, y dar sus límites en $\pm\infty$ y en ambos
 lados de cada \omterm{def:b1:fractions:field}{polo}.
 \item Deduce que por cada $\lambda > 0$, la ecuación $F(x) =
 \lambda $ tiene exactamente $r $ soluciones reales, una en cada
 intervalo $\intoo{p_i}{p_{i+1}}$ y uno posterior a$p_r$.
 (Tales ecuaciones gobiernan las perturbaciones de valores propios;
 El teorema del valor intermedio se utiliza en el nivel de escuela secundaria.
 aquí y demostrado en \cref{ch:b1:continuity}.)
\end{enumerate}
\end{exercise}

\section{Problema: fracciones parciales como motor.}

\begin{problem}\label{pb:b1:fractions:1}
La descomposición en fracciones parciales se parece a la contabilidad; este problema
muestra que es un motor. Alimentado con la fracción
$\frac1{X(X+1)\cdots(X+k)}$, telescopio familias enteras de sumas
en forma cerrada; alimentado con $\frac1{X^n - 1}$, produce
identidades trigonométricas como
\[
\sum_{k=1}^{n-1}\frac{1}{\sin^2\frac{k\pi}{n}} = \frac{n^2-1}{3} ;
\]
y, llevado un paso más allá, esa identidad exprime uno de los
Fórmulas más famosas de las matemáticas: la de Euler.
\[
\sum_{k=1}^{\infty}\frac1{k^2} = \frac{\pi^2}{6}
\]
--- aquí obtenido sin nada más allá del álgebra de este capítulo y
high-school trigonometry.\index{telescópico}\index{Problema de Basilea}
En todo momento, $\omega = \eu^{2\iu\pi/n}$; Los límites de las secuencias son.
utilizados a nivel de Secundaria (\cref{ch:b1:seq} los formaliza).

\medskip
\noindent\textbf{Parte I --- The telescope.}
\begin{enumerate}
 \item Descomponga $\frac1{X(X+1)}$ y calcule $\sum_{n=1}^{N}
 \frac1{n(n+1)}$ exactamente; concluir que la suma tiende a
 $1$.
 \item Lo mismo para $\frac1{X(X+2)}$: mostrar $\sum_{n=1}^{N}
 \frac1{n(n+2)} = \frac12\bigl(\frac32 - \frac1{N+1} -
 \frac1{N+2}\bigr) \to \frac34$. (Con un hueco, \emph{dos}
 Los términos de frontera sobreviven en cada extremo.)
 \item Formalizar el mecanismo: si $F(X) = G(X) - G(X+1)$ para
 algún $G$ racional sin \omterm{def:b1:fractions:field}{polos} en $\intco1{+\infty}$,
 luego $\sum_{n=1}^N F(n) = G(1) - G(N+1)$. recuperar el
 valor $\frac14$ de \cref{exo:b1:fractions:8} exhibiendo
 el testigo $G$ para $F = \frac1{X(X+1)(X+2)}$.
 \item Demuestre el telescopio factorial general: para $k \geq 1$,
 \[
 \frac{1}{X(X+1)\cdots(X+k)}
 = \frac1k\biggl(\frac{1}{X(X+1)\cdots(X+k-1)}
 - \frac{1}{(X+1)\cdots(X+k)}\biggr),
 \]
 y deducir
 \[
 \sum_{n=1}^{\infty}\frac{1}{n(n+1)\cdots(n+k)}
 = \frac{1}{k \cdot k!} .
 \]
 Verifique el caso $k = 2$ con la pregunta 3.
 \item Evaluar la descomposición de \cref{exo:b1:fractions:6}
 en puntos bien elegidos para demostrar
 \[
 \sum_{j=0}^{n}(-1)^j\binom nj\,\frac1{j+1}
 = \frac1{n+1},
 \qquad
 \sum_{j=0}^{n}(-1)^j\binom nj\,\frac1{j+2}
 = \frac1{(n+1)(n+2)} .
 \]
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The fraction $1/(X^n - 1)$.}
\begin{enumerate}[resume]
 \item Demuestre mediante la fórmula de encubrimiento que
 \[
 \frac{1}{X^n - 1}
 = \frac1n\sum_{k=0}^{n-1}\frac{\omega^k}{X - \omega^k} .
 \]
 \item Dos comprobaciones de cordura: verificar la fórmula directamente para $n =
 2$ y mostrar que la suma de los coeficientes $n $
 desaparece para $n \geq 2$ --- explique por qué debe (considere
 $xF(x)$ como $x \to \infty $).
 \item Rederivar por encubrimiento la identidad de
 \cref{exo:b1:fractions:7}:
 $\frac{nX^{n-1}}{X^n-1} = \sum_k \frac1{X - \omega^k}$.
 \item Grupo conjugado \omterm{def:b1:fractions:field}{polos} para probar la descomposición real:
 con $\theta_k = \frac{2k\pi}n$,
 \[
 \frac{\omega^k}{X - \omega^k} +
 \frac{\omega^{n-k}}{X - \omega^{n-k}}
 = \frac{2\cos\theta_k\,X - 2}
 {X^2 - 2\cos\theta_k\,X + 1} ,
 \]
 y escriba la descomposición real completa de
 $\frac1{X^n-1}$(distinguir $n$ impar y $n$ par).
 \item Especialícese en $n = 4$ y compare con
 \cref{exo:b1:fractions:4}.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Trigonometric sums, and Euler's
$\pi^2/6$.} Sea $P = 1 + X + \dots + X^{n-1}$, cuyas raíces son
$\omega, \omega^2, \dots, \omega^{n-1}$(todo sencillo).
\begin{enumerate}[resume]
 \item Usando $\frac{P'}P = \sum_{k=1}^{n-1}\frac1{X - \omega^k}$
 (\cref{exo:b1:poly:11}), probar
 \[
 \sum_{k=1}^{n-1}\frac{1}{1 - \omega^k} = \frac{n-1}2 .
 \]
 \item Demostrar $\dfrac1{1 - \eu^{\iu\theta}} = \dfrac12 +
 \dfrac\iu2\,\frac{\cos(\theta/2)}{\sin(\theta/2)}$ para
 $\theta \notin 2\pi\Z$(factorización de medio ángulo,
 \cref{met:b1:complex:trig}), y deducir de la pregunta 11
 que $\sum_{k=1}^{n-1}
 \frac{\cos(k\pi/n)}{\sin(k\pi/n)} = 0$ --- también visible
 por la simetría $k \leftrightarrow n - k$.
 \item Diferenciando la identidad de la pregunta 11 (es decir,\ usando
 $\bigl(\frac{P'}P\bigr)' = \frac{P''}P -
 \bigl(\frac{P'}P\bigr)^2$ evaluado en $X = 1$), probar
 \[
 \sum_{k=1}^{n-1}\frac{1}{(1 - \omega^k)^2}
 = \frac{(n-1)(5-n)}{12} .
 \]
 \item Escribiendo $\cot t = \frac{\cos t}{\sin t}$, deducir de
 preguntas 12--13 las dos formas cerradas
 \[
 \sum_{k=1}^{n-1}\cot^2\frac{k\pi}n = \frac{(n-1)(n-2)}3,
 \qquad
 \sum_{k=1}^{n-1}\frac1{\sin^2\frac{k\pi}n}
 = \frac{n^2-1}3 .
 \]
 \item Verifique ambas fórmulas a mano para $n = 3$ y $n = 4$.
 \item Demostrar las desigualdades $\cot t < \frac1t < \frac1{\sin t}$
 para $t \in \intoo0{\frac\pi2}$(de $\sin t < t < \tan
 t $), y deducir, para $n = 2m + 1$ y $1 \leq k \leq m $:
 \[
 \cot^2\frac{k\pi}n \;<\; \frac{n^2}{k^2\pi^2}
 \;<\; \frac1{\sin^2\frac{k\pi}n} .
 \]
 \item Sume estas desigualdades para $k = 1, \dots, m $(usando el
 simetría $k \leftrightarrow n - k$ para reducir a la mitad las fórmulas
 de la pregunta 14) y aprieta:
 \[
 \sum_{k=1}^{\infty}\frac1{k^2} = \frac{\pi^2}6 .
 \]
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Higher bricks.}
\begin{enumerate}[resume]
 \item Al elevar al cuadrado la descomposición de $\frac1{X^2-1}$ y
 re-descomponiendo el término cruzado, demuestre
 \[
 \frac1{(X^2-1)^2}
 = \frac14\Bigl(\frac1{(X-1)^2} + \frac1{(X+1)^2}\Bigr)
 - \frac14\Bigl(\frac1{X-1} - \frac1{X+1}\Bigr),
 \]
 y compruébalo en $X = 0$.
 \item Combine la pregunta 18, el telescopio y el valor de Euler
 (pregunta 17) para demostrar
 \[
 \sum_{n=2}^{\infty}\frac1{(n^2-1)^2}
 = \frac{\pi^2}{12} - \frac{11}{16} ,
 \]
 y confirmar el valor numéricamente con tres decimales.
 \item Diferenciar la identidad de la pregunta 8 para obtener una
 formulario cerrado para $\sum_{k=0}^{n-1}\frac1{(X -
 \omega^k)^2}$ y compruébelo en $X = 2$,$n = 2$.
 \item Demuestre que para $k \geq 2$,
 \[
 \sum_{n=k}^{\infty}\frac{1}{\binom nk} = \frac{k}{k-1} .
 \]
 \emph{(Reduce to question 4 by writing $1/\binom nk$ with
 factorials.)}
 \item Para $k = 3$, proporcione la suma parcial exacta
 $\sum_{n=3}^{N}\frac1{\binom n3}$ y su límite.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Synthesis.}
\begin{enumerate}[resume]
 \item Como cálculo final, escriba el valor real completo
 descomposición de $\dfrac1{X^6 - 1}$ y compruébelo en $X =
 0$.
 \item ¿Dónde exactamente se utilizó el problema: (i) la unicidad de
 la descomposición; (ii) el \omterm{def:b1:complex:unity}{raíces de la unidad} de
 \cref{ch:b1:complex}; (iii) la derivada logarítmica
 de \cref{exo:b1:poly:11}? Una frase cada uno.
 \item Síntesis, en un breve párrafo: una identidad algebraica
 --- romper una fracción en ladrillos --- generó exactamente
 sumas, identidades trigonométricas y $\pi^2/6$. Comentar en
 la división del trabajo entre álgebra (exacta
 descomposiciones, válidas en todas partes) y análisis (límites,
 apretando), y señale donde está cada hilo
 industrializados: telescopado y comparación en
 \cref{ch:b1:series}, integración de los ladrillos en
 \cref{ch:b1:integration}.
\end{enumerate}
\end{problem}
