\chapter{Secuencias}\label{ch:b1:seq}

Las secuencias fueron manipuladas en el volumen de High School con el límite
Concepto tomado a medias por fe. Aquí la teoría se reconstruye sobre la base de
integridad de $\R$(\cref{ch:b1:reals}): cada teorema clásico
--- convergencia monótona, \omterm{thm:b1:seq:adjacent}{secuencias adyacentes}, Bolzano--Weierstrass,
el criterio de Cauchy --- es una cara de ese único axioma. el capitulo
Finaliza con el estudio práctico de las secuencias definidas por $u_{n+1} =
f(u_n)$.

\section{Convergencia}

\begin{definition}[Límite de una secuencia]\label{def:b1:seq:limit}
Una secuencia $(u_n)$ de reales \emph{converge} a$\ell \in \R$
cuando\index{límite!de una secuencia}
\[
\forall \varepsilon > 0,\ \exists N \in \N,\ \forall n \geq N,
\qquad \abs{u_n - \ell} \leq \varepsilon .
\]
Se escribe $u_n \to \ell$ o $\lim u_n = \ell$. Una secuencia que hace
no converge (a ningún real) \emph{diverge}. Divergencia \emph{to
$+\infty$}: $\forall M,\ \exists N,\ \forall n \geq N,\ u_n \geq M$
(de manera similar $-\infty$).
\end{definition}

\begin{example}[Una prueba $\varepsilon$--$N$, escrita una vez]\label{ex:b1:seq:epsilonN}
Reclamación: $u_n = \dfrac{n^2 + 1}{2n^2 - 3} \to \dfrac12$. primero
aislar el error:
\[
\Bigl| u_n - \frac12 \Bigr|
= \Bigl| \frac{2(n^2 + 1) - (2n^2 - 3)}{2(2n^2 - 3)} \Bigr|
= \frac{5}{2\,\abs{2n^2 - 3}}
= \frac{5}{2\,(2n^2 - 3)} \quad (n \geq 2).
\]
Luego domínelo con algo simple: para $n \geq 2$,$2n^2 - 3
\geq n^2$, entonces el error es $\leq \frac{5}{2n^2} \leq \frac 5{2n}$.
Dado $\varepsilon > 0$, \omterm{thm:b1:reals:archimedes}{propiedad de Arquímedes} proporciona $N \geq
\max\bigl(2, \frac{5}{2\varepsilon}\bigr)$; para $n \geq N $ el
El error es $\leq \varepsilon$. Hecho. La idea final: una
$\varepsilon$--$N$ la prueba tiene exactamente tres movimientos --- calcula el
error, limítelo por una expresión elemental decreciente, resuelva para
el umbral --- y después de los teoremas de este capítulo (operaciones,
apretar) casi nunca se vuelve a escribir una demostración de este tipo: los teoremas
empaquete los tres movimientos de una vez por todas.
\end{example}

\begin{example}[Divergencia al infinito, certificado]\label{ex:b1:seq:toinfinity}
Reclamación: $u_n = n^2 - 100n \to +\infty$. Factoriza el término dominante:
$u_n = n^2\bigl(1 - \frac{100}{n}\bigr) \geq \frac{n^2}{2}$ para
$n \geq 200$. Dado $M$, tome $N = \max\bigl(200,
\lceil\sqrt{2M}\rceil\bigr)$: para $n \geq N $,$ u_n \geq
\frac{n^2}{2} \geq M $. Se muestran dos hábitos: el término dominante
El factoring convierte una competencia ($n^2$ contra$-100n$) en una
escala única multiplicada por un factor que tiende a $1$; y el umbral
puede ser enorme ($u_{100} = 0$, la secuencia es incluso negativa antes
$n = 100$) --- la divergencia con $+\infty $ es una \omterm{def:b1:logic:statement}{enunciado} sobre el
cola, indiferente a cualquier cantidad finita de mala conducta.
\end{example}

\begin{proposition}[Primeras propiedades]\label{prop:b1:seq:first}
\begin{enumerate}
 \item El límite, si existe, es único.
 \item Una secuencia convergente está acotada.
 \item Si $u_n \to \ell$, cada modificación de un número finito de términos
 deja la convergencia y el límite sin cambios.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) Si $u_n \to \ell$ y $u_n \to \ell'$ con $\ell \neq \ell'$,
tomar $\varepsilon = \frac{\abs{\ell - \ell'}}{3}$: más allá de los dos
umbrales, $\abs{\ell - \ell'} \leq \abs{\ell - u_n} + \abs{u_n -
\ell'} \leq 2\varepsilon = \frac23 \abs{\ell - \ell'}$, absurdo.

(2) Con $\varepsilon = 1$: más allá de $N$,$\abs{u_n} \leq \abs\ell +
1$; los un número finito de términos anteriores también están acotados, por lo que $\abs{u_n}
\leq \max(\abs{u_0}, \dots, \abs{u_{N-1}}, \abs\ell + 1)$.

(3) En detalle: supongamos $v_n = u_n$ para $n \geq n_0$ y $u_n \to
\ell $. Dado $\varepsilon > 0$, tome el umbral $N $ para
$(u_n)$: para $n \geq \max(N, n_0)$,$\abs{v_n - \ell} = \abs{u_n
- \ell} \leq \varepsilon$. Entonces $v_n \to \ell $: la definición
cuantifica sólo sobre $n \geq N$, y cualquier prefijo finito puede ser
sobrescrito a costa de ampliar el umbral. (Esta es la razón por la que
Las hipótesis "para todos los grandes $n$" son suficientes en todas partes en este
capítulo.)
\end{proof}

\begin{theorem}[Operaciones en límites]\label{thm:b1:seq:operations}
Si $u_n \to \ell$ y $v_n \to m$, entonces
\[
u_n + v_n \to \ell + m, \qquad u_n v_n \to \ell m, \qquad
\frac{u_n}{v_n} \to \frac{\ell}{m} \ (\text{if } m \neq 0),
\qquad \abs{u_n} \to \abs\ell .
\]
\end{theorem}

\begin{proof}
\emph{Suma:} $\abs{(u_n + v_n) - (\ell + m)} \leq \abs{u_n - \ell} +
\abs{v_n - m} \leq 2\varepsilon$ más allá del umbral mayor.
\emph{Producto:} escribir
\[
u_n v_n - \ell m = (u_n - \ell)\,v_n + \ell\,(v_n - m);
\]
$(v_n)$ está limitado por algún $B$(\cref{prop:b1:seq:first}), por lo que el
el lado derecho es $\leq B\abs{u_n - \ell} + \abs{\ell}\,\abs{v_n - m}$,
arbitrariamente pequeño. \emph{Cociente:} basta con tratar $\frac
1{v_n}$. Con $\varepsilon = \frac{\abs m}{2}$: más allá de algunos $N_0$,
$\abs{v_n} \geq \frac{\abs m}{2}$, entonces
\[
\Bigl| \frac{1}{v_n} - \frac 1m \Bigr|
= \frac{\abs{m - v_n}}{\abs{v_n m}}
\leq \frac{2}{m^2}\,\abs{v_n - m} \longrightarrow 0 .
\]
\emph{Valor absoluto:} $\bigl|\abs{u_n} - \abs\ell\bigr| \leq
\abs{u_n - \ell}$(desigualdad del triángulo inverso,
\cref{prop:b1:complex:rules}).
\end{proof}

\begin{example}[Operaciones más un truco algebraico]\label{ex:b1:seq:conjugatetrick}
Calcule $\lim\,\bigl(\sqrt{n^2 + n} - n\bigr)$. las dos piezas
tienden por separado a $+\infty$: el teorema de operaciones no dice nada
sobre su diferencia (un \emph{forma indeterminada}). multiplicar
por el conjugado:
\[
\sqrt{n^2 + n} - n
= \frac{(n^2 + n) - n^2}{\sqrt{n^2 + n} + n}
= \frac{n}{\sqrt{n^2+n} + n}
= \frac{1}{\sqrt{1 + \frac1n} + 1} .
\]
Ahora todo converge: $\sqrt{1 + \frac1n} \to 1$, porque
$0 \leq \sqrt{1 + h} - 1 = \frac{h}{\sqrt{1+h} + 1} \leq h$
(conjugar nuevamente, luego apretar con $h = \frac1n$); entonces el
El teorema de operaciones da el límite $\frac{1}{1 + 1} =
\frac12$. La idea final: el teorema de las operaciones no es un
calculadora para todos los límites --- las formas indeterminadas ($\infty -
\infty $,$\frac00$,$0 \times \infty $,$1^\infty $) primero deben ser
\emph{transformado} por álgebra (conjugados, factorizando el dominante
término) hasta que cada pieza converja; la máquina sistemática para
casos resistentes es la expansión asintótica de
\cref{ch:b1:taylor}.
\end{example}

\begin{theorem}[Límites y orden]\label{thm:b1:seq:order}
\begin{enumerate}
 \item Si $u_n \leq v_n$ es para todos los $n$ grandes y ambos convergen, entonces
 $\lim u_n \leq \lim v_n$. (Las desigualdades estrictas \emph{no}
 pasar al límite: $\frac 1n > 0$ pero $\lim = 0$.)
 \item (Teorema de compresión\index{teorema de compresión}) Si $u_n \leq w_n
 \leq v_n $ para todos los grandes $n $ y $u_n, v_n \to \ell$, entonces
 $w_n \to \ell$.
 \item Si $u_n \to \ell > 0$, entonces $u_n > \frac\ell2 > 0$ para todos
 grande $n$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Supongamos $\ell = \lim u_n > m = \lim v_n$; con $\varepsilon =
\frac{\ell - m}{3}$, los términos grandes satisfacen $v_n \leq m + \varepsilon <
\ell - \varepsilon \leq u_n $ y contradicen $u_n \leq v_n $.

(2) Más allá de los umbrales: $\ell - \varepsilon \leq u_n \leq w_n \leq
v_n \leq \ell + \varepsilon $.

(3) es \cref{def:b1:seq:limit} con $\varepsilon = \frac\ell2$.
\end{proof}

\begin{example}[dos apretones]\label{ex:b1:seq:squeezes}
(i) $\dfrac{\sin n}{n} \to 0$: desde $-\frac1n \leq \frac{\sin
n}{n} \leq \frac1n $, ambas paredes colapsan en $0$ --- no es necesario
entender el numerador errático en absoluto. (ii) $(2^n +
3^n)^{1/n} \to 3$: enmarcar el interior,
\[
3^n \leq 2^n + 3^n \leq 2\cdot3^n
\quad\Longrightarrow\quad
3 \leq (2^n + 3^n)^{1/n} \leq 3\cdot 2^{1/n} ,
\]
y $2^{1/n} = \eu^{\frac{\ln 2}{n}} \to 1$(como para $5^{1/n}$ en
\cref{exo:b1:seq:2}): el apretón entrega $3$. el cierre
Idea: una suma de exponenciales en competencia se comporta como su
Término \emph{más grande} --- los más pequeños son absorbidos por un
factor constante inofensivo, que luego la raíz $n$-ésima borra.
\end{example}

\section{Secuencias monótonas}

\begin{theorem}[Teorema del límite monótono]\label{thm:b1:seq:monotone}
Una secuencia creciente acotada arriba converge a $\sup\{u_n : n \in
\N\}$; una secuencia creciente no acotada arriba diverge a
$+\infty$. (Espejo \omterm{def:b1:logic:statement}{enunciados} para disminuir
secuencias.)\index{teorema del límite monótono}
\end{theorem}

\begin{proof}
Sea $s = \sup\{u_n\}$(\cref{thm:b1:reals:sup}). Dado $\varepsilon >
0$, la caracterización $\varepsilon$
(\cref{prop:b1:reals:epsilon}) produce $N$ con $u_N > s -
\varepsilon $; por monotonicidad,$s - \varepsilon < u_N \leq u_n \leq
s $ para todo $n \geq N $: convergencia a$s $. Si es ilimitado: por cada
$M$ algunos $u_N > M$, y la monotonicidad mantiene todos los términos posteriores por encima de $M$.
\end{proof}

\begin{example}[El teorema monótono como máquina de existencia]\label{ex:b1:seq:existencemachine}
Sea $u_n = \prod_{k=1}^{n} \bigl(1 + \frac{1}{2^k}\bigr)$. cada uno
El factor excede $1$, por lo que $(u_n)$ está aumentando. ¿Limitado arriba? tomar
logaritmos y uso $\ln(1 + x) \leq x$
(\cref{ex:b1:derivative:convexineq} lo anticipa; o el
crudo $1 + x \leq \eu^x$ del volumen de High School):
\[
\ln u_n = \sum_{k=1}^{n} \ln\Bigl(1 + \frac{1}{2^k}\Bigr)
\leq \sum_{k=1}^{n} \frac{1}{2^k} < 1 ,
\]
entonces $u_n < \eu$. Creciente y acotado: $(u_n)$ converge a algunos
$\ell \in \intoc{u_1}{\eu}$ --- un real perfectamente bien definido
número sin forma cerrada a la vista ($\ell = 2.384\dots$). el
Idea final: el teorema del límite monótono es el más barato
máquina de existencia en análisis; se llamó $\eu$.
(\cref{ex:b1:seq:e} a continuación), y en \cref{ch:b1:series} será
decidir la convergencia de cada serie positiva por simple
limitación.
\end{example}

\begin{theorem}[Secuencias adyacentes]\label{thm:b1:seq:adjacent}
Sea $(a_n)$ creciente,$(b_n)$ decreciente y $b_n - a_n \to
0$. Luego ambos convergen, a un límite \emph{común},$\ell$ y $ a_n
\leq \ell \leq b_n $ para todos los $n $.\index{secuencias adyacentes}.
\end{theorem}

\begin{proof}
Primero, $a_n \leq b_n$ para todos los $n$: la secuencia $(b_n - a_n)$ es
decreciente y tiende a $0$, por lo que es $\geq 0$(un término negativo
congelarlo por debajo de $0$). Entonces $(a_n)$ es creciente y está acotado arriba por
$b_0$: converge a unos $\ell $(\cref{thm:b1:seq:monotone});
igualmente $(b_n) \to \ell'$; y $\ell' - \ell = \lim (b_n - a_n) =
0$. Las desigualdades $a_n \leq \ell \leq b_n $ se derivan de
monotonicidad ($\ell = \sup a_k \geq a_n$, etc.).
\end{proof}

\begin{example}[El número $\eu$]\label{ex:b1:seq:e}
Configure $a_n = \sum_{k=0}^{n} \frac{1}{k!}$ y $b_n = a_n + \frac{1}{n
\cdot n!}$($n \geq 1$). Entonces $(a_n)$ aumenta; y
\[
b_{n+1} - b_n = \frac{1}{(n+1)!} + \frac{1}{(n+1)(n+1)!} -
\frac{1}{n\,n!}
= \frac{n(n+1) + n - (n+1)^2}{n(n+1)(n+1)!}
= \frac{-1}{n(n+1)(n+1)!} < 0 ,
\]
entonces $(b_n)$ disminuye y $b_n - a_n \to 0$: adyacente. Su común
el límite es (por definición aquí) el número $\eu \approx 2.71828$; el
Las desigualdades $a_n < \eu < b_n$ son lo suficientemente agudas como para demostrar $\eu \notin
\Q $(\cref{exo:b1:seq:9}).
\end{example}

\section{Subsecuencias y Bolzano--Weierstrass}

\begin{definition}[Subsecuencia]\label{def:b1:seq:subsequence}
Un \emph{subsecuencia}\index{subsecuencia} de $(u_n)$ es una secuencia
$(u_{\varphi(n)})$ donde $\varphi \colon \N \to \N$ es estrictamente
creciente (nota $\varphi(n) \geq n$, por inducción).
\end{definition}

\begin{proposition}\label{prop:b1:seq:subsequences}
Si $u_n \to \ell $($\ell \in \R$ o $\pm\infty $), cada \omterm{def:b1:seq:subsequence}{subsecuencia}
tiende a $\ell$. En consecuencia, una secuencia con dos \omterm{def:b1:seq:subsequence}{subsecuencias} de
diferentes límites divergen. Por el contrario, si $(u_{2n})$ y
$(u_{2n+1})$ convergen a \emph{mismo},$\ell$ y luego a$u_n \to
\ell $.
\end{proposition}

\begin{proof}
Más allá del umbral $N$ para $(u_n)$, todos los índices $\varphi(n) \geq n
\geq N $califican (la desigualdad $\varphi(n) \geq n $ es la
inducción observada en \cref{def:b1:seq:subsequence}: fuerzas $\varphi(0)
\geq 0$ y $\varphi(n+1) > \varphi(n) \geq n $
$\varphi(n+1) \geq n + 1$). Por el contrario: dado
$\varepsilon$, tome los dos umbrales $N_0$(pares) y $N_1$
(probabilidades); un índice arbitrario $n \geq \max(2N_0, 2N_1 + 1)$ es
ya sea par, $n = 2k$ con $k \geq N_0$, o impar,$n = 2k+1$ con
$k \geq N_1$--- en ambos casos $\abs{u_n - \ell} \leq
\varepsilon$: cada índice está cubierto por uno de los dos
\omterm{def:b1:seq:subsequence}{subsecuencias}, y ese es el punto.
\end{proof}

\begin{example}[Límites subsecuentes]\label{ex:b1:seq:sublimits}
Para $u_n = (-1)^n \frac{n}{n+1}$: el par \omterm{def:b1:seq:subsequence}{subsecuencia} tiende a
$1$, el impar a$-1$, por lo que la secuencia diverge --- pero lo hace
de forma organizada, agrupados en torno a los dos valores $\pm 1$.
Para $u_n = \cos\frac{2\pi n}{3}$: los tres \omterm{def:b1:seq:subsequence}{subsecuencias} de
los índices $3k$,$3k + 1$,$3k + 2$ son constantes, iguales a$1$,
$-\frac12$,$-\frac12$; el \omterm{def:b1:logic:sets}{conjunto} de límites subsiguientes es $\{1,
-\frac12\}$. La idea final: una secuencia \emph{encerrado}
converge exactamente cuando tiene un único límite subsiguiente
(\cref{exo:b1:seq:8}); divergencia de una secuencia acotada siempre
significa al menos dos grupos, y Bolzano--Weierstrass debajo
garantiza que hay al menos uno.
\end{example}

\begin{theorem}[Bolzano--Weierstrass]\label{thm:b1:seq:bw}
Toda secuencia acotada de reales tiene un convergente
\omterm{def:b1:seq:subsequence}{subsecuencia}.\index{Teorema de Bolzano-Weierstrass}
\end{theorem}

\begin{proof}
Sea $u_n \in \intcc{a}{b}$ para todos los $n$. Construir segmentos anidados por
dicotomía: establecer $\intcc{a_0}{b_0} = \intcc{a}{b}$; dado
$\intcc{a_k}{b_k}$ que contiene $u_n$ para una infinidad de $n$, uno de
sus dos mitades todavía contienen $u_n$ para una cantidad infinita de $n$ --- llame
es $\intcc{a_{k+1}}{b_{k+1}}$. Las secuencias $(a_k)$,$(b_k)$ son
adyacente ($b_k - a_k = \frac{b-a}{2^k} \to 0$), con límite común
$\ell $(\cref{thm:b1:seq:adjacent}).

Extracto: elegir $\varphi(0)$ con $u_{\varphi(0)} \in
\intcc{a_0}{b_0}$, luego, inductivamente,$\varphi(k+1) > \varphi(k)$
con $u_{\varphi(k+1)} \in \intcc{a_{k+1}}{b_{k+1}}$ --- posible
ya que ese segmento contiene infinitos términos. Entonces $a_k \leq
u_{\varphi(k)} \leq b_k $, y el teorema de compresión da
$u_{\varphi(k)} \to \ell$.
\end{proof}

\begin{remark}[Lo que Bolzano--Weierstrass hace y no dice]
Dice \emph{hace}: sólo de la delimitación, algunos \omterm{def:b1:seq:subsequence}{subsecuencia}
converge --- existencia sin fórmula, como prueba de la dicotomía
deja claro (nada nos dice que los índices \emph{cual} sobrevivan). eso
\emph{no} dice que el límite es único? $((-1)^n)$ tiene
\omterm{def:b1:seq:subsequence}{subsecuencias} convergiendo a $1$ y a$-1$, y el \omterm{def:b1:logic:sets}{conjunto} de
Los límites subsiguientes pueden incluso ser infinitos.
(\cref{ex:b1:seq:sublimits}, y todos los Cantor \omterm{def:b1:logic:sets}{conjunto} en
\cref{pb:b1:topology:1}). No sobrevive a la ilimitación:
$(n)$ no tiene ningún \omterm{def:b1:seq:subsequence}{subsecuencia} convergente --- aunque se puede
extraiga siempre un \omterm{def:b1:seq:subsequence}{subsecuencia} tendiendo a $+\infty$ o $-\infty$
de cualquier secuencia ilimitada (elija $\varphi(k)$ con
$u_{\varphi(k)} \geq k$, digamos). Usado correctamente, el teorema es un
\emph{bomba de existencia}: aparece en el centro del Cauchy
criterio siguiente, del teorema de Heine y del valor extremo
teorema --- siempre para producir un punto que no sea explícito
ofertas de construcción.
\end{remark}

\section{Secuencias de Cauchy y completitud.}

\begin{definition}\label{def:b1:seq:cauchy}
Una secuencia $(u_n)$ es una \emph{secuencia de cauchy}\index{secuencia de cauchy}
cuando sus términos se vuelven arbitrariamente cercanos \emph{el uno al otro}:
\[
\forall \varepsilon > 0,\ \exists N,\ \forall p, q \geq N,
\qquad \abs{u_p - u_q} \leq \varepsilon .
\]
\end{definition}

\begin{example}[Verificando la propiedad Cauchy a mano]\label{ex:b1:seq:cauchyhand}
Dejemos $u_n = \sum_{k=0}^{n} \frac{\cos k}{2^k}$ --- sin monotonicidad,
sin límite adivinable. Para $p > q$:
\[
\abs{u_p - u_q}
= \Bigl| \sum_{k=q+1}^{p} \frac{\cos k}{2^k} \Bigr|
\leq \sum_{k=q+1}^{p} \frac{1}{2^k}
< \frac{1}{2^{q}} ,
\]
por la desigualdad del triángulo, $\abs{\cos k} \leq 1$ y un finito
suma geométrica. Dado $\varepsilon > 0$, elija $N$ con $2^{-N}
\leq \varepsilon $: todos los espacios más allá de $N $ son $\leq \varepsilon $,
la secuencia es Cauchy, por lo tanto converge --- a un límite que nadie
puedo nombrar en forma cerrada, que es exactamente el punto. el cierre
Insight: el dominio geométrico de los incrementos es el estándar.
forma de ganar la propiedad de Cauchy, y \cref{ch:b1:series}
embotella el argumento como "la convergencia absoluta implica
convergencia''.
\end{example}

\begin{theorem}[Integración de $\R$]\label{thm:b1:seq:complete}
Una secuencia de reales converge si y sólo si es \omterm{def:b1:seq:cauchy}{cauchy
secuencia}.\index{completeness of R@completeness of $\R$}
\end{theorem}

\begin{proof}
($\Rightarrow$) Si$u_n \to \ell $: más allá del umbral para
$\frac\varepsilon2$,$\abs{u_p - u_q} \leq \abs{u_p - \ell} +
\abs{\ell - u_q} \leq \varepsilon$.

($\Leftarrow$) Sea $(u_n)$ Cauchy. \emph{esta limitado}: con
$\varepsilon = 1$, más allá de $N$ todos los términos se encuentran dentro de $1$ de $u_N$, y
la cabeza es finita. \emph{Extracto}: por
\cref{thm:b1:seq:bw}, algunos \omterm{def:b1:seq:subsequence}{subsecuencia} $u_{\varphi(n)} \to \ell$.
\emph{Concluir}: dado $\varepsilon > 0$, tome $N$(Cauchy, para
$\frac\varepsilon2$) y $n \geq N$ con $\abs{u_{\varphi(n)} - \ell}
\leq \frac\varepsilon2$ y $\varphi(n) \geq N $; luego para cada $p
\geq N $:
\[
\abs{u_p - \ell} \leq \abs{u_p - u_{\varphi(n)}} +
\abs{u_{\varphi(n)} - \ell} \leq \varepsilon . \qedhere
\]
\end{proof}

\begin{remark}
El valor del criterio: certifica la convergencia \emph{sin
nombrando el límite}. Falla en $\Q$(los truncamientos decimales de
$\sqrt 2$ forma un \omterm{def:b1:seq:cauchy}{secuencia de cauchy} de racionales sin racional
límite): la integridad es una propiedad de $\R$, equivalente al axioma \omterm{def:b1:reals:bounds}{superior
atado}. También es el caballo de batalla detrás de la convergencia de
serie (\cref{ch:b1:series}).
\end{remark}

\begin{example}[Una secuencia de Cauchy con un límite invisible]\label{ex:b1:seq:basel}
Sea $S_n = \sum_{k=1}^{n} \frac{1}{k^2}$. Para $p > q \geq 1$:
\[
S_p - S_q = \sum_{k=q+1}^{p} \frac{1}{k^2}
\leq \sum_{k=q+1}^{p} \frac{1}{k(k-1)}
= \sum_{k=q+1}^{p} \Bigl(\frac{1}{k-1} - \frac 1k\Bigr)
= \frac 1q - \frac 1p < \frac 1q ,
\]
por lo que más allá de $N > \frac1\varepsilon$ todos los espacios son $\leq \varepsilon$:
$(S_n)$ es Cauchy, por lo tanto converge. Note lo que acaba de suceder: nosotros
demostró que existe un número real específico sin tener ningún nombre
por ello. (Es $\frac{\pi^2}{6}$ --- una célebre identidad de
Euler, demostrado en el volumen del año 2; nada en este capítulo podría
díganos eso.) Esta división del trabajo --- existencia ahora,
identificación más tarde, si es que alguna vez, es el criterio de Cauchy
punto, y el motor de la teoría de series en
\cref{ch:b1:series}.
\end{example}

\section{Secuencias recurrentes}

\begin{method}[Estudiando $u_{n+1} = f(u_n)$]\label{met:b1:seq:recurrent}
Dado $f$ y un punto de partida $u_0$:\index{secuencia recurrente}
\begin{enumerate}
 \item \emph{Stable \omterm{prop:b1:reals:intervals}{intervalo}:} encuentra un \omterm{prop:b1:reals:intervals}{intervalo} $I$ con $f(I)
 \subseteq I $ que contiene $u_0$: luego todo $u_n \in I $(por
 inducción).
 \item \emph{Límites de candidatos:} si $u_n \to \ell \in I$ y $f$ son
 continuo en $\ell $(\cref{ch:b1:continuity}), luego $\ell$
 es un \emph{punto fijo}: $f(\ell) = \ell$. Resuelva $f(x) = x$.
 \item \emph{Monotonicidad:} si $f$ aumenta en $I$, entonces
 $(u_n)$ es monótono (aumenta si $u_1 \geq u_0$,
 disminuyendo en caso contrario); combinado con limitación,
 \cref{thm:b1:seq:monotone} concluye. Si $f$ está disminuyendo,
 estudie los dos \omterm{def:b1:seq:subsequence}{subsecuencias} $(u_{2n})$ y $(u_{2n+1})$,
 que son monótonos para $f \circ f$.
 \item \emph{Control de errores:} una desigualdad $\abs{f(x) - \ell} \leq
 k\abs{x - \ell}$ con $ k < 1$ da $\abs{u_n - \ell} \leq
 k^n \abs{u_0 - \ell} \to 0$ directamente.
\end{enumerate}
\end{method}

\begin{example}[Método de la garza]\label{ex:b1:seq:heron}
Seamos $u_0 = 2$ y $u_{n+1} = \dfrac12\Bigl(u_n +
\dfrac{2}{u_n}\Bigr)$: el antiguo algoritmo para
$\sqrt 2$.\index{método de garza}
\begin{itemize}
 \item \emph{Estabilidad:} para $x > 0$, la media aritmética-geométrica
 la desigualdad da $\frac12(x + \frac2x) \geq \sqrt{x \cdot
 \frac 2x} = \sqrt 2$; entonces $I = \intco{\sqrt 2}{+\infty}$ es
 estable y contiene $u_1$(de hecho,$u_1 = \frac32 \geq
 \sqrt 2$).
 \item \emph{Monotonicidad:} para $x \geq \sqrt 2$,$\;x - f(x) =
 \frac{x^2 - 2}{2x} \geq 0$: la secuencia disminuye de
 $u_1$ en adelante, y está limitado por abajo por $\sqrt 2$: converge.
 \item \emph{Límite:} los puntos fijos resuelven $x = \frac12(x +
 \frac2x)$, es decir $x^2 = 2$: en $I $,$\ell = \sqrt 2$.
 \item\emph{Velocidad:}$u_{n+1} - \sqrt 2 = \frac{(u_n -
 \sqrt2)^2}{2u_n} \leq \frac{(u_n - \sqrt 2)^2}{2\sqrt 2}$:
 el número de dígitos correctos aproximadamente \emph{dobles} en cada
 paso (convergencia cuadrática).
\end{itemize}
\end{example}

\begin{remark}[Errores comunes con límites]
Cuatro clásicos. (yo) \emph{Pequeños pasos no implican convergencia}:
$u_{n+1} - u_n \to 0$ es mucho más débil que la propiedad de Cauchy ---
las sumas armónicas $H_n$ tienen pasos $\frac{1}{n+1} \to 0$ todavía
divergir a $+\infty $(\cref{exo:b1:seq:5}); la condición de Cauchy
controla $\abs{u_p - u_q}$ para \emph{todo} pares grandes, no
los consecutivos. (ii) \emph{Las desigualdades estrictas desaparecen en el
límite}: de $u_n < v_n$ para todo $n$ se obtiene solo $\lim u_n
\leq \lim v_n $(\cref{thm:b1:seq:order});$\frac1n > 0$ y todavía
$\lim = 0$. (iii) \emph{Acotado no es convergente}:$((-1)^n)$
está limitado y diverge; limitación más \emph{monotonicidad}
converge, la cota por sí sola sólo garantiza una convergencia
\omterm{def:b1:seq:subsequence}{subsecuencia} (\cref{thm:b1:seq:bw}). (iv) \emph{El punto fijo
La ecuación es la segunda, no la primera.}: para $u_{n+1} = f(u_n)$,
resolviendo $f(\ell) = \ell$ identifica el límite \emph{sólo después}
La convergencia está demostrada. La recurrencia $u_{n+1} = 2u_n$ tiene la
punto fijo único $\ell = 0$, pero desde $u_0 = 1$ la secuencia
corre a $+\infty$: la ecuación $\ell = 2\ell$ nunca fue
tiene derecho a un límite. Orden de operaciones, siempre: existencia
primero (\cref{met:b1:seq:recurrent}, pasos 1--3), identificación
segundo.
\end{remark}

\begin{example}[Un $f$ decreciente: la recurrencia dorada]\label{ex:b1:seq:golden}
Sean $u_0 = 1$ y $u_{n+1} = \dfrac{1}{1 + u_n}$. Aquí $f(x) =
\frac{1}{1+x}$ es \emph{decreciente}, por lo que la secuencia no es
monótono (alterna alrededor de su límite); el paso de contracción
de \cref{met:b1:seq:recurrent} es la herramienta adecuada. Estabilidad: si
$x \in \intcc{\frac12}{1}$ luego $1 + x \in
\intcc{\frac32}{2}$, por lo que $f(x) \in \intcc{\frac12}{\frac23}
\subseteq \intcc{\frac12}{1}$ y $ u_1 = \frac12$ ponen el
toda la secuencia allí. Punto fijo: $\ell = \frac{1}{1+\ell}$
con $\ell > 0$ da $\ell^2 + \ell - 1 = 0$, es decir
\[
\ell = \frac{\sqrt5 - 1}{2} = 0.6180\dots
\]
(la proporción áurea es inversa). Contracción: para $x, y \in
\intcc{\frac12}{1}$,
\[
\abs{f(x) - f(y)} = \frac{\abs{x - y}}{(1+x)(1+y)}
\leq \frac{\abs{x-y}}{(3/2)^2} = \frac49\,\abs{x - y} ,
\]
entonces $\abs{u_n - \ell} \leq \bigl(\frac49\bigr)^{n-1}\abs{u_1 -
\ell} \to 0$: convergencia, con velocidad geométrica, no
Se necesita monotonía. La idea final: métodos monótonos y
Los métodos de contracción dividen el mundo recurrente entre ellos ---
aumentar $f$ da órbitas monótonas, disminuir $f$ da
órbitas alternas domesticadas por una constante de Lipschitz $< 1$(la
teoría sistemática es \cref{exo:b1:derivative:11}).
\end{example}

\begin{remark}[Perspectivas dentro de este volumen]
Las secuencias son el instrumento de medida del resto del volumen.
soporta cada objeto. En \cref{ch:b1:topology} ellos
\emph{caracterizar} cierre y compacidad; en
\cref{ch:b1:continuity} transportan límites de funciones; en
\cref{ch:b1:integration} Las sumas de Riemann son secuencias convergentes
a la integral; \cref{ch:b1:series} \emph{is} la teoría de uno
clase especial de secuencias, las sumas parciales. Incluso el álgebra
Los capítulos los consumen: las iteraciones de una matriz en
\cref{ch:b1:matrices} forman secuencias cuyo comportamiento
(convergencia de $A^n$) es una pregunta de álgebra lineal con esta
vocabulario del capítulo. Los dos teoremas para llevar a todas partes:
límite monótono (existencia a partir del orden) y Bolzano--Weierstrass
(existencia desde la limitación) --- entre ellos, casi todos
Nace el límite en este libro.
\end{remark}

\begin{remark}[Secuencias complejas]
Una secuencia $(z_n)$ de números complejos converge a$\ell$ cuando
$\abs{z_n - \ell} \to 0$; equivalentemente, cuando $\Re(z_n) \to \Re(\ell)$
y $\Im(z_n) \to \Im(\ell)$(compárese $\abs{z}$ con $\abs{\Re z} +
\abs{\Im z}$). Los teoremas que no involucran orden --- operaciones,
Bolzano--Weierstrass (extracto dos veces), criterio de Cauchy --- prórroga
palabra por palabra.
\end{remark}

\section{Problema: Cesàro, Stolz y la lenta caída del seno}

\begin{problem}[{Problema del fin de semana --- el teorema de Cesàro--Stolz
y las asintóticas $u_n \sim \sqrt{3/n}$ para $u_{n+1} = \sin
u_n $}]
\label{pb:b1:seq:1}
El teorema de Cesàro--Stolz es la regla discreta de l'Hospital: a
encontrar el límite de un cociente $a_n/b_n$, basta con encontrar el
límite del cociente de \emph{incrementos} $(a_{n+1} -
a_n)/(b_{n+1} - b_n)$. Este problema prueba el teorema, cosechas
lo clásico lo limita y luego lo apunta a un objetivo famoso:
la secuencia $u_{n+1} = \sin u_n$, que avanza lentamente hasta $0$ en el
velocidad exactamente computable $u_n \sim \sqrt{3/n}$.\index{Cesaro--Stolz
theorem@Cesàro--Stolz theorem} Dos datos de la escuela secundaria
El volumen se concede aquí y se vuelve a demostrar honestamente más adelante en este
volumen: la desigualdad de la recta tangente
\[
\tag{G1} \eu^{u} \geq 1 + u \quad (u \in \R),
\]
re-probado por convexidad en \cref{ch:b1:derivative}, y el seno
horquillado
\[
\tag{G2} x - \frac{x^3}{6} \;\leq\; \sin x \;\leq\; x -
\frac{x^3}{6} + \frac{x^5}{120} \quad (0 \leq x \leq 1), \qquad
\abs{\sin x} \leq \abs{x} \quad (x \in \R),
\]
demostrado nuevamente por la fórmula de Taylor en \cref{ch:b1:taylor}.

\medskip
\noindent\textbf{Parte I --- Sums without closed formulas.}
\begin{enumerate}
 \item Usando $1 + 2 + \dots + n = \frac{n(n+1)}{2}$ y $1^2 +
 \dots + n^2 = \frac{n(n+1)(2n+1)}{6}$, calcule
 $\lim \frac{1 + 2 + \dots + n}{n^2}$ y $\lim \frac{1^2 +
 \dots + n^2}{n^3}$.
 \item Vamos $T_n = \sum_{k=1}^n \sqrt k$, para el cual no hay cerrado
 La fórmula existe. Demostrar el bracketing
 \[
 \frac{1}{2\sqrt 2}\,n^{3/2} \;\leq\; T_n \;\leq\;
 n^{3/2}
 \]
 \emph{(keep only the terms $k > \frac n2$ for the lower
 bound)}. Entonces $T_n$ tiene el orden $n^{3/2}$ --- pero cual
 constante? Mantenga la pregunta hasta la pregunta 8.
 \item (Lema telescópico) Sea $(b_n)$ estrictamente creciente
 y supongamos que para todos los $k \geq N$,
 \[
 m \;\leq\; \frac{a_{k+1} - a_k}{b_{k+1} - b_k} \;\leq\;
 M .
 \]
 Demuestre que $m \leq \dfrac{a_n - a_N}{b_n - b_N} \leq M$
 por cada $n > N$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The Cesàro--Stolz theorem.} Deja
$(b_n)$ será estrictamente creciente con $b_n \to +\infty $, y
supongamos $\dfrac{a_{n+1} - a_n}{b_{n+1} - b_n} \to \ell \in \R$.
\begin{enumerate}[resume]
 \item Reparar $\varepsilon > 0$. Mostrar que hay $N$ tal que
 $\ell - \varepsilon \leq \dfrac{a_n - a_N}{b_n - b_N}
 \leq \ell + \varepsilon $ para todos los $n > N $.
 \item Establecer, para $n > N$, la identidad
 \[
 \frac{a_n}{b_n} - \ell
 = \frac{a_N - \ell\,b_N}{b_n}
 + \Bigl(1 - \frac{b_N}{b_n}\Bigr)
 \Bigl(\frac{a_n - a_N}{b_n - b_N} - \ell\Bigr),
 \]
 y concluir el teorema: $\dfrac{a_n}{b_n} \to \ell$.
 \item Demuestre la variante $+\infty$: si $\dfrac{a_{n+1} -
 a_n}{b_{n+1} - b_n} \to +\infty $(mismas hipótesis sobre
 $(b_n)$), luego $\dfrac{a_n}{b_n} \to +\infty$.
 \item Tome $b_n = n$: recupere el teorema de la media de Cesàro
 \cref{exo:b1:seq:10}. Luego muestra lo contrario de
 Cesàro--Stolz falla: para $a_n = (-1)^n$,$b_n = n$, el
 el cociente $a_n/b_n$ converge mientras que el cociente de
 los incrementos no. Stolz es una calle de sentido único.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- First dividends.}
\begin{enumerate}[resume]
 \item Demuestre $(1+h)^{3/2} - 1 = \dfrac{3h + 3h^2 +
 h^3}{(1+h)^{3/2} + 1}$ por conjugación, deduzca
 $n\bigl((1 + \tfrac1n)^{3/2} - 1\bigr) \to \tfrac32$,
 y concluir con Cesàro--Stolz:
 \[
 T_n = \sum_{k=1}^{n} \sqrt k \;\sim\; \tfrac23\,
 n^{3/2} ,
 \]
 resolviendo el suspenso de la pregunta 2.
 \item Solo a partir de (G1), derivar el paréntesis del logaritmo
 \[
 \frac{t}{1 + t} \;\leq\; \ln(1 + t) \;\leq\; t
 \qquad (t > -1)
 \]
 \emph{(apply (G1) at $u = \ln(1+t)$ and at $u =
 -t/(1+t)$)}.
 \item Demuestre que $b_n = \ln n$ es estrictamente creciente con
 $\ln n \to +\infty$, y probar con Cesàro--Stolz y
 pregunta 9 que
 \[
 H_n = \sum_{k=1}^{n} \frac 1k \;\sim\; \ln n .
 \]
 (La estructura más fina $H_n = \ln n + \gamma + o(1)$ es la
 problema de fin de semana de \cref{ch:b1:series}.)
 \item (De proporciones a raíces) Sea $u_n > 0$ con
 $\frac{u_{n+1}}{u_n} \to L > 0$. Usando la pregunta 9, muestre
 $\ln\frac{u_{n+1}}{u_n} \to \ln L$; aplicar Cesàro a
 concluir $\frac{\ln u_n}{n} \to \ln L$, luego, con (G1),
 que $u_n^{1/n} \to L$. Aplicación: calcular
 $\lim\,\binom{2n}{n}^{1/n}$.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The slow fall of the sine.} Deja
$u_0 \in \R$ y $u_{n+1} = \sin u_n$.
\begin{enumerate}[resume]
 \item Desde (G2), muestre $0 < \sin x < x$ para $0 < x \leq 1$.
 Deducir: $u_1 \in \intcc{-1}{1}$; si $u_1 = 0$ el
 la secuencia es cero desde el rango $1$; y si $u_1 > 0$(el
 caso $u_1 < 0$ es simétrico,$\sin $ impar), entonces
 $(u_n)_{n \geq 1}$ es estrictamente decreciente, positivo y
 converge a $0$\emph{(identify the limit via $\ell =
 \sin \ell $, using $\abs{\sin a - \sin b} \leq \abs{a -
 b}$, itself a consequence of (G2) and the product-to-sum
 formula)}.
 \item Asumir de ahora en adelante $u_1 \in \intoc{0}{1}$. Mostrar por
 apretando, usando (G2):
 \[
 \frac{\sin u_n}{u_n} \to 1
 \qquad\text{and}\qquad
 \frac{u_n - \sin u_n}{u_n^{3}} \to \frac16 .
 \]
 \item Demostrar la factorización
 \[
 w_n := \frac{1}{u_{n+1}^{2}} - \frac{1}{u_n^{2}}
 = \frac{u_n - \sin u_n}{u_n^{3}} \cdot
 \frac{u_n + \sin u_n}{u_n} \cdot
 \Bigl(\frac{u_n}{\sin u_n}\Bigr)^{2},
 \]
 y deducir $w_n \to \frac13$.
 \item Concluir con \cref{exo:b1:seq:10} (versión incremental)
 que $\frac{1}{n\,u_n^{2}} \to \frac13$, entonces, por un
 argumento de conjugación para la raíz cuadrada, el titular:
 \[
 \sqrt n\;u_n \longrightarrow \sqrt 3 ,
 \qquad\text{i.e.}\qquad
 u_n \sim \sqrt{\frac 3n} .
 \]
 \item Cuantifique la lentitud: muestre que eventualmente $\sqrt{2/n}
 \leq u_n \leq 2/\sqrt n$, de modo que llegar a$u_n \leq
 10^{-2}$ requiere más de $20\,000$ iteraciones (aproximadamente
 $30\,000$, por las asintóticas). Contraste con \omterm{ex:b1:seq:heron}{garza
 método} (\cref{ex:b1:seq:heron}) y explique la
 Razón estructural: en el punto fijo $0$, la pendiente de
 $\sin $ es $1$(un punto fijo \emph{neutral}), mientras que
 las iteraciones de reducción de errores necesitan una pendiente de \omterm{def:b1:complex:field}{módulo} $< 1$.
 \item Muestre que para \emph{cada} punto de partida $u_0 \in \R$,
 ya sea $u_n = 0$ desde el rango $1$ en adelante, o $\abs{u_n} \sim
 \sqrt{3/n}$ --- la caída es universal, solo el signo
 recuerda $u_0$.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The general principle.} El seno es
una instancia de una máquina.
\begin{enumerate}[resume]
 \item Sea $u_n > 0$,$u_n \to 0$ y $\dfrac{u_n -
 u_{n+1}}{u_n^{2}} \to a > 0$. Demuestre sucesivamente:
 $\frac{u_{n+1}}{u_n} \to 1$; luego $\frac{1}{u_{n+1}} -
 \frac{1}{u_n} \to a$; luego $n\,u_n \to \frac1a $.
 \item (modelo exacto) Para $u_{n+1} = \dfrac{u_n}{1 + u_n}$,$u_0
 > 0$: muestre $\frac{1}{u_n}$ es aritmético, resuelva exactamente,
 y coteje la conclusión de la pregunta 18 con la exacta
 fórmula.
 \item Para $u_{n+1} = u_n \eu^{-u_n}$,$u_0 > 0$: muestre $u_n \to
 0$, use (G1) para apretar $\frac{1 - \eu^{-t}}{t}$ entre
 $\frac{1}{1+t}$ y $1$ para $t > 0$, y concluye $u_n
 \sim \frac 1n $.
 \item (Contacto cúbico, telescopio cuadrado) Sea $u_n > 0$,$u_n
 \to 0$,$\dfrac{u_n - u_{n+1}}{u_n^{3}} \to a > 0$.
 Adapte la factorización de la pregunta 14 para mostrar
 $\frac{1}{u_{n+1}^2} - \frac{1}{u_n^2} \to 2a$, y
 concluir $n\,u_n^{2} \to \frac{1}{2a}$. Compruebe que $a =
 \frac16$ recupere la Parte IV.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Limits of the method, and morals.}
\begin{enumerate}[resume]
 \item Demuestre que la hipótesis $b_n \to +\infty$ no puede ser
 eliminado: para $a_n = 2 - 2^{-n}$ y $b_n = 1 - 2^{-n}$,
 el cociente incremental tiende a $1$ mientras que $\frac{a_n}
 {b_n} \to 2$. Señale la línea exacta de las preguntas 5.
 prueba que se rompe.
 \item (Stolz dos veces) Prueba $\sum_{k=1}^{n} H_k \sim n \ln n$\emph{(one application of Cesàro--Stolz, then question
 10; bound $(n+1)\ln(n+1) - n\ln n$ using question 9)}.
 \item (Medios geométricos) Si $u_n > 0$ y $u_n \to \ell > 0$,
 mostrar $(u_1 u_2 \cdots u_n)^{1/n} \to \ell$; si $u_n \to
 +\infty $, muestre $(u_1 \cdots u_n)^{1/n} \to +\infty $.
 Deducir $(n!)^{1/n} \to +\infty$.
 \item Síntesis, una oración cada una: (i) ¿dónde exactamente
 la integridad entra en este problema; (ii) en qué sentido
 Cesàro--Stolz una regla discreta del Hospital (su
 El gemelo diferencial se basa en el teorema del valor medio de
 \cref{ch:b1:derivative}); (iii) enunciar la heurística
 vinculando el orden de contacto de $f$ en un punto neutro fijo
 señale el exponente de desintegración de $u_{n+1} = f(u_n)$;
 (iv) rastrear la constante $3$ de $\sqrt{3/n}$ hasta
 la tubería $\frac16 \to \frac13 \to 3$.
\end{enumerate}
\end{problem}
