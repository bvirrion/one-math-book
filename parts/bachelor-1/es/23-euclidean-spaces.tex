\chapter{Espacios euclidianos}\label{ch:b1:euclid}

Agregar un \omterm{def:b1:euclid:def}{producto interno} a un \omterm{def:b1:vspaces:def}{espacio vectorial} real compra el valor geométrico
nociones --- longitudes, ángulos, ortogonalidad, distancias --- y una
teorema que domina el capítulo: cada \omterm{def:b1:vspaces:subspace}{subespacio} admite una
\omterm{thm:b1:euclid:projection}{proyección ortogonal}, computable por Gram--Schmidt, al darse cuenta de la
distancia más corta. Las isometrías planas cierran el capítulo y el
geometría del año.

En todo momento, $E$ es un \emph{real} \omterm{def:b1:vspaces:def}{espacio vectorial}.

\section{productos internos}

\begin{definition}\label{def:b1:euclid:def}
Un \emph{producto interno}\index{producto interno} en $E$ es una \omterm{def:b1:logic:map}{aplicación}
$\langle\cdot,\cdot\rangle \colon E \times E \to \R$ eso es
bilineal, simétrico y definido positivo ($\langle x, x\rangle >
0$ para $ x \neq 0$). Un espacio \omterm{def:b1:findim:def}{de dimensión finita} así equipado es un
\emph{espacio euclidiano}\index{espacio euclidiano}. El \emph{norma} de
$x$ es $\norm{x} = \sqrt{\langle x, x\rangle}$ y $d(x, y) =
\norm{x - y}$.
\end{definition}

\begin{example}\label{ex:b1:euclid:examples}
En $\R^n$: el producto canónico $\langle x, y\rangle = \sum x_i
y_i $. En$C(\intcc{a}{b})$:$\langle f, g \rangle = \int_a^b fg $
(la precisión positiva es \cref{thm:b1:integration:props}~(4)). encendido
$\R_n[X]$:$\langle P, Q\rangle = \int_0^1 PQ$ o $\sum_{i}
P(x_i)Q(x_i)$ sobre $ n+1$ puntos distintos.
\end{example}

\begin{example}[El ángulo entre dos polinomios]
\label{ex:b1:euclid:polyangle}
Una vez que se elige un \omterm{def:b1:euclid:def}{producto interno}, \emph{cualquier} dos vectores distintos de cero
tener un ángulo, a través de $\cos\theta = \frac{\langle x,
y\rangle}{\norm x\,\norm y}$(un coseno legítimo por

\[
\langle X, X^2\rangle = \frac14, \qquad
\norm X = \frac1{\sqrt3}, \qquad \norm{X^2} = \frac1{\sqrt5},
\qquad
\cos\theta = \frac{1/4}{1/\sqrt{15}} = \frac{\sqrt{15}}{4}
\approx 0.968 :
\]
un ángulo de aproximadamente $14.5$ grados --- en $\intcc{0}{1}$, el
Los gráficos de $x$ y $x^2$ son "casi paralelos" en el
sentido de media cuadrática, por lo que eliminar ese valor compartido
dirección (Gram--Schmidt, abajo) deja sólo la pequeña

\end{example}

\begin{theorem}[Cauchy—Negro; propiedades de norma]\label{thm:b1:euclid:cs}
Para todos $x, y \in E$:\index{Desigualdad de Cauchy-Schwarz}
\[
\abs{\langle x, y\rangle} \leq \norm x\, \norm y ,
\]
con igualdad si $x, y $ son proporcionales. En consecuencia $\norm\cdot $
satisface la desigualdad del triángulo $\norm{x + y} \leq \norm x + \norm
y $(y $\norm{\lambda x} = \abs\lambda \norm x$,$\norm x = 0 \iff
x = 0$). Además:
\[
\norm{x+y}^2 = \norm x^2 + 2\langle x, y\rangle + \norm y^2,
\qquad
\langle x, y \rangle = \tfrac14\bigl(\norm{x+y}^2 -
\norm{x-y}^2\bigr) .
\]
\end{theorem}

\begin{proof}
Si $ y = 0$, todo es trivial. De lo contrario, la cuadrática $t
\mapsto \norm{x + ty}^2 = \norm x^2 + 2t\langle x, y\rangle + t^2
\norm y^2$ es $\geq 0$ para todo $t $: su discriminante es $\leq 0$,
que es Cauchy--Schwarz; igualdad significa una raíz doble $t_0$, es decir $x + t_0 y = 0$(definición): proporcionalidad. Triangulo
desigualdad: expandir,
\[
\norm{x + y}^2 = \norm x^2 + 2\langle x, y\rangle + \norm y^2
\leq \norm x^2 + 2\norm x\,\norm y + \norm y^2
= \bigl(\norm x + \norm y\bigr)^2 ,
\]
el paso intermedio es Cauchy-Schwarz; la igualdad fuerza $\langle
x, y\rangle = \norm x\norm y $, la igualdad \emph{positivo}
caso, es decir proporcionalidad con relación no negativa ---
geométricamente, el triángulo degenera sólo cuando los dos
los vectores apuntan en la misma dirección. los dos ultimos
Las identidades son expansiones directas (la segunda, la
\emph{identidad de polarización}, recupera el producto de la norma).
\end{proof}

\section{Ortogonalidad}

\begin{definition}\label{def:b1:euclid:orthogonal}
$x \perp y $ cuando $\langle x, y \rangle = 0$. una familia es
\emph{ortogonal} cuando sus vectores son ortogonales por pares,
\emph{ortonormal}\index{familia ortonormal} cuando además cada uno tiene
norma $1$. El \emph{complemento ortogonal} de un \omterm{def:b1:vspaces:subspace}{subespacio}$F$ es
\[
F^{\perp} = \{x \in E : \forall y \in F,\ \langle x, y\rangle =
0\},
\]
un \omterm{def:b1:vspaces:subspace}{subespacio} de $E$.\index{complemento ortogonal}
\end{definition}

\begin{proposition}\label{prop:b1:euclid:orthfree}
(Pitágoras\index{teorema de pitágoras}) Si $x \perp y $ entonces
$\norm{x+y}^2 = \norm x^2 + \norm y^2$. Una familia \omterm{def:b1:euclid:orthogonal}{ortogonal} de
vectores distintos de cero es \omterm{def:b1:vspaces:free}{libre}. En un \omterm{def:b1:euclid:orthogonal}{ortonormal} \omterm{def:b1:vspaces:free}{base} $(e_1, \dots,
e_n)$, \omterm{prop:b1:vspaces:coordinates}{coordenadas} y los productos son
\[
x = \sum_{i} \langle x, e_i\rangle\, e_i,
\qquad
\langle x, y \rangle = \sum_i \langle x, e_i\rangle \langle y,
e_i\rangle,
\qquad
\norm x^2 = \sum_i \langle x, e_i\rangle^2 .
\]
\end{proposition}

\begin{proof}
Pitágoras: ampliar. Libertad: tomar $\langle\,\cdot\,, x_j\rangle$
de una combinación nula: $\lambda_j \norm{x_j}^2 = 0$. \omterm{prop:b1:vspaces:coordinates}{Coordenadas}:
escribe $x = \sum \lambda_i e_i$ y toma el producto con $e_j$:
$\lambda_j = \langle x, e_j\rangle$; las dos fórmulas siguen por
bilinealidad.
\end{proof}

\begin{example}[Coordenadas ortonormales, con comprobación de Parseval]
\label{ex:b1:euclid:parsevalcheck}
Expanda $x = (1, 2, 3)$ en el \omterm{def:b1:euclid:orthogonal}{ortonormal} \omterm{def:b1:vspaces:free}{base} de
\cref{exo:b1:euclid:3},
\[
e_1 = \tfrac{1}{\sqrt2}(1,1,0), \quad
e_2 = \tfrac{1}{\sqrt6}(1,-1,2), \quad
e_3 = \tfrac{1}{\sqrt3}(-1,1,1).
\]
No hay sistema para resolver --- tres \omterm{def:b1:euclid:def}{productos internos}:
\[
\langle x, e_1\rangle = \frac{3}{\sqrt2},
\qquad
\langle x, e_2\rangle = \frac{1 - 2 + 6}{\sqrt6} =
\frac{5}{\sqrt6},
\qquad
\langle x, e_3\rangle = \frac{-1 + 2 + 3}{\sqrt3} =
\frac{4}{\sqrt3}.
\]
Certificación por la fórmula normativa de la propuesta:
\[
\frac{9}{2} + \frac{25}{6} + \frac{16}{3}
= \frac{27 + 25 + 32}{6} = 14 = \norm{x}^2 = 1 + 4 + 9 .
\]
Esta verificación de suma de coordenadas cuadradas (una prueba finita de Parseval
identidad) cuesta segundos y capta la señal y la normalización
errores con casi certeza: conviértalo en un hábito cada vez que un
Se calcula la expansión \omterm{def:b1:euclid:orthogonal}{ortonormal}; es de dimensión infinita
versión, para los coeficientes de Fourier de
\cref{ex:b1:euclid:trigortho}, es un teorema del Año~3
volumen.
\end{example}

\begin{theorem}[Abuela--Schmidt]\label{thm:b1:euclid:gramschmidt}
Cada \omterm{def:b1:euclid:def}{espacio euclidiano} tiene bases \omterm{def:b1:euclid:orthogonal}{ortonormal}. Explícitamente, de cualquier
\omterm{def:b1:vspaces:free}{base} $(v_1, \dots, v_n)$, la receta\index{Proceso de Gram-Schmidt}
\[
w_k = v_k - \sum_{i=1}^{k-1} \langle v_k, e_i\rangle\, e_i ,
\qquad
e_k = \frac{w_k}{\norm{w_k}}
\]
produce un \omterm{def:b1:euclid:orthogonal}{ortonormal} \omterm{def:b1:vspaces:free}{base} $(e_1, \dots, e_n)$ con
$\operatorname{Vect}(e_1, \dots, e_k) = \operatorname{Vect}(v_1,
\dots, v_k)$ por cada $k $.
\end{theorem}

\begin{proof}
Inducción en $k$. Suponiendo $(e_1, \dots, e_{k-1})$\omterm{def:b1:euclid:orthogonal}{ortonormal}
que abarca $\operatorname{Vect}(v_1, \dots, v_{k-1})$: el vector
$w_k$ es \omterm{def:b1:euclid:orthogonal}{ortogonal} para cada $e_j $($j < k$) por construcción
($\langle w_k, e_j \rangle = \langle v_k, e_j\rangle - \langle v_k,
e_j\rangle $) y $w_k \neq 0$ desde $ v_k \notin
\operatorname{Vect}(v_1, \dots, v_{k-1})$. La normalización mantiene
ortogonalidad; el \omterm{def:b1:vspaces:span}{engendrado} \omterm{def:b1:logic:statement}{enunciado} se mantiene ya que $e_k$ es un
combinación de $v_k$ y $e_i$ anteriores, de forma invertible.
\end{proof}

\begin{example}[Gram--Schmidt sobre polinomios, completo]
\label{ex:b1:euclid:legendre}
Ortonormaliza $(1, X, X^2)$ en $\R_2[X]$ con $\langle P,
Q\rangle = \int_0^1 PQ $. \emph{Paso 1}:$\norm{1}^2 = 1$, entonces
$e_1 = 1$. \emph{Paso 2}:$w_2 = X - \langle X, 1\rangle\,1 = X -
\frac12$ y $\norm{w_2}^2 = \int_0^1\bigl(x - \frac12\bigr)^2
\dd x = \frac1{12}$:$e_2 = \sqrt{12}\,\bigl(X - \frac12\bigr)$.
\emph{Paso 3}: $\langle X^2, e_1\rangle = \frac13$ y
\[
\langle X^2, e_2\rangle
= \sqrt{12}\int_0^1 x^2\Bigl(x - \frac12\Bigr)\dd x
= \frac{\sqrt{12}}{12},
\qquad\text{so}\qquad
w_3 = X^2 - \frac13 - \Bigl(X - \frac12\Bigr)
= X^2 - X + \frac16 .
\]
Su norma se calculó en \cref{exo:b1:euclid:9}: $\norm{w_3}^2 =
\frac1{180}$, de donde $e_3 = \sqrt{180}\,\bigl(X^2 - X +
\frac16\bigr)$. Los \omterm{def:b1:poly:def}{polinomios}$1$,$X - \frac12$,$ X^2 - X +
\frac16$ son, a escala, los primeros \emph{Legendre \omterm{def:b1:poly:def}{polinomios}}
del \omterm{prop:b1:reals:intervals}{intervalo} $\intcc{0}{1}$; la construcción continúa una
grado a la vez, cada nuevo \omterm{def:b1:poly:def}{polinomio} \omterm{def:b1:euclid:orthogonal}{ortogonal} a todos sus
predecesores. Observe cómo el algoritmo recicla el trabajo anterior: el
\omterm{def:b1:linmaps:projection}{proyección} restado en el paso 3 es exactamente el mejor afín
aproximación de $X^2$ encontrada en \cref{ex:b1:euclid:bestapprox}
--- Gramo--Schmidt \emph{is} iteró \omterm{thm:b1:euclid:projection}{proyección ortogonal}.
\end{example}

\begin{theorem}[Proyección ortogonal]\label{thm:b1:euclid:projection}
Sea $F$ un \omterm{def:b1:vspaces:subspace}{subespacio} del \omterm{def:b1:euclid:def}{espacio euclidiano}$E$. entonces
\[
E = F \oplus F^{\perp},
\]
y el \omterm{def:b1:linmaps:projection}{proyección} $p_F$ asociado a$F$(el \emph{ortogonal
proyección}\index{proyección ortogonal}) se proporciona, en cualquier
\omterm{def:b1:euclid:orthogonal}{ortonormal} \omterm{def:b1:vspaces:free}{base} $(e_1, \dots, e_k)$ de $F$, por $p_F(x) = \sum_i
\langle x, e_i\rangle e_i $. Realiza la distancia a$F $:
para todos los $ y \in F$,
\[
\norm{x - p_F(x)} \leq \norm{x - y},
\]
con igualdad sólo para $ y = p_F(x)$; uno escribe $d(x, F) = \norm{x -
p_F(x)}$.
\end{theorem}

\begin{proof}
Tome un \omterm{def:b1:euclid:orthogonal}{ortonormal} \omterm{def:b1:vspaces:free}{base} $(e_i)_{i \leq k}$ de $F$
(\cref{thm:b1:euclid:gramschmidt} dentro de $F$) y configure $\pi(x) =
\sum \langle x, e_i\rangle e_i \in F $. Entonces $x - \pi(x) \perp e_j $
para cada $j $(misma cancelación que arriba), por lo tanto $x - \pi(x) \in
F^\perp $:$E = F + F^\perp $. Y$F \cap F^\perp = \{0\}$: tal
El vector satisface $\langle x, x\rangle = 0$. Entonces la suma es directa y
$\pi = p_F$.

Distancia: para $ y \in F$, descomponga las piezas $x - y = (x - p_F(x)) + (p_F(x) -
y)$, \omterm{def:b1:euclid:orthogonal}{ortogonal} ($F^\perp$ y $F$); Pitágoras:
\[
\norm{x - y}^2 = \norm{x - p_F(x)}^2 + \norm{p_F(x) - y}^2
\geq \norm{x - p_F(x)}^2,
\]
igualdad si $ y = p_F(x)$.
\end{proof}

\begin{example}[Las proyecciones nunca se alargan]
\label{ex:b1:euclid:bessel}
Aplicando Pitágoras a la división $x = p_F(x) + (x - p_F(x))$:
\[
\norm{p_F(x)}^2 = \norm x^2 - \norm{x - p_F(x)}^2 \leq
\norm x^2 ,
\]
con igualdad si $x \in F$. En un \omterm{def:b1:euclid:orthogonal}{ortonormal} \omterm{def:b1:vspaces:free}{base}$(e_1,
\dots, e_k)$ de $ F $se lee $\sum_{i \leq k}\langle x,
e_i\rangle^2 \leq \norm x^2$(un \emph{Desigualdad de Bessel}):
por muchas direcciones \omterm{def:b1:euclid:orthogonal}{ortonormal} que uno mida, el cuadrado
\omterm{prop:b1:vspaces:coordinates}{coordenadas} nunca exceda la longitud al cuadrado --- compare la
igualdad exacta de \cref{ex:b1:euclid:parsevalcheck} cuando el
La familia es un \omterm{def:b1:vspaces:free}{base} completo. Esta desigualdad unifilar es lo que hace
Coeficientes de Fourier resumibles en el volumen del Año~3; aqui esta
ya explica por qué agregar más funciones \omterm{def:b1:vspaces:free}{base} al menos
El ajuste de cuadrados solo puede disminuir el residual.
\end{example}

\begin{example}[Mejor aproximación cuadrática]\label{ex:b1:euclid:bestapprox}
En $C(\intcc{0}{1})$ con $\langle f, g\rangle = \int_0^1 fg $, el
\omterm{def:b1:poly:def}{polinomio} de grado $\leq 1$ más cercano a$f(x) = x^2$ en el
la distancia asociada (media cuadrática) es $p_F(f)$ donde $F = \R_1[X]$.
Abuela--Schmidt en $(1, X)$:$e_1 = 1$,$w_2 = X - \frac12$,
$\norm{w_2}^2 = \int_0^1 (x - \frac12)^2 = \frac{1}{12}$,$e_2 =
\sqrt{12}\,(X - \tfrac12)$. entonces
\[
p_F(f) = \langle f, e_1\rangle e_1 + \langle f, e_2\rangle e_2
= \frac13 + 12\Bigl(\int_0^1 x^2\bigl(x - \tfrac12\bigr)\dd
x\Bigr)\bigl(X - \tfrac12\bigr)
= X - \frac{1}{6},
\]
utilizando $\int_0^1 x^2(x - \frac12)\dd x = \frac14 - \frac16 =
\frac{1}{12}$. La idea de "mínimos cuadrados" en una línea de \omterm{def:b1:linmaps:def}{lineal}
álgebra.
\end{example}

\begin{method}[Tres rutas a distancia $d(x, F)$]
\label{met:b1:euclid:distance}
\begin{enumerate}
 \item \emph{\omterm{def:b1:euclid:orthogonal}{ortonormal} \omterm{def:b1:vspaces:free}{base} of $F$}: luego $p_F(x) = \sum_i
 \langle x, e_i\rangle e_i $ y, por Pitágoras,
 \[
 d(x, F)^2 = \norm{x}^2 - \norm{p_F(x)}^2
 = \norm x^2 - \sum_i \langle x, e_i\rangle^2 ,
 \]
 a menudo más barato que calcular el propio $x - p_F(x)$.
 \item \emph{Ecuaciones normales}: con cualquier familia generadora de
 $F$, resuelve $\langle x - p, v_j\rangle = 0$ para el
 coeficientes de $p $(\cref{exo:b1:euclid:5}) --- no
 Se necesita ortonormalización.
 \item \emph{A través del complemento}: si $F^\perp$ es más pequeño
 que $F$(por ejemplo $F$a \omterm{def:b1:linmaps:forms}{hiperplano},$F^\perp $ una línea
 $\operatorname{Vect}(n)$), proyectar en $F^\perp $
 en lugar de:
 \[
 d(x, F) = \norm{p_{F^\perp}(x)} =
 \frac{\abs{\langle x, n\rangle}}{\norm n} ,
 \]
 cuál es la fórmula clásica de la distancia a un plano
 (\cref{exo:b1:multivar:8} lo usa).
\end{enumerate}
La ruta 3 es un caso especial de reflejo general: proyectar siempre
en cualquiera de $F$,$F^\perp $ que tenga la dimensión más pequeña.
\end{method}

\begin{example}[Ortogonalidad trigonométrica: una vista previa de Fourier]
\label{ex:b1:euclid:trigortho}
En $C(\intcc{0}{2\pi})$ con $\langle f, g\rangle =
\frac1\pi\int_0^{2\pi} fg $, la familia
\[
\Bigl(\frac{1}{\sqrt2},\ \cos x,\ \sin x,\ \cos 2x,\ \sin 2x,\
\dots\Bigr)
\]
es \omterm{def:b1:euclid:orthogonal}{ortonormal}: por ejemplo $\langle \cos px, \cos qx\rangle =
\frac1\pi\int_0^{2\pi}\cos px\cos qx\,\dd x = 0$ para $ p \neq q $
(linealizar el producto en $\frac12[\cos(p{-}q)x +
\cos(p{+}q)x]$ e integrarlo en períodos completos), mientras
$\frac1\pi\int_0^{2\pi}\cos^2 px\,\dd x = 1$. El \omterm{thm:b1:euclid:projection}{ortogonal
proyección} en el \omterm{def:b1:vspaces:span}{engendrado} del primer $2N + 1$ de estas funciones
por lo tanto tiene \omterm{prop:b1:vspaces:coordinates}{coordenadas} $\langle f, e_i\rangle$ --- \omterm{thm:b1:integration:def}{integrales}
contra cosenos y senos. Estos son el \emph{Fourier
coeficientes} de $f$, y el \omterm{def:b1:linmaps:projection}{proyección} es su mejor
aproximación trigonométrica de media cuadrática; el volumen del año ~ 3
estudia su convergencia. La ortogonalidad hace todo el trabajo:
las fórmulas para los coeficientes son

\end{example}

\section{Isometrias del avion}

\begin{definition}\label{def:b1:euclid:isometry}
Un endomorfismo $u$ de un \omterm{def:b1:euclid:def}{espacio euclidiano} es un
\emph{isometria}\index{isometria} (o \omterm{def:b1:euclid:orthogonal}{ortogonal} \omterm{def:b1:logic:map}{aplicación}) cuando
conserva la norma: $\norm{u(x)} = \norm x$ para todos $x$ ---
de manera equivalente (polarización) conserva el \omterm{def:b1:euclid:def}{producto interno};
de manera equivalente su matriz $A$ en un \omterm{def:b1:euclid:orthogonal}{ortonormal} \omterm{def:b1:vspaces:free}{base} satisface
$A^{\mathsf T} A = I$. Las isometrías forman un \omterm{def:b1:structures:group}{grupo}, el \omterm{def:b1:euclid:orthogonal}{ortogonal} \omterm{def:b1:structures:group}{grupo}
$O(E)$.
\end{definition}

\begin{example}[Reconociendo una isometría a la vista]
\label{ex:b1:euclid:recognize}
Es $A = \dfrac15\begin{pmatrix} 3 & -4\\ 4 & 3\end{pmatrix}$\omterm{def:b1:euclid:orthogonal}{ortogonal}? Columnas: normas $\frac15\sqrt{9 + 16} = 1$ y
$\frac15\sqrt{16 + 9} = 1$; producto $\frac1{25}(3\cdot(-4) +
4\cdot3) = 0$. Sí --- y $\det A = \frac{9 + 16}{25} = 1$, entonces
es la rotación $R_\theta$ con $\cos\theta = \frac35$,
$\sin\theta = \frac45$(la rotación ``$3$-$4$-$5$'', cuyo
El ángulo no es una fracción notable de $\pi$). Por el contrario,$B =
\frac{1}{\sqrt2}\begin{pmatrix} 1 & 1\\ 0 & 1\end{pmatrix}$ tiene
Aspecto escalado por determinante de unidad pero primera columna sin unidad
($\frac1{\sqrt2}$): no \omterm{def:b1:euclid:orthogonal}{ortogonal} --- determinante $\pm1$ solo
no certifica nada, hay que comprobar las columnas.
\end{example}

\begin{theorem}[Isometrías planas]\label{thm:b1:euclid:plane}
En un \omterm{def:b1:euclid:orthogonal}{ortonormal} \omterm{def:b1:vspaces:free}{base} de un plano euclidiano, las matrices de
las isometrías son exactamente
\[
R_\theta = \begin{pmatrix}
\cos\theta & -\sin\theta\\ \sin\theta & \cos\theta
\end{pmatrix}
\quad (\text{rotation of angle } \theta,\ \det = 1),
\]
\[
S_\theta = \begin{pmatrix}
\cos\theta & \sin\theta\\ \sin\theta & -\cos\theta
\end{pmatrix}
\quad (\det = -1),
\]
siendo este último el reflejo en el ángulo que forma la línea.
$\frac\theta2$ con el primer vector \omterm{def:b1:vspaces:free}{base}.
\end{theorem}

\begin{proof}
Sea $A = \begin{pmatrix} a & c\\ b & d\end{pmatrix}$ con
$A^{\mathsf T}A = I$: las columnas son unidad y \omterm{def:b1:euclid:orthogonal}{ortogonal}. el primero
la columna es $(\cos\theta, \sin\theta)$ para algunos $\theta$; el segundo,
unidad y \omterm{def:b1:euclid:orthogonal}{ortogonal} a ella, es $\pm(-\sin\theta, \cos\theta)$. el
firmar $+$ da $R_\theta $; el signo$-$ da $S_\theta $. uno comprueba
$S_\theta^2 = I$ y que el vector $(\cos\frac\theta2,
\sin\frac\theta2)$ es fijo mientras que su \omterm{def:b1:euclid:orthogonal}{ortogonal} está invertido: a
reflexión. (Y $R_\alpha R_\beta = R_{\alpha+\beta}$: el
la rotación \omterm{def:b1:structures:group}{grupo} es el ángulo \omterm{def:b1:structures:group}{grupo} --- comparar
\cref{thm:b1:complex:funceq}.)
\end{proof}

\begin{remark}[Errores comunes]
\label{rem:b1:euclid:pitfalls}
\emph{The \omterm{def:b1:linmaps:projection}{proyección} formula needs an \omterm{def:b1:euclid:orthogonal}{ortonormal} \omterm{def:b1:vspaces:free}{base}}: para un
simplemente generando la familia $(v_i)$ de $F$, la suma $\sum_i\langle
x, v_i\rangle v_i$ es \emph{no}$p_F(x)$(prueba $F = \R^2$,$ v_1
= e_1$,$v_2 = e_1 + e_2$); con una familia \omterm{def:b1:euclid:orthogonal}{no ortonormal}, resuelve
las ecuaciones normales en su lugar
(\cref{met:b1:euclid:distance}~(2)).
\emph{\omterm{def:b1:euclid:orthogonal}{Ortogonal} families must avoid $0$ to be \omterm{def:b1:vspaces:free}{libre}}: el cero
el vector es \omterm{def:b1:euclid:orthogonal}{ortogonal} para todo, incluido él mismo ---
la libertad en \cref{prop:b1:euclid:orthfree} requiere un valor distinto de cero
vectores.
\emph{$F^\perp$ depends on the \omterm{def:b1:euclid:def}{producto interno}}: en $\R_1[X]$, el
complemento de $\operatorname{Vect}(X)$ para $\int_0^1 PQ$ no es
las constantes pero $\operatorname{Vect}(1 - \frac32 X)$ ---
calcular $\int_0^1 x(1 - \frac32 x) = \frac12 - \frac12 = 0$;
"perpendicular" no tiene sentido hasta que se nombra el producto.
\emph{Do not expand $\norm{x + y}$ linearly}: lo correcto
la identidad es $\norm{x+y}^2 = \norm x^2 + 2\langle x, y\rangle +
\norm y^2$; el término cruzado desaparece sólo bajo ortogonalidad
(Pitágoras), y la desigualdad del triángulo es una
\emph{desigualdad}.
\emph{Enviar vectores unitarios a vectores unitarios no es suficiente}: $u(x,
y) = (x + y,\ 0)$\omterm{def:b1:logic:map}{aplicaciones} ambos vectores canónicos \omterm{def:b1:vspaces:free}{base} a la unidad
vector $(1, 0)$, aún $\norm{u(1,1)} = 2 \neq \sqrt2$: no
\omterm{def:b1:euclid:isometry}{isometria}. La definición exige $\norm{u(x)} = \norm x$ para
\emph{todo}$x$; en términos matriciales $A^{\mathsf T}A = I$, es decir columnas que son la unidad \emph{and pairwise \omterm{def:b1:euclid:orthogonal}{ortogonal}} --- ambas
condiciones, verificadas juntas.
\end{remark}

\begin{remark}[Donde va el producto interior]
\label{rem:b1:euclid:whereused}
\omterm{thm:b1:euclid:projection}{Proyección ortogonal} es el teorema más aplicado del capítulo:
subyace a los mínimos cuadrados (el problema del fin de semana de
\cref{ch:b1:multivar} construye líneas de regresión), Fourier
coeficientes (\cref{ex:b1:euclid:trigortho}), y el normal
ecuaciones de \cref{exo:b1:euclid:5}, cuyo análisis numérico
resuelve a escala. La clasificación $R_\theta / S_\theta$ es
completado a continuación: el problema del fin de semana clasifica \emph{todo}
transformaciones del plano que preservan la distancia, \omterm{def:b1:linmaps:def}{lineal} o no,
y su finito \omterm{def:b1:structures:group}{grupos} --- la matemática de rosetas y
polígonos regulares. En el volumen Year~2, el \omterm{def:b1:euclid:def}{producto interno} se encuentra
teoría de valores propios (matrices simétricas, formas cuadráticas); en
Año ~ 3, la geometría euclidiana de dimensión infinita se convierte en Hilbert
Teoría del espacio.
\end{remark}

\section{Problema: isometrías planas y teorema de Leonardo}

\begin{problem}\label{pb:b1:euclid:1}
Un \emph{\omterm{def:b1:euclid:isometry}{isometria} of the plane} es cualquier \omterm{def:b1:logic:map}{aplicación} $f \colon \R^2 \to
\R^2$ preservando las distancias:$\norm{f(x) - f(y)} = \norm{x - y}$
para todos $x, y $ --- no se asume linealidad. Este problema demuestra
que tales \omterm{def:b1:logic:map}{aplicaciones} son exactamente las traslaciones, rotaciones,
reflexiones y \omterm{pb:b1:euclid:1}{reflejos de deslizamiento} (la clasificación del plano
isometrías), calcula sus composiciones y determina todas
su \emph{finito} \omterm{def:b1:structures:group}{grupos}: el teorema de Leonardo, las matemáticas
detrás de patrones de rosetas. A partir de la Parte II identificamos $\R^2$ con
$\C$(\cref{ch:b1:complex}): el canónico \omterm{def:b1:euclid:def}{producto interno} es
$\langle z, w\rangle = \operatorname{Re}(z\conj w)$ y la norma
es el \omterm{def:b1:complex:field}{módulo}.\index{isometria}\index{reflejo de deslizamiento}
\index{teorema de leonardo}\index{grupo diédrico}

\medskip
\noindent\textbf{Parte I --- Every \omterm{def:b1:euclid:isometry}{isometria} is affine.}
\begin{enumerate}
 \item Compruebe que las traducciones $t_a(x) = x + a$, \omterm{def:b1:linmaps:def}{lineal}
 isometrías, y todas sus composiciones son isometrías,
 y que las isometrías forman una composición \omterm{def:b1:structures:group}{grupo}.
 \item Sea $f$ un \omterm{def:b1:euclid:isometry}{isometria} con $f(0) = 0$. Muestra que $f$
 preserva las normas, entonces --- por polarización,
 \cref{thm:b1:euclid:cs} --- ese $\langle f(x),
 f(y)\rangle = \langle x, y\rangle $ para todos los $x, y $.
 \item Aún con $f(0) = 0$: expanda $\norm{f(x + y) - f(x) -
 f(y)}^2$ y $\norm{f(\lambda x) - \lambda f(x)}^2$
 usando la pregunta 2, y concluya que $f$ es
 \emph{\omterm{def:b1:linmaps:def}{lineal}}: $f \in O(\R^2)$.
 \item Deduzca que cada \omterm{def:b1:euclid:isometry}{isometria} $f$ escribe \emph{únicamente} como
 $f = t_a \circ g$ con $a = f(0)$ y $g$ a \omterm{def:b1:linmaps:def}{lineal}
 \omterm{def:b1:euclid:isometry}{isometria} (el \emph{\omterm{def:b1:linmaps:def}{lineal} part} de $f$).
 \item Llame a $f$\emph{directo} si $\det g = 1$, \emph{indirecto}
 si $\det g = -1$. Demuestre que la parte \omterm{def:b1:linmaps:def}{lineal} de un
 La composición es la composición de las partes \omterm{def:b1:linmaps:def}{lineal}, y
 enuncie la regla de signos resultante
 (composición directa/indirecta como $+1/-1$).
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The four types.} vía
\cref{thm:b1:euclid:plane}, las isometrías \omterm{def:b1:linmaps:def}{lineal} de $\C$ son $z
\mapsto az $ y $z \mapsto a\conj z$ con $\abs a = 1$; entonces cada
el avión \omterm{def:b1:euclid:isometry}{isometria} es
\[
f(z) = a z + b \quad (\text{direct}) \qquad\text{or}\qquad
f(z) = a\conj z + b \quad (\text{indirect}), \qquad \abs a = 1 .
\]
\begin{enumerate}[resume]
 \item Verifique el diccionario: $R_\theta$ es $z \mapsto
 \eu^{\iu\theta}z $ y $S_\theta$ es $ z \mapsto
 \eu^{\iu\theta}\conj z $(verifique ambos en $1$ y $\iu$).
 \item (Caso directo) Sea $f(z) = az + b$,$\abs a = 1$. Mostrar:
 si $a = 1$,$f$ es una traducción; si $a \neq 1$,$f$ tiene
 el punto fijo único $z_0 = b/(1 - a)$ y $f(z) - z_0
 = a(z - z_0)$: un \emph{rotación} de centro $z_0$ y
 ángulo $\arg a$.
 \item (caso indirecto) Sea $f(z) = a\conj z + b$ y $v =
 a\conj b + b $. Mostrar $f \circ f = t_v $ y $ f \circ t_v =
 t_v \circ f $. Si$v = 0$: muestra que el punto medio de $z $
 y $f(z)$ es un punto fijo, y que $f$ es un
 \emph{reflexión} en una línea. Si $v \neq 0$: muestre que $r
 = t_{-v/2}\circ f $ es una reflexión cuyo eje es
 paralelo a $v$, de modo que $f = t_{v/2} \circ r$ es un
 \emph{\omterm{pb:b1:euclid:1}{reflejo de deslizamiento}}. Concluye: cada avión \omterm{def:b1:euclid:isometry}{isometria}
 es una traslación, una rotación, una reflexión o un \omterm{pb:b1:euclid:1}{deslizarse
 reflexión} (la clasificación de isometrías planas).
 \item (Composiciones) Mostrar: la composición de rotaciones de
 Los ángulos $\alpha$ y $\beta$ son una rotación del ángulo.
 $\alpha + \beta $(una traducción si $\alpha + \beta \in
 2\pi\Z $); la composición de dos reflexiones es una
 rotación del ángulo dos veces el ángulo entre los ejes (a
 traslación si los ejes son paralelos).
 \item Deduzca que todo plano \omterm{def:b1:euclid:isometry}{isometria} es una composición de al menos
 La mayoría de las tres reflexiones.
\end{enumerate}

\begin{enumerate}[resume]
 \item Clasificar completamente $f(z) = \iu\conj z + 1 + \iu$:

\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Finite \omterm{def:b1:structures:group}{grupos}: Leonardo's theorem.}
Sea $G$ un \emph{finito} \omterm{def:b1:structures:group}{grupo} de isometrías planas.
\begin{enumerate}[resume]
 \item Demuestre que las isometrías preservan los baricentros: si
 $\lambda_1 + \dots + \lambda_m = 1$ y $f = t_a \circ
 g $($g$\omterm{def:b1:linmaps:def}{lineal}), luego $f\bigl(\sum_i \lambda_i
 x_i\bigr) = \sum_i \lambda_i f(x_i)$.
 \item Configure $c = \frac{1}{\abs G}\sum_{g \in G} g(x_0)$ para cualquier
 elegido $x_0$. Demuestre que cada $h \in G$ corrige $c$: a
 El \omterm{def:b1:structures:group}{grupo} finito de isometrías tiene un punto fijo común.
 \item Deducir que, luego de conjugar por $t_{-c}$, se puede
 supongamos $G \subseteq O(\R^2)$. Sea $G^{+} = \{g \in G :
 \det g = 1\}$; mostrar que $G = G^{+}$ o $ G^{+}$
 tiene exactamente el índice $2$ en $G$(exhibe una biyección $G^+
 \to G \setminus G^+$).
 \item Demuestre que un \omterm{def:b1:structures:group}{grupo} finito de \emph{rotaciones} sobre $c$
 es cíclico: entre sus elementos elige la rotación

 \item Supongamos $G \neq G^{+}$ y elija un reflejo $s \in G$.
 Mostrar $G = G^{+} \cup sG^{+}$, ese $s r s = r^{-1}$ para

 \item Concluye (\emph{teorema de leonardo}): todo finito \omterm{def:b1:structures:group}{grupo}

\end{enumerate}

\noindent\textbf{Part V --- Dividends, and synthesis.}
\begin{enumerate}[resume]
 \item Muestre directamente que un \omterm{def:b1:structures:group}{grupo} finito de isometrías puede
