\section{Conjuntos abiertos, conjuntos cerrados.}

\begin{definition}[Barrio, conjunto abierto]\label{def:b1:topology:open}
Un \omterm{def:b1:logic:sets}{conjunto} $V \subseteq \R$ es un \emph{entorno}\index{entorno}
de $x \in \R$ cuando contiene un \omterm{prop:b1:reals:intervals}{intervalo}$\intoo{x - r}{x + r}$ para
algunos $r > 0$. A \omterm{def:b1:logic:sets}{conjunto}$U \subseteq \R$ es \emph{abierto}\index{conjunto abierto}
cuando es una vecindad de cada uno de sus puntos:
\[
\forall x \in U,\ \exists r > 0, \quad \intoo{x - r}{x + r}
\subseteq U .
\]
\end{definition}

\begin{example}\label{ex:b1:topology:open}
\omterm{def:b1:topology:open}{Abierto} \omterm{prop:b1:reals:intervals}{intervalos} son \omterm{def:b1:topology:open}{abierto}: para $x \in \intoo{a}{b}$, tome $r = \min(x -
a,\, b - x) > 0$. Las medias líneas $\intoo{a}{+\infty}$ son \omterm{def:b1:topology:open}{abierto};$\R$ y
$\emptyset$ son \omterm{def:b1:topology:open}{abierto} (este último de forma vacía).$\intcc{0}{1}$ es
\emph{no} \omterm{def:b1:topology:open}{abierto}: no \omterm{prop:b1:reals:intervals}{intervalo} alrededor de $0$ permanece adentro.
\end{example}

\begin{proposition}[Estabilidad de conjuntos abiertos]\label{prop:b1:topology:openstable}
Cualquier unión de \omterm{def:b1:topology:open}{conjuntos abiertos} es \omterm{def:b1:topology:open}{abierto}; una intersección \emph{finito} de \omterm{def:b1:topology:open}{abierto
conjuntos} es \omterm{def:b1:topology:open}{abierto}. Infinitas intersecciones pueden fallar:
$\bigcap_{n \geq 1} \intoo{-\frac 1n}{\frac 1n} = \{0\}$, no \omterm{def:b1:topology:open}{abierto}.
\end{proposition}

\begin{proof}
Unión: si $x \in \bigcup_i U_i$, entonces $x \in U_{i_0}$ para algunos $i_0$,
y el \omterm{prop:b1:reals:intervals}{intervalo} proporcionado por $U_{i_0}$ se encuentra dentro de la unión. finito
intersección: si $x \in U_1 \cap \dots \cap U_k$, tome $r = \min(r_1,
\dots, r_k) > 0$ de los radios proporcionados por cada $U_j $. Para el
contraejemplo: cualquier \omterm{prop:b1:reals:intervals}{intervalo} alrededor de $0$ contiene algo de $\frac{1}{n}$
(Arquímedes), por lo tanto sale de la intersección.
\end{proof}

Los estilos importan: el estructural (construido a partir del conocido \omterm{def:b1:topology:open}{abierto
conjuntos} mediante uniones e intersecciones finitas) se escala mejor, el

\end{example}

\begin{definition}[conjunto cerrado]\label{def:b1:topology:closed}
Un \omterm{def:b1:logic:sets}{conjunto} $F \subseteq \R$ es \emph{cerrado}\index{conjunto cerrado} cuando
El complemento $\R \setminus F$ es \omterm{def:b1:topology:open}{abierto}. Por de Morgan y
\cref{prop:b1:topology:openstable}: cualquier intersección de \omterm{def:b1:logic:sets}{conjuntos} cerrada
está cerrado, las uniones finitas de \omterm{def:b1:logic:sets}{conjuntos} cerrado están cerradas.
\end{definition}

\begin{theorem}[Caracterización secuencial de conjuntos cerrados]\label{thm:b1:topology:seqclosed}
$F$ es \omterm{def:b1:topology:closed}{cerrado} si y sólo si: para cada secuencia $(u_n)$ de puntos de
$F$ converge con algún $\ell \in \R$, el límite al que pertenece $\ell$
$F$. (``\omterm{def:b1:topology:closed}{Cerrado}''$=$ ``estable bajo límites''.)
\end{theorem}

\begin{proof}
($\Rightarrow$) Sea $F$\omterm{def:b1:topology:closed}{cerrado},$u_n \in F$,$u_n \to \ell $ y
supongamos $\ell \notin F$. El complemento es \omterm{def:b1:topology:open}{abierto}: algunos
$\intoo{\ell - r}{\ell + r}$ evita $F$. Pero la convergencia pone $u_n$
en que \omterm{prop:b1:reals:intervals}{intervalo} para $n$ grande: contradicción con $u_n \in F$.

($\Leftarrow$) Supongamos que $F$ no es \omterm{def:b1:topology:closed}{cerrado}: el complemento no es \omterm{def:b1:topology:open}{abierto},
entonces algunos $x \notin F$ no tienen \omterm{prop:b1:reals:intervals}{intervalo}$\intoo{x - r}{x + r}$ dentro
el complemento; tomando $r = \frac{1}{n+1}$, elija $u_n \in F$ con
$\abs{u_n - x} < \frac{1}{n+1}$. Entonces $u_n \in F$,$u_n \to x \notin
F $: la propiedad secuencial falla.
\end{proof}

\section{Interior, cierre, densidad.}

\begin{definition}[Interior, cierre, límite]\label{def:b1:topology:closure}
Sea $A \subseteq \R$.
\begin{itemize}
 \item Un punto $x$ es \emph{interior} a$A$ cuando $A$ es un
 \omterm{def:b1:topology:open}{entorno} de $x$; el \emph{interior}
 $\mathring{A}$\index{interior} es el \omterm{def:b1:logic:sets}{conjunto} de puntos interiores.
 \item Un punto $x$ es \emph{adherente} a$A$ cuando cada \omterm{def:b1:topology:open}{entorno}
 de $x$ cumple con $A$; el \emph{cierre}
 $\overline{A}$\index{cierre} es el \omterm{def:b1:logic:sets}{conjunto} de puntos adherentes.
 \item El \emph{límite} es $\partial A = \overline A \setminus
 \mathring A $.\index{límite}
\end{itemize}
Entonces $\mathring A \subseteq A \subseteq \overline A$.
\end{definition}

\begin{proposition}[Propiedades principales]\label{prop:b1:topology:closureprops}
\begin{enumerate}
 \item $\mathring A$ es el \omterm{def:b1:topology:open}{conjunto abierto} más grande contenido en $A$;$A$
 es \omterm{def:b1:topology:open}{abierto} y si $A = \mathring A$.
 \item $\overline A$ es el \omterm{def:b1:topology:closed}{conjunto cerrado} más pequeño que contiene $A$;$A$
 es \omterm{def:b1:topology:closed}{cerrado} y si $A = \overline A$.
 \item (Caracterización secuencial de la adherencia) $x \in \overline
 A $ si y sólo si $x $ es el límite de una secuencia de puntos de
 $A$.
 \item La complementación intercambia las nociones: $\R \setminus
 \overline A = \bigl(\R \setminus A\bigr)^{\!\circ}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(4) $x \notin \overline A$ $\iff$ algunos \omterm{def:b1:topology:open}{entorno} de $x$ evitan $A$
$\iff$ algunos \omterm{prop:b1:reals:intervals}{intervalo} alrededor de $x$ se encuentran en $\R \setminus A$ $\iff$ $x$
es \omterm{def:b1:topology:closure}{interior} a $\R \setminus A$.

(1) $\mathring A$ es \omterm{def:b1:topology:open}{abierto}: si $x \in \mathring A$, algunos
$\intoo{x-r}{x+r} \subseteq A$; cada punto$ y $ de ese \omterm{prop:b1:reals:intervals}{intervalo} tiene
un \omterm{prop:b1:reals:intervals}{intervalo} más pequeño alrededor de él dentro, por lo tanto dentro de $A$: todo
\omterm{prop:b1:reals:intervals}{intervalo} está en $\mathring A$. Cualquier \omterm{def:b1:topology:open}{abierto}$U \subseteq A$ consta de
\omterm{def:b1:topology:closure}{interior} puntos de $A$, por lo que $U \subseteq \mathring A$: mayor. el
A continuación se presenta la caracterización de la apertura.

(2) En detalle. Por (4), $\R \setminus \overline A$ es el \omterm{def:b1:topology:closure}{interior}
de $\R \setminus A$, un \omterm{def:b1:topology:open}{conjunto abierto} por (1): entonces $\overline A$ es
\omterm{def:b1:topology:closed}{cerrado} y contiene $A$. Minimidad: sea $F \supseteq A$\omterm{def:b1:topology:closed}{cerrado}. Entonces $\R \setminus F$ es \omterm{def:b1:topology:open}{abierto} y está contenido en $\R
\setminus A $, por lo que según la maximalidad en (1),
\[
\R \setminus F \subseteq \bigl(\R \setminus A\bigr)^{\!\circ}
= \R \setminus \overline A ,
\]
y tomando complementos nuevamente: $\overline A \subseteq F$. Así
$\overline A$ es el superconjunto \omterm{def:b1:topology:closed}{cerrado} más pequeño. Caracterización:
si $A = \overline A$ entonces $A$ es \omterm{def:b1:topology:closed}{cerrado} (se acaba de mostrar); si $A$ es
\omterm{def:b1:topology:closed}{cerrado}, es en sí mismo un superconjunto \omterm{def:b1:topology:closed}{cerrado} de $A$, por lo que la minimalidad
fuerzas $\overline A \subseteq A$, e igualdad.

(3) Si $u_n \in A$,$u_n \to x$: cada \omterm{def:b1:topology:open}{entorno} de $x$ contiene
algunos $u_n \in A$, entonces $x \in \overline A$. Por el contrario, si $x \in
\overline A $: cada \omterm{prop:b1:reals:intervals}{intervalo}$\intoo{x - \frac{1}{n+1}}{x +
\frac{1}{n+1}}$ se encuentra con $A $ en algún $u_n $ y $u_n \to x$.
\end{proof}

\begin{example}[Una anatomía completa]\label{ex:b1:topology:anatomy}
Sea $A = \intoc{0}{1} \,\cup\, \bigl(\Q \cap \intoo{2}{3}\bigr)
\,\cup\, \{4\}$. Calculamos los tres \omterm{def:b1:logic:sets}{conjuntos} de

La idea final: las tres operaciones actúan \emph{en la zona} ---
cada pieza de $A$ contribuye según su propia naturaleza (un
El sólido \omterm{prop:b1:reals:intervals}{intervalo} mantiene su interior, una pieza densa pero porosa se vuelve

\begin{remark}[Errores comunes en el razonamiento de conjuntos de puntos.]
(i) \emph{``Not \omterm{def:b1:topology:open}{abierto}'' does not mean ``\omterm{def:b1:topology:closed}{cerrado}''}: la mayoría de \omterm{def:b1:logic:sets}{conjuntos} son
ninguno ($\intoc{0}{1}$), y dos \omterm{def:b1:logic:sets}{conjuntos} son ambos
($\emptyset$,$\R$) --- \omterm{def:b1:topology:open}{abierto} y \omterm{def:b1:topology:closed}{cerrado} no son opuestos pero
duales mediante complementación. (ii) \emph{\omterm{def:b1:topology:closure}{Interior} and \omterm{def:b1:topology:closure}{cierre}
do not commute}: para $A = \Q$,
\[
\overline{\mathring A} = \overline\emptyset = \emptyset
\qquad\text{while}\qquad
\bigl(\,\overline A\,\bigr)^{\!\circ} = \mathring \R = \R :
\]
los dos operadores iterados difieren tanto como puede hacerlo \omterm{def:b1:logic:sets}{conjuntos}. (iii)
\emph{Infinite unions of \omterm{def:b1:topology:closed}{conjuntos cerrados} can fail to be \omterm{def:b1:topology:closed}{cerrado}}:
$\bigcup_{n\geq1} \intcc{\frac1n}{1} = \intoc{0}{1}$ --- el
espejo de la intersección contraejemplo de
\cref{prop:b1:topology:openstable}. (iv) \emph{\omterm{def:b1:topology:dense}{Denso} does not
mean big}: $\Q$ es \omterm{def:b1:topology:dense}{denso}, contable, con \omterm{def:b1:topology:closure}{interior} vacío, y
su complemento también es \omterm{def:b1:topology:dense}{denso}; \omterm{def:b1:topology:dense}{densidad} dice ``arbitrariamente cerca de
todo'', no ``casi todo'' --- el problema del fin de semana
\omterm{pb:b1:topology:1}{conjunto de cantantes} (\cref{pb:b1:topology:1}) hace el punto opuesto,
un \omterm{def:b1:logic:sets}{conjunto} topológicamente pequeño que es incontablemente grande.
\end{remark}

\begin{definition}[Densidad, forma topológica]\label{def:b1:topology:dense}
$A$ es \emph{denso}\index{densidad} en $\R$ cuando $\overline A = \R$
--- de manera equivalente, cada \omterm{def:b1:topology:open}{abierto} \omterm{prop:b1:reals:intervals}{intervalo} que no esté vacío cumple con $A$;
equivalentemente (por \cref{prop:b1:topology:closureprops}~(3)), cada
real es un límite de elementos de $A$. Ejemplos:$\Q$,$\R \setminus
\Q $, las diádicas (\cref{exo:b1:reals:8}), densa \omterm{def:b1:structures:subgroup}{subgrupos}
(\cref{exo:b1:reals:9}).
\end{definition}

en su lugar, sea \omterm{def:b1:topology:dense}{denso} solo en una parte de la línea. Las diádicas de
$\intcc{0}{1}$, es decir $D \cap \intcc{0}{1}$
(\cref{exo:b1:reals:8}), cumple con todos los \omterm{def:b1:topology:open}{abierto} \omterm{prop:b1:reals:intervals}{intervalo} incluidos en
$\intcc{0}{1}$ pero, por supuesto, extrañamos por completo a$\intoo{2}{3}$: ellos
son \omterm{def:b1:topology:dense}{denso} \emph{in} $\intcc{0}{1}$, es decir $\overline{D \cap

\end{example}

\begin{example}[Densidad de manejo]\label{ex:b1:topology:densitytricks}
Tres movimientos rápidos que se repiten constantemente. \emph{Ampliando}: si
$A$ es \omterm{def:b1:topology:dense}{denso} y $A \subseteq B$, luego $B$ es \omterm{def:b1:topology:dense}{denso} (cada
\omterm{prop:b1:reals:intervals}{intervalo} ya cumple con $A$). \emph{Transporte}: si $A$ es
\omterm{def:b1:topology:dense}{denso}, también lo es $\lambda A + \mu$ para $\lambda \neq 0$ --- un
\omterm{prop:b1:reals:intervals}{intervalo} $I$ se encuentra con $\lambda A + \mu $ si el \omterm{prop:b1:reals:intervals}{intervalo}
$\frac{I - \mu}{\lambda}$ se encuentra con $A$; por tanto, los múltiplos impares de
$10^{-9}$, digamos, son \omterm{def:b1:topology:dense}{denso}. \emph{La intersección falla}: dos \omterm{def:b1:topology:dense}{denso}
\omterm{def:b1:logic:sets}{conjuntos} pueden extrañarse por completo ($\Q$ y $\R \setminus \Q$):
\omterm{def:b1:topology:dense}{densidad} sobrevive a uniones y afines \omterm{def:b1:logic:map}{aplicaciones}, nunca a intersecciones.
\end{example}

\section{Compacidad de los segmentos}

\begin{theorem}[Los segmentos son compactos secuencialmente]\label{thm:b1:topology:compact}
Sea $a \leq b$. Cada secuencia de puntos de $\intcc{a}{b}$ tiene un
\omterm{def:b1:seq:subsequence}{subsecuencia} convergiendo a un punto \emph{of}
$\intcc{a}{b}$.\index{compacidad}

De manera más general, los subconjuntos de $\R$ con esta propiedad (cada
secuencia tiene un \omterm{def:b1:seq:subsequence}{subsecuencia} que converge en el \omterm{def:b1:logic:sets}{conjunto}) son exactamente los
\emph{\omterm{def:b1:topology:closed}{cerrado} and bounded}\omterm{def:b1:logic:sets}{conjuntos}.
\end{theorem}

\begin{proof}
Una secuencia en $\intcc{a}{b}$ está acotada, por lo que Bolzano--Weierstrass
(\cref{thm:b1:seq:bw}) extrae un \omterm{def:b1:seq:subsequence}{subsecuencia} convergente; su limite
permanece en $\intcc{a}{b}$ porque los segmentos son \omterm{def:b1:topology:closed}{cerrado}
(\cref{thm:b1:topology:seqclosed}).

Caso general. \emph{(\omterm{def:b1:topology:closed}{Cerrado} bounded $\Rightarrow$ compact)}: deja
$F$ será \omterm{def:b1:topology:closed}{cerrado} y acotado, y $(u_n)$ una secuencia en $F$.
La cota de $F$ limita la secuencia, por lo que Bolzano--Weierstrass
extrae $u_{\varphi(n)} \to \ell$; y $\ell \in F$ porque $F$
es \omterm{def:b1:topology:closed}{cerrado} y \omterm{def:b1:seq:subsequence}{subsecuencia} es una secuencia convergente de puntos
de $F$(\cref{thm:b1:topology:seqclosed}): las dos hipótesis
se consumen uno cada uno, la cota por la existencia del límite,
cierre para sus miembros. \emph{(Compact $\Rightarrow$\omterm{def:b1:topology:closed}{cerrado} and
bounded)}: si $F$ no está limitado, elija $u_n \in F$ con $\abs{u_n} \geq
n$; cada \omterm{def:b1:seq:subsequence}{subsecuencia} es ilimitado, por lo tanto divergente
(\cref{prop:b1:seq:first}): no hay ningún \omterm{def:b1:seq:subsequence}{subsecuencia} convergente. Si $F$
no es \omterm{def:b1:topology:closed}{cerrado}, tome $u_n \in F$ con $u_n \to \ell \notin F$
(\cref{thm:b1:topology:seqclosed}): cada \omterm{def:b1:seq:subsequence}{subsecuencia} converge a
$\ell \notin F$, por lo que ningún \omterm{def:b1:seq:subsequence}{subsecuencia} converge \emph{in}$F$.
\end{proof}

\begin{example}[Conjuntos compactos anidados]\label{ex:b1:topology:nested}
Un primer ejercicio para el teorema. Sean $K_0 \supseteq K_1 \supseteq
K_2 \supseteq \dots $ subconjuntos compactos no vacíos (\omterm{def:b1:topology:closed}{cerrado} acotados)
de $\R$. Entonces $\bigcap_n K_n \neq \emptyset$. De hecho, elija $x_n
\in K_n $ para cada $n $: la secuencia se encuentra en el compacto $K_0$,
entonces un \omterm{def:b1:seq:subsequence}{subsecuencia} $x_{\varphi(n)}$ converge a algún $x$
(\cref{thm:b1:topology:compact}). Para cada $m$ fijo, los términos
$x_{\varphi(n)}$ con $\varphi(n) \geq m$ se encuentran todos en el \omterm{def:b1:topology:closed}{cerrado}
establezca $K_m$, por lo que el límite $x$ se encuentra en $K_m$
(\cref{thm:b1:topology:seqclosed}); ya que $m$ era arbitrario,$x \in
\bigcap_n K_n $. Un compañero útil: si un \omterm{def:b1:topology:open}{conjunto abierto}$U$ contiene
$\bigcap_n K_n$, luego $U \supseteq K_n$ para algunos $n$ --- aplicar
el mismo argumento para los puntos $x_n \in K_n \setminus U$; el limite
$x$ estaría en $\bigcap K_n \subseteq U$, pero $U$\omterm{def:b1:topology:open}{abierto} fuerza
$x_{\varphi(n)} \in U$ en definitiva, una contradicción. ambos
\omterm{def:b1:logic:statement}{enunciados} fallan sin compacidad: $\bigcap_n \intoo{0}{\frac
1n} = \emptyset$ y $\bigcap_n \intco{n}{+\infty} = \emptyset$.
La idea final: la compacidad convierte una cadena infinita de
afirmaciones de no vacuidad en un único punto límite --- es el
herramienta que sobrevive al paso a la intersección infinita, y la
El problema del fin de semana (\cref{pb:b1:topology:1}) se apoyará en él dos veces.
\end{example}

\section{Problema: El set de Cantor, pequeño y enorme a la vez}

\begin{problem}[{Problema del fin de semana --- los tercios medios de Cantor
\omterm{def:b1:logic:sets}{conjunto}: longitud cero, incontable, perfecto y $C + C =
\intcc{0}{2}$}]
\label{pb:b1:topology:1}
Retire de $\intcc{0}{1}$ su tercio medio \omterm{def:b1:topology:open}{abierto}, luego el tercio medio.
tercio de cada segmento restante, y repetir para siempre: ¿qué
sobrevive es el \emph{\omterm{pb:b1:topology:1}{conjunto de cantantes}} $C$, el fundamental
fábrica de análisis de contraejemplos. Este problema lo construye,
lo lee a través de la maquinaria base-$3$ de \cref{pb:b1:reals:1},
y establece su retrato paradójico: longitud total cero, aún
incontable; vacío \omterm{def:b1:topology:closure}{interior}, pero ningún punto aislado; totalmente
desconectado, pero $C + C$ llena todo el segmento
$\intcc{0}{2}$.\index{conjunto de cantantes} Formalmente:$C_0 = \intcc{0}{1}$,
y $C_{n+1}$ se obtiene de $C_n$ eliminando el medio \omterm{def:b1:topology:open}{abierto}
tercio de cada segmento de $C_n$; finalmente $C = \bigcap_{n \geq 0}
C_n $. En todo momento, un \emph{código ternario} de $x \in \intcc{0}{1}$
es cualquier cadena de dígitos $(d_k)_{k\geq1}$ con $d_k \in \{0, 1, 2\}$
cuyo valor $\sup_n \sum_{k=1}^n d_k 3^{-k}$ es igual a$x$ ---
se permiten códigos incorrectos (eventualmente $2$); por
\cref{pb:b1:reals:1} (preguntas 9--11), cada $x \in
\intcc{0}{1}$ tiene uno o dos códigos, dos exactamente cuando $x = m/3^N
\in \intoo{0}{1}$.

\medskip
\noindent\textbf{Parte I --- The construction.}
\begin{enumerate}
 \item Describe $C_1$ y $C_2$ explícitamente como uniones de
 segmentos, y demostrar por inducción: $C_n$ es un disjunto
 unión de segmentos $2^n$\omterm{def:b1:topology:closed}{cerrado}, cada uno de longitud $3^{-n}$.
 \item Demuestre que $C$ es \omterm{def:b1:topology:closed}{cerrado}, acotado --- por lo tanto compacto
 (\cref{thm:b1:topology:compact}) --- no vacío, y que
 cada punto final de cada segmento de cada $C_n$ pertenece
 a $C$.
 \item La longitud total de $C_n$ es $\bigl(\frac23\bigr)^n$.
 Deduzca que por cada $\varepsilon > 0$, el \omterm{def:b1:logic:sets}{conjunto}$C$ puede
 estar cubierto por un número finito de segmentos de longitud total
 $\leq \varepsilon$: el \omterm{pb:b1:topology:1}{conjunto de cantantes} tiene \emph{longitud
 cero}.
 \item Muestre que un \omterm{prop:b1:reals:intervals}{intervalo} contenido en $C$ tiene una longitud $\leq
 3^{-n}$ para cada $n $, por lo tanto, es un singleton o está vacío:
 $\mathring C = \emptyset$. Siendo \omterm{def:b1:topology:closed}{cerrado} con vacío
 \omterm{def:b1:topology:closure}{interior}, $C$ es \emph{nowhere \omterm{def:b1:topology:dense}{denso}}.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The ternary code.}
\begin{enumerate}[resume]
 \item Demostrar la recursividad de autosimilitud
 \[
 C_{n+1} = \tfrac13 C_n \,\cup\,
 \bigl(\tfrac23 + \tfrac13 C_n\bigr),
 \qquad\text{hence}\qquad
 C = \tfrac13 C \,\cup\, \bigl(\tfrac23 + \tfrac13
 C\bigr),
 \]
 las dos piezas están separadas: $C$ son dos copias de
 sí mismo a escala $\frac13$.
 \item Demostrar por inducción en $n$:$x \in C_n$ si y sólo si
 $x$ tiene un código ternario cuyos primeros dígitos $n$ se encuentran en
 $\{0, 2\}$. Deducir, utilizando el hecho de que $x$ tiene como máximo
 dos códigos: $x \in C$ si y sólo si $x$ tiene un código con
 \emph{ningún dígito es igual a $1$} (un código \emph{$1$ libre}).
 \item Códigos en acción: proporcione códigos gratuitos $1$ para $0$,$1$,
 $\frac13$,$\frac23$; mostrar $\frac14 = (0.\overline{02})_3$
 y $\frac34 = (0.\overline{20})_3$, por lo que ambos pertenecen a
 $C$; y verifique que $\frac14$ sea \emph{no} un punto final
 de cualquier $C_n $(los puntos finales tienen el formato $m/3^n$).
 \item Demuestre que cada $x \in C$ tiene \emph{exactamente uno}$1$-libre
 código \emph{(cuando $x$ tiene dos códigos, demuestra que exactamente
 uno del par contiene el dígito $1$)}. Concluir: el
 El valor \omterm{def:b1:logic:map}{aplicación} es una biyección de cadenas $\{0,2\}$ en
 $C$.
 \item (Diagonal) Sea $k \mapsto x_k$ cualquier \omterm{def:b1:logic:map}{aplicación}$\N^* \to C$.
 Cree una cadena $\{0, 2\}$ que difiera en el índice $k$ de
 el código de $x_k$ y concluir que $C$ es incontable
 --- mientras que, por el contrario, la pregunta 3 dice que es
 métricamente insignificante.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Topological portrait.}
\begin{enumerate}[resume]
 \item Reúna el registro hasta el momento: $C$ es compacto, incontable,
 de longitud cero, en ninguna parte \omterm{def:b1:topology:dense}{denso}. ¿Qué contención única
 $C \subseteq C_n$ lleva cada propiedad?
 \item ($C$ es perfecto) Sea $x \in C$ con código libre $1$
 $(d_k)$. Al invertir el dígito $d_n $($0 \leftrightarrow
 2$) se produce $x_n \in C $ con $\abs{x_n - x} = 2 \cdot
 3^{-n}$. Concluimos que $C $ no tiene un punto aislado: cada
 punto de $C$ es un límite de \emph{otro} puntos de $C$.
 \item Demuestre que los puntos finales de la pregunta 2 forman un contable
 \omterm{def:b1:topology:dense}{denso} subconjunto de $C$\emph{(truncate the code after $n$
 digits and continue with $0$ s; countability as in
 \cref{exo:b1:topology:10})}. Conclusión: el punto típico
 de $C$--- como $\frac14$ --- es \emph{no} un punto final:
 Los puntos finales son un esqueleto contable dentro de un incontable.
 cuerpo.
 \item (Totalmente desconectado) Sea $x < y $ en $C$. Elija $n$
 con $3^{-n} < y - x$ y producir un punto $z \in
 \intoo{x}{y}$ con $ z \notin C $. Concluir que el único
 Los \omterm{prop:b1:reals:intervals}{intervalos} no vacíos contenidos en $C$ son singletons.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Arithmetic of $C$.}
\begin{enumerate}[resume]
 \item Mostrar $1 - C = C$\emph{(what does $x \mapsto 1 - x$ do
 to a $1$-free code? recall $1 = (0.\overline{2})_3$)}.
 \item (Adición de códigos) Mostrar que si $x$,$x'$ tienen códigos
 $(a_k)$,$(b_k)$, luego $x + x' = \lim_n\,(t_n + t'_n)$
 donde $t_n, t'_n$ son las sumas parciales. Deducir: cada $ y
 \in \intcc{0}{1}$ es el \emph{punto medio} de dos puntos
 de $C$--- dado un código $(e_k)$ de $ y $, elija dígitos
 $a_k, b_k \in \{0, 2\}$ con $\frac{a_k + b_k}{2} =
 e_k$.
 \item Concluye $C + C = \intcc{0}{2}$ y, con la pregunta 14,
 $C - C = \intcc{-1}{1}$. Ejemplo concreto: escriba $1$ como
 una suma de los dos criterios no finales encontrados en la pregunta 7.
 \item Refleja: un \omterm{def:b1:logic:sets}{conjunto} de longitud cero cuya diferencia \omterm{def:b1:logic:sets}{conjunto} llena
 $\intcc{-1}{1}$. ¿Por qué no hay contradicción entre
 ``$C$ es métricamente insignificante'' y ``$C + C$ tiene
 longitud''? (Una frase; piensa en qué longitud
 controla y lo que no controla.)
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Members, rational and irrational.}
\begin{enumerate}[resume]
 \item Combine la pregunta 8 con el criterio de periodicidad de
 \cref{pb:b1:reals:1}: un punto de $C$ es racional si y
 sólo si su código libre $1$ es eventualmente periódico. correr
 la división larga base-$3$ para verificar $\frac1{13} =
 (0.\overline{002})_3 \in C $.
 \item Produzca explícitamente un miembro \emph{irracional} de $C$:
 el valor del código con $d_k = 2$ en el triangular
 posiciones $k = \frac{j(j+1)}{2}$ y $d_k = 0$
 en otro lugar. Justificar la irracionalidad por las crecientes brechas
 argumento de \cref{pb:b1:reals:1} (pregunta 20).
 \item (En un segmento completo) Considere $h$ mapear el punto de
 $C$ con $1$ código libre $(d_k)$ al valor del
 \emph{binario} cadena $\bigl(\frac{d_k}2\bigr)$, es decir $h(x) = \sup_n \sum_{k=1}^n \frac{d_k}{2}\,2^{-k}$.
 Muestre que $h$\omterm{def:b1:logic:map}{aplicaciones}$C$\emph{sobre}$\intcc{0}{1}$. entonces
 el insignificante $C$ se proyecta en un segmento de lleno
 longitud --- una segunda prueba de que $C$ es incontable.
 \item (Longitud autosimilar) Supongamos alguna noción de longitud $L$
 se definieron para $C$ y sus copias reducidas, respetando
 escala ($L(\lambda A) = \lambda L(A)$), traducción
 invariancia y aditividad sobre la descomposición disjunta
 de la pregunta 5. Demuestre que entonces $L(C) = \frac23\,L(C)$,
 forzando $L(C) = 0$: autosemejanza sola ya
 sentencias $C$ de longitud cero.
\end{enumerate}
