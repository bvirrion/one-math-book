\section{Leyes de composición}

\begin{definition}\label{def:b1:structures:law}
Un \emph{ley de composición}\index{ley de composición} en un \omterm{def:b1:logic:sets}{conjunto} $E$ es
a \omterm{def:b1:logic:map}{aplicación} $E \times E \to E$, escrito $(x, y) \mapsto x * y $. es
\emph{de asociación} cuando $(x*y)*z = x*(y*z)$ siempre, \emph{conmutativo}
cuando $x * y = y * x$ siempre. Un elemento $e$ es un \emph{identidad} cuando
$e * x = x * e = x$ para todos los $x$; entonces $x'$ es un \emph{inverso} de $x$
cuando $x * x' = x' * x = e$.
\end{definition}

\begin{proposition}[Singularidad]\label{prop:b1:structures:unique}
Una ley tiene como máximo una identidad; por una ley asociativa con identidad,
cada elemento tiene como máximo una inversa.
\end{proposition}

\begin{proof}
Si $e$ y $e'$ son identidades:$e = e * e' = e'$. Si$x'$ y $x''$
invertir $x$:$x' = x' * e = x' * (x * x'') = (x' * x) * x'' = e * x''
= x''$.
\end{proof}

\section{Grupos}

\begin{definition}[grupo]\label{def:b1:structures:group}
Un \emph{grupo}\index{grupo} $(G, *)$ es un \omterm{def:b1:logic:sets}{conjunto} con ley asociativa
admitiendo una identidad y en la que cada elemento tiene una inversa. el
grupo es \emph{abeliano}\index{grupo abeliano} cuando la ley es
conmutativo.
\end{definition}

\begin{example}\label{ex:b1:structures:groups}
$(\Z, +)$,$(\Q, +)$,$(\R, +)$,$(\C, +)$;$(\Q^*, \times)$,
$(\R^*, \times)$,$(\C^*, \times)$,$(\mathbb{U}_n, \times)$(\omterm{def:b1:complex:unity}{raíces de
unidad}, \cref{def:b1:complex:unity}); el \omterm{def:b1:logic:sets}{conjunto} $\mathfrak{S}(E)$ de
biyecciones de un \omterm{def:b1:logic:sets}{conjunto} $E$ sobre sí mismo, bajo composición --- el
\emph{grupo simétrico}\index{grupo simétrico} de $E$, \omterm{def:b1:structures:group}{no abeliano} como
tan pronto como $\abs E \geq 3$. No \omterm{def:b1:structures:group}{grupos}:$(\N, +)$(sin inversos),
$(\Z, \times)$(solo $\pm 1$ invertible).
\end{example}

\begin{proposition}[reglas de calculo]\label{prop:b1:structures:rules}
En un \omterm{def:b1:structures:group}{grupo} $G$(escrito multiplicativamente, identidad $e$):
\begin{enumerate}
 \item Cancelación : $ax = ay \implies x = y $ y $xa = ya \implies
 x = y $;
 \item $(ab)^{-1} = b^{-1} a^{-1}$ y $(a^{-1})^{-1} = a$;
 \item para $a, b \in G$, cada ecuación$ ax = b$ y $xa = b$ tiene una
 solución única ($x = a^{-1}b$, resp.$x = b a^{-1}$).
\end{enumerate}
\end{proposition}

\begin{proof}
(1) Multiplicar por $a^{-1}$ en el lado apropiado, usando
asociatividad. (2) $(b^{-1}a^{-1})(ab) = b^{-1}(a^{-1}a)b = b^{-1}b =
e $ y simétricamente; concluye la unicidad de lo inverso; el segundo
El punto es \cref{prop:b1:structures:unique} aplicado a $a^{-1}$. (3)
Sustituya y utilice (1) para unicidad.
\end{proof}

\begin{definition}[Subgrupo]\label{def:b1:structures:subgroup}
Un subconjunto $H$ de un \omterm{def:b1:structures:group}{grupo}$G$ es un \emph{subgrupo}\index{subgrupo}
(escrito $H \leq G$) cuando contiene $e$, es estable ante la ley
y bajo inversión. Entonces $H$ es en sí mismo un \omterm{def:b1:structures:group}{grupo}.

\emph{Criterio:} un $H \subseteq G$ no vacío es un subgrupo si y sólo
si
\[
\forall x, y \in H, \quad x y^{-1} \in H .
\]
\end{definition}

\begin{proof}[Prueba del criterio]
Un \omterm{def:b1:structures:subgroup}{subgrupo} obviamente lo satisface. Por el contrario, dejemos que $H \neq \emptyset$
satisfacerlo y elegir $x_0 \in H$. Entonces $e = x_0 x_0^{-1} \in H$; para
$ y \in H$,$ y^{-1} = e\,y^{-1} \in H$; y para $x, y \in H$,$xy =
x (y^{-1})^{-1} \in H $.
\end{proof}

\begin{example}\label{ex:b1:structures:subgroups}
$\mathbb{U}_n \leq (\C^*, \times)$: no vacío y para $z, w \in
\mathbb{U}_n $,$(zw^{-1})^n = z^n (w^n)^{-1} = 1$. El \omterm{def:b1:structures:subgroup}{subgrupos} de
$(\Z, +)$ son exactamente los $n\Z$(probados en
\cref{thm:b1:arith:gcd}). Una intersección de \omterm{def:b1:structures:subgroup}{subgrupos} es siempre una
\omterm{def:b1:structures:subgroup}{subgrupo}, pero un sindicato casi nunca lo es
(\cref{exo:b1:structures:6}).
\end{example}

\begin{remark}[Errores comunes con las estructuras.]\label{rem:b1:structures:pitfalls}
\begin{enumerate}
 \item \emph{La estabilidad ante la ley no es suficiente.} $\N$ es
 estable bajo adición dentro de $\Z$, contiene $0$, pero es
 no \omterm{def:b1:structures:subgroup}{subgrupo}: faltan inversos. el criterio
 $xy^{-1} \in H$ prueba todo a la vez --- pero solo
 después de comprobar $H \neq \emptyset$.
 \item \emph{Reflejos nobelianos.} En general \omterm{def:b1:structures:group}{grupo}, $(ab)^2 =
 abab $, que es $a^2b^2$ solo cuando $a $ y $b$ conmutan;
 Asimismo $(ab)^{-1} = b^{-1}a^{-1}$, \omterm{def:b1:structures:order}{orden} invertidos.
 Toda identidad importada del álgebra escolar debe ser
 redirigido de los axiomas o marcado como conmutativo.
 \item \emph{\omterm{def:b1:structures:morphism}{Núcleo} versus image.} $\ker f$ vive en el
 \emph{fuente}, $\operatorname{im} f$ en el objetivo;
 ``$f$\omterm{def:b1:logic:inj}{inyectivo} si $\ker f$ trivial''
 (\cref{prop:b1:structures:kernel}) no tiene análogo con
 la imagen ($\operatorname{im} f = G'$ es sobreyectividad).
 \item \emph{\omterm{def:b1:structures:ring}{Anillos} are not \omterm{def:b1:structures:group}{grupos} for $\times$.} En un \omterm{def:b1:structures:ring}{anillo}, la mayoría
 No es necesario que los elementos sean invertibles y que se cancelen mediante $a$.
 requiere que $a$ sea una unidad o que \omterm{def:b1:structures:ring}{anillo} sea un \omterm{def:b1:structures:field}{integral
 dominio}: en $\Z/12\Z$,$\overline3\,\overline2 =
 \overline3\,\overline6$ y $\overline2 \neq \overline6$
 (\cref{ex:b1:structures:zncomputation}).
\end{enumerate}
\end{remark}

\begin{definition}[Morfismo de grupo]\label{def:b1:structures:morphism}
Sean $(G, *)$ y $(G', \star)$\omterm{def:b1:structures:group}{grupos}. A \omterm{def:b1:logic:map}{aplicación}$f \colon G \to G'$
es un \emph{morfismo}\index{morfismo!de grupos} cuando
\[
\forall x, y \in G, \qquad f(x * y) = f(x) \star f(y).
\]
Luego $f(e_G) = e_{G'}$ y $f(x^{-1}) = f(x)^{-1}$. el
\emph{núcleo}\index{núcleo} y \emph{imagen} de $f$ son
\[
\ker f = f^{-1}(\{e_{G'}\}) \leq G,
\qquad
\operatorname{im} f = f(G) \leq G' .
\]
Un morfismo \omterm{def:b1:logic:inj}{biyectivo} es un \emph{isomorfismo}; su inverso \omterm{def:b1:logic:map}{aplicación} es entonces
automáticamente un morfismo.
\end{definition}

\begin{proof}[Prueba de las afirmaciones]
$f(e) = f(e * e) = f(e)\star f(e)$, y cancelar $f(e)$ da
$e_{G'} = f(e)$. Entonces $f(x)\star f(x^{-1}) = f(x x^{-1}) = e_{G'}$
identifica $f(x^{-1})$ como lo inverso. \omterm{def:b1:structures:morphism}{Núcleo}:$e \in \ker f$; si
$x, y \in \ker f$,$f(xy^{-1}) = f(x)f(y)^{-1} = e$; el criterio
aplica. Imagen: mismo criterio con $f(x)f(y)^{-1} = f(xy^{-1})$.
Inverso de un isomorfismo: para $u, v \in G'$, escriba $u = f(x)$,$v =
f(y)$; luego $f^{-1}(u \star v) = f^{-1}(f(xy)) = xy =
f^{-1}(u) f^{-1}(v)$.
\end{proof}

\begin{proposition}[Inyectividad a través del kernel]\label{prop:b1:structures:kernel}
A \omterm{def:b1:structures:group}{grupo} \omterm{def:b1:structures:morphism}{morfismo} $f$ es \omterm{def:b1:logic:inj}{inyectivo} si y sólo si $\ker f =
\{e\}$.
\end{proposition}

\begin{proof}
Si $f$ es \omterm{def:b1:logic:inj}{inyectivo},$\ker f$ solo puede contener la \omterm{def:b1:logic:map}{imagen inversa} de
$e_{G'}$, que es $e$. Por el contrario, si $\ker f = \{e\}$ y $f(x) =
f(y)$, entonces $f(xy^{-1}) = f(x) f(y)^{-1} = e_{G'}$, entonces $xy^{-1} = e $,
es decir,\ $x = y $.
\end{proof}

\begin{example}\label{ex:b1:structures:expmorphism}
$\exp \colon (\R, +) \to (\R_+^*, \times)$ es un \omterm{def:b1:structures:morphism}{morfismo}
($\eu^{x+y} = \eu^x \eu^y $), \omterm{def:b1:logic:inj}{biyectivo}
(\cref{prop:b1:functions:expln}): el aditivo y multiplicativo
Las estructuras son isomórficas --- la razón de ser histórica de
logaritmos. Otro \omterm{def:b1:structures:morphism}{morfismo}: $\theta \mapsto \eu^{\iu\theta}$ de
$(\R, +)$ en el círculo unitario $(\mathbb{U}, \times)$, con \omterm{def:b1:structures:morphism}{núcleo}
$2\pi\Z$.
\end{example}

\begin{example}[El morfismo de signo]\label{ex:b1:structures:signmorphism}
El envío \omterm{def:b1:logic:map}{aplicación} $s \colon (\R^*, \times) \to (\{\pm1\}, \times)$
$x$ a su signo es un \omterm{def:b1:structures:morphism}{morfismo}: el signo de un producto es el

\begin{definition}[Potencias, orden de un elemento]\label{def:b1:structures:order}
En un \omterm{def:b1:structures:group}{grupo} $G$(notación multiplicativa), configure $x^0 = e$,$x^{k+1} =
x^k x $ y $x^{-k} = (x^k)^{-1}$ para $k \in \N $; luego $x^{k+l} = x^k
x^l $ para todos los $k, l \in \Z $, por lo que $k \mapsto x^k $ es un \omterm{def:b1:structures:morphism}{morfismo}$(\Z,
+) \to G $cuya imagen $\langle x \rangle = \{x^k : k \in \Z\}$ es un
\omterm{def:b1:structures:subgroup}{subgrupo}, el \omterm{def:b1:structures:subgroup}{subgrupo} \emph{generado} por $x$. el
\emph{orden}\index{orden de un elemento} de $x$ es el mínimo $m \geq
1$ con $ x^m = e $ si existe (entonces $\langle x\rangle = \{e, x,
\dots, x^{m-1}\}$ tiene exactamente elementos $m $ y $ x^k = e \iff m
\mid k $), y $\infty$ en caso contrario.
\end{definition}

\begin{example}\label{ex:b1:structures:order}
En $(\C^*, \times)$:$\iu$ tiene \omterm{def:b1:structures:order}{orden}$4$, con $\langle \iu \rangle
= \{1, \iu, -1, -\iu\} = \mathbb{U}_4$; de manera más general,$\omega =
\eu^{2\iu\pi/n}$ tiene \omterm{def:b1:structures:order}{orden}$n$ y $\langle\omega\rangle =
\mathbb{U}_n $. En $(\Z, +)$, cada $x \neq 0$ tiene \omterm{def:b1:structures:order}{orden} infinito.
Por qué se mantienen las afirmaciones en la definición: si $x$ tiene \omterm{def:b1:structures:order}{orden}$m$, divida
cualquier $k$ por $m $($k = mq + r$,$0 \leq r < m$,
\cref{thm:b1:arith:division}): $x^k = (x^m)^q x^r = x^r$, por lo que el
ciclo de energía con período $m$, los elementos enumerados están en pares
distinto por la minimalidad de $m$, y $x^k = e$ fuerza a$r = 0$. \omterm{def:b1:structures:order}{Órdenes}
de \omterm{def:b1:counting:objects}{permutaciones} se calculan en el siguiente problema de fin de semana.
\end{example}

\begin{example}[Pedidos dentro de $\mathbb U_{12}$]\label{ex:b1:structures:orderU12}
¿Qué es el \omterm{def:b1:structures:order}{orden} de $\omega^k$ en $\mathbb U_n$, para $\omega =
\eu^{2\iu\pi/n}$? Uno tiene $(\omega^k)^m = 1$ si $ n \mid km $, y
escribiendo $d = \gcd(n, k)$,$n = dn'$,$k = dk'$ con $\gcd(n', k')
= 1$:$n \mid km \iff n' \mid k'm \iff n' \mid m $(Gauss
lema, \cref{thm:b1:arith:gauss}). Lo mínimo como $m \geq 1$ es
$n' = \frac{n}{\gcd(n,k)}$. En$\mathbb U_{12}$ por ejemplo,
$\omega^8$ tiene \omterm{def:b1:structures:order}{orden}$\frac{12}{\gcd(12,8)} = 3$(de hecho
$\omega^8 = \eu^{4\iu\pi/3} \in \mathbb U_3$), mientras que $\omega^5$
tiene \omterm{def:b1:structures:order}{orden} $12$: genera el \omterm{def:b1:structures:group}{grupo} completo, aunque no es
el generador "estándar". Contando los generadores --- el $k$
con $\gcd(k, n) = 1$ --- recupera los conteos \omterm{cor:b1:arith:bezout}{coprimo} de
\cref{ex:b1:counting:coprime}: \omterm{def:b1:structures:group}{grupo} encuentro de teoría y conteo.
\end{example}

\section{Anillos y campos}

\begin{definition}[anillo]\label{def:b1:structures:ring}
Un \emph{anillo}\index{anillo} $(A, +, \times)$ es un \omterm{def:b1:logic:sets}{conjunto} con dos leyes tales
que: $(A, +)$ es un \omterm{def:b1:structures:group}{grupo abeliano} (identidad $0$);$\times$ es
asociativo con una identidad $1$; y $\times$ distribuye sobre $+$ en
ambos lados. El anillo es \emph{conmutativo} cuando lo es $\times$. un
el elemento $a$ es \emph{reversible} (un \emph{unidad}) cuando $ab = ba = 1$
para algunos $b$; las unidades forman un \omterm{def:b1:structures:group}{grupo}$(A^\times, \times)$.
\end{definition}

\begin{proof}[Prueba de que las unidades forman un grupo]
Estabilidad: si $a, a'$ son unidades con inversas $b, b'$, entonces
\[
(aa')(b'b) = a(a'b')b = a\,1\,b = ab = 1,
\qquad (b'b)(aa') = 1
\]
simétricamente, por lo que $aa'$ es una unidad. El elemento $1$ es una unidad (su
propia inversa), la asociatividad se hereda de $A$, y la
la inversa $b$ de una unidad $a$ es en sí misma una unidad (con la inversa $a$).
Entonces $(A^\times, \times)$ satisface todos los axiomas de \omterm{def:b1:structures:group}{grupo}. Cada \omterm{def:b1:structures:group}{grupo}
en este libro que no está construido a partir de \omterm{def:b1:counting:objects}{permutaciones} surge de esta manera:
$\Q^* = \Q^\times$,$\R^*$,$\C^*$, las unidades de $\Z/n\Z$ a continuación,
y posteriormente las matrices invertibles (\cref{ch:b1:matrices}).
\end{proof}

\begin{example}\label{ex:b1:structures:rings}
$\Z, \Q, \R, \C$ son conmutativos \omterm{def:b1:structures:ring}{anillos};$\Z^\times = \{1, -1\}$,
$\Q^\times = \Q^*$. Posteriormente: polinomio \omterm{def:b1:structures:ring}{anillos}$K[X]$
(\cref{ch:b1:poly}), matriz \omterm{def:b1:structures:ring}{anillos} (no conmutativa,
\cref{ch:b1:matrices}) y $\Z/n\Z$a continuación. En cada \omterm{def:b1:structures:ring}{anillo},$0 \times
a = 0$(de distributividad:$0a = (0+0)a = 0a + 0a $), y
$(-1)a = -a$.
\end{example}

\begin{example}[Idempotentes: nuevos fenómenos en nuevos anillos]\label{ex:b1:structures:idempotents}
En $\Z$, la ecuación $x^2 = x$, es decir $x(x - 1) = 0$, sólo tiene
las soluciones $0$ y $1$. En$\Z/6\Z$, probando todas las clases:
$\overline0^2 = \overline0$,$\overline1^2 = \overline1$,
$\overline3^2 = \overline9 = \overline3$ y $\overline4^2 =
\overline{16} = \overline4$ --- \emph{cuatro} idempotentes. los dos
los exóticos provienen de divisores cero: $\overline3\,(\overline3 -
\overline1) = \overline3 \times \overline2 = \overline6 =
\overline0$ sin factor cero. Tales cálculos calibran
los instintos: hechos familiares sobre las ecuaciones sobreviven en
\omterm{def:b1:structures:field}{dominios integrales} y \omterm{def:b1:structures:field}{campos}, pero un \omterm{def:b1:structures:ring}{anillo} general puede y lo hace
comportarse de manera diferente --- consulte también el booleano \omterm{def:b1:structures:ring}{anillos} de
\cref{exo:b1:structures:10}, donde el elemento \emph{cada} es
idempotente.
\end{example}

\begin{proposition}[Teorema del binomio en un anillo conmutativo]\label{prop:b1:structures:binomial}
Si $a, b$ son elementos de un \omterm{def:b1:structures:ring}{anillo} conmutativo (más generalmente, si
$ab = ba$), luego para $n \in \N$:
\[
(a+b)^n = \sum_{k=0}^n \binom nk a^k b^{n-k},
\qquad
a^n - b^n = (a - b) \sum_{k=0}^{n-1} a^k b^{\,n-1-k} .
\]
\end{proposition}

\begin{proof}
Las pruebas de \cref{thm:b1:counting:binomial} y de la geometría.
identidad utiliza solo asociatividad, conmutatividad de los dos elementos,
y distributividad --- se aplican palabra por palabra.
\end{proof}

\begin{example}[El teorema del binomio en un anillo desconocido]\label{ex:b1:structures:binomialring}
Dos pagos rápidos de la generalidad. En $\Z/p\Z$($p$\omterm{def:b1:arith:prime}{primo}), el
medio \omterm{def:b1:counting:objects}{coeficientes binomiales} desaparecer
(El primer paso de \cref{thm:b1:arith:fermat}), por lo que el teorema
colapsa al \emph{el sueño del estudiante de primer año}
\[

\]
una identidad genuina allí, por criminal que parezca $\R$.
Y en cualquier \omterm{def:b1:structures:ring}{anillo} conmutativo que contenga un elemento $\varepsilon$
con $\varepsilon^2 = 0$, el teorema se trunca:
$(a + \varepsilon)^n = a^n + n\,a^{n-1}\varepsilon$, todos superiores
términos que llevan un factor $\varepsilon^2 = 0$. el coeficiente
$n\,a^{n-1}$ de $\varepsilon$ es el derivado de $x^n$ ---
no es casualidad, y un primer indicio de que las derivadas son álgebra como

\end{example}

\begin{definition}[Dominio integral, campo]\label{def:b1:structures:field}
Un \omterm{def:b1:structures:ring}{anillo} $A \neq \{0\}$ conmutativo es un \emph{integral
dominio}\index{dominio integral} cuando no tiene divisores de cero: $ab = 0
\implies a = 0$ o $ b = 0$. Es un \emph{campo}\index{campo} cuando
todo elemento distinto de cero es invertible. Cada campo es una integral.
dominio ($ab = 0$ y $a \neq 0$ dan $b = a^{-1}ab = 0$).
\end{definition}

\begin{example}
$\Q$,$\R$,$\C$ son \omterm{def:b1:structures:field}{campos};$\Z$ es un \omterm{def:b1:structures:field}{dominio integral} pero no un

ac $ y $a \neq 0$ implican $b = c $.
\end{example}

\section{El anillo $\Z/n\Z$}

\begin{definition}\label{def:b1:structures:zn}
Reparar $n \in \N^*$. Las clases \omterm{def:b1:arith:congruence}{congruencia} mod $n$
(\cref{ex:b1:logic:congruence}) forma un \omterm{def:b1:logic:sets}{conjunto}
$\Z/n\Z$\index{Z/nZ@$\Z/n\Z$} de elementos $n$, escrito $\overline 0,
\overline 1, \dots, \overline{n-1}$. las operaciones
\[
\overline a + \overline b = \overline{a + b},
\qquad
\overline a \times \overline b = \overline{ab}
\]
están bien definidos --- las clases de los resultados no dependen del
representantes, precisamente porque \omterm{def:b1:arith:congruence}{congruencia} es compatible con $+$
y $\times $(\cref{def:b1:arith:congruence}) --- y haga de $\Z/n\Z$ un
conmutativo \omterm{def:b1:structures:ring}{anillo}.
\end{definition}

\begin{theorem}[Unidades de $\Z/n\Z$; los campos $\Z/p\Z$]\label{thm:b1:structures:zp}
\begin{enumerate}
 \item $\overline a$ es invertible en $\Z/n\Z$ si y sólo si
 $\gcd(a, n) = 1$.
 \item $\Z/n\Z$ es \omterm{def:b1:structures:field}{campo} si y sólo si $n$ es \omterm{def:b1:arith:prime}{primo}.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) se reescribe \cref{prop:b1:arith:invmod} con clases.

(2) Si $n = p$ es \omterm{def:b1:arith:prime}{primo}, cada $\overline a \neq \overline 0$ tiene $p
\nmid a $, entonces $\gcd(a, p) = 1$: invertible por (1) --- a \omterm{def:b1:structures:field}{campo}. Si$n =
ab $ con $1 < a, b < n $, entonces $\overline a\, \overline b = \overline
n = \overline 0$ con $\overline a, \overline b \neq \overline 0$:
cero divisores, por lo que ni siquiera un \omterm{def:b1:structures:field}{dominio integral}; y $n = 1$ da el
cero \omterm{def:b1:structures:ring}{anillo}, excluido.
\end{proof}

 $\tau$ on the right swaps the values in positions $i$ and
 $i + 1$; exactly one pair changes its inversion
 status.)}
 \item Deduzca, utilizando la pregunta 6, que para \emph{cualquier}
 transposición $\tau$,$\varepsilon(\sigma\tau) =
 -\varepsilon(\sigma)$; concluir que si $\sigma$ es un
 producto de las transposiciones $p$, entonces $\varepsilon(\sigma)
 = (-1)^p $--- en particular la paridad de $p $ depende solo
 en $\sigma$, no en la factorización elegida --- y que
 $\varepsilon \colon \mathfrak S_n \to \{\pm 1\}$ es un
 \omterm{def:b1:structures:group}{grupo}\omterm{def:b1:structures:morphism}{morfismo}.
 \item Mostrar que un ciclo $k$ tiene la firma $(-1)^{k-1}$, y
 que en general $\varepsilon(\sigma) = (-1)^{n -
 c(\sigma)}$, donde $c(\sigma)$ es el número de órbitas de
 $\sigma $(puntos fijos incluidos).
 \item El \emph{\omterm{pb:b1:structures:1}{grupo alterno}} es $\mathfrak A_n =
 \ker\varepsilon $. Justifique que es un \omterm{def:b1:structures:subgroup}{subgrupo} y demuestre
 $\abs{\mathfrak A_n} = \frac{n!}2$ para $n \geq 2$.
 \emph{(Fix a transposition $\tau_0$ and consider $\sigma
 \mapsto \sigma\tau_0$.)}
 \item Comprobación de coherencia en $\sigma = [4, 1, 5, 2, 3, 7, 8, 6]$:
 calcular $\varepsilon(\sigma)$ de tres maneras --- contando
 inversiones, del tipo de ciclo a través de la pregunta 10, y de
 su transposición cuenta en la pregunta 5.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- $\mathfrak A_n$ is generated by
$3$-cycles.}
\begin{enumerate}[resume]
 \item Sea $a, b, c, d$ distinto por pares. verificar los dos
 identidades
 \[
 (a\ b)(a\ c) = (a\ c\ b),
 \qquad
 (a\ b)(c\ d) = (a\ c\ b)(a\ c\ d) .
 \]
 \item Demuestre que para $n \geq 3$, cada elemento de $\mathfrak
 A_n $ es un producto de ciclos $3$. \emph{(An even
 \omterm{def:b1:counting:objects}{permutación} is a product of an even number of
 transpositions; absorb them two at a time.)}
 \item Escribe $(1\ 2)(3\ 4)$ y el ciclo $5$ $(1\ 2\ 3\ 4\ 5)$
 explícitamente como productos de ciclos $3$.
 \item Demuestre la fórmula de conjugación: por cada $\sigma \in
 \mathfrak S_n $,
 \[
 \sigma\,(a_1\ \dots\ a_k)\,\sigma^{-1}
 = \bigl(\sigma(a_1)\ \dots\ \sigma(a_k)\bigr) .
 \]
\end{enumerate}

\noindent\textbf{Parte IV --- The 8-puzzle.} Azulejos $1, \dots, 8$
deslícese en un cuadro $3 \times 3$ con una celda vacía; un \emph{mover}
desliza una ficha adyacente a la celda vacía dentro de ella. Numera las celdas
$1, \dots, 9$(fila por fila; la posición resuelta tiene el mosaico $i$ en
celda $i$ y la celda vacía en la celda $9$). Trate la celda vacía como
un noveno mosaico, por lo que una posición es \omterm{def:b1:counting:objects}{permutación} $\sigma \in \mathfrak
S_9$(el mosaico $\sigma(i)$ se encuentra en la celda $i $).
\begin{enumerate}[resume]
 \item Muestra que un movimiento reemplaza $\sigma$ por $\sigma \circ \tau $
 donde $\tau$ es la transposición de las dos celdas
 involucrado; deducir que cada movimiento invierte
 $\varepsilon(\sigma)$.
 \item Sea $d(\sigma)$ la distancia del taxi (filas más
 columnas) entre la celda actual de la celda vacía y su
 celular de casa $9$. Demuestre que cada movimiento cambia $d$ por $\pm1$,
 por lo que cada movimiento también invierte $(-1)^{d(\sigma)}$. Concluir que
 \[
 I(\sigma) = \varepsilon(\sigma)\cdot(-1)^{d(\sigma)}
 \]
 es \emph{invariante} debajo de cada movimiento.
 \item Demuestra la imposibilidad clásica del rompecabezas: la posición
 que intercambia los mosaicos $7$ y $8$ y deja todo lo demás
 (incluida la celda vacía) en su lugar no se puede alcanzar
 desde la posición resuelta.
 \item Admitimos lo contrario (su prueba es instructiva pero
 inducción prolongada): cada posición con $I = +1$ es
 accesible. Deduzca que exactamente la mitad de las posiciones $8!$
 con la celda vacía en casa se pueden resolver, es decir,\
 $\frac{8!}2 = 20\,160$.
 \item Deduzca de la pregunta 20 que el mosaico accesible
 Los arreglos con la celda vacía en casa forman exactamente el mismo
 \omterm{def:b1:structures:subgroup}{subgrupo}$\mathfrak A_8 \leq \mathfrak S_8$.
 \item Aplicaciones del invariante: ¿se puede alcanzar (a) la
 posición donde los mosaicos $1, 2, 3$ se permutan cíclicamente
 y todo lo demás, incluida la celda vacía, ¿está en casa?
 (b) la posición donde el mosaico $5$ y la celda vacía tienen
 ¿Los lugares intercambiados y todos los demás mosaicos están en casa? justificar
 ambas respuestas con $I$.
\end{enumerate}
