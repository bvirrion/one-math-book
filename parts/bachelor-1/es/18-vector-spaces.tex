\section{Definición y ejemplos}

\begin{definition}[Espacio vectorial]\label{def:b1:vspaces:def}
Un \emph{$K$-espacio vectorial}\index{espacio vectorial} es un \omterm{def:b1:logic:sets}{conjunto}$E$ con un
suma haciendo $(E, +)$ un \omterm{def:b1:structures:group}{grupo abeliano} (cero escrito $0_E$ o
$0$), y una multiplicación escalar $K \times E \to E$ tal que, para
todos $\lambda, \mu \in K$ y $x, y \in E$:
\[
\lambda(x + y) = \lambda x + \lambda y,\quad
(\lambda + \mu) x = \lambda x + \mu x,\quad
\lambda(\mu x) = (\lambda\mu) x,\quad
1\,x = x .
\]
Consecuencias: $0\,x = 0_E$,$\lambda\,0_E = 0_E$,$(-1)x = -x$ y
$\lambda x = 0_E \implies \lambda = 0$ o $x = 0_E$(multiplicar por
$\lambda^{-1}$).
\end{definition}

\begin{proof}[Prueba de las consecuencias]
Para $0\,x = 0_E$: de $(0 + 0)x = 0x + 0x$ y $(0+0)x = 0x$,
cancelar $0x$ en el \omterm{def:b1:structures:group}{grupo}$(E, +)$. Para $\lambda\,0_E$: igual
truco en $\lambda(0_E + 0_E)$. Para $(-1)x$: agregue $x$,
\[
x + (-1)x = 1\,x + (-1)x = \bigl(1 + (-1)\bigr)x = 0\,x = 0_E ,
\]
entonces $(-1)x$ es el inverso aditivo de $x$. Finalmente si $\lambda x
= 0_E $ con $\lambda \neq 0$: multiplicar por $\lambda^{-1}$(el
los escalares forman un \omterm{def:b1:structures:field}{campo}) y usan los dos axiomas
$\lambda^{-1}(\lambda x) = (\lambda^{-1}\lambda)x = 1x = x$
junto con $\lambda^{-1}0_E = 0_E$:$x = 0_E$. Pequeños como ellos
son, estas cuatro reglas se utilizan silenciosamente en cada página que
sigue --- y el último es exactamente donde se necesitan \omterm{def:b1:structures:field}{campos}:
sobre los escalares $\Z$, el ``espacio''$\Z/2\Z$ lo violaría
con $2\,x = 0$.
\end{proof}

\begin{example}[Subespacio o no: cuatro sentencias, argumentadas]
\label{ex:b1:vspaces:subspaceornot}

\section{Span, sumas, sumas directas}

\begin{definition}[Combinaciones lineales, subespacio engendrado]\label{def:b1:vspaces:span}
Una \emph{combinación lineal} de la familia $(x_1, \dots, x_p)$ de
vectores de $E$ es cualquier $\lambda_1 x_1 + \dots + \lambda_p x_p$
($\lambda_i \in K$). El \omterm{def:b1:logic:sets}{conjunto} de todas ellas es el
\emph{subespacio engendrado}\index{subespacio engendrado}
$\operatorname{Vect}(x_1, \dots, x_p)$: es un
\omterm{def:b1:vspaces:subspace}{subespacio}, el más pequeño que contiene la
familia.
\end{definition}

\begin{proof}[Prueba de las dos afirmaciones]
Estabilidad: la suma de dos combinaciones lineales
\[
\sum\lambda_i x_i + \sum\mu_i x_i
= \sum(\lambda_i + \mu_i)x_i
\]
vuelve a serlo, e igual un múltiplo escalar
$\mu\sum\lambda_i x_i = \sum(\mu\lambda_i)x_i$; la combinación nula
muestra que $0$ pertenece: el
\omterm{def:b1:vspaces:span}{engendrado} es un
\omterm{def:b1:vspaces:subspace}{subespacio}.
Minimalidad: sea $H$ cualquier
\omterm{def:b1:vspaces:subspace}{subespacio} que contenga
$x_1,\dots,x_p$. Por estabilidad bajo la multiplicación escalar,
cada $\lambda_i x_i \in H$, y por estabilidad bajo la suma su suma
está en $H$: toda combinación lineal pertenece a $H$, es decir
$\operatorname{Vect}(x_1,\dots,x_p)\subseteq H$. Así el
\omterm{def:b1:vspaces:span}{engendrado} está contenido en todo
\omterm{def:b1:vspaces:subspace}{subespacio} que contenga la familia:
es el más pequeño.
\end{proof}

\begin{definition}[suma, suma directa]\label{def:b1:vspaces:sum}
Para \omterm{def:b1:vspaces:subspace}{subespacios} $F, G$ de $E$:
\[
F + G = \{\,u + v : u \in F,\ v \in G\,\}
\]
es un \omterm{def:b1:vspaces:subspace}{subespacio} (el más pequeño contiene $F \cup G$). la suma es
\emph{directo}\index{suma directa}, escrito $F \oplus G$, cuando cada
el elemento de $F + G$ descompone \emph{únicamente} como $u + v$;
de manera equivalente (ver más abajo) cuando $F \cap G = \{0\}$. Cuando $E = F \oplus
G $, \omterm{def:b1:vspaces:subspace}{subespacios} son \emph{suplementario}\index{suplementario
subespacios} en $E$.
\end{definition}

\begin{example}[Una suma de dos lineas]
\label{ex:b1:vspaces:sumoflines}
En $\R^3$, dejemos que $F = \operatorname{Vect}\bigl((1,0,1)\bigr)$ y
$G = \operatorname{Vect}\bigl((0,1,1)\bigr)$. Su suma es
\[
F + G = \{\,a(1,0,1) + b(0,1,1)\,\}
= \{(a,\ b,\ a + b)\} = \{(x, y, z) : z = x + y\},
\]
el plano que pasa por el origen que contiene ambas rectas. es
estrictamente mayor que la unión $F \cup G$(la mera cruz de la
dos líneas): el vector $(1, 1, 2) = (1,0,1) + (0,1,1)$ se encuentra en
la suma pero en ninguna de las líneas. Y $F \cap G = \{0\}$(un común
vector requiere $a(1,0,1) = b(0,1,1)$, cuyos primeros dos
\omterm{prop:b1:vspaces:coordinates}{coordenadas} fuerza $a = b = 0$): la suma es directa, y $F \oplus
G $ es exactamente ese plano.
\end{example}

\begin{proposition}\label{prop:b1:vspaces:directsum}
$F + G$ es directo si y sólo si $F \cap G = \{0\}$.
\end{proposition}

\begin{proof}
Si algunos $w \neq 0$ se encuentran en $F \cap G$:$w = w + 0 = 0 + w$ son dos
descomposiciones de $w$. Por el contrario, si $u + v = u' + v'$ con $u, u'
\in F $,$v, v' \in G$, entonces $u - u' = v' - v$ pertenece a $F \cap G =
\{0\}$: las descomposiciones son únicas.
\end{proof}

\begin{method}[Probando $E = F \oplus G$]
\label{met:b1:vspaces:directsum}
Dos cosas para comprobar, cada una con su movimiento de apertura estándar.
\begin{enumerate}
 \item \emph{Intersección trivial.} Toma $x \in F \cap G$,
 escriba ambas condiciones de membresía y presione $x =
 0$. (Nunca discutas por imagen: cf.\ los peligros que aparecen a continuación.)
 \item \emph{La suma lo es todo.} Tome un $x \in E$ arbitrario
 y \emph{producir} la descomposición $x = f + g$ ---
 ya sea adivinando $f$ del objetivo ($f$ debe satisfacer
 la propiedad definitoria de $F$, que generalmente dicta su
 fórmula) o resolviendo el sistema lineal que expresa $x$
 contra generadores de $F$ y $G$.
\end{enumerate}
Cuando se adivina la fórmula de descomposición, la unicidad es
automático desde el paso 1; cuando sólo la existencia no está clara, el paso 2 es
donde vive la obra. Los dos ejemplos siguientes ejecutan el método: para
funciones pares/impares la fórmula para $f$ se fuerza evaluando
la identidad deseada en $x$ y $-x$; para \omterm{def:b1:poly:def}{polinomios} desapareciendo
en un punto, evaluando en $a$.
\end{method}

\section{Familias libres, familias generadoras, bases.}

\begin{definition}\label{def:b1:vspaces:free}
Una familia $(x_1, \dots, x_p)$ de vectores de $E$ es:
\begin{itemize}
 \item \emph{generando} (de $E$) cuando
 $\operatorname{Vect}(x_1,\dots,x_p) = E$;
 \item \emph{libre}\index{familia libre} (sus vectores \emph{linealmente
 independiente}) cuando
 \[
 \lambda_1 x_1 + \dots + \lambda_p x_p = 0
 \implies \lambda_1 = \dots = \lambda_p = 0 ;
 \]
 de lo contrario \emph{vinculado};
 \item a \emph{base}\index{base} cuando esté libre y generando.
\end{itemize}
\end{definition}

\begin{proposition}[coordinadas]\label{prop:b1:vspaces:coordinates}
$(e_1, \dots, e_n)$ es un \omterm{def:b1:vspaces:free}{base} de $E$ si y sólo si cada $x \in E$
es \emph{únicamente} una combinación $x = \lambda_1 e_1 + \dots +
\lambda_n e_n $; los escalares $\lambda_i$ son los
\emph{coordenadas}\index{coordenadas} de $x$ en el \omterm{def:b1:vspaces:free}{base}.
\end{proposition}

\begin{proof}
Generando $=$existencia de la descomposición. Singularidad$=$
libertad: dos descomposiciones del mismo $x$ se diferencian por una combinación
igual a $0$; la libertad fuerza todos sus coeficientes --- el
diferencias de \omterm{prop:b1:vspaces:coordinates}{coordenadas} --- desaparecer. Por el contrario, una cuestión no trivial
La combinación nula da las dos descomposiciones $0 = \sum \lambda_i
e_i = \sum 0\,e_i $.
\end{proof}

\begin{example}\label{ex:b1:vspaces:bases}
El \emph{canonical \omterm{def:b1:vspaces:free}{base}} de $K^n$:$e_i = (0, \dots, 1, \dots, 0)$
($1$ en la ranura $i$). Los monomios $(1, X, X^2, \dots, X^n)$: a \omterm{def:b1:vspaces:free}{base}
de $K_n[X]$(libre: una combinación nula es el cero \omterm{def:b1:poly:def}{polinomio}, por lo que
todos los coeficientes desaparecen, \cref{def:b1:poly:def}). en
$\C$ sobre $\R$: el \omterm{def:b1:vspaces:free}{base}$(1, \iu)$.
\end{example}

K_n[X]$,$Q \neq 0$, de grado $d $, con coeficiente principal $a $,
y sea $b \neq 0$ el coeficiente principal de $P_d$. Entonces $Q -
\frac ab P_d $ tiene el grado$< d $(los términos superiores se cancelan); reemplazando
$Q$ por esta diferencia e iterando, después de como máximo los pasos $n + 1$
se llega al cero \omterm{def:b1:poly:def}{polinomio}, y desenrollando las restas
expresa $Q$ como una combinación de $P_k$. Dos escaleras
ya se cumplen: las potencias desplazadas $\bigl((X-a)^k\bigr)_{0 \leq k
\leq n}$(\cref{exo:b1:vspaces:4}) y los productos de Newton
$\bigl((X - x_0)(X - x_1)\cdots(X - x_{k-1})\bigr)_{0 \leq k \leq
n}$, puesto a trabajar en el problema del fin de semana.
\end{example}

\begin{example}[Libertad en espacios funcionales]\label{ex:b1:vspaces:functionfree}
En $\mathcal{F}(\R,\R)$, la familia $(\eu^{a_1 x}, \dots, \eu^{a_p
x})$ con $ a_1 < \dots < a_p $ es \omterm{def:b1:vspaces:free}{libre}: dividir una combinación nula por
$\eu^{a_p x}$ y deja que $x \to +\infty $; el último coeficiente muere,
y uno cae en cascada (\cref{exo:b1:vspaces:8} detalla esto y
variantes). La libertad de funciones queda demostrada por \emph{evaluando}:
en puntos bien elegidos, en el infinito o después de diferenciar.
\end{example}

\begin{example}[Una relación oculta se reduce un lapso]
\label{ex:b1:vspaces:hiddenrelation}
En $\mathcal{F}(\R, \R)$, ¿qué es?
$\operatorname{Vect}(1,\ \cos^2,\ \sin^2)$? la identidad
$\cos^2 + \sin^2 = 1$ es una combinación nula no trivial
\[
1\cdot\mathbf{1} + (-1)\cos^2 + (-1)\sin^2 = 0 :
\]
la familia está vinculada y el \omterm{def:b1:vspaces:span}{engendrado} ya está generado por
$(1, \cos^2)$ solo ($\sin^2 = 1 - \cos^2$). Esa familia más pequeña
es \omterm{def:b1:vspaces:free}{libre}: $a + b\cos^2 x = 0$ para todos los $x$, en $x = 0$ y
$x = \frac\pi2$:$a + b = 0$ y $a = 0$. Entonces el \omterm{def:b1:vspaces:span}{engendrado} es un
\emph{avión} dentro del espacio funcional --- y también contiene
$\cos 2x = 2\cos^2 x - 1$: familias de aspecto lineal de
las funciones trigonométricas colapsan habitualmente bajo identidades,
por lo que la libertad debe ser \emph{probado}, nunca asumida desde
la longitud de la lista.
\end{example}

\begin{remark}[Errores comunes]
\label{rem:b1:vspaces:pitfalls}
Cuatro trampas clásicas.
\emph{Por parejas no es suficiente}: en $\R^2$, los vectores $(1,0)$,
$(0,1)$,$(1,1)$ son pares no proporcionales, pero están vinculados ---
La libertad es una propiedad de la familia \emph{entero}, probada por uno.
combinación global, nunca de dos en dos.
\emph{El vector cero lo envenena todo}: cualquier familia que contenga
$0$ está vinculado ($1\cdot 0 = 0$ es una relación no trivial), sin embargo
inocentes los otros vectores.
\emph{La unión no es suma.}: $F \cup G$ casi nunca es un \omterm{def:b1:vspaces:subspace}{subespacio}
(\cref{def:b1:vspaces:subspace}); el más pequeño \omterm{def:b1:vspaces:subspace}{subespacio}
que contiene ambos es $F + G$, generalmente mucho más grande que la unión
--- en $\R^2$, dos líneas distintas tienen unión en cruz, suma la
avión entero.
\emph{Directo requiere intersección trivial, no desunión}:
dos \omterm{def:b1:vspaces:subspace}{subespacios} nunca son separados (ambos contienen $0$); lo correcto
la condición es $F \cap G = \{0\}$ y debe ser \emph{probado},
no leer un dibujo --- cf.\
\cref{ex:b1:vspaces:manysupplements}, donde muchos $G$ diferentes
trabajo.
\emph{La libertad depende de los escalares.}: el par $(1, \iu)$ es
\omterm{def:b1:vspaces:free}{libre} en $\C$ visto como un espacio vectorial $\R$, pero vinculado en $\C$
visto como un espacio vectorial $\C$($\iu\cdot 1 + (-1)\cdot\iu = 0$).
Sepa siempre qué \omterm{def:b1:structures:field}{campo} está actuando antes de declarar una familia \omterm{def:b1:vspaces:free}{libre}
--- el problema del fin de semana de \cref{ch:b1:findim} gira exactamente
esta sensibilidad en pruebas de irracionalidad.
\end{remark}

\begin{remark}[Adónde va este idioma]
\label{rem:b1:vspaces:whereused}
Todo lo que sigue a este capítulo habla el idioma \omterm{def:b1:logic:sets}{conjunto} aquí.
\cref{ch:b1:findim} cuenta los vectores \omterm{def:b1:vspaces:free}{base} y convierte ``\omterm{def:b1:vspaces:free}{libre}'' y
"generando" desigualdades en un solo número entero, la
dimensión. \cref{ch:b1:linmaps} estudia el \omterm{def:b1:logic:map}{aplicaciones} compatible con
las dos operaciones; \omterm{def:b1:vspaces:sum}{sumas directas} se convierten en proyectores allí.
\cref{ch:b1:matrices} codifica vectores por su \omterm{prop:b1:vspaces:coordinates}{coordenadas} en un
\omterm{def:b1:vspaces:free}{base} --- \cref{prop:b1:vspaces:coordinates} es la licencia para
esa codificación --- y \cref{ch:b1:euclid} agrega longitudes y ángulos
encima de la estructura lineal. En el volumen Year~2 lo mismo.
axiomas, palabra por palabra, repasados arbitrarios \omterm{def:b1:structures:field}{campos} y en infinito
dimensión; nada en este capítulo utilizó la finitud en ninguna parte.
\end{remark}

\section{Problema: interpolación, tres bases para un espacio.}

\begin{problem}\label{pb:b1:vspaces:1}
Reparar $n + 1$\emph{distinto} puntos $x_0, x_1, \dots, x_n$ de $\R$.
Este problema vuelve a visitar \omterm{thm:b1:poly:lagrange}{interpolación de Lagrange}
(\cref{thm:b1:poly:lagrange}) a los ojos de este capítulo: el
El espacio $\R_n[X]$ lleva tres bases naturales: Lagrange,
Newton y (para puntos equidistantes) el binomio \omterm{def:b1:vspaces:free}{base} ---
y cada \omterm{def:b1:vspaces:free}{base} facilita una pregunta. El camino termina en un auténtico
teorema aritmético: caracterización de Pólya del \omterm{def:b1:poly:def}{polinomios}
mapeo de $\Z$ en $\Z$.\index{interpolación de Lagrange}
\index{diferencias divididas}\index{polinomio de valores enteros}

\medskip
\noindent\textbf{Parte I --- The Lagrange \omterm{def:b1:vspaces:free}{base}.} Para conjunto $0 \leq i
\leq n $\[
L_i \;=\; \prod_{j \neq i} \frac{X - x_j}{x_i - x_j} \;\in\;
\R_n[X].
\]
\begin{enumerate}
 \item Compruebe que $\deg L_i = n$ y $L_i(x_j) = 1$ si $j =
 i $, y $0$ si $j \neq i $.
 \item Demuestre que la familia $(L_0, \dots, L_n)$ es \omterm{def:b1:vspaces:free}{libre}.
 \item Demuestre que por cada $P \in \R_n[X]$,
 \[
 P \;=\; \sum_{i=0}^{n} P(x_i)\, L_i ,
 \]
 y concluir que $(L_0, \dots, L_n)$ es un \omterm{def:b1:vspaces:free}{base} de
 $\R_n[X]$. \emph{(Consider the difference of the two sides
 and count its roots, \cref{cor:b1:poly:nroots}.)}
 \item Deduzca el teorema de interpolación: para cualquier valor $y_0,
 \dots, y_n \in \R $existe un \emph{único}$P \in
 \R_n[X]$ con $P(x_i) = y_i $ para todo $i $. En Lagrange
 \omterm{def:b1:vspaces:free}{base}, ¿cuáles son los \omterm{prop:b1:vspaces:coordinates}{coordenadas} de un \omterm{def:b1:poly:def}{polinomio} $P$?
 \item Probar las identidades
 \[
 \sum_{i=0}^{n} L_i = 1
 \qquad\text{and, for } 0 \leq k \leq n,\qquad
 \sum_{i=0}^{n} x_i^{k}\, L_i = X^{k} .
 \]
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The Newton \omterm{def:b1:vspaces:free}{base} and \omterm{pb:b1:vspaces:1}{dividido
diferencias}.} Configure $N_0 = 1$ y $N_k = (X - x_0)(X - x_1) \cdots
(X - x_{k-1})$ para $1 \leq k \leq n $. Para una función $f $ definida
en los nodos, defina el \emph{\omterm{pb:b1:vspaces:1}{diferencias divididas}} por
$f[x_i] = f(x_i)$ y
\[
f[x_i, \dots, x_{i+k}] \;=\;
\frac{f[x_{i+1}, \dots, x_{i+k}] - f[x_i, \dots, x_{i+k-1}]}
{x_{i+k} - x_i} .
\]
\begin{enumerate}[resume]
 \item Demuestre que $(N_0, N_1, \dots, N_n)$ es un \omterm{def:b1:vspaces:free}{base} de
 $\R_n[X]$.
 \item Calcule $f[x_0, x_1]$ y $f[x_0, x_1, x_2]$ en términos de
 los valores de $f$, luego calcule todos los \omterm{pb:b1:vspaces:1}{diferencias divididas} de
 $f(x) = x^2$ en tres nodos arbitrarios.
 \item (lema de Aitken) Dejemos que $R$ interpola $f$ en $x_0, \dots,
 x_{n-1}$ y $ Q$ interpola $f$ en $x_1, \dots, x_n$,
 ambos de grado $\leq n - 1$. demostrar que
 \[
 S \;=\; \frac{(X - x_0)\,Q - (X - x_n)\,R}{x_n - x_0}
 \]
 interpola $f$ en $x_0, x_1, \dots, x_n$.
 \item Deducir, por inducción sobre el número de nodos, que el
 El coeficiente de $X^{k}$ en el interpolante de $f$ en $x_0,
 \dots, x_k $ es exactamente $f[x_0, \dots, x_k]$.
 \item Demuestre \emph{Fórmula de interpolación de Newton}: el
 interpolante de $f$ en $x_0, \dots, x_n$ es
 \[
 P \;=\; \sum_{k=0}^{n} f[x_0, \dots, x_k]\, N_k ,
 \]
 y derivar la fórmula cerrada
 \[
 f[x_0, \dots, x_k] \;=\; \sum_{i=0}^{k}
 \frac{f(x_i)}{\prod_{j \neq i,\, j \leq k} (x_i - x_j)} ,
 \]
 lo que muestra que $f[x_0, \dots, x_k]$ no depende de
 el orden de los nodos.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Equally spaced nodes: the difference
operator.} De ahora en adelante los nodos son $0, 1, 2, \dots$ y, por un
\omterm{def:b1:poly:def}{polinomio} $P$, configuramos
\[
\Delta P(X) = P(X + 1) - P(X),
\qquad
B_k = \frac{X(X-1)\cdots(X-k+1)}{k!} \quad (B_0 = 1).
\]
\begin{enumerate}[resume]
 \item Demuestre que si $\deg P = m \geq 1$ tiene coeficiente principal
 $a$, luego $\deg \Delta P = m - 1$ con coeficiente principal
 $m\,a$, y que $\Delta$ mata constantes.
 \item Demuestre que $(B_0, B_1, \dots, B_n)$ es un \omterm{def:b1:vspaces:free}{base} de
 $\R_n[X]$ y ese $\Delta B_k = B_{k-1}$ para $k \geq 1$.
 \item (fórmula de diferencia directa de Newton) Demuestre que cada $P
 \in \R_n[X]$ satisface
 \[
 P \;=\; \sum_{k=0}^{n} \bigl(\Delta^{k} P\bigr)(0)\, B_k .
 \]
 \item Demuestre que por cada $k \geq 0$,
 \[
 \bigl(\Delta^{k} P\bigr)(0) \;=\;
 \sum_{j=0}^{k} (-1)^{k-j} \binom{k}{j} P(j) .
 \]
 \item Demuestre que si $\deg P = n$ tiene el coeficiente principal $a_n$,
 entonces $\Delta^{n} P$ es la constante $n!\,a_n$ y
 $\Delta^{n+1} P = 0$.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- \omterm{pb:b1:vspaces:1}{Polinomios con valores enteros}.}A
\omterm{def:b1:poly:def}{polinomio} $P \in \R[X]$ es \emph{de valor entero} cuando $P(m) \in
\Z $ por cada $m \in \Z $.
\begin{enumerate}[resume]
 \item Demuestre que cada $B_k$ tiene un valor entero. \emph{(Treat $m
 \geq k $,$0 \leq m < k $and $m < 0$ separately; for $m =
 -q < 0$, show $B_k(-q) = (-1)^k \binom{q + k - 1}{k}$.)}
 \item Demostrar \emph{Pólya's characterization}: $P \in \R_n[X]$
 tiene un valor entero si y sólo si es \omterm{prop:b1:vspaces:coordinates}{coordenadas} en el
 \omterm{def:b1:vspaces:free}{base} $(B_0, \dots, B_n)$ son números enteros.
 \item Deducir: si $P \in \R_n[X]$toma valores enteros en $n + 1$\emph{consecutivo} enteros $a, a+1, \dots, a+n$, luego $P$
 tiene un valor entero. \emph{(Shift: aplicar el estudio a $Q(X)
 = P(X + a)$.)}
 \item Deduzca de la pregunta 16 que un producto de $k$ consecutivo
 Los números enteros siempre son divisibles por $k!$.
 \item Vamos $P = \dfrac{X(X+1)(2X+1)}{6}$. Calcular su Newton
 tabla en $0, 1, 2, 3$, escriba $P$ en \omterm{def:b1:vspaces:free}{base}$(B_k)$, y
 concluir que $P$ tiene un valor entero aunque ninguno de sus
 los coeficientes monomios son un número entero. Verifique $\Delta P =
 (X+1)^2$ y deduzca $P(m) = 1^2 + 2^2 + \dots + m^2$ para
 $m \in \N$.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends.}
\begin{enumerate}[resume]
 \item Dejemos que $P \in \R_n[X]$interpole los valores $2^i$ en $i =
 0, 1, \dots, n $. Muestra que $P = B_0 + B_1 + \dots + B_n $
 y que $P(n + 1) = 2^{n+1} - 1$: la ``duplicación
 El patrón'' siempre se rompe en el siguiente punto.
 \item (Antiderivada discreta) Demuestre que para todos los números enteros $m
 \geq 1$ y $ k \geq 0$,
 \[
 \sum_{j=0}^{m-1} B_k(j) \;=\; B_{k+1}(m),
 \]
 es decir,\ la identidad del palo de hockey $\sum_{j=k}^{m-1}
 \binom{j}{k} = \binom{m}{k+1}$.
 \item Expanda $X^2$ y $X^3$ en \omterm{def:b1:vspaces:free}{base}$(B_k)$ y deduzca
 fórmulas cerradas para $\sum_{j=0}^{m-1} j^2$ y
 $\sum_{j=0}^{m-1} j^3$; recuperar la identidad de Nicómaco
 $1^3 + \dots + m^3 = (1 + \dots + m)^2$.
 \item Tome $n = 2$ y los nodos $0, 1, 2$. Escribe el \omterm{prop:b1:vspaces:coordinates}{coordenadas} de
 $X^2$ en las tres bases de este problema: el monomio
 \omterm{def:b1:vspaces:free}{base}, el Lagrange \omterm{def:b1:vspaces:free}{base}, el Newton \omterm{def:b1:vspaces:free}{base}. Compruebe el
