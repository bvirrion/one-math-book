\chapter{Aplicaciones lineales}\label{ch:b1:linmaps}

Los \omterm{def:b1:logic:map}{aplicaciones} que vale la pena estudiar entre \omterm{def:b1:vspaces:def}{espacios vectoriales} son los compatibles
con la estructura: \emph{\omterm{def:b1:linmaps:def}{aplicaciones lineales}}. Sus dos fundamentales
\omterm{def:b1:vspaces:subspace}{subespacios} --- \omterm{def:b1:linmaps:kerim}{núcleo} y \omterm{def:b1:linmaps:kerim}{imagen} --- miden la inyectividad y
sobreyectividad, y en \omterm{def:b1:findim:def}{dimensión finita} el teorema de nulidad de rango
une sus tamaños en una ley de conservación. \omterm{def:b1:linmaps:projection}{Proyecciones} y
simetrías, luego \omterm{def:b1:linmaps:forms}{formas lineales} y \omterm{def:b1:linmaps:forms}{hiperplanos}, cierran el capítulo.

\section{Definiciones y primeras propiedades.}

\begin{definition}\label{def:b1:linmaps:def}
Sea $E, F$ espacios vectoriales $K$. Una \omterm{def:b1:logic:map}{aplicación}$u \colon E \to F$ es
\emph{lineal}\index{aplicación lineal} cuando
\[
\forall x, y \in E,\ \forall \lambda \in K, \qquad
u(x + \lambda y) = u(x) + \lambda u(y).
\]
Luego $u(0) = 0$ y $u(\sum \lambda_i x_i) = \sum \lambda_i u(x_i)$.
El \omterm{def:b1:logic:sets}{conjunto} $\mathcal{L}(E, F)$ del \omterm{def:b1:logic:map}{aplicaciones} lineal es en sí mismo un \omterm{def:b1:vspaces:def}{espacio vectorial};
La composición del lineal \omterm{def:b1:logic:map}{aplicaciones} es lineal y bilineal en el par. un
\emph{endomorfismo} es un $E \to E$ lineal; un \emph{isomorfismo} es
un \omterm{def:b1:logic:inj}{biyectivo} lineal \omterm{def:b1:logic:map}{aplicación} (su inversa es entonces automáticamente lineal);
$u^{-1}$ de un isomorfismo, composiciones de isomorfismos, son
isomorfismos.
\end{definition}

\begin{proof}[Demostración de que la inversa es lineal]
Sea $u$\omterm{def:b1:linmaps:def}{lineal} \omterm{def:b1:logic:inj}{biyectivo},$ y, y' \in F$ y $\lambda \in K$.
Ponga $x = u^{-1}(y)$ y $x' = u^{-1}(y')$. entonces
\[
u\bigl(x + \lambda x'\bigr) = u(x) + \lambda u(x')
= y + \lambda y' ,
\]
y aplicando $u^{-1}$ en ambos extremos:$u^{-1}(y + \lambda y') = x
+ \lambda x' = u^{-1}(y) + \lambda\,u^{-1}(y')$. Nada sobre
Se calculó $u^{-1}$: la linealidad se transporta a través del
propiedad definitoria de $u$ sola --- un patrón que vale la pena recordar,
ya que la estructura a menudo viaja a lo largo de biyecciones de forma gratuita.
\end{proof}

\begin{proposition}[Una aplicación lineal queda determinada por la base]\label{prop:b1:linmaps:basis}
Sea $(e_1, \dots, e_n)$ un \omterm{def:b1:vspaces:free}{base} de $E$ y $(v_1, \dots, v_n)$
vectores arbitrarios de $F$. Hay exactamente una \omterm{def:b1:linmaps:def}{aplicación lineal}$u \colon
E \to F $ con $u(e_i) = v_i $ para todos los $i $. Además:
\[
u \text{ injective} \iff (v_i) \text{ free};
\qquad
u \text{ surjective} \iff (v_i) \text{ generates } F .
\]
\end{proposition}

\begin{proof}
Existencia/singularidad: cada $x$ tiene \omterm{prop:b1:vspaces:coordinates}{coordenadas}$x = \sum
\lambda_i e_i $(\cref{prop:b1:vspaces:coordinates}); linealidad
fuerza $u(x) = \sum \lambda_i v_i$, y esta fórmula define un
\omterm{def:b1:linmaps:def}{aplicación lineal}.

Inyectividad: por \cref{prop:b1:linmaps:kernel} más abajo, $u$ es
\omterm{def:b1:logic:inj}{inyectiva} si y solo si su
\omterm{def:b1:linmaps:kerim}{núcleo} es trivial. Ahora bien,
$u\bigl(\sum\lambda_i e_i\bigr) = 0$ significa exactamente
$\sum\lambda_i v_i = 0$. Si $(v_i)$ es
\omterm{def:b1:vspaces:free}{libre}, esto fuerza cada $\lambda_i = 0$,
es decir el \omterm{def:b1:linmaps:kerim}{núcleo} se reduce a $\{0\}$:
\omterm{def:b1:logic:inj}{inyectiva}. Si $(v_i)$ es ligada, una
relación no trivial $\sum\lambda_i v_i = 0$ produce el vector no nulo
$\sum\lambda_i e_i$ en el \omterm{def:b1:linmaps:kerim}{núcleo}
($(e_i)$ es \omterm{def:b1:vspaces:free}{libre}): no es
\omterm{def:b1:logic:inj}{inyectiva}. Las dos condiciones coinciden
término a término.

Sobreyectividad: la \omterm{def:b1:linmaps:kerim}{imagen} de $u$ es el
\omterm{def:b1:logic:sets}{conjunto} de todas las $\sum\lambda_i v_i$,
es decir exactamente $\operatorname{Vect}(v_1, \dots, v_n)$, que
coincide con $F$ si y solo si la familia genera.
\end{proof}

\begin{definition}[Kernel e imagen]\label{def:b1:linmaps:kerim}
Para $u \in \mathcal{L}(E, F)$:
\[
\ker u = \{x \in E : u(x) = 0\} \subseteq E,
\qquad
\operatorname{im} u = u(E) \subseteq F ,
\]
tanto \omterm{def:b1:vspaces:subspace}{subespacios} (verificación directa con el
criterio).\index{núcleo}\index{imagen}
\end{definition}

\begin{method}[Kernel e imagen, en la práctica]
\label{met:b1:linmaps:kerim}
\emph{\omterm{def:b1:linmaps:kerim}{Núcleo}}: escribe $u(x) = 0$ como sistema en el \omterm{prop:b1:vspaces:coordinates}{coordenadas}
(o coeficientes) de $x$, resolver y parametrizar --- el \omterm{def:b1:linmaps:kerim}{núcleo}
sale con un \omterm{def:b1:vspaces:free}{base} adjunto
(\cref{met:b1:findim:computedim}). \emph{\omterm{def:b1:linmaps:kerim}{Imagen}}: es el \omterm{def:b1:vspaces:span}{engendrado}
del \omterm{def:b1:linmaps:kerim}{imágenes} de \emph{cualquier} familia generadora de $E$ ---
normalmente un \omterm{def:b1:vspaces:free}{base}, por lo que $\operatorname{im} u =
\operatorname{Vect}\bigl(u(e_1), \dots, u(e_n)\bigr)$; entonces
elimine \omterm{def:b1:linmaps:kerim}{imágenes} redundante para extraer un \omterm{def:b1:vspaces:free}{base}. \emph{Atajo}:
Calcula cuál de los dos es más fácil y obtén el del otro.
dimensión libre por rango - nulidad
(\cref{thm:b1:linmaps:ranknullity}); cuando un candidato plausible
Se conoce \omterm{def:b1:vspaces:subspace}{subespacio} para \omterm{def:b1:linmaps:kerim}{imagen}, comparando las actualizaciones de dimensiones
la fácil inclusión a la igualdad
(\cref{thm:b1:findim:subspaces}). Ambos atajos se utilizan en
\cref{ex:b1:linmaps:delta} a continuación.
\end{method}

\begin{proposition}\label{prop:b1:linmaps:kernel}
$u$ is \omterm{def:b1:logic:inj}{inyectivo}$\iff$ $\ker u = \{0\}$;$u$ is \omterm{def:b1:logic:inj}{sobreyectivo}$\iff$
$\operatorname{im} u = F$.
\end{proposition}

\begin{proof}
En cuanto a \omterm{def:b1:structures:group}{grupos} (\cref{prop:b1:structures:kernel}): $u(x) = u(y) \iff
u(x - y) = 0 \iff x - y \in \ker u $. El segundo punto es el
definición.
\end{proof}

\section{El teorema de rango-nulidad}

\begin{definition}\label{def:b1:linmaps:rank}
El \emph{rango}\index{rango!de una aplicación lineal} de $u \in \mathcal{L}(E,
F)$(con $E $\omterm{def:b1:findim:def}{de dimensión finita}) es $\operatorname{rk} u = \dim
\operatorname{im} u $ --- también el rango de la familia
$\bigl(u(e_1), \dots, u(e_n)\bigr)$ para cualquier \omterm{def:b1:vspaces:free}{base}$(e_i)$ de $E$.
\end{definition}

\begin{theorem}[rango--nulidad]\label{thm:b1:linmaps:ranknullity}
Sea $E$\omterm{def:b1:findim:def}{de dimensión finita} y $u \in \mathcal{L}(E, F)$.
Entonces\index{rango - teorema de nulidad}
\[
\dim E = \dim \ker u + \operatorname{rk} u .
\]
Más precisamente, si $S$ es cualquier \omterm{def:b1:vspaces:sum}{suplementario} \omterm{def:b1:vspaces:subspace}{subespacio} de $\ker u$ en
$E$, luego $u$ restringe a \emph{isomorfismo} desde $S$ a
$\operatorname{im} u$.
\end{theorem}

\begin{proof}
Deja que $S$ satisfaga a$E = \ker u \oplus S$
(\cref{thm:b1:findim:subspaces}), y sea $v \colon S \to
\operatorname{im} u $ la restricción de $u $.

$v$ is \omterm{def:b1:logic:inj}{inyectivo}:$\ker v = S \cap \ker u = \{0\}$.

$v$ es \omterm{def:b1:logic:inj}{sobreyectivo}: cualquier $u(x)$ con $x = k + s $($k \in \ker u$,$s
\in S $) es igual a$u(s) = v(s)$.

Entonces $v$ es un isomorfismo; un isomorfismo envía un \omterm{def:b1:vspaces:free}{base} a un \omterm{def:b1:vspaces:free}{base}
(\cref{prop:b1:linmaps:basis}), por lo que concluyen $\dim S = \dim\operatorname{im}
u $ y $\dim E = \dim\ker u + \dim S $.
\end{proof}

\begin{example}[Construir una aplicación a partir de especificaciones]
\label{ex:b1:linmaps:tospec}
Construya $u \in \mathcal{L}(\R^3)$ con $\ker u =
\operatorname{Vect}(1,1,1)$ y $\operatorname{im} u = \{z =
0\}$. La cordura primero: rango - demandas de nulidad $1 + 2 = 3$ ---
consistente, por lo que puede existir una solución. Elija un \omterm{def:b1:vspaces:free}{base} adaptado a
el \omterm{def:b1:linmaps:kerim}{núcleo}, digamos $\bigl((1,1,1),\ e_1,\ e_2\bigr)$
(\cref{ex:b1:findim:completion}), y prescribir \omterm{def:b1:linmaps:kerim}{imágenes}
(\cref{prop:b1:linmaps:basis}):
\[
u(1,1,1) = 0, \qquad u(e_1) = e_1, \qquad u(e_2) = e_2 .
\]
Entonces $\ker u \supseteq \operatorname{Vect}(1,1,1)$ y
$\operatorname{im} u = \operatorname{Vect}(e_1, e_2) = \{z =
0\}$; La nulidad de rango fuerza a$\dim\ker u = 1$, por lo que \omterm{def:b1:linmaps:kerim}{núcleo} es
exactamente la línea prescrita. Explícitamente, descomponiendo $(x, y, z)
= z(1,1,1) + (x - z)e_1 + (y - z)e_2$:
\[
u(x, y, z) = (x - z,\ y - z,\ 0).
\]
La receta generaliza: una \omterm{def:b1:linmaps:def}{aplicación lineal} con prescrito \omterm{def:b1:linmaps:kerim}{núcleo} $N$
y \omterm{def:b1:linmaps:kerim}{imagen} $I$ existe exactamente cuando $\dim N + \dim I = \dim E$
--- necesidad es rango --nulidad, suficiencia es esto
construcción.
\end{example}

\begin{corollary}\label{cor:b1:linmaps:samedim}
Si $\dim E = \dim F$(finito), entonces para $u \in \mathcal{L}(E, F)$:
\[
u \text{ injective} \iff u \text{ surjective} \iff u \text{
bijective}.
\]
En particular, esto es válido para los endomorfismos en \omterm{def:b1:findim:def}{dimensión finita}. (Es
falla en dimensión infinita: en $K[X]$, el \omterm{def:b1:derivative:def}{derivado} es \omterm{def:b1:logic:inj}{sobreyectivo}
pero no \omterm{def:b1:logic:inj}{inyectivo}, y $P \mapsto XP$ es \omterm{def:b1:logic:inj}{inyectivo} pero no
\omterm{def:b1:logic:inj}{sobreyectivo}.)
\end{corollary}

\begin{proof}
\omterm{def:b1:logic:inj}{inyectivo} $\iff \dim\ker u = 0 \iff \operatorname{rk} u = \dim E =
\dim F \iff \operatorname{im} u = F $(un \omterm{def:b1:vspaces:subspace}{subespacio} de dimensión completa
es todo, \cref{thm:b1:findim:subspaces}) $\iff$\omterm{def:b1:logic:inj}{sobreyectivo}.
\end{proof}

\begin{example}[Interpolación estructural]\label{ex:b1:linmaps:interpolation}
Fíjense $x_0, \dots, x_n$ distintos y sea
$u \colon \R_n[X] \to \R^{n+1}$,
$P \mapsto (P(x_0), \dots, P(x_n))$: es
\omterm{def:b1:linmaps:def}{lineal}. Su
\omterm{def:b1:linmaps:kerim}{núcleo} es
$\{P : \deg P \leq n,\ n+1 \text{ raíces}\} = \{0\}$
(\cref{cor:b1:poly:nroots}). Las dimensiones coinciden ($n+1$): $u$
es un isomorfismo --- existencia \emph{y} unicidad del interpolante
de Lagrange (\cref{thm:b1:poly:lagrange}) de un golpe.

El mismo patrón unifilar maneja valores de mezcla de datos y
\omterm{def:b1:derivative:def}{derivados}: $v \colon \R_3[X] \to \R^4$,$P \mapsto \bigl(P(0),
P'(0), P(1), P'(1)\bigr)$ es \omterm{def:b1:linmaps:def}{lineal} y su \omterm{def:b1:linmaps:kerim}{núcleo} consta de
grado-$\leq 3$\omterm{def:b1:poly:def}{polinomios} con raíces dobles en $0$\emph{y}
$1$, es decir divisible por $X^2(X-1)^2$ de grado $4$: sólo $P =
0$. Dimensiones iguales nuevamente: cada cuádruple de datos $(P(0),
P'(0), P(1), P'(1))$ se realiza exactamente en un cúbico ---
Interpolación de Hermite, otorgada por un cálculo \omterm{def:b1:linmaps:kerim}{núcleo} antes
cualquier fórmula está escrita (el problema del fin de semana de \cref{ch:b1:det}
cumple con su determinante).
\end{example}

\begin{example}[Rango - nulidad en funcionamiento: el operador de diferencia]
\label{ex:b1:linmaps:delta}
Sea $\Delta \colon \R_n[X] \to \R_n[X]$,$P \mapsto P(X+1) -
P(X)$: \omterm{def:b1:linmaps:def}{lineal}. \omterm{def:b1:linmaps:kerim}{Núcleo}: si $\Delta P = 0$, entonces $P(0) = P(1) =
P(2) = \dots $, entonces $P - P(0)$ tiene infinitas raíces y
desaparece (\cref{cor:b1:poly:nroots}): $\ker\Delta$ es la línea de
constantes. Rango--nulidad: $\operatorname{rk}\Delta = (n + 1) - 1
= n$. Desde $\deg \Delta P < \deg P $ para $P $ no constante (la parte superior
términos cancelar), $\operatorname{im}\Delta \subseteq \R_{n-1}[X]$,
que tiene dimensión exactamente $n$: la inclusión es una igualdad.
Conclusión, sin cálculo de \omterm{def:b1:logic:map}{imágenes inversas}: \emph{cada}
\omterm{def:b1:poly:def}{polinomio} $Q$ de grado $\leq n - 1$ es una diferencia $Q = P(X+1)
- P(X)$ --- existe la antiderivada discreta. (Compare el
problema de fin de semana de \cref{ch:b1:vspaces}, donde estaba $\Delta$
invertido explícitamente en el binomio \omterm{def:b1:vspaces:free}{base}.)
\end{example}

\begin{example}[Contabilidad de rangos en una composición]
\label{ex:b1:linmaps:rankbookkeeping}
En $\R_2[X]$, compóngase el
\omterm{def:b1:derivative:def}{derivado} $D(P) = P'$
(\omterm{def:b1:linmaps:rank}{rango}~$2$:
\omterm{def:b1:linmaps:kerim}{imagen} $\R_1[X]$,
\omterm{def:b1:linmaps:kerim}{núcleo} las constantes) consigo mismo.
Entonces $D \circ D = D^2$ es la
\omterm{def:b1:logic:map}{aplicación} $P \mapsto P''$, con
\omterm{def:b1:linmaps:kerim}{imagen} $\R_0[X]$:
\omterm{def:b1:linmaps:rank}{rango}~$1$.
Compárese con las cotas generales: la cota cruda da
$\operatorname{rk} D^2 \leq \min(2, 2) = 2$; la fórmula exacta de
\cref{exo:b1:linmaps:12} cuenta la pérdida con precisión,
\[
\operatorname{rk} D^2 = \operatorname{rk} D -
\dim\bigl(\ker D \cap \operatorname{im} D\bigr) = 2 - 1 = 1 ,
\]
pues las constantes (\omterm{def:b1:linmaps:kerim}{núcleo} del $D$ exterior)
están dentro de $\R_1[X]$ (\omterm{def:b1:linmaps:kerim}{imagen} del $D$
interior) con dimensión~$1$. El \omterm{def:b1:linmaps:rank}{rango} es
perdido exactamente donde el \omterm{def:b1:linmaps:kerim}{núcleo} externo tiende una emboscada al \omterm{def:b1:linmaps:kerim}{imagen} interno
--- la frase para recordar cuando la composición \omterm{def:b1:linmaps:rank}{rangos} se porta mal.
\end{example}

\begin{example}[Kernel e imagen del operador Euler]
\label{ex:b1:linmaps:euler}
En $\R_n[X]$, dejemos que $u(P) = X\,P'$(\omterm{def:b1:linmaps:def}{lineal}: diferenciación y
multiplicación por $X$ son). \emph{\omterm{def:b1:linmaps:kerim}{Núcleo}}:$XP' = 0$ fuerza a$P' =
0$(un producto de \omterm{def:b1:poly:def}{polinomios} desaparece sólo si un factor lo hace),
entonces $\ker u$ es la línea de constantes. \emph{\omterm{def:b1:linmaps:kerim}{Imagen}}: en el
monomio \omterm{def:b1:vspaces:free}{base},
\[
u(X^k) = k\,X^{k} \qquad (k = 0, 1, \dots, n),
\]
entonces $\operatorname{im} u = \operatorname{Vect}(X, 2X^2, \dots,
nX^n) = \operatorname{Vect}(X, X^2, \dots, X^n)$: el
\omterm{def:b1:poly:def}{polinomios} con término constante cero. Verificar contra rango - nulidad:
$\operatorname{rk} u = (n + 1) - 1 = n$, que efectivamente es el
dimensión encontrada. Dos comentarios que vale la pena conservar. Primero, aquí
$\operatorname{im} u \oplus \ker u = \R_n[X]$ --- pero eso es un
\emph{feliz accidente} de este operador, no un teorema: para el
$v(P) = P'$ en $\R_1[X]$,$\ker v = \operatorname{im}
v = \R_0[X]$ y la suma no es directa. En segundo lugar, la relación
$u(X^k) = kX^k$ dice que cada monomio simplemente es reescalado por $u$ ---
un \omterm{def:b1:vspaces:free}{base} adaptado al \omterm{def:b1:logic:map}{aplicación}, el germen de la idea del valor propio
desarrollado en el volumen Year~2.
\end{example}

\section{Proyecciones y simetrías.}

\begin{definition}\label{def:b1:linmaps:projection}
Sea $E = F \oplus G$. El \emph{proyección sobre $F$ a lo largo
$g$}\index{proyección} \omterm{def:b1:logic:map}{aplicaciones}$x = f + g $(descomposición única) para
$p(x) = f$; el \emph{simetría}\index{simetría} asociado es $s(x)
= f - g $. Ambos son \omterm{def:b1:linmaps:def}{lineal} y $s = 2p - \mathrm{id}$.
\end{definition}

\begin{theorem}[Caracterización algebraica]\label{thm:b1:linmaps:projchar}
\begin{enumerate}
 \item Un endomorfismo $p$ es un \omterm{def:b1:linmaps:projection}{proyección} (en algún $F$ a lo largo
 algunos $G$) si y sólo si $p \circ p = p$; luego $F =
 \operatorname{im} p = \ker(p - \mathrm{id})$ y $ G = \ker
 p $.
 \item Un endomorfismo $s$ es una simetría si y sólo si $s \circ s
 = \mathrm{id}$; luego $E = \ker(s - \mathrm{id}) \oplus
 \ker(s + \mathrm{id})$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) A \omterm{def:b1:linmaps:projection}{proyección} satisface $p(f + g) = f$ y $p(f) = f$:$p^2 = p$.
Por el contrario, sea $p^2 = p$; establezca $F = \operatorname{im} p$,$G = \ker
p $. Cada $x $escribe $x = p(x) + (x - p(x))$ con $p(x) \in F $ y
$p\bigl(x - p(x)\bigr) = p(x) - p^2(x) = 0$:$E = F + G$. Si$ y \in
F \cap G $:$ y = p(z)$ y $ p(y) = 0$, entonces $ y = p(z) = p^2(z) = p(y) =
0$: \omterm{def:b1:vspaces:sum}{suma directa} y $p $ es \omterm{def:b1:linmaps:projection}{proyección} en $F $a lo largo de $G $.
Finalmente en $F$:$ y = p(z)$ da $p(y) = y $, por lo que $F \subseteq
\ker(p - \mathrm{id})$ y, a la inversa,$p(y) = y $pone$ y $ en el
\omterm{def:b1:linmaps:kerim}{imagen}.

(2) La correspondencia $s = 2p - \mathrm{id}$,$p = \frac{s +
\mathrm{id}}2$ es una biyección entre endomorfismos, y bajo ella
\[
s^2 = 4p^2 - 4p + \mathrm{id} = \mathrm{id}
\iff 4p^2 = 4p \iff p^2 = p :
\]
Las simetrías corresponden exactamente a \omterm{def:b1:linmaps:projection}{proyecciones}. Traduciendo el
\omterm{def:b1:vspaces:subspace}{subespacios}: $s(x) = x \iff p(x) = x$, entonces $\ker(s - \mathrm{id}) =
\operatorname{im} p = F $; y $s(x) = -x \iff 2p(x) = 0 \iff x
\in \ker p = G $, entonces $\ker(s + \mathrm{id}) = G $. El \omterm{def:b1:vspaces:sum}{suma directa}
$E = F \oplus G$ del punto (1) pasa a ser la descomposición anunciada
en los vectores fijos y los vectores invertidos de $s$.
\end{proof}

\begin{example}[Una proyección y su simetría, explícitamente]
\label{ex:b1:linmaps:projexplicit}
En $\R^2$, proyecte sobre $F = \operatorname{Vect}(1,1)$ a lo largo de $G =
\operatorname{Vect}(0,1)$. Descomponer $(x, y) = a(1,1) + b(0,1)$:
la primera coordenada da $a = x$, la segunda $b = y - x$.
Por lo tanto
\[
p(x, y) = (x, x),
\qquad
s(x, y) = 2p(x,y) - (x,y) = (x,\ 2x - y).
\]
Verifique el álgebra: $p(p(x,y)) = p(x,x) = (x,x)$ y $s(s(x,y)) =
s(x, 2x - y) = (x, 2x - (2x - y)) = (x, y)$. Geométricamente,$s $
es la ``reflexión oblicua'' a través de la línea $ y = x$ en el
dirección vertical: fija $F$ puntualmente e invierte $G$. tenia
Si proyectamos el mismo $F$a lo largo de $G' =
\operatorname{Vect}(1,-1)$, la fórmula cambiaría a
$p'(x,y) = \bigl(\frac{x+y}2, \frac{x+y}2\bigr)$: un \omterm{def:b1:linmaps:projection}{proyección} es

\end{example}

\section{Formas lineales e hiperplanos.}

\begin{definition}\label{def:b1:linmaps:forms}
Un \emph{forma lineal}\index{forma lineal} en $E$ es una \omterm{def:b1:linmaps:def}{aplicación lineal}
$\varphi \colon E \to K$. A \emph{hiperplano}\index{hiperplano} de
$E$($\dim E = n$) es un \omterm{def:b1:vspaces:subspace}{subespacio} de dimensión $n - 1$.
\end{definition}

\begin{example}[Un formulario de evaluación y su hiperplano.]
\label{ex:b1:linmaps:evaluationform}
En $\R_2[X]$, la evaluación $\varphi(P) = P(2)$ es \omterm{def:b1:linmaps:forms}{lineal
forma}, distinta de cero ($\varphi(1) = 1$). Su \omterm{def:b1:linmaps:kerim}{núcleo} es el \omterm{def:b1:linmaps:forms}{hiperplano}
de \omterm{def:b1:poly:def}{polinomios} que desaparece en $2$, es decir,\ (teorema del factor,
\cref{thm:b1:poly:factor}) los múltiplos de $X - 2$ dentro
$\R_2[X]$:
\[
\ker\varphi = \operatorname{Vect}\bigl(X - 2,\ X(X - 2)\bigr),
\qquad \dim = 2 .
\]
En \omterm{prop:b1:vspaces:coordinates}{coordenadas} en $(1, X, X^2)$,$\varphi(a + bX + cX^2) = a +
2b + 4c $: cada \omterm{def:b1:linmaps:forms}{forma lineal} en un espacio \omterm{def:b1:findim:def}{de dimensión finita} es,
una vez que se fija un \omterm{def:b1:vspaces:free}{base}, una expresión lineal fija en el
\omterm{prop:b1:vspaces:coordinates}{coordenadas} --- las formas son ``vectores de fila'', como
\cref{ch:b1:matrices} se hará literal y el coeficiente
La fila aquí, $(1, 2, 4)$, es una fila de Vandermonde: formularios de evaluación.
Así es como funciona la teoría de interpolación del fin de semana de \cref{ch:b1:det}.
El problema entra en álgebra lineal.
\end{example}

\begin{theorem}\label{thm:b1:linmaps:hyperplanes}
El \omterm{def:b1:linmaps:forms}{hiperplanos} de $E$ es exactamente el \omterm{def:b1:linmaps:kerim}{granos} de \omterm{def:b1:linmaps:forms}{lineal
formas} distinto de cero. Dos formas distintas de cero tienen el mismo \omterm{def:b1:linmaps:kerim}{núcleo} si y sólo si
son proporcionales.
\end{theorem}

\begin{proof}
Si $\varphi \neq 0$:$\operatorname{rk}\varphi = 1$(\omterm{def:b1:linmaps:kerim}{imagen} un valor distinto de cero
\omterm{def:b1:vspaces:subspace}{subespacio} de $K$), por lo que $\dim\ker\varphi = n - 1$: a \omterm{def:b1:linmaps:forms}{hiperplano}.
Por el contrario, sea $H$ un \omterm{def:b1:linmaps:forms}{hiperplano},$(e_1, \dots, e_{n-1})$ un \omterm{def:b1:vspaces:free}{base}.
de $H$ completado por $e_n$: la forma ``última coordenada'' tiene \omterm{def:b1:linmaps:kerim}{núcleo}
$H$.

Las formas proporcionales comparten su \omterm{def:b1:linmaps:kerim}{núcleo}. Por el contrario, supongamos
$\ker\varphi = \ker\psi = H$ y elige $a \notin H$: cada $x$ escribe
$x = h + \lambda a $(como $E = H \oplus Ka$), y
\[
\varphi(x) = \lambda \varphi(a), \qquad \psi(x) = \lambda\psi(a):
\]
entonces $\varphi = \frac{\varphi(a)}{\psi(a)}\,\psi$.
\end{proof}

\begin{example}\label{ex:b1:linmaps:hyperplane}
En $K^n$, \omterm{def:b1:linmaps:forms}{hiperplano} es una solución \omterm{def:b1:logic:sets}{conjunto}$\{a_1 x_1 + \dots + a_n
x_n = 0\}$ sin que todo $a_i $ sea cero --- la familiar ecuación de a
plano que pasa por el origen en $\R^3$. En espacios funcionales, evaluación.
los formularios $P \mapsto P(1)$ o $f \mapsto \int_0^1 f$ definen \omterm{def:b1:linmaps:forms}{hiperplanos}
de $\R_n[X]$, de $C(\intcc{0}{1})$(cf.\ \cref{exo:b1:findim:6}).
\end{example}

\begin{example}[Un hiperplano, funcionó de tres maneras.]
\label{ex:b1:linmaps:hyperplanerun}
Tome $\varphi(x, y, z) = x - 2y + 3z$ en $\R^3$ y $H =

por ejemplo\ $a = (1, 0, 0)$($\varphi(a) = 1 \neq 0$); el

\begin{remark}[Errores comunes]
\label{rem:b1:linmaps:pitfalls}
\emph{\omterm{def:b1:linmaps:kerim}{Núcleo} and \omterm{def:b1:linmaps:kerim}{imagen} live in different spaces}: $\ker u
\subseteq E $,$\operatorname{im} u \subseteq F$; la suma $\ker u
+ \operatorname{im} u $ solo tiene sentido para endomorfismos, y
aun así no es necesario que sea directo ($u(x, y) = (y, 0)$ tiene $\ker u
= \operatorname{im} u $; \cref{exo:b1:linmaps:7} caracteriza
cuando se mantiene la franqueza).
\emph{$u^2 = 0$ no significa $u = 0$}: el mismo $u(x,y) =
(y, 0)$ cuadra a cero sin desaparecer --- qué $u^2 = 0$
realmente dice que es $\operatorname{im} u \subseteq \ker u$
(\cref{exo:b1:linmaps:5}).
\emph{\omterm{def:b1:logic:inj}{inyectivo} $\iff$\omterm{def:b1:logic:inj}{sobreyectivo} needs equal \omterm{def:b1:findim:def}{finito
dimensiones}}: en $K[X]$, el \omterm{def:b1:derivative:def}{derivado} es \omterm{def:b1:logic:inj}{sobreyectivo} y no
\omterm{def:b1:logic:inj}{inyectivo}, $P \mapsto XP$\omterm{def:b1:logic:inj}{inyectivo} y no \omterm{def:b1:logic:inj}{sobreyectivo}
(\cref{cor:b1:linmaps:samedim}); y entre espacios de
\emph{diferente} dimensiones, una implicación es simplemente
imposible ($\operatorname{rk} u \leq \min(\dim E, \dim F)$).
\emph{Prescribing \omterm{def:b1:linmaps:kerim}{imágenes} works on a \omterm{def:b1:vspaces:free}{base}, not on any family}:
exigente $u(1, 0) = a$,$u(0, 1) = b$,$u(1, 1) = c$
sobredetermina $u$a menos que $c = a + b$; una \omterm{def:b1:linmaps:def}{aplicación lineal} es \omterm{def:b1:vspaces:free}{libre} en un
\omterm{def:b1:vspaces:free}{base}, esclavizado en todos lados.
\emph{\omterm{def:b1:linmaps:rank}{Rango} is not preserved by composition}: sólo puede caer,
$\operatorname{rk}(vu) \leq \min(\operatorname{rk} u,
\operatorname{rk} v)$(\cref{exo:b1:linmaps:4}), con pérdida exacta
medido en \cref{exo:b1:linmaps:12}.
\end{remark}

\begin{remark}[Adónde van estas aplicaciones]
\label{rem:b1:linmaps:whereused}
\omterm{def:b1:linmaps:def}{aplicaciones lineales} están a punto de convertirse en \emph{matrices}: una vez que las bases estén
fijo, \cref{ch:b1:matrices} codifica cada $u \in \mathcal{L}(E,
F)$ mediante una matriz rectangular, y la composición se convierte en la matriz
producto --- rango --nulidad impulsa la teoría de \omterm{def:b1:linmaps:def}{lineal}
sistemas en \cref{ch:b1:det}. \omterm{def:b1:linmaps:projection}{Proyecciones} regreso en
\cref{ch:b1:euclid} en su caso especial más útil, el
\emph{ortogonal} \omterm{def:b1:linmaps:projection}{proyección}, donde se elige el \omterm{def:b1:linmaps:kerim}{núcleo}
perpendicular al \omterm{def:b1:linmaps:kerim}{imagen}. El siguiente problema del fin de semana empuja al
álgebra del proyector hasta donde llegan las herramientas del primer año, hasta
Lema de apropiado; el volumen Year~2 va más allá con la traza
y con la teoría de valores propios, para la cual los proyectores sobre estables
\omterm{def:b1:vspaces:subspace}{subespacios} son los componentes básicos.
\end{remark}

\begin{remark}[Perspectivas dentro del Libro 3: rango--nulidad tres veces más]
\label{rem:b1:linmaps:perspectives}
La ley de conservación $\dim E = \dim\ker u + \operatorname{rk} u$
Se volverá a leer tres veces antes de que termine el volumen. en
\cref{ch:b1:det} se convierte en la forma de la solución \omterm{def:b1:logic:sets}{conjuntos}: a
sistema compatible con $p$ desconocidos y \omterm{def:b1:linmaps:rank}{rango}$r$ tiene una solución
\omterm{def:b1:logic:sets}{conjunto} de dimensión $p - r$ --- dimensión \omterm{def:b1:linmaps:kerim}{núcleo} disfrazada. en
\cref{ch:b1:euclid} se divide ortogonalmente, $\dim F + \dim
F^\perp = \dim E $, y potencia todos los cálculos de distancia. en
el problema del fin de semana de \cref{ch:b1:multivar}, es el
contador de mínimos cuadrados: $n$ observaciones,$2$ ajustado
parámetros, dimensiones residuales $n - 2$ y el método pitagórico.
identidad $\norm b^2 = \norm p^2 + \norm{b - p}^2$ es
rango: la sombra euclidiana de la nulidad. Un teorema, cuatro disfraces.
\end{remark}

\section{Problema: cálculo del proyector y lema de adaptación}

\begin{problem}\label{pb:b1:linmaps:1}
\omterm{def:b1:linmaps:projection}{Proyecciones} son los endomorfismos que produce \omterm{def:b1:vspaces:sum}{sumas directas}, y
por el contrario: cada identidad $E = F_1 \oplus \dots \oplus F_k$ es
En secreto una familia de proyectores sumando la identidad. esto
problema desarrolla ese diccionario --- el álgebra de uno
proyector, de dos, de $k$ --- luego se aplica lo mismo
ideas de estabilización a un endomorfismo arbitrario y demuestra
\emph{Lema de montaje}: todo endomorfismo de un
\omterm{def:b1:findim:def}{de dimensión finita} el espacio se divide en una parte nilpotente y una
parte reversible. En todo momento, $E$ es un espacio vectorial $K$ de
dimensión $n$ y \emph{proyector} significa $p \in \mathcal{L}(E)$
con $p^2 = p $(\cref{thm:b1:linmaps:projchar}).
\index{proyección}\index{Lema de montaje}\index{nilpotente}

\medskip
\noindent\textbf{Parte I --- The algebra around one projector.}
Sea $p$ un proyector,$p \neq 0$,$p \neq \mathrm{id}$.
\begin{enumerate}
 \item Demuestre que $\mathrm{id} - p$ es un proyector e identifique
 $\operatorname{im}(\mathrm{id} - p)$ y
 $\ker(\mathrm{id} - p)$.
 \item Calcular $(\lambda\,\mathrm{id} + \mu\,p)^2$ y determinar
 todos los pares $(\lambda, \mu) \in K^2$ para los cuales
 $\lambda\,\mathrm{id} + \mu\,p$ es un proyector.
 \item Muestra que el avión $\operatorname{Vect}(\mathrm{id}, p)$
 de $\mathcal{L}(E)$ es estable bajo composición, y que
 por cada \omterm{def:b1:poly:def}{polinomio} $Q \in K[X]$,
 \[
 Q(p) = Q(0)\,\mathrm{id} + \bigl(Q(1) -
 Q(0)\bigr)\,p .
 \]
 \item Determinar para qué $(\lambda, \mu)$ la \omterm{def:b1:logic:map}{aplicación}
 $\lambda\,\mathrm{id} + \mu\,p$ es invertible y da
 su inverso en la forma $\alpha\,\mathrm{id} + \beta\,p$.
 Interpretar la respuesta mediante la acción de
 $\lambda\,\mathrm{id} + \mu\,p$ en $\operatorname{im} p$
 y en $\ker p$.
 \item Sea $p'$ sea otro proyector con el \emph{same \omterm{def:b1:linmaps:kerim}{imagen}}
 $\operatorname{im} p' = \operatorname{im} p$. mostrar eso
 $p\,p' = p'$ y $p'\,p = p$. ¿Qué significan estas identidades?
 decir sobre componer \omterm{def:b1:linmaps:projection}{proyecciones} en el mismo \omterm{def:b1:vspaces:subspace}{subespacio}
 ¿A lo largo de diferentes \omterm{def:b1:linmaps:kerim}{granos}?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Two projectors.} Sea $p, q$
proyectores $E$; asumir que la característica no es $2$(verdadero
para $K = \R, \C$).
\begin{enumerate}[resume]
 \item Supongamos que $p + q$ es un proyector. Expandiendo $(p + q)^2$,
 mostrar $pq + qp = 0$; componiendo con $p$ a la izquierda, luego
 a la derecha, deduzca $pq = qp$ y concluya $pq = qp =
 0$.
 \item Por el contrario, supongamos $pq = qp = 0$. Demuestre que $p + q$ es un
 proyector, con
 \[
 \operatorname{im}(p + q) = \operatorname{im} p \oplus
 \operatorname{im} q,
 \qquad
 \ker(p + q) = \ker p \cap \ker q .
 \]
 \item Demuestre que $p - q$ es un proyector si y sólo si $pq = qp
 = q $. \emph{(Apply questions 6--7 to $\mathrm{id} - p$
 and $q$.)}
 \item Muestre el significado geométrico de $pq = qp = q$: se cumple si
 y sólo si $\operatorname{im} q \subseteq
 \operatorname{im} p$ y $\ker p \subseteq \ker q $. (Uno
 luego escribe $q \leq p$: ``$q$ se proyecta en menos, junto
 más ''.)
 \item Ahora deja que $p, q$ conmute. Recordar de
 \cref{exo:b1:linmaps:6} que $pq$ es el proyector en
 $\operatorname{im} p \cap \operatorname{im} q$ junto
 $\ker p + \ker q$. Demuestre que $r = p + q - pq $ es un
 proyector con
 \[
 \operatorname{im} r = \operatorname{im} p +
 \operatorname{im} q,
 \qquad
 \ker r = \ker p \cap \ker q .
 \]
 \emph{(Consider $\mathrm{id} - r = (\mathrm{id} -
 p)(\mathrm{id} - q)$.)}
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Decompositions of the identity.}
\begin{enumerate}[resume]
 \item Sea $E = F_1 \oplus \dots \oplus F_k$ y, para $x = x_1 +
