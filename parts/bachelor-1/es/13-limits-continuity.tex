\section{Límites de funciones}

\begin{definition}[Límite en un punto]\label{def:b1:continuity:limit}
Sea $x_0 \in \overline{I}$ y $\ell \in \R$. Entonces $f(x) \to \ell $ como
$x \to x_0$ cuando\index{límite!de una función}
\[
\forall \varepsilon > 0,\ \exists \delta > 0,\ \forall x \in I,
\qquad \abs{x - x_0} \leq \delta \implies \abs{f(x) - \ell} \leq
\varepsilon .
\]
Los límites en $\pm\infty$ y los límites infinitos están definidos por el mismo
patrón ($\abs{x - x_0} \leq \delta$ se convierte en $x \geq M$;$\abs{f(x) -
\ell} \leq \varepsilon$ se convierte en $f(x) \geq M'$). Límites unilaterales
restrinja $x$a$x > x_0$(escrito $x \to x_0^+$) o $x < x_0$. el
El límite es único cuando existe (la misma prueba que
\cref{prop:b1:seq:first}).
\end{definition}

\begin{example}[Un límite mediante el presionado]\label{ex:b1:continuity:floorlimit}
Calcule $\lim_{x \to 0} x\,\bigl\lfloor \frac1x \bigr\rfloor$.
El \omterm{thm:b1:reals:floor}{piso} entre corchetes $\frac1x - 1 < \lfloor \frac1x \rfloor \leq
\frac1x $ da, después de multiplicar por $x $(¡cuidado con el signo!):
\[
1 - x < x\Bigl\lfloor \frac1x \Bigr\rfloor \leq 1 \quad (x > 0),
\qquad
1 \leq x\Bigl\lfloor \frac1x \Bigr\rfloor < 1 - x \quad (x < 0),
\]
y ambos apretones unilaterales se cierran en $1$: el límite es $1$. Nota
qué pasó: $\lfloor \frac1x\rfloor$ solo tiene saltos salvajes cerca
$0$, pero el factor $x$ controla cada salto ($x$ veces un salto unitario es
pequeño), y sólo sobrevive el paréntesis. La idea final:
límites de productos de un factor pequeño con una oscilación acotada
Los factores son problemas de compresión, nunca problemas de teoremas de operación.
el teorema de operaciones necesita factores \emph{ambos} para converger.
\end{example}

\begin{theorem}[Caracterización secuencial]\label{thm:b1:continuity:seqchar}
$f(x) \to \ell$ como $x \to x_0$ si y sólo si: para \emph{cada}
secuencia $(u_n)$ de puntos de $I$ con $u_n \to x_0$, se tiene
$f(u_n) \to \ell$.
\end{theorem}

\begin{proof}
($\Rightarrow$) Sea $u_n \to x_0$ y $\varepsilon > 0$. tomar
$\delta$ de la definición, luego $N$ con $\abs{u_n - x_0} \leq
\delta$ para $n \geq N $: más allá de $N $,$\abs{f(u_n) - \ell} \leq
\varepsilon$.

($\Leftarrow$) Contrapositivo. Si$f \not\to \ell $: algunos
$\varepsilon_0 > 0$ derrota a todos los $\delta$; elegir $\delta =
\frac{1}{n+1}$ produce$ u_n \in I $ con $\abs{u_n - x_0} \leq
\frac{1}{n+1}$ y $\abs{f(u_n) - \ell} > \varepsilon_0$. Entonces $u_n
\to x_0$ pero $ f(u_n) \not\to \ell $.
\end{proof}

\begin{corollary}[Operaciones, composición, orden.]\label{cor:b1:continuity:operations}
Se comportan sumas, productos, cocientes (límite distinto de cero en la parte inferior) de límites
en cuanto a secuencias; si $f \to \ell$ en $x_0$ y $g \to m$ en $\ell$,
y $g$ se define alrededor de $\ell$ con $g(\ell) = m$ o $f \neq \ell $
cerca de $x_0$, luego $g \circ f \to m$ en $x_0$; los límites se mantienen amplios
desigualdades y se cumple el teorema de la compresión.
\end{corollary}

\begin{proof}
Cada \omterm{def:b1:logic:statement}{enunciado} se transfiere a través de
\cref{thm:b1:continuity:seqchar} a su secuencia análoga
(\cref{thm:b1:seq:operations,thm:b1:seq:order}). Para la composición,
Vale la pena escribir el encadenamiento directo una vez: dado $\varepsilon >
0$, el límite de $g $ en $\ell$ proporciona a$\eta > 0$\[
\abs{y - \ell} \leq \eta \implies \abs{g(y) - m} \leq
\varepsilon \quad (y \text{ in the domain of } g),
\]
donde se cubre el caso $ y = \ell$ porque $g(\ell) = m$; entonces
el límite de $f$ en $x_0$ proporciona a$\delta > 0$ con $\abs{x -
x_0} \leq \delta \implies \abs{f(x) - \ell} \leq \eta$; encadenamiento,
$\abs{x - x_0} \leq \delta$ da $\abs{g(f(x)) - m} \leq
\varepsilon$. En la hipótesis alternativa ($f \neq \ell$ cerca
$x_0$), el valor$ y = \ell $nunca se envía a$g$ y su valor
hay irrelevante.
\end{proof}

\begin{example}[¿Por qué existe la condición de composición?]\label{ex:b1:continuity:compositiontrap}
Sea $g(y) = 0$ para $ y \neq 0$ y $g(0) = 1$, y sea $f$
idénticamente $0$. Luego $f(x) \to 0$ como $x \to 0$ y $g(y) \to 0$.
como $ y \to 0$; sin embargo,$g(f(x)) = g(0) = 1$ por cada $x$, entonces $g
\circ f \to 1 \neq 0$. La función interna se encuentra \emph{exactamente en}
el valor prohibido $\ell = 0$ para siempre y el límite de $g$ en $0$
ignora lo que hace $g$ en $0$. la condición de
\cref{cor:b1:continuity:operations} --- ya sea $g(\ell) = m$
(es decir,\ $g$\omterm{def:b1:continuity:continuous}{continuo} en $\ell$), o $f \neq \ell $cerca de $x_0$ ---
es precisamente lo que descarta esto. La idea final: en
práctica uno compone funciones \emph{\omterm{def:b1:continuity:continuous}{continuo}} y el
la condición es gratuita; muerde sólo cuando se llevan límites
perforado \omterm{def:b1:topology:open}{barrios}, por lo que la definición de
$\lim_{x \to x_0}$ utilizado en este libro incluye el punto en el que
está en el dominio.
\end{example}

\section{Continuidad}

\begin{definition}\label{def:b1:continuity:continuous}
$f$ es \emph{continuous at $x_0 \in I$}\index{continuidad} cuando $f(x)
\to f(x_0)$ es $ x \to x_0$; \emph{continuo en $I $} cuando es
continuo en cada punto. Por
\cref{thm:b1:continuity:seqchar}: $f$ es continuo en $x_0$ si y así
$f(u_n) \to f(x_0)$ para cada secuencia $u_n \to x_0$ en $I$.
\end{definition}

\begin{example}[Una taxonomía de discontinuidades.]\label{ex:b1:continuity:taxonomy}
Tres formas de fallar en un punto, cada vez con mayor gravedad.
\emph{Desmontable}: $f(x) = \frac{\sin x}{x}$ en $\R^*$ tiene límite
$1$ en $0$; definiendo $f(0) = 1$ lo repara --- la discontinuidad
Era un agujero, no una característica. \emph{Saltar}: $\lfloor x \rfloor$ en
un número entero tiene límites unilaterales distintos ($n - 1$ y $n$); no
La elección del valor puede conciliarlos, pero ambos límites existen.
\emph{Básico}: $\sin\frac1x$ en $0$ no tiene límite unilateral en
todos (\cref{exo:b1:continuity:1}) --- oscilación sin
liquidación. Las funciones monótonas sólo pueden producir el tipo medio.
(sus límites unilaterales siempre existen, siendo suprema e ínfima),
por eso sus discontinuidades \omterm{def:b1:logic:sets}{conjuntos} son como mucho contables ---
un racional por salto. Derivadas, según el teorema de Darboux
(\cref{exo:b1:derivative:10}), sólo puede producir el último tipo:
una función con una discontinuidad de salto nunca es la derivada de
cualquier cosa.
\end{example}

\begin{example}[Max y min de funciones continuas]\label{ex:b1:continuity:maxmin}
Si $f, g$ son \omterm{def:b1:continuity:continuous}{continuo}, también lo son $\max(f, g)$ y $\min(f, g)$:
no es necesario analizar el caso, gracias a las identidades
\[
\max(f, g) = \frac{f + g + \abs{f - g}}{2},
\qquad
\min(f, g) = \frac{f + g - \abs{f - g}}{2},
\]
y el \omterm{def:b1:continuity:continuous}{continuidad} de sumas y del $\abs{\,\cdot\,}$
(\cref{prop:b1:continuity:algebra}). En particular $f^+ =
\max(f, 0)$ y $ f^- = \max(-f, 0)$ son \omterm{def:b1:continuity:continuous}{continuo} con $f =
f^+ - f^-$: la división de signos utilizada para series
(\cref{ch:b1:series}) y, a escala completa, en la integración
teoría del volumen del Año 3, no cuesta nada en regularidad.
\end{example}

\begin{theorem}[Teorema del valor intermedio]\label{thm:b1:continuity:ivt}
Sea $f$\omterm{def:b1:continuity:continuous}{continuo} en $\intcc{a}{b}$ con $f(a) \leq 0 \leq
f(b)$. Luego $f(c) = 0$ para algunos $c \in \intcc{a}{b}$.
En consecuencia, una función \omterm{def:b1:continuity:continuous}{continuo} en un \omterm{prop:b1:reals:intervals}{intervalo} toma todos los valores
entre dos de sus valores: $f(I)$ es un
\omterm{prop:b1:reals:intervals}{intervalo}.\index{teorema del valor intermedio}
\end{theorem}

\begin{proof}
Dicotomía. Establezca $a_0 = a$,$b_0 = b$. Dado $\intcc{a_n}{b_n}$ con
$f(a_n) \leq 0 \leq f(b_n)$, sea $m$ el punto medio: si $f(m) \leq
0$ conserve $\intcc{m}{b_n}$, en caso contrario conserve $\intcc{a_n}{m}$; el signo
las condiciones persisten. Las secuencias $(a_n), (b_n)$ son adyacentes ($b_n -
a_n = \frac{b-a}{2^n}$), con límite común $c $
(\cref{thm:b1:seq:adjacent}). Por \omterm{def:b1:continuity:continuous}{continuidad} y
\cref{thm:b1:seq:order}: $f(c) = \lim f(a_n) \leq 0$ y $f(c) =
\lim f(b_n) \geq 0$, entonces $f(c) = 0$.

Para la consecuencia: valores dados $f(u) < v < f(w)$, aplique lo anterior
a $x \mapsto f(x) - v$ en el segmento con puntos finales $u$ y $w$;
por lo tanto $f(I)$ es convexo, es decir, un \omterm{prop:b1:reals:intervals}{intervalo}
(\cref{prop:b1:reals:intervals}).
\end{proof}

\begin{example}[Una ecuación, el protocolo completo.]\label{ex:b1:continuity:protocol}
Resuelva $\eu^x = 3 - x$ sobre $\R$: existencia, unicidad,
ubicación. Configure $g(x) = \eu^x + x - 3$, \omterm{def:b1:continuity:continuous}{continuo}. Ubicación y
existencia: $g(0) = -2 < 0$ y $g(1) = \eu - 2 > 0$, por lo que el
El teorema del valor intermedio plantea una solución en
$\intoo{0}{1}$. Unicidad:$g$ es una suma de lo estrictamente
aumentando $\eu^x$ y $x - 3$, por lo que aumenta estrictamente en
$\R$; una función estrictamente monótona toma cada valor como máximo
una vez, por lo que la solución es única en todo $\R$(no solo en el
\omterm{prop:b1:reals:intervals}{intervalo} sondeado). El protocolo --- reorganizar a $g = 0$, firmar
cambio por existencia, monotonicidad por unicidad --- se asienta
la mayoría de las preguntas de "cuántas soluciones" en tres líneas, y
Tablas de variaciones de \cref{ch:b1:derivative}
extiéndalo a $g$ no monótono cortando $\R$ en monótono
ramas.
\end{example}

\begin{example}[Dicotomía como algoritmo]\label{ex:b1:continuity:dichotomy}
Se calcula la prueba de \cref{thm:b1:continuity:ivt}. Tome $f(x) =
x^3 + x - 1$:$f(0) = -1 < 0 < 1 = f(1)$, por lo que una raíz se encuentra en
$\intoo{0}{1}$. Reducir a la mitad:
\[
f(0.5) = -0.375 < 0, \qquad
f(0.75) = 0.171875 > 0, \qquad
f(0.625) = -0.130859375 < 0 ,
\]
entonces la raíz queda atrapada sucesivamente en $\intoo{0.5}{1}$, luego
$\intoo{0.5}{0.75}$, luego $\intoo{0.625}{0.75}$(valor verdadero:
$c \approx 0.6823$). Después de los pasos $n$ el error es como máximo
$\frac{b - a}{2^n}$: diez pasos dan tres decimales, veinte dan
seis. La idea final: el teorema del valor intermedio no es
sólo una existencia \omterm{def:b1:logic:statement}{enunciado} --- su prueba de dicotomía es una
algoritmo de búsqueda de raíces garantizado, aunque lento, contra el cual el
El método Newton rápido pero local de \cref{ch:b1:derivative} debería
ser medido.
\end{example}

\begin{theorem}[Teorema del valor extremo]\label{thm:b1:continuity:evt}
Una función \omterm{def:b1:continuity:continuous}{continuo} en un segmento $\intcc{a}{b}$ está acotada y
alcanza sus límites: hay $c, d \in \intcc{a}{b}$ con
\[
f(c) = \inf_{\intcc{a}{b}} f,
\qquad
f(d) = \sup_{\intcc{a}{b}} f .
\]
Combinado con \cref{thm:b1:continuity:ivt}: \emph{the \omterm{def:b1:continuity:continuous}{continuo}
image of a segment is a segment} $\intcc{f(c)}{f(d)}$.
\index{teorema del valor extremo}
\end{theorem}

\begin{proof}
\emph{Acotado arriba:}, de lo contrario, elija $u_n$ con $f(u_n) \geq n$. Por
compacidad del segmento (\cref{thm:b1:topology:compact}), una
\omterm{def:b1:seq:subsequence}{subsecuencia}$u_{\varphi(n)} \to x \in \intcc{a}{b}$; \omterm{def:b1:continuity:continuous}{continuidad} da
$f(u_{\varphi(n)}) \to f(x)$, pero $f(u_{\varphi(n)}) \geq \varphi(n)
\to +\infty $: contradicción.

\emph{Sup alcanzado:} deja $M = \sup f$ y elige $v_n$ con $f(v_n) >
M - \frac{1}{n+1}$(\cref{prop:b1:reals:epsilon}). Extracto
$v_{\varphi(n)} \to d \in \intcc{a}{b}$: luego $f(d) = \lim
f(v_{\varphi(n)}) = M $ por el apretón. El \omterm{def:b1:reals:bounds}{ínfimo} es manejado por
$-f$.
\end{proof}

\begin{example}[Mínimo positivo en un segmento]\label{ex:b1:continuity:posmin}
Sea $f$\omterm{def:b1:continuity:continuous}{continuo} en $\intcc{0}{1}$ con $f(x) > 0$ para
cada $x$. Entonces $\inf f > 0$: por el teorema del valor extremo el
\omterm{def:b1:reals:bounds}{ínfimo} es un \emph{valor} $f(c)$ y $f(c) > 0$ por hipótesis.
Entonces una función positiva \omterm{def:b1:continuity:continuous}{continuo} en un segmento está acotada
de $0$ --- un argumento de dos líneas usado una docena de veces en el
próximos capítulos (denominadores bajo control, función escalonada
encuadres, límites de error). En un \omterm{prop:b1:reals:intervals}{intervalo} no compacto esto falla
espectacularmente: $f(x) = x$ en $\intoc{0}{1}$ es \omterm{def:b1:continuity:continuous}{continuo} y
positivo con $\inf f = 0$, no alcanzado. La idea final:
"positivo" se actualiza a "uniformemente positivo" exactamente cuando el
el dominio es compacto; cada hipótesis del teorema del valor extremo
es portante.
\end{example}

\begin{remark}[Errores comunes en torno a los tres teoremas]
(i) \emph{\omterm{def:b1:continuity:continuous}{Continuo} images}: sólo \emph{segmentos} son robustos.
La imagen \omterm{def:b1:continuity:continuous}{continuo} de un \omterm{def:b1:topology:open}{abierto} \omterm{prop:b1:reals:intervals}{intervalo} no necesita ser \omterm{def:b1:topology:open}{abierto}
($x \mapsto x^2$\omterm{def:b1:logic:map}{aplicaciones}$\intoo{-1}{1}$ sobre $\intco{0}{1}$), el
La imagen de un \omterm{def:b1:topology:closed}{conjunto cerrado} no necesita ser \omterm{def:b1:topology:closed}{cerrado} ($\arctan$\omterm{def:b1:logic:map}{aplicaciones} el
\omterm{def:b1:topology:closed}{cerrado} $\R$ hacia \omterm{def:b1:topology:open}{abierto}
$\intoo{-\frac\pi2}{\frac\pi2}$); pero la imagen de un segmento es
un segmento (\cref{thm:b1:continuity:evt}). (ii) \emph{The
intermediate value theorem needs an \omterm{prop:b1:reals:intervals}{intervalo}}: la función $x
\mapsto \frac1x $, \omterm{def:b1:continuity:continuous}{continuo} en $\intco{-1}{0} \cup
\intoc{0}{1}$, toma los valores$-1$ y $1$ pero nunca
desaparece --- su dominio son dos piezas disjuntas, y el valor
$0$ cae en el hueco; nombre siempre el \omterm{prop:b1:reals:intervals}{intervalo} en el que se
se aplica el teorema. (iii) \emph{\omterm{def:b1:continuity:uniform}{Continuidad uniforme} is a property of the
pair (function, \omterm{def:b1:logic:sets}{conjunto})}: $x^2$ es \omterm{def:b1:continuity:uniform}{uniformemente continuo} en cada
segmento aún no en $\R$
(\cref{ex:b1:continuity:notuniform}) --- las palabras ``\omterm{def:b1:continuity:uniform}{uniformemente
continuo}'' sin un dominio no tienen sentido. (v)
\emph{\omterm{def:b1:continuity:continuous}{Continuidad} of the inverse is not formal}: aguanta
\omterm{prop:b1:reals:intervals}{intervalos} vía monotonicidad
(\cref{thm:b1:continuity:bijection}), pero una biyección \omterm{def:b1:continuity:continuous}{continuo}
entre uniones de \omterm{prop:b1:reals:intervals}{intervalos} puede tener una inversa discontinua ---
la observación después de ese teorema está ahí porque los estudiantes lo citan
sin la hipótesis \omterm{prop:b1:reals:intervals}{intervalo}.
\end{remark}

\section{Funciones monótonas y funciones inversas.}

\begin{theorem}[Teorema de biyección monotona]\label{thm:b1:continuity:bijection}
Sea $f$\omterm{def:b1:continuity:continuous}{continuo} y estrictamente monótono en un \omterm{prop:b1:reals:intervals}{intervalo}$I$. entonces
\begin{enumerate}
 \item $f$ es una biyección de $I$ sobre \omterm{prop:b1:reals:intervals}{intervalo}$J = f(I)$;
 \item el inverso $f^{-1} \colon J \to I$ es estrictamente monótono (en
 la misma dirección) y \emph{\omterm{def:b1:continuity:continuous}{continuo}}.
\end{enumerate}
\end{theorem}

\begin{proof}
Digamos que $f$ es estrictamente creciente. (1) La inyectividad es inmediata desde
estricta monotonicidad; la sobreyectividad sobre $f(I)$ es trivial y $f(I)$
es un \omterm{prop:b1:reals:intervals}{intervalo} de \cref{thm:b1:continuity:ivt}.

(2) $f^{-1}$ es estrictamente creciente: si $ y < y'$ en $J$ pero $f^{-1}(y)
\geq f^{-1}(y')$, aplicando el $f $creciente da $ y \geq y'$,
absurdo. \omterm{def:b1:continuity:continuous}{Continuidad} de $f^{-1}$ en $y_0 = f(x_0) \in J$: vamos
$\varepsilon > 0$. Supongamos primero $x_0$\omterm{def:b1:topology:closure}{interior} a$I$, y
encoger $\varepsilon$ para que $x_0 \pm \varepsilon \in I$: su
las imágenes satisfacen $f(x_0 - \varepsilon) < y_0 < f(x_0 +
\varepsilon)$. tomar
$\delta = \min\bigl(y_0 - f(x_0 - \varepsilon),\, f(x_0 +
\varepsilon) - y_0\bigr) > 0$: para $\abs{y - y_0} \leq \delta$,
la monotonicidad de $f^{-1}$ aprieta a$f^{-1}(y)$ entre $x_0 -
\varepsilon $ y $x_0 + \varepsilon$. Si $x_0$ es, digamos, la izquierda
punto final de $I$, solo $x_0 + \varepsilon $ está disponible: entonces

\section{Continuidad uniforme}

\begin{definition}\label{def:b1:continuity:uniform}
$f \colon I \to \R$ es \emph{uniformemente continuo}\index{uniforme
continuidad} cuando
\[
\forall \varepsilon > 0,\ \exists \delta > 0,\ \forall x, y \in I,
\qquad \abs{x - y} \leq \delta \implies \abs{f(x) - f(y)} \leq
\varepsilon .
\]
El punto: $\delta$ depende sólo de $\varepsilon$, no del
ubicación en $I$. Uniforme \omterm{def:b1:continuity:continuous}{continuidad} implica \omterm{def:b1:continuity:continuous}{continuidad}; un
Función \emph{Lipschitz} ($\abs{f(x) - f(y)} \leq k \abs{x - y}$)
es uniformemente \omterm{def:b1:continuity:continuous}{continuo} ($\delta = \varepsilon/k$).
\end{definition}

\begin{example}\label{ex:b1:continuity:notuniform}
$x \mapsto x^2$ es \omterm{def:b1:continuity:continuous}{continuo} en $\R$ pero \emph{no} \omterm{def:b1:continuity:uniform}{uniformemente
continuo}: $\abs{(n + \frac 1n)^2 - n^2} = 2 + \frac{1}{n^2} \geq 2$
aunque los argumentos están a distancia $\frac 1n \to 0$. En cualquier
acotado \omterm{prop:b1:reals:intervals}{intervalo} es Lipschitz, por lo tanto \omterm{def:b1:continuity:uniform}{uniformemente continuo} ---
consistente con el teorema de Heine a continuación.
\end{example}

\begin{example}[Módulos uniformes, específicamente]\label{ex:b1:continuity:moduli}
En un segmento, Heine garantiza un $\delta$ uniforme; a menudo uno puede
también \emph{calcular}. Para $f(x) = x^2$ en $\intcc{0}{10}$:
\[
\abs{x^2 - y^2} = \abs{x + y}\,\abs{x - y} \leq 20\,\abs{x - y},
\]
entonces $\delta = \frac{\varepsilon}{20}$ funciona uniformemente (a
Lipschitz \omterm{def:b1:complex:field}{módulo}, lineal en $\varepsilon$). Para $\sqrt x$ en
$\intcc{0}{1}$:$\abs{\sqrt x - \sqrt y} \leq \sqrt{\abs{x -
y}}$(\cref{exo:b1:continuity:9}), entonces $\delta = \varepsilon^2$
funciona --- uniforme pero \emph{no} lineal: cerca de $0$ el cuadrado
la raíz es pronunciada y el precio aparece en el exponente de
$\varepsilon$, no por falta de uniformidad. el cierre
Insight: \omterm{def:b1:continuity:uniform}{continuidad uniforme} es un espectro, no un sí/no ---
la función $\delta(\varepsilon)$, llamada \omterm{def:b1:complex:field}{módulo}, mide
lo cara que es la uniformidad, y Lipschitz es simplemente su
mejor nota.
\end{example}

\begin{theorem}[Heine]\label{thm:b1:continuity:heine}
Una función \omterm{def:b1:continuity:continuous}{continuo} en un \emph{segmento} $\intcc{a}{b}$ es \omterm{def:b1:continuity:uniform}{uniformemente
continuo}.\index{teorema de heine}
\end{theorem}

\begin{proof}
Por contradicción: supongamos que algún $\varepsilon_0 > 0$ vence a todos.
$\delta$. Con $\delta = \frac{1}{n+1}$, elija $x_n, y_n \in
\intcc{a}{b}$ con $\abs{x_n - y_n} \leq \frac{1}{n+1}$ y
$\abs{f(x_n) - f(y_n)} > \varepsilon_0$. Compacidad
(\cref{thm:b1:topology:compact}) extrae $x_{\varphi(n)} \to c \in
\intcc{a}{b}$; luego $y_{\varphi(n)} \to c $ también (presione $\abs{x -
y}$). \omterm{def:b1:continuity:continuous}{Continuidad} en $c $ da $f(x_{\varphi(n)}) \to f(c)$ y
$f(y_{\varphi(n)}) \to f(c)$, entonces $\abs{f(x_{\varphi(n)}) -
f(y_{\varphi(n)})} \to 0 < \varepsilon_0$: contradicción.
\end{proof}

\begin{example}[Acotado, continuo, pero no uniforme.]\label{ex:b1:continuity:sinx2}
La función $f(x) = \sin(x^2)$ es \omterm{def:b1:continuity:continuous}{continuo} y está limitada por
$\R$, pero no \omterm{def:b1:continuity:uniform}{uniformemente continuo}. tomar
\[
x_n = \sqrt{2\pi n}, \qquad y_n = \sqrt{2\pi n + \tfrac\pi2}:
\qquad
y_n - x_n = \frac{\pi/2}{\sqrt{2\pi n + \frac\pi2} +
\sqrt{2\pi n}} \longrightarrow 0 ,
\]
sin embargo, $f(y_n) - f(x_n) = \sin\bigl(2\pi n + \frac\pi2\bigr) -
\sin(2\pi n) = 1 - 0 = 1$ por cada $n $: ningún $\delta$ puede
sirva $\varepsilon = \frac12$ en todas partes. Geométricamente, el
oscilaciones de $\sin(x^2)$\emph{acelerar}: el gráfico

\end{example}

Las funciones aquí son \omterm{def:b1:continuity:continuous}{continuo}.

\noindent\textbf{Part V --- Jensen and midpoint convexity.}
\begin{enumerate}[resume]
 \item (ecuación de Jensen) Sea $f \colon \R \to \R$\omterm{def:b1:continuity:continuous}{continuo} con $f\bigl(\frac{x+y}{2}\bigr) = \frac{f(x)
 + f(y)}{2}$. Demuestre que $g = f - f(0)$ satisface
 $g\bigl(\frac x2\bigr) = \frac{g(x)}{2}$, deduce que
 $g$ es aditivo y concluye $f(x) = cx + d$.
 \item Supongamos ahora solo el \emph{desigualdad}: $f$\omterm{def:b1:continuity:continuous}{continuo}
 con
 \[
 f\Bigl(\frac{x + y}{2}\Bigr) \leq \frac{f(x) +
 f(y)}{2} \qquad (x, y \in \R).
 \]
 Demuestre por inducción en $n$ que para todos los pesos diádicos
 $\lambda = \frac{k}{2^n} \in \intcc{0}{1}$:
 \[
 f\bigl(\lambda x + (1 - \lambda)y\bigr) \leq \lambda
 f(x) + (1 - \lambda) f(y) .
 \]
 \item Ampliar por \omterm{def:b1:continuity:continuous}{continuidad} y \omterm{def:b1:topology:dense}{densidad} de las diádicas
 (\cref{exo:b1:reals:8}) a cada $\lambda \in
 \intcc{0}{1}$: una función convexa de punto medio \omterm{def:b1:continuity:continuous}{continuo}
 satisface la desigualdad de convexidad total (la noción
 estudiado sistemáticamente en \cref{ch:b1:derivative}).
 \item Mostrar \omterm{def:b1:continuity:continuous}{continuidad} no se puede eliminar: un aditivo no lineal

 propiedad (diádica), la convexidad es continua;
 \omterm{def:b1:continuity:continuous}{continuidad} es el puente --- exactamente como en las Partes I--II.
\end{enumerate}
