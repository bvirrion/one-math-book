\section{El avión $\R^2$; continuidad}

\begin{definition}\label{def:b1:multivar:topology}
En $\R^2$, utilice la norma euclidiana $\norm{(x,y)} = \sqrt{x^2 + y^2}$
(\cref{ch:b1:euclid}). Las bolas \omterm{def:b1:topology:open}{Abierto}, \omterm{def:b1:topology:open}{barrios}, \emph{\omterm{def:b1:topology:open}{abierto}
subsets} de $\R^2$ se definen exactamente como en
\cref{ch:b1:topology}, con bolas en lugar de \omterm{prop:b1:reals:intervals}{intervalos}. una función
$f \colon U \to \R$($U \subseteq \R^2$\omterm{def:b1:topology:open}{abierto}) es \emph{\omterm{def:b1:continuity:continuous}{continuo}
at} $a \in U$ cuando
\[
\forall\varepsilon > 0,\ \exists\delta > 0, \quad
\norm{X - a} \leq \delta \implies \abs{f(X) - f(a)} \leq
\varepsilon,
\]
con la misma caracterización secuencial que en una variable.
Sums, products, quotients and compositions with \omterm{def:b1:continuity:continuous}{continuo}
las funciones de una variable conservan \omterm{def:b1:continuity:continuous}{continuidad}; las coordenadas \omterm{def:b1:logic:map}{aplicaciones} son
\omterm{def:b1:continuity:continuous}{continuo}, por lo tanto también lo son \omterm{def:b1:poly:def}{polinomios} en $(x,y)$.
\end{definition}

\begin{example}[El límite polar, la forma limpia de demostrar un límite]
\label{ex:b1:multivar:polarbound}
Muestre que $f(x, y) = \dfrac{x^2y^2}{x^2 + y^2}$(con $f(0,0) =
0$) es \omterm{def:b1:continuity:continuous}{continuo} en el origen. En polar \omterm{prop:b1:vspaces:coordinates}{coordenadas}$x =
\rho\cos\theta $,$ y = \rho\sin\theta$:
\[
\abs{f} = \frac{\rho^4\cos^2\theta\sin^2\theta}{\rho^2}
= \rho^2\,(\cos\theta\sin\theta)^2 \leq \frac{\rho^2}{4}
\xrightarrow[\rho \to 0]{} 0 ,
\]
un atado \emph{independent of $\theta$}: cualquiera que sea la dirección
de enfoque, los valores se comprimen a $0$. Esa uniformidad en
$\theta$ es el punto --- un límite como $\abs g =
\abs{\cos\theta\sin\theta}$(no queda $\rho$) no prueba nada,
y de hecho que $g$ es la trampa radial discontinua del
siguiente ejemplo.
\end{example}

\begin{example}[La trampa radial]\label{ex:b1:multivar:trap}
Sea $f(x, y) = \dfrac{xy}{x^2 + y^2}$ por $(x,y) \neq (0,0)$,$f(0,
0) = 0$. A lo largo de cada eje,$f = 0 \to 0$; sino a lo largo de la diagonal$ y
= x $,$f(x, x) = \frac12 \not\to 0$. \emph{Sin límite en el origen}:
acercándose a lo largo de cada línea, e incluso encontrando el mismo límite a lo largo
cada uno, no es suficiente (aquí los límites de la línea no coinciden; peores ejemplos
están de acuerdo en todos los aspectos pero fracasan en una parábola,
\cref{exo:b1:multivar:3}). \omterm{def:b1:continuity:continuous}{Continuidad} en cada variable por separado
no implica \omterm{def:b1:continuity:continuous}{continuidad}.
\end{example}

\section{Derivadas parciales}

\begin{definition}\label{def:b1:multivar:partial}
El \emph{derivadas parciales}\index{derivada parcial} de $f$ en
$(a, b)$ son los \omterm{def:b1:derivative:def}{derivados} de una variable a lo largo de los ejes:
\[
\frac{\partial f}{\partial x}(a,b)
= \lim_{h \to 0} \frac{f(a + h, b) - f(a,b)}{h},
\qquad
\frac{\partial f}{\partial y}(a,b)
= \lim_{k \to 0} \frac{f(a, b + k) - f(a,b)}{k}.
\]
$f$ es de clase $C^1$ en $U$ cuando ambos existen y son \omterm{def:b1:continuity:continuous}{continuo} en
$U$. El \emph{gradiente}\index{gradiente} es $\nabla f(a,b) =
\bigl(\frac{\partial f}{\partial x},\, \frac{\partial f}{\partial
y}\bigr)(a,b)$.
\end{definition}

\begin{theorem}[$C^1$ implica un plano tangente]\label{thm:b1:multivar:c1}
Sea $f$ $C^1$ en $U$ y $(a,b) \in U$. Entonces, como $(h, k) \to
(0,0)$:
\[
f(a + h, b + k) = f(a, b)
+ h\,\frac{\partial f}{\partial x}(a,b)
+ k\,\frac{\partial f}{\partial y}(a,b)
+ o\bigl(\norm{(h,k)}\bigr) .
\]
En particular $f$ es \omterm{def:b1:continuity:continuous}{continuo}, y el gráfico $z = f(x,y)$ tiene al menos
cada punto del plano tangente lee la fórmula.
\end{theorem}

\begin{example}[Aproximación lineal en acción]
\label{ex:b1:multivar:linapprox}
Calcule $f(1.02,\ 0.99)$ para $f(x, y) = x^3y^2$. En $(1, 1)$:
$f = 1$,$\frac{\partial f}{\partial x} = 3x^2y^2 = 3$,
$\frac{\partial f}{\partial y} = 2x^3y = 2$, entonces
\cref{thm:b1:multivar:c1} da
\[
f(1.02,\ 0.99) \approx 1 + 3\,(0.02) + 2\,(-0.01) = 1.04 ,
\]
contra el valor real $1.02^3 \times 0.99^2 = 1.04006\dots$ ---
el error es de segundo orden en los incrementos, ya que el
$ o(\norm{(h,k)})$ promesas. El plano tangente a la gráfica en
$(1, 1, 1)$ es $z = 1 + 3(x - 1) + 2(y - 1)$, la ecuación
implícito en la estimación.
\end{example}

\begin{proof}
Mueva una coordenada a la vez:
\[
f(a+h, b+k) - f(a,b)
= \bigl[f(a+h, b+k) - f(a, b+k)\bigr] + \bigl[f(a, b+k) -
f(a,b)\bigr].
\]
Por el teorema del valor medio de una variable
(\cref{thm:b1:derivative:mvt}), el primer corchete es $h\,
\frac{\partial f}{\partial x}(a + \theta h,\, b + k)$ para algunos
$\theta \in \intoo{0}{1}$, y el segundo es $k\,\frac{\partial
f}{\partial y}(a,\, b + \theta' k)$. \omterm{def:b1:continuity:continuous}{Continuidad} de los parciales en
$(a,b)$ nos deja escribir cada uno como (valor en $(a,b)$)$+$
(error $\to 0$); el error total es $h\,\varepsilon_1 +
k\,\varepsilon_2 = o(\norm{(h,k)})$ desde $\abs h, \abs k \leq
\norm{(h,k)}$.
\end{proof}

\begin{theorem}[regla de cadena]\label{thm:b1:multivar:chain}
Sea $f$ $C^1$ en $U$ y $t \mapsto (x(t), y(t))$ sea $C^1$ de
un \omterm{prop:b1:reals:intervals}{intervalo} en $U$. Entonces $g(t) = f\bigl(x(t), y(t)\bigr)$ es
$C^1$, con\index{regla de la cadena!dos variables}
\[
g'(t) = x'(t)\,\frac{\partial f}{\partial x}\bigl(x(t),y(t)\bigr)
+ y'(t)\,\frac{\partial f}{\partial y}\bigl(x(t),y(t)\bigr)
= \bigl\langle \nabla f,\ (x', y')\bigr\rangle .
\]
\end{theorem}

\begin{proof}
Aplicar \cref{thm:b1:multivar:c1} en $(x(t), y(t))$ con $(h, k) =
(x(t+s) - x(t),\, y(t+s) - y(t))$: como $s \to 0$, una variable
la diferenciabilidad da $h = s\,x'(t) + o(s)$ y $k = s\,y'(t)
+ o(s)$, por lo que $\norm{(h, k)} = O(s)$ y
\[
g(t+s) - g(t)
= h\,\frac{\partial f}{\partial x} + k\,\frac{\partial
f}{\partial y} + o\bigl(\norm{(h,k)}\bigr)
= s\,\Bigl(x'\,\frac{\partial f}{\partial x} +
y'\,\frac{\partial f}{\partial y}\Bigr) + o(s) ,
\]
el error final absorbe los $ o(s)$ de $h$ y $k$
(multiplicado por los valores fijos de los parciales) y el
$ o(O(s))$ de la estimación del plano tangente. Dividir por $s$ y dejar
$s \to 0$. \omterm{def:b1:continuity:continuous}{Continuidad} de $g'$ se sigue del de todos
ingredientes.
\end{proof}

\begin{example}[La regla de la cadena, verificada en ambos sentidos.]
\label{ex:b1:multivar:chainrun}
Sean $f(x, y) = x^2 y $ y $g(t) = f(t, t^2)$. Directamente:$g(t) =
t^2\cdot t^2 = t^4$, entonces $g'(t) = 4t^3$. Por la regla de la cadena:
$\frac{\partial f}{\partial x} = 2xy$ y $\frac{\partial
f}{\partial y} = x^2$, evaluados a lo largo de la curva $(t, t^2)$:
\[
g'(t) = 1\cdot 2t\cdot t^2 + 2t\cdot t^2 = 2t^3 + 2t^3 = 4t^3 .
\]
Los dos cálculos concuerdan y la división es significativa: $2t^3$
del crecimiento proviene de mover \emph{a la derecha} a través del
$x$-pendiente,$2t^3$ de mover \emph{hacia arriba} a través del
$ y $-pendiente. En curvas donde no existe una forma cerrada para $g$, solo
el segundo cálculo sobrevive: ese es el punto del teorema.
\end{example}

\begin{remark}[Leyendo el gradiente]
A lo largo de una dirección unitaria $u$, la regla de la cadena en $t \mapsto f(a + tu)$
da el \emph{directional \omterm{def:b1:derivative:def}{derivado}} $\langle \nabla f(a),
u\rangle $: máximo cuando $u $apunta a lo largo de $\nabla f(a)$(por
Cauchy--Schwarz, \cref{thm:b1:euclid:cs}). El \omterm{def:b1:multivar:partial}{gradiente} es el
dirección del ascenso más pronunciado, y es \omterm{def:b1:euclid:orthogonal}{ortogonal} hasta el nivel
curvas $\{f = c\}$(diferenciar $f$ a lo largo de una curva dibujada en un
nivel \omterm{def:b1:logic:sets}{conjunto}: la regla de la cadena da $\langle\nabla f,\,
\text{tangent}\rangle = 0$).
\end{remark}

\begin{example}[Curvas de nivel y gradientes, en una sola función]
\label{ex:b1:multivar:levelcurves}
Tome $f(x, y) = x^2 - y^2$. Su nivel \omterm{def:b1:logic:sets}{conjuntos}:$\{f = c\}$ es un
hipérbola que se abre de izquierda a derecha para $c > 0$, de arriba a abajo para $c < 0$,
y el par de líneas cruzadas $ y = \pm x$ para $c = 0$ --- el
contorno \omterm{def:b1:logic:map}{aplicación} de un puerto de montaña, con el \omterm{met:b1:multivar:monge}{punto de silla} en el
origen donde se cruzan las dos líneas de nivel cero. \omterm{def:b1:multivar:partial}{Gradiente}: $\nabla
f = (2x, -2y)$. En el punto $(2, 1)$(en el nivel $c = 3$):
$\nabla f = (4, -2)$, mientras que el vector tangente del nivel
curva, parametrizada cerca de ese punto por $\bigl(t,
\sqrt{t^2 - 3}\bigr)$, es $\bigl(1, \frac{t}{\sqrt{t^2 -
3}}\bigr) = (1, 2)$ en $ t = 2$ --- y de hecho
\[
\langle (4, -2),\ (1, 2)\rangle = 4 - 4 = 0 :
\]
\omterm{def:b1:multivar:partial}{gradiente} perpendicular al contorno, apuntando hacia arriba
valores de $f $(aquí: lejos del eje$ y $). dos mas
lecturas: el \omterm{def:b1:multivar:partial}{gradiente} desaparece exactamente en el sillín, donde
el contorno \omterm{def:b1:logic:map}{aplicación} pellizca; y la tangente \emph{línea} a la
la curva de nivel en $(2,1)$ es $4(x - 2) - 2(y - 1) = 0$, es decir $2x -

\end{example}

\begin{theorem}[negro]\label{thm:b1:multivar:schwarz}
Si $f$ es de clase $C^2$(los parciales de parciales existen y son
\omterm{def:b1:continuity:continuous}{continuo}), luego\index{teorema de schwarz}
\[
\frac{\partial^2 f}{\partial x\,\partial y}
= \frac{\partial^2 f}{\partial y\,\partial x} .
\]
\end{theorem}
\admitted

\section{Extremos locales}

\begin{method}[Estudios extremos, organizados]
\label{met:b1:multivar:extremamethod}
\begin{enumerate}
 \item \emph{Solve $\nabla f = 0$ completely.} Factorizar cada uno
 parcial siempre que sea posible (productos de factores \omterm{def:b1:linmaps:def}{lineal}
 dividir el sistema en casos transparentes, como en
 \cref{ex:b1:multivar:fourpoints} a continuación); un olvidado
 El caso es un \omterm{thm:b1:multivar:critical}{punto crítico} olvidado.
 \item \emph{Clasifica cada punto} con los datos de Monge $r, s,
 t $ --- recalculado \emph{en cada punto}, nunca una vez por
 todos.
 \item \emph{If $rt - s^2 = 0$}, examine $f$ directamente a lo largo
 curvas bien elegidas que pasan por el punto (líneas primero, luego
 parábolas), buscando dos signos (sin extremo)
 o para un letrero cerrado con un argumento que cubra todos
 direcciones.
 \item \emph{Un paso atrás para la imagen global}: comprobar el
 comportamiento en el infinito (un mínimo local puede no ser global
 uno), y si el dominio no es \omterm{def:b1:topology:open}{abierto}, trate su \omterm{def:b1:topology:closure}{límite}
 por separado (\cref{exo:b1:multivar:12}) --- el teorema \omterm{thm:b1:multivar:critical}{crítico
 punto} solo ve puntos \omterm{def:b1:topology:closure}{interior}.
\end{enumerate}
\end{method}

\begin{theorem}[Puntos críticos]\label{thm:b1:multivar:critical}
Si $f $($C^1$ en \omterm{def:b1:topology:open}{conjunto abierto}$U$) tiene un extremo local en $(a,b)
\in U $, entonces $\nabla f(a,b) = (0,0)$: el punto es
\emph{crítico}\index{punto crítico}.
\end{theorem}

\begin{proof}
Las funciones de una variable $x \mapsto f(x, b)$ y $ y \mapsto f(a,
y)$ tienen extremos locales \omterm{def:b1:topology:closure}{interior} en $a $, resp.$b$:
\cref{prop:b1:derivative:fermat} mata ambos parciales.
\end{proof}

\begin{method}[Prueba de segundo orden (notación Monge)]\label{met:b1:multivar:monge}
En un \omterm{thm:b1:multivar:critical}{punto crítico} de una función $C^2$, configure
\[
r = \frac{\partial^2 f}{\partial x^2},
\qquad
s = \frac{\partial^2 f}{\partial x \partial y},
\qquad
t = \frac{\partial^2 f}{\partial y^2}
\qquad (\text{values at the point}).
\]
\begin{itemize}
 \item Si $rt - s^2 > 0$: extremo local --- mínimo para $r > 0$,
 máximo para $r < 0$;
 \item si $rt - s^2 < 0$: sin extremo (a \emph{silla de montar
 punto}\index{punto de silla});
 \item si $rt - s^2 = 0$: la prueba es silenciosa; examinar directamente.
\end{itemize}
(Justificación: una expansión de Taylor-Young en el orden $2$ y la
estudio de signos de la forma cuadrática $r h^2 + 2shk + tk^2$ --- es
realizado en el segundo año; La prueba se utiliza aquí como prueba de trabajo.
herramienta.)
\end{method}

\begin{example}\label{ex:b1:multivar:extrema}
$f(x,y) = x^3 + y^3 - 3xy$. \omterm{thm:b1:multivar:critical}{Puntos críticos}: $\nabla f = (3x^2 -
3y,\; 3y^2 - 3x) = 0$ da $ y = x^2$ y $x = y^2$, entonces $x = x^4$:
$x \in \{0, 1\}$: puntos $(0,0)$ y $(1,1)$.

Segundo \omterm{def:b1:derivative:def}{derivados}: $r = 6x$,$s = -3$,$t = 6y$. En $(0,0)$:$rt -
s^2 = -9 < 0$: silla de montar. En $(1,1)$:$rt - s^2 = 36 - 9 > 0$,$ r = 6
> 0$: mínimo local,$f(1,1) = -1$. (No global:$f(x, 0) = x^3 \to
-\infty $).
\end{example}

\begin{example}[Un estudio de cuatro puntos, completo.]
\label{ex:b1:multivar:fourpoints}
$f(x, y) = xy\,(3 - x - y) = 3xy - x^2y - xy^2$. \omterm{def:b1:multivar:partial}{Gradiente}:
\[
\frac{\partial f}{\partial x} = y\,(3 - 2x - y),
\qquad
\frac{\partial f}{\partial y} = x\,(3 - x - 2y).
\]
\omterm{thm:b1:multivar:critical}{Puntos críticos}: si $ y = 0$, la segunda ecuación da $x \in
\{0, 3\}$; si $x = 0$, el primero da $ y \in \{0, 3\}$; si $xy
\neq 0$, resuelva $2x + y = 3$,$x + 2y = 3$:$x = y = 1$. cuatro
puntos: $(0,0)$,$(3,0)$,$(0,3)$,$(1,1)$. Segundo \omterm{def:b1:derivative:def}{derivados}:
$r = -2y$,$s = 3 - 2x - 2y$,$t = -2x$.
\begin{itemize}
 \item $(1,1)$:$r = -2$,$s = -1$,$t = -2$:$rt - s^2 = 3 >
 0$,$r < 0$: máximo local,$f(1,1) = 1$.
 \item $(0,0)$:$r = t = 0$,$s = 3$:$rt - s^2 = -9 < 0$:
 silla de montar; así mismo $(3, 0)$($s = -3$) y $(0, 3)$: tres
 sillas de montar.
\end{itemize}
El máximo es sólo local: $f(-T, -T) = T^2(3 + 2T) \to
+\infty $. Verificación de simetría:$f(x, y) = f(y, x)$, y de hecho el
crítico \omterm{def:b1:logic:sets}{conjunto} y la clasificación son simétricas en $x
\leftrightarrow y $. Interpretación: entre rectángulos con holgura
$x, y \geq 0$,$x + y \leq 3$, el producto$ xy(3 - x - y)$ de
las tres "partes" de $3$ son mayores cuando las partes son iguales
--- una sombra de dos variables del principio AM--GM.
\end{example}

\begin{remark}[Errores comunes]
\label{rem:b1:multivar:pitfalls}
\emph{\omterm{def:b1:multivar:partial}{Derivadas parciales} may exist at a point of
discontinuity}: la trampa radial $g(x,y) = \frac{xy}{x^2+y^2}$ de
\cref{ex:b1:multivar:trap} tiene $\frac{\partial g}{\partial
x}(0,0) = \frac{\partial g}{\partial y}(0,0) = 0$(ambos ejes
las restricciones desaparecen de manera idéntica), sin embargo, $g$ no tiene límite en el
origen --- sonda parcial sólo en dos direcciones, \omterm{def:b1:continuity:continuous}{continuidad} necesita
todos ellos; sólo la hipótesis $C^1$ restablece el orden
(\cref{thm:b1:multivar:c1}).
\emph{Los límites de línea nunca son suficientes}: \cref{exo:b1:multivar:3}
la función tiene un límite $0$ en cada línea y aún no tiene límite ---
Siempre intente con parábolas (o límites polares válidos uniformemente en
$\theta$).
\emph{Lo crítico es necesario, no suficiente}: abundan las sillas de montar
(tres de cuatro puntos en \cref{ex:b1:multivar:fourpoints}); y
el teorema se cumple solo en \emph{\omterm{def:b1:topology:open}{abierto}} \omterm{def:b1:logic:sets}{conjuntos} --- en un disco \omterm{def:b1:topology:closed}{cerrado},
los extremos pueden sentarse en el \omterm{def:b1:topology:closure}{límite} con \omterm{def:b1:multivar:partial}{gradiente} distinto de cero
(\cref{exo:b1:multivar:12}).
\emph{El caso silencioso $rt - s^2 = 0$ es genuinamente silencioso}:$x^4
+ y^4$(mínimo) y $x^3 + y^3$(ninguno) tienen $r = s = t
= 0$ en el origen; sólo un estudio de signos directos decide
(\cref{exo:b1:multivar:6}, función $h$).
\emph{The \omterm{def:b1:multivar:partial}{gradiente} is \omterm{def:b1:euclid:orthogonal}{ortogonal} to level curves, not along
them}: para seguir una línea de contorno, muévase perpendicular a $\nabla
f $; para subir más rápido, muévete a lo largo de él --- mezclando los dos reversos
la geometría de cada contorno \omterm{def:b1:logic:map}{aplicación}.
\end{remark}

\begin{remark}[Adónde conducen dos variables]
\label{rem:b1:multivar:whereused}
Este capítulo es una puerta. El \omterm{def:b1:multivar:partial}{gradiente} y la regla de la cadena se extienden
palabra por palabra a las variables $n$ en el volumen Año~2, donde
$ o(\norm{(h,k)})$ de \cref{thm:b1:multivar:c1} se convierte en el
\emph{diferencial} y la prueba de Monge se demuestra en su totalidad mediante el
Fórmula de Taylor de orden dos y formas cuadráticas. El caso especial
que se puede resolver \emph{este} año --- funciones cuadráticas,
para el cual la expansión de segundo orden es exacta --- es el tema
del problema del fin de semana, y resulta que es el caso que se ejecuta
Ajuste de datos del mundo: regresión \omterm{pb:b1:multivar:1}{mínimos cuadrados}. restringido
extrema (\cref{exo:b1:multivar:5} fue un adelanto) se convierte en Lagrange
multiplicadores en el año ~2; funciones armónicas
(\cref{exo:b1:multivar:7}) regresa en el complejo análisis de la
Año ~ 3 volumen.
\end{remark}

\begin{exercise}[$\star$]\label{exo:b1:multivar:1}
Calcule el \omterm{def:b1:multivar:partial}{derivadas parciales}:
$f(x,y) = x^2 y + \eu^{xy}$;$\;g(x,y) = \ln(x^2 + y^2)$(en
$\R^2\setminus\{0\}$);$\;h(x,y) = \arctan\frac yx $(en $x > 0$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:multivar:2}
Estudie el \omterm{def:b1:continuity:continuous}{continuidad} en $(0,0)$(con el valor $0$ allí) de:
\[
f(x,y) = \frac{x^2 y}{x^2 + y^2},
\qquad
g(x,y) = \frac{xy}{x^2 + y^2},
\qquad
h(x,y) = \frac{x^3 + y^3}{x^2 + y^2}.
\]
\emph{(Polar \omterm{prop:b1:vspaces:coordinates}{coordenadas} $x = \rho\cos\theta$, $ y = \rho\sin\theta$
help: bound by a function of $\rho$ alone when possible.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:3}
Sea $f(x,y) = \dfrac{x^2 y}{x^4 + y^2}$($f(0,0) = 0$). demostrar que
$f$ tiene el límite $0$ en el origen \emph{a lo largo de cada línea recta},
pero que $f\bigl(x, x^2\bigr) = \frac12$:$f$ no es \omterm{def:b1:continuity:continuous}{continuo} en
$(0,0)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:multivar:4}
Verifique el teorema de Schwarz manualmente en $f(x, y) = x^3 y^2 + \sin(xy)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:5}
Sea $f$ $C^1$ en $\R^2$ y $g(t) = f(\cos t, \sin t)$. expreso
$g'(t)$ mediante la regla de la cadena. Deducir que $f$ restringido a la unidad
círculo alcanza extremos en puntos donde $\nabla f$ es paralelo a
el vector de radio.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:6}
Encuentra y clasifica el \omterm{thm:b1:multivar:critical}{puntos críticos} de:
\[
f(x, y) = x^2 + xy + y^2 - 3x,
\qquad
g(x, y) = x^2 - y^2 + 4y,
\qquad
h(x, y) = x^4 + y^4 - 2(x - y)^2 .
\]
\emph{(Para $h$, la prueba del determinante no dice nada en el origen:
examine $h(x, x)$ y $h(x, -x)$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:7}
Una función $f$ es \emph{armónico} cuando $\frac{\partial^2 f}{\partial
x^2} + \frac{\partial^2 f}{\partial y^2} = 0$. Compruebe que $x^2 -
y^2$,$xy$, $\eu^x\cos y $ y $\ln(x^2 + y^2)$(fuera del origen) estén
armónico.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:multivar:8}
Encuentra el punto del avión $\{x + 2y - z = 4\} \subseteq \R^3$
más cercano al origen, de dos maneras: por \omterm{thm:b1:euclid:projection}{proyección ortogonal}
(\cref{ch:b1:euclid}), y minimizando la función de dos variables
$f(x, y) = x^2 + y^2 + (x + 2y - 4)^2$ obtenido eliminando $z$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:multivar:9}
(Laplaciano en polar \omterm{prop:b1:vspaces:coordinates}{coordenadas}, primer contacto) Sea $f$ $C^2$ en
$\R^2 \setminus \{0\}$ y \emph{radial}:$f(x, y) =
\varphi\bigl(\sqrt{x^2+y^2}\bigr)$ con $\varphi$ de clase $C^2$ en
$\intoo{0}{+\infty}$. demostrar que
\[
\frac{\partial^2 f}{\partial x^2} + \frac{\partial^2 f}{\partial
y^2} = \varphi''(\rho) + \frac{\varphi'(\rho)}{\rho},
\qquad \rho = \sqrt{x^2 + y^2},
\]
y encuentre todas las funciones armónicas radiales en el plano perforado.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:10}
Dé la ecuación del plano tangente a la gráfica $z = xy$ en
el punto $(1, 1, 1)$. Luego encuentre todos los puntos de la gráfica de
$f(x,y) = x^3 + y^3 - 3xy$ donde está el plano tangente
\emph{horizontal}, y relacionar la respuesta con
\cref{ex:b1:multivar:extrema}.
\end{exercise}

 \item Encuentra todas las soluciones $C^1$ de la ecuación.
 $\frac{\partial f}{\partial x} = \frac{\partial

\section{Problema: mínimos cuadrados y línea de regresión}

\begin{problem}\label{pb:b1:multivar:1}
Dados los puntos de datos $n$, ¿qué línea recta pasa "más cerca" de
todos ellos? Legendre y Gauss respondieron: la línea que minimiza la
suma de los errores verticales \emph{al cuadrado} --- porque eso
la minimización se puede resolver exactamente mediante álgebra lineal. este problema
prueba por primera vez la prueba de Monge de \cref{met:b1:multivar:monge}
honestamente para funciones cuadráticas (el único caso donde el
la expansión de segundo orden es exacta), luego construye \omterm{pb:b1:multivar:1}{normales
ecuaciones}, \omterm{pb:b1:multivar:1}{línea de regresión} y el coeficiente de correlación
encima de la geometría euclidiana de
\cref{ch:b1:euclid}.\index{mínimos cuadrados}\index{línea de regresión}
\index{ecuaciones normales}\index{correlación}

\medskip
\noindent\textbf{Parte I --- Quadratic functions: the Monge test,
proved.} Arregla los reales $r, s, t$ y deja que $q(h, k) = r h^2 + 2s hk +
t k^2$.
\begin{enumerate}
 \item Supongamos $r \neq 0$. Establecer la forma del cuadrado completo.
 \[
 q(h, k) = r\Bigl(h + \frac{s}{r}k\Bigr)^{\!2}
 + \frac{rt - s^2}{r}\,k^2 ,
 \]
 y deducir: si $rt - s^2 > 0$,$q$ tiene el signo estricto
 de $r$ fuera del origen; si $rt - s^2 < 0$,$q$ toma ambos
 señales.
 \item Resolver los casos restantes: $r = 0$,$t \neq 0$
 (simétrico); y $r = t = 0$($q = 2shk$). Concluir:
 $q$ toma ambos signos si $rt - s^2 < 0$ y $q$
 desaparece sólo en el origen si $rt - s^2 > 0$.
 \item Sea ahora $f(x, y) = \frac12\bigl(r x^2 + 2s xy + t
 y^2\bigr) + \beta x + \gamma y + \delta $ una cuadrática
 funcionar con un \omterm{thm:b1:multivar:critical}{punto crítico} $X_0 = (a, b)$. demostrar el
 \emph{exacto} expansión
 \[
 f(X_0 + (h,k)) = f(X_0) + \tfrac12\,q(h, k)
 \qquad (\text{no remainder}),
 \]
 y deducir la clasificación de Monge para cuadrática
 funciones: mínimo global estricto si $rt - s^2 > 0$,$r >
 0$; máximo global estricto si $rt - s^2 > 0$,$r < 0$;
 sillín si $rt - s^2 < 0$.
 \item Supongamos $rt - s^2 > 0$ y $r > 0$(también $t > 0$).
 Demostrar la coercitividad explícita ligada
 \[
 q(h, k) \geq c\,(h^2 + k^2),
 \qquad
 c = \frac{rt - s^2}{r + t} > 0 .
 \]
 \emph{(Demuestre que la forma cuadrática $q - c(h^2 + k^2)$,
 con coeficientes $r - c$,$s$,$t - c$, todavía tiene
 datos de pruebas discriminantes no negativas.)}
 \item Deducir que una función cuadrática con definida positiva
 la parte cuadrática tiende a $+\infty$ como $\norm{(x,y)} \to
 \infty$ y, por lo tanto, tiene un \emph{único global}
 minimizador: es \omterm{thm:b1:multivar:critical}{punto crítico}. Contraste con
 \cref{ex:b1:multivar:extrema}, donde un mínimo local de
 una función no cuadrática no era global.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The \omterm{pb:b1:multivar:1}{ecuaciones normales}.} Sea $C_1,
C_2, b \in \R^n $(canónico \omterm{def:b1:euclid:def}{producto interno}), y
\[
f(u, v) = \norm{u\,C_1 + v\,C_2 - b}^2 .
\]
\begin{enumerate}[resume]
 \item Expanda $f$ y muestre que es una función cuadrática de $(u,
 v)$ con coeficientes de parte cuadrática $r = 2\norm{C_1}^2$,
 $s = 2\langle C_1, C_2\rangle$, $t = 2\norm{C_2}^2$.
 \item Demuestre que $rt - s^2 > 0$ si y sólo si $(C_1, C_2)$ es
 \omterm{def:b1:vspaces:free}{libre}\emph{(the equality case of Cauchy--Schwarz,
 \cref{thm:b1:euclid:cs})}; asuma esto de ahora en adelante.
 \item Demuestre que las ecuaciones críticas $\nabla f = 0$ son las
 \emph{\omterm{pb:b1:multivar:1}{ecuaciones normales}}
 \[
 \begin{pmatrix}
 \norm{C_1}^2 & \langle C_1, C_2\rangle\\
 \langle C_1, C_2\rangle & \norm{C_2}^2
 \end{pmatrix}
 \begin{pmatrix} u\\ v\end{pmatrix}
 =
 \begin{pmatrix} \langle C_1, b\rangle\\ \langle C_2,
 b\rangle\end{pmatrix}
 \]
 --- la matriz de Gram de \cref{exo:b1:euclid:11} en el
 izquierda --- y que dicen exactamente: $b - (uC_1 + vC_2)
 \perp C_1, C_2$. Concluir con la Parte I y
 \cref{thm:b1:euclid:projection}: el minimizador único
 le da a $p = uC_1 + vC_2 =$ el \omterm{thm:b1:euclid:projection}{proyección ortogonal} de
 $b$ sobre $\operatorname{Vect}(C_1, C_2)$.
 \item Demuestre que el valor mínimo es $\norm{b}^2 - \langle p,
 b\rangle = \norm{b - p}^2$ y dibuje el pitagórico.
 imagen.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The \omterm{pb:b1:multivar:1}{línea de regresión}.} Puntos de datos
$(x_1, y_1), \dots, (x_n, y_n)$, no todos los $x_i$ son iguales. minimizar
\[
E(\alpha, \beta) = \sum_{i=1}^{n}\bigl(y_i - \alpha - \beta
x_i\bigr)^2 .
\]
Escriba $\bar x = \frac1n\sum x_i$,$\bar y = \frac1n\sum y_i$,
$v_x = \frac1n\sum x_i^2 - \bar x^2$,$v_y = \frac1n\sum y_i^2 -
\bar y^2$,$c_{xy} = \frac1n\sum x_i y_i - \bar x\bar y $.
\begin{enumerate}[resume]
 \item Reconocer la Parte II con $C_1 = (1, \dots, 1)$,$C_2 =
 (x_1, \dots, x_n)$,$b = (y_1, \dots, y_n)$ y escribir
 el \omterm{pb:b1:multivar:1}{ecuaciones normales}
 \[
 n\alpha + \Bigl(\sum x_i\Bigr)\beta = \sum y_i,
 \qquad
 \Bigl(\sum x_i\Bigr)\alpha + \Bigl(\sum
 x_i^2\Bigr)\beta = \sum x_i y_i .
 \]
 \item Resolverlos:
 \[
 \beta = \frac{c_{xy}}{v_x},
 \qquad
 \alpha = \bar y - \beta\,\bar x ,
 \]
 y observar que el \emph{\omterm{pb:b1:multivar:1}{línea de regresión}} $ y = \alpha
 + \beta x $pasa por el punto medio $(\bar x, \bar
 y)$.
 \item Verifique $v_x > 0$ exactamente cuando los $x_i$ no estén todos
 igual, y relacione esto con la pregunta 7.
 \item Demuestre que el error mínimo es
 \[
 E_{\min} = n\Bigl(v_y - \frac{c_{xy}^2}{v_x}\Bigr)
 = n\,v_y\,(1 - \rho^2),
 \qquad
 \rho = \frac{c_{xy}}{\sqrt{v_x v_y}}
 \quad (v_y \neq 0).
 \]
 Deducir $\abs\rho \leq 1$, con $\abs\rho = 1$ si y
 sólo si los datos están perfectamente alineados.
 \item Ejemplo resuelto: para los datos $(0,0)$,$(1,1)$,$(2,3)$,
 $(3,4)$, calcular $\bar x, \bar y, v_x, c_{xy}$, el
 \omterm{pb:b1:multivar:1}{línea de regresión}, los cuatro residuales y $E_{\min}$.
 \item Demuestre que los residuos $\varepsilon_i = y_i - \alpha -
 \beta x_i$ de la línea óptima satisfacen $\sum_i
 \varepsilon_i = 0$ y $\sum_i x_i\varepsilon_i = 0$,
 e interpretar ambos vía ortogonalidad.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Variations and applications.}
\begin{enumerate}[resume]
 \item (Mejor constante) Demuestre que la constante $\alpha$
 minimizar $\sum_i (y_i - \alpha)^2$ es la media $\bar
 y $, y que el valor mínimo es $n\,v_y $: la varianza
 Mide la falla de los datos para ser constantes.
 \item (Línea que pasa por el origen) Demuestre que la pendiente que minimiza
 $\sum_i (y_i - \beta x_i)^2$ es $\beta_0 = \frac{\sum
 x_iy_i}{\sum x_i^2}$ y proporciona una condición sobre los datos.
 para que $\beta_0$ coincida con la pendiente
 $c_{xy}/v_x$ de la pregunta 11.
 \item (Distancia a una línea, nuevamente) Para un punto $M$ y la línea
 $D$a$P_0$ dirigido por el vector unitario $w$,
 minimizar $g(\tau) = \norm{P_0 + \tau w - M}^2$ y
 deducir $d(M, D)^2 = \norm{M - P_0}^2 - \langle M - P_0,
 w\rangle^2$; recupera la fórmula $d = \frac{\abs{a x_0 +
 b y_0 + c}}{\sqrt{a^2 + b^2}}$ para la línea$ ax + by + c
 = 0$ en el plano.
 \item (Análisis de varianza) Demuestre la descomposición $v_y =
 \beta^2 v_x + \frac{E_{\min}}n $: la varianza de la
 $y_i$ se divide en la parte explicada por la línea más
 la varianza residual.
 \item (Ajuste parabólico) Para ajustar $ y = a + bx + cx^2$, demuestre que
 el \omterm{pb:b1:multivar:1}{ecuaciones normales} es el sistema $3 \times 3$ con
 la matriz de momento
 \[
 \begin{pmatrix}
 n & \sum x_i & \sum x_i^2\\
 \sum x_i & \sum x_i^2 & \sum x_i^3\\
 \sum x_i^2 & \sum x_i^3 & \sum x_i^4
 \end{pmatrix},
 \]
 una matriz de Gram que es invertible tan pronto como tres de
 los $x_i$ son distintos \emph{(freeness of $(1, X, X^2)$
 sampled at the data, \cref{exo:b1:euclid:11}; compare
 the moment matrices of the problema de fin de semana of
 \cref{ch:b1:det})}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Robustness, and synthesis.}
\begin{enumerate}[resume]
 \item Clasifica el \omterm{thm:b1:multivar:critical}{puntos críticos} de los tres cuadráticos
 funciones
 \[
 x^2 + xy + y^2, \qquad x^2 + 3xy + y^2, \qquad
 x^2 + 2xy + y^2 ,
 \]
 por la Parte I, tratando el tercer caso degenerado a mano
 (¿dónde se alcanza el mínimo?).
 \item (Experimento de valores atípicos) Agregue el punto $(10, 0)$ al
 datos de la pregunta 14 y recalcular la pendiente $\beta$.
 ¿Qué sucedió y por qué el error \emph{al cuadrado} es tan grande?
 sensible a un punto distante?
 \item (Ponderado \omterm{pb:b1:multivar:1}{mínimos cuadrados}) Dados los pesos $w_i > 0$,
 minimizar $\sum_i w_i(y_i - \alpha - \beta x_i)^2$: mostrar
 la solución viene dada por las mismas fórmulas que la pregunta
 11 con medias ponderadas, varianza y covarianza
 (definirlos).
 \item Demuestre que para la línea óptima, $E_{\min} = 0$ fuerza
 todos los puntos se encuentren exactamente sobre él y se conecten con el
 caso de igualdad $\abs\rho = 1$ de la pregunta 13; prueba en el
 datos alineados $(1,1), (2,3), (3,5)$.
 \item Síntesis, en cuatro frases: por qué funciones cuadráticas
 son la única clase donde el segundo orden de este capítulo
 la prueba no necesita teorema admitido; cómo el \omterm{pb:b1:multivar:1}{ecuaciones normales}
 identificar la minimización analítica con el \omterm{thm:b1:euclid:projection}{ortogonal
 proyección} de \cref{ch:b1:euclid}; cuál es la correlación
 medidas de coeficientes y qué desigualdad lo limita;
 y qué partes de los Capítulos 18-23 (bases, Gram y
 matrices de momento, \omterm{def:b1:linmaps:projection}{proyecciones}) reaparecieron. Nombra el
 método y las ecuaciones estudiadas en las Partes II--III.
\end{enumerate}
\end{problem}
