\chapter{Integración en un segmento}\label{ch:b1:integration}

El \omterm{thm:b1:integration:def}{integral} del volumen High School se fundó en áreas tomadas
intuitivamente. Este capítulo lo construye: primero para \omterm{def:b1:integration:step}{funciones escalonadas},
donde \omterm{thm:b1:integration:def}{integral} es una suma finita, entonces para \omterm{def:b1:continuity:continuous}{continuo} (y
por partes \omterm{def:b1:continuity:continuous}{continuo}) funciona por aproximación uniforme --- el
lugar donde gana el teorema de Heine (\cref{thm:b1:continuity:heine})
es mantener. El teorema fundamental del cálculo conecta entonces el
construcción a primitivas, y \omterm{thm:b1:integration:riemann}{sumas de riemann} conectarlo a discreto
promedios.

En todo momento, $a < b$ son reales.

\section{Funciones de paso}

\begin{definition}\label{def:b1:integration:step}
$\varphi \colon \intcc{a}{b} \to \R$ es un \emph{paso
función}\index{función de paso} cuando existe una subdivisión $a = x_0
< x_1 < \dots < x_n = b $tal que $\varphi$ es constante, igual a
$c_i$, en cada \omterm{def:b1:topology:open}{abierto} \omterm{prop:b1:reals:intervals}{intervalo}$\intoo{x_{i-1}}{x_i}$(valores en el
los nodos no están restringidos). Su \omterm{thm:b1:integration:def}{integral} es
\[
\int_a^b \varphi = \sum_{i=1}^{n} c_i\,(x_i - x_{i-1}),
\]
independiente de la subdivisión elegida (refinar dos subdivisiones por
su común: cada lado permanece sin cambios bajo refinamiento).
\end{definition}

\begin{proposition}\label{prop:b1:integration:stepprops}
En \omterm{def:b1:integration:step}{funciones escalonadas}, el \omterm{thm:b1:integration:def}{integral} es lineal, creciente ($\varphi \leq
\psi \implies \int\varphi \leq \int\psi $) y satisface el Chasles
relación $\int_a^b = \int_a^c + \int_c^b$ para $a < c < b$.
\end{proposition}

\begin{proof}
El motor es \emph{invariancia de refinamiento}, indicado en el
definición: insertar un nodo adicional $t \in \intoo{x_{i-1}}{x_i}$
en una subdivisión reemplaza el término $c_i(x_i - x_{i-1})$ por
$c_i(t - x_{i-1}) + c_i(x_i - t)$ --- el mismo número --- entonces el
\omterm{thm:b1:integration:def}{integral} no se modifica mediante ningún refinamiento finito. Ahora toma
$\varphi$ con subdivisión $\sigma$ y $\psi$ con subdivisión
$\sigma'$: sobre el refinamiento común $\sigma \cup \sigma'$ ambos
son \omterm{def:b1:integration:step}{funciones escalonadas} con los nodos \emph{mismo}, y en cada pieza
$\varphi + \lambda\psi$ es constante igual a $c_i + \lambda
d_i $: la linealidad se reduce a linealidad de sumas finitas. Incrementar:
$c_i \leq d_i$ en cada pieza da $\sum c_i \Delta_i \leq \sum
d_i \Delta_i $(longitudes $\Delta_i \geq 0$). Chasles: insertar el
nodo $c$ y divida la suma en él.
\end{proof}

\section{Integral de una función continua}

\begin{theorem}[aproximación uniforme]\label{thm:b1:integration:approx}
Sea $f$\omterm{def:b1:continuity:continuous}{continuo} en $\intcc{a}{b}$. Por cada $\varepsilon > 0$
hay \omterm{def:b1:integration:step}{funciones escalonadas} $\varphi, \psi$ con
\[
\varphi \leq f \leq \psi
\qquad\text{and}\qquad
\psi - \varphi \leq \varepsilon \text{ on } \intcc{a}{b}.
\]
\end{theorem}

\begin{proof}
Según el teorema de Heine (\cref{thm:b1:continuity:heine}), $f$ es \omterm{def:b1:continuity:uniform}{uniformemente
continuo}: elija $\delta$ para $\varepsilon$, y una subdivisión de
malla $< \delta $(equivalentemente espaciadas, digamos, con piezas $n > \frac{b -
a}{\delta}$). En cada pieza \omterm{def:b1:topology:closed}{cerrado}$\intcc{x_{i-1}}{x_i}$,$f$
alcanza un mínimo $m_i$ y un máximo $M_i$
(\cref{thm:b1:continuity:evt}) y $M_i - m_i \leq \varepsilon$
(los dos puntos extremos están dentro de $\delta$). Definir $\varphi = m_i$
y $\psi = M_i$ en $\intoo{x_{i-1}}{x_i}$(y $\varphi = \psi = f$
en los nodos).
\end{proof}

\begin{theorem}[Definición de integral]\label{thm:b1:integration:def}
Sea $f$\omterm{def:b1:continuity:continuous}{continuo} en $\intcc{a}{b}$. los dos numeros
\[
I_-(f) = \sup\Bigl\{\int_a^b \varphi : \varphi \text{ step},\
\varphi \leq f\Bigr\},
\qquad
I_+(f) = \inf\Bigl\{\int_a^b \psi : \psi \text{ step},\ \psi \geq
f\Bigr\}
\]
son iguales; su valor común es el
\emph{integral}\index{integral} $\int_a^b f $(también escrito $\int_a^b
f(t)\,\dd t$). Coincide con la noción anterior en \omterm{def:b1:integration:step}{paso
funciones} y se extiende a las funciones \omterm{def:b1:continuity:continuous}{continuo} por partes mediante
dividiendo $\intcc{a}{b}$ en las discontinuidades (Chasles como
definición allí).
\end{theorem}

\begin{proof}
Ambos \omterm{def:b1:logic:sets}{conjuntos} no están vacíos ($f$ está limitado) y cada paso inferior \omterm{thm:b1:integration:def}{integral}
es $\leq$ cada uno superior (monotonicidad en \omterm{def:b1:integration:step}{funciones escalonadas}): entonces
$I_-(f) \leq I_+(f)$. Por \cref{thm:b1:integration:approx}, por cada
$\varepsilon$ hay un par con $\int\psi - \int\varphi \leq
\varepsilon(b - a)$: el sup y el inf están juntos,$I_- =
I_+$.
\end{proof}

\begin{example}[Continuo por partes, sin drama]\label{ex:b1:integration:piecewise}
El \omterm{thm:b1:reals:floor}{función de piso} en $\intcc{0}{3}$ es un \omterm{def:b1:integration:step}{función de paso} en
disfraz: partiéndose en sus saltos,
\[
\int_0^3 \lfloor t \rfloor\,\dd t
= \int_0^1 0 + \int_1^2 1 + \int_2^3 2 = 0 + 1 + 2 = 3 ,
\]
y los valores \emph{at} los puntos de salto $1, 2$ son irrelevantes:
cambiar una función en un número finito de puntos no cambia \omterm{thm:b1:integration:def}{integral}
(el encuadre \omterm{def:b1:integration:step}{funciones escalonadas} no se ve afectado). este es el todo
contenido de la extensión ``por trozos \omterm{def:b1:continuity:continuous}{continuo}'': cortado en la
un número finito de discontinuidades, integra cada pieza \omterm{def:b1:continuity:continuous}{continuo},
agregue --- Chasles como definición.
\end{example}

\begin{example}[La definición se calcula una vez.]\label{ex:b1:integration:defcompute}
Sea $f(x) = x$ sobre $\intcc{0}{1}$ y córtelo en $n$ en trozos iguales.
La mejor constante \omterm{def:b1:integration:step}{funciones escalonadas} en las piezas es $\varphi =
\frac{k-1}{n}$ y $\psi = \frac kn $ en la pieza $k $, con
\[
\int_0^1 \varphi = \sum_{k=1}^{n} \frac{k-1}{n}\cdot\frac1n
= \frac{n-1}{2n},
\qquad
\int_0^1 \psi = \sum_{k=1}^{n} \frac{k}{n}\cdot\frac1n
= \frac{n+1}{2n} .
\]
Cada \omterm{thm:b1:integration:def}{integral} inferior es $\leq I_-(f) \leq I_+(f) \leq$ cada
superior, por lo que $\frac{n-1}{2n} \leq I_-(f) \leq I_+(f) \leq
\frac{n+1}{2n}$ para todos los $n $: ambos se aprietan en $\frac12$, y
$\int_0^1 x\,\dd x = \frac12$ directamente de la definición. el
Perspectiva final: esta es la primera y última vez que integramos
de la definición --- el teorema fundamental a continuación reemplaza
todos estos cálculos mediante una búsqueda de antiderivada, que es
todo el punto económico de este capítulo.
\end{example}

\begin{theorem}[Propiedades]\label{thm:b1:integration:props}
Para $f, g$\omterm{def:b1:continuity:continuous}{continuo} (o \omterm{def:b1:continuity:continuous}{continuo} por partes) en $\intcc{a}{b}$ y
$\lambda \in \R$:
\begin{enumerate}
 \item linealidad: $\int (f + \lambda g) = \int f + \lambda \int g$;
 \item monotonicidad: $f \leq g \implies \int_a^b f \leq \int_a^b g$;
 y $\bigl|\int_a^b f\bigr| \leq \int_a^b \abs f \leq (b -
 a)\, \sup\abs f $;
 \item Chasles: $\int_a^b = \int_a^c + \int_c^b $(con el
 convención $\int_b^a = -\int_a^b$, válida para cualquier orden de
 los límites);
 \item positividad estricta: si $f$ es \omterm{def:b1:continuity:continuous}{continuo},$f \geq 0$ y
 $\int_a^b f = 0$, luego $f = 0$ en todas partes
 $\intcc{a}{b}$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1)--(3) pasar de \omterm{def:b1:integration:step}{funciones escalonadas} al límite por el
definición sup/inf. La linealidad merece los detalles una vez: dado
$\varepsilon > 0$, marco $\varphi_f \leq f \leq \psi_f$ y
$\varphi_g \leq g \leq \psi_g$ con huecos $\leq \varepsilon $
(\cref{thm:b1:integration:approx}). Para $\lambda \geq 0$,
$\varphi_f + \lambda\varphi_g \leq f + \lambda g \leq \psi_f +
\lambda\psi_g$ es un encuadre de \omterm{def:b1:integration:step}{funciones escalonadas} con espacio $\leq
(1 + \lambda)\varepsilon $, y su paso \omterm{thm:b1:integration:def}{integrales} es igual a$\int
\varphi_f + \lambda\int\varphi_g $, etc.\
(\cref{prop:b1:integration:stepprops}): dejar que $\varepsilon \to
0$ apriete $\int(f + \lambda g)$ sobre $\int f + \lambda\int g $.
Para $\lambda < 0$, multiplicando por $\lambda$\emph{invierte} el
encuadre de $g$--- el \omterm{def:b1:integration:step}{función de paso} inferior de $\lambda g$ es
$\lambda \psi_g$ --- y el mismo apretón corre con los roles
intercambiado. La cota $\abs{\int f} \leq \int \abs f$ proviene de
$-\abs f \leq f \leq \abs f$ y monotonía.

(4) Contrapositivo: si $f(x_0) = m > 0$, \omterm{def:b1:continuity:continuous}{continuidad} proporciona un
subintervalo de longitud $\eta > 0$ en el que $f \geq \frac m2$; el
\omterm{def:b1:integration:step}{función de paso} vale $\frac m2$ allí y $0$ en otros lugares es $\leq f$,
entonces $\int f \geq \frac{m\eta}{2} > 0$.
\end{proof}

\begin{example}[Chasles en el trabajo: integrales con valores absolutos]\label{ex:b1:integration:chasleswork}
Para integrar un valor absoluto, corte donde cambia el signo.
\[
\int_0^2 \abs{x - 1}\,\dd x
= \int_0^1 (1 - x)\,\dd x + \int_1^2 (x - 1)\,\dd x
= \frac12 + \frac12 = 1 ,
\]
y, cortando $\intcc{0}{2\pi}$ en $\pi$:
\[
\int_0^{2\pi} \abs{\sin t}\,\dd t
= \int_0^{\pi} \sin t\,\dd t - \int_{\pi}^{2\pi} \sin t\,\dd t
= 2 + 2 = 4 ,
\]
mientras que $\int_0^{2\pi} \sin t\,\dd t = 0$: la cancelación es real,
y por eso la estricta positividad \omterm{def:b1:logic:statement}{enunciado}
(\cref{thm:b1:integration:props}~(4)) lleva la hipótesis $f
\geq 0$ --- sin ella, un \omterm{thm:b1:integration:def}{integral} que desaparece no prueba nada
sobre $f$. La idea final: medidas $\int \abs f$\emph{área}, $\int f$ mide \emph{saldo firmado}; el
desigualdad $\abs{\int f} \leq \int\abs f$ es el registro exacto de
lo que la cancelación puede destruir.
\end{example}

\section{El teorema fundamental del cálculo.}

\begin{theorem}[Teorema fundamental del calculo]\label{thm:b1:integration:ftc}
Sea $f$\omterm{def:b1:continuity:continuous}{continuo} en un \omterm{prop:b1:reals:intervals}{intervalo}$I$ y $a \in I$. la funcion
\[
F(x) = \int_a^x f(t)\, \dd t
\]
es de clase $C^1$ en $I$, con $F' = f$: cada función \omterm{def:b1:continuity:continuous}{continuo}
en un \omterm{prop:b1:reals:intervals}{intervalo} tiene primitivas. En consecuencia, para \emph{cualquier}
primitiva $G$ de $f$:\index{teorema fundamental del cálculo}
\[
\int_a^b f(t)\,\dd t = G(b) - G(a) .
\]
\end{theorem}

\begin{proof}
Reparar $x_0 \in I$ y $\varepsilon > 0$; \omterm{def:b1:continuity:continuous}{continuidad} en $x_0$ proporciona
$\delta$ con $\abs{f(t) - f(x_0)} \leq \varepsilon$ para $\abs{t -
x_0} \leq \delta$. Para $0 < \abs{h} \leq \delta $(y $x_0 + h \in
I $), Chasles ofrece
\[
\frac{F(x_0 + h) - F(x_0)}{h} - f(x_0)
= \frac 1h \int_{x_0}^{x_0+h} \bigl(f(t) - f(x_0)\bigr)\dd t ,
\]
cuyo valor absoluto es $\leq \frac{1}{\abs h}\cdot \abs h\,
\varepsilon = \varepsilon $(cota (2), válido para cualquier orden del
límites). Entonces $F'(x_0) = f(x_0)$;$F' = f$ es \omterm{def:b1:continuity:continuous}{continuo}:$F$ es
$C^1$. Si$G' = f$ también, entonces $(G - F)' = 0$ en \omterm{prop:b1:reals:intervals}{intervalo}, entonces $G =
F + c $(\cref{cor:b1:derivative:monotone}) y $G(b) - G(a) = F(b)
- F(a) = \int_a^b f $.
\end{proof}

\begin{example}[Simetría antes del cálculo]\label{ex:b1:integration:parity}
En un \omterm{prop:b1:reals:intervals}{intervalo} simétrico, la paridad hace el trabajo: si $f$ es impar,
la sustitución $t \mapsto -t$ envía $\int_{-a}^{0} f$ a
$-\int_0^a f$, entonces
\[
\int_{-a}^{a} f(t)\,\dd t = 0 ;
\qquad\text{if $f$ is even,}\quad
\int_{-a}^{a} f = 2\int_0^a f .
\]
Así $\int_{-1}^{1} \frac{t^3\cos t}{1 + t^4}\,\dd t = 0$ con
no hay primitivos a la vista (el integrando es impar), y
$\int_{-\pi}^{\pi} t^2\cos t\,\dd t = 2\int_0^\pi t^2\cos t\,
\dd t$. Compruebe la simetría antes de recurrir a las técnicas: el
El \omterm{thm:b1:integration:def}{integral} más rápido es el que nunca se calcula.
\end{example}

\begin{example}[Reconociendo un derivado a la vista]\label{ex:b1:integration:halfangle}
Calcule $\displaystyle\int_0^{\pi/2} \frac{\dd x}{1 + \cos x}$.
La identidad del medio ángulo $1 + \cos x = 2\cos^2\frac x2$ convierte el
integrando en $\frac{1}{2}\bigl(1 + \tan^2\frac
x2\bigr)$, que es exactamente el \omterm{def:b1:derivative:def}{derivado} de $\tan\frac x2$:
\[
\int_0^{\pi/2} \frac{\dd x}{1 + \cos x}
= \Bigl[\tan\frac x2\Bigr]_0^{\pi/2} = \tan\frac\pi4 = 1 .
\]
No se necesitaba ninguna maquinaria de sustitución, sólo el reflejo de
leer un integrando como el \omterm{def:b1:derivative:def}{derivado} de alguien, el fundamental
teorema haciendo el resto. (La herramienta sistemática detrás de tal
trigonométrico \omterm{thm:b1:integration:def}{integrales}, la sustitución $t = \tan\frac x2$,
pertenece al conjunto de herramientas estándar creado a partir de
\cref{thm:b1:integration:parts}~(2).)
\end{example}

\begin{example}[Funciones definidas por integrales]\label{ex:b1:integration:definedby}
El teorema fundamental fabrica funciones. dejar
\[
F(x) = \int_0^x \eu^{-t^2}\,\dd t .
\]
Ninguna combinación de funciones clásicas tiene \omterm{def:b1:derivative:def}{derivado}
$\eu^{-t^2}$(un teorema de Liouville, admitido); sin embargo,$F$ existe,
es $C^1$ con $F'(x) = \eu^{-x^2} > 0$, estrictamente creciente, impar
(sustituya $t \mapsto -t$), y acotado: para $x \geq 1$,
\[
F(x) - F(1) = \int_1^x \eu^{-t^2}\dd t
\leq \int_1^x \eu^{-t}\dd t \leq \eu^{-1} ,
\]
entonces $F \leq F(1) + \eu^{-1} \leq 1 + \eu^{-1}$. (El límite exacto,
$\frac{\sqrt\pi}{2}$, se computa con el doble \omterm{thm:b1:integration:def}{integrales} en el
Volumen del año 3.) Regla de la cadena para límites móviles: $\frac{\dd}{\dd
x}\int_x^{x^2} \eu^{-t^2}\dd t = 2x\,\eu^{-x^4} - \eu^{-x^2}$.
La idea final: integración \emph{crea} nuevas funciones
de los antiguos, con todas sus propiedades legibles desde el
integrando --- el primitivo que no puedes escribir sigue siendo un
función que usted controla completamente.
\end{example}

\begin{example}[Estimando sin evaluar]\label{ex:b1:integration:estimate}
Los \omterm{thm:b1:integration:def}{integrales} $R_n = \int_0^1 \frac{t^n}{1 + t}\,\dd t$ no tienen
agradable forma cerrada, pero la monotonicidad los fija precisamente: en
$\intcc{0}{1}$,$\frac12 \leq \frac{1}{1+t} \leq 1$, entonces
\[
\frac{1}{2(n+1)} = \frac12\int_0^1 t^n\,\dd t
\;\leq\; R_n \;\leq\; \int_0^1 t^n \,\dd t = \frac{1}{n+1} :
\]
el orden exacto de desintegración ($R_n \sim$ un múltiplo de $\frac1n$,
de hecho $R_n \sim \frac{1}{2n}$) con dos líneas y sin
antiderivada. Los problemas de fin de semana de este capítulo y los
próxima ejecución exactamente en tales encuadres: la primera vez que el analista
El instinto antes de un \omterm{thm:b1:integration:def}{integral} debe ser \emph{lo ató}, y solo
luego, si es necesario, calculelo.
\end{example}

\begin{example}[Valores medios]\label{ex:b1:integration:average}
El \emph{significar} de un \omterm{def:b1:continuity:continuous}{continuo} $f$ sobre $\intcc{a}{b}$ es
$\frac{1}{b-a}\int_a^b f$. Para el arco del seno:
\[
\frac{1}{\pi}\int_0^\pi \sin t\,\dd t
= \frac{1}{\pi}\bigl[-\cos t\bigr]_0^\pi = \frac{2}{\pi}
\approx 0.637 :
\]
un arco positivo completo no promedia $\frac12$ sino
$\frac2\pi$ --- la curva pasa más tiempo en alto que un triángulo
lo haría. Por \cref{exo:b1:integration:11} (teorema del valor medio para
\omterm{thm:b1:integration:def}{integrales}, $g = 1$), la media es un \emph{valor}:$\sin c =
\frac2\pi $ para algunos $c \in \intoo{0}{\pi}$. Y según la \omterm{thm:b1:integration:riemann}{Riemann
sumas} de este capítulo, la media es el límite de las ordinarias.
promedios de muestras $n$ --- el puente entre las muestras discretas
media de datos y la media \omterm{def:b1:continuity:continuous}{continuo} de una señal, que es como
el \omterm{thm:b1:integration:def}{integral} entra en física.
\end{example}

\begin{theorem}[Integración por partes; reemplazo]\label{thm:b1:integration:parts}
\begin{enumerate}
 \item Si $u, v$ son $C^1$ en $\intcc{a}{b}$:\index{integración por
 partes}
 \[
 \int_a^b u'v = \bigl[uv\bigr]_a^b - \int_a^b uv' .
 \]
 \item Si $\varphi$ es $C^1$ en $\intcc{\alpha}{\beta}$ y $f$\omterm{def:b1:continuity:continuous}{continuo} en $\varphi(\intcc{\alpha}{\beta})$:
 \index{sustitución}
 \[
 \int_{\alpha}^{\beta} f\bigl(\varphi(t)\bigr)\,\varphi'(t)\,
 \dd t = \int_{\varphi(\alpha)}^{\varphi(\beta)} f(x)\, \dd x .
 \]
\end{enumerate}
\end{theorem}

\begin{proof}
(1) $(uv)' = u'v + uv'$; integrar sobre $\intcc{a}{b}$ y aplicar el
teorema fundamental de la función $C^1$ $uv$.

(2) Sea $F$ una primitiva de $f$ en la imagen \omterm{prop:b1:reals:intervals}{intervalo}
(\cref{thm:b1:integration:ftc}). Entonces $(F \circ \varphi)' = (f
\circ \varphi)\,\varphi'$(regla de la cadena), por lo que ambos lados son iguales
$F(\varphi(\beta)) - F(\varphi(\alpha))$.
\end{proof}

\begin{example}\label{ex:b1:integration:parts}
$\int_0^1 t\,\eu^t \dd t = \bigl[t\,\eu^t\bigr]_0^1 - \int_0^1
\eu^t\dd t = \eu - (\eu - 1) = 1$. Y con la sustitución $x =
\sin t $($t \in \intcc{0}{\frac\pi2}$):
\[
\int_0^1 \sqrt{1 - x^2}\, \dd x
= \int_0^{\pi/2} \cos t \cdot \cos t \, \dd t
= \int_0^{\pi/2} \frac{1 + \cos 2t}{2}\, \dd t = \frac\pi4 ,
\]
--- un cuarto de la unidad de disco, según lo exige la geometría. (Linealización
de \cref{met:b1:complex:trig} en el trabajo).
\end{example}

\begin{remark}[Errores comunes en el cálculo integral]
(yo) \emph{Substitutions must be $C^1$ on the whole \omterm{prop:b1:reals:intervals}{intervalo}}:
el cambio $x = \frac1t$ es ilegal en $0$; aplicado a ciegas
a $\int_{-1}^{1}\frac{\dd x}{1 + x^2}$ "prueba" que el
\omterm{thm:b1:integration:def}{integral} es igual a su propio negativo. Cuando una sustitución tiene un
singularidad, corte primero el \omterm{prop:b1:reals:intervals}{intervalo} (Chasles), sustitúyalo por
cada pieza, y solo luego recombinarlas. (ii) \emph{logarítmico
las primitivas necesitan valores absolutos}: $\int \frac{\dd x}{x - 2} =
\ln\abs{x - 2} + C $ en cada lado de $2$ por separado --- escritura
$\ln(x - 2)$ en $\intoo{0}{1}$ está escribiendo el logaritmo de un
número negativo; y la constante $C$ puede diferir en los dos
lados de la singularidad. (iii) \emph{A vanishing \omterm{thm:b1:integration:def}{integral} does
not kill the function}: $\int_0^{2\pi}\sin = 0$; positividad de
el integrando es necesario antes de concluir $f = 0$
(\cref{ex:b1:integration:chasleswork}). (iv) \emph{\omterm{thm:b1:integration:riemann}{sumas de riemann}
must be calibrated}: en $\frac{b-a}{n}\sum f\bigl(a +
k\frac{b-a}{n}\bigr)$, el paso exterior y los puntos interiores
debe coincidir con la misma subdivisión --- el error frecuente es una suma
$\sum_{k=1}^{n} f\bigl(\frac kn\bigr)$\emph{sin} el factor
$\frac1n$, que diverge en lugar de converger a$\int_0^1
f$. Lista de verificación antes de invocar \cref{thm:b1:integration:riemann}:
factorice $\frac1n$, reescriba el sumando como $f$ de $\frac kn $,
nombre $f$ y verifique que sea \omterm{def:b1:continuity:continuous}{continuidad}.
\end{remark}

\begin{example}[Adivina, diferencia, ajusta]\label{ex:b1:integration:guessdiff}
¿Qué es $\int_1^x (\ln t)^2\,\dd t$? Adivina una primitiva de la
formar $t\,P(\ln t)$ con $P$\omterm{def:b1:poly:def}{polinomio} y diferenciar:
\[
\bigl(t\,P(\ln t)\bigr)' = P(\ln t) + P'(\ln t) .
\]
Necesitamos $P(u) + P'(u) = u^2$: toma $P(u) = u^2 - 2u + 2$
(coeficientes coincidentes hacia abajo desde $u^2$). Por lo tanto
\[
\int_1^x (\ln t)^2\,\dd t
= \bigl[t\bigl((\ln t)^2 - 2\ln t + 2\bigr)\bigr]_1^x
= x(\ln x)^2 - 2x\ln x + 2x - 2 ,
\]
resultado que de otro modo se alcanzaría mediante dos integraciones por partes. el
Información final: para integrandos de la forma.
(\omterm{def:b1:poly:def}{polinomio} en $\ln t$) o (\omterm{def:b1:poly:def}{polinomio} veces $\eu^{\lambda t}$),
el primitivo tiene la misma forma --- diferenciando una forma
Supongo que convierte la integración en álgebra lineal sobre coeficientes,
más rápido y menos propenso a errores que las partes iteradas.
\end{example}

\begin{example}[El boomerang integral]\label{ex:b1:integration:boomerang}
Calcule $I = \int_0^{\pi/2} \eu^x \cos x\,\dd x$. Integrar por
partes dos veces, diferenciando el factor trigonométrico cada vez:
\[
I = \bigl[\eu^x \sin x\bigr]_0^{\pi/2}
- \int_0^{\pi/2} \eu^x \sin x\,\dd x
= \eu^{\pi/2} - J,
\qquad
J = \bigl[-\eu^x\cos x\bigr]_0^{\pi/2}
+ \int_0^{\pi/2} \eu^x\cos x\,\dd x = 1 + I .
\]
El \omterm{thm:b1:integration:def}{integral} ha vuelto a sí mismo: $I = \eu^{\pi/2} - 1 - I$,
de donde
\[
I = \frac{\eu^{\pi/2} - 1}{2} .
\]
La idea final: cuando el integrando es producto de dos
funciones que se reproducen bajo diferenciación
($\eu^{ax}$,$\cos bx $,$\sin bx $), dos integraciones por partes
produce una ecuación lineal para la incógnita \omterm{thm:b1:integration:def}{integral} --- resuélvela
en lugar de integrar; equivalentemente, pasar a través
$\eu^{(a + \iu b)x}$(\cref{ch:b1:complex}) y toma real
partes. Ambos caminos dan la misma respuesta, y comprobando que
hacer es una prueba de cordura gratuita.
\end{example}

\section{sumas de riemann}

\begin{theorem}[sumas de riemann]\label{thm:b1:integration:riemann}
Sea $f$\omterm{def:b1:continuity:continuous}{continuo} en $\intcc{a}{b}$. Entonces\index{sumas de riemann}
\[
S_n = \frac{b - a}{n} \sum_{k=0}^{n-1}
f\Bigl(a + k\,\frac{b-a}{n}\Bigr)
\xrightarrow[n \to \infty]{} \int_a^b f(t)\, \dd t ,
\]
y lo mismo con cualquier punto de evaluación dentro de los subintervalos.
\end{theorem}

\begin{proof}
$S_n$ es el \omterm{thm:b1:integration:def}{integral} del \omterm{def:b1:integration:step}{función de paso}$\varphi_n$ igual a
$f(a + k\frac{b-a}{n})$ en el subintervalo $k$. dado
$\varepsilon > 0$, \omterm{def:b1:continuity:uniform}{continuidad uniforme} (Heine) proporciona $\delta$; para
$n > \frac{b-a}{\delta}$, cada punto de un subintervalo está dentro
$\delta$ de su punto de evaluación, entonces $\abs{f - \varphi_n} \leq
\varepsilon$ en $\intcc{a}{b}$, de donde
\[
\Bigl| \int_a^b f - S_n \Bigr|
= \Bigl| \int_a^b (f - \varphi_n) \Bigr|
\leq (b-a)\,\varepsilon . \qedhere
\]
\end{proof}

\begin{example}\label{ex:b1:integration:riemannexample}
$\displaystyle\sum_{k=1}^{n} \frac{1}{n + k} = \frac 1n
\sum_{k=1}^{n} \frac{1}{1 + k/n} \xrightarrow[n\to\infty]{}
\int_0^1 \frac{\dd x}{1 + x} = \ln 2$: un límite opaco para cotas
elementales, transparente como suma de Riemann.
\end{example}

\begin{example}[Una segunda suma de Riemann, con calibración]\label{ex:b1:integration:riemann2}
Calcúlese $\lim_{n\to\infty} \sum_{k=1}^{n} \dfrac{n}{(n+k)^2}$.
Calibración:
\[
\sum_{k=1}^{n} \frac{n}{(n+k)^2}
= \frac{1}{n}\sum_{k=1}^{n} \frac{n^2}{(n+k)^2}
= \frac1n \sum_{k=1}^{n} \frac{1}{\bigl(1 + \frac kn\bigr)^2} ,
\]
suma de Riemann de la función
\omterm{def:b1:continuity:continuous}{continua}
$f(x) = \frac{1}{(1+x)^2}$ en $\intcc{0}{1}$: el límite es
\[
\int_0^1 \frac{\dd x}{(1 + x)^2}
= \Bigl[-\frac{1}{1+x}\Bigr]_0^1 = \frac12 .
\]
La idea final: todo el arte es la línea media ---
forzar el sumando a la forma $f(\frac kn)$ a costa de
extrayendo exactamente un factor $\frac1n$; una vez que la forma es
Bien, el teorema hace el análisis y el teorema fundamental.
hace la aritmética.
\end{example}

\begin{remark}[Dónde funciona la integral a continuación]
Las construcciones del capítulo tienen cada una una secuela. \omterm{thm:b1:integration:riemann}{sumas de riemann}
regresa en \cref{ch:b1:series} como puente entre la serie y
\omterm{thm:b1:integration:def}{integrales} (comparación de $\sum \frac{1}{n^\alpha}$ con $\int
\frac{\dd t}{t^\alpha}$); el resto \omterm{thm:b1:integration:def}{integral} es el más nítido
forma de la fórmula de Taylor en \cref{ch:b1:taylor}; el
La definición basada en $\sup$ es el prototipo de Lebesgue
\omterm{thm:b1:integration:def}{integral} del volumen del Año 3, donde las mismas tres propiedades
(linealidad, monotonicidad, un teorema de convergencia) se reconstruyen sobre una
clase de funciones mucho mayor. Y el problema del fin de semana a continuación.
convierte \omterm{thm:b1:integration:parts}{integración por partes} en aritmética: la irracionalidad de
$\pi^2$.
\end{remark}

\section{Problema: la máquina de irracionalidad integral}

\begin{problem}[{Problema del fin de semana --- $\eu$ y $\pi^2$ son
irracional, $\eu$ con seis decimales y $\frac{22}{7} > \pi$ con
prueba}]
\label{pb:b1:integration:1}
Un mecanismo impulsa todo este problema: \emph{una expresión que
debe ser un número entero positivo, pero probablemente sea menor que $1$,
no puede existir}. \Cref{exo:b1:integration:9} (Niven) lo ejecutó una vez para
probar $\pi \notin \Q$; aquí lo industrializamos. la maquina
necesita tres partes: una entrada \emph{integridad} (valores de punto final
de \omterm{def:b1:poly:def}{polinomios} bien elegido), una entrada \emph{pequeñez} (un factor
$\frac{1}{n!}$ aplastando el \omterm{thm:b1:integration:def}{integral}), y un \emph{puente}
(\omterm{thm:b1:integration:parts}{integración por partes}) conectándolos. Probamos que $\eu$ es
irracional y calcularlo con error certificado, probar la teoría de Legendre.
teorema más claro de que $\pi^2$ es irracional, y termina con el
\omterm{thm:b1:integration:def}{integral} más encantador en el análisis: $\int_0^1 \frac{x^4(1 -
x)^4}{1 + x^2}\dd x = \frac{22}{7} - \pi$, que entre corchetes $\pi$
a mano.\index{irracionalidad!of pi squared@of
$\pi^2$}\index{irracionalidad!of e@of $\eu$}

\medskip
\noindent\textbf{Parte I --- Fuel.}
\begin{enumerate}
 \item Demuestre que $\dfrac{c^{\,n}}{n!} \to 0$ por cada $c
 > 0$\emph{(beyond $n \geq 2c$, each step at least halves
 the term)} fijo.
 \item (Beta \omterm{thm:b1:integration:def}{integrales}) Demostrar, por inducción en $m$ con
 \omterm{thm:b1:integration:parts}{integración por partes}:
 \[
 \int_0^1 x^{\,k}\,(1 - x)^{\,m}\,\dd x =
 \frac{k!\,m!}{(k + m + 1)!}
 \qquad (k, m \in \N).
 \]
 \item Deduce $\displaystyle\int_0^1 \bigl(x(1-x)\bigr)^n \dd x
 = \frac{1}{(2n+1)\binom{2n}{n}}$, y --- comparando con
 La cota $x(1 - x) \leq \frac14$ --- la estimación
 $\binom{2n}{n} \geq \dfrac{4^n}{2n+1}$, coincidente
 $\binom{2n}{n}^{1/n} \to 4$ de \cref{pb:b1:seq:1}.
 \item Demuestre el lema de pequeñez usado dos veces a continuación: por cada
 \omterm{def:b1:continuity:continuous}{continuo} $g > 0$ en $\intoo{0}{1}$,
 \[
 0 < \int_0^1 \bigl(x(1-x)\bigr)^n g(x)\,\dd x
 \leq \frac{\sup_{\intcc{0}{1}}\abs g}{4^{\,n}} .
 \]
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- $\eu$: irrationality, then six
decimals.} Establecer $A_n = \displaystyle\int_0^1 x^n \eu^x \dd x$.
\begin{enumerate}[resume]
 \item Calcule $A_0$ y $A_1$, pruebe la recurrencia $A_n = \eu
 - n A_{n-1}$ y los límites $0 < A_n \leq
 \dfrac{\eu}{n+1}$.
 \item Mostrar por inducción que $A_n = \alpha_n + \beta_n \eu$
 con $\alpha_n, \beta_n \in \Z$.
 \item Deduzca que $\eu$ es irracional \emph{(if $\eu = \frac
 pq $, then $q A_n $ is an integer trapped in
 $\intoo{0}{1}$ for $n$ large)}. Comparar con la prueba
 de \cref{exo:b1:seq:9}: mismo remate, diferente combustible.
 \item Demostrar, por inducción y \omterm{thm:b1:integration:parts}{integración por partes}, la exactitud
 fórmula del resto
 \[
 \eu = \sum_{k=0}^{n} \frac{1}{k!} + R_n,
 \qquad
 R_n = \frac{1}{n!}\int_0^1 (1 - t)^n\,\eu^{\,t}\,\dd t,
 \qquad
 \frac{1}{(n+1)!} \leq R_n \leq \frac{\eu}{(n+1)!} .
 \]
 \item Tome $n = 9$: cota $R_9$ usando $\eu < 2.75$(de
 $b_2 = 2.75$ en \cref{ex:b1:seq:e}), evaluar la suma,
 y concluir el paréntesis certificado $2.7182818 \leq
 \eu \leq 2.7182823$--- seis decimales,$\eu \approx
 2.718282$, con comprobante.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Legendre's theorem: $\pi^2$ is
irrational.} Sea $f(x) = \dfrac{x^n (1 - x)^n}{n!}$, y supongamos
$\pi^2 = \frac ab$ con $a, b \in \N^*$.
\begin{enumerate}[resume]
 \item Mostrar $f(1 - x) = f(x)$ y $0 < f \leq
 \dfrac{1}{4^n\,n!}$ en $\intoo{0}{1}$.
 \item Demuestre que $f^{(k)}(0)$ y $f^{(k)}(1)$ son números enteros para
 cada $k \geq 0$\emph{(expand $x^n(1-x)^n$ with integer
 coefficients; $\frac{k!}{n!} \in \Z$ for $k \geq n$;
 then use the symmetry)}.
 \item Definir
 \[
 G = b^{\,n} \sum_{k=0}^{n} (-1)^k\,
 \pi^{2n - 2k} f^{(2k)} .
 \]
 Demuestre que $G(0)$ y $G(1)$ son números enteros \emph{(each
 $b^n \pi^{2n-2k} = a^{\,n-k}\,b^{\,k}$)}.
 \item Verifique el $G'' + \pi^2 G = b^n \pi^{2n+2} f
 = \pi^2 a^n f $ telescópico, luego
 \[
 \frac{\dd}{\dd x}\Bigl(G'(x)\sin \pi x - \pi\,G(x)\cos
 \pi x\Bigr) = \pi^2 a^n f(x)\sin \pi x .
 \]
 \item Integrar sobre $\intcc{0}{1}$ y concluir
 \[
 \pi\,a^n \int_0^1 f(x)\sin(\pi x)\,\dd x = G(0) + G(1)
 \in \Z ,
 \]
 un entero \emph{positivo} limitado por $\dfrac{\pi
 a^n}{4^n\,n!}$.
 \item Concluya con la pregunta 1 que $\pi^2$ es irracional.
 (Legendre, 1794), y que esto fortalece
 \cref{exo:b1:integration:9}: ¿por qué la irracionalidad
 de $\pi^2$ implican el de $\pi$, y no a la inversa?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Understanding the machine.}
\begin{enumerate}[resume]
 \item Localice las dos fuerzas opuestas (integralidad del punto final
 datos; pequeñez analítica del \omterm{thm:b1:integration:def}{integral}) y la
 puente, en las Partes II y III. Luego explica por qué el factor
 $\frac{1}{n!}$ en $f$ es el quid: si se elimina,
 la integralidad sobrevive, pero la desigualdad muere, y por
 que reclamó las fracciones $\frac ab$ hace la prueba entonces
 fallar?
 \item Efectividad: supongamos que alguien reclama $\pi^2 = \frac ab$
 con $a \leq 10$. Demuestre que la contradicción ya
 aterriza en $n = 7$: calcula $\pi\,(10/4)^7/7! \approx
 0.38 < 1$. La máquina no se limita a refutar; eso
 refuta mediante una etapa fija y computable.
 \item Control de cordura del puente incondicionalmente: prueba por dos
 integraciones por partes que
 \[
 \int_0^1 x(1 - x)\sin(\pi x)\,\dd x = \frac{4}{\pi^3},
 \]
 y conciliar con la pregunta 14 en $n = 1$(mantener
 $\pi^2$ simbólico: la identidad telescópica dice $\pi^3
 \int_0^1 f_1 \sin \pi x = -(f_1''(0) + f_1''(1)) = 4$).
 \item ¿Qué hace que $\eu^x$ y $\sin \pi x$ sean elegibles como
 ¿Los núcleos de la máquina? Identifique la propiedad (cada una satisface
 una ecuación diferencial lineal con constante
 coeficientes, por lo que se repiten ciclos \omterm{thm:b1:integration:parts}{integración por partes}
 volver al inicio), y nombrar la frontera: lo mismo
 máquina, perfeccionada por Hermite y Lindemann, demuestra $\eu$
 y $\pi$\emph{trascendental} --- más allá de este volumen.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- $\frac{22}{7}$ versus $\pi$, and the
moral.}
\begin{enumerate}[resume]
 \item Establecer la división \omterm{def:b1:poly:def}{polinomio}
 \[
 \frac{x^4(1-x)^4}{1 + x^2}
 = x^6 - 4x^5 + 5x^4 - 4x^2 + 4 - \frac{4}{1 + x^2},
 \]
 y deducir la célebre identidad
 \[
 \int_0^1 \frac{x^4 (1-x)^4}{1 + x^2}\,\dd x
 = \frac{22}{7} - \pi .
 \]
 \item El integrando es positivo: concluir $\pi <
 \frac{22}{7}$. Luego, delimitando $\frac{1}{1+x^2}$ entre
 $\frac12$ y $1$ y usando $\int_0^1 (x(1-x))^4 =
 \frac{1}{630}$(pregunta 3), pruebe
 \[
 \frac{22}{7} - \frac{1}{630} \;\leq\; \pi \;\leq\;
 \frac{22}{7} - \frac{1}{1260} ,
 \]
 es decir $3.14126 \leq \pi \leq 3.14207$: dos correctos
 decimales, a mano.
 \item Generalizar: dividiendo $x^{4m}(1-x)^{4m}$ por $1 + x^2$,
 muestre que el resto es la constante $(-4)^m$\emph{(work
 modulo $x^2 + 1$:$(1-x)^2 \equiv -2x$)}, deduzca
 racionales $r_m$ con
 \[
 \abs{\pi - r_m} \leq 4^{\,1 - 5m} ,
 \]
 y verifique que $m = 1$ reproduzca la pregunta 20--21.
 \item Confronta estos racionales con la teoría de la aproximación.
 de \cref{pb:b1:derivative:1}: calcular $\abs{\pi -
 \frac{22}{7}} \approx 1.26\cdot10^{-3}$ contra el
 Garantía Dirichlet $\frac{1}{49}$ y cotización.
 $\abs{\pi - \frac{355}{113}} \approx 2.7\cdot10^{-7}$
 contra $\frac{1}{113^2} \approx 7.8\cdot10^{-5}$:
 Existen aproximaciones racionales excepcionalmente buenas para
 $\pi$--- consistente, ya que no se sabe que $\pi$ sea
 mal aproximado.
 \item (La trampa de los enteros, resumida) Demuestre el lema que
 unifica todo: \emph{if $x \in \R$ and there exist
 integers $a_n, b_n$ with $0 < \abs{a_n + b_n x} \to 0$,
 then $x$ is irrational.} Enumere sus instancias en este
 problema, en \cref{exo:b1:integration:9}, en
 \cref{exo:b1:seq:9} y en \cref{pb:b1:derivative:1}.
 \item Síntesis, una frase cada una: (i) los tres de la máquina
 partes y dónde vive cada una en la caja de herramientas de este
 capítulo; (ii) qué aporta el \omterm{thm:b1:integration:def}{integral} que el
 teorema del valor medio de \cref{pb:b1:derivative:1} podría
 no; (iii) el inventario de resultados extraídos (dos
 irracionalidades, una constante de seis decimales, una
 horquillado de $\pi$, un límite binomial); (iv) el
 frontera (Hermite, Lindemann; y la misma trampa, corre sobre
 $\zeta(2)$ y $\zeta(3)$, en el siglo XX
 aritmética).
\end{enumerate}
