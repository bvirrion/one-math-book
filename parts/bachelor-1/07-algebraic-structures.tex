\chapter{Algebraic Structures}\label{ch:b1:structures}

The same computation rules keep reappearing: integers, real numbers,
complex numbers, \omterm{def:b1:arith:congruence}{congruence} classes, and soon polynomials
(\cref{ch:b1:poly}), vectors and matrices
(\cref{ch:b1:vspaces,ch:b1:matrices}). Algebra extracts the common
patterns and names them: \emph{\omterm{def:b1:structures:group}{group}}, \emph{\omterm{def:b1:structures:ring}{ring}}, \emph{\omterm{def:b1:structures:field}{field}}.
Proving a fact once, at the level of the structure, proves it for
every example at once.

\section{Laws of composition}

\begin{definition}\label{def:b1:structures:law}
A \emph{law of composition}\index{law of composition} on a \omterm{def:b1:logic:sets}{set} $E$ is
a \omterm{def:b1:logic:map}{map} $E \times E \to E$, written $(x, y) \mapsto x * y$. It is
\emph{associative} when $(x*y)*z = x*(y*z)$ always, \emph{commutative}
when $x * y = y * x$ always. An element $e$ is an \emph{identity} when
$e * x = x * e = x$ for all $x$; then $x'$ is an \emph{inverse} of $x$
when $x * x' = x' * x = e$.
\end{definition}

\begin{proposition}[Uniqueness]\label{prop:b1:structures:unique}
A law has at most one identity; for an associative law with identity,
each element has at most one inverse.
\end{proposition}

\begin{proof}
If $e$ and $e'$ are identities: $e = e * e' = e'$. If $x'$ and $x''$
invert $x$: $x' = x' * e = x' * (x * x'') = (x' * x) * x'' = e * x''
= x''$.
\end{proof}

\section{Groups}

\begin{definition}[Group]\label{def:b1:structures:group}
A \emph{group}\index{group} $(G, *)$ is a \omterm{def:b1:logic:sets}{set} with an associative law
admitting an identity and in which every element has an inverse. The
group is \emph{abelian}\index{abelian group} when the law is
commutative.
\end{definition}

\begin{example}\label{ex:b1:structures:groups}
$(\Z, +)$, $(\Q, +)$, $(\R, +)$, $(\C, +)$; $(\Q^*, \times)$,
$(\R^*, \times)$, $(\C^*, \times)$, $(\mathbb{U}_n, \times)$ (\omterm{def:b1:complex:unity}{roots of
unity}, \cref{def:b1:complex:unity}); the \omterm{def:b1:logic:sets}{set} $\mathfrak{S}(E)$ of
bijections of a \omterm{def:b1:logic:sets}{set} $E$ onto itself, under composition --- the
\emph{symmetric group}\index{symmetric group} of $E$, \omterm{def:b1:structures:group}{non-abelian} as
soon as $\abs E \geq 3$. Not \omterm{def:b1:structures:group}{groups}: $(\N, +)$ (no inverses),
$(\Z, \times)$ (only $\pm 1$ invertible).
\end{example}

\begin{proposition}[Computation rules]\label{prop:b1:structures:rules}
In a \omterm{def:b1:structures:group}{group} $G$ (written multiplicatively, identity $e$):
\begin{enumerate}
  \item cancellation: $ax = ay \implies x = y$ and $xa = ya \implies
        x = y$;
  \item $(ab)^{-1} = b^{-1} a^{-1}$ and $(a^{-1})^{-1} = a$;
  \item for $a, b \in G$, each equation $ax = b$ and $xa = b$ has a
        unique solution ($x = a^{-1}b$, resp.\ $x = b a^{-1}$).
\end{enumerate}
\end{proposition}

\begin{proof}
(1) Multiply by $a^{-1}$ on the appropriate side, using
associativity. (2) $(b^{-1}a^{-1})(ab) = b^{-1}(a^{-1}a)b = b^{-1}b =
e$ and symmetrically; uniqueness of the inverse concludes; the second
point is \cref{prop:b1:structures:unique} applied to $a^{-1}$. (3)
Substitute and use (1) for uniqueness.
\end{proof}

\begin{example}[The symmetries of a rectangle]\label{ex:b1:structures:klein}
A (non-square) rectangle admits exactly four isometries onto
itself: the identity $e$, the horizontal-axis reflection $h$, the
vertical-axis reflection $v$, and the half-turn $r$ about the
center. Composition makes this four-element \omterm{def:b1:logic:sets}{set} a \omterm{def:b1:structures:group}{group}: each
element is its own inverse ($h^2 = v^2 = r^2 = e$), and the
product of any two distinct non-identity elements is the third
($hv = vh = r$: reflecting in both axes is the half-turn). The
full table is symmetric, so the \omterm{def:b1:structures:group}{group} is \omterm{def:b1:structures:group}{abelian} --- yet it is
\emph{not} the same \omterm{def:b1:structures:group}{group} as the rotations $\mathbb U_4$ of
\cref{ex:b1:structures:order}: there, $\iu$ has \omterm{def:b1:structures:order}{order} $4$, while
here every element has \omterm{def:b1:structures:order}{order} $\leq 2$. Two \omterm{def:b1:structures:group}{groups} of the same
size can thus have genuinely different multiplication structures
--- the figure below displays both tables side by side. This
four-element \omterm{def:b1:structures:group}{group} returns as $\{\pm1\} \times \{\pm1\}$, and
\cref{exo:b1:structures:7} explains why any \omterm{def:b1:structures:group}{group} with all
squares trivial must, like this one, be \omterm{def:b1:structures:group}{abelian}.
\end{example}

\begin{definition}[Subgroup]\label{def:b1:structures:subgroup}
A subset $H$ of a \omterm{def:b1:structures:group}{group} $G$ is a \emph{subgroup}\index{subgroup}
(written $H \leq G$) when it contains $e$, is stable under the law
and under inversion. Then $H$ is itself a \omterm{def:b1:structures:group}{group}.

\emph{Criterion:} a nonempty $H \subseteq G$ is a subgroup if and only
if
\[
\forall x, y \in H, \quad x y^{-1} \in H .
\]
\end{definition}

\begin{proof}[Proof of the criterion]
A \omterm{def:b1:structures:subgroup}{subgroup} obviously satisfies it. Conversely, let $H \neq \emptyset$
satisfy it, and pick $x_0 \in H$. Then $e = x_0 x_0^{-1} \in H$; for
$y \in H$, $y^{-1} = e\,y^{-1} \in H$; and for $x, y \in H$, $xy =
x (y^{-1})^{-1} \in H$.
\end{proof}

\begin{example}\label{ex:b1:structures:subgroups}
$\mathbb{U}_n \leq (\C^*, \times)$: nonempty, and for $z, w \in
\mathbb{U}_n$, $(zw^{-1})^n = z^n (w^n)^{-1} = 1$. The \omterm{def:b1:structures:subgroup}{subgroups} of
$(\Z, +)$ are exactly the $n\Z$ (proved in
\cref{thm:b1:arith:gcd}). An intersection of \omterm{def:b1:structures:subgroup}{subgroups} is always a
\omterm{def:b1:structures:subgroup}{subgroup}, but a union almost never is
(\cref{exo:b1:structures:6}).
\end{example}

\begin{remark}[Common pitfalls with structures]\label{rem:b1:structures:pitfalls}
\begin{enumerate}
  \item \emph{Stability under the law is not enough.} $\N$ is
        stable under addition inside $\Z$, contains $0$, yet is
        no \omterm{def:b1:structures:subgroup}{subgroup}: inverses are missing. The criterion
        $xy^{-1} \in H$ tests everything at once --- but only
        after checking $H \neq \emptyset$.
  \item \emph{Nonabelian reflexes.} In a general \omterm{def:b1:structures:group}{group}, $(ab)^2 =
        abab$, which is $a^2b^2$ only when $a$ and $b$ commute;
        likewise $(ab)^{-1} = b^{-1}a^{-1}$, \omterm{def:b1:structures:order}{order} reversed.
        Every identity imported from school algebra must be
        re-derived from the axioms or flagged as commutative.
  \item \emph{\omterm{def:b1:structures:morphism}{Kernel} versus image.} $\ker f$ lives in the
        \emph{source}, $\operatorname{im} f$ in the target;
        ``$f$ \omterm{def:b1:logic:inj}{injective} iff $\ker f$ trivial''
        (\cref{prop:b1:structures:kernel}) has no analogue with
        the image ($\operatorname{im} f = G'$ is surjectivity).
  \item \emph{\omterm{def:b1:structures:ring}{Rings} are not \omterm{def:b1:structures:group}{groups} for $\times$.} In a \omterm{def:b1:structures:ring}{ring}, most
        elements need not be invertible, and cancellation by $a$
        requires $a$ to be a unit or the \omterm{def:b1:structures:ring}{ring} to be an \omterm{def:b1:structures:field}{integral
        domain}: in $\Z/12\Z$, $\overline3\,\overline2 =
        \overline3\,\overline6$ yet $\overline2 \neq \overline6$
        (\cref{ex:b1:structures:zncomputation}).
\end{enumerate}
\end{remark}

\begin{definition}[Group morphism]\label{def:b1:structures:morphism}
Let $(G, *)$ and $(G', \star)$ be \omterm{def:b1:structures:group}{groups}. A \omterm{def:b1:logic:map}{map} $f \colon G \to G'$
is a \emph{morphism}\index{morphism!of groups} when
\[
\forall x, y \in G, \qquad f(x * y) = f(x) \star f(y).
\]
Then $f(e_G) = e_{G'}$ and $f(x^{-1}) = f(x)^{-1}$. The
\emph{kernel}\index{kernel} and \emph{image} of $f$ are
\[
\ker f = f^{-1}(\{e_{G'}\}) \leq G,
\qquad
\operatorname{im} f = f(G) \leq G' .
\]
A \omterm{def:b1:logic:inj}{bijective} morphism is an \emph{isomorphism}; its inverse \omterm{def:b1:logic:map}{map} is then
automatically a morphism.
\end{definition}

\begin{proof}[Proof of the assertions]
$f(e) = f(e * e) = f(e)\star f(e)$, and cancelling $f(e)$ gives
$e_{G'} = f(e)$. Then $f(x)\star f(x^{-1}) = f(x x^{-1}) = e_{G'}$
identifies $f(x^{-1})$ as the inverse. \omterm{def:b1:structures:morphism}{Kernel}: $e \in \ker f$; if
$x, y \in \ker f$, $f(xy^{-1}) = f(x)f(y)^{-1} = e$; the criterion
applies. Image: same criterion with $f(x)f(y)^{-1} = f(xy^{-1})$.
Inverse of an isomorphism: for $u, v \in G'$, write $u = f(x)$, $v =
f(y)$; then $f^{-1}(u \star v) = f^{-1}(f(xy)) = xy =
f^{-1}(u) f^{-1}(v)$.
\end{proof}

\begin{proposition}[Injectivity via the kernel]\label{prop:b1:structures:kernel}
A \omterm{def:b1:structures:group}{group} \omterm{def:b1:structures:morphism}{morphism} $f$ is \omterm{def:b1:logic:inj}{injective} if and only if $\ker f =
\{e\}$.
\end{proposition}

\begin{proof}
If $f$ is \omterm{def:b1:logic:inj}{injective}, $\ker f$ can only contain the one \omterm{def:b1:logic:map}{preimage} of
$e_{G'}$, which is $e$. Conversely, if $\ker f = \{e\}$ and $f(x) =
f(y)$, then $f(xy^{-1}) = f(x) f(y)^{-1} = e_{G'}$, so $xy^{-1} = e$,
i.e.\ $x = y$.
\end{proof}

\begin{example}\label{ex:b1:structures:expmorphism}
$\exp \colon (\R, +) \to (\R_+^*, \times)$ is a \omterm{def:b1:structures:morphism}{morphism}
($\eu^{x+y} = \eu^x \eu^y$), \omterm{def:b1:logic:inj}{bijective}
(\cref{prop:b1:functions:expln}): the additive and multiplicative
structures are isomorphic --- the historical raison d'être of
logarithms. Another \omterm{def:b1:structures:morphism}{morphism}: $\theta \mapsto \eu^{\iu\theta}$ from
$(\R, +)$ onto the unit circle $(\mathbb{U}, \times)$, with \omterm{def:b1:structures:morphism}{kernel}
$2\pi\Z$.
\end{example}

\begin{example}[The sign morphism]\label{ex:b1:structures:signmorphism}
The \omterm{def:b1:logic:map}{map} $s \colon (\R^*, \times) \to (\{\pm1\}, \times)$ sending
$x$ to its sign is a \omterm{def:b1:structures:morphism}{morphism}: the sign of a product is the
product of the signs. Its \omterm{def:b1:structures:morphism}{kernel} is $\intoo0{+\infty}$ (a
\omterm{def:b1:structures:subgroup}{subgroup}, as \cref{def:b1:structures:morphism} promises), its
image all of $\{\pm1\}$: \omterm{def:b1:logic:inj}{surjective}, massively \omterm{def:b1:logic:inj}{non-injective}.
Two general lessons in miniature. First, a \omterm{def:b1:structures:morphism}{morphism} may crush
information: $s$ remembers nothing of $x$ but one bit, and that
is its virtue --- sign arguments are exactly the computations
that factor through $s$. Second, \omterm{def:b1:structures:morphism}{morphisms} to $\{\pm1\}$ are the
simplest ``invariants'': the signature of \omterm{def:b1:counting:objects}{permutations}, built in
this chapter's weekend problem, is the same phenomenon on the
\omterm{def:b1:structures:group}{group} $\mathfrak S_n$, and the parity arguments it powers all
descend through such a two-valued \omterm{def:b1:structures:morphism}{morphism}.
\end{example}

\begin{definition}[Powers, order of an element]\label{def:b1:structures:order}
In a \omterm{def:b1:structures:group}{group} $G$ (multiplicative notation), set $x^0 = e$, $x^{k+1} =
x^k x$ and $x^{-k} = (x^k)^{-1}$ for $k \in \N$; then $x^{k+l} = x^k
x^l$ for all $k, l \in \Z$, so $k \mapsto x^k$ is a \omterm{def:b1:structures:morphism}{morphism} $(\Z,
+) \to G$ whose image $\langle x \rangle = \{x^k : k \in \Z\}$ is a
\omterm{def:b1:structures:subgroup}{subgroup}, the \omterm{def:b1:structures:subgroup}{subgroup} \emph{generated} by $x$. The
\emph{order}\index{order of an element} of $x$ is the least $m \geq
1$ with $x^m = e$ if one exists (then $\langle x\rangle = \{e, x,
\dots, x^{m-1}\}$ has exactly $m$ elements, and $x^k = e \iff m
\mid k$), and $\infty$ otherwise.
\end{definition}

\begin{example}\label{ex:b1:structures:order}
In $(\C^*, \times)$: $\iu$ has \omterm{def:b1:structures:order}{order} $4$, with $\langle \iu \rangle
= \{1, \iu, -1, -\iu\} = \mathbb{U}_4$; more generally $\omega =
\eu^{2\iu\pi/n}$ has \omterm{def:b1:structures:order}{order} $n$ and $\langle\omega\rangle =
\mathbb{U}_n$. In $(\Z, +)$, every $x \neq 0$ has infinite \omterm{def:b1:structures:order}{order}.
Why the claims in the definition hold: if $x$ has \omterm{def:b1:structures:order}{order} $m$, divide
any $k$ by $m$ ($k = mq + r$, $0 \leq r < m$,
\cref{thm:b1:arith:division}): $x^k = (x^m)^q x^r = x^r$, so the
powers cycle with period $m$, the listed elements are pairwise
distinct by minimality of $m$, and $x^k = e$ forces $r = 0$. \omterm{def:b1:structures:order}{Orders}
of \omterm{def:b1:counting:objects}{permutations} are computed in the weekend problem below.
\end{example}

\begin{example}[Orders inside $\mathbb U_{12}$]\label{ex:b1:structures:orderU12}
What is the \omterm{def:b1:structures:order}{order} of $\omega^k$ in $\mathbb U_n$, for $\omega =
\eu^{2\iu\pi/n}$? One has $(\omega^k)^m = 1$ iff $n \mid km$, and
writing $d = \gcd(n, k)$, $n = dn'$, $k = dk'$ with $\gcd(n', k')
= 1$: $n \mid km \iff n' \mid k'm \iff n' \mid m$ (Gauss's
lemma, \cref{thm:b1:arith:gauss}). The least such $m \geq 1$ is
$n' = \frac{n}{\gcd(n,k)}$. In $\mathbb U_{12}$ for instance,
$\omega^8$ has \omterm{def:b1:structures:order}{order} $\frac{12}{\gcd(12,8)} = 3$ (indeed
$\omega^8 = \eu^{4\iu\pi/3} \in \mathbb U_3$), while $\omega^5$
has \omterm{def:b1:structures:order}{order} $12$: it generates the whole \omterm{def:b1:structures:group}{group}, though it is not
the ``standard'' generator. Counting the generators --- the $k$
with $\gcd(k, n) = 1$ --- recovers the \omterm{cor:b1:arith:bezout}{coprime} counts of
\cref{ex:b1:counting:coprime}: \omterm{def:b1:structures:group}{group} theory and counting meet.
\end{example}

\section{Rings and fields}

\begin{definition}[Ring]\label{def:b1:structures:ring}
A \emph{ring}\index{ring} $(A, +, \times)$ is a \omterm{def:b1:logic:sets}{set} with two laws such
that: $(A, +)$ is an \omterm{def:b1:structures:group}{abelian group} (identity $0$); $\times$ is
associative with an identity $1$; and $\times$ distributes over $+$ on
both sides. The ring is \emph{commutative} when $\times$ is. An
element $a$ is \emph{invertible} (a \emph{unit}) when $ab = ba = 1$
for some $b$; the units form a \omterm{def:b1:structures:group}{group} $(A^\times, \times)$.
\end{definition}

\begin{proof}[Proof that the units form a group]
Stability: if $a, a'$ are units with inverses $b, b'$, then
\[
(aa')(b'b) = a(a'b')b = a\,1\,b = ab = 1,
\qquad (b'b)(aa') = 1
\]
symmetrically, so $aa'$ is a unit. The element $1$ is a unit (its
own inverse), associativity is inherited from $A$, and the
inverse $b$ of a unit $a$ is itself a unit (with inverse $a$).
So $(A^\times, \times)$ satisfies all \omterm{def:b1:structures:group}{group} axioms. Every \omterm{def:b1:structures:group}{group}
in this book that is not built from \omterm{def:b1:counting:objects}{permutations} arises this way:
$\Q^* = \Q^\times$, $\R^*$, $\C^*$, the units of $\Z/n\Z$ below,
and later the invertible matrices (\cref{ch:b1:matrices}).
\end{proof}

\begin{example}\label{ex:b1:structures:rings}
$\Z, \Q, \R, \C$ are commutative \omterm{def:b1:structures:ring}{rings}; $\Z^\times = \{1, -1\}$,
$\Q^\times = \Q^*$. Later: polynomial \omterm{def:b1:structures:ring}{rings} $K[X]$
(\cref{ch:b1:poly}), matrix \omterm{def:b1:structures:ring}{rings} (non-commutative,
\cref{ch:b1:matrices}), and $\Z/n\Z$ below. In every \omterm{def:b1:structures:ring}{ring}, $0 \times
a = 0$ (from distributivity: $0a = (0+0)a = 0a + 0a$), and
$(-1)a = -a$.
\end{example}

\begin{example}[Idempotents: new phenomena in new rings]\label{ex:b1:structures:idempotents}
In $\Z$, the equation $x^2 = x$, i.e.\ $x(x - 1) = 0$, has only
the solutions $0$ and $1$. In $\Z/6\Z$, testing all classes:
$\overline0^2 = \overline0$, $\overline1^2 = \overline1$,
$\overline3^2 = \overline9 = \overline3$ and $\overline4^2 =
\overline{16} = \overline4$ --- \emph{four} idempotents. The two
exotic ones come from zero divisors: $\overline3\,(\overline3 -
\overline1) = \overline3 \times \overline2 = \overline6 =
\overline0$ with neither factor zero. Such computations calibrate
one's instincts: familiar facts about equations survive in
\omterm{def:b1:structures:field}{integral domains} and \omterm{def:b1:structures:field}{fields}, but a general \omterm{def:b1:structures:ring}{ring} can and does
behave differently --- see also the Boolean \omterm{def:b1:structures:ring}{rings} of
\cref{exo:b1:structures:10}, where \emph{every} element is
idempotent.
\end{example}

\begin{proposition}[Binomial theorem in a commutative ring]\label{prop:b1:structures:binomial}
If $a, b$ are elements of a commutative \omterm{def:b1:structures:ring}{ring} (more generally, if
$ab = ba$), then for $n \in \N$:
\[
(a+b)^n = \sum_{k=0}^n \binom nk a^k b^{n-k},
\qquad
a^n - b^n = (a - b) \sum_{k=0}^{n-1} a^k b^{\,n-1-k} .
\]
\end{proposition}

\begin{proof}
The proofs of \cref{thm:b1:counting:binomial} and of the geometric
identity use only associativity, commutativity of the two elements,
and distributivity --- they apply verbatim.
\end{proof}

\begin{example}[The binomial theorem in an unfamiliar ring]\label{ex:b1:structures:binomialring}
Two quick payoffs of the generality. In $\Z/p\Z$ ($p$ \omterm{def:b1:arith:prime}{prime}), the
middle \omterm{def:b1:counting:objects}{binomial coefficients} vanish
(\cref{thm:b1:arith:fermat}'s first step), so the theorem
collapses to the \emph{freshman's dream}
\[
(a + b)^p = a^p + b^p \qquad \text{in } \Z/p\Z ,
\]
a genuine identity there, however criminal it looks over $\R$.
And in any commutative \omterm{def:b1:structures:ring}{ring} containing an element $\varepsilon$
with $\varepsilon^2 = 0$, the theorem truncates:
$(a + \varepsilon)^n = a^n + n\,a^{n-1}\varepsilon$, all higher
terms carrying a factor $\varepsilon^2 = 0$. The coefficient
$n\,a^{n-1}$ of $\varepsilon$ is the derivative of $x^n$ ---
no accident, and a first hint that derivatives are algebra as
much as analysis (compare the formal derivative of
\cref{ch:b1:poly}).
\end{example}

\begin{definition}[Integral domain, field]\label{def:b1:structures:field}
A commutative \omterm{def:b1:structures:ring}{ring} $A \neq \{0\}$ is an \emph{integral
domain}\index{integral domain} when it has no zero divisors: $ab = 0
\implies a = 0$ or $b = 0$. It is a \emph{field}\index{field} when
every nonzero element is invertible. Every field is an integral
domain ($ab = 0$ and $a \neq 0$ give $b = a^{-1}ab = 0$).
\end{definition}

\begin{example}
$\Q$, $\R$, $\C$ are \omterm{def:b1:structures:field}{fields}; $\Z$ is an \omterm{def:b1:structures:field}{integral domain} but not a
\omterm{def:b1:structures:field}{field}. In an \omterm{def:b1:structures:field}{integral domain}, cancellation holds for $\times$: $ab =
ac$ and $a \neq 0$ imply $b = c$.
\end{example}

\section{The ring $\Z/n\Z$}

\begin{definition}\label{def:b1:structures:zn}
Fix $n \in \N^*$. The \omterm{def:b1:arith:congruence}{congruence} classes mod $n$
(\cref{ex:b1:logic:congruence}) form a \omterm{def:b1:logic:sets}{set}
$\Z/n\Z$\index{Z/nZ@$\Z/n\Z$} of $n$ elements, written $\overline 0,
\overline 1, \dots, \overline{n-1}$. The operations
\[
\overline a + \overline b = \overline{a + b},
\qquad
\overline a \times \overline b = \overline{ab}
\]
are well defined --- the classes of the results do not depend on the
representatives, precisely because \omterm{def:b1:arith:congruence}{congruence} is compatible with $+$
and $\times$ (\cref{def:b1:arith:congruence}) --- and make $\Z/n\Z$ a
commutative \omterm{def:b1:structures:ring}{ring}.
\end{definition}

\begin{theorem}[Units of $\Z/n\Z$; the fields $\Z/p\Z$]\label{thm:b1:structures:zp}
\begin{enumerate}
  \item $\overline a$ is invertible in $\Z/n\Z$ if and only if
        $\gcd(a, n) = 1$.
  \item $\Z/n\Z$ is a \omterm{def:b1:structures:field}{field} if and only if $n$ is \omterm{def:b1:arith:prime}{prime}.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) is \cref{prop:b1:arith:invmod} rewritten with classes.

(2) If $n = p$ is \omterm{def:b1:arith:prime}{prime}, every $\overline a \neq \overline 0$ has $p
\nmid a$, so $\gcd(a, p) = 1$: invertible by (1) --- a \omterm{def:b1:structures:field}{field}. If $n =
ab$ with $1 < a, b < n$, then $\overline a\, \overline b = \overline
n = \overline 0$ with $\overline a, \overline b \neq \overline 0$:
zero divisors, so not even an \omterm{def:b1:structures:field}{integral domain}; and $n = 1$ gives the
zero \omterm{def:b1:structures:ring}{ring}, excluded.
\end{proof}

\begin{example}[How many square roots of $1$?]\label{ex:b1:structures:sqrtone}
Solve $x^2 = \overline 1$ in $\Z/8\Z$ and in $\Z/7\Z$. Testing the
eight classes mod $8$: $1^2 = 1$, $3^2 = 9 \equiv 1$, $5^2 = 25
\equiv 1$, $7^2 = 49 \equiv 1$ --- \emph{four} solutions
$\{\overline1, \overline3, \overline5, \overline7\}$, even though
the polynomial $X^2 - 1$ has degree $2$. In the \omterm{def:b1:structures:field}{field} $\Z/7\Z$,
by contrast, $x^2 = \overline1$ means $(x - \overline1)(x +
\overline1) = \overline0$, and a \omterm{def:b1:structures:field}{field} has no zero divisors: $x =
\pm\overline1$, two solutions only. The failure mod $8$ is
traceable: $(3-1)(3+1) = 2 \times 4 = 8 \equiv 0$ without either
factor vanishing. Moral: the familiar rule ``a degree-$d$
equation has at most $d$ roots'' is a theorem about
\emph{\omterm{def:b1:structures:field}{integral domains}} (\cref{cor:b1:poly:nroots} proves it over
\omterm{def:b1:structures:field}{fields}); in \omterm{def:b1:structures:ring}{rings} with zero divisors it silently fails --- which
is exactly why the pairing proof of Wilson's theorem
(\cref{exo:b1:arith:11}) needed $p$ \omterm{def:b1:arith:prime}{prime}.
\end{example}

\begin{example}[Computing in $\Z/n\Z$]\label{ex:b1:structures:zncomputation}
In $\Z/12\Z$: the units are $\overline 1, \overline 5, \overline 7,
\overline{11}$ (the classes \omterm{cor:b1:arith:bezout}{coprime} to $12$), and each is its own
inverse ($5^2 = 25 \equiv 1$, $7^2 = 49 \equiv 1$, $11^2 = 121 \equiv
1$). The equation $\overline 3\, x = \overline 6$ has \emph{three}
solutions ($x \in \{\overline 2, \overline 6, \overline{10}\}$):
without invertibility, no cancellation. In $\Z/11\Z$ by contrast,
every equation $\overline a x = \overline b$ with $\overline a \neq
\overline 0$ has exactly one solution.
\end{example}

\begin{example}[Group axioms as a solving license]\label{ex:b1:structures:solving}
In the \omterm{def:b1:structures:group}{group} $\bigl((\Z/7\Z)^*, \times\bigr)$, solve $\overline
3\,x = \overline 5$. By \cref{prop:b1:structures:rules}~(3) the
solution exists, is unique, and equals $\overline3^{-1}\,
\overline5$; since $\overline3 \times \overline5 = \overline{15}
= \overline1$, the inverse of $\overline 3$ is $\overline 5$, so
\[
x = \overline5 \times \overline5 = \overline{25} = \overline4,
\qquad\text{check: } \overline3 \times \overline4 =
\overline{12} = \overline5 .
\]
The point is less the answer than the guarantee: in a \omterm{def:b1:structures:group}{group},
every such equation is solvable uniquely \emph{before} any
computation, so a solving procedure can never run into ``no
solution'' or ``several''. Compare $\overline3\,x = \overline6$
in $\Z/12\Z$ above, where the guarantee fails --- knowing which
structure one is in is knowing what one may take for granted.
\end{example}

\begin{example}[Direct products]\label{ex:b1:structures:product}
If $G$ and $H$ are \omterm{def:b1:structures:group}{groups}, the product \omterm{def:b1:logic:sets}{set} $G \times H$ with the
componentwise law $(g, h)(g', h') = (gg', hh')$ is a \omterm{def:b1:structures:group}{group}:
axioms are checked coordinate by coordinate, with identity
$(e_G, e_H)$ and inverses $(g^{-1}, h^{-1})$. \omterm{def:b1:structures:order}{Orders} combine by
the lcm: $(g, h)^m = (g^m, h^m)$ is the identity iff the \omterm{def:b1:structures:order}{order}
of $g$ and the \omterm{def:b1:structures:order}{order} of $h$ both divide $m$. Thus in $\Z/2\Z
\times \Z/2\Z$ (additive) every nonzero element has \omterm{def:b1:structures:order}{order} $2$
--- this is exactly the rectangle \omterm{def:b1:structures:group}{group} of
\cref{ex:b1:structures:klein} in coordinates --- while $\Z/4\Z$
has an element of \omterm{def:b1:structures:order}{order} $4$: a second, computation-free proof
that the two \omterm{def:b1:structures:group}{groups} of size $4$ are not isomorphic (an
isomorphism preserves \omterm{def:b1:structures:order}{orders}). Products are the easiest way to
manufacture new \omterm{def:b1:structures:group}{groups} from old, and the plane $\R^2 = \R \times
\R$ of \cref{ch:b1:vspaces} is the construction's most important
instance.
\end{example}

\begin{remark}[Fermat, structurally]
In the \omterm{def:b1:structures:field}{field} $\Z/p\Z$, the nonzero classes form a multiplicative \omterm{def:b1:structures:group}{group}
with $p - 1$ elements, and Fermat's little theorem
(\cref{thm:b1:arith:fermat}) says: every element $x$ of this \omterm{def:b1:structures:group}{group}
satisfies $x^{p-1} = \overline 1$. This is an instance of a general
fact about finite \omterm{def:b1:structures:group}{groups} (Lagrange's theorem), proved in the second
year; the pairing proof of Wilson's theorem
(\cref{exo:b1:arith:11}) already had this group-theoretic flavor.
\end{remark}

\begin{remark}[Interlude: what abstraction buys]\label{rem:b1:structures:interlude}
It is fair to ask what was gained by proving, say,
\cref{prop:b1:structures:unique} for an abstract law rather than
for numbers. The answer is leverage. That two-line argument now
covers, at once: inverses of functions under composition
(\cref{thm:b1:logic:inverse}, whose uniqueness proof it repeats
word for word), inverses mod $n$
(\cref{prop:b1:arith:invmod}), inverses of nonzero reals, of
units in any \omterm{def:b1:structures:ring}{ring}, and --- sight unseen --- of the invertible
matrices of \cref{ch:b1:matrices}, where uniqueness of $A^{-1}$
will not need a single line of proof. The same economy holds for
\cref{prop:b1:structures:kernel} (one injectivity criterion,
reused for linear \omterm{def:b1:logic:map}{maps} in \cref{ch:b1:linmaps}) and for the
\omterm{def:b1:structures:subgroup}{subgroup} criterion. Abstraction here is not generality for its
own sake: it is the refusal to prove the same lemma five times
under five names. The price --- keeping track of which axioms
each \omterm{def:b1:logic:statement}{statement} really used --- is exactly what the exercises of
this chapter train.
\end{remark}

\begin{remark}[Where this chapter is used]\label{rem:b1:structures:whereused}
The vocabulary of this chapter is the grammar of the rest of the
volume. \omterm{def:b1:structures:ring}{Rings} and \omterm{def:b1:structures:field}{fields} organize \cref{ch:b1:poly} ($K[X]$ is a
\omterm{def:b1:structures:ring}{ring} imitating $\Z$) and \cref{ch:b1:fractions} ($K(X)$ is its
\omterm{def:b1:structures:field}{field} of fractions); vector spaces (\cref{ch:b1:vspaces}) are
\omterm{def:b1:structures:group}{abelian groups} with a \omterm{def:b1:structures:field}{field} acting on them; matrices
(\cref{ch:b1:matrices}) form the volume's first seriously
non-commutative \omterm{def:b1:structures:ring}{ring}, and their invertible elements a \omterm{def:b1:structures:group}{group} whose
study is linear algebra itself. \omterm{def:b1:structures:morphism}{Morphisms} and \omterm{def:b1:structures:morphism}{kernels} return as
linear \omterm{def:b1:logic:map}{maps} and null spaces in \cref{ch:b1:linmaps} ---
\cref{prop:b1:structures:kernel} \emph{is} the injectivity
criterion of that chapter, proved once and for all here. The
\omterm{ex:b1:structures:groups}{symmetric group}, star of the weekend problem below, supplies the
signature on which determinants are built in \cref{ch:b1:det}.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:structures:1}
On $E = \R \setminus \{1\}$, define $x * y = x + y - xy$. Prove that
$(E, *)$ is an \omterm{def:b1:structures:group}{abelian group}. \emph{(Identify the identity and the
inverse of $x$; check stability: why is $x * y \neq 1$?)}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:structures:2}
Which of the following are \omterm{def:b1:structures:group}{groups}?
\begin{enumerate}
  \item $(\intoo{0}{+\infty}, \times)$;
  \item $(\{-1, 0, 1\}, +)$;
  \item $(\Q^*, \times)$;
  \item the \omterm{def:b1:logic:sets}{set} of odd integers under addition.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:structures:3}
Write the composition table of the \omterm{ex:b1:structures:groups}{symmetric group} $\mathfrak{S}_3$
of $\{1,2,3\}$ (six bijections: identity, three transpositions, two
$3$-cycles), and exhibit two elements that do not commute.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:structures:4}
Prove that $H = \{z \in \C^* : \abs z = 1\}$ is a \omterm{def:b1:structures:subgroup}{subgroup} of $(\C^*,
\times)$, and that $\R_+^*$ is another; is $H \cup \R_+^*$ a
\omterm{def:b1:structures:subgroup}{subgroup}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:structures:5}
Let $f \colon (\R, +) \to (\C^*, \times)$, $\theta \mapsto
\eu^{\iu\theta}$. Prove that $f$ is a \omterm{def:b1:structures:morphism}{morphism}, compute $\ker f$ and
$\operatorname{im} f$, and deduce from
\cref{prop:b1:structures:kernel} that $f$ is not \omterm{def:b1:logic:inj}{injective}. Restrict
the domain to make it \omterm{def:b1:logic:inj}{injective} on an interval as large as possible.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:structures:6}
Let $H, K$ be \omterm{def:b1:structures:subgroup}{subgroups} of $G$. Prove that $H \cap K$ is a \omterm{def:b1:structures:subgroup}{subgroup},
and that $H \cup K$ is a \omterm{def:b1:structures:subgroup}{subgroup} \emph{only} when $H \subseteq K$ or
$K \subseteq H$. \emph{(If $h \in H \setminus K$ and $k \in K
\setminus H$, where can $hk$ live?)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:structures:7}
A \omterm{def:b1:structures:group}{group} $G$ satisfies $x^2 = e$ for all $x \in G$. Prove that $G$ is
\omterm{def:b1:structures:group}{abelian}. \emph{(Expand $(xy)^2$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:structures:8}
In $\Z/18\Z$: list the units and find the inverse of $\overline 5$;
solve $\overline 5\, x = \overline 7$; solve $\overline 6\, x =
\overline 3$ and $\overline 6\, x = \overline{12}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:structures:9}
Prove that the \omterm{def:b1:logic:sets}{set} $\Z[\sqrt 2] = \{a + b\sqrt 2 : a, b \in \Z\}$ is
a \omterm{def:b1:structures:ring}{ring} (a subring of $\R$), and that $1 + \sqrt 2$ is a unit of it
with infinitely many distinct powers --- so $\Z[\sqrt 2]^\times$ is
infinite, unlike $\Z^\times$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:structures:10}
(Boolean \omterm{def:b1:structures:ring}{rings}) Let $A$ be a \omterm{def:b1:structures:ring}{ring} in which $x^2 = x$ for every $x$.
Prove that $x + x = 0$ for all $x$, and that $A$ is commutative.
\emph{(Expand $(x+x)^2$ and $(x+y)^2$.)} Give an example of such a
\omterm{def:b1:structures:ring}{ring} with $\mathcal{P}(E)$, taking the symmetric difference as
addition and intersection as multiplication.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:structures:11}
Let $G$ be a \omterm{def:b1:structures:group}{group} in which, for some fixed $n \geq 1$, $(xy)^n =
x^n y^n$, $(xy)^{n+1} = x^{n+1}y^{n+1}$ and $(xy)^{n+2} =
x^{n+2}y^{n+2}$ for all $x, y$. Prove that $G$ is \omterm{def:b1:structures:group}{abelian}.
\emph{(From the three identities, derive first $y^n x = x y^n$, then
$y^{n+1} x = x y^{n+1}$, and conclude.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:structures:12}
\begin{enumerate}
  \item Determine all \omterm{def:b1:structures:group}{group} \omterm{def:b1:structures:morphism}{morphisms} from $(\Z, +)$ to $(\Z, +)$.
  \item Prove that the only \omterm{def:b1:structures:group}{group} \omterm{def:b1:structures:morphism}{morphism} from $(\Q, +)$ to
        $(\Z, +)$ is the zero \omterm{def:b1:structures:morphism}{morphism}. \emph{(For $x \in \Q$ and
        $n \in \N^*$, compare $f(x)$ and $n\,f(x/n)$.)}
\end{enumerate}
\end{exercise}

\section{Problem: The symmetric group and the 8-puzzle}

\begin{problem}\label{pb:b1:structures:1}
The \omterm{def:b1:structures:group}{group} $\mathfrak S_n$ of \omterm{def:b1:counting:objects}{permutations} of $\intint1n$ is the
oldest \omterm{def:b1:structures:group}{group} in mathematics and still the most instructive. This
problem builds its structure theory from scratch --- cycles,
generation by transpositions, the \emph{signature} \omterm{def:b1:structures:morphism}{morphism}
$\varepsilon \colon \mathfrak S_n \to \{\pm1\}$ (whose existence is
genuinely nontrivial), and the \omterm{pb:b1:structures:1}{alternating group} $\mathfrak A_n$
generated by $3$-cycles --- then cashes it in on a classical
puzzle: in the $3 \times 3$ sliding-tile game, no sequence of moves
can swap two tiles and leave everything else in place.
\index{symmetric group}\index{signature}\index{alternating group}
\omterm{def:b1:counting:objects}{Permutations} act on $\intint1n$; products $\sigma\tau$ mean
``apply $\tau$ first''; $[\,v_1, \dots, v_n]$ denotes the
\omterm{def:b1:counting:objects}{permutation} sending $i$ to $v_i$.

\medskip
\noindent\textbf{Part I --- Cycles and transpositions.}
\begin{enumerate}
  \item Justify $\abs{\mathfrak S_n} = n!$
        (\cref{thm:b1:counting:counts}). In $\mathfrak S_3$,
        compute both products of $\sigma = [2, 3, 1]$ and $\tau =
        [1, 3, 2]$, and conclude that $\mathfrak S_3$ is not
        \omterm{def:b1:structures:group}{abelian}.
  \item A \emph{$k$-cycle} $(a_1\ a_2\ \dots\ a_k)$ ($k \geq 2$,
        the $a_i$ pairwise distinct) sends $a_1 \mapsto a_2
        \mapsto \dots \mapsto a_k \mapsto a_1$ and fixes
        everything else; its \emph{support} is $\{a_1, \dots,
        a_k\}$. Prove that two cycles with disjoint supports
        commute.
  \item Prove that every $\sigma \in \mathfrak S_n$ is a product
        of cycles with pairwise disjoint supports, and that this
        decomposition is unique up to the \omterm{def:b1:structures:order}{order} of the factors.
        \emph{(Consider, for each $i$, the sequence $i, \sigma(i),
        \sigma^2(i), \dots$: it must return to $i$; the resulting
        \emph{orbits} \omterm{thm:b1:logic:partition}{partition} $\intint1n$, and $\sigma$ acts on
        each as a cycle.)}
  \item Decompose $\sigma = [4, 1, 5, 2, 3, 7, 8, 6] \in
        \mathfrak S_8$ into disjoint cycles. Defining the
        \emph{\omterm{def:b1:structures:order}{order}} of $\sigma$ as in
        \cref{def:b1:structures:order}, prove that the \omterm{def:b1:structures:order}{order} of a
        product of disjoint cycles is the lcm of their lengths,
        and compute the \omterm{def:b1:structures:order}{order} of this $\sigma$.
  \item Prove the telescoping identity
        \[
        (a_1\ a_2\ \dots\ a_k)
        = (a_1\ a_k)(a_1\ a_{k-1})\cdots(a_1\ a_2) ,
        \]
        and conclude that every \omterm{def:b1:counting:objects}{permutation} is a product of
        transpositions. Write the $\sigma$ of question 4 as such a
        product.
  \item Show further that the \emph{adjacent} transpositions
        $(i\ \ i{+}1)$ suffice: for $a < b$,
        \[
        (a\ b) = (a\ \ a{+}1)(a{+}1\ \ a{+}2)\cdots(b{-}1\ \ b)
        \cdots(a{+}1\ \ a{+}2)(a\ \ a{+}1),
        \]
        a product of $2(b - a) - 1$ adjacent transpositions ---
        an \emph{odd} number (this parity will matter twice
        below).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The signature exists.} For $\sigma
\in \mathfrak S_n$, let
\[
N(\sigma) = \#\bigl\{(i, j) : i < j,\ \sigma(i) >
\sigma(j)\bigr\}
\]
be its number of \emph{inversions}, and set $\varepsilon(\sigma) =
(-1)^{N(\sigma)}$.
\begin{enumerate}[resume]
  \item Compute $N$ and $\varepsilon$ for the identity, for a
        transposition $(i\ \ i{+}1)$, and for $[2, 3, 1]$.
  \item Prove that for every $\sigma$ and every adjacent
        transposition $\tau = (i\ \ i{+}1)$:
        $N(\sigma\tau) = N(\sigma) \pm 1$. \emph{(Composing with
        $\tau$ on the right swaps the values in positions $i$ and
        $i + 1$; exactly one pair changes its inversion
        status.)}
  \item Deduce, using question 6, that for \emph{any}
        transposition $\tau$, $\varepsilon(\sigma\tau) =
        -\varepsilon(\sigma)$; conclude that if $\sigma$ is a
        product of $p$ transpositions, then $\varepsilon(\sigma)
        = (-1)^p$ --- in particular the parity of $p$ depends only
        on $\sigma$, not on the chosen factorization --- and that
        $\varepsilon \colon \mathfrak S_n \to \{\pm 1\}$ is a
        \omterm{def:b1:structures:group}{group} \omterm{def:b1:structures:morphism}{morphism}.
  \item Show that a $k$-cycle has signature $(-1)^{k-1}$, and
        that in general $\varepsilon(\sigma) = (-1)^{n -
        c(\sigma)}$, where $c(\sigma)$ is the number of orbits of
        $\sigma$ (fixed points included).
  \item The \emph{\omterm{pb:b1:structures:1}{alternating group}} is $\mathfrak A_n =
        \ker\varepsilon$. Justify that it is a \omterm{def:b1:structures:subgroup}{subgroup}, and prove
        $\abs{\mathfrak A_n} = \frac{n!}2$ for $n \geq 2$.
        \emph{(Fix a transposition $\tau_0$ and consider $\sigma
        \mapsto \sigma\tau_0$.)}
  \item Consistency check on $\sigma = [4, 1, 5, 2, 3, 7, 8, 6]$:
        compute $\varepsilon(\sigma)$ three ways --- by counting
        inversions, from the cycle type via question 10, and from
        your transposition count in question 5.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- $\mathfrak A_n$ is generated by
$3$-cycles.}
\begin{enumerate}[resume]
  \item Let $a, b, c, d$ be pairwise distinct. Verify the two
        identities
        \[
        (a\ b)(a\ c) = (a\ c\ b),
        \qquad
        (a\ b)(c\ d) = (a\ c\ b)(a\ c\ d) .
        \]
  \item Prove that for $n \geq 3$, every element of $\mathfrak
        A_n$ is a product of $3$-cycles. \emph{(An even
        \omterm{def:b1:counting:objects}{permutation} is a product of an even number of
        transpositions; absorb them two at a time.)}
  \item Write $(1\ 2)(3\ 4)$ and the $5$-cycle $(1\ 2\ 3\ 4\ 5)$
        explicitly as products of $3$-cycles.
  \item Prove the conjugation formula: for every $\sigma \in
        \mathfrak S_n$,
        \[
        \sigma\,(a_1\ \dots\ a_k)\,\sigma^{-1}
        = \bigl(\sigma(a_1)\ \dots\ \sigma(a_k)\bigr) .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The 8-puzzle.} Tiles $1, \dots, 8$
slide in a $3 \times 3$ frame with one empty cell; a \emph{move}
slides a tile adjacent to the empty cell into it. Number the cells
$1, \dots, 9$ (row by row; the solved position has tile $i$ in
cell $i$ and the empty cell at cell $9$). Treat the empty cell as
a ninth tile, so a position is a \omterm{def:b1:counting:objects}{permutation} $\sigma \in \mathfrak
S_9$ (tile $\sigma(i)$ sits in cell $i$).
\begin{enumerate}[resume]
  \item Show that a move replaces $\sigma$ by $\sigma \circ \tau$
        where $\tau$ is the transposition of the two cells
        involved; deduce that each move flips
        $\varepsilon(\sigma)$.
  \item Let $d(\sigma)$ be the taxicab distance (rows plus
        columns) between the empty cell's current cell and its
        home cell $9$. Show that each move changes $d$ by $\pm1$,
        so each move also flips $(-1)^{d(\sigma)}$. Conclude that
        \[
        I(\sigma) = \varepsilon(\sigma)\cdot(-1)^{d(\sigma)}
        \]
        is \emph{invariant} under every move.
  \item Prove the puzzle's classical impossibility: the position
        that swaps tiles $7$ and $8$ and leaves everything else
        (including the empty cell) in place cannot be reached
        from the solved position.
  \item We admit the converse (its proof is an instructive but
        lengthy induction): every position with $I = +1$ is
        reachable. Deduce that exactly half of the $8!$ positions
        with the empty cell at home are solvable, i.e.\
        $\frac{8!}2 = 20\,160$.
  \item Deduce from question 20 that the reachable tile
        arrangements with the empty cell at home form exactly the
        \omterm{def:b1:structures:subgroup}{subgroup} $\mathfrak A_8 \leq \mathfrak S_8$.
  \item Applications of the invariant: can one reach (a) the
        position where tiles $1, 2, 3$ are cyclically permuted
        and everything else, empty cell included, is at home?
        (b) the position where tile $5$ and the empty cell have
        exchanged places and all other tiles are at home? Justify
        both answers with $I$.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Synthesis.}
\begin{enumerate}[resume]
  \item Prove that for $n \geq 3$ the only \omterm{def:b1:structures:group}{group} \omterm{def:b1:structures:morphism}{morphisms} $f
        \colon \mathfrak S_n \to \{\pm 1\}$ are the constant
        \omterm{def:b1:structures:morphism}{morphism} and $\varepsilon$. \emph{(Using question 16 and
        commutativity of $\{\pm1\}$, show $f$ takes the same
        value on all transpositions.)}
