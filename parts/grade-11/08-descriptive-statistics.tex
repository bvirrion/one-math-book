\chapter{Descriptive Statistics}\label{ch:g11:stat}

Statistics compresses a list of numbers into a few meaningful summaries:
where the data sits (\omterm{def:g11:stat:mean}{mean}, \omterm{def:g11:stat:median}{median}), and how spread out it is (\omterm{def:g11:stat:quartiles}{quartiles},
\omterm{def:g11:stat:variance}{variance}, \omterm{def:g11:stat:variance}{standard deviation}). This chapter works everything out on one
running example. The same summaries reappear for random variables in
\cref{ch:g11:prob}, and their probabilistic laws in \cref{ch:g12:sums}.

Throughout the chapter, our running dataset is the marks (out of $20$) of
a class of $12$ students:
\[
8,\ 9,\ 10,\ 10,\ 11,\ 12,\ 12,\ 12,\ 13,\ 14,\ 16,\ 17 .
\]

\section{Mean and median}

\begin{definition}[Mean]\label{def:g11:stat:mean}
The \emph{mean}\index{mean} of the values $x_1, x_2, \dots, x_n$ is
\[
\bar x = \frac{x_1 + x_2 + \dots + x_n}{n}.
\]
If the value $x_i$ appears with \omterm{def:g10:stats:series}{frequency} $n_i$ (with
$n_1 + \dots + n_k = n$), then
$\bar x = \frac{n_1 x_1 + \dots + n_k x_k}{n}$.
\end{definition}

\begin{definition}[Median]\label{def:g11:stat:median}
A \emph{median}\index{median} of a sorted dataset is a value that splits
it in half: at least half the values are $\leq$ it, and at least half are
$\geq$ it. In practice, sort the $n$ values; if $n$ is odd, the median is
the middle value; if $n$ is even, take the \omterm{prop:g10:coordgeom:midpoint}{midpoint} of the two middle
values.
\end{definition}

\begin{example}\label{ex:g11:stat:meanmedian}
For the running dataset, the sum is $144$, so $\bar x = \frac{144}{12} =
12$. The two middle values ($6$th and $7$th of the sorted list) are both
$12$, so the \omterm{def:g11:stat:median}{median} is $12$ too. \omterm{def:g11:stat:mean}{Mean} and \omterm{def:g11:stat:median}{median} need not coincide: the
\omterm{def:g11:stat:median}{median} ignores \emph{how far} the extreme values are, the \omterm{def:g11:stat:mean}{mean} does not.
Replacing the top mark $17$ by $20$ moves the \omterm{def:g11:stat:mean}{mean} to $12.25$ but leaves
the \omterm{def:g11:stat:median}{median} at $12$.
\end{example}

\section{Quartiles}

\begin{definition}[Quartiles, interquartile range]\label{def:g11:stat:quartiles}
The \emph{first quartile}\index{quartile} $Q_1$ is the smallest value such
that at least a quarter of the data is $\leq Q_1$; the \emph{third
quartile} $Q_3$ is the smallest value such that at least three quarters of
the data is $\leq Q_3$. The \emph{interquartile range}\index{interquartile range} is $Q_3 - Q_1$: the
width of the central half of the data.
\end{definition}

\begin{example}
$n = 12$: a quarter is $3$, so $Q_1$ is the $3$rd sorted value:
$Q_1 = 10$. Three quarters is $9$, so $Q_3$ is the $9$th value:
$Q_3 = 13$. The \omterm{def:g11:stat:quartiles}{interquartile range} is $3$: the central half of the class
sits within $3$ marks.
\end{example}

\section{Variance and standard deviation}

\begin{definition}[Variance, standard deviation]\label{def:g11:stat:variance}
The \emph{variance}\index{variance} of the dataset is the \omterm{def:g11:stat:mean}{mean} of the
squared deviations from the \omterm{def:g11:stat:mean}{mean}:
\[
V = \frac{1}{n}\sum_{i=1}^{n} (x_i - \bar x)^2,
\]
and the \emph{standard deviation}\index{standard deviation} is
$\sigma = \sqrt{V}$, expressed in the same unit as the data.
\end{definition}

\begin{proposition}[Shortcut formula]\label{prop:g11:stat:shortcut}
The \omterm{def:g11:stat:variance}{variance} is the \omterm{def:g11:stat:mean}{mean} of the squares minus the square of the \omterm{def:g11:stat:mean}{mean}:
\[
V = \frac{1}{n}\sum_{i=1}^{n} x_i^2 \;-\; \bar x^{\,2}.
\]
\end{proposition}

\begin{proof}
Expand each squared deviation:
$(x_i - \bar x)^2 = x_i^2 - 2\bar x\, x_i + \bar x^2$. Summing and
dividing by $n$:
\[
V = \frac1n \sum x_i^2 - 2\bar x \cdot \underbrace{\frac1n \sum
x_i}_{=\ \bar x} + \frac1n \cdot n\bar x^2
= \frac1n\sum x_i^2 - 2\bar x^2 + \bar x^2
= \frac1n\sum x_i^2 - \bar x^2 .
\]
The same identity, for random variables, is
\cref{prop:g11:prob:konig}.
\end{proof}

\begin{example}\label{ex:g11:stat:variance}
For the running dataset, the deviations from $\bar x = 12$ are $-4$,
$-3$, $-2$, $-2$, $-1$, $0$, $0$, $0$, $1$, $2$, $4$, $5$; their squares
sum to
\[
16 + 9 + 4 + 4 + 1 + 0 + 0 + 0 + 1 + 4 + 16 + 25 = 80 .
\]
Hence
\[
V = \frac{80}{12} = \frac{20}{3} \approx 6.67,
\qquad
\sigma = \sqrt{20/3} \approx 2.6 \text{ marks}.
\]
Roughly speaking, a typical mark sits about $2.6$ marks away from the
\omterm{def:g11:stat:mean}{mean}.
\end{example}

\begin{proposition}[Change of units]\label{prop:g11:stat:affine}
If every value is transformed by $y_i = a x_i + b$, then
\[
\bar y = a \bar x + b,
\qquad
V_y = a^2\, V_x,
\qquad
\sigma_y = \abs{a}\,\sigma_x .
\]
\end{proposition}

\begin{proof}
$\bar y = \frac1n \sum (a x_i + b) = a\left(\frac1n\sum x_i\right) +
\frac1n \cdot nb = a\bar x + b$. Then
$y_i - \bar y = a(x_i - \bar x)$: each deviation is scaled by $a$
(the shift $b$ cancels), so each squared deviation is scaled by $a^2$,
and so is their \omterm{def:g11:stat:mean}{mean}. Taking square roots gives
$\sigma_y = \abs a\,\sigma_x$.
\end{proof}

\begin{example}
Temperatures recorded in degrees Celsius with \omterm{def:g11:stat:mean}{mean} $20$ and \omterm{def:g11:stat:variance}{standard
deviation} $3$ convert to Fahrenheit by $F = 1.8\,C + 32$: the \omterm{def:g11:stat:mean}{mean}
becomes $1.8 \times 20 + 32 = 68$, the \omterm{def:g11:stat:variance}{standard deviation}
$1.8 \times 3 = 5.4$. Shifting a dataset does not change its spread;
rescaling it rescales the spread.
\end{example}

\begin{method}[Summarizing and comparing datasets]\label{met:g11:stat:compare}
To compare two datasets, put side by side: \omterm{def:g11:stat:mean}{mean} (or \omterm{def:g11:stat:median}{median}) for
\emph{position}, \omterm{def:g11:stat:variance}{standard deviation} (or \omterm{def:g11:stat:quartiles}{interquartile range}) for
\emph{spread}. The pair (\omterm{def:g11:stat:median}{median}, \omterm{def:g11:stat:quartiles}{interquartile range}) is robust to
extreme values; the pair (\omterm{def:g11:stat:mean}{mean}, \omterm{def:g11:stat:variance}{standard deviation}) uses every value and
feeds \omterm{def:g10:proba:distribution}{probability} theory.
\end{method}

\begin{example}
Two archers each shoot $10$ arrows; scores average $8.5$ for both, but
the \omterm{def:g11:stat:variance}{standard deviations} are $0.5$ and $2.1$. Same level on average --- but
the first archer is far more consistent.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g11:stat:1}
Compute the \omterm{def:g11:stat:mean}{mean}, \omterm{def:g11:stat:median}{median}, \omterm{def:g11:stat:quartiles}{quartiles} and \omterm{def:g11:stat:quartiles}{interquartile range} of the
dataset
\[
5,\ 7,\ 7,\ 8,\ 9,\ 10,\ 11,\ 14 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:stat:2}
Compute the \omterm{def:g11:stat:variance}{variance} and \omterm{def:g11:stat:variance}{standard deviation} of the dataset
$2,\ 4,\ 4,\ 4,\ 5,\ 5,\ 7,\ 9$ (\omterm{def:g11:stat:mean}{mean} first, then the shortcut formula of
\cref{prop:g11:stat:shortcut}).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:stat:3}
A die is rolled $50$ times; the outcomes are summarized by frequencies:
\begin{center}
\begin{tabular}{c|cccccc}
outcome & 1 & 2 & 3 & 4 & 5 & 6\\
\hline
\omterm{def:g10:stats:series}{frequency} & 6 & 9 & 8 & 10 & 9 & 8
\end{tabular}
\end{center}
Compute the \omterm{def:g11:stat:mean}{mean} outcome and the \omterm{def:g11:stat:median}{median}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:stat:4}
A class of $15$ students has \omterm{def:g11:stat:mean}{mean} mark $11.2$. A $16$th student joins and
scores $18$. What is the new class \omterm{def:g11:stat:mean}{mean}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:stat:5}
The \omterm{def:g11:stat:mean}{mean} of the dataset $6, 8, x, 12, 14$ is $10$. Find $x$, then compute
the \omterm{def:g11:stat:variance}{standard deviation} of the completed dataset.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:stat:6}
Marks with \omterm{def:g11:stat:mean}{mean} $12.4$ and \omterm{def:g11:stat:variance}{standard deviation} $2.5$ are rescaled by
$y = \frac{x}{2} + 4$. Give the new \omterm{def:g11:stat:mean}{mean} and the new \omterm{def:g11:stat:variance}{standard deviation}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:stat:7}
Group A ($20$ values) has \omterm{def:g11:stat:mean}{mean} $10$; group B ($30$ values) has \omterm{def:g11:stat:mean}{mean} $15$.
Compute the \omterm{def:g11:stat:mean}{mean} of the $50$ values pooled together. Is it the \omterm{prop:g10:coordgeom:midpoint}{midpoint}
$12.5$? Why not?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:stat:8}
Two production lines fill $1$-kg sugar bags. \omterm{def:g10:proba:sample}{Samples} give:
line 1: \omterm{def:g11:stat:mean}{mean} $1.002$ kg, \omterm{def:g11:stat:variance}{standard deviation} $0.006$ kg; line 2: \omterm{def:g11:stat:mean}{mean}
$1.001$ kg, \omterm{def:g11:stat:variance}{standard deviation} $0.019$ kg. Which line should be
recalibrated first, and why?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:stat:9}
A dataset of $n = 10$ values has $\sum x_i = 70$ and
$\sum x_i^2 = 540$. Compute its \omterm{def:g11:stat:mean}{mean}, \omterm{def:g11:stat:variance}{variance} and \omterm{def:g11:stat:variance}{standard deviation}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:stat:10}
The dataset $1,\ 2,\ 2,\ 3,\ 3,\ 3,\ 4,\ 4,\ 5,\ 23$ contains an
\emph{outlier}.
\begin{enumerate}
  \item Compute the \omterm{def:g11:stat:mean}{mean} and \omterm{def:g11:stat:median}{median} with and without the value $23$.
  \item Compute the \omterm{def:g11:stat:quartiles}{interquartile range} with and without $23$.
  \item Which summaries are robust to the outlier? Conclude with a
        recommendation.
\end{enumerate}
\end{exercise}
