\chapter{Funções e equações quadráticas}\label{ch:g11:quad}

Funções quadráticas são os mais simples funções depois dos lineares, e
o primeiro cujo \omterm{def:g10:functions:graph}{gráficos} são genuinamente curvados. Este capítulo desenvolve o
caixa de ferramentas completa para eles: o \omterm{thm:g11:quad:canonical}{forma canônica}, o \omterm{def:g11:quad:discriminant}{discriminante}, o sinal
de uma expressão quadrática, e a geometria do \omterm{def:g11:quad:parabola}{parábola}.

\section{A forma canônica}

\begin{definition}[Função quadrática]\label{def:g11:quad:quadratic}
UM \emph{função quadrática}\index{função quadrática} é um \omterm{def:g10:functions:function}{função} do
formulário
\[
f(x) = ax^2 + bx + c, \qquad a \neq 0,
\]
onde $a $, $ b $, $ c $ são \omterm{def:g10:numbers:sets}{números reais}. A expressão $ ax^2 + bx + c $ é
também chamado de \emph{trinômio quadrático}.
\end{definition}

\begin{theorem}[Forma canônica]\label{thm:g11:quad:canonical}
Todo \omterm{def:g11:quad:quadratic}{função quadrática} pode ser escrito exatamente de uma maneira como
\[
f(x) = a(x - \alpha)^2 + \beta,
\qquad\text{where } \alpha = -\frac{b}{2a} \text{ and } \beta = f(\alpha).
\]
Este é o \emph{forma canônica}\index{forma canônica} de $ f $.
\end{theorem}

\begin{proof}
Nós \emph{complete o quadrado}. Desde $ a \neq 0$, fatore-o do primeiro
dois termos:
\[
ax^2 + bx + c = a\left(x^2 + \frac{b}{a}x\right) + c .
\]
Agora $ x^2 + \frac{b}{a}x $ é o começo do quadrado
$\left(x + \frac{b}{2a}\right)^2 = x^2 + \frac ba x + \frac{b^2}{4a^2}$,
então $ x^2 + \frac{b}{a}x = \left(x + \frac{b}{2a}\right)^2 -
\frac{b^2}{4a^2}$. Substituindo:
\[
f(x) = a\left(x + \frac{b}{2a}\right)^2 - \frac{b^2}{4a} + c
= a\bigl(x - \alpha\bigr)^2 + \beta,
\]
com $\alpha = -\frac{b}{2a}$ e $\beta = c - \frac{b^2}{4a}$. Avaliando
em $ x = \alpha $ mata o termo ao quadrado, então $\beta = f(\alpha)$. Singularidade:
expandindo $ a(x-\alpha)^2 + \beta $ e identificando os coeficientes de $ x $
forças $\alpha = -\frac{b}{2a}$, e então $\beta $ está determinado.
\end{proof}

\begin{example}\label{ex:g11:quad:canonical}
Deixar $ f(x) = 2x^2 - 8x + 3$. Então $\alpha = -\frac{-8}{2 \times 2} = 2$ e
$\beta = f(2) = 8 - 16 + 3 = -5$, então
\[
f(x) = 2(x - 2)^2 - 5 .
\]
Verifique por expandindo: $2(x^2 - 4x + 4) - 5 = 2x^2 - 8x + 8 - 5 =
2x^2 - 8x + 3$.
\end{example}

\begin{proposition}[Extremos e variações]\label{prop:g11:quad:variations}
Deixar $ f(x) = a(x-\alpha)^2 + \beta $.
\begin{enumerate}
  \item Se $ a > 0$, então $ f $ é diminuindo sobre $\intoc{-\infty}{\alpha}$ e
        \omterm{def:g10:functions:variations}{aumentando} sobre $\intco{\alpha}{+\infty}$: $ f $ tem um \emph{\omterm{def:g10:functions:extrema}{mínimo}}
        $\beta $, alcançado em $ x = \alpha $.
  \item Se $ a < 0$, as variações são invertidas: $ f $ tem um \emph{\omterm{def:g10:functions:extrema}{máximo}}
        $\beta $ no $ x = \alpha $.
\end{enumerate}
\end{proposition}

\begin{proof}
Suponha $ a > 0$ e deixe $\alpha \leq u < v $. Então
\[
f(v) - f(u) = a\bigl((v-\alpha)^2 - (u-\alpha)^2\bigr)
= a\,(v - u)\,\bigl((v-\alpha) + (u-\alpha)\bigr).
\]
Cada fator é positivo: $ a > 0$, $ v - u > 0$, e
$(v-\alpha) + (u-\alpha) > 0$ desde $ v > \alpha $ e $ u \geq \alpha $. Então
$ f(v) > f(u)$: $ f $ é \omterm{def:g10:functions:variations}{aumentando} sobre $\intco{\alpha}{+\infty}$. Para
$ u < v \leq \alpha $ o último fator é negativo, então $ f(v) < f(u)$: $ f $ é
diminuindo lá. Finalmente $ f(x) - f(\alpha) = a(x-\alpha)^2 \geq 0$ para todos
$ x $, então $\beta = f(\alpha)$ é o \omterm{def:g10:functions:extrema}{mínimo}. O caso $ a < 0$ segue por
aplicando o acima para $-f $.
\end{proof}

\begin{example}
Um agricultor tem $100$ metros de cerca para cercar um campo retangular contra uma
parede reta (não é necessária cerca ao longo da parede). Se $ x $ é a largura, o
área fechada é $ A(x) = x(100 - 2x) = -2x^2 + 100x $. Aqui $\alpha =
-\frac{100}{2\times(-2)} = 25$ e $ A(25) = 25 \times 50 = 1250$: a área
é maior para uma largura de $25$ m, dando $1250$ eu $^2$.
\end{example}

\section{Raízes e fatoração}

\begin{definition}[Raiz, discriminante]\label{def:g11:quad:discriminant}
UM \emph{raiz}\index{raiz} do trinômio $ ax^2+bx+c $ é uma solução do
equação $ ax^2 + bx + c = 0$. O \emph{discriminante}\index{discriminante}
do trinômio é o número
\[
\Delta = b^2 - 4ac .
\]
\end{definition}

\begin{theorem}[Resolvendo $ ax^2+bx+c=0$]\label{thm:g11:quad:roots}
\begin{enumerate}
  \item Se $\Delta > 0$, o equação tem dois distintos raízes
        \[
        x_1 = \frac{-b - \sqrt{\Delta}}{2a},
        \qquad
        x_2 = \frac{-b + \sqrt{\Delta}}{2a},
        \]
        e $ ax^2+bx+c = a(x - x_1)(x - x_2)$.
  \item Se $\Delta = 0$, o equação tem o único \omterm{def:g11:quad:discriminant}{raiz}
        $ x_0 = -\frac{b}{2a}$, e $ ax^2+bx+c = a(x - x_0)^2$.
  \item Se $\Delta < 0$, o equação não tem solução real, e o
        trinômio não pode ser fatorado com coeficientes reais.
\end{enumerate}
\end{theorem}

\begin{proof}
Comece do \omterm{thm:g11:quad:canonical}{forma canônica} calculado em \cref{thm:g11:quad:canonical}:
\[
ax^2+bx+c
= a\left(x + \frac{b}{2a}\right)^2 + c - \frac{b^2}{4a}
= a\left[\left(x + \frac{b}{2a}\right)^2 - \frac{\Delta}{4a^2}\right],
\]
desde $ c - \frac{b^2}{4a} = \frac{4ac - b^2}{4a} = -\frac{\Delta}{4a} =
a \times \left(-\frac{\Delta}{4a^2}\right)$.

\emph{1.} Se $\Delta > 0$, então $\frac{\Delta}{4a^2} =
\left(\frac{\sqrt\Delta}{2a}\right)^2$ e o colchete é uma diferença de
dois quadrados:
\[
ax^2+bx+c = a\left(x + \frac{b}{2a} - \frac{\sqrt\Delta}{2a}\right)
\left(x + \frac{b}{2a} + \frac{\sqrt\Delta}{2a}\right)
= a(x - x_2)(x - x_1).
\]
Um produto é zero exatamente quando um de seus fatores é zero, dando aos dois
raízes, distinto porque $\sqrt\Delta \neq 0$.

\emph{2.} Se $\Delta = 0$, o colchete é
$\left(x + \frac{b}{2a}\right)^2$, que desaparece apenas em
$ x_0 = -\frac{b}{2a}$.

\emph{3.} Se $\Delta < 0$, então $-\frac{\Delta}{4a^2} > 0$, então o colchete
é um quadrado mais um número positivo: nunca desaparece, e o equação tem
nenhuma solução real. Uma fatoração $ a(x-x_1)(x-x_2)$ produziria raízes, então
não existe nenhum.
\end{proof}

\begin{example}\label{ex:g11:quad:solving}
Resolver $2x^2 - 8x + 3 = 0$. Aqui $\Delta = (-8)^2 - 4 \times 2 \times 3 =
64 - 24 = 40 > 0$, então há dois raízes:
\[
x_{1,2} = \frac{8 \pm \sqrt{40}}{4} = \frac{8 \pm 2\sqrt{10}}{4}
= 2 \pm \frac{\sqrt{10}}{2}.
\]
Para $ x^2 + x + 1 = 0$: $\Delta = 1 - 4 = -3 < 0$, nenhuma solução real.
\end{example}

\begin{proposition}[Soma e produto das raízes]\label{prop:g11:quad:vieta}
Se $\Delta \geq 0$ e $ x_1, x_2$ são os raízes (igual quando $\Delta = 0$),
então
\[
x_1 + x_2 = -\frac{b}{a},
\qquad
x_1 x_2 = \frac{c}{a} .
\]
Por outro lado, dois números com soma $ s $ e produto $ p $ são as soluções de
$ x^2 - sx + p = 0$.
\end{proposition}

\begin{proof}
Expanda a fatoração de \cref{thm:g11:quad:roots}:
\[
a(x - x_1)(x - x_2) = a x^2 - a(x_1 + x_2)x + a\,x_1 x_2 .
\]
Identificando os coeficientes com $ ax^2 + bx + c $ dá
$-a(x_1+x_2) = b $ e $ a\,x_1x_2 = c $. Pelo contrário, se $ u + v = s $ e
$ uv = p $, então $(x-u)(x-v) = x^2 - sx + p $, então $ u $ e $ v $ são seus raízes.
\end{proof}

\begin{example}
Encontre dois números com soma $10$ e produto $21$. Eles resolvem
$ x^2 - 10x + 21 = 0$; $\Delta = 100 - 84 = 16$, raízes
$\frac{10 \pm 4}{2} = 7$ e $3$. \emph{Rápido \omterm{def:g11:quad:discriminant}{raiz} mancha:} desde o
coeficientes de $ x^2 - 5x + 4$ satisfazer $1 - 5 + 4 = 0$, o número $1$ é um
\omterm{def:g11:quad:discriminant}{raiz}, e o outro é $\frac{c}{a} = 4$ (seu produto deve ser
$\frac ca $).
\end{example}

\section{Sinal de uma quadrática}

\begin{theorem}[Sinal de um trinômio]\label{thm:g11:quad:sign}
Deixar $ f(x) = ax^2+bx+c $ com $ a \neq 0$.
\begin{enumerate}
  \item Se $\Delta > 0$, com raízes $ x_1 < x_2$: $ f(x)$ tem o sinal de $ a $
        fora $\intcc{x_1}{x_2}$, e o sinal de $-a $ sobre
        $\intoo{x_1}{x_2}$.
  \item Se $\Delta = 0$: $ f(x)$ tem o sinal de $ a $ para todos $ x \neq x_0$,
        e desaparece em $ x_0$.
  \item Se $\Delta < 0$: $ f(x)$ tem o sinal de $ a $ para todos $ x $.
\end{enumerate}
Resumidamente: \emph{um trinômio tem o sinal de $ a $, exceto entre seus raízes}.
\end{theorem}

\begin{proof}
\emph{1.} Escrever $ f(x) = a(x-x_1)(x-x_2)$ e siga as placas do
fatores. Se $ x < x_1$: ambos $ x - x_1$ e $ x - x_2$ são negativos, seus
produto é positivo, então $ f(x)$ tem o sinal de $ a $. Se $ x_1 < x < x_2$: o
fatores têm sinais opostos, o produto é negativo e $ f(x)$ tem o
sinal de $-a $. Se $ x > x_2$: ambos os fatores são positivos e $ f(x)$ tem o
sinal de $ a $ de novo.

\emph{2.} $ f(x) = a(x - x_0)^2$ e $(x-x_0)^2 > 0$ para $ x \neq x_0$.

\emph{3.} Da prova de \cref{thm:g11:quad:roots},
$ f(x) = a\left[\left(x + \frac{b}{2a}\right)^2 +
\left(-\frac{\Delta}{4a^2}\right)\right]$, e o colchete é sempre
positivo.
\end{proof}

\begin{method}[Resolvendo desigualdades quadráticas]\label{met:g11:quad:inequalities}
Para resolver $ ax^2 + bx + c \leq 0$ (ou $\geq $, $<$, $>$):
\begin{enumerate}
  \item calcular $\Delta $ e o raízes, se houver;
  \item desenhe o \omterm{def:g10:algebra:signtable}{mesa de sinalização} usando \cref{thm:g11:quad:sign} (sinal de $ a $
        fora do raízes);
  \item leia a solução \omterm{def:g10:numbers:interval}{intervalo}(s), prestando atenção se o
        raízes eles próprios estão incluídos (grande desigualdade) ou não (estrito).
\end{enumerate}
\end{method}

\begin{example}\label{ex:g11:quad:inequality}
Resolver $-x^2 + 3x + 4 \geq 0$. Aqui $\Delta = 9 + 16 = 25$, e o raízes
são $\frac{-3 \pm 5}{-2}$, aquilo é $ x_1 = -1$ e $ x_2 = 4$. Desde
$ a = -1 < 0$, o trinômio é negativo fora do raízes e positivo
entre eles. Com a ampla desigualdade, o conjunto solução é
$\intcc{-1}{4}$.
\end{example}

\section{A parábola}

\begin{definition}[Parábola]\label{def:g11:quad:parabola}
O \omterm{def:g10:functions:graph}{gráfico} de um \omterm{def:g11:quad:quadratic}{função quadrática} $ f(x) = a(x-\alpha)^2+\beta $ é um
\emph{parábola}\index{parábola} com \emph{vértice} $ S(\alpha, \beta)$. Isso
abre para cima se $ a > 0$, para baixo se $ a < 0$, e é simétrico em relação ao
linha vertical $ x = \alpha $.
\end{definition}

\begin{remark}
A simetria é fácil de verificar: para qualquer $ h $,
$ f(\alpha + h) = ah^2 + \beta = f(\alpha - h)$, então dois pontos iguais
distância horizontal da linha $ x = \alpha $ têm a mesma altura.
\end{remark}

\begin{method}[Lendo uma parábola]\label{met:g11:quad:reading}
Para esboçar $ f(x) = ax^2+bx+c $: encontre a orientação (sinal de $ a $), o
vértice $\left(-\frac{b}{2a},\, f\!\left(-\frac{b}{2a}\right)\right)$, o
\omterm{def:g10:numbers:interunion}{interseção} $(0, c)$ com o $ y $-eixo, e o raízes, se houver. O eixo de
simetria $ x = -\frac{b}{2a}$ em seguida, coloca cada ponto computado duas vezes.
\end{method}

\begin{example}
Para $ f(x) = x^2 - 4x + 3$: para cima, vértice $(2, -1)$, atravessa o $ y $-eixo
em $(0,3)$, raízes $1$ e $3$ (identificado: $1 - 4 + 3 = 0$). Por simetria, o
ponto $(4, 3)$ também está na curva, espelhando $(0, 3)$.
\end{example}

\section{Problema: o ponto secreto da parábola}

\begin{problem}[{Problema de fim de semana --- foco e diretriz: por que
antenas parabólicas, faróis e pontes suspensas atraem o
mesma curva, da qual existe essencialmente apenas uma}]
\label{pb:g11:quad:1}
Cada antena parabólica em cada varanda aponta seu receptor para um
ponto interior preciso; cada farol esconde sua lâmpada no
mesmo local especial. Esse ponto --- o \emph{\omterm{pb:g11:quad:1}{foco}} --- é um
segredo o \omterm{def:g11:quad:parabola}{parábola} manteve desde a geometria grega, e o
a fórmula da distância é suficiente para extraí-lo. Ao longo do caminho, isso
problema perfura o \omterm{def:g11:quad:discriminant}{discriminante}, joga projéteis nas paredes,
suspende uma ponte e prova um fato final surpreendente: até
zoom, \emph{há apenas um \omterm{def:g11:quad:parabola}{parábola} no mundo}.

\medskip
\noindent\textbf{Part I --- Trinomial fluency.}
\begin{enumerate}
  \item Resolver $ x^2 - 5x + 6 = 0$, então $2x^2 + x - 6 = 0$, então
        $ x^2 + x + 1 = 0$ (\cref{thm:g11:quad:roots}).
  \item Escrever $2x^2 - 4x - 6$ na forma fatorada e em canônico
        formulário (\cref{thm:g11:quad:canonical}). Juntamente com o
        forma expandida, você agora possui três fantasias de um
        \omterm{def:g10:functions:function}{função}: diga qual recurso cada um exibe rapidamente
        (raízes, vértice, $ y $-interceptar).
  \item Resolver $-x^2 + 4x - 3 \geq 0$ com a regra do sinal
        (\cref{thm:g11:quad:sign}).
  \item Viète como detetive (\cref{prop:g11:quad:vieta},
        continuando Babilônia em \cref{pb:g10:algebra:1}): encontre dois
        números com soma $7$ e produto $12$ escrevendo seus
        equação. Então: um \omterm{def:g11:quad:discriminant}{raiz} de $ x^2 - 5x + c = 0$ é igual
        $2$; encontrar $ c $ e o outro \omterm{def:g11:quad:discriminant}{raiz} sem resolver
        qualquer coisa.
  \item Para quais valores de $ m $ faz $ x^2 + mx + 9 = 0$ tenha um
        duplo \omterm{def:g11:quad:discriminant}{raiz}? Não \omterm{def:g11:quad:discriminant}{raiz}? Dois raízes?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The \omterm{pb:g11:quad:1}{foco} unveiled.}
Trabalhe no \omterm{def:g11:quad:parabola}{parábola} $ y = x^2$, o ponto
$ F\left(0, \frac14\right)$ e a linha horizontal $\delta $ de
equação $ y = -\frac14$.
\begin{enumerate}[resume]
  \item Por um ponto $ P(x, x^2)$ do \omterm{def:g11:quad:parabola}{parábola}, expandir
        $ PF^2 = x^2 + \left(x^2 - \frac14\right)^2$ e mostre isso
        é o quadrado perfeito $\left(x^2 + \frac14\right)^2$.
        Deduzir $ PF $.
  \item Calcule a distância de $ P $ para a linha $\delta $ e
        concluir: \emph{cada ponto do \omterm{def:g11:quad:parabola}{parábola} é exatamente
        tão longe de $ F $ a partir de $\delta $}. ($ F $ é o
        \emph{foco}\index{foco}, $\delta $ o
        \emph{diretora}\index{diretora}.)
  \item Prove o inverso: um ponto $ P(x, y)$ com
        $ PF = \operatorname{dist}(P, \delta)$ deve satisfazer
        $ y = x^2$. (Eleve ao quadrado ambas as distâncias e simplifique.) O
        \omterm{def:g11:quad:parabola}{parábola} é assim um \emph{local}, exatamente como o círculo
        estava em \cref{pb:g10:coordgeom:1}: equidistância de um
        ponto e uma linha, em vez de um ponto.
  \item Para o mais amplo \omterm{def:g11:quad:parabola}{parábola} $ y = \frac{x^2}{4f}$ (com
        $ f > 0$), adapte a questão 6 para mostrar a \omterm{pb:g11:quad:1}{foco} é
        $(0, f)$ e o diretriz $ y = -f $. Aplicação: um
        antena parabólica é $60$ cm de largura e $9$ cm de profundidade --- é
        o aro passa $(30, 9)$. Encontrar $ f $: quão alto acima
        o vértice o receptor deve travar?
  \item A razão pela qual os pratos funcionam: raios que chegam paralelos ao
        eixo todos refletem \emph{através do \omterm{pb:g11:quad:1}{foco}} (e uma lâmpada em
        o \omterm{pb:g11:quad:1}{foco} lança um feixe paralelo --- os faróis são
        pratos correm para trás). Explique por que a questão 7
        a equidistância torna isso plausível (pense nos raios como
        frentes de onda em marcha) e diga qual ferramenta, entrando
        \cref{ch:g11:deriv} (ver \cref{exo:g11:quad:11}), vira
        plausibilidade em prova.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- \omterm{def:g11:quad:parabola}{Parábolas} at work.}
\begin{enumerate}[resume]
  \item Uma bola segue a trajetória
        $ y = x - \frac{x^2}{20}$ ($ x $ e $ y $ em metros). Encontrar
        onde pousa, a posição horizontal do seu ponto mais alto
        ponto e a altura máxima
        (\cref{prop:g11:quad:variations}).
  \item Na mesma trajetória, uma parede fica em $ x = 4$.
        Calcule a altura da bola ali: ela passa por um $3$ eu
        parede? UM $4$ minha parede?
  \item Um arco parabólico abrange $40$ m com altura máxima
        $10$ m: no centro \omterm{def:g10:coordgeom:system}{coordenadas},
        $ y = 10 - \frac{x^2}{40}$. Um caminhão pode $7$ m passe alto
        em $10$ m do centro? No $15$ eu?
  \item A receita de um teatro pelo preço do ingresso $ p $ euros é
        $ R(p) = p\,(120 - 2p)$ (a demanda cai à medida que o preço sobe).
        Encontre o preço que maximiza a receita e o \omterm{def:g10:functions:extrema}{máximo}.
  \item Os principais cabos da Golden Gate são parabólicos
        excelente precisão (uma carga uniforme no convés produz um
        \omterm{def:g11:quad:parabola}{parábola} --- uma corrente pendurada livremente faz \emph{não}: é
        curva, a catenária, precisa da exponencial da nota 12).
        Com vão $1\,280$ m e sag $143$ m, o cabo segue
        $ y = 143\left(\frac{x}{640}\right)^2$ do centro.
        Qual é a altura do cabo acima do seu ponto mais baixo em
        $ x = 320$ eu?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Lines, tangents, and the last
surprise.}
\begin{enumerate}[resume]
  \item Completo \cref{exo:g11:quad:11}: as linhas
        $ y = mx - 1$ através do ponto externo $(0, -1)$ encontrar
        $ y = x^2$ de acordo com o sinal de um \omterm{def:g11:quad:discriminant}{discriminante}.
        Encontre as duas retas tangentes de $(0, -1)$ e seus
        pontos de contato.
  \item Agora as linhas $ y = mx + 1$ através do \emph{interior}
        apontar $(0, 1)$: calcular o \omterm{def:g11:quad:discriminant}{discriminante} do
        \omterm{def:g10:numbers:interunion}{interseção} equação e mostre que cada uma dessas linhas corta o
        \omterm{def:g11:quad:parabola}{parábola} duas vezes. Moral: de dentro, sem tangentes ---
        assim como para um círculo.
  \item A corda focal: \omterm{def:g10:numbers:interunion}{interseção} $ y = x^2$ com o
        linha horizontal que atravessa \omterm{pb:g11:quad:1}{foco}, $ y = \frac14$. O que
        é o comprimento do acorde? Confira a regra geral (vale
        lembrando para pratos): para $ y = \frac{x^2}{4f}$, o
        acorde através do \omterm{pb:g11:quad:1}{foco} paralelo ao diretriz tem
        comprimento $4f $.
  \item A última surpresa: aplique o zoom
        $(x, y) \mapsto (kx, ky)$ para o \omterm{def:g11:quad:parabola}{parábola} $ y = x^2$ e
        mostrar o \omterm{def:g10:functions:function}{imagem} é exatamente $ y = \frac{x^2}{k}$. Concluir:
        cada \omterm{def:g11:quad:parabola}{parábola} $ y = \frac{x^2}{4f}$ é uma unidade ampliada
        \omterm{def:g11:quad:parabola}{parábola} --- ao contrário de círculos e elipses de diferentes
        proporções, \emph{todos \omterm{def:g11:quad:parabola}{parábolas} são semelhantes}.
  \item Final: o parábolaretrato em cinco linhas --- um
        equação com três fantasias (questão 2), um locus
        (questões 7 a 8), um refletor de engenheiro (questões 9,
        10), a trajetória de um projétil e o cabo de uma ponte
        (questões 11 a 15) e uma única curva para cima para ampliar
        (questão 19). Uma frase cada.
\end{enumerate}
\end{problem}
