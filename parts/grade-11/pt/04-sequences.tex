\chapter{Sequências: um primeiro curso}\label{ch:g11:seq}

UM \omterm{def:g11:seq:sequence}{sequência} é uma lista de números produzida por uma regra: os saldos sucessivos
de uma conta poupança, o tamanho de uma população ano após ano. Isto
capítulo estuda as duas famílias que dominam as aplicações ---
\emph{\omterm{def:g11:seq:arithmetic}{aritmética}} \omterm{def:g11:seq:sequence}{sequências}, que crescem em etapas iguais, e
\emph{\omterm{def:g11:seq:geometric}{geométrico}} \omterm{def:g11:seq:sequence}{sequências}, que crescem em proporções iguais. A teoria rigorosa
de limites é desenvolvido em \cref{ch:g12:seq}.

\section{Definindo uma sequência}

\begin{definition}[Sequência]\label{def:g11:seq:sequence}
UM \emph{sequência}\index{sequência} $(u_n)$ atribui a cada inteiro
$ n \geq 0$ (ou $ n \geq 1$) um número real $ u_n $, isso é \emph{termo do índice
$ n $}. Uma sequência pode ser dada
\begin{itemize}
  \item \emph{explicitamente}, por uma fórmula para $ u_n $ em termos de $ n $: por exemplo\
        $ u_n = n^2 + 1$;
  \item \emph{recursivamente}, pelo seu primeiro termo e uma regra para passar de cada
        termo para o próximo: por exemplo\ $ u_0 = 3$ e $ u_{n+1} = 2u_n - 1$.
\end{itemize}
\end{definition}

\begin{example}
Para $ u_n = n^2 + 1$: $ u_0 = 1$, $ u_1 = 2$, $ u_2 = 5$, e $ u_{10} = 101$
diretamente. Para $ u_0 = 3$, $ u_{n+1} = 2u_n - 1$: $ u_1 = 5$, $ u_2 = 9$,
$ u_3 = 17$ --- cada termo precisa do anterior; alcançando $ u_{10}$ leva
dez passos (ou uma fórmula geral, veja \cref{exo:g11:seq:11}).
\end{example}

\section{Sequências aritméticas}

\begin{definition}[Sequência aritmética]\label{def:g11:seq:arithmetic}
UM \omterm{def:g11:seq:sequence}{sequência} é \emph{aritmética}\index{aritmética} com
\emph{diferença comum}\index{diferença comum} $ d $ se cada termo for obtido do anterior
um adicionando $ d $:
\[
u_{n+1} = u_n + d \quad \text{for all } n.
\]
Equivalentemente: a diferença $ u_{n+1} - u_n $ é constante, igual a $ d $.
\end{definition}

\begin{theorem}[Termo geral]\label{thm:g11:seq:arithgeneral}
Se $(u_n)$ é \omterm{def:g11:seq:arithmetic}{aritmética} com primeiro mandato $ u_0$ e \omterm{def:g11:seq:arithmetic}{diferença comum} $ d $,
então
\[
u_n = u_0 + n\,d \quad \text{for all } n \geq 0,
\qquad\text{and more generally } u_n = u_p + (n - p)\,d .
\]
\end{theorem}

\begin{proof}
Para ir de $ u_0$ para $ u_n $, a regra ``adicionar $ d $'' é aplicado $ n $ vezes: um
passo dá $ u_1 = u_0 + d $, dois passos dão $ u_2 = u_0 + 2d $, e depois $ n $
etapas que cada aplicativo contribuiu com uma $ d $, então $ u_n = u_0 + nd $.
(Este “e assim por diante” é tornado rigoroso por indução em \cref{ch:g12:seq}.)
A fórmula geral segue contando o $ n - p $ passos de $ u_p $ para
$ u_n $.
\end{proof}

\begin{theorem}[Soma de inteiros consecutivos]\label{thm:g11:seq:intsum}
Para cada inteiro $ n \geq 1$:
\[
1 + 2 + \dots + n = \frac{n(n+1)}{2}.
\]
Mais genericamente, uma soma de termos consecutivos de um \omterm{def:g11:seq:sequence}{sequência} \omterm{def:g11:seq:arithmetic}{aritmética}
é igual
\[
(\text{number of terms}) \times
\frac{\text{first term} + \text{last term}}{2}.
\]
\end{theorem}

\begin{proof}
Escreva a soma $ S $ duas vezes, a segunda vez na ordem inversa, e adicione
coluna por coluna:
\[
\begin{array}{ccccccccc}
S & = & 1 & + & 2 & + & \dots & + & n\\
S & = & n & + & (n-1) & + & \dots & + & 1\\
\hline
2S & = & (n+1) & + & (n+1) & + & \dots & + & (n+1)
\end{array}
\]
Há $ n $ colunas, cada uma somando $ n + 1$, então $2S = n(n+1)$. Por um
geral \omterm{def:g11:seq:sequence}{sequência} \omterm{def:g11:seq:arithmetic}{aritmética} o mesmo emparelhamento funciona: primeiro $+$ durar $=$
segundo $+$ penúltimo $= \dots $, porque dar um passo à frente em
a extremidade esquerda ($+d $) é compensado por um passo para trás na extremidade direita
($-d $).
\end{proof}

\begin{example}
$1 + 2 + \dots + 100 = \frac{100 \times 101}{2} = 5050$. A soma do
números ímpares $1 + 3 + \dots + 99$ ($50$ termos) é
$50 \times \frac{1 + 99}{2} = 2500$.
\end{example}

\section{Sequências geométricas}

\begin{definition}[Sequência geométrica]\label{def:g11:seq:geometric}
UM \omterm{def:g11:seq:sequence}{sequência} é \emph{geométrico}\index{geométrico} com \emph{comum
proporção}\index{comum
proporção} $ q \neq 0$ se cada termo for obtido do anterior por
multiplicando por $ q $:
\[
u_{n+1} = q\,u_n \quad \text{for all } n.
\]
Equivalentemente, quando nenhum termo desaparece: a razão $\frac{u_{n+1}}{u_n}$ é
constante, igual a $ q $.
\end{definition}

\begin{theorem}[Termo geral]\label{thm:g11:seq:geomgeneral}
Se $(u_n)$ é \omterm{def:g11:seq:geometric}{geométrico} com primeiro mandato $ u_0$ e proporção $ q $, então
\[
u_n = u_0\, q^n \quad \text{for all } n \geq 0,
\qquad\text{and more generally } u_n = u_p\, q^{\,n-p} .
\]
\end{theorem}

\begin{proof}
A mesma contagem de passos que em \cref{thm:g11:seq:arithgeneral}: de $ u_0$ para
$ u_n $, a regra ``multiplicar por $ q $'' é aplicado $ n $ vezes, contribuindo com
fator $ q^n $.
\end{proof}

\begin{theorem}[Soma geométrica]\label{thm:g11:seq:geomsum}
Para cada real $ q \neq 1$ e inteiro $ n \geq 0$:
\[
1 + q + q^2 + \dots + q^n = \frac{1 - q^{\,n+1}}{1 - q}.
\]
\end{theorem}

\begin{proof}
Deixar $ S = 1 + q + \dots + q^n $. Multiplicar por $ q $:
$ qS = q + q^2 + \dots + q^{n+1}$. Subtrair:
\[
S - qS = \bigl(1 + q + \dots + q^n\bigr)
- \bigl(q + q^2 + \dots + q^{n+1}\bigr) = 1 - q^{\,n+1},
\]
porque todo termo intermediário aparece uma vez em cada soma e é cancelado.
Por isso $(1 - q)S = 1 - q^{\,n+1}$, e dividindo por $1 - q \neq 0$ dá o
fórmula.
\end{proof}

\begin{example}\label{ex:g11:seq:chessboard}
$1 + 2 + 4 + \dots + 2^{10} = \frac{1 - 2^{11}}{1 - 2} = 2^{11} - 1 =
2047$: duplicar grãos de arroz nas casas de um tabuleiro de xadrez sobrecarrega
any granary well before the $64$ o quadrado, onde o total é
$2^{64} - 1 \approx 1.8 \times 10^{19}$.
\end{example}

\begin{method}[Reconhecendo o tipo de uma sequência]\label{met:g11:seq:recognize}
Calcular $ u_{n+1} - u_n $ e simplificar. Se o resultado for uma constante $ d $, o
\omterm{def:g11:seq:sequence}{sequência} é \omterm{def:g11:seq:arithmetic}{aritmética}. Caso contrário, calcule $\frac{u_{n+1}}{u_n}$ (termos
diferente de zero) e simplificar: uma constante $ q $ significa \omterm{def:g11:seq:geometric}{geométrico}. Se nenhum dos dois for
constante, o \omterm{def:g11:seq:sequence}{sequência} não é de nenhum tipo --- nunca conclua do
apenas os primeiros termos.
\end{method}

\begin{example}
Para $ u_n = 3 \times 5^n $:
$\frac{u_{n+1}}{u_n} = \frac{3 \times 5^{n+1}}{3 \times 5^n} = 5$ para todos
$ n $: \omterm{def:g11:seq:geometric}{geométrico} com proporção $5$. Para $ u_n = n^2$:
$ u_1 - u_0 = 1$ mas $ u_2 - u_1 = 3$, e
$\frac{u_1}{u_0}$ nem está definido --- nem \omterm{def:g11:seq:arithmetic}{aritmética} nem
\omterm{def:g11:seq:geometric}{geométrico}.
\end{example}

\section{Monotonicidade}

\begin{definition}[Sequência monotônica]\label{def:g11:seq:monotone}
UM \omterm{def:g11:seq:sequence}{sequência} $(u_n)$ é \emph{\omterm{def:g11:func:monotone}{aumentando}} se $ u_{n+1} \geq u_n $ para todos $ n $,
e \emph{diminuindo} se $ u_{n+1} \leq u_n $ para todos $ n $.
\end{definition}

\begin{method}[Estudando a monotonicidade]\label{met:g11:seq:monotonicity}
Estude o sinal de $ u_{n+1} - u_n $. Para \omterm{def:g11:seq:sequence}{sequências} com termos positivos, um
em vez disso, pode comparar $\frac{u_{n+1}}{u_n}$ para $1$.
\end{method}

\begin{example}
Um \omterm{def:g11:seq:sequence}{sequência} \omterm{def:g11:seq:arithmetic}{aritmética} é \omterm{def:g11:func:monotone}{aumentando} quando $ d \geq 0$
($ u_{n+1} - u_n = d $), diminuindo quando $ d \leq 0$. UM \omterm{def:g11:seq:sequence}{sequência} geométrica
com $ u_0 > 0$ e $ q > 1$ é \omterm{def:g11:func:monotone}{aumentando}:
$ u_{n+1} - u_n = u_0 q^n (q - 1) > 0$; com $ u_0 > 0$ e $0 < q < 1$ isso é diminuindo.
\end{example}

\section{Comportamento de longo prazo, informalmente}

O que acontece com $ u_n $ quando $ n $ fica muito grande? Por um \omterm{def:g11:seq:arithmetic}{aritmética}
\omterm{def:g11:seq:sequence}{sequência} com $ d > 0$, os termos $ u_0 + nd $ exceder qualquer número fixo
eventualmente. Por um \omterm{def:g11:seq:sequence}{sequência} geométrica com $0 < q < 1$, os termos
$ u_0 q^n $ encolher em direção $0$: multiplicando repetidamente por $0.9$, digamos, corrói
qualquer valor inicial. E para $ q > 1$ os termos explodem, como em
\cref{ex:g11:seq:chessboard}.

\begin{remark}
Estas declarações podem ser perfeitamente precisas --- ``os termos eventualmente
permaneça dentro de qualquer distância $0$'' --- e provou. Essa é a teoria
de \emph{limites}, tema de abertura de \cref{ch:g12:seq}.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g11:seq:1}
Para cada \omterm{def:g11:seq:sequence}{sequência}, calcular $ u_1$, $ u_2$, $ u_3$:
\[
u_n = \frac{n}{n+1}; \qquad
u_0 = 5,\ u_{n+1} = 3u_n - 2; \qquad
u_n = (-1)^n\,n .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:seq:2}
$(u_n)$ é \omterm{def:g11:seq:arithmetic}{aritmética} com $ u_0 = 7$ e $ d = -3$. Calcular $ u_{10}$ e
$ u_{25}$. $(v_n)$ é \omterm{def:g11:seq:arithmetic}{aritmética} com $ v_3 = 11$ e $ v_8 = 26$. Encontre o
\omterm{def:g11:seq:arithmetic}{diferença comum} e $ v_0$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:seq:3}
$(u_n)$ é \omterm{def:g11:seq:geometric}{geométrico} com $ u_0 = 5$ e $ q = 2$. Calcular $ u_8$. $(v_n)$ é
\omterm{def:g11:seq:geometric}{geométrico} com termos positivos, $ v_2 = 12$ e $ v_4 = 48$. Encontre a proporção

\end{exercise}

Calcular

1 + 2 + 3 + \dots + 500, \qquad

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:seq:5}
Determine se cada \omterm{def:g11:seq:sequence}{sequência} é \omterm{def:g11:seq:arithmetic}{aritmética}, \omterm{def:g11:seq:geometric}{geométrico}, ou nenhum:

u_n = 4n - 1; \qquad
v_n = \frac{2^n}{3^{n+1}}; \qquad
w_n = n^2 + n .

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:seq:6}
Um teatro tem $20$ linhas: $16$ assentos na primeira fila, e cada fila tem

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:seq:7}

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:seq:8}
Todo mês, um poupador deposita $100$ euros numa conta que paga
$0.2\%$ juros por mês sobre o saldo existente (os juros são creditados
pouco antes do depósito). Deixar $ c_n $ seja o equilíbrio logo após o $ n $-º
depósito, então $ c_1 = 100$ e $ c_{n+1} = 1.002\,c_n + 100$. Calcular $ c_2$
e $ c_3$ e explique por que $(c_n)$ não é nenhum dos dois \omterm{def:g11:seq:arithmetic}{aritmética} nem \omterm{def:g11:seq:geometric}{geométrico}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:seq:9}
Estude a monotonicidade do \omterm{def:g11:seq:sequence}{sequências}

v_n = \frac{3^n}{n!}\ (n \geq 1),
\]
onde $ n! = 1 \times 2 \times \dots \times n $. (Para $(v_n)$, comparar

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:seq:10}

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:seq:11}
Deixar $ u_0 = 3$ e $ u_{n+1} = 2u_n - 1$.
\begin{enumerate}

  \item Deixar $ v_n = u_n - 1$. Mostre isso $(v_n)$ é \omterm{def:g11:seq:geometric}{geométrico}, dê o seu

  \item Deduza uma fórmula explícita para $ u_n $ e verifique sua conjectura.
\end{enumerate}
\end{exercise}

\section{Problema: A torre dos coelhos de Brahma e Fibonacci}

\begin{problem}[{Problema de fim de semana --- duas recorrências lendárias:
a torre que acaba com o mundo, a sequência que cresce como ouro,
e o truque auxiliar que doma os empréstimos}]
\label{pb:g11:seq:1}
Dois \omterm{def:g11:seq:sequence}{sequências} governar o folclore da matemática. Um conta o
movimentos da torre de Brahma --- sessenta e quatro discos dourados cujos
transferência, diz a lenda, acabará com o mundo. A outra conta
Coelhos de Fibonacci e esconde a proporção áurea. Nem é
\omterm{def:g11:seq:arithmetic}{aritmética}, nem é \omterm{def:g11:seq:geometric}{geométrico} --- e ambos se rendem a isso
armas do capítulo: recorrências, \omterm{def:g11:seq:geometric}{geométrico} somas
(\cref{thm:g11:seq:geomsum}) e o \omterm{def:g11:seq:sequence}{sequência} auxiliar truque de
\cref{exo:g11:seq:11}, que também calcula sua hipoteca.

\medskip
\noindent\textbf{Part I --- The tower of Brahma.}
O quebra-cabeça: $ n $ discos de diminuindo tamanho são empilhados no pino A;
mova toda a pilha para o pino C, um disco de cada vez, nunca colocando
um disco maior em um menor (o pino B pode ajudar). Deixar $ h_n $ seja o
número \omterm{def:g10:functions:extrema}{mínimo} de movimentos.
\begin{enumerate}
  \item Jogue (com moedas) e grave $ h_1$, $ h_2$, $ h_3$.
  \item Explique a estratégia por trás da recorrência
        $ h_{n+1} = 2h_n + 1$: o que deve acontecer antes e depois
        o maior disco se move?
  \item Resolva a recorrência com o truque de
        \cref{exo:g11:seq:11}: definir $ v_n = h_n + 1$, mostrar
        $(v_n)$ é \omterm{def:g11:seq:geometric}{geométrico}, e concluir $ h_n = 2^n - 1$.
  \item A torre da lenda tem $64$ discos, e os monges se movem
        um disco por segundo. Usando $2^{10} = 1024 \approx 10^3$,
        estimar o tempo de transferência em anos (um ano é cerca de
        $3 \times 10^7$ segundos; comparar
        \cref{ex:g11:seq:chessboard}, o mesmo gigante em outro
        história). Devemos nos preocupar?
  \item Por que nenhuma estratégia pode vencer $2^n - 1$ se move? Argumente que
        \emph{qualquer} solução obedece
        $ h_{n+1} \geq 2 h_n + 1$: o que deve ser verdade para o topo
        $ n $ discos logo antes e logo depois do disco inferior
        mover?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Fibonacci.}
Definir $ F_1 = F_2 = 1$ e $ F_{n+2} = F_{n+1} + F_n $ (cada termo
a soma dos dois antes --- a regra de contagem de ritmo do
Volume do Ensino Médio, agora com nome europeu).
\begin{enumerate}[resume]
  \item Lista $ F_1$ através $ F_{12}$.
  \item Mostre isso $(F_n)$ não é nenhum dos dois \omterm{def:g11:seq:arithmetic}{aritmética} nem \omterm{def:g11:seq:geometric}{geométrico},
        mas que é estritamente \omterm{def:g11:func:monotone}{aumentando} de $ n = 2$ sobre
        (\cref{met:g11:seq:monotonicity} e a recorrência).
  \item Prove a identidade da soma
        \[
        F_1 + F_2 + \dots + F_n = F_{n+2} - 1
        \]
        telescópica: escreva cada $ F_k $ como
        $ F_{k+2} - F_{k+1}$ e observe a soma desmoronar. Verifique
        isso para $ n = 6$.
  \item Prove a identidade dos quadrados
        $ F_1^2 + F_2^2 + \dots + F_n^2 = F_n F_{n+1}$,
        telescópica com
        $ F_k F_{k+1} - F_{k-1} F_k = F_k^2$. Verifique para
        $ n = 4$. (Foto: quadrados de lados
        $1, 1, 2, 3, 5, \dots $ ladrilhar um retângulo --- o famoso
        Esqueleto da espiral de Fibonacci.)
  \item Estados de identidade da Cassini
        $ F_{n+1} F_{n-1} - F_n^2 = (-1)^n $. Verifique para
        $ n = 4, 5, 6$ --- e reconhecer o motor do
        truque do quadrado de fuga praticado no ensino médio
        problema de áreas do volume.
  \item Mostre pela recorrência que
        $ F_{n+2} \geq 2 F_n $: Fibonacci pelo menos dobra a cada
        duas etapas --- cresce pelo menos tão rápido quanto um \omterm{def:g11:seq:geometric}{geométrico}
        \omterm{def:g11:seq:sequence}{sequência} de proporção $\sqrt2$.
  \item Calcule as proporções $ r_n = \frac{F_{n+1}}{F_n}$ para
        $ n = 3$ para $10$ (três decimais). Admitindo que eles
        estabelecer um limite $ L $, passe a relação
        $ r_{n+1} = 1 + \frac{1}{r_n}$ até o limite e resolva:
        qual número de \cref{pb:g10:algebra:1} faça os coelhos
        adoração?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The auxiliary trick, at the
bank.}
\begin{enumerate}[resume]
  \item Generalizar \cref{exo:g11:seq:11}: para
        $ u_{n+1} = a\,u_n + b $ com $ a \neq 1$, deixar
        $\ell = \frac{b}{1 - a}$ (o \omterm{pb:g10:functions:1}{ponto fixo}). Mostrar
        $ v_n = u_n - \ell $ é \omterm{def:g11:seq:geometric}{geométrico} de proporção $ a $ e
        concluir
        $ u_n = a^n (u_0 - \ell) + \ell $.
  \item Um empréstimo: $10\,000$ euros em $1\,\%$ juros por mês,
        reembolsado $300$ euros por mês, então a dívida obedece
        $ d_{n+1} = 1.01\,d_n - 300$. Aplique a questão 13 (fixo
        ponto primeiro!) para obter uma fórmula explícita para $ d_n $.
  \item Com uma calculadora, encontre o primeiro mês em que o
        a dívida é liquidada e o valor total reembolsado. Quanto
        o empréstimo em si custou?
  \item Uma cidade de $50\,000$ habitantes cresce $2\,\%$ um ano
        e dá as boas-vindas $1\,000$ recém-chegados além de:
        $ p_{n+1} = 1.02\,p_n + 1000$. Dê a fórmula explícita
        e a população depois $10$ anos.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The two royal families.}
\begin{enumerate}[resume]
  \item Calcular $1 + 2 + 3 + \dots + 1000$
        (\cref{thm:g11:seq:intsum} --- soma do pequeno Gauss de
        o volume do Ensino Médio, agora oficial), e
        $1 + 2 + 4 + \dots + 2^{19}$
        (\cref{thm:g11:seq:geomsum}).
  \item Calcule a soma dos \omterm{def:g11:seq:sequence}{sequência} \omterm{def:g11:seq:arithmetic}{aritmética}
        $7, 12, 17, \dots, 502$ (quantos termos?).
  \item Plano de poupança: $100$ euros depositados todos os meses, rendendo
        $0.5\,\%$ mensal; depois do $ n $-º depositar o saldo
        é $100\left(1.005^{n-1} + \dots + 1.005 + 1\right)$.
        Calcule o saldo depois $5$ anos ($ n = 60$).
  \item Finale --- o kit do domador de \omterm{def:g11:seq:sequence}{sequência}: explícito contra
        descrições recorrentes; as duas famílias reais e
        suas fórmulas de soma; o auxiliar \omterm{def:g11:seq:sequence}{sequência} isso vira
        recorrências afins em \omterm{def:g11:seq:geometric}{geométrico} uns; e Fibonacci,
        primeiro cidadão fora de ambas as famílias, hoje domesticado por
        identidades e matrizes de espera (nota 12) e limites
        para captura completa. Uma frase cada.
\end{enumerate}
