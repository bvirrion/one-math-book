\chapter{Trigonometria: O Círculo Unitário}\label{ch:g11:trigo}

A trigonometria começa em triângulos retângulos, mas seu verdadeiro lar é o
\emph{círculo unitário}, onde \omterm{def:g11:trigo:cossin}{cosseno} e \omterm{def:g11:trigo:cossin}{seno} tornar-se funções de uma arbitrariedade
número real. Este capítulo instala o \omterm{def:g11:trigo:radian}{radiano}, o enrolamento do real
linha ao redor do círculo, os valores exatos para saber de cor e o
simetrias que geram todas as identidades clássicas. O estudo de
$\cos $ e $\sin $ como funções --- variações, derivados --- é transportado
fora em \cref{ch:g12:trigo}.

\section{Radianos e o enrolamento}

\begin{definition}[Radiano]\label{def:g11:trigo:radian}
Deixar $\mathcal C $ seja o círculo de raio $1$ centrado na origem $ O $,
e $ I(1, 0)$. O \emph{radiano}\index{radiano} medida de um ângulo no
centro de $\mathcal C $ é o comprimento do arco que ele corta $\mathcal C $.
Como o círculo completo tem circunferência $2\pi $:
\[
360^\circ = 2\pi \text{ rad},
\qquad
180^\circ = \pi \text{ rad},
\qquad
1 \text{ rad} = \frac{180^\circ}{\pi} \approx 57.3^\circ .
\]
\end{definition}

\begin{method}[Convertendo]\label{met:g11:trigo:convert}
Graus e \omterm{def:g11:trigo:radian}{radianos} são proporcionais: multiplique por $\frac{\pi}{180}$ ir
de graus a \omterm{def:g11:trigo:radian}{radianos}, por $\frac{180}{\pi}$ o outro lado. Por exemplo
$60^\circ = 60 \times \frac{\pi}{180} = \frac{\pi}{3}$, e
$\frac{3\pi}{4} = \frac{3 \times 180^\circ}{4} = 135^\circ $.
\end{method}

\begin{definition}[Enrolando a linha no círculo]\label{def:g11:trigo:winding}
Para cada um número real $ t $, associe o ponto $ M(t)$ de $\mathcal C $
obtido caminhando uma distância $\abs t $ ao longo do círculo de $ I $,
sentido anti-horário se $ t \geq 0$, no sentido horário se $ t < 0$. Desde o
circunferência é $2\pi $, os números $ t $ e $ t + 2k\pi $ ($ k \in \Z $)
chegar ao mesmo ponto: $ M(t + 2k\pi) = M(t)$.
\end{definition}

\begin{example}
$ M(0) = I $; $ M\!\left(\frac{\pi}{2}\right) = (0, 1)$, o topo do
círculo; $ M(\pi) = (-1, 0)$; $ M\!\left(-\frac{\pi}{2}\right) = (0, -1)$;
e $ M\!\left(\frac{9\pi}{4}\right) = M\!\left(\frac{\pi}{4} +
2\pi\right) = M\!\left(\frac{\pi}{4}\right)$.
\end{example}

\section{Cosseno e seno}

\begin{definition}[Cosseno e seno]\label{def:g11:trigo:cossin}
Para cada real $ t $, o \emph{cosseno}\index{cosseno} e
\emph{seno}\index{seno} de $ t $ são os \omterm{def:g10:coordgeom:system}{coordenadas} do ponto $ M(t)$ de
o círculo unitário:
\[
M(t) = (\cos t,\ \sin t).
\]
\end{definition}

\begin{proposition}[Primeiras propriedades]\label{prop:g11:trigo:first}
Para todos $ t \in \R $ e $ k \in \Z $:
\[
-1 \leq \cos t \leq 1, \qquad
-1 \leq \sin t \leq 1, \qquad
\cos^2 t + \sin^2 t = 1,
\]
\[
\cos(t + 2k\pi) = \cos t, \qquad \sin(t + 2k\pi) = \sin t .
\]
\end{proposition}

\begin{proof}
$ M(t)$ está no círculo unitário, então é \omterm{def:g10:coordgeom:system}{coordenadas} estão entre $-1$ e
$1$, e eles satisfazem o círculo equação $ x^2 + y^2 = 1$, que é o
identidade $\cos^2 t + \sin^2 t = 1$. Periodicidade reafirmada
$ M(t + 2k\pi) = M(t)$ (\cref{def:g11:trigo:winding}).
\end{proof}

\begin{example}\label{ex:g11:trigo:identity}
Se $\sin t = \frac35$ e $ t \in \intcc{\frac\pi2}{\pi}$ (segundo
quadrante), então $\cos^2 t = 1 - \frac{9}{25} = \frac{16}{25}$, então
$\cos t = \pm\frac45$; no segundo quadrante o \omterm{def:g10:coordgeom:system}{abscissa} é negativo,
daí $\cos t = -\frac45$.
\end{example}

Os valores para saber de cor:
\begin{center}
\begin{tabular}{c|ccccc}
$ t $ & $0$ & $\dfrac{\pi}{6}$ & $\dfrac{\pi}{4}$ & $\dfrac{\pi}{3}$ & $\dfrac{\pi}{2}$\\[6pt]
\hline\rule{0pt}{14pt}%
$\cos t $ & $1$ & $\dfrac{\sqrt3}{2}$ & $\dfrac{\sqrt2}{2}$ & $\dfrac12$ & $0$\\[6pt]
$\sin t $ & $0$ & $\dfrac12$ & $\dfrac{\sqrt2}{2}$ & $\dfrac{\sqrt3}{2}$ & $1$
\end{tabular}
\end{center}

\begin{remark}
Um auxílio à memória: o \omterm{def:g11:trigo:cossin}{senos} ler
$\frac{\sqrt0}{2}, \frac{\sqrt1}{2}, \frac{\sqrt2}{2}, \frac{\sqrt3}{2},
\frac{\sqrt4}{2}$, e o \omterm{def:g11:trigo:cossin}{cossenos} são a mesma lista invertida.
\end{remark}

\section{Ângulos associados}

\begin{proposition}[Ângulos associados]\label{prop:g11:trigo:associated}
\index{ângulos associados}
Para todos $ t \in \R $:
\[
\begin{aligned}
\cos(-t) &= \cos t, & \sin(-t) &= -\sin t,\\
\cos(\pi - t) &= -\cos t, & \sin(\pi - t) &= \sin t,\\
\cos(\pi + t) &= -\cos t, & \sin(\pi + t) &= -\sin t,\\
\cos\left(\tfrac{\pi}{2} - t\right) &= \sin t, &
\sin\left(\tfrac{\pi}{2} - t\right) &= \cos t .
\end{aligned}
\]
\end{proposition}

\begin{proof}
Cada linha expressa uma simetria do círculo.

$ M(-t)$ é o reflexo de $ M(t)$ no $ x $-eixo: o \omterm{def:g10:coordgeom:system}{abscissa} é
inalterado, o \omterm{def:g10:coordgeom:system}{ordenar} muda de sinal.

$ M(\pi - t)$ é o reflexo de $ M(t)$ no $ y $-eixo: caminhando
para trás de $\pi $ por $ t $ pousa oposta (horizontalmente) à caminhada
para frente de $0$ por $ t $. O \omterm{def:g10:coordgeom:system}{abscissa} muda de sinal, o \omterm{def:g10:coordgeom:system}{ordenar} é
inalterado.

$ M(\pi + t)$ é diametralmente oposto $ M(t)$ (meia volta): ambos
\omterm{def:g10:coordgeom:system}{coordenadas} mudar de sinal.

$ M\!\left(\frac\pi2 - t\right)$ é o reflexo de $ M(t)$ no
linha diagonal $ y = x $, que troca os dois \omterm{def:g10:coordgeom:system}{coordenadas}.
\end{proof}

\begin{example}
$\cos\frac{2\pi}{3} = \cos\left(\pi - \frac\pi3\right)
= -\cos\frac\pi3 = -\frac12$, e
$\sin\frac{7\pi}{6} = \sin\left(\pi + \frac\pi6\right)
= -\sin\frac\pi6 = -\frac12$. A tabela de cinco valores, ampliada pelo
simetrias, cobre todo o círculo.
\end{example}

\section{Resolvendo equações trigonométricas}

\begin{theorem}[As equações $\cos t = \cos a $ e $\sin t = \sin a $]
\label{thm:g11:trigo:equations}
Para \omterm{def:g10:numbers:sets}{números reais} $ t $ e $ a $:
\[
\cos t = \cos a \iff t = a + 2k\pi \ \text{or}\ t = -a + 2k\pi
\quad (k \in \Z),
\]
\[
\sin t = \sin a \iff t = a + 2k\pi \ \text{or}\ t = \pi - a + 2k\pi
\quad (k \in \Z).
\]
\end{theorem}

\begin{proof}
Dois pontos do círculo unitário têm a mesma \omterm{def:g10:coordgeom:system}{abscissa} exatamente quando eles estão
iguais ou reflexos um do outro no $ x $-eixo; por
\cref{prop:g11:trigo:associated}, o reflexo de $ M(a)$ é $ M(-a)$. Então
$\cos t = \cos a $ significa $ M(t) = M(a)$ ou $ M(t) = M(-a)$, ou seja\
$ t = \pm a $ até um múltiplo de $2\pi $. Da mesma forma, igual ordenadas significa
pontos iguais ou reflexões no $ y $-eixo e o reflexo de $ M(a)$
é $ M(\pi - a)$.
\end{proof}

\begin{method}[Resolvendo $\cos t = c $ em um intervalo]\label{met:g11:trigo:solve}
\begin{enumerate}
  \item Encontrar \emph{um} ângulo $ a $ com $\cos a = c $, da tabela de
        valores conhecidos.
  \item Escreva as soluções gerais $ t = \pm a + 2k\pi $
        (\cref{thm:g11:trigo:equations}).
  \item Escolha o \omterm{def:g10:numbers:sets}{inteiros} $ k $ que pousam dentro do solicitado \omterm{def:g10:numbers:interval}{intervalo},
        e liste as soluções.
\end{enumerate}
Proceda da mesma forma para $\sin t = c $ com $ t = a + 2k\pi $ ou
$ t = \pi - a + 2k\pi $.
\end{method}

\begin{example}\label{ex:g11:trigo:solve}
Resolver $\cos t = \frac12$ sobre $\intoc{-\pi}{\pi}$. Uma solução de
$\cos a = \frac12$ é $ a = \frac{\pi}{3}$. As soluções gerais são
$ t = \frac\pi3 + 2k\pi $ e $ t = -\frac\pi3 + 2k\pi $. Dentro
$\intoc{-\pi}{\pi}$, apenas $ k = 0$ contribui:
$ t \in \left\{-\frac\pi3,\ \frac\pi3\right\}$.

Resolver $\sin t = -\frac{\sqrt2}{2}$ sobre $\intco{0}{2\pi}$. Um ângulo é
$ a = -\frac\pi4$; as soluções são $ t = -\frac\pi4 + 2k\pi $ e
$ t = \pi + \frac\pi4 + 2k\pi = \frac{5\pi}{4} + 2k\pi $. Em
$\intco{0}{2\pi}$: $ t = \frac{7\pi}{4}$ (de $ k = 1$) e
$ t = \frac{5\pi}{4}$.
\end{example}

\section{Problema: A roda gigante e a maré}

\begin{problem}[{Problema de fim de semana --- radianos como distância percorrida,
e o círculo unitário como o relógio mestre de cada período
fenômeno}]
\label{pb:g11:trigo:1}
Uma roda gigante leva você ao redor de um círculo; a maré carrega o
a água do porto para cima e para baixo no mesmo círculo matemático. Cada
fenômeno periódico --- rodas, marés, som, estações --- lê
sua programação está fora do círculo unitário deste capítulo
(\cref{def:g11:trigo:cossin}) e o equações de
\cref{met:g11:trigo:solve} diga os horários exatos. Este problema
converte, enrola, modela e resolve --- e se estabelece no caminho por que
matemáticos medem ângulos em \omterm{def:g11:trigo:radian}{radianos}.

\medskip
\noindent\textbf{Part I --- \omterm{def:g11:trigo:radian}{Radianos}, the rolled distance.}
\begin{enumerate}
  \item Converter para \omterm{def:g11:trigo:radian}{radianos}: $30^\circ $, $135^\circ $,
        $300^\circ $; e em graus: $\frac{3\pi}{4}$,
        $\frac{5\pi}{6}$ (\cref{met:g11:trigo:convert}).
  \item Todo o ponto de \omterm{def:g11:trigo:radian}{radianos}: um arco de ângulo $ t $ (em
        \omterm{def:g11:trigo:radian}{radianos}) em um círculo de raio $ r $ tem comprimento exatamente
        $ r\,t $. De que arco forma um ângulo $2$ \omterm{def:g11:trigo:radian}{radianos} corte em um
        círculo de raio $3$ eu? (Não $\pi $ em qualquer lugar --- isto é
        o ponto.)
  \item Uma roda de raio $30$ cm rolos $1$ km sem
        escorregando. Através de quantos \emph{\omterm{def:g11:trigo:radian}{radianos}} virou,
        e quantas voltas completas são essas?
  \item Coloque no círculo unitário e avalie exatamente:
        $\cos\frac{2\pi}{3}$, $\sin\left(-\frac{\pi}{4}\right)$,
        e $\cos\frac{19\pi}{6}$ (reduzir módulo $2\pi $ primeiro;
        \cref{prop:g11:trigo:associated}).
  \item Um ângulo $ t \in \intoo{\frac\pi2}{\pi}$ satisfaz
        $\sin t = \frac35$. Usando
        $\cos^2 t + \sin^2 t = 1$ e o quadrante, encontre
        $\cos t $ e $\tan t $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The Ferris wheel.}
Uma roda gigante tem raio $60$ m, seu hub $65$ m acima do
chão, e gira uma vez a cada $30$ minutos. Você embarca no
ponto mais baixo no tempo $ t = 0$ (em minutos), então sua altura obedece
\[
h(t) = 65 - 60 \cos\!\left(\frac{\pi t}{15}\right).
\]
\begin{enumerate}[resume]
  \item Justifique a fórmula: verifique o ângulo virado após $ t $
        minutos, a altura em $ t = 0$, e explique o menos
        sinal.
  \item Calcule sua altura em $ t = 7.5$, $ t = 15$ e
        $ t = 20$ minutos (valores exatos).
  \item A que horas você está \emph{primeiro} no $95$ eu? (Resolver
        $ h(t) = 95$ com \cref{met:g11:trigo:solve}.)
  \item Durante qual tempo \omterm{def:g10:numbers:interval}{intervalo} do passeio você está pelo menos
        $95$ m de altura --- e por quanto tempo no total? (Comparar
        \cref{exo:g11:trigo:11}.)
  \item Onde a altura muda \emph{mais rápido}? Compare o
        subir durante o primeiro minuto
        ($ h(1) - h(0)$) com a subida entre minutos $7$ e
        $8$, e formular a observação (a ferramenta que faz
        é exato --- o derivado da sinusóide --- chega
        na 12ª série).
  \item A vez do engenheiro: projete a fórmula da altura de uma roda
        com altura máxima $40$ m, altura mínima $4$ eu e
        período $12$ minutos, embarque no fundo às
        $ t = 0$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The tide and the circle's
symmetries.}
Num porto, a profundidade da água em metros, $ t $ horas depois
meia-noite, é
\[
d(t) = 7 + 3 \sin\!\left(\frac{\pi t}{6}\right).
\]
\begin{enumerate}[resume]
  \item Dê as profundidades máxima e mínima, os tempos em que
        eles ocorrem primeiro e o período da maré.
  \item Um navio cargueiro precisa $8.5$ m de água. Resolver
        $ d(t) \geq 8.5$ sobre $\intco{0}{12}$: durante qual janela
        ele pode entrar e qual é o comprimento da janela? Quando é que
        próxima janela aberta?
  \item Derive novamente duas identidades de \omterm{prop:g11:trigo:associated}{ângulos associados} diretamente de
        simetrias de círculo --- $\sin(\pi - t) = \sin t $ (espelho
        ao longo do eixo vertical) e $\cos(\pi + t) = -\cos t $
        (meia volta) --- e indique a identidade complementar
        $\cos\left(\frac\pi2 - t\right) = \sin t $.
  \item Paridade (\cref{pb:g11:func:1} vocabulário): classificar
        $\sin $, $\cos $, e $ t \mapsto \sin t + t^3$ como par ou
        estranho, com provas de uma linha do círculo.
  \item Resolver $\intco{0}{2\pi}$:
        $2\sin^2 t - \sin t - 1 = 0$. (Uma quadrática em
        disfarce: fatore como em \cref{pb:g11:quad:1}, então
        leia o círculo.)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The mathematician's angle.}
\begin{enumerate}[resume]
  \item Quantos graus é $1$ \omterm{def:g11:trigo:radian}{radiano}? Em seguida, uma armadilha da calculadora:
        que é maior, $\sin(1)$ (\omterm{def:g11:trigo:radian}{radiano} modo) ou
        $\sin(1^\circ)$ --- e aproximadamente por qual fator? Moral
        para configurações da calculadora?
  \item Resolver $\cos t = \sin t $ sobre $\intco{0}{2\pi}$, usando o
        círculo (onde estão \omterm{def:g10:coordgeom:system}{abscissa} e \omterm{def:g10:coordgeom:system}{ordenar} igual?).
  \item Para ângulos pequenos, o arco e sua \omterm{def:g11:trigo:cossin}{seno} quase coincidem:
        comparar $\sin(0.1)$ com $0.1$ (cinco casas decimais) e
        forneça o erro relativo. Este ``pequeno ângulo
        \omterm{def:g10:numbers:approx}{aproximação}'' dirige navios e pêndulos; o teorema
        atrás dele ($\frac{\sin t}{t} \to 1$) é uma estrela do
        capítulo de limites da 12ª série.
  \item Finale --- o círculo unitário como relógio mestre, uma frase
        por item: \omterm{def:g11:trigo:radian}{radianos} transformar ângulos em números honestos (arco
        $=$ raio $\times $ ângulo); \omterm{def:g11:trigo:cossin}{cosseno} e \omterm{def:g11:trigo:cossin}{seno} são os
        sombra e altura do ponteiro do relógio; as simetrias do círculo
        são as identidades; resolvendo $\cos t = c$ lê vezes
        fora do horário; e todo fenômeno periódico --- roda,
        maré, som --- este relógio está fantasiado.
\end{enumerate}
\end{problem}
