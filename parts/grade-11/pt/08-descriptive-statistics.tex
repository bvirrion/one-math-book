\chapter{Estatística Descritiva}\label{ch:g11:stat}

As estatísticas compactam uma lista de números em alguns resumos \omterm{def:g11:stat:mean}{significativos}:
onde estão os dados (significar, \omterm{def:g11:stat:median}{mediana}) e quão espalhado ele é (quartis,
\omterm{def:g11:stat:variance}{variação}, \omterm{def:g11:stat:variance}{desvio padrão}). Este capítulo resolve tudo de uma só vez
exemplo de execução. Os mesmos resumos reaparecem para variáveis aleatórias em
\cref{ch:g11:prob}, e suas leis probabilísticas em \cref{ch:g12:sums}.

Ao longo do capítulo, nosso conjunto de dados em execução são as marcas (de $20$) de
uma aula de $12$ alunos:
\[
8,\ 9,\ 10,\ 10,\ 11,\ 12,\ 12,\ 12,\ 13,\ 14,\ 16,\ 17 .
\]

\section{Média e mediana}

\begin{definition}[Significar]\label{def:g11:stat:mean}
O \emph{significativo}\index{significativo} dos valores $ x_1, x_2, \dots, x_n $ é
\[
\bar x = \frac{x_1 + x_2 + \dots + x_n}{n}.
\]
Se o valor $ x_i $ aparece com freqüência $ n_i $ (com
$ n_1 + \dots + n_k = n $), então
$\bar x = \frac{n_1 x_1 + \dots + n_k x_k}{n}$.
\end{definition}

\begin{definition}[Mediana]\label{def:g11:stat:median}
UM \emph{mediana}\index{mediana} de um conjunto de dados classificado é um valor que divide
pela metade: pelo menos metade dos valores são $\leq $ isso, e pelo menos metade são
$\geq $ isto. Na prática, classifique o $ n $ valores; se $ n $ é ímpar, a mediana é
o valor médio; se $ n $ é par, pegue o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} dos dois médios
valores.
\end{definition}

\begin{example}\label{ex:g11:stat:meanmedian}
Para o conjunto de dados em execução, a soma é $144$, então $\bar x = \frac{144}{12} =
12$. Os dois valores intermediários ($6$ o e $7$ da lista ordenada) são ambos
$12$, então o \omterm{def:g11:stat:median}{mediana} é $12$ também. Significar e \omterm{def:g11:stat:median}{mediana} não precisa coincidir: o
\omterm{def:g11:stat:median}{mediana} ignora \emph{quão longe} os valores extremos são, o significar não.
Substituindo a marca superior $17$ por $20$ move o significar para $12.25$ mas vai embora
o \omterm{def:g11:stat:median}{mediana} no $12$.
\end{example}

\section{Quartis}

\begin{definition}[Quartis, intervalo interquartil]\label{def:g11:stat:quartiles}
O \emph{primeiro quartil}\index{primeiro quartil} $ Q_1$ é o menor valor tal
que pelo menos um quarto dos dados seja $\leq Q_1$; o \emph{terceiro
quartil} $ Q_3$ é o menor valor tal que pelo menos três quartos da
os dados são $\leq Q_3$. O \emph{intervalo interquartil}\index{intervalo interquartil} é $ Q_3 - Q_1$: o
largura da metade central dos dados.
\end{definition}

\begin{example}
$ n = 12$: um quarto é $3$, então $ Q_1$ é o $3$ Terceiro valor classificado:
$ Q_1 = 10$. Três quartos é $9$, então $ Q_3$ é o $9$ o valor:
$ Q_3 = 13$. O \omterm{def:g11:stat:quartiles}{intervalo interquartil} é $3$: a metade central da classe
senta-se dentro $3$ marcas.
\end{example}

\section{Variância e desvio padrão}

\begin{definition}[Variância, desvio padrão]\label{def:g11:stat:variance}
O \emph{variação}\index{variação} do conjunto de dados é o significar do
desvios quadrados do significar:
\[
V = \frac{1}{n}\sum_{i=1}^{n} (x_i - \bar x)^2,
\]
e o \emph{desvio padrão}\index{desvio padrão} é
$\sigma = \sqrt{V}$, expresso na mesma unidade dos dados.
\end{definition}

\begin{proposition}[Fórmula de atalho]\label{prop:g11:stat:shortcut}
O \omterm{def:g11:stat:variance}{variação} é o significar dos quadrados menos o quadrado do significar:
\[
V = \frac{1}{n}\sum_{i=1}^{n} x_i^2 \;-\; \bar x^{\,2}.
\]
\end{proposition}

\begin{proof}
Expanda cada desvio quadrático:
$(x_i - \bar x)^2 = x_i^2 - 2\bar x\, x_i + \bar x^2$. Resumindo e
dividindo por $ n $:
\[
V = \frac1n \sum x_i^2 - 2\bar x \cdot \underbrace{\frac1n \sum
x_i}_{=\ \bar x} + \frac1n \cdot n\bar x^2
= \frac1n\sum x_i^2 - 2\bar x^2 + \bar x^2
= \frac1n\sum x_i^2 - \bar x^2 .
\]
A mesma identidade, para variáveis aleatórias, é
\cref{prop:g11:prob:konig}.
\end{proof}

\begin{example}\label{ex:g11:stat:variance}
Para o conjunto de dados em execução, os desvios de $\bar x = 12$ são $-4$,
$-3$, $-2$, $-2$, $-1$, $0$, $0$, $0$, $1$, $2$, $4$, $5$; seus quadrados
soma para
\[
16 + 9 + 4 + 4 + 1 + 0 + 0 + 0 + 1 + 4 + 16 + 25 = 80 .
\]
Por isso
\[
V = \frac{80}{12} = \frac{20}{3} \approx 6.67,
\qquad
\sigma = \sqrt{20/3} \approx 2.6 \text{ marks}.
\]
Grosso modo, uma marca típica fica $2.6$ marcas longe do
significar.
\end{example}

\begin{proposition}[Mudança de unidades]\label{prop:g11:stat:affine}
Se todo valor for transformado por $ y_i = a x_i + b $, então
\[
\bar y = a \bar x + b,
\qquad
V_y = a^2\, V_x,
\qquad
\sigma_y = \abs{a}\,\sigma_x .
\]
\end{proposition}

\begin{proof}
$\bar y = \frac1n \sum (a x_i + b) = a\left(\frac1n\sum x_i\right) +
\frac1n \cdot nb = a\bar x + b $. Então
$ y_i - \bar y = a(x_i - \bar x)$: cada desvio é dimensionado por $ a $
(a mudança $ b $ cancela), então cada desvio quadrático é escalonado por $ a^2$,
e assim é o deles significar. Tirar raízes quadradas dá
$\sigma_y = \abs a\,\sigma_x $.
\end{proof}

\begin{example}
Temperaturas registradas em graus Celsius com significar $20$ e padrão
desvio $3$ converter para Fahrenheit por $ F = 1.8\,C + 32$: o significar
torna-se $1.8 \times 20 + 32 = 68$, o \omterm{def:g11:stat:variance}{desvio padrão}
$1.8 \times 3 = 5.4$. A mudança de um conjunto de dados não altera sua distribuição;
redimensionar redimensiona o spread.
\end{example}

\begin{method}[Resumindo e comparando conjuntos de dados]\label{met:g11:stat:compare}
Para comparar dois conjuntos de dados, coloque lado a lado: significar (ou \omterm{def:g11:stat:median}{mediana}) para
\emph{posição}, \omterm{def:g11:stat:variance}{desvio padrão} (ou \omterm{def:g11:stat:quartiles}{intervalo interquartil}) para
\emph{espalhar}. O par (\omterm{def:g11:stat:median}{mediana}, \omterm{def:g11:stat:quartiles}{intervalo interquartil}) é robusto para
valores extremos; o par (significar, \omterm{def:g11:stat:variance}{desvio padrão}) usa todos os valores e
alimenta \omterm{def:g10:proba:distribution}{probabilidade} teoria.
\end{method}

\begin{example}
Dois arqueiros disparam cada $10$ flechas; média de pontuação $8.5$ para ambos, mas
o desvios padrão são $0.5$ e $2.1$. Mesmo nível em média --- mas
o primeiro arqueiro é muito mais consistente.
\end{example}

\section{Problema: O escore z, uma régua universal}

\begin{problem}[{Problema de fim de semana --- comparação de desvios padrão
maçãs com laranjas, a desigualdade de Chebyshev vira $\sigma $ em um
garantia, e a média ganha sua própria medalha}]
\label{pb:g11:stat:1}
É $15/20$ em um exame difícil melhor do que $13/20$ em um fácil? É
um $23$ em uma lista de pequenos números, um acidente ou uma exceção? Cru
os números não podem dizer --- mas divididos pela direita padrão
desvio todos falam uma língua, a \emph{\omterm{pb:g11:stat:1}{pontuação z}}. Isto
problema domina o \omterm{def:g11:stat:variance}{variação} maquinaria
(\cref{def:g11:stat:variance},
\cref{prop:g11:stat:shortcut}), constrói o governante universal, e
prova a notável desigualdade que torna $\sigma $ de um
número descritivo em uma garantia.

\medskip
\noindent\textbf{Part I --- Variância mechanics.}
\begin{enumerate}
  \item Para o conjunto de dados $4, 8, 6, 5, 2$: calcular o significar, então
        o \omterm{def:g11:stat:variance}{variação} direto da definição, então
        $\sigma $.
  \item Recalcular o \omterm{def:g11:stat:variance}{variação} com a fórmula de atalho
        (\cref{prop:g11:stat:shortcut}) --- significar de quadrados
        menos quadrado de significar --- e verifique o acordo.
  \item Com \omterm{def:g10:stats:series}{frequências}: marcas $8$, $10$, $14$ obtido por
        $3$, $5$, $2$ estudantes. Calcular significar, \omterm{def:g11:stat:variance}{variação} e
        $\sigma $ (duas casas decimais).
  \item As temperaturas têm significar $20$ e $\sigma = 3$ (graus
        Celsius). Dê o significar e $\sigma $ em Fahrenheit
        ($ F = 1.8\,C + 32$; \cref{prop:g11:stat:affine}).
  \item Prove as duas regras que acabamos de usar: adicionar uma constante $ b $ para
        cada valor muda o significar por $ b $ e folhas $\sigma $
        inalterado; multiplicando por $ a $ multiplica o significar por $ a $
        e $\sigma $ por $\abs a $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The universal ruler.}
O \emph{pontuação z}\index{pontuação z} de um valor $ x $ em um conjunto de dados de
significar $\bar x $ e \omterm{def:g11:stat:variance}{desvio padrão} $\sigma $ é
$ z = \dfrac{x - \bar x}{\sigma}$: o número de padrão
desvios separando $ x $ do significar.
\begin{enumerate}[resume]
  \item O exame A tem significar $12$ e $\sigma = 2$; exame B tem significar
        $8$ e $\sigma = 4$. Alice marcou $15$ em A, Bob $13$
        em B. Calcule ambos pontuações z: cujo desempenho se destaca
        mais acima da multidão?
  \item Calcule os cinco pontuações z do conjunto de dados da pergunta 1,
        e verifique se eles têm significar $0$ e padrão
        desvio $1$. Explique a partir da pergunta 5 por que
        padronização \emph{sempre} produz significar $0$ e
        $\sigma = 1$.
  \item Um teste de aptidão tem significar $100$ e $\sigma = 15$. O que
        pontuação bruta tem $ z = 2$? Qual é o \omterm{pb:g11:stat:1}{pontuação z} de $76$?
  \item Um pediatra registra o peso de um bebê como
        ``$ z = -2.5$''. Diga o que isso significa e por que o crescimento é
        rastreado em pontuações z em vez de quilogramas como a criança
        idades.
  \item O estranho de \cref{exo:g11:stat:10} (conjunto de dados
        $1, 2, 2, 3, 3, 3, 4, 4, 5, 23$): calcule o significar,
        $\sigma $, e o \omterm{pb:g11:stat:1}{pontuação z} de $23$. O alarme clássico
        limite é $\abs z \geq 3$: veredicto?
  \item Recalcular $\sigma $ \emph{sem} o valor $23$ e
        o \omterm{pb:g11:stat:1}{pontuação z} que $23$ teria contra o limpo
        dados significar e $\sigma $. O que o outlier fez
        o próprio governante pretendia medi-lo, e o que significa
        \cref{exo:g11:stat:10} sugerir como contramedida?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Chebyshev's guarantee.}
\begin{enumerate}[resume]
  \item Prove o Bienaymé-Desigualdade de Chebyshev para um
        conjunto de dados: se uma proporção $ p $ dos valores satisfazem
        $\abs{x - \bar x} \geq k\sigma $, então, mantendo apenas
        esses termos no \omterm{def:g11:stat:variance}{variação} soma,
        \[
        \sigma^2 \geq p \cdot (k\sigma)^2,
        \qquad\text{hence}\qquad
        p \leq \frac{1}{k^2} .
        \]
  \item O que a desigualdade garante sobre a proporção
        de qualquer conjunto de dados além $2\sigma $ do seu significar? Além
        $3\sigma $? Verifique o $2\sigma $ declaração sobre o conjunto de dados
        da questão 1.
  \item Uma fábrica produz parafusos com significar $50$ milímetros e
        $\sigma = 0.1$ mm. O que Chebyshev garante sobre
        a proporção de parafusos fora
        $\intcc{49.7}{50.3}$? (A produção real é muito melhor
        --- a curva do sino da 12ª série torna isso cauteloso
        $\frac19$ em uma fração de um por cento.)
  \item Dois fundos retornam $5\,\%$ em média por ano, com
        $\sigma = 2\,\%$ para o fundo A e $\sigma = 12\,\%$ para
        fundo B. Para cada fundo, o que Chebyshev diz sobre
        a proporção probabilística de anos com retornos
        abaixo $-19\,\%$? O que é $\sigma $ nome financeiro?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The significar's medal, and a warning.}
\begin{enumerate}[resume]
  \item Deixar $ f(c)$ seja o significar dos desvios quadrados
        $(x_i - c)^2$ de um centro arbitrário $ c $. Prove o
        identidade
        \[
        f(c) = \sigma^2 + (\bar x - c)^2 ,
        \]
        e concluo: o significar é o único minimizador de significar
        desvio quadrado --- sua medalha, correspondendo ao medianade
        minimização de desvios absolutos em
        \cref{pb:g10:stats:1}.
  \item Verifique a identidade no conjunto de dados da pergunta 1 com
        $ c = 6$: calcular $ f(6)$ diretamente e por meio da fórmula.
  \item Mesmos resumos, dados diferentes: verifique isso
        $ P = \{3, 5, 7, 9, 11\}$ e $ Q = \{1, 7, 7, 7, 13\}$
        compartilhe o mesmo significar mas não é o mesmo \omterm{def:g11:stat:variance}{variação}; então
        construir um conjunto de dados do formulário
        $\{7 - a, 7, 7, 7, 7 + a\}$ exatamente com o significar e
        \omterm{def:g11:stat:variance}{variação} de $ P $ --- ainda com uma forma própria.
  \item Prove isso $\sigma = 0$ força todos os valores iguais. (O que
        deve ser todo termo de uma soma zero de quadrados?)
  \item Finale --- a escada do estatístico, uma linha por
        degrau: centros (significar, \omterm{def:g11:stat:median}{mediana}), spreads ($\sigma $, AIQ),
        o governante universal ($ z$), a garantia (Chebyshev),
        e a advertência da questão 18: resumos compactados,
        e a compressão perde --- com a curva em forma de sino do grau
        12 esperando para virar pontuações z em \omterm{def:g10:proba:distribution}{probabilidades} exatas.
\end{enumerate}
\end{problem}
