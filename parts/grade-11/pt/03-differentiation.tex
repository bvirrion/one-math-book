\chapter{Diferenciação: Um Primeiro Curso}\label{ch:g11:deriv}

Quão rápido um \omterm{def:g11:func:function}{função} mudar? A resposta em um único ponto é a
\emph{derivado}, o número mais útil em todos os aplicados
matemática. Este capítulo o constrói a partir de taxas médias de mudança,
calcula para o \omterm{def:g11:func:reference}{funções de referência}, e usa seu sinal para ler o
variações de um \omterm{def:g11:func:function}{função}. Tudo aqui é ampliado e aprofundado em
\cref{ch:g12:deriv}.

\section{A taxa média de mudança}

\begin{definition}[Taxa média de mudança]\label{def:g11:deriv:rate}
Deixar $f $ ser definido em um \omterm{def:g10:numbers:interval}{intervalo} $ I $, e $ a \neq b $ em $ I $. O
\emph{taxa média de mudança}\index{taxa média de mudança} de $ f $ entre
$ a $ e $ b $ é
\[
\frac{f(b) - f(a)}{b - a}.
\]
É o declive do \emph{secar}\index{secar} linha através dos pontos
$\bigl(a, f(a)\bigr)$ e $\bigl(b, f(b)\bigr)$ da curva.
\end{definition}

\begin{example}
Um carro viaja $140$ km entre 14h e 16h: sua velocidade média é
$\frac{140}{2} = 70$ km/h. O velocímetro, no entanto, exibe um
\emph{instantâneo} velocidade a cada momento --- essa é exatamente a noção
que o derivado torna preciso.
\end{example}

\section{A derivada em um ponto}

\begin{definition}[Derivada em um ponto]\label{def:g11:deriv:derivative}
Deixar $ f $ ser definido em um \omterm{def:g10:numbers:interval}{intervalo} contendo $ a $. Para $ h \neq 0$, formar o
\omterm{def:g11:deriv:rate}{taxa média de mudança} entre $ a $ e $ a + h $:
\[
r(h) = \frac{f(a+h) - f(a)}{h}.
\]
Se $ r(h)$ se aproxima de um único número fixo como $ h $ fica arbitrariamente perto de
$0$, dizemos que $ f $ é \emph{diferenciável em $ a $}\index{diferenciável em $ a $}\index{diferenciável};
esse número é o \emph{derivado de $ f $ no $ a $}, escrito $ f'(a)$:
\[
f'(a) = \lim_{h \to 0} \frac{f(a+h) - f(a)}{h}.
\]
\end{definition}

\begin{example}\label{ex:g11:deriv:atpoint}
Deixar $ f(x) = x^2$ e $ a = 1$. Para $ h \neq 0$:
\[
\frac{f(1+h) - f(1)}{h}
= \frac{(1+h)^2 - 1}{h}
= \frac{1 + 2h + h^2 - 1}{h}
= \frac{h(2 + h)}{h}
= 2 + h .
\]
Como $ h $ abordagens $0$, isso se aproxima $2$: o \omterm{def:g11:func:function}{função} $ x^2$ é
\omterm{def:g11:deriv:derivative}{diferenciável} no $1$ e $ f'(1) = 2$. Observe a estratégia: \emph{simplificar
o quociente até $ h $ não aparece mais em um denominador}, então deixe
$ h \to 0$.
\end{example}

\begin{example}[Um ponto não diferenciável]
Deixar $ f(x) = \abs{x}$ e $ a = 0$. Então
$\frac{f(h) - f(0)}{h} = \frac{\abs h}{h}$, que é igual $1$ para $ h > 0$
e $-1$ para $ h < 0$: nenhum número único é aproximado, e $\abs x $ não é
\omterm{def:g11:deriv:derivative}{diferenciável} no $0$. Sua curva tem \emph{canto} lá.
\end{example}

\section{A linha tangente}

\begin{definition}[Tangente]\label{def:g11:deriv:tangent}
Se $ f $ é \omterm{def:g11:deriv:derivative}{diferenciável} no $ a $, o \emph{tangente}\index{tangente} para o
curva no ponto $ A = \bigl(a, f(a)\bigr)$ é a linha que passa $ A $ com
declive $ f'(a)$. Isso é equação é
\[
y = f(a) + f'(a)(x - a).
\]
\end{definition}

O \omterm{def:g11:deriv:tangent}{tangente} é a posição limite do secantes através $ A $: como
$ h \to 0$, o segundo ponto $\bigl(a+h, f(a+h)\bigr)$ desliza ao longo do
curva em direção $ A $, e o secantede declive $\frac{f(a+h)-f(a)}{h}$
abordagens $ f'(a)$.

\begin{example}\label{ex:g11:deriv:tangenteq}
Para $ f(x) = x^2$ no $ a = 1$: $ f(1) = 1$ e $ f'(1) = 2$
(\cref{ex:g11:deriv:atpoint}), então o \omterm{def:g11:deriv:tangent}{tangente} é
\[
y = 1 + 2(x - 1) = 2x - 1 .
\]
Esta é a linha $ d_2$ encontrado por álgebra pura em \cref{exo:g11:quad:11}.
\end{example}

\section{Derivadas das funções de referência}

\begin{proposition}\label{prop:g11:deriv:refderivatives}
Em seu domínios de diferenciabilidade:
\[
(c)' = 0, \quad
(x)' = 1, \quad
(x^2)' = 2x, \quad
(x^3)' = 3x^2, \quad
\left(\frac1x\right)' = -\frac{1}{x^2}, \quad
(\sqrt x\,)' = \frac{1}{2\sqrt x} \ (x > 0).
\]
Mais geralmente, $(x^n)' = n\,x^{n-1}$ para cada inteiro $ n \geq 1$.
\end{proposition}

\begin{proof}
Cada prova segue a estratégia de \cref{ex:g11:deriv:atpoint}: simplificar
a taxa, então deixe $ h \to 0$. Consertar $ a $ no domínio.

\emph{Constante e identidade.} $\frac{c - c}{h} = 0$ e
$\frac{(a+h) - a}{h} = 1$ para todos $ h $: os limites são $0$ e $1$.

\emph{Quadrado.} $\frac{(a+h)^2 - a^2}{h} = \frac{2ah + h^2}{h} = 2a + h
\to 2a $.

\emph{Cubo.} $(a+h)^3 = a^3 + 3a^2h + 3ah^2 + h^3$, então a taxa é
$3a^2 + 3ah + h^2 \to 3a^2$.

\emph{Recíproca.} Para $ a \neq 0$ e $ h $ pequeno o suficiente para que
$ a + h \neq 0$:
\[
\frac{1}{h}\left(\frac{1}{a+h} - \frac{1}{a}\right)
= \frac{1}{h} \cdot \frac{a - (a+h)}{a(a+h)}
= \frac{-1}{a(a+h)}
\xrightarrow[h \to 0]{} -\frac{1}{a^2}.
\]

\emph{\omterm{def:g11:quad:discriminant}{Raiz} quadrada.} Para $ a > 0$, multiplique pelo conjugado:
\[
\frac{\sqrt{a+h} - \sqrt a}{h}
= \frac{(a + h) - a}{h\,(\sqrt{a+h} + \sqrt a)}
= \frac{1}{\sqrt{a+h} + \sqrt a}
\xrightarrow[h \to 0]{} \frac{1}{2\sqrt a}.
\]

\emph{Poder geral.} O padrão $2a $, $3a^2$ continua: expandindo
$(a+h)^n $, os termos depois $ a^n + n\,a^{n-1}h $ todos carregam $ h^2$, então o
taxa tende a $ n\,a^{n-1}$. (Uma indução completa é dada em
\cref{ch:g12:deriv}.)
\end{proof}

\section{Regras de diferenciação}

\begin{theorem}[Operações]\label{thm:g11:deriv:rules}
Deixar $ u $ e $ v $ ser \omterm{def:g11:deriv:derivative}{diferenciável} sobre $ I $ e $\lambda \in \R $. Então
$ u + v $, $\lambda u $, $ uv $ são \omterm{def:g11:deriv:derivative}{diferenciável} sobre $ I $, como é
$\frac{u}{v}$ onde quer que $ v \neq 0$, e
\[
(u+v)' = u' + v', \qquad
(\lambda u)' = \lambda u', \qquad
(uv)' = u'v + uv', \qquad
\left(\frac{u}{v}\right)' = \frac{u'v - uv'}{v^2}.
\]
\end{theorem}

\begin{proof}[Prova das três primeiras regras]
Correção $ a \in I $ e abreviar $\Delta u = u(a+h) - u(a)$,
$\Delta v = v(a+h) - v(a)$.

\emph{Soma.} A taxa de $ u + v $ é
$\frac{\Delta u + \Delta v}{h} = \frac{\Delta u}{h} + \frac{\Delta v}{h}
\to u'(a) + v'(a)$.

\emph{Múltiplo.} A taxa de $\lambda u $ é
$\lambda \frac{\Delta u}{h} \to \lambda u'(a)$.

\emph{Produto.} Adicione e subtraia o termo misto $ u(a+h)\,v(a)$:
\begin{align*}
\frac{u(a+h)v(a+h) - u(a)v(a)}{h}
&= \frac{u(a+h)v(a+h) - u(a+h)v(a) + u(a+h)v(a) - u(a)v(a)}{h}\\
&= u(a+h)\,\frac{\Delta v}{h} + v(a)\,\frac{\Delta u}{h}.
\end{align*}
Como $ h \to 0$: $\frac{\Delta v}{h} \to v'(a)$, $\frac{\Delta u}{h} \to
u'(a)$, e $ u(a+h) \to u(a)$ (sua taxa tem um limite, então
$\Delta u = h \cdot \frac{\Delta u}{h} \to 0$). A taxa tende, portanto,
para $ u(a)v'(a) + v(a)u'(a)$.

A regra do quociente é provada da mesma forma; isso é \emph{admitido em
este nível} e provou em \cref{ch:g12:deriv}.
\end{proof}

\begin{example}\label{ex:g11:deriv:computations}
\emph{Polinomial.} $ f(x) = 3x^3 - 5x^2 + 7x - 2$:
\[
f'(x) = 3 \times 3x^2 - 5 \times 2x + 7 = 9x^2 - 10x + 7 .
\]

\emph{Quociente.} $ g(x) = \frac{2x+1}{x^2+1}$: com $ u = 2x + 1$, $ v = x^2
+ 1$,
\[
g'(x) = \frac{2(x^2+1) - (2x+1)\,2x}{(x^2+1)^2}
= \frac{-2x^2 - 2x + 2}{(x^2+1)^2}.
\]

\emph{Produto.} $ k(x) = x\sqrt x $ para $ x > 0$:
$ k'(x) = 1 \times \sqrt x + x \times \frac{1}{2\sqrt x}
= \sqrt x + \frac{\sqrt x}{2} = \frac{3\sqrt x}{2}$.
\end{example}

\section{Sinal da derivada e variações}

\begin{theorem}[Derivada e variações]\label{thm:g11:deriv:variations}
Deixar $ f $ ser \omterm{def:g11:deriv:derivative}{diferenciável} em um \omterm{def:g10:numbers:interval}{intervalo} $ I $.
\begin{enumerate}
  \item Se $ f' \geq 0$ sobre $ I $, então $ f $ é \omterm{def:g11:func:monotone}{aumentando} sobre $ I $.
  \item Se $ f' \leq 0$ sobre $ I $, então $ f $ é diminuindo sobre $ I $.
  \item Se $ f' > 0$ sobre $ I $ (exceto possivelmente em um número finito de pontos onde
        desaparece), então $ f $ é estritamente \omterm{def:g11:func:monotone}{aumentando} sobre $ I $.
\end{enumerate}
\end{theorem}

\begin{proof}
A intuição é clara --- uma curva cuja \omterm{def:g11:deriv:tangent}{tangentes} todos apontam para cima devem
subir --- mas uma prova rigorosa precisa do significar teorema do valor. O resultado é
\emph{admitido neste nível}; ver \cref{thm:g12:deriv:variations}.
\end{proof}

\begin{proposition}[Extrema]\label{prop:g11:deriv:extrema}
Se $ f'$ muda o sinal em $ a $ de $+$ para $-$, então $ f $ tem um local \omterm{def:g10:functions:extrema}{máximo}
no $ a $; de $-$ para $+$, um local \omterm{def:g10:functions:extrema}{mínimo}. Em tal ponto, $ f'(a) = 0$ e
o \omterm{def:g11:deriv:tangent}{tangente} é horizontal.
\end{proposition}

\begin{proof}
Se $ f' \geq 0$ em um \omterm{def:g10:numbers:interval}{intervalo} terminando em $ a $ e $ f' \leq 0$ em um \omterm{def:g10:numbers:interval}{intervalo}
começando em $ a $, então $ f $ aumenta até $ f(a)$ e diminui depois
(\cref{thm:g11:deriv:variations}): $ f(a)$ é o maior próximo. Desde $ f'$
muda o sinal em $ a $ e é definido lá, $ f'(a) = 0$.
\end{proof}

\begin{method}[Estudando uma função]\label{met:g11:deriv:study}
\begin{enumerate}
  \item Calcular $ f'(x)$ e simplificá-lo --- idealmente em uma forma fatorada.
  \item Determine o sinal de $ f'(x)$ em cada \omterm{def:g10:numbers:interval}{intervalo} (para um quadrático
        $ f'$, usar \cref{thm:g11:quad:sign}).
  \item Registre os resultados em um \emph{\omterm{def:g10:functions:table}{tabela de variação}}: uma linha para o
        sinal de $ f'$, uma linha de setas para $ f $, com os valores de $ f $ em
        os pontos de viragem.
  \item Conclua: extremos, número de soluções de $ f(x) = k $, desigualdades.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g11:deriv:study}
Estudar $ f(x) = x^3 - 3x + 1$ sobre $\R $. Primeiro,
$ f'(x) = 3x^2 - 3 = 3(x-1)(x+1)$: uma quadrática com raízes $-1$ e $1$,
positivo fora deles. Então $ f $ é \omterm{def:g11:func:monotone}{aumentando} sobre $\intoc{-\infty}{-1}$,
diminuindo sobre $\intcc{-1}{1}$, \omterm{def:g11:func:monotone}{aumentando} sobre $\intco{1}{+\infty}$, com um
locais \omterm{def:g10:functions:extrema}{máximo} $ f(-1) = -1 + 3 + 1 = 3$ e um local \omterm{def:g10:functions:extrema}{mínimo}
$ f(1) = 1 - 3 + 1 = -1$.
\end{example}

\begin{example}[Otimização]\label{ex:g11:deriv:box}
Uma caixa aberta é feita de $12 \times 12$ folha de cm, cortando um quadrado de
lado $ x $ de cada canto e dobrando. Seu volume é
$ V(x) = x(12 - 2x)^2$ para $0 < x < 6$. \omterm{def:g10:algebra:expand}{Expandindo},
$ V(x) = 4x^3 - 48x^2 + 144x $, então
\[
V'(x) = 12x^2 - 96x + 144 = 12(x^2 - 8x + 12) = 12(x - 2)(x - 6).
\]
Sobre $\intoo{0}{6}$, o fator $ x - 6$ é negativo, então $ V'(x)$ tem o sinal
de $-(x-2)$: positivo antes $2$, negativo depois. O volume é \omterm{def:g10:functions:extrema}{máximo}
para $ x = 2$ cm, dando $ V(2) = 2 \times 8^2 = 128$ cm $^3$.
\end{example}

\section{Problema: os três empregos da tangente}

\begin{problem}[{Problema de fim de semana --- o espelho parabólico provou,
A máquina de encontrar raízes de Newton e a viga mais forte da tora}]
\label{pb:g11:deriv:1}
UM derivado empates \omterm{def:g11:deriv:tangent}{tangentes}; isso parece um talento modesto.
Este problema mostra a \omterm{def:g11:deriv:tangent}{tangente} mantendo três empregos ao mesmo tempo:
prova o teorema do farol deixado pendente em
\cref{pb:g11:quad:1} (cada raio paralelo a um parábolaeixo do
reflete através do \omterm{pb:g11:quad:1}{foco}), ele alimenta o mais rápido
algoritmo de resolução de equações de uso comum --- dentro de cada
calculadora e rede neural --- e encontra o mais forte
viga retangular que uma tora redonda pode produzir.

\medskip
\noindent\textbf{Part I --- \omterm{def:g11:deriv:tangent}{Tangente} fluency.}
\begin{enumerate}
  \item Diferenciar $ f(x) = 3x^2 - 5x + 1$ e
        $ g(x) = x^3 - 12x $
        (\cref{thm:g11:deriv:rules}); e recuperar $\left(x^2
        \right)'$ no $ x = 1$ da definição (taxa de
        mudar, então $ h \to 0$).
  \item Encontre o \omterm{def:g11:deriv:tangent}{tangente} para $ y = x^3 - 12x $ no $ a = 2$. O que é
        especial sobre isso, e o que isso sinaliza
        (\cref{prop:g11:deriv:extrema})?
  \item Estabeleça o geral \omterm{def:g11:deriv:tangent}{tangente} para o \omterm{def:g11:quad:parabola}{parábola} $ y = x^2$
        no ponto $(a, a^2)$:
        \[
        y = 2ax - a^2 .
        \]
  \item Onde é que isso \omterm{def:g11:deriv:tangent}{tangente} atravessar o $ y $-eixo? Deduza o
        receita do desenhista: desenhar o \omterm{def:g11:deriv:tangent}{tangente} em um ponto $ P $
        de altura $ h $ no \omterm{def:g11:quad:parabola}{parábola}, juntar $ P $ ao ponto de
        o eixo em profundidade\,\dots\ abaixo da origem. Declare isso.
  \item Construa o completo \omterm{def:g10:functions:table}{tabela de variação} de
        $ g(x) = x^3 - 12x $ (sinal de $ g'$, extremos locais e
        seus valores, \cref{met:g11:deriv:study}).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The mirror theorem.}
Deixar $ P(a, a^2)$ ser um ponto do \omterm{def:g11:quad:parabola}{parábola} $ y = x^2$ (pegar
$ a \neq 0$), $ F\left(0, \frac14\right)$ o \omterm{pb:g11:quad:1}{foco}, e $ T $ o
ponto onde o \omterm{def:g11:deriv:tangent}{tangente} no $ P $ atravessa o $ y $-eixo (pergunta
4). Lembrar de \cref{pb:g11:quad:1} que
$ PF = a^2 + \frac14$.
\begin{enumerate}[resume]
  \item Calcule a distância $ FT $.
  \item Conclua que o triângulo $ FPT $ é isósceles em $ F $.
        Quais são os dois ângulos que são, portanto, iguais?
  \item O raio que chega através $ P $, paralelo ao eixo, é
        paralelo à linha $(TF)$ (ambos verticais). Usando
        ângulos alternados, mostre que \omterm{def:g11:deriv:tangent}{tangente} no $ P $ faz
        \emph{ângulos iguais} com o raio vertical de entrada e
        com o segmento $[PF]$.
  \item Física: a luz reflete em uma curva como em seu \omterm{def:g11:deriv:tangent}{tangente},
        saindo no mesmo ângulo em que chegou (a lei da
        reflexão). Conclua o \emph{teorema do espelho}: cada
        raio paralelo ao eixo reflete através do \omterm{pb:g11:quad:1}{foco} ---
        e, invertendo o tempo, uma lâmpada no \omterm{pb:g11:quad:1}{foco} irradia
        luz paralela. O prato e o farol de
        \cref{pb:g11:quad:1} agora são teoremas.
  \item Verifique o triângulo isósceles numericamente em $ a = 1$:
        calcular $ T $, $ FP $ e $ FT $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Newton's machine.}
Para resolver $ f(x) = 0$, Newton propôs: comece com um palpite
$ x_0$, siga o \omterm{def:g11:deriv:tangent}{tangente} no $ x_0$ até o eixo e deixe
é $ x $-intercept seja o próximo palpite.
\begin{enumerate}[resume]
  \item Derive a fórmula do método:
        \[
        x_{n + 1} = x_n - \frac{f(x_n)}{f'(x_n)} .
        \]
  \item Execute-o $ f(x) = x^2 - 2$ de $ x_0 = 1$: calcular
        $ x_1$, $ x_2$, $ x_3$ como frações exatas. Qual
        sequência de dois mil anos --- e qual máquina
        de \cref{pb:g10:functions:1} --- você acabou de
        reconstruído? Prove a identidade por trás da coincidência:
        \[
        x - \frac{x^2 - 2}{2x} = \frac12\left(x +
        \frac2x\right).
        \]
  \item Resolver $ x^3 = 100$ (o cruzamento de poupança de
        \cref{pb:g10:reffunc:1}): aplique duas etapas de Newton para
        $ f(x) = x^3 - 100$ de $ x_0 = 5$ (quatro casas decimais) e
        compare com a calculadora $\sqrt[3]{100}$.
  \item Sabotagem: tente iniciar o método em
        $ f(x) = x^3 - 12x $ no $ x_0 = 2$. O que dá errado e
        por que (pergunta 2)? Faça a advertência prática.
  \item Na questão 12, os decimais corretos foram aproximadamente
        $1 \to 3 \to 6$. Compare com bissecção (dividir um
        \omterm{def:g10:numbers:interval}{intervalo} cada etapa) e diga em uma frase por que
        calculadoras, receptores GPS e treinamento em redes neurais
        todos funcionam com base na ideia de Newton.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The strongest beam.}
Uma viga retangular de largura $ w $ e profundidade $ d $ é serrado de um
tora redonda de diâmetro $30$ centímetros, então $ w^2 + d^2 = 900$. Uma viga
rigidez é proporcional a $ w d^2$ (a profundidade conta duas vezes:
as vigas ficam no limite!).
\begin{enumerate}[resume]
  \item Expresse a rigidez como \omterm{def:g11:func:function}{função} de $ w $ sozinho:
        $ S(w) = 900w - w^3$, em que \omterm{def:g10:numbers:interval}{intervalo}?
  \item Diferenciar, construir o \omterm{def:g10:functions:table}{tabela de variação}, e encontre o
        ideal $ w $ e $ d $ (valores exatos, então para o
        milímetro).
  \item Mostre que as proporções ótimas satisfazem
        $ d = w\sqrt2$ --- a proporção dos antigos carpinteiros. (Bônus
        tradição: dividir o diâmetro da tora em três partes iguais
        e elevando perpendiculares às construções do círculo
        exatamente este retângulo.)
  \item A escolha ingénua é a \emph{quadrado} feixe
        ($ w = d$). Calcule sua rigidez e o valor ideal
        da viga e forneça o ganho do carpinteiro em porcentagem.
  \item Final --- o tangentesão três trabalhos, uma frase cada:
        geometria (o teorema do espelho), numérica (teorema de Newton
        máquina), otimização (o feixe), tudo alimentado pelo
        mesmo objeto. E o ponteiro para frente: a 12ª série acrescenta
        convexidade --- o tangentelado da curva --- para
        afie todos os três.
\end{enumerate}
\end{problem}
