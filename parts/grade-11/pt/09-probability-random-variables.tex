\chapter{Probabilidade e variáveis aleatórias}\label{ch:g11:prob}

\omterm{def:g10:proba:distribution}{Probabilidade} modela experimentos cujo resultado é incerto: um lançamento de dados, um
carta desenhada, um jogo jogado. Este capítulo estabelece o vocabulário sobre finitos
conjuntos e, em seguida, apresenta o personagem central de todos \omterm{def:g10:proba:distribution}{probabilidade} teoria:
o \emph{variável aleatória} e seu \emph{\omterm{def:g11:prob:expectation}{expectativa}}. Essas noções são
revisado e aprofundado em \cref{ch:g12:randvar}e condicional
as \omterm{def:g10:proba:distribution}{probabilidades} seguem \cref{ch:g12:condprob}.

\section{Probabilidade em um conjunto finito}

\begin{definition}[Modelo de probabilidade]\label{def:g11:prob:model}
Um experimento com um número finito de resultados é modelado por seu \emph{amostra
espaço}\index{amostra
espaço} $\Omega = \{\omega_1, \dots, \omega_n\}$ (o
conjunto de resultados) juntamente com um \emph{\omterm{def:g10:proba:distribution}{probabilidade}} $\P $ atribuindo a cada
resultado $\omega_i $ um número $ p_i \geq 0$, com $ p_1 + \dots + p_n = 1$. Um
\emph{evento}\index{evento} é um subconjunto $ A \subseteq \Omega $, e
\[
\P(A) = \sum_{\omega_i \in A} p_i .
\]
Quando todos os resultados são igualmente prováveis ($ p_i = \frac1n $: o \emph{uniforme
modelo}\index{uniforme
modelo}),
\[
\P(A) = \frac{\text{number of outcomes in } A}
{\text{number of outcomes in } \Omega}.
\]
\end{definition}

\begin{example}\label{ex:g11:prob:twodice}
Jogue dois dados distinguíveis: $\Omega $ é o conjunto do $36$ pares ordenados
$(i, j)$ com $1 \leq i, j \leq 6$, todos igualmente prováveis. O \omterm{def:g11:prob:model}{evento}
$ S $ = ``a soma é $7$'' contém o $6$ pares
$(1,6)$, $(2,5)$, $(3,4)$, $(4,3)$, $(5,2)$, $(6,1)$, então
$\P(S) = \frac{6}{36} = \frac16$. O \omterm{def:g11:prob:model}{evento} ``a soma é $12$''contém
apenas $(6,6)$: \omterm{def:g10:proba:distribution}{probabilidade} $\frac{1}{36}$. Escolhendo o \emph{certo}
\omterm{def:g10:proba:events}{espaço amostral} --- pares ordenados, não desordenados --- é o que torna o
modelo uniforme aplicável.
\end{example}

\begin{proposition}[Regras de probabilidade]\label{prop:g11:prob:rules}
Para \omterm{def:g11:prob:model}{eventos} $ A, B \subseteq \Omega $:
\[
\P(\emptyset) = 0, \qquad \P(\Omega) = 1, \qquad
\P(\bar A) = 1 - \P(A),
\]
\[
\P(A \cup B) = \P(A) + \P(B) - \P(A \cap B),
\]
onde $\bar A $ é o \emph{\omterm{def:g10:proba:operations}{complemento}} de $ A $ (todos os resultados não $ A $).
Em particular, $\P(A \cup B) = \P(A) + \P(B)$ quando $ A $ e $ B $ são
\emph{\omterm{def:g10:proba:operations}{incompatível}} ($ A \cap B = \emptyset $).
\end{proposition}

\begin{proof}
O vazio \omterm{def:g11:prob:model}{evento} não contém resultado (soma vazia) e $\Omega $ contém
todos eles (soma total $1$). Cada resultado está exatamente em um dos $ A $,
$\bar A $, então $\P(A) + \P(\bar A) = 1$. Para o \omterm{def:g10:numbers:interunion}{união}: somando o $ p_i $
sobre $ A $ e então acabou $ B $ conta os resultados de $ A \cap B $ duas vezes;
subtraindo $\P(A \cap B)$ corrige a contagem dupla.
\end{proof}

\begin{example}
Em uma aula, $60\%$ estudar música ($ M $), $45\%$ praticar um esporte ($ S $), e
$25\%$ faça ambos. A proporção que faz pelo menos um é
$\P(M \cup S) = 0.60 + 0.45 - 0.25 = 0.80$, então $20\%$ não faça nada
(\omterm{def:g10:proba:operations}{complemento}).
\end{example}

\section{Variáveis aleatórias}

\begin{definition}[Variável aleatória, distribuição]\label{def:g11:prob:rv}
UM \emph{variável calculada}\index{variável calculada} sobre $\Omega $ é um \omterm{def:g11:func:function}{função}
$ X $ atribuindo um número a cada resultado. Isso é
\emph{distribuição}\index{distribuição} é a tabela do seu possível
valores $ x_1, \dots, x_k $ juntamente com suas \omterm{def:g10:proba:distribution}{probabilidades}
\[
\P(X = x_i) = \P\bigl(\{\omega : X(\omega) = x_i\}\bigr),
\qquad
\sum_{i=1}^k \P(X = x_i) = 1 .
\]
\end{definition}

\begin{example}\label{ex:g11:prob:game}
Um jogo: pague $2$ euros, jogue um dado justo e receba o valor mostrado se
é pelo menos $5$, nada de outra forma. Deixar $ G $ seja o \emph{ganho} (recebido
menos pago). Os resultados $1, 2, 3, 4$ dar $ G = -2$; o resultado $5$
dá $ G = 3$; o resultado $6$ dá $ G = 4$. O \omterm{def:g11:prob:rv}{distribuição} de $ G $:
\begin{center}
\begin{tabular}{c|ccc}
$ g $ & $-2$ & $3$ & $4$\\
\hline\rule{0pt}{11pt}%
$\P(G = g)$ & $\dfrac46$ & $\dfrac16$ & $\dfrac16$
\end{tabular}
\end{center}
\end{example}

\begin{method}[Construindo uma tabela de distribuição]\label{met:g11:prob:table}
Liste os possíveis valores de $ X $; para cada valor, colete os resultados
produzi-lo e somar suas \omterm{def:g10:proba:distribution}{probabilidades}; verifique se a linha de
somas de \omterm{def:g10:proba:distribution}{probabilidades} para $1$.
\end{method}

\section{Expectativa}

\begin{definition}[Expectativa]\label{def:g11:prob:expectation}
O \emph{expectativa}\index{expectativa} (ou significar) de $ X $ é a média
de seus valores ponderados por suas \omterm{def:g10:proba:distribution}{probabilidades}:
\[
\E(X) = \sum_{i=1}^{k} \P(X = x_i)\; x_i .
\]
\end{definition}

\begin{remark}
$\E(X)$ é o que significar de um grande número de repetições independentes de
o experimento seria --- a versão precisa desta afirmação é a
lei dos grandes números, provada em \cref{ch:g12:sums}. Observe o formal
analogia com o significar de um conjunto de dados com \omterm{def:g10:stats:series}{frequências}
(\cref{def:g11:stat:mean}): as \omterm{def:g10:proba:distribution}{probabilidades} desempenham o papel do relativo
\omterm{def:g10:stats:series}{frequências} $\frac{n_i}{n}$.
\end{remark}

\begin{example}\label{ex:g11:prob:gameexp}
Para o jogo de \cref{ex:g11:prob:game}:
\[
\E(G) = \frac46 \times (-2) + \frac16 \times 3 + \frac16 \times 4
= \frac{-8 + 3 + 4}{6} = -\frac16 \approx -0.17 .
\]
Em média o jogador perde cerca de $17$ centavos por jogo: o jogo é
desfavorável. Um jogo é chamado \emph{justo} quando o ganho esperado é $0$.
\end{example}

\begin{proposition}[Linearidade]\label{prop:g11:prob:linearity}
Para todos os reais $ a, b $:
\[
\E(aX + b) = a\,\E(X) + b .
\]
\end{proposition}

\begin{proof}
A variável $ aX + b $ leva o valor $ a x_i + b $ com \omterm{def:g10:proba:distribution}{probabilidade}
$\P(X = x_i)$, então
\[
\E(aX + b) = \sum_i \P(X = x_i)(a x_i + b)
= a \sum_i \P(X = x_i)\,x_i + b \underbrace{\sum_i \P(X =
x_i)}_{=\,1} = a\,\E(X) + b .
\]
\end{proof}

\section{Variância e desvio padrão}

\begin{definition}[Variância]\label{def:g11:prob:variance}
O \emph{variação}\index{variação} de $ X $ mede a propagação de sua
valores ao redor \omterm{def:g11:prob:expectation}{expectativa} $ m = \E(X)$:
\[
\V(X) = \E\bigl((X - m)^2\bigr)
= \sum_{i=1}^{k} \P(X = x_i)\,(x_i - m)^2,
\qquad
\sigma(X) = \sqrt{\V(X)} .
\]
\end{definition}

\begin{proposition}[Fórmula de atalho]\label{prop:g11:prob:konig}
\[
\V(X) = \E(X^2) - \E(X)^2 .
\]
\end{proposition}

\begin{proof}
Expandir $(x_i - m)^2 = x_i^2 - 2m\,x_i + m^2$ dentro da soma:
\[
\V(X) = \sum_i \P(X = x_i)\,x_i^2
- 2m \sum_i \P(X = x_i)\,x_i + m^2\sum_i \P(X = x_i)
= \E(X^2) - 2m^2 + m^2 ,
\]
o que é $\E(X^2) - m^2$. Esta é a versão de variável aleatória do
identidade comprovada para conjuntos de dados em \cref{prop:g11:stat:shortcut}.
\end{proof}

\begin{proposition}[Transformação afim]\label{prop:g11:prob:affine}
$\V(aX + b) = a^2\,\V(X)$ e $\sigma(aX + b) = \abs a\,\sigma(X)$.
\end{proposition}

\begin{proof}
Por linearidade, $ aX + b $ se desvia do seu \omterm{def:g11:prob:expectation}{expectativa} $ a m + b $ por
$(aX + b) - (am + b) = a(X - m)$: cada desvio é dimensionado por $ a $, portanto
cada desvio quadrado por $ a^2$, e assim é o seu peso significar. O
mudança $ b $ desapareceu --- mudar uma variável não altera sua
espalhar.
\end{proof}

\begin{example}\label{ex:g11:prob:variance}
Para o jogo de \cref{ex:g11:prob:game}:
$\E(G^2) = \frac46 \times 4 + \frac16 \times 9 + \frac16 \times 16
= \frac{16 + 9 + 16}{6} = \frac{41}{6}$, então
\[
\V(G) = \frac{41}{6} - \left(-\frac16\right)^2
= \frac{41}{6} - \frac{1}{36} = \frac{245}{36} \approx 6.8,
\qquad
\sigma(G) \approx 2.6 .
\]
A perda de $0.17$ por jogo, em média, vem com oscilações de tamanho típico
$2.6$ euros --- \omterm{def:g11:prob:expectation}{expectativa} sem \omterm{def:g11:stat:variance}{desvio padrão} conta apenas metade
a história.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g11:prob:1}
Uma carta é retirada de um padrão $52$-baralho de cartas. Calcule o \omterm{def:g10:proba:distribution}{probabilidade}
de desenho: um coração; um rei; um coração ou um rei; nem um coração nem um
rei.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:prob:2}
Dois dados justos são lançados e sua soma $ S $ está gravado. Usando o
$36$-modelo de resultado de \cref{ex:g11:prob:twodice}, calcular $\P(S = 6)$,
$\P(S \geq 10)$ e $\P(S \text{ is even})$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:prob:3}
UM variável aleatória tem \omterm{def:g11:prob:rv}{distribuição}
\begin{center}
\begin{tabular}{c|cccc}
$ x $ & $-1$ & $0$ & $2$ & $5$\\
\hline
$\P(X=x)$ & $0.3$ & $0.2$ & $0.4$ & $ p $
\end{tabular}
\end{center}
Encontrar $ p $, então calcule $\E(X)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:prob:4}
$ X $ satisfaz $\E(X) = 3$ e $\V(X) = 4$. Dê o \omterm{def:g11:prob:expectation}{expectativa}, \omterm{def:g11:prob:variance}{variação}
e \omterm{def:g11:stat:variance}{desvio padrão} de $ Y = 2X - 5$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:prob:5}
Uma roda da fortuna tem $10$ setores iguais: $5$ mostrar $0$ euros, $3$ mostrar
$2$ euros, $2$ mostrar $5$ euros. Custos de jogo $2$ euros. Dê o
\omterm{def:g11:prob:rv}{distribuição} do ganho $ G $ (ganhos menos aposta), calcule $\E(G)$ e
decidir se o jogo é favorável.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:prob:6}
Uma urna contém $3$ vermelho e $2$ bolas azuis; duas bolas são sorteadas
\emph{sem} substituição. Deixar $ X $ seja o número de bolas vermelhas sorteadas.
Construa o \omterm{def:g11:prob:rv}{distribuição} de $ X $ (enumere o $20$ sorteios ordenados ou use um
\omterm{met:g10:proba:tree}{árvore}) e calcular $\E(X)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:prob:7}
Para a roda de \cref{exo:g11:prob:5}, calcular $\V(G)$ e $\sigma(G)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:prob:8}
Uma companhia de seguros vende uma apólice por $80$ euros. Com \omterm{def:g10:proba:distribution}{probabilidade}
$0.05$ o cliente afirma $1000$ euros, com \omterm{def:g10:proba:distribution}{probabilidade} $0.01$ o cliente
reivindicações $5000$ euros, caso contrário nada. Deixar $ B $ ser o benefício da empresa
em uma política. Dê o \omterm{def:g11:prob:rv}{distribuição} de $ B $, calcular $\E(B)$ e
interpretar.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:prob:9}
Uma questão de múltipla escolha tem $4$ respostas, uma correta. Uma resposta correta
pontuações $+1$; para desencorajar a adivinhação, uma resposta errada pontua $ x < 0$.
Um aluno responde aleatoriamente. Para qual valor de $ x $ é do aluno
pontuação esperada zero?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:prob:10}
Duas moedas honestas são lançadas. Ann propõe: ``se ambas as moedas corresponderem, eu pago a você
$1$ euro; se eles forem diferentes, você me paga $1$ euro.'' Calcule o valor esperado
ganho de cada jogador. O jogo é justo? A mesma pergunta com três moedas,
onde Ann paga $2$ euros se todos os três corresponderem.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:prob:11}
Uma loteria vende $1000$ ingressos em $5$ euros cada. Um ingresso ganha
$2000$ euros, cinco bilhetes ganham $200$ euros, vinte bilhetes ganham $25$
euros.
\begin{enumerate}
  \item Dê o \omterm{def:g11:prob:rv}{distribuição} da receita total do organizador menos
        prêmios e do ganho de um único jogador $ G $.
  \item Calcular $\E(G)$ e o lucro total esperado do organizador.
        Verifique se as duas respostas são consistentes.
\end{enumerate}
\end{exercise}

\section{Problema: o jogo inacabado}

\begin{problem}[{Problema de fim de semana --- Pascal, Fermat e a feira
divisão de uma aposta interrompida: o nascimento da expectativa, a magia
de linearidade, e o que a expectativa não pode ver}]
\label{pb:g11:prob:1}
Em 1654, o Cavaleiro de M.éré --- o mesmo jogador cujos dados
apostas abriram o volume do ensino médio \omterm{def:g10:proba:distribution}{probabilidade} problema ---
fez a Blaise Pascal uma pergunta mais profunda: dois igualmente qualificados
os jogadores devem parar sua série no placar de $5$ para $3$, seis
ganha levando o $64$- pote de pistola. \emph{Como deve ser o pote
dividir de forma justa?} As cartas entre Pascal e Fermat respondendo
fundou o conceito no centro deste capítulo: \omterm{def:g11:prob:expectation}{expectativa}
(\cref{def:g11:prob:expectation}). Você irá derivar novamente seus
responda nos dois sentidos, então conheça o truque da festa da linearidade e
expectativaé um ponto cego.

\medskip
\noindent\textbf{Part I --- \omterm{def:g11:prob:expectation}{Expectativa} fluency.}
\begin{enumerate}
  \item Jogue um dado justo; seu ganho é (cara $- 3$) euros. Construir
        o \omterm{def:g11:prob:rv}{distribuição} tabela do ganho $ G $
        (\cref{met:g11:prob:table}) e calcular $\E(G)$.
  \item Calcular $ V(G)$ com a fórmula de atalho
        (\cref{prop:g11:prob:konig}) e $\sigma(G)$ (dois
        decimais).
  \item Sem nenhuma tabela nova, dê $\E(2G - 1)$ e
        $ V(2G - 1)$ (\cref{prop:g11:prob:affine}).
  \item Uma barraca cobra $ m $ euros para lançar dois dados e bolsos
        você a soma em euros. Qual é o \emph{justo} preço
        $ m $? A barraca cobra $8$: qual é a sua perda esperada
        por jogo?
  \item Um laptop que vale $800$ euros tem um $5\,\%$ \omterm{def:g11:func:parity}{chance} de
        destruição a cada ano. Calcule o seguro justo
        prêmio e o lucro esperado da seguradora por apólice
        se cobra $60$ euros.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Splitting the pot.}
Pontuação $5$--$3$, primeiro a $6$ ganha o $64$ pistolas, cartuchos são
lançamentos justos de moedas.
\begin{enumerate}[resume]
  \item Duas respostas erradas tentadoras: dividir $5 : 3$
        (proporcionalmente às rodadas vencidas), ou dar tudo para
        o líder. Diga em uma frase cada o que é injusto
        sobre eles.
  \item Método de Fermat: o jogador A precisa $1$ mais vitória, B precisa
        $3$; imagine jogar exatamente $3$ mais rodadas independentemente.
        Liste o $8$ resultados igualmente prováveis, marque quem ganha o
        série em cada um, e deduza a divisão justa do $64$
        pistolas.
  \item O ponto sutil que Fermat teve que defender: o verdadeiro jogo
        pode parar após uma rodada --- por que mesmo assim
        legítimo raciocinar sobre todos $3$ rodadas imaginárias?
  \item Método de Pascal, de trás para frente: calcule A's
        parcela justa nas pontuações (hipotéticas) $5$--$5$, então
        $5$--$4$, então $5$--$3$, cada vez calculando a média dos dois
        futuros igualmente prováveis. Compare com a questão 7.
  \item A mesma pergunta no placar $4$--$3$: quantos imaginários
        rodadas, quais resultados dão a série a B e qual é
        a divisão justa?
  \item Declare o princípio que os dois métodos compartilham --- o justo
        O valor de um futuro incerto é o seu \emph{\omterm{def:g11:prob:expectation}{expectativa}}
        --- e cite duas indústrias modernas que funcionam inteiramente com
        (a questão 5 mostra uma).
  \item Uma verificação de sanidade para cada pontuação: explique, com o
        linearidade de \omterm{def:g11:prob:expectation}{expectativa}
        (\cref{prop:g11:prob:linearity}), por que os dois jogadores
        as partes justas devem sempre somar exatamente o
        $64$- pote de pistola.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Linearity's party trick.}
\begin{enumerate}[resume]
  \item Recuperar $\E(\text{sum of two dice}) = 7$ em uma linha
        com linearidade (\cref{prop:g11:prob:linearity}), não
        $36$- tabela de células. Para o \emph{\omterm{def:g11:prob:variance}{variação}} da soma,
        a regra $ V(X + Y) = V(X) + V(Y)$ precisa que os dados sejam
        independente (admitido): calcule-o.
  \item A festa do chapéu: $ n $ convidados jogam seus chapéus em um
        pilha e cada um pega um de volta aleatoriamente. Deixar $ X $ contar
        os convidados que recuperam o próprio chapéu. Para um fixo
        convidado, qual é o \omterm{def:g10:proba:distribution}{probabilidade} de recuperar o seu próprio
        chapéu? Resumindo estes $ n $ \omterm{def:g11:func:parity}{chances} (linearidade!), calcule
        $\E(X)$. A resposta deveria surpreendê-lo: não
        depender de $ n $.
  \item Verificar $\E(X) = 1$ pela força bruta para $ n = 2$ e
        $ n = 3$ (listar os $2$ e $6$ igualmente provável
        atribuições de chapéus e média das contagens).
  \item As recuperações dos hóspedes estão longe de ser independentes (se
        $ n - 1$ os convidados recuperaram seus próprios chapéus, o último
        também deve). Onde exatamente foi feito o cálculo da questão 14
        \emph{não} requer independência? Linearidade de estado
        superpotência.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- What \omterm{def:g11:prob:expectation}{expectativa} cannot see.}
\begin{enumerate}[resume]
  \item Dois $5$- raspadinhas de euro: a carta A ganha $10$ euros com
        \omterm{def:g10:proba:distribution}{probabilidade} $\frac12$; carta B ganha $1\,000$ euros com
        \omterm{def:g10:proba:distribution}{probabilidade} $0.005$. Verifique se ambos os ganhos têm o
        mesmo \omterm{def:g11:prob:expectation}{expectativa}, então calcule ambos variações. Qual
        cartão as pessoas reais compram e o que estão comprando?
  \item O jogo de São Petersburgo: virar até a primeira coroa;
        se a primeira coroa ocorrer na virada $ n $ você recebe
        $2^n $ euros. Calcule a contribuição de cada $ n $ para o \omterm{def:g11:prob:expectation}{expectativa} e o \omterm{def:g11:prob:expectation}{expectativa} em si. Você faria
        pagar $1\,000$ euros para jogar uma vez? O que o confronto
        ensinar sobre \omterm{def:g11:prob:expectation}{expectativa}?
  \item Seguro por parte do cliente: pague $20$ euros, ou
        enfrentar um $1\,\%$ risco de perder $1\,000$? Compare os dois
        expectativase explique por que comprar o seguro pode
        ainda ser racional (o que uma única família não pode
        ``média acima''?).
  \item Final --- expectativabiografia, uma frase por
        capítulo: nascido em 1654 para dividir uma panela; seu superpoder,
