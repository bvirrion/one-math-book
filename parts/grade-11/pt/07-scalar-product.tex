\chapter{O produto escalar no plano}\label{ch:g11:scal}

O \omterm{def:g11:scal:dot}{produto escalar} anexa um \emph{número} para um par de \omterm{def:g11:vect:vector}{vetores}, e isso
número vê geometria: ele desaparece exatamente em direções perpendiculares,
e mede ângulos e comprimentos. É o motor por trás da lei da
\omterm{def:g11:trigo:cossin}{cossenos}, equações de círculos e cálculos de perpendicularidade. O mesmo
ferramenta, no espaço, é objeto de \cref{ch:g12:space}.

\section{Definição e primeiros cálculos}

\begin{definition}[Produto escalar]\label{def:g11:scal:dot}
O \emph{produto escalar}\index{produto escalar} de dois \omterm{def:g11:vect:vector}{vetores} $\vec u $
e $\vec v $ é o número
\[
\vec u \cdot \vec v
= \tfrac12\left(\norm{\vec u + \vec v}^2 - \norm{\vec u}^2
- \norm{\vec v}^2\right),
\]
onde $\norm{\vec u}$ denota o comprimento (\emph{norma}\index{norma}) de $\vec u $.
\end{definition}

\begin{theorem}[As quatro faces do produto escalar]\label{thm:g11:scal:faces}
Deixar $\vec u\,(x, y)$ e $\vec v\,(x', y')$ ser diferente de zero \omterm{def:g11:vect:vector}{vetores}, e
$\theta $ o ângulo entre eles. Então
\begin{enumerate}
  \item \emph{(\omterm{def:g10:coordgeom:system}{coordenadas})} $\vec u \cdot \vec v = xx' + yy'$;
  \item \emph{(\omterm{def:g11:trigo:cossin}{cosseno})} $\vec u \cdot \vec v =
        \norm{\vec u}\,\norm{\vec v}\cos\theta $;
  \item \emph{(projeção)} $\vec u \cdot \vec v = \vec u \cdot \vec v'$,
        onde $\vec v'$ é o \omterm{def:g11:scal:orthogonal}{ortogonal} projeção de $\vec v $ no
        direção de $\vec u $; se $\vec u = \vect{OA}$ e
        $\vec v = \vect{OB}$, então
        $\vec u \cdot \vec v = \overline{OA} \times \overline{OH}$, um
        produto de comprimentos sinalizados, onde $ H $ é o pé do
        perpendicularmente de $ B $ para a linha $(OA)$;
  \item \emph{(\omterm{def:g11:scal:dot}{normas})} a fórmula definidora de \cref{def:g11:scal:dot}.
\end{enumerate}
\end{theorem}

\begin{proof}[Prova de 1 e 2]
\emph{1.} Em \omterm{def:g10:coordgeom:system}{coordenadas}, $\norm{\vec u}^2 = x^2 + y^2$ (Pitágoras) e
$\vec u + \vec v $ tem \omterm{def:g10:coordgeom:system}{coordenadas} $(x + x', y + y')$, então
\begin{align*}
2\,\vec u \cdot \vec v
&= (x + x')^2 + (y + y')^2 - (x^2 + y^2) - (x'^2 + y'^2)\\
&= 2xx' + 2yy',
\end{align*}
depois expandindo os quadrados e cancelando.

\emph{2.} Escolha o sistema de \omterm{def:g10:coordgeom:system}{coordenadas} para que $\vec u $ pontos ao longo do
positivo $ x $-eixo: $\vec u\,(\norm{\vec u}, 0)$ e
$\vec v\,(\norm{\vec v}\cos\theta,\ \norm{\vec v}\sin\theta)$
(\cref{ch:g11:trigo}). A Fórmula 1 então dá
$\vec u \cdot \vec v = \norm{\vec u}\,\norm{\vec v}\cos\theta + 0$.
A Fórmula 3 apresenta o mesmo cálculo: a projeção de $\vec v $ no $ x $-axis tem comprimento assinado $\norm{\vec v}\cos\theta =
\overline{OH}$.
\end{proof}

\begin{example}\label{ex:g11:scal:compute}
$\vec u\,(3, 1)$ e $\vec v\,(2, -4)$:
$\vec u \cdot \vec v = 6 - 4 = 2$. O \omterm{def:g11:scal:dot}{normas} são
$\norm{\vec u} = \sqrt{10}$ e $\norm{\vec v} = \sqrt{20}$, então
$\cos\theta = \frac{2}{\sqrt{10}\sqrt{20}} = \frac{2}{10\sqrt2}
= \frac{\sqrt2}{10}$: o ângulo é um pouco menor que $82^\circ $.
\end{example}

\begin{proposition}[Regras algébricas]\label{prop:g11:scal:rules}
Para todos \omterm{def:g11:vect:vector}{vetores} e tudo $\lambda \in \R $:
\[
\vec u \cdot \vec v = \vec v \cdot \vec u, \qquad
\vec u \cdot (\vec v + \vec w) = \vec u \cdot \vec v + \vec u \cdot
\vec w, \qquad
(\lambda \vec u)\cdot \vec v = \lambda\,(\vec u \cdot \vec v),
\]
\[
\vec u \cdot \vec u = \norm{\vec u}^2,
\qquad
\norm{\vec u + \vec v}^2 = \norm{\vec u}^2 + 2\,\vec u\cdot\vec v +
\norm{\vec v}^2 .
\]
\end{proposition}

\begin{proof}
Todos seguem a fórmula de \omterm{def:g10:coordgeom:system}{coordenadas}: por exemplo
$\vec u \cdot (\vec v + \vec w) = x(x' + x'') + y(y' + y'') =
(xx' + yy') + (xx'' + yy'')$. A última identidade se expande
$\norm{\vec u + \vec v}^2 = (\vec u + \vec v)\cdot(\vec u + \vec v)$
usando as três primeiras regras --- o \omterm{def:g11:scal:dot}{produto escalar} se comporta exatamente como um
produto comum, então as identidades algébricas usuais se aplicam a \omterm{def:g11:vect:vector}{vetores}.
\end{proof}

\section{Ortogonalidade}

\begin{definition}[Vetores ortogonais]\label{def:g11:scal:orthogonal}
Dois \omterm{def:g11:vect:vector}{vetores} são \emph{ortogonal}\index{ortogonal}, escrito
$\vec u \perp \vec v $, se $\vec u \cdot \vec v = 0$.
\end{definition}

\begin{remark}
Para diferente de zero \omterm{def:g11:vect:vector}{vetores}, fórmula 2 de \cref{thm:g11:scal:faces} mostra
$\vec u \cdot \vec v = 0 \iff \cos\theta = 0 \iff $ as direções são
perpendicular. O zero vetor é \omterm{def:g11:scal:orthogonal}{ortogonal} para tudo.
\end{remark}

\begin{example}
$\vec u\,(3, 2)$ e $\vec v\,(-4, 6)$: $\vec u \cdot \vec v = -12 + 12 =
0$: \omterm{def:g11:scal:orthogonal}{ortogonal}. Em geral, $\vec u\,(a, b)$ é sempre \omterm{def:g11:scal:orthogonal}{ortogonal} para
$\vec v\,(-b, a)$ --- uma rotação de um quarto de volta.
\end{example}

\section{A lei dos cossenos}

\begin{theorem}[Lei dos cossenos]\label{thm:g11:scal:cosines}
\index{lei dos cossenos}
Em um triângulo $ ABC $ com lados $ a = BC $, $ b = CA $, $ c = AB $ e ângulo
$\widehat A $ no vértice $ A $:
\[
a^2 = b^2 + c^2 - 2bc\,\cos\widehat A .
\]
\end{theorem}

\begin{proof}
Escrever $\vect{BC} = \vect{BA} + \vect{AC} = \vect{AC} - \vect{AB}$ e
expandir com \cref{prop:g11:scal:rules}:
\[
a^2 = \norm{\vect{BC}}^2
= \norm{\vect{AC}}^2 - 2\,\vect{AC}\cdot\vect{AB} + \norm{\vect{AB}}^2
= b^2 + c^2 - 2\,\vect{AB}\cdot\vect{AC}.
\]
Pelo \omterm{def:g11:trigo:cossin}{cosseno} fórmula,
$\vect{AB}\cdot\vect{AC} = c\,b\,\cos\widehat A $.
\end{proof}

\begin{remark}
Quando $\widehat A $ é um ângulo reto, $\cos\widehat A = 0$ e a lei de
\omterm{def:g11:trigo:cossin}{cossenos} reduz ao teorema de Pitágoras $ a^2 = b^2 + c^2$: é
Pitágoras com um termo de correção para ângulos não retos.
\end{remark}

\begin{example}\label{ex:g11:scal:cosines}
Medida de dois lados de um triângulo $ b = 5$ e $ c = 8$, encerrando um ângulo
$\widehat A = \frac\pi3$. Então
\[
a^2 = 25 + 64 - 2 \times 5 \times 8 \times \frac12 = 89 - 40 = 49,
\]
então $ a = 7$.
\end{example}

\section{Aplicações: círculos e linhas perpendiculares}

\begin{proposition}[Círculo com um determinado diâmetro]\label{prop:g11:scal:diameter}
Um ponto $ M $ encontra-se no círculo de diâmetro $[AB]$ se e somente se
\[
\vect{MA} \cdot \vect{MB} = 0 .
\]
\end{proposition}

\begin{proof}
Se $ M \neq A, B $, a condição indica o ângulo em $ M $ no triângulo
$ AMB $ está certo, e os pontos que vêem um segmento sob um ângulo reto são
exatamente aqueles do círculo com esse diâmetro. Aqui está uma prova direta:
deixe $ O $ seja o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de $[AB]$ e $ r = \frac{AB}{2}$. Então
$\vect{MA} = \vect{MO} + \vect{OA}$ e
$\vect{MB} = \vect{MO} + \vect{OB} = \vect{MO} - \vect{OA}$, então
\[
\vect{MA}\cdot\vect{MB}
= \norm{\vect{MO}}^2 - \norm{\vect{OA}}^2 = MO^2 - r^2,
\]
que desaparece exatamente quando $ MO = r $, ou seja,\ quando $ M $ está no círculo de
centro $ O $ e raio $ r $.
\end{proof}

\begin{proposition}[Vetor normal de uma linha]\label{prop:g11:scal:normal}
O vetor $\vec n\,(a, b)$ é \omterm{def:g11:scal:orthogonal}{ortogonal} para cada \omterm{def:g11:vect:direction}{vetor de direção} do
linha $ d : ax + by + c = 0$; um diz $\vec n $ é um \emph{normais
vetores}\index{normais
vetores} de $ d $. Duas linhas são exatamente perpendiculares
quando seu normal (ou direção) \omterm{def:g11:vect:vector}{vetores} são \omterm{def:g11:scal:orthogonal}{ortogonal}.
\end{proposition}

\begin{proof}
UM \omterm{def:g11:vect:direction}{vetor de direção} de $ d $ é $\vec u\,(-b, a)$
(\cref{thm:g11:vect:cartesian}), e
$\vec n \cdot \vec u = a(-b) + b(a) = 0$.
\end{proof}

\begin{method}[Usando vetores normais]\label{met:g11:scal:normal}
\begin{itemize}
  \item Linha através $ A $ com normal $\vec n\,(a,b)$: escrever
        $\vect{AM} \cdot \vec n = 0$, que se expande para $ ax + by + c = 0$
        com $ c $ determinado por $ A $.
  \item Perpendicularidade de $ d: ax+by+c = 0$ e $ d': a'x+b'y+c' = 0$:
        verificar $ aa' + bb' = 0$.
\end{itemize}
\end{method}

\begin{example}\label{ex:g11:scal:perpline}
A linha através $ P(2, -1)$ perpendicular a $ d : 3x - y + 4 = 0$: um
\omterm{def:g11:vect:direction}{vetor de direção} de $ d $ é $(1, 3)$, que serve como \emph{normal} vetor
da perpendicular. Equação: $ x + 3y + c = 0$ com
$2 - 3 + c = 0$, então $ c = 1$: a linha é $ x + 3y + 1 = 0$.
\end{example}

\begin{example}[Equação de um círculo]\label{ex:g11:scal:circle}
O círculo do centro $\Omega(1, 2)$ e raio $3$ é o conjunto de pontos
$ M(x,y)$ com $\Omega M^2 = 9$:
\[
(x - 1)^2 + (y - 2)^2 = 9 .
\]
Por outro lado, $ x^2 + y^2 - 2x - 4y - 4 = 0$ reescreve, completando o
quadrados (\cref{ch:g11:quad}), como $(x-1)^2 + (y-2)^2 = 9$: o mesmo
círculo.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g11:scal:1}
Calcular $\vec u \cdot \vec v $ para
\[
\vec u\,(2, 5),\ \vec v\,(3, -1); \qquad
\vec u\,(1, -4),\ \vec v\,(8, 2); \qquad
\norm{\vec u} = 3,\ \norm{\vec v} = 4,\ \theta = \frac{\pi}{3} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:scal:2}
$ ABC $ é um triângulo equilátero de lado $6$. Calcular
$\vect{AB} \cdot \vect{AC}$ e $\vect{AB} \cdot \vect{BC}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:scal:3}
Para qual valor de $ t $ são $\vec u\,(3, t)$ e $\vec v\,(t - 8, 1)$
\omterm{def:g11:scal:orthogonal}{ortogonal}? E para quais valores de $ t $ é $\vec u\,(t, 2)$ \omterm{def:g11:scal:orthogonal}{ortogonal}
para si mesmo menos $\vec v\,(4, 2)$, ou seja\ $\vec u \perp (\vec u - \vec v)$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:scal:4}
Calcule o ângulo $\theta $ entre $\vec u\,(1, 2)$ e $\vec v\,(3, 1)$
(dar $\cos\theta $ exatamente, então $\theta $ para o grau mais próximo).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:scal:5}
Um triângulo tem lados $ b = 4$, $ c = 6$ e ângulo
$\widehat A = \frac{2\pi}{3}$ entre eles. Calcule o terceiro lado $ a $.
Então, em um triângulo com lados $ a = 7$, $ b = 5$, $ c = 4$, calcular
$\cos\widehat A $ e decidir se $\widehat A $ é agudo ou obtuso.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:scal:6}
Deixar $ A(1, 1)$, $ B(5, -1)$ e $ C(3, 5)$. Calcular
$\vect{AB} \cdot \vect{AC}$; é o ângulo $\widehat A $ certo? Calcule o
comprimentos de três lados e verifique sua resposta com a lei de \omterm{def:g11:trigo:cossin}{cossenos}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:scal:7}
Dê um equação do círculo de centro $\Omega(-2, 3)$ e raio $5$.
Em seguida, determine o centro e o raio do círculo
\[
x^2 + y^2 - 6x + 2y - 6 = 0 .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:scal:8}
Deixar $ A(-3, 0)$ e $ B(0, 2)$. Usando \cref{prop:g11:scal:diameter}, encontrar
um equação do círculo de diâmetro $[AB]$, e verifique se a origem
$ O(0,0)$ mente sobre isso.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:scal:9}
Encontre um equação da linha através $ P(1, 3)$ perpendicular a
$ d : 2x + y - 7 = 0$, e o \omterm{def:g10:coordgeom:system}{coordenadas} do pé $ H $ do
perpendicular (o \omterm{def:g10:numbers:interunion}{interseção} das duas linhas). Deduza a distância
de $ P $ para $ d $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:scal:10}
Usando $\norm{\vec u + \vec v}^2 = \norm{\vec u}^2 + 2\vec u\cdot\vec v +
\norm{\vec v}^2$ e seu análogo para $\vec u - \vec v $, comprove o
\emph{identidade do paralelogramo}:
\[
\norm{\vec u + \vec v}^2 + \norm{\vec u - \vec v}^2
= 2\norm{\vec u}^2 + 2\norm{\vec v}^2 ,
\]
e interpretá-lo em um paralelogramo (diagonais versus lados).
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:scal:11}
Deixar $ A $ e $ B $ ser dois pontos com $ AB = 4$, e $ I $ o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de
$[AB]$.
\begin{enumerate}
  \item Mostre isso para cada ponto $ M $:
        $\vect{MA}\cdot\vect{MB} = MI^2 - 4$.
        (Inserir $ I $: $\vect{MA} = \vect{MI} + \vect{IA}$.)
  \item Deduza o conjunto de pontos $ M $ tal que
        $\vect{MA}\cdot\vect{MB} = 12$.
\end{enumerate}
\end{exercise}

\section{Problema: Dois presentes do produto escalar}

\begin{problem}[{Problema de fim de semana --- as fórmulas de adição caem
de um produto escalar, a distância até uma linha de outro,
e as três altitudes se encontram como um bônus}]
\label{pb:g11:scal:1}
O \omterm{def:g11:scal:dot}{produto escalar} parece contabilidade --- multiplicar, adicionar ---
no entanto, entrega, quase gratuitamente, dois tesouros que
resistiu a séculos de perseguição: o \emph{fórmulas de adição}
de trigonometria (calcular $\cos 15^\circ $ exatamente!) e o
\emph{distância de um ponto a uma reta} em uma fração limpa. Como
um presente de despedida prova que as três altitudes de cada
triângulo passa por um ponto. Todas as quatro faces de
\cref{thm:g11:scal:faces} apresentar-se-á para o serviço.

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item Para $\vec u(3, 4)$ e $\vec v(-1, 2)$: calcular
        $\vec u \cdot \vec v $, os dois \omterm{def:g11:scal:dot}{normas}, e o ângulo
        entre o \omterm{def:g11:vect:vector}{vetores} (até o décimo de grau).
  \item Encontrar $ k $ para que $(2, k)$ e $(3, -4)$ são \omterm{def:g11:scal:orthogonal}{ortogonal}
        (\cref{def:g11:scal:orthogonal}).
  \item Cara de física: um trenó é puxado $10$ m por um $50$ Força N
        em $60^\circ $ para o chão. Calcule o trabalho
        $ W = \vec F \cdot \vec d $.
  \item Lei de \omterm{def:g11:trigo:cossin}{cossenos} (\cref{thm:g11:scal:cosines}): dois lados
        de $5$ e $7$ delimitar um ângulo de $60^\circ $; encontre o
        exatamente o terceiro lado.
  \item Reverso: um triângulo tem lados $4$, $6$, $8$. Calcule o
        \omterm{def:g11:trigo:cossin}{cosseno} do ângulo oposto ao lado $8$, classifique o
        ângulo e dê-o ao décimo de grau.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- First gift: the addition
formulas.}
Deixar $\vec u = (\cos a, \sin a)$ e $\vec v = (\cos b, \sin b)$:
duas unidades \omterm{def:g11:vect:vector}{vetores} em ângulos $ a $ e $ b $.
\begin{enumerate}[resume]
  \item Calcular $\vec u \cdot \vec v $ duas vezes --- uma vez com
        \omterm{def:g10:coordgeom:system}{coordenadas}, uma vez com \omterm{def:g11:scal:dot}{normas} e o ângulo entre
        \omterm{def:g11:vect:vector}{vetores} --- e conclua a fórmula mãe:
        \[
        \cos(a - b) = \cos a \cos b + \sin a \sin b .
        \]
  \item Substituir $-b $ para $ b $ (paridades de \omterm{def:g11:trigo:cossin}{seno} e \omterm{def:g11:trigo:cossin}{cosseno}!)
        para obter $\cos(a + b)$.
  \item Obter $\sin(a + b)$ da identidade complementar
        $\sin x = \cos\left(\frac\pi2 - x\right)$ aplicado a
        $ x = a + b $.
  \item Finalmente valores exatos: calcular $\cos 15^\circ $ (como
        $\cos(45^\circ - 30^\circ)$) e $\sin 75^\circ $. Verifique
        ambos numericamente.
  \item Definir $ b = a $ para derivar as fórmulas de ângulo duplo:
        $\cos 2a $ (em suas três formas) e $\sin 2a $.
  \item De $\cos 2a = 1 - 2\sin^2 a $, calcular
        $\sin^2 15^\circ $, então $\sin 15^\circ $ como um aninhado
        radical --- e verifique se seu quadrado concorda com o
        valor $\frac{\sqrt6 - \sqrt2}{4}$ aquela pergunta 9
        método dá.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Second gift: the distance to a
line.}
\begin{enumerate}[resume]
  \item Deixar $ d $ seja a linha $ ax + by + c = 0$, com normais
        vetor $\vec n(a, b)$ (\cref{prop:g11:scal:normal}),
        e $ P(x_0, y_0)$ um ponto. Tomando qualquer ponto $ A $ sobre $ d $,
        explique por que a distância de $ P $ para $ d $ é
        $\dfrac{\abs{\vect{AP} \cdot \vec n}}{\norm{\vec n}}$,
        e deduza a fórmula
        \[
        \operatorname{dist}(P, d)
        = \frac{\abs{a x_0 + b y_0 + c}}{\sqrt{a^2 + b^2}} .
        \]
        Calcule a distância de $ P(5, 6)$ para
        $3x + 4y - 12 = 0$.
  \item Escreva o equação do círculo centrado em $(5, 6)$
        \omterm{def:g11:deriv:tangent}{tangente} para essa linha.
  \item Verifique a fórmula: calcule a distância do
        origem para $3x + 4y - 12 = 0$, então verifique-o por
        calculando o pé da perpendicular à origem
        explicitamente.
  \item Triângulo $ A(0,0)$, $ B(6,0)$, $ C(2,5)$: calcule sua área
        usando a base $[AB]$; então calcule a distância de
        $ B $ para a linha $(AC)$ e verifique se reproduz o mesmo
        área.
  \item O círculo do centro $(2, 1)$ e raio $2$, contra
        a linha $ x + y - 6 = 0$: calcule a distância do centro
        para a linha e classificar sua posição (secante,
        \omterm{def:g11:deriv:tangent}{tangente}, disjunto).
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The bonus: altitudes concur.}
\begin{enumerate}[resume]
  \item Generalizar \cref{exo:g11:scal:11}: para quaisquer dois pontos
        $ A $, $ B $ com \omterm{prop:g10:coordgeom:midpoint}{ponto médio} $ I $, provar
        \[
        \vect{MA} \cdot \vect{MB} = MI^2 - \frac{AB^2}{4}
        \quad\text{for every point } M .
        \]
  \item Deduzir \cref{prop:g11:scal:diameter} em uma linha: o
        conjunto de pontos $ M $ com
        $\vect{MA} \cdot \vect{MB} = 0$ é o círculo de
        diâmetro $[AB]$.
  \item As altitudes: deixe $ H $ seja o \omterm{def:g10:numbers:interunion}{interseção} do
        altitudes de $ A $ e de $ B $ em um triângulo $ ABC $
        (então $\vect{HA} \cdot \vect{BC} = 0$ e
        $\vect{HB} \cdot \vect{CA} = 0$). Prove a identidade
        \[
        \vect{HA} \cdot \vect{BC} + \vect{HB} \cdot \vect{CA}
        + \vect{HC} \cdot \vect{AB} = 0
        \]
        --- expandir tudo de $ H $ com Chasles --- e
        concluir que $ H $ também fica na altitude de $ C$:
        as três altitudes são simultâneas (no
        \emph{ortocentro}\index{ortocentro}).
  \item Final: quatro faces, três troféus. Qual cara de
        \cref{thm:g11:scal:faces} carregava as fórmulas de adição,
        qual a fórmula da distância, qual o \omterm{pb:g11:scal:1}{ortocentro} ---
        e por que “calcular a mesma quantidade duas vezes” é o lema
        de todo o capítulo?
\end{enumerate}
\end{problem}
