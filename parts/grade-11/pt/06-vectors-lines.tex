\section{Vetores: uma recapitulaĂ§ĂŁo}

\begin{definition}[Vetor, coordenadas]\label{def:g11:vect:vector}
UM \emph{vetores}\index{vetores} $\vec u $ representa um deslocamento: todos
setas com a mesma direĂ§ĂŁo, orientaĂ§ĂŁo e comprimento representam o mesmo
vetor. Em um sistema de \omterm{def:g10:coordgeom:system}{coordenadas}, $\vec u\,(x, y)$ registra o deslocamento
($ x $ horizontalmente, $ y $ verticalmente); por pontos $ A(x_A, y_A)$ e
$ B(x_B, y_B)$,
\[
\vect{AB}\,\bigl(x_B - x_A,\ y_B - y_A\bigr).
\]
Vetores somam \omterm{def:g10:coordgeom:system}{coordenadas} e $\lambda\vec u $ tem \omterm{def:g10:coordgeom:system}{coordenadas}
$(\lambda x, \lambda y)$.
\end{definition}

\begin{proposition}[Ponto mĂ©dio]\label{prop:g11:vect:midpoint}
O \omterm{prop:g10:coordgeom:midpoint}{ponto mĂ©dio} $ M $ do segmento $[AB]$ tem \omterm{def:g10:coordgeom:system}{coordenadas}
$\left(\frac{x_A + x_B}{2},\ \frac{y_A + y_B}{2}\right)$.
\end{proposition}

\begin{proof}
$ M $ Ă© caracterizado por $\vect{AM} = \frac12\vect{AB}$: em termos de \omterm{def:g10:coordgeom:system}{coordenadas},
$ x_M - x_A = \frac12(x_B - x_A)$, entĂŁo $ x_M = \frac{x_A + x_B}{2}$ e
da mesma forma para $ y_M $.
\end{proof}

\section{Colinearidade e o determinante}

\begin{definition}[Vetores colineares]\label{def:g11:vect:collinear}
Dois \omterm{def:g11:vect:vector}{vetores} sĂŁo \emph{colinear}\index{colinear} se alguĂ©m Ă© um
mĂșltiplo escalar do outro: $\vec v = \lambda \vec u $ para alguns
$\lambda \in \R $ (ou $\vec u = \vec 0$). Geometricamente: eles tĂȘm o mesmo
direĂ§ĂŁo, ou um deles Ă© zero.
\end{definition}

\begin{definition}[Determinante]\label{def:g11:vect:det}
O \emph{determinante}\index{determinante} de $\vec u\,(x, y)$ e
$\vec v\,(x', y')$ Ă©
\[
\det(\vec u, \vec v) =
\begin{vmatrix} x & x' \\ y & y' \end{vmatrix}
= x y' - x' y .
\]
\end{definition}

\begin{theorem}[CritĂ©rio de colinearidade]\label{thm:g11:vect:collinearity}
Dois \omterm{def:g11:vect:vector}{vetores} $\vec u $ e $\vec v $ sĂŁo \omterm{def:g11:vect:collinear}{colinear} se e somente se
$\det(\vec u, \vec v) = 0$.
\end{theorem}

\begin{proof}
($\Rightarrow $) Se $\vec v = \lambda\vec u $, entĂŁo $ x' = \lambda x $ e
$ y' = \lambda y $, entĂŁo
$\det(\vec u, \vec v) = x(\lambda y) - (\lambda x)y = 0$; se
$\vec u = \vec 0$, o \omterm{def:g11:vect:det}{determinante} Ă© trivialmente $0$.

($\Leftarrow $) Suponha $ xy' - x'y = 0$ e $\vec u \neq \vec 0$, dizer
$ x \neq 0$ (o caso $ y \neq 0$ Ă© simĂ©trico). Definir
$\lambda = \frac{x'}{x}$. EntĂŁo $ x' = \lambda x $ por construĂ§ĂŁo, e o
relaĂ§ĂŁo $ xy' = x'y = \lambda x y $ dĂĄ, depois de dividir por $ x \neq 0$,
$ y' = \lambda y $. Por isso $\vec v = \lambda \vec u $.
\end{proof}

\begin{example}\label{ex:g11:vect:collinear}
$\vec u\,(3, -2)$ e $\vec v\,(-6, 4)$:
$\det = 3 \times 4 - (-6) \times (-2) = 12 - 12 = 0$, \omterm{def:g11:vect:collinear}{colinear} (de fato
$\vec v = -2\vec u $). $\vec u\,(3, -2)$ e $\vec w\,(2, 1)$:
$\det = 3 + 4 = 7 \neq 0$, nĂŁo \omterm{def:g11:vect:collinear}{colinear}.
\end{example}

\begin{method}[Alinhamento de trĂȘs pontos]\label{met:g11:vect:alignment}
$ A $, $ B $, $ C $ estĂŁo alinhados se e somente se $\vect{AB}$ e $\vect{AC}$ sĂŁo
\omterm{def:g11:vect:collinear}{colinear}: calcular ambos \omterm{def:g11:vect:vector}{vetores} e verifique
$\det\bigl(\vect{AB}, \vect{AC}\bigr) = 0$.
\end{method}

\section{EquaĂ§Ă”es cartesianas de retas}

\begin{definition}[Vetor de direĂ§ĂŁo]\label{def:g11:vect:direction}

\begin{theorem}[EquaĂ§ĂŁo cartesiana]\label{thm:g11:vect:cartesian}

\]

$\vec u\,(-b, a)$.
\end{theorem}

\begin{proof}

ponto $ M(x, y)$ mente $ d $ exatamente quando $\vect{AM}$ Ă© \omterm{def:g11:vect:collinear}{colinear} com
$\vec u $, ou seja,\ (\cref{thm:g11:vect:collinearity})
\[

= (x - x_A)\,a - (-b)(y - y_A)
= ax + by - (a x_A + b y_A),
\]

os pontos $\left(x, -\frac{a x + c}{b}\right)$: subtraindo dois deles
mostra que cada acorde Ă© \omterm{def:g11:vect:collinear}{colinear} com $(-b, a)$, e uma soluĂ§ĂŁo sempre
existe --- Ă© a linha atravĂ©s dele dirigida por $(-b, a)$.
\end{proof}

$\vect{AB}\,(x_B - x_A, y_B - y_A)$; combine com $(-b, a)$, ou seja,\ pegue
$ a = y_B - y_A $ e $ b = -(x_B - x_A)$; determinar $ c $ conectando o
\omterm{def:g10:coordgeom:system}{coordenadas} de $ A $.
\end{method}

\section{PosiĂ§Ă”es relativas de duas linhas}

\begin{proposition}[Paralelo ou cruzando]\label{prop:g11:vect:positions}
Duas linhas com \omterm{def:g11:vect:vector}{vetores} de direĂ§ĂŁo $\vec u $ e $\vec v $ sĂŁo paralelos se e
somente se $\det(\vec u, \vec v) = 0$. Em equaĂ§ĂŁo forma: $ d : ax + by + c =

$ ab' - a'b = 0$. Retas nĂŁo paralelas se encontram exatamente em um ponto.
\end{proposition}

\begin{proof}

\cref{thm:g11:vect:collinearity}; com $\vec u\,(-b, a)$ e
$\vec v\,(-b', a')$, o \omterm{def:g11:vect:det}{determinante} Ă© $(-b)a' - (-b')a = ab' - a'b $.
Se as linhas nĂŁo forem paralelas, resolvendo os dois equaĂ§Ă”es simultaneamente
(por substituiĂ§ĂŁo ou combinaĂ§ĂŁo) leva a uma soluĂ§ĂŁo Ășnica: o sistema
se comporta como um \omterm{def:g11:vect:det}{determinante} diferente de zero $2 \times 2$ sistema.
\end{proof}

\begin{method}[InterseĂ§ĂŁo de duas linhas]\label{met:g11:vect:intersection}
Resolva o sistema dos dois equaĂ§Ă”es: isolar um desconhecido em um equaĂ§ĂŁo
e substituir no outro, ou combinar o equaĂ§Ă”es para eliminar um
desconhecido. Sempre verifique o ponto resultante em \emph{ambos} equaĂ§Ă”es.
\end{method}

\begin{example}\label{ex:g11:vect:intersection}
Cruzar $ d: x - 3y + 5 = 0$ e $ d': 2x + y - 4 = 0$. De $ d'$:
$ y = 4 - 2x $. Substituindo em $ d $:
\[

\iff x - 12 + 6x + 5 = 0
\iff 7x = 7
\iff x = 1,
\]

\end{example}
