\begin{solution}{exo:g11:vect:1}
$\vect{AB}\,(3, 4)$, $\vect{AC}\,(-3, 2)$,
$\vect{AB} + \vect{AC}\,(0, 6)$,
$2\vect{AB} - 3\vect{AC}\,(6 + 9,\ 8 - 6) = (15, 2)$. The \omterm{prop:g10:coordgeom:midpoint}{midpoint} of
$[BC]$ is $\left(\frac{5 + (-1)}{2}, \frac{3+1}{2}\right) = (2, 2)$.
\end{solution}

\begin{solution}{exo:g11:vect:2}
$\det = 4 \times 3 - (-2) \times (-6) = 12 - 12 = 0$: \omterm{def:g11:vect:collinear}{collinear} (indeed
$\vec v = -\frac12 \vec u$).

$\det = 2 \times 7 - 3 \times 5 = -1 \neq 0$: not \omterm{def:g11:vect:collinear}{collinear}.

$\det = 0 \times (-8) - 0 \times 3 = 0$: \omterm{def:g11:vect:collinear}{collinear} (both vertical).
\end{solution}

\begin{solution}{exo:g11:vect:3}
Through $A(2,1)$, direction $(3, -1)$: matching $(-b, a) = (3, -1)$ gives
$a = -1$, $b = -3$, so $-x - 3y + c = 0$; plugging in $A$: $-2 - 3 + c =
0$, $c = 5$. Multiplying by $-1$: $x + 3y - 5 = 0$.

Through $B(0,4)$ and $C(2,0)$: $\vect{BC}\,(2, -4)$, so $a = -4$,
$b = -2$: $-4x - 2y + c = 0$; plugging in $B$: $-8 + c = 0$, $c = 8$.
Simplifying by $-2$: $2x + y - 4 = 0$. Check with $C$: $4 + 0 - 4 = 0$.
\end{solution}

\begin{solution}{exo:g11:vect:4}
A \omterm{def:g11:vect:direction}{direction vector} is $(-b, a) = (2, 3)$. \omterm{def:g10:numbers:interunion}{Intersections}: $x = 0$ gives
$y = 3$, point $(0, 3)$; $y = 0$ gives $x = -2$, point $(-2, 0)$. For
$P(2,6)$: $3 \times 2 - 2 \times 6 + 6 = 0$, so $P \in d$. For $Q(1,4)$:
$3 - 8 + 6 = 1 \neq 0$, so $Q \notin d$.
\end{solution}

\begin{solution}{exo:g11:vect:5}
$\vect{AB}\,(3, 6)$ and $\vect{AC}\,(-3, -6)$:
$\det = 3 \times (-6) - (-3) \times 6 = 0$, so $A$, $B$, $C$ are aligned.

$\vect{DE}\,(2, 3)$ and $\vect{DF}\,(5, 8)$:
$\det = 2 \times 8 - 5 \times 3 = 1 \neq 0$: $D$, $E$, $F$ are not
aligned.
\end{solution}

\begin{solution}{exo:g11:vect:6}
First pair: adding the two \omterm{def:g10:algebra:equation}{equations}, $(2x + y - 7) + (x - y + 1) =
3x - 6 = 0$, so $x = 2$, then $y = x + 1 = 3$: they meet at $(2, 3)$
(check: $4 + 3 - 7 = 0$).

Second pair: $ab' - a'b = 1 \times 4 - (-2) \times (-2) = 0$, so the
lines are parallel; multiplying $e$ by $-2$ gives $-2x + 4y - 6 = 0$,
which differs from $e'$ ($-6 \neq 1$): strictly parallel, no
\omterm{def:g10:numbers:interunion}{intersection}.
\end{solution}

\begin{solution}{exo:g11:vect:7}
For $d: 2x - y + 3 = 0$: direction $(-b, a) = (1, 2)$ and the point
$(0, 3)$ lies on $d$, so $d = \{(t,\ 3 + 2t) : t \in \R\}$.

For the parametric line through $(1, -3)$ with direction $(2, 1)$:
$a = 1$, $b = -2$, so $x - 2y + c = 0$; plugging in $(1, -3)$:
$1 + 6 + c = 0$, $c = -7$. \omterm{def:g10:algebra:equation}{Equation}: $x - 2y - 7 = 0$ (check $t = 1$,
point $(3, -2)$: $3 + 4 - 7 = 0$).
\end{solution}

\begin{solution}{exo:g11:vect:8}
A parallel line has the same $(a, b)$: $4x - 3y + c = 0$; through
$P(1, -2)$: $4 + 6 + c = 0$, so $c = -10$ and the line is
$4x - 3y - 10 = 0$.

Parallelism of $mx + 2y - 5 = 0$ with $d$ requires
$ab' - a'b = 4 \times 2 - m \times (-3) = 8 + 3m = 0$, so
$m = -\frac83$.
\end{solution}

\begin{solution}{exo:g11:vect:9}
The \omterm{prop:g10:coordgeom:midpoint}{midpoint} of $[AB]$ is $M\!\left(\frac{-1+3}{2}, \frac{0+2}{2}\right)
= (1, 1)$. The \omterm{def:g10:stats:median}{median} is the line $(CM)$ with $C(1, -4)$: both points
have \omterm{def:g10:coordgeom:system}{abscissa} $1$, so the \omterm{def:g10:stats:median}{median} is the vertical line
\[
x - 1 = 0 .
\]
\end{solution}

\begin{solution}{exo:g11:vect:10}
$\det(\vec u, \vec v) = m(m - 1) - 3 \times 2 = m^2 - m - 6$. It vanishes
when $m^2 - m - 6 = 0$: $\Delta = 1 + 24 = 25$, \omterm{def:g11:quad:discriminant}{roots}
$m = \frac{1 \pm 5}{2}$, i.e.\ $m = 3$ or $m = -2$.
\end{solution}

\begin{solution}{exo:g11:vect:11}
\emph{1.} $I = \left(\frac12, 0\right)$. Then
$\vect{DI} = \left(\frac12 - 0,\ 0 - 1\right) = \left(\frac12, -1\right)$
and $\vect{DJ} = \frac23\vect{DI} = \left(\frac13, -\frac23\right)$, so
\[
J = D + \vect{DJ} = \left(\frac13,\ \frac13\right).
\]

\emph{2.} $\vect{AJ}\left(\frac13, \frac13\right)$ and
$\vect{AC}\,(1, 1)$:
$\det = \frac13 \times 1 - 1 \times \frac13 = 0$, so $A$, $J$, $C$ are
aligned --- in fact $\vect{AJ} = \frac13\vect{AC}$: the line $(DI)$ cuts
the diagonal $(AC)$ at exactly one third of its length from $A$.
\end{solution}

\begin{solution}{pb:g11:vect:1}
\textbf{1.} $x = 1 + 3t$, $y = 2 - t$. From the second,
$t = 2 - y$; substituting: $x = 1 + 3(2 - y) = 7 - 3y$:
Cartesian \omterm{def:g10:algebra:equation}{equation} $x + 3y - 7 = 0$.

\textbf{2.} For $ax + by + c = 0$, a \omterm{def:g11:vect:direction}{direction vector} is
$(-b, a)$: here $(5, 2)$. A point: $y = 1$ gives $x = 1$:
$(1, 1)$. Parametric: $x = 1 + 5t$, $y = 1 + 2t$.

\textbf{3.} From $x = 7 - 3y$ in $2x - 5y + 3 = 0$:
$14 - 6y - 5y + 3 = 0$, so $y = \frac{17}{11}$ and
$x = 7 - \frac{51}{11} = \frac{26}{11}$: the point
$\left(\frac{26}{11}, \frac{17}{11}\right)$.

\textbf{4.} Directions $(5, 2)$ and $(3, -1)$:
$\det = 5 \times (-1) - 2 \times 3 = -11 \neq 0$: intersecting,
as found. For the second pair: directions $(5, 2)$ and
$(10, 4)$ have $\det = 0$ (parallel), and doubling
$2x - 5y + 3 = 0$ gives $-4x + 10y - 6 = 0 \neq -4x + 10y + 1
= 0$: strictly parallel.

\textbf{5.} $\vect{AB}(3, 2)$, $\vect{AC}(9, 6)$:
$\det = 18 - 18 = 0$: aligned.

\textbf{6.} $P$: from $x = 2t$, $y = 3 + t$: $y = 3 +
\frac x2$. $Q$: from $x = 10 - t$: $t = 10 - x$, so
$y = 2(10 - x) - 1 = 19 - 2x$. Crossing:
$3 + \frac x2 = 19 - 2x$ gives $x = 6.4$, $y = 6.2$: the paths
cross at $(6.4,\ 6.2)$.

\textbf{7.} $P$ passes there when $2t = 6.4$: $t = 3.2$ h. $Q$
when $10 - t = 6.4$: $t = 3.6$ h. Twenty-four minutes apart: no
collision. Warning: a chart shows \emph{paths}; only the
parametric clocks say whether two motions occupy the same point
at the same instant.

\textbf{8.} $D^2(t) = (3t - 10)^2 + (4 - t)^2 =
10t^2 - 68t + 116$: vertex at $t = \frac{68}{20} = 3.4$ h,
where $D^2 = 0.4$: closest approach
$D = \sqrt{0.4} \approx 0.63$ mile at $t = 3.4$.

\textbf{9.} Violated: $0.63 < 1$. $D^2(t) < 1$ reads
$10t^2 - 68t + 115 < 0$: $\Delta = 4624 - 4600 = 24$, \omterm{def:g11:quad:discriminant}{roots}
$\frac{68 \pm \sqrt{24}}{20} = 3.4 \pm 0.245$: the ships are
too close from $t \approx 3.16$ to $t \approx 3.64$ --- about
$29$ minutes of infraction. One of them must alter course.

\textbf{10.} Each coordinate of each ship is an \emph{affine}
\omterm{def:g11:func:function}{function} of $t$, so each difference of \omterm{def:g10:coordgeom:system}{coordinates} is affine,
and $D^2$ --- a sum of two squared \omterm{def:g10:reffunc:affine}{affine functions} --- is a
quadratic (with nonnegative leading coefficient). Closest
approach is therefore always read off a vertex.

\textbf{11.} Meeting $P$ at $t = 3$, i.e.\ at $(6, 6)$:
$(3a, 3b) = (6, 6)$ gives $(a, b) = (2, 2)$, speed
$\sqrt{8} \approx 2.8$ knots.

\textbf{12.} $D(0, 1)$, $I\left(\frac12, 0\right)$: direction
$\left(\frac12, -1\right)$, i.e.\ $(1, -2)$: \omterm{def:g10:algebra:equation}{equation}
$2x + y = 1$.

\textbf{13.} $(AC)$: $y = x$. Then $2x + x = 1$:
$x = \frac13$: the point $\left(\frac13, \frac13\right)$, one
third of the way from $A$ to $C$.

\textbf{14.} $B(1, 0)$, $K\left(\frac12, 1\right)$: direction
$\left(-\frac12, 1\right)$, \omterm{def:g10:algebra:equation}{equation} $2x + y = 2$. With
$y = x$: $x = \frac23$: the point
$\left(\frac23, \frac23\right)$. The two cevians cut $(AC)$ at
its two trisection points: theorem proved, twice three lines
of arithmetic.

\textbf{15.} Any parallelogram is carried onto the unit square
by choosing $A$ as origin and $\vect{AB}$, $\vect{AD}$ as the
two axes' units. That choice preserves everything the theorem
speaks of --- \omterm{prop:g10:coordgeom:midpoint}{midpoints}, alignment, and ratios along a given
line (all expressible with \omterm{def:g11:vect:vector}{vectors} and scalars) --- so proving
the statement in the square proves it for every parallelogram.
(It would \emph{not} transport lengths or angles: those the
frame distorts.)

\textbf{16.} $E\left(1, \frac12\right)$: line $(AE)$:
$y = \frac x2$. Diagonal $(DB)$: from $(0,1)$ to $(1,0)$:
$y = 1 - x$. \omterm{def:g10:numbers:interunion}{Intersection}: $\frac x2 = 1 - x$: $x = \frac23$:
the point $\left(\frac23, \frac13\right)$ --- again a
trisection point, now of the other diagonal. \omterm{prop:g10:coordgeom:midpoint}{Midpoint} cevians
love thirds.

\textbf{17.} First two: $x + y = 4$ and $2x - y = 2$ give
$3x = 6$: $(2, 2)$. Third: $2 - 4 = -2$: satisfied. Concurrent
at $(2, 2)$.

\textbf{18.} $mx - y + 2 - 3m = m(x - 3) + (2 - y) = 0$: for
the \omterm{def:g10:algebra:equation}{equation} to hold for \emph{every} $m$, take $x = 3$,
$y = 2$ --- and indeed every $d_m$ contains $(3, 2)$. A pencil
of lines through one point, one line per \omterm{def:g10:reffunc:affine}{slope} $m$.

\textbf{19.} Parallel to $y = 2x + 7$: \omterm{def:g10:reffunc:affine}{slope} $m = 2$, the line
$2x - y - 4 = 0$. Through the origin: $2 - 3m = 0$:
$m = \frac23$.

\textbf{20.} Cartesian: plug in a point, membership answered.
Reduced: \omterm{def:g10:reffunc:affine}{slope} and intercept visible, \omterm{def:g10:functions:graph}{graph} immediate.
Parametric: the clock built in --- collisions, closest
approach, interception. The \omterm{def:g11:vect:det}{determinant}: one multiplication
cross, and parallelism is decided. The chosen frame: a
parallelogram becomes a unit square, and a classical trisection
theorem becomes three \omterm{def:g10:numbers:interunion}{intersections} of lines.
\end{solution}
