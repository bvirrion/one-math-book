\begin{solution}{exo:g11:vect:1}
$\vect{AB}\,(3, 4)$, $\vect{AC}\,(-3, 2)$,
$\vect{AB} + \vect{AC}\,(0, 6)$,
$2\vect{AB} - 3\vect{AC}\,(6 + 9,\ 8 - 6) = (15, 2)$. The \omterm{prop:g10:coordgeom:midpoint}{midpoint} of

\end{solution}

\begin{solution}{exo:g11:vect:2}
$\det = 4 \times 3 - (-2) \times (-6) = 12 - 12 = 0$: \omterm{def:g11:vect:collinear}{collinear} (indeed
$\vec v = -\frac12 \vec u$).

$\det = 2 \times 7 - 3 \times 5 = -1 \neq 0$: not \omterm{def:g11:vect:collinear}{collinear}.

$\det = 0 \times (-8) - 0 \times 3 = 0$: \omterm{def:g11:vect:collinear}{collinear} (both vertical).
\end{solution}

\begin{solution}{exo:g11:vect:3}
Through $A(2,1)$, direction $(3, -1)$: matching $(-b, a) = (3, -1)$ gives
$a = -1$, $b = -3$, so $-x - 3y + c = 0$; plugging in $A$: $-2 - 3 + c =
0$, $c = 5$. Multiplying by $-1$: $x + 3y - 5 = 0$.

Through $B(0,4)$ and $C(2,0)$: $\vect{BC}\,(2, -4)$, so $a = -4$,
$b = -2$: $-4x - 2y + c = 0$; plugging in $B$: $-8 + c = 0$, $c = 8$.
Simplifying by $-2$: $2x + y - 4 = 0$. Check with $C$: $4 + 0 - 4 = 0$.
\end{solution}

\begin{solution}{exo:g11:vect:4}
A \omterm{def:g11:vect:direction}{direction vector} is $(-b, a) = (2, 3)$. \omterm{def:g10:numbers:interunion}{Intersections}: $x = 0$ gives
$y = 3$, point $(0, 3)$; $y = 0$ gives $x = -2$, point $(-2, 0)$. For
$P(2,6)$: $3 \times 2 - 2 \times 6 + 6 = 0$, so $P \in d$. For $Q(1,4)$:
$3 - 8 + 6 = 1 \neq 0$, so $Q \notin d$.
\end{solution}

\begin{solution}{exo:g11:vect:5}
$\vect{AB}\,(3, 6)$ and $\vect{AC}\,(-3, -6)$:
$\det = 3 \times (-6) - (-3) \times 6 = 0$, so $A$, $B$, $C$ are aligned.

$\vect{DE}\,(2, 3)$ and $\vect{DF}\,(5, 8)$:
$\det = 2 \times 8 - 5 \times 3 = 1 \neq 0$: $D$, $E$, $F$ are not
aligned.
\end{solution}

\begin{solution}{exo:g11:vect:6}
First pair: adding the two \omterm{def:g10:algebra:equation}{equations}, $(2x + y - 7) + (x - y + 1) =
3x - 6 = 0$, so $x = 2$, then $y = x + 1 = 3$: they meet at $(2, 3)$
(check: $4 + 3 - 7 = 0$).

Second pair: $ab' - a'b = 1 \times 4 - (-2) \times (-2) = 0$, so the
lines are parallel; multiplying $e$ by $-2$ gives $-2x + 4y - 6 = 0$,
which differs from $e'$ ($-6 \neq 1$): strictly parallel, no
\omterm{def:g10:numbers:interunion}{intersection}.
\end{solution}

\begin{solution}{exo:g11:vect:7}
For $d: 2x - y + 3 = 0$: direction $(-b, a) = (1, 2)$ and the point
$(0, 3)$ lies on $d$, so $d = \{(t,\ 3 + 2t) : t \in \R\}$.

For the parametric line through $(1, -3)$ with direction $(2, 1)$:
$a = 1$, $b = -2$, so $x - 2y + c = 0$; plugging in $(1, -3)$:
$1 + 6 + c = 0$, $c = -7$. \omterm{def:g10:algebra:equation}{Equation}: $x - 2y - 7 = 0$ (check $t = 1$,
point $(3, -2)$: $3 + 4 - 7 = 0$).
\end{solution}

\begin{solution}{exo:g11:vect:8}
A parallel line has the same $(a, b)$: $4x - 3y + c = 0$; through
$P(1, -2)$: $4 + 6 + c = 0$, so $c = -10$ and the line is
$4x - 3y - 10 = 0$.

Parallelism of $mx + 2y - 5 = 0$ with $d$ requires
$ab' - a'b = 4 \times 2 - m \times (-3) = 8 + 3m = 0$, so
$m = -\frac83$.
\end{solution}

\begin{solution}{exo:g11:vect:9}
The \omterm{prop:g10:coordgeom:midpoint}{midpoint} of $[AB]$ is $M\!\left(\frac{-1+3}{2}, \frac{0+2}{2}\right)
= (1, 1)$. The \omterm{def:g10:stats:median}{median} is the line $(CM)$ with $C(1, -4)$: both points
have \omterm{def:g10:coordgeom:system}{abscissa} $1$, so the \omterm{def:g10:stats:median}{median} is the vertical line
\[
x - 1 = 0 .
\]
\end{solution}

\begin{solution}{exo:g11:vect:10}
$\det(\vec u, \vec v) = m(m - 1) - 3 \times 2 = m^2 - m - 6$. It vanishes
when $m^2 - m - 6 = 0$: $\Delta = 1 + 24 = 25$, \omterm{def:g11:quad:discriminant}{roots}
$m = \frac{1 \pm 5}{2}$, i.e.\ $m = 3$ or $m = -2$.
\end{solution}

\begin{solution}{exo:g11:vect:11}
\emph{1.} $I = \left(\frac12, 0\right)$. Then
$\vect{DI} = \left(\frac12 - 0,\ 0 - 1\right) = \left(\frac12, -1\right)$
and $\vect{DJ} = \frac23\vect{DI} = \left(\frac13, -\frac23\right)$, so
\[
J = D + \vect{DJ} = \left(\frac13,\ \frac13\right).
\]

\emph{2.} $\vect{AJ}\left(\frac13, \frac13\right)$ and
$\vect{AC}\,(1, 1)$:
$\det = \frac13 \times 1 - 1 \times \frac13 = 0$, so $A$, $J$, $C$ are
aligned --- in fact $\vect{AJ} = \frac13\vect{AC}$: the line $(DI)$ cuts
the diagonal $(AC)$ at exactly one third of its length from $A$.
\end{solution}
