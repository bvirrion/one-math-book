\begin{solution}{exo:g11:prob:1}
Uniform model on $52$ cards. Hearts: $\frac{13}{52} = \frac14$. Kings:
$\frac{4}{52} = \frac{1}{13}$. Heart or king:
$\frac{13 + 4 - 1}{52} = \frac{16}{52} = \frac{4}{13}$ (the king of
hearts is counted once). Neither: $1 - \frac{4}{13} = \frac{9}{13}$.
\end{solution}

\begin{solution}{exo:g11:prob:2}
$S = 6$: the pairs $(1,5), (2,4), (3,3), (4,2), (5,1)$, so
$\P(S = 6) = \frac{5}{36}$.

$S \geq 10$: three pairs for $10$, two for $11$, one for $12$:
$\P(S \geq 10) = \frac{6}{36} = \frac16$.

$S$ even: sums $2, 4, 6, 8, 10, 12$ have $1 + 3 + 5 + 5 + 3 + 1 = 18$
pairs: probability $\frac{18}{36} = \frac12$.
\end{solution}

\begin{solution}{exo:g11:prob:3}
The probabilities sum to $1$: $p = 1 - 0.3 - 0.2 - 0.4 = 0.1$. Then
\[
\E(X) = 0.3 \times (-1) + 0.2 \times 0 + 0.4 \times 2 + 0.1 \times 5
= -0.3 + 0.8 + 0.5 = 1 .
\]
\end{solution}

\begin{solution}{exo:g11:prob:4}
$\E(Y) = 2 \times 3 - 5 = 1$; $\V(Y) = 2^2 \times 4 = 16$;
$\sigma(Y) = 4$.
\end{solution}

\begin{solution}{exo:g11:prob:5}
The winnings are $0$, $2$, $5$ euros with probabilities $0.5$, $0.3$,
$0.2$; subtracting the $2$-euro stake, $G$ takes the values $-2, 0, 3$:
\begin{center}
\begin{tabular}{c|ccc}
$g$ & $-2$ & $0$ & $3$\\
\hline
$\P(G=g)$ & $0.5$ & $0.3$ & $0.2$
\end{tabular}
\end{center}
$\E(G) = -1 + 0 + 0.6 = -0.4$: the player loses $40$ cents per game on
average --- unfavorable.
\end{solution}

\begin{solution}{exo:g11:prob:6}
Drawing in order: $\P(X = 2) = \frac35 \times \frac24 = \frac{3}{10}$
(red then red); $\P(X = 0) = \frac25 \times \frac14 = \frac{1}{10}$
(blue then blue); hence
$\P(X = 1) = 1 - \frac{3}{10} - \frac{1}{10} = \frac{6}{10}$. Then
\[
\E(X) = 0 \times \frac1{10} + 1 \times \frac6{10} + 2 \times \frac3{10}
= \frac{12}{10} = 1.2 ,
\]
consistent with intuition: $\frac35$ of the two draws are red on average.
\end{solution}

\begin{solution}{exo:g11:prob:7}
$\E(G^2) = 0.5 \times 4 + 0.3 \times 0 + 0.2 \times 9 = 3.8$, so by the
shortcut formula
$\V(G) = 3.8 - (-0.4)^2 = 3.8 - 0.16 = 3.64$ and
$\sigma(G) \approx 1.9$ euros.
\end{solution}

\begin{solution}{exo:g11:prob:8}
The benefit is $80$ minus the claim:
\begin{center}
\begin{tabular}{c|ccc}
$b$ & $80$ & $-920$ & $-4920$\\
\hline
$\P(B=b)$ & $0.94$ & $0.05$ & $0.01$
\end{tabular}
\end{center}
$\E(B) = 75.2 - 46 - 49.2 = -20$: at this price the company \emph{loses}
$20$ euros per policy on average. The premium is too low for the risks
covered (it would need to exceed $100$ euros for a positive expected
benefit).
\end{solution}

\begin{solution}{exo:g11:prob:9}
The random score is $+1$ with probability $\frac14$ and $x$ with
probability $\frac34$:
\[
\E = \frac14 + \frac{3x}{4} = 0 \iff x = -\frac13 .
\]
With the penalty $-\frac13$, random guessing brings nothing on average.
\end{solution}

\begin{solution}{exo:g11:prob:10}
Two coins: they match (HH or TT) with probability $\frac24 \times 2 =
\frac12$. The player's gain is $+1$ or $-1$ with equal probabilities:
expectation $0$ for both players --- the game is fair.

Three coins: all match (HHH or TTT) with probability $\frac{2}{8} =
\frac14$. The player receives $2$ with probability $\frac14$ and pays $1$
with probability $\frac34$:
$\E = \frac24 - \frac34 = -\frac14$. The game favours Ann by $25$ cents
per round; it would be fair if she paid $3$ euros for a triple match.
\end{solution}

\begin{solution}{exo:g11:prob:11}
\emph{1.} The organizer receives $1000 \times 5 = 5000$ euros and pays
out $2000 + 5 \times 200 + 20 \times 25 = 3500$ euros in all: the profit
is the \emph{constant} $1500$ (no randomness once all tickets are sold).
A single player's gain $G$ (prize minus the $5$-euro ticket):
\begin{center}
\begin{tabular}{c|cccc}
$g$ & $1995$ & $195$ & $20$ & $-5$\\
\hline\rule{0pt}{11pt}%
$\P(G=g)$ & $\dfrac{1}{1000}$ & $\dfrac{5}{1000}$ & $\dfrac{20}{1000}$
& $\dfrac{974}{1000}$
\end{tabular}
\end{center}

\emph{2.}
\[
\E(G) = \frac{1995 + 5 \times 195 + 20 \times 20 - 974 \times 5}{1000}
= \frac{1995 + 975 + 400 - 4870}{1000} = -1.5 .
\]
Each player expects to lose $1.50$ euros; over $1000$ players that is
$1500$ euros --- exactly the organizer's profit. The books balance:
what the players lose on average is precisely what the organizer
collects.
\end{solution}
