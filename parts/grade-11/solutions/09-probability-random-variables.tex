\section*{Chapter \ref{ch:g11:prob} --- Probability and Random Variables}

\begin{solution}{exo:g11:prob:1}
\omterm{def:g11:prob:model}{Uniform model} on $52$ cards. Hearts: $\frac{13}{52} = \frac14$. Kings:
$\frac{4}{52} = \frac{1}{13}$. Heart or king:
$\frac{13 + 4 - 1}{52} = \frac{16}{52} = \frac{4}{13}$ (the king of
hearts is counted once). Neither: $1 - \frac{4}{13} = \frac{9}{13}$.
\end{solution}

\begin{solution}{exo:g11:prob:2}
$S = 6$: the pairs $(1,5), (2,4), (3,3), (4,2), (5,1)$, so
$\P(S = 6) = \frac{5}{36}$.

$S \geq 10$: three pairs for $10$, two for $11$, one for $12$:
$\P(S \geq 10) = \frac{6}{36} = \frac16$.

$S$ even: sums $2, 4, 6, 8, 10, 12$ have $1 + 3 + 5 + 5 + 3 + 1 = 18$
pairs: \omterm{def:g10:proba:distribution}{probability} $\frac{18}{36} = \frac12$.
\end{solution}

\begin{solution}{exo:g11:prob:3}
The probabilities sum to $1$: $p = 1 - 0.3 - 0.2 - 0.4 = 0.1$. Then
\[
\E(X) = 0.3 \times (-1) + 0.2 \times 0 + 0.4 \times 2 + 0.1 \times 5
= -0.3 + 0.8 + 0.5 = 1 .
\]
\end{solution}

\begin{solution}{exo:g11:prob:4}
$\E(Y) = 2 \times 3 - 5 = 1$; $\V(Y) = 2^2 \times 4 = 16$;
$\sigma(Y) = 4$.
\end{solution}

\begin{solution}{exo:g11:prob:5}
The winnings are $0$, $2$, $5$ euros with probabilities $0.5$, $0.3$,
$0.2$; subtracting the $2$-euro stake, $G$ takes the values $-2, 0, 3$:
\begin{center}
\begin{tabular}{c|ccc}
$g$ & $-2$ & $0$ & $3$\\
\hline
$\P(G=g)$ & $0.5$ & $0.3$ & $0.2$
\end{tabular}
\end{center}
$\E(G) = -1 + 0 + 0.6 = -0.4$: the player loses $40$ cents per game on
average --- unfavorable.
\end{solution}

\begin{solution}{exo:g11:prob:6}
Drawing in order: $\P(X = 2) = \frac35 \times \frac24 = \frac{3}{10}$
(red then red); $\P(X = 0) = \frac25 \times \frac14 = \frac{1}{10}$
(blue then blue); hence
$\P(X = 1) = 1 - \frac{3}{10} - \frac{1}{10} = \frac{6}{10}$. Then
\[
\E(X) = 0 \times \frac1{10} + 1 \times \frac6{10} + 2 \times \frac3{10}
= \frac{12}{10} = 1.2 ,
\]
consistent with intuition: $\frac35$ of the two draws are red on average.
\end{solution}

\begin{solution}{exo:g11:prob:7}
$\E(G^2) = 0.5 \times 4 + 0.3 \times 0 + 0.2 \times 9 = 3.8$, so by the
shortcut formula
$\V(G) = 3.8 - (-0.4)^2 = 3.8 - 0.16 = 3.64$ and
$\sigma(G) \approx 1.9$ euros.
\end{solution}

\begin{solution}{exo:g11:prob:8}
The benefit is $80$ minus the claim:
\begin{center}
\begin{tabular}{c|ccc}
$b$ & $80$ & $-920$ & $-4920$\\
\hline
$\P(B=b)$ & $0.94$ & $0.05$ & $0.01$
\end{tabular}
\end{center}
$\E(B) = 75.2 - 46 - 49.2 = -20$: at this price the company \emph{loses}
$20$ euros per policy on average. The premium is too low for the risks
covered (it would need to exceed $100$ euros for a positive expected
benefit).
\end{solution}

\begin{solution}{exo:g11:prob:9}
The random score is $+1$ with \omterm{def:g10:proba:distribution}{probability} $\frac14$ and $x$ with
\omterm{def:g10:proba:distribution}{probability} $\frac34$:
\[
\E = \frac14 + \frac{3x}{4} = 0 \iff x = -\frac13 .
\]
With the penalty $-\frac13$, random guessing brings nothing on average.
\end{solution}

\begin{solution}{exo:g11:prob:10}
Two coins: they match (HH or TT) with \omterm{def:g10:proba:distribution}{probability} $\frac24 \times 2 =
\frac12$. The player's gain is $+1$ or $-1$ with equal probabilities:
\omterm{def:g11:prob:expectation}{expectation} $0$ for both players --- the game is fair.

Three coins: all match (HHH or TTT) with \omterm{def:g10:proba:distribution}{probability} $\frac{2}{8} =
\frac14$. The player receives $2$ with \omterm{def:g10:proba:distribution}{probability} $\frac14$ and pays $1$
with \omterm{def:g10:proba:distribution}{probability} $\frac34$:
$\E = \frac24 - \frac34 = -\frac14$. The game favours Ann by $25$ cents
per round; it would be fair if she paid $3$ euros for a triple match.
\end{solution}

\begin{solution}{exo:g11:prob:11}
\emph{1.} The organizer receives $1000 \times 5 = 5000$ euros and pays
out $2000 + 5 \times 200 + 20 \times 25 = 3500$ euros in all: the profit
is the \emph{constant} $1500$ (no randomness once all tickets are sold).
A single player's gain $G$ (prize minus the $5$-euro ticket):
\begin{center}
\begin{tabular}{c|cccc}
$g$ & $1995$ & $195$ & $20$ & $-5$\\
\hline\rule{0pt}{11pt}%
$\P(G=g)$ & $\dfrac{1}{1000}$ & $\dfrac{5}{1000}$ & $\dfrac{20}{1000}$
& $\dfrac{974}{1000}$
\end{tabular}
\end{center}

\emph{2.}
\[
\E(G) = \frac{1995 + 5 \times 195 + 20 \times 20 - 974 \times 5}{1000}
= \frac{1995 + 975 + 400 - 4870}{1000} = -1.5 .
\]
Each player expects to lose $1.50$ euros; over $1000$ players that is
$1500$ euros --- exactly the organizer's profit. The books balance:
what the players lose on average is precisely what the organizer
collects.
\end{solution}

\begin{solution}{pb:g11:prob:1}
\textbf{1.} $G$ takes $-2, -1, 0, 1, 2, 3$, each with
\omterm{def:g10:proba:distribution}{probability} $\frac16$: $\E(G) = \frac{-2 - 1 + 0 + 1 + 2 +
3}{6} = \frac12$ euro.

\textbf{2.} $\E(G^2) = \frac{4 + 1 + 0 + 1 + 4 + 9}{6} =
\frac{19}{6}$; $V(G) = \frac{19}{6} - \frac14 = \frac{35}{12}
\approx 2.92$; $\sigma(G) \approx 1.71$.

\textbf{3.} $\E(2G - 1) = 2 \times \frac12 - 1 = 0$;
$V(2G - 1) = 4\,V(G) = \frac{35}{3}$.

\textbf{4.} $\E(\text{sum}) = 7$: fair price $7$ euros. At $8$,
the expected loss is $1$ euro per game.

\textbf{5.} Fair premium: $0.05 \times 800 = 40$ euros.
Charging $60$: expected profit $20$ euros per policy.

\textbf{6.} Splitting $5:3$ rewards the \emph{past} --- but the
pot belongs to whoever wins the \emph{future}, and B might
still win. Giving all to A treats a probable win as certain ---
B's small chance is worth something.

\textbf{7.} Rounds to play: $3$; outcomes (A-win/B-win per
round): AAA, AAB, ABA, ABB, BAA, BAB, BBA, BBB. Player B takes
the series only by winning all three: BBB alone. A wins the
series in $7$ of $8$ cases: fair split
$64 \times \frac78 = 56$ pistoles for A, $8$ for B.

\textbf{8.} Playing the extra, pointless rounds changes
nobody's victory: once A has her sixth win, later rounds are
ceremony. But fixing the horizon at exactly $3$ rounds makes
all $8$ outcomes equally likely --- honest counting needs
same-length stories, and padding them out costs nothing.

\textbf{9.} At $5$--$5$: one deciding round, $32$ each. At
$5$--$4$: A gets $\frac{64 + 32}{2} = 48$. At $5$--$3$: the
next round leads to $64$ (A wins it) or to the $5$--$4$
position worth $48$: A's share $\frac{64 + 48}{2} = 56$ ---
Fermat's $56 : 8$, re-derived by walking backwards.

\textbf{10.} A needs $2$, B needs $3$: play $4$ imaginary
rounds ($16$ outcomes). B takes the series with $3$ or $4$
B-wins: $4 + 1 = 5$ outcomes. Split:
$B$: $64 \times \frac{5}{16} = 20$; $A$: $44$.

\textbf{11.} The fair present value of an uncertain future
payoff is its \omterm{def:g11:prob:expectation}{expectation} --- probability-weighted average of
what may come. Insurance (question 5) and finance (pricing
uncertain payoffs) are this sentence turned into industries;
so are casinos, from the other side of the table.

\textbf{12.} The pot always ends up somewhere: the two shares
are \omterm{def:g11:prob:rv}{random variables} summing to the constant $64$, so by
linearity $\E_A + \E_B = 64$ --- fair shares exhaust the pot,
at every score.

\textbf{13.} Linearity: $\E(X + Y) = 3.5 + 3.5 = 7$, table-free
--- true even for glued-together dice. \omterm{def:g11:prob:variance}{Variance} of one die:
$\E(X^2) - 3.5^2 = \frac{91}{6} - \frac{49}{4} = \frac{35}{12}$;
for independent dice
$V(X + Y) = \frac{35}{12} + \frac{35}{12} = \frac{35}{6}$.

\textbf{14.} Each fixed guest recovers their own hat with
\omterm{def:g10:proba:distribution}{probability} $\frac1n$ (their hat is one among $n$). Summing
over the $n$ guests:
$\E(X) = n \times \frac1n = 1$ --- one lucky guest on average,
whether the party has two hats or two thousand.

\textbf{15.} $n = 2$: assignments: both right ($X = 2$) or
swapped ($X = 0$): $\E = 1$. $n = 3$: the six assignments give
$X = 3, 1, 1, 1, 0, 0$: $\E = \frac{6}{6} = 1$.

\textbf{16.} The computation added the \omterm{def:g11:prob:expectation}{expectations} of the $n$
individual indicators (``guest $i$ got their hat''), and
linearity adds \omterm{def:g11:prob:expectation}{expectations} \emph{unconditionally} ---
dependence would wreck a product or a \omterm{def:g11:prob:variance}{variance}, but never a
sum of \omterm{def:g11:prob:expectation}{expectations}. That is linearity's superpower: no
independence questions asked.

\textbf{17.} Both gains: $\E = 0.5 \times 10 - 5 = 0$ and
$0.005 \times 1000 - 5 = 0$. \omterm{def:g11:prob:variance}{Variances} (of the gain):
A: $\E(G^2) = 0.5 \times 25 + 0.5 \times 25 = 25$;
B: $0.005 \times 995^2 + 0.995 \times 25 \approx 4\,975$:
$\sigma_A = 5$, $\sigma_B \approx 70.5$. People buy B: what
they purchase is not \omterm{def:g11:prob:expectation}{expectation} (zero in both) but the
\emph{spread} --- a small chance of a life-changing right
tail.

\textbf{18.} First tails at flip $n$ has \omterm{def:g10:proba:distribution}{probability}
$2^{-n}$ and pays $2^n$: each $n$ contributes
$2^{-n} \times 2^n = 1$ to the \omterm{def:g11:prob:expectation}{expectation}, which is therefore
$1 + 1 + 1 + \dots$: infinite. Yet no sane person pays
$1\,000$ to play once: with \omterm{def:g10:proba:distribution}{probability} $\frac{15}{16}$ the
payout is at most $16$. \omterm{def:g11:prob:expectation}{Expectation} summarizes the long run of
\emph{repeatable} bets; a once-in-a-lifetime game with a wild
tail is not priced by its \omterm{def:g11:stat:mean}{mean}.

\textbf{19.} \omterm{def:g11:prob:expectation}{Expectations}: insure $-20$; go bare
$-0.01 \times 1000 = -10$. The naked strategy wins on average
--- for someone who can absorb the loss and repeat the bet
many times. A family that cannot survive the $-1\,000$ is not
in the averaging business: it pays $10$ extra to delete the
\omterm{def:g11:prob:variance}{variance}. Insurers average; households buy certainty.

\textbf{20.} Born splitting a pot between two impatient
gamblers; empowered by linearity, which sums hopes without
asking permission of dependence; blind to risk, which \omterm{def:g11:prob:variance}{variance}
--- the lotteries, the Petersburg game, the family's policy
--- must measure instead; and crowned next year by the law of
large numbers, which explains why the averages always,
eventually, collect.
\end{solution}
