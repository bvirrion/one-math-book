\section*{Capítulo \ref{ch:g11:binom} --- A distribuição binomial}

\begin{solution}{exo:g11:binom:1}
Linhas $6$ e $7$:
\[
1,\ 6,\ 15,\ 20,\ 15,\ 6,\ 1
\qquad\text{and}\qquad
1,\ 7,\ 21,\ 35,\ 35,\ 21,\ 7,\ 1 .
\]
Por isso $\binom62 = 15$, $\binom73 = 35$, $\binom74 = 35$ (a simetria
$\binom73 = \binom74$ reflete sucessos e fracassos de troca).
\end{solution}

\begin{solution}{exo:g11:binom:2}
Número fixo de tentativas ($4$ lançamentos), dois resultados por lançamento (seis ou não,
$ p = \frac16$), rolos independentes: $ X \sim \mathcal B(4, \frac16)$.
\[
\P(X=0) = \left(\frac56\right)^4 = \frac{625}{1296} \approx 0.48,
\qquad
\P(X=1) = 4 \times \frac16\left(\frac56\right)^3 = \frac{500}{1296}
\approx 0.39,
\]
\[
\P(X \geq 2) = 1 - \frac{625 + 500}{1296} = \frac{171}{1296}
\approx 0.13 .
\]
\end{solution}

\begin{solution}{exo:g11:binom:3}
\emph{1.} Binômio $\mathcal B(20, \frac12)$: fixo $ n $, mesmo $ p $,
lançamentos independentes.

\emph{2.} Não binomial: as cartas são distribuídas \emph{sem substituição},
então o \omterm{def:g10:proba:distribution}{probabilidade} de um ás muda de carta para carta e os draws são
não independente.

\emph{3.} Binômio $\mathcal B(7, 0.3)$ pela independência declarada
suposição.
\end{solution}

\begin{solution}{exo:g11:binom:4}
$\E(X) = 50 \times 0.2 = 10$;
$\V(X) = 50 \times 0.2 \times 0.8 = 8$;
$\sigma(X) = 2\sqrt2 \approx 2.83$.
\end{solution}

\begin{solution}{exo:g11:binom:5}
$ X \sim \mathcal B(6, 0.7)$.
\[
\P(X = 4) = \binom64 (0.7)^4 (0.3)^2 = 15 \times 0.2401 \times 0.09
\approx 0.324 .
\]
\[
\P(X \geq 5) = \binom65 (0.7)^5(0.3) + (0.7)^6
= 6 \times 0.16807 \times 0.3 + 0.117649 \approx 0.420 .
\]
\end{solution}

\begin{solution}{exo:g11:binom:6}
$ X \sim \mathcal B\left(8, \frac12\right)$; todo caminho tem \omterm{def:g10:proba:distribution}{probabilidade}
$\frac{1}{256}$, então
\[
\P(X \geq 6) = \frac{\binom86 + \binom87 + \binom88}{256}
= \frac{28 + 8 + 1}{256} = \frac{37}{256} \approx 0.14 .
\]
A adivinhação ocorre cerca de uma vez em sete.
\end{solution}

\begin{solution}{exo:g11:binom:7}
O \omterm{def:g10:proba:operations}{complemento} de ``pelo menos um de cada'' é ``todos os cinco idênticos'':
todos A ou todos B, cada um com \omterm{def:g10:proba:distribution}{probabilidade} $\left(\frac12\right)^5 =
\frac1{32}$. Por isso
\[
\P(\text{one of each kind}) = 1 - \frac{2}{32} = \frac{15}{16} .
\]
\end{solution}

\begin{solution}{exo:g11:binom:8}
$\P(X = 3) = p^3$ e $\P(X \geq 1) = 1 - (1-p)^3$. Resolvendo
$ p^3 = \frac{27}{64} = \left(\frac34\right)^3$ dá $ p = \frac34$ (o
cubo \omterm{def:g11:func:function}{função} é estritamente \omterm{def:g11:func:monotone}{aumentando}, \cref{ch:g11:func}, então a solução
é único).
\end{solution}

\begin{solution}{exo:g11:binom:9}
$\P(\text{at least one head}) = 1 - \left(\frac12\right)^n $, então o
condição é $\left(\frac12\right)^n < 0.01$, ou seja\ $2^n > 100$. Desde
$2^6 = 64$ e $2^7 = 128$: de $ n = 7$ joga.
\end{solution}

\begin{solution}{exo:g11:binom:10}
A soma da linha $ n $ conta todos os caminhos do $ n $-julgamento \omterm{met:g10:proba:tree}{árvore}, classificado por
seu número de sucessos. Mas o \omterm{met:g10:proba:tree}{árvore} dobra seus caminhos a cada tentativa
(cada caminho se divide em S e F), então tem $2^n $ caminhos no total. Por isso
$\sum_{k} \binom nk = 2^n $. Alternativamente, por indução: linha $0$ somas para
$1 = 2^0$, e a regra de Pascal faz com que cada entrada da linha $ n $ contribuir para
exatamente duas entradas de linha $ n+1$, então a linha soma o dobro.
\end{solution}

\begin{solution}{exo:g11:binom:11}
\emph{1.} Sob a reivindicação, $ X \sim \mathcal B(10, 0.6)$. Resumindo o
primeiros termos:
\begin{align*}
\P(X \leq 3)
&= (0.4)^{10} + 10(0.6)(0.4)^9 + 45(0.6)^2(0.4)^8
+ 120(0.6)^3(0.4)^7\\
&\approx 0.0001 + 0.0016 + 0.0106 + 0.0425 = 0.0548 .
\end{align*}

\emph{2.} Um resultado pelo menos tão extremo quanto o observado ($3$ ou menos
aprovações) tem \omterm{def:g10:proba:distribution}{probabilidade} sobre $5.5\%$ --- apenas \emph{acima} o $5\%$
limite. Aplicando estritamente a regra, a observação é (mal)
compatível com a reivindicação e não se pode rejeitá-la. O exemplo mostra
quão sensíveis são as decisões limítrofes à escolha do limite: com um
$6\%$ convenção, a conclusão mudaria.
\end{solution}

\begin{solution}{pb:g11:binom:1}
\textbf{1.} Linhas: $1$; $1\,1$; $1\,2\,1$; $1\,3\,3\,1$;
$1\,4\,6\,4\,1$; $1\,5\,10\,10\,5\,1$;
$1\,6\,15\,20\,15\,6\,1$. Simetria: escolhendo qual $ k $ objetos
tomar é o mesmo ato que escolher qual $ n - k $ sair
atrás.

\textbf{2.} $1 + 4 + 6 + 4 + 1 = 16 = 2^4$;
$1 + 5 + 10 + 10 + 5 + 1 = 32 = 2^5$. Prova: o
$\binom nk $ contar os subconjuntos de cada tamanho de um $ n $-elemento
conjunto, e os subconjuntos juntos numeram $2^n $ (cada elemento em
ou fora, independentemente).

\textbf{3.} Comitês de $ k $ escolhido entre $ n + 1$ pessoas, um
de quem é Zoe: aqueles sem Zoe são $\binom nk $ (escolha todos
$ k $ entre outros); aqueles com Zoe são $\binom{n}{k-1}$
(escolha ela $ k - 1$ colegas). Total:
$\binom nk + \binom{n}{k-1}$.

\textbf{4.} Linha triangular $7$: $1\,7\,21\,35\,\dots $: $35$.
Fórmula: $\frac{7 \times 6 \times 5}{3 \times 2 \times 1} =
35$.

\textbf{5.} $1 + 3 + 6 + 10 = 20 = \binom63$. Cascata:
$\binom63 = \binom52 + \binom53 = \binom52 + \binom42 +
\binom43 = \binom52 + \binom42 + \binom32 + \binom33$ ---
cada aplicação da regra de Pascal elimina um termo do
escada.

\textbf{6.} Número fixo $ n $ de saltos; cada um salta um
independente \omterm{def:g11:binom:bernoulli}{Julgamento de Bernoulli} com o mesmo $ p = \frac12$; o
o número do slot conta os sucessos (rejeições para a direita): todos os três
caixas de \cref{met:g11:binom:recognize} marcado:
$\mathcal B\!\left(n, \frac12\right)$.

\textbf{7.} \omterm{def:g10:proba:distribution}{Probabilidades} $\frac{1}{16}, \frac{4}{16},
\frac{6}{16}, \frac{4}{16}, \frac{1}{16}$ para slots
$0, \dots, 4$: o slot central $2$ é o mais lotado.

\textbf{8.} Contagens esperadas $= 1024 \times
\binom{10}{k}/1024 = \binom{10}{k}$: slot do meio
$\binom{10}{5} = 252$ bolas; slot $7$: $\binom{10}{7} = 120$;
cada slot de borda: $1$ bola. Um centro alto caindo
simetricamente às bordas finas: o sino.

\textbf{9.} $\E = np = 5$; $ V = np(1 - p) = 2.5$;
$\sigma \approx 1.58$. Dentro de $2\sigma $: slots $2$ para $8$
carregar
\[
\frac{45 + 120 + 210 + 252 + 210 + 120 + 45}{1024}
= \frac{1002}{1024} \approx 98\,\%
\]
das bolas --- muito melhor
do que o multifuncional de Chebyshev $75\,\%$
(\cref{pb:g11:stat:1}): formas de sino concentram-se fortemente.

\textbf{10.} $\E = 6$, $ V = 10 \times 0.6 \times 0.4 = 2.4$,
$\sigma \approx 1.55$: a pilha mantém a forma de sino, mas
desliza seu cume para encaixar $6$ --- uma placa inclinada é carregada
moeda tornada visível.

\textbf{11.} O número do slot é um \emph{soma} de muitos pequenos,
impulsos aleatórios independentes e do mesmo tamanho --- e essas somas sempre
organizam-se na campainha: a maioria dos empurrões cancelam, os extremos
exigem unanimidade. Alturas, erros de medição e inúmeras
quantidades naturais são também somas de muitas pequenas quantidades independentes
efeitos, razão pela qual a mesma silhueta aparece em todos os lugares;
o teorema que o certifica é o teorema do limite central.

\textbf{12.} Uma raspagem: uma equipe vence todas $4$:
$2 \times \left(\frac12\right)^4 = \frac18$.

\textbf{13.} Sete jogos exigem $3$--$3$ depois das seis:
$\binom63 \left(\frac12\right)^6 = \frac{20}{64} =
\frac{5}{16}$.

\textbf{14.} Termina em $5$: o vencedor leva o jogo 5 e $3$ de
o primeiro $4$:
$2 \times \binom43 \left(\frac12\right)^5 = \frac14$. Termina em
$6$: $2 \times \binom53 \left(\frac12\right)^6 = \frac{5}{16}$.
\omterm{def:g11:prob:rv}{Distribuição} sobre $4, 5, 6, 7$:
$\frac18, \frac14, \frac{5}{16}, \frac{5}{16}$ (soma $1$).
Comprimento esperado: $4 \cdot \frac18 + 5 \cdot \frac14 + 6 \cdot
\frac{5}{16} + 7 \cdot \frac{5}{16} = 5.8125$ jogos. Seis e
séries de sete jogos são as mais prováveis --- o drama está embutido
o formato.

\textbf{15.} Ganhe em $4$: $0.6^4 = 0.1296$; em $5$:
$\binom43\,0.6^3 \times 0.4 \times 0.6 = 0.2074$; em $6$:
$\binom53\,0.6^3 \times 0.4^2 \times 0.6 = 0.2074$; em $7$:
$\binom63\,0.6^3 \times 0.4^3 \times 0.6 = 0.1659$. Total:
sobre $0.710$: um $60\,\%$-vitórias da equipe por jogo $71\,\%$ de
série --- a série amplifica a vantagem.

\textbf{16.} Final única: $60\,\%$. Melhor de 3:
$ p^2 + 2p^2 q = 0.36 + 0.288 = 0.648$. A escada
$60\,\% \to 65\,\% \to 71\,\%$ continua com comprimento: mais
os jogos calculam a média da sorte (a lei dos grandes números em miniatura),
tão longas finais coroam a habilidade --- que é exatamente o que as ligas
vender.

\textbf{17.} Moeda justa: $\E = 50$, $\sigma = \sqrt{25} = 5$;
$ z = \frac{62 - 50}{5} = 2.4$: além do $2\sigma $
convenção --- a moeda merece uma investigação.

\textbf{18.} $\P(X = 0) = 0.98^{50} \approx 0.364$;
$\P(X = 1) = 50 \times 0.02 \times 0.98^{49} \approx 0.372$;
$\P(X = 2) = \binom{50}{2} 0.02^2 \times 0.98^{48} \approx
0.186$. Então $\P(X \geq 3) \approx 1 - 0.922 = 0.078$: sobre
$7.8\,\%$ --- acima do $5\,\%$ limite, então não (ainda)
o suficiente para rejeitar a reivindicação; um segundo lote ruim mudaria
a história.

\textbf{19.} $\P(\text{at least one win}) = 1 - 0.999^{1000}
\approx 1 - 0.368 = 0.632$: mil ingressos de uma só vez
mil não dão certeza, mas $63\,\%$. O recorrente
$0.368$ é $\frac1e $ disfarçado --- a constante $ e $ faz
sua entrada oficial na 12ª série.

\textbf{20.} Reconhecimento: fixo $ n $, independência, constante
$ p $ --- então e só então, binomial. Mesa: Pascal
triângulo, linha $ n $. Forma: o sino, simétrico para
$ p = \frac12$, mudou de outra forma. Bússola: centro $ np $,
espalhar $\sqrt{np(1-p)}$ --- o pontuações z de decisões.
Herdeiros: a curva suave em forma de sino à qual as pilhas se aproximam e o
lei dos grandes números que explica por que grandes tabuleiros nunca mentem.
\end{solution}
