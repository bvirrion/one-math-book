\section*{Capítulo \ref{ch:g11:func} --- Funções e Variações}

\begin{solution}{exo:g11:func:1}
$f $: precisar $3 - x \geq 0$, domínio $\intoc{-\infty}{3}$.

$ g $: precisar $ x^2 - 4 \neq 0$, ou seja\ $ x \neq \pm 2$: domínio
$\R \setminus \{-2, 2\}$.

$ h $: precisar $ x \geq 0$ e $ x \neq 5$: domínio
$\intco{0}{5} \cup \intoo{5}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:2}
$\sqrt{11} < \sqrt{13}$ desde $\sqrt{}$ é \omterm{def:g11:func:monotone}{aumentando} e $11 < 13$.

$(-2.1)^2 > (-2.05)^2$ desde $ x^2$ é diminuindo sobre
$\intoc{-\infty}{0}$ e $-2.1 < -2.05$.

$\frac{1}{0.9} > \frac{1}{1.1}$ desde $\frac1x $ é diminuindo sobre
$\intoo{0}{+\infty}$ e $0.9 < 1.1$.

$(-1.01)^3 < (-0.99)^3$ desde $ x^3$ é \omterm{def:g11:func:monotone}{aumentando} e $-1.01 < -0.99$.
\end{solution}

\begin{solution}{exo:g11:func:3}
$ f(-x) = 3(-x)^4 - (-x)^2 = 3x^4 - x^2 = f(x)$: até.

$ g(-x) = -x^3 - x = -g(x)$: \omterm{def:g11:func:parity}{chance}.

$ h(1) = 2$ e $ h(-1) = 1 - 1 = 0$, então $ h(-1)$ não é nenhum dos dois $ h(1)$ nem
$-h(1)$: nenhum até nem \omterm{def:g11:func:parity}{chance}.

$ k(-x) = \frac{-x}{x^2+1} = -k(x)$: \omterm{def:g11:func:parity}{chance}.
\end{solution}

\begin{solution}{exo:g11:func:4}
Leia o \omterm{def:g10:functions:table}{tabela de variação}. Sobre $\intcc{-3}{1}$, $ f $ sobe de $-2$ para $4$
e cruza o valor $1$ exatamente uma vez; sobre $\intcc{1}{5}$ isso desce
de $4$ para $0$ e cruzes $1$ mais uma vez: o equação $ f(x) = 1$ tem
$2$ soluções. O \omterm{def:g10:functions:extrema}{máximo} de $ f $ é $ f(1) = 4 < 5$, então $ f(x) = 5$ não tem
solução.
\end{solution}

\begin{solution}{exo:g11:func:5}
Deixar $2 \leq u < v $. Então
\[
f(v) - f(u) = v^2 - 4v - u^2 + 4u = (v-u)(v+u) - 4(v-u)
= (v - u)(u + v - 4).
\]
Ambos os fatores são positivos: $ v - u > 0$, e $ u + v > 2 + 2 = 4$ desde
$ u \geq 2$ e $ v > 2$. Por isso $ f(v) > f(u)$: $ f $ é estritamente \omterm{def:g11:func:monotone}{aumentando}
sobre $\intco{2}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:6}
$ f = \sqrt{u}$ com $ u(x) = 5 - x $ diminuindo e não negativo em
$\intoc{-\infty}{5}$: $ f $ é diminuindo lá
(\cref{prop:g11:func:transforms}).

$ g = \frac1u $ com $ u(x) = x^2 + 1$ \omterm{def:g11:func:monotone}{aumentando} e positivo em
$\intco{0}{+\infty}$: $ g $ é diminuindo lá.

$ h = 3 - 2\sqrt x $: $\sqrt x $ é \omterm{def:g11:func:monotone}{aumentando}, multiplicando por $-2 < 0$
inverte, adicionando $3$ não muda nada: $ h $ é diminuindo sobre
$\intco{0}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:7}
\emph{1.} $ f $ está em \omterm{thm:g11:quad:canonical}{forma canônica}: diminuindo sobre $\intoc{-\infty}{2}$,
\omterm{def:g11:func:monotone}{aumentando} sobre $\intco{2}{+\infty}$, \omterm{def:g10:functions:extrema}{mínimo} $ f(2) = -3$.

\emph{2.} $ g = \frac{1}{u}$ com $ u(x) = (x-2)^2 + 1 = f(x) + 4$. O
mudar por $4$ mantém as variações de $ f $, e $ u \geq 1 > 0$. Tomando o
recíproco os inverte: $ g $ é \omterm{def:g11:func:monotone}{aumentando} sobre $\intoc{-\infty}{2}$ e
diminuindo sobre $\intco{2}{+\infty}$, com \omterm{def:g10:functions:extrema}{máximo}
$ g(2) = \frac{1}{1} = 1$.
\end{solution}

\begin{solution}{exo:g11:func:8}
Qualquer curva em forma de colina através dos três pontos funciona. Então:
$ f + 2$ passa $(0,3)$, $(2,5)$, $(4,3)$, mesmas variações
(deslocamento vertical); $-f $ passa $(0,-1)$, $(2,-3)$, $(4,-1)$ com
\emph{invertido} variações (o pico vira vale); $2f $ passes
através $(0,2)$, $(2,6)$, $(4,2)$, mesmas variações, mas amplificadas
verticalmente.
\end{solution}

\begin{solution}{exo:g11:func:9}
$2 - \frac{1}{x+1} = \frac{2(x+1) - 1}{x+1} = \frac{2x+1}{x+1} = f(x)$.
Sobre $\intoo{-1}{+\infty}$, $ u(x) = x + 1$ é \omterm{def:g11:func:monotone}{aumentando} e positivo, então
$\frac1u $ é diminuindo, $-\frac1u $ é aumentandoe adicionando $2$ mudanças
nada: $ f $ é estritamente \omterm{def:g11:func:monotone}{aumentando} sobre $\intoo{-1}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:10}
\emph{1. Verdadeiro.} Se $ u < v $, então
$(f+g)(v) - (f+g)(u) = \bigl(f(v)-f(u)\bigr) + \bigl(g(v)-g(u)\bigr)$ é
uma soma de dois números não negativos.

\emph{2. Falso.} $ f(x) = g(x) = x $ são \omterm{def:g11:func:monotone}{aumentando} sobre $\R $, mas seus
produto $ x^2$ não é \omterm{def:g11:func:monotone}{aumentando} sobre $\R $ (diminui em
$\intoc{-\infty}{0}$). (A afirmação torna-se verdadeira se ambos funções são
também não negativo.)

\emph{3. Falso.} Mesmo contra-exemplo: $ f(x) = x $ é \omterm{def:g11:func:monotone}{aumentando} sobre $\R $
mas $ f^2(x) = x^2$ não é.
\end{solution}

\begin{solution}{exo:g11:func:11}
\emph{1.} Para $0 < u < v $:
\[
f(v) - f(u) = (v - u) + \frac1v - \frac1u
= (v - u) + \frac{u - v}{uv}
= (v - u)\left(1 - \frac{1}{uv}\right)
= (v - u)\,\frac{uv - 1}{uv}.
\]

\emph{2.} Sempre $ v - u > 0$ e $ uv > 0$. Se $0 < u < v \leq 1$, então
$ uv < 1$, então $ uv - 1 < 0$ e $ f(v) < f(u)$: $ f $ é estritamente diminuindo
sobre $\intoc{0}{1}$. Se $1 \leq u < v $, então $ uv > 1$ e $ f(v) > f(u)$:
$ f $ é estritamente \omterm{def:g11:func:monotone}{aumentando} sobre $\intco{1}{+\infty}$.

\emph{3.} $ f $ diminui antes $1$ e aumenta depois, então
$ f(x) \geq f(1) = 2$ para todos $ x > 0$, com igualdade exatamente em $ x = 1$.
(Essa desigualdade reaparece sob muitas formas; compare
\cref{ex:g11:quad:canonical} e as desigualdades da linha tangente de
\cref{ch:g12:deriv}.)
\end{solution}

\begin{solution}{pb:g11:func:1}
\textbf{1.} $ x^4 - 3x^2$: até ($(-x)^4 - 3(-x)^2 = x^4 -
3x^2$). $ x^3 - x $: \omterm{def:g11:func:parity}{chance}. $ x^2 + x $: nenhum ($ f(-1) = 0 \neq
f(1) = 2$ e $\neq -f(1)$). $\abs x $: até. $\frac1x $: \omterm{def:g11:func:parity}{chance}.

\textbf{2.} Até: a curva é simétrica ao longo do $ y $-eixo
(espelho). \omterm{def:g11:func:parity}{Chance}: simétrico em relação à origem (meia volta).

\textbf{3.} Se $ f $ e $ g $ são \omterm{def:g11:func:parity}{chance}:
$(fg)(-x) = f(-x)g(-x) = (-f(x))(-g(x)) = (fg)(x)$: até.
Mesa: até $\times $ até $=$ até; até $\times $ \omterm{def:g11:func:parity}{chance} $=$ \omterm{def:g11:func:parity}{chance};
\omterm{def:g11:func:parity}{chance} $\times $ \omterm{def:g11:func:parity}{chance} $=$ até --- a regra do sinal com paridades.

\textbf{4.} \omterm{def:g11:func:parity}{Chance} no $0$: $ f(0) = -f(0)$, então $ f(0) = 0$. Até:
$ f(-3) = f(3) = 7$. Ambos até e \omterm{def:g11:func:parity}{chance}:
$ f(x) = f(-x) = -f(x)$ para todos $ x $, então $2f(x) = 0$: $ f $ é o
zero \omterm{def:g11:func:function}{função}.

\textbf{5.} A metade direita ($ x \geq 0$) determina um mesmo
\omterm{def:g11:func:function}{função} completamente: a metade esquerda é o seu espelho. Nenhum até
nem \omterm{def:g11:func:parity}{chance}: $ x^2 + x $ (questão 1) --- suas duas simetrias falham em
$ x = 1$.

\textbf{6.} $ E(x) = \frac{(x^3 + x^2 + 1) + (-x^3 + x^2 +
1)}{2} = x^2 + 1$ (até), $ O(x) = \frac{(x^3 + x^2 + 1) -
(-x^3 + x^2 + 1)}{2} = x^3$ (\omterm{def:g11:func:parity}{chance}); a soma deles é $ f $.

\textbf{7.} $ E(-x) = \frac{f(-x) + f(x)}{2} = E(x)$: até.
$ O(-x) = \frac{f(-x) - f(x)}{2} = -O(x)$: \omterm{def:g11:func:parity}{chance}. E
$ E(x) + O(x) = f(x)$ por adição direta: cada \omterm{def:g11:func:function}{função}
divide.

\textbf{8.} De $ E_1 + O_1 = E_2 + O_2$:
$ E_1 - E_2 = O_2 - O_1$. O lado esquerdo é até, o lado direito
\omterm{def:g11:func:parity}{chance} --- então isso \omterm{def:g11:func:function}{função} é ambos, portanto zero (questão 4):
$ E_1 = E_2$ e $ O_1 = O_2$. Apenas uma divisão.

\textbf{9.} $ E(x) = \frac12\left(\frac{1}{1 - x} + \frac{1}{1 +
x}\right) = \frac12 \cdot \frac{(1 + x) + (1 - x)}{1 - x^2}
= \frac{1}{1 - x^2}$, e
$ O(x) = \frac12 \cdot \frac{(1 + x) - (1 - x)}{1 - x^2}
= \frac{x}{1 - x^2}$. Verificar: $ E $ é até, $ O $ é chancee
a soma deles é $\frac{1 + x}{1 - x^2} =
\frac{1 + x}{(1 - x)(1 + x)} = \frac{1}{1 - x} = f(x)$.

\textbf{10.} Como cada peça obedece a uma simetria, metade da
trabalho (e metade dos dados) é suficiente para cada um: resolva de um lado,
espelho para o outro --- e muitas operações (integração,
série, análise de sinal) tratam peças simétricas muito mais
simplesmente do que cru funções.

\textbf{11.} $(x-3)^2$: mudou $3$ certo. $ x^2 + 2$: mudou
$2$ acima. $(x-3)^2 + 2$: ambos. $-x^2$: virado de cabeça para baixo.
$2x^2$: esticado verticalmente por $2$ (tigela mais estreita).

\textbf{12.} Para $ x \geq 2$: $(x - 2) + (x + 2) = 2x $. Para
$-2 \leq x \leq 2$: $(2 - x) + (x + 2) = 4$. Para $ x \leq -2$:
$-2x $. A curva desce com declive $-2$, corre plano em altura
$4$ entre $-2$ e $2$, então sobe com declive $2$: um vale
com piso plano.

\textbf{13.} Deslocamento vertical: $ f(x) + 2$ fica até
($ f(-x) + 2 = f(x) + 2$). Alongamento vertical: $2f(x)$ fica
até. Mudança horizontal: $(x - 3)^2$ não é até
($ f(-1) = 16 \neq 4 = f(1)$): deslizar lateralmente destrói o
espelho.

\textbf{14.} No piso plano o valor é $4 \neq 6$. Para
$ x \geq 2$: $2x = 6$, $ x = 3$. Para $ x \leq -2$: $-2x = 6$,
$ x = -3$. Soluções: $\pm 3$.

\textbf{15.} $ f(x) = a(x - 2)^2 - 3$ com $ f(0) = 4a - 3 = 1$:
$ a = 1$, então $ f(x) = (x - 2)^2 - 3$.

\textbf{16.} $ g(v) - g(u) = (v - u) + 4\,\frac{u - v}{uv}
= (v - u)\,\frac{uv - 4}{uv}$: negativo para $ u < v $ com
$ uv < 4$, positivo com $ uv > 4$: diminuindo sobre
$\intoc{0}{2}$, \omterm{def:g11:func:monotone}{aumentando} sobre $\intco{2}{+\infty}$; \omterm{def:g10:functions:extrema}{mínimo} no
$ x = 2$, de valor $ g(2) = 4$.

\textbf{17.} Lados $ x $ e $\frac{16}{x}$:
$ P(x) = 2\left(x + \frac{16}{x}\right)$, minimizado (pergunta
Máquina de 16 anos com $4 \to 16$, ponto de viragem $\sqrt{16} = 4$)
no $ x = 4$: o $4 \times 4$ \emph{quadrado}, perímetro $16$ ---
A resposta de Dido, agora provada para todos os comprimentos laterais reais.

\textbf{18.} Sobre $\intco{0}{+\infty}$: $ x \mapsto x^2 + 1$
aumenta, e $\sqrt{\phantom{x}}$ aumenta, então o
aumentos compostos. E $ h(-x) = \sqrt{x^2 + 1} = h(x)$: $ h $
é até, então, por simetria espelhada, diminui em
$\intoc{-\infty}{0}$. \omterm{def:g10:functions:extrema}{Mínimo} $ h(0) = 1$.

\textbf{19.} Distância quadrada até $(3, 0)$ do ponto
$(x, \sqrt x)$:
$ d^2(x) = (x - 3)^2 + x = x^2 - 5x + 9$, \omterm{def:g10:functions:extrema}{mínimo} em
$ x = \frac52$: ponto mais próximo
$\left(\frac52, \sqrt{\frac52}\right)$, distância
$\sqrt{\frac{11}{4}} = \frac{\sqrt{11}}{2}$. Minimizando $ d^2$
é legítimo porque a quadratura é estritamente \omterm{def:g11:func:monotone}{aumentando} ligado
números positivos: $ d $ e $ d^2$ fundo no mesmo lugar.

\textbf{20.} Estoque bruto: as curvas de referência, conhecidas de cor.
Máquinas: muda, vira e estica montando novas \omterm{def:g10:functions:graph}{gráficos}
desde antigo. Controle de qualidade: paridade e divisão par-ímpar,
que reduzem pela metade todos os estudos. Bancada de medição: \omterm{def:g10:algebra:expand}{factoring}
$ f(v) - f(u)$, que lê variações e \omterm{def:g10:functions:extrema}{mínimos} diretamente. O
ferramenta elétrica do próximo capítulo: o \emph{derivado}, onde se lê
eles de relance.
\end{solution}
