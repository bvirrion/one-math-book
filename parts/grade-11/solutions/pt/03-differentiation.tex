\section*{Capítulo \ref{ch:g11:deriv} --- Diferenciação: Um Primeiro Curso}

\begin{solution}{exo:g11:deriv:1}
Para $f(x) = x^2$ no $ a = 2$:
\[
\frac{(2+h)^2 - 4}{h} = \frac{4h + h^2}{h} = 4 + h
\xrightarrow[h\to0]{} 4,
\]
então $ f'(2) = 4$. Para $ g(x) = x^2 + 3x $ no $ a = 1$: $ g(1) = 4$ e
\[
\frac{(1+h)^2 + 3(1+h) - 4}{h} = \frac{5h + h^2}{h} = 5 + h
\xrightarrow[h\to0]{} 5,
\]
então $ g'(1) = 5$.
\end{solution}

\begin{solution}{exo:g11:deriv:2}
$ f'(x) = 12x^2 - 4x + 1$.

$ g'(x) = \frac{5}{2\sqrt x} - \frac{3}{x^2}$ (sobre $\intoo{0}{+\infty}$).

$ h'(x) = 2(x^2 - 3) + (2x+1)(2x) = 2x^2 - 6 + 4x^2 + 2x
= 6x^2 + 2x - 6$ (regra do produto).
\end{solution}

\begin{solution}{exo:g11:deriv:3}
$ f(x) = x^2 - 3x $, $ a = 2$: $ f(2) = -2$, $ f'(x) = 2x - 3$ então $ f'(2) = 1$;
\omterm{def:g11:deriv:tangent}{tangente} $ y = -2 + (x - 2) = x - 4$.

$ f(x) = \frac1x $, $ a = 1$: $ f(1) = 1$, $ f'(1) = -1$; \omterm{def:g11:deriv:tangent}{tangente}
$ y = 1 - (x - 1) = 2 - x $.

$ f(x) = \sqrt x $, $ a = 4$: $ f(4) = 2$, $ f'(4) = \frac{1}{2\sqrt4} =
\frac14$; \omterm{def:g11:deriv:tangent}{tangente} $ y = 2 + \frac14(x - 4) = \frac{x}{4} + 1$.
\end{solution}

\begin{solution}{exo:g11:deriv:4}
$ f'(x) = \frac{(x-1) - (x+2)}{(x-1)^2} = \frac{-3}{(x-1)^2}$.

$ g'(x) = \frac{2x(x^2+1) - x^2 \times 2x}{(x^2+1)^2}
= \frac{2x}{(x^2+1)^2}$.

$ h'(x) = -\frac{2x+1}{(x^2+x+1)^2}$ (regra do quociente com $ u = 1$).
\end{solution}

\begin{solution}{exo:g11:deriv:5}
$ f'(x) = 3x^2 - 12x + 9 = 3(x^2 - 4x + 3) = 3(x-1)(x-3)$: positivo
fora $\intcc{1}{3}$, negativo por dentro. Então $ f $ é \omterm{def:g11:func:monotone}{aumentando} sobre
$\intoc{-\infty}{1}$, diminuindo sobre $\intcc{1}{3}$, \omterm{def:g11:func:monotone}{aumentando} sobre
$\intco{3}{+\infty}$, com um local \omterm{def:g10:functions:extrema}{máximo} $ f(1) = 1 - 6 + 9 - 2 = 2$ e
um local \omterm{def:g10:functions:extrema}{mínimo} $ f(3) = 27 - 54 + 27 - 2 = -2$.
\end{solution}

\begin{solution}{exo:g11:deriv:6}
Sobre $\intoo{-1}{+\infty}$:
\[
f'(x) = \frac{2x(x+1) - (x^2+3)}{(x+1)^2}
= \frac{x^2 + 2x - 3}{(x+1)^2}
= \frac{(x+3)(x-1)}{(x+1)^2}.
\]
Para $ x > -1$, ambos $ x + 3$ e $(x+1)^2$ são positivos, então $ f'(x)$ tem o
sinal de $ x - 1$: $ f $ é diminuindo sobre $\intoc{-1}{1}$ e \omterm{def:g11:func:monotone}{aumentando} sobre
$\intco{1}{+\infty}$, com \omterm{def:g10:functions:extrema}{mínimo} $ f(1) = \frac{4}{2} = 2$.
\end{solution}

\begin{solution}{exo:g11:deriv:7}
$ f'(x) = 2x - 4$.

\emph{1.} Horizontal \omterm{def:g11:deriv:tangent}{tangente}: $ f'(x) = 0$ no $ x = 2$; o ponto é
$(2, f(2)) = (2, 1)$ (o vértice do \omterm{def:g11:quad:parabola}{parábola}).

\emph{2.} Paralelo a $ y = 2x + 1$ significa declive $2$: $2x - 4 = 2$, então
$ x = 3$, dando o ponto $(3, f(3)) = (3, 2)$.
\end{solution}

\begin{solution}{exo:g11:deriv:8}
Com largura $ x $ (dois lados cercados) e comprimento $60 - 2x $:
$ A(x) = x(60 - 2x) = 60x - 2x^2$ sobre $\intoo{0}{30}$. Então
$ A'(x) = 60 - 4x $, positivo antes $ x = 15$ e negativo depois: a área
é \omterm{def:g10:functions:extrema}{máximo} para $ x = 15$, dando um $15 \times 30$ m recinto de área
$450$ eu $^2$. A fórmula do vértice $\alpha = -\frac{60}{2 \times (-2)} = 15$
de \cref{ch:g11:quad} dá a mesma resposta sem cálculo.
\end{solution}

\begin{solution}{exo:g11:deriv:9}
Deixar $ f(x) = \frac{x+1}{2} - \sqrt x $ sobre $\intoo{0}{+\infty}$. Então
\[
f'(x) = \frac12 - \frac{1}{2\sqrt x} = \frac{\sqrt x - 1}{2\sqrt x},
\]
que tem o sinal de $\sqrt x - 1$: negativo ligado $\intoo{0}{1}$, positivo
ligado $\intoo{1}{+\infty}$. Então $ f $ diminui e depois aumenta, com \omterm{def:g10:functions:extrema}{mínimo}
$ f(1) = 1 - 1 = 0$. Por isso $ f(x) \geq 0$ para todos $ x > 0$, aquilo é
$\sqrt x \leq \frac{x+1}{2}$, com igualdade apenas em $ x = 1$.
\end{solution}

\begin{solution}{exo:g11:deriv:10}
De $\pi r^2 h = 500$, $ h = \frac{500}{\pi r^2}$, então
\[
S(r) = 2\pi r^2 + 2\pi r \cdot \frac{500}{\pi r^2}
= 2\pi r^2 + \frac{1000}{r},
\qquad
S'(r) = 4\pi r - \frac{1000}{r^2}.
\]
$ S'(r) = 0$ quando $4\pi r^3 = 1000$, ou seja\ $ r^3 = \frac{250}{\pi}$, então
$ r = \sqrt[3]{250/\pi}$ ($\approx 4.3$ cm). Desde $ S'(r) =
\frac{4\pi r^3 - 1000}{r^2}$ é negativo antes deste valor e positivo
depois, é um \omterm{def:g10:functions:extrema}{mínimo}.
\end{solution}

\begin{solution}{exo:g11:deriv:11}
O \omterm{def:g11:deriv:tangent}{tangente} para $ y = x^2$ no ponto de \omterm{def:g10:coordgeom:system}{abscissa} $ a $ é
$ y = a^2 + 2a(x - a) = 2ax - a^2$. Ele passa $ P(1, -3)$ quando
$-3 = 2a - a^2$, ou seja\ $ a^2 - 2a - 3 = 0$, cujo raízes são $ a = 3$ e
$ a = -1$ ($\Delta = 16$). Existem, portanto, \emph{dois} \omterm{def:g11:deriv:tangent}{tangentes} através
$ P $:
\[
y = 6x - 9 \quad (a = 3),
\qquad
y = -2x - 1 \quad (a = -1).
\]
\end{solution}

\begin{solution}{exo:g11:deriv:12}
De \cref{ex:g11:deriv:study}, $ f $ aumenta até o local \omterm{def:g10:functions:extrema}{máximo}
$ f(-1) = 3$, diminui para o local \omterm{def:g10:functions:extrema}{mínimo} $ f(1) = -1$, então aumenta
novamente; para $ x $ grande positivo (resp.\ negativo), $ x^3$ domina e $ f $
assume valores arbitrariamente grandes (resp.\ pequenos). Lendo linhas horizontais
$ y = k $ em todo o \omterm{def:g10:functions:table}{tabela de variação}:
\begin{itemize}
  \item $ k > 3$ ou $ k < -1$: a linha encontra a curva uma vez
        --- $1$ solução;
  \item $ k = 3$ ou $ k = -1$: a linha toca um ponto de viragem e cruza
        um outro ramo --- $2$ soluções;
  \item $-1 < k < 3$: a linha cruza todos os três ramos --- $3$
        soluções.
\end{itemize}
\end{solution}

\begin{solution}{pb:g11:deriv:1}
\textbf{1.} $ f'(x) = 6x - 5$; $ g'(x) = 3x^2 - 12$. Definição:
$\frac{(1 + h)^2 - 1}{h} = 2 + h \to 2$: o derivado de
$ x^2$ no $1$ é $2$.

\textbf{2.} $ g(2) = 8 - 24 = -16$ e $ g'(2) = 12 - 12 = 0$:
o \omterm{def:g11:deriv:tangent}{tangente} é a linha horizontal $ y = -16$. Uma horizontal
\omterm{def:g11:deriv:tangent}{tangente} sinaliza um ponto crítico --- aqui um local \omterm{def:g10:functions:extrema}{mínimo}, como
o \omterm{def:g10:functions:table}{tabela de variação} da questão 5 confirma.

\textbf{3.} Declive $2a $ no $(a, a^2)$:
$ y = a^2 + 2a(x - a) = 2ax - a^2$.

\textbf{4.} No $ x = 0$: $ y = -a^2$: o \omterm{def:g11:deriv:tangent}{tangente} atravessa o
eixo em profundidade $ a^2$ abaixo da origem --- exatamente tão longe
\emph{abaixo} zero como $ P $ fica acima dele. Receita: para desenhar o
\omterm{def:g11:deriv:tangent}{tangente} em um ponto de altura $ h $, junte-o ao ponto do eixo em
profundidade $ h $ abaixo da origem. (Não derivado necessário no
prancheta.)

\textbf{5.} $ g'(x) = 3(x^2 - 4)$: positivo ligado
$\intoo{-\infty}{-2}$, negativo em $\intoo{-2}{2}$, positivo
além. $ g $ aumenta para um local \omterm{def:g10:functions:extrema}{máximo} $ g(-2) = 16$, cai
para um local \omterm{def:g10:functions:extrema}{mínimo} $ g(2) = -16$, então aumenta.

\textbf{6.} $ F\left(0, \frac14\right)$ e $ T(0, -a^2)$:
$ FT = \frac14 + a^2$.

\textbf{7.} $ FP = a^2 + \frac14 = FT $: isósceles em $ F $, então o
os ângulos da base são iguais: $\widehat{FTP} = \widehat{FPT}$.

\textbf{8.} O raio que chega em $ P $ é vertical, portanto paralelo
para a linha $(TF)$ (o $ y $-eixo). O \omterm{def:g11:deriv:tangent}{tangente} $(TP)$ cruzes
ambas paralelas, então o ângulo entre o raio e o \omterm{def:g11:deriv:tangent}{tangente}
é igual $\widehat{FTP}$ (ângulos alternados) --- que é igual
$\widehat{FPT}$, o ângulo entre $[PF]$ e o \omterm{def:g11:deriv:tangent}{tangente}.
Ângulos iguais nos dois lados do \omterm{def:g11:deriv:tangent}{tangente} no $ P $.

\textbf{9.} Pela lei da reflexão, o raio que chega
verticalmente em $ P $ sai ao longo da linha fazendo o ângulo igual
do outro lado --- pela pergunta 8, junto $[PF]$: através do
\omterm{pb:g11:quad:1}{foco}, qualquer que seja $ a $ é. Invertida, luz nascida em $ F $ folhas
cada ponto do espelho paralelo ao eixo. Pratos coletam,
feixe de faróis: comprovado.

\textbf{10.} $ a = 1$: \omterm{def:g11:deriv:tangent}{tangente} $ y = 2x - 1$, $ T(0, -1)$;
$ FP = \sqrt{1 + \left(1 - \frac14\right)^2} =
\sqrt{\frac{25}{16}} = \frac54$; $ FT = \frac14 + 1 = \frac54$:
isósceles, como prometido.

\textbf{11.} O \omterm{def:g11:deriv:tangent}{tangente} no $ x_n $ é
$ y = f(x_n) + f'(x_n)(x - x_n)$; atravessa $ y = 0$ onde
$ x = x_n - \frac{f(x_n)}{f'(x_n)}$.

\textbf{12.} $ x_1 = 1 - \frac{-1}{2} = \frac32$;
$ x_2 = \frac32 - \frac{1/4}{3} = \frac{17}{12}$;
$ x_3 = \frac{577}{408}$: Garça aproximações de $\sqrt2$,
ou seja, a máquina de \omterm{pb:g10:functions:1}{ponto fixo} de \cref{pb:g10:functions:1}.
Identidade: $ x - \frac{x^2 - 2}{2x} = \frac{2x^2 - x^2 + 2}{2x}
= \frac{x^2 + 2}{2x} = \frac12\left(x + \frac2x\right)$ ---
Receita de garça \emph{é} Método de Newton aplicado a
$ x^2 - 2$, dois milênios antes.

\textbf{13.} $ x_1 = 5 - \frac{125 - 100}{75} = \frac{14}{3}
\approx 4.6667$; $ x_2 \approx 4.6667 - \frac{1.63}{65.33}
\approx 4.6417$. Calculadora: $\sqrt[3]{100} = 4.6416\ldots $ ---
quatro casas decimais corretas em duas etapas.

\textbf{14.} $ f'(2) = 0$: o \omterm{def:g11:deriv:tangent}{tangente} no ponto de partida é
horizontal (questão 2) e nunca cruza o eixo --- o
fórmula divide por zero. Advertência: comece Newton longe de
pontos críticos (e, mais geralmente, perto o suficiente do \omterm{def:g11:quad:discriminant}{raiz}
você quiser).

\textbf{15.} A bissecção ganha um dígito binário por etapa; Newton
aproximadamente \emph{dobra} o número de dígitos corretos em cada etapa
($1 \to 3 \to 6$ aqui). Quando cada etapa custa --- milhões
de parâmetros, navegação em tempo real --- duplicação de batidas
avançando, razão pela qual o \omterm{def:g11:deriv:tangent}{tangente} a linha corre tanto por dentro
silício.

\textbf{16.} $ d^2 = 900 - w^2$, então
$ S(w) = w(900 - w^2) = 900w - w^3$, para $0 < w < 30$.

\textbf{17.} $ S'(w) = 900 - 3w^2 = 0$ no
$ w = \sqrt{300} = 10\sqrt3 \approx 17.3$ cm ($ S' > 0$ antes,
$< 0$ depois: um \omterm{def:g10:functions:extrema}{máximo}). Então
$ d = \sqrt{600} = 10\sqrt6 \approx 24.5$ cm.

\textbf{18.} $\frac{d^2}{w^2} = \frac{600}{300} = 2$, então
$ d = w\sqrt2$: profundidade é largura vezes $\sqrt2$ --- o
proporção de carpinteiro (e a construção do diâmetro em terços
entrega com bússola, sem álgebra no local).

\textbf{19.} Feixe quadrado: $ w = d = \sqrt{450} \approx 21.2$
centímetros, rigidez $ w^3 = 450^{3/2} \approx 9\,546$. Feixe ideal:
$ S = 10\sqrt3 \times 600 = 6\,000\sqrt3 \approx 10\,392$. Ganho:
sobre $8.9\,\%$ mais rígido do mesmo tronco --- o derivado
paga a serraria.

\textbf{20.} Geometria: a tangenteos ângulos iguais viraram um
parabolóide em um coletor de luz. Numéricos: deslizando para baixo
\omterm{def:g11:deriv:tangent}{tangentes} resolve equações na velocidade de duplicação de dígitos.
Otimização: um desaparecimento derivado encontrou o tronco mais forte
retângulo. Um objeto, três negociações --- e nota 12
a convexidade dirá, além disso, de que lado de cada
\omterm{def:g11:deriv:tangent}{tangente} a curva está.
\end{solution}
