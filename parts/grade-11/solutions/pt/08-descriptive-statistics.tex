\section*{Capítulo \ref{ch:g11:stat} --- Estatística Descritiva}

\begin{solution}{exo:g11:stat:1}
$n = 8$, soma $71$: significar $\bar x = \frac{71}{8} = 8.875$. \omterm{def:g11:stat:median}{Mediana}: \omterm{prop:g10:coordgeom:midpoint}{ponto médio}
do $4$ o e $5$ os valores, $\frac{8 + 9}{2} = 8.5$. Quartis: um
quarto de $8$ é $2$, então $ Q_1$ é o $2$ segundo valor, $7$; três quartos é
$6$, então $ Q_3$ é o $6$ o valor, $10$. \omterm{def:g11:stat:quartiles}{Intervalo interquartil}:
$10 - 7 = 3$.
\end{solution}

\begin{solution}{exo:g11:stat:2}
Soma $40$, então $\bar x = 5$. Soma dos quadrados:
$4 + 16 + 16 + 16 + 25 + 25 + 49 + 81 = 232$. Pela fórmula de atalho,
\[
V = \frac{232}{8} - 5^2 = 29 - 25 = 4,
\qquad
\sigma = 2 .
\]
\end{solution}

\begin{solution}{exo:g11:stat:3}
Significar:
\[
\bar x = \frac{1 \times 6 + 2 \times 9 + 3 \times 8 + 4 \times 10
+ 5 \times 9 + 6 \times 8}{50}
= \frac{181}{50} = 3.62 .
\]
\omterm{def:g11:stat:median}{Mediana}: as \omterm{def:g10:stats:series}{frequências} cumulativas são $6, 15, 23, 33, \dots $, então o
$25$ o e $26$ os resultados classificados são ambos $4$: o \omterm{def:g11:stat:median}{mediana} é $4$.
\end{solution}

\begin{solution}{exo:g11:stat:4}
O total do $15$ marcas é $15 \times 11.2 = 168$; com o
recém-chegado, $\frac{168 + 18}{16} = \frac{186}{16} = 11.625$.
\end{solution}

\begin{solution}{exo:g11:stat:5}
$\frac{6 + 8 + x + 12 + 14}{5} = 10$ dá $40 + x = 50$, então $ x = 10$.
Os desvios de $6, 8, 10, 12, 14$ do significar $10$ são
$-4, -2, 0, 2, 4$, com quadrados somando $40$:
$ V = \frac{40}{5} = 8$ e $\sigma = 2\sqrt2 \approx 2.83$.
\end{solution}

\begin{solution}{exo:g11:stat:6}
Por \cref{prop:g11:stat:affine} com $ a = \frac12$, $ b = 4$: novo significar
$\frac{12.4}{2} + 4 = 10.2$; novo \omterm{def:g11:stat:variance}{desvio padrão}
$\frac{2.5}{2} = 1.25$.
\end{solution}

\begin{solution}{exo:g11:stat:7}
O pool significar é o \emph{ponderado} média
\[
\frac{20 \times 10 + 30 \times 15}{50} = \frac{650}{50} = 13,
\]
não $12.5$: o grupo B contribui com mais valores, por isso puxa o pool significar
em direção ao seu próprio.
\end{solution}

\begin{solution}{exo:g11:stat:8}
Ambas as linhas estão quase no alvo, em média ($1.002$ e $1.001$ kg), mas
linha 2 \omterm{def:g11:stat:variance}{desvio padrão} é três vezes maior: suas malas variam muito
mais, produzindo sacos muito mais cheios ou insuficientemente cheios. Linha 2
deve ser tratado primeiro --- consistência, não apenas o significar, é o que
precisa de conserto.
\end{solution}

\begin{solution}{exo:g11:stat:9}
$\bar x = \frac{70}{10} = 7$;
$ V = \frac{540}{10} - 7^2 = 54 - 49 = 5$;
$\sigma = \sqrt5 \approx 2.24$.
\end{solution}

\begin{solution}{exo:g11:stat:10}
\emph{1.} Com $23$: soma $50$, significar $5$; \omterm{def:g11:stat:median}{mediana} = \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de $5$ o e
$6$ os valores $= 3$. Sem $23$: soma $27$, significar $3$; \omterm{def:g11:stat:median}{mediana} (5 de 9
valores) $= 3$.

\emph{2.} Com $23$ ($ n = 10$): $ Q_1$ é o $3$ terceiro valor ($2$), $ Q_3$ o
$8$ o ($4$): \omterm{def:g11:stat:quartiles}{intervalo interquartil} $2$. Sem ($ n = 9$): $ Q_1$ é o
$3$ terceiro valor ($2$), $ Q_3$ o $7$ o ($4$): ainda $2$.

\emph{3.} O único outlier arrasta o significar de $3$ até $5$ mas vai embora
o \omterm{def:g11:stat:median}{mediana} e o \omterm{def:g11:stat:quartiles}{intervalo interquartil} inalterados: esses dois são
\emph{robusto}. Quando um conjunto de dados puder conter valores aberrantes, relate o
\omterm{def:g11:stat:median}{mediana} e \omterm{def:g11:stat:quartiles}{intervalo interquartil} em vez do significar e padrão
desvio.
\end{solution}

\begin{solution}{pb:g11:stat:1}
\textbf{1.} Significar $\bar x = \frac{25}{5} = 5$. Desvios
ao quadrado: $1, 9, 1, 0, 9$: \omterm{def:g11:stat:variance}{variação} $\frac{20}{5} = 4$, então
$\sigma = 2$.

\textbf{2.} Significar de quadrados:
$\frac{16 + 64 + 36 + 25 + 4}{5} = 29$; menos
$\bar x^2 = 25$: \omterm{def:g11:stat:variance}{variação} $4$. Mesma resposta, metade do trabalho.

\textbf{3.} Significar $= \frac{3 \times 8 + 5 \times 10 + 2 \times
14}{10} = 10.2$; \omterm{def:g11:stat:variance}{variação} $= \frac{3 \times 64 + 5 \times 100 +
2 \times 196}{10} - 10.2^2 = 108.4 - 104.04 = 4.36$;
$\sigma \approx 2.09$.

\textbf{4.} Significar: $1.8 \times 20 + 32 = 68\,^\circ $ F;
$\sigma $: $1.8 \times 3 = 5.4\,^\circ $ F --- a mudança $+32$
não espalha nada.

\textbf{5.} Adicionando $ b $: cada desvio $ x_i + b - (\bar x + b)
= x_i - \bar x $ permanece inalterado, então o \omterm{def:g11:stat:variance}{variação} é. Multiplicando
por $ a $: os desvios são multiplicados por $ a $, seus quadrados por
$ a^2$, o \omterm{def:g11:stat:variance}{variação} por $ a^2$, e $\sigma $ por
$\sqrt{a^2} = \abs a $.

\textbf{6.} $ z_A = \frac{15 - 12}{2} = 1.5$;
$ z_B = \frac{13 - 8}{4} = 1.25$. Alice está mais acima dela
multidão: o exame mais difícil $15$ supera os exames mais fáceis $13$.

\textbf{7.} $ z $-pontuações: $-0.5,\ 1.5,\ 0.5,\ 0,\ -1.5$: significar
$0$, \omterm{def:g11:stat:variance}{desvio padrão} $1$. Em geral, a padronização
subtrai $\bar x $ (fazendo o significar $0$, $\sigma $ intocado)
então divide por $\sigma $ (fazendo o novo $\sigma $ igual a
$1$): questão 5 duas vezes.

\textbf{8.} $ z = 2$: $100 + 2 \times 15 = 130$. E
$ z(76) = \frac{76 - 100}{15} = -1.6$.

\textbf{9.} O bebê é $2.5$ desvios padrão abaixo do
significar peso \emph{de bebês da sua idade} --- alarmantemente leve para
sua coorte. Os quilogramas brutos não podem ser comparados entre idades (todos
os bebês ganham peso); pontuações z medir cada criança em relação ao seu
própria \omterm{def:g10:stats:quartiles}{faixa} etária, uma régua para todas as idades.

\textbf{10.} Significar $5$; significar de quadrados $\frac{622}{10} = 62.2$,
\omterm{def:g11:stat:variance}{variação} $37.2$, $\sigma \approx 6.10$;
$ z(23) = \frac{18}{6.10} \approx 2.95$: logo abaixo do alarme
limite --- o teste mal pisca.

\textbf{11.} Sem $23$: significar $3$, \omterm{def:g11:stat:variance}{variação}
$\frac{93}{9} - 9 \approx 1.33$, $\sigma \approx 1.15$;
contra o governante limpo, $23$ pontuações
$ z \approx \frac{20}{1.15} \approx 17$: um monstro. O estranho
inflou tanto o significar e $\sigma $, camuflando-se ---
é por isso \cref{exo:g11:stat:10} recomenda robusto
resumos (\omterm{def:g11:stat:median}{mediana}, AIQ) quando houver ameaça de contaminação.

\textbf{12.} No \omterm{def:g11:stat:variance}{variação} soma
$\sigma^2 = \frac1N \sum (x_i - \bar x)^2$, guarde apenas o
termos com $\abs{x_i - \bar x} \geq k\sigma $: há $ pN $
deles, cada um pelo menos $(k\sigma)^2$, então
$\sigma^2 \geq \frac{1}{N} \cdot pN \cdot k^2\sigma^2 =
p\,k^2\sigma^2$: dividir por $ k^2 \sigma^2$ (spread diferente de zero):
$ p \leq \frac{1}{k^2}$.

\textbf{13.} Além $2\sigma $: no \omterm{def:g10:functions:extrema}{máximo} $\frac14 = 25\,\%$ de
qualquer conjunto de dados; além $3\sigma $: no \omterm{def:g10:functions:extrema}{máximo}
$\frac19 \approx 11\,\%$. Dados da questão 1: além
$2\sigma = 4$ do significar $5$ significa $\leq 1$ ou $\geq 9$:
nenhum dos cinco valores --- $0\,\% \leq 25\,\%$, como prometido.

\textbf{14.} Fora $\intcc{49.7}{50.3}$ significa além
$3\sigma $: no \omterm{def:g10:functions:extrema}{máximo} $\frac19$ de produção, \emph{qualquer que seja}
a forma da distribuição --- Chebyshev não pergunta nada além de uma
significar e um $\sigma $. A produção em forma de sino concentra-se muito
melhor (cerca $0.3\,\%$ além $3\sigma $): a lei normal,
12ª série.

\textbf{15.} $-19\,\%$ senta $24$ pontos abaixo do significar $5$.
Fundo B: $24 = 2\sigma_B $: Chebyshev permite até
$\frac14$ de anos tão ruins. Fundo A: $24 = 12\sigma_A $: em
mais $\frac{1}{144}$. A mesma média, perigos incomparáveis ---
$\sigma $ é o que as finanças chamam \emph{risco} (ou volatilidade).

\textbf{16.} Expandir ao redor $\bar x $:
$(x_i - c)^2 = \left((x_i - \bar x) + (\bar x - c)\right)^2
= (x_i - \bar x)^2 + 2(\bar x - c)(x_i - \bar x) +
(\bar x - c)^2$. Média: o termo médio morre (desvios
do significar soma zero), deixando
$ f(c) = \sigma^2 + (\bar x - c)^2$ --- \omterm{def:g10:functions:extrema}{mínimo} exatamente em
$ c = \bar x $, com \omterm{def:g10:functions:extrema}{mínimo} $\sigma^2$. O significar é o
centro dos \omterm{def:g10:functions:extrema}{mínimos} quadrados, como o \omterm{def:g11:stat:median}{mediana} é o \omterm{def:g10:functions:extrema}{mínimo} absoluto
centro (\cref{pb:g10:stats:1}).

\textbf{17.} Diretamente: desvios de $6$ são
$-2, 2, 0, -1, -4$, quadrados $4, 4, 0, 1, 16$:
$ f(6) = \frac{25}{5} = 5$. Fórmula:
$\sigma^2 + (5 - 6)^2 = 4 + 1 = 5$. Acordo.

\textbf{18.} Ambos têm significar $7$; variações:
$ P $: $\frac{16 + 4 + 0 + 4 + 16}{5} = 8$;
$ Q $: $\frac{36 + 0 + 0 + 0 + 36}{5} = 14.4$: spreads diferentes
atrás de um significar. Para $\{7 - a, 7, 7, 7, 7 + a\}$: \omterm{def:g11:stat:variance}{variação}
$\frac{2a^2}{5} = 8$ dá $ a = \sqrt{20} \approx 4.47$: o
conjunto de dados $\{2.53, 7, 7, 7, 11.47\}$ carrega o exato significar e
\omterm{def:g11:stat:variance}{variação} de $ P $ com um perfil completamente diferente ---
os resumos são compactados e a compactação é perdida.

\textbf{19.} $\sigma^2 = 0$ significa uma soma de quadrados
$(x_i - \bar x)^2$ desaparece; quadrados são $\geq 0$, então um zero
soma força todos os termos a serem zero: $ x_i = \bar x $ para todos $ i $
--- todos os valores iguais.

\textbf{20.} Os centros dizem onde estão os dados; spreads diga como
vagamente; o \omterm{pb:g11:stat:1}{pontuação z} estabelece uma régua em todos os conjuntos de dados,
exames, idades e moedas; Chebyshev se converte $\sigma$ em
uma garantia incondicional; e a pergunta 18 nos mantém honestos
--- nenhum punhado de números contém a história toda. Próximo
ano, a curva do sino aguça a régua em precisão
\omterm{def:g10:proba:distribution}{probabilidades}.
\end{solution}
