\section*{Capítulo \ref{ch:g11:trigo} --- Trigonometria: O Círculo Unitário}

\begin{solution}{exo:g11:trigo:1}
Para \omterm{def:g11:trigo:radian}{radianos}: $30^\circ = \frac{\pi}{6}$, $45^\circ = \frac{\pi}{4}$,
$120^\circ = \frac{2\pi}{3}$, $270^\circ = \frac{3\pi}{2}$.

Em graus: $\frac{\pi}{5} = 36^\circ $, $\frac{5\pi}{6} = 150^\circ $,
$\frac{7\pi}{4} = 315^\circ $, $\frac{2\pi}{9} = 40^\circ $.
\end{solution}

\begin{solution}{exo:g11:trigo:2}
$\frac{2\pi}{3}$ e $-\frac{\pi}{4}$ já estão em
$\intoc{-\pi}{\pi}$: segundo quadrante e quarto quadrante respectivamente.

$\frac{17\pi}{6} - 2\pi = \frac{5\pi}{6}$: segundo quadrante, próximo ao
negativo $ x $-eixo.

$-\frac{7\pi}{2} + 4\pi = \frac{\pi}{2}$: o topo do círculo,
$(0, 1)$.
\end{solution}

\begin{solution}{exo:g11:trigo:3}
$\cos\frac{3\pi}{4} = \cos\left(\pi - \frac\pi4\right)
= -\cos\frac\pi4 = -\frac{\sqrt2}{2}$.

$\sin\frac{5\pi}{6} = \sin\left(\pi - \frac\pi6\right)
= \sin\frac\pi6 = \frac12$.

$\cos\left(-\frac\pi3\right) = \cos\frac\pi3 = \frac12$.

$\sin\frac{4\pi}{3} = \sin\left(\pi + \frac\pi3\right)
= -\sin\frac\pi3 = -\frac{\sqrt3}{2}$.

$\cos\frac{11\pi}{6} = \cos\left(-\frac\pi6 + 2\pi\right)
= \cos\frac\pi6 = \frac{\sqrt3}{2}$.
\end{solution}

\begin{solution}{exo:g11:trigo:4}
$\sin^2 t = 1 - \cos^2 t = 1 - \frac{25}{169} = \frac{144}{169}$, então
$\sin t = \pm\frac{12}{13}$. Sobre $\intoo{-\frac\pi2}{0}$ (quarto quadrante)
o \omterm{def:g10:coordgeom:system}{ordenar} é negativo: $\sin t = -\frac{12}{13}$.
\end{solution}

\begin{solution}{exo:g11:trigo:5}
Usando \cref{prop:g11:trigo:associated}:
\[
A = (-\cos x) + \cos x + \cos x - \cos x = 0 ,
\]
\[
B = \sin x + (-\sin x) + (-\sin x) = -\sin x .
\]
\end{solution}

\begin{solution}{exo:g11:trigo:6}
$\cos t = \frac{\sqrt3}{2}$: uma solução é $\frac\pi6$, então
$ t = \pm\frac\pi6$ (as mudanças por $2k\pi $ deixe o \omterm{def:g10:numbers:interval}{intervalo}).

$\sin t = \frac12$: $ t = \frac\pi6$ ou $ t = \pi - \frac\pi6 =
\frac{5\pi}{6}$.

$\cos t = -1$: o único ponto de \omterm{def:g10:coordgeom:system}{abscissa} $-1$ é $ M(\pi)$, então $ t = \pi $.
\end{solution}

\begin{solution}{exo:g11:trigo:7}
$\sin t = -\frac{\sqrt3}{2}$: de $ a = -\frac\pi3$, as famílias são
$-\frac\pi3 + 2k\pi $ e $\pi + \frac\pi3 + 2k\pi $; em
$\intco{0}{2\pi}$: $ t = \frac{4\pi}{3}$ e $ t = \frac{5\pi}{3}$.

$\cos t = 0$: $ t = \frac\pi2$ e $ t = \frac{3\pi}{2}$.

$2\sin t + 1 = 0$ significa $\sin t = -\frac12$: de $ a = -\frac\pi6$, em
$\intco{0}{2\pi}$: $ t = \frac{7\pi}{6}$ e $ t = \frac{11\pi}{6}$.
\end{solution}

\begin{solution}{exo:g11:trigo:8}
Por \cref{thm:g11:trigo:equations}, $\cos 2t = \cos t $ significa
$2t = t + 2k\pi $ ou $2t = -t + 2k\pi $, ou seja\ $ t = 2k\pi $ ou
$ t = \frac{2k\pi}{3}$ ($ k \in \Z $). A primeira família está contida no
segundo (tomar $ k $ um múltiplo de $3$), então o conjunto solução é
\[
S = \left\{\frac{2k\pi}{3} : k \in \Z\right\}
\]
--- no círculo, os três pontos $ M(0)$,
$ M\!\left(\frac{2\pi}{3}\right)$, $ M\!\left(\frac{4\pi}{3}\right)$.
\end{solution}

\begin{solution}{exo:g11:trigo:9}
\omterm{def:g10:algebra:expand}{Expandindo} ambos os quadrados e usando $\cos^2 x + \sin^2 x = 1$:
\[
(\cos^2 x + 2\sin x\cos x + \sin^2 x)
+ (\cos^2 x - 2\sin x\cos x + \sin^2 x)
= 1 + 1 = 2 .
\]
\end{solution}

\begin{solution}{exo:g11:trigo:10}
Rolar sem escorregar significa que o comprimento do arco é igual à distância
viajou: $ r\theta = 150$ cm dá
$\theta = \frac{150}{30} = 5$ ótimo $= 5 \times \frac{180}{\pi} \approx
286.5^\circ $. Uma volta completa avança a bicicleta pela circunferência
$2\pi \times 30 = 60\pi \approx 188.5$ centímetros, cerca $1.88$ m.
\end{solution}

\begin{solution}{exo:g11:trigo:11}
No círculo unitário, os pontos de \omterm{def:g10:coordgeom:system}{abscissa} exatamente $\frac12$ são
$ M\!\left(\pm\frac\pi3\right)$. Os pontos com \omterm{def:g10:coordgeom:system}{abscissa} \emph{no \omterm{def:g10:functions:extrema}{máximo}}
$\frac12$ formar o arco à esquerda da linha vertical $ x = \frac12$,
viajou como $ t $ vai de $\frac\pi3$ até $\pi $, ou de $-\pi $ até
$-\frac\pi3$. Daí, em $\intoc{-\pi}{\pi}$:
\[
S = \intoc{-\pi}{-\frac{\pi}{3}} \cup \intcc{\frac{\pi}{3}}{\pi}.
\]
(Os pontos finais $\pm\frac\pi3$ estão incluídos porque a desigualdade é ampla.)
\end{solution}

\begin{solution}{pb:g11:trigo:1}
\textbf{1.} $30^\circ = \frac{\pi}{6}$; $135^\circ =
\frac{3\pi}{4}$; $300^\circ = \frac{5\pi}{3}$. E
$\frac{3\pi}{4} = 135^\circ $; $\frac{5\pi}{6} = 150^\circ $.

\textbf{2.} Comprimento $ r\,t = 3 \times 2 = 6$ m. \omterm{def:g11:trigo:radian}{Radianos} fazer arco
comprimento de um produto simples --- graus arrastariam um
$\frac{\pi}{180}$ em cada fórmula.

\textbf{3.} \omterm{def:g11:trigo:radian}{Radianos} virou $= \frac{\text{distance}}
{\text{radius}} = \frac{1000}{0.30} \approx 3\,333$ ótimo;
dividido por $2\pi $: sobre $530$ voltas completas.

\textbf{4.} $\cos\frac{2\pi}{3} = -\frac12$;
$\sin\left(-\frac\pi4\right) = -\frac{\sqrt2}{2}$;
$\frac{19\pi}{6} - 2\pi = \frac{7\pi}{6}$, então
$\cos\frac{19\pi}{6} = \cos\frac{7\pi}{6} =
-\frac{\sqrt3}{2}$.

\textbf{5.} $\cos^2 t = 1 - \frac{9}{25} = \frac{16}{25}$; em
o segundo quadrante o \omterm{def:g11:trigo:cossin}{cosseno} é negativo:
$\cos t = -\frac45$, e
$\tan t = \frac{3/5}{-4/5} = -\frac34$.

\textbf{6.} Em $ t $ minutos a roda gira
$\frac{t}{30} \times 2\pi = \frac{\pi t}{15}$. No $ t = 0$:
$ h = 65 - 60 = 5$ m --- a plataforma de embarque na parte inferior
(altura do cubo menos raio). O sinal de menos coloca você \emph{abaixo}
o cubo quando o ângulo girado é $0$.

\textbf{7.} $ h(7.5) = 65 - 60\cos\frac{\pi}{2} = 65$ m (centro
altura); $ h(15) = 65 + 60 = 125$ m (o topo);
$ h(20) = 65 - 60\cos\frac{4\pi}{3} = 65 + 30 = 95$ m.

\textbf{8.} $65 - 60\cos\frac{\pi t}{15} = 95$ dá
$\cos\frac{\pi t}{15} = -\frac12$: primeira solução
$\frac{\pi t}{15} = \frac{2\pi}{3}$, ou seja\ $ t = 10$ minutos.

\textbf{9.} $\cos\frac{\pi t}{15} \leq -\frac12$ para
$\frac{\pi t}{15} \in \intcc{\frac{2\pi}{3}}{\frac{4\pi}{3}}$,
ou seja\ $ t \in \intcc{10}{20}$: dez minutos de viagem acima
$95$ m, simétrico em relação ao cume em $ t = 15$.

\textbf{10.} Primeiro minuto:
\[
h(1) - h(0) = 60\left(1 - \cos\frac{\pi}{15}\right)
\approx 1.3 \text{ m}.
\]
Entre minutos $7$ e $8$:
\[
60\left(\cos\frac{7\pi}{15} - \cos\frac{8\pi}{15}\right)
\approx 12.5 \text{ m}
\]
--- quase dez vezes mais. A altura muda mais rapidamente ao passar pelo nível do cubo
e para perto da parte inferior e superior: a sinusóide é íngreme em sua
meio, plano em seus extremos.

\textbf{11.} Raio $= \frac{40 - 4}{2} = 18$ m, cubo em
$4 + 18 = 22$ m, ponto final $12$ min:
$ h(t) = 22 - 18\cos\left(\frac{\pi t}{6}\right)$.

\textbf{12.} \omterm{def:g10:functions:extrema}{Máximo} $7 + 3 = 10$ estou quando
$\sin\frac{\pi t}{6} = 1$: primeiro em $ t = 3$ (3h);
\omterm{def:g10:functions:extrema}{mínimo} $4$ sou o primeiro em $ t = 9$; período
$\frac{2\pi}{\pi/6} = 12$ horas.

\textbf{13.} $\sin\frac{\pi t}{6} \geq \frac12$ para
$\frac{\pi t}{6} \in \intcc{\frac\pi6}{\frac{5\pi}{6}}$:
$ t \in \intcc{1}{5}$ --- uma janela de quatro horas da 1h às 5h
da manhã, reabrindo um período depois, $ t \in \intcc{13}{17}$
(13h às 17h).

\textbf{14.} O espelho através do eixo vertical envia o ponto em
ângulo $ t $ ao ponto em ângulo $\pi - t $, preservando o
\omterm{def:g10:coordgeom:system}{ordenar}: $\sin(\pi - t) = \sin t $. A meia volta em torno do
centro envia $ t $ para $\pi + t $, negando ambos \omterm{def:g10:coordgeom:system}{coordenadas}:
$\cos(\pi + t) = -\cos t $. E o espelho na diagonal
trocas \omterm{def:g10:coordgeom:system}{abscissa} e \omterm{def:g10:coordgeom:system}{ordenar}:
$\cos\left(\frac\pi2 - t\right) = \sin t $ --- o
``co'' de ângulos complementares.

\textbf{15.} $\sin(-t) = -\sin t $: ímpar (simetria de meia volta de
é \omterm{def:g10:functions:graph}{gráfico}); $\cos(-t) = \cos t $: par (espelho). E
$\sin t + t^3$ é uma soma de dois funções estranhas: \omterm{def:g11:func:parity}{chance}.

\textbf{16.} Fator: $(2\sin t + 1)(\sin t - 1) = 0$:
$\sin t = 1$ dá $ t = \frac\pi2$;
$\sin t = -\frac12$ dá $ t = \frac{7\pi}{6}$ e
$ t = \frac{11\pi}{6}$. Três soluções no círculo.

\textbf{17.} $1$ ótimo $= \frac{180}{\pi} \approx 57.3^\circ $.
Então $\sin(1) \approx 0.841$ enquanto $\sin(1^\circ) \approx
0.0175$: um fator de quase $50$. Moral: verifique a luz do modo
antes de confiar em qualquer exibição trigonométrica.

\textbf{18.} \omterm{def:g10:coordgeom:system}{Abscissa} é igual \omterm{def:g10:coordgeom:system}{ordenar} na diagonal:
$ t = \frac\pi4$ e $ t = \frac{5\pi}{4}$.

\textbf{19.} $\sin(0.1) = 0.09983\ldots $ contra $0.1$:
erro relativo sobre $0.17\,\%$. Para ângulos pequenos a corda
abraça o arco --- e o limite da 12ª série
$\frac{\sin t}{t} \to 1$ é exatamente esta observação feita
um teorema.

\textbf{20.} \omterm{def:g11:trigo:radian}{Radianos}: ângulo $=$ arco no círculo unitário, então
ângulos calculam comprimentos semelhantes. \omterm{def:g11:trigo:cossin}{Cosseno} e \omterm{def:g11:trigo:cossin}{seno}: o \omterm{def:g10:coordgeom:system}{coordenadas}
do ponteiro do relógio --- sombra no chão, altura no
parede. Simetrias do círculo: todo o catálogo de
identidades, ler espelhos e meias voltas. Equações
$\cos t = c $: o horário do relógio, com o círculo mostrando
todas as soluções de uma só vez. Trajes: a roda usava
$65 - 60\cos $, a maré desgastou $7 + 3\sin$ --- mesmo relógio, mesmo
matemática.
\end{solution}
