\section*{Capítulo \ref{ch:g11:scal} --- O produto escalar no plano}

\begin{solution}{exo:g11:scal:1}
$(2,5)\cdot(3,-1) = 6 - 5 = 1$.

$(1,-4)\cdot(8,2) = 8 - 8 = 0$ (\omterm{def:g11:vect:vector}{vetores} ortogonais).

$\vec u\cdot\vec v = 3 \times 4 \times \cos\frac\pi3 = 12 \times \frac12
= 6$.
\end{solution}

\begin{solution}{exo:g11:scal:2}
$\vect{AB}\cdot\vect{AC} = 6 \times 6 \times \cos 60^\circ = 18$.

O \omterm{def:g11:vect:vector}{vetores} $\vect{AB}$ e $\vect{BC}$ faça um ângulo de
$180^\circ - 60^\circ = 120^\circ $ (coloque-os cauda com cauda em $ B $), então
$\vect{AB}\cdot\vect{BC} = 36 \cos 120^\circ = -18$.
\end{solution}

\begin{solution}{exo:g11:scal:3}
$\vec u \cdot \vec v = 3(t - 8) + t = 4t - 24 = 0$ dá $ t = 6$.

$\vec u - \vec v = (t - 4,\ 0)$, então
$\vec u \cdot (\vec u - \vec v) = t(t-4) + 0 = t^2 - 4t $, que desaparece
para $ t = 0$ ou $ t = 4$.
\end{solution}

\begin{solution}{exo:g11:scal:4}
$\vec u \cdot \vec v = 3 + 2 = 5$, $\norm{\vec u} = \sqrt5$,
$\norm{\vec v} = \sqrt{10}$, então
\[
\cos\theta = \frac{5}{\sqrt5\,\sqrt{10}} = \frac{5}{5\sqrt2}
= \frac{\sqrt2}{2},
\]
por isso $\theta = 45^\circ $ exatamente.
\end{solution}

\begin{solution}{exo:g11:scal:5}
$ a^2 = 16 + 36 - 2 \times 4 \times 6 \times \cos\frac{2\pi}{3}
= 52 - 48 \times \left(-\frac12\right) = 76$, então $ a = \sqrt{76} =
2\sqrt{19} \approx 8.7$.

Para o segundo triângulo, a lei de \omterm{def:g11:trigo:cossin}{cossenos} resolvido para o ângulo dá
\[
\cos\widehat A = \frac{b^2 + c^2 - a^2}{2bc}
= \frac{25 + 16 - 49}{2 \times 5 \times 4} = -\frac{8}{40} = -\frac15,
\]
negativo: $\widehat A $ é obtuso.
\end{solution}

\begin{solution}{exo:g11:scal:6}
$\vect{AB}\,(4, -2)$ e $\vect{AC}\,(2, 4)$:
$\vect{AB}\cdot\vect{AC} = 8 - 8 = 0$, então o ângulo em $ A $ está certo.
Lados: $ AB = \sqrt{16 + 4} = \sqrt{20}$, $ AC = \sqrt{4 + 16} =
\sqrt{20}$, e $\vect{BC}\,(-2, 6)$ dá $ BC = \sqrt{40}$. Consistência:
$ BC^2 = 40 = 20 + 20 = AB^2 + AC^2$, a lei de \omterm{def:g11:trigo:cossin}{cossenos} com
$\cos\widehat A = 0$ (Pitágoras).
\end{solution}

\begin{solution}{exo:g11:scal:7}
Forma centro-raio: $(x + 2)^2 + (y - 3)^2 = 25$.

Para o segundo círculo, complete os quadrados:
\[
x^2 - 6x + y^2 + 2y = 6
\iff (x - 3)^2 - 9 + (y + 1)^2 - 1 = 6
\iff (x-3)^2 + (y+1)^2 = 16 :
\]
centro $(3, -1)$, raio $4$.
\end{solution}

\begin{solution}{exo:g11:scal:8}
$ M(x,y)$ está no círculo quando $\vect{MA}\cdot\vect{MB} = 0$ com
$\vect{MA}\,(-3 - x,\ -y)$ e $\vect{MB}\,(-x,\ 2 - y)$:
\[
(-3 - x)(-x) + (-y)(2 - y) = x^2 + 3x + y^2 - 2y = 0 .
\]
Para a origem: $0 + 0 + 0 - 0 = 0$, então $ O $ encontra-se no círculo ---
geometricamente, $\vect{OA}\,(-3,0)$ e $\vect{OB}\,(0,2)$ são \omterm{def:g11:scal:orthogonal}{ortogonal},
então $ O $ vê $[AB]$ sob um ângulo reto. (Completando os quadrados:
centro $\left(-\frac32, 1\right)$, raio $\frac{\sqrt{13}}{2}$.)
\end{solution}

\begin{solution}{exo:g11:scal:9}
A perpendicular através $ P $ tem o \omterm{def:g11:vect:direction}{vetor de direção} de $ d $, nomeadamente
$(-1, 2)$, normalmente vetor: equação $-x + 2y + c = 0$; através
$ P(1,3)$: $-1 + 6 + c = 0$, $ c = -5$, então $ x - 2y + 5 = 0$.

\omterm{def:g10:numbers:interunion}{Interseção} com $ d $: de $2x + y = 7$, $ y = 7 - 2x $; substituindo,
$ x - 2(7 - 2x) + 5 = 5x - 9 = 0$, então $ x = \frac95$, $ y = \frac{17}5$:
$ H\left(\frac95, \frac{17}5\right)$. Então
$\vect{PH}\left(\frac45, \frac25\right)$ e
\[
PH = \sqrt{\frac{16}{25} + \frac{4}{25}} = \frac{\sqrt{20}}{5}
= \frac{2\sqrt5}{5} \approx 0.89 .
\]
\end{solution}

\begin{solution}{exo:g11:scal:10}
Adicionando
\[
\norm{\vec u + \vec v}^2 = \norm{\vec u}^2 + 2\vec u\cdot\vec v +
\norm{\vec v}^2
\quad\text{and}\quad
\norm{\vec u - \vec v}^2 = \norm{\vec u}^2 - 2\vec u\cdot\vec v +
\norm{\vec v}^2,
\]
os termos cruzados cancelam e
$\norm{\vec u + \vec v}^2 + \norm{\vec u - \vec v}^2 = 2\norm{\vec u}^2 +
2\norm{\vec v}^2$. Em um paralelogramo com lados $\vec u $ e $\vec v $,
as diagonais são $\vec u + \vec v $ e $\vec u - \vec v $: a soma do
quadrados das duas diagonais é igual à soma dos quadrados das quatro
lados.
\end{solution}

\begin{solution}{exo:g11:scal:11}
\emph{1.} Desde $ I $ é o \omterm{prop:g10:coordgeom:midpoint}{ponto médio}, $\vect{IB} = -\vect{IA}$ e
$ IA = 2$. Então
\[
\vect{MA}\cdot\vect{MB}
= (\vect{MI} + \vect{IA})\cdot(\vect{MI} - \vect{IA})
= \norm{\vect{MI}}^2 - \norm{\vect{IA}}^2
= MI^2 - 4 .
\]

\emph{2.} A condição $\vect{MA}\cdot\vect{MB} = 12$ torna-se
$ MI^2 = 16$, ou seja\ $ MI = 4$: o conjunto é o círculo de centro $ I $ e
raio $4$.
\end{solution}

\begin{solution}{pb:g11:scal:1}
\textbf{1.} $\vec u \cdot \vec v = -3 + 8 = 5$;
$\norm{\vec u} = 5$, $\norm{\vec v} = \sqrt5$;
$\cos\theta = \frac{5}{5\sqrt5} = \frac{1}{\sqrt5}$:
$\theta \approx 63.4^\circ $.

\textbf{2.} $6 - 4k = 0$: $ k = \frac32$.

\textbf{3.} $ W = 50 \times 10 \times \cos 60^\circ = 250$
joules: apenas a componente ao longo do movimento funciona.

\textbf{4.} $ c^2 = 25 + 49 - 2 \times 35 \times \frac12 = 39$:
$ c = \sqrt{39}$.

\textbf{5.} $\cos\gamma = \frac{16 + 36 - 64}{2 \times 4
\times 6} = -\frac14$: negativo, o ângulo é obtuso:
$\gamma \approx 104.5^\circ $.

\textbf{6.} \omterm{def:g10:coordgeom:system}{Coordenadas}: $\vec u \cdot \vec v = \cos a \cos b +
\sin a \sin b $. Geometria: duas unidades \omterm{def:g11:vect:vector}{vetores},
$\vec u \cdot \vec v = 1 \times 1 \times \cos(a - b)$.
Igualando os dois cálculos do mesmo número:
$\cos(a - b) = \cos a \cos b + \sin a \sin b $.

\textbf{7.} $\cos(a + b) = \cos(a - (-b)) = \cos a \cos(-b) +
\sin a \sin(-b) = \cos a \cos b - \sin a \sin b $.

\textbf{8.} $\sin(a + b) = \cos\left(\frac\pi2 - a - b\right)
= \cos\left(\left(\frac\pi2 - a\right) - b\right)
= \cos\left(\frac\pi2 - a\right)\cos b +
\sin\left(\frac\pi2 - a\right)\sin b
= \sin a \cos b + \cos a \sin b $.

\textbf{9.} $\cos 15^\circ = \cos 45^\circ \cos 30^\circ +
\sin 45^\circ \sin 30^\circ = \frac{\sqrt2}{2} \cdot
\frac{\sqrt3}{2} + \frac{\sqrt2}{2} \cdot \frac12 =
\frac{\sqrt6 + \sqrt2}{4} \approx 0.966$. E
$\sin 75^\circ = \cos 15^\circ $: mesmo valor.

\textbf{10.} $ b = a $ nas questões 7 e 8:
$\cos 2a = \cos^2 a - \sin^2 a = 2\cos^2 a - 1 = 1 - 2\sin^2 a $
(usando $\cos^2 + \sin^2 = 1$), e
$\sin 2a = 2 \sin a \cos a $.

\textbf{11.} $\sin^2 15^\circ = \frac{1 - \cos 30^\circ}{2}
= \frac{2 - \sqrt3}{4}$, então
$\sin 15^\circ = \frac{\sqrt{2 - \sqrt3}}{2}$. Quadratura
$\frac{\sqrt6 - \sqrt2}{4}$:
$\frac{6 - 2\sqrt{12} + 2}{16} = \frac{8 - 4\sqrt3}{16} =
\frac{2 - \sqrt3}{4}$: o mesmo número em duas fantasias.

\textbf{12.} $\vect{AP}$ se decompõe em uma parte ao longo $ d $
(invisível para $\vec n $) e uma parte perpendicular a $ d $, cujo
comprimento é a distância procurada; pontilhando com a unidade normal
$\frac{\vec n}{\norm{\vec n}}$ mata a primeira parte e
mede o segundo. Com $ A $ sobre $ d $:
$\vect{AP} \cdot \vec n = a x_0 + b y_0 - (a x_A + b y_A) =
a x_0 + b y_0 + c $, de onde vem a fórmula. Exemplo:
$\frac{\abs{15 + 24 - 12}}{5} = \frac{27}{5} = 5.4$.

\textbf{13.} Raio $= 5.4$:
$(x - 5)^2 + (y - 6)^2 = 29.16$.

\textbf{14.} Distância $= \frac{\abs{-12}}{5} = 2.4$. Pé: o
perpendicular à origem é $(3t, 4t)$; na linha:
$9t + 16t = 12$, $ t = \frac{12}{25}$: pé
$\left(\frac{36}{25}, \frac{48}{25}\right)$, à distância
$5t = \frac{60}{25} = 2.4$. A fórmula e o honesto
cálculo concorda.

\textbf{15.} Base $ AB = 6$, altura $=$ distância de $ C $ para
$(AB)$ (o eixo $ y = 0$): $5$; área $15$. Linha $(AC)$:
$5x - 2y = 0$, e
$\operatorname{dist}(B, (AC)) = \frac{\abs{30}}{\sqrt{29}}$;
então $\frac12 \times AC \times \frac{30}{\sqrt{29}} =
\frac12 \times \sqrt{29} \times \frac{30}{\sqrt{29}} = 15$:
mesma área, em qualquer altitude.

\textbf{16.} $\operatorname{dist} = \frac{\abs{2 + 1 - 6}}
{\sqrt2} = \frac{3}{\sqrt2} \approx 2.12 > 2$: a linha passa
claro do círculo --- disjunto.

\textbf{17.} Inserir $ I $:
$\vect{MA} \cdot \vect{MB} = (\vect{MI} + \vect{IA}) \cdot
(\vect{MI} + \vect{IB}) = MI^2 + \vect{MI} \cdot (\vect{IA} +
\vect{IB}) + \vect{IA} \cdot \vect{IB}$. O meio termo morre
($\vect{IA} + \vect{IB} = \vec 0$), e
$\vect{IA} \cdot \vect{IB} = -IA^2 = -\frac{AB^2}{4}$
(oposto \omterm{def:g11:vect:vector}{vetores} de comprimento $\frac{AB}2$).

\textbf{18.} $\vect{MA} \cdot \vect{MB} = 0 \iff MI^2 =
\frac{AB^2}{4} \iff MI = \frac{AB}{2}$: o círculo do centro
$ I $ e raio $\frac{AB}2$ --- o círculo com diâmetro
$[AB]$.

\textbf{19.} \omterm{def:g10:algebra:expand}{Expandindo} cada par de $ H $ ($\vect{BC} =
\vect{HC} - \vect{HB}$, etc.):
\[
\vect{HA} \cdot \vect{HC} - \vect{HA} \cdot \vect{HB}
+ \vect{HB} \cdot \vect{HA} - \vect{HB} \cdot \vect{HC}
+ \vect{HC} \cdot \vect{HB} - \vect{HC} \cdot \vect{HA} = 0 :
\]
tudo se cancela em pares --- a identidade vale para
\emph{qualquer} quatro pontos. Para o nosso $ H $, os dois primeiros pontos
produtos desaparecem por hipótese, então
$\vect{HC} \cdot \vect{AB} = 0$: $ H $ fica na altitude de
$ C$. Três altitudes, um ponto.

\textbf{20.} As fórmulas de adição vieram do cálculo de um ponto
produto com o \emph{coordenada} cara e o
\emph{\omterm{def:g11:scal:dot}{normas} e ângulo} face; a fórmula da distância do
\emph{projeção} face; o \omterm{pb:g11:scal:1}{ortocentro} do
\emph{bilinearidade} (Chasles-e-expandir) rosto. Calcule o mesmo
quantidade duas vezes, iguale, e um teorema aparece --- o escalar
produtoé todo o comércio.
\end{solution}
