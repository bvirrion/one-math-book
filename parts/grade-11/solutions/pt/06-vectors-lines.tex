\begin{solution}{exo:g11:vect:1}
$\vect{AB}\,(3, 4)$, $\vect{AC}\,(-3, 2)$,
$\vect{AB} + \vect{AC}\,(0, 6)$,
$2\vect{AB} - 3\vect{AC}\,(6 + 9,\ 8 - 6) = (15, 2)$. O \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de
$[BC]$ é $\left(\frac{5 + (-1)}{2}, \frac{3+1}{2}\right) = (2, 2)$.
\end{solution}

\begin{solution}{exo:g11:vect:2}
$\det = 4 \times 3 - (-2) \times (-6) = 12 - 12 = 0$: \omterm{def:g11:vect:collinear}{colinear} (de fato
$\vec v = -\frac12 \vec u $).

$\det = 2 \times 7 - 3 \times 5 = -1 \neq 0$: não \omterm{def:g11:vect:collinear}{colinear}.

$\det = 0 \times (-8) - 0 \times 3 = 0$: \omterm{def:g11:vect:collinear}{colinear} (ambos verticais).
\end{solution}

\begin{solution}{exo:g11:vect:3}
Através $ A(2,1)$, direção $(3, -1)$: correspondência $(-b, a) = (3, -1)$ dá
$ a = -1$, $ b = -3$, então $-x - 3y + c = 0$; conectando $ A $: $-2 - 3 + c =
0$, $ c = 5$. Multiplicando por $-1$: $ x + 3y - 5 = 0$.

Através $ B(0,4)$ e $ C(2,0)$: $\vect{BC}\,(2, -4)$, então $ a = -4$,
$ b = -2$: $-4x - 2y + c = 0$; conectando $ B $: $-8 + c = 0$, $ c = 8$.
Simplificando por $-2$: $2x + y - 4 = 0$. Verifique com $ C $: $4 + 0 - 4 = 0$.
\end{solution}

\begin{solution}{exo:g11:vect:4}
UM \omterm{def:g11:vect:direction}{vetor de direção} é $(-b, a) = (2, 3)$. Interseções: $ x = 0$ dá
$ y = 3$, apontar $(0, 3)$; $ y = 0$ dá $ x = -2$, apontar $(-2, 0)$. Para
$ P(2,6)$: $3 \times 2 - 2 \times 6 + 6 = 0$, então $ P \in d $. Para $ Q(1,4)$:
$3 - 8 + 6 = 1 \neq 0$, então $ Q \notin d $.
\end{solution}

\begin{solution}{exo:g11:vect:5}
$\vect{AB}\,(3, 6)$ e $\vect{AC}\,(-3, -6)$:
$\det = 3 \times (-6) - (-3) \times 6 = 0$, então $ A $, $ B $, $ C $ estão alinhados.

$\vect{DE}\,(2, 3)$ e $\vect{DF}\,(5, 8)$:
$\det = 2 \times 8 - 5 \times 3 = 1 \neq 0$: $ D $, $ E $, $ F $ não são
alinhado.
\end{solution}

\begin{solution}{exo:g11:vect:6}
Primeiro par: somando os dois equações, $(2x + y - 7) + (x - y + 1) =
3x - 6 = 0$, então $ x = 2$, então $ y = x + 1 = 3$: eles se encontram em $(2, 3)$
(verificar: $4 + 3 - 7 = 0$).

Segundo par: $ ab' - a'b = 1 \times 4 - (-2) \times (-2) = 0$, então o
as linhas são paralelas; multiplicando $ e $ por $-2$ dá $-2x + 4y - 6 = 0$,
que difere de $ e'$ ($-6 \neq 1$): estritamente paralelo, não
\omterm{def:g10:numbers:interunion}{interseção}.
\end{solution}

\begin{solution}{exo:g11:vect:7}
Para $ d: 2x - y + 3 = 0$: direção $(-b, a) = (1, 2)$ e o ponto
$(0, 3)$ mente $ d $, então $ d = \{(t,\ 3 + 2t) : t \in \R\}$.

Para a linha paramétrica através $(1, -3)$ com direção $(2, 1)$:
$ a = 1$, $ b = -2$, então $ x - 2y + c = 0$; conectando $(1, -3)$:
$1 + 6 + c = 0$, $ c = -7$. Equação: $ x - 2y - 7 = 0$ (verificar $ t = 1$,
apontar $(3, -2)$: $3 + 4 - 7 = 0$).
\end{solution}

\begin{solution}{exo:g11:vect:8}
Uma linha paralela tem o mesmo $(a, b)$: $4x - 3y + c = 0$; através
$ P(1, -2)$: $4 + 6 + c = 0$, então $ c = -10$ e a linha é
$4x - 3y - 10 = 0$.

Paralelismo de $ mx + 2y - 5 = 0$ com $ d $ requer
$ ab' - a'b = 4 \times 2 - m \times (-3) = 8 + 3m = 0$, então
$ m = -\frac83$.
\end{solution}

\begin{solution}{exo:g11:vect:9}
O \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de $[AB]$ é $ M\!\left(\frac{-1+3}{2}, \frac{0+2}{2}\right)
= (1, 1)$. O \omterm{def:g10:stats:median}{mediana} é a linha $(CM)$ com $ C(1, -4)$: ambos os pontos
tem \omterm{def:g10:coordgeom:system}{abscissa} $1$, então o \omterm{def:g10:stats:median}{mediana} é a linha vertical
\[
x - 1 = 0 .
\]
\end{solution}

\begin{solution}{exo:g11:vect:10}
$\det(\vec u, \vec v) = m(m - 1) - 3 \times 2 = m^2 - m - 6$. Ele desaparece
quando $ m^2 - m - 6 = 0$: $\Delta = 1 + 24 = 25$, raízes
$ m = \frac{1 \pm 5}{2}$, ou seja\ $ m = 3$ ou $ m = -2$.
\end{solution}

\begin{solution}{exo:g11:vect:11}
\emph{1.} $ I = \left(\frac12, 0\right)$. Então
$\vect{DI} = \left(\frac12 - 0,\ 0 - 1\right) = \left(\frac12, -1\right)$
e $\vect{DJ} = \frac23\vect{DI} = \left(\frac13, -\frac23\right)$, então
\[
J = D + \vect{DJ} = \left(\frac13,\ \frac13\right).
\]

\emph{2.} $\vect{AJ}\left(\frac13, \frac13\right)$ e
$\vect{AC}\,(1, 1)$:
$\det = \frac13 \times 1 - 1 \times \frac13 = 0$, então $ A $, $ J $, $ C $ são
alinhado --- na verdade $\vect{AJ} = \frac13\vect{AC}$: a linha $(DI)$ cortes
a diagonal $(AC)$ exatamente a um terço do seu comprimento $ A $.
\end{solution}

\begin{solution}{pb:g11:vect:1}
\textbf{1.} $ x = 1 + 3t $, $ y = 2 - t $. A partir do segundo,
$ t = 2 - y $; substituindo: $ x = 1 + 3(2 - y) = 7 - 3y $:
Cartesiano equação $ x + 3y - 7 = 0$.

\textbf{2.} Para $ ax + by + c = 0$, um \omterm{def:g11:vect:direction}{vetor de direção} é
$(-b, a)$: aqui $(5, 2)$. Um ponto: $ y = 1$ dá $ x = 1$:
$(1, 1)$. Paramétrico: $ x = 1 + 5t $, $ y = 1 + 2t $.

\textbf{3.} De $ x = 7 - 3y $ em $2x - 5y + 3 = 0$:
$14 - 6y - 5y + 3 = 0$, então $ y = \frac{17}{11}$ e
$ x = 7 - \frac{51}{11} = \frac{26}{11}$: o ponto
$\left(\frac{26}{11}, \frac{17}{11}\right)$.

\textbf{4.} Instruções $(5, 2)$ e $(3, -1)$:
$\det = 5 \times (-1) - 2 \times 3 = -11 \neq 0$: cruzando,
conforme encontrado. Para o segundo par: direções $(5, 2)$ e
$(10, 4)$ ter $\det = 0$ (paralelo) e duplicando
$2x - 5y + 3 = 0$ dá $-4x + 10y - 6 = 0 \neq -4x + 10y + 1
= 0$: estritamente paralelo.

\textbf{5.} $\vect{AB}(3, 2)$, $\vect{AC}(9, 6)$:
$\det = 18 - 18 = 0$: alinhado.

\textbf{6.} $ P $: de $ x = 2t $, $ y = 3 + t $: $ y = 3 +
\frac x2$. $ Q $: de $ x = 10 - t $: $ t = 10 - x $, então
$ y = 2(10 - x) - 1 = 19 - 2x $. Cruzamento:
$3 + \frac x2 = 19 - 2x $ dá $ x = 6.4$, $ y = 6.2$: os caminhos
cruzar em $(6.4,\ 6.2)$.

\textbf{7.} $ P $ passa lá quando $2t = 6.4$: $ t = 3.2$ h. $ Q $
quando $10 - t = 6.4$: $ t = 3.6$ h. Vinte e quatro minutos de \omterm{def:g10:numbers:interval}{intervalo}: não
colisão. Aviso: um \omterm{def:g10:functions:graph}{gráfico} mostra \emph{caminhos}; apenas o
relógios paramétricos dizem se dois movimentos ocupam o mesmo ponto
no mesmo instante.

\textbf{8.} $ D^2(t) = (3t - 10)^2 + (4 - t)^2 =
10t^2 - 68t + 116$: vértice em $ t = \frac{68}{20} = 3.4$ h,
onde $ D^2 = 0.4$: abordagem mais próxima
$ D = \sqrt{0.4} \approx 0.63$ milha em $ t = 3.4$.

\textbf{9.} Violado: $0.63 < 1$. $ D^2(t) < 1$ lê
$10t^2 - 68t + 115 < 0$: $\Delta = 4624 - 4600 = 24$, raízes
$\frac{68 \pm \sqrt{24}}{20} = 3.4 \pm 0.245$: os navios são
muito perto de $ t \approx 3.16$ para $ t \approx 3.64$ --- sobre
$29$ minutos de infração. Um deles deve alterar o rumo.

\textbf{10.} Cada coordenada de cada navio é um \emph{afim}
\omterm{def:g11:func:function}{função} de $ t $, então cada diferença de \omterm{def:g10:coordgeom:system}{coordenadas} é afim,
e $ D^2$ --- uma soma de dois ao quadrado funções afins --- é um
quadrático (com coeficiente líder não negativo). Mais próximo
abordagem é, portanto, sempre lida em um vértice.

\textbf{11.} Reunião $ P $ no $ t = 3$, ou seja,\ em $(6, 6)$:
$(3a, 3b) = (6, 6)$ dá $(a, b) = (2, 2)$, velocidade
$\sqrt{8} \approx 2.8$ nós.

\textbf{12.} $ D(0, 1)$, $ I\left(\frac12, 0\right)$: direção
$\left(\frac12, -1\right)$, ou seja\ $(1, -2)$: equação
$2x + y = 1$.

\textbf{13.} $(AC)$: $ y = x $. Então $2x + x = 1$:
$ x = \frac13$: o ponto $\left(\frac13, \frac13\right)$, um
terço do caminho de $ A $ para $ C $.

\textbf{14.} $ B(1, 0)$, $ K\left(\frac12, 1\right)$: direção
$\left(-\frac12, 1\right)$, equação $2x + y = 2$. Com
$ y = x $: $ x = \frac23$: o ponto
$\left(\frac23, \frac23\right)$. As duas cevianas cortadas $(AC)$ em
seus dois pontos de trissecção: teorema provado, duas vezes três linhas
de \omterm{def:g11:seq:arithmetic}{aritmética}.

\textbf{15.} Qualquer paralelogramo é transportado para o quadrado unitário
escolhendo $ A $ como origem e $\vect{AB}$, $\vect{AD}$ como o
unidades de dois eixos. Essa escolha preserva tudo o teorema
fala de --- pontos médios, alinhamento e proporções ao longo de um determinado
linha (todas expressáveis com \omterm{def:g11:vect:vector}{vetores} e escalares) --- então provando
a afirmação no quadrado prova isso para cada paralelogramo.
(Seria \emph{não} comprimentos ou ângulos de transporte: aqueles que
quadro distorce.)

\textbf{16.} $ E\left(1, \frac12\right)$: linha $(AE)$:
$ y = \frac x2$. Diagonal $(DB)$: de $(0,1)$ para $(1,0)$:
$ y = 1 - x $. \omterm{def:g10:numbers:interunion}{Interseção}: $\frac x2 = 1 - x $: $ x = \frac23$:
o ponto $\left(\frac23, \frac13\right)$ --- novamente um
ponto de trissecção, agora da outra diagonal. \omterm{prop:g10:coordgeom:midpoint}{Ponto médio} cevianas
amo terços.

\textbf{17.} Os dois primeiros: $ x + y = 4$ e $2x - y = 2$ dar
$3x = 6$: $(2, 2)$. Terceiro: $2 - 4 = -2$: satisfeito. Simultâneo
em $(2, 2)$.

\textbf{18.} $ mx - y + 2 - 3m = m(x - 3) + (2 - y) = 0$: para
o equação segurar por \emph{todo} $ m $, pegar $ x = 3$,
$ y = 2$ --- e de fato cada $ d_m $ contém $(3, 2)$. Um lápis
de linhas que passam por um ponto, uma linha por declive $ m $.

\textbf{19.} Paralelo a $ y = 2x + 7$: declive $ m = 2$, a linha
$2x - y - 4 = 0$. Através da origem: $2 - 3m = 0$:
$ m = \frac23$.

\textbf{20.} Cartesiano: conecte um ponto, adesão respondida.
Reduzido: declive e interceptar visível, \omterm{def:g10:functions:graph}{gráfico} imediato.
Paramétrico: o relógio integrado --- colisões, mais próximo
abordagem, interceptação. O \omterm{def:g11:vect:det}{determinante}: uma multiplicação
cruzada, e o paralelismo é decidido. O quadro escolhido: um
paralelogramo torna-se um quadrado unitário e uma trissecção clássica
teorema torna-se três cruzamentos de linhas.
\end{solution}
