\section*{Hoofdstuk \ref{ch:g11:prob} --- Kansrekening en stochastische variabelen}

\begin{solution}{exo:g11:prob:1}
\omterm{def:g11:prob:model}{Uniform model} op $52$ kaarten. Harten: $\frac{13}{52} = \frac14$. Heren:
$\frac{4}{52} = \frac{1}{13}$. Harten of heer:
$\frac{13 + 4 - 1}{52} = \frac{16}{52} = \frac{4}{13}$ (de hartenheer
wordt één keer geteld). Geen van beide: $1 - \frac{4}{13} = \frac{9}{13}$.
\end{solution}

\begin{solution}{exo:g11:prob:2}
$S = 6$: de paren $(1,5), (2,4), (3,3), (4,2), (5,1)$, dus
$\P(S = 6) = \frac{5}{36}$.

$S \geq 10$: drie paren voor $10$, twee voor $11$, één voor $12$:
$\P(S \geq 10) = \frac{6}{36} = \frac16$.

$S$ even: sommen $2, 4, 6, 8, 10, 12$ hebben $1 + 3 + 5 + 5 + 3 + 1 = 18$
paren: \omterm{def:g11:prob:model}{kans} $\frac{18}{36} = \frac12$.
\end{solution}

\begin{solution}{exo:g11:prob:3}
De kansen sommen tot $1$: $p = 1 - 0.3 - 0.2 - 0.4 = 0.1$. Dan
\[
\E(X) = 0.3 \times (-1) + 0.2 \times 0 + 0.4 \times 2 + 0.1 \times 5
= -0.3 + 0.8 + 0.5 = 1 .
\]
\end{solution}

\begin{solution}{exo:g11:prob:4}
$\E(Y) = 2 \times 3 - 5 = 1$; $\V(Y) = 2^2 \times 4 = 16$;
$\sigma(Y) = 4$.
\end{solution}

\begin{solution}{exo:g11:prob:5}
De winsten zijn $0$, $2$, $5$ euro met kansen $0.5$, $0.3$, $0.2$;
na aftrek van de $2$-euro inzet neemt $G$ de waarden $-2, 0, 3$ aan:
\begin{center}
\begin{tabular}{c|ccc}
$g$ & $-2$ & $0$ & $3$\\
\hline
$\P(G=g)$ & $0.5$ & $0.3$ & $0.2$
\end{tabular}
\end{center}
$\E(G) = -1 + 0 + 0.6 = -0.4$: de speler verliest $40$ cent per spel
gemiddeld --- ongunstig.
\end{solution}

\begin{solution}{exo:g11:prob:6}
Trekking in volgorde: $\P(X = 2) = \frac35 \times \frac24 = \frac{3}{10}$
(rood dan rood); $\P(X = 0) = \frac25 \times \frac14 = \frac{1}{10}$
(blauw dan blauw); dus
$\P(X = 1) = 1 - \frac{3}{10} - \frac{1}{10} = \frac{6}{10}$. Dan
\[
\E(X) = 0 \times \frac1{10} + 1 \times \frac6{10} + 2 \times \frac3{10}
= \frac{12}{10} = 1.2 ,
\]
consistent met de intuïtie: $\frac35$ van de twee trekkingen is rood
gemiddeld.
\end{solution}

\begin{solution}{exo:g11:prob:7}
$\E(G^2) = 0.5 \times 4 + 0.3 \times 0 + 0.2 \times 9 = 3.8$, dus met de
snelle formule
$\V(G) = 3.8 - (-0.4)^2 = 3.8 - 0.16 = 3.64$ en
$\sigma(G) \approx 1.9$ euro.
\end{solution}

\begin{solution}{exo:g11:prob:8}
Het voordeel is $80$ minus de claim:
\begin{center}
\begin{tabular}{c|ccc}
$b$ & $80$ & $-920$ & $-4920$\\
\hline
$\P(B=b)$ & $0.94$ & $0.05$ & $0.01$
\end{tabular}
\end{center}
$\E(B) = 75.2 - 46 - 49.2 = -20$: tegen deze prijs \emph{verliest} de
maatschappij $20$ euro per polis gemiddeld. De premie is te laag voor de
gedekte risico's (ze zou $100$ euro moeten overschrijden voor een
positief verwacht voordeel).
\end{solution}

\begin{solution}{exo:g11:prob:9}
De stochastische score is $+1$ met \omterm{def:g11:prob:model}{kans} $\frac14$ en $x$ met \omterm{def:g11:prob:model}{kans}
$\frac34$:
\[
\E = \frac14 + \frac{3x}{4} = 0 \iff x = -\frac13 .
\]
Met de straf $-\frac13$ brengt willekeurig raden gemiddeld niets op.
\end{solution}

\begin{solution}{exo:g11:prob:10}
Twee munten: ze matchen (KK of MM) met \omterm{def:g11:prob:model}{kans} $\frac24 \times 2 =
\frac12$. De winst van de speler is $+1$ of $-1$ met gelijke kansen:
\omterm{def:g11:prob:expectation}{verwachting} $0$ voor beide spelers --- het spel is eerlijk.

Drie munten: alle matchen (KKK of MMM) met \omterm{def:g11:prob:model}{kans} $\frac{2}{8} =
\frac14$. De speler ontvangt $2$ met \omterm{def:g11:prob:model}{kans} $\frac14$ en betaalt $1$ met
\omterm{def:g11:prob:model}{kans} $\frac34$:
$\E = \frac24 - \frac34 = -\frac14$. Het spel begunstigt Ann met
$25$ cent per ronde; het zou eerlijk zijn als ze $3$ euro betaalde voor
een driedubbele match.
\end{solution}

\begin{solution}{exo:g11:prob:11}
\emph{1.} De organisator ontvangt $1000 \times 5 = 5000$ euro en keert
$2000 + 5 \times 200 + 20 \times 25 = 3500$ euro uit in totaal: de
winst is de \emph{constante} $1500$ (geen toeval zodra alle loten
verkocht zijn). De winst $G$ van één speler (prijs minus het $5$-euro
lot):
\begin{center}
\begin{tabular}{c|cccc}
$g$ & $1995$ & $195$ & $20$ & $-5$\\
\hline\rule{0pt}{11pt}%
$\P(G=g)$ & $\dfrac{1}{1000}$ & $\dfrac{5}{1000}$ & $\dfrac{20}{1000}$
& $\dfrac{974}{1000}$
\end{tabular}
\end{center}

\emph{2.}
\[
\E(G) = \frac{1995 + 5 \times 195 + 20 \times 20 - 974 \times 5}{1000}
= \frac{1995 + 975 + 400 - 4870}{1000} = -1.5 .
\]
Elke speler verwacht $1.50$ euro te verliezen; over $1000$ spelers is dat
$1500$ euro --- precies de winst van de organisator. De boeken kloppen:
wat de spelers gemiddeld verliezen is precies wat de organisator
opstrijkt.
\end{solution}
