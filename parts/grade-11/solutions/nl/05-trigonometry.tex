\section*{Hoofdstuk \ref{ch:g11:trigo} --- Trigonometrie: de eenheidscirkel}

\begin{solution}{exo:g11:trigo:1}
Naar radialen: $30^\circ = \frac{\pi}{6}$, $45^\circ = \frac{\pi}{4}$,
$120^\circ = \frac{2\pi}{3}$, $270^\circ = \frac{3\pi}{2}$.

Naar graden: $\frac{\pi}{5} = 36^\circ$, $\frac{5\pi}{6} = 150^\circ$,
$\frac{7\pi}{4} = 315^\circ$, $\frac{2\pi}{9} = 40^\circ$.
\end{solution}

\begin{solution}{exo:g11:trigo:2}
$\frac{2\pi}{3}$ en $-\frac{\pi}{4}$ liggen al in
$\intoc{-\pi}{\pi}$: tweede kwadrant respectievelijk vierde kwadrant.

$\frac{17\pi}{6} - 2\pi = \frac{5\pi}{6}$: tweede kwadrant, dicht bij de
negatieve $x$-as.

$-\frac{7\pi}{2} + 4\pi = \frac{\pi}{2}$: de top van de cirkel,
$(0, 1)$.
\end{solution}

\begin{solution}{exo:g11:trigo:3}
$\cos\frac{3\pi}{4} = \cos\left(\pi - \frac\pi4\right)
= -\cos\frac\pi4 = -\frac{\sqrt2}{2}$.

$\sin\frac{5\pi}{6} = \sin\left(\pi - \frac\pi6\right)
= \sin\frac\pi6 = \frac12$.

$\cos\left(-\frac\pi3\right) = \cos\frac\pi3 = \frac12$.

$\sin\frac{4\pi}{3} = \sin\left(\pi + \frac\pi3\right)
= -\sin\frac\pi3 = -\frac{\sqrt3}{2}$.

$\cos\frac{11\pi}{6} = \cos\left(-\frac\pi6 + 2\pi\right)
= \cos\frac\pi6 = \frac{\sqrt3}{2}$.
\end{solution}

\begin{solution}{exo:g11:trigo:4}
$\sin^2 t = 1 - \cos^2 t = 1 - \frac{25}{169} = \frac{144}{169}$, dus
$\sin t = \pm\frac{12}{13}$. Op $\intoo{-\frac\pi2}{0}$ (vierde kwadrant)
is de ordinaat negatief: $\sin t = -\frac{12}{13}$.
\end{solution}

\begin{solution}{exo:g11:trigo:5}
Met \cref{prop:g11:trigo:associated}:
\[
A = (-\cos x) + \cos x + \cos x - \cos x = 0 ,
\]
\[
B = \sin x + (-\sin x) + (-\sin x) = -\sin x .
\]
\end{solution}

\begin{solution}{exo:g11:trigo:6}
$\cos t = \frac{\sqrt3}{2}$: één oplossing is $\frac\pi6$, dus
$t = \pm\frac\pi6$ (de verschuivingen met $2k\pi$ verlaten het
interval).

$\sin t = \frac12$: $t = \frac\pi6$ of $t = \pi - \frac\pi6 =
\frac{5\pi}{6}$.

$\cos t = -1$: het enige punt met abscis $-1$ is $M(\pi)$, dus $t = \pi$.
\end{solution}

\begin{solution}{exo:g11:trigo:7}
$\sin t = -\frac{\sqrt3}{2}$: vanaf $a = -\frac\pi3$ zijn de families
$-\frac\pi3 + 2k\pi$ en $\pi + \frac\pi3 + 2k\pi$; in
$\intco{0}{2\pi}$: $t = \frac{4\pi}{3}$ en $t = \frac{5\pi}{3}$.

$\cos t = 0$: $t = \frac\pi2$ en $t = \frac{3\pi}{2}$.

$2\sin t + 1 = 0$ betekent $\sin t = -\frac12$: vanaf $a = -\frac\pi6$, in
$\intco{0}{2\pi}$: $t = \frac{7\pi}{6}$ en $t = \frac{11\pi}{6}$.
\end{solution}

\begin{solution}{exo:g11:trigo:8}
Volgens \cref{thm:g11:trigo:equations} betekent $\cos 2t = \cos t$ dat
$2t = t + 2k\pi$ of $2t = -t + 2k\pi$, d.w.z.\ $t = 2k\pi$ of
$t = \frac{2k\pi}{3}$ ($k \in \Z$). De eerste familie is bevat in de
tweede (neem $k$ een veelvoud van $3$), dus de oplossingsverzameling is
\[
S = \left\{\frac{2k\pi}{3} : k \in \Z\right\}
\]
--- op de cirkel de drie punten $M(0)$,
$M\!\left(\frac{2\pi}{3}\right)$, $M\!\left(\frac{4\pi}{3}\right)$.
\end{solution}

\begin{solution}{exo:g11:trigo:9}
Beide kwadraten ontwikkelen en $\cos^2 x + \sin^2 x = 1$ gebruiken:
\[
(\cos^2 x + 2\sin x\cos x + \sin^2 x)
+ (\cos^2 x - 2\sin x\cos x + \sin^2 x)
= 1 + 1 = 2 .
\]
\end{solution}

\begin{solution}{exo:g11:trigo:10}
Rollen zonder slippen betekent dat de booglengte gelijk is aan de
afgelegde afstand: $r\theta = 150$ cm geeft
$\theta = \frac{150}{30} = 5$ rad $= 5 \times \frac{180}{\pi} \approx
286.5^\circ$. Eén volledige omwenteling brengt de fiets vooruit over de
omtrek $2\pi \times 30 = 60\pi \approx 188.5$ cm, ongeveer $1.88$ m.
\end{solution}

\begin{solution}{exo:g11:trigo:11}
Op de eenheidscirkel zijn de punten met abscis precies $\frac12$
$M\!\left(\pm\frac\pi3\right)$. De punten met abscis \emph{ten hoogste}
$\frac12$ vormen de boog links van de verticale lijn $x = \frac12$,
doorlopen als $t$ van $\frac\pi3$ tot $\pi$ gaat, of van $-\pi$ tot
$-\frac\pi3$. Dus, op $\intoc{-\pi}{\pi}$:
\[
S = \intoc{-\pi}{-\frac{\pi}{3}} \cup \intcc{\frac{\pi}{3}}{\pi}.
\]
(De eindpunten $\pm\frac\pi3$ zijn inbegrepen omdat de ongelijkheid
breed is.)
\end{solution}
