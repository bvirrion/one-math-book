\section*{Hoofdstuk \ref{ch:g11:quad} --- Kwadratische functies en vergelijkingen}

\begin{solution}{exo:g11:quad:1}
$x^2-5x+6$: $\Delta = 25 - 24 = 1$, wortels $\frac{5 \pm 1}{2}$, d.w.z.\
$2$ en $3$.

$2x^2+3x-2$: $\Delta = 9 + 16 = 25$, wortels $\frac{-3 \pm 5}{4}$, d.w.z.\
$\frac12$ en $-2$.

$x^2+2x+5$: $\Delta = 4 - 20 = -16 < 0$, geen reĂ«le oplossing.

$9x^2-6x+1$: $\Delta = 36 - 36 = 0$, Ă©Ă©n (dubbele) wortel
$x_0 = \frac{6}{18} = \frac13$.
\end{solution}

\begin{solution}{exo:g11:quad:2}
$f(x) = x^2 - 6x + 11 = (x-3)^2 - 9 + 11 = (x-3)^2 + 2$: top $(3, 2)$.

$g(x) = -2x^2+4x+1 = -2(x^2 - 2x) + 1 = -2\bigl((x-1)^2 - 1\bigr) + 1
= -2(x-1)^2 + 3$: top $(1, 3)$.

$h(x) = 3x^2 + 6x = 3(x^2 + 2x) = 3\bigl((x+1)^2 - 1\bigr)
= 3(x+1)^2 - 3$: top $(-1, -3)$.
\end{solution}

\begin{solution}{exo:g11:quad:3}
$x^2 - 3x - 10$: $\Delta = 9 + 40 = 49$, wortels $\frac{3 \pm 7}{2} = -2$
en $5$; positieve leidende coĂ«fficiĂ«nt, dus de trinomiaal is strikt
negatief tussen de wortels: de oplossingsverzameling is
$\intoo{-2}{5}$.

$2x^2 + x + 3$: $\Delta = 1 - 24 = -23 < 0$ en $a = 2 > 0$: altijd
positief, de oplossingsverzameling is $\R$.

$-x^2 + 6x - 9 = -(x-3)^2$: ze is $\geq 0$ alleen waar $(x-3)^2 \leq 0$,
d.w.z.\ bij $x = 3$. De oplossingsverzameling is $\{3\}$.
\end{solution}

\begin{solution}{exo:g11:quad:4}
$x^2 - 7x + 6$: $1 - 7 + 6 = 0$, dus $1$ is een wortel; het product van
de wortels is $6$, dus de andere wortel is $6$.

$2x^2 + 5x - 7$: $2 + 5 - 7 = 0$, dus $1$ is een wortel; het product is
$\frac{c}{a} = -\frac72$, dus de andere wortel is $-\frac72$.

$x^2 + 4x - 5$: $1 + 4 - 5 = 0$, dus $1$ is een wortel; het product is
$-5$, dus de andere wortel is $-5$.
\end{solution}

\begin{solution}{exo:g11:quad:5}
De twee getallen lossen $x^2 - 14x + 45 = 0$ op
(\cref{prop:g11:quad:vieta}); $\Delta = 196 - 180 = 16$, wortels
$\frac{14 \pm 4}{2} = 9$ en $5$.

Voor de rechthoek: half de omtrek geeft lengte $+$ breedte $= 17$, en de
oppervlakte geeft hun product $60$. Ze lossen $x^2 - 17x + 60 = 0$ op;
$\Delta = 289 - 240 = 49$, wortels $\frac{17 \pm 7}{2} = 12$ en $5$. De
rechthoek is $12 \times 5$.
\end{solution}

\begin{solution}{exo:g11:quad:6}
$\Delta = (m-2)^2 - 4$. Precies Ă©Ă©n oplossing wanneer $\Delta = 0$:
$(m-2)^2 = 4$, dus $m - 2 = \pm 2$, d.w.z.\ $m = 0$ of $m = 4$. Geen
oplossing wanneer $\Delta < 0$: $(m-2)^2 < 4$, d.w.z.\
$-2 < m - 2 < 2$, dat is $0 < m < 4$.
\end{solution}

\begin{solution}{exo:g11:quad:7}
De top ligt bij $t = -\frac{20}{2 \times (-5)} = 2$, en
$h(2) = -20 + 40 + 1 = 21$: maximale hoogte $21$ m, bereikt na $2$
seconden.

$h(t) \geq 16$ betekent $-5t^2 + 20t - 15 \geq 0$, d.w.z.\ (delen door
$-5$, wat de ongelijkheid omkeert) $t^2 - 4t + 3 \leq 0$. De wortels van
$t^2-4t+3$ zijn $1$ en $3$ ($1 - 4 + 3 = 0$, product $3$), dus
$t^2 - 4t + 3 \leq 0$ precies op $\intcc{1}{3}$: de bal is minstens
$16$ m hoog tussen de tijden $t = 1$ s en $t = 3$ s.
\end{solution}

\begin{solution}{exo:g11:quad:8}
Met $X = x^2$: $X^2 - 13X + 36 = 0$, $\Delta = 169 - 144 = 25$, dus
$X = \frac{13 \pm 5}{2} = 9$ of $4$. Terug naar $x$: $x^2 = 9$ geeft
$x = \pm 3$, en $x^2 = 4$ geeft $x = \pm 2$. De oplossingsverzameling is
$\{-3, -2, 2, 3\}$.
\end{solution}

\begin{solution}{exo:g11:quad:9}
De voorwaarde is $x^2 < x + 2$, d.w.z.\ $x^2 - x - 2 < 0$. Omdat
$x^2 - x - 2 = (x+1)(x-2)$ (wortels $-1$ en $2$), geldt de ongelijkheid
strikt tussen de wortels: $x \in \intoo{-1}{2}$. Op een schets duikt de
parabool $y = x^2$ onder de lijn $y = x+2$ precies tussen hun twee
snijpunten $(-1, 1)$ en $(2, 4)$.
\end{solution}

\begin{solution}{exo:g11:quad:10}
Laat $x > 0$ voldoen aan $x + \frac1x = \frac{13}{6}$. Vermenigvuldigen
met $6x > 0$: $6x^2 + 6 = 13x$, d.w.z.\ $6x^2 - 13x + 6 = 0$. Dan
$\Delta = 169 - 144 = 25$ en $x = \frac{13 \pm 5}{12}$, wat
$x = \frac32$ of $x = \frac23$ geeft. Beide zijn positief en elkaars
omgekeerden, zoals verwacht: het getal is $\frac32$ (of $\frac23$).
\end{solution}

\begin{solution}{exo:g11:quad:11}
\emph{1.} Snijpunten komen overeen met oplossingen van $x^2 = mx - 1$,
d.w.z.\ $x^2 - mx + 1 = 0$, waarvan de discriminant $\Delta = m^2 - 4$
is. Twee punten wanneer $\abs{m} > 2$; precies Ă©Ă©n wanneer $m = \pm 2$;
geen wanneer $-2 < m < 2$.

\emph{2.} Voor $m = 2$: $x^2 - 2x + 1 = (x-1)^2 = 0$, dus het unieke
raakpunt is $(1, 1)$. Voor $m = -2$: $(x+1)^2 = 0$, raakpunt $(-1, 1)$.
In beide gevallen ontmoet de lijn de parabool in Ă©Ă©n punt zonder haar te
kruisen --- het is de raaklijn daar, en inderdaad is haar
richtingscoĂ«fficiĂ«nt $m = \pm 2$ gelijk aan de waarde bij $x = \pm 1$ van
de afgeleide $2x$ van $x^2$ (\cref{ch:g11:deriv}).
\end{solution}
