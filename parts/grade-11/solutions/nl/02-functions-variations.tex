\section*{Hoofdstuk \ref{ch:g11:func} --- Functies en variaties}

\begin{solution}{exo:g11:func:1}
$f$: eist $3 - x \geq 0$, \omterm{def:g11:func:function}{domein} $\intoc{-\infty}{3}$.

$g$: eist $x^2 - 4 \neq 0$, d.w.z.\ $x \neq \pm 2$: \omterm{def:g11:func:function}{domein}
$\R \setminus \{-2, 2\}$.

$h$: eist $x \geq 0$ en $x \neq 5$: \omterm{def:g11:func:function}{domein}
$\intco{0}{5} \cup \intoo{5}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:2}
$\sqrt{11} < \sqrt{13}$ omdat $\sqrt{}$ \omterm{def:g11:func:monotone}{stijgend} is en $11 < 13$.

$(-2.1)^2 > (-2.05)^2$ omdat $x^2$ \omterm{def:g10:functions:variations}{dalend} is op
$\intoc{-\infty}{0}$ en $-2.1 < -2.05$.

$\frac{1}{0.9} > \frac{1}{1.1}$ omdat $\frac1x$ \omterm{def:g10:functions:variations}{dalend} is op
$\intoo{0}{+\infty}$ en $0.9 < 1.1$.

$(-1.01)^3 < (-0.99)^3$ omdat $x^3$ \omterm{def:g11:func:monotone}{stijgend} is en $-1.01 < -0.99$.
\end{solution}

\begin{solution}{exo:g11:func:3}
$f(-x) = 3(-x)^4 - (-x)^2 = 3x^4 - x^2 = f(x)$: \omterm{def:g11:func:parity}{even}.

$g(-x) = -x^3 - x = -g(x)$: \omterm{def:g11:func:parity}{oneven}.

$h(1) = 2$ en $h(-1) = 1 - 1 = 0$, dus $h(-1)$ is noch $h(1)$ noch
$-h(1)$: noch \omterm{def:g11:func:parity}{even} noch \omterm{def:g11:func:parity}{oneven}.

$k(-x) = \frac{-x}{x^2+1} = -k(x)$: \omterm{def:g11:func:parity}{oneven}.
\end{solution}

\begin{solution}{exo:g11:func:4}
Lees de \omterm{def:g10:functions:table}{veranderingstabel}. Op $\intcc{-3}{1}$ klimt $f$ van $-2$ naar
$4$ en kruist de waarde $1$ precies één keer; op $\intcc{1}{5}$ daalt ze
van $4$ naar $0$ en kruist $1$ nog eens: de \omterm{def:g10:algebra:equation}{vergelijking} $f(x) = 1$ heeft
$2$ oplossingen. Het \omterm{def:g10:functions:extrema}{maximum} van $f$ is $f(1) = 4 < 5$, dus $f(x) = 5$
heeft geen oplossing.
\end{solution}

\begin{solution}{exo:g11:func:5}
Laat $2 \leq u < v$. Dan
\[
f(v) - f(u) = v^2 - 4v - u^2 + 4u = (v-u)(v+u) - 4(v-u)
= (v - u)(u + v - 4).
\]
Beide factoren zijn positief: $v - u > 0$, en $u + v > 2 + 2 = 4$ omdat
$u \geq 2$ en $v > 2$. Dus $f(v) > f(u)$: $f$ is strikt \omterm{def:g11:func:monotone}{stijgend} op
$\intco{2}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:6}
$f = \sqrt{u}$ met $u(x) = 5 - x$ \omterm{def:g10:functions:variations}{dalend} en niet-negatief op
$\intoc{-\infty}{5}$: $f$ is daar \omterm{def:g10:functions:variations}{dalend}
(\cref{prop:g11:func:transforms}).

$g = \frac1u$ met $u(x) = x^2 + 1$ \omterm{def:g11:func:monotone}{stijgend} en positief op
$\intco{0}{+\infty}$: $g$ is daar \omterm{def:g10:functions:variations}{dalend}.

$h = 3 - 2\sqrt x$: $\sqrt x$ is \omterm{def:g11:func:monotone}{stijgend}, vermenigvuldigen met $-2 < 0$
keert om, $3$ optellen verandert niets: $h$ is \omterm{def:g10:functions:variations}{dalend} op
$\intco{0}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:7}
\emph{1.} $f$ staat in \omterm{thm:g11:quad:canonical}{canonieke vorm}: \omterm{def:g10:functions:variations}{dalend} op $\intoc{-\infty}{2}$,
\omterm{def:g11:func:monotone}{stijgend} op $\intco{2}{+\infty}$, \omterm{def:g10:functions:extrema}{minimum} $f(2) = -3$.

\emph{2.} $g = \frac{1}{u}$ met $u(x) = (x-2)^2 + 1 = f(x) + 4$. De
verschuiving met $4$ houdt de variaties van $f$, en $u \geq 1 > 0$.
Omkeren keert ze om: $g$ is \omterm{def:g11:func:monotone}{stijgend} op $\intoc{-\infty}{2}$ en \omterm{def:g10:functions:variations}{dalend}
op $\intco{2}{+\infty}$, met \omterm{def:g10:functions:extrema}{maximum} $g(2) = \frac{1}{1} = 1$.
\end{solution}

\begin{solution}{exo:g11:func:8}
Elke heuvelvormige kromme door de drie punten werkt. Dan:
$f + 2$ gaat door $(0,3)$, $(2,5)$, $(4,3)$, dezelfde variaties
(verticale verschuiving); $-f$ gaat door $(0,-1)$, $(2,-3)$, $(4,-1)$ met
\emph{omgekeerde} variaties (de top wordt een dal); $2f$ gaat door
$(0,2)$, $(2,6)$, $(4,2)$, dezelfde variaties maar verticaal versterkt.
\end{solution}

\begin{solution}{exo:g11:func:9}
$2 - \frac{1}{x+1} = \frac{2(x+1) - 1}{x+1} = \frac{2x+1}{x+1} = f(x)$.
Op $\intoo{-1}{+\infty}$ is $u(x) = x + 1$ \omterm{def:g11:func:monotone}{stijgend} en positief, dus
$\frac1u$ is \omterm{def:g10:functions:variations}{dalend}, $-\frac1u$ is \omterm{def:g11:func:monotone}{stijgend}, en $2$ optellen verandert
niets: $f$ is strikt \omterm{def:g11:func:monotone}{stijgend} op $\intoo{-1}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:10}
\emph{1. Waar.} Als $u < v$, dan is
$(f+g)(v) - (f+g)(u) = \bigl(f(v)-f(u)\bigr) + \bigl(g(v)-g(u)\bigr)$
een som van twee niet-negatieve getallen.

\emph{2. Onwaar.} $f(x) = g(x) = x$ zijn \omterm{def:g11:func:monotone}{stijgend} op $\R$, maar hun
product $x^2$ is niet \omterm{def:g11:func:monotone}{stijgend} op $\R$ (het daalt op
$\intoc{-\infty}{0}$). (De uitspraak wordt waar als beide \omterm{def:g11:func:function}{functies} ook
niet-negatief zijn.)

\emph{3. Onwaar.} Hetzelfde tegenvoorbeeld: $f(x) = x$ is \omterm{def:g11:func:monotone}{stijgend} op
$\R$ maar $f^2(x) = x^2$ niet.
\end{solution}

\begin{solution}{exo:g11:func:11}
\emph{1.} Voor $0 < u < v$:
\[
f(v) - f(u) = (v - u) + \frac1v - \frac1u
= (v - u) + \frac{u - v}{uv}
= (v - u)\left(1 - \frac{1}{uv}\right)
= (v - u)\,\frac{uv - 1}{uv}.
\]

\emph{2.} Altijd $v - u > 0$ en $uv > 0$. Als $0 < u < v \leq 1$, dan
$uv < 1$, dus $uv - 1 < 0$ en $f(v) < f(u)$: $f$ is strikt \omterm{def:g10:functions:variations}{dalend} op
$\intoc{0}{1}$. Als $1 \leq u < v$, dan $uv > 1$ en $f(v) > f(u)$: $f$
is strikt \omterm{def:g11:func:monotone}{stijgend} op $\intco{1}{+\infty}$.

\emph{3.} $f$ daalt vóór $1$ en stijgt erna, dus $f(x) \geq f(1) = 2$
voor alle $x > 0$, met gelijkheid precies bij $x = 1$. (Deze
ongelijkheid keert in vele gedaanten terug; vergelijk
\cref{ex:g11:quad:canonical} en de raaklijnongelijkheden van
\cref{ch:g12:deriv}.)
\end{solution}
