\section*{Hoofdstuk \ref{ch:g11:vect} --- Vectoren en lijnen in het vlak}

\begin{solution}{exo:g11:vect:1}
$\vect{AB}\,(3, 4)$, $\vect{AC}\,(-3, 2)$,
$\vect{AB} + \vect{AC}\,(0, 6)$,
$2\vect{AB} - 3\vect{AC}\,(6 + 9,\ 8 - 6) = (15, 2)$. Het midden van
$[BC]$ is $\left(\frac{5 + (-1)}{2}, \frac{3+1}{2}\right) = (2, 2)$.
\end{solution}

\begin{solution}{exo:g11:vect:2}
$\det = 4 \times 3 - (-2) \times (-6) = 12 - 12 = 0$: collineair
(inderdaad $\vec v = -\frac12 \vec u$).

$\det = 2 \times 7 - 3 \times 5 = -1 \neq 0$: niet collineair.

$\det = 0 \times (-8) - 0 \times 3 = 0$: collineair (beide verticaal).
\end{solution}

\begin{solution}{exo:g11:vect:3}
Door $A(2,1)$, richting $(3, -1)$: matchen met $(-b, a) = (3, -1)$
geeft $a = -1$, $b = -3$, dus $-x - 3y + c = 0$; invullen van $A$:
$-2 - 3 + c = 0$, $c = 5$. Vermenigvuldigen met $-1$: $x + 3y - 5 = 0$.

Door $B(0,4)$ en $C(2,0)$: $\vect{BC}\,(2, -4)$, dus $a = -4$,
$b = -2$: $-4x - 2y + c = 0$; invullen van $B$: $-8 + c = 0$, $c = 8$.
Vereenvoudigen met $-2$: $2x + y - 4 = 0$. Controle met $C$:
$4 + 0 - 4 = 0$.
\end{solution}

\begin{solution}{exo:g11:vect:4}
Een richtingsvector is $(-b, a) = (2, 3)$. Snijpunten: $x = 0$ geeft
$y = 3$, punt $(0, 3)$; $y = 0$ geeft $x = -2$, punt $(-2, 0)$. Voor
$P(2,6)$: $3 \times 2 - 2 \times 6 + 6 = 0$, dus $P \in d$. Voor
$Q(1,4)$: $3 - 8 + 6 = 1 \neq 0$, dus $Q \notin d$.
\end{solution}

\begin{solution}{exo:g11:vect:5}
$\vect{AB}\,(3, 6)$ en $\vect{AC}\,(-3, -6)$:
$\det = 3 \times (-6) - (-3) \times 6 = 0$, dus $A$, $B$, $C$ zijn
collineair.

$\vect{DE}\,(2, 3)$ en $\vect{DF}\,(5, 8)$:
$\det = 2 \times 8 - 5 \times 3 = 1 \neq 0$: $D$, $E$, $F$ zijn niet
collineair.
\end{solution}

\begin{solution}{exo:g11:vect:6}
Eerste paar: optellen van de twee vergelijkingen,
$(2x + y - 7) + (x - y + 1) = 3x - 6 = 0$, dus $x = 2$, dan
$y = x + 1 = 3$: ze snijden in $(2, 3)$ (controle: $4 + 3 - 7 = 0$).

Tweede paar: $ab' - a'b = 1 \times 4 - (-2) \times (-2) = 0$, dus de
lijnen zijn evenwijdig; vermenigvuldigen van $e$ met $-2$ geeft
$-2x + 4y - 6 = 0$, wat verschilt van $e'$ ($-6 \neq 1$): strikt
evenwijdig, geen snijpunt.
\end{solution}

\begin{solution}{exo:g11:vect:7}
Voor $d: 2x - y + 3 = 0$: richting $(-b, a) = (1, 2)$ en het punt
$(0, 3)$ ligt op $d$, dus $d = \{(t,\ 3 + 2t) : t \in \R\}$.

Voor de parametrische lijn door $(1, -3)$ met richting $(2, 1)$:
$a = 1$, $b = -2$, dus $x - 2y + c = 0$; invullen van $(1, -3)$:
$1 + 6 + c = 0$, $c = -7$. Vergelijking: $x - 2y - 7 = 0$ (controle
$t = 1$, punt $(3, -2)$: $3 + 4 - 7 = 0$).
\end{solution}

\begin{solution}{exo:g11:vect:8}
Een evenwijdige lijn heeft dezelfde $(a, b)$: $4x - 3y + c = 0$; door
$P(1, -2)$: $4 + 6 + c = 0$, dus $c = -10$ en de lijn is
$4x - 3y - 10 = 0$.

Evenwijdigheid van $mx + 2y - 5 = 0$ met $d$ eist
$ab' - a'b = 4 \times 2 - m \times (-3) = 8 + 3m = 0$, dus
$m = -\frac83$.
\end{solution}

\begin{solution}{exo:g11:vect:9}
Het midden van $[AB]$ is $M\!\left(\frac{-1+3}{2}, \frac{0+2}{2}\right)
= (1, 1)$. De zwaartelijn is de lijn $(CM)$ met $C(1, -4)$: beide
punten hebben abscis $1$, dus de zwaartelijn is de verticale lijn
\[
x - 1 = 0 .
\]
\end{solution}

\begin{solution}{exo:g11:vect:10}
$\det(\vec u, \vec v) = m(m - 1) - 3 \times 2 = m^2 - m - 6$. Die
verdwijnt wanneer $m^2 - m - 6 = 0$: $\Delta = 1 + 24 = 25$, wortels
$m = \frac{1 \pm 5}{2}$, d.w.z.\ $m = 3$ of $m = -2$.
\end{solution}

\begin{solution}{exo:g11:vect:11}
\emph{1.} $I = \left(\frac12, 0\right)$. Dan
$\vect{DI} = \left(\frac12 - 0,\ 0 - 1\right) = \left(\frac12, -1\right)$
en $\vect{DJ} = \frac23\vect{DI} = \left(\frac13, -\frac23\right)$, dus
\[
J = D + \vect{DJ} = \left(\frac13,\ \frac13\right).
\]

\emph{2.} $\vect{AJ}\left(\frac13, \frac13\right)$ en
$\vect{AC}\,(1, 1)$:
$\det = \frac13 \times 1 - 1 \times \frac13 = 0$, dus $A$, $J$, $C$ zijn
collineair --- in feite $\vect{AJ} = \frac13\vect{AC}$: de lijn $(DI)$
snijdt de diagonaal $(AC)$ op precies een derde van de lengte vanaf $A$.
\end{solution}
