\section*{Hoofdstuk \ref{ch:g11:deriv} --- Differentiaalrekening: een eerste cursus}

\begin{solution}{exo:g11:deriv:1}
Voor $f(x) = x^2$ in $a = 2$:
\[
\frac{(2+h)^2 - 4}{h} = \frac{4h + h^2}{h} = 4 + h
\xrightarrow[h\to0]{} 4,
\]
dus $f'(2) = 4$. Voor $g(x) = x^2 + 3x$ in $a = 1$: $g(1) = 4$ en
\[
\frac{(1+h)^2 + 3(1+h) - 4}{h} = \frac{5h + h^2}{h} = 5 + h
\xrightarrow[h\to0]{} 5,
\]
dus $g'(1) = 5$.
\end{solution}

\begin{solution}{exo:g11:deriv:2}
$f'(x) = 12x^2 - 4x + 1$.

$g'(x) = \frac{5}{2\sqrt x} - \frac{3}{x^2}$ (op $\intoo{0}{+\infty}$).

$h'(x) = 2(x^2 - 3) + (2x+1)(2x) = 2x^2 - 6 + 4x^2 + 2x
= 6x^2 + 2x - 6$ (productregel).
\end{solution}

\begin{solution}{exo:g11:deriv:3}
$f(x) = x^2 - 3x$, $a = 2$: $f(2) = -2$, $f'(x) = 2x - 3$ dus $f'(2) = 1$;
raaklijn $y = -2 + (x - 2) = x - 4$.

$f(x) = \frac1x$, $a = 1$: $f(1) = 1$, $f'(1) = -1$; raaklijn
$y = 1 - (x - 1) = 2 - x$.

$f(x) = \sqrt x$, $a = 4$: $f(4) = 2$, $f'(4) = \frac{1}{2\sqrt4} =
\frac14$; raaklijn $y = 2 + \frac14(x - 4) = \frac{x}{4} + 1$.
\end{solution}

\begin{solution}{exo:g11:deriv:4}
$f'(x) = \frac{(x-1) - (x+2)}{(x-1)^2} = \frac{-3}{(x-1)^2}$.

$g'(x) = \frac{2x(x^2+1) - x^2 \times 2x}{(x^2+1)^2}
= \frac{2x}{(x^2+1)^2}$.

$h'(x) = -\frac{2x+1}{(x^2+x+1)^2}$ (quotiëntrregel met $u = 1$).
\end{solution}

\begin{solution}{exo:g11:deriv:5}
$f'(x) = 3x^2 - 12x + 9 = 3(x^2 - 4x + 3) = 3(x-1)(x-3)$: positief
buiten $\intcc{1}{3}$, negatief erin. Dus $f$ is stijgend op
$\intoc{-\infty}{1}$, dalend op $\intcc{1}{3}$, stijgend op
$\intco{3}{+\infty}$, met een lokaal maximum $f(1) = 1 - 6 + 9 - 2 = 2$
en een lokaal minimum $f(3) = 27 - 54 + 27 - 2 = -2$.
\end{solution}

\begin{solution}{exo:g11:deriv:6}
Op $\intoo{-1}{+\infty}$:
\[
f'(x) = \frac{2x(x+1) - (x^2+3)}{(x+1)^2}
= \frac{x^2 + 2x - 3}{(x+1)^2}
= \frac{(x+3)(x-1)}{(x+1)^2}.
\]
Voor $x > -1$ zijn zowel $x + 3$ als $(x+1)^2$ positief, dus $f'(x)$ heeft
het teken van $x - 1$: $f$ is dalend op $\intoc{-1}{1}$ en stijgend op
$\intco{1}{+\infty}$, met minimum $f(1) = \frac{4}{2} = 2$.
\end{solution}

\begin{solution}{exo:g11:deriv:7}
$f'(x) = 2x - 4$.

\emph{1.} Horizontale raaklijn: $f'(x) = 0$ bij $x = 2$; het punt is
$(2, f(2)) = (2, 1)$ (de top van de parabool).

\emph{2.} Evenwijdig met $y = 2x + 1$ betekent richtingscoëfficiënt $2$:
$2x - 4 = 2$, dus $x = 3$, wat het punt $(3, f(3)) = (3, 2)$ geeft.
\end{solution}

\begin{solution}{exo:g11:deriv:8}
Met breedte $x$ (twee omheinde zijden) en lengte $60 - 2x$:
$A(x) = x(60 - 2x) = 60x - 2x^2$ op $\intoo{0}{30}$. Dan
$A'(x) = 60 - 4x$, positief vóór $x = 15$ en negatief erna: de
oppervlakte is maximaal voor $x = 15$, wat een omheining van
$15 \times 30$ m met oppervlakte $450$ m$^2$ geeft. De topformule
$\alpha = -\frac{60}{2 \times (-2)} = 15$ van \cref{ch:g11:quad} geeft
hetzelfde antwoord zonder differentiaalrekening.
\end{solution}

\begin{solution}{exo:g11:deriv:9}
Laat $f(x) = \frac{x+1}{2} - \sqrt x$ op $\intoo{0}{+\infty}$. Dan
\[
f'(x) = \frac12 - \frac{1}{2\sqrt x} = \frac{\sqrt x - 1}{2\sqrt x},
\]
wat het teken van $\sqrt x - 1$ heeft: negatief op $\intoo{0}{1}$,
positief op $\intoo{1}{+\infty}$. Dus $f$ daalt dan stijgt, met minimum
$f(1) = 1 - 1 = 0$. Dus $f(x) \geq 0$ voor alle $x > 0$, d.w.z.\
$\sqrt x \leq \frac{x+1}{2}$, met gelijkheid alleen bij $x = 1$.
\end{solution}

\begin{solution}{exo:g11:deriv:10}
Uit $\pi r^2 h = 500$, $h = \frac{500}{\pi r^2}$, dus
\[
S(r) = 2\pi r^2 + 2\pi r \cdot \frac{500}{\pi r^2}
= 2\pi r^2 + \frac{1000}{r},
\qquad
S'(r) = 4\pi r - \frac{1000}{r^2}.
\]
$S'(r) = 0$ wanneer $4\pi r^3 = 1000$, d.w.z.\ $r^3 = \frac{250}{\pi}$,
dus $r = \sqrt[3]{250/\pi}$ ($\approx 4.3$ cm). Omdat $S'(r) =
\frac{4\pi r^3 - 1000}{r^2}$ negatief is vóór deze waarde en positief
erna, is het een minimum.
\end{solution}

\begin{solution}{exo:g11:deriv:11}
De raaklijn aan $y = x^2$ in het punt met abscis $a$ is
$y = a^2 + 2a(x - a) = 2ax - a^2$. Ze gaat door $P(1, -3)$ wanneer
$-3 = 2a - a^2$, d.w.z.\ $a^2 - 2a - 3 = 0$, waarvan de wortels $a = 3$
en $a = -1$ zijn ($\Delta = 16$). Er zijn dus \emph{twee} raaklijnen door
$P$:
\[
y = 6x - 9 \quad (a = 3),
\qquad
y = -2x - 1 \quad (a = -1).
\]
\end{solution}

\begin{solution}{exo:g11:deriv:12}
Uit \cref{ex:g11:deriv:study} stijgt $f$ tot het lokale maximum
$f(-1) = 3$, daalt tot het lokale minimum $f(1) = -1$, en stijgt
daarna opnieuw; voor grote positieve (resp.\ negatieve) $x$ domineert
$x^3$ en neemt $f$ willekeurig grote (resp.\ kleine) waarden aan. Horizontale
lijnen $y = k$ aflezen op de veranderingstabel:
\begin{itemize}
  \item $k > 3$ of $k < -1$: de lijn snijdt de kromme één keer
        --- $1$ oplossing;
  \item $k = 3$ of $k = -1$: de lijn raakt een keerpunt en kruist één
        andere tak --- $2$ oplossingen;
  \item $-1 < k < 3$: de lijn kruist alle drie de takken --- $3$
        oplossingen.
\end{itemize}
\end{solution}
