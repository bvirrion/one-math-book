\section*{Hoofdstuk \ref{ch:g11:stat} --- Beschrijvende statistiek}

\begin{solution}{exo:g11:stat:1}
$n = 8$, som $71$: gemiddelde $\bar x = \frac{71}{8} = 8.875$. Mediaan:
midden van de $4$de en $5$de waarden, $\frac{8 + 9}{2} = 8.5$.
Kwartielen: een kwart van $8$ is $2$, dus $Q_1$ is de $2$de waarde, $7$;
drie kwart is $6$, dus $Q_3$ is de $6$de waarde, $10$.
Interkwartielafstand: $10 - 7 = 3$.
\end{solution}

\begin{solution}{exo:g11:stat:2}
Som $40$, dus $\bar x = 5$. Som van kwadraten:
$4 + 16 + 16 + 16 + 25 + 25 + 49 + 81 = 232$. Met de snelle formule,
\[
V = \frac{232}{8} - 5^2 = 29 - 25 = 4,
\qquad
\sigma = 2 .
\]
\end{solution}

\begin{solution}{exo:g11:stat:3}
Gemiddelde:
\[
\bar x = \frac{1 \times 6 + 2 \times 9 + 3 \times 8 + 4 \times 10
+ 5 \times 9 + 6 \times 8}{50}
= \frac{181}{50} = 3.62 .
\]
Mediaan: de cumulatieve frequenties zijn $6, 15, 23, 33, \dots$, dus de
$25$ste en $26$ste gesorteerde uitkomsten zijn beide $4$: de mediaan is
$4$.
\end{solution}

\begin{solution}{exo:g11:stat:4}
Het totaal van de $15$ cijfers is $15 \times 11.2 = 168$; met de
nieuwkomer, $\frac{168 + 18}{16} = \frac{186}{16} = 11.625$.
\end{solution}

\begin{solution}{exo:g11:stat:5}
$\frac{6 + 8 + x + 12 + 14}{5} = 10$ geeft $40 + x = 50$, dus $x = 10$.
De afwijkingen van $6, 8, 10, 12, 14$ van het gemiddelde $10$ zijn
$-4, -2, 0, 2, 4$, met kwadraten sommerend tot $40$:
$V = \frac{40}{5} = 8$ en $\sigma = 2\sqrt2 \approx 2.83$.
\end{solution}

\begin{solution}{exo:g11:stat:6}
Volgens \cref{prop:g11:stat:affine} met $a = \frac12$, $b = 4$: nieuw
gemiddelde $\frac{12.4}{2} + 4 = 10.2$; nieuwe standaardafwijking
$\frac{2.5}{2} = 1.25$.
\end{solution}

\begin{solution}{exo:g11:stat:7}
Het samengevoegde gemiddelde is het \emph{gewogen} gemiddelde
\[
\frac{20 \times 10 + 30 \times 15}{50} = \frac{650}{50} = 13,
\]
niet $12.5$: groep B draagt meer waarden bij, dus trekt ze het
samengevoegde gemiddelde naar haar eigen.
\end{solution}

\begin{solution}{exo:g11:stat:8}
Beide lijnen liggen gemiddeld bijna op target ($1.002$ en $1.001$ kg),
maar de standaardafwijking van lijn 2 is drie keer groter: haar zakken
variëren veel meer, wat veel meer merkbaar te weinig en te veel gevulde
zakken oplevert. Lijn 2 moet eerst aangepakt worden --- consistentie,
niet alleen het gemiddelde, moet hersteld worden.
\end{solution}

\begin{solution}{exo:g11:stat:9}
$\bar x = \frac{70}{10} = 7$;
$V = \frac{540}{10} - 7^2 = 54 - 49 = 5$;
$\sigma = \sqrt5 \approx 2.24$.
\end{solution}

\begin{solution}{exo:g11:stat:10}
\emph{1.} Met $23$: som $50$, gemiddelde $5$; mediaan = midden van $5$de
en $6$de waarden $= 3$. Zonder $23$: som $27$, gemiddelde $3$; mediaan
($5$de van $9$ waarden) $= 3$.

\emph{2.} Met $23$ ($n = 10$): $Q_1$ is de $3$de waarde ($2$), $Q_3$ de
$8$ste ($4$): interkwartielafstand $2$. Zonder ($n = 9$): $Q_1$ is de
$3$de waarde ($2$), $Q_3$ de $7$de ($4$): nog steeds $2$.

\emph{3.} De ene uitbijter trekt het gemiddelde van $3$ naar $5$ maar
laat de mediaan en de interkwartielafstand onveranderd: die twee zijn
\emph{robuust}. Wanneer een dataset afwijkende waarden kan bevatten,
rapporteer mediaan en interkwartielafstand in plaats van gemiddelde en
standaardafwijking.
\end{solution}
