\section*{Hoofdstuk \ref{ch:g11:binom} --- De binomiale verdeling}

\begin{solution}{exo:g11:binom:1}
Rijen $6$ en $7$:
\[
1,\ 6,\ 15,\ 20,\ 15,\ 6,\ 1
\qquad\text{en}\qquad
1,\ 7,\ 21,\ 35,\ 35,\ 21,\ 7,\ 1 .
\]
Dus $\binom62 = 15$, $\binom73 = 35$, $\binom74 = 35$ (de symmetrie
$\binom73 = \binom74$ weerspiegelt successen en mislukkingen
verwisselen).
\end{solution}

\begin{solution}{exo:g11:binom:2}
Vast aantal proeven ($4$ worpen), twee uitkomsten per worp (zes of
niet, $p = \frac16$), onafhankelijke worpen: $X \sim \mathcal B(4,
\frac16)$.
\[
\P(X=0) = \left(\frac56\right)^4 = \frac{625}{1296} \approx 0.48,
\qquad
\P(X=1) = 4 \times \frac16\left(\frac56\right)^3 = \frac{500}{1296}
\approx 0.39,
\]
\[
\P(X \geq 2) = 1 - \frac{625 + 500}{1296} = \frac{171}{1296}
\approx 0.13 .
\]
\end{solution}

\begin{solution}{exo:g11:binom:3}
\emph{1.} Binomiaal $\mathcal B(20, \frac12)$: vaste $n$, zelfde $p$,
onafhankelijke worpen.

\emph{2.} Niet binomiaal: de kaarten worden \emph{zonder teruglegging}
gedeeld, dus de kans op een aas verandert van kaart tot kaart en de
trekkingen zijn niet onafhankelijk.

\emph{3.} Binomiaal $\mathcal B(7, 0.3)$ door de gestelde
onafhankelijkheids-aanname.
\end{solution}

\begin{solution}{exo:g11:binom:4}
$\E(X) = 50 \times 0.2 = 10$;
$\V(X) = 50 \times 0.2 \times 0.8 = 8$;
$\sigma(X) = 2\sqrt2 \approx 2.83$.
\end{solution}

\begin{solution}{exo:g11:binom:5}
$X \sim \mathcal B(6, 0.7)$.
\[
\P(X = 4) = \binom64 (0.7)^4 (0.3)^2 = 15 \times 0.2401 \times 0.09
\approx 0.324 .
\]
\[
\P(X \geq 5) = \binom65 (0.7)^5(0.3) + (0.7)^6
= 6 \times 0.16807 \times 0.3 + 0.117649 \approx 0.420 .
\]
\end{solution}

\begin{solution}{exo:g11:binom:6}
$X \sim \mathcal B\left(8, \frac12\right)$; elk pad heeft kans
$\frac{1}{256}$, dus
\[
\P(X \geq 6) = \frac{\binom86 + \binom87 + \binom88}{256}
= \frac{28 + 8 + 1}{256} = \frac{37}{256} \approx 0.14 .
\]
Raden slaagt ongeveer één op de zeven keer.
\end{solution}

\begin{solution}{exo:g11:binom:7}
Het complement van ``minstens één van elk'' is ``alle vijf identiek'':
alle A of alle B, elk met kans $\left(\frac12\right)^5 =
\frac1{32}$. Dus
\[
\P(\text{één van elke soort}) = 1 - \frac{2}{32} = \frac{15}{16} .
\]
\end{solution}

\begin{solution}{exo:g11:binom:8}
$\P(X = 3) = p^3$ en $\P(X \geq 1) = 1 - (1-p)^3$. Oplossen van
$p^3 = \frac{27}{64} = \left(\frac34\right)^3$ geeft $p = \frac34$ (de
derdemachtfunctie is strikt stijgend, \cref{ch:g11:func}, dus de
oplossing is uniek).
\end{solution}

\begin{solution}{exo:g11:binom:9}
$\P(\text{minstens één kop}) = 1 - \left(\frac12\right)^n$, dus de
voorwaarde is $\left(\frac12\right)^n < 0.01$, d.w.z.\ $2^n > 100$.
Omdat $2^6 = 64$ en $2^7 = 128$: vanaf $n = 7$ worpen.
\end{solution}

\begin{solution}{exo:g11:binom:10}
De som van rij $n$ telt alle paden van de $n$-proevenboom, gesorteerd op
aantal successen. Maar de boom verdubbelt haar paden bij elke proef (elk
pad splitst in S en M), dus ze heeft $2^n$ paden in totaal. Dus
$\sum_{k} \binom nk = 2^n$. Alternatief, door inductie: rij $0$ somt tot
$1 = 2^0$, en de regel van Pascal zorgt dat elke entry van rij $n$
bijdraagt aan precies twee entries van rij $n+1$, dus de rijsommen
verdubbelen.
\end{solution}

\begin{solution}{exo:g11:binom:11}
\emph{1.} Onder de claim, $X \sim \mathcal B(10, 0.6)$. Sommeren van de
eerste termen:
\begin{align*}
\P(X \leq 3)
&= (0.4)^{10} + 10(0.6)(0.4)^9 + 45(0.6)^2(0.4)^8
+ 120(0.6)^3(0.4)^7\\
&\approx 0.0001 + 0.0016 + 0.0106 + 0.0425 = 0.0548 .
\end{align*}

\emph{2.} Een resultaat minstens zo extreem als het waargenomene ($3$ of
minder goedkeuringen) heeft kans ongeveer $5.5\%$ --- net \emph{boven}
de drempel van $5\%$. Strikt de regel toepassend is de waarneming
(nauwelijks) verenigbaar met de claim en verwerpt men haar niet. Het
voorbeeld toont hoe gevoelig grensbeslissingen zijn voor de keuze van de
drempel: met een conventie van $6\%$ zou de conclusie omslaan.
\end{solution}
