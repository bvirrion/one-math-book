\begin{solution}{exo:g11:deriv:1}
Para $f(x) = x^2$ en $a = 2$:
\[
\frac{(2+h)^2 - 4}{h} = \frac{4h + h^2}{h} = 4 + h
\xrightarrow[h\to0]{} 4,
\]
Entonces $f'(2) = 4$. Para $g(x) = x^2 + 3x$ en $a = 1$: $g(1) = 4$ y
\[
\frac{(1+h)^2 + 3(1+h) - 4}{h} = \frac{5h + h^2}{h} = 5 + h
\xrightarrow[h\to0]{} 5,
\]
entonces $g'(1) = 5$.
\end{solution}

\begin{solution}{exo:g11:deriv:2}
$f'(x) = 12x^2 - 4x + 1$.

$g'(x) = \frac{5}{2\sqrt x} - \frac{3}{x^2}$ (on $\intoo{0}{+\infty}$).

$h'(x) = 2(x^2 - 3) + (2x+1)(2x) = 2x^2 - 6 + 4x^2 + 2x
= 6x^2 + 2x - 6$ (regla de producto).
\end{solution}

\begin{solution}{exo:g11:deriv:3}
$f(x) = x^2 - 3x$, $a = 2$: $f(2) = -2$, $f'(x) = 2x - 3$ so $f'(2) = 1$;
\omterm{def:g11:deriv:tangent}{tangente} $y = -2 + (x - 2) = x - 4$.

$f(x) = \frac1x$, $a = 1$: $f(1) = 1$, $f'(1) = -1$; \omterm{def:g11:deriv:tangent}{tangente}
$y = 1 - (x - 1) = 2 - x$.

$f(x) = \sqrt x$, $a = 4$: $f(4) = 2$, $f'(4) = \frac{1}{2\sqrt4} =
\frac14$; \omterm{def:g11:deriv:tangent}{tangente} $y = 2 + \frac14(x - 4) = \frac{x}{4} + 1$.
\end{solution}

\begin{solution}{exo:g11:deriv:4}
$f'(x) = \frac{(x-1) - (x+2)}{(x-1)^2} = \frac{-3}{(x-1)^2}$.

$g'(x) = \frac{2x(x^2+1) - x^2 \times 2x}{(x^2+1)^2}
= \frac{2x}{(x^2+1)^2}$.

$h'(x) = -\frac{2x+1}{(x^2+x+1)^2}$ (regla del cociente con $u = 1$).
\end{solution}

\begin{solution}{exo:g11:deriv:5}
$f'(x) = 3x^2 - 12x + 9 = 3(x^2 - 4x + 3) = 3(x-1)(x-3)$: positivo
exterior $\intcc{1}{3}$, negativo interior. Entonces $f$ está en \omterm{def:g11:func:monotone}{creciente}.
$\intoc{-\infty}{1}$, \omterm{def:g10:functions:variations}{decreciente} en $\intcc{1}{3}$, \omterm{def:g11:func:monotone}{creciente} en
$\intco{3}{+\infty}$, con un local \omterm{def:g10:functions:extrema}{máximo} $f(1) = 1 - 6 + 9 - 2 = 2$ y
un local \omterm{def:g10:functions:extrema}{mínimo} $f(3) = 27 - 54 + 27 - 2 = -2$.
\end{solution}

\begin{solution}{exo:g11:deriv:6}
En $\intoo{-1}{+\infty}$:
\[
f'(x) = \frac{2x(x+1) - (x^2+3)}{(x+1)^2}
= \frac{x^2 + 2x - 3}{(x+1)^2}
= \frac{(x+3)(x-1)}{(x+1)^2}.
\]
Para $x > -1$, tanto $x + 3$ como $(x+1)^2$ son positivos, por lo que $f'(x)$ tiene el
signo de $x - 1$: $f$ es \omterm{def:g10:functions:variations}{decreciente} en $\intoc{-1}{1}$ y \omterm{def:g11:func:monotone}{creciente} en
$\intco{1}{+\infty}$, con \omterm{def:g10:functions:extrema}{mínimo} $f(1) = \frac{4}{2} = 2$.
\end{solution}

\begin{solution}{exo:g11:deriv:7}
$f'(x) = 2x - 4$.

\emph{1.} Horizontal \omterm{def:g11:deriv:tangent}{tangente}: $f'(x) = 0$ en $x = 2$; el punto es
$(2, f(2)) = (2, 1)$ (el vértice del \omterm{def:g11:quad:parabola}{parábola}).

\emph{2.} Paralelo a $y = 2x + 1$ significa \omterm{def:g10:reffunc:affine}{pendiente} $2$: $2x - 4 = 2$, entonces
$x = 3$, dando el punto $(3, f(3)) = (3, 2)$.
\end{solution}

\begin{solution}{exo:g11:deriv:8}
Con ancho $x$ (dos lados vallados) y largo $60 - 2x$:
$A(x) = x(60 - 2x) = 60x - 2x^2$ en $\intoo{0}{30}$. entonces
$A'(x) = 60 - 4x$, positivo antes de $x = 15$ y negativo después: el área
es máximo para $x = 15$, lo que da un recinto de $15 \times 30$ m de área
$450$ m$^2$. La fórmula del vértice $\alpha = -\frac{60}{2 \times (-2)} = 15$
de \cref{ch:g11:quad} da la misma respuesta sin cálculo.
\end{solution}

\begin{solution}{exo:g11:deriv:9}
Deje $f(x) = \frac{x+1}{2} - \sqrt x$ en $\intoo{0}{+\infty}$. entonces
\[
f'(x) = \frac12 - \frac{1}{2\sqrt x} = \frac{\sqrt x - 1}{2\sqrt x},
\]
que tiene el signo de $\sqrt x - 1$: negativo en $\intoo{0}{1}$, positivo
en $\intoo{1}{+\infty}$. Entonces $f$ disminuye y luego aumenta, con \omterm{def:g10:functions:extrema}{mínimo}
$f(1) = 1 - 1 = 0$. Por lo tanto $f(x) \geq 0$ para todo $x > 0$, es decir
$\sqrt x \leq \frac{x+1}{2}$, con igualdad sólo en $x = 1$.
\end{solution}

\begin{solution}{exo:g11:deriv:10}
De $\pi r^2 h = 500$, $h = \frac{500}{\pi r^2}$, entonces
\[
S(r) = 2\pi r^2 + 2\pi r \cdot \frac{500}{\pi r^2}
= 2\pi r^2 + \frac{1000}{r},
\qquad
S'(r) = 4\pi r - \frac{1000}{r^2}.
\]
$S'(r) = 0$ cuando $4\pi r^3 = 1000$, es decir\ $r^3 = \frac{250}{\pi}$, entonces
$r = \sqrt[3]{250/\pi}$ ($\approx 4.3$cm). Dado que $S'(r) =
\frac{4\pi r^3 - 1000}{r^2}$ es negativo antes de este valor y positivo
después, es un \omterm{def:g10:functions:extrema}{mínimo}.
\end{solution}

\begin{solution}{exo:g11:deriv:11}
El \omterm{def:g11:deriv:tangent}{tangente} a $y = x^2$ en el punto de \omterm{def:g10:coordgeom:system}{abscisa} $a$ es
$y = a^2 + 2a(x - a) = 2ax - a^2$. Pasa por $P(1, -3)$ cuando
$-3 = 2a - a^2$, es decir$a^2 - 2a - 3 = 0$, cuyos \omterm{def:g11:quad:discriminant}{raíces} son $a = 3$ y
$a = -1$ ($\Delta = 16$). Por lo tanto, existen \emph{dos} \omterm{def:g11:deriv:tangent}{tangentes} hasta
$P$:
\[
y = 6x - 9 \quad (a = 3),
\qquad
y = -2x - 1 \quad (a = -1).
\]
\end{solution}

\begin{solution}{exo:g11:deriv:12}
Desde \cref{ex:g11:deriv:study}, $f$ aumenta hasta el local \omterm{def:g10:functions:extrema}{máximo}
$f(-1) = 3$, disminuye al \omterm{def:g10:functions:extrema}{mínimo} local $f(1) = -1$, luego aumenta
otra vez; para $x$ positivo grande (resp.\ negativo), $x^3$ domina y $f$
toma valores arbitrariamente grandes (o pequeños). Leyendo líneas horizontales
$y = k$ a través del \omterm{def:g10:functions:table}{tabla de variaciones}:
\begin{itemize}
  \item $k > 3$ o $k < -1$: la línea se encuentra con la curva una vez
        --- solución $1$;
  \item $k = 3$ o $k = -1$: la línea toca un punto de inflexión y cruza
        otra rama --- soluciones $2$;
  \item $-1 < k < 3$: la línea cruza los tres ramales --- $3$
        soluciones.
\end{itemize}
\end{solution}

\begin{solution}{pb:g11:deriv:1}
\textbf{1.}$f'(x) = 6x - 5$; $g'(x) = 3x^2 - 12$. Definición:
$\frac{(1 + h)^2 - 1}{h} = 2 + h \to 2$: el \omterm{def:g11:deriv:derivative}{derivado} de
$x^2$ en $1$ es $2$.

\textbf{2.} $g(2) = 8 - 24 = -16$ y $g'(2) = 12 - 12 = 0$:
el \omterm{def:g11:deriv:tangent}{tangente} es la línea horizontal $y = -16$. una horizontal
\omterm{def:g11:deriv:tangent}{tangente} señala un punto crítico --- aquí un \omterm{def:g10:functions:extrema}{mínimo} local, como
el \omterm{def:g10:functions:table}{tabla de variaciones} de la pregunta 5 lo confirma.

\textbf{3.} \omterm{def:g10:reffunc:affine}{pendiente} $2a$ at $(a, a^2)$:
$y = a^2 + 2a(x - a) = 2ax - a^2$.

\textbf{4.} En $x = 0$: $y = -a^2$: el \omterm{def:g11:deriv:tangent}{tangente} cruza el
eje en la profundidad $a^2$ debajo del origen --- exactamente tan lejos
\emph{abajo} cero ya que $P$ se encuentra encima de él. Receta: dibujar el
\omterm{def:g11:deriv:tangent}{tangente} en un punto de altura $h$, unirlo al punto del eje en
profundidad $h$ debajo del origen. (No se necesita \omterm{def:g11:deriv:derivative}{derivado} en el
tablero de dibujo.)

\textbf{5.} $g'(x) = 3(x^2 - 4)$: positivo encendido
$\intoo{-\infty}{-2}$, negativo en $\intoo{-2}{2}$, positivo
más allá. $g$ aumenta a \omterm{def:g10:functions:extrema}{máximo} local $g(-2) = 16$, cae
a un \omterm{def:g10:functions:extrema}{mínimo} $g(2) = -16$ local, luego aumenta.

\textbf{6.} $F\left(0, \frac14\right)$ y $T(0, -a^2)$:
$FT = \frac14 + a^2$.

\textbf{7.} $FP = a^2 + \frac14 = FT$: isósceles en $F$, por lo que el
Los ángulos de la base son iguales: $\widehat{FTP} = \widehat{FPT}$.

\textbf{8.} El rayo entrante en $P$ es vertical, por lo tanto paralelo
a la línea $(TF)$ (el eje $y$). El \omterm{def:g11:deriv:tangent}{tangente} $(TP)$ cruza
ambos paralelos, por lo que el ángulo entre el rayo y el \omterm{def:g11:deriv:tangent}{tangente}
es igual a $\widehat{FTP}$ (ángulos alternos) --- que es igual
$\widehat{FPT}$, el ángulo entre $[PF]$ y \omterm{def:g11:deriv:tangent}{tangente}.
Ángulos iguales en los dos lados del \omterm{def:g11:deriv:tangent}{tangente} en $P$.

\textbf{9.} Por la ley de la reflexión, el rayo que llega
verticalmente en $P$ hojas a lo largo de la línea que forma el ángulo igual
del otro lado --- por la pregunta 8, por $[PF]$: a través del
\omterm{pb:g11:quad:1}{enfocar}, sea lo que sea $a$. Invertida, luz nacida en las hojas $F$
cada punto del espejo paralelo al eje. Los platos recogen,
Haz de luces: probado.

\textbf{10.} $a = 1$: \omterm{def:g11:deriv:tangent}{tangente} $y = 2x - 1$, $T(0, -1)$;
$FP = \sqrt{1 + \left(1 - \frac14\right)^2} =
\sqrt{\frac{25}{16}} = \frac54$; $FT = \frac14 + 1 = \frac54$:
isósceles, como prometí.

\textbf{11.} El \omterm{def:g11:deriv:tangent}{tangente} en $x_n$ es
$y = f(x_n) + f'(x_n)(x - x_n)$; cruza $y = 0$ donde
$x = x_n - \frac{f(x_n)}{f'(x_n)}$.

\textbf{12.}$x_1 = 1 - \frac{-1}{2} = \frac32$;
$x_2 = \frac32 - \frac{1/4}{3} = \frac{17}{12}$;
$x_3 = \frac{577}{408}$: \omterm{def:g10:numbers:approx}{aproximaciones} de Garza de $\sqrt2$,
es decir,\ la máquina de punto fijo de \cref{pb:g10:functions:1}.
Identidad: $x - \frac{x^2 - 2}{2x} = \frac{2x^2 - x^2 + 2}{2x}
= \frac{x^2 + 2}{2x} = \frac12\left(x + \frac2x\right)$ ---
Receta de Heron \emph{is} Método de Newton aplicado a
$x^2 - 2$, dos milenios antes.

\textbf{13.}$x_1 = 5 - \frac{125 - 100}{75} = \frac{14}{3}
\approx 4.6667$; $x_2 \approx 4.6667 - \frac{1.63}{65.33}
\approx 4.6417$. Calculadora: $\sqrt[3]{100} = 4.6416\ldots$ ---
cuatro decimales correctos en dos pasos.

\textbf{14.} $f'(2) = 0$: el \omterm{def:g11:deriv:tangent}{tangente} en el punto de partida es
horizontal (pregunta 2) y nunca cruza el eje --- el
la fórmula divide por cero. Advertencia: comience a Newton lejos de
puntos críticos (y, en general, lo suficientemente cerca del \omterm{def:g11:quad:discriminant}{raíz}
quieres).

\textbf{15.} La bisección gana un dígito binario por paso; Newton
aproximadamente \emph{dobles} el número de dígitos correctos en cada paso
($1 \to 3 \to 6$ aquí). Cuando cada paso es costoso: millones
de parámetros, navegación en tiempo real --- duplicación de tiempos
avanzando lentamente, razón por la cual la línea \omterm{def:g11:deriv:tangent}{tangente} corre tanto hacia adentro
silicio.

\textbf{16.} $d^2 = 900 - w^2$, entonces
$S(w) = w(900 - w^2) = 900w - w^3$, para $0 < w < 30$.

\textbf{17.} $S'(w) = 900 - 3w^2 = 0$ en
$w = \sqrt{300} = 10\sqrt3 \approx 17.3$ cm ($S' > 0$ antes,
$< 0$ después: a \omterm{def:g10:functions:extrema}{máximo}). entonces
$d = \sqrt{600} = 10\sqrt6 \approx 24.5$cm.

\textbf{18.} $\frac{d^2}{w^2} = \frac{600}{300} = 2$, entonces
$d = w\sqrt2$: la profundidad es el ancho multiplicado por $\sqrt2$ --- el
proporción de carpintero (y el diámetro en tercios de construcción
lo entrega con una brújula, sin álgebra en el sitio).

\textbf{19.} Viga cuadrada: $w = d = \sqrt{450} \approx 21.2$
cm, rigidez $w^3 = 450^{3/2} \approx 9\,546$. Haz óptimo:
$S = 10\sqrt3 \times 600 = 6\,000\sqrt3 \approx 10\,392$. Ganancia:
aproximadamente $8.9\,\%$ más rígido del mismo tronco --- el \omterm{def:g11:deriv:derivative}{derivado}
paga el aserradero.

Geometría \textbf{20.}: los ángulos iguales del \omterm{def:g11:deriv:tangent}{tangente} se convirtieron en
paraboloide en un captador de luz. Numéricos: deslizarse hacia abajo
\omterm{def:g11:deriv:tangent}{tangentes} resuelve \omterm{def:g10:algebra:equation}{ecuaciones} a una velocidad de duplicación de dígitos.
Optimización: un \omterm{def:g11:deriv:derivative}{derivado} que desapareció encontró el registro más fuerte
rectángulo. Un objeto, tres oficios --- y grado 12
La convexidad dirá, además, de qué lado de cada
\omterm{def:g11:deriv:tangent}{tangente} la curva está.
\end{solution}
