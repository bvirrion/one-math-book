\begin{solution}{exo:g11:vect:3}
A través de $A(2,1)$, dirección $(3, -1)$: coincidencia $(-b, a) = (3, -1)$ da
$a = -1$, $b = -3$, entonces $-x - 3y + c = 0$; conectando $A$: $-2 - 3 + c =
0$, $c = 5$. Multiplicando por $-1$: $x + 3y - 5 = 0$.

A través de $B(0,4)$ y $C(2,0)$: $\vect{BC}\,(2, -4)$, entonces $a = -4$,
$b = -2$: $-4x - 2y + c = 0$; conectando $B$: $-8 + c = 0$, $c = 8$.
Simplificando por $-2$: $2x + y - 4 = 0$. Consulte con $C$: $4 + 0 - 4 = 0$.
\end{solution}

\begin{solution}{exo:g11:vect:4}

$y = 3$, punto $(0, 3)$; $y = 0$ da $x = -2$, punto $(-2, 0)$. Para

\end{solution}

\begin{solution}{exo:g11:vect:5}
$\vect{AB}\,(3, 6)$ y $\vect{AC}\,(-3, -6)$:
$\det = 3 \times (-6) - (-3) \times 6 = 0$, por lo que $A$, $B$, $C$ están alineados.

$\vect{DE}\,(2, 3)$ y $\vect{DF}\,(5, 8)$:
$\det = 2 \times 8 - 5 \times 3 = 1 \neq 0$: $D$, $E$, $F$ no son
alineado.
\end{solution}

\begin{solution}{exo:g11:vect:6}
Primer par: sumando los dos \omterm{def:g10:algebra:equation}{ecuaciones}, $(2x + y - 7) + (x - y + 1) =
3x - 6 = 0$, entonces $x = 2$, luego $y = x + 1 = 3$: se encuentran en $(2, 3)$
(verificar: $4 + 3 - 7 = 0$).

Segundo par: $ab' - a'b = 1 \times 4 - (-2) \times (-2) = 0$, por lo que el
las líneas son paralelas; multiplicar $e$ por $-2$ da $-2x + 4y - 6 = 0$,
que difiere de $e'$ ($-6 \neq 1$): estrictamente paralelo, no
\omterm{def:g10:numbers:interunion}{intersección}.
\end{solution}

\begin{solution}{exo:g11:vect:7}
Para $d: 2x - y + 3 = 0$: dirección $(-b, a) = (1, 2)$ y el punto
$(0, 3)$ se encuentra en $d$, por lo que $d = \{(t,\ 3 + 2t) : t \in \R\}$.

Para la línea paramétrica que pasa por $(1, -3)$ con dirección $(2, 1)$:
$a = 1$, $b = -2$, entonces $x - 2y + c = 0$; conectando $(1, -3)$:
$1 + 6 + c = 0$, $c = -7$. \omterm{def:g10:algebra:equation}{Ecuación}: $x - 2y - 7 = 0$ (verifique $t = 1$,
punto $(3, -2)$: $3 + 4 - 7 = 0$).
\end{solution}

\begin{solution}{exo:g11:vect:8}
Una línea paralela tiene el mismo $(a, b)$: $4x - 3y + c = 0$; a través de
$P(1, -2)$: $4 + 6 + c = 0$, entonces $c = -10$ y la línea es
$4x - 3y - 10 = 0$.

El paralelismo de $mx + 2y - 5 = 0$ con $d$ requiere
$ab' - a'b = 4 \times 2 - m \times (-3) = 8 + 3m = 0$, entonces
$m = -\frac83$.
\end{solution}

\begin{solution}{exo:g11:vect:9}

\end{solution}

\begin{solution}{exo:g11:vect:10}
$\det(\vec u, \vec v) = m(m - 1) - 3 \times 2 = m^2 - m - 6$. se desvanece

\end{solution}

\begin{solution}{exo:g11:vect:11}
\emph{1.} $I = \left(\frac12, 0\right)$. entonces
$\vect{DI} = \left(\frac12 - 0,\ 0 - 1\right) = \left(\frac12, -1\right)$
y $\vect{DJ} = \frac23\vect{DI} = \left(\frac13, -\frac23\right)$, entonces
\[
J = D + \vect{DJ} = \left(\frac13,\ \frac13\right).
\]

\emph{2.} $\vect{AJ}\left(\frac13, \frac13\right)$ y
$\vect{AC}\,(1, 1)$:
$\det = \frac13 \times 1 - 1 \times \frac13 = 0$, por lo que $A$, $J$, $C$ son
alineado --- de hecho $\vect{AJ} = \frac13\vect{AC}$: la línea $(DI)$ corta
la diagonal $(AC)$ exactamente a un tercio de su longitud de $A$.
\end{solution}

\begin{solution}{pb:g11:vect:1}

\textbf{2.} Para $ax + by + c = 0$, un \omterm{def:g11:vect:direction}{vector de direccion} es
$(-b, a)$: aquí $(5, 2)$. Un punto: $y = 1$ le da a $x = 1$:
$(1, 1)$. Paramétrico: $x = 1 + 5t$, $y = 1 + 2t$.

\textbf{3.} De $x = 7 - 3y$ en $2x - 5y + 3 = 0$:
$14 - 6y - 5y + 3 = 0$, entonces $y = \frac{17}{11}$ y
