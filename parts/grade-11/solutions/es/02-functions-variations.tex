\begin{solution}{exo:g11:func:2}
$\sqrt{11} < \sqrt{13}$ ya que $\sqrt{}$ es \omterm{def:g11:func:monotone}{creciente} y $11 < 13$.

$(-2.1)^2 > (-2.05)^2$ ya que $x^2$ es \omterm{def:g10:functions:variations}{decreciente} en
$\intoc{-\infty}{0}$ y $-2.1 < -2.05$.

$\frac{1}{0.9} > \frac{1}{1.1}$ ya que $\frac1x$ es \omterm{def:g10:functions:variations}{decreciente} en
$\intoo{0}{+\infty}$ y $0.9 < 1.1$.

$(-1.01)^3 < (-0.99)^3$ ya que $x^3$ es \omterm{def:g11:func:monotone}{creciente} y $-1.01 < -0.99$.
\end{solution}

\begin{solution}{exo:g11:func:3}
$f(-x) = 3(-x)^4 - (-x)^2 = 3x^4 - x^2 = f(x)$: \omterm{def:g11:func:parity}{incluso}.

$g(-x) = -x^3 - x = -g(x)$: \omterm{def:g11:func:parity}{extraño}.

$h(1) = 2$ y $h(-1) = 1 - 1 = 0$, por lo que $h(-1)$ no es ni $h(1)$ ni
$-h(1)$: ni \omterm{def:g11:func:parity}{incluso} ni \omterm{def:g11:func:parity}{extraño}.

$k(-x) = \frac{-x}{x^2+1} = -k(x)$: \omterm{def:g11:func:parity}{extraño}.
\end{solution}

\begin{solution}{exo:g11:func:4}
Lea el \omterm{def:g10:functions:table}{tabla de variaciones}. En $\intcc{-3}{1}$, $f$ sube de $-2$ a $4$
y cruza el valor $1$ exactamente una vez; en $\intcc{1}{5}$ baja
de $4$ a $0$ y cruza una vez más $1$: el \omterm{def:g10:algebra:equation}{ecuación} $f(x) = 1$ tiene
$2$ soluciones. El \omterm{def:g10:functions:extrema}{máximo} de $f$ es $f(1) = 4 < 5$, por lo que $f(x) = 5$ no tiene
solución.
\end{solution}

\begin{solution}{exo:g11:func:5}
Deje $2 \leq u < v$. entonces
\[
f(v) - f(u) = v^2 - 4v - u^2 + 4u = (v-u)(v+u) - 4(v-u)
= (v - u)(u + v - 4).
\]
Ambos factores son positivos: $v - u > 0$ y $u + v > 2 + 2 = 4$ desde
$u \geq 2$ y $v > 2$. Por lo tanto $f(v) > f(u)$: $f$ es estrictamente \omterm{def:g11:func:monotone}{creciente}.
en $\intco{2}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:6}
$f = \sqrt{u}$ con $u(x) = 5 - x$ \omterm{def:g10:functions:variations}{decreciente} y no negativo en
$\intoc{-\infty}{5}$: $f$ está \omterm{def:g10:functions:variations}{decreciente} ahí
(\cref{prop:g11:func:transforms}).

$g = \frac1u$ con $u(x) = x^2 + 1$ \omterm{def:g11:func:monotone}{creciente} y positivo en
$\intco{0}{+\infty}$: $g$ está \omterm{def:g10:functions:variations}{decreciente} allí.

$h = 3 - 2\sqrt x$: $\sqrt x$ es \omterm{def:g11:func:monotone}{creciente}, multiplicado por $-2 < 0$
se invierte, agregar $3$ no cambia nada: $h$ es \omterm{def:g10:functions:variations}{decreciente} en
$\intco{0}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:7}
\emph{1.} $f$ is in \omterm{thm:g11:quad:canonical}{forma canónica}: \omterm{def:g10:functions:variations}{decreciente} on $\intoc{-\infty}{2}$,
\omterm{def:g11:func:monotone}{creciente} on $\intco{2}{+\infty}$, \omterm{def:g10:functions:extrema}{mínimo} $f(2) = -3$.

\emph{2.} $g = \frac{1}{u}$ con $u(x) = (x-2)^2 + 1 = f(x) + 4$. el
shift by $4$ mantiene las variaciones de $f$ y $u \geq 1 > 0$. Tomando el
recíproco los invierte: $g$ es \omterm{def:g11:func:monotone}{creciente} en $\intoc{-\infty}{2}$ y
\omterm{def:g10:functions:variations}{decreciente} en $\intco{2}{+\infty}$, con \omterm{def:g10:functions:extrema}{máximo}
$g(2) = \frac{1}{1} = 1$.
\end{solution}

\begin{solution}{exo:g11:func:8}
Cualquier curva en forma de colina que pase por los tres puntos funciona. Entonces:
$f + 2$ pasa por $(0,3)$, $(2,5)$, $(4,3)$, mismas variaciones
(desplazamiento vertical); $-f$ pasa por $(0,-1)$, $(2,-3)$, $(4,-1)$ con
\emph{invertido} variaciones (el pico se convierte en valle); $2f$ pases
hasta $(0,2)$, $(2,6)$, $(4,2)$, mismas variaciones pero amplificadas
verticalmente.
\end{solution}

\begin{solution}{exo:g11:func:9}
$2 - \frac{1}{x+1} = \frac{2(x+1) - 1}{x+1} = \frac{2x+1}{x+1} = f(x)$.
En $\intoo{-1}{+\infty}$, $u(x) = x + 1$ es \omterm{def:g11:func:monotone}{creciente} y positivo, por lo que
$\frac1u$ es \omterm{def:g10:functions:variations}{decreciente}, $-\frac1u$ es \omterm{def:g11:func:monotone}{creciente} y se agregan cambios $2$
nada: $f$ es estrictamente \omterm{def:g11:func:monotone}{creciente} en $\intoo{-1}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:10}
\emph{1. Verdadero.} Si $u < v$, entonces
$(f+g)(v) - (f+g)(u) = \bigl(f(v)-f(u)\bigr) + \bigl(g(v)-g(u)\bigr)$ es
una suma de dos números no negativos.

\emph{2. Falso.} $f(x) = g(x) = x$ son \omterm{def:g11:func:monotone}{creciente} en $\R$, pero sus
el producto $x^2$ no es \omterm{def:g11:func:monotone}{creciente} en $\R$ (disminuye en
$\intoc{-\infty}{0}$). (La afirmación se vuelve verdadera si ambos \omterm{def:g11:func:function}{funciones} son
también no negativo.)

\emph{3. Falso.} Mismo contraejemplo: $f(x) = x$ es \omterm{def:g11:func:monotone}{creciente} en $\R$
pero $f^2(x) = x^2$ no lo es.
\end{solution}

\begin{solution}{exo:g11:func:11}
\emph{1.} Para $0 < u < v$:
\[
f(v) - f(u) = (v - u) + \frac1v - \frac1u
= (v - u) + \frac{u - v}{uv}
= (v - u)\left(1 - \frac{1}{uv}\right)
= (v - u)\,\frac{uv - 1}{uv}.
\]

\emph{2.} Siempre $v - u > 0$ y $uv > 0$. Si $0 < u < v \leq 1$, entonces
$uv < 1$, entonces $uv - 1 < 0$ y $f(v) < f(u)$: $f$ es estrictamente \omterm{def:g10:functions:variations}{decreciente}
en $\intoc{0}{1}$. Si $1 \leq u < v$, entonces $uv > 1$ y $f(v) > f(u)$:
$f$ es estrictamente \omterm{def:g11:func:monotone}{creciente} en $\intco{1}{+\infty}$.

\emph{3.} $f$ disminuye antes de $1$ y aumenta después, por lo que
$f(x) \geq f(1) = 2$ para todos los $x > 0$, con igualdad exactamente en $x = 1$.
(Esta desigualdad reaparece de muchas formas; compárese
\cref{ex:g11:quad:canonical} y las desigualdades de la recta tangente de
\cref{ch:g12:deriv}.)
\end{solution}

\begin{solution}{pb:g11:func:1}
\textbf{1.} $x^4 - 3x^2$: \omterm{def:g11:func:parity}{incluso} ($(-x)^4 - 3(-x)^2 = x^4 -
3x^2$). $x^3 - x$: \omterm{def:g11:func:parity}{extraño}. $x^2 + x$: ninguno ($f(-1) = 0 \neq
f(1) = 2$ y $\neq -f(1)$). $\abs x$: \omterm{def:g11:func:parity}{incluso}. $\frac1x$: \omterm{def:g11:func:parity}{extraño}.

\textbf{2.} \omterm{def:g11:func:parity}{Incluso}: la curva es simétrica a través del eje $y$
(espejo). \omterm{def:g11:func:parity}{Extraño}: simétrico respecto al origen (media vuelta).

\textbf{3.} Si $f$ y $g$ son \omterm{def:g11:func:parity}{extraño}:
$(fg)(-x) = f(-x)g(-x) = (-f(x))(-g(x)) = (fg)(x)$: \omterm{def:g11:func:parity}{incluso}.
Tabla: \omterm{def:g11:func:parity}{incluso} $\times$ \omterm{def:g11:func:parity}{incluso} $=$ \omterm{def:g11:func:parity}{incluso}; \omterm{def:g11:func:parity}{incluso} $\times$ \omterm{def:g11:func:parity}{extraño} $=$ \omterm{def:g11:func:parity}{extraño};
\omterm{def:g11:func:parity}{extraño} $\times$ \omterm{def:g11:func:parity}{extraño} $=$ \omterm{def:g11:func:parity}{incluso} --- la regla de los signos con paridades.

\textbf{4.} \omterm{def:g11:func:parity}{Extraño} en $0$: $f(0) = -f(0)$, entonces $f(0) = 0$. \omterm{def:g11:func:parity}{Incluso}:
$f(-3) = f(3) = 7$. Tanto \omterm{def:g11:func:parity}{incluso} como \omterm{def:g11:func:parity}{extraño}:
$f(x) = f(-x) = -f(x)$ para todos los $x$, por lo que $2f(x) = 0$: $f$ es el
cero \omterm{def:g11:func:function}{función}.

\textbf{5.} La mitad derecha ($x \geq 0$) determina completamente un \omterm{def:g11:func:parity}{incluso
función}: la mitad izquierda es su espejo. Ni \omterm{def:g11:func:parity}{incluso}
ni \omterm{def:g11:func:parity}{extraño}: $x^2 + x$ (pregunta 1) --- sus dos simetrías fallan en
$x = 1$.

\textbf{6.} $E(x) = \frac{(x^3 + x^2 + 1) + (-x^3 + x^2 +
1)}{2} = x^2 + 1$ (\omterm{def:g11:func:parity}{incluso}), $O(x) = \frac{(x^3 + x^2 + 1) -
(-x^3 + x^2 + 1)}{2} = x^3$ (\omterm{def:g11:func:parity}{extraño}); su suma es $f$.

\textbf{7.} $E(-x) = \frac{f(-x) + f(x)}{2} = E(x)$: \omterm{def:g11:func:parity}{incluso}.
$O(-x) = \frac{f(-x) - f(x)}{2} = -O(x)$: \omterm{def:g11:func:parity}{extraño}. y
$E(x) + O(x) = f(x)$ por suma directa: cada \omterm{def:g11:func:function}{función}
divisiones.

\textbf{8.} De $E_1 + O_1 = E_2 + O_2$:
$E_1 - E_2 = O_2 - O_1$. El lado izquierdo es \omterm{def:g11:func:parity}{incluso}, el lado derecho
\omterm{def:g11:func:parity}{extraño} --- entonces este \omterm{def:g11:func:function}{función} es ambos, por lo tanto cero (pregunta 4):
$E_1 = E_2$ y $O_1 = O_2$. Una sola división.

\textbf{9.} $E(x) = \frac12\left(\frac{1}{1 - x} + \frac{1}{1 +
x}\right) = \frac12 \cdot \frac{(1 + x) + (1 - x)}{1 - x^2}
= \frac{1}{1 - x^2}$, y
$O(x) = \frac12 \cdot \frac{(1 + x) - (1 - x)}{1 - x^2}
= \frac{x}{1 - x^2}$. Verifique: $E$ es \omterm{def:g11:func:parity}{incluso}, $O$ es \omterm{def:g11:func:parity}{extraño} y
su suma es $\frac{1 + x}{1 - x^2} =
\frac{1 + x}{(1 - x)(1 + x)} = \frac{1}{1 - x} = f(x)$.

\textbf{10.} Debido a que cada pieza obedece a una simetría, la mitad del
el trabajo (y la mitad de los datos) es suficiente para cada uno: resolver por un lado,
espejo del otro --- y muchas operaciones (integración,
series, análisis de señales) tratan piezas simétricas mucho más
simplemente que \omterm{def:g11:func:function}{funciones} sin procesar.

\textbf{11.} $(x-3)^2$: $3$ desplazado a la derecha. $x^2 + 2$: desplazado
$2$ arriba. $(x-3)^2 + 2$: ambos. $-x^2$: volteado al revés.
$2x^2$: estirado verticalmente por $2$ (cuenco más estrecho).

\textbf{12.} Para $x \geq 2$: $(x - 2) + (x + 2) = 2x$. Para
$-2 \leq x \leq 2$: $(2 - x) + (x + 2) = 4$. Para $x \leq -2$:
$-2x$. La curva desciende con \omterm{def:g10:reffunc:affine}{pendiente} $-2$, discurre plana en altura
$4$ entre $-2$ y $2$, luego sube con \omterm{def:g10:reffunc:affine}{pendiente} $2$: un valle
con piso plano.

\textbf{13.} Desplazamiento vertical: $f(x) + 2$ permanece \omterm{def:g11:func:parity}{incluso}
($f(-x) + 2 = f(x) + 2$). Estiramiento vertical: $2f(x)$ estancias
\omterm{def:g11:func:parity}{incluso}. Desplazamiento horizontal: $(x - 3)^2$ no es \omterm{def:g11:func:parity}{incluso}
($f(-1) = 16 \neq 4 = f(1)$): el deslizamiento lateral destruye el
espejo.

\textbf{14.} En piso plano el valor es $4 \neq 6$. Para
$x \geq 2$: $2x = 6$, $x = 3$. Para $x \leq -2$: $-2x = 6$,
$x = -3$. Soluciones: $\pm 3$.

\textbf{15.} $f(x) = a(x - 2)^2 - 3$ con $f(0) = 4a - 3 = 1$:
$a = 1$, entonces $f(x) = (x - 2)^2 - 3$.

\textbf{16.} $g(v) - g(u) = (v - u) + 4\,\frac{u - v}{uv}
= (v - u)\,\frac{uv - 4}{uv}$: negativo para $u < v$ con
$uv < 4$, positivo con $uv > 4$: \omterm{def:g10:functions:variations}{decreciente} encendido
$\intoc{0}{2}$, \omterm{def:g11:func:monotone}{creciente} en $\intco{2}{+\infty}$; \omterm{def:g10:functions:extrema}{mínimo} en
$x = 2$, de valor $g(2) = 4$.

\textbf{17.} Lados $x$ y $\frac{16}{x}$:
$P(x) = 2\left(x + \frac{16}{x}\right)$, minimizado (pregunta
Máquina de 16 con $4 \to 16$, punto de inflexión $\sqrt{16} = 4$)
en $x = 4$: el $4 \times 4$ \emph{cuadrado}, perímetro $16$ ---
La respuesta de Dido, ahora demostrada para cada longitud de lado real.

\textbf{18.} En $\intco{0}{+\infty}$: $x \mapsto x^2 + 1$
aumenta, y $\sqrt{\phantom{x}}$ aumenta, por lo que el
aumentos compuestos. Y $h(-x) = \sqrt{x^2 + 1} = h(x)$: $h$
es \omterm{def:g11:func:parity}{incluso}, por lo que por simetría especular disminuye en
$\intoc{-\infty}{0}$. \omterm{def:g10:functions:extrema}{Mínimo} $h(0) = 1$.

\textbf{19.} Distancia al cuadrado a $(3, 0)$ del punto
$(x, \sqrt x)$:
$d^2(x) = (x - 3)^2 + x = x^2 - 5x + 9$, mínimo en
$x = \frac52$: punto más cercano
$\left(\frac52, \sqrt{\frac52}\right)$, distancia
$\sqrt{\frac{11}{4}} = \frac{\sqrt{11}}{2}$. Minimizando $d^2$
es legítimo porque elevar al cuadrado es estrictamente \omterm{def:g11:func:monotone}{creciente} en
Números positivos: $d$ y $d^2$ tocan fondo en el mismo lugar.

\textbf{20.} Stock en bruto: las curvas de referencia, conocidas de memoria.
Máquinas: desplaza, voltea y estira montaje nuevo \omterm{def:g10:functions:graph}{graficos}
de viejo. Control de calidad: paridad y reparto par-impar,
que reducen a la mitad cada estudio. Banco de medición: \omterm{def:g10:algebra:expand}{factorización}
$f(v) - f(u)$, que lee variaciones y mínimos directamente. el
herramienta eléctrica del próximo capítulo: el \emph{derivado}, que dice
ellos de un vistazo.
\end{solution}
