\section*{Capítulo \ref{ch:g11:trigo} --- Trigonometría: el círculo unitario}

\begin{solution}{exo:g11:trigo:1}
To \omterm{def:g11:trigo:radian}{radianes}: $30^\circ = \frac{\pi}{6}$, $45^\circ = \frac{\pi}{4}$,
$120^\circ = \frac{2\pi}{3}$, $270^\circ = \frac{3\pi}{2}$.

A grados: $\frac{\pi}{5} = 36^\circ$, $\frac{5\pi}{6} = 150^\circ$,
$\frac{7\pi}{4} = 315^\circ$, $\frac{2\pi}{9} = 40^\circ$.
\end{solution}

\begin{solution}{exo:g11:trigo:2}
$\frac{2\pi}{3}$ y $-\frac{\pi}{4}$ ya están en
$\intoc{-\pi}{\pi}$: segundo cuadrante y cuarto cuadrante respectivamente.

$\frac{17\pi}{6} - 2\pi = \frac{5\pi}{6}$: segundo cuadrante, cercano al
eje $x$ negativo.

$-\frac{7\pi}{2} + 4\pi = \frac{\pi}{2}$: la parte superior del círculo,
$(0, 1)$.
\end{solution}

\begin{solution}{exo:g11:trigo:3}
$\cos\frac{3\pi}{4} = \cos\left(\pi - \frac\pi4\right)
= -\cos\frac\pi4 = -\frac{\sqrt2}{2}$.

$\sin\frac{5\pi}{6} = \sin\left(\pi - \frac\pi6\right)
= \sin\frac\pi6 = \frac12$.

$\cos\left(-\frac\pi3\right) = \cos\frac\pi3 = \frac12$.

$\sin\frac{4\pi}{3} = \sin\left(\pi + \frac\pi3\right)
= -\sin\frac\pi3 = -\frac{\sqrt3}{2}$.

$\cos\frac{11\pi}{6} = \cos\left(-\frac\pi6 + 2\pi\right)
= \cos\frac\pi6 = \frac{\sqrt3}{2}$.
\end{solution}

\begin{solution}{exo:g11:trigo:4}
$\sin^2 t = 1 - \cos^2 t = 1 - \frac{25}{169} = \frac{144}{169}$, entonces
$\sin t = \pm\frac{12}{13}$. En $\intoo{-\frac\pi2}{0}$ (cuarto cuadrante)
el \omterm{def:g10:coordgeom:system}{ordenada} es negativo: $\sin t = -\frac{12}{13}$.
\end{solution}

\begin{solution}{exo:g11:trigo:5}
Usando \cref{prop:g11:trigo:associated}:
\[
A = (-\cos x) + \cos x + \cos x - \cos x = 0 ,
\]
\[
B = \sin x + (-\sin x) + (-\sin x) = -\sin x .
\]
\end{solution}

\begin{solution}{exo:g11:trigo:6}
$\cos t = \frac{\sqrt3}{2}$: una solución es $\frac\pi6$, entonces
$t = \pm\frac\pi6$ (los desplazamientos por $2k\pi$ salen del \omterm{def:g10:numbers:interval}{intervalo}).

$\sin t = \frac12$: $t = \frac\pi6$ or $t = \pi - \frac\pi6 =
\frac{5\pi}{6}$.

$\cos t = -1$: el único punto de \omterm{def:g10:coordgeom:system}{abscisa} $-1$ es $M(\pi)$, por lo que $t = \pi$.
\end{solution}

\begin{solution}{exo:g11:trigo:7}
$\sin t = -\frac{\sqrt3}{2}$: a partir de $a = -\frac\pi3$, las familias son
$-\frac\pi3 + 2k\pi$ y $\pi + \frac\pi3 + 2k\pi$; en
$\intco{0}{2\pi}$: $t = \frac{4\pi}{3}$ y $t = \frac{5\pi}{3}$.

$\cos t = 0$: $t = \frac\pi2$ y $t = \frac{3\pi}{2}$.

$2\sin t + 1 = 0$ significa $\sin t = -\frac12$: de $a = -\frac\pi6$, en
$\intco{0}{2\pi}$: $t = \frac{7\pi}{6}$ y $t = \frac{11\pi}{6}$.
\end{solution}

\begin{solution}{exo:g11:trigo:8}
Por \cref{thm:g11:trigo:equations}, $\cos 2t = \cos t$ significa
$2t = t + 2k\pi$ o $2t = -t + 2k\pi$, es decir,\ $t = 2k\pi$ o
$t = \frac{2k\pi}{3}$ ($k \in \Z$). La primera familia está contenida en el
segundo (tome $k$ un múltiplo de $3$), por lo que el conjunto de soluciones es
\[
S = \left\{\frac{2k\pi}{3} : k \in \Z\right\}
\]
--- en el círculo, los tres puntos $M(0)$,
$M\!\left(\frac{2\pi}{3}\right)$, $M\!\left(\frac{4\pi}{3}\right)$.
\end{solution}

\begin{solution}{exo:g11:trigo:9}
\omterm{def:g10:algebra:expand}{En expansión} ambos cuadrados y usando $\cos^2 x + \sin^2 x = 1$:
\[
(\cos^2 x + 2\sin x\cos x + \sin^2 x)
+ (\cos^2 x - 2\sin x\cos x + \sin^2 x)
= 1 + 1 = 2 .
\]
\end{solution}

\begin{solution}{exo:g11:trigo:10}
Rodar sin deslizarse significa que la longitud del arco es igual a la distancia
recorrido: $r\theta = 150$ cm da
$\theta = \frac{150}{30} = 5$ rad $= 5 \times \frac{180}{\pi} \approx
286.5^\circ$. Una vuelta completa hace avanzar la bicicleta por la circunferencia.
$2\pi \times 30 = 60\pi \approx 188.5$ cm, aproximadamente $1.88$ m.
\end{solution}

\begin{solution}{exo:g11:trigo:11}
En el círculo unitario, los puntos de \omterm{def:g10:coordgeom:system}{abscisa} exactamente $\frac12$ son
$M\!\left(\pm\frac\pi3\right)$. Los puntos con \omterm{def:g10:coordgeom:system}{abscisa} \emph{a lo sumo}
$\frac12$ forma el arco a la izquierda de la línea vertical $x = \frac12$,
viajado como $t$ va desde $\frac\pi3$ hasta $\pi$, o desde $-\pi$ hasta
$-\frac\pi3$. Por lo tanto, en $\intoc{-\pi}{\pi}$:
\[
S = \intoc{-\pi}{-\frac{\pi}{3}} \cup \intcc{\frac{\pi}{3}}{\pi}.
\]
(Se incluyen los puntos finales $\pm\frac\pi3$ ya que la desigualdad es amplia).
\end{solution}

\begin{solution}{pb:g11:trigo:1}
\textbf{1.}$30^\circ = \frac{\pi}{6}$; $135^\circ =
\frac{3\pi}{4}$; $300^\circ = \frac{5\pi}{3}$. y
$\frac{3\pi}{4} = 135^\circ$; $\frac{5\pi}{6} = 150^\circ$.

\textbf{2.} Longitud $r\,t = 3 \times 2 = 6$ m. \omterm{def:g11:trigo:radian}{radianes} hacer arco
longitud de un producto simple --- los grados arrastrarían un
$\frac{\pi}{180}$ en cada fórmula.

\textbf{3.} \omterm{def:g11:trigo:radian}{radianes} se convirtió en $= \frac{\text{distance}}
{\text{radius}} = \frac{1000}{0.30} \approx 3\,333$ rad;
dividido por $2\pi$: aproximadamente $530$ vueltas completas.

\textbf{4.} $\cos\frac{2\pi}{3} = -\frac12$;
$\sin\left(-\frac\pi4\right) = -\frac{\sqrt2}{2}$;
$\frac{19\pi}{6} - 2\pi = \frac{7\pi}{6}$, so
$\cos\frac{19\pi}{6} = \cos\frac{7\pi}{6} =
-\frac{\sqrt3}{2}$.

\textbf{5.}$\cos^2 t = 1 - \frac{9}{25} = \frac{16}{25}$; en
el segundo cuadrante el \omterm{def:g11:trigo:cossin}{coseno} es negativo:
$\cos t = -\frac45$, y
$\tan t = \frac{3/5}{-4/5} = -\frac34$.

\textbf{6.} En $t$ minutos la rueda gira
$\frac{t}{30} \times 2\pi = \frac{\pi t}{15}$. En $t = 0$:
$h = 65 - 60 = 5$ m --- la plataforma de embarque en la parte inferior
(altura del cubo menos radio). El signo menos te pone \emph{abajo}
el cubo cuando el ángulo girado es $0$.

\textbf{7.} $h(7.5) = 65 - 60\cos\frac{\pi}{2} = 65$ m (cubo
altura); $h(15) = 65 + 60 = 125$ m (la parte superior);
$h(20) = 65 - 60\cos\frac{4\pi}{3} = 65 + 30 = 95$m.

\textbf{8.} $65 - 60\cos\frac{\pi t}{15} = 95$ da
$\cos\frac{\pi t}{15} = -\frac12$: primera solución
$\frac{\pi t}{15} = \frac{2\pi}{3}$, es decir,\ $t = 10$ minutos.

\textbf{9.} $\cos\frac{\pi t}{15} \leq -\frac12$ para
$\frac{\pi t}{15} \in \intcc{\frac{2\pi}{3}}{\frac{4\pi}{3}}$,
es decir,\ $t \in \intcc{10}{20}$: diez minutos del viaje anterior
$95$ m, simétrico con respecto a la cumbre en $t = 15$.

\textbf{10.} Primer minuto:
\[
h(1) - h(0) = 60\left(1 - \cos\frac{\pi}{15}\right)
\approx 1.3 \text{ m}.
\]
Entre los minutos $7$ y $8$:
\[
60\left(\cos\frac{7\pi}{15} - \cos\frac{8\pi}{15}\right)
\approx 12.5 \text{ m}
\]
--- casi diez veces más. La altura cambia más rápidamente al pasar el nivel del buje.
y se detiene cerca de la parte inferior y superior: la sinusoide es empinada en su
medio, plano en sus extremos.

\textbf{11.} Radio $= \frac{40 - 4}{2} = 18$ m, cubo en
$4 + 18 = 22$ m, período $12$ min:
$h(t) = 22 - 18\cos\left(\frac{\pi t}{6}\right)$.

\textbf{12.} \omterm{def:g10:functions:extrema}{Máximo} $7 + 3 = 10$ m cuando
$\sin\frac{\pi t}{6} = 1$: primero a las $t = 3$ (3 a. m.);
\omterm{def:g10:functions:extrema}{mínimo} $4$ m primero en $t = 9$; período
$\frac{2\pi}{\pi/6} = 12$ horas.

\textbf{13.} $\sin\frac{\pi t}{6} \geq \frac12$ para
$\frac{\pi t}{6} \in \intcc{\frac\pi6}{\frac{5\pi}{6}}$:
$t \in \intcc{1}{5}$ --- un período de cuatro horas de 1 a.m. a 5
a.m., reabriendo un período más tarde, $t \in \intcc{13}{17}$
(de 13 a 17 horas).

\textbf{14.} El espejo a través del eje vertical envía el punto en
ángulo $t$ hasta el punto en el ángulo $\pi - t$, preservando el
\omterm{def:g10:coordgeom:system}{ordenada}: $\sin(\pi - t) = \sin t$. La media vuelta sobre el
El centro envía $t$ a $\pi + t$, negando ambos \omterm{def:g10:coordgeom:system}{coordenadas}:
$\cos(\pi + t) = -\cos t$. Y el espejo en diagonal
intercambia \omterm{def:g10:coordgeom:system}{abscisa} y \omterm{def:g10:coordgeom:system}{ordenada}:
$\cos\left(\frac\pi2 - t\right) = \sin t$ --- el
``co'' de ángulos complementarios.

\textbf{15.} $\sin(-t) = -\sin t$: impar (simetría de media vuelta de
es \omterm{def:g10:functions:graph}{gráfico}); $\cos(-t) = \cos t$: par (espejo). y
$\sin t + t^3$ es una suma de dos \omterm{def:g11:func:parity}{funciones impares}: impar.

\textbf{16.} Factor: $(2\sin t + 1)(\sin t - 1) = 0$:
$\sin t = 1$ da $t = \frac\pi2$;
$\sin t = -\frac12$ da $t = \frac{7\pi}{6}$ y
$t = \frac{11\pi}{6}$. Tres soluciones en el círculo.

\textbf{17.} $1$ rad $= \frac{180}{\pi} \approx 57.3^\circ$.
Entonces $\sin(1) \approx 0.841$ mientras que $\sin(1^\circ) \approx
0.0175$: un factor de casi $50$. Moraleja: comprueba la luz de modo.
antes de confiar en cualquier visualización trigonométrica.

\textbf{18.} \omterm{def:g10:coordgeom:system}{Abscisa} es igual a \omterm{def:g10:coordgeom:system}{ordenada} en la diagonal:
$t = \frac\pi4$ y $t = \frac{5\pi}{4}$.

\textbf{19.} $\sin(0.1) = 0.09983\ldots$ contra $0.1$:
error relativo sobre $0.17\,\%$. Para ángulos pequeños la cuerda
abraza el arco --- y el límite del grado 12
$\frac{\sin t}{t} \to 1$ es exactamente esta observación hecha en
un teorema.

\textbf{20.} \omterm{def:g11:trigo:radian}{radianes}: ángulo $=$ arco en el círculo unitario, por lo que
los ángulos se calculan como longitudes. \omterm{def:g11:trigo:cossin}{Coseno} y \omterm{def:g11:trigo:cossin}{seno}: el \omterm{def:g10:coordgeom:system}{coordenadas}
de la manecilla del reloj --- sombra en el suelo, altura en el
pared. Simetrías del círculo: todo el catálogo de
identidades, leer espejos y medias vueltas. \omterm{def:g10:algebra:equation}{Ecuaciones}
$\cos t = c$: horario del reloj, con el círculo mostrando
cada solución a la vez. Disfraces: la rueda llevaba
$65 - 60\cos$, la marea llevaba $7 + 3\sin$ --- mismo reloj, mismo
matemáticas.
\end{solution}
