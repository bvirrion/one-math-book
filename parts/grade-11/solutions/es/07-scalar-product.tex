\begin{solution}{exo:g11:scal:1}
$(2,5)\cdot(3,-1) = 6 - 5 = 1$.

\begin{solution}{exo:g11:scal:2}
$\vect{AB}\cdot\vect{AC} = 6 \times 6 \times \cos 60^\circ = 18$.

El \omterm{def:g11:vect:vector}{vectores} $\vect{AB}$ y $\vect{BC}$ forman un ángulo de
$180^\circ - 60^\circ = 120^\circ$ (colóquelos cola con cola en $B$), de modo que
$\vect{AB}\cdot\vect{BC} = 36 \cos 120^\circ = -18$.
\end{solution}

\begin{solution}{exo:g11:scal:3}
$\vec u \cdot \vec v = 3(t - 8) + t = 4t - 24 = 0$ da $t = 6$.

$\vec u - \vec v = (t - 4,\ 0)$, entonces
$\vec u \cdot (\vec u - \vec v) = t(t-4) + 0 = t^2 - 4t$, que desaparece
para $t = 0$ o $t = 4$.
\end{solution}

\begin{solution}{exo:g11:scal:4}
$\vec u \cdot \vec v = 3 + 2 = 5$, $\norm{\vec u} = \sqrt5$,
$\norm{\vec v} = \sqrt{10}$, entonces
\[
\cos\theta = \frac{5}{\sqrt5\,\sqrt{10}} = \frac{5}{5\sqrt2}
= \frac{\sqrt2}{2},
\]
por lo tanto $\theta = 45^\circ$ exactamente.
\end{solution}

\begin{solution}{exo:g11:scal:5}
$a^2 = 16 + 36 - 2 \times 4 \times 6 \times \cos\frac{2\pi}{3}
= 52 - 48 \times \left(-\frac12\right) = 76$, so $a = \sqrt{76} =
2\sqrt{19} \approx 8.7$.

Para el segundo triángulo, la ley de \omterm{def:g11:trigo:cossin}{cosenos} resuelta para el ángulo da
\[
\cos\widehat A = \frac{b^2 + c^2 - a^2}{2bc}
= \frac{25 + 16 - 49}{2 \times 5 \times 4} = -\frac{8}{40} = -\frac15,
\]
negativo: $\widehat A$ es obtuso.
\end{solution}

\begin{solution}{exo:g11:scal:6}
$\vect{AB}\,(4, -2)$ y $\vect{AC}\,(2, 4)$:
$\vect{AB}\cdot\vect{AC} = 8 - 8 = 0$, entonces el ángulo en $A$ es correcto.
Lados: $AB = \sqrt{16 + 4} = \sqrt{20}$, $AC = \sqrt{4 + 16} =

\end{solution}

\begin{solution}{exo:g11:scal:7}
Forma centro-radio: $(x + 2)^2 + (y - 3)^2 = 25$.

Para el segundo círculo, completa los cuadrados:
\[
x^2 - 6x + y^2 + 2y = 6
\iff (x - 3)^2 - 9 + (y + 1)^2 - 1 = 6
\iff (x-3)^2 + (y+1)^2 = 16 :
\]
centro $(3, -1)$, radio $4$.
\end{solution}

\begin{solution}{exo:g11:scal:8}
$M(x,y)$ está en el círculo cuando $\vect{MA}\cdot\vect{MB} = 0$ con
$\vect{MA}\,(-3 - x,\ -y)$ y $\vect{MB}\,(-x,\ 2 - y)$:
\[
(-3 - x)(-x) + (-y)(2 - y) = x^2 + 3x + y^2 - 2y = 0 .
\]
Para el origen: $0 + 0 + 0 - 0 = 0$, entonces $O$ se encuentra en el círculo ---
geométricamente, $\vect{OA}\,(-3,0)$ y $\vect{OB}\,(0,2)$ son \omterm{def:g11:scal:orthogonal}{ortogonal},
entonces $O$ ve $[AB]$ en ángulo recto. (Completando los cuadrados:
centro $\left(-\frac32, 1\right)$, radio $\frac{\sqrt{13}}{2}$.)
\end{solution}

\begin{solution}{exo:g11:scal:9}
La perpendicular que pasa por $P$ tiene el \omterm{def:g11:vect:direction}{vector de direccion} de $d$, es decir
$(-1, 2)$, como de costumbre \omterm{def:g11:vect:vector}{vector}: \omterm{def:g10:algebra:equation}{ecuación} $-x + 2y + c = 0$; a través de
$P(1,3)$: $-1 + 6 + c = 0$, $c = -5$, entonces $x - 2y + 5 = 0$.

\omterm{def:g10:numbers:interunion}{Intersección} con $d$: de $2x + y = 7$, $y = 7 - 2x$; sustituyendo,
$x - 2(7 - 2x) + 5 = 5x - 9 = 0$, entonces $x = \frac95$, $y = \frac{17}5$:
$H\left(\frac95, \frac{17}5\right)$. entonces
$\vect{PH}\left(\frac45, \frac25\right)$ y
\[
PH = \sqrt{\frac{16}{25} + \frac{4}{25}} = \frac{\sqrt{20}}{5}
= \frac{2\sqrt5}{5} \approx 0.89 .
\]
\end{solution}

\begin{solution}{exo:g11:scal:10}
Añadiendo
\[
\norm{\vec u + \vec v}^2 = \norm{\vec u}^2 + 2\vec u\cdot\vec v +
\norm{\vec v}^2
\quad\text{and}\quad
\norm{\vec u - \vec v}^2 = \norm{\vec u}^2 - 2\vec u\cdot\vec v +
\norm{\vec v}^2,
\]
los términos cruzados se cancelan y
$\norm{\vec u + \vec v}^2 + \norm{\vec u - \vec v}^2 = 2\norm{\vec u}^2 +
2\norm{\vec v}^2$. En un paralelogramo con lados $\vec u$ y $\vec v$,
las diagonales son $\vec u + \vec v$ y $\vec u - \vec v$: la suma de las
los cuadrados de las dos diagonales son iguales a la suma de los cuadrados de las cuatro
lados.
\end{solution}

\begin{solution}{exo:g11:scal:11}
\emph{1.} Dado que $I$ es \omterm{prop:g10:coordgeom:midpoint}{punto medio}, $\vect{IB} = -\vect{IA}$ y
$IA = 2$. entonces
\[
\vect{MA}\cdot\vect{MB}
= (\vect{MI} + \vect{IA})\cdot(\vect{MI} - \vect{IA})
= \norm{\vect{MI}}^2 - \norm{\vect{IA}}^2
= MI^2 - 4 .
\]

\emph{2.} La condición $\vect{MA}\cdot\vect{MB} = 12$ se convierte en
$MI^2 = 16$, es decir$MI = 4$: el conjunto es el círculo de centro $I$ y
radio $4$.
\end{solution}

\begin{solution}{pb:g11:scal:1}
\textbf{1.} $\vec u \cdot \vec v = -3 + 8 = 5$;
$\norm{\vec u} = 5$, $\norm{\vec v} = \sqrt5$;
$\cos\theta = \frac{5}{5\sqrt5} = \frac{1}{\sqrt5}$:
$\theta \approx 63.4^\circ$.

\textbf{2.} $6 - 4k = 0$: $k = \frac32$.

\textbf{3.}$W = 50 \times 10 \times \cos 60^\circ = 250$
Julios: sólo funciona el componente a lo largo del movimiento.

\textbf{4.} $c^2 = 25 + 49 - 2 \times 35 \times \frac12 = 39$:
$c = \sqrt{39}$.

\textbf{5.} $\cos\gamma = \frac{16 + 36 - 64}{2 \times 4
\times 6} = -\frac14$: negativo, el ángulo es obtuso:
$\gamma \approx 104.5^\circ$.

\textbf{6.} \omterm{def:g10:coordgeom:system}{Coordenadas}: $\vec u \cdot \vec v = \cos a \cos b +
\sin a \sin b$. Geometría: dos unidades \omterm{def:g11:vect:vector}{vectores},
$\vec u \cdot \vec v = 1 \times 1 \times \cos(a - b)$.
Igualando los dos cálculos del mismo número:
$\cos(a - b) = \cos a \cos b + \sin a \sin b$.

\textbf{7.} $\cos(a + b) = \cos(a - (-b)) = \cos a \cos(-b) +
\sin a \sin(-b) = \cos a \cos b - \sin a \sin b$.

\textbf{8.} $\sin(a + b) = \cos\left(\frac\pi2 - a - b\right)
= \cos\left(\left(\frac\pi2 - a\right) - b\right)
= \cos\left(\frac\pi2 - a\right)\cos b +
\sin\left(\frac\pi2 - a\right)\sin b
= \sin a \cos b + \cos a \sin b$.

\textbf{9.}$\cos 15^\circ = \cos 45^\circ \cos 30^\circ +
\sin 45^\circ \sin 30^\circ = \frac{\sqrt2}{2} \cdot
\frac{\sqrt3}{2} + \frac{\sqrt2}{2} \cdot \frac12 =
\frac{\sqrt6 + \sqrt2}{4} \approx 0.966$. y
$\sin 75^\circ = \cos 15^\circ$: mismo valor.

\textbf{10.} $b = a$ en las preguntas 7 y 8:
$\cos 2a = \cos^2 a - \sin^2 a = 2\cos^2 a - 1 = 1 - 2\sin^2 a$
(usando $\cos^2 + \sin^2 = 1$), y
$\sin 2a = 2 \sin a \cos a$.

\textbf{11.} $\sin^2 15^\circ = \frac{1 - \cos 30^\circ}{2}
= \frac{2 - \sqrt3}{4}$, entonces
$\sin 15^\circ = \frac{\sqrt{2 - \sqrt3}}{2}$. cuadratura
$\frac{\sqrt6 - \sqrt2}{4}$:
$\frac{6 - 2\sqrt{12} + 2}{16} = \frac{8 - 4\sqrt3}{16} =
\frac{2 - \sqrt3}{4}$: el mismo número con dos disfraces.

\textbf{12.} $\vect{AP}$ se descompone en una parte a lo largo de $d$
(invisible para $\vec n$) y una parte perpendicular a $d$, cuyo
longitud es la distancia buscada; salpicando con la unidad normal
$\frac{\vec n}{\norm{\vec n}}$ mata la primera parte y
mide el segundo. Con $A$ en $d$:
$\vect{AP} \cdot \vec n = a x_0 + b y_0 - (a x_A + b y_A) =
a x_0 + b y_0 + c$, de ahí la fórmula. Ejemplo:
$\frac{\abs{15 + 24 - 12}}{5} = \frac{27}{5} = 5.4$.

\textbf{13.} Radio $= 5.4$:
$(x - 5)^2 + (y - 6)^2 = 29.16$.

\textbf{14.} Distancia $= \frac{\abs{-12}}{5} = 2.4$. Pie: el
perpendicular a través del origen es $(3t, 4t)$; en la linea:
$9t + 16t = 12$, $t = \frac{12}{25}$: pie
$\left(\frac{36}{25}, \frac{48}{25}\right)$, a distancia
$5t = \frac{60}{25} = 2.4$. La fórmula y lo honesto.
el cálculo está de acuerdo.

\textbf{15.} Base $AB = 6$, altura $=$ distancia de $C$ a
$(AB)$ (el eje $y = 0$): $5$; área $15$. Línea $(AC)$:
$5x - 2y = 0$, y
$\operatorname{dist}(B, (AC)) = \frac{\abs{30}}{\sqrt{29}}$;
entonces $\frac12 \times AC \times \frac{30}{\sqrt{29}} =
\frac12 \times \sqrt{29} \times \frac{30}{\sqrt{29}} = 15$:
misma zona, ya sea altitud.

\textbf{16.} $\operatorname{dist} = \frac{\abs{2 + 1 - 6}}
{\sqrt2} = \frac{3}{\sqrt2} \approx 2.12 > 2$: la línea pasa
claro del círculo --- disjunto.

\textbf{17.} Insertar $I$:
$\vect{MA} \cdot \vect{MB} = (\vect{MI} + \vect{IA}) \cdot
(\vect{MI} + \vect{IB}) = MI^2 + \vect{MI} \cdot (\vect{IA} +
\vect{IB}) + \vect{IA} \cdot \vect{IB}$. El término medio muere
($\vect{IA} + \vect{IB} = \vec 0$), y
$\vect{IA} \cdot \vect{IB} = -IA^2 = -\frac{AB^2}{4}$
(frente a \omterm{def:g11:vect:vector}{vectores} de longitud $\frac{AB}2$).

\textbf{18.} $\vect{MA} \cdot \vect{MB} = 0 \iff MI^2 =
\frac{AB^2}{4} \iff MI = \frac{AB}{2}$: el círculo del centro
$I$ y radio $\frac{AB}2$ --- el círculo con diámetro
$[AB]$.

\textbf{19.} \omterm{def:g10:algebra:expand}{En expansión} cada par de $H$ ($\vect{BC} =
\vect{HC} - \vect{HB}$, etc.):
\[
\vect{HA} \cdot \vect{HC} - \vect{HA} \cdot \vect{HB}
+ \vect{HB} \cdot \vect{HA} - \vect{HB} \cdot \vect{HC}
+ \vect{HC} \cdot \vect{HB} - \vect{HC} \cdot \vect{HA} = 0 :
\]
todo se cancela en pares --- la identidad se mantiene para
\emph{cualquier} cuatro puntos. Para nuestro $H$, los dos primeros puntos
Los productos desaparecen por hipótesis, por lo que
$\vect{HC} \cdot \vect{AB} = 0$: $H$ se encuentra en la altitud desde
$C$. Tres altitudes, un punto.

\textbf{20.} Las fórmulas de suma provienen de calcular un punto.
producto con la cara \emph{coordinar} y el
\emph{normas y ángulos} cara; la fórmula de la distancia desde el
\emph{proyección} cara; el \omterm{pb:g11:scal:1}{ortocentro} del
Cara \emph{bilinealidad} (Chasles-and-expand). Calcular lo mismo
cantidad dos veces, se iguala y se cae un teorema: todo el comercio del \omterm{def:g11:scal:dot}{escalar
producto}.
\end{solution}
