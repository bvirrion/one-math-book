\section*{Capítulo \ref{ch:g11:quad} --- Funciones y ecuaciones cuadráticas}

\begin{solution}{exo:g11:quad:1}
$x^2-5x+6$: $\Delta = 25 - 24 = 1$, \omterm{def:g11:quad:discriminant}{raíces} $\frac{5 \pm 1}{2}$, es decir\ $2$
y $3$.

$2x^2+3x-2$: $\Delta = 9 + 16 = 25$, \omterm{def:g11:quad:discriminant}{raíces} $\frac{-3 \pm 5}{4}$, es decir\
$\frac12$ y $-2$.

$x^2+2x+5$: $\Delta = 4 - 20 = -16 < 0$, no hay una solución real.

$9x^2-6x+1$: $\Delta = 36 - 36 = 0$, uno (doble) \omterm{def:g11:quad:discriminant}{raíz}
$x_0 = \frac{6}{18} = \frac13$.
\end{solution}

\begin{solution}{exo:g11:quad:2}
$f(x) = x^2 - 6x + 11 = (x-3)^2 - 9 + 11 = (x-3)^2 + 2$: vértice $(3, 2)$.

$g(x) = -2x^2+4x+1 = -2(x^2 - 2x) + 1 = -2\bigl((x-1)^2 - 1\bigr) + 1
= -2(x-1)^2 + 3$: vértice $(1, 3)$.

$h(x) = 3x^2 + 6x = 3(x^2 + 2x) = 3\bigl((x+1)^2 - 1\bigr)
= 3(x+1)^2 - 3$: vértice $(-1, -3)$.
\end{solution}

\begin{solution}{exo:g11:quad:3}
$x^2 - 3x - 10$: $\Delta = 9 + 40 = 49$, \omterm{def:g11:quad:discriminant}{raíces} $\frac{3 \pm 7}{2} = -2$
y $5$; coeficiente principal positivo, por lo que el trinomio es negativo
estrictamente entre \omterm{def:g11:quad:discriminant}{raíces}: el conjunto de soluciones es $\intoo{-2}{5}$.

$2x^2 + x + 3$: $\Delta = 1 - 24 = -23 < 0$ y $a = 2 > 0$: siempre
positivo, el conjunto de soluciones es $\R$.

$-x^2 + 6x - 9 = -(x-3)^2$: es $\geq 0$ solo donde $(x-3)^2 \leq 0$,
eso es en $x = 3$. El conjunto de soluciones es $\{3\}$.
\end{solution}

\begin{solution}{exo:g11:quad:4}
$x^2 - 7x + 6$: $1 - 7 + 6 = 0$, por lo que $1$ es un \omterm{def:g11:quad:discriminant}{raíz}; el producto de la
\omterm{def:g11:quad:discriminant}{raíces} es $6$, por lo que el otro \omterm{def:g11:quad:discriminant}{raíz} es $6$.

$2x^2 + 5x - 7$: $2 + 5 - 7 = 0$, por lo que $1$ es un \omterm{def:g11:quad:discriminant}{raíz}; el producto es
$\frac{c}{a} = -\frac72$, por lo que el otro \omterm{def:g11:quad:discriminant}{raíz} es $-\frac72$.

$x^2 + 4x - 5$: $1 + 4 - 5 = 0$, por lo que $1$ es un \omterm{def:g11:quad:discriminant}{raíz}; el producto es $-5$, entonces
el otro \omterm{def:g11:quad:discriminant}{raíz} es $-5$.
\end{solution}

\begin{solution}{exo:g11:quad:6}
$\Delta = (m-2)^2 - 4$. Exactamente una solución cuando $\Delta = 0$:
$(m-2)^2 = 4$, entonces $m - 2 = \pm 2$, es decir\ $m = 0$ o $m = 4$. Ninguna solución
cuando $\Delta < 0$: $(m-2)^2 < 4$, es decir\ $-2 < m - 2 < 2$, es decir
$0 < m < 4$.
\end{solution}

\begin{solution}{exo:g11:quad:7}
El vértice está en $t = -\frac{20}{2 \times (-5)} = 2$ y
$h(2) = -20 + 40 + 1 = 21$: altura máxima $21$ m, alcanzada después de $2$
segundos.

$h(t) \geq 16$ significa $-5t^2 + 20t - 15 \geq 0$, es decir\ (dividiendo por $-5$,
lo que invierte la desigualdad) $t^2 - 4t + 3 \leq 0$. El \omterm{def:g11:quad:discriminant}{raíces} de
$t^2-4t+3$ son $1$ y $3$ ($1 - 4 + 3 = 0$, producto $3$), por lo que
$t^2 - 4t + 3 \leq 0$ exactamente en $\intcc{1}{3}$: la pelota está al menos
$16$ m de altura entre los tiempos $t = 1$ s y $t = 3$ s.
\end{solution}

\begin{solution}{exo:g11:quad:8}
Con $X = x^2$: $X^2 - 13X + 36 = 0$, $\Delta = 169 - 144 = 25$, entonces
$X = \frac{13 \pm 5}{2} = 9$ o $4$. Volver a $x$: $x^2 = 9$ da
$x = \pm 3$ y $x^2 = 4$ dan $x = \pm 2$. El conjunto solución es
$\{-3, -2, 2, 3\}$.
\end{solution}

\begin{solution}{exo:g11:quad:9}
La condición es $x^2 < x + 2$, es decir,$x^2 - x - 2 < 0$. desde
$x^2 - x - 2 = (x+1)(x-2)$ (\omterm{def:g11:quad:discriminant}{raíces} $-1$ y $2$), la desigualdad se cumple
estrictamente entre \omterm{def:g11:quad:discriminant}{raíces}: $x \in \intoo{-1}{2}$. En un boceto, el
\omterm{def:g11:quad:parabola}{parábola} $y = x^2$ cae por debajo de la línea $y = x+2$ exactamente entre sus dos
\omterm{def:g10:numbers:interunion}{intersección} puntos $(-1, 1)$ y $(2, 4)$.
\end{solution}

\begin{solution}{exo:g11:quad:10}
Deje que $x > 0$ satisfaga a $x + \frac1x = \frac{13}{6}$. Multiplicando por $6x > 0$:
$6x^2 + 6 = 13x$, es decir$6x^2 - 13x + 6 = 0$. entonces
$\Delta = 169 - 144 = 25$ y $x = \frac{13 \pm 5}{12}$, dando
$x = \frac32$ o $x = \frac23$. Ambos son positivos y recíprocos de
entre sí, como se esperaba: el número es $\frac32$ (o $\frac23$).
\end{solution}

\begin{solution}{exo:g11:quad:11}
\emph{1.} \omterm{def:g10:numbers:interunion}{Intersecciones} corresponden a soluciones de $x^2 = mx - 1$, es decir\
$x^2 - mx + 1 = 0$, cuyo \omterm{def:g11:quad:discriminant}{discriminante} es $\Delta = m^2 - 4$. Dos puntos
cuando $\abs{m} > 2$; exactamente uno cuando $m = \pm 2$; ninguno cuando $-2 < m < 2$.

\emph{2.} Para $m = 2$: $x^2 - 2x + 1 = (x-1)^2 = 0$, por lo que el contacto único
El punto es $(1, 1)$. Para $m = -2$: $(x+1)^2 = 0$, punto de contacto $(-1, 1)$.
En ambos casos la línea se encuentra con \omterm{def:g11:quad:parabola}{parábola} en un solo punto sin
cruzándolo --- es la tangente allí, y de hecho su \omterm{def:g10:reffunc:affine}{pendiente} $m = \pm 2$
es igual al valor en $x = \pm 1$ de la derivada $2x$ de $x^2$
(\cref{ch:g11:deriv}).
\end{solution}

\begin{solution}{pb:g11:quad:1}
\textbf{1.} $x^2 - 5x + 6$: $\Delta = 1$, \omterm{def:g11:quad:discriminant}{raíces} $2$ y $3$.
$2x^2 + x - 6$: $\Delta = 1 + 48 = 49$, \omterm{def:g11:quad:discriminant}{raíces}
$\frac{-1 \pm 7}{4}$: $\frac32$ y $-2$. $x^2 + x + 1$:
$\Delta = -3 < 0$: no hay una solución real.

\textbf{2.} $\Delta = 16 + 48 = 64$: \omterm{def:g11:quad:discriminant}{raíces} $3$ y $-1$, entonces
$2x^2 - 4x - 6 = 2(x - 3)(x + 1)$; \omterm{thm:g11:quad:canonical}{forma canónica}
$2(x - 1)^2 - 8$. La forma factorizada muestra el \omterm{def:g11:quad:discriminant}{raíces}, el
\omterm{thm:g11:quad:canonical}{forma canónica} el vértice $(1, -8)$, la forma expandida
$y$-intercepción $-6$.

\textbf{3.} \omterm{def:g11:quad:discriminant}{Raíces} $1$ y $3$, coeficiente principal negativo:
el trinomio es $\geq 0$ \emph{entre} el \omterm{def:g11:quad:discriminant}{raíces}:
$x \in \intcc{1}{3}$.

\textbf{4.} Los números son el \omterm{def:g11:quad:discriminant}{raíces} de $x^2 - 7x + 12 = 0$:
$3$ y $4$. Si $2$ es un \omterm{def:g11:quad:discriminant}{raíz} de $x^2 - 5x + c$, el otro \omterm{def:g11:quad:discriminant}{raíz}
$r$ satisface $2 + r = 5$ (suma), por lo que $r = 3$, y
$c = 2 \times 3 = 6$ (producto): Viète lo lee.

\textbf{5.} $\Delta = m^2 - 36$: doble \omterm{def:g11:quad:discriminant}{raíz} para $m = \pm 6$
(\omterm{def:g11:quad:discriminant}{raíz}$\mp 3$); no \omterm{def:g11:quad:discriminant}{raíz} para $-6 < m < 6$; dos \omterm{def:g11:quad:discriminant}{raíces} para
$\abs m > 6$.

\textbf{6.} $PF^2 = x^2 + \left(x^2 - \frac14\right)^2
= x^2 + x^4 - \frac{x^2}{2} + \frac{1}{16}
= x^4 + \frac{x^2}{2} + \frac{1}{16}
= \left(x^2 + \frac14\right)^2$. Por lo tanto
$PF = x^2 + \frac14$ (ambos factores positivos).

\textbf{7.} La distancia desde $P(x, x^2)$ a la horizontal
la línea $y = -\frac14$ es $x^2 + \frac14$ --- exactamente $PF$. cada
El punto del \omterm{def:g11:quad:parabola}{parábola} es equidistante del \omterm{pb:g11:quad:1}{enfocar} y del
\omterm{pb:g11:quad:1}{directora}.

\textbf{8.} $PF^2 = x^2 + \left(y - \frac14\right)^2$ y
$\operatorname{dist}^2 = \left(y + \frac14\right)^2$. Igualdad
da $x^2 + y^2 - \frac y2 + \frac{1}{16} = y^2 + \frac y2 +
\frac1{16}$, entonces $x^2 = y$: el lugar geométrico es exactamente \omterm{def:g11:quad:parabola}{parábola}.
Un punto y una recta generan la curva, como un punto y una
La distancia generó el círculo.

\textbf{9.} Para $y = \frac{x^2}{4f}$, la misma expansión con
$F(0, f)$ da $PF^2 = x^2 + \left(\frac{x^2}{4f} - f\right)^2
= \left(\frac{x^2}{4f} + f\right)^2$: la distancia a $F$ es igual
distancia a $y = -f$. Plato: $9 = \frac{30^2}{4f}$ da
$4f = 100$, $f = 25$ cm: el receptor cuelga un cuarto de metro
encima del vértice --- fuera del cuenco $9$ cm, como platos reales
espectáculo.

\textbf{10.} Un frente de onda que llega paralelo al eje tiene
viajó distancias iguales para alcanzar el nivel del \omterm{pb:g11:quad:1}{directora};
La equidistancia convierte ``distancia igual al \omterm{pb:g11:quad:1}{directora}'' en
``distancia igual a $F$'': todas las señales reflejadas llegan a
$F$ \emph{al paso}, reforzando --- ahí es donde escuchar.
La prueba honesta necesita la \omterm{def:g10:reffunc:affine}{pendiente} de la tangente en cada punto ---
el derivado de \cref{ch:g11:deriv}, vislumbrado en
\cref{exo:g11:quad:11}.

\textbf{11.} $y = x\left(1 - \frac{x}{20}\right)$: aterriza en
$x = 20$m. Vértice medio: $x = 10$, altura
$y(10) = 10 - 5 = 5$m.

\textbf{12.} $y(4) = 4 - \frac{16}{20} = 3.2$ m: borra $3$ m
con espacio, choca contra una pared de $4$ m.

\textbf{13.} $y(10) = 10 - \frac{100}{40} = 7.5$ m $> 7$: el
el camión pasa a $10$ m descentrado. $y(15) = 10 - \frac{225}{40}
= 4.375$ m: no en $15$ m.

\textbf{14.} $R(p) = -2p^2 + 120p$: vértice en
$p = \frac{120}{4} = 30$ euros, ingresos
$R(30) = 30 \times 60 = 1\,800$ euros.

\textbf{15.} $y(320) = 143 \times \left(\frac{320}{640}\right)^2
= \frac{143}{4} \approx 36$ m por encima del punto más bajo.

\textbf{16.} $x^2 = mx - 1$ da $x^2 - mx + 1 = 0$,
$\Delta = m^2 - 4$: dos puntos por $\abs m > 2$, ninguno por
$\abs m < 2$, tangencia para $m = \pm 2$ con contacto
$x = \frac m2 = \pm 1$: tangentes $y = 2x - 1$ en $(1, 1)$ y
$y = -2x - 1$ en $(-1, 1)$.

\textbf{17.} $x^2 = mx + 1$ da $\Delta = m^2 + 4 > 0$ por
cada $m$: dos \omterm{def:g10:numbers:interunion}{intersecciones} siempre. Desde un punto interior,
cada línea cruza la curva dos veces --- sólo se dibujan tangentes
desde fuera, exactamente como en el caso de los círculos.

\textbf{18.} $y = \frac14$ se encuentra con $y = x^2$ en $x = \pm\frac12$:
longitud de la cuerda $1$. Para $y = \frac{x^2}{4f}$: $y = f$ da
$x^2 = 4f^2$, $x = \pm 2f$: longitud $4f$ --- el ancho focal es
cuatro veces la distancia focal (los diseñadores de antenas lo llaman el
lado recto).

\textbf{19.} Un punto de $y = x^2$ es $(a, a^2)$; su \omterm{def:g10:functions:function}{imagen} es
$(X, Y) = (ka, ka^2)$ y $\frac{X^2}{k} = \frac{k^2a^2}{k} =
ka^2 = Y$: la curva \omterm{def:g10:functions:function}{imagen} es exactamente $y = \frac{x^2}{k}$.
Al elegir $k = 4f$ se llega a todos los \omterm{def:g11:quad:parabola}{parábola}: son todos
Ampliaciones de una curva maestra. (Los círculos son todos similares
también --- pero las elipses no lo son: aplastar importa, hacer zoom no.)

\textbf{20.} Un \omterm{def:g10:algebra:equation}{ecuación} con tres disfraces legibles; el
lugar de equidistancia entre un punto y una recta; el reflector
que recoge rayos paralelos en su \omterm{pb:g11:quad:1}{enfocar} y el cable que
lleva una cubierta uniforme; el recorrido de cada bola lanzada; y,
hasta aumento, una sola curva --- una \omterm{def:g11:quad:parabola}{parábola},
en todas partes.
\end{solution}
