\section*{Chapter \ref{ch:g11:stat} --- Descriptive Statistics}

\begin{solution}{exo:g11:stat:1}
$n = 8$, sum $71$: \omterm{def:g11:stat:mean}{mean} $\bar x = \frac{71}{8} = 8.875$. \omterm{def:g11:stat:median}{Median}: \omterm{prop:g10:coordgeom:midpoint}{midpoint}
of the $4$th and $5$th values, $\frac{8 + 9}{2} = 8.5$. \omterm{def:g11:stat:quartiles}{Quartiles}: a
quarter of $8$ is $2$, so $Q_1$ is the $2$nd value, $7$; three quarters is
$6$, so $Q_3$ is the $6$th value, $10$. \omterm{def:g11:stat:quartiles}{Interquartile range}:
$10 - 7 = 3$.
\end{solution}

\begin{solution}{exo:g11:stat:2}
Sum $40$, so $\bar x = 5$. Sum of squares:
$4 + 16 + 16 + 16 + 25 + 25 + 49 + 81 = 232$. By the shortcut formula,
\[
V = \frac{232}{8} - 5^2 = 29 - 25 = 4,
\qquad
\sigma = 2 .
\]
\end{solution}

\begin{solution}{exo:g11:stat:3}
\omterm{def:g11:stat:mean}{Mean}:
\[
\bar x = \frac{1 \times 6 + 2 \times 9 + 3 \times 8 + 4 \times 10
+ 5 \times 9 + 6 \times 8}{50}
= \frac{181}{50} = 3.62 .
\]
\omterm{def:g11:stat:median}{Median}: the cumulative frequencies are $6, 15, 23, 33, \dots$, so the
$25$th and $26$th sorted outcomes are both $4$: the \omterm{def:g11:stat:median}{median} is $4$.
\end{solution}

\begin{solution}{exo:g11:stat:4}
The total of the $15$ marks is $15 \times 11.2 = 168$; with the
newcomer, $\frac{168 + 18}{16} = \frac{186}{16} = 11.625$.
\end{solution}

\begin{solution}{exo:g11:stat:5}
$\frac{6 + 8 + x + 12 + 14}{5} = 10$ gives $40 + x = 50$, so $x = 10$.
The deviations of $6, 8, 10, 12, 14$ from the \omterm{def:g11:stat:mean}{mean} $10$ are
$-4, -2, 0, 2, 4$, with squares summing to $40$:
$V = \frac{40}{5} = 8$ and $\sigma = 2\sqrt2 \approx 2.83$.
\end{solution}

\begin{solution}{exo:g11:stat:6}
By \cref{prop:g11:stat:affine} with $a = \frac12$, $b = 4$: new \omterm{def:g11:stat:mean}{mean}
$\frac{12.4}{2} + 4 = 10.2$; new \omterm{def:g11:stat:variance}{standard deviation}
$\frac{2.5}{2} = 1.25$.
\end{solution}

\begin{solution}{exo:g11:stat:7}
The pooled \omterm{def:g11:stat:mean}{mean} is the \emph{weighted} average
\[
\frac{20 \times 10 + 30 \times 15}{50} = \frac{650}{50} = 13,
\]
not $12.5$: group B contributes more values, so it pulls the pooled \omterm{def:g11:stat:mean}{mean}
towards its own.
\end{solution}

\begin{solution}{exo:g11:stat:8}
Both lines are nearly on target on average ($1.002$ and $1.001$ kg), but
line 2's \omterm{def:g11:stat:variance}{standard deviation} is three times larger: its bags vary much
more, producing far more noticeably under- and over-filled bags. Line 2
should be dealt with first --- consistency, not just the \omterm{def:g11:stat:mean}{mean}, is what
needs fixing.
\end{solution}

\begin{solution}{exo:g11:stat:9}
$\bar x = \frac{70}{10} = 7$;
$V = \frac{540}{10} - 7^2 = 54 - 49 = 5$;
$\sigma = \sqrt5 \approx 2.24$.
\end{solution}

\begin{solution}{exo:g11:stat:10}
\emph{1.} With $23$: sum $50$, \omterm{def:g11:stat:mean}{mean} $5$; \omterm{def:g11:stat:median}{median} = \omterm{prop:g10:coordgeom:midpoint}{midpoint} of $5$th and
$6$th values $= 3$. Without $23$: sum $27$, \omterm{def:g11:stat:mean}{mean} $3$; \omterm{def:g11:stat:median}{median} (5th of 9
values) $= 3$.

\emph{2.} With $23$ ($n = 10$): $Q_1$ is the $3$rd value ($2$), $Q_3$ the
$8$th ($4$): \omterm{def:g11:stat:quartiles}{interquartile range} $2$. Without ($n = 9$): $Q_1$ is the
$3$rd value ($2$), $Q_3$ the $7$th ($4$): still $2$.

\emph{3.} The single outlier drags the \omterm{def:g11:stat:mean}{mean} from $3$ up to $5$ but leaves
the \omterm{def:g11:stat:median}{median} and the \omterm{def:g11:stat:quartiles}{interquartile range} unchanged: those two are
\emph{robust}. When a dataset may contain aberrant values, report the
\omterm{def:g11:stat:median}{median} and \omterm{def:g11:stat:quartiles}{interquartile range} rather than the \omterm{def:g11:stat:mean}{mean} and \omterm{def:g11:stat:variance}{standard
deviation}.
\end{solution}

\begin{solution}{pb:g11:stat:1}
\textbf{1.} \omterm{def:g11:stat:mean}{Mean} $\bar x = \frac{25}{5} = 5$. Deviations
squared: $1, 9, 1, 0, 9$: \omterm{def:g11:stat:variance}{variance} $\frac{20}{5} = 4$, so
$\sigma = 2$.

\textbf{2.} \omterm{def:g11:stat:mean}{Mean} of squares:
$\frac{16 + 64 + 36 + 25 + 4}{5} = 29$; minus
$\bar x^2 = 25$: \omterm{def:g11:stat:variance}{variance} $4$. Same answer, half the work.

\textbf{3.} \omterm{def:g11:stat:mean}{Mean} $= \frac{3 \times 8 + 5 \times 10 + 2 \times
14}{10} = 10.2$; \omterm{def:g11:stat:variance}{variance} $= \frac{3 \times 64 + 5 \times 100 +
2 \times 196}{10} - 10.2^2 = 108.4 - 104.04 = 4.36$;
$\sigma \approx 2.09$.

\textbf{4.} \omterm{def:g11:stat:mean}{Mean}: $1.8 \times 20 + 32 = 68\,^\circ$F;
$\sigma$: $1.8 \times 3 = 5.4\,^\circ$F --- the shift $+32$
spreads nothing.

\textbf{5.} Adding $b$: every deviation $x_i + b - (\bar x + b)
= x_i - \bar x$ is unchanged, so the \omterm{def:g11:stat:variance}{variance} is. Multiplying
by $a$: deviations are multiplied by $a$, their squares by
$a^2$, the \omterm{def:g11:stat:variance}{variance} by $a^2$, and $\sigma$ by
$\sqrt{a^2} = \abs a$.

\textbf{6.} $z_A = \frac{15 - 12}{2} = 1.5$;
$z_B = \frac{13 - 8}{4} = 1.25$. Alice stands further above her
crowd: the harder exam's $15$ beats the easier exam's $13$.

\textbf{7.} $z$-scores: $-0.5,\ 1.5,\ 0.5,\ 0,\ -1.5$: \omterm{def:g11:stat:mean}{mean}
$0$, \omterm{def:g11:stat:variance}{standard deviation} $1$. In general, standardizing
subtracts $\bar x$ (making the \omterm{def:g11:stat:mean}{mean} $0$, $\sigma$ untouched)
then divides by $\sigma$ (making the new $\sigma$ equal to
$1$): question 5 twice.

\textbf{8.} $z = 2$: $100 + 2 \times 15 = 130$. And
$z(76) = \frac{76 - 100}{15} = -1.6$.

\textbf{9.} The baby is $2.5$ \omterm{def:g11:stat:variance}{standard deviations} below the
\omterm{def:g11:stat:mean}{mean} weight \emph{of babies its age} --- alarmingly light for
its cohort. Raw kilograms cannot be compared across ages (all
babies gain weight); \omterm{pb:g11:stat:1}{z-scores} measure each child against its
own age group, one ruler for all ages.

\textbf{10.} \omterm{def:g11:stat:mean}{Mean} $5$; \omterm{def:g11:stat:mean}{mean} of squares $\frac{622}{10} = 62.2$,
\omterm{def:g11:stat:variance}{variance} $37.2$, $\sigma \approx 6.10$;
$z(23) = \frac{18}{6.10} \approx 2.95$: just under the alarm
threshold --- the test barely blinks.

\textbf{11.} Without $23$: \omterm{def:g11:stat:mean}{mean} $3$, \omterm{def:g11:stat:variance}{variance}
$\frac{93}{9} - 9 \approx 1.33$, $\sigma \approx 1.15$;
against the clean ruler, $23$ scores
$z \approx \frac{20}{1.15} \approx 17$: a monster. The outlier
inflated both the \omterm{def:g11:stat:mean}{mean} and $\sigma$, camouflaging itself ---
which is why \cref{exo:g11:stat:10} recommends robust
summaries (\omterm{def:g11:stat:median}{median}, IQR) when contamination threatens.

\textbf{12.} In the \omterm{def:g11:stat:variance}{variance} sum
$\sigma^2 = \frac1N \sum (x_i - \bar x)^2$, keep only the
terms with $\abs{x_i - \bar x} \geq k\sigma$: there are $pN$
of them, each at least $(k\sigma)^2$, so
$\sigma^2 \geq \frac{1}{N} \cdot pN \cdot k^2\sigma^2 =
p\,k^2\sigma^2$: divide by $k^2 \sigma^2$ (nonzero spread):
$p \leq \frac{1}{k^2}$.

\textbf{13.} Beyond $2\sigma$: at most $\frac14 = 25\,\%$ of
any dataset whatsoever; beyond $3\sigma$: at most
$\frac19 \approx 11\,\%$. Question 1's data: beyond
$2\sigma = 4$ from the \omterm{def:g11:stat:mean}{mean} $5$ means $\leq 1$ or $\geq 9$:
none of the five values --- $0\,\% \leq 25\,\%$, as promised.

\textbf{14.} Outside $\intcc{49.7}{50.3}$ means beyond
$3\sigma$: at most $\frac19$ of production, \emph{whatever}
the shape of the distribution --- Chebyshev asks nothing but a
\omterm{def:g11:stat:mean}{mean} and a $\sigma$. Bell-shaped production concentrates far
better (about $0.3\,\%$ beyond $3\sigma$): the normal law,
grade 12.

\textbf{15.} $-19\,\%$ sits $24$ points below the \omterm{def:g11:stat:mean}{mean} $5$.
Fund B: $24 = 2\sigma_B$: Chebyshev allows up to
$\frac14$ of years that bad. Fund A: $24 = 12\sigma_A$: at
most $\frac{1}{144}$. Same average, incomparable dangers ---
$\sigma$ is what finance calls \emph{risk} (or volatility).

\textbf{16.} Expand around $\bar x$:
$(x_i - c)^2 = \left((x_i - \bar x) + (\bar x - c)\right)^2
= (x_i - \bar x)^2 + 2(\bar x - c)(x_i - \bar x) +
(\bar x - c)^2$. Averaging: the middle term dies (deviations
from the \omterm{def:g11:stat:mean}{mean} sum to zero), leaving
$f(c) = \sigma^2 + (\bar x - c)^2$ --- minimal exactly at
$c = \bar x$, with \omterm{def:g10:functions:extrema}{minimum} $\sigma^2$. The \omterm{def:g11:stat:mean}{mean} is the
least-squares center, as the \omterm{def:g11:stat:median}{median} is the least-absolute
center (\cref{pb:g10:stats:1}).

\textbf{17.} Directly: deviations from $6$ are
$-2, 2, 0, -1, -4$, squares $4, 4, 0, 1, 16$:
$f(6) = \frac{25}{5} = 5$. Formula:
$\sigma^2 + (5 - 6)^2 = 4 + 1 = 5$. Agreement.

\textbf{18.} Both have \omterm{def:g11:stat:mean}{mean} $7$; \omterm{def:g11:stat:variance}{variances}:
$P$: $\frac{16 + 4 + 0 + 4 + 16}{5} = 8$;
$Q$: $\frac{36 + 0 + 0 + 0 + 36}{5} = 14.4$: different spreads
behind one \omterm{def:g11:stat:mean}{mean}. For $\{7 - a, 7, 7, 7, 7 + a\}$: \omterm{def:g11:stat:variance}{variance}
$\frac{2a^2}{5} = 8$ gives $a = \sqrt{20} \approx 4.47$: the
dataset $\{2.53, 7, 7, 7, 11.47\}$ carries the exact \omterm{def:g11:stat:mean}{mean} and
\omterm{def:g11:stat:variance}{variance} of $P$ with a completely different profile ---
summaries compress, and compression loses.

\textbf{19.} $\sigma^2 = 0$ means a sum of squares
$(x_i - \bar x)^2$ vanishes; squares are $\geq 0$, so a zero
sum forces every term to be zero: $x_i = \bar x$ for all $i$
--- all values equal.

\textbf{20.} Centers say where the data lives; spreads say how
loosely; the \omterm{pb:g11:stat:1}{z-score} lays one ruler across all datasets,
exams, ages and currencies; Chebyshev converts $\sigma$ into
an unconditional guarantee; and question 18 keeps us honest
--- no handful of numbers ever contains the whole story. Next
year the bell curve sharpens the ruler into exact
probabilities.
\end{solution}
