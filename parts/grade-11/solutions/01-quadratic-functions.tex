\section*{Chapter \ref{ch:g11:quad} --- Quadratic Functions and Equations}

\begin{solution}{exo:g11:quad:1}
$x^2-5x+6$: $\Delta = 25 - 24 = 1$, roots $\frac{5 \pm 1}{2}$, i.e.\ $2$
and $3$.

$2x^2+3x-2$: $\Delta = 9 + 16 = 25$, roots $\frac{-3 \pm 5}{4}$, i.e.\
$\frac12$ and $-2$.

$x^2+2x+5$: $\Delta = 4 - 20 = -16 < 0$, no real solution.

$9x^2-6x+1$: $\Delta = 36 - 36 = 0$, one (double) root
$x_0 = \frac{6}{18} = \frac13$.
\end{solution}

\begin{solution}{exo:g11:quad:2}
$f(x) = x^2 - 6x + 11 = (x-3)^2 - 9 + 11 = (x-3)^2 + 2$: vertex $(3, 2)$.

$g(x) = -2x^2+4x+1 = -2(x^2 - 2x) + 1 = -2\bigl((x-1)^2 - 1\bigr) + 1
= -2(x-1)^2 + 3$: vertex $(1, 3)$.

$h(x) = 3x^2 + 6x = 3(x^2 + 2x) = 3\bigl((x+1)^2 - 1\bigr)
= 3(x+1)^2 - 3$: vertex $(-1, -3)$.
\end{solution}

\begin{solution}{exo:g11:quad:3}
$x^2 - 3x - 10$: $\Delta = 9 + 40 = 49$, roots $\frac{3 \pm 7}{2} = -2$
and $5$; positive leading coefficient, so the trinomial is negative
strictly between the roots: the solution set is $\intoo{-2}{5}$.

$2x^2 + x + 3$: $\Delta = 1 - 24 = -23 < 0$ and $a = 2 > 0$: always
positive, the solution set is $\R$.

$-x^2 + 6x - 9 = -(x-3)^2$: it is $\geq 0$ only where $(x-3)^2 \leq 0$,
that is at $x = 3$. The solution set is $\{3\}$.
\end{solution}

\begin{solution}{exo:g11:quad:4}
$x^2 - 7x + 6$: $1 - 7 + 6 = 0$, so $1$ is a root; the product of the
roots is $6$, so the other root is $6$.

$2x^2 + 5x - 7$: $2 + 5 - 7 = 0$, so $1$ is a root; the product is
$\frac{c}{a} = -\frac72$, so the other root is $-\frac72$.

$x^2 + 4x - 5$: $1 + 4 - 5 = 0$, so $1$ is a root; the product is $-5$, so
the other root is $-5$.
\end{solution}

\begin{solution}{exo:g11:quad:5}
The two numbers solve $x^2 - 14x + 45 = 0$ (\cref{prop:g11:quad:vieta});
$\Delta = 196 - 180 = 16$, roots $\frac{14 \pm 4}{2} = 9$ and $5$.

For the rectangle: half the perimeter gives length $+$ width $= 17$, and
the area gives their product $60$. They solve $x^2 - 17x + 60 = 0$;
$\Delta = 289 - 240 = 49$, roots $\frac{17 \pm 7}{2} = 12$ and $5$. The
rectangle is $12 \times 5$.
\end{solution}

\begin{solution}{exo:g11:quad:6}
$\Delta = (m-2)^2 - 4$. Exactly one solution when $\Delta = 0$:
$(m-2)^2 = 4$, so $m - 2 = \pm 2$, i.e.\ $m = 0$ or $m = 4$. No solution
when $\Delta < 0$: $(m-2)^2 < 4$, i.e.\ $-2 < m - 2 < 2$, that is
$0 < m < 4$.
\end{solution}

\begin{solution}{exo:g11:quad:7}
The vertex is at $t = -\frac{20}{2 \times (-5)} = 2$, and
$h(2) = -20 + 40 + 1 = 21$: maximal height $21$ m, reached after $2$
seconds.

$h(t) \geq 16$ means $-5t^2 + 20t - 15 \geq 0$, i.e.\ (dividing by $-5$,
which reverses the inequality) $t^2 - 4t + 3 \leq 0$. The roots of
$t^2-4t+3$ are $1$ and $3$ ($1 - 4 + 3 = 0$, product $3$), so
$t^2 - 4t + 3 \leq 0$ exactly on $\intcc{1}{3}$: the ball is at least
$16$ m high between the times $t = 1$ s and $t = 3$ s.
\end{solution}

\begin{solution}{exo:g11:quad:8}
With $X = x^2$: $X^2 - 13X + 36 = 0$, $\Delta = 169 - 144 = 25$, so
$X = \frac{13 \pm 5}{2} = 9$ or $4$. Back to $x$: $x^2 = 9$ gives
$x = \pm 3$, and $x^2 = 4$ gives $x = \pm 2$. The solution set is
$\{-3, -2, 2, 3\}$.
\end{solution}

\begin{solution}{exo:g11:quad:9}
The condition is $x^2 < x + 2$, i.e.\ $x^2 - x - 2 < 0$. Since
$x^2 - x - 2 = (x+1)(x-2)$ (roots $-1$ and $2$), the inequality holds
strictly between the roots: $x \in \intoo{-1}{2}$. On a sketch, the
parabola $y = x^2$ dips below the line $y = x+2$ exactly between their two
intersection points $(-1, 1)$ and $(2, 4)$.
\end{solution}

\begin{solution}{exo:g11:quad:10}
Let $x > 0$ satisfy $x + \frac1x = \frac{13}{6}$. Multiplying by $6x > 0$:
$6x^2 + 6 = 13x$, i.e.\ $6x^2 - 13x + 6 = 0$. Then
$\Delta = 169 - 144 = 25$ and $x = \frac{13 \pm 5}{12}$, giving
$x = \frac32$ or $x = \frac23$. Both are positive and are reciprocals of
each other, as expected: the number is $\frac32$ (or $\frac23$).
\end{solution}

\begin{solution}{exo:g11:quad:11}
\emph{1.} Intersections correspond to solutions of $x^2 = mx - 1$, i.e.\
$x^2 - mx + 1 = 0$, whose discriminant is $\Delta = m^2 - 4$. Two points
when $\abs{m} > 2$; exactly one when $m = \pm 2$; none when $-2 < m < 2$.

\emph{2.} For $m = 2$: $x^2 - 2x + 1 = (x-1)^2 = 0$, so the unique contact
point is $(1, 1)$. For $m = -2$: $(x+1)^2 = 0$, contact point $(-1, 1)$.
In both cases the line meets the parabola at a single point without
crossing it --- it is the tangent there, and indeed its slope $m = \pm 2$
equals the value at $x = \pm 1$ of the derivative $2x$ of $x^2$
(\cref{ch:g11:deriv}).
\end{solution}
