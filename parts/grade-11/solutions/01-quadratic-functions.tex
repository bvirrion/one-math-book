\section*{Chapter \ref{ch:g11:quad} --- Quadratic Functions and Equations}

\begin{solution}{exo:g11:quad:1}
$x^2-5x+6$: $\Delta = 25 - 24 = 1$, \omterm{def:g11:quad:discriminant}{roots} $\frac{5 \pm 1}{2}$, i.e.\ $2$
and $3$.

$2x^2+3x-2$: $\Delta = 9 + 16 = 25$, \omterm{def:g11:quad:discriminant}{roots} $\frac{-3 \pm 5}{4}$, i.e.\
$\frac12$ and $-2$.

$x^2+2x+5$: $\Delta = 4 - 20 = -16 < 0$, no real solution.

$9x^2-6x+1$: $\Delta = 36 - 36 = 0$, one (double) \omterm{def:g11:quad:discriminant}{root}
$x_0 = \frac{6}{18} = \frac13$.
\end{solution}

\begin{solution}{exo:g11:quad:2}
$f(x) = x^2 - 6x + 11 = (x-3)^2 - 9 + 11 = (x-3)^2 + 2$: vertex $(3, 2)$.

$g(x) = -2x^2+4x+1 = -2(x^2 - 2x) + 1 = -2\bigl((x-1)^2 - 1\bigr) + 1
= -2(x-1)^2 + 3$: vertex $(1, 3)$.

$h(x) = 3x^2 + 6x = 3(x^2 + 2x) = 3\bigl((x+1)^2 - 1\bigr)
= 3(x+1)^2 - 3$: vertex $(-1, -3)$.
\end{solution}

\begin{solution}{exo:g11:quad:3}
$x^2 - 3x - 10$: $\Delta = 9 + 40 = 49$, \omterm{def:g11:quad:discriminant}{roots} $\frac{3 \pm 7}{2} = -2$
and $5$; positive leading coefficient, so the trinomial is negative
strictly between the \omterm{def:g11:quad:discriminant}{roots}: the solution set is $\intoo{-2}{5}$.

$2x^2 + x + 3$: $\Delta = 1 - 24 = -23 < 0$ and $a = 2 > 0$: always
positive, the solution set is $\R$.

$-x^2 + 6x - 9 = -(x-3)^2$: it is $\geq 0$ only where $(x-3)^2 \leq 0$,
that is at $x = 3$. The solution set is $\{3\}$.
\end{solution}

\begin{solution}{exo:g11:quad:4}
$x^2 - 7x + 6$: $1 - 7 + 6 = 0$, so $1$ is a \omterm{def:g11:quad:discriminant}{root}; the product of the
\omterm{def:g11:quad:discriminant}{roots} is $6$, so the other \omterm{def:g11:quad:discriminant}{root} is $6$.

$2x^2 + 5x - 7$: $2 + 5 - 7 = 0$, so $1$ is a \omterm{def:g11:quad:discriminant}{root}; the product is
$\frac{c}{a} = -\frac72$, so the other \omterm{def:g11:quad:discriminant}{root} is $-\frac72$.

$x^2 + 4x - 5$: $1 + 4 - 5 = 0$, so $1$ is a \omterm{def:g11:quad:discriminant}{root}; the product is $-5$, so
the other \omterm{def:g11:quad:discriminant}{root} is $-5$.
\end{solution}

\begin{solution}{exo:g11:quad:5}
The two numbers solve $x^2 - 14x + 45 = 0$ (\cref{prop:g11:quad:vieta});
$\Delta = 196 - 180 = 16$, \omterm{def:g11:quad:discriminant}{roots} $\frac{14 \pm 4}{2} = 9$ and $5$.

For the rectangle: half the perimeter gives length $+$ width $= 17$, and
the area gives their product $60$. They solve $x^2 - 17x + 60 = 0$;
$\Delta = 289 - 240 = 49$, \omterm{def:g11:quad:discriminant}{roots} $\frac{17 \pm 7}{2} = 12$ and $5$. The
rectangle is $12 \times 5$.
\end{solution}

\begin{solution}{exo:g11:quad:6}
$\Delta = (m-2)^2 - 4$. Exactly one solution when $\Delta = 0$:
$(m-2)^2 = 4$, so $m - 2 = \pm 2$, i.e.\ $m = 0$ or $m = 4$. No solution
when $\Delta < 0$: $(m-2)^2 < 4$, i.e.\ $-2 < m - 2 < 2$, that is
$0 < m < 4$.
\end{solution}

\begin{solution}{exo:g11:quad:7}
The vertex is at $t = -\frac{20}{2 \times (-5)} = 2$, and
$h(2) = -20 + 40 + 1 = 21$: maximal height $21$ m, reached after $2$
seconds.

$h(t) \geq 16$ means $-5t^2 + 20t - 15 \geq 0$, i.e.\ (dividing by $-5$,
which reverses the inequality) $t^2 - 4t + 3 \leq 0$. The \omterm{def:g11:quad:discriminant}{roots} of
$t^2-4t+3$ are $1$ and $3$ ($1 - 4 + 3 = 0$, product $3$), so
$t^2 - 4t + 3 \leq 0$ exactly on $\intcc{1}{3}$: the ball is at least
$16$ m high between the times $t = 1$ s and $t = 3$ s.
\end{solution}

\begin{solution}{exo:g11:quad:8}
With $X = x^2$: $X^2 - 13X + 36 = 0$, $\Delta = 169 - 144 = 25$, so
$X = \frac{13 \pm 5}{2} = 9$ or $4$. Back to $x$: $x^2 = 9$ gives
$x = \pm 3$, and $x^2 = 4$ gives $x = \pm 2$. The solution set is
$\{-3, -2, 2, 3\}$.
\end{solution}

\begin{solution}{exo:g11:quad:9}
The condition is $x^2 < x + 2$, i.e.\ $x^2 - x - 2 < 0$. Since
$x^2 - x - 2 = (x+1)(x-2)$ (\omterm{def:g11:quad:discriminant}{roots} $-1$ and $2$), the inequality holds
strictly between the \omterm{def:g11:quad:discriminant}{roots}: $x \in \intoo{-1}{2}$. On a sketch, the
\omterm{def:g11:quad:parabola}{parabola} $y = x^2$ dips below the line $y = x+2$ exactly between their two
\omterm{def:g10:numbers:interunion}{intersection} points $(-1, 1)$ and $(2, 4)$.
\end{solution}

\begin{solution}{exo:g11:quad:10}
Let $x > 0$ satisfy $x + \frac1x = \frac{13}{6}$. Multiplying by $6x > 0$:
$6x^2 + 6 = 13x$, i.e.\ $6x^2 - 13x + 6 = 0$. Then
$\Delta = 169 - 144 = 25$ and $x = \frac{13 \pm 5}{12}$, giving
$x = \frac32$ or $x = \frac23$. Both are positive and are reciprocals of
each other, as expected: the number is $\frac32$ (or $\frac23$).
\end{solution}

\begin{solution}{exo:g11:quad:11}
\emph{1.} \omterm{def:g10:numbers:interunion}{Intersections} correspond to solutions of $x^2 = mx - 1$, i.e.\
$x^2 - mx + 1 = 0$, whose \omterm{def:g11:quad:discriminant}{discriminant} is $\Delta = m^2 - 4$. Two points
when $\abs{m} > 2$; exactly one when $m = \pm 2$; none when $-2 < m < 2$.

\emph{2.} For $m = 2$: $x^2 - 2x + 1 = (x-1)^2 = 0$, so the unique contact
point is $(1, 1)$. For $m = -2$: $(x+1)^2 = 0$, contact point $(-1, 1)$.
In both cases the line meets the \omterm{def:g11:quad:parabola}{parabola} at a single point without
crossing it --- it is the tangent there, and indeed its \omterm{def:g10:reffunc:affine}{slope} $m = \pm 2$
equals the value at $x = \pm 1$ of the derivative $2x$ of $x^2$
(\cref{ch:g11:deriv}).
\end{solution}

\begin{solution}{pb:g11:quad:1}
\textbf{1.} $x^2 - 5x + 6$: $\Delta = 1$, \omterm{def:g11:quad:discriminant}{roots} $2$ and $3$.
$2x^2 + x - 6$: $\Delta = 1 + 48 = 49$, \omterm{def:g11:quad:discriminant}{roots}
$\frac{-1 \pm 7}{4}$: $\frac32$ and $-2$. $x^2 + x + 1$:
$\Delta = -3 < 0$: no real solution.

\textbf{2.} $\Delta = 16 + 48 = 64$: \omterm{def:g11:quad:discriminant}{roots} $3$ and $-1$, so
$2x^2 - 4x - 6 = 2(x - 3)(x + 1)$; \omterm{thm:g11:quad:canonical}{canonical form}
$2(x - 1)^2 - 8$. The factored form shows the \omterm{def:g11:quad:discriminant}{roots}, the
\omterm{thm:g11:quad:canonical}{canonical form} the vertex $(1, -8)$, the expanded form the
$y$-intercept $-6$.

\textbf{3.} \omterm{def:g11:quad:discriminant}{Roots} $1$ and $3$, leading coefficient negative:
the trinomial is $\geq 0$ \emph{between} the \omterm{def:g11:quad:discriminant}{roots}:
$x \in \intcc{1}{3}$.

\textbf{4.} The numbers are the \omterm{def:g11:quad:discriminant}{roots} of $x^2 - 7x + 12 = 0$:
$3$ and $4$. If $2$ is a \omterm{def:g11:quad:discriminant}{root} of $x^2 - 5x + c$, the other \omterm{def:g11:quad:discriminant}{root}
$r$ satisfies $2 + r = 5$ (sum), so $r = 3$, and
$c = 2 \times 3 = 6$ (product): Vi\`ete reads it off.

\textbf{5.} $\Delta = m^2 - 36$: double \omterm{def:g11:quad:discriminant}{root} for $m = \pm 6$
(\omterm{def:g11:quad:discriminant}{root} $\mp 3$); no \omterm{def:g11:quad:discriminant}{root} for $-6 < m < 6$; two \omterm{def:g11:quad:discriminant}{roots} for
$\abs m > 6$.

\textbf{6.} $PF^2 = x^2 + \left(x^2 - \frac14\right)^2
= x^2 + x^4 - \frac{x^2}{2} + \frac{1}{16}
= x^4 + \frac{x^2}{2} + \frac{1}{16}
= \left(x^2 + \frac14\right)^2$. Hence
$PF = x^2 + \frac14$ (both factors positive).

\textbf{7.} The distance from $P(x, x^2)$ to the horizontal
line $y = -\frac14$ is $x^2 + \frac14$ --- exactly $PF$. Every
point of the \omterm{def:g11:quad:parabola}{parabola} is equidistant from the \omterm{pb:g11:quad:1}{focus} and the
\omterm{pb:g11:quad:1}{directrix}.

\textbf{8.} $PF^2 = x^2 + \left(y - \frac14\right)^2$ and
$\operatorname{dist}^2 = \left(y + \frac14\right)^2$. Equality
gives $x^2 + y^2 - \frac y2 + \frac{1}{16} = y^2 + \frac y2 +
\frac1{16}$, so $x^2 = y$: the locus is exactly the \omterm{def:g11:quad:parabola}{parabola}.
One point and one line generate the curve, as one point and one
distance generated the circle.

\textbf{9.} For $y = \frac{x^2}{4f}$, the same expansion with
$F(0, f)$ gives $PF^2 = x^2 + \left(\frac{x^2}{4f} - f\right)^2
= \left(\frac{x^2}{4f} + f\right)^2$: distance to $F$ equals
distance to $y = -f$. Dish: $9 = \frac{30^2}{4f}$ gives
$4f = 100$, $f = 25$ cm: the receiver hangs a quarter meter
above the vertex --- outside the $9$ cm bowl, as real dishes
show.

\textbf{10.} A wavefront arriving parallel to the axis has
travelled equal distances to reach the \omterm{pb:g11:quad:1}{directrix}'s level;
equidistance converts ``equal distance to the \omterm{pb:g11:quad:1}{directrix}'' into
``equal distance to $F$'': all the reflected signals arrive at
$F$ \emph{in step}, reinforcing --- that is where to listen.
The honest proof needs the tangent's \omterm{def:g10:reffunc:affine}{slope} at each point ---
the derivative of \cref{ch:g11:deriv}, glimpsed in
\cref{exo:g11:quad:11}.

\textbf{11.} $y = x\left(1 - \frac{x}{20}\right)$: lands at
$x = 20$ m. Vertex midway: $x = 10$, height
$y(10) = 10 - 5 = 5$ m.

\textbf{12.} $y(4) = 4 - \frac{16}{20} = 3.2$ m: clears $3$ m
with room, smacks into a $4$ m wall.

\textbf{13.} $y(10) = 10 - \frac{100}{40} = 7.5$ m $> 7$: the
truck passes at $10$ m off-center. $y(15) = 10 - \frac{225}{40}
= 4.375$ m: not at $15$ m.

\textbf{14.} $R(p) = -2p^2 + 120p$: vertex at
$p = \frac{120}{4} = 30$ euros, revenue
$R(30) = 30 \times 60 = 1\,800$ euros.

\textbf{15.} $y(320) = 143 \times \left(\frac{320}{640}\right)^2
= \frac{143}{4} \approx 36$ m above the lowest point.

\textbf{16.} $x^2 = mx - 1$ gives $x^2 - mx + 1 = 0$,
$\Delta = m^2 - 4$: two points for $\abs m > 2$, none for
$\abs m < 2$, tangency for $m = \pm 2$ with contact
$x = \frac m2 = \pm 1$: tangents $y = 2x - 1$ at $(1, 1)$ and
$y = -2x - 1$ at $(-1, 1)$.

\textbf{17.} $x^2 = mx + 1$ gives $\Delta = m^2 + 4 > 0$ for
every $m$: two \omterm{def:g10:numbers:interunion}{intersections} always. From an interior point,
every line crosses the curve twice --- tangents are drawn only
from outside, exactly as for circles.

\textbf{18.} $y = \frac14$ meets $y = x^2$ at $x = \pm\frac12$:
chord length $1$. For $y = \frac{x^2}{4f}$: $y = f$ gives
$x^2 = 4f^2$, $x = \pm 2f$: length $4f$ --- the focal width is
four times the focal distance (dish designers call it the
latus rectum).

\textbf{19.} A point of $y = x^2$ is $(a, a^2)$; its \omterm{def:g10:functions:function}{image} is
$(X, Y) = (ka, ka^2)$, and $\frac{X^2}{k} = \frac{k^2a^2}{k} =
ka^2 = Y$: the \omterm{def:g10:functions:function}{image} curve is exactly $y = \frac{x^2}{k}$.
Choosing $k = 4f$ reaches every \omterm{def:g11:quad:parabola}{parabola}: they are all
magnifications of one master curve. (Circles are all similar
too --- but ellipses are not: squash matters, zoom does not.)

\textbf{20.} An \omterm{def:g10:algebra:equation}{equation} with three readable costumes; the
locus of equidistance from a point and a line; the reflector
that gathers parallel rays at its \omterm{pb:g11:quad:1}{focus} and the cable that
carries a uniform deck; the path of every thrown ball; and,
up to magnification, a single curve --- one \omterm{def:g11:quad:parabola}{parabola},
everywhere.
\end{solution}
