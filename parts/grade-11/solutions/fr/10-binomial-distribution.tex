\section*{Chapitre \ref{ch:g11:binom} --- La loi binomiale}

\begin{solution}{exo:g11:binom:1}
Lignes $6$ et $7$~:
\[
1,\ 6,\ 15,\ 20,\ 15,\ 6,\ 1
\qquad\text{et}\qquad
1,\ 7,\ 21,\ 35,\ 35,\ 21,\ 7,\ 1 .
\]
D'où $\binom62 = 15$, $\binom73 = 35$, $\binom74 = 35$ (la symétrie
$\binom73 = \binom74$ reflète l'échange des succès et des échecs).
\end{solution}

\begin{solution}{exo:g11:binom:2}
Nombre fixe d'épreuves ($4$ lancers), deux issues par lancer (six ou
non, $p = \frac16$), lancers indépendants~:
$X \sim \mathcal B(4, \frac16)$.
\[
\P(X=0) = \left(\frac56\right)^4 = \frac{625}{1296} \approx 0.48,
\qquad
\P(X=1) = 4 \times \frac16\left(\frac56\right)^3 = \frac{500}{1296}
\approx 0.39,
\]
\[
\P(X \geq 2) = 1 - \frac{625 + 500}{1296} = \frac{171}{1296}
\approx 0.13 .
\]
\end{solution}

\begin{solution}{exo:g11:binom:3}
\emph{1.} Binomiale $\mathcal B(20, \frac12)$~: $n$ fixe, même $p$,
lancers indépendants.

\emph{2.} Pas binomiale~: les cartes sont distribuées \emph{sans
remise}, donc la probabilité d'un as change d'une carte à l'autre et les
tirages ne sont pas indépendants.

\emph{3.} Binomiale $\mathcal B(7, 0.3)$ par l'hypothèse d'indépendance
énoncée.
\end{solution}

\begin{solution}{exo:g11:binom:4}
$\E(X) = 50 \times 0.2 = 10$~;
$\V(X) = 50 \times 0.2 \times 0.8 = 8$~;
$\sigma(X) = 2\sqrt2 \approx 2.83$.
\end{solution}

\begin{solution}{exo:g11:binom:5}
$X \sim \mathcal B(6, 0.7)$.
\[
\P(X = 4) = \binom64 (0.7)^4 (0.3)^2 = 15 \times 0.2401 \times 0.09
\approx 0.324 .
\]
\[
\P(X \geq 5) = \binom65 (0.7)^5(0.3) + (0.7)^6
= 6 \times 0.16807 \times 0.3 + 0.117649 \approx 0.420 .
\]
\end{solution}

\begin{solution}{exo:g11:binom:6}
$X \sim \mathcal B\left(8, \frac12\right)$~; chaque chemin a pour
probabilité $\frac{1}{256}$, donc
\[
\P(X \geq 6) = \frac{\binom86 + \binom87 + \binom88}{256}
= \frac{28 + 8 + 1}{256} = \frac{37}{256} \approx 0.14 .
\]
Deviner fait réussir environ une fois sur sept.
\end{solution}

\begin{solution}{exo:g11:binom:7}
Le complémentaire de «~au moins une de chaque~» est «~les cinq
identiques~»~: toutes A ou toutes B, chacune de probabilité
$\left(\frac12\right)^5 = \frac1{32}$. D'où
\[
\P(\text{un de chaque type}) = 1 - \frac{2}{32} = \frac{15}{16} .
\]
\end{solution}

\begin{solution}{exo:g11:binom:8}
$\P(X = 3) = p^3$ et $\P(X \geq 1) = 1 - (1-p)^3$. Résoudre
$p^3 = \frac{27}{64} = \left(\frac34\right)^3$ donne $p = \frac34$ (la
fonction cube est strictement croissante, \cref{ch:g11:func}, donc la
solution est unique).
\end{solution}

\begin{solution}{exo:g11:binom:9}
$\P(\text{au moins une face}) = 1 - \left(\frac12\right)^n$, donc la
condition est $\left(\frac12\right)^n < 0.01$, c'est-à-dire
$2^n > 100$. Comme $2^6 = 64$ et $2^7 = 128$~: à partir de $n = 7$
lancers.
\end{solution}

\begin{solution}{exo:g11:binom:10}
La somme de la ligne~$n$ compte tous les chemins de l'arbre à $n$
épreuves, classés par leur nombre de succès. Mais l'arbre double ses
chemins à chaque épreuve (chaque chemin se scinde en S et E), donc il a
$2^n$ chemins au total. D'où $\sum_{k} \binom nk = 2^n$.
Alternativement, par récurrence~: la ligne~$0$ somme à $1 = 2^0$, et la
règle de Pascal fait que chaque entrée de la ligne~$n$ contribue à
exactement deux entrées de la ligne~$n+1$, donc les sommes de lignes
doublent.
\end{solution}

\begin{solution}{exo:g11:binom:11}
\emph{1.} Sous l'affirmation, $X \sim \mathcal B(10, 0.6)$. En sommant
les premiers termes~:
\begin{align*}
\P(X \leq 3)
&= (0.4)^{10} + 10(0.6)(0.4)^9 + 45(0.6)^2(0.4)^8
+ 120(0.6)^3(0.4)^7\\
&\approx 0.0001 + 0.0016 + 0.0106 + 0.0425 = 0.0548 .
\end{align*}

\emph{2.} Un résultat au moins aussi extrême que celui observé ($3$
approbations ou moins) a une probabilité d'environ $5.5\,\%$ --- juste
\emph{au-dessus} du seuil de $5\,\%$. En appliquant strictement la règle,
l'observation est (de justesse) compatible avec l'affirmation et on ne
la rejette pas. L'exemple montre à quel point les décisions limites sont
sensibles au choix du seuil~: avec une convention à $6\,\%$ la
conclusion s'inverserait.
\end{solution}
