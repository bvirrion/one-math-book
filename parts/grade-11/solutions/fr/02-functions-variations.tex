\section*{Chapitre \ref{ch:g11:func} --- Fonctions et variations}

\begin{solution}{exo:g11:func:1}
$f$~: il faut $3 - x \geq 0$, \omterm{def:g11:func:function}{ensemble de définition}
$\intoc{-\infty}{3}$.

$g$~: il faut $x^2 - 4 \neq 0$, c'est-à-dire $x \neq \pm 2$~: \omterm{def:g11:func:function}{ensemble
de définition} $\R \setminus \{-2, 2\}$.

$h$~: il faut $x \geq 0$ et $x \neq 5$~: \omterm{def:g11:func:function}{ensemble de définition}
$\intco{0}{5} \cup \intoo{5}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:2}
$\sqrt{11} < \sqrt{13}$ car $\sqrt{}$ est \omterm{def:g11:func:monotone}{croissante} et $11 < 13$.

$(-2.1)^2 > (-2.05)^2$ car $x^2$ est \omterm{def:g10:functions:variations}{décroissante} sur
$\intoc{-\infty}{0}$ et $-2.1 < -2.05$.

$\frac{1}{0.9} > \frac{1}{1.1}$ car $\frac1x$ est \omterm{def:g10:functions:variations}{décroissante} sur
$\intoo{0}{+\infty}$ et $0.9 < 1.1$.

$(-1.01)^3 < (-0.99)^3$ car $x^3$ est \omterm{def:g11:func:monotone}{croissante} et $-1.01 < -0.99$.
\end{solution}

\begin{solution}{exo:g11:func:3}
$f(-x) = 3(-x)^4 - (-x)^2 = 3x^4 - x^2 = f(x)$~: \omterm{def:g11:func:parity}{paire}.

$g(-x) = -x^3 - x = -g(x)$~: \omterm{def:g11:func:parity}{impaire}.

$h(1) = 2$ et $h(-1) = 1 - 1 = 0$, donc $h(-1)$ n'est ni $h(1)$ ni
$-h(1)$~: ni \omterm{def:g11:func:parity}{paire} ni \omterm{def:g11:func:parity}{impaire}.

$k(-x) = \frac{-x}{x^2+1} = -k(x)$~: \omterm{def:g11:func:parity}{impaire}.
\end{solution}

\begin{solution}{exo:g11:func:4}
Lire le \omterm{def:g10:functions:table}{tableau de variations}. Sur $\intcc{-3}{1}$, $f$ monte de $-2$ à
$4$ et croise la valeur $1$ exactement une fois~; sur $\intcc{1}{5}$
elle descend de $4$ à $0$ et croise $1$ une fois encore~: l'\omterm{def:g10:algebra:equation}{équation}
$f(x) = 1$ a $2$ solutions. Le \omterm{def:g10:functions:extrema}{maximum} de $f$ est $f(1) = 4 < 5$, donc
$f(x) = 5$ n'a aucune solution.
\end{solution}

\begin{solution}{exo:g11:func:5}
Soit $2 \leq u < v$. Alors
\[
f(v) - f(u) = v^2 - 4v - u^2 + 4u = (v-u)(v+u) - 4(v-u)
= (v - u)(u + v - 4).
\]
Les deux facteurs sont positifs~: $v - u > 0$, et $u + v > 2 + 2 = 4$
car $u \geq 2$ et $v > 2$. Donc $f(v) > f(u)$~: $f$ est strictement
\omterm{def:g11:func:monotone}{croissante} sur $\intco{2}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:6}
$f = \sqrt{u}$ avec $u(x) = 5 - x$ \omterm{def:g10:functions:variations}{décroissante} et positive ou nulle sur
$\intoc{-\infty}{5}$~: $f$ est \omterm{def:g10:functions:variations}{décroissante} là
(\cref{prop:g11:func:transforms}).

$g = \frac1u$ avec $u(x) = x^2 + 1$ \omterm{def:g11:func:monotone}{croissante} et positive sur
$\intco{0}{+\infty}$~: $g$ est \omterm{def:g10:functions:variations}{décroissante} là.

$h = 3 - 2\sqrt x$~: $\sqrt x$ est \omterm{def:g11:func:monotone}{croissante}, multiplier par $-2 < 0$
inverse, ajouter $3$ ne change rien~: $h$ est \omterm{def:g10:functions:variations}{décroissante} sur
$\intco{0}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:7}
\emph{1.} $f$ est sous \omterm{thm:g11:quad:canonical}{forme canonique}~: \omterm{def:g10:functions:variations}{décroissante} sur
$\intoc{-\infty}{2}$, \omterm{def:g11:func:monotone}{croissante} sur $\intco{2}{+\infty}$, \omterm{def:g10:functions:extrema}{minimum}
$f(2) = -3$.

\emph{2.} $g = \frac{1}{u}$ avec $u(x) = (x-2)^2 + 1 = f(x) + 4$. Le
décalage de $4$ conserve les variations de $f$, et $u \geq 1 > 0$.
Prendre l'inverse les inverse~: $g$ est \omterm{def:g11:func:monotone}{croissante} sur
$\intoc{-\infty}{2}$ et \omterm{def:g10:functions:variations}{décroissante} sur $\intco{2}{+\infty}$, avec
\omterm{def:g10:functions:extrema}{maximum} $g(2) = \frac{1}{1} = 1$.
\end{solution}

\begin{solution}{exo:g11:func:8}
Toute courbe en forme de colline passant par les trois points convient.
Puis~: $f + 2$ passe par $(0,3)$, $(2,5)$, $(4,3)$, mêmes variations
(décalage vertical)~; $-f$ passe par $(0,-1)$, $(2,-3)$, $(4,-1)$ avec
des variations \emph{inversées} (le pic devient une vallée)~; $2f$
passe par $(0,2)$, $(2,6)$, $(4,2)$, mêmes variations mais amplifiées
verticalement.
\end{solution}

\begin{solution}{exo:g11:func:9}
$2 - \frac{1}{x+1} = \frac{2(x+1) - 1}{x+1} = \frac{2x+1}{x+1} = f(x)$.
Sur $\intoo{-1}{+\infty}$, $u(x) = x + 1$ est \omterm{def:g11:func:monotone}{croissante} et positive,
donc $\frac1u$ est \omterm{def:g10:functions:variations}{décroissante}, $-\frac1u$ est \omterm{def:g11:func:monotone}{croissante}, et ajouter
$2$ ne change rien~: $f$ est strictement \omterm{def:g11:func:monotone}{croissante} sur
$\intoo{-1}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:10}
\emph{1.~Vrai.} Si $u < v$, alors
$(f+g)(v) - (f+g)(u) = \bigl(f(v)-f(u)\bigr) + \bigl(g(v)-g(u)\bigr)$
est une somme de deux nombres positifs ou nuls.

\emph{2.~Faux.} $f(x) = g(x) = x$ sont \omterm{def:g11:func:monotone}{croissantes} sur $\R$, mais leur
produit $x^2$ n'est pas croissant sur $\R$ (il décroît sur
$\intoc{-\infty}{0}$). (L'assertion devient vraie si les deux \omterm{def:g11:func:function}{fonctions}
sont aussi positives ou nulles.)

\emph{3.~Faux.} Même contre-exemple~: $f(x) = x$ est \omterm{def:g11:func:monotone}{croissante} sur
$\R$ mais $f^2(x) = x^2$ ne l'est pas.
\end{solution}

\begin{solution}{exo:g11:func:11}
\emph{1.} Pour $0 < u < v$~:
\[
f(v) - f(u) = (v - u) + \frac1v - \frac1u
= (v - u) + \frac{u - v}{uv}
= (v - u)\left(1 - \frac{1}{uv}\right)
= (v - u)\,\frac{uv - 1}{uv}.
\]

\emph{2.} Toujours $v - u > 0$ et $uv > 0$. Si $0 < u < v \leq 1$,
alors $uv < 1$, donc $uv - 1 < 0$ et $f(v) < f(u)$~: $f$ est strictement
\omterm{def:g10:functions:variations}{décroissante} sur $\intoc{0}{1}$. Si $1 \leq u < v$, alors $uv > 1$ et
$f(v) > f(u)$~: $f$ est strictement \omterm{def:g11:func:monotone}{croissante} sur
$\intco{1}{+\infty}$.

\emph{3.} $f$ décroît avant $1$ et croît après, donc
$f(x) \geq f(1) = 2$ pour tout $x > 0$, avec égalité exactement en
$x = 1$. (Cette inégalité réapparaît sous de nombreuses formes~;
comparer l'\cref{ex:g11:quad:canonical} et les inégalités de tangente
du \cref{ch:g12:deriv}.)
\end{solution}
