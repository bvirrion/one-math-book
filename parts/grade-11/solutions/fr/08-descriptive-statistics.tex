\section*{Chapitre \ref{ch:g11:stat} --- Statistique descriptive}

\begin{solution}{exo:g11:stat:1}
$n = 8$, somme $71$~: moyenne $\bar x = \frac{71}{8} = 8.875$. Médiane~:
milieu des $4$-ième et $5$-ième valeurs, $\frac{8 + 9}{2} = 8.5$.
Quartiles~: un quart de $8$ est $2$, donc $Q_1$ est la $2$-ième
valeur, $7$~; trois quarts est $6$, donc $Q_3$ est la $6$-ième valeur,
$10$. Écart interquartile~: $10 - 7 = 3$.
\end{solution}

\begin{solution}{exo:g11:stat:2}
Somme $40$, donc $\bar x = 5$. Somme des carrés~:
$4 + 16 + 16 + 16 + 25 + 25 + 49 + 81 = 232$. Par la formule de
König--Huygens,
\[
V = \frac{232}{8} - 5^2 = 29 - 25 = 4,
\qquad
\sigma = 2 .
\]
\end{solution}

\begin{solution}{exo:g11:stat:3}
Moyenne~:
\[
\bar x = \frac{1 \times 6 + 2 \times 9 + 3 \times 8 + 4 \times 10
+ 5 \times 9 + 6 \times 8}{50}
= \frac{181}{50} = 3.62 .
\]
Médiane~: les effectifs cumulés sont $6, 15, 23, 33, \dots$, donc les
$25$-ième et $26$-ième issues triées sont toutes deux $4$~: la médiane
est $4$.
\end{solution}

\begin{solution}{exo:g11:stat:4}
Le total des $15$ notes est $15 \times 11.2 = 168$~; avec le nouvel
arrivant, $\frac{168 + 18}{16} = \frac{186}{16} = 11.625$.
\end{solution}

\begin{solution}{exo:g11:stat:5}
$\frac{6 + 8 + x + 12 + 14}{5} = 10$ donne $40 + x = 50$, donc
$x = 10$. Les écarts de $6, 8, 10, 12, 14$ à la moyenne $10$ sont
$-4, -2, 0, 2, 4$, de carrés sommant à $40$~:
$V = \frac{40}{5} = 8$ et $\sigma = 2\sqrt2 \approx 2.83$.
\end{solution}

\begin{solution}{exo:g11:stat:6}
Par la \cref{prop:g11:stat:affine} avec $a = \frac12$, $b = 4$~: nouvelle
moyenne $\frac{12.4}{2} + 4 = 10.2$~; nouvel écart type
$\frac{2.5}{2} = 1.25$.
\end{solution}

\begin{solution}{exo:g11:stat:7}
La moyenne regroupée est la moyenne \emph{pondérée}
\[
\frac{20 \times 10 + 30 \times 15}{50} = \frac{650}{50} = 13,
\]
pas $12.5$~: le groupe B contribue plus de valeurs, donc tire la moyenne
regroupée vers la sienne.
\end{solution}

\begin{solution}{exo:g11:stat:8}
Les deux chaînes sont presque dans la cible en moyenne ($1.002$ et
$1.001$~kg), mais l'écart type de la chaîne~2 est trois fois plus grand~:
ses sacs varient beaucoup plus, produisant bien plus de sacs
manifestement sous- et sur-remplis. La chaîne~2 doit être traitée en
premier --- c'est la régularité, pas seulement la moyenne, qu'il faut
corriger.
\end{solution}

\begin{solution}{exo:g11:stat:9}
$\bar x = \frac{70}{10} = 7$~;
$V = \frac{540}{10} - 7^2 = 54 - 49 = 5$~;
$\sigma = \sqrt5 \approx 2.24$.
\end{solution}

\begin{solution}{exo:g11:stat:10}
\emph{1.} Avec $23$~: somme $50$, moyenne $5$~; médiane $=$ milieu des
$5$-ième et $6$-ième valeurs $= 3$. Sans $23$~: somme $27$, moyenne
$3$~; médiane ($5$-ième sur $9$ valeurs) $= 3$.

\emph{2.} Avec $23$ ($n = 10$)~: $Q_1$ est la $3$-ième valeur ($2$),
$Q_3$ la $8$-ième ($4$)~: écart interquartile $2$. Sans ($n = 9$)~:
$Q_1$ est la $3$-ième valeur ($2$), $Q_3$ la $7$-ième ($4$)~: encore
$2$.

\emph{3.} L'unique valeur aberrante tire la moyenne de $3$ à $5$ mais
laisse la médiane et l'écart interquartile inchangés~: ces deux-là sont
\emph{robustes}. Lorsqu'un jeu de données peut contenir des valeurs
aberrantes, rapporter la médiane et l'écart interquartile plutôt que la
moyenne et l'écart type.
\end{solution}
