\section*{Chapitre \ref{ch:g11:vect} --- Vecteurs et droites dans le plan}

\begin{solution}{exo:g11:vect:1}
$\vect{AB}\,(3, 4)$, $\vect{AC}\,(-3, 2)$,
$\vect{AB} + \vect{AC}\,(0, 6)$,
$2\vect{AB} - 3\vect{AC}\,(6 + 9,\ 8 - 6) = (15, 2)$. Le milieu de
$[BC]$ est $\left(\frac{5 + (-1)}{2}, \frac{3+1}{2}\right) = (2, 2)$.
\end{solution}

\begin{solution}{exo:g11:vect:2}
$\det = 4 \times 3 - (-2) \times (-6) = 12 - 12 = 0$~: colinéaires (en
effet $\vec v = -\frac12 \vec u$).

$\det = 2 \times 7 - 3 \times 5 = -1 \neq 0$~: non colinéaires.

$\det = 0 \times (-8) - 0 \times 3 = 0$~: colinéaires (tous deux
verticaux).
\end{solution}

\begin{solution}{exo:g11:vect:3}
Par $A(2,1)$, direction $(3, -1)$~: identifier $(-b, a) = (3, -1)$ donne
$a = -1$, $b = -3$, donc $-x - 3y + c = 0$~; en injectant $A$~:
$-2 - 3 + c = 0$, $c = 5$. En multipliant par $-1$~: $x + 3y - 5 = 0$.

Par $B(0,4)$ et $C(2,0)$~: $\vect{BC}\,(2, -4)$, donc $a = -4$,
$b = -2$~: $-4x - 2y + c = 0$~; en injectant $B$~: $-8 + c = 0$,
$c = 8$. En simplifiant par $-2$~: $2x + y - 4 = 0$. Vérification avec
$C$~: $4 + 0 - 4 = 0$.
\end{solution}

\begin{solution}{exo:g11:vect:4}
Un vecteur directeur est $(-b, a) = (2, 3)$. Intersections~: $x = 0$
donne $y = 3$, point $(0, 3)$~; $y = 0$ donne $x = -2$, point $(-2, 0)$.
Pour $P(2,6)$~: $3 \times 2 - 2 \times 6 + 6 = 0$, donc $P \in d$. Pour
$Q(1,4)$~: $3 - 8 + 6 = 1 \neq 0$, donc $Q \notin d$.
\end{solution}

\begin{solution}{exo:g11:vect:5}
$\vect{AB}\,(3, 6)$ et $\vect{AC}\,(-3, -6)$~:
$\det = 3 \times (-6) - (-3) \times 6 = 0$, donc $A$, $B$, $C$ sont
alignés.

$\vect{DE}\,(2, 3)$ et $\vect{DF}\,(5, 8)$~:
$\det = 2 \times 8 - 5 \times 3 = 1 \neq 0$~: $D$, $E$, $F$ ne sont pas
alignés.
\end{solution}

\begin{solution}{exo:g11:vect:6}
Premier couple~: en additionnant les deux équations,
$(2x + y - 7) + (x - y + 1) = 3x - 6 = 0$, donc $x = 2$, puis
$y = x + 1 = 3$~: elles se coupent en $(2, 3)$ (vérification~:
$4 + 3 - 7 = 0$).

Second couple~: $ab' - a'b = 1 \times 4 - (-2) \times (-2) = 0$, donc les
droites sont parallèles~; multiplier $e$ par $-2$ donne
$-2x + 4y - 6 = 0$, qui diffère de $e'$ ($-6 \neq 1$)~: strictement
parallèles, pas d'intersection.
\end{solution}

\begin{solution}{exo:g11:vect:7}
Pour $d: 2x - y + 3 = 0$~: direction $(-b, a) = (1, 2)$ et le point
$(0, 3)$ est sur $d$, donc $d = \{(t,\ 3 + 2t) : t \in \R\}$.

Pour la droite paramétrique par $(1, -3)$ de direction $(2, 1)$~:
$a = 1$, $b = -2$, donc $x - 2y + c = 0$~; en injectant $(1, -3)$~:
$1 + 6 + c = 0$, $c = -7$. Équation~: $x - 2y - 7 = 0$ (vérification
$t = 1$, point $(3, -2)$~: $3 + 4 - 7 = 0$).
\end{solution}

\begin{solution}{exo:g11:vect:8}
Une droite parallèle a le même $(a, b)$~: $4x - 3y + c = 0$~; par
$P(1, -2)$~: $4 + 6 + c = 0$, donc $c = -10$ et la droite est
$4x - 3y - 10 = 0$.

Le parallélisme de $mx + 2y - 5 = 0$ avec $d$ exige
$ab' - a'b = 4 \times 2 - m \times (-3) = 8 + 3m = 0$, donc
$m = -\frac83$.
\end{solution}

\begin{solution}{exo:g11:vect:9}
Le milieu de $[AB]$ est $M\!\left(\frac{-1+3}{2}, \frac{0+2}{2}\right)
= (1, 1)$. La médiane est la droite $(CM)$ avec $C(1, -4)$~: les deux
points ont pour abscisse $1$, donc la médiane est la droite verticale
\[
x - 1 = 0 .
\]
\end{solution}

\begin{solution}{exo:g11:vect:10}
$\det(\vec u, \vec v) = m(m - 1) - 3 \times 2 = m^2 - m - 6$. Il
s'annule lorsque $m^2 - m - 6 = 0$~: $\Delta = 1 + 24 = 25$, racines
$m = \frac{1 \pm 5}{2}$, c'est-à-dire $m = 3$ ou $m = -2$.
\end{solution}

\begin{solution}{exo:g11:vect:11}
\emph{1.} $I = \left(\frac12, 0\right)$. Alors
$\vect{DI} = \left(\frac12 - 0,\ 0 - 1\right) = \left(\frac12, -1\right)$
et $\vect{DJ} = \frac23\vect{DI} = \left(\frac13, -\frac23\right)$,
donc
\[
J = D + \vect{DJ} = \left(\frac13,\ \frac13\right).
\]

\emph{2.} $\vect{AJ}\left(\frac13, \frac13\right)$ et
$\vect{AC}\,(1, 1)$~:
$\det = \frac13 \times 1 - 1 \times \frac13 = 0$, donc $A$, $J$, $C$
sont alignés --- en fait $\vect{AJ} = \frac13\vect{AC}$~: la droite
$(DI)$ coupe la diagonale $(AC)$ à exactement un tiers de sa longueur
depuis $A$.
\end{solution}
