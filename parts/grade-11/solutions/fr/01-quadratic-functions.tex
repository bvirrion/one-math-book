\section*{Chapitre \ref{ch:g11:quad} --- Fonctions et équations du second degré}

\begin{solution}{exo:g11:quad:1}
$x^2-5x+6$~: $\Delta = 25 - 24 = 1$, \omterm{def:g11:quad:discriminant}{racines} $\frac{5 \pm 1}{2}$,
c'est-à-dire $2$ et $3$.

$2x^2+3x-2$~: $\Delta = 9 + 16 = 25$, \omterm{def:g11:quad:discriminant}{racines} $\frac{-3 \pm 5}{4}$,
c'est-à-dire $\frac12$ et $-2$.

$x^2+2x+5$~: $\Delta = 4 - 20 = -16 < 0$, pas de solution réelle.

$9x^2-6x+1$~: $\Delta = 36 - 36 = 0$, une (double) \omterm{def:g11:quad:discriminant}{racine}
$x_0 = \frac{6}{18} = \frac13$.
\end{solution}

\begin{solution}{exo:g11:quad:2}
$f(x) = x^2 - 6x + 11 = (x-3)^2 - 9 + 11 = (x-3)^2 + 2$~: sommet
$(3, 2)$.

$g(x) = -2x^2+4x+1 = -2(x^2 - 2x) + 1 = -2\bigl((x-1)^2 - 1\bigr) + 1
= -2(x-1)^2 + 3$~: sommet $(1, 3)$.

$h(x) = 3x^2 + 6x = 3(x^2 + 2x) = 3\bigl((x+1)^2 - 1\bigr)
= 3(x+1)^2 - 3$~: sommet $(-1, -3)$.
\end{solution}

\begin{solution}{exo:g11:quad:3}
$x^2 - 3x - 10$~: $\Delta = 9 + 40 = 49$, \omterm{def:g11:quad:discriminant}{racines}
$\frac{3 \pm 7}{2} = -2$ et $5$~; coefficient dominant positif, donc le
trinôme est négatif strictement entre les \omterm{def:g11:quad:discriminant}{racines}~: l'ensemble des
solutions est $\intoo{-2}{5}$.

$2x^2 + x + 3$~: $\Delta = 1 - 24 = -23 < 0$ et $a = 2 > 0$~: toujours
positif, l'ensemble des solutions est $\R$.

$-x^2 + 6x - 9 = -(x-3)^2$~: il est $\geq 0$ seulement là où
$(x-3)^2 \leq 0$, c'est-à-dire en $x = 3$. L'ensemble des solutions est
$\{3\}$.
\end{solution}

\begin{solution}{exo:g11:quad:4}
$x^2 - 7x + 6$~: $1 - 7 + 6 = 0$, donc $1$ est \omterm{def:g11:quad:discriminant}{racine}~; le produit des
\omterm{def:g11:quad:discriminant}{racines} est $6$, donc l'autre \omterm{def:g11:quad:discriminant}{racine} est $6$.

$2x^2 + 5x - 7$~: $2 + 5 - 7 = 0$, donc $1$ est \omterm{def:g11:quad:discriminant}{racine}~; le produit est
$\frac{c}{a} = -\frac72$, donc l'autre \omterm{def:g11:quad:discriminant}{racine} est $-\frac72$.

$x^2 + 4x - 5$~: $1 + 4 - 5 = 0$, donc $1$ est \omterm{def:g11:quad:discriminant}{racine}~; le produit est
$-5$, donc l'autre \omterm{def:g11:quad:discriminant}{racine} est $-5$.
\end{solution}

\begin{solution}{exo:g11:quad:5}
Les deux nombres résolvent $x^2 - 14x + 45 = 0$
(\cref{prop:g11:quad:vieta})~; $\Delta = 196 - 180 = 16$, \omterm{def:g11:quad:discriminant}{racines}
$\frac{14 \pm 4}{2} = 9$ et $5$.

Pour le rectangle~: la moitié du périmètre donne longueur $+$ largeur
$= 17$, et l'aire donne leur produit $60$. Ils résolvent
$x^2 - 17x + 60 = 0$~; $\Delta = 289 - 240 = 49$, \omterm{def:g11:quad:discriminant}{racines}
$\frac{17 \pm 7}{2} = 12$ et $5$. Le rectangle est $12 \times 5$.
\end{solution}

\begin{solution}{exo:g11:quad:6}
$\Delta = (m-2)^2 - 4$. Exactement une solution lorsque $\Delta = 0$~:
$(m-2)^2 = 4$, donc $m - 2 = \pm 2$, c'est-à-dire $m = 0$ ou $m = 4$.
Aucune solution lorsque $\Delta < 0$~: $(m-2)^2 < 4$, c'est-à-dire
$-2 < m - 2 < 2$, soit $0 < m < 4$.
\end{solution}

\begin{solution}{exo:g11:quad:7}
Le sommet est en $t = -\frac{20}{2 \times (-5)} = 2$, et
$h(2) = -20 + 40 + 1 = 21$~: hauteur maximale $21$~m, atteinte après
$2$ secondes.

$h(t) \geq 16$ signifie $-5t^2 + 20t - 15 \geq 0$, c'est-à-dire (en
divisant par $-5$, ce qui inverse l'inégalité) $t^2 - 4t + 3 \leq 0$.
Les \omterm{def:g11:quad:discriminant}{racines} de $t^2-4t+3$ sont $1$ et $3$ ($1 - 4 + 3 = 0$, produit
$3$), donc $t^2 - 4t + 3 \leq 0$ exactement sur $\intcc{1}{3}$~: la
balle est à au moins $16$~m de hauteur entre les instants $t = 1$~s et
$t = 3$~s.
\end{solution}

\begin{solution}{exo:g11:quad:8}
Avec $X = x^2$~: $X^2 - 13X + 36 = 0$, $\Delta = 169 - 144 = 25$, donc
$X = \frac{13 \pm 5}{2} = 9$ ou $4$. Retour à $x$~: $x^2 = 9$ donne
$x = \pm 3$, et $x^2 = 4$ donne $x = \pm 2$. L'ensemble des solutions est
$\{-3, -2, 2, 3\}$.
\end{solution}

\begin{solution}{exo:g11:quad:9}
La condition est $x^2 < x + 2$, c'est-à-dire $x^2 - x - 2 < 0$. Comme
$x^2 - x - 2 = (x+1)(x-2)$ (\omterm{def:g11:quad:discriminant}{racines} $-1$ et $2$), l'inégalité est
réalisée strictement entre les \omterm{def:g11:quad:discriminant}{racines}~: $x \in \intoo{-1}{2}$. Sur un
croquis, la \omterm{def:g11:quad:parabola}{parabole} $y = x^2$ passe sous la droite $y = x+2$ exactement
entre leurs deux points d'\omterm{def:g10:numbers:interunion}{intersection} $(-1, 1)$ et $(2, 4)$.
\end{solution}

\begin{solution}{exo:g11:quad:10}
Soit $x > 0$ tel que $x + \frac1x = \frac{13}{6}$. Multiplier par
$6x > 0$~: $6x^2 + 6 = 13x$, c'est-à-dire $6x^2 - 13x + 6 = 0$. Alors
$\Delta = 169 - 144 = 25$ et $x = \frac{13 \pm 5}{12}$, d'où
$x = \frac32$ ou $x = \frac23$. Les deux sont positifs et inverses l'un
de l'autre, comme attendu~: le nombre est $\frac32$ (ou $\frac23$).
\end{solution}

\begin{solution}{exo:g11:quad:11}
\emph{1.} Les \omterm{def:g10:numbers:interunion}{intersections} correspondent aux solutions de
$x^2 = mx - 1$, c'est-à-dire $x^2 - mx + 1 = 0$, dont le \omterm{def:g11:quad:discriminant}{discriminant} est
$\Delta = m^2 - 4$. Deux points lorsque $\abs{m} > 2$~; exactement un
lorsque $m = \pm 2$~; aucun lorsque $-2 < m < 2$.

\emph{2.} Pour $m = 2$~: $x^2 - 2x + 1 = (x-1)^2 = 0$, donc l'unique
point de contact est $(1, 1)$. Pour $m = -2$~: $(x+1)^2 = 0$, point de
contact $(-1, 1)$. Dans les deux cas la droite rencontre la \omterm{def:g11:quad:parabola}{parabole} en
un seul point sans la traverser --- c'est la tangente en ce point, et en
effet sa pente $m = \pm 2$ égale la valeur en $x = \pm 1$ de la dérivée
$2x$ de $x^2$ (\cref{ch:g11:deriv}).
\end{solution}
