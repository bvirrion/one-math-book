\section*{Chapitre \ref{ch:g11:seq} --- Suites~: un premier cours}

\begin{solution}{exo:g11:seq:1}
$u_n = \frac{n}{n+1}$~: $u_1 = \frac12$, $u_2 = \frac23$,
$u_3 = \frac34$.

$u_0 = 5$, $u_{n+1} = 3u_n - 2$~: $u_1 = 13$, $u_2 = 37$, $u_3 = 109$.

$u_n = (-1)^n n$~: $u_1 = -1$, $u_2 = 2$, $u_3 = -3$.
\end{solution}

\begin{solution}{exo:g11:seq:2}
$u_{10} = 7 + 10 \times (-3) = -23$ et $u_{25} = 7 - 75 = -68$.

Pour $(v_n)$~: $v_8 = v_3 + 5d$ donne $26 = 11 + 5d$, donc $d = 3$~;
puis $v_0 = v_3 - 3d = 11 - 9 = 2$.
\end{solution}

\begin{solution}{exo:g11:seq:3}
$u_8 = 5 \times 2^8 = 1280$.

Pour $(v_n)$~: $v_4 = v_2\, q^2$ donne $48 = 12 q^2$, donc $q^2 = 4$ et
$q = 2$ (les termes sont positifs). Puis
$v_0 = \frac{v_2}{q^2} = \frac{12}{4} = 3$.
\end{solution}

\begin{solution}{exo:g11:seq:4}
$1 + \dots + 500 = \frac{500 \times 501}{2} = 125\,250$.

$4 + 7 + \dots + 61$ est arithmétique de raison $3$ avec
$\frac{61 - 4}{3} + 1 = 20$ termes~: somme
$20 \times \frac{4 + 61}{2} = 650$.

$1 + \frac12 + \dots + \frac{1}{2^{10}}$ est géométrique de raison
$\frac12$ avec $11$ termes~:
$\frac{1 - (1/2)^{11}}{1 - 1/2} = 2\left(1 - \frac{1}{2048}\right)
= \frac{2047}{1024}$.
\end{solution}

\begin{solution}{exo:g11:seq:5}
$u_{n+1} - u_n = 4(n+1) - 1 - 4n + 1 = 4$~: arithmétique de raison $4$.

$\frac{v_{n+1}}{v_n} = \frac{2^{n+1}}{3^{n+2}} \cdot \frac{3^{n+1}}{2^n}
= \frac23$~: géométrique de raison $\frac23$.

$w_0 = 0$, $w_1 = 2$, $w_2 = 6$~: les différences $2$ et $4$ diffèrent,
donc non arithmétique~; $\frac{w_1}{w_0}$ n'est même pas défini, et les
rapports $\frac{w_2}{w_1} = 3 \neq \frac{w_3}{w_2} = 2$~: ni l'un ni
l'autre.
\end{solution}

\begin{solution}{exo:g11:seq:6}
Les tailles des rangées sont arithmétiques~: premier terme $16$, raison
$2$. La dernière ($20$-ième) rangée a $16 + 19 \times 2 = 54$ sièges. Le
total est $20 \times \frac{16 + 54}{2} = 700$ sièges.
\end{solution}

\begin{solution}{exo:g11:seq:7}
Après $n$ heures la population est $500 \times 2^n$. À 20~h, $n = 8$~:
$500 \times 256 = 128\,000$ bactéries. Il faut $500 \times 2^n >
10^6$, c'est-à-dire $2^n > 2000$~: comme $2^{10} = 1024$ et
$2^{11} = 2048$, la population dépasse d'abord un million après $11$
heures complètes, à 23~h.
\end{solution}

\begin{solution}{exo:g11:seq:8}
$c_2 = 1.002 \times 100 + 100 = 200.20$ et
$c_3 = 1.002 \times 200.20 + 100 \approx 300.60$. Les différences
$c_2 - c_1 = 100.20$ et $c_3 - c_2 \approx 100.40$ ne sont pas égales,
donc $(c_n)$ n'est pas arithmétique~; les rapports
$\frac{c_2}{c_1} = 2.002$ et $\frac{c_3}{c_2} \approx 1.50$ ne sont pas
égaux non plus, donc ce n'est pas géométrique. (Les récurrences mixtes
«~multiplier puis ajouter~» comme celle-ci se résolvent par l'astuce de
suite auxiliaire de l'\cref{exo:g11:seq:11}.)
\end{solution}

\begin{solution}{exo:g11:seq:9}
$u_{n+1} - u_n = (n+1)^2 - 8(n+1) - n^2 + 8n = 2n - 7$~: négatif pour
$n \leq 3$, positif pour $n \geq 4$. Donc $(u_n)$ décroît jusqu'à
$u_4 = 16 - 32 = -16$, puis croît~: elle n'est pas monotone.

$(v_n)$ a des termes positifs et
\[
\frac{v_{n+1}}{v_n} = \frac{3^{n+1}}{(n+1)!} \cdot \frac{n!}{3^n}
= \frac{3}{n+1},
\]
qui est $> 1$ pour $n \leq 1$, $= 1$ pour $n = 2$, et $< 1$ pour
$n \geq 3$~: la suite croît jusqu'à $v_2 = v_3 = \frac92$, puis
décroît.
\end{solution}

\begin{solution}{exo:g11:seq:10}
Les $n$ premiers termes sont $u_0, \dots, u_{n-1}$, avec $u_0 = 3$ et
$u_{n-1} = 3 + 4(n-1) = 4n - 1$. Leur somme est
\[
n \times \frac{3 + (4n-1)}{2} = n(2n + 1) = 903,
\]
donc $2n^2 + n - 903 = 0$. Ici $\Delta = 1 + 4 \times 2 \times 903 =
7225 = 85^2$, et $n = \frac{-1 + 85}{4} = 21$ (la racine négative est
rejetée). Vérification~: $21 \times 43 = 903$.
\end{solution}

\begin{solution}{exo:g11:seq:11}
\emph{1.} $u_1 = 5$, $u_2 = 9$, $u_3 = 17$~: chaque terme est un de plus
que $4, 8, 16$, ce qui suggère $u_n = 2^{n+1} + 1$.

\emph{2.} Avec $v_n = u_n - 1$~:
\[
v_{n+1} = u_{n+1} - 1 = 2u_n - 1 - 1 = 2(u_n - 1) = 2v_n,
\]
donc $(v_n)$ est géométrique de raison $2$ et de premier terme
$v_0 = u_0 - 1 = 2$.

\emph{3.} D'où $v_n = 2 \times 2^n = 2^{n+1}$ et
$u_n = v_n + 1 = 2^{n+1} + 1$, confirmant la conjecture. (Le nombre $1$
soustrait dans $v_n$ est le point fixe de $x \mapsto 2x - 1$~; la même
idée réapparaît pour $u_{n+1} = au_n + b$ dans le \cref{ch:g12:seq}.)
\end{solution}
