\section*{Chapter \ref{ch:g11:binom} --- The Binomial Distribution}

\begin{solution}{exo:g11:binom:1}
Rows $6$ and $7$:
\[
1,\ 6,\ 15,\ 20,\ 15,\ 6,\ 1
\qquad\text{and}\qquad
1,\ 7,\ 21,\ 35,\ 35,\ 21,\ 7,\ 1 .
\]
Hence $\binom62 = 15$, $\binom73 = 35$, $\binom74 = 35$ (the symmetry
$\binom73 = \binom74$ reflects swapping successes and failures).
\end{solution}

\begin{solution}{exo:g11:binom:2}
Fixed number of trials ($4$ rolls), two outcomes per roll (six or not,
$p = \frac16$), independent rolls: $X \sim \mathcal B(4, \frac16)$.
\[
\P(X=0) = \left(\frac56\right)^4 = \frac{625}{1296} \approx 0.48,
\qquad
\P(X=1) = 4 \times \frac16\left(\frac56\right)^3 = \frac{500}{1296}
\approx 0.39,
\]
\[
\P(X \geq 2) = 1 - \frac{625 + 500}{1296} = \frac{171}{1296}
\approx 0.13 .
\]
\end{solution}

\begin{solution}{exo:g11:binom:3}
\emph{1.} Binomial $\mathcal B(20, \frac12)$: fixed $n$, same $p$,
independent tosses.

\emph{2.} Not binomial: the cards are dealt \emph{without replacement},
so the \omterm{def:g10:proba:distribution}{probability} of an ace changes from card to card and the draws are
not independent.

\emph{3.} Binomial $\mathcal B(7, 0.3)$ by the stated independence
assumption.
\end{solution}

\begin{solution}{exo:g11:binom:4}
$\E(X) = 50 \times 0.2 = 10$;
$\V(X) = 50 \times 0.2 \times 0.8 = 8$;
$\sigma(X) = 2\sqrt2 \approx 2.83$.
\end{solution}

\begin{solution}{exo:g11:binom:5}
$X \sim \mathcal B(6, 0.7)$.
\[
\P(X = 4) = \binom64 (0.7)^4 (0.3)^2 = 15 \times 0.2401 \times 0.09
\approx 0.324 .
\]
\[
\P(X \geq 5) = \binom65 (0.7)^5(0.3) + (0.7)^6
= 6 \times 0.16807 \times 0.3 + 0.117649 \approx 0.420 .
\]
\end{solution}

\begin{solution}{exo:g11:binom:6}
$X \sim \mathcal B\left(8, \frac12\right)$; every path has \omterm{def:g10:proba:distribution}{probability}
$\frac{1}{256}$, so
\[
\P(X \geq 6) = \frac{\binom86 + \binom87 + \binom88}{256}
= \frac{28 + 8 + 1}{256} = \frac{37}{256} \approx 0.14 .
\]
Guessing passes about one time in seven.
\end{solution}

\begin{solution}{exo:g11:binom:7}
The \omterm{def:g10:proba:operations}{complement} of ``at least one of each'' is ``all five identical'':
all A or all B, each with \omterm{def:g10:proba:distribution}{probability} $\left(\frac12\right)^5 =
\frac1{32}$. Hence
\[
\P(\text{one of each kind}) = 1 - \frac{2}{32} = \frac{15}{16} .
\]
\end{solution}

\begin{solution}{exo:g11:binom:8}
$\P(X = 3) = p^3$ and $\P(X \geq 1) = 1 - (1-p)^3$. Solving
$p^3 = \frac{27}{64} = \left(\frac34\right)^3$ gives $p = \frac34$ (the
cube \omterm{def:g11:func:function}{function} is strictly \omterm{def:g11:func:monotone}{increasing}, \cref{ch:g11:func}, so the solution
is unique).
\end{solution}

\begin{solution}{exo:g11:binom:9}
$\P(\text{at least one head}) = 1 - \left(\frac12\right)^n$, so the
condition is $\left(\frac12\right)^n < 0.01$, i.e.\ $2^n > 100$. Since
$2^6 = 64$ and $2^7 = 128$: from $n = 7$ tosses on.
\end{solution}

\begin{solution}{exo:g11:binom:10}
The sum of row $n$ counts all paths of the $n$-trial \omterm{met:g10:proba:tree}{tree}, sorted by
their number of successes. But the \omterm{met:g10:proba:tree}{tree} doubles its paths at each trial
(each path splits into S and F), so it has $2^n$ paths in total. Hence
$\sum_{k} \binom nk = 2^n$. Alternatively, by induction: row $0$ sums to
$1 = 2^0$, and Pascal's rule makes each entry of row $n$ contribute to
exactly two entries of row $n+1$, so the row sums double.
\end{solution}

\begin{solution}{exo:g11:binom:11}
\emph{1.} Under the claim, $X \sim \mathcal B(10, 0.6)$. Summing the
first terms:
\begin{align*}
\P(X \leq 3)
&= (0.4)^{10} + 10(0.6)(0.4)^9 + 45(0.6)^2(0.4)^8
+ 120(0.6)^3(0.4)^7\\
&\approx 0.0001 + 0.0016 + 0.0106 + 0.0425 = 0.0548 .
\end{align*}

\emph{2.} A result at least as extreme as the one observed ($3$ or fewer
approvals) has \omterm{def:g10:proba:distribution}{probability} about $5.5\%$ --- just \emph{above} the $5\%$
threshold. Strictly applying the rule, the observation is (barely)
compatible with the claim and one does not reject it. The example shows
how sensitive borderline decisions are to the choice of threshold: with a
$6\%$ convention the conclusion would flip.
\end{solution}

\begin{solution}{pb:g11:binom:1}
\textbf{1.} Rows: $1$; $1\,1$; $1\,2\,1$; $1\,3\,3\,1$;
$1\,4\,6\,4\,1$; $1\,5\,10\,10\,5\,1$;
$1\,6\,15\,20\,15\,6\,1$. Symmetry: choosing which $k$ objects
to take is the same act as choosing which $n - k$ to leave
behind.

\textbf{2.} $1 + 4 + 6 + 4 + 1 = 16 = 2^4$;
$1 + 5 + 10 + 10 + 5 + 1 = 32 = 2^5$. Proof: the
$\binom nk$ count the subsets of each size of an $n$-element
set, and the subsets altogether number $2^n$ (each element in
or out, independently).

\textbf{3.} Committees of $k$ chosen among $n + 1$ people, one
of whom is Zoe: those without Zoe are $\binom nk$ (choose all
$k$ among the others); those with Zoe are $\binom{n}{k-1}$
(choose her $k - 1$ colleagues). Total:
$\binom nk + \binom{n}{k-1}$.

\textbf{4.} Triangle row $7$: $1\,7\,21\,35\,\dots$: $35$.
Formula: $\frac{7 \times 6 \times 5}{3 \times 2 \times 1} =
35$.

\textbf{5.} $1 + 3 + 6 + 10 = 20 = \binom63$. Cascade:
$\binom63 = \binom52 + \binom53 = \binom52 + \binom42 +
\binom43 = \binom52 + \binom42 + \binom32 + \binom33$ ---
each application of Pascal's rule peels off one term of the
staircase.

\textbf{6.} Fixed number $n$ of bounces; each bounce an
independent \omterm{def:g11:binom:bernoulli}{Bernoulli trial} with the same $p = \frac12$; the
slot number counts the successes (right bounces): all three
boxes of \cref{met:g11:binom:recognize} ticked:
$\mathcal B\!\left(n, \frac12\right)$.

\textbf{7.} Probabilities $\frac{1}{16}, \frac{4}{16},
\frac{6}{16}, \frac{4}{16}, \frac{1}{16}$ for slots
$0, \dots, 4$: the center slot $2$ is the most crowded.

\textbf{8.} Expected counts $= 1024 \times
\binom{10}{k}/1024 = \binom{10}{k}$: middle slot
$\binom{10}{5} = 252$ balls; slot $7$: $\binom{10}{7} = 120$;
each edge slot: $1$ ball. A tall center falling away
symmetrically to whisper-thin edges: the bell.

\textbf{9.} $\E = np = 5$; $V = np(1 - p) = 2.5$;
$\sigma \approx 1.58$. Within $2\sigma$: slots $2$ to $8$
carry
\[
\frac{45 + 120 + 210 + 252 + 210 + 120 + 45}{1024}
= \frac{1002}{1024} \approx 98\,\%
\]
of the balls --- far better
than Chebyshev's all-purpose $75\,\%$
(\cref{pb:g11:stat:1}): bell shapes concentrate hard.

\textbf{10.} $\E = 6$, $V = 10 \times 0.6 \times 0.4 = 2.4$,
$\sigma \approx 1.55$: the pile keeps its bell shape but
slides its summit to slot $6$ --- a tilted board is a loaded
coin made visible.

\textbf{11.} The slot number is a \emph{sum} of many small,
independent, same-sized chance pushes --- and such sums always
organize themselves into the bell: most pushes cancel, extremes
require unanimity. Heights, measurement errors and countless
natural quantities are likewise sums of many small independent
effects, which is why the same silhouette appears everywhere;
the theorem certifying it is the central limit theorem.

\textbf{12.} A sweep: one team wins all $4$:
$2 \times \left(\frac12\right)^4 = \frac18$.

\textbf{13.} Seven games require $3$--$3$ after six:
$\binom63 \left(\frac12\right)^6 = \frac{20}{64} =
\frac{5}{16}$.

\textbf{14.} Ends in $5$: the winner takes game 5 and $3$ of
the first $4$:
$2 \times \binom43 \left(\frac12\right)^5 = \frac14$. Ends in
$6$: $2 \times \binom53 \left(\frac12\right)^6 = \frac{5}{16}$.
\omterm{def:g11:prob:rv}{Distribution} over $4, 5, 6, 7$:
$\frac18, \frac14, \frac{5}{16}, \frac{5}{16}$ (sum $1$).
Expected length: $4 \cdot \frac18 + 5 \cdot \frac14 + 6 \cdot
\frac{5}{16} + 7 \cdot \frac{5}{16} = 5.8125$ games. Six- and
seven-game series are the most likely --- drama is built into
the format.

\textbf{15.} Win in $4$: $0.6^4 = 0.1296$; in $5$:
$\binom43\,0.6^3 \times 0.4 \times 0.6 = 0.2074$; in $6$:
$\binom53\,0.6^3 \times 0.4^2 \times 0.6 = 0.2074$; in $7$:
$\binom63\,0.6^3 \times 0.4^3 \times 0.6 = 0.1659$. Total:
about $0.710$: a $60\,\%$-per-game team wins $71\,\%$ of
series --- the series amplifies the edge.

\textbf{16.} Single final: $60\,\%$. Best-of-3:
$p^2 + 2p^2 q = 0.36 + 0.288 = 0.648$. The ladder
$60\,\% \to 65\,\% \to 71\,\%$ continues with length: more
games average out luck (the law of large numbers in miniature),
so long finals crown skill --- which is precisely what leagues
sell.

\textbf{17.} Fair coin: $\E = 50$, $\sigma = \sqrt{25} = 5$;
$z = \frac{62 - 50}{5} = 2.4$: beyond the $2\sigma$
convention --- the coin deserves an investigation.

\textbf{18.} $\P(X = 0) = 0.98^{50} \approx 0.364$;
$\P(X = 1) = 50 \times 0.02 \times 0.98^{49} \approx 0.372$;
$\P(X = 2) = \binom{50}{2} 0.02^2 \times 0.98^{48} \approx
0.186$. So $\P(X \geq 3) \approx 1 - 0.922 = 0.078$: about
$7.8\,\%$ --- above the $5\,\%$ threshold, so not (yet)
enough to reject the claim; a second bad batch would change
the story.

\textbf{19.} $\P(\text{at least one win}) = 1 - 0.999^{1000}
\approx 1 - 0.368 = 0.632$: a thousand tickets at one in a
thousand give not certainty but $63\,\%$. The recurring
$0.368$ is $\frac1e$ in disguise --- the constant $e$ makes
its official entrance in grade 12.

\textbf{20.} Recognition: fixed $n$, independence, constant
$p$ --- then and only then, binomial. Table: \omterm{prop:g11:binom:pascal}{Pascal's
triangle}, row $n$. Shape: the bell, symmetric for
$p = \frac12$, shifted otherwise. Compass: center $np$,
spread $\sqrt{np(1-p)}$ --- the \omterm{pb:g11:stat:1}{z-scores} of decisions.
Heirs: the smooth bell curve that the piles approach, and the
law of large numbers that explains why big boards never lie.
\end{solution}
