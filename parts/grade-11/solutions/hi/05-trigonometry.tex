\section*{अध्याय \ref{ch:g11:trigo} --- त्रिकोणमिति: इकाई वृत्त}

\begin{solution}{exo:g11:trigo:1}
को रेडियंस: $30^\circ = \frac{\pi}{6}$, $45^\circ = \frac{\pi}{4}$,
$120^\circ = \frac{2\pi}{3}$, $270^\circ = \frac{3\pi}{2}$.

डिग्री के लिए: $\frac{\pi}{5} = 36^\circ $, $\frac{5\pi}{6} = 150^\circ $,
$\frac{7\pi}{4} = 315^\circ $, $\frac{2\pi}{9} = 40^\circ $.
\end{solution}

\begin{solution}{exo:g11:trigo:2}
$\frac{2\pi}{3}$ और $-\frac{\pi}{4}$ पहले से ही अंदर हैं
$\intoc{-\pi}{\pi}$: क्रमशः दूसरा चतुर्थांश और चौथा चतुर्थांश।

$\frac{17\pi}{6} - 2\pi = \frac{5\pi}{6}$: दूसरा चतुर्थांश, के करीब
नकारात्मक $ x $-अक्ष.

$-\frac{7\pi}{2} + 4\pi = \frac{\pi}{2}$: वृत्त का शीर्ष,
$(0, 1)$.
\end{solution}

\begin{solution}{exo:g11:trigo:3}
$\cos\frac{3\pi}{4} = \cos\left(\pi - \frac\pi4\right)
= -\cos\frac\pi4 = -\frac{\sqrt2}{2}$.

$\sin\frac{5\pi}{6} = \sin\left(\pi - \frac\pi6\right)
= \sin\frac\pi6 = \frac12$.

$\cos\left(-\frac\pi3\right) = \cos\frac\pi3 = \frac12$.

$\sin\frac{4\pi}{3} = \sin\left(\pi + \frac\pi3\right)
= -\sin\frac\pi3 = -\frac{\sqrt3}{2}$.

$\cos\frac{11\pi}{6} = \cos\left(-\frac\pi6 + 2\pi\right)
= \cos\frac\pi6 = \frac{\sqrt3}{2}$.
\end{solution}

\begin{solution}{exo:g11:trigo:4}
$\sin^2 t = 1 - \cos^2 t = 1 - \frac{25}{169} = \frac{144}{169}$, इसलिए
$\sin t = \pm\frac{12}{13}$. पर $\intoo{-\frac\pi2}{0}$ (चौथा चतुर्थांश)
द \omterm{def:g10:coordgeom:system}{तालमेल} नकारात्मक है: $\sin t = -\frac{12}{13}$.
\end{solution}

\begin{solution}{exo:g11:trigo:5}
का उपयोग करते हुए \cref{prop:g11:trigo:associated}:
\[
A = (-\cos x) + \cos x + \cos x - \cos x = 0 ,
\]
\[
B = \sin x + (-\sin x) + (-\sin x) = -\sin x .
\]
\end{solution}

\begin{solution}{exo:g11:trigo:6}
$\cos t = \frac{\sqrt3}{2}$: एक समाधान है $\frac\pi6$, इसलिए
$ t = \pm\frac\pi6$ (द्वारा बदलाव $2k\pi $ छोड़ दो \omterm{def:g10:numbers:interval}{अंतराल}).

$\sin t = \frac12$: $ t = \frac\pi6$ या $ t = \pi - \frac\pi6 =
\frac{5\pi}{6}$.

$\cos t = -1$: का एकमात्र बिंदु \omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज} $-1$ है $ M(\pi)$, इसलिए $ t = \pi $.
\end{solution}

\begin{solution}{exo:g11:trigo:7}
$\sin t = -\frac{\sqrt3}{2}$: से $ a = -\frac\pi3$, परिवार हैं
$-\frac\pi3 + 2k\pi $ और $\pi + \frac\pi3 + 2k\pi $; में
$\intco{0}{2\pi}$: $ t = \frac{4\pi}{3}$ और $ t = \frac{5\pi}{3}$.

$\cos t = 0$: $ t = \frac\pi2$ और $ t = \frac{3\pi}{2}$.

$2\sin t + 1 = 0$ मतलब $\sin t = -\frac12$: से $ a = -\frac\pi6$, में
$\intco{0}{2\pi}$: $ t = \frac{7\pi}{6}$ और $ t = \frac{11\pi}{6}$.
\end{solution}

\begin{solution}{exo:g11:trigo:8}
द्वारा \cref{thm:g11:trigo:equations}, $\cos 2t = \cos t $ मतलब
$2t = t + 2k\pi $ या $2t = -t + 2k\pi $, यानी\ $ t = 2k\pi $ या
$ t = \frac{2k\pi}{3}$ ($ k \in \Z $). प्रथम परिवार में समाहित है
दूसरा (ले लो $ k $ का एक गुणक $3$), तो समाधान सेट है
\[
S = \left\{\frac{2k\pi}{3} : k \in \Z\right\}
\]
--- वृत्त पर, तीन बिंदु $ M(0)$,
$ M\!\left(\frac{2\pi}{3}\right)$, $ M\!\left(\frac{4\pi}{3}\right)$.
\end{solution}

\begin{solution}{exo:g11:trigo:9}
\omterm{def:g10:algebra:expand}{विस्तार} दोनों वर्ग और प्रयोग $\cos^2 x + \sin^2 x = 1$:
\[
(\cos^2 x + 2\sin x\cos x + \sin^2 x)
+ (\cos^2 x - 2\sin x\cos x + \sin^2 x)
= 1 + 1 = 2 .
\]
\end{solution}

\begin{solution}{exo:g11:trigo:10}
बिना फिसले लुढ़कने का मतलब है कि चाप की लंबाई दूरी के बराबर है
यात्रा की: $ r\theta = 150$ सेमी देता है
$\theta = \frac{150}{30} = 5$ रेड $= 5 \times \frac{180}{\pi} \approx
286.5^\circ $. एक पूरा चक्कर बाइक को परिधि से आगे बढ़ाता है
$2\pi \times 30 = 60\pi \approx 188.5$ सेमी, के बारे में $1.88$ एम।
\end{solution}

\begin{solution}{exo:g11:trigo:11}
यूनिट सर्कल पर, के बिंदु \omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज} बिल्कुल $\frac12$ हैं
$ M\!\left(\pm\frac\pi3\right)$. के साथ अंक \omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज} \emph{अधिक से अधिक}
$\frac12$ ऊर्ध्वाधर रेखा के बाईं ओर चाप बनाएं $ x = \frac12$,
के रूप में यात्रा की $ t $ से चला जाता है $\frac\pi3$ तक $\pi $, या से $-\pi $ तक
$-\frac\pi3$. इसलिए, पर $\intoc{-\pi}{\pi}$:
\[
S = \intoc{-\pi}{-\frac{\pi}{3}} \cup \intcc{\frac{\pi}{3}}{\pi}.
\]
(अंतिमबिंदु $\pm\frac\pi3$ शामिल हैं क्योंकि असमानता व्यापक है।)
\end{solution}

\begin{solution}{pb:g11:trigo:1}
\textbf{1.} $30^\circ = \frac{\pi}{6}$; $135^\circ =
\frac{3\pi}{4}$; $300^\circ = \frac{5\pi}{3}$. और
$\frac{3\pi}{4} = 135^\circ $; $\frac{5\pi}{6} = 150^\circ $.

\textbf{2.} लंबाई $ r\,t = 3 \times 2 = 6$ एम। रेडियंस चाप बनाओ
लंबाई एक सादा उत्पाद---डिग्री खींचेगी
$\frac{\pi}{180}$ हर सूत्र में.

\textbf{3.} रेडियंस चालू $= \frac{\text{distance}}
{\text{radius}} = \frac{1000}{0.30} \approx 3\,333$ रेड;
द्वारा विभाजित $2\pi $: के बारे में $530$ पूर्ण मोड़.

\textbf{4.} $\cos\frac{2\pi}{3} = -\frac12$;
$\sin\left(-\frac\pi4\right) = -\frac{\sqrt2}{2}$;
$\frac{19\pi}{6} - 2\pi = \frac{7\pi}{6}$, इसलिए
$\cos\frac{19\pi}{6} = \cos\frac{7\pi}{6} =
-\frac{\sqrt3}{2}$.

\textbf{5.} $\cos^2 t = 1 - \frac{9}{25} = \frac{16}{25}$; में
दूसरा चतुर्थांश \omterm{def:g11:trigo:cossin}{कोज्या} नकारात्मक है:
$\cos t = -\frac45$, और
$\tan t = \frac{3/5}{-4/5} = -\frac34$.

\textbf{6.} में $ t $ मिनट पहिया घूमता है
$\frac{t}{30} \times 2\pi = \frac{\pi t}{15}$. पर $ t = 0$:
$ h = 65 - 60 = 5$ मी --- नीचे बोर्डिंग प्लेटफार्म
(हब की ऊंचाई शून्य से त्रि\omterm{def:g11:trigo:cossin}{ज्या})। ऋण चिह्न आपको डालता है \emph{नीचे}
जब कोण घुमाया जाता है तो हब होता है $0$.

\textbf{7.} $ h(7.5) = 65 - 60\cos\frac{\pi}{2} = 65$ मी (हब
ऊँचाई); $ h(15) = 65 + 60 = 125$ मी (शीर्ष);
$ h(20) = 65 - 60\cos\frac{4\pi}{3} = 65 + 30 = 95$ एम।

\textbf{8.} $65 - 60\cos\frac{\pi t}{15} = 95$ देता है
$\cos\frac{\pi t}{15} = -\frac12$: पहला समाधान
$\frac{\pi t}{15} = \frac{2\pi}{3}$, यानी\ $ t = 10$ मिनट।

\textbf{9.} $\cos\frac{\pi t}{15} \leq -\frac12$ के लिए
$\frac{\pi t}{15} \in \intcc{\frac{2\pi}{3}}{\frac{4\pi}{3}}$,
यानी\ $ t \in \intcc{10}{20}$: ऊपर की यात्रा के दस मिनट
$95$ मी, शिखर के बारे में सममित $ t = 15$.

\textbf{10.} पहला मिनट:
\[
h(1) - h(0) = 60\left(1 - \cos\frac{\pi}{15}\right)
\approx 1.3 \text{ m}.
\]
मिनटों के बीच $7$ और $8$:
\[
60\left(\cos\frac{7\pi}{15} - \cos\frac{8\pi}{15}\right)
\approx 12.5 \text{ m}
\]
--- लगभग दस गुना अधिक. हब स्तर पार करते समय ऊंचाई सबसे तेजी से बदलती है
और नीचे और ऊपर के पास रुक जाता है: साइनसॉइड अपनी ओर से तीव्र होता है
\omterm{prop:g10:coordgeom:midpoint}{मध्य}, अपने चरम पर सपाट।

\textbf{11.} RADIUS $= \frac{40 - 4}{2} = 18$ मी, हब एट
$4 + 18 = 22$ मी, अवधि $12$ मिनट:
$ h(t) = 22 - 18\cos\left(\frac{\pi t}{6}\right)$.

\textbf{12.} \omterm{def:g10:functions:extrema}{अधिकतम} $7 + 3 = 10$ मी कब
$\sin\frac{\pi t}{6} = 1$: पहले पर $ t = 3$ (सुबह 3 बजे);
\omterm{def:g10:functions:extrema}{न्यूनतम} $4$ मैं पहले पर $ t = 9$; अवधि
$\frac{2\pi}{\pi/6} = 12$ घंटे।

\textbf{13.} $\sin\frac{\pi t}{6} \geq \frac12$ के लिए
$\frac{\pi t}{6} \in \intcc{\frac\pi6}{\frac{5\pi}{6}}$:
$ t \in \intcc{1}{5}$ --- दोपहर 1 बजे से शाम 5 बजे तक चार घंटे की विंडो
पूर्वाह्न, एक अवधि बाद फिर से खुल रहा है, $ t \in \intcc{13}{17}$
(दोपहर 1 बजे से शाम 5 बजे तक)।

\textbf{14.} ऊर्ध्वाधर अक्ष पर दर्पण बिंदु को भेजता है
कोण $ t $ कोण पर बिंदु तक $\pi - t $, का संरक्षण
\omterm{def:g10:coordgeom:system}{तालमेल}: $\sin(\pi - t) = \sin t $. के बारे में आधा मोड़
\omterm{pb:g11:quad:1}{केंद्र} भेजता है $ t $ को $\pi + t $, दोनों को नकारते हुए \omterm{def:g10:coordgeom:system}{COORDINATES}:
$\cos(\pi + t) = -\cos t $. और विकर्ण के पार दर्पण
अदला-बदली \omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज} और \omterm{def:g10:coordgeom:system}{तालमेल}:
$\cos\left(\frac\pi2 - t\right) = \sin t $ --- द
\omterm{def:g10:proba:operations}{पूरक} कोणों का ``सह''।

\textbf{15.} $\sin(-t) = -\sin t $: \omterm{def:g11:func:parity}{विषम} (अर्ध-मोड़ समरूपता
इसका \omterm{def:g10:functions:graph}{ग्राफ}); $\cos(-t) = \cos t $: सम (दर्पण). और
$\sin t + t^3$ दो का योग है अजीब कार्य: \omterm{def:g11:func:parity}{विषम}।

\textbf{16.} कारक: $(2\sin t + 1)(\sin t - 1) = 0$:
$\sin t = 1$ देता है $ t = \frac\pi2$;
$\sin t = -\frac12$ देता है $ t = \frac{7\pi}{6}$ और
$ t = \frac{11\pi}{6}$. वृत्त पर तीन समाधान.

\textbf{17.} $1$ रेड $= \frac{180}{\pi} \approx 57.3^\circ $.
इसलिए $\sin(1) \approx 0.841$ जबकि $\sin(1^\circ) \approx
0.0175$: लगभग का एक कारक $50$. नैतिक: मोड लाइट की जाँच करें
किसी भी त्रिकोणमितीय प्रदर्शन पर भरोसा करने से पहले।

\textbf{18.} \omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज} के बराबर होती है \omterm{def:g10:coordgeom:system}{तालमेल} विकर्ण पर:
$ t = \frac\pi4$ और $ t = \frac{5\pi}{4}$.

\textbf{19.} $\sin(0.1) = 0.09983\ldots $ ख़िलाफ़ $0.1$:
के बारे में सापेक्ष त्रुटि $0.17\,\%$. छोटे कोणों के लिए जीवा
चाप को गले लगाता है --- और ग्रेड-12 की सीमा
$\frac{\sin t}{t} \to 1$ वास्तव में यही अवलोकन किया गया है
एक प्रमेय.

\textbf{20.} रेडियंस: कोण $=$ यूनिट सर्कल पर चाप, इसलिए
कोणों की गणना लंबाई की तरह की जाती है। \omterm{def:g11:trigo:cossin}{कोज्या} और \omterm{def:g11:trigo:cossin}{ज्या}: द \omterm{def:g10:coordgeom:system}{COORDINATES}
घड़ी की सुई की --- छाया ज़मीन पर, ऊँचाई पर
दीवार. वृत्त की समरूपताएँ: की संपूर्ण सूची
पहचान, दर्पण और आधे-मोड़ को पढ़ें। \omterm{def:g10:algebra:equation}{समीकरण}
$\cos t = c $: घड़ी की समय सारिणी, वृत्त दिखाने के साथ
हर समाधान एक ही बार में. वेशभूषा: पहिया पहना
$65 - 60\cos $, ज्वार घिस गया $7 + 3\sin$ --- वही घड़ी, वही
गणित.
\end{solution}
