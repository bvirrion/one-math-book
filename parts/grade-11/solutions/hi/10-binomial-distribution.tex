\section*{अध्याय \ref{ch:g11:binom} --- द्विपद वितरण}

\begin{solution}{exo:g11:binom:1}
पंक्तियों $6$ और $7$:
\[
1,\ 6,\ 15,\ 20,\ 15,\ 6,\ 1
\qquad\text{and}\qquad
1,\ 7,\ 21,\ 35,\ 35,\ 21,\ 7,\ 1 .
\]
इस तरह $\binom62 = 15$, $\binom73 = 35$, $\binom74 = 35$ (समरूपता
$\binom73 = \binom74$ सफलताओं और असफलताओं की अदला-बदली को दर्शाता है)।
\end{solution}

\begin{solution}{exo:g11:binom:2}
परीक्षणों की निश्चित संख्या ($4$ रोल), प्रति रोल दो परिणाम (छह या नहीं,
$ p = \frac16$), स्वतंत्र रोल: $ X \sim \mathcal B(4, \frac16)$.
\[
\P(X=0) = \left(\frac56\right)^4 = \frac{625}{1296} \approx 0.48,
\qquad
\P(X=1) = 4 \times \frac16\left(\frac56\right)^3 = \frac{500}{1296}
\approx 0.39,
\]
\[
\P(X \geq 2) = 1 - \frac{625 + 500}{1296} = \frac{171}{1296}
\approx 0.13 .
\]
\end{solution}

\begin{solution}{exo:g11:binom:3}
\emph{1.} द्विपद $\mathcal B(20, \frac12)$: तय $ n $, वही $ p $,
स्वतंत्र टॉस।

\emph{2.} द्विपद नहीं: कार्ड बांटे जाते हैं \emph{स्थानापन्न के बिना},
इतना \omterm{def:g10:proba:distribution}{संभावना} एक इक्के का कार्ड दर कार्ड बदलता है और ड्रा होता है
स्वतंत्र नहीं.

\emph{3.} द्विपद $\mathcal B(7, 0.3)$ घोषित स्वतंत्रता से
धारणा.
\end{solution}

\begin{solution}{exo:g11:binom:4}
$\E(X) = 50 \times 0.2 = 10$;
$\V(X) = 50 \times 0.2 \times 0.8 = 8$;
$\sigma(X) = 2\sqrt2 \approx 2.83$.
\end{solution}

\begin{solution}{exo:g11:binom:5}
$ X \sim \mathcal B(6, 0.7)$.
\[
\P(X = 4) = \binom64 (0.7)^4 (0.3)^2 = 15 \times 0.2401 \times 0.09
\approx 0.324 .
\]
\[
\P(X \geq 5) = \binom65 (0.7)^5(0.3) + (0.7)^6
= 6 \times 0.16807 \times 0.3 + 0.117649 \approx 0.420 .
\]
\end{solution}

\begin{solution}{exo:g11:binom:6}
$ X \sim \mathcal B\left(8, \frac12\right)$; हर पथ में है \omterm{def:g10:proba:distribution}{संभावना}
$\frac{1}{256}$, इसलिए
\[
\P(X \geq 6) = \frac{\binom86 + \binom87 + \binom88}{256}
= \frac{28 + 8 + 1}{256} = \frac{37}{256} \approx 0.14 .
\]
अनुमान लगाना सात में से एक बार गुजरता है।
\end{solution}

\begin{solution}{exo:g11:binom:7}
 \omterm{def:g10:proba:operations}{पूरक} ``प्रत्येक में से कम से कम एक'' ``सभी पाँच समान'' हैं:
सभी ए या सभी बी, प्रत्येक के साथ \omterm{def:g10:proba:distribution}{संभावना} $\left(\frac12\right)^5 =
\frac1{32}$. इस तरह
\[
\P(\text{one of each kind}) = 1 - \frac{2}{32} = \frac{15}{16} .
\]
\end{solution}

\begin{solution}{exo:g11:binom:8}
$\P(X = 3) = p^3$ और $\P(X \geq 1) = 1 - (1-p)^3$. हल
$ p^3 = \frac{27}{64} = \left(\frac34\right)^3$ देता है $ p = \frac34$ (घन \omterm{def:g11:func:function}{समारोह} सख्ती से है \omterm{def:g11:func:monotone}{की बढ़ती}, \cref{ch:g11:func}, तो समाधान
अद्वितीय है)।
\end{solution}

\begin{solution}{exo:g11:binom:9}
$\P(\text{at least one head}) = 1 - \left(\frac12\right)^n $, तो
शर्त है $\left(\frac12\right)^n < 0.01$, यानी\ $2^n > 100$. तब से
$2^6 = 64$ और $2^7 = 128$: से $ n = 7$ उछालता है.
\end{solution}

\begin{solution}{exo:g11:binom:10}
पंक्ति का योग $ n $ के सभी पथों को गिनता है $ n $-परीक्षण \omterm{met:g10:proba:tree}{पेड़}, द्वारा क्रमबद्ध
उनकी सफलताओं की संख्या. लेकिन \omterm{met:g10:proba:tree}{पेड़} प्रत्येक परीक्षण में अपने पथ को दोगुना कर देता है
(प्रत्येक पथ एस और एफ में विभाजित होता है), इसलिए ऐसा हुआ है $2^n $ कुल मिलाकर पथ. इस तरह
$\sum_{k} \binom nk = 2^n $. वैकल्पिक रूप से, प्रेरण द्वारा: पंक्ति $0$ का योग है
$1 = 2^0$, और पास्कल का नियम पंक्ति की प्रत्येक प्रविष्टि बनाता है $ n $ योगदान करें
पंक्ति की बिल्कुल दो प्रविष्टियाँ $ n+1$, इसलिए पंक्ति का योग दोगुना है।
\end{solution}

\begin{solution}{exo:g11:binom:11}
\emph{1.} दावे के तहत, $ X \sim \mathcal B(10, 0.6)$. संक्षेप में
पहली शर्तें:
\begin{align*}
\P(X \leq 3)
&= (0.4)^{10} + 10(0.6)(0.4)^9 + 45(0.6)^2(0.4)^8
+ 120(0.6)^3(0.4)^7\\
&\approx 0.0001 + 0.0016 + 0.0106 + 0.0425 = 0.0548 .
\end{align*}

\emph{2.} परिणाम कम से कम उतना ही चरम जैसा कि देखा गया ($3$ या उससे कम
अनुमोदन) है \omterm{def:g10:proba:distribution}{संभावना} के बारे में $5.5\%$ --- अभी \emph{ऊपर}  $5\%$
सीमा। नियम को सख्ती से लागू करने पर, अवलोकन (बमुश्किल) होता है
दावे के अनुकूल है और कोई इसे अस्वीकार नहीं करता है। उदाहरण से पता चलता है
सीमा रेखा के निर्णय सीमा के चयन के प्रति कितने संवेदनशील हैं: a के साथ
$6\%$ सम्मलेन से निष्कर्ष उलट जाएगा।
\end{solution}

\begin{solution}{pb:g11:binom:1}
\textbf{1.} पंक्तियाँ: $1$; $1\,1$; $1\,2\,1$; $1\,3\,3\,1$;
$1\,4\,6\,4\,1$; $1\,5\,10\,10\,5\,1$;
$1\,6\,15\,20\,15\,6\,1$. समरूपता: कौन सा चुनना $ k $ वस्तुएं
लेना किसे चुनने जैसा ही कार्य है $ n - k $ पीछे छोड़ देना।

\textbf{2.} $1 + 4 + 6 + 4 + 1 = 16 = 2^4$;
$1 + 5 + 10 + 10 + 5 + 1 = 32 = 2^5$. प्रमाण:
$\binom nk $ के प्रत्येक आकार के उपसमुच्चय गिनें $ n $-तत्व
सेट, और उपसमुच्चय कुल मिलाकर संख्या $2^n $ (प्रत्येक तत्व में
या बाहर, स्वतंत्र रूप से)।

\textbf{3.} की समितियाँ $ k $ के बीच चुना गया $ n + 1$ लोग, एक
ज़ो किसमें से है: जो ज़ो के बिना हैं वे हैं $\binom nk $ (सभी चुनिए
$ k $ दूसरों के बीच); ज़ो वाले लोग हैं $\binom{n}{k-1}$
(उसे चुनें $ k - 1$ सहकर्मी)। कुल:
$\binom nk + \binom{n}{k-1}$.

\textbf{4.} त्रिभुज पंक्ति $7$: $1\,7\,21\,35\,\dots $: $35$.
सूत्र: $\frac{7 \times 6 \times 5}{3 \times 2 \times 1} =
35$.

\textbf{5.} $1 + 3 + 6 + 10 = 20 = \binom63$. कैस्केड:
$\binom63 = \binom52 + \binom53 = \binom52 + \binom42 +
\binom43 = \binom52 + \binom42 + \binom32 + \binom33$ ---
पास्कल के नियम का प्रत्येक अनुप्रयोग एक पद को हटा देता है
सीढ़ी.

\textbf{6.} निर्धारित अंक $ n $ उछाल का; प्रत्येक उछाल एक
स्वतंत्र \omterm{def:g11:binom:bernoulli}{बर्नौली परीक्षण} उसी के साथ $ p = \frac12$; द
स्लॉट संख्या सफलताओं को गिनती है (दाएं बाउंस): तीनों
के बक्से \cref{met:g11:binom:recognize} टिक किया गया:
$\mathcal B\!\left(n, \frac12\right)$.

\textbf{7.} संभावनाओं $\frac{1}{16}, \frac{4}{16},
\frac{6}{16}, \frac{4}{16}, \frac{1}{16}$ स्लॉट के लिए
$0, \dots, 4$: \omterm{pb:g11:quad:1}{केंद्र} स्लॉट $2$ सबसे अधिक भीड़ है.

\textbf{8.} अपेक्षित गिनती $= 1024 \times
\binom{10}{k}/1024 = \binom{10}{k}$: \omterm{prop:g10:coordgeom:midpoint}{मध्य} स्लॉट
$\binom{10}{5} = 252$ गेंदें; छेद $7$: $\binom{10}{7} = 120$;
प्रत्येक किनारे का स्लॉट: $1$ गेंद। एक ऊँचा \omterm{pb:g11:quad:1}{केंद्र} दूर गिर रहा है
सममित रूप से फुसफुसाते हुए-पतले किनारे: घंटी।

\textbf{9.} $\E = np = 5$; $ V = np(1 - p) = 2.5$;
$\sigma \approx 1.58$. अंदर $2\sigma $: स्लॉट $2$ को $8$
ढोना
\[
\frac{45 + 120 + 210 + 252 + 210 + 120 + 45}{1024}
= \frac{1002}{1024} \approx 98\,\%
\]
गेंदों की --- कहीं बेहतर
चेबीशेव के सर्व-उद्देश्य की तुलना में $75\,\%$
(\cref{pb:g11:stat:1}): घंटी की आकृतियाँ कड़ी मेहनत से \omterm{pb:g11:quad:1}{केंद्र}ित होती हैं।

\textbf{10.} $\E = 6$, $ V = 10 \times 0.6 \times 0.4 = 2.4$,
$\sigma \approx 1.55$: ढेर अपनी घंटी के आकार को बरकरार रखता है लेकिन
अपने शिखर को स्लॉट की ओर खिसकाता है $6$ --- एक झुका हुआ बोर्ड एक भरा हुआ है
सिक्का दिखाई देने लगा.

\textbf{11.} स्लॉट संख्या एक है \emph{\omterm{def:g10:vectors:sum}{जोड़}} बहुत से छोटे,
स्वतंत्र, समान आकार का मौका धक्का देता है --- और ऐसी रकम हमेशा
अपने आप को घंटी में व्यवस्थित करें: अधिकांश धक्का रद्द, चरम सीमा
सर्वसम्मति की आवश्यकता है. ऊँचाई, माप त्रुटियाँ और अनगिनत
प्राकृतिक मात्राएँ इसी प्रकार कई छोटी स्वतंत्र राशियों का योग होती हैं
प्रभाव, यही कारण है कि एक ही छाया हर जगह दिखाई देती है;
इसे प्रमाणित करने वाला प्रमेय \omterm{pb:g11:quad:1}{केंद्र}ीय सीमा प्रमेय है।

\textbf{12.} एक स्वीप: एक टीम सभी को जीतती है $4$:
$2 \times \left(\frac12\right)^4 = \frac18$.

\textbf{13.} सात खेलों की आवश्यकता है $3$--$3$ छह के बाद:
$\binom63 \left(\frac12\right)^6 = \frac{20}{64} =
\frac{5}{16}$.

\textbf{14.} में समाप्त होता है $5$: विजेता खेल 5 लेता है और $3$ का
पहला $4$:
$2 \times \binom43 \left(\frac12\right)^5 = \frac14$. में समाप्त होता है
$6$: $2 \times \binom53 \left(\frac12\right)^6 = \frac{5}{16}$.
\omterm{def:g11:prob:rv}{वितरण} ऊपर $4, 5, 6, 7$:
$\frac18, \frac14, \frac{5}{16}, \frac{5}{16}$ (\omterm{def:g10:vectors:sum}{जोड़} $1$).
अपेक्षित लंबाई: $4 \cdot \frac18 + 5 \cdot \frac14 + 6 \cdot
\frac{5}{16} + 7 \cdot \frac{5}{16} = 5.8125$ खेल. छह- और
सात-गेम श्रृंखलाओं की सबसे अधिक \omterm{def:g10:proba:distribution}{संभावना} है --- नाटक का निर्माण किया गया है
प्रारूप.

\textbf{15.} में जीतो $4$: $0.6^4 = 0.1296$; में $5$:
$\binom43\,0.6^3 \times 0.4 \times 0.6 = 0.2074$; में $6$:
$\binom53\,0.6^3 \times 0.4^2 \times 0.6 = 0.2074$; में $7$:
$\binom63\,0.6^3 \times 0.4^3 \times 0.6 = 0.1659$. कुल:
के बारे में $0.710$: ए $60\,\%$-प्रति-गेम टीम जीतती है $71\,\%$ का
शृंखला --- शृंखला बढ़त को बढ़ाती है।

\textbf{16.} एकल अंतिम: $60\,\%$. सर्वश्रेष्ठ-3:
$ p^2 + 2p^2 q = 0.36 + 0.288 = 0.648$. सीढ़ी
$60\,\% \to 65\,\% \to 71\,\%$ लंबाई के साथ जारी है: अधिक
खेल भाग्य को औसत करते हैं (लघु रूप में बड़ी संख्या का नियम),
इतने लंबे फाइनल में कौशल का ताज --- जो वास्तव में लीग है
बेचो.

\textbf{17.} उचित सिक्का: $\E = 50$, $\sigma = \sqrt{25} = 5$;
$ z = \frac{62 - 50}{5} = 2.4$: परे $2\sigma $
सम्मेलन--- सिक्का जांच का पात्र है।

\textbf{18.} $\P(X = 0) = 0.98^{50} \approx 0.364$;
$\P(X = 1) = 50 \times 0.02 \times 0.98^{49} \approx 0.372$;
$\P(X = 2) = \binom{50}{2} 0.02^2 \times 0.98^{48} \approx
0.186$. इसलिए $\P(X \geq 3) \approx 1 - 0.922 = 0.078$: के बारे में
$7.8\,\%$ --- के ऊपर $5\,\%$ दहलीज, इसलिए नहीं (अभी तक)
दावे को अस्वीकार करने के लिए पर्याप्त; दूसरा ख़राब बैच बदल जाएगा
कहानी.

\textbf{19.} $\P(\text{at least one win}) = 1 - 0.999^{1000}
\approx 1 - 0.368 = 0.632$: एक में एक हजार टिकट
हज़ार यकीन नहीं देते लेकिन $63\,\%$. आवर्ती
$0.368$ है $\frac1e $ भेष में --- स्थिरांक $ e $ बनाता है
12वीं कक्षा में इसका आधिकारिक प्रवेश।

\textbf{20.} मान्यता: निश्चित $ n $, स्वतंत्रता, स्थिर
$ p $ ---तब और केवल तभी, द्विपद. मेज़: \omterm{prop:g11:binom:pascal}{पास्कल का
त्रिकोण}, पंक्ति $ n $. आकार: घंटी, के लिए सममित
$ p = \frac12$, अन्यथा स्थानांतरित कर दिया गया। कम्पास: \omterm{pb:g11:quad:1}{केंद्र} $ np $,
फैलाना $\sqrt{np(1-p)}$ --- द z-स्कोर निर्णयों का.
वारिस: चिकनी घंटी वक्र जो बवासीर के पास पहुंचती है, और
बड़ी संख्या का नियम बताता है कि बड़े बोर्ड कभी झूठ क्यों नहीं बोलते।
\end{solution}
