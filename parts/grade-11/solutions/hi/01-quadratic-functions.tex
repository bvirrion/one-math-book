\section*{अध्याय \ref{ch:g11:quad} --- द्विघात कार्य और समीकरण}

\begin{solution}{exo:g11:quad:1}
$x^2-5x+6$: $\Delta = 25 - 24 = 1$, \omterm{def:g11:quad:discriminant}{जड़}ों $\frac{5 \pm 1}{2}$, यानी\ $2$
और $3$.

$2x^2+3x-2$: $\Delta = 9 + 16 = 25$, \omterm{def:g11:quad:discriminant}{जड़}ों $\frac{-3 \pm 5}{4}$, \omterm{def:g10:stats:mean}{अर्थ}ात\
$\frac12$ और $-2$.

$ x^2+2x+5$: $\Delta = 4 - 20 = -16 < 0$, कोई वास्तविक समाधान नहीं.

$9x^2-6x+1$: $\Delta = 36 - 36 = 0$, एक (डबल) \omterm{def:g11:quad:discriminant}{जड़}
$ x_0 = \frac{6}{18} = \frac13$.
\end{solution}

\begin{solution}{exo:g11:quad:2}
$ f(x) = x^2 - 6x + 11 = (x-3)^2 - 9 + 11 = (x-3)^2 + 2$: शिखर $(3, 2)$.

$ g(x) = -2x^2+4x+1 = -2(x^2 - 2x) + 1 = -2\bigl((x-1)^2 - 1\bigr) + 1
= -2(x-1)^2 + 3$: शिखर $(1, 3)$.

$ h(x) = 3x^2 + 6x = 3(x^2 + 2x) = 3\bigl((x+1)^2 - 1\bigr)
= 3(x+1)^2 - 3$: शिखर $(-1, -3)$.
\end{solution}

\begin{solution}{exo:g11:quad:3}
$ x^2 - 3x - 10$: $\Delta = 9 + 40 = 49$, \omterm{def:g11:quad:discriminant}{जड़}ों $\frac{3 \pm 7}{2} = -2$
और $5$; सकारात्मक अग्रणी गुणांक, इसलिए त्रिपद नकारात्मक है
के बीच सख्ती से \omterm{def:g11:quad:discriminant}{जड़}ों: समाधान सेट है $\intoo{-2}{5}$.

$2x^2 + x + 3$: $\Delta = 1 - 24 = -23 < 0$ और $ a = 2 > 0$: सदैव
सकारात्मक, समाधान सेट है $\R $.

$-x^2 + 6x - 9 = -(x-3)^2$: यह है $\geq 0$ केवल कहाँ $(x-3)^2 \leq 0$,
वह पर है $ x = 3$. समाधान सेट है $\{3\}$.
\end{solution}

\begin{solution}{exo:g11:quad:4}
$ x^2 - 7x + 6$: $1 - 7 + 6 = 0$, इसलिए $1$ एक है \omterm{def:g11:quad:discriminant}{जड़}; का उत्पाद
\omterm{def:g11:quad:discriminant}{जड़}ों है $6$, तो दूसरा \omterm{def:g11:quad:discriminant}{जड़} है $6$.

$2x^2 + 5x - 7$: $2 + 5 - 7 = 0$, इसलिए $1$ एक है \omterm{def:g11:quad:discriminant}{जड़}; उत्पाद है
$\frac{c}{a} = -\frac72$, तो दूसरा \omterm{def:g11:quad:discriminant}{जड़} है $-\frac72$.

$ x^2 + 4x - 5$: $1 + 4 - 5 = 0$, इसलिए $1$ एक है \omterm{def:g11:quad:discriminant}{जड़}; उत्पाद है $-5$, तो
दूसरा \omterm{def:g11:quad:discriminant}{जड़} है $-5$.
\end{solution}

\begin{solution}{exo:g11:quad:5}
दो संख्याएँ हल करती हैं $ x^2 - 14x + 45 = 0$ (\cref{prop:g11:quad:vieta});
$\Delta = 196 - 180 = 16$, \omterm{def:g11:quad:discriminant}{जड़}ों $\frac{14 \pm 4}{2} = 9$ और $5$.

आयत के लिए: आधी परिधि लंबाई देती है $+$ चौड़ाई $= 17$, और
क्षेत्र अपना उत्पाद देता है $60$. वे समाधान करते हैं $ x^2 - 17x + 60 = 0$;
$\Delta = 289 - 240 = 49$, \omterm{def:g11:quad:discriminant}{जड़}ों $\frac{17 \pm 7}{2} = 12$ और $5$. द
आयत है $12 \times 5$.
\end{solution}

\begin{solution}{exo:g11:quad:6}
$\Delta = (m-2)^2 - 4$. बिल्कुल एक समाधान जब $\Delta = 0$:
$(m-2)^2 = 4$, इसलिए $ m - 2 = \pm 2$, यानी\ $ m = 0$ या $ m = 4$. कोई समाधान नहीं
कब $\Delta < 0$: $(m-2)^2 < 4$, यानी\ $-2 < m - 2 < 2$, वह है
$0 < m < 4$.
\end{solution}

\begin{solution}{exo:g11:quad:7}
शिखर पर है $ t = -\frac{20}{2 \times (-5)} = 2$, और
$ h(2) = -20 + 40 + 1 = 21$: \omterm{def:g10:functions:extrema}{अधिकतम} ऊंचाई $21$ मी, के बाद पहुंचा $2$
सेकंड.

$ h(t) \geq 16$ मतलब $-5t^2 + 20t - 15 \geq 0$, यानी\ (से विभाजित करना $-5$,
जो असमानता को उलट देता है) $ t^2 - 4t + 3 \leq 0$. \omterm{def:g11:quad:discriminant}{जड़}ों का
$ t^2-4t+3$ हैं $1$ और $3$ ($1 - 4 + 3 = 0$, उत्पाद $3$), इसलिए
$ t^2 - 4t + 3 \leq 0$ बिलकुल चालू $\intcc{1}{3}$: गेंद तो है ही
$16$ समय के बीच मी ऊँचा $ t = 1$ रेत $ t = 3$ एस।
\end{solution}

\begin{solution}{exo:g11:quad:8}
साथ $ X = x^2$: $ X^2 - 13X + 36 = 0$, $\Delta = 169 - 144 = 25$, इसलिए
$ X = \frac{13 \pm 5}{2} = 9$ या $4$. वापस जाएँ $ x $: $ x^2 = 9$ देता है
$ x = \pm 3$, और $ x^2 = 4$ देता है $ x = \pm 2$. समाधान सेट है
$\{-3, -2, 2, 3\}$.
\end{solution}

\begin{solution}{exo:g11:quad:9}
शर्त यह है $ x^2 < x + 2$, यानी\ $ x^2 - x - 2 < 0$. तब से
$ x^2 - x - 2 = (x+1)(x-2)$ (\omterm{def:g11:quad:discriminant}{जड़}ों $-1$ और $2$), असमानता कायम है
के बीच सख्ती से \omterm{def:g11:quad:discriminant}{जड़}ों: $ x \in \intoo{-1}{2}$. एक स्केच पर,
\omterm{def:g11:quad:parabola}{परवलय} $ y = x^2$ रेखा के नीचे गिरता है $ y = x+2$ बिल्कुल उन दोनों के बीच
\omterm{def:g10:numbers:interunion}{चौराहा} अंक $(-1, 1)$ और $(2, 4)$.
\end{solution}

\begin{solution}{exo:g11:quad:10}
होने देना $ x > 0$ संतुष्ट $ x + \frac1x = \frac{13}{6}$. से गुणा करना $6x > 0$:
$6x^2 + 6 = 13x $, यानी\ $6x^2 - 13x + 6 = 0$. तब
$\Delta = 169 - 144 = 25$ और $ x = \frac{13 \pm 5}{12}$, देना
$ x = \frac32$ या $ x = \frac23$. दोनों सकारात्मक हैं और पारस्परिक हैं
एक दूसरे, जैसा कि अपेक्षित था: संख्या है $\frac32$ (या $\frac23$).
\end{solution}

\begin{solution}{exo:g11:quad:11}
\emph{1.} चौराहों के समाधान के अनुरूप है $ x^2 = mx - 1$, \omterm{def:g10:stats:mean}{अर्थ}ात\
$ x^2 - mx + 1 = 0$, किसका \omterm{def:g11:quad:discriminant}{विभेदक} है $\Delta = m^2 - 4$. दो अंक
कब $\abs{m} > 2$; बिल्कुल एक जब $ m = \pm 2$; कोई नहीं जब $-2 < m < 2$.

\emph{2.} के लिए $ m = 2$: $ x^2 - 2x + 1 = (x-1)^2 = 0$, तो अद्वितीय संपर्क
बिंदु है $(1, 1)$. के लिए $ m = -2$: $(x+1)^2 = 0$, संपर्क बिंदु $(-1, 1)$.
दोनों ही स्थितियों में रेखा मिलती है \omterm{def:g11:quad:parabola}{परवलय} बिना एक बिंदु पर
इसे पार करना --- यह वहां स्पर्श रेखा है, और वास्तव में यह है \omterm{def:g10:reffunc:affine}{ढलान} $ m = \pm 2$
पर मान के बराबर है $ x = \pm 1$ व्युत्पन्न का $2x $ का $ x^2$
(\cref{ch:g11:deriv}).
\end{solution}

\begin{solution}{pb:g11:quad:1}
\textbf{1.} $ x^2 - 5x + 6$: $\Delta = 1$, \omterm{def:g11:quad:discriminant}{जड़}ों $2$ और $3$.
$2x^2 + x - 6$: $\Delta = 1 + 48 = 49$, \omterm{def:g11:quad:discriminant}{जड़}ों
$\frac{-1 \pm 7}{4}$: $\frac32$ और $-2$. $ x^2 + x + 1$:
$\Delta = -3 < 0$: कोई वास्तविक समाधान नहीं.

\textbf{2.} $\Delta = 16 + 48 = 64$: \omterm{def:g11:quad:discriminant}{जड़}ों $3$ और $-1$, इसलिए
$2x^2 - 4x - 6 = 2(x - 3)(x + 1)$; \omterm{thm:g11:quad:canonical}{कानूनी फॉर्म}
$2(x - 1)^2 - 8$. गुणनखंडित प्रपत्र दर्शाता है \omterm{def:g11:quad:discriminant}{जड़}ों, द
\omterm{thm:g11:quad:canonical}{कानूनी फॉर्म} शिखर $(1, -8)$, \omterm{def:g10:algebra:expand}{विस्तार}ित रूप
$ y $संवाद $-6$.

\textbf{3.} \omterm{def:g11:quad:discriminant}{जड़}ों $1$ और $3$, अग्रणी गुणांक नकारात्मक:
त्रिपद है $\geq 0$ \emph{बीच में}  \omterm{def:g11:quad:discriminant}{जड़}ों:
$ x \in \intcc{1}{3}$.

\textbf{4.} संख्याएँ हैं \omterm{def:g11:quad:discriminant}{जड़}ों का $ x^2 - 7x + 12 = 0$:
$3$ और $4$. अगर $2$ एक है \omterm{def:g11:quad:discriminant}{जड़} का $ x^2 - 5x + c $, दूसरा \omterm{def:g11:quad:discriminant}{जड़}
$ r $ संतुष्ट $2 + r = 5$ (योग), तो $ r = 3$, और
$ c = 2 \times 3 = 6$ (उत्पाद): वी.आईète इसे पढ़ता है।

\textbf{5.} $\Delta = m^2 - 36$: दोहरा \omterm{def:g11:quad:discriminant}{जड़} के लिए $ m = \pm 6$
(\omterm{def:g11:quad:discriminant}{जड़} $\mp 3$); नहीं \omterm{def:g11:quad:discriminant}{जड़} के लिए $-6 < m < 6$; दो \omterm{def:g11:quad:discriminant}{जड़}ों के लिए
$\abs m > 6$.

\textbf{6.} $ PF^2 = x^2 + \left(x^2 - \frac14\right)^2
= x^2 + x^4 - \frac{x^2}{2} + \frac{1}{16}
= x^4 + \frac{x^2}{2} + \frac{1}{16}
= \left(x^2 + \frac14\right)^2$. इस तरह
$ PF = x^2 + \frac14$ (दोनों कारक सकारात्मक)।

\textbf{7.} से दूरी $ P(x, x^2)$ क्षैतिज तक
लाइन $ y = -\frac14$ है $ x^2 + \frac14$ --- बिल्कुल $ PF $. हर
का बिंदु \omterm{def:g11:quad:parabola}{परवलय} से समान दूरी पर है \omterm{pb:g11:quad:1}{केंद्र} और यह
\omterm{pb:g11:quad:1}{नियता}.

\textbf{8.} $ PF^2 = x^2 + \left(y - \frac14\right)^2$ और
$\operatorname{dist}^2 = \left(y + \frac14\right)^2$. समानता
देता है $ x^2 + y^2 - \frac y2 + \frac{1}{16} = y^2 + \frac y2 +
\frac1{16}$, इसलिए $ x^2 = y $: लोकस बिल्कुल वही है \omterm{def:g11:quad:parabola}{परवलय}.
एक बिंदु और एक रेखा एक बिंदु और एक के रूप में वक्र उत्पन्न करते हैं
दूरी ने वृत्त उत्पन्न किया।

\textbf{9.} के लिए $ y = \frac{x^2}{4f}$, के साथ समान \omterm{def:g10:algebra:expand}{विस्तार}
$ F(0, f)$ देता है $ PF^2 = x^2 + \left(\frac{x^2}{4f} - f\right)^2
= \left(\frac{x^2}{4f} + f\right)^2$: की दूरी $ F $ बराबर
की दूरी $ y = -f $. व्यंजन: $9 = \frac{30^2}{4f}$ देता है
$4f = 100$, $ f = 25$ सेमी: रिसीवर एक चौथाई मीटर लटका रहता है
शीर्ष के ऊपर --- के बाहर $9$ सेमी कटोरा, असली व्यंजन के रूप में
दिखाओ.

\textbf{10.} अक्ष के समानांतर आने वाला एक तरंगाग्र है
तक पहुँचने के लिए समान दूरी तय की नियताका स्तर;
समान दूरी `` को समान दूरी में बदल देती है \omterm{pb:g11:quad:1}{नियता}'' में
``समान दूरी $ F $'': सभी परावर्तित संकेत पहुंचते हैं
$ F $ \emph{चरण में}, सुदृढ़ीकरण --- यही वह जगह है जहां सुनना है।
ईमानदार प्रमाण के लिए स्पर्शरेखा की आवश्यकता होती है \omterm{def:g10:reffunc:affine}{ढलान} प्रत्येक बिंदु पर ---
का व्युत्पन्न \cref{ch:g11:deriv}, झलक मिली
\cref{exo:g11:quad:11}.

\textbf{11.} $ y = x\left(1 - \frac{x}{20}\right)$: पर भूमि
$ x = 20$ एम। शीर्ष \omterm{prop:g10:coordgeom:midpoint}{मध्य} मार्ग: $ x = 10$, ऊंचाई
$ y(10) = 10 - 5 = 5$ एम।

\textbf{12.} $ y(4) = 4 - \frac{16}{20} = 3.2$ मी: साफ़ करता है $3$ एम
कमरे के साथ, एक में स्मैक $4$ एम दीवार.

\textbf{13.} $ y(10) = 10 - \frac{100}{40} = 7.5$ एम $> 7$: द
ट्रक गुजरता है $10$ मैं \omterm{pb:g11:quad:1}{केंद्र} से बाहर हूं। $ y(15) = 10 - \frac{225}{40}
= 4.375$ म: नहीं पर $15$ एम।

\textbf{14.} $ R(p) = -2p^2 + 120p $: शीर्ष पर
$ p = \frac{120}{4} = 30$ यूरो, राजस्व
$ R(30) = 30 \times 60 = 1\,800$ यूरो.

\textbf{15.} $ y(320) = 143 \times \left(\frac{320}{640}\right)^2
= \frac{143}{4} \approx 36$ निम्नतम बिंदु से ऊपर मी.

\textbf{16.} $ x^2 = mx - 1$ देता है $ x^2 - mx + 1 = 0$,
$\Delta = m^2 - 4$: के लिए दो अंक $\abs m > 2$, के लिए कोई नहीं
$\abs m < 2$, के लिए स्पर्शरेखा $ m = \pm 2$ संपर्क के साथ
$ x = \frac m2 = \pm 1$: स्पर्शरेखा $ y = 2x - 1$ पर $(1, 1)$ और
$ y = -2x - 1$ पर $(-1, 1)$.

\textbf{17.} $ x^2 = mx + 1$ देता है $\Delta = m^2 + 4 > 0$ हरएक के लिए $ m $: दो चौराहों हमेशा। एक आंतरिक बिंदु से,
प्रत्येक रेखा वक्र को दो बार काटती है --- केवल स्पर्श रेखाएँ खींची जाती हैं
बाहर से, बिल्कुल वृत्तों की तरह।

\textbf{18.} $ y = \frac14$ की बैठक $ y = x^2$ पर $ x = \pm\frac12$:
तार की लंबाई $1$. के लिए $ y = \frac{x^2}{4f}$: $ y = f $ देता है
$ x^2 = 4f^2$, $ x = \pm 2f $: लंबाई $4f $ ---फोकल चौड़ाई है
फोकल दूरी का चार गुना (डिश डिजाइनर इसे कहते हैं
लैटस रेक्टम)।

\textbf{19.} का एक बिंदु $ y = x^2$ है $(a, a^2)$; इसका \omterm{def:g10:functions:function}{छवि} है
$(X, Y) = (ka, ka^2)$, और $\frac{X^2}{k} = \frac{k^2a^2}{k} =
ka^2 = Y $: द \omterm{def:g10:functions:function}{छवि} वक्र बिलकुल है $ y = \frac{x^2}{k}$.
का चयन $ k = 4f$ हर तक पहुंचता है \omterm{def:g11:quad:parabola}{परवलय}: वे सब हैं
एक मास्टर वक्र का आवर्धन। (वृत्त सभी समान हैं
भी --- लेकिन दीर्घवृत्त नहीं हैं: स्क्वैश मायने रखता है, ज़ूम मायने नहीं रखता।)

\textbf{20.} एक \omterm{def:g10:algebra:equation}{समीकरण} तीन पठनीय वेशभूषा के साथ; द
एक बिंदु और एक रेखा से समान दूरी का स्थान; परावर्तक
जो अपने पास समानांतर किरणें एकत्रित करता है \omterm{pb:g11:quad:1}{केंद्र} और वह केबल
एक समान डेक रखता है; प्रत्येक फेंकी गई गेंद का पथ; और,
आवर्धन तक, एक एकल वक्र --- एक \omterm{def:g11:quad:parabola}{परवलय},
हर जगह.
\end{solution}
