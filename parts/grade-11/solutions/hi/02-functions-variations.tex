\begin{solution}{exo:g11:func:5}
होने देना $2 \leq u < v $. तब
\[
f(v) - f(u) = v^2 - 4v - u^2 + 4u = (v-u)(v+u) - 4(v-u)
= (v - u)(u + v - 4).
\]
दोनों कारक सकारात्मक हैं: $ v - u > 0$, और $ u + v > 2 + 2 = 4$ तब से
$ u \geq 2$ और $ v > 2$. इस तरह $ f(v) > f(u)$: $ f $ सख्ती से है \omterm{def:g11:func:monotone}{की बढ़ती}
पर $\intco{2}{+\infty}$.
\end{solution}

\begin{solution}{exo:g11:func:10}
\emph{1. सत्य.} अगर $ u < v $, तब
$(f+g)(v) - (f+g)(u) = \bigl(f(v)-f(u)\bigr) + \bigl(g(v)-g(u)\bigr)$ है
दो अऋणात्मक संख्याओं का योग।

\emph{2. मिथ्या।} $ f(x) = g(x) = x $ हैं \omterm{def:g11:func:monotone}{की बढ़ती} पर $\R $, लेकिन उनके
उत्पाद $ x^2$ क्या नहीं है \omterm{def:g11:func:monotone}{की बढ़ती} पर $\R $ (यह घटता जाता है
$\intoc{-\infty}{0}$). (यदि दोनों हों तो कथन सत्य हो जाता है कार्य हैं
गैर-नकारात्मक भी।)

\emph{3. मिथ्या।} वही प्रति उदाहरण: $ f(x) = x $ है \omterm{def:g11:func:monotone}{की बढ़ती} पर $\R $
लेकिन $ f^2(x) = x^2$ क्या नहीं है।
\end{solution}

\begin{solution}{exo:g11:func:11}
\emph{1.} के लिए $0 < u < v $:
\[
f(v) - f(u) = (v - u) + \frac1v - \frac1u
= (v - u) + \frac{u - v}{uv}
= (v - u)\left(1 - \frac{1}{uv}\right)
= (v - u)\,\frac{uv - 1}{uv}.
\]

\emph{2.} हमेशा $ v - u > 0$ और $ uv > 0$. अगर $0 < u < v \leq 1$, तब
$ uv < 1$, इसलिए $ uv - 1 < 0$ और $ f(v) < f(u)$: $ f $ सख्ती से है \omterm{def:g10:functions:variations}{घटते}
पर $\intoc{0}{1}$. अगर $1 \leq u < v $, तब $ uv > 1$ और $ f(v) > f(u)$:
$ f $ सख्ती से है \omterm{def:g11:func:monotone}{की बढ़ती} पर $\intco{1}{+\infty}$.

\emph{3.} $ f $ पहले घट जाती है $1$ और उसके बाद बढ़ता है, इसलिए
$ f(x) \geq f(1) = 2$ सभी के लिए $ x > 0$, बिल्कुल समानता के साथ $ x = 1$.
(यह असमानता कई रूपों में पुनः प्रकट होती है; तुलना करें
\cref{ex:g11:quad:canonical} और स्पर्शरेखा-रेखा असमानताएँ
\cref{ch:g12:deriv}.)
\end{solution}

\begin{solution}{pb:g11:func:1}
\textbf{1.} $ x^4 - 3x^2$: \omterm{def:g11:func:parity}{यहां तक की} ($(-x)^4 - 3(-x)^2 = x^4 -
3x^2$). $ x^3 - x $: \omterm{def:g11:func:parity}{विषम}. $ x^2 + x $: कोई भी नहीं ($ f(-1) = 0 \neq
f(1) = 2$ और $\neq -f(1)$). $\abs x $: \omterm{def:g11:func:parity}{यहां तक की}. $\frac1x $: \omterm{def:g11:func:parity}{विषम}.

\textbf{2.} \omterm{def:g11:func:parity}{यहां तक की}: वक्र सममित है $ y $-अक्ष
(दर्पण). \omterm{def:g11:func:parity}{विषम}: मूल बिंदु के बारे में सममित (आधा-मोड़)।

\textbf{3.} अगर $ f $ और $ g $ हैं \omterm{def:g11:func:parity}{विषम}:
$(fg)(-x) = f(-x)g(-x) = (-f(x))(-g(x)) = (fg)(x)$: \omterm{def:g11:func:parity}{यहां तक की}.
मेज़: \omterm{def:g11:func:parity}{यहां तक की} $\times $ \omterm{def:g11:func:parity}{यहां तक की} $=$ \omterm{def:g11:func:parity}{यहां तक की}; \omterm{def:g11:func:parity}{यहां तक की} $\times $ \omterm{def:g11:func:parity}{विषम} $=$ \omterm{def:g11:func:parity}{विषम};
\omterm{def:g11:func:parity}{विषम} $\times $ \omterm{def:g11:func:parity}{विषम} $=$ \omterm{def:g11:func:parity}{यहां तक की} --- समता के साथ चिह्न नियम.

\textbf{4.} \omterm{def:g11:func:parity}{विषम} पर $0$: $ f(0) = -f(0)$, इसलिए $ f(0) = 0$. \omterm{def:g11:func:parity}{यहां तक की}:
$ f(-3) = f(3) = 7$. दोनों \omterm{def:g11:func:parity}{यहां तक की} और \omterm{def:g11:func:parity}{विषम}:
$ f(x) = f(-x) = -f(x)$ सभी के लिए $ x $, इसलिए $2f(x) = 0$: $ f $ है
शून्य \omterm{def:g11:func:function}{समारोह}.

\textbf{5.} दाहिना आधा ($ x \geq 0$) एक निर्धारित करता है यहां तक कि \omterm{def:g11:func:function}{समारोह} पूर्णतः: बायां आधा भाग इसका दर्पण है। कोई भी नहीं \omterm{def:g11:func:parity}{यहां तक की}
और न \omterm{def:g11:func:parity}{विषम}: $ x^2 + x $ (प्रश्न 1) --- इसकी दो समरूपताएँ विफल हो जाती हैं
$ x = 1$.

\textbf{6.} $ E(x) = \frac{(x^3 + x^2 + 1) + (-x^3 + x^2 +
1)}{2} = x^2 + 1$ (\omterm{def:g11:func:parity}{यहां तक की}), $ O(x) = \frac{(x^3 + x^2 + 1) -
(-x^3 + x^2 + 1)}{2} = x^3$ (\omterm{def:g11:func:parity}{विषम}); उनका योग है $ f $.

\textbf{7.} $ E(-x) = \frac{f(-x) + f(x)}{2} = E(x)$: \omterm{def:g11:func:parity}{यहां तक की}.
$ O(-x) = \frac{f(-x) - f(x)}{2} = -O(x)$: \omterm{def:g11:func:parity}{विषम}. और
$ E(x) + O(x) = f(x)$ प्रत्यक्ष \omterm{def:g10:vectors:sum}{जोड़} द्वारा: प्रत्येक \omterm{def:g11:func:function}{समारोह}
विभाजन

\textbf{8.} से $ E_1 + O_1 = E_2 + O_2$:
$ E_1 - E_2 = O_2 - O_1$. बाईं ओर है \omterm{def:g11:func:parity}{यहां तक की}, दाहिनी ओर
\omterm{def:g11:func:parity}{विषम} --- इसलिए इस \omterm{def:g11:func:function}{समारोह} दोनों है, इसलिए शून्य (प्रश्न 4):
$ E_1 = E_2$ और $ O_1 = O_2$. केवल एक विभाजन.

\textbf{9.} $ E(x) = \frac12\left(\frac{1}{1 - x} + \frac{1}{1 +
x}\right) = \frac12 \cdot \frac{(1 + x) + (1 - x)}{1 - x^2}
= \frac{1}{1 - x^2}$, और
$ O(x) = \frac12 \cdot \frac{(1 + x) - (1 - x)}{1 - x^2}
= \frac{x}{1 - x^2}$. जाँच करना: $ E $ है \omterm{def:g11:func:parity}{यहां तक की}, $ O $ है \omterm{def:g11:func:parity}{विषम}, और
उनका योग है $\frac{1 + x}{1 - x^2} =
\frac{1 + x}{(1 - x)(1 + x)} = \frac{1}{1 - x} = f(x)$.

\textbf{10.} क्योंकि प्रत्येक टुकड़ा एक समरूपता का पालन करता है, आधा
कार्य (और आधा डेटा) प्रत्येक के लिए पर्याप्त है: एक तरफ हल करें,
दूसरे को दर्पण --- और कई ऑपरेशन (एकीकरण,
श्रृंखला, सिग्नल विश्लेषण) सममित टुकड़ों का कहीं अधिक इलाज करते हैं
बस कच्चे से कार्य.

\textbf{11.} $(x-3)^2$: स्थानांतरित $3$ सही। $ x^2 + 2$: स्थानांतरित
$2$ ऊपर। $(x-3)^2 + 2$: दोनों। $-x^2$: उलटा हो गया।
$2x^2$: द्वारा लंबवत रूप से फैला हुआ $2$ (संकीर्ण कटोरा).

\textbf{12.} के लिए $ x \geq 2$: $(x - 2) + (x + 2) = 2x $. के लिए
$-2 \leq x \leq 2$: $(2 - x) + (x + 2) = 4$. के लिए $ x \leq -2$:
$-2x $. वक्र के साथ उतरता है \omterm{def:g10:reffunc:affine}{ढलान} $-2$, ऊंचाई पर सपाट चलता है
$4$ बीच में $-2$ और $2$, फिर साथ चढ़ता है \omterm{def:g10:reffunc:affine}{ढलान} $2$: एक घाटी
समतल फर्श के साथ.

\textbf{13.} लंबवत बदलाव: $ f(x) + 2$ रहता है \omterm{def:g11:func:parity}{यहां तक की}
($ f(-x) + 2 = f(x) + 2$). लंबवत खिंचाव: $2f(x)$ रहता है
\omterm{def:g11:func:parity}{यहां तक की}. क्षैतिज बदलाव: $(x - 3)^2$ क्या नहीं है \omterm{def:g11:func:parity}{यहां तक की}
($ f(-1) = 16 \neq 4 = f(1)$): बग़ल में फिसलने से नष्ट हो जाता है
दर्पण.

\textbf{14.} समतल फर्श पर मूल्य है $4 \neq 6$. के लिए
$ x \geq 2$: $2x = 6$, $ x = 3$. के लिए $ x \leq -2$: $-2x = 6$,
$ x = -3$. समाधान: $\pm 3$.

\textbf{15.} $ f(x) = a(x - 2)^2 - 3$ साथ $ f(0) = 4a - 3 = 1$:
$ a = 1$, इसलिए $ f(x) = (x - 2)^2 - 3$.

\textbf{16.} $ g(v) - g(u) = (v - u) + 4\,\frac{u - v}{uv}
= (v - u)\,\frac{uv - 4}{uv}$: के लिए नकारात्मक $ u < v $ साथ
$ uv < 4$, के साथ सकारात्मक $ uv > 4$: \omterm{def:g10:functions:variations}{घटते} पर
$\intoc{0}{2}$, \omterm{def:g11:func:monotone}{की बढ़ती} पर $\intco{2}{+\infty}$; \omterm{def:g10:functions:extrema}{न्यूनतम} पर
$ x = 2$, मूल्य का $ g(2) = 4$.

\textbf{17.} पक्षों $ x $ और $\frac{16}{x}$:
$ P(x) = 2\left(x + \frac{16}{x}\right)$, \omterm{def:g10:functions:extrema}{न्यूनतम} (प्रश्न)
16 की मशीन के साथ $4 \to 16$, मोड़ $\sqrt{16} = 4$)
पर $ x = 4$: द $4 \times 4$ \emph{वर्ग}, परिधि $16$ ---
डिडो का उत्तर, अब प्रत्येक वास्तविक पार्श्व लंबाई के लिए सिद्ध हो गया है।

\textbf{18.} पर $\intco{0}{+\infty}$: $ x \mapsto x^2 + 1$
बढ़ता है, और $\sqrt{\phantom{x}}$ बढ़ता है, इसलिए
समग्र वृद्धि. और $ h(-x) = \sqrt{x^2 + 1} = h(x)$: $ h $
है \omterm{def:g11:func:parity}{यहां तक की}, इसलिए दर्पण समरूपता से यह घटता जाता है
$\intoc{-\infty}{0}$. \omterm{def:g10:functions:extrema}{न्यूनतम} $ h(0) = 1$.

\textbf{19.} वर्ग दूरी से $(3, 0)$ मुद्दे की
$(x, \sqrt x)$:
$ d^2(x) = (x - 3)^2 + x = x^2 - 5x + 9$, \omterm{def:g10:functions:extrema}{न्यूनतम} पर
$ x = \frac52$: निकटतम बिंदु
$\left(\frac52, \sqrt{\frac52}\right)$, दूरी
$\sqrt{\frac{11}{4}} = \frac{\sqrt{11}}{2}$. कम से कम $ d^2$
वैध है क्योंकि वर्ग करना सख्ती से है \omterm{def:g11:func:monotone}{की बढ़ती} पर
सकारात्मक संख्याएँ: $ d $ और $ d^2$ एक ही स्थान पर नीचे से बाहर।
