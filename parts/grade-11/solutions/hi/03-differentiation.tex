\section*{अध्याय \ref{ch:g11:deriv} --- विभेदन: एक पहला कोर्स}

\begin{solution}{exo:g11:deriv:1}
के लिए $f(x) = x^2$ पर $ a = 2$:
\[
\frac{(2+h)^2 - 4}{h} = \frac{4h + h^2}{h} = 4 + h
\xrightarrow[h\to0]{} 4,
\]
इसलिए $ f'(2) = 4$. के लिए $ g(x) = x^2 + 3x $ पर $ a = 1$: $ g(1) = 4$ और
\[
\frac{(1+h)^2 + 3(1+h) - 4}{h} = \frac{5h + h^2}{h} = 5 + h
\xrightarrow[h\to0]{} 5,
\]
इसलिए $ g'(1) = 5$.
\end{solution}

\begin{solution}{exo:g11:deriv:2}
$ f'(x) = 12x^2 - 4x + 1$.

$ g'(x) = \frac{5}{2\sqrt x} - \frac{3}{x^2}$ (पर $\intoo{0}{+\infty}$).

$ h'(x) = 2(x^2 - 3) + (2x+1)(2x) = 2x^2 - 6 + 4x^2 + 2x
= 6x^2 + 2x - 6$ (प्रॉडक्ट नियम)।
\end{solution}

\begin{solution}{exo:g11:deriv:3}
$ f(x) = x^2 - 3x $, $ a = 2$: $ f(2) = -2$, $ f'(x) = 2x - 3$ इसलिए $ f'(2) = 1$;
\omterm{def:g11:deriv:tangent}{स्पर्शरेखा} $ y = -2 + (x - 2) = x - 4$.

$ f(x) = \frac1x $, $ a = 1$: $ f(1) = 1$, $ f'(1) = -1$; \omterm{def:g11:deriv:tangent}{स्पर्शरेखा}
$ y = 1 - (x - 1) = 2 - x $.

$ f(x) = \sqrt x $, $ a = 4$: $ f(4) = 2$, $ f'(4) = \frac{1}{2\sqrt4} =
\frac14$; \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} $ y = 2 + \frac14(x - 4) = \frac{x}{4} + 1$.
\end{solution}

\begin{solution}{exo:g11:deriv:4}
$ f'(x) = \frac{(x-1) - (x+2)}{(x-1)^2} = \frac{-3}{(x-1)^2}$.

$ g'(x) = \frac{2x(x^2+1) - x^2 \times 2x}{(x^2+1)^2}
= \frac{2x}{(x^2+1)^2}$.

$ h'(x) = -\frac{2x+1}{(x^2+x+1)^2}$ (भागफल नियम के साथ $ u = 1$).
\end{solution}

\begin{solution}{exo:g11:deriv:5}
$ f'(x) = 3x^2 - 12x + 9 = 3(x^2 - 4x + 3) = 3(x-1)(x-3)$: सकारात्मक
बाहर $\intcc{1}{3}$, अंदर से नकारात्मक। इसलिए $ f $ है \omterm{def:g11:func:monotone}{की बढ़ती} पर
$\intoc{-\infty}{1}$, \omterm{def:g10:functions:variations}{घटते} पर $\intcc{1}{3}$, \omterm{def:g11:func:monotone}{की बढ़ती} पर
$\intco{3}{+\infty}$, एक स्थानीय के साथ \omterm{def:g10:functions:extrema}{अधिकतम} $ f(1) = 1 - 6 + 9 - 2 = 2$ और
एक स्थानीय \omterm{def:g10:functions:extrema}{न्यूनतम} $ f(3) = 27 - 54 + 27 - 2 = -2$.
\end{solution}

\begin{solution}{exo:g11:deriv:6}
पर $\intoo{-1}{+\infty}$:
\[
f'(x) = \frac{2x(x+1) - (x^2+3)}{(x+1)^2}
= \frac{x^2 + 2x - 3}{(x+1)^2}
= \frac{(x+3)(x-1)}{(x+1)^2}.
\]
के लिए $ x > -1$, दोनों $ x + 3$ और $(x+1)^2$ सकारात्मक हैं, इसलिए $ f'(x)$ के पास है
का संकेत $ x - 1$: $ f $ है \omterm{def:g10:functions:variations}{घटते} पर $\intoc{-1}{1}$ और \omterm{def:g11:func:monotone}{की बढ़ती} पर
$\intco{1}{+\infty}$, साथ \omterm{def:g10:functions:extrema}{न्यूनतम} $ f(1) = \frac{4}{2} = 2$.
\end{solution}

\begin{solution}{exo:g11:deriv:7}
$ f'(x) = 2x - 4$.

\emph{1.} क्षैतिज \omterm{def:g11:deriv:tangent}{स्पर्शरेखा}: $ f'(x) = 0$ पर $ x = 2$; बात यह है
$(2, f(2)) = (2, 1)$ (के शीर्ष \omterm{def:g11:quad:parabola}{परवलय}).

\emph{2.} के समानांतर $ y = 2x + 1$ मतलब \omterm{def:g10:reffunc:affine}{ढलान} $2$: $2x - 4 = 2$, इसलिए
$ x = 3$, बात बता रहा हूँ $(3, f(3)) = (3, 2)$.
\end{solution}

\begin{solution}{exo:g11:deriv:8}
चौड़ाई के साथ $ x $ (दो बाड़े वाले किनारे) और लंबाई $60 - 2x $:
$ A(x) = x(60 - 2x) = 60x - 2x^2$ पर $\intoo{0}{30}$. तब
$ A'(x) = 60 - 4x $, पहले सकारात्मक $ x = 15$ और इसके बाद नकारात्मक: क्षेत्र
के लिए \omterm{def:g10:functions:extrema}{अधिकतम} है $ x = 15$, एक देते हुए $15 \times 30$ मी क्षेत्र का घेरा
$450$ एम $^2$. शिखर सूत्र $\alpha = -\frac{60}{2 \times (-2)} = 15$
का \cref{ch:g11:quad} बिना गणना के वही उत्तर देता है।
\end{solution}

\begin{solution}{exo:g11:deriv:9}
होने देना $ f(x) = \frac{x+1}{2} - \sqrt x $ पर $\intoo{0}{+\infty}$. तब
\[
f'(x) = \frac12 - \frac{1}{2\sqrt x} = \frac{\sqrt x - 1}{2\sqrt x},
\]
जिसका चिन्ह है $\sqrt x - 1$: नकारात्मक पर $\intoo{0}{1}$, सकारात्मक
पर $\intoo{1}{+\infty}$. इसलिए $ f $ घटता है फिर बढ़ता है, साथ \omterm{def:g10:functions:extrema}{न्यूनतम}
$ f(1) = 1 - 1 = 0$. इस तरह $ f(x) \geq 0$ सभी के लिए $ x > 0$, वह है
$\sqrt x \leq \frac{x+1}{2}$, केवल समानता के साथ $ x = 1$.
\end{solution}

\begin{solution}{exo:g11:deriv:10}
से $\pi r^2 h = 500$, $ h = \frac{500}{\pi r^2}$, इसलिए
\[
S(r) = 2\pi r^2 + 2\pi r \cdot \frac{500}{\pi r^2}
= 2\pi r^2 + \frac{1000}{r},
\qquad
S'(r) = 4\pi r - \frac{1000}{r^2}.
\]
$ S'(r) = 0$ कब $4\pi r^3 = 1000$, यानी\ $ r^3 = \frac{250}{\pi}$, इसलिए
$ r = \sqrt[3]{250/\pi}$ ($\approx 4.3$ सेमी)। तब से $ S'(r) =
\frac{4\pi r^3 - 1000}{r^2}$ इस मान से पहले ऋणात्मक और धनात्मक है
के बाद, यह एक है \omterm{def:g10:functions:extrema}{न्यूनतम}.
\end{solution}

\begin{solution}{exo:g11:deriv:11}
 \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} को $ y = x^2$ के बिंदु पर \omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज} $ a $ है
$ y = a^2 + 2a(x - a) = 2ax - a^2$. यह गुजरता है $ P(1, -3)$ कब
$-3 = 2a - a^2$, यानी\ $ a^2 - 2a - 3 = 0$, किसका \omterm{def:g11:quad:discriminant}{जड़}ों हैं $ a = 3$ और
$ a = -1$ ($\Delta = 16$). इसलिए हैं \emph{दो} \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} के माध्यम से
$ P $:
\[
y = 6x - 9 \quad (a = 3),
\qquad
y = -2x - 1 \quad (a = -1).
\]
\end{solution}

\begin{solution}{exo:g11:deriv:12}
से \cref{ex:g11:deriv:study}, $ f $ स्थानीय तक बढ़ जाता है \omterm{def:g10:functions:extrema}{अधिकतम}
$ f(-1) = 3$, स्थानीय तक घट जाती है \omterm{def:g10:functions:extrema}{न्यूनतम} $ f(1) = -1$, फिर बढ़ जाता है
पुनः; के लिए $ x $ बड़ा सकारात्मक (सम्मान)\ नकारात्मक), $ x^3$ हावी है और $ f $
मनमाने ढंग से बड़ा लेता है (सम्मान)\ छोटे) मान। क्षैतिज रेखाएँ पढ़ना
$ y = k $ के आर - पार \omterm{def:g10:functions:table}{विविधता तालिका}:
\begin{itemize}
  \item $ k > 3$ या $ k < -1$: रेखा वक्र से एक बार मिलती है
        --- $1$ समाधान;
  \item $ k = 3$ या $ k = -1$: रेखा एक मोड़ को छूती है और काटती है
        एक अन्य शाखा --- $2$ समाधान;
  \item $-1 < k < 3$: रेखा तीनों शाखाओं को पार करती है --- $3$
        समाधान.
\end{itemize}
\end{solution}

\begin{solution}{pb:g11:deriv:1}
\textbf{1.} $ f'(x) = 6x - 5$; $ g'(x) = 3x^2 - 12$. परिभाषा:
$\frac{(1 + h)^2 - 1}{h} = 2 + h \to 2$: द यौगिक का
$ x^2$ पर $1$ है $2$.

\textbf{2.} $ g(2) = 8 - 24 = -16$ और $ g'(2) = 12 - 12 = 0$:
द \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} क्षैतिज रेखा है $ y = -16$. एक क्षैतिज
\omterm{def:g11:deriv:tangent}{स्पर्शरेखा} एक महत्वपूर्ण बिंदु का संकेत देता है --- यहाँ एक स्थानीय है \omterm{def:g10:functions:extrema}{न्यूनतम}, जैसे
द \omterm{def:g10:functions:table}{विविधता तालिका} प्रश्न 5 की पुष्टि करता है।

\textbf{3.} \omterm{def:g10:reffunc:affine}{ढलान} $2a $ पर $(a, a^2)$:
$ y = a^2 + 2a(x - a) = 2ax - a^2$.

\textbf{4.} पर $ x = 0$: $ y = -a^2$: द \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} को पार कर जाता है
गहराई पर अक्ष $ a^2$ उद्गम स्थल से नीचे --- ठीक उतनी ही दूर
\emph{नीचे} शून्य के रूप में $ P $ इसके ऊपर बैठता है. विधि: आकर्षित करने के लिए
\omterm{def:g11:deriv:tangent}{स्पर्शरेखा} ऊंचाई के एक बिंदु पर $ h $, इसे अक्ष बिंदु पर \omterm{def:g10:vectors:sum}{जोड़}ें
गहराई $ h $ मूल के नीचे. (नहीं यौगिक पर आवश्यकता है
ड्राइंग बोर्ड.)

\textbf{5.} $ g'(x) = 3(x^2 - 4)$: सकारात्मक पर
$\intoo{-\infty}{-2}$, नकारात्मक पर $\intoo{-2}{2}$, सकारात्मक
परे. $ g $ स्थानीय तक बढ़ जाता है \omterm{def:g10:functions:extrema}{अधिकतम} $ g(-2) = 16$, गिर जाता है
एक स्थानीय को \omterm{def:g10:functions:extrema}{न्यूनतम} $ g(2) = -16$, फिर बढ़ जाता है।

\textbf{6.} $ F\left(0, \frac14\right)$ और $ T(0, -a^2)$:
$ FT = \frac14 + a^2$.

\textbf{7.} $ FP = a^2 + \frac14 = FT $: समद्विबाहु पर $ F $, तो
आधार कोण बराबर हैं: $\widehat{FTP} = \widehat{FPT}$.

\textbf{8.} आने वाली किरण पर $ P $ ऊर्ध्वाधर है, इसलिए समानांतर है
लाइन तक $(TF)$ (द $ y $-अक्ष). \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} $(TP)$ पार
दोनों समानांतर हैं, इसलिए किरण और के बीच का कोण है \omterm{def:g11:deriv:tangent}{स्पर्शरेखा}
के बराबर होती है $\widehat{FTP}$ (वैकल्पिक कोण) --- जो बराबर हो
$\widehat{FPT}$, बीच का कोण $[PF]$ और यह \omterm{def:g11:deriv:tangent}{स्पर्शरेखा}.
के दोनों पक्षों पर समान कोण \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} पर $ P $.

\textbf{9.} परावर्तन के नियम के अनुसार, किरण का आगमन
लंबवत पर $ P $ रेखा के अनुदिश समान कोण बनाती हुई पत्तियाँ
दूसरी ओर --- प्रश्न 8 द्वारा, साथ में $[PF]$: के माध्यम से
\omterm{pb:g11:quad:1}{केंद्र}, जो कुछ भी $ a $ है। उलटा, प्रकाश का जन्म हुआ $ F $ पत्तियां
दर्पण का प्रत्येक बिंदु अक्ष के समानांतर होता है। बर्तन इकट्ठा करो,
हेडलाइट्स बीम: सिद्ध।

\textbf{10.} $ a = 1$: \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} $ y = 2x - 1$, $ T(0, -1)$;
$ FP = \sqrt{1 + \left(1 - \frac14\right)^2} =
\sqrt{\frac{25}{16}} = \frac54$; $ FT = \frac14 + 1 = \frac54$:
समद्विबाहु, जैसा कि वादा किया गया था।

\textbf{11.}  \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} पर $ x_n $ है
$ y = f(x_n) + f'(x_n)(x - x_n)$; यह पार हो जाता है $ y = 0$ कहाँ
$ x = x_n - \frac{f(x_n)}{f'(x_n)}$.

\textbf{12.} $ x_1 = 1 - \frac{-1}{2} = \frac32$;
$ x_2 = \frac32 - \frac{1/4}{3} = \frac{17}{12}$;
$ x_3 = \frac{577}{408}$: बगुला का अनुमान का $\sqrt2$,
यानी\ की फिक्स्ड-प्वाइंट मशीन \cref{pb:g10:functions:1}.
पहचान: $ x - \frac{x^2 - 2}{2x} = \frac{2x^2 - x^2 + 2}{2x}
= \frac{x^2 + 2}{2x} = \frac12\left(x + \frac2x\right)$ ---
बगुला का नुस्खा \emph{है} न्यूटन की विधि लागू की गई
$ x^2 - 2$, दो सहस्राब्दी पहले।

\textbf{13.} $ x_1 = 5 - \frac{125 - 100}{75} = \frac{14}{3}
\approx 4.6667$; $ x_2 \approx 4.6667 - \frac{1.63}{65.33}
\approx 4.6417$. कैलकुलेटर: $\sqrt[3]{100} = 4.6416\ldots $ ---
दो चरणों में चार सही दशमलव।

\textbf{14.} $ f'(2) = 0$: द \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} आरंभिक बिंदु पर है
क्षैतिज (प्रश्न 2) और कभी भी अक्ष को पार नहीं करता---
सूत्र शून्य से विभाजित होता है. चेतावनी: न्यूटन से दूर शुरुआत करें
महत्वपूर्ण बिंदु (और, अधिक सामान्यतः, काफी करीब \omterm{def:g11:quad:discriminant}{जड़}
आप चाहते हैं)

\textbf{15.} द्विभाजन प्रति चरण एक बाइनरी अंक प्राप्त करता है; न्यूटन
मोटे तौर पर \emph{दोगुना हो जाता है} प्रत्येक चरण में सही अंकों की संख्या
($1 \to 3 \to 6$ यहाँ)। जब प्रत्येक कदम महँगा हो --- लाखों
मापदंडों की, वास्तविक समय नेविगेशन --- धड़कनों को दोगुना करना
इंचिंग, यही कारण है कि \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} लाइन इतनी अंदर तक चलती है
सिलिकॉन.

\textbf{16.} $ d^2 = 900 - w^2$, इसलिए
$ S(w) = w(900 - w^2) = 900w - w^3$, के लिए $0 < w < 30$.

\textbf{17.} $ S'(w) = 900 - 3w^2 = 0$ पर
$ w = \sqrt{300} = 10\sqrt3 \approx 17.3$ सेमी ($ S' > 0$ पहले,
$< 0$ बाद एक \omterm{def:g10:functions:extrema}{अधिकतम}). तब
$ d = \sqrt{600} = 10\sqrt6 \approx 24.5$ सेमी।

\textbf{18.} $\frac{d^2}{w^2} = \frac{600}{300} = 2$, इसलिए
$ d = w\sqrt2$: गहराई चौड़ाई गुना है $\sqrt2$ --- द
बढ़ई का अनुपात (और व्यास-तिहाई निर्माण
इसे कम्पास के साथ वितरित करता है, साइट पर कोई बीजगणित नहीं है)।

\textbf{19.} चौकोर बीम: $ w = d = \sqrt{450} \approx 21.2$
सेमी, कठोरता $ w^3 = 450^{3/2} \approx 9\,546$. इष्टतम किरण:
$ S = 10\sqrt3 \times 600 = 6\,000\sqrt3 \approx 10\,392$. लाभ:
के बारे में $8.9\,\%$ एक ही लॉग से कठोर --- यौगिक

\textbf{20.} ज्यामिति: स्पर्शरेखाके बराबर कोण बन गए

\omterm{def:g11:deriv:tangent}{स्पर्शरेखा} हल करती है \omterm{def:g10:algebra:equation}{समीकरण} अंकों को दोगुना करने की गति से।

\end{solution}
