\section*{अध्याय \ref{ch:g11:prob} --- संभाव्यता और यादृच्छिक चर}

\begin{solution}{exo:g11:prob:1}
\omterm{def:g11:prob:model}{वर्दी मॉडल} पर $52$ कार्ड. दिल: $\frac{13}{52} = \frac14$. राजा:
$\frac{4}{52} = \frac{1}{13}$. हृदय या राजा:
$\frac{13 + 4 - 1}{52} = \frac{16}{52} = \frac{4}{13}$ (के राजा
दिल एक बार गिने जाते हैं)। कोई भी नहीं: $1 - \frac{4}{13} = \frac{9}{13}$.
\end{solution}

\begin{solution}{exo:g11:prob:2}

$\P(S = 6) = \frac{5}{36}$.

$ S \geq 10$: के लिए तीन \omterm{def:g10:vectors:sum}{जोड़}े $10$, दो के लिए $11$, एक के लिए $12$:
$\P(S \geq 10) = \frac{6}{36} = \frac16$.

$ S $ सम: रकम $2, 4, 6, 8, 10, 12$ पास होना $1 + 3 + 5 + 5 + 3 + 1 = 18$
\omterm{def:g10:vectors:sum}{जोड़}े: \omterm{def:g10:proba:distribution}{संभावना} $\frac{18}{36} = \frac12$.
\end{solution}

\end{solution}

\begin{solution}{exo:g11:prob:5}
जीत हैं $0$, $2$, $5$ संभावनाओं के साथ यूरो $0.5$, $0.3$,
$0.2$; घटाना $2$-यूरो हिस्सेदारी, $ G $ मान लेता है $-2, 0, 3$:
\begin{center}
\begin{tabular}{c|ccc}
$ g $ & $-2$ & $0$ & $3$\\
\hline
$\P(G=g)$ & $0.5$ & $0.3$ & $0.2$
\end{tabular}
\end{center}
$\E(G) = -1 + 0 + 0.6 = -0.4$: खिलाड़ी हार जाता है $40$ प्रति गेम सेंट
औसत---प्रतिकूल.
\end{solution}

\begin{solution}{exo:g11:prob:6}
क्रम में आरेखण: $\P(X = 2) = \frac35 \times \frac24 = \frac{3}{10}$
(लाल फिर लाल); $\P(X = 0) = \frac25 \times \frac14 = \frac{1}{10}$
(नीला फिर नीला); इस तरह
$\P(X = 1) = 1 - \frac{3}{10} - \frac{1}{10} = \frac{6}{10}$. तब
\[
\E(X) = 0 \times \frac1{10} + 1 \times \frac6{10} + 2 \times \frac3{10}
= \frac{12}{10} = 1.2 ,
\]
अंतर्ज्ञान के अनुरूप: $\frac35$ दो में से दो ड्रा औसतन लाल रहे।
\end{solution}

\begin{solution}{exo:g11:prob:7}
$\E(G^2) = 0.5 \times 4 + 0.3 \times 0 + 0.2 \times 9 = 3.8$, तो द्वारा
शॉर्टकट फॉर्मूला
$\V(G) = 3.8 - (-0.4)^2 = 3.8 - 0.16 = 3.64$ और
$\sigma(G) \approx 1.9$ यूरो.
\end{solution}

\begin{solution}{exo:g11:prob:8}
फायदा यह है $80$ दावा घटा:
\begin{center}
\begin{tabular}{c|ccc}
$ b $ & $80$ & $-920$ & $-4920$\\
\hline
$\P(B=b)$ & $0.94$ & $0.05$ & $0.01$

लाभ)
\end{solution}

\begin{solution}{exo:g11:prob:9}
यादृच्छिक स्कोर है $+1$ साथ \omterm{def:g10:proba:distribution}{संभावना} $\frac14$ और $ x $ साथ
\omterm{def:g10:proba:distribution}{संभावना} $\frac34$:
\[
\E = \frac14 + \frac{3x}{4} = 0 \iff x = -\frac13 .
\]
जुर्माने के साथ $-\frac13$, यादृच्छिक अनुमान लगाने से औसतन कुछ नहीं मिलता।
\end{solution}

\begin{solution}{exo:g11:prob:10}
दो सिक्के: वे (HH या TT) से मेल खाते हैं \omterm{def:g10:proba:distribution}{संभावना} $\frac24 \times 2 =
\frac12$. खिलाड़ी का लाभ है $+1$ या $-1$ समान संभावनाओं के साथ:
\omterm{def:g11:prob:expectation}{अपेक्षा} $0$ दोनों खिलाड़ियों के लिए --- खेल निष्पक्ष है।

तीन सिक्के: सभी मेल खाते हैं (HHH या TTT)। \omterm{def:g10:proba:distribution}{संभावना} $\frac{2}{8} =
\frac14$. खिलाड़ी को प्राप्त होता है $2$ साथ \omterm{def:g10:proba:distribution}{संभावना} $\frac14$ और भुगतान करता है $1$
साथ \omterm{def:g10:proba:distribution}{संभावना} $\frac34$:
$\E = \frac24 - \frac34 = -\frac14$. खेल ऐन के पक्ष में है $25$ सेंट
प्रति चक्कर; यदि वह भुगतान कर दे तो यह उचित होगा $3$ ट्रिपल मैच के लिए यूरो.
\end{solution}

\begin{solution}{exo:g11:prob:11}
\emph{1.} आयोजक प्राप्त करता है $1000 \times 5 = 5000$ यूरो और भुगतान
बाहर $2000 + 5 \times 200 + 20 \times 25 = 3500$ कुल मिलाकर यूरो: लाभ
है \emph{स्थिर} $1500$ (सभी टिकटें बिक जाने के बाद कोई यादृच्छिकता नहीं)।
एक अकेले खिलाड़ी का लाभ $ G $ (पुरस्कार घटा) $5$-यूरो टिकट):
\begin{center}
\begin{tabular}{c|cccc}
$ g $ & $1995$ & $195$ & $20$ & $-5$\\
\hline\rule{0pt}{11pt}%
$\P(G=g)$ & $\dfrac{1}{1000}$ & $\dfrac{5}{1000}$ & $\dfrac{20}{1000}$
& $\dfrac{974}{1000}$
\end{tabular}
\end{center}

\emph{2.}
\[
\E(G) = \frac{1995 + 5 \times 195 + 20 \times 20 - 974 \times 5}{1000}
= \frac{1995 + 975 + 400 - 4870}{1000} = -1.5 .
\]
प्रत्येक खिलाड़ी हार की आशा करता है $1.50$ यूरो; ऊपर $1000$ खिलाड़ी जो है
$1500$ यूरो --- बिल्कुल आयोजक का लाभ। पुस्तकें संतुलन:
खिलाड़ी औसतन जो खोते हैं वही आयोजक का होता है
एकत्र करता है.
\end{solution}

\begin{solution}{pb:g11:prob:1}
\textbf{1.} $ G $ लेता है $-2, -1, 0, 1, 2, 3$, प्रत्येक के साथ
\omterm{def:g10:proba:distribution}{संभावना} $\frac16$: $\E(G) = \frac{-2 - 1 + 0 + 1 + 2 +
3}{6} = \frac12$ यूरो.

\textbf{2.} $\E(G^2) = \frac{4 + 1 + 0 + 1 + 4 + 9}{6} =
\frac{19}{6}$; $ V(G) = \frac{19}{6} - \frac14 = \frac{35}{12}
\approx 2.92$; $\sigma(G) \approx 1.71$.

\textbf{3.} $\E(2G - 1) = 2 \times \frac12 - 1 = 0$;
$ V(2G - 1) = 4\,V(G) = \frac{35}{3}$.

\textbf{4.} $\E(\text{sum}) = 7$: उचित मूल्य $7$ यूरो. पर $8$,
अपेक्षित हानि है $1$ प्रति खेल यूरो.

\textbf{5.} उचित प्रीमियम: $0.05 \times 800 = 40$ यूरो.
चार्ज $60$: अपेक्षित लाभ $20$ प्रति पॉलिसी यूरो.

\textbf{6.} विभाजन $5:3$ को पुरस्कृत करता है \emph{अतीत} --- लेकिन
पॉट उसी का होता है जो जीतता है \emph{भविष्य}, और बी हो सकता है
फिर भी जीतो. ए को सब कुछ देने से संभावित जीत निश्चित मानी जाती है ---
बी का छोटा सा मौका कुछ मायने रखता है।

\textbf{7.} खेलने के लिए राउंड: $3$; परिणाम (ए-जीत/बी-जीत प्रति
दौर): एएए, एएबी, एबीए, एबीबी, बीएए, बीएबी, बीबीए, बीबीबी। खिलाड़ी बी लेता है
तीनों को जीतकर ही सीरीज: अकेले बीबीबी। A जीतता है
श्रृंखला में $7$ का $8$ मामले: निष्पक्ष विभाजन
$64 \times \frac78 = 56$ ए के लिए पिस्तौल, $8$ बी के लिए

\textbf{8.} अतिरिक्त, निरर्थक राउंड खेलने से परिवर्तन होता है
किसी की जीत नहीं: एक बार जब ए की छठी जीत होती है, तो बाद के दौर होते हैं
\omterm{def:g11:func:function}{समारोह}. लेकिन क्षितिज को बिल्कुल ठीक कर रहा हूँ $3$ चक्कर लगाता है
सब $8$ परिणाम समान रूप से संभावित --- ईमानदार गिनती की जरूरत है
समान-लंबाई वाली कहानियाँ, और उन्हें प्रस्तुत करने में कुछ भी खर्च नहीं होता है।

\textbf{9.} पर $5$--$5$: एक निर्णायक दौर, $32$ प्रत्येक। पर
$5$--$4$: ए को मिलता है $\frac{64 + 32}{2} = 48$. पर $5$--$3$: द
अगले दौर की ओर ले जाता है $64$ (ए इसे जीतता है) या को $5$--$4$
पद के लायक $48$: ए का हिस्सा $\frac{64 + 48}{2} = 56$ ---
फ़र्मेट का $56 : 8$, पीछे की ओर चलकर पुनः प्राप्त किया गया।

\textbf{10.} एक की जरूरत है $2$, बी की जरूरत है $3$: खेल $4$ काल्पनिक
राउंड ($16$ परिणाम)। B श्रृंखला को अपने साथ ले जाता है $3$ या $4$
बी-जीत: $4 + 1 = 5$ परिणाम. विभाजित करना:
$ B $: $64 \times \frac{5}{16} = 20$; $ A $: $44$.

\textbf{11.} अनिश्चित भविष्य का उचित वर्तमान मूल्य
अदायगी इसकी है \omterm{def:g11:prob:expectation}{अपेक्षा} --- संभाव्यता-भारित औसत
क्या आ सकता है. बीमा (प्रश्न 5) और वित्त (मूल्य निर्धारण)।
अनिश्चित अदायगी) क्या यह वाक्य उद्योगों में बदल गया है;
टेबल के दूसरी ओर से कैसीनो भी ऐसा ही है।

\textbf{12.} बर्तन हमेशा कहीं न कहीं समाप्त होता है: दो हिस्से
हैं यादृच्छिक चर स्थिरांक का योग $64$, तो द्वारा
रैखिकता $\E_A + \E_B = 64$ --- उचित शेयर बर्तन को ख़त्म कर देते हैं,
हर स्कोर पर.

\textbf{13.} रैखिकता: $\E(X + Y) = 3.5 + 3.5 = 7$, टेबल-मुक्त
--- एक साथ चिपके हुए पासों के लिए भी सत्य। \omterm{def:g11:prob:variance}{झगड़ा} एक मरने का:
$\E(X^2) - 3.5^2 = \frac{91}{6} - \frac{49}{4} = \frac{35}{12}$;
स्वतंत्र पासे के लिए
$ V(X + Y) = \frac{35}{12} + \frac{35}{12} = \frac{35}{6}$.

\textbf{14.} प्रत्येक निश्चित अतिथि अपनी-अपनी टोपी वापस ले लेता है
\omterm{def:g10:proba:distribution}{संभावना} $\frac1n $ (उनकी टोपी उनमें से एक है $ n $). संक्षेप
के ऊपर $ n $ मेहमान:
$\E(X) = n \times \frac1n = 1$ --- औसतन एक भाग्यशाली अतिथि,
चाहे पार्टी में दो टोपियाँ हों या दो हजार।

\textbf{15.} $ n = 2$: असाइनमेंट: दोनों सही ($ X = 2$) या
अदला-बदली ($ X = 0$): $\E = 1$. $ n = 3$: छह कार्य देते हैं
$ X = 3, 1, 1, 1, 0, 0$: $\E = \frac{6}{6} = 1$.

\textbf{16.} गणना में \omterm{def:g10:vectors:sum}{जोड़}ा गया अपेक्षाएं की $ n $
व्यक्तिगत संकेतक (``अतिथि $ i $ उनकी टोपी मिल गई''), और
रैखिकता जुड़ती है अपेक्षाएं \emph{बिना शर्त} ---
निर्भरता किसी उत्पाद या उत्पाद को बर्बाद कर देगी \omterm{def:g11:prob:variance}{झगड़ा}, लेकिन कभी नहीं
का योग अपेक्षाएं. वह रैखिकता की महाशक्ति है: नहीं
स्वतंत्रता प्रश्न पूछे गए.

\textbf{17.} दोनों लाभ: $\E = 0.5 \times 10 - 5 = 0$ और
$0.005 \times 1000 - 5 = 0$. प्रसरण (लाभ का):
ए: $\E(G^2) = 0.5 \times 25 + 0.5 \times 25 = 25$;
बी: $0.005 \times 995^2 + 0.995 \times 25 \approx 4\,975$:
$\sigma_A = 5$, $\sigma_B \approx 70.5$. लोग बी खरीदते हैं: क्या
वे खरीद नहीं है \omterm{def:g11:prob:expectation}{अपेक्षा} (दोनों में शून्य) लेकिन
\emph{फैलाना} --- जीवन बदलने वाले अधिकार का एक छोटा सा मौका
पूंछ.

\textbf{18.} फ्लिप पर पहली पूंछ $ n $ है \omterm{def:g10:proba:distribution}{संभावना}
$2^{-n}$ और भुगतान करता है $2^n $: प्रत्येक $ n $ योगदान
$2^{-n} \times 2^n = 1$ तक \omterm{def:g11:prob:expectation}{अपेक्षा}, जो इसलिए है
$1 + 1 + 1 + \dots $: अनंत। फिर भी कोई समझदार व्यक्ति भुगतान नहीं करता
$1\,000$ एक बार खेलने के लिए: साथ \omterm{def:g10:proba:distribution}{संभावना} $\frac{15}{16}$ द
भुगतान \omterm{def:g10:functions:extrema}{अधिकतम} है $16$. \omterm{def:g11:prob:expectation}{अपेक्षा} के दीर्घकाल का सारांश प्रस्तुत करता है
\emph{repeatable} दांव; जंगली के साथ जीवन में एक बार होने वाला खेल
पूँछ की कीमत उसके हिसाब से नहीं है \omterm{def:g11:stat:mean}{अर्थ}.

\textbf{19.} अपेक्षाएं: बीमा करें $-20$; नंगे हो जाओ
$-0.01 \times 1000 = -10$. नग्न रणनीति औसतन जीतती है
--- किसी ऐसे व्यक्ति के लिए जो नुकसान सह सकता है और दांव दोहरा सकता है
कई बार. एक परिवार जो जीवित नहीं रह सकता $-1\,000$ नहीं है
औसत व्यवसाय में: यह भुगतान करता है $10$ हटाने के लिए अतिरिक्त
\omterm{def:g11:prob:variance}{झगड़ा}. बीमाकर्ताओं का औसत; परिवार निश्चितता खरीदते हैं।

\textbf{20.} दो अधीर के बीच एक बर्तन को विभाजित करते हुए पैदा हुआ
जुआरी; रैखिकता द्वारा सशक्त, जो बिना आशाओं का योग है
निर्भरता की अनुमति मांगना; जोखिम के प्रति अंध, जो \omterm{def:g11:prob:variance}{झगड़ा}
--- लॉटरी, पीटर्सबर्ग खेल, परिवार की नीति
--- इसके बजाय मापना चाहिए; और अगले वर्ष के कानून द्वारा ताज पहनाया गया
बड़ी संख्याएँ, जो बताती हैं कि हमेशा औसत क्यों होते हैं,
अंततः, एकत्र करें।
\end{solution}
