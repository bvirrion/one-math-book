\section*{अध्याय \ref{ch:g11:stat} --- वर्णनात्मक आँकड़े}

\begin{solution}{exo:g11:stat:1}
$n = 8$, \omterm{def:g10:vectors:sum}{जोड़} $71$: \omterm{def:g11:stat:mean}{अर्थ} $\bar x = \frac{71}{8} = 8.875$. मंझला: \omterm{prop:g10:coordgeom:midpoint}{मध्य}
की $4$ वें और $5$ वें मान, $\frac{8 + 9}{2} = 8.5$. चतुर्थक: के एक चौथाई $8$ है $2$, इसलिए $ Q_1$ है $2$ दूसरा मूल्य, $7$; तीन चौथाई है
$6$, इसलिए $ Q_3$ है $6$ वें मान, $10$. \omterm{def:g11:stat:quartiles}{अंतरचतुर्थक सीमा}:
$10 - 7 = 3$.
\end{solution}

\begin{solution}{exo:g11:stat:2}
\omterm{def:g10:vectors:sum}{जोड़} $40$, इसलिए $\bar x = 5$. वर्गों का योग:
$4 + 16 + 16 + 16 + 25 + 25 + 49 + 81 = 232$. शॉर्टकट फ़ॉर्मूले से,
\[
V = \frac{232}{8} - 5^2 = 29 - 25 = 4,
\qquad
\sigma = 2 .
\]
\end{solution}

\begin{solution}{exo:g11:stat:3}
\omterm{def:g11:stat:mean}{अर्थ}:
\[
\bar x = \frac{1 \times 6 + 2 \times 9 + 3 \times 8 + 4 \times 10
+ 5 \times 9 + 6 \times 8}{50}
= \frac{181}{50} = 3.62 .
\]
मंझला: संचयी आवृत्तियाँ हैं $6, 15, 23, 33, \dots $, इतना
$25$ वें और $26$ वें क्रमबद्ध परिणाम दोनों हैं $4$: द \omterm{def:g11:stat:median}{MEDIAN} है $4$.
\end{solution}

\begin{solution}{exo:g11:stat:4}
का कुल $15$ मार्क्स है $15 \times 11.2 = 168$; के साथ
नवागंतुक, $\frac{168 + 18}{16} = \frac{186}{16} = 11.625$.
\end{solution}

\begin{solution}{exo:g11:stat:5}
$\frac{6 + 8 + x + 12 + 14}{5} = 10$ देता है $40 + x = 50$, इसलिए $ x = 10$.
के विचलन $6, 8, 10, 12, 14$ से \omterm{def:g11:stat:mean}{अर्थ} $10$ हैं
$-4, -2, 0, 2, 4$, वर्गों के योग के साथ $40$:
$ V = \frac{40}{5} = 8$ और $\sigma = 2\sqrt2 \approx 2.83$.
\end{solution}

\begin{solution}{exo:g11:stat:6}
द्वारा \cref{prop:g11:stat:affine} साथ $ a = \frac12$, $ b = 4$: नया \omterm{def:g11:stat:mean}{अर्थ}
$\frac{12.4}{2} + 4 = 10.2$; नया \omterm{def:g11:stat:variance}{मानक विचलन}
$\frac{2.5}{2} = 1.25$.
\end{solution}

\begin{solution}{exo:g11:stat:7}
जमा हुआ \omterm{def:g11:stat:mean}{अर्थ} है \emph{भारित} औसत
\[
\frac{20 \times 10 + 30 \times 15}{50} = \frac{650}{50} = 13,
\]
नहीं $12.5$: समूह बी अधिक मूल्यों का योगदान देता है, इसलिए यह पूल को खींचता है \omterm{def:g11:stat:mean}{अर्थ}
अपनी ओर.
\end{solution}

\begin{solution}{exo:g11:stat:8}
दोनों लाइनें औसतन लगभग लक्ष्य पर हैं ($1.002$ और $1.001$ किग्रा), लेकिन
पंक्ति 2 \omterm{def:g11:stat:variance}{मानक विचलन} तीन गुना बड़ा है: इसके बैग बहुत भिन्न होते हैं
अधिक, कम और अधिक भरी हुई थैलियों का उत्पादन कहीं अधिक स्पष्ट रूप से होता है। पंक्ति 2
सबसे पहले निपटा जाना चाहिए --- स्थिरता, न कि केवल \omterm{def:g11:stat:mean}{अर्थ}, क्या है
ठीक करने की जरूरत है.
\end{solution}

\begin{solution}{exo:g11:stat:9}
$\bar x = \frac{70}{10} = 7$;
$ V = \frac{540}{10} - 7^2 = 54 - 49 = 5$;
$\sigma = \sqrt5 \approx 2.24$.
\end{solution}

\begin{solution}{exo:g11:stat:10}
\emph{1.} साथ $23$: \omterm{def:g10:vectors:sum}{जोड़} $50$, \omterm{def:g11:stat:mean}{अर्थ} $5$; \omterm{def:g11:stat:median}{MEDIAN} = \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $5$ वें और
$6$ वें मान $= 3$. बिना $23$: \omterm{def:g10:vectors:sum}{जोड़} $27$, \omterm{def:g11:stat:mean}{अर्थ} $3$; \omterm{def:g11:stat:median}{MEDIAN} (9 में से 5वां
मान) $= 3$.

\emph{2.} साथ $23$ ($ n = 10$): $ Q_1$ है $3$ तीसरा मान ($2$), $ Q_3$ 
$8$ वां ($4$): \omterm{def:g11:stat:quartiles}{अंतरचतुर्थक सीमा} $2$. बिना ($ n = 9$): $ Q_1$ है
$3$ तीसरा मान ($2$), $ Q_3$  $7$ वां ($4$): फिर भी $2$.

\emph{3.} एकल बाह्यता खींचती है \omterm{def:g11:stat:mean}{अर्थ} से $3$ तक $5$ लेकिन छोड़ देता है
द \omterm{def:g11:stat:median}{MEDIAN} और यह \omterm{def:g11:stat:quartiles}{अंतरचतुर्थक सीमा} अपरिवर्तित: वे दो हैं
\emph{मज़बूत}. जब किसी डेटासेट में असामान्य मान हो सकते हैं, तो रिपोर्ट करें
\omterm{def:g11:stat:median}{MEDIAN} और \omterm{def:g11:stat:quartiles}{अंतरचतुर्थक सीमा} इसके बजाय \omterm{def:g11:stat:mean}{अर्थ} और \omterm{def:g11:stat:variance}{मानक
विचलन}.
\end{solution}

\begin{solution}{pb:g11:stat:1}
\textbf{1.} \omterm{def:g11:stat:mean}{अर्थ} $\bar x = \frac{25}{5} = 5$. विचलन
चुकता: $1, 9, 1, 0, 9$: \omterm{def:g11:stat:variance}{झगड़ा} $\frac{20}{5} = 4$, इसलिए
$\sigma = 2$.

\textbf{2.} \omterm{def:g11:stat:mean}{अर्थ} वर्गों का:
$\frac{16 + 64 + 36 + 25 + 4}{5} = 29$; ऋण
$\bar x^2 = 25$: \omterm{def:g11:stat:variance}{झगड़ा} $4$. जवाब वही, काम आधा।

\textbf{3.} \omterm{def:g11:stat:mean}{अर्थ} $= \frac{3 \times 8 + 5 \times 10 + 2 \times
14}{10} = 10.2$; \omterm{def:g11:stat:variance}{झगड़ा} $= \frac{3 \times 64 + 5 \times 100 +
2 \times 196}{10} - 10.2^2 = 108.4 - 104.04 = 4.36$;
$\sigma \approx 2.09$.

\textbf{4.} \omterm{def:g11:stat:mean}{अर्थ}: $1.8 \times 20 + 32 = 68\,^\circ $ एफ;
$\sigma $: $1.8 \times 3 = 5.4\,^\circ $ एफ --- बदलाव $+32$
कुछ भी नहीं फैलाता.

\textbf{5.} \omterm{def:g10:vectors:sum}{जोड़}ा जा रहा है $ b $: हर विचलन $ x_i + b - (\bar x + b)
= x_i - \bar x $ अपरिवर्तित है, इसलिए \omterm{def:g11:stat:variance}{झगड़ा} है। गुणा करना
द्वारा $ a $: विचलनों को गुणा किया जाता है $ a $, उनके वर्ग द्वारा
$ a^2$, द \omterm{def:g11:stat:variance}{झगड़ा} द्वारा $ a^2$, और $\sigma $ द्वारा
$\sqrt{a^2} = \abs a $.

\textbf{6.} $ z_A = \frac{15 - 12}{2} = 1.5$;
$ z_B = \frac{13 - 8}{4} = 1.25$. ऐलिस उससे भी ऊपर खड़ी है
भीड़: परीक्षा जितनी कठिन होगी $15$ आसान परीक्षा को मात देता है $13$.

\textbf{7.} $ z $-अंक: $-0.5,\ 1.5,\ 0.5,\ 0,\ -1.5$: \omterm{def:g11:stat:mean}{अर्थ}
$0$, \omterm{def:g11:stat:variance}{मानक विचलन} $1$. सामान्य तौर पर, मानकीकरण
घटाना $\bar x $ (बनाना \omterm{def:g11:stat:mean}{अर्थ} $0$, $\sigma $ अछूता)
फिर विभाजित करता है $\sigma $ (नया बनाना $\sigma $ के बराबर
$1$): प्रश्न 5 दो बार।

\textbf{8.} $ z = 2$: $100 + 2 \times 15 = 130$. और
$ z(76) = \frac{76 - 100}{15} = -1.6$.

\textbf{9.} बच्चा है $2.5$ \omterm{def:g11:stat:variance}{मानक विचलन} के नीचे
\omterm{def:g11:stat:mean}{अर्थ} वज़न \emph{बच्चों की उम्र इसकी है} --- के लिए चिंताजनक रूप से प्रकाश
इसका समूह. कच्चे किलोग्राम की तुलना सभी उम्र में नहीं की जा सकती
बच्चों का वजन बढ़ता है); z-स्कोर प्रत्येक बच्चे को उसके मुकाबले मापें
अपना आयु समूह, सभी उम्र के लिए एक शासक।

\textbf{10.} \omterm{def:g11:stat:mean}{अर्थ} $5$; \omterm{def:g11:stat:mean}{अर्थ} वर्गों का $\frac{622}{10} = 62.2$,
\omterm{def:g11:stat:variance}{झगड़ा} $37.2$, $\sigma \approx 6.10$;
$ z(23) = \frac{18}{6.10} \approx 2.95$: ठीक अलार्म के नीचे
दहलीज --- परीक्षण मुश्किल से झपकाता है।

\textbf{11.} बिना $23$: \omterm{def:g11:stat:mean}{अर्थ} $3$, \omterm{def:g11:stat:variance}{झगड़ा}
$\frac{93}{9} - 9 \approx 1.33$, $\sigma \approx 1.15$;
स्वच्छ शासक के विरुद्ध, $23$ स्कोर
$ z \approx \frac{20}{1.15} \approx 17$: एक राक्षस. बाहरी
दोनों को फुलाया \omterm{def:g11:stat:mean}{अर्थ} और $\sigma $, खुद को छुपाना ---
यही कारण है \cref{exo:g11:stat:10} मजबूत की सिफ़ारिश करता है
सारांश (\omterm{def:g11:stat:median}{MEDIAN}, IQR) जब संदूषण का खतरा हो।

\textbf{12.} में \omterm{def:g11:stat:variance}{झगड़ा} \omterm{def:g10:vectors:sum}{जोड़}
$\sigma^2 = \frac1N \sum (x_i - \bar x)^2$, केवल रखें
के साथ शर्तें $\abs{x_i - \bar x} \geq k\sigma $: वहाँ हैं $ pN $
उनमें से प्रत्येक, कम से कम $(k\sigma)^2$, इसलिए
$\sigma^2 \geq \frac{1}{N} \cdot pN \cdot k^2\sigma^2 =
p\,k^2\sigma^2$: से विभाजित करें $ k^2 \sigma^2$ (शून्येतर प्रसार):
$ p \leq \frac{1}{k^2}$.

\textbf{13.} आगे $2\sigma $: अधिक से अधिक $\frac14 = 25\,\%$ का
कोई भी डेटासेट; आगे $3\sigma $: अधिक से अधिक
$\frac19 \approx 11\,\%$. प्रश्न 1 का डेटा: परे
$2\sigma = 4$ से \omterm{def:g11:stat:mean}{अर्थ} $5$ मतलब $\leq 1$ या $\geq 9$:
पांच मूल्यों में से कोई नहीं --- $0\,\% \leq 25\,\%$, वादे के अनुसार।

\textbf{14.} बाहर $\intcc{49.7}{50.3}$ मतलब परे
$3\sigma $: अधिक से अधिक $\frac19$ उत्पादन का, \emph{जो कुछ भी}
वितरण का आकार --- चेबीशेव कुछ नहीं बल्कि एक पूछता है
\omterm{def:g11:stat:mean}{अर्थ} और ए $\sigma $. बेल के आकार का उत्पादन दूर तक \omterm{pb:g11:quad:1}{केंद्र}ित होता है
बेहतर (के बारे में) $0.3\,\%$ आगे $3\sigma $): सामान्य कानून,
ग्रेड 12.

\textbf{15.} $-19\,\%$ बैठता $24$ के नीचे बिंदु \omterm{def:g11:stat:mean}{अर्थ} $5$.
फंड बी: $24 = 2\sigma_B $: चेबीशेव तक की अनुमति देता है
$\frac14$ इतने बुरे साल। फंड ए: $24 = 12\sigma_A $: अधिक से अधिक $\frac{1}{144}$. वही औसत, अतुलनीय खतरे ---
$\sigma $ वित्त यही कहता है \emph{जोखिम} (या अस्थिरता).

\textbf{16.} चारों ओर \omterm{def:g10:algebra:expand}{विस्तार} करें $\bar x $:
$(x_i - c)^2 = \left((x_i - \bar x) + (\bar x - c)\right)^2
= (x_i - \bar x)^2 + 2(\bar x - c)(x_i - \bar x) +
(\bar x - c)^2$. औसत: \omterm{prop:g10:coordgeom:midpoint}{मध्य} पद समाप्त हो जाता है (विचलन)।
से \omterm{def:g11:stat:mean}{अर्थ} योग शून्य तक), जा रहा है
$ f(c) = \sigma^2 + (\bar x - c)^2$ --- बिल्कुल \omterm{def:g10:functions:extrema}{न्यूनतम}
$ c = \bar x $, साथ \omterm{def:g10:functions:extrema}{न्यूनतम} $\sigma^2$. \omterm{def:g11:stat:mean}{अर्थ} है
न्यूनतम-वर्ग \omterm{pb:g11:quad:1}{केंद्र}, के रूप में \omterm{def:g11:stat:median}{MEDIAN} न्यूनतम-निरपेक्ष है
\omterm{pb:g11:quad:1}{केंद्र} (\cref{pb:g10:stats:1}).

\textbf{17.} सीधे: से विचलन $6$ हैं
$-2, 2, 0, -1, -4$, वर्ग $4, 4, 0, 1, 16$:
$ f(6) = \frac{25}{5} = 5$. सूत्र:
$\sigma^2 + (5 - 6)^2 = 4 + 1 = 5$. समझौता।

\textbf{18.} दोनों के पास है \omterm{def:g11:stat:mean}{अर्थ} $7$; प्रसरण:
$ P $: $\frac{16 + 4 + 0 + 4 + 16}{5} = 8$;
$ Q $: $\frac{36 + 0 + 0 + 0 + 36}{5} = 14.4$: विभिन्न फैलाव
एक के पीछे \omterm{def:g11:stat:mean}{अर्थ}. के लिए $\{7 - a, 7, 7, 7, 7 + a\}$: \omterm{def:g11:stat:variance}{झगड़ा}
$\frac{2a^2}{5} = 8$ देता है $ a = \sqrt{20} \approx 4.47$: द
डेटासेट $\{2.53, 7, 7, 7, 11.47\}$ सटीक रखता है \omterm{def:g11:stat:mean}{अर्थ} और
\omterm{def:g11:stat:variance}{झगड़ा} का $ P $ बिल्कुल अलग प्रोफ़ाइल के साथ ---
सारांश संपीड़ित होता है, और संपीड़न खो जाता है।

\textbf{19.} $\sigma^2 = 0$ मतलब वर्गों का योग
$(x_i - \bar x)^2$ गायब हो जाता है; वर्ग हैं $\geq 0$, तो एक शून्य
योग प्रत्येक पद को शून्य होने के लिए बाध्य करता है: $ x_i = \bar x $ सभी के लिए $ i $
--- सभी मान समान.

\textbf{20.} \omterm{pb:g11:quad:1}{केंद्र} कहते हैं कि डेटा कहाँ रहता है; स्प्रेड कहते हैं कैसे
शिथिलता से; \omterm{pb:g11:stat:1}{z के स्कोर} सभी डेटासेट पर एक रूलर रखता है,
परीक्षा, उम्र और मुद्राएँ; चेबीशेव धर्म परिवर्तन करता है $\sigma$ में
बिना शर्त गारंटी; और प्रश्न 18 हमें ईमानदार रखता है
--- किसी भी मुट्ठी भर संख्याओं में कभी भी पूरी कहानी नहीं होती। अगला
वर्ष घंटी वक्र शासक को सटीक में तेज कर देता है
सम्भावनाएँ
\end{solution}
