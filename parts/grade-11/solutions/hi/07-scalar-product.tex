\begin{solution}{exo:g11:scal:2}

 वैक्टर $\vect{AB}$ और $\vect{BC}$ का एक कोण बनाएं

\begin{solution}{exo:g11:scal:3}
$\vec u \cdot \vec v = 3(t - 8) + t = 4t - 24 = 0$ देता है $ t = 6$.

$\vec u - \vec v = (t - 4,\ 0)$, इसलिए
$\vec u \cdot (\vec u - \vec v) = t(t-4) + 0 = t^2 - 4t $, जो गायब हो जाता है
के लिए $ t = 0$ या $ t = 4$.
\end{solution}

\begin{solution}{exo:g11:scal:4}

\begin{solution}{exo:g11:scal:5}
$ a^2 = 16 + 36 - 2 \times 4 \times 6 \times \cos\frac{2\pi}{3}
= 52 - 48 \times \left(-\frac12\right) = 76$, इसलिए $ a = \sqrt{76} =
2\sqrt{19} \approx 8.7$.

दूसरे त्रिभुज के लिए, का नियम कोसाइन कोण के लिए हल दिया गया है
\[
\cos\widehat A = \frac{b^2 + c^2 - a^2}{2bc}
= \frac{25 + 16 - 49}{2 \times 5 \times 4} = -\frac{8}{40} = -\frac15,
\]
नकारात्मक: $\widehat A $ कुंठित है.
\end{solution}

\begin{solution}{exo:g11:scal:6}
$\vect{AB}\,(4, -2)$ और $\vect{AC}\,(2, 4)$:
$\vect{AB}\cdot\vect{AC} = 8 - 8 = 0$, तो कोण पर $ A $ सही है।
पक्ष: $ AB = \sqrt{16 + 4} = \sqrt{20}$, $ AC = \sqrt{4 + 16} =
\sqrt{20}$, और $\vect{BC}\,(-2, 6)$ देता है $ BC = \sqrt{40}$. स्थिरता:
$ BC^2 = 40 = 20 + 20 = AB^2 + AC^2$, का कानून कोसाइन साथ
$\cos\widehat A = 0$ (पाइथागोरस)।
\end{solution}

\begin{solution}{exo:g11:scal:7}

दूसरे वृत्त के लिए, वर्ग पूरे करें:

\end{solution}

\begin{solution}{exo:g11:scal:8}
$ M(x,y)$ सर्कल पर है जब $\vect{MA}\cdot\vect{MB} = 0$ साथ
$\vect{MA}\,(-3 - x,\ -y)$ और $\vect{MB}\,(-x,\ 2 - y)$:
\[
(-3 - x)(-x) + (-y)(2 - y) = x^2 + 3x + y^2 - 2y = 0 .
\]
उत्पत्ति के लिए: $0 + 0 + 0 - 0 = 0$, इसलिए $ O $ वृत्त पर स्थित है ---
ज्यामितीय रूप से, $\vect{OA}\,(-3,0)$ और $\vect{OB}\,(0,2)$ हैं \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल},
इसलिए $ O $ देखता है $[AB]$ एक समकोण के नीचे. (वर्गों को पूरा करना:
\omterm{pb:g11:quad:1}{केंद्र} $\left(-\frac32, 1\right)$, त्रि\omterm{def:g11:trigo:cossin}{ज्या} $\frac{\sqrt{13}}{2}$.)
\end{solution}

\begin{solution}{exo:g11:scal:9}
के माध्यम से लंबवत $ P $ के पास है \omterm{def:g11:vect:direction}{दिशा सदिश} का $ d $, \omterm{def:g10:stats:mean}{अर्थ}ात्

$ P(1,3)$: $-1 + 6 + c = 0$, $ c = -5$, इसलिए $ x - 2y + 5 = 0$.

\omterm{def:g10:numbers:interunion}{चौराहा} साथ $ d $: से $2x + y = 7$, $ y = 7 - 2x $; प्रतिस्थापित करना,
$ x - 2(7 - 2x) + 5 = 5x - 9 = 0$, इसलिए $ x = \frac95$, $ y = \frac{17}5$:

\]
\end{solution}

\begin{solution}{exo:g11:scal:10}
\omterm{def:g10:vectors:sum}{जोड़}ा जा रहा है
\[
\norm{\vec u + \vec v}^2 = \norm{\vec u}^2 + 2\vec u\cdot\vec v +
\norm{\vec v}^2
\quad\text{and}\quad
\norm{\vec u - \vec v}^2 = \norm{\vec u}^2 - 2\vec u\cdot\vec v +
\norm{\vec v}^2,
\]
क्रॉस शर्तें रद्द करें और
$\norm{\vec u + \vec v}^2 + \norm{\vec u - \vec v}^2 = 2\norm{\vec u}^2 +
2\norm{\vec v}^2$. भुजाओं वाले समांतर चतुर्भुज में $\vec u $ और $\vec v $,
विकर्ण हैं $\vec u + \vec v $ और $\vec u - \vec v $: का योग
दो विकर्णों का वर्ग चारों के वर्गों के योग के बराबर होता है
पक्ष.
\end{solution}

\begin{solution}{exo:g11:scal:11}
\emph{1.} तब से $ I $ है \omterm{prop:g10:coordgeom:midpoint}{मध्य}, $\vect{IB} = -\vect{IA}$ और
$ IA = 2$. तब
\[
\vect{MA}\cdot\vect{MB}
= (\vect{MI} + \vect{IA})\cdot(\vect{MI} - \vect{IA})
= \norm{\vect{MI}}^2 - \norm{\vect{IA}}^2
= MI^2 - 4 .
\]

\emph{2.} स्थिति $\vect{MA}\cdot\vect{MB} = 12$ बन जाता है
$ MI^2 = 16$, यानी\ $ MI = 4$: समुच्चय \omterm{pb:g11:quad:1}{केंद्र} का वृत्त है $ I $ और
त्रि\omterm{def:g11:trigo:cossin}{ज्या} $4$.
\end{solution}

\begin{solution}{pb:g11:scal:1}
\textbf{1.} $\vec u \cdot \vec v = -3 + 8 = 5$;
$\norm{\vec u} = 5$, $\norm{\vec v} = \sqrt5$;
$\cos\theta = \frac{5}{5\sqrt5} = \frac{1}{\sqrt5}$:
$\theta \approx 63.4^\circ $.

\textbf{2.} $6 - 4k = 0$: $ k = \frac32$.

\textbf{3.} $ W = 50 \times 10 \times \cos 60^\circ = 250$
जूल: गति के साथ केवल घटक काम करता है।

\textbf{4.} $ c^2 = 25 + 49 - 2 \times 35 \times \frac12 = 39$:
$ c = \sqrt{39}$.

\textbf{5.} $\cos\gamma = \frac{16 + 36 - 64}{2 \times 4
\times 6} = -\frac14$: ऋणात्मक, कोण अधिक है:
$\gamma \approx 104.5^\circ $.

\textbf{6.} \omterm{def:g10:coordgeom:system}{COORDINATES}: $\vec u \cdot \vec v = \cos a \cos b +
\sin a \sin b $. ज्यामिति: दो इकाई वैक्टर,
$\vec u \cdot \vec v = 1 \times 1 \times \cos(a - b)$.
एक ही संख्या की दो गणनाओं को बराबर करना:
$\cos(a - b) = \cos a \cos b + \sin a \sin b $.

\textbf{7.} $\cos(a + b) = \cos(a - (-b)) = \cos a \cos(-b) +
\sin a \sin(-b) = \cos a \cos b - \sin a \sin b $.

\textbf{8.} $\sin(a + b) = \cos\left(\frac\pi2 - a - b\right)
= \cos\left(\left(\frac\pi2 - a\right) - b\right)
= \cos\left(\frac\pi2 - a\right)\cos b +
\sin\left(\frac\pi2 - a\right)\sin b
= \sin a \cos b + \cos a \sin b $.

\textbf{9.} $\cos 15^\circ = \cos 45^\circ \cos 30^\circ +
\sin 45^\circ \sin 30^\circ = \frac{\sqrt2}{2} \cdot
\frac{\sqrt3}{2} + \frac{\sqrt2}{2} \cdot \frac12 =
\frac{\sqrt6 + \sqrt2}{4} \approx 0.966$. और
$\sin 75^\circ = \cos 15^\circ $: समान मूल्य.

\textbf{10.} $ b = a $ प्रश्न 7 और 8 में:
$\cos 2a = \cos^2 a - \sin^2 a = 2\cos^2 a - 1 = 1 - 2\sin^2 a $
(का उपयोग करते हुए $\cos^2 + \sin^2 = 1$), और
$\sin 2a = 2 \sin a \cos a $.

\textbf{11.} $\sin^2 15^\circ = \frac{1 - \cos 30^\circ}{2}
= \frac{2 - \sqrt3}{4}$, इसलिए
$\sin 15^\circ = \frac{\sqrt{2 - \sqrt3}}{2}$. बराबरी
$\frac{\sqrt6 - \sqrt2}{4}$:
$\frac{6 - 2\sqrt{12} + 2}{16} = \frac{8 - 4\sqrt3}{16} =
\frac{2 - \sqrt3}{4}$: दो पोशाकों में एक ही नंबर।

\textbf{12.} $\vect{AP}$ साथ ही एक भाग में विघटित हो जाता है $ d $
(अदृश्य के लिए $\vec n $) और एक भाग लंबवत $ d $, किसका
लंबाई मांगी गई दूरी है; इकाई के साथ बिंदु सामान्य है
$\frac{\vec n}{\norm{\vec n}}$ पहले भाग को मारता है और
दूसरा मापता है. साथ $ A $ पर $ d $:
$\vect{AP} \cdot \vec n = a x_0 + b y_0 - (a x_A + b y_A) =
a x_0 + b y_0 + c $, सूत्र कहां से। उदाहरण:
$\frac{\abs{15 + 24 - 12}}{5} = \frac{27}{5} = 5.4$.

\textbf{13.} RADIUS $= 5.4$:
$(x - 5)^2 + (y - 6)^2 = 29.16$.

\textbf{14.} दूरी $= \frac{\abs{-12}}{5} = 2.4$. पैर: द
मूल बिन्दु से होकर लम्बवत है $(3t, 4t)$; लाइन पर:
$9t + 16t = 12$, $ t = \frac{12}{25}$: पैर
$\left(\frac{36}{25}, \frac{48}{25}\right)$, दूरी पर
$5t = \frac{60}{25} = 2.4$. सूत्र और ईमानदार
गणना सहमत.

\textbf{15.} आधार $ AB = 6$, ऊंचाई $=$ से दूरी $ C $ को
$(AB)$ (धुरी $ y = 0$): $5$; क्षेत्र $15$. रेखा $(AC)$:
$5x - 2y = 0$, और
$\operatorname{dist}(B, (AC)) = \frac{\abs{30}}{\sqrt{29}}$;
तब $\frac12 \times AC \times \frac{30}{\sqrt{29}} =
\frac12 \times \sqrt{29} \times \frac{30}{\sqrt{29}} = 15$:
समान क्षेत्र, या तो ऊंचाई।

\textbf{16.} $\operatorname{dist} = \frac{\abs{2 + 1 - 6}}
{\sqrt2} = \frac{3}{\sqrt2} \approx 2.12 > 2$: लाइन गुजरती है
वृत्त से मुक्त --- असंयुक्त।

\textbf{17.} डालना $ I $:
$\vect{MA} \cdot \vect{MB} = (\vect{MI} + \vect{IA}) \cdot
(\vect{MI} + \vect{IB}) = MI^2 + \vect{MI} \cdot (\vect{IA} +
\vect{IB}) + \vect{IA} \cdot \vect{IB}$. \omterm{prop:g10:coordgeom:midpoint}{मध्य} पद मर जाता है
($\vect{IA} + \vect{IB} = \vec 0$), और
$\vect{IA} \cdot \vect{IB} = -IA^2 = -\frac{AB^2}{4}$
(विलोम वैक्टर लम्बाई का $\frac{AB}2$).

\textbf{18.} $\vect{MA} \cdot \vect{MB} = 0 \iff MI^2 =
\frac{AB^2}{4} \iff MI = \frac{AB}{2}$: \omterm{pb:g11:quad:1}{केंद्र} का वृत्त
$ I $ और त्रि\omterm{def:g11:trigo:cossin}{ज्या} $\frac{AB}2$ --- व्यास वाला वृत्त
$[AB]$.

\textbf{19.} \omterm{def:g10:algebra:expand}{विस्तार} प्रत्येक \omterm{def:g10:vectors:sum}{जोड़}ी से $ H $ ($\vect{BC} =
\vect{HC} - \vect{HB}$, वगैरह।):
\[
\vect{HA} \cdot \vect{HC} - \vect{HA} \cdot \vect{HB}
+ \vect{HB} \cdot \vect{HA} - \vect{HB} \cdot \vect{HC}
+ \vect{HC} \cdot \vect{HB} - \vect{HC} \cdot \vect{HA} = 0 :
\]
सब कुछ \omterm{def:g10:vectors:sum}{जोड़}े में रद्द हो जाता है---पहचान कायम रहती है
\emph{कोई} चार अंक. हमारे लिए $ H $, पहले दो बिंदु
उत्पाद परिकल्पना से गायब हो जाते हैं, इसलिए
$\vect{HC} \cdot \vect{AB} = 0$: $ H $ से ऊँचाई पर स्थित है
$ C$. तीन ऊँचाइयाँ, एक बिंदु।

\textbf{20.} \omterm{def:g10:vectors:sum}{जोड़} सूत्र एक बिंदु की गणना से आए
के साथ उत्पाद \emph{कोआर्डिनेट} चेहरा और
\emph{मानदंड-और-कोण} चेहरा; से दूरी सूत्र
\emph{अनुमान} चेहरा; \omterm{pb:g11:scal:1}{ऑर्थोसेंटर} से
\emph{द्विरेखीयता} (चेसल्स-एंड-एक्सपैंड) चेहरा। वही गणना करें
मात्रा दो बार, बराबर करें, और एक प्रमेय निकल जाता है --- अदिश
उत्पादका पूरा व्यापार.
\end{solution}
