\section*{Chapter \ref{ch:g11:seq} --- Sequences: A First Course}

\begin{solution}{exo:g11:seq:1}
$u_n = \frac{n}{n+1}$: $u_1 = \frac12$, $u_2 = \frac23$,
$u_3 = \frac34$.

$u_0 = 5$, $u_{n+1} = 3u_n - 2$: $u_1 = 13$, $u_2 = 37$, $u_3 = 109$.

$u_n = (-1)^n n$: $u_1 = -1$, $u_2 = 2$, $u_3 = -3$.
\end{solution}

\begin{solution}{exo:g11:seq:2}
$u_{10} = 7 + 10 \times (-3) = -23$ and $u_{25} = 7 - 75 = -68$.

For $(v_n)$: $v_8 = v_3 + 5d$ gives $26 = 11 + 5d$, so $d = 3$; then
$v_0 = v_3 - 3d = 11 - 9 = 2$.
\end{solution}

\begin{solution}{exo:g11:seq:3}
$u_8 = 5 \times 2^8 = 1280$.

For $(v_n)$: $v_4 = v_2\, q^2$ gives $48 = 12 q^2$, so $q^2 = 4$ and
$q = 2$ (the terms are positive). Then
$v_0 = \frac{v_2}{q^2} = \frac{12}{4} = 3$.
\end{solution}

\begin{solution}{exo:g11:seq:4}
$1 + \dots + 500 = \frac{500 \times 501}{2} = 125\,250$.

$4 + 7 + \dots + 61$ is arithmetic with $d = 3$ and
$\frac{61 - 4}{3} + 1 = 20$ terms: sum
$20 \times \frac{4 + 61}{2} = 650$.

$1 + \frac12 + \dots + \frac{1}{2^{10}}$ is geometric with $q = \frac12$
and $11$ terms:
$\frac{1 - (1/2)^{11}}{1 - 1/2} = 2\left(1 - \frac{1}{2048}\right)
= \frac{2047}{1024}$.
\end{solution}

\begin{solution}{exo:g11:seq:5}
$u_{n+1} - u_n = 4(n+1) - 1 - 4n + 1 = 4$: arithmetic with $d = 4$.

$\frac{v_{n+1}}{v_n} = \frac{2^{n+1}}{3^{n+2}} \cdot \frac{3^{n+1}}{2^n}
= \frac23$: geometric with $q = \frac23$.

$w_0 = 0$, $w_1 = 2$, $w_2 = 6$: the differences $2$ and $4$ differ, so
not arithmetic; $\frac{w_1}{w_0}$ is not even defined, and the ratios
$\frac{w_2}{w_1} = 3 \neq \frac{w_3}{w_2} = 2$: neither.
\end{solution}

\begin{solution}{exo:g11:seq:6}
The row sizes are arithmetic: first term $16$, difference $2$. The last
(20th) row has $16 + 19 \times 2 = 54$ seats. The total is
$20 \times \frac{16 + 54}{2} = 700$ seats.
\end{solution}

\begin{solution}{exo:g11:seq:7}
After $n$ hours the population is $500 \times 2^n$. At 8 p.m., $n = 8$:
$500 \times 256 = 128\,000$ bacteria. We need $500 \times 2^n >
10^6$, i.e.\ $2^n > 2000$: since $2^{10} = 1024$ and $2^{11} = 2048$, the
population first exceeds one million after $11$ full hours, at 11 p.m.
\end{solution}

\begin{solution}{exo:g11:seq:8}
$c_2 = 1.002 \times 100 + 100 = 200.20$ and
$c_3 = 1.002 \times 200.20 + 100 \approx 300.60$. The differences
$c_2 - c_1 = 100.20$ and $c_3 - c_2 \approx 100.40$ are not equal, so
$(c_n)$ is not arithmetic; the ratios $\frac{c_2}{c_1} = 2.002$ and
$\frac{c_3}{c_2} \approx 1.50$ are not equal either, so it is not
geometric. (Mixed ``multiply then add'' recursions like this one are
solved by the auxiliary-sequence trick of \cref{exo:g11:seq:11}.)
\end{solution}

\begin{solution}{exo:g11:seq:9}
$u_{n+1} - u_n = (n+1)^2 - 8(n+1) - n^2 + 8n = 2n - 7$: negative for
$n \leq 3$, positive for $n \geq 4$. So $(u_n)$ decreases down to
$u_4 = 16 - 32 = -16$, then increases: it is not monotonic.

$(v_n)$ has positive terms and
\[
\frac{v_{n+1}}{v_n} = \frac{3^{n+1}}{(n+1)!} \cdot \frac{n!}{3^n}
= \frac{3}{n+1},
\]
which is $> 1$ for $n \leq 1$, $= 1$ for $n = 2$, and $< 1$ for
$n \geq 3$: the sequence increases up to $v_2 = v_3 = \frac92$, then
decreases.
\end{solution}

\begin{solution}{exo:g11:seq:10}
The first $n$ terms are $u_0, \dots, u_{n-1}$, with $u_0 = 3$ and
$u_{n-1} = 3 + 4(n-1) = 4n - 1$. Their sum is
\[
n \times \frac{3 + (4n-1)}{2} = n(2n + 1) = 903,
\]
so $2n^2 + n - 903 = 0$. Here $\Delta = 1 + 4 \times 2 \times 903 =
7225 = 85^2$, and $n = \frac{-1 + 85}{4} = 21$ (the negative root is
rejected). Check: $21 \times 43 = 903$.
\end{solution}

\begin{solution}{exo:g11:seq:11}
\emph{1.} $u_1 = 5$, $u_2 = 9$, $u_3 = 17$: each term is one more than
$4, 8, 16$, suggesting $u_n = 2^{n+1} + 1$.

\emph{2.} With $v_n = u_n - 1$:
\[
v_{n+1} = u_{n+1} - 1 = 2u_n - 1 - 1 = 2(u_n - 1) = 2v_n,
\]
so $(v_n)$ is geometric with ratio $2$ and first term
$v_0 = u_0 - 1 = 2$.

\emph{3.} Hence $v_n = 2 \times 2^n = 2^{n+1}$ and
$u_n = v_n + 1 = 2^{n+1} + 1$, confirming the conjecture. (The number $1$
subtracted in $v_n$ is the fixed point of $x \mapsto 2x - 1$; the same
idea reappears for $u_{n+1} = au_n + b$ in \cref{ch:g12:seq}.)
\end{solution}
