\chapter{Funciones y ecuaciones cuadráticas}\label{ch:g11:quad}

\omterm{def:g11:quad:quadratic}{Funciones cuadráticas} son los \omterm{def:g10:functions:function}{funciones} más simples después de los lineales, y
los primeros cuyos \omterm{def:g10:functions:graph}{graficos} son genuinamente curvos. Este capítulo desarrolla la
caja de herramientas completa para ellos: el \omterm{thm:g11:quad:canonical}{forma canónica}, el \omterm{def:g11:quad:discriminant}{discriminante}, el letrero
de una expresión cuadrática y la geometría del \omterm{def:g11:quad:parabola}{parábola}.

\section{la forma canónica}

\begin{definition}[Función cuadrática]\label{def:g11:quad:quadratic}
Un \emph{función cuadrática}\index{función cuadrática} es un \omterm{def:g10:functions:function}{función} del
forma
\[
f(x) = ax^2 + bx + c, \qquad a \neq 0,
\]
donde $a$, $b$, $c$ son \omterm{def:g10:numbers:sets}{números reales}. La expresión $ax^2 + bx + c$ es
también llamado \emph{trinomio cuadrático}.
\end{definition}

\begin{theorem}[forma canónica]\label{thm:g11:quad:canonical}
Cada \omterm{def:g11:quad:quadratic}{función cuadrática} se puede escribir exactamente de una manera como
\[
f(x) = a(x - \alpha)^2 + \beta,
\qquad\text{where } \alpha = -\frac{b}{2a} \text{ and } \beta = f(\alpha).
\]
Este es el \emph{forma canónica}\index{forma canónica} de $f$.
\end{theorem}

\begin{proof}
Nosotros \emph{completar el cuadrado}. Desde $a \neq 0$, factorízalo del primero
dos términos:
\[
ax^2 + bx + c = a\left(x^2 + \frac{b}{a}x\right) + c .
\]
Ahora $x^2 + \frac{b}{a}x$ es el comienzo del cuadrado.
$\left(x + \frac{b}{2a}\right)^2 = x^2 + \frac ba x + \frac{b^2}{4a^2}$,
entonces $x^2 + \frac{b}{a}x = \left(x + \frac{b}{2a}\right)^2 -
\frac{b^2}{4a^2}$. Sustituyendo:
\[
f(x) = a\left(x + \frac{b}{2a}\right)^2 - \frac{b^2}{4a} + c
= a\bigl(x - \alpha\bigr)^2 + \beta,
\]
con $\alpha = -\frac{b}{2a}$ y $\beta = c - \frac{b^2}{4a}$. evaluando
en $x = \alpha$ mata el término al cuadrado, por lo que $\beta = f(\alpha)$. Unicidad:
\omterm{def:g10:algebra:expand}{en expansión} $a(x-\alpha)^2 + \beta$ e identificando los coeficientes de $x$
fuerza $\alpha = -\frac{b}{2a}$ y luego se determina $\beta$.
\end{proof}

\begin{example}\label{ex:g11:quad:canonical}
Deje $f(x) = 2x^2 - 8x + 3$. Luego $\alpha = -\frac{-8}{2 \times 2} = 2$ y
$\beta = f(2) = 8 - 16 + 3 = -5$, entonces
\[
f(x) = 2(x - 2)^2 - 5 .
\]
Consultar por \omterm{def:g10:algebra:expand}{en expansión}: $2(x^2 - 4x + 4) - 5 = 2x^2 - 8x + 8 - 5 =
2x^2 - 8x + 3$.
\end{example}

\begin{proposition}[Extremos y variaciones]\label{prop:g11:quad:variations}
Deje $f(x) = a(x-\alpha)^2 + \beta$.
\begin{enumerate}
  \item Si $a > 0$, entonces $f$ es \omterm{def:g10:functions:variations}{decreciente} en $\intoc{-\infty}{\alpha}$ y
        \omterm{def:g10:functions:variations}{creciente} en $\intco{\alpha}{+\infty}$: $f$ tiene un \emph{\omterm{def:g10:functions:extrema}{mínimo}}
        $\beta$, obtenido en $x = \alpha$.
  \item Si $a < 0$, las variaciones se invierten: $f$ tiene un \emph{\omterm{def:g10:functions:extrema}{máximo}}
        $\beta$ en $x = \alpha$.
\end{enumerate}
\end{proposition}

\begin{proof}
Supongamos $a > 0$ y dejemos que $\alpha \leq u < v$. entonces
\[
f(v) - f(u) = a\bigl((v-\alpha)^2 - (u-\alpha)^2\bigr)
= a\,(v - u)\,\bigl((v-\alpha) + (u-\alpha)\bigr).
\]
Cada factor es positivo: $a > 0$, $v - u > 0$ y
$(v-\alpha) + (u-\alpha) > 0$ desde $v > \alpha$ y $u \geq \alpha$. entonces
$f(v) > f(u)$: $f$ es \omterm{def:g10:functions:variations}{creciente} en $\intco{\alpha}{+\infty}$. Para
$u < v \leq \alpha$ el último factor es negativo, por lo que $f(v) < f(u)$: $f$ es
\omterm{def:g10:functions:variations}{decreciente} ahí. Finalmente $f(x) - f(\alpha) = a(x-\alpha)^2 \geq 0$ para todos
$x$, por lo que $\beta = f(\alpha)$ es \omterm{def:g10:functions:extrema}{mínimo}. El caso $a < 0$ sigue por
aplicando lo anterior a $-f$.
\end{proof}

\begin{example}
Un agricultor tiene $100$ metros de cerca para cercar un campo rectangular contra un
pared recta (no se necesita valla a lo largo de la pared). Si $x$ es el ancho, el
El área cerrada es $A(x) = x(100 - 2x) = -2x^2 + 100x$. Aquí $\alpha =
-\frac{100}{2\times(-2)} = 25$ y $A(25) = 25 \times 50 = 1250$: el área
es mayor para un ancho de $25$ m, lo que da $1250$ m$^2$.
\end{example}

\section{Raíces y factorización}

\begin{definition}[Raíz, discriminante]\label{def:g11:quad:discriminant}
Un \emph{raíz}\index{raíz} del trinomio $ax^2+bx+c$ es una solución del
\omterm{def:g10:algebra:equation}{ecuación} $ax^2 + bx + c = 0$. El \emph{discriminante}\index{discriminante}
del trinomio es el número
\[
\Delta = b^2 - 4ac .
\]
\end{definition}

\begin{theorem}[Resolviendo $ax^2+bx+c=0$]\label{thm:g11:quad:roots}
\begin{enumerate}
  \item Si $\Delta > 0$, el \omterm{def:g10:algebra:equation}{ecuación} tiene dos \omterm{def:g11:quad:discriminant}{raíces} distintos.
        \[
        x_1 = \frac{-b - \sqrt{\Delta}}{2a},
        \qquad
        x_2 = \frac{-b + \sqrt{\Delta}}{2a},
        \]
        y $ax^2+bx+c = a(x - x_1)(x - x_2)$.
  \item Si $\Delta = 0$, el \omterm{def:g10:algebra:equation}{ecuación} tiene el único \omterm{def:g11:quad:discriminant}{raíz}
        $x_0 = -\frac{b}{2a}$ y $ax^2+bx+c = a(x - x_0)^2$.
  \item Si $\Delta < 0$, el \omterm{def:g10:algebra:equation}{ecuación} no tiene una solución real y el
        El trinomio no se puede factorizar con coeficientes reales.
\end{enumerate}
\end{theorem}

\begin{proof}
Comience desde \omterm{thm:g11:quad:canonical}{forma canónica} calculado en \cref{thm:g11:quad:canonical}:
\[
ax^2+bx+c
= a\left(x + \frac{b}{2a}\right)^2 + c - \frac{b^2}{4a}
= a\left[\left(x + \frac{b}{2a}\right)^2 - \frac{\Delta}{4a^2}\right],
\]
desde $c - \frac{b^2}{4a} = \frac{4ac - b^2}{4a} = -\frac{\Delta}{4a} =
a \times \left(-\frac{\Delta}{4a^2}\right)$.

\emph{1.} Si $\Delta > 0$, entonces $\frac{\Delta}{4a^2} =
\left(\frac{\sqrt\Delta}{2a}\right)^2$ y el corchete es una diferencia de
dos cuadrados:
\[
ax^2+bx+c = a\left(x + \frac{b}{2a} - \frac{\sqrt\Delta}{2a}\right)
\left(x + \frac{b}{2a} + \frac{\sqrt\Delta}{2a}\right)
= a(x - x_2)(x - x_1).
\]
Un producto es cero exactamente cuando uno de sus factores es cero, dando a los dos
\omterm{def:g11:quad:discriminant}{raíces}, distinto porque $\sqrt\Delta \neq 0$.

\emph{2.} Si $\Delta = 0$, el soporte es
$\left(x + \frac{b}{2a}\right)^2$, que desaparece sólo en
$x_0 = -\frac{b}{2a}$.

\emph{3.} Si $\Delta < 0$, entonces $-\frac{\Delta}{4a^2} > 0$, entonces el soporte
es un cuadrado más un número positivo: nunca desaparece, y el \omterm{def:g10:algebra:equation}{ecuación} tiene
ninguna solución real. Una factorización $a(x-x_1)(x-x_2)$ produciría \omterm{def:g11:quad:discriminant}{raíces}, por lo que
ninguno existe.
\end{proof}

\begin{example}\label{ex:g11:quad:solving}
Resuelve $2x^2 - 8x + 3 = 0$. Aquí $\Delta = (-8)^2 - 4 \times 2 \times 3 =
64 - 24 = 40 > 0$, entonces hay dos \omterm{def:g11:quad:discriminant}{raíces}:
\[
x_{1,2} = \frac{8 \pm \sqrt{40}}{4} = \frac{8 \pm 2\sqrt{10}}{4}
= 2 \pm \frac{\sqrt{10}}{2}.
\]
Para $x^2 + x + 1 = 0$: $\Delta = 1 - 4 = -3 < 0$, no hay una solución real.
\end{example}

\begin{proposition}[Suma y producto de las raíces.]\label{prop:g11:quad:vieta}
Si $\Delta \geq 0$ y $x_1, x_2$ son \omterm{def:g11:quad:discriminant}{raíces} (igual cuando $\Delta = 0$),
entonces
\[
x_1 + x_2 = -\frac{b}{a},
\qquad
x_1 x_2 = \frac{c}{a} .
\]
Por el contrario, dos números con suma $s$ y producto $p$ son las soluciones de
$x^2 - sx + p = 0$.
\end{proposition}

\begin{proof}
Ampliar la factorización de \cref{thm:g11:quad:roots}:
\[
a(x - x_1)(x - x_2) = a x^2 - a(x_1 + x_2)x + a\,x_1 x_2 .
\]
Identificar los coeficientes con $ax^2 + bx + c$ da
$-a(x_1+x_2) = b$ y $a\,x_1x_2 = c$. Por el contrario, si $u + v = s$ y
$uv = p$, luego $(x-u)(x-v) = x^2 - sx + p$, por lo que $u$ y $v$ son su \omterm{def:g11:quad:discriminant}{raíces}.
\end{proof}

\begin{example}
Encuentra dos números con suma $10$ y producto $21$. Ellos resuelven
$x^2 - 10x + 21 = 0$; $\Delta = 100 - 84 = 16$, \omterm{def:g11:quad:discriminant}{raíces}
$\frac{10 \pm 4}{2} = 7$ y $3$. \emph{Quick \omterm{def:g11:quad:discriminant}{raíz} spotting:} desde el
coeficientes de $x^2 - 5x + 4$ satisfacen $1 - 5 + 4 = 0$, el número $1$ es un
\omterm{def:g11:quad:discriminant}{raíz}, y el otro es $\frac{c}{a} = 4$ (su producto debe ser
$\frac ca$).
\end{example}

\section{Signo de una cuadrática}

\begin{theorem}[Signo de un trinomio]\label{thm:g11:quad:sign}
Deje $f(x) = ax^2+bx+c$ con $a \neq 0$.
\begin{enumerate}
  \item Si $\Delta > 0$, con \omterm{def:g11:quad:discriminant}{raíces} $x_1 < x_2$: $f(x)$ tiene el signo de $a$
        fuera de $\intcc{x_1}{x_2}$, y el signo de $-a$ en
        $\intoo{x_1}{x_2}$.
  \item Si $\Delta = 0$: $f(x)$ tiene el signo de $a$ para todo $x \neq x_0$,
        y desaparece en $x_0$.
  \item Si $\Delta < 0$: $f(x)$ tiene el signo de $a$ para todo $x$.
\end{enumerate}
En resumen: \emph{a trinomial has the sign of $a$, except between its \omterm{def:g11:quad:discriminant}{raíces}}.
\end{theorem}

\begin{proof}
\emph{1.} Escribe $f(x) = a(x-x_1)(x-x_2)$ y sigue las indicaciones del
factores. Si $x < x_1$: tanto $x - x_1$ como $x - x_2$ son negativos, sus
El producto es positivo, por lo que $f(x)$ tiene el signo de $a$. Si $x_1 < x < x_2$: el
Los factores tienen signos opuestos, el producto es negativo y $f(x)$ tiene el
signo de $-a$. Si $x > x_2$: ambos factores son positivos y $f(x)$ tiene el
signo de $a$ nuevamente.

\emph{2.} $f(x) = a(x - x_0)^2$ y $(x-x_0)^2 > 0$ para $x \neq x_0$.

\emph{3.} De la prueba de \cref{thm:g11:quad:roots},
$f(x) = a\left[\left(x + \frac{b}{2a}\right)^2 +
\left(-\frac{\Delta}{4a^2}\right)\right]$, y el soporte siempre es
positivo.
\end{proof}

\begin{method}[Resolver desigualdades cuadráticas]\label{met:g11:quad:inequalities}
Para resolver $ax^2 + bx + c \leq 0$ (o $\geq$, $<$, $>$):
\begin{enumerate}
  \item calcula $\Delta$ y \omterm{def:g11:quad:discriminant}{raíces}, si los hubiera;
  \item dibuja el \omterm{def:g10:algebra:signtable}{mesa de signos} usando \cref{thm:g11:quad:sign} (signo de $a$
        fuera del \omterm{def:g11:quad:discriminant}{raíces});
  \item lea la solución \omterm{def:g10:numbers:interval}{intervalo}(s), prestando atención a si la
        Los propios \omterm{def:g11:quad:discriminant}{raíces} están incluidos (amplia desigualdad) o no (estricta).
\end{enumerate}
\end{method}

\begin{example}\label{ex:g11:quad:inequality}
Resuelve $-x^2 + 3x + 4 \geq 0$. Aquí $\Delta = 9 + 16 = 25$ y \omterm{def:g11:quad:discriminant}{raíces}.
son $\frac{-3 \pm 5}{-2}$, es decir $x_1 = -1$ y $x_2 = 4$. desde
$a = -1 < 0$, el trinomio es negativo fuera del \omterm{def:g11:quad:discriminant}{raíces} y positivo
entre ellos. Con la desigualdad amplia, el conjunto solución es
$\intcc{-1}{4}$.
\end{example}

\section{la parábola}

\begin{definition}[Parábola]\label{def:g11:quad:parabola}
El \omterm{def:g10:functions:graph}{gráfico} de un \omterm{def:g11:quad:quadratic}{función cuadrática} $f(x) = a(x-\alpha)^2+\beta$ es un
\emph{parábola}\index{parábola} con \emph{vértice} $S(\alpha, \beta)$. eso
se abre hacia arriba si $a > 0$, hacia abajo si $a < 0$, y es simétrico con respecto al
línea vertical $x = \alpha$.
\end{definition}

\begin{remark}
La simetría es fácil de comprobar: para cualquier $h$,
$f(\alpha + h) = ah^2 + \beta = f(\alpha - h)$, por lo que dos puntos son iguales
La distancia horizontal desde la línea $x = \alpha$ tiene la misma altura.
\end{remark}

\begin{method}[Leyendo una parábola]\label{met:g11:quad:reading}
Para dibujar $f(x) = ax^2+bx+c$: busque la orientación (signo de $a$), el
vértice $\left(-\frac{b}{2a},\, f\!\left(-\frac{b}{2a}\right)\right)$, el
\omterm{def:g10:numbers:interunion}{intersección} $(0, c)$ con el eje $y$ y el \omterm{def:g11:quad:discriminant}{raíces}, si lo hubiera. El eje de
La simetría $x = -\frac{b}{2a}$ luego coloca cada punto calculado dos veces.
\end{method}

\begin{example}
Para $f(x) = x^2 - 4x + 3$: hacia arriba, el vértice $(2, -1)$, cruza el eje $y$
en $(0,3)$, \omterm{def:g11:quad:discriminant}{raíces} $1$ y $3$ (manchado: $1 - 4 + 3 = 0$). Por simetría, el
El punto $(4, 3)$ también está en la curva, reflejando $(0, 3)$.
\end{example}

\section{Problema: el punto secreto de la parábola}

\begin{problem}[{Problema de fin de semana --- foco y directriz: por qué
antenas parabólicas, faros y puentes colgantes dibujan el
misma curva, de la cual esencialmente solo hay una}]
\label{pb:g11:quad:1}
Cada antena parabólica en cada balcón apunta su receptor a una
punto interior preciso; Cada faro esconde su bombilla en el
mismo lugar especial. Ese punto --- el \emph{\omterm{pb:g11:quad:1}{enfocar}} --- es un
secreto que el \omterm{def:g11:quad:parabola}{parábola} ha mantenido desde la geometría griega, y el
La fórmula de distancia es suficiente para extraerla. En el camino, este
El problema perfora el \omterm{def:g11:quad:discriminant}{discriminante}, lanza proyectiles por encima de las paredes,
suspende un puente, y demuestra un hecho final sorprendente: hasta
zoom, \emph{there is only one \omterm{def:g11:quad:parabola}{parábola} in the world}.

\medskip
\noindent\textbf{Parte I --- Trinomial fluency.}
\begin{enumerate}
  \item Resuelve $x^2 - 5x + 6 = 0$, luego $2x^2 + x - 6 = 0$, luego
        $x^2 + x + 1 = 0$ (\cref{thm:g11:quad:roots}).
  \item Escribe $2x^2 - 4x - 6$ en forma factorizada y en \omterm{thm:g11:quad:canonical}{canónico
        forma} (\cref{thm:g11:quad:canonical}). junto con el
        forma expandida, ahora tienes tres disfraces de uno
        \omterm{def:g10:functions:function}{función}: diga qué característica muestra cada uno de un vistazo
        (\omterm{def:g11:quad:discriminant}{raíces}, vértice, $y$-intersección).
  \item Resuelve $-x^2 + 4x - 3 \geq 0$ con la regla de los signos
        (\cref{thm:g11:quad:sign}).
  \item Viète como detective (\cref{prop:g11:quad:vieta},
        continuando Babilonia en \cref{pb:g10:algebra:1}): encuentra dos
        números con suma $7$ y producto $12$ escribiendo su
        \omterm{def:g10:algebra:equation}{ecuación}. Entonces: un \omterm{def:g11:quad:discriminant}{raíz} de $x^2 - 5x + c = 0$ es igual
        $2$; encuentra $c$ y el otro \omterm{def:g11:quad:discriminant}{raíz} sin resolver
        cualquier cosa.
  \item ¿Para qué valores de $m$ tiene $x^2 + mx + 9 = 0$ un
        doble \omterm{def:g11:quad:discriminant}{raíz}? No \omterm{def:g11:quad:discriminant}{raíz}? ¿Dos \omterm{def:g11:quad:discriminant}{raíces}?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The \omterm{pb:g11:quad:1}{enfocar} unveiled.}
Trabajar en el \omterm{def:g11:quad:parabola}{parábola} $y = x^2$, el punto
$F\left(0, \frac14\right)$ y la línea horizontal $\delta$ de
\omterm{def:g10:algebra:equation}{ecuación} $y = -\frac14$.
\begin{enumerate}[resume]
  \item Para un punto $P(x, x^2)$ del \omterm{def:g11:quad:parabola}{parábola}, expandir
        $PF^2 = x^2 + \left(x^2 - \frac14\right)^2$ y muéstralo
        es el cuadrado perfecto $\left(x^2 + \frac14\right)^2$.
        Deducir $PF$.
  \item Calcule la distancia desde $P$ hasta la línea $\delta$, y
        concluir: \emph{every point of the \omterm{def:g11:quad:parabola}{parábola} is exactly
        as far from $F$ as from $\delta$}. ($F$ es el
        \emph{enfocar}\index{enfocar}, $\delta$ el
        \emph{directora}\index{directora}.)
  \item Demuestre lo contrario: un punto $P(x, y)$ con
        $PF = \operatorname{dist}(P, \delta)$ debe satisfacer
        $y = x^2$. (Eleva al cuadrado ambas distancias y simplifica.)
        \omterm{def:g11:quad:parabola}{parábola} es por tanto un \emph{lugar}, exactamente como el círculo
        estaba en \cref{pb:g10:coordgeom:1}: equidistancia desde un
        punto y una línea, en lugar de desde un punto.
  \item Para el \omterm{def:g11:quad:parabola}{parábola} $y = \frac{x^2}{4f}$ más ancho (con
        $f > 0$), adapte la pregunta 6 para mostrar que \omterm{pb:g11:quad:1}{enfocar} es
        $(0, f)$ y \omterm{pb:g11:quad:1}{directora} $y = -f$. Aplicación: una
        la antena parabólica tiene $60$ cm de ancho y $9$ cm de profundidad ---
        la llanta pasa por $(30, 9)$. Encuentra $f$: ¿a qué altura está arriba?
        ¿En qué vértice debe colgarse el receptor?
  \item La razón por la que las antenas funcionan: los rayos llegan paralelos a la
        Todos los ejes reflejan \emph{through the \omterm{pb:g11:quad:1}{enfocar}} (y una bombilla en
        el \omterm{pb:g11:quad:1}{enfocar} lanza un haz paralelo --- los faros están
        los platos corren hacia atrás). Explica por qué la pregunta 7
        La equidistancia hace que esto sea plausible (piense en los rayos como
        frentes de onda en marcha), y decir qué herramienta, entrando
        \cref{ch:g11:deriv} (ver \cref{exo:g11:quad:11}), gira
        verosimilitud en prueba.
\end{enumerate}

\begin{enumerate}[resume]

  \item En la misma trayectoria, se levanta un muro en $x = 4$.

\noindent\textbf{Parte IV --- Lines, tangents, and the last
surprise.}
\begin{enumerate}[resume]
  \item Completo \cref{exo:g11:quad:11}: las líneas
        $y = mx - 1$ por el punto exterior $(0, -1)$ se encuentra
        $y = x^2$ según el signo de un \omterm{def:g11:quad:discriminant}{discriminante}.
        Encuentra las dos rectas tangentes de $(0, -1)$ y sus
        puntos de contacto.
  \item Ahora las líneas $y = mx + 1$ hasta la \emph{interior}
        punto $(0, 1)$: calcular el \omterm{def:g11:quad:discriminant}{discriminante} del
        \omterm{def:g10:numbers:interunion}{intersección} \omterm{def:g10:algebra:equation}{ecuación} y muestra que cada línea corta el
        \omterm{def:g11:quad:parabola}{parábola} dos veces. Moraleja: desde dentro, sin tangentes ---
        al igual que para un círculo.
  \item La cuerda focal: intersecta $y = x^2$ con la
        línea horizontal que pasa por \omterm{pb:g11:quad:1}{enfocar}, $y = \frac14$. que
        Cuál es la longitud del acorde? Consulta la regla general (vale la pena
        recordando para los platos): para $y = \frac{x^2}{4f}$, el
        cuerda a través del \omterm{pb:g11:quad:1}{enfocar} paralelo al \omterm{pb:g11:quad:1}{directora} tiene
        longitud $4f$.
  \item La última sorpresa: aplica el zoom
        $(x, y) \mapsto (kx, ky)$ al \omterm{def:g11:quad:parabola}{parábola} $y = x^2$ y
        muestre que \omterm{def:g10:functions:function}{imagen} es exactamente $y = \frac{x^2}{k}$. Concluir:
        cada \omterm{def:g11:quad:parabola}{parábola} $y = \frac{x^2}{4f}$ es una unidad ampliada
        \omterm{def:g11:quad:parabola}{parábola} --- a diferencia de círculos y elipses de diferentes
        proporciones, \emph{all \omterm{def:g11:quad:parabola}{parábolas} are similar}.
  \item Final: el retrato del \omterm{def:g11:quad:parabola}{parábola} en cinco líneas --- un
        \omterm{def:g10:algebra:equation}{ecuación} con tres disfraces (pregunta 2), un locus
        (preguntas 7--8), un reflector de ingeniero (preguntas 9,
        10), la trayectoria de un proyectil y el cable de un puente
        (preguntas 11 a 15) y una sola curva hacia arriba para ampliar
        (pregunta 19). Una frase cada uno.
\end{enumerate}
