\chapter{El producto escalar en el plano}\label{ch:g11:scal}

El \omterm{def:g11:scal:dot}{producto escalar} conecta un \emph{número} a un par de \omterm{def:g11:vect:vector}{vectores}, y eso
el número ve la geometría: desaparece exactamente en direcciones perpendiculares,
y mide ángulos y longitudes. Es el motor detrás de la ley de
\omterm{def:g11:trigo:cossin}{cosenos}, \omterm{def:g10:algebra:equation}{ecuaciones} de círculos y cálculos de perpendicularidad. lo mismo
herramienta, en el espacio, es el tema de \cref{ch:g12:space}.

\section{Definición y primeros cálculos.}

\begin{definition}[Escalar producto]\label{def:g11:scal:dot}
El \emph{producto escalar}\index{producto escalar} de dos \omterm{def:g11:vect:vector}{vectores} $\vec u$
y $\vec v$ es el número
\[
\vec u \cdot \vec v
= \tfrac12\left(\norm{\vec u + \vec v}^2 - \norm{\vec u}^2
- \norm{\vec v}^2\right),
\]
donde $\norm{\vec u}$ indica la longitud (\emph{norma}\index{norma}) de $\vec u$.
\end{definition}

\begin{theorem}[Las cuatro caras del producto escalar]\label{thm:g11:scal:faces}
Sean $\vec u\,(x, y)$ y $\vec v\,(x', y')$ distintos de cero \omterm{def:g11:vect:vector}{vectores}, y
$\theta$ el ángulo entre ellos. entonces
\begin{enumerate}
  \item \emph{(\omterm{def:g10:coordgeom:system}{coordenadas})} $\vec u \cdot \vec v = xx' + yy'$;
  \item \emph{(\omterm{def:g11:trigo:cossin}{coseno})} $\vec u \cdot \vec v =
        \norm{\vec u}\,\norm{\vec v}\cos\theta$;
  \item \emph{(proyección)} $\vec u \cdot \vec v = \vec u \cdot \vec v'$,
        donde $\vec v'$ es la proyección \omterm{def:g11:scal:orthogonal}{ortogonal} de $\vec v$ en el
        dirección de $\vec u$; si $\vec u = \vect{OA}$ y
        $\vec v = \vect{OB}$, entonces
        $\vec u \cdot \vec v = \overline{OA} \times \overline{OH}$, un
        producto de longitudes firmadas, donde $H$ es el pie del
        perpendicular desde $B$ a la línea $(OA)$;
  \item \emph{(\omterm{def:g11:scal:dot}{normas})} la fórmula que define \cref{def:g11:scal:dot}.
\end{enumerate}
\end{theorem}

\begin{proof}[Prueba de 1 y 2]
\emph{1.} En \omterm{def:g10:coordgeom:system}{coordenadas}, $\norm{\vec u}^2 = x^2 + y^2$ (Pitágoras) y
$\vec u + \vec v$ tiene \omterm{def:g10:coordgeom:system}{coordenadas} $(x + x', y + y')$, entonces
\begin{align*}
2\,\vec u \cdot \vec v
&= (x + x')^2 + (y + y')^2 - (x^2 + y^2) - (x'^2 + y'^2)\\
&= 2xx' + 2yy',
\end{align*}
después de \omterm{def:g10:algebra:expand}{en expansión} los cuadrados y cancelando.

\emph{2.} Elija \omterm{def:g10:coordgeom:system}{sistema de coordenadas} para que $\vec u$ apunte a lo largo del
positivo $x$-eje: $\vec u\,(\norm{\vec u}, 0)$ y
$\vec v\,(\norm{\vec v}\cos\theta,\ \norm{\vec v}\sin\theta)$
(\cref{ch:g11:trigo}). La Fórmula 1 entonces da
$\vec u \cdot \vec v = \norm{\vec u}\,\norm{\vec v}\cos\theta + 0$.
La Fórmula 3 muestra el mismo cálculo: la proyección de $\vec v$ en
el eje $x$ tiene una longitud con signo $\norm{\vec v}\cos\theta =
\overline{OH}$.
\end{proof}

\begin{example}\label{ex:g11:scal:compute}
$\vec u\,(3, 1)$ y $\vec v\,(2, -4)$:
$\vec u \cdot \vec v = 6 - 4 = 2$. Los \omterm{def:g11:scal:dot}{normas} son
$\norm{\vec u} = \sqrt{10}$ y $\norm{\vec v} = \sqrt{20}$, entonces
$\cos\theta = \frac{2}{\sqrt{10}\sqrt{20}} = \frac{2}{10\sqrt2}
= \frac{\sqrt2}{10}$: el ángulo es un poco menor que $82^\circ$.
\end{example}

\begin{proposition}[reglas algebraicas]\label{prop:g11:scal:rules}
Para todos los \omterm{def:g11:vect:vector}{vectores} y todos los $\lambda \in \R$:
\[
\vec u \cdot \vec v = \vec v \cdot \vec u, \qquad
\vec u \cdot (\vec v + \vec w) = \vec u \cdot \vec v + \vec u \cdot
\vec w, \qquad
(\lambda \vec u)\cdot \vec v = \lambda\,(\vec u \cdot \vec v),
\]
\[
\vec u \cdot \vec u = \norm{\vec u}^2,
\qquad
\norm{\vec u + \vec v}^2 = \norm{\vec u}^2 + 2\,\vec u\cdot\vec v +
\norm{\vec v}^2 .
\]
\end{proposition}

\begin{proof}
Todos se derivan de la fórmula de coordenadas: por ejemplo
$\vec u \cdot (\vec v + \vec w) = x(x' + x'') + y(y' + y'') =
(xx' + yy') + (xx'' + yy'')$. La última identidad se expande.
$\norm{\vec u + \vec v}^2 = (\vec u + \vec v)\cdot(\vec u + \vec v)$
usando las primeras tres reglas --- el \omterm{def:g11:scal:dot}{producto escalar} se comporta exactamente como un
producto ordinario, por lo que las identidades algebraicas habituales se aplican a \omterm{def:g11:vect:vector}{vectores}.
\end{proof}

\section{Ortogonalidad}

\begin{definition}[Vectores ortogonales]\label{def:g11:scal:orthogonal}
Dos \omterm{def:g11:vect:vector}{vectores} son \emph{ortogonal}\index{vectores ortogonales}, escritos
$\vec u \perp \vec v$, si $\vec u \cdot \vec v = 0$.
\end{definition}

\begin{remark}
Para \omterm{def:g11:vect:vector}{vectores} distinto de cero, se muestra la fórmula 2 de \cref{thm:g11:scal:faces}
$\vec u \cdot \vec v = 0 \iff \cos\theta = 0 \iff$ las direcciones son
perpendicular. El cero \omterm{def:g11:vect:vector}{vector} es \omterm{def:g11:scal:orthogonal}{ortogonal} para todo.
\end{remark}

\begin{example}
$\vec u\,(3, 2)$ y $\vec v\,(-4, 6)$: $\vec u \cdot \vec v = -12 + 12 =
0$: \omterm{def:g11:scal:orthogonal}{ortogonal}. En general, $\vec u\,(a, b)$ es siempre \omterm{def:g11:scal:orthogonal}{ortogonal} para
$\vec v\,(-b, a)$ --- una rotación de un cuarto de vuelta.
\end{example}

\section{La ley de los cosenos}

\begin{theorem}[Ley de los Coseños]\label{thm:g11:scal:cosines}
\index{ley de los cosenos}
En un triángulo $ABC$ con lados $a = BC$, $b = CA$, $c = AB$ y ángulo
$\widehat A$ en el vértice $A$:
\[
a^2 = b^2 + c^2 - 2bc\,\cos\widehat A .
\]
\end{theorem}

\begin{proof}
Escribe $\vect{BC} = \vect{BA} + \vect{AC} = \vect{AC} - \vect{AB}$ y
ampliar con \cref{prop:g11:scal:rules}:
\[
a^2 = \norm{\vect{BC}}^2
= \norm{\vect{AC}}^2 - 2\,\vect{AC}\cdot\vect{AB} + \norm{\vect{AB}}^2
= b^2 + c^2 - 2\,\vect{AB}\cdot\vect{AC}.
\]
Según la fórmula \omterm{def:g11:trigo:cossin}{coseno},
$\vect{AB}\cdot\vect{AC} = c\,b\,\cos\widehat A$.
\end{proof}

\begin{remark}
Cuando $\widehat A$ es un ángulo recto, $\cos\widehat A = 0$ y la ley de
\omterm{def:g11:trigo:cossin}{cosenos} se reduce al teorema de Pitágoras $a^2 = b^2 + c^2$: es
Pitágoras con un término de corrección para ángulos no rectos.
\end{remark}

\begin{example}\label{ex:g11:scal:cosines}
Dos lados de un triángulo miden $b = 5$ y $c = 8$ y encierran un ángulo.
$\widehat A = \frac\pi3$. entonces
\[
a^2 = 25 + 64 - 2 \times 5 \times 8 \times \frac12 = 89 - 40 = 49,
\]
entonces $a = 7$.
\end{example}

\section{Aplicaciones: círculos y líneas perpendiculares.}

\begin{proposition}[Círculo con un diámetro determinado]\label{prop:g11:scal:diameter}
Un punto $M$ se encuentra en el círculo de diámetro $[AB]$ si y sólo si
\[
\vect{MA} \cdot \vect{MB} = 0 .
\]
\end{proposition}

\begin{proof}
Si $M \neq A, B$, la condición dice el ángulo en $M$ en el triángulo
$AMB$ tiene razón y los puntos que ven un segmento bajo un ángulo recto son
exactamente los del círculo con ese diámetro. Aquí hay una prueba directa:
sea $O$ el \omterm{prop:g10:coordgeom:midpoint}{punto medio} de $[AB]$ y $r = \frac{AB}{2}$. entonces
$\vect{MA} = \vect{MO} + \vect{OA}$ y
$\vect{MB} = \vect{MO} + \vect{OB} = \vect{MO} - \vect{OA}$, entonces
\[
\vect{MA}\cdot\vect{MB}
= \norm{\vect{MO}}^2 - \norm{\vect{OA}}^2 = MO^2 - r^2,
\]
que desaparece exactamente cuando $MO = r$, es decir,\ cuando $M$ está en el círculo de
centro $O$ y radio $r$.
\end{proof}

\begin{proposition}[Vector normal de una recta]\label{prop:g11:scal:normal}
El \omterm{def:g11:vect:vector}{vector} $\vec n\,(a, b)$ es \omterm{def:g11:scal:orthogonal}{ortogonal} para cada \omterm{def:g11:vect:direction}{vector de direccion} del
línea $d : ax + by + c = 0$; uno dice $\vec n$ es un \emph{normales
vector}\index{vectores normales} de $d$. Dos rectas son exactamente perpendiculares
cuando su normal (o dirección) \omterm{def:g11:vect:vector}{vectores} es \omterm{def:g11:scal:orthogonal}{ortogonal}.
\end{proposition}

\begin{proof}
Un \omterm{def:g11:vect:direction}{vector de direccion} de $d$ es $\vec u\,(-b, a)$
(\cref{thm:g11:vect:cartesian}), y
$\vec n \cdot \vec u = a(-b) + b(a) = 0$.
\end{proof}

\begin{method}[Usando vectores normales]\label{met:g11:scal:normal}
\begin{itemize}
  \item Línea a través de $A$ con $\vec n\,(a,b)$ normal: escribir
        $\vect{AM} \cdot \vec n = 0$, que se expande a $ax + by + c = 0$
        con $c$ determinado por $A$.
  \item Perpendicularidad de $d: ax+by+c = 0$ y $d': a'x+b'y+c' = 0$:
        marque $aa' + bb' = 0$.
\end{itemize}
\end{method}

\begin{example}\label{ex:g11:scal:perpline}
La línea que pasa por $P(2, -1)$ perpendicular a $d : 3x - y + 4 = 0$: a
\omterm{def:g11:vect:direction}{vector de direccion} de $d$ es $(1, 3)$, que sirve como \emph{normal} \omterm{def:g11:vect:vector}{vector}
de la perpendicular. \omterm{def:g10:algebra:equation}{Ecuación}: $x + 3y + c = 0$ con
$2 - 3 + c = 0$, entonces $c = 1$: la línea es $x + 3y + 1 = 0$.
\end{example}

\begin{example}[Ecuación de un círculo]\label{ex:g11:scal:circle}
La circunferencia de centro $\Omega(1, 2)$ y radio $3$ es el conjunto de puntos
$M(x,y)$ con $\Omega M^2 = 9$:
\[
(x - 1)^2 + (y - 2)^2 = 9 .
\]
Por el contrario, $x^2 + y^2 - 2x - 4y - 4 = 0$ reescribe, completando el
cuadrados (\cref{ch:g11:quad}), como $(x-1)^2 + (y-2)^2 = 9$: lo mismo
círculo.
\end{example}

\section{Problema: Dos regalos del producto escalar}

\begin{problem}[{Problema de fin de semana --- las fórmulas de suma caen
de un producto escalar, la distancia a una línea de otra,
y las tres altitudes se encuentran como un bono}]
\label{pb:g11:scal:1}
El \omterm{def:g11:scal:dot}{producto escalar} parece contabilidad --- multiplicar, sumar ---
sin embargo, entrega, casi gratuitamente, dos tesoros que
resistió siglos de persecución de ángulos: el \emph{fórmulas de suma}
de trigonometría (¡calcule $\cos 15^\circ$ exactamente!) y el
\emph{distancia de un punto a una recta} en una fracción limpia. como
un regalo de despedida demuestra que las tres altitudes de cada
El triángulo pasa por un punto. Las cuatro caras de
\cref{thm:g11:scal:faces} se presentará a trabajar.

\medskip
\noindent\textbf{Parte I --- Fluency.}
\begin{enumerate}
  \item Para $\vec u(3, 4)$ y $\vec v(-1, 2)$: calcular
        $\vec u \cdot \vec v$, los dos \omterm{def:g11:scal:dot}{normas} y el ángulo
        entre el \omterm{def:g11:vect:vector}{vectores} (hasta la décima de grado).
  \item Encuentre $k$ para que $(2, k)$ y $(3, -4)$ sean \omterm{def:g11:scal:orthogonal}{ortogonal}
        (\cref{def:g11:scal:orthogonal}).
  \item Cara de física: un trineo es arrastrado $10$ m por una fuerza $50$ N
        en $60^\circ$ al suelo. calcular el trabajo
        $W = \vec F \cdot \vec d$.
  \item Ley de \omterm{def:g11:trigo:cossin}{cosenos} (\cref{thm:g11:scal:cosines}): dos lados
        de $5$ y $7$ encierran un ángulo de $60^\circ$; encontrar el
        tercer lado exactamente.
  \item Inversa: un triángulo tiene lados $4$, $6$, $8$. Calcular el
        \omterm{def:g11:trigo:cossin}{coseno} del ángulo opuesto al lado $8$, clasifica el
        ángulo y dale a la décima de grado.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- First gift: the addition
formulas.}
Sean $\vec u = (\cos a, \sin a)$ y $\vec v = (\cos b, \sin b)$:
dos unidades \omterm{def:g11:vect:vector}{vectores} en los ángulos $a$ y $b$.
\begin{enumerate}[resume]
  \item Calcular $\vec u \cdot \vec v$ dos veces --- una vez con
        \omterm{def:g10:coordgeom:system}{coordenadas}, una vez con \omterm{def:g11:scal:dot}{normas} y el ángulo entre los
        \omterm{def:g11:vect:vector}{vectores} --- y concluir la fórmula madre:
        \[
        \cos(a - b) = \cos a \cos b + \sin a \sin b .
        \]
  \item Sustituya $-b$ por $b$ (¡paridades de \omterm{def:g11:trigo:cossin}{seno} y \omterm{def:g11:trigo:cossin}{coseno}!)
        para obtener $\cos(a + b)$.
  \item Obtener $\sin(a + b)$ de la identidad complementaria
        $\sin x = \cos\left(\frac\pi2 - x\right)$ aplicado a
        $x = a + b$.
  \item Por fin valores exactos: calcular $\cos 15^\circ$ (como
        $\cos(45^\circ - 30^\circ)$) y $\sin 75^\circ$. comprobar
        ambos numéricamente.
  \item Configure $b = a$ para derivar las fórmulas de doble ángulo:
        $\cos 2a$ (en sus tres formas) y $\sin 2a$.
  \item Desde $\cos 2a = 1 - 2\sin^2 a$, calcule
        $\sin^2 15^\circ$, luego $\sin 15^\circ$ como anidado
        radical --- y verificar que su cuadrado concuerde con el
        valor $\frac{\sqrt6 - \sqrt2}{4}$ esa pregunta 9's
        método da.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Second gift: the distance to a
line.}
\begin{enumerate}[resume]
  \item Sea $d$ la línea $ax + by + c = 0$, con normal
        \omterm{def:g11:vect:vector}{vector} $\vec n(a, b)$ (\cref{prop:g11:scal:normal}),
        y $P(x_0, y_0)$ un punto. Tomando cualquier punto $A$ en $d$,
        explique por qué la distancia de $P$ a $d$ es
        $\dfrac{\abs{\vect{AP} \cdot \vec n}}{\norm{\vec n}}$,
        y deducir la fórmula
        \[
        \operatorname{dist}(P, d)
        = \frac{\abs{a x_0 + b y_0 + c}}{\sqrt{a^2 + b^2}} .
        \]
        Calcula la distancia desde $P(5, 6)$ hasta
        $3x + 4y - 12 = 0$.
  \item Escribe el \omterm{def:g10:algebra:equation}{ecuación} del círculo centrado en $(5, 6)$
        \omterm{def:g11:deriv:tangent}{tangente} a esa línea.
  \item Verifique la fórmula: calcule la distancia desde el
        origen a $3x + 4y - 12 = 0$, luego verifíquelo mediante
        calcular el pie de la perpendicular desde el origen
        explícitamente.
  \item Triángulo $A(0,0)$, $B(6,0)$, $C(2,5)$: calcula su área
        utilizando la base $[AB]$; luego calcula la distancia desde
        $B$ a la línea $(AC)$ y comprueba que reproduce lo mismo
        zona.
  \item El círculo de centro $(2, 1)$ y radio $2$, contra
        la línea $x + y - 6 = 0$: calcula la distancia del centro
        a la línea y clasificar su posición (\omterm{def:g11:deriv:rate}{secante},
        \omterm{def:g11:deriv:tangent}{tangente}, disjunto).
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The bonus: altitudes concur.}
\begin{enumerate}[resume]
  \item Generalizar \cref{exo:g11:scal:11}: para dos puntos cualesquiera
        $A$, $B$ con \omterm{prop:g10:coordgeom:midpoint}{punto medio} $I$, probar
        \[
        \vect{MA} \cdot \vect{MB} = MI^2 - \frac{AB^2}{4}
        \quad\text{for every point } M .
        \]
  \item Deduce \cref{prop:g11:scal:diameter} en una línea: el
        conjunto de puntos $M$ con
        $\vect{MA} \cdot \vect{MB} = 0$ es el círculo de
        diámetro $[AB]$.
  \item Las altitudes: sea $H$ el \omterm{def:g10:numbers:interunion}{intersección} del
        altitudes desde $A$ y desde $B$ en un triángulo $ABC$
        (entonces $\vect{HA} \cdot \vect{BC} = 0$ y
        $\vect{HB} \cdot \vect{CA} = 0$). demostrar la identidad
        \[
        \vect{HA} \cdot \vect{BC} + \vect{HB} \cdot \vect{CA}
        + \vect{HC} \cdot \vect{AB} = 0
        \]
        --- ampliar todo desde $H$ con Chasles --- y
        concluya que $H$ también se encuentra en la altitud de $C$:
        las tres altitudes son concurrentes (en el
        \emph{ortocentro}\index{ortocentro}).
  \item Final: cuatro caras, tres trofeos. ¿Qué cara de
        \cref{thm:g11:scal:faces} llevaba las fórmulas de suma,
        cuál la fórmula de distancia, cuál el \omterm{pb:g11:scal:1}{ortocentro} ---
        ¿Y por qué el lema es "calcular la misma cantidad dos veces"?
        de todo el capitulo?
\end{enumerate}
\end{problem}
