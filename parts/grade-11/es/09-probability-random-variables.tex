\section{Probabilidad en un conjunto finito}

\begin{definition}[Modelo de probabilidad]\label{def:g11:prob:model}
Un experimento con un número finito de resultados está modelado por su \emph{muestra
espacio}\index{espacio muestral} $\Omega = \{\omega_1, \dots, \omega_n\}$ (el
conjunto de resultados) junto con una asignación \emph{\omterm{def:g10:proba:distribution}{probabilidad}} $\P$ a cada
resultado $\omega_i$ un número $p_i \geq 0$, con $p_1 + \dots + p_n = 1$. un
\emph{evento}\index{evento} es un subconjunto $A \subseteq \Omega$, y
\[
\P(A) = \sum_{\omega_i \in A} p_i .
\]
Cuando todos los resultados son igualmente probables ($p_i = \frac1n$: el \emph{uniforme
modelo}\index{modelo uniforme}),
\[
\P(A) = \frac{\text{number of outcomes in } A}
{\text{number of outcomes in } \Omega}.
\]
\end{definition}

\begin{example}\label{ex:g11:prob:twodice}
Tira dos dados distinguibles: $\Omega$ es el conjunto de los pares ordenados $36$
$(i, j)$ con $1 \leq i, j \leq 6$, todos igualmente probables. El \omterm{def:g11:prob:model}{evento}
$S$ = ``la suma es $7$'' contiene los pares $6$
$(1,6)$, $(2,5)$, $(3,4)$, $(4,3)$, $(5,2)$, $(6,1)$, entonces
$\P(S) = \frac{6}{36} = \frac16$. El \omterm{def:g11:prob:model}{evento} ``la suma es $12$'' contiene
sólo $(6,6)$: \omterm{def:g10:proba:distribution}{probabilidad} $\frac{1}{36}$. Eligiendo el \emph{bien}
\omterm{def:g11:prob:model}{espacio muestral} --- pares ordenados, no desordenados --- es lo que hace que el
\omterm{def:g11:prob:model}{modelo uniforme} aplicable.
\end{example}

\begin{proposition}[Reglas de probabilidad]\label{prop:g11:prob:rules}
Para \omterm{def:g11:prob:model}{eventos} $A, B \subseteq \Omega$:
\[
\P(\emptyset) = 0, \qquad \P(\Omega) = 1, \qquad
\P(\bar A) = 1 - \P(A),
\]
\[
\P(A \cup B) = \P(A) + \P(B) - \P(A \cap B),
\]
donde $\bar A$ es \emph{\omterm{def:g10:proba:operations}{complementar}} de $A$ (todos los resultados no están en $A$).
En particular, $\P(A \cup B) = \P(A) + \P(B)$ cuando $A$ y $B$ son
\emph{\omterm{def:g10:proba:operations}{incompatible}} ($A \cap B = \emptyset$).
\end{proposition}

\begin{proof}
El \omterm{def:g11:prob:model}{evento} vacío no contiene ningún resultado (suma vacía) y el $\Omega$ contiene
todos ellos (suma total $1$). Cada resultado está exactamente en uno de $A$,
$\bar A$, entonces $\P(A) + \P(\bar A) = 1$. Para el \omterm{def:g10:numbers:interunion}{unión}: sumando el $p_i$
sobre $A$ y luego sobre $B$ cuenta los resultados de $A \cap B$ dos veces;
restar $\P(A \cap B)$ corrige el conteo doble.
\end{proof}

\begin{example}
En una clase, $60\%$ estudia música ($M$), $45\%$ practica un deporte ($S$) y
$25\%$ haz ambas cosas. La proporción que hace al menos uno es
$\P(M \cup S) = 0.60 + 0.45 - 0.25 = 0.80$, entonces $20\%$ no hace ninguna de las dos cosas
(\omterm{def:g10:proba:operations}{complementar}).
\end{example}

\section{variables aleatorias}

\begin{definition}[Variable aleatoria, distribución.]\label{def:g11:prob:rv}
Un \emph{variable aleatoria}\index{variable aleatoria} en $\Omega$ es un \omterm{def:g11:func:function}{función}
$X$ asignando un número a cada resultado. su
\emph{distribución}\index{distribución} es la tabla de sus posibles
valores $x_1, \dots, x_k$ junto con sus probabilidades
\[
\P(X = x_i) = \P\bigl(\{\omega : X(\omega) = x_i\}\bigr),
\qquad
\sum_{i=1}^k \P(X = x_i) = 1 .
\]
\end{definition}

\begin{example}\label{ex:g11:prob:game}
Un juego: paga $2$ euros, tira un dado justo, recibe el valor mostrado si
es al menos $5$, nada más. Sea $G$ el \emph{ganar} (recibido
menos pagado). Los resultados $1, 2, 3, 4$ dan $G = -2$; el resultado $5$
da $G = 3$; el resultado $6$ da $G = 4$. El \omterm{def:g11:prob:rv}{distribución} de $G$:
\begin{center}
\begin{tabular}{c|ccc}
$g$ & $-2$ & $3$ & $4$\\
\hline\rule{0pt}{11pt}%
$\P(G = g)$ & $\dfrac46$ & $\dfrac16$ & $\dfrac16$
\end{tabular}
\end{center}
\end{example}

\begin{method}[Construyendo una tabla de distribución]\label{met:g11:prob:table}
Enumere los posibles valores de $X$; Para cada valor, recopile los resultados.
producirlo y sumar sus probabilidades; comprobar que la fila de
Las probabilidades suman $1$.
\end{method}

\section{Expectativa}

\begin{definition}[expectativa]\label{def:g11:prob:expectation}
El \emph{expectativa}\index{expectativa} (o \omterm{def:g11:stat:mean}{significar}) de $X$ es el promedio
de sus valores ponderados por sus probabilidades:
\[
\E(X) = \sum_{i=1}^{k} \P(X = x_i)\; x_i .
\]
\end{definition}

\begin{remark}
$\E(X)$ es lo que el \omterm{def:g11:stat:mean}{significar} de una enorme cantidad de repeticiones independientes de
el experimento sería --- la versión precisa de esta afirmación es la
ley de los grandes números, demostrada en \cref{ch:g12:sums}. Tenga en cuenta lo formal
analogía con el \omterm{def:g11:stat:mean}{significar} de un conjunto de datos con frecuencias
(\cref{def:g11:stat:mean}): las probabilidades juegan el papel de relativo
frecuencias $\frac{n_i}{n}$.
\end{remark}

\begin{example}\label{ex:g11:prob:gameexp}
Para el juego de \cref{ex:g11:prob:game}:
\[
\E(G) = \frac46 \times (-2) + \frac16 \times 3 + \frac16 \times 4
= \frac{-8 + 3 + 4}{6} = -\frac16 \approx -0.17 .
\]
En promedio, el jugador pierde aproximadamente $17$ centavos por juego: el juego es
desfavorable. Un juego se llama \emph{justo} cuando la ganancia esperada es $0$.
\end{example}

\begin{proposition}[Linealidad]\label{prop:g11:prob:linearity}
Para todos los reales $a, b$:
\[
\E(aX + b) = a\,\E(X) + b .
\]
\end{proposition}

\begin{proof}
La variable $aX + b$ toma el valor $a x_i + b$ con \omterm{def:g10:proba:distribution}{probabilidad}
$\P(X = x_i)$, entonces
\[
\E(aX + b) = \sum_i \P(X = x_i)(a x_i + b)
= a \sum_i \P(X = x_i)\,x_i + b \underbrace{\sum_i \P(X =
x_i)}_{=\,1} = a\,\E(X) + b .
\]
\end{proof}

\section{Varianza y desviación estándar}

\begin{definition}[variación]\label{def:g11:prob:variance}
El \emph{diferencia}\index{diferencia} de $X$ mide la propagación de su
valores alrededor de \omterm{def:g11:prob:expectation}{expectativa} $m = \E(X)$:
\[
\V(X) = \E\bigl((X - m)^2\bigr)
= \sum_{i=1}^{k} \P(X = x_i)\,(x_i - m)^2,
\qquad
\sigma(X) = \sqrt{\V(X)} .
\]
\end{definition}

\begin{proposition}[Fórmula de acceso directo]\label{prop:g11:prob:konig}
\[
\V(X) = \E(X^2) - \E(X)^2 .
\]
\end{proposition}

\begin{proof}
Expande $(x_i - m)^2 = x_i^2 - 2m\,x_i + m^2$ dentro de la suma:
\[
\V(X) = \sum_i \P(X = x_i)\,x_i^2
- 2m \sum_i \P(X = x_i)\,x_i + m^2\sum_i \P(X = x_i)
= \E(X^2) - 2m^2 + m^2 ,
\]
que es $\E(X^2) - m^2$. Esta es la versión de variable aleatoria del
Identidad probada para conjuntos de datos en \cref{prop:g11:stat:shortcut}.
\end{proof}

\begin{proposition}[Transformación afín]\label{prop:g11:prob:affine}
$\V(aX + b) = a^2\,\V(X)$ y $\sigma(aX + b) = \abs a\,\sigma(X)$.
\end{proposition}

\begin{proof}
Por linealidad, $aX + b$ se desvía de su \omterm{def:g11:prob:expectation}{expectativa} $a m + b$ en
$(aX + b) - (am + b) = a(X - m)$: cada desviación se escala en $a$, por lo tanto
cada desviación al cuadrado por $a^2$, y también lo es su \omterm{def:g11:stat:mean}{significar} ponderado. el
shift $b$ ha desaparecido --- cambiar una variable no cambia su
difundir.
\end{proof}

\begin{example}\label{ex:g11:prob:variance}
Para el juego de \cref{ex:g11:prob:game}:
$\E(G^2) = \frac46 \times 4 + \frac16 \times 9 + \frac16 \times 16
= \frac{16 + 9 + 16}{6} = \frac{41}{6}$, entonces
\[
\V(G) = \frac{41}{6} - \left(-\frac16\right)^2
= \frac{41}{6} - \frac{1}{36} = \frac{245}{36} \approx 6.8,
\qquad
\sigma(G) \approx 2.6 .
\]
La pérdida de $0.17$ por juego en promedio viene con cambios de tamaño típicos.
$2.6$ euros --- \omterm{def:g11:prob:expectation}{expectativa} sin \omterm{def:g11:stat:variance}{desviación estándar} dice solo la mitad
la historia.
\end{example}
