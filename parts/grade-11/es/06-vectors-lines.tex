\chapter{Vectores y rectas en el plano}\label{ch:g11:vect}

\omterm{def:g11:vect:vector}{Vectores} codifica desplazamientos; las líneas son sus trayectorias. La novedad clave
herramienta de este capítulo es el \emph{\omterm{def:g11:vect:det}{determinante}}, un número único que
decide si dos \omterm{def:g11:vect:vector}{vectores} son \omterm{def:g11:vect:collinear}{colineal} --- y, a partir de ahí, produce
\omterm{def:g10:algebra:equation}{ecuaciones} de líneas y resuelve cuestiones de alineación y paralelismo por puro
cálculo. Las mismas ideas se extienden al espacio en \cref{ch:g12:space}.

\section{Vectores: un resumen}

\begin{definition}[Coordenadas vectoriales]\label{def:g11:vect:vector}
A \emph{vector}\index{vector} $\vec u$ representa un desplazamiento: todos
Las flechas con la misma dirección, orientación y longitud representan lo mismo.
vector. En un \omterm{def:g10:coordgeom:system}{sistema de coordenadas}, $\vec u\,(x, y)$ registra el desplazamiento
($x$ horizontalmente, $y$ verticalmente); para los puntos $A(x_A, y_A)$ y
$B(x_B, y_B)$,
\[
\vect{AB}\,\bigl(x_B - x_A,\ y_B - y_A\bigr).
\]
Los vectores se suman en coordenadas, y $\lambda\vec u$ tiene \omterm{def:g10:coordgeom:system}{coordenadas}
$(\lambda x, \lambda y)$.
\end{definition}

\begin{proposition}[punto medio]\label{prop:g11:vect:midpoint}
El \omterm{prop:g10:coordgeom:midpoint}{punto medio} $M$ del segmento $[AB]$ tiene \omterm{def:g10:coordgeom:system}{coordenadas}
$\left(\frac{x_A + x_B}{2},\ \frac{y_A + y_B}{2}\right)$.
\end{proposition}

\begin{proof}
$M$ se caracteriza por $\vect{AM} = \frac12\vect{AB}$: en cuanto a coordenadas,
$x_M - x_A = \frac12(x_B - x_A)$, entonces $x_M = \frac{x_A + x_B}{2}$, y
de manera similar para $y_M$.
\end{proof}

\section{Colinealidad y el determinante.}

\begin{definition}[Vectores colineales]\label{def:g11:vect:collinear}
Dos \omterm{def:g11:vect:vector}{vectores} son \emph{colineal}\index{Vectores colineales} si uno es un
múltiplo escalar del otro: $\vec v = \lambda \vec u$ para algunos
$\lambda \in \R$ (o $\vec u = \vec 0$). Geométricamente: tienen lo mismo
dirección, o uno de ellos es cero.
\end{definition}

\begin{definition}[Determinante]\label{def:g11:vect:det}
El \emph{determinante}\index{determinante} de $\vec u\,(x, y)$ y
$\vec v\,(x', y')$ es
\[
\det(\vec u, \vec v) =
\begin{vmatrix} x & x' \\ y & y' \end{vmatrix}
= x y' - x' y .
\]
\end{definition}

\begin{theorem}[Criterio de colinealidad]\label{thm:g11:vect:collinearity}
Dos \omterm{def:g11:vect:vector}{vectores} $\vec u$ y $\vec v$ son \omterm{def:g11:vect:collinear}{colineal} si y solo si
$\det(\vec u, \vec v) = 0$.
\end{theorem}

\begin{proof}
($\Rightarrow$) Si $\vec v = \lambda\vec u$, entonces $x' = \lambda x$ y
$y' = \lambda y$, entonces
$\det(\vec u, \vec v) = x(\lambda y) - (\lambda x)y = 0$; si
$\vec u = \vec 0$, el \omterm{def:g11:vect:det}{determinante} es trivialmente $0$.

($\Leftarrow$) Supongamos que $xy' - x'y = 0$ y $\vec u \neq \vec 0$, digamos
$x \neq 0$ (el caso $y \neq 0$ es simétrico). conjunto
$\lambda = \frac{x'}{x}$. Luego $x' = \lambda x$ por construcción, y el
la relación $xy' = x'y = \lambda x y$ da, después de dividir por $x \neq 0$,
$y' = \lambda y$. Por lo tanto $\vec v = \lambda \vec u$.
\end{proof}

\begin{example}\label{ex:g11:vect:collinear}
$\vec u\,(3, -2)$ y $\vec v\,(-6, 4)$:
$\det = 3 \times 4 - (-6) \times (-2) = 12 - 12 = 0$, \omterm{def:g11:vect:collinear}{colineal} (de hecho
$\vec v = -2\vec u$). $\vec u\,(3, -2)$ y $\vec w\,(2, 1)$:
$\det = 3 + 4 = 7 \neq 0$, no \omterm{def:g11:vect:collinear}{colineal}.
\end{example}

\begin{method}[Alineación de tres puntos]\label{met:g11:vect:alignment}
$A$, $B$, $C$ están alineados si y sólo si $\vect{AB}$ y $\vect{AC}$ están
\omterm{def:g11:vect:collinear}{colineal}: calcula \omterm{def:g11:vect:vector}{vectores} y verifica
$\det\bigl(\vect{AB}, \vect{AC}\bigr) = 0$.
\end{method}

\section{Ecuaciones cartesianas de rectas.}

\begin{definition}[Vector de dirección]\label{def:g11:vect:direction}
Un \emph{vector de direccion}\index{vector de direccion} de una línea $d$ es cualquier
distinto de cero \omterm{def:g11:vect:vector}{vector} $\vec u$ tal que $\vect{AB}$ es \omterm{def:g11:vect:collinear}{colineal} con $\vec u$
para todos los puntos $A, B$ de $d$.
\end{definition}

\begin{theorem}[Ecuación cartesiana]\label{thm:g11:vect:cartesian}
Una línea con \omterm{def:g11:vect:direction}{vector de direccion} $\vec u\,(-b, a)$ admite un \omterm{def:g10:algebra:equation}{ecuación} del
forma
\[
ax + by + c = 0
\qquad (a, b) \neq (0, 0).
\]
Por el contrario, cada uno de estos \omterm{def:g10:algebra:equation}{ecuación} describe una línea con \omterm{def:g11:vect:direction}{vector de direccion}
$\vec u\,(-b, a)$.
\end{theorem}

\begin{proof}
Deje que $d$ pase por $A(x_A, y_A)$ con dirección $\vec u\,(-b, a)$. un
El punto $M(x, y)$ se encuentra en $d$ exactamente cuando $\vect{AM}$ es \omterm{def:g11:vect:collinear}{colineal} con
$\vec u$, es decir\ (\cref{thm:g11:vect:collinearity})
\[
0 = \det\bigl(\vect{AM}, \vec u\bigr)
= (x - x_A)\,a - (-b)(y - y_A)
= ax + by - (a x_A + b y_A),
\]
un \omterm{def:g10:algebra:equation}{ecuación} de la forma anunciada con $c = -(a x_A + b y_A)$.
Por el contrario, las soluciones de $ax + by + c = 0$ con, digamos, $b \neq 0$ son
los puntos $\left(x, -\frac{a x + c}{b}\right)$: restando dos de ellos
muestra que cada acorde es \omterm{def:g11:vect:collinear}{colineal} con $(-b, a)$, y siempre hay una solución
existe --- es la línea que lo atraviesa dirigida por $(-b, a)$.
\end{proof}

\begin{method}[Línea que pasa por dos puntos]\label{met:g11:vect:twopoints}
Para encontrar un \omterm{def:g10:algebra:equation}{ecuación} de la línea $(AB)$: calcule el \omterm{def:g11:vect:direction}{vector de direccion}
$\vect{AB}\,(x_B - x_A, y_B - y_A)$; combinarlo con $(-b, a)$, es decir, \ tomar
$a = y_B - y_A$ y $b = -(x_B - x_A)$; determine $c$ conectando el
\omterm{def:g10:coordgeom:system}{coordenadas} de $A$.
\end{method}

\begin{example}\label{ex:g11:vect:twopoints}
Línea a través de $A(1, 2)$ y $B(4, 3)$: $\vect{AB}\,(3, 1)$, por lo que
$a = 1$, $b = -3$ y \omterm{def:g10:algebra:equation}{ecuación} es $x - 3y + c = 0$. Conectando $A$:
$1 - 6 + c = 0$, entonces $c = 5$. \omterm{def:g10:algebra:equation}{Ecuación}: $x - 3y + 5 = 0$. Consulte con $B$:
$4 - 9 + 5 = 0$.
\end{example}

\begin{remark}
Cuando $b \neq 0$, el \omterm{def:g10:algebra:equation}{ecuación} se puede resolver para $y$: $y = mx + p$ con
\omterm{def:g10:reffunc:affine}{pendiente} $m = -\frac ab$. La forma cartesiana $ax + by + c = 0$ es más
general: también cubre líneas verticales ($b = 0$), que no tienen \omterm{def:g10:reffunc:affine}{pendiente}.
\end{remark}

\section{Representación paramétrica}

\begin{definition}[Representación paramétrica]\label{def:g11:vect:parametric}
La línea que pasa por $A(x_A, y_A)$ con dirección $\vec u\,(\alpha, \beta)$ es
el conjunto de puntos
\[
M\bigl(x_A + t\alpha,\ y_A + t\beta\bigr), \qquad t \in \R .
\]
El número $t$ es el \emph{parámetro}: cada valor de $t$ produce un
punto de la línea, como si viajara a lo largo de ella con velocidad constante $\vec u$.
\end{definition}

\begin{example}
La línea de \cref{ex:g11:vect:twopoints} también es
$\{(1 + 3t,\ 2 + t) : t \in \R\}$: $t = 0$ da $A$, $t = 1$ da $B$.
Al eliminar $t$ ($t = y - 2$, por lo que $x = 1 + 3(y-2)$) se recupera
$x - 3y + 5 = 0$.
\end{example}

\section{Posiciones relativas de dos líneas.}

\begin{proposition}[Paralelo o intersectante]\label{prop:g11:vect:positions}
Dos líneas con \omterm{def:g11:vect:direction}{vectores de direccion} $\vec u$ y $\vec v$ son paralelas si y
sólo si $\det(\vec u, \vec v) = 0$. En forma \omterm{def:g10:algebra:equation}{ecuación}: $d : ax + by + c =
0$ y $d' : a'x + b'y + c' = 0$ son paralelos si y sólo si
$ab' - a'b = 0$. Las rectas no paralelas se encuentran exactamente en un punto.
\end{proposition}

\begin{proof}
Paralelismo significa direcciones \omterm{def:g11:vect:collinear}{colineal}, que es
\cref{thm:g11:vect:collinearity}; con $\vec u\,(-b, a)$ y
$\vec v\,(-b', a')$, el \omterm{def:g11:vect:det}{determinante} es $(-b)a' - (-b')a = ab' - a'b$.
Si las rectas no son paralelas, resolviendo las dos \omterm{def:g10:algebra:equation}{ecuaciones} simultáneamente
(por sustitución o combinación) conduce a una solución única: el sistema
se comporta como un sistema \omterm{def:g11:vect:det}{determinante distinto de cero} $2 \times 2$.
\end{proof}

\begin{method}[Intersección de dos líneas]\label{met:g11:vect:intersection}
Resuelve el sistema de los dos \omterm{def:g10:algebra:equation}{ecuaciones}: aísla una incógnita en un \omterm{def:g10:algebra:equation}{ecuación}
y sustituir por el otro, o combinar el \omterm{def:g10:algebra:equation}{ecuaciones} para eliminar uno
desconocido. Verifique siempre el punto resultante en \emph{ambos} \omterm{def:g10:algebra:equation}{ecuaciones}.
\end{method}

\begin{example}\label{ex:g11:vect:intersection}
Intersección $d: x - 3y + 5 = 0$ y $d': 2x + y - 4 = 0$. De $d'$:
$y = 4 - 2x$. Sustituyendo en $d$:
\[
x - 3(4 - 2x) + 5 = 0
\iff x - 12 + 6x + 5 = 0
\iff 7x = 7
\iff x = 1,
\]
luego $y = 2$. Las líneas se encuentran en $(1, 2)$. Registro en $d$: $1 - 6 + 5 = 0$.
\end{example}

\section{Problema: rumbos de colisión y coordenadas inteligentes}

\begin{problem}[{Problema de fin de semana --- cruzar caminos no es
colisión: aproximación más cercana en el mar y teoremas demostrados por
elegir el marco correcto}]
\label{pb:g11:vect:1}
Los rumbos rectos de dos barcos se cruzan. ¿Deben chocar los barcos? no
a menos que lleguen al cruce \emph{al mismo tiempo}: los caminos están
conjuntos de puntos, los movimientos son paramétricos y confundirlos ha
barcos reales hundidos. Este problema gestiona un servicio de tráfico de embarcaciones pequeñas.
en líneas paramétricas (\cref{def:g11:vect:parametric}), luego gira
a pura geometría con el otro superpoder de \omterm{def:g10:coordgeom:system}{coordenadas}:
\emph{eligiendo} ellos hábilmente, como en \cref{exo:g11:vect:11}, para
hacer que los teoremas clásicos se calculen solos.

\medskip
\noindent\textbf{Parte I --- Lines, three ways.}
\begin{enumerate}
  \item Dar una representación paramétrica de la línea que pasa
        $A(1, 2)$ con \omterm{def:g11:vect:direction}{vector de direccion} $\vec u(3, -1)$, luego
        elimine el parámetro para obtener un cartesiano \omterm{def:g10:algebra:equation}{ecuación}.
  \item Para la línea $2x - 5y + 3 = 0$: lea \omterm{def:g11:vect:direction}{dirección
        vector} (\cref{thm:g11:vect:cartesian}), encuentre un punto,
        y escribe una representación paramétrica.
  \item Calcular el punto \omterm{def:g10:numbers:interunion}{intersección} de las dos líneas de
        preguntas 1 y 2 (\cref{met:g11:vect:intersection}).
  \item Confirmar con el \omterm{def:g11:vect:det}{determinante}
        (\cref{thm:g11:vect:collinearity}) que esas dos líneas
        tuvo que cruzarse; luego muestre que $2x - 5y + 3 = 0$ y

\end{enumerate}
