\chapter{Funciones y variaciones}\label{ch:g11:func}

Este capítulo organiza lo que uno necesita saber sobre \omterm{def:g11:func:function}{funciones} antes
cálculo: la pequeña familia de \emph{\omterm{def:g11:func:reference}{funciones de referencia}} cuyo \omterm{def:g10:functions:graph}{graficos} debe
saber de memoria el significado preciso de \emph{\omterm{def:g11:func:monotone}{creciente}} y
\emph{\omterm{def:g10:functions:variations}{decreciente}} y cómo se comportan las variaciones cuando se combinan \omterm{def:g11:func:function}{funciones}.

\section{Funciones y curvas}

\begin{definition}[Función, dominio, curva]\label{def:g11:func:function}
A \emph{función}\index{función} $f$ asigna a cada número $x$ de un conjunto
$D \subseteq \R$ (es \emph{dominio}\index{dominio}) exactamente un número
$f(x)$. Su \emph{curva} (o \omterm{def:g10:functions:graph}{gráfico}) es el conjunto de puntos
$\bigl(x, f(x)\bigr)$ para $x \in D$.
\end{definition}

\begin{example}
La fórmula $f(x) = \frac{\sqrt{x+2}}{x-1}$ requiere $x + 2 \geq 0$ (para
la raíz cuadrada) y $x \neq 1$ (denominador distinto de cero): el \omterm{def:g11:func:function}{dominio} es
$\intco{-2}{1} \cup \intoo{1}{+\infty}$.
\end{example}

\section{Variaciones en un intervalo}

\begin{definition}[Creciente, decreciente]\label{def:g11:func:monotone}
Dejemos que $f$ se defina en un \omterm{def:g10:numbers:interval}{intervalo} $I$. El \omterm{def:g11:func:function}{función} $f$ es
\emph{creciente}\index{función creciente} en $I$ si conserva el orden:
\[
\text{for all } u, v \in I: \quad u < v \implies f(u) \leq f(v),
\]
y \emph{\omterm{def:g10:functions:variations}{decreciente}} si invierte el orden ($u < v \implies f(u) \geq
f(v)$). Con desigualdades estrictas a la derecha, $f$ es \emph{estrictamente}
aumentando (resp.\ \omterm{def:g10:functions:variations}{decreciente}). Un \omterm{def:g11:func:function}{función} que está aumentando o
\omterm{def:g10:functions:variations}{decreciente} en $I$ es \emph{monótono} en $I$.
\end{definition}

\begin{method}[Demostrando una variación de la definición]\label{met:g11:func:variation}
Tome dos números arbitrarios $u < v$ en \omterm{def:g10:numbers:interval}{intervalo} y determine el signo
de la diferencia $f(v) - f(u)$, generalmente por \emph{\omterm{def:g10:algebra:expand}{factorización}}. si el
El signo es constante en el \omterm{def:g10:numbers:interval}{intervalo}, concluye.
\end{method}

\begin{proposition}[Variaciones de las funciones de referencia.]\label{prop:g11:func:refvariations}
\begin{enumerate}
  \item $x^2$ es estrictamente \omterm{def:g10:functions:variations}{decreciente} en $\intoc{-\infty}{0}$ y estrictamente
        \omterm{def:g11:func:monotone}{creciente} en $\intco{0}{+\infty}$.
  \item $x^3$ es estrictamente \omterm{def:g11:func:monotone}{creciente} en $\R$.
  \item $\sqrt x$ es estrictamente \omterm{def:g11:func:monotone}{creciente} en $\intco{0}{+\infty}$.
  \item $\frac1x$ es estrictamente \omterm{def:g10:functions:variations}{decreciente} en $\intoo{-\infty}{0}$ y en
        $\intoo{0}{+\infty}$ (pero \emph{no} en su \omterm{def:g10:numbers:interunion}{unión}).
  \item $\abs x$ es estrictamente \omterm{def:g10:functions:variations}{decreciente} en $\intoc{-\infty}{0}$ y
        estrictamente \omterm{def:g11:func:monotone}{creciente} en $\intco{0}{+\infty}$.
\end{enumerate}
\end{proposition}

\begin{proof}
Deje $u < v$. Factorizamos $f(v) - f(u)$ en cada caso.

\emph{1.}$v^2 - u^2 = (v-u)(v+u)$. En $\intco{0}{+\infty}$: $v - u > 0$
y $v + u > 0$ (como $v > u \geq 0$), entonces $v^2 > u^2$. encendido
$\intoc{-\infty}{0}$: $v + u < 0$, entonces $v^2 < u^2$.

\emph{2.} $v^3 - u^3 = (v - u)(v^2 + uv + u^2)$, y el segundo factor
es igual a $\left(v + \frac u2\right)^2 + \frac{3u^2}{4} > 0$ a menos que
$u = v = 0$. Por lo tanto $v^3 > u^3$ siempre que $v > u$.

\emph{3.} Para $0 \leq u < v$:
$\sqrt v - \sqrt u = \frac{(\sqrt v - \sqrt u)(\sqrt v + \sqrt u)}
{\sqrt v + \sqrt u} = \frac{v - u}{\sqrt v + \sqrt u} > 0$.

\emph{4.} Para $0 < u < v$:
$\frac1v - \frac1u = \frac{u - v}{uv} < 0$ desde $u - v < 0$ y $uv > 0$;
Mismo cálculo para $u < v < 0$. En el \omterm{def:g10:numbers:interunion}{unión} el reclamo falla:
$-1 < 1$ pero $\frac{1}{-1} = -1 < 1 = \frac11$.

\emph{5.} En $\intco{0}{+\infty}$, $\abs x = x$ es estrictamente \omterm{def:g11:func:monotone}{creciente}; en
$\intoc{-\infty}{0}$, $\abs x = -x$ es estrictamente \omterm{def:g10:functions:variations}{decreciente}.
\end{proof}

\begin{remark}
El punto 4 es una trampa clásica: ``\omterm{def:g10:functions:variations}{decreciente} en $\intoo{-\infty}{0}$ y en
$\intoo{0}{+\infty}$'' es \emph{no} igual que ``\omterm{def:g10:functions:variations}{decreciente} en
$\R \setminus \{0\}$''. Las variaciones sólo tienen sentido en \omterm{def:g10:numbers:interval}{intervalos}.
\end{remark}

\begin{example}[Comparando sin computadora]
Dado que $\sqrt x$ es estrictamente \omterm{def:g11:func:monotone}{creciente}, $\sqrt{7} < \sqrt{8}$; desde
$x^2$ es estrictamente \omterm{def:g10:functions:variations}{decreciente} en $\intoc{-\infty}{0}$,
$(-3)^2 > (-2)^2$. La monotonicidad convierte las comparaciones de \omterm{def:g10:functions:function}{imágenes} en
comparaciones de argumentos.
\end{example}

\section{Nuevas funciones de las antiguas}

\begin{proposition}[Transformaciones]\label{prop:g11:func:transforms}
Sea $u$ un \omterm{def:g11:func:function}{función} definido en un \omterm{def:g10:numbers:interval}{intervalo} $I$ y $k \in \R$.
\begin{enumerate}
  \item $u + k$ tiene las variaciones \emph{mismo} como $u$ en $I$; su curva es
        el de $u$ se desplazó verticalmente en $k$.
  \item Para $\lambda > 0$, $\lambda u$ tiene las mismas variaciones que $u$; para
        $\lambda < 0$, las variaciones se invierten.
  \item Si $u \geq 0$ en $I$, entonces $\sqrt u$ tiene las mismas variaciones que
        $u$.
  \item Si $u \neq 0$ y $u$ tienen signo constante en $I$, entonces $\frac1u$ tiene
        las variaciones \emph{opuesto} a las de $u$.
\end{enumerate}
\end{proposition}

\begin{proof}
Dejemos $u < v$ en $I$ (abusando de la notación, comparamos $x$-valores $u < v$; escribimos
$s = u(x_1)$, $t = u(x_2)$ para el \omterm{def:g10:functions:function}{imágenes} de $x_1 < x_2$).

\emph{1.} $(u+k)(x_2) - (u+k)(x_1) = t - s$: la diferencia de \omterm{def:g10:functions:function}{imágenes} es
no cambia, por lo que su signo no cambia.

\emph{2.} $\lambda t - \lambda s = \lambda(t - s)$: el signo se conserva si
$\lambda > 0$, volteado si $\lambda < 0$.

\emph{3.} Si $u$ es \omterm{def:g11:func:monotone}{creciente} y $x_1 < x_2$, entonces $0 \leq s \leq t$, y
$\sqrt{}$ siendo \omterm{def:g11:func:monotone}{creciente} (\cref{prop:g11:func:refvariations}) da
$\sqrt s \leq \sqrt t$. Mismo argumento en el caso \omterm{def:g10:functions:variations}{decreciente}.

\emph{4.} Si $u$ es \omterm{def:g11:func:monotone}{creciente}, $0 < s \leq t$ (o $s \leq t < 0$), entonces
$\frac1s \geq \frac1t$ porque $x \mapsto \frac1x$ es \omterm{def:g10:functions:variations}{decreciente} en cada
lado de $0$: el orden se invierte.
\end{proof}

\begin{example}\label{ex:g11:func:composite}
Estudio $f(x) = \sqrt{x^2 + 1}$ sobre $\R$. El interior \omterm{def:g11:func:function}{función}
$u(x) = x^2 + 1$ es positivo, \omterm{def:g10:functions:variations}{decreciente} en $\intoc{-\infty}{0}$ y
\omterm{def:g11:func:monotone}{creciente} en $\intco{0}{+\infty}$ (un cuadrado desplazado). Por el punto 3, $f$
tiene las mismas variaciones: \omterm{def:g10:functions:variations}{decreciente} luego \omterm{def:g11:func:monotone}{creciente}, con \omterm{def:g10:functions:extrema}{mínimo}
$f(0) = 1$.
\end{example}

\section{Funciones pares e impares}

\begin{definition}[Paridad]\label{def:g11:func:parity}
Dejemos que $f$ se defina en un \omterm{def:g11:func:function}{dominio} $D$ simétrico con respecto a $0$ (es decir, \ $-x \in D$
siempre que sea $x \in D$). El \omterm{def:g11:func:function}{función} $f$ es \emph{incluso}\index{incluso función}
si $f(-x) = f(x)$ para todos los $x \in D$, y \emph{extraño}\index{función impar} si
$f(-x) = -f(x)$ para todos los $x \in D$.
\end{definition}

\begin{proposition}[Significado geométrico]\label{prop:g11:func:paritysym}
La curva de un \omterm{def:g11:func:parity}{incluso función} es simétrica con respecto al eje $y$; la curva
de un \omterm{def:g11:func:parity}{función impar} es simétrico con respecto al origen.
\end{proposition}

\begin{proof}
Si $f$ es \omterm{def:g11:func:parity}{incluso} y $(x, y)$ está en la curva (por lo tanto, $y = f(x)$), entonces
$f(-x) = f(x) = y$: el punto especular $(-x, y)$ también está en la curva. si
$f$ es \omterm{def:g11:func:parity}{extraño}, $f(-x) = -y$: el punto $(-x, -y)$, simétrico de $(x,y)$ aproximadamente
el origen, está en la curva.
\end{proof}

\begin{example}
$x^2$ y $\abs x$ son \omterm{def:g11:func:parity}{incluso}; $x^3$ y $\frac1x$ son \omterm{def:g11:func:parity}{extraño};
$f(x) = x^2 + x$ no es: $f(-1) = 0$ mientras que $f(1) = 2$, entonces
$f(-1) \neq \pm f(1)$. La paridad reduce el trabajo a la mitad: un \omterm{def:g11:func:parity}{incluso} o un \omterm{def:g11:func:parity}{función impar}
solo necesita ser estudiado en $\intco{0}{+\infty}$.
\end{example}

\begin{exercise}[$\star\star$]\label{exo:g11:func:10}
verdadero o falso (justifique o dé un contraejemplo):
\begin{enumerate}
  \item La suma de dos \omterm{def:g11:func:monotone}{funciones crecientes} en $I$ es \omterm{def:g11:func:monotone}{creciente} en $I$.
  \item El producto de dos \omterm{def:g11:func:monotone}{funciones crecientes} en $I$ es \omterm{def:g11:func:monotone}{creciente} en

  \item Si $f$ es \omterm{def:g11:func:monotone}{creciente} en $I$, entonces $f^2$ ($= f \times f$) es

\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:func:11}
Deje $f(x) = x + \frac{1}{x}$ por $x > 0$.
\begin{enumerate}
  \item Muestre que para $0 < u < v$,
        $f(v) - f(u) = (v - u)\,\frac{uv - 1}{uv}$.
  \item Deduzca que $f$ es estrictamente \omterm{def:g10:functions:variations}{decreciente} en $\intoc{0}{1}$ y
        estrictamente \omterm{def:g11:func:monotone}{creciente} en $\intco{1}{+\infty}$.
  \item Concluir que $x + \frac1x \geq 2$ para todos $x > 0$, con igualdad
        sólo en $x = 1$.
\end{enumerate}
\end{exercise}

\section{Problema: la fábrica de simetría}

\begin{problem}[{Problema de fin de semana --- cada función es par
parte más una parte impar: reglas de paridad, el juego de transformación y
mínimos ganados sin cálculo}]
\label{pb:g11:func:1}
Algunos \omterm{def:g11:func:function}{funciones} son simétricos en espejo, otros son de media vuelta.
simétricos, la mayoría no lo son, pero es un pequeño milagro algebraico
dice que \emph{cada} \omterm{def:g11:func:function}{función} es, exactamente de una manera, la suma
de una pieza simétrica espejo y una pieza simétrica de media vuelta. esto
problema demuestra el milagro, juega el juego de la transformación en el
curvas de referencia (\cref{prop:g11:func:transforms}), y cierra
con la técnica \omterm{def:g10:algebra:expand}{factorización de diferencias} de
\cref{exo:g11:func:11}, que gana los problemas de minimización por completo
capítulo antes de que llegue el derivado.

\medskip
\noindent\textbf{Parte I --- Parity fluency.}
\begin{enumerate}
  \item \omterm{def:g11:func:parity}{Incluso}, \omterm{def:g11:func:parity}{extraño} o ninguno (\cref{def:g11:func:parity})?
        Justifique cada uno: $x^4 - 3x^2$; $x^3 - x$; $x^2 + x$; \
        $\abs x$; $\frac1x$.
  \item Indique el significado geométrico de cada veredicto.
        (\cref{prop:g11:func:paritysym}): ¿qué simetría hace?
        ¿Qué curva tiene un \omterm{def:g11:func:parity}{incluso función}? ¿De un \omterm{def:g11:func:parity}{extraño}?
  \item La paridad multiplica como signos: demuestra que el producto de
        dos \omterm{def:g11:func:parity}{funciones impares} es \omterm{def:g11:func:parity}{incluso}, y completa el conjunto
        tabla de multiplicar (\omterm{def:g11:func:parity}{incluso} $\times$ \omterm{def:g11:func:parity}{incluso}, \omterm{def:g11:func:parity}{incluso} $\times$
        \omterm{def:g11:func:parity}{extraño}, \omterm{def:g11:func:parity}{extraño} $\times$ \omterm{def:g11:func:parity}{extraño}).
  \item Consecuencias rápidas: si $f$ es \omterm{def:g11:func:parity}{extraño} y se define en $0$,
        ¿Qué es $f(0)$? Si $f$ es \omterm{def:g11:func:parity}{incluso} con $f(3) = 7$, ¿cuál es
        $f(-3)$? Y demuestre que el único \omterm{def:g11:func:function}{función} tanto \omterm{def:g11:func:parity}{incluso} como
        \omterm{def:g11:func:parity}{extraño} es el cero \omterm{def:g11:func:function}{función}.
  \item ¿Qué parte de la curva de \omterm{def:g11:func:parity}{incluso función} determina todos los
        eso? Proporcione un \omterm{def:g11:func:function}{función} que no sea ni \omterm{def:g11:func:parity}{incluso} ni \omterm{def:g11:func:parity}{extraño}, y
        explica lo que falla.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The decomposition theorem.}
Para un \omterm{def:g11:func:function}{función} $f$ definido en un \omterm{def:g11:func:function}{dominio} simétrico respecto a $0$, establezca
\[
E(x) = \frac{f(x) + f(-x)}{2},
\qquad
O(x) = \frac{f(x) - f(-x)}{2} .
\]
\begin{enumerate}[resume]
  \item Calcular $E$ y $O$ para $f(x) = x^3 + x^2 + 1$, y
        comprobar: $E$ \omterm{def:g11:func:parity}{incluso}, $O$ \omterm{def:g11:func:parity}{extraño}, $E + O = f$.
  \item Demuestre la afirmación general: para cualquier $f$, el \omterm{def:g11:func:function}{función}
        $E$ es \omterm{def:g11:func:parity}{incluso}, $O$ es \omterm{def:g11:func:parity}{extraño} y $f = E + O$ --- \emph{every
        \omterm{def:g11:func:function}{función} splits into an \omterm{def:g11:func:parity}{incluso} part and an \omterm{def:g11:func:parity}{extraño} part}.
  \item Demostrar que la división es única: si
        $f = E_1 + O_1 = E_2 + O_2$ con $E_i$ \omterm{def:g11:func:parity}{incluso} y $O_i$
        \omterm{def:g11:func:parity}{extraño}, utilice la pregunta 4 en $E_1 - E_2$.
  \item Descomponer $f(x) = \dfrac{1}{1 - x}$ (para
        $x \neq \pm 1$): ponga $E$ y $O$ sobre el común
        denominador $1 - x^2$ y simplificar.
  \item Un vistazo al futuro, en una frase cada uno: el \omterm{def:g11:func:parity}{incluso} y el \omterm{def:g11:func:parity}{extraño}
        partes del exponencial \omterm{def:g11:func:function}{función} serán el gemelo
        \omterm{def:g11:func:function}{funciones} de grado 12 (coseno y seno hiperbólicos); y
        el mismo instinto --- dividir un objeto en simétrico
        piezas y tratar cada una con su propia simetría --- poderes
        Análisis de sonidos y señales de los volúmenes universitarios.
        ¿Por qué hay tal división \emph{útil}, en general?
\end{enumerate}

\noindent\textbf{Parte III --- The transformation game.}
\begin{enumerate}[resume]
  \item Partiendo de la curva de $x \mapsto x^2$, describa
        precisamente las curvas de $(x - 3)^2$, $x^2 + 2$,\
        $(x - 3)^2 + 2$, $-x^2$,$2x^2$
        (\cref{prop:g11:func:transforms}).
  \item Reducir $f(x) = \abs{x - 2} + \abs{x + 2}$ a un
        fórmula por partes en los tres \omterm{def:g10:numbers:interval}{intervalos} cortados por $-2$
        y $2$, y describir su curva (tiene un famoso plano
        abajo).
  \item ¿Qué transformaciones respetan la uniformidad? decidir, con
        prueba o contraejemplo: desplazamiento horizontal
        $f(x - 3)$; desplazamiento vertical $f(x) + 2$; estiramiento vertical
        $2 f(x)$ (todo aplicado a un \omterm{def:g11:func:parity}{incluso} $f$).
  \item Resuelva $\abs{x - 2} + \abs{x + 2} = 6$ usando el
        Fórmula por partes de la pregunta 12.
  \item Ingeniería inversa: un \omterm{def:g11:quad:parabola}{parábola} tiene el vértice $(2, -3)$
        y pasa por $(0, 1)$. Reconstruir su \omterm{def:g10:algebra:equation}{ecuación} en
        \omterm{thm:g11:quad:canonical}{forma canónica}.
\end{enumerate}

\noindent\textbf{Parte IV --- Minima without calculus.}
\begin{enumerate}[resume]
  \item Espejo \cref{exo:g11:func:11} para
        $g(x) = x + \dfrac4x$ en $\intoo{0}{+\infty}$: mostrar
        $g(v) - g(u) = (v - u)\,\dfrac{uv - 4}{uv}$, localice el
        \omterm{def:g10:functions:extrema}{mínimo}, y dar su valor.
  \item Aplicar: entre todos los rectángulos del área $16$, que tiene el
        ¿Mínimo perímetro? (Escriba $P = 2\left(x +
        \frac{16}{x}\right)$ y utilice el contestador de la pregunta 16).
        concluye un viejo amigo.
  \item Determinar las variaciones de
        $h(x) = \sqrt{x^2 + 1}$ en todo $\R$, combinando un
        argumento de composición en $\intco{0}{+\infty}$ con un
        argumento de paridad para el resto.
  \item El punto más cercano: encuentra el punto de la curva.
        $y = \sqrt x$ más cercano a $(3, 0)$. (Minimizar el
        Distancia \emph{al cuadrado} --- una cuadrática en $x$ --- y
        explique por qué minimizar el cuadrado es legítimo).
  \item Final, el recorrido por la fábrica: las curvas de referencia son las crudas
        valores; transforma las máquinas; paridad y la
        descomposición del control de calidad; \omterm{def:g10:algebra:expand}{factorización de diferencias}
        el banco de medición. Una oración en cada --- y nombre
        la herramienta eléctrica que ofrece el siguiente capítulo.
\end{enumerate}
