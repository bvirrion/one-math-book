\chapter{Trigonometría: el círculo unitario}\label{ch:g11:trigo}

La trigonometría comienza en triángulos rectángulos, pero su verdadero hogar es el
\emph{círculo unitario}, donde \omterm{def:g11:trigo:cossin}{coseno} y \omterm{def:g11:trigo:cossin}{seno} se convierten en \omterm{def:g11:func:function}{funciones} de un valor arbitrario.
\omterm{def:g10:numbers:sets}{numero real}. Este capítulo instala el \omterm{def:g11:trigo:radian}{radián}, el devanado del real
línea alrededor del círculo, los valores exactos que debe saber de memoria y el
simetrías que generan todas las identidades clásicas. el estudio de
$\cos$ y $\sin$ como \omterm{def:g11:func:function}{funciones} --- variaciones, \omterm{def:g11:deriv:derivative}{derivados} --- se lleva
en \cref{ch:g12:trigo}.

\section{Radianes y el devanado}

\begin{definition}[Radián]\label{def:g11:trigo:radian}
Sea $\mathcal C$ el círculo de radio $1$ centrado en el origen $O$,
y $I(1, 0)$. La medida \emph{radián}\index{radián} de un ángulo en el
El centro de $\mathcal C$ es la longitud del arco que corta en $\mathcal C$.
Como el círculo completo tiene circunferencia $2\pi$:
\[
360^\circ = 2\pi \text{ rad},
\qquad
180^\circ = \pi \text{ rad},
\qquad
1 \text{ rad} = \frac{180^\circ}{\pi} \approx 57.3^\circ .
\]
\end{definition}

\begin{method}[Conversión]\label{met:g11:trigo:convert}
Los grados y \omterm{def:g11:trigo:radian}{radianes} son proporcionales: multiplica por $\frac{\pi}{180}$ para ir
desde grados hasta \omterm{def:g11:trigo:radian}{radianes}, por $\frac{180}{\pi}$ al revés. por ejemplo
$60^\circ = 60 \times \frac{\pi}{180} = \frac{\pi}{3}$, y
$\frac{3\pi}{4} = \frac{3 \times 180^\circ}{4} = 135^\circ$.
\end{method}

\begin{definition}[Inscribiendo la línea en el círculo]\label{def:g11:trigo:winding}
A cada \omterm{def:g10:numbers:sets}{numero real} $t$ asociar el punto $M(t)$ de $\mathcal C$
obtenido caminando una distancia $\abs t$ a lo largo del círculo desde $I$,
en sentido antihorario si $t \geq 0$, en sentido horario si $t < 0$. desde el
la circunferencia es $2\pi$, los números $t$ y $t + 2k\pi$ ($k \in \Z$)
llegar al mismo punto: $M(t + 2k\pi) = M(t)$.
\end{definition}

\begin{example}
$M(0) = I$; $M\!\left(\frac{\pi}{2}\right) = (0, 1)$, la parte superior del
círculo; $M(\pi) = (-1, 0)$; $M\!\left(-\frac{\pi}{2}\right) = (0, -1)$;
y $M\!\left(\frac{9\pi}{4}\right) = M\!\left(\frac{\pi}{4} +
2\pi\right) = M\!\left(\frac{\pi}{4}\right)$.
\end{example}

\section{Coseno y seno}

\begin{definition}[Coseño y seno]\label{def:g11:trigo:cossin}
Por cada $t$ real, el \emph{coseno}\index{coseno} y
\emph{seno}\index{seno} de $t$ son los \omterm{def:g10:coordgeom:system}{coordenadas} del punto $M(t)$ de
el círculo unitario:
\[
M(t) = (\cos t,\ \sin t).
\]
\end{definition}

\begin{proposition}[Primeras propiedades]\label{prop:g11:trigo:first}
Para todos los $t \in \R$ y $k \in \Z$:
\[
-1 \leq \cos t \leq 1, \qquad
-1 \leq \sin t \leq 1, \qquad
\cos^2 t + \sin^2 t = 1,
\]
\[
\cos(t + 2k\pi) = \cos t, \qquad \sin(t + 2k\pi) = \sin t .
\]
\end{proposition}

\begin{proof}
$M(t)$ se encuentra en el círculo unitario, por lo que sus \omterm{def:g10:coordgeom:system}{coordenadas} están entre $-1$ y
$1$, y satisfacen el \omterm{def:g10:algebra:equation}{ecuación} $x^2 + y^2 = 1$ del círculo, que es el
identidad $\cos^2 t + \sin^2 t = 1$. Replanteos de periodicidad
$M(t + 2k\pi) = M(t)$ (\cref{def:g11:trigo:winding}).
\end{proof}

\begin{example}\label{ex:g11:trigo:identity}
Si $\sin t = \frac35$ y $t \in \intcc{\frac\pi2}{\pi}$ (segundo
cuadrante), luego $\cos^2 t = 1 - \frac{9}{25} = \frac{16}{25}$, entonces
$\cos t = \pm\frac45$; en el segundo cuadrante el \omterm{def:g10:coordgeom:system}{abscisa} es negativo,
por lo tanto $\cos t = -\frac45$.
\end{example}

Los valores a saber de memoria:
\begin{center}
\begin{tabular}{c|ccccc}
$t$ & $0$ & $\dfrac{\pi}{6}$ & $\dfrac{\pi}{4}$ & $\dfrac{\pi}{3}$ & $\dfrac{\pi}{2}$\\[6pt]
\hline\rule{0pt}{14pt}%
$\cos t$ & $1$ & $\dfrac{\sqrt3}{2}$ & $\dfrac{\sqrt2}{2}$ & $\dfrac12$ & $0$\\[6pt]
$\sin t$ & $0$ & $\dfrac12$ & $\dfrac{\sqrt2}{2}$ & $\dfrac{\sqrt3}{2}$ & $1$
\end{tabular}
\end{center}

\begin{remark}
Una ayuda para la memoria: la lectura \omterm{def:g11:trigo:cossin}{senos}
$\frac{\sqrt0}{2}, \frac{\sqrt1}{2}, \frac{\sqrt2}{2}, \frac{\sqrt3}{2},
\frac{\sqrt4}{2}$ y \omterm{def:g11:trigo:cossin}{cosenos} son la misma lista invertida.
\end{remark}

\section{Ángulos asociados}

\begin{proposition}[Ángulos asociados]\label{prop:g11:trigo:associated}
\index{ángulos asociados}
Para todos los $t \in \R$:
\[
\begin{aligned}
\cos(-t) &= \cos t, & \sin(-t) &= -\sin t,\\
\cos(\pi - t) &= -\cos t, & \sin(\pi - t) &= \sin t,\\
\cos(\pi + t) &= -\cos t, & \sin(\pi + t) &= -\sin t,\\
\cos\left(\tfrac{\pi}{2} - t\right) &= \sin t, &
\sin\left(\tfrac{\pi}{2} - t\right) &= \cos t .
\end{aligned}
\]
\end{proposition}

\begin{proof}
Cada línea expresa una simetría del círculo.

$M(-t)$ es el reflejo de $M(t)$ en el eje $x$: el \omterm{def:g10:coordgeom:system}{abscisa} es
sin cambios, el \omterm{def:g10:coordgeom:system}{ordenada} cambia de signo.

$M(\pi - t)$ es el reflejo de $M(t)$ en el eje $y$: caminar
hacia atrás desde $\pi$ por $t$ aterriza en sentido opuesto (horizontalmente) a caminar
reenvía desde $0$ por $t$. El \omterm{def:g10:coordgeom:system}{abscisa} cambia de signo, el \omterm{def:g10:coordgeom:system}{ordenada} cambia de signo.
sin cambios.

$M(\pi + t)$ está diametralmente opuesto a $M(t)$ (media vuelta): ambos
\omterm{def:g10:coordgeom:system}{coordenadas} cambio de signo.

$M\!\left(\frac\pi2 - t\right)$ es el reflejo de $M(t)$ en el
línea diagonal $y = x$, que intercambia los dos \omterm{def:g10:coordgeom:system}{coordenadas}.
\end{proof}

\begin{example}
$\cos\frac{2\pi}{3} = \cos\left(\pi - \frac\pi3\right)
= -\cos\frac\pi3 = -\frac12$, y
$\sin\frac{7\pi}{6} = \sin\left(\pi + \frac\pi6\right)
= -\sin\frac\pi6 = -\frac12$. La tabla de cinco valores, ampliada por el
simetrías, cubre todo el círculo.
\end{example}

\section{Resolver ecuaciones trigonométricas}

\begin{theorem}[Las ecuaciones $\cos t = \cos a$ y $\sin t = \sin a$]
\label{thm:g11:trigo:equations}
Para \omterm{def:g10:numbers:sets}{números reales} $t$ y $a$:
\[
\cos t = \cos a \iff t = a + 2k\pi \ \text{or}\ t = -a + 2k\pi
\quad (k \in \Z),
\]
\[
\sin t = \sin a \iff t = a + 2k\pi \ \text{or}\ t = \pi - a + 2k\pi
\quad (k \in \Z).
\]
\end{theorem}

\begin{proof}
Dos puntos del círculo unitario tienen el mismo \omterm{def:g10:coordgeom:system}{abscisa} exactamente cuando están
iguales o reflejos entre sí en el eje $x$; por
\cref{prop:g11:trigo:associated}, el reflejo de $M(a)$ es $M(-a)$. entonces
$\cos t = \cos a$ significa $M(t) = M(a)$ o $M(t) = M(-a)$, es decir\
$t = \pm a$ hasta un múltiplo de $2\pi$. Del mismo modo, igual \omterm{def:g10:coordgeom:system}{ordenadas} media
puntos iguales o reflejos en el eje $y$, y el reflejo de $M(a)$
es $M(\pi - a)$.
\end{proof}

\begin{method}[Resolviendo $\cos t = c$ en un intervalo]\label{met:g11:trigo:solve}
\begin{enumerate}
  \item Encuentre el ángulo \emph{uno} $a$ con $\cos a = c$, de la tabla de
        valores conocidos.
  \item Escribe las soluciones generales $t = \pm a + 2k\pi$.
        (\cref{thm:g11:trigo:equations}).
  \item Elija el \omterm{def:g10:numbers:sets}{números enteros} $k$ que aterrice dentro del \omterm{def:g10:numbers:interval}{intervalo} solicitado,
        y enumere las soluciones.
\end{enumerate}
Proceder de la misma manera para $\sin t = c$ con $t = a + 2k\pi$ o
$t = \pi - a + 2k\pi$.
\end{method}

\begin{example}\label{ex:g11:trigo:solve}
Resuelva $\cos t = \frac12$ en $\intoc{-\pi}{\pi}$. Una solución de
$\cos a = \frac12$ es $a = \frac{\pi}{3}$. Las soluciones generales son
$t = \frac\pi3 + 2k\pi$ y $t = -\frac\pi3 + 2k\pi$. Dentro
$\intoc{-\pi}{\pi}$, sólo $k = 0$ aporta:
$t \in \left\{-\frac\pi3,\ \frac\pi3\right\}$.

Resuelva $\sin t = -\frac{\sqrt2}{2}$ en $\intco{0}{2\pi}$. Un ángulo es
$a = -\frac\pi4$; las soluciones son $t = -\frac\pi4 + 2k\pi$ y
$t = \pi + \frac\pi4 + 2k\pi = \frac{5\pi}{4} + 2k\pi$. en
$\intco{0}{2\pi}$: $t = \frac{7\pi}{4}$ (de $k = 1$) y
$t = \frac{5\pi}{4}$.
\end{example}

\section{Problema: La noria y la marea}

\begin{problem}[{Problema del fin de semana --- radianes como distancia recorrida,
y el círculo unitario como reloj maestro de cada periódico.
fenómeno}]
\label{pb:g11:trigo:1}
Una noria te lleva alrededor de un círculo; la marea lleva el
el agua del puerto arriba y abajo en el mismo círculo matemático. cada
fenómeno periódico --- ruedas, mareas, sonido, estaciones --- lee
su horario fuera del círculo unitario de este capítulo
(\cref{def:g11:trigo:cossin}), y el \omterm{def:g10:algebra:equation}{ecuaciones} de
\cref{met:g11:trigo:solve} indica las horas exactas. este problema
convierte, enrolla, modela y resuelve --- y se establece en el camino por qué
Los matemáticos miden ángulos en \omterm{def:g11:trigo:radian}{radianes}.

\medskip
\noindent\textbf{Parte I --- \omterm{def:g11:trigo:radian}{radianes}, the rolled distance.}
\begin{enumerate}
  \item Convertir a \omterm{def:g11:trigo:radian}{radianes}: $30^\circ$, $135^\circ$,
        $300^\circ$; y a grados: $\frac{3\pi}{4}$,
        $\frac{5\pi}{6}$ (\cref{met:g11:trigo:convert}).
  \item El punto completo de \omterm{def:g11:trigo:radian}{radianes}: un arco de ángulo $t$ (en
        \omterm{def:g11:trigo:radian}{radianes}) en un círculo de radio $r$ tiene longitud exactamente
        $r\,t$. ¿Qué arco corta un ángulo de $2$ \omterm{def:g11:trigo:radian}{radianes} en una
        círculo de radio $3$ m? (No hay $\pi$ en ninguna parte --- es decir
        el punto.)
  \item Una rueda de radio $30$ cm rueda $1$ km sin
        resbalando. ¿Por cuántos \emph{\omterm{def:g11:trigo:radian}{radianes}} ha girado?
        y cuantas vueltas completas son?
  \item Colocar en el círculo unitario y evaluar exactamente:
        $\cos\frac{2\pi}{3}$, $\sin\left(-\frac{\pi}{4}\right)$,
        y $\cos\frac{19\pi}{6}$ (reduzca primero el módulo $2\pi$;
        \cref{prop:g11:trigo:associated}).
  \item Un ángulo $t \in \intoo{\frac\pi2}{\pi}$ satisface
        $\sin t = \frac35$. Usando
        $\cos^2 t + \sin^2 t = 1$ y el cuadrante, encuentre
        $\cos t$ y $\tan t$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The Ferris wheel.}
Una rueda gigante tiene radio $60$ m, su cubo $65$ m por encima del
tierra y gira una vez cada $30$ minutos. Usted aborda en el
punto más bajo en el momento $t = 0$ (en minutos), por lo que su altura obedece
\[
h(t) = 65 - 60 \cos\!\left(\frac{\pi t}{15}\right).
\]
\begin{enumerate}[resume]
  \item Justifique la fórmula: verifique el ángulo girado después de $t$
        minutos, la altura en $t = 0$, y explique el signo negativo.
        firmar.
  \item Calcula tu altura en $t = 7.5$, $t = 15$ y
        $t = 20$ minutos (valores exactos).
  \item ¿A qué hora estás \emph{primero} en $95$ m? (Resolver
        $h(t) = 95$ con \cref{met:g11:trigo:solve}.)
  \item ¿Durante qué tiempo \omterm{def:g10:numbers:interval}{intervalo} del viaje estás al menos
        $95$ m de alto --- ¿y por cuánto tiempo en total? (Comparar
        \cref{exo:g11:trigo:11}.)
  \item ¿Dónde cambia la altura \emph{lo más rápido}? comparar el
        subir durante el primer minuto
        ($h(1) - h(0)$) con la subida entre los minutos $7$ y
        $8$, y formular la observación (la herramienta que hace
        es exacto --- llega el \omterm{def:g11:deriv:derivative}{derivado} de la sinusoide ---
        en el grado 12).
  \item El turno del ingeniero: diseñar la fórmula de altura de una rueda
        con altura máxima $40$ m, altura mínima $4$ m y
        periodo $12$ minutos, embarque por la parte inferior a las
        $t = 0$.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The tide and the circle's
symmetries.}
En un puerto, la profundidad del agua en metros, $t$ horas después
medianoche, es
\[
d(t) = 7 + 3 \sin\!\left(\frac{\pi t}{6}\right).
\]
\begin{enumerate}[resume]
  \item Indique las profundidades máxima y mínima, los momentos en que
        aparecen por primera vez y el período de la marea.
  \item Un buque de carga necesita $8.5$ m de agua. resolver
        $d(t) \geq 8.5$ en $\intco{0}{12}$: durante qué ventana
        ¿Puede entrar y cuánto mide la ventana? ¿Cuándo
        siguiente ventana abierta?
  \item Volver a derivar dos identidades de ángulos asociados directamente desde
        simetrías circulares --- $\sin(\pi - t) = \sin t$ (espejo
        a lo largo del eje vertical) y $\cos(\pi + t) = -\cos t$
        (media vuelta) --- y enuncie la identidad complementaria
        $\cos\left(\frac\pi2 - t\right) = \sin t$.
  \item Paridad (vocabulario \cref{pb:g11:func:1}): clasificar
        $\sin$, $\cos$ y $t \mapsto \sin t + t^3$ como par o
        impar, con pruebas de una línea del círculo.
  \item Resolver en $\intco{0}{2\pi}$:
        $2\sin^2 t - \sin t - 1 = 0$. (Una cuadrática en
        disfraz: factorícelo como en \cref{pb:g11:quad:1}, luego
        lee el círculo.)
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The mathematician's angle.}
\begin{enumerate}[resume]
  \item ¿Cuántos grados tiene $1$ \omterm{def:g11:trigo:radian}{radián}? Luego una trampa de calculadora:
        que es más grande, $\sin(1)$ (modo \omterm{def:g11:trigo:radian}{radián}) o
        $\sin(1^\circ)$ --- ¿y aproximadamente por qué factor? moraleja
        para la configuración de la calculadora?
  \item Resuelva $\cos t = \sin t$ en $\intco{0}{2\pi}$, usando el
        círculo (¿dónde son iguales \omterm{def:g10:coordgeom:system}{abscisa} y \omterm{def:g10:coordgeom:system}{ordenada}?).
  \item Para ángulos pequeños, el arco y su \omterm{def:g11:trigo:cossin}{seno} casi coinciden:
        comparar $\sin(0.1)$ con $0.1$ (cinco decimales), y
        dar el error relativo. Este "ángulo pequeño"
        \omterm{def:g10:numbers:approx}{aproximación}'' gobierna barcos y péndulos; el teorema
        detrás de él ($\frac{\sin t}{t} \to 1$) hay una estrella del
        Capítulo de límites de grado 12.
  \item Finale --- el círculo unitario como reloj maestro, una frase
        por artículo: \omterm{def:g11:trigo:radian}{radianes} convierte ángulos en números honestos (arco
        $=$ radio $\times$ ángulo); \omterm{def:g11:trigo:cossin}{coseno} y \omterm{def:g11:trigo:cossin}{seno} son los
        sombra y altura de la manecilla del reloj; las simetrias del circulo
        son las identidades; resolviendo $\cos t = c$ tiempos de lectura
        fuera del horario; y todo fenómeno periódico --- rueda,
        marea, sonido --- este reloj lleva un disfraz.
\end{enumerate}
\end{problem}
