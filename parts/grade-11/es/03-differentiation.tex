\chapter{Diferenciación: un primer curso}\label{ch:g11:deriv}

¿Qué tan rápido cambia un \omterm{def:g11:func:function}{función}? La respuesta en un solo punto es la
\emph{\omterm{def:g11:deriv:derivative}{derivado}}, el número más útil de todos los aplicados
matemáticas. Este capítulo lo construye a partir de tasas de cambio promedio,
lo calcula para \omterm{def:g11:func:reference}{funciones de referencia} y usa su signo para leer el
variaciones de un \omterm{def:g11:func:function}{función}. Todo aquí se amplía y profundiza en
\cref{ch:g12:deriv}.

\section{La tasa promedio de cambio}

\begin{definition}[Tasa de cambio promedio]\label{def:g11:deriv:rate}
Dejemos que $f$ se defina en \omterm{def:g10:numbers:interval}{intervalo} $I$ y $a \neq b$ en $I$. el
\emph{tasa de cambio promedio}\index{tasa de cambio promedio} de $f$ entre
$a$ y $b$ son
\[
\frac{f(b) - f(a)}{b - a}.
\]
Es el \omterm{def:g10:reffunc:affine}{pendiente} de la recta \emph{secante}\index{secante} que pasa por los puntos
$\bigl(a, f(a)\bigr)$ y $\bigl(b, f(b)\bigr)$ de la curva.
\end{definition}

\begin{example}
Un automóvil recorre $140$ km entre las 14:00 y las 16:00 horas: su velocidad media es
$\frac{140}{2} = 70$ kilómetros por hora. El velocímetro, sin embargo, muestra una
\emph{instantáneo} velocidad en cada momento --- esa es exactamente la noción
que el \omterm{def:g11:deriv:derivative}{derivado} hace preciso.
\end{example}

\section{La derivada en un punto}

\begin{definition}[Derivada en un punto]\label{def:g11:deriv:derivative}
Dejemos que $f$ se defina en un \omterm{def:g10:numbers:interval}{intervalo} que contiene $a$. Para $h \neq 0$, forme el
\omterm{def:g11:deriv:rate}{tasa de cambio promedio} entre $a$ y $a + h$:
\[
r(h) = \frac{f(a+h) - f(a)}{h}.
\]
Si $r(h)$ se acerca a un único número fijo mientras $h$ se acerca arbitrariamente a
$0$, decimos que $f$ es \emph{diferenciable en $a$}\index{derivado}\index{diferenciable};
este número es el \emph{derivada de $f$ en $a$}, escrito $f'(a)$:
\[
f'(a) = \lim_{h \to 0} \frac{f(a+h) - f(a)}{h}.
\]
\end{definition}

\begin{example}\label{ex:g11:deriv:atpoint}
Sean $f(x) = x^2$ y $a = 1$. Para $h \neq 0$:
\[
\frac{f(1+h) - f(1)}{h}
= \frac{(1+h)^2 - 1}{h}
= \frac{1 + 2h + h^2 - 1}{h}
= \frac{h(2 + h)}{h}
= 2 + h .
\]
A medida que $h$ se aproxima a $0$, esto se aproxima a $2$: el \omterm{def:g11:func:function}{función} $x^2$ es
\omterm{def:g11:deriv:derivative}{diferenciable} en $1$ y $f'(1) = 2$. Tenga en cuenta la estrategia: \emph{simplificar
el cociente hasta que $h$ ya no aparezca en un denominador}, luego deje
$h \to 0$.
\end{example}

\begin{example}[Un punto no diferenciable]
Sean $f(x) = \abs{x}$ y $a = 0$. entonces
$\frac{f(h) - f(0)}{h} = \frac{\abs h}{h}$, que equivale a $1$ para $h > 0$
y $-1$ para $h < 0$: no se acerca a ningún número único, y $\abs x$ no es
\omterm{def:g11:deriv:derivative}{diferenciable} en $0$. Su curva tiene allí un \emph{esquina}.
\end{example}

\section{la recta tangente}

\begin{definition}[tangente]\label{def:g11:deriv:tangent}
Si $f$ es \omterm{def:g11:deriv:derivative}{diferenciable} en $a$, el \emph{tangente}\index{tangente} al
curva en el punto $A = \bigl(a, f(a)\bigr)$ es la línea que pasa por $A$ con
\omterm{def:g10:reffunc:affine}{pendiente} $f'(a)$. Su \omterm{def:g10:algebra:equation}{ecuación} es
\[
y = f(a) + f'(a)(x - a).
\]
\end{definition}

El \omterm{def:g11:deriv:tangent}{tangente} es la posición límite del \omterm{def:g11:deriv:rate}{secantes} al $A$: como
$h \to 0$, el segundo punto $\bigl(a+h, f(a+h)\bigr)$ se desliza a lo largo del
curva hacia $A$, y el \omterm{def:g11:deriv:rate}{secante} \omterm{def:g10:reffunc:affine}{pendiente} $\frac{f(a+h)-f(a)}{h}$
se aproxima a $f'(a)$.

\begin{example}\label{ex:g11:deriv:tangenteq}
Para $f(x) = x^2$ en $a = 1$: $f(1) = 1$ y $f'(1) = 2$
(\cref{ex:g11:deriv:atpoint}), por lo que \omterm{def:g11:deriv:tangent}{tangente} es
\[
y = 1 + 2(x - 1) = 2x - 1 .
\]
Esta es la línea $d_2$ encontrada por álgebra pura en \cref{exo:g11:quad:11}.
\end{example}

\section{Derivadas de las funciones de referencia}

\begin{proposition}\label{prop:g11:deriv:refderivatives}
Sobre su \omterm{def:g11:func:function}{dominios} de diferenciabilidad:
\[
(c)' = 0, \quad
(x)' = 1, \quad
(x^2)' = 2x, \quad
(x^3)' = 3x^2, \quad
\left(\frac1x\right)' = -\frac{1}{x^2}, \quad
(\sqrt x\,)' = \frac{1}{2\sqrt x} \ (x > 0).
\]
De manera más general, $(x^n)' = n\,x^{n-1}$ por cada \omterm{def:g10:numbers:sets}{entero} $n \geq 1$.
\end{proposition}

\begin{proof}
Cada prueba sigue la estrategia de \cref{ex:g11:deriv:atpoint}: simplificar
la tasa, luego deje $h \to 0$. Reparar $a$ en \omterm{def:g11:func:function}{dominio}.

\emph{Constante e identidad.} $\frac{c - c}{h} = 0$ y
$\frac{(a+h) - a}{h} = 1$ para todos los $h$: los límites son $0$ y $1$.

\emph{Cuadrado.} $\frac{(a+h)^2 - a^2}{h} = \frac{2ah + h^2}{h} = 2a + h
\to 2a$.

\emph{Cubo.} $(a+h)^3 = a^3 + 3a^2h + 3ah^2 + h^3$, por lo que la tasa es
$3a^2 + 3ah + h^2 \to 3a^2$.

\emph{Recíproca.} Para $a \neq 0$ y $h$ lo suficientemente pequeño como para
$a + h \neq 0$:
\[
\frac{1}{h}\left(\frac{1}{a+h} - \frac{1}{a}\right)
= \frac{1}{h} \cdot \frac{a - (a+h)}{a(a+h)}
= \frac{-1}{a(a+h)}
\xrightarrow[h \to 0]{} -\frac{1}{a^2}.
\]

\emph{Raíz cuadrada.} Para $a > 0$, multiplica por el conjugado:
\[
\frac{\sqrt{a+h} - \sqrt a}{h}
= \frac{(a + h) - a}{h\,(\sqrt{a+h} + \sqrt a)}
= \frac{1}{\sqrt{a+h} + \sqrt a}
\xrightarrow[h \to 0]{} \frac{1}{2\sqrt a}.
\]

\emph{Poder general.} El patrón $2a$, $3a^2$ continúa: \omterm{def:g10:algebra:expand}{en expansión}
$(a+h)^n$, todos los términos posteriores a $a^n + n\,a^{n-1}h$ llevan $h^2$, por lo que el
la tasa tiende a $n\,a^{n-1}$. (Se da una inducción completa en
\cref{ch:g12:deriv}.)
\end{proof}

\section{Reglas de diferenciación}

\begin{theorem}[Operaciones]\label{thm:g11:deriv:rules}
Sean $u$ y $v$ \omterm{def:g11:deriv:derivative}{diferenciable} en $I$ y $\lambda \in \R$. entonces
$u + v$, $\lambda u$, $uv$ son \omterm{def:g11:deriv:derivative}{diferenciable} en $I$, tal como están
$\frac{u}{v}$ dondequiera que sea $v \neq 0$, y
\[
(u+v)' = u' + v', \qquad
(\lambda u)' = \lambda u', \qquad
(uv)' = u'v + uv', \qquad
\left(\frac{u}{v}\right)' = \frac{u'v - uv'}{v^2}.
\]
\end{theorem}

\begin{proof}[Prueba de las tres primeras reglas]
Corrija $a \in I$ y abrevie $\Delta u = u(a+h) - u(a)$,
$\Delta v = v(a+h) - v(a)$.

\emph{Suma.} La tasa de $u + v$ es
$\frac{\Delta u + \Delta v}{h} = \frac{\Delta u}{h} + \frac{\Delta v}{h}
\to u'(a) + v'(a)$.

\emph{Múltiple.} La tasa de $\lambda u$ es
$\lambda \frac{\Delta u}{h} \to \lambda u'(a)$.

\emph{Producto.} Suma y resta del término mixto $u(a+h)\,v(a)$:
\begin{align*}
\frac{u(a+h)v(a+h) - u(a)v(a)}{h}
&= \frac{u(a+h)v(a+h) - u(a+h)v(a) + u(a+h)v(a) - u(a)v(a)}{h}\\
&= u(a+h)\,\frac{\Delta v}{h} + v(a)\,\frac{\Delta u}{h}.
\end{align*}
Como $h \to 0$: $\frac{\Delta v}{h} \to v'(a)$, $\frac{\Delta u}{h} \to
u'(a)$ y $u(a+h) \to u(a)$ (su tarifa tiene un límite, por lo que
$\Delta u = h \cdot \frac{\Delta u}{h} \to 0$). Por lo tanto, la tasa tiende
a $u(a)v'(a) + v(a)u'(a)$.

La regla del cociente se demuestra del mismo modo; es \emph{admitido en
este nivel} y se prueba en \cref{ch:g12:deriv}.
\end{proof}

\begin{example}\label{ex:g11:deriv:computations}
\emph{Polinomio.} $f(x) = 3x^3 - 5x^2 + 7x - 2$:
\[
f'(x) = 3 \times 3x^2 - 5 \times 2x + 7 = 9x^2 - 10x + 7 .
\]

\emph{Cociente.} $g(x) = \frac{2x+1}{x^2+1}$: con $u = 2x + 1$, $v = x^2
+ 1$,
\[
g'(x) = \frac{2(x^2+1) - (2x+1)\,2x}{(x^2+1)^2}
= \frac{-2x^2 - 2x + 2}{(x^2+1)^2}.
\]

\emph{Producto.} $k(x) = x\sqrt x$ para $x > 0$:
$k'(x) = 1 \times \sqrt x + x \times \frac{1}{2\sqrt x}
= \sqrt x + \frac{\sqrt x}{2} = \frac{3\sqrt x}{2}$.
\end{example}

\section{Signo de la derivada y variaciones.}

\begin{theorem}[Derivada y variaciones]\label{thm:g11:deriv:variations}
Sea $f$ \omterm{def:g11:deriv:derivative}{diferenciable} en un \omterm{def:g10:numbers:interval}{intervalo} $I$.
\begin{enumerate}
  \item Si $f' \geq 0$ en $I$, entonces $f$ es \omterm{def:g11:func:monotone}{creciente} en $I$.
  \item Si $f' \leq 0$ en $I$, entonces $f$ es \omterm{def:g10:functions:variations}{decreciente} en $I$.
  \item Si $f' > 0$ en $I$ (excepto posiblemente en un número finito de puntos donde
        desaparece), entonces $f$ es estrictamente \omterm{def:g11:func:monotone}{creciente} en $I$.
\end{enumerate}
\end{theorem}

\begin{proof}
La intuición es clara: una curva cuyo \omterm{def:g11:deriv:tangent}{tangentes} apunte hacia arriba debe
subir --- pero una demostración rigurosa necesita el teorema del valor \omterm{def:g10:stats:mean}{significar}. El resultado es
\emph{admitido a este nivel}; ver \cref{thm:g12:deriv:variations}.
\end{proof}

\begin{proposition}[extremo]\label{prop:g11:deriv:extrema}
Si $f'$ cambia de signo en $a$ de $+$ a $-$, entonces $f$ tiene un \omterm{def:g10:functions:extrema}{máximo} local.
en $a$; de $-$ a $+$, un \omterm{def:g10:functions:extrema}{mínimo} local. En tal punto, $f'(a) = 0$ y
el \omterm{def:g11:deriv:tangent}{tangente} es horizontal.
\end{proposition}

\begin{proof}
Si $f' \geq 0$ en un \omterm{def:g10:numbers:interval}{intervalo} termina en $a$ y $f' \leq 0$ en un \omterm{def:g10:numbers:interval}{intervalo}
comenzando en $a$, luego $f$ aumenta hasta $f(a)$ y luego disminuye
(\cref{thm:g11:deriv:variations}): $f(a)$ es el más grande en las cercanías. Desde $f'$
cambia de signo en $a$ y se define allí, $f'(a) = 0$.
\end{proof}

\begin{method}[Estudiando una función]\label{met:g11:deriv:study}
\begin{enumerate}
  \item Calcule $f'(x)$ y simplifíquelo --- idealmente en una forma factorizada.
  \item Determine el signo de $f'(x)$ en cada \omterm{def:g10:numbers:interval}{intervalo} (para una ecuación cuadrática
        $f'$, utilice \cref{thm:g11:quad:sign}).
  \item Registre los resultados en un \emph{\omterm{def:g10:functions:table}{tabla de variaciones}}: una fila para el
        signo de $f'$, una fila de flechas para $f$, con los valores de $f$ en
        los puntos de inflexión.
  \item Concluye: extremos, número de soluciones de $f(x) = k$, desigualdades.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g11:deriv:study}
Estudio $f(x) = x^3 - 3x + 1$ sobre $\R$. Primero,
$f'(x) = 3x^2 - 3 = 3(x-1)(x+1)$: una cuadrática con \omterm{def:g11:quad:discriminant}{raíces} $-1$ y $1$,
positivo fuera de ellos. Entonces $f$ es \omterm{def:g11:func:monotone}{creciente} en $\intoc{-\infty}{-1}$,
\omterm{def:g10:functions:variations}{decreciente} en $\intcc{-1}{1}$, \omterm{def:g11:func:monotone}{creciente} en $\intco{1}{+\infty}$, con un
local \omterm{def:g10:functions:extrema}{máximo} $f(-1) = -1 + 3 + 1 = 3$ y un local \omterm{def:g10:functions:extrema}{mínimo}
$f(1) = 1 - 3 + 1 = -1$.
\end{example}

\begin{example}[Optimización]\label{ex:g11:deriv:box}
Se fabrica una caja abierta a partir de una lámina de $12 \times 12$ cm cortando un cuadrado de
lado $x$ desde cada esquina y plegable. Su volumen es
$V(x) = x(12 - 2x)^2$ para $0 < x < 6$. \omterm{def:g10:algebra:expand}{En expansión},
$V(x) = 4x^3 - 48x^2 + 144x$, entonces
\[
V'(x) = 12x^2 - 96x + 144 = 12(x^2 - 8x + 12) = 12(x - 2)(x - 6).
\]
En $\intoo{0}{6}$, el factor $x - 6$ es negativo, por lo que $V'(x)$ tiene el signo
de $-(x-2)$: positivo antes de $2$, negativo después. El volumen es máximo.
para $x = 2$ cm, dando $V(2) = 2 \times 8^2 = 128$ cm$^3$.
\end{example}

\section{Problema: los tres trabajos de la tangente}

\begin{problem}[{Problema del fin de semana --- el espejo parabólico demostró,
La máquina de búsqueda de raíces de Newton y la viga más fuerte del tronco}]
\label{pb:g11:deriv:1}
A \omterm{def:g11:deriv:derivative}{derivado} dibuja \omterm{def:g11:deriv:tangent}{tangentes}; Eso suena como un talento modesto.
Este problema muestra que el \omterm{def:g11:deriv:tangent}{tangente} tiene tres trabajos a la vez:
demuestra el teorema de los faros que quedó pendiente en
\cref{pb:g11:quad:1} (cada rayo paralelo al eje de \omterm{def:g11:quad:parabola}{parábola}
refleja a través del \omterm{pb:g11:quad:1}{enfocar}), alimenta el más rápido
algoritmo de resolución de ecuaciones de uso común --- dentro de cada
calculadora y red neuronal --- y encuentra el más fuerte
viga rectangular que puede producir un tronco redondo.

\medskip
\noindent\textbf{Parte I --- \omterm{def:g11:deriv:tangent}{Tangente} fluency.}
\begin{enumerate}
  \item Diferenciar $f(x) = 3x^2 - 5x + 1$ y
        $g(x) = x^3 - 12x$
        (\cref{thm:g11:deriv:rules}); y recuperar $\left(x^2

\noindent\textbf{Parte IV --- The strongest beam.}
Una viga rectangular de ancho $w$ y profundidad $d$ se corta desde una
tronco redondo de diámetro $30$ cm, por lo que $w^2 + d^2 = 900$. una viga
la rigidez es proporcional a $w d^2$ (la profundidad cuenta dos veces:
¡Las vigas están de canto!).
\begin{enumerate}[resume]
  \item Exprese la rigidez como \omterm{def:g11:func:function}{función} de $w$ solo:
        $S(w) = 900w - w^3$, ¿en qué \omterm{def:g10:numbers:interval}{intervalo}?
  \item Diferenciar, construir el \omterm{def:g10:functions:table}{tabla de variaciones} y encontrar el
        $w$ y $d$ óptimos (valores exactos, luego a la
        milímetro).
  \item Demuestre que las proporciones óptimas satisfacen
        $d = w\sqrt2$ --- la proporción de los viejos carpinteros. (Bonificación
        tradición: dividir el diámetro del tronco en tres partes iguales
        y elevando perpendiculares a las construcciones circulares.
        exactamente este rectángulo.)
  \item La elección ingenua es el haz \emph{cuadrado}
        ($w = d$). Calcule su rigidez y el óptimo.
        viga y dé la ganancia del carpintero en porcentaje.
  \item Finale --- los tres trabajos del \omterm{def:g11:deriv:tangent}{tangente}, una frase cada uno:
