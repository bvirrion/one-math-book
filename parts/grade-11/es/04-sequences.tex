\chapter{Secuencias: un primer curso}\label{ch:g11:seq}

Un \omterm{def:g11:seq:sequence}{secuencia} es una lista de números producida por una regla: los saldos sucesivos
de una cuenta de ahorros, el tamaño de una población año tras año. esto
El capítulo estudia las dos familias que dominan las aplicaciones.
\emph{\omterm{def:g11:seq:arithmetic}{aritmética}} \omterm{def:g11:seq:sequence}{secuencias}, que crecen en pasos iguales, y
\emph{\omterm{def:g11:seq:geometric}{geométrico}} \omterm{def:g11:seq:sequence}{secuencias}, que crecen en proporciones iguales. La teoría rigurosa
de límites se desarrolla en \cref{ch:g12:seq}.

\section{Definiendo una secuencia}

\begin{definition}[Secuencia]\label{def:g11:seq:sequence}
A \emph{secuencia}\index{secuencia} $(u_n)$ asigna a cada \omterm{def:g10:numbers:sets}{entero}
$n \geq 0$ (o $n \geq 1$) a \omterm{def:g10:numbers:sets}{numero real} $u_n$, es \emph{término de índice
$n$}. Se puede dar una secuencia
\begin{itemize}
  \item \emph{explícitamente}, mediante una fórmula para $u_n$ en términos de $n$: por ejemplo\
        $u_n = n^2 + 1$;
  \item \emph{recursivamente}, por su primer término y una regla a pasar de cada uno
        término al siguiente: por ejemplo\ $u_0 = 3$ y $u_{n+1} = 2u_n - 1$.
\end{itemize}
\end{definition}

\begin{example}
Para $u_n = n^2 + 1$: $u_0 = 1$, $u_1 = 2$, $u_2 = 5$ y $u_{10} = 101$
directamente. Para $u_0 = 3$, $u_{n+1} = 2u_n - 1$: $u_1 = 5$, $u_2 = 9$,
$u_3 = 17$ --- cada término necesita del anterior; llegar a $u_{10}$ toma
diez pasos (o una fórmula general, ver \cref{exo:g11:seq:11}).
\end{example}

\section{secuencias aritméticas}

\begin{definition}[Secuencia aritmética]\label{def:g11:seq:arithmetic}
Un \omterm{def:g11:seq:sequence}{secuencia} es \emph{aritmética}\index{secuencia aritmética} con
\emph{diferencia común}\index{diferencia común} $d$ si cada término se obtiene del anterior
uno agregando $d$:
\[
u_{n+1} = u_n + d \quad \text{for all } n.
\]
Equivalentemente: la diferencia $u_{n+1} - u_n$ es constante, igual a $d$.
\end{definition}

\begin{theorem}[término general]\label{thm:g11:seq:arithgeneral}
Si $(u_n)$ es \omterm{def:g11:seq:arithmetic}{aritmética} con el primer término $u_0$ y \omterm{def:g11:seq:arithmetic}{diferencia común} $d$,
entonces
\[
u_n = u_0 + n\,d \quad \text{for all } n \geq 0,
\qquad\text{and more generally } u_n = u_p + (n - p)\,d .
\]
\end{theorem}

\begin{proof}
Para pasar de $u_0$ a $u_n$ se aplica la regla ``añadir $d$'' $n$ veces: una
el paso da $u_1 = u_0 + d$, dos pasos dan $u_2 = u_0 + 2d$ y después $n$
pasos cada aplicación ha contribuido con un $d$, por lo que $u_n = u_0 + nd$.
(Este ``y demás'' se vuelve riguroso por inducción en \cref{ch:g12:seq}.)
La fórmula general sigue contando los pasos $n - p$ desde $u_p$ hasta
$u_n$.
\end{proof}

\begin{theorem}[Suma de enteros consecutivos]\label{thm:g11:seq:intsum}
Por cada \omterm{def:g10:numbers:sets}{entero} $n \geq 1$:
\[
1 + 2 + \dots + n = \frac{n(n+1)}{2}.
\]
De manera más general, una suma de términos consecutivos de un \omterm{def:g11:seq:arithmetic}{secuencia aritmética}
es igual
\[
(\text{number of terms}) \times
\frac{\text{first term} + \text{last term}}{2}.
\]
\end{theorem}

\begin{proof}
Escribe la suma $S$ dos veces, la segunda vez en orden inverso, y suma
columna por columna:
\[
\begin{array}{ccccccccc}
S & = & 1 & + & 2 & + & \dots & + & n\\
S & = & n & + & (n-1) & + & \dots & + & 1\\
\hline
2S & = & (n+1) & + & (n+1) & + & \dots & + & (n+1)
\end{array}
\]
Hay columnas $n$, cada una de las cuales suma $n + 1$, por lo que $2S = n(n+1)$. por un
general \omterm{def:g11:seq:arithmetic}{secuencia aritmética} funciona el mismo emparejamiento: primero $+$ último $=$
segundo $+$ penúltimo $= \dots$, porque avanzar un paso en
el extremo izquierdo ($+d$) se compensa con un paso hacia atrás en el extremo derecho
($-d$).
\end{proof}

\begin{example}
$1 + 2 + \dots + 100 = \frac{100 \times 101}{2} = 5050$. la suma de la
números impares $1 + 3 + \dots + 99$ (términos $50$) es
$50 \times \frac{1 + 99}{2} = 2500$.
\end{example}

\section{Secuencias geométricas}

\begin{definition}[secuencia geométrica]\label{def:g11:seq:geometric}
Un \omterm{def:g11:seq:sequence}{secuencia} es \emph{geométrico}\index{secuencia geométrica} con \emph{común
relación}\index{proporción común} $q \neq 0$ si cada término se obtiene del anterior por
multiplicando por $q$:
\[
u_{n+1} = q\,u_n \quad \text{for all } n.
\]
De manera equivalente, cuando ningún término desaparece: la relación $\frac{u_{n+1}}{u_n}$ es
constante, igual a $q$.
\end{definition}

\begin{theorem}[término general]\label{thm:g11:seq:geomgeneral}
Si $(u_n)$ es \omterm{def:g11:seq:geometric}{geométrico} con el primer término $u_0$ y la relación $q$, entonces
\[
u_n = u_0\, q^n \quad \text{for all } n \geq 0,
\qquad\text{and more generally } u_n = u_p\, q^{\,n-p} .
\]
\end{theorem}

\begin{proof}
Mismo conteo de pasos que en \cref{thm:g11:seq:arithgeneral}: de $u_0$ a
$u_n$, se aplica $n$ veces la regla ``multiplicar por $q$'', aportando un
factor $q^n$.
\end{proof}

\begin{theorem}[suma geométrica]\label{thm:g11:seq:geomsum}
Para cada $q \neq 1$ y \omterm{def:g10:numbers:sets}{entero} $n \geq 0$ reales:
\[
1 + q + q^2 + \dots + q^n = \frac{1 - q^{\,n+1}}{1 - q}.
\]
\end{theorem}

\begin{proof}
Deje $S = 1 + q + \dots + q^n$. Multiplicar por $q$:
$qS = q + q^2 + \dots + q^{n+1}$. Restar:
\[
S - qS = \bigl(1 + q + \dots + q^n\bigr)
- \bigl(q + q^2 + \dots + q^{n+1}\bigr) = 1 - q^{\,n+1},
\]
porque cada término intermedio aparece una vez en cada suma y se cancela.
Por lo tanto $(1 - q)S = 1 - q^{\,n+1}$, y al dividir por $1 - q \neq 0$ se obtiene
fórmula.
\end{proof}

\begin{example}\label{ex:g11:seq:chessboard}
$1 + 2 + 4 + \dots + 2^{10} = \frac{1 - 2^{11}}{1 - 2} = 2^{11} - 1 =
2047$: doblar granos de arroz en las casillas de un tablero de ajedrez abruma
cualquier granero mucho antes de la casilla $64$, donde el total es
$2^{64} - 1 \approx 1.8 \times 10^{19}$.
\end{example}

\begin{method}[Reconociendo el tipo de una secuencia]\label{met:g11:seq:recognize}
Calcule $u_{n+1} - u_n$ y simplifique. Si el resultado es una constante $d$, el
\omterm{def:g11:seq:sequence}{secuencia} es \omterm{def:g11:seq:arithmetic}{aritmética}. De lo contrario, calcule $\frac{u_{n+1}}{u_n}$ (términos
distinto de cero) y simplificar: una constante $q$ significa \omterm{def:g11:seq:geometric}{geométrico}. Si ninguno lo es
constante, el \omterm{def:g11:seq:sequence}{secuencia} no es de ningún tipo --- nunca concluya del
primeros términos solos.
\end{method}

\begin{example}
Para $u_n = 3 \times 5^n$:
$\frac{u_{n+1}}{u_n} = \frac{3 \times 5^{n+1}}{3 \times 5^n} = 5$ para todos
$n$: \omterm{def:g11:seq:geometric}{geométrico} con relación $5$. Para $u_n = n^2$:
$u_1 - u_0 = 1$ pero $u_2 - u_1 = 3$, y
$\frac{u_1}{u_0}$ ni siquiera está definido --- ni \omterm{def:g11:seq:arithmetic}{aritmética} ni
\omterm{def:g11:seq:geometric}{geométrico}.
\end{example}

\section{monotonicidad}

\begin{definition}[Secuencia monótona]\label{def:g11:seq:monotone}
A \omterm{def:g11:seq:sequence}{secuencia} $(u_n)$ es \emph{\omterm{def:g11:func:monotone}{creciente}} si $u_{n+1} \geq u_n$ para todos los $n$,
y \emph{\omterm{def:g10:functions:variations}{decreciente}} si $u_{n+1} \leq u_n$ para todos los $n$.
\end{definition}

\begin{method}[Estudiando la monotonía]\label{met:g11:seq:monotonicity}
Estudie el signo de $u_{n+1} - u_n$. Para \omterm{def:g11:seq:sequence}{secuencias} con términos positivos, uno
en su lugar, puede comparar $\frac{u_{n+1}}{u_n}$ con $1$.
\end{method}

\begin{example}
Un \omterm{def:g11:seq:arithmetic}{secuencia aritmética} es \omterm{def:g11:func:monotone}{creciente} cuando $d \geq 0$
($u_{n+1} - u_n = d$), \omterm{def:g10:functions:variations}{decreciente} cuando $d \leq 0$. A \omterm{def:g11:seq:geometric}{secuencia geométrica}
con $u_0 > 0$ y $q > 1$ es \omterm{def:g11:func:monotone}{creciente}:
$u_{n+1} - u_n = u_0 q^n (q - 1) > 0$; con $u_0 > 0$ y $0 < q < 1$
es \omterm{def:g10:functions:variations}{decreciente}.
\end{example}

\section{Comportamiento a largo plazo, informalmente}

¿Qué le sucede a $u_n$ cuando $n$ se vuelve muy grande? Para un \omterm{def:g11:seq:arithmetic}{aritmética
secuencia} con $d > 0$, los términos $u_0 + nd$ exceden cualquier número fijo
eventualmente. Para un \omterm{def:g11:seq:geometric}{secuencia geométrica} con $0 < q < 1$, los términos
$u_0 q^n$ se reduce hacia $0$: multiplicar repetidamente por $0.9$, digamos, erosiona
cualquier valor inicial. Y para $q > 1$ los términos explotan, como en
\cref{ex:g11:seq:chessboard}.

\begin{remark}
Estas afirmaciones pueden hacerse perfectamente precisas: "los términos eventualmente
manténgase dentro de cualquier distancia determinada de $0$'' --- y probado. esa es la teoria
de \emph{límites}, el tema de apertura de \cref{ch:g12:seq}.
\end{remark}

\section{Problema: La torre de Brahma y los conejos de Fibonacci}

\begin{problem}[{Problema del fin de semana --- dos recurrencias legendarias:
la torre que acaba con el mundo, la secuencia que crece como el oro,
y el truco auxiliar que domina los préstamos}]
\label{pb:g11:seq:1}
Dos \omterm{def:g11:seq:sequence}{secuencias} gobiernan el folklore de las matemáticas. Uno cuenta el
movimientos de la torre de Brahma --- sesenta y cuatro discos dorados cuyos
La transferencia, dice la leyenda, acabará con el mundo. el otro cuenta
Los conejos de Fibonacci y esconde la proporción áurea. Tampoco lo es
\omterm{def:g11:seq:arithmetic}{aritmética}, tampoco lo es \omterm{def:g11:seq:geometric}{geométrico} --- y ambos se entregan a esto
Armas del capítulo: recurrencias, sumas \omterm{def:g11:seq:geometric}{geométrico}.
(\cref{thm:g11:seq:geomsum}), y el truco \omterm{def:g11:seq:sequence}{secuencia auxiliar} de
\cref{exo:g11:seq:11}, que también computa tu hipoteca.

\medskip
\noindent\textbf{Parte I --- The tower of Brahma.}
El rompecabezas: los discos $n$ de tamaño \omterm{def:g10:functions:variations}{decreciente} están apilados en la clavija A;
mueva toda la pila a la clavija C, un disco a la vez, sin colocar nunca
un disco más grande en uno más pequeño (la clavija B puede ayudar). Sea $h_n$ el
número mínimo de movimientos.
\begin{enumerate}
  \item Juega (con monedas) y registra $h_1$, $h_2$, $h_3$.
  \item Explica la estrategia detrás de la recurrencia.
        $h_{n+1} = 2h_n + 1$: lo que debe pasar antes y después
        ¿Se mueve el disco más grande?
  \item Resuelve la recurrencia con el truco de
        \cref{exo:g11:seq:11}: establecer $v_n = h_n + 1$, mostrar
        $(v_n)$ es \omterm{def:g11:seq:geometric}{geométrico} y concluye $h_n = 2^n - 1$.
  \item La torre de la leyenda tiene discos $64$ y los monjes se mueven.
        un disco por segundo. Usando $2^{10} = 1024 \approx 10^3$,
        estimar el tiempo de transferencia en años (un año es aproximadamente
        $3 \times 10^7$ segundos; comparar
        \cref{ex:g11:seq:chessboard}, el mismo gigante en otro
        historia). ¿Deberíamos preocuparnos?
  \item ¿Por qué ninguna estrategia puede superar los movimientos $2^n - 1$? argumentar que
        \emph{cualquier} la solución obedece
        $h_{n+1} \geq 2 h_n + 1$: lo que debe ser cierto de la cima
        Discos $n$ justo antes y justo después del disco inferior.
        moverse?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Fibonacci.}
Defina $F_1 = F_2 = 1$ y $F_{n+2} = F_{n+1} + F_n$ (cada término
la suma de los dos anteriores --- la regla de conteo de ritmo del
Volumen Middle School, ahora con nombre europeo).
\begin{enumerate}[resume]
  \item Lista $F_1$ a $F_{12}$.
  \item Demuestre que $(F_n)$ no es ni \omterm{def:g11:seq:arithmetic}{aritmética} ni \omterm{def:g11:seq:geometric}{geométrico},
        pero que es estrictamente \omterm{def:g11:func:monotone}{creciente} de $n = 2$ en adelante
        (\cref{met:g11:seq:monotonicity} y la recurrencia).
  \item Demostrar la identidad de la suma
        \[
        F_1 + F_2 + \dots + F_n = F_{n+2} - 1
        \]
        telescópico: escriba cada $F_k$ como
        $F_{k+2} - F_{k+1}$ y observa cómo colapsa la suma. verificar
        para $n = 6$.
  \item Demostrar la identidad de los cuadrados.
        $F_1^2 + F_2^2 + \dots + F_n^2 = F_n F_{n+1}$,
        telescópico con
        $F_k F_{k+1} - F_{k-1} F_k = F_k^2$. Verificar para
        $n = 4$. (Imagen: cuadrados de lados
        $1, 1, 2, 3, 5, \dots$ mosaico de un rectángulo --- el famoso
        El esqueleto de la espiral de Fibonacci.)
  \item Estados de identidad de Cassini
        $F_{n+1} F_{n-1} - F_n^2 = (-1)^n$. Verifíquelo para
        $n = 4, 5, 6$ --- y reconocer el motor del
        Truco del cuadrado fugaz jugado en la escuela secundaria
        Problema de áreas de volumen.
  \item Demuestre por la recurrencia que
        $F_{n+2} \geq 2 F_n$: Fibonacci al menos se duplica cada
        dos pasos --- crece al menos tan rápido como un \omterm{def:g11:seq:geometric}{geométrico
        secuencia} de proporción $\sqrt2$.
  \item Calcule las razones $r_n = \frac{F_{n+1}}{F_n}$ para
        $n = 3$ a $10$ (tres decimales). Admitiendo que ellos
        establecer un límite $L$, pasar la relación
        $r_{n+1} = 1 + \frac{1}{r_n}$ al límite y resuelve:
        ¿cual numero de \cref{pb:g10:algebra:1} hacen los conejos?
        adoración?
\end{enumerate}

\noindent\textbf{Parte III --- The auxiliary trick, at the
bank.}
\begin{enumerate}[resume]
  \item Generalizar \cref{exo:g11:seq:11}: para
        $u_{n+1} = a\,u_n + b$ con $a \neq 1$, vamos
        $\ell = \frac{b}{1 - a}$ (el \omterm{pb:g10:functions:1}{punto fijo}). Mostrar

        concluir
        $u_n = a^n (u_0 - \ell) + \ell$.
  \item Un préstamo: $10\,000$ euros a $1\,\%$ intereses al mes,
        reembolsado $300$ euros al mes, por lo que la deuda obedece

        punto} primero!) para obtener una fórmula explícita para $d_n$.
  \item Con una calculadora, encuentre el primer mes en el que
        La deuda se liquida y se reembolsa el importe total. cuanto

\end{enumerate}
