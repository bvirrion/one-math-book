\chapter{Estadísticas descriptivas}\label{ch:g11:stat}

Estadísticas comprime una lista de números en algunos resúmenes significativos:
dónde se encuentran los datos (\omterm{def:g11:stat:mean}{significar}, \omterm{def:g11:stat:median}{mediana}) y qué tan dispersos están (\omterm{def:g11:stat:quartiles}{cuartiles},
\omterm{def:g11:stat:variance}{diferencia}, \omterm{def:g11:stat:variance}{desviación estándar}). Este capítulo resuelve todo en uno.
ejemplo de ejecución. Los mismos resúmenes reaparecen para variables aleatorias en
\cref{ch:g11:prob}, y sus leyes probabilísticas en \cref{ch:g12:sums}.

A lo largo del capítulo, nuestro conjunto de datos en ejecución son las marcas (de $20$) de
una clase de estudiantes $12$:
\[
8,\ 9,\ 10,\ 10,\ 11,\ 12,\ 12,\ 12,\ 13,\ 14,\ 16,\ 17 .
\]

\section{Media y mediana}

\begin{definition}[Medio]\label{def:g11:stat:mean}
El \emph{significar}\index{significar} de los valores $x_1, x_2, \dots, x_n$ es
\[
\bar x = \frac{x_1 + x_2 + \dots + x_n}{n}.
\]
Si el valor $x_i$ aparece con \omterm{def:g10:stats:series}{frecuencia} $n_i$ (con
$n_1 + \dots + n_k = n$), entonces
$\bar x = \frac{n_1 x_1 + \dots + n_k x_k}{n}$.
\end{definition}

\begin{definition}[mediana]\label{def:g11:stat:median}
Un \emph{mediana}\index{mediana} de un conjunto de datos ordenados es un valor que se divide
por la mitad: al menos la mitad de los valores son $\leq$ y al menos la mitad son
$\geq$. En la práctica, ordene los valores $n$; si $n$ es impar, la mediana es
el valor medio; si $n$ es par, tome el \omterm{prop:g10:coordgeom:midpoint}{punto medio} de los dos medios
valores.
\end{definition}

\begin{example}\label{ex:g11:stat:meanmedian}
Para el conjunto de datos en ejecución, la suma es $144$, por lo que $\bar x = \frac{144}{12} =
12$. Los dos valores medios ($6$th y $7$th de la lista ordenada) son ambos
$12$, por lo que \omterm{def:g11:stat:median}{mediana} también es $12$. \omterm{def:g11:stat:mean}{Significar} y \omterm{def:g11:stat:median}{mediana} no tienen por qué coincidir: el
\omterm{def:g11:stat:median}{mediana} ignora \emph{Hasta dónde}, los valores extremos son, \omterm{def:g11:stat:mean}{significar} no.
Reemplazar la marca superior $17$ por $20$ mueve el \omterm{def:g11:stat:mean}{significar} a $12.25$ pero deja
el \omterm{def:g11:stat:median}{mediana} en $12$.
\end{example}

\section{Cuartiles}

\begin{definition}[Cuartiles, rango intercuartil]\label{def:g11:stat:quartiles}
El \emph{primer cuartil}\index{cuartilla} $Q_1$ es el valor más pequeño tal
que al menos una cuarta parte de los datos sean $\leq Q_1$; el \emph{tercero
cuartil} $Q_3$ es el valor más pequeño tal que al menos tres cuartos de
el dato es $\leq Q_3$. El \emph{rango intercuartil}\index{rango intercuartil} es $Q_3 - Q_1$: el
ancho de la mitad central de los datos.
\end{definition}

\begin{example}
$n = 12$: una cuarta parte es $3$, por lo que $Q_1$ es el tercer valor ordenado $3$:
$Q_1 = 10$. Tres cuartos es $9$, por lo que $Q_3$ es el valor $9$:
$Q_3 = 13$. El \omterm{def:g11:stat:quartiles}{rango intercuartil} es $3$: la mitad central de la clase.
se encuentra dentro de las marcas $3$.
\end{example}

\section{Varianza y desviación estándar}

\begin{definition}[Varianza, desviación estándar]\label{def:g11:stat:variance}
El \emph{diferencia}\index{diferencia} del conjunto de datos es el \omterm{def:g11:stat:mean}{significar} del
desviaciones al cuadrado del \omterm{def:g11:stat:mean}{significar}:
\[
V = \frac{1}{n}\sum_{i=1}^{n} (x_i - \bar x)^2,
\]
y el \emph{desviación estándar}\index{desviación estándar} es
$\sigma = \sqrt{V}$, expresado en la misma unidad que los datos.
\end{definition}

\begin{proposition}[Fórmula de acceso directo]\label{prop:g11:stat:shortcut}
El \omterm{def:g11:stat:variance}{diferencia} es el \omterm{def:g11:stat:mean}{significar} de los cuadrados menos el cuadrado del \omterm{def:g11:stat:mean}{significar}:
\[
V = \frac{1}{n}\sum_{i=1}^{n} x_i^2 \;-\; \bar x^{\,2}.
\]
\end{proposition}

\begin{proof}
Expande cada desviación al cuadrado:
$(x_i - \bar x)^2 = x_i^2 - 2\bar x\, x_i + \bar x^2$. sumando y
dividiendo por $n$:
\[
V = \frac1n \sum x_i^2 - 2\bar x \cdot \underbrace{\frac1n \sum
x_i}_{=\ \bar x} + \frac1n \cdot n\bar x^2
= \frac1n\sum x_i^2 - 2\bar x^2 + \bar x^2
= \frac1n\sum x_i^2 - \bar x^2 .
\]
La misma identidad, para variables aleatorias, es
\cref{prop:g11:prob:konig}.
\end{proof}

\begin{example}\label{ex:g11:stat:variance}
Para el conjunto de datos en ejecución, las desviaciones de $\bar x = 12$ son $-4$,
$-3$, $-2$, $-2$, $-1$, $0$, $0$, $0$, $1$, $2$, $4$, $5$; sus cuadrados
suma a
\[
16 + 9 + 4 + 4 + 1 + 0 + 0 + 0 + 1 + 4 + 16 + 25 = 80 .
\]
Por lo tanto
\[
V = \frac{80}{12} = \frac{20}{3} \approx 6.67,
\qquad
\sigma = \sqrt{20/3} \approx 2.6 \text{ marks}.
\]
En términos generales, una marca típica se encuentra a unas $2.6$ marcas de distancia de la
\omterm{def:g11:stat:mean}{significar}.
\end{example}

\begin{proposition}[Cambio de unidades]\label{prop:g11:stat:affine}
Si cada valor es transformado por $y_i = a x_i + b$, entonces
\[
\bar y = a \bar x + b,
\qquad
V_y = a^2\, V_x,
\qquad
\sigma_y = \abs{a}\,\sigma_x .
\]
\end{proposition}

\begin{proof}
$\bar y = \frac1n \sum (a x_i + b) = a\left(\frac1n\sum x_i\right) +
\frac1n \cdot nb = a\bar x + b$. entonces
$y_i - \bar y = a(x_i - \bar x)$: cada desviación se escala en $a$
(el desplazamiento $b$ se cancela), por lo que cada desviación al cuadrado se escala en $a^2$,
y también lo es su \omterm{def:g11:stat:mean}{significar}. Sacar raíces cuadradas da
$\sigma_y = \abs a\,\sigma_x$.
\end{proof}

\begin{example}
Temperaturas registradas en grados Celsius con \omterm{def:g11:stat:mean}{significar} $20$ y \omterm{def:g11:stat:variance}{estándar
desviación} $3$ convertidas a Fahrenheit por $F = 1.8\,C + 32$: el \omterm{def:g11:stat:mean}{significar}
se convierte en $1.8 \times 20 + 32 = 68$, el \omterm{def:g11:stat:variance}{desviación estándar}
$1.8 \times 3 = 5.4$. Cambiar un conjunto de datos no cambia su distribución;
reescalarlo reescala la extensión.
\end{example}

\begin{method}[Resumen y comparación de conjuntos de datos.]\label{met:g11:stat:compare}
Para comparar dos conjuntos de datos, colóquelos uno al lado del otro: \omterm{def:g11:stat:mean}{significar} (o \omterm{def:g11:stat:median}{mediana}) para
\emph{posición}, \omterm{def:g11:stat:variance}{desviación estándar} (o \omterm{def:g11:stat:quartiles}{rango intercuartil}) para
\emph{desparramar}. El par (\omterm{def:g11:stat:median}{mediana}, \omterm{def:g11:stat:quartiles}{rango intercuartil}) es robusto a
valores extremos; el par (\omterm{def:g11:stat:mean}{significar}, \omterm{def:g11:stat:variance}{desviación estándar}) usa todos los valores y
alimenta la teoría \omterm{def:g10:proba:distribution}{probabilidad}.
\end{method}

\begin{example}
Dos arqueros disparan flechas $10$ cada uno; puntuaciones promedio $8.5$ para ambos, pero
los \omterm{def:g11:stat:variance}{desviaciones estándar} son $0.5$ y $2.1$. Mismo nivel en promedio --- pero
el primer arquero es mucho más consistente.
\end{example}

\section{Problema: el puntaje z, una regla universal}

\begin{problem}[{Problema de fin de semana --- comparación de desviaciones estándar
manzanas con naranjas, la desigualdad de Chebyshev convierte a $\sigma$ en una
garantía, y la media gana su propia medalla}]
\label{pb:g11:stat:1}
¿$15/20$ en un examen difícil es mejor que $13/20$ en uno fácil? es
¿Un $23$ en una lista de números pequeños es un accidente o un valor atípico? crudo
los números no pueden decir --- pero divididos por la derecha \omterm{def:g11:stat:variance}{estándar
desviación} todos hablan un idioma, el \emph{\omterm{pb:g11:stat:1}{puntuación z}}. esto
El problema domina la maquinaria \omterm{def:g11:stat:variance}{diferencia}.
(\cref{def:g11:stat:variance},
\cref{prop:g11:stat:shortcut}), construye la regla universal, y
demuestra la notable desigualdad que convierte a $\sigma$ de un
número descriptivo en una garantía.

\medskip
\noindent\textbf{Parte I --- \omterm{def:g11:stat:variance}{Diferencia} mechanics.}
\begin{enumerate}
  \item Para el conjunto de datos $4, 8, 6, 5, 2$: calcule \omterm{def:g11:stat:mean}{significar}, luego
        el \omterm{def:g11:stat:variance}{diferencia} directamente de la definición, entonces
        $\sigma$.
  \item Vuelva a calcular el \omterm{def:g11:stat:variance}{diferencia} con la fórmula de acceso directo
        (\cref{prop:g11:stat:shortcut}) --- \omterm{def:g11:stat:mean}{significar} de cuadrados
        menos el cuadrado de \omterm{def:g11:stat:mean}{significar} --- y verifique el acuerdo.
  \item Con frecuencias: marcas $8$, $10$, $14$ obtenidas por
        Estudiantes $3$, $5$, $2$. Calcular \omterm{def:g11:stat:mean}{significar}, \omterm{def:g11:stat:variance}{diferencia} y
        $\sigma$ (dos decimales).
  \item Las temperaturas tienen \omterm{def:g11:stat:mean}{significar} $20$ y $\sigma = 3$ (grados
        Celsius). Indique \omterm{def:g11:stat:mean}{significar} y $\sigma$ en Fahrenheit
        ($F = 1.8\,C + 32$; \cref{prop:g11:stat:affine}).
  \item Demuestre las dos reglas que acabamos de usar: agregar una constante $b$ a
        cada valor desplaza \omterm{def:g11:stat:mean}{significar} por $b$ y deja $\sigma$
        sin cambios; multiplicar por $a$ multiplica el \omterm{def:g11:stat:mean}{significar} por $a$
        y $\sigma$ por $\abs a$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The universal ruler.}
El \emph{puntuación z}\index{puntuación z} de un valor $x$ en un conjunto de datos de
\omterm{def:g11:stat:mean}{significar} $\bar x$ y \omterm{def:g11:stat:variance}{desviación estándar} $\sigma$ es
$z = \dfrac{x - \bar x}{\sigma}$: el número de \omterm{def:g11:stat:variance}{estándar
desviaciones} que separa $x$ del \omterm{def:g11:stat:mean}{significar}.
\begin{enumerate}[resume]
  \item El examen A tiene \omterm{def:g11:stat:mean}{significar} $12$ y $\sigma = 2$; el examen B tiene \omterm{def:g11:stat:mean}{significar}
        $8$ y $\sigma = 4$. Alice obtuvo $15$ en A, Bob $13$
        en B. Calcule ambos \omterm{pb:g11:stat:1}{puntuaciones z}: cuyo rendimiento se mantiene
        ¿Más por encima de la multitud?
  \item Calcule los cinco \omterm{pb:g11:stat:1}{puntuaciones z} del conjunto de datos de la pregunta 1,
        y verificar que tengan \omterm{def:g11:stat:mean}{significar} $0$ y \omterm{def:g11:stat:variance}{estándar
        desviación} $1$. Explique a partir de la pregunta 5 por qué
        la estandarización \emph{siempre} produce \omterm{def:g11:stat:mean}{significar} $0$ y
        $\sigma = 1$.
  \item Una prueba de aptitud tiene \omterm{def:g11:stat:mean}{significar} $100$ y $\sigma = 15$. que
        ¿La puntuación bruta tiene $z = 2$? ¿Qué es el \omterm{pb:g11:stat:1}{puntuación z} de $76$?
  \item Un pediatra registra el peso de un bebé como
        ``$z = -2.5$''. Diga qué significa esto y por qué el crecimiento es
        rastreado en \omterm{pb:g11:stat:1}{puntuaciones z} en lugar de kilogramos como el niño
        edades.
  \item El valor atípico de \cref{exo:g11:stat:10} (conjunto de datos
        $1, 2, 2, 3, 3, 3, 4, 4, 5, 23$): calcula el \omterm{def:g11:stat:mean}{significar},
        $\sigma$ y \omterm{pb:g11:stat:1}{puntuación z} de $23$. La alarma clásica
        el umbral es $\abs z \geq 3$: ¿veredicto?
  \item Vuelva a calcular $\sigma$ \emph{sin} el valor $23$, y
        el \omterm{pb:g11:stat:1}{puntuación z} que tendría $23$ contra el clean
        \omterm{def:g11:stat:mean}{significar} y $\sigma$ de datos. ¿Qué hizo el caso atípico para
        el mismo gobernante pretendía medirlo, y ¿qué significa
        \cref{exo:g11:stat:10} sugiere como contramedida?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Chebyshev's guarantee.}
\begin{enumerate}[resume]
  \item Demuestre la desigualdad de Bienaymé--Chebyshev para a
        conjunto de datos: si una proporción $p$ de los valores satisface
        $\abs{x - \bar x} \geq k\sigma$, entonces, quedando sólo
        esos términos en la suma \omterm{def:g11:stat:variance}{diferencia},
        \[
        \sigma^2 \geq p \cdot (k\sigma)^2,
        \qquad\text{hence}\qquad
        p \leq \frac{1}{k^2} .
        \]
  \item ¿Qué garantiza la desigualdad sobre la proporción?
        de cualquier conjunto de datos más allá de $2\sigma$ de su \omterm{def:g11:stat:mean}{significar}? Más allá
        $3\sigma$? Verifique la declaración $2\sigma$ en el conjunto de datos
        de la pregunta 1.
  \item Una fábrica produce pernos con \omterm{def:g11:stat:mean}{significar} $50$ mm y
        $\sigma = 0.1$ mm. ¿Qué garantiza Chebyshev sobre
        la proporción de tornillos fuera
        $\intcc{49.7}{50.3}$? (La producción real funciona mucho mejor
        --- la curva de campana del grado 12 se vuelve así de cautelosa
        $\frac19$ en una fracción de porcentaje.)
  \item Dos fondos devuelven $5\,\%$ por año en promedio, con
        $\sigma = 2\,\%$ para el fondo A y $\sigma = 12\,\%$ para
        fondo B. Para cada fondo, ¿qué dice Chebyshev sobre
        la proporción similar a la probabilidad de años con rendimientos
        debajo de $-19\,\%$? ¿Cuál es el nombre financiero de $\sigma$?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The \omterm{def:g11:stat:mean}{significar}'s medal, and a warning.}
\begin{enumerate}[resume]
  \item Sea $f(c)$ el \omterm{def:g11:stat:mean}{significar} de las desviaciones al cuadrado.
        $(x_i - c)^2$ de un centro arbitrario $c$. demostrar el
        identidad
        \[
        f(c) = \sigma^2 + (\bar x - c)^2 ,
        \]
        y concluimos: el \omterm{def:g11:stat:mean}{significar} es el único minimizador de \omterm{def:g11:stat:mean}{significar}
        desviación al cuadrado --- su medalla, que coincide con la del \omterm{def:g11:stat:median}{mediana}
        minimización de las desviaciones absolutas en
        \cref{pb:g10:stats:1}.
  \item Verifique la identidad en el conjunto de datos de la pregunta 1 con
        $c = 6$: calcula $f(6)$ directamente y mediante la fórmula.
  \item Mismos resúmenes, diferentes datos: comprueba que
        $P = \{3, 5, 7, 9, 11\}$ y $Q = \{1, 7, 7, 7, 13\}$
        comparten el mismo \omterm{def:g11:stat:mean}{significar} pero no el mismo \omterm{def:g11:stat:variance}{diferencia}; entonces
        construir un conjunto de datos de la forma
        $\{7 - a, 7, 7, 7, 7 + a\}$ con exactamente el \omterm{def:g11:stat:mean}{significar} y
        \omterm{def:g11:stat:variance}{diferencia} de $P$ --- pero con una forma propia.
  \item Demuestre que $\sigma = 0$ fuerza que todos los valores sean iguales. (¿Qué
        ¿Debe serlo cada término de una suma cero de cuadrados?)
  \item Finale --- la escalera del estadístico, una línea por
        peldaño: centros (\omterm{def:g11:stat:mean}{significar}, \omterm{def:g11:stat:median}{mediana}), diferenciales ($\sigma$, IQR),
        el gobernante universal ($z$), la garantía (Chebyshev),
        y la advertencia de la pregunta 18: los resúmenes se comprimen,
        y la compresión pierde --- con la curva de campana de grado
        12 esperando convertir \omterm{pb:g11:stat:1}{puntuaciones z} en probabilidades exactas.
\end{enumerate}
