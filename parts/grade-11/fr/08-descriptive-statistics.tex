\chapter{Statistique descriptive}\label{ch:g11:stat}

La statistique comprime une liste de nombres en quelques résumés
signifiants~: où se situent les données (\omterm{def:g11:stat:mean}{moyenne}, \omterm{def:g11:stat:median}{médiane}), et comment
elles sont dispersées (\omterm{def:g11:stat:quartiles}{quartiles}, \omterm{def:g11:stat:variance}{variance}, \omterm{def:g11:stat:variance}{écart type}). Ce chapitre
calcule tout sur un exemple fil rouge. Les mêmes résumés réapparaissent
pour les variables aléatoires dans le \cref{ch:g11:prob}, et leurs lois
probabilistes dans le \cref{ch:g12:sums}.

Tout au long du chapitre, notre jeu de données fil rouge est les notes
(sur $20$) d'une classe de $12$ élèves~:
\[
8,\ 9,\ 10,\ 10,\ 11,\ 12,\ 12,\ 12,\ 13,\ 14,\ 16,\ 17 .
\]

\section{Moyenne et médiane}

\begin{definition}[Moyenne]\label{def:g11:stat:mean}
La \emph{moyenne}\index{moyenne} des valeurs $x_1, x_2, \dots, x_n$ est
\[
\bar x = \frac{x_1 + x_2 + \dots + x_n}{n}.
\]
Si la valeur $x_i$ apparaît avec l'effectif $n_i$ (avec
$n_1 + \dots + n_k = n$), alors
$\bar x = \frac{n_1 x_1 + \dots + n_k x_k}{n}$.
\end{definition}

\begin{definition}[Médiane]\label{def:g11:stat:median}
Une \emph{médiane}\index{médiane} d'un jeu de données trié est une valeur
qui le partage en deux~: au moins la moitié des valeurs sont $\leq$ à
elle, et au moins la moitié sont $\geq$ à elle. En pratique, trier les
$n$ valeurs~; si $n$ est impair, la médiane est la valeur du \omterm{prop:g10:coordgeom:midpoint}{milieu}~; si
$n$ est pair, prendre le \omterm{prop:g10:coordgeom:midpoint}{milieu} des deux valeurs centrales.
\end{definition}

\begin{example}\label{ex:g11:stat:meanmedian}
Pour le jeu fil rouge, la somme est $144$, donc $\bar x = \frac{144}{12}
= 12$. Les deux valeurs centrales ($6$-ième et $7$-ième de la liste
triée) sont toutes deux $12$, donc la \omterm{def:g11:stat:median}{médiane} est $12$ aussi. \omterm{def:g11:stat:mean}{Moyenne} et
\omterm{def:g11:stat:median}{médiane} n'ont pas besoin de coïncider~: la \omterm{def:g11:stat:median}{médiane} ignore \emph{à quel
point} les valeurs extrêmes sont éloignées, la \omterm{def:g11:stat:mean}{moyenne} non. Remplacer la
note haute $17$ par $20$ déplace la \omterm{def:g11:stat:mean}{moyenne} à $12.25$ mais laisse la
\omterm{def:g11:stat:median}{médiane} à $12$.
\end{example}

\section{Quartiles}

\begin{definition}[Quartiles, écart interquartile]\label{def:g11:stat:quartiles}
Le \emph{premier quartile}\index{quartile} $Q_1$ est la plus petite
valeur telle qu'au moins un quart des données est $\leq Q_1$~; le
\emph{troisième quartile} $Q_3$ est la plus petite valeur telle qu'au
moins les trois quarts des données sont $\leq Q_3$. L'\emph{écart
interquartile}\index{écart interquartile} est $Q_3 - Q_1$~: la largeur de la moitié centrale des
données.
\end{definition}

\begin{example}
$n = 12$~: un quart vaut $3$, donc $Q_1$ est la $3$-ième valeur
triée~: $Q_1 = 10$. Trois quarts valent $9$, donc $Q_3$ est la $9$-ième
valeur~: $Q_3 = 13$. L'\omterm{def:g11:stat:quartiles}{écart interquartile} est $3$~: la moitié centrale
de la classe se trouve dans un \omterm{def:g10:numbers:interval}{intervalle} de $3$ points.
\end{example}

\section{Variance et écart type}

\begin{definition}[Variance, écart type]\label{def:g11:stat:variance}
La \emph{variance}\index{variance} du jeu de données est la \omterm{def:g11:stat:mean}{moyenne} des
écarts au carré à la \omterm{def:g11:stat:mean}{moyenne}~:
\[
V = \frac{1}{n}\sum_{i=1}^{n} (x_i - \bar x)^2,
\]
et l'\emph{écart type}\index{écart type} est $\sigma = \sqrt{V}$, exprimé
dans la même unité que les données.
\end{definition}

\begin{proposition}[Formule de König--Huygens]\label{prop:g11:stat:shortcut}
La \omterm{def:g11:stat:variance}{variance} est la \omterm{def:g11:stat:mean}{moyenne} des carrés moins le carré de la \omterm{def:g11:stat:mean}{moyenne}~:
\[
V = \frac{1}{n}\sum_{i=1}^{n} x_i^2 \;-\; \bar x^{\,2}.
\]
\end{proposition}

\begin{proof}
Développer chaque écart au carré~:
$(x_i - \bar x)^2 = x_i^2 - 2\bar x\, x_i + \bar x^2$. En sommant et en
divisant par $n$~:
\[
V = \frac1n \sum x_i^2 - 2\bar x \cdot \underbrace{\frac1n \sum
x_i}_{=\ \bar x} + \frac1n \cdot n\bar x^2
= \frac1n\sum x_i^2 - 2\bar x^2 + \bar x^2
= \frac1n\sum x_i^2 - \bar x^2 .
\]
La même identité, pour les variables aléatoires, est la
\cref{prop:g11:prob:konig}.
\end{proof}

\begin{example}\label{ex:g11:stat:variance}
Pour le jeu fil rouge, les écarts à $\bar x = 12$ sont $-4$, $-3$, $-2$,
$-2$, $-1$, $0$, $0$, $0$, $1$, $2$, $4$, $5$~; leurs carrés somment à
\[
16 + 9 + 4 + 4 + 1 + 0 + 0 + 0 + 1 + 4 + 16 + 25 = 80 .
\]
D'où
\[
V = \frac{80}{12} = \frac{20}{3} \approx 6.67,
\qquad
\sigma = \sqrt{20/3} \approx 2.6 \text{ points}.
\]
En gros, une note typique se situe à environ $2.6$ points de la \omterm{def:g11:stat:mean}{moyenne}.
\end{example}

\begin{proposition}[Changement d'unités]\label{prop:g11:stat:affine}
Si chaque valeur est transformée par $y_i = a x_i + b$, alors
\[
\bar y = a \bar x + b,
\qquad
V_y = a^2\, V_x,
\qquad
\sigma_y = \abs{a}\,\sigma_x .
\]
\end{proposition}

\begin{proof}
$\bar y = \frac1n \sum (a x_i + b) = a\left(\frac1n\sum x_i\right) +
\frac1n \cdot nb = a\bar x + b$. Puis
$y_i - \bar y = a(x_i - \bar x)$~: chaque écart est multiplié par $a$
(le décalage $b$ s'annule), donc chaque écart au carré est multiplié par
$a^2$, et sa \omterm{def:g11:stat:mean}{moyenne} aussi. En prenant les racines carrées,
$\sigma_y = \abs a\,\sigma_x$.
\end{proof}

\begin{example}
Des températures enregistrées en degrés Celsius de \omterm{def:g11:stat:mean}{moyenne} $20$ et
d'\omterm{def:g11:stat:variance}{écart type} $3$ se convertissent en Fahrenheit par
$F = 1.8\,C + 32$~: la \omterm{def:g11:stat:mean}{moyenne} devient $1.8 \times 20 + 32 = 68$,
l'\omterm{def:g11:stat:variance}{écart type} $1.8 \times 3 = 5.4$. Décaler un jeu de données ne change
pas sa dispersion~; le redimensionner redimensionne la dispersion.
\end{example}

\begin{method}[Résumer et comparer des jeux de données]\label{met:g11:stat:compare}
Pour comparer deux jeux de données, mettre côte à côte~: \omterm{def:g11:stat:mean}{moyenne} (ou
\omterm{def:g11:stat:median}{médiane}) pour la \emph{position}, \omterm{def:g11:stat:variance}{écart type} (ou \omterm{def:g11:stat:quartiles}{écart interquartile})
pour la \emph{dispersion}. Le couple (\omterm{def:g11:stat:median}{médiane}, \omterm{def:g11:stat:quartiles}{écart interquartile}) est
robuste aux valeurs extrêmes~; le couple (\omterm{def:g11:stat:mean}{moyenne}, \omterm{def:g11:stat:variance}{écart type}) utilise
chaque valeur et alimente la théorie des \omterm{def:g10:proba:distribution}{probabilités}.
\end{method}

\begin{example}
Deux archers tirent chacun $10$ flèches~; les scores moyennent $8.5$
pour les deux, mais les \omterm{def:g11:stat:variance}{écarts types} sont $0.5$ et $2.1$. Même niveau en
\omterm{def:g11:stat:mean}{moyenne} --- mais le premier archer est bien plus régulier.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g11:stat:1}
Calculer la \omterm{def:g11:stat:mean}{moyenne}, la \omterm{def:g11:stat:median}{médiane}, les \omterm{def:g11:stat:quartiles}{quartiles} et l'\omterm{def:g11:stat:quartiles}{écart interquartile}
du jeu de données
\[
5,\ 7,\ 7,\ 8,\ 9,\ 10,\ 11,\ 14 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:stat:2}
Calculer la \omterm{def:g11:stat:variance}{variance} et l'\omterm{def:g11:stat:variance}{écart type} du jeu de données
$2,\ 4,\ 4,\ 4,\ 5,\ 5,\ 7,\ 9$ (\omterm{def:g11:stat:mean}{moyenne} d'abord, puis la formule de
König--Huygens de la \cref{prop:g11:stat:shortcut}).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:stat:3}
Un dé est lancé $50$ fois~; les issues sont résumées par des
effectifs~:
\begin{center}
\begin{tabular}{c|cccccc}
issue & 1 & 2 & 3 & 4 & 5 & 6\\
\hline
effectif & 6 & 9 & 8 & 10 & 9 & 8
\end{tabular}
\end{center}
Calculer la \omterm{def:g11:stat:mean}{moyenne} des issues et la \omterm{def:g11:stat:median}{médiane}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:stat:4}
Une classe de $15$ élèves a une \omterm{def:g11:stat:mean}{moyenne} de $11.2$. Un $16$-ième élève
arrive et obtient $18$. Quelle est la nouvelle \omterm{def:g11:stat:mean}{moyenne} de la classe~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:stat:5}
La \omterm{def:g11:stat:mean}{moyenne} du jeu de données $6, 8, x, 12, 14$ est $10$. Trouver $x$,
puis calculer l'\omterm{def:g11:stat:variance}{écart type} du jeu complété.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:stat:6}
Des notes de \omterm{def:g11:stat:mean}{moyenne} $12.4$ et d'\omterm{def:g11:stat:variance}{écart type} $2.5$ sont redimensionnées
par $y = \frac{x}{2} + 4$. Donner la nouvelle \omterm{def:g11:stat:mean}{moyenne} et le nouvel
\omterm{def:g11:stat:variance}{écart type}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:stat:7}
Le groupe A ($20$ valeurs) a pour \omterm{def:g11:stat:mean}{moyenne} $10$~; le groupe B ($30$
valeurs) a pour \omterm{def:g11:stat:mean}{moyenne} $15$. Calculer la \omterm{def:g11:stat:mean}{moyenne} des $50$ valeurs mises
ensemble. Est-ce le \omterm{prop:g10:coordgeom:midpoint}{milieu} $12.5$~? Pourquoi pas~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:stat:8}
Deux chaînes de production remplissent des sacs de sucre de $1$~kg. Des
\omterm{def:g10:proba:sample}{échantillons} donnent~: chaîne~1~: \omterm{def:g11:stat:mean}{moyenne} $1.002$~kg, \omterm{def:g11:stat:variance}{écart type}
$0.006$~kg~; chaîne~2~: \omterm{def:g11:stat:mean}{moyenne} $1.001$~kg, \omterm{def:g11:stat:variance}{écart type} $0.019$~kg.
Quelle chaîne recalibrer en premier, et pourquoi~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:stat:9}
Un jeu de données de $n = 10$ valeurs a $\sum x_i = 70$ et
$\sum x_i^2 = 540$. Calculer sa \omterm{def:g11:stat:mean}{moyenne}, sa \omterm{def:g11:stat:variance}{variance} et son \omterm{def:g11:stat:variance}{écart type}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:stat:10}
Le jeu de données $1,\ 2,\ 2,\ 3,\ 3,\ 3,\ 4,\ 4,\ 5,\ 23$ contient une
\emph{valeur aberrante}.
\begin{enumerate}
  \item Calculer la \omterm{def:g11:stat:mean}{moyenne} et la \omterm{def:g11:stat:median}{médiane} avec et sans la valeur $23$.
  \item Calculer l'\omterm{def:g11:stat:quartiles}{écart interquartile} avec et sans $23$.
  \item Quels résumés sont robustes à la valeur aberrante~? Conclure
        avec une recommandation.
\end{enumerate}
\end{exercise}
