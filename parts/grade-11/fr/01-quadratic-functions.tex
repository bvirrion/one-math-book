\chapter{Fonctions et équations du second degré}\label{ch:g11:quad}

Les fonctions du second degré sont les fonctions les plus simples après
les linéaires, et les premières dont les graphiques sont véritablement
courbes. Ce chapitre développe la boîte à outils complète pour elles~: la
forme canonique, le discriminant, le signe d'un trinôme, et la géométrie
de la parabole.

\section{La forme canonique}

\begin{definition}[Fonction du second degré]\label{def:g11:quad:quadratic}
Une \emph{fonction du second degré}\index{fonction du second degré} est une
fonction de la forme
\[
f(x) = ax^2 + bx + c, \qquad a \neq 0,
\]
où $a$, $b$, $c$ sont des nombres réels. L'expression $ax^2 + bx + c$ est
aussi appelée un \emph{trinôme du second degré}.
\end{definition}

\begin{theorem}[Forme canonique]\label{thm:g11:quad:canonical}
Toute fonction du second degré peut s'écrire d'exactement une façon comme
\[
f(x) = a(x - \alpha)^2 + \beta,
\qquad\text{où } \alpha = -\frac{b}{2a} \text{ et } \beta = f(\alpha).
\]
C'est la \emph{forme canonique}\index{forme canonique} de $f$.
\end{theorem}

\begin{proof}
On \emph{complète le carré}. Comme $a \neq 0$, factoriser $a$ dans les
deux premiers termes~:
\[
ax^2 + bx + c = a\left(x^2 + \frac{b}{a}x\right) + c .
\]
Or $x^2 + \frac{b}{a}x$ est le début du carré
$\left(x + \frac{b}{2a}\right)^2 = x^2 + \frac ba x + \frac{b^2}{4a^2}$,
donc $x^2 + \frac{b}{a}x = \left(x + \frac{b}{2a}\right)^2 -
\frac{b^2}{4a^2}$. En substituant~:
\[
f(x) = a\left(x + \frac{b}{2a}\right)^2 - \frac{b^2}{4a} + c
= a\bigl(x - \alpha\bigr)^2 + \beta,
\]
avec $\alpha = -\frac{b}{2a}$ et $\beta = c - \frac{b^2}{4a}$. Évaluer en
$x = \alpha$ annule le terme au carré, donc $\beta = f(\alpha)$.
Unicité~: développer $a(x-\alpha)^2 + \beta$ et identifier les
coefficients de $x$ force $\alpha = -\frac{b}{2a}$, puis $\beta$ est
déterminé.
\end{proof}

\begin{example}\label{ex:g11:quad:canonical}
Soit $f(x) = 2x^2 - 8x + 3$. Alors $\alpha = -\frac{-8}{2 \times 2} = 2$
et $\beta = f(2) = 8 - 16 + 3 = -5$, donc
\[
f(x) = 2(x - 2)^2 - 5 .
\]
Vérification en développant~: $2(x^2 - 4x + 4) - 5 = 2x^2 - 8x + 8 - 5 =
2x^2 - 8x + 3$.
\end{example}

\begin{proposition}[Extremum et variations]\label{prop:g11:quad:variations}
Soit $f(x) = a(x-\alpha)^2 + \beta$.
\begin{enumerate}
  \item Si $a > 0$, alors $f$ est décroissante sur
        $\intoc{-\infty}{\alpha}$ et croissante sur
        $\intco{\alpha}{+\infty}$~: $f$ a un \emph{minimum} $\beta$,
        atteint en $x = \alpha$.
  \item Si $a < 0$, les variations sont inversées~: $f$ a un
        \emph{maximum} $\beta$ en $x = \alpha$.
\end{enumerate}
\end{proposition}

\begin{proof}
Supposons $a > 0$ et soit $\alpha \leq u < v$. Alors
\[
f(v) - f(u) = a\bigl((v-\alpha)^2 - (u-\alpha)^2\bigr)
= a\,(v - u)\,\bigl((v-\alpha) + (u-\alpha)\bigr).
\]
Chaque facteur est positif~: $a > 0$, $v - u > 0$, et
$(v-\alpha) + (u-\alpha) > 0$ car $v > \alpha$ et $u \geq \alpha$. Donc
$f(v) > f(u)$~: $f$ est croissante sur $\intco{\alpha}{+\infty}$. Pour
$u < v \leq \alpha$ le dernier facteur est négatif, donc
$f(v) < f(u)$~: $f$ est décroissante là. Enfin
$f(x) - f(\alpha) = a(x-\alpha)^2 \geq 0$ pour tout $x$, donc
$\beta = f(\alpha)$ est le minimum. Le cas $a < 0$ s'obtient en appliquant
ce qui précède à $-f$.
\end{proof}

\begin{example}
Un fermier a $100$ mètres de clôture pour enfermer un champ rectangulaire
contre un mur droit (pas de clôture le long du mur). Si $x$ est la
largeur, l'aire enclose est $A(x) = x(100 - 2x) = -2x^2 + 100x$. Ici
$\alpha = -\frac{100}{2\times(-2)} = 25$ et $A(25) = 25 \times 50 =
1250$~: l'aire est maximale pour une largeur de $25$~m, donnant
$1250$~m$^2$.
\end{example}

\section{Racines et factorisation}

\begin{definition}[Racine, discriminant]\label{def:g11:quad:discriminant}
Une \emph{racine}\index{racine} du trinôme $ax^2+bx+c$ est une solution de
l'équation $ax^2 + bx + c = 0$. Le
\emph{discriminant}\index{discriminant} du trinôme est le nombre
\[
\Delta = b^2 - 4ac .
\]
\end{definition}

\begin{theorem}[Résoudre $ax^2+bx+c=0$]\label{thm:g11:quad:roots}
\begin{enumerate}
  \item Si $\Delta > 0$, l'équation a deux racines distinctes
        \[
        x_1 = \frac{-b - \sqrt{\Delta}}{2a},
        \qquad
        x_2 = \frac{-b + \sqrt{\Delta}}{2a},
        \]
        et $ax^2+bx+c = a(x - x_1)(x - x_2)$.
  \item Si $\Delta = 0$, l'équation a l'unique racine
        $x_0 = -\frac{b}{2a}$, et $ax^2+bx+c = a(x - x_0)^2$.
  \item Si $\Delta < 0$, l'équation n'a pas de solution réelle, et le
        trinôme ne se factorise pas avec des coefficients réels.
\end{enumerate}
\end{theorem}

\begin{proof}
Partir de la forme canonique calculée dans le
\cref{thm:g11:quad:canonical}~:
\[
ax^2+bx+c
= a\left(x + \frac{b}{2a}\right)^2 + c - \frac{b^2}{4a}
= a\left[\left(x + \frac{b}{2a}\right)^2 - \frac{\Delta}{4a^2}\right],
\]
car $c - \frac{b^2}{4a} = \frac{4ac - b^2}{4a} = -\frac{\Delta}{4a} =
a \times \left(-\frac{\Delta}{4a^2}\right)$.

\emph{1.} Si $\Delta > 0$, alors $\frac{\Delta}{4a^2} =
\left(\frac{\sqrt\Delta}{2a}\right)^2$ et le crochet est une différence de
deux carrés~:
\[
ax^2+bx+c = a\left(x + \frac{b}{2a} - \frac{\sqrt\Delta}{2a}\right)
\left(x + \frac{b}{2a} + \frac{\sqrt\Delta}{2a}\right)
= a(x - x_2)(x - x_1).
\]
Un produit est nul exactement lorsqu'un de ses facteurs est nul, d'où les
deux racines, distinctes car $\sqrt\Delta \neq 0$.

\emph{2.} Si $\Delta = 0$, le crochet est
$\left(x + \frac{b}{2a}\right)^2$, qui s'annule seulement en
$x_0 = -\frac{b}{2a}$.

\emph{3.} Si $\Delta < 0$, alors $-\frac{\Delta}{4a^2} > 0$, donc le
crochet est un carré plus un nombre positif~: il ne s'annule jamais, et
l'équation n'a pas de solution réelle. Une factorisation
$a(x-x_1)(x-x_2)$ produirait des racines, donc aucune n'existe.
\end{proof}

\begin{example}\label{ex:g11:quad:solving}
Résoudre $2x^2 - 8x + 3 = 0$. Ici
$\Delta = (-8)^2 - 4 \times 2 \times 3 = 64 - 24 = 40 > 0$, donc deux
racines~:
\[
x_{1,2} = \frac{8 \pm \sqrt{40}}{4} = \frac{8 \pm 2\sqrt{10}}{4}
= 2 \pm \frac{\sqrt{10}}{2}.
\]
Pour $x^2 + x + 1 = 0$~: $\Delta = 1 - 4 = -3 < 0$, pas de solution
réelle.
\end{example}

\begin{proposition}[Somme et produit des racines]\label{prop:g11:quad:vieta}
Si $\Delta \geq 0$ et $x_1, x_2$ sont les racines (égales lorsque
$\Delta = 0$), alors
\[
x_1 + x_2 = -\frac{b}{a},
\qquad
x_1 x_2 = \frac{c}{a} .
\]
Réciproquement, deux nombres de somme $s$ et de produit $p$ sont les
solutions de $x^2 - sx + p = 0$.
\end{proposition}

\begin{proof}
Développer la factorisation du \cref{thm:g11:quad:roots}~:
\[
a(x - x_1)(x - x_2) = a x^2 - a(x_1 + x_2)x + a\,x_1 x_2 .
\]
Identifier les coefficients avec $ax^2 + bx + c$ donne
$-a(x_1+x_2) = b$ et $a\,x_1x_2 = c$. Pour la réciproque, si $u + v = s$
et $uv = p$, alors $(x-u)(x-v) = x^2 - sx + p$, donc $u$ et $v$ en sont
les racines.
\end{proof}

\begin{example}
Trouver deux nombres de somme $10$ et de produit $21$. Ils résolvent
$x^2 - 10x + 21 = 0$~; $\Delta = 100 - 84 = 16$, racines
$\frac{10 \pm 4}{2} = 7$ et $3$. \emph{Repérage rapide de racines~:}
comme les coefficients de $x^2 - 5x + 4$ satisfont $1 - 5 + 4 = 0$, le
nombre $1$ est racine, et l'autre est $\frac{c}{a} = 4$ (leur produit doit
être $\frac ca$).
\end{example}

\section{Signe d'un trinôme}

\begin{theorem}[Signe d'un trinôme]\label{thm:g11:quad:sign}
Soit $f(x) = ax^2+bx+c$ avec $a \neq 0$.
\begin{enumerate}
  \item Si $\Delta > 0$, avec racines $x_1 < x_2$~: $f(x)$ a le signe de
        $a$ à l'extérieur de $\intcc{x_1}{x_2}$, et le signe de $-a$ sur
        $\intoo{x_1}{x_2}$.
  \item Si $\Delta = 0$~: $f(x)$ a le signe de $a$ pour tout
        $x \neq x_0$, et s'annule en $x_0$.
  \item Si $\Delta < 0$~: $f(x)$ a le signe de $a$ pour tout $x$.
\end{enumerate}
En bref~: \emph{un trinôme a le signe de $a$, sauf entre ses racines}.
\end{theorem}

\begin{proof}
\emph{1.} Écrire $f(x) = a(x-x_1)(x-x_2)$ et suivre les signes des
facteurs. Si $x < x_1$~: $x - x_1$ et $x - x_2$ sont tous deux négatifs,
leur produit est positif, donc $f(x)$ a le signe de $a$. Si
$x_1 < x < x_2$~: les facteurs ont des signes opposés, le produit est
négatif, et $f(x)$ a le signe de $-a$. Si $x > x_2$~: les deux facteurs
sont positifs et $f(x)$ a à nouveau le signe de $a$.

\emph{2.} $f(x) = a(x - x_0)^2$ et $(x-x_0)^2 > 0$ pour $x \neq x_0$.

\emph{3.} D'après la démonstration du \cref{thm:g11:quad:roots},
$f(x) = a\left[\left(x + \frac{b}{2a}\right)^2 +
\left(-\frac{\Delta}{4a^2}\right)\right]$, et le crochet est toujours
positif.
\end{proof}

\begin{method}[Résoudre des inéquations du second degré]\label{met:g11:quad:inequalities}
Pour résoudre $ax^2 + bx + c \leq 0$ (ou $\geq$, $<$, $>$)~:
\begin{enumerate}
  \item calculer $\Delta$ et les racines, s'il y en a~;
  \item dresser le tableau de signes en utilisant le
        \cref{thm:g11:quad:sign} (signe de $a$ à l'extérieur des
        racines)~;
  \item lire le ou les intervalles de solutions, en veillant à ce que les
        racines elles-mêmes soient incluses (inégalité large) ou non
        (stricte).
\end{enumerate}
\end{method}

\begin{example}\label{ex:g11:quad:inequality}
Résoudre $-x^2 + 3x + 4 \geq 0$. Ici $\Delta = 9 + 16 = 25$, et les
racines sont $\frac{-3 \pm 5}{-2}$, c'est-à-dire $x_1 = -1$ et
$x_2 = 4$. Comme $a = -1 < 0$, le trinôme est négatif à l'extérieur des
racines et positif entre elles. Avec l'inégalité large, l'ensemble des
solutions est $\intcc{-1}{4}$.
\end{example}

\section{La parabole}

\begin{definition}[Parabole]\label{def:g11:quad:parabola}
Le graphique d'une fonction du second degré
$f(x) = a(x-\alpha)^2+\beta$ est une \emph{parabole}\index{parabole} de
\emph{sommet} $S(\alpha, \beta)$. Elle s'ouvre vers le haut si $a > 0$,
vers le bas si $a < 0$, et est symétrique par rapport à la droite
verticale $x = \alpha$.
\end{definition}

\begin{remark}
La symétrie est facile à vérifier~: pour tout $h$,
$f(\alpha + h) = ah^2 + \beta = f(\alpha - h)$, donc deux points à égale
distance horizontale de la droite $x = \alpha$ ont la même hauteur.
\end{remark}

\begin{method}[Lire une parabole]\label{met:g11:quad:reading}
Pour croquer $f(x) = ax^2+bx+c$~: trouver l'orientation (signe de $a$), le
sommet $\left(-\frac{b}{2a},\, f\!\left(-\frac{b}{2a}\right)\right)$,
l'intersection $(0, c)$ avec l'axe des $y$, et les racines, s'il y en a.
L'axe de symétrie $x = -\frac{b}{2a}$ place alors chaque point calculé en
double.
\end{method}

\begin{example}
Pour $f(x) = x^2 - 4x + 3$~: tournée vers le haut, sommet $(2, -1)$,
coupe l'axe des $y$ en $(0,3)$, racines $1$ et $3$ (repérées~:
$1 - 4 + 3 = 0$). Par symétrie, le point $(4, 3)$ est aussi sur la
courbe, miroir de $(0, 3)$.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g11:quad:1}
Résoudre les équations
\[
x^2 - 5x + 6 = 0, \qquad
2x^2 + 3x - 2 = 0, \qquad
x^2 + 2x + 5 = 0, \qquad
9x^2 - 6x + 1 = 0 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:quad:2}
Écrire sous forme canonique et donner le sommet de la parabole~:
\[
f(x) = x^2 - 6x + 11, \qquad
g(x) = -2x^2 + 4x + 1, \qquad
h(x) = 3x^2 + 6x .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:quad:3}
Résoudre les inéquations
\[
x^2 - 3x - 10 < 0, \qquad
2x^2 + x + 3 > 0, \qquad
-x^2 + 6x - 9 \geq 0 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:quad:4}
Sans calculer $\Delta$, trouver une racine évidente, puis l'autre~:
\[
x^2 - 7x + 6 = 0, \qquad
2x^2 + 5x - 7 = 0, \qquad
x^2 + 4x - 5 = 0 .
\]
(Indication~: tester $x = 1$ et $x = -1$, puis utiliser le produit des
racines.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:5}
Trouver deux nombres dont la somme est $14$ et le produit $45$. Puis
trouver les dimensions d'un rectangle de périmètre $34$ et d'aire $60$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:6}
Soit $f(x) = x^2 + (m-2)x + 1$, où $m$ est un paramètre réel. Pour
quelles valeurs de $m$ l'équation $f(x) = 0$ a-t-elle exactement une
solution~? Aucune solution~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:7}
Une balle est lancée vers le haut~; sa hauteur après $t$ secondes est
$h(t) = -5t^2 + 20t + 1$ (en mètres). Quelle est la hauteur maximale
atteinte, et à quel instant~? Pendant quel intervalle de temps la balle
est-elle à au moins $16$~m de hauteur~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:8}
Résoudre l'équation $x^4 - 13x^2 + 36 = 0$ en posant $X = x^2$ (une
équation \emph{bicarrée}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:9}
Pour quelles valeurs de $x$ le point $M(x, x^2)$ de la parabole
$y = x^2$ est-il strictement sous la droite $y = x + 2$~? Interpréter sur
un croquis.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:10}
La somme d'un nombre positif et de son inverse est $\frac{13}{6}$.
Trouver le nombre. (Se ramener à une équation du second degré.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:quad:11}
Soit $\mathcal P$ la parabole $y = x^2$ et soit $d_m$ la droite
$y = mx - 1$, où $m$ est un paramètre réel.
\begin{enumerate}
  \item Pour quelles valeurs de $m$ $\mathcal P$ et $d_m$ se coupent-elles
        en deux points~? Un point~? Aucun point~?
  \item Lorsqu'elles se touchent en exactement un point, calculer le
        point de contact. (Dans le \cref{ch:g11:deriv} on reconnaîtra
        $d_m$ comme la tangente à $\mathcal P$ en ce point.)
\end{enumerate}
\end{exercise}
