\chapter{La loi binomiale}\label{ch:g11:binom}

Répéter la même expérience oui/non plusieurs fois, indépendamment, et
compter les succès~: la loi résultante --- la \emph{binomiale} --- est la
plus importante de toutes les lois discrètes. Ce chapitre la construit
avec des arbres et le comptage de chemins~; la formule fermée pour les
comptages de chemins (avec factorielles) vient avec les outils de
dénombrement du \cref{ch:g12:comb}, et la loi est revue dans le
\cref{ch:g12:randvar}.

\section{Épreuves de Bernoulli}

\begin{definition}[Épreuve de Bernoulli]\label{def:g11:binom:bernoulli}
Une \emph{épreuve de Bernoulli}\index{épreuve de Bernoulli} est une
expérience à exactement deux issues~: \emph{succès}, de probabilité $p$,
et \emph{échec}, de probabilité $1 - p$. La variable aléatoire $X$ égale
à $1$ en cas de succès et $0$ en cas d'échec suit la \emph{loi de
Bernoulli} $\mathcal B(p)$~; alors
\[
\E(X) = p,
\qquad
\V(X) = p(1 - p).
\]
\end{definition}

\begin{proof}[Démonstration des deux formules]
$\E(X) = p \times 1 + (1-p) \times 0 = p$~; et comme $X^2 = X$ ($0$ et
$1$ sont leurs propres carrés), $\E(X^2) = p$, donc par la
\cref{prop:g11:prob:konig}, $\V(X) = p - p^2 = p(1-p)$.
\end{proof}

\begin{definition}[Épreuves indépendantes répétées]\label{def:g11:binom:repeated}
Répéter une épreuve de Bernoulli $n$ fois \emph{indépendamment} signifie~:
l'issue de chaque épreuve n'a aucune influence sur les autres, et la
probabilité de toute suite complète d'issues est le \emph{produit} des
probabilités le long du chemin correspondant de l'arbre --- $p$ pour
chaque succès, $1 - p$ pour chaque échec.
\end{definition}

\begin{example}\label{ex:g11:binom:threetrials}
Trois épreuves indépendantes de probabilité de succès $p$. La suite SES
(succès, échec, succès) a pour probabilité $p(1-p)p = p^2(1-p)$ --- et
il en va de même pour \emph{toute} suite avec exactement deux succès,
quelles que soient les positions~: seul le \emph{nombre} de S et de E
compte.
\end{example}

\section{Comptage de chemins et coefficients binomiaux}

\begin{definition}[Coefficient binomial]\label{def:g11:binom:coefficient}
Dans l'arbre de $n$ épreuves indépendantes, le \emph{coefficient
binomial}\index{coefficient binomial} $\binom{n}{k}$ (lu «~$n$ parmi
$k$~») est le nombre de chemins contenant exactement $k$ succès.
\end{definition}

\begin{example}
$\binom{3}{2} = 3$~: les chemins SSE, SES, ESS. De même
$\binom{3}{0} = 1$ (le chemin EEE), $\binom{3}{1} = 3$ et
$\binom{3}{3} = 1$. Par convention et par l'arbre,
$\binom n0 = \binom nn = 1$ pour tout $n$.
\end{example}

\begin{proposition}[Règle de Pascal]\label{prop:g11:binom:pascal}
\index{triangle de Pascal}
Pour $1 \leq k \leq n - 1$~:
\[
\binom{n}{k} = \binom{n-1}{k-1} + \binom{n-1}{k}.
\]
\end{proposition}

\begin{proof}
Classer les chemins de l'arbre à $n$ épreuves avec $k$ succès selon leur
\emph{dernière} épreuve. Ceux qui se terminent par un succès s'obtiennent
d'un chemin des $n-1$ premières épreuves avec $k - 1$ succès~: il y en a
$\binom{n-1}{k-1}$. Ceux qui se terminent par un échec prolongent un
chemin avec $k$ succès parmi les $n-1$ premières~: $\binom{n-1}{k}$.
Tout chemin est d'exactement l'un des deux types.
\end{proof}

La règle de Pascal engendre les coefficients ligne par ligne --- chaque
entrée est la somme des deux au-dessus d'elle~:
\[
\begin{array}{ccccccccccc}
&&&&&1&&&&&\\
&&&&1&&1&&&&\\
&&&1&&2&&1&&&\\
&&1&&3&&3&&1&&\\
&1&&4&&6&&4&&1&\\
1&&5&&10&&10&&5&&1
\end{array}
\]

\begin{remark}
Une formule fermée, $\binom nk = \frac{n!}{k!(n-k)!}$, avec une théorie
systématique du dénombrement, est établie dans le \cref{ch:g12:comb}. À
ce niveau, le triangle de Pascal calcule tout coefficient dont on a
besoin.
\end{remark}

\section{La loi binomiale}

\begin{theorem}[Loi binomiale]\label{thm:g11:binom:binomial}
Soit $X$ le nombre de succès dans $n$ épreuves de Bernoulli
indépendantes de paramètre $p$. Alors $X$ suit la \emph{loi
binomiale}\index{loi binomiale} $\mathcal B(n, p)$~:
\[
\P(X = k) = \binom{n}{k}\, p^k (1-p)^{n-k},
\qquad k = 0, 1, \dots, n .
\]
\end{theorem}

\begin{proof}
L'événement $X = k$ est la collection de tous les chemins avec
exactement $k$ succès. Chaque tel chemin a pour probabilité
$p^k(1-p)^{n-k}$~: le produit le long du chemin contient $k$ facteurs
$p$ et $n - k$ facteurs $1-p$, dans un certain ordre
(\cref{def:g11:binom:repeated}). Il y a $\binom nk$ tels chemins
(\cref{def:g11:binom:coefficient}), et leurs probabilités s'additionnent.
\end{proof}

\begin{example}\label{ex:g11:binom:quiz}
Un quiz a $5$ questions indépendantes, chacune à $4$ choix~; un élève
répond au hasard, donc chaque question est un succès avec
$p = \frac14$. Le nombre $X$ de bonnes réponses suit
$\mathcal B\left(5, \frac14\right)$, et, en utilisant la ligne~$5$ du
triangle de Pascal~:
\[
\P(X = 2) = \binom52 \left(\frac14\right)^2\left(\frac34\right)^3
= 10 \times \frac{1}{16} \times \frac{27}{64}
= \frac{270}{1024} \approx 0.26 .
\]
La probabilité d'au moins une bonne réponse utilise le complémentaire~:
$\P(X \geq 1) = 1 - \P(X = 0) = 1 - \left(\frac34\right)^5 \approx
0.76$.
\end{example}

\begin{proposition}[Espérance et variance]\label{prop:g11:binom:expectation}
Si $X \sim \mathcal B(n, p)$~:
\[
\E(X) = np,
\qquad
\V(X) = np(1-p).
\]
\end{proposition}

\begin{proof}[Justification]
Écrire $X = X_1 + X_2 + \dots + X_n$, où $X_i$ vaut $1$ si la $i$-ième
épreuve réussit~: chaque $X_i$ est une variable de Bernoulli d'espérance
$p$ (\cref{def:g11:binom:bernoulli}). Les moyennes s'additionnent ---
sommer les $n$ contributions donne $\E(X) = np$. Que les \emph{variances}
s'additionnent aussi pour des variables indépendantes est vrai mais plus
délicat~: la formule de variance est \emph{admise à ce niveau} et
prouvée dans le \cref{ch:g12:sums}.
\end{proof}

\begin{method}[Reconnaître une situation binomiale]\label{met:g11:binom:recognize}
Avant d'écrire $X \sim \mathcal B(n, p)$, vérifier trois ingrédients~:
un \emph{nombre fixe} $n$ d'épreuves, décidé à l'avance~; chaque épreuve
a \emph{deux issues} avec la \emph{même} probabilité de succès $p$~; les
épreuves sont \emph{indépendantes} (avec remise, ou dispositifs
séparés). Tirer sans remise dans une petite population n'est \emph{pas}
binomial --- la probabilité change à chaque tirage
(\cref{exo:g11:prob:6}).
\end{method}

\section{Échantillonnage~: l'observation est-elle surprenante~?}

La loi binomiale répond à une question très pratique~: \emph{si la
probabilité de succès est vraiment $p$, quels comptes de succès sont
plausibles~?}

\begin{example}\label{ex:g11:binom:decision}
Une machine est censée produire au plus $10\,\%$ de pièces défectueuses.
Dans un lot de $10$ pièces, $4$ sont défectueuses. Malchance ou machine
cassée~? Si la machine est en bon état, le nombre de défectueuses suit
$\mathcal B(10, 0.1)$, et
\[
\P(X \geq 4) = 1 - \P(X \leq 3) \approx 1 - 0.987 = 0.013 :
\]
environ une chance sur $80$. Observer un événement aussi improbable est
un signal fort --- on \emph{rejette} l'hypothèse que la machine fonctionne
encore à $10\,\%$, tout en gardant à l'esprit que la décision pourrait
être fausse avec une probabilité d'environ $0.013$.
\end{example}

\begin{method}[Règle de décision à partir d'un modèle binomial]\label{met:g11:binom:decision}
Pour juger un nombre observé $k$ de succès contre une hypothèse
$X \sim \mathcal B(n, p)$~: calculer la probabilité, sous l'hypothèse,
d'un résultat \emph{au moins aussi extrême} que $k$. Si cette
probabilité est très petite (une convention courante~: sous $5\,\%$),
rejeter l'hypothèse~; sinon l'observation est compatible avec elle. Le
seuil est un choix, pas un théorème --- la statistique quantifie le
risque, et l'utilisateur l'accepte.
\end{method}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g11:binom:1}
Prolonger le triangle de Pascal jusqu'à la ligne~$7$, et donner les
valeurs de $\binom62$, $\binom{7}{3}$ et $\binom74$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:binom:2}
Un dé équilibré est lancé $4$ fois~; $X$ compte les six. Justifier que
$X \sim \mathcal B\left(4, \frac16\right)$ et calculer $\P(X = 0)$,
$\P(X = 1)$ et $\P(X \geq 2)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:binom:3}
Lequel des cas suivants est binomial~? Justifier.
\begin{enumerate}
  \item Nombre de faces dans $20$ lancers d'une pièce équilibrée.
  \item Nombre d'as dans $5$ cartes distribuées d'un jeu.
  \item Nombre de jours de pluie la semaine prochaine, si chaque jour est
        pluvieux avec probabilité $0.3$ indépendamment.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:binom:4}
$X \sim \mathcal B(50, 0.2)$. Donner $\E(X)$, $\V(X)$ et $\sigma(X)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:binom:5}
Un archer touche la cible avec probabilité $0.7$ à chaque tir,
indépendamment. En $6$ tirs, calculer la probabilité d'exactement $4$
touches, et d'au moins $5$ touches.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:binom:6}
Un test vrai/faux a $8$ questions~; un élève devine chaque réponse.
Quelle est la probabilité de réussir (au moins $6$ bonnes réponses)~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:binom:7}
Chaque boîte de céréales achetée contient la figurine A ou la figurine B,
avec probabilité $\frac12$ chacune, indépendamment. Un collectionneur
achète $5$ boîtes. Calculer la probabilité qu'il obtienne au moins une
figurine de chaque type. (Complémentaire~: toutes A ou toutes B.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:binom:8}
Un basketteur marque des lancers francs avec probabilité $p$,
indépendamment. Soit $X \sim \mathcal B(3, p)$ le nombre de réussites en
trois tirs. Exprimer $\P(X = 3)$ et $\P(X \geq 1)$ en fonction de $p$,
et trouver pour quel $p$ la probabilité de marquer les trois égale
$\frac{27}{64}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:binom:9}
Combien de fois faut-il lancer une pièce équilibrée pour que la
probabilité d'obtenir au moins une face dépasse $0.99$~?
(Complémentaire, puis essayer des valeurs successives de $n$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:binom:10}
En utilisant la règle de Pascal (\cref{prop:g11:binom:pascal}) et
$\binom n0 = \binom nn = 1$, prouver que les entrées de chaque ligne du
triangle de Pascal somment à $2^n$~: interpréter les deux côtés comme le
comptage de tous les chemins de l'arbre.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:binom:11}
Un politicien prétend avoir $60\,\%$ d'approbation. Dans un échantillon
aléatoire de $10$ personnes, seulement $3$ approuvent.
\begin{enumerate}
  \item Sous l'affirmation, quelle loi suit le nombre $X$
        d'approbations dans l'échantillon~? Calculer $\P(X \leq 3)$.
  \item En utilisant la règle de décision de la
        \cref{met:g11:binom:decision} avec un seuil de $5\,\%$,
        l'observation est-elle compatible avec l'affirmation~?
\end{enumerate}
\end{exercise}
