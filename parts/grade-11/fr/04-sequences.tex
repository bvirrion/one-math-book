\chapter{Suites~: un premier cours}\label{ch:g11:seq}

Une \omterm{def:g11:seq:sequence}{suite} est une liste de nombres produits par une règle~: les soldes
successifs d'un compte d'épargne, les tailles d'une population année
après année. Ce chapitre étudie les deux familles qui dominent les
applications --- les \omterm{def:g11:seq:sequence}{suites} \emph{\omterm{def:g11:seq:arithmetic}{arithmétiques}}, qui croissent par pas
égaux, et les \omterm{def:g11:seq:sequence}{suites} \emph{\omterm{def:g11:seq:geometric}{géométriques}}, qui croissent par rapports
égaux. La théorie rigoureuse des limites est développée dans le
\cref{ch:g12:seq}.

\section{Définir une suite}

\begin{definition}[Suite]\label{def:g11:seq:sequence}
Une \emph{suite}\index{suite} $(u_n)$ associe à chaque entier $n \geq 0$
(ou $n \geq 1$) un \omterm{def:g10:numbers:sets}{nombre réel} $u_n$, son \emph{terme d'indice $n$}. Une
suite peut être donnée
\begin{itemize}
  \item \emph{explicitement}, par une formule de $u_n$ en \omterm{def:g11:func:function}{fonction} de
        $n$~: par exemple $u_n = n^2 + 1$~;
  \item \emph{par récurrence}, par son premier terme et une règle pour
        passer de chaque terme au suivant~: par exemple $u_0 = 3$ et
        $u_{n+1} = 2u_n - 1$.
\end{itemize}
\end{definition}

\begin{example}
Pour $u_n = n^2 + 1$~: $u_0 = 1$, $u_1 = 2$, $u_2 = 5$, et $u_{10} = 101$
directement. Pour $u_0 = 3$, $u_{n+1} = 2u_n - 1$~: $u_1 = 5$, $u_2 = 9$,
$u_3 = 17$ --- chaque terme a besoin du précédent~; atteindre $u_{10}$
prend dix étapes (ou une formule générale, voir
l'\cref{exo:g11:seq:11}).
\end{example}

\section{Suites arithmétiques}

\begin{definition}[Suite arithmétique]\label{def:g11:seq:arithmetic}
Une \omterm{def:g11:seq:sequence}{suite} est \emph{arithmétique}\index{suite arithmétique} de
\emph{raison}\index{raison} $d$ si chaque terme s'obtient du précédent en ajoutant
$d$~:
\[
u_{n+1} = u_n + d \quad \text{pour tout } n.
\]
De façon équivalente~: la différence $u_{n+1} - u_n$ est constante,
égale à $d$.
\end{definition}

\begin{theorem}[Terme général]\label{thm:g11:seq:arithgeneral}
Si $(u_n)$ est \omterm{def:g11:seq:arithmetic}{arithmétique} de premier terme $u_0$ et de raison $d$,
alors
\[
u_n = u_0 + n\,d \quad \text{pour tout } n \geq 0,
\qquad\text{et plus généralement } u_n = u_p + (n - p)\,d .
\]
\end{theorem}

\begin{proof}
Pour aller de $u_0$ à $u_n$, la règle «~ajouter $d$~» est appliquée $n$
fois~: une étape donne $u_1 = u_0 + d$, deux étapes donnent
$u_2 = u_0 + 2d$, et après $n$ étapes chaque application a contribué un
$d$, donc $u_n = u_0 + nd$. (Ce «~et ainsi de \omterm{def:g11:seq:sequence}{suite}~» est rendu
rigoureux par récurrence dans le \cref{ch:g12:seq}.) La formule générale
suit en comptant les $n - p$ étapes de $u_p$ à $u_n$.
\end{proof}

\begin{theorem}[Somme des entiers consécutifs]\label{thm:g11:seq:intsum}
Pour tout entier $n \geq 1$~:
\[
1 + 2 + \dots + n = \frac{n(n+1)}{2}.
\]
Plus généralement, une somme de termes consécutifs d'une \omterm{def:g11:seq:arithmetic}{suite
arithmétique} égale
\[
(\text{nombre de termes}) \times
\frac{\text{premier terme} + \text{dernier terme}}{2}.
\]
\end{theorem}

\begin{proof}
Écrire la somme $S$ deux fois, la seconde à l'envers, et additionner
colonne par colonne~:
\[
\begin{array}{ccccccccc}
S & = & 1 & + & 2 & + & \dots & + & n\\
S & = & n & + & (n-1) & + & \dots & + & 1\\
\hline
2S & = & (n+1) & + & (n+1) & + & \dots & + & (n+1)
\end{array}
\]
Il y a $n$ colonnes, chacune sommant à $n + 1$, donc $2S = n(n+1)$. Pour
une \omterm{def:g11:seq:arithmetic}{suite arithmétique} générale le même appariement fonctionne~: premier
$+$ dernier $=$ second $+$ avant-dernier $= \dots$, car avancer d'un pas
à gauche ($+d$) est compensé par reculer d'un pas à droite ($-d$).
\end{proof}

\begin{example}
$1 + 2 + \dots + 100 = \frac{100 \times 101}{2} = 5050$. La somme des
impairs $1 + 3 + \dots + 99$ ($50$ termes) est
$50 \times \frac{1 + 99}{2} = 2500$.
\end{example}

\section{Suites géométriques}

\begin{definition}[Suite géométrique]\label{def:g11:seq:geometric}
Une \omterm{def:g11:seq:sequence}{suite} est \emph{géométrique}\index{suite géométrique} de
\emph{raison}\index{raison} $q \neq 0$ si chaque terme s'obtient du précédent en
multipliant par $q$~:
\[
u_{n+1} = q\,u_n \quad \text{pour tout } n.
\]
De façon équivalente, lorsqu'aucun terme ne s'annule~: le rapport
$\frac{u_{n+1}}{u_n}$ est constant, égal à $q$.
\end{definition}

\begin{theorem}[Terme général]\label{thm:g11:seq:geomgeneral}
Si $(u_n)$ est \omterm{def:g11:seq:geometric}{géométrique} de premier terme $u_0$ et de raison $q$,
alors
\[
u_n = u_0\, q^n \quad \text{pour tout } n \geq 0,
\qquad\text{et plus généralement } u_n = u_p\, q^{\,n-p} .
\]
\end{theorem}

\begin{proof}
Même comptage d'étapes que dans le \cref{thm:g11:seq:arithgeneral}~: de
$u_0$ à $u_n$, la règle «~multiplier par $q$~» est appliquée $n$ fois,
contribuant un facteur $q^n$.
\end{proof}

\begin{theorem}[Somme géométrique]\label{thm:g11:seq:geomsum}
Pour tout réel $q \neq 1$ et tout entier $n \geq 0$~:
\[
1 + q + q^2 + \dots + q^n = \frac{1 - q^{\,n+1}}{1 - q}.
\]
\end{theorem}

\begin{proof}
Soit $S = 1 + q + \dots + q^n$. Multiplier par $q$~:
$qS = q + q^2 + \dots + q^{n+1}$. Soustraire~:
\[
S - qS = \bigl(1 + q + \dots + q^n\bigr)
- \bigl(q + q^2 + \dots + q^{n+1}\bigr) = 1 - q^{\,n+1},
\]
car chaque terme intermédiaire apparaît une fois dans chaque somme et
s'annule. Donc $(1 - q)S = 1 - q^{\,n+1}$, et diviser par $1 - q \neq 0$
donne la formule.
\end{proof}

\begin{example}\label{ex:g11:seq:chessboard}
$1 + 2 + 4 + \dots + 2^{10} = \frac{1 - 2^{11}}{1 - 2} = 2^{11} - 1 =
2047$~: doubler des grains de riz sur les cases d'un échiquier
submerge n'importe quel grenier bien avant la $64$-ième case, où le
total est $2^{64} - 1 \approx 1.8 \times 10^{19}$.
\end{example}

\begin{method}[Reconnaître le type d'une suite]\label{met:g11:seq:recognize}
Calculer $u_{n+1} - u_n$ et simplifier. Si le résultat est une constante
$d$, la \omterm{def:g11:seq:sequence}{suite} est \omterm{def:g11:seq:arithmetic}{arithmétique}. Sinon calculer $\frac{u_{n+1}}{u_n}$
(termes non nuls) et simplifier~: une constante $q$ signifie
\omterm{def:g11:seq:geometric}{géométrique}. Si ni l'un ni l'autre n'est constant, la \omterm{def:g11:seq:sequence}{suite} n'est d'aucun
des deux types --- ne jamais conclure des seuls premiers termes.
\end{method}

\begin{example}
Pour $u_n = 3 \times 5^n$~:
$\frac{u_{n+1}}{u_n} = \frac{3 \times 5^{n+1}}{3 \times 5^n} = 5$ pour
tout $n$~: \omterm{def:g11:seq:geometric}{géométrique} de raison $5$. Pour $u_n = n^2$~:
$u_1 - u_0 = 1$ mais $u_2 - u_1 = 3$, et $\frac{u_1}{u_0}$ n'est même
pas défini --- ni \omterm{def:g11:seq:arithmetic}{arithmétique} ni \omterm{def:g11:seq:geometric}{géométrique}.
\end{example}

\section{Monotonie}

\begin{definition}[Suite monotone]\label{def:g11:seq:monotone}
Une \omterm{def:g11:seq:sequence}{suite} $(u_n)$ est \emph{\omterm{def:g11:func:monotone}{croissante}} si $u_{n+1} \geq u_n$ pour tout
$n$, et \emph{\omterm{def:g10:functions:variations}{décroissante}} si $u_{n+1} \leq u_n$ pour tout $n$.
\end{definition}

\begin{method}[Étudier la monotonie]\label{met:g11:seq:monotonicity}
Étudier le signe de $u_{n+1} - u_n$. Pour des \omterm{def:g11:seq:sequence}{suites} à termes positifs,
on peut à la place comparer $\frac{u_{n+1}}{u_n}$ à $1$.
\end{method}

\begin{example}
Une \omterm{def:g11:seq:arithmetic}{suite arithmétique} est \omterm{def:g11:func:monotone}{croissante} lorsque $d \geq 0$
($u_{n+1} - u_n = d$), \omterm{def:g10:functions:variations}{décroissante} lorsque $d \leq 0$. Une \omterm{def:g11:seq:geometric}{suite
géométrique} avec $u_0 > 0$ et $q > 1$ est \omterm{def:g11:func:monotone}{croissante}~:
$u_{n+1} - u_n = u_0 q^n (q - 1) > 0$~; avec $u_0 > 0$ et $0 < q < 1$
elle est \omterm{def:g10:functions:variations}{décroissante}.
\end{example}

\section{Comportement à long terme, informellement}

Que devient $u_n$ lorsque $n$ devient très grand~? Pour une \omterm{def:g11:seq:arithmetic}{suite
arithmétique} avec $d > 0$, les termes $u_0 + nd$ dépassent tout nombre
fixé éventuellement. Pour une \omterm{def:g11:seq:geometric}{suite géométrique} avec $0 < q < 1$, les
termes $u_0 q^n$ se contractent vers $0$~: multiplier à répétition par
$0.9$, disons, érode toute valeur de départ. Et pour $q > 1$ les termes
explosent, comme dans l'\cref{ex:g11:seq:chessboard}.

\begin{remark}
Ces affirmations peuvent être rendues parfaitement précises --- «~les
termes finissent par rester à toute distance donnée de $0$~» --- et
prouvées. C'est la théorie des \emph{limites}, le thème d'ouverture du
\cref{ch:g12:seq}.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g11:seq:1}
Pour chaque \omterm{def:g11:seq:sequence}{suite}, calculer $u_1$, $u_2$, $u_3$~:
\[
u_n = \frac{n}{n+1}; \qquad
u_0 = 5,\ u_{n+1} = 3u_n - 2; \qquad
u_n = (-1)^n\,n .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:seq:2}
$(u_n)$ est \omterm{def:g11:seq:arithmetic}{arithmétique} avec $u_0 = 7$ et $d = -3$. Calculer $u_{10}$
et $u_{25}$. $(v_n)$ est \omterm{def:g11:seq:arithmetic}{arithmétique} avec $v_3 = 11$ et $v_8 = 26$.
Trouver la raison et $v_0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:seq:3}
$(u_n)$ est \omterm{def:g11:seq:geometric}{géométrique} avec $u_0 = 5$ et $q = 2$. Calculer $u_8$.
$(v_n)$ est \omterm{def:g11:seq:geometric}{géométrique} à termes positifs, $v_2 = 12$ et $v_4 = 48$.
Trouver la raison et $v_0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:seq:4}
Calculer
\[
1 + 2 + 3 + \dots + 500, \qquad
4 + 7 + 10 + \dots + 61, \qquad
1 + \frac12 + \frac14 + \dots + \frac{1}{2^{10}} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:seq:5}
Déterminer si chaque \omterm{def:g11:seq:sequence}{suite} est \omterm{def:g11:seq:arithmetic}{arithmétique}, \omterm{def:g11:seq:geometric}{géométrique}, ou ni l'une
ni l'autre~:
\[
u_n = 4n - 1; \qquad
v_n = \frac{2^n}{3^{n+1}}; \qquad
w_n = n^2 + n .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:seq:6}
Un théâtre a $20$ rangées~: $16$ sièges dans la première, et chaque
rangée a $2$ sièges de plus que la précédente. Combien de sièges dans
la dernière rangée~? Dans tout le théâtre~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:seq:7}
Une population de bactéries double toutes les heures~; à midi il y en a
$500$. Combien y en a-t-il à 20~h~? Au bout de combien d'heures
complètes la population dépasse-t-elle d'abord un million~? (Résoudre
en essayant des puissances successives de $2$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:seq:8}
Chaque mois, un épargnant dépose $100$ euros sur un compte qui rapporte
$0.2\,\%$ d'intérêts par mois sur le solde existant (les intérêts sont
crédités juste avant le dépôt). Soit $c_n$ le solde juste après le
$n$-ième dépôt, donc $c_1 = 100$ et $c_{n+1} = 1.002\,c_n + 100$.
Calculer $c_2$ et $c_3$, et expliquer pourquoi $(c_n)$ n'est ni
\omterm{def:g11:seq:arithmetic}{arithmétique} ni \omterm{def:g11:seq:geometric}{géométrique}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:seq:9}
Étudier la monotonie des \omterm{def:g11:seq:sequence}{suites}
\[
u_n = n^2 - 8n \ (n \geq 0), \qquad
v_n = \frac{3^n}{n!}\ (n \geq 1),
\]
où $n! = 1 \times 2 \times \dots \times n$. (Pour $(v_n)$, comparer
$\frac{v_{n+1}}{v_n}$ à $1$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:seq:10}
La somme des $n$ premiers termes d'une \omterm{def:g11:seq:arithmetic}{suite arithmétique} avec
$u_0 = 3$ et $d = 4$ égale $903$. Trouver $n$. (Mettre en place une
\omterm{def:g10:algebra:equation}{équation} du second degré en $n$ et utiliser le \cref{ch:g11:quad}.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:seq:11}
Soit $u_0 = 3$ et $u_{n+1} = 2u_n - 1$.
\begin{enumerate}
  \item Calculer $u_1, u_2, u_3$ et conjecturer une formule pour $u_n$.
  \item Soit $v_n = u_n - 1$. Montrer que $(v_n)$ est \omterm{def:g11:seq:geometric}{géométrique}, donner
        sa raison et son premier terme.
  \item En déduire une formule explicite pour $u_n$ et vérifier la
        conjecture.
\end{enumerate}
\end{exercise}
