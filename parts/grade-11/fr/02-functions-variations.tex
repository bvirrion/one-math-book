\chapter{Fonctions et variations}\label{ch:g11:func}

Ce chapitre organise ce qu'il faut savoir sur les \omterm{def:g11:func:function}{fonctions} avant le
calcul différentiel~: la petite famille des \emph{\omterm{def:g11:func:reference}{fonctions de référence}}
dont les \omterm{def:g10:functions:graph}{graphiques} doivent être connus par cœur, le sens précis de
\emph{\omterm{def:g11:func:monotone}{croissante}} et \emph{\omterm{def:g10:functions:variations}{décroissante}}, et le comportement des
variations lorsque l'on combine des \omterm{def:g11:func:function}{fonctions}.

\section{Fonctions et courbes}

\begin{definition}[Fonction, ensemble de définition, courbe]\label{def:g11:func:function}
Une \emph{fonction}\index{fonction} $f$ associe à chaque nombre $x$ d'un
ensemble $D \subseteq \R$ (son \emph{ensemble de
définition}\index{ensemble de définition}) exactement un nombre $f(x)$.
Sa \emph{courbe} (ou \omterm{def:g10:functions:graph}{graphique}) est l'ensemble des points
$\bigl(x, f(x)\bigr)$ pour $x \in D$.
\end{definition}

\begin{example}
La formule $f(x) = \frac{\sqrt{x+2}}{x-1}$ exige $x + 2 \geq 0$ (pour la
racine carrée) et $x \neq 1$ (dénominateur non nul)~: l'\omterm{def:g11:func:function}{ensemble de
définition} est $\intco{-2}{1} \cup \intoo{1}{+\infty}$.
\end{example}

\section{Les fonctions de référence}

Les cinq \omterm{def:g11:func:function}{fonctions} suivantes et leurs courbes doivent devenir des
réflexes.

\begin{definition}[Fonctions de référence]\label{def:g11:func:reference}
\index{fonctions de référence}
\[
x \mapsto x^2 \text{ (carré)}, \quad
x \mapsto x^3 \text{ (cube)}, \quad
x \mapsto \sqrt{x}, \quad
x \mapsto \frac{1}{x}, \quad
x \mapsto \abs{x}.
\]
Leurs \omterm{def:g11:func:function}{ensembles de définition} sont $\R$, $\R$, $\intco{0}{+\infty}$,
$\R \setminus \{0\}$ et $\R$ respectivement. La courbe de
$x \mapsto \frac1x$ est une \emph{hyperbole}\index{hyperbole}.
\end{definition}

\section{Variations sur un intervalle}

\begin{definition}[Croissante, décroissante]\label{def:g11:func:monotone}
Soit $f$ définie sur un \omterm{def:g10:numbers:interval}{intervalle} $I$. La \omterm{def:g11:func:function}{fonction} $f$ est
\emph{croissante}\index{fonction croissante} sur $I$ si elle préserve
l'ordre~:
\[
\text{pour tous } u, v \in I: \quad u < v \implies f(u) \leq f(v),
\]
et \emph{\omterm{def:g10:functions:variations}{décroissante}} si elle inverse l'ordre ($u < v \implies
f(u) \geq f(v)$). Avec des inégalités strictes à droite, $f$ est
\emph{strictement} croissante (resp.\ \omterm{def:g10:functions:variations}{décroissante}). Une \omterm{def:g11:func:function}{fonction} qui
est croissante ou \omterm{def:g10:functions:variations}{décroissante} sur $I$ est \emph{monotone} sur $I$.
\end{definition}

\begin{method}[Démontrer une variation à partir de la définition]\label{met:g11:func:variation}
Prendre deux nombres arbitraires $u < v$ dans l'\omterm{def:g10:numbers:interval}{intervalle}, et
déterminer le signe de la différence $f(v) - f(u)$, en général en la
\emph{factorisant}. Si le signe est constant sur l'\omterm{def:g10:numbers:interval}{intervalle}, conclure.
\end{method}

\begin{proposition}[Variations des fonctions de référence]\label{prop:g11:func:refvariations}
\begin{enumerate}
  \item $x^2$ est strictement \omterm{def:g10:functions:variations}{décroissante} sur $\intoc{-\infty}{0}$ et
        strictement \omterm{def:g11:func:monotone}{croissante} sur $\intco{0}{+\infty}$.
  \item $x^3$ est strictement \omterm{def:g11:func:monotone}{croissante} sur $\R$.
  \item $\sqrt x$ est strictement \omterm{def:g11:func:monotone}{croissante} sur $\intco{0}{+\infty}$.
  \item $\frac1x$ est strictement \omterm{def:g10:functions:variations}{décroissante} sur $\intoo{-\infty}{0}$
        et sur $\intoo{0}{+\infty}$ (mais \emph{pas} sur leur \omterm{def:g10:numbers:interunion}{réunion}).
  \item $\abs x$ est strictement \omterm{def:g10:functions:variations}{décroissante} sur $\intoc{-\infty}{0}$ et
        strictement \omterm{def:g11:func:monotone}{croissante} sur $\intco{0}{+\infty}$.
\end{enumerate}
\end{proposition}

\begin{proof}
Soit $u < v$. On factorise $f(v) - f(u)$ dans chaque cas.

\emph{1.} $v^2 - u^2 = (v-u)(v+u)$. Sur $\intco{0}{+\infty}$~:
$v - u > 0$ et $v + u > 0$ (car $v > u \geq 0$), donc $v^2 > u^2$. Sur
$\intoc{-\infty}{0}$~: $v + u < 0$, donc $v^2 < u^2$.

\emph{2.} $v^3 - u^3 = (v - u)(v^2 + uv + u^2)$, et le second facteur
égale $\left(v + \frac u2\right)^2 + \frac{3u^2}{4} > 0$ sauf si
$u = v = 0$. Donc $v^3 > u^3$ dès que $v > u$.

\emph{3.} Pour $0 \leq u < v$~:
$\sqrt v - \sqrt u = \frac{(\sqrt v - \sqrt u)(\sqrt v + \sqrt u)}
{\sqrt v + \sqrt u} = \frac{v - u}{\sqrt v + \sqrt u} > 0$.

\emph{4.} Pour $0 < u < v$~:
$\frac1v - \frac1u = \frac{u - v}{uv} < 0$ car $u - v < 0$ et $uv > 0$~;
même calcul pour $u < v < 0$. Sur la \omterm{def:g10:numbers:interunion}{réunion} l'assertion échoue~:
$-1 < 1$ mais $\frac{1}{-1} = -1 < 1 = \frac11$.

\emph{5.} Sur $\intco{0}{+\infty}$, $\abs x = x$ est strictement
\omterm{def:g11:func:monotone}{croissante}~; sur $\intoc{-\infty}{0}$, $\abs x = -x$ est strictement
\omterm{def:g10:functions:variations}{décroissante}.
\end{proof}

\begin{remark}
Le point~4 est un piège classique~: «~\omterm{def:g10:functions:variations}{décroissante} sur
$\intoo{-\infty}{0}$ et sur $\intoo{0}{+\infty}$~» n'est \emph{pas} la
même chose que «~\omterm{def:g10:functions:variations}{décroissante} sur $\R \setminus \{0\}$~». Les variations
n'ont de sens que sur des \omterm{def:g10:numbers:interval}{intervalles}.
\end{remark}

\begin{example}[Comparer sans calculer]
Comme $\sqrt x$ est strictement \omterm{def:g11:func:monotone}{croissante}, $\sqrt{7} < \sqrt{8}$~; comme
$x^2$ est strictement \omterm{def:g10:functions:variations}{décroissante} sur $\intoc{-\infty}{0}$,
$(-3)^2 > (-2)^2$. La monotonie transforme les comparaisons d'\omterm{def:g10:functions:function}{images} en
comparaisons d'arguments.
\end{example}

\section{Nouvelles fonctions à partir d'anciennes}

\begin{proposition}[Transformations]\label{prop:g11:func:transforms}
Soit $u$ une \omterm{def:g11:func:function}{fonction} définie sur un \omterm{def:g10:numbers:interval}{intervalle} $I$ et $k \in \R$.
\begin{enumerate}
  \item $u + k$ a les \emph{mêmes} variations que $u$ sur $I$~; sa courbe
        est celle de $u$ décalée verticalement de $k$.
  \item Pour $\lambda > 0$, $\lambda u$ a les mêmes variations que $u$~;
        pour $\lambda < 0$, les variations sont inversées.
  \item Si $u \geq 0$ sur $I$, alors $\sqrt u$ a les mêmes variations que
        $u$.
  \item Si $u \neq 0$ et $u$ a un signe constant sur $I$, alors
        $\frac1u$ a les variations \emph{opposées} à celles de $u$.
\end{enumerate}
\end{proposition}

\begin{proof}
Soient $u < v$ dans $I$ (en abusant de notation, on compare des valeurs
de $x$ $u < v$~; écrire $s = u(x_1)$, $t = u(x_2)$ pour les \omterm{def:g10:functions:function}{images} de
$x_1 < x_2$).

\emph{1.} $(u+k)(x_2) - (u+k)(x_1) = t - s$~: la différence des \omterm{def:g10:functions:function}{images}
est inchangée, donc son signe l'est aussi.

\emph{2.} $\lambda t - \lambda s = \lambda(t - s)$~: le signe est
conservé si $\lambda > 0$, inversé si $\lambda < 0$.

\emph{3.} Si $u$ est \omterm{def:g11:func:monotone}{croissante} et $x_1 < x_2$, alors $0 \leq s \leq t$,
et $\sqrt{}$ étant \omterm{def:g11:func:monotone}{croissante} (\cref{prop:g11:func:refvariations}) donne
$\sqrt s \leq \sqrt t$. Même argument dans le cas décroissant.

\emph{4.} Si $u$ est \omterm{def:g11:func:monotone}{croissante}, $0 < s \leq t$ (ou $s \leq t < 0$),
alors $\frac1s \geq \frac1t$ car $x \mapsto \frac1x$ est \omterm{def:g10:functions:variations}{décroissante}
de chaque côté de $0$~: l'ordre est inversé.
\end{proof}

\begin{example}\label{ex:g11:func:composite}
Étudier $f(x) = \sqrt{x^2 + 1}$ sur $\R$. La \omterm{def:g11:func:function}{fonction} intérieure
$u(x) = x^2 + 1$ est positive, \omterm{def:g10:functions:variations}{décroissante} sur $\intoc{-\infty}{0}$ et
\omterm{def:g11:func:monotone}{croissante} sur $\intco{0}{+\infty}$ (un carré décalé). Par le point~3,
$f$ a les mêmes variations~: \omterm{def:g10:functions:variations}{décroissante} puis \omterm{def:g11:func:monotone}{croissante}, avec \omterm{def:g10:functions:extrema}{minimum}
$f(0) = 1$.
\end{example}

\section{Fonctions paires et impaires}

\begin{definition}[Parité]\label{def:g11:func:parity}
Soit $f$ définie sur un \omterm{def:g11:func:function}{ensemble de définition} $D$ symétrique par
rapport à $0$ (c'est-à-dire $-x \in D$ dès que $x \in D$). La \omterm{def:g11:func:function}{fonction}
$f$ est \emph{paire}\index{fonction paire} si $f(-x) = f(x)$ pour tout
$x \in D$, et \emph{impaire}\index{fonction impaire} si
$f(-x) = -f(x)$ pour tout $x \in D$.
\end{definition}

\begin{proposition}[Sens géométrique]\label{prop:g11:func:paritysym}
La courbe d'une \omterm{def:g11:func:parity}{fonction paire} est symétrique par rapport à l'axe des
$y$~; la courbe d'une \omterm{def:g11:func:parity}{fonction impaire} est symétrique par rapport à
l'origine.
\end{proposition}

\begin{proof}
Si $f$ est \omterm{def:g11:func:parity}{paire} et $(x, y)$ est sur la courbe (donc $y = f(x)$), alors
$f(-x) = f(x) = y$~: le point miroir $(-x, y)$ est aussi sur la courbe.
Si $f$ est \omterm{def:g11:func:parity}{impaire}, $f(-x) = -y$~: le point $(-x, -y)$, symétrique de
$(x,y)$ par rapport à l'origine, est sur la courbe.
\end{proof}

\begin{example}
$x^2$ et $\abs x$ sont \omterm{def:g11:func:parity}{paires}~; $x^3$ et $\frac1x$ sont \omterm{def:g11:func:parity}{impaires}~;
$f(x) = x^2 + x$ n'est ni l'une ni l'autre~: $f(-1) = 0$ tandis que
$f(1) = 2$, donc $f(-1) \neq \pm f(1)$. La parité divise le travail par
deux~: une \omterm{def:g11:func:parity}{fonction paire} ou \omterm{def:g11:func:parity}{impaire} n'a besoin d'être étudiée que sur
$\intco{0}{+\infty}$.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g11:func:1}
Donner l'\omterm{def:g11:func:function}{ensemble de définition} de chaque \omterm{def:g11:func:function}{fonction}~:
\[
f(x) = \sqrt{3 - x}, \qquad
g(x) = \frac{1}{x^2 - 4}, \qquad
h(x) = \frac{\sqrt{x}}{x - 5} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:func:2}
En utilisant les variations des \omterm{def:g11:func:reference}{fonctions de référence}, comparer sans
calculatrice~:
\[
\sqrt{11} \text{ et } \sqrt{13}; \qquad
(-2.1)^2 \text{ et } (-2.05)^2; \qquad
\frac{1}{0.9} \text{ et } \frac{1}{1.1}; \qquad
(-1.01)^3 \text{ et } (-0.99)^3 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:func:3}
Déterminer si chaque \omterm{def:g11:func:function}{fonction} est \omterm{def:g11:func:parity}{paire}, \omterm{def:g11:func:parity}{impaire}, ou ni l'une ni
l'autre~:
\[
f(x) = 3x^4 - x^2, \qquad
g(x) = x^3 + x, \qquad
h(x) = x^2 + x^3, \qquad
k(x) = \frac{x}{x^2+1} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:func:4}
Une \omterm{def:g11:func:function}{fonction} $f$ a le comportement en variations suivant~: \omterm{def:g11:func:monotone}{croissante} sur
$\intcc{-3}{1}$ de $f(-3) = -2$ à $f(1) = 4$, puis \omterm{def:g10:functions:variations}{décroissante} sur
$\intcc{1}{5}$ jusqu'à $f(5) = 0$. Combien de solutions l'\omterm{def:g10:algebra:equation}{équation}
$f(x) = 1$ a-t-elle sur $\intcc{-3}{5}$~? Et $f(x) = 5$~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:func:5}
En utilisant la \cref{met:g11:func:variation} (signe de $f(v) - f(u)$),
prouver que $f(x) = x^2 - 4x$ est strictement \omterm{def:g11:func:monotone}{croissante} sur
$\intco{2}{+\infty}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:func:6}
Sans calculer aucune dérivée, donner les variations de
\begin{gather*}
f(x) = \sqrt{5 - x} \ \text{ sur } \intoc{-\infty}{5},
\qquad
g(x) = \frac{1}{x^2 + 1} \ \text{ sur } \intco{0}{+\infty},
\\
h(x) = 3 - 2\sqrt{x} \ \text{ sur } \intco{0}{+\infty}.
\end{gather*}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:func:7}
Soit $f(x) = (x-2)^2 - 3$ sur $\R$.
\begin{enumerate}
  \item Donner les variations de $f$ et son \omterm{def:g10:functions:extrema}{minimum}.
  \item En déduire les variations de $g(x) = \frac{1}{(x-2)^2+1}$ et son
        \omterm{def:g10:functions:extrema}{maximum}. (Noter $g = \frac{1}{f + 4}$.)
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:func:8}
La courbe d'une \omterm{def:g11:func:function}{fonction} $f$ passe par les points $(0, 1)$, $(2, 3)$ et
$(4, 1)$, et $f$ est \omterm{def:g11:func:monotone}{croissante} sur $\intcc{0}{2}$ et \omterm{def:g10:functions:variations}{décroissante} sur
$\intcc{2}{4}$. Croquer une courbe possible, puis croquer les courbes de
$f + 2$, de $-f$, et de $2f$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:func:9}
Montrer que la \omterm{def:g11:func:function}{fonction} $f(x) = \frac{2x+1}{x+1}$ peut s'écrire
$f(x) = 2 - \frac{1}{x+1}$, et en déduire ses variations sur
$\intoo{-1}{+\infty}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:func:10}
Vrai ou faux (justifier ou donner un contre-exemple)~:
\begin{enumerate}
  \item La somme de deux \omterm{def:g11:func:monotone}{fonctions croissantes} sur $I$ est \omterm{def:g11:func:monotone}{croissante} sur
        $I$.
  \item Le produit de deux \omterm{def:g11:func:monotone}{fonctions croissantes} sur $I$ est croissant
        sur $I$.
  \item Si $f$ est \omterm{def:g11:func:monotone}{croissante} sur $I$, alors $f^2$ ($= f \times f$) est
        \omterm{def:g11:func:monotone}{croissante} sur $I$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:func:11}
Soit $f(x) = x + \frac{1}{x}$ pour $x > 0$.
\begin{enumerate}
  \item Montrer que pour $0 < u < v$,
        $f(v) - f(u) = (v - u)\,\frac{uv - 1}{uv}$.
  \item En déduire que $f$ est strictement \omterm{def:g10:functions:variations}{décroissante} sur
        $\intoc{0}{1}$ et strictement \omterm{def:g11:func:monotone}{croissante} sur
        $\intco{1}{+\infty}$.
  \item Conclure que $x + \frac1x \geq 2$ pour tout $x > 0$, avec
        égalité seulement en $x = 1$.
\end{enumerate}
\end{exercise}
