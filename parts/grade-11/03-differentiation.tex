\chapter{Differentiation: A First Course}\label{ch:g11:deriv}

How fast does a \omterm{def:g11:func:function}{function} change? The answer at a single point is the
\emph{\omterm{def:g11:deriv:derivative}{derivative}}, the single most useful number in all of applied
mathematics. This chapter constructs it from average rates of change,
computes it for the \omterm{def:g11:func:reference}{reference functions}, and uses its sign to read the
variations of a \omterm{def:g11:func:function}{function}. Everything here is extended and deepened in
\cref{ch:g12:deriv}.

\section{The average rate of change}

\begin{definition}[Average rate of change]\label{def:g11:deriv:rate}
Let $f$ be defined on an \omterm{def:g10:numbers:interval}{interval} $I$, and $a \neq b$ in $I$. The
\emph{average rate of change}\index{average rate of change} of $f$ between
$a$ and $b$ is
\[
\frac{f(b) - f(a)}{b - a}.
\]
It is the \omterm{def:g10:reffunc:affine}{slope} of the \emph{secant}\index{secant} line through the points
$\bigl(a, f(a)\bigr)$ and $\bigl(b, f(b)\bigr)$ of the curve.
\end{definition}

\begin{example}
A car travels $140$ km between 2 p.m.\ and 4 p.m.: its average speed is
$\frac{140}{2} = 70$ km/h. The speedometer, however, displays an
\emph{instantaneous} speed at every moment --- that is exactly the notion
that the \omterm{def:g11:deriv:derivative}{derivative} makes precise.
\end{example}

\section{The derivative at a point}

\begin{definition}[Derivative at a point]\label{def:g11:deriv:derivative}
Let $f$ be defined on an \omterm{def:g10:numbers:interval}{interval} containing $a$. For $h \neq 0$, form the
\omterm{def:g11:deriv:rate}{average rate of change} between $a$ and $a + h$:
\[
r(h) = \frac{f(a+h) - f(a)}{h}.
\]
If $r(h)$ approaches a single fixed number as $h$ gets arbitrarily close to
$0$, we say that $f$ is \emph{differentiable at $a$}\index{derivative}\index{differentiable};
this number is the \emph{derivative of $f$ at $a$}, written $f'(a)$:
\[
f'(a) = \lim_{h \to 0} \frac{f(a+h) - f(a)}{h}.
\]
\end{definition}

\begin{example}\label{ex:g11:deriv:atpoint}
Let $f(x) = x^2$ and $a = 1$. For $h \neq 0$:
\[
\frac{f(1+h) - f(1)}{h}
= \frac{(1+h)^2 - 1}{h}
= \frac{1 + 2h + h^2 - 1}{h}
= \frac{h(2 + h)}{h}
= 2 + h .
\]
As $h$ approaches $0$, this approaches $2$: the \omterm{def:g11:func:function}{function} $x^2$ is
\omterm{def:g11:deriv:derivative}{differentiable} at $1$ and $f'(1) = 2$. Note the strategy: \emph{simplify
the quotient until $h$ no longer appears in a denominator}, then let
$h \to 0$.
\end{example}

\begin{example}[A non-differentiable point]
Let $f(x) = \abs{x}$ and $a = 0$. Then
$\frac{f(h) - f(0)}{h} = \frac{\abs h}{h}$, which equals $1$ for $h > 0$
and $-1$ for $h < 0$: no single number is approached, and $\abs x$ is not
\omterm{def:g11:deriv:derivative}{differentiable} at $0$. Its curve has a \emph{corner} there.
\end{example}

\section{The tangent line}

\begin{definition}[Tangent]\label{def:g11:deriv:tangent}
If $f$ is \omterm{def:g11:deriv:derivative}{differentiable} at $a$, the \emph{tangent}\index{tangent} to the
curve at the point $A = \bigl(a, f(a)\bigr)$ is the line through $A$ with
\omterm{def:g10:reffunc:affine}{slope} $f'(a)$. Its \omterm{def:g10:algebra:equation}{equation} is
\[
y = f(a) + f'(a)(x - a).
\]
\end{definition}

The \omterm{def:g11:deriv:tangent}{tangent} is the limiting position of the \omterm{def:g11:deriv:rate}{secants} through $A$: as
$h \to 0$, the second point $\bigl(a+h, f(a+h)\bigr)$ slides along the
curve towards $A$, and the \omterm{def:g11:deriv:rate}{secant}'s \omterm{def:g10:reffunc:affine}{slope} $\frac{f(a+h)-f(a)}{h}$
approaches $f'(a)$.

For $f(x) = x^2$ at $a = 1$: $f(1) = 1$ and $f'(1) = 2$
(\cref{ex:g11:deriv:atpoint}), so the \omterm{def:g11:deriv:tangent}{tangent} is
\[
y = 1 + 2(x - 1) = 2x - 1 .
\]
This is the line $d_2$ found by pure algebra in \cref{exo:g11:quad:11}.
\end{example}

\section{Derivatives of the reference functions}

\begin{proposition}\label{prop:g11:deriv:refderivatives}
On their \omterm{def:g11:func:function}{domains} of differentiability:
\[
(c)' = 0, \quad
(x)' = 1, \quad
(x^2)' = 2x, \quad
(x^3)' = 3x^2, \quad
\left(\frac1x\right)' = -\frac{1}{x^2}, \quad
(\sqrt x\,)' = \frac{1}{2\sqrt x} \ (x > 0).
\]
More generally, $(x^n)' = n\,x^{n-1}$ for every \omterm{def:g10:numbers:sets}{integer} $n \geq 1$.
\end{proposition}

\begin{proof}
Each proof follows the strategy of \cref{ex:g11:deriv:atpoint}: simplify
the rate, then let $h \to 0$. Fix $a$ in the \omterm{def:g11:func:function}{domain}.

\emph{Constant and identity.} $\frac{c - c}{h} = 0$ and
$\frac{(a+h) - a}{h} = 1$ for all $h$: the limits are $0$ and $1$.

\emph{Square.} $\frac{(a+h)^2 - a^2}{h} = \frac{2ah + h^2}{h} = 2a + h
\to 2a$.

\emph{Cube.} $(a+h)^3 = a^3 + 3a^2h + 3ah^2 + h^3$, so the rate is
$3a^2 + 3ah + h^2 \to 3a^2$.

\emph{Reciprocal.} For $a \neq 0$ and $h$ small enough that
$a + h \neq 0$:
\[
\frac{1}{h}\left(\frac{1}{a+h} - \frac{1}{a}\right)
= \frac{1}{h} \cdot \frac{a - (a+h)}{a(a+h)}
= \frac{-1}{a(a+h)}
\xrightarrow[h \to 0]{} -\frac{1}{a^2}.
\]

\emph{Square root.} For $a > 0$, multiply by the conjugate:
\[
\frac{\sqrt{a+h} - \sqrt a}{h}
= \frac{(a + h) - a}{h\,(\sqrt{a+h} + \sqrt a)}
= \frac{1}{\sqrt{a+h} + \sqrt a}
\xrightarrow[h \to 0]{} \frac{1}{2\sqrt a}.
\]

\emph{General power.} The pattern $2a$, $3a^2$ continues: \omterm{def:g10:algebra:expand}{expanding}
$(a+h)^n$, the terms after $a^n + n\,a^{n-1}h$ all carry $h^2$, so the
rate tends to $n\,a^{n-1}$. (A complete induction is given in
\cref{ch:g12:deriv}.)
\end{proof}

\section{Differentiation rules}

\begin{theorem}[Operations]\label{thm:g11:deriv:rules}
Let $u$ and $v$ be \omterm{def:g11:deriv:derivative}{differentiable} on $I$ and $\lambda \in \R$. Then
$u + v$, $\lambda u$, $uv$ are \omterm{def:g11:deriv:derivative}{differentiable} on $I$, as is
$\frac{u}{v}$ wherever $v \neq 0$, and
\[
(u+v)' = u' + v', \qquad
(\lambda u)' = \lambda u', \qquad
(uv)' = u'v + uv', \qquad
\left(\frac{u}{v}\right)' = \frac{u'v - uv'}{v^2}.
\]
\end{theorem}

\begin{proof}[Proof of the first three rules]
Fix $a \in I$ and abbreviate $\Delta u = u(a+h) - u(a)$,
$\Delta v = v(a+h) - v(a)$.

\emph{Sum.} The rate of $u + v$ is
$\frac{\Delta u + \Delta v}{h} = \frac{\Delta u}{h} + \frac{\Delta v}{h}
\to u'(a) + v'(a)$.

\emph{Multiple.} The rate of $\lambda u$ is
$\lambda \frac{\Delta u}{h} \to \lambda u'(a)$.

\emph{Product.} Add and subtract the mixed term $u(a+h)\,v(a)$:
\begin{align*}
\frac{u(a+h)v(a+h) - u(a)v(a)}{h}
&= \frac{u(a+h)v(a+h) - u(a+h)v(a) + u(a+h)v(a) - u(a)v(a)}{h}\\
&= u(a+h)\,\frac{\Delta v}{h} + v(a)\,\frac{\Delta u}{h}.
\end{align*}
As $h \to 0$: $\frac{\Delta v}{h} \to v'(a)$, $\frac{\Delta u}{h} \to
u'(a)$, and $u(a+h) \to u(a)$ (its rate has a limit, so
$\Delta u = h \cdot \frac{\Delta u}{h} \to 0$). The rate therefore tends
to $u(a)v'(a) + v(a)u'(a)$.

The quotient rule is proved along the same lines; it is \emph{admitted at
this level} and proved in \cref{ch:g12:deriv}.
\end{proof}

\begin{example}\label{ex:g11:deriv:computations}
\emph{Polynomial.} $f(x) = 3x^3 - 5x^2 + 7x - 2$:
\[
f'(x) = 3 \times 3x^2 - 5 \times 2x + 7 = 9x^2 - 10x + 7 .
\]

\emph{Quotient.} $g(x) = \frac{2x+1}{x^2+1}$: with $u = 2x + 1$, $v = x^2
+ 1$,
\[
g'(x) = \frac{2(x^2+1) - (2x+1)\,2x}{(x^2+1)^2}
= \frac{-2x^2 - 2x + 2}{(x^2+1)^2}.
\]

\emph{Product.} $k(x) = x\sqrt x$ for $x > 0$:
$k'(x) = 1 \times \sqrt x + x \times \frac{1}{2\sqrt x}
= \sqrt x + \frac{\sqrt x}{2} = \frac{3\sqrt x}{2}$.
\end{example}

\section{Sign of the derivative and variations}

\begin{theorem}[Derivative and variations]\label{thm:g11:deriv:variations}
Let $f$ be \omterm{def:g11:deriv:derivative}{differentiable} on an \omterm{def:g10:numbers:interval}{interval} $I$.
\begin{enumerate}
  \item If $f' \geq 0$ on $I$, then $f$ is \omterm{def:g11:func:monotone}{increasing} on $I$.
  \item If $f' \leq 0$ on $I$, then $f$ is \omterm{def:g10:functions:variations}{decreasing} on $I$.
  \item If $f' > 0$ on $I$ (except possibly at finitely many points where
        it vanishes), then $f$ is strictly \omterm{def:g11:func:monotone}{increasing} on $I$.
\end{enumerate}
\end{theorem}

\begin{proof}
The intuition is clear --- a curve whose \omterm{def:g11:deriv:tangent}{tangents} all point upward must
climb --- but a rigorous proof needs the \omterm{def:g10:stats:mean}{mean} value theorem. The result is
\emph{admitted at this level}; see \cref{thm:g12:deriv:variations}.
\end{proof}

\begin{proposition}[Extrema]\label{prop:g11:deriv:extrema}
If $f'$ changes sign at $a$ from $+$ to $-$, then $f$ has a local \omterm{def:g10:functions:extrema}{maximum}
at $a$; from $-$ to $+$, a local \omterm{def:g10:functions:extrema}{minimum}. At such a point, $f'(a) = 0$ and
the \omterm{def:g11:deriv:tangent}{tangent} is horizontal.
\end{proposition}

\begin{proof}
If $f' \geq 0$ on an \omterm{def:g10:numbers:interval}{interval} ending at $a$ and $f' \leq 0$ on an \omterm{def:g10:numbers:interval}{interval}
starting at $a$, then $f$ increases up to $f(a)$ and decreases afterwards
(\cref{thm:g11:deriv:variations}): $f(a)$ is largest nearby. Since $f'$
changes sign at $a$ and is defined there, $f'(a) = 0$.
\end{proof}

\begin{method}[Studying a function]\label{met:g11:deriv:study}
\begin{enumerate}
  \item Compute $f'(x)$ and simplify it --- ideally into a factored form.
  \item Determine the sign of $f'(x)$ on each \omterm{def:g10:numbers:interval}{interval} (for a quadratic
        $f'$, use \cref{thm:g11:quad:sign}).
  \item Record the results in a \emph{\omterm{def:g10:functions:table}{variation table}}: one row for the
        sign of $f'$, one row of arrows for $f$, with the values of $f$ at
        the turning points.
  \item Conclude: extrema, number of solutions of $f(x) = k$, inequalities.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g11:deriv:study}
Study $f(x) = x^3 - 3x + 1$ on $\R$. First,
$f'(x) = 3x^2 - 3 = 3(x-1)(x+1)$: a quadratic with \omterm{def:g11:quad:discriminant}{roots} $-1$ and $1$,
positive outside them. So $f$ is \omterm{def:g11:func:monotone}{increasing} on $\intoc{-\infty}{-1}$,
\omterm{def:g10:functions:variations}{decreasing} on $\intcc{-1}{1}$, \omterm{def:g11:func:monotone}{increasing} on $\intco{1}{+\infty}$, with a
local \omterm{def:g10:functions:extrema}{maximum} $f(-1) = -1 + 3 + 1 = 3$ and a local \omterm{def:g10:functions:extrema}{minimum}
$f(1) = 1 - 3 + 1 = -1$.
\end{example}

\begin{example}[Optimization]\label{ex:g11:deriv:box}
An open box is made from a $12 \times 12$ cm sheet by cutting a square of
side $x$ from each corner and folding. Its volume is
$V(x) = x(12 - 2x)^2$ for $0 < x < 6$. \omterm{def:g10:algebra:expand}{Expanding},
$V(x) = 4x^3 - 48x^2 + 144x$, so
\[
V'(x) = 12x^2 - 96x + 144 = 12(x^2 - 8x + 12) = 12(x - 2)(x - 6).
\]
On $\intoo{0}{6}$, the factor $x - 6$ is negative, so $V'(x)$ has the sign
of $-(x-2)$: positive before $2$, negative after. The volume is maximal
for $x = 2$ cm, giving $V(2) = 2 \times 8^2 = 128$ cm$^3$.
\end{example}

\section{Problem: The tangent's three jobs}

\begin{problem}[{Weekend problem --- the parabolic mirror proved,
Newton's root-finding machine, and the strongest beam in the log}]
\label{pb:g11:deriv:1}
A \omterm{def:g11:deriv:derivative}{derivative} draws \omterm{def:g11:deriv:tangent}{tangents}; that sounds like a modest talent.
This problem shows the \omterm{def:g11:deriv:tangent}{tangent} holding three jobs at once: it
proves the headlight theorem left pending in
\cref{pb:g11:quad:1} (every ray parallel to a \omterm{def:g11:quad:parabola}{parabola}'s axis
reflects through the \omterm{pb:g11:quad:1}{focus}), it powers the fastest
equation-solving algorithm in common use --- inside every
calculator and neural network --- and it finds the strongest
rectangular beam that a round log can yield.

\medskip
\noindent\textbf{Part I --- \omterm{def:g11:deriv:tangent}{Tangent} fluency.}
\begin{enumerate}
  \item Differentiate $f(x) = 3x^2 - 5x + 1$ and
        $g(x) = x^3 - 12x$
        (\cref{thm:g11:deriv:rules}); and recover $\left(x^2
        \right)'$ at $x = 1$ from the definition (rate of
        change, then $h \to 0$).
  \item Find the \omterm{def:g11:deriv:tangent}{tangent} to $y = x^3 - 12x$ at $a = 2$. What is
        special about it, and what does that signal
        (\cref{prop:g11:deriv:extrema})?
  \item Establish the general \omterm{def:g11:deriv:tangent}{tangent} to the \omterm{def:g11:quad:parabola}{parabola} $y = x^2$
        at the point $(a, a^2)$:
        \[
        y = 2ax - a^2 .
        \]
  \item Where does that \omterm{def:g11:deriv:tangent}{tangent} cross the $y$-axis? Deduce the
        draftsman's recipe: to draw the \omterm{def:g11:deriv:tangent}{tangent} at a point $P$
        of height $h$ on the \omterm{def:g11:quad:parabola}{parabola}, join $P$ to the point of
        the axis at depth\,\dots\ below the origin. State it.
  \item Build the complete \omterm{def:g10:functions:table}{variation table} of
        $g(x) = x^3 - 12x$ (sign of $g'$, local extrema and
        their values, \cref{met:g11:deriv:study}).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The mirror theorem.}
Let $P(a, a^2)$ be a point of the \omterm{def:g11:quad:parabola}{parabola} $y = x^2$ (take
$a \neq 0$), $F\left(0, \frac14\right)$ the \omterm{pb:g11:quad:1}{focus}, and $T$ the
point where the \omterm{def:g11:deriv:tangent}{tangent} at $P$ crosses the $y$-axis (question
4). Recall from \cref{pb:g11:quad:1} that
$PF = a^2 + \frac14$.
\begin{enumerate}[resume]
  \item Compute the distance $FT$.
  \item Conclude that the triangle $FPT$ is isosceles at $F$.
        Which two angles are therefore equal?
  \item The incoming ray through $P$, parallel to the axis, is
        parallel to the line $(TF)$ (both vertical). Using
        alternate angles, show that the \omterm{def:g11:deriv:tangent}{tangent} at $P$ makes
        \emph{equal angles} with the incoming vertical ray and
        with the segment $[PF]$.
  \item Physics: light reflects off a curve as off its \omterm{def:g11:deriv:tangent}{tangent},
        leaving at the same angle it arrived (the law of
        reflection). Conclude the \emph{mirror theorem}: every
        ray parallel to the axis reflects through the \omterm{pb:g11:quad:1}{focus} ---
        and, reversing time, a bulb at the \omterm{pb:g11:quad:1}{focus} beams out
        parallel light. The dish and the headlight of
        \cref{pb:g11:quad:1} are now theorems.
  \item Check the isosceles triangle numerically at $a = 1$:
        compute $T$, $FP$ and $FT$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Newton's machine.}
To solve $f(x) = 0$, Newton proposed: start from a guess
$x_0$, follow the \omterm{def:g11:deriv:tangent}{tangent} at $x_0$ down to the axis, and let
its $x$-intercept be the next guess.
\begin{enumerate}[resume]
  \item Derive the formula of the method:
        \[
        x_{n + 1} = x_n - \frac{f(x_n)}{f'(x_n)} .
        \]
  \item Run it on $f(x) = x^2 - 2$ from $x_0 = 1$: compute
        $x_1$, $x_2$, $x_3$ as exact fractions. Which
        two-thousand-year-old sequence --- and which machine
        from \cref{pb:g10:functions:1} --- have you just
        rebuilt? Prove the identity behind the coincidence:
        \[
        x - \frac{x^2 - 2}{2x} = \frac12\left(x +
        \frac2x\right).
        \]
  \item Solve $x^3 = 100$ (the savings crossover of
        \cref{pb:g10:reffunc:1}): apply two Newton steps to
        $f(x) = x^3 - 100$ from $x_0 = 5$ (four decimals), and
        compare with the calculator's $\sqrt[3]{100}$.
  \item Sabotage: try to launch the method on
        $f(x) = x^3 - 12x$ at $x_0 = 2$. What goes wrong, and
        why (question 2)? State the practical caveat.
  \item In question 12 the correct decimals went roughly
        $1 \to 3 \to 6$. Compare with bisection (halving an
        \omterm{def:g10:numbers:interval}{interval} each step) and say in one sentence why
        calculators, GPS receivers and neural-network training
        all run on Newton's idea.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The strongest beam.}
A rectangular beam of width $w$ and depth $d$ is sawn from a
round log of diameter $30$ cm, so $w^2 + d^2 = 900$. A beam's
stiffness is proportional to $w d^2$ (depth counts twice:
joists stand on edge!).
\begin{enumerate}[resume]
  \item Express the stiffness as a \omterm{def:g11:func:function}{function} of $w$ alone:
        $S(w) = 900w - w^3$, on which \omterm{def:g10:numbers:interval}{interval}?
  \item Differentiate, build the \omterm{def:g10:functions:table}{variation table}, and find the
        optimal $w$ and $d$ (exact values, then to the
        millimeter).
  \item Show that the optimal proportions satisfy
        $d = w\sqrt2$ --- the old carpenters' ratio. (Bonus
        lore: dividing the log's diameter in three equal parts
        and raising perpendiculars to the circle constructs
        exactly this rectangle.)
  \item The naive choice is the \emph{square} beam
        ($w = d$). Compute its stiffness and the optimal
        beam's, and give the carpenter's gain in percent.
  \item Finale --- the \omterm{def:g11:deriv:tangent}{tangent}'s three jobs, one sentence each:
        geometry (the mirror theorem), numerics (Newton's
        machine), optimization (the beam), all powered by the
        same object. And the forward pointer: grade 12 adds
        convexity --- the \omterm{def:g11:deriv:tangent}{tangent}'s side of the curve --- to
        sharpen all three.
\end{enumerate}
\end{problem}
