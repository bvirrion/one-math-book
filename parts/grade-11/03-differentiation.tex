\chapter{Differentiation: A First Course}\label{ch:g11:deriv}

How fast does a function change? The answer at a single point is the
\emph{derivative}, the single most useful number in all of applied
mathematics. This chapter constructs it from average rates of change,
computes it for the reference functions, and uses its sign to read the
variations of a function. Everything here is extended and deepened in
\cref{ch:g12:deriv}.

\section{The average rate of change}

\begin{definition}[Average rate of change]\label{def:g11:deriv:rate}
Let $f$ be defined on an interval $I$, and $a \neq b$ in $I$. The
\emph{average rate of change}\index{average rate of change} of $f$ between
$a$ and $b$ is
\[
\frac{f(b) - f(a)}{b - a}.
\]
It is the slope of the \emph{secant} line through the points
$\bigl(a, f(a)\bigr)$ and $\bigl(b, f(b)\bigr)$ of the curve.
\end{definition}

\begin{example}
A car travels $140$ km between 2 p.m.\ and 4 p.m.: its average speed is
$\frac{140}{2} = 70$ km/h. The speedometer, however, displays an
\emph{instantaneous} speed at every moment --- that is exactly the notion
that the derivative makes precise.
\end{example}

\section{The derivative at a point}

\begin{definition}[Derivative at a point]\label{def:g11:deriv:derivative}
Let $f$ be defined on an interval containing $a$. For $h \neq 0$, form the
average rate of change between $a$ and $a + h$:
\[
r(h) = \frac{f(a+h) - f(a)}{h}.
\]
If $r(h)$ approaches a single fixed number as $h$ gets arbitrarily close to
$0$, we say that $f$ is \emph{differentiable at $a$}\index{derivative};
this number is the \emph{derivative of $f$ at $a$}, written $f'(a)$:
\[
f'(a) = \lim_{h \to 0} \frac{f(a+h) - f(a)}{h}.
\]
\end{definition}

\begin{example}\label{ex:g11:deriv:atpoint}
Let $f(x) = x^2$ and $a = 1$. For $h \neq 0$:
\[
\frac{f(1+h) - f(1)}{h}
= \frac{(1+h)^2 - 1}{h}
= \frac{1 + 2h + h^2 - 1}{h}
= \frac{h(2 + h)}{h}
= 2 + h .
\]
As $h$ approaches $0$, this approaches $2$: the function $x^2$ is
differentiable at $1$ and $f'(1) = 2$. Note the strategy: \emph{simplify
the quotient until $h$ no longer appears in a denominator}, then let
$h \to 0$.
\end{example}

\begin{example}[A non-differentiable point]
Let $f(x) = \abs{x}$ and $a = 0$. Then
$\frac{f(h) - f(0)}{h} = \frac{\abs h}{h}$, which equals $1$ for $h > 0$
and $-1$ for $h < 0$: no single number is approached, and $\abs x$ is not
differentiable at $0$. Its curve has a \emph{corner} there.
\end{example}

\section{The tangent line}

\begin{definition}[Tangent]\label{def:g11:deriv:tangent}
If $f$ is differentiable at $a$, the \emph{tangent}\index{tangent} to the
curve at the point $A = \bigl(a, f(a)\bigr)$ is the line through $A$ with
slope $f'(a)$. Its equation is
\[
y = f(a) + f'(a)(x - a).
\]
\end{definition}

The tangent is the limiting position of the secants through $A$: as
$h \to 0$, the second point $\bigl(a+h, f(a+h)\bigr)$ slides along the
curve towards $A$, and the secant's slope $\frac{f(a+h)-f(a)}{h}$
approaches $f'(a)$.

\begin{example}\label{ex:g11:deriv:tangenteq}
For $f(x) = x^2$ at $a = 1$: $f(1) = 1$ and $f'(1) = 2$
(\cref{ex:g11:deriv:atpoint}), so the tangent is
\[
y = 1 + 2(x - 1) = 2x - 1 .
\]
This is the line $d_2$ found by pure algebra in \cref{exo:g11:quad:11}.
\end{example}

\section{Derivatives of the reference functions}

\begin{proposition}\label{prop:g11:deriv:refderivatives}
On their domains of differentiability:
\[
(c)' = 0, \quad
(x)' = 1, \quad
(x^2)' = 2x, \quad
(x^3)' = 3x^2, \quad
\left(\frac1x\right)' = -\frac{1}{x^2}, \quad
(\sqrt x\,)' = \frac{1}{2\sqrt x} \ (x > 0).
\]
More generally, $(x^n)' = n\,x^{n-1}$ for every integer $n \geq 1$.
\end{proposition}

\begin{proof}
Each proof follows the strategy of \cref{ex:g11:deriv:atpoint}: simplify
the rate, then let $h \to 0$. Fix $a$ in the domain.

\emph{Constant and identity.} $\frac{c - c}{h} = 0$ and
$\frac{(a+h) - a}{h} = 1$ for all $h$: the limits are $0$ and $1$.

\emph{Square.} $\frac{(a+h)^2 - a^2}{h} = \frac{2ah + h^2}{h} = 2a + h
\to 2a$.

\emph{Cube.} $(a+h)^3 = a^3 + 3a^2h + 3ah^2 + h^3$, so the rate is
$3a^2 + 3ah + h^2 \to 3a^2$.

\emph{Reciprocal.} For $a \neq 0$ and $h$ small enough that
$a + h \neq 0$:
\[
\frac{1}{h}\left(\frac{1}{a+h} - \frac{1}{a}\right)
= \frac{1}{h} \cdot \frac{a - (a+h)}{a(a+h)}
= \frac{-1}{a(a+h)}
\xrightarrow[h \to 0]{} -\frac{1}{a^2}.
\]

\emph{Square root.} For $a > 0$, multiply by the conjugate:
\[
\frac{\sqrt{a+h} - \sqrt a}{h}
= \frac{(a + h) - a}{h\,(\sqrt{a+h} + \sqrt a)}
= \frac{1}{\sqrt{a+h} + \sqrt a}
\xrightarrow[h \to 0]{} \frac{1}{2\sqrt a}.
\]

\emph{General power.} The pattern $2a$, $3a^2$ continues: expanding
$(a+h)^n$, the terms after $a^n + n\,a^{n-1}h$ all carry $h^2$, so the
rate tends to $n\,a^{n-1}$. (A complete induction is given in
\cref{ch:g12:deriv}.)
\end{proof}

\section{Differentiation rules}

\begin{theorem}[Operations]\label{thm:g11:deriv:rules}
Let $u$ and $v$ be differentiable on $I$ and $\lambda \in \R$. Then
$u + v$, $\lambda u$, $uv$ are differentiable on $I$, as is
$\frac{u}{v}$ wherever $v \neq 0$, and
\[
(u+v)' = u' + v', \qquad
(\lambda u)' = \lambda u', \qquad
(uv)' = u'v + uv', \qquad
\left(\frac{u}{v}\right)' = \frac{u'v - uv'}{v^2}.
\]
\end{theorem}

\begin{proof}[Proof of the first three rules]
Fix $a \in I$ and abbreviate $\Delta u = u(a+h) - u(a)$,
$\Delta v = v(a+h) - v(a)$.

\emph{Sum.} The rate of $u + v$ is
$\frac{\Delta u + \Delta v}{h} = \frac{\Delta u}{h} + \frac{\Delta v}{h}
\to u'(a) + v'(a)$.

\emph{Multiple.} The rate of $\lambda u$ is
$\lambda \frac{\Delta u}{h} \to \lambda u'(a)$.

\emph{Product.} Add and subtract the mixed term $u(a+h)\,v(a)$:
\begin{align*}
\frac{u(a+h)v(a+h) - u(a)v(a)}{h}
&= \frac{u(a+h)v(a+h) - u(a+h)v(a) + u(a+h)v(a) - u(a)v(a)}{h}\\
&= u(a+h)\,\frac{\Delta v}{h} + v(a)\,\frac{\Delta u}{h}.
\end{align*}
As $h \to 0$: $\frac{\Delta v}{h} \to v'(a)$, $\frac{\Delta u}{h} \to
u'(a)$, and $u(a+h) \to u(a)$ (its rate has a limit, so
$\Delta u = h \cdot \frac{\Delta u}{h} \to 0$). The rate therefore tends
to $u(a)v'(a) + v(a)u'(a)$.

The quotient rule is proved along the same lines; it is \emph{admitted at
this level} and proved in \cref{ch:g12:deriv}.
\end{proof}

\begin{example}\label{ex:g11:deriv:computations}
\emph{Polynomial.} $f(x) = 3x^3 - 5x^2 + 7x - 2$:
\[
f'(x) = 3 \times 3x^2 - 5 \times 2x + 7 = 9x^2 - 10x + 7 .
\]

\emph{Quotient.} $g(x) = \frac{2x+1}{x^2+1}$: with $u = 2x + 1$, $v = x^2
+ 1$,
\[
g'(x) = \frac{2(x^2+1) - (2x+1)\,2x}{(x^2+1)^2}
= \frac{-2x^2 - 2x + 2}{(x^2+1)^2}.
\]

\emph{Product.} $k(x) = x\sqrt x$ for $x > 0$:
$k'(x) = 1 \times \sqrt x + x \times \frac{1}{2\sqrt x}
= \sqrt x + \frac{\sqrt x}{2} = \frac{3\sqrt x}{2}$.
\end{example}

\section{Sign of the derivative and variations}

\begin{theorem}[Derivative and variations]\label{thm:g11:deriv:variations}
Let $f$ be differentiable on an interval $I$.
\begin{enumerate}
  \item If $f' \geq 0$ on $I$, then $f$ is increasing on $I$.
  \item If $f' \leq 0$ on $I$, then $f$ is decreasing on $I$.
  \item If $f' > 0$ on $I$ (except possibly at finitely many points where
        it vanishes), then $f$ is strictly increasing on $I$.
\end{enumerate}
\end{theorem}

\begin{proof}
The intuition is clear --- a curve whose tangents all point upward must
climb --- but a rigorous proof needs the mean value theorem. The result is
\emph{admitted at this level}; see \cref{thm:g12:deriv:variations}.
\end{proof}

\begin{proposition}[Extrema]\label{prop:g11:deriv:extrema}
If $f'$ changes sign at $a$ from $+$ to $-$, then $f$ has a local maximum
at $a$; from $-$ to $+$, a local minimum. At such a point, $f'(a) = 0$ and
the tangent is horizontal.
\end{proposition}

\begin{proof}
If $f' \geq 0$ on an interval ending at $a$ and $f' \leq 0$ on an interval
starting at $a$, then $f$ increases up to $f(a)$ and decreases afterwards
(\cref{thm:g11:deriv:variations}): $f(a)$ is largest nearby. Since $f'$
changes sign at $a$ and is defined there, $f'(a) = 0$.
\end{proof}

\begin{method}[Studying a function]\label{met:g11:deriv:study}
\begin{enumerate}
  \item Compute $f'(x)$ and simplify it --- ideally into a factored form.
  \item Determine the sign of $f'(x)$ on each interval (for a quadratic
        $f'$, use \cref{thm:g11:quad:sign}).
  \item Record the results in a \emph{variation table}: one row for the
        sign of $f'$, one row of arrows for $f$, with the values of $f$ at
        the turning points.
  \item Conclude: extrema, number of solutions of $f(x) = k$, inequalities.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g11:deriv:study}
Study $f(x) = x^3 - 3x + 1$ on $\R$. First,
$f'(x) = 3x^2 - 3 = 3(x-1)(x+1)$: a quadratic with roots $-1$ and $1$,
positive outside them. So $f$ is increasing on $\intoc{-\infty}{-1}$,
decreasing on $\intcc{-1}{1}$, increasing on $\intco{1}{+\infty}$, with a
local maximum $f(-1) = -1 + 3 + 1 = 3$ and a local minimum
$f(1) = 1 - 3 + 1 = -1$.
\end{example}

\begin{example}[Optimization]\label{ex:g11:deriv:box}
An open box is made from a $12 \times 12$ cm sheet by cutting a square of
side $x$ from each corner and folding. Its volume is
$V(x) = x(12 - 2x)^2$ for $0 < x < 6$. Expanding,
$V(x) = 4x^3 - 48x^2 + 144x$, so
\[
V'(x) = 12x^2 - 96x + 144 = 12(x^2 - 8x + 12) = 12(x - 2)(x - 6).
\]
On $\intoo{0}{6}$, the factor $x - 6$ is negative, so $V'(x)$ has the sign
of $-(x-2)$: positive before $2$, negative after. The volume is maximal
for $x = 2$ cm, giving $V(2) = 2 \times 8^2 = 128$ cm$^3$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g11:deriv:1}
Using the definition (rate of change, then $h \to 0$), compute $f'(2)$ for
$f(x) = x^2$, then $g'(1)$ for $g(x) = x^2 + 3x$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:deriv:2}
Differentiate:
\[
f(x) = 4x^3 - 2x^2 + x - 7, \qquad
g(x) = 5\sqrt{x} + \frac{3}{x}, \qquad
h(x) = (2x+1)(x^2 - 3) .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:deriv:3}
Give the equation of the tangent to the curve of $f$ at the point of
abscissa $a$:
\[
f(x) = x^2 - 3x, \ a = 2; \qquad
f(x) = \frac1x, \ a = 1; \qquad
f(x) = \sqrt x, \ a = 4 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:deriv:4}
Differentiate the quotients:
\[
f(x) = \frac{x+2}{x-1}, \qquad
g(x) = \frac{x^2}{x^2+1}, \qquad
h(x) = \frac{1}{x^2 + x + 1} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:deriv:5}
Study the variations of $f(x) = x^3 - 6x^2 + 9x - 2$ on $\R$ (variation
table, local extrema).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:deriv:6}
Study the variations of $f(x) = \frac{x^2 + 3}{x + 1}$ on
$\intoo{-1}{+\infty}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:deriv:7}
Let $f(x) = x^2 - 4x + 5$.
\begin{enumerate}
  \item Find the point of the curve where the tangent is horizontal.
  \item Find the point(s) where the tangent is parallel to the line
        $y = 2x + 1$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:deriv:8}
A rectangular enclosure is built against a wall with $60$ m of fence
(three sides). Express the area as a function of the width $x$, and use
the derivative to find the dimensions of maximal area. (Compare with the
vertex method of \cref{ch:g11:quad}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:deriv:9}
Show that for all $x > 0$, $\sqrt{x} \leq \frac{x + 1}{2}$, by studying
the function $f(x) = \frac{x+1}{2} - \sqrt x$ on $\intoo{0}{+\infty}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:deriv:10}
A cylindrical can must hold $500$ cm$^3$. Its surface area (two disks plus
the side) is $S = 2\pi r^2 + 2\pi r h$ with $\pi r^2 h = 500$. Express $S$
as a function of $r$ alone, and show that $S'(r) = 4\pi r -
\frac{1000}{r^2}$. Deduce the radius that minimizes the surface, as an
exact expression.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:deriv:11}
How many tangents to the parabola $y = x^2$ pass through the external
point $P(1, -3)$? Determine their equations. (Let $a$ be the abscissa of
the contact point and express that the tangent at $a$ passes through $P$.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:deriv:12}
Let $f(x) = x^3 - 3x + 1$ (see \cref{ex:g11:deriv:study}). Using the
variation table, determine the number of solutions of the equation
$f(x) = k$ according to the value of the real number $k$.
\end{exercise}
