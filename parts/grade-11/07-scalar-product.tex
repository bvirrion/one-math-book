\chapter{The Scalar Product in the Plane}\label{ch:g11:scal}

The \omterm{def:g11:scal:dot}{scalar product} attaches a \emph{number} to a pair of \omterm{def:g11:vect:vector}{vectors}, and that
number sees geometry: it vanishes exactly for perpendicular directions,
and it measures angles and lengths. It is the engine behind the law of
\omterm{def:g11:trigo:cossin}{cosines}, \omterm{def:g10:algebra:equation}{equations} of circles, and perpendicularity computations. The same
tool, in space, is the subject of \cref{ch:g12:space}.

\section{Definition and first computations}

\begin{definition}[Scalar product]\label{def:g11:scal:dot}
The \emph{scalar product}\index{scalar product} of two \omterm{def:g11:vect:vector}{vectors} $\vec u$
and $\vec v$ is the number
\[
\vec u \cdot \vec v
= \tfrac12\left(\norm{\vec u + \vec v}^2 - \norm{\vec u}^2
- \norm{\vec v}^2\right),
\]
where $\norm{\vec u}$ denotes the length (\emph{norm}\index{norm}) of $\vec u$.
\end{definition}

\begin{theorem}[The four faces of the scalar product]\label{thm:g11:scal:faces}
Let $\vec u\,(x, y)$ and $\vec v\,(x', y')$ be nonzero \omterm{def:g11:vect:vector}{vectors}, and
$\theta$ the angle between them. Then
\begin{enumerate}
  \item \emph{(\omterm{def:g10:coordgeom:system}{coordinates})} $\vec u \cdot \vec v = xx' + yy'$;
  \item \emph{(\omterm{def:g11:trigo:cossin}{cosine})} $\vec u \cdot \vec v =
        \norm{\vec u}\,\norm{\vec v}\cos\theta$;
  \item \emph{(projection)} $\vec u \cdot \vec v = \vec u \cdot \vec v'$,
        where $\vec v'$ is the \omterm{def:g11:scal:orthogonal}{orthogonal} projection of $\vec v$ on the
        direction of $\vec u$; if $\vec u = \vect{OA}$ and
        $\vec v = \vect{OB}$, then
        $\vec u \cdot \vec v = \overline{OA} \times \overline{OH}$, a
        product of signed lengths, where $H$ is the foot of the
        perpendicular from $B$ to the line $(OA)$;
  \item \emph{(\omterm{def:g11:scal:dot}{norms})} the defining formula of \cref{def:g11:scal:dot}.
\end{enumerate}
\end{theorem}

\begin{proof}[Proof of 1 and 2]
\emph{1.} In \omterm{def:g10:coordgeom:system}{coordinates}, $\norm{\vec u}^2 = x^2 + y^2$ (Pythagoras) and
$\vec u + \vec v$ has \omterm{def:g10:coordgeom:system}{coordinates} $(x + x', y + y')$, so
\begin{align*}
2\,\vec u \cdot \vec v
&= (x + x')^2 + (y + y')^2 - (x^2 + y^2) - (x'^2 + y'^2)\\
&= 2xx' + 2yy',
\end{align*}
after \omterm{def:g10:algebra:expand}{expanding} the squares and cancelling.

\emph{2.} Choose the \omterm{def:g10:coordgeom:system}{coordinate system} so that $\vec u$ points along the
positive $x$-axis: $\vec u\,(\norm{\vec u}, 0)$ and
$\vec v\,(\norm{\vec v}\cos\theta,\ \norm{\vec v}\sin\theta)$
(\cref{ch:g11:trigo}). Formula 1 then gives
$\vec u \cdot \vec v = \norm{\vec u}\,\norm{\vec v}\cos\theta + 0$.
Formula 3 reads off the same computation: the projection of $\vec v$ on
the $x$-axis has signed length $\norm{\vec v}\cos\theta =
\overline{OH}$.
\end{proof}

\begin{example}\label{ex:g11:scal:compute}
$\vec u\,(3, 1)$ and $\vec v\,(2, -4)$:
$\vec u \cdot \vec v = 6 - 4 = 2$. The \omterm{def:g11:scal:dot}{norms} are
$\norm{\vec u} = \sqrt{10}$ and $\norm{\vec v} = \sqrt{20}$, so
$\cos\theta = \frac{2}{\sqrt{10}\sqrt{20}} = \frac{2}{10\sqrt2}
= \frac{\sqrt2}{10}$: the angle is a bit less than $82^\circ$.
\end{example}

\begin{proposition}[Algebraic rules]\label{prop:g11:scal:rules}
For all \omterm{def:g11:vect:vector}{vectors} and all $\lambda \in \R$:
\[
\vec u \cdot \vec v = \vec v \cdot \vec u, \qquad
\vec u \cdot (\vec v + \vec w) = \vec u \cdot \vec v + \vec u \cdot
\vec w, \qquad
(\lambda \vec u)\cdot \vec v = \lambda\,(\vec u \cdot \vec v),
\]
\[
\vec u \cdot \vec u = \norm{\vec u}^2,
\qquad
\norm{\vec u + \vec v}^2 = \norm{\vec u}^2 + 2\,\vec u\cdot\vec v +
\norm{\vec v}^2 .
\]
\end{proposition}

\begin{proof}
All follow from the coordinate formula: for instance
$\vec u \cdot (\vec v + \vec w) = x(x' + x'') + y(y' + y'') =
(xx' + yy') + (xx'' + yy'')$. The last identity expands
$\norm{\vec u + \vec v}^2 = (\vec u + \vec v)\cdot(\vec u + \vec v)$
using the first three rules --- the \omterm{def:g11:scal:dot}{scalar product} behaves exactly like an
ordinary product, so the usual algebraic identities apply to \omterm{def:g11:vect:vector}{vectors}.
\end{proof}

\section{Orthogonality}

\begin{definition}[Orthogonal vectors]\label{def:g11:scal:orthogonal}
Two \omterm{def:g11:vect:vector}{vectors} are \emph{orthogonal}\index{orthogonal vectors}, written
$\vec u \perp \vec v$, if $\vec u \cdot \vec v = 0$.
\end{definition}

\begin{remark}
For nonzero \omterm{def:g11:vect:vector}{vectors}, formula 2 of \cref{thm:g11:scal:faces} shows
$\vec u \cdot \vec v = 0 \iff \cos\theta = 0 \iff$ the directions are
perpendicular. The zero \omterm{def:g11:vect:vector}{vector} is \omterm{def:g11:scal:orthogonal}{orthogonal} to everything.
\end{remark}

\begin{example}
$\vec u\,(3, 2)$ and $\vec v\,(-4, 6)$: $\vec u \cdot \vec v = -12 + 12 =
0$: \omterm{def:g11:scal:orthogonal}{orthogonal}. In general, $\vec u\,(a, b)$ is always \omterm{def:g11:scal:orthogonal}{orthogonal} to
$\vec v\,(-b, a)$ --- a quarter-turn rotation.
\end{example}

\section{The law of cosines}

\begin{theorem}[Law of cosines]\label{thm:g11:scal:cosines}
\index{law of cosines}
In a triangle $ABC$ with sides $a = BC$, $b = CA$, $c = AB$ and angle
$\widehat A$ at the vertex $A$:
\[
a^2 = b^2 + c^2 - 2bc\,\cos\widehat A .
\]
\end{theorem}

\begin{proof}
Write $\vect{BC} = \vect{BA} + \vect{AC} = \vect{AC} - \vect{AB}$ and
expand with \cref{prop:g11:scal:rules}:
\[
a^2 = \norm{\vect{BC}}^2
= \norm{\vect{AC}}^2 - 2\,\vect{AC}\cdot\vect{AB} + \norm{\vect{AB}}^2
= b^2 + c^2 - 2\,\vect{AB}\cdot\vect{AC}.
\]
By the \omterm{def:g11:trigo:cossin}{cosine} formula,
$\vect{AB}\cdot\vect{AC} = c\,b\,\cos\widehat A$.
\end{proof}

\begin{remark}
When $\widehat A$ is a right angle, $\cos\widehat A = 0$ and the law of
\omterm{def:g11:trigo:cossin}{cosines} reduces to the Pythagorean theorem $a^2 = b^2 + c^2$: it is
Pythagoras with a correction term for non-right angles.
\end{remark}

\begin{example}\label{ex:g11:scal:cosines}
Two sides of a triangle measure $b = 5$ and $c = 8$, enclosing an angle
$\widehat A = \frac\pi3$. Then
\[
a^2 = 25 + 64 - 2 \times 5 \times 8 \times \frac12 = 89 - 40 = 49,
\]
so $a = 7$.
\end{example}

\section{Applications: circles and perpendicular lines}

\begin{proposition}[Circle with a given diameter]\label{prop:g11:scal:diameter}
A point $M$ lies on the circle of diameter $[AB]$ if and only if
\[
\vect{MA} \cdot \vect{MB} = 0 .
\]
\end{proposition}

\begin{proof}
If $M \neq A, B$, the condition says the angle at $M$ in the triangle
$AMB$ is right, and the points seeing a segment under a right angle are
exactly those of the circle with that diameter. Here is a direct proof:
let $O$ be the \omterm{prop:g10:coordgeom:midpoint}{midpoint} of $[AB]$ and $r = \frac{AB}{2}$. Then
$\vect{MA} = \vect{MO} + \vect{OA}$ and
$\vect{MB} = \vect{MO} + \vect{OB} = \vect{MO} - \vect{OA}$, so
\[
\vect{MA}\cdot\vect{MB}
= \norm{\vect{MO}}^2 - \norm{\vect{OA}}^2 = MO^2 - r^2,
\]
which vanishes exactly when $MO = r$, i.e.\ when $M$ is on the circle of
center $O$ and radius $r$.
\end{proof}

\begin{proposition}[Normal vector of a line]\label{prop:g11:scal:normal}
The \omterm{def:g11:vect:vector}{vector} $\vec n\,(a, b)$ is \omterm{def:g11:scal:orthogonal}{orthogonal} to every \omterm{def:g11:vect:direction}{direction vector} of the
line $d : ax + by + c = 0$; one says $\vec n$ is a \emph{normal
vector}\index{normal vector} of $d$. Two lines are perpendicular exactly
when their normal (or direction) \omterm{def:g11:vect:vector}{vectors} are \omterm{def:g11:scal:orthogonal}{orthogonal}.
\end{proposition}

\begin{proof}
A \omterm{def:g11:vect:direction}{direction vector} of $d$ is $\vec u\,(-b, a)$
(\cref{thm:g11:vect:cartesian}), and
$\vec n \cdot \vec u = a(-b) + b(a) = 0$.
\end{proof}

\begin{method}[Using normal vectors]\label{met:g11:scal:normal}
\begin{itemize}
  \item Line through $A$ with normal $\vec n\,(a,b)$: write
        $\vect{AM} \cdot \vec n = 0$, which expands to $ax + by + c = 0$
        with $c$ determined by $A$.
  \item Perpendicularity of $d: ax+by+c = 0$ and $d': a'x+b'y+c' = 0$:
        check $aa' + bb' = 0$.
\end{itemize}
\end{method}

\begin{example}\label{ex:g11:scal:perpline}
The line through $P(2, -1)$ perpendicular to $d : 3x - y + 4 = 0$: a
\omterm{def:g11:vect:direction}{direction vector} of $d$ is $(1, 3)$, which serves as \emph{normal} \omterm{def:g11:vect:vector}{vector}
of the perpendicular. \omterm{def:g10:algebra:equation}{Equation}: $x + 3y + c = 0$ with
$2 - 3 + c = 0$, so $c = 1$: the line is $x + 3y + 1 = 0$.
\end{example}

\begin{example}[Equation of a circle]\label{ex:g11:scal:circle}
The circle of center $\Omega(1, 2)$ and radius $3$ is the set of points
$M(x,y)$ with $\Omega M^2 = 9$:
\[
(x - 1)^2 + (y - 2)^2 = 9 .
\]
Conversely, $x^2 + y^2 - 2x - 4y - 4 = 0$ rewrites, completing the
squares (\cref{ch:g11:quad}), as $(x-1)^2 + (y-2)^2 = 9$: the same
circle.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g11:scal:1}
Compute $\vec u \cdot \vec v$ for
\[
\vec u\,(2, 5),\ \vec v\,(3, -1); \qquad
\vec u\,(1, -4),\ \vec v\,(8, 2); \qquad
\norm{\vec u} = 3,\ \norm{\vec v} = 4,\ \theta = \frac{\pi}{3} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:scal:2}
$ABC$ is an equilateral triangle of side $6$. Compute
$\vect{AB} \cdot \vect{AC}$ and $\vect{AB} \cdot \vect{BC}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:scal:3}
For which value of $t$ are $\vec u\,(3, t)$ and $\vec v\,(t - 8, 1)$
\omterm{def:g11:scal:orthogonal}{orthogonal}? And for which value(s) of $t$ is $\vec u\,(t, 2)$ \omterm{def:g11:scal:orthogonal}{orthogonal}
to itself minus $\vec v\,(4, 2)$, i.e.\ $\vec u \perp (\vec u - \vec v)$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:scal:4}
Compute the angle $\theta$ between $\vec u\,(1, 2)$ and $\vec v\,(3, 1)$
(give $\cos\theta$ exactly, then $\theta$ to the nearest degree).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:5}
A triangle has sides $b = 4$, $c = 6$ and angle
$\widehat A = \frac{2\pi}{3}$ between them. Compute the third side $a$.
Then, in a triangle with sides $a = 7$, $b = 5$, $c = 4$, compute
$\cos\widehat A$ and decide whether $\widehat A$ is acute or obtuse.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:6}
Let $A(1, 1)$, $B(5, -1)$ and $C(3, 5)$. Compute
$\vect{AB} \cdot \vect{AC}$; is the angle $\widehat A$ right? Compute the
three side lengths and check your answer with the law of \omterm{def:g11:trigo:cossin}{cosines}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:7}
Give an \omterm{def:g10:algebra:equation}{equation} of the circle of center $\Omega(-2, 3)$ and radius $5$.
Then determine the center and radius of the circle
\[
x^2 + y^2 - 6x + 2y - 6 = 0 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:8}
Let $A(-3, 0)$ and $B(0, 2)$. Using \cref{prop:g11:scal:diameter}, find
an \omterm{def:g10:algebra:equation}{equation} of the circle of diameter $[AB]$, and check that the origin
$O(0,0)$ lies on it.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:9}
Find an \omterm{def:g10:algebra:equation}{equation} of the line through $P(1, 3)$ perpendicular to
$d : 2x + y - 7 = 0$, and the \omterm{def:g10:coordgeom:system}{coordinates} of the foot $H$ of the
perpendicular (the \omterm{def:g10:numbers:interunion}{intersection} of the two lines). Deduce the distance
from $P$ to $d$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:10}
Using $\norm{\vec u + \vec v}^2 = \norm{\vec u}^2 + 2\vec u\cdot\vec v +
\norm{\vec v}^2$ and its analogue for $\vec u - \vec v$, prove the
\emph{parallelogram identity}:
\[
\norm{\vec u + \vec v}^2 + \norm{\vec u - \vec v}^2
= 2\norm{\vec u}^2 + 2\norm{\vec v}^2 ,
\]
and interpret it in a parallelogram (diagonals versus sides).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:scal:11}
Let $A$ and $B$ be two points with $AB = 4$, and $I$ the \omterm{prop:g10:coordgeom:midpoint}{midpoint} of
$[AB]$.
\begin{enumerate}
  \item Show that for every point $M$:
        $\vect{MA}\cdot\vect{MB} = MI^2 - 4$.
        (Insert $I$: $\vect{MA} = \vect{MI} + \vect{IA}$.)
  \item Deduce the set of points $M$ such that
        $\vect{MA}\cdot\vect{MB} = 12$.
\end{enumerate}
\end{exercise}

\section{Problem: Two gifts of the scalar product}

\begin{problem}[{Weekend problem --- the addition formulas fall
out of one dot product, the distance to a line out of another,
and the three altitudes meet as a bonus}]
\label{pb:g11:scal:1}
The \omterm{def:g11:scal:dot}{scalar product} looks like bookkeeping --- multiply, add ---
yet it hands over, almost free of charge, two treasures that
resisted centuries of angle-chasing: the \emph{addition formulas}
of trigonometry (compute $\cos 15^\circ$ exactly!) and the
\emph{distance from a point to a line} in one clean fraction. As
a parting gift it proves that the three altitudes of every
triangle pass through one point. All four faces of
\cref{thm:g11:scal:faces} will report for duty.

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item For $\vec u(3, 4)$ and $\vec v(-1, 2)$: compute
        $\vec u \cdot \vec v$, the two \omterm{def:g11:scal:dot}{norms}, and the angle
        between the \omterm{def:g11:vect:vector}{vectors} (to the tenth of a degree).
  \item Find $k$ so that $(2, k)$ and $(3, -4)$ are \omterm{def:g11:scal:orthogonal}{orthogonal}
        (\cref{def:g11:scal:orthogonal}).
  \item Physics face: a sled is pulled $10$ m by a $50$ N force
        at $60^\circ$ to the ground. Compute the work
        $W = \vec F \cdot \vec d$.
  \item Law of \omterm{def:g11:trigo:cossin}{cosines} (\cref{thm:g11:scal:cosines}): two sides
        of $5$ and $7$ enclose an angle of $60^\circ$; find the
        third side exactly.
  \item Reverse: a triangle has sides $4$, $6$, $8$. Compute the
        \omterm{def:g11:trigo:cossin}{cosine} of the angle opposite the side $8$, classify the
        angle, and give it to the tenth of a degree.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- First gift: the addition
formulas.}
Let $\vec u = (\cos a, \sin a)$ and $\vec v = (\cos b, \sin b)$:
two unit \omterm{def:g11:vect:vector}{vectors} at angles $a$ and $b$.
\begin{enumerate}[resume]
  \item Compute $\vec u \cdot \vec v$ twice --- once with
        \omterm{def:g10:coordgeom:system}{coordinates}, once with \omterm{def:g11:scal:dot}{norms} and the angle between the
        \omterm{def:g11:vect:vector}{vectors} --- and conclude the mother formula:
        \[
        \cos(a - b) = \cos a \cos b + \sin a \sin b .
        \]
  \item Substitute $-b$ for $b$ (parities of \omterm{def:g11:trigo:cossin}{sine} and \omterm{def:g11:trigo:cossin}{cosine}!)
        to obtain $\cos(a + b)$.
  \item Obtain $\sin(a + b)$ from the complementary identity
        $\sin x = \cos\left(\frac\pi2 - x\right)$ applied to
        $x = a + b$.
  \item Exact values at last: compute $\cos 15^\circ$ (as
        $\cos(45^\circ - 30^\circ)$) and $\sin 75^\circ$. Check
        both numerically.
  \item Set $b = a$ to derive the double-angle formulas:
        $\cos 2a$ (in its three forms) and $\sin 2a$.
  \item From $\cos 2a = 1 - 2\sin^2 a$, compute
        $\sin^2 15^\circ$, then $\sin 15^\circ$ as a nested
        radical --- and verify that its square agrees with the
        value $\frac{\sqrt6 - \sqrt2}{4}$ that question 9's
        method gives.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Second gift: the distance to a
line.}
\begin{enumerate}[resume]
  \item Let $d$ be the line $ax + by + c = 0$, with normal
        \omterm{def:g11:vect:vector}{vector} $\vec n(a, b)$ (\cref{prop:g11:scal:normal}),
        and $P(x_0, y_0)$ a point. Taking any point $A$ on $d$,
        explain why the distance from $P$ to $d$ is
        $\dfrac{\abs{\vect{AP} \cdot \vec n}}{\norm{\vec n}}$,
        and deduce the formula
        \[
        \operatorname{dist}(P, d)
        = \frac{\abs{a x_0 + b y_0 + c}}{\sqrt{a^2 + b^2}} .
        \]
        Compute the distance from $P(5, 6)$ to
        $3x + 4y - 12 = 0$.
  \item Write the \omterm{def:g10:algebra:equation}{equation} of the circle centered at $(5, 6)$
        \omterm{def:g11:deriv:tangent}{tangent} to that line.
  \item Cross-check the formula: compute the distance from the
        origin to $3x + 4y - 12 = 0$, then verify it by
        computing the foot of the perpendicular from the origin
        explicitly.
  \item Triangle $A(0,0)$, $B(6,0)$, $C(2,5)$: compute its area
        using the base $[AB]$; then compute the distance from
        $B$ to the line $(AC)$ and check it reproduces the same
        area.
  \item The circle of center $(2, 1)$ and radius $2$, against
        the line $x + y - 6 = 0$: compute the center's distance
        to the line and classify their position (\omterm{def:g11:deriv:rate}{secant},
        \omterm{def:g11:deriv:tangent}{tangent}, disjoint).
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The bonus: altitudes concur.}
\begin{enumerate}[resume]
  \item Generalize \cref{exo:g11:scal:11}: for any two points
        $A$, $B$ with \omterm{prop:g10:coordgeom:midpoint}{midpoint} $I$, prove
        \[
        \vect{MA} \cdot \vect{MB} = MI^2 - \frac{AB^2}{4}
        \quad\text{for every point } M .
        \]
  \item Deduce \cref{prop:g11:scal:diameter} in one line: the
        set of points $M$ with
        $\vect{MA} \cdot \vect{MB} = 0$ is the circle of
        diameter $[AB]$.
  \item The altitudes: let $H$ be the \omterm{def:g10:numbers:interunion}{intersection} of the
        altitudes from $A$ and from $B$ in a triangle $ABC$
        (so $\vect{HA} \cdot \vect{BC} = 0$ and
        $\vect{HB} \cdot \vect{CA} = 0$). Prove the identity
        \[
        \vect{HA} \cdot \vect{BC} + \vect{HB} \cdot \vect{CA}
        + \vect{HC} \cdot \vect{AB} = 0
        \]
        --- expand everything from $H$ with Chasles --- and
        conclude that $H$ also lies on the altitude from $C$:
        the three altitudes are concurrent (at the
        \emph{orthocenter}\index{orthocenter}).
  \item Finale: four faces, three trophies. Which face of
        \cref{thm:g11:scal:faces} carried the addition formulas,
        which the distance formula, which the \omterm{pb:g11:scal:1}{orthocenter} ---
        and why is ``compute the same quantity twice'' the motto
        of the whole chapter?
\end{enumerate}
\end{problem}
