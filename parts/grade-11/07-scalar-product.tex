\chapter{The Scalar Product in the Plane}\label{ch:g11:scal}

The scalar product attaches a \emph{number} to a pair of vectors, and that
number sees geometry: it vanishes exactly for perpendicular directions,
and it measures angles and lengths. It is the engine behind the law of
cosines, equations of circles, and perpendicularity computations. The same
tool, in space, is the subject of \cref{ch:g12:space}.

\section{Definition and first computations}

\begin{definition}[Scalar product]\label{def:g11:scal:dot}
The \emph{scalar product}\index{scalar product} of two vectors $\vec u$
and $\vec v$ is the number
\[
\vec u \cdot \vec v
= \tfrac12\left(\norm{\vec u + \vec v}^2 - \norm{\vec u}^2
- \norm{\vec v}^2\right),
\]
where $\norm{\vec u}$ denotes the length (\emph{norm}) of $\vec u$.
\end{definition}

\begin{theorem}[The four faces of the scalar product]\label{thm:g11:scal:faces}
Let $\vec u\,(x, y)$ and $\vec v\,(x', y')$ be nonzero vectors, and
$\theta$ the angle between them. Then
\begin{enumerate}
  \item \emph{(coordinates)} $\vec u \cdot \vec v = xx' + yy'$;
  \item \emph{(cosine)} $\vec u \cdot \vec v =
        \norm{\vec u}\,\norm{\vec v}\cos\theta$;
  \item \emph{(projection)} $\vec u \cdot \vec v = \vec u \cdot \vec v'$,
        where $\vec v'$ is the orthogonal projection of $\vec v$ on the
        direction of $\vec u$; if $\vec u = \vect{OA}$ and
        $\vec v = \vect{OB}$, then
        $\vec u \cdot \vec v = \overline{OA} \times \overline{OH}$, a
        product of signed lengths, where $H$ is the foot of the
        perpendicular from $B$ to the line $(OA)$;
  \item \emph{(norms)} the defining formula of \cref{def:g11:scal:dot}.
\end{enumerate}
\end{theorem}

\begin{proof}[Proof of 1 and 2]
\emph{1.} In coordinates, $\norm{\vec u}^2 = x^2 + y^2$ (Pythagoras) and
$\vec u + \vec v$ has coordinates $(x + x', y + y')$, so
\begin{align*}
2\,\vec u \cdot \vec v
&= (x + x')^2 + (y + y')^2 - (x^2 + y^2) - (x'^2 + y'^2)\\
&= 2xx' + 2yy',
\end{align*}
after expanding the squares and cancelling.

\emph{2.} Choose the coordinate system so that $\vec u$ points along the
positive $x$-axis: $\vec u\,(\norm{\vec u}, 0)$ and
$\vec v\,(\norm{\vec v}\cos\theta,\ \norm{\vec v}\sin\theta)$
(\cref{ch:g11:trigo}). Formula 1 then gives
$\vec u \cdot \vec v = \norm{\vec u}\,\norm{\vec v}\cos\theta + 0$.
Formula 3 reads off the same computation: the projection of $\vec v$ on
the $x$-axis has signed length $\norm{\vec v}\cos\theta =
\overline{OH}$.
\end{proof}

\begin{example}\label{ex:g11:scal:compute}
$\vec u\,(3, 1)$ and $\vec v\,(2, -4)$:
$\vec u \cdot \vec v = 6 - 4 = 2$. The norms are
$\norm{\vec u} = \sqrt{10}$ and $\norm{\vec v} = \sqrt{20}$, so
$\cos\theta = \frac{2}{\sqrt{10}\sqrt{20}} = \frac{2}{10\sqrt2}
= \frac{\sqrt2}{10}$: the angle is a bit less than $82^\circ$.
\end{example}

\begin{proposition}[Algebraic rules]\label{prop:g11:scal:rules}
For all vectors and all $\lambda \in \R$:
\[
\vec u \cdot \vec v = \vec v \cdot \vec u, \qquad
\vec u \cdot (\vec v + \vec w) = \vec u \cdot \vec v + \vec u \cdot
\vec w, \qquad
(\lambda \vec u)\cdot \vec v = \lambda\,(\vec u \cdot \vec v),
\]
\[
\vec u \cdot \vec u = \norm{\vec u}^2,
\qquad
\norm{\vec u + \vec v}^2 = \norm{\vec u}^2 + 2\,\vec u\cdot\vec v +
\norm{\vec v}^2 .
\]
\end{proposition}

\begin{proof}
All follow from the coordinate formula: for instance
$\vec u \cdot (\vec v + \vec w) = x(x' + x'') + y(y' + y'') =
(xx' + yy') + (xx'' + yy'')$. The last identity expands
$\norm{\vec u + \vec v}^2 = (\vec u + \vec v)\cdot(\vec u + \vec v)$
using the first three rules --- the scalar product behaves exactly like an
ordinary product, so the usual algebraic identities apply to vectors.
\end{proof}

\section{Orthogonality}

\begin{definition}[Orthogonal vectors]\label{def:g11:scal:orthogonal}
Two vectors are \emph{orthogonal}\index{orthogonal vectors}, written
$\vec u \perp \vec v$, if $\vec u \cdot \vec v = 0$.
\end{definition}

\begin{remark}
For nonzero vectors, formula 2 of \cref{thm:g11:scal:faces} shows
$\vec u \cdot \vec v = 0 \iff \cos\theta = 0 \iff$ the directions are
perpendicular. The zero vector is orthogonal to everything.
\end{remark}

\begin{example}
$\vec u\,(3, 2)$ and $\vec v\,(-4, 6)$: $\vec u \cdot \vec v = -12 + 12 =
0$: orthogonal. In general, $\vec u\,(a, b)$ is always orthogonal to
$\vec v\,(-b, a)$ --- a quarter-turn rotation.
\end{example}

\section{The law of cosines}

\begin{theorem}[Law of cosines]\label{thm:g11:scal:cosines}
\index{law of cosines}
In a triangle $ABC$ with sides $a = BC$, $b = CA$, $c = AB$ and angle
$\widehat A$ at the vertex $A$:
\[
a^2 = b^2 + c^2 - 2bc\,\cos\widehat A .
\]
\end{theorem}

\begin{proof}
Write $\vect{BC} = \vect{BA} + \vect{AC} = \vect{AC} - \vect{AB}$ and
expand with \cref{prop:g11:scal:rules}:
\[
a^2 = \norm{\vect{BC}}^2
= \norm{\vect{AC}}^2 - 2\,\vect{AC}\cdot\vect{AB} + \norm{\vect{AB}}^2
= b^2 + c^2 - 2\,\vect{AB}\cdot\vect{AC}.
\]
By the cosine formula,
$\vect{AB}\cdot\vect{AC} = c\,b\,\cos\widehat A$.
\end{proof}

\begin{remark}
When $\widehat A$ is a right angle, $\cos\widehat A = 0$ and the law of
cosines reduces to the Pythagorean theorem $a^2 = b^2 + c^2$: it is
Pythagoras with a correction term for non-right angles.
\end{remark}

\begin{example}\label{ex:g11:scal:cosines}
Two sides of a triangle measure $b = 5$ and $c = 8$, enclosing an angle
$\widehat A = \frac\pi3$. Then
\[
a^2 = 25 + 64 - 2 \times 5 \times 8 \times \frac12 = 89 - 40 = 49,
\]
so $a = 7$.
\end{example}

\section{Applications: circles and perpendicular lines}

\begin{proposition}[Circle with a given diameter]\label{prop:g11:scal:diameter}
A point $M$ lies on the circle of diameter $[AB]$ if and only if
\[
\vect{MA} \cdot \vect{MB} = 0 .
\]
\end{proposition}

\begin{proof}
If $M \neq A, B$, the condition says the angle at $M$ in the triangle
$AMB$ is right, and the points seeing a segment under a right angle are
exactly those of the circle with that diameter. Here is a direct proof:
let $O$ be the midpoint of $[AB]$ and $r = \frac{AB}{2}$. Then
$\vect{MA} = \vect{MO} + \vect{OA}$ and
$\vect{MB} = \vect{MO} + \vect{OB} = \vect{MO} - \vect{OA}$, so
\[
\vect{MA}\cdot\vect{MB}
= \norm{\vect{MO}}^2 - \norm{\vect{OA}}^2 = MO^2 - r^2,
\]
which vanishes exactly when $MO = r$, i.e.\ when $M$ is on the circle of
center $O$ and radius $r$.
\end{proof}

\begin{proposition}[Normal vector of a line]\label{prop:g11:scal:normal}
The vector $\vec n\,(a, b)$ is orthogonal to every direction vector of the
line $d : ax + by + c = 0$; one says $\vec n$ is a \emph{normal
vector}\index{normal vector} of $d$. Two lines are perpendicular exactly
when their normal (or direction) vectors are orthogonal.
\end{proposition}

\begin{proof}
A direction vector of $d$ is $\vec u\,(-b, a)$
(\cref{thm:g11:vect:cartesian}), and
$\vec n \cdot \vec u = a(-b) + b(a) = 0$.
\end{proof}

\begin{method}[Using normal vectors]\label{met:g11:scal:normal}
\begin{itemize}
  \item Line through $A$ with normal $\vec n\,(a,b)$: write
        $\vect{AM} \cdot \vec n = 0$, which expands to $ax + by + c = 0$
        with $c$ determined by $A$.
  \item Perpendicularity of $d: ax+by+c = 0$ and $d': a'x+b'y+c' = 0$:
        check $aa' + bb' = 0$.
\end{itemize}
\end{method}

\begin{example}\label{ex:g11:scal:perpline}
The line through $P(2, -1)$ perpendicular to $d : 3x - y + 4 = 0$: a
direction vector of $d$ is $(1, 3)$, which serves as \emph{normal} vector
of the perpendicular. Equation: $x + 3y + c = 0$ with
$2 - 3 + c = 0$, so $c = 1$: the line is $x + 3y + 1 = 0$.
\end{example}

\begin{example}[Equation of a circle]\label{ex:g11:scal:circle}
The circle of center $\Omega(1, 2)$ and radius $3$ is the set of points
$M(x,y)$ with $\Omega M^2 = 9$:
\[
(x - 1)^2 + (y - 2)^2 = 9 .
\]
Conversely, $x^2 + y^2 - 2x - 4y - 4 = 0$ rewrites, completing the
squares (\cref{ch:g11:quad}), as $(x-1)^2 + (y-2)^2 = 9$: the same
circle.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g11:scal:1}
Compute $\vec u \cdot \vec v$ for
\[
\vec u\,(2, 5),\ \vec v\,(3, -1); \qquad
\vec u\,(1, -4),\ \vec v\,(8, 2); \qquad
\norm{\vec u} = 3,\ \norm{\vec v} = 4,\ \theta = \frac{\pi}{3} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:scal:2}
$ABC$ is an equilateral triangle of side $6$. Compute
$\vect{AB} \cdot \vect{AC}$ and $\vect{AB} \cdot \vect{BC}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:scal:3}
For which value of $t$ are $\vec u\,(3, t)$ and $\vec v\,(t - 8, 1)$
orthogonal? And for which value(s) of $t$ is $\vec u\,(t, 2)$ orthogonal
to itself minus $\vec v\,(4, 2)$, i.e.\ $\vec u \perp (\vec u - \vec v)$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:scal:4}
Compute the angle $\theta$ between $\vec u\,(1, 2)$ and $\vec v\,(3, 1)$
(give $\cos\theta$ exactly, then $\theta$ to the nearest degree).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:5}
A triangle has sides $b = 4$, $c = 6$ and angle
$\widehat A = \frac{2\pi}{3}$ between them. Compute the third side $a$.
Then, in a triangle with sides $a = 7$, $b = 5$, $c = 4$, compute
$\cos\widehat A$ and decide whether $\widehat A$ is acute or obtuse.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:6}
Let $A(1, 1)$, $B(5, -1)$ and $C(3, 5)$. Compute
$\vect{AB} \cdot \vect{AC}$; is the angle $\widehat A$ right? Compute the
three side lengths and check your answer with the law of cosines.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:7}
Give an equation of the circle of center $\Omega(-2, 3)$ and radius $5$.
Then determine the center and radius of the circle
\[
x^2 + y^2 - 6x + 2y - 6 = 0 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:8}
Let $A(-3, 0)$ and $B(0, 2)$. Using \cref{prop:g11:scal:diameter}, find
an equation of the circle of diameter $[AB]$, and check that the origin
$O(0,0)$ lies on it.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:9}
Find an equation of the line through $P(1, 3)$ perpendicular to
$d : 2x + y - 7 = 0$, and the coordinates of the foot $H$ of the
perpendicular (the intersection of the two lines). Deduce the distance
from $P$ to $d$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:10}
Using $\norm{\vec u + \vec v}^2 = \norm{\vec u}^2 + 2\vec u\cdot\vec v +
\norm{\vec v}^2$ and its analogue for $\vec u - \vec v$, prove the
\emph{parallelogram identity}:
\[
\norm{\vec u + \vec v}^2 + \norm{\vec u - \vec v}^2
= 2\norm{\vec u}^2 + 2\norm{\vec v}^2 ,
\]
and interpret it in a parallelogram (diagonals versus sides).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:scal:11}
Let $A$ and $B$ be two points with $AB = 4$, and $I$ the midpoint of
$[AB]$.
\begin{enumerate}
  \item Show that for every point $M$:
        $\vect{MA}\cdot\vect{MB} = MI^2 - 4$.
        (Insert $I$: $\vect{MA} = \vect{MI} + \vect{IA}$.)
  \item Deduce the set of points $M$ such that
        $\vect{MA}\cdot\vect{MB} = 12$.
\end{enumerate}
\end{exercise}
