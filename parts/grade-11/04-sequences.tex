\chapter{Sequences: A First Course}\label{ch:g11:seq}

A \omterm{def:g11:seq:sequence}{sequence} is a list of numbers produced by a rule: the successive balances
of a savings account, the sizes of a population year after year. This
chapter studies the two families that dominate applications ---
\emph{\omterm{def:g11:seq:arithmetic}{arithmetic}} \omterm{def:g11:seq:sequence}{sequences}, which grow by equal steps, and
\emph{\omterm{def:g11:seq:geometric}{geometric}} \omterm{def:g11:seq:sequence}{sequences}, which grow by equal ratios. The rigorous theory
of limits is developed in \cref{ch:g12:seq}.

\section{Defining a sequence}

\begin{definition}[Sequence]\label{def:g11:seq:sequence}
A \emph{sequence}\index{sequence} $(u_n)$ assigns to each \omterm{def:g10:numbers:sets}{integer}
$n \geq 0$ (or $n \geq 1$) a \omterm{def:g10:numbers:sets}{real number} $u_n$, its \emph{term of index
$n$}. A sequence can be given
\begin{itemize}
  \item \emph{explicitly}, by a formula for $u_n$ in terms of $n$: e.g.\
        $u_n = n^2 + 1$;
  \item \emph{recursively}, by its first term and a rule to pass from each
        term to the next: e.g.\ $u_0 = 3$ and $u_{n+1} = 2u_n - 1$.
\end{itemize}
\end{definition}

\begin{example}
For $u_n = n^2 + 1$: $u_0 = 1$, $u_1 = 2$, $u_2 = 5$, and $u_{10} = 101$
directly. For $u_0 = 3$, $u_{n+1} = 2u_n - 1$: $u_1 = 5$, $u_2 = 9$,
$u_3 = 17$ --- each term needs the previous one; reaching $u_{10}$ takes
ten steps (or a general formula, see \cref{exo:g11:seq:11}).
\end{example}

\section{Arithmetic sequences}

\begin{definition}[Arithmetic sequence]\label{def:g11:seq:arithmetic}
A \omterm{def:g11:seq:sequence}{sequence} is \emph{arithmetic}\index{arithmetic sequence} with
\emph{common difference}\index{common difference} $d$ if each term is obtained from the previous
one by adding $d$:
\[
u_{n+1} = u_n + d \quad \text{for all } n.
\]
Equivalently: the difference $u_{n+1} - u_n$ is constant, equal to $d$.
\end{definition}

\begin{theorem}[General term]\label{thm:g11:seq:arithgeneral}
If $(u_n)$ is \omterm{def:g11:seq:arithmetic}{arithmetic} with first term $u_0$ and \omterm{def:g11:seq:arithmetic}{common difference} $d$,
then
\[
u_n = u_0 + n\,d \quad \text{for all } n \geq 0,
\qquad\text{and more generally } u_n = u_p + (n - p)\,d .
\]
\end{theorem}

\begin{proof}
To go from $u_0$ to $u_n$, the rule ``add $d$'' is applied $n$ times: one
step gives $u_1 = u_0 + d$, two steps give $u_2 = u_0 + 2d$, and after $n$
steps every application has contributed one $d$, so $u_n = u_0 + nd$.
(This ``and so on'' is made rigorous by induction in \cref{ch:g12:seq}.)
The general formula follows by counting the $n - p$ steps from $u_p$ to
$u_n$.
\end{proof}

\begin{theorem}[Sum of consecutive integers]\label{thm:g11:seq:intsum}
For every \omterm{def:g10:numbers:sets}{integer} $n \geq 1$:
\[
1 + 2 + \dots + n = \frac{n(n+1)}{2}.
\]
More generally, a sum of consecutive terms of an \omterm{def:g11:seq:arithmetic}{arithmetic sequence}
equals
\[
(\text{number of terms}) \times
\frac{\text{first term} + \text{last term}}{2}.
\]
\end{theorem}

\begin{proof}
Write the sum $S$ twice, the second time in reverse order, and add
column by column:
\[
\begin{array}{ccccccccc}
S & = & 1 & + & 2 & + & \dots & + & n\\
S & = & n & + & (n-1) & + & \dots & + & 1\\
\hline
2S & = & (n+1) & + & (n+1) & + & \dots & + & (n+1)
\end{array}
\]
There are $n$ columns, each summing to $n + 1$, so $2S = n(n+1)$. For a
general \omterm{def:g11:seq:arithmetic}{arithmetic sequence} the same pairing works: first $+$ last $=$
second $+$ second-to-last $= \dots$, because moving one step forward at
the left end ($+d$) is compensated by one step backward at the right end
($-d$).
\end{proof}

\begin{example}
$1 + 2 + \dots + 100 = \frac{100 \times 101}{2} = 5050$. The sum of the
odd numbers $1 + 3 + \dots + 99$ ($50$ terms) is
$50 \times \frac{1 + 99}{2} = 2500$.
\end{example}

\section{Geometric sequences}

\begin{definition}[Geometric sequence]\label{def:g11:seq:geometric}
A \omterm{def:g11:seq:sequence}{sequence} is \emph{geometric}\index{geometric sequence} with \emph{common
ratio}\index{common ratio} $q \neq 0$ if each term is obtained from the previous one by
multiplying by $q$:
\[
u_{n+1} = q\,u_n \quad \text{for all } n.
\]
Equivalently, when no term vanishes: the ratio $\frac{u_{n+1}}{u_n}$ is
constant, equal to $q$.
\end{definition}

\begin{theorem}[General term]\label{thm:g11:seq:geomgeneral}
If $(u_n)$ is \omterm{def:g11:seq:geometric}{geometric} with first term $u_0$ and ratio $q$, then
\[
u_n = u_0\, q^n \quad \text{for all } n \geq 0,
\qquad\text{and more generally } u_n = u_p\, q^{\,n-p} .
\]
\end{theorem}

\begin{proof}
Same step-counting as in \cref{thm:g11:seq:arithgeneral}: from $u_0$ to
$u_n$, the rule ``multiply by $q$'' is applied $n$ times, contributing a
factor $q^n$.
\end{proof}

\begin{theorem}[Geometric sum]\label{thm:g11:seq:geomsum}
For every real $q \neq 1$ and \omterm{def:g10:numbers:sets}{integer} $n \geq 0$:
\[
1 + q + q^2 + \dots + q^n = \frac{1 - q^{\,n+1}}{1 - q}.
\]
\end{theorem}

\begin{proof}
Let $S = 1 + q + \dots + q^n$. Multiply by $q$:
$qS = q + q^2 + \dots + q^{n+1}$. Subtract:
\[
S - qS = \bigl(1 + q + \dots + q^n\bigr)
- \bigl(q + q^2 + \dots + q^{n+1}\bigr) = 1 - q^{\,n+1},
\]
because every intermediate term appears once in each sum and cancels.
Hence $(1 - q)S = 1 - q^{\,n+1}$, and dividing by $1 - q \neq 0$ gives the
formula.
\end{proof}

\begin{example}\label{ex:g11:seq:chessboard}
$1 + 2 + 4 + \dots + 2^{10} = \frac{1 - 2^{11}}{1 - 2} = 2^{11} - 1 =
2047$: doubling grains of rice on the squares of a chessboard overwhelms
any granary well before the $64$th square, where the total is
$2^{64} - 1 \approx 1.8 \times 10^{19}$.
\end{example}

\begin{method}[Recognizing the type of a sequence]\label{met:g11:seq:recognize}
Compute $u_{n+1} - u_n$ and simplify. If the result is a constant $d$, the
\omterm{def:g11:seq:sequence}{sequence} is \omterm{def:g11:seq:arithmetic}{arithmetic}. Otherwise compute $\frac{u_{n+1}}{u_n}$ (terms
nonzero) and simplify: a constant $q$ means \omterm{def:g11:seq:geometric}{geometric}. If neither is
constant, the \omterm{def:g11:seq:sequence}{sequence} is of neither type --- never conclude from the
first few terms alone.
\end{method}

\begin{example}
For $u_n = 3 \times 5^n$:
$\frac{u_{n+1}}{u_n} = \frac{3 \times 5^{n+1}}{3 \times 5^n} = 5$ for all
$n$: \omterm{def:g11:seq:geometric}{geometric} with ratio $5$. For $u_n = n^2$:
$u_1 - u_0 = 1$ but $u_2 - u_1 = 3$, and
$\frac{u_1}{u_0}$ is not even defined --- neither \omterm{def:g11:seq:arithmetic}{arithmetic} nor
\omterm{def:g11:seq:geometric}{geometric}.
\end{example}

\section{Monotonicity}

\begin{definition}[Monotonic sequence]\label{def:g11:seq:monotone}
A \omterm{def:g11:seq:sequence}{sequence} $(u_n)$ is \emph{\omterm{def:g11:func:monotone}{increasing}} if $u_{n+1} \geq u_n$ for all $n$,
and \emph{\omterm{def:g10:functions:variations}{decreasing}} if $u_{n+1} \leq u_n$ for all $n$.
\end{definition}

\begin{method}[Studying monotonicity]\label{met:g11:seq:monotonicity}
Study the sign of $u_{n+1} - u_n$. For \omterm{def:g11:seq:sequence}{sequences} with positive terms, one
may instead compare $\frac{u_{n+1}}{u_n}$ to $1$.
\end{method}

\begin{example}
An \omterm{def:g11:seq:arithmetic}{arithmetic sequence} is \omterm{def:g11:func:monotone}{increasing} when $d \geq 0$
($u_{n+1} - u_n = d$), \omterm{def:g10:functions:variations}{decreasing} when $d \leq 0$. A \omterm{def:g11:seq:geometric}{geometric sequence}
with $u_0 > 0$ and $q > 1$ is \omterm{def:g11:func:monotone}{increasing}:
$u_{n+1} - u_n = u_0 q^n (q - 1) > 0$; with $u_0 > 0$ and $0 < q < 1$ it
is \omterm{def:g10:functions:variations}{decreasing}.
\end{example}

\section{Long-run behaviour, informally}

What happens to $u_n$ when $n$ becomes very large? For an \omterm{def:g11:seq:arithmetic}{arithmetic
sequence} with $d > 0$, the terms $u_0 + nd$ exceed any fixed number
eventually. For a \omterm{def:g11:seq:geometric}{geometric sequence} with $0 < q < 1$, the terms
$u_0 q^n$ shrink towards $0$: multiplying repeatedly by $0.9$, say, erodes
any starting value. And for $q > 1$ the terms explode, as in
\cref{ex:g11:seq:chessboard}.

\begin{remark}
These statements can be made perfectly precise --- ``the terms eventually
stay within any given distance of $0$'' --- and proved. That is the theory
of \emph{limits}, the opening theme of \cref{ch:g12:seq}.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g11:seq:1}
For each \omterm{def:g11:seq:sequence}{sequence}, compute $u_1$, $u_2$, $u_3$:
\[
u_n = \frac{n}{n+1}; \qquad
u_0 = 5,\ u_{n+1} = 3u_n - 2; \qquad
u_n = (-1)^n\,n .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:seq:2}
$(u_n)$ is \omterm{def:g11:seq:arithmetic}{arithmetic} with $u_0 = 7$ and $d = -3$. Compute $u_{10}$ and
$u_{25}$. $(v_n)$ is \omterm{def:g11:seq:arithmetic}{arithmetic} with $v_3 = 11$ and $v_8 = 26$. Find the
\omterm{def:g11:seq:arithmetic}{common difference} and $v_0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:seq:3}
$(u_n)$ is \omterm{def:g11:seq:geometric}{geometric} with $u_0 = 5$ and $q = 2$. Compute $u_8$. $(v_n)$ is
\omterm{def:g11:seq:geometric}{geometric} with positive terms, $v_2 = 12$ and $v_4 = 48$. Find the ratio
and $v_0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:seq:4}
Compute
\[
1 + 2 + 3 + \dots + 500, \qquad
4 + 7 + 10 + \dots + 61, \qquad
1 + \frac12 + \frac14 + \dots + \frac{1}{2^{10}} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:seq:5}
Determine whether each \omterm{def:g11:seq:sequence}{sequence} is \omterm{def:g11:seq:arithmetic}{arithmetic}, \omterm{def:g11:seq:geometric}{geometric}, or neither:
\[
u_n = 4n - 1; \qquad
v_n = \frac{2^n}{3^{n+1}}; \qquad
w_n = n^2 + n .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:seq:6}
A theater has $20$ rows: $16$ seats in the first row, and each row has
$2$ more seats than the one before it. How many seats are in the last
row? In the whole theater?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:seq:7}
A population of bacteria doubles every hour; at noon there are $500$
bacteria. How many are there at 8 p.m.? After how many full hours does the
population first exceed one million? (Solve by trying successive powers of
$2$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:seq:8}
Every month, a saver deposits $100$ euros into an account that pays
$0.2\%$ interest per month on the existing balance (interest is credited
just before the deposit). Let $c_n$ be the balance right after the $n$-th
deposit, so $c_1 = 100$ and $c_{n+1} = 1.002\,c_n + 100$. Compute $c_2$
and $c_3$, and explain why $(c_n)$ is neither \omterm{def:g11:seq:arithmetic}{arithmetic} nor \omterm{def:g11:seq:geometric}{geometric}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:seq:9}
Study the monotonicity of the \omterm{def:g11:seq:sequence}{sequences}
\[
u_n = n^2 - 8n \ (n \geq 0), \qquad
v_n = \frac{3^n}{n!}\ (n \geq 1),
\]
where $n! = 1 \times 2 \times \dots \times n$. (For $(v_n)$, compare
$\frac{v_{n+1}}{v_n}$ to $1$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:seq:10}
The sum of the first $n$ terms of an \omterm{def:g11:seq:arithmetic}{arithmetic sequence} with $u_0 = 3$
and $d = 4$ equals $903$. Find $n$. (Set up a quadratic \omterm{def:g10:algebra:equation}{equation} in $n$
and use \cref{ch:g11:quad}.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:seq:11}
Let $u_0 = 3$ and $u_{n+1} = 2u_n - 1$.
\begin{enumerate}
  \item Compute $u_1, u_2, u_3$ and conjecture a formula for $u_n$.
  \item Let $v_n = u_n - 1$. Show that $(v_n)$ is \omterm{def:g11:seq:geometric}{geometric}, give its
        ratio and first term.
  \item Deduce an explicit formula for $u_n$ and verify your conjecture.
\end{enumerate}
\end{exercise}

\section{Problem: The tower of Brahma and Fibonacci's rabbits}

\begin{problem}[{Weekend problem --- two legendary recurrences:
the tower that ends the world, the sequence that grows like gold,
and the auxiliary trick that tames loans}]
\label{pb:g11:seq:1}
Two \omterm{def:g11:seq:sequence}{sequences} rule the folklore of mathematics. One counts the
moves of the tower of Brahma --- sixty-four golden disks whose
transfer, legend says, will end the world. The other counts
Fibonacci's rabbits and hides the golden ratio. Neither is
\omterm{def:g11:seq:arithmetic}{arithmetic}, neither is \omterm{def:g11:seq:geometric}{geometric} --- and both surrender to this
chapter's weapons: recurrences, \omterm{def:g11:seq:geometric}{geometric} sums
(\cref{thm:g11:seq:geomsum}), and the \omterm{def:g11:seq:sequence}{auxiliary-sequence} trick of
\cref{exo:g11:seq:11}, which also computes your mortgage.

\medskip
\noindent\textbf{Part I --- The tower of Brahma.}
The puzzle: $n$ disks of \omterm{def:g10:functions:variations}{decreasing} size are stacked on peg A;
move the whole stack to peg C, one disk at a time, never placing
a larger disk on a smaller one (peg B may help). Let $h_n$ be the
minimal number of moves.
\begin{enumerate}
  \item Play (with coins) and record $h_1$, $h_2$, $h_3$.
  \item Explain the strategy behind the recurrence
        $h_{n+1} = 2h_n + 1$: what must happen before and after
        the largest disk moves?
  \item Solve the recurrence with the trick of
        \cref{exo:g11:seq:11}: set $v_n = h_n + 1$, show
        $(v_n)$ is \omterm{def:g11:seq:geometric}{geometric}, and conclude $h_n = 2^n - 1$.
  \item The legend's tower has $64$ disks, and the monks move
        one disk per second. Using $2^{10} = 1024 \approx 10^3$,
        estimate the transfer time in years (a year is about
        $3 \times 10^7$ seconds; compare
        \cref{ex:g11:seq:chessboard}, the same giant in another
        story). Should we worry?
  \item Why can no strategy beat $2^n - 1$ moves? Argue that
        \emph{any} solution obeys
        $h_{n+1} \geq 2 h_n + 1$: what must be true of the top
        $n$ disks just before, and just after, the bottom disk's
        move?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Fibonacci.}
Define $F_1 = F_2 = 1$ and $F_{n+2} = F_{n+1} + F_n$ (each term
the sum of the two before --- the rhythm-counting rule of the
Middle School volume, now with its European name).
\begin{enumerate}[resume]
  \item List $F_1$ through $F_{12}$.
  \item Show that $(F_n)$ is neither \omterm{def:g11:seq:arithmetic}{arithmetic} nor \omterm{def:g11:seq:geometric}{geometric},
        but that it is strictly \omterm{def:g11:func:monotone}{increasing} from $n = 2$ on
        (\cref{met:g11:seq:monotonicity} and the recurrence).
  \item Prove the sum identity
        \[
        F_1 + F_2 + \dots + F_n = F_{n+2} - 1
        \]
        by telescoping: write each $F_k$ as
        $F_{k+2} - F_{k+1}$ and watch the sum collapse. Verify
        it for $n = 6$.
  \item Prove the squares identity
        $F_1^2 + F_2^2 + \dots + F_n^2 = F_n F_{n+1}$,
        telescoping with
        $F_k F_{k+1} - F_{k-1} F_k = F_k^2$. Verify for
        $n = 4$. (Picture: squares of sides
        $1, 1, 2, 3, 5, \dots$ tile a rectangle --- the famous
        Fibonacci spiral's skeleton.)
  \item Cassini's identity states
        $F_{n+1} F_{n-1} - F_n^2 = (-1)^n$. Verify it for
        $n = 4, 5, 6$ --- and recognize the engine of the
        vanishing-square trick played in the Middle School
        volume's areas problem.
  \item Show from the recurrence that
        $F_{n+2} \geq 2 F_n$: Fibonacci at least doubles every
        two steps --- it grows at least as fast as a \omterm{def:g11:seq:geometric}{geometric
        sequence} of ratio $\sqrt2$.
  \item Compute the ratios $r_n = \frac{F_{n+1}}{F_n}$ for
        $n = 3$ to $10$ (three decimals). Admitting that they
        settle on a limit $L$, pass the relation
        $r_{n+1} = 1 + \frac{1}{r_n}$ to the limit and solve:
        which number of \cref{pb:g10:algebra:1} do the rabbits
        worship?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The auxiliary trick, at the
bank.}
\begin{enumerate}[resume]
  \item Generalize \cref{exo:g11:seq:11}: for
        $u_{n+1} = a\,u_n + b$ with $a \neq 1$, let
        $\ell = \frac{b}{1 - a}$ (the \omterm{pb:g10:functions:1}{fixed point}). Show
        $v_n = u_n - \ell$ is \omterm{def:g11:seq:geometric}{geometric} of ratio $a$, and
        conclude
        $u_n = a^n (u_0 - \ell) + \ell$.
  \item A loan: $10\,000$ euros at $1\,\%$ interest per month,
        repaid $300$ euros per month, so the debt obeys
        $d_{n+1} = 1.01\,d_n - 300$. Apply question 13 (\omterm{pb:g10:functions:1}{fixed
        point} first!) to get an explicit formula for $d_n$.
  \item With a calculator, find the first month at which the
        debt is cleared, and the total amount repaid. How much
        did the borrowing itself cost?
  \item A town of $50\,000$ inhabitants grows by $2\,\%$ a year
        and welcomes $1\,000$ newcomers besides:
        $p_{n+1} = 1.02\,p_n + 1000$. Give the explicit formula
        and the population after $10$ years.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The two royal families.}
\begin{enumerate}[resume]
  \item Compute $1 + 2 + 3 + \dots + 1000$
        (\cref{thm:g11:seq:intsum} --- little Gauss's sum from
        the Middle School volume, now official), and
        $1 + 2 + 4 + \dots + 2^{19}$
        (\cref{thm:g11:seq:geomsum}).
  \item Compute the sum of the \omterm{def:g11:seq:arithmetic}{arithmetic sequence}
        $7, 12, 17, \dots, 502$ (how many terms?).
  \item Savings plan: $100$ euros deposited each month, earning
        $0.5\,\%$ monthly; after the $n$-th deposit the balance
        is $100\left(1.005^{n-1} + \dots + 1.005 + 1\right)$.
        Compute the balance after $5$ years ($n = 60$).
  \item Finale --- the sequence-tamer's kit: explicit against
        recurrent descriptions; the two royal families and
        their sum formulas; the auxiliary \omterm{def:g11:seq:sequence}{sequence} that turns
        affine recurrences into \omterm{def:g11:seq:geometric}{geometric} ones; and Fibonacci,
        first citizen outside both families, tamed today by
        identities and awaiting matrices (grade 12) and limits
        for full capture. One sentence each.
\end{enumerate}
\end{problem}
