\chapter{Differentiaalrekening: een eerste cursus}\label{ch:g11:deriv}

Hoe snel verandert een functie? Het antwoord in één punt is de
\emph{afgeleide}, het nuttigste getal in alle toegepaste wiskunde. Dit
hoofdstuk construeert haar uit gemiddelde veranderingsverhoudingen,
berekent haar voor de referentiefuncties, en gebruikt haar teken om de
variaties van een functie te lezen. Alles hier wordt uitgebreid en
verdiept in \cref{ch:g12:deriv}.

\section{De gemiddelde veranderingsverhouding}

\begin{definition}[Gemiddelde veranderingsverhouding]\label{def:g11:deriv:rate}
Laat $f$ gedefinieerd zijn op een interval $I$, en $a \neq b$ in $I$. De
\emph{gemiddelde veranderingsverhouding}\index{gemiddelde
veranderingsverhouding} van $f$ tussen $a$ en $b$ is
\[
\frac{f(b) - f(a)}{b - a}.
\]
Het is de richtingscoëfficiënt van de \emph{secant} door de punten
$\bigl(a, f(a)\bigr)$ en $\bigl(b, f(b)\bigr)$ van de kromme.
\end{definition}

\begin{example}
Een auto rijdt $140$ km tussen 14u en 16u: haar gemiddelde snelheid is
$\frac{140}{2} = 70$ km/h. De snelheidsmeter toont echter een
\emph{ogenblikkelijke} snelheid op elk moment --- dat is precies het
begrip dat de afgeleide precies maakt.
\end{example}

\section{De afgeleide in een punt}

\begin{definition}[Afgeleide in een punt]\label{def:g11:deriv:derivative}
Laat $f$ gedefinieerd zijn op een interval dat $a$ bevat. Voor $h \neq 0$,
vorm de gemiddelde veranderingsverhouding tussen $a$ en $a + h$:
\[
r(h) = \frac{f(a+h) - f(a)}{h}.
\]
Als $r(h)$ een enkel vast getal nadert wanneer $h$ willekeurig dicht bij
$0$ komt, zeggen we dat $f$ \emph{afleidbaar is in $a$}\index{afgeleide};
dit getal is de \emph{afgeleide van $f$ in $a$}, genoteerd $f'(a)$:
\[
f'(a) = \lim_{h \to 0} \frac{f(a+h) - f(a)}{h}.
\]
\end{definition}

\begin{example}\label{ex:g11:deriv:atpoint}
Laat $f(x) = x^2$ en $a = 1$. Voor $h \neq 0$:
\[
\frac{f(1+h) - f(1)}{h}
= \frac{(1+h)^2 - 1}{h}
= \frac{1 + 2h + h^2 - 1}{h}
= \frac{h(2 + h)}{h}
= 2 + h .
\]
Als $h$ naar $0$ nadert, nadert dit $2$: de functie $x^2$ is afleidbaar in
$1$ en $f'(1) = 2$. Let op de strategie: \emph{vereenvoudig het quotiënt
tot $h$ niet meer in een noemer voorkomt}, en laat daarna $h \to 0$.
\end{example}

\begin{example}[Een niet-afleidbaar punt]
Laat $f(x) = \abs{x}$ en $a = 0$. Dan
$\frac{f(h) - f(0)}{h} = \frac{\abs h}{h}$, wat $1$ is voor $h > 0$ en
$-1$ voor $h < 0$: er wordt geen enkel getal genaderd, en $\abs x$ is
niet afleidbaar in $0$. Haar kromme heeft daar een \emph{hoek}.
\end{example}

\section{De raaklijn}

\begin{definition}[Raaklijn]\label{def:g11:deriv:tangent}
Als $f$ afleidbaar is in $a$, is de \emph{raaklijn}\index{raaklijn} aan de
kromme in het punt $A = \bigl(a, f(a)\bigr)$ de lijn door $A$ met
richtingscoëfficiënt $f'(a)$. Haar vergelijking is
\[
y = f(a) + f'(a)(x - a).
\]
\end{definition}

De raaklijn is de limietpositie van de secanten door $A$: als $h \to 0$,
schuift het tweede punt $\bigl(a+h, f(a+h)\bigr)$ langs de kromme naar
$A$, en de richtingscoëfficiënt $\frac{f(a+h)-f(a)}{h}$ van de secant
nadert $f'(a)$.

\begin{example}\label{ex:g11:deriv:tangenteq}
Voor $f(x) = x^2$ in $a = 1$: $f(1) = 1$ en $f'(1) = 2$
(\cref{ex:g11:deriv:atpoint}), dus de raaklijn is
\[
y = 1 + 2(x - 1) = 2x - 1 .
\]
Dit is de lijn $d_2$ die puur algebraïsch gevonden werd in
\cref{exo:g11:quad:11}.
\end{example}

\section{Afgeleiden van de referentiefuncties}

\begin{proposition}\label{prop:g11:deriv:refderivatives}
Op hun domeinen van afleidbaarheid:
\[
(c)' = 0, \quad
(x)' = 1, \quad
(x^2)' = 2x, \quad
(x^3)' = 3x^2, \quad
\left(\frac1x\right)' = -\frac{1}{x^2}, \quad
(\sqrt x\,)' = \frac{1}{2\sqrt x} \ (x > 0).
\]
Algemener, $(x^n)' = n\,x^{n-1}$ voor elk geheel getal $n \geq 1$.
\end{proposition}

\begin{proof}
Elk bewijs volgt de strategie van \cref{ex:g11:deriv:atpoint}:
vereenvoudig de verhouding, laat daarna $h \to 0$. Fixeer $a$ in het
domein.

\emph{Constante en identiteit.} $\frac{c - c}{h} = 0$ en
$\frac{(a+h) - a}{h} = 1$ voor alle $h$: de limieten zijn $0$ en $1$.

\emph{Kwadraat.} $\frac{(a+h)^2 - a^2}{h} = \frac{2ah + h^2}{h} = 2a + h
\to 2a$.

\emph{Derdemacht.} $(a+h)^3 = a^3 + 3a^2h + 3ah^2 + h^3$, dus de
verhouding is $3a^2 + 3ah + h^2 \to 3a^2$.

\emph{Omgekeerde.} Voor $a \neq 0$ en $h$ klein genoeg dat
$a + h \neq 0$:
\[
\frac{1}{h}\left(\frac{1}{a+h} - \frac{1}{a}\right)
= \frac{1}{h} \cdot \frac{a - (a+h)}{a(a+h)}
= \frac{-1}{a(a+h)}
\xrightarrow[h \to 0]{} -\frac{1}{a^2}.
\]

\emph{Vierkantswortel.} Voor $a > 0$, vermenigvuldig met de toegevoegde:
\[
\frac{\sqrt{a+h} - \sqrt a}{h}
= \frac{(a + h) - a}{h\,(\sqrt{a+h} + \sqrt a)}
= \frac{1}{\sqrt{a+h} + \sqrt a}
\xrightarrow[h \to 0]{} \frac{1}{2\sqrt a}.
\]

\emph{Algemene macht.} Het patroon $2a$, $3a^2$ zet door: bij ontwikkelen
van $(a+h)^n$ dragen de termen na $a^n + n\,a^{n-1}h$ alle $h^2$, dus de
verhouding neigt naar $n\,a^{n-1}$. (Een volledige inductie staat in
\cref{ch:g12:deriv}.)
\end{proof}

\section{Differentiatieregels}

\begin{theorem}[Bewerkingen]\label{thm:g11:deriv:rules}
Laat $u$ en $v$ afleidbaar zijn op $I$ en $\lambda \in \R$. Dan zijn
$u + v$, $\lambda u$, $uv$ afleidbaar op $I$, evenals $\frac{u}{v}$ waar
$v \neq 0$, en
\[
(u+v)' = u' + v', \qquad
(\lambda u)' = \lambda u', \qquad
(uv)' = u'v + uv', \qquad
\left(\frac{u}{v}\right)' = \frac{u'v - uv'}{v^2}.
\]
\end{theorem}

\begin{proof}[Bewijs van de eerste drie regels]
Fixeer $a \in I$ en noteer kort $\Delta u = u(a+h) - u(a)$,
$\Delta v = v(a+h) - v(a)$.

\emph{Som.} De verhouding van $u + v$ is
$\frac{\Delta u + \Delta v}{h} = \frac{\Delta u}{h} + \frac{\Delta v}{h}
\to u'(a) + v'(a)$.

\emph{Veelvoud.} De verhouding van $\lambda u$ is
$\lambda \frac{\Delta u}{h} \to \lambda u'(a)$.

\emph{Product.} Tel op en trek af de gemengde term $u(a+h)\,v(a)$:
\begin{align*}
\frac{u(a+h)v(a+h) - u(a)v(a)}{h}
&= \frac{u(a+h)v(a+h) - u(a+h)v(a) + u(a+h)v(a) - u(a)v(a)}{h}\\
&= u(a+h)\,\frac{\Delta v}{h} + v(a)\,\frac{\Delta u}{h}.
\end{align*}
Als $h \to 0$: $\frac{\Delta v}{h} \to v'(a)$, $\frac{\Delta u}{h} \to
u'(a)$, en $u(a+h) \to u(a)$ (haar verhouding heeft een limiet, dus
$\Delta u = h \cdot \frac{\Delta u}{h} \to 0$). De verhouding neigt dus
naar $u(a)v'(a) + v(a)u'(a)$.

De quotiëntrregel wordt op dezelfde wijze bewezen; ze is \emph{toegegeven
op dit niveau} en bewezen in \cref{ch:g12:deriv}.
\end{proof}

\begin{example}\label{ex:g11:deriv:computations}
\emph{Polynoom.} $f(x) = 3x^3 - 5x^2 + 7x - 2$:
\[
f'(x) = 3 \times 3x^2 - 5 \times 2x + 7 = 9x^2 - 10x + 7 .
\]

\emph{Quotiënt.} $g(x) = \frac{2x+1}{x^2+1}$: met $u = 2x + 1$, $v = x^2
+ 1$,
\[
g'(x) = \frac{2(x^2+1) - (2x+1)\,2x}{(x^2+1)^2}
= \frac{-2x^2 - 2x + 2}{(x^2+1)^2}.
\]

\emph{Product.} $k(x) = x\sqrt x$ voor $x > 0$:
$k'(x) = 1 \times \sqrt x + x \times \frac{1}{2\sqrt x}
= \sqrt x + \frac{\sqrt x}{2} = \frac{3\sqrt x}{2}$.
\end{example}

\section{Teken van de afgeleide en variaties}

\begin{theorem}[Afgeleide en variaties]\label{thm:g11:deriv:variations}
Laat $f$ afleidbaar zijn op een interval $I$.
\begin{enumerate}
  \item Als $f' \geq 0$ op $I$, dan is $f$ stijgend op $I$.
  \item Als $f' \leq 0$ op $I$, dan is $f$ dalend op $I$.
  \item Als $f' > 0$ op $I$ (behalve eventueel in eindig veel punten waar
        ze verdwijnt), dan is $f$ strikt stijgend op $I$.
\end{enumerate}
\end{theorem}

\begin{proof}
De intuïtie is duidelijk --- een kromme waarvan alle raaklijnen omhoog
wijzen moet klimmen --- maar een rigoureus bewijs heeft de
middelwaardestelling nodig. Het resultaat is \emph{toegegeven op dit
niveau}; zie \cref{thm:g12:deriv:variations}.
\end{proof}

\begin{proposition}[Extrema]\label{prop:g11:deriv:extrema}
Als $f'$ van teken wisselt in $a$ van $+$ naar $-$, dan heeft $f$ een
lokaal maximum in $a$; van $-$ naar $+$, een lokaal minimum. In zo'n punt
is $f'(a) = 0$ en de raaklijn is horizontaal.
\end{proposition}

\begin{proof}
Als $f' \geq 0$ op een interval dat eindigt in $a$ en $f' \leq 0$ op een
interval dat in $a$ begint, dan stijgt $f$ tot $f(a)$ en daalt daarna
(\cref{thm:g11:deriv:variations}): $f(a)$ is het grootst in de buurt.
Omdat $f'$ van teken wisselt in $a$ en daar gedefinieerd is, is
$f'(a) = 0$.
\end{proof}

\begin{method}[Een functie bestuderen]\label{met:g11:deriv:study}
\begin{enumerate}
  \item Bereken $f'(x)$ en vereenvoudig haar --- idealiter tot een
        ontbonden vorm.
  \item Bepaal het teken van $f'(x)$ op elk interval (voor een
        kwadratische $f'$, gebruik \cref{thm:g11:quad:sign}).
  \item Noteer de resultaten in een \emph{veranderingstabel}: één rij
        voor het teken van $f'$, één rij pijlen voor $f$, met de
        waarden van $f$ in de keerpunten.
  \item Concludeer: extrema, aantal oplossingen van $f(x) = k$,
        ongelijkheden.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g11:deriv:study}
Bestudeer $f(x) = x^3 - 3x + 1$ op $\R$. Eerst
$f'(x) = 3x^2 - 3 = 3(x-1)(x+1)$: een kwadratische met wortels $-1$ en
$1$, positief erbuiten. Dus $f$ is stijgend op $\intoc{-\infty}{-1}$,
dalend op $\intcc{-1}{1}$, stijgend op $\intco{1}{+\infty}$, met een
lokaal maximum $f(-1) = -1 + 3 + 1 = 3$ en een lokaal minimum
$f(1) = 1 - 3 + 1 = -1$.
\end{example}

\begin{example}[Optimalisatie]\label{ex:g11:deriv:box}
Een open doos wordt gemaakt van een blad van $12 \times 12$ cm door een
vierkant van zijde $x$ uit elke hoek te snijden en om te vouwen. Haar
inhoud is $V(x) = x(12 - 2x)^2$ voor $0 < x < 6$. Ontwikkelen geeft
$V(x) = 4x^3 - 48x^2 + 144x$, dus
\[
V'(x) = 12x^2 - 96x + 144 = 12(x^2 - 8x + 12) = 12(x - 2)(x - 6).
\]
Op $\intoo{0}{6}$ is de factor $x - 6$ negatief, dus $V'(x)$ heeft het
teken van $-(x-2)$: positief vóór $2$, negatief erna. De inhoud is
maximaal voor $x = 2$ cm, wat $V(2) = 2 \times 8^2 = 128$ cm$^3$ geeft.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g11:deriv:1}
Bereken met de definitie (veranderingsverhouding, daarna $h \to 0$)
$f'(2)$ voor $f(x) = x^2$, daarna $g'(1)$ voor $g(x) = x^2 + 3x$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:deriv:2}
Differentieer:
\[
f(x) = 4x^3 - 2x^2 + x - 7, \qquad
g(x) = 5\sqrt{x} + \frac{3}{x}, \qquad
h(x) = (2x+1)(x^2 - 3) .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:deriv:3}
Geef de vergelijking van de raaklijn aan de kromme van $f$ in het punt met
abscis $a$:
\[
f(x) = x^2 - 3x, \ a = 2; \qquad
f(x) = \frac1x, \ a = 1; \qquad
f(x) = \sqrt x, \ a = 4 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:deriv:4}
Differentieer de quotiënten:
\[
f(x) = \frac{x+2}{x-1}, \qquad
g(x) = \frac{x^2}{x^2+1}, \qquad
h(x) = \frac{1}{x^2 + x + 1} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:deriv:5}
Bestudeer de variaties van $f(x) = x^3 - 6x^2 + 9x - 2$ op $\R$
(veranderingstabel, lokale extrema).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:deriv:6}
Bestudeer de variaties van $f(x) = \frac{x^2 + 3}{x + 1}$ op
$\intoo{-1}{+\infty}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:deriv:7}
Laat $f(x) = x^2 - 4x + 5$.
\begin{enumerate}
  \item Vind het punt van de kromme waar de raaklijn horizontaal is.
  \item Vind het/de punt(en) waar de raaklijn evenwijdig is met de lijn
        $y = 2x + 1$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:deriv:8}
Een rechthoekige omheining wordt tegen een muur gebouwd met $60$ m hek
(drie zijden). Druk de oppervlakte uit als functie van de breedte $x$, en
gebruik de afgeleide om de afmetingen van maximale oppervlakte te vinden.
(Vergelijk met de topmethode van \cref{ch:g11:quad}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:deriv:9}
Toon aan dat voor alle $x > 0$, $\sqrt{x} \leq \frac{x + 1}{2}$, door de
functie $f(x) = \frac{x+1}{2} - \sqrt x$ te bestuderen op
$\intoo{0}{+\infty}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:deriv:10}
Een cilindrisch blik moet $500$ cm$^3$ bevatten. Haar oppervlakte (twee
schijven plus de zijkant) is $S = 2\pi r^2 + 2\pi r h$ met
$\pi r^2 h = 500$. Druk $S$ uit als functie van $r$ alleen, en toon aan
dat $S'(r) = 4\pi r - \frac{1000}{r^2}$. Leid de straal af die de
oppervlakte minimaliseert, als exacte uitdrukking.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:deriv:11}
Hoeveel raaklijnen aan de parabool $y = x^2$ gaan door het externe punt
$P(1, -3)$? Bepaal hun vergelijkingen. (Laat $a$ de abscis van het
raakpunt zijn en druk uit dat de raaklijn in $a$ door $P$ gaat.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:deriv:12}
Laat $f(x) = x^3 - 3x + 1$ (zie \cref{ex:g11:deriv:study}). Bepaal met de
veranderingstabel het aantal oplossingen van de vergelijking $f(x) = k$
naargelang de waarde van het reële getal $k$.
\end{exercise}
