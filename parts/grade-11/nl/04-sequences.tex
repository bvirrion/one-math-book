\chapter{Rijen: een eerste cursus}\label{ch:g11:seq}

Een \omterm{def:g11:seq:sequence}{rij} is een lijst getallen voortgebracht door een regel: de
opeenvolgende saldi van een spaarrekening, de groottes van een populatie
jaar na jaar. Dit hoofdstuk bestudeert de twee families die toepassingen
domineren --- \emph{rekenkundige} \omterm{def:g11:seq:sequence}{rijen}, die met gelijke stappen groeien,
en \emph{meetkundige} \omterm{def:g11:seq:sequence}{rijen}, die met gelijke verhoudingen groeien. De
rigoureuze theorie van limieten wordt ontwikkeld in \cref{ch:g12:seq}.

\section{Een rij definiĂ«ren}

\begin{definition}[Rij]\label{def:g11:seq:sequence}
Een \emph{rij}\index{rij} $(u_n)$ koppelt aan elk \omterm{def:g10:numbers:sets}{geheel getal} $n \geq 0$
(of $n \geq 1$) een \omterm{def:g10:numbers:sets}{reĂ«el getal} $u_n$, haar \emph{term van index $n$}.
Een rij kan gegeven worden
\begin{itemize}
  \item \emph{expliciet}, door een formule voor $u_n$ in termen van $n$:
        bijv.\ $u_n = n^2 + 1$;
  \item \emph{recursief}, door haar eerste term en een regel om van elke
        term naar de volgende te gaan: bijv.\ $u_0 = 3$ en
        $u_{n+1} = 2u_n - 1$.
\end{itemize}
\end{definition}

\begin{example}
Voor $u_n = n^2 + 1$: $u_0 = 1$, $u_1 = 2$, $u_2 = 5$, en $u_{10} = 101$
direct. Voor $u_0 = 3$, $u_{n+1} = 2u_n - 1$: $u_1 = 5$, $u_2 = 9$,
$u_3 = 17$ --- elke term heeft de vorige nodig; $u_{10}$ bereiken kost
tien stappen (of een algemene formule, zie \cref{exo:g11:seq:11}).
\end{example}

\section{Rekenkundige rijen}

\begin{definition}[Rekenkundige rij]\label{def:g11:seq:arithmetic}
Een \omterm{def:g11:seq:sequence}{rij} is \emph{rekenkundig}\index{rekenkundige rij} met
\emph{gemeenschappelijk verschil}\index{gemeenschappelijk verschil} $d$ als elke term uit de vorige
verkregen wordt door $d$ op te tellen:
\[
u_{n+1} = u_n + d \quad \text{voor alle } n.
\]
Equivalent: het verschil $u_{n+1} - u_n$ is constant, gelijk aan $d$.
\end{definition}

\begin{theorem}[Algemene term]\label{thm:g11:seq:arithgeneral}
Als $(u_n)$ \omterm{def:g11:seq:arithmetic}{rekenkundig} is met eerste term $u_0$ en \omterm{def:g11:seq:arithmetic}{gemeenschappelijk
verschil} $d$, dan
\[
u_n = u_0 + n\,d \quad \text{voor alle } n \geq 0,
\qquad\text{en algemener } u_n = u_p + (n - p)\,d .
\]
\end{theorem}

\begin{proof}
Om van $u_0$ naar $u_n$ te gaan wordt de regel ``tel $d$ op'' $n$ keer
toegepast: Ă©Ă©n stap geeft $u_1 = u_0 + d$, twee stappen geven
$u_2 = u_0 + 2d$, en na $n$ stappen heeft elke toepassing Ă©Ă©n $d$
bijgedragen, dus $u_n = u_0 + nd$. (Dit ``enzovoort'' wordt rigoriseerd
door inductie in \cref{ch:g12:seq}.) De algemene formule volgt door de
$n - p$ stappen van $u_p$ naar $u_n$ te tellen.
\end{proof}

\begin{theorem}[Som van opeenvolgende gehele getallen]\label{thm:g11:seq:intsum}
Voor elk \omterm{def:g10:numbers:sets}{geheel getal} $n \geq 1$:
\[
1 + 2 + \dots + n = \frac{n(n+1)}{2}.
\]
Algemener is een som van opeenvolgende termen van een \omterm{def:g11:seq:arithmetic}{rekenkundige rij}
gelijk aan
\[
(\text{aantal termen}) \times
\frac{\text{eerste term} + \text{laatste term}}{2}.
\]
\end{theorem}

\begin{proof}
Schrijf de som $S$ twee keer, de tweede keer in omgekeerde volgorde, en
tel kolom voor kolom op:
\[
\begin{array}{ccccccccc}
S & = & 1 & + & 2 & + & \dots & + & n\\
S & = & n & + & (n-1) & + & \dots & + & 1\\
\hline
2S & = & (n+1) & + & (n+1) & + & \dots & + & (n+1)
\end{array}
\]
Er zijn $n$ kolommen, elk sommerend tot $n + 1$, dus $2S = n(n+1)$. Voor
een algemene \omterm{def:g11:seq:arithmetic}{rekenkundige rij} werkt hetzelfde paren: eerste $+$ laatste
$=$ tweede $+$ voorlaatste $= \dots$, omdat Ă©Ă©n stap vooruit aan het
linkereind ($+d$) gecompenseerd wordt door Ă©Ă©n stap achteruit aan het
rechtereind ($-d$).
\end{proof}

\begin{example}
$1 + 2 + \dots + 100 = \frac{100 \times 101}{2} = 5050$. De som van de
oneven getallen $1 + 3 + \dots + 99$ ($50$ termen) is
$50 \times \frac{1 + 99}{2} = 2500$.
\end{example}

\section{Meetkundige rijen}

\begin{definition}[Meetkundige rij]\label{def:g11:seq:geometric}
Een \omterm{def:g11:seq:sequence}{rij} is \emph{meetkundig}\index{meetkundige rij} met
\emph{gemeenschappelijke verhouding}\index{gemeenschappelijke verhouding} $q \neq 0$ als elke term uit de
vorige verkregen wordt door met $q$ te vermenigvuldigen:
\[
u_{n+1} = q\,u_n \quad \text{voor alle } n.
\]
Equivalent, wanneer geen term verdwijnt: de verhouding
$\frac{u_{n+1}}{u_n}$ is constant, gelijk aan $q$.
\end{definition}

\begin{theorem}[Algemene term]\label{thm:g11:seq:geomgeneral}
Als $(u_n)$ \omterm{def:g11:seq:geometric}{meetkundig} is met eerste term $u_0$ en verhouding $q$, dan
\[
u_n = u_0\, q^n \quad \text{voor alle } n \geq 0,
\qquad\text{en algemener } u_n = u_p\, q^{\,n-p} .
\]
\end{theorem}

\begin{proof}
Zelfde stappentelling als in \cref{thm:g11:seq:arithgeneral}: van $u_0$
naar $u_n$ wordt de regel ``vermenigvuldig met $q$'' $n$ keer toegepast,
wat een factor $q^n$ bijdraagt.
\end{proof}

\begin{theorem}[Meetkundige som]\label{thm:g11:seq:geomsum}
Voor elke reĂ«le $q \neq 1$ en \omterm{def:g10:numbers:sets}{geheel getal} $n \geq 0$:
\[
1 + q + q^2 + \dots + q^n = \frac{1 - q^{\,n+1}}{1 - q}.
\]
\end{theorem}

\begin{proof}
Laat $S = 1 + q + \dots + q^n$. Vermenigvuldig met $q$:
$qS = q + q^2 + \dots + q^{n+1}$. Trek af:
\[
S - qS = \bigl(1 + q + \dots + q^n\bigr)
- \bigl(q + q^2 + \dots + q^{n+1}\bigr) = 1 - q^{\,n+1},
\]
omdat elke tussenterm Ă©Ă©n keer in elke som voorkomt en wegvalt. Dus
$(1 - q)S = 1 - q^{\,n+1}$, en delen door $1 - q \neq 0$ geeft de
formule.
\end{proof}

\begin{example}\label{ex:g11:seq:chessboard}
$1 + 2 + 4 + \dots + 2^{10} = \frac{1 - 2^{11}}{1 - 2} = 2^{11} - 1 =
2047$: rijstkorrels verdubbelen op de velden van een schaakbord overweldigt
elke graanschuur ruim vĂłĂłr het $64$ste veld, waar het totaal
$2^{64} - 1 \approx 1.8 \times 10^{19}$ is.
\end{example}

\begin{method}[Het type van een rij herkennen]\label{met:g11:seq:recognize}
Bereken $u_{n+1} - u_n$ en vereenvoudig. Als het resultaat een constante
$d$ is, is de \omterm{def:g11:seq:sequence}{rij} \omterm{def:g11:seq:arithmetic}{rekenkundig}. Anders bereken $\frac{u_{n+1}}{u_n}$
(termen niet-nul) en vereenvoudig: een constante $q$ betekent \omterm{def:g11:seq:geometric}{meetkundig}.
Als geen van beide constant is, is de \omterm{def:g11:seq:sequence}{rij} van geen van beide types ---
concludeer nooit alleen uit de eerste paar termen.
\end{method}

\begin{example}
Voor $u_n = 3 \times 5^n$:
$\frac{u_{n+1}}{u_n} = \frac{3 \times 5^{n+1}}{3 \times 5^n} = 5$ voor
alle $n$: \omterm{def:g11:seq:geometric}{meetkundig} met verhouding $5$. Voor $u_n = n^2$:
$u_1 - u_0 = 1$ maar $u_2 - u_1 = 3$, en $\frac{u_1}{u_0}$ is zelfs niet
gedefinieerd --- noch \omterm{def:g11:seq:arithmetic}{rekenkundig} noch \omterm{def:g11:seq:geometric}{meetkundig}.
\end{example}

\section{Monotonie}

\begin{definition}[Monotone rij]\label{def:g11:seq:monotone}
Een \omterm{def:g11:seq:sequence}{rij} $(u_n)$ is \emph{\omterm{def:g11:func:monotone}{stijgend}} als $u_{n+1} \geq u_n$ voor alle $n$,
en \emph{\omterm{def:g10:functions:variations}{dalend}} als $u_{n+1} \leq u_n$ voor alle $n$.
\end{definition}

\begin{method}[Monotonie bestuderen]\label{met:g11:seq:monotonicity}
Bestudeer het teken van $u_{n+1} - u_n$. Voor \omterm{def:g11:seq:sequence}{rijen} met positieve termen
mag men in plaats daarvan $\frac{u_{n+1}}{u_n}$ met $1$ vergelijken.
\end{method}

\begin{example}
Een \omterm{def:g11:seq:arithmetic}{rekenkundige rij} is \omterm{def:g11:func:monotone}{stijgend} wanneer $d \geq 0$
($u_{n+1} - u_n = d$), \omterm{def:g10:functions:variations}{dalend} wanneer $d \leq 0$. Een \omterm{def:g11:seq:geometric}{meetkundige rij}
met $u_0 > 0$ en $q > 1$ is \omterm{def:g11:func:monotone}{stijgend}:
$u_{n+1} - u_n = u_0 q^n (q - 1) > 0$; met $u_0 > 0$ en $0 < q < 1$ is ze
\omterm{def:g10:functions:variations}{dalend}.
\end{example}

\section{Lange-termijngedrag, informeel}

Wat gebeurt er met $u_n$ wanneer $n$ zeer groot wordt? Voor een
\omterm{def:g11:seq:arithmetic}{rekenkundige rij} met $d > 0$ overschrijden de termen $u_0 + nd$
uiteindelijk elk vast getal. Voor een \omterm{def:g11:seq:geometric}{meetkundige rij} met $0 < q < 1$
krimpen de termen $u_0 q^n$ naar $0$: herhaaldelijk vermenigvuldigen met
$0.9$, zeg, erodeert elke startwaarde. En voor $q > 1$ exploderen de
termen, zoals in \cref{ex:g11:seq:chessboard}.

\begin{remark}
Deze uitspraken kunnen perfect precies gemaakt worden --- ``de termen
blijven uiteindelijk binnen elke gegeven afstand van $0$'' --- en
bewezen. Dat is de theorie van \emph{limieten}, het openings thema van
\cref{ch:g12:seq}.
\end{remark}

Bereken voor elke \omterm{def:g11:seq:sequence}{rij} $u_1$, $u_2$, $u_3$:

u_0 = 5,\ u_{n+1} = 3u_n - 2; \qquad

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:seq:2}
$(u_n)$ is \omterm{def:g11:seq:arithmetic}{rekenkundig} met $u_0 = 7$ en $d = -3$. Bereken $u_{10}$ en
$u_{25}$. $(v_n)$ is \omterm{def:g11:seq:arithmetic}{rekenkundig} met $v_3 = 11$ en $v_8 = 26$. Vind het
\omterm{def:g11:seq:arithmetic}{gemeenschappelijk verschil} en $v_0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:seq:3}
$(u_n)$ is \omterm{def:g11:seq:geometric}{meetkundig} met $u_0 = 5$ en $q = 2$. Bereken $u_8$. $(v_n)$ is

\begin{exercise}[$\star$]\label{exo:g11:seq:5}

\begin{exercise}[$\star\star$]\label{exo:g11:seq:6}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:seq:8}
Elke maand stort een spaarder $100$ euro op een rekening die $0.2\%$

en $c_3$, en leg uit waarom $(c_n)$ noch \omterm{def:g11:seq:arithmetic}{rekenkundig} noch \omterm{def:g11:seq:geometric}{meetkundig} is.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:seq:9}
Bestudeer de monotonie van de \omterm{def:g11:seq:sequence}{rijen}

\begin{exercise}[$\star\star$]\label{exo:g11:seq:10}
