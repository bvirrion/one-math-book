\chapter{Vectoren en lijnen in het vlak}\label{ch:g11:vect}

Vectoren coderen verplaatsingen; lijnen zijn hun banen. Het nieuwe
sleutelinstrument van dit hoofdstuk is de \emph{determinant}, één getal
dat beslist of twee vectoren collineair zijn --- en van daaruit
vergelijkingen van lijnen produceert en collineariteits- en
evenwijdigheidsvragen door pure berekening beslist. Dezelfde ideeën
worden uitgebreid naar de ruimte in \cref{ch:g12:space}.

\section{Vectoren: een herhaling}

\begin{definition}[Vector, coördinaten]\label{def:g11:vect:vector}
Een \emph{vector}\index{vector} $\vec u$ stelt een verplaatsing voor:
alle pijlen met dezelfde richting, zin en lengte stellen dezelfde
vector voor. In een coördinatenstelsel noteert $\vec u\,(x, y)$ de
verplaatsing ($x$ horizontaal, $y$ verticaal); voor punten $A(x_A, y_A)$
en $B(x_B, y_B)$ is
\[
\vect{AB}\,\bigl(x_B - x_A,\ y_B - y_A\bigr).
\]
Vectoren worden coördinaatsgewijs opgeteld, en $\lambda\vec u$ heeft
coördinaten $(\lambda x, \lambda y)$.
\end{definition}

\begin{proposition}[Midden]\label{prop:g11:vect:midpoint}
Het midden $M$ van het lijnstuk $[AB]$ heeft coördinaten
$\left(\frac{x_A + x_B}{2},\ \frac{y_A + y_B}{2}\right)$.
\end{proposition}

\begin{proof}
$M$ wordt gekarakteriseerd door $\vect{AM} = \frac12\vect{AB}$:
coördinaatsgewijs $x_M - x_A = \frac12(x_B - x_A)$, dus
$x_M = \frac{x_A + x_B}{2}$, en evenzo voor $y_M$.
\end{proof}

\section{Collineariteit en de determinant}

\begin{definition}[Collineaire vectoren]\label{def:g11:vect:collinear}
Twee vectoren zijn \emph{collineair}\index{collineaire vectoren} als de
ene een scalaïr veelvoud is van de andere: $\vec v = \lambda \vec u$
voor zekere $\lambda \in \R$ (of $\vec u = \vec 0$). Meetkundig: ze
hebben dezelfde richting, of één van beide is nul.
\end{definition}

\begin{definition}[Determinant]\label{def:g11:vect:det}
De \emph{determinant}\index{determinant} van $\vec u\,(x, y)$ en
$\vec v\,(x', y')$ is
\[
\det(\vec u, \vec v) =
\begin{vmatrix} x & x' \\ y & y' \end{vmatrix}
= x y' - x' y .
\]
\end{definition}

\begin{theorem}[Collineariteitscriterium]\label{thm:g11:vect:collinearity}
Twee vectoren $\vec u$ en $\vec v$ zijn collineair dan en slechts dan
als $\det(\vec u, \vec v) = 0$.
\end{theorem}

\begin{proof}
($\Rightarrow$) Als $\vec v = \lambda\vec u$, dan $x' = \lambda x$ en
$y' = \lambda y$, dus
$\det(\vec u, \vec v) = x(\lambda y) - (\lambda x)y = 0$; als
$\vec u = \vec 0$, is de determinant triviaal $0$.

($\Leftarrow$) Stel $xy' - x'y = 0$ en $\vec u \neq \vec 0$, zeg
$x \neq 0$ (het geval $y \neq 0$ is symmetrisch). Stel
$\lambda = \frac{x'}{x}$. Dan $x' = \lambda x$ per constructie, en de
relatie $xy' = x'y = \lambda x y$ geeft, na delen door $x \neq 0$,
$y' = \lambda y$. Dus $\vec v = \lambda \vec u$.
\end{proof}

\begin{example}\label{ex:g11:vect:collinear}
$\vec u\,(3, -2)$ en $\vec v\,(-6, 4)$:
$\det = 3 \times 4 - (-6) \times (-2) = 12 - 12 = 0$, collineair
(inderdaad $\vec v = -2\vec u$). $\vec u\,(3, -2)$ en $\vec w\,(2, 1)$:
$\det = 3 + 4 = 7 \neq 0$, niet collineair.
\end{example}

\begin{method}[Collineariteit van drie punten]\label{met:g11:vect:alignment}
$A$, $B$, $C$ zijn collineair dan en slechts dan als $\vect{AB}$ en
$\vect{AC}$ collineair zijn: bereken beide vectoren en controleer
$\det\bigl(\vect{AB}, \vect{AC}\bigr) = 0$.
\end{method}

\section{Cartesiaanse vergelijkingen van lijnen}

\begin{definition}[Richtingsvector]\label{def:g11:vect:direction}
Een \emph{richtingsvector}\index{richtingsvector} van een lijn $d$ is
elke niet-nul vector $\vec u$ zodat $\vect{AB}$ collineair is met
$\vec u$ voor alle punten $A, B$ van $d$.
\end{definition}

\begin{theorem}[Cartesiaanse vergelijking]\label{thm:g11:vect:cartesian}
Een lijn met richtingsvector $\vec u\,(-b, a)$ laat een vergelijking toe
van de vorm
\[
ax + by + c = 0
\qquad (a, b) \neq (0, 0).
\]
Omgekeerd beschrijft elke zulke vergelijking een lijn met
richtingsvector $\vec u\,(-b, a)$.
\end{theorem}

\begin{proof}
Laat $d$ door $A(x_A, y_A)$ gaan met richting $\vec u\,(-b, a)$. Een
punt $M(x, y)$ ligt op $d$ precies wanneer $\vect{AM}$ collineair is
met $\vec u$, d.w.z.\ (\cref{thm:g11:vect:collinearity})
\[
0 = \det\bigl(\vect{AM}, \vec u\bigr)
= (x - x_A)\,a - (-b)(y - y_A)
= ax + by - (a x_A + b y_A),
\]
een vergelijking van de aangekondigde vorm met
$c = -(a x_A + b y_A)$. Omgekeerd zijn de oplossingen van
$ax + by + c = 0$ met, zeg, $b \neq 0$ de punten
$\left(x, -\frac{a x + c}{b}\right)$: aftrekken van twee ervan toont dat
elke koorde collineair is met $(-b, a)$, en er bestaat altijd een
oplossing --- het is de lijn erdoor met richting $(-b, a)$.
\end{proof}

\begin{method}[Lijn door twee punten]\label{met:g11:vect:twopoints}
Om een vergelijking van de lijn $(AB)$ te vinden: bereken de
richtingsvector $\vect{AB}\,(x_B - x_A, y_B - y_A)$; match die met
$(-b, a)$, d.w.z.\ neem $a = y_B - y_A$ en $b = -(x_B - x_A)$; bepaal
$c$ door de coördinaten van $A$ in te vullen.
\end{method}

\begin{example}\label{ex:g11:vect:twopoints}
Lijn door $A(1, 2)$ en $B(4, 3)$: $\vect{AB}\,(3, 1)$, dus $a = 1$,
$b = -3$, en de vergelijking is $x - 3y + c = 0$. Invullen van $A$:
$1 - 6 + c = 0$, dus $c = 5$. Vergelijking: $x - 3y + 5 = 0$. Controle
met $B$: $4 - 9 + 5 = 0$.
\end{example}

\begin{remark}
Wanneer $b \neq 0$ kan de vergelijking naar $y$ opgelost worden:
$y = mx + p$ met richtingscoëfficiënt $m = -\frac ab$. De cartesiaanse
vorm $ax + by + c = 0$ is algemener: ze dekt ook verticale lijnen
($b = 0$), die geen richtingscoëfficiënt hebben.
\end{remark}

\section{Parametrische voorstelling}

\begin{definition}[Parametrische voorstelling]\label{def:g11:vect:parametric}
De lijn door $A(x_A, y_A)$ met richting $\vec u\,(\alpha, \beta)$ is de
verzameling punten
\[
M\bigl(x_A + t\alpha,\ y_A + t\beta\bigr), \qquad t \in \R .
\]
Het getal $t$ is de \emph{parameter}: elke waarde van $t$ produceert één
punt van de lijn, alsof men er met constante snelheid $\vec u$ over
reist.
\end{definition}

\begin{example}
De lijn van \cref{ex:g11:vect:twopoints} is ook
$\{(1 + 3t,\ 2 + t) : t \in \R\}$: $t = 0$ geeft $A$, $t = 1$ geeft $B$.
Elimineren van $t$ ($t = y - 2$, dus $x = 1 + 3(y-2)$) levert
$x - 3y + 5 = 0$ terug.
\end{example}

\section{Onderlinge standen van twee lijnen}

\begin{proposition}[Evenwijdig of snijdend]\label{prop:g11:vect:positions}
Twee lijnen met richtingsvectoren $\vec u$ en $\vec v$ zijn evenwijdig
dan en slechts dan als $\det(\vec u, \vec v) = 0$. In
vergelijkingsvorm: $d : ax + by + c = 0$ en $d' : a'x + b'y + c' = 0$
zijn evenwijdig dan en slechts dan als $ab' - a'b = 0$. Niet-evenwijdige
lijnen snijden elkaar in precies één punt.
\end{proposition}

\begin{proof}
Evenwijdigheid betekent collineaire richtingen, wat
\cref{thm:g11:vect:collinearity} is; met $\vec u\,(-b, a)$ en
$\vec v\,(-b', a')$ is de determinant
$(-b)a' - (-b')a = ab' - a'b$. Als de lijnen niet evenwijdig zijn, leidt
het gelijktijdig oplossen van de twee vergelijkingen (door substitutie of
combinatie) tot een unieke oplossing: het stelsel gedraagt zich als een
$2 \times 2$-stelsel met niet-nul determinant.
\end{proof}

\begin{method}[Snijpunt van twee lijnen]\label{met:g11:vect:intersection}
Los het stelsel van de twee vergelijkingen op: isoleer één onbekende in
één vergelijking en substitueer in de andere, of combineer de
vergelijkingen om één onbekende te elimineren. Controleer altijd het
resulterende punt in \emph{beide} vergelijkingen.
\end{method}

\begin{example}\label{ex:g11:vect:intersection}
Snijd $d: x - 3y + 5 = 0$ en $d': 2x + y - 4 = 0$. Uit $d'$:
$y = 4 - 2x$. Substitueren in $d$:
\[
x - 3(4 - 2x) + 5 = 0
\iff x - 12 + 6x + 5 = 0
\iff 7x = 7
\iff x = 1,
\]
dan $y = 2$. De lijnen snijden in $(1, 2)$. Controle in $d$:
$1 - 6 + 5 = 0$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g11:vect:1}
Gegeven $A(2, -1)$, $B(5, 3)$ en $C(-1, 1)$, bereken de coördinaten van
$\vect{AB}$, $\vect{AC}$, $\vect{AB} + \vect{AC}$ en $2\vect{AB} -
3\vect{AC}$, en het midden van $[BC]$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:vect:2}
Zijn de vectoren collineair?
\[
\vec u\,(4, -6) \text{ en } \vec v\,(-2, 3); \qquad
\vec u\,(2, 5) \text{ en } \vec v\,(3, 7); \qquad
\vec u\,(0, 3) \text{ en } \vec v\,(0, -8).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:vect:3}
Vind een cartesiaanse vergelijking van de lijn door $A(2, 1)$ met
richtingsvector $\vec u\,(3, -1)$, daarna van de lijn door $B(0, 4)$ en
$C(2, 0)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:vect:4}
Geef voor de lijn $d: 3x - 2y + 6 = 0$ een richtingsvector, de
snijpunten met de twee coördinaatassen, en beslis of de punten $P(2, 6)$
en $Q(1, 4)$ tot $d$ behoren.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:vect:5}
Zijn de punten $A(1, 2)$, $B(4, 8)$, $C(-2, -4)$ collineair? Zelfde
vraag voor $D(0, 1)$, $E(2, 4)$, $F(5, 9)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:vect:6}
Bepaal het snijpunt, indien aanwezig:
\[
d: 2x + y - 7 = 0 \ \text{ en }\ d': x - y + 1 = 0;
\qquad
e: x - 2y + 3 = 0 \ \text{ en }\ e': -2x + 4y + 1 = 0 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:vect:7}
Geef een parametrische voorstelling van de lijn $d: 2x - y + 3 = 0$, en
een cartesiaanse vergelijking van de lijn
$\bigl\{(1 + 2t,\ -3 + t) : t \in \R\bigr\}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:vect:8}
Vind de vergelijking van de lijn door $P(1, -2)$ evenwijdig met
$d: 4x - 3y + 2 = 0$, en de waarde van $m$ waarvoor de lijn
$mx + 2y - 5 = 0$ evenwijdig is met $d$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:vect:9}
Laat $A(-1, 0)$, $B(3, 2)$ en $C(1, -4)$. Vind een cartesiaanse
vergelijking van de \emph{zwaartelijn} van de driehoek $ABC$ vanuit $C$
(de lijn die $C$ verbindt met het midden van $[AB]$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:vect:10}
Voor welke waarde(n) van de parameter $m$ zijn de vectoren
$\vec u\,(m, 2)$ en $\vec v\,(3, m - 1)$ collineair?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:vect:11}
Laat $ABCD$ een parallellogram zijn, $I$ het midden van $[AB]$, en $J$
het punt gedefinieerd door $\vect{DJ} = \frac23 \vect{DI}$. Werkend in
het coördinatenstelsel waarin $A(0,0)$, $B(1,0)$, $D(0,1)$ (dus
$C(1,1)$):
\begin{enumerate}
  \item Bereken de coördinaten van $I$ en $J$.
  \item Toon aan dat $A$, $J$ en $C$ collineair zijn. (Een mooi feit: de
        lijn $(DI)$ snijdt de diagonaal $(AC)$ op precies een derde van
        de lengte vanaf $A$.)
\end{enumerate}
\end{exercise}
