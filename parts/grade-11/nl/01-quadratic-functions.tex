\chapter{Kwadratische functies en vergelijkingen}\label{ch:g11:quad}

\omterm{def:g11:quad:quadratic}{Kwadratische functies} zijn de eenvoudigste \omterm{def:g10:functions:function}{functies} na de lineaire, en de
eerste waarvan de \omterm{def:g10:functions:graph}{grafieken} echt gekromd zijn. Dit hoofdstuk ontwikkelt
het volledige gereedschap ervoor: de \omterm{thm:g11:quad:canonical}{canonieke vorm}, de \omterm{def:g11:quad:discriminant}{discriminant}, het
teken van een kwadratische uitdrukking, en de meetkunde van de \omterm{def:g11:quad:parabola}{parabool}.

\section{De canonieke vorm}

\begin{definition}[Kwadratische functie]\label{def:g11:quad:quadratic}
Een \emph{kwadratische functie}\index{kwadratische functie} is een
\omterm{def:g10:functions:function}{functie} van de vorm
\[
f(x) = ax^2 + bx + c, \qquad a \neq 0,
\]
waar $a$, $b$, $c$ \omterm{def:g10:numbers:sets}{reële getallen} zijn. De uitdrukking $ax^2 + bx + c$
heet ook een \emph{kwadratische trinomiaal}.
\end{definition}

\begin{theorem}[Canonieke vorm]\label{thm:g11:quad:canonical}
Elke \omterm{def:g11:quad:quadratic}{kwadratische functie} kan op precies één manier geschreven worden als
\[
f(x) = a(x - \alpha)^2 + \beta,
\qquad\text{waar } \alpha = -\frac{b}{2a} \text{ en } \beta = f(\alpha).
\]
Dit is de \emph{canonieke vorm}\index{canonieke vorm} van $f$.
\end{theorem}

\begin{proof}
We \emph{voltooien het kwadraat}. Omdat $a \neq 0$, factoreren we $a$ uit
de eerste twee termen:
\[
ax^2 + bx + c = a\left(x^2 + \frac{b}{a}x\right) + c .
\]
Nu is $x^2 + \frac{b}{a}x$ het begin van het kwadraat
$\left(x + \frac{b}{2a}\right)^2 = x^2 + \frac ba x + \frac{b^2}{4a^2}$,
dus $x^2 + \frac{b}{a}x = \left(x + \frac{b}{2a}\right)^2 -
\frac{b^2}{4a^2}$. Substitueren:
\[
f(x) = a\left(x + \frac{b}{2a}\right)^2 - \frac{b^2}{4a} + c
= a\bigl(x - \alpha\bigr)^2 + \beta,
\]
met $\alpha = -\frac{b}{2a}$ en $\beta = c - \frac{b^2}{4a}$. Evalueren
in $x = \alpha$ doodt de gekwadrateerde term, dus $\beta = f(\alpha)$.
Uniciteit: \omterm{def:g10:algebra:expand}{ontwikkelen} van $a(x-\alpha)^2 + \beta$ en identificeren van
de coëfficiënten van $x$ forceert $\alpha = -\frac{b}{2a}$, en dan is
$\beta$ bepaald.
\end{proof}

\begin{example}\label{ex:g11:quad:canonical}
Laat $f(x) = 2x^2 - 8x + 3$. Dan $\alpha = -\frac{-8}{2 \times 2} = 2$
en $\beta = f(2) = 8 - 16 + 3 = -5$, dus
\[
f(x) = 2(x - 2)^2 - 5 .
\]
Controle door te \omterm{def:g10:algebra:expand}{ontwikkelen}: $2(x^2 - 4x + 4) - 5 = 2x^2 - 8x + 8 - 5 =
2x^2 - 8x + 3$.
\end{example}

\begin{proposition}[Extremum en variaties]\label{prop:g11:quad:variations}
Laat $f(x) = a(x-\alpha)^2 + \beta$.
\begin{enumerate}
  \item Als $a > 0$, dan is $f$ \omterm{def:g10:functions:variations}{dalend} op $\intoc{-\infty}{\alpha}$ en
        \omterm{def:g10:functions:variations}{stijgend} op $\intco{\alpha}{+\infty}$: $f$ heeft een
        \emph{\omterm{def:g10:functions:extrema}{minimum}} $\beta$, bereikt bij $x = \alpha$.
  \item Als $a < 0$, zijn de variaties omgekeerd: $f$ heeft een
        \emph{\omterm{def:g10:functions:extrema}{maximum}} $\beta$ bij $x = \alpha$.
\end{enumerate}
\end{proposition}

\begin{proof}
Stel $a > 0$ en laat $\alpha \leq u < v$. Dan
\[
f(v) - f(u) = a\bigl((v-\alpha)^2 - (u-\alpha)^2\bigr)
= a\,(v - u)\,\bigl((v-\alpha) + (u-\alpha)\bigr).
\]
Elke factor is positief: $a > 0$, $v - u > 0$, en
$(v-\alpha) + (u-\alpha) > 0$ omdat $v > \alpha$ en $u \geq \alpha$. Dus
$f(v) > f(u)$: $f$ is \omterm{def:g10:functions:variations}{stijgend} op $\intco{\alpha}{+\infty}$. Voor
$u < v \leq \alpha$ is de laatste factor negatief, dus $f(v) < f(u)$: $f$
is daar \omterm{def:g10:functions:variations}{dalend}. Ten slotte $f(x) - f(\alpha) = a(x-\alpha)^2 \geq 0$
voor alle $x$, dus $\beta = f(\alpha)$ is het \omterm{def:g10:functions:extrema}{minimum}. Het geval
$a < 0$ volgt door het bovenstaande toe te passen op $-f$.
\end{proof}

\begin{example}
Een boer heeft $100$ meter hek om een rechthoekig veld af te sluiten tegen
een rechte muur (geen hek nodig langs de muur). Als $x$ de breedte is, is
de ingesloten oppervlakte $A(x) = x(100 - 2x) = -2x^2 + 100x$. Hier
$\alpha = -\frac{100}{2\times(-2)} = 25$ en $A(25) = 25 \times 50 = 1250$:
de oppervlakte is het grootst bij een breedte van $25$ m, wat
$1250$ m$^2$ geeft.
\end{example}

\section{Wortels en ontbinding}

\begin{definition}[Wortel, discriminant]\label{def:g11:quad:discriminant}
Een \emph{wortel}\index{wortel} van de trinomiaal $ax^2+bx+c$ is een
oplossing van de \omterm{def:g10:algebra:equation}{vergelijking} $ax^2 + bx + c = 0$. De
\emph{discriminant}\index{discriminant} van de trinomiaal is het getal
\[
\Delta = b^2 - 4ac .
\]
\end{definition}

\begin{theorem}[Oplossen van $ax^2+bx+c=0$]\label{thm:g11:quad:roots}
\begin{enumerate}
  \item Als $\Delta > 0$, heeft de \omterm{def:g10:algebra:equation}{vergelijking} twee verschillende
        \omterm{def:g11:quad:discriminant}{wortels}
        \[
        x_1 = \frac{-b - \sqrt{\Delta}}{2a},
        \qquad
        x_2 = \frac{-b + \sqrt{\Delta}}{2a},
        \]
        en $ax^2+bx+c = a(x - x_1)(x - x_2)$.
  \item Als $\Delta = 0$, heeft de \omterm{def:g10:algebra:equation}{vergelijking} de enige \omterm{def:g11:quad:discriminant}{wortel}
        $x_0 = -\frac{b}{2a}$, en $ax^2+bx+c = a(x - x_0)^2$.
  \item Als $\Delta < 0$, heeft de \omterm{def:g10:algebra:equation}{vergelijking} geen reële oplossing, en
        de trinomiaal kan niet ontbonden worden met reële coëfficiënten.
\end{enumerate}
\end{theorem}

\begin{proof}
Start vanuit de \omterm{thm:g11:quad:canonical}{canonieke vorm} berekend in \cref{thm:g11:quad:canonical}:
\[
ax^2+bx+c
= a\left(x + \frac{b}{2a}\right)^2 + c - \frac{b^2}{4a}
= a\left[\left(x + \frac{b}{2a}\right)^2 - \frac{\Delta}{4a^2}\right],
\]
omdat $c - \frac{b^2}{4a} = \frac{4ac - b^2}{4a} = -\frac{\Delta}{4a} =
a \times \left(-\frac{\Delta}{4a^2}\right)$.

\emph{1.} Als $\Delta > 0$, dan $\frac{\Delta}{4a^2} =
\left(\frac{\sqrt\Delta}{2a}\right)^2$ en de haak is een verschil van
twee kwadraten:
\[
ax^2+bx+c = a\left(x + \frac{b}{2a} - \frac{\sqrt\Delta}{2a}\right)
\left(x + \frac{b}{2a} + \frac{\sqrt\Delta}{2a}\right)
= a(x - x_2)(x - x_1).
\]
Een product is nul precies wanneer één van de factoren nul is, wat de
twee \omterm{def:g11:quad:discriminant}{wortels} geeft, verschillend omdat $\sqrt\Delta \neq 0$.

\emph{2.} Als $\Delta = 0$, is de haak $\left(x + \frac{b}{2a}\right)^2$,
die alleen verdwijnt bij $x_0 = -\frac{b}{2a}$.

\emph{3.} Als $\Delta < 0$, dan $-\frac{\Delta}{4a^2} > 0$, dus de haak
is een kwadraat plus een positief getal: hij verdwijnt nooit, en de
\omterm{def:g10:algebra:equation}{vergelijking} heeft geen reële oplossing. Een ontbinding
$a(x-x_1)(x-x_2)$ zou \omterm{def:g11:quad:discriminant}{wortels} produceren, dus bestaat die niet.
\end{proof}

\begin{example}\label{ex:g11:quad:solving}
Los $2x^2 - 8x + 3 = 0$ op. Hier $\Delta = (-8)^2 - 4 \times 2 \times 3 =
64 - 24 = 40 > 0$, dus er zijn twee \omterm{def:g11:quad:discriminant}{wortels}:
\[
x_{1,2} = \frac{8 \pm \sqrt{40}}{4} = \frac{8 \pm 2\sqrt{10}}{4}
= 2 \pm \frac{\sqrt{10}}{2}.
\]
Voor $x^2 + x + 1 = 0$: $\Delta = 1 - 4 = -3 < 0$, geen reële oplossing.
\end{example}

\begin{proposition}[Som en product van de wortels]\label{prop:g11:quad:vieta}
Als $\Delta \geq 0$ en $x_1, x_2$ de \omterm{def:g11:quad:discriminant}{wortels} zijn (gelijk wanneer
$\Delta = 0$), dan
\[
x_1 + x_2 = -\frac{b}{a},
\qquad
x_1 x_2 = \frac{c}{a} .
\]
Omgekeerd zijn twee getallen met som $s$ en product $p$ de oplossingen
van $x^2 - sx + p = 0$.
\end{proposition}

\begin{proof}
Ontwikkel de ontbinding van \cref{thm:g11:quad:roots}:
\[
a(x - x_1)(x - x_2) = a x^2 - a(x_1 + x_2)x + a\,x_1 x_2 .
\]
Identificeren van de coëfficiënten met $ax^2 + bx + c$ geeft
$-a(x_1+x_2) = b$ en $a\,x_1x_2 = c$. Voor het omgekeerde: als
$u + v = s$ en $uv = p$, dan $(x-u)(x-v) = x^2 - sx + p$, dus $u$ en $v$
zijn de \omterm{def:g11:quad:discriminant}{wortels}.
\end{proof}

\begin{example}
Vind twee getallen met som $10$ en product $21$. Ze lossen
$x^2 - 10x + 21 = 0$ op; $\Delta = 100 - 84 = 16$, \omterm{def:g11:quad:discriminant}{wortels}
$\frac{10 \pm 4}{2} = 7$ en $3$. \emph{Snelle wortelherkenning:} omdat de
coëfficiënten van $x^2 - 5x + 4$ voldoen aan $1 - 5 + 4 = 0$, is het
getal $1$ een \omterm{def:g11:quad:discriminant}{wortel}, en de andere is $\frac{c}{a} = 4$ (hun product moet
$\frac ca$ zijn).
\end{example}

\section{Teken van een kwadratische}

\begin{theorem}[Teken van een trinomiaal]\label{thm:g11:quad:sign}
Laat $f(x) = ax^2+bx+c$ met $a \neq 0$.
\begin{enumerate}
  \item Als $\Delta > 0$, met \omterm{def:g11:quad:discriminant}{wortels} $x_1 < x_2$: $f(x)$ heeft het teken
        van $a$ buiten $\intcc{x_1}{x_2}$, en het teken van $-a$ op
        $\intoo{x_1}{x_2}$.
  \item Als $\Delta = 0$: $f(x)$ heeft het teken van $a$ voor alle
        $x \neq x_0$, en verdwijnt bij $x_0$.
  \item Als $\Delta < 0$: $f(x)$ heeft het teken van $a$ voor alle $x$.
\end{enumerate}
Kortom: \emph{een trinomiaal heeft het teken van $a$, behalve tussen
haar \omterm{def:g11:quad:discriminant}{wortels}}.
\end{theorem}

\begin{proof}
\emph{1.} Schrijf $f(x) = a(x-x_1)(x-x_2)$ en volg de tekens van de
factoren. Als $x < x_1$: zowel $x - x_1$ als $x - x_2$ zijn negatief, hun
product is positief, dus $f(x)$ heeft het teken van $a$. Als
$x_1 < x < x_2$: de factoren hebben tegengestelde tekens, het product is
negatief, en $f(x)$ heeft het teken van $-a$. Als $x > x_2$: beide
factoren zijn positief en $f(x)$ heeft opnieuw het teken van $a$.

\emph{2.} $f(x) = a(x - x_0)^2$ en $(x-x_0)^2 > 0$ voor $x \neq x_0$.

\emph{3.} Uit het bewijs van \cref{thm:g11:quad:roots},
$f(x) = a\left[\left(x + \frac{b}{2a}\right)^2 +
\left(-\frac{\Delta}{4a^2}\right)\right]$, en de haak is altijd
positief.
\end{proof}

\begin{method}[Kwadratische ongelijkheden oplossen]\label{met:g11:quad:inequalities}
Om $ax^2 + bx + c \leq 0$ (of $\geq$, $<$, $>$) op te lossen:
\begin{enumerate}
  \item bereken $\Delta$ en de \omterm{def:g11:quad:discriminant}{wortels}, indien aanwezig;
  \item teken de \omterm{def:g10:algebra:signtable}{tekentabel} met \cref{thm:g11:quad:sign} (teken van $a$
        buiten de \omterm{def:g11:quad:discriminant}{wortels});
  \item lees de oplossingsinterval(len) af, met aandacht of de \omterm{def:g11:quad:discriminant}{wortels}
        zelf inbegrepen zijn (brede ongelijkheid) of niet (strikte).
\end{enumerate}
\end{method}

\begin{example}\label{ex:g11:quad:inequality}
Los $-x^2 + 3x + 4 \geq 0$ op. Hier $\Delta = 9 + 16 = 25$, en de \omterm{def:g11:quad:discriminant}{wortels}
zijn $\frac{-3 \pm 5}{-2}$, d.w.z.\ $x_1 = -1$ en $x_2 = 4$. Omdat
$a = -1 < 0$, is de trinomiaal negatief buiten de \omterm{def:g11:quad:discriminant}{wortels} en positief
ertussen. Met de brede ongelijkheid is de oplossingsverzameling
$\intcc{-1}{4}$.
\end{example}

\section{De parabool}

\begin{definition}[Parabool]\label{def:g11:quad:parabola}
De \omterm{def:g10:functions:graph}{grafiek} van een \omterm{def:g11:quad:quadratic}{kwadratische functie} $f(x) = a(x-\alpha)^2+\beta$ is
een \emph{parabool}\index{parabool} met \emph{top} $S(\alpha, \beta)$.
Ze opent omhoog als $a > 0$, omlaag als $a < 0$, en is symmetrisch t.o.v.\
de verticale lijn $x = \alpha$.
\end{definition}

\begin{remark}
De symmetrie is eenvoudig te controleren: voor elke $h$ is
$f(\alpha + h) = ah^2 + \beta = f(\alpha - h)$, dus twee punten op
gelijke horizontale afstand van de lijn $x = \alpha$ hebben dezelfde
hoogte.
\end{remark}

\begin{method}[Een parabool lezen]\label{met:g11:quad:reading}
Om $f(x) = ax^2+bx+c$ te schetsen: vind de oriëntatie (teken van $a$), de
top $\left(-\frac{b}{2a},\, f\!\left(-\frac{b}{2a}\right)\right)$, het
snijpunt $(0, c)$ met de $y$-as, en de \omterm{def:g11:quad:discriminant}{wortels}, indien aanwezig. De
symmetrie-as $x = -\frac{b}{2a}$ plaatst elk berekend punt twee keer.
\end{method}

\begin{example}
Voor $f(x) = x^2 - 4x + 3$: omhoog, top $(2, -1)$, snijdt de $y$-as in
$(0,3)$, \omterm{def:g11:quad:discriminant}{wortels} $1$ en $3$ (gespot: $1 - 4 + 3 = 0$). Door symmetrie ligt
het punt $(4, 3)$ ook op de kromme, spiegelbeeld van $(0, 3)$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g11:quad:1}
Los de \omterm{def:g10:algebra:equation}{vergelijkingen} op
\[
x^2 - 5x + 6 = 0, \qquad
2x^2 + 3x - 2 = 0, \qquad
x^2 + 2x + 5 = 0, \qquad
9x^2 - 6x + 1 = 0 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:quad:2}
Schrijf in \omterm{thm:g11:quad:canonical}{canonieke vorm} en geef de top van de \omterm{def:g11:quad:parabola}{parabool}:
\[
f(x) = x^2 - 6x + 11, \qquad
g(x) = -2x^2 + 4x + 1, \qquad
h(x) = 3x^2 + 6x .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:quad:3}
Los de ongelijkheden op
\[
x^2 - 3x - 10 < 0, \qquad
2x^2 + x + 3 > 0, \qquad
-x^2 + 6x - 9 \geq 0 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:quad:4}
Vind zonder $\Delta$ te berekenen één voor de hand liggende \omterm{def:g11:quad:discriminant}{wortel}, daarna
de andere:
\[
x^2 - 7x + 6 = 0, \qquad
2x^2 + 5x - 7 = 0, \qquad
x^2 + 4x - 5 = 0 .
\]
(Hint: test $x = 1$ en $x = -1$, gebruik daarna het product van de
\omterm{def:g11:quad:discriminant}{wortels}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:5}
Vind twee getallen waarvan de som $14$ is en het product $45$. Vind
daarna de afmetingen van een rechthoek met omtrek $34$ en oppervlakte
$60$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:6}
Laat $f(x) = x^2 + (m-2)x + 1$, waar $m$ een reële parameter is. Voor
welke waarden van $m$ heeft de \omterm{def:g10:algebra:equation}{vergelijking} $f(x) = 0$ precies één
oplossing? Geen oplossing?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:7}
Een bal wordt omhoog gegooid; haar hoogte na $t$ seconden is
$h(t) = -5t^2 + 20t + 1$ (in meters). Wat is de maximale bereikte
hoogte, en op welk moment? Gedurende welk tijdsinterval is de bal
minstens $16$ m hoog?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:8}
Los de \omterm{def:g10:algebra:equation}{vergelijking} $x^4 - 13x^2 + 36 = 0$ op door $X = x^2$ te stellen
(een \emph{biquadratische} \omterm{def:g10:algebra:equation}{vergelijking}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:9}
Voor welke waarden van $x$ ligt het punt $M(x, x^2)$ van de \omterm{def:g11:quad:parabola}{parabool}
$y = x^2$ strikt onder de lijn $y = x + 2$? Interpreteer op een schets.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:10}
De som van een positief getal en zijn omgekeerde is $\frac{13}{6}$. Vind
het getal. (Herleid tot een kwadratische \omterm{def:g10:algebra:equation}{vergelijking}.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:quad:11}
Laat $\mathcal P$ de \omterm{def:g11:quad:parabola}{parabool} $y = x^2$ zijn en laat $d_m$ de lijn
$y = mx - 1$ zijn, waar $m$ een reële parameter is.
\begin{enumerate}
  \item Voor welke waarden van $m$ snijden $\mathcal P$ en $d_m$ elkaar
        in twee punten? Eén punt? Geen punt?
  \item Wanneer ze in precies één punt raken, bereken het raakpunt. (In
        \cref{ch:g11:deriv} zul je $d_m$ herkennen als de raaklijn aan
