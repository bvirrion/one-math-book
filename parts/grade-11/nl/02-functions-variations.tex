\chapter{Functies en variaties}\label{ch:g11:func}

Dit hoofdstuk ordent wat men over \omterm{def:g11:func:function}{functies} moet weten vóór de analyse: de
kleine familie van \emph{\omterm{def:g11:func:reference}{referentiefuncties}} waarvan de \omterm{def:g10:functions:graph}{grafieken} uit het
hoofd bekend moeten zijn, de precieze betekenis van \emph{\omterm{def:g11:func:monotone}{stijgend}} en
\emph{\omterm{def:g10:functions:variations}{dalend}}, en hoe variaties zich gedragen wanneer \omterm{def:g11:func:function}{functies} gecombineerd
worden.

\section{Functies en krommen}

\begin{definition}[Functie, domein, kromme]\label{def:g11:func:function}
Een \emph{functie}\index{functie} $f$ koppelt aan elk getal $x$ van een
verzameling $D \subseteq \R$ (haar \emph{domein}\index{domein}) precies
één getal $f(x)$. Haar \emph{kromme} (of \omterm{def:g10:functions:graph}{grafiek}) is de verzameling van
punten $\bigl(x, f(x)\bigr)$ voor $x \in D$.
\end{definition}

\begin{example}
De formule $f(x) = \frac{\sqrt{x+2}}{x-1}$ eist $x + 2 \geq 0$ (voor de
vierkantswortel) en $x \neq 1$ (niet-nul noemer): het \omterm{def:g11:func:function}{domein} is
$\intco{-2}{1} \cup \intoo{1}{+\infty}$.
\end{example}

\section{De referentiefuncties}

De volgende vijf \omterm{def:g11:func:function}{functies} en hun krommen moeten reflexen worden.

\begin{definition}[Referentiefuncties]\label{def:g11:func:reference}
\index{referentiefuncties}
\[
x \mapsto x^2 \text{ (kwadraat)}, \quad
x \mapsto x^3 \text{ (derdemacht)}, \quad
x \mapsto \sqrt{x}, \quad
x \mapsto \frac{1}{x}, \quad
x \mapsto \abs{x}.
\]
Hun \omterm{def:g11:func:function}{domeinen} zijn $\R$, $\R$, $\intco{0}{+\infty}$, $\R \setminus \{0\}$
en $\R$ respectievelijk. De kromme van $x \mapsto \frac1x$ is een
\emph{hyperbool}\index{hyperbool}.
\end{definition}

\section{Variaties op een interval}

\begin{definition}[Stijgend, dalend]\label{def:g11:func:monotone}
Laat $f$ gedefinieerd zijn op een \omterm{def:g10:numbers:interval}{interval} $I$. De \omterm{def:g11:func:function}{functie} $f$ is
\emph{stijgend}\index{stijgende functie} op $I$ als ze de volgorde
behoudt:
\[
\text{voor alle } u, v \in I: \quad u < v \implies f(u) \leq f(v),
\]
en \emph{\omterm{def:g10:functions:variations}{dalend}} als ze de volgorde omkeert ($u < v \implies f(u) \geq
f(v)$). Met strikte ongelijkheden rechts is $f$ \emph{strikt} stijgend
(resp.\ \omterm{def:g10:functions:variations}{dalend}). Een \omterm{def:g11:func:function}{functie} die stijgend of \omterm{def:g10:functions:variations}{dalend} is op $I$ is
\emph{monotoon} op $I$.
\end{definition}

\begin{method}[Een variatie bewijzen vanuit de definitie]\label{met:g11:func:variation}
Neem twee willekeurige getallen $u < v$ in het \omterm{def:g10:numbers:interval}{interval}, en bepaal het
teken van het verschil $f(v) - f(u)$, meestal door het te
\emph{\omterm{def:g10:algebra:expand}{ontbinden}}. Als het teken constant is op het \omterm{def:g10:numbers:interval}{interval}, concludeer.
\end{method}

\begin{proposition}[Variaties van de referentiefuncties]\label{prop:g11:func:refvariations}
\begin{enumerate}
  \item $x^2$ is strikt \omterm{def:g10:functions:variations}{dalend} op $\intoc{-\infty}{0}$ en strikt \omterm{def:g11:func:monotone}{stijgend}
        op $\intco{0}{+\infty}$.
  \item $x^3$ is strikt \omterm{def:g11:func:monotone}{stijgend} op $\R$.
  \item $\sqrt x$ is strikt \omterm{def:g11:func:monotone}{stijgend} op $\intco{0}{+\infty}$.
  \item $\frac1x$ is strikt \omterm{def:g10:functions:variations}{dalend} op $\intoo{-\infty}{0}$ en op
        $\intoo{0}{+\infty}$ (maar \emph{niet} op hun \omterm{def:g10:numbers:interunion}{vereniging}).
  \item $\abs x$ is strikt \omterm{def:g10:functions:variations}{dalend} op $\intoc{-\infty}{0}$ en strikt
        \omterm{def:g11:func:monotone}{stijgend} op $\intco{0}{+\infty}$.
\end{enumerate}
\end{proposition}

\begin{proof}
Laat $u < v$. We \omterm{def:g10:algebra:expand}{ontbinden} $f(v) - f(u)$ in elk geval.

\emph{1.} $v^2 - u^2 = (v-u)(v+u)$. Op $\intco{0}{+\infty}$: $v - u > 0$
en $v + u > 0$ (omdat $v > u \geq 0$), dus $v^2 > u^2$. Op
$\intoc{-\infty}{0}$: $v + u < 0$, dus $v^2 < u^2$.

\emph{2.} $v^3 - u^3 = (v - u)(v^2 + uv + u^2)$, en de tweede factor
is gelijk aan $\left(v + \frac u2\right)^2 + \frac{3u^2}{4} > 0$ tenzij
$u = v = 0$. Dus $v^3 > u^3$ wanneer $v > u$.

\emph{3.} Voor $0 \leq u < v$:
$\sqrt v - \sqrt u = \frac{(\sqrt v - \sqrt u)(\sqrt v + \sqrt u)}
{\sqrt v + \sqrt u} = \frac{v - u}{\sqrt v + \sqrt u} > 0$.

\emph{4.} Voor $0 < u < v$:
$\frac1v - \frac1u = \frac{u - v}{uv} < 0$ omdat $u - v < 0$ en $uv > 0$;
zelfde berekening voor $u < v < 0$. Op de \omterm{def:g10:numbers:interunion}{vereniging} faalt de claim:
$-1 < 1$ maar $\frac{1}{-1} = -1 < 1 = \frac11$.

\emph{5.} Op $\intco{0}{+\infty}$ is $\abs x = x$ strikt \omterm{def:g11:func:monotone}{stijgend}; op
$\intoc{-\infty}{0}$ is $\abs x = -x$ strikt \omterm{def:g10:functions:variations}{dalend}.
\end{proof}

\begin{remark}
Punt 4 is een klassieke valkuil: ``\omterm{def:g10:functions:variations}{dalend} op $\intoo{-\infty}{0}$ en op
$\intoo{0}{+\infty}$'' is \emph{niet} hetzelfde als ``\omterm{def:g10:functions:variations}{dalend} op
$\R \setminus \{0\}$''. Variaties hebben alleen zin op intervallen.
\end{remark}

\begin{example}[Vergelijken zonder te rekenen]
Omdat $\sqrt x$ strikt \omterm{def:g11:func:monotone}{stijgend} is, is $\sqrt{7} < \sqrt{8}$; omdat
$x^2$ strikt \omterm{def:g10:functions:variations}{dalend} is op $\intoc{-\infty}{0}$, is
$(-3)^2 > (-2)^2$. Monotonie zet vergelijkingen van beelden om in
vergelijkingen van argumenten.
\end{example}

\section{Nieuwe functies uit oude}

\begin{proposition}[Transformaties]\label{prop:g11:func:transforms}
Laat $u$ een \omterm{def:g11:func:function}{functie} zijn gedefinieerd op een \omterm{def:g10:numbers:interval}{interval} $I$ en $k \in \R$.
\begin{enumerate}
  \item $u + k$ heeft dezelfde variaties als $u$ op $I$; haar kromme is
        die van $u$ verticaal verschoven met $k$.
  \item Voor $\lambda > 0$ heeft $\lambda u$ dezelfde variaties als $u$;
        voor $\lambda < 0$ zijn de variaties omgekeerd.
  \item Als $u \geq 0$ op $I$, dan heeft $\sqrt u$ dezelfde variaties als
        $u$.
  \item Als $u \neq 0$ en $u$ constant teken heeft op $I$, dan heeft
        $\frac1u$ de \emph{tegengestelde} variaties van $u$.
\end{enumerate}
\end{proposition}

\begin{proof}
Laat $u < v$ in $I$ (we misbruiken notatie en vergelijken $x$-waarden
$u < v$; schrijf $s = u(x_1)$, $t = u(x_2)$ voor de \omterm{def:g10:functions:function}{beelden} van
$x_1 < x_2$).

\emph{1.} $(u+k)(x_2) - (u+k)(x_1) = t - s$: het verschil van \omterm{def:g10:functions:function}{beelden}
blijft onveranderd, dus het teken ook.

\emph{2.} $\lambda t - \lambda s = \lambda(t - s)$: het teken blijft als
$\lambda > 0$, keert om als $\lambda < 0$.

\emph{3.} Als $u$ \omterm{def:g11:func:monotone}{stijgend} is en $x_1 < x_2$, dan $0 \leq s \leq t$, en
omdat $\sqrt{}$ \omterm{def:g11:func:monotone}{stijgend} is (\cref{prop:g11:func:refvariations}) volgt
$\sqrt s \leq \sqrt t$. Hetzelfde argument in het dalende geval.

\emph{4.} Als $u$ \omterm{def:g11:func:monotone}{stijgend} is, $0 < s \leq t$ (of $s \leq t < 0$), dan
$\frac1s \geq \frac1t$ omdat $x \mapsto \frac1x$ \omterm{def:g10:functions:variations}{dalend} is aan weerszijden
van $0$: de volgorde is omgekeerd.
\end{proof}

\begin{example}\label{ex:g11:func:composite}
Bestudeer $f(x) = \sqrt{x^2 + 1}$ op $\R$. De binnelfunctie
$u(x) = x^2 + 1$ is positief, \omterm{def:g10:functions:variations}{dalend} op $\intoc{-\infty}{0}$ en \omterm{def:g11:func:monotone}{stijgend}
op $\intco{0}{+\infty}$ (een verschoven kwadraat). Volgens punt 3 heeft
$f$ dezelfde variaties: \omterm{def:g10:functions:variations}{dalend} dan \omterm{def:g11:func:monotone}{stijgend}, met \omterm{def:g10:functions:extrema}{minimum} $f(0) = 1$.
\end{example}

\section{Even en oneven functies}

\begin{definition}[Pariteit]\label{def:g11:func:parity}
Laat $f$ gedefinieerd zijn op een \omterm{def:g11:func:function}{domein} $D$ symmetrisch t.o.v.\ $0$
(d.w.z.\ $-x \in D$ wanneer $x \in D$). De \omterm{def:g11:func:function}{functie} $f$ is
\emph{even}\index{even functie} als $f(-x) = f(x)$ voor alle $x \in D$,
en \emph{oneven}\index{oneven functie} als $f(-x) = -f(x)$ voor alle
$x \in D$.
\end{definition}

\begin{proposition}[Meetkundige betekenis]\label{prop:g11:func:paritysym}
De kromme van een \omterm{def:g11:func:parity}{even functie} is symmetrisch t.o.v.\ de $y$-as; de
kromme van een \omterm{def:g11:func:parity}{oneven functie} is symmetrisch t.o.v.\ de oorsprong.
\end{proposition}

\begin{proof}
Als $f$ \omterm{def:g11:func:parity}{even} is en $(x, y)$ op de kromme ligt (dus $y = f(x)$), dan
$f(-x) = f(x) = y$: het spiegelpunt $(-x, y)$ ligt ook op de kromme. Als
$f$ \omterm{def:g11:func:parity}{oneven} is, $f(-x) = -y$: het punt $(-x, -y)$, symmetrisch van $(x,y)$
t.o.v.\ de oorsprong, ligt op de kromme.
\end{proof}

\begin{example}
$x^2$ en $\abs x$ zijn \omterm{def:g11:func:parity}{even}; $x^3$ en $\frac1x$ zijn \omterm{def:g11:func:parity}{oneven};
$f(x) = x^2 + x$ is geen van beide: $f(-1) = 0$ terwijl $f(1) = 2$, dus
$f(-1) \neq \pm f(1)$. Pariteit halveert het werk: een \omterm{def:g11:func:parity}{even} of \omterm{def:g11:func:parity}{oneven
functie} hoeft alleen bestudeerd te worden op $\intco{0}{+\infty}$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g11:func:1}
Geef het \omterm{def:g11:func:function}{domein} van elke \omterm{def:g11:func:function}{functie}:
\[
f(x) = \sqrt{3 - x}, \qquad
g(x) = \frac{1}{x^2 - 4}, \qquad
h(x) = \frac{\sqrt{x}}{x - 5} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:func:2}
Vergelijk met de variaties van de \omterm{def:g11:func:reference}{referentiefuncties} zonder rekenmachine:
\[
\sqrt{11} \text{ en } \sqrt{13}; \qquad
(-2.1)^2 \text{ en } (-2.05)^2; \qquad
\frac{1}{0.9} \text{ en } \frac{1}{1.1}; \qquad
(-1.01)^3 \text{ en } (-0.99)^3 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:func:3}
Bepaal of elke \omterm{def:g11:func:function}{functie} \omterm{def:g11:func:parity}{even}, \omterm{def:g11:func:parity}{oneven}, of geen van beide is:
\[
f(x) = 3x^4 - x^2, \qquad
g(x) = x^3 + x, \qquad
h(x) = x^2 + x^3, \qquad
k(x) = \frac{x}{x^2+1} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:func:4}
Een \omterm{def:g11:func:function}{functie} $f$ heeft het volgende variatiegedrag: \omterm{def:g11:func:monotone}{stijgend} op
$\intcc{-3}{1}$ van $f(-3) = -2$ tot $f(1) = 4$, daarna \omterm{def:g10:functions:variations}{dalend} op
$\intcc{1}{5}$ tot $f(5) = 0$. Hoeveel oplossingen heeft de \omterm{def:g10:algebra:equation}{vergelijking}
$f(x) = 1$ op $\intcc{-3}{5}$? En $f(x) = 5$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:func:5}
Bewijs met \cref{met:g11:func:variation} (teken van $f(v) - f(u)$) dat
$f(x) = x^2 - 4x$ strikt \omterm{def:g11:func:monotone}{stijgend} is op $\intco{2}{+\infty}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:func:6}
Geef zonder enige afgeleide te berekenen de variaties van
\[
f(x) = \sqrt{5 - x} \ \text{ op } \intoc{-\infty}{5},
\qquad
g(x) = \frac{1}{x^2 + 1} \ \text{ op } \intco{0}{+\infty},
\qquad
h(x) = 3 - 2\sqrt{x} \ \text{ op } \intco{0}{+\infty}.
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:func:7}
Laat $f(x) = (x-2)^2 - 3$ op $\R$.
\begin{enumerate}
  \item Geef de variaties van $f$ en haar \omterm{def:g10:functions:extrema}{minimum}.
  \item Leid de variaties van $g(x) = \frac{1}{(x-2)^2+1}$ en haar
        \omterm{def:g10:functions:extrema}{maximum} af. (Merk op $g = \frac{1}{f + 4}$.)
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:func:8}
De kromme van een \omterm{def:g11:func:function}{functie} $f$ gaat door de punten $(0, 1)$, $(2, 3)$ en
$(4, 1)$, en $f$ is \omterm{def:g11:func:monotone}{stijgend} op $\intcc{0}{2}$ en \omterm{def:g10:functions:variations}{dalend} op
$\intcc{2}{4}$. Schets een mogelijke kromme, daarna de krommen van
$f + 2$, van $-f$, en van $2f$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:func:9}
Toon aan dat de \omterm{def:g11:func:function}{functie} $f(x) = \frac{2x+1}{x+1}$ geschreven kan worden
als $f(x) = 2 - \frac{1}{x+1}$, en leid haar variaties af op
$\intoo{-1}{+\infty}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:func:10}
Waar of onwaar (rechtvaardig of geef een tegenvoorbeeld):
\begin{enumerate}
  \item De som van twee \omterm{def:g11:func:monotone}{stijgende functies} op $I$ is \omterm{def:g11:func:monotone}{stijgend} op $I$.
  \item Het product van twee \omterm{def:g11:func:monotone}{stijgende functies} op $I$ is \omterm{def:g11:func:monotone}{stijgend} op
        $I$.
  \item Als $f$ \omterm{def:g11:func:monotone}{stijgend} is op $I$, dan is $f^2$ ($= f \times f$)
        \omterm{def:g11:func:monotone}{stijgend} op $I$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:func:11}
Laat $f(x) = x + \frac{1}{x}$ voor $x > 0$.
\begin{enumerate}
  \item Toon aan dat voor $0 < u < v$,
        $f(v) - f(u) = (v - u)\,\frac{uv - 1}{uv}$.
  \item Leid af dat $f$ strikt \omterm{def:g10:functions:variations}{dalend} is op $\intoc{0}{1}$ en strikt
        \omterm{def:g11:func:monotone}{stijgend} op $\intco{1}{+\infty}$.
  \item Concludeer dat $x + \frac1x \geq 2$ voor alle $x > 0$, met
        gelijkheid alleen bij $x = 1$.
\end{enumerate}
\end{exercise}
