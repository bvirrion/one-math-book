\chapter{Het inwendig product in het vlak}\label{ch:g11:scal}

Het inwendig product (ook \emph{scalair product} genoemd) koppelt een
\emph{getal} aan een paar vectoren, en dat getal ziet meetkunde: het
verdwijnt precies bij loodrechte richtingen, en het meet hoeken en
lengtes. Het is de motor achter de cosinusregel, vergelijkingen van
cirkels, en loodrechtheidsberekeningen. Hetzelfde instrument, in de
ruimte, is het onderwerp van \cref{ch:g12:space}.

\section{Definitie en eerste berekeningen}

\begin{definition}[Inwendig product]\label{def:g11:scal:dot}
Het \emph{inwendig product}\index{inwendig product}\index{scalair
product} van twee vectoren $\vec u$ en $\vec v$ is het getal
\[
\vec u \cdot \vec v
= \tfrac12\left(\norm{\vec u + \vec v}^2 - \norm{\vec u}^2
- \norm{\vec v}^2\right),
\]
waar $\norm{\vec u}$ de lengte (\emph{norm}) van $\vec u$ aanduidt.
\end{definition}

\begin{theorem}[De vier gezichten van het inwendig product]\label{thm:g11:scal:faces}
Laat $\vec u\,(x, y)$ en $\vec v\,(x', y')$ niet-nul vectoren zijn, en
$\theta$ de hoek ertussen. Dan
\begin{enumerate}
  \item \emph{(coördinaten)} $\vec u \cdot \vec v = xx' + yy'$;
  \item \emph{(cosinus)} $\vec u \cdot \vec v =
        \norm{\vec u}\,\norm{\vec v}\cos\theta$;
  \item \emph{(projectie)} $\vec u \cdot \vec v = \vec u \cdot \vec v'$,
        waar $\vec v'$ de orthogonale projectie is van $\vec v$ op de
        richting van $\vec u$; als $\vec u = \vect{OA}$ en
        $\vec v = \vect{OB}$, dan
        $\vec u \cdot \vec v = \overline{OA} \times \overline{OH}$, een
        product van gerichte lengtes, waar $H$ de voet is van de
        loodlijn vanuit $B$ op de lijn $(OA)$;
  \item \emph{(normen)} de definiërende formule van
        \cref{def:g11:scal:dot}.
\end{enumerate}
\end{theorem}

\begin{proof}[Bewijs van 1 en 2]
\emph{1.} In coördinaten is $\norm{\vec u}^2 = x^2 + y^2$ (Pythagoras)
en $\vec u + \vec v$ heeft coördinaten $(x + x', y + y')$, dus
\begin{align*}
2\,\vec u \cdot \vec v
&= (x + x')^2 + (y + y')^2 - (x^2 + y^2) - (x'^2 + y'^2)\\
&= 2xx' + 2yy',
\end{align*}
na ontwikkelen van de kwadraten en wegstrepen.

\emph{2.} Kies het coördinatenstelsel zodat $\vec u$ langs de positieve
$x$-as wijst: $\vec u\,(\norm{\vec u}, 0)$ en
$\vec v\,(\norm{\vec v}\cos\theta,\ \norm{\vec v}\sin\theta)$
(\cref{ch:g11:trigo}). Formule 1 geeft dan
$\vec u \cdot \vec v = \norm{\vec u}\,\norm{\vec v}\cos\theta + 0$.
Formule 3 leest dezelfde berekening af: de projectie van $\vec v$ op de
$x$-as heeft gerichte lengte $\norm{\vec v}\cos\theta =
\overline{OH}$.
\end{proof}

\begin{example}\label{ex:g11:scal:compute}
$\vec u\,(3, 1)$ en $\vec v\,(2, -4)$:
$\vec u \cdot \vec v = 6 - 4 = 2$. De normen zijn
$\norm{\vec u} = \sqrt{10}$ en $\norm{\vec v} = \sqrt{20}$, dus
$\cos\theta = \frac{2}{\sqrt{10}\sqrt{20}} = \frac{2}{10\sqrt2}
= \frac{\sqrt2}{10}$: de hoek is iets minder dan $82^\circ$.
\end{example}

\begin{proposition}[Algebraïsche regels]\label{prop:g11:scal:rules}
Voor alle vectoren en alle $\lambda \in \R$:
\[
\vec u \cdot \vec v = \vec v \cdot \vec u, \qquad
\vec u \cdot (\vec v + \vec w) = \vec u \cdot \vec v + \vec u \cdot
\vec w, \qquad
(\lambda \vec u)\cdot \vec v = \lambda\,(\vec u \cdot \vec v),
\]
\[
\vec u \cdot \vec u = \norm{\vec u}^2,
\qquad
\norm{\vec u + \vec v}^2 = \norm{\vec u}^2 + 2\,\vec u\cdot\vec v +
\norm{\vec v}^2 .
\]
\end{proposition}

\begin{proof}
Alles volgt uit de coördinatenformule: bijvoorbeeld
$\vec u \cdot (\vec v + \vec w) = x(x' + x'') + y(y' + y'') =
(xx' + yy') + (xx'' + yy'')$. De laatste identiteit ontwikkelt
$\norm{\vec u + \vec v}^2 = (\vec u + \vec v)\cdot(\vec u + \vec v)$
met de eerste drie regels --- het inwendig product gedraagt zich precies
als een gewoon product, dus de gebruikelijke algebraïsche identiteiten
gelden voor vectoren.
\end{proof}

\section{Orthogonaliteit}

\begin{definition}[Orthogonale vectoren]\label{def:g11:scal:orthogonal}
Twee vectoren zijn \emph{orthogonaal}\index{orthogonale vectoren},
genoteerd $\vec u \perp \vec v$, als $\vec u \cdot \vec v = 0$.
\end{definition}

\begin{remark}
Voor niet-nul vectoren toont formule 2 van \cref{thm:g11:scal:faces}
$\vec u \cdot \vec v = 0 \iff \cos\theta = 0 \iff$ de richtingen
loodrecht zijn. De nulvector is orthogonaal op alles.
\end{remark}

\begin{example}
$\vec u\,(3, 2)$ en $\vec v\,(-4, 6)$: $\vec u \cdot \vec v = -12 + 12 =
0$: orthogonaal. In het algemeen is $\vec u\,(a, b)$ altijd orthogonaal
op $\vec v\,(-b, a)$ --- een kwartslagrotatie.
\end{example}

\section{De cosinusregel}

\begin{theorem}[Cosinusregel]\label{thm:g11:scal:cosines}
\index{cosinusregel}
In een driehoek $ABC$ met zijden $a = BC$, $b = CA$, $c = AB$ en hoek
$\widehat A$ in het hoekpunt $A$:
\[
a^2 = b^2 + c^2 - 2bc\,\cos\widehat A .
\]
\end{theorem}

\begin{proof}
Schrijf $\vect{BC} = \vect{BA} + \vect{AC} = \vect{AC} - \vect{AB}$ en
ontwikkel met \cref{prop:g11:scal:rules}:
\[
a^2 = \norm{\vect{BC}}^2
= \norm{\vect{AC}}^2 - 2\,\vect{AC}\cdot\vect{AB} + \norm{\vect{AB}}^2
= b^2 + c^2 - 2\,\vect{AB}\cdot\vect{AC}.
\]
Volgens de cosinusformule is
$\vect{AB}\cdot\vect{AC} = c\,b\,\cos\widehat A$.
\end{proof}

\begin{remark}
Wanneer $\widehat A$ een rechte hoek is, is $\cos\widehat A = 0$ en
reduceert de cosinusregel tot de stelling van Pythagoras
$a^2 = b^2 + c^2$: het is Pythagoras met een correctieterm voor
niet-rechte hoeken.
\end{remark}

\begin{example}\label{ex:g11:scal:cosines}
Twee zijden van een driehoek meten $b = 5$ en $c = 8$, met ingesloten
hoek $\widehat A = \frac\pi3$. Dan
\[
a^2 = 25 + 64 - 2 \times 5 \times 8 \times \frac12 = 89 - 40 = 49,
\]
dus $a = 7$.
\end{example}

\section{Toepassingen: cirkels en loodrechte lijnen}

\begin{proposition}[Cirkel met gegeven diameter]\label{prop:g11:scal:diameter}
Een punt $M$ ligt op de cirkel met diameter $[AB]$ dan en slechts dan als
\[
\vect{MA} \cdot \vect{MB} = 0 .
\]
\end{proposition}

\begin{proof}
Als $M \neq A, B$, zegt de voorwaarde dat de hoek in $M$ in de driehoek
$AMB$ recht is, en de punten die een lijnstuk onder een rechte hoek zien
zijn precies die van de cirkel met die diameter. Hier een direct bewijs:
laat $O$ het midden van $[AB]$ zijn en $r = \frac{AB}{2}$. Dan
$\vect{MA} = \vect{MO} + \vect{OA}$ en
$\vect{MB} = \vect{MO} + \vect{OB} = \vect{MO} - \vect{OA}$, dus
\[
\vect{MA}\cdot\vect{MB}
= \norm{\vect{MO}}^2 - \norm{\vect{OA}}^2 = MO^2 - r^2,
\]
wat precies verdwijnt wanneer $MO = r$, d.w.z.\ wanneer $M$ op de
cirkel met middelpunt $O$ en straal $r$ ligt.
\end{proof}

\begin{proposition}[Normaalvector van een lijn]\label{prop:g11:scal:normal}
De vector $\vec n\,(a, b)$ is orthogonaal op elke richtingsvector van de
lijn $d : ax + by + c = 0$; men zegt dat $\vec n$ een
\emph{normaalvector}\index{normaalvector} van $d$ is. Twee lijnen zijn
loodrecht precies wanneer hun normaal- (of richtings)vectoren
orthogonaal zijn.
\end{proposition}

\begin{proof}
Een richtingsvector van $d$ is $\vec u\,(-b, a)$
(\cref{thm:g11:vect:cartesian}), en
$\vec n \cdot \vec u = a(-b) + b(a) = 0$.
\end{proof}

\begin{method}[Normaalvectoren gebruiken]\label{met:g11:scal:normal}
\begin{itemize}
  \item Lijn door $A$ met normaal $\vec n\,(a,b)$: schrijf
        $\vect{AM} \cdot \vec n = 0$, wat uitwerkt tot $ax + by + c = 0$
        met $c$ bepaald door $A$.
  \item Loodrechtheid van $d: ax+by+c = 0$ en $d': a'x+b'y+c' = 0$:
        controleer $aa' + bb' = 0$.
\end{itemize}
\end{method}

\begin{example}\label{ex:g11:scal:perpline}
De lijn door $P(2, -1)$ loodrecht op $d : 3x - y + 4 = 0$: een
richtingsvector van $d$ is $(1, 3)$, die dient als \emph{normaal}vector
van de loodlijn. Vergelijking: $x + 3y + c = 0$ met
$2 - 3 + c = 0$, dus $c = 1$: de lijn is $x + 3y + 1 = 0$.
\end{example}

\begin{example}[Vergelijking van een cirkel]\label{ex:g11:scal:circle}
De cirkel met middelpunt $\Omega(1, 2)$ en straal $3$ is de verzameling
punten $M(x,y)$ met $\Omega M^2 = 9$:
\[
(x - 1)^2 + (y - 2)^2 = 9 .
\]
Omgekeerd herschrijft $x^2 + y^2 - 2x - 4y - 4 = 0$, door de kwadraten
te voltooien (\cref{ch:g11:quad}), tot $(x-1)^2 + (y-2)^2 = 9$:
dezelfde cirkel.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g11:scal:1}
Bereken $\vec u \cdot \vec v$ voor
\[
\vec u\,(2, 5),\ \vec v\,(3, -1); \qquad
\vec u\,(1, -4),\ \vec v\,(8, 2); \qquad
\norm{\vec u} = 3,\ \norm{\vec v} = 4,\ \theta = \frac{\pi}{3} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:scal:2}
$ABC$ is een gelijkzijdige driehoek met zijde $6$. Bereken
$\vect{AB} \cdot \vect{AC}$ en $\vect{AB} \cdot \vect{BC}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:scal:3}
Voor welke waarde van $t$ zijn $\vec u\,(3, t)$ en $\vec v\,(t - 8, 1)$
orthogonaal? En voor welke waarde(n) van $t$ is $\vec u\,(t, 2)$
orthogonaal op zichzelf minus $\vec v\,(4, 2)$, d.w.z.\
$\vec u \perp (\vec u - \vec v)$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:scal:4}
Bereken de hoek $\theta$ tussen $\vec u\,(1, 2)$ en $\vec v\,(3, 1)$
(geef $\cos\theta$ exact, daarna $\theta$ op de dichtstbijzijnde graad).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:5}
Een driehoek heeft zijden $b = 4$, $c = 6$ en hoek
$\widehat A = \frac{2\pi}{3}$ ertussen. Bereken de derde zijde $a$.
Bereken daarna, in een driehoek met zijden $a = 7$, $b = 5$, $c = 4$,
$\cos\widehat A$ en beslis of $\widehat A$ scherp of stomp is.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:6}
Laat $A(1, 1)$, $B(5, -1)$ en $C(3, 5)$. Bereken
$\vect{AB} \cdot \vect{AC}$; is de hoek $\widehat A$ recht? Bereken de
drie zijdelengtes en controleer je antwoord met de cosinusregel.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:7}
Geef een vergelijking van de cirkel met middelpunt $\Omega(-2, 3)$ en
straal $5$. Bepaal daarna het middelpunt en de straal van de cirkel
\[
x^2 + y^2 - 6x + 2y - 6 = 0 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:8}
Laat $A(-3, 0)$ en $B(0, 2)$. Vind met \cref{prop:g11:scal:diameter} een
vergelijking van de cirkel met diameter $[AB]$, en controleer dat de
oorsprong $O(0,0)$ erop ligt.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:9}
Vind een vergelijking van de lijn door $P(1, 3)$ loodrecht op
$d : 2x + y - 7 = 0$, en de coördinaten van de voet $H$ van de loodlijn
(het snijpunt van de twee lijnen). Leid de afstand van $P$ tot $d$ af.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:scal:10}
Bewijs met $\norm{\vec u + \vec v}^2 = \norm{\vec u}^2 + 2\vec u\cdot\vec v +
\norm{\vec v}^2$ en haar analogon voor $\vec u - \vec v$ de
\emph{parallellogramidentiteit}:
\[
\norm{\vec u + \vec v}^2 + \norm{\vec u - \vec v}^2
= 2\norm{\vec u}^2 + 2\norm{\vec v}^2 ,
\]
en interpreteer die in een parallellogram (diagonalen versus zijden).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:scal:11}
Laat $A$ en $B$ twee punten zijn met $AB = 4$, en $I$ het midden van
$[AB]$.
\begin{enumerate}
  \item Toon aan dat voor elk punt $M$:
        $\vect{MA}\cdot\vect{MB} = MI^2 - 4$.
        (Voeg $I$ in: $\vect{MA} = \vect{MI} + \vect{IA}$.)
  \item Leid de verzameling punten $M$ af zodat
        $\vect{MA}\cdot\vect{MB} = 12$.
\end{enumerate}
\end{exercise}
