\chapter{Trigonometrie: de eenheidscirkel}\label{ch:g11:trigo}

Trigonometrie begint in rechthoekige driehoeken, maar haar echte thuis is
de \emph{eenheidscirkel}, waar \omterm{def:g11:trigo:cossin}{cosinus} en \omterm{def:g11:trigo:cossin}{sinus} \omterm{def:g11:func:function}{functies} worden van een
willekeurig \omterm{def:g10:numbers:sets}{reëel getal}. Dit hoofdstuk installeert de \omterm{def:g11:trigo:radian}{radiaal}, het
opwinden van de reële lijn op de cirkel, de exacte waarden om uit het
hoofd te kennen, en de symmetrieën die alle klassieke identiteiten
voortbrengen. De studie van $\cos$ en $\sin$ als \omterm{def:g11:func:function}{functies} --- variaties,
\omterm{def:g11:deriv:derivative}{afgeleiden} --- gebeurt in \cref{ch:g12:trigo}.

\section{Radialen en het opwinden}

\begin{definition}[Radiaal]\label{def:g11:trigo:radian}
Laat $\mathcal C$ de cirkel van straal $1$ met middelpunt de oorsprong
$O$ zijn, en $I(1, 0)$. De \emph{radiaal}\index{radiaal}-maat van een
hoek in het middelpunt van $\mathcal C$ is de lengte van de boog die ze
op $\mathcal C$ snijdt. Omdat de volledige cirkel omtrek $2\pi$ heeft:
\[
360^\circ = 2\pi \text{ rad},
\qquad
180^\circ = \pi \text{ rad},
\qquad
1 \text{ rad} = \frac{180^\circ}{\pi} \approx 57.3^\circ .
\]
\end{definition}

\begin{method}[Omrekenen]\label{met:g11:trigo:convert}
Graden en radialen zijn evenredig: vermenigvuldig met $\frac{\pi}{180}$
om van graden naar radialen te gaan, met $\frac{180}{\pi}$ de andere
kant. Bijvoorbeeld $60^\circ = 60 \times \frac{\pi}{180} = \frac{\pi}{3}$,
en $\frac{3\pi}{4} = \frac{3 \times 180^\circ}{4} = 135^\circ$.
\end{method}

\begin{definition}[De lijn op de cirkel opwinden]\label{def:g11:trigo:winding}
Aan elk \omterm{def:g10:numbers:sets}{reëel getal} $t$ koppelt men het punt $M(t)$ van $\mathcal C$
verkregen door vanaf $I$ een afstand $\abs t$ langs de cirkel te lopen,
tegen de klok in als $t \geq 0$, met de klok mee als $t < 0$. Omdat de
omtrek $2\pi$ is, bereiken de getallen $t$ en $t + 2k\pi$ ($k \in \Z$)
hetzelfde punt: $M(t + 2k\pi) = M(t)$.
\end{definition}

\begin{example}
$M(0) = I$; $M\!\left(\frac{\pi}{2}\right) = (0, 1)$, de top van de
cirkel; $M(\pi) = (-1, 0)$; $M\!\left(-\frac{\pi}{2}\right) = (0, -1)$;
en $M\!\left(\frac{9\pi}{4}\right) = M\!\left(\frac{\pi}{4} +
2\pi\right) = M\!\left(\frac{\pi}{4}\right)$.
\end{example}

\section{Cosinus en sinus}

\begin{definition}[Cosinus en sinus]\label{def:g11:trigo:cossin}
Voor elke reële $t$ zijn de \emph{cosinus}\index{cosinus} en
\emph{sinus}\index{sinus} van $t$ de \omterm{def:g10:coordgeom:system}{coördinaten} van het punt $M(t)$ van
de eenheidscirkel:
\[
M(t) = (\cos t,\ \sin t).
\]
\end{definition}

\begin{proposition}[Eerste eigenschappen]\label{prop:g11:trigo:first}
Voor alle $t \in \R$ en $k \in \Z$:
\[
-1 \leq \cos t \leq 1, \qquad
-1 \leq \sin t \leq 1, \qquad
\cos^2 t + \sin^2 t = 1,
\]
\[
\cos(t + 2k\pi) = \cos t, \qquad \sin(t + 2k\pi) = \sin t .
\]
\end{proposition}

\begin{proof}
$M(t)$ ligt op de eenheidscirkel, dus haar \omterm{def:g10:coordgeom:system}{coördinaten} liggen tussen
$-1$ en $1$, en ze voldoen aan de cirkelvergelijking $x^2 + y^2 = 1$, wat
de identiteit $\cos^2 t + \sin^2 t = 1$ is. Periodiciteit herhaalt
$M(t + 2k\pi) = M(t)$ (\cref{def:g11:trigo:winding}).
\end{proof}

\begin{example}\label{ex:g11:trigo:identity}
Als $\sin t = \frac35$ en $t \in \intcc{\frac\pi2}{\pi}$ (tweede
kwadrant), dan $\cos^2 t = 1 - \frac{9}{25} = \frac{16}{25}$, dus
$\cos t = \pm\frac45$; in het tweede kwadrant is de \omterm{def:g10:coordgeom:system}{abscis} negatief,
dus $\cos t = -\frac45$.
\end{example}

De waarden om uit het hoofd te kennen:
\begin{center}
\begin{tabular}{c|ccccc}
$t$ & $0$ & $\dfrac{\pi}{6}$ & $\dfrac{\pi}{4}$ & $\dfrac{\pi}{3}$ & $\dfrac{\pi}{2}$\\[6pt]
\hline\rule{0pt}{14pt}%
$\cos t$ & $1$ & $\dfrac{\sqrt3}{2}$ & $\dfrac{\sqrt2}{2}$ & $\dfrac12$ & $0$\\[6pt]
$\sin t$ & $0$ & $\dfrac12$ & $\dfrac{\sqrt2}{2}$ & $\dfrac{\sqrt3}{2}$ & $1$
\end{tabular}
\end{center}

\begin{remark}
Een geheugensteuntje: de sinussen lezen
$\frac{\sqrt0}{2}, \frac{\sqrt1}{2}, \frac{\sqrt2}{2}, \frac{\sqrt3}{2},
\frac{\sqrt4}{2}$, en de cosinussen zijn dezelfde lijst omgekeerd.
\end{remark}

\section{Geassocieerde hoeken}

\begin{proposition}[Geassocieerde hoeken]\label{prop:g11:trigo:associated}
\index{geassocieerde hoeken}
Voor alle $t \in \R$:
\[
\begin{aligned}
\cos(-t) &= \cos t, & \sin(-t) &= -\sin t,\\
\cos(\pi - t) &= -\cos t, & \sin(\pi - t) &= \sin t,\\
\cos(\pi + t) &= -\cos t, & \sin(\pi + t) &= -\sin t,\\
\cos\left(\tfrac{\pi}{2} - t\right) &= \sin t, &
\sin\left(\tfrac{\pi}{2} - t\right) &= \cos t .
\end{aligned}
\]
\end{proposition}

\begin{proof}
Elke regel drukt een symmetrie van de cirkel uit.

$M(-t)$ is de spiegeling van $M(t)$ in de $x$-as: de \omterm{def:g10:coordgeom:system}{abscis} blijft
onveranderd, de \omterm{def:g10:coordgeom:system}{ordinaat} wisselt van teken.

$M(\pi - t)$ is de spiegeling van $M(t)$ in de $y$-as: achteruit lopen
vanaf $\pi$ met $t$ landt horizontaal tegenover vooruit lopen vanaf $0$
met $t$. De \omterm{def:g10:coordgeom:system}{abscis} wisselt van teken, de \omterm{def:g10:coordgeom:system}{ordinaat} blijft onveranderd.

$M(\pi + t)$ ligt diametraal tegenover $M(t)$ (halve draai): beide
\omterm{def:g10:coordgeom:system}{coördinaten} wisselen van teken.

$M\!\left(\frac\pi2 - t\right)$ is de spiegeling van $M(t)$ in de
diagonaal $y = x$, die de twee \omterm{def:g10:coordgeom:system}{coördinaten} verwisselt.
\end{proof}

{\small De vier geassocieerde punten: $M(-t)$ spiegelt $M(t)$ in de

tegenover. Elke identiteit van \cref{prop:g11:trigo:associated} wordt van
dit plaatje afgelezen.}
\end{omfigure}

\begin{example}
$\cos\frac{2\pi}{3} = \cos\left(\pi - \frac\pi3\right)
= -\cos\frac\pi3 = -\frac12$, en
$\sin\frac{7\pi}{6} = \sin\left(\pi + \frac\pi6\right)
= -\sin\frac\pi6 = -\frac12$. De tabel van vijf waarden, uitgebreid met
de symmetrieën, dekt de hele cirkel.
\end{example}

\section{Trigonometrische vergelijkingen oplossen}

\begin{theorem}[De vergelijkingen $\cos t = \cos a$ en $\sin t = \sin a$]
\label{thm:g11:trigo:equations}
Voor \omterm{def:g10:numbers:sets}{reële getallen} $t$ en $a$:
\[
\cos t = \cos a \iff t = a + 2k\pi \ \text{of}\ t = -a + 2k\pi
\quad (k \in \Z),
\]
\[
\sin t = \sin a \iff t = a + 2k\pi \ \text{of}\ t = \pi - a + 2k\pi
\quad (k \in \Z).
\]
\end{theorem}

\begin{proof}
Twee punten van de eenheidscirkel hebben dezelfde \omterm{def:g10:coordgeom:system}{abscis} precies wanneer
ze gelijk zijn of spiegelbeelden van elkaar in de $x$-as; volgens
\cref{prop:g11:trigo:associated} is het spiegelbeeld van $M(a)$ gelijk
aan $M(-a)$. Dus $\cos t = \cos a$ betekent $M(t) = M(a)$ of
$M(t) = M(-a)$, d.w.z.\ $t = \pm a$ op een veelvoud van $2\pi$ na.
Evenzo betekenen gelijke ordinaten gelijke punten of spiegelbeelden in de
$y$-as, en het spiegelbeeld van $M(a)$ is $M(\pi - a)$.
\end{proof}

\begin{method}[$\cos t = c$ oplossen op een interval]\label{met:g11:trigo:solve}
\begin{enumerate}
  \item Vind \emph{één} hoek $a$ met $\cos a = c$, uit de tabel van
        bekende waarden.
  \item Schrijf de algemene oplossingen $t = \pm a + 2k\pi$
        (\cref{thm:g11:trigo:equations}).
  \item Kies de gehele $k$ die binnen het gevraagde \omterm{def:g10:numbers:interval}{interval} vallen, en
        som de oplossingen op.
\end{enumerate}
Ga evenzo te werk voor $\sin t = c$ met $t = a + 2k\pi$ of
$t = \pi - a + 2k\pi$.
\end{method}

\begin{example}\label{ex:g11:trigo:solve}
Los $\cos t = \frac12$ op op $\intoc{-\pi}{\pi}$. Eén oplossing van
$\cos a = \frac12$ is $a = \frac{\pi}{3}$. De algemene oplossingen zijn
$t = \frac\pi3 + 2k\pi$ en $t = -\frac\pi3 + 2k\pi$. Binnen
$\intoc{-\pi}{\pi}$ draagt alleen $k = 0$ bij:
$t \in \left\{-\frac\pi3,\ \frac\pi3\right\}$.

Los $\sin t = -\frac{\sqrt2}{2}$ op op $\intco{0}{2\pi}$. Eén hoek is
$a = -\frac\pi4$; de oplossingen zijn $t = -\frac\pi4 + 2k\pi$ en
$t = \pi + \frac\pi4 + 2k\pi = \frac{5\pi}{4} + 2k\pi$. In
$\intco{0}{2\pi}$: $t = \frac{7\pi}{4}$ (van $k = 1$) en
$t = \frac{5\pi}{4}$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g11:trigo:1}
Zet om naar radialen: $30^\circ$, $45^\circ$, $120^\circ$, $270^\circ$.
Zet om naar graden: $\frac{\pi}{5}$, $\frac{5\pi}{6}$, $\frac{7\pi}{4}$,
$\frac{2\pi}{9}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:trigo:2}
Plaats op de eenheidscirkel de punten $M(t)$ voor
\[
t = \frac{2\pi}{3},\quad
t = -\frac{\pi}{4},\quad
t = \frac{17\pi}{6},\quad
t = -\frac{7\pi}{2},
\]
na elke te herleiden tot een waarde in $\intoc{-\pi}{\pi}$ modulo
$2\pi$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:trigo:3}
Geef met de tabel en de \omterm{prop:g11:trigo:associated}{geassocieerde hoeken} de exacte waarden van
\[
\cos\frac{3\pi}{4}, \quad
\sin\frac{5\pi}{6}, \quad
\cos\left(-\frac{\pi}{3}\right), \quad
\sin\frac{4\pi}{3}, \quad
\cos\frac{11\pi}{6} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:trigo:4}
Gegeven $\cos t = \frac{5}{13}$ met $t \in \intoo{-\frac\pi2}{0}$, bereken
$\sin t$ exact.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:trigo:5}
Vereenvoudig, voor elke reële $x$:
\[
A = \cos(\pi + x) + \cos(-x) + \sin\left(\frac\pi2 - x\right) - \cos x,
\qquad
B = \sin(\pi - x) + \sin(\pi + x) + \sin(-x) .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:trigo:6}
Los op op $\intoc{-\pi}{\pi}$:
\[
\cos t = \frac{\sqrt3}{2}, \qquad
\sin t = \frac12, \qquad
\cos t = -1 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:trigo:7}
Los op op $\intco{0}{2\pi}$:
\[
\sin t = -\frac{\sqrt3}{2}, \qquad
\cos t = 0, \qquad
2\sin t + 1 = 0 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:trigo:8}
Los op $\R$ de \omterm{def:g10:algebra:equation}{vergelijking} $\cos 2t = \cos t$ op. (Gebruik
\cref{thm:g11:trigo:equations} met $a = t$, en bespreek de twee
families van oplossingen.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:trigo:9}
Toon aan dat voor alle reële $x$,
\[
\bigl(\cos x + \sin x\bigr)^2 + \bigl(\cos x - \sin x\bigr)^2 = 2 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:trigo:10}
Een wiel van straal $30$ cm rolt zonder slippen. Door welke hoek (in
radialen, daarna in graden) draait het wanneer de fiets $1.5$ m
vooruitgaat? Hoe ver gaat de fiets vooruit tijdens één volledige
omwenteling van het wiel?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:trigo:11}
Los op $\intoc{-\pi}{\pi}$ de ongelijkheid $\cos t \leq \frac12$ op.
(Schets de eenheidscirkel, markeer de regio waar de \omterm{def:g10:coordgeom:system}{abscis} ten hoogste
$\frac12$ is, en lees het \omterm{def:g10:numbers:interval}{interval} van $t$-waarden af.)
\end{exercise}
