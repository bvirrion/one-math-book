\chapter{Beschrijvende statistiek}\label{ch:g11:stat}

Statistiek comprimeert een lijst getallen tot een paar betekenisvolle
samenvattingen: waar de data zit (gemiddelde, mediaan), en hoe gespreid
ze is (kwartielen, variantie, standaardafwijking). Dit hoofdstuk werkt
alles uit op één doorlopend voorbeeld. Dezelfde samenvattingen keren
terug voor stochastische variabelen in \cref{ch:g11:prob}, en hun
kanswetten in \cref{ch:g12:sums}.

Doorheen het hoofdstuk is onze doorlopende dataset de cijfers (op $20$)
van een klas van $12$ leerlingen:
\[
8,\ 9,\ 10,\ 10,\ 11,\ 12,\ 12,\ 12,\ 13,\ 14,\ 16,\ 17 .
\]

\section{Gemiddelde en mediaan}

\begin{definition}[Gemiddelde]\label{def:g11:stat:mean}
Het \emph{gemiddelde}\index{gemiddelde} van de waarden
$x_1, x_2, \dots, x_n$ is
\[
\bar x = \frac{x_1 + x_2 + \dots + x_n}{n}.
\]
Als de waarde $x_i$ voorkomt met frequentie $n_i$ (met
$n_1 + \dots + n_k = n$), dan
$\bar x = \frac{n_1 x_1 + \dots + n_k x_k}{n}$.
\end{definition}

\begin{definition}[Mediaan]\label{def:g11:stat:median}
Een \emph{mediaan}\index{mediaan} van een gesorteerde dataset is een
waarde die haar in tweeën deelt: minstens de helft van de waarden is
$\leq$ haar, en minstens de helft is $\geq$ haar. In de praktijk:
sorteer de $n$ waarden; als $n$ oneven is, is de mediaan de middelste
waarde; als $n$ even is, neem het midden van de twee middelste waarden.
\end{definition}

\begin{example}\label{ex:g11:stat:meanmedian}
Voor de doorlopende dataset is de som $144$, dus
$\bar x = \frac{144}{12} = 12$. De twee middelste waarden ($6$de en
$7$de van de gesorteerde lijst) zijn beide $12$, dus de mediaan is ook
$12$. Gemiddelde en mediaan hoeven niet samen te vallen: de mediaan
negeert \emph{hoe ver} de extreme waarden liggen, het gemiddelde niet.
Vervangen van het topcijfer $17$ door $20$ verplaatst het gemiddelde
naar $12.25$ maar laat de mediaan op $12$.
\end{example}

\section{Kwartielen}

\begin{definition}[Kwartielen, interkwartielafstand]\label{def:g11:stat:quartiles}
Het \emph{eerste kwartiel}\index{kwartiel} $Q_1$ is de kleinste waarde
zodat minstens een kwart van de data $\leq Q_1$ is; het \emph{derde
kwartiel} $Q_3$ is de kleinste waarde zodat minstens drie kwart van de
data $\leq Q_3$ is. De \emph{interkwartielafstand} is $Q_3 - Q_1$: de
breedte van de centrale helft van de data.
\end{definition}

\begin{example}
$n = 12$: een kwart is $3$, dus $Q_1$ is de $3$de gesorteerde waarde:
$Q_1 = 10$. Drie kwart is $9$, dus $Q_3$ is de $9$de waarde:
$Q_3 = 13$. De interkwartielafstand is $3$: de centrale helft van de
klas zit binnen $3$ punten.
\end{example}

\section{Variantie en standaardafwijking}

\begin{definition}[Variantie, standaardafwijking]\label{def:g11:stat:variance}
De \emph{variantie}\index{variantie} van de dataset is het gemiddelde van
de gekwadrateerde afwijkingen van het gemiddelde:
\[
V = \frac{1}{n}\sum_{i=1}^{n} (x_i - \bar x)^2,
\]
en de \emph{standaardafwijking}\index{standaardafwijking} is
$\sigma = \sqrt{V}$, uitgedrukt in dezelfde eenheid als de data.
\end{definition}

\begin{proposition}[Snelle formule]\label{prop:g11:stat:shortcut}
De variantie is het gemiddelde van de kwadraten minus het kwadraat van
het gemiddelde:
\[
V = \frac{1}{n}\sum_{i=1}^{n} x_i^2 \;-\; \bar x^{\,2}.
\]
\end{proposition}

\begin{proof}
Ontwikkel elke gekwadrateerde afwijking:
$(x_i - \bar x)^2 = x_i^2 - 2\bar x\, x_i + \bar x^2$. Sommeren en
delen door $n$:
\[
V = \frac1n \sum x_i^2 - 2\bar x \cdot \underbrace{\frac1n \sum
x_i}_{=\ \bar x} + \frac1n \cdot n\bar x^2
= \frac1n\sum x_i^2 - 2\bar x^2 + \bar x^2
= \frac1n\sum x_i^2 - \bar x^2 .
\]
Dezelfde identiteit, voor stochastische variabelen, is
\cref{prop:g11:prob:konig}.
\end{proof}

\begin{example}\label{ex:g11:stat:variance}
Voor de doorlopende dataset zijn de afwijkingen van $\bar x = 12$
$-4$, $-3$, $-2$, $-2$, $-1$, $0$, $0$, $0$, $1$, $2$, $4$, $5$; hun
kwadraten sommen tot
\[
16 + 9 + 4 + 4 + 1 + 0 + 0 + 0 + 1 + 4 + 16 + 25 = 80 .
\]
Dus
\[
V = \frac{80}{12} = \frac{20}{3} \approx 6.67,
\qquad
\sigma = \sqrt{20/3} \approx 2.6 \text{ punten}.
\]
Globaal genomen ligt een typisch cijfer ongeveer $2.6$ punten van het
gemiddelde.
\end{example}

\begin{proposition}[Eenhedenwisseling]\label{prop:g11:stat:affine}
Als elke waarde getransformeerd wordt door $y_i = a x_i + b$, dan
\[
\bar y = a \bar x + b,
\qquad
V_y = a^2\, V_x,
\qquad
\sigma_y = \abs{a}\,\sigma_x .
\]
\end{proposition}

\begin{proof}
$\bar y = \frac1n \sum (a x_i + b) = a\left(\frac1n\sum x_i\right) +
\frac1n \cdot nb = a\bar x + b$. Dan
$y_i - \bar y = a(x_i - \bar x)$: elke afwijking wordt geschaald met $a$
(de verschuiving $b$ valt weg), dus elk gekwadrateerd afwijking wordt
geschaald met $a^2$, en hun gemiddelde ook. Vierkantswortels nemen geeft
$\sigma_y = \abs a\,\sigma_x$.
\end{proof}

\begin{example}
Temperaturen in graden Celsius met gemiddelde $20$ en
standaardafwijking $3$ converteren naar Fahrenheit via
$F = 1.8\,C + 32$: het gemiddelde wordt $1.8 \times 20 + 32 = 68$, de
standaardafwijking $1.8 \times 3 = 5.4$. Verschuiven van een dataset
verandert haar spreiding niet; herschalen herschaalt de spreiding.
\end{example}

\begin{method}[Datasets samenvatten en vergelijken]\label{met:g11:stat:compare}
Om twee datasets te vergelijken, zet naast elkaar: gemiddelde (of
mediaan) voor \emph{positie}, standaardafwijking (of
interkwartielafstand) voor \emph{spreiding}. Het paar (mediaan,
interkwartielafstand) is robuust tegen extreme waarden; het paar
(gemiddelde, standaardafwijking) gebruikt elke waarde en voedt de
kansrekening.
\end{method}

\begin{example}
Twee boogschutters schieten elk $10$ pijlen; scores middelen $8.5$ voor
beide, maar de standaardafwijkingen zijn $0.5$ en $2.1$. Zelfde niveau
gemiddeld --- maar de eerste schutter is veel consistenter.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g11:stat:1}
Bereken het gemiddelde, de mediaan, de kwartielen en de
interkwartielafstand van de dataset
\[
5,\ 7,\ 7,\ 8,\ 9,\ 10,\ 11,\ 14 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:stat:2}
Bereken de variantie en standaardafwijking van de dataset
$2,\ 4,\ 4,\ 4,\ 5,\ 5,\ 7,\ 9$ (eerst het gemiddelde, daarna de snelle
formule van \cref{prop:g11:stat:shortcut}).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:stat:3}
Een dobbelsteen wordt $50$ keer gegooid; de uitkomsten worden
samengevat door frequenties:
\begin{center}
\begin{tabular}{c|cccccc}
uitkomst & 1 & 2 & 3 & 4 & 5 & 6\\
\hline
frequentie & 6 & 9 & 8 & 10 & 9 & 8
\end{tabular}
\end{center}
Bereken de gemiddelde uitkomst en de mediaan.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:stat:4}
Een klas van $15$ leerlingen heeft gemiddeld cijfer $11.2$. Een $16$de
leerling komt erbij en scoort $18$. Wat is het nieuwe klasgemiddelde?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:stat:5}
Het gemiddelde van de dataset $6, 8, x, 12, 14$ is $10$. Vind $x$,
bereken daarna de standaardafwijking van de volledige dataset.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:stat:6}
Cijfers met gemiddelde $12.4$ en standaardafwijking $2.5$ worden
herschald door $y = \frac{x}{2} + 4$. Geef het nieuwe gemiddelde en de
nieuwe standaardafwijking.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:stat:7}
Groep A ($20$ waarden) heeft gemiddelde $10$; groep B ($30$ waarden)
heeft gemiddelde $15$. Bereken het gemiddelde van de $50$ waarden
samengevoegd. Is het het midden $12.5$? Waarom niet?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:stat:8}
Twee productielijnen vullen suikerzakken van $1$ kg. Steekproeven geven:
lijn 1: gemiddelde $1.002$ kg, standaardafwijking $0.006$ kg; lijn 2:
gemiddelde $1.001$ kg, standaardafwijking $0.019$ kg. Welke lijn moet
eerst herijkt worden, en waarom?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:stat:9}
Een dataset van $n = 10$ waarden heeft $\sum x_i = 70$ en
$\sum x_i^2 = 540$. Bereken haar gemiddelde, variantie en
standaardafwijking.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:stat:10}
De dataset $1,\ 2,\ 2,\ 3,\ 3,\ 3,\ 4,\ 4,\ 5,\ 23$ bevat een
\emph{uitbijter}.
\begin{enumerate}
  \item Bereken het gemiddelde en de mediaan met en zonder de waarde
        $23$.
  \item Bereken de interkwartielafstand met en zonder $23$.
  \item Welke samenvattingen zijn robuust tegen de uitbijter?
        Concludeer met een aanbeveling.
\end{enumerate}
\end{exercise}
