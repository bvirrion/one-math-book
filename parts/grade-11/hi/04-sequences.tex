\chapter{अनुक्रम: पहला कोर्स}\label{ch:g11:seq}

ए \omterm{def:g11:seq:sequence}{अनुक्रम} एक नियम द्वारा निर्मित संख्याओं की एक सूची है: क्रमिक शेष
एक बचत खाते का, वर्ष दर वर्ष जनसंख्या का आकार। यह
अध्याय उन दो परिवारों का अध्ययन करता है जो अनुप्रयोगों पर हावी हैं ---
\emph{\omterm{def:g11:seq:arithmetic}{अंकगणित}} दृश्यों, जो समान चरणों से बढ़ता है, और
\emph{\omterm{def:g11:seq:geometric}{ज्यामितिक}} दृश्यों, जो समान अनुपात से बढ़ता है। कठोर सिद्धांत
की सीमाएँ विकसित की गई हैं \cref{ch:g12:seq}.

\section{अनुक्रम को परिभाषित करना}

\begin{definition}[अनुक्रम]\label{def:g11:seq:sequence}
ए \emph{अनुक्रम}\index{अनुक्रम} $(u_n)$ प्रत्येक को असाइन करता है पूर्णांक
$ n \geq 0$ (या $ n \geq 1$) ए \omterm{def:g10:numbers:sets}{वास्तविक संख्या} $ u_n $, इसका \emph{सूचकांक की अवधि
$ n $}. एक क्रम दिया जा सकता है
\begin{itemize}
  \item \emph{स्पष्ट रूप से}, के लिए एक सूत्र द्वारा $ u_n $ के अनुसार $ n $: उदा.\
        $ u_n = n^2 + 1$;
  \item \emph{रिकर्सिवली}, इसके पहले पद और प्रत्येक से पारित होने के नियम के अनुसार
        अगले के लिए पद: उदा.\ $ u_0 = 3$ और $ u_{n+1} = 2u_n - 1$.
\end{itemize}
\end{definition}

\begin{example}
के लिए $ u_n = n^2 + 1$: $ u_0 = 1$, $ u_1 = 2$, $ u_2 = 5$, और $ u_{10} = 101$
सीधे. के लिए $ u_0 = 3$, $ u_{n+1} = 2u_n - 1$: $ u_1 = 5$, $ u_2 = 9$,
$ u_3 = 17$ --- प्रत्येक पद को पिछले शब्द की आवश्यकता होती है; तक पहुँचने $ u_{10}$ लेता है
दस चरण (या एक सामान्य सूत्र, देखें \cref{exo:g11:seq:11}).
\end{example}

\section{अंकगणित अनुक्रम}

\begin{definition}[अंकगणित अनुक्रम]\label{def:g11:seq:arithmetic}
ए \omterm{def:g11:seq:sequence}{अनुक्रम} है \emph{अंकगणित}\index{अंकगणित} साथ
\emph{सामान्य अंतर}\index{सामान्य अंतर} $ d $ यदि प्रत्येक पद पिछले से प्राप्त किया गया है
एक जोड़कर $ d $:
\[
u_{n+1} = u_n + d \quad \text{for all } n.
\]
समतुल्य: अंतर $ u_{n+1} - u_n $ स्थिर है, बराबर है $ d $.
\end{definition}

\begin{theorem}[सामान्य शब्द]\label{thm:g11:seq:arithgeneral}
अगर $(u_n)$ है \omterm{def:g11:seq:arithmetic}{अंकगणित} प्रथम पद के साथ $ u_0$ और \omterm{def:g11:seq:arithmetic}{सामान्य अंतर} $ d $,
तब
\[
u_n = u_0 + n\,d \quad \text{for all } n \geq 0,
\qquad\text{and more generally } u_n = u_p + (n - p)\,d .
\]
\end{theorem}

\begin{proof}
से जाना है $ u_0$ को $ u_n $, नियम ``\omterm{def:g10:vectors:sum}{जोड़}ें $ d $'' लागू की गई है $ n $ समय: एक
कदम देता है $ u_1 = u_0 + d $, दो चरण देते हैं $ u_2 = u_0 + 2d $, और उसके बाद $ n $
प्रत्येक एप्लिकेशन ने एक चरण में योगदान दिया है $ d $, इसलिए $ u_n = u_0 + nd $.
(इस ``और इसी तरह'' को शामिल करके कठोर बनाया गया है \cref{ch:g12:seq}.)
सामान्य सूत्र गिनकर अनुसरण करता है $ n - p $ से कदम $ u_p $ को
$ u_n $.
\end{proof}

\begin{theorem}[क्रमागत पूर्णांकों का योग]\label{thm:g11:seq:intsum}
हरएक के लिए पूर्णांक $ n \geq 1$:
\[
1 + 2 + \dots + n = \frac{n(n+1)}{2}.
\]
अधिक सामान्यतः, a के क्रमागत पदों का योग \omterm{def:g11:seq:arithmetic}{अंकगणित} \omterm{def:g11:seq:sequence}{अनुक्रम}
के बराबर होती है
\[
(\text{number of terms}) \times
\frac{\text{first term} + \text{last term}}{2}.
\]
\end{theorem}

\begin{proof}
योग लिखो $ S $ दो बार, दूसरी बार उल्टे क्रम में, और \omterm{def:g10:vectors:sum}{जोड़}ें
कॉलम दर कॉलम:
\[
\begin{array}{ccccccccc}
S & = & 1 & + & 2 & + & \dots & + & n\\
S & = & n & + & (n-1) & + & \dots & + & 1\\
\hline
2S & = & (n+1) & + & (n+1) & + & \dots & + & (n+1)
\end{array}
\]
वहाँ हैं $ n $ कॉलम, प्रत्येक का योग $ n + 1$, इसलिए $2S = n(n+1)$. एक के लिए
सामान्य \omterm{def:g11:seq:arithmetic}{अंकगणित} \omterm{def:g11:seq:sequence}{अनुक्रम} वही \omterm{def:g10:vectors:sum}{जोड़}ी काम करती है: पहला $+$ अंतिम $=$
दूसरा $+$ आखरी से पहले $= \dots $, क्योंकि एक कदम आगे बढ़ना
बायां छोर ($+d $) की भरपाई दाहिने छोर पर एक कदम पीछे की ओर की जाती है
($-d $).
\end{proof}

\begin{example}
$1 + 2 + \dots + 100 = \frac{100 \times 101}{2} = 5050$. का योग
\omterm{def:g11:func:parity}{विषम} संख्याएँ $1 + 3 + \dots + 99$ ($50$ शर्तें) है
$50 \times \frac{1 + 99}{2} = 2500$.
\end{example}

\section{ज्यामितीय अनुक्रम}

\begin{definition}[ज्यामितीय क्रम]\label{def:g11:seq:geometric}
ए \omterm{def:g11:seq:sequence}{अनुक्रम} है \emph{ज्यामितिक}\index{ज्यामितिक} साथ \emph{सामान्य
अनुपात}\index{सामान्य
अनुपात} $ q \neq 0$ यदि प्रत्येक पद पिछले एक से प्राप्त किया जाता है
से गुणा करना $ q $:
\[
u_{n+1} = q\,u_n \quad \text{for all } n.
\]
समान रूप से, जब कोई पद लुप्त नहीं होता: अनुपात $\frac{u_{n+1}}{u_n}$ है
स्थिर, बराबर $ q $.
\end{definition}

\begin{theorem}[सामान्य शब्द]\label{thm:g11:seq:geomgeneral}
अगर $(u_n)$ है \omterm{def:g11:seq:geometric}{ज्यामितिक} प्रथम पद के साथ $ u_0$ और अनुपात $ q $, तब
\[
u_n = u_0\, q^n \quad \text{for all } n \geq 0,
\qquad\text{and more generally } u_n = u_p\, q^{\,n-p} .
\]
\end{theorem}

\begin{proof}
चरण-गिनती के समान ही \cref{thm:g11:seq:arithgeneral}: से $ u_0$ को
$ u_n $, नियम `` गुणा करें $ q $'' लागू की गई है $ n $ समय, योगदान ए
कारक $ q^n $.
\end{proof}

\begin{theorem}[ज्यामितीय योग]\label{thm:g11:seq:geomsum}
हर वास्तविक के लिए $ q \neq 1$ और पूर्णांक $ n \geq 0$:
\[
1 + q + q^2 + \dots + q^n = \frac{1 - q^{\,n+1}}{1 - q}.
\]
\end{theorem}

\begin{proof}
होने देना $ S = 1 + q + \dots + q^n $. गुणा करके $ q $:
$ qS = q + q^2 + \dots + q^{n+1}$. घटाएँ:
\[
S - qS = \bigl(1 + q + \dots + q^n\bigr)
- \bigl(q + q^2 + \dots + q^{n+1}\bigr) = 1 - q^{\,n+1},
\]
क्योंकि प्रत्येक मध्यवर्ती पद प्रत्येक योग में एक बार आता है और रद्द हो जाता है।
इस तरह $(1 - q)S = 1 - q^{\,n+1}$, और से विभाजित करना $1 - q \neq 0$ देता है
सूत्र.
\end{proof}

\begin{example}\label{ex:g11:seq:chessboard}
$1 + 2 + 4 + \dots + 2^{10} = \frac{1 - 2^{11}}{1 - 2} = 2^{11} - 1 =
2047$: शतरंज की बिसात पर चावल के दानों को दोगुना करना भारी पड़ जाता है
किसी भी अन्न भंडार से बहुत पहले $64$ वां वर्ग, जहां कुल है
$2^{64} - 1 \approx 1.8 \times 10^{19}$.
\end{example}

\begin{method}[अनुक्रम के प्रकार को पहचानना]\label{met:g11:seq:recognize}
गणना करना $ u_{n+1} - u_n $ और सरल करें. यदि परिणाम स्थिर है $ d $, द
\omterm{def:g11:seq:sequence}{अनुक्रम} है \omterm{def:g11:seq:arithmetic}{अंकगणित}. अन्यथा गणना करें $\frac{u_{n+1}}{u_n}$ (शर्तें
शून्येतर) और सरल करें: एक स्थिरांक $ q $ मतलब \omterm{def:g11:seq:geometric}{ज्यामितिक}. यदि कोई नहीं है
स्थिर, \omterm{def:g11:seq:sequence}{अनुक्रम} किसी भी प्रकार का नहीं है --- से कभी निष्कर्ष न निकालें
पहले कुछ शब्द अकेले।
\end{method}

\begin{example}
के लिए $ u_n = 3 \times 5^n $:
$\frac{u_{n+1}}{u_n} = \frac{3 \times 5^{n+1}}{3 \times 5^n} = 5$ सभी के लिए
$ n $: \omterm{def:g11:seq:geometric}{ज्यामितिक} अनुपात के साथ $5$. के लिए $ u_n = n^2$:
$ u_1 - u_0 = 1$ लेकिन $ u_2 - u_1 = 3$, और
$\frac{u_1}{u_0}$ परिभाषित भी नहीं है --- न ही \omterm{def:g11:seq:arithmetic}{अंकगणित} और न
\omterm{def:g11:seq:geometric}{ज्यामितिक}.
\end{example}

\section{दिष्टता}

\begin{definition}[मोनोटोनिक अनुक्रम]\label{def:g11:seq:monotone}
ए \omterm{def:g11:seq:sequence}{अनुक्रम} $(u_n)$ है \emph{\omterm{def:g11:func:monotone}{की बढ़ती}} अगर $ u_{n+1} \geq u_n $ सभी के लिए $ n $,
और \emph{\omterm{def:g10:functions:variations}{घटते}} अगर $ u_{n+1} \leq u_n $ सभी के लिए $ n $.
\end{definition}

\begin{method}[एकरसता का अध्ययन]\label{met:g11:seq:monotonicity}
के चिन्ह का अध्ययन करें $ u_{n+1} - u_n $. के लिए दृश्यों सकारात्मक शर्तों के साथ, एक
इसके बजाय तुलना कर सकते हैं $\frac{u_{n+1}}{u_n}$ को $1$.
\end{method}

\begin{example}
एक \omterm{def:g11:seq:arithmetic}{अंकगणित} \omterm{def:g11:seq:sequence}{अनुक्रम} है \omterm{def:g11:func:monotone}{की बढ़ती} कब $ d \geq 0$
($ u_{n+1} - u_n = d $), \omterm{def:g10:functions:variations}{घटते} कब $ d \leq 0$. ए ज्यामितीय क्रम
साथ $ u_0 > 0$ और $ q > 1$ है \omterm{def:g11:func:monotone}{की बढ़ती}:
$ u_{n+1} - u_n = u_0 q^n (q - 1) > 0$; साथ $ u_0 > 0$ और $0 < q < 1$ यह है \omterm{def:g10:functions:variations}{घटते}.
\end{example}

\section{लंबे समय तक चलने वाला व्यवहार, अनौपचारिक रूप से}

क्या होता है $ u_n $ कब $ n $ बहुत बड़ा हो जाता है? एक के लिए \omterm{def:g11:seq:arithmetic}{अंकगणित}
क्रम साथ $ d > 0$, शर्तें $ u_0 + nd $ किसी भी निश्चित संख्या से अधिक
अंततः. एक के लिए ज्यामितीय क्रम साथ $0 < q < 1$, शर्तें
$ u_0 q^n $ की ओर सिकुड़ना $0$: से बार-बार गुणा करना $0.9$, कहते हैं, नष्ट हो जाता है
कोई आरंभिक मान. और के लिए $ q > 1$ शब्द विस्फोटित होते हैं, जैसे कि
\cref{ex:g11:seq:chessboard}.

\begin{remark}
इन कथनों को बिल्कुल सटीक बनाया जा सकता है --- ``शर्तें अंततः
की किसी भी दूरी के भीतर रहें $0$'' --- और साबित हुआ। यही सिद्धांत है
का \emph{सीमा}, का उद्घाटन विषय \cref{ch:g12:seq}.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g11:seq:1}
प्रत्येक के लिए \omterm{def:g11:seq:sequence}{अनुक्रम}, गणना करें $ u_1$, $ u_2$, $ u_3$:
\[
u_n = \frac{n}{n+1}; \qquad
u_0 = 5,\ u_{n+1} = 3u_n - 2; \qquad
u_n = (-1)^n\,n .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:seq:2}
$(u_n)$ है \omterm{def:g11:seq:arithmetic}{अंकगणित} साथ $ u_0 = 7$ और $ d = -3$. गणना करना $ u_{10}$ और
$ u_{25}$. $(v_n)$ है \omterm{def:g11:seq:arithmetic}{अंकगणित} साथ $ v_3 = 11$ और $ v_8 = 26$. खोजें
\omterm{def:g11:seq:arithmetic}{सामान्य अंतर} और $ v_0$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:seq:3}
$(u_n)$ है \omterm{def:g11:seq:geometric}{ज्यामितिक} साथ $ u_0 = 5$ और $ q = 2$. गणना करना $ u_8$. $(v_n)$ है
\omterm{def:g11:seq:geometric}{ज्यामितिक} सकारात्मक शर्तों के साथ, $ v_2 = 12$ और $ v_4 = 48$. अनुपात ज्ञात कीजिये
और $ v_0$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:seq:4}
गणना करना
\[
1 + 2 + 3 + \dots + 500, \qquad
4 + 7 + 10 + \dots + 61, \qquad
1 + \frac12 + \frac14 + \dots + \frac{1}{2^{10}} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:seq:5}
निर्धारित करें कि क्या प्रत्येक \omterm{def:g11:seq:sequence}{अनुक्रम} है \omterm{def:g11:seq:arithmetic}{अंकगणित}, \omterm{def:g11:seq:geometric}{ज्यामितिक}, या न ही:
\[
u_n = 4n - 1; \qquad
v_n = \frac{2^n}{3^{n+1}}; \qquad
w_n = n^2 + n .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:seq:6}
एक थिएटर है $20$ पंक्तियाँ: $16$ पहली पंक्ति में सीटें, और प्रत्येक पंक्ति में हैं
$2$ पहले से ज्यादा सीटें. आखिर में कितनी सीटें हैं
पंक्ति? पूरे थिएटर में?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:seq:7}
बैक्टीरिया की आबादी हर घंटे दोगुनी हो जाती है; दोपहर के समय वहाँ हैं $500$
बैक्टीरिया. रात 8 बजे कितने लोग हैं? पूरे कितने घंटे बाद
जनसंख्या पहले दस लाख से अधिक हो गई? (क्रमिक शक्तियों को आज़माकर हल करें
$2$.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:seq:8}
हर महीने एक बचतकर्ता जमा करता है $100$ भुगतान करने वाले खाते में यूरो
$0.2\%$ मौजूदा शेष राशि पर प्रति माह ब्याज (ब्याज जमा किया जाता है
जमा करने से ठीक पहले)। होने देना $ c_n $ के ठीक बाद संतुलन हो $ n $-वें
जमा, तो $ c_1 = 100$ और $ c_{n+1} = 1.002\,c_n + 100$. गणना करना $ c_2$
और $ c_3$, और समझाएं क्यों $(c_n)$ नहीं है \omterm{def:g11:seq:arithmetic}{अंकगणित} और न \omterm{def:g11:seq:geometric}{ज्यामितिक}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:seq:9}
की एकरसता का अध्ययन करें दृश्यों
\[
u_n = n^2 - 8n \ (n \geq 0), \qquad
v_n = \frac{3^n}{n!}\ (n \geq 1),
\]
कहाँ $ n! = 1 \times 2 \times \dots \times n $. (के लिए $(v_n)$, तुलना करना
$\frac{v_{n+1}}{v_n}$ को $1$.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:seq:10}
प्रथम का योग $ n $ एक की शर्तें \omterm{def:g11:seq:arithmetic}{अंकगणित} \omterm{def:g11:seq:sequence}{अनुक्रम} साथ $ u_0 = 3$
और $ d = 4$ के बराबर होती है $903$. खोजो $ n $. (एक द्विघात स्थापित करें \omterm{def:g10:algebra:equation}{समीकरण} में $ n $
और उपयोग करें \cref{ch:g11:quad}.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:seq:11}
होने देना $ u_0 = 3$ और $ u_{n+1} = 2u_n - 1$.
\begin{enumerate}
  \item गणना करना $ u_1, u_2, u_3$ और इसके लिए एक सूत्र का अनुमान लगाएं $ u_n $.
  \item होने देना $ v_n = u_n - 1$. बताते हैं कि $(v_n)$ है \omterm{def:g11:seq:geometric}{ज्यामितिक}, उसे दे दो
        अनुपात और प्रथम पद.
  \item के लिए एक स्पष्ट सूत्र निकालें $ u_n $ और अपने अनुमान को सत्यापित करें.
\end{enumerate}
\end{exercise}

\section{समस्या: ब्रह्मा और फाइबोनैचि के खरगोशों का टॉवर}

\begin{problem}[{सप्ताहांत समस्या --- दो पौराणिक पुनरावृत्तियाँ: वह मीनार जो दुनिया को समाप्त करती है, वह क्रम जो सोने की तरह बढ़ता है, और सहायक युक्ति जो ऋणों को वश में करती है}]
\label{pb:g11:seq:1}
दो दृश्यों गणित की लोककथाओं पर शासन करें। एक गिनता है
ब्रह्मा की मीनार की चालें --- चौंसठ स्वर्ण डिस्क जिनकी
किंवदंती कहती है, स्थानांतरण से दुनिया ख़त्म हो जाएगी। अन्य मायने रखता है
फाइबोनैचि के खरगोश और सुनहरे अनुपात को छुपाते हैं। ना ही
\omterm{def:g11:seq:arithmetic}{अंकगणित}, ना ही \omterm{def:g11:seq:geometric}{ज्यामितिक} --- और दोनों इस पर समर्पण कर देते हैं
अध्याय के हथियार: पुनरावृत्ति, \omterm{def:g11:seq:geometric}{ज्यामितिक} रकम
(\cref{thm:g11:seq:geomsum}), और यह सहा\omterm{def:g11:seq:sequence}{यक-अनुक्रम} की चाल
\cref{exo:g11:seq:11}, जो आपके बंधक की गणना भी करता है।

\medskip
\noindent\textbf{Part I --- The tower of Brahma.}
पहेली: $ n $ की डिस्क \omterm{def:g10:functions:variations}{घटते} आकार खूंटी ए पर रखे गए हैं;
पूरे स्टैक को खूंटी सी पर ले जाएं, एक समय में एक डिस्क, कभी भी रखें नहीं
छोटी डिस्क पर बड़ी डिस्क (खूंटी बी मदद कर सकती है)। होने देना $ h_n $ हो
चालों की \omterm{def:g10:functions:extrema}{न्यूनतम} संख्या.
\begin{enumerate}
  \item (सिक्कों के साथ) खेलें और रिकॉर्ड करें $ h_1$, $ h_2$, $ h_3$.
  \item पुनरावृत्ति के पीछे की रणनीति स्पष्ट करें
        $ h_{n+1} = 2h_n + 1$: पहले और बाद में क्या होना चाहिए
        सबसे बड़ी डिस्क चाल?
  \item की ट्रिक से पुनरावृत्ति का समाधान करें
        \cref{exo:g11:seq:11}: तय करना $ v_n = h_n + 1$, दिखाओ
        $(v_n)$ है \omterm{def:g11:seq:geometric}{ज्यामितिक}, और निष्कर्ष निकालें $ h_n = 2^n - 1$.
  \item किंवदंती का टॉवर है $64$ डिस्क, और भिक्षु चलते हैं
        प्रति सेकंड एक डिस्क. का उपयोग करते हुए $2^{10} = 1024 \approx 10^3$,
        वर्षों में स्थानांतरण समय का अनुमान लगाएं (एक वर्ष लगभग है)।
        $3 \times 10^7$ सेकंड; तुलना करना
        \cref{ex:g11:seq:chessboard}, दूसरे में वही विशाल
        कहानी) क्या हमें चिंता करनी चाहिए?
  \item कोई रणनीति क्यों नहीं हरा सकती $2^n - 1$ चाल? उस पर बहस करें
        \emph{कोई} समाधान पालन करता है
        $ h_{n+1} \geq 2 h_n + 1$: शीर्ष के बारे में क्या सत्य होना चाहिए
        $ n $ निचली डिस्क के ठीक पहले और ठीक बाद की डिस्क
        हटो?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Fibonacci.}
परिभाषित करना $ F_1 = F_2 = 1$ और $ F_{n+2} = F_{n+1} + F_n $ (प्रत्येक पद
पहले दोनों का योग --- लय-गिनती नियम
मिडिल स्कूल वॉल्यूम, अब अपने यूरोपीय नाम के साथ)।
\begin{enumerate}[resume]
  \item सूची $ F_1$ के माध्यम से $ F_{12}$.
  \item बताते हैं कि $(F_n)$ नहीं है \omterm{def:g11:seq:arithmetic}{अंकगणित} और न \omterm{def:g11:seq:geometric}{ज्यामितिक},
        लेकिन यह सख्ती से है \omterm{def:g11:func:monotone}{की बढ़ती} से $ n = 2$ पर
        (\cref{met:g11:seq:monotonicity} और पुनरावृत्ति)।
  \item योग पहचान सिद्ध करें
        \[
        F_1 + F_2 + \dots + F_n = F_{n+2} - 1
        \]
        टेलीस्कोपिंग द्वारा: प्रत्येक को लिखें $ F_k $ जैसा
        $ F_{k+2} - F_{k+1}$ और योग को ढहते हुए देखिये। सत्यापित करें
        इसके लिए $ n = 6$.
  \item वर्गों की पहचान सिद्ध करें
        $ F_1^2 + F_2^2 + \dots + F_n^2 = F_n F_{n+1}$,
        दूरबीन के साथ
        $ F_k F_{k+1} - F_{k-1} F_k = F_k^2$. के लिए सत्यापित करें
        $ n = 4$. (चित्र: भुजाओं का वर्ग
        $1, 1, 2, 3, 5, \dots $ एक आयत टाइल करें --- प्रसिद्ध
        फाइबोनैचि सर्पिल का कंकाल।)
  \item कैसिनी की पहचान बताती है
        $ F_{n+1} F_{n-1} - F_n^2 = (-1)^n $. इसके लिए इसे सत्यापित करें
        $ n = 4, 5, 6$ --- और के इंजन को पहचानें
        मिडिल स्कूल में खेला गया लुप्त-वर्ग चाल
        वॉल्यूम के क्षेत्र की समस्या.
  \item पुनरावृत्ति से दिखाओ कि
        $ F_{n+2} \geq 2 F_n $: फाइबोनैचि प्रत्येक को कम से कम दोगुना कर देती है
        दो चरण --- यह कम से कम उतनी ही तेजी से बढ़ता है ज्यामितीय
        क्रम अनुपात का $\sqrt2$.
  \item अनुपातों की गणना करें $ r_n = \frac{F_{n+1}}{F_n}$ के लिए
        $ n = 3$ को $10$ (तीन दशमलव). यह स्वीकार करते हुए कि वे
        एक सीमा तय करो $ L $, संबंध पारित करें
        $ r_{n+1} = 1 + \frac{1}{r_n}$ सीमा तक और हल करें:
        किस संख्या में \cref{pb:g10:algebra:1} खरगोश करो
        पूजा?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The auxiliary trick, at the
bank.}
\begin{enumerate}[resume]
  \item सामान्यीकरण \cref{exo:g11:seq:11}: के लिए
        $ u_{n+1} = a\,u_n + b $ साथ $ a \neq 1$, होने देना
        $\ell = \frac{b}{1 - a}$ (द \omterm{pb:g10:functions:1}{नियत बिन्दु}). दिखाओ
        $ v_n = u_n - \ell $ है \omterm{def:g11:seq:geometric}{ज्यामितिक} अनुपात का $ a $, और
        निष्कर्ष
        $ u_n = a^n (u_0 - \ell) + \ell $.
  \item एक ऋण: $10\,000$ यूरो पर $1\,\%$ प्रति माह ब्याज,
        चुकाया $300$ यूरो प्रति माह, इसलिए ऋण का पालन होता है
        $ d_{n+1} = 1.01\,d_n - 300$. प्रश्न 13 लागू करें (\omterm{pb:g10:functions:1}{नियत बिन्दु} पहले!) के लिए एक स्पष्ट सूत्र प्राप्त करने के लिए $ d_n $.
  \item कैलकुलेटर की सहायता से, वह पहला महीना ज्ञात करें जिसमें
        कर्ज़ चुका दिया गया है, और कुल राशि चुका दी गई है। कितना
        क्या उधार लेने की लागत स्वयं थी?
  \item का एक शहर $50\,000$ निवासी बढ़ते हैं $2\,\%$ एक वर्ष
        और स्वागत करता है $1\,000$ नवागंतुकों के अलावा:
        $ p_{n+1} = 1.02\,p_n + 1000$. स्पष्ट सूत्र दीजिए
        और उसके बाद की जनसंख्या $10$ साल।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The two royal families.}
\begin{enumerate}[resume]
  \item गणना करना $1 + 2 + 3 + \dots + 1000$
        (\cref{thm:g11:seq:intsum} --- थोड़ा गॉस का योग से
        मिडिल स्कूल वॉल्यूम, अब आधिकारिक), और
        $1 + 2 + 4 + \dots + 2^{19}$
        (\cref{thm:g11:seq:geomsum}).
  \item के योग की गणना करें \omterm{def:g11:seq:arithmetic}{अंकगणित} \omterm{def:g11:seq:sequence}{अनुक्रम}
        $7, 12, 17, \dots, 502$ (कितने पद?)।
  \item बचत योजना: $100$ हर महीने जमा किये गये यूरो, कमाई
        $0.5\,\%$ महीने के; के बाद $ n $-वें शेष राशि जमा करें
        है $100\left(1.005^{n-1} + \dots + 1.005 + 1\right)$.
        इसके बाद शेष राशि की गणना करें $5$ साल ($ n = 60$).
  \item समापन --- अनुक्रम-टैमर की किट: स्पष्ट विरुद्ध
        आवर्ती विवरण; दो शाही परिवार और
        उनके योग सूत्र; सहायक \omterm{def:g11:seq:sequence}{अनुक्रम} वह बदल जाता है
        एफ़िन की पुनरावृत्ति होती है \omterm{def:g11:seq:geometric}{ज्यामितिक} वाले; और फाइबोनैचि,
        दोनों परिवारों के बाहर प्रथम नागरिक, द्वारा आज वश में किया गया
        पहचान और प्रतीक्षारत मैट्रिक्स (ग्रेड 12) और सीमाएँ
        पूर्ण कब्जा के लिए. एक-एक वाक्य.
\end{enumerate}
\end{problem}
