\chapter{त्रिकोणमिति: इकाई वृत्त}\label{ch:g11:trigo}

त्रिकोणमिति समकोण त्रिभुज में शुरू होती है, लेकिन इसका असली घर वही है
\emph{इकाई चक्र}, कहाँ \omterm{def:g11:trigo:cossin}{कोज्या} और \omterm{def:g11:trigo:cossin}{ज्या} बनना कार्य एक मनमानी का
\omterm{def:g10:numbers:sets}{वास्तविक संख्या}. यह अध्याय स्थापित करता है \omterm{def:g11:trigo:radian}{कांति}, असली की घुमावदार
वृत्त के चारों ओर रेखा, हृदय से जानने योग्य सटीक मान, और
समरूपताएं जो सभी शास्त्रीय पहचान उत्पन्न करती हैं। का अध्ययन
$\cos $ और $\sin $ जैसा कार्य --- विविधताएँ, डेरिवेटिव --- ले जाया जाता है
बाहर अंदर \cref{ch:g12:trigo}.

\section{रेडियन और वाइंडिंग}

\begin{definition}[कांति]\label{def:g11:trigo:radian}
होने देना $\mathcal C $ त्रि\omterm{def:g11:trigo:cossin}{ज्या} का वृत्त हो $1$ मूल पर केन्द्रित $ O $,
और $ I(1, 0)$. \emph{कांति}\index{कांति} पर एक कोण का माप
का \omterm{pb:g11:quad:1}{केंद्र} $\mathcal C $ यह उस चाप की लंबाई है जिस पर यह कटता है $\mathcal C $.
चूँकि पूर्ण वृत्त की परिधि होती है $2\pi $:
\[
360^\circ = 2\pi \text{ rad},
\qquad
180^\circ = \pi \text{ rad},
\qquad
1 \text{ rad} = \frac{180^\circ}{\pi} \approx 57.3^\circ .
\]
\end{definition}

\begin{method}[परिवर्तित]\label{met:g11:trigo:convert}
डिग्री और रेडियंस आनुपातिक हैं: से गुणा करें $\frac{\pi}{180}$ जाना
डिग्री से लेकर रेडियंस, द्वारा $\frac{180}{\pi}$ दूसरे तरीके से. उदाहरण के लिए
$60^\circ = 60 \times \frac{\pi}{180} = \frac{\pi}{3}$, और
$\frac{3\pi}{4} = \frac{3 \times 180^\circ}{4} = 135^\circ $.
\end{method}

\begin{definition}[वृत्त पर रेखा को घुमाना]\label{def:g11:trigo:winding}
प्रत्येक के लिए \omterm{def:g10:numbers:sets}{वास्तविक संख्या} $ t $, बिंदु को संबद्ध करें $ M(t)$ का $\mathcal C $
की दूरी पैदल चलकर प्राप्त की गई $\abs t $ से वृत्त के साथ $ I $,
यदि वामावर्त $ t \geq 0$, यदि दक्षिणावर्त $ t < 0$. के बाद से
परिधि है $2\pi $, संख्या $ t $ और $ t + 2k\pi $ ($ k \in \Z $)
उसी बिंदु पर पहुँचें: $ M(t + 2k\pi) = M(t)$.
\end{definition}

\begin{example}
$ M(0) = I $; $ M\!\left(\frac{\pi}{2}\right) = (0, 1)$, का शीर्ष
वृत्त; $ M(\pi) = (-1, 0)$; $ M\!\left(-\frac{\pi}{2}\right) = (0, -1)$;
और $ M\!\left(\frac{9\pi}{4}\right) = M\!\left(\frac{\pi}{4} +
2\pi\right) = M\!\left(\frac{\pi}{4}\right)$.
\end{example}

\section{कोज्या और ज्या}

\begin{definition}[कोज्या और ज्या]\label{def:g11:trigo:cossin}
हर वास्तविक के लिए $ t $, द \emph{कोज्या}\index{कोज्या} और
\emph{ज्या}\index{ज्या} का $ t $ हैं \omterm{def:g10:coordgeom:system}{COORDINATES} मुद्दे की $ M(t)$ का
यूनिट सर्कल:
\[
M(t) = (\cos t,\ \sin t).
\]
\end{definition}

$ I $, बिंदु पर उतरता है $ M(t)$ किसका \omterm{def:g10:coordgeom:system}{COORDINATES} \emph{परिभाषित करना} $\cos t $
और $\sin t $.}
\end{omfigure}

\begin{proposition}[प्रथम गुण]\label{prop:g11:trigo:first}
सभी के लिए $ t \in \R $ और $ k \in \Z $:
\[
-1 \leq \cos t \leq 1, \qquad
-1 \leq \sin t \leq 1, \qquad
\cos^2 t + \sin^2 t = 1,
\]
\[
\cos(t + 2k\pi) = \cos t, \qquad \sin(t + 2k\pi) = \sin t .
\]
\end{proposition}

\begin{proof}
$ M(t)$ इकाई वृत्त पर स्थित है, इसलिए यह है \omterm{def:g10:coordgeom:system}{COORDINATES} के बीच हैं $-1$ और
$1$, और वे मंडली को संतुष्ट करते हैं \omterm{def:g10:algebra:equation}{समीकरण} $ x^2 + y^2 = 1$, जो है
पहचान $\cos^2 t + \sin^2 t = 1$. आवधिकता पुनः स्थापित होती है
$ M(t + 2k\pi) = M(t)$ (\cref{def:g11:trigo:winding}).
\end{proof}

\begin{example}\label{ex:g11:trigo:identity}
अगर $\sin t = \frac35$ और $ t \in \intcc{\frac\pi2}{\pi}$ (दूसरा
चतुर्थांश), फिर $\cos^2 t = 1 - \frac{9}{25} = \frac{16}{25}$, इसलिए
$\cos t = \pm\frac45$; दूसरे चतुर्थांश में \omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज} नकारात्मक है,
इसलिए $\cos t = -\frac45$.
\end{example}

हृदय से जानने योग्य मूल्य:
\begin{center}
\begin{tabular}{c|ccccc}
$ t $ & $0$ & $\dfrac{\pi}{6}$ & $\dfrac{\pi}{4}$ & $\dfrac{\pi}{3}$ & $\dfrac{\pi}{2}$\\[6pt]
\hline\rule{0pt}{14pt}%
$\cos t $ & $1$ & $\dfrac{\sqrt3}{2}$ & $\dfrac{\sqrt2}{2}$ & $\dfrac12$ & $0$\\[6pt]
$\sin t $ & $0$ & $\dfrac12$ & $\dfrac{\sqrt2}{2}$ & $\dfrac{\sqrt3}{2}$ & $1$
\end{tabular}
\end{center}

\begin{remark}
एक स्मृति सहायता: जीवाओं पढ़ना
$\frac{\sqrt0}{2}, \frac{\sqrt1}{2}, \frac{\sqrt2}{2}, \frac{\sqrt3}{2},
\frac{\sqrt4}{2}$, और यह कोसाइन क्या वही सूची उलट दी गई है?
\end{remark}

\section{संबद्ध कोण}

\begin{proposition}[संबद्ध कोण]\label{prop:g11:trigo:associated}
\index{संबद्ध कोण}
सभी के लिए $ t \in \R $:
\[
\begin{aligned}
\cos(-t) &= \cos t, & \sin(-t) &= -\sin t,\\
\cos(\pi - t) &= -\cos t, & \sin(\pi - t) &= \sin t,\\
\cos(\pi + t) &= -\cos t, & \sin(\pi + t) &= -\sin t,\\
\cos\left(\tfrac{\pi}{2} - t\right) &= \sin t, &
\sin\left(\tfrac{\pi}{2} - t\right) &= \cos t .
\end{aligned}
\]
\end{proposition}

\begin{proof}
प्रत्येक रेखा वृत्त की समरूपता व्यक्त करती है।

$ M(-t)$ का प्रतिबिम्ब है $ M(t)$ में $ x $-अक्ष: \omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज} है
अपरिवर्तित, \omterm{def:g10:coordgeom:system}{तालमेल} संकेत बदलता है.

$ M(\pi - t)$ का प्रतिबिम्ब है $ M(t)$ में $ y $-अक्ष: चलना
से पीछे की ओर $\pi $ द्वारा $ t $ चलने के विपरीत (क्षैतिज रूप से) भूमि
से आगे $0$ द्वारा $ t $. \omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज} परिवर्तन चिह्न, \omterm{def:g10:coordgeom:system}{तालमेल} है
अपरिवर्तित.

$ M(\pi + t)$ एकदम विपरीत है $ M(t)$ (आधा मोड़): दोनों
\omterm{def:g10:coordgeom:system}{COORDINATES} संकेत बदलें.

$ M\!\left(\frac\pi2 - t\right)$ का प्रतिबिम्ब है $ M(t)$ में
विकर्ण रेखा $ y = x $, जो दोनों की अदला-बदली करता है \omterm{def:g10:coordgeom:system}{COORDINATES}.
\end{proof}

$\cos\frac{2\pi}{3} = \cos\left(\pi - \frac\pi3\right)
= -\cos\frac\pi3 = -\frac12$, और
$\sin\frac{7\pi}{6} = \sin\left(\pi + \frac\pi6\right)
= -\sin\frac\pi6 = -\frac12$. पाँच मानों की तालिका, द्वारा \omterm{def:g10:algebra:expand}{विस्तार}ित
समरूपता, पूरे वृत्त को कवर करती है।
\end{example}

\section{त्रिकोणमितीय समीकरणों को हल करना}

\begin{theorem}[समीकरण $\cos t = \cos a $ और $\sin t = \sin a $]
\label{thm:g11:trigo:equations}
के लिए \omterm{def:g10:numbers:sets}{वास्तविक संख्या} $ t $ और $ a $:
\[
\cos t = \cos a \iff t = a + 2k\pi \ \text{or}\ t = -a + 2k\pi
\quad (k \in \Z),
\]
\[
\sin t = \sin a \iff t = a + 2k\pi \ \text{or}\ t = \pi - a + 2k\pi
\quad (k \in \Z).
\]
\end{theorem}

\begin{proof}
यूनिट सर्कल के दो बिंदु समान हैं \omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज} ठीक उसी समय जब वे हों
में एक दूसरे के बराबर या प्रतिबिंब $ x $-अक्ष; द्वारा
\cref{prop:g11:trigo:associated}, का प्रतिबिंब $ M(a)$ है $ M(-a)$. इसलिए
$\cos t = \cos a $ मतलब $ M(t) = M(a)$ या $ M(t) = M(-a)$, \omterm{def:g10:stats:mean}{अर्थ}ात\
$ t = \pm a $ के गुणक तक $2\pi $. इसी प्रकार, बराबर \omterm{def:g10:coordgeom:system}{तालमेल} मतलब
में समान बिंदु या प्रतिबिंब $ y $-अक्ष, और का प्रतिबिंब $ M(a)$
है $ M(\pi - a)$.
\end{proof}

\begin{method}[हल $\cos t = c $ एक अंतराल पर]\label{met:g11:trigo:solve}
\begin{enumerate}
  \item खोजो \emph{एक} कोण $ a $ साथ $\cos a = c $, की तालिका से
        ज्ञात मूल्य.
  \item सामान्य समाधान लिखिए $ t = \pm a + 2k\pi $
        (\cref{thm:g11:trigo:equations}).
  \item चुने \omterm{def:g10:numbers:sets}{पूर्णांकों} $ k $ वह भूमि अंदर अनुरोधित है \omterm{def:g10:numbers:interval}{अंतराल},
        और समाधान सूचीबद्ध करें.
\end{enumerate}
इसी तरह आगे बढ़ें $\sin t = c $ साथ $ t = a + 2k\pi $ या
$ t = \pi - a + 2k\pi $.
\end{method}

\begin{example}\label{ex:g11:trigo:solve}
हल करना $\cos t = \frac12$ पर $\intoc{-\pi}{\pi}$. का एक समाधान
$\cos a = \frac12$ है $ a = \frac{\pi}{3}$. सामान्य समाधान हैं
$ t = \frac\pi3 + 2k\pi $ और $ t = -\frac\pi3 + 2k\pi $. अंदर
$\intoc{-\pi}{\pi}$, केवल $ k = 0$ योगदान देता है:
$ t \in \left\{-\frac\pi3,\ \frac\pi3\right\}$.

हल करना $\sin t = -\frac{\sqrt2}{2}$ पर $\intco{0}{2\pi}$. एक कोण है
$ a = -\frac\pi4$; समाधान हैं $ t = -\frac\pi4 + 2k\pi $ और
$ t = \pi + \frac\pi4 + 2k\pi = \frac{5\pi}{4} + 2k\pi $. में
$\intco{0}{2\pi}$: $ t = \frac{7\pi}{4}$ (से $ k = 1$) और
$ t = \frac{5\pi}{4}$.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g11:trigo:1}
में बदलो रेडियंस: $30^\circ $, $45^\circ $, $120^\circ $, $270^\circ $.
डिग्री में कनवर्ट करें: $\frac{\pi}{5}$, $\frac{5\pi}{6}$, $\frac{7\pi}{4}$,
$\frac{2\pi}{9}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:trigo:2}
इकाई वृत्त पर बिंदुओं को रखें $ M(t)$ के लिए
\[
t = \frac{2\pi}{3},\quad
t = -\frac{\pi}{4},\quad
t = \frac{17\pi}{6},\quad
t = -\frac{7\pi}{2},
\]
प्रत्येक को एक मान में घटाने के बाद $\intoc{-\pi}{\pi}$ सापेक्ष $2\pi $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:trigo:3}
तालिका और का उपयोग करना \omterm{prop:g11:trigo:associated}{संबद्ध कोण}, का सटीक मान दें
\[
\cos\frac{3\pi}{4}, \quad
\sin\frac{5\pi}{6}, \quad
\cos\left(-\frac{\pi}{3}\right), \quad
\sin\frac{4\pi}{3}, \quad
\cos\frac{11\pi}{6} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:trigo:4}
दिया गया $\cos t = \frac{5}{13}$ साथ $ t \in \intoo{-\frac\pi2}{0}$, गणना करें
$\sin t $ बिल्कुल।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:trigo:5}
किसी भी वास्तविक के लिए सरल बनाएं $ x $:
\[
A = \cos(\pi + x) + \cos(-x) + \sin\left(\frac\pi2 - x\right) - \cos x,
\qquad
B = \sin(\pi - x) + \sin(\pi + x) + \sin(-x) .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:trigo:6}
पर समाधान करें $\intoc{-\pi}{\pi}$:
\[
\cos t = \frac{\sqrt3}{2}, \qquad
\sin t = \frac12, \qquad
\cos t = -1 .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:trigo:7}
पर समाधान करें $\intco{0}{2\pi}$:
\[
\sin t = -\frac{\sqrt3}{2}, \qquad
\cos t = 0, \qquad
2\sin t + 1 = 0 .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:trigo:8}
पर समाधान करें $\R $  \omterm{def:g10:algebra:equation}{समीकरण} $\cos 2t = \cos t $. (उपयोग
\cref{thm:g11:trigo:equations} साथ $ a = t $, और दोनों परिवारों पर चर्चा करें
समाधान के.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:trigo:9}
इसे वास्तविक रूप से दिखाओ $ x $,
\[
\bigl(\cos x + \sin x\bigr)^2 + \bigl(\cos x - \sin x\bigr)^2 = 2 .
\]
\end{exercise}

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:trigo:11}
पर समाधान करें $\intoc{-\pi}{\pi}$ असमानता
$\cos t \leq \frac12$. (यूनिट सर्कल को स्केच करें, उस क्षेत्र को चिह्नित करें जहां
\omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज} \omterm{def:g10:functions:extrema}{अधिकतम} है $\frac12$, और पढ़ें \omterm{def:g10:numbers:interval}{अंतराल} के मूल्यों का
$ t $.)
\end{exercise}

\section{समस्या: फ़ेरिस व्हील और ज्वार}

\begin{problem}[{सप्ताहांत समस्या --- रोल्ड दूरी के रूप में रेडियन, और प्रत्येक आवधिक की मास्टर घड़ी के रूप में इकाई वृत्त घटना}]
\label{pb:g11:trigo:1}
एक फ़ेरिस व्हील आपको एक घेरे में ले जाता है; ज्वार वहन करता है
बंदरगाह का पानी ऊपर और नीचे एक ही गणितीय वृत्त। हर
आवधिक घटना --- पहिये, ज्वार, ध्वनि, ऋतु --- पढ़ता है
इसका शेड्यूल इस अध्याय के यूनिट सर्कल से अलग है
(\cref{def:g11:trigo:cossin}), और यह \omterm{def:g10:algebra:equation}{समीकरण} का
\cref{met:g11:trigo:solve} सही समय बताओ. यह समस्या
परिवर्तित करता है, घुमाता है, मॉडल बनाता है और हल करता है --- और रास्ते में क्यों तय करता है
गणितज्ञ कोणों को मापते हैं रेडियंस.

\medskip
\noindent\textbf{Part I --- रेडियंस, the rolled distance.}
\begin{enumerate}
  \item में बदलो रेडियंस: $30^\circ $, $135^\circ $,
        $300^\circ $; और डिग्री के लिए: $\frac{3\pi}{4}$,
        $\frac{5\pi}{6}$ (\cref{met:g11:trigo:convert}).
  \item का पूरा बिंदु रेडियंस: कोण का एक चाप $ t $ (में
        रेडियंस) त्रि\omterm{def:g11:trigo:cossin}{ज्या} के एक वृत्त पर $ r $ बिल्कुल लंबाई है
        $ r\,t $. एक कोण किस चाप का होता है $2$ रेडियंस ए पर काटें
        त्रि\omterm{def:g11:trigo:cossin}{ज्या} का वृत्त $3$ एम? (नहीं $\pi $ कहीं भी --- यानी
        बात.)
  \item त्रि\omterm{def:g11:trigo:cossin}{ज्या} का एक पहिया $30$ सेमी रोल $1$ किमी बिना
        फिसल रहा है. कितने के माध्यम से \emph{रेडियंस} क्या यह बदल गया है,
        और वह कितने पूर्ण मोड़ हैं?
  \item यूनिट सर्कल पर रखें और सटीक मूल्यांकन करें:
        $\cos\frac{2\pi}{3}$, $\sin\left(-\frac{\pi}{4}\right)$,
        और $\cos\frac{19\pi}{6}$ (मॉड्यूलो कम करें $2\pi $ पहला;
        \cref{prop:g11:trigo:associated}).
  \item एक कोण $ t \in \intoo{\frac\pi2}{\pi}$ संतुष्ट
        $\sin t = \frac35$. का उपयोग करते हुए
        $\cos^2 t + \sin^2 t = 1$ और चतुर्थांश, खोजें
        $\cos t $ और $\tan t $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The Ferris wheel.}
एक विशाल पहिये की त्रि\omterm{def:g11:trigo:cossin}{ज्या} होती है $60$ मी, इसका हब $65$ के ऊपर मी
जमीन, और एक-एक बार मुड़ता है $30$ मिनट। आप पर बोर्ड करें
समय पर निम्नतम बिंदु $ t = 0$ (मिनटों में), इसलिए आपकी ऊंचाई आज्ञा का पालन करती है
\[
h(t) = 65 - 60 \cos\!\left(\frac{\pi t}{15}\right).
\]
\begin{enumerate}[resume]
  \item सूत्र का औचित्य सिद्ध करें: बाद में बने कोण की जाँच करें $ t $
        मिनट, ऊंचाई पर $ t = 0$, और माइनस समझाएं
        संकेत.
  \item अपनी ऊंचाई की गणना करें $ t = 7.5$, $ t = 15$ और
        $ t = 20$ मिनट (सटीक मान)।
  \item आप किस समय पर हैं \emph{पहला} पर $95$ एम? (हल करना
        $ h(t) = 95$ साथ \cref{met:g11:trigo:solve}.)
  \item जिस दौरान \omterm{def:g10:numbers:interval}{अंतराल} सवारी में आप कम से कम हैं
        $95$ मी ऊँचा --- और कुल कितने समय तक? (तुलना करना
        \cref{exo:g11:trigo:11}.)
  \item ऊंचाई कहां बदलती है \emph{सबसे तेज़}? तुलना करें
        पहले मिनट के दौरान चढ़ें
        ($ h(1) - h(0)$) मिनटों के बीच चढ़ाई के साथ $7$ और
        $8$, और अवलोकन तैयार करें (वह उपकरण जो बनाता है
        यह सटीक --- है यौगिक साइनसॉइड का --- आता है
        कक्षा 12 में)।
  \item इंजीनियर की बारी: एक पहिये की ऊंचाई का फॉर्मूला डिज़ाइन करें
        \omterm{def:g10:functions:extrema}{अधिकतम} ऊंचाई के साथ $40$ मी, \omterm{def:g10:functions:extrema}{न्यूनतम} ऊँचाई $4$ मी और
        अवधि $12$ मिनट, नीचे बोर्डिंग
        $ t = 0$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The tide and the circle's
symmetries.}
एक बंदरगाह में, पानी की गहराई मीटर में, $ t $ घंटों बाद
आधी रात, है
\[
d(t) = 7 + 3 \sin\!\left(\frac{\pi t}{6}\right).
\]
\begin{enumerate}[resume]
  \item \omterm{def:g10:functions:extrema}{अधिकतम} और \omterm{def:g10:functions:extrema}{न्यूनतम} गहराई, वह समय बताएं जिस पर
        वे पहले घटित होते हैं, और ज्वार की अवधि।
  \item एक मालवाहक जहाज की जरूरत है $8.5$ पानी का मी. हल करना
        $ d(t) \geq 8.5$ पर $\intco{0}{12}$: किस विंडो के दौरान
        क्या यह प्रवेश कर सकता है, और खिड़की कितनी लंबी है? कब होता है
        अगली विंडो खुली?
  \item दो संबद्ध-कोण सर्वसमिकाओं को सीधे पुनः प्राप्त करें
        वृत्त समरूपता --- $\sin(\pi - t) = \sin t $ (दर्पण
        ऊर्ध्वाधर अक्ष के पार) और $\cos(\pi + t) = -\cos t $
        (आधा-मोड़) --- और \omterm{def:g10:proba:operations}{पूरक} पहचान बताएं
        $\cos\left(\frac\pi2 - t\right) = \sin t $.
  \item समता (\cref{pb:g11:func:1} शब्दावली): वर्गीकृत करें
        $\sin $, $\cos $, और $ t \mapsto \sin t + t^3$ सम या के रूप में
        \omterm{def:g11:func:parity}{विषम}, वृत्त से एक-पंक्ति प्रमाण के साथ।
  \item पर समाधान करें $\intco{0}{2\pi}$:
        $2\sin^2 t - \sin t - 1 = 0$. (एक द्विघात में
        भेष बदलना: इसे इस रूप में प्रस्तुत करना \cref{pb:g11:quad:1}, फिर
        वृत्त पढ़ें.)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The mathematician's angle.}
\begin{enumerate}[resume]
  \item कितनी डिग्री होती है $1$ \omterm{def:g11:trigo:radian}{कांति}? फिर एक कैलकुलेटर जाल:
        जो बड़ा है, $\sin(1)$ (\omterm{def:g11:trigo:radian}{कांति} मोड) या
        $\sin(1^\circ)$ --- और मोटे तौर पर किस कारक से? नैतिक
        कैलकुलेटर सेटिंग्स के लिए?
  \item हल करना $\cos t = \sin t $ पर $\intco{0}{2\pi}$, का उपयोग करते हुए
        वृत्त (कहां हैं \omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज} और \omterm{def:g10:coordgeom:system}{तालमेल} बराबर?)।
  \item छोटे कोणों के लिए, चाप और उसके \omterm{def:g11:trigo:cossin}{ज्या} लगभग मेल खाता है:
        तुलना करें $\sin(0.1)$ साथ $0.1$ (पांच दशमलव), और
        सापेक्ष त्रुटि दें. यह ``छोटा-कोण
        \omterm{def:g10:numbers:approx}{सन्निकटन}''जहाजों और पेंडुलम को चलाता है; प्रमेय
        इसके पीछे ($\frac{\sin t}{t} \to 1$) का एक सितारा है
        ग्रेड-12 सीमा अध्याय।
  \item समापन --- मास्टर घड़ी के रूप में यूनिट सर्कल, एक वाक्य
        प्रति आइटम: रेडियंस कोणों को ईमानदार संख्याओं में बदलें (चाप
        $=$ RADIUS $\times $ कोण); \omterm{def:g11:trigo:cossin}{कोज्या} और \omterm{def:g11:trigo:cossin}{ज्या} हैं
        घड़ी की सुई की छाया और ऊंचाई; वृत्त की समरूपता
        पहचान हैं; सुलझाने $\cos t = c$ बार पढ़ता है
        घड़ी से बाहर; और प्रत्येक आवधिक घटना --- पहिया,
        ज्वार, ध्वनि --- क्या यह घड़ी पोशाक पहने हुए है।
\end{enumerate}
\end{problem}
