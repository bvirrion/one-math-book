\chapter{वर्णनात्मक आँकड़े}\label{ch:g11:stat}

सांख्यिकी संख्याओं की सूची को कुछ सार्थक सारांशों में संपीड़ित करती है:
जहां डेटा बैठता है (\omterm{def:g11:stat:mean}{अर्थ}, \omterm{def:g11:stat:median}{MEDIAN}), और यह कितना फैला हुआ है (चतुर्थक,
\omterm{def:g11:stat:variance}{झगड़ा}, \omterm{def:g11:stat:variance}{मानक विचलन}). यह अध्याय सब कुछ एक पर आधारित करता है
चल रहा उदाहरण. यादृच्छिक चरों के लिए वही सारांश पुनः प्रकट होते हैं
\cref{ch:g11:prob}, और उनके संभाव्य कानून \cref{ch:g12:sums}.

पूरे अध्याय में, हमारा चल रहा डेटासेट अंक (में से) है $20$) का
का एक वर्ग $12$ छात्र:
\[
8,\ 9,\ 10,\ 10,\ 11,\ 12,\ 12,\ 12,\ 13,\ 14,\ 16,\ 17 .
\]

\section{माध्य और माध्यिका}

\begin{definition}[अर्थ]\label{def:g11:stat:mean}
 \emph{अर्थ}\index{अर्थ} मूल्यों का $ x_1, x_2, \dots, x_n $ है
\[
\bar x = \frac{x_1 + x_2 + \dots + x_n}{n}.
\]
यदि मान $ x_i $ के साथ प्रकट होता है \omterm{def:g10:stats:series}{आवृत्ति} $ n_i $ (साथ
$ n_1 + \dots + n_k = n $), तब
$\bar x = \frac{n_1 x_1 + \dots + n_k x_k}{n}$.
\end{definition}

\begin{definition}[मंझला]\label{def:g11:stat:median}
ए \emph{MEDIAN}\index{MEDIAN} क्रमबद्ध डेटासेट का एक मान है जो विभाजित होता है
इसे आधे में: कम से कम आधे मान हैं $\leq $ यह, और कम से कम आधे हैं
$\geq $ यह। व्यवहार में, क्रमबद्ध करें $ n $ मूल्य; अगर $ n $ \omterm{def:g11:func:parity}{विषम} है, माध्यिका है
\omterm{prop:g10:coordgeom:midpoint}{मध्य} मान; अगर $ n $ सम है, ले लो \omterm{prop:g10:coordgeom:midpoint}{मध्य} दो \omterm{prop:g10:coordgeom:midpoint}{मध्य} में से
मूल्य.
\end{definition}

\begin{example}\label{ex:g11:stat:meanmedian}
चल रहे डेटासेट के लिए, योग है $144$, इसलिए $\bar x = \frac{144}{12} =
12$. दो \omterm{prop:g10:coordgeom:midpoint}{मध्य} मान ($6$ वें और $7$ क्रमबद्ध सूची का वां भाग) दोनों हैं
$12$, इतना \omterm{def:g11:stat:median}{MEDIAN} है $12$ बहुत। \omterm{def:g11:stat:mean}{अर्थ} और \omterm{def:g11:stat:median}{MEDIAN} मेल खाने की जरूरत नहीं: द
\omterm{def:g11:stat:median}{MEDIAN} ध्यान न दी \emph{कितनी दूर} चरम मूल्य हैं, \omterm{def:g11:stat:mean}{अर्थ} नहीं करता.
शीर्ष चिह्न को बदलना $17$ द्वारा $20$ को स्थानांतरित करता है \omterm{def:g11:stat:mean}{अर्थ} को $12.25$ लेकिन छोड़ देता है
द \omterm{def:g11:stat:median}{MEDIAN} पर $12$.
\end{example}

\section{चतुर्थक}

\begin{definition}[चतुर्थक, अंतरचतुर्थक सीमा]\label{def:g11:stat:quartiles}
 \emph{प्रथम चतुर्थक}\index{प्रथम चतुर्थक} $ Q_1$ ऐसा सबसे छोटा मान है
कि कम से कम एक चौथाई डेटा है $\leq Q_1$; \emph{तीसरा
चतुर्थक} $ Q_3$ सबसे छोटा मान है जैसे कि कम से कम तीन चौथाई
डेटा है $\leq Q_3$. \emph{अंतरचतुर्थक सीमा}\index{अंतरचतुर्थक सीमा} है $ Q_3 - Q_1$: द
डेटा के \omterm{prop:g10:coordgeom:midpoint}{मध्य} भाग की चौड़ाई.
\end{definition}

\begin{example}
$ n = 12$: एक चौथाई है $3$, इसलिए $ Q_1$ है $3$ तीसरा क्रमबद्ध मान:
$Q_1 = 10$. तीन चौथाई है $9$, इसलिए $ Q_3$ है $9$ वां मान:
$Q_3 = 13$. \omterm{def:g11:stat:quartiles}{अंतरचतुर्थक सीमा} है $3$: कक्षा का \omterm{prop:g10:coordgeom:midpoint}{मध्य} भाग
भीतर बैठता है $3$ निशान.
\end{example}

\section{विचरण और मानक विचलन}

\begin{definition}[विचरण, मानक विचलन]\label{def:g11:stat:variance}
 \emph{झगड़ा}\index{झगड़ा} डेटासेट का है \omterm{def:g11:stat:mean}{अर्थ} का
से वर्ग विचलन \omterm{def:g11:stat:mean}{अर्थ}:
\[
V = \frac{1}{n}\sum_{i=1}^{n} (x_i - \bar x)^2,
\]
और यह \emph{मानक विचलन}\index{मानक विचलन} है
$\sigma = \sqrt{V}$, डेटा के समान इकाई में व्यक्त किया गया है।
\end{definition}

\begin{proposition}[शॉर्टकट फॉर्मूला]\label{prop:g11:stat:shortcut}
 \omterm{def:g11:stat:variance}{झगड़ा} है \omterm{def:g11:stat:mean}{अर्थ} वर्गों का घटा वर्ग का \omterm{def:g11:stat:mean}{अर्थ}:
\[
V = \frac{1}{n}\sum_{i=1}^{n} x_i^2 \;-\; \bar x^{\,2}.
\]
\end{proposition}

\begin{proof}
प्रत्येक वर्ग विचलन का \omterm{def:g10:algebra:expand}{विस्तार} करें:
$(x_i - \bar x)^2 = x_i^2 - 2\bar x\, x_i + \bar x^2$. संक्षेप और
द्वारा विभाजित करना $ n $:
\[
V = \frac1n \sum x_i^2 - 2\bar x \cdot \underbrace{\frac1n \sum
x_i}_{=\ \bar x} + \frac1n \cdot n\bar x^2
= \frac1n\sum x_i^2 - 2\bar x^2 + \bar x^2
= \frac1n\sum x_i^2 - \bar x^2 .
\]
यादृच्छिक चरों के लिए भी यही पहचान है
\cref{prop:g11:prob:konig}.
\end{proof}

\begin{example}\label{ex:g11:stat:variance}
चल रहे डेटासेट के लिए, से विचलन $\bar x = 12$ हैं $-4$,
$-3$, $-2$, $-2$, $-1$, $0$, $0$, $0$, $1$, $2$, $4$, $5$; उनके वर्ग
का योग
\[
16 + 9 + 4 + 4 + 1 + 0 + 0 + 0 + 1 + 4 + 16 + 25 = 80 .
\]
इस तरह
\[
V = \frac{80}{12} = \frac{20}{3} \approx 6.67,
\qquad
\sigma = \sqrt{20/3} \approx 2.6 \text{ marks}.
\]
मोटे तौर पर कहें तो एक विशिष्ट चिह्न बैठता है $2.6$ से दूर निशान
\omterm{def:g11:stat:mean}{अर्थ}.
\end{example}

\begin{proposition}[इकाइयों का परिवर्तन]\label{prop:g11:stat:affine}
यदि प्रत्येक मान को रूपांतरित किया जाता है $ y_i = a x_i + b $, तब
\[
\bar y = a \bar x + b,
\qquad
V_y = a^2\, V_x,
\qquad
\sigma_y = \abs{a}\,\sigma_x .
\]
\end{proposition}

\begin{proof}
$\bar y = \frac1n \sum (a x_i + b) = a\left(\frac1n\sum x_i\right) +
\frac1n \cdot nb = a\bar x + b $. तब
$ y_i - \bar y = a(x_i - \bar x)$: प्रत्येक विचलन को इसके द्वारा मापा जाता है $ a $
(स्थानान्तरित करना $ b $ रद्द करता है), इसलिए प्रत्येक वर्ग विचलन को इसके द्वारा मापा जाता है $ a^2$,
और उनका भी ऐसा ही है \omterm{def:g11:stat:mean}{अर्थ}. वर्गमूल लेने पर प्राप्त होता है
$\sigma_y = \abs a\,\sigma_x $.
\end{proof}

\begin{example}
तापमान डिग्री सेल्सियस में दर्ज किया गया \omterm{def:g11:stat:mean}{अर्थ} $20$ और \omterm{def:g11:stat:variance}{मानक
विचलन} $3$ द्वारा फ़ारेनहाइट में परिवर्तित करें $ F = 1.8\,C + 32$: द \omterm{def:g11:stat:mean}{अर्थ}
बन जाता है $1.8 \times 20 + 32 = 68$, द \omterm{def:g11:stat:variance}{मानक विचलन}
$1.8 \times 3 = 5.4$. किसी डेटासेट को स्थानांतरित करने से उसका प्रसार नहीं बदलता है;
इसे पुनः स्केल करने से प्रसार पुनः स्केल हो जाता है।
\end{example}

\begin{method}[डेटासेट का सारांश और तुलना करना]\label{met:g11:stat:compare}
दो डेटासेट की तुलना करने के लिए, उन्हें एक साथ रखें: \omterm{def:g11:stat:mean}{अर्थ} (या \omterm{def:g11:stat:median}{MEDIAN}) के लिए
\emph{पद}, \omterm{def:g11:stat:variance}{मानक विचलन} (या \omterm{def:g11:stat:quartiles}{अंतरचतुर्थक सीमा}) के लिए
\emph{फैलाना}. \omterm{def:g10:vectors:sum}{जोड़}ी (\omterm{def:g11:stat:median}{MEDIAN}, \omterm{def:g11:stat:quartiles}{अंतरचतुर्थक सीमा}) के लिए मजबूत है
अत्यधिक मूल्य; \omterm{def:g10:vectors:sum}{जोड़}ी (\omterm{def:g11:stat:mean}{अर्थ}, \omterm{def:g11:stat:variance}{मानक विचलन}) प्रत्येक मान का उपयोग करता है और
फ़ीड \omterm{def:g10:proba:distribution}{संभावना} लिखित।
\end{method}

\begin{example}
प्रत्येक में दो तीरंदाज निशाना लगाते हैं $10$ तीर; औसत अंक प्राप्त करता है $8.5$ दोनों के लिए, लेकिन
द \omterm{def:g11:stat:variance}{मानक विचलन} हैं $0.5$ और $2.1$. औसतन समान स्तर --- लेकिन
पहला तीरंदाज़ कहीं अधिक सुसंगत है।
\end{example}

\section{समस्या: z-स्कोर, एक सार्वभौमिक शासक}

\begin{problem}[{सप्ताहांत समस्या --- मानक विचलन की तुलना संतरे के साथ सेब, चेबीशेव की असमानता बदल जाती है $\sigma $ एक में गारंटी, और माध्य अपना पदक जीतता है}]
\label{pb:g11:stat:1}
है $15/20$ कठिन परीक्षा से बेहतर $13/20$ एक आसान पर? एक है $23$ छोटी संख्याओं की सूची में कोई दुर्घटना या कोई बाहरी घटना? कच्चा
संख्याएँ नहीं कह सकतीं --- लेकिन दाएँ से विभाजित होती हैं \omterm{def:g11:stat:variance}{मानक
विचलन} वे सभी एक ही भाषा बोलते हैं \emph{\omterm{pb:g11:stat:1}{z के स्कोर}}. यह
समस्या पर काबू पा लेता है \omterm{def:g11:stat:variance}{झगड़ा} मशीनरी
(\cref{def:g11:stat:variance},
\cref{prop:g11:stat:shortcut}), सार्वभौमिक शासक का निर्माण करता है, और
उल्लेखनीय असमानता को साबित करता है जो बदल जाती है $\sigma $ ए से
गारंटी में वर्णनात्मक संख्या.

\medskip
\noindent\textbf{Part I --- \omterm{def:g11:stat:variance}{झगड़ा} mechanics.}
\begin{enumerate}
  \item डेटासेट के लिए $4, 8, 6, 5, 2$: गणना करें \omterm{def:g11:stat:mean}{अर्थ}, फिर
        द \omterm{def:g11:stat:variance}{झगड़ा} फिर, सीधे परिभाषा से
        $\sigma $.
  \item पुनः गणना करें \omterm{def:g11:stat:variance}{झगड़ा} शॉर्टकट फ़ॉर्मूले के साथ
        (\cref{prop:g11:stat:shortcut}) --- \omterm{def:g11:stat:mean}{अर्थ} वर्गों का
        का शून्य से वर्ग \omterm{def:g11:stat:mean}{अर्थ} --- और समझौते की जांच करें.
  \item आवृत्तियों के साथ: निशान $8$, $10$, $14$ प्राप्तकर्ता
        $3$, $5$, $2$ छात्र. गणना करना \omterm{def:g11:stat:mean}{अर्थ}, \omterm{def:g11:stat:variance}{झगड़ा} और
        $\sigma $ (दो दशमलव).
  \item तापमान है \omterm{def:g11:stat:mean}{अर्थ} $20$ और $\sigma = 3$ (डिग्री
        सेल्सियस). दे दो \omterm{def:g11:stat:mean}{अर्थ} और $\sigma $ फ़ारेनहाइट में
        ($ F = 1.8\,C + 32$; \cref{prop:g11:stat:affine}).
  \item अभी उपयोग किए गए दो नियमों को \omterm{def:g11:vect:det}{सिद्ध} करें: एक स्थिरांक जोड़ना $ b $ को
        प्रत्येक मान बदलता है \omterm{def:g11:stat:mean}{अर्थ} द्वारा $ b $ और चला जाता है $\sigma $
        अपरिवर्तित; से गुणा करना $ a $ को गुणा करता है \omterm{def:g11:stat:mean}{अर्थ} द्वारा $ a $
        और $\sigma $ द्वारा $\abs a $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The universal ruler.}
 \emph{z के स्कोर}\index{z के स्कोर} एक मूल्य का $ x $ के डेटासेट में
\omterm{def:g11:stat:mean}{अर्थ} $\bar x $ और \omterm{def:g11:stat:variance}{मानक विचलन} $\sigma $ है
$ z = \dfrac{x - \bar x}{\sigma}$: की संख्या \omterm{def:g11:stat:variance}{मानक विचलन} पृथक करना $ x $ से \omterm{def:g11:stat:mean}{अर्थ}.
\begin{enumerate}[resume]
  \item परीक्षा ए है \omterm{def:g11:stat:mean}{अर्थ} $12$ और $\sigma = 2$; परीक्षा बी है \omterm{def:g11:stat:mean}{अर्थ}
        $8$ और $\sigma = 4$. ऐलिस ने स्कोर किया $15$ ए, बॉब पर $13$
        बी पर दोनों की गणना करें z-स्कोर: जिसका प्रदर्शन कायम है
        भीड़ से और ऊपर?
  \item पांचों की गणना करें z-स्कोर प्रश्न 1 के डेटासेट का,
        और सत्यापित करें कि उनके पास है \omterm{def:g11:stat:mean}{अर्थ} $0$ और \omterm{def:g11:stat:variance}{मानक
        विचलन} $1$. प्रश्न 5 से स्पष्ट करें कि क्यों
        मानकीकरण \emph{हमेशा} का उत्पादन \omterm{def:g11:stat:mean}{अर्थ} $0$ और
        $\sigma = 1$.
  \item एक योग्यता परीक्षण है \omterm{def:g11:stat:mean}{अर्थ} $100$ और $\sigma = 15$. क्या
        कच्चा स्कोर है $ z = 2$? क्या है \omterm{pb:g11:stat:1}{z के स्कोर} का $76$?
  \item एक बाल रोग विशेषज्ञ एक बच्चे का वजन इस प्रकार रिकॉर्ड करता है
        ``$ z = -2.5$''. बताएं कि इसका क्या मतलब है और विकास क्यों है
        में ट्रैक किया गया z-स्कोर बच्चे के वजन के बजाय किलोग्राम
        उम्र
  \item का बाह्य \cref{exo:g11:stat:10} (डेटासेट
        $1, 2, 2, 3, 3, 3, 4, 4, 5, 23$): गणना करें \omterm{def:g11:stat:mean}{अर्थ},
        $\sigma $, और यह \omterm{pb:g11:stat:1}{z के स्कोर} का $23$. क्लासिक अलार्म
        दहलीज है $\abs z \geq 3$: निर्णय?
  \item पुन: परिकलन करें $\sigma $ \emph{बिना} मूल्य $23$, और यह \omterm{pb:g11:stat:1}{z के स्कोर} वह $23$ साफ के खिलाफ होगा
        डेटा का \omterm{def:g11:stat:mean}{अर्थ} और $\sigma $. बाहरी ने क्या किया
        वही शासक इसे मापना चाहता था, और क्या करता है
        \cref{exo:g11:stat:10} प्रतिकार के रूप में सुझाव दें?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Chebyshev's guarantee.}
\begin{enumerate}[resume]
  \item बायनेयम को \omterm{def:g11:vect:det}{सिद्ध} करें\'ई--ए के लिए चेबीशेव असमानता
        डेटासेट: यदि एक अनुपात $ p $ के मूल्य संतुष्ट करते हैं
        $\abs{x - \bar x} \geq k\sigma $, फिर, केवल रखते हुए
        उन शर्तों में \omterm{def:g11:stat:variance}{झगड़ा} \omterm{def:g10:vectors:sum}{जोड़},
        \[
        \sigma^2 \geq p \cdot (k\sigma)^2,
        \qquad\text{hence}\qquad
        p \leq \frac{1}{k^2} .
        \]
  \item असमानता अनुपात के बारे में क्या गारंटी देती है?
        किसी भी डेटासेट से परे $2\sigma $ उसके जैसा \omterm{def:g11:stat:mean}{अर्थ}? आगे
        $3\sigma $? जाँचें $2\sigma $ डेटासेट पर बयान
        प्रश्न 1 का.
  \item एक फैक्ट्री बोल्ट का उत्पादन करती है \omterm{def:g11:stat:mean}{अर्थ} $50$ मिमी और
        $\sigma = 0.1$ मिमी. चेबीशेव किस बारे में गारंटी देता है?
        बाहर बोल्ट का अनुपात
        $\intcc{49.7}{50.3}$? (वास्तविक उत्पादन कहीं बेहतर होता है
        --- ग्रेड 12 का घंटी वक्र इसे सतर्क कर देता है
        $\frac19$ प्रतिशत के एक अंश में।)
  \item दो फंड दोनों वापस आते हैं $5\,\%$ प्रति वर्ष औसतन, साथ
        $\sigma = 2\,\%$ फंड ए के लिए और $\sigma = 12\,\%$ के लिए
        फंड बी. प्रत्येक फंड के लिए, चेबीशेव किस बारे में कहता है
        रिटर्न के साथ वर्षों का संभाव्यता-जैसा अनुपात
        नीचे $-19\,\%$? क्या है $\sigma $ का वित्तीय नाम?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The \omterm{def:g11:stat:mean}{अर्थ}'s medal, and a warning.}
\begin{enumerate}[resume]
  \item होने देना $f(c)$ हो \omterm{def:g11:stat:mean}{अर्थ} वर्ग विचलन का
        $(x_i - c)^2$ एक मनमाना \omterm{pb:g11:quad:1}{केंद्र} से $ c $. \omterm{def:g11:vect:det}{सिद्ध} करो
        पहचान
        \[
        f(c) = \sigma^2 + (\bar x - c)^2 ,
        \]
        और निष्कर्ष निकालें: \omterm{def:g11:stat:mean}{अर्थ} का अनोखा मिनिमाइज़र है \omterm{def:g11:stat:mean}{अर्थ}
        वर्ग विचलन --- इसका पदक, मेल खाता है MEDIANका
        में निरपेक्ष विचलन को \omterm{def:g10:functions:extrema}{न्यूनतम} करना
        \cref{pb:g10:stats:1}.
  \item प्रश्न 1 के डेटासेट पर पहचान सत्यापित करें
        $ c = 6$: गणना करें $ f(6)$ सीधे और सूत्र के माध्यम से.
  \item समान सारांश, भिन्न डेटा: इसकी जाँच करें
        $ P = \{3, 5, 7, 9, 11\}$ और $ Q = \{1, 7, 7, 7, 13\}$
        वही साझा करें \omterm{def:g11:stat:mean}{अर्थ} लेकिन वैसा नहीं \omterm{def:g11:stat:variance}{झगड़ा}; फिर
        प्रपत्र का एक डेटासेट बनाएं
        $\{7 - a, 7, 7, 7, 7 + a\}$ बिल्कुल के साथ \omterm{def:g11:stat:mean}{अर्थ} और
        \omterm{def:g11:stat:variance}{झगड़ा} का $ P $ --- फिर भी एक आकार अपना है।
  \item साबित करें कि $\sigma = 0$ सभी मूल्यों को समान बनाता है। (क्या
        क्या वर्गों के शून्य योग का प्रत्येक पद होना चाहिए?)
  \item समापन---सांख्यिकीविद् की सीढ़ी, एक पंक्ति प्रति
        पायदान: \omterm{pb:g11:quad:1}{केंद्र} (\omterm{def:g11:stat:mean}{अर्थ}, \omterm{def:g11:stat:median}{MEDIAN}), फैलता है ($\sigma $, आईक्यूआर),
        सार्वभौमिक शासक ($ z$), गारंटी (चेबीशेव),
        और प्रश्न 18 की चेतावनी: सारांश संक्षिप्त करें,
        और ग्रेड के घंटी वक्र के साथ संपीड़न खो जाता है
        12 बारी का इंतजार कर रहे हैं z-स्कोर सटीक संभावनाओं में.
\end{enumerate}
\end{problem}
