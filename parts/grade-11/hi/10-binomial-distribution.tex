\chapter{द्विपद वितरण}\label{ch:g11:binom}

एक ही हां/नहीं प्रयोग को स्वतंत्र रूप से कई बार दोहराएं और गिनती करें
सफलताएँ: परिणामी \omterm{def:g11:prob:rv}{वितरण} --- द \emph{द्विपद} --- है
सबसे महत्वपूर्ण असतत \omterm{def:g11:prob:rv}{वितरण} के सभी। यह अध्याय इसका निर्माण करता है
के साथ \omterm{met:g10:proba:tree}{पेड़} और पथ गणना; पथ गणना के लिए बंद सूत्र (के साथ)
फैक्टोरियल) गिनती के उपकरणों के साथ आता है \cref{ch:g12:comb}, और यह
\omterm{def:g11:prob:rv}{वितरण} में दोबारा गौर किया गया है \cref{ch:g12:randvar}.

\section{बर्नौली परीक्षण}

\begin{definition}[बर्नौली परीक्षण]\label{def:g11:binom:bernoulli}
ए \emph{बर्नौली परीक्षण}\index{बर्नौली परीक्षण} के साथ एक प्रयोग है
बिल्कुल दो परिणाम: \emph{सफलता}, साथ \omterm{def:g10:proba:distribution}{संभावना} $p $, और
\emph{असफलता}, साथ \omterm{def:g10:proba:distribution}{संभावना} $1 - p $. \omterm{def:g11:prob:rv}{अनियमित परिवर्तनशील वस्तु} $ X $ के बराबर $1$ सफलता पर और $0$ असफल होने पर इसका पालन करने को कहा जाता है
\emph{Bernoulli \omterm{def:g11:prob:rv}{वितरण}} $\mathcal B(p)$; तब
\[
\E(X) = p,
\qquad
\V(X) = p(1 - p).
\]
\end{definition}

\begin{proof}[दो सूत्रों का प्रमाण]
$\E(X) = p \times 1 + (1-p) \times 0 = p $; और तबसे $ X^2 = X $ (दोनों $0$
और $1$ उनके अपने वर्ग हैं), $\E(X^2) = p $, तो द्वारा
\cref{prop:g11:prob:konig}, $\V(X) = p - p^2 = p(1-p)$.
\end{proof}

\begin{definition}[बार-बार स्वतंत्र परीक्षण]\label{def:g11:binom:repeated}
दोहराते हुए ए \omterm{def:g11:binom:bernoulli}{बर्नौली परीक्षण} $ n $ टाइम्स \emph{स्वतंत्र रूप से} का \omterm{def:g11:stat:mean}{अर्थ} है: द
प्रत्येक परीक्षण के परिणाम का दूसरों पर कोई प्रभाव नहीं पड़ता है, और \omterm{def:g10:proba:distribution}{संभावना}
किसी भी पूर्ण का \omterm{def:g11:seq:sequence}{अनुक्रम} परिणामों का है \emph{उत्पाद} का
के संगत पथ पर संभाव्यताएँ \omterm{met:g10:proba:tree}{पेड़} --- $ p $ प्रत्येक के लिए
सफलता, $1 - p $ प्रत्येक विफलता के लिए.
\end{definition}

\begin{example}\label{ex:g11:binom:threetrials}
सफलता के साथ तीन स्वतंत्र परीक्षण \omterm{def:g10:proba:distribution}{संभावना} $ p $. \omterm{def:g11:seq:sequence}{अनुक्रम} एसएफएस
(सफलता, असफलता, सफलता) है \omterm{def:g10:proba:distribution}{संभावना} $ p(1-p)p = p^2(1-p)$ --- और
ऐसा ही होता है \emph{प्रत्येक} \omterm{def:g11:seq:sequence}{अनुक्रम} बिल्कुल दो सफलताओं के साथ, चाहे कुछ भी हो
पद: केवल \emph{संख्या} एस और एफ के मामले।
\end{example}

\section{पथ गणना और द्विपद गुणांक}

\begin{definition}[द्विपद गुणांक]\label{def:g11:binom:coefficient}
में \omterm{met:g10:proba:tree}{पेड़} का $ n $ स्वतंत्र परीक्षण, \emph{द्विपद
गुणांक}\index{द्विपद
गुणांक} $\binom{n}{k}$ (पढ़ें ``$ n $
चुनना $ k $'') सटीक रूप से शामिल पथों की संख्या है $ k $ सफलताएँ
\end{definition}

\begin{example}
$\binom{3}{2} = 3$: पथ एसएसएफ, एसएफएस, एफएसएस। वैसे ही $\binom{3}{0} = 1$
(पथ एफएफएफ), $\binom{3}{1} = 3$ और $\binom{3}{3} = 1$. द्वारा
सम्मेलन और द्वारा \omterm{met:g10:proba:tree}{पेड़}, $\binom n0 = \binom nn = 1$ हरएक के लिए $ n $.
\end{example}

\begin{proposition}[पास्कल का नियम]\label{prop:g11:binom:pascal}
\index{पास्कल का त्रिकोण}
के लिए $1 \leq k \leq n - 1$:
\[
\binom{n}{k} = \binom{n-1}{k-1} + \binom{n-1}{k}.
\]
\end{proposition}

\begin{proof}
के पथों को क्रमबद्ध करें $ n $-परीक्षण \omterm{met:g10:proba:tree}{पेड़} साथ $ k $ के अनुसार सफलताएँ
उनके \emph{अंतिम} परीक्षण। जो सफलता में समाप्त होते हैं वे a से प्राप्त होते हैं
पहले का पथ $ n-1$ के साथ परीक्षण $ k - 1$ सफलताएँ: वहाँ हैं
$\binom{n-1}{k-1}$ उनमें से. जिनका अंत असफलता में होता है वे एक मार्ग का \omterm{def:g10:algebra:expand}{विस्तार} करते हैं
$ k $ प्रथम के बीच सफलताएँ $ n-1$ परीक्षण: $\binom{n-1}{k}$ उनमें से.
प्रत्येक पथ बिल्कुल दो प्रकार का होता है।
\end{proof}

पास्कल का नियम पंक्ति दर पंक्ति गुणांक उत्पन्न करता है --- प्रत्येक प्रविष्टि है
इसके ऊपर दोनों का योग:
\[
\begin{array}{ccccccccccc}
&&&&&1&&&&&\\
&&&&1&&1&&&&\\
&&&1&&2&&1&&&\\
&&1&&3&&3&&1&&\\
&1&&4&&6&&4&&1&\\
1&&5&&10&&10&&5&&1
\end{array}
\]

\begin{remark}
एक बंद सूत्र, $\binom nk = \frac{n!}{k!(n-k)!}$, साथ में ए
गिनती का व्यवस्थित \omterm{def:g11:vect:det}{सिद्ध}ांत स्थापित किया गया है \cref{ch:g12:comb}. पर
इस स्तर पर, \omterm{prop:g11:binom:pascal}{पास्कल का त्रिकोण} हमारे लिए आवश्यक प्रत्येक गुणांक की गणना करता है।
\end{remark}

\section{द्विपद वितरण}

\begin{theorem}[द्विपद वितरण]\label{thm:g11:binom:binomial}
होने देना $ X $ सफलताओं को गिनें $ n $ स्वतंत्र \omterm{def:g11:binom:bernoulli}{बर्नौली परीक्षण} का
पैरामीटर $ p $. तब $ X $ का अनुसरण करता है \emph{द्विपद
वितरण}\index{द्विपद
वितरण} $\mathcal B(n, p)$:
\[
\P(X = k) = \binom{n}{k}\, p^k (1-p)^{n-k},
\qquad k = 0, 1, \dots, n .
\]
\end{theorem}

\begin{proof}
 \omterm{def:g11:prob:model}{आयोजन} $ X = k $ सटीक रूप से सभी पथों का संग्रह है $ k $
सफलताएँ प्रत्येक ऐसे पथ है \omterm{def:g10:proba:distribution}{संभावना} $ p^k(1-p)^{n-k}$: उत्पाद
रास्ते में शामिल हैं $ k $ कारकों $ p $ और $ n - k $ कारकों $1-p $, में
कुछ आदेश (\cref{def:g11:binom:repeated}). वहाँ हैं $\binom nk $ ऐसे
पथ (\cref{def:g11:binom:coefficient}), और उनकी संभावनाएँ जुड़ जाती हैं।
\end{proof}

\begin{example}\label{ex:g11:binom:quiz}
एक प्रश्नोत्तरी है $5$ स्वतंत्र प्रश्न, प्रत्येक के साथ $4$ विकल्प; एक छात्र
उत्तर यादृच्छिक होते हैं, इसलिए प्रत्येक प्रश्न सफल होता है $ p = \frac14$. जो नंबर $ X $ सही उत्तर इस प्रकार हैं $\mathcal B\left(5, \frac14\right)$,
और, पंक्ति का उपयोग करके $5$ का \omterm{prop:g11:binom:pascal}{पास्कल का त्रिकोण}:
\[
\P(X = 2) = \binom52 \left(\frac14\right)^2\left(\frac34\right)^3
= 10 \times \frac{1}{16} \times \frac{27}{64}
= \frac{270}{1024} \approx 0.26 .
\]
 \omterm{def:g10:proba:distribution}{संभावना} कम से कम एक सही उत्तर का उपयोग करता है \omterm{def:g10:proba:operations}{पूरक}:
$\P(X \geq 1) = 1 - \P(X = 0) = 1 - \left(\frac34\right)^5 \approx
0.76$.
\end{example}

\begin{proposition}[अपेक्षा और भिन्नता]\label{prop:g11:binom:expectation}
अगर $ X \sim \mathcal B(n, p)$:
\[
\E(X) = np,
\qquad
\V(X) = np(1-p).
\]
\end{proposition}

\begin{proof}[औचित्य]
लिखो $ X = X_1 + X_2 + \dots + X_n $, कहाँ $ X_i $ के बराबर होती है $1$ यदि
$ i $-वां परीक्षण सफल: प्रत्येक $ X_i $ का एक बर्नौली वैरिएबल है \omterm{def:g11:prob:expectation}{अपेक्षा}
$ p $ (\cref{def:g11:binom:bernoulli}). औसत \omterm{def:g10:vectors:sum}{जोड़}ें --- योग करें $ n $
योगदान देता है $\E(X) = np $. वह \emph{प्रसरण} के लिए भी \omterm{def:g10:vectors:sum}{जोड़}ें
स्वतंत्र चर सत्य है लेकिन अधिक नाजुक है: \omterm{def:g11:prob:variance}{झगड़ा} सूत्र है
\emph{इस स्तर पर स्वीकार किया गया} और में साबित हुआ \cref{ch:g12:sums}.
\end{proof}

\begin{method}[द्विपद स्थिति को पहचानना]\label{met:g11:binom:recognize}
लिखने से पहले $ X \sim \mathcal B(n, p)$, तीन सामग्रियों की जाँच करें: a
\emph{निर्धारित अंक} $ n $ परीक्षणों का, पहले से निर्णय लिया गया; प्रत्येक परीक्षण में है
\emph{दो परिणाम} साथ \emph{वही} सफलता \omterm{def:g10:proba:distribution}{संभावना} $ p $; द
परीक्षण हैं \emph{स्वतंत्र} (प्रतिस्थापन के साथ, या अलग उपकरणों के साथ)।
एक छोटी आबादी से प्रतिस्थापन के बिना चित्रण करना है \emph{नहीं}
द्विपद --- द \omterm{def:g10:proba:distribution}{संभावना} प्रत्येक ड्रा पर परिवर्तन
(\cref{exo:g11:prob:6}).
\end{method}

\section{नमूनाकरण: क्या अवलोकन आश्चर्यजनक है?}

द्विपद \omterm{def:g11:prob:rv}{वितरण} एक बहुत ही व्यावहारिक प्रश्न का उत्तर देता है: \emph{यदि
सफलता \omterm{def:g10:proba:distribution}{संभावना} वास्तव में है $ p $, सफलताओं की गिनती क्या है
प्रशंसनीय?}

\begin{example}\label{ex:g11:binom:decision}
एक मशीन को उत्पादन करना चाहिए $10\%$ अधिक से अधिक दोषपूर्ण वस्तुएँ। ए में
का बैच $10$ सामान, $4$ ख़राब हैं. दुर्भाग्य या टूटी हुई मशीन? यदि
मशीन ठीक है, खराबी की संख्या इस प्रकार है
$\mathcal B(10, 0.1)$, और
\[
\P(X \geq 4) = 1 - \P(X \leq 3) \approx 1 - 0.987 = 0.013 :
\]
में एक अवसर के बारे में $80$. एक का अवलोकन करना \omterm{def:g11:prob:model}{आयोजन} यह असंभावित एक मजबूत है
संकेत---एक \emph{अस्वीकृत} यह परिकल्पना कि मशीन अभी भी काम करती है
पर $10\%$, जबकि यह ध्यान में रखते हुए कि निर्णय गलत हो सकता है
\omterm{def:g10:proba:distribution}{संभावना} के बारे में $0.013$.
\end{example}

\begin{method}[द्विपद मॉडल से निर्णय नियम]\label{met:g11:binom:decision}
किसी प्रेक्षित संख्या का निर्णय करना $ k $ एक परिकल्पना के विरुद्ध सफलताओं का
$ X \sim \mathcal B(n, p)$: गणना करें \omterm{def:g10:proba:distribution}{संभावना}, परिकल्पना के तहत,
एक परिणाम का \emph{कम से कम अतिवादी के रूप में} जैसा $ k $. कि अगर \omterm{def:g10:proba:distribution}{संभावना} है
बहुत छोटा (एक सामान्य परंपरा: नीचे $5\%$), परिकल्पना को अस्वीकार करें;
अन्यथा अवलोकन इसके अनुकूल है। दहलीज एक है
विकल्प, प्रमेय नहीं --- आँकड़े जोखिम और उपयोगकर्ता की मात्रा निर्धारित करते हैं
इसे स्वीकार करता है.
\end{method}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g11:binom:1}
बढ़ाना \omterm{prop:g11:binom:pascal}{पास्कल का त्रिकोण} पंक्ति $7$, और का मान दें
$\binom62$, $\binom{7}{3}$ और $\binom74$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:binom:2}
एक निष्पक्ष पासा फेंका गया है $4$ समय; $ X $ छक्के गिनता है. उसे उचित ठहराओ
$ X \sim \mathcal B\left(4, \frac16\right)$ और गणना करें $\P(X = 0)$,
$\P(X = 1)$ और $\P(X \geq 2)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:binom:3}
निम्नलिखित में से कौन सा द्विपद है? औचित्य।
\begin{enumerate}
  \item में प्रमुखों की संख्या $20$ एक उचित सिक्के का उछाल.
  \item में इक्के की संख्या $5$ कार्ड एक डेक से बांटे गए।
  \item अगले सप्ताह बरसात के दिनों की संख्या, यदि प्रत्येक दिन बरसात हो
        \omterm{def:g10:proba:distribution}{संभावना} $0.3$ स्वतंत्र रूप से।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:binom:4}
$ X \sim \mathcal B(50, 0.2)$. देना $\E(X)$, $\V(X)$ और $\sigma(X)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:binom:5}
एक तीरंदाज लक्ष्य पर प्रहार करता है \omterm{def:g10:proba:distribution}{संभावना} $0.7$ प्रत्येक शॉट पर,
स्वतंत्र रूप से। में $6$ शॉट्स, गणना करें \omterm{def:g10:proba:distribution}{संभावना} बिल्कुल $4$ हिट,
और कम से कम $5$ हिट.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:binom:6}
एक सच्चा/झूठा परीक्षण होता है $8$ प्रश्न; एक विद्यार्थी प्रत्येक उत्तर का अनुमान लगाता है।
क्या है \omterm{def:g10:proba:distribution}{संभावना} उत्तीर्ण होने का (कम से कम) $6$ सही उत्तर)?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:binom:7}
प्रत्येक खरीदे गए अनाज के डिब्बे में मूर्ति ए या मूर्ति बी होती है
\omterm{def:g10:proba:distribution}{संभावना} $\frac12$ प्रत्येक, स्वतंत्र रूप से। एक संग्राहक खरीदता है $5$ बक्से.
की गणना करें \omterm{def:g10:proba:distribution}{संभावना} कि संग्राहक को कम से कम एक मूर्ति मिले
प्रत्येक प्रकार. (\omterm{def:g10:proba:operations}{पूरक}: सभी ए या सभी बी.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:binom:8}
एक बास्केटबॉल खिलाड़ी फ्री थ्रो पर स्कोर करता है \omterm{def:g10:proba:distribution}{संभावना} $ p $,
स्वतंत्र रूप से। होने देना $ X \sim \mathcal B(3, p)$ में अंकों की संख्या हो
तीन फेंके. अभिव्यक्त करना $\P(X = 3)$ और $\P(X \geq 1)$ जैसा कार्य का $ p $,
और किसके लिए खोजें $ p $  \omterm{def:g10:proba:distribution}{संभावना} तीनों का स्कोर बराबर
$\frac{27}{64}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:binom:9}
एक उचित सिक्के को कितनी बार उछाला जाना चाहिए? \omterm{def:g10:proba:distribution}{संभावना} पाने का
कम से कम एक सिर से अधिक होना $0.99$? (\omterm{def:g10:proba:operations}{पूरक}, फिर लगातार प्रयास करें
के मान $ n $.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:binom:10}
पास्कल के नियम का उपयोग करना (\cref{prop:g11:binom:pascal}) और
$\binom n0 = \binom nn = 1$, साबित करें कि प्रत्येक पंक्ति की प्रविष्टियाँ
\omterm{prop:g11:binom:pascal}{पास्कल का त्रिकोण} का योग $2^n $: दोनों पक्षों को सभी की गिनती के रूप में व्याख्या करें
के पथ \omterm{met:g10:proba:tree}{पेड़}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:binom:11}
एक राजनेता का दावा है $60\%$ अनुमोदन। बेतरतीब ढंग से \omterm{def:g10:proba:sample}{नमूना} का $10$ लोग,
केवल $3$ मंज़ूरी देना।
\begin{enumerate}
  \item दावे के तहत, क्या \omterm{def:g11:prob:rv}{वितरण} संख्या करता है $ X $ का
        में अनुमोदन \omterm{def:g10:proba:sample}{नमूना} अनुसरण करना? गणना करना $\P(X \leq 3)$.
  \item के निर्णय नियम का उपयोग करना \cref{met:g11:binom:decision} के साथ
        $5\%$ सीमा, क्या अवलोकन दावे के अनुकूल है?
\end{enumerate}
\end{exercise}

\section{समस्या: गैल्टन बोर्ड}

\begin{problem}[{सप्ताहांत समस्या --- गेंदें, खूंटियां और पास्कल त्रिकोण: घंटी का आकार कैसे पैदा होता है, प्लेऑफ़ श्रृंखला क्यों अनुकूल है मजबूत टीम, और कब बेईमानी का रोना है}]
\label{pb:g11:binom:1}
खूंटियों की एक जाली के माध्यम से एक हजार गेंदें गिराएं, प्रत्येक गेंद उछलती है
निष्पक्ष बाएँ या दाएँ सिक्के को उछालें, और नीचे दिए गए स्लॉट एक में भर जाएँ
चिकनी, सममित घंटी --- हर बार। मशीन को कहा जाता है
गैल्टन बोर्ड, और इसका गणित बिल्कुल इसी अध्याय का है
द्विपद \omterm{def:g11:prob:rv}{वितरण} (\cref{thm:g11:binom:binomial}). यह
समस्या त्रिकोण बनाती है, बोर्ड चलाती है, रेफरी ए
सात में से सर्वश्रेष्ठ शृंखला, और वहीं समाप्त होती है जहां द्विपद अपनी कमाई करता है
वेतन: यह तय करना कि कब किसी अवलोकन से हमें संदेह होना चाहिए
दावा.

\medskip
\noindent\textbf{Part I --- The triangle.}
\begin{enumerate}
  \item निर्माण \omterm{prop:g11:binom:pascal}{पास्कल का त्रिकोण} पंक्ति से नीचे $6$
        (\cref{prop:g11:binom:pascal}). बताएं और समझाएं
        समरूपता $\binom nk = \binom{n}{n-k}$ एक वाक्य में
        (चुनना $ k $ वस्तुओं के समान ही है\,\dots).
  \item पंक्तियों पर सत्यापित करें $4$ और $5$ प्रत्येक पंक्ति का योग है $2^n $,
        और इसे साबित करें: सभी क्या करते हैं $\binom nk $ एक साथ
        गिनती?
  \item पास्कल का नियम पुनः प्राप्त करें
        $\binom{n+1}{k} = \binom nk + \binom{n}{k-1}$ द्वारा
        समिति का तर्क: एक प्रतिष्ठित व्यक्ति को ठीक करें और
        उस व्यक्ति के भाग्य के अनुसार समितियों का विभाजन करें।
  \item गणना करना $\binom73$ दो बार: त्रिभुज से, और से
        भाज्य सूत्र.
  \item सीढ़ी की पहचान की जाँच करें
        $\binom22 + \binom32 + \binom42 + \binom52 =
        \binom63$, और इसे पास्कल के नियम को कैस्केडिंग करके समझाएं
        से $\binom63$ नीचे की ओर.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The board.}
एक गेंद गिरती है $ n $ खूंटियों की पंक्तियाँ; प्रत्येक खूंटी पर यह उछलता है
बाएँ या दाएँ के साथ \omterm{def:g10:proba:distribution}{संभावना} $\frac12$, स्वतंत्र रूप से। संख्या
स्लॉट $0$ को $ n $ सही बाउंस की गिनती से.
\begin{enumerate}[resume]
  \item की जाँच सूची सहित समझाइये
        \cref{met:g11:binom:recognize}, स्लॉट संख्या क्यों
        द्विपद का अनुसरण करता है \omterm{def:g11:prob:rv}{वितरण}
        $\mathcal B\!\left(n, \frac12\right)$.
  \item एक छोटे बोर्ड के लिए ($ n = 4$): पांच स्लॉट दें
        सम्भावनाएँ कौन सा स्लॉट सबसे अधिक भीड़भाड़ वाला है?
  \item अब $ n = 10$ और $1\,024$ गेंदें: क्या हैं
        \emph{अपेक्षित} गेंद \omterm{prop:g10:coordgeom:midpoint}{मध्य} स्लॉट में, स्लॉट में गिना जाता है
        $7$, और प्रत्येक किनारे के स्लॉट में? ढेर के आकार का वर्णन करें.
  \item के लिए $ X \sim \mathcal B\!\left(10, \frac12\right)$:
        गणना करें $\E(X)$, $ V(X)$ और $\sigma $
        (\cref{prop:g11:binom:expectation}); फिर गणना करें
        भीतर अपेक्षित गेंदों का अनुपात $2\sigma $ का
        \omterm{pb:g11:quad:1}{केंद्र} (स्लॉट $2$ को $8$) और चेबीशेव के साथ तुलना करें
        से गारंटी \cref{pb:g11:stat:1}.
  \item एक झुका हुआ बोर्ड दाईं ओर उछलता है \omterm{def:g10:proba:distribution}{संभावना} $0.6$:
        देना $\E $, $ V $ और $\sigma $ के लिए $ n = 10$, और वर्णन करें
        ढेर का क्या होता है.
  \item एक या दो वाक्यों में: क्या, बोर्ड के डिज़ाइन में,
        घंटी के आकार का निर्माण करता है --- और इतने सारे लोग ऐसा क्यों करते हैं
        वास्तविक दुनिया की मात्राएँ (ऊँचाई, माप त्रुटियाँ)
        उसी तरह ढेर लगाओ? (दोनों के पीछे गहरा प्रमेय है
        \omterm{pb:g11:quad:1}{केंद्र}ीय सीमा प्रमेय, विश्वविद्यालय का शिखर
        खंड' \omterm{def:g10:proba:distribution}{संभावना} अवधि।)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Best of seven.}
दो टीमें एक शृंखला खेलती हैं: पहली से $4$ जीत खिताब लेता है;
खेल स्वतंत्र हैं.
\begin{enumerate}[resume]
  \item बराबर टीमें ($ p = \frac12$): गणना करें \omterm{def:g10:proba:distribution}{संभावना}
        कि श्रृंखला एक स्वीप में समाप्त होती है (बिल्कुल $4$ खेल)।
  \item की गणना करें \omterm{def:g10:proba:distribution}{संभावना} कि शृंखला पूरी चले
        $7$ खेल (अंत के बाद स्कोर क्या होना चाहिए $6$?).
  \item को पूर्ण करो \omterm{def:g11:prob:rv}{वितरण} श्रृंखला की लंबाई का
        ($4$, $5$, $6$ या $7$ खेल) समान टीमों के लिए, और
        अपेक्षित लंबाई की गणना करें। कौन सी लंबाई सबसे अधिक है
        \omterm{def:g10:proba:distribution}{संभावना}?
  \item अब प्रत्येक खेल में एक टीम जीतती है $ p = 0.6$. इसकी गणना करें
        \omterm{def:g10:proba:distribution}{संभावना} शृंखला पर कब्ज़ा करने का (जीतने का)। $4$, $5$, $6$
        या $7$: प्रत्येक मामले में टीम आखिरी गेम जीतती है और
        $3$ पिछले वाले में से)। सीरीज ने क्या किया
        प्रति गेम बढ़त?
  \item एकल फाइनल से तुलना करें ($60\,\%$) और एक सर्वश्रेष्ठ-3
        (इसकी गणना करें)। श्रृंखला की लंबाई का सामान्य प्रभाव बताएं
        भाग्य के विरुद्ध कौशल पर --- और लीग लंबे समय को क्यों पसंद करते हैं
        फाइनल.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- When to cry foul.}
\begin{enumerate}[resume]
  \item एक सिक्का उछाला जाता है $100$ समय और शो $62$ सिर. के लिए
        एक उचित सिक्का, दे दो $\E $, $\sigma $, और यह \omterm{pb:g11:stat:1}{z के स्कोर}
        (\cref{pb:g11:stat:1}) अवलोकन का. फैसला
        के अंतर्गत $2\sigma $ सम्मेलन?
  \item एक सप्लायर का दावा है $2\,\%$ दोषपूर्ण भाग. एक बैच में
        का $50$ आप पाते हैं $3$ दोषपूर्ण. गणना करना
        $\P(X \geq 3)$ दावे के तहत
        ($ X \sim \mathcal B(50, 0.02)$; के माध्यम से जाना
        $\P(X = 0), \P(X = 1), \P(X = 2)$). पर चिंताजनक
        $5\,\%$ सीमा (\cref{met:g11:binom:decision},
        \cref{exo:g11:binom:11})?
  \item लॉटरी की दृढ़ता: प्रत्येक टिकट (कुछ) जीतता है
        \omterm{def:g10:proba:distribution}{संभावना} $\frac{1}{1000}$. की गणना करें \omterm{def:g10:proba:distribution}{संभावना}
        कम से कम एक जीत के साथ $1\,000$ टिकट. उत्तर
        ($\approx 63\,\%$, नहीं $100\,\%$!) एक प्रसिद्ध छुपाता है
        स्थिरांक: गणना करें $0.999^{1000}$ और नंबर रखें
        $0.368$ 12वीं कक्षा के लिए मन में।
  \item समापन--- द्विपद का चित्र: पहचान
        चेकलिस्ट (निश्चित) $ n $, स्वतंत्रता, स्थिर $ p $);
        \omterm{prop:g11:binom:pascal}{पास्कल का त्रिकोण} इसकी मेज के रूप में; घंटी अपने आकार के अनुसार;
        $ np $ और $ np(1-p)$ इसके कम्पास के रूप में; और इसके दो वारिस
        कक्षा 12 में प्रतीक्षारत --- चिकनी घंटी वक्र और
        बड़ी संख्या का नियम. एक-एक वाक्य.
\end{enumerate}
\end{problem}
