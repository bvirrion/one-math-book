\chapter{समतल में अदिश गुणनफल}\label{ch:g11:scal}

 \omterm{def:g11:scal:dot}{अदिश उत्पाद} एक जोड़ता है \emph{संख्या} की एक \omterm{def:g10:vectors:sum}{जोड़}ी के लिए वैक्टर, और वह
संख्या ज्यामिति देखती है: यह लंबवत दिशाओं के लिए बिल्कुल गायब हो जाती है,
और यह कोणों और लंबाई को मापता है। यह कानून के पीछे का इंजन है
कोसाइन, \omterm{def:g10:algebra:equation}{समीकरण} वृत्तों की, और लंबवतता की गणना। वैसा ही
उपकरण, अंतरिक्ष में, का विषय है \cref{ch:g12:space}.

\section{परिभाषा और पहली गणना}

\begin{definition}[अदिश गुणनफल]\label{def:g11:scal:dot}
 \emph{अदिश उत्पाद}\index{अदिश उत्पाद} दोनों में से वैक्टर $\vec u $
और $\vec v $ संख्या है
\[
\vec u \cdot \vec v
= \tfrac12\left(\norm{\vec u + \vec v}^2 - \norm{\vec u}^2
- \norm{\vec v}^2\right),
\]
कहाँ $\norm{\vec u}$ लंबाई को दर्शाता है (\emph{आदर्श}\index{आदर्श}) का $\vec u $.
\end{definition}

\begin{theorem}[अदिश उत्पाद के चार फलक]\label{thm:g11:scal:faces}
होने देना $\vec u\,(x, y)$ और $\vec v\,(x', y')$ शून्येतर हो वैक्टर, और
$\theta $ उनके बीच का कोण. तब
\begin{enumerate}
  \item \emph{(\omterm{def:g10:coordgeom:system}{COORDINATES})} $\vec u \cdot \vec v = xx' + yy'$;
  \item \emph{(\omterm{def:g11:trigo:cossin}{कोज्या})} $\vec u \cdot \vec v =
        \norm{\vec u}\,\norm{\vec v}\cos\theta $;
  \item \emph{(प्रक्षेपण)} $\vec u \cdot \vec v = \vec u \cdot \vec v'$,
        कहाँ $\vec v'$ है \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल} का प्रक्षेपण $\vec v $ पर
        की दिशा $\vec u $; अगर $\vec u = \vect{OA}$ और
        $\vec v = \vect{OB}$, तब
        $\vec u \cdot \vec v = \overline{OA} \times \overline{OH}$, ए
        हस्ताक्षरित लंबाई का उत्पाद, जहां $ H $ का पैर है
        से लंबवत $ B $ लाइन तक $(OA)$;
  \item \emph{(मानदंड)} का परिभाषित सूत्र \cref{def:g11:scal:dot}.
\end{enumerate}
\end{theorem}

\begin{proof}[1 और 2 का प्रमाण]
\emph{1.} में \omterm{def:g10:coordgeom:system}{COORDINATES}, $\norm{\vec u}^2 = x^2 + y^2$ (पाइथागोरस) और
$\vec u + \vec v $ है \omterm{def:g10:coordgeom:system}{COORDINATES} $(x + x', y + y')$, इसलिए
\begin{align*}
2\,\vec u \cdot \vec v
&= (x + x')^2 + (y + y')^2 - (x^2 + y^2) - (x'^2 + y'^2)\\
&= 2xx' + 2yy',
\end{align*}
बाद का \omterm{def:g10:algebra:expand}{विस्तार} वर्ग और रद्द करना।

\emph{2.} चुने \omterm{def:g10:coordgeom:system}{निर्देशांक तरीका} ताकि $\vec u $ के साथ अंक
सकारात्मक $ x $-अक्ष: $\vec u\,(\norm{\vec u}, 0)$ और
$\vec v\,(\norm{\vec v}\cos\theta,\ \norm{\vec v}\sin\theta)$
(\cref{ch:g11:trigo}). फॉर्मूला 1 तब देता है
$\vec u \cdot \vec v = \norm{\vec u}\,\norm{\vec v}\cos\theta + 0$.
फॉर्मूला 3 उसी गणना को पढ़ता है: का प्रक्षेपण $\vec v $ पर $ x $-अक्ष ने लंबाई पर हस्ताक्षर किए हैं $\norm{\vec v}\cos\theta =
\overline{OH}$.
\end{proof}

\begin{example}\label{ex:g11:scal:compute}
$\vec u\,(3, 1)$ और $\vec v\,(2, -4)$:
$\vec u \cdot \vec v = 6 - 4 = 2$. मानदंड हैं
$\norm{\vec u} = \sqrt{10}$ और $\norm{\vec v} = \sqrt{20}$, इसलिए
$\cos\theta = \frac{2}{\sqrt{10}\sqrt{20}} = \frac{2}{10\sqrt2}
= \frac{\sqrt2}{10}$: कोण थोड़ा कम है $82^\circ $.
\end{example}

\begin{proposition}[बीजगणितीय नियम]\label{prop:g11:scal:rules}
सभी के लिए वैक्टर और सब $\lambda \in \R $:
\[
\vec u \cdot \vec v = \vec v \cdot \vec u, \qquad
\vec u \cdot (\vec v + \vec w) = \vec u \cdot \vec v + \vec u \cdot
\vec w, \qquad
(\lambda \vec u)\cdot \vec v = \lambda\,(\vec u \cdot \vec v),
\]
\[
\vec u \cdot \vec u = \norm{\vec u}^2,
\qquad
\norm{\vec u + \vec v}^2 = \norm{\vec u}^2 + 2\,\vec u\cdot\vec v +
\norm{\vec v}^2 .
\]
\end{proposition}

\begin{proof}
सभी समन्वय सूत्र का पालन करते हैं: उदाहरण के लिए
$\vec u \cdot (\vec v + \vec w) = x(x' + x'') + y(y' + y'') =
(xx' + yy') + (xx'' + yy'')$. अंतिम पहचान का \omterm{def:g10:algebra:expand}{विस्तार} होता है
$\norm{\vec u + \vec v}^2 = (\vec u + \vec v)\cdot(\vec u + \vec v)$
पहले तीन नियमों का उपयोग करते हुए --- \omterm{def:g11:scal:dot}{अदिश उत्पाद} बिल्कुल एक जैसा व्यवहार करता है
साधारण उत्पाद, इसलिए सामान्य बीजगणितीय सर्वसमिकाएँ लागू होती हैं वैक्टर.
\end{proof}

\section{ओर्थोगोनालिटी}

\begin{definition}[ऑर्थोगोनल वैक्टर]\label{def:g11:scal:orthogonal}
दो वैक्टर हैं \emph{ओर्थोगोनल}\index{ओर्थोगोनल}, लिखा हुआ
$\vec u \perp \vec v $, अगर $\vec u \cdot \vec v = 0$.
\end{definition}

\begin{remark}
शून्येतर के लिए वैक्टर, का सूत्र 2 \cref{thm:g11:scal:faces} दिखाता है
$\vec u \cdot \vec v = 0 \iff \cos\theta = 0 \iff $ दिशाएँ हैं
लंबवत. शून्य \omterm{def:g11:vect:vector}{वेक्टर} है \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल} सबकुछ में।
\end{remark}

\begin{example}
$\vec u\,(3, 2)$ और $\vec v\,(-4, 6)$: $\vec u \cdot \vec v = -12 + 12 =
0$: \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल}. सामान्य तौर पर, $\vec u\,(a, b)$ हमेशा से रहा है \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल} को
$\vec v\,(-b, a)$ --- एक चौथाई-मोड़ रोटेशन।
\end{example}

\section{कोसाइन का नियम}

\begin{theorem}[कोसाइन का नियम]\label{thm:g11:scal:cosines}
\index{कोसाइन का नियम}
एक त्रिकोण में $ ABC $ पक्षों के साथ $ a = BC $, $ b = CA $, $ c = AB $ और कोण
$\widehat A $ शिखर पर $ A $:
\[
a^2 = b^2 + c^2 - 2bc\,\cos\widehat A .
\]
\end{theorem}

\begin{proof}
लिखना $\vect{BC} = \vect{BA} + \vect{AC} = \vect{AC} - \vect{AB}$ और
के साथ \omterm{def:g10:algebra:expand}{विस्तार} करें \cref{prop:g11:scal:rules}:
\[
a^2 = \norm{\vect{BC}}^2
= \norm{\vect{AC}}^2 - 2\,\vect{AC}\cdot\vect{AB} + \norm{\vect{AB}}^2
= b^2 + c^2 - 2\,\vect{AB}\cdot\vect{AC}.
\]
से \omterm{def:g11:trigo:cossin}{कोज्या} सूत्र,
$\vect{AB}\cdot\vect{AC} = c\,b\,\cos\widehat A $.
\end{proof}

\begin{remark}
कब $\widehat A $ एक समकोण है, $\cos\widehat A = 0$ और का कानून
कोसाइन पाइथागोरस प्रमेय को कम करता है $ a^2 = b^2 + c^2$: यह है
पाइथागोरस गैर-समकोण के लिए एक सुधार शब्द के साथ।
\end{remark}

{\small का कानून कोसाइन: दो पक्षों को जानना $ b $, $ c $ और कोण
उनके बीच तीसरा पक्ष निर्धारित होता है $ a $.}
\end{omfigure}

\begin{example}\label{ex:g11:scal:cosines}
एक त्रिभुज की दो भुजाएँ मापी जाती हैं $ b = 5$ और $ c = 8$, एक कोण घेरना
$\widehat A = \frac\pi3$. तब
\[
a^2 = 25 + 64 - 2 \times 5 \times 8 \times \frac12 = 89 - 40 = 49,
\]
इसलिए $ a = 7$.
\end{example}

\section{अनुप्रयोग: वृत्त और लंबवत रेखाएँ}

\begin{proposition}[दिए गए व्यास वाला वृत्त]\label{prop:g11:scal:diameter}
एक बिंदु $ M $ व्यास के वृत्त पर स्थित है $[AB]$ अगर और केवल अगर
\[
\vect{MA} \cdot \vect{MB} = 0 .
\]
\end{proposition}

\begin{proof}
अगर $ M \neq A, B $, शर्त कोण कहती है $ M $ त्रिकोण में
$ AMB $ सही है, और एक खंड को समकोण के नीचे देखने वाले बिंदु हैं
बिल्कुल उस व्यास वाले वृत्त के। यहाँ इसका प्रत्यक्ष प्रमाण है:
चलो $ O $ हो \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[AB]$ और $ r = \frac{AB}{2}$. तब
$\vect{MA} = \vect{MO} + \vect{OA}$ और
$\vect{MB} = \vect{MO} + \vect{OB} = \vect{MO} - \vect{OA}$, इसलिए
\[
\vect{MA}\cdot\vect{MB}
= \norm{\vect{MO}}^2 - \norm{\vect{OA}}^2 = MO^2 - r^2,
\]
जो बिल्कुल कब गायब हो जाता है $ MO = r $, यानी\ कब $ M $ के घेरे पर है
\omterm{pb:g11:quad:1}{केंद्र} $ O $ और त्रि\omterm{def:g11:trigo:cossin}{ज्या} $ r $.
\end{proof}

\begin{proposition}[एक रेखा का सामान्य सदिश]\label{prop:g11:scal:normal}
 \omterm{def:g11:vect:vector}{वेक्टर} $\vec n\,(a, b)$ है \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल} प्रत्येक को \omterm{def:g11:vect:direction}{दिशा सदिश} का
लाइन $ d : ax + by + c = 0$; एक कहता है $\vec n $ एक है \emph{सामान्य
वेक्टर}\index{सामान्य
वेक्टर} का $ d $. दो रेखाएँ बिल्कुल लंबवत हैं
जब उनकी सामान्य (या दिशा) वैक्टर हैं \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल}.
\end{proposition}

\begin{proof}
ए \omterm{def:g11:vect:direction}{दिशा सदिश} का $ d $ है $\vec u\,(-b, a)$
(\cref{thm:g11:vect:cartesian}), और
$\vec n \cdot \vec u = a(-b) + b(a) = 0$.
\end{proof}

\begin{method}[सामान्य वैक्टर का उपयोग करना]\label{met:g11:scal:normal}
\begin{itemize}
  \item के माध्यम से रेखा $ A $ सामान्य के साथ $\vec n\,(a,b)$: लिखना
        $\vect{AM} \cdot \vec n = 0$, जिसका \omterm{def:g10:algebra:expand}{विस्तार} होता है $ ax + by + c = 0$
        साथ $ c $ द्वारा निर्धारित $ A $.
  \item की लम्बवतता $ d: ax+by+c = 0$ और $ d': a'x+b'y+c' = 0$:
        जाँच करना $ aa' + bb' = 0$.
\end{itemize}
\end{method}

\begin{example}\label{ex:g11:scal:perpline}
के माध्यम से लाइन $ P(2, -1)$ के लिए सीधा $ d : 3x - y + 4 = 0$: ए
\omterm{def:g11:vect:direction}{दिशा सदिश} का $ d $ है $(1, 3)$, जो के रूप में कार्य करता है \emph{सामान्य} \omterm{def:g11:vect:vector}{वेक्टर}
लम्बवत् का. \omterm{def:g10:algebra:equation}{समीकरण}: $ x + 3y + c = 0$ साथ
$2 - 3 + c = 0$, इसलिए $ c = 1$: लाइन है $ x + 3y + 1 = 0$.
\end{example}

\begin{example}[एक वृत्त का समीकरण]\label{ex:g11:scal:circle}
\omterm{pb:g11:quad:1}{केंद्र} का घेरा $\Omega(1, 2)$ और त्रि\omterm{def:g11:trigo:cossin}{ज्या} $3$ बिंदुओं का समूह है
$ M(x,y)$ साथ $\Omega M^2 = 9$:
\[
(x - 1)^2 + (y - 2)^2 = 9 .
\]
इसके विपरीत, $ x^2 + y^2 - 2x - 4y - 4 = 0$ को पूरा करते हुए पुनः लिखता है
वर्ग (\cref{ch:g11:quad}), जैसा $(x-1)^2 + (y-2)^2 = 9$: वही
वृत्त.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g11:scal:1}
गणना करना $\vec u \cdot \vec v $ के लिए
\[
\vec u\,(2, 5),\ \vec v\,(3, -1); \qquad
\vec u\,(1, -4),\ \vec v\,(8, 2); \qquad
\norm{\vec u} = 3,\ \norm{\vec v} = 4,\ \theta = \frac{\pi}{3} .
\]
\end{exercise}

$ ABC $ भुजा का एक समबाहु त्रिभुज है $6$. गणना करना

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:scal:3}
किस मूल्य के लिए $ t $ हैं $\vec u\,(3, t)$ और $\vec v\,(t - 8, 1)$
\omterm{def:g11:scal:orthogonal}{ओर्थोगोनल}? और किस मान के लिए $ t $ है $\vec u\,(t, 2)$ \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल}
अपने आप को शून्य से $\vec v\,(4, 2)$, यानी\ $\vec u \perp (\vec u - \vec v)$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:scal:4}
कोण की गणना करें $\theta $ बीच में $\vec u\,(1, 2)$ और $\vec v\,(3, 1)$
(देना $\cos\theta $ बिल्कुल, तो फिर $\theta $ निकटतम डिग्री तक)।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:scal:5}
एक त्रिभुज की भुजाएँ होती हैं $ b = 4$, $ c = 6$ और कोण

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:scal:6}

तीन भुजाओं की लंबाई और अपने उत्तर की जाँच के नियम से करें कोसाइन.

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:scal:9}
एक खोजें \omterm{def:g10:algebra:equation}{समीकरण} लाइन के माध्यम से $ P(1, 3)$ के लिए सीधा
$ d : 2x + y - 7 = 0$, और यह \omterm{def:g10:coordgeom:system}{COORDINATES} पैर का $ H $ का
लंबवत (द \omterm{def:g10:numbers:interunion}{चौराहा} दो पंक्तियों में से)। दूरी कम करें
से $ P $ को $ d $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:scal:10}
का उपयोग करते हुए $\norm{\vec u + \vec v}^2 = \norm{\vec u}^2 + 2\vec u\cdot\vec v +
\norm{\vec v}^2$ और इसके लिए एनालॉग $\vec u - \vec v $, \omterm{def:g11:vect:det}{सिद्ध} करो
\emph{समांतर चतुर्भुज पहचान}:
\[
\norm{\vec u + \vec v}^2 + \norm{\vec u - \vec v}^2
= 2\norm{\vec u}^2 + 2\norm{\vec v}^2 ,
\]
और इसे एक समांतर चतुर्भुज (विकर्ण बनाम भुजाएँ) में व्याख्या करें।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:scal:11}
होने देना $ A $ और $ B $ के साथ दो अंक हो $ AB = 4$, और $ I $  \omterm{prop:g10:coordgeom:midpoint}{मध्य} का
$[AB]$.
\begin{enumerate}
  \item प्रत्येक बिंदु के लिए इसे दिखाएँ $ M $:
        $\vect{MA}\cdot\vect{MB} = MI^2 - 4$.
        (डालना $ I $: $\vect{MA} = \vect{MI} + \vect{IA}$.)
  \item अंकों का सेट निकालें $ M $ ऐसा है कि
        $\vect{MA}\cdot\vect{MB} = 12$.
\end{enumerate}
\end{exercise}

\section{समस्या: अदिश गुणनफल के दो उपहार}

\begin{problem}[{सप्ताहांत समस्या---जोड़ सूत्र गिरते हैं एक डॉट उत्पाद से, दूसरे से एक रेखा की दूरी, और तीन ऊंचाईयां बोनस के रूप में मिलती हैं}]
\label{pb:g11:scal:1}
 \omterm{def:g11:scal:dot}{अदिश उत्पाद} बहीखाता पद्धति जैसा दिखता है --- गुणा करें, \omterm{def:g10:vectors:sum}{जोड़}ें ---
फिर भी यह लगभग नि:शुल्क, दो ख़ज़ाने सौंपता है
सदियों से चले आ रहे कोण-पीछा का विरोध किया: \emph{अतिरिक्त सूत्र}
त्रिकोणमिति (गणना) $\cos 15^\circ $ बिल्कुल!) और
\emph{एक बिंदु से एक रेखा की दूरी} एक साफ़ अंश में. जैसे
एक बिदाई उपहार यह साबित करता है कि प्रत्येक की तीन ऊँचाइयाँ
त्रिभुज एक बिंदु से होकर गुजरता है। के चारों मुख
\cref{thm:g11:scal:faces} ड्यूटी के लिए रिपोर्ट करेंगे.

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item के लिए $\vec u(3, 4)$ और $\vec v(-1, 2)$: गणना करें
        $\vec u \cdot \vec v $, दो मानदंड, और कोण
        के बीच वैक्टर (डिग्री के दसवें भाग तक)।
  \item खोजो $ k $ ताकि $(2, k)$ और $(3, -4)$ हैं \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल}
        (\cref{def:g11:scal:orthogonal}).
  \item भौतिकी चेहरा: एक स्लेज खींची जाती है $10$ एम बाय ए $50$ एन बल
        पर $60^\circ $ भूमि पर। कार्य की गणना करें
        $ W = \vec F \cdot \vec d $.
  \item का कानून कोसाइन (\cref{thm:g11:scal:cosines}): दो पक्ष
        का $5$ और $7$ का एक कोण संलग्न करें $60^\circ $; खोजें
        बिल्कुल तीसरा पक्ष.
  \item उलटा: एक त्रिभुज की भुजाएँ होती हैं $4$, $6$, $8$. की गणना करें
        \omterm{def:g11:trigo:cossin}{कोज्या} भुजा के विपरीत कोण का $8$, वर्गीकृत करें
        कोण, और इसे एक डिग्री के दसवें हिस्से तक दें।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- First gift: the addition
formulas.}
होने देना $\vec u = (\cos a, \sin a)$ और $\vec v = (\cos b, \sin b)$:
दो इकाई वैक्टर कोणों पर $ a $ और $ b $.
\begin{enumerate}[resume]
  \item गणना करना $\vec u \cdot \vec v $ दो बार --- एक बार साथ
        \omterm{def:g10:coordgeom:system}{COORDINATES}, एक बार साथ मानदंड और के बीच का कोण
        वैक्टर --- और मातृ सूत्र समाप्त करें:
        \[
        \cos(a - b) = \cos a \cos b + \sin a \sin b .
        \]
  \item विकल्प $-b $ के लिए $ b $ (की समानताएं \omterm{def:g11:trigo:cossin}{ज्या} और \omterm{def:g11:trigo:cossin}{कोज्या}!)
        प्राप्त करने के लिए $\cos(a + b)$.
  \item प्राप्त $\sin(a + b)$ \omterm{def:g10:proba:operations}{पूरक} पहचान से
        $\sin x = \cos\left(\frac\pi2 - x\right)$ के लिए आवेदन किया
        $ x = a + b $.
  \item आख़िरकार सटीक मान: गणना करें $\cos 15^\circ $ (जैसा
        $\cos(45^\circ - 30^\circ)$) और $\sin 75^\circ $. जांचें
        दोनों संख्यात्मक रूप से.
  \item तय करना $ b = a $ द्विकोण सूत्र प्राप्त करने के लिए:
        $\cos 2a $ (इसके तीन रूपों में) और $\sin 2a $.
  \item से $\cos 2a = 1 - 2\sin^2 a $, गणना करें
        $\sin^2 15^\circ $, तब $\sin 15^\circ $ एक नेस्टेड के रूप में
        रेडिकल --- और सत्यापित करें कि इसका वर्ग इससे सहमत है
        मूल्य $\frac{\sqrt6 - \sqrt2}{4}$ वह प्रश्न 9 का है
        विधि देता है.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Second gift: the distance to a
line.}
\begin{enumerate}[resume]
  \item होने देना $ d $ पंक्ति हो $ ax + by + c = 0$, सामान्य के साथ
        \omterm{def:g11:vect:vector}{वेक्टर} $\vec n(a, b)$ (\cref{prop:g11:scal:normal}),
        और $ P(x_0, y_0)$ एक बिंदु. कोई भी बिंदु लेना $ A $ पर $ d $,
        बताएं कि दूरी क्यों है? $ P $ को $ d $ है
        $\dfrac{\abs{\vect{AP} \cdot \vec n}}{\norm{\vec n}}$,
        और सूत्र निकालें
        \[
        \operatorname{dist}(P, d)
        = \frac{\abs{a x_0 + b y_0 + c}}{\sqrt{a^2 + b^2}} .
        \]
        से दूरी की गणना करें $ P(5, 6)$ को
        $3x + 4y - 12 = 0$.
  \item लिखना \omterm{def:g10:algebra:equation}{समीकरण} केन्द्रित वृत्त का $(5, 6)$
        \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} उस पंक्ति को.
  \item सूत्र को क्रॉस-चेक करें: से दूरी की गणना करें
        की उत्पत्ति $3x + 4y - 12 = 0$, फिर इसे सत्यापित करें
        मूल बिंदु से लम्ब के पाद की गणना करना
        स्पष्ट रूप से.
  \item त्रिकोण $ A(0,0)$, $ B(6,0)$, $ C(2,5)$: इसके क्षेत्रफल की गणना करें
        आधार का उपयोग करना $[AB]$; फिर से दूरी की गणना करें
        $ B $ लाइन तक $(AC)$ और जांचें कि यह वही पुन: उत्पन्न करता है
        क्षेत्र.
  \item \omterm{pb:g11:quad:1}{केंद्र} का घेरा $(2, 1)$ और त्रि\omterm{def:g11:trigo:cossin}{ज्या} $2$, विरुद्ध
        रेखा $ x + y - 6 = 0$: \omterm{pb:g11:quad:1}{केंद्र} की दूरी की गणना करें
        पंक्ति में जाएँ और उनकी स्थिति वर्गीकृत करें (\omterm{def:g11:deriv:rate}{काटनेवाला},
        \omterm{def:g11:deriv:tangent}{स्पर्शरेखा}, असंयुक्त)।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The bonus: altitudes concur.}
\begin{enumerate}[resume]
  \item सामान्यीकरण \cref{exo:g11:scal:11}: किन्हीं दो बिंदुओं के लिए
        $ A $, $ B $ साथ \omterm{prop:g10:coordgeom:midpoint}{मध्य} $ I $, \omterm{def:g11:vect:det}{सिद्ध} करना
        \[
        \vect{MA} \cdot \vect{MB} = MI^2 - \frac{AB^2}{4}
        \quad\text{for every point } M .
        \]
  \item परिणाम निकालना \cref{prop:g11:scal:diameter} एक पंक्ति में:
        अंकों का सेट $ M $ साथ
        $\vect{MA} \cdot \vect{MB} = 0$ का चक्र है
        व्यास $[AB]$.
  \item ऊंचाई: चलो $ H $ हो \omterm{def:g10:numbers:interunion}{चौराहा} का
        से ऊंचाई $ A $ और से $ B $ एक त्रिकोण में $ ABC $
        (इसलिए $\vect{HA} \cdot \vect{BC} = 0$ और
        $\vect{HB} \cdot \vect{CA} = 0$). पहचान साबित करें
        \[
        \vect{HA} \cdot \vect{BC} + \vect{HB} \cdot \vect{CA}
        + \vect{HC} \cdot \vect{AB} = 0
        \]
        --- से सब कुछ का \omterm{def:g10:algebra:expand}{विस्तार} करें $ H $ चैसल्स के साथ --- और
        यह निष्कर्ष निकालें $ H $ से ऊँचाई पर भी स्थित है $ C$:
        तीन ऊंचाईयां समवर्ती हैं (पर)
        \emph{ऑर्थोसेंटर}\index{ऑर्थोसेंटर}).
  \item समापन: चार चेहरे, तीन ट्राफियां। कौन सा चेहरा
        \cref{thm:g11:scal:faces} \omterm{def:g10:vectors:sum}{जोड़} सूत्र ले गए,
        जो दूरी सूत्र, जो \omterm{pb:g11:scal:1}{ऑर्थोसेंटर} ---
        और क्यों ``समान मात्रा की गणना दो बार करें'' \omterm{def:g11:scal:dot}{आदर्श} वाक्य है
        पूरे अध्याय का?
\end{enumerate}
\end{problem}
