\chapter{संभाव्यता और यादृच्छिक चर}\label{ch:g11:prob}

\omterm{def:g10:proba:distribution}{संभावना} मॉडल प्रयोग जिनका परिणाम अनिश्चित है: एक पासा रोल, ए
कार्ड निकाला गया, एक खेल खेला गया। यह अध्याय शब्दावली को परिमित पर स्थापित करता है
सेट, फिर सभी के \omterm{pb:g11:quad:1}{केंद्र}ीय चरित्र का परिचय देता है \omterm{def:g10:proba:distribution}{संभावना} \omterm{def:g11:vect:det}{सिद्ध}ांत:
द \emph{\omterm{def:g11:prob:rv}{अनियमित परिवर्तनशील वस्तु}} और इसके \emph{\omterm{def:g11:prob:expectation}{अपेक्षा}}. ये धारणाएँ हैं
समीक्षा की गई और गहनता से विचार किया गया \cref{ch:g12:randvar}, और सशर्त
सम्भावनाएँ अनुसरण करती हैं \cref{ch:g12:condprob}.

\section{एक परिमित समुच्चय पर प्रायिकता}

\begin{definition}[संभाव्यता मॉडल]\label{def:g11:prob:model}
सीमित रूप से अनेक परिणामों वाला एक प्रयोग इसके आधार पर तैयार किया जाता है \emph{नमूना अंतरिक्ष}\index{नमूना अंतरिक्ष} $\Omega = \{\omega_1, \dots, \omega_n\}$ (द
परिणामों का सेट) साथ में ए \emph{\omterm{def:g10:proba:distribution}{संभावना}} $\P $ प्रत्येक को असाइन करना
परिणाम $\omega_i $ एक संख्या $ p_i \geq 0$, साथ $ p_1 + \dots + p_n = 1$. एक
\emph{आयोजन}\index{आयोजन} एक उपसमुच्चय है $ A \subseteq \Omega $, और
\[
\P(A) = \sum_{\omega_i \in A} p_i .
\]
जब सभी परिणाम समान \omterm{def:g10:proba:distribution}{संभावना} वाले हों ($ p_i = \frac1n $: द \emph{वर्दी
मॉडल}\index{वर्दी
मॉडल}),
\[
\P(A) = \frac{\text{number of outcomes in } A}
{\text{number of outcomes in } \Omega}.
\]
\end{definition}

\begin{example}\label{ex:g11:prob:twodice}
दो अलग-अलग पासे फेंकें: $\Omega $ का सेट है $36$ आदेशित \omterm{def:g10:vectors:sum}{जोड़}े
$(i, j)$ साथ $1 \leq i, j \leq 6$, सभी समान रूप से संभावित हैं। \omterm{def:g11:prob:model}{आयोजन}
$ S $ = ``योग है $7$'' शामिल है $6$ \omterm{def:g10:vectors:sum}{जोड़}े
$(1,6)$, $(2,5)$, $(3,4)$, $(4,3)$, $(5,2)$, $(6,1)$, इसलिए
$\P(S) = \frac{6}{36} = \frac16$. \omterm{def:g11:prob:model}{आयोजन} ``योग है $12$'' शामिल है
केवल $(6,6)$: \omterm{def:g10:proba:distribution}{संभावना} $\frac{1}{36}$. का चयन करना \emph{सही}
\omterm{def:g11:prob:model}{नमूना अंतरिक्ष} --- क्रमबद्ध \omterm{def:g10:vectors:sum}{जोड़}े, अव्यवस्थित नहीं --- यही बनाता है
एकसमान मॉडल लागू.
\end{example}

\begin{proposition}[संभाव्यता के नियम]\label{prop:g11:prob:rules}
के लिए घटनाएँ $ A, B \subseteq \Omega $:
\[
\P(\emptyset) = 0, \qquad \P(\Omega) = 1, \qquad
\P(\bar A) = 1 - \P(A),
\]
\[
\P(A \cup B) = \P(A) + \P(B) - \P(A \cap B),
\]
कहाँ $\bar A $ है \emph{\omterm{def:g10:proba:operations}{पूरक}} का $ A $ (सभी नतीजे अंदर नहीं हैं $ A $).
विशेष रूप से, $\P(A \cup B) = \P(A) + \P(B)$ कब $ A $ और $ B $ हैं
\emph{\omterm{def:g10:proba:operations}{असंगत}} ($ A \cap B = \emptyset $).
\end{proposition}

\begin{proof}
द एम्प्टी \omterm{def:g11:prob:model}{आयोजन} इसमें कोई परिणाम नहीं है (खाली राशि), और $\Omega $ शामिल है
वे सभी (कुल योग) $1$). प्रत्येक परिणाम बिल्कुल एक में है $ A $,
$\bar A $, इसलिए $\P(A) + \P(\bar A) = 1$. के लिए \omterm{def:g10:numbers:interunion}{मिलन}: संक्षेप में $ p_i $
ऊपर $ A $ और फिर खत्म $ B $ के परिणामों को गिनता है $ A \cap B $ दो बार;
घटाने $\P(A \cap B)$ दोहरी गिनती को ठीक करता है।
\end{proof}

\begin{example}
एक कक्षा में, $60\%$ संगीत का अध्ययन करें ($ M $), $45\%$ कोई खेल खेलें ($ S $), और
$25\%$ दोनों करें। अनुपात कम से कम एक कर रहा है
$\P(M \cup S) = 0.60 + 0.45 - 0.25 = 0.80$, इसलिए $20\%$ न ही करो
(\omterm{def:g10:proba:operations}{पूरक}).
\end{example}

\section{यादृच्छिक चर}

\begin{definition}[यादृच्छिक चर, वितरण]\label{def:g11:prob:rv}
ए \emph{अनियमित परिवर्तनशील वस्तु}\index{अनियमित परिवर्तनशील वस्तु} पर $\Omega $ एक है \omterm{def:g11:func:function}{समारोह}
$ X $ प्रत्येक परिणाम के लिए एक संख्या निर्दिष्ट करना। इसका
\emph{वितरण}\index{वितरण} इसकी \omterm{def:g10:proba:distribution}{संभावना} की तालिका है
मूल्य $ x_1, \dots, x_k $ उनकी संभावनाओं के साथ
\[
\P(X = x_i) = \P\bigl(\{\omega : X(\omega) = x_i\}\bigr),
\qquad
\sum_{i=1}^k \P(X = x_i) = 1 .
\]
\end{definition}

\begin{example}\label{ex:g11:prob:game}
एक खेल: भुगतान करें $2$ यूरो, एक उचित पासा रोल करें, यदि ऐसा हो तो दिखाया गया मूल्य प्राप्त करें
कम से कम है $5$, अन्यथा कुछ नहीं. होने देना $ G $ हो \emph{पाना} (प्राप्त
माइनस भुगतान)। परिणाम $1, 2, 3, 4$ देना $ G = -2$; ये परिणाम $5$
देता है $ G = 3$; ये परिणाम $6$ देता है $ G = 4$. \omterm{def:g11:prob:rv}{वितरण} का $ G $:
\begin{center}
\begin{tabular}{c|ccc}
$ g $ & $-2$ & $3$ & $4$\\
\hline\rule{0pt}{11pt}%
$\P(G = g)$ & $\dfrac46$ & $\dfrac16$ & $\dfrac16$
\end{tabular}
\end{center}
\end{example}

\begin{method}[एक वितरण तालिका का निर्माण]\label{met:g11:prob:table}
के संभावित मानों की सूची बनाएं $ X $; प्रत्येक मान के लिए, परिणाम एकत्र करें
इसे तैयार करना और उनकी संभावनाओं को जोड़ना; जाँच करें कि की पंक्ति
संभावनाओं का योग है $1$.
\end{method}

\section{अपेक्षा}

\begin{definition}[अपेक्षा]\label{def:g11:prob:expectation}
 \emph{अपेक्षा}\index{अपेक्षा} (या \omterm{def:g11:stat:mean}{अर्थ}) का $ X $ औसत है
इसके मूल्यों को उनकी संभावनाओं द्वारा भारित किया गया:
\[
\E(X) = \sum_{i=1}^{k} \P(X = x_i)\; x_i .
\]
\end{definition}

\begin{remark}
$\E(X)$ क्या है \omterm{def:g11:stat:mean}{अर्थ} की स्वतंत्र पुनरावृत्ति की एक बड़ी संख्या
प्रयोग होगा --- इस कथन का सटीक संस्करण है
बड़ी संख्या का नियम, में \omterm{def:g11:vect:det}{सिद्ध} हुआ \cref{ch:g12:sums}. औपचारिकता पर ध्यान दें
के साथ सादृश्य \omterm{def:g11:stat:mean}{अर्थ} आवृत्तियों वाले डेटासेट का
(\cref{def:g11:stat:mean}): संभावनाएँ रिश्तेदार की भूमिका निभाती हैं
आवृत्तियाँ $\frac{n_i}{n}$.
\end{remark}

\begin{example}\label{ex:g11:prob:gameexp}
के खेल के लिए \cref{ex:g11:prob:game}:
\[
\E(G) = \frac46 \times (-2) + \frac16 \times 3 + \frac16 \times 4
= \frac{-8 + 3 + 4}{6} = -\frac16 \approx -0.17 .
\]
औसतन खिलाड़ी हार जाता है $17$ प्रति गेम सेंट: गेम है
प्रतिकूल. एक गेम कहा जाता है \emph{गोरा} जब अपेक्षित लाभ हो $0$.
\end{example}

\begin{proposition}[रैखिकता]\label{prop:g11:prob:linearity}
सभी वास्तविकताओं के लिए $ a, b $:
\[
\E(aX + b) = a\,\E(X) + b .
\]
\end{proposition}

\begin{proof}
परिवर्तनशील $ aX + b $ मूल्य लेता है $ a x_i + b $ साथ \omterm{def:g10:proba:distribution}{संभावना}
$\P(X = x_i)$, इसलिए
\[
\E(aX + b) = \sum_i \P(X = x_i)(a x_i + b)
= a \sum_i \P(X = x_i)\,x_i + b \underbrace{\sum_i \P(X =
x_i)}_{=\,1} = a\,\E(X) + b .
\]
\end{proof}

\section{विचरण और मानक विचलन}

\begin{definition}[झगड़ा]\label{def:g11:prob:variance}
 \emph{झगड़ा}\index{झगड़ा} का $ X $ इसके प्रसार को मापता है
चारों ओर मूल्य \omterm{def:g11:prob:expectation}{अपेक्षा} $ m = \E(X)$:
\[
\V(X) = \E\bigl((X - m)^2\bigr)
= \sum_{i=1}^{k} \P(X = x_i)\,(x_i - m)^2,
\qquad
\sigma(X) = \sqrt{\V(X)} .
\]
\end{definition}

\begin{proposition}[शॉर्टकट फॉर्मूला]\label{prop:g11:prob:konig}
\[
\V(X) = \E(X^2) - \E(X)^2 .
\]
\end{proposition}

\begin{proof}
बढ़ाना $(x_i - m)^2 = x_i^2 - 2m\,x_i + m^2$ योग के अंदर:
\[
\V(X) = \sum_i \P(X = x_i)\,x_i^2
- 2m \sum_i \P(X = x_i)\,x_i + m^2\sum_i \P(X = x_i)
= \E(X^2) - 2m^2 + m^2 ,
\]
जो है $\E(X^2) - m^2$. यह का यादृच्छिक-चर संस्करण है
में डेटासेट के लिए पहचान साबित हुई \cref{prop:g11:stat:shortcut}.
\end{proof}

\begin{proposition}[एफ़िन परिवर्तन]\label{prop:g11:prob:affine}
$\V(aX + b) = a^2\,\V(X)$ और $\sigma(aX + b) = \abs a\,\sigma(X)$.
\end{proposition}

\begin{proof}
रैखिकता से, $ aX + b $ अपने से भटक जाता है \omterm{def:g11:prob:expectation}{अपेक्षा} $ a m + b $ द्वारा
$(aX + b) - (am + b) = a(X - m)$: प्रत्येक विचलन को इसके द्वारा मापा जाता है $ a $, इसलिए
प्रत्येक वर्ग विचलन द्वारा $ a^2$, और इसी तरह उनका भार भी है \omterm{def:g11:stat:mean}{अर्थ}. स्थानान्तरित करना $ b $ गायब हो गया है --- किसी वेरिएबल को स्थानांतरित करने से उसका परिवर्तन नहीं होता है
फैलाव.
\end{proof}

\begin{example}\label{ex:g11:prob:variance}
के खेल के लिए \cref{ex:g11:prob:game}:
$\E(G^2) = \frac46 \times 4 + \frac16 \times 9 + \frac16 \times 16
= \frac{16 + 9 + 16}{6} = \frac{41}{6}$, इसलिए
\[
\V(G) = \frac{41}{6} - \left(-\frac16\right)^2
= \frac{41}{6} - \frac{1}{36} = \frac{245}{36} \approx 6.8,
\qquad
\sigma(G) \approx 2.6 .
\]
का नुकसान $0.17$ प्रति गेम औसतन विशिष्ट आकार के उतार-चढ़ाव आते हैं
$2.6$ यूरो --- \omterm{def:g11:prob:expectation}{अपेक्षा} बिना \omterm{def:g11:stat:variance}{मानक विचलन} आधा ही बताता है
कहानी.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g11:prob:1}
एक कार्ड एक मानक से निकाला जाता है $52$-कार्ड डेक। की गणना करें \omterm{def:g10:proba:distribution}{संभावना}
ड्राइंग का: एक दिल; एक राजा; एक दिल या एक राजा; न तो एक दिल और न ही एक
राजा.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:prob:2}
दो निष्पक्ष पासे फेंके गए और उनका योग $ S $ अभिलिखित है। का उपयोग
$36$-परिणाम मॉडल \cref{ex:g11:prob:twodice}, गणना करें $\P(S = 6)$,
$\P(S \geq 10)$ और $\P(S \text{ is even})$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:prob:3}
ए \omterm{def:g11:prob:rv}{अनियमित परिवर्तनशील वस्तु} है \omterm{def:g11:prob:rv}{वितरण}
\begin{center}
\begin{tabular}{c|cccc}
$ x $ & $-1$ & $0$ & $2$ & $5$\\
\hline
$\P(X=x)$ & $0.3$ & $0.2$ & $0.4$ & $ p $
\end{tabular}
\end{center}
खोजो $ p $, फिर गणना करें $\E(X)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:prob:4}
$ X $ संतुष्ट $\E(X) = 3$ और $\V(X) = 4$. दे दो \omterm{def:g11:prob:expectation}{अपेक्षा}, \omterm{def:g11:prob:variance}{झगड़ा}
और \omterm{def:g11:stat:variance}{मानक विचलन} का $ Y = 2X - 5$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:prob:5}
भाग्य का एक पहिया है $10$ समान क्षेत्र: $5$ दिखाओ $0$ यूरो, $3$ दिखाओ
$2$ यूरो, $2$ दिखाओ $5$ यूरो. खेलने की लागत $2$ यूरो. दे दो
\omterm{def:g11:prob:rv}{वितरण} लाभ का $ G $ (जीत घटा हिस्सेदारी), गणना करें $\E(G)$, और
तय करें कि खेल अनुकूल है या नहीं।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:prob:6}
एक कलश में शामिल है $3$ लाल और $2$ नीली गेंदें; दो गेंदें निकाली गईं
\emph{बिना} प्रतिस्थापन। होने देना $ X $ निकाली गई लाल गेंदों की संख्या हो।
का निर्माण करें \omterm{def:g11:prob:rv}{वितरण} का $ X $ (गिनाओ $20$ आदेशित ड्रा, या उपयोग करें
\omterm{met:g10:proba:tree}{पेड़}), और गणना करें $\E(X)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:prob:7}
के पहिये के लिए \cref{exo:g11:prob:5}, गणना करें $\V(G)$ और $\sigma(G)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:prob:8}
एक बीमा कंपनी किसके लिए पॉलिसी बेचती है? $80$ यूरो. साथ \omterm{def:g10:proba:distribution}{संभावना}
$0.05$ ग्राहक का दावा है $1000$ यूरो, के साथ \omterm{def:g10:proba:distribution}{संभावना} $0.01$ ग्राहक
दावा $5000$ यूरो, अन्यथा कुछ भी नहीं। होने देना $ B $ कंपनी का लाभ हो
एक नीति पर. दे दो \omterm{def:g11:prob:rv}{वितरण} का $ B $, गणना करें $\E(B)$, और
व्याख्या करें।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:prob:9}
एक बहुविकल्पीय प्रश्न है $4$ उत्तर, एक सही। एक सही उत्तर
स्कोर $+1$; अनुमान लगाने को हतोत्साहित करने के लिए, गलत उत्तर पर अंक मिलते हैं $ x < 0$.
एक विद्यार्थी यादृच्छिक रूप से उत्तर देता है। किस मूल्य के लिए $ x $ छात्र का है
अपेक्षित स्कोर शून्य?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:prob:10}
दो निष्पक्ष सिक्के उछाले गए। ऐन का प्रस्ताव है: ``यदि दोनों सिक्के मेल खाते हैं, तो मैं आपको भुगतान करूंगा
$1$ यूरो; यदि वे भिन्न हैं, तो आप मुझे भुगतान करें $1$ यूरो.'' अपेक्षित की गणना करें
प्रत्येक खिलाड़ी का लाभ. क्या खेल निष्पक्ष है? तीन सिक्कों के साथ भी वही सवाल,
जहां ऐन भुगतान करती है $2$ यदि तीनों मेल खाते हैं तो यूरो।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:prob:11}
एक लॉटरी बिकती है $1000$ पर टिकट $5$ प्रत्येक यूरो. एक टिकट जीतता है
$2000$ यूरो, पाँच टिकट जीते $200$ यूरो, बीस टिकट जीते $25$
यूरो.
\begin{enumerate}
  \item दे दो \omterm{def:g11:prob:rv}{वितरण} आयोजक के कुल राजस्व का घटा
        पुरस्कार, और एकल खिलाड़ी का लाभ $ G $.
  \item गणना करना $\E(G)$ और आयोजक का कुल अपेक्षित लाभ।
        जांचें कि दोनों उत्तर सुसंगत हैं।
\end{enumerate}
\end{exercise}

\section{समस्या: अधूरा खेल}

\begin{problem}[{सप्ताहांत समस्या---पास्कल, फ़र्मेट और मेला बाधित शर्त का विभाजन: उम्मीद का जन्म, जादू रैखिकता की, और क्या अपेक्षा नहीं देख सकते}]
\label{pb:g11:prob:1}
1654 में शेवेलियर डी एम\'एर\'ई---वही जुआरी जिसका पासा
बेट्स ने मिडिल स्कूल वॉल्यूम खोला \omterm{def:g10:proba:distribution}{संभावना} समस्या ---
ब्लेज़ पास्कल से एक गहरा प्रश्न पूछा: दो समान रूप से कुशल
के स्कोर पर खिलाड़ियों को अपनी श्रृंखला रोकनी होगी $5$ को $3$, छह
लेने में जीतता है $64$-पिस्तौल पात्र. \emph{गमला कैसा होना चाहिए
निष्पक्ष रूप से विभाजित?} पास्कल और फ़र्मेट के बीच के पत्र इसका उत्तर दे रहे हैं
इस अध्याय के मूल में इस अवधारणा को स्थापित किया गया: \omterm{def:g11:prob:expectation}{अपेक्षा}
(\cref{def:g11:prob:expectation}). आप उन्हें पुनः प्राप्त करेंगे
दोनों तरीकों से उत्तर दें, फिर रैखिकता की पार्टी चाल को पूरा करें और
अपेक्षायह एक अंधा स्थान है.

\medskip
\noindent\textbf{Part I --- \omterm{def:g11:prob:expectation}{अपेक्षा} fluency.}
\begin{enumerate}
  \item एक निष्पक्ष पासा रोल करें; आपका लाभ है (चेहरा $- 3$) यूरो. निर्माण
        द \omterm{def:g11:prob:rv}{वितरण} लाभ की तालिका $ G $
        (\cref{met:g11:prob:table}) और गणना करें $\E(G)$.
  \item गणना करना $ V(G)$ शॉर्टकट फ़ॉर्मूले के साथ
        (\cref{prop:g11:prob:konig}) और $\sigma(G)$ (दो
        दशमलव)।
  \item बिना किसी नई टेबल के दे दो $\E(2G - 1)$ और
        $ V(2G - 1)$ (\cref{prop:g11:prob:affine}).
  \item एक स्टॉल शुल्क लेता है $ m $ दो पासे और जेबें घुमाने के लिए यूरो
        आप राशि यूरो में. क्या है \emph{गोरा} कीमत
        $ m $? स्टॉल शुल्क $8$: आपका अपेक्षित नुकसान क्या है?
        प्रति गेम?
  \item एक लैपटॉप लायक $800$ यूरो में एक है $5\,\%$ का मौका
        हर साल विनाश. उचित बीमा की गणना करें
        प्रीमियम, और बीमाकर्ता का प्रति पॉलिसी अपेक्षित लाभ
        अगर यह चार्ज करता है $60$ यूरो.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Splitting the pot.}
अंक $5$--$3$, किसी कार्य के लिए पहला होना $6$ जीतता है $64$ पिस्तौलें, गोलियाँ हैं
निष्पक्ष सिक्का उछाल.
\begin{enumerate}[resume]
  \item दो आकर्षक गलत उत्तर: विभाजन $5 : 3$
        (जीते राउंड के आनुपातिक रूप से), या सब कुछ दे दो
        नेता. एक-एक वाक्य में बताएं कि क्या अनुचित है
        उनके बारे में.
  \item फ़र्मेट की विधि: खिलाड़ी ए की आवश्यकता है $1$ अधिक जीत, बी की जरूरत है
        $3$; बिल्कुल खेलने की कल्पना करें $3$ परवाह किए बिना अधिक राउंड।
        लिस्ट $8$ समान रूप से संभावित परिणाम, चिह्नित करें कि कौन जीतता है
        प्रत्येक में श्रृंखला, और का उचित विभाजन निकालें $64$
        पिस्तौलें
  \item फ़र्मेट को जिस सूक्ष्म बिंदु का बचाव करना था: वास्तविक खेल
        एक दौर के बाद रुक सकता है --- फिर भी ऐसा क्यों है
        सभी पर तर्क करना वैध है $3$ काल्पनिक दौर?
  \item पास्कल की विधि, अंत से पीछे की ओर: ए की गणना करें
        (काल्पनिक) स्कोर पर उचित हिस्सा $5$--$5$, तब
        $5$--$4$, तब $5$--$3$, हर बार दोनों का औसत
        समान रूप से संभावित भविष्य। प्रश्न 7 से तुलना करें।
  \item स्कोर पर वही प्रश्न $4$--$3$: कितने काल्पनिक
        राउंड, कौन से परिणाम बी को श्रृंखला देते हैं, और क्या है
        उचित विभाजन?
  \item वह \omterm{def:g11:vect:det}{सिद्ध}ांत बताएं जो दोनों विधियां साझा करती हैं --- निष्पक्ष
        अनिश्चित भविष्य का मूल्य यही है \emph{\omterm{def:g11:prob:expectation}{अपेक्षा}}
        --- और पूरी तरह से चलने वाले दो आधुनिक उद्योगों के नाम बताएं
        यह (प्रश्न 5 एक दिखाता है)।
  \item प्रत्येक अंक के लिए एक विवेक जांच: समझाएं, साथ में
        की रैखिकता \omterm{def:g11:prob:expectation}{अपेक्षा}
        (\cref{prop:g11:prob:linearity}), क्यों दो खिलाड़ियों का
        उचित शेयरों को हमेशा बिल्कुल ठीक तक जोड़ना चाहिए
        $64$-पिस्तौल पात्र.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Linearity's party trick.}
\begin{enumerate}[resume]
  \item वापस पाना $\E(\text{sum of two dice}) = 7$ एक पंक्ति में
        रैखिकता के साथ (\cref{prop:g11:prob:linearity}), नहीं
        $36$-सेल टेबल. के लिए \emph{\omterm{def:g11:prob:variance}{झगड़ा}} योग का,
        नियम $ V(X + Y) = V(X) + V(Y)$ पासे की जरूरत है
        स्वतंत्र (स्वीकृत): इसकी गणना करें।
  \item हैट-चेक पार्टी: $ n $ मेहमान अपनी टोपियाँ फेंकते हैं
        ढेर और प्रत्येक यादृच्छिक रूप से एक वापस ले लेता है। होने देना $ X $ गिनती
        मेहमान जो अपनी टोपी पुनः प्राप्त करते हैं। एक के लिए तय किया गया
        अतिथि, क्या है \omterm{def:g10:proba:distribution}{संभावना} अपने आप को पुनः प्राप्त करने का
        टोपी? इन्हें संक्षेप में बताएं $ n $ संभावनाएँ (रैखिकता!), गणना करें
        $\E(X)$. उत्तर आपको आश्चर्यचकित करना चाहिए: ऐसा नहीं है
        पर निर्भर है $ n $.
  \item सत्यापित करें $\E(X) = 1$ के लिए क्रूर बल द्वारा $ n = 2$ और
        $ n = 3$ (लिस्ट $2$ और $6$ समान रूप से संभावित
        हैट-असाइनमेंट और औसत गिनती)।
  \item मेहमानों की वसूली स्वतंत्र (यदि) से बहुत दूर है
        $ n - 1$ मेहमानों को उनकी आखिरी टोपी वापस मिल गई
        अवश्य भी)। प्रश्न 14 की गणना वास्तव में कहां की गई
        \emph{नहीं} स्वतंत्रता की आवश्यकता है? राज्य रैखिकता
        महाशक्ति.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- What \omterm{def:g11:prob:expectation}{अपेक्षा} cannot see.}
\begin{enumerate}[resume]
  \item दो $5$-यूरो स्क्रैच कार्ड: कार्ड ए जीतता है $10$ यूरो के साथ
        \omterm{def:g10:proba:distribution}{संभावना} $\frac12$; कार्ड बी जीतता है $1\,000$ यूरो के साथ
        \omterm{def:g10:proba:distribution}{संभावना} $0.005$. जांचें कि दोनों लाभ हैं
        वही \omterm{def:g11:prob:expectation}{अपेक्षा}, फिर दोनों की गणना करें प्रसरण. जो
        क्या वास्तविक लोग कार्ड खरीदते हैं, और वे क्या खरीद रहे हैं?
  \item सेंट पीटर्सबर्ग खेल: पहली पूंछ तक पलटें;
        यदि पहली पट फ्लिप पर होती है $ n $ आपको प्राप्त हुया
        $2^n $ यूरो. प्रत्येक के योगदान की गणना करें $ n $ तक \omterm{def:g11:prob:expectation}{अपेक्षा} और यह \omterm{def:g11:prob:expectation}{अपेक्षा} स्वयं. क्या आप
        भुगतान करें $1\,000$ यूरो एक बार खेलने के लिए? टकराव क्या होता है
        के बारे में सिखाओ \omterm{def:g11:prob:expectation}{अपेक्षा}?
  \item ग्राहक की ओर से बीमा: भुगतान करें $20$ यूरो, या
        चेहरा ए $1\,\%$ खोने का जोखिम $1\,000$? दोनों की तुलना करें
        अपेक्षाएं, और बताएं कि बीमा क्यों खरीदा जा सकता है
        फिर भी तर्कसंगत रहें (एकल परिवार क्या नहीं कर सकता
        ``औसत से अधिक''?)।
  \item समापन --- अपेक्षाकी जीवनी, प्रति एक वाक्य
        अध्याय: 1654 में एक बर्तन फोड़ने के लिए जन्म; इसकी महाशक्ति,
        रैखिकता, जो जोड़ती है अपेक्षाएं बिना पूछे
        स्वतंत्रता (टोपियाँ); इसका अंधा स्थान, जोखिम, जो
        \omterm{def:g11:prob:variance}{झगड़ा} उपाय (कार्ड, सेंट पीटर्सबर्ग, बीमा);
        और इसका राज्याभिषेक अगले वर्ष होगा, जब बड़े पैमाने का विधान होगा
        संख्याएँ बिल्कुल स्पष्ट करती हैं कि घर हमेशा क्यों जीतता है।
\end{enumerate}
\end{problem}
