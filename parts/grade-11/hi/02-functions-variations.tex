\chapter{कार्य और विविधताएँ}\label{ch:g11:func}

यह अध्याय वह व्यवस्थित करता है जिसके बारे में किसी को जानना आवश्यक है कार्य पहले
कैलकुलस: का छोटा परिवार \emph{\omterm{def:g11:func:reference}{संदर्भ कार्य}} किसका रेखांकन अवश्य
दिल से जाना जाए, इसका सटीक \omterm{def:g10:stats:mean}{अर्थ} \emph{\omterm{def:g11:func:monotone}{की बढ़ती}} और
\emph{\omterm{def:g10:functions:variations}{घटते}}, और विविधताएँ कब कैसे व्यवहार करती हैं कार्य संयुक्त हैं.

\section{कार्य और वक्र}

\begin{definition}[फ़ंक्शन, डोमेन, वक्र]\label{def:g11:func:function}
ए \emph{समारोह}\index{समारोह} $f $ प्रत्येक नंबर को असाइन करता है $ x $ एक सेट का
$ D \subseteq \R $ (इसका \emph{कार्यक्षेत्र}\index{कार्यक्षेत्र}) बिल्कुल एक संख्या
$ f(x)$. इसका \emph{वक्र} (या \omterm{def:g10:functions:graph}{ग्राफ}) बिंदुओं का समूह है
$\bigl(x, f(x)\bigr)$ के लिए $ x \in D $.
\end{definition}

\begin{example}
सूत्र $ f(x) = \frac{\sqrt{x+2}}{x-1}$ आवश्यक है $ x + 2 \geq 0$ (के लिए
वर्गमूल) और $ x \neq 1$ (अशून्य हर): द \omterm{def:g11:func:function}{कार्यक्षेत्र} है
$\intco{-2}{1} \cup \intoo{1}{+\infty}$.
\end{example}

\section{एक अंतराल पर बदलाव}

\begin{definition}[बढ़ रहा है, घट रहा है]\label{def:g11:func:monotone}
होने देना $ f $ एक पर परिभाषित किया जाए \omterm{def:g10:numbers:interval}{अंतराल} $ I $. \omterm{def:g11:func:function}{समारोह} $ f $ है
\emph{की बढ़ती}\index{की बढ़ती} पर $ I $ यदि यह व्यवस्था बरकरार रखता है:
\[
\text{for all } u, v \in I: \quad u < v \implies f(u) \leq f(v),
\]
और \emph{\omterm{def:g10:functions:variations}{घटते}} यदि यह क्रम उलट देता है ($ u < v \implies f(u) \geq
f(v)$). दाहिनी ओर सख्त असमानताओं के साथ, $ f $ है \emph{कठोरता से}
बढ़ रहा है (सम्मान)\ \omterm{def:g10:functions:variations}{घटते}). ए \omterm{def:g11:func:function}{समारोह} वह बढ़ रहा है या
\omterm{def:g10:functions:variations}{घटते} पर $ I $ है \emph{एकरस} पर $ I $.
\end{definition}

\begin{method}[परिभाषा से भिन्नता सिद्ध करना]\label{met:g11:func:variation}
दो मनमानी संख्याएँ लें $ u < v $ में \omterm{def:g10:numbers:interval}{अंतराल}, और चिन्ह निर्धारित करें
अंतर का $ f(v) - f(u)$, आमतौर पर द्वारा \emph{\omterm{def:g10:algebra:expand}{फैक्टरिंग}} यह। यदि
पर चिन्ह स्थिर है \omterm{def:g10:numbers:interval}{अंतराल}, निष्कर्ष निकालें।
\end{method}

\begin{proposition}[संदर्भ कार्यों की विविधताएँ]\label{prop:g11:func:refvariations}
\begin{enumerate}
  \item $ x^2$ सख्ती से है \omterm{def:g10:functions:variations}{घटते} पर $\intoc{-\infty}{0}$ और सख्ती से
        \omterm{def:g11:func:monotone}{की बढ़ती} पर $\intco{0}{+\infty}$.
  \item $ x^3$ सख्ती से है \omterm{def:g11:func:monotone}{की बढ़ती} पर $\R $.
  \item $\sqrt x $ सख्ती से है \omterm{def:g11:func:monotone}{की बढ़ती} पर $\intco{0}{+\infty}$.
  \item $\frac1x $ सख्ती से है \omterm{def:g10:functions:variations}{घटते} पर $\intoo{-\infty}{0}$ और पर
        $\intoo{0}{+\infty}$ (लेकिन \emph{नहीं} पर उनके \omterm{def:g10:numbers:interunion}{मिलन}).
  \item $\abs x $ सख्ती से है \omterm{def:g10:functions:variations}{घटते} पर $\intoc{-\infty}{0}$ और
        सख्ती से \omterm{def:g11:func:monotone}{की बढ़ती} पर $\intco{0}{+\infty}$.
\end{enumerate}
\end{proposition}

\begin{proof}
होने देना $ u < v $. हम कारक हैं $ f(v) - f(u)$ हर मामले में।

\emph{1.} $ v^2 - u^2 = (v-u)(v+u)$. पर $\intco{0}{+\infty}$: $ v - u > 0$
और $ v + u > 0$ (जैसा $ v > u \geq 0$), इसलिए $ v^2 > u^2$. पर
$\intoc{-\infty}{0}$: $ v + u < 0$, इसलिए $ v^2 < u^2$.

\emph{2.} $ v^3 - u^3 = (v - u)(v^2 + uv + u^2)$, और दूसरा कारक
बराबर $\left(v + \frac u2\right)^2 + \frac{3u^2}{4} > 0$ जब तक
$ u = v = 0$. इस तरह $ v^3 > u^3$ जब कभी भी $ v > u $.

\emph{3.} के लिए $0 \leq u < v $:
$\sqrt v - \sqrt u = \frac{(\sqrt v - \sqrt u)(\sqrt v + \sqrt u)}
{\sqrt v + \sqrt u} = \frac{v - u}{\sqrt v + \sqrt u} > 0$.

\emph{4.} के लिए $0 < u < v $:
$\frac1v - \frac1u = \frac{u - v}{uv} < 0$ तब से $ u - v < 0$ और $ uv > 0$;
के लिए समान गणना $ u < v < 0$. पर \omterm{def:g10:numbers:interunion}{मिलन} दावा विफल रहता है:
$-1 < 1$ लेकिन $\frac{1}{-1} = -1 < 1 = \frac11$.

\emph{5.} पर $\intco{0}{+\infty}$, $\abs x = x $ सख्ती से है \omterm{def:g11:func:monotone}{की बढ़ती}; पर
$\intoc{-\infty}{0}$, $\abs x = -x $ सख्ती से है \omterm{def:g10:functions:variations}{घटते}.
\end{proof}

\begin{remark}
बिंदु 4 एक क्लासिक जाल है: ``\omterm{def:g10:functions:variations}{घटते} पर $\intoo{-\infty}{0}$ और पर
$\intoo{0}{+\infty}$'' है \emph{नहीं} `` के समानघटते पर
$\R \setminus \{0\}$''. विविधताएं केवल तभी समझ में आती हैं \omterm{def:g10:numbers:interval}{अंतराल}.
\end{remark}

\begin{example}[गणना के बिना तुलना करना]
तब से $\sqrt x $ सख्ती से है \omterm{def:g11:func:monotone}{की बढ़ती}, $\sqrt{7} < \sqrt{8}$; तब से
$ x^2$ सख्ती से है \omterm{def:g10:functions:variations}{घटते} पर $\intoc{-\infty}{0}$,
$(-3)^2 > (-2)^2$. एकरसता तुलना को बदल देती है इमेजिस में
तर्कों की तुलना.
\end{example}

\section{पुराने से नए कार्य}

\begin{proposition}[परिवर्तनों]\label{prop:g11:func:transforms}
होने देना $ u $ एक हो \omterm{def:g11:func:function}{समारोह} एक पर परिभाषित \omterm{def:g10:numbers:interval}{अंतराल} $ I $ और $ k \in \R $.
\begin{enumerate}
  \item $ u + k $ के पास है \emph{वही} विविधताओं के रूप में $ u $ पर $ I $; इसका वक्र है
        का $ u $ द्वारा लंबवत रूप से स्थानांतरित किया गया $ k $.
  \item के लिए $\lambda > 0$, $\lambda u $ जैसी ही विविधताएँ हैं $ u $; के लिए
        $\lambda < 0$, विविधताएँ उलट जाती हैं।
  \item अगर $ u \geq 0$ पर $ I $, तब $\sqrt u $ जैसी ही विविधताएँ हैं
        $ u $.
  \item अगर $ u \neq 0$ और $ u $ पर लगातार साइन ऑन है $ I $, तब $\frac1u $ है
        विविधताएँ \emph{विलोम} उन लोगों के लिए $ u $.
\end{enumerate}
\end{proposition}

\begin{proof}
होने देना $ u < v $ में $ I $ (नोटेशन का दुरुपयोग करते हुए, हम तुलना करते हैं $ x $-मूल्य $ u < v $; लिखना
$ s = u(x_1)$, $ t = u(x_2)$ के लिए इमेजिस का $ x_1 < x_2$).

\emph{1.} $(u+k)(x_2) - (u+k)(x_1) = t - s $: का अंतर इमेजिस है
अपरिवर्तित है, इसलिए इसका चिन्ह अपरिवर्तित है।

\emph{2.} $\lambda t - \lambda s = \lambda(t - s)$: चिन्ह रखा जाता है यदि
$\lambda > 0$, अगर फ़्लिप किया गया $\lambda < 0$.

\emph{3.} अगर $ u $ है \omterm{def:g11:func:monotone}{की बढ़ती} और $ x_1 < x_2$, तब $0 \leq s \leq t $, और
$\sqrt{}$ प्राणी \omterm{def:g11:func:monotone}{की बढ़ती} (\cref{prop:g11:func:refvariations}) देता है
$\sqrt s \leq \sqrt t $. में भी यही तर्क \omterm{def:g10:functions:variations}{घटते} मामला।

\emph{4.} अगर $ u $ है \omterm{def:g11:func:monotone}{की बढ़ती}, $0 < s \leq t $ (या $ s \leq t < 0$), तब
$\frac1s \geq \frac1t $ क्योंकि $ x \mapsto \frac1x $ है \omterm{def:g10:functions:variations}{घटते} प्रत्येक पर
की ओर $0$: क्रम उलट गया है.
\end{proof}

\begin{example}\label{ex:g11:func:composite}
अध्ययन $ f(x) = \sqrt{x^2 + 1}$ पर $\R $. आंतरिक \omterm{def:g11:func:function}{समारोह}
$ u(x) = x^2 + 1$ सकारात्मक है, \omterm{def:g10:functions:variations}{घटते} पर $\intoc{-\infty}{0}$ और
\omterm{def:g11:func:monotone}{की बढ़ती} पर $\intco{0}{+\infty}$ (एक स्थानांतरित वर्ग)। बिंदु 3 से, $ f $
समान विविधताएँ हैं: \omterm{def:g10:functions:variations}{घटते} तब \omterm{def:g11:func:monotone}{की बढ़ती}, साथ \omterm{def:g10:functions:extrema}{न्यूनतम}
$ f(0) = 1$.
\end{example}

\section{सम और विषम कार्य}

\begin{definition}[समता]\label{def:g11:func:parity}
होने देना $ f $ ए पर परिभाषित किया जाए \omterm{def:g11:func:function}{कार्यक्षेत्र} $ D $ के बारे में सममित $0$ (\omterm{def:g10:stats:mean}{अर्थ}ात्\ $-x \in D $
जब कभी भी $ x \in D $). \omterm{def:g11:func:function}{समारोह} $ f $ है \emph{यहां तक की}\index{यहां तक की}
अगर $ f(-x) = f(x)$ सभी के लिए $ x \in D $, और \emph{विषम}\index{विषम} अगर
$ f(-x) = -f(x)$ सभी के लिए $ x \in D $.
\end{definition}

\begin{proposition}[ज्यामितीय अर्थ]\label{prop:g11:func:paritysym}
एक का वक्र यहां तक कि \omterm{def:g11:func:function}{समारोह} के बारे में सममित है $ y $-अक्ष; वक्र
एक का पुराना फंक्शन मूल के बारे में सममित है।
\end{proposition}

\begin{proof}
अगर $ f $ है \omterm{def:g11:func:parity}{यहां तक की} और $(x, y)$ वक्र पर है (इसलिए) $ y = f(x)$), तब
$ f(-x) = f(x) = y $: दर्पण बिंदु $(-x, y)$ भी वक्र पर है. अगर
$ f $ है \omterm{def:g11:func:parity}{विषम}, $ f(-x) = -y $: मुद्दा $(-x, -y)$, का सममित $(x,y)$ के बारे में
मूल, वक्र पर है.
\end{proof}

\begin{example}
$ x^2$ और $\abs x $ हैं \omterm{def:g11:func:parity}{यहां तक की}; $ x^3$ और $\frac1x $ हैं \omterm{def:g11:func:parity}{विषम};
$ f(x) = x^2 + x $ नहीं है: $ f(-1) = 0$ जबकि $ f(1) = 2$, इसलिए
$ f(-1) \neq \pm f(1)$. समता कार्य को आधा कर देती है: a \omterm{def:g11:func:parity}{यहां तक की} या पुराना फंक्शन
केवल इस पर अध्ययन करने की आवश्यकता है $\intco{0}{+\infty}$.
\end{example}

\section{अभ्यास}

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:func:6}
किसी भी व्युत्पन्न की गणना किए बिना, इसकी विविधताएं बताएं
\[
f(x) = \sqrt{5 - x} \ \text{ on } \intoc{-\infty}{5},
\quad
g(x) = \frac{1}{x^2 + 1} \ \text{ on } \intco{0}{+\infty},
\quad
h(x) = 3 - 2\sqrt{x} \ \text{ on } \intco{0}{+\infty}.
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:func:7}
होने देना $ f(x) = (x-2)^2 - 3$ पर $\R $.
\begin{enumerate}
  \item की विविधताएँ दीजिए $ f $ और इसके \omterm{def:g10:functions:extrema}{न्यूनतम}.
  \item की विविधताएँ निकालें $ g(x) = \frac{1}{(x-2)^2+1}$ और इसके
        \omterm{def:g10:functions:extrema}{अधिकतम}. (टिप्पणी $ g = \frac{1}{f + 4}$.)
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:func:8}
ए का वक्र \omterm{def:g11:func:function}{समारोह} $ f $ बिन्दुओं से होकर गुजरता है $(0, 1)$, $(2, 3)$
और $(4, 1)$, और $ f $ है \omterm{def:g11:func:monotone}{की बढ़ती} पर $\intcc{0}{2}$ और \omterm{def:g10:functions:variations}{घटते} पर
$\intcc{2}{4}$. एक संभावित वक्र का रेखाचित्र बनाएं, फिर उसके वक्रों का रेखाचित्र बनाएं
$ f + 2$, का $-f $, और का $2f $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:func:9}
दिखाओ कि \omterm{def:g11:func:function}{समारोह} $ f(x) = \frac{2x+1}{x+1}$ लिखा जा सकता है
$ f(x) = 2 - \frac{1}{x+1}$, और इसकी विविधताएं निकालें
$\intoo{-1}{+\infty}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:func:10}
सही या गलत (उचित ठहराएं या प्रति उदाहरण दें):
\begin{enumerate}
  \item दो का योग कार्यों में वृद्धि पर $ I $ है \omterm{def:g11:func:monotone}{की बढ़ती} पर $ I $.
  \item दो का उत्पाद कार्यों में वृद्धि पर $ I $ है \omterm{def:g11:func:monotone}{की बढ़ती} पर
        $ I $.
  \item अगर $ f $ है \omterm{def:g11:func:monotone}{की बढ़ती} पर $ I $, तब $ f^2$ ($= f \times f $) है
        \omterm{def:g11:func:monotone}{की बढ़ती} पर $ I $.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:func:11}
होने देना $ f(x) = x + \frac{1}{x}$ के लिए $ x > 0$.
\begin{enumerate}
  \item उसके लिए दिखाओ $0 < u < v $,
        $ f(v) - f(u) = (v - u)\,\frac{uv - 1}{uv}$.
  \item उसका निष्कर्ष निकालो $ f $ सख्ती से है \omterm{def:g10:functions:variations}{घटते} पर $\intoc{0}{1}$ और
        सख्ती से \omterm{def:g11:func:monotone}{की बढ़ती} पर $\intco{1}{+\infty}$.
  \item समाप्त करें कि $ x + \frac1x \geq 2$ सभी के लिए $ x > 0$, समानता के साथ
        केवल पर $ x = 1$.
\end{enumerate}
\end{exercise}

\section{समस्या: समरूपता का कारखाना}

\begin{problem}[{सप्ताहांत समस्या --- प्रत्येक फ़ंक्शन सम है भाग प्लस एक विषम भाग: समता नियम, परिवर्तन खेल, और मिनिमा बिना गणना के जीता}]
\label{pb:g11:func:1}
कुछ कार्य दर्पण-सममित हैं, कुछ अर्ध-मोड़ हैं
सममित, अधिकांश न तो हैं --- फिर भी एक छोटा बीजगणितीय चमत्कार है
ऐसा कहते हैं \emph{प्रत्येक} \omterm{def:g11:func:function}{समारोह} बिल्कुल एक तरह से, योग है
एक दर्पण-सममित टुकड़ा और एक आधा-मोड़-सममित टुकड़ा। यह
समस्या चमत्कार साबित करती है, परिवर्तन का खेल खेलती है
संदर्भ वक्र (\cref{prop:g11:func:transforms}), और बंद हो जाता है
के साथ अंतर-कारक की तकनीक
\cref{exo:g11:func:11}, जो न्यूनतमकरण समस्याओं को पूर्ण रूप से जीतता है
व्युत्पन्न आने से पहले अध्याय.

\medskip
\noindent\textbf{Part I --- Parity fluency.}
\begin{enumerate}
  \item \omterm{def:g11:func:parity}{यहां तक की}, \omterm{def:g11:func:parity}{विषम}, या न ही (\cref{def:g11:func:parity})?
        प्रत्येक को उचित ठहराएँ: $ x^4 - 3x^2$; \ $ x^3 - x $; \ $ x^2 + x $; \
        $\abs x $; \ $\frac1x $.
  \item प्रत्येक फैसले का ज्यामितीय \omterm{def:g10:stats:mean}{अर्थ} बताएं
        (\cref{prop:g11:func:paritysym}): कौन सा समरूपता करता है
        एक का वक्र यहां तक कि \omterm{def:g11:func:function}{समारोह} पास होना? की एक \omterm{def:g11:func:parity}{विषम} एक?
  \item समानता संकेतों की तरह गुणा करती है: सिद्ध करें कि का गुणनफल
        दो अजीब कार्य है \omterm{def:g11:func:parity}{यहां तक की}, और पूरा पूरा करें
        गुणन सारणी (\omterm{def:g11:func:parity}{यहां तक की} $\times $ \omterm{def:g11:func:parity}{यहां तक की}, \omterm{def:g11:func:parity}{यहां तक की} $\times $
        \omterm{def:g11:func:parity}{विषम}, \omterm{def:g11:func:parity}{विषम} $\times $ \omterm{def:g11:func:parity}{विषम}).
  \item त्वरित परिणाम: यदि $ f $ है \omterm{def:g11:func:parity}{विषम} और परिभाषित किया गया है $0$,
        क्या है $ f(0)$? अगर $ f $ है \omterm{def:g11:func:parity}{यहां तक की} साथ $ f(3) = 7$, क्या है
        $ f(-3)$? और वही दिखाओ \omterm{def:g11:func:function}{समारोह} दोनों \omterm{def:g11:func:parity}{यहां तक की} और
        \omterm{def:g11:func:parity}{विषम} शून्य है \omterm{def:g11:func:function}{समारोह}.
  \item कितना एक यहां तक कि समारोहका वक्र सभी को निर्धारित करता है
        यह? दे दो \omterm{def:g11:func:function}{समारोह} वह कुछ भी नहीं है \omterm{def:g11:func:parity}{यहां तक की} और न \omterm{def:g11:func:parity}{विषम}, और
        समझाएं कि क्या विफल रहता है.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The decomposition theorem.}
एक के लिए \omterm{def:g11:func:function}{समारोह} $ f $ ए पर परिभाषित \omterm{def:g11:func:function}{कार्यक्षेत्र} के बारे में सममित $0$, तय करना
\[
E(x) = \frac{f(x) + f(-x)}{2},
\qquad
O(x) = \frac{f(x) - f(-x)}{2} .
\]
\begin{enumerate}[resume]
  \item गणना करना $ E $ और $ O $ के लिए $ f(x) = x^3 + x^2 + 1$, और
        जांचें: $ E $ \omterm{def:g11:func:parity}{यहां तक की}, $ O $ \omterm{def:g11:func:parity}{विषम}, $ E + O = f $.
  \item सामान्य कथन सिद्ध करें: किसी के लिए $ f $, द \omterm{def:g11:func:function}{समारोह}
        $ E $ है \omterm{def:g11:func:parity}{यहां तक की}, $ O $ है \omterm{def:g11:func:parity}{विषम}, और $ f = E + O $ --- \emph{प्रत्येक
        \omterm{def:g11:func:function}{समारोह} एक में विभाजित हो जाता है \omterm{def:g11:func:parity}{यहां तक की} भाग और एक \omterm{def:g11:func:parity}{विषम} भाग}.
  \item साबित करें कि विभाजन अद्वितीय है: यदि
        $ f = E_1 + O_1 = E_2 + O_2$ साथ $ E_i $ \omterm{def:g11:func:parity}{यहां तक की} और $ O_i $
        \omterm{def:g11:func:parity}{विषम}, प्रश्न 4 का उपयोग करें $ E_1 - E_2$.
  \item विघटित $ f(x) = \dfrac{1}{1 - x}$ (के लिए
        $ x \neq \pm 1$): रखना $ E $ और $ O $ आम से ऊपर
        हर $1 - x^2$ और सरल करें.
  \item आगे एक नज़र, प्रत्येक एक वाक्य में: \omterm{def:g11:func:parity}{यहां तक की} और \omterm{def:g11:func:parity}{विषम}
        घातांक के भाग \omterm{def:g11:func:function}{समारोह} जुड़वाँ होगा
        कार्य ग्रेड 12 (हाइपरबोलिक कोसाइन और साइन); और
        वही वृत्ति --- किसी वस्तु को सममित में विभाजित करती है
        टुकड़े और प्रत्येक को उसकी अपनी समरूपता --- शक्तियों के साथ व्यवहार करें
        विश्वविद्यालय के संस्करणों में ध्वनि और संकेतों का विश्लेषण।
        ऐसा बंटवारा क्यों है \emph{उपयोगी}, सामान्य तौर पर?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The transformation game.}
\begin{enumerate}[resume]
  \item के वक्र से शुरू हो रहा है $ x \mapsto x^2$, वर्णन करें
        बिल्कुल के वक्र $(x - 3)^2$, \ $ x^2 + 2$, \
        $(x - 3)^2 + 2$, \ $-x^2$, \ $2x^2$
        (\cref{prop:g11:func:transforms}).
  \item कम करना $ f(x) = \abs{x - 2} + \abs{x + 2}$ एक को
        तीनों पर टुकड़ा-टुकड़ा फॉर्मूला \omterm{def:g10:numbers:interval}{अंतराल} द्वारा काटा गया $-2$
        और $2$, और इसके वक्र का वर्णन करें (इसमें एक प्रसिद्ध फ्लैट है
        नीचे)।
  \item कौन से परिवर्तन समता का सम्मान करते हैं? निर्णय लें, साथ में
        प्रमाण या प्रतिउदाहरण: क्षैतिज बदलाव
        $ f(x - 3)$; ऊर्ध्वाधर बदलाव $ f(x) + 2$; ऊर्ध्वाधर खिंचाव
        $2 f(x)$ (सभी एक पर लागू होते हैं \omterm{def:g11:func:parity}{यहां तक की} $ f $).
  \item हल करना $\abs{x - 2} + \abs{x + 2} = 6$ का उपयोग कर रहा हूँ
        प्रश्न 12 का टुकड़ावार सूत्र।
  \item रिवर्स इंजीनियरिंग: ए \omterm{def:g11:quad:parabola}{परवलय} शीर्ष है $(2, -3)$
        और गुजरता है $(0, 1)$. इसका पुनर्निर्माण करें \omterm{def:g10:algebra:equation}{समीकरण} में
        \omterm{thm:g11:quad:canonical}{कानूनी फॉर्म}.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Minima without calculus.}
\begin{enumerate}[resume]
  \item आईना \cref{exo:g11:func:11} के लिए
        $ g(x) = x + \dfrac4x $ पर $\intoo{0}{+\infty}$: दिखाओ
        $ g(v) - g(u) = (v - u)\,\dfrac{uv - 4}{uv}$, का पता लगाएं
        \omterm{def:g10:functions:extrema}{न्यूनतम}, और इसका मूल्य बताइये।
  \item लागू करें: क्षेत्रफल के सभी आयतों के बीच $16$, जिसके पास है
        \omterm{def:g10:functions:extrema}{न्यूनतम} परिधि? (लिखना $ P = 2\left(x +
        \frac{16}{x}\right)$ और प्रश्न 16 की मशीन का उपयोग करें।) एक
        पुराने मित्र ने निष्कर्ष निकाला.
  \item की विविधताएँ निर्धारित करें
        $ h(x) = \sqrt{x^2 + 1}$ सब पर $\R $, संयोजन ए
        रचना तर्क पर $\intco{0}{+\infty}$ एक के साथ
        बाकी के लिए समता तर्क.
  \item निकटतम बिंदु: वक्र का बिंदु ज्ञात करें
        $ y = \sqrt x $ के सबसे नजदीक $(3, 0)$. (\omterm{def:g10:functions:extrema}{न्यूनतम} करें
        \emph{Squared} दूरी--- एक द्विघात इंच $ x$ --- और
        बताएं कि वर्ग को छोटा करना वैध क्यों है।)
  \item समापन, फ़ैक्टरी दौरा: संदर्भ वक्र कच्चे हैं
        भण्डार; मशीनों का रूपांतरण; समता और
        गुणवत्ता नियंत्रण का विघटन; अंतर-कारक
        मापने की बेंच. प्रत्येक पर एक वाक्य --- और नाम
        अगला अध्याय बिजली उपकरण प्रदान करता है।
\end{enumerate}
\end{problem}
