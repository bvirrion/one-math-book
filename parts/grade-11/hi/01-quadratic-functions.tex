\chapter{द्विघात कार्य और समीकरण}\label{ch:g11:quad}

\omterm{def:g11:quad:quadratic}{द्विघात कार्य} सबसे सरल हैं कार्य रैखिक वाले के बाद, और
पहला किसका रेखांकन वास्तव में घुमावदार हैं. यह अध्याय विकसित करता है
उनके लिए पूरा टूलबॉक्स: \omterm{thm:g11:quad:canonical}{कानूनी फॉर्म}, द \omterm{def:g11:quad:discriminant}{विभेदक}, संकेत
एक द्विघात अभिव्यक्ति की, और की ज्यामिति \omterm{def:g11:quad:parabola}{परवलय}.

\section{विहित रूप}

\begin{definition}[द्विघात फलन]\label{def:g11:quad:quadratic}
ए \emph{द्विघात कार्य}\index{द्विघात कार्य} एक है \omterm{def:g10:functions:function}{समारोह} का
प्रपत्र
\[
f(x) = ax^2 + bx + c, \qquad a \neq 0,
\]
कहाँ $a $, $ b $, $ c $ हैं \omterm{def:g10:numbers:sets}{वास्तविक संख्या}. इजहार $ ax^2 + bx + c $ है
ए भी कहा जाता है \emph{द्विघात त्रिपद}.
\end{definition}

\begin{theorem}[कानूनी फॉर्म]\label{thm:g11:quad:canonical}
प्रत्येक \omterm{def:g11:quad:quadratic}{द्विघात कार्य} ठीक उसी तरह लिखा जा सकता है जैसे
\[
f(x) = a(x - \alpha)^2 + \beta,
\qquad\text{where } \alpha = -\frac{b}{2a} \text{ and } \beta = f(\alpha).
\]
ये है \emph{कानूनी फॉर्म}\index{कानूनी फॉर्म} का $ f $.
\end{theorem}

\begin{proof}
हम \emph{वर्ग पूरा करो}. तब से $ a \neq 0$, इसे पहले से गुणनखंडित करें
दो शर्तें:
\[
ax^2 + bx + c = a\left(x^2 + \frac{b}{a}x\right) + c .
\]
अब $ x^2 + \frac{b}{a}x $ वर्ग की शुरुआत है
$\left(x + \frac{b}{2a}\right)^2 = x^2 + \frac ba x + \frac{b^2}{4a^2}$,
इसलिए $ x^2 + \frac{b}{a}x = \left(x + \frac{b}{2a}\right)^2 -
\frac{b^2}{4a^2}$. प्रतिस्थापन:
\[
f(x) = a\left(x + \frac{b}{2a}\right)^2 - \frac{b^2}{4a} + c
= a\bigl(x - \alpha\bigr)^2 + \beta,
\]
साथ $\alpha = -\frac{b}{2a}$ और $\beta = c - \frac{b^2}{4a}$. मूल्यांकन करना
पर $ x = \alpha $ चुकता पद को ख़त्म कर देता है, इसलिए $\beta = f(\alpha)$. विशिष्टता:
का \omterm{def:g10:algebra:expand}{विस्तार} $ a(x-\alpha)^2 + \beta $ और के गुणांकों की पहचान करना $ x $
बल $\alpha = -\frac{b}{2a}$, और तब $\beta $ निर्धारित है.
\end{proof}

\begin{example}\label{ex:g11:quad:canonical}
होने देना $ f(x) = 2x^2 - 8x + 3$. तब $\alpha = -\frac{-8}{2 \times 2} = 2$ और
$\beta = f(2) = 8 - 16 + 3 = -5$, इसलिए
\[
f(x) = 2(x - 2)^2 - 5 .
\]
द्वारा जाँच का \omterm{def:g10:algebra:expand}{विस्तार}: $2(x^2 - 4x + 4) - 5 = 2x^2 - 8x + 8 - 5 =
2x^2 - 8x + 3$.
\end{example}

\begin{proposition}[चरम और विविधताएँ]\label{prop:g11:quad:variations}
होने देना $ f(x) = a(x-\alpha)^2 + \beta $.
\begin{enumerate}
  \item अगर $ a > 0$, तब $ f $ है \omterm{def:g10:functions:variations}{घटते} पर $\intoc{-\infty}{\alpha}$ और
        \omterm{def:g10:functions:variations}{की बढ़ती} पर $\intco{\alpha}{+\infty}$: $ f $ एक \emph{\omterm{def:g10:functions:extrema}{न्यूनतम}}
        $\beta $, पर प्राप्त किया गया $ x = \alpha $.
  \item अगर $ a < 0$, विविधताएं उलट हैं: $ f $ एक \emph{\omterm{def:g10:functions:extrema}{अधिकतम}}
        $\beta $ पर $ x = \alpha $.
\end{enumerate}
\end{proposition}

\begin{proof}
कल्पना करना $ a > 0$ और चलो $\alpha \leq u < v $. तब
\[
f(v) - f(u) = a\bigl((v-\alpha)^2 - (u-\alpha)^2\bigr)
= a\,(v - u)\,\bigl((v-\alpha) + (u-\alpha)\bigr).
\]
प्रत्येक कारक सकारात्मक है: $ a > 0$, $ v - u > 0$, और
$(v-\alpha) + (u-\alpha) > 0$ तब से $ v > \alpha $ और $ u \geq \alpha $. इसलिए
$ f(v) > f(u)$: $ f $ है \omterm{def:g10:functions:variations}{की बढ़ती} पर $\intco{\alpha}{+\infty}$. के लिए
$ u < v \leq \alpha $ अंतिम कारक नकारात्मक है, इसलिए $ f(v) < f(u)$: $ f $ है
\omterm{def:g10:functions:variations}{घटते} वहाँ। अंत में $ f(x) - f(\alpha) = a(x-\alpha)^2 \geq 0$ सभी के लिए
$ x $, इसलिए $\beta = f(\alpha)$ है \omterm{def:g10:functions:extrema}{न्यूनतम}. मामला $ a < 0$ द्वारा अनुसरण करता है
उपरोक्त को लागू करना $-f $.
\end{proof}

\begin{example}
एक किसान के पास है $100$ एक आयताकार मैदान को घेरने के लिए मीटर की बाड़
सीधी दीवार (दीवार के साथ बाड़ की आवश्यकता नहीं)। अगर $ x $ चौड़ाई है,
संलग्न क्षेत्र है $ A(x) = x(100 - 2x) = -2x^2 + 100x $. यहाँ $\alpha =
-\frac{100}{2\times(-2)} = 25$ और $ A(25) = 25 \times 50 = 1250$: क्षेत्र
की चौड़ाई के लिए सबसे बड़ा है $25$ मी, दे रहा हूँ $1250$ एम $^2$.
\end{example}

\section{जड़ें और कारकीकरण}

\begin{definition}[जड़, विवेकशील]\label{def:g11:quad:discriminant}
ए \emph{जड़}\index{जड़} त्रिपद का $ ax^2+bx+c $ का एक समाधान है
\omterm{def:g10:algebra:equation}{समीकरण} $ ax^2 + bx + c = 0$. \emph{विभेदक}\index{विभेदक}
त्रिपद की संख्या है
\[
\Delta = b^2 - 4ac .
\]
\end{definition}

\begin{theorem}[हल $ ax^2+bx+c=0$]\label{thm:g11:quad:roots}
\begin{enumerate}
  \item अगर $\Delta > 0$, द \omterm{def:g10:algebra:equation}{समीकरण} दो अलग-अलग हैं \omterm{def:g11:quad:discriminant}{जड़}ों
        \[
        x_1 = \frac{-b - \sqrt{\Delta}}{2a},
        \qquad
        x_2 = \frac{-b + \sqrt{\Delta}}{2a},
        \]
        और $ ax^2+bx+c = a(x - x_1)(x - x_2)$.
  \item अगर $\Delta = 0$, द \omterm{def:g10:algebra:equation}{समीकरण} एकल है \omterm{def:g11:quad:discriminant}{जड़}
        $ x_0 = -\frac{b}{2a}$, और $ ax^2+bx+c = a(x - x_0)^2$.
  \item अगर $\Delta < 0$, द \omterm{def:g10:algebra:equation}{समीकरण} कोई वास्तविक समाधान नहीं है, और
        त्रिपद को वास्तविक गुणांकों से गुणनखंडित नहीं किया जा सकता।
\end{enumerate}
\end{theorem}

\begin{proof}
से शुरू करें \omterm{thm:g11:quad:canonical}{कानूनी फॉर्म} में गणना की गई \cref{thm:g11:quad:canonical}:
\[
ax^2+bx+c
= a\left(x + \frac{b}{2a}\right)^2 + c - \frac{b^2}{4a}
= a\left[\left(x + \frac{b}{2a}\right)^2 - \frac{\Delta}{4a^2}\right],
\]
तब से $ c - \frac{b^2}{4a} = \frac{4ac - b^2}{4a} = -\frac{\Delta}{4a} =
a \times \left(-\frac{\Delta}{4a^2}\right)$.

\emph{1.} अगर $\Delta > 0$, तब $\frac{\Delta}{4a^2} =
\left(\frac{\sqrt\Delta}{2a}\right)^2$ और ब्रैकेट का अंतर है
दो वर्ग:
\[
ax^2+bx+c = a\left(x + \frac{b}{2a} - \frac{\sqrt\Delta}{2a}\right)
\left(x + \frac{b}{2a} + \frac{\sqrt\Delta}{2a}\right)
= a(x - x_2)(x - x_1).
\]
एक उत्पाद ठीक उसी समय शून्य होता है जब उसका एक कारक शून्य होता है, जिससे दोनों मिलते हैं
\omterm{def:g11:quad:discriminant}{जड़}ों, विशिष्ट क्योंकि $\sqrt\Delta \neq 0$.

\emph{2.} अगर $\Delta = 0$, ब्रैकेट है
$\left(x + \frac{b}{2a}\right)^2$, जो केवल पर ही लुप्त हो जाता है
$ x_0 = -\frac{b}{2a}$.

\emph{3.} अगर $\Delta < 0$, तब $-\frac{\Delta}{4a^2} > 0$, तो ब्रैकेट
एक वर्ग और एक धनात्मक संख्या है: यह कभी लुप्त नहीं होती, और \omterm{def:g10:algebra:equation}{समीकरण} है
कोई वास्तविक समाधान नहीं. एक गुणनखंडन $ a(x-x_1)(x-x_2)$ उत्पादन करेगा \omterm{def:g11:quad:discriminant}{जड़}ों, तो
कोई भी मौजूद नहीं है.
\end{proof}

\begin{example}\label{ex:g11:quad:solving}
हल करना $2x^2 - 8x + 3 = 0$. यहाँ $\Delta = (-8)^2 - 4 \times 2 \times 3 =
64 - 24 = 40 > 0$, तो दो हैं \omterm{def:g11:quad:discriminant}{जड़}ों:
\[
x_{1,2} = \frac{8 \pm \sqrt{40}}{4} = \frac{8 \pm 2\sqrt{10}}{4}
= 2 \pm \frac{\sqrt{10}}{2}.
\]
के लिए $ x^2 + x + 1 = 0$: $\Delta = 1 - 4 = -3 < 0$, कोई वास्तविक समाधान नहीं.
\end{example}

\begin{proposition}[जड़ों का योग और गुणनफल]\label{prop:g11:quad:vieta}
अगर $\Delta \geq 0$ और $ x_1, x_2$ हैं \omterm{def:g11:quad:discriminant}{जड़}ों (जब बराबर हो $\Delta = 0$),
तब
\[
x_1 + x_2 = -\frac{b}{a},
\qquad
x_1 x_2 = \frac{c}{a} .
\]
इसके विपरीत, योग सहित दो संख्याएँ $ s $ और उत्पाद $ p $ के समाधान हैं
$ x^2 - sx + p = 0$.
\end{proposition}

\begin{proof}
के गुणनखंडन का \omterm{def:g10:algebra:expand}{विस्तार} करें \cref{thm:g11:quad:roots}:
\[
a(x - x_1)(x - x_2) = a x^2 - a(x_1 + x_2)x + a\,x_1 x_2 .
\]
के साथ गुणांकों की पहचान करना $ ax^2 + bx + c $ देता है
$-a(x_1+x_2) = b $ और $ a\,x_1x_2 = c $. बातचीत के लिए, यदि $ u + v = s $ और
$ uv = p $, तब $(x-u)(x-v) = x^2 - sx + p $, इसलिए $ u $ और $ v $ इसके हैं \omterm{def:g11:quad:discriminant}{जड़}ों.
\end{proof}

\begin{example}
योग सहित दो संख्याएँ ज्ञात कीजिए $10$ और उत्पाद $21$. वे समाधान करते हैं
$ x^2 - 10x + 21 = 0$; $\Delta = 100 - 84 = 16$, \omterm{def:g11:quad:discriminant}{जड़}ों
$\frac{10 \pm 4}{2} = 7$ और $3$. \emph{जल्दी \omterm{def:g11:quad:discriminant}{जड़} खोलना:} के बाद से
के गुणांक $ x^2 - 5x + 4$ संतुष्ट $1 - 5 + 4 = 0$, जो नंबर $1$ एक है
\omterm{def:g11:quad:discriminant}{जड़}, और दूसरा है $\frac{c}{a} = 4$ (उनका उत्पाद होना चाहिए
$\frac ca $).
\end{example}

\section{द्विघात का चिन्ह}

\begin{theorem}[त्रिपद का चिन्ह]\label{thm:g11:quad:sign}
होने देना $ f(x) = ax^2+bx+c $ साथ $ a \neq 0$.
\begin{enumerate}
  \item अगर $\Delta > 0$, साथ \omterm{def:g11:quad:discriminant}{जड़}ों $ x_1 < x_2$: $ f(x)$ का चिन्ह है $ a $
        बाहर $\intcc{x_1}{x_2}$, और का चिन्ह $-a $ पर
        $\intoo{x_1}{x_2}$.
  \item अगर $\Delta = 0$: $ f(x)$ का चिन्ह है $ a $ सभी के लिए $ x \neq x_0$,
        और गायब हो जाता है $ x_0$.
  \item अगर $\Delta < 0$: $ f(x)$ का चिन्ह है $ a $ सभी के लिए $ x $.
\end{enumerate}
संक्षेप में: \emph{त्रिपद का चिन्ह होता है $ a $, इसके बीच को छोड़कर \omterm{def:g11:quad:discriminant}{जड़}ों}.
\end{theorem}

\begin{proof}
\emph{1.} लिखना $ f(x) = a(x-x_1)(x-x_2)$ और के संकेतों का पालन करें
कारक. अगर $ x < x_1$: दोनों $ x - x_1$ और $ x - x_2$ नकारात्मक हैं, उनके
उत्पाद सकारात्मक है, इसलिए $ f(x)$ का चिन्ह है $ a $. अगर $ x_1 < x < x_2$: द
कारकों के विपरीत संकेत हैं, उत्पाद नकारात्मक है, और $ f(x)$ के पास है
का संकेत $-a $. अगर $ x > x_2$: दोनों कारक सकारात्मक हैं और $ f(x)$ के पास है
का संकेत $ a $ दोबारा।

\emph{2.} $ f(x) = a(x - x_0)^2$ और $(x-x_0)^2 > 0$ के लिए $ x \neq x_0$.

\emph{3.} के प्रमाण से \cref{thm:g11:quad:roots},
$ f(x) = a\left[\left(x + \frac{b}{2a}\right)^2 +
\left(-\frac{\Delta}{4a^2}\right)\right]$, और ब्रैकेट हमेशा है
सकारात्मक.
\end{proof}

\begin{method}[द्विघात असमानताओं को हल करना]\label{met:g11:quad:inequalities}
समाधान करना $ ax^2 + bx + c \leq 0$ (या $\geq $, $<$, $>$):
\begin{enumerate}
  \item गणना $\Delta $ और यह \omterm{def:g11:quad:discriminant}{जड़}ों, यदि कोई;
  \item खींचना \omterm{def:g10:algebra:signtable}{साइन टेबल} का उपयोग करते हुए \cref{thm:g11:quad:sign} (हस्ताक्षर करना $ a $
        इसके बाहर \omterm{def:g11:quad:discriminant}{जड़}ों);
  \item समाधान पढ़ें \omterm{def:g10:numbers:interval}{अंतराल}(ओं), इस बात पर ध्यान देना कि क्या
        \omterm{def:g11:quad:discriminant}{जड़}ों स्वयं शामिल हैं (व्यापक असमानता) या नहीं (सख्त)।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g11:quad:inequality}
हल करना $-x^2 + 3x + 4 \geq 0$. यहाँ $\Delta = 9 + 16 = 25$, और यह \omterm{def:g11:quad:discriminant}{जड़}ों
हैं $\frac{-3 \pm 5}{-2}$, वह है $ x_1 = -1$ और $ x_2 = 4$. तब से
$ a = -1 < 0$, त्रिपद के बाहर नकारात्मक है \omterm{def:g11:quad:discriminant}{जड़}ों और सकारात्मक
उनके बीच. व्यापक असमानता के साथ, समाधान सेट है
$\intcc{-1}{4}$.
\end{example}

\section{परवलय}

\begin{definition}[परवलय]\label{def:g11:quad:parabola}
 \omterm{def:g10:functions:graph}{ग्राफ} एक का \omterm{def:g11:quad:quadratic}{द्विघात कार्य} $ f(x) = a(x-\alpha)^2+\beta $ एक है
\emph{परवलय}\index{परवलय} साथ \emph{शिखर} $ S(\alpha, \beta)$. यह
ऊपर की ओर खुलता है यदि $ a > 0$, नीचे की ओर यदि $ a < 0$, और के बारे में सममित है
ऊर्ध्वाधर रेखा $ x = \alpha $.
\end{definition}

\begin{remark}
समरूपता की जांच करना आसान है: किसी के लिए भी $ h $,
$ f(\alpha + h) = ah^2 + \beta = f(\alpha - h)$, तो दो अंक बराबर हैं
रेखा से क्षैतिज दूरी $ x = \alpha $ समान ऊंचाई हो.
\end{remark}

\begin{method}[एक परवलय पढ़ना]\label{met:g11:quad:reading}
रेखाचित्र बनाना $ f(x) = ax^2+bx+c $: अभिविन्यास खोजें (का संकेत)। $ a $), द
शिखर $\left(-\frac{b}{2a},\, f\!\left(-\frac{b}{2a}\right)\right)$, द
\omterm{def:g10:numbers:interunion}{चौराहा} $(0, c)$ साथ $ y $-अक्ष, और \omterm{def:g11:quad:discriminant}{जड़}ों, यदि कोई। की धुरी
समरूपता $ x = -\frac{b}{2a}$ फिर प्रत्येक परिकलित बिंदु को दो बार रखता है।
\end{method}

\begin{example}
के लिए $ f(x) = x^2 - 4x + 3$: ऊपर की ओर, शीर्ष पर $(2, -1)$, को पार करता है $ y $-अक्ष
पर $(0,3)$, \omterm{def:g11:quad:discriminant}{जड़}ों $1$ और $3$ (चित्तीदार: $1 - 4 + 3 = 0$). समरूपता से,
बिंदु $(4, 3)$ वक्र पर भी है, प्रतिबिम्बित $(0, 3)$.
\end{example}

\section{समस्या: परवलय का गुप्त बिंदु}

\begin{problem}[{सप्ताहांत समस्या --- फोकस और डायरेक्ट्रिक्स: क्यों सैटेलाइट डिश, हेडलाइट्स और सस्पेंशन ब्रिज सभी आकर्षित करते हैं वही वक्र, जिसमें मूलतः केवल एक ही है}]
\label{pb:g11:quad:1}
प्रत्येक बालकनी पर प्रत्येक उपग्रह डिश का रिसीवर एक पर लक्षित होता है
सटीक आंतरिक बिंदु; प्रत्येक हेडलाइट अपने बल्ब को छिपाती है
वही विशेष स्थान. वह बिंदु--- \emph{\omterm{pb:g11:quad:1}{केंद्र}} --- एक है
रहस्य \omterm{def:g11:quad:parabola}{परवलय} ग्रीक ज्यामिति के बाद से रखा गया है, और
इसे निकालने के लिए दूरी का फार्मूला ही काफी है. रास्ते में, यह
समस्या ड्रिल करती है \omterm{def:g11:quad:discriminant}{विभेदक}, दीवारों पर प्रक्षेप्य फेंकता है,
एक पुल को निलंबित कर देता है, और एक आश्चर्यजनक अंतिम तथ्य साबित करता है: तक
ज़ूम करें, \emph{वहां सिर्फ एक ही है \omterm{def:g11:quad:parabola}{परवलय} इस दुनिया में}.

\medskip
\noindent\textbf{Part I --- Trinomial fluency.}
\begin{enumerate}
  \item हल करना $ x^2 - 5x + 6 = 0$, तब $2x^2 + x - 6 = 0$, तब
        $ x^2 + x + 1 = 0$ (\cref{thm:g11:quad:roots}).
  \item लिखना $2x^2 - 4x - 6$ तथ्यात्मक रूप में और में \omterm{thm:g11:quad:canonical}{कानूनी फॉर्म} (\cref{thm:g11:quad:canonical}). साथ में
        \omterm{def:g10:algebra:expand}{विस्तार}ित रूप में, अब आपके पास एक की तीन पोशाकें हैं
        \omterm{def:g10:functions:function}{समारोह}: बताएं कि प्रत्येक एक नज़र में कौन सी सुविधा प्रदर्शित करता है
        (\omterm{def:g11:quad:discriminant}{जड़}ों, शिखर, $ y $-अवरोधन).
  \item हल करना $-x^2 + 4x - 3 \geq 0$ साइन नियम के साथ
        (\cref{thm:g11:quad:sign}).
  \item छठी\`जासूस के रूप में ete (\cref{prop:g11:quad:vieta},
        बाबुल में जारी है \cref{pb:g10:algebra:1}): दो खोजें
        योग के साथ संख्याएँ $7$ और उत्पाद $12$ उनके लिखकर
        \omterm{def:g10:algebra:equation}{समीकरण}. तब एक \omterm{def:g11:quad:discriminant}{जड़} का $ x^2 - 5x + c = 0$ के बराबर होती है
        $2$; खोजो $ c $ और दूसरा \omterm{def:g11:quad:discriminant}{जड़} बिना हल किये
        कुछ भी.
  \item किस मान के लिए $ m $ करता है $ x^2 + mx + 9 = 0$ एक है
        दोगुना \omterm{def:g11:quad:discriminant}{जड़}? नहीं \omterm{def:g11:quad:discriminant}{जड़}? दो \omterm{def:g11:quad:discriminant}{जड़}ों?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The \omterm{pb:g11:quad:1}{केंद्र} unveiled.}
पर काम करें \omterm{def:g11:quad:parabola}{परवलय} $ y = x^2$, बात
$ F\left(0, \frac14\right)$ और क्षैतिज रेखा $\delta $ का
\omterm{def:g10:algebra:equation}{समीकरण} $ y = -\frac14$.
\begin{enumerate}[resume]
  \item एक बिंदु के लिए $ P(x, x^2)$ की \omterm{def:g11:quad:parabola}{परवलय}, बढ़ाना
        $ PF^2 = x^2 + \left(x^2 - \frac14\right)^2$ और दिखाओ
        पूर्ण वर्ग है $\left(x^2 + \frac14\right)^2$.
        परिणाम निकालना $ PF $.
  \item से दूरी की गणना करें $ P $ लाइन तक $\delta $, और
        निष्कर्ष: \emph{के हर बिंदु \omterm{def:g11:quad:parabola}{परवलय} बिलकुल है
        जितना दूर $ F $ से के रूप में $\delta $}. ($ F $ है
        \emph{केंद्र}\index{केंद्र}, $\delta $ 
        \emph{नियता}\index{नियता}.)
  \item व्युत्क्रम सिद्ध करें: एक बिंदु $ P(x, y)$ साथ
        $ PF = \operatorname{dist}(P, \delta)$ संतुष्ट होना चाहिए
        $ y = x^2$. (दोनों दूरियों का वर्ग करें और सरल करें।)
        \omterm{def:g11:quad:parabola}{परवलय} इस प्रकार एक है \emph{ठिकाना}, बिलकुल वृत्त के समान
        में था \cref{pb:g10:coordgeom:1}: ए से समान दूरी
        एक बिंदु के बजाय बिंदु और एक रेखा।
  \item व्यापक के लिए \omterm{def:g11:quad:parabola}{परवलय} $ y = \frac{x^2}{4f}$ (साथ
        $ f > 0$), दिखाने के लिए प्रश्न 6 को अनुकूलित करें \omterm{pb:g11:quad:1}{केंद्र} है
        $(0, f)$ और यह \omterm{pb:g11:quad:1}{नियता} $ y = -f $. आवेदन: ए
        सैटेलाइट डिश है $60$ सेमी चौड़ा और $9$ सेमी गहरा --- इसका
        रिम गुजरता है $(30, 9)$. खोजो $ f $: कितना ऊपर
        रिसीवर को शीर्ष पर लटका देना चाहिए?
  \item व्यंजन के काम करने का कारण: किरणों का समानांतर आगमन
        अक्ष सभी प्रतिबिंबित करते हैं \emph{के माध्यम से \omterm{pb:g11:quad:1}{केंद्र}} (और एक बल्ब
        द \omterm{pb:g11:quad:1}{केंद्र} एक समानांतर किरण फेंकता है --- हेडलाइट्स हैं
        बर्तन पीछे की ओर चलते हैं)। बताएं कि प्रश्न 7 क्यों है
        समान दूरी इसे प्रशंसनीय बनाती है (किरणों के बारे में सोचें
        मार्चिंग वेवफ्रंट्स), और बताएं कि कौन सा उपकरण, अंदर आ रहा है
        \cref{ch:g11:deriv} (देखना \cref{exo:g11:quad:11}), बदल जाता है
        प्रमाण में संभाव्यता.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- \omterm{def:g11:quad:parabola}{परवलय} at work.}
\begin{enumerate}[resume]
  \item एक गेंद प्रक्षेप पथ का अनुसरण करती है
        $ y = x - \frac{x^2}{20}$ ($ x $ और $ y $ मीटर में)। खोजें
        जहां यह उतरता है, उसकी क्षैतिज स्थिति सबसे ऊंची होती है
        बिंदु, और \omterm{def:g10:functions:extrema}{अधिकतम} ऊंचाई
        (\cref{prop:g11:quad:variations}).
  \item उसी पथ पर एक दीवार खड़ी है $ x = 4$.
        वहां गेंद की ऊंचाई की गणना करें: क्या यह स्पष्ट है? $3$ एम
        दीवार? ए $4$ एम दीवार?
  \item एक \omterm{def:g11:quad:parabola}{परवलय}िक मेहराब फैला हुआ है $40$ \omterm{def:g10:functions:extrema}{अधिकतम} ऊंचाई के साथ मी
        $10$ मी: केन्द्रित में \omterm{def:g10:coordgeom:system}{COORDINATES},
        $ y = 10 - \frac{x^2}{40}$. एक ट्रक कर सकते हैं $7$ मी लंबा पास
        पर $10$ मैं \omterm{pb:g11:quad:1}{केंद्र} से? पर $15$ एम?
  \item टिकट की कीमत पर थिएटर का राजस्व $ p $ यूरो है
        $ R(p) = p\,(120 - 2p)$ (कीमत बढ़ने पर मांग गिरती है)।
        राजस्व को \omterm{def:g10:functions:extrema}{अधिकतम} करने वाली कीमत ज्ञात करें, और \omterm{def:g10:functions:extrema}{अधिकतम}.
  \item गोल्डन गेट के मुख्य केबल \omterm{def:g11:quad:parabola}{परवलय}िक हैं
        उत्कृष्ट सटीकता (एक समान डेक लोड उत्पन्न करता है
        \omterm{def:g11:quad:parabola}{परवलय} --- एक स्वतंत्र रूप से लटकी हुई चेन होती है \emph{नहीं}: इसका
        वक्र, कैटेनरी, को ग्रेड 12 के घातांक की आवश्यकता है)।
        \omterm{def:g10:algebra:expand}{विस्तार} के साथ $1\,280$ मी और शिथिलता $143$ मी, केबल इस प्रकार है
        $ y = 143\left(\frac{x}{640}\right)^2$ \omterm{pb:g11:quad:1}{केंद्र} से.
        केबल अपने निम्नतम बिंदु से कितनी ऊंचाई पर है
        $ x = 320$ एम?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Lines, tangents, and the last
surprise.}
\begin{enumerate}[resume]
  \item पूरा \cref{exo:g11:quad:11}: पंक्तियाँ
        $ y = mx - 1$ बाह्य बिंदु के माध्यम से $(0, -1)$ मिलो
        $ y = x^2$ ए के संकेत के अनुसार \omterm{def:g11:quad:discriminant}{विभेदक}.
        से दो स्पर्श रेखाएँ ज्ञात कीजिए $(0, -1)$ और उनके
        संपर्क बिंदु.
  \item अब पंक्तियाँ $ y = mx + 1$ के माध्यम से \emph{आंतरिक भाग}
        बिंदु $(0, 1)$: गणना करें \omterm{def:g11:quad:discriminant}{विभेदक} की
        \omterm{def:g10:numbers:interunion}{चौराहा} \omterm{def:g10:algebra:equation}{समीकरण} और दिखाएँ कि ऐसी प्रत्येक रेखा काटती है
        \omterm{def:g11:quad:parabola}{परवलय} दो बार। नैतिक: अंदर से, कोई स्पर्शरेखा नहीं ---
        जैसे एक वृत्त के लिए.
  \item फोकल कॉर्ड: प्रतिच्छेद $ y = x^2$ के साथ
        के माध्यम से क्षैतिज रेखा \omterm{pb:g11:quad:1}{केंद्र}, $ y = \frac14$. क्या
        क्या तार की लंबाई है? सामान्य नियम (मूल्य) की जाँच करें
        व्यंजन के लिए याद रखना): के लिए $ y = \frac{x^2}{4f}$, द
        के माध्यम से तार \omterm{pb:g11:quad:1}{केंद्र} के समानांतर \omterm{pb:g11:quad:1}{नियता} है
        लंबाई $4f $.
  \item अंतिम आश्चर्य: ज़ूम लागू करें
        $(x, y) \mapsto (kx, ky)$ तक \omterm{def:g11:quad:parabola}{परवलय} $ y = x^2$ और
        दिखाओ \omterm{def:g10:functions:function}{छवि} बिलकुल है $ y = \frac{x^2}{k}$. निष्कर्ष:
        प्रत्येक \omterm{def:g11:quad:parabola}{परवलय} $ y = \frac{x^2}{4f}$ एक आवर्धित इकाई है
        \omterm{def:g11:quad:parabola}{परवलय} --- विभिन्न वृत्तों और दीर्घवृत्तों के विपरीत
        अनुपात, \emph{सभी \omterm{def:g11:quad:parabola}{परवलय} समान हैं}.
  \item समापन: परवलयका चित्र पांच पंक्तियों में --- एक
        \omterm{def:g10:algebra:equation}{समीकरण} तीन वेशभूषा के साथ (प्रश्न 2), एक लोकस
        (प्रश्न 7-8), एक इंजीनियर का परावर्तक (प्रश्न 9,
        10), एक प्रक्षेप्य पथ और एक पुल की केबल
        (प्रश्न 11-15), और ज़ूम करने के लिए एक एकल वक्र
        (प्रश्न 19). एक-एक वाक्य.
\end{enumerate}
\end{problem}
