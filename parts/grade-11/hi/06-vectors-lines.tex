\chapter{समतल में सदिश और रेखाएँ}\label{ch:g11:vect}

वैक्टर विस्थापन को एन्कोड करें; रेखाएँ उनके प्रक्षेप पथ हैं। कुंजी नई
इस अध्याय का उपकरण है \emph{\omterm{def:g11:vect:det}{सिद्ध}}, एक एकल संख्या जो
तय करता है कि दो वैक्टर हैं \omterm{def:g11:vect:collinear}{समरेख} --- और, वहां से, उत्पादन करता है
\omterm{def:g10:algebra:equation}{समीकरण} रेखाओं का और शुद्ध रूप से संरेखण और समानता के प्रश्नों का समाधान करता है
गणना. वही विचार अंतरिक्ष तक \omterm{def:g10:algebra:expand}{विस्तार}ित हैं \cref{ch:g12:space}.

\section{वेक्टर: एक पुनर्कथन}

\begin{definition}[वेक्टर, निर्देशांक]\label{def:g11:vect:vector}
ए \emph{वेक्टर}\index{वेक्टर} $\vec u $ एक विस्थापन का प्रतिनिधित्व करता है: सभी
समान दिशा, अभिविन्यास और लंबाई वाले तीर समान दर्शाते हैं
वेक्टर. में एक \omterm{def:g10:coordgeom:system}{निर्देशांक तरीका}, $\vec u\,(x, y)$ विस्थापन को रिकार्ड करता है
($ x $ क्षैतिज रूप से, $ y $ लंबवत); अंक के लिए $ A(x_A, y_A)$ और
$ B(x_B, y_B)$,
\[
\vect{AB}\,\bigl(x_B - x_A,\ y_B - y_A\bigr).
\]
वेक्टर समन्वय-वार जोड़ते हैं, और $\lambda\vec u $ है \omterm{def:g10:coordgeom:system}{COORDINATES}
$(\lambda x, \lambda y)$.
\end{definition}

\begin{proposition}[मध्य]\label{prop:g11:vect:midpoint}
 \omterm{prop:g10:coordgeom:midpoint}{मध्य} $ M $ खंड का $[AB]$ है \omterm{def:g10:coordgeom:system}{COORDINATES}
$\left(\frac{x_A + x_B}{2},\ \frac{y_A + y_B}{2}\right)$.
\end{proposition}

\begin{proof}
$ M $ द्वारा चित्रित है $\vect{AM} = \frac12\vect{AB}$: समन्वय-वार,
$ x_M - x_A = \frac12(x_B - x_A)$, इसलिए $ x_M = \frac{x_A + x_B}{2}$, और
इसी तरह के लिए $ y_M $.
\end{proof}

\section{संरेखता और निर्धारक}

\begin{definition}[संरेख सदिश]\label{def:g11:vect:collinear}
दो वैक्टर हैं \emph{समरेख}\index{समरेख} यदि कोई है
अन्य का अदिश गुणज: $\vec v = \lambda \vec u $ कुछ के लिए
$\lambda \in \R $ (या $\vec u = \vec 0$). ज्यामितीय रूप से: उनके पास समान है
दिशा, या उनमें से एक शून्य है.
\end{definition}

\begin{definition}[सिद्ध]\label{def:g11:vect:det}
 \emph{सिद्ध}\index{सिद्ध} का $\vec u\,(x, y)$ और
$\vec v\,(x', y')$ है
\[
\det(\vec u, \vec v) =
\begin{vmatrix} x & x' \\ y & y' \end{vmatrix}
= x y' - x' y .
\]
\end{definition}

\begin{theorem}[संरेखता मानदंड]\label{thm:g11:vect:collinearity}
दो वैक्टर $\vec u $ और $\vec v $ हैं \omterm{def:g11:vect:collinear}{समरेख} अगर और केवल अगर
$\det(\vec u, \vec v) = 0$.
\end{theorem}

\begin{proof}
($\Rightarrow $) अगर $\vec v = \lambda\vec u $, तब $ x' = \lambda x $ और
$ y' = \lambda y $, इसलिए
$\det(\vec u, \vec v) = x(\lambda y) - (\lambda x)y = 0$; अगर
$\vec u = \vec 0$, द \omterm{def:g11:vect:det}{सिद्ध} तुच्छ है $0$.

($\Leftarrow $) कल्पना करना $ xy' - x'y = 0$ और $\vec u \neq \vec 0$, कहना
$ x \neq 0$ (मामला $ y \neq 0$ सममित है)। तय करना
$\lambda = \frac{x'}{x}$. तब $ x' = \lambda x $ निर्माण द्वारा, और
संबंध $ xy' = x'y = \lambda x y $ से विभाजित करके देता है $ x \neq 0$,
$ y' = \lambda y $. इस तरह $\vec v = \lambda \vec u $.
\end{proof}

\begin{example}\label{ex:g11:vect:collinear}
$\vec u\,(3, -2)$ और $\vec v\,(-6, 4)$:
$\det = 3 \times 4 - (-6) \times (-2) = 12 - 12 = 0$, \omterm{def:g11:vect:collinear}{समरेख} (वास्तव में
$\vec v = -2\vec u $). $\vec u\,(3, -2)$ और $\vec w\,(2, 1)$:
$\det = 3 + 4 = 7 \neq 0$, नहीं \omterm{def:g11:vect:collinear}{समरेख}.
\end{example}

\begin{method}[तीन बिंदुओं का संरेखण]\label{met:g11:vect:alignment}
$ A $, $ B $, $ C $ यदि और केवल यदि संरेखित हैं $\vect{AB}$ और $\vect{AC}$ हैं
\omterm{def:g11:vect:collinear}{समरेख}: दोनों की गणना करें वैक्टर और जांचें
$\det\bigl(\vect{AB}, \vect{AC}\bigr) = 0$.
\end{method}

\section{रेखाओं के कार्तीय समीकरण}

\begin{definition}[दिशा सदिश]\label{def:g11:vect:direction}
ए \emph{दिशा सदिश}\index{दिशा सदिश} एक पंक्ति का $ d $ कोई है
शून्येतर \omterm{def:g11:vect:vector}{वेक्टर} $\vec u $ ऐसा है कि $\vect{AB}$ है \omterm{def:g11:vect:collinear}{समरेख} साथ $\vec u $
सभी बिंदुओं के लिए $ A, B $ का $ d $.
\end{definition}

\begin{theorem}[कार्तीय समीकरण]\label{thm:g11:vect:cartesian}
के साथ एक पंक्ति \omterm{def:g11:vect:direction}{दिशा सदिश} $\vec u\,(-b, a)$ एक स्वीकार करता है \omterm{def:g10:algebra:equation}{समीकरण} का
प्रपत्र
\[
ax + by + c = 0
\qquad (a, b) \neq (0, 0).
\]
इसके विपरीत, प्रत्येक ऐसे \omterm{def:g10:algebra:equation}{समीकरण} के साथ एक पंक्ति का वर्णन करता है \omterm{def:g11:vect:direction}{दिशा सदिश}
$\vec u\,(-b, a)$.
\end{theorem}

\begin{proof}
होने देना $ d $ निकासी $ A(x_A, y_A)$ दिशा के साथ $\vec u\,(-b, a)$. ए
बिंदु $ M(x, y)$ पर स्थित है $ d $ बिल्कुल कब $\vect{AM}$ है \omterm{def:g11:vect:collinear}{समरेख} साथ
$\vec u $, यानी\ (\cref{thm:g11:vect:collinearity})
\[
0 = \det\bigl(\vect{AM}, \vec u\bigr)
= (x - x_A)\,a - (-b)(y - y_A)
= ax + by - (a x_A + b y_A),
\]
एक \omterm{def:g10:algebra:equation}{समीकरण} घोषित फॉर्म के साथ $ c = -(a x_A + b y_A)$.
इसके विपरीत, के समाधान $ ax + by + c = 0$ साथ, कहो, $ b \neq 0$ हैं
अंक $\left(x, -\frac{a x + c}{b}\right)$: उनमें से दो को घटाना
दिखाता है कि हर राग है \omterm{def:g11:vect:collinear}{समरेख} साथ $(-b, a)$, और हमेशा एक समाधान
मौजूद है --- यह इसके माध्यम से निर्देशित रेखा है $(-b, a)$.
\end{proof}

\begin{method}[दो बिंदुओं से होकर रेखा]\label{met:g11:vect:twopoints}
एक खोजने के लिए \omterm{def:g10:algebra:equation}{समीकरण} पंक्ति का $(AB)$: गणना करें \omterm{def:g11:vect:direction}{दिशा सदिश}
$\vect{AB}\,(x_B - x_A, y_B - y_A)$; इसके साथ मिलान करें $(-b, a)$, यानी\ लेना
$ a = y_B - y_A $ और $ b = -(x_B - x_A)$; ठानना $ c $ प्लग इन करके
\omterm{def:g10:coordgeom:system}{COORDINATES} का $ A $.
\end{method}

\begin{example}\label{ex:g11:vect:twopoints}
के माध्यम से रेखा $ A(1, 2)$ और $ B(4, 3)$: $\vect{AB}\,(3, 1)$, इसलिए
$ a = 1$, $ b = -3$, और यह \omterm{def:g10:algebra:equation}{समीकरण} है $ x - 3y + c = 0$. प्लग इन करना $ A $:
$1 - 6 + c = 0$, इसलिए $ c = 5$. \omterm{def:g10:algebra:equation}{समीकरण}: $ x - 3y + 5 = 0$. के साथ जांच $ B $:
$4 - 9 + 5 = 0$.
\end{example}

\begin{remark}
कब $ b \neq 0$, द \omterm{def:g10:algebra:equation}{समीकरण} के लिए हल किया जा सकता है $ y $: $ y = mx + p $ साथ
\omterm{def:g10:reffunc:affine}{ढलान} $ m = -\frac ab $. कार्तीय रूप $ ax + by + c = 0$ अधिक है
सामान्य: इसमें ऊर्ध्वाधर रेखाएँ भी शामिल हैं ($ b = 0$), जिसमें कोई नहीं है \omterm{def:g10:reffunc:affine}{ढलान}.
\end{remark}

\section{पैरामीट्रिक प्रतिनिधित्व}

\begin{definition}[पैरामीट्रिक प्रतिनिधित्व]\label{def:g11:vect:parametric}
के माध्यम से लाइन $ A(x_A, y_A)$ दिशा के साथ $\vec u\,(\alpha, \beta)$ है
अंकों का सेट
\[
M\bigl(x_A + t\alpha,\ y_A + t\beta\bigr), \qquad t \in \R .
\]
जो नंबर $ t $ है \emph{पैरामीटर}: का प्रत्येक मान $ t $ एक पैदा करता है
रेखा का बिंदु, मानो उस पर स्थिर गति से यात्रा कर रहा हो $\vec u $.
\end{definition}

\begin{example}
की पंक्ति \cref{ex:g11:vect:twopoints} ई आल्सो
$\{(1 + 3t,\ 2 + t) : t \in \R\}$: $ t = 0$ देता है $ A $, $ t = 1$ देता है $ B $.
खत्म करना $ t $ ($ t = y - 2$, इसलिए $ x = 1 + 3(y-2)$) ठीक हो जाता है
$ x - 3y + 5 = 0$.
\end{example}

\section{दो रेखाओं की सापेक्ष स्थिति}

\begin{proposition}[समानांतर या प्रतिच्छेदी]\label{prop:g11:vect:positions}
दो पंक्तियों के साथ \omterm{def:g11:vect:direction}{दिशा सदिश} $\vec u $ और $\vec v $ यदि और समानांतर हैं
केवल यदि $\det(\vec u, \vec v) = 0$. में \omterm{def:g10:algebra:equation}{समीकरण} रूप: $ d : ax + by + c =
0$ और $ d' : a'x + b'y + c' = 0$ यदि और केवल यदि समानान्तर हैं
$ ab' - a'b = 0$. गैर-समानांतर रेखाएँ बिल्कुल एक बिंदु पर मिलती हैं।
\end{proposition}

\begin{proof}
समांतरता का \omterm{def:g10:stats:mean}{अर्थ} है \omterm{def:g11:vect:collinear}{समरेख} दिशाएँ, जो है
\cref{thm:g11:vect:collinearity}; साथ $\vec u\,(-b, a)$ और
$\vec v\,(-b', a')$, द \omterm{def:g11:vect:det}{सिद्ध} है $(-b)a' - (-b')a = ab' - a'b $.
यदि रेखाएं समानांतर नहीं हैं, तो दोनों को हल करें \omterm{def:g10:algebra:equation}{समीकरण} एक साथ
(प्रतिस्थापन या संयोजन द्वारा) एक अद्वितीय समाधान की ओर ले जाता है: प्रणाली
की तरह व्यवहार करता है अशून्य-निर्धारक $2 \times 2$ प्रणाली।
\end{proof}

\begin{method}[दो रेखाओं का प्रतिच्छेदन]\label{met:g11:vect:intersection}
दोनों की प्रणाली को हल करें \omterm{def:g10:algebra:equation}{समीकरण}: एक अज्ञात को एक में अलग करें \omterm{def:g10:algebra:equation}{समीकरण}
और दूसरे में प्रतिस्थापित करें, या संयोजित करें \omterm{def:g10:algebra:equation}{समीकरण} एक को ख़त्म करना
अज्ञात. परिणामी बिंदु को हमेशा जांचें \emph{दोनों} \omterm{def:g10:algebra:equation}{समीकरण}.
\end{method}

\begin{example}\label{ex:g11:vect:intersection}
इंटरसेक्ट $ d: x - 3y + 5 = 0$ और $ d': 2x + y - 4 = 0$. से $ d'$:
$ y = 4 - 2x $. में प्रतिस्थापित करना $ d $:
\[
x - 3(4 - 2x) + 5 = 0
\iff x - 12 + 6x + 5 = 0
\iff 7x = 7
\iff x = 1,
\]
तब $ y = 2$. रेखाएं मिलती हैं $(1, 2)$. चेक इन $ d $: $1 - 6 + 5 = 0$.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g11:vect:1}
दिया गया $ A(2, -1)$, $ B(5, 3)$ और $ C(-1, 1)$, गणना करें \omterm{def:g10:coordgeom:system}{COORDINATES} का
$\vect{AB}$, $\vect{AC}$, $\vect{AB} + \vect{AC}$ और $2\vect{AB} -
3\vect{AC}$, और यह \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[BC]$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:vect:2}
हैं वैक्टर \omterm{def:g11:vect:collinear}{समरेख}?
\[
\vec u\,(4, -6) \text{ and } \vec v\,(-2, 3); \qquad
\vec u\,(2, 5) \text{ and } \vec v\,(3, 7); \qquad
\vec u\,(0, 3) \text{ and } \vec v\,(0, -8).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:vect:3}
एक कार्टेशियन खोजें \omterm{def:g10:algebra:equation}{समीकरण} लाइन के माध्यम से $ A(2, 1)$ साथ \omterm{prop:g10:lines:parallel}{दिशा
वेक्टर} $\vec u\,(3, -1)$, फिर लाइन के माध्यम से $ B(0, 4)$ और
$ C(2, 0)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:vect:4}
लाइन के लिए $ d: 3x - 2y + 6 = 0$, दे दो \omterm{def:g11:vect:direction}{दिशा सदिश}, द
चौराहों दो समन्वय अक्षों के साथ, और तय करें कि क्या बिंदु हैं
$ P(2, 6)$ और $ Q(1, 4)$ के संबंधित $ d $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:vect:5}
बिंदु हैं $ A(1, 2)$, $ B(4, 8)$, $ C(-2, -4)$ संरेखित? वही सवाल
के लिए $ D(0, 1)$, $ E(2, 4)$, $ F(5, 9)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:vect:6}
निर्धारित करें \omterm{def:g10:numbers:interunion}{चौराहा} बिंदु, यदि कोई हो:
\[
d: 2x + y - 7 = 0 \ \text{ and }\ d': x - y + 1 = 0;
\qquad
e: x - 2y + 3 = 0 \ \text{ and }\ e': -2x + 4y + 1 = 0 .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:vect:7}
रेखा का पैरामीट्रिक निरूपण दीजिए $ d: 2x - y + 3 = 0$, और ए
कार्तीय \omterm{def:g10:algebra:equation}{समीकरण} पंक्ति का
$\bigl\{(1 + 2t,\ -3 + t) : t \in \R\bigr\}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:vect:8}
खोजें \omterm{def:g10:algebra:equation}{समीकरण} लाइन के माध्यम से $ P(1, -2)$ के समानांतर
$ d: 4x - 3y + 2 = 0$, और का मूल्य $ m $ जिसके लिए लाइन
$ mx + 2y - 5 = 0$ के समानांतर है $ d $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:vect:9}
होने देना $ A(-1, 0)$, $ B(3, 2)$ और $ C(1, -4)$. एक कार्टेशियन खोजें \omterm{def:g10:algebra:equation}{समीकरण} की
\emph{\omterm{def:g10:stats:median}{MEDIAN}} त्रिकोण का $ ABC $ से जारी किया गया $ C $ (जोड़ने वाली रेखा $ C $
तक \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[AB]$).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:vect:10}
पैरामीटर के किस मान के लिए $ m $ हैं वैक्टर
$\vec u\,(m, 2)$ और $\vec v\,(3, m - 1)$ \omterm{def:g11:vect:collinear}{समरेख}?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:vect:11}
होने देना $ ABCD $ एक समांतर चतुर्भुज हो, $ I $  \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[AB]$, और $ J $ द
बिंदु द्वारा परिभाषित $\vect{DJ} = \frac23 \vect{DI}$. में काम कर रहा हूँ
\omterm{def:g10:coordgeom:system}{निर्देशांक तरीका} कहाँ $ A(0,0)$, $ B(1,0)$, $ D(0,1)$ (इसलिए $ C(1,1)$):
\begin{enumerate}
  \item की गणना करें \omterm{def:g10:coordgeom:system}{COORDINATES} का $ I $ और $ J $.
  \item बताते हैं कि $ A $, $ J $ और $ C $ संरेखित हैं. (एक अच्छा तथ्य: पंक्ति
        $(DI)$ विकर्ण को काटता है $(AC)$ रास्ते का एक तिहाई ऊपर।)
\end{enumerate}
\end{exercise}

\section{समस्या: टकराव पाठ्यक्रम और चतुर निर्देशांक}

\begin{problem}[{सप्ताहान्त समस्या --- रास्ता पार करना नहीं है टकराना: समुद्र में निकटतम दृष्टिकोण, और प्रमेय द्वारा सिद्ध सही फ़्रेम चुनना}]
\label{pb:g11:vect:1}
दो जहाज़ सीधे रास्ते पार करते हैं --- क्या जहाज़ों को टकराना चाहिए? नहीं
जब तक वे क्रॉसिंग तक नहीं पहुंच जाते \emph{एक ही समय पर}: पथ हैं
बिंदुओं के सेट, गतियाँ पैरामीट्रिक हैं, और दोनों को भ्रमित करने वाला है
असली जहाज डूब गए. यह समस्या एक छोटी जहाज-यातायात सेवा चलाती है
पैरामीट्रिक लाइनों पर (\cref{def:g11:vect:parametric}), फिर बदल जाता है
की अन्य महाशक्ति के साथ शुद्ध ज्यामिति की ओर \omterm{def:g10:coordgeom:system}{COORDINATES}:
\emph{का चयन} उन्हें चतुराई से, जैसे कि \cref{exo:g11:vect:11}, को
शास्त्रीय प्रमेयों की गणना स्वयं करें।

\medskip
\noindent\textbf{Part I --- Lines, three ways.}
\begin{enumerate}
  \item इसके माध्यम से जाने वाली रेखा का पैरामीट्रिक निरूपण दीजिए
        $ A(1, 2)$ साथ \omterm{def:g11:vect:direction}{दिशा सदिश} $\vec u(3, -1)$, फिर
        कार्टेशियन प्राप्त करने के लिए पैरामीटर को हटा दें \omterm{def:g10:algebra:equation}{समीकरण}.
  \item लाइन के लिए $2x - 5y + 3 = 0$: पढ़ो ए \omterm{prop:g10:lines:parallel}{दिशा
        वेक्टर} (\cref{thm:g11:vect:cartesian}), एक बिंदु खोजें,
        और एक पैरामीट्रिक प्रतिनिधित्व लिखें।
  \item की गणना करें \omterm{def:g10:numbers:interunion}{चौराहा} की दो पंक्तियों का बिंदु
        प्रश्न 1 और 2 (\cref{met:g11:vect:intersection}).
  \item से पुष्टि करें \omterm{def:g11:vect:det}{सिद्ध}
        (\cref{thm:g11:vect:collinearity}) वह दो पंक्तियाँ
        काटना पड़ा; तो फिर दिखाओ $2x - 5y + 3 = 0$ और
        $-4x + 10y + 1 = 0$ बिल्कुल समानांतर हैं.
  \item हैं $ A(2, 1)$, $ B(5, 3)$, $ C(11, 7)$ संरेखित
        (\cref{met:g11:vect:alignment})?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The vessel-traffic service.}
समय पर $ t $ (घंटे), जहाज़ $ P $ पर है $(2t,\ 3 + t)$ और जहाज $ Q $
पर $(10 - t,\ 2t - 1)$ (दूरियां समुद्री मील में)।
\begin{enumerate}[resume]
  \item कार्टेशियन खोजें \omterm{def:g10:algebra:equation}{समीकरण} दोनों में से \emph{के रास्ते},
        और गणना करें कि रास्ते कहां से गुजरते हैं।
  \item प्रत्येक जहाज किस समय क्रॉसिंग पॉइंट से गुजरता है? करो
        जहाज़ टकराते हैं? सामान्य चेतावनी एक में बताएं
        वाक्य: बैठक के प्रस्तावों के विरुद्ध रास्ता पार करना।
  \item वर्ग दूरी की गणना करें $ D^2(t)$ जहाजों के बीच
        समय पर $ t $, और इसे एक द्विघात में घटाएं $ t $. पर
        जहाज़ किस समय निकटतम हैं, और वे कितने निकट हैं
        आओ (\cref{pb:g11:quad:1} वर्टेक्स टेक्नोलॉजी)?
  \item विनियम कम से कम अलगाव की मांग करते हैं $1$ मील.
        क्या इसका उल्लंघन हुआ है? यदि हां तो समाधान करें $ D^2(t) < 1$ और दे दो
        उल्लंघन की अवधि.
  \item स्पष्ट करें क्यों, के लिए \emph{कोई} दो जहाज सीधे
        निरंतर गति से पाठ्यक्रम, $ D^2(t)$ हमेशा एक है
        \omterm{def:g11:quad:quadratic}{द्विघात कार्य} का $ t $ --- तो ``निकटतम बिंदु
        दृष्टिकोण'' हमेशा एक शीर्ष गणना है।
  \item अवरोधन: एक गश्ती नाव उद्गम स्थल से प्रारंभ होती है
        $ t = 0$ निरंतर वेग के साथ $(a, b)$ और \omterm{def:g10:numbers:interunion}{मिलन}ा चाहिए
        जहाज़ $ P $ बिलकुल पर $ t = 3$. आवश्यक की गणना करें
        $(a, b)$ और गश्ती नाव की गति।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Clever \omterm{def:g10:coordgeom:system}{COORDINATES}.}
के फ्रेम में काम करें \cref{exo:g11:vect:11}: समांतर चतुर्भुज
$ ABCD $ बन जाता है $ A(0,0)$, $ B(1,0)$, $ C(1,1)$, $ D(0,1)$; होने देना $ I $
हो \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[AB]$ और $ K $  \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[CD]$.
\begin{enumerate}[resume]
  \item कार्टेशियन की गणना करें \omterm{def:g10:algebra:equation}{समीकरण} पंक्ति का $(DI)$.
  \item इंटरसेक्ट $(DI)$ विकर्ण के साथ $(AC)$: वापस पाना
        \cref{exo:g11:vect:11}तथ्य यह है--- कट बिलकुल झूठ है
        विकर्ण तक एक तिहाई रास्ता।
  \item अब लाइन $(BK)$: इसका पता लगाएं \omterm{def:g10:algebra:equation}{समीकरण}, के साथ प्रतिच्छेद करें
        $(AC)$, और पूर्ण शास्त्रीय प्रमेय समाप्त करें:
        \emph{सेवियन $(DI)$ और $(BK)$ विकर्ण को त्रिविभाजित करें
        $(AC)$}.
  \item इसमें प्रमेय को \omterm{def:g11:vect:det}{सिद्ध} करना वैध क्यों था?
        विशेष फ्रेम? (इसका चुनाव क्या करता है? \omterm{def:g10:coordgeom:system}{COORDINATES}
        संरक्षित करें, और कथन में क्या शामिल है?) उत्तर
        दो या तीन वाक्यों में.
  \item अभ्यास के लिए एक और ट्राइसेक्शन: चलो $ E $ हो
        \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[BC]$. रेखा कहाँ है $(AE)$ काटो
        \emph{अन्य} विकर्ण $(DB)$?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Position puzzles.}
\begin{enumerate}[resume]
  \item तीन पंक्तियाँ हैं $ x + y = 4$, \ $2x - y = 2$, \
        $ x - 2y = -2$ समवर्ती? (दो को काटें, परीक्षण करें
        तीसरा.)
  \item प्रत्येक वास्तविक के लिए $ m $, होने देना $ d_m $ पंक्ति हो
        $ mx - y + 2 - 3m = 0$. बताते हैं कि \emph{प्रत्येक} $ d_m $
        एक से गुजरता है \omterm{pb:g10:functions:1}{नियत बिन्दु}. (शर्तों को समूह में रखें
        $ m $.)
  \item परिवार में $(d_m)$: कौन सा सदस्य समानांतर है
        $ y = 2x + 7$? जो मूल से होकर गुजरता है?
  \item समापन --- एक पंक्ति की तीन पोशाकें: कार्टेशियन
        (एक प्रतिस्थापन में सदस्यता परीक्षण), कम (\omterm{def:g10:reffunc:affine}{ढलान}
        और \omterm{def:g10:functions:graph}{ग्राफ} दृष्टि में), पैरामीट्रिक (गति, समय,
        अवरोधन); \omterm{def:g11:vect:det}{सिद्ध} समानता दैवज्ञ के रूप में;
        और प्रमेय कारखाने के रूप में अच्छी तरह से चुना गया फ्रेम। एक
        प्रत्येक वाक्य.
\end{enumerate}
\end{problem}
