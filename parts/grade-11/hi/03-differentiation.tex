\chapter{विभेदन: एक पहला कोर्स}\label{ch:g11:deriv}

कितनी तेजी से करता है ए \omterm{def:g11:func:function}{समारोह} परिवर्तन? एक ही बिंदु पर उत्तर है
\emph{यौगिक}, लागू सभी में से एकमात्र सबसे उपयोगी संख्या
गणित. यह अध्याय \omterm{def:g11:deriv:rate}{परिवर्तन की औसत दर}ों से इसका निर्माण करता है,
के लिए इसकी गणना करता है \omterm{def:g11:func:reference}{संदर्भ कार्य}, और पढ़ने के लिए इसके चिह्न का उपयोग करता है
ए की विविधताएँ \omterm{def:g11:func:function}{समारोह}. यहां सब कुछ विस्तृत और गहरा है
\cref{ch:g12:deriv}.

\section{परिवर्तन की औसत दर}

\begin{definition}[परिवर्तन की औसत दर]\label{def:g11:deriv:rate}
होने देना $f $ एक पर परिभाषित किया जाए \omterm{def:g10:numbers:interval}{अंतराल} $ I $, और $ a \neq b $ में $ I $.
\emph{परिवर्तन की औसत दर}\index{परिवर्तन की औसत दर} का $ f $ बीच में
$ a $ और $ b $ है
\[
\frac{f(b) - f(a)}{b - a}.
\]
यह है \omterm{def:g10:reffunc:affine}{ढलान} की \emph{काटनेवाला}\index{काटनेवाला} बिंदुओं के माध्यम से रेखा
$\bigl(a, f(a)\bigr)$ और $\bigl(b, f(b)\bigr)$ वक्र का.
\end{definition}

\begin{example}
एक कार यात्रा करती है $140$ दोपहर 2 बजे के बीच किमी.\ और शाम 4 बजे: इसकी औसत गति है
$\frac{140}{2} = 70$ किमी/घंटा. हालाँकि, स्पीडोमीटर एक प्रदर्शित करता है
\emph{तात्कालिक} हर क्षण गति --- बिल्कुल यही धारणा है
वह यौगिक सटीक बनाता है.
\end{example}

\section{एक बिंदु पर व्युत्पन्न}

\begin{definition}[एक बिंदु पर व्युत्पन्न]\label{def:g11:deriv:derivative}
होने देना $ f $ एक पर परिभाषित किया जाए \omterm{def:g10:numbers:interval}{अंतराल} युक्त $ a $. के लिए $ h \neq 0$, इससे
\omterm{def:g11:deriv:rate}{परिवर्तन की औसत दर} बीच में $ a $ और $ a + h $:
\[
r(h) = \frac{f(a+h) - f(a)}{h}.
\]
अगर $ r(h)$ एक निश्चित संख्या के पास पहुँचता है $ h $ मनमाने ढंग से करीब आ जाता है
$0$, हम ऐसा कहते हैं $ f $ है \emph{पर भिन्न $ a $}\index{पर भिन्न $ a $}\index{विभेदक};
यह संख्या है \emph{का व्युत्पन्न $ f $ पर $ a $}, लिखा हुआ $ f'(a)$:
\[
f'(a) = \lim_{h \to 0} \frac{f(a+h) - f(a)}{h}.
\]
\end{definition}

\begin{example}\label{ex:g11:deriv:atpoint}
होने देना $ f(x) = x^2$ और $ a = 1$. के लिए $ h \neq 0$:
\[
\frac{f(1+h) - f(1)}{h}
= \frac{(1+h)^2 - 1}{h}
= \frac{1 + 2h + h^2 - 1}{h}
= \frac{h(2 + h)}{h}
= 2 + h .
\]
जैसा $ h $ दृष्टिकोण $0$, यह दृष्टिकोण $2$: द \omterm{def:g11:func:function}{समारोह} $ x^2$ है
\omterm{def:g11:deriv:derivative}{विभेदक} पर $1$ और $ f'(1) = 2$. रणनीति पर ध्यान दें: \emph{सरल करें
तक भागफल $ h $ अब हर में प्रकट नहीं होता}, तो करने दें
$ h \to 0$.
\end{example}

\begin{example}[एक अभेद्य बिंदु]
होने देना $ f(x) = \abs{x}$ और $ a = 0$. तब
$\frac{f(h) - f(0)}{h} = \frac{\abs h}{h}$, जो बराबर है $1$ के लिए $ h > 0$
और $-1$ के लिए $ h < 0$: किसी एक नंबर पर संपर्क नहीं किया गया है, और $\abs x $ क्या नहीं है
\omterm{def:g11:deriv:derivative}{विभेदक} पर $0$. इसके वक्र में एक है \emph{कोना} वहाँ।
\end{example}

\section{स्पर्श रेखा}

\begin{definition}[स्पर्शरेखा]\label{def:g11:deriv:tangent}
अगर $ f $ है \omterm{def:g11:deriv:derivative}{विभेदक} पर $ a $, द \emph{स्पर्शरेखा}\index{स्पर्शरेखा} को
बिंदु पर वक्र $ A = \bigl(a, f(a)\bigr)$ के माध्यम से लाइन है $ A $ साथ
\omterm{def:g10:reffunc:affine}{ढलान} $ f'(a)$. इसका \omterm{def:g10:algebra:equation}{समीकरण} है
\[
y = f(a) + f'(a)(x - a).
\]
\end{definition}

 \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} की सीमित स्थिति है सेकेंट्स के माध्यम से $ A $: जैसा
$ h \to 0$, दूसरा बिंदु $\bigl(a+h, f(a+h)\bigr)$ के साथ स्लाइड करता है
की ओर वक्र $ A $, और यह \omterm{def:g11:deriv:rate}{काटनेवाला}'एस \omterm{def:g10:reffunc:affine}{ढलान} $\frac{f(a+h)-f(a)}{h}$
दृष्टिकोण $ f'(a)$.

\begin{example}\label{ex:g11:deriv:tangenteq}
के लिए $ f(x) = x^2$ पर $ a = 1$: $ f(1) = 1$ और $ f'(1) = 2$
(\cref{ex:g11:deriv:atpoint}), इतना \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} है
\[
y = 1 + 2(x - 1) = 2x - 1 .
\]
ये लाइन है $ d_2$ में शुद्ध बीजगणित द्वारा पाया गया \cref{exo:g11:quad:11}.
\end{example}

\section{संदर्भ कार्यों के व्युत्पन्न}

\begin{proposition}\label{prop:g11:deriv:refderivatives}
पर उनके डोमेन भिन्नता की:
\[
(c)' = 0, \quad
(x)' = 1, \quad
(x^2)' = 2x, \quad
(x^3)' = 3x^2, \quad
\left(\frac1x\right)' = -\frac{1}{x^2}, \quad
(\sqrt x\,)' = \frac{1}{2\sqrt x} \ (x > 0).
\]
आम तौर पर अधिक, $(x^n)' = n\,x^{n-1}$ हरएक के लिए पूर्णांक $ n \geq 1$.
\end{proposition}

\begin{proof}
प्रत्येक प्रमाण की रणनीति का अनुसरण करता है \cref{ex:g11:deriv:atpoint}: सरल करें
दर, तो चलो $ h \to 0$. हल करना $ a $ में \omterm{def:g11:func:function}{कार्यक्षेत्र}.

\emph{निरंतर और पहचान.} $\frac{c - c}{h} = 0$ और
$\frac{(a+h) - a}{h} = 1$ सभी के लिए $ h $: सीमाएँ हैं $0$ और $1$.

\emph{वर्ग।} $\frac{(a+h)^2 - a^2}{h} = \frac{2ah + h^2}{h} = 2a + h
\to 2a $.

\emph{घन.} $(a+h)^3 = a^3 + 3a^2h + 3ah^2 + h^3$, तो यह है रेट
$3a^2 + 3ah + h^2 \to 3a^2$.

\emph{पारस्परिक।} के लिए $ a \neq 0$ और $ h $ इतना छोटा कि
$ a + h \neq 0$:
\[
\frac{1}{h}\left(\frac{1}{a+h} - \frac{1}{a}\right)
= \frac{1}{h} \cdot \frac{a - (a+h)}{a(a+h)}
= \frac{-1}{a(a+h)}
\xrightarrow[h \to 0]{} -\frac{1}{a^2}.
\]

\emph{वर्गमूल.} के लिए $ a > 0$, संयुग्म द्वारा गुणा करें:
\[
\frac{\sqrt{a+h} - \sqrt a}{h}
= \frac{(a + h) - a}{h\,(\sqrt{a+h} + \sqrt a)}
= \frac{1}{\sqrt{a+h} + \sqrt a}
\xrightarrow[h \to 0]{} \frac{1}{2\sqrt a}.
\]

\emph{सामान्य शक्ति.} \omterm{def:g10:proba:sample}{नमूना} $2a $, $3a^2$ जारी है: का \omterm{def:g10:algebra:expand}{विस्तार}
$(a+h)^n $, बाद की शर्तें $ a^n + n\,a^{n-1}h $ सभी ले जाते हैं $ h^2$, तो
दर की प्रवृत्ति होती है $ n\,a^{n-1}$. (पूर्ण प्रेरण इसमें दिया गया है
\cref{ch:g12:deriv}.)
\end{proof}

\section{विभेदन नियम}

\begin{theorem}[संचालन]\label{thm:g11:deriv:rules}
होने देना $ u $ और $ v $ होना \omterm{def:g11:deriv:derivative}{विभेदक} पर $ I $ और $\lambda \in \R $. तब
$ u + v $, $\lambda u $, $ uv $ हैं \omterm{def:g11:deriv:derivative}{विभेदक} पर $ I $, जैसा है
$\frac{u}{v}$ जहाँ भी $ v \neq 0$, और
\[
(u+v)' = u' + v', \qquad
(\lambda u)' = \lambda u', \qquad
(uv)' = u'v + uv', \qquad
\left(\frac{u}{v}\right)' = \frac{u'v - uv'}{v^2}.
\]
\end{theorem}

\begin{proof}[पहले तीन नियमों का प्रमाण]
ठीक करो $ a \in I $ और संक्षिप्त करें $\Delta u = u(a+h) - u(a)$,
$\Delta v = v(a+h) - v(a)$.

\emph{\omterm{def:g10:vectors:sum}{जोड़}।} की दर $ u + v $ है
$\frac{\Delta u + \Delta v}{h} = \frac{\Delta u}{h} + \frac{\Delta v}{h}
\to u'(a) + v'(a)$.

\emph{एकाधिक.} की दर $\lambda u $ है
$\lambda \frac{\Delta u}{h} \to \lambda u'(a)$.

\emph{उत्पाद।} मिश्रित पद को \omterm{def:g10:vectors:sum}{जोड़}ें और घटाएँ $ u(a+h)\,v(a)$:
\begin{align*}
\frac{u(a+h)v(a+h) - u(a)v(a)}{h}
&= \frac{u(a+h)v(a+h) - u(a+h)v(a) + u(a+h)v(a) - u(a)v(a)}{h}\\
&= u(a+h)\,\frac{\Delta v}{h} + v(a)\,\frac{\Delta u}{h}.
\end{align*}
जैसा $ h \to 0$: $\frac{\Delta v}{h} \to v'(a)$, $\frac{\Delta u}{h} \to
u'(a)$, और $ u(a+h) \to u(a)$ (इसकी दर की एक सीमा है, इसलिए)
$\Delta u = h \cdot \frac{\Delta u}{h} \to 0$). इसलिए दर बढ़ती है
को $ u(a)v'(a) + v(a)u'(a)$.

भागफल नियम उसी तर्ज पर सिद्ध होता है; यह है \emph{पर भर्ती कराया गया
इस स्तर} और में साबित हुआ \cref{ch:g12:deriv}.
\end{proof}

\begin{example}\label{ex:g11:deriv:computations}
\emph{बहुपद.} $ f(x) = 3x^3 - 5x^2 + 7x - 2$:
\[
f'(x) = 3 \times 3x^2 - 5 \times 2x + 7 = 9x^2 - 10x + 7 .
\]

\emph{भागफल।} $ g(x) = \frac{2x+1}{x^2+1}$: साथ $ u = 2x + 1$, $ v = x^2
+ 1$,
\[
g'(x) = \frac{2(x^2+1) - (2x+1)\,2x}{(x^2+1)^2}
= \frac{-2x^2 - 2x + 2}{(x^2+1)^2}.
\]

\emph{उत्पाद।} $ k(x) = x\sqrt x $ के लिए $ x > 0$:
$ k'(x) = 1 \times \sqrt x + x \times \frac{1}{2\sqrt x}
= \sqrt x + \frac{\sqrt x}{2} = \frac{3\sqrt x}{2}$.
\end{example}

\section{व्युत्पन्न और विविधता का संकेत}

\begin{theorem}[व्युत्पन्न और विविधताएँ]\label{thm:g11:deriv:variations}
होने देना $ f $ होना \omterm{def:g11:deriv:derivative}{विभेदक} एक पर \omterm{def:g10:numbers:interval}{अंतराल} $ I $.
\begin{enumerate}
  \item अगर $ f' \geq 0$ पर $ I $, तब $ f $ है \omterm{def:g11:func:monotone}{की बढ़ती} पर $ I $.
  \item अगर $ f' \leq 0$ पर $ I $, तब $ f $ है \omterm{def:g10:functions:variations}{घटते} पर $ I $.
  \item अगर $ f' > 0$ पर $ I $ (संभवतः बहुत से बिंदुओं को छोड़कर जहां
        यह गायब हो जाता है), फिर $ f $ सख्ती से है \omterm{def:g11:func:monotone}{की बढ़ती} पर $ I $.
\end{enumerate}
\end{theorem}

\begin{proof}
अंतर्ज्ञान स्पष्ट है --- एक वक्र जिसका \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} सभी बिंदु ऊपर की ओर होने चाहिए
चढ़ो --- लेकिन एक कठोर प्रमाण की आवश्यकता है \omterm{def:g10:stats:mean}{अर्थ} मूल्य प्रमेय. परिणाम है
\emph{इस स्तर पर स्वीकार किया गया}; देखना \cref{thm:g12:deriv:variations}.
\end{proof}

\begin{proposition}[एक्सट्रीमा]\label{prop:g11:deriv:extrema}
अगर $ f'$ पर परिवर्तन चिह्न $ a $ से $+$ को $-$, तब $ f $ एक स्थानीय है \omterm{def:g10:functions:extrema}{अधिकतम}
पर $ a $; से $-$ को $+$, एक स्थानीय \omterm{def:g10:functions:extrema}{न्यूनतम}. ऐसे मोड़ पर, $ f'(a) = 0$ और यह \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} क्षैतिज है.
\end{proposition}

\begin{proof}
अगर $ f' \geq 0$ एक पर \omterm{def:g10:numbers:interval}{अंतराल} पर समाप्त हो रहा है $ a $ और $ f' \leq 0$ एक पर \omterm{def:g10:numbers:interval}{अंतराल}
से शुरू हो रहा है $ a $, तब $ f $ तक बढ़ जाता है $ f(a)$ और बाद में घट जाती है
(\cref{thm:g11:deriv:variations}): $ f(a)$ पास में सबसे बड़ा है. तब से $ f'$
पर परिवर्तन चिह्न $ a $ और वहां परिभाषित किया गया है, $ f'(a) = 0$.
\end{proof}

\begin{method}[किसी फ़ंक्शन का अध्ययन करना]\label{met:g11:deriv:study}
\begin{enumerate}
  \item गणना करना $ f'(x)$ और इसे सरल बनाएं --- आदर्श रूप से एक गुणनखंडित रूप में।
  \item का चिन्ह निर्धारित करें $ f'(x)$ सभी के ऊपर \omterm{def:g10:numbers:interval}{अंतराल} (एक द्विघात के लिए
        $ f'$, उपयोग \cref{thm:g11:quad:sign}).
  \item परिणामों को ए में रिकार्ड करें \emph{\omterm{def:g10:functions:table}{विविधता तालिका}}: के लिए एक पंक्ति
        का संकेत $ f'$, के लिए तीरों की एक पंक्ति $ f $, के मूल्यों के साथ $ f $ पर
        निर्णायक मोड़.
  \item निष्कर्ष: एक्स्ट्रेमा, समाधानों की संख्या $ f(x) = k $, असमानताएँ।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g11:deriv:study}
अध्ययन $ f(x) = x^3 - 3x + 1$ पर $\R $. पहला,
$ f'(x) = 3x^2 - 3 = 3(x-1)(x+1)$: एक द्विघात के साथ \omterm{def:g11:quad:discriminant}{जड़}ों $-1$ और $1$,
उनके बाहर सकारात्मक. इसलिए $ f $ है \omterm{def:g11:func:monotone}{की बढ़ती} पर $\intoc{-\infty}{-1}$,
\omterm{def:g10:functions:variations}{घटते} पर $\intcc{-1}{1}$, \omterm{def:g11:func:monotone}{की बढ़ती} पर $\intco{1}{+\infty}$, एक के साथ
स्थानीय \omterm{def:g10:functions:extrema}{अधिकतम} $ f(-1) = -1 + 3 + 1 = 3$ और एक स्थानीय \omterm{def:g10:functions:extrema}{न्यूनतम}
$ f(1) = 1 - 3 + 1 = -1$.
\end{example}

\omterm{def:g10:functions:extrema}{अधिकतम} $(-1, 3)$ और स्थानीय \omterm{def:g10:functions:extrema}{न्यूनतम} $(1, -1)$, कहाँ $ f'$ परिवर्तन
संकेत.}
\end{omfigure}

\begin{example}[अनुकूलन]\label{ex:g11:deriv:box}
एक खुला बक्सा किससे बनाया जाता है? $12 \times 12$ सेमी शीट का एक वर्ग काटकर
ओर $ x $ प्रत्येक कोने से और तह से. इसका आयतन है
$ V(x) = x(12 - 2x)^2$ के लिए $0 < x < 6$. \omterm{def:g10:algebra:expand}{विस्तार},
$ V(x) = 4x^3 - 48x^2 + 144x $, इसलिए
\[
V'(x) = 12x^2 - 96x + 144 = 12(x^2 - 8x + 12) = 12(x - 2)(x - 6).
\]
पर $\intoo{0}{6}$, कारक $ x - 6$ नकारात्मक है, इसलिए $ V'(x)$ का चिन्ह है
का $-(x-2)$: पहले सकारात्मक $2$, बाद में नकारात्मक। आयतन \omterm{def:g10:functions:extrema}{अधिकतम} है
के लिए $ x = 2$ सेमी, दे रहा हूँ $ V(2) = 2 \times 8^2 = 128$ सेमी $^3$.
\end{example}

\begin{exercise}[$\star\star $]\label{exo:g11:deriv:10}

\frac{1000}{r^2}$. उस त्रिज्या को घटाएं जो सतह को \omterm{def:g10:functions:extrema}{न्यूनतम} करती है, एक के रूप में
सटीक अभिव्यक्ति.
\end{exercise}

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:deriv:12}

\section{समस्या: स्पर्शरेखा के तीन कार्य}

\begin{problem}[{सप्ताहांत समस्या --- परवलयिक दर्पण सिद्ध हुआ, न्यूटन की जड़-खोजने वाली मशीन, और लॉग में सबसे मजबूत किरण}]
\label{pb:g11:deriv:1}
ए यौगिक खींचता \omterm{def:g11:deriv:tangent}{स्पर्शरेखा}; यह एक मामूली प्रतिभा की तरह लगता है.
यह समस्या दर्शाती है \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} एक साथ तीन नौकरियाँ धारण करना: यह
लंबित छोड़े गए हेडलाइट प्रमेय को सिद्ध करता है
\cref{pb:g11:quad:1} (प्रत्येक किरण a के समानांतर है परवलयकी धुरी
के माध्यम से प्रतिबिंबित करता है \omterm{pb:g11:quad:1}{केंद्र}), यह सबसे तेज़ शक्ति देता है
सामान्य उपयोग में समीकरण-समाधान एल्गोरिदम --- प्रत्येक के अंदर
कैलकुलेटर और तंत्रिका नेटवर्क --- और यह सबसे मजबूत पाता है
आयताकार बीम जो एक गोल लॉग से प्राप्त हो सकता है।

\medskip
\noindent\textbf{Part I --- \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} fluency.}
\begin{enumerate}
  \item अंतर $ f(x) = 3x^2 - 5x + 1$ और
        $ g(x) = x^3 - 12x $
        (\cref{thm:g11:deriv:rules}); और ठीक हो जाओ $\left(x^2
        \right)'$ पर $ x = 1$ परिभाषा से (की दर)
        फिर बदलो $ h \to 0$).
  \item खोजें \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} को $ y = x^3 - 12x $ पर $ a = 2$. क्या है?
        इसके बारे में विशेष, और वह क्या संकेत देता है
        (\cref{prop:g11:deriv:extrema})?
  \item सामान्य स्थापित करें \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} तक \omterm{def:g11:quad:parabola}{परवलय} $ y = x^2$
        बिंदु पर $(a, a^2)$:
        \[
        y = 2ax - a^2 .
        \]
  \item वह कहां है \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} पार करो $ y $-अक्ष? निष्कर्ष निकालें
        ड्राफ्ट्समैन का नुस्खा: आकर्षित करने के लिए \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} एक बिंदु पर $ P $
        ऊंचाई का $ h $ पर \omterm{def:g11:quad:parabola}{परवलय}, जोड़ना $ P $ के बिंदु तक
        गहराई पर अक्ष\,\dots\ मूल के नीचे. वर्णन करें।
  \item पूरा निर्माण करें \omterm{def:g10:functions:table}{विविधता तालिका} का
        $ g(x) = x^3 - 12x $ (हस्ताक्षर करना $ g'$, स्थानीय एक्स्ट्रेमा और
        उनके मूल्य, \cref{met:g11:deriv:study}).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The mirror theorem.}
होने देना $ P(a, a^2)$ का एक बिंदु हो \omterm{def:g11:quad:parabola}{परवलय} $ y = x^2$ (लेना
$ a \neq 0$), $ F\left(0, \frac14\right)$  \omterm{pb:g11:quad:1}{केंद्र}, और $ T $ द
बिंदु कहाँ है \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} पर $ P $ को पार कर जाता है $ y $-अक्ष (प्रश्न
4). से स्मरण करें \cref{pb:g11:quad:1} वह
$ PF = a^2 + \frac14$.
\begin{enumerate}[resume]
  \item दूरी की गणना करें $ FT $.
  \item निष्कर्ष निकालिए कि त्रिभुज $ FPT $ समद्विबाहु है $ F $.
        अतः कौन से दो कोण बराबर हैं?
  \item आने वाली किरण के माध्यम से $ P $, अक्ष के समानांतर है
        रेखा के समानांतर $(TF)$ (दोनों लंबवत)। उपयोग करना
        वैकल्पिक कोण, दिखाएँ कि \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} पर $ P $ बनाता है
        \emph{समान कोण} आने वाली ऊर्ध्वाधर किरण के साथ और
        खंड के साथ $[PF]$.
  \item भौतिकी: प्रकाश वक्र से परावर्तित होता है जैसे कि वक्र से \omterm{def:g11:deriv:tangent}{स्पर्शरेखा},
        उसी कोण पर छोड़ते हुए वह आया (का नियम)।
        प्रतिबिंब)। निष्कर्ष निकालें \emph{दर्पण प्रमेय}: हर
        अक्ष के समांतर किरण परावर्तित होती है \omterm{pb:g11:quad:1}{केंद्र} ---
        और, समय को उलटते हुए, एक बल्ब \omterm{pb:g11:quad:1}{केंद्र} बाहर किरणें
        समानांतर प्रकाश. की थाली और हेडलाइट
        \cref{pb:g11:quad:1} अब प्रमेय हैं.
  \item समद्विबाहु त्रिभुज को संख्यात्मक रूप से जाँचें $ a = 1$:
        गणना करें $ T $, $ FP $ और $ FT $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Newton's machine.}
समाधान करना $ f(x) = 0$, न्यूटन ने प्रस्तावित किया: एक अनुमान से शुरुआत करें
$ x_0$, का पीछा करो \omterm{def:g11:deriv:tangent}{स्पर्शरेखा} पर $ x_0$ अक्ष के नीचे, और चलो
इसका $ x $-अगला अनुमान अवरोधन होगा।
\begin{enumerate}[resume]
  \item विधि का सूत्र व्युत्पन्न करें:
        \[
        x_{n + 1} = x_n - \frac{f(x_n)}{f'(x_n)} .
        \]
  \item इसे चालू करें $ f(x) = x^2 - 2$ से $ x_0 = 1$: गणना करें
        $ x_1$, $ x_2$, $ x_3$ सटीक भिन्नों के रूप में. जो
        दो हजार साल पुराना क्रम --- और कौन सी मशीन
        से \cref{pb:g10:functions:1} --- क्या आपके पास बस है
        पुनर्निर्माण? संयोग के पीछे की पहचान सिद्ध करें:
        \[
        x - \frac{x^2 - 2}{2x} = \frac12\left(x +
        \frac2x\right).
        \]
  \item हल करना $ x^3 = 100$ (की बचत क्रॉसओवर
        \cref{pb:g10:reffunc:1}): दो न्यूटन चरण लागू करें
        $ f(x) = x^3 - 100$ से $ x_0 = 5$ (चार दशमलव), और
        कैलकुलेटर से तुलना करें $\sqrt[3]{100}$.
  \item तोड़फोड़: विधि को चालू करने का प्रयास करें
        $ f(x) = x^3 - 12x $ पर $ x_0 = 2$. क्या गलत होता है, और
        क्यों (प्रश्न 2)? व्यावहारिक चेतावनी बताएं.
  \item प्रश्न 12 में सही दशमलव मोटे तौर पर गए
        $1 \to 3 \to 6$. द्विभाजन (आधा करने) से तुलना करें
        \omterm{def:g10:numbers:interval}{अंतराल} प्रत्येक चरण) और एक वाक्य में बताएं क्यों
        कैलकुलेटर, जीपीएस रिसीवर और तंत्रिका-नेटवर्क प्रशिक्षण
        सभी न्यूटन के विचार पर चलते हैं।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The strongest beam.}
चौड़ाई का एक आयताकार बीम $ w $ और गहराई $ d $ ए से काटा जाता है
व्यास का गोल लॉग $30$ सेमी, तो $ w^2 + d^2 = 900$. एक किरण का
कठोरता आनुपातिक है $ w d^2$ (गहराई दो बार गिना जाता है:
जॉयिस्ट किनारे पर खड़े हैं!)
\begin{enumerate}[resume]
  \item कठोरता को इस प्रकार व्यक्त करें \omterm{def:g11:func:function}{समारोह} का $ w $ अकेला:
        $ S(w) = 900w - w^3$, जिस पर \omterm{def:g10:numbers:interval}{अंतराल}?
  \item अंतर करें, निर्माण करें \omterm{def:g10:functions:table}{विविधता तालिका}, और खोजें
        इष्टतम $ w $ और $ d $ (सटीक मान, फिर से
        मिलीमीटर).
  \item दिखाएँ कि इष्टतम अनुपात संतुष्ट करते हैं
        $ d = w\sqrt2$ --- पुराने बढ़ई का अनुपात. (बोनस
        विद्या: लॉग के व्यास को तीन बराबर भागों में विभाजित करना
        और वृत्त निर्माणों पर लंबों को ऊपर उठाना
        बिल्कुल यही आयत।)
  \item भोली भाली पसंद है \emph{वर्ग} खुशी से उछलना
        ($ w = d$). इसकी कठोरता और इष्टतम की गणना करें
        बीम का, और बढ़ई का लाभ प्रतिशत में दें।
  \item समापन--- द स्पर्शरेखाके तीन कार्य, प्रत्येक एक वाक्य:
        ज्यामिति (दर्पण प्रमेय), अंकगणित (न्यूटन का)।
        मशीन), अनुकूलन (बीम), सभी द्वारा संचालित
        एक ही वस्तु. और आगे का सूचक: ग्रेड 12 जोड़ता है
        उत्तलता --- स्पर्शरेखावक्र के किनारे --- को
        तीनों को तेज करो.
\end{enumerate}
\end{problem}
