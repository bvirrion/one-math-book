\chapter{The Binomial Distribution}\label{ch:g11:binom}

Repeat the same yes/no experiment several times, independently, and count
the successes: the resulting \omterm{def:g11:prob:rv}{distribution} --- the \emph{binomial} --- is
the most important discrete \omterm{def:g11:prob:rv}{distribution} of all. This chapter builds it
with \omterm{met:g10:proba:tree}{trees} and path counting; the closed formula for the path counts (with
factorials) comes with the counting tools of \cref{ch:g12:comb}, and the
\omterm{def:g11:prob:rv}{distribution} is revisited in \cref{ch:g12:randvar}.

\section{Bernoulli trials}

\begin{definition}[Bernoulli trial]\label{def:g11:binom:bernoulli}
A \emph{Bernoulli trial}\index{Bernoulli trial} is an experiment with
exactly two outcomes: \emph{success}, with \omterm{def:g10:proba:distribution}{probability} $p$, and
\emph{failure}, with \omterm{def:g10:proba:distribution}{probability} $1 - p$. The \omterm{def:g11:prob:rv}{random variable} $X$ equal
to $1$ on success and $0$ on failure is said to follow the
\emph{Bernoulli \omterm{def:g11:prob:rv}{distribution}} $\mathcal B(p)$; then
\[
\E(X) = p,
\qquad
\V(X) = p(1 - p).
\]
\end{definition}

\begin{proof}[Proof of the two formulas]
$\E(X) = p \times 1 + (1-p) \times 0 = p$; and since $X^2 = X$ (both $0$
and $1$ are their own squares), $\E(X^2) = p$, so by
\cref{prop:g11:prob:konig}, $\V(X) = p - p^2 = p(1-p)$.
\end{proof}

\begin{definition}[Repeated independent trials]\label{def:g11:binom:repeated}
Repeating a \omterm{def:g11:binom:bernoulli}{Bernoulli trial} $n$ times \emph{independently} means: the
outcome of each trial has no influence on the others, and the \omterm{def:g10:proba:distribution}{probability}
of any complete \omterm{def:g11:seq:sequence}{sequence} of outcomes is the \emph{product} of the
probabilities along the corresponding path of the \omterm{met:g10:proba:tree}{tree} --- $p$ for each
success, $1 - p$ for each failure.
\end{definition}

\begin{example}\label{ex:g11:binom:threetrials}
Three independent trials with success \omterm{def:g10:proba:distribution}{probability} $p$. The \omterm{def:g11:seq:sequence}{sequence} SFS
(success, failure, success) has \omterm{def:g10:proba:distribution}{probability} $p(1-p)p = p^2(1-p)$ --- and
so does \emph{every} \omterm{def:g11:seq:sequence}{sequence} with exactly two successes, regardless of
the positions: only the \emph{number} of S's and F's matters.
\end{example}

\section{Path counts and binomial coefficients}

\begin{definition}[Binomial coefficient]\label{def:g11:binom:coefficient}
In the \omterm{met:g10:proba:tree}{tree} of $n$ independent trials, the \emph{binomial
coefficient}\index{binomial coefficient} $\binom{n}{k}$ (read ``$n$
choose $k$'') is the number of paths containing exactly $k$ successes.
\end{definition}

\begin{example}
$\binom{3}{2} = 3$: the paths SSF, SFS, FSS. Likewise $\binom{3}{0} = 1$
(the path FFF), $\binom{3}{1} = 3$ and $\binom{3}{3} = 1$. By
convention and by the \omterm{met:g10:proba:tree}{tree}, $\binom n0 = \binom nn = 1$ for every $n$.
\end{example}

\begin{proposition}[Pascal's rule]\label{prop:g11:binom:pascal}
\index{Pascal's triangle}
For $1 \leq k \leq n - 1$:
\[
\binom{n}{k} = \binom{n-1}{k-1} + \binom{n-1}{k}.
\]
\end{proposition}

\begin{proof}
Sort the paths of the $n$-trial \omterm{met:g10:proba:tree}{tree} with $k$ successes according to
their \emph{last} trial. Those ending in a success are obtained from a
path of the first $n-1$ trials with $k - 1$ successes: there are
$\binom{n-1}{k-1}$ of them. Those ending in a failure extend a path with
$k$ successes among the first $n-1$ trials: $\binom{n-1}{k}$ of them.
Every path is of exactly one of the two kinds.
\end{proof}

Pascal's rule generates the coefficients row by row --- each entry is the
sum of the two above it:
\[
\begin{array}{ccccccccccc}
&&&&&1&&&&&\\
&&&&1&&1&&&&\\
&&&1&&2&&1&&&\\
&&1&&3&&3&&1&&\\
&1&&4&&6&&4&&1&\\
1&&5&&10&&10&&5&&1
\end{array}
\]

\begin{remark}
A closed formula, $\binom nk = \frac{n!}{k!(n-k)!}$, together with a
systematic theory of counting, is established in \cref{ch:g12:comb}. At
this level, \omterm{prop:g11:binom:pascal}{Pascal's triangle} computes every coefficient we need.
\end{remark}

\section{The binomial distribution}

\begin{theorem}[Binomial distribution]\label{thm:g11:binom:binomial}
Let $X$ count the successes in $n$ independent \omterm{def:g11:binom:bernoulli}{Bernoulli trials} of
parameter $p$. Then $X$ follows the \emph{binomial
distribution}\index{binomial distribution} $\mathcal B(n, p)$:
\[
\P(X = k) = \binom{n}{k}\, p^k (1-p)^{n-k},
\qquad k = 0, 1, \dots, n .
\]
\end{theorem}

\begin{proof}
The \omterm{def:g11:prob:model}{event} $X = k$ is the collection of all paths with exactly $k$
successes. Each such path has \omterm{def:g10:proba:distribution}{probability} $p^k(1-p)^{n-k}$: the product
along the path contains $k$ factors $p$ and $n - k$ factors $1-p$, in
some order (\cref{def:g11:binom:repeated}). There are $\binom nk$ such
paths (\cref{def:g11:binom:coefficient}), and their probabilities add up.
\end{proof}

\begin{example}\label{ex:g11:binom:quiz}
A quiz has $5$ independent questions, each with $4$ choices; a student
answers at random, so each question is a success with $p = \frac14$. The
number $X$ of correct answers follows $\mathcal B\left(5, \frac14\right)$,
and, using row $5$ of \omterm{prop:g11:binom:pascal}{Pascal's triangle}:
\[
\P(X = 2) = \binom52 \left(\frac14\right)^2\left(\frac34\right)^3
= 10 \times \frac{1}{16} \times \frac{27}{64}
= \frac{270}{1024} \approx 0.26 .
\]
The \omterm{def:g10:proba:distribution}{probability} of at least one correct answer uses the \omterm{def:g10:proba:operations}{complement}:
$\P(X \geq 1) = 1 - \P(X = 0) = 1 - \left(\frac34\right)^5 \approx
0.76$.
\end{example}

\begin{proposition}[Expectation and variance]\label{prop:g11:binom:expectation}
If $X \sim \mathcal B(n, p)$:
\[
\E(X) = np,
\qquad
\V(X) = np(1-p).
\]
\end{proposition}

\begin{proof}[Justification]
Write $X = X_1 + X_2 + \dots + X_n$, where $X_i$ equals $1$ if the
$i$-th trial succeeds: each $X_i$ is a Bernoulli variable of \omterm{def:g11:prob:expectation}{expectation}
$p$ (\cref{def:g11:binom:bernoulli}). Averages add --- summing the $n$
contributions gives $\E(X) = np$. That \emph{\omterm{def:g11:prob:variance}{variances}} also add for
independent variables is true but more delicate: the \omterm{def:g11:prob:variance}{variance} formula is
\emph{admitted at this level} and proved in \cref{ch:g12:sums}.
\end{proof}

\begin{method}[Recognizing a binomial situation]\label{met:g11:binom:recognize}
Before writing $X \sim \mathcal B(n, p)$, check three ingredients: a
\emph{fixed number} $n$ of trials, decided in advance; each trial has
\emph{two outcomes} with the \emph{same} success \omterm{def:g10:proba:distribution}{probability} $p$; the
trials are \emph{independent} (with replacement, or separate devices).
Drawing without replacement from a small population is \emph{not}
binomial --- the \omterm{def:g10:proba:distribution}{probability} changes at each draw
(\cref{exo:g11:prob:6}).
\end{method}

\section{Sampling: is the observation surprising?}

The binomial \omterm{def:g11:prob:rv}{distribution} answers a very practical question: \emph{if the
success \omterm{def:g10:proba:distribution}{probability} really is $p$, what counts of successes are
plausible?}

\begin{example}\label{ex:g11:binom:decision}
A machine is supposed to produce $10\%$ defective items at most. In a
batch of $10$ items, $4$ are defective. Bad luck or broken machine? If
the machine is fine, the number of defectives follows
$\mathcal B(10, 0.1)$, and
\[
\P(X \geq 4) = 1 - \P(X \leq 3) \approx 1 - 0.987 = 0.013 :
\]
about one chance in $80$. Observing an \omterm{def:g11:prob:model}{event} this unlikely is a strong
signal --- one \emph{rejects} the hypothesis that the machine still works
at $10\%$, while keeping in mind that the decision could be wrong with
\omterm{def:g10:proba:distribution}{probability} about $0.013$.
\end{example}

\begin{method}[Decision rule from a binomial model]\label{met:g11:binom:decision}
To judge an observed number $k$ of successes against a hypothesis
$X \sim \mathcal B(n, p)$: compute the \omterm{def:g10:proba:distribution}{probability}, under the hypothesis,
of a result \emph{at least as extreme} as $k$. If that \omterm{def:g10:proba:distribution}{probability} is
very small (a common convention: below $5\%$), reject the hypothesis;
otherwise the observation is compatible with it. The threshold is a
choice, not a theorem --- statistics quantifies the risk, and the user
accepts it.
\end{method}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g11:binom:1}
Extend \omterm{prop:g11:binom:pascal}{Pascal's triangle} to row $7$, and give the values of
$\binom62$, $\binom{7}{3}$ and $\binom74$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:binom:2}
A fair die is rolled $4$ times; $X$ counts the sixes. Justify that
$X \sim \mathcal B\left(4, \frac16\right)$ and compute $\P(X = 0)$,
$\P(X = 1)$ and $\P(X \geq 2)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:binom:3}
Which of the following is binomial? Justify.
\begin{enumerate}
  \item Number of heads in $20$ tosses of a fair coin.
  \item Number of aces in $5$ cards dealt from one deck.
  \item Number of rainy days next week, if each day is rainy with
        \omterm{def:g10:proba:distribution}{probability} $0.3$ independently.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:binom:4}
$X \sim \mathcal B(50, 0.2)$. Give $\E(X)$, $\V(X)$ and $\sigma(X)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:binom:5}
An archer hits the target with \omterm{def:g10:proba:distribution}{probability} $0.7$ at each shot,
independently. In $6$ shots, compute the \omterm{def:g10:proba:distribution}{probability} of exactly $4$ hits,
and of at least $5$ hits.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:binom:6}
A true/false test has $8$ questions; a student guesses each answer.
What is the \omterm{def:g10:proba:distribution}{probability} of passing (at least $6$ correct answers)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:binom:7}
Each purchased cereal box contains figurine A or figurine B, with
\omterm{def:g10:proba:distribution}{probability} $\frac12$ each, independently. A collector buys $5$ boxes.
Compute the \omterm{def:g10:proba:distribution}{probability} that the collector gets at least one figurine of
each kind. (\omterm{def:g10:proba:operations}{Complement}: all A or all B.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:binom:8}
A basketball player scores free throws with \omterm{def:g10:proba:distribution}{probability} $p$,
independently. Let $X \sim \mathcal B(3, p)$ be the number of scores in
three throws. Express $\P(X = 3)$ and $\P(X \geq 1)$ as \omterm{def:g11:func:function}{functions} of $p$,
and find for which $p$ the \omterm{def:g10:proba:distribution}{probability} of scoring all three equals
$\frac{27}{64}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:binom:9}
How many times must a fair coin be tossed for the \omterm{def:g10:proba:distribution}{probability} of getting
at least one head to exceed $0.99$? (\omterm{def:g10:proba:operations}{Complement}, then try successive
values of $n$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:binom:10}
Using Pascal's rule (\cref{prop:g11:binom:pascal}) and
$\binom n0 = \binom nn = 1$, prove that the entries of each row of
\omterm{prop:g11:binom:pascal}{Pascal's triangle} sum to $2^n$: interpret both sides as counting all the
paths of the \omterm{met:g10:proba:tree}{tree}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:binom:11}
A politician claims $60\%$ approval. In a random \omterm{def:g10:proba:sample}{sample} of $10$ people,
only $3$ approve.
\begin{enumerate}
  \item Under the claim, what \omterm{def:g11:prob:rv}{distribution} does the number $X$ of
        approvals in the \omterm{def:g10:proba:sample}{sample} follow? Compute $\P(X \leq 3)$.
  \item Using the decision rule of \cref{met:g11:binom:decision} with a
        $5\%$ threshold, is the observation compatible with the claim?
\end{enumerate}
\end{exercise}

\section{Problem: The Galton board}

\begin{problem}[{Weekend problem --- balls, pegs and Pascal's
triangle: how the bell shape is born, why playoff series favor
the stronger team, and when to cry foul}]
\label{pb:g11:binom:1}
Drop a thousand balls through a lattice of pegs, each bounce a
fair left-or-right coin flip, and the slots below fill into a
smooth, symmetric bell --- every time. The machine is called a
Galton board, and its mathematics is exactly this chapter's
binomial \omterm{def:g11:prob:rv}{distribution} (\cref{thm:g11:binom:binomial}). This
problem builds the triangle, runs the board, referees a
best-of-seven series, and ends where the binomial earns its
salary: deciding when an observation should make us doubt a
claim.

\medskip
\noindent\textbf{Part I --- The triangle.}
\begin{enumerate}
  \item Build \omterm{prop:g11:binom:pascal}{Pascal's triangle} down to row $6$
        (\cref{prop:g11:binom:pascal}). State and explain the
        symmetry $\binom nk = \binom{n}{n-k}$ in one sentence
        (choosing $k$ objects is the same as\,\dots).
  \item Verify on rows $4$ and $5$ that each row sums to $2^n$,
        and prove it: what do all the $\binom nk$ together
        count?
  \item Re-derive Pascal's rule
        $\binom{n+1}{k} = \binom nk + \binom{n}{k-1}$ by the
        committee argument: fix one distinguished person and
        split the committees according to that person's fate.
  \item Compute $\binom73$ twice: from the triangle, and from
        the factorial formula.
  \item Check the staircase identity
        $\binom22 + \binom32 + \binom42 + \binom52 =
        \binom63$, and explain it by cascading Pascal's rule
        from $\binom63$ downward.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The board.}
A ball falls through $n$ rows of pegs; at each peg it bounces
left or right with \omterm{def:g10:proba:distribution}{probability} $\frac12$, independently. Number
the slots $0$ to $n$ by the count of right bounces.
\begin{enumerate}[resume]
  \item Explain, with the checklist of
        \cref{met:g11:binom:recognize}, why the slot number
        follows the binomial \omterm{def:g11:prob:rv}{distribution}
        $\mathcal B\!\left(n, \frac12\right)$.
  \item For a small board ($n = 4$): give the five slot
        probabilities. Which slot is the most crowded?
  \item Now $n = 10$ and $1\,024$ balls: what are the
        \emph{expected} ball counts in the middle slot, in slot
        $7$, and in each edge slot? Describe the pile's shape.
  \item For $X \sim \mathcal B\!\left(10, \frac12\right)$:
        compute $\E(X)$, $V(X)$ and $\sigma$
        (\cref{prop:g11:binom:expectation}); then compute the
        proportion of balls expected within $2\sigma$ of the
        center (slots $2$ to $8$) and compare with Chebyshev's
        guarantee from \cref{pb:g11:stat:1}.
  \item A tilted board bounces right with \omterm{def:g10:proba:distribution}{probability} $0.6$:
        give $\E$, $V$ and $\sigma$ for $n = 10$, and describe
        what happens to the pile.
  \item In one or two sentences: what, in the board's design,
        manufactures the bell shape --- and why do so many
        real-world quantities (heights, measurement errors)
        pile up the same way? (The deep theorem behind both is
        the central limit theorem, the summit of the university
        volumes' \omterm{def:g10:proba:distribution}{probability} course.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Best of seven.}
Two teams play a series: first to $4$ wins takes the title;
games are independent.
\begin{enumerate}[resume]
  \item Equal teams ($p = \frac12$): compute the \omterm{def:g10:proba:distribution}{probability}
        that the series ends in a sweep (exactly $4$ games).
  \item Compute the \omterm{def:g10:proba:distribution}{probability} that the series goes the full
        $7$ games (what must the score be after $6$?).
  \item Complete the \omterm{def:g11:prob:rv}{distribution} of the series' length
        ($4$, $5$, $6$ or $7$ games) for equal teams, and
        compute the expected length. Which lengths are most
        likely?
  \item Now one team wins each game with $p = 0.6$. Compute its
        \omterm{def:g10:proba:distribution}{probability} of taking the series (win in $4$, $5$, $6$
        or $7$: in each case the team wins the last game and
        $3$ of the previous ones). What did the series do to
        the per-game edge?
  \item Compare with a single final ($60\,\%$) and a best-of-3
        (compute it). State the general effect of series length
        on skill against luck --- and why leagues prefer long
        finals.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- When to cry foul.}
\begin{enumerate}[resume]
  \item A coin is flipped $100$ times and shows $62$ heads. For
        a fair coin, give $\E$, $\sigma$, and the \omterm{pb:g11:stat:1}{z-score}
        (\cref{pb:g11:stat:1}) of the observation. Verdict
        under the $2\sigma$ convention?
  \item A supplier claims $2\,\%$ defective parts. In a batch
        of $50$ you find $3$ defective. Compute
        $\P(X \geq 3)$ under the claim
        ($X \sim \mathcal B(50, 0.02)$; go through
        $\P(X = 0), \P(X = 1), \P(X = 2)$). Alarming at the
        $5\,\%$ threshold (\cref{met:g11:binom:decision},
        \cref{exo:g11:binom:11})?
  \item Lottery persistence: each ticket wins (something) with
        \omterm{def:g10:proba:distribution}{probability} $\frac{1}{1000}$. Compute the \omterm{def:g10:proba:distribution}{probability}
        of at least one win with $1\,000$ tickets. The answer
        ($\approx 63\,\%$, not $100\,\%$!) hides a famous
        constant: compute $0.999^{1000}$ and keep the number
        $0.368$ in mind for grade 12.
  \item Finale --- the binomial's portrait: the recognition
        checklist (fixed $n$, independence, constant $p$);
        \omterm{prop:g11:binom:pascal}{Pascal's triangle} as its table; the bell as its shape;
        $np$ and $np(1-p)$ as its compass; and its two heirs
        awaiting in grade 12 --- the smooth bell curve and the
        law of large numbers. One sentence each.
\end{enumerate}
\end{problem}
