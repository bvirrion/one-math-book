\chapter{Quadratic Functions and Equations}\label{ch:g11:quad}

\omterm{def:g11:quad:quadratic}{Quadratic functions} are the simplest \omterm{def:g10:functions:function}{functions} after the linear ones, and
the first whose \omterm{def:g10:functions:graph}{graphs} are genuinely curved. This chapter develops the
complete toolbox for them: the \omterm{thm:g11:quad:canonical}{canonical form}, the \omterm{def:g11:quad:discriminant}{discriminant}, the sign
of a quadratic expression, and the geometry of the \omterm{def:g11:quad:parabola}{parabola}.

\section{The canonical form}

\begin{definition}[Quadratic function]\label{def:g11:quad:quadratic}
A \emph{quadratic function}\index{quadratic function} is a \omterm{def:g10:functions:function}{function} of the
form
\[
f(x) = ax^2 + bx + c, \qquad a \neq 0,
\]
where $a$, $b$, $c$ are \omterm{def:g10:numbers:sets}{real numbers}. The expression $ax^2 + bx + c$ is
also called a \emph{quadratic trinomial}.
\end{definition}

\begin{theorem}[Canonical form]\label{thm:g11:quad:canonical}
Every \omterm{def:g11:quad:quadratic}{quadratic function} can be written in exactly one way as
\[
f(x) = a(x - \alpha)^2 + \beta,
\qquad\text{where } \alpha = -\frac{b}{2a} \text{ and } \beta = f(\alpha).
\]
This is the \emph{canonical form}\index{canonical form} of $f$.
\end{theorem}

\begin{proof}
We \emph{complete the square}. Since $a \neq 0$, factor it out of the first
two terms:
\[
ax^2 + bx + c = a\left(x^2 + \frac{b}{a}x\right) + c .
\]
Now $x^2 + \frac{b}{a}x$ is the beginning of the square
$\left(x + \frac{b}{2a}\right)^2 = x^2 + \frac ba x + \frac{b^2}{4a^2}$,
so $x^2 + \frac{b}{a}x = \left(x + \frac{b}{2a}\right)^2 -
\frac{b^2}{4a^2}$. Substituting:
\[
f(x) = a\left(x + \frac{b}{2a}\right)^2 - \frac{b^2}{4a} + c
= a\bigl(x - \alpha\bigr)^2 + \beta,
\]
with $\alpha = -\frac{b}{2a}$ and $\beta = c - \frac{b^2}{4a}$. Evaluating
at $x = \alpha$ kills the squared term, so $\beta = f(\alpha)$. Uniqueness:
\omterm{def:g10:algebra:expand}{expanding} $a(x-\alpha)^2 + \beta$ and identifying the coefficients of $x$
forces $\alpha = -\frac{b}{2a}$, and then $\beta$ is determined.
\end{proof}

\begin{example}\label{ex:g11:quad:canonical}
Let $f(x) = 2x^2 - 8x + 3$. Then $\alpha = -\frac{-8}{2 \times 2} = 2$ and
$\beta = f(2) = 8 - 16 + 3 = -5$, so
\[
f(x) = 2(x - 2)^2 - 5 .
\]
Check by \omterm{def:g10:algebra:expand}{expanding}: $2(x^2 - 4x + 4) - 5 = 2x^2 - 8x + 8 - 5 =
2x^2 - 8x + 3$.
\end{example}

\begin{proposition}[Extremum and variations]\label{prop:g11:quad:variations}
Let $f(x) = a(x-\alpha)^2 + \beta$.
\begin{enumerate}
  \item If $a > 0$, then $f$ is \omterm{def:g10:functions:variations}{decreasing} on $\intoc{-\infty}{\alpha}$ and
        \omterm{def:g10:functions:variations}{increasing} on $\intco{\alpha}{+\infty}$: $f$ has a \emph{\omterm{def:g10:functions:extrema}{minimum}}
        $\beta$, attained at $x = \alpha$.
  \item If $a < 0$, the variations are reversed: $f$ has a \emph{\omterm{def:g10:functions:extrema}{maximum}}
        $\beta$ at $x = \alpha$.
\end{enumerate}
\end{proposition}

\begin{proof}
Suppose $a > 0$ and let $\alpha \leq u < v$. Then
\[
f(v) - f(u) = a\bigl((v-\alpha)^2 - (u-\alpha)^2\bigr)
= a\,(v - u)\,\bigl((v-\alpha) + (u-\alpha)\bigr).
\]
Each factor is positive: $a > 0$, $v - u > 0$, and
$(v-\alpha) + (u-\alpha) > 0$ since $v > \alpha$ and $u \geq \alpha$. So
$f(v) > f(u)$: $f$ is \omterm{def:g10:functions:variations}{increasing} on $\intco{\alpha}{+\infty}$. For
$u < v \leq \alpha$ the last factor is negative, so $f(v) < f(u)$: $f$ is
\omterm{def:g10:functions:variations}{decreasing} there. Finally $f(x) - f(\alpha) = a(x-\alpha)^2 \geq 0$ for all
$x$, so $\beta = f(\alpha)$ is the \omterm{def:g10:functions:extrema}{minimum}. The case $a < 0$ follows by
applying the above to $-f$.
\end{proof}

\begin{example}
A farmer has $100$ meters of fence to enclose a rectangular field against a
straight wall (no fence needed along the wall). If $x$ is the width, the
enclosed area is $A(x) = x(100 - 2x) = -2x^2 + 100x$. Here $\alpha =
-\frac{100}{2\times(-2)} = 25$ and $A(25) = 25 \times 50 = 1250$: the area
is largest for a width of $25$ m, giving $1250$ m$^2$.
\end{example}

\section{Roots and factorization}

\begin{definition}[Root, discriminant]\label{def:g11:quad:discriminant}
A \emph{root}\index{root} of the trinomial $ax^2+bx+c$ is a solution of the
\omterm{def:g10:algebra:equation}{equation} $ax^2 + bx + c = 0$. The \emph{discriminant}\index{discriminant}
of the trinomial is the number
\[
\Delta = b^2 - 4ac .
\]
\end{definition}

\begin{theorem}[Solving $ax^2+bx+c=0$]\label{thm:g11:quad:roots}
\begin{enumerate}
  \item If $\Delta > 0$, the \omterm{def:g10:algebra:equation}{equation} has two distinct \omterm{def:g11:quad:discriminant}{roots}
        \[
        x_1 = \frac{-b - \sqrt{\Delta}}{2a},
        \qquad
        x_2 = \frac{-b + \sqrt{\Delta}}{2a},
        \]
        and $ax^2+bx+c = a(x - x_1)(x - x_2)$.
  \item If $\Delta = 0$, the \omterm{def:g10:algebra:equation}{equation} has the single \omterm{def:g11:quad:discriminant}{root}
        $x_0 = -\frac{b}{2a}$, and $ax^2+bx+c = a(x - x_0)^2$.
  \item If $\Delta < 0$, the \omterm{def:g10:algebra:equation}{equation} has no real solution, and the
        trinomial cannot be factored with real coefficients.
\end{enumerate}
\end{theorem}

\begin{proof}
Start from the \omterm{thm:g11:quad:canonical}{canonical form} computed in \cref{thm:g11:quad:canonical}:
\[
ax^2+bx+c
= a\left(x + \frac{b}{2a}\right)^2 + c - \frac{b^2}{4a}
= a\left[\left(x + \frac{b}{2a}\right)^2 - \frac{\Delta}{4a^2}\right],
\]
since $c - \frac{b^2}{4a} = \frac{4ac - b^2}{4a} = -\frac{\Delta}{4a} =
a \times \left(-\frac{\Delta}{4a^2}\right)$.

\emph{1.} If $\Delta > 0$, then $\frac{\Delta}{4a^2} =
\left(\frac{\sqrt\Delta}{2a}\right)^2$ and the bracket is a difference of
two squares:
\[
ax^2+bx+c = a\left(x + \frac{b}{2a} - \frac{\sqrt\Delta}{2a}\right)
\left(x + \frac{b}{2a} + \frac{\sqrt\Delta}{2a}\right)
= a(x - x_2)(x - x_1).
\]
A product is zero exactly when one of its factors is zero, giving the two
\omterm{def:g11:quad:discriminant}{roots}, distinct because $\sqrt\Delta \neq 0$.

\emph{2.} If $\Delta = 0$, the bracket is
$\left(x + \frac{b}{2a}\right)^2$, which vanishes only at
$x_0 = -\frac{b}{2a}$.

\emph{3.} If $\Delta < 0$, then $-\frac{\Delta}{4a^2} > 0$, so the bracket
is a square plus a positive number: it never vanishes, and the \omterm{def:g10:algebra:equation}{equation} has
no real solution. A factorization $a(x-x_1)(x-x_2)$ would produce \omterm{def:g11:quad:discriminant}{roots}, so
none exists.
\end{proof}

\begin{example}\label{ex:g11:quad:solving}
Solve $2x^2 - 8x + 3 = 0$. Here $\Delta = (-8)^2 - 4 \times 2 \times 3 =
64 - 24 = 40 > 0$, so there are two \omterm{def:g11:quad:discriminant}{roots}:
\[
x_{1,2} = \frac{8 \pm \sqrt{40}}{4} = \frac{8 \pm 2\sqrt{10}}{4}
= 2 \pm \frac{\sqrt{10}}{2}.
\]
For $x^2 + x + 1 = 0$: $\Delta = 1 - 4 = -3 < 0$, no real solution.
\end{example}

\begin{proposition}[Sum and product of the roots]\label{prop:g11:quad:vieta}
If $\Delta \geq 0$ and $x_1, x_2$ are the \omterm{def:g11:quad:discriminant}{roots} (equal when $\Delta = 0$),
then
\[
x_1 + x_2 = -\frac{b}{a},
\qquad
x_1 x_2 = \frac{c}{a} .
\]
Conversely, two numbers with sum $s$ and product $p$ are the solutions of
$x^2 - sx + p = 0$.
\end{proposition}

\begin{proof}
Expand the factorization of \cref{thm:g11:quad:roots}:
\[
a(x - x_1)(x - x_2) = a x^2 - a(x_1 + x_2)x + a\,x_1 x_2 .
\]
Identifying the coefficients with $ax^2 + bx + c$ gives
$-a(x_1+x_2) = b$ and $a\,x_1x_2 = c$. For the converse, if $u + v = s$ and
$uv = p$, then $(x-u)(x-v) = x^2 - sx + p$, so $u$ and $v$ are its \omterm{def:g11:quad:discriminant}{roots}.
\end{proof}

\begin{example}
Find two numbers with sum $10$ and product $21$. They solve
$x^2 - 10x + 21 = 0$; $\Delta = 100 - 84 = 16$, \omterm{def:g11:quad:discriminant}{roots}
$\frac{10 \pm 4}{2} = 7$ and $3$. \emph{Quick \omterm{def:g11:quad:discriminant}{root} spotting:} since the
coefficients of $x^2 - 5x + 4$ satisfy $1 - 5 + 4 = 0$, the number $1$ is a
\omterm{def:g11:quad:discriminant}{root}, and the other one is $\frac{c}{a} = 4$ (their product must be
$\frac ca$).
\end{example}

\section{Sign of a quadratic}

\begin{theorem}[Sign of a trinomial]\label{thm:g11:quad:sign}
Let $f(x) = ax^2+bx+c$ with $a \neq 0$.
\begin{enumerate}
  \item If $\Delta > 0$, with \omterm{def:g11:quad:discriminant}{roots} $x_1 < x_2$: $f(x)$ has the sign of $a$
        outside $\intcc{x_1}{x_2}$, and the sign of $-a$ on
        $\intoo{x_1}{x_2}$.
  \item If $\Delta = 0$: $f(x)$ has the sign of $a$ for all $x \neq x_0$,
        and vanishes at $x_0$.
  \item If $\Delta < 0$: $f(x)$ has the sign of $a$ for all $x$.
\end{enumerate}
In short: \emph{a trinomial has the sign of $a$, except between its \omterm{def:g11:quad:discriminant}{roots}}.
\end{theorem}

\begin{proof}
\emph{1.} Write $f(x) = a(x-x_1)(x-x_2)$ and follow the signs of the
factors. If $x < x_1$: both $x - x_1$ and $x - x_2$ are negative, their
product is positive, so $f(x)$ has the sign of $a$. If $x_1 < x < x_2$: the
factors have opposite signs, the product is negative, and $f(x)$ has the
sign of $-a$. If $x > x_2$: both factors are positive and $f(x)$ has the
sign of $a$ again.

\emph{2.} $f(x) = a(x - x_0)^2$ and $(x-x_0)^2 > 0$ for $x \neq x_0$.

\emph{3.} From the proof of \cref{thm:g11:quad:roots},
$f(x) = a\left[\left(x + \frac{b}{2a}\right)^2 +
\left(-\frac{\Delta}{4a^2}\right)\right]$, and the bracket is always
positive.
\end{proof}

\begin{method}[Solving quadratic inequalities]\label{met:g11:quad:inequalities}
To solve $ax^2 + bx + c \leq 0$ (or $\geq$, $<$, $>$):
\begin{enumerate}
  \item compute $\Delta$ and the \omterm{def:g11:quad:discriminant}{roots}, if any;
  \item draw the \omterm{def:g10:algebra:signtable}{sign table} using \cref{thm:g11:quad:sign} (sign of $a$
        outside the \omterm{def:g11:quad:discriminant}{roots});
  \item read off the solution \omterm{def:g10:numbers:interval}{interval}(s), paying attention to whether the
        \omterm{def:g11:quad:discriminant}{roots} themselves are included (wide inequality) or not (strict).
\end{enumerate}
\end{method}

\begin{example}\label{ex:g11:quad:inequality}
Solve $-x^2 + 3x + 4 \geq 0$. Here $\Delta = 9 + 16 = 25$, and the \omterm{def:g11:quad:discriminant}{roots}
are $\frac{-3 \pm 5}{-2}$, that is $x_1 = -1$ and $x_2 = 4$. Since
$a = -1 < 0$, the trinomial is negative outside the \omterm{def:g11:quad:discriminant}{roots} and positive
between them. With the wide inequality, the solution set is
$\intcc{-1}{4}$.
\end{example}

\section{The parabola}

\begin{definition}[Parabola]\label{def:g11:quad:parabola}
The \omterm{def:g10:functions:graph}{graph} of a \omterm{def:g11:quad:quadratic}{quadratic function} $f(x) = a(x-\alpha)^2+\beta$ is a
\emph{parabola}\index{parabola} with \emph{vertex} $S(\alpha, \beta)$. It
opens upward if $a > 0$, downward if $a < 0$, and is symmetric about the
vertical line $x = \alpha$.
\end{definition}

\begin{remark}
The symmetry is easy to check: for any $h$,
$f(\alpha + h) = ah^2 + \beta = f(\alpha - h)$, so two points at equal
horizontal distance from the line $x = \alpha$ have the same height.
\end{remark}

\begin{method}[Reading a parabola]\label{met:g11:quad:reading}
To sketch $f(x) = ax^2+bx+c$: find the orientation (sign of $a$), the
vertex $\left(-\frac{b}{2a},\, f\!\left(-\frac{b}{2a}\right)\right)$, the
\omterm{def:g10:numbers:interunion}{intersection} $(0, c)$ with the $y$-axis, and the \omterm{def:g11:quad:discriminant}{roots}, if any. The axis of
symmetry $x = -\frac{b}{2a}$ then places every computed point twice.
\end{method}

\begin{example}
For $f(x) = x^2 - 4x + 3$: upward, vertex $(2, -1)$, crosses the $y$-axis
at $(0,3)$, \omterm{def:g11:quad:discriminant}{roots} $1$ and $3$ (spotted: $1 - 4 + 3 = 0$). By symmetry, the
point $(4, 3)$ is also on the curve, mirroring $(0, 3)$.
\end{example}

2x^2 + 3x - 2 = 0, \qquad

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:quad:2}
Write in \omterm{thm:g11:quad:canonical}{canonical form} and give the vertex of the \omterm{def:g11:quad:parabola}{parabola}:

f(x) = x^2 - 6x + 11, \qquad
g(x) = -2x^2 + 4x + 1, \qquad

\end{exercise}

Solve the inequalities

x^2 - 3x - 10 < 0, \qquad
2x^2 + x + 3 > 0, \qquad
-x^2 + 6x - 9 \geq 0 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:quad:4}
Without computing $\Delta$, find one obvious \omterm{def:g11:quad:discriminant}{root}, then the other:
\[
x^2 - 7x + 6 = 0, \qquad
2x^2 + 5x - 7 = 0, \qquad
x^2 + 4x - 5 = 0 .
\]
(Hint: test $x = 1$ and $x = -1$, then use the product of the \omterm{def:g11:quad:discriminant}{roots}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:5}
Find two numbers whose sum is $14$ and whose product is $45$. Then find the

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:6}
Let $f(x) = x^2 + (m-2)x + 1$, where $m$ is a real parameter. For which

solution?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:7}
A ball is thrown upward; its height after $t$ seconds is
$h(t) = -5t^2 + 20t + 1$ (in meters). What is the maximal height reached,
and at what time? During which time \omterm{def:g10:numbers:interval}{interval} is the ball at least $16$ m
high?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:8}
Solve the \omterm{def:g10:algebra:equation}{equation} $x^4 - 13x^2 + 36 = 0$ by setting $X = x^2$

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:9}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:10}

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:quad:11}
Let $\mathcal P$ be the \omterm{def:g11:quad:parabola}{parabola} $y = x^2$ and let $d_m$ be the line

\begin{enumerate}
  \item For which values of $m$ do $\mathcal P$ and $d_m$ intersect in two

  \item When they touch in exactly one point, compute the contact point.

\end{enumerate}
\end{exercise}

\section{Problem: The parabola's secret point}

\begin{problem}[{Weekend problem --- focus and directrix: why
satellite dishes, headlights and suspension bridges all draw the
same curve, of which there is essentially only one}]
\label{pb:g11:quad:1}
Every satellite dish on every balcony aims its receiver at one
precise interior point; every headlight hides its bulb at the
same special spot. That point --- the \emph{\omterm{pb:g11:quad:1}{focus}} --- is a
secret the \omterm{def:g11:quad:parabola}{parabola} has kept since Greek geometry, and the
distance formula is enough to extract it. Along the way, this
problem drills the \omterm{def:g11:quad:discriminant}{discriminant}, throws projectiles over walls,
suspends a bridge, and proves an astonishing final fact: up to
zoom, \emph{there is only one \omterm{def:g11:quad:parabola}{parabola} in the world}.

\medskip
\noindent\textbf{Part I --- Trinomial fluency.}
\begin{enumerate}
  \item Solve $x^2 - 5x + 6 = 0$, then $2x^2 + x - 6 = 0$, then
        $x^2 + x + 1 = 0$ (\cref{thm:g11:quad:roots}).
  \item Write $2x^2 - 4x - 6$ in factored form and in \omterm{thm:g11:quad:canonical}{canonical
        form} (\cref{thm:g11:quad:canonical}). Together with the
        expanded form, you now hold three costumes of one
        \omterm{def:g10:functions:function}{function}: say which feature each displays at a glance
        (\omterm{def:g11:quad:discriminant}{roots}, vertex, $y$-intercept).
  \item Solve $-x^2 + 4x - 3 \geq 0$ with the sign rule
        (\cref{thm:g11:quad:sign}).
  \item Vi\`ete as detective (\cref{prop:g11:quad:vieta},
        continuing Babylon in \cref{pb:g10:algebra:1}): find two
        numbers with sum $7$ and product $12$ by writing their
        \omterm{def:g10:algebra:equation}{equation}. Then: one \omterm{def:g11:quad:discriminant}{root} of $x^2 - 5x + c = 0$ equals
        $2$; find $c$ and the other \omterm{def:g11:quad:discriminant}{root} without solving
        anything.
  \item For which values of $m$ does $x^2 + mx + 9 = 0$ have a
        double \omterm{def:g11:quad:discriminant}{root}? No \omterm{def:g11:quad:discriminant}{root}? Two \omterm{def:g11:quad:discriminant}{roots}?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The \omterm{pb:g11:quad:1}{focus} unveiled.}
Work on the \omterm{def:g11:quad:parabola}{parabola} $y = x^2$, the point
$F\left(0, \frac14\right)$ and the horizontal line $\delta$ of
\omterm{def:g10:algebra:equation}{equation} $y = -\frac14$.
\begin{enumerate}[resume]
  \item For a point $P(x, x^2)$ of the \omterm{def:g11:quad:parabola}{parabola}, expand
        $PF^2 = x^2 + \left(x^2 - \frac14\right)^2$ and show it
        is the perfect square $\left(x^2 + \frac14\right)^2$.
        Deduce $PF$.
  \item Compute the distance from $P$ to the line $\delta$, and
        conclude: \emph{every point of the \omterm{def:g11:quad:parabola}{parabola} is exactly
        as far from $F$ as from $\delta$}. ($F$ is the
        \emph{focus}\index{focus}, $\delta$ the
        \emph{directrix}\index{directrix}.)
  \item Prove the converse: a point $P(x, y)$ with
        $PF = \operatorname{dist}(P, \delta)$ must satisfy
        $y = x^2$. (Square both distances and simplify.) The
        \omterm{def:g11:quad:parabola}{parabola} is thus a \emph{locus}, exactly as the circle
        was in \cref{pb:g10:coordgeom:1}: equidistance from a
        point and a line, instead of from one point.
  \item For the wider \omterm{def:g11:quad:parabola}{parabola} $y = \frac{x^2}{4f}$ (with
        $f > 0$), adapt question 6 to show the \omterm{pb:g11:quad:1}{focus} is
        $(0, f)$ and the \omterm{pb:g11:quad:1}{directrix} $y = -f$. Application: a
        satellite dish is $60$ cm wide and $9$ cm deep --- its
        rim passes through $(30, 9)$. Find $f$: how high above
        the vertex must the receiver hang?
  \item The reason dishes work: rays arriving parallel to the
        axis all reflect \emph{through the \omterm{pb:g11:quad:1}{focus}} (and a bulb at
        the \omterm{pb:g11:quad:1}{focus} throws a parallel beam --- headlights are
        dishes run backwards). Explain why question 7's
        equidistance makes this plausible (think of the rays as
        marching wavefronts), and say which tool, coming in
        \cref{ch:g11:deriv} (see \cref{exo:g11:quad:11}), turns
        plausibility into proof.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- \omterm{def:g11:quad:parabola}{Parabolas} at work.}
\begin{enumerate}[resume]
  \item A ball follows the trajectory
        $y = x - \frac{x^2}{20}$ ($x$ and $y$ in meters). Find
        where it lands, the horizontal position of its highest
        point, and the maximal height
        (\cref{prop:g11:quad:variations}).
  \item On the same trajectory, a wall stands at $x = 4$.
        Compute the ball's height there: does it clear a $3$ m
        wall? A $4$ m wall?
  \item A parabolic arch spans $40$ m with maximal height
        $10$ m: in centered \omterm{def:g10:coordgeom:system}{coordinates},
        $y = 10 - \frac{x^2}{40}$. Can a truck $7$ m tall pass
        at $10$ m from the center? At $15$ m?
  \item A theater's revenue at ticket price $p$ euros is
        $R(p) = p\,(120 - 2p)$ (demand falls as price rises).
        Find the price maximizing revenue, and the \omterm{def:g10:functions:extrema}{maximum}.
  \item The Golden Gate's main cables are parabolic to
        excellent accuracy (a uniform deck load produces a
        \omterm{def:g11:quad:parabola}{parabola} --- a freely hanging chain does \emph{not}: its
        curve, the catenary, needs the exponential of grade 12).
        With span $1\,280$ m and sag $143$ m, the cable follows
        $y = 143\left(\frac{x}{640}\right)^2$ from the center.
        How high is the cable above its lowest point at
        $x = 320$ m?
\end{enumerate}

\noindent\textbf{Part IV --- Lines, tangents, and the last
surprise.}
\begin{enumerate}[resume]
  \item Complete \cref{exo:g11:quad:11}: the lines
        $y = mx - 1$ through the external point $(0, -1)$ meet
        $y = x^2$ according to the sign of a \omterm{def:g11:quad:discriminant}{discriminant}.
        Find the two tangent lines from $(0, -1)$ and their
        contact points.
  \item Now the lines $y = mx + 1$ through the \emph{interior}
        point $(0, 1)$: compute the \omterm{def:g11:quad:discriminant}{discriminant} of the
        \omterm{def:g10:numbers:interunion}{intersection} \omterm{def:g10:algebra:equation}{equation} and show every such line cuts the
        \omterm{def:g11:quad:parabola}{parabola} twice. Moral: from inside, no tangents ---
        just as for a circle.
  \item The focal chord: intersect $y = x^2$ with the
        horizontal line through the \omterm{pb:g11:quad:1}{focus}, $y = \frac14$. What
        is the chord's length? Check the general rule (worth
        remembering for dishes): for $y = \frac{x^2}{4f}$, the
        chord through the \omterm{pb:g11:quad:1}{focus} parallel to the \omterm{pb:g11:quad:1}{directrix} has
        length $4f$.
  \item The last surprise: apply the zoom
        $(x, y) \mapsto (kx, ky)$ to the \omterm{def:g11:quad:parabola}{parabola} $y = x^2$ and
        show the \omterm{def:g10:functions:function}{image} is exactly $y = \frac{x^2}{k}$. Conclude:
        every \omterm{def:g11:quad:parabola}{parabola} $y = \frac{x^2}{4f}$ is a magnified unit
        \omterm{def:g11:quad:parabola}{parabola} --- unlike circles and ellipses of different
        proportions, \emph{all \omterm{def:g11:quad:parabola}{parabolas} are similar}.
  \item Finale: the \omterm{def:g11:quad:parabola}{parabola}'s portrait in five lines --- an
        \omterm{def:g10:algebra:equation}{equation} with three costumes (question 2), a locus
        (questions 7--8), an engineer's reflector (questions 9,
        10), a projectile's path and a bridge's cable
        (questions 11--15), and a single curve up to zoom
        (question 19). One sentence each.
\end{enumerate}
\end{problem}
