\chapter{Quadratic Functions and Equations}\label{ch:g11:quad}

Quadratic functions are the simplest functions after the linear ones, and
the first whose graphs are genuinely curved. This chapter develops the
complete toolbox for them: the canonical form, the discriminant, the sign
of a quadratic expression, and the geometry of the parabola.

\section{The canonical form}

\begin{definition}[Quadratic function]\label{def:g11:quad:quadratic}
A \emph{quadratic function}\index{quadratic function} is a function of the
form
\[
f(x) = ax^2 + bx + c, \qquad a \neq 0,
\]
where $a$, $b$, $c$ are real numbers. The expression $ax^2 + bx + c$ is
also called a \emph{quadratic trinomial}.
\end{definition}

\begin{theorem}[Canonical form]\label{thm:g11:quad:canonical}
Every quadratic function can be written in exactly one way as
\[
f(x) = a(x - \alpha)^2 + \beta,
\qquad\text{where } \alpha = -\frac{b}{2a} \text{ and } \beta = f(\alpha).
\]
This is the \emph{canonical form}\index{canonical form} of $f$.
\end{theorem}

\begin{proof}
We \emph{complete the square}. Since $a \neq 0$, factor it out of the first
two terms:
\[
ax^2 + bx + c = a\left(x^2 + \frac{b}{a}x\right) + c .
\]
Now $x^2 + \frac{b}{a}x$ is the beginning of the square
$\left(x + \frac{b}{2a}\right)^2 = x^2 + \frac ba x + \frac{b^2}{4a^2}$,
so $x^2 + \frac{b}{a}x = \left(x + \frac{b}{2a}\right)^2 -
\frac{b^2}{4a^2}$. Substituting:
\[
f(x) = a\left(x + \frac{b}{2a}\right)^2 - \frac{b^2}{4a} + c
= a\bigl(x - \alpha\bigr)^2 + \beta,
\]
with $\alpha = -\frac{b}{2a}$ and $\beta = c - \frac{b^2}{4a}$. Evaluating
at $x = \alpha$ kills the squared term, so $\beta = f(\alpha)$. Uniqueness:
expanding $a(x-\alpha)^2 + \beta$ and identifying the coefficients of $x$
forces $\alpha = -\frac{b}{2a}$, and then $\beta$ is determined.
\end{proof}

\begin{example}\label{ex:g11:quad:canonical}
Let $f(x) = 2x^2 - 8x + 3$. Then $\alpha = -\frac{-8}{2 \times 2} = 2$ and
$\beta = f(2) = 8 - 16 + 3 = -5$, so
\[
f(x) = 2(x - 2)^2 - 5 .
\]
Check by expanding: $2(x^2 - 4x + 4) - 5 = 2x^2 - 8x + 8 - 5 =
2x^2 - 8x + 3$.
\end{example}

\begin{proposition}[Extremum and variations]\label{prop:g11:quad:variations}
Let $f(x) = a(x-\alpha)^2 + \beta$.
\begin{enumerate}
  \item If $a > 0$, then $f$ is decreasing on $\intoc{-\infty}{\alpha}$ and
        increasing on $\intco{\alpha}{+\infty}$: $f$ has a \emph{minimum}
        $\beta$, attained at $x = \alpha$.
  \item If $a < 0$, the variations are reversed: $f$ has a \emph{maximum}
        $\beta$ at $x = \alpha$.
\end{enumerate}
\end{proposition}

\begin{proof}
Suppose $a > 0$ and let $\alpha \leq u < v$. Then
\[
f(v) - f(u) = a\bigl((v-\alpha)^2 - (u-\alpha)^2\bigr)
= a\,(v - u)\,\bigl((v-\alpha) + (u-\alpha)\bigr).
\]
Each factor is positive: $a > 0$, $v - u > 0$, and
$(v-\alpha) + (u-\alpha) > 0$ since $v > \alpha$ and $u \geq \alpha$. So
$f(v) > f(u)$: $f$ is increasing on $\intco{\alpha}{+\infty}$. For
$u < v \leq \alpha$ the last factor is negative, so $f(v) < f(u)$: $f$ is
decreasing there. Finally $f(x) - f(\alpha) = a(x-\alpha)^2 \geq 0$ for all
$x$, so $\beta = f(\alpha)$ is the minimum. The case $a < 0$ follows by
applying the above to $-f$.
\end{proof}

\begin{example}
A farmer has $100$ meters of fence to enclose a rectangular field against a
straight wall (no fence needed along the wall). If $x$ is the width, the
enclosed area is $A(x) = x(100 - 2x) = -2x^2 + 100x$. Here $\alpha =
-\frac{100}{2\times(-2)} = 25$ and $A(25) = 25 \times 50 = 1250$: the area
is largest for a width of $25$ m, giving $1250$ m$^2$.
\end{example}

\section{Roots and factorization}

\begin{definition}[Root, discriminant]\label{def:g11:quad:discriminant}
A \emph{root}\index{root} of the trinomial $ax^2+bx+c$ is a solution of the
equation $ax^2 + bx + c = 0$. The \emph{discriminant}\index{discriminant}
of the trinomial is the number
\[
\Delta = b^2 - 4ac .
\]
\end{definition}

\begin{theorem}[Solving $ax^2+bx+c=0$]\label{thm:g11:quad:roots}
\begin{enumerate}
  \item If $\Delta > 0$, the equation has two distinct roots
        \[
        x_1 = \frac{-b - \sqrt{\Delta}}{2a},
        \qquad
        x_2 = \frac{-b + \sqrt{\Delta}}{2a},
        \]
        and $ax^2+bx+c = a(x - x_1)(x - x_2)$.
  \item If $\Delta = 0$, the equation has the single root
        $x_0 = -\frac{b}{2a}$, and $ax^2+bx+c = a(x - x_0)^2$.
  \item If $\Delta < 0$, the equation has no real solution, and the
        trinomial cannot be factored with real coefficients.
\end{enumerate}
\end{theorem}

\begin{proof}
Start from the canonical form computed in \cref{thm:g11:quad:canonical}:
\[
ax^2+bx+c
= a\left(x + \frac{b}{2a}\right)^2 + c - \frac{b^2}{4a}
= a\left[\left(x + \frac{b}{2a}\right)^2 - \frac{\Delta}{4a^2}\right],
\]
since $c - \frac{b^2}{4a} = \frac{4ac - b^2}{4a} = -\frac{\Delta}{4a} =
a \times \left(-\frac{\Delta}{4a^2}\right)$.

\emph{1.} If $\Delta > 0$, then $\frac{\Delta}{4a^2} =
\left(\frac{\sqrt\Delta}{2a}\right)^2$ and the bracket is a difference of
two squares:
\[
ax^2+bx+c = a\left(x + \frac{b}{2a} - \frac{\sqrt\Delta}{2a}\right)
\left(x + \frac{b}{2a} + \frac{\sqrt\Delta}{2a}\right)
= a(x - x_2)(x - x_1).
\]
A product is zero exactly when one of its factors is zero, giving the two
roots, distinct because $\sqrt\Delta \neq 0$.

\emph{2.} If $\Delta = 0$, the bracket is
$\left(x + \frac{b}{2a}\right)^2$, which vanishes only at
$x_0 = -\frac{b}{2a}$.

\emph{3.} If $\Delta < 0$, then $-\frac{\Delta}{4a^2} > 0$, so the bracket
is a square plus a positive number: it never vanishes, and the equation has
no real solution. A factorization $a(x-x_1)(x-x_2)$ would produce roots, so
none exists.
\end{proof}

\begin{example}\label{ex:g11:quad:solving}
Solve $2x^2 - 8x + 3 = 0$. Here $\Delta = (-8)^2 - 4 \times 2 \times 3 =
64 - 24 = 40 > 0$, so there are two roots:
\[
x_{1,2} = \frac{8 \pm \sqrt{40}}{4} = \frac{8 \pm 2\sqrt{10}}{4}
= 2 \pm \frac{\sqrt{10}}{2}.
\]
For $x^2 + x + 1 = 0$: $\Delta = 1 - 4 = -3 < 0$, no real solution.
\end{example}

\begin{proposition}[Sum and product of the roots]\label{prop:g11:quad:vieta}
If $\Delta \geq 0$ and $x_1, x_2$ are the roots (equal when $\Delta = 0$),
then
\[
x_1 + x_2 = -\frac{b}{a},
\qquad
x_1 x_2 = \frac{c}{a} .
\]
Conversely, two numbers with sum $s$ and product $p$ are the solutions of
$x^2 - sx + p = 0$.
\end{proposition}

\begin{proof}
Expand the factorization of \cref{thm:g11:quad:roots}:
\[
a(x - x_1)(x - x_2) = a x^2 - a(x_1 + x_2)x + a\,x_1 x_2 .
\]
Identifying the coefficients with $ax^2 + bx + c$ gives
$-a(x_1+x_2) = b$ and $a\,x_1x_2 = c$. For the converse, if $u + v = s$ and
$uv = p$, then $(x-u)(x-v) = x^2 - sx + p$, so $u$ and $v$ are its roots.
\end{proof}

\begin{example}
Find two numbers with sum $10$ and product $21$. They solve
$x^2 - 10x + 21 = 0$; $\Delta = 100 - 84 = 16$, roots
$\frac{10 \pm 4}{2} = 7$ and $3$. \emph{Quick root spotting:} since the
coefficients of $x^2 - 5x + 4$ satisfy $1 - 5 + 4 = 0$, the number $1$ is a
root, and the other one is $\frac{c}{a} = 4$ (their product must be
$\frac ca$).
\end{example}

\section{Sign of a quadratic}

\begin{theorem}[Sign of a trinomial]\label{thm:g11:quad:sign}
Let $f(x) = ax^2+bx+c$ with $a \neq 0$.
\begin{enumerate}
  \item If $\Delta > 0$, with roots $x_1 < x_2$: $f(x)$ has the sign of $a$
        outside $\intcc{x_1}{x_2}$, and the sign of $-a$ on
        $\intoo{x_1}{x_2}$.
  \item If $\Delta = 0$: $f(x)$ has the sign of $a$ for all $x \neq x_0$,
        and vanishes at $x_0$.
  \item If $\Delta < 0$: $f(x)$ has the sign of $a$ for all $x$.
\end{enumerate}
In short: \emph{a trinomial has the sign of $a$, except between its roots}.
\end{theorem}

\begin{proof}
\emph{1.} Write $f(x) = a(x-x_1)(x-x_2)$ and follow the signs of the
factors. If $x < x_1$: both $x - x_1$ and $x - x_2$ are negative, their
product is positive, so $f(x)$ has the sign of $a$. If $x_1 < x < x_2$: the
factors have opposite signs, the product is negative, and $f(x)$ has the
sign of $-a$. If $x > x_2$: both factors are positive and $f(x)$ has the
sign of $a$ again.

\emph{2.} $f(x) = a(x - x_0)^2$ and $(x-x_0)^2 > 0$ for $x \neq x_0$.

\emph{3.} From the proof of \cref{thm:g11:quad:roots},
$f(x) = a\left[\left(x + \frac{b}{2a}\right)^2 +
\left(-\frac{\Delta}{4a^2}\right)\right]$, and the bracket is always
positive.
\end{proof}

\begin{method}[Solving quadratic inequalities]\label{met:g11:quad:inequalities}
To solve $ax^2 + bx + c \leq 0$ (or $\geq$, $<$, $>$):
\begin{enumerate}
  \item compute $\Delta$ and the roots, if any;
  \item draw the sign table using \cref{thm:g11:quad:sign} (sign of $a$
        outside the roots);
  \item read off the solution interval(s), paying attention to whether the
        roots themselves are included (wide inequality) or not (strict).
\end{enumerate}
\end{method}

\begin{example}\label{ex:g11:quad:inequality}
Solve $-x^2 + 3x + 4 \geq 0$. Here $\Delta = 9 + 16 = 25$, and the roots
are $\frac{-3 \pm 5}{-2}$, that is $x_1 = -1$ and $x_2 = 4$. Since
$a = -1 < 0$, the trinomial is negative outside the roots and positive
between them. With the wide inequality, the solution set is
$\intcc{-1}{4}$.
\end{example}

\section{The parabola}

\begin{definition}[Parabola]\label{def:g11:quad:parabola}
The graph of a quadratic function $f(x) = a(x-\alpha)^2+\beta$ is a
\emph{parabola}\index{parabola} with \emph{vertex} $S(\alpha, \beta)$. It
opens upward if $a > 0$, downward if $a < 0$, and is symmetric about the
vertical line $x = \alpha$.
\end{definition}

\begin{remark}
The symmetry is easy to check: for any $h$,
$f(\alpha + h) = ah^2 + \beta = f(\alpha - h)$, so two points at equal
horizontal distance from the line $x = \alpha$ have the same height.
\end{remark}

\begin{method}[Reading a parabola]\label{met:g11:quad:reading}
To sketch $f(x) = ax^2+bx+c$: find the orientation (sign of $a$), the
vertex $\left(-\frac{b}{2a},\, f\!\left(-\frac{b}{2a}\right)\right)$, the
intersection $(0, c)$ with the $y$-axis, and the roots, if any. The axis of
symmetry $x = -\frac{b}{2a}$ then places every computed point twice.
\end{method}

\begin{example}
For $f(x) = x^2 - 4x + 3$: upward, vertex $(2, -1)$, crosses the $y$-axis
at $(0,3)$, roots $1$ and $3$ (spotted: $1 - 4 + 3 = 0$). By symmetry, the
point $(4, 3)$ is also on the curve, mirroring $(0, 3)$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g11:quad:1}
Solve the equations
\[
x^2 - 5x + 6 = 0, \qquad
2x^2 + 3x - 2 = 0, \qquad
x^2 + 2x + 5 = 0, \qquad
9x^2 - 6x + 1 = 0 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:quad:2}
Write in canonical form and give the vertex of the parabola:
\[
f(x) = x^2 - 6x + 11, \qquad
g(x) = -2x^2 + 4x + 1, \qquad
h(x) = 3x^2 + 6x .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:quad:3}
Solve the inequalities
\[
x^2 - 3x - 10 < 0, \qquad
2x^2 + x + 3 > 0, \qquad
-x^2 + 6x - 9 \geq 0 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:quad:4}
Without computing $\Delta$, find one obvious root, then the other:
\[
x^2 - 7x + 6 = 0, \qquad
2x^2 + 5x - 7 = 0, \qquad
x^2 + 4x - 5 = 0 .
\]
(Hint: test $x = 1$ and $x = -1$, then use the product of the roots.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:5}
Find two numbers whose sum is $14$ and whose product is $45$. Then find the
dimensions of a rectangle of perimeter $34$ and area $60$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:6}
Let $f(x) = x^2 + (m-2)x + 1$, where $m$ is a real parameter. For which
values of $m$ does the equation $f(x) = 0$ have exactly one solution? No
solution?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:7}
A ball is thrown upward; its height after $t$ seconds is
$h(t) = -5t^2 + 20t + 1$ (in meters). What is the maximal height reached,
and at what time? During which time interval is the ball at least $16$ m
high?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:8}
Solve the equation $x^4 - 13x^2 + 36 = 0$ by setting $X = x^2$
(a \emph{biquadratic} equation).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:9}
For which values of $x$ is the point $M(x, x^2)$ of the parabola $y = x^2$
strictly below the line $y = x + 2$? Interpret on a sketch.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:quad:10}
The sum of a positive number and its reciprocal is $\frac{13}{6}$. Find the
number. (Reduce to a quadratic equation.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:quad:11}
Let $\mathcal P$ be the parabola $y = x^2$ and let $d_m$ be the line
$y = mx - 1$, where $m$ is a real parameter.
\begin{enumerate}
  \item For which values of $m$ do $\mathcal P$ and $d_m$ intersect in two
        points? One point? No point?
  \item When they touch in exactly one point, compute the contact point.
        (In \cref{ch:g11:deriv} you will recognize $d_m$ as the tangent to
        $\mathcal P$ at that point.)
\end{enumerate}
\end{exercise}
