\chapter{Fractions: First Steps}\label{ch:g6:fractions}

Sharing three pizzas fairly among four people gives each person\,\dots\
not a whole number of pizzas. \omterm{def:g6:fractions:def}{Fractions} are the numbers invented for
sharing. This chapter builds the picture: a \omterm{def:g6:fractions:def}{fraction} as a part of a
whole, as a point on the number line, and as the exact result of a
division.

\section{What a fraction means}

\begin{definition}[Fraction]\label{def:g6:fractions:def}
Cut a unit into $b$ equal parts, and take $a$ of them: the quantity
obtained is the \emph{fraction}\index{fraction}
\[
\frac{a}{b}
\qquad
\begin{array}{l}
a \text{: the \emph{numerator} --- how many parts we take;}\\
b \text{: the \emph{denominator} --- into how many parts the unit is cut.}
\end{array}
\]
\end{definition}

\begin{example}\label{ex:g6:fractions:morethanone}
A \omterm{def:g6:fractions:def}{fraction} can be bigger than $1$: $\frac74$ means seven \omterm{def:g3:division:half}{quarters} ---
one whole unit (four \omterm{def:g3:division:half}{quarters}) and three \omterm{def:g3:division:half}{quarters} more:
\[
\frac74 = 1 + \frac34 .
\]
\end{example}

\begin{proposition}[Fractions on the number line]\label{prop:g6:fractions:line}
To place $\frac ab$ on the number line, cut each unit interval into $b$
equal parts and count $a$ parts from $0$.
\end{proposition}

\begin{theorem}[A fraction is a division]\label{thm:g6:fractions:quotient}
The \omterm{def:g6:fractions:def}{fraction} $\frac ab$ is the number which, multiplied by $b$, gives
$a$:
\[
\frac ab \times b = a .
\]
In other words, $\frac ab$ is the exact \omterm{def:g3:division:remainder}{quotient} of $a$ by $b$ --- even
when the division ``does not come out even''.
\end{theorem}
\admitted

\begin{example}\label{ex:g6:fractions:pizzas}
Three pizzas for four people: each person gets $\frac34$ of a pizza.
Check with the theorem: $4$ shares of $\frac34$ make
$\frac34 \times 4 = 3$ pizzas. Some \omterm{def:g6:fractions:def}{fractions} are \omterm{def:g6:decimals:places}{decimal numbers}
($\frac34 = 0.75$), others are not: $\frac13 = 0.333\dots$ never stops.
The \omterm{def:g6:fractions:def}{fraction} is the \emph{exact} value; $0.33$ is only an approximation.
\end{example}

\section{Equal fractions}

\begin{proposition}[Equivalent fractions]\label{prop:g6:fractions:equal}
A \omterm{def:g6:fractions:def}{fraction} does not change when its \omterm{def:g4:fractions:def}{numerator} and \omterm{def:g4:fractions:def}{denominator} are both
multiplied (or divided) by the same nonzero number:
\[
\frac{a}{b} = \frac{a \times k}{b \times k} .
\]
\end{proposition}

\begin{proof}[Idea of proof]
Cutting each part of the sharing into $k$ smaller pieces multiplies both
the number of pieces taken and the total number of pieces by $k$, but
the quantity taken is the same. A picture says it best:
\end{proof}

\begin{omfigure}
\begin{tikzpicture}[scale=0.6]
  \foreach \i in {0,1,2} \draw[omExo] (\i*2.4,0) rectangle (\i*2.4+2.4,1.2);
  \foreach \i in {0,1} \fill[omDef!40] (\i*2.4+0.05,0.05)
    rectangle (\i*2.4+2.35,1.15);
  \draw[thick] (0,0) rectangle (7.2,1.2);
  \node[below, font=\small] at (3.6,-0.15) {$\frac23$};
  \begin{scope}[xshift=10.5cm]
  \foreach \i in {0,...,5} \draw[omExo] (\i*1.2,0) rectangle (\i*1.2+1.2,1.2);
  \foreach \i in {0,...,3} \fill[omDef!40] (\i*1.2+0.05,0.05)
    rectangle (\i*1.2+1.15,1.15);
  \draw[thick] (0,0) rectangle (7.2,1.2);
  \node[below, font=\small] at (3.6,-0.15) {$\frac46$};
  \end{scope}
\end{tikzpicture}

{\small $\frac23 = \frac46$: cutting each third in two doubles the
counts, top and bottom, without changing the shaded amount.}
\end{omfigure}

\begin{method}[Simplifying a fraction]\label{met:g6:fractions:simplify}
Look for a number dividing the \omterm{def:g4:fractions:def}{numerator} \emph{and} the \omterm{def:g4:fractions:def}{denominator},
and divide both. Repeat until no common \omterm{def:g5:division:multiple}{divisor} remains:
\[
\frac{12}{18} = \frac{12 \div 2}{18 \div 2} = \frac{6}{9}
= \frac{6 \div 3}{9 \div 3} = \frac{2}{3}.
\]
\end{method}

\section{Taking a fraction of a quantity}

\begin{method}[Fraction of a quantity]\label{met:g6:fractions:ofquantity}
To compute $\frac ab$ of a quantity, divide the quantity by $b$ (one
part), then multiply by $a$ (the number of parts):
\[
\frac ab \text{ of } Q = (Q \div b) \times a .
\]
(Multiplying first by $a$ and then dividing by $b$ gives the same
result --- choose the order that makes the numbers nicer.)
\end{method}

\begin{example}\label{ex:g6:fractions:ofquantity}
Compute $\frac35$ of $40$ euros, step by step:
\begin{enumerate}
  \item one fifth of $40$: $40 \div 5 = 8$;
  \item three fifths: $8 \times 3 = 24$.
\end{enumerate}
So $\frac35$ of $40$ euros is $24$ euros. Check: the remaining
$\frac25$ is $16$ euros, and $24 + 16 = 40$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g6:fractions:1}
For each picture description, write the \omterm{def:g6:fractions:def}{fraction}: a cake cut in $8$
slices, $3$ eaten; a chocolate bar of $12$ squares, $7$ eaten; a pie
cut in $6$, all $6$ eaten.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:2}
Draw a bar cut into $5$ equal parts and color $\frac45$ of it. Is
$\frac45$ smaller or larger than $1$? And $\frac65$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:3}
Place on a number line graduated in thirds: $\frac13$; \ $\frac53$;
\ $2$; \ $\frac73$. Which two of these numbers are equal distances from
$2$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:4}
Write each \omterm{def:g6:fractions:def}{fraction} as a whole number plus a \omterm{def:g6:fractions:def}{fraction} smaller than $1$
(as in $\frac74 = 1 + \frac34$):
\[
\frac{9}{4}, \qquad \frac{13}{5}, \qquad \frac{12}{6}, \qquad
\frac{25}{8} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:5}
Complete so the \omterm{def:g6:fractions:def}{fractions} are equal:
\[
\frac{2}{3} = \frac{?}{12}, \qquad
\frac{15}{20} = \frac{3}{?}, \qquad
\frac{7}{10} = \frac{21}{?}, \qquad
\frac{?}{6} = \frac{10}{12}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:6}
Simplify as much as possible: $\dfrac{6}{8}$; \quad $\dfrac{15}{25}$;
\quad $\dfrac{18}{24}$; \quad $\dfrac{35}{7}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:7}
Compute: $\frac12$ of $86$; \quad $\frac34$ of $60$; \quad $\frac25$ of
$35$; \quad $\frac{7}{10}$ of $250$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:8}
Which is larger: $\frac12$ or $\frac13$ (of the same cake)? $\frac25$
or $\frac35$? $\frac34$ or $\frac78$ (write both with \omterm{def:g4:fractions:def}{denominator}
$8$)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:fractions:9}
Give the decimal value of the \omterm{def:g6:fractions:def}{fractions} that have one, and say which
\omterm{def:g6:fractions:def}{fraction} does not: $\frac12$; \ $\frac34$; \ $\frac13$; \ $\frac{7}{10}$;
\ $\frac{9}{4}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:fractions:10}
A class has $28$ students. Three \omterm{def:g3:division:half}{quarters} of them come to school by
bus, and the rest walk. How many students walk? Solve step by step, and
check that your two group sizes add up to $28$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:fractions:11}
Leo spent $\frac23$ of his savings on a game that cost $18$ euros. How
much money did Leo have before? (One third of his savings is \omterm{def:g3:division:half}{half} of
$18$\,\dots\ think about why, or find the size of one part first.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:fractions:12}
A tank is filled to $\frac58$ of its \omterm{def:g2:measure:units}{capacity}. After using $10$ liters,
it is \omterm{def:g3:division:half}{half} full. What is the \omterm{def:g2:measure:units}{capacity} of the tank? (Which \omterm{def:g6:fractions:def}{fraction} of
the tank do the $10$ liters represent?)
\end{exercise}

\section{Problem: The chocolate bar you can never finish}

\begin{problem}[{Weekend problem --- halves of halves: the sum
$\frac12 + \frac14 + \frac18 + \dots$ creeps up to $1$ without ever
touching it}]
\label{pb:g6:fractions:1}
Take a chocolate bar. Eat \omterm{def:g3:division:half}{half} of it. Then eat \omterm{def:g3:division:half}{half} of \emph{what
is left}. Then again \omterm{def:g3:division:half}{half} of what is left, and so on. Two things
seem to be true at once: you never finish the bar (something always
remains), and yet you eat almost all of it. This problem turns both
feelings into exact statements about \omterm{def:g6:fractions:def}{fractions} --- and meets, on
the way, a runner from ancient Greece and a sharing puzzle solved
entirely with a knife that only cuts things in \omterm{def:g3:division:half}{half}.

\medskip
\noindent\textbf{Part I --- Halves of halves.}
The bar is a rectangle of $16$ equal squares.
\begin{enumerate}
  \item First bite: you eat \omterm{def:g3:division:half}{half} the bar. How many squares is
        that? Write the bite as a \omterm{def:g6:fractions:def}{fraction} of the bar in two ways,
        with \omterm{def:g4:fractions:def}{denominator} $2$ and with \omterm{def:g4:fractions:def}{denominator} $16$
        (\cref{prop:g6:fractions:equal}).
  \item Second bite: \omterm{def:g3:division:half}{half} of what is left. How many squares? Show,
        with the picture or with equivalent \omterm{def:g6:fractions:def}{fractions}, that this
        bite is $\frac14$ of the whole bar: \emph{\omterm{def:g3:division:half}{half} of a \omterm{def:g3:division:half}{half}
        is a \omterm{def:g3:division:half}{quarter}}.
  \item Third and fourth bites, same rule: give each one in
        squares and as a \omterm{def:g6:fractions:def}{fraction} of the bar. What happens to the
        \omterm{def:g4:fractions:def}{denominator} from one bite to the next?
  \item After the four bites, how many squares have been eaten in
        all? What \omterm{def:g6:fractions:def}{fraction} of the bar is that, and what \omterm{def:g6:fractions:def}{fraction}
        is left?
  \item Question 4 says, written in sixteenths:
        \[
        \frac{8}{16} + \frac{4}{16} + \frac{2}{16} + \frac{1}{16}
        = \frac{15}{16},
        \qquad\text{that is}\qquad
        \frac12 + \frac14 + \frac18 + \frac{1}{16}
        = \frac{15}{16} .
        \]
        Check each of the four rewritings
        ($\frac12 = \frac{8}{16}$, and so on).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The staircase towards $1$.}
To keep biting, imagine a finer bar of $64$ squares.
\begin{enumerate}[resume]
  \item List the sizes of the first six bites, in squares
        ($32$, then\,\dots), and check that after the sixth bite
        exactly one square is left. What \omterm{def:g6:fractions:def}{fraction} of the bar
        remains?
  \item After each bite, what \omterm{def:g6:fractions:def}{fraction} of the bar remains? Write
        the list ($\frac12$, $\frac14$, \dots) down to the sixth
        bite, and describe the pattern. If an imaginary knife
        could halve \omterm{def:g3:numbers:evenodd}{even} the last square, what would remain after
        a seventh bite?
  \item Explain why the eaten total can \emph{never} reach $1$,
        no matter how many bites are taken. (Compare with the
        twenty nines of \cref{pb:g6:decimals:1}: same story?)
  \item Yet the total passes any target below $1$. A friend
        challenges you to eat more than $\frac{99}{100}$ of the
        bar. Which is smaller, $\frac{1}{128}$ or
        $\frac{1}{100}$ --- and after which bite is the challenge
        won?
  \item On a number line from $0$ to $1$ graduated in sixteenths,
        place the eaten totals after each of the first four
        bites: $\frac12$, $\frac34$, $\frac78$, $\frac{15}{16}$.
        Where does each new total land, compared with the
        previous one and with $1$?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Zeno's runner, and a knife that
only halves.}
\begin{enumerate}[resume]
  \item Twenty-four centuries ago, the Greek philosopher Zeno
        told this story: to reach a wall, a runner must first
        cover \omterm{def:g3:division:half}{half} the distance, then \omterm{def:g3:division:half}{half} of what remains, then
        \omterm{def:g3:division:half}{half} of what remains again\,\dots\ ``so the runner never
        reaches the wall!'' Using questions 8 and 9, say what is
        \emph{right} in Zeno's story, and where the trap is.
        (Does the runner really need forever to make all those
        steps?)
  \item The sharing puzzle: share $15$ identical chocolate bars
        fairly among $16$ children, using only cuts into halves
        (of a bar, of a half-bar, of a quarter-bar\,\dots).
        Describe how to cut: how many bars are cut into halves,
        how many into \omterm{def:g3:division:half}{quarters}, into eighths, into sixteenths,
        so that every child receives one piece of each size.
  \item Check your sharing: write each child's share as a \omterm{def:g1:addition:def}{sum} of
        \omterm{def:g6:fractions:def}{fractions}, compute it in sixteenths, and verify that the
        $16$ shares together use up exactly the $15$ bars.
  \item In the class next door, $7$ bars are shared fairly among
        $8$ children. Which child receives more chocolate --- one
        of the $16$, or one of the $8$? (Write both shares with
        \omterm{def:g4:fractions:def}{denominator} $16$, \cref{exo:g6:fractions:8}.)
  \item The finale: a friend claims that with \emph{infinitely}
        many bites one would eat \emph{exactly} the whole bar.
        Draw a square and shade the bites inside it: \omterm{def:g3:division:half}{half} the
        square, then a \omterm{def:g3:division:half}{quarter}, then an eighth\,\dots\ What do
        you observe about the unshaded corner? State carefully
        the two true facts of this whole problem: what remains
        after every finite number of bites, and how small that
        \omterm{def:g3:division:remainder}{remainder} becomes. (What an ``infinite \omterm{def:g1:addition:def}{sum}'' exactly
        means is a story for the High School volume.)
\end{enumerate}
\end{problem}
