\chapter{दशमलव संख्याएं}\label{ch:g6:decimals}

बीच में $2$ और $3$ वहाँ बहुत सारी संख्याएँ हैं: $2.5$, $2.71$,
$2.999$\,\dots\ दशमलव संख्याएं स्थान-मूल्य प्रणाली का विस्तार करें
इकाइयों का अधिकार, और वे पूर्ण संख्याओं के समान नियमों का पालन करते हैं ---
यह अध्याय आपको कुछ \omterm{def:g5:solids:net}{जाल}ों से बचना सिखाएगा।

\section{दशमलव लेखन}

\begin{definition}[दशमलव स्थानों]\label{def:g6:decimals:places}
ए \emph{दशमलव संख्या}\index{दशमलव संख्या} एक पूरा हिस्सा है और एक
दशमलव भाग, द्वारा अलग किया गया \omterm{def:g4:decimals:point}{दशमलव बिंदु}. प्रत्येक स्थान दाहिनी ओर
बिंदु का मूल्य पहले वाले से दस गुना कम है: दसवां हिस्सा,
सौवाँ, हज़ारवाँ। उदाहरण के लिए
\[
13.407 = 13 + \frac{4}{10} + \frac{0}{100} + \frac{7}{1000}
= 1 \times 10 + 3 + 4 \times \frac{1}{10} + 7 \times \frac{1}{1000}.
\]
\end{definition}

\begin{remark}[शून्य वो मायने रखते हैं, वो शून्य जो मायने नहीं रखते]
\omterm{def:g1:addition:def}{जोड़}ा जा रहा है \omterm{def:g1:counting:zero}{शून्य} पर \emph{अंत} दशमलव भाग में कुछ भी परिवर्तन नहीं होता:
$2.5 = 2.50 = 2.500$. लेकिन \omterm{def:g1:counting:zero}{शून्य} \emph{बीच में} अंक आवश्यक हैं:
$13.407 \neq 13.47$. और $0.5$ से बहुत अलग है $0.05$!
\end{remark}

\begin{method}[दशमलव संख्याओं की तुलना करना]\label{met:g6:decimals:compare}
\begin{enumerate}
  \item पहले पूरे भागों की तुलना करें: $7.2 > 5.99$ क्योंकि $7 > 5$.
  \item यदि पूरे भाग बराबर हैं, तो दशमलव भागों के अंकों की तुलना करें
        बाएँ से अंक --- अंतिम के साथ पैडिंग \omterm{def:g1:counting:zero}{शून्य} मदद करता है:
        तुलना करना $3.4$ और $3.15$, लिखना $3.40$ और $3.15$;
        तब से $40 > 15$ सौवां, $3.4 > 3.15$.
\end{enumerate}
सावधान: \emph{लंबे का मतलब बड़ा नहीं है}. $3.15$ से अधिक अंक हैं
$3.4$ लेकिन छोटा है.
\end{method}

\section{जोड़ना, घटाना, गुणा करना}

\begin{method}[दशमलव के साथ स्तंभ गणना]\label{met:g6:decimals:columns}
जोड़ना या घटाना दशमलव संख्याएं, संरेखित करें \emph{दशमलव अंक}
(इसलिए इकाइयाँ इकाइयों के नीचे बैठती हैं, दसवें को दसवें के नीचे), अंतिम के साथ पैडिंग
\omterm{def:g1:counting:zero}{शून्य} यदि ज़रूरत हो तो; फिर पूर्ण संख्याओं की तरह गणना करें और बिंदु रखें
दूसरों के अधीन परिणाम का.
\end{method}

\begin{example}\label{ex:g6:decimals:addsub}
गणना करना $13.7 + 2.85$, बिंदुओं को संरेखित करना ($13.7 = 13.70$):
\[
\begin{array}{r}
1\,3.7\,0 \\
+\ \ \,2.8\,5 \\
\hline
1\,6.5\,5
\end{array}
\]
गणना करना $6 - 2.35$ (लिखना $6 = 6.00$):
$6.00 - 2.35 = 3.65$. जाँच करना: $3.65 + 2.35 = 6$.
\end{example}

\begin{example}[दशमलव को गुणा करना]\label{ex:g6:decimals:mult}
गणना करना $2.3 \times 1.4$: पूर्ण संख्याओं से गुणा करें,
$23 \times 14 = 322$; फिर गुणनखंडों के दशमलव अंक गिनें
($1 + 1 = 2$), और बिंदु को इस प्रकार रखें कि परिणाम में इतने सारे हों:
$2.3 \times 1.4 = 3.22$. आकारों पर विवेक की जाँच: $2.3 \times 1.4$
से थोड़ा अधिक होना चाहिए $2.3 \times 1 = 2.3$ --- और $3.22$ है।
\end{example}

\section{10, 100, 1000 से गुणा और भाग करना}

\begin{proposition}[बात बदल रही है]\label{prop:g6:decimals:shift}
ए को गुणा करना \omterm{def:g6:decimals:places}{दशमलव संख्या} द्वारा $10$, $100$, $1000$ इसे चलाता है \omterm{def:g4:decimals:point}{दशमलव
बिंदु} $1$, $2$, $3$ स्थानों के लिए \emph{सही}; विभाजन इसे ले जाता है
\emph{बाएं} (साथ पैडिंग \omterm{def:g1:counting:zero}{शून्य} जब जरूरत है):
\[
3.75 \times 100 = 375, \qquad
42.1 \div 1000 = 0.0421 .
\]
\end{proposition}

\begin{proof}
से गुणा करना $10$ प्रत्येक अंक का मूल्य दस गुना अधिक: दसवां बनाता है
इकाइयाँ बन जाती हैं, इकाइयाँ दहाई बन जाती हैं, और इसी तरह --- प्रत्येक अंक एक चलता है
स्थान-मान तालिका में बाईं ओर कॉलम, जो चलने के समान है
बिंदु एक स्थान दाईं ओर. विभाजन इसका उलटा करता है।
\end{proof}

\begin{example}[माप की इकाइयां]\label{ex:g6:decimals:units}
इकाइयों को परिवर्तित करना वास्तव में यही खेल है: चूँकि $1$ एम $= 100$ सेमी,
\[
3.75 \text{ m} = 3.75 \times 100 \text{ cm} = 375 \text{ cm},
\qquad
42 \text{ mm} = 42 \div 10 \text{ cm} = 4.2 \text{ cm}.
\]
\end{example}

\section{गोलाई}

\begin{definition}[गोलाई]\label{def:g6:decimals:rounding}
 \emph{गोलाई}\index{गोलाई} किसी संख्या की इकाई (या दसवां,
सौवां, \dots) उस परिशुद्धता के साथ निकटतम संख्या है। देखो
अगला अंक: यदि यह है $0,1,2,3,4$, \omterm{def:g4:numbers:round}{गोल} नीचे (रखें); अगर यह है
$5,6,7,8,9$, \omterm{def:g4:numbers:round}{गोल} ऊपर।
\end{definition}

\begin{example}
$7.38$ इकाई तक पूर्णांकित है $7$ (अगला अंक $3$: रखना); तक \omterm{def:g4:numbers:round}{गोल}ाकार
यह दसवां है $7.4$ (अगला अंक $8$: \omterm{def:g4:numbers:round}{गोल} ऊपर)। क़ीमत
$4.996$ सौवें तक पूर्णांकित है $5.00$ --- \omterm{def:g6:decimals:rounding}{गोलाई} बदल सकता है
हर अंक!
\end{example}

\section{समस्या: 3 के ठीक बाद वाली संख्या मौजूद नहीं है}

\begin{problem}[{सप्ताहांत समस्या---किसी भी दो दशमलव के बीच संख्याएँ हमेशा भिन्न होती हैं: संख्या रेखा पर ज़ूम करना}]
\label{pb:g6:decimals:1}
पूर्ण संख्याओं की सीढ़ी पर, प्रत्येक संख्या का एक अगला द्वार होता है
पड़ोसी: ठीक बाद $7$ आता है $8$, और कुछ भी अंदर नहीं रहता है
के बीच. दशमलव संख्याएं एक पूरी तरह से अलग दुनिया हैं. क्या है?
ठीक बाद का नंबर $3$? यह है $3.1$? $3.01$? $3.001$? यह
समस्या विकसित होती है \emph{ज़ूम} संख्या रेखा पर तकनीक
(जारी \cref{exo:g6:decimals:11}) और एक प्रसिद्ध तक पहुंचता है
निष्कर्ष: ठीक बाद की संख्या $3$ \emph{मौजूद नहीं} ---
किन्हीं दो के बीच दशमलव संख्याएं, चाहे कितना भी करीब हो , हमेशा होता है
और अधिक के लिए जगह.

\medskip
\noindent\textbf{Part I --- Zooming in.}
\begin{enumerate}
  \item जो बड़ा है, $2.999$ या $3$? की गणना करें \omterm{ex:g1:subtraction:difference}{अंतर}
        उन दोनों के बीच।
  \item से एक संख्या रेखा खींचिए $7.4$ को $7.5$, में स्नातक किया
        सौवां, और स्थान $7.42$, $7.45$ और $7.48$ उस पर.
        सूची \emph{सभी} दो दशमलव अंकों वाली संख्याएँ
        बीच में सख्ती से झूठ बोलें $7.4$ और $7.5$. कितने हैं?
  \item फिर से ज़ूम करें: बीच में $7.44$ और $7.45$, सभी की सूची बनाएं
        तीन दशमलव अंकों वाली संख्याएँ। कितने हैं?
        इस बार?
  \item उसी विचार का उपयोग करते हुए, संख्याओं को गिनने का तरीका बताएं
        जिसके ठीक बीच में तीन दशमलव अंक हों
        $7.4$ और $7.5$, और वह गिनती दीजिए।
  \item प्रश्न 2-4 के पीछे छुपे नुस्खे का वर्णन करें
        \emph{ज़ूम}): दो संख्याएँ दी गई हैं जो समान दिखती हैं
        पड़ोसी, जैसे $5.67$ और $5.68$, लेखन कैसे होता है
        एक और दशमलव स्थान हमेशा एक संख्या को सख्ती से प्रकट करता है
        उनके बीच? अपना नुस्खा लागू करें $5.67$ और $5.68$,
        फिर को $0.1999$ और $0.2$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The missing neighbour.}
\begin{enumerate}[resume]
  \item टॉम का दावा है: `` ठीक बाद का नंबर $3$ है $3.1$.'' सिद्ध करो
        बीच में एक संख्या का नामकरण करके उसे गलत बताया गया $3$ और
        $3.1$. टॉम पीछे हट गया $3.01$, फिर को $3.001$. प्रत्येक को मारो
        उसके उम्मीदवारों की.
  \item समझाइए क्यों \emph{कोई नहीं} आपके विरुद्ध यह गेम जीत सकते हैं:
        जो भी संख्या पूर्णतया उससे अधिक हो $3$ टॉम का प्रस्ताव है,
        प्रश्न 5 का ज़ूम नुस्खा कड़ाई से एक संख्या उत्पन्न करता है
        के बीच $3$ और उसका प्रस्ताव. इससे क्या सिद्ध होता है
        `` ठीक बाद का नंबर $3$''?
  \item अब दूसरा पक्ष: टॉम सिर्फ नंबर की तलाश में है
        \emph{पहले} $3$ और कोशिश करता है $2.9$, तब $2.99$, तब
        $2.999$. उसके तीन उम्मीदवारों को हराया. फिर गणना करें
        $3 - 2.999$, और वर्णन करें (स्तंभों में गणना किए बिना)
        द \omterm{ex:g1:subtraction:difference}{अंतर} बीच में $3$ और a से लिखा गया नंबर
        $2$, एक बिंदु, और बीस अंक $9$.
  \item इनमें से कोई भी काम क्यों नहीं करता \emph{साबुत} संख्याएँ?
        एक या दो वाक्यों में स्पष्ट करें कि पूर्णांक क्यों नहीं है
        के बीच सख्ती से संख्या $7$ और $8$, भले ही वहाँ
        बहुत सारे हैं दशमलव संख्याएं वहाँ।
  \item एक लंबाई की घोषणा `` के रूप में की जाती है $3.7$ सेमी, की ओर \omterm{def:g4:numbers:round}{गोल}ाकार
        दसवां'' (\cref{def:g6:decimals:rounding}). दे दो
        सबसे छोटी लंबाई वह राउंड को $3.7$ सेमी, और समझाएं क्यों
        की एक लंबाई $3.75$ सेमी करता है \emph{नहीं} \omterm{def:g4:numbers:round}{गोल} को $3.7$ ---
        इसलिए वास्तविक लंबाई कम से कम है $3.65$ सेमी और सख्ती से
        नीचे $3.75$ सेमी।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Games with digits and points.}
\begin{enumerate}[resume]
  \item सबसे छोटे से सबसे बड़े तक ऑर्डर करें: $0.6$; \ $0.58$; \
        $0.123$; \ $0.0999$. फिर किसी सहपाठी को एक में समझाएं
        वाक्य, क्यों $0.0999 < 0.6$ भले ही यह लिखा हो
        अधिक अंकों के साथ (\cref{met:g6:decimals:compare}).
  \item किसी दुकान में कीमतें हमेशा दशमलव के ठीक दो अंकों की होती हैं।
        वहां एक \emph{कीमत} सख्ती से बीच में $4.99$ और $5$?
        प्रश्न 9 से तुलना करें: कीमतें और पूर्णांक क्या कहते हैं
        में समानता है?
  \item प्रत्येक अंक का उपयोग करना $2$, $5$, $8$ बिल्कुल एक बार और
        एक \omterm{def:g4:decimals:point}{दशमलव बिंदु} (पूरा भाग खाली नहीं है, जैसा कि अंदर है
        \cref{exo:g6:decimals:12}), निकटतम संख्या ज्ञात कीजिए
        $6$. अपने सर्वोत्तम की दूरी की गणना करके औचित्य सिद्ध करें
        उम्मीदवारों को $6$.
  \item गणित का सबसे प्रसिद्ध अंक, $\pi $, संतुष्ट करता है
        $3.141 < \pi < 3.142$. नाम ए \omterm{def:g6:decimals:places}{दशमलव संख्या} सख्ती से
        के बीच $3.141$ और $3.142$; फिर एक के बीच सख्ती से
        $3.1415$ और $3.1416$. (गणितज्ञ निचोड़ते हैं $\pi $
        बिल्कुल इसी तरह --- प्रत्येक नया दशमलव अंक एक और होता है
        ज़ूम करें. तुम मिलोगे $\pi $ में काम पर
        \cref{ch:g6:measure}.)
  \item समापन: समझाएँ कि वहाँ क्यों हैं \emph{अधिक दशमलव संख्याएं बीच में $0$ और $1$ किसी भी संख्या से जो आप कर सकते हैं
        नाम}. (एक ज़ूम कितने उत्पन्न करता है? क्या ज़ूम कर सकता है
        कभी रुकें?)
\end{enumerate}
\end{problem}
