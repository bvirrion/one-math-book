\chapter{त्रिभुज और चतुर्भुज}\label{ch:g6:shapes}

\omterm{def:g3:shapes:shapes}{त्रिकोण} और चतुर्भुज ज्यामिति की मूल आकृतियाँ हैं। लक्ष्य
इस अध्याय का उद्देश्य उनके वि\omterm{def:g3:division:remainder}{शेष} परिवारों को जानना है --- \omterm{def:g6:shapes:triangles}{समद्विबाहु},
\omterm{def:g6:shapes:triangles}{समभुज}, \omterm{def:g3:shapes:rightangle}{समकोण} त्रिभुज; आयत, समचतुर्भुज, वर्ग --- द्वारा
उनके \emph{गुणों को परिभाषित करना}, और उनके साथ सटीक रूप से निर्माण करना
शासक और कम्पास.

\section{त्रिकोण}

\begin{definition}[विशेष त्रिकोण]\label{def:g6:shapes:triangles}
एक \omterm{def:g3:shapes:shapes}{त्रिकोण} $ABC $ तीन हैं \omterm{def:g5:solids:def}{कोने} और तीन पक्ष. यह है:
\begin{itemize}
  \item \emph{समद्विबाहु}\index{समद्विबाहु} पर $ A $ कब
        $ AB = AC $ (दो बराबर भुजाएँ);
  \item \emph{समभुज}\index{समभुज} कब
        $ AB = BC = CA $ (तीन बराबर भुजाएँ);
  \item \emph{समकोण का}\index{समकोण का} पर $ A $ जब कोण
        $\widehat{BAC}$ एक है \omterm{def:g3:shapes:rightangle}{समकोण}. ओर $[BC]$ के विपरीत
        \omterm{def:g3:shapes:rightangle}{समकोण} कहा जाता है \emph{कर्ण}.
\end{itemize}
\end{definition}

\begin{method}[इसकी तीन भुजाओं से एक त्रिभुज की रचना करना]\label{met:g6:shapes:construct}
निर्मित करना $ ABC $ साथ $ AB = 5$ सेमी, $ AC = 4$ सेमी, $ BC = 3$ सेमी:
\begin{enumerate}
  \item खींचना \omterm{def:g6:lines:objects}{खंड} $[AB]$ का $5$ शासक के साथ सेमी;
  \item केंद्र के वृत्त का एक चाप बनाएं $ A $ और \omterm{def:g3:shapes:shapes}{RADIUS} $4$ सेमी
        (सभी बिंदुओं पर $4$ सेमी से $ A $);
  \item केंद्र के वृत्त का एक चाप बनाएं $ B $ और \omterm{def:g3:shapes:shapes}{RADIUS} $3$ सेमी;
  \item मुद्दा $ C $ वह स्थान है जहां दो चाप प्रतिच्छेद करते हैं; इससे जुड़ें $ A $ और
        $ B $.
\end{enumerate}
\end{method}

\section{चतुर्भुज}

\begin{definition}[विशेष चतुर्भुज]\label{def:g6:shapes:quadrilaterals}
ए \omterm{def:g4:geometry:polygon}{चतुष्कोष} चार हैं \omterm{def:g5:solids:def}{कोने} और चार भुजाएँ. यह एक है:
\begin{itemize}
  \item \emph{आयत}\index{आयत} जब इसके चार कोण हों
        सही कोण;
  \item \emph{विषमकोण}\index{विषमकोण} जब कि उसकी चारों भुजाएं एक समान हों
        लंबाई;
  \item \emph{वर्ग}\index{वर्ग} जब यह दोनों हो: चार सही कोण
        \emph{और} चार बराबर भुजाएँ.
\end{itemize}
\end{definition}

\begin{proposition}[विकर्णों के गुण]\label{prop:g6:shapes:diagonals}
इन मे चतुर्भुज, दोनों विकर्ण एक दूसरे को अपने-अपने स्थान पर काटते हैं
सामान्य मध्यबिंदु, और इसके अलावा:
\begin{itemize}
  \item एक आयत में, विकर्णों का होता है \emph{एक ही लंबाई};
  \item में एक \omterm{def:g6:shapes:quadrilaterals}{विषमकोण}, विकर्ण हैं \emph{\omterm{def:g6:lines:perp}{सीधा}};
  \item एक चौक में, दोनों एक साथ।
\end{itemize}
\end{proposition}
\admitted

\begin{remark}
एक वर्ग स्वचालित रूप से एक आयत होता है (इसमें चार होते हैं)। सही कोण) और
स्वचालित रूप से ए \omterm{def:g6:shapes:quadrilaterals}{विषमकोण} (चार बराबर भुजाएँ)। तो ``आयत'' नहीं है
मतलब ``वर्ग नहीं''! गणित में आकार अधिक वि\omterm{def:g3:division:remainder}{शेष} होता है
अधिक सामान्य परिवार के लिए.
\end{remark}

\begin{example}[किसी आकृति को उसके गुणों से पहचानना]\label{ex:g6:shapes:recognize}
ए \omterm{def:g4:geometry:polygon}{चतुष्कोष} लंबाई की चार भुजाएँ हैं $4$ सेमी और एक \omterm{def:g3:shapes:rightangle}{समकोण}.
यह क्या है? चार बराबर भुजाएँ इसे बनाती हैं a \omterm{def:g6:shapes:quadrilaterals}{विषमकोण}. में एक \omterm{def:g6:shapes:quadrilaterals}{विषमकोण}, विपरीत
कोण बराबर होते हैं और चारों कोणों का योग होता है $360^\circ $; एक \omterm{def:g3:shapes:rightangle}{समकोण} फिर चारों को सही होने के लिए मजबूर करता है (यह साबित हो जाएगा)।
में समांतर चतुर्भुज \cref{ch:g7:central}). तो आकृति एक वर्ग है.
\end{example}

\begin{method}[निर्माण कार्यक्रम]\label{met:g6:shapes:program}
ए \emph{निर्माण कार्यक्रम} निर्देशों की एक सूची है जो कोई भी कर सकता है
किसी आकृति को हूबहू पुन: प्रस्तुत करने के लिए अनुसरण करें। प्रति पंक्ति एक निर्देश लिखें,
प्रत्येक बिंदु को बनाते समय उसे नाम दें, और केवल उपयोग करें: ``आकर्षित करें \omterm{def:g6:lines:objects}{खंड}
/ \omterm{def:g6:lines:objects}{रेखा} / वृत्त\dots'', ``बिंदु को वहां रखें \dots\ पार करना''। परीक्षण
अपने कार्यक्रम का शब्दशः पालन करके।
\end{method}

\begin{example}\label{ex:g6:shapes:program}
एक वर्ग के लिए कार्यक्रम $ ABCD $ इस पक्ष की $4$ सेमी:
\begin{enumerate}
  \item चित्रित करो \omterm{def:g6:lines:objects}{खंड} $[AB]$ साथ $ AB = 4$ सेमी।
  \item खींचना \omterm{def:g6:lines:perp}{सीधा} को $(AB)$ पर $ A $; उस पर, जगह $ D $ पर
        $4$ सेमी से $ A $ (\omterm{def:g6:lines:objects}{रेखा} का एक पक्ष चुनें और उस पर टिके रहें)।
  \item खींचना \omterm{def:g6:lines:perp}{सीधा} को $(AB)$ पर $ B $; उस पर, जगह $ C $ पर
        $4$ सेमी से $ B $, उसी तरफ $ D $.
  \item जोड़ना $ C $ और $ D $.
\end{enumerate}
रूलर से जांचें: $ DC $ मापना चाहिए $4$ सेमी भी.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g6:shapes:1}
एक त्रिभुज का निर्माण करें $ ABC $ साथ $ AB = 6$ सेमी, $ AC = 5$ सेमी और
$ BC = 4$ सेमी, निम्नलिखित \cref{met:g6:shapes:construct}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:shapes:2}
एक का निर्माण करें \omterm{def:g6:shapes:triangles}{समद्विबाहु} त्रिभुज $ DEF $ साथ $ DE = DF = 5$ सेमी और
$ EF = 3$ सेमी। इस त्रिभुज के कौन से दो कोण बराबर दिखते हैं? उन्हें मापें
जाँच करना.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:shapes:3}
एक का निर्माण करें समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण} इस पक्ष की $4$ केवल रूलर का उपयोग करके सेमी
दिशा सूचक यंत्र। इसके एक कोण को मापें.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:shapes:4}
प्रत्येक त्रिभुज को उसके डेटा से वर्गीकृत करें (\omterm{def:g6:shapes:triangles}{समद्विबाहु}, \omterm{def:g6:shapes:triangles}{समभुज},
\omterm{def:g6:shapes:triangles}{समकोण का} --- कई उत्तर लागू हो सकते हैं):
(ए) पक्ष $7$, $7$, $7$; \
(बी) पक्ष $5$, $5$, $8$; \
(सी) कोण $90^\circ $, $45^\circ $, $45^\circ $; \
(डी) कोण $60^\circ $, $60^\circ $, $60^\circ $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:shapes:5}
में एक सही \omterm{def:g3:shapes:shapes}{त्रिकोण}, कर्ण किस तरफ है? एक \omterm{def:g3:shapes:shapes}{त्रिकोण} बनाएं
$ MNP $ \omterm{def:g6:shapes:triangles}{समकोण का} पर $ N $, और इसके कर्ण का नाम बताएं। कौन शिखर करता है
कर्ण कभी स्पर्श नहीं करता?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:shapes:6}
सच या झूठ? प्रत्येक उत्तर को एक वाक्य में उचित ठहराएँ।
\begin{enumerate}
  \item प्रत्येक वर्ग एक आयत है.
  \item प्रत्येक आयत एक वर्ग है.
  \item प्रत्येक वर्ग एक है \omterm{def:g6:shapes:quadrilaterals}{विषमकोण}.
  \item ए \omterm{def:g6:shapes:quadrilaterals}{विषमकोण} के साथ \omterm{def:g3:shapes:rightangle}{समकोण} एक वर्ग है.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:shapes:7}
एक आयत का निर्माण करें $ ABCD $ साथ $ AB = 5$ सेमी और $ BC = 3$ सेमी। उपाय
इसके दो विकर्ण: आप क्या देखते हैं?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:shapes:8}
ए का निर्माण करें \omterm{def:g6:shapes:quadrilaterals}{विषमकोण} के विकर्णों के साथ $6$ सेमी और $4$ सेमी। (उपयोग
\cref{prop:g6:shapes:diagonals}: पहले दो विकर्ण बनाएं ---
\omterm{def:g6:lines:perp}{सीधा}, उनके मध्य बिंदुओं को पार करते हुए --- फिर चारों से जुड़ें
समाप्त होता है।)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:shapes:9}
के लिए एक निर्माण कार्यक्रम लिखें \omterm{def:g6:shapes:triangles}{समद्विबाहु} त्रिभुज आधार के साथ
$4$ सेमी और बराबर भुजाएँ $6$ सेमी, की शैली में
\cref{ex:g6:shapes:program}. सहपाठी के साथ आदान-प्रदान: क्या वे अनुसरण कर सकते हैं
यह आपका फिगर देखे बिना?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:shapes:10}
केंद्र का एक वृत्त बनाएं $ O $ और इसका एक \omterm{def:g4:geometry:circle}{व्यास} $[AB]$. कोई भी रखें
बिंदु $ M $ वृत्त पर त्रिभुज बनाएं $ AMB $. कोण मापें
$\widehat{AMB}$. कदम $ M $ और फिर से मापें. आप क्या अनुमान लगाते हैं?
(यह आश्चर्यजनक तथ्य सिद्ध होता है \cref{ch:g8:cosine}.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:shapes:11}
ए \omterm{def:g4:geometry:polygon}{चतुष्कोष} समान लंबाई के विकर्ण हैं जो एक दूसरे को काटते हैं
उनका सामान्य मध्यबिंदु. इसका उपयोग करके क्रमबद्ध करें कि यह किस आकार का होना चाहिए
\cref{prop:g6:shapes:diagonals}: आयत, \omterm{def:g6:shapes:quadrilaterals}{विषमकोण} या चौकोर?
(सावधान: केवल एक ही संपत्ति दी गई है।)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g6:shapes:12}
एक पर $4 \times 4$ बिंदीदार ग्रिड, आप कितने अलग-अलग वर्ग बना सकते हैं
किसके \omterm{def:g5:solids:def}{कोने} ग्रिड बिंदु हैं --- ``झुके हुए'' वर्गों की भी गिनती?
कम से कम एक झुका हुआ वर्ग बनाएं और बताएं कि उसकी भुजाएं बराबर क्यों हैं।
\end{exercise}

\section{समस्या: विकर्ण क्या जानते हैं}

\begin{problem}[{सप्ताहांत समस्या---प्रत्येक विशेष का पुनर्निर्माण चतुर्भुज को केवल उसके विकर्णों से, और विकर्णों को गिनकर किसी भी बहुभुज का}]
\label{pb:g6:shapes:1}
\Cref{prop:g6:shapes:diagonals} एक आकृति पढ़ता है और आपको उसके बारे में बताता है
इसके विकर्ण. यह समस्या खेल खेलती है \emph{पीछे की ओर}: दिया गया
केवल दो क्रॉसिंग \omterm{def:g6:lines:objects}{खंड}ों --- भविष्य के विकर्ण --- किस आकार के
जब आप उनके चार सिरों को जोड़ते हैं तो प्रकट होता है? आप पूरा निर्माण करेंगे
``विकर्ण शब्दकोश'', समझे \emph{क्यों}  \omterm{def:g6:shapes:quadrilaterals}{विषमकोण}
निर्माण का \cref{exo:g6:shapes:8} काम करता है, एक आकृति से मिलें
अभी तक कोई नाम नहीं है, और विकर्णों की गिनती करके समाप्त करें \omterm{def:g4:geometry:polygon}{बहुभुज}
चार से अधिक भुजाओं वाला।

\medskip
\noindent\textbf{Part I --- The diagonal dictionary.}
प्रत्येक प्रश्न के लिए, दो रेखाएँ बनाएँ \omterm{def:g6:lines:objects}{खंड}ों जैसा कि बताया गया है, उनसे जुड़ें
चार सिरों को क्रम में रखें और पहचानें \omterm{def:g4:geometry:polygon}{चतुष्कोष} आपको मिला।
\begin{enumerate}
  \item दो \omterm{def:g6:lines:objects}{खंड}ों का $6$ सेमी प्रत्येक, \omterm{def:g6:lines:perp}{सीधा}, पर पार करना
        उनका सामान्य मध्यबिंदु. चारों भुजाओं को मापें और
        परिणामी आकृति के चार कोण: यह क्या है?
  \item दो \omterm{def:g6:lines:objects}{खंड}ों का $8$ सेमी और $5$ सेमी, \omterm{def:g6:lines:perp}{सीधा},
        अपने सामान्य मध्यबिंदु पर पार करना। कौन सी आकृति दिखाई देती है
        (\cref{exo:g6:shapes:8})?
  \item दो \omterm{def:g6:lines:objects}{खंड}ों का $7$ सेमी प्रत्येक --- बराबर लेकिन \emph{नहीं}
        \omterm{def:g6:lines:perp}{सीधा} --- अपने सामान्य मध्यबिंदु पर पार करना।
        भुजाओं और कोणों को मापें: आप क्या अनुमान लगाते हैं
        आकार होना चाहिए?
  \item दो \omterm{def:g6:lines:objects}{खंड}ों का $8$ सेमी और $5$ सेमी, नहीं \omterm{def:g6:lines:perp}{सीधा},
        अपने सामान्य मध्यबिंदु पर पार करना। आपको जो आकार मिलता है वह है
        तीनों में से कोई नहीं
        \cref{def:g6:shapes:quadrilaterals} --- वर्णन करें क्या
        आप देखते हैं (कौन सी भुजाएँ समान दिखती हैं? \omterm{def:g6:lines:perp}{समानांतर}?). यह
        \omterm{def:g4:geometry:polygon}{चतुष्कोष} में अपना नाम प्राप्त करता है \cref{ch:g7:central}.
  \item प्रश्न 1-4 को शब्दकोश में चार के साथ सारांशित करें
        प्रविष्टियाँ: विकर्ण \emph{बराबर / \omterm{def:g6:lines:perp}{सीधा} / न तो दोनों} (हमेशा अपने मध्यबिंदुओं पर पार करें)
        $\rightarrow $ आकृति का नाम (या विवरण)।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Why the constructions work.}
\begin{enumerate}[resume]
  \item प्रश्न 2 में, दो विकर्ण आकृति को चार भागों में काटते हैं
        \omterm{def:g3:shapes:rightangle}{समकोण} त्रिभुज क्रॉसिंग पॉइंट के आसपास. क्या हैं
        इनमें से प्रत्येक त्रिभुज की दो टांगों की लंबाई? स्पष्ट करें क्यों
        सभी चार त्रिभुज एक दूसरे की सटीक प्रतिलिपियाँ हैं, और
        निष्कर्ष निकालें कि आकृति की चारों भुजाएँ बराबर क्यों हैं ---
        का रहस्य \cref{exo:g6:shapes:8}.
  \item केवल-कम्पास परीक्षण निकालें: किसी भी आकार के निर्माण के बाद
        इसके विकर्णों से, आप कंपास से कैसे जांच कर सकते हैं
        अकेले (कोई शासक माप नहीं) कि चार भुजाएँ हैं
        बराबर?
  \item एक निर्माण कार्यक्रम लिखें
        (\cref{met:g6:shapes:program}) एक वर्ग के लिए जिसका
        \emph{विकर्णों} उपाय $6$ सेमी --- इसकी भुजाएँ नहीं।
  \item प्रश्न 8 के वर्ग में भुजा लम्बी है अथवा
        से छोटा $3$ सेमी --- \omterm{def:g3:division:half}{आधा} विकर्ण? उपाय करें
        पता करो. एक \omterm{def:g3:division:half}{आधा} उत्तर का पहले से ही हो सकता है
        \emph{साबित}: के माध्यम से एक \omterm{def:g5:solids:def}{कोने} से दूसरे \omterm{def:g5:solids:def}{कोने} तक जाना
        केंद्र एक है $3 + 3$ सेमी चक्कर, और एक \omterm{def:g6:lines:perp}{सीधा} रास्ता है
        हमेशा एक चक्कर से छोटा --- इसलिए पक्ष छोटा होता है
        से $6$ सेमी। (यह साबित करते हुए कि यह धड़कता भी है $3$ सेमी, और
        इसकी सटीक गणना करना एक काम है
        \cref{ch:g8:pythagoras}.)
  \item शब्दकोश के लिए एक नई प्रविष्टि: ड्रा ए \omterm{def:g6:lines:objects}{खंड} $[AC]$ का
        $8$ सेमी; उस पर बिंदु अंकित करें $ H $ पर $3$ सेमी से $ A $;
        के माध्यम से खींचो $ H $  \omterm{def:g6:lines:perp}{सीधा} \omterm{def:g6:lines:objects}{खंड} $[BD]$ का
        $5$ सेमी के साथ $ H $ जैसा \emph{इसका} मध्यबिंदु. जोड़ना $ ABCD $. यह
        आकार एक है \emph{पतंग}\index{पतंग} (एक विकर्ण समद्विभाजित करता है
        अन्य, एक पर \omterm{def:g3:shapes:rightangle}{समकोण}). \omterm{def:g5:solids:def}{कोने} वाले त्रिभुजों का उपयोग करना
        प्रश्न 6 में, खोजें कि a की कौन-सी भुजाएँ हैं \omterm{pb:g6:shapes:1}{पतंग} के बराबर हैं
        जो.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Counting diagonals.}
ए \emph{विकर्ण} एक का \omterm{def:g4:geometry:polygon}{बहुभुज} दो को जोड़ता है \omterm{def:g5:solids:def}{कोने} वह नहीं हैं
पड़ोसी.
\begin{enumerate}[resume]
  \item चित्रित करो \omterm{def:g4:geometry:polygon}{चतुष्कोष}, ए \omterm{def:g4:geometry:polygon}{पंचकोण} (पाँच \omterm{def:g5:solids:def}{कोने}) और ए
        \omterm{def:g4:geometry:polygon}{षट्भुज} (छह \omterm{def:g5:solids:def}{कोने}), और उनके सभी विकर्ण बनाएं।
        उन्हें गिनें: प्रत्येक मामले में कितने?
  \item ड्राइंग के बिना गिनती, के लिए \omterm{def:g4:geometry:polygon}{षट्भुज}: एक से
        शिखर, कितने विकर्ण छोड़ते हैं ? (कितने को
        छह \omterm{def:g5:solids:def}{कोने} क्या यह कर सकता है \emph{नहीं} एक विकर्ण से जुड़ें?)
        छह से गुणा करना \omterm{def:g5:solids:def}{कोने} प्रत्येक विकर्ण को गिनता है
        बिल्कुल दो बार---क्यों? अपने प्रश्नों की संख्या पुनः प्राप्त करें
        11. (हाथ मिलाने में भी वही दोहरी गिनती दिखाई दी
        का \cref{pb:g6:wholes:1}.)
  \item के विकर्ण गिनने के लिए प्रश्न 12 की विधि का प्रयोग करें
        ए \omterm{def:g4:geometry:polygon}{बहुभुज} साथ $10$ \omterm{def:g5:solids:def}{कोने}, फिर साथ $20$ \omterm{def:g5:solids:def}{कोने}.
  \item ए \omterm{def:g4:geometry:polygon}{बहुभुज} बिलकुल है $44$ विकर्ण. कितने \omterm{def:g5:solids:def}{कोने}
        क्या इसमें है? (कुछ पर प्रश्न 13 की विधि आज़माएँ
        उम्मीदवार।)
  \item अपनी विधि को छोटे से छोटे पर जाँचें \omterm{def:g4:geometry:polygon}{बहुभुज} सबमें से: कैसे
        एक त्रिभुज में कितने विकर्ण होते हैं, और क्या होता है?
        प्रश्न 12 उत्तर की गिनती विधि? एक के साथ समापन करें
        हैंडशेक और विकर्णों में क्या शामिल है, इस पर वाक्य
        सामान्य.
\end{enumerate}
\end{problem}
