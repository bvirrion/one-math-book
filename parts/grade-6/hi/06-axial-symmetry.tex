\chapter{अक्षीय समरूपता}\label{ch:g6:symmetry}

कागज की एक शीट को एक \omterm{def:g6:lines:objects}{रेखा} के अनुदिश मोड़ें: प्रत्येक बिंदु दूसरे बिंदु पर आ जाता है,
इसकी दर्पण छवि. ये फोल्डिंग है \emph{अक्षीय समरूपता} (या
प्रतिबिंब), पुस्तक का पहला ज्यामितीय परिवर्तन ---
दूसरा, केंद्रीय समरूपता, आती है \cref{ch:g7:central}.

\section{कुछ विचार}

\begin{definition}[सममित बिंदु]\label{def:g6:symmetry:def}
होने देना $d $ एक पंक्ति हो. \emph{सममित}\index{सममित} एक का
बिंदु $ M $ आर-पार $ d $ मुद्दा यह है $ M'$ ऐसा है कि $ d $ है
\omterm{def:g6:symmetry:bisector}{लंब समद्विभाजक} का $[MM']$: द \omterm{def:g6:lines:objects}{खंड} $[MM']$ है \omterm{def:g6:lines:perp}{सीधा}
को $ d $, और $ d $ इसे इसके मध्यबिंदु पर काटता है। एक बिंदु \emph{पर} $ d $ इसका है
स्वयं सममित.
\end{definition}

\begin{method}[एक सममित बिंदु का निर्माण]\label{met:g6:symmetry:construct}
का निर्माण करने के लिए \omterm{def:g6:symmetry:def}{सममित} $ M'$ का $ M $ पूरी पंक्ति पर $ d $:
\begin{enumerate}
  \item खींचना \omterm{def:g6:lines:perp}{सीधा} को $ d $ के माध्यम से $ M $ (गुनिया); पुकारना
        $ H $ वह बिंदु जहां यह पार होता है $ d $;
  \item उपाय $ MH $ कम्पास या शासक के साथ;
  \item जगह $ M'$ उसी पर \omterm{def:g6:lines:perp}{सीधा}, के दूसरी तरफ $ d $,
        साथ $ HM' = MH $.
\end{enumerate}
ग्रिड पेपर पर यह तेज़ है: वर्गों को गिनें $ M $ अक्ष पर, और
दूसरी तरफ भी वही संख्या गिनें।
\end{method}

\begin{example}[एक ग्रिड पर]\label{ex:g6:symmetry:grid}
यदि अक्ष एक ऊर्ध्वाधर ग्रिड \omterm{def:g6:lines:objects}{रेखा} है और $ M $ है $3$ इसके बाईं ओर वर्ग,
फिर $ M'$ है $3$ इसके दाहिनी ओर, एक ही पंक्ति में वर्ग। एक विकर्ण के लिए
अक्ष, गिनती हमेशा विकर्णों के साथ काम करती है \omterm{def:g6:lines:perp}{सीधा} को
अक्ष.
\end{example}

\begin{proposition}[एक प्रतिबिंब क्या सुरक्षित रखता है]\label{prop:g6:symmetry:preserved}
 \omterm{def:g6:symmetry:def}{सममित} एक \omterm{def:g6:lines:objects}{रेखा} के पार एक आकृति का एक आकृति है \emph{वही
आकार और समान आकार}: लंबाई, कोण, \omterm{def:g4:measure:perimeter}{परिधि} और \omterm{def:g4:measure:area}{क्षेत्र}ों सभी हैं
संरक्षित. \omterm{def:g6:symmetry:def}{सममित} एक का \omterm{def:g6:lines:objects}{खंड} एक है \omterm{def:g6:lines:objects}{खंड}, एक वृत्त का ए
उसी का \omterm{def:g6:lines:circle}{घेरा} \omterm{def:g3:shapes:shapes}{RADIUS}, एक लाइन की एक लाइन.
\end{proposition}
\admitted

\begin{remark}
एक बात है \emph{नहीं} संरक्षित: अभिविन्यास. ए प्रतिबिंबित ``बी''
एक ``डी'' बन जाता है --- दर्पण लेखन। यदि आपका फिगर ट्रेस किया जा रहा है और फ़्लिप किया जा रहा है
ट्रेसिंग पेपर छवि को पुन: प्रस्तुत करता है, प्रतिबिंब सही है।
\end{remark}

\section{किसी आकृति की समरूपता के अक्ष}

\begin{definition}[समरूपता की धुरी]\label{def:g6:symmetry:axis}
एक पंक्ति $ d $ एक \emph{समरूपता की धुरी}\index{समरूपता की धुरी} एक का
चित्रा जब प्रतिबिंब भर में $ d $ चित्र को बिल्कुल सही भेजता है
स्वयं ---साथ मोड़ना $ d $ दोनों हिस्सों को मेल कराता है।
\end{definition}

\begin{example}\label{ex:g6:symmetry:shapes}
समरूपता के अक्षों की गिनती:
\begin{itemize}
  \item \omterm{def:g6:shapes:triangles}{समद्विबाहु} त्रिभुज: $1$ (शीर्ष के माध्यम से शिखर और मध्य
        आधार का); समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण}: $3$;
  \item आयत: $2$ (दो ``मध्य रेखाएँ'', \emph{नहीं} द
        विकर्ण); \omterm{def:g6:shapes:quadrilaterals}{विषमकोण}: $2$ (इसके विकर्ण!); वर्ग: $4$;
  \item वृत्त: केंद्र से होकर जाने वाली प्रत्येक रेखा---अनंत रूप से अनेक।
\end{itemize}
\end{example}

\begin{example}[समरूपता समानता सिद्ध करती है]\label{ex:g6:symmetry:proves}
एक के दो आधार कोण क्यों हैं? \omterm{def:g6:shapes:triangles}{समद्विबाहु} त्रिभुज बराबर? मोड़ो
इसके अनुदिश त्रिभुज \omterm{def:g6:symmetry:axis}{समरूपता की धुरी}: छोड़ा \omterm{def:g3:division:half}{आधा} बिलकुल पर उतरता है
ठीक है \omterm{def:g3:division:half}{आधा}, इसलिए बायां आधार कोण दाएं आधार कोण पर पड़ता है ---
उनका माप समान होना चाहिए
(\cref{prop:g6:symmetry:preserved}: प्रतिबिंब कोणों को संरक्षित करते हैं)।
\end{example}

\section{लम्बवत् समद्विभाजक}

\begin{definition}[लम्बवत् समद्विभाजक]\label{def:g6:symmetry:bisector}
 \emph{लंब समद्विभाजक}\index{लंब समद्विभाजक} एक का
\omterm{def:g6:lines:objects}{खंड} $[AB]$ लाइन है \omterm{def:g6:lines:perp}{सीधा} को $[AB]$ इसके माध्यम से
मध्यबिंदु --- यह है \omterm{def:g6:symmetry:axis}{समरूपता की धुरी} वह अदला-बदली करता है $ A $ और $ B $.
\end{definition}

\begin{proposition}[समान दूरी]\label{prop:g6:symmetry:bisector}
एक बिंदु $ M $ के ऊपर है \omterm{def:g6:symmetry:bisector}{लंब समद्विभाजक} का $[AB]$ बिलकुल जब यह
से समान दूरी पर है $ A $ और से $ B $: $ MA = MB $.
\end{proposition}
\admitted

\begin{method}[कम्पास निर्माण]\label{met:g6:symmetry:bisector}
खींचने के लिए \omterm{def:g6:symmetry:bisector}{लंब समद्विभाजक} का $[AB]$ बिना मापे:
\begin{enumerate}
  \item कम्पास को अधिक से अधिक खोलें \omterm{def:g3:division:half}{आधा} का $ AB $;
  \item केन्द्रित दो चाप बनाएं $ A $, ऊपर और नीचे \omterm{def:g6:lines:objects}{खंड};
  \item साथ \emph{वही} खोलते हुए, केन्द्रित दो चाप बनाएं $ B $;
  \item दो क्रॉसिंग बिंदुओं को \omterm{def:g1:addition:def}{जोड़}ें: यह \omterm{def:g6:lines:objects}{रेखा} है लंबवत
        द्वि\omterm{def:g5:division:multiple}{भाजक} (प्रत्येक क्रॉसिंग बिंदु से समान दूरी पर है $ A $ और $ B $).
\end{enumerate}
\end{method}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g6:symmetry:1}
ग्रिड पेपर पर कॉपी करें: एक ऊर्ध्वाधर अक्ष $ d $, एक बिंदु $ A $ इसके दो वर्ग
बाएँ, एक बिंदु $ B $ इसके बायीं ओर पाँच वर्ग और एक वर्ग ऊँचा।
उनका निर्माण करें \omterm{def:g6:symmetry:def}{सममित} अंक $ A'$ और $ B'$ आर-पार $ d $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:symmetry:2}
एक \omterm{def:g6:lines:objects}{रेखा} खींचो $ d $ और एक बिंदु $ M $ पर $3$ सेमी से $ d $ (ग्रिड पर नहीं
कागज). का निर्माण करें \omterm{def:g6:symmetry:def}{सममित} $ M'$ का $ M $ आर-पार $ d $ सेट स्क्वायर के साथ
और शासक, निम्नलिखित \cref{met:g6:symmetry:construct}. कितनी दूरी है
हैं $ M $ और $ M'$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:symmetry:3}
वर्णमाला के कौन से बड़े अक्षर, उनके सरलतम रूप में लिखे गए हैं,
एक ऊर्ध्वाधर है \omterm{def:g6:symmetry:axis}{समरूपता की धुरी} (एक तरह से)? एक क्षैतिज (बी की तरह)?
दोनों?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:symmetry:4}
\omterm{def:g6:symmetry:def}{सममित}ि के कितने अक्ष होते हैं: a समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण}? एक आयत
(वह एक वर्ग नहीं है)? ए \omterm{def:g6:shapes:quadrilaterals}{विषमकोण} (वर्ग नहीं)? एक वर्ग? एक चक्र?
प्रत्येक आकृति को उसके अक्षों से खींचिए।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:symmetry:5}
सच या झूठ? ``एक आयत के विकर्ण \omterm{def:g6:symmetry:def}{सममित}ि के अक्ष होते हैं
आयत।'' एक मोड़ने वाले तर्क या चित्र के साथ समझाएं।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:symmetry:6}
एक \omterm{def:g3:shapes:shapes}{त्रिकोण} $ ABC $ एक \omterm{def:g6:lines:objects}{रेखा} पर प्रतिबिंबित होता है $ d $ में $ A'B'C'$. दिया गया
$ AB = 4$ सेमी, $\widehat{BAC} = 70^\circ $ और यह \omterm{def:g4:measure:perimeter}{परिधि} का $ ABC $ है
$12$ सेमी: दे दो, बिना किसी निर्माण के, $ A'B'$,
$\widehat{B'A'C'}$ और यह \omterm{def:g4:measure:perimeter}{परिधि} का $ A'B'C'$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:symmetry:7}
चित्रित करो \omterm{def:g6:lines:objects}{खंड} $[AB]$ का $7$ सेमी और इसका निर्माण करें लंबवत
द्वि\omterm{def:g5:division:multiple}{भाजक} कम्पास के साथ (\cref{met:g6:symmetry:bisector}). कोई भी चुनें
बिंदु $ M $ उस पर और रूलर से जाँच करें $ MA = MB $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:symmetry:8}
दो पेड़ बिंदु पर खड़े हैं $ A $ और $ B $ एक (सपाट) बगीचे का। कोई कहाँ कर सकता है
एक फव्वारा लगाएं ताकि वह दोनों पेड़ों से समान दूरी पर हो?
वर्णन करना \emph{सभी} संभावित स्थान.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:symmetry:9}
ग्रिड पेपर पर अक्ष बनाएं $ d $ (ग्रिड का एक विकर्ण
$45^\circ $) और एक तरफ एक छोटी एल-आकार की आकृति। इसका निर्माण करें
\omterm{def:g6:symmetry:def}{सममित} पार समझना $ d $. (प्रत्येक को प्रतिबिंबित करें शिखर: एक के लिए $45^\circ $
अक्ष, एक बिंदु $2$ अक्ष के दाईं ओर का वर्ग भूमि है $2$ ऊपर वर्ग
यह.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:symmetry:10}
केंद्र का एक वृत्त बनाएं $ O $, एक पंक्ति $ d $ के माध्यम से $ O $, और एक बिंदु $ M $ पर
\omterm{def:g6:lines:circle}{घेरा}. जहां है \omterm{def:g6:symmetry:def}{सममित} का $ M $ आर-पार $ d $? स्पष्ट करें कि क्यों
प्रतिबिंब वृत्त को स्वयं पर भेजता है।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:symmetry:11}
समान दूरी संपत्ति का उपयोग करना (\cref{prop:g6:symmetry:bisector}),
स्पष्ट करें कि चाप के दो प्रतिच्छेदन बिंदु अंदर क्यों हैं
\cref{met:g6:symmetry:bisector} वास्तव में पर झूठ बोलते हैं लंबवत
द्वि\omterm{def:g5:division:multiple}{भाजक} का $[AB]$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g6:symmetry:12}
बिंदु पर एक बिलियर्ड गेंद $ A $ सीधी दीवार से उछलना चाहिए $ d $ और
पहुंच बिंदु $ B $ (दोनों दीवार के एक ही तरफ)। प्रतिबिंबित होना $ B $
आर-पार $ d $ में $ B'$, और ड्रा करें \omterm{def:g6:lines:objects}{खंड} $[AB']$. स्पष्ट करें कि क्यों
सबसे अच्छा बाउंसिंग पॉइंट वह है जहां $[AB']$ दीवार पार कर जाती है. (विचार: पथ
$ A \to P \to B $ की लंबाई भी उतनी ही है $ A \to P \to B'$.)
\end{exercise}

\section{समस्या: फव्वारा, टूटी प्लेट, और चलना
दर्पण}

\begin{problem}[{सप्ताहांत समस्या---तीन लंबवत् त्रिभुज के समद्विभाजक एक बिंदु पर मिलते हैं: तीन से होकर जाने वाला वृत्त अंक, और दो दर्पण एक साथ क्या करते हैं}]
\label{pb:g6:symmetry:1}
में \cref{exo:g6:symmetry:8} एक फव्वारे को बराबर खड़ा करना पड़ा
से दूरी \emph{दो} पेड़, और उत्तर की एक पूरी पंक्ति थी
धब्बे. यह समस्या पौधे ए \emph{तीसरा} पेड़ --- और संपूर्ण
\omterm{def:g6:lines:objects}{रेखा} एक एकल, पूर्णतः निर्धारित बिंदु पर ढह जाती है। वह
बिंदु ज्यामिति का खजाना छुपाता है: यह आपको एक का चित्र बनाने की सुविधा देता है
तीन दिए गए बिंदुओं से होकर गुजरने वाला वृत्त, एक पूरी प्लेट का पुनर्निर्माण करें
एक टूटे हुए टुकड़े से, और यह हर \omterm{def:g3:shapes:shapes}{त्रिकोण} का है। आखिरी
भाग दो दर्पणों के साथ एक प्रयोग करता है जिसका परिणाम
अगले वर्ष की ज्यामिति की घोषणा करता है।

\medskip
\noindent\textbf{Part I --- Two trees, then a map of
territories.}
\begin{enumerate}
  \item दो बिंदु बनाएं $ A $ और $ B $ साथ $ AB = 6$ सेमी और
        का निर्माण करें \omterm{def:g6:symmetry:bisector}{लंब समद्विभाजक} का $[AB]$ के साथ
        दिशा सूचक यंत्र (\cref{met:g6:symmetry:bisector}). एक बिंदु चुनें
        $ M $ उस पर और रूलर से जाँच करें $ MA = MB $.
  \item \Cref{prop:g6:symmetry:bisector} दो बातें कहता है
        एक बार: हर बिंदु \emph{पर} समद्वि\omterm{def:g5:division:multiple}{भाजक} समदूरस्थ है
        से $ A $ और $ B $, और प्रत्येक समदूरस्थ बिंदु \emph{है
        पर} द्वि\omterm{def:g5:division:multiple}{भाजक}. दोनों में से कौन फव्वारे का उत्तर देता है?
        का प्रश्न \cref{exo:g6:symmetry:8}? क्या करता है
        प्रस्ताव एक बिंदु के बारे में कहता है \emph{नहीं} पर
        द्वि\omterm{def:g5:division:multiple}{भाजक}?
  \item एक बिंदु लीजिए $ M $ समद्वि\omterm{def:g5:division:multiple}{भाजक} के समान तरफ
        $ A $. \omterm{def:g6:lines:objects}{खंड} $[MB]$ एक बिंदु पर समद्वि\omterm{def:g5:division:multiple}{भाजक} को पार करता है
        $ P $. पथ की तुलना करें $ M \to P \to B $ पथ के साथ
        $ M \to P \to A $, और समझाएं क्यों $ M $ के करीब है $ A $
        की तुलना में $ B $: समद्वि\omterm{def:g5:division:multiple}{भाजक} शीट को एक में विभाजित करता है
        ``$ A $-साइड'' और एक ``$ B $-ओर''।
  \item दो स्कूल $ A $ और $ B $ एक शहर की सेवा करो. हर बच्चा चलता है
        नजदीकी स्कूल में. के साथ एक छोटा सा मानचित्र बनाएं $ A $ और $ B $,
        और स्कूल द्वारा संचालित शहर के सटीक \omterm{def:g4:measure:area}{क्षेत्र} को छायांकित करें
        $ A $. कौन सी \omterm{def:g6:lines:objects}{रेखा} सीमा बनाती है?
  \item एक तीसरा स्कूल $ C $ खुलता है (इसे तीन स्कूलों में रखें
        एक \omterm{def:g3:shapes:shapes}{त्रिकोण} बनाओ)। का निर्माण करें \omterm{def:g6:symmetry:bisector}{लंब समद्विभाजक}
        का $[AB]$ और यह \omterm{def:g6:symmetry:bisector}{लंब समद्विभाजक} का $[BC]$, और
        बुलाओ $ O $ उनका क्रॉसिंग पॉइंट. क्या दो समानताएं हैं
        दूरियाँ करती है $ O $ संतुष्ट करें, और क्यों?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Three trees, one point, one
circle.}
\begin{enumerate}[resume]
  \item प्रश्न 5 से निष्कर्ष निकालें कि $ OA = OC $ --- और इसलिए
        वह $ O $ पर भी पड़ा है \omterm{def:g6:symmetry:bisector}{लंब समद्विभाजक} का
        $[AC]$, जो आपने कभी नहीं बनाया। निष्कर्ष: \emph{तीन
        लंबवत समद्वि\omterm{def:g5:division:multiple}{भाजक} एक त्रिभुज की सभी भुजाओं का
        एक बिंदु से गुजरें.}
  \item बताएं कि केंद्र का वृत्त क्यों है? $ O $ और \omterm{def:g3:shapes:shapes}{RADIUS} $ OA $
        से गुजरता है $ B $ और के माध्यम से $ C $ साथ ही: यह है
        \emph{} तीन बिंदुओं के माध्यम से \omterm{def:g4:numbers:round}{गोल}ा बनाएं ---
        \emph{परिबद्ध घेरा}\index{परिबद्ध घेरा}
        \omterm{def:g3:shapes:shapes}{त्रिकोण} का $ ABC $.
  \item भुजाओं वाला एक त्रिभुज बनाइये $6$ सेमी, $5$ सेमी और
        $4$ सेमी (\cref{met:g6:shapes:construct}), तो यह है
        \omterm{pb:g6:symmetry:1}{परिबद्ध घेरा}. केवल चित्र बनाना ही पर्याप्त क्यों है?
        \emph{दो} लंबवत समद्वि\omterm{def:g5:division:multiple}{भाजक} केंद्र खोजने के लिए?
  \item टूटी हुई प्लेट: एक पुरातत्ववेत्ता ने एक टुकड़ा खोदा जिसका
        एकमात्र अक्षुण्ण भाग प्लेट के \omterm{def:g4:numbers:round}{गोल}ाकार रिम का एक टुकड़ा है।
        एक ऐसे नुस्खे का वर्णन करें जो केंद्र और को पुनः प्राप्त करता है
        \omterm{def:g3:shapes:shapes}{RADIUS} मूल प्लेट का --- और उसका परीक्षण करें: एक चित्र बनाएं
        अपने कंपास से वृत्त का चाप, केंद्र को ``भूलें'',
        चाप पर तीन बिंदु चिह्नित करें, और इसका पुनर्निर्माण करें।
  \item क्या कभी तीन अंक हो सकते हैं \emph{नहीं} के माध्यम से घूमना
        उन्हें? कल्पना करना $ A $, $ B $, $ C $ संरेखित हैं. क्या हो सकता है
        के बारे में कहा लंबवत समद्वि\omterm{def:g5:division:multiple}{भाजक} का $[AB]$ और का
        $[BC]$ (\cref{prop:g6:lines:facts})? निष्कर्ष.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Axes everywhere, and the walking
mirror.}
\begin{enumerate}[resume]
  \item \Cref{ex:g6:symmetry:shapes} गिना गया $3$ के लिए कुल्हाड़ियों
        समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण} और $4$ वर्ग के लिए. अनुमान
        एक नियमित के लिए गिनती \omterm{def:g4:geometry:polygon}{पंचकोण} और एक नियमित \omterm{def:g4:geometry:polygon}{षट्भुज},
        और वर्णन करें कि प्रत्येक मामले में कुल्हाड़ियाँ कहाँ से गुजरती हैं (अजीब
        संख्या पक्षों के विरुद्ध \omterm{def:g3:numbers:evenodd}{यहां तक की}: दो स्थितियाँ
        भिन्न)।
  \item कागज की एक शीट को एक बार मोड़ें और दोनों में से एक आकृति काट लें
        परतें; प्रकट करना. छेद क्यों होता है \emph{हमेशा} एक है
        \omterm{def:g6:symmetry:axis}{समरूपता की धुरी}, जो कुछ भी आप काटते हैं, और कौन सी \omterm{def:g6:lines:objects}{रेखा} है
        धुरी?
  \item शीट को दो बार मोड़ें --- दूसरी बार मोड़ें \omterm{def:g6:lines:perp}{सीधा}
        पहले तक --- काटें, और खोलें। के कितने अक्ष
        छेद में समरूपता कम से कम है, और कितनी है
        क्या आप अपने कट की प्रतियां देखते हैं? (पेपर स्नोफ्लेक्स हैं
        यह विचार और भी आगे बढ़ गया।)
  \item वॉकिंग मिरर: ग्रिड पेपर पर, एक ऊर्ध्वाधर अक्ष बनाएं
        $ d_1$, एक छोटा सा झंडा $ F $ इसके बायीं ओर, और एक दूसरा
        ऊर्ध्वाधर अक्ष $ d_2$, $4$ के दाईं ओर वर्ग $ d_1$.
        प्रतिबिंबित होना $ F $ आर-पार $ d_1$ (छवि $ F'$), फिर प्रतिबिंबित करें
        $ F'$ आर-पार $ d_2$ (छवि $ F''$). तुलना करना $ F $ और $ F''$:
        समान अभिविन्यास या प्रतिबिंबित? कितने वर्गों से किया
        आकृति चाल? के बीच की दूरी से तुलना करें
        कुल्हाड़ियाँ
  \item के साथ भी वैसा ही प्रयोग $ d_2$ \emph{\omterm{def:g6:lines:perp}{सीधा}} को
        $ d_1$: फिर, ध्वज को ऊर्ध्वाधर अक्ष पर प्रतिबिंबित करें
        छवि क्षैतिज के पार। वर्णन करें कैसे $ F''$
        के सापेक्ष बैठता है $ F$ और के पार बिंदु तक
        दो अक्ष. (आपने अभी अगले वर्ष की खोज की है
        परिवर्तन: \cref{ch:g7:central}.)
\end{enumerate}
\end{problem}
