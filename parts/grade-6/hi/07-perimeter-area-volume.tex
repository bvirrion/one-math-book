\chapter{परिधि, क्षेत्रफल, आयतन}\label{ch:g6:measure}

बाड़ कितनी लंबी है, मैदान कितना बड़ा है, उसमें कितना पानी समा सकता है
टैंक? तीन अलग-अलग प्रश्न, तीन अलग-अलग मात्राएँ ---
\omterm{def:g6:measure:perimeter}{परिधि}, \omterm{def:g6:measure:area}{क्षेत्र}, \omterm{def:g6:measure:volume}{आयतन} --- प्रत्येक की अपनी इकाइयाँ हैं। उन्हें भ्रमित करना है
ज्यामिति में सबसे आम गलती; यह अध्याय उनका निराकरण करता है
अच्छा.

\section{लंबाई और परिधि}

\begin{definition}[परिधि]\label{def:g6:measure:perimeter}
 \emph{परिधि}\index{परिधि} एक आकृति की कुल लंबाई है
इसकी सीमा का. इसे लंबाई की इकाइयों में मापा जाता है: मिली\omterm{def:g4:fractions:def}{मीटर} (मिमी),
सेंटी\omterm{def:g4:fractions:def}{मीटर} (सेमी), \omterm{def:g4:fractions:def}{मीटर} (एम), किलो\omterm{def:g4:fractions:def}{मीटर} (किमी), साथ
\[
1 \text{ km} = 1000 \text{ m}, \qquad
1 \text{ m} = 100 \text{ cm}, \qquad
1 \text{ cm} = 10 \text{ mm}.
\]
\end{definition}

\begin{example}\label{ex:g6:measure:perimeter}
लंबाई का एक आयत $L $ और चौड़ाई $ w $ है \omterm{def:g6:measure:perimeter}{परिधि}
\[
P = L + w + L + w = 2 \times (L + w).
\]
एक के लिए $7$ सेमी द्वारा $4$ सेमी आयत: $ P = 2 \times (7 + 4) = 2 \times 11 =
22$ सेमी। भुजा का एक वर्ग $ c $ है \omterm{def:g6:measure:perimeter}{परिधि} $4c $.
\end{example}

\begin{proposition}[एक वृत्त की परिधि]\label{prop:g6:measure:circle}
 \omterm{def:g6:measure:perimeter}{परिधि} (या \emph{\omterm{def:g6:measure:perimeter}{परिधि}}) के एक वृत्त का \omterm{def:g3:shapes:shapes}{RADIUS} $ r $ है
\[
P = 2 \pi r ,
\]
कहाँ $\pi \approx 3.14$ प्रत्येक सर्कल के लिए समान संख्या है.
\end{proposition}
\admitted

\begin{example}
एक \omterm{def:g4:numbers:round}{गोल}ाकार तालाब है \omterm{def:g3:shapes:shapes}{RADIUS} $5$ एम। इसकी सीमा मापी जाती है
$2 \times \pi \times 5 = 10\pi \approx 31.4$ एम। सटीक मान रखें
$10\pi $ जब तक संभव है; \omterm{def:g4:numbers:round}{गोल} केवल अंत में.
\end{example}

\section{क्षेत्रों}

\begin{definition}[क्षेत्र]\label{def:g6:measure:area}
 \emph{क्षेत्र}\index{क्षेत्र} एक आकृति का माप उस सतह को मापता है जिसे वह कवर करती है:
कितने इकाई वर्ग अंदर फिट होते हैं. इकाइयाँ: सेमी$^2$ (भुजा का एक वर्ग
$1$ सेमी), मी $^2$, किमी $^2$, \dots\ सावधान:
\[
1 \text{ m}^2 = 100 \times 100 \text{ cm}^2 = 10\,000 \text{ cm}^2
\]
--- एक वर्ग \omterm{def:g4:fractions:def}{मीटर} एक है $100$ सेमी द्वारा $100$ सेमी वर्ग, इसलिए प्रत्येक चरण
यूनिट सीढ़ी लायक है $100$, नहीं $10$.
\end{definition}

तो $7 \times 4 = 28$ कुल मिलाकर वर्ग. इसलिए \omterm{def:g6:measure:area}{क्षेत्र} $=$ लंबाई
$\times $ चौड़ाई।}
\end{omfigure}

\begin{proposition}[मूल क्षेत्र सूत्र]\label{prop:g6:measure:formulas}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{l|l}
figure & \omterm{def:g6:measure:area}{क्षेत्र} \\
\hline
rectangle ($ L \times w $) & $ A = L \times w $ \\
square (side $ c $) & $ A = c^2$ \\[4pt]
सही \omterm{def:g3:shapes:shapes}{त्रिकोण} (legs $ a $, $ b $) & $ A = \dfrac{a \times b}{2}$ \\[4pt]
disk (\omterm{def:g3:shapes:shapes}{RADIUS} $ r $) & $ A = \pi r^2$ \\
\end{tabular}
\end{center}
\end{proposition}

\begin{proof}[समकोण त्रिभुज का प्रमाण]
ए की दो प्रतियां सही \omterm{def:g3:shapes:shapes}{त्रिकोण} पैरों के साथ $ a $ और $ b $, साथ चिपका हुआ
कर्ण, रूप ए $ a \times b $ आयत। तो \omterm{def:g3:shapes:shapes}{त्रिकोण} का \omterm{def:g6:measure:area}{क्षेत्र} है
\omterm{def:g3:division:half}{आधा} आयत का: $\frac{ab}{2}$. (आयत सूत्र है
ऊपर इकाई-वर्ग गिनती; डिस्क सूत्र स्वीकृत है, देखें
\cref{ch:g7:areas}.)
\end{proof}

$\frac{a\times b}{2}$.}

\begin{example}[समग्र आंकड़े]\label{ex:g6:measure:composite}
एक एल आकार का कमरा है $6$ एम $\times $ $4$ ए के साथ एम आयत $2$ एम
$\times $ $2$ मी वर्गाकार कोना हटा दिया गया। इसका \omterm{def:g6:measure:area}{क्षेत्र}, क्रमशः:
\begin{enumerate}
  \item पूर्ण आयत: $6 \times 4 = 24$ एम $^2$;
  \item हटाया गया वर्ग: $2 \times 2 = 4$ एम $^2$;
  \item \omterm{def:g3:division:remainder}{शेष} \omterm{def:g6:measure:area}{क्षेत्र}: $24 - 4 = 20$ एम $^2$.
\end{enumerate}
इसका \emph{\omterm{def:g6:measure:perimeter}{परिधि}} क्या नहीं है $20$ कुछ भी: एल के चारों ओर घूमना, द
सीमा अभी भी मापती है $6 + 4 + 6 + 4 = 20$ मी --- के दो कट
\omterm{def:g5:solids:def}{कोने} को दीवार के दो बराबर टुकड़ों से बदलें। संयोग से वही संख्या,
लेकिन एक के लिए वर्ग \omterm{def:g4:fractions:def}{मीटर}, दूसरे के लिए \omterm{def:g4:fractions:def}{मीटर}!
\end{example}

\begin{remark}[एक ही परिधि, अलग-अलग क्षेत्र]
दो आकृतियाँ समान हो सकती हैं \omterm{def:g6:measure:perimeter}{परिधि} और बहुत अलग \omterm{def:g6:measure:area}{क्षेत्र}ों: ए
$5 \times 5$ वर्ग और ए $9 \times 1$ आयत दोनों के पास है \omterm{def:g6:measure:perimeter}{परिधि}
$20$, लेकिन \omterm{def:g6:measure:area}{क्षेत्र}ों $25$ और $9$. \omterm{def:g6:measure:perimeter}{परिधि} और \omterm{def:g6:measure:area}{क्षेत्र} वास्तव में स्वतंत्र हैं
मात्राएँ.
\end{remark}

\section{संस्करणों}

\begin{definition}[आयतन]\label{def:g6:measure:volume}
 \emph{आयतन}\index{आयतन} एक का \omterm{def:g5:solids:def}{ठोस} उसके द्वारा भरे गए स्थान को मापता है:
कितनी यूनिट क्यूब्स अंदर फिट हो गया। इकाइयाँ: सेमी $^3$, एम $^3$, \dots\ साथ
$1$ एम $^3 = 1\,000\,000$ सेमी $^3$ (सीढ़ी का प्रत्येक चरण मूल्यवान है
$1000$). तरल पदार्थों के लिए भी इसका उपयोग किया जाता है \emph{लीटर}:
\[
1 \text{ L} = 1 \text{ dm}^3 = 1000 \text{ cm}^3,
\qquad
1 \text{ m}^3 = 1000 \text{ L}.
\]
\end{definition}

\begin{proposition}[एक डिब्बे का आयतन]\label{prop:g6:measure:box}
एक आयताकार \omterm{def:g5:solids:def}{डिब्बा} (ए \emph{आयताकार आयता}) लंबाई का $ L $, चौड़ाई $ w $
और ऊंचाई $ h $ है \omterm{def:g6:measure:volume}{आयतन}
\[
V = L \times w \times h .
\]
\end{proposition}

\begin{proof}
निचली परत में शामिल है $ L \times w $ इकाई क्यूब्स
(\cref{def:g6:measure:area} चित्र, के साथ क्यूब्स), और वहाँ हैं $ h $
ऐसी परतें.
\end{proof}

      \foreach \z in {0,1} {

    }
  }
  \node[below, font=\small] at (2.5,-0.3)

\end{tikzpicture}

{\small ए \omterm{def:g5:solids:def}{डिब्बा} इकाई से भरा हुआ क्यूब्स: $4 \times 2$ क्यूब्स प्रति परत, दो
परतें.}
\end{omfigure}

\begin{example}\label{ex:g6:measure:aquarium}
एक मछलीघर उपाय $60$ सेमी द्वारा $30$ सेमी द्वारा $40$ सेमी (ऊंचाई). \omterm{def:g6:measure:volume}{आयतन}:
\[
V = 60 \times 30 \times 40 = 72\,000 \text{ cm}^3 = 72 \text{ dm}^3
= 72 \text{ L}.
\]
रूपांतरण के लिए चरण दर चरण: $1000$ सेमी $^3$ एक लीटर बनाओ, और
$72\,000 \div 1000 = 72$.
\end{example}

\section{समस्या: रानी डिडो की बाड़}

\begin{problem}[{सप्ताहांत समस्या --- बाड़ की एक निश्चित लंबाई के साथ, कौन सी आकृति सबसे अधिक भूमि को घेरती है? वर्ग प्रत्येक को हराता है आयत, वृत्त वर्ग को हरा देता है, और एक खलिहान की दीवार बदल जाती है सब कुछ}]
\label{pb:g6:measure:1}
किंवदंती कहती है कि रानी डिडो, अफ्रीका के तट पर उतरी थी
''जितनी ज़मीन एक बैल की खाल घेर सकती है'' दी गई --- इसलिए उसने काट दी
खाल को एक बेहद लंबी पतली पट्टी में बाँट लें और पर्याप्त रूप से बंद कर दें
कार्थेज शहर की स्थापना के लिए भूमि। उसकी समस्या अब आपकी है: a
किसान बिल्कुल मालिक है $20$ बाड़ का मी. \Cref{exo:g6:measure:8}
दिखाया कि दो पेन एक जैसे हैं \omterm{def:g6:measure:perimeter}{परिधि} भिन्न हो सकते हैं
\omterm{def:g6:measure:area}{क्षेत्र}ों; यह समस्या ढूँढती है \emph{श्रेष्ठ} कलम --- और पता चलता है
जब एक खलिहान की दीवार मुफ्त में उधार देती है तो उत्तर पूरी तरह से बदल जाता है
ओर.

\medskip
\noindent\textbf{Part I --- Twenty meters of fence.}
\begin{enumerate}
  \item सभी का उपयोग करके पेन एक आयत होना चाहिए $20$ बाड़ का मी.
        जाँच करें कि ए $1 \times 9$ कलम और ए $2 \times 8$ दोनों कलम
        अर्हता प्राप्त करें, और उनकी गणना करें \omterm{def:g6:measure:area}{क्षेत्र}ों.
  \item बताएं कि प्रत्येक क्वालीफाइंग पेन की लंबाई और चौड़ाई क्यों होती है
        तक \omterm{def:g1:addition:def}{जोड़}ें $10$ एम। फिर इसकी पूरी तालिका बनाएं
        पूर्णांक पेन ($1 \times 9$ तक $5 \times 5$) साथ उनके \omterm{def:g6:measure:area}{क्षेत्र}ों. कौन सा सर्वोत्तम है?
  \item क्या दशमलव पक्ष आज़माने लायक हैं? की गणना करें \omterm{def:g6:measure:area}{क्षेत्र}ों की
        $4.5 \times 5.5$ कलम और का $4.9 \times 5.1$ कलम, और
        के साथ तुलना करें $5 \times 5$ वर्ग। आप क्या करते हैं?
        अनुमान?
  \item प्रश्न 2 ने बाड़ समस्या को शुद्ध संख्या में बदल दिया
        प्रश्न: \emph{जोड़ने वाली संख्याओं के सभी युग्मों के बीच
        $10$, कौन सा \omterm{def:g1:addition:def}{जोड़}ा सबसे बड़ा है \omterm{def:g2:mult:def}{उत्पाद}?} इसका उत्तर दीजिये
        अपनी तालिका, और आपके द्वारा पालन किए जाने वाले सामान्य नियम को बताएं।
  \item यही कारण है कि वर्ग से दूर जाने पर हमेशा हानि होती है,
        बीजगणित के बजाय कैंची से: से प्रारंभ करें
        $5 \times 5$ कलम और इसे में बदलें $6 \times 4$ कलम
        एक पट्टी हटाकर और दूसरी पट्टी पीछे चिपकाकर। जो
        पट्टी हटा दी जाती है, जो \omterm{def:g1:addition:def}{जोड़}ा जाता है, और क्यों \omterm{def:g1:addition:def}{जोड़}ा जाता है
        एक्सचेंज ठीक एक वर्ग \omterm{def:g4:fractions:def}{मीटर} खो देता है? स्पष्ट करें कि क्यों ए
        आगे का कदम (को $7 \times 3$) हार जाता है \omterm{def:g3:numbers:evenodd}{यहां तक की} अधिक।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The barn, and the circle.}
\begin{enumerate}[resume]
  \item कलम अब एक लंबी खलिहान की दीवार के खिलाफ बनाया गया है: दीवार
        पेन की एक लंबाई को बदल देता है, और $20$ बाड़ का मी
        केवल अन्य तीन पक्षों को ढकें। की एक तालिका बनायें
        पूर्णांक पेन (चौड़ाई) $ w $, लंबाई $ L $ दीवार के साथ,
        $ w + L + w = 20$) और उनके \omterm{def:g6:measure:area}{क्षेत्र}ों. अब कौन सी कलम जीतती है ---
        और क्या यह एक वर्ग है?
  \item विजेता को दर्पण से समझाओ
        (\cref{ch:g6:symmetry}): खलिहान के पार कलम को प्रतिबिंबित करें
        दीवार और दोहरी कलम पर विचार करें। कितनी बाड़ लगाती है
        भाग I के अनुसार डबल पेन का उपयोग, कौन सा डबल पेन सबसे अच्छा है,
        और इससे मूल कलम क्या बनती है?
  \item डिडो ने एक आयत नहीं बनाया. मोड़ो $20$ बाड़ का मी
        एक घेरे में: का उपयोग करना $\pi \approx 3.14$, इसकी गणना करें
        \omterm{def:g3:shapes:shapes}{RADIUS} (सेमी तक), फिर यह \omterm{def:g6:measure:area}{क्षेत्र} (उन्हें $^2$)
        (\cref{prop:g6:measure:circle},
        \cref{prop:g6:measure:formulas}). से तुलना करें
        $5 \times 5$ वर्ग।
  \item उलटी समस्या: की एक कलम \omterm{def:g6:measure:area}{क्षेत्र} बिल्कुल $36$ एम $^2$ है
        जितना संभव हो उतना कम बाड़ लगाना चाहता था। तुलना करें
        आयत $1 \times 36$, $2 \times 18$, $3 \times 12$,
        $4 \times 9$ और $6 \times 6$: \omterm{def:g6:measure:perimeter}{परिधि}? कौन सी आकृति है
        सबसे सस्ता, और यह प्रश्न कैसा दर्पण प्रतिबिम्ब है
        भाग I?
  \item एक वाक्य में: डिब्बे, पाइप और पानी की टंकियाँ ऐसी क्यों हैं?
        अक्सर \emph{\omterm{def:g4:numbers:round}{गोल}}?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- \omterm{def:g6:measure:perimeter}{परिधि} and \omterm{def:g6:measure:area}{क्षेत्र} are strangers.}
\begin{enumerate}[resume]
  \item एक आयत खोजें जिसका \omterm{def:g6:measure:perimeter}{परिधि} से अधिक है $100$ मी लेकिन
        किसका \omterm{def:g6:measure:area}{क्षेत्र} मै रुक जाना $1$ एम $^2$. (बहुत लंबा और बहुत
        पतला. दशमलव की अनुमति है.)
  \item सही या ग़लत: ``दो आकृतियों में से, एक बड़ी आकृति वाली
        \omterm{def:g6:measure:perimeter}{परिधि} बड़ा है \omterm{def:g6:measure:area}{क्षेत्र}.'' एक प्रतिउदाहरण दीजिए
        इस समस्या से.
  \item ले लो $6 \times 4$ आयत और कट ए $1 \times 1$
        एक लंबी भुजा के मध्य में वर्गाकार पायदान (नॉच)।
        बाहर की ओर खुलता है, टूटे हुए दांत की तरह)। की गणना करें \omterm{def:g6:measure:area}{क्षेत्र}
        और यह \omterm{def:g6:measure:perimeter}{परिधि} नोकदार आकृति का, और दोनों की तुलना करें
        मूल के साथ. प्रत्येक को क्या हुआ?
  \item मानचित्र निर्माता एक अजीब तथ्य जानते हैं: मानचित्र जितना अधिक विस्तृत होगा,
        द \emph{अब} एक समुद्र तट मापता है --- प्रत्येक ज़ूम
        नए छोटे निशानों का पता चलता है, और प्रश्न 13 क्या दिखाता है
        प्रत्येक पायदान करता है. एक या दो वाक्यों में स्पष्ट करें कि ऐसा क्यों है
        ``ब्रिटनी के तट की लंबाई'' फिसलन भरी है
        संख्या, जबकि ``द \omterm{def:g6:measure:area}{क्षेत्र} ब्रिटनी का'' नहीं है.
  \item किसान की परीक्षा. साथ $36$ बाड़ और खलिहान का मी
        दीवार उपलब्ध है, सबसे अच्छा आयताकार पेन ढूंढें (उपयोग करें)।
        प्रश्न 7 का दर्पण), उसका दो \omterm{def:g6:measure:area}{क्षेत्र}, और तुलना करें
        सबसे अच्छा पेन बनाया गया $36$ खुले मैदान में हूँ. कितना
        क्या खलिहान की दीवार से किसान की कमाई होती है?
\end{enumerate}
\end{problem}
