\chapter{भिन्न: पहला चरण}\label{ch:g6:fractions}

चार लोगों के बीच तीन पिज़्ज़ा बाँटने से प्रत्येक व्यक्ति को लाभ मिलता है\,\dots\
पिज़्ज़ा की पूरी संख्या नहीं. भिन्न संख्याओं का आविष्कार किसके लिए किया गया है?
साझा करना. यह अध्याय चित्र बनाता है: a \omterm{def:g6:fractions:def}{अंश} एक के एक भाग के रूप में
संपूर्ण, संख्या रेखा पर एक बिंदु के रूप में, और a के सटीक परिणाम के रूप में
प्रभाग.

\section{भिन्न का क्या अर्थ है}

\begin{definition}[अंश]\label{def:g6:fractions:def}
में एक इकाई काटें $b $ बराबर भागों में, और ले लो $ a $ उनमें से: मात्रा
प्राप्त है \emph{अंश}\index{अंश}
\[
\frac{a}{b}
\qquad
\begin{array}{l}
a \text{: the \emph{मीटर} --- how many parts we take;}\\
b \text{: the \emph{भाजक} --- into how many parts the unit is cut.}
\end{array}
\]
\end{definition}

\begin{example}\label{ex:g6:fractions:morethanone}
ए \omterm{def:g6:fractions:def}{अंश} से बड़ा हो सकता है $1$: $\frac74$ मतलब सात क्वार्टरों ---
एक पूरी इकाई (चार क्वार्टरों) और तीन क्वार्टरों अधिक:
\[
\frac74 = 1 + \frac34 .
\]
\end{example}

\begin{proposition}[संख्या रेखा पर भिन्न]\label{prop:g6:fractions:line}
रखना $\frac ab $ संख्या रेखा पर प्रत्येक इकाई \omterm{ex:g1:subtraction:difference}{अंतर}ाल को काटें $ b $
बराबर भाग और गिनती $ a $ से भाग $0$.
\end{proposition}

\begin{theorem}[भिन्न एक विभाजन है]\label{thm:g6:fractions:quotient}
 \omterm{def:g6:fractions:def}{अंश} $\frac ab $ वह संख्या है, जिसे गुणा किया जाता है $ b $, देता है
$ a $:
\[
\frac ab \times b = a .
\]
दूसरे शब्दों में, $\frac ab $ सटीक है \omterm{def:g3:division:remainder}{भागफल} का $ a $ द्वारा $ b $ --- यहां तक कि
जब बँटवारा ``निकलता ही नहीं''।
\end{theorem}
\admitted

\begin{example}\label{ex:g6:fractions:pizzas}
चार लोगों के लिए तीन पिज़्ज़ा: प्रत्येक व्यक्ति को मिलता है $\frac34$ एक पिज़्ज़ा का.
प्रमेय से जाँचें: $4$ के शेयर $\frac34$ बनाना
$\frac34 \times 4 = 3$ पिज्जा. कुछ \omterm{def:g6:fractions:def}{अंश}ों हैं दशमलव संख्याएं
($\frac34 = 0.75$), अन्य नहीं हैं: $\frac13 = 0.333\dots $ कभी नहीं रुकता.
 \omterm{def:g6:fractions:def}{अंश} है \emph{एकदम सही} कीमत; $0.33$ केवल एक अनुमान है.
\end{example}

\section{समान भिन्न}

\begin{proposition}[समतुल्य भिन्न]\label{prop:g6:fractions:equal}
ए \omterm{def:g6:fractions:def}{अंश} जब यह बदलता नहीं है \omterm{def:g4:fractions:def}{मीटर} और \omterm{def:g5:division:multiple}{भाजक} दोनों हैं
समान अशून्य संख्या से गुणा (या विभाजित):
\[
\frac{a}{b} = \frac{a \times k}{b \times k} .
\]
\end{proposition}

\begin{proof}[प्रमाण का विचार]
साझाकरण के प्रत्येक भाग को काटना $ k $ छोटे टुकड़े दोनों को गुणा करते हैं
लिए गए टुकड़ों की संख्या और टुकड़ों की कुल संख्या $ k $, लेकिन
ली गई मात्रा समान है. एक तस्वीर इसे सबसे अच्छी तरह कहती है:
\end{proof}

\begin{method}[भिन्न को सरल बनाना]\label{met:g6:fractions:simplify}
को विभाजित करने वाली संख्या खोजें \omterm{def:g4:fractions:def}{मीटर} \emph{और}  \omterm{def:g5:division:multiple}{भाजक},
और दोनों को बांट दो. जब तक कोई सामान्य न हो तब तक दोहराएँ \omterm{def:g5:division:multiple}{भाजक} अवशेष:
\[
\frac{12}{18} = \frac{12 \div 2}{18 \div 2} = \frac{6}{9}
= \frac{6 \div 3}{9 \div 3} = \frac{2}{3}.
\]
\end{method}

\section{किसी मात्रा का अंश लेना}

\begin{method}[किसी मात्रा का अंश]\label{met:g6:fractions:ofquantity}
गणना करना $\frac ab $ किसी मात्रा का, मात्रा को इससे विभाजित करें $ b $ (एक
भाग), फिर से गुणा करें $ a $ (भागों की संख्या):
\[
\frac ab \text{ of } Q = (Q \div b) \times a .
\]
(पहले से गुणा करना $ a $ और फिर से विभाजित करना $ b $ वही देता है
परिणाम --- वह क्रम चुनें जो संख्याओं को बेहतर बनाता है।)
\end{method}

\begin{example}\label{ex:g6:fractions:ofquantity}
गणना करना $\frac35$ का $40$ यूरो, चरण दर चरण:
\begin{enumerate}
  \item का पांचवां हिस्सा $40$: $40 \div 5 = 8$;
  \item तीन पाँचवाँ: $8 \times 3 = 24$.
\end{enumerate}
इसलिए $\frac35$ का $40$ यूरो है $24$ यूरो. जांचें: \omterm{def:g3:division:remainder}{शेष}
$\frac25$ है $16$ यूरो, और $24 + 16 = 40$.
\end{example}

\section{समस्या: चॉकलेट बार जिसे आप कभी ख़त्म नहीं कर सकते}

\begin{problem}[{सप्ताहांत समस्या---आधे का आधा: योग $\frac12 + \frac14 + \frac18 + \dots $ तक रेंगता है $1$ बिना कभी इसे छूना}]
\label{pb:g6:fractions:1}
एक चॉकलेट बार लें. खाओ \omterm{def:g3:division:half}{आधा} इसका. तो खाओ \omterm{def:g3:division:half}{आधा} का \emph{क्या
बाकी है}. तो फिर \omterm{def:g3:division:half}{आधा} क्या बचा है, इत्यादि। दो बातें
यह एक बार में सच प्रतीत होता है: आप कभी भी बार को पूरा नहीं करते (हमेशा कुछ न कुछ)।
रहता है), और फिर भी आप इसका लगभग सारा हिस्सा खा लेते हैं। यह समस्या दोनों को बदल देती है
भावनाओं के बारे में सटीक कथनों में \omterm{def:g6:fractions:def}{अंश}ों --- और मिलते हैं, पर
रास्ता, प्राचीन ग्रीस के एक धावक और एक साझा पहेली का समाधान
पूरी तरह से एक चाकू से जो केवल चीज़ों को काटता है \omterm{def:g3:division:half}{आधा}.

\medskip
\noindent\textbf{Part I --- Halves of halves.}
बार एक आयत है $16$ समान वर्ग.
\begin{enumerate}
  \item पहला निवाला: तुम खाओ \omterm{def:g3:division:half}{आधा} बार। कितने वर्ग है
        वह? काटने को इस प्रकार लिखें \omterm{def:g6:fractions:def}{अंश} बार की दो तरह से,
        के साथ \omterm{def:g5:division:multiple}{भाजक} $2$ और साथ में \omterm{def:g5:division:multiple}{भाजक} $16$
        (\cref{prop:g6:fractions:equal}).
  \item दूसरा दंश: \omterm{def:g3:division:half}{आधा} जो बचा है उसका. कितने वर्ग? दिखाओ,
        चित्र के साथ या समकक्ष के साथ \omterm{def:g6:fractions:def}{अंश}ों, कि यह
        काटना है $\frac14$ पूरे बार का: \emph{\omterm{def:g3:division:half}{आधा} एक का \omterm{def:g3:division:half}{आधा}
        एक है \omterm{def:g3:division:half}{तिमाही}}.
  \item तीसरा और चौथा निवाला, एक ही नियम: प्रत्येक को अंदर दें
        वर्ग और एक के रूप में \omterm{def:g6:fractions:def}{अंश} बार का. क्या होता है
        \omterm{def:g5:division:multiple}{भाजक} एक काटने से दूसरे काटने तक?
  \item चार काटने के बाद, कितने वर्ग में खाया गया है
        सब? क्या \omterm{def:g6:fractions:def}{अंश} बार का वह है, और क्या \omterm{def:g6:fractions:def}{अंश}
        \omterm{def:g3:division:remainder}{शेष} है?
  \item प्रश्न 4 कहता है, सोलहवें में लिखा गया:
        \[
        \frac{8}{16} + \frac{4}{16} + \frac{2}{16} + \frac{1}{16}
        = \frac{15}{16},
        \qquad\text{that is}\qquad
        \frac12 + \frac14 + \frac18 + \frac{1}{16}
        = \frac{15}{16} .
        \]
        चार पुनर्लेखनों में से प्रत्येक की जाँच करें
        ($\frac12 = \frac{8}{16}$, और इसी तरह)।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The staircase towards $1$.}
काटते रहने के लिए, एक बेहतर बार की कल्पना करें $64$ वर्ग।
\begin{enumerate}[resume]
  \item पहले छह काटने के आकारों को वर्गों में सूचीबद्ध करें
        ($32$, तब\,\dots), और छठे काटने के बाद इसकी जांच करें
        ठीक एक वर्ग \omterm{def:g3:division:remainder}{शेष} है. क्या \omterm{def:g6:fractions:def}{अंश} बार का
        रहता है?
  \item प्रत्येक काटने के बाद, क्या \omterm{def:g6:fractions:def}{अंश} बार का अवशेष? लिखो
        सूची ($\frac12$, $\frac14$, \dots) छठे तक नीचे
        काटो और पैटर्न का वर्णन करो। यदि एक काल्पनिक चाकू
        \omterm{def:g3:division:half}{आधा} हो सकता है \omterm{def:g3:numbers:evenodd}{यहां तक की} अंतिम वर्ग, उसके बाद क्या बचेगा?
        सातवाँ दंश?
  \item बताएं कि खाया हुआ कुल क्यों हो सकता है \emph{कभी नहीं} पहुँचना $1$,
        चाहे कितने भी काटे जाएं. (तुलना करें
        उनतीस के \cref{pb:g6:decimals:1}: वही कहानी?)
  \item फिर भी कुल योग नीचे दिए गए किसी भी लक्ष्य को पार कर जाता है $1$. एक दोस्त
        आपको अधिक खाने की चुनौती देता है $\frac{99}{100}$ का
        बार. जो छोटा है, $\frac{1}{128}$ या
        $\frac{1}{100}$ --- और इसके बाद कौन सा दंश चुनौती है
        जीत गए?
  \item से एक संख्या रेखा पर $0$ को $1$ सोलहवीं में स्नातक,
        पहले चार में से प्रत्येक के बाद खाए गए योग को रखें
        काटने: $\frac12$, $\frac34$, $\frac78$, $\frac{15}{16}$.
        प्रत्येक नई कुल भूमि कहां है, इसकी तुलना में
        पिछले एक और साथ $1$?
\end{enumerate}

\noindent\textbf{Part III --- Zeno's runner, and a knife that
only halves.}
\begin{enumerate}[resume]
  \item चौबीस शताब्दी पहले, यूनानी दार्शनिक ज़ेनो
        यह कहानी बताई: एक दीवार तक पहुंचने के लिए, एक धावक को सबसे पहले दौड़ना होगा
        कवर \omterm{def:g3:division:half}{आधा} फिर दूरी \omterm{def:g3:division:half}{आधा} फिर जो बचता है
        \omterm{def:g3:division:half}{आधा} फिर क्या बचता है\,\dots\ ``तो धावक कभी नहीं
        दीवार तक पहुँचता है!'' प्रश्न 8 और 9 का प्रयोग करते हुए बताएं कि क्या है
        \emph{सही} ज़ेनो की कहानी में, और \omterm{def:g5:solids:net}{जाल} कहाँ है।
        (क्या धावक को वास्तव में वह सब बनाने की हमेशा के लिए ज़रूरत है
        कदम?)
  \item साझा करने की पहेली: साझा करें $15$ समान चॉकलेट बार
        काफी बीच में $16$ बच्चे, केवल हिस्सों में कटौती का उपयोग कर रहे हैं
        (एक बार का, आधे बार का, एक चौथाई बार का\,\dots).
        वर्णन करें कि कैसे काटें: कितनी छड़ों को आधा-आधा काटा जाता है,
        कितने में क्वार्टरों, आठवें में, सोलहवें में,
        ताकि प्रत्येक बच्चे को प्रत्येक आकार का एक टुकड़ा मिले।
  \item अपना साझाकरण जांचें: प्रत्येक बच्चे का हिस्सा एक के रूप में लिखें \omterm{def:g1:addition:def}{जोड़} का
        \omterm{def:g6:fractions:def}{अंश}ों, इसे सोलहवें भाग में गिनें, और सत्यापित करें कि
        $16$ शेयर एक साथ बिल्कुल उपयोग करते हैं $15$ बार.
  \item बगल की कक्षा में, $7$ बारों को आपस में उचित रूप से साझा किया जाता है
        $8$ बच्चे। किस बच्चे को अधिक चॉकलेट मिलती है --- एक
        का $16$, या इनमें से एक $8$? (दोनों शेयर साथ लिखें
        \omterm{def:g5:division:multiple}{भाजक} $16$, \cref{exo:g6:fractions:8}.)
  \item समापन: एक मित्र का दावा है कि साथ \emph{असीम}
        एक व्यक्ति कई निवाले खाएगा \emph{बिल्कुल} पूरी बार.
        एक वर्ग बनाएं और उसके अंदर काटे गए स्थानों को छायांकित करें: \omterm{def:g3:division:half}{आधा} द
        वर्ग, फिर ए \omterm{def:g3:division:half}{तिमाही}, फिर आठवां\,\dots\ क्या करें?
        क्या आप अछाया वाले \omterm{def:g5:solids:def}{कोने} के बारे में ध्यान देते हैं? ध्यान से बताइये
        इस पूरी समस्या के दो सच्चे तथ्य: क्या बचा है
        काटने की प्रत्येक सीमित संख्या के बाद, और वह कितना छोटा है
        \omterm{def:g3:division:remainder}{शेष} बन जाता है. (क्या ``अनंत है \omterm{def:g1:addition:def}{जोड़}'' बिल्कुल
        मीन्स हाई स्कूल खंड के लिए एक कहानी है।)
\end{enumerate}
\end{problem}
