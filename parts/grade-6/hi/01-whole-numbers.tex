\chapter{पूर्ण संख्याएं}\label{ch:g6:wholes}

गणित में हर चीज़ गिनती से शुरू होती है। यह अध्याय समीक्षा करता है कि कैसे
पूर्ण संख्याओं को लिखा जाता है, तुलना की जाती है और संयोजित किया जाता है --- और सावधानी बरतनी पड़ती है
विभाजन को देखें, वह संक्रिया जो दोनों a देती है \omterm{def:g3:division:remainder}{भागफल} \emph{और} ए
\omterm{def:g3:division:remainder}{शेष}.

\section{पूर्ण संख्याओं को लिखना एवं उनकी तुलना करना}

\begin{definition}[जगह की मूल्य]\label{def:g6:wholes:place}
संख्याओं को लिखने का हमारा तरीका दस अंकों का उपयोग करता है ($0$ को $9$), और यह
\emph{किसी अंक का मान उसके स्थान पर निर्भर करता है}\index{किसी अंक का मान उसके स्थान पर निर्भर करता है}: में
$5\,208$, अंक $5$ हजारों की गिनती, $2$ सैकड़ों की गिनती,
$0$ दहाई की गिनती होती है (कोई नहीं है) और $8$ इकाइयाँ गिनता है:
\[
5\,208 = 5 \times 1000 + 2 \times 100 + 0 \times 10 + 8 .
\]
\end{definition}

\begin{method}[दो पूर्ण संख्याओं की तुलना करना]\label{met:g6:wholes:compare}
\begin{enumerate}
  \item अधिक अंकों वाली संख्या बड़ी होती है
        ($1\,203 > 989$).
  \item यदि उनके अंकों की संख्या समान है, तो अंक दर अंक तुलना करें
        से \emph{बाएं}; पहला \omterm{ex:g1:subtraction:difference}{अंतर} तय करता है:
        $6\,742 > 6\,715$ क्योंकि दहाई के स्थान पर, $4 > 1$.
\end{enumerate}
प्रतीक हैं $<$ (``इससे कम'') और $>$ (``इससे भी बड़ा''); द
प्रतीक का छोटा सिरा छोटी संख्या की ओर इंगित करता है।
\end{method}

\begin{example}\label{ex:g6:wholes:order}
सबसे छोटे से सबसे बड़े तक ऑर्डर करें: $908$; $89$; $1\,001$; $980$.
सबसे पहले अंकों की गिनती से: $89$ (दो अंक) पहले आता है, $1\,001$ (चार
अंक) अंतिम। बीच में $908$ और $980$: वही सैकड़ों अंक $9$, फिर
दसियों $0 < 8$, इसलिए $908 < 980$. अंतिम उत्तर:
\[
89 < 908 < 980 < 1\,001 .
\]
\end{example}

\section{जोड़, घटाव, गुणा}

\begin{example}[स्तंभ गणना]\label{ex:g6:wholes:columns}
जोड़ने या घटाने के लिए, इकाइयों को इकाइयों के नीचे, दहाई को के नीचे संरेखित करें
दसियों, और दाएं से बाएं काम करें, जरूरत पड़ने पर ले जाएं:
\[
\begin{array}{r}
4\,5\,7 \\
+\ 2\,8\,6 \\
\hline
7\,4\,3
\end{array}
\qquad\qquad
\begin{array}{r}
6\,0\,3 \\
-\ 1\,4\,7 \\
\hline
4\,5\,6
\end{array}
\]
इसके अतिरिक्त: $7 + 6 = 13$, लिखना $3$, ढोना $1$; तब
$5 + 8 + 1 = 14$, लिखना $4$, ढोना $1$; तब $4 + 2 + 1 = 7$.
वापस जोड़कर घटाव की जाँच करें: $456 + 147 = 603$.
\end{example}

\begin{remark}[शब्दावली]
\omterm{def:g1:addition:def}{जोड़} का परिणाम है a \emph{जोड़}\index{जोड़}; घटाव का,
ए \emph{अंतर}\index{अंतर}; गुणन का, ए
\emph{उत्पाद}\index{उत्पाद}; एक प्रभाग का, ए
\emph{भागफल}\index{भागफल}. संयुक्त की जा रही संख्याएँ हैं
\emph{शर्तें} (के लिए $+$, $-$) या \emph{कारकों} (के लिए $\times $).
\end{remark}

\begin{example}[स्मार्ट गणना]\label{ex:g6:wholes:smart}
नियमों या कारकों को पुनः व्यवस्थित करने से अक्सर काम बच जाता है:
\[
17 + 58 + 3 = (17 + 3) + 58 = 20 + 58 = 78 ,
\]
\[
25 \times 17 \times 4 = (25 \times 4) \times 17 = 100 \times 17 = 1700 .
\]
\end{example}

\section{विभाजन}

\begin{theorem}[यूक्लिडियन प्रभाग]\label{thm:g6:wholes:division}
दो पूर्ण संख्याएँ दी गई हैं $ a $ (द \emph{लाभांश}) और $ b \neq 0$ (द
\emph{\omterm{def:g5:division:multiple}{भाजक}}), लिखने का बिल्कुल एक ही तरीका है
\[
a = b \times q + r
\qquad\text{with}\qquad
0 \leq r < b .
\]
जो नंबर $ q $ है \emph{भागफल}\index{भागफल} और $ r $ 
\emph{शेष}\index{शेष} के विभाजन का $ a $ द्वारा $ b $.
\end{theorem}
\admitted

\begin{example}\label{ex:g6:wholes:division}
विभाजित करना $137$ द्वारा $4$, क्रमशः।
\begin{enumerate}
  \item कितनी बार करता है $4$ अंदर जाएं $13$? तीन बार
        ($4 \times 3 = 12$), \omterm{def:g3:division:remainder}{शेष} $13 - 12 = 1$.
  \item नीचे लाओ $7$: कितनी बार करता है $4$ अंदर जाएं $17$?
        चार बार ($4 \times 4 = 16$), \omterm{def:g3:division:remainder}{शेष} $1$.
\end{enumerate}
इसलिए $ q = 34$ और $ r = 1$:
\[
137 = 4 \times 34 + 1, \qquad\text{and indeed } 0 \leq 1 < 4 .
\]
जाँच करना: $4 \times 34 = 136$, और $136 + 1 = 137$.
\end{example}

(पहुंचना $136$), और $1$ बचा हुआ है --- \omterm{def:g3:division:remainder}{शेष} हमेशा है
से छोटा \omterm{def:g5:division:multiple}{भाजक}.}
\end{omfigure}

\begin{definition}[भाज्य]\label{def:g6:wholes:divisible}
जब \omterm{def:g3:division:remainder}{शेष} है $0$, हम ऐसा कहते हैं $ a $ है
\emph{भाज्य}\index{भाज्य} द्वारा $ b $: उदाहरण के लिए $135$ है
से विभाज्य $5$ तब से $135 = 5 \times 27 + 0$.
\end{definition}

\begin{method}[विभाजन के साथ शब्द समस्याएँ]\label{met:g6:wholes:problems}
गणना करने के बाद $ a = bq + r $, हमेशा प्रश्न पर लौटें:
\begin{enumerate}
  \item ``कितने पूर्ण बॉक्स/टीम/पेज?'' --- उत्तर है
        \omterm{def:g3:division:remainder}{भागफल} $ q $;
  \item ``कितने बचे हैं?'' --- उत्तर है \omterm{def:g3:division:remainder}{शेष}
        $ r $;
  \item ``हर किसी के लिए कितने बक्सों की आवश्यकता है?'' --- यदि $ r > 0$, एक
        \emph{अधिक} की तुलना में \omterm{def:g3:division:remainder}{भागफल}: $ q + 1$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g6:wholes:bus}
$137$ की बसों में यात्रा पर जाते हैं छात्र $40$ सीटें. विभाजन:
$137 = 40 \times 3 + 17$. तीन बसें ले जाती हैं $120$ छात्र, और $17$
छात्र रहते हैं---उन्हें भी बस की जरूरत है! इसलिए $4$ बसों की भी जरूरत है
यद्यपि \omterm{def:g3:division:remainder}{भागफल} है $3$.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g6:wholes:1}
अंकों में लिखें: ``तीन हजार सैंतालीस''; ``बीस हजार

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:wholes:2}
संख्या में $47\,209$: सैकड़े का अंक क्या है? का अंक
इकाइयाँ? अंक क्या दर्शाता है $4$ गिनती करना? संख्या को a के रूप में लिखें \omterm{def:g1:addition:def}{जोड़} का
\omterm{def:g5:division:multiple}{गुणकों} का $1$, $10$, $100$, \dots\ के रूप में
\cref{def:g6:wholes:place}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:wholes:3}
कॉपी करें और पूरा करें $<$ या $>$:
\[
507 \;?\; 570, \qquad
1\,099 \;?\; 989, \qquad
23\,456 \;?\; 23\,465, \qquad
9\,999 \;?\; 10\,000 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:wholes:4}
सबसे छोटे से सबसे बड़े तक ऑर्डर करें: $2\,043$; $978$; $2\,304$; $2\,034$;

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:wholes:5}
कॉलम में गणना करें: $348 + 576$; \quad $805 - 268$; \quad
$307 \times 26$. \omterm{def:g1:addition:def}{जोड़} के साथ घटाव की जाँच करें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:wholes:6}
शब्दों या कारकों को समूहीकृत करते हुए चतुराई से गणना करें:
\[
38 + 45 + 12 + 5, \qquad
2 \times 83 \times 50, \qquad
4 \times 78 \times 25 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:wholes:7}
प्रत्येक प्रभाग के लिए, दें \omterm{def:g3:division:remainder}{भागफल} और यह \omterm{def:g3:division:remainder}{शेष}, और लिखें
समानता $ a = b \times q + r $:
\[
95 \div 7, \qquad
230 \div 6, \qquad
504 \div 8 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:wholes:8}
बिना बांटे बताएं कि समानता क्यों? $173 = 8 \times 20 + 13$ है
\emph{नहीं} का यूक्लिडियन प्रभाग $173$ द्वारा $8$, और सही खोजें
\omterm{def:g3:division:remainder}{भागफल} और \omterm{def:g3:division:remainder}{शेष}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:wholes:9}
अंडे को डिब्बों में पैक किया जाता है $12$.

  \item एक खेत है $2\,000$ अंडे. यह कितने पूर्ण बक्से भर सकता है, और

  \item एक बेकरी की जरूरत है $150$ अंडे. इसे कितने डिब्बे खरीदने होंगे?

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:wholes:10}

\begin{exercise}[$\star\star $]\label{exo:g6:wholes:11}

\begin{exercise}[$\star\star\star $]\label{exo:g6:wholes:12}

\section{समस्या: थोड़ा गॉस और पहले सौ का योग
नंबर}

\begin{problem}[{सप्ताहांत समस्या---जोड़ना $1 + 2 + 3 + \dots + 100$ दस सेकंड में, और त्रिकोणीय संख्याएँ}]
\label{pb:g6:wholes:1}
एक प्रसिद्ध कहानी: अपनी कक्षा को व्यस्त रखने के लिए, एक स्कूल मास्टर ने उससे पूछा
विद्यार्थियों को सभी संख्याओं को जोड़ना है $1$ को $100$. एक लड़के ने लिखा
$5\,050$ अपनी स्लेट पर लगभग तुरंत --- कार्ल फ्रेडरिक गॉस, नौ
वर्षों का, जो आगे चलकर महानतम गणितज्ञों में से एक बनेगा
हर समय का. उसका रहस्य और कुछ नहीं था \emph{स्मार्ट गणना}
(\cref{ex:g6:wholes:smart}): शब्दों को समूहीकृत करना ताकि प्रत्येक समूह
आसान है. यह समस्या उसकी चाल को फिर से खोजती है, इसे हर काम में लाती है
स्थिति, और दिखाता है कि ये कितनी बार होते हैं रकम दैनिक जीवन में दिखाई देते हैं।

\medskip
\noindent\textbf{Part I --- Gauss's pairs.}
\begin{enumerate}
  \item गणना करना $1 + 2 + 3 + 4 + 5$ शब्दों को चतुराई से समूहीकृत करके।
  \item के लिए $1 + 2 + 3 + \dots + 10$: पहले पद को के साथ \omterm{def:g1:addition:def}{जोड़}ें
        अंतिम ($1 + 10$), अगले-से-अंतिम के साथ दूसरा
        ($2 + 9$), और इसी तरह। प्रत्येक \omterm{def:g1:addition:def}{जोड़}ी का मूल्य क्या है? कितने
        \omterm{def:g1:addition:def}{जोड़}े हैं? गणना समाप्त करें.
  \item के लिए वही तरीका $1 + 2 + \dots + 20$: प्रत्येक \omterm{def:g1:addition:def}{जोड़}ी क्या है
        मूल्य, कितने \omterm{def:g1:addition:def}{जोड़}े, कुल कितना?
  \item अब स्कूल मास्टर का \omterm{def:g1:addition:def}{जोड़}, $1 + 2 + \dots + 100$: समझाओ
        \omterm{def:g1:addition:def}{जोड़}े क्यों $1 + 100$, $2 + 99$, $3 + 98, \dots $ सभी हैं
        वही मूल्य है, और वास्तव में क्यों हैं $50$ उनमें से,
        आखिरी वाला है $50 + 51$.
  \item गॉस के उत्तर को एकल गुणन के रूप में लिखें, और
        इसकी गणना करें.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- A trick that never fails.}
\begin{enumerate}[resume]
  \item \omterm{def:g1:addition:def}{जोड़}ी बनाने का प्रयास करें $1 + 2 + \dots + 9$: इस बार एक
        नंबर बिना पार्टनर के रह गया है. कौन सा? चारों ओर \omterm{def:g1:addition:def}{जोड़}ी
        यह और गणना करें \omterm{def:g1:addition:def}{जोड़}.
  \item यहां उस ट्रिक का एक संस्करण है जो काम करता है
        पदों की गिनती है \omterm{def:g3:numbers:evenodd}{यहां तक की} या \omterm{def:g3:numbers:evenodd}{विषम}: लिखना \omterm{def:g1:addition:def}{जोड़} दो बार, एक बार
        एक बार आगे और एक बार पीछे की ओर, एक रेखा के नीचे दूसरी रेखा,
        \[
        \begin{array}{ccccccc}
        1 & + & 2 & + & \dots & + & 9 \\
        9 & + & 8 & + & \dots & + & 1
        \end{array}
        \]
        और \omterm{def:g1:addition:def}{जोड़}ें \emph{कॉलम दर कॉलम}. प्रत्येक कॉलम क्या है
        लायक? वहां कितने कॉलम हैं? स्पष्ट करें कि क्यों \omterm{def:g1:addition:def}{जोड़} का
        दो पंक्तियाँ है $9 \times 10 = 90$, और निष्कर्ष निकालो
        $1 + 2 + \dots + 9 = 45$ --- जैसा ही उत्तर है
        प्रश्न 6.
  \item गणना करने के लिए दो-पंक्ति युक्ति का उपयोग करें $1 + 2 + \dots + 50$,
        तब $1 + 2 + \dots + 200$.
  \item की गणना करें \omterm{def:g1:addition:def}{जोड़} पहले सौ में से \emph{\omterm{def:g3:numbers:evenodd}{यहां तक की}} संख्याएँ,
        $2 + 4 + 6 + \dots + 200$. (प्रत्येक पद की तुलना एक पद से करें
        गॉस का \omterm{def:g1:addition:def}{जोड़}: किसी नए युग्म की आवश्यकता नहीं है।)
  \item गणना करना $101 + 102 + \dots + 200$, दो का उपयोग करना रकम आप
        पहले से ही पता है और एक घटाव.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- \omterm{def:g3:shapes:shapes}{त्रिकोण}ीय संख्याएँ everywhere.}
कुल $1$, $3$, $6$, $10$, $15, \dots $ की रकम
$1 + 2 + \dots + n $ कहा जाता है \emph{त्रिकोणीय
नंबर}\index{त्रिकोणीय
नंबर}, क्योंकि वे वस्तुओं को गिनते हैं
एक \omterm{def:g3:shapes:shapes}{त्रिकोण} में रखा गया.
\begin{enumerate}[resume]
  \item आठ मित्र मिलते हैं, और प्रत्येक एक ठीक एक बार हाथ मिलाता है
        एक दूसरे के साथ. बताएं कि संख्या क्यों है?
        हाथ मिलाना है $7 + 6 + 5 + 4 + 3 + 2 + 1$, और इसकी गणना करें।
  \item एक पंसारी एक \omterm{def:g3:shapes:shapes}{त्रिकोण}ीय बनाता है \omterm{def:g5:solids:def}{पिरामिड} डिब्बे का: $12$ डिब्बे में
        निचली पंक्ति, $11$ अगले में, और इसी तरह एकल तक
        शीर्ष पर कर सकते हैं. उसे कितने डिब्बे चाहिए? (की युक्ति का प्रयोग करें
        प्रश्न 7.)
  \item खींचना \omterm{def:g1:addition:def}{जोड़} $1 + 2 + 3 + 4$ चौकों की सीढ़ी के रूप में:
        पहले कॉलम में एक वर्ग, दूसरे में दो, तीन
        तीसरे में, चौथे में चार. अपने चित्र पर दिखाएँ
        वह \emph{दो} ऐसी सीढ़ियाँ, उनमें से एक उलटी हो गई
        नीचे, बिल्कुल एक आयत में एक साथ फिट हों $4$ की पंक्तियाँ
        $5$ वर्ग --- और समझाएं कि यह दो-पंक्ति वाली चाल क्यों है
        प्रश्न 7 में लिखित के स्थान पर खींचा गया है।
  \item एक घड़ी बजती है $1$ समय एक बजे, $2$ दो बार
        बजे, \dots, $12$ कई बार बारह बजे. कितने
        क्या हड़ताल बारह घंटों से अधिक है? और कुल मिलाकर
        दिन?
  \item गॉस के लिए अंतिम: $5\,050$ सेकंड --- क्या वह अधिक है या
        एक घंटे से भी कम और ए \omterm{def:g3:division:half}{आधा}? इसे घंटों में बदलें,
        यूक्लिडियन डिवीजन का उपयोग करते हुए मिनट और सेकंड
        (\cref{thm:g6:wholes:division}) दो बार।
\end{enumerate}
\end{problem}
