\chapter{रेखाएं, वृत्त और कोण}\label{ch:g6:lines}

ज्यामिति एक रूलर, एक सेट स्क्वायर और एक कम्पास से शुरू होती है। यह अध्याय
शब्दावली को ठीक करता है --- अंक, \omterm{def:g6:lines:objects}{खंड}ों, किरणें, रेखाएं, वृत्त ---
रेखाओं की दो वि\omterm{def:g3:division:remainder}{शेष} स्थितियों का परिचय देता है (\omterm{def:g6:lines:perp}{समानांतर} और
\omterm{def:g6:lines:perp}{सीधा}), और कोणों को मापना और बनाना सिखाता है।

\section{बिंदु, खंड, रेखाएँ}

\begin{definition}[बुनियादी वस्तुएँ]\label{def:g6:lines:objects}
दो अलग-अलग बिंदुओं के माध्यम से $A $ और $ B $ उत्तीर्ण:
\begin{itemize}
  \item  \emph{खंड}\index{खंड} $[AB]$: पंक्ति का भाग
        के बीच $ A $ और $ B $ (इसकी लंबाई है, लिखा है $ AB $);
  \item  \emph{रे} $[AB)$: से प्रारंभ होता है $ A $, के माध्यम से जाता है $ B $ और
        सदैव जारी रहता है;
  \item  \emph{रेखा}\index{रेखा} $(AB)$: दोनों पर हमेशा के लिए विस्तारित होता है
        पक्ष. दो बिंदु बिल्कुल एक रेखा निर्धारित करते हैं।
\end{itemize}
एक ही रेखा पर स्थित बिन्दु कहलाते हैं \emph{गठबंधन}.
\end{definition}

\section{समानांतर और लंबवत रेखाएँ}

\begin{definition}[लम्बवत्, समानान्तर]\label{def:g6:lines:perp}
दो पंक्तियाँ हैं \emph{सीधा}\index{सीधा} जब वे
ए पर पार करें \omterm{def:g3:shapes:rightangle}{समकोण}; हम लिखते हैं $ d_1 \perp d_2$ और चिन्हित करें \omterm{def:g3:shapes:rightangle}{समकोण} एक छोटे वर्ग के साथ. दो पंक्तियाँ हैं
\emph{समानांतर}\index{समानांतर} जब वे कभी नहीं मिलते, चाहे कितनी भी दूर हों
वे विस्तारित हैं; हम लिखते हैं $ d_1 \parallel d_2$.
\end{definition}

समानताएं: एक ही दिशा, कोई क्रॉसिंग पॉइंट नहीं।}
\end{omfigure}

\begin{proposition}[दो उपयोगी तथ्य]\label{prop:g6:lines:facts}
\begin{enumerate}
  \item यदि दो रेखाएं दोनों हैं \omterm{def:g6:lines:perp}{सीधा} तीसरी पंक्ति में, वे हैं
        \omterm{def:g6:lines:perp}{समानांतर} एक दूसरे से।
  \item यदि दो पंक्तियाँ हैं \omterm{def:g6:lines:perp}{समानांतर}, हर पंक्ति \omterm{def:g6:lines:perp}{सीधा} एक को है
        \omterm{def:g6:lines:perp}{सीधा} अन्य के लिए।
\end{enumerate}
\end{proposition}
\admitted

\begin{method}[सेट स्क्वायर के साथ ड्राइंग]\label{met:g6:lines:setsquare}
खींचने के लिए \omterm{def:g6:lines:perp}{सीधा} एक पंक्ति के लिए $ d $ एक बिंदु के माध्यम से $ P $:
\begin{enumerate}
  \item का एक किनारा रखें \omterm{def:g3:shapes:rightangle}{समकोण} सेट स्क्वायर के साथ $ d $;
  \item सेट स्क्वायर को साथ में स्लाइड करें $ d $ जब तक उसका दूसरा किनारा न पहुँच जाए
        $ P $;
  \item उस किनारे पर \omterm{def:g6:lines:objects}{रेखा} खींचें, और चिह्नित करें \omterm{def:g3:shapes:rightangle}{समकोण}.
\end{enumerate}
एक के लिए \omterm{def:g6:lines:perp}{समानांतर} के माध्यम से $ P $: चित्रित करो \omterm{def:g6:lines:perp}{सीधा} को $ d $, फिर
\omterm{def:g6:lines:perp}{सीधा} को \emph{वह} के माध्यम से \omterm{def:g6:lines:objects}{रेखा} $ P $
(\cref{prop:g6:lines:facts}).
\end{method}

\section{मंडलियां}

\begin{definition}[घेरा]\label{def:g6:lines:circle}
 \emph{घेरा}\index{घेरा} केंद्र का $ O $ और \omterm{def:g3:shapes:shapes}{RADIUS} $ r $ सेट है
दूरी पर स्थित सभी बिंदुओं की बिल्कुल सटीक गणना $ r $ से $ O $. ए \omterm{def:g6:lines:objects}{खंड} से
वृत्त का केंद्र एक है \emph{\omterm{def:g3:shapes:shapes}{RADIUS}}; ए \omterm{def:g6:lines:objects}{खंड} दो बिंदुओं को जोड़ना
केंद्र से गुजरने वाले वृत्त का एक भाग है \emph{\omterm{def:g4:geometry:circle}{व्यास}} ---इसकी लम्बाई है
$2r $; ए \omterm{def:g6:lines:objects}{खंड} वृत्त के दो बिंदुओं को मिलाना एक है \emph{तार}.
\end{definition}

``केंद्र का वृत्त खींचिए $ O $ के माध्यम से गुजरते हुए $ A $'': कम्पास खोलें

इस वृत्त का बिंदु समान दूरी पर है $ O $ जैसा $ A $.
\end{example}

\section{एंगल्स}

\begin{definition}[कोण]\label{def:g6:lines:angle}
दो किरणें $[AB)$ और $[AC)$ एक ही शुरुआती बिंदु के साथ $ A $ इससे
\emph{कोण}\index{कोण} $\widehat{BAC}$; मुद्दा $ A $ इसका है
\emph{शिखर} (हमेशा मध्य अक्षर!) कोणों को मापा जाता है
\emph{डिग्री} ($^\circ $), से $0^\circ $ को $360^\circ $ पूर्ण के लिए
बारी. एक कोण है:
\begin{itemize}
  \item \emph{सही} यदि यह मापता है $90^\circ $;
  \item \emph{तीव्र} यदि इसका माप इससे कम है $90^\circ $;
  \item \emph{कुंठित} यदि यह बीच में मापता है $90^\circ $ और
        $180^\circ $;
  \item \emph{\omterm{def:g6:lines:perp}{सीधा}} यदि यह मापता है $180^\circ $ (दो किरणें बनती हैं
        एक पंक्ति).
\end{itemize}
\end{definition}

\begin{method}[चाँदे से कोण मापना]\label{met:g6:lines:protractor}
\begin{enumerate}
  \item चांदा के केंद्र को ठीक पर रखें शिखर का
        कोण;
  \item इसे संरेखित करें $0^\circ $ कोण के एक तरफ से \omterm{def:g6:lines:objects}{रेखा};
  \item दूसरी ओर से क्रॉस किए गए ग्रेजुएशन को पढ़ें --- का उपयोग करके
        वह पैमाना जहां से शुरू होता है $0$ संरेखित पक्ष पर;
  \item आँख से विवेक की जाँच करें: एक न्यून कोण को इससे कम पढ़ना चाहिए
        $90^\circ $, एक और कुंठित।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g6:lines:estimate}
मापने से पहले अनुमान लगा लें! \omterm{def:g5:solids:def}{कोने} से थोड़ा अधिक खुला कोण
कागज का एक टुकड़ा थोड़ा सा खत्म हो गया है $90^\circ $; \omterm{def:g3:division:half}{आधा} ए \omterm{def:g3:shapes:rightangle}{समकोण} है
$45^\circ $; एक का एक तिहाई \omterm{def:g3:shapes:rightangle}{समकोण} है $30^\circ $. यदि आपका चांदा
कहते हैं $150^\circ $ एक ऐसे कोण के लिए जो तीव्र दिखता है, आप गलत पढ़ते हैं
पैमाना: सही माप है $180^\circ - 150^\circ = 30^\circ $.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g6:lines:1}
तीन बिंदु बनाएं $ A $, $ B $, $ C $ संरेखित नहीं. विभिन्न रंगों में ड्रा करें:
द \omterm{def:g6:lines:objects}{खंड} $[AB]$, किरण $[CA)$, \omterm{def:g6:lines:objects}{रेखा} $(BC)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:2}
सच या झूठ? ``$[AB]$ और $[BA]$ वही हैं \omterm{def:g6:lines:objects}{खंड}.''
``$[AB)$ और $[BA)$ एक ही किरण हैं.'' ``$(AB)$ और $(BA)$ हैं
एक ही पंक्ति।'' प्रत्येक उत्तर स्पष्ट करें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:3}
एक \omterm{def:g6:lines:objects}{रेखा} खींचो $ d $ और एक बिंदु $ P $ पर नहीं $ d $. सेट के साथ निर्माण करें
वर्ग: द \omterm{def:g6:lines:perp}{सीधा} को $ d $ के माध्यम से $ P $, फिर \omterm{def:g6:lines:perp}{समानांतर} को $ d $
के माध्यम से $ P $. अपने कदमों का वर्णन करें.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:4}
पंक्तियां $ a $ और $ b $ दोनों \omterm{def:g6:lines:perp}{सीधा} एक पंक्ति के लिए $ c $, और एक चौथाई
लाइन $ e $ है \omterm{def:g6:lines:perp}{सीधा} को $ a $. आप इसके बारे में क्या कह सकते हैं $ a $ और $ b $?
के बारे में $ e $ और $ c $? के साथ औचित्य सिद्ध करें \cref{prop:g6:lines:facts}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:5}
केंद्र का एक वृत्त बनाएं $ O $ साथ \omterm{def:g3:shapes:shapes}{RADIUS} $3$ सेमी। एक बिंदु रखें $ M $ पर
वृत्त, एक बिंदु $ N $ अंदर, एक बिंदु $ P $ बाहर। आप क्या कह सकते हैं
दूरियों के बारे में $ OM $, $ ON $, $ OP $ के साथ तुलना $3$ सेमी?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:6}
एक वृत्त है \omterm{def:g4:geometry:circle}{व्यास} $9$ सेमी। यह क्या है \omterm{def:g3:shapes:shapes}{RADIUS}? दूसरे के पास है \omterm{def:g3:shapes:shapes}{RADIUS}
$2.6$ सेमी: यह क्या है \omterm{def:g4:geometry:circle}{व्यास}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:7}
अंकित कोण को तीन अक्षरों में नाम दें, फिर उसका वर्गीकरण करें (न्यून, दाएँ,
कुंठित, \omterm{def:g6:lines:perp}{सीधा}): का एक कोण $72^\circ $ पर शिखर $ R $ किरणों के बीच
की ओर $ S $ और $ T $; का एक कोण $148^\circ $ पर शिखर $ B $ किरणों के बीच
की ओर $ A $ और $ C $; का एक कोण $90^\circ $ पर शिखर $ O $ किरणों के बीच
की ओर $ M $ और $ N $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:8}
अनुमान लगाएं, फिर चांदे से तीनों कोणों को मापें
त्रिभुज आप स्वयं बनाएं। तीन उपाय \omterm{def:g1:addition:def}{जोड़}ें: आप क्या पाते हैं?
(अपना उत्तर अपने पास रखें \cref{ch:g7:triangles}.)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:9}
एक कोण बनाएं $\widehat{xOy}$ का $65^\circ $ एक प्रोट्रैक्टर के साथ, फिर एक
का कोण $130^\circ $. आप पहले से दूसरा कैसे प्राप्त कर सकते हैं?
चांदे के बिना?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:lines:10}
दो गांव $ A $ और $ B $ मानचित्र पर खींचे गए हैं। वो बिंदु कहां हैं
पर हैं $3$ सेमी से $ A $ \emph{और} पर $2$ सेमी से $ B $? एक चित्र बनाएं
दिखा रहा है कि ऐसे कितने बिंदु हो सकते हैं ($0$, $1$ या $2$ पर निर्भर करता है
दूरी $ AB $).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:lines:11}
एक घड़ी 3 बजे दिखाती है: दोनों सुइयों के बीच का कोण कितना है? वही
5 बजे प्रश्न, और 6 बजे प्रश्न। (एक पूर्ण मोड़ है $360^\circ $
के लिए $12$ घंटे।)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g6:lines:12}
चित्रित करो \omterm{def:g6:lines:objects}{खंड} $[AB]$ का $6$ सेमी। बिंदु का निर्माण करें $ C $ ऐसा है कि
$ AC = BC = 6$ सेमी, केवल कंपास का उपयोग करके, और कोण मापें
$\widehat{CAB}$. आपने कौन सा त्रिभुज बनाया और आप क्या बनाते हैं?
इसके कोणों के बारे में अनुमान?
\end{exercise}

\section{समस्या: घड़ी के मुख की ज्यामिति}

\begin{problem}[{सप्ताहांत समस्या --- कोण a के भिन्नों के रूप में बारी: उन्हें घड़ी पर पढ़ना, उन क्षणों का शिकार करना जब सुईयाँ मिलें, और एक दिन को पाई में काटें}]
\label{pb:g6:lines:1}
घड़ी एक चाँदा है जो समय बताता है: इसका चेहरा पूर्ण है
की बारी $360^\circ $, बारह घंटे के अंकों को बारह में काटें
समान कोण. \Cref{exo:g6:lines:11} हाथों को मापा
3 बजे और 6 बजे; यह समस्या संपूर्ण सिद्धांत का निर्माण करती है
--- उस समय सहित जब \emph{कोई भी नहीं} हाथ एक निशान की ओर इशारा करता है
--- एक प्रश्न का उत्तर कुछ वयस्क सही देते हैं (``कितनी बार ऐसा करते हैं
दो हाथ बिल्कुल एक दूसरे के ऊपर बैठते हैं?''), और समाप्त होता है
पूरे दिन को पाई चार्ट में काटना।

\medskip
\noindent\textbf{Part I --- भिन्न of a turn.}
\begin{enumerate}
  \item एक पूर्ण मोड़ उपाय $360^\circ $. डिग्री कितनी होती है
        \omterm{def:g3:division:half}{आधा} एक मोड़, ए \omterm{def:g3:division:half}{तिमाही} एक मोड़ का, एक मोड़ का बारहवां?
  \item बारह घंटे के निशान ने घड़ी काट दी चेहरा बारह में
        केंद्र पर समान कोण. दो के बीच कितनी डिग्री
        पड़ोसी निशान? का उत्तर पुनर्प्राप्त करें
        \cref{exo:g6:lines:11} 3 बजे तक अंक गिनकर।
  \item 1 बजे, पर सुइयों के बीच का कोण बताएं
        4 बजे और 7 बजे. (सात बजे हाथ
        अलग करें चेहरा दो कोणों में; ``बीच का कोण
        हाथ'' का हमेशा मतलब होता है \emph{छोटे} एक।)
  \item मिनट की सुई पूरा चक्कर लगाती है $60$ मिनट। कैसे
        यह प्रति मिनट कितने डिग्री तक घूमती है?
  \item घंटे की सुई एक निशान से दूसरे निशान तक यात्रा करती है ---
        $30^\circ $ --- में $60$ मिनट। कितनी डिग्री होती है
        \emph{यह} प्रति मिनट स्वीप? (ए \omterm{def:g6:decimals:places}{दशमलव संख्या},
        \cref{ch:g6:decimals}.)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Times when nothing points at a
mark.}
\begin{enumerate}[resume]
  \item 3:30 बजे, मिनट की सुई इंगित करती है $6$. बिल्कुल कहाँ
        क्या घंटे की सुई है? से प्रत्येक हाथ के कोण की गणना करें
        $12$ (घड़ी की दिशा में मापते हुए), और बीच का कोण निकालें
        हाथ.
  \item 6:30 बजे कई लोग अनुमान लगाते हैं कि हाथ प्रत्येक के ऊपर हैं
        अन्य. पहले अनुमान लगाएं, फिर प्रश्न के अनुसार कोण की गणना करें
        6. कौन सही था?
  \item 9:15 बजे हाथों के बीच के कोण की गणना करें।
  \item स्पष्ट करें कि, 1:00 और 1:10 के बीच, दोनों क्यों
        हाथ बिल्कुल एक दूसरे के ऊपर होने चाहिए: कहाँ है
        1:00 बजे घंटे की सुई के सापेक्ष मिनट की सुई, और कहाँ
        1:10 पर? कौन सा हाथ तेजी से चलता है, और वह क्यों व्यवस्थित होता है?
        यह?
  \item बारह घंटे में दोनों हाथ कितनी बार बैठते हैं
        बिल्कुल एक दूसरे के ऊपर? (एक मीटिंग होती है
        क्रमिक घंटों के बीच प्रत्येक खिंचाव --- एक के साथ
        अपवाद: 11 और 12 के बीच क्या होता है? सूचीबद्ध करें
        अनुमानित बैठक समय और गिनती।) एक में कितनी बार
        पूरा दिन?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Angles around a point: pie
charts.}
\begin{enumerate}[resume]
  \item घड़ी के बारह \omterm{def:g4:measure:area}{क्षेत्र} चेहरा एक साथ भरें
        चेहरा: उनके कोणों का योग होता है $360^\circ $. स्पष्ट करें क्यों
        ये सच है \emph{कोई} कोणों का संग्रह
        साझा करें ए शिखर और इसके चारों ओर एक पूरा चक्कर भरें, नहीं के साथ
        ओवरलैप और कोई गैप नहीं.
  \item ज़ो अपना दिन रिकॉर्ड करती है $24$ घंटे: नींद $9$ एच, स्कूल
        $6$ ज, खेलो $3$ एच, भोजन $2$ ज, बाकी सब कुछ $4$ एच।
        वह चाहती है \emph{पाई चार्ट}: एक डिस्क जहां प्रत्येक गतिविधि
        समय के समानुपाती कोणों वाला एक त्रिज्यखंड मिलता है।
        क्या \omterm{def:g6:fractions:def}{अंश} प्रत्येक गतिविधि दिन की कितनी है, और कितनी है
        क्या उसके \omterm{def:g4:measure:area}{क्षेत्र} को डिग्रियाँ मिलती हैं? पाँच कोणों के योग की जाँच करें
        को $360^\circ $.
  \item ज़ो का पाई चार्ट कैसे बनाएं, इसका चरण दर चरण वर्णन करें
        कम्पास और प्रोट्रैक्टर
        (\cref{met:g6:lines:protractor}): जहां पहला \omterm{def:g3:shapes:shapes}{RADIUS}
        जाता है, और कैसे प्रत्येक नया \omterm{def:g4:measure:area}{क्षेत्र} पिछले से शुरू होता है
        एक समाप्त होता है.
  \item एक दिन के दूसरे पाई चार्ट में, एक सेक्टर को मापा जाता है
        बिलकुल $90^\circ $. क्या \omterm{def:g6:fractions:def}{अंश} डिस्क का वह हिस्सा है,
        और यह कितने घंटों का प्रतिनिधित्व करता है?
  \item समापन, घड़ी पर वापस: में $24$ घंटे, कितने
        क्या घंटे की सुई पूरे चक्कर लगाती है? मिनट की सुई? द
        सेकेंड हैंड? (इनमें से एक उत्तर एक हजार से अधिक है।)
\end{enumerate}
\end{problem}
