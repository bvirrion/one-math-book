\chapter{Fractions~: premiers pas}\label{ch:g6:fractions}

Partager équitablement trois pizzas entre quatre personnes donne à
chacun\,\dots\ pas un nombre entier de pizzas. Les \omterm{def:g6:fractions:def}{fractions} sont les
nombres inventés pour le partage. Ce chapitre construit le tableau~: une
\omterm{def:g6:fractions:def}{fraction} comme part d'un tout, comme point sur la droite graduée, et
comme résultat exact d'une division.

\section{Ce qu'une fraction signifie}

\begin{definition}[Fraction]\label{def:g6:fractions:def}
On découpe une unité en $b$ parts égales, et on en prend $a$~: la
quantité obtenue est la \emph{fraction}\index{fraction}
\[
\frac{a}{b}
\qquad
\begin{array}{l}
a \text{~: le \emph{numérateur} --- combien de parts on prend~;}\\
b \text{~: le \emph{dénominateur} --- en combien de parts l'unité est découpée.}
\end{array}
\]
\end{definition}

\begin{example}\label{ex:g6:fractions:morethanone}
Une \omterm{def:g6:fractions:def}{fraction} peut être plus grande que $1$~: $\frac74$ signifie sept
\omterm{def:g3:division:half}{quarts} --- une unité entière (quatre \omterm{def:g3:division:half}{quarts}) et trois \omterm{def:g3:division:half}{quarts} de
plus~:
\[
\frac74 = 1 + \frac34 .
\]
\end{example}

\begin{proposition}[Fractions sur la droite graduée]\label{prop:g6:fractions:line}
Pour placer $\frac ab$ sur la droite graduée, on découpe chaque
intervalle unitaire en $b$ parts égales et on compte $a$ parts depuis
$0$.
\end{proposition}

\begin{theorem}[Une fraction est une division]\label{thm:g6:fractions:quotient}
La \omterm{def:g6:fractions:def}{fraction} $\frac ab$ est le nombre qui, multiplié par $b$, donne
$a$~:
\[
\frac ab \times b = a .
\]
Autrement dit, $\frac ab$ est le \omterm{def:g3:division:remainder}{quotient} exact de $a$ par $b$ ---
même quand la division «~ne tombe pas juste~».
\end{theorem}
\admitted

\begin{example}\label{ex:g6:fractions:pizzas}
Trois pizzas pour quatre personnes~: chacune obtient $\frac34$ d'une
pizza. Vérification avec le théorème~: $4$ parts de $\frac34$ font
$\frac34 \times 4 = 3$ pizzas. Certaines \omterm{def:g6:fractions:def}{fractions} sont des nombres
décimaux ($\frac34 = 0.75$), d'autres non~: $\frac13 = 0.333\dots$ ne
s'arrête jamais. La \omterm{def:g6:fractions:def}{fraction} est la valeur \emph{exacte}~; $0.33$
n'est qu'une approximation.
\end{example}

\section{Fractions égales}

\begin{proposition}[Fractions équivalentes]\label{prop:g6:fractions:equal}
Une \omterm{def:g6:fractions:def}{fraction} ne change pas lorsque son \omterm{def:g4:fractions:def}{numérateur} et son \omterm{def:g4:fractions:def}{dénominateur}
sont tous deux multipliés (ou divisés) par le même nombre non nul~:
\[
\frac{a}{b} = \frac{a \times k}{b \times k} .
\]
\end{proposition}

\begin{proof}[Idée de démonstration]
Découper chaque part du partage en $k$ plus petites pièces multiplie
à la fois le nombre de pièces prises et le nombre total de pièces par
$k$, mais la quantité prise \omterm{def:g3:division:remainder}{reste} la même. Un dessin le dit le mieux~:
\end{proof}

\begin{method}[Simplifier une fraction]\label{met:g6:fractions:simplify}
Chercher un nombre divisant le \omterm{def:g4:fractions:def}{numérateur} \emph{et} le \omterm{def:g4:fractions:def}{dénominateur},
et diviser les deux. Répéter jusqu'à ce qu'il ne reste plus de
\omterm{def:g5:division:multiple}{diviseur} commun~:
\[
\frac{12}{18} = \frac{12 \div 2}{18 \div 2} = \frac{6}{9}
= \frac{6 \div 3}{9 \div 3} = \frac{2}{3}.
\]
\end{method}

\section{Prendre une fraction d'une quantité}

\begin{method}[Fraction d'une quantité]\label{met:g6:fractions:ofquantity}
Pour calculer $\frac ab$ d'une quantité, diviser la quantité par $b$
(une part), puis multiplier par $a$ (le nombre de parts)~:
\[
\frac ab \text{ de } Q = (Q \div b) \times a .
\]
(Multiplier d'abord par $a$ puis diviser par $b$ donne le même
résultat --- choisir l'ordre qui rend les nombres plus jolis.)
\end{method}

\begin{example}\label{ex:g6:fractions:ofquantity}
Calculer $\frac35$ de $40$ euros, étape par étape~:
\begin{enumerate}
  \item un cinquième de $40$~: $40 \div 5 = 8$~;
  \item trois cinquièmes~: $8 \times 3 = 24$.
\end{enumerate}
Donc $\frac35$ de $40$ euros, c'est $24$ euros. Vérif.~: les
$\frac25$ restants font $16$ euros, et $24 + 16 = 40$.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g6:fractions:1}
Pour chaque description d'image, écrire la \omterm{def:g6:fractions:def}{fraction}~: un gâteau
découpé en $8$ parts, $3$ mangées~; une tablette de chocolat de
$12$ carrés, $7$ mangés~; une tarte découpée en $6$, les $6$ mangées.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:2}
Dessiner une barre découpée en $5$ parts égales et colorier $\frac45$.
$\frac45$ est-elle plus petite ou plus grande que $1$~? Et
$\frac65$~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:3}
Placer sur une droite graduée en tiers~: $\frac13$~; \ $\frac53$~;
\ $2$~; \ $\frac73$. Lesquels de ces nombres sont à égale distance de
$2$~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:4}
Écrire chaque \omterm{def:g6:fractions:def}{fraction} comme un entier plus une \omterm{def:g6:fractions:def}{fraction} plus petite
que $1$ (comme dans $\frac74 = 1 + \frac34$)~:
\[
\frac{9}{4}, \qquad \frac{13}{5}, \qquad \frac{12}{6}, \qquad
\frac{25}{8} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:5}
Compléter pour que les \omterm{def:g6:fractions:def}{fractions} soient égales~:
\[
\frac{2}{3} = \frac{?}{12}, \qquad
\frac{15}{20} = \frac{3}{?}, \qquad
\frac{7}{10} = \frac{21}{?}, \qquad
\frac{?}{6} = \frac{10}{12}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:6}
Simplifier au maximum~: $\dfrac{6}{8}$~; \quad $\dfrac{15}{25}$~;
\quad $\dfrac{18}{24}$~; \quad $\dfrac{35}{7}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:7}
Calculer~: $\frac12$ de $86$~; \quad $\frac34$ de $60$~; \quad
$\frac25$ de $35$~; \quad $\frac{7}{10}$ de $250$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:8}
Laquelle est la plus grande~: $\frac12$ ou $\frac13$ (du même
gâteau)~? $\frac25$ ou $\frac35$~? $\frac34$ ou $\frac78$ (écrire
les deux avec le \omterm{def:g4:fractions:def}{dénominateur} $8$)~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:fractions:9}
Donner la valeur décimale des \omterm{def:g6:fractions:def}{fractions} qui en ont une, et dire
laquelle n'en a pas~: $\frac12$~; \ $\frac34$~; \ $\frac13$~; \
$\frac{7}{10}$~; \ $\frac{9}{4}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:fractions:10}
Une classe a $28$ élèves. Les trois \omterm{def:g3:division:half}{quarts} viennent à l'école en bus,
et le reste à pied. Combien d'élèves viennent à pied~? Résoudre
étape par étape, et vérifier que les tailles des deux groupes
additionnent $28$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:fractions:11}
Léo a dépensé $\frac23$ de ses économies pour un jeu qui a coûté
$18$ euros. Combien d'argent Léo avait-il avant~? (Un tiers de ses
économies est la \omterm{def:g3:division:half}{moitié} de $18$\,\dots\ réfléchir pourquoi, ou trouver
d'abord la taille d'une part.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:fractions:12}
Un réservoir est rempli aux $\frac58$ de sa capacité. Après avoir
utilisé $10$ litres, il est à \omterm{def:g3:division:half}{moitié} plein. Quelle est la capacité du
réservoir~? (Quelle \omterm{def:g6:fractions:def}{fraction} du réservoir les $10$ litres
représentent-ils~?)
\end{exercise}
