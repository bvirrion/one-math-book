\chapter{Les nombres entiers}\label{ch:g6:wholes}

Tout en mathématiques commence par le comptage. Ce chapitre revoit
comment les nombres entiers s'écrivent, se comparent et se combinent ---
et regarde de près la division, l'opération qui donne à la fois un
quotient \emph{et} un reste.

\section{Écrire et comparer des entiers}

\begin{definition}[Valeur de position]\label{def:g6:wholes:place}
Notre façon d'écrire les nombres utilise dix chiffres ($0$ à $9$), et
la \emph{valeur d'un chiffre dépend de sa place}\index{valeur de
position}~: dans $5\,208$, le chiffre $5$ compte les milliers, le $2$
les centaines, le $0$ les dizaines (il n'y en a pas) et le $8$ les
unités~:
\[
5\,208 = 5 \times 1000 + 2 \times 100 + 0 \times 10 + 8 .
\]
\end{definition}

\begin{method}[Comparer deux entiers]\label{met:g6:wholes:compare}
\begin{enumerate}
  \item Le nombre avec le plus de chiffres est le plus grand
        ($1\,203 > 989$).
  \item S'ils ont le même nombre de chiffres, on compare chiffre par
        chiffre depuis la \emph{gauche}~; la première différence
        décide~: $6\,742 > 6\,715$ car à la place des dizaines,
        $4 > 1$.
\end{enumerate}
Les symboles sont $<$ («~plus petit que~») et $>$ («~plus grand
que~»)~; le petit côté du symbole pointe vers le petit nombre.
\end{method}

\begin{example}\label{ex:g6:wholes:order}
Ranger du plus petit au plus grand~: $908$~; $89$~; $1\,001$~; $980$.
D'abord par le nombre de chiffres~: $89$ (deux chiffres) vient en
premier, $1\,001$ (quatre chiffres) en dernier. Entre $908$ et
$980$~: même chiffre des centaines $9$, puis dizaines $0 < 8$, donc
$908 < 980$. Réponse finale~:
\[
89 < 908 < 980 < 1\,001 .
\]
\end{example}

\section{Addition, soustraction, multiplication}

\begin{example}[Calculs en colonnes]\label{ex:g6:wholes:columns}
Pour additionner ou soustraire, on aligne les unités sous les unités,
les dizaines sous les dizaines, et on travaille de droite à gauche, en
retenant au besoin~:
\[
\begin{array}{r}
4\,5\,7 \\
+\ 2\,8\,6 \\
\hline
7\,4\,3
\end{array}
\qquad\qquad
\begin{array}{r}
6\,0\,3 \\
-\ 1\,4\,7 \\
\hline
4\,5\,6
\end{array}
\]
Dans l'addition~: $7 + 6 = 13$, on écrit $3$, on retient $1$~; puis
$5 + 8 + 1 = 14$, on écrit $4$, on retient $1$~; puis $4 + 2 + 1 = 7$.
Vérifier la soustraction en additionnant~: $456 + 147 = 603$.
\end{example}

\begin{remark}[Vocabulaire]
Le résultat d'une addition est une \emph{somme}\index{somme}~; d'une
soustraction, une \emph{différence}\index{différence}~; d'une
multiplication, un \emph{produit}\index{produit}~; d'une division, un
\emph{quotient}\index{quotient}. Les nombres combinés sont les
\emph{termes} (pour $+$, $-$) ou \emph{facteurs} (pour $\times$).
\end{remark}

\begin{example}[Calcul malin]\label{ex:g6:wholes:smart}
Réordonner les termes ou facteurs économise souvent du travail~:
\[
17 + 58 + 3 = (17 + 3) + 58 = 20 + 58 = 78 ,
\]
\[
25 \times 17 \times 4 = (25 \times 4) \times 17 = 100 \times 17 = 1700 .
\]
\end{example}

\section{Division}

\begin{theorem}[Division euclidienne]\label{thm:g6:wholes:division}
Étant donnés deux entiers $a$ (le \emph{dividende}) et $b \neq 0$ (le
\emph{diviseur}), il y a exactement une façon d'écrire
\[
a = b \times q + r
\qquad\text{avec}\qquad
0 \leq r < b .
\]
Le nombre $q$ est le \emph{quotient}\index{quotient} et $r$ le
\emph{reste}\index{reste} de la division de $a$ par $b$.
\end{theorem}
\admitted

\begin{example}\label{ex:g6:wholes:division}
Diviser $137$ par $4$, étape par étape.
\begin{enumerate}
  \item Combien de fois $4$ rentre-t-il dans $13$~? Trois fois
        ($4 \times 3 = 12$), reste $13 - 12 = 1$.
  \item On abaisse le $7$~: combien de fois $4$ rentre-t-il dans
        $17$~? Quatre fois ($4 \times 4 = 16$), reste $1$.
\end{enumerate}
Donc $q = 34$ et $r = 1$~:
\[
137 = 4 \times 34 + 1, \qquad\text{et bien } 0 \leq 1 < 4 .
\]
Vérification~: $4 \times 34 = 136$, et $136 + 1 = 137$.
\end{example}

\begin{definition}[Divisible]\label{def:g6:wholes:divisible}
Quand le reste est $0$, on dit que $a$ est
\emph{divisible}\index{divisible} par $b$~: par exemple $135$ est
divisible par $5$ car $135 = 5 \times 27 + 0$.
\end{definition}

\begin{method}[Problèmes de division en mots]\label{met:g6:wholes:problems}
Après avoir calculé $a = bq + r$, toujours revenir à la question~:
\begin{enumerate}
  \item «~Combien de boîtes/équipes/pages complètes~?~» --- la
        réponse est le quotient $q$~;
  \item «~Combien reste-t-il~?~» --- la réponse est le reste $r$~;
  \item «~Combien de boîtes faut-il pour tout le monde~?~» --- si
        $r > 0$, une de \emph{plus} que le quotient~: $q + 1$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g6:wholes:bus}
$137$ élèves partent en voyage en bus de $40$ places. Division~:
$137 = 40 \times 3 + 17$. Trois bus transportent $120$ élèves, et
$17$ élèves restent --- il leur faut un bus aussi~! Donc $4$ bus sont
nécessaires, même si le quotient est $3$.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g6:wholes:1}
Écrire en chiffres~: «~trois mille quarante-sept~»~; «~vingt mille
cinq cents~»~; «~cent quatre~». Puis écrire $60\,013$ en lettres.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:2}
Dans le nombre $47\,209$~: quel est le chiffre des centaines~? Celui
des unités~? Que compte le chiffre $4$~? Écrire le nombre comme une
somme de multiples de $1$, $10$, $100$, \dots\ comme dans
\cref{def:g6:wholes:place}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:3}
Recopier et compléter avec $<$ ou $>$~:
\[
507 \;?\; 570, \qquad
1\,099 \;?\; 989, \qquad
23\,456 \;?\; 23\,465, \qquad
9\,999 \;?\; 10\,000 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:4}
Ranger du plus petit au plus grand~: $2\,043$~; $978$~; $2\,304$~;
$2\,034$~; $1\,001$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:5}
Calculer en colonnes~: $348 + 576$~; \quad $805 - 268$~; \quad
$307 \times 26$. Vérifier la soustraction avec une addition.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:6}
Calculer malinement, en regroupant termes ou facteurs~:
\[
38 + 45 + 12 + 5, \qquad
2 \times 83 \times 50, \qquad
4 \times 78 \times 25 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:7}
Pour chaque division, donner le quotient et le reste, et écrire
l'égalité $a = b \times q + r$~:
\[
95 \div 7, \qquad
230 \div 6, \qquad
504 \div 8 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:8}
Sans diviser, expliquer pourquoi l'égalité $173 = 8 \times 20 + 13$
\emph{n'est pas} la division euclidienne de $173$ par $8$, et trouver
le quotient et le reste corrects.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:wholes:9}
Les œufs sont emballés par boîtes de $12$.
\begin{enumerate}
  \item Une ferme a $2\,000$ œufs. Combien de boîtes pleines
        peut-elle remplir, et combien d'œufs restent~?
  \item Une boulangerie a besoin de $150$ œufs. Combien de boîtes
        doit-elle acheter~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:wholes:10}
Aujourd'hui, c'est mardi. Quel jour de la semaine sera-t-ce dans
$100$ jours~? (Les semaines ont $7$ jours~: diviser et penser au
reste.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:wholes:11}
Dans une division par $9$, le quotient est $56$ et le reste est le
plus grand possible. Quel est le dividende~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:wholes:12}
Je suis un nombre à trois chiffres. Mon chiffre des centaines est le
double de mon chiffre des unités, mon chiffre des dizaines est $0$, et
la somme de mes chiffres est $9$. Qui suis-je~? (Raisonner étape par
étape, puis vérifier.)
\end{exercise}
