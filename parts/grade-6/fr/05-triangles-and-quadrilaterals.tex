\chapter{Triangles et quadrilatères}\label{ch:g6:shapes}

Les triangles et les quadrilatères sont les formes de base de la
géométrie. Le but de ce chapitre est de connaître leurs familles
spéciales --- isocèles, équilatéraux, rectangles~; rectangles,
losanges, carrés --- par leurs \emph{propriétés définissantes}, et de
les construire précisément à la règle et au compas.

\section{Triangles}

\begin{definition}[Triangles spéciaux]\label{def:g6:shapes:triangles}
Un triangle $ABC$ a trois sommets et trois côtés. Il est~:
\begin{itemize}
  \item \emph{isocèle}\index{triangle isocèle} en $A$ lorsque
        $AB = AC$ (deux côtés égaux)~;
  \item \emph{équilatéral}\index{triangle équilatéral} lorsque
        $AB = BC = CA$ (trois côtés égaux)~;
  \item \emph{rectangle}\index{triangle rectangle} en $A$ lorsque
        l'angle $\widehat{BAC}$ est un angle droit. Le côté $[BC]$
        opposé à l'angle droit s'appelle l'\emph{hypoténuse}.
\end{itemize}
\end{definition}

\begin{method}[Construire un triangle à partir de ses trois côtés]\label{met:g6:shapes:construct}
Pour construire $ABC$ avec $AB = 5$~cm, $AC = 4$~cm, $BC = 3$~cm~:
\begin{enumerate}
  \item tracer le segment $[AB]$ de $5$~cm à la règle~;
  \item tracer un arc du cercle de centre $A$ et de rayon $4$~cm
        (tous les points à $4$~cm de $A$)~;
  \item tracer un arc du cercle de centre $B$ et de rayon $3$~cm~;
  \item le point $C$ est où les deux arcs se croisent~; le joindre à
        $A$ et $B$.
\end{enumerate}
\end{method}

\section{Quadrilatères}

\begin{definition}[Quadrilatères spéciaux]\label{def:g6:shapes:quadrilaterals}
Un quadrilatère a quatre sommets et quatre côtés. C'est un~:
\begin{itemize}
  \item \emph{rectangle}\index{rectangle} lorsque ses quatre angles
        sont des angles droits~;
  \item \emph{losange}\index{losange} lorsque ses quatre côtés ont la
        même longueur~;
  \item \emph{carré}\index{carré} lorsqu'il est les deux~: quatre
        angles droits \emph{et} quatre côtés égaux.
\end{itemize}
\end{definition}

\begin{proposition}[Propriétés des diagonales]\label{prop:g6:shapes:diagonals}
Dans ces quadrilatères, les deux diagonales se coupent en leur
milieu commun, et de plus~:
\begin{itemize}
  \item dans un rectangle, les diagonales ont la \emph{même
        longueur}~;
  \item dans un losange, les diagonales sont \emph{perpendiculaires}~;
  \item dans un carré, les deux à la fois.
\end{itemize}
\end{proposition}
\admitted

\begin{remark}
Un carré est automatiquement un rectangle (il a quatre angles droits)
et automatiquement un losange (quatre côtés égaux). Donc «~rectangle~»
ne signifie pas «~pas un carré~»~! En mathématiques, la forme plus
spéciale appartient à la famille plus générale.
\end{remark}

\begin{example}[Reconnaître une forme d'après ses propriétés]\label{ex:g6:shapes:recognize}
Un quadrilatère a quatre côtés de longueur $4$~cm et un angle droit.
Qu'est-ce~? Quatre côtés égaux en font un losange. Dans un losange,
les angles opposés sont égaux et les quatre angles additionnent
$360^\circ$~; un angle droit force alors les quatre à être droits
(ce sera prouvé avec les parallélogrammes dans
\cref{ch:g7:central}). Donc la forme est un carré.
\end{example}

\begin{method}[Programmes de construction]\label{met:g6:shapes:program}
Un \emph{programme de construction} est une liste d'instructions que
n'importe qui peut suivre pour reproduire une figure exactement.
Écrire une instruction par ligne, nommer chaque point quand il est
créé, et n'utiliser que~: «~tracer le segment / la droite / le
cercle\,\dots~», «~placer le point où \dots\ se croisent~». Tester
son programme en le suivant littéralement, mot pour mot.
\end{method}

\begin{example}\label{ex:g6:shapes:program}
Programme pour un carré $ABCD$ de côté $4$~cm~:
\begin{enumerate}
  \item Tracer un segment $[AB]$ avec $AB = 4$~cm.
  \item Tracer la perpendiculaire à $(AB)$ en $A$~; dessus, placer
        $D$ à $4$~cm de $A$ (choisir un côté de la droite et s'y
        tenir).
  \item Tracer la perpendiculaire à $(AB)$ en $B$~; dessus, placer
        $C$ à $4$~cm de $B$, du même côté que $D$.
  \item Joindre $C$ et $D$.
\end{enumerate}
Vérifier à la règle~: $DC$ doit aussi mesurer $4$~cm.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g6:shapes:1}
Construire un triangle $ABC$ avec $AB = 6$~cm, $AC = 5$~cm et
$BC = 4$~cm, en suivant \cref{met:g6:shapes:construct}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:2}
Construire un triangle isocèle $DEF$ avec $DE = DF = 5$~cm et
$EF = 3$~cm. Quels deux angles de ce triangle semblent égaux~? Les
mesurer pour vérifier.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:3}
Construire un triangle équilatéral de côté $4$~cm en n'utilisant que
la règle et le compas. Mesurer un de ses angles.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:4}
Classer chaque triangle d'après ses données (isocèle, équilatéral,
rectangle --- plusieurs réponses peuvent s'appliquer)~:
(a) côtés $7$, $7$, $7$~; \
(b) côtés $5$, $5$, $8$~; \
(c) angles $90^\circ$, $45^\circ$, $45^\circ$~; \
(d) angles $60^\circ$, $60^\circ$, $60^\circ$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:5}
Dans un triangle rectangle, quel côté est l'hypoténuse~? Dessiner un
triangle $MNP$ rectangle en $N$, et nommer son hypoténuse. Quel sommet
l'hypoténuse ne touche-t-elle jamais~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:6}
Vrai ou faux~? Justifier chaque réponse en une phrase.
\begin{enumerate}
  \item Tout carré est un rectangle.
  \item Tout rectangle est un carré.
  \item Tout carré est un losange.
  \item Un losange avec un angle droit est un carré.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:7}
Construire un rectangle $ABCD$ avec $AB = 5$~cm et $BC = 3$~cm.
Mesurer ses deux diagonales~: que remarque-t-on~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:8}
Construire un losange de diagonales $6$~cm et $4$~cm. (Utiliser
\cref{prop:g6:shapes:diagonals}~: tracer d'abord les deux diagonales ---
perpendiculaires, se croisant en leurs milieux --- puis joindre les
quatre extrémités.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:shapes:9}
Écrire un programme de construction pour un triangle isocèle de base
$4$~cm et de côtés égaux $6$~cm, dans le style de
\cref{ex:g6:shapes:program}. Échanger avec un camarade~: peut-il le
suivre sans voir la figure~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:shapes:10}
Tracer un cercle de centre $O$ et un de ses diamètres $[AB]$. Placer
un point $M$ quelconque sur le cercle et tracer le triangle $AMB$.
Mesurer l'angle $\widehat{AMB}$. Déplacer $M$ et mesurer encore. Que
conjecture-t-on~? (Ce fait surprenant est prouvé dans
\cref{ch:g8:cosine}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:shapes:11}
Un quadrilatère a des diagonales de même longueur qui se coupent en
leur milieu commun. Déterminer quelle forme ce doit être, en utilisant
\cref{prop:g6:shapes:diagonals}~: rectangle, losange ou carré~?
(Attention~: une seule propriété est donnée.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:shapes:12}
Sur un quadrillage pointé $4 \times 4$, combien de carrés différents
peut-on dessiner dont les coins sont des points du quadrillage --- en
comptant aussi les carrés «~penchés~»~? Dessiner au moins un carré
penché et expliquer pourquoi ses côtés sont égaux.
\end{exercise}
