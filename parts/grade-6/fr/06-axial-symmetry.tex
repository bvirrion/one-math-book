\chapter{La symétrie axiale}\label{ch:g6:symmetry}

Plier une feuille le long d'une droite~: chaque point se pose sur un
autre point, son image miroir. Ce pliage est la \emph{symétrie
axiale} (ou réflexion), la première transformation géométrique du
livre --- la seconde, la symétrie centrale, arrive dans
\cref{ch:g7:central}.

\section{Réflexions}

\begin{definition}[Point symétrique]\label{def:g6:symmetry:def}
Soit $d$ une droite. Le \emph{symétrique}\index{symétrie axiale}
d'un point $M$ par rapport à $d$ est le point $M'$ tel que $d$ est la
médiatrice de $[MM']$~: le segment $[MM']$ est perpendiculaire à $d$,
et $d$ le coupe en son milieu. Un point \emph{sur} $d$ est son propre
symétrique.
\end{definition}

\begin{method}[Construire un point symétrique]\label{met:g6:symmetry:construct}
Pour construire le symétrique $M'$ de $M$ par rapport à la droite
$d$~:
\begin{enumerate}
  \item tracer la perpendiculaire à $d$ passant par $M$ (équerre)~;
        appeler $H$ le point où elle croise $d$~;
  \item mesurer $MH$ au compas ou à la règle~;
  \item placer $M'$ sur la même perpendiculaire, de l'autre côté de
        $d$, avec $HM' = MH$.
\end{enumerate}
Sur papier quadrillé, c'est plus rapide~: compter les carreaux de $M$
à l'axe, et compter le même nombre de l'autre côté.
\end{method}

\begin{example}[Sur un quadrillage]\label{ex:g6:symmetry:grid}
Si l'axe est une ligne verticale du quadrillage et $M$ est à
$3$ carreaux à sa gauche, alors $M'$ est à $3$ carreaux à sa droite,
sur la même ligne. Pour un axe diagonal, le comptage se fait le long
des diagonales --- toujours perpendiculairement à l'axe.
\end{example}

\begin{proposition}[Ce qu'une réflexion conserve]\label{prop:g6:symmetry:preserved}
Le symétrique d'une figure par rapport à une droite est une figure de
la \emph{même forme et de la même taille}~: longueurs, angles,
périmètres et aires sont tous conservés. Le symétrique d'un segment
est un segment, d'un cercle un cercle de même rayon, d'une droite une
droite.
\end{proposition}
\admitted

\begin{remark}
Une chose n'est \emph{pas} conservée~: l'orientation. Un «~b~»
réfléchi devient un «~d~» --- l'écriture en miroir. Si calquer sa
figure et retourner le calque reproduit l'image, la réflexion est
correcte.
\end{remark}

\section{Axes de symétrie d'une figure}

\begin{definition}[Axe de symétrie]\label{def:g6:symmetry:axis}
Une droite $d$ est un \emph{axe de symétrie}\index{axe de symétrie}
d'une figure lorsque la réflexion par rapport à $d$ envoie la figure
exactement sur elle-même --- plier le long de $d$ fait coïncider les
deux moitiés.
\end{definition}

{\small Axes de symétrie de formes familières. Attention au
rectangle~: ses diagonales ne sont \emph{pas} des axes de symétrie
(plier le long d'une diagonale --- les moitiés ne coïncident pas, sauf
si c'est un carré).}
\end{omfigure}

\begin{example}\label{ex:g6:symmetry:shapes}
Compter les axes de symétrie~:
\begin{itemize}
  \item triangle isocèle~: $1$ (par le sommet et le milieu de la
        base)~; triangle équilatéral~: $3$~;
  \item rectangle~: $2$ (les deux «~médianes~», \emph{pas} les
        diagonales)~; losange~: $2$ (ses diagonales~!)~; carré~: $4$~;
  \item cercle~: toute droite passant par le centre --- une infinité.
\end{itemize}
\end{example}

\begin{example}[La symétrie prouve des égalités]\label{ex:g6:symmetry:proves}
Pourquoi les deux angles à la base d'un triangle isocèle sont-ils
égaux~? Plier le triangle le long de son axe de symétrie~: la moitié
gauche se pose exactement sur la moitié droite, donc l'angle de gauche
se pose sur l'angle de droite --- ils doivent avoir la même mesure
(\cref{prop:g6:symmetry:preserved}~: les réflexions conservent les
angles).
\end{example}

\section{Médiatrice}

\begin{definition}[Médiatrice]\label{def:g6:symmetry:bisector}
La \emph{médiatrice}\index{médiatrice} d'un segment $[AB]$ est la
droite perpendiculaire à $[AB]$ passant par son milieu --- c'est l'axe
de symétrie qui échange $A$ et $B$.
\end{definition}

\begin{proposition}[Équidistance]\label{prop:g6:symmetry:bisector}
Un point $M$ est sur la médiatrice de $[AB]$ exactement quand il est
à la même distance de $A$ et de $B$~: $MA = MB$.
\end{proposition}
\admitted

\begin{method}[Construction au compas]\label{met:g6:symmetry:bisector}
Pour tracer la médiatrice de $[AB]$ sans mesurer~:
\begin{enumerate}
  \item ouvrir le compas à plus de la moitié de $AB$~;
  \item tracer deux arcs centrés en $A$, au-dessus et en dessous du
        segment~;
  \item avec la \emph{même} ouverture, tracer deux arcs centrés en
        $B$~;
  \item joindre les deux points de croisement~: cette droite est la
        médiatrice (chaque point de croisement est équidistant de $A$
        et de $B$).
\end{enumerate}
\end{method}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:1}
Recopier sur papier quadrillé~: un axe vertical $d$, un point $A$ à
deux carreaux à sa gauche, un point $B$ à cinq carreaux à sa gauche et
un carreau plus haut. Construire leurs points symétriques $A'$ et
$B'$ par rapport à $d$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:2}
Tracer une droite $d$ et un point $M$ à $3$~cm de $d$ (pas sur
papier quadrillé). Construire le symétrique $M'$ de $M$ par rapport à
$d$ avec équerre et règle, en suivant
\cref{met:g6:symmetry:construct}. Quelle distance sépare $M$ et
$M'$~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:3}
Quelles lettres capitales de l'alphabet, écrites sous leur forme la
plus simple, ont un axe de symétrie vertical (comme A)~? Un horizontal
(comme B)~? Les deux~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:4}
Combien d'axes de symétrie a~: un triangle équilatéral~? un rectangle
(qui n'est pas un carré)~? un losange (pas un carré)~? un carré~? un
cercle~? Dessiner chaque figure avec ses axes.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:5}
Vrai ou faux~? «~Les diagonales d'un rectangle sont des axes de
symétrie du rectangle.~» Expliquer avec un argument de pliage ou un
dessin.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:6}
Un triangle $ABC$ est réfléchi par rapport à une droite $d$ en
$A'B'C'$. Sachant $AB = 4$~cm, $\widehat{BAC} = 70^\circ$ et le
périmètre de $ABC$ est $12$~cm~: donner, sans aucune construction,
$A'B'$, $\widehat{B'A'C'}$ et le périmètre de $A'B'C'$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:7}
Tracer un segment $[AB]$ de $7$~cm et construire sa médiatrice au
compas (\cref{met:g6:symmetry:bisector}). Choisir un point $M$
quelconque dessus et vérifier à la règle que $MA = MB$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:symmetry:8}
Deux arbres se dressent aux points $A$ et $B$ d'un jardin (plat). Où
peut-on planter une fontaine pour qu'elle soit à la même distance des
deux arbres~? Décrire \emph{tous} les emplacements possibles.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:symmetry:9}
Sur papier quadrillé, tracer l'axe $d$ (une diagonale du quadrillage à
$45^\circ$) et une petite figure en L d'un côté. Construire sa
figure symétrique par rapport à $d$. (Réfléchir chaque sommet~: pour un
axe à $45^\circ$, un point à $2$ carreaux à droite de l'axe se pose à
$2$ carreaux au-dessus.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:symmetry:10}
Tracer un cercle de centre $O$, une droite $d$ passant par $O$, et un
point $M$ sur le cercle. Où se trouve le symétrique de $M$ par rapport
à $d$~? Expliquer pourquoi la réflexion envoie le cercle sur
lui-même.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:symmetry:11}
En utilisant la propriété d'équidistance
(\cref{prop:g6:symmetry:bisector}), expliquer pourquoi les deux points
de croisement des arcs dans \cref{met:g6:symmetry:bisector} se
trouvent bien sur la médiatrice de $[AB]$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:symmetry:12}
Une bille de billard au point $A$ doit rebondir sur un mur droit $d$
et atteindre le point $B$ (tous deux du même côté du mur). Réfléchir
$B$ par rapport à $d$ en $B'$, et tracer le segment $[AB']$. Expliquer
pourquoi le meilleur point de rebond est où $[AB']$ croise le mur.
(Idée~: le chemin $A \to P \to B$ a la même longueur que
$A \to P \to B'$.)
\end{exercise}
