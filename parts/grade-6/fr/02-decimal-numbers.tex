\chapter{Les nombres décimaux}\label{ch:g6:decimals}

Entre $2$ et $3$, il y a plein de nombres~: $2.5$, $2.71$,
$2.999$\,\dots\ Les nombres décimaux étendent le système de
numération à droite des unités, et ils suivent les mêmes règles que
les entiers --- avec quelques pièges que ce chapitre apprend à
éviter.

\section{Écriture décimale}

\begin{definition}[Positions décimales]\label{def:g6:decimals:places}
Un \emph{nombre décimal}\index{nombre décimal} a une partie entière
et une partie décimale, séparées par la virgule. Chaque place à
droite de la virgule vaut dix fois moins que la précédente~:
dixièmes, centièmes, millièmes. Par exemple
\[
13.407 = 13 + \frac{4}{10} + \frac{0}{100} + \frac{7}{1000}
= 1 \times 10 + 3 + 4 \times \frac{1}{10} + 7 \times \frac{1}{1000}.
\]
\end{definition}

\begin{remark}[Zéros qui comptent, zéros qui ne comptent pas]
Ajouter des zéros à la \emph{fin} de la partie décimale ne change
rien~: $2.5 = 2.50 = 2.500$. Mais les zéros \emph{entre} les chiffres
sont essentiels~: $13.407 \neq 13.47$. Et $0.5$ est très différent de
$0.05$~!
\end{remark}

\begin{method}[Comparer des nombres décimaux]\label{met:g6:decimals:compare}
\begin{enumerate}
  \item Comparer d'abord les parties entières~: $7.2 > 5.99$ car
        $7 > 5$.
  \item Si les parties entières sont égales, comparer les parties
        décimales chiffre par chiffre depuis la gauche --- compléter
        avec des zéros finaux aide~: pour comparer $3.4$ et $3.15$,
        écrire $3.40$ et $3.15$~; comme $40 > 15$ centièmes,
        $3.4 > 3.15$.
\end{enumerate}
Attention~: \emph{plus long ne veut pas dire plus grand}. $3.15$ a
plus de chiffres que $3.4$ mais est plus petit.
\end{method}

\section{Additionner, soustraire, multiplier}

\begin{method}[Calculs en colonnes avec des décimaux]\label{met:g6:decimals:columns}
Pour additionner ou soustraire des nombres décimaux, aligner les
\emph{virgules} (ainsi les unités sont sous les unités, les dixièmes
sous les dixièmes), en complétant avec des zéros finaux si besoin~;
puis calculer comme avec les entiers, et placer la virgule du
résultat sous les autres.
\end{method}

\begin{example}\label{ex:g6:decimals:addsub}
Calculer $13.7 + 2.85$, en alignant les virgules ($13.7 = 13.70$)~:
\[
\begin{array}{r}
1\,3.7\,0 \\
+\ \ \,2.8\,5 \\
\hline
1\,6.5\,5
\end{array}
\]
Calculer $6 - 2.35$ (écrire $6 = 6.00$)~:
$6.00 - 2.35 = 3.65$. Vérif.~: $3.65 + 2.35 = 6$.
\end{example}

\begin{example}[Multiplier des décimaux]\label{ex:g6:decimals:mult}
Pour calculer $2.3 \times 1.4$~: multiplier comme des entiers,
$23 \times 14 = 322$~; puis compter les chiffres décimaux des
facteurs ($1 + 1 = 2$), et placer la virgule pour que le résultat en
ait autant~: $2.3 \times 1.4 = 3.22$. Vérification d'ordre de
grandeur~: $2.3 \times 1.4$ doit être un peu plus que
$2.3 \times 1 = 2.3$ --- et $3.22$ l'est.
\end{example}

\section{Multiplier et diviser par 10, 100, 1000}

\begin{proposition}[Décaler la virgule]\label{prop:g6:decimals:shift}
Multiplier un nombre décimal par $10$, $100$, $1000$ déplace sa
virgule de $1$, $2$, $3$ places vers la \emph{droite}~; diviser la
déplace vers la \emph{gauche} (en complétant avec des zéros au
besoin)~:
\[
3.75 \times 100 = 375, \qquad
42.1 \div 1000 = 0.0421 .
\]
\end{proposition}

\begin{proof}
Multiplier par $10$ fait valoir chaque chiffre dix fois plus~: les
dixièmes deviennent des unités, les unités des dizaines, et ainsi de
suite --- chaque chiffre se déplace d'une colonne à gauche dans le
tableau de numération, ce qui revient à déplacer la virgule d'une
place à droite. Diviser inverse ce mouvement.
\end{proof}

\begin{example}[Unités de mesure]\label{ex:g6:decimals:units}
Convertir des unités, c'est exactement ce jeu~: comme $1$~m $= 100$~cm,
\[
3.75 \text{ m} = 3.75 \times 100 \text{ cm} = 375 \text{ cm},
\qquad
42 \text{ mm} = 42 \div 10 \text{ cm} = 4.2 \text{ cm}.
\]
\end{example}

\section{Arrondi}

\begin{definition}[Arrondi]\label{def:g6:decimals:rounding}
L'\emph{arrondi}\index{arrondi} d'un nombre à l'unité (ou au dixième,
au centième, \dots) est le nombre le plus proche avec cette
précision. On regarde le chiffre suivant~: s'il vaut $0,1,2,3,4$, on
arrondit vers le bas (on garde)~; s'il vaut $5,6,7,8,9$, on arrondit
vers le haut.
\end{definition}

\begin{example}
$7.38$ arrondi à l'unité est $7$ (chiffre suivant $3$~: on garde)~;
arrondi au dixième, c'est $7.4$ (chiffre suivant $8$~: on arrondit
vers le haut). Le prix $4.996$ arrondi au centième est $5.00$ ---
l'arrondi peut changer tous les chiffres~!
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g6:decimals:1}
Écrire comme un nombre décimal~: $5 + \dfrac{3}{10} + \dfrac{7}{100}$~;
\quad $\dfrac{9}{10}$~; \quad $12 + \dfrac{4}{1000}$~;
\quad «~huit unités et cinq centièmes~».
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:2}
Dans $86.354$~: quel est le chiffre des dixièmes~? Des centièmes~?
Que compte le chiffre $8$~? Écrire ce nombre comme dans
\cref{def:g6:decimals:places}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:3}
Recopier et compléter avec $<$, $>$ ou $=$~:
\[
5.3 \;?\; 5.29, \qquad
0.7 \;?\; 0.70, \qquad
2.09 \;?\; 2.9, \qquad
14.5 \;?\; 14.49 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:4}
Ranger du plus petit au plus grand~:
$4.2$~; \ $4.05$~; \ $4.51$~; \ $4.15$~; \ $4.5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:5}
Quels nombres décimaux correspondent aux points $A$, $B$, $C$ sur une
droite graduée en dixièmes, si $A$ est $3$ graduations après $6$, $B$
est $7$ graduations après $6$, et $C$ est $2$ graduations après $7$~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:6}
Calculer en colonnes~: $45.8 + 7.65$~; \quad $23.4 - 8.72$~; \quad
$5.6 \times 2.4$ (compter les chiffres décimaux~!).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:7}
Calculer sans aucun calcul écrit~:
\[
6.42 \times 10, \qquad
0.35 \times 1000, \qquad
78.1 \div 100, \qquad
5 \div 1000 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:8}
Convertir~: $2.4$~m en cm~; \quad $370$~g en kg~; \quad $0.85$~km en
m~; \quad $56$~mm en m.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:9}
Arrondir $23.867$~: à l'unité~; au dixième~; au centième. Arrondir
$9.97$ au dixième.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:decimals:10}
Une baguette coûte $1.15$. Léna en achète trois et paie avec un billet
de $5$. Écrire les deux calculs nécessaires, et donner sa monnaie.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:decimals:11}
Trouver un nombre décimal strictement entre $7.4$ et $7.5$~; puis
entre $3.99$ et $4$. Combien de tels nombres y a-t-il dans chaque
cas~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:decimals:12}
En utilisant chacun des chiffres $2$, $5$, $8$ exactement une fois et
une virgule, écrire le plus grand nombre possible, puis le plus
petit. (Les nombres comme $.58$ ne sont pas autorisés~: la partie
entière doit contenir au moins un chiffre.)
\end{exercise}
