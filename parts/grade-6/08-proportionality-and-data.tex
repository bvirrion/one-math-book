\chapter{Proportionality and Data}\label{ch:g6:propdata}

If three notebooks cost $6$ euros, six notebooks cost $12$: double the
notebooks, double the price. Quantities behaving like this are
\emph{\omterm{def:g6:propdata:def}{proportional}} --- one of the most useful ideas in all of
mathematics, developed further in \cref{ch:g7:prop}. The chapter ends
with reading and drawing simple data charts.

\section{Proportional quantities}

\begin{definition}[Proportionality]\label{def:g6:propdata:def}
Two quantities are \emph{proportional}\index{proportionality} when the
values of one are obtained from the values of the other by multiplying
always by the \emph{same} number, called the \emph{proportionality
coefficient}.
\end{definition}

\begin{example}\label{ex:g6:propdata:table}
Notebooks at $2$ euros each:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cccc}
notebooks & $3$ & $5$ & $8$ & $12$ \\
\hline
price (euros) & $6$ & $10$ & $16$ & $24$ \\
\end{tabular}
\end{center}
Each price is the number of notebooks $\times 2$: the coefficient is
$2$ (the unit price). A quick check that a table is \omterm{def:g6:propdata:def}{proportional}:
all the ``column \omterm{def:g3:division:remainder}{quotients}'' $\frac{6}{3}, \frac{10}{5}, \frac{16}{8},
\frac{24}{12}$ must be equal --- here they all equal $2$.
\end{example}

\begin{example}[A non-example]\label{ex:g6:propdata:nonexample}
Age and height are not \omterm{def:g6:propdata:def}{proportional}: a $12$-year-old is not twice as
tall as a $6$-year-old. Check on numbers: $1.50$ m at $12$ years and
$1.15$ m at $6$ years give \omterm{def:g3:division:remainder}{quotients} $\frac{1.50}{12} = 0.125$ and
$\frac{1.15}{6} \approx 0.19$ --- not equal.
\end{example}

\begin{method}[Completing a proportionality table]\label{met:g6:propdata:complete}
Three ways, to be chosen freely:
\begin{enumerate}
  \item \emph{coefficient}: find the multiplier from one complete
        column, apply it to the others;
  \item \emph{columns}: a column can be multiplied by a number, or two
        columns can be added, to produce a new column;
  \item \emph{back to the unit}: find the value for \emph{one} item
        first, then multiply.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g6:propdata:methods}
Five identical mugs cost $15$ euros; how much do eight mugs cost?

\emph{Back to the unit}: one mug costs $15 \div 5 = 3$ euros, so eight
mugs cost $8 \times 3 = 24$ euros.

\emph{Columns}: $8 = 5 + 3$; three mugs cost $\frac{3}{5}$ of $15$,
i.e.\ $9$; so eight mugs cost $15 + 9 = 24$ euros. Same answer, as it
must be.
\end{example}

\section{Percentages}

\begin{definition}[Percentage]\label{def:g6:propdata:percent}
``$t\,\%$ of a quantity'' means the \omterm{def:g6:fractions:def}{fraction} $\frac{t}{100}$ of it:
$25\,\%$ of $60$ is $\frac{25}{100} \times 60 = 15$. A percentage is a
\omterm{def:g6:propdata:def}{proportionality} with coefficient $\frac{t}{100}$.
\end{definition}

\begin{example}\label{ex:g6:propdata:percent}
A class of $30$ students contains $40\,\%$ girls. Number of girls, step
by step: $40\,\%$ means $\frac{40}{100} = 0.4$, and
$0.4 \times 30 = 12$ girls. Useful landmarks:
$50\,\%$ is one \omterm{def:g3:division:half}{half}, $25\,\%$ one \omterm{def:g3:division:half}{quarter}, $10\,\%$ one tenth ---
so $10\,\%$ of $30$ is $3$, and $40\,\% = 4 \times 10\,\%$ is
$4 \times 3 = 12$: same answer, computed mentally.
\end{example}

\section{Reading and drawing charts}

\begin{method}[Reading a table or chart]\label{met:g6:propdata:read}
Whatever the picture --- table, bar chart, line graph:
\begin{enumerate}
  \item read the title and the units on both axes first;
  \item to answer a question, locate the right bar/column, then read
        the value carefully against the graduation;
  \item to compare, compare bars by height --- but check the axis
        starts at $0$, or the picture can mislead.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g6:propdata:chart}
The bar chart below shows the number of books borrowed at the school
library during one week.

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[
  width=8.6cm, height=5.2cm,
  ybar, bar width=16pt,
  symbolic x coords={Mon,Tue,Wed,Thu,Fri},
  xtick=data,
  ymin=0, ymax=22,
  ytick={5,10,15,20},
  ylabel={books borrowed},
  tick label style={font=\small},
  label style={font=\small}]
  \addplot[fill=omDef!30, draw=omDef] coordinates
    {(Mon,12) (Tue,8) (Wed,17) (Thu,6) (Fri,15)};
\end{axis}
\end{tikzpicture}

{\small A bar chart: one bar per day, height equal to the count.}
\end{omfigure}

Readings: the busiest day is Wednesday ($17$ books). Tuesday and
Thursday together account for $8 + 6 = 14$ books --- fewer than Friday
alone ($15$). Total for the week:
$12 + 8 + 17 + 6 + 15 = 58$ books.
\end{example}

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[omaxis,
  width=8.2cm, height=5.4cm,
  xmin=0, xmax=9.4, ymin=0, ymax=28,
  xlabel={notebooks}, ylabel={price (euros)},
  xtick={1,...,9}, ytick={4,8,12,16,20,24}]
  \addplot[omDef, thick, domain=0:9] {3*x};
  \addplot[omDef, only marks, mark=*, mark size=1.5pt] coordinates
    {(1,3) (2,6) (3,9) (4,12) (5,15) (6,18) (7,21) (8,24)};
  \draw[dashed, omThm] (axis cs:6,0) -- (axis cs:6,18)
    -- (axis cs:0,18);
\end{axis}
\end{tikzpicture}

{\small \omterm{def:g6:propdata:def}{Proportional} quantities have a very recognizable graph: the
points line up on a straight line through the origin (here notebooks at
$3$ euros each; the dashes read the price of $6$ notebooks: $18$
euros).}
\end{omfigure}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g6:propdata:1}
Is the table \omterm{def:g6:propdata:def}{proportional}? Justify by computing \omterm{def:g3:division:remainder}{quotients}.
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|ccc}
kg of apples & $2$ & $3$ & $5$ \\
\hline
price (euros) & $5$ & $7.5$ & $12.5$ \\
\end{tabular}
\qquad
\begin{tabular}{l|ccc}
age (years) & $2$ & $4$ & $8$ \\
\hline
height (cm) & $85$ & $103$ & $130$ \\
\end{tabular}
\end{center}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:propdata:2}
Four croissants cost $4.80$ euros. Find the price of one croissant,
then of seven croissants.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:propdata:3}
Complete the \omterm{def:g6:propdata:def}{proportionality} table (coefficient first!):
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|cccc}
liters of fuel & $10$ & $25$ & $?$ & $60$ \\
\hline
price (euros) & $18$ & $?$ & $72$ & $?$ \\
\end{tabular}
\end{center}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:propdata:4}
A car uses $6$ L of fuel per $100$ km. How much fuel for $250$ km? For
$350$ km? How far can it go with $27$ L?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:propdata:5}
Compute mentally, using the landmarks of \cref{ex:g6:propdata:percent}:
$50\,\%$ of $84$; \quad $25\,\%$ of $200$; \quad $10\,\%$ of $63$;
\quad $30\,\%$ of $70$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:propdata:6}
In a school of $450$ students, $60\,\%$ eat at the cafeteria. How many
students is that? How many do not?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:propdata:7}
Using the bar chart of \cref{ex:g6:propdata:chart}: on which days were
fewer than $10$ books borrowed? How many more books were borrowed on
Wednesday than on Thursday?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:propdata:8}
The temperatures at noon from Monday to Friday were $14$, $16$, $13$,
$17$, $20$ degrees. Draw a bar chart of these data (choose a sensible
graduation), and read off the warmest day.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:propdata:9}
A recipe for $6$ people needs $450$ g of flour and $3$ eggs. Adapt it
for $10$ people. (For the eggs, think before writing a \omterm{def:g6:decimals:places}{decimal number}
of eggs!)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:propdata:10}
At a constant speed, a cyclist rides $24$ km in $1$ hour.
\begin{enumerate}
  \item How far does she ride in $2$ h? In \omterm{def:g3:division:half}{half} an hour? In
        $1$ h $30$ min?
  \item How long does she need for $60$ km?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:propdata:11}
A shop offers ``$20\,\%$ off everything''. Compute the discount and
the new price for: a ball at $15$ euros; a racket at $40$ euros. Is the
new price \omterm{def:g6:propdata:def}{proportional} to the old price? What is the coefficient?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:propdata:12}
Two candles of the same height are lit at the same time. The thick one
burns down completely in $6$ hours, the thin one in $4$ hours, each at
its own constant rate. After how much time is the thin candle exactly
\omterm{def:g3:division:half}{half} as tall as the thick one? (Express the remaining heights after
$t$ hours as \omterm{def:g6:fractions:def}{fractions} of the initial height.)
\end{exercise}
