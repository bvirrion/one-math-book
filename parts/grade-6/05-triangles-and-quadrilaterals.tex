\chapter{Triangles and Quadrilaterals}\label{ch:g6:shapes}

Triangles and \omterm{def:g4:geometry:polygon}{quadrilaterals} are the basic shapes of geometry. The goal
of this chapter is to know their special families --- \omterm{def:g6:shapes:triangles}{isosceles},
\omterm{def:g6:shapes:triangles}{equilateral}, \omterm{def:g6:shapes:triangles}{right triangles}; rectangles, rhombuses, squares --- by
their \emph{defining properties}, and to construct them precisely with
ruler and compass.

\section{Triangles}

\begin{definition}[Special triangles]\label{def:g6:shapes:triangles}
A triangle $ABC$ has three \omterm{def:g5:solids:def}{vertices} and three sides. It is:
\begin{itemize}
  \item \emph{isosceles}\index{isosceles triangle} at $A$ when
        $AB = AC$ (two equal sides);
  \item \emph{equilateral}\index{equilateral triangle} when
        $AB = BC = CA$ (three equal sides);
  \item \emph{right-angled}\index{right triangle} at $A$ when the angle
        $\widehat{BAC}$ is a \omterm{def:g3:shapes:rightangle}{right angle}. The side $[BC]$ opposite the
        \omterm{def:g3:shapes:rightangle}{right angle} is called the \emph{hypotenuse}.
\end{itemize}
\end{definition}

\begin{method}[Constructing a triangle from its three sides]\label{met:g6:shapes:construct}
To construct $ABC$ with $AB = 5$ cm, $AC = 4$ cm, $BC = 3$ cm:
\begin{enumerate}
  \item draw the \omterm{def:g6:lines:objects}{segment} $[AB]$ of $5$ cm with the ruler;
  \item draw an arc of the circle of center $A$ and \omterm{def:g3:shapes:shapes}{radius} $4$ cm
        (all points at $4$ cm from $A$);
  \item draw an arc of the circle of center $B$ and \omterm{def:g3:shapes:shapes}{radius} $3$ cm;
  \item the point $C$ is where the two arcs cross; join it to $A$ and
        $B$.
\end{enumerate}
\end{method}

\section{Quadrilaterals}

\begin{definition}[Special quadrilaterals]\label{def:g6:shapes:quadrilaterals}
A \omterm{def:g4:geometry:polygon}{quadrilateral} has four \omterm{def:g5:solids:def}{vertices} and four sides. It is a:
\begin{itemize}
  \item \emph{rectangle}\index{rectangle} when its four angles are
        \omterm{def:g3:shapes:rightangle}{right angles};
  \item \emph{rhombus}\index{rhombus} when its four sides have the same
        length;
  \item \emph{square}\index{square} when it is both: four \omterm{def:g3:shapes:rightangle}{right angles}
        \emph{and} four equal sides.
\end{itemize}
\end{definition}

\begin{proposition}[Properties of the diagonals]\label{prop:g6:shapes:diagonals}
In these \omterm{def:g4:geometry:polygon}{quadrilaterals}, the two diagonals cut each other at their
common midpoint, and moreover:
\begin{itemize}
  \item in a rectangle, the diagonals have the \emph{same length};
  \item in a \omterm{def:g6:shapes:quadrilaterals}{rhombus}, the diagonals are \emph{\omterm{def:g6:lines:perp}{perpendicular}};
  \item in a square, both at once.
\end{itemize}
\end{proposition}
\admitted

\begin{remark}
A square is automatically a rectangle (it has four \omterm{def:g3:shapes:rightangle}{right angles}) and
automatically a \omterm{def:g6:shapes:quadrilaterals}{rhombus} (four equal sides). So ``rectangle'' does not
mean ``not a square''! In mathematics, the more special shape belongs
to the more general family.
\end{remark}

\begin{example}[Recognizing a shape from its properties]\label{ex:g6:shapes:recognize}
A \omterm{def:g4:geometry:polygon}{quadrilateral} has four sides of length $4$ cm and one \omterm{def:g3:shapes:rightangle}{right angle}.
What is it? Four equal sides make it a \omterm{def:g6:shapes:quadrilaterals}{rhombus}. In a \omterm{def:g6:shapes:quadrilaterals}{rhombus}, opposite
angles are equal and the four angles add up to $360^\circ$; one \omterm{def:g3:shapes:rightangle}{right
angle} then forces all four to be right (this will be proved with
parallelograms in \cref{ch:g7:central}). So the shape is a square.
\end{example}

\begin{method}[Construction programs]\label{met:g6:shapes:program}
A \emph{construction program} is a list of instructions that anyone can
follow to reproduce a figure exactly. Write one instruction per line,
name every point when it is created, and only use: ``draw the \omterm{def:g6:lines:objects}{segment}
/ line / circle\dots'', ``place the point where \dots\ cross''. Test
your program by following it literally, word for word.
\end{method}

\begin{example}\label{ex:g6:shapes:program}
Program for a square $ABCD$ of side $4$ cm:
\begin{enumerate}
  \item Draw a \omterm{def:g6:lines:objects}{segment} $[AB]$ with $AB = 4$ cm.
  \item Draw the \omterm{def:g6:lines:perp}{perpendicular} to $(AB)$ at $A$; on it, place $D$ at
        $4$ cm from $A$ (choose one side of the line and stick to it).
  \item Draw the \omterm{def:g6:lines:perp}{perpendicular} to $(AB)$ at $B$; on it, place $C$ at
        $4$ cm from $B$, on the same side as $D$.
  \item Join $C$ and $D$.
\end{enumerate}
Check with the ruler: $DC$ should measure $4$ cm too.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g6:shapes:1}
Construct a triangle $ABC$ with $AB = 6$ cm, $AC = 5$ cm and
$BC = 4$ cm, following \cref{met:g6:shapes:construct}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:2}
Construct an \omterm{def:g6:shapes:triangles}{isosceles triangle} $DEF$ with $DE = DF = 5$ cm and
$EF = 3$ cm. Which two angles of this triangle look equal? Measure them
to check.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:3}
Construct an \omterm{def:g6:shapes:triangles}{equilateral triangle} of side $4$ cm using only ruler and
compass. Measure one of its angles.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:4}
Classify each triangle from its data (\omterm{def:g6:shapes:triangles}{isosceles}, \omterm{def:g6:shapes:triangles}{equilateral},
\omterm{def:g6:shapes:triangles}{right-angled} --- several answers may apply):
(a) sides $7$, $7$, $7$; \
(b) sides $5$, $5$, $8$; \
(c) angles $90^\circ$, $45^\circ$, $45^\circ$; \
(d) angles $60^\circ$, $60^\circ$, $60^\circ$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:5}
In a \omterm{def:g6:shapes:triangles}{right triangle}, which side is the hypotenuse? Draw a triangle
$MNP$ \omterm{def:g6:shapes:triangles}{right-angled} at $N$, and name its hypotenuse. Which \omterm{def:g5:solids:def}{vertex} does
the hypotenuse never touch?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:6}
True or false? Justify each answer in one sentence.
\begin{enumerate}
  \item Every square is a rectangle.
  \item Every rectangle is a square.
  \item Every square is a \omterm{def:g6:shapes:quadrilaterals}{rhombus}.
  \item A \omterm{def:g6:shapes:quadrilaterals}{rhombus} with a \omterm{def:g3:shapes:rightangle}{right angle} is a square.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:7}
Construct a rectangle $ABCD$ with $AB = 5$ cm and $BC = 3$ cm. Measure
its two diagonals: what do you observe?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:8}
Construct a \omterm{def:g6:shapes:quadrilaterals}{rhombus} with diagonals of $6$ cm and $4$ cm. (Use
\cref{prop:g6:shapes:diagonals}: draw the two diagonals first ---
\omterm{def:g6:lines:perp}{perpendicular}, crossing at their midpoints --- then join the four
ends.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:shapes:9}
Write a construction program for an \omterm{def:g6:shapes:triangles}{isosceles triangle} with base
$4$ cm and equal sides of $6$ cm, in the style of
\cref{ex:g6:shapes:program}. Exchange with a classmate: can they follow
it without seeing your figure?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:shapes:10}
Draw a circle of center $O$ and one of its \omterm{def:g4:geometry:circle}{diameters} $[AB]$. Place any
point $M$ on the circle and draw the triangle $AMB$. Measure the angle
$\widehat{AMB}$. Move $M$ and measure again. What do you conjecture?
(This surprising fact is proved in \cref{ch:g8:cosine}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:shapes:11}
A \omterm{def:g4:geometry:polygon}{quadrilateral} has diagonals of the same length that cut each other at
their common midpoint. Sort out which shape it must be, using
\cref{prop:g6:shapes:diagonals}: rectangle, \omterm{def:g6:shapes:quadrilaterals}{rhombus} or square?
(Careful: only one property is given.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:shapes:12}
On a $4 \times 4$ dotted grid, how many different squares can you draw
whose corners are grid dots --- counting also the ``tilted'' squares?
Draw at least one tilted square and explain why its sides are equal.
\end{exercise}

\section{Problem: What the diagonals know}

\begin{problem}[{Weekend problem --- rebuilding every special
quadrilateral from its diagonals alone, and counting the diagonals
of any polygon}]
\label{pb:g6:shapes:1}
\Cref{prop:g6:shapes:diagonals} reads a shape and tells you about
its diagonals. This problem plays the game \emph{backwards}: given
only two crossing \omterm{def:g6:lines:objects}{segments} --- the future diagonals --- what shape
appears when you join their four ends? You will build a complete
``diagonal dictionary'', understand \emph{why} the \omterm{def:g6:shapes:quadrilaterals}{rhombus}
construction of \cref{exo:g6:shapes:8} works, meet a shape that
has no name yet, and finish by counting the diagonals of \omterm{def:g4:geometry:polygon}{polygons}
with many more than four sides.

\medskip
\noindent\textbf{Part I --- The diagonal dictionary.}
For each question, draw the two \omterm{def:g6:lines:objects}{segments} as described, join their
four ends in order, and identify the \omterm{def:g4:geometry:polygon}{quadrilateral} you get.
\begin{enumerate}
  \item Two \omterm{def:g6:lines:objects}{segments} of $6$ cm each, \omterm{def:g6:lines:perp}{perpendicular}, crossing at
        their common midpoint. Measure the four sides and the
        four angles of the resulting shape: what is it?
  \item Two \omterm{def:g6:lines:objects}{segments} of $8$ cm and $5$ cm, \omterm{def:g6:lines:perp}{perpendicular},
        crossing at their common midpoint. What shape appears
        (\cref{exo:g6:shapes:8})?
  \item Two \omterm{def:g6:lines:objects}{segments} of $7$ cm each --- equal but \emph{not}
        \omterm{def:g6:lines:perp}{perpendicular} --- crossing at their common midpoint.
        Measure the sides and angles: what do you conjecture the
        shape to be?
  \item Two \omterm{def:g6:lines:objects}{segments} of $8$ cm and $5$ cm, not \omterm{def:g6:lines:perp}{perpendicular},
        crossing at their common midpoint. The shape you get is
        none of the three of
        \cref{def:g6:shapes:quadrilaterals} --- describe what
        you see (which sides look equal? \omterm{def:g6:lines:perp}{parallel}?). This
        \omterm{def:g4:geometry:polygon}{quadrilateral} receives its name in \cref{ch:g7:central}.
  \item Summarize questions 1--4 in a dictionary with four
        entries: diagonals \emph{equal / \omterm{def:g6:lines:perp}{perpendicular} / both /
        neither} (always crossing at their midpoints)
        $\rightarrow$ name (or description) of the shape.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Why the constructions work.}
\begin{enumerate}[resume]
  \item In question 2, the two diagonals cut the shape into four
        \omterm{def:g6:shapes:triangles}{right triangles} around the crossing point. What are the
        two leg lengths of each of these triangles? Explain why
        all four triangles are exact copies of one another, and
        deduce why the four sides of the shape are equal --- the
        secret of \cref{exo:g6:shapes:8}.
  \item Deduce a compass-only test: after building any shape
        from its diagonals, how can you check with the compass
        alone (no ruler measurements) that the four sides are
        equal?
  \item Write a construction program
        (\cref{met:g6:shapes:program}) for a square whose
        \emph{diagonals} measure $6$ cm --- not its sides.
  \item In the square of question 8, is the side longer or
        shorter than $3$ cm --- \omterm{def:g3:division:half}{half} the diagonal? Measure to
        find out. One \omterm{def:g3:division:half}{half} of the answer can already be
        \emph{proved}: going from one corner to the next via the
        center is a $3 + 3$ cm detour, and a straight path is
        always shorter than a detour --- so the side is shorter
        than $6$ cm. (Proving that it also beats $3$ cm, and
        computing it exactly, is a job for
        \cref{ch:g8:pythagoras}.)
  \item A new entry for the dictionary: draw a \omterm{def:g6:lines:objects}{segment} $[AC]$ of
        $8$ cm; on it, mark the point $H$ at $3$ cm from $A$;
        draw through $H$ the \omterm{def:g6:lines:perp}{perpendicular} \omterm{def:g6:lines:objects}{segment} $[BD]$ of
        $5$ cm with $H$ as \emph{its} midpoint. Join $ABCD$. This
        shape is a \emph{kite}\index{kite} (one diagonal bisects
        the other, at a \omterm{def:g3:shapes:rightangle}{right angle}). Using corner triangles as
        in question 6, find which sides of a \omterm{pb:g6:shapes:1}{kite} are equal to
        which.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Counting diagonals.}
A \emph{diagonal} of a \omterm{def:g4:geometry:polygon}{polygon} joins two \omterm{def:g5:solids:def}{vertices} that are not
neighbours.
\begin{enumerate}[resume]
  \item Draw a \omterm{def:g4:geometry:polygon}{quadrilateral}, a \omterm{def:g4:geometry:polygon}{pentagon} (five \omterm{def:g5:solids:def}{vertices}) and a
        \omterm{def:g4:geometry:polygon}{hexagon} (six \omterm{def:g5:solids:def}{vertices}), and draw all their diagonals.
        Count them: how many in each case?
  \item Counting without drawing, for the \omterm{def:g4:geometry:polygon}{hexagon}: from one
        \omterm{def:g5:solids:def}{vertex}, how many diagonals leave? (To how many of the
        six \omterm{def:g5:solids:def}{vertices} can it \emph{not} be joined by a diagonal?)
        Multiplying by the six \omterm{def:g5:solids:def}{vertices} counts every diagonal
        exactly twice --- why? Recover your count of question
        11. (The same double-counting appeared in the handshakes
        of \cref{pb:g6:wholes:1}.)
  \item Use the method of question 12 to count the diagonals of
        a \omterm{def:g4:geometry:polygon}{polygon} with $10$ \omterm{def:g5:solids:def}{vertices}, then with $20$ \omterm{def:g5:solids:def}{vertices}.
  \item A \omterm{def:g4:geometry:polygon}{polygon} has exactly $44$ diagonals. How many \omterm{def:g5:solids:def}{vertices}
        does it have? (Try the method of question 13 on a few
        candidates.)
  \item Check your method on the smallest \omterm{def:g4:geometry:polygon}{polygon} of all: how
        many diagonals does a triangle have, and what does the
        counting method of question 12 answer? Conclude with one
        sentence on what handshakes and diagonals have in
        common.
\end{enumerate}
\end{problem}
