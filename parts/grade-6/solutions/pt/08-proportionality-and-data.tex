\section*{Capítulo \ref{ch:g6:propdata} --- Proporcionalidade e Dados}

\begin{solution}{exo:g6:propdata:1}
Maçãs: \omterm{def:g3:division:remainder}{quocientes} $\frac52 = 2.5$, $\frac{7.5}{3} = 2.5$,
$\frac{12.5}{5} = 2.5$ --- todos iguais: proporcional (preço
$2.50$ por kg).

Idade/altura: $\frac{85}{2} = 42.5$ mas $\frac{103}{4} = 25.75$ --- não
igual: não proporcional.
\end{solution}

\begin{solution}{exo:g6:propdata:2}
Um croissant: $4.80 \div 4 = 1.20$ euros. Sete:
$7 \times 1.20 = 8.40$ euros.
\end{solution}

\begin{solution}{exo:g6:propdata:3}
Coeficiente: $18 \div 10 = 1.8$ euros por litro. Então
$25$ Custo L $25 \times 1.8 = 45$ euros; $72$ comprar euros
$72 \div 1.8 = 40$ eu; $60$ Custo L $60 \times 1.8 = 108$ euros.
\end{solution}

\begin{solution}{exo:g6:propdata:4}
Para $250$ quilômetros: $2.5$ vezes o combustível de $100$ km, então
$2.5 \times 6 = 15$ L. Para $350$ quilômetros: $3.5 \times 6 = 21$ L. Com
$27$ eu: $27 \div 6 = 4.5$ centenas de km, ou seja,\ $450$ km.
\end{solution}

\begin{solution}{exo:g6:propdata:5}
$50\,\%$ de $84$: \omterm{def:g3:division:half}{metade}, $42$. \quad
$25\,\%$ de $200$: um \omterm{def:g3:division:half}{trimestre}, $50$. \quad
$10\,\%$ de $63$: um décimo, $6.3$. \quad
$30\,\%$ de $70$: $3 \times 7 = 21$.
\end{solution}

\begin{solution}{exo:g6:propdata:6}
$60\,\%$ de $450$: $\frac{60}{100} \times 450 = 270$ os alunos comem em
o refeitório; $450 - 270 = 180$ não.
\end{solution}

\begin{solution}{exo:g6:propdata:7}
Menos que $10$ livros: terça-feira ($8$) e quinta-feira ($6$). Quarta-feira
menos quinta-feira: $17 - 6 = 11$ mais livros.
\end{solution}

\begin{solution}{exo:g6:propdata:8}
Gráfico de barras com cinco barras de altura $14$, $16$, $13$, $17$, $20$
(formatura a cada $2$ ou $5$ graus funciona bem). Dia mais quente: sexta-feira
($20$ graus).
\end{solution}

\begin{solution}{exo:g6:propdata:9}
Para $10$ pessoas, multiplique por $\frac{10}{6} = \frac53$: farinha
$450 \times \frac{10}{6} = 750$ g. Ovos:
$3 \times \frac{10}{6} = 5$ --- felizmente um número inteiro. (Se fosse
não, alguém faria redondo \emph{acima} ter o suficiente.)
\end{solution}

\begin{solution}{exo:g6:propdata:10}
\emph{1.} Em $2$ h: $48$ km. Em \omterm{def:g3:division:half}{metade} uma hora: $12$ km. Em
$1$ h $30$: $24 + 12 = 36$ km.

\emph{2.} $60 \div 24 = 2.5$ horas, ou seja,\ $2$ h $30$ min.
\end{solution}

\begin{solution}{exo:g6:propdata:11}
Bola: desconto $20\,\%$ de $15 = 3$ euros, novo preço $12$ euros.
Raquete: desconto $8$ euros, novo preço $32$ euros. O novo preço é
$80\,\%$ do antigo em todos os casos: proporcional, coeficiente
$0.8$.
\end{solution}

\begin{solution}{exo:g6:propdata:12}
Depois $ t $ horas, a vela grossa queimou $\frac t6$ da sua altura:
está em $1 - \frac t6$ da altura inicial; o magro em
$1 - \frac t4$. Nós queremos
\[
1 - \frac t4 = \frac12 \left(1 - \frac t6\right)
\quad\text{i.e.}\quad
1 - \frac t4 = \frac12 - \frac t{12}.
\]
Então $\frac12 = \frac t4 - \frac t{12} = \frac{3t - t}{12} =
\frac{t}{6}$, então $ t = 3$ horas. Verifique: depois $3$ h a vela grossa
fica em $1 - \frac36 = \frac12$ e o magro em
$1 - \frac34 = \frac14$ --- de fato \omterm{def:g3:division:half}{metade} disso.
\end{solution}

\begin{solution}{pb:g6:propdata:1}
\textbf{1.} $100\,000$ cm $= 1\,000$ eu $= 1$ km: um centímetro
no mapa é um quilômetro no chão.

\textbf{2.} $7.5$ cm $\rightarrow 7.5$ km entre as aldeias;
$12$ cm $\rightarrow 12$ km de margem do lago.

\textbf{3.} $20$ quilômetros $\rightarrow 20$ cm no mapa.

\textbf{4.} $5$ quilômetros $= 500\,000$ cm, então a escala é
$1:500\,000$. O mapa de caminhadas ($1:100\,000$) é o mais
detalhado: usa $5$ cm de papel onde o outro gasta apenas
$1$ cm.

\textbf{5.}
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|ccc}
map (cm) & $1$ & $2$ & $7.5$ \\
\hline
real (km) & $1$ & $2$ & $7.5$ \\
\end{tabular}
\end{center}
Distância real $=$ distância do mapa $\times 1$ (nestas unidades):
sempre o \emph{mesmo} multiplicador, que é exatamente o
definição de proporcionalidade
(\cref{def:g6:propdata:def}). O coeficiente aqui é $1$
quilômetro por centímetro.

\textbf{6.} $4$ eu $= 400$ centímetros e $400 \div 50 = 8$;
$3$ eu $= 300$ centímetros e $300 \div 50 = 6$: o plano mostra um
$8$ cm $\times $ $6$ cm retângulo.

\textbf{7.} Cama: $200 \div 50 = 4$ cm por $90 \div 50 = 1.8$ cm.
Porta: $80 \div 50 = 1.6$ cm.

\textbf{8.} Real \omterm{def:g6:measure:area}{área}: $4 \times 3 = 12$ eu $^2$. Plano \omterm{def:g6:measure:area}{área}:
$8 \times 6 = 48$ cm $^2$. Convertendo:
$12$ eu $^2 = 12 \times 10\,000 = 120\,000$ cm $^2$, e
\[
120\,000 \div 48 = 2\,500 .
\]
O \omterm{def:g3:division:remainder}{quociente} não é $50$ mas $2\,500 = 50 \times 50$.

\textbf{9.} Um centímetro $^2$ de papel representa um quadrado real de
$50$ cm por $50$ cm, que contém
$50 \times 50 = 2\,500$ cm $^2$. Então, quando os comprimentos são divididos por
$50$, \omterm{def:g6:measure:area}{áreas} são divididos por $50 \times 50 = 2\,500$: \omterm{def:g6:measure:area}{áreas} siga
o \emph{quadrado} da escala.

\textbf{10.} $6 \times 2\,500 = 15\,000$ cm $^2$, e
$15\,000 \div 10\,000 = 1.5$ eu $^2$.

\textbf{11.} A imagem sugere domingo vendido \emph{duas vezes} como
muito (uma barra com o dobro da altura). Verdade: o aumento é $5$ saída de euros
de $105$, e $5 \div 105 \approx 0.048$: sobre $5\,\%$ mais,
não $100\,\%$ mais. O gerente ignorou o aviso para verificar
que o eixo começa em $0$
(\cref{met:g6:propdata:read}).

\textbf{12.} Com o eixo de $0$, as barras sobem para $105$ e
$110$ unidades pequenas: duas barras de quase a mesma altura, o
imagem honesta de um $5\,\%$ \omterm{ex:g1:subtraction:difference}{diferença}.

\textbf{13.} Para cima: o aumento é $20$ selos desde o início de
$40$: $\frac{20}{40} = 50\,\%$. Para baixo: a diminuição é $20$
selos desde o início de $60$: $\frac{20}{60} = \frac13 \approx
33\,\%$. Mesmo $20$ selos, valores iniciais diferentes --- a
porcentagem é sempre um \omterm{def:g6:fractions:def}{fração} \emph{de algo}, e o
“alguma coisa” mudou.

\textbf{14.} Depois $20\,\%$ desligado: $100 - 20 = 80$ euros. O
extra $10\,\%$ aplica-se a $80$: desconto $8$ euros, preço final
$72$ euros. Desconto total: $28$ euros de $100$, então
$28\,\%$ --- não $30\,\%$. O segundo desconto incidiu sobre o
preço já reduzido, por isso os seus euros são menores: percentagens de
quantidades diferentes não adicionam.

\textbf{15.} Nesse mapa, $1$ cm significa $5$ km, então $1$
cm $^2$ significa $5 \times 5 = 25$ quilômetros $^2$ (regra da questão 9).
A floresta: $3 \times 25 = 75$ quilômetros $^2$. Os comprimentos seguem o
escala; \omterm{def:g6:measure:area}{áreas} siga seu \emph{quadrado}.
\end{solution}
