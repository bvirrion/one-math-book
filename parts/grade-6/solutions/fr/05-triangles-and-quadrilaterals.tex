\section*{Chapitre \ref{ch:g6:shapes} --- Triangles et quadrilatères}

\begin{solution}{exo:g6:shapes:1}
Suivre la méthode~: \omterm{def:g6:lines:objects}{segment} $[AB]$ de $6$~cm, arc de \omterm{def:g3:shapes:shapes}{rayon} $5$~cm
centré en $A$, arc de \omterm{def:g3:shapes:shapes}{rayon} $4$~cm centré en $B$~; le point de
croisement est $C$.
\end{solution}

\begin{solution}{exo:g6:shapes:2}
Base $[EF]$ de $3$~cm, puis deux arcs de \omterm{def:g3:shapes:shapes}{rayon} $5$~cm centrés en $E$
et $F$~; leur croisement est $D$. Les deux angles à la base
$\widehat{DEF}$ et $\widehat{DFE}$ sont égaux (environ $72.5^\circ$
chacun) --- symétrie du \omterm{def:g6:shapes:triangles}{triangle isocèle}
(\cref{ex:g6:symmetry:proves}).
\end{solution}

\begin{solution}{exo:g6:shapes:3}
\omterm{def:g6:lines:objects}{Segment} de $4$~cm, puis deux arcs de \omterm{def:g3:shapes:shapes}{rayon} $4$~cm centrés en ses
extrémités. Chaque angle mesure $60^\circ$.
\end{solution}

\begin{solution}{exo:g6:shapes:4}
(a) \omterm{def:g6:shapes:triangles}{équilatéral} (donc aussi \omterm{def:g6:shapes:triangles}{isocèle}). \quad
(b) \omterm{def:g6:shapes:triangles}{isocèle} ($5 = 5$). \quad
(c) rectangle et \omterm{def:g6:shapes:triangles}{isocèle} (deux angles égaux de $45^\circ$ font face à
deux côtés égaux). \quad
(d) \omterm{def:g6:shapes:triangles}{équilatéral} (trois angles égaux).
\end{solution}

\begin{solution}{exo:g6:shapes:5}
L'hypoténuse est le côté opposé à l'\omterm{def:g3:shapes:rightangle}{angle droit}, toujours le plus long
côté. Dans $MNP$ rectangle en $N$, l'hypoténuse est $[MP]$~: elle ne
touche jamais le \omterm{def:g5:solids:def}{sommet} de l'\omterm{def:g3:shapes:rightangle}{angle droit}, $N$.
\end{solution}

\begin{solution}{exo:g6:shapes:6}
\emph{1. Vrai}~: un carré a quatre \omterm{def:g3:shapes:rightangle}{angles droits}, ce qui est la
définition d'un rectangle.

\emph{2. Faux}~: un rectangle $5 \times 3$ a des côtés inégaux.

\emph{3. Vrai}~: un carré a quatre côtés égaux, définition d'un
\omterm{def:g6:shapes:quadrilaterals}{losange}.

\emph{4. Vrai}~: voir \cref{ex:g6:shapes:recognize}.
\end{solution}

\begin{solution}{exo:g6:shapes:7}
Construction avec des \omterm{def:g6:lines:perp}{perpendiculaires}, comme dans
\cref{ex:g6:shapes:program}. Les deux diagonales mesurent environ
$5.8$~cm~: les diagonales d'un rectangle ont la même longueur
(\cref{prop:g6:shapes:diagonals}).
\end{solution}

\begin{solution}{exo:g6:shapes:8}
Tracer un \omterm{def:g6:lines:objects}{segment} de $6$~cm~; son milieu $O$~; la \omterm{def:g6:lines:perp}{perpendiculaire} en
$O$~; dessus, deux points à $2$~cm de $O$ de chaque côté. Joindre
les quatre extrémités donne le \omterm{def:g6:shapes:quadrilaterals}{losange} (les quatre \omterm{def:g5:solids:def}{sommets} sont aux
distances $3$~cm et $2$~cm de $O$ le long de \omterm{def:g6:lines:perp}{droites perpendiculaires},
donc les quatre côtés sont égaux par symétrie).
\end{solution}

\begin{solution}{exo:g6:shapes:9}
Un programme correct~:
1. tracer un \omterm{def:g6:lines:objects}{segment} $[BC]$ avec $BC = 4$~cm~;
2. tracer l'arc de centre $B$ et de \omterm{def:g3:shapes:shapes}{rayon} $6$~cm au-dessus de
$[BC]$~;
3. tracer l'arc de centre $C$ et de \omterm{def:g3:shapes:shapes}{rayon} $6$~cm au-dessus de
$[BC]$~;
4. nommer $A$ le point de croisement des deux arcs~;
5. tracer les \omterm{def:g6:lines:objects}{segments} $[AB]$ et $[AC]$.
\end{solution}

\begin{solution}{exo:g6:shapes:10}
Quelle que soit la position de $M$ sur le cercle, l'angle
$\widehat{AMB}$ mesure $90^\circ$~: le triangle $AMB$ est rectangle
en $M$ dès que $[AB]$ est un \omterm{def:g4:geometry:circle}{diamètre}. La démonstration attend
\cref{ch:g8:cosine}.
\end{solution}

\begin{solution}{exo:g6:shapes:11}
Des diagonales qui se coupent en leur milieu et ont des longueurs
égales~: c'est la liste des propriétés du \emph{rectangle}
(\cref{prop:g6:shapes:diagonals}). Rien n'oblige les diagonales à
être \omterm{def:g6:lines:perp}{perpendiculaires}, donc ce n'est pas forcément un \omterm{def:g6:shapes:quadrilaterals}{losange} ou un
carré --- un rectangle est la bonne réponse (un carré la
satisferait aussi, mais n'est pas forcé).
\end{solution}

\begin{solution}{exo:g6:shapes:12}
Avec $4 \times 4$ points, il y a $3 \times 3$ cellules de
quadrillage. Carrés alignés~: côtés $1$, $2$ ou $3$ donnent
$9 + 4 + 1 = 14$ carrés. Des carrés penchés existent aussi~: les
petits dont les côtés sont des diagonales de cellules $1 \times 1$
(comme $(1,0)$, $(2,1)$, $(1,2)$, $(0,1)$) --- quatre d'entre eux~;
et les plus grands dont les côtés sont des diagonales de rectangles
$2 \times 1$, comme $(1,0)$, $(3,1)$, $(2,3)$, $(0,2)$ --- deux
d'entre eux. Les quatre côtés d'un tel carré sont égaux parce qu'ils
sont des diagonales de rectangles identiques. Total~:
$14 + 4 + 2 = 20$ carrés. (Tout compte avec un exemple penché correct
et les $14$ alignés est une réussite à ce niveau.)
\end{solution}
