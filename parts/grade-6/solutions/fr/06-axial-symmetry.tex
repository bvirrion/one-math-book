\section*{Chapitre \ref{ch:g6:symmetry} --- La symétrie axiale}

\begin{solution}{exo:g6:symmetry:1}
$A'$ est à deux carreaux à \emph{droite} de $d$, sur la même ligne
que $A$~; $B'$ est à cinq carreaux à droite, un carreau plus haut ---
positions miroir.
\end{solution}

\begin{solution}{exo:g6:symmetry:2}
\omterm{def:g6:lines:perp}{Perpendiculaire} à $d$ par $M$, pied $H$~; puis $M'$ de l'autre côté
avec $HM' = MH = 3$~cm. La distance $MM'$ est $3 + 3 = 6$~cm.
\end{solution}

\begin{solution}{exo:g6:symmetry:3}
Axe vertical (comme A)~: A, H, I, M, O, T, U, V, W, X, Y.
Axe horizontal (comme B)~: B, C, D, E, H, I, K, O, X.
Les deux~: H, I, O, X.
(Les réponses exactes dépendent un peu de la façon dont les lettres
sont dessinées.)
\end{solution}

\begin{solution}{exo:g6:symmetry:4}
\omterm{def:g6:shapes:triangles}{Triangle équilatéral}~: $3$ axes. Rectangle~: $2$ (les \omterm{def:g6:symmetry:bisector}{médiatrices} de
ses côtés). \omterm{def:g6:shapes:quadrilaterals}{Losange}~: $2$ (ses diagonales). Carré~: $4$. Cercle~:
une infinité (toute droite passant par le centre).
\end{solution}

\begin{solution}{exo:g6:symmetry:5}
\emph{Faux.} Plier un rectangle (non carré) le long d'une diagonale
ne fait pas coïncider les deux \omterm{def:g3:division:half}{moitiés}~: les deux triangles ont la
même forme mais se placent différemment --- essayer avec une feuille
de papier. Seul pour un carré les diagonales fonctionnent comme axes.
\end{solution}

\begin{solution}{exo:g6:symmetry:6}
Les réflexions conservent les longueurs, les angles et les \omterm{def:g4:measure:perimeter}{périmètres}
(\cref{prop:g6:symmetry:preserved})~: $A'B' = 4$~cm,
$\widehat{B'A'C'} = 70^\circ$, \omterm{def:g4:measure:perimeter}{périmètre} de $A'B'C' = 12$~cm.
\end{solution}

\begin{solution}{exo:g6:symmetry:7}
Construction au compas comme dans \cref{met:g6:symmetry:bisector}.
Pour tout $M$ sur la droite, la règle confirme $MA = MB$
(\cref{prop:g6:symmetry:bisector}).
\end{solution}

\begin{solution}{exo:g6:symmetry:8}
Tous les points à égale distance de $A$ et de $B$ forment la
\omterm{def:g6:symmetry:bisector}{médiatrice} de $[AB]$ (\cref{prop:g6:symmetry:bisector}). La fontaine
peut se placer n'importe où sur cette droite (dans le jardin).
\end{solution}

\begin{solution}{exo:g6:symmetry:9}
Chaque \omterm{def:g5:solids:def}{sommet} est réfléchi perpendiculairement à l'axe~: pour l'axe à
$45^\circ$, un point à $k$ carreaux à droite de l'axe se pose à $k$
carreaux au-dessus (et inversement) --- les déplacements horizontaux
et verticaux sont échangés. Réfléchir les \omterm{def:g5:solids:def}{sommets} un par un, puis les
joindre dans l'ordre.
\end{solution}

\begin{solution}{exo:g6:symmetry:10}
Le \omterm{def:g6:symmetry:def}{symétrique} de $M$ est la seconde intersection du cercle avec la
\omterm{def:g6:lines:perp}{perpendiculaire} à $d$ par $M$ --- il est sur le cercle. Raison~: la
réflexion conserve les distances et fixe $O$ (qui est sur $d$), donc
l'image de tout point à distance $r$ de $O$ est encore à distance $r$
de $O$~: le cercle se transforme en lui-même.
\end{solution}

\begin{solution}{exo:g6:symmetry:11}
Chaque point de croisement $P$ a été tracé avec la même ouverture de
compas $\rho$ depuis $A$ et depuis $B$~: donc $PA = \rho = PB$. Par
\cref{prop:g6:symmetry:bisector}, un point équidistant de $A$ et de
$B$ se trouve sur la \omterm{def:g6:symmetry:bisector}{médiatrice} de $[AB]$~; deux tels points
déterminent la droite.
\end{solution}

\begin{solution}{exo:g6:symmetry:12}
Pour tout point de rebond $P$ sur le mur, $PB = PB'$ (la réflexion
par rapport à $d$ conserve les distances), donc la longueur du chemin
$AP + PB = AP + PB'$. La ligne brisée $A \to P \to B'$ est la plus
courte quand elle est droite, c'est-à-dire quand $P$ est sur le
\omterm{def:g6:lines:objects}{segment} $[AB']$. Donc le meilleur point de rebond est l'intersection
de $[AB']$ avec le mur.
\end{solution}
