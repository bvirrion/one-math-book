\section*{Chapitre \ref{ch:g6:propdata} --- Proportionnalité et données}

\begin{solution}{exo:g6:propdata:1}
Pommes~: \omterm{def:g3:division:remainder}{quotients} $\frac52 = 2.5$, $\frac{7.5}{3} = 2.5$,
$\frac{12.5}{5} = 2.5$ --- tous égaux~: proportionnel (prix
$2.50$ par kg).

Âge/taille~: $\frac{85}{2} = 42.5$ mais $\frac{103}{4} = 25.75$ ---
pas égaux~: non proportionnel.
\end{solution}

\begin{solution}{exo:g6:propdata:2}
Un croissant~: $4.80 \div 4 = 1.20$ euros. Sept~:
$7 \times 1.20 = 8.40$ euros.
\end{solution}

\begin{solution}{exo:g6:propdata:3}
Coefficient~: $18 \div 10 = 1.8$ euros par litre. Puis
$25$~L coûtent $25 \times 1.8 = 45$ euros~; $72$ euros achètent
$72 \div 1.8 = 40$~L~; $60$~L coûtent $60 \times 1.8 = 108$ euros.
\end{solution}

\begin{solution}{exo:g6:propdata:4}
Pour $250$~km~: $2.5$ fois le carburant de $100$~km, donc
$2.5 \times 6 = 15$~L. Pour $350$~km~: $3.5 \times 6 = 21$~L. Avec
$27$~L~: $27 \div 6 = 4.5$ centaines de km, c'est-à-dire $450$~km.
\end{solution}

\begin{solution}{exo:g6:propdata:5}
$50\,\%$ de $84$~: \omterm{def:g3:division:half}{moitié}, $42$. \quad
$25\,\%$ de $200$~: un \omterm{def:g3:division:half}{quart}, $50$. \quad
$10\,\%$ de $63$~: un dixième, $6.3$. \quad
$30\,\%$ de $70$~: $3 \times 7 = 21$.
\end{solution}

\begin{solution}{exo:g6:propdata:6}
$60\,\%$ de $450$~: $\frac{60}{100} \times 450 = 270$ élèves mangent
à la cantine~; $450 - 270 = 180$ n'y mangent pas.
\end{solution}

\begin{solution}{exo:g6:propdata:7}
Moins de $10$ livres~: mardi ($8$) et jeudi ($6$). Mercredi moins
jeudi~: $17 - 6 = 11$ livres de plus.
\end{solution}

\begin{solution}{exo:g6:propdata:8}
Diagramme en barres avec cinq barres de hauteurs $14$, $16$, $13$,
$17$, $20$ (graduation tous les $2$ ou $5$ degrés convient). Jour le
plus chaud~: vendredi ($20$ degrés).
\end{solution}

\begin{solution}{exo:g6:propdata:9}
Pour $10$ personnes, multiplier par $\frac{10}{6} = \frac53$~: farine
$450 \times \frac{10}{6} = 750$~g. Œufs~:
$3 \times \frac{10}{6} = 5$ --- heureusement un entier. (Si ce
n'était pas le cas, on arrondirait \emph{vers le haut} pour en avoir
assez.)
\end{solution}

\begin{solution}{exo:g6:propdata:10}
\emph{1.} En $2$~h~: $48$~km. En une demi-heure~: $12$~km. En
$1$~h $30$~: $24 + 12 = 36$~km.

\emph{2.} $60 \div 24 = 2.5$ heures, c'est-à-dire $2$~h $30$~min.
\end{solution}

\begin{solution}{exo:g6:propdata:11}
Ballon~: réduction $20\,\%$ de $15 = 3$ euros, nouveau prix $12$
euros. Raquette~: réduction $8$ euros, nouveau prix $32$ euros. Le
nouveau prix est $80\,\%$ de l'ancien dans tous les cas~:
proportionnel, coefficient $0.8$.
\end{solution}

\begin{solution}{exo:g6:propdata:12}
Après $t$ heures, la grosse bougie a brûlé $\frac t6$ de sa hauteur~:
elle se dresse à $1 - \frac t6$ de la hauteur initiale~; la fine à
$1 - \frac t4$. On veut
\[
1 - \frac t4 = \frac12 \left(1 - \frac t6\right)
\quad\text{c'est-à-dire}\quad
1 - \frac t4 = \frac12 - \frac t{12}.
\]
Alors $\frac12 = \frac t4 - \frac t{12} = \frac{3t - t}{12} =
\frac{t}{6}$, donc $t = 3$ heures. Vérif.~: après $3$~h la grosse
bougie se dresse à $1 - \frac36 = \frac12$ et la fine à
$1 - \frac34 = \frac14$ --- bien la \omterm{def:g3:division:half}{moitié}.
\end{solution}
