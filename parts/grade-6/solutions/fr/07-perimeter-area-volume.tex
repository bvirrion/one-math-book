\section*{Chapitre \ref{ch:g6:measure} --- Périmètre, aire, volume}

\begin{solution}{exo:g6:measure:1}
$3.5$~m $= 350$~cm~; \quad $420$~mm $= 42$~cm~; \quad
$0.8$~km $= 800$~m~; \quad $25\,000$~m $= 25$~km.
\end{solution}

\begin{solution}{exo:g6:measure:2}
Rectangle~: $2 \times (8 + 3.5) = 2 \times 11.5 = 23$~cm.
Carré~: $4 \times 6.2 = 24.8$~cm.
Triangle~: $5 + 7 + 9 = 21$~cm.
\end{solution}

\begin{solution}{exo:g6:measure:3}
Un tour~: $2\pi \times 50 = 100\pi \approx 314$~m. Pour $2$~km
$= 2000$~m~: $2000 \div 314 \approx 6.4$, donc $7$ tours complets
sont nécessaires ($6$ tours ne font qu'environ $1\,885$~m).
\end{solution}

\begin{solution}{exo:g6:measure:4}
Rectangle~: $9 \times 4 = 36$~cm$^2$. Carré~: $7^2 = 49$~m$^2$.
\omterm{def:g6:shapes:triangles}{Triangle rectangle}~: $\frac{6 \times 10}{2} = 30$~cm$^2$.
Disque~: $\pi \times 3^2 = 9\pi \approx 28$~cm$^2$.
\end{solution}

\begin{solution}{exo:g6:measure:5}
$3$~m$^2 = 30\,000$~cm$^2$~; \quad
$45\,000$~cm$^2 = 4.5$~m$^2$~; \quad
$2.5$~L $= 2\,500$~cm$^3$~; \quad
$4\,500$~L $= 4.5$~m$^3$.
\end{solution}

\begin{solution}{exo:g6:measure:6}
Clôture (\omterm{def:g6:measure:perimeter}{périmètre})~: $2 \times (120 + 85) = 2 \times 205 = 410$~m.
\omterm{def:g6:measure:area}{Aire}~: $120 \times 85 = 10\,200$~m$^2$.
\end{solution}

\begin{solution}{exo:g6:measure:7}
\omterm{def:g5:solids:def}{Pavé}~: $5 \times 4 \times 10 = 200$~cm$^3$. \omterm{def:g5:solids:def}{Cube}~:
$3 \times 3 \times 3 = 27$~cm$^3$.
\end{solution}

\begin{solution}{exo:g6:measure:8}
\omterm{def:g6:measure:perimeter}{Périmètre} $16$~cm signifie longueur $+$ largeur $= 8$~cm. Par
exemple $6 \times 2$ (\omterm{def:g6:measure:area}{aire} $12$~cm$^2$) et $5 \times 3$ (\omterm{def:g6:measure:area}{aire}
$15$~cm$^2$) --- ou le carré $4 \times 4$ (\omterm{def:g6:measure:area}{aire} $16$~cm$^2$). Plus
on se rapproche d'un carré, plus l'\omterm{def:g6:measure:area}{aire} est grande.
\end{solution}

\begin{solution}{exo:g6:measure:9}
\omterm{def:g6:measure:area}{Aire}~: $10 \times 2 + 2 \times 6 = 20 + 12 = 32$~cm$^2$.

\omterm{def:g6:measure:perimeter}{Périmètre} (tige centrée sous la barre)~: en parcourant le bord,
\[
10 + 2 + 4 + 6 + 2 + 6 + 4 + 2 = 36 \text{ cm}
\]
(haut~; extrémité droite de la barre~; dessous droit~; droite de la
tige~; bas de la tige~; gauche de la tige~; dessous gauche~;
extrémité gauche de la barre).
\end{solution}

\begin{solution}{exo:g6:measure:10}
\emph{1.} $V = 25 \times 10 \times 2 = 500$~m$^3$.

\emph{2.} $500$~m$^3 = 500 \times 1000 = 500\,000$~L.

\emph{3.} $500\,000 \div 50\,000 = 10$ heures.
\end{solution}

\begin{solution}{exo:g6:measure:11}
Jardin~: $20^2 = 400$~m$^2$. Étang~: $\pi \times 3^2 = 9\pi$~m$^2$.
Gazon~: $400 - 9\pi \approx 400 - 28.3 \approx 372$~m$^2$.
\end{solution}

\begin{solution}{exo:g6:measure:12}
\emph{1.} Petite tablette~: $15 \times 6 \times 1 = 90$~cm$^3$.
Grande~: $30 \times 12 \times 2 = 720$~cm$^3$. Le \omterm{def:g6:measure:volume}{volume} est
multiplié par $720 \div 90 = 8$ --- doubler chacune des trois
dimensions multiplie le \omterm{def:g6:measure:volume}{volume} par $2 \times 2 \times 2 = 8$.

\emph{2.} Huit fois le chocolat pour quatre fois le prix~: oui, la
grande tablette est deux fois plus avantageuse au gramme.
\end{solution}
