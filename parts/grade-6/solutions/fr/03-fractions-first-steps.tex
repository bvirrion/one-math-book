\section*{Chapitre \ref{ch:g6:fractions} --- Fractions~: premiers pas}

\begin{solution}{exo:g6:fractions:1}
Gâteau~: $\frac38$ mangés. \quad Chocolat~: $\frac{7}{12}$. \quad
Tarte~: $\frac66 = 1$ (toute la tarte).
\end{solution}

\begin{solution}{exo:g6:fractions:2}
Quatre parts sur cinq sont coloriées. $\frac45 < 1$ (quatre parts sur
cinq, c'est moins que la barre entière)~; $\frac65 > 1$ (six cinquièmes,
c'est une barre entière et un cinquième de plus).
\end{solution}

\begin{solution}{exo:g6:fractions:3}
Sur la droite graduée en tiers~: $\frac13$ est une graduation après
$0$~; $\frac53$ est une graduation avant $2$ (car $2 = \frac63$)~;
$\frac73$ est une graduation après $2$. Donc $\frac53$ et $\frac73$
sont tous deux à distance $\frac13$ de $2$.
\end{solution}

\begin{solution}{exo:g6:fractions:4}
$\frac94 = 2 + \frac14$~; \quad
$\frac{13}{5} = 2 + \frac35$~; \quad
$\frac{12}{6} = 2$ (exactement)~; \quad
$\frac{25}{8} = 3 + \frac18$.
\end{solution}

\begin{solution}{exo:g6:fractions:5}
$\frac23 = \frac{8}{12}$ (multiplier par $4$)~; \quad
$\frac{15}{20} = \frac34$ (diviser par $5$)~; \quad
$\frac{7}{10} = \frac{21}{30}$ (multiplier par $3$)~; \quad
$\frac56 = \frac{10}{12}$ (diviser par $2$).
\end{solution}

\begin{solution}{exo:g6:fractions:6}
$\frac68 = \frac34$~; \quad
$\frac{15}{25} = \frac35$~; \quad
$\frac{18}{24} = \frac{9}{12} = \frac34$~; \quad
$\frac{35}{7} = 5$ (un entier).
\end{solution}

\begin{solution}{exo:g6:fractions:7}
$\frac12$ de $86$~: $86 \div 2 = 43$.

$\frac34$ de $60$~: $60 \div 4 = 15$, puis $15 \times 3 = 45$.

$\frac25$ de $35$~: $35 \div 5 = 7$, puis $7 \times 2 = 14$.

$\frac{7}{10}$ de $250$~: $250 \div 10 = 25$, puis $25 \times 7 = 175$.
\end{solution}

\begin{solution}{exo:g6:fractions:8}
$\frac12 > \frac13$~: les \omterm{def:g3:division:half}{moitiés} sont des parts plus grandes que les
tiers. $\frac35 > \frac25$~: mêmes parts, plus d'entre elles.
$\frac34 = \frac68 < \frac78$~: sept huitièmes battent six huitièmes.
\end{solution}

\begin{solution}{exo:g6:fractions:9}
$\frac12 = 0.5$~; \quad $\frac34 = 0.75$~; \quad
$\frac{7}{10} = 0.7$~; \quad $\frac94 = 2.25$. L'intrus est
$\frac13 = 0.333\dots$, dont l'écriture décimale ne s'arrête jamais~:
ce n'est pas un \omterm{def:g6:decimals:places}{nombre décimal}.
\end{solution}

\begin{solution}{exo:g6:fractions:10}
Usagers du bus~: $\frac34$ de $28$~: $28 \div 4 = 7$, puis
$7 \times 3 = 21$ élèves. À pied~: $28 - 21 = 7$ élèves (c'est le
\omterm{def:g3:division:half}{quart} restant~: $28 \div 4 = 7$). Vérif.~: $21 + 7 = 28$.
\end{solution}

\begin{solution}{exo:g6:fractions:11}
Les deux tiers des économies valent $18$ euros, donc \emph{un} tiers
vaut $18 \div 2 = 9$ euros, et les économies entières (trois tiers)
étaient $9 \times 3 = 27$ euros. Vérif.~: $\frac23$ de $27$ est $18$.
\end{solution}

\begin{solution}{exo:g6:fractions:12}
L'eau utilisée est passée de $\frac58$ à $\frac48$ (une \omterm{def:g3:division:half}{moitié}, c'est
quatre huitièmes) du réservoir~: les $10$ litres représentent
$\frac58 - \frac48 = \frac18$ de la capacité. Donc le réservoir plein
tient $10 \times 8 = 80$ litres. Vérif.~: $\frac58$ de $80$ est $50$,
moins $10$ laisse $40$, qui est bien la \omterm{def:g3:division:half}{moitié} de $80$.
\end{solution}
