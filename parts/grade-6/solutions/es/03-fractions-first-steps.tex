\begin{solution}{exo:g6:fractions:7}
$\frac12$ of $86$: $86 \div 2 = 43$.

$\frac34$ de $60$: $60 \div 4 = 15$, luego $15 \times 3 = 45$.

$\frac25$ de $35$: $35 \div 5 = 7$, luego $7 \times 2 = 14$.

$\frac{7}{10}$ de $250$: $250 \div 10 = 25$, luego $25 \times 7 = 175$.
\end{solution}

\begin{solution}{exo:g6:fractions:10}
Usuarios de autobús: $\frac34$ de $28$: $28 \div 4 = 7$, luego
$7 \times 3 = 21$ estudiantes. Caminantes: $28 - 21 = 7$ estudiantes (es decir
el restante \omterm{def:g3:division:half}{cuarto}: $28 \div 4 = 7$). Verificar: $21 + 7 = 28$.
\end{solution}

\begin{solution}{exo:g6:fractions:11}
Dos tercios del ahorro son $18$ euros, por lo que \emph{uno} tercio es
$18 \div 2 = 9$ euros, y el ahorro total (tres tercios) se
$9 \times 3 = 27$ euros. Verifique: $\frac23$ de $27$ es $18$.
\end{solution}

\begin{solution}{exo:g6:fractions:12}

$\frac58 - \frac48 = \frac18$ del \omterm{def:g2:measure:units}{capacidad}. Entonces el tanque lleno aguanta

\end{solution}

\begin{solution}{pb:g6:fractions:1}
\textbf{1.} \omterm{def:g3:division:half}{Medio} de $16$ cuadrados es $8$ cuadrados. Como \omterm{def:g6:fractions:def}{fracción}:
$\frac12$ de la barra, y dado que cada \omterm{def:g3:division:half}{medio} es $8$ cuadrados fuera de
$16$, también $\frac{8}{16}$ --- el mismo \omterm{def:g6:fractions:def}{fracción} escrito de dos maneras
(\cref{prop:g6:fractions:equal}).

\textbf{2.} \omterm{def:g3:division:half}{Medio} de los $8$ cuadrados restantes son $4$ cuadrados,
que es $\frac{4}{16}$ de la barra. Simplificando por $4$:
$\frac{4}{16} = \frac14$. En la foto: cortando el resto
\omterm{def:g3:division:half}{medio} en dos produce dos \omterm{def:g3:division:half}{cuarteles} de la barra original --- \omterm{def:g3:division:half}{medio} de
un \omterm{def:g3:division:half}{medio} es un \omterm{def:g3:division:half}{cuarto}.

\textbf{3.} Tercer bocado: \omterm{def:g3:division:half}{medio} de $4$ cuadrados son $2$ cuadrados, es decir\
$\frac{2}{16} = \frac18$ de la barra. Cuarto bocado: $1$ cuadrado,

\emph{dobles}: $2$, $4$, $8$, $16$.

\textbf{4.} Comido: $8 + 4 + 2 + 1 = 15$ cuadrados de $16$, que
