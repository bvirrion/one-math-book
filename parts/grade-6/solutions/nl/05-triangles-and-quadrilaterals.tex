\section*{Hoofdstuk \ref{ch:g6:shapes} --- Driehoeken en vierhoeken}

\begin{solution}{exo:g6:shapes:1}
Volg de methode: \omterm{def:g6:lines:objects}{lijnstuk} $[AB]$ van $6$ cm, boog met \omterm{def:g3:shapes:shapes}{straal} $5$ cm
en middelpunt $A$, boog met \omterm{def:g3:shapes:shapes}{straal} $4$ cm en middelpunt $B$; het
snijpunt is $C$.
\end{solution}

\begin{solution}{exo:g6:shapes:2}
Basis $[EF]$ van $3$ cm, daarna twee bogen met \omterm{def:g3:shapes:shapes}{straal} $5$ cm en
middelpunten $E$ en $F$; hun snijpunt is $D$. De twee basishoeken
$\widehat{DEF}$ en $\widehat{DFE}$ zijn gelijk (ongeveer $72.5^\circ$
elk) --- symmetrie van de \omterm{def:g6:shapes:triangles}{gelijkbenige driehoek}
(\cref{ex:g6:symmetry:proves}).
\end{solution}

\begin{solution}{exo:g6:shapes:3}
\omterm{def:g6:lines:objects}{Lijnstuk} van $4$ cm, daarna twee bogen met \omterm{def:g3:shapes:shapes}{straal} $4$ cm en
middelpunten in de uiteinden. Elke hoek meet $60^\circ$.
\end{solution}

\begin{solution}{exo:g6:shapes:4}
(a) \omterm{def:g6:shapes:triangles}{gelijkzijdig} (dus ook \omterm{def:g6:shapes:triangles}{gelijkbenig}). \quad
(b) \omterm{def:g6:shapes:triangles}{gelijkbenig} ($5 = 5$). \quad
(c) \omterm{def:g6:shapes:triangles}{rechthoekig} en \omterm{def:g6:shapes:triangles}{gelijkbenig} (twee gelijke hoeken van $45^\circ$
staan tegenover twee gelijke zijden). \quad
(d) \omterm{def:g6:shapes:triangles}{gelijkzijdig} (drie gelijke hoeken).
\end{solution}

\begin{solution}{exo:g6:shapes:5}
De schuine zijde is de zijde tegenover de \omterm{def:g3:shapes:rightangle}{rechte hoek}, altijd de
langste zijde. In $MNP$ \omterm{def:g6:shapes:triangles}{rechthoekig} in $N$ is de schuine zijde
$[MP]$: zij raakt nooit het \omterm{def:g5:solids:def}{hoekpunt} van de \omterm{def:g3:shapes:rightangle}{rechte hoek}, $N$.
\end{solution}

\begin{solution}{exo:g6:shapes:6}
\emph{1. Waar}: een vierkant heeft vier \omterm{def:g3:shapes:rightangle}{rechte hoeken}, wat de
definitie van een rechthoek is.

\emph{2. Onwaar}: een rechthoek $5 \times 3$ heeft ongelijke zijden.

\emph{3. Waar}: een vierkant heeft vier gelijke zijden, de definitie
van een \omterm{def:g6:shapes:quadrilaterals}{ruit}.

\emph{4. Waar}: zie \cref{ex:g6:shapes:recognize}.
\end{solution}

\begin{solution}{exo:g6:shapes:7}
Constructie met loodlijnen, zoals in \cref{ex:g6:shapes:program}.
Beide diagonalen meten ongeveer $5.8$ cm: de diagonalen van een
rechthoek hebben dezelfde lengte (\cref{prop:g6:shapes:diagonals}).
\end{solution}

\begin{solution}{exo:g6:shapes:8}
Teken een \omterm{def:g6:lines:objects}{lijnstuk} van $6$ cm; zijn midden $O$; de loodlijn in $O$;
daarop twee punten op $2$ cm van $O$ aan weerszijden. De vier
uiteinden verbinden geeft de \omterm{def:g6:shapes:quadrilaterals}{ruit} (alle vier \omterm{def:g5:solids:def}{hoekpunten} liggen op
afstanden $3$ cm en $2$ cm van $O$ langs \omterm{def:g6:lines:perp}{loodrechte lijnen}, dus de
vier zijden zijn gelijk door symmetrie).
\end{solution}

\begin{solution}{exo:g6:shapes:9}
Een correct programma:
1. teken een \omterm{def:g6:lines:objects}{lijnstuk} $[BC]$ met $BC = 4$ cm;
2. teken de boog met middelpunt $B$ en \omterm{def:g3:shapes:shapes}{straal} $6$ cm boven $[BC]$;
3. teken de boog met middelpunt $C$ en \omterm{def:g3:shapes:shapes}{straal} $6$ cm boven $[BC]$;
4. noem $A$ het snijpunt van de twee bogen;
5. teken de lijnstukken $[AB]$ en $[AC]$.
\end{solution}

\begin{solution}{exo:g6:shapes:10}
Welke positie $M$ op de cirkel ook heeft, de hoek $\widehat{AMB}$
meet $90^\circ$: de driehoek $AMB$ is \omterm{def:g6:shapes:triangles}{rechthoekig} in $M$ wanneer
$[AB]$ een middellijn is. Het bewijs wacht op \cref{ch:g8:cosine}.
\end{solution}

\begin{solution}{exo:g6:shapes:11}
Diagonalen die elkaar in hun midden snijden en gelijke lengten hebben:
dat is de eigenschappenlijst van de \emph{rechthoek}
(\cref{prop:g6:shapes:diagonals}). Niets dwingt de diagonalen
\omterm{def:g6:lines:perp}{loodrecht} te zijn, dus het hoeft geen \omterm{def:g6:shapes:quadrilaterals}{ruit} of vierkant te zijn ---
een rechthoek is het juiste antwoord (een vierkant zou het ook
voldoen, maar is niet verplicht).
\end{solution}

\begin{solution}{exo:g6:shapes:12}
Met $4 \times 4$ stippen zijn er $3 \times 3$ roostercellen.
Roostergelijnde vierkanten: zijden $1$, $2$ of $3$ geven
$9 + 4 + 1 = 14$ vierkanten. Schuine vierkanten bestaan ook: de
kleine waarvan de zijden diagonalen van $1 \times 1$-cellen zijn
(zoals $(1,0)$, $(2,1)$, $(1,2)$, $(0,1)$) --- vier ervan; en de
grotere waarvan de zijden diagonalen van $2 \times 1$-rechthoeken
zijn, zoals $(1,0)$, $(3,1)$, $(2,3)$, $(0,2)$ --- twee ervan. Alle
vier zijden van zo'n vierkant zijn gelijk omdat ze diagonalen van
identieke rechthoeken zijn. Totaal: $14 + 4 + 2 = 20$ vierkanten.
(Elke telling met een correct schuin voorbeeld en de $14$
uitgelijnde telt als geslaagd op dit niveau.)
\end{solution}
