\section*{Hoofdstuk \ref{ch:g6:decimals} --- Decimale getallen}

\begin{solution}{exo:g6:decimals:1}
$5.37$; \quad $0.9$; \quad $12.004$; \quad $8.05$.
\end{solution}

\begin{solution}{exo:g6:decimals:2}
Tienden-cijfer: $3$; honderdsten-cijfer: $5$; de $8$ telt de
tientallen.
\[
86.354 = 8 \times 10 + 6 + 3 \times \frac{1}{10} + 5 \times
\frac{1}{100} + 4 \times \frac{1}{1000}.
\]
\end{solution}

\begin{solution}{exo:g6:decimals:3}
$5.3 > 5.29$ (vergelijk $5.30$ met $5.29$); \quad
$0.7 = 0.70$ (eindnullen veranderen niets); \quad
$2.09 < 2.9$ (tienden: $0 < 9$); \quad
$14.5 > 14.49$.
\end{solution}

\begin{solution}{exo:g6:decimals:4}
Vul aan tot twee decimalen: $4.20$; $4.05$; $4.51$; $4.15$; $4.50$.
Volgorde:
\[
4.05 < 4.15 < 4.2 < 4.5 < 4.51 .
\]
\end{solution}

\begin{solution}{exo:g6:decimals:5}
Elke streep is één tiende. $A$: $6.3$; \quad $B$: $6.7$; \quad
$C$: $7.2$.
\end{solution}

\begin{solution}{exo:g6:decimals:6}
$45.80 + 7.65 = 53.45$.

$23.40 - 8.72 = 14.68$ (controle: $14.68 + 8.72 = 23.4$).

$56 \times 24 = 1344$, en twee decimale cijfers in de factoren:
$5.6 \times 2.4 = 13.44$.
\end{solution}

\begin{solution}{exo:g6:decimals:7}
$6.42 \times 10 = 64.2$; \quad
$0.35 \times 1000 = 350$; \quad
$78.1 \div 100 = 0.781$; \quad
$5 \div 1000 = 0.005$.
\end{solution}

\begin{solution}{exo:g6:decimals:8}
$2.4$ m $= 240$ cm; \quad $370$ g $= 0.37$ kg (deel door $1000$);
\quad $0.85$ km $= 850$ m; \quad $56$ mm $= 0.056$ m.
\end{solution}

\begin{solution}{exo:g6:decimals:9}
Op de eenheid: $24$ (volgend cijfer $8$: naar boven). Op de tiende:
$23.9$. Op de honderdste: $23.87$. En $9.97$ op de tiende: het
volgende cijfer is $7$, dus de $9$ tienden afronden naar boven ---
$10.0$.
\end{solution}

\begin{solution}{exo:g6:decimals:10}
Prijs van de stokbroden: $3 \times 1.15 = 3.45$. Wisselgeld:
$5 - 3.45 = 1.55$.
\end{solution}

\begin{solution}{exo:g6:decimals:11}
Tussen $7.4$ en $7.5$: bijvoorbeeld $7.45$ (of $7.41$, $7.499$,
\dots). Tussen $3.99$ en $4$: bijvoorbeeld $3.995$. In beide gevallen
zijn er \emph{oneindig veel} zulke getallen: je kunt altijd meer
decimalen toevoegen.
\end{solution}

\begin{solution}{exo:g6:decimals:12}
Grootste: zet de grootste cijfers vooraan en de punt zo laat mogelijk:
$85.2$. Kleinste: kleinste cijfers vooraan en de punt zo vroeg
mogelijk: $2.58$.
\end{solution}
