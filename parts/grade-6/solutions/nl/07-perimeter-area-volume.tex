\section*{Hoofdstuk \ref{ch:g6:measure} --- Omtrek, oppervlakte, volume}

\begin{solution}{exo:g6:measure:1}
$3.5$ m $= 350$ cm; \quad $420$ mm $= 42$ cm; \quad
$0.8$ km $= 800$ m; \quad $25\,000$ m $= 25$ km.
\end{solution}

\begin{solution}{exo:g6:measure:2}
Rechthoek: $2 \times (8 + 3.5) = 2 \times 11.5 = 23$ cm.
Vierkant: $4 \times 6.2 = 24.8$ cm.
Driehoek: $5 + 7 + 9 = 21$ cm.
\end{solution}

\begin{solution}{exo:g6:measure:3}
Eén ronde: $2\pi \times 50 = 100\pi \approx 314$ m. Voor $2$ km
$= 2000$ m: $2000 \div 314 \approx 6.4$, dus $7$ volle rondes zijn
nodig ($6$ rondes maken slechts ongeveer $1\,885$ m).
\end{solution}

\begin{solution}{exo:g6:measure:4}
Rechthoek: $9 \times 4 = 36$ cm$^2$. Vierkant: $7^2 = 49$ m$^2$.
\omterm{def:g6:shapes:triangles}{Rechthoekige driehoek}: $\frac{6 \times 10}{2} = 30$ cm$^2$.
Schijf: $\pi \times 3^2 = 9\pi \approx 28$ cm$^2$.
\end{solution}

\begin{solution}{exo:g6:measure:5}
$3$ m$^2 = 30\,000$ cm$^2$; \quad
$45\,000$ cm$^2 = 4.5$ m$^2$; \quad
$2.5$ L $= 2\,500$ cm$^3$; \quad
$4\,500$ L $= 4.5$ m$^3$.
\end{solution}

\begin{solution}{exo:g6:measure:6}
Hek (\omterm{def:g6:measure:perimeter}{omtrek}): $2 \times (120 + 85) = 2 \times 205 = 410$ m.
\omterm{def:g6:measure:area}{Oppervlakte}: $120 \times 85 = 10\,200$ m$^2$.
\end{solution}

\begin{solution}{exo:g6:measure:7}
\omterm{def:g5:solids:def}{Balk}: $5 \times 4 \times 10 = 200$ cm$^3$. \omterm{def:g5:solids:def}{Kubus}:
$3 \times 3 \times 3 = 27$ cm$^3$.
\end{solution}

\begin{solution}{exo:g6:measure:8}
\omterm{def:g6:measure:perimeter}{Omtrek} $16$ cm betekent lengte $+$ breedte $= 8$ cm. Bijvoorbeeld
$6 \times 2$ (\omterm{def:g6:measure:area}{oppervlakte} $12$ cm$^2$) en $5 \times 3$ (\omterm{def:g6:measure:area}{oppervlakte}
$15$ cm$^2$) --- of het vierkant $4 \times 4$ (\omterm{def:g6:measure:area}{oppervlakte}
$16$ cm$^2$). Hoe dichter bij een vierkant, hoe groter de
\omterm{def:g6:measure:area}{oppervlakte}.
\end{solution}

\begin{solution}{exo:g6:measure:9}
\omterm{def:g6:measure:area}{Oppervlakte}: $10 \times 2 + 2 \times 6 = 20 + 12 = 32$ cm$^2$.

\omterm{def:g6:measure:perimeter}{Omtrek} (steel gecentreerd onder de balk): rondlopend,
\[
10 + 2 + 4 + 6 + 2 + 6 + 4 + 2 = 36 \text{ cm}
\]
(bovenkant; rechteruiteinde van de balk; onderkant rechts; rechterkant
van de steel; onderkant van de steel; linkerkant van de steel;
onderkant links; linkeruiteinde van de balk).
\end{solution}

\begin{solution}{exo:g6:measure:10}
\emph{1.} $V = 25 \times 10 \times 2 = 500$ m$^3$.

\emph{2.} $500$ m$^3 = 500 \times 1000 = 500\,000$ L.

\emph{3.} $500\,000 \div 50\,000 = 10$ uur.
\end{solution}

\begin{solution}{exo:g6:measure:11}
Tuin: $20^2 = 400$ m$^2$. Vijver: $\pi \times 3^2 = 9\pi$ m$^2$.
Gras: $400 - 9\pi \approx 400 - 28.3 \approx 372$ m$^2$.
\end{solution}

\begin{solution}{exo:g6:measure:12}
\emph{1.} Kleine reep: $15 \times 6 \times 1 = 90$ cm$^3$. Grote
reep: $30 \times 12 \times 2 = 720$ cm$^3$. Het \omterm{def:g6:measure:volume}{volume} wordt
vermenigvuldigd met $720 \div 90 = 8$ --- elke van de drie afmetingen
verdubbelen vermenigvuldigt het \omterm{def:g6:measure:volume}{volume} met $2 \times 2 \times 2 = 8$.

\emph{2.} Acht keer de chocolade voor vier keer de prijs: ja, de
grote reep is tweemaal zo voordelig per gram.
\end{solution}
