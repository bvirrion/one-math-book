\section*{Hoofdstuk \ref{ch:g6:fractions} --- Breuken: eerste stappen}

\begin{solution}{exo:g6:fractions:1}
Taart: $\frac38$ opgegeten. \quad Chocolade: $\frac{7}{12}$. \quad
Taart: $\frac66 = 1$ (de hele taart).
\end{solution}

\begin{solution}{exo:g6:fractions:2}
Vier van de vijf delen zijn gekleurd. $\frac45 < 1$ (vier van de vijf
delen is minder dan de hele balk); $\frac65 > 1$ (zes vijfden is één
hele balk en nog één vijfde).
\end{solution}

\begin{solution}{exo:g6:fractions:3}
Op de lijn in derden: $\frac13$ is één streep na $0$;
$\frac53$ is één streep vóór $2$ (want $2 = \frac63$); $\frac73$ is
één streep na $2$. Dus $\frac53$ en $\frac73$ liggen beide op afstand
$\frac13$ van $2$.
\end{solution}

\begin{solution}{exo:g6:fractions:4}
$\frac94 = 2 + \frac14$; \quad
$\frac{13}{5} = 2 + \frac35$; \quad
$\frac{12}{6} = 2$ (exact); \quad
$\frac{25}{8} = 3 + \frac18$.
\end{solution}

\begin{solution}{exo:g6:fractions:5}
$\frac23 = \frac{8}{12}$ (vermenigvuldig met $4$); \quad
$\frac{15}{20} = \frac34$ (deel door $5$); \quad
$\frac{7}{10} = \frac{21}{30}$ (vermenigvuldig met $3$); \quad
$\frac56 = \frac{10}{12}$ (deel door $2$).
\end{solution}

\begin{solution}{exo:g6:fractions:6}
$\frac68 = \frac34$; \quad
$\frac{15}{25} = \frac35$; \quad
$\frac{18}{24} = \frac{9}{12} = \frac34$; \quad
$\frac{35}{7} = 5$ (een geheel getal).
\end{solution}

\begin{solution}{exo:g6:fractions:7}
$\frac12$ van $86$: $86 \div 2 = 43$.

$\frac34$ van $60$: $60 \div 4 = 15$, daarna $15 \times 3 = 45$.

$\frac25$ van $35$: $35 \div 5 = 7$, daarna $7 \times 2 = 14$.

$\frac{7}{10}$ van $250$: $250 \div 10 = 25$, daarna $25 \times 7 = 175$.
\end{solution}

\begin{solution}{exo:g6:fractions:8}
$\frac12 > \frac13$: helen zijn grotere delen dan derden.
$\frac35 > \frac25$: dezelfde delen, meer ervan.
$\frac34 = \frac68 < \frac78$: zeven achtsten winnen van zes achtsten.
\end{solution}

\begin{solution}{exo:g6:fractions:9}
$\frac12 = 0.5$; \quad $\frac34 = 0.75$; \quad
$\frac{7}{10} = 0.7$; \quad $\frac94 = 2.25$. De uitzondering is
$\frac13 = 0.333\dots$, waarvan de decimale schrijfwijze nooit stopt:
het is geen decimaal getal.
\end{solution}

\begin{solution}{exo:g6:fractions:10}
Busreizigers: $\frac34$ van $28$: $28 \div 4 = 7$, daarna
$7 \times 3 = 21$ leerlingen. Lopenden: $28 - 21 = 7$ leerlingen (dat
is het resterende kwart: $28 \div 4 = 7$). Controle: $21 + 7 = 28$.
\end{solution}

\begin{solution}{exo:g6:fractions:11}
Twee derden van het spaargeld zijn $18$ euro, dus \emph{één} derde is
$18 \div 2 = 9$ euro, en het hele spaargeld (drie derden) was
$9 \times 3 = 27$ euro. Controle: $\frac23$ van $27$ is $18$.
\end{solution}

\begin{solution}{exo:g6:fractions:12}
Het gebruikte water ging van $\frac58$ naar $\frac48$ (een half is
vier achtsten) van de tank: de $10$ liter stellen
$\frac58 - \frac48 = \frac18$ van de capaciteit voor. Dus de volle
tank houdt $10 \times 8 = 80$ liter. Controle: $\frac58$ van $80$ is
$50$, min $10$ blijft $40$, wat inderdaad de helft van $80$ is.
\end{solution}
