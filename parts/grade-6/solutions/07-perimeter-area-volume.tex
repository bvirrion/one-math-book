\section*{Chapter \ref{ch:g6:measure} --- Perimeter, Area, Volume}

\begin{solution}{exo:g6:measure:1}
$3.5$ m $= 350$ cm; \quad $420$ mm $= 42$ cm; \quad
$0.8$ km $= 800$ m; \quad $25\,000$ m $= 25$ km.
\end{solution}

\begin{solution}{exo:g6:measure:2}
Rectangle: $2 \times (8 + 3.5) = 2 \times 11.5 = 23$ cm.
Square: $4 \times 6.2 = 24.8$ cm.
Triangle: $5 + 7 + 9 = 21$ cm.
\end{solution}

\begin{solution}{exo:g6:measure:3}
One lap: $2\pi \times 50 = 100\pi \approx 314$ m. For $2$ km
$= 2000$ m: $2000 \div 314 \approx 6.4$, so $7$ full laps are needed
($6$ laps only make about $1\,885$ m).
\end{solution}

\begin{solution}{exo:g6:measure:4}
Rectangle: $9 \times 4 = 36$ cm$^2$. Square: $7^2 = 49$ m$^2$.
\omterm{def:g6:shapes:triangles}{Right triangle}: $\frac{6 \times 10}{2} = 30$ cm$^2$.
Disk: $\pi \times 3^2 = 9\pi \approx 28$ cm$^2$.
\end{solution}

\begin{solution}{exo:g6:measure:5}
$3$ m$^2 = 30\,000$ cm$^2$; \quad
$45\,000$ cm$^2 = 4.5$ m$^2$; \quad
$2.5$ L $= 2\,500$ cm$^3$; \quad
$4\,500$ L $= 4.5$ m$^3$.
\end{solution}

\begin{solution}{exo:g6:measure:6}
Fence (\omterm{def:g6:measure:perimeter}{perimeter}): $2 \times (120 + 85) = 2 \times 205 = 410$ m.
\omterm{def:g6:measure:area}{Area}: $120 \times 85 = 10\,200$ m$^2$.
\end{solution}

\begin{solution}{exo:g6:measure:7}
\omterm{def:g5:solids:def}{Box}: $5 \times 4 \times 10 = 200$ cm$^3$. \omterm{def:g5:solids:def}{Cube}:
$3 \times 3 \times 3 = 27$ cm$^3$.
\end{solution}

\begin{solution}{exo:g6:measure:8}
\omterm{def:g6:measure:perimeter}{Perimeter} $16$ cm means length $+$ width $= 8$ cm. For instance
$6 \times 2$ (\omterm{def:g6:measure:area}{area} $12$ cm$^2$) and $5 \times 3$ (\omterm{def:g6:measure:area}{area} $15$ cm$^2$)
--- or the square $4 \times 4$ (\omterm{def:g6:measure:area}{area} $16$ cm$^2$). The closer to a
square, the larger the \omterm{def:g6:measure:area}{area}.
\end{solution}

\begin{solution}{exo:g6:measure:9}
\omterm{def:g6:measure:area}{Area}: $10 \times 2 + 2 \times 6 = 20 + 12 = 32$ cm$^2$.

\omterm{def:g6:measure:perimeter}{Perimeter} (stem centered under the bar): walking around,

\begin{solution}{exo:g6:measure:10}
\emph{1.} $V = 25 \times 10 \times 2 = 500$ m$^3$.

\emph{2.} $500$ m$^3 = 500 \times 1000 = 500\,000$ L.

\emph{3.} $500\,000 \div 50\,000 = 10$ hours.
\end{solution}

\begin{solution}{exo:g6:measure:11}
Garden: $20^2 = 400$ m$^2$. Pond: $\pi \times 3^2 = 9\pi$ m$^2$.
Grass: $400 - 9\pi \approx 400 - 28.3 \approx 372$ m$^2$.
\end{solution}

\begin{solution}{exo:g6:measure:12}
\emph{1.} Small bar: $15 \times 6 \times 1 = 90$ cm$^3$. Big bar:
$30 \times 12 \times 2 = 720$ cm$^3$. The \omterm{def:g6:measure:volume}{volume} is multiplied by
$720 \div 90 = 8$ --- doubling each of the three dimensions multiplies
the \omterm{def:g6:measure:volume}{volume} by $2 \times 2 \times 2 = 8$.

\emph{2.} Eight times the chocolate for four times the price: yes, the
big bar is twice as good a deal per gram.
\end{solution}

\begin{solution}{pb:g6:measure:1}
\textbf{1.} $1 \times 9$: \omterm{def:g6:measure:perimeter}{perimeter} $2 \times (1 + 9) = 20$ m,
\omterm{def:g6:measure:area}{area} $9$ m$^2$. \ $2 \times 8$: \omterm{def:g6:measure:perimeter}{perimeter} $2 \times 10 = 20$ m,
\omterm{def:g6:measure:area}{area} $16$ m$^2$.

\textbf{2.} The \omterm{def:g6:measure:perimeter}{perimeter} is twice (length $+$ width)
(\cref{ex:g6:measure:perimeter}), so length $+$ width
$= 20 \div 2 = 10$ m. The table:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccccc}
pen & $1 \times 9$ & $2 \times 8$ & $3 \times 7$ & $4 \times 6$
& $5 \times 5$ \\
\hline
\omterm{def:g6:measure:area}{area} (m$^2$) & $9$ & $16$ & $21$ & $24$ & $25$ \\
\end{tabular}
\end{center}
The square $5 \times 5$ is best.

\textbf{3.} $4.5 \times 5.5 = 24.75$ and $4.9 \times 5.1 =
24.99$: closer and closer to $25$, but still below. Conjecture:
the square beats \emph{every} rectangle of \omterm{def:g6:measure:perimeter}{perimeter} $20$,
decimal sides included.

\textbf{4.} From the table: the \omterm{def:g2:mult:def}{product} of two numbers with \omterm{def:g1:addition:def}{sum}
$10$ is largest when the two numbers are \emph{equal} ($5$ and
$5$). General rule: for a fixed \omterm{def:g1:addition:def}{sum}, the \omterm{def:g2:mult:def}{product} is largest for
equal parts --- the more unequal the pair, the smaller the
\omterm{def:g2:mult:def}{product}.

\textbf{5.} From the $5 \times 5$ square, remove the top row
--- a strip of $5$ unit squares --- leaving a $5 \times 4$ pen;
then glue a column of $4$ unit squares onto one end, making the
pen $6 \times 4$. Five squares were taken away and only four
came back: the added strip lies along the $4$-side, which is
shorter than the $5$-side the removed strip covered. \omterm{def:g5:solids:net}{Net} loss:
one square meter ($25 \to 24$). The next step, from
$6 \times 4$ to $7 \times 3$, trades a row of $6$ for a column
of $3$ and loses three more ($24 \to 21$): the further the pen
is from the square, the longer the strip it gives up and the
shorter the one it gets back.

\textbf{6.} With $w + L + w = 20$, i.e.\ $L = 20 - 2w$:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccccccc}
$w$ & $1$ & $2$ & $3$ & $4$ & $5$ & $6$ & $7$ \\
\hline
$L$ & $18$ & $16$ & $14$ & $12$ & $10$ & $8$ & $6$ \\
\omterm{def:g6:measure:area}{area} & $18$ & $32$ & $42$ & $48$ & $50$ & $48$ & $42$ \\
\end{tabular}
\end{center}
The winner is $5 \times 10$, of \omterm{def:g6:measure:area}{area} $50$ m$^2$ --- twice as
long (along the wall) as it is wide: \emph{not} a square.

\textbf{7.} Reflect the pen across the wall: pen plus mirror
image form a doubled pen using the fence twice, $40$ m of
fence all around (the wall side is interior now). By Part I, the
best rectangle with \omterm{def:g6:measure:perimeter}{perimeter} $40$ is the square, $10 \times
10$, of \omterm{def:g6:measure:area}{area} $100$ m$^2$. The best real pen is \omterm{def:g3:division:half}{half} of that
best doubled pen: $10$ m along the wall, $5$ m deep, \omterm{def:g6:measure:area}{area}
$50$ m$^2$ --- exactly the table's winner, now explained.

\textbf{8.} Circumference $20 = 2 \times \pi \times r$, so
$r = 20 \div 6.28 \approx 3.18$ m. \omterm{def:g6:measure:area}{Area}:
$\pi r^2 \approx 3.14 \times 3.18 \times 3.18 \approx
31.8$ m$^2$ --- comfortably more than the square's $25$ m$^2$.
Dido knew what she was doing: for a given length of border, the
circle encloses the most.

\textbf{9.} \omterm{def:g6:measure:perimeter}{Perimeters}: $1 \times 36$: $74$ m; \ $2 \times 18$:
$40$ m; \ $3 \times 12$: $30$ m; \ $4 \times 9$: $26$ m; \
$6 \times 6$: $24$ m. The square again --- least fence for the
given \omterm{def:g6:measure:area}{area}. It is Part I backwards: fixed \omterm{def:g6:measure:perimeter}{perimeter}, biggest
\omterm{def:g6:measure:area}{area}, or fixed \omterm{def:g6:measure:area}{area}, smallest \omterm{def:g6:measure:perimeter}{perimeter}: both crown the square
(and, beyond all rectangles, the circle).

\textbf{10.} A \omterm{def:g4:numbers:round}{round} wall is the shortest border for the space
it encloses (question 8), so a \omterm{def:g4:numbers:round}{round} can, pipe or tank holds the
most content for the least metal --- material is money.

\textbf{11.} For instance $50$ m $\times$ $0.01$ m: \omterm{def:g6:measure:perimeter}{perimeter}
$2 \times (50 + 0.01) = 100.02$ m, \omterm{def:g6:measure:area}{area}
$50 \times 0.01 = 0.5$ m$^2$. Long and thin: kilometers of
border, hardly any land.

\textbf{12.} False. The pen of question 11 has \omterm{def:g6:measure:perimeter}{perimeter}
$100.02$ m and \omterm{def:g6:measure:area}{area} $0.5$ m$^2$, while the $5 \times 5$ pen has
\omterm{def:g6:measure:perimeter}{perimeter} $20$ m and \omterm{def:g6:measure:area}{area} $25$ m$^2$: larger \omterm{def:g6:measure:perimeter}{perimeter}, far
smaller \omterm{def:g6:measure:area}{area}.

\textbf{13.} \omterm{def:g6:measure:area}{Area}: one unit square is missing,
$24 - 1 = 23$ cm$^2$. \omterm{def:g6:measure:perimeter}{Perimeter}: walking around, the notch
replaces $1$ cm of straight wall by three sides of the little
square, $1 + 1 + 1 = 3$ cm: the \omterm{def:g6:measure:perimeter}{perimeter} grows from $20$ cm to
$20 - 1 + 3 = 22$ cm. Removing material \emph{lengthened} the
border.

\textbf{14.} Every zoom on a real coast reveals new bays, rocks
and creeks --- notches upon notches, and question 13 shows that
each notch adds border length while barely changing the \omterm{def:g6:measure:area}{area}.
So the measured length keeps growing with the level of detail
(``the coastline paradox''), while the measured \omterm{def:g6:measure:area}{area} settles
down: \omterm{def:g6:measure:perimeter}{perimeter} and \omterm{def:g6:measure:area}{area} are truly independent quantities.

\textbf{15.} Mirror argument: the doubled pen would use
$2 \times 36 = 72$ m of fence, and the best rectangle of
\omterm{def:g6:measure:perimeter}{perimeter} $72$ is the $18 \times 18$ square. So the best pen is
$18$ m along the wall and $9$ m deep: fence check
$9 + 18 + 9 = 36$ m, \omterm{def:g6:measure:area}{area} $18 \times 9 = 162$ m$^2$. In open
field, the best is the $9 \times 9$ square, \omterm{def:g6:measure:area}{area} $81$ m$^2$.
The barn wall exactly \emph{doubles} the farmer's land.
\end{solution}
