\section*{Chapter \ref{ch:g6:decimals} --- Decimal Numbers}

\begin{solution}{exo:g6:decimals:1}
$5.37$; \quad $0.9$; \quad $12.004$; \quad $8.05$.
\end{solution}

\begin{solution}{exo:g6:decimals:2}
Tenths digit: $3$; hundredths digit: $5$; the $8$ counts the tens.
\[
86.354 = 8 \times 10 + 6 + 3 \times \frac{1}{10} + 5 \times
\frac{1}{100} + 4 \times \frac{1}{1000}.
\]
\end{solution}

\begin{solution}{exo:g6:decimals:3}
$5.3 > 5.29$ (compare $5.30$ with $5.29$); \quad
$0.7 = 0.70$ (final zeros change nothing); \quad
$2.09 < 2.9$ (tenths: $0 < 9$); \quad
$14.5 > 14.49$.
\end{solution}

\begin{solution}{exo:g6:decimals:4}
Pad to two decimals: $4.20$; $4.05$; $4.51$; $4.15$; $4.50$. Order:
\[
4.05 < 4.15 < 4.2 < 4.5 < 4.51 .
\]
\end{solution}

\begin{solution}{exo:g6:decimals:5}
Each tick is one tenth. $A$: $6.3$; \quad $B$: $6.7$; \quad
$C$: $7.2$.
\end{solution}

\begin{solution}{exo:g6:decimals:6}
$45.80 + 7.65 = 53.45$.

$23.40 - 8.72 = 14.68$ (check: $14.68 + 8.72 = 23.4$).

$56 \times 24 = 1344$, and two decimal digits in the factors:
$5.6 \times 2.4 = 13.44$.
\end{solution}

\begin{solution}{exo:g6:decimals:7}
$6.42 \times 10 = 64.2$; \quad
$0.35 \times 1000 = 350$; \quad
$78.1 \div 100 = 0.781$; \quad
$5 \div 1000 = 0.005$.
\end{solution}

\begin{solution}{exo:g6:decimals:8}
$2.4$ m $= 240$ cm; \quad $370$ g $= 0.37$ kg (divide by $1000$);
\quad $0.85$ km $= 850$ m; \quad $56$ mm $= 0.056$ m.
\end{solution}

\begin{solution}{exo:g6:decimals:9}
To the unit: $24$ (next digit $8$: round up). To the tenth: $23.9$.
To the hundredth: $23.87$. And $9.97$ to the tenth: the next digit is
$7$, so round up the $9$ tenths --- $10.0$.
\end{solution}

\begin{solution}{exo:g6:decimals:10}
Price of the baguettes: $3 \times 1.15 = 3.45$. Change:
$5 - 3.45 = 1.55$.
\end{solution}

\begin{solution}{exo:g6:decimals:11}
Between $7.4$ and $7.5$: for instance $7.45$ (or $7.41$, $7.499$,
\dots). Between $3.99$ and $4$: for instance $3.995$. In both cases
there are \emph{infinitely many} such numbers: one can always add more
decimal places.
\end{solution}

\begin{solution}{exo:g6:decimals:12}
Largest: put the biggest digits first and the point as late as
possible: $85.2$. Smallest: smallest digits first and the point as
early as possible: $2.58$.
\end{solution}
