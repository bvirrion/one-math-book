\section*{अध्याय \ref{ch:g6:measure} --- परिधि, क्षेत्रफल, आयतन}

\begin{solution}{exo:g6:measure:1}
$3.5$ एम $= 350$ सेमी; \quad $420$ मिमी $= 42$ सेमी; \quad
$0.8$ किमी $= 800$ एम; \quad $25\,000$ एम $= 25$ किमी.
\end{solution}

\begin{solution}{exo:g6:measure:2}
आयत: $2 \times (8 + 3.5) = 2 \times 11.5 = 23$ सेमी।
वर्ग: $4 \times 6.2 = 24.8$ सेमी।
त्रिभुज: $5 + 7 + 9 = 21$ सेमी।
\end{solution}

\begin{solution}{exo:g6:measure:3}
एक गोद: $2\pi \times 50 = 100\pi \approx 314$ एम। के लिए $2$ किमी
$= 2000$ एम: $2000 \div 314 \approx 6.4$, इसलिए $7$ पूर्ण गोद की जरूरत है
($6$ गोद केवल बनाते हैं $1\,885$ एम)।
\end{solution}

\begin{solution}{exo:g6:measure:4}
आयत: $9 \times 4 = 36$ सेमी $^2$. वर्ग: $7^2 = 49$ एम $^2$.
सही \omterm{def:g3:shapes:shapes}{त्रिकोण}: $\frac{6 \times 10}{2} = 30$ सेमी $^2$.
डिस्क: $\pi \times 3^2 = 9\pi \approx 28$ सेमी $^2$.
\end{solution}

\begin{solution}{exo:g6:measure:5}
$3$ एम $^2 = 30\,000$ सेमी $^2$; \quad
$45\,000$ सेमी $^2 = 4.5$ एम $^2$; \quad
$2.5$ एल $= 2\,500$ सेमी $^3$; \quad
$4\,500$ एल $= 4.5$ एम $^3$.
\end{solution}

\begin{solution}{exo:g6:measure:6}
बाड़ (\omterm{def:g6:measure:perimeter}{परिधि}): $2 \times (120 + 85) = 2 \times 205 = 410$ एम।
\omterm{def:g6:measure:area}{क्षेत्र}: $120 \times 85 = 10\,200$ एम $^2$.
\end{solution}

\begin{solution}{exo:g6:measure:7}
\omterm{def:g5:solids:def}{डिब्बा}: $5 \times 4 \times 10 = 200$ सेमी $^3$. \omterm{def:g5:solids:def}{घनक्षेत्र}:
$3 \times 3 \times 3 = 27$ सेमी $^3$.
\end{solution}

\begin{solution}{exo:g6:measure:8}
\omterm{def:g6:measure:perimeter}{परिधि} $16$ सेमी का अर्थ है लंबाई $+$ चौड़ाई $= 8$ सेमी। उदाहरण के लिए
$6 \times 2$ (\omterm{def:g6:measure:area}{क्षेत्र} $12$ सेमी $^2$) और $5 \times 3$ (\omterm{def:g6:measure:area}{क्षेत्र} $15$ सेमी $^2$)
--- या वर्ग $4 \times 4$ (\omterm{def:g6:measure:area}{क्षेत्र} $16$ सेमी $^2$). ए के जितना करीब
वर्ग, उतना बड़ा \omterm{def:g6:measure:area}{क्षेत्र}.
\end{solution}

\begin{solution}{exo:g6:measure:9}
\omterm{def:g6:measure:area}{क्षेत्र}: $10 \times 2 + 2 \times 6 = 20 + 12 = 32$ सेमी $^2$.

\omterm{def:g6:measure:perimeter}{परिधि} (बार के नीचे केन्द्रित तना): चारों ओर घूमना,
\[
10 + 2 + 4 + 6 + 2 + 6 + 4 + 2 = 36 \text{ cm}
\]
(ऊपर; बार का दाहिना सिरा; निचला दाहिना भाग; तने का दाहिना भाग; निचला भाग
तना; तने का बायां भाग; नीचे बाईं ओर; बार का बायां सिरा)।
\end{solution}

\begin{solution}{exo:g6:measure:10}
\emph{1.} $ V = 25 \times 10 \times 2 = 500$ एम $^3$.

\emph{2.} $500$ एम $^3 = 500 \times 1000 = 500\,000$ एल

\emph{3.} $500\,000 \div 50\,000 = 10$ घंटे।
\end{solution}

\begin{solution}{exo:g6:measure:11}
बगीचा: $20^2 = 400$ एम $^2$. तालाब: $\pi \times 3^2 = 9\pi $ एम$^2$.
घास: $400 - 9\pi \approx 400 - 28.3 \approx 372$ एम $^2$.
\end{solution}

\begin{solution}{exo:g6:measure:12}
\emph{1.} छोटी बार: $15 \times 6 \times 1 = 90$ सेमी $^3$. बड़ी बार:
$30 \times 12 \times 2 = 720$ सेमी $^3$. \omterm{def:g6:measure:volume}{आयतन} से गुणा किया जाता है
$720 \div 90 = 8$ --- तीन आयामों में से प्रत्येक को दोगुना करने से गुणन होता है
द \omterm{def:g6:measure:volume}{आयतन} द्वारा $2 \times 2 \times 2 = 8$.

\emph{2.} चार गुना कीमत पर आठ गुना चॉकलेट: हाँ,
बिग बार प्रति ग्राम से दोगुना अच्छा सौदा है।
\end{solution}

\begin{solution}{pb:g6:measure:1}
\textbf{1.} $1 \times 9$: \omterm{def:g6:measure:perimeter}{परिधि} $2 \times (1 + 9) = 20$ एम,
\omterm{def:g6:measure:area}{क्षेत्र} $9$ एम $^2$. \ $2 \times 8$: \omterm{def:g6:measure:perimeter}{परिधि} $2 \times 10 = 20$ एम,
\omterm{def:g6:measure:area}{क्षेत्र} $16$ एम $^2$.

\textbf{2.}  \omterm{def:g6:measure:perimeter}{परिधि} दोगुनी है (लंबाई $+$ चौड़ाई)
(\cref{ex:g6:measure:perimeter}), इतनी लंबाई $+$ चौड़ाई
$= 20 \div 2 = 10$ एम। टेबल:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccccc}
pen & $1 \times 9$ & $2 \times 8$ & $3 \times 7$ & $4 \times 6$
& $5 \times 5$ \\
\hline
\omterm{def:g6:measure:area}{क्षेत्र} (m $^2$) & $9$ & $16$ & $21$ & $24$ & $25$ \\
\end{tabular}
\end{center}
चौराहा $5 \times 5$ सबसे अच्छा है।

\textbf{3.} $4.5 \times 5.5 = 24.75$ और $4.9 \times 5.1 =
24.99$: करीब और करीब $25$, लेकिन फिर भी नीचे। अनुमान:
वर्ग धड़कता है \emph{प्रत्येक} का आयत \omterm{def:g6:measure:perimeter}{परिधि} $20$,
दशमलव पक्ष शामिल हैं.

\textbf{4.} तालिका से: \omterm{def:g2:mult:def}{उत्पाद} दो संख्याओं के साथ \omterm{def:g1:addition:def}{जोड़}
$10$ दो संख्याएँ होने पर सबसे बड़ा होता है \emph{बराबर} ($5$ और
$5$). सामान्य नियम: एक निश्चित के लिए \omterm{def:g1:addition:def}{जोड़}, द \omterm{def:g2:mult:def}{उत्पाद} के लिए सबसे बड़ा है
बराबर भाग--जोड़ी जितनी अधिक असमान होगी, \omterm{def:g1:addition:def}{जोड़}ी उतनी ही छोटी होगी
\omterm{def:g2:mult:def}{उत्पाद}.

\textbf{5.} से $5 \times 5$ वर्गाकार, शीर्ष पंक्ति हटा दें
--- की एक पट्टी $5$ इकाई वर्ग---ए छोड़कर $5 \times 4$ कलम;
फिर एक कॉलम को गोंद दें $4$ इकाई को एक छोर पर वर्गाकार किया जाता है, जिससे बनता है
कलम $6 \times 4$. पाँच वर्ग छीन लिये गये और केवल चार
वापस आया: \omterm{def:g1:addition:def}{जोड़}ी गई पट्टी साथ में है $4$-पक्ष, जो है
से छोटा $5$-हटाई गई पट्टी को किनारे से ढक दें। \omterm{def:g5:solids:net}{जाल} हानि:
एक वर्ग \omterm{def:g4:fractions:def}{मीटर} ($25 \to 24$). अगला कदम, से
$6 \times 4$ को $7 \times 3$, की एक पंक्ति का व्यापार करता है $6$ एक कॉलम के लिए
का $3$ और तीन और हारे ($24 \to 21$): जितनी आगे कलम
वर्ग से है, जितनी लंबी पट्टी वह छोड़ता है और
जितना छोटा यह वापस मिलता है।

\textbf{6.} साथ $ w + L + w = 20$, यानी\ $ L = 20 - 2w $:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccccccc}
$ w $ & $1$ & $2$ & $3$ & $4$ & $5$ & $6$ & $7$ \\
\hline
$ L $ & $18$ & $16$ & $14$ & $12$ & $10$ & $8$ & $6$ \\
\omterm{def:g6:measure:area}{क्षेत्र} & $18$ & $32$ & $42$ & $48$ & $50$ & $48$ & $42$ \\
\end{tabular}
\end{center}
विजेता है $5 \times 10$, का \omterm{def:g6:measure:area}{क्षेत्र} $50$ एम $^2$ --- दो बार के रूप में
लम्बा (दीवार के साथ) जितना चौड़ा है: \emph{नहीं} एक वर्ग.

\textbf{7.} दीवार पर कलम को प्रतिबिंबित करें: कलम प्लस दर्पण
बाड़ का दो बार उपयोग करके दोहरी कलम से छवि बनाएं, $40$ का मी
चारों ओर बाड़ (दीवार का हिस्सा अब आंतरिक है)। भाग I द्वारा,
के साथ सबसे अच्छा आयत \omterm{def:g6:measure:perimeter}{परिधि} $40$ वर्ग है, $10 \times
10$, का \omterm{def:g6:measure:area}{क्षेत्र} $100$ एम $^2$. सबसे अच्छा असली पेन है \omterm{def:g3:division:half}{आधा} उसमें से
सर्वोत्तम दोहरी कलम: $10$ मैं दीवार के साथ, $5$ मी गहरा, \omterm{def:g6:measure:area}{क्षेत्र}
$50$ एम $^2$ --- बिल्कुल तालिका का विजेता, अब समझाया गया।

\textbf{8.} \omterm{def:g6:measure:perimeter}{परिधि} $20 = 2 \times \pi \times r $, इसलिए
$ r = 20 \div 6.28 \approx 3.18$ एम। \omterm{def:g6:measure:area}{क्षेत्र}:
$\pi r^2 \approx 3.14 \times 3.18 \times 3.18 \approx
31.8$ एम $^2$ --- आराम से वर्ग से भी ज्यादा $25$ एम $^2$.
डिडो को पता था कि वह क्या कर रही थी: सीमा की एक निश्चित लंबाई के लिए
वृत्त सबसे अधिक घेरता है।

\textbf{9.} \omterm{def:g6:measure:perimeter}{परिधि}: $1 \times 36$: $74$ एम; \ $2 \times 18$:
$40$ एम; \ $3 \times 12$: $30$ एम; \ $4 \times 9$: $26$ एम; \
$6 \times 6$: $24$ एम। वर्ग फिर से --- के लिए कम से कम बाड़
दिया गया \omterm{def:g6:measure:area}{क्षेत्र}. यह भाग I पीछे की ओर है: निश्चित \omterm{def:g6:measure:perimeter}{परिधि}, सबसे बड़ा
\omterm{def:g6:measure:area}{क्षेत्र}, या निश्चित \omterm{def:g6:measure:area}{क्षेत्र}, सबसे छोटा \omterm{def:g6:measure:perimeter}{परिधि}: दोनों वर्ग का ताज पहनते हैं
(और, सभी आयतों से परे, वृत्त)।

\textbf{10.} ए \omterm{def:g4:numbers:round}{गोल} दीवार \omterm{ex:g1:subtraction:difference}{अंतर}िक्ष की सबसे छोटी सीमा है
यह संलग्न है (प्रश्न 8), इसलिए a \omterm{def:g4:numbers:round}{गोल} कैन, पाइप या टैंक धारण करता है
कम से कम धातु के लिए सबसे अधिक सामग्री --- सामग्री पैसा है।

\textbf{11.} उदाहरण के लिए $50$ एम $\times $ $0.01$ एम: \omterm{def:g6:measure:perimeter}{परिधि}
$2 \times (50 + 0.01) = 100.02$ एम, \omterm{def:g6:measure:area}{क्षेत्र}
$50 \times 0.01 = 0.5$ एम $^2$. लंबा और पतला: किलो\omterm{def:g4:fractions:def}{मीटर}
सीमा, शायद ही कोई ज़मीन।

\textbf{12.} असत्य। प्रश्न 11 का पेन है \omterm{def:g6:measure:perimeter}{परिधि}
$100.02$ मी और \omterm{def:g6:measure:area}{क्षेत्र} $0.5$ एम $^2$, जब $5 \times 5$ कलम है
\omterm{def:g6:measure:perimeter}{परिधि} $20$ मी और \omterm{def:g6:measure:area}{क्षेत्र} $25$ एम $^2$: बड़ा \omterm{def:g6:measure:perimeter}{परिधि}, दूर
छोटा \omterm{def:g6:measure:area}{क्षेत्र}.

\textbf{13.} \omterm{def:g6:measure:area}{क्षेत्र}: एक इकाई वर्ग गायब है,
$24 - 1 = 23$ सेमी $^2$. \omterm{def:g6:measure:perimeter}{परिधि}: चारों ओर घूमना, पायदान
प्रतिस्थापित करता है $1$ छोटी सी तीन तरफ से सीधी दीवार का सेमी
वर्ग, $1 + 1 + 1 = 3$ सेमी: द \omterm{def:g6:measure:perimeter}{परिधि} से बढ़ता है $20$ सेमी से
$20 - 1 + 3 = 22$ सेमी। सामग्री हटाना \emph{लम्बे} द
सीमा.

\textbf{14.} वास्तविक तट पर प्रत्येक ज़ूम नई खाड़ियों, चट्टानों को प्रकट करता है
और खाड़ियाँ --- पायदान पर पायदान, और प्रश्न 13 यह दर्शाता है
प्रत्येक पायदान बमुश्किल बदलते हुए सीमा की लंबाई जोड़ता है \omterm{def:g6:measure:area}{क्षेत्र}.
इसलिए मापी गई लंबाई विस्तार के स्तर के साथ बढ़ती रहती है
(``तटरेखा विरोधाभास''), जबकि मापा गया \omterm{def:g6:measure:area}{क्षेत्र} बसाता है
नीचे: \omterm{def:g6:measure:perimeter}{परिधि} और \omterm{def:g6:measure:area}{क्षेत्र} वास्तव में स्वतंत्र मात्राएँ हैं।

\textbf{15.} दर्पण तर्क: दोगुनी कलम का प्रयोग होगा
$2 \times 36 = 72$ बाड़ का मी, और सबसे अच्छा आयत
\omterm{def:g6:measure:perimeter}{परिधि} $72$ है $18 \times 18$ वर्ग। तो सबसे अच्छा पेन है
$18$ दीवार के साथ मी और $9$ मी गहराई: बाड़ की जाँच
$9 + 18 + 9 = 36$ एम, \omterm{def:g6:measure:area}{क्षेत्र} $18 \times 9 = 162$ एम $^2$. खुले में
\omterm{def:g6:measure:area}{क्षेत्र}, सबसे अच्छा है $9 \times 9$ वर्ग, \omterm{def:g6:measure:area}{क्षेत्र} $81$ एम $^2$.
बिल्कुल खलिहान की दीवार \emph{दोगुना हो जाता है} किसान की जमीन.
\end{solution}
