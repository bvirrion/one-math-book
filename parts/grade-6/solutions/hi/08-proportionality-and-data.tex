\section*{अध्याय \ref{ch:g6:propdata} --- आनुपातिकता और डेटा}

\begin{solution}{exo:g6:propdata:1}
सेब: \omterm{def:g3:division:remainder}{भागफल} $\frac52 = 2.5$, $\frac{7.5}{3} = 2.5$,
$\frac{12.5}{5} = 2.5$ --- सभी समान: \omterm{def:g6:propdata:def}{आनुपातिक} (कीमत
$2.50$ प्रति किलो)।

आयु/ऊंचाई: $\frac{85}{2} = 42.5$ लेकिन $\frac{103}{4} = 25.75$ --- नहीं
बराबर: नहीं \omterm{def:g6:propdata:def}{आनुपातिक}.
\end{solution}

\begin{solution}{exo:g6:propdata:2}
एक क्रोइसैन: $4.80 \div 4 = 1.20$ यूरो. सात:
$7 \times 1.20 = 8.40$ यूरो.
\end{solution}

\begin{solution}{exo:g6:propdata:3}
गुणांक: $18 \div 10 = 1.8$ यूरो प्रति लीटर. तब
$25$ एल लागत $25 \times 1.8 = 45$ यूरो; $72$ यूरो खरीदें
$72 \div 1.8 = 40$ एल; $60$ एल लागत $60 \times 1.8 = 108$ यूरो.
\end{solution}

\begin{solution}{exo:g6:propdata:4}
के लिए $250$ किमी: $2.5$ के ईंधन का समय $100$ किमी, तो
$2.5 \times 6 = 15$ एल. के लिए $350$ किमी: $3.5 \times 6 = 21$ एल. के साथ
$27$ एल: $27 \div 6 = 4.5$ सैकड़ों किमी, यानी\ $450$ किमी.
\end{solution}

\begin{solution}{exo:g6:propdata:5}
$50\,\%$ का $84$: \omterm{def:g3:division:half}{आधा}, $42$. \quad
$25\,\%$ का $200$: ए \omterm{def:g3:division:half}{तिमाही}, $50$. \quad
$10\,\%$ का $63$: दसवां हिस्सा, $6.3$. \quad
$30\,\%$ का $70$: $3 \times 7 = 21$.
\end{solution}

\begin{solution}{exo:g6:propdata:6}
$60\,\%$ का $450$: $\frac{60}{100} \times 450 = 270$ छात्र यहाँ खाते हैं
कैफेटेरिया; $450 - 270 = 180$ ऐसा न करें।
\end{solution}

\begin{solution}{exo:g6:propdata:7}
से भी कम $10$ पुस्तकें: मंगलवार ($8$) और गुरुवार ($6$). बुधवार
\omterm{def:g1:counting:zero}{शून्य} गुरुवार: $17 - 6 = 11$ अधिक पुस्तकें.
\end{solution}

\begin{solution}{exo:g6:propdata:8}
ऊँचाई की पाँच पट्टियों वाला बार चार्ट $14$, $16$, $13$, $17$, $20$
(स्नातक प्रत्येक $2$ या $5$ डिग्री अच्छी तरह से काम करती है)। सबसे गर्म दिन: शुक्रवार
($20$ डिग्री)।
\end{solution}

\begin{solution}{exo:g6:propdata:9}
के लिए $10$ लोग, गुणा करें $\frac{10}{6} = \frac53$: आटा
$450 \times \frac{10}{6} = 750$ जी। अंडे:
$3 \times \frac{10}{6} = 5$ --- सौभाग्य से एक पूर्ण संख्या. (अगर ऐसा होता
नहीं, कोई होगा \omterm{def:g4:numbers:round}{गोल} \emph{ऊपर} पर्याप्त होना।)
\end{solution}

\begin{solution}{exo:g6:propdata:10}
\emph{1.} में $2$ एच: $48$ किमी. में \omterm{def:g3:division:half}{आधा} एक घंटा: $12$ किमी. में
$1$ एच $30$: $24 + 12 = 36$ किमी.

\emph{2.} $60 \div 24 = 2.5$ घंटे, यानी\ $2$ एच $30$ मि.
\end{solution}

\begin{solution}{exo:g6:propdata:11}
गेंद: छूट $20\,\%$ का $15 = 3$ यूरो, नई कीमत $12$ यूरो.
रैकेट: छूट $8$ यूरो, नई कीमत $32$ यूरो. नई कीमत है
$80\,\%$ प्रत्येक मामले में पुराने का: \omterm{def:g6:propdata:def}{आनुपातिक}, गुणांक
$0.8$.
\end{solution}

\begin{solution}{exo:g6:propdata:12}
बाद $ t $ घंटे, मोटी मोमबत्ती जल गई है $\frac t6$ इसकी ऊंचाई का:
यह पर खड़ा है $1 - \frac t6$ प्रारंभिक ऊँचाई का; पतला वाला
$1 - \frac t4$. हम चाहते हैं
\[
1 - \frac t4 = \frac12 \left(1 - \frac t6\right)
\quad\text{i.e.}\quad
1 - \frac t4 = \frac12 - \frac t{12}.
\]
तब $\frac12 = \frac t4 - \frac t{12} = \frac{3t - t}{12} =
\frac{t}{6}$, इसलिए $ t = 3$ घंटे। जांचें: बाद में $3$ h मोटी मोमबत्ती
पर खड़ा है $1 - \frac36 = \frac12$ और पतला वाला
$1 - \frac34 = \frac14$ --- वास्तव में \omterm{def:g3:division:half}{आधा} इसका.
\end{solution}

\begin{solution}{pb:g6:propdata:1}
\textbf{1.} $100\,000$ सेमी $= 1\,000$ एम $= 1$ किमी: एक सेंटी\omterm{def:g4:fractions:def}{मीटर}
मानचित्र पर जमीन पर एक किलो\omterm{def:g4:fractions:def}{मीटर} है।

\textbf{2.} $7.5$ सेमी $\rightarrow 7.5$ गांवों के बीच किमी;
$12$ सेमी $\rightarrow 12$ झील किनारे का किमी.

\textbf{3.} $20$ किमी $\rightarrow 20$ मानचित्र पर सेमी.

\textbf{4.} $5$ किमी $= 500\,000$ सेमी, तो पैमाना है
$1:500\,000$. लंबी पैदल यात्रा मानचित्र ($1:100\,000$) अधिक है
विस्तृत: यह उपयोग करता है $5$ कागज का सेमी जहाँ दूसरा ही खर्च करता है
$1$ सेमी।

\textbf{5.}
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|ccc}
map (cm) & $1$ & $2$ & $7.5$ \\
\hline
real (km) & $1$ & $2$ & $7.5$ \\
\end{tabular}
\end{center}
वास्तविक दूरी $=$ मानचित्र दूरी $\times 1$ (इन इकाइयों में):
हमेशा \emph{वही} गुणक, जो वास्तव में है
की परिभाषा समानता
(\cref{def:g6:propdata:def}). यहाँ गुणांक है $1$
किलो\omterm{def:g4:fractions:def}{मीटर} प्रति सेंटी\omterm{def:g4:fractions:def}{मीटर}.

\textbf{6.} $4$ एम $= 400$ सेमी और $400 \div 50 = 8$;
$3$ एम $= 300$ सेमी और $300 \div 50 = 6$: योजना एक दिखाती है
$8$ सेमी $\times $ $6$ सेमी आयत.

\textbf{7.} बिस्तर: $200 \div 50 = 4$ सेमी द्वारा $90 \div 50 = 1.8$ सेमी।
द्वार: $80 \div 50 = 1.6$ सेमी।

\textbf{8.} असली \omterm{def:g6:measure:area}{क्षेत्र}: $4 \times 3 = 12$ एम $^2$. योजना \omterm{def:g6:measure:area}{क्षेत्र}:
$8 \times 6 = 48$ सेमी $^2$. परिवर्तित करना:
$12$ एम $^2 = 12 \times 10\,000 = 120\,000$ सेमी $^2$, और
\[
120\,000 \div 48 = 2\,500 .
\]
 \omterm{def:g3:division:remainder}{भागफल} क्या नहीं है $50$ लेकिन $2\,500 = 50 \times 50$.

\textbf{9.} एक सेमी $^2$ कागज का मतलब वास्तविक वर्ग है
$50$ सेमी द्वारा $50$ सेमी, जिसमें शामिल है
$50 \times 50 = 2\,500$ सेमी $^2$. तो जब लंबाई को विभाजित किया जाता है
$50$, \omterm{def:g6:measure:area}{क्षेत्र}ों द्वारा विभाजित हैं $50 \times 50 = 2\,500$: \omterm{def:g6:measure:area}{क्षेत्र}ों का पीछा करो \emph{वर्ग} पैमाने का.

\textbf{10.} $6 \times 2\,500 = 15\,000$ सेमी $^2$, और
$15\,000 \div 10\,000 = 1.5$ एम $^2$.

\textbf{11.} चित्र से पता चलता है कि रविवार को बेचा गया \emph{दो बार} जैसे
बहुत अधिक (एक बार दोगुना लंबा)। सत्य: वृद्धि है $5$ यूरो बाहर
का $105$, और $5 \div 105 \approx 0.048$: के बारे में $5\,\%$ अधिक,
नहीं $100\,\%$ अधिक। मैनेजर ने जांच करने की चेतावनी को नजरअंदाज कर दिया
वह अक्ष जहां से प्रारंभ होता है $0$
(\cref{met:g6:propdata:read}).

\textbf{12.} से अक्ष के साथ $0$, सलाखें ऊपर उठती हैं $105$ और
$110$ छोटी इकाइयाँ: लगभग समान ऊँचाई की दो पट्टियाँ
एक की ईमानदार तस्वीर $5\,\%$ \omterm{ex:g1:subtraction:difference}{अंतर}.

\textbf{13.} ऊपर: वृद्धि है $20$ की शुरुआत से टिकट
$40$: $\frac{20}{40} = 50\,\%$. नीचे: कमी है $20$
की शुरुआत से टिकट $60$: $\frac{20}{60} = \frac13 \approx
33\,\%$. वही $20$ टिकटें, विभिन्न आरंभिक मान --- a
प्रतिशत हमेशा एक होता है \omterm{def:g6:fractions:def}{अंश} \emph{किसी चीज़ का}, और
``कुछ'' बदल गया.

\textbf{14.} बाद $20\,\%$ बंद: $100 - 20 = 80$ यूरो. अतिरिक्त $10\,\%$ पर लागू होता है $80$: छूट $8$ यूरो, अंतिम कीमत
$72$ यूरो. कुल छूट: $28$ यूरो बाहर $100$, इसलिए
$28\,\%$ --- नहीं $30\,\%$. दूसरी छूट पर कार्रवाई की
पहले से ही कम कीमत, इसलिए इसके यूरो छोटे हैं: प्रतिशत
अलग-अलग मात्राएं न \omterm{def:g1:addition:def}{जोड़}ें.

\textbf{15.} उस मानचित्र पर, $1$ सेमी का मतलब है $5$ किमी, तो $1$
सेमी $^2$ के लिए खड़ा है $5 \times 5 = 25$ किमी $^2$ (प्रश्न 9 का नियम).
वन: $3 \times 25 = 75$ किमी $^2$. लम्बाई अनुसरण करती है
पैमाना; \omterm{def:g6:measure:area}{क्षेत्र}ों इसका पालन करें \emph{वर्ग}.
\end{solution}
