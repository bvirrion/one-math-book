\section*{Chapter \ref{ch:g6:symmetry} --- Axial Symmetry}

\begin{solution}{exo:g6:symmetry:1}
$A'$ is two squares to the \emph{right} of $d$, on the same row as
$A$; $B'$ is five squares to the right, one square higher --- mirror
positions.
\end{solution}

\begin{solution}{exo:g6:symmetry:2}
\omterm{def:g6:lines:perp}{Perpendicular} to $d$ through $M$, foot $H$; then $M'$ on the other
side with $HM' = MH = 3$ cm. The distance $MM'$ is
$3 + 3 = 6$ cm.
\end{solution}

\begin{solution}{exo:g6:symmetry:3}
Vertical axis (like A): A, H, I, M, O, T, U, V, W, X, Y.
Horizontal axis (like B): B, C, D, E, H, I, K, O, X.
Both: H, I, O, X.
(Exact answers depend slightly on how the letters are drawn.)
\end{solution}

\begin{solution}{exo:g6:symmetry:4}
\omterm{def:g6:shapes:triangles}{Equilateral triangle}: $3$ axes. Rectangle: $2$ (the \omterm{def:g6:symmetry:bisector}{perpendicular
bisectors} of its sides). \omterm{def:g6:shapes:quadrilaterals}{Rhombus}: $2$ (its diagonals). Square: $4$.
Circle: infinitely many (every line through the center).
\end{solution}

\begin{solution}{exo:g6:symmetry:5}
\emph{False.} Folding a (non-square) rectangle along a diagonal does
not make the two halves coincide: the two triangles have the same
shape but sit differently --- try it with a sheet of paper. Only for a
square do the diagonals work as axes.
\end{solution}

\begin{solution}{exo:g6:symmetry:6}
Reflections preserve lengths, angles and \omterm{def:g4:measure:perimeter}{perimeters}
(\cref{prop:g6:symmetry:preserved}): $A'B' = 4$ cm,
$\widehat{B'A'C'} = 70^\circ$, \omterm{def:g4:measure:perimeter}{perimeter} of $A'B'C' = 12$ cm.
\end{solution}

\begin{solution}{exo:g6:symmetry:7}
Compass construction as in \cref{met:g6:symmetry:bisector}. For any
$M$ on the line, the ruler confirms $MA = MB$
(\cref{prop:g6:symmetry:bisector}).
\end{solution}

\begin{solution}{exo:g6:symmetry:8}
All the points at equal distance from $A$ and $B$ form the
\omterm{def:g6:symmetry:bisector}{perpendicular bisector} of $[AB]$ (\cref{prop:g6:symmetry:bisector}).
The fountain can go anywhere on that line (within the garden).
\end{solution}

\begin{solution}{exo:g6:symmetry:9}
Each \omterm{def:g5:solids:def}{vertex} is reflected \omterm{def:g6:lines:perp}{perpendicular} to the axis: for the
$45^\circ$ axis, a point $k$ squares to the right of the axis lands
$k$ squares above it (and vice versa) --- horizontal and vertical
displacements are exchanged. Reflect the \omterm{def:g5:solids:def}{vertices} one by one, then
join them in order.
\end{solution}

\begin{solution}{exo:g6:symmetry:10}
The \omterm{def:g6:symmetry:def}{symmetric} of $M$ is the second intersection of the circle with the
\omterm{def:g6:lines:perp}{perpendicular} to $d$ through $M$ --- it lies on the circle. Reason:
the reflection preserves distances and fixes $O$ (which is on $d$), so
the image of any point at distance $r$ from $O$ is again at distance
$r$ from $O$: the circle maps onto itself.
\end{solution}

\begin{solution}{exo:g6:symmetry:11}
Each crossing point $P$ was drawn with the same compass opening $\rho$
from $A$ and from $B$: so $PA = \rho = PB$. By
\cref{prop:g6:symmetry:bisector}, a point equidistant from $A$ and $B$
lies on the \omterm{def:g6:symmetry:bisector}{perpendicular bisector} of $[AB]$; two such points determine
the line.
\end{solution}

\begin{solution}{exo:g6:symmetry:12}
For any bounce point $P$ on the wall, $PB = PB'$ (reflection across
$d$ preserves distances), so the path length
$AP + PB = AP + PB'$. The broken line $A \to P \to B'$ is shortest
when it is straight, i.e.\ when $P$ is on the \omterm{def:g6:lines:objects}{segment} $[AB']$. So the
best bounce point is the intersection of $[AB']$ with the wall.
\end{solution}
