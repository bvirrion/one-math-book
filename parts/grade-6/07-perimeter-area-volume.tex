\chapter{Perimeter, Area, Volume}\label{ch:g6:measure}

How long is the fence, how big is the field, how much water fits in the
tank? Three different questions, three different quantities ---
\omterm{def:g6:measure:perimeter}{perimeter}, \omterm{def:g6:measure:area}{area}, \omterm{def:g6:measure:volume}{volume} --- each with its own units. Confusing them is
the most common mistake in geometry; this chapter sorts them out for
good.

\section{Lengths and perimeter}

\begin{definition}[Perimeter]\label{def:g6:measure:perimeter}
The \emph{perimeter}\index{perimeter} of a figure is the total length
of its border. It is measured in units of length: millimeters (mm),
centimeters (cm), meters (m), kilometers (km), with
\[
1 \text{ km} = 1000 \text{ m}, \qquad
1 \text{ m} = 100 \text{ cm}, \qquad
1 \text{ cm} = 10 \text{ mm}.
\]
\end{definition}

\begin{example}\label{ex:g6:measure:perimeter}
A rectangle of length $L$ and width $w$ has \omterm{def:g6:measure:perimeter}{perimeter}
\[
P = L + w + L + w = 2 \times (L + w).
\]
For a $7$ cm by $4$ cm rectangle: $P = 2 \times (7 + 4) = 2 \times 11 =
22$ cm. A square of side $c$ has \omterm{def:g6:measure:perimeter}{perimeter} $4c$.
\end{example}

\begin{proposition}[Circumference of a circle]\label{prop:g6:measure:circle}
The \omterm{def:g6:measure:perimeter}{perimeter} (or \emph{circumference}) of a circle of \omterm{def:g3:shapes:shapes}{radius} $r$ is
\[
P = 2 \pi r ,
\]
where $\pi \approx 3.14$ is the same number for every circle.
\end{proposition}
\admitted

\begin{example}
A circular pond has \omterm{def:g3:shapes:shapes}{radius} $5$ m. Its border measures
$2 \times \pi \times 5 = 10\pi \approx 31.4$ m. Keep the exact value
$10\pi$ as long as possible; \omterm{def:g4:numbers:round}{round} only at the end.
\end{example}

\section{Areas}

\begin{definition}[Area]\label{def:g6:measure:area}
The \emph{area}\index{area} of a figure measures the surface it covers:
how many unit squares fit inside. Units: cm$^2$ (a square of side
$1$ cm), m$^2$, km$^2$, \dots\ Careful:
\[
1 \text{ m}^2 = 100 \times 100 \text{ cm}^2 = 10\,000 \text{ cm}^2
\]
--- one square meter is a $100$ cm by $100$ cm square, so each step of
the units ladder is worth $100$, not $10$.
\end{definition}

{\small The \omterm{def:g6:measure:area}{area} of a rectangle: $7$ columns of $4$ unit squares each,
so $7 \times 4 = 28$ squares in total. This is why \omterm{def:g6:measure:area}{area} $=$ length
$\times$ width.}
\end{omfigure}

\begin{proposition}[Basic area formulas]\label{prop:g6:measure:formulas}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{l|l}
figure & \omterm{def:g6:measure:area}{area} \\
\hline
rectangle ($L \times w$) & $A = L \times w$ \\
square (side $c$) & $A = c^2$ \\[4pt]
\omterm{def:g6:shapes:triangles}{right triangle} (legs $a$, $b$) & $A = \dfrac{a \times b}{2}$ \\[4pt]
disk (\omterm{def:g3:shapes:shapes}{radius} $r$) & $A = \pi r^2$ \\
\end{tabular}
\end{center}
\end{proposition}

\begin{proof}[Proof for the right triangle]
Two copies of a \omterm{def:g6:shapes:triangles}{right triangle} with legs $a$ and $b$, glued along the
hypotenuse, form an $a \times b$ rectangle. So the triangle's \omterm{def:g6:measure:area}{area} is
\omterm{def:g3:division:half}{half} the rectangle's: $\frac{ab}{2}$. (The rectangle formula is the
unit-square counting above; the disk formula is admitted, see
\cref{ch:g7:areas}.)
\end{proof}

\begin{tikzpicture}[scale=0.75]
  \draw[thick, fill=omDef!15] (0,0) -- (4,0) -- (0,2.5) -- cycle;
  \draw[thick, omExo, dashed] (4,0) -- (4,2.5) -- (0,2.5);
  \draw[thin] (0.3,0) -- (0.3,0.3) -- (0,0.3);
  \node[below, font=\small] at (2,-0.1) {$a$};
  \node[left, font=\small] at (0,1.25) {$b$};
  \node[font=\small, omExo] at (2.8,1.7) {second copy};
\end{tikzpicture}

{\small A \omterm{def:g6:shapes:triangles}{right triangle} is \omterm{def:g3:division:half}{half} a rectangle: hence the formula
$\frac{a\times b}{2}$.}
\end{omfigure}

\begin{example}[Composite figures]\label{ex:g6:measure:composite}
An L-shaped room is a $6$ m $\times$ $4$ m rectangle with a $2$ m
$\times$ $2$ m square corner removed. Its \omterm{def:g6:measure:area}{area}, step by step:
\begin{enumerate}
  \item full rectangle: $6 \times 4 = 24$ m$^2$;
  \item removed square: $2 \times 2 = 4$ m$^2$;
  \item remaining \omterm{def:g6:measure:area}{area}: $24 - 4 = 20$ m$^2$.
\end{enumerate}
Its \emph{\omterm{def:g6:measure:perimeter}{perimeter}} is not $20$ anything: walking around the L, the
border still measures $6 + 4 + 6 + 4 = 20$ m --- the two cuts of the
corner replace two equal pieces of wall. Same number by coincidence,
but square meters for one, meters for the other!
\end{example}

\begin{remark}[Same perimeter, different areas]
Two figures can have the same \omterm{def:g6:measure:perimeter}{perimeter} and very different \omterm{def:g6:measure:area}{areas}: a
$5 \times 5$ square and a $9 \times 1$ rectangle both have \omterm{def:g6:measure:perimeter}{perimeter}
$20$, but \omterm{def:g6:measure:area}{areas} $25$ and $9$. \omterm{def:g6:measure:perimeter}{Perimeter} and \omterm{def:g6:measure:area}{area} are truly independent
quantities.
\end{remark}

\section{Volumes}

\begin{definition}[Volume]\label{def:g6:measure:volume}
The \emph{volume}\index{volume} of a \omterm{def:g5:solids:def}{solid} measures the space it fills:
how many unit \omterm{def:g5:solids:def}{cubes} fit inside. Units: cm$^3$, m$^3$, \dots\ with
$1$ m$^3 = 1\,000\,000$ cm$^3$ (each step of the ladder is worth
$1000$). For liquids one also uses the \emph{liter}:
\[
1 \text{ L} = 1 \text{ dm}^3 = 1000 \text{ cm}^3,
\qquad
1 \text{ m}^3 = 1000 \text{ L}.
\]
\end{definition}

\begin{proposition}[Volume of a box]\label{prop:g6:measure:box}
A rectangular \omterm{def:g5:solids:def}{box} (a \emph{rectangular prism}) of length $L$, width $w$
and height $h$ has \omterm{def:g6:measure:volume}{volume}
\[
V = L \times w \times h .
\]
\end{proposition}

\begin{proof}
The bottom layer contains $L \times w$ unit \omterm{def:g5:solids:def}{cubes}
(\cref{def:g6:measure:area} picture, with \omterm{def:g5:solids:def}{cubes}), and there are $h$
such layers.
\end{proof}

            -- cycle;
          \draw[fill=omDef!28] (1,0) -- (1.45,0.3) -- (1.45,1.3) -- (1,1)
            -- cycle;
        \end{scope}
      }
    }

  \node[below, font=\small] at (2.5,-0.3)

\end{tikzpicture}

{\small A \omterm{def:g5:solids:def}{box} filled with unit \omterm{def:g5:solids:def}{cubes}: $4 \times 2$ \omterm{def:g5:solids:def}{cubes} per layer, two
layers.}
\end{omfigure}

\begin{example}\label{ex:g6:measure:aquarium}
An aquarium measures $60$ cm by $30$ cm by $40$ cm (height). \omterm{def:g6:measure:volume}{Volume}:
\[
V = 60 \times 30 \times 40 = 72\,000 \text{ cm}^3 = 72 \text{ dm}^3
= 72 \text{ L}.
\]
Step by step for the conversion: $1000$ cm$^3$ make one liter, and
$72\,000 \div 1000 = 72$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g6:measure:1}
Convert: $3.5$ m into cm; $420$ mm into cm; $0.8$ km into m;
$25\,000$ m into km.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:measure:2}
Compute the \omterm{def:g6:measure:perimeter}{perimeter} of: a rectangle $8$ cm $\times$ $3.5$ cm; a
square of side $6.2$ cm; a triangle with sides $5$ cm, $7$ cm and
$9$ cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:measure:3}
A circular running track has \omterm{def:g3:shapes:shapes}{radius} $50$ m. How long is one lap
(exact value with $\pi$, then rounded to the meter)? How many laps make
at least $2$ km?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:measure:4}
Compute the \omterm{def:g6:measure:area}{area} of: a rectangle $9$ cm $\times$ $4$ cm; a square of
side $7$ m; a \omterm{def:g6:shapes:triangles}{right triangle} with legs $6$ cm and $10$ cm; a disk of
\omterm{def:g3:shapes:shapes}{radius} $3$ cm (exact value, then rounded to the cm$^2$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:measure:5}
Convert: $3$ m$^2$ into cm$^2$; $45\,000$ cm$^2$ into m$^2$; $2.5$ L
into cm$^3$; $4\,500$ L into m$^3$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:measure:6}
A rectangular field is $120$ m long and $85$ m wide. How many meters of
fence are needed to enclose it? What is its \omterm{def:g6:measure:area}{area}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:measure:7}
Compute the \omterm{def:g6:measure:volume}{volume} of a \omterm{def:g5:solids:def}{box} $5$ cm $\times$ $4$ cm $\times$ $10$ cm,
and of a \omterm{def:g5:solids:def}{cube} of edge $3$ cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:measure:8}
Draw two different rectangles with \omterm{def:g6:measure:perimeter}{perimeter} $16$ cm, and compute their
\omterm{def:g6:measure:area}{areas}. Which of your rectangles has the larger \omterm{def:g6:measure:area}{area}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:measure:9}
A T-shaped figure is made of a $10 \times 2$ horizontal rectangle on
top of a $2 \times 6$ vertical one (measurements in cm). Compute its
\omterm{def:g6:measure:area}{area}, then its \omterm{def:g6:measure:perimeter}{perimeter} (walk around the border carefully, adding
every edge).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:measure:10}
A swimming pool is a rectangular \omterm{def:g5:solids:def}{box} $25$ m long, $10$ m wide and
$2$ m deep.
\begin{enumerate}
  \item How many cubic meters of water does it hold when full?
  \item How many liters is that?
  \item The pool is filled at $50\,000$ L per hour. How long does the
        filling take?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:measure:11}
A garden is a square of side $20$ m containing a circular pond of
\omterm{def:g3:shapes:shapes}{radius} $3$ m. What \omterm{def:g6:measure:area}{area} of grass is there (exact value, then rounded
to the m$^2$)?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:measure:12}
A chocolate bar measures $15$ cm $\times$ $6$ cm $\times$ $1$ cm.
The maker doubles \emph{all three} dimensions.
\begin{enumerate}
  \item By how much is the \omterm{def:g6:measure:volume}{volume} multiplied? Verify by computing both
        \omterm{def:g6:measure:volume}{volumes}.
  \item The price is multiplied by $4$. Is the big bar a better deal
        than the small one?
\end{enumerate}
\end{exercise}

\section{Problem: Queen Dido's fence}

\begin{problem}[{Weekend problem --- with a fixed length of fence,
which shape encloses the most land? The square beats every
rectangle, the circle beats the square, and a barn wall changes
everything}]
\label{pb:g6:measure:1}
Legend says that queen Dido, landing on the coast of Africa, was
granted ``as much land as an ox hide can enclose'' --- so she cut
the hide into one immensely long thin strip and enclosed enough
ground to found the city of Carthage. Her problem is now yours: a
farmer owns exactly $20$ m of fence. \Cref{exo:g6:measure:8}
showed that two pens with the same \omterm{def:g6:measure:perimeter}{perimeter} can have different
\omterm{def:g6:measure:area}{areas}; this problem finds the \emph{best} pen --- and discovers
that the answer changes completely when a barn wall lends a free
side.

\medskip
\noindent\textbf{Part I --- Twenty meters of fence.}
\begin{enumerate}
  \item The pen must be a rectangle using all $20$ m of fence.
        Check that a $1 \times 9$ pen and a $2 \times 8$ pen both
        qualify, and compute their \omterm{def:g6:measure:area}{areas}.
  \item Explain why the length and width of every qualifying pen
        add up to $10$ m. Then make the complete table of the
        whole-number pens ($1 \times 9$ up to $5 \times 5$) with
        their \omterm{def:g6:measure:area}{areas}. Which is best?
  \item Are decimal sides worth trying? Compute the \omterm{def:g6:measure:area}{areas} of the
        $4.5 \times 5.5$ pen and of the $4.9 \times 5.1$ pen, and
        compare with the $5 \times 5$ square. What do you
        conjecture?
  \item Question 2 turned the fence problem into a pure number
        question: \emph{among all pairs of numbers adding up to
        $10$, which pair has the largest \omterm{def:g2:mult:def}{product}?} Answer it from
        your table, and state the general rule you observe.
  \item Here is why moving away from the square always loses,
        with scissors instead of algebra: start from the
        $5 \times 5$ pen and change it into the $6 \times 4$ pen
        by removing a strip and gluing another one back. Which
        strip is removed, which is added, and why does the
        exchange lose exactly one square meter? Explain why a
        further step (to $7 \times 3$) loses \omterm{def:g3:numbers:evenodd}{even} more.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The barn, and the circle.}
\begin{enumerate}[resume]
  \item The pen is now built against a long barn wall: the wall
        replaces one length of the pen, and the $20$ m of fence
        cover only the other three sides. Make a table of the
        whole-number pens (width $w$, length $L$ along the wall,
        $w + L + w = 20$) and their \omterm{def:g6:measure:area}{areas}. Which pen wins now ---
        and is it a square?
  \item Explain the winner with a mirror
        (\cref{ch:g6:symmetry}): reflect the pen across the barn
        wall and consider the doubled pen. How much fence does
        the doubled pen use, which doubled pen is best by Part I,
        and what does that make the original pen?
  \item Dido did not build a rectangle. Bend the $20$ m of fence
        into a circle: using $\pi \approx 3.14$, compute its
        \omterm{def:g3:shapes:shapes}{radius} (to the cm), then its \omterm{def:g6:measure:area}{area} (to the m$^2$)
        (\cref{prop:g6:measure:circle},
        \cref{prop:g6:measure:formulas}). Compare with the
        $5 \times 5$ square.
  \item The reverse problem: a pen of \omterm{def:g6:measure:area}{area} exactly $36$ m$^2$ is
        wanted, with as little fence as possible. Compare the
        rectangles $1 \times 36$, $2 \times 18$, $3 \times 12$,
        $4 \times 9$ and $6 \times 6$: \omterm{def:g6:measure:perimeter}{perimeters}? Which shape is
        cheapest, and how is this question the mirror image of
        Part I?
  \item In one sentence: why are cans, pipes and water tanks so
        often \emph{\omterm{def:g4:numbers:round}{round}}?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- \omterm{def:g6:measure:perimeter}{Perimeter} and \omterm{def:g6:measure:area}{area} are strangers.}
\begin{enumerate}[resume]
  \item Find a rectangle whose \omterm{def:g6:measure:perimeter}{perimeter} exceeds $100$ m but
        whose \omterm{def:g6:measure:area}{area} is less than $1$ m$^2$. (Very long and very
        thin. Decimals allowed.)
  \item True or false: ``of two figures, the one with the larger
        \omterm{def:g6:measure:perimeter}{perimeter} has the larger \omterm{def:g6:measure:area}{area}.'' Give a counterexample
        from this problem.
  \item Take the $6 \times 4$ rectangle and cut a $1 \times 1$
        square notch into the middle of one long side (the notch
        opens outwards, like a missing tooth). Compute the \omterm{def:g6:measure:area}{area}
        and the \omterm{def:g6:measure:perimeter}{perimeter} of the notched figure, and compare both
        with the original. What happened to each?
  \item Mapmakers know a strange fact: the more detailed the map,
        the \emph{longer} a coastline measures --- every zoom
        reveals new little notches, and question 13 shows what
        each notch does. Explain in one or two sentences why
        ``the length of the coast of Brittany'' is a slippery
        number, while ``the \omterm{def:g6:measure:area}{area} of Brittany'' is not.
  \item The farmer's exam. With $36$ m of fence and the barn
        wall available, find the best rectangular pen (use the
        mirror of question 7), give its \omterm{def:g6:measure:area}{area}, and compare with
        the best pen built with $36$ m in open field. How much
        does the barn wall earn the farmer?
\end{enumerate}
\end{problem}
