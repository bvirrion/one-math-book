\chapter{Whole Numbers}\label{ch:g6:wholes}

Everything in mathematics starts with counting. This chapter reviews how
whole numbers are written, compared and combined --- and takes a careful
look at division, the operation that gives both a \omterm{def:g3:division:remainder}{quotient} \emph{and} a
\omterm{def:g3:division:remainder}{remainder}.

\section{Writing and comparing whole numbers}

\begin{definition}[Place value]\label{def:g6:wholes:place}
Our way of writing numbers uses ten digits ($0$ to $9$), and the
\emph{value of a digit depends on its place}\index{place value}: in
$5\,208$, the digit $5$ counts thousands, the $2$ counts hundreds, the
$0$ counts tens (there are none) and the $8$ counts units:
\[
5\,208 = 5 \times 1000 + 2 \times 100 + 0 \times 10 + 8 .
\]
\end{definition}

\begin{method}[Comparing two whole numbers]\label{met:g6:wholes:compare}
\begin{enumerate}
  \item The number with more digits is the larger one
        ($1\,203 > 989$).
  \item If they have the same number of digits, compare digit by digit
        from the \emph{left}; the first difference decides:
        $6\,742 > 6\,715$ because at the tens place, $4 > 1$.
\end{enumerate}
The symbols are $<$ (``less than'') and $>$ (``greater than''); the
small end of the symbol points at the small number.
\end{method}

\begin{example}\label{ex:g6:wholes:order}
Order from smallest to largest: $908$; $89$; $1\,001$; $980$.
First by digit count: $89$ (two digits) comes first, $1\,001$ (four
digits) last. Between $908$ and $980$: same hundreds digit $9$, then
tens $0 < 8$, so $908 < 980$. Final answer:
\[
89 < 908 < 980 < 1\,001 .
\]
\end{example}

\section{Addition, subtraction, multiplication}

\begin{example}[Column computations]\label{ex:g6:wholes:columns}
To add or subtract, align the units under the units, the tens under the
tens, and work from right to left, carrying when needed:
\[
\begin{array}{r}
4\,5\,7 \\
+\ 2\,8\,6 \\
\hline
7\,4\,3
\end{array}
\qquad\qquad
\begin{array}{r}
6\,0\,3 \\
-\ 1\,4\,7 \\
\hline
4\,5\,6
\end{array}
\]
In the addition: $7 + 6 = 13$, write $3$, carry $1$; then
$5 + 8 + 1 = 14$, write $4$, carry $1$; then $4 + 2 + 1 = 7$.
Check the subtraction by adding back: $456 + 147 = 603$.
\end{example}

\begin{remark}[Vocabulary]
The result of an addition is a \emph{sum}\index{sum}; of a subtraction,
a \emph{difference}\index{difference}; of a multiplication, a
\emph{product}\index{product}; of a division, a
\emph{quotient}\index{quotient}. The numbers being combined are the
\emph{terms} (for $+$, $-$) or \emph{factors} (for $\times$).
\end{remark}

\begin{example}[Smart computation]\label{ex:g6:wholes:smart}
Reordering terms or factors often saves work:
\[
17 + 58 + 3 = (17 + 3) + 58 = 20 + 58 = 78 ,
\]
\[
25 \times 17 \times 4 = (25 \times 4) \times 17 = 100 \times 17 = 1700 .
\]
\end{example}

\section{Division}

\begin{theorem}[Euclidean division]\label{thm:g6:wholes:division}
Given two whole numbers $a$ (the \emph{dividend}) and $b \neq 0$ (the
\emph{\omterm{def:g5:division:multiple}{divisor}}), there is exactly one way to write
\[
a = b \times q + r
\qquad\text{with}\qquad
0 \leq r < b .
\]
The number $q$ is the \emph{quotient}\index{quotient} and $r$ the
\emph{remainder}\index{remainder} of the division of $a$ by $b$.
\end{theorem}
\admitted

\begin{example}\label{ex:g6:wholes:division}
Divide $137$ by $4$, step by step.
\begin{enumerate}
  \item How many times does $4$ go into $13$? Three times
        ($4 \times 3 = 12$), \omterm{def:g3:division:remainder}{remainder} $13 - 12 = 1$.
  \item Bring down the $7$: how many times does $4$ go into $17$?
        Four times ($4 \times 4 = 16$), \omterm{def:g3:division:remainder}{remainder} $1$.
\end{enumerate}
So $q = 34$ and $r = 1$:
\[
137 = 4 \times 34 + 1, \qquad\text{and indeed } 0 \leq 1 < 4 .
\]
Check: $4 \times 34 = 136$, and $136 + 1 = 137$.
\end{example}

\begin{definition}[Divisible]\label{def:g6:wholes:divisible}
When the \omterm{def:g3:division:remainder}{remainder} is $0$, we say that $a$ is
\emph{divisible}\index{divisible} by $b$: for instance $135$ is
divisible by $5$ since $135 = 5 \times 27 + 0$.
\end{definition}

\begin{method}[Word problems with division]\label{met:g6:wholes:problems}
After computing $a = bq + r$, always return to the question:
\begin{enumerate}
  \item ``How many full boxes/teams/pages?'' --- the answer is the
        \omterm{def:g3:division:remainder}{quotient} $q$;
  \item ``How many are left over?'' --- the answer is the \omterm{def:g3:division:remainder}{remainder}
        $r$;
  \item ``How many boxes are needed for everyone?'' --- if $r > 0$, one
        \emph{more} than the \omterm{def:g3:division:remainder}{quotient}: $q + 1$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g6:wholes:bus}
$137$ students go on a trip in buses of $40$ seats. Division:
$137 = 40 \times 3 + 17$. Three buses carry $120$ students, and $17$
students remain --- they need a bus too! So $4$ buses are needed, even
though the \omterm{def:g3:division:remainder}{quotient} is $3$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g6:wholes:1}
Write in digits: ``three thousand and forty-seven''; ``twenty thousand
five hundred''; ``one hundred four''. Then write $60\,013$ in words.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:2}
In the number $47\,209$: what is the digit of the hundreds? The digit of
the units? What does the digit $4$ count? Write the number as a \omterm{def:g1:addition:def}{sum} of
\omterm{def:g5:division:multiple}{multiples} of $1$, $10$, $100$, \dots\ as in
\cref{def:g6:wholes:place}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:3}
Copy and complete with $<$ or $>$:
\[
507 \;?\; 570, \qquad
1\,099 \;?\; 989, \qquad
23\,456 \;?\; 23\,465, \qquad
9\,999 \;?\; 10\,000 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:4}
Order from smallest to largest: $2\,043$; $978$; $2\,304$; $2\,034$;
$1\,001$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:5}
Compute in columns: $348 + 576$; \quad $805 - 268$; \quad
$307 \times 26$. Check the subtraction with an addition.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:6}
Compute smartly, grouping terms or factors:
\[
38 + 45 + 12 + 5, \qquad
2 \times 83 \times 50, \qquad
4 \times 78 \times 25 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:7}
For each division, give the \omterm{def:g3:division:remainder}{quotient} and the \omterm{def:g3:division:remainder}{remainder}, and write the
equality $a = b \times q + r$:
\[
95 \div 7, \qquad
230 \div 6, \qquad
504 \div 8 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:8}
Without dividing, explain why the equality $173 = 8 \times 20 + 13$ is
\emph{not} the Euclidean division of $173$ by $8$, and find the correct
\omterm{def:g3:division:remainder}{quotient} and \omterm{def:g3:division:remainder}{remainder}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:wholes:9}
Eggs are packed in boxes of $12$.
\begin{enumerate}
  \item A farm has $2\,000$ eggs. How many full boxes can it fill, and
        how many eggs are left over?
  \item A bakery needs $150$ eggs. How many boxes must it buy?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:wholes:10}
Today is a Tuesday. What day of the week will it be in $100$ days?
(Weeks have $7$ days: divide and think about the \omterm{def:g3:division:remainder}{remainder}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:wholes:11}
In a division by $9$, the \omterm{def:g3:division:remainder}{quotient} is $56$ and the \omterm{def:g3:division:remainder}{remainder} is the
largest it can possibly be. What is the dividend?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:wholes:12}
I am a three-digit number. My hundreds digit is twice my units digit,
my tens digit is $0$, and the \omterm{def:g1:addition:def}{sum} of my digits is $9$. Who am I?
(Reason step by step, then check.)
\end{exercise}

\section{Problem: Little Gauss and the sum of the first hundred
numbers}

\begin{problem}[{Weekend problem --- adding $1 + 2 + 3 + \dots +
100$ in ten seconds, and the triangular numbers}]
\label{pb:g6:wholes:1}
A famous story: to keep his class busy, a schoolmaster asked his
pupils to add up all the numbers from $1$ to $100$. One boy wrote
$5\,050$ on his slate almost at once --- Carl Friedrich Gauss, nine
years old, who would grow into one of the greatest mathematicians
of all time. His secret was nothing but \emph{smart computation}
(\cref{ex:g6:wholes:smart}): grouping the terms so that every group
is easy. This problem rediscovers his trick, makes it work in every
situation, and shows how often these \omterm{def:g1:addition:def}{sums} appear in daily life.

\medskip
\noindent\textbf{Part I --- Gauss's pairs.}
\begin{enumerate}
  \item Compute $1 + 2 + 3 + 4 + 5$ by grouping terms cleverly.
  \item For $1 + 2 + 3 + \dots + 10$: pair the first term with the
        last ($1 + 10$), the second with the next-to-last
        ($2 + 9$), and so on. What is each pair worth? How many
        pairs are there? Finish the computation.
  \item Same method for $1 + 2 + \dots + 20$: what is each pair
        worth, how many pairs, what total?
  \item Now the schoolmaster's \omterm{def:g1:addition:def}{sum}, $1 + 2 + \dots + 100$: explain
        why the pairs $1 + 100$, $2 + 99$, $3 + 98, \dots$ are all
        worth the same, and why there are exactly $50$ of them,
        the last one being $50 + 51$.
  \item Write Gauss's answer as a single multiplication, and
        compute it.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- A trick that never fails.}
\begin{enumerate}[resume]
  \item Try the pairing on $1 + 2 + \dots + 9$: this time one
        number is left without a partner. Which one? Pair around
        it and compute the \omterm{def:g1:addition:def}{sum}.
  \item Here is a version of the trick that works whether the
        count of terms is \omterm{def:g3:numbers:evenodd}{even} or \omterm{def:g3:numbers:evenodd}{odd}: write the \omterm{def:g1:addition:def}{sum} twice, once
        forwards and once backwards, one line under the other,
        \[
        \begin{array}{ccccccc}
        1 & + & 2 & + & \dots & + & 9 \\
        9 & + & 8 & + & \dots & + & 1
        \end{array}
        \]
        and add \emph{column by column}. What is each column
        worth? How many columns are there? Explain why the \omterm{def:g1:addition:def}{sum} of
        the two lines is $9 \times 10 = 90$, and conclude that
        $1 + 2 + \dots + 9 = 45$ --- the same answer as in
        question 6.
  \item Use the two-line trick to compute $1 + 2 + \dots + 50$,
        then $1 + 2 + \dots + 200$.
  \item Compute the \omterm{def:g1:addition:def}{sum} of the first hundred \emph{\omterm{def:g3:numbers:evenodd}{even}} numbers,
        $2 + 4 + 6 + \dots + 200$. (Compare each term with a term
        of Gauss's \omterm{def:g1:addition:def}{sum}: no new pairing is needed.)
  \item Compute $101 + 102 + \dots + 200$, using two \omterm{def:g1:addition:def}{sums} you
        already know and a subtraction.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- \omterm{pb:g6:wholes:1}{Triangular numbers} everywhere.}
The totals $1$, $3$, $6$, $10$, $15, \dots$ of the \omterm{def:g1:addition:def}{sums}
$1 + 2 + \dots + n$ are called \emph{triangular
numbers}\index{triangular numbers}, because they count objects
stacked in a triangle.
\begin{enumerate}[resume]
  \item Eight friends meet, and each one shakes hands exactly once
        with each of the others. Explain why the number of
        handshakes is $7 + 6 + 5 + 4 + 3 + 2 + 1$, and compute it.
  \item A grocer builds a triangular \omterm{def:g5:solids:def}{pyramid} of cans: $12$ cans in
        the bottom row, $11$ in the next, and so on up to a single
        can on top. How many cans does she need? (Use the trick of
        question 7.)
  \item Draw the \omterm{def:g1:addition:def}{sum} $1 + 2 + 3 + 4$ as a staircase of squares:
        one square in the first column, two in the second, three
        in the third, four in the fourth. Show on your drawing
        that \emph{two} such staircases, one of them turned upside
        down, fit together exactly into a rectangle of $4$ rows of
        $5$ squares --- and explain why this is the two-line trick
        of question 7, drawn instead of written.
  \item A clock strikes $1$ time at one o'clock, $2$ times at two
        o'clock, \dots, $12$ times at twelve o'clock. How many
        strikes is that over the twelve hours? And over a whole
        day?
  \item A last one for Gauss: $5\,050$ seconds --- is that more or
        less than an hour and a \omterm{def:g3:division:half}{half}? Convert it into hours,
        minutes and seconds, using Euclidean division
        (\cref{thm:g6:wholes:division}) twice.
\end{enumerate}
\end{problem}
