\chapter{Axial Symmetry}\label{ch:g6:symmetry}

Fold a sheet of paper along a line: every point lands on another point,
its mirror image. This folding is \emph{\omterm{def:g6:symmetry:def}{axial symmetry}} (or
reflection), the first geometric transformation of the book --- the
second, central symmetry, comes in \cref{ch:g7:central}.

\section{Reflections}

\begin{definition}[Symmetric point]\label{def:g6:symmetry:def}
Let $d$ be a line. The \emph{symmetric}\index{axial symmetry} of a
point $M$ across $d$ is the point $M'$ such that $d$ is the
\omterm{def:g6:symmetry:bisector}{perpendicular bisector} of $[MM']$: the \omterm{def:g6:lines:objects}{segment} $[MM']$ is \omterm{def:g6:lines:perp}{perpendicular}
to $d$, and $d$ cuts it at its midpoint. A point \emph{on} $d$ is its
own symmetric.
\end{definition}

\begin{method}[Constructing a symmetric point]\label{met:g6:symmetry:construct}
To construct the \omterm{def:g6:symmetry:def}{symmetric} $M'$ of $M$ across the line $d$:
\begin{enumerate}
  \item draw the \omterm{def:g6:lines:perp}{perpendicular} to $d$ through $M$ (set square); call
        $H$ the point where it crosses $d$;
  \item measure $MH$ with the compass or ruler;
  \item place $M'$ on the same \omterm{def:g6:lines:perp}{perpendicular}, on the other side of $d$,
        with $HM' = MH$.
\end{enumerate}
On grid paper it is faster: count the squares from $M$ to the axis, and
count the same number on the other side.
\end{method}

\begin{example}[On a grid]\label{ex:g6:symmetry:grid}
If the axis is a vertical grid line and $M$ is $3$ squares to its left,
then $M'$ is $3$ squares to its right, on the same row. For a diagonal
axis, the counting works along diagonals --- always \omterm{def:g6:lines:perp}{perpendicular} to
the axis.
\end{example}

\begin{proposition}[What a reflection preserves]\label{prop:g6:symmetry:preserved}
The \omterm{def:g6:symmetry:def}{symmetric} of a figure across a line is a figure of the \emph{same
shape and same size}: lengths, angles, \omterm{def:g4:measure:perimeter}{perimeters} and \omterm{def:g4:measure:area}{areas} are all
preserved. The \omterm{def:g6:symmetry:def}{symmetric} of a \omterm{def:g6:lines:objects}{segment} is a \omterm{def:g6:lines:objects}{segment}, of a circle a
circle of the same \omterm{def:g3:shapes:shapes}{radius}, of a line a line.
\end{proposition}
\admitted

\begin{remark}
One thing is \emph{not} preserved: orientation. A reflected ``b''
becomes a ``d'' --- mirror writing. If tracing your figure and flipping
the tracing paper reproduces the image, the reflection is correct.
\end{remark}

\section{Axes of symmetry of a figure}

\begin{definition}[Axis of symmetry]\label{def:g6:symmetry:axis}
A line $d$ is an \emph{axis of symmetry}\index{axis of symmetry} of a
figure when the reflection across $d$ sends the figure exactly onto
itself --- folding along $d$ makes the two halves match.
\end{definition}

\begin{omfigure}
\begin{tikzpicture}[scale=0.62]
  % isosceles triangle
  \draw[thick] (0,0) -- (2.4,0) -- (1.2,2.6) -- cycle;
  \draw[omThm, dashed] (1.2,-0.5) -- (1.2,3.0);
  \node[below, font=\small] at (1.2,-0.7) {isosceles: $1$ axis};
  % rectangle
  \begin{scope}[xshift=5.6cm]
  \draw[thick] (0,0.3) rectangle (3.2,2.3);
  \draw[omThm, dashed] (1.6,-0.4) -- (1.6,3.0);
  \draw[omThm, dashed] (-0.5,1.3) -- (3.7,1.3);
  \node[below, font=\small] at (1.6,-0.7) {rectangle: $2$ axes};
  \end{scope}
  % square
  \begin{scope}[xshift=12.0cm]
  \draw[thick] (0.3,0.1) rectangle (2.7,2.5);
  \draw[omThm, dashed] (1.5,-0.5) -- (1.5,3.1);
  \draw[omThm, dashed] (-0.3,1.3) -- (3.3,1.3);
  \draw[omThm, dashed] (0.0,-0.2) -- (3.0,2.8);
  \draw[omThm, dashed] (0.0,2.8) -- (3.0,-0.2);
  \node[below, font=\small] at (1.5,-0.7) {square: $4$ axes};
  \end{scope}
\end{tikzpicture}

{\small Axes of symmetry of familiar shapes. Careful with the
rectangle: its diagonals are \emph{not} axes of symmetry (fold along a
diagonal --- the halves do not match unless it is a square).}
\end{omfigure}

\begin{example}\label{ex:g6:symmetry:shapes}
Counting axes of symmetry:
\begin{itemize}
  \item \omterm{def:g6:shapes:triangles}{isosceles triangle}: $1$ (through the top \omterm{def:g5:solids:def}{vertex} and the middle
        of the base); \omterm{def:g6:shapes:triangles}{equilateral triangle}: $3$;
  \item rectangle: $2$ (the two ``middle lines'', \emph{not} the
        diagonals); \omterm{def:g6:shapes:quadrilaterals}{rhombus}: $2$ (its diagonals!); square: $4$;
  \item circle: every line through the center --- infinitely many.
\end{itemize}
\end{example}

\begin{example}[Symmetry proves equalities]\label{ex:g6:symmetry:proves}
Why are the two base angles of an \omterm{def:g6:shapes:triangles}{isosceles triangle} equal? Fold the
triangle along its \omterm{def:g6:symmetry:axis}{axis of symmetry}: the left \omterm{def:g3:division:half}{half} lands exactly on the
right \omterm{def:g3:division:half}{half}, so the left base angle lands on the right base angle ---
they must have the same measure
(\cref{prop:g6:symmetry:preserved}: reflections preserve angles).
\end{example}

\section{Perpendicular bisector}

\begin{definition}[Perpendicular bisector]\label{def:g6:symmetry:bisector}
The \emph{perpendicular bisector}\index{perpendicular bisector} of a
\omterm{def:g6:lines:objects}{segment} $[AB]$ is the line \omterm{def:g6:lines:perp}{perpendicular} to $[AB]$ through its
midpoint --- it is the \omterm{def:g6:symmetry:axis}{axis of symmetry} that swaps $A$ and $B$.
\end{definition}

\begin{proposition}[Equidistance]\label{prop:g6:symmetry:bisector}
A point $M$ is on the \omterm{def:g6:symmetry:bisector}{perpendicular bisector} of $[AB]$ exactly when it
is at the same distance from $A$ and from $B$: $MA = MB$.
\end{proposition}
\admitted

\begin{method}[Compass construction]\label{met:g6:symmetry:bisector}
To draw the \omterm{def:g6:symmetry:bisector}{perpendicular bisector} of $[AB]$ without measuring:
\begin{enumerate}
  \item open the compass to more than \omterm{def:g3:division:half}{half} of $AB$;
  \item draw two arcs centered at $A$, above and below the \omterm{def:g6:lines:objects}{segment};
  \item with the \emph{same} opening, draw two arcs centered at $B$;
  \item join the two crossing points: this line is the \omterm{def:g6:symmetry:bisector}{perpendicular
        bisector} (each crossing point is equidistant from $A$ and $B$).
\end{enumerate}
\end{method}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:1}
Copy on grid paper: a vertical axis $d$, a point $A$ two squares to its
left, a point $B$ five squares to its left and one square higher.
Construct their \omterm{def:g6:symmetry:def}{symmetric} points $A'$ and $B'$ across $d$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:2}
Draw a line $d$ and a point $M$ at $3$ cm from $d$ (not on grid
paper). Construct the \omterm{def:g6:symmetry:def}{symmetric} $M'$ of $M$ across $d$ with set square
and ruler, following \cref{met:g6:symmetry:construct}. How far apart
are $M$ and $M'$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:3}
Which capital letters of the alphabet, written in their simplest form,
have a vertical \omterm{def:g6:symmetry:axis}{axis of symmetry} (like A)? A horizontal one (like B)?
Both?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:4}
How many axes of symmetry has: an \omterm{def:g6:shapes:triangles}{equilateral triangle}? a rectangle
(that is not a square)? a \omterm{def:g6:shapes:quadrilaterals}{rhombus} (not a square)? a square? a circle?
Draw each figure with its axes.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:5}
True or false? ``The diagonals of a rectangle are axes of symmetry of
the rectangle.'' Explain with a folding argument or a drawing.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:6}
A triangle $ABC$ is reflected across a line $d$ into $A'B'C'$. Given
$AB = 4$ cm, $\widehat{BAC} = 70^\circ$ and the \omterm{def:g4:measure:perimeter}{perimeter} of $ABC$ is
$12$ cm: give, without any construction, $A'B'$,
$\widehat{B'A'C'}$ and the \omterm{def:g4:measure:perimeter}{perimeter} of $A'B'C'$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:7}
Draw a \omterm{def:g6:lines:objects}{segment} $[AB]$ of $7$ cm and construct its \omterm{def:g6:symmetry:bisector}{perpendicular
bisector} with the compass (\cref{met:g6:symmetry:bisector}). Choose any
point $M$ on it and check with the ruler that $MA = MB$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:symmetry:8}
Two trees stand at points $A$ and $B$ of a (flat) garden. Where can one
plant a fountain so that it is at the same distance from both trees?
Describe \emph{all} the possible spots.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:symmetry:9}
On grid paper, draw the axis $d$ (a diagonal of the grid at
$45^\circ$) and a small L-shaped figure on one side. Construct its
\omterm{def:g6:symmetry:def}{symmetric} figure across $d$. (Reflect each \omterm{def:g5:solids:def}{vertex}: for a $45^\circ$
axis, a point $2$ squares right of the axis lands $2$ squares above
it.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:symmetry:10}
Draw a circle of center $O$, a line $d$ through $O$, and a point $M$ on
the circle. Where is the \omterm{def:g6:symmetry:def}{symmetric} of $M$ across $d$? Explain why the
reflection sends the circle onto itself.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:symmetry:11}
Using the equidistance property (\cref{prop:g6:symmetry:bisector}),
explain why the two crossing points of the arcs in
\cref{met:g6:symmetry:bisector} really do lie on the \omterm{def:g6:symmetry:bisector}{perpendicular
bisector} of $[AB]$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:symmetry:12}
A billiard ball at point $A$ must bounce off a straight wall $d$ and
reach point $B$ (both on the same side of the wall). Reflect $B$
across $d$ into $B'$, and draw the \omterm{def:g6:lines:objects}{segment} $[AB']$. Explain why the
best bouncing point is where $[AB']$ crosses the wall. (Idea: the path
$A \to P \to B$ has the same length as $A \to P \to B'$.)
\end{exercise}

\section{Problem: The fountain, the broken plate, and the walking
mirror}

\begin{problem}[{Weekend problem --- the three perpendicular
bisectors of a triangle meet at one point: the circle through three
points, and what two mirrors do together}]
\label{pb:g6:symmetry:1}
In \cref{exo:g6:symmetry:8} a fountain had to stand at equal
distance from \emph{two} trees, and the answer was a whole line of
spots. This problem plants a \emph{third} tree --- and the whole
line collapses to a single, perfectly determined point. That
point hides a treasure of geometry: it lets you draw the one
circle passing through three given points, rebuild a whole plate
from a broken shard, and it belongs to every triangle. The last
part performs an experiment with two mirrors whose result
announces next year's geometry.

\medskip
\noindent\textbf{Part I --- Two trees, then a map of
territories.}
\begin{enumerate}
  \item Draw two points $A$ and $B$ with $AB = 6$ cm and
        construct the \omterm{def:g6:symmetry:bisector}{perpendicular bisector} of $[AB]$ with the
        compass (\cref{met:g6:symmetry:bisector}). Pick a point
        $M$ on it and check with the ruler that $MA = MB$.
  \item \Cref{prop:g6:symmetry:bisector} says two things at
        once: every point \emph{on} the bisector is equidistant
        from $A$ and $B$, and every equidistant point \emph{is
        on} the bisector. Which of the two answers the fountain
        question of \cref{exo:g6:symmetry:8}? What does the
        proposition say about a point that is \emph{not} on the
        bisector?
  \item Take a point $M$ on the same side of the bisector as
        $A$. The \omterm{def:g6:lines:objects}{segment} $[MB]$ crosses the bisector at a point
        $P$. Compare the path $M \to P \to B$ with the path
        $M \to P \to A$, and explain why $M$ is closer to $A$
        than to $B$: the bisector splits the sheet into an
        ``$A$-side'' and a ``$B$-side''.
  \item Two schools $A$ and $B$ serve a town. Every child walks
        to the closer school. Draw a small map with $A$ and $B$,
        and shade the exact region of the town served by school
        $A$. What line forms the border?
  \item A third school $C$ opens (place it so the three schools
        make a triangle). Construct the \omterm{def:g6:symmetry:bisector}{perpendicular bisector}
        of $[AB]$ and the \omterm{def:g6:symmetry:bisector}{perpendicular bisector} of $[BC]$, and
        call $O$ their crossing point. What two equalities of
        distances does $O$ satisfy, and why?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Three trees, one point, one
circle.}
\begin{enumerate}[resume]
  \item Deduce from question 5 that $OA = OC$ --- and therefore
        that $O$ also lies on the \omterm{def:g6:symmetry:bisector}{perpendicular bisector} of
        $[AC]$, which you never drew. Conclude: \emph{the three
        \omterm{def:g6:symmetry:bisector}{perpendicular bisectors} of the sides of a triangle all
        pass through one point.}
  \item Explain why the circle of center $O$ and \omterm{def:g3:shapes:shapes}{radius} $OA$
        passes through $B$ and through $C$ as well: it is
        \emph{the} circle through the three points --- the
        \emph{circumscribed circle}\index{circumscribed circle}
        of the triangle $ABC$.
  \item Construct a triangle with sides $6$ cm, $5$ cm and
        $4$ cm (\cref{met:g6:shapes:construct}), then its
        \omterm{pb:g6:symmetry:1}{circumscribed circle}. Why is it enough to draw only
        \emph{two} \omterm{def:g6:symmetry:bisector}{perpendicular bisectors} to find the center?
  \item The broken plate: an archaeologist digs up a shard whose
        only intact part is a piece of the plate's circular rim.
        Describe a recipe that recovers the center and the
        \omterm{def:g3:shapes:shapes}{radius} of the original plate --- and test it: draw an
        arc of circle with your compass, ``forget'' the center,
        mark three points on the arc, and rebuild it.
  \item Could three points ever have \emph{no} circle through
        them? Suppose $A$, $B$, $C$ are aligned. What can be
        said of the \omterm{def:g6:symmetry:bisector}{perpendicular bisectors} of $[AB]$ and of
        $[BC]$ (\cref{prop:g6:lines:facts})? Conclude.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Axes everywhere, and the walking
mirror.}
\begin{enumerate}[resume]
  \item \Cref{ex:g6:symmetry:shapes} counted $3$ axes for the
        \omterm{def:g6:shapes:triangles}{equilateral triangle} and $4$ for the square. Conjecture
        the count for a regular \omterm{def:g4:geometry:polygon}{pentagon} and a regular \omterm{def:g4:geometry:polygon}{hexagon},
        and describe where the axes pass in each case (\omterm{def:g3:numbers:evenodd}{odd
        number} of sides against \omterm{def:g3:numbers:evenodd}{even}: the two situations
        differ).
  \item Fold a sheet of paper once and cut a shape through both
        layers; unfold. Why does the hole \emph{always} have an
        \omterm{def:g6:symmetry:axis}{axis of symmetry}, whatever you cut, and which line is
        the axis?
  \item Fold the sheet twice --- the second fold \omterm{def:g6:lines:perp}{perpendicular}
        to the first --- cut, and unfold. How many axes of
        symmetry does the hole have at least, and how many
        copies of your cut do you see? (Paper snowflakes are
        this idea folded further.)
  \item The walking mirror: on grid paper, draw a vertical axis
        $d_1$, a small flag $F$ to its left, and a second
        vertical axis $d_2$, $4$ squares to the right of $d_1$.
        Reflect $F$ across $d_1$ (image $F'$), then reflect
        $F'$ across $d_2$ (image $F''$). Compare $F$ and $F''$:
        same orientation or mirrored? By how many squares did
        the figure move? Compare with the distance between the
        axes.
  \item Same experiment with $d_2$ \emph{\omterm{def:g6:lines:perp}{perpendicular}} to
        $d_1$: reflect the flag across the vertical axis, then
        the image across the horizontal one. Describe how $F''$
        sits relative to $F$ and to the crossing point of the
        two axes. (You have just discovered next year's
        transformation: \cref{ch:g7:central}.)
\end{enumerate}
\end{problem}
