\chapter{Fracciones: primeros pasos}\label{ch:g6:fractions}

Compartir tres pizzas de manera justa entre cuatro personas le da a cada persona\,\dots\
No un número entero de pizzas. \omterm{def:g6:fractions:def}{fracciones} son los números inventados para
compartiendo. Este capítulo construye la imagen: un \omterm{def:g6:fractions:def}{fracción} como parte de un
entero, como un punto en la recta numérica y como el resultado exacto de una
división.

\section{que significa una fraccion}

\begin{definition}[fracción]\label{def:g6:fractions:def}
Cortar una unidad en $b$ partes iguales, y de ellas tomar $a$: la cantidad
obtenido es el \emph{fracción}\index{fracción}
\[
\frac{a}{b}
\qquad
\begin{array}{l}
a \text{: the \emph{numerador} --- how many parts we take;}\\
b \text{: the \emph{denominador} --- into how many parts the unit is cut.}
\end{array}
\]
\end{definition}

\begin{example}\label{ex:g6:fractions:morethanone}
Un \omterm{def:g6:fractions:def}{fracción} puede ser más grande que $1$: $\frac74$ significa siete \omterm{def:g3:division:half}{cuarteles} ---
una unidad completa (cuatro \omterm{def:g3:division:half}{cuarteles}) y tres \omterm{def:g3:division:half}{cuarteles} más:
\[
\frac74 = 1 + \frac34 .
\]
\end{example}

\begin{proposition}[Fracciones en la recta numérica]\label{prop:g6:fractions:line}
Para colocar $\frac ab$ en la recta numérica, corte cada intervalo unitario en $b$
partes iguales y cuente $a$ partes de $0$.
\end{proposition}

\begin{theorem}[Una fracción es una división]\label{thm:g6:fractions:quotient}
El \omterm{def:g6:fractions:def}{fracción} $\frac ab$ es el número que multiplicado por $b$ da
$a$:
\[
\frac ab \times b = a .
\]
En otras palabras, $\frac ab$ es exactamente el \omterm{def:g3:division:remainder}{cociente} de $a$ por $b$ --- incluso
cuando la división ``no sale igualada''.
\end{theorem}
\admitted

\begin{example}\label{ex:g6:fractions:pizzas}
Tres pizzas para cuatro personas: cada persona recibe $\frac34$ de una pizza.
Verifique con el teorema: $4$ las acciones de $\frac34$ hacen
$\frac34 \times 4 = 3$ pizzas. Algunos \omterm{def:g6:fractions:def}{fracciones} son \omterm{def:g6:decimals:places}{numeros decimales}
($\frac34 = 0.75$), otros no: $\frac13 = 0.333\dots$ nunca se detiene.
\omterm{def:g6:fractions:def}{fracción} es el valor \emph{exacto}; $0.33$ es sólo una aproximación.
\end{example}

\section{fracciones iguales}

\begin{proposition}[Fracciones equivalentes]\label{prop:g6:fractions:equal}
Un \omterm{def:g6:fractions:def}{fracción} no cambia cuando sus \omterm{def:g4:fractions:def}{numerador} y \omterm{def:g4:fractions:def}{denominador} son ambos
multiplicado (o dividido) por el mismo número distinto de cero:
\[
\frac{a}{b} = \frac{a \times k}{b \times k} .
\]
\end{proposition}

\begin{proof}[Idea de prueba]
Cortar cada parte del intercambio en $k$ pedazos más pequeños multiplica ambos
el número de piezas tomadas y el número total de piezas por $k$, pero
La cantidad tomada es la misma. Una imagen lo dice mejor:
\end{proof}

\begin{method}[Simplificando una fracción]\label{met:g6:fractions:simplify}
Busque un número que divida el \omterm{def:g4:fractions:def}{numerador} \emph{y} del \omterm{def:g4:fractions:def}{denominador},
y dividir ambos. Repita hasta que no quede ningún \omterm{def:g5:division:multiple}{divisor} común:
\[
\frac{12}{18} = \frac{12 \div 2}{18 \div 2} = \frac{6}{9}
= \frac{6 \div 3}{9 \div 3} = \frac{2}{3}.
\]
\end{method}

\section{Tomando una fracción de una cantidad}

\begin{method}[Fraccion de una cantidad]\label{met:g6:fractions:ofquantity}
Para calcular $\frac ab$ de una cantidad, divida la cantidad por $b$ (uno
parte), luego multiplique por $a$ (el número de partes):
\[
\frac ab \text{ of } Q = (Q \div b) \times a .
\]
(Multiplicar primero por $a$ y luego dividir por $b$ da el mismo
resultado --- elija el orden que haga que los números sean más agradables.)
\end{method}

\begin{example}\label{ex:g6:fractions:ofquantity}
Calcular $\frac35$ de $40$ euros, paso a paso:
\begin{enumerate}
  \item una quinta parte de $40$: $40 \div 5 = 8$;
  \item tres quintos: $8 \times 3 = 24$.
\end{enumerate}
Entonces $\frac35$ de $40$ euros son $24$ euros. Verificar: el restante
$\frac25$ es $16$ euros y $24 + 16 = 40$.
\end{example}

\section{Problema: la barra de chocolate que nunca podrás terminar}

\begin{problem}[{Problema de fin de semana --- mitades de mitades: la suma
$\frac12 + \frac14 + \frac18 + \dots$ se arrastra hasta $1$ sin
tocándolo}]
\label{pb:g6:fractions:1}
Toma una barra de chocolate. Come \omterm{def:g3:division:half}{medio}. Luego come \omterm{def:g3:division:half}{medio} de \emph{que
queda}. Luego nuevamente \omterm{def:g3:division:half}{medio} de lo que queda, y así sucesivamente. Dos cosas
Parece ser cierto a la vez: nunca terminas la barra (algo que siempre
restos), y aun así te lo comes casi todo. Este problema convierte a ambos
sentimientos en declaraciones exactas sobre \omterm{def:g6:fractions:def}{fracciones} --- y se encuentra, en
El camino, un corredor de la antigua Grecia y un rompecabezas para compartir resuelto
enteramente con un cuchillo que solo corta cosas en \omterm{def:g3:division:half}{medio}.

\medskip
\noindent\textbf{Parte I --- Halves of halves.}
La barra es un rectángulo de $16$ cuadrados iguales.
\begin{enumerate}
  \item Primer bocado: te comes \omterm{def:g3:division:half}{medio} la barra. cuantos cuadrados son
        eso? Escribe el bocado como \omterm{def:g6:fractions:def}{fracción} de la barra de dos maneras,
        con \omterm{def:g4:fractions:def}{denominador} $2$ y con \omterm{def:g4:fractions:def}{denominador} $16$
        (\cref{prop:g6:fractions:equal}).
  \item Segundo bocado: \omterm{def:g3:division:half}{medio} de lo que queda. ¿Cuantos cuadrados? mostrar,
        con la imagen o con equivalente \omterm{def:g6:fractions:def}{fracciones}, que este
        El bocado es $\frac14$ de toda la barra: \emph{\omterm{def:g3:division:half}{medio} of a \omterm{def:g3:division:half}{medio}
        is a \omterm{def:g3:division:half}{cuarto}}.
  \item Tercer y cuarto bocado, misma regla: dar cada uno en
        cuadrados y como \omterm{def:g6:fractions:def}{fracción} de la barra. ¿Qué pasa con el
        \omterm{def:g4:fractions:def}{denominador} de un bocado a otro?
  \item Después de los cuatro bocados, ¿cuántos cuadritos se han comido en
        todos? Que \omterm{def:g6:fractions:def}{fracción} de la barra es eso, y que \omterm{def:g6:fractions:def}{fracción}
        queda?
  \item La pregunta 4 dice, escrita en semicorcheas:
        \[
        \frac{8}{16} + \frac{4}{16} + \frac{2}{16} + \frac{1}{16}
        = \frac{15}{16},
        \qquad\text{that is}\qquad
        \frac12 + \frac14 + \frac18 + \frac{1}{16}
        = \frac{15}{16} .
        \]
        Consulta cada una de las cuatro reescrituras.
        ($\frac12 = \frac{8}{16}$, etc.).
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The staircase towards $1$.}
Para seguir mordiendo, imagina una barra más fina de cuadrados $64$.
\begin{enumerate}[resume]
  \item Enumere los tamaños de los primeros seis bocados, en cuadrados.
        ($32$, luego\,\dots), y verifique que después del sexto bocado
        queda exactamente un cuadrado. Lo \omterm{def:g6:fractions:def}{fracción} de la barra
        queda?
  \item Después de cada bocado, ¿qué \omterm{def:g6:fractions:def}{fracción} queda de la barra? escribir
        la lista ($\frac12$, $\frac14$, \dots) hasta el sexto
        muerde y describe el patrón. Si un cuchillo imaginario
        podría reducir a la mitad \omterm{def:g3:numbers:evenodd}{incluso} el último cuadrado, lo que quedaría después
        ¿un séptimo bocado?
  \item Explique por qué el total consumido puede llegar a \emph{nunca} y $1$.
        no importa cuántos bocados se den. (Compárese con el
        veintinueve de \cref{pb:g6:decimals:1}: ¿la misma historia?)
  \item Sin embargo, el total supera cualquier objetivo inferior a $1$. un amigo
        te desafía a comer más de $\frac{99}{100}$ del
        barra. ¿Cuál es más pequeño, $\frac{1}{128}$ o
        $\frac{1}{100}$ --- y después de qué morder es el desafío
        ganó?
  \item En una recta numérica de $0$ a $1$ se graduó en dieciseisavos,
        coloque los totales comidos después de cada uno de los primeros cuatro
        picaduras: $\frac12$, $\frac34$, $\frac78$, $\frac{15}{16}$.
        ¿Dónde queda cada nueva tierra total, en comparación con la
        anterior y con $1$?
\end{enumerate}
