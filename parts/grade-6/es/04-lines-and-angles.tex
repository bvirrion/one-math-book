\chapter{Líneas, círculos y ángulos}\label{ch:g6:lines}

La geometría comienza con una regla, una escuadra y un compás. este capitulo
corrige el vocabulario --- puntos, \omterm{def:g6:lines:objects}{segmentos}, rayos, líneas, círculos ---
introduce las dos posiciones especiales de líneas (\omterm{def:g6:lines:perp}{paralelo} y
\omterm{def:g6:lines:perp}{perpendicular}), y enseña cómo medir y dibujar ángulos.

\section{Puntos, segmentos, rectas.}

\begin{definition}[Objetos básicos]\label{def:g6:lines:objects}
Por dos puntos distintos pasan $A$ y $B$:
\begin{itemize}
  \item o \emph{segmento}\index{segmento} $[AB]$: la parte de la línea
        entre $A$ y $B$ (tiene una longitud, escrita $AB$);
  \item el \emph{rayo} $[AB)$: comienza en $A$, pasa por $B$ y
        continúa para siempre;
  \item o \emph{línea}\index{línea} $(AB)$: se extiende para siempre en ambos
        lados. Dos puntos determinan exactamente una recta.
\end{itemize}
Los puntos de la misma recta se denominan \emph{alineado}.
\end{definition}

\section{Rectas paralelas y perpendiculares}

\begin{definition}[Perpendicular, paralela]\label{def:g6:lines:perp}
Dos líneas son \emph{perpendicular}\index{lineas perpendiculares} cuando
cruce en \omterm{def:g3:shapes:rightangle}{ángulo recto}; Escribimos $d_1 \perp d_2$ y marcamos el \omterm{def:g3:shapes:rightangle}{correcto
ángulo} con un pequeño cuadrado. Dos líneas son
\emph{paralelo}\index{lineas paralelas} cuando nunca se encuentran, por lejos que sean
están extendidos; escribimos $d_1 \parallel d_2$.
\end{definition}

\begin{proposition}[Dos datos útiles]\label{prop:g6:lines:facts}
\begin{enumerate}
  \item Si dos líneas son \omterm{def:g6:lines:perp}{perpendicular} a una tercera línea, son
        \omterm{def:g6:lines:perp}{paralelo} entre sí.
  \item Si dos líneas son \omterm{def:g6:lines:perp}{paralelo}, cada línea \omterm{def:g6:lines:perp}{perpendicular} a una es
        \omterm{def:g6:lines:perp}{perpendicular} al otro.
\end{enumerate}
\end{proposition}
\admitted

\begin{method}[Dibujo con la escuadra]\label{met:g6:lines:setsquare}
Para dibujar \omterm{def:g6:lines:perp}{perpendicular} hasta una línea $d$ que pasa por un punto $P$:
\begin{enumerate}
  \item coloque un borde del \omterm{def:g3:shapes:rightangle}{ángulo recto} de la escuadra a lo largo de $d$;
  \item deslice la escuadra a lo largo de $d$ hasta que su otro borde llegue
        $P$;
  \item dibuje la línea a lo largo de ese borde y marque \omterm{def:g3:shapes:rightangle}{ángulo recto}.
\end{enumerate}
Para \omterm{def:g6:lines:perp}{paralelo} a $P$: dibuje un \omterm{def:g6:lines:perp}{perpendicular} a $d$, luego el
Línea \omterm{def:g6:lines:perp}{perpendicular} a \emph{eso} hasta $P$
(\cref{prop:g6:lines:facts}).
\end{method}

\section{circulos}

\begin{definition}[circulo]\label{def:g6:lines:circle}
El \emph{círculo}\index{círculo} del centro $O$ y \omterm{def:g3:shapes:shapes}{radio} $r$ es el conjunto
de todos los puntos a la distancia exactamente $r$ de $O$. Un \omterm{def:g6:lines:objects}{segmento} del
el centro del círculo es un \emph{\omterm{def:g3:shapes:shapes}{radio}}; a \omterm{def:g6:lines:objects}{segmento} uniendo dos puntos
del círculo que pasa por el centro es un \emph{\omterm{def:g4:geometry:circle}{diámetro}} --- su longitud es
$2r$; un \omterm{def:g6:lines:objects}{segmento} que une dos puntos del círculo es un \emph{acorde}.
\end{definition}

``Dibujar el círculo de centro $O$ que pasa por $A$'': abrir la brújula
de $O$ a $A$ --- el \omterm{def:g3:shapes:shapes}{radio} es la distancia $OA$ --- y gira. cada
El punto de este círculo está a la misma distancia de $O$ que de $A$.
\end{example}

\section{Anglos}

\begin{definition}[angulo]\label{def:g6:lines:angle}
Dos rayos $[AB)$ y $[AC)$ con el mismo punto de partida $A$ forman el
\emph{ángulo}\index{ángulo} $\widehat{BAC}$; el punto $A$ es su
\emph{\omterm{def:g5:solids:def}{vértice}} (¡siempre la letra del medio!). Los ángulos se miden en
\emph{grados} ($^\circ$), de $0^\circ$ a $360^\circ$ para un total
girar. Un ángulo es:
\begin{itemize}
  \item \emph{bien} si mide $90^\circ$;
  \item \emph{agudo} si mide menos de $90^\circ$;
  \item \emph{obtuso} si mide entre $90^\circ$ y
        $180^\circ$;
  \item \emph{derecho} si mide $180^\circ$ (los dos rayos forman
        una línea).
\end{itemize}
\end{definition}

\begin{method}[Medición de un ángulo con transportador]\label{met:g6:lines:protractor}
\begin{enumerate}
  \item Coloque el centro del transportador exactamente en el \omterm{def:g5:solids:def}{vértice} del
        ángulo;
  \item alinea su línea $0^\circ$ con un lado del ángulo;
  \item leer la graduación cruzada por el otro lado --- usando el
        escala que comienza en $0$ en el lado alineado;
  \item control de cordura con el ojo: un ángulo agudo debe leer menos de
        $90^\circ$, uno obtuso más.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g6:lines:estimate}
¡Antes de medir, estima! Un ángulo ligeramente más abierto que la esquina.
de una hoja de papel es un poco más de $90^\circ$; \omterm{def:g3:division:half}{medio} y \omterm{def:g3:shapes:rightangle}{ángulo recto} son
$45^\circ$; un tercio de un \omterm{def:g3:shapes:rightangle}{ángulo recto} es $30^\circ$. Si tu transportador
dice $150^\circ$ para un ángulo que parece agudo, leíste mal
Escala: la medida correcta es $180^\circ - 150^\circ = 30^\circ$.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g6:lines:1}
Dibuja tres puntos $A$, $B$, $C$ no alineados. Dibuja en diferentes colores:
el \omterm{def:g6:lines:objects}{segmento} $[AB]$, el rayo $[CA)$, la línea $(BC)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:2}
¿Verdadero o falso? ``$[AB]$ y $[BA]$ son el mismo \omterm{def:g6:lines:objects}{segmento}.''
``$[AB)$ y $[BA)$ son el mismo rayo.'' ``$(AB)$ y $(BA)$ son los
misma línea." Explique cada respuesta.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:3}
Dibuja una línea $d$ y un punto $P$ que no esté en $d$. Construir con el conjunto
cuadrado: \omterm{def:g6:lines:perp}{perpendicular} a $d$ a $P$, luego \omterm{def:g6:lines:perp}{paralelo} a $d$
hasta $P$. Describe tus pasos.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:4}
Las líneas $a$ y $b$ son \omterm{def:g6:lines:perp}{perpendicular} para una línea $c$ y una cuarta
la línea $e$ es \omterm{def:g6:lines:perp}{perpendicular} a $a$. ¿Qué puedes decir sobre $a$ y $b$?
¿Acerca de $e$ y $c$? Justifique con \cref{prop:g6:lines:facts}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:5}
Dibuja un círculo de centro $O$ con \omterm{def:g3:shapes:shapes}{radio} $3$ cm. Coloque un punto $M$ en
el círculo, un punto $N$ adentro, un punto $P$ afuera. ¿Qué puedes decir?
¿Acerca de las distancias $OM$, $ON$, $OP$ en comparación con $3$ cm?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:6}
Un círculo tiene \omterm{def:g4:geometry:circle}{diámetro} $9$ cm. ¿Cuál es su \omterm{def:g3:shapes:shapes}{radio}? Otro tiene \omterm{def:g3:shapes:shapes}{radio}
$2.6$ cm: ¿cuál es su \omterm{def:g4:geometry:circle}{diámetro}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:7}
Nombra el ángulo marcado con tres letras, luego clasifícalo (agudo, recto,
obtuso, recto): un ángulo de $72^\circ$ en \omterm{def:g5:solids:def}{vértice} $R$ entre rayos
hacia $S$ y $T$; un ángulo de $148^\circ$ en \omterm{def:g5:solids:def}{vértice} $B$ entre rayos
hacia $A$ y $C$; un ángulo de $90^\circ$ en \omterm{def:g5:solids:def}{vértice} $O$ entre rayos
hacia $M$ y $N$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:8}
Estima y luego mide con un transportador los tres ángulos de un
triángulo que dibujas tú mismo. Suma las tres medidas: ¿qué encuentras?
(Guarde su respuesta para \cref{ch:g7:triangles}).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:9}
Dibuja un ángulo $\widehat{xOy}$ de $65^\circ$ con un transportador, luego un
ángulo de $130^\circ$. ¿Cómo pudiste sacar el segundo del primero?
sin el transportador?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:lines:10}
En un mapa se dibujan dos pueblos $A$ y $B$. ¿Dónde están los puntos que
están en $3$ cm de $A$ \emph{y} en $2$ cm de $B$? hacer un dibujo
mostrando cuántos puntos de este tipo puede haber ($0$, $1$ o $2$ dependiendo de
la distancia $AB$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:lines:11}
Un reloj marca las 3: ¿cuál es el ángulo entre las dos manecillas? lo mismo
pregunta a las 5 en punto y a las 6 en punto. (Un giro completo es $360^\circ$
durante $12$ horas.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:lines:12}
Dibujar un \omterm{def:g6:lines:objects}{segmento} $[AB]$ de $6$ cm. Construya el punto $C$ tal que
$AC = BC = 6$ cm, usando sólo el compás, y mide el ángulo
$\widehat{CAB}$. ¿Qué triángulo construiste y qué
¿Conjetura sobre sus ángulos?
\end{exercise}

\section{Problema: la geometría de la esfera del reloj}

\begin{problem}[{Problema de fin de semana --- ángulos como fracciones de un
turno: leerlos en un reloj, cazar los momentos en que las manecillas
reunir y dividir un día en un pastel}]
\label{pb:g6:lines:1}
Un reloj es un transportador que indica la hora: su \omterm{def:g5:solids:def}{rostro} es un transportador completo
turno de $360^\circ$, cortado por las marcas de doce horas en doce
ángulos iguales. \Cref{exo:g6:lines:11} midió las manos en
las 3 y las 6 en punto; este problema construye la teoría completa
--- incluyendo los momentos en que la mano \emph{ni} apunta a una marca
--- responde una pregunta que pocos adultos responden correctamente ("¿con qué frecuencia
¿Dos manos están exactamente una encima de la otra?"), y termina por
dividir un día entero en un gráfico circular.

\medskip
\noindent\textbf{Parte I --- \omterm{def:g6:fractions:def}{fracciones} of a turn.}
\begin{enumerate}
  \item Una vuelta completa mide $360^\circ$. cuantos grados son
        \omterm{def:g3:division:half}{medio} una vuelta, una \omterm{def:g3:division:half}{cuarto} de una vuelta, una doceava de vuelta?
  \item Las marcas de doce horas cortan el reloj \omterm{def:g5:solids:def}{rostro} en doce
        ángulos iguales en el centro. cuantos grados hay entre dos
        marcas vecinas? recuperar la respuesta de
        \cref{exo:g6:lines:11} para las 3 en punto contando marcas.
  \item Indique el ángulo entre las manecillas a la 1 en punto, a la
        A las 4 y a las 7. (A las 7 en punto las manecillas
        separe el \omterm{def:g5:solids:def}{rostro} en dos ángulos; ``el ángulo entre
        las manos'' siempre significa la \emph{menor}.)
  \item El minutero da una vuelta completa en $60$ minutos. como
        cuantos grados barre por minuto?
  \item La manecilla de las horas viaja de una marca a la siguiente ---
        $30^\circ$ --- en $60$ minutos. cuantos grados tiene
        \emph{it} ¿barrido por minuto? (A \omterm{def:g6:decimals:places}{numero decimal},
        \cref{ch:g6:decimals}.)
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Times when nothing points at a
mark.}
\begin{enumerate}[resume]
  \item A las 3:30, el minutero apunta al $6$. donde exactamente
        es la manecilla de las horas? Calcula el ángulo de cada mano a partir de la
        $12$ (medindo en el sentido de las agujas del reloj) y deduzca el ángulo entre
        las manos.
  \item A las 6:30 mucha gente adivina que las manos están encima de cada una.
        otro. Primero adivina, luego calcula el ángulo como en cuestión.
        6. ¿Quién tenía razón?
  \item Calcula el ángulo entre las manos a las 9:15.
  \item Explique por qué, en algún momento entre la 1:00 y la 1:10, los dos
        las manos deben estar exactamente una encima de la otra: ¿dónde está la
        manecilla de minutos en relación con la manecilla de horas a la 1:00, y donde
        a las 1:10? ¿Qué mano corre más rápido y por qué se asienta?
        eso?
  \item En doce horas, ¿cuántas veces se sientan las dos manos?
        exactamente uno encima del otro? (Una reunión ocurre en
        cada tramo entre horas sucesivas --- con uno
        excepción: ¿qué pasa entre las 11 y las 12? Enumere el
        tiempos aproximados de reunión y conteo.) ¿Cuántas veces en un
        todo el día?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Angles around a point: pie
charts.}
\begin{enumerate}[resume]
  \item Los doce sectores del reloj \omterm{def:g5:solids:def}{rostro} juntos llenan el
        \omterm{def:g5:solids:def}{rostro}: sus ángulos suman $360^\circ$. Explica por qué
        esto es cierto para la colección \emph{cualquier} de ángulos que
        comparte un \omterm{def:g5:solids:def}{vértice} y completa una vuelta completa a su alrededor, sin
        superposición y sin espacio.
  \item Zoe registra su día de $24$ horas: sueño $9$ h, escuela
        $6$ h, jugar $3$ h, comidas $2$ h, todo lo demás $4$ h.
        Quiere un \emph{gráfico circular}: un disco donde cada actividad
        obtiene un sector, con ángulos proporcionales a los tiempos.
        ¿Qué \omterm{def:g6:fractions:def}{fracción} del día es cada actividad y cuántas
        ¿Qué grados obtiene su sector? Comprueba que los cinco ángulos suman
        a $360^\circ$.
  \item Describe, paso a paso, cómo dibujar el gráfico circular de Zoe con
        compás y transportador
        (\cref{met:g6:lines:protractor}): donde el primer \omterm{def:g3:shapes:shapes}{radio}
        va, y cómo cada nuevo sector comienza donde el anterior
        uno termina.
  \item En otro gráfico circular de un día, un sector mide
        exactamente $90^\circ$. ¿Qué \omterm{def:g6:fractions:def}{fracción} del disco es eso?
        y cuantas horas representa?
  \item El final, volviendo al reloj: en $24$ horas, ¿cuántas
        vueltas completas da la manecilla de las horas? ¿El minutero? el
        segunda mano? (Una de estas respuestas es más de mil).
\end{enumerate}
\end{problem}
