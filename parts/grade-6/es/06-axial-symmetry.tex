\chapter{Simetría axial}\label{ch:g6:symmetry}

Dobla una hoja de papel a lo largo de una línea: cada punto aterriza en otro punto,
su imagen especular. Este plegado es \emph{\omterm{def:g6:symmetry:def}{simetría axial}} (o
reflexión), la primera transformación geométrica del libro --- la
el segundo, simetría central, viene en \cref{ch:g7:central}.

\section{Reflexiones}

\begin{definition}[Punto simetrico]\label{def:g6:symmetry:def}
Sea $d$ una línea. El \emph{simétrico}\index{simetría axial} de un
El punto $M$ a través de $d$ es el punto $M'$ tal que $d$ es el
\omterm{def:g6:symmetry:bisector}{bisectriz perpendicular} de $[MM']$: el \omterm{def:g6:lines:objects}{segmento} $[MM']$ es \omterm{def:g6:lines:perp}{perpendicular}
a $d$, y $d$ lo corta por su punto medio. Un punto \emph{on} $d$ es su
propia simétrica.
\end{definition}

\begin{method}[Construyendo un punto simétrico]\label{met:g6:symmetry:construct}
Para construir el \omterm{def:g6:symmetry:def}{simétrico} $M'$ de $M$ a través de la línea $d$:
\begin{enumerate}
  \item dibuje \omterm{def:g6:lines:perp}{perpendicular} a $d$ a $M$ (escuadra); llamar
        $H$ el punto donde cruza $d$;
  \item mida $MH$ con el compás o regla;
  \item coloque $M'$ en el mismo \omterm{def:g6:lines:perp}{perpendicular}, del otro lado de $d$,
        con $HM' = MH$.
\end{enumerate}
En papel cuadriculado es más rápido: cuenta los cuadrados desde $M$ hasta el eje, y
cuenta el mismo número en el otro lado.
\end{method}

\begin{example}[En una cuadrícula]\label{ex:g6:symmetry:grid}
Si el eje es una línea de cuadrícula vertical y $M$ son $3$ cuadrados a su izquierda,
entonces $M'$ es $3$ cuadrados a su derecha, en la misma fila. para una diagonal
eje, el conteo funciona a lo largo de diagonales --- siempre \omterm{def:g6:lines:perp}{perpendicular} a
el eje.
\end{example}

\begin{proposition}[Que reflejo conserva]\label{prop:g6:symmetry:preserved}
El \omterm{def:g6:symmetry:def}{simétrico} de una figura a través de una línea es una figura del \emph{lo mismo
forma y mismo tamaño}: longitudes, ángulos, \omterm{def:g4:measure:perimeter}{perímetros} y \omterm{def:g4:measure:area}{áreas} son todos
conservado. El \omterm{def:g6:symmetry:def}{simétrico} de un \omterm{def:g6:lines:objects}{segmento} es un \omterm{def:g6:lines:objects}{segmento}, de un círculo a
círculo del mismo \omterm{def:g3:shapes:shapes}{radio}, de una línea una línea.
\end{proposition}
\admitted

\begin{remark}
Una cosa se conserva \emph{no}: la orientación. A reflejada ``b''
se convierte en una ``d'' --- escritura en espejo. Si trazas tu figura y volteas
el papel de calco reproduce la imagen, el reflejo es correcto.
\end{remark}

\section{Ejes de simetría de una figura.}

\begin{definition}[Eje de simetria]\label{def:g6:symmetry:axis}
Una línea $d$ es una \emph{eje de simetría}\index{eje de simetría} de un
figura cuando la reflexión a través de $d$ envía la figura exactamente a
sí mismo --- doblar a lo largo de $d$ hace que las dos mitades coincidan.
\end{definition}

\begin{example}\label{ex:g6:symmetry:shapes}
Contando ejes de simetría:
\begin{itemize}
  \item \omterm{def:g6:shapes:triangles}{triangulo isósceles}: $1$ (a través del \omterm{def:g5:solids:def}{vértice} superior y el medio
        de la base); \omterm{def:g6:shapes:triangles}{triangulo equilatero}: $3$;
  \item Rectángulo : $2$ (las dos ``líneas medias'', \emph{no} la
        diagonales); \omterm{def:g6:shapes:quadrilaterals}{rombo}: $2$ (¡sus diagonales!); cuadrado: $4$;
  \item círculo: cada línea que pasa por el centro --- infinitas.
\end{itemize}
\end{example}

\begin{example}[La simetría demuestra igualdades]\label{ex:g6:symmetry:proves}
¿Por qué son iguales los dos ángulos base de un \omterm{def:g6:shapes:triangles}{triangulo isósceles}? Doblar el
triángulo a lo largo de su \omterm{def:g6:symmetry:axis}{eje de simetría}: el \omterm{def:g3:division:half}{medio} izquierdo aterriza exactamente en el
derecha \omterm{def:g3:division:half}{medio}, por lo que el ángulo de la base izquierda aterriza en el ángulo de la base derecha ---
deben tener la misma medida
(\cref{prop:g6:symmetry:preserved}: los reflejos preservan los ángulos).
\end{example}

\section{bisectriz perpendicular}

\begin{definition}[Bisectriz perpendicular]\label{def:g6:symmetry:bisector}
El \emph{bisectriz perpendicular}\index{bisectriz perpendicular} de un
\omterm{def:g6:lines:objects}{segmento} $[AB]$ es la línea \omterm{def:g6:lines:perp}{perpendicular} a $[AB]$ a través de su
punto medio --- es el \omterm{def:g6:symmetry:axis}{eje de simetría} el que intercambia $A$ y $B$.
\end{definition}

\begin{proposition}[Equidistancia]\label{prop:g6:symmetry:bisector}
Un punto $M$ está en el \omterm{def:g6:symmetry:bisector}{bisectriz perpendicular} de $[AB]$ exactamente cuando
está a la misma distancia de $A$ y de $B$: $MA = MB$.
\end{proposition}
\admitted

\begin{method}[Construcción de la brújula]\label{met:g6:symmetry:bisector}
Para dibujar el \omterm{def:g6:symmetry:bisector}{bisectriz perpendicular} de $[AB]$ sin medir:
\begin{enumerate}
  \item abre la brújula a más de \omterm{def:g3:division:half}{medio} de $AB$;
  \item dibuje dos arcos centrados en $A$, arriba y abajo de \omterm{def:g6:lines:objects}{segmento};
  \item con la apertura \emph{mismo}, dibuje dos arcos centrados en $B$;
  \item une los dos cruces: esta línea es la \omterm{def:g6:symmetry:bisector}{perpendicular
        bisectriz} (cada cruce equidista de $A$ y $B$).
\end{enumerate}
\end{method}
