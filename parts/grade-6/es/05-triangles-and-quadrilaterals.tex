\chapter{Triángulos y Cuadriláteros}\label{ch:g6:shapes}

Los triángulos y \omterm{def:g4:geometry:polygon}{cuadriláteros} son las formas básicas de la geometría. el objetivo
de este capítulo es conocer sus familias especiales --- \omterm{def:g6:shapes:triangles}{isósceles},
\omterm{def:g6:shapes:triangles}{equilátero}, \omterm{def:g6:shapes:triangles}{triangulos rectángulos}; rectángulos, rombos, cuadrados --- por
su \emph{definir propiedades}, y construirlos precisamente con
regla y compás.

\section{Triangulos}

\begin{definition}[Triángulos especiales]\label{def:g6:shapes:triangles}
Un triángulo $ABC$ tiene tres \omterm{def:g5:solids:def}{vértices} y tres lados. Es:
\begin{itemize}
  \item \emph{isósceles}\index{triangulo isósceles} en $A$ cuando
        $AB = AC$ (dos lados iguales);
  \item \emph{equilátero}\index{triangulo equilatero} cuando
        $AB = BC = CA$ (tres lados iguales);
  \item \emph{en ángulo recto}\index{triangulo rectángulo} en $A$ cuando el ángulo
        $\widehat{BAC}$ es un \omterm{def:g3:shapes:rightangle}{ángulo recto}. El lado $[BC]$ opuesto al
        \omterm{def:g3:shapes:rightangle}{ángulo recto} se llama \emph{hipotenusa}.
\end{itemize}
\end{definition}

\begin{method}[Construyendo un triángulo a partir de sus tres lados]\label{met:g6:shapes:construct}
Para construir $ABC$ con $AB = 5$ cm, $AC = 4$ cm, $BC = 3$ cm:
\begin{enumerate}
  \item dibuja el \omterm{def:g6:lines:objects}{segmento} $[AB]$ de $5$ cm con la regla;
  \item dibuja un arco del círculo de centro $A$ y \omterm{def:g3:shapes:shapes}{radio} $4$ cm
        (todos los puntos en $4$ cm desde $A$);
  \item dibuja un arco del círculo de centro $B$ y \omterm{def:g3:shapes:shapes}{radio} $3$ cm;
  \item el punto $C$ es donde se cruzan los dos arcos; únelo a $A$ y
        $B$.
\end{enumerate}
\end{method}

\section{Cuadriláteros}

\begin{definition}[cuadriláteros especiales]\label{def:g6:shapes:quadrilaterals}
Un \omterm{def:g4:geometry:polygon}{cuadrilátero} tiene cuatro \omterm{def:g5:solids:def}{vértices} y cuatro lados. Es un:
\begin{itemize}
  \item \emph{rectángulo}\index{rectángulo} cuando sus cuatro ángulos son
        \omterm{def:g3:shapes:rightangle}{ángulos rectos};
  \item \emph{rombo}\index{rombo} cuando sus cuatro lados tienen iguales
        longitud;
  \item \emph{cuadrado}\index{cuadrado} cuando son ambos: cuatro \omterm{def:g3:shapes:rightangle}{ángulos rectos}
        \emph{y} cuatro lados iguales.
\end{itemize}
\end{definition}

\begin{proposition}[Propiedades de las diagonales]\label{prop:g6:shapes:diagonals}
En estos \omterm{def:g4:geometry:polygon}{cuadriláteros}, las dos diagonales se cortan en sus
punto medio común, y además:
\begin{itemize}
  \item en un rectángulo, las diagonales tienen el \emph{misma longitud};
  \item en un \omterm{def:g6:shapes:quadrilaterals}{rombo}, las diagonales son \emph{\omterm{def:g6:lines:perp}{perpendicular}};
  \item en un cuadrado, ambos a la vez.
\end{itemize}
\end{proposition}
\admitted

\begin{remark}
Un cuadrado es automáticamente un rectángulo (tiene cuatro \omterm{def:g3:shapes:rightangle}{ángulos rectos}) y
automáticamente un \omterm{def:g6:shapes:quadrilaterals}{rombo} (cuatro lados iguales). Entonces "rectángulo" no
significa "no un cuadrado"! En matemáticas, la forma más especial pertenece
a la familia más general.
\end{remark}

\begin{example}[Reconocer una forma por sus propiedades.]\label{ex:g6:shapes:recognize}
Un \omterm{def:g4:geometry:polygon}{cuadrilátero} tiene cuatro lados de longitud $4$ cm y un \omterm{def:g3:shapes:rightangle}{ángulo recto}.
¿Qué es? Cuatro lados iguales lo convierten en \omterm{def:g6:shapes:quadrilaterals}{rombo}. En un \omterm{def:g6:shapes:quadrilaterals}{rombo}, enfrente
los ángulos son iguales y los cuatro ángulos suman $360^\circ$; un \omterm{def:g3:shapes:rightangle}{correcto
ángulo} luego obliga a los cuatro a tener razón (esto se demostrará con
paralelogramos en \cref{ch:g7:central}). Entonces la forma es un cuadrado.
\end{example}

\begin{method}[Programas de construcción]\label{met:g6:shapes:program}
Un \emph{programa de construcción} es una lista de instrucciones que cualquiera puede
siga para reproducir una figura exactamente. Escriba una instrucción por línea,
nombra cada punto cuando se crea, y solo usa: ``dibuja el \omterm{def:g6:lines:objects}{segmento}
/line/circle\dots'', ``coloca el punto donde \dots\ cruza''. prueba
su programa siguiéndolo literalmente, palabra por palabra.
\end{method}

\begin{example}\label{ex:g6:shapes:program}
Programa para un cuadrado $ABCD$ de lado $4$ cm:
\begin{enumerate}
  \item Dibuja un \omterm{def:g6:lines:objects}{segmento} $[AB]$ con $AB = 4$ cm.
  \item Dibuja el \omterm{def:g6:lines:perp}{perpendicular} a $(AB)$ en $A$; en él, coloque $D$ en
        $4$ cm de $A$ (elige un lado de la línea y cíñete a él).
  \item Dibuja el \omterm{def:g6:lines:perp}{perpendicular} a $(AB)$ en $B$; en él, coloque $C$ en
        $4$ cm de $B$, del mismo lado que $D$.
  \item Únete a $C$ y $D$.
\end{enumerate}
Verifique con la regla: $DC$ debe medir $4$ cm también.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g6:shapes:1}
Construya un triángulo $ABC$ con $AB = 6$ cm, $AC = 5$ cm y
$BC = 4$ cm, siguiendo a \cref{met:g6:shapes:construct}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:2}
Construya un \omterm{def:g6:shapes:triangles}{triangulo isósceles} $DEF$ con $DE = DF = 5$ cm y
$EF = 3$cm. ¿Cuáles dos ángulos de este triángulo parecen iguales? Mídelos
para comprobarlo.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:3}
Construya un \omterm{def:g6:shapes:triangles}{triangulo equilatero} de lado $4$ cm usando solo regla y
brújula. Mide uno de sus ángulos.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:4}
Clasifica cada triángulo a partir de sus datos (\omterm{def:g6:shapes:triangles}{isósceles}, \omterm{def:g6:shapes:triangles}{equilátero},
\omterm{def:g6:shapes:triangles}{en ángulo recto} --- pueden aplicarse varias respuestas):
(a) lados $7$, $7$, $7$; \
(b) lados $5$, $5$, $8$; \
(c) ángulos $90^\circ$, $45^\circ$, $45^\circ$; \
(d) ángulos $60^\circ$, $60^\circ$, $60^\circ$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:5}
En un \omterm{def:g6:shapes:triangles}{triangulo rectángulo}, ¿de qué lado está la hipotenusa? dibuja un triangulo
$MNP$ \omterm{def:g6:shapes:triangles}{en ángulo recto} en $N$, y nombra su hipotenusa. ¿Qué hace \omterm{def:g5:solids:def}{vértice}?
¿La hipotenusa nunca se toca?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:6}
¿Verdadero o falso? Justifica cada respuesta en una frase.
\begin{enumerate}
  \item Todo cuadrado es un rectángulo.
  \item Todo rectángulo es un cuadrado.
  \item Cada cuadrado es un \omterm{def:g6:shapes:quadrilaterals}{rombo}.
  \item Un \omterm{def:g6:shapes:quadrilaterals}{rombo} con un \omterm{def:g3:shapes:rightangle}{ángulo recto} es un cuadrado.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:7}
Construya un rectángulo $ABCD$ con $AB = 5$ cm y $BC = 3$ cm. Medida
sus dos diagonales: ¿qué observas?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:8}
Construya un \omterm{def:g6:shapes:quadrilaterals}{rombo} con diagonales de $6$ cm y $4$ cm. (Usar
\cref{prop:g6:shapes:diagonals}: dibuja primero las dos diagonales ---
\omterm{def:g6:lines:perp}{perpendicular}, cruzando por sus puntos medios --- luego unir los cuatro
termina.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:shapes:9}
Escribir un programa de construcción para un \omterm{def:g6:shapes:triangles}{triangulo isósceles} con base.
$4$ cm y lados iguales de $6$ cm, al estilo de
\cref{ex:g6:shapes:program}. Intercambio con un compañero: ¿pueden seguir?
¿Sin ver tu figura?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:shapes:10}
Dibuja un círculo de centro $O$ y uno de sus \omterm{def:g4:geometry:circle}{diámetros} $[AB]$. Coloque cualquier
Punto $M$ en el círculo y dibuja el triángulo $AMB$. medir el angulo
$\widehat{AMB}$. Mueva $M$ y mida nuevamente. ¿Qué conjeturas?
(Este hecho sorprendente se prueba en \cref{ch:g8:cosine}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:shapes:11}
Un \omterm{def:g4:geometry:polygon}{cuadrilátero} tiene diagonales de la misma longitud que se cortan entre sí en
su punto medio común. Determina qué forma debe tener, usando
\cref{prop:g6:shapes:diagonals}: ¿rectángulo, \omterm{def:g6:shapes:quadrilaterals}{rombo} o cuadrado?
(Cuidado: sólo se proporciona una propiedad).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:shapes:12}
En una cuadrícula de puntos $4 \times 4$, ¿cuántos cuadrados diferentes puedes dibujar?
¿Cuyas esquinas son puntos de la cuadrícula, contando también los cuadrados "inclinados"?
Dibuja al menos un cuadrado inclinado y explica por qué sus lados son iguales.
\end{exercise}

\section{Problema: lo que saben las diagonales}

\begin{problem}[{Problema del fin de semana --- reconstruyendo cada especial
cuadrilátero sólo por sus diagonales y contando las diagonales
de cualquier polígono}]
\label{pb:g6:shapes:1}
\Cref{prop:g6:shapes:diagonals} lee una forma y te informa sobre
sus diagonales. Este problema juega el juego \emph{hacia atrás}: dado
solo dos se cruzan \omterm{def:g6:lines:objects}{segmentos} --- las futuras diagonales --- que forma
aparece cuando unes sus cuatro extremos? Construirás un completo
``diccionario diagonal'', entiende \emph{por qué} o \omterm{def:g6:shapes:quadrilaterals}{rombo}
construcción de las obras \cref{exo:g6:shapes:8}, cumplen con una forma que
aún no tiene nombre, y terminamos contando las diagonales de \omterm{def:g4:geometry:polygon}{polígonos}
con muchos más de cuatro lados.

\medskip
\noindent\textbf{Parte I --- The diagonal dictionary.}
Para cada pregunta, dibuja los dos \omterm{def:g6:lines:objects}{segmentos} como se describe, une sus
cuatro extremos en orden e identifique el \omterm{def:g4:geometry:polygon}{cuadrilátero} que obtiene.
\begin{enumerate}
  \item Dos \omterm{def:g6:lines:objects}{segmentos} de $6$ cm cada uno, \omterm{def:g6:lines:perp}{perpendicular}, cruce en
        su punto medio común. Mida los cuatro lados y el
        cuatro ángulos de la forma resultante: ¿qué es?
  \item Dos \omterm{def:g6:lines:objects}{segmentos} de $8$ cm y $5$ cm, \omterm{def:g6:lines:perp}{perpendicular},
        cruce en su punto medio común. ¿Qué forma aparece?
        (\cref{exo:g6:shapes:8})?
  \item Dos \omterm{def:g6:lines:objects}{segmentos} de $7$ cm cada uno --- iguales pero \emph{no}
        \omterm{def:g6:lines:perp}{perpendicular} --- cruce en su punto medio común.
        Mide los lados y los ángulos: ¿cuál conjeturas
        forma de ser?
  \item Dos \omterm{def:g6:lines:objects}{segmentos} de $8$ cm y $5$ cm, no \omterm{def:g6:lines:perp}{perpendicular},
        cruce en su punto medio común. La forma que obtienes es
        ninguno de los tres
        \cref{def:g6:shapes:quadrilaterals} --- describe lo que
        Ya ves (¿qué lados parecen iguales? \omterm{def:g6:lines:perp}{paralelo}?). esto
        \omterm{def:g4:geometry:polygon}{cuadrilátero} recibe su nombre en \cref{ch:g7:central}.
  \item Resume las preguntas 1 a 4 en un diccionario con cuatro
        entradas: diagonales \emph{equal / \omterm{def:g6:lines:perp}{perpendicular} / both /
        neither} (siempre cruzando en sus puntos medios)
        $\rightarrow$ nombre (o descripción) de la forma.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Why the constructions work.}
\begin{enumerate}[resume]
  \item En la pregunta 2, las dos diagonales cortan la forma en cuatro
        \omterm{def:g6:shapes:triangles}{triangulos rectángulos} alrededor del punto de cruce. ¿Cuáles son los
        ¿La longitud de dos catetos de cada uno de estos triángulos? Explica por qué
        los cuatro triángulos son copias exactas uno del otro, y
        deduce por qué los cuatro lados de la figura son iguales --- el
        secreto de \cref{exo:g6:shapes:8}.
  \item Prueba de deducir solo con brújula: después de construir cualquier forma
        desde sus diagonales, ¿cómo se puede comprobar con la brújula?
        solo (sin medidas de regla) que los cuatro lados son
        ¿iguales?
  \item Escribir un programa de construcción.
        (\cref{met:g6:shapes:program}) para un cuadrado cuyo
        \emph{diagonales} mide $6$ cm --- no sus lados.
  \item En el cuadrado de la pregunta 8, ¿el lado es más largo o
        más corto que $3$ cm --- \omterm{def:g3:division:half}{medio} la diagonal? Medir hasta
        descúbrelo. Un \omterm{def:g3:division:half}{medio} de la respuesta ya puede ser
        \emph{probado}: yendo de una esquina a la siguiente por el
        centro es un desvío $3 + 3$ cm, y un camino recto es
        siempre más corto que un desvío --- entonces el lado es más corto
        de $6$ cm. (Demostrando que también supera a $3$ cm, y
        calcularlo exactamente, es un trabajo para
        \cref{ch:g8:pythagoras}.)
  \item Una nueva entrada para el diccionario: dibuja un \omterm{def:g6:lines:objects}{segmento} $[AC]$ de
        $8$cm; en él marcar el punto $H$ a $3$ cm de $A$;
        dibujar a través de $H$ el \omterm{def:g6:lines:perp}{perpendicular} \omterm{def:g6:lines:objects}{segmento} $[BD]$ de
        $5$ cm con $H$ como punto medio \emph{es}. Únase a $ABCD$. esto
        La forma es \emph{cometa}\index{cometa} (una diagonal biseca
        el otro, en un \omterm{def:g3:shapes:rightangle}{ángulo recto}). Usando triángulos de esquina como
        en la pregunta 6, encuentra qué lados de un \omterm{pb:g6:shapes:1}{cometa} son iguales a
        cual.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Counting diagonals.}
Un \emph{diagonal} de un \omterm{def:g4:geometry:polygon}{polígono} une dos \omterm{def:g5:solids:def}{vértices} que no están
vecinos.
\begin{enumerate}[resume]
  \item Dibuja un \omterm{def:g4:geometry:polygon}{cuadrilátero}, un \omterm{def:g4:geometry:polygon}{pentágono} (cinco \omterm{def:g5:solids:def}{vértices}) y un
        \omterm{def:g4:geometry:polygon}{hexágono} (seis \omterm{def:g5:solids:def}{vértices}), y dibuja todas sus diagonales.
        Cuéntalos: ¿cuántos en cada caso?
  \item Contar sin dibujar, para el \omterm{def:g4:geometry:polygon}{hexágono}: desde uno
        \omterm{def:g5:solids:def}{vértice}, ¿cuántas diagonales quedan? (¿A cuántos de los
        seis \omterm{def:g5:solids:def}{vértices} ¿se puede unir \emph{no} por una diagonal?)
        Multiplicar por seis \omterm{def:g5:solids:def}{vértices} cuenta cada diagonal
        exactamente dos veces --- ¿por qué? Recupera tu conteo de pregunta
        11. (La misma doble contabilización apareció en los apretones de manos
        de \cref{pb:g6:wholes:1}.)
  \item Utilice el método de la pregunta 12 para contar las diagonales de
        a \omterm{def:g4:geometry:polygon}{polígono} con $10$ \omterm{def:g5:solids:def}{vértices}, luego con $20$ \omterm{def:g5:solids:def}{vértices}.
  \item A \omterm{def:g4:geometry:polygon}{polígono} tiene exactamente $44$ diagonales. Cuantos \omterm{def:g5:solids:def}{vértices}
        tiene? (Pruebe el método de la pregunta 13 en algunos
        candidatos.)
  \item Comprueba tu método en el \omterm{def:g4:geometry:polygon}{polígono} más pequeño de todos: cómo
        ¿Cuántas diagonales tiene un triángulo y cuál es?
        ¿Método de conteo de la respuesta a la pregunta 12? concluir con uno
        frase sobre lo que tienen los apretones de manos y las diagonales
        común.
\end{enumerate}
\end{problem}
