\chapter{Decimale getallen}\label{ch:g6:decimals}

Tussen $2$ en $3$ liggen heel wat getallen: $2.5$, $2.71$,
$2.999$\,\dots\ Met decimale getallen breiden we het
plaatswaardesysteem uit naar rechts van de eenheden, en ze volgen
dezelfde regels als gehele getallen --- met een paar valkuilen die dit
hoofdstuk je leert vermijden.

\section{Decimale schrijfwijze}

\begin{definition}[Decimalen]\label{def:g6:decimals:places}
Een \emph{decimaal getal}\index{decimaal getal} heeft een geheel deel
en een decimaal deel, gescheiden door de \omterm{def:g4:decimals:point}{decimale punt}. Elke plaats
rechts van de punt is tien keer minder waard dan de vorige: tienden,
honderdsten, duizendsten. Bijvoorbeeld
\[
13.407 = 13 + \frac{4}{10} + \frac{0}{100} + \frac{7}{1000}
= 1 \times 10 + 3 + 4 \times \frac{1}{10} + 7 \times \frac{1}{1000}.
\]
\end{definition}

\begin{remark}[Nullen die tellen, nullen die niet tellen]
Nullen \emph{aan het eind} van het decimale deel veranderen niets:
$2.5 = 2.50 = 2.500$. Maar nullen \emph{tussen} cijfers zijn
essentieel: $13.407 \neq 13.47$. En $0.5$ is heel anders dan $0.05$!
\end{remark}

\begin{method}[Decimale getallen vergelijken]\label{met:g6:decimals:compare}
\begin{enumerate}
  \item Vergelijk eerst de gehele delen: $7.2 > 5.99$ want $7 > 5$.
  \item Als de gehele delen gelijk zijn, vergelijk de decimale delen
        cijfer voor cijfer van links --- bijvullen met eindnullen helpt:
        om $3.4$ en $3.15$ te vergelijken, schrijf $3.40$ en $3.15$;
        omdat $40 > 15$ honderdsten, is $3.4 > 3.15$.
\end{enumerate}
Let op: \emph{langer betekent niet groter}. $3.15$ heeft meer cijfers
dan $3.4$ maar is kleiner.
\end{method}

\section{Optellen, aftrekken, vermenigvuldigen}

\begin{method}[Kolomberekeningen met decimalen]\label{met:g6:decimals:columns}
Om decimale getallen op te tellen of af te trekken, lijn je de
\emph{\omterm{def:g4:decimals:point}{decimale punten}} uit (zodat eenheden onder eenheden staan,
tienden onder tienden), vul eventueel aan met eindnullen; reken daarna
als bij gehele getallen, en zet de punt van het resultaat onder de
andere.
\end{method}

\begin{example}\label{ex:g6:decimals:addsub}
Bereken $13.7 + 2.85$, met uitgelijnde punten ($13.7 = 13.70$):
\[
\begin{array}{r}
1\,3.7\,0 \\
+\ \ \,2.8\,5 \\
\hline
1\,6.5\,5
\end{array}
\]
Bereken $6 - 2.35$ (schrijf $6 = 6.00$):
$6.00 - 2.35 = 3.65$. Controle: $3.65 + 2.35 = 6$.
\end{example}

\begin{example}[Decimale getallen vermenigvuldigen]\label{ex:g6:decimals:mult}
Om $2.3 \times 1.4$ te berekenen: vermenigvuldig als gehele getallen,
$23 \times 14 = 322$; tel daarna de decimale cijfers van de factoren
($1 + 1 = 2$), en plaats de punt zodat het resultaat zoveel decimalen
heeft: $2.3 \times 1.4 = 3.22$. Controle op grootteorde:
$2.3 \times 1.4$ moet iets meer zijn dan $2.3 \times 1 = 2.3$ --- en
$3.22$ is dat.
\end{example}

\section{Vermenigvuldigen en delen door 10, 100, 1000}

\begin{proposition}[De punt verschuiven]\label{prop:g6:decimals:shift}
Een \omterm{def:g6:decimals:places}{decimaal getal} vermenigvuldigen met $10$, $100$, $1000$
verschuift de \omterm{def:g4:decimals:point}{decimale punt} $1$, $2$, $3$ plaatsen naar
\emph{rechts}; delen verschuift hem naar \emph{links} (met nullen
aanvullen indien nodig):
\[
3.75 \times 100 = 375, \qquad
42.1 \div 1000 = 0.0421 .
\]
\end{proposition}

\begin{proof}
Vermenigvuldigen met $10$ maakt elk cijfer tien keer meer waard:
tienden worden eenheden, eenheden worden tientallen, enzovoort ---
elk cijfer schuift één kolom naar links in de plaatswaardetabel, wat
hetzelfde is als de punt één plaats naar rechts te schuiven. Delen
keert dit om.
\end{proof}

\begin{example}[Meeteenheden]\label{ex:g6:decimals:units}
Eenheden omrekenen is precies dit spel: omdat $1$ m $= 100$ cm,
\[
3.75 \text{ m} = 3.75 \times 100 \text{ cm} = 375 \text{ cm},
\qquad
42 \text{ mm} = 42 \div 10 \text{ cm} = 4.2 \text{ cm}.
\]
\end{example}

\section{Afronden}

\begin{definition}[Afronden]\label{def:g6:decimals:rounding}
Het \emph{afronden}\index{afronden} van een getal op de eenheid (of
tiende, honderdste, \dots) is het dichtstbijzijnde getal met die
nauwkeurigheid. Kijk naar het volgende cijfer: als het $0,1,2,3,4$ is,
rond af naar beneden (behouden); als het $5,6,7,8,9$ is, rond af naar
boven.
\end{definition}

\begin{example}
$7.38$ afgerond op de eenheid is $7$ (volgend cijfer $3$: behouden);
afgerond op de tiende is het $7.4$ (volgend cijfer $8$: naar boven).
De prijs $4.996$ afgerond op de honderdste is $5.00$ --- \omterm{def:g6:decimals:rounding}{afronden} kan
elk cijfer veranderen!
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g6:decimals:1}
Schrijf als \omterm{def:g6:decimals:places}{decimaal getal}: $5 + \dfrac{3}{10} + \dfrac{7}{100}$;
\quad $\dfrac{9}{10}$; \quad $12 + \dfrac{4}{1000}$;
\quad ``acht eenheden en vijf honderdsten''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:2}
In $86.354$: wat is het cijfer van de tienden? Van de honderdsten?
Wat telt het cijfer $8$? Schrijf dit getal zoals in
\cref{def:g6:decimals:places}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:3}
Kopieer en vul in met $<$, $>$ of $=$:
\[
5.3 \;?\; 5.29, \qquad
0.7 \;?\; 0.70, \qquad
2.09 \;?\; 2.9, \qquad
14.5 \;?\; 14.49 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:4}
Orden van klein naar groot:
$4.2$; \ $4.05$; \ $4.51$; \ $4.15$; \ $4.5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:5}
Welke decimale getallen horen bij de punten $A$, $B$, $C$ op een
getallenlijn met tienden, als $A$ $3$ streepjes na $6$ ligt, $B$
$7$ streepjes na $6$, en $C$ $2$ streepjes na $7$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:6}
Bereken in kolommen: $45.8 + 7.65$; \quad $23.4 - 8.72$; \quad
$5.6 \times 2.4$ (tel de decimale cijfers!).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:7}
Bereken zonder schriftelijk rekenwerk:
\[
6.42 \times 10, \qquad
0.35 \times 1000, \qquad
78.1 \div 100, \qquad
5 \div 1000 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:8}
Reken om: $2.4$ m naar cm; \quad $370$ g naar kg; \quad $0.85$ km naar
m; \quad $56$ mm naar m.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:9}
Rond $23.867$ af: op de eenheid; op de tiende; op de honderdste. Rond
$9.97$ af op de tiende.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:decimals:10}
Een stokbrood kost $1.15$. Lena koopt drie stokbroden en betaalt met
een biljet van $5$. Schrijf de twee benodigde berekeningen, en geef
haar wisselgeld.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:decimals:11}
Vind een \omterm{def:g6:decimals:places}{decimaal getal} strikt tussen $7.4$ en $7.5$; daarna tussen
$3.99$ en $4$. Hoeveel zulke getallen zijn er in elk geval?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:decimals:12}
Gebruik elk van de cijfers $2$, $5$, $8$ precies één keer en één
\omterm{def:g4:decimals:point}{decimale punt}, en schrijf het grootst mogelijke getal, daarna het
kleinst mogelijke. (Getallen zoals $.58$ zijn niet toegelaten: het
gehele deel moet minstens één cijfer bevatten.)
\end{exercise}
