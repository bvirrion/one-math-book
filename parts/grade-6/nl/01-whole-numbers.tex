\chapter{Gehele getallen}\label{ch:g6:wholes}

Alles in de wiskunde begint met tellen. Dit hoofdstuk herhaalt hoe
gehele getallen worden geschreven, vergeleken en gecombineerd --- en
bekijkt de deling van dichtbij, de bewerking die zowel een \omterm{def:g3:division:remainder}{quotiënt}
\emph{als} een \omterm{def:g3:division:remainder}{rest} geeft.

\section{Schrijven en vergelijken van gehele getallen}

\begin{definition}[Plaatswaarde]\label{def:g6:wholes:place}
Onze manier om getallen te schrijven gebruikt tien cijfers ($0$ tot $9$),
en de \emph{waarde van een cijfer hangt af van zijn plaats}\index{plaatswaarde}:
in $5\,208$ telt het cijfer $5$ de duizendtallen, de $2$ de
honderdtallen, de $0$ de tientallen (er zijn er geen) en de $8$ de
eenheden:
\[
5\,208 = 5 \times 1000 + 2 \times 100 + 0 \times 10 + 8 .
\]
\end{definition}

\begin{method}[Twee gehele getallen vergelijken]\label{met:g6:wholes:compare}
\begin{enumerate}
  \item Het getal met meer cijfers is het grootste
        ($1\,203 > 989$).
  \item Als ze evenveel cijfers hebben, vergelijk cijfer voor cijfer
        van \emph{links}; het eerste \omterm{ex:g1:subtraction:difference}{verschil} beslist:
        $6\,742 > 6\,715$ want op de plaats van de tientallen is
        $4 > 1$.
\end{enumerate}
De symbolen zijn $<$ (``kleiner dan'') en $>$ (``groter dan''); de
kleine kant van het symbool wijst naar het kleine getal.
\end{method}

\begin{example}\label{ex:g6:wholes:order}
Orden van klein naar groot: $908$; $89$; $1\,001$; $980$.
Eerst op aantal cijfers: $89$ (twee cijfers) komt eerst, $1\,001$
(vier cijfers) als laatste. Tussen $908$ en $980$: hetzelfde
honderdtalcijfer $9$, daarna tientallen $0 < 8$, dus $908 < 980$.
Eindantwoord:
\[
89 < 908 < 980 < 1\,001 .
\]
\end{example}

\section{Optellen, aftrekken, vermenigvuldigen}

\begin{example}[Kolomberekeningen]\label{ex:g6:wholes:columns}
Om op te tellen of af te trekken, zet je de eenheden onder de
eenheden, de tientallen onder de tientallen, en werk je van rechts naar
links, met onthouden indien nodig:
\[
\begin{array}{r}
4\,5\,7 \\
+\ 2\,8\,6 \\
\hline
7\,4\,3
\end{array}
\qquad\qquad
\begin{array}{r}
6\,0\,3 \\
-\ 1\,4\,7 \\
\hline
4\,5\,6
\end{array}
\]
Bij de optelling: $7 + 6 = 13$, schrijf $3$, onthoud $1$; dan
$5 + 8 + 1 = 14$, schrijf $4$, onthoud $1$; dan $4 + 2 + 1 = 7$.
Controleer de aftrekking door terug op te tellen: $456 + 147 = 603$.
\end{example}

\begin{remark}[Woordenlijst]
Het resultaat van een optelling is een \emph{som}\index{som}; van een
aftrekking een \emph{verschil}\index{verschil}; van een
vermenigvuldiging een \emph{product}\index{product}; van een deling
een \emph{quotiënt}\index{quoti\"ent}. De gecombineerde getallen heten
de \emph{termen} (bij $+$, $-$) of \emph{factoren} (bij $\times$).
\end{remark}

\begin{example}[Slim rekenen]\label{ex:g6:wholes:smart}
Termen of factoren herschikken bespaart vaak werk:
\[
17 + 58 + 3 = (17 + 3) + 58 = 20 + 58 = 78 ,
\]
\[
25 \times 17 \times 4 = (25 \times 4) \times 17 = 100 \times 17 = 1700 .
\]
\end{example}

\section{Deling}

\begin{theorem}[Euclidische deling]\label{thm:g6:wholes:division}
Gegeven twee gehele getallen $a$ (het \emph{deeltal}) en $b \neq 0$
(de \emph{\omterm{def:g5:division:multiple}{deler}}), is er precies één manier om te schrijven
\[
a = b \times q + r
\qquad\text{met}\qquad
0 \leq r < b .
\]
Het getal $q$ is het \emph{quotiënt}\index{quoti\"ent} en $r$ de
\emph{rest}\index{rest} van de deling van $a$ door $b$.
\end{theorem}
\admitted

\begin{example}\label{ex:g6:wholes:division}
Deel $137$ door $4$, stap voor stap.
\begin{enumerate}
  \item Hoe vaak past $4$ in $13$? Drie keer
        ($4 \times 3 = 12$), \omterm{def:g3:division:remainder}{rest} $13 - 12 = 1$.
  \item Laat de $7$ zakken: hoe vaak past $4$ in $17$?
        Vier keer ($4 \times 4 = 16$), \omterm{def:g3:division:remainder}{rest} $1$.
\end{enumerate}
Dus $q = 34$ en $r = 1$:
\[
137 = 4 \times 34 + 1, \qquad\text{en inderdaad } 0 \leq 1 < 4 .
\]
Controle: $4 \times 34 = 136$, en $136 + 1 = 137$.
\end{example}

\begin{definition}[Deelbaar]\label{def:g6:wholes:divisible}
Wanneer de \omterm{def:g3:division:remainder}{rest} $0$ is, zeggen we dat $a$ \emph{deelbaar}\index{deelbaar}
is door $b$: bijvoorbeeld is $135$ deelbaar door $5$ want
$135 = 5 \times 27 + 0$.
\end{definition}

\begin{method}[Woordproblemen met deling]\label{met:g6:wholes:problems}
Na het berekenen van $a = bq + r$, ga altijd terug naar de vraag:
\begin{enumerate}
  \item ``Hoeveel volle dozen/teams/pagina's?'' --- het antwoord is het
        \omterm{def:g3:division:remainder}{quotiënt} $q$;
  \item ``Hoeveel blijven er over?'' --- het antwoord is de \omterm{def:g3:division:remainder}{rest}
        $r$;
  \item ``Hoeveel dozen zijn er nodig voor iedereen?'' --- als $r > 0$,
        \emph{één meer} dan het \omterm{def:g3:division:remainder}{quotiënt}: $q + 1$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g6:wholes:bus}
$137$ leerlingen gaan op uitstap in bussen van $40$ plaatsen. Deling:
$137 = 40 \times 3 + 17$. Drie bussen vervoeren $120$ leerlingen, en
$17$ leerlingen blijven over --- ook zij hebben een bus nodig! Dus
zijn er $4$ bussen nodig, hoewel het \omterm{def:g3:division:remainder}{quotiënt} $3$ is.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g6:wholes:1}
Schrijf in cijfers: ``drieduizend zevenenveertig''; ``twintigduizend
vijfhonderd''; ``honderdvier''. Schrijf daarna $60\,013$ in woorden.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:2}
In het getal $47\,209$: wat is het cijfer van de honderdtallen? Het
cijfer van de eenheden? Wat telt het cijfer $4$? Schrijf het getal als
een \omterm{def:g1:addition:def}{som} van \omterm{def:g5:division:multiple}{veelvouden} van $1$, $10$, $100$, \dots\ zoals in
\cref{def:g6:wholes:place}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:3}
Kopieer en vul in met $<$ of $>$:
\[
507 \;?\; 570, \qquad
1\,099 \;?\; 989, \qquad
23\,456 \;?\; 23\,465, \qquad
9\,999 \;?\; 10\,000 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:4}
Orden van klein naar groot: $2\,043$; $978$; $2\,304$; $2\,034$;
$1\,001$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:5}
Bereken in kolommen: $348 + 576$; \quad $805 - 268$; \quad
$307 \times 26$. Controleer de aftrekking met een optelling.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:6}
Reken slim, door termen of factoren te groeperen:
\[
38 + 45 + 12 + 5, \qquad
2 \times 83 \times 50, \qquad
4 \times 78 \times 25 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:7}
Geef voor elke deling het \omterm{def:g3:division:remainder}{quotiënt} en de \omterm{def:g3:division:remainder}{rest}, en schrijf de
gelijkheid $a = b \times q + r$:
\[
95 \div 7, \qquad
230 \div 6, \qquad
504 \div 8 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:wholes:8}
Zonder te delen, leg uit waarom de gelijkheid $173 = 8 \times 20 + 13$
\emph{niet} de Euclidische deling van $173$ door $8$ is, en vind het
juiste \omterm{def:g3:division:remainder}{quotiënt} en de juiste \omterm{def:g3:division:remainder}{rest}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:wholes:9}
Eieren worden verpakt in dozen van $12$.
\begin{enumerate}
  \item Een boerderij heeft $2\,000$ eieren. Hoeveel volle dozen kan
        ze vullen, en hoeveel eieren blijven over?
  \item Een bakkerij heeft $150$ eieren nodig. Hoeveel dozen moet ze
        kopen?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:wholes:10}
Vandaag is het dinsdag. Welke weekdag is het over $100$ dagen?
(Weken hebben $7$ dagen: deel en denk na over de \omterm{def:g3:division:remainder}{rest}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:wholes:11}
Bij een deling door $9$ is het \omterm{def:g3:division:remainder}{quotiënt} $56$ en de \omterm{def:g3:division:remainder}{rest} zo groot
mogelijk. Wat is het deeltal?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:wholes:12}
Ik ben een getal van drie cijfers. Mijn honderdtalcijfer is het dubbele
van mijn eenheidcijfer, mijn tientalcijfer is $0$, en de \omterm{def:g1:addition:def}{som} van mijn
cijfers is $9$. Wie ben ik?
(Redeneer stap voor stap, en controleer daarna.)
\end{exercise}
