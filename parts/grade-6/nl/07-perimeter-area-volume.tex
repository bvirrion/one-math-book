\chapter{Omtrek, oppervlakte, volume}\label{ch:g6:measure}

Hoe lang is het hek, hoe groot is het veld, hoeveel water past in de
tank? Drie verschillende vragen, drie verschillende grootheden ---
omtrek, oppervlakte, volume --- elk met zijn eigen eenheden. Ze
verwarren is de meest voorkomende fout in de meetkunde; dit hoofdstuk
zet ze definitief recht.

\section{Lengten en omtrek}

\begin{definition}[Omtrek]\label{def:g6:measure:perimeter}
De \emph{omtrek}\index{omtrek} van een figuur is de totale lengte van
haar rand. Hij wordt gemeten in lengte-eenheden: millimeters (mm),
centimeters (cm), meters (m), kilometers (km), met
\[
1 \text{ km} = 1000 \text{ m}, \qquad
1 \text{ m} = 100 \text{ cm}, \qquad
1 \text{ cm} = 10 \text{ mm}.
\]
\end{definition}

\begin{example}\label{ex:g6:measure:perimeter}
Een rechthoek met lengte $L$ en breedte $w$ heeft omtrek
\[
P = L + w + L + w = 2 \times (L + w).
\]
Voor een rechthoek van $7$ cm bij $4$ cm: $P = 2 \times (7 + 4) =
2 \times 11 = 22$ cm. Een vierkant met zijde $c$ heeft omtrek $4c$.
\end{example}

\begin{proposition}[Omtrek van een cirkel]\label{prop:g6:measure:circle}
De omtrek (of \emph{cirkelomtrek}) van een cirkel met straal $r$ is
\[
P = 2 \pi r ,
\]
waarbij $\pi \approx 3.14$ hetzelfde getal is voor elke cirkel.
\end{proposition}
\admitted

\begin{example}
Een ronde vijver heeft straal $5$ m. Zijn rand meet
$2 \times \pi \times 5 = 10\pi \approx 31.4$ m. Houd de exacte
waarde $10\pi$ zo lang mogelijk; rond pas op het einde af.
\end{example}

\section{Oppervlakten}

\begin{definition}[Oppervlakte]\label{def:g6:measure:area}
De \emph{oppervlakte}\index{oppervlakte} van een figuur meet de
oppervlakte die ze bedekt: hoeveel eenheidsvierkanten erin passen.
Eenheden: cm$^2$ (een vierkant met zijde $1$ cm), m$^2$, km$^2$,
\dots\ Let op:
\[
1 \text{ m}^2 = 100 \times 100 \text{ cm}^2 = 10\,000 \text{ cm}^2
\]
--- één vierkante meter is een vierkant van $100$ cm bij $100$ cm,
dus elke stap van de eenhedenladder is $100$ waard, niet $10$.
\end{definition}

\begin{proposition}[Basisformules voor oppervlakte]\label{prop:g6:measure:formulas}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{l|l}
figuur & oppervlakte \\
\hline
rechthoek ($L \times w$) & $A = L \times w$ \\
vierkant (zijde $c$) & $A = c^2$ \\[4pt]
rechthoekige driehoek (benen $a$, $b$) & $A = \dfrac{a \times b}{2}$ \\[4pt]
cirkelschijf (straal $r$) & $A = \pi r^2$ \\
\end{tabular}
\end{center}
\end{proposition}

\begin{proof}[Bewijs voor de rechthoekige driehoek]
Twee kopieën van een rechthoekige driehoek met benen $a$ en $b$,
aan elkaar geplakt langs de schuine zijde, vormen een rechthoek
$a \times b$. Dus de oppervlakte van de driehoek is de helft van die
van de rechthoek: $\frac{ab}{2}$. (De rechthoekformule is het
tellen van eenheidsvierkanten hierboven; de schijfformule is
toegegeven, zie \cref{ch:g7:areas}.)
\end{proof}

\begin{example}[Samengestelde figuren]\label{ex:g6:measure:composite}
Een L-vormige kamer is een rechthoek van $6$ m $\times$ $4$ m met een
vierkante hoek van $2$ m $\times$ $2$ m weggenomen. Haar oppervlakte,
stap voor stap:
\begin{enumerate}
  \item volle rechthoek: $6 \times 4 = 24$ m$^2$;
  \item weggenomen vierkant: $2 \times 2 = 4$ m$^2$;
  \item resterende oppervlakte: $24 - 4 = 20$ m$^2$.
\end{enumerate}
Haar \emph{omtrek} is niet $20$ iets: rond de L lopen meet de rand
nog altijd $6 + 4 + 6 + 4 = 20$ m --- de twee sneden van de hoek
vervangen twee gelijke stukken muur. Hetzelfde getal bij toeval, maar
vierkante meters voor de ene, meters voor de andere!
\end{example}

\begin{remark}[Zelfde omtrek, verschillende oppervlakten]
Twee figuren kunnen dezelfde omtrek en heel verschillende
oppervlakten hebben: een vierkant $5 \times 5$ en een rechthoek
$9 \times 1$ hebben beide omtrek $20$, maar oppervlakten $25$ en $9$.
Omtrek en oppervlakte zijn echt onafhankelijke grootheden.
\end{remark}

\section{Volumes}

\begin{definition}[Volume]\label{def:g6:measure:volume}
Het \emph{volume}\index{volume} van een lichaam meet de ruimte die het
vult: hoeveel eenheidskubussen erin passen. Eenheden: cm$^3$, m$^3$,
\dots\ met $1$ m$^3 = 1\,000\,000$ cm$^3$ (elke stap van de ladder is
$1000$ waard). Voor vloeistoffen gebruikt men ook de \emph{liter}:
\[
1 \text{ L} = 1 \text{ dm}^3 = 1000 \text{ cm}^3,
\qquad
1 \text{ m}^3 = 1000 \text{ L}.
\]
\end{definition}

\begin{proposition}[Volume van een balk]\label{prop:g6:measure:box}
Een rechthoekige balk (een \emph{rechthoekig prisma}) met lengte $L$,
breedte $w$ en hoogte $h$ heeft volume
\[
V = L \times w \times h .
\]
\end{proposition}

\begin{proof}
De onderste laag bevat $L \times w$ eenheidskubussen
(\cref{def:g6:measure:area}-plaatje, met kubussen), en er zijn $h$
zulke lagen.
\end{proof}

\begin{example}\label{ex:g6:measure:aquarium}
Een aquarium meet $60$ cm bij $30$ cm bij $40$ cm (hoogte). Volume:
\[
V = 60 \times 30 \times 40 = 72\,000 \text{ cm}^3 = 72 \text{ dm}^3
= 72 \text{ L}.
\]
Stap voor stap voor de omrekening: $1000$ cm$^3$ maken één liter, en
$72\,000 \div 1000 = 72$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g6:measure:1}
Reken om: $3.5$ m naar cm; $420$ mm naar cm; $0.8$ km naar m;
$25\,000$ m naar km.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:measure:2}
Bereken de omtrek van: een rechthoek $8$ cm $\times$ $3.5$ cm; een
vierkant met zijde $6.2$ cm; een driehoek met zijden $5$ cm, $7$ cm en
$9$ cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:measure:3}
Een ronde renbaan heeft straal $50$ m. Hoe lang is één ronde
(exacte waarde met $\pi$, daarna afgerond op de meter)? Hoeveel
rondes maken minstens $2$ km?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:measure:4}
Bereken de oppervlakte van: een rechthoek $9$ cm $\times$ $4$ cm; een
vierkant met zijde $7$ m; een rechthoekige driehoek met benen $6$ cm
en $10$ cm; een cirkelschijf met straal $3$ cm (exacte waarde, daarna
afgerond op de cm$^2$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:measure:5}
Reken om: $3$ m$^2$ naar cm$^2$; $45\,000$ cm$^2$ naar m$^2$; $2.5$ L
naar cm$^3$; $4\,500$ L naar m$^3$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:measure:6}
Een rechthoekig veld is $120$ m lang en $85$ m breed. Hoeveel meter
hek is nodig om het af te sluiten? Wat is zijn oppervlakte?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:measure:7}
Bereken het volume van een balk $5$ cm $\times$ $4$ cm $\times$
$10$ cm, en van een kubus met ribbe $3$ cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:measure:8}
Teken twee verschillende rechthoeken met omtrek $16$ cm, en bereken
hun oppervlakten. Welke van je rechthoeken heeft de grotere
oppervlakte?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:measure:9}
Een T-vormige figuur bestaat uit een horizontale rechthoek $10 \times 2$
bovenop een verticale $2 \times 6$ (maten in cm). Bereken zijn
oppervlakte, daarna zijn omtrek (loop zorgvuldig rond de rand en tel
elke rand op).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:measure:10}
Een zwembad is een rechthoekige bak van $25$ m lang, $10$ m breed en
$2$ m diep.
\begin{enumerate}
  \item Hoeveel kubieke meters water houdt het als het vol is?
  \item Hoeveel liters is dat?
  \item Het bad wordt gevuld met $50\,000$ L per uur. Hoe lang duurt
        het vullen?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:measure:11}
Een tuin is een vierkant met zijde $20$ m met daarin een ronde vijver
van straal $3$ m. Welke oppervlakte gras is er (exacte waarde, daarna
afgerond op de m$^2$)?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:measure:12}
Een chocoladereep meet $15$ cm $\times$ $6$ cm $\times$ $1$ cm.
De fabrikant verdubbelt \emph{alle drie} de afmetingen.
\begin{enumerate}
  \item Met hoeveel wordt het volume vermenigvuldigd? Controleer door
        beide volumes te berekenen.
  \item De prijs wordt met $4$ vermenigvuldigd. Is de grote reep een
        betere deal dan de kleine?
\end{enumerate}
\end{exercise}
