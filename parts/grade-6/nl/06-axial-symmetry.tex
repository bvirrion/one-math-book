\chapter{Spiegeling in een as}\label{ch:g6:symmetry}

Vouw een vel papier langs een lijn: elk punt landt op een ander punt,
zijn \omterm{def:g6:symmetry:def}{spiegelbeeld}. Dit vouwen is \emph{\omterm{def:g6:symmetry:def}{spiegeling in een as}} (of
axiale symmetrie), de eerste meetkundige transformatie van het boek ---
de tweede, centrale spiegeling, komt in \cref{ch:g7:central}.

\section{Spiegelbeelden}

\begin{definition}[Spiegelbeeldpunt]\label{def:g6:symmetry:def}
Laat $d$ een \omterm{def:g6:lines:objects}{rechte} zijn. Het \emph{spiegelbeeld}\index{spiegeling in
een as} van een punt $M$ ten opzichte van $d$ is het punt $M'$ zo dat
$d$ de \omterm{def:g6:symmetry:bisector}{middelloodlijn} is van $[MM']$: het \omterm{def:g6:lines:objects}{lijnstuk} $[MM']$ staat
\omterm{def:g6:lines:perp}{loodrecht} op $d$, en $d$ snijdt het in zijn midden. Een punt \emph{op}
$d$ is zijn eigen spiegelbeeld.
\end{definition}

\begin{method}[Een spiegelbeeldpunt construeren]\label{met:g6:symmetry:construct}
Om het \omterm{def:g6:symmetry:def}{spiegelbeeld} $M'$ van $M$ ten opzichte van de \omterm{def:g6:lines:objects}{rechte} $d$ te
construeren:
\begin{enumerate}
  \item teken de loodlijn op $d$ door $M$ (geodriehoek); noem $H$ het
        punt waar die $d$ snijdt;
  \item meet $MH$ met passer of liniaal;
  \item plaats $M'$ op dezelfde loodlijn, aan de andere kant van $d$,
        met $HM' = MH$.
\end{enumerate}
Op ruitjespapier gaat het sneller: tel de hokjes van $M$ tot de as, en
tel hetzelfde aantal aan de andere kant.
\end{method}

\begin{example}[Op een rooster]\label{ex:g6:symmetry:grid}
Als de as een verticale roosterlijn is en $M$ $3$ hokjes links ervan
ligt, dan ligt $M'$ $3$ hokjes rechts, op dezelfde rij. Voor een
diagonale as werkt het tellen langs diagonalen --- altijd \omterm{def:g6:lines:perp}{loodrecht} op
de as.
\end{example}

\begin{proposition}[Wat een spiegeling behoudt]\label{prop:g6:symmetry:preserved}
Het \omterm{def:g6:symmetry:def}{spiegelbeeld} van een figuur ten opzichte van een \omterm{def:g6:lines:objects}{rechte} is een
figuur van \emph{dezelfde vorm en dezelfde grootte}: lengten, hoeken,
omtrekken en \omterm{def:g4:measure:area}{oppervlakten} blijven allemaal behouden. Het \omterm{def:g6:symmetry:def}{spiegelbeeld}
van een \omterm{def:g6:lines:objects}{lijnstuk} is een \omterm{def:g6:lines:objects}{lijnstuk}, van een cirkel een cirkel met
dezelfde \omterm{def:g3:shapes:shapes}{straal}, van een \omterm{def:g6:lines:objects}{rechte} een \omterm{def:g6:lines:objects}{rechte}.
\end{proposition}
\admitted

\begin{remark}
Eén ding blijft \emph{niet} behouden: de oriëntatie. Een gespiegelde
``b'' wordt een ``d'' --- spiegelSchrift. Als het calqueren van je
figuur en omdraaien van het calqueerpapier het beeld reproduceert, is
de spiegeling correct.
\end{remark}

\section{Symmetrieassen van een figuur}

\begin{definition}[Symmetrieas]\label{def:g6:symmetry:axis}
Een \omterm{def:g6:lines:objects}{rechte} $d$ is een \emph{symmetrieas}\index{symmetrieas} van een
figuur wanneer de spiegeling ten opzichte van $d$ de figuur exact op
zichzelf stuurt --- vouwen langs $d$ laat de twee \omterm{def:g3:division:half}{helften} samenvallen.
\end{definition}

\begin{example}\label{ex:g6:symmetry:shapes}
Symmetrieassen tellen:
\begin{itemize}
  \item \omterm{def:g6:shapes:triangles}{gelijkbenige driehoek}: $1$ (door het tophoekpunt en het midden
        van de basis); \omterm{def:g6:shapes:triangles}{gelijkzijdige driehoek}: $3$;
  \item rechthoek: $2$ (de twee ``middenlijnen'', \emph{niet} de
        diagonalen); \omterm{def:g6:shapes:quadrilaterals}{ruit}: $2$ (zijn diagonalen!); vierkant: $4$;
  \item cirkel: elke \omterm{def:g6:lines:objects}{rechte} door het middelpunt --- oneindig veel.
\end{itemize}
\end{example}

\begin{example}[Symmetrie bewijst gelijkheden]\label{ex:g6:symmetry:proves}
Waarom zijn de twee basishoeken van een \omterm{def:g6:shapes:triangles}{gelijkbenige driehoek} gelijk?
Vouw de driehoek langs zijn \omterm{def:g6:symmetry:axis}{symmetrieas}: de linkerhelft landt exact op
de rechterhelft, dus de linker basishoek landt op de rechter basishoek
--- ze moeten dezelfde maat hebben
(\cref{prop:g6:symmetry:preserved}: spiegelingen behouden hoeken).
\end{example}

\section{Middelloodlijn}

\begin{definition}[Middelloodlijn]\label{def:g6:symmetry:bisector}
De \emph{middelloodlijn}\index{middelloodlijn} van een \omterm{def:g6:lines:objects}{lijnstuk} $[AB]$
is de \omterm{def:g6:lines:objects}{rechte} \omterm{def:g6:lines:perp}{loodrecht} op $[AB]$ door zijn midden --- het is de
\omterm{def:g6:symmetry:axis}{symmetrieas} die $A$ en $B$ verwisselt.
\end{definition}

\begin{proposition}[Gelijkheid van afstanden]\label{prop:g6:symmetry:bisector}
Een punt $M$ ligt op de \omterm{def:g6:symmetry:bisector}{middelloodlijn} van $[AB]$ precies wanneer het
op gelijke afstand van $A$ en van $B$ ligt: $MA = MB$.
\end{proposition}
\admitted

\begin{method}[Passerconstructie]\label{met:g6:symmetry:bisector}
Om de \omterm{def:g6:symmetry:bisector}{middelloodlijn} van $[AB]$ te tekenen zonder te meten:
\begin{enumerate}
  \item open de passer tot meer dan de \omterm{def:g3:division:half}{helft} van $AB$;
  \item teken twee bogen met middelpunt $A$, boven en onder het
        \omterm{def:g6:lines:objects}{lijnstuk};
  \item met \emph{dezelfde} opening, teken twee bogen met middelpunt
        $B$;
  \item verbind de twee snijpunten: deze \omterm{def:g6:lines:objects}{rechte} is de \omterm{def:g6:symmetry:bisector}{middelloodlijn}
        (elk snijpunt ligt even ver van $A$ als van $B$).
\end{enumerate}
\end{method}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:1}
Kopieer op ruitjespapier: een verticale as $d$, een punt $A$ twee
hokjes links ervan, een punt $B$ vijf hokjes links en één hokje hoger.
Construeer hun \omterm{def:g6:symmetry:def}{spiegelbeelden} $A'$ en $B'$ ten opzichte van $d$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:2}
Teken een \omterm{def:g6:lines:objects}{rechte} $d$ en een punt $M$ op $3$ cm van $d$ (niet op
ruitjespapier). Construeer het \omterm{def:g6:symmetry:def}{spiegelbeeld} $M'$ van $M$ ten opzichte
van $d$ met geodriehoek en liniaal, volgens
\cref{met:g6:symmetry:construct}. Hoe ver liggen $M$ en $M'$ uit
elkaar?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:3}
Welke hoofdletters van het alfabet, in hun eenvoudigste vorm
geschreven, hebben een verticale \omterm{def:g6:symmetry:axis}{symmetrieas} (zoals A)? Een
horizontale (zoals B)? Beide?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:4}
Hoeveel symmetrieassen heeft: een \omterm{def:g6:shapes:triangles}{gelijkzijdige driehoek}? een
rechthoek (die geen vierkant is)? een \omterm{def:g6:shapes:quadrilaterals}{ruit} (geen vierkant)? een
vierkant? een cirkel? Teken elke figuur met zijn assen.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:5}
Waar of onwaar? ``De diagonalen van een rechthoek zijn symmetrieassen
van de rechthoek.'' Leg uit met een vouwargument of een tekening.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:6}
Een driehoek $ABC$ wordt gespiegeld ten opzichte van een \omterm{def:g6:lines:objects}{rechte} $d$
naar $A'B'C'$. Gegeven $AB = 4$ cm, $\widehat{BAC} = 70^\circ$ en de
\omterm{def:g4:measure:perimeter}{omtrek} van $ABC$ is $12$ cm: geef, zonder enige constructie, $A'B'$,
$\widehat{B'A'C'}$ en de \omterm{def:g4:measure:perimeter}{omtrek} van $A'B'C'$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:symmetry:7}
Teken een \omterm{def:g6:lines:objects}{lijnstuk} $[AB]$ van $7$ cm en construeer zijn \omterm{def:g6:symmetry:bisector}{middelloodlijn}
met de passer (\cref{met:g6:symmetry:bisector}). Kies een willekeurig
punt $M$ erop en controleer met de liniaal dat $MA = MB$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:symmetry:8}
Twee bomen staan in de punten $A$ en $B$ van een (vlakke) tuin. Waar
kan men een fontein planten zodat die op gelijke afstand van beide
bomen staat? Beschrijf \emph{alle} mogelijke plekken.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:symmetry:9}
Teken op ruitjespapier de as $d$ (een diagonaal van het rooster onder
$45^\circ$) en een kleine L-vormige figuur aan één kant. Construeer
zijn \omterm{def:g6:symmetry:def}{spiegelbeeld} ten opzichte van $d$. (Spiegel elk \omterm{def:g5:solids:def}{hoekpunt}: voor
een as van $45^\circ$ landt een punt $2$ hokjes rechts van de as $2$
hokjes erboven.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:symmetry:10}
Teken een cirkel met middelpunt $O$, een \omterm{def:g6:lines:objects}{rechte} $d$ door $O$, en een
punt $M$ op de cirkel. Waar is het \omterm{def:g6:symmetry:def}{spiegelbeeld} van $M$ ten opzichte
van $d$? Leg uit waarom de spiegeling de cirkel op zichzelf stuurt.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:symmetry:11}
Gebruik de eigenschap van gelijke afstanden
(\cref{prop:g6:symmetry:bisector}) om uit te leggen waarom de twee
snijpunten van de bogen in \cref{met:g6:symmetry:bisector} echt op de
\omterm{def:g6:symmetry:bisector}{middelloodlijn} van $[AB]$ liggen.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:symmetry:12}
Een biljartbal in punt $A$ moet van een \omterm{def:g6:lines:objects}{rechte} muur $d$ stuiten en
punt $B$ bereiken (beide aan dezelfde kant van de muur). Spiegel $B$
ten opzichte van $d$ naar $B'$, en teken het \omterm{def:g6:lines:objects}{lijnstuk} $[AB']$. Leg uit
waarom het beste stuitpunt is waar $[AB']$ de muur snijdt. (Idee: het
pad $A \to P \to B$ heeft dezelfde lengte als $A \to P \to B'$.)
\end{exercise}
