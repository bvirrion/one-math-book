\chapter{Driehoeken en vierhoeken}\label{ch:g6:shapes}

Driehoeken en vierhoeken zijn de basisfiguren van de meetkunde. Het
doel van dit hoofdstuk is hun speciale families te kennen ---
gelijkbenig, gelijkzijdig, rechthoekig; rechthoeken, ruiten, vierkanten
--- via hun \emph{definiërende eigenschappen}, en ze precies te
construeren met liniaal en passer.

\section{Driehoeken}

\begin{definition}[Speciale driehoeken]\label{def:g6:shapes:triangles}
Een driehoek $ABC$ heeft drie hoekpunten en drie zijden. Hij is:
\begin{itemize}
  \item \emph{gelijkbenig}\index{gelijkbenige driehoek} in $A$ wanneer
        $AB = AC$ (twee gelijke zijden);
  \item \emph{gelijkzijdig}\index{gelijkzijdige driehoek} wanneer
        $AB = BC = CA$ (drie gelijke zijden);
  \item \emph{rechthoekig}\index{rechthoekige driehoek} in $A$ wanneer
        de hoek $\widehat{BAC}$ een rechte hoek is. De zijde $[BC]$
        tegenover de rechte hoek heet de \emph{schuine zijde}
        (hypotenusa).
\end{itemize}
\end{definition}

\begin{method}[Een driehoek construeren uit zijn drie zijden]\label{met:g6:shapes:construct}
Om $ABC$ te construeren met $AB = 5$ cm, $AC = 4$ cm, $BC = 3$ cm:
\begin{enumerate}
  \item teken het lijnstuk $[AB]$ van $5$ cm met de liniaal;
  \item teken een boog van de cirkel met middelpunt $A$ en straal
        $4$ cm (alle punten op $4$ cm van $A$);
  \item teken een boog van de cirkel met middelpunt $B$ en straal
        $3$ cm;
  \item het punt $C$ is waar de twee bogen snijden; verbind het met
        $A$ en $B$.
\end{enumerate}
\end{method}

\section{Vierhoeken}

\begin{definition}[Speciale vierhoeken]\label{def:g6:shapes:quadrilaterals}
Een vierhoek heeft vier hoekpunten en vier zijden. Het is een:
\begin{itemize}
  \item \emph{rechthoek}\index{rechthoek} wanneer zijn vier hoeken
        rechte hoeken zijn;
  \item \emph{ruit}\index{ruit} wanneer zijn vier zijden dezelfde
        lengte hebben;
  \item \emph{vierkant}\index{vierkant} wanneer beide: vier rechte
        hoeken \emph{en} vier gelijke zijden.
\end{itemize}
\end{definition}

\begin{proposition}[Eigenschappen van de diagonalen]\label{prop:g6:shapes:diagonals}
In deze vierhoeken snijden de twee diagonalen elkaar in hun gemeenschappelijk
midden, en bovendien:
\begin{itemize}
  \item in een rechthoek hebben de diagonalen \emph{dezelfde lengte};
  \item in een ruit staan de diagonalen \emph{loodrecht} op elkaar;
  \item in een vierkant allebei tegelijk.
\end{itemize}
\end{proposition}
\admitted

\begin{remark}
Een vierkant is automatisch een rechthoek (het heeft vier rechte
hoeken) en automatisch een ruit (vier gelijke zijden). Dus ``rechthoek''
betekent niet ``geen vierkant''! In de wiskunde hoort de specialere
vorm bij de algemenere familie.
\end{remark}

\begin{example}[Een vorm herkennen aan zijn eigenschappen]\label{ex:g6:shapes:recognize}
Een vierhoek heeft vier zijden van lengte $4$ cm en één rechte hoek.
Wat is het? Vier gelijke zijden maken het een ruit. In een ruit zijn
overstaande hoeken gelijk en tellen de vier hoeken op tot $360^\circ$;
één rechte hoek dwingt dan alle vier recht te zijn (dit wordt bewezen
met parallellogrammen in \cref{ch:g7:central}). Dus de vorm is een
vierkant.
\end{example}

\begin{method}[Constructieprogramma's]\label{met:g6:shapes:program}
Een \emph{constructieprogramma} is een lijst instructies die iedereen
kan volgen om een figuur exact te reproduceren. Schrijf één instructie
per regel, noem elk punt wanneer het ontstaat, en gebruik alleen:
``teken het lijnstuk / de rechte / de cirkel\dots'', ``plaats het punt
waar \dots\ snijden''. Test je programma door het letterlijk te
volgen.
\end{method}

\begin{example}\label{ex:g6:shapes:program}
Programma voor een vierkant $ABCD$ met zijde $4$ cm:
\begin{enumerate}
  \item Teken een lijnstuk $[AB]$ met $AB = 4$ cm.
  \item Teken de loodlijn op $(AB)$ in $A$; plaats daarop $D$ op
        $4$ cm van $A$ (kies één kant van de lijn en blijf daarbij).
  \item Teken de loodlijn op $(AB)$ in $B$; plaats daarop $C$ op
        $4$ cm van $B$, aan dezelfde kant als $D$.
  \item Verbind $C$ en $D$.
\end{enumerate}
Controleer met de liniaal: $DC$ moet ook $4$ cm meten.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g6:shapes:1}
Construeer een driehoek $ABC$ met $AB = 6$ cm, $AC = 5$ cm en
$BC = 4$ cm, volgens \cref{met:g6:shapes:construct}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:2}
Construeer een gelijkbenige driehoek $DEF$ met $DE = DF = 5$ cm en
$EF = 3$ cm. Welke twee hoeken van deze driehoek lijken gelijk? Meet
ze om te controleren.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:3}
Construeer een gelijkzijdige driehoek met zijde $4$ cm met alleen
liniaal en passer. Meet één van zijn hoeken.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:4}
Classificeer elke driehoek aan de hand van de gegevens (gelijkbenig,
gelijkzijdig, rechthoekig --- meerdere antwoorden mogelijk):
(a) zijden $7$, $7$, $7$; \
(b) zijden $5$, $5$, $8$; \
(c) hoeken $90^\circ$, $45^\circ$, $45^\circ$; \
(d) hoeken $60^\circ$, $60^\circ$, $60^\circ$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:5}
In een rechthoekige driehoek, welke zijde is de schuine zijde? Teken
een driehoek $MNP$ rechthoekig in $N$, en noem zijn schuine zijde.
Welk hoekpunt raakt de schuine zijde nooit?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:6}
Waar of onwaar? Motiveer elk antwoord in één zin.
\begin{enumerate}
  \item Elk vierkant is een rechthoek.
  \item Elke rechthoek is een vierkant.
  \item Elk vierkant is een ruit.
  \item Een ruit met een rechte hoek is een vierkant.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:7}
Construeer een rechthoek $ABCD$ met $AB = 5$ cm en $BC = 3$ cm. Meet
zijn twee diagonalen: wat observeer je?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:shapes:8}
Construeer een ruit met diagonalen van $6$ cm en $4$ cm. (Gebruik
\cref{prop:g6:shapes:diagonals}: teken eerst de twee diagonalen ---
loodrecht, snijdend in hun middens --- en verbind daarna de vier
uiteinden.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:shapes:9}
Schrijf een constructieprogramma voor een gelijkbenige driehoek met
basis $4$ cm en gelijke zijden van $6$ cm, in de stijl van
\cref{ex:g6:shapes:program}. Wissel met een klasgenoot: kunnen zij het
volgen zonder je figuur te zien?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:shapes:10}
Teken een cirkel met middelpunt $O$ en één van zijn middellijnen
$[AB]$. Plaats een willekeurig punt $M$ op de cirkel en teken de
driehoek $AMB$. Meet de hoek $\widehat{AMB}$. Verplaats $M$ en meet
opnieuw. Wat vermoed je? (Dit verrassende feit wordt bewezen in
\cref{ch:g8:cosine}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:shapes:11}
Een vierhoek heeft diagonalen van dezelfde lengte die elkaar in hun
gemeenschappelijk midden snijden. Zoek uit welke vorm het moet zijn,
met \cref{prop:g6:shapes:diagonals}: rechthoek, ruit of vierkant?
(Let op: er is slechts één eigenschap gegeven.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:shapes:12}
Op een $4 \times 4$ stippenrooster, hoeveel verschillende vierkanten
kun je tekenen waarvan de hoeken roosterpunten zijn --- ook de
``schuine'' vierkanten meetellend? Teken minstens één schuin vierkant
en leg uit waarom zijn zijden gelijk zijn.
\end{exercise}
