\chapter{Lijnen, cirkels en hoeken}\label{ch:g6:lines}

Meetkunde begint met een liniaal, een geodriehoek en een passer. Dit
hoofdstuk legt de woordenschat vast --- punten, lijnstukken, halfrechten,
rechten, cirkels --- introduceert de twee speciale posities van lijnen
(evenwijdig en loodrecht), en leert hoeken meten en tekenen.

\section{Punten, lijnstukken, rechten}

\begin{definition}[Basisobjecten]\label{def:g6:lines:objects}
Door twee verschillende punten $A$ en $B$ gaan:
\begin{itemize}
  \item het \emph{lijnstuk}\index{lijnstuk} $[AB]$: het deel van de
        rechte tussen $A$ en $B$ (het heeft een lengte, genoteerd
        $AB$);
  \item de \emph{halfrechte} $[AB)$: start in $A$, gaat door $B$ en
        loopt oneindig door;
  \item de \emph{rechte}\index{rechte} $(AB)$: loopt oneindig door
        aan beide kanten. Twee punten bepalen precies één rechte.
\end{itemize}
Punten op dezelfde rechte heten \emph{gealigneerd} (op één lijn).
\end{definition}

\section{Evenwijdige en loodrechte lijnen}

\begin{definition}[Loodrecht, evenwijdig]\label{def:g6:lines:perp}
Twee rechten zijn \emph{loodrecht}\index{loodrechte lijnen} wanneer ze
snijden onder een rechte hoek; we schrijven $d_1 \perp d_2$ en
markeren de rechte hoek met een klein vierkantje. Twee rechten zijn
\emph{evenwijdig}\index{evenwijdige lijnen} wanneer ze elkaar nooit
ontmoeten, hoe ver ze ook verlengd worden; we schrijven
$d_1 \parallel d_2$.
\end{definition}

\begin{proposition}[Twee nuttige feiten]\label{prop:g6:lines:facts}
\begin{enumerate}
  \item Als twee rechten beide loodrecht staan op een derde rechte, dan
        zijn ze evenwijdig aan elkaar.
  \item Als twee rechten evenwijdig zijn, is elke rechte loodrecht op
        de ene ook loodrecht op de andere.
\end{enumerate}
\end{proposition}
\admitted

\begin{method}[Tekenen met de geodriehoek]\label{met:g6:lines:setsquare}
Om de loodlijn op een rechte $d$ door een punt $P$ te tekenen:
\begin{enumerate}
  \item leg één been van de rechte hoek van de geodriehoek langs $d$;
  \item schuif de geodriehoek langs $d$ tot het andere been $P$
        bereikt;
  \item teken de lijn langs dat been, en markeer de rechte hoek.
\end{enumerate}
Voor een evenwijdige door $P$: teken een loodlijn op $d$, daarna de
loodlijn op \emph{die} lijn door $P$
(\cref{prop:g6:lines:facts}).
\end{method}

\section{Cirkels}

\begin{definition}[Cirkel]\label{def:g6:lines:circle}
De \emph{cirkel}\index{cirkel} met middelpunt $O$ en straal $r$ is de
verzameling van alle punten op afstand precies $r$ van $O$. Een
lijnstuk van het middelpunt naar de cirkel is een \emph{straal}; een
lijnstuk dat twee punten van de cirkel verbindt door het middelpunt is
een \emph{middellijn} --- zijn lengte is $2r$; een lijnstuk dat twee
punten van de cirkel verbindt is een \emph{koorde}.
\end{definition}

\begin{example}\label{ex:g6:lines:circle}
``Teken de cirkel met middelpunt $O$ door $A$'': open de passer van $O$
tot $A$ --- de straal is de afstand $OA$ --- en draai. Elk punt van
deze cirkel ligt op dezelfde afstand van $O$ als $A$.
\end{example}

\section{Hoeken}

\begin{definition}[Hoek]\label{def:g6:lines:angle}
Twee halfrechten $[AB)$ en $[AC)$ met hetzelfde beginpunt $A$ vormen
de \emph{hoek}\index{hoek} $\widehat{BAC}$; het punt $A$ is zijn
\emph{hoekpunt} (altijd de middelste letter!). Hoeken worden gemeten in
\emph{graden} ($^\circ$), van $0^\circ$ tot $360^\circ$ voor een
volledige omwenteling. Een hoek is:
\begin{itemize}
  \item \emph{recht} als hij $90^\circ$ meet;
  \item \emph{scherp} als hij minder dan $90^\circ$ meet;
  \item \emph{stomp} als hij tussen $90^\circ$ en $180^\circ$ meet;
  \item \emph{gestrekt} als hij $180^\circ$ meet (de twee halfrechten
        vormen een rechte).
\end{itemize}
\end{definition}

\begin{method}[Een hoek meten met een gradenboog]\label{met:g6:lines:protractor}
\begin{enumerate}
  \item Plaats het middelpunt van de gradenboog precies op het
        hoekpunt;
  \item lijn de $0^\circ$-lijn uit met één been van de hoek;
  \item lees de graad af die het andere been snijdt --- op de schaal
        die bij $0$ begint op het uitgelijnde been;
  \item controleer met het oog: een scherpe hoek moet minder dan
        $90^\circ$ aangeven, een stompe meer.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g6:lines:estimate}
Schat vóór je meet! Een hoek die iets opener is dan de hoek van een
vel papier is iets meer dan $90^\circ$; de helft van een rechte hoek
is $45^\circ$; een derde van een rechte hoek is $30^\circ$. Als je
gradenboog $150^\circ$ aangeeft voor een hoek die scherp lijkt, heb
je de verkeerde schaal gelezen: de juiste maat is
$180^\circ - 150^\circ = 30^\circ$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g6:lines:1}
Teken drie punten $A$, $B$, $C$ die niet gealigneerd zijn. Teken in
verschillende kleuren: het lijnstuk $[AB]$, de halfrechte $[CA)$, de
rechte $(BC)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:2}
Waar of onwaar? ``$[AB]$ en $[BA]$ zijn hetzelfde lijnstuk.''
``$[AB)$ en $[BA)$ zijn dezelfde halfrechte.'' ``$(AB)$ en $(BA)$
zijn dezelfde rechte.'' Leg elk antwoord uit.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:3}
Teken een rechte $d$ en een punt $P$ niet op $d$. Construeer met de
geodriehoek: de loodlijn op $d$ door $P$, daarna de evenwijdige aan $d$
door $P$. Beschrijf je stappen.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:4}
Rechten $a$ en $b$ staan beide loodrecht op een rechte $c$, en een
vierde rechte $e$ staat loodrecht op $a$. Wat kun je zeggen over $a$
en $b$? Over $e$ en $c$? Motiveer met \cref{prop:g6:lines:facts}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:5}
Teken een cirkel met middelpunt $O$ en straal $3$ cm. Plaats een punt
$M$ op de cirkel, een punt $N$ erin, een punt $P$ erbuiten. Wat kun
je zeggen over de afstanden $OM$, $ON$, $OP$ vergeleken met $3$ cm?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:6}
Een cirkel heeft middellijn $9$ cm. Wat is zijn straal? Een andere
heeft straal $2.6$ cm: wat is zijn middellijn?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:7}
Noem de gemarkeerde hoek met drie letters, en classificeer hem (scherp,
recht, stomp, gestrekt): een hoek van $72^\circ$ in hoekpunt $R$ tussen
halfrechten naar $S$ en $T$; een hoek van $148^\circ$ in hoekpunt $B$
tussen halfrechten naar $A$ en $C$; een hoek van $90^\circ$ in hoekpunt
$O$ tussen halfrechten naar $M$ en $N$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:8}
Schat, en meet daarna met een gradenboog, de drie hoeken van een
driehoek die je zelf tekent. Tel de drie maten op: wat vind je?
(Bewaar je antwoord voor \cref{ch:g7:triangles}.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:lines:9}
Teken een hoek $\widehat{xOy}$ van $65^\circ$ met een gradenboog,
daarna een hoek van $130^\circ$. Hoe kun je de tweede uit de eerste
halen zonder de gradenboog?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:lines:10}
Twee dorpen $A$ en $B$ staan op een kaart. Waar liggen de punten die
op $3$ cm van $A$ \emph{en} op $2$ cm van $B$ liggen? Teken een
figuur die laat zien hoeveel zulke punten er kunnen zijn ($0$, $1$ of
$2$ afhankelijk van de afstand $AB$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:lines:11}
Een klok wijst 3 uur: wat is de hoek tussen de twee wijzers? Zelfde
vraag om 5 uur, en om 6 uur. (Een volledige omwenteling is $360^\circ$
voor $12$ uren.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:lines:12}
Teken een lijnstuk $[AB]$ van $6$ cm. Construeer het punt $C$ zo dat
$AC = BC = 6$ cm, met alleen de passer, en meet de hoek
$\widehat{CAB}$. Welke driehoek heb je gebouwd, en wat vermoed je over
zijn hoeken?
\end{exercise}
