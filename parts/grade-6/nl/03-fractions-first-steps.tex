\chapter{Breuken: eerste stappen}\label{ch:g6:fractions}

Drie pizza's eerlijk verdelen onder vier personen geeft elke
persoon\,\dots\ geen geheel aantal pizza's. Breuken zijn de getallen
die uitgevonden zijn om te delen. Dit hoofdstuk bouwt het beeld op: een
breuk als deel van een geheel, als punt op de getallenlijn, en als het
exacte resultaat van een deling.

\section{Wat een breuk betekent}

\begin{definition}[Breuk]\label{def:g6:fractions:def}
Snijd een eenheid in $b$ gelijke delen, en neem er $a$ van: de
verkregen hoeveelheid is de \emph{breuk}\index{breuk}
\[
\frac{a}{b}
\qquad
\begin{array}{l}
a \text{: de \emph{teller} --- hoeveel delen we nemen;}\\
b \text{: de \emph{noemer} --- in hoeveel delen de eenheid is
gesneden.}
\end{array}
\]
\end{definition}

\begin{example}\label{ex:g6:fractions:morethanone}
Een breuk kan groter zijn dan $1$: $\frac74$ betekent zeven kwarten ---
één hele eenheid (vier kwarten) en nog drie kwarten:
\[
\frac74 = 1 + \frac34 .
\]
\end{example}

\begin{proposition}[Breuken op de getallenlijn]\label{prop:g6:fractions:line}
Om $\frac ab$ op de getallenlijn te plaatsen, snijd elk eenheidsinterval
in $b$ gelijke delen en tel $a$ delen vanaf $0$.
\end{proposition}

\begin{theorem}[Een breuk is een deling]\label{thm:g6:fractions:quotient}
De breuk $\frac ab$ is het getal dat, vermenigvuldigd met $b$, $a$
geeft:
\[
\frac ab \times b = a .
\]
Met andere woorden is $\frac ab$ het exacte quotiënt van $a$ door $b$
--- ook wanneer de deling ``niet uitkomt''.
\end{theorem}
\admitted

\begin{example}\label{ex:g6:fractions:pizzas}
Drie pizza's voor vier personen: elke persoon krijgt $\frac34$ pizza.
Controle met de stelling: $4$ porties van $\frac34$ maken
$\frac34 \times 4 = 3$ pizza's. Sommige breuken zijn decimale getallen
($\frac34 = 0.75$), andere niet: $\frac13 = 0.333\dots$ stopt nooit.
De breuk is de \emph{exacte} waarde; $0.33$ is slechts een benadering.
\end{example}

\section{Gelijke breuken}

\begin{proposition}[Equivalente breuken]\label{prop:g6:fractions:equal}
Een breuk verandert niet wanneer teller en noemer beide vermenigvuldigd
(of gedeeld) worden met hetzelfde getal ongelijk aan nul:
\[
\frac{a}{b} = \frac{a \times k}{b \times k} .
\]
\end{proposition}

zowel het aantal genomen stukken als het totaal aantal stukken met $k$,

\end{proof}

\begin{method}[Een breuk vereenvoudigen]\label{met:g6:fractions:simplify}
Zoek een getal dat de teller \emph{en} de noemer deelt, en deel beide.
Herhaal tot er geen gemeenschappelijke deler meer is:
\[
\frac{12}{18} = \frac{12 \div 2}{18 \div 2} = \frac{6}{9}
= \frac{6 \div 3}{9 \div 3} = \frac{2}{3}.
\]
\end{method}

\section{Een breuk van een hoeveelheid nemen}

\begin{method}[Breuk van een hoeveelheid]\label{met:g6:fractions:ofquantity}
Om $\frac ab$ van een hoeveelheid te berekenen, deel de hoeveelheid door
$b$ (één deel), en vermenigvuldig daarna met $a$ (het aantal delen):
\[
\frac ab \text{ van } Q = (Q \div b) \times a .
\]
(Eerst vermenigvuldigen met $a$ en daarna delen door $b$ geeft hetzelfde
resultaat --- kies de volgorde die de getallen prettiger maakt.)
\end{method}

\begin{example}\label{ex:g6:fractions:ofquantity}
Bereken $\frac35$ van $40$ euro, stap voor stap:
\begin{enumerate}
  \item één vijfde van $40$: $40 \div 5 = 8$;
  \item drie vijfden: $8 \times 3 = 24$.
\end{enumerate}
Dus $\frac35$ van $40$ euro is $24$ euro. Controle: de resterende
$\frac25$ is $16$ euro, en $24 + 16 = 40$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g6:fractions:1}
Schrijf voor elke beschrijving de breuk: een taart in $8$ stukken,
$3$ opgegeten; een chocoladereep van $12$ vakjes, $7$ opgegeten; een
taart in $6$ stukken, alle $6$ opgegeten.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:2}
Teken een balk in $5$ gelijke delen en kleur $\frac45$ ervan. Is
$\frac45$ kleiner of groter dan $1$? En $\frac65$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:3}
Plaats op een getallenlijn afgestreept in derden: $\frac13$; \
$\frac53$; \ $2$; \ $\frac73$. Welke twee van deze getallen liggen
op gelijke afstand van $2$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:4}
Schrijf elke breuk als een geheel getal plus een breuk kleiner dan $1$
(zoals $\frac74 = 1 + \frac34$):
\[
\frac{9}{4}, \qquad \frac{13}{5}, \qquad \frac{12}{6}, \qquad
\frac{25}{8} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:5}
Vul in zodat de breuken gelijk zijn:
\[
\frac{2}{3} = \frac{?}{12}, \qquad
\frac{15}{20} = \frac{3}{?}, \qquad
\frac{7}{10} = \frac{21}{?}, \qquad
\frac{?}{6} = \frac{10}{12}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:6}
Vereenvoudig zoveel mogelijk: $\dfrac{6}{8}$; \quad $\dfrac{15}{25}$;
\quad $\dfrac{18}{24}$; \quad $\dfrac{35}{7}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:7}
Bereken: $\frac12$ van $86$; \quad $\frac34$ van $60$; \quad $\frac25$
van $35$; \quad $\frac{7}{10}$ van $250$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:fractions:8}
Welke is groter: $\frac12$ of $\frac13$ (van dezelfde taart)?
$\frac25$ of $\frac35$? $\frac34$ of $\frac78$ (schrijf beide met
noemer $8$)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:fractions:9}
Geef de decimale waarde van de breuken die er een hebben, en zeg welke
breuk dat niet heeft: $\frac12$; \ $\frac34$; \ $\frac13$; \
$\frac{7}{10}$; \ $\frac{9}{4}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:fractions:10}
Een klas heeft $28$ leerlingen. Driekwart van hen komt met de bus naar
school, en de rest loopt. Hoeveel leerlingen lopen? Los stap voor stap
op, en controleer dat de twee groepsgroottes samen $28$ zijn.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:fractions:11}
Leo gaf $\frac23$ van zijn spaargeld uit aan een spel dat $18$ euro
kostte. Hoeveel geld had Leo ervoor? (Eén derde van zijn spaargeld is
de helft van $18$\,\dots\ denk na waarom, of vind eerst de grootte van
één deel.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:fractions:12}
Een tank is tot $\frac58$ van zijn capaciteit gevuld. Na gebruik van
$10$ liter is hij halfvol. Wat is de capaciteit van de tank? (Welk
deel van de tank stellen de $10$ liter voor?)
\end{exercise}
