\chapter{Evenredigheid en gegevens}\label{ch:g6:propdata}

Als drie schriften $6$ euro kosten, kosten zes schriften $12$:
dubbel zoveel schriften, dubbel de prijs. Grootheden die zich zo
gedragen zijn \emph{evenredig} --- een van de nuttigste ideeën in de
hele wiskunde, verder uitgewerkt in \cref{ch:g7:prop}. Het hoofdstuk
eindigt met het lezen en tekenen van eenvoudige gegevensgrafieken.

\section{Evenredige grootheden}

\begin{definition}[Evenredigheid]\label{def:g6:propdata:def}
Twee grootheden zijn \emph{evenredig}\index{evenredigheid} wanneer de
waarden van de ene uit die van de andere ontstaan door altijd met
\emph{hetzelfde} getal te vermenigvuldigen, de \emph{evenredigheidscoëfficiënt}.
\end{definition}

\begin{example}\label{ex:g6:propdata:table}
Schriften aan $2$ euro per stuk:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cccc}
schriften & $3$ & $5$ & $8$ & $12$ \\
\hline
prijs (euro) & $6$ & $10$ & $16$ & $24$ \\
\end{tabular}
\end{center}
Elke prijs is het aantal schriften $\times 2$: de coëfficiënt is $2$
(de eenheidsprijs). Een snelle controle dat een tabel evenredig is:
alle ``kolomquotiënten'' $\frac{6}{3}, \frac{10}{5}, \frac{16}{8},
\frac{24}{12}$ moeten gelijk zijn --- hier zijn ze allemaal $2$.
\end{example}

\begin{example}[Een tegenvoorbeeld]\label{ex:g6:propdata:nonexample}
Leeftijd en lengte zijn niet evenredig: een $12$-jarige is niet twee
keer zo lang als een $6$-jarige. Controleer met getallen: $1.50$ m op
$12$ jaar en $1.15$ m op $6$ jaar geven quotiënten
$\frac{1.50}{12} = 0.125$ en $\frac{1.15}{6} \approx 0.19$ --- niet
gelijk.
\end{example}

\begin{method}[Een evenredigheidstabel invullen]\label{met:g6:propdata:complete}
Drie manieren, vrij te kiezen:
\begin{enumerate}
  \item \emph{coëfficiënt}: vind de vermenigvuldiger uit één complete
        kolom, pas hem toe op de andere;
  \item \emph{kolommen}: een kolom mag met een getal vermenigvuldigd
        worden, of twee kolommen mogen opgeteld worden, om een nieuwe
        kolom te maken;
  \item \emph{terug naar de eenheid}: vind eerst de waarde voor
        \emph{één} item, en vermenigvuldig daarna.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g6:propdata:methods}
Vijf identieke mokken kosten $15$ euro; hoeveel kosten acht mokken?

\emph{Terug naar de eenheid}: één mok kost $15 \div 5 = 3$ euro, dus
acht mokken kosten $8 \times 3 = 24$ euro.

\emph{Kolommen}: $8 = 5 + 3$; drie mokken kosten $\frac{3}{5}$ van
$15$, d.w.z.\ $9$; dus acht mokken kosten $15 + 9 = 24$ euro.
Hetzelfde antwoord, zoals het moet.
\end{example}

\section{Percentages}

\begin{definition}[Percentage]\label{def:g6:propdata:percent}
``$t\,\%$ van een hoeveelheid'' betekent de breuk $\frac{t}{100}$
ervan: $25\,\%$ van $60$ is $\frac{25}{100} \times 60 = 15$. Een
percentage is een evenredigheid met coëfficiënt $\frac{t}{100}$.
\end{definition}

\begin{example}\label{ex:g6:propdata:percent}
Een klas van $30$ leerlingen telt $40\,\%$ meisjes. Aantal meisjes,
stap voor stap: $40\,\%$ betekent $\frac{40}{100} = 0.4$, en
$0.4 \times 30 = 12$ meisjes. Nuttige ankerpunten:
$50\,\%$ is de helft, $25\,\%$ een kwart, $10\,\%$ een tiende ---
dus $10\,\%$ van $30$ is $3$, en $40\,\% = 4 \times 10\,\%$ is
$4 \times 3 = 12$: hetzelfde antwoord, uit het hoofd.
\end{example}

\section{Grafieken lezen en tekenen}

\begin{method}[Een tabel of grafiek lezen]\label{met:g6:propdata:read}
Welk plaatje ook --- tabel, staafdiagram, lijndiagram:
\begin{enumerate}
  \item lees eerst de titel en de eenheden op beide assen;
  \item om een vraag te beantwoorden, zoek de juiste staaf/kolom, en
        lees de waarde zorgvuldig af tegen de schaalverdeling;
  \item om te vergelijken, vergelijk staven op hoogte --- maar
        controleer of de as bij $0$ begint, anders kan het plaatje
        misleiden.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g6:propdata:chart}
Het staafdiagram hieronder toont het aantal boeken dat in één week
uitgeleend is in de schoolbibliotheek.

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[
  width=8.6cm, height=5.2cm,
  ybar, bar width=16pt,
  symbolic x coords={ma,di,wo,do,vr},
  xtick=data,
  ymin=0, ymax=22,
  ytick={5,10,15,20},
  ylabel={uitgeleende boeken},
  tick label style={font=\small},
  label style={font=\small}]
  \addplot[fill=omDef!30, draw=omDef] coordinates
    {(ma,12) (di,8) (wo,17) (do,6) (vr,15)};
\end{axis}
\end{tikzpicture}

{\small Een staafdiagram: één staaf per dag, hoogte gelijk aan het
aantal.}
\end{omfigure}

Aflezingen: de drukste dag is woensdag ($17$ boeken). Dinsdag en
donderdag samen zijn $8 + 6 = 14$ boeken --- minder dan vrijdag
alleen ($15$). Totaal voor de week:
$12 + 8 + 17 + 6 + 15 = 58$ boeken.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g6:propdata:1}
Is de tabel evenredig? Motiveer door quotiënten te berekenen.
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|ccc}
kg appels & $2$ & $3$ & $5$ \\
\hline
prijs (euro) & $5$ & $7.5$ & $12.5$ \\
\end{tabular}
\qquad
\begin{tabular}{l|ccc}
leeftijd (jaar) & $2$ & $4$ & $8$ \\
\hline
lengte (cm) & $85$ & $103$ & $130$ \\
\end{tabular}
\end{center}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:propdata:2}
Vier croissants kosten $4.80$ euro. Vind de prijs van één croissant,
daarna van zeven croissants.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:propdata:3}
Vul de evenredigheidstabel in (eerst de coëfficiënt!):
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|cccc}
liters brandstof & $10$ & $25$ & $?$ & $60$ \\
\hline
prijs (euro) & $18$ & $?$ & $72$ & $?$ \\
\end{tabular}
\end{center}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:propdata:4}
Een auto verbruikt $6$ L brandstof per $100$ km. Hoeveel brandstof
voor $250$ km? Voor $350$ km? Hoe ver komt hij met $27$ L?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:propdata:5}
Reken uit het hoofd, met de ankerpunten van
\cref{ex:g6:propdata:percent}: $50\,\%$ van $84$; \quad $25\,\%$ van
$200$; \quad $10\,\%$ van $63$; \quad $30\,\%$ van $70$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:propdata:6}
In een school van $450$ leerlingen eet $60\,\%$ in de kantine. Hoeveel
leerlingen is dat? Hoeveel niet?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:propdata:7}
Gebruik het staafdiagram van \cref{ex:g6:propdata:chart}: op welke
dagen werden minder dan $10$ boeken uitgeleend? Hoeveel boeken meer
werden er op woensdag uitgeleend dan op donderdag?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:propdata:8}
De temperaturen om twaalf uur van maandag tot vrijdag waren $14$,
$16$, $13$, $17$, $20$ graden. Teken een staafdiagram van deze
gegevens (kies een zinvolle schaalverdeling), en lees de warmste dag
af.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:propdata:9}
Een recept voor $6$ personen heeft $450$ g meel en $3$ eieren nodig.
Pas het aan voor $10$ personen. (Voor de eieren: denk na vóór je een
decimaal aantal eieren opschrijft!)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:propdata:10}
Met constante snelheid fietst een fietser $24$ km in $1$ uur.
\begin{enumerate}
  \item Hoe ver fietst ze in $2$ u? In een half uur? In
        $1$ u $30$ min?
  \item Hoe lang heeft ze nodig voor $60$ km?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:propdata:11}
Een winkel biedt ``$20\,\%$ korting op alles''. Bereken de korting en
de nieuwe prijs voor: een bal van $15$ euro; een racket van $40$ euro.
Is de nieuwe prijs evenredig met de oude prijs? Wat is de
coëfficiënt?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:propdata:12}
Twee kaarsen van dezelfde hoogte worden tegelijk aangestoken. De dikke
brandt volledig op in $6$ uur, de dunne in $4$ uur, elk met zijn
eigen constante snelheid. Na hoeveel tijd is de dunne kaars exact half
zo hoog als de dikke? (Druk de resterende hoogten na $t$ uur uit als
breuken van de beginhoogte.)
\end{exercise}
