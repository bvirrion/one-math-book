\chapter{Números decimais}\label{ch:g6:decimals}

Entre $2$ e $3$ há muitos números: $2.5$, $2.71$,
$2.999$\,\dots\ Números decimais estender o sistema de valor posicional para o
à direita das unidades e seguem as mesmas regras dos números inteiros ---
com algumas armadilhas que este capítulo ensinará você a evitar.

\section{Escrita decimal}

\begin{definition}[Casas decimais]\label{def:g6:decimals:places}
UM \emph{número decimal}\index{número decimal} tem uma parte inteira e uma
parte decimal, separada por \omterm{def:g4:decimals:point}{ponto decimal}. Cada lugar à direita
do ponto vale dez vezes menos que o anterior: décimos,
centésimos, milésimos. Por exemplo
\[
13.407 = 13 + \frac{4}{10} + \frac{0}{100} + \frac{7}{1000}
= 1 \times 10 + 3 + 4 \times \frac{1}{10} + 7 \times \frac{1}{1000}.
\]
\end{definition}

\begin{remark}[Zeros que importam, zeros que não importam]
Adicionando \omterm{def:g1:counting:zero}{zeros} no \emph{fim} da parte decimal não muda nada:
$2.5 = 2.50 = 2.500$. Mas \omterm{def:g1:counting:zero}{zeros} \emph{entre} dígitos são essenciais:
$13.407 \neq 13.47$. E $0.5$ é muito diferente de $0.05$!
\end{remark}

\begin{method}[Comparando números decimais]\label{met:g6:decimals:compare}
\begin{enumerate}
  \item Compare as partes inteiras primeiro: $7.2 > 5.99$ porque $7 > 5$.
  \item Se as partes inteiras forem iguais, compare o dígito das partes decimais por
        dígito da esquerda --- preenchimento com final \omterm{def:g1:counting:zero}{zeros} ajuda:
        comparar $3.4$ e $3.15$, escrever $3.40$ e $3.15$;
        desde $40 > 15$ centésimos, $3.4 > 3.15$.
\end{enumerate}
Cuidado: \emph{mais não significa maior}. $3.15$ tem mais dígitos do que
$3.4$ mas é menor.
\end{method}

\section{Adição, subtração, multiplicação}

\begin{method}[Cálculos de coluna com decimais]\label{met:g6:decimals:columns}
Para adicionar ou subtrair números decimais, alinhe o \emph{pontos decimais}
(portanto, as unidades ficam abaixo das unidades, décimos abaixo dos décimos), preenchimento com final
\omterm{def:g1:counting:zero}{zeros} se necessário; então calcule como com números inteiros e coloque o ponto
do resultado sob os demais.
\end{method}

\begin{example}\label{ex:g6:decimals:addsub}
Calcular $13.7 + 2.85$, alinhando os pontos ($13.7 = 13.70$):
\[
\begin{array}{r}
1\,3.7\,0 \\
+\ \ \,2.8\,5 \\
\hline
1\,6.5\,5
\end{array}
\]
Calcular $6 - 2.35$ (escrever $6 = 6.00$):
$6.00 - 2.35 = 3.65$. Verificar: $3.65 + 2.35 = 6$.
\end{example}

\begin{example}[Multiplicando decimais]\label{ex:g6:decimals:mult}
Para calcular $2.3 \times 1.4$: multiplique como números inteiros,
$23 \times 14 = 322$; então conte os dígitos decimais dos fatores
($1 + 1 = 2$) e coloque o ponto de forma que o resultado tenha esse número:
$2.3 \times 1.4 = 3.22$. Verificação de sanidade nos tamanhos: $2.3 \times 1.4$
deveria ser um pouco mais do que $2.3 \times 1 = 2.3$ --- e $3.22$ é.
\end{example}

\section{Multiplicando e dividindo por 10, 100, 1000}

\begin{proposition}[Mudando o ponto]\label{prop:g6:decimals:shift}
Multiplicando um \omterm{def:g6:decimals:places}{número decimal} por $10$, $100$, $1000$ move seu decimal
ponto $1$, $2$, $3$ lugares para o \emph{certo}; a divisão move-o para o
\emph{esquerda} (preenchimento com \omterm{def:g1:counting:zero}{zeros} quando necessário):
\[
3.75 \times 100 = 375, \qquad
42.1 \div 1000 = 0.0421 .
\]
\end{proposition}

\begin{proof}
Multiplicando por $10$ faz com que cada dígito valha dez vezes mais: décimos
tornam-se unidades, as unidades tornam-se dezenas e assim por diante --- cada dígito se move um
coluna à esquerda na tabela de valores posicionais, que é o mesmo que mover
o ponto um lugar à direita. A divisão inverte isso.
\end{proof}

\begin{example}[Unidades de medida]\label{ex:g6:decimals:units}
Converter unidades é exatamente este jogo: já que $1$ eu $= 100$ cm,
\[
3.75 \text{ m} = 3.75 \times 100 \text{ cm} = 375 \text{ cm},
\qquad
42 \text{ mm} = 42 \div 10 \text{ cm} = 4.2 \text{ cm}.
\]
\end{example}

\section{Arredondamento}

\begin{definition}[Arredondamento]\label{def:g6:decimals:rounding}
O \emph{arredondamento}\index{arredondamento} de um número elevado à unidade (ou décimo,
centésimo, \dots) é o número mais próximo com essa precisão. Olhe para o
próximo dígito: se for $0,1,2,3,4$, redondo para baixo (manter); se for
$5,6,7,8,9$, redondo acima.
\end{definition}

\begin{example}
$7.38$ \omterm{def:g4:numbers:round}{arredondado} para a unidade é $7$ (próximo dígito $3$: manter); \omterm{def:g4:numbers:round}{arredondado} para
o décimo é $7.4$ (próximo dígito $8$: redondo acima). O preço
$4.996$ \omterm{def:g4:numbers:round}{arredondado} para o centésimo é $5.00$ --- \omterm{def:g6:decimals:rounding}{arredondamento} pode mudar
cada dígito!
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g6:decimals:1}
Escreva como um \omterm{def:g6:decimals:places}{número decimal}: $5 + \dfrac{3}{10} + \dfrac{7}{100}$;
\quad $\dfrac{9}{10}$; \quad $12 + \dfrac{4}{1000}$;
\quad ``oito unidades e cinco centésimos''.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:decimals:2}
Em $86.354$: qual é o algarismo das décimas? Dos centésimos? O que
o dígito $8$ contar? Escreva este número como em
\cref{def:g6:decimals:places}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:decimals:3}
Copie e complete com $<$, $>$ ou $=$:
\[
5.3 \;?\; 5.29, \qquad
0.7 \;?\; 0.70, \qquad
2.09 \;?\; 2.9, \qquad
14.5 \;?\; 14.49 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:decimals:4}
Ordene do menor para o maior:
$4.2$; \ $4.05$; \ $4.51$; \ $4.15$; \ $4.5$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:decimals:5}
Qual números decimais correspondem aos pontos $ A $, $ B $, $ C $ em um
reta numérica graduada em décimos, se $ A $ é $3$ carrapatos depois $6$, $ B $ é
$7$ carrapatos depois $6$, e $ C $ é $2$ carrapatos depois $7$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:decimals:6}
Calcule em colunas: $45.8 + 7.65$; \quad $23.4 - 8.72$; \quad
$5.6 \times 2.4$ (conte os dígitos decimais!).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:decimals:7}
Calcule sem nenhum trabalho escrito:
\[
6.42 \times 10, \qquad
0.35 \times 1000, \qquad
78.1 \div 100, \qquad
5 \div 1000 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:decimals:8}
Converter: $2.4$ m em cm; \quad $370$ g em kg; \quad $0.85$ km em
m; \quad $56$ mm em m.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:decimals:9}
Redondo $23.867$: para a unidade; ao décimo; ao centésimo. Redondo
$9.97$ ao décimo.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:decimals:10}
Uma baguete custa $1.15$. Lena compra três baguetes e paga com uma
$5$ conta. Escreva os dois cálculos necessários e dê-lhe o troco.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:decimals:11}
Encontre um \omterm{def:g6:decimals:places}{número decimal} estritamente entre $7.4$ e $7.5$; então entre
$3.99$ e $4$. Quantos desses números existem em cada caso?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g6:decimals:12}
Usando cada um dos dígitos $2$, $5$, $8$ exatamente uma vez e uma decimal
ponto, escreva o maior número possível e depois o menor possível
um. (Números como $.58$ não são permitidos: a parte inteira deve conter
pelo menos um dígito.)
\end{exercise}

\section{Problema: O número logo após 3 não existe}

\begin{problem}[{Problema de fim de semana --- entre quaisquer duas casas decimais
números, sempre há outro: ampliar a reta numérica}]
\label{pb:g6:decimals:1}
Na escala dos números inteiros, cada número tem uma porta ao lado
vizinho: logo depois $7$ vem $8$, e nada vive em
entre. Números decimais são um mundo completamente diferente. O que é
o número logo depois $3$? É isso $3.1$? $3.01$? $3.001$? Isto
problema desenvolve um \emph{zoom} técnica na reta numérica
(continuando \cref{exo:g6:decimals:11}) e alcança um famoso
conclusão: o número logo após $3$ \emph{não existe} ---
entre quaisquer dois números decimais, por mais próximo que esteja, há sempre
espaço para mais.

\medskip
\noindent\textbf{Part I --- Zooming in.}
\begin{enumerate}
  \item Qual é maior, $2.999$ ou $3$? Calcule o \omterm{ex:g1:subtraction:difference}{diferença}
        entre eles.
  \item Desenhe uma reta numérica a partir $7.4$ para $7.5$, formou-se em
        centésimos e lugar $7.42$, $7.45$ e $7.48$ nisso.
        Lista \emph{todos} os números com duas casas decimais que
        mentir estritamente entre $7.4$ e $7.5$. Quantos são?
  \item Zoom novamente: entre $7.44$ e $7.45$, liste todos os
        números com três dígitos decimais. Quantos existem
        desta vez?
  \item Usando a mesma ideia, explique como contar os números
        com exatamente três dígitos decimais situados estritamente entre
        $7.4$ e $7.5$, e dê essa contagem.
  \item Descreva a receita escondida atrás das questões 2 a 4 (o
        \emph{ampliar}): dados dois números que se parecem com
        vizinhos, como $5.67$ e $5.68$, como é que a escrita
        mais uma casa decimal sempre revela um número estritamente
        entre eles? Aplique sua receita a $5.67$ e $5.68$,
        então para $0.1999$ e $0.2$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The missing neighbour.}
\begin{enumerate}[resume]
  \item Tom afirma: ``o número logo após $3$ é $3.1$.'' Provar
        ele errou ao nomear um número estritamente entre $3$ e
        $3.1$. Tom recua para $3.01$, então para $3.001$. Bata cada
        dos seus candidatos.
  \item Explique por que \emph{ninguém} pode ganhar este jogo contra você:
        qualquer número estritamente maior que $3$ Tom propõe,
        a receita de zoom da pergunta 5 produz um número estritamente
        entre $3$ e sua proposta. O que isso prova
        ``o número logo depois $3$''?
  \item Agora o outro lado: Tom procura o número apenas
        \emph{antes} $3$ e tenta $2.9$, então $2.99$, então
        $2.999$. Vencer seus três candidatos. Então calcule
        $3 - 2.999$ e descrever (sem calcular em colunas)
        o \omterm{ex:g1:subtraction:difference}{diferença} entre $3$ e o número escrito com um
        $2$, um ponto e vinte dígitos $9$.
  \item Por que nada disso funciona para \emph{todo} números?
        Explique em uma ou duas frases por que não existe um todo
        número estritamente entre $7$ e $8$, embora haja
        são muitos números decimais lá.
  \item Um comprimento é anunciado como ``$3.7$ cm, \omterm{def:g4:numbers:round}{arredondado} para
        décimo'' (\cref{def:g6:decimals:rounding}). Dê o
        menor comprimento que rodadas para $3.7$ cm e explique por que
        um comprimento de $3.75$ cm faz \emph{não} redondo para $3.7$ ---
        então o comprimento verdadeiro é pelo menos $3.65$ cm e estritamente
        abaixo $3.75$ cm.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Games with digits and points.}
\begin{enumerate}[resume]
  \item Ordene do menor para o maior: $0.6$; \ $0.58$; \
        $0.123$; \ $0.0999$. Em seguida, explique a um colega, em um
        sentença, por que $0.0999 < 0.6$ mesmo que esteja escrito
        com mais dígitos (\cref{met:g6:decimals:compare}).
  \item Os preços em uma loja sempre têm exatamente dois dígitos decimais.
        Tem alguma \emph{preço} estritamente entre $4.99$ e $5$?
        Compare com a pergunta 9: o que significam os preços e os números inteiros?
        têm em comum?
  \item Usando cada um dos dígitos $2$, $5$, $8$ exatamente uma vez e
        um \omterm{def:g4:decimals:point}{ponto decimal} (parte inteira não vazia, como em
        \cref{exo:g6:decimals:12}), encontre o número mais próximo de
        $6$. Justifique calculando a distância do seu melhor
        candidatos para $6$.
  \item O número mais famoso da matemática, $\pi $, satisfaz
        $3.141 < \pi < 3.142$. Nomeie um \omterm{def:g6:decimals:places}{número decimal} estritamente
        entre $3.141$ e $3.142$; então um estritamente entre
        $3.1415$ e $3.1416$. (Os matemáticos apertam $\pi $
        exatamente assim --- cada novo dígito decimal é mais um
        zoom. Você vai conhecer $\pi $ no trabalho em
        \cref{ch:g6:measure}.)
  \item O final: explique por que existem \emph{mais decimal
        números entre $0$ e $1$ do que qualquer número que você puder
        nome}. (Quantos um zoom produz? O zoom pode
        já parou?)
\end{enumerate}
\end{problem}
