\chapter{Linhas, círculos e ângulos}\label{ch:g6:lines}

A geometria começa com uma régua, um esquadro e um compasso. Este capítulo
corrige o vocabulário --- pontos, \omterm{def:g6:lines:objects}{segmentos}, \omterm{def:g3:shapes:shapes}{raios}, \omterm{def:g6:lines:objects}{linhas}, círculos ---
introduz as duas posições especiais de \omterm{def:g6:lines:objects}{linhas} (\omterm{def:g6:lines:perp}{paralelo} e
\omterm{def:g6:lines:perp}{perpendicular}) e ensina como medir e desenhar ângulos.

\section{Pontos, segmentos, linhas}

\begin{definition}[Objetos básicos]\label{def:g6:lines:objects}
Através de dois pontos distintos $A $ e $ B $ passar:
\begin{itemize}
  \item o \emph{segmento}\index{segmento} $[AB]$: a parte da linha
        entre $ A $ e $ B $ (tem um comprimento, escrito $ AB $);
  \item o \emph{\omterm{def:g3:shapes:shapes}{raio}} $[AB)$: começa às $ A $, passa $ B $ e
        continua para sempre;
  \item o \emph{linha}\index{linha} $(AB)$: se estende para sempre em ambos
        lados. Dois pontos determinam exatamente uma linha.
\end{itemize}
Os pontos na mesma reta são chamados \emph{alinhado}.
\end{definition}

\section{Linhas paralelas e perpendiculares}

\begin{definition}[Perpendicular, paralelo]\label{def:g6:lines:perp}
Duas \omterm{def:g6:lines:objects}{linhas} são \emph{perpendicular}\index{perpendicular} quando eles
cruzar em um \omterm{def:g3:shapes:rightangle}{ângulo reto}; nós escrevemos $ d_1 \perp d_2$ e marque o certo
ângulo com um pequeno quadrado. Duas \omterm{def:g6:lines:objects}{linhas} são
\emph{paralelo}\index{paralelo} quando eles nunca se encontram, por mais longe
eles são estendidos; nós escrevemos $ d_1 \parallel d_2$.
\end{definition}

\begin{proposition}[Dois fatos úteis]\label{prop:g6:lines:facts}
\begin{enumerate}
  \item Se duas \omterm{def:g6:lines:objects}{linhas} são ambas \omterm{def:g6:lines:perp}{perpendicular} para uma terceira \omterm{def:g6:lines:objects}{linha}, eles são
        \omterm{def:g6:lines:perp}{paralelo} um para o outro.
  \item Se duas \omterm{def:g6:lines:objects}{linhas} forem \omterm{def:g6:lines:perp}{paralelo}, cada \omterm{def:g6:lines:objects}{linha} \omterm{def:g6:lines:perp}{perpendicular} para um é
        \omterm{def:g6:lines:perp}{perpendicular} para o outro.
\end{enumerate}
\end{proposition}
\admitted

\begin{method}[Desenhando com o esquadro]\label{met:g6:lines:setsquare}
Para desenhar o \omterm{def:g6:lines:perp}{perpendicular} para uma \omterm{def:g6:lines:objects}{linha} $ d $ através de um ponto $ P $:
\begin{enumerate}
  \item coloque uma borda do \omterm{def:g3:shapes:rightangle}{ângulo reto} do esquadro ao longo $ d $;
  \item deslize o esquadro ao longo $ d $ até que sua outra borda alcance
        $ P $;
  \item desenhe a \omterm{def:g6:lines:objects}{linha} ao longo dessa borda e marque o \omterm{def:g3:shapes:rightangle}{ângulo reto}.
\end{enumerate}
Por um \omterm{def:g6:lines:perp}{paralelo} através $ P $: desenhe um \omterm{def:g6:lines:perp}{perpendicular} para $ d $, então o
\omterm{def:g6:lines:perp}{perpendicular} para \emph{que} \omterm{def:g6:lines:objects}{linha} através $ P $
(\cref{prop:g6:lines:facts}).
\end{method}

\section{Círculos}

\begin{definition}[Círculo]\label{def:g6:lines:circle}
O \emph{círculo}\index{círculo} do centro $ O $ e \omterm{def:g3:shapes:shapes}{raio} $ r $ é o conjunto
de todos os pontos a uma distância exata $ r $ de $ O $. UM \omterm{def:g6:lines:objects}{segmento} do
centro do círculo é um \emph{\omterm{def:g3:shapes:shapes}{raio}}; um \omterm{def:g6:lines:objects}{segmento} juntando dois pontos
do círculo que passa pelo centro é um \emph{\omterm{def:g4:geometry:circle}{diâmetro}} --- seu comprimento é
$2r $; um \omterm{def:g6:lines:objects}{segmento} unir dois pontos do círculo é um \emph{acorde}.
\end{definition}

\begin{example}\label{ex:g6:lines:circle}
``Desenhe o círculo do centro $ O $ passando $ A $'': abra a bússola
de $ O $ para $ A $ --- o \omterm{def:g3:shapes:shapes}{raio} é a distância $ OA $ --- e vire. Cada
ponto deste círculo está à mesma distância de $ O $ como $ A $.
\end{example}

\section{Ângulos}

\begin{definition}[Ângulo]\label{def:g6:lines:angle}
Dois \omterm{def:g3:shapes:shapes}{raios} $[AB)$ e $[AC)$ com o mesmo ponto de partida $ A $ formar o
\emph{ângulo}\index{ângulo} $\widehat{BAC}$; o ponto $ A $ é o seu
\emph{vértice} (sempre a letra do meio!). Os ângulos são medidos em
\emph{graus} ($^\circ $), de $0^\circ $ para $360^\circ $ por um completo
vire. Um ângulo é:
\begin{itemize}
  \item \emph{certo} se medisse $90^\circ $;
  \item \emph{agudo} se medir menos que $90^\circ $;
  \item \emph{obtuso} se medir entre $90^\circ $ e
        $180^\circ $;
  \item \emph{direto} se medisse $180^\circ $ (os dois \omterm{def:g3:shapes:shapes}{raios} formam
        uma \omterm{def:g6:lines:objects}{linha}).
\end{itemize}
\end{definition}

\begin{method}[Medindo um ângulo com um transferidor]\label{met:g6:lines:protractor}
\begin{enumerate}
  \item Coloque o centro do transferidor exatamente no vértice do
        ângulo;
  \item alinhe seu $0^\circ $ alinhe com um lado do canto;
  \item leia a graduação atravessada do outro lado --- usando o
        escala que começa em $0$ no lado alinhado;
  \item verificação de sanidade com o olho: um ângulo agudo deve ler menos que
        $90^\circ $, mais um obtuso.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g6:lines:estimate}
Antes de medir, estime! Um ângulo ligeiramente mais aberto que o canto
de uma folha de papel está um pouco acima $90^\circ $; \omterm{def:g3:division:half}{metade} um \omterm{def:g3:shapes:rightangle}{ângulo reto} é
$45^\circ $; um terço de um \omterm{def:g3:shapes:rightangle}{ângulo reto} é $30^\circ $. Se o seu transferidor
diz $150^\circ $ para um ângulo que parece agudo, você leu errado
escala: a medida correta é $180^\circ - 150^\circ = 30^\circ $.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g6:lines:1}
Desenhe três pontos $ A $, $ B $, $ C $ não alinhado. Desenhe em cores diferentes:
o \omterm{def:g6:lines:objects}{segmento} $[AB]$, o \omterm{def:g3:shapes:shapes}{raio} $[CA)$, a \omterm{def:g6:lines:objects}{linha} $(BC)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:2}
Verdadeiro ou falso? ``$[AB]$ e $[BA]$ são iguais \omterm{def:g6:lines:objects}{segmento}.''
``$[AB)$ e $[BA)$ são o mesmo \omterm{def:g3:shapes:shapes}{raio}.'' ``$(AB)$ e $(BA)$ são os
mesma \omterm{def:g6:lines:objects}{linha}. '' Explique cada resposta.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:3}
Desenhe uma \omterm{def:g6:lines:objects}{linha} $ d $ e um ponto $ P $ não ligado $ d $. Construa com o conjunto
quadrado: o \omterm{def:g6:lines:perp}{perpendicular} para $ d $ através $ P $, então o \omterm{def:g6:lines:perp}{paralelo} para $ d $
através $ P $. Descreva seus passos.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:4}
\omterm{def:g6:lines:objects}{Linhas} $ a $ e $ b $ são ambos \omterm{def:g6:lines:perp}{perpendicular} para uma \omterm{def:g6:lines:objects}{linha} $ c $, e um quarto
\omterm{def:g6:lines:objects}{linha} $ e $ é \omterm{def:g6:lines:perp}{perpendicular} para $ a $. O que você pode dizer sobre $ a $ e $ b $?
Sobre $ e $ e $ c $? Justifique com \cref{prop:g6:lines:facts}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:5}
Desenhe um círculo de centro $ O $ com \omterm{def:g3:shapes:shapes}{raio} $3$ cm. Coloque um ponto $ M $ ligado
o círculo, um ponto $ N $ dentro, um ponto $ P $ fora. O que você pode dizer
sobre as distâncias $ OM $, $ ON $, $ OP $ comparado com $3$ cm?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:6}
Um círculo tem \omterm{def:g4:geometry:circle}{diâmetro} $9$ cm. Qual é o seu \omterm{def:g3:shapes:shapes}{raio}? Outro tem \omterm{def:g3:shapes:shapes}{raio}
$2.6$ cm: qual é o seu \omterm{def:g4:geometry:circle}{diâmetro}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:7}
Nomeie o ângulo marcado em três letras e depois classifique-o (agudo, direito,
obtuso, reto): um ângulo de $72^\circ $ no vértice $ R $ entre \omterm{def:g3:shapes:shapes}{raios}
em direção $ S $ e $ T $; um ângulo de $148^\circ $ no vértice $ B $ entre \omterm{def:g3:shapes:shapes}{raios}
em direção $ A $ e $ C $; um ângulo de $90^\circ $ no vértice $ O $ entre \omterm{def:g3:shapes:shapes}{raios}
em direção $ M $ e $ N $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:8}
Estime e depois meça com um transferidor os três ângulos de um
triângulo que você mesmo desenha. Some as três medidas: o que você encontra?
(Guarde sua resposta para \cref{ch:g7:triangles}.)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:lines:9}
Desenhe um ângulo $\widehat{xOy}$ de $65^\circ $ com um transferidor, depois um
ângulo de $130^\circ $. Como você conseguiu o segundo do primeiro
sem o transferidor?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:lines:10}
Duas aldeias $ A $ e $ B $ são desenhados em um mapa. Onde estão os pontos que
estão em $3$ cm de $ A $ \emph{e} no $2$ cm de $ B $? Desenhe uma imagem
mostrando quantos desses pontos pode haver ($0$, $1$ ou $2$ dependendo de
a distância $ AB $).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:lines:11}
Um relógio marca 3 horas: qual é o ângulo entre os dois ponteiros? Igual
pergunta às 5 horas e às 6 horas. (Uma volta completa é $360^\circ $
para $12$ horas.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g6:lines:12}
Desenhe um \omterm{def:g6:lines:objects}{segmento} $[AB]$ de $6$ cm. Construa o ponto $ C $ tal que
$ AC = BC = 6$ cm, usando apenas a bússola, e meça o ângulo
$\widehat{CAB}$. Que triângulo você construiu e o que você
conjectura sobre seus ângulos?
\end{exercise}

\section{Problema: A geometria do mostrador do relógio}

\begin{problem}[{Problema de fim de semana --- ângulos como frações de um
turno: lendo-os em um relógio, caçando os momentos em que os ponteiros
conhecer e fatiar um dia em uma torta}]
\label{pb:g6:lines:1}
Um relógio é um transferidor que indica a hora: é face é um completo
vez de $360^\circ $, cortado pelas marcas de doze horas em doze
ângulos iguais. \Cref{exo:g6:lines:11} mediu as mãos em
3h e 6h; este problema constrói a teoria completa
--- incluindo os momentos em que \emph{nenhum} mão aponta para uma marca
--- responde a uma pergunta que poucos adultos acertam (``com que frequência o
duas mãos ficam exatamente uma em cima da outra?''), e termina
fatiando um dia inteiro em um gráfico de pizza.

\medskip
\noindent\textbf{Part I --- Frações of a turn.}
\begin{enumerate}
  \item Uma volta completa mede $360^\circ $. Quantos graus são
        \omterm{def:g3:division:half}{metade} uma volta, um \omterm{def:g3:division:half}{trimestre} de volta, um duodécimo de volta?
  \item As marcas de doze horas cortam o relógio face em doze
        ângulos iguais no centro. Quantos graus entre dois
        marcas vizinhas? Recuperar a resposta de
        \cref{exo:g6:lines:11} às 3 horas contando marcas.
  \item Dê o ângulo entre os ponteiros à 1 hora, à
        4 horas e às 7 horas. (Às 7 horas os ponteiros
        separar o face em dois ângulos; ``o ângulo entre
        as mãos'' sempre significam o \emph{menor} um.)
  \item O ponteiro dos minutos dá uma volta completa $60$ minutos. Como
        varre muitos graus por minuto?
  \item O ponteiro das horas viaja de uma marca para a próxima ---
        $30^\circ $ --- em $60$ minutos. Quantos graus tem
        \emph{isto} varredura por minuto? (UM \omterm{def:g6:decimals:places}{número decimal},
        \cref{ch:g6:decimals}.)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Times when nothing points at a
mark.}
\begin{enumerate}[resume]
  \item Às 3h30, o ponteiro dos minutos aponta para o $6$. Onde exatamente
        é o ponteiro das horas? Calcule o ângulo de cada mão a partir do
        $12$ (medindo no sentido horário) e deduza o ângulo entre
        as mãos.
  \item Às 6h30, muitas pessoas acham que os ponteiros estão em cima de cada um
        outro. Adivinhe primeiro e depois calcule o ângulo como na pergunta
        6. Quem estava certo?
  \item Calcule o ângulo entre os ponteiros às 9:15.
  \item Explique por que, em algum momento entre 13h e 13h10, os dois
        as mãos devem estar exatamente uma em cima da outra: onde está o
        ponteiro dos minutos em relação ao ponteiro das horas às 1:00, e onde
        às 13h10? Qual mão corre mais rápido e por que isso resolve
        isso?
  \item Em doze horas, quantas vezes as duas mãos se sentam
        exatamente um em cima do outro? (Uma reunião acontece em
        cada trecho entre horas sucessivas --- com um
        exceção: o que acontece entre 11 e 12? Liste o
        horários e contagem aproximados das reuniões.) Quantas vezes em um
        o dia inteiro?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Angles around a point: pie
charts.}
\begin{enumerate}[resume]
  \item Os doze setores do relógio face juntos preencham o
        face: seus ângulos somam $360^\circ $. Explique por que
        isso é verdade \emph{qualquer} coleção de ângulos que
        compartilhe um vértice e dar uma volta completa em torno dele, sem
        sobreposição e nenhuma lacuna.
  \item Zoe registra seu dia de $24$ horas: dormir $9$ h, escola
        $6$ h, jogue $3$ h, refeições $2$ h, todo o resto $4$ h.
        Ela quer um \emph{gráfico de pizza}: um disco onde cada atividade
        obtém um setor, com ângulos proporcionais aos tempos.
        O que fração do dia é cada atividade e quantas
        graus seu setor obtém? Verifique a soma dos cinco ângulos
        para $360^\circ $.
  \item Descreva, passo a passo, como desenhar o gráfico de pizza da Zoe com
        bússola e transferidor
        (\cref{met:g6:lines:protractor}): onde o primeiro raio
        vai, e como cada novo setor começa onde o anterior
        um termina.
  \item Em outro gráfico de pizza de um dia, um setor mede
        exatamente $90^\circ $. O que fração do disco é que,
        e quantas horas isso representa?
  \item O final, de volta ao relógio: em $24$ horas, quantas
        voltas completas o ponteiro das horas dá? O ponteiro dos minutos? O
        segunda mão? (Uma dessas respostas é superior a mil.)
\end{enumerate}
\end{problem}
