\chapter{Frações: primeiros passos}\label{ch:g6:fractions}

Dividir três pizzas de forma justa entre quatro pessoas dá a cada pessoa\,\dots\
não um número inteiro de pizzas. Frações os números foram inventados para
compartilhando. Este capítulo constrói a imagem: uma \omterm{def:g6:fractions:def}{fração} como parte de um
inteiro, como um ponto na reta numérica e como o resultado exato de uma
divisão.

\section{O que significa uma fração}

\begin{definition}[Fração]\label{def:g6:fractions:def}
Corte uma unidade em $b $ partes iguais e tome $ a $ deles: a quantidade
obtido é o \emph{fração}\index{fração}
\[
\frac{a}{b}
\qquad
\begin{array}{l}
a \text{: the \emph{numerador} --- how many parts we take;}\\
b \text{: the \emph{denominador} --- into how many parts the unit is cut.}
\end{array}
\]
\end{definition}

\begin{example}\label{ex:g6:fractions:morethanone}
UM \omterm{def:g6:fractions:def}{fração} pode ser maior que $1$: $\frac74$ significa sete \omterm{def:g3:division:half}{trimestres} ---
uma unidade inteira (quatro \omterm{def:g3:division:half}{trimestres}) e três \omterm{def:g3:division:half}{trimestres} mais:
\[
\frac74 = 1 + \frac34 .
\]
\end{example}

\begin{proposition}[Frações na reta numérica]\label{prop:g6:fractions:line}
Para colocar $\frac ab $ na reta numérica, corte cada intervalo de unidade em $ b $
partes iguais e conte $ a $ peças de $0$.
\end{proposition}

\begin{theorem}[Uma fração é uma divisão]\label{thm:g6:fractions:quotient}
O \omterm{def:g6:fractions:def}{fração} $\frac ab $ é o número que, multiplicado por $ b $, dá
$ a $:
\[
\frac ab \times b = a .
\]
Em outras palavras, $\frac ab $ é o exato \omterm{def:g3:division:remainder}{quociente} de $ a $ por $ b $ --- mesmo
quando a divisão ``não sai \omterm{def:g3:numbers:evenodd}{par}''.
\end{theorem}
\admitted

\begin{example}\label{ex:g6:fractions:pizzas}
Três pizzas para quatro pessoas: cada pessoa ganha $\frac34$ de uma pizza.
Verifique com o teorema: $4$ ações de $\frac34$ fazer
$\frac34 \times 4 = 3$ pizzas. Alguns frações são números decimais
($\frac34 = 0.75$), outros não são: $\frac13 = 0.333\dots $ nunca para.
O \omterm{def:g6:fractions:def}{fração} é o \emph{exato} valor; $0.33$ é apenas uma aproximação.
\end{example}

\section{Frações iguais}

\begin{proposition}[Frações equivalentes]\label{prop:g6:fractions:equal}
UM \omterm{def:g6:fractions:def}{fração} não muda quando é \omterm{def:g4:fractions:def}{numerador} e \omterm{def:g4:fractions:def}{denominador} são ambos
multiplicado (ou dividido) pelo mesmo número diferente de \omterm{def:g1:counting:zero}{zero}:
\[
\frac{a}{b} = \frac{a \times k}{b \times k} .
\]
\end{proposition}

\begin{proof}[Ideia de prova]
Cortando cada parte do compartilhamento em $ k $ pedaços menores multiplica ambos
o número de peças retiradas e o número total de peças por $ k $, mas
a quantidade retirada é a mesma. Uma imagem diz melhor:
\end{proof}

\end{omfigure}

\begin{method}[Simplificando uma fração]\label{met:g6:fractions:simplify}
Procure um número dividindo \omterm{def:g4:fractions:def}{numerador} \emph{e} o \omterm{def:g4:fractions:def}{denominador},
e divida ambos. Repita até que não haja comum \omterm{def:g5:division:multiple}{divisor} restos:
\[
\frac{12}{18} = \frac{12 \div 2}{18 \div 2} = \frac{6}{9}
= \frac{6 \div 3}{9 \div 3} = \frac{2}{3}.
\]
\end{method}

\section{Tomando uma fração de uma quantidade}

\begin{method}[Fração de uma quantidade]\label{met:g6:fractions:ofquantity}
Para calcular $\frac ab $ de uma quantidade, divida a quantidade por $ b $ (um
parte), depois multiplique por $ a $ (o número de peças):
\[
\frac ab \text{ of } Q = (Q \div b) \times a .
\]
(Multiplicando primeiro por $ a $ e depois dividindo por $ b $ dá o mesmo
resultado --- escolha a ordem que torna os números mais bonitos.)
\end{method}

\begin{example}\label{ex:g6:fractions:ofquantity}
Calcular $\frac35$ de $40$ euros, passo a passo:
\begin{enumerate}
  \item um quinto de $40$: $40 \div 5 = 8$;
  \item três quintos: $8 \times 3 = 24$.
\end{enumerate}
Então $\frac35$ de $40$ euros é $24$ euros. Confira: o \omterm{def:g3:division:remainder}{restante}
$\frac25$ é $16$ euros, e $24 + 16 = 40$.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g6:fractions:1}
Para cada descrição de imagem, escreva o \omterm{def:g6:fractions:def}{fração}: um bolo cortado $8$
fatias, $3$ comido; uma barra de chocolate de $12$ quadrados, $7$ comido; uma torta
cortar $6$, todos $6$ comido.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:fractions:2}
Desenhe uma barra cortada em $5$ partes iguais e cor $\frac45$ disso. É
$\frac45$ menor ou maior que $1$? E $\frac65$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:fractions:3}
Coloque em uma reta numérica graduada em terços: $\frac13$; \ $\frac53$;
\ $2$; \ $\frac73$. Quais dois desses números estão a distâncias iguais de
$2$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:fractions:4}
Escreva cada \omterm{def:g6:fractions:def}{fração} como um número inteiro mais um \omterm{def:g6:fractions:def}{fração} menor que $1$
(como em $\frac74 = 1 + \frac34$):
\[
\frac{9}{4}, \qquad \frac{13}{5}, \qquad \frac{12}{6}, \qquad
\frac{25}{8} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:fractions:5}
Complete então o frações são iguais:
\[
\frac{2}{3} = \frac{?}{12}, \qquad
\frac{15}{20} = \frac{3}{?}, \qquad
\frac{7}{10} = \frac{21}{?}, \qquad
\frac{?}{6} = \frac{10}{12}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:fractions:6}
Simplifique o máximo possível: $\dfrac{6}{8}$; \quad $\dfrac{15}{25}$;
\quad $\dfrac{18}{24}$; \quad $\dfrac{35}{7}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:fractions:7}
Calcular: $\frac12$ de $86$; \quad $\frac34$ de $60$; \quad $\frac25$ de
$35$; \quad $\frac{7}{10}$ de $250$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:fractions:8}
O que é maior: $\frac12$ ou $\frac13$ (do mesmo bolo)? $\frac25$
ou $\frac35$? $\frac34$ ou $\frac78$ (escreva ambos com \omterm{def:g4:fractions:def}{denominador}
$8$)?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:fractions:9}
Dê o valor decimal do frações que tem um e diga qual
\omterm{def:g6:fractions:def}{fração} não: $\frac12$; \ $\frac34$; \ $\frac13$; \ $\frac{7}{10}$;
\ $\frac{9}{4}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:fractions:10}
Uma aula tem $28$ estudantes. Três \omterm{def:g3:division:half}{trimestres} deles vêm para a escola
ônibus e o resto a pé. Quantos alunos caminham? Resolva passo a passo e
verifique se os tamanhos dos seus dois grupos somam $28$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:fractions:11}
Leão gastou $\frac23$ de suas economias em um jogo que custou $18$ euros. Como
quanto dinheiro Leo tinha antes? (Um terço de suas economias é \omterm{def:g3:division:half}{metade} de
$18$\,\dots\ pense no porquê ou descubra primeiro o tamanho de uma peça.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g6:fractions:12}
Um tanque é cheio até $\frac58$ do seu \omterm{def:g2:measure:units}{capacidade}. Depois de usar $10$ litros,
é \omterm{def:g3:division:half}{metade} completo. Qual é o \omterm{def:g2:measure:units}{capacidade} do tanque? (Qual \omterm{def:g6:fractions:def}{fração} de
o tanque faz o $10$ litros representam?)
\end{exercise}

\section{Problema: a barra de chocolate que você nunca consegue terminar}

\begin{problem}[{Problema de fim de semana --- metades de metades: a soma
$\frac12 + \frac14 + \frac18 + \dots $ rasteja até $1$ sem nunca
tocando-o}]
\label{pb:g6:fractions:1}
Pegue uma barra de chocolate. Comer \omterm{def:g3:division:half}{metade} disso. Então coma \omterm{def:g3:division:half}{metade} de \emph{o que
resta}. Então novamente \omterm{def:g3:division:half}{metade} do que resta, e assim por diante. Duas coisas
parece ser verdade de uma vez: você nunca termina a barra (algo sempre
permanece), e ainda assim você come quase tudo. Este problema transforma ambos
sentimentos em declarações exatas sobre frações --- e atende, em
a propósito, um corredor da Grécia antiga e um quebra-cabeça compartilhado resolvido
inteiramente com uma faca que só corta coisas \omterm{def:g3:division:half}{metade}.

\medskip
\noindent\textbf{Part I --- Halves of halves.}
A barra é um retângulo de $16$ quadrados iguais.
\begin{enumerate}
  \item Primeira mordida: você come \omterm{def:g3:division:half}{metade} a barra. Quantos quadrados tem
        isso? Escreva a mordida como um \omterm{def:g6:fractions:def}{fração} da barra de duas maneiras,
        com \omterm{def:g4:fractions:def}{denominador} $2$ e com \omterm{def:g4:fractions:def}{denominador} $16$
        (\cref{prop:g6:fractions:equal}).
  \item Segunda mordida: \omterm{def:g3:division:half}{metade} do que resta. Quantos quadrados? Mostrar,
        com a imagem ou com equivalente frações, que isso
        mordida é $\frac14$ de toda a barra: \emph{\omterm{def:g3:division:half}{metade} de um \omterm{def:g3:division:half}{metade}
        é um \omterm{def:g3:division:half}{trimestre}}.
  \item Terceira e quarta mordidas, mesma regra: dar cada uma
        quadrados e como \omterm{def:g6:fractions:def}{fração} da barra. O que acontece com o
        \omterm{def:g4:fractions:def}{denominador} de uma mordida para outra?
  \item Após as quatro mordidas, quantos quadrados foram comidos em
        tudo? O que \omterm{def:g6:fractions:def}{fração} do bar é isso, e o que \omterm{def:g6:fractions:def}{fração}
        resta?
  \item A questão 4 diz, escrita em décimos sextos:
        \[
        \frac{8}{16} + \frac{4}{16} + \frac{2}{16} + \frac{1}{16}
        = \frac{15}{16},
        \qquad\text{that is}\qquad
        \frac12 + \frac14 + \frac18 + \frac{1}{16}
        = \frac{15}{16} .
        \]
        Confira cada uma das quatro reescritas
        ($\frac12 = \frac{8}{16}$, e assim por diante).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The staircase towards $1$.}
Para continuar mordendo, imagine uma barra mais fina de $64$ quadrados.
\begin{enumerate}[resume]
  \item Liste os tamanhos das primeiras seis mordidas, em quadrados
        ($32$, então\,\dots), e verifique se após a sexta mordida
        resta exatamente um quadrado. O que \omterm{def:g6:fractions:def}{fração} da barra
        permanece?
  \item Depois de cada mordida, o que \omterm{def:g6:fractions:def}{fração} da barra permanece? Escreva
        a lista ($\frac12$, $\frac14$, \dots) até o sexto
        morda e descreva o padrão. Se uma faca imaginária
        poderia reduzir pela \omterm{def:g3:division:half}{metade} \omterm{def:g3:numbers:evenodd}{par} o último quadrado, o que restaria depois
        uma sétima mordida?
  \item Explique por que o total consumido pode \emph{nunca} alcançar $1$,
        não importa quantas mordidas sejam dadas. (Compare com o
        vinte noves de \cref{pb:g6:decimals:1}: mesma história?)
  \item No entanto, o total ultrapassa qualquer meta abaixo $1$. Um amigo
        desafia você a comer mais do que $\frac{99}{100}$ do
        barra. Qual é menor, $\frac{1}{128}$ ou
        $\frac{1}{100}$ --- e depois disso a mordida é o desafio
        ganhou?
  \item Em uma reta numérica de $0$ para $1$ graduou-se em décimo sexto,
        coloque os totais consumidos após cada um dos primeiros quatro
        mordidas: $\frac12$, $\frac34$, $\frac78$, $\frac{15}{16}$.
        Onde fica cada nova terra total, em comparação com o
        anterior e com $1$?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Zeno's runner, and a knife that
only halves.}
\begin{enumerate}[resume]
  \item Há vinte e quatro séculos, o filósofo grego Zenão
        contou esta história: para alcançar uma parede, um corredor deve primeiro
        capa \omterm{def:g3:division:half}{metade} a distância, então \omterm{def:g3:division:half}{metade} do que resta, então
        \omterm{def:g3:division:half}{metade} do que resta novamente\,\dots\ ``então o corredor nunca
        atinge a parede!'' Usando as questões 8 e 9, diga o que é
        \emph{certo} na história de Zenão e onde está a armadilha.
        (Será que o corredor realmente precisa de uma eternidade para fazer todos aqueles
        passos?)
  \item O quebra-cabeça do compartilhamento: compartilhe $15$ barras de chocolate idênticas
        razoavelmente entre $16$ crianças, usando apenas cortes ao meio
        (de uma barra, de uma meia barra, de um quarto de barra\,\dots).
        Descreva como cortar: quantas barras são cortadas ao meio,
        quantos em \omterm{def:g3:division:half}{trimestres}, em colcheias, em semicolcheias,
        para que cada criança receba uma peça de cada tamanho.
  \item Verifique seu compartilhamento: escreva o compartilhamento de cada criança como um \omterm{def:g1:addition:def}{soma} de
        frações, calcule-o em dezesseis décimos e verifique se o
        $16$ ações juntas consomem exatamente o $15$ bares.
  \item Na aula ao lado, $7$ as barras são compartilhadas de forma justa entre
        $8$ crianças. Qual criança recebe mais chocolate --- um
        do $16$, ou um dos $8$? (Escreva ambas as ações com
        \omterm{def:g4:fractions:def}{denominador} $16$, \cref{exo:g6:fractions:8}.)
  \item O final: um amigo afirma que com \emph{infinitamente}
        muitas mordidas que alguém comeria \emph{exatamente} toda a barra.
        Desenhe um quadrado e sombreie as mordidas dentro dele: \omterm{def:g3:division:half}{metade} o
        quadrado, então um \omterm{def:g3:division:half}{trimestre}, então um oitavo\,\dots\ O que fazer
        você observa sobre o canto sem sombra? Declare com cuidado
        os dois factos verdadeiros de todo este problema: o que resta
        após cada número finito de mordidas, e quão pequeno isso
        \omterm{def:g3:division:remainder}{restante} torna-se. (Que ``infinito \omterm{def:g1:addition:def}{soma}'' exatamente
        significa é uma história para o \omterm{def:g5:solids:volume}{volume} do Ensino Médio.)
\end{enumerate}
\end{problem}
