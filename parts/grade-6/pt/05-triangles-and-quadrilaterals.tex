\chapter{Triângulos e Quadriláteros}\label{ch:g6:shapes}

Triângulos e \omterm{def:g4:geometry:polygon}{quadriláteros} são as formas básicas da geometria. O objetivo
deste capítulo é conhecer suas famílias especiais --- \omterm{def:g6:shapes:triangles}{isósceles},
\omterm{def:g6:shapes:triangles}{equilátero}, triângulos retângulos; retângulos, losangos, quadrados --- por
seu \emph{definindo propriedades}, e construí-los precisamente com
régua e compasso.

\section{Triângulos}

\begin{definition}[Triângulos especiais]\label{def:g6:shapes:triangles}
Um triângulo $ABC $ tem três \omterm{def:g5:solids:def}{vértices} e três lados. Isso é:
\begin{itemize}
  \item \emph{isósceles}\index{isósceles} no $ A $ quando
        $ AB = AC $ (dois lados iguais);
  \item \emph{equilátero}\index{equilátero} quando
        $ AB = BC = CA $ (três lados iguais);
  \item \emph{em ângulo reto}\index{em ângulo reto} no $ A $ quando o ângulo
        $\widehat{BAC}$ é um \omterm{def:g3:shapes:rightangle}{ângulo reto}. O lado $[BC]$ oposto ao
        \omterm{def:g3:shapes:rightangle}{ângulo reto} é chamado de \emph{hipotenusa}.
\end{itemize}
\end{definition}

\begin{method}[Construindo um triângulo a partir de seus três lados]\label{met:g6:shapes:construct}
Para construir $ ABC $ com $ AB = 5$ cm, $ AC = 4$ cm, $ BC = 3$ cm:
\begin{enumerate}
  \item desenhe o \omterm{def:g6:lines:objects}{segmento} $[AB]$ de $5$ cm com a régua;
  \item desenhe um arco do círculo do centro $ A $ e \omterm{def:g3:shapes:shapes}{raio} $4$ centímetros
        (todos os pontos em $4$ cm de $ A $);
  \item desenhe um arco do círculo do centro $ B $ e \omterm{def:g3:shapes:shapes}{raio} $3$ cm;
  \item o ponto $ C $ é onde os dois arcos se cruzam; junte-se a isso $ A $ e
        $ B $.
\end{enumerate}
\end{method}

\section{Quadriláteros}

\begin{definition}[Quadriláteros especiais]\label{def:g6:shapes:quadrilaterals}
UM \omterm{def:g4:geometry:polygon}{quadrilátero} tem quatro \omterm{def:g5:solids:def}{vértices} e quatro lados. É um:
\begin{itemize}
  \item \emph{retângulo}\index{retângulo} quando seus quatro ângulos são
        ângulos retos;
  \item \emph{Losango}\index{Losango} quando seus quatro lados têm o mesmo
        comprimento;
  \item \emph{quadrado}\index{quadrado} quando são ambos: quatro ângulos retos
        \emph{e} quatro lados iguais.
\end{itemize}
\end{definition}

\begin{proposition}[Propriedades das diagonais]\label{prop:g6:shapes:diagonals}
Nestes \omterm{def:g4:geometry:polygon}{quadriláteros}, as duas diagonais se cortam em seus
ponto médio comum e, além disso:
\begin{itemize}
  \item em um retângulo, as diagonais têm o \emph{mesmo comprimento};
  \item em um losango, as diagonais são \emph{\omterm{def:g6:lines:perp}{perpendicular}};
  \item em um quadrado, ambos ao mesmo tempo.
\end{itemize}
\end{proposition}
\admitted

\begin{remark}
Um quadrado é automaticamente um retângulo (possui quatro ângulos retos) e
automaticamente um losango (quatro lados iguais). Então ``retângulo'' não
significa ``não é um quadrado''! Em matemática, a forma mais especial pertence
para a família mais geral.
\end{remark}

\begin{example}[Reconhecendo uma forma a partir de suas propriedades]\label{ex:g6:shapes:recognize}
UM \omterm{def:g4:geometry:polygon}{quadrilátero} tem quatro lados de comprimento $4$ cm e um \omterm{def:g3:shapes:rightangle}{ângulo reto}.
O que é? Quatro lados iguais fazem dele um losango. Em um losango, oposto
os ângulos são iguais e a \omterm{def:g1:addition:def}{soma} dos quatro ângulos resulta $360^\circ $; um certo
ângulo então força todos os quatro a estarem certos (isso será provado com
paralelogramos em \cref{ch:g7:central}). Portanto a forma é um quadrado.
\end{example}

\begin{method}[Programas de construção]\label{met:g6:shapes:program}
UM \emph{programa de construção} é uma lista de instruções que qualquer um pode
siga para reproduzir uma figura com exatidão. Escreva uma instrução por \omterm{def:g6:lines:objects}{linha},
nomeie cada ponto quando ele for criado e use apenas: ``desenhe o \omterm{def:g6:lines:objects}{segmento}
/ \omterm{def:g6:lines:objects}{linha} / círculo\dots'', ``coloque o ponto onde \dots\ cruzar''. Teste
seu programa seguindo-o literalmente, palavra por palavra.
\end{method}

\begin{example}\label{ex:g6:shapes:program}
Programa para um quadrado $ ABCD $ de lado $4$ cm:
\begin{enumerate}
  \item Desenhe um \omterm{def:g6:lines:objects}{segmento} $[AB]$ com $ AB = 4$ cm.
  \item Desenhe o \omterm{def:g6:lines:perp}{perpendicular} para $(AB)$ no $ A $; nele, coloque $ D $ no
        $4$ cm de $ A $ (escolha um lado da \omterm{def:g6:lines:objects}{linha} e cumpra-o).
  \item Desenhe o \omterm{def:g6:lines:perp}{perpendicular} para $(AB)$ no $ B $; nele, coloque $ C $ no
        $4$ cm de $ B $, do mesmo lado que $ D $.
  \item Juntar $ C $ e $ D $.
\end{enumerate}
Verifique com a régua: $ DC $ deveria medir $4$ cm também.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g6:shapes:1}
Construa um triângulo $ ABC $ com $ AB = 6$ cm, $ AC = 5$ centímetros e
$ BC = 4$ centímetros, seguindo \cref{met:g6:shapes:construct}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:shapes:2}
Construa um triângulo \omterm{def:g6:shapes:triangles}{isósceles} $ DEF $ com $ DE = DF = 5$ centímetros e
$ EF = 3$ cm. Quais são os dois ângulos deste triângulo que parecem iguais? Meça-os
para verificar.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:shapes:3}
Construa um triângulo \omterm{def:g6:shapes:triangles}{equilátero} de lado $4$ cm usando apenas régua e
bússola. Meça um de seus ângulos.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:shapes:4}
Classifique cada triângulo a partir de seus dados (\omterm{def:g6:shapes:triangles}{isósceles}, \omterm{def:g6:shapes:triangles}{equilátero},
\omterm{def:g6:shapes:triangles}{em ângulo reto} --- várias respostas podem ser aplicadas):
(a) lados $7$, $7$, $7$; \
(b) lados $5$, $5$, $8$; \
(c) ângulos $90^\circ $, $45^\circ $, $45^\circ $; \
(d) ângulos $60^\circ $, $60^\circ $, $60^\circ $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:shapes:5}
Em um triângulo retângulo, de que lado está a hipotenusa? Desenhe um triângulo
$ MNP $ \omterm{def:g6:shapes:triangles}{em ângulo reto} no $ N $ e nomeie sua hipotenusa. Qual vértice faz
a hipotenusa nunca toca?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:shapes:6}
Verdadeiro ou falso? Justifique cada resposta em uma frase.
\begin{enumerate}
  \item Todo quadrado é um retângulo.
  \item Todo retângulo é um quadrado.
  \item Cada quadrado é um losango.
  \item UM losango com um \omterm{def:g3:shapes:rightangle}{ângulo reto} é um quadrado.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:shapes:7}
Construa um retângulo $ ABCD $ com $ AB = 5$ centímetros e $ BC = 3$ cm. Medir
suas duas diagonais: o que você observa?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:shapes:8}
Construa um losango com diagonais de $6$ centímetros e $4$ cm. (Usar
\cref{prop:g6:shapes:diagonals}: desenhe as duas diagonais primeiro ---
\omterm{def:g6:lines:perp}{perpendicular}, cruzando em seus pontos médios --- então junte-se aos quatro
termina.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:shapes:9}
Escreva um programa de construção para um triângulo \omterm{def:g6:shapes:triangles}{isósceles} com base
$4$ cm e lados iguais de $6$ cm, no estilo de
\cref{ex:g6:shapes:program}. Troque com um colega de turma: ele consegue acompanhar
sem ver sua figura?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:shapes:10}
Desenhe um círculo de centro $ O $ e um de seus \omterm{def:g4:geometry:circle}{diâmetros} $[AB]$. Coloque qualquer
ponto $ M $ no círculo e desenhe o triângulo $ AMB $. Meça o ângulo
$\widehat{AMB}$. Mover $ M $ e meça novamente. O que você conjectura?
(Este fato surpreendente é provado em \cref{ch:g8:cosine}.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:shapes:11}
UM \omterm{def:g4:geometry:polygon}{quadrilátero} tem diagonais de mesmo comprimento que se cortam em
seu ponto médio comum. Classifique qual formato ele deve ter, usando
\cref{prop:g6:shapes:diagonals}: retângulo, losango ou quadrado?
(Cuidado: apenas uma propriedade é fornecida.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g6:shapes:12}
Em um $4 \times 4$ grade pontilhada, quantos quadrados diferentes você pode desenhar
cujos cantos são pontos da grade --- contando também os quadrados ``inclinados''?
Desenhe pelo menos um quadrado inclinado e explique por que seus lados são iguais.
\end{exercise}

\section{Problema: O que as diagonais sabem}

\begin{problem}[{Problema de fim de semana --- reconstruindo todos os especiais
quadrilátero apenas a partir de suas diagonais e contando as diagonais
de qualquer polígono}]
\label{pb:g6:shapes:1}
\Cref{prop:g6:shapes:diagonals} lê uma forma e fala sobre
suas diagonais. Este problema joga o jogo \emph{para trás}: dado
apenas duas travessias \omterm{def:g6:lines:objects}{segmentos} --- as futuras diagonais --- qual a forma
aparece quando você junta suas quatro pontas? Você construirá um completo
``dicionário diagonal'', entenda \emph{por que} o losango
construção de \cref{exo:g6:shapes:8} funciona, conheça uma forma que
ainda não tem nome e termine contando as diagonais de \omterm{def:g4:geometry:polygon}{polígonos}
com muito mais de quatro lados.

\medskip
\noindent\textbf{Part I --- The diagonal dictionary.}
Para cada pergunta, desenhe os dois \omterm{def:g6:lines:objects}{segmentos} conforme descrito, junte-se a eles
quatro extremidades em ordem e identifique o \omterm{def:g4:geometry:polygon}{quadrilátero} você consegue.
\begin{enumerate}
  \item Dois \omterm{def:g6:lines:objects}{segmentos} de $6$ cm cada, \omterm{def:g6:lines:perp}{perpendicular}, cruzando em
        seu ponto médio comum. Meça os quatro lados e o
        quatro ângulos da forma resultante: o que é?
  \item Dois \omterm{def:g6:lines:objects}{segmentos} de $8$ centímetros e $5$ cm, \omterm{def:g6:lines:perp}{perpendicular},
        cruzando em seu ponto médio comum. Que forma aparece
        (\cref{exo:g6:shapes:8})?
  \item Dois \omterm{def:g6:lines:objects}{segmentos} de $7$ cm cada --- igual, mas \emph{não}
        \omterm{def:g6:lines:perp}{perpendicular} --- cruzando em seu ponto médio comum.
        Meça os lados e ângulos: o que você conjetura
        forma para ser?
  \item Dois \omterm{def:g6:lines:objects}{segmentos} de $8$ centímetros e $5$ centímetros, não \omterm{def:g6:lines:perp}{perpendicular},
        cruzando em seu ponto médio comum. A forma que você obtém é
        nenhum dos três
        \cref{def:g6:shapes:quadrilaterals} --- descreva o que
        você vê (quais lados parecem iguais? \omterm{def:g6:lines:perp}{paralelo}?). Esse
        \omterm{def:g4:geometry:polygon}{quadrilátero} recebe seu nome em \cref{ch:g7:central}.
  \item Resuma as questões 1 a 4 em um dicionário com quatro
        entradas: diagonais \emph{igual / \omterm{def:g6:lines:perp}{perpendicular} /ambos/
        nem} (sempre cruzando em seus pontos médios)
        $\rightarrow $ nome (ou descrição) da forma.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Why the constructions work.}
\begin{enumerate}[resume]
  \item Na questão 2, as duas diagonais cortam a forma em quatro
        triângulos retângulos em torno do ponto de passagem. Quais são os
        dois comprimentos de perna de cada um desses triângulos? Explique por que
        todos os quatro triângulos são cópias exatas um do outro, e
        deduzir por que os quatro lados da forma são iguais --- o
        segredo de \cref{exo:g6:shapes:8}.
  \item Deduza um teste apenas com bússola: depois de construir qualquer forma
        de suas diagonais, como você pode verificar com a bússola
        sozinho (sem medidas de régua) que os quatro lados são
        igual?
  \item Escreva um programa de construção
        (\cref{met:g6:shapes:program}) para um quadrado cujo
        \emph{diagonais} medir $6$ cm --- não seus lados.
  \item No quadrado da questão 8, o lado é maior ou
        mais curto que $3$ cm --- \omterm{def:g3:division:half}{metade} a diagonal? Medir para
        descubra. Um \omterm{def:g3:division:half}{metade} da resposta já pode ser
        \emph{comprovado}: indo de um canto ao outro através do
        centro é um $3 + 3$ cm desvio, e um caminho reto é
        sempre mais curto que um desvio --- então o lado é mais curto
        do que $6$ cm. (Provando que também bate $3$ cm, e
        computá-lo exatamente, é um trabalho para
        \cref{ch:g8:pythagoras}.)
  \item Uma nova entrada para o dicionário: desenhe um \omterm{def:g6:lines:objects}{segmento} $[AC]$ de
        $8$ cm; nele, marque o ponto $ H $ no $3$ cm de $ A $;
        desenhar $ H $ o \omterm{def:g6:lines:perp}{perpendicular} \omterm{def:g6:lines:objects}{segmento} $[BD]$ de
        $5$ centímetros com $ H $ como \emph{isso é} ponto médio. Juntar $ ABCD $. Isto
        forma é um \emph{pipa}\index{pipa} (uma diagonal divide
        o outro, em um \omterm{def:g3:shapes:rightangle}{ângulo reto}). Usando triângulos de canto como
        na questão 6, descubra quais lados de um \omterm{pb:g6:shapes:1}{pipa} são iguais a
        qual.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Counting diagonals.}
UM \emph{diagonal} de um \omterm{def:g4:geometry:polygon}{polígono} junta-se a dois \omterm{def:g5:solids:def}{vértices} que não são
vizinhos.
\begin{enumerate}[resume]
  \item Desenhe um \omterm{def:g4:geometry:polygon}{quadrilátero}, um \omterm{def:g4:geometry:polygon}{Pentágono} (cinco \omterm{def:g5:solids:def}{vértices}) e um
        \omterm{def:g4:geometry:polygon}{hexágono} (seis \omterm{def:g5:solids:def}{vértices}) e desenhe todas as suas diagonais.
        Conte-os: quantos em cada caso?
  \item Contar sem desenhar, para o \omterm{def:g4:geometry:polygon}{hexágono}: de um
        vértice, quantas diagonais restam? (Para quantos
        seis \omterm{def:g5:solids:def}{vértices} pode \emph{não} ser unidos por uma diagonal?)
        Multiplicando por seis \omterm{def:g5:solids:def}{vértices} conta cada diagonal
        exatamente duas vezes --- por quê? Recupere sua contagem de perguntas
        11. (A mesma contagem dupla apareceu nos apertos de mão
        de \cref{pb:g6:wholes:1}.)
  \item Use o método da questão 12 para contar as diagonais de
        um \omterm{def:g4:geometry:polygon}{polígono} com $10$ \omterm{def:g5:solids:def}{vértices}, então com $20$ \omterm{def:g5:solids:def}{vértices}.
  \item UM \omterm{def:g4:geometry:polygon}{polígono} tem exatamente $44$ diagonais. Quantos \omterm{def:g5:solids:def}{vértices}
        isso tem? (Experimente o método da pergunta 13 em alguns
        candidatos.)
  \item Verifique seu método no menor \omterm{def:g4:geometry:polygon}{polígono} de tudo: como
        quantas diagonais um triângulo tem, e qual é a
        método de contagem da resposta da pergunta 12? Conclua com um
        frase sobre o que apertos de mão e diagonais têm em
        comum.
\end{enumerate}
\end{problem}
