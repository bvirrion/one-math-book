\chapter{Números inteiros}\label{ch:g6:wholes}

Tudo em matemática começa com a contagem. Este capítulo analisa como
números inteiros são escritos, comparados e combinados --- e é preciso um cuidado
veja a divisão, a operação que dá a ambos um \omterm{def:g3:division:remainder}{quociente} \emph{e} um
\omterm{def:g3:division:remainder}{restante}.

\section{Escrevendo e comparando números inteiros}

\begin{definition}[Valor posicional]\label{def:g6:wholes:place}
Nossa maneira de escrever números usa dez dígitos ($0$ para $9$) e o
\emph{o valor de um dígito depende de sua posição}\index{o valor de um dígito depende de sua posição}: em
$5\,208$, o dígito $5$ conta milhares, o $2$ conta centenas, o
$0$ conta dezenas (não há nenhuma) e o $8$ conta unidades:
\[
5\,208 = 5 \times 1000 + 2 \times 100 + 0 \times 10 + 8 .
\]
\end{definition}

\begin{method}[Comparando dois números inteiros]\label{met:g6:wholes:compare}
\begin{enumerate}
  \item O número com mais algarismos é o maior
        ($1\,203 > 989$).
  \item Se eles tiverem o mesmo número de dígitos, compare dígito por dígito
        do \emph{esquerda}; a primeira \omterm{ex:g1:subtraction:difference}{diferença} decide:
        $6\,742 > 6\,715$ porque na casa das dezenas, $4 > 1$.
\end{enumerate}
Os símbolos são $<$ (``menos que'') e $>$ (``maior que''); o
a extremidade menor do símbolo aponta para o número pequeno.
\end{method}

\begin{example}\label{ex:g6:wholes:order}
Ordene do menor para o maior: $908$; $89$; $1\,001$; $980$.
Primeiro por contagem de dígitos: $89$ (dois dígitos) vem primeiro, $1\,001$ (quatro
dígitos) por último. Entre $908$ e $980$: mesmo dígito das centenas $9$, então
dezenas $0 < 8$, então $908 < 980$. Resposta final:
\[
89 < 908 < 980 < 1\,001 .
\]
\end{example}

\section{Adição, subtração, multiplicação}

\begin{example}[Cálculos de coluna]\label{ex:g6:wholes:columns}
Para somar ou subtrair, alinhe as unidades sob as unidades, as dezenas sob as
dezenas, e trabalhe da direita para a esquerda, carregando quando necessário:
\[
\begin{array}{r}
4\,5\,7 \\
+\ 2\,8\,6 \\
\hline
7\,4\,3
\end{array}
\qquad\qquad
\begin{array}{r}
6\,0\,3 \\
-\ 1\,4\,7 \\
\hline
4\,5\,6
\end{array}
\]
Além disso: $7 + 6 = 13$, escrever $3$, carregar $1$; então
$5 + 8 + 1 = 14$, escrever $4$, carregar $1$; então $4 + 2 + 1 = 7$.
Verifique a subtração adicionando novamente: $456 + 147 = 603$.
\end{example}

\begin{remark}[Vocabulário]
O resultado de uma adição é um \emph{soma}\index{soma}; de uma subtração,
um \emph{diferença}\index{diferença}; de uma multiplicação, um
\emph{produto}\index{produto}; de uma divisão, um
\emph{quociente}\index{quociente}. Os números que estão sendo combinados são os
\emph{termos} (para $+$, $-$) ou \emph{fatores} (para $\times $).
\end{remark}

\begin{example}[Computação inteligente]\label{ex:g6:wholes:smart}
Reordenar termos ou fatores geralmente economiza trabalho:
\[
17 + 58 + 3 = (17 + 3) + 58 = 20 + 58 = 78 ,
\]
\[
25 \times 17 \times 4 = (25 \times 4) \times 17 = 100 \times 17 = 1700 .
\]
\end{example}

\section{Divisão}

\begin{theorem}[Divisão euclidiana]\label{thm:g6:wholes:division}
Dados dois números inteiros $ a $ (o \emph{dividendo}) e $ b \neq 0$ (o
\emph{\omterm{def:g5:division:multiple}{divisor}}), existe exatamente uma maneira de escrever
\[
a = b \times q + r
\qquad\text{with}\qquad
0 \leq r < b .
\]
O número $ q $ é o \emph{quociente}\index{quociente} e $ r $ o
\emph{restante}\index{restante} da divisão de $ a $ por $ b $.
\end{theorem}
\admitted

\begin{example}\label{ex:g6:wholes:division}
Dividir $137$ por $4$, passo a passo.
\begin{enumerate}
  \item Quantas vezes $4$ entrar $13$? Três vezes
        ($4 \times 3 = 12$), \omterm{def:g3:division:remainder}{restante} $13 - 12 = 1$.
  \item Derrubar o $7$: quantas vezes $4$ entrar $17$?
        Quatro vezes ($4 \times 4 = 16$), \omterm{def:g3:division:remainder}{restante} $1$.
\end{enumerate}
Então $ q = 34$ e $ r = 1$:
\[
137 = 4 \times 34 + 1, \qquad\text{and indeed } 0 \leq 1 < 4 .
\]
Verificar: $4 \times 34 = 136$, e $136 + 1 = 137$.
\end{example}

\begin{definition}[Divisível]\label{def:g6:wholes:divisible}
Quando o \omterm{def:g3:division:remainder}{restante} é $0$, dizemos que $ a $ é
\emph{divisível}\index{divisível} por $ b $: por exemplo $135$ é
divisível por $5$ desde $135 = 5 \times 27 + 0$.
\end{definition}

\begin{method}[Problemas de palavras com divisão]\label{met:g6:wholes:problems}
Depois de computar $ a = bq + r $, sempre retorne à pergunta:
\begin{enumerate}
  \item ``Quantas \omterm{def:g5:solids:def}{caixas}/equipes/páginas completas?'' --- a resposta é a
        \omterm{def:g3:division:remainder}{quociente} $ q $;
  \item ``Quantos sobraram?'' --- a resposta é a \omterm{def:g3:division:remainder}{restante}
        $ r $;
  \item ``Quantas \omterm{def:g5:solids:def}{caixas} são necessárias para todos?'' --- se $ r > 0$, um
        \emph{mais} do que o \omterm{def:g3:division:remainder}{quociente}: $ q + 1$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g6:wholes:bus}
$137$ estudantes viajam em ônibus de $40$ assentos. Divisão:
$137 = 40 \times 3 + 17$. Três ônibus transportam $120$ estudantes, e $17$
os alunos permanecem --- eles também precisam de um ônibus! Então $4$ são necessários ônibus, mesmo
embora o \omterm{def:g3:division:remainder}{quociente} é $3$.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g6:wholes:1}
Escreva em algarismos: ``três mil e quarenta e sete''; ``vinte mil
quinhentos''; ``cento e quatro''. Então escreva $60\,013$ em palavras.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:wholes:2}
No número $47\,209$: qual é o algarismo das centenas? O dígito de
as unidades? O que o dígito $4$ contar? Escreva o número como \omterm{def:g1:addition:def}{soma} de
\omterm{def:g5:division:multiple}{múltiplos} de $1$, $10$, $100$, \dots\ como em
\cref{def:g6:wholes:place}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:wholes:3}
Copie e complete com $<$ ou $>$:
\[
507 \;?\; 570, \qquad
1\,099 \;?\; 989, \qquad
23\,456 \;?\; 23\,465, \qquad
9\,999 \;?\; 10\,000 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:wholes:4}
Ordene do menor para o maior: $2\,043$; $978$; $2\,304$; $2\,034$;
$1\,001$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:wholes:5}
Calcule em colunas: $348 + 576$; \quad $805 - 268$; \quad
$307 \times 26$. Verifique a subtração com uma adição.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:wholes:6}
Calcule de forma inteligente, agrupando termos ou fatores:
\[
38 + 45 + 12 + 5, \qquad
2 \times 83 \times 50, \qquad
4 \times 78 \times 25 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:wholes:7}
Para cada divisão, dê o \omterm{def:g3:division:remainder}{quociente} e o \omterm{def:g3:division:remainder}{restante}, e escreva o
igualdade $ a = b \times q + r $:
\[
95 \div 7, \qquad
230 \div 6, \qquad
504 \div 8 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:wholes:8}
Sem dividir, explique por que a igualdade $173 = 8 \times 20 + 13$ é
\emph{não} a divisão euclidiana de $173$ por $8$, e encontre o correto
\omterm{def:g3:division:remainder}{quociente} e \omterm{def:g3:division:remainder}{restante}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:wholes:9}
Os ovos são embalados em \omterm{def:g5:solids:def}{caixas} de $12$.
\begin{enumerate}
  \item Uma fazenda tem $2\,000$ ovos. Quantas \omterm{def:g5:solids:def}{caixas} cheias ele pode preencher e
        quantos ovos sobraram?
  \item Uma padaria precisa $150$ ovos. Quantas \omterm{def:g5:solids:def}{caixas} ele deve comprar?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:wholes:10}
Hoje é terça-feira. Em que dia da semana será $100$ dias?
(Semanas têm $7$ dias: divida e pense no \omterm{def:g3:division:remainder}{restante}.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:wholes:11}
Numa divisão por $9$, o \omterm{def:g3:division:remainder}{quociente} é $56$ e o \omterm{def:g3:division:remainder}{restante} é o
maior que possa ser. Qual é o dividendo?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g6:wholes:12}
Eu sou um número de três dígitos. Meu algarismo das centenas é o dobro do meu algarismo das unidades,
meu algarismo das dezenas é $0$, e o \omterm{def:g1:addition:def}{soma} dos meus dígitos é $9$. Quem sou eu?
(Raciocine passo a passo e depois verifique.)
\end{exercise}

\section{Problema: Little Gauss e a soma dos primeiros cem
números}

\begin{problem}[{Problema de fim de semana --- adicionando $1 + 2 + 3 + \dots +
100$ em dez segundos, e os números triangulares}]
\label{pb:g6:wholes:1}
Uma história famosa: para manter a turma ocupada, um professor pediu ao seu
os alunos devem somar todos os números de $1$ para $100$. Um menino escreveu
$5\,050$ em sua lousa quase imediatamente --- Carl Friedrich Gauss, nove
anos de idade, que se tornaria um dos maiores matemáticos
de todos os tempos. Seu segredo nada mais era do que \emph{computação inteligente}
(\cref{ex:g6:wholes:smart}): agrupando os termos para que cada grupo
é fácil. Este problema redescobre o seu truque, faz com que funcione em todos os
situação, e mostra com que frequência estes \omterm{def:g1:addition:def}{somas} aparecem na vida diária.

\medskip
\noindent\textbf{Part I --- Gauss's pairs.}
\begin{enumerate}
  \item Calcular $1 + 2 + 3 + 4 + 5$ agrupando termos de forma inteligente.
  \item Para $1 + 2 + 3 + \dots + 10$: emparelhe o primeiro termo com o
        último ($1 + 10$), o segundo com o penúltimo
        ($2 + 9$) e assim por diante. Quanto vale cada \omterm{def:g3:numbers:evenodd}{par}? Quantos
        \omterm{def:g3:numbers:evenodd}{pares} existem? Conclua o cálculo.
  \item Mesmo método para $1 + 2 + \dots + 20$: o que é cada \omterm{def:g3:numbers:evenodd}{par}
        vale a pena, quantos \omterm{def:g3:numbers:evenodd}{pares}, qual o total?
  \item Agora o professor \omterm{def:g1:addition:def}{soma}, $1 + 2 + \dots + 100$: explicar
        por que os \omterm{def:g3:numbers:evenodd}{pares} $1 + 100$, $2 + 99$, $3 + 98, \dots $ são todos
        vale o mesmo, e por que existem exatamente $50$ deles,
        sendo o último $50 + 51$.
  \item Escreva a resposta de Gauss como uma única multiplicação e
        calcule-o.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- A trick that never fails.}
\begin{enumerate}[resume]
  \item Experimente o emparelhamento $1 + 2 + \dots + 9$: desta vez um
        número fica sem parceiro. Qual deles? \omterm{def:g3:numbers:evenodd}{Par} ao redor
        e calcule o \omterm{def:g1:addition:def}{soma}.
  \item Aqui está uma versão do truque que funciona quer o
        contagem de termos é \omterm{def:g3:numbers:evenodd}{par} ou \omterm{def:g3:numbers:evenodd}{ímpar}: escreva o \omterm{def:g1:addition:def}{soma} duas vezes, uma vez
        para frente e uma vez para trás, uma linha abaixo da outra,
        \[
        \begin{array}{ccccccc}
        1 & + & 2 & + & \dots & + & 9 \\
        9 & + & 8 & + & \dots & + & 1
        \end{array}
        \]
        e adicione \emph{coluna por coluna}. O que é cada coluna
        vale a pena? Quantas colunas existem? Explique por que \omterm{def:g1:addition:def}{soma} de
        as duas linhas são $9 \times 10 = 90$, e concluir que
        $1 + 2 + \dots + 9 = 45$ --- a mesma resposta que em
        questão 6.
  \item Use o truque das duas linhas para calcular $1 + 2 + \dots + 50$,
        então $1 + 2 + \dots + 200$.
  \item Calcule o \omterm{def:g1:addition:def}{soma} dos primeiros cem \emph{\omterm{def:g3:numbers:evenodd}{par}} números,
        $2 + 4 + 6 + \dots + 200$. (Compare cada termo com um termo
        de Gauss \omterm{def:g1:addition:def}{soma}: nenhum novo emparelhamento é necessário.)
  \item Calcular $101 + 102 + \dots + 200$, usando dois \omterm{def:g1:addition:def}{somas} você
        já sabe e uma subtração.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- \omterm{pb:g6:wholes:1}{Números triangulares} everywhere.}
Os totais $1$, $3$, $6$, $10$, $15, \dots $ do \omterm{def:g1:addition:def}{somas}
$1 + 2 + \dots + n $ são chamados \emph{triangular
números}\index{triangular
números}, porque eles contam objetos
empilhados em um triângulo.
\begin{enumerate}[resume]
  \item Oito amigos se encontram e cada um aperta a mão exatamente uma vez
        com cada um dos outros. Explique por que o número de
        apertos de mão é $7 + 6 + 5 + 4 + 3 + 2 + 1$ e calcule-o.
  \item Um dono da mercearia constrói um triângulo \omterm{def:g5:solids:def}{pirâmide} de latas: $12$ latas em
        a linha inferior, $11$ no próximo, e assim por diante até um único
        pode por cima. De quantas latas ela precisa? (Use o truque de
        questão 7.)
  \item Desenhe o \omterm{def:g1:addition:def}{soma} $1 + 2 + 3 + 4$ como uma escada de quadrados:
        um quadrado na primeira coluna, dois na segunda, três
        no terceiro, quatro no quarto. Mostre no seu desenho
        isso \emph{dois} tais escadas, uma delas virou de cabeça
        para baixo, encaixem-se exatamente em um retângulo de $4$ linhas de
        $5$ quadrados --- e explique por que esse é o truque das duas linhas
        da questão 7, desenhada em vez de escrita.
  \item Um relógio bate $1$ hora à uma hora, $2$ vezes às duas
        horas, \dots, $12$ vezes às doze horas. Quantos
        greves é isso ao longo das doze horas? E ao longo de um todo
        dia?
  \item Uma última para Gauss: $5\,050$ segundos --- isso é mais ou
        menos de uma hora e uma \omterm{def:g3:division:half}{metade}? Converta em horas,
        minutos e segundos, usando a divisão euclidiana
        (\cref{thm:g6:wholes:division}) duas vezes.
\end{enumerate}
\end{problem}
