\chapter{Simetria Axial}\label{ch:g6:symmetry}

Dobre uma folha de papel ao longo de uma \omterm{def:g6:lines:objects}{linha}: cada ponto pousa em outro ponto,
sua imagem espelhada. Essa dobradura é \emph{simetria axial} (ou
reflexão), a primeira transformação geométrica do livro --- o
segundo, simetria central, entra \cref{ch:g7:central}.

\section{Reflexões}

\begin{definition}[Ponto simétrico]\label{def:g6:symmetry:def}
Deixar $d $ seja uma \omterm{def:g6:lines:objects}{linha}. O \emph{simétrico}\index{simétrico} de um
ponto $ M $ entre $ d $ é o ponto $ M'$ tal que $ d $ é o
\omterm{def:g6:symmetry:bisector}{bissetriz perpendicular} de $[MM']$: o \omterm{def:g6:lines:objects}{segmento} $[MM']$ é \omterm{def:g6:lines:perp}{perpendicular}
para $ d $, e $ d $ corta-o em seu ponto médio. Um ponto \emph{sobre} $ d $ é o seu
própria simétrica.
\end{definition}

\begin{method}[Construindo um ponto simétrico]\label{met:g6:symmetry:construct}
Para construir o \omterm{def:g6:symmetry:def}{simétrico} $ M'$ de $ M $ através da \omterm{def:g6:lines:objects}{linha} $ d $:
\begin{enumerate}
  \item desenhe o \omterm{def:g6:lines:perp}{perpendicular} para $ d $ através $ M $ (quadrado); chamar
        $ H $ o ponto onde ele cruza $ d $;
  \item medir $ MH $ com o compasso ou régua;
  \item lugar $ M'$ no mesmo \omterm{def:g6:lines:perp}{perpendicular}, do outro lado $ d $,
        com $ HM' = MH $.
\end{enumerate}
Em papel quadriculado é mais rápido: conte os quadrados a partir $ M $ ao eixo e
conte o mesmo número do outro lado.
\end{method}

\begin{example}[Em uma grade]\label{ex:g6:symmetry:grid}
Se o eixo for uma \omterm{def:g6:lines:objects}{linha} de grade vertical e $ M $ é $3$ quadrados à sua esquerda,
então $ M'$ é $3$ quadrados à sua direita, na mesma \omterm{def:g6:lines:objects}{linha}. Para uma diagonal
eixo, a contagem funciona ao longo das diagonais --- sempre \omterm{def:g6:lines:perp}{perpendicular} para
o eixo.
\end{example}

\begin{proposition}[O que uma reflexão preserva]\label{prop:g6:symmetry:preserved}
O \omterm{def:g6:symmetry:def}{simétrico} de uma figura através de uma \omterm{def:g6:lines:objects}{linha} é uma figura do \emph{mesmo
formato e mesmo tamanho}: comprimentos, ângulos, \omterm{def:g4:measure:perimeter}{perímetros} e \omterm{def:g4:measure:area}{áreas} são todos
preservado. O \omterm{def:g6:symmetry:def}{simétrico} de um \omterm{def:g6:lines:objects}{segmento} é um \omterm{def:g6:lines:objects}{segmento}, de um círculo a
círculo do mesmo \omterm{def:g3:shapes:shapes}{raio}, de uma \omterm{def:g6:lines:objects}{linha} por \omterm{def:g6:lines:objects}{linha}.
\end{proposition}
\admitted

\begin{remark}
Uma coisa é \emph{não} preservado: orientação. Um ``b'' refletido
torna-se um ``d'' --- escrita espelhada. Se traçar sua figura e virar
o papel vegetal reproduz a imagem, o reflexo está correto.
\end{remark}

\section{Eixos de simetria de uma figura}

\begin{definition}[Eixo de simetria]\label{def:g6:symmetry:axis}
Uma \omterm{def:g6:lines:objects}{linha} $ d $ é um \emph{eixo de simetria}\index{eixo de simetria} de um
figura quando a reflexão através $ d $ envia a figura exatamente para
em si --- dobrando-se $ d $ faz com que as duas \omterm{def:g3:division:half}{metades} combinem.
\end{definition}

{\small Eixos de simetria de formas familiares. Cuidado com o
retângulo: suas diagonais são \emph{não} eixos de simetria (dobra ao longo de um

\begin{example}\label{ex:g6:symmetry:shapes}
Contando eixos de simetria:
\begin{itemize}
  \item triângulo \omterm{def:g6:shapes:triangles}{isósceles}: $1$ (pelo topo vértice e o meio
        da base); triângulo \omterm{def:g6:shapes:triangles}{equilátero}: $3$;
  \item retângulo: $2$ (as duas ``\omterm{def:g6:lines:objects}{linhas} intermediárias'', \emph{não} o
        diagonais); losango: $2$ (suas diagonais!); quadrado: $4$;
  \item círculo: cada \omterm{def:g6:lines:objects}{linha} que passa pelo centro --- infinitas.
\end{itemize}
\end{example}

\begin{example}[Simetria prova igualdades]\label{ex:g6:symmetry:proves}
Por que os dois ângulos da base de um triângulo \omterm{def:g6:shapes:triangles}{isósceles} igual? Dobre o
triângulo ao longo de seu \omterm{def:g6:symmetry:axis}{eixo de simetria}: a esquerda \omterm{def:g3:division:half}{metade} pousa exatamente no
certo \omterm{def:g3:division:half}{metade}, então o ângulo da base esquerda pousa no ângulo da base direita ---
eles devem ter a mesma medida
(\cref{prop:g6:symmetry:preserved}: os reflexos preservam os ângulos).
\end{example}

\section{Bissetriz perpendicular}

\begin{definition}[Bissetriz perpendicular]\label{def:g6:symmetry:bisector}
O \emph{bissetriz perpendicular}\index{bissetriz perpendicular} de um
\omterm{def:g6:lines:objects}{segmento} $[AB]$ é a \omterm{def:g6:lines:objects}{linha} \omterm{def:g6:lines:perp}{perpendicular} para $[AB]$ através do seu
ponto médio --- é o \omterm{def:g6:symmetry:axis}{eixo de simetria} que troca $ A $ e $ B $.
\end{definition}

\begin{proposition}[Equidistância]\label{prop:g6:symmetry:bisector}
Um ponto $ M $ está no \omterm{def:g6:symmetry:bisector}{bissetriz perpendicular} de $[AB]$ exatamente quando
está à mesma distância de $ A $ e de $ B $: $ MA = MB $.
\end{proposition}
\admitted

\begin{method}[Construção de bússola]\label{met:g6:symmetry:bisector}
Para desenhar o \omterm{def:g6:symmetry:bisector}{bissetriz perpendicular} de $[AB]$ sem medir:
\begin{enumerate}
  \item abra a bússola para mais de \omterm{def:g3:division:half}{metade} de $ AB $;
  \item desenhe dois arcos centrados em $ A $, acima e abaixo do \omterm{def:g6:lines:objects}{segmento};
  \item com o \emph{mesmo} abertura, desenhe dois arcos centrados em $ B $;
  \item juntar os dois pontos de cruzamento: esta \omterm{def:g6:lines:objects}{linha} é o \omterm{def:g6:lines:perp}{perpendicular}
        bissetriz (cada ponto de cruzamento é equidistante de $ A $ e $ B $).
\end{enumerate}
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g6:symmetry:1}
Cópia em papel quadriculado: um eixo vertical $ d $, um ponto $ A $ dois quadrados ao seu
esquerda, um ponto $ B $ cinco quadrados à sua esquerda e um quadrado acima.
Construa seu \omterm{def:g6:symmetry:def}{simétrico} pontos $ A'$ e $ B'$ entre $ d $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:symmetry:2}
Desenhe uma \omterm{def:g6:lines:objects}{linha} $ d $ e um ponto $ M $ no $3$ cm de $ d $ (não na grade
papel). Construa o \omterm{def:g6:symmetry:def}{simétrico} $ M'$ de $ M $ entre $ d $ com esquadro
e governante, seguindo \cref{met:g6:symmetry:construct}. A que distância
são $ M $ e $ M'$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:symmetry:3}
Quais letras maiúsculas do alfabeto, escritas em sua forma mais simples,
tem uma vertical \omterm{def:g6:symmetry:axis}{eixo de simetria} (como A)? Um horizontal (como B)?
Ambos?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:symmetry:4}
Quantos eixos de simetria tem: um triângulo \omterm{def:g6:shapes:triangles}{equilátero}? um retângulo
(isso não é um quadrado)? um losango (não é um quadrado)? um quadrado? um círculo?
Desenhe cada figura com seus eixos.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:symmetry:5}
Verdadeiro ou falso? ``As diagonais de um retângulo são eixos de simetria de
o retângulo.'' Explique com um argumento dobrável ou um desenho.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:symmetry:6}
Um triângulo $ ABC $ é refletido através de uma \omterm{def:g6:lines:objects}{linha} $ d $ em $ A'B'C'$. Dado
$ AB = 4$ cm, $\widehat{BAC} = 70^\circ $ e o \omterm{def:g4:measure:perimeter}{perímetro} de $ ABC $ é
$12$ cm: dar, sem nenhuma construção, $ A'B'$,
$\widehat{B'A'C'}$ e o \omterm{def:g4:measure:perimeter}{perímetro} de $ A'B'C'$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:symmetry:7}
Desenhe um \omterm{def:g6:lines:objects}{segmento} $[AB]$ de $7$ cm e construa seu \omterm{def:g6:lines:perp}{perpendicular}
bissetriz com a bússola (\cref{met:g6:symmetry:bisector}). Escolha qualquer
ponto $ M $ nele e verifique com a régua se $ MA = MB $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:symmetry:8}
Duas árvores estão em pontos $ A $ e $ B $ de um jardim (plano). Onde alguém pode
plantar uma fonte de forma que fique à mesma distância das duas árvores?
Descrever \emph{todos} os possíveis pontos.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:symmetry:9}
Em papel quadriculado, desenhe o eixo $ d $ (uma diagonal da grade em
$45^\circ $) e uma pequena figura em forma de L de um lado. Construa seu
\omterm{def:g6:symmetry:def}{simétrico} figura através $ d $. (Refletir cada vértice: para um $45^\circ $
eixo, um ponto $2$ quadrados à direita do eixo pousam $2$ quadrados acima
isso.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:symmetry:10}
Desenhe um círculo de centro $ O $, uma \omterm{def:g6:lines:objects}{linha} $ d $ através $ O $, e um ponto $ M $ ligado
o círculo. Onde está o \omterm{def:g6:symmetry:def}{simétrico} de $ M $ entre $ d $? Explique por que
a reflexão envia o círculo sobre si mesmo.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:symmetry:11}
Usando a propriedade de equidistância (\cref{prop:g6:symmetry:bisector}),
explique por que os dois pontos de cruzamento dos arcos em
\cref{met:g6:symmetry:bisector} realmente minto no \omterm{def:g6:lines:perp}{perpendicular}
bissetriz de $[AB]$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g6:symmetry:12}
Uma bola de bilhar no ponto $ A $ deve quicar em uma parede reta $ d $ e
ponto de alcance $ B $ (ambos do mesmo lado da parede). Refletir $ B $
entre $ d $ em $ B'$, e desenhe o \omterm{def:g6:lines:objects}{segmento} $[AB']$. Explique por que
o melhor ponto de salto é onde $[AB']$ atravessa a parede. (Ideia: o caminho
$ A \to P \to B $ tem o mesmo comprimento que $ A \to P \to B'$.)
\end{exercise}

\section{Problema: A fonte, o prato quebrado e a caminhada
espelho}

\begin{problem}[{Problema de fim de semana --- as três perpendiculares
bissetrizes de um triângulo se encontram em um ponto: o círculo que passa por três
pontos e o que dois espelhos fazem juntos}]
\label{pb:g6:symmetry:1}
Em \cref{exo:g6:symmetry:8} uma fonte tinha que ficar no mesmo nível
distância de \emph{dois} árvores, e a resposta foi toda uma \omterm{def:g6:lines:objects}{linha} de
manchas. Este problema planta uma \emph{terceiro} árvore --- e o todo
a \omterm{def:g6:lines:objects}{linha} colapsa até um ponto único e perfeitamente determinado. Isso
point esconde um tesouro de geometria: permite desenhar aquele
círculo passando por três pontos dados, reconstrua uma placa inteira
de um fragmento quebrado e pertence a todos os triângulos. O último
parte realiza um experimento com dois espelhos cujo resultado
anuncia a geometria do próximo ano.

\medskip
\noindent\textbf{Part I --- Two trees, then a map of
territories.}
\begin{enumerate}
  \item Desenhe dois pontos $ A $ e $ B $ com $ AB = 6$ centímetros e
        construir o \omterm{def:g6:symmetry:bisector}{bissetriz perpendicular} de $[AB]$ com o
        bússola (\cref{met:g6:symmetry:bisector}). Escolha um ponto
        $ M $ nele e verifique com a régua se $ MA = MB $.
  \item \Cref{prop:g6:symmetry:bisector} diz duas coisas em
        uma vez: cada ponto \emph{sobre} a bissetriz é equidistante
        de $ A $ e $ B $, e todo ponto equidistante \emph{é
        ligado} a bissetriz. Qual dos dois responde à fonte
        questão de \cref{exo:g6:symmetry:8}? O que o
        proposição diz sobre um ponto que é \emph{não} no
        bissetriz?
  \item Tome um ponto $ M $ no mesmo lado da bissetriz que
        $ A $. O \omterm{def:g6:lines:objects}{segmento} $[MB]$ cruza a bissetriz em um ponto
        $ P $. Compare o caminho $ M \to P \to B $ com o caminho
        $ M \to P \to A $ e explique por que $ M $ está mais perto de $ A $
        do que $ B $: a bissetriz divide a folha em um
        ``$ A $-lado'' e um ``$ B $-lado''.
  \item Duas escolas $ A $ e $ B $ servir uma cidade. Toda criança caminha
        para a escola mais próxima. Desenhe um pequeno mapa com $ A $ e $ B $,
        e sombrear a região exata da cidade atendida pela escola
        $ A $. Que \omterm{def:g6:lines:objects}{linha} forma a fronteira?
  \item Uma terceira escola $ C $ abre (coloque-o de forma que as três escolas
        faça um triângulo). Construa o \omterm{def:g6:symmetry:bisector}{bissetriz perpendicular}
        de $[AB]$ e o \omterm{def:g6:symmetry:bisector}{bissetriz perpendicular} de $[BC]$ e
        ligar $ O $ seu ponto de passagem. Quais são as duas igualdades de
        distâncias fazem $ O $ satisfazer e por quê?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Three trees, one point, one
circle.}
\begin{enumerate}[resume]
  \item Deduza da questão 5 que $ OA = OC $ --- e portanto
        isso $ O $ também está no \omterm{def:g6:symmetry:bisector}{bissetriz perpendicular} de
        $[AC]$, que você nunca desenhou. Concluir: \emph{os três
        bissetoras \omterm{def:g6:lines:perp}{perpendiculares} dos lados de um triângulo todos
        passar por um ponto.}
  \item Explique por que o círculo de centro $ O $ e \omterm{def:g3:shapes:shapes}{raio} $ OA $
        passa $ B $ e através $ C $ também: é
        \emph{o} circule através dos três pontos --- o
        \emph{círculo circunscrito}\index{círculo circunscrito}
        do triângulo $ ABC $.
  \item Construa um triângulo com lados $6$ cm, $5$ centímetros e
        $4$ cm (\cref{met:g6:shapes:construct}), então é
        \omterm{pb:g6:symmetry:1}{círculo circunscrito}. Por que é suficiente desenhar apenas
        \emph{dois} bissetoras \omterm{def:g6:lines:perp}{perpendiculares} encontrar o centro?
  \item A placa quebrada: um arqueólogo desenterra um fragmento cujo
        a única parte intacta é um pedaço da borda circular da placa.
        Descreva uma receita que recupere o centro e a
        \omterm{def:g3:shapes:shapes}{raio} da placa original --- e teste: desenhe um
        arco de círculo com sua bússola, ``esqueça'' o centro,
        marque três pontos no arco e reconstrua-o.
  \item Poderiam três pontos ter \emph{não} circular através
        eles? Suponha $ A $, $ B $, $ C $ estão alinhados. O que pode ser
        disse do bissetoras \omterm{def:g6:lines:perp}{perpendiculares} de $[AB]$ e de
        $[BC]$ (\cref{prop:g6:lines:facts})? Concluir.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Axes everywhere, and the walking
mirror.}
\begin{enumerate}[resume]
  \item \Cref{ex:g6:symmetry:shapes} contado $3$ eixos para o
        triângulo \omterm{def:g6:shapes:triangles}{equilátero} e $4$ para a praça. Conjectura
        a contagem para um regular \omterm{def:g4:geometry:polygon}{Pentágono} e um regular \omterm{def:g4:geometry:polygon}{hexágono},
        e descreva onde os eixos passam em cada caso (número \omterm{def:g3:numbers:evenodd}{ímpar} dos lados contra \omterm{def:g3:numbers:evenodd}{par}: as duas situações
        diferem).
  \item Dobre uma folha de papel uma vez e corte uma forma em ambos
        camadas; desdobrar. Por que o buraco \emph{sempre} tenha um
        \omterm{def:g6:symmetry:axis}{eixo de simetria}, o que quer que você corte e qual \omterm{def:g6:lines:objects}{linha} é
        o eixo?
  \item Dobre a folha duas vezes --- a segunda dobra \omterm{def:g6:lines:perp}{perpendicular}
        para o primeiro --- corte e desdobre. Quantos eixos de
        simetria o buraco tem pelo menos e quantos
        cópias do seu corte você vê? (Flocos de neve de papel são
        essa ideia se desenvolveu ainda mais.)
  \item O espelho ambulante: em papel quadriculado, desenhe um eixo vertical
        $ d_1$, uma pequena bandeira $ F $ à sua esquerda, e um segundo
        eixo vertical $ d_2$, $4$ quadrados à direita de $ d_1$.
        Refletir $ F $ entre $ d_1$ (imagem $ F'$), então reflita
        $ F'$ entre $ d_2$ (imagem $ F''$). Comparar $ F $ e $ F''$:
        mesma orientação ou espelhado? Por quantos quadrados fez
        a figura se move? Compare com a distância entre o
        eixos.
  \item O mesmo experimento com $ d_2$ \emph{\omterm{def:g6:lines:perp}{perpendicular}} para
        $ d_1$: reflete a bandeira no eixo vertical e, em seguida,
        a imagem na horizontal. Descreva como $ F''$
        senta-se em relação a $ F$ e ao ponto de passagem do
        dois eixos. (Você acabou de descobrir o próximo ano
        transformação: \cref{ch:g7:central}.)
\end{enumerate}
\end{problem}
