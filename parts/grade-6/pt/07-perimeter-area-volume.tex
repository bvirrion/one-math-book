\chapter{Perímetro, Área, Volume}\label{ch:g6:measure}

Qual é o comprimento da cerca, qual o tamanho do campo, quanta água cabe no
tanque? Três perguntas diferentes, três quantidades diferentes ---
\omterm{def:g6:measure:perimeter}{perímetro}, \omterm{def:g6:measure:area}{área}, \omterm{def:g6:measure:volume}{volume} --- cada um com suas próprias unidades. Confundi-los é
o erro mais comum em geometria; este capítulo os classifica para
bom.

\section{Comprimentos e perímetro}

\begin{definition}[Perímetro]\label{def:g6:measure:perimeter}
O \emph{perímetro}\index{perímetro} de uma figura é o comprimento total
da sua fronteira. É medido em unidades de comprimento: milímetros (mm),
centímetros (cm), metros (m), quilômetros (km), com
\[
1 \text{ km} = 1000 \text{ m}, \qquad
1 \text{ m} = 100 \text{ cm}, \qquad
1 \text{ cm} = 10 \text{ mm}.
\]
\end{definition}

\begin{example}\label{ex:g6:measure:perimeter}
Um retângulo de comprimento $L $ e largura $ w $ tem \omterm{def:g6:measure:perimeter}{perímetro}
\[
P = L + w + L + w = 2 \times (L + w).
\]
Por um $7$ cm por $4$ cm retângulo: $ P = 2 \times (7 + 4) = 2 \times 11 =
22$ cm. Um quadrado de lado $ c $ tem \omterm{def:g6:measure:perimeter}{perímetro} $4c $.
\end{example}

\begin{proposition}[Circunferência de um círculo]\label{prop:g6:measure:circle}
O \omterm{def:g6:measure:perimeter}{perímetro} (ou \emph{circunferência}) de um círculo de \omterm{def:g3:shapes:shapes}{raio} $ r $ é
\[
P = 2 \pi r ,
\]
onde $\pi \approx 3.14$ é o mesmo número para cada círculo.
\end{proposition}
\admitted

\begin{example}
Uma lagoa circular tem \omterm{def:g3:shapes:shapes}{raio} $5$ m. Suas medidas fronteiriças
$2 \times \pi \times 5 = 10\pi \approx 31.4$ m. Mantenha o valor exato
$10\pi $ tanto tempo quanto possível; redondo só no final.
\end{example}

\section{Áreas}

\begin{definition}[Área]\label{def:g6:measure:area}
O \emph{área}\index{área} de uma figura mede a superfície que ela cobre:
quantos quadrados unitários cabem dentro. Unidades: centímetros $^2$ (um quadrado de lado
$1$ centímetros), m $^2$, quilômetros $^2$, \dots\ Cuidadoso:
\[
1 \text{ m}^2 = 100 \times 100 \text{ cm}^2 = 10\,000 \text{ cm}^2
\]
--- um metro quadrado é um $100$ cm por $100$ cm quadrado, então cada passo de
a escada das unidades vale $100$, não $10$.
\end{definition}

\begin{proposition}[Fórmulas básicas de área]\label{prop:g6:measure:formulas}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{l|l}
figure & \omterm{def:g6:measure:area}{área} \\
\hline
rectangle ($ L \times w $) & $ A = L \times w $ \\
square (side $ c $) & $ A = c^2$ \\[4pt]
triângulo retângulo (legs $ a $, $ b $) & $ A = \dfrac{a \times b}{2}$ \\[4pt]
disk (\omterm{def:g3:shapes:shapes}{raio} $ r $) & $ A = \pi r^2$ \\
\end{tabular}
\end{center}
\end{proposition}

\begin{proof}[Prova para o triângulo retângulo]
Duas cópias de um triângulo retângulo com pernas $ a $ e $ b $, colado ao longo do
hipotenusa, forma uma $ a \times b $ retângulo. Então o triângulo é \omterm{def:g6:measure:area}{área} é
\omterm{def:g3:division:half}{metade} o retângulo: $\frac{ab}{2}$. (A fórmula do retângulo é a
contagem de unidades quadradas acima; a fórmula do disco é admitida, veja
\cref{ch:g7:areas}.)
\end{proof}

\begin{example}[Figuras compostas]\label{ex:g6:measure:composite}
Uma sala em forma de L é uma $6$ eu $\times $ $4$ m retângulo com um $2$ eu
$\times $ $2$ m canto quadrado removido. Isso é \omterm{def:g6:measure:area}{área}, passo a passo:
\begin{enumerate}
  \item retângulo completo: $6 \times 4 = 24$ eu $^2$;
  \item quadrado removido: $2 \times 2 = 4$ eu $^2$;
  \item \omterm{def:g3:division:remainder}{restante} \omterm{def:g6:measure:area}{área}: $24 - 4 = 20$ eu $^2$.
\end{enumerate}
Isso é \emph{\omterm{def:g6:measure:perimeter}{perímetro}} não é $20$ qualquer coisa: andando pelo L, o
fronteira ainda medidas $6 + 4 + 6 + 4 = 20$ m --- os dois cortes do
canto substitua duas peças iguais de parede. Mesmo número por coincidência,
mas metros quadrados para um, metros para o outro!
\end{example}

\begin{remark}[Mesmo perímetro, áreas diferentes]
Duas figuras podem ter o mesmo \omterm{def:g6:measure:perimeter}{perímetro} e muito diferente \omterm{def:g6:measure:area}{áreas}: um
$5 \times 5$ quadrado e um $9 \times 1$ retângulo ambos têm \omterm{def:g6:measure:perimeter}{perímetro}
$20$, mas \omterm{def:g6:measure:area}{áreas} $25$ e $9$. \omterm{def:g6:measure:perimeter}{Perímetro} e \omterm{def:g6:measure:area}{área} são verdadeiramente independentes
quantidades.
\end{remark}

\section{Volumes}

\begin{definition}[Volume]\label{def:g6:measure:volume}
O \emph{volume}\index{volume} de um \omterm{def:g5:solids:def}{sólido} mede o espaço que preenche:
quantas unidades \omterm{def:g5:solids:def}{cubos} caber dentro. Unidades: centímetros $^3$, m $^3$, \dots\ com
$1$ eu $^3 = 1\,000\,000$ cm $^3$ (cada degrau da escada vale
$1000$). Para \omterm{def:g5:solids:net}{líquidos} também se usa o \emph{litro}:
\[
1 \text{ L} = 1 \text{ dm}^3 = 1000 \text{ cm}^3,
\qquad
1 \text{ m}^3 = 1000 \text{ L}.
\]
\end{definition}

\begin{proposition}[Volume de uma caixa]\label{prop:g6:measure:box}
Um retangular \omterm{def:g5:solids:def}{caixa} (um \emph{prisma retangular}) de comprimento $ L $, largura $ w $
e altura $ h $ tem \omterm{def:g6:measure:volume}{volume}
\[
V = L \times w \times h .
\]
\end{proposition}

\begin{proof}
A camada inferior contém $ L \times w $ unidade \omterm{def:g5:solids:def}{cubos}
(\cref{def:g6:measure:area} foto, com \omterm{def:g5:solids:def}{cubos}), e há $ h $
tais camadas.
\end{proof}

camadas.}
\end{omfigure}

\begin{example}\label{ex:g6:measure:aquarium}
Um aquário mede $60$ cm por $30$ cm por $40$ centímetros (altura). \omterm{def:g6:measure:volume}{Volume}:
\[
V = 60 \times 30 \times 40 = 72\,000 \text{ cm}^3 = 72 \text{ dm}^3
= 72 \text{ L}.
\]
Passo a passo para a conversão: $1000$ cm $^3$ faça um litro e
$72\,000 \div 1000 = 72$.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g6:measure:1}
Converter: $3.5$ m em cm; $420$ mm em cm; $0.8$ km em m;
$25\,000$ m em km.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:measure:2}
Calcule o \omterm{def:g6:measure:perimeter}{perímetro} de: um retângulo $8$ cm $\times $ $3.5$ cm; um
quadrado de lado $6.2$ cm; um triângulo com lados $5$ cm, $7$ centímetros e
$9$ cm.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:measure:3}
Uma pista de corrida circular tem \omterm{def:g3:shapes:shapes}{raio} $50$ m. Quanto tempo dura uma volta
(valor exato com $\pi $, depois \omterm{def:g4:numbers:round}{arredondado} para o metro)? Quantas voltas fazem
pelo menos $2$ quilômetros?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:measure:4}
Calcule o \omterm{def:g6:measure:area}{área} de: um retângulo $9$ cm $\times $ $4$ cm; um quadrado de
lado $7$ m; um triângulo retângulo com pernas $6$ centímetros e $10$ cm; um disco de
\omterm{def:g3:shapes:shapes}{raio} $3$ cm (valor exato, depois \omterm{def:g4:numbers:round}{arredondado} para cm $^2$).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:measure:5}
Converter: $3$ eu $^2$ em cm $^2$; $45\,000$ cm $^2$ em m $^2$; $2.5$ eu
em cm $^3$; $4\,500$ L em m $^3$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:measure:6}
Um campo retangular é $120$ sou longo e $85$ m de largura. Quantos metros de
são necessárias cercas para encerrá-lo? Qual é o seu \omterm{def:g6:measure:area}{área}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:measure:7}
Calcule o \omterm{def:g6:measure:volume}{volume} de um \omterm{def:g5:solids:def}{caixa} $5$ cm $\times $ $4$ cm $\times $ $10$ cm,
e de um \omterm{def:g5:solids:def}{cubo} de borda $3$ cm.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g6:measure:8}
Desenhe dois retângulos diferentes com \omterm{def:g6:measure:perimeter}{perímetro} $16$ cm e calcule seus
\omterm{def:g6:measure:area}{áreas}. Qual dos seus retângulos tem o maior \omterm{def:g6:measure:area}{área}?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:measure:9}
Uma figura em forma de T é feita de $10 \times 2$ retângulo horizontal em
topo de um $2 \times 6$ vertical (medidas em cm). Calcule seu
\omterm{def:g6:measure:area}{área}, então é \omterm{def:g6:measure:perimeter}{perímetro} (ande pela fronteira com cuidado, adicionando
cada borda).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:measure:10}
Uma piscina é retangular \omterm{def:g5:solids:def}{caixa} $25$ sou longo, $10$ m de largura e
$2$ estou profundo.
\begin{enumerate}
  \item Quantos metros cúbicos de água ele contém quando cheio?
  \item Quantos litros são isso?
  \item A piscina está cheia às $50\,000$ L por hora. Quanto tempo dura o
        enchimento levar?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g6:measure:11}
Um jardim é um quadrado de lado $20$ m contendo uma lagoa circular de
\omterm{def:g3:shapes:shapes}{raio} $3$ m. O que \omterm{def:g6:measure:area}{área} de grama está lá (valor exato, depois \omterm{def:g4:numbers:round}{arredondado}
para o m $^2$)?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g6:measure:12}
Uma barra de chocolate mede $15$ cm $\times $ $6$ cm $\times $ $1$ cm.
O criador dobra \emph{todos os três} dimensões.
\begin{enumerate}
  \item Por quanto é o \omterm{def:g6:measure:volume}{volume} multiplicado? Verifique calculando ambos
        \omterm{def:g6:measure:volume}{volumes}.
  \item O preço é multiplicado por $4$. O grande bar é um negócio melhor
        do que o pequeno?
\end{enumerate}
\end{exercise}

\section{Problema: cerca da Rainha Dido}

\begin{problem}[{Problema de fim de semana --- com um comprimento fixo de cerca,
qual forma envolve mais terra? A praça bate a cada
retângulo, o círculo vence o quadrado e a parede de um celeiro muda
tudo}]
\label{pb:g6:measure:1}
Reza a lenda que a rainha Dido, ao desembarcar na costa de África, foi
concedeu “tanta terra quanto uma pele de boi pode conter” --- então ela cortou
a pele em uma tira fina imensamente longa e fechada o suficiente
terreno para fundar a cidade de Cartago. O problema dela agora é seu: um
o agricultor possui exatamente $20$ m de cerca. \Cref{exo:g6:measure:8}
mostrou que duas canetas com o mesmo \omterm{def:g6:measure:perimeter}{perímetro} pode ter diferentes
\omterm{def:g6:measure:area}{áreas}; esse problema encontra o \emph{melhor} caneta --- e descobre
que a resposta muda completamente quando a parede de um celeiro oferece espaço livre
lado.

\medskip
\noindent\textbf{Part I --- Twenty meters of fence.}
\begin{enumerate}
  \item A caneta deve ser um retângulo usando todos $20$ m de cerca.
        Verifique se um $1 \times 9$ caneta e um $2 \times 8$ caneta ambos
        qualificar e calcular seus \omterm{def:g6:measure:area}{áreas}.
  \item Explique por que o comprimento e a largura de cada curral qualificado
        somar a $10$ m. Depois faça a tabela completa do
        canetas de números inteiros ($1 \times 9$ até $5 \times 5$) com
        seu \omterm{def:g6:measure:area}{áreas}. Qual é o melhor?
  \item Vale a pena tentar os lados decimais? Calcule o \omterm{def:g6:measure:area}{áreas} do
        $4.5 \times 5.5$ caneta e do $4.9 \times 5.1$ caneta, e
        comparar com o $5 \times 5$ quadrado. O que você
        conjectura?
  \item A questão 2 transformou o problema da cerca em um número puro
        pergunta: \emph{entre todos os \omterm{def:g3:numbers:evenodd}{pares} de números que somam
        $10$, qual \omterm{def:g3:numbers:evenodd}{par} tem o maior \omterm{def:g2:mult:def}{produto}?} Responda de
        sua tabela e indique a regra geral que você observa.
  \item Eis por que sair da praça sempre perde,
        com tesoura em vez de álgebra: comece do
        $5 \times 5$ caneta e transforme-a no $6 \times 4$ caneta
        removendo uma tira e colando outra de volta. Qual
        tira é removida, qual é adicionada, e por que a
        troca perder exatamente um metro quadrado? Explique por que um
        passo adicional (para $7 \times 3$) perde \omterm{def:g3:numbers:evenodd}{par} mais.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The barn, and the circle.}
\begin{enumerate}[resume]
  \item O cercado agora está construído contra uma longa parede do celeiro: a parede
        substitui um comprimento da caneta e o $20$ m de cerca
        cubra apenas os outros três lados. Faça uma tabela do
        canetas de números inteiros (largura $ w $, comprimento $ L $ ao longo da parede,
        $ w + L + w = 20$) e seus \omterm{def:g6:measure:area}{áreas}. Qual caneta ganha agora ---
        e é um quadrado?
  \item Explique o vencedor com um espelho
        (\cref{ch:g6:symmetry}): reflete a caneta em todo o celeiro
        parede e considere a caneta duplicada. Quanto custa a cerca
        o uso de caneta dupla, qual caneta dupla é melhor na Parte I,
        e o que isso torna a caneta original?
  \item Dido não construiu um retângulo. Dobre o $20$ m de cerca
        em um círculo: usando $\pi \approx 3.14$, calcule seu
        \omterm{def:g3:shapes:shapes}{raio} (para o cm), então é \omterm{def:g6:measure:area}{área} (para o m $^2$)
        (\cref{prop:g6:measure:circle},
        \cref{prop:g6:measure:formulas}). Compare com o
        $5 \times 5$ quadrado.
  \item O problema inverso: uma caneta de \omterm{def:g6:measure:area}{área} exatamente $36$ eu $^2$ é
        queria, com o mínimo de cerca possível. Compare o
        retângulos $1 \times 36$, $2 \times 18$, $3 \times 12$,
        $4 \times 9$ e $6 \times 6$: \omterm{def:g6:measure:perimeter}{perímetros}? Qual forma é
        mais barato, e como esta questão é a imagem espelhada de
        Parte I?
  \item Em uma frase: por que latas, canos e \omterm{def:g5:solids:def}{caixas} d’água são tão
        frequentemente \emph{redondo}?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- \omterm{def:g6:measure:perimeter}{Perímetro} and \omterm{def:g6:measure:area}{área} are strangers.}
\begin{enumerate}[resume]
  \item Encontre um retângulo cujo \omterm{def:g6:measure:perimeter}{perímetro} excede $100$ eu mas
        cujo \omterm{def:g6:measure:area}{área} é menor que $1$ eu $^2$. (Muito longo e muito
        magro. Decimais permitidos.)
  \item Verdadeiro ou falso: ``de duas figuras, aquela com maior
        \omterm{def:g6:measure:perimeter}{perímetro} tem o maior \omterm{def:g6:measure:area}{área}.'' Dê um contra-exemplo
        deste problema.
  \item Pegue o $6 \times 4$ retângulo e corte um $1 \times 1$
        entalhe quadrado no meio de um lado longo (o entalhe
        abre para fora, como um dente perdido). Calcule o \omterm{def:g6:measure:area}{área}
        e o \omterm{def:g6:measure:perimeter}{perímetro} da figura entalhada e compare ambos
        com o original. O que aconteceu com cada um?
  \item Os cartógrafos conhecem um fato estranho: quanto mais detalhado o mapa,
        o \emph{mais longo} um litoral mede --- cada zoom
        revela novos pequenos entalhes, e a pergunta 13 mostra o que
        cada entalhe faz. Explique em uma ou duas frases por que
        ``a extensão da costa da Bretanha'' é escorregadia
        número, enquanto ``o \omterm{def:g6:measure:area}{área} da Bretanha'' não é.
  \item O exame do agricultor. Com $36$ m de cerca e celeiro
        parede disponível, encontre a melhor caneta retangular (use o
        espelho da questão 7), dê o seu \omterm{def:g6:measure:area}{área}, e compare com
        a melhor caneta construída com $36$ m em campo aberto. Quanto
        a parede do celeiro ganha o agricultor?
\end{enumerate}
\end{problem}
