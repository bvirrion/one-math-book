\chapter{Decimal Numbers}\label{ch:g6:decimals}

Between $2$ and $3$ there are plenty of numbers: $2.5$, $2.71$,
$2.999$\,\dots\ \omterm{def:g6:decimals:places}{Decimal numbers} extend the place-value system to the
right of the units, and they follow the same rules as whole numbers ---
with a few traps this chapter will teach you to avoid.

\section{Decimal writing}

\begin{definition}[Decimal places]\label{def:g6:decimals:places}
A \emph{decimal number}\index{decimal number} has a whole part and a
decimal part, separated by the \omterm{def:g4:decimals:point}{decimal point}. Each place to the right
of the point is worth ten times less than the one before: tenths,
hundredths, thousandths. For instance
\[
13.407 = 13 + \frac{4}{10} + \frac{0}{100} + \frac{7}{1000}
= 1 \times 10 + 3 + 4 \times \frac{1}{10} + 7 \times \frac{1}{1000}.
\]
\end{definition}

\begin{remark}[Zeros that matter, zeros that don't]
Adding \omterm{def:g1:counting:zero}{zeros} at the \emph{end} of the decimal part changes nothing:
$2.5 = 2.50 = 2.500$. But \omterm{def:g1:counting:zero}{zeros} \emph{between} digits are essential:
$13.407 \neq 13.47$. And $0.5$ is very different from $0.05$!
\end{remark}

\begin{method}[Comparing decimal numbers]\label{met:g6:decimals:compare}
\begin{enumerate}
  \item Compare the whole parts first: $7.2 > 5.99$ because $7 > 5$.
  \item If the whole parts are equal, compare the decimal parts digit by
        digit from the left --- padding with final \omterm{def:g1:counting:zero}{zeros} helps:
        to compare $3.4$ and $3.15$, write $3.40$ and $3.15$;
        since $40 > 15$ hundredths, $3.4 > 3.15$.
\end{enumerate}
Beware: \emph{longer does not mean larger}. $3.15$ has more digits than
$3.4$ but is smaller.
\end{method}

\section{Adding, subtracting, multiplying}

\begin{method}[Column computations with decimals]\label{met:g6:decimals:columns}
To add or subtract \omterm{def:g6:decimals:places}{decimal numbers}, align the \emph{\omterm{def:g4:decimals:point}{decimal points}}
(so units sit under units, tenths under tenths), padding with final
\omterm{def:g1:counting:zero}{zeros} if needed; then compute as with whole numbers, and place the point
of the result under the others.
\end{method}

\begin{example}\label{ex:g6:decimals:addsub}
Compute $13.7 + 2.85$, aligning the points ($13.7 = 13.70$):
\[
\begin{array}{r}
1\,3.7\,0 \\
+\ \ \,2.8\,5 \\
\hline
1\,6.5\,5
\end{array}
\]
Compute $6 - 2.35$ (write $6 = 6.00$):
$6.00 - 2.35 = 3.65$. Check: $3.65 + 2.35 = 6$.
\end{example}

\begin{example}[Multiplying decimals]\label{ex:g6:decimals:mult}
To compute $2.3 \times 1.4$: multiply as whole numbers,
$23 \times 14 = 322$; then count the decimal digits of the factors
($1 + 1 = 2$), and place the point so the result has that many:
$2.3 \times 1.4 = 3.22$. Sanity check on sizes: $2.3 \times 1.4$
should be a bit more than $2.3 \times 1 = 2.3$ --- and $3.22$ is.
\end{example}

\section{Multiplying and dividing by 10, 100, 1000}

\begin{proposition}[Shifting the point]\label{prop:g6:decimals:shift}
Multiplying a \omterm{def:g6:decimals:places}{decimal number} by $10$, $100$, $1000$ moves its \omterm{def:g4:decimals:point}{decimal
point} $1$, $2$, $3$ places to the \emph{right}; dividing moves it to the
\emph{left} (padding with \omterm{def:g1:counting:zero}{zeros} when needed):
\[
3.75 \times 100 = 375, \qquad
42.1 \div 1000 = 0.0421 .
\]
\end{proposition}

\begin{proof}
Multiplying by $10$ makes each digit worth ten times more: tenths
become units, units become tens, and so on --- every digit moves one
column to the left in the place-value table, which is the same as moving
the point one place to the right. Dividing reverses this.
\end{proof}

\begin{example}[Units of measurement]\label{ex:g6:decimals:units}
Converting units is exactly this game: since $1$ m $= 100$ cm,
\[
3.75 \text{ m} = 3.75 \times 100 \text{ cm} = 375 \text{ cm},
\qquad
42 \text{ mm} = 42 \div 10 \text{ cm} = 4.2 \text{ cm}.
\]
\end{example}

\section{Rounding}

\begin{definition}[Rounding]\label{def:g6:decimals:rounding}
The \emph{rounding}\index{rounding} of a number to the unit (or tenth,
hundredth, \dots) is the closest number with that precision. Look at the
next digit: if it is $0,1,2,3,4$, \omterm{def:g4:numbers:round}{round} down (keep); if it is
$5,6,7,8,9$, \omterm{def:g4:numbers:round}{round} up.
\end{definition}

\begin{example}
$7.38$ rounded to the unit is $7$ (next digit $3$: keep); rounded to
the tenth it is $7.4$ (next digit $8$: \omterm{def:g4:numbers:round}{round} up). The price
$4.996$ rounded to the hundredth is $5.00$ --- \omterm{def:g6:decimals:rounding}{rounding} can change
every digit!
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g6:decimals:1}
Write as a \omterm{def:g6:decimals:places}{decimal number}: $5 + \dfrac{3}{10} + \dfrac{7}{100}$;
\quad $\dfrac{9}{10}$; \quad $12 + \dfrac{4}{1000}$;
\quad ``eight units and five hundredths''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:2}
In $86.354$: what is the digit of the tenths? Of the hundredths? What
does the digit $8$ count? Write this number as in
\cref{def:g6:decimals:places}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:3}
Copy and complete with $<$, $>$ or $=$:
\[
5.3 \;?\; 5.29, \qquad
0.7 \;?\; 0.70, \qquad
2.09 \;?\; 2.9, \qquad
14.5 \;?\; 14.49 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:4}
Order from smallest to largest:
$4.2$; \ $4.05$; \ $4.51$; \ $4.15$; \ $4.5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:5}
Which \omterm{def:g6:decimals:places}{decimal numbers} correspond to the points $A$, $B$, $C$ on a
number line graduated in tenths, if $A$ is $3$ ticks after $6$, $B$ is
$7$ ticks after $6$, and $C$ is $2$ ticks after $7$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:6}
Compute in columns: $45.8 + 7.65$; \quad $23.4 - 8.72$; \quad
$5.6 \times 2.4$ (count the decimal digits!).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:7}
Compute without any written work:
\[
6.42 \times 10, \qquad
0.35 \times 1000, \qquad
78.1 \div 100, \qquad
5 \div 1000 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:8}
Convert: $2.4$ m into cm; \quad $370$ g into kg; \quad $0.85$ km into
m; \quad $56$ mm into m.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g6:decimals:9}
Round $23.867$: to the unit; to the tenth; to the hundredth. Round
$9.97$ to the tenth.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:decimals:10}
A baguette costs $1.15$. Lena buys three baguettes and pays with a
$5$ bill. Write the two computations needed, and give her change.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g6:decimals:11}
Find a \omterm{def:g6:decimals:places}{decimal number} strictly between $7.4$ and $7.5$; then between
$3.99$ and $4$. How many such numbers are there in each case?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g6:decimals:12}
Using each of the digits $2$, $5$, $8$ exactly once and one \omterm{def:g4:decimals:point}{decimal
point}, write the largest possible number, then the smallest possible
one. (Numbers like $.58$ are not allowed: the whole part must contain
at least one digit.)
\end{exercise}
