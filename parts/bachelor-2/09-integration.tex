\chapter{Integration}\label{ch:b2:integration}

The Year 1 volume built the integral on a segment. This chapter
extends it to arbitrary intervals (improper integrals, with the full
comparison toolkit), then studies integrals \emph{depending on a
parameter} --- continuity and differentiation under the integral
sign --- powered by the dominated convergence theorem, the one
result of this chapter taken on trust. The $\Gamma$ function serves
as the running example, and as the gateway to half of the special
functions of mathematics.

\section{Integrals on an arbitrary interval}

\begin{definition}\label{def:b2:integration:improper}
Let $f$ be piecewise continuous on $\intco{a}{b}$ ($b \in \R$ or
$+\infty$). The integral \emph{converges}\index{improper integral}
when $\lim_{x \to b^-} \int_a^x f$ exists; one then writes
$\int_a^b f$ for the limit. (Similarly on $\intoc{a}{b}$, and on
$\intoo{a}{b}$ by splitting at an interior point --- the choice does
not matter, by Chasles.) The integral \emph{converges absolutely}
when $\int_a^b \abs f$ converges; absolute convergence implies
convergence, by the Cauchy criterion:
\[
\Bigl| \int_x^{y} f \Bigr| \leq \int_x^{y} \abs f
\]
and completeness of $\R$ (the primitive has the Cauchy property).
In detail: let $F(x) = \int_a^x f$ and $G(x) = \int_a^x \abs
f$. If $\int^b\abs f$ converges, $G$ has a limit at $b^-$, so
for every $\varepsilon > 0$ there is $c < b$ with $G(y) - G(x)
\leq \varepsilon$ whenever $c \leq x \leq y < b$; the display
transfers this Cauchy property to $F$. For any sequence $x_n
\to b^-$ the values $F(x_n)$ then form a Cauchy sequence of
reals, convergent by completeness, and interlacing two such
sequences shows the limit is the same for all of them: $F$ has
a limit at $b^-$.
\end{definition}

\begin{theorem}[Positive comparison toolkit]\label{thm:b2:integration:comparison}
For $f, g \geq 0$ piecewise continuous on $\intco{a}{b}$:
\begin{enumerate}
  \item $\int_a^b f$ converges iff the primitive $x \mapsto \int_a^x
        f$ is bounded;
  \item $f \leq g$: convergence of $\int g$ forces that of $\int f$;
        divergence transfers the other way;
  \item $f \sim g$ at $b$: the two integrals have the same nature;
  \item the reference scales: at $+\infty$, $\int^{\infty}
        \frac{\dd t}{t^\alpha}$ converges iff $\alpha > 1$, and
        $\int^\infty \frac{\dd t}{t(\ln t)^\beta}$ iff $\beta > 1$;
        at a finite endpoint $b$, $\int^b \frac{\dd t}{(b -
        t)^\alpha}$ converges iff $\alpha < 1$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) The primitive $F(x) = \int_a^x f$ is nondecreasing ($f \geq
0$). If it is bounded, $\ell = \sup_{x < b}F$ is finite and
$F(x) \to \ell$: given $\varepsilon > 0$, some $F(x_0) > \ell -
\varepsilon$, and monotonicity traps $F(x) \in
\intoc{\ell - \varepsilon}{\ell}$ for $x_0 \leq x < b$. If it is
unbounded, $F \to +\infty$: divergence.

(2) From $f \leq g$: $\int_a^x f \leq \int_a^x g$ for all $x$;
if $\int^b g$ converges, the right side is bounded, hence so is
the left, and (1) concludes. Contraposition transfers divergence
the other way.

(3) $f \sim g$ at $b$ provides $c < b$ with
\[
\tfrac12\,g(t) \;\leq\; f(t) \;\leq\; 2\,g(t)
\qquad (c \leq t < b) :
\]
by (2) applied both ways on $\intco{c}{b}$, the two integrals
have the same nature; the initial piece $\intcc{a}{c}$ is a
proper integral and changes nothing.

(4) Explicit primitives: for $\alpha \neq 1$ and $\beta \neq 1$,
\[
\int_c^x \frac{\dd t}{t^\alpha}
= \frac{x^{1-\alpha} - c^{1-\alpha}}{1 - \alpha},
\qquad
\int_c^x \frac{\dd t}{t(\ln t)^\beta}
= \frac{(\ln x)^{1-\beta} - (\ln c)^{1-\beta}}{1 - \beta},
\]
with logarithms in the excluded cases: bounded as $x \to
+\infty$ exactly when $\alpha > 1$, resp.\ $\beta > 1$. At a
finite endpoint, the substitution $u = b - t$ reduces to the
scale $\int_0 u^{-\alpha}\,\dd u$, bounded iff $\alpha < 1$.
Apply (1) each time.
\end{proof}

\begin{example}[Two warm-ups, worked to the end]\label{ex:b2:integration:warmups}
\emph{(a)} $\displaystyle\int_0^1 \ln t\,\dd t$: the integrand
blows up at $0^+$, but $\abs{\ln t} = o\bigl(t^{-1/2}\bigr)$
there (logarithms lose to powers), and $\int_0 t^{-1/2}$
converges: absolute convergence. The value, by parts on
$\intcc{\varepsilon}{1}$:
\[
\int_\varepsilon^1 \ln t\,\dd t
= \bigl[t\ln t - t\bigr]_\varepsilon^1
= -1 - \varepsilon\ln\varepsilon + \varepsilon
\xrightarrow[\varepsilon\to0^+]{} -1 .
\]
\emph{(b)} $\displaystyle\int_0^\infty \frac{\ln t}{1 +
t^2}\,\dd t$: trouble at both ends, so split at $1$. Near $0$:
$\abs{\ln t}$ integrable as in (a); near $\infty$: $\frac{\ln
t}{1+t^2} = o(t^{-3/2})$: absolutely convergent. The
substitution $t = \frac1u$ maps $\intoo{0}{1}$ onto
$\intoo{1}{\infty}$ and
\[
\int_0^1 \frac{\ln t}{1+t^2}\,\dd t
= \int_1^{\infty} \frac{-\ln u}{1 + u^{-2}}\cdot
\frac{\dd u}{u^2}
= -\int_1^\infty \frac{\ln u}{1+u^2}\,\dd u :
\]
the two halves cancel, and the integral is $0$. Closing
insight: symmetry under $t \mapsto \frac1t$ is worth a page of
computation --- the same trick already powered
\cref{exo:b2:integration:3}.
\end{example}

\begin{example}[One value, three integrals]\label{ex:b2:integration:onecos}
Study $I = \displaystyle\int_0^{\infty} \frac{1 - \cos
t}{t^2}\,\dd t$. At $0$: $1 - \cos t \sim \frac{t^2}2$, so the
integrand extends continuously by the value $\frac12$ --- no
singularity at all. At $\infty$: $0 \leq \frac{1 - \cos t}{t^2}
\leq \frac{2}{t^2}$: absolute convergence
(\cref{thm:b2:integration:comparison}). Value: integrate by
parts on $\intcc{\varepsilon}{M}$ with $u = 1 - \cos t$, $v' =
t^{-2}$:
\[
\int_\varepsilon^M \frac{1 - \cos t}{t^2}\,\dd t
= \Bigl[-\frac{1 - \cos t}{t}\Bigr]_\varepsilon^M
+ \int_\varepsilon^M \frac{\sin t}{t}\,\dd t .
\]
The bracket vanishes at both ends ($\frac{1 -
\cos\varepsilon}{\varepsilon} \sim \frac\varepsilon2$; bounded
numerator at $M$), and the integral tends to the Dirichlet value
$\frac\pi2$ (\cref{exo:b2:integration:10}): $I = \frac\pi2$.
Closing insight: with $1 - \cos t = 2\sin^2\frac t2$ and $u =
\frac t2$,
\[
I = \int_0^\infty \frac{2\sin^2 u}{(2u)^2}\;2\,\dd u
= \int_0^\infty \Bigl(\frac{\sin u}{u}\Bigr)^{\!2}\dd u :
\]
the three classics $\int_0^\infty\frac{\sin t}{t}\dd t$,
$\int_0^\infty\bigl(\frac{\sin t}{t}\bigr)^2\dd t$
(\cref{exo:b2:integration:11}) and $I$ all share the value
$\frac\pi2$, passed around by parts and substitution --- and
only the first is semi-convergent: integration by parts traded
away the absolute convergence for a simpler integrand.
\end{example}

\begin{example}[A semi-convergent integral]\label{ex:b2:integration:sint}
$\displaystyle\int_1^{\infty} \frac{\sin t}{t}\,\dd t$ converges:
integrate by parts,
\[
\int_1^x \frac{\sin t}{t}\dd t
= \Bigl[\frac{-\cos t}{t}\Bigr]_1^x - \int_1^x \frac{\cos
t}{t^2}\dd t ,
\]
where the bracket has a limit and the last integral converges
absolutely ($\abs{\cos t}/t^2 \leq t^{-2}$). But not absolutely:
from $\abs{\sin t} \geq \sin^2 t$,
\[
\int_1^x \frac{\abs{\sin t}}{t}\,\dd t
\;\geq\; \int_1^x \frac{\sin^2t}{t}\,\dd t
= \underbrace{\int_1^x \frac{\dd t}{2t}}_{=\ \frac12\ln x
\ \to\ \infty}
\;-\; \underbrace{\int_1^x \frac{\cos 2t}{2t}\,\dd
t}_{\text{convergent}} ,
\]
where the last integral converges by the \emph{same}
integration by parts as above (with $\sin 2t$ in the bracket):
a divergent piece minus a convergent one diverges. So
$\int_1^\infty\frac{\sin t}{t}\dd t$ converges without
converging absolutely --- the integral analogue of the
alternating series, with integration by parts playing the role
of the alternating test.
\end{example}

\section{The convergence theorem}

\begin{theorem}[Dominated convergence]\label{thm:b2:integration:dominated}
Let $(f_n)$ be piecewise continuous on an interval $I$, converging
pointwise to a piecewise continuous $f$, and suppose there is a
fixed integrable $\varphi \geq 0$ ($\int_I \varphi < \infty$)
with\index{dominated convergence theorem}
\[
\abs{f_n(t)} \leq \varphi(t) \qquad (t \in I,\ n \in \N).
\]
Then all $\int_I f_n$ and $\int_I f$ converge absolutely, and
\[
\int_I f_n \xrightarrow[n \to \infty]{} \int_I f .
\]
\end{theorem}
\admitted

\begin{remark}
The honest proof belongs to the Lebesgue integration theory of Year
3; the statement, however, is used constantly from now on. The
\emph{domination} hypothesis is the whole point: pointwise
convergence alone does not suffice ($f_n = n\,\mathbf{1}_{\intoo{0}{1/n}}$,
sliding bumps: $\int f_n = 1 \not\to 0 = \int f$). The theorem
also holds for a continuous parameter ($f_\lambda$, $\lambda \to
\lambda_0$), by the sequential characterization of limits.
\end{remark}

\begin{example}[A Gaussian limit, by domination]\label{ex:b2:integration:wallislimit}
Compute $\displaystyle\lim_{n\to\infty} I_n$ where $I_n =
\int_0^\infty \Bigl(1 + \frac{t^2}{n}\Bigr)^{\!-n}\dd t$.
Pointwise, $(1 + t^2/n)^n \to \eu^{t^2}$ (compound-interest
limit), so the integrands tend to $\eu^{-t^2}$. Domination: the
sequence $n \mapsto (1 + u/n)^n$ is nondecreasing for $u \geq 0$
(AM--GM on the $n + 1$ factors $1, 1 + \frac un, \dots, 1 +
\frac un$ gives $(1 + \frac u{n+1})^{n+1} \geq (1 + \frac
un)^n$), so for $n \geq 2$:
\[
\Bigl(1 + \frac{t^2}{n}\Bigr)^{\!-n}
\leq \Bigl(1 + \frac{t^2}{2}\Bigr)^{\!-2},
\]
an integrable dominator ($\sim 4t^{-4}$ at infinity). Dominated
convergence:
\[
I_n \xrightarrow[n\to\infty]{} \int_0^\infty \eu^{-t^2}\dd t
= \frac{\sqrt\pi}{2}
\]
(the Gaussian integral of \cref{exo:b2:integration:8}). Closing
check: the substitution $t = \sqrt n\tan\theta$ computes $I_n$
exactly, $I_n = \sqrt n\int_0^{\pi/2}\cos^{2n-2}\theta\,\dd\theta
= \sqrt n\,W_{2n-2}$, and the Wallis asymptotics $W_m \sim
\sqrt{\pi/(2m)}$ (\cref{lem:b2:comparison:wallis}) give $\sqrt
n\,W_{2n-2} \to \frac{\sqrt\pi}2$ again: the two pillars of this
chapter and the last agree.
\end{example}

\begin{example}[Dominated convergence, continuous parameter]
\label{ex:b2:integration:arctanlimit}
Compute
\[
\lim_{x\to+\infty}\int_0^\infty
\frac{\arctan(xt)}{1+t^2}\,\dd t .
\]
For each $t > 0$,
$\arctan(xt) \to \frac\pi2$ as $x \to \infty$; and the
domination
\[
\Bigl|\frac{\arctan(xt)}{1+t^2}\Bigr|
\leq \frac{\pi/2}{1+t^2},
\qquad\text{integrable, independent of } x,
\]
holds for all $x$. By the continuous-parameter form of
\cref{thm:b2:integration:dominated} (sequential
characterization: test along every $x_n \to \infty$),
\[
\int_0^\infty\frac{\arctan(xt)}{1+t^2}\,\dd t
\xrightarrow[x\to+\infty]{}
\frac\pi2\int_0^\infty\frac{\dd t}{1+t^2}
= \frac{\pi^2}{4} .
\]
Closing insight: the single point $t = 0$, where the pointwise
limit is $0$ rather than $\frac\pi2$, changes nothing --- the
limit function only enters through its integral, one of the
quiet mercies of the theorem.
\end{example}

\section{Integrals with a parameter}

\begin{theorem}[Continuity under the integral sign]\label{thm:b2:integration:continuity}
Let $f \colon A \times I \to \R$ ($A$ a metric space, $I$ an
interval) with: $t \mapsto f(x, t)$ piecewise continuous for each
$x$; $x \mapsto f(x, t)$ continuous for each $t$; and a
\emph{domination} $\abs{f(x,t)} \leq \varphi(t)$ ($\varphi$
integrable on $I$, independent of $x$). Then
\[
F(x) = \int_I f(x, t)\,\dd t
\]
is defined and continuous on $A$.\index{parameter integral}
\end{theorem}

\begin{proof}
Definedness: domination gives absolute convergence. Continuity at
$x_0$: for any sequence $x_n \to x_0$, the functions $g_n(t) =
f(x_n, t)$ converge pointwise to $f(x_0, t)$ (continuity in $x$)
under the fixed domination $\varphi$: dominated convergence gives
$F(x_n) \to F(x_0)$; conclude by the sequential characterization of
continuity (\cref{def:b2:metric:continuity}).
\end{proof}

\begin{theorem}[Differentiation under the integral sign]\label{thm:b2:integration:leibnizrule}
Let $f \colon J \times I \to \R$ ($J$ an interval of parameters)
with: $t \mapsto f(x,t)$ integrable on $I$ for each $x$; $x \mapsto
f(x,t)$ of class $C^1$ for each $t$, the partial derivative
$\frac{\partial f}{\partial x}$ being piecewise continuous in $t$
and dominated: $\bigl|\frac{\partial f}{\partial x}(x,t)\bigr| \leq
\psi(t)$ with $\psi$ integrable. Then $F(x) = \int_I f(x,t)\dd t$
is $C^1$ on $J$ and
\[
F'(x) = \int_I \frac{\partial f}{\partial x}(x, t)\,\dd t .
\]
\end{theorem}

\begin{proof}
Fix $x$ and $h_n \to 0$. The difference quotients
\[
\frac{F(x + h_n) - F(x)}{h_n}
= \int_I \frac{f(x + h_n, t) - f(x, t)}{h_n}\,\dd t
\]
have integrands converging pointwise to $\frac{\partial f}{\partial
x}(x, t)$, and dominated by $\psi(t)$: by the mean value
inequality applied in $x$ at fixed $t$,
\[
\Bigl|\frac{f(x + h_n, t) - f(x,t)}{h_n}\Bigr|
\leq \sup_{\xi} \Bigl|\frac{\partial f}{\partial x}(\xi, t)\Bigr|
\leq \psi(t) .
\]
Dominated convergence gives the limit $\int_I \frac{\partial
f}{\partial x}(x,t)\dd t$ of the quotients: $F$ is differentiable
with the announced derivative, which is continuous by
\cref{thm:b2:integration:continuity} applied to $\frac{\partial
f}{\partial x}$.
\end{proof}

\begin{example}[A parameter integral checked against a formula]\label{ex:b2:integration:arctanparam}
Let $F(x) = \displaystyle\int_0^\infty \frac{\dd t}{t^2 + x}$
for $x > 0$. On every $\intcc{a}{b} \subset \intoo{0}{\infty}$,
the integrand is dominated by $\frac{1}{t^2 + a}$, integrable
and independent of $x$: $F$ is continuous
(\cref{thm:b2:integration:continuity}). Here the theorem can be
checked against an explicit value:
\[
F(x) = \Bigl[\frac{1}{\sqrt x}\arctan\frac{t}{\sqrt
x}\Bigr]_0^\infty = \frac{\pi}{2\sqrt x} ,
\]
visibly continuous. Now differentiate under the integral: the
$x$-derivative $-\frac{1}{(t^2+x)^2}$ is dominated on
$\intcc ab$ by $\frac{1}{(t^2+a)^2}$, integrable:
\cref{thm:b2:integration:leibnizrule} gives
\[
F'(x) = -\int_0^\infty \frac{\dd t}{(t^2 + x)^2}
\qquad\text{while}\qquad
F'(x) = -\frac{\pi}{4}\,x^{-3/2} ,
\]
so we have \emph{computed} a new integral for free:
$\int_0^\infty\frac{\dd t}{(t^2+x)^2} = \frac{\pi}{4x^{3/2}}$.
Closing insight: differentiating a known parameter integral is a
factory of new formulas --- iterating gives
$\int_0^\infty\frac{\dd t}{(t^2+1)^n}$ for every $n$, with no
trigonometric substitutions.
\end{example}

\begin{method}[Studying an improper integral]\label{met:b2:integration:study}
Given $\int_a^b f$:
\begin{enumerate}
  \item Locate the trouble: list the endpoints (or interior
        points) where $f$ is unbounded or the interval is
        infinite, and split so that each piece has exactly one
        troublesome end.
  \item If $f$ has constant sign near that end, find an
        equivalent and compare with the reference scales of
        \cref{thm:b2:integration:comparison}.
  \item If $f$ oscillates, test $\abs f$ first (absolute
        convergence). If $\int\abs f$ diverges, try integration
        by parts to trade the oscillation for decay, as in
        \cref{ex:b2:integration:sint}; minorations like
        $\abs{\sin t} \geq \sin^2t$ detect genuine
        semi-convergence.
  \item For a value, not just the nature: parts, substitution,
        or a parameter (differentiate a simpler integral, as in
        \cref{ex:b2:integration:arctanparam} and
        \cref{ex:b2:integration:laplace}).
  \item Sanity checks on any computed value: sign and rough
        size against a crude bound ($\int_0^\infty
        \eu^{-t^2}\dd t \in \intoo{0}{1 + \int_1^\infty
        \eu^{-t}}$, so $\frac{\sqrt\pi}{2} \approx 0.886$ is
        plausible); and dimensional consistency under scaling
        ($t \mapsto \lambda t$ must rescale both sides the
        same way --- the fastest detector of a lost factor).
\end{enumerate}
\end{method}

\begin{remark}[Common pitfalls]
Three recurring errors. \emph{(i) Parameter-dependent
dominators:} the domination $\abs{f(x,t)} \leq \varphi(t)$ must
be uniform in $x$ on the set considered; it usually holds on
segments $\intcc ab$ but not globally --- for
$\int_0^\infty\eu^{-xt}\dd t$ there is no integrable dominator
valid for all $x > 0$, yet dominating on $x \geq a > 0$ is
enough to work on the whole open half-line, since continuity
and derivatives are local notions. \emph{(ii) Comparing signed
integrands:} the comparison toolkit is for \emph{nonnegative}
functions; from $\abs f \leq g$ with $\int g$ divergent one may
conclude nothing --- $\int_1^\infty\frac{\sin t}t\,\dd t$
converges although every comparison with $\frac1t$ fails.
\emph{(iii) Forgetting half the trouble:} on
$\intoo{0}{\infty}$ always study both ends separately;
$\int_0^\infty\frac{\dd t}{t}$ diverges at \emph{both}, and a
convergent-looking split can silently cancel two infinities.
The safe reflex is the checklist of
\cref{met:b2:integration:study}.
\end{remark}

\begin{example}[A Bertrand boundary case, to the digit]\label{ex:b2:integration:bertrandboundary}
The scale $\int^\infty\frac{\dd t}{t(\ln t)^\beta}$ of
\cref{thm:b2:integration:comparison} sits exactly on the edge
of the power scales; its boundary cases deserve one full
computation. For $\beta = 2$:
\[
\int_\eu^{\infty}\frac{\dd t}{t(\ln t)^2}
= \Bigl[-\frac{1}{\ln t}\Bigr]_\eu^{\infty}
= 0 - (-1) = 1 ,
\]
a convergent integral with a pleasantly exact value; while for
$\beta = 1$,
\[
\int_\eu^{x}\frac{\dd t}{t\ln t}
= \bigl[\ln\ln t\bigr]_\eu^{x} = \ln\ln x
\longrightarrow \infty ,
\]
divergent --- but so slowly that reaching $\ln\ln x = 10$
requires $x = \eu^{\eu^{10}} \approx 10^{9566}$. Closing
insight: between ``every power $t^{-1-\varepsilon}$
converges'' and ``$t^{-1}$ diverges'' lives an infinite ladder
of logarithmic scales, each refining the last; the substitution
$u = \ln t$ collapses each rung onto the previous one, which
is why the Bertrand criteria echo the Riemann ones one level
up.
\end{example}

\begin{definition}[The $\Gamma$ function]\label{def:b2:integration:gamma}
For $x > 0$:\index{Gamma function}
\[
\Gamma(x) = \int_0^{\infty} t^{x-1}\,\eu^{-t}\,\dd t ,
\]
convergent at both ends ($t^{x-1}$ integrable at $0^+$ for $x > 0$;
exponential decay at $\infty$).
\end{definition}

\begin{theorem}\label{thm:b2:integration:gammaprops}
$\Gamma$ is continuous on $\intoo{0}{+\infty}$, satisfies the
functional equation
\[
\Gamma(x + 1) = x\,\Gamma(x),
\qquad \Gamma(1) = 1,
\qquad\text{hence}\qquad \Gamma(n + 1) = n! ,
\]
and is of class $C^1$ (indeed $C^\infty$) with $\Gamma'(x) =
\int_0^\infty t^{x-1}\eu^{-t}\ln t\,\dd t$.
\end{theorem}

\begin{proof}
Functional equation: integrate by parts on $\intcc{\varepsilon}{M}$
and let the ends go: $\int t^{x}\eu^{-t} = [-t^x\eu^{-t}] + x\int
t^{x-1}\eu^{-t}$, boundary terms vanishing --- indeed
$\varepsilon^x\eu^{-\varepsilon} \to 0$ as $\varepsilon \to
0^+$ because $x > 0$, and $M^x\eu^{-M} \to 0$ as $M \to \infty$
because the exponential beats every power; both truncated
integrals converge to their improper values by the convergence
established in \cref{def:b2:integration:gamma}. $\Gamma(1) = \int
\eu^{-t} = 1$; induction gives the factorial.

Continuity on $\intcc{a}{b} \subset \intoo{0}{\infty}$: dominate
$t^{x-1}\eu^{-t}$ by $\varphi(t) = (t^{a-1} + t^{b-1})\eu^{-t}$,
integrable and independent of $x \in \intcc{a}{b}$:
\cref{thm:b2:integration:continuity} applies on every such segment,
hence on the whole half-line. Differentiability: the $x$-derivative
$t^{x-1}\eu^{-t}\ln t$ is dominated on $\intcc{a}{b}$ by $(t^{a-1}
+ t^{b-1})\eu^{-t}\,\abs{\ln t}$, still integrable:
\cref{thm:b2:integration:leibnizrule}; iterating gives all
derivatives (each adds a power of $\ln t$, harmless).
\end{proof}

\begin{example}[Half-integer factorials]\label{ex:b2:integration:halffactorial}
The functional equation and $\Gamma\bigl(\frac12\bigr) =
\sqrt\pi$ (a substitution away from
\cref{exo:b2:integration:8}: set $t = u^2$ in the defining
integral) generate all half-integer values:
\[
\Gamma\Bigl(\frac32\Bigr) = \frac12\,\Gamma\Bigl(\frac12\Bigr)
= \frac{\sqrt\pi}{2},
\qquad
\Gamma\Bigl(\frac52\Bigr) = \frac32\cdot\frac{\sqrt\pi}{2}
= \frac{3\sqrt\pi}{4},
\qquad
\Gamma\Bigl(\frac72\Bigr) = \frac{15\sqrt\pi}{8} .
\]
Since $\Gamma(n+1) = n!$, it is fair to say ``$\frac12! =
\frac{\sqrt\pi}{2} \approx 0.886$'': the factorial has been
interpolated, and the interpolating curve dips below $1$
between $0! = 1$ and $1! = 1$ (its minimum $\approx 0.8856$ at
$x \approx 1.4616$ matches the convexity picture of the
weekend problem's Part I). Closing insight: nothing in the
integral $\int_0^\infty t^{x-1}\eu^{-t}\dd t$ privileges
integers --- the factorial's discreteness was an accident of
counting, and $\sqrt\pi$ is what lives between $1$ and $1$.
\end{example}

\begin{remark}[Where $\Gamma$ goes from here]
The weekend problem of this chapter builds the whole Euler
calculus around $\Gamma$: the Beta function, its integration-by-parts
recursions, the Wallis integrals as Beta values, and Gauss's
limit formula. The chapter on multiple integrals proves Euler's
Beta--Gamma formula for all arguments by a double integral; the
probability chapters meet $\Gamma$ again in the normalization of
the most common densities and in the moments of waiting times.
The Year 3 volume rebuilds $\Gamma$ on Lebesgue foundations,
proves the Bohr--Mollerup uniqueness theorem, and extends
Stirling's formula from integers to the real half-line by
dominated convergence.
\end{remark}

\begin{example}[A classical computation by differentiation]\label{ex:b2:integration:laplace}
For $x \in \R$, let $F(x) = \int_0^{\infty} \eu^{-t^2}\cos(xt)\,\dd
t$ (absolutely convergent, dominated by $\eu^{-t^2}$). By
\cref{thm:b2:integration:leibnizrule} (domination of the
$x$-derivative by $t\,\eu^{-t^2}$, integrable):
\[
F'(x) = -\int_0^\infty t\,\eu^{-t^2}\sin(xt)\,\dd t
= \Bigl[\frac{\eu^{-t^2}}{2}\sin(xt)\Bigr]_0^\infty
- \frac x2\int_0^\infty \eu^{-t^2}\cos(xt)\,\dd t
= -\frac x2\,F(x),
\]
(parts with $u' = t\eu^{-t^2}$). The differential equation $F' =
-\frac x2 F$ integrates to $F(x) = F(0)\,\eu^{-x^2/4}$: the
Gaussian-type integral reproduces itself. The constant $F(0) =
\int_0^\infty \eu^{-t^2}\dd t = \frac{\sqrt\pi}{2}$ is computed in
\cref{exo:b2:integration:8} --- and again, by double integration,
in \cref{ch:b2:multint}.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b2:integration:1}
Nature of:
$\displaystyle\int_0^1 \frac{\dd t}{\sqrt{t(1-t)}}$;
$\;\displaystyle\int_1^\infty \frac{\ln t}{t^2}\dd t$;
$\;\displaystyle\int_0^\infty \frac{\dd t}{1 + t^2\sin^2 t}$
\emph{(compare with the divergent harmonic-type behavior near $t =
n\pi$)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:integration:2}
Compute $\displaystyle\int_0^\infty t^n \eu^{-\lambda t}\,\dd t$
($\lambda > 0$) via $\Gamma$, and
$\displaystyle\int_0^1 (\ln t)^n \dd t$ via the substitution $t =
\eu^{-u}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:integration:3}
Prove that $\displaystyle\int_0^{\infty} \frac{\dd t}{(1 +
t^2)(1 + t^x)}$ is well defined for every $x \in \R$ and
independent of $x$. \emph{(Substitute $t \mapsto \frac1t$ and
average the two expressions.)} What is its value?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:4}
(Bertrand integrals at a finite endpoint) For which $(\alpha,
\beta)$ does $\displaystyle\int_0^{1/2}
\frac{\dd t}{t^\alpha\,\abs{\ln t}^\beta}$ converge?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:5}
Let $F(x) = \displaystyle\int_0^{\infty} \frac{\eu^{-xt}}{1 +
t^2}\,\dd t$ for $x \geq 0$. Prove that $F$ is continuous on
$\intco{0}{\infty}$, $C^2$ on $\intoo{0}{\infty}$, satisfies $F'' +
F = \frac1x$ there, and that $F(x) \to 0$ as $x \to +\infty$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:6}
(Frullani) Let $f$ be continuous on $\intco{0}{+\infty}$ with a
finite limit $f(\infty)$ at $+\infty$. Prove that for $a, b > 0$:
\[
\int_0^{\infty} \frac{f(at) - f(bt)}{t}\,\dd t
= \bigl(f(0) - f(\infty)\bigr)\,\ln\frac ba .
\]
\emph{(On $\intcc{\varepsilon}{M}$, substitute in each piece and
regroup into $\int_{a\varepsilon}^{b\varepsilon} -
\int_{aM}^{bM}$ of $\frac{f(u)}u\,\dd u$; squeeze using the
continuity at $0$ and the limit at $\infty$.)} Compute
$\int_0^\infty \frac{\eu^{-t} - \eu^{-2t}}{t}\dd t$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:7}
Justify and compute $\lim_{n\to\infty} \displaystyle\int_0^n
\Bigl(1 - \frac tn\Bigr)^{\!n} t^{x-1}\,\dd t$ for $x > 0$
\emph{(dominated convergence with $\varphi(t) = \eu^{-t}t^{x-1}$,
using $(1 - t/n)^n \leq \eu^{-t}$; the limit is $\Gamma(x)$)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:integration:8}
(The Gaussian integral by a parameter trick) For $x \geq 0$ set
\[
G(x) = \Bigl(\int_0^x \eu^{-t^2}\dd t\Bigr)^{\!2},
\qquad
H(x) = \int_0^1 \frac{\eu^{-x^2(1+t^2)}}{1 + t^2}\,\dd t .
\]
Prove that $G' + H' = 0$ (differentiate $H$ under the integral and
substitute $u = xt$ in the resulting integral), deduce $G(x) + H(x)
= \frac\pi4$ for all $x$, and conclude
\[
\int_0^{\infty} \eu^{-t^2}\,\dd t = \frac{\sqrt\pi}{2} .
\]
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:integration:9}
Prove that $\Gamma$ is \emph{log-convex}: $\ln\Gamma$ is convex on
$\intoo{0}{\infty}$. \emph{(Cauchy--Schwarz for integrals applied
to $t^{(x+y)/2 - 1}\eu^{-t} = \bigl(t^{x-1}\eu^{-t}\bigr)^{1/2}
\bigl(t^{y-1}\eu^{-t}\bigr)^{1/2}$ gives
$\Gamma\bigl(\frac{x+y}{2}\bigr)^2 \leq \Gamma(x)\Gamma(y)$;
combine with continuity and \cref{exo:b2:realfun:8}.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:integration:10}
(Dirichlet integral) Set $F(x) = \displaystyle\int_0^{\infty}
\frac{\sin t}{t}\,\eu^{-xt}\,\dd t$ for $x > 0$.
\begin{enumerate}
  \item Justify $F'(x) = -\frac{1}{1 + x^2}$ (differentiate under
        the integral; compute $\int_0^\infty \eu^{-xt}\sin t\,\dd
        t$ by two integrations by parts).
  \item Prove $F(x) \to 0$ as $x \to +\infty$ and deduce $F(x) =
        \frac\pi2 - \arctan x$.
  \item Admitting the continuity of $F$ at $0^+$ (an Abel-type
        theorem), conclude the value of the semi-convergent
        integral:
        \[
        \int_0^{\infty} \frac{\sin t}{t}\,\dd t = \frac{\pi}{2}.
        \]
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:11}
Justify the convergence of $\displaystyle\int_0^\infty
\Bigl(\frac{\sin t}{t}\Bigr)^{\!2}\dd t$, then compute it by one
integration by parts and \cref{exo:b2:integration:10}:
\[
\int_0^\infty \Bigl(\frac{\sin t}{t}\Bigr)^{\!2}\dd t
= \frac{\pi}{2} .
\]
(The same value as $\int_0^\infty \frac{\sin t}{t}\dd t$ --- but
this time the convergence is absolute.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:integration:12}
(The Gaussian tail) For $x > 0$ set $T(x) = \displaystyle
\int_x^\infty \eu^{-t^2}\dd t$.
\begin{enumerate}
  \item Writing $\eu^{-t^2} = \frac{1}{-2t}\cdot(-2t\,\eu^{-t^2})$,
        integrate by parts twice to obtain
        \[
        T(x) = \eu^{-x^2}\Bigl(\frac{1}{2x} -
        \frac{1}{4x^3}\Bigr) +
        \frac34\int_x^\infty \frac{\eu^{-t^2}}{t^4}\,\dd t .
        \]
  \item Bound the remainder: $0 \leq \frac34\int_x^\infty
        t^{-4}\eu^{-t^2}\dd t \leq \frac{3}{8x^5}\,\eu^{-x^2}$,
        and deduce the bracketing
        \[
        \eu^{-x^2}\Bigl(\frac{1}{2x} - \frac{1}{4x^3}\Bigr)
        \leq T(x) \leq \frac{\eu^{-x^2}}{2x},
        \qquad\text{hence}\qquad
        T(x) \sim \frac{\eu^{-x^2}}{2x} \quad (x \to +\infty).
        \]
  \item Why can the full alternating series obtained by iterating
        the parts never converge for fixed $x$? \emph{(Compare the
        growth of the coefficients $1\cdot3\cdots(2k-1)$ with the
        powers $(2x^2)^k$.)}
\end{enumerate}
\end{exercise}

\section{Problem: Euler's integrals --- Beta, Gamma, and Gauss's
limit formula}

\begin{problem}\label{pb:b2:integration:1}
The $\Gamma$ function of \cref{def:b2:integration:gamma} is one
half of Euler's calculus of integrals; the other half is the
\emph{Beta function}\index{Beta function}
\[
B(x, y) = \int_0^1 t^{x-1}(1 - t)^{y-1}\,\dd t .
\]
This problem develops the pair $(\Gamma, B)$ with the tools of
this chapter only --- integration by parts, substitution,
dominated convergence --- and culminates in \emph{Euler's
Beta--Gamma formula} $B(x,y) = \frac{\Gamma(x)\Gamma(y)}
{\Gamma(x+y)}$ on the half-integers and in \emph{Gauss's limit
formula}\index{Gauss's limit formula} for $\Gamma$. Along the way
the Wallis integrals of \cref{lem:b2:comparison:wallis} reappear
as Beta values, and Legendre's duplication formula drops out.

\medskip
\noindent\textbf{Part I --- Fine structure of $\Gamma$.}
\begin{enumerate}
  \item Recall why $\Gamma(x) = \int_0^\infty t^{x-1}\eu^{-t}\dd
        t$ converges exactly for $x > 0$, and show
        \[
        \Gamma(x) \sim \frac1x \qquad (x \to 0^+)
        \]
        \emph{(functional equation plus continuity of $\Gamma$ at
        $1$)}.
  \item Prove $\Gamma\bigl(\tfrac12\bigr) = \sqrt\pi$
        \emph{(substitute $t = u^2$ and invoke
        \cref{exo:b2:integration:8})}, and deduce
        $\int_\R \eu^{-u^2/2}\dd u = \sqrt{2\pi}$.
  \item Show by induction, for $n \in \N$:
        \[
        \Gamma\Bigl(n + \frac12\Bigr)
        = \frac{(2n)!}{4^n\,n!}\,\sqrt\pi .
        \]
  \item Justify $\Gamma''(x) = \int_0^\infty
        t^{x-1}\eu^{-t}(\ln t)^2\dd t > 0$, and deduce that
        $\Gamma$ is strictly convex, attains a unique minimum at
        some $x_0 \in \intoo{1}{2}$ \emph{($\Gamma(1) = \Gamma(2)
        = 1$ and Rolle)}, decreases on $\intoo{0}{x_0}$ and
        increases on $\intoo{x_0}{\infty}$.
  \item Show that $\Gamma$ beats every power: for each $k \in
        \N$, $x^k = o\bigl(\Gamma(x)\bigr)$ as $x \to +\infty$
        \emph{(squeeze $x$ between integers and use $\Gamma(n+1)
        = n!$ with the monotonicity of question 4)}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The Beta function, by parts.}
\begin{enumerate}[resume]
  \item Show that $B(x,y)$ converges exactly for $x > 0$ and $y >
        0$, and that $B(x,y) = B(y,x)$.
  \item Compute $B(x, 1) = \frac1x$, and prove by integration by
        parts, for $x, y > 0$:
        \[
        B(x, y+1) = \frac{y}{x}\,B(x+1, y) .
        \]
  \item From the splitting $t^{x-1}(1-t)^{y-1} =
        t^{x}(1-t)^{y-1} + t^{x-1}(1-t)^{y}$ deduce $B(x,y) =
        B(x+1,y) + B(x,y+1)$, and combine with question 7 into
        the \emph{descent relations}
        \[
        B(x, y+1) = \frac{y}{x+y}\,B(x,y),
        \qquad
        B(x+1, y) = \frac{x}{x+y}\,B(x,y) .
        \]
  \item Deduce, for $m, n \geq 1$ integers:
        \[
        B(m, n) = \frac{(m-1)!\,(n-1)!}{(m+n-1)!}
        = \frac{1}{(m+n-1)\binom{m+n-2}{m-1}} .
        \]
  \item Prove \emph{Euler's formula for one integer argument}:
        for every $x > 0$ and $n \in \N^*$,
        \[
        B(x, n) = \frac{\Gamma(x)\,\Gamma(n)}{\Gamma(x + n)}
        \]
        \emph{(induction on $n$: both sides equal $\frac1x$ at $n
        = 1$ and obey the same descent relation)}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Wallis integrals as Beta values.}
\begin{enumerate}[resume]
  \item Substitute $t = \sin^2\theta$ to obtain the trigonometric
        form
        \[
        B(x, y) = 2\int_0^{\pi/2}
        \sin^{2x-1}\theta\,\cos^{2y-1}\theta\,\dd\theta .
        \]
  \item Deduce $W_n = \frac12\,B\bigl(\frac{n+1}2, \frac12\bigr)$
        for the Wallis integral $W_n = \int_0^{\pi/2}\sin^n
        \theta\,\dd\theta$, and recover the recurrence $W_n =
        \frac{n-1}{n}W_{n-2}$ of \cref{lem:b2:comparison:wallis}
        from the descent relations of question 8 alone.
  \item Compute $B\bigl(\frac12, \frac12\bigr) = 2W_0 = \pi$ and
        check it against $\Gamma\bigl(\frac12\bigr)^2/\Gamma(1)$:
        Euler's formula holds at $\bigl(\frac12,
        \frac12\bigr)$.
  \item Derive the closed form $W_{2n} = \frac\pi2\,
        \frac{(2n)!}{4^n(n!)^2}$ from the recurrence, and verify
        \[
        B\Bigl(n + \frac12, \frac12\Bigr)
        = \frac{\Gamma\bigl(n + \frac12\bigr)\Gamma\bigl(
        \frac12\bigr)}{\Gamma(n+1)} .
        \]
        Conclude, by induction with the descent relations, that
        Euler's formula $B(x,y) =
        \frac{\Gamma(x)\Gamma(y)}{\Gamma(x+y)}$ holds whenever
        $2x$ and $2y$ are positive integers.
  \item Substitute $u = \frac{t}{1-t}$ to obtain the third
        classical form
        \[
        B(x,y) = \int_0^\infty
        \frac{u^{x-1}}{(1+u)^{x+y}}\,\dd u ,
        \]
        and check the case $x = y = \frac12$ directly ($u = v^2$
        reduces it to $\int_0^\infty\frac{2\,\dd v}{1+v^2}$).
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Gauss's limit formula.}
\begin{enumerate}[resume]
  \item For $x > 0$ and $n \in \N^*$, prove by $n$ successive
        integrations by parts:
        \[
        \int_0^n \Bigl(1 - \frac tn\Bigr)^{\!n} t^{x-1}\,\dd t
        = \frac{n!\;n^x}{x(x+1)\cdots(x+n)} .
        \]
  \item Conclude with \cref{exo:b2:integration:7} (dominated
        convergence) \emph{Gauss's limit formula}:
        \[
        \Gamma(x) = \lim_{n\to\infty}
        \frac{n!\;n^x}{x(x+1)\cdots(x+n)}
        \qquad (x > 0).
        \]
  \item Taking logarithms, show that for $x > 0$:
        \[
        \ln\Gamma(x) = -\ln x - \gamma x +
        \sum_{k=1}^{\infty}\Bigl(\frac xk -
        \ln\Bigl(1 + \frac xk\Bigr)\Bigr),
        \]
        where $\gamma$ is Euler's constant
        (\cref{ex:b2:comparison:harmonic}); justify the
        convergence of the series \emph{(the general term is
        $\sim \frac{x^2}{2k^2}$)}.
  \item Use Gauss's formula at $x = \frac12$ and the central
        binomial asymptotics $\binom{2n}{n} \sim
        \frac{4^n}{\sqrt{\pi n}}$
        (\cref{ex:b2:comparison:centralbinomial}) to recompute
        $\Gamma\bigl(\frac12\bigr) = \sqrt\pi$: Stirling's
        constant and the Gaussian integral are the same number
        in two disguises.
  \item Check that Gauss's formula reproves the functional
        equation: from the exact identity
        \[
        \frac{n!\,n^{x+1}}{(x+1)\cdots(x+n+1)}
        = \frac{n!\,n^{x}}{x(x+1)\cdots(x+n)}\cdot
        \frac{n\,x}{x+n+1},
        \]
        conclude $\Gamma(x+1) = x\,\Gamma(x)$ again. (Gauss's
        formula determines $\Gamma$ outright; the Year 3 volume
        proves the sharper Bohr--Mollerup theorem: the
        functional equation plus log-convexity already pin
        $\Gamma$ down.)
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends.}
\begin{enumerate}[resume]
  \item For $a > 0$ show $\int_0^\infty \eu^{-t^a}\dd t =
        \Gamma\bigl(1 + \frac1a\bigr)$, and compute the limit as
        $a \to +\infty$ by dominated convergence \emph{(pointwise
        limit $\mathbf 1_{t < 1}$; dominate by $1$ on
        $\intoc{0}{1}$ and by $\eu^{-t^2}$ beyond, for $a \geq
        2$)}. Check the answer against the continuity of
        $\Gamma$.
  \item For $n \geq 1$ show
        \[
        \int_0^1 \frac{\dd t}{\sqrt{1 - t^n}}
        = \frac1n\,B\Bigl(\frac1n, \frac12\Bigr),
        \]
        and recover the values $2$ ($n = 1$) and $\frac\pi2$ ($n
        = 2$). (For $n = 4$ this is the \emph{lemniscate
        constant}, which has no elementary closed form; its
        story belongs to the theory of elliptic integrals.)
  \item (Moments) For $x > 0$ and $k \in \N$, show
        \[
        \frac{1}{\Gamma(x)}\int_0^\infty
        t^{k}\,t^{x-1}\eu^{-t}\,\dd t
        = \frac{\Gamma(x+k)}{\Gamma(x)}
        = x(x+1)\cdots(x+k-1),
        \]
        the rising factorial; check that $x = 1$ gives $k!$. (In
        the probability chapters this is the $k$-th moment of a
        standard waiting-time density.)
  \item Prove the Beta identity, valid for \emph{all} $x > 0$:
        \[
        B(x, x) = 2^{1-2x}\,B\Bigl(x, \frac12\Bigr)
        \]
        \emph{(substitute $t = \frac{1+s}2$, exploit the
        symmetry in $s$, then set $s = \sqrt v$)}. Deduce, for
        $2x \in \N^*$, \emph{Legendre's duplication formula}
        \[
        \Gamma(x)\,\Gamma\Bigl(x + \frac12\Bigr)
        = 2^{1-2x}\,\sqrt\pi\;\Gamma(2x),
        \]
        and verify it directly at $x = n$ via question 3. (For
        general $x$ it follows from the same identity once
        Euler's formula is known for all arguments --- the
        double-integral proof in the chapter on multiple
        integrals.)
  \item Synthesis. In one sentence each: (i) where integration
        by parts carried the whole of Part II; (ii) where
        dominated convergence entered Parts IV and V; (iii) which
        asymptotic inputs were imported from the comparison
        chapter; (iv) what is now proved of Euler's formula
        $B(x,y) = \Gamma(x)\Gamma(y)/\Gamma(x+y)$, and what
        remains for the double integral to settle.
\end{enumerate}
\end{problem}
