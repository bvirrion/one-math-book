\chapter{Integration}\label{ch:b2:integration}

The Year 1 volume built the integral on a segment. This chapter
extends it to arbitrary intervals (improper integrals, with the full
comparison toolkit), then studies integrals \emph{depending on a
parameter} --- continuity and differentiation under the integral
sign --- powered by the dominated convergence theorem, the one
result of this chapter taken on trust. The $\Gamma$ function serves
as the running example, and as the gateway to half of the special
functions of mathematics.

\section{Integrals on an arbitrary interval}

\begin{definition}\label{def:b2:integration:improper}
Let $f$ be piecewise continuous on $\intco{a}{b}$ ($b \in \R$ or
$+\infty$). The integral \emph{converges}\index{improper integral}
when $\lim_{x \to b^-} \int_a^x f$ exists; one then writes
$\int_a^b f$ for the limit. (Similarly on $\intoc{a}{b}$, and on
$\intoo{a}{b}$ by splitting at an interior point --- the choice does
not matter, by Chasles.) The integral \emph{converges absolutely}
when $\int_a^b \abs f$ converges; absolute convergence implies
convergence, by the Cauchy criterion:
\[
\Bigl| \int_x^{y} f \Bigr| \leq \int_x^{y} \abs f
\]
and completeness of $\R$ (the primitive has the Cauchy property).
\end{definition}

\begin{theorem}[Positive comparison toolkit]\label{thm:b2:integration:comparison}
For $f, g \geq 0$ piecewise continuous on $\intco{a}{b}$:
\begin{enumerate}
  \item $\int_a^b f$ converges iff the primitive $x \mapsto \int_a^x
        f$ is bounded;
  \item $f \leq g$: convergence of $\int g$ forces that of $\int f$;
        divergence transfers the other way;
  \item $f \sim g$ at $b$: the two integrals have the same nature;
  \item the reference scales: at $+\infty$, $\int^{\infty}
        \frac{\dd t}{t^\alpha}$ converges iff $\alpha > 1$, and
        $\int^\infty \frac{\dd t}{t(\ln t)^\beta}$ iff $\beta > 1$;
        at a finite endpoint $b$, $\int^b \frac{\dd t}{(b -
        t)^\alpha}$ converges iff $\alpha < 1$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) The primitive is increasing; bounded increasing functions have
limits (monotone limit theorem transferred to functions: the
supremum works). (2) Compare primitives. (3) For large $x$,
$\frac12 g \leq f \leq 2g$: comparison both ways. (4) Compute the
primitives explicitly ($\frac{x^{1-\alpha}}{1-\alpha}$, $\ln$,
$\frac{(\ln x)^{1-\beta}}{1-\beta}$) and apply (1).
\end{proof}

\begin{example}[A semi-convergent integral]\label{ex:b2:integration:sint}
$\displaystyle\int_1^{\infty} \frac{\sin t}{t}\,\dd t$ converges:
integrate by parts,
\[
\int_1^x \frac{\sin t}{t}\dd t
= \Bigl[\frac{-\cos t}{t}\Bigr]_1^x - \int_1^x \frac{\cos
t}{t^2}\dd t ,
\]
where the bracket has a limit and the last integral converges
absolutely ($\abs{\cos t}/t^2 \leq t^{-2}$). But not absolutely:
$\abs{\sin t} \geq \sin^2 t = \frac{1 - \cos 2t}{2}$, and
$\int \frac{1 - \cos 2t}{2t}$ diverges (the $\frac{1}{2t}$ part
diverges; the $\frac{\cos 2t}{2t}$ part converges by the same
integration by parts). The integral analogue of the alternating
series.
\end{example}

\section{The convergence theorem}

\begin{theorem}[Dominated convergence]\label{thm:b2:integration:dominated}
Let $(f_n)$ be piecewise continuous on an interval $I$, converging
pointwise to a piecewise continuous $f$, and suppose there is a
fixed integrable $\varphi \geq 0$ ($\int_I \varphi < \infty$)
with\index{dominated convergence theorem}
\[
\abs{f_n(t)} \leq \varphi(t) \qquad (t \in I,\ n \in \N).
\]
Then all $\int_I f_n$ and $\int_I f$ converge absolutely, and
\[
\int_I f_n \xrightarrow[n \to \infty]{} \int_I f .
\]
\end{theorem}
\admitted

\begin{remark}
The honest proof belongs to the Lebesgue integration theory of Year
3; the statement, however, is used constantly from now on. The
\emph{domination} hypothesis is the whole point: pointwise
convergence alone does not suffice ($f_n = n\,\mathbf{1}_{\intoo{0}{1/n}}$,
sliding bumps: $\int f_n = 1 \not\to 0 = \int f$). The theorem
also holds for a continuous parameter ($f_\lambda$, $\lambda \to
\lambda_0$), by the sequential characterization of limits.
\end{remark}

\section{Integrals with a parameter}

\begin{theorem}[Continuity under the integral sign]\label{thm:b2:integration:continuity}
Let $f \colon A \times I \to \R$ ($A$ a metric space, $I$ an
interval) with: $t \mapsto f(x, t)$ piecewise continuous for each
$x$; $x \mapsto f(x, t)$ continuous for each $t$; and a
\emph{domination} $\abs{f(x,t)} \leq \varphi(t)$ ($\varphi$
integrable on $I$, independent of $x$). Then
\[
F(x) = \int_I f(x, t)\,\dd t
\]
is defined and continuous on $A$.\index{parameter integral}
\end{theorem}

\begin{proof}
Definedness: domination gives absolute convergence. Continuity at
$x_0$: for any sequence $x_n \to x_0$, the functions $g_n(t) =
f(x_n, t)$ converge pointwise to $f(x_0, t)$ (continuity in $x$)
under the fixed domination $\varphi$: dominated convergence gives
$F(x_n) \to F(x_0)$; conclude by the sequential characterization of
continuity (\cref{def:b2:metric:continuity}).
\end{proof}

\begin{theorem}[Differentiation under the integral sign]\label{thm:b2:integration:leibnizrule}
Let $f \colon J \times I \to \R$ ($J$ an interval of parameters)
with: $t \mapsto f(x,t)$ integrable on $I$ for each $x$; $x \mapsto
f(x,t)$ of class $C^1$ for each $t$, the partial derivative
$\frac{\partial f}{\partial x}$ being piecewise continuous in $t$
and dominated: $\bigl|\frac{\partial f}{\partial x}(x,t)\bigr| \leq
\psi(t)$ with $\psi$ integrable. Then $F(x) = \int_I f(x,t)\dd t$
is $C^1$ on $J$ and
\[
F'(x) = \int_I \frac{\partial f}{\partial x}(x, t)\,\dd t .
\]
\end{theorem}

\begin{proof}
Fix $x$ and $h_n \to 0$. The difference quotients
\[
\frac{F(x + h_n) - F(x)}{h_n}
= \int_I \frac{f(x + h_n, t) - f(x, t)}{h_n}\,\dd t
\]
have integrands converging pointwise to $\frac{\partial f}{\partial
x}(x, t)$, and dominated by $\psi(t)$: by the mean value
inequality applied in $x$ at fixed $t$,
\[
\Bigl|\frac{f(x + h_n, t) - f(x,t)}{h_n}\Bigr|
\leq \sup_{\xi} \Bigl|\frac{\partial f}{\partial x}(\xi, t)\Bigr|
\leq \psi(t) .
\]
Dominated convergence gives the limit $\int_I \frac{\partial
f}{\partial x}(x,t)\dd t$ of the quotients: $F$ is differentiable
with the announced derivative, which is continuous by
\cref{thm:b2:integration:continuity} applied to $\frac{\partial
f}{\partial x}$.
\end{proof}

\begin{definition}[The $\Gamma$ function]\label{def:b2:integration:gamma}
For $x > 0$:\index{Gamma function}
\[
\Gamma(x) = \int_0^{\infty} t^{x-1}\,\eu^{-t}\,\dd t ,
\]
convergent at both ends ($t^{x-1}$ integrable at $0^+$ for $x > 0$;
exponential decay at $\infty$).
\end{definition}

\begin{theorem}\label{thm:b2:integration:gammaprops}
$\Gamma$ is continuous on $\intoo{0}{+\infty}$, satisfies the
functional equation
\[
\Gamma(x + 1) = x\,\Gamma(x),
\qquad \Gamma(1) = 1,
\qquad\text{hence}\qquad \Gamma(n + 1) = n! ,
\]
and is of class $C^1$ (indeed $C^\infty$) with $\Gamma'(x) =
\int_0^\infty t^{x-1}\eu^{-t}\ln t\,\dd t$.
\end{theorem}

\begin{proof}
Functional equation: integrate by parts on $\intcc{\varepsilon}{M}$
and let the ends go: $\int t^{x}\eu^{-t} = [-t^x\eu^{-t}] + x\int
t^{x-1}\eu^{-t}$, boundary terms vanishing. $\Gamma(1) = \int
\eu^{-t} = 1$; induction gives the factorial.

Continuity on $\intcc{a}{b} \subset \intoo{0}{\infty}$: dominate
$t^{x-1}\eu^{-t}$ by $\varphi(t) = (t^{a-1} + t^{b-1})\eu^{-t}$,
integrable and independent of $x \in \intcc{a}{b}$:
\cref{thm:b2:integration:continuity} applies on every such segment,
hence on the whole half-line. Differentiability: the $x$-derivative
$t^{x-1}\eu^{-t}\ln t$ is dominated on $\intcc{a}{b}$ by $(t^{a-1}
+ t^{b-1})\eu^{-t}\,\abs{\ln t}$, still integrable:
\cref{thm:b2:integration:leibnizrule}; iterating gives all
derivatives (each adds a power of $\ln t$, harmless).
\end{proof}

\begin{example}[A classical computation by differentiation]\label{ex:b2:integration:laplace}
For $x \in \R$, let $F(x) = \int_0^{\infty} \eu^{-t^2}\cos(xt)\,\dd
t$ (absolutely convergent, dominated by $\eu^{-t^2}$). By
\cref{thm:b2:integration:leibnizrule} (domination of the
$x$-derivative by $t\,\eu^{-t^2}$, integrable):
\[
F'(x) = -\int_0^\infty t\,\eu^{-t^2}\sin(xt)\,\dd t
= \Bigl[\frac{\eu^{-t^2}}{2}\sin(xt)\Bigr]_0^\infty
- \frac x2\int_0^\infty \eu^{-t^2}\cos(xt)\,\dd t
= -\frac x2\,F(x),
\]
(parts with $u' = t\eu^{-t^2}$). The differential equation $F' =
-\frac x2 F$ integrates to $F(x) = F(0)\,\eu^{-x^2/4}$: the
Gaussian-type integral reproduces itself. The constant $F(0) =
\int_0^\infty \eu^{-t^2}\dd t = \frac{\sqrt\pi}{2}$ is computed in
\cref{exo:b2:integration:8} --- and again, by double integration,
in \cref{ch:b2:multint}.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b2:integration:1}
Nature of:
$\displaystyle\int_0^1 \frac{\dd t}{\sqrt{t(1-t)}}$;
$\;\displaystyle\int_1^\infty \frac{\ln t}{t^2}\dd t$;
$\;\displaystyle\int_0^\infty \frac{\dd t}{1 + t^2\sin^2 t}$
\emph{(compare with the divergent harmonic-type behavior near $t =
n\pi$)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:integration:2}
Compute $\displaystyle\int_0^\infty t^n \eu^{-\lambda t}\,\dd t$
($\lambda > 0$) via $\Gamma$, and
$\displaystyle\int_0^1 (\ln t)^n \dd t$ via the substitution $t =
\eu^{-u}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:integration:3}
Prove that $\displaystyle\int_0^{\infty} \frac{\dd t}{(1 +
t^2)(1 + t^x)}$ is well defined for every $x \in \R$ and
independent of $x$. \emph{(Substitute $t \mapsto \frac1t$ and
average the two expressions.)} What is its value?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:4}
(Bertrand integrals at a finite endpoint) For which $(\alpha,
\beta)$ does $\displaystyle\int_0^{1/2}
\frac{\dd t}{t^\alpha\,\abs{\ln t}^\beta}$ converge?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:5}
Let $F(x) = \displaystyle\int_0^{\infty} \frac{\eu^{-xt}}{1 +
t^2}\,\dd t$ for $x \geq 0$. Prove that $F$ is continuous on
$\intco{0}{\infty}$, $C^2$ on $\intoo{0}{\infty}$, satisfies $F'' +
F = \frac1x$ there, and that $F(x) \to 0$ as $x \to +\infty$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:6}
(Frullani) Let $f$ be continuous on $\intco{0}{+\infty}$ with a
finite limit $f(\infty)$ at $+\infty$. Prove that for $a, b > 0$:
\[
\int_0^{\infty} \frac{f(at) - f(bt)}{t}\,\dd t
= \bigl(f(0) - f(\infty)\bigr)\,\ln\frac ba .
\]
\emph{(On $\intcc{\varepsilon}{M}$, substitute in each piece and
regroup into $\int_{a\varepsilon}^{b\varepsilon} -
\int_{aM}^{bM}$ of $\frac{f(u)}u\,\dd u$; squeeze using the
continuity at $0$ and the limit at $\infty$.)} Compute
$\int_0^\infty \frac{\eu^{-t} - \eu^{-2t}}{t}\dd t$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:7}
Justify and compute $\lim_{n\to\infty} \displaystyle\int_0^n
\Bigl(1 - \frac tn\Bigr)^{\!n} t^{x-1}\,\dd t$ for $x > 0$
\emph{(dominated convergence with $\varphi(t) = \eu^{-t}t^{x-1}$,
using $(1 - t/n)^n \leq \eu^{-t}$; the limit is $\Gamma(x)$)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:integration:8}
(The Gaussian integral by a parameter trick) For $x \geq 0$ set
\[
G(x) = \Bigl(\int_0^x \eu^{-t^2}\dd t\Bigr)^{\!2},
\qquad
H(x) = \int_0^1 \frac{\eu^{-x^2(1+t^2)}}{1 + t^2}\,\dd t .
\]
Prove that $G' + H' = 0$ (differentiate $H$ under the integral and
substitute $u = xt$ in the resulting integral), deduce $G(x) + H(x)
= \frac\pi4$ for all $x$, and conclude
\[
\int_0^{\infty} \eu^{-t^2}\,\dd t = \frac{\sqrt\pi}{2} .
\]
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:integration:9}
Prove that $\Gamma$ is \emph{log-convex}: $\ln\Gamma$ is convex on
$\intoo{0}{\infty}$. \emph{(Cauchy--Schwarz for integrals applied
to $t^{(x+y)/2 - 1}\eu^{-t} = \bigl(t^{x-1}\eu^{-t}\bigr)^{1/2}
\bigl(t^{y-1}\eu^{-t}\bigr)^{1/2}$ gives
$\Gamma\bigl(\frac{x+y}{2}\bigr)^2 \leq \Gamma(x)\Gamma(y)$;
combine with continuity and \cref{exo:b2:realfun:8}.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:integration:10}
(Dirichlet integral) Set $F(x) = \displaystyle\int_0^{\infty}
\frac{\sin t}{t}\,\eu^{-xt}\,\dd t$ for $x > 0$.
\begin{enumerate}
  \item Justify $F'(x) = -\frac{1}{1 + x^2}$ (differentiate under
        the integral; compute $\int_0^\infty \eu^{-xt}\sin t\,\dd
        t$ by two integrations by parts).
  \item Prove $F(x) \to 0$ as $x \to +\infty$ and deduce $F(x) =
        \frac\pi2 - \arctan x$.
  \item Admitting the continuity of $F$ at $0^+$ (an Abel-type
        theorem), conclude the value of the semi-convergent
        integral:
        \[
        \int_0^{\infty} \frac{\sin t}{t}\,\dd t = \frac{\pi}{2}.
        \]
\end{enumerate}
\end{exercise}
