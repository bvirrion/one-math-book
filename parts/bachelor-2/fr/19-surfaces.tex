\chapter{Surfaces}\label{ch:b2:surfaces}

Après les courbes, les surfaces~: des objets à deux paramètres
dans $\R^3$. Le calcul différentiel du~\cref{ch:b2:diffcalc}
fournit tout ce dont nous avons besoin --- les dérivées
partielles donnent les vecteurs tangents, le produit vectoriel
donne la normale, les \omterm{def:b2:linalg:det}{déterminants} donnent les \omterm{def:b2:surfaces:area}{aires}. Nous
définissons les \omterm{def:b2:surfaces:param}{surfaces paramétrées} régulières, leurs \omterm{def:b2:surfaces:tangent}{plans
tangents} et la \emph{\omterm{def:b2:surfaces:fff}{première forme fondamentale}}, qui encode
toutes les mesures de \omterm{def:b2:curves:length}{longueur} et d'\omterm{def:b2:surfaces:area}{aire} sur la surface. Les
surfaces apparaissent aussi comme ensembles de niveau
$f(x, y, z) = c$~; le gradient dirige alors la normale.

\section{Surfaces paramétrées}

\begin{definition}[Surface paramétrée régulière]\label{def:b2:surfaces:param}
Soit $U \subseteq \R^2$ un \omterm{def:b2:metric:topology}{ouvert}. Une \emph{surface
paramétrée}\index{surface paramétrée} de
classe $\mathcal{C}^k$ ($k \geq 1$) est une application $\sigma \colon U \to
\R^3$, $(u, v) \mapsto \sigma(u, v)$, de classe $\mathcal{C}^k$. Un
point est \emph{régulier} si les vecteurs dérivées partielles
\[
\sigma_u(u, v) = \frac{\partial\sigma}{\partial u}(u,v),
\qquad
\sigma_v(u, v) = \frac{\partial\sigma}{\partial v}(u,v)
\]
sont linéairement indépendants, c.-à-d.\ $\sigma_u \wedge \sigma_v \neq 0$~;
la surface est \emph{régulière} si tout point l'est.
\end{definition}

\begin{example}[Les trois descriptions standard]\label{ex:b2:surfaces:standard}
\leavevmode
\begin{enumerate}
\item \emph{Graphe~:} $\sigma(u, v) = (u,\ v,\ f(u, v))$ pour $f \in
\mathcal{C}^1(U)$. Toujours régulière~: $\sigma_u = (1, 0, f_u)$ et
$\sigma_v = (0, 1, f_v)$ sont indépendants.
\item \emph{Sphère (coordonnées sphériques)~:} pour la sphère de
rayon $R$,
\[
\sigma(\theta, \varphi)
= (R\cos\theta\cos\varphi,\ R\sin\theta\cos\varphi,\
R\sin\varphi),
\qquad
(\theta, \varphi) \in \R \times
\bigl(-\tfrac\pi2, \tfrac\pi2\bigr),
\]
avec $\theta$ la longitude et $\varphi$ la latitude. On vérifie
$\norm{\sigma_\theta \wedge \sigma_\varphi} = R^2\cos\varphi > 0$~:
régulière hors des pôles (que cette carte omet).
\item \emph{Ensemble de niveau~:} $S = \{(x,y,z) : f(x, y, z) = c\}$ où
$f$ est $\mathcal{C}^1$ et $\nabla f \neq 0$ sur $S$. Au voisinage de chaque
point, une coordonnée peut s'exprimer en fonction des deux
autres par le théorème des fonctions implicites (\cref{ch:b2:diffcalc}), donc
$S$ est localement un graphe.
\end{enumerate}
\end{example}

\begin{example}[De l'ensemble de niveau au graphe]
\label{ex:b2:surfaces:levelgraph}
Le théorème des fonctions implicites du point 3 mérite une mise
en œuvre explicite. Prenons la sphère $x^2 + y^2 + z^2 = R^2$ au
voisinage de son pôle nord $(0, 0, R)$~: là $\frac{\partial f}{\partial z} = 2z =
2R \neq 0$, et la résolution en $z$ donne la carte-graphe
\[
z = \sqrt{R^2 - x^2 - y^2},
\qquad x^2 + y^2 < R^2 ,
\]
régulière partout sur son domaine (\omterm{def:b2:metric:topology}{ouvert}) --- y compris le
pôle que la carte sphérique manquait. Au voisinage d'un point de
l'équateur comme $(R, 0, 0)$, le même théorème résout en $x$ à la
place ($\frac{\partial f}{\partial x} = 2R \neq 0$). La règle
empirique~: une surface de niveau est un graphe au-dessus du plan
de coordonnées \emph{orthogonal à la plus grande composante du
gradient}, et en recouvrant la sphère par six telles
cartes-graphes on vérifie sa régularité partout sans aucune
trigonométrie.
\end{example}

\section{Plan tangent et normale}

\begin{definition}[Plan tangent]\label{def:b2:surfaces:tangent}
Soit $\sigma$ régulière en $(u_0, v_0)$, $M_0 = \sigma(u_0, v_0)$.
Le \emph{plan tangent}\index{plan tangent} $T_{M_0}S$ est le plan passant par $M_0$
dirigé par $\operatorname{Vect}(\sigma_u, \sigma_v)$ (dérivées
partielles en $(u_0, v_0)$). La \emph{normale unitaire}\index{normale unitaire} est
\[
n(u_0, v_0)
= \frac{\sigma_u \wedge \sigma_v}
       {\norm{\sigma_u \wedge \sigma_v}} .
\]
\end{definition}

\begin{proposition}[Les vecteurs tangents sont des vecteurs
vitesse]\label{prop:b2:surfaces:velocity}
La direction de $T_{M_0}S$ est exactement l'ensemble des vecteurs
$\gamma'(0)$, où $\gamma = \sigma \circ c$ parcourt les courbes
$\mathcal{C}^1$ tracées sur la surface passant par $M_0$ (c.-à-d.\
$c \colon (-\varepsilon, \varepsilon) \to U$ est $\mathcal{C}^1$
avec $c(0) = (u_0, v_0)$).
\end{proposition}

\begin{proof}
Si $c(t) = (u(t), v(t))$, la règle de dérivation en chaîne
(\cref{ch:b2:diffcalc}) donne
\[
\gamma'(0) = u'(0)\,\sigma_u + v'(0)\,\sigma_v
\in \operatorname{Vect}(\sigma_u, \sigma_v).
\]
Réciproquement, le vecteur $a\sigma_u + b\sigma_v$ est atteint par la
courbe $c(t) = (u_0 + at,\ v_0 + bt)$, qui reste dans l'\omterm{def:b2:metric:topology}{ouvert}
$U$ pour $\abs t$ petit.
\end{proof}

\begin{example}[Le plan tangent de l'hélicoïde]
\label{ex:b2:surfaces:helicoidtangent}
Pour l'hélicoïde $\sigma(u, v) = (v\cos u,\ v\sin u,\ au)$,
au point $\sigma(0, 1) = (1, 0, 0)$~:
\[
\sigma_u = (0,\ 1,\ a), \qquad \sigma_v = (1,\ 0,\ 0),
\qquad
\sigma_u \wedge \sigma_v = (0,\ a,\ -1),
\]
donc le \omterm{def:b2:surfaces:tangent}{plan tangent} est $a\,y = z$. Il contient toute la
génératrice horizontale $\{(t, 0, 0)\}$ (direction $\sigma_v$)~:
comme pour le cône de l'\cref{exo:b2:surfaces:1}, une surface
réglée par des droites a chacune de ses génératrices contenue
dans le \omterm{def:b2:surfaces:tangent}{plan tangent} le long de celle-ci. L'autre direction
\omterm{def:b2:curves:arc}{tangente} $\sigma_u$ est la vitesse de l'hélice $u \mapsto
\sigma(u, 1)$~: une carte, deux courbes tracées, et tout le \omterm{def:b2:surfaces:tangent}{plan
tangent} est engendré --- la~\cref{prop:b2:surfaces:velocity} en
action.
\end{example}

\begin{proposition}[Normale d'une surface de niveau]\label{prop:b2:surfaces:gradient}
Soit $S = \{f = c\}$ avec $f$ de classe $\mathcal{C}^1$ et $\nabla
f(M_0) \neq 0$. Alors le \omterm{def:b2:surfaces:tangent}{plan tangent} de $S$ en $M_0$ est le
plan passant par $M_0$ orthogonal à $\nabla f(M_0)$~:
\[
T_{M_0}S :\quad
\langle \nabla f(M_0),\ M - M_0\rangle = 0 .
\]
\end{proposition}

\begin{proof}
Pour toute courbe $\gamma$ tracée sur $S$ passant par $M_0$, $f(\gamma(t)) =
c$ identiquement, donc la règle de dérivation en chaîne donne $\langle
\nabla f(M_0), \gamma'(0)\rangle = 0$~: tous les vecteurs vitesse sont
orthogonaux au gradient, donc la direction \omterm{def:b2:curves:arc}{tangente} est contenue
dans le plan $\nabla f(M_0)^\perp$. Les deux sont des sous-espaces
de dimension $2$ --- la direction \omterm{def:b2:curves:arc}{tangente} parce que $S$ est
localement un graphe régulier (\cref{ex:b2:surfaces:standard}),
l'orthogonal parce que $\nabla f(M_0) \neq 0$ --- donc ils sont
égaux.
\end{proof}

\begin{example}\label{ex:b2:surfaces:spheretangent}
Pour la sphère $x^2 + y^2 + z^2 = R^2$~: $\nabla f = 2(x, y, z)$,
donc le \omterm{def:b2:surfaces:tangent}{plan tangent} en $M_0$ est orthogonal au rayon
$\vect{OM_0}$ --- le fait classique que rayon et \omterm{def:b2:surfaces:tangent}{plan tangent}
sont perpendiculaires, d'équation $\langle M_0, M\rangle = R^2$.
\end{example}

\begin{example}[Plan tangent d'un graphe]\label{ex:b2:surfaces:graphtangent}
Pour $z = f(x, y)$ en $(x_0, y_0)$~: en appliquant la
\cref{prop:b2:surfaces:gradient} à $F(x,y,z) = f(x,y) - z$,
\[
z = f(x_0, y_0)
+ f_x(x_0, y_0)(x - x_0)
+ f_y(x_0, y_0)(y - y_0) ,
\]
la partie \omterm{def:b2:affine:subspace}{affine} du développement de Taylor à l'\omterm{def:b2:structures:generated}{ordre} un --- le
\omterm{def:b2:surfaces:tangent}{plan tangent} est le graphe de la \omterm{def:b2:diffcalc:differential}{différentielle}, comme il se doit.
\end{example}

\begin{example}[Le point le plus proche d'une
surface]\label{ex:b2:surfaces:closest}
Quel point du paraboloïde $z = x^2 + y^2$ est le plus proche de
$P = (0, 0, 1)$~? Minimisons le carré de la distance le long de la
surface~: avec $\rho^2 = x^2 + y^2$,
\[
g(\rho^2) = \rho^2 + (\rho^2 - 1)^2,
\qquad
g'(\rho^2) = 1 + 2(\rho^2 - 1) = 0
\iff \rho^2 = \tfrac12 ,
\]
donnant le cercle des points à hauteur $z = \frac12$ et
distance $\sqrt{\tfrac12 + \tfrac14} = \frac{\sqrt3}2$. La
signature géométrique de la minimalité~: en un tel point $M$, le
vecteur $\vect{MP}$ doit être \emph{normal} à la surface ---
sinon glisser le long d'une courbe tracée avec une vitesse ayant
une composante vers $P$ diminuerait la distance. Vérification~:
$\nabla(z - x^2 - y^2) = (-2x, -2y, 1)$ en $M = (x, y,
\tfrac12)$, tandis que $\vect{MP} = (-x, -y, \tfrac12) =
\tfrac12(-2x, -2y, 1)$~: parallèles, comme prédit. La
condition du premier \omterm{def:b2:structures:generated}{ordre} «~pied de la perpendiculaire~» est la
même qui pilotera les extrema sur les ensembles de niveau dans
l'\cref{exo:b2:surfaces:12}.
\end{example}

\begin{example}[Plans tangents des quadriques~: la
règle de polarisation]\label{ex:b2:surfaces:polarization}
Soit $S : xy + yz + zx = 1$ et $M_0 = (1, 1, 0) \in S$. Ici
$\nabla f = (y + z,\ x + z,\ x + y)$, donc $\nabla f(M_0) = (1,
1, 2)$ et le \omterm{def:b2:surfaces:tangent}{plan tangent} est
\[
(x - 1) + (y - 1) + 2z = 0,
\qquad\text{c.-à-d.}\qquad x + y + 2z = 2 .
\]
La même réponse provient de la \emph{règle de \omterm{def:b2:quadratic:def}{polarisation}} qui
généralise l'\cref{ex:b2:surfaces:spheretangent} et
l'\cref{exo:b2:surfaces:2}~: dans l'équation de la \omterm{pb:b2:surfaces:1}{quadrique},
remplacer $x^2$ par $x_0x$, et chaque produit $xy$ par
$\frac{x_0y + y_0x}2$ (et cycliquement)~:
\[
\frac{x_0y + y_0x}2 + \frac{y_0z + z_0y}2 + \frac{z_0x +
x_0z}2 = 1
\;\xrightarrow{\ M_0 = (1,1,0)\ }\;
\frac{x + y}2 + \frac z2 + \frac z2 = 1 ,
\]
soit $x + y + 2z = 2$ à nouveau. La règle fonctionne parce que
$\nabla$ d'une \omterm{def:b2:quadratic:def}{forme quadratique} est la forme bilinéaire associée
évaluée contre le point de base --- la tangence à une \omterm{pb:b2:surfaces:1}{quadrique}
est la \omterm{def:b2:quadratic:def}{polarisation}, une facette de plus du~\cref{ch:b2:quadratic}.
\end{example}

\section{La première forme fondamentale}

\begin{definition}[Première forme fondamentale]\label{def:b2:surfaces:fff}
Soit $\sigma \colon U \to \R^3$ une surface $\mathcal{C}^1$
régulière. Sa \emph{première forme fondamentale}\index{première forme
fondamentale} en $(u,v)$ est la
\omterm{def:b2:quadratic:def}{forme quadratique} définie positive sur $\R^2$
\[
I(h, k) = \norm{h\,\sigma_u + k\,\sigma_v}^2
= E\,h^2 + 2F\,hk + G\,k^2,
\]
où
\[
E = \norm{\sigma_u}^2,
\qquad
F = \langle \sigma_u, \sigma_v\rangle,
\qquad
G = \norm{\sigma_v}^2 .
\]
\end{definition}

\begin{remark}
$I$ est la restriction du produit scalaire euclidien ambiant au
\omterm{def:b2:surfaces:tangent}{plan tangent}, lue dans la base $(\sigma_u, \sigma_v)$~: elle est
définie positive précisément parce que $\sigma_u, \sigma_v$ sont
indépendants (\cref{ch:b2:quadratic}). Toute quantité métrique sur la
surface --- \omterm{def:b2:curves:length}{longueurs} des courbes tracées, angles entre elles, \omterm{def:b2:surfaces:area}{aires} ---
se calcule à partir de $E, F, G$ seuls. Deux surfaces ayant les mêmes $E,
F, G$ dans des paramètres convenables sont \emph{isométriques} même si
elles sont placées différemment dans l'espace~: c'est le point de
départ de la géométrie intrinsèque.
\end{remark}

\begin{example}[Angles entre courbes
coordonnées]\label{ex:b2:surfaces:coordangle}
La \omterm{def:b2:surfaces:fff}{première forme fondamentale} mesure aussi les angles~: les
courbes coordonnées $u \mapsto \sigma(u, v_0)$ et $v \mapsto
\sigma(u_0, v)$ se rencontrent selon l'angle $\theta$ avec
\[
\cos\theta = \frac{\langle\sigma_u,
\sigma_v\rangle}{\norm{\sigma_u}\,\norm{\sigma_v}}
= \frac{F}{\sqrt{EG}} :
\]
le seul coefficient $F$ décide de l'orthogonalité du
réseau de paramètres. Pour la carte sphérique et l'hélicoïde, $F =
0$~: les méridiens coupent les parallèles, et les hélices coupent
les génératrices horizontales, à angle droit --- ce qui explique
pourquoi leurs intégrandes d'\omterm{def:b2:surfaces:area}{aire} se réduisaient à $\sqrt{EG}$.
Pour une carte-graphe, $F = f_xf_y$ ne s'annule que là où une
dérivée partielle s'annule~: le réseau de coordonnées d'un graphe
incliné n'est \emph{pas} orthogonal, même si le réseau $(x, y)$ en
dessous l'est. Quand les calculs sur une surface paraissent
lourds, le premier réflexe est de chercher une carte avec $F = 0$.
\end{example}

\begin{example}[La carte de la selle]\label{ex:b2:surfaces:saddlechart}
Pour la selle $z = xy$ avec la carte $\sigma(u, v) = (u, v,
uv)$~:
\[
\sigma_u = (1, 0, v), \qquad \sigma_v = (0, 1, u),
\qquad
E = 1 + v^2, \quad F = uv, \quad G = 1 + u^2 ,
\]
et $EG - F^2 = 1 + u^2 + v^2 > 0$~: régulière partout. Les
deux courbes coordonnées passant par un point sont des
\emph{droites} de $\R^3$ (fixer $u$ ou fixer $v$~: les
génératrices de la selle doublement réglée), pourtant $F \neq 0$
hors des axes~: les génératrices passant par un point générique
ne sont pas orthogonales. Les deux génératrices sont dans le \omterm{def:b2:surfaces:tangent}{plan
tangent}, qu'elles engendrent --- donc le \omterm{def:b2:surfaces:tangent}{plan tangent} coupe la
surface le long de deux droites entières, l'exact opposé de la
sphère, dont les \omterm{def:b2:surfaces:tangent}{plans tangents} touchent en un seul point. Le
signe du «~contact d'\omterm{def:b2:structures:generated}{ordre} deux~» entre une surface et ses \omterm{def:b2:surfaces:tangent}{plans
tangents} est une histoire de \omterm{thm:b2:curves:frenet2d}{courbure}, reprise dans le volume de
troisième année.
\end{example}

\begin{proposition}[Longueur d'une courbe tracée sur une
surface]\label{prop:b2:surfaces:curvelength}
Si $\gamma(t) = \sigma(u(t), v(t))$, $t \in [a, b]$, est
$\mathcal{C}^1$, alors
\[
L(\gamma) = \int_a^b
\sqrt{E\,u'^2 + 2F\,u'v' + G\,v'^2}\;\dd t ,
\]
avec $E, F, G$ évalués en $(u(t), v(t))$.
\end{proposition}

\begin{proof}
$\gamma' = u'\sigma_u + v'\sigma_v$ par la règle de dérivation en
chaîne, donc $\norm{\gamma'}^2 = I(u', v') = Eu'^2 + 2Fu'v' + Gv'^2$~;
intégrer $\norm{\gamma'}$ (\cref{def:b2:curves:length}).
\end{proof}

\begin{example}[Pourquoi les avions de ligne survolent le pôle]\label{ex:b2:surfaces:polarroute}
Deux aéroports sont à la latitude $\varphi_0$ et à des longitudes
opposées~: $A = \sigma(0, \varphi_0)$ et $B = \sigma(\pi,
\varphi_0)$ sur la sphère de rayon $R$. Le long du parallèle
($\varphi \equiv \varphi_0$), la \omterm{def:b2:curves:length}{longueur} est
$\int_0^\pi R\cos\varphi_0\,\dd\theta = \pi R\cos\varphi_0$.
Le long de la route par le pôle (en montant le méridien $\theta = 0$,
en descendant le méridien $\theta = \pi$), elle vaut $2R(\frac\pi2 -
\varphi_0)$. À la latitude $\varphi_0 = \frac\pi3$ (soixante
degrés)~: route par le parallèle $\pi R/2 \approx 1.571\,R$, route
polaire $\pi R/3 \approx 1.047\,R$ --- un tiers plus courte. En fait
$\pi\cos\varphi_0 \geq \pi - 2\varphi_0$ sur
$\intcc0{\pi/2}$ (la fonction $\pi\cos\varphi - \pi +
2\varphi$ s'annule aux deux bornes et sa dérivée $2 -
\pi\sin\varphi$ change de signe une fois, donc elle est d'abord
croissante puis décroissante, donc positive)~: la route polaire
ne perd jamais. La \omterm{def:b2:surfaces:fff}{première forme fondamentale} a transformé une
question de navigation en deux intégrales d'une ligne~;
l'\cref{exo:b2:surfaces:6} pousse l'idée jusqu'à une véritable
preuve de minimalité pour les méridiens.
\end{example}

\begin{lemma}[Identité de Lagrange]\label{lem:b2:surfaces:lagrange}
Pour tous $a, b \in \R^3$~:
$\norm{a \wedge b}^2 = \norm a^2 \norm b^2 - \langle a, b\rangle^2$.
En particulier
\[
\norm{\sigma_u \wedge \sigma_v} = \sqrt{EG - F^2}.
\]
\end{lemma}

\begin{proof}
Les deux membres sont inchangés si l'on remplace $b$ par sa composante $b_\perp
= b - \frac{\langle a, b\rangle}{\norm a^2}a$ orthogonale à $a$
(pour $a \neq 0$~; le cas $a = 0$ est trivial)~: le membre de gauche
parce que $a \wedge a = 0$, celui de droite en développant $\norm
{b_\perp}^2 = \norm b^2 - \frac{\langle a,b\rangle^2}{\norm a^2}$
et $\langle a, b_\perp\rangle = 0$. Il suffit donc de prouver
l'identité pour $a, b$ orthogonaux, où elle s'écrit $\norm{a \wedge
b} = \norm a \norm b$~: vrai, puisque pour des vecteurs orthogonaux le
produit vectoriel a pour \omterm{def:b2:nvs:norm}{norme} $\norm a\norm b\,\abs{\sin\frac\pi2}$. La
formule affichée est le cas $a = \sigma_u$, $b = \sigma_v$.
\end{proof}

\begin{remark}
L'identité de Lagrange dit que $EG - F^2 = \det\begin{psmallmatrix}
E & F\\ F & G\end{psmallmatrix}$ est le \emph{\omterm{def:b2:linalg:det}{déterminant} de
Gram} de $(\sigma_u, \sigma_v)$~: le carré de l'\omterm{def:b2:surfaces:area}{aire} du
parallélogramme qu'ils engendrent. La régularité, la
définie-positivité de la \omterm{def:b2:surfaces:fff}{première forme fondamentale} et la
positivité du \omterm{def:b2:linalg:det}{déterminant} de Gram sont trois formulations d'une
même condition --- ce qui explique pourquoi l'intégrande d'\omterm{def:b2:surfaces:area}{aire}
ci-dessous ne s'annule jamais sur une carte régulière.
\end{remark}

\begin{definition}[Aire]\label{def:b2:surfaces:area}
Soit $\sigma \colon U \to \R^3$ une surface $\mathcal{C}^1$
régulière injective et $K \subseteq U$ un domaine \omterm{def:b2:metric:compact}{compact} sur
lequel les intégrales doubles ont un sens (\cref{ch:b2:multint}). L'\emph{aire}\index{aire!d'une surface} du morceau $\sigma(K)$ est
\[
\mathcal{A}
= \iint_K \norm{\sigma_u \wedge \sigma_v}\,\dd u\,\dd v
= \iint_K \sqrt{EG - F^2}\;\dd u\,\dd v .
\]
\end{definition}

\begin{remark}[Pourquoi cette formule]
Le rectangle $[u, u + \dd u] \times [v, v + \dd v]$ est envoyé, au
premier \omterm{def:b2:structures:generated}{ordre}, sur le parallélogramme engendré par $\sigma_u\,\dd u$
et $\sigma_v\,\dd v$, dont l'\omterm{def:b2:surfaces:area}{aire} est $\norm{\sigma_u \wedge
\sigma_v}\,\dd u\,\dd v$~: la définition intègre le facteur local de
distorsion d'\omterm{def:b2:surfaces:area}{aire}, exactement comme la \omterm{def:b2:curves:length}{longueur d'arc} intègre la
vitesse locale. La cohérence avec le \omterm{def:b2:curves:reparam}{changement de paramètres} est
l'\cref{exo:b2:surfaces:8}~; la cohérence avec la formule de
changement de variables pour les intégrales doubles est discutée
dans le~\cref{ch:b2:multint}.
\end{remark}

\begin{example}[Deux graphes différents, une même
aire]\label{ex:b2:surfaces:samearea}
Sur le disque unité, comparons le bol $z = \frac12(x^2 + y^2)$
et la selle $z = xy$. Leurs intégrandes d'\omterm{def:b2:surfaces:area}{aire}
(\cref{exo:b2:surfaces:5}) sont
\[
\sqrt{1 + x^2 + y^2}
\qquad\text{et}\qquad
\sqrt{1 + y^2 + x^2} :
\]
identiques. Les deux surfaces --- l'une se courbant de la même
façon dans toutes les directions, l'autre en forme de selle ---
ont des \omterm{def:b2:surfaces:area}{aires} exactement égales sur tout domaine, $\frac{2\pi}3(2\sqrt2 - 1)$
sur le disque unité. L'élément d'\omterm{def:b2:surfaces:area}{aire} ne voit que la
\emph{\omterm{def:b2:curves:length}{longueur}} du gradient, pas la disposition de la \omterm{thm:b2:curves:frenet2d}{courbure}~;
distinguer le bol de la selle exige des données du second \omterm{def:b2:structures:generated}{ordre}
(la structure de signe exhibée dans la~\cref{fig:b2:surfaces:saddle}),
qu'aucune mesure d'\omterm{def:b2:surfaces:area}{aire} ne détecte. \omterm{def:b2:surfaces:fff}{Première forme fondamentale}~:
métrique, aveugle à la forme~; la seconde forme, qui voit la
forme, appartient à la troisième année.
\end{example}

\begin{example}[Aire de la sphère]\label{ex:b2:surfaces:spherearea}
Pour la carte sphérique de l'\cref{ex:b2:surfaces:standard}~:
\[
\sigma_\theta = R(-\sin\theta\cos\varphi,\
\cos\theta\cos\varphi,\ 0),
\qquad
\sigma_\varphi = R(-\cos\theta\sin\varphi,\
-\sin\theta\sin\varphi,\ \cos\varphi),
\]
donc $E = R^2\cos^2\varphi$, $F = 0$, $G = R^2$ et $\sqrt{EG - F^2}
= R^2\cos\varphi$. D'où
\[
\mathcal{A}
= \int_{-\pi/2}^{\pi/2}\!\!\int_0^{2\pi}
R^2\cos\varphi\;\dd\theta\,\dd\varphi
= 2\pi R^2\,\bigl[\sin\varphi\bigr]_{-\pi/2}^{\pi/2}
= \boxed{4\pi R^2} .
\]
\end{example}

\begin{example}[Le cône, confronté à la formule du
lycée]\label{ex:b2:surfaces:conearea}
Pour le cône $z = \sqrt{x^2 + y^2}$ au-dessus de l'anneau $a \leq
\rho \leq b$, la formule du graphe de
l'\cref{exo:b2:surfaces:5} donne $1 + f_x^2 + f_y^2 = 2$
(calculer $f_x = x/\rho$, $f_y = y/\rho$), donc
\[
\mathcal A = \sqrt2\,\pi\,(b^2 - a^2) .
\]
Vérification de cohérence avec la formule de l'\omterm{def:b2:surfaces:area}{aire} latérale
$\pi\rho\ell$ de l'\cref{ex:b2:surfaces:revolution}~: les cônes
pleins de rayons de base $b$ et $a$ ont des \omterm{def:b2:surfaces:area}{aires} latérales
$\pi b\cdot b\sqrt2$ et $\pi a\cdot a\sqrt2$, dont la différence
est exactement $\sqrt2\pi(b^2 - a^2)$. Deux cartes, deux formules,
une même \omterm{def:b2:surfaces:area}{aire} --- l'invariance prouvée dans
l'\cref{exo:b2:surfaces:8}, vue en action.
\end{example}

\begin{remark}[Vérifications rapides pour les aires]
Trois vérifications instantanées attrapent la plupart des erreurs
dans un calcul d'\omterm{def:b2:surfaces:area}{aire}. \emph{Homogénéité}~: dilater une surface
d'un facteur $\lambda$ multiplie $E, F, G$ par $\lambda^2$ et
l'\omterm{def:b2:surfaces:area}{aire} par $\lambda^2$ --- une réponse dont la dépendance en $R$
n'est pas quadratique (comme $4\pi R^2$) est fausse. \emph{Positivité de
l'élément}~: $\sqrt{EG - F^2}$ doit être strictement positif à
l'intérieur de la carte~; une valeur nulle signale une
dégénérescence de la carte, à exciser comme aux pôles de la
sphère. \emph{Symétrie}~: un calcul sur un morceau \omterm{def:b2:quadratic:adjoint}{symétrique} doit
être cohérent avec la somme de ses parties congruentes ---
l'hémisphère a intérêt à donner $2\pi R^2$.
\end{remark}

\begin{example}[Surface de révolution]\label{ex:b2:surfaces:revolution}
Faire tourner la courbe $z \mapsto (r(z), 0, z)$, $r > 0$ de classe
$\mathcal{C}^1$, autour de l'axe $z$~:
\[
\sigma(\theta, z) = (r(z)\cos\theta,\ r(z)\sin\theta,\ z).
\]
Alors $E = r(z)^2$, $F = 0$, $G = 1 + r'(z)^2$, donc
\[
\mathcal{A} = \int_a^b\!\!\int_0^{2\pi}
r(z)\sqrt{1 + r'(z)^2}\;\dd\theta\,\dd z
= 2\pi\int_a^b r(z)\sqrt{1 + r'(z)^2}\,\dd z ,
\]
la formule classique (circonférence $2\pi r$ fois l'élément de
\omterm{def:b2:curves:length}{longueur} oblique). Pour le cône $r(z) = kz$, $z \in [0, h]$~: $\mathcal
A = 2\pi k\sqrt{1 + k^2}\,\frac{h^2}{2} = \pi \rho \ell$ avec
$\rho = kh$ le rayon de base et $\ell = h\sqrt{1 + k^2}$ la
génératrice --- la formule du lycée, désormais démontrée plutôt
qu'admise.
\end{example}

\begin{example}[Le caténoïde]\label{ex:b2:surfaces:catenoid}
Faire tourner la chaînette $r(z) = \cosh z$, $z \in \intcc{-1}{1}$,
autour de son axe~: le \emph{caténoïde} résultant a, par la
formule de révolution et $1 + \sinh^2 z = \cosh^2 z$,
\[
\mathcal A = 2\pi\int_{-1}^1\cosh z\,\sqrt{1 + \sinh^2z}\,
\dd z = 2\pi\int_{-1}^1\cosh^2z\,\dd z
= \pi\bigl[z + \sinh z\cosh z\bigr]_{-1}^{1},
\]
soit
\[
\mathcal A = 2\pi + \pi\sinh 2 \approx 17.68 .
\]
Le facteur oblique $\sqrt{1 + r'^2}$ s'est fondu avec le rayon
en un carré parfait --- la même identité qui rendait élémentaire
la \omterm{def:b2:curves:length}{longueur d'arc} de la chaînette dans le chapitre sur les
courbes. Ce n'est pas un hasard algébrique~: parmi toutes les
surfaces de révolution s'appuyant sur les deux cercles de bord,
le caténoïde minimise l'\omterm{def:b2:surfaces:area}{aire} (c'est la forme d'un film de savon
entre deux anneaux), et cette propriété variationnelle est
précisément ce qui distingue $\cosh$~; le volume de troisième
année le prouve avec le calcul des variations.
\end{example}

\begin{remark}[Pièges courants]
(i) \emph{Les singularités de carte ne sont pas des singularités
de surface}~: la carte sphérique dégénère aux pôles
($\cos\varphi = 0$), mais la sphère y est parfaitement régulière
--- une autre carte (échanger les rôles des axes) est régulière
aux pôles. Avant de déclarer un point singulier, essayer un second
paramétrage. (ii) \emph{La régularité de $\sigma$ concerne
le paramétrage, pas l'image}~: $\sigma(u, v) = (u^3, v,
0)$ est non régulière le long de $u = 0$ bien que son image soit un
plan. (iii) La formule de l'\omterm{def:b2:surfaces:area}{aire} requiert $\sigma$
\emph{injective} sur $K$~: une carte couvrant un morceau deux fois
le compte deux fois ($\theta$ parcourant $\intcc0{4\pi}$
double l'\omterm{def:b2:surfaces:area}{aire} de la sphère). (iv) La \omterm{def:b2:surfaces:tangent}{normale unitaire} est définie
au signe près par la surface mais est \emph{choisie} par la carte
(\omterm{def:b2:structures:generated}{ordre} de $u, v$)~; les énoncés faisant intervenir l'orientation
doivent fixer ce choix. (v) Enfin, $EG - F^2 > 0$ n'est pas une
hypothèse supplémentaire~: c'est exactement la régularité, par
l'identité de Lagrange --- si elle s'annule quelque part, le
problème est la carte, et aucune formule d'\omterm{def:b2:surfaces:area}{aire} ou de \omterm{def:b2:surfaces:tangent}{plan tangent}
ne s'y applique.
\end{remark}

\begin{remark}[Perspectives dans ce volume]
Liens vers l'avant à partir d'ici. L'élément d'\omterm{def:b2:surfaces:area}{aire} $\sqrt{EG -
F^2}\,\dd u\,\dd v$ est un jacobien bidimensionnel déguisé, et le~\cref{ch:b2:multint} rend l'analogie exacte avec le théorème de
changement de variables --- les intégrales de surface là-bas sont
les \omterm{def:b2:surfaces:area}{aires} de ce chapitre munies d'un intégrande. La \omterm{def:b2:surfaces:fff}{première forme
fondamentale} est un champ de \omterm{def:b2:quadratic:def}{formes quadratiques} définies
positives, traité point par point par les outils du~\cref{ch:b2:quadratic}, dont le théorème spectral alimente aussi
la classification des \omterm{pb:b2:surfaces:1}{quadriques} du week-end de ce chapitre. Et la
droite normale pilote les problèmes d'extremum sur les ensembles
de contraintes (\cref{ex:b2:surfaces:closest}), le germe
géométrique de la méthode des multiplicateurs de Lagrange
esquissée avec le~\cref{thm:b2:diffcalc:extrema}.
\end{remark}

\begin{remark}[Où cela est utilisé]
L'élément d'\omterm{def:b2:surfaces:area}{aire} $\norm{\sigma_u\wedge\sigma_v}\,\dd u\,\dd v$
est la mesure d'intégration de surface du~\cref{ch:b2:multint}, où
il rencontre la formule de Green~; la \omterm{def:b2:surfaces:fff}{première forme fondamentale}
est le prototype d'un champ de \omterm{def:b2:quadratic:def}{formes quadratiques}, étudié point
par point avec les outils du~\cref{ch:b2:quadratic}~; et le
problème du week-end de ce chapitre classe toutes les surfaces
\omterm{pb:b2:surfaces:1}{quadriques} avec le théorème spectral. Le volume de troisième année
revient aux surfaces avec les formes \omterm{def:b2:diffcalc:differential}{différentielles} et le
théorème de la divergence, et la \omterm{thm:b2:curves:frenet2d}{courbure} intrinsèque --- ce que
$E, F, G$ savent de la \omterm{thm:b2:curves:frenet2d}{courbure} --- est la porte d'entrée de la
géométrie \omterm{def:b2:diffcalc:differential}{différentielle} proprement dite.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b2:surfaces:1}
Montrer que les \omterm{def:b2:surfaces:tangent}{plans tangents} du cône $z = \sqrt{x^2 + y^2}$
(privé de son sommet) passent tous par le sommet. \emph{(Paramétrer par
$\sigma(\theta, r) = (r\cos\theta, r\sin\theta, r)$, $r > 0$.)}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:surfaces:2}
Trouver le \omterm{def:b2:surfaces:tangent}{plan tangent} de l'ellipsoïde $\frac{x^2}{a^2} +
\frac{y^2}{b^2} + \frac{z^2}{c^2} = 1$ en un point $(x_0, y_0,
z_0)$ de la surface.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:surfaces:3}
Calculer $E, F, G$ pour l'hélicoïde $\sigma(u, v) = (v\cos u,\
v\sin u,\ au)$, $a > 0$, et l'\omterm{def:b2:surfaces:area}{aire} du morceau $0 \leq u \leq
2\pi$, $0 \leq v \leq 1$, comme intégrale (l'évaluer en utilisant
$\int\sqrt{v^2 + a^2}\,\dd v = \frac12\bigl(v\sqrt{v^2+a^2} +
a^2\ln(v + \sqrt{v^2 + a^2})\bigr) + C$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:surfaces:4}
(Tore) Paramétrer le tore obtenu en faisant tourner le cercle de
centre $(R, 0, 0)$ et de rayon $r < R$ dans le plan $xz$ autour de
l'axe $z$~:
\[
\sigma(\theta, \psi) = \bigl((R + r\cos\psi)\cos\theta,\
(R + r\cos\psi)\sin\theta,\ r\sin\psi\bigr).
\]
Calculer $E, F, G$, vérifier la régularité, et montrer que l'\omterm{def:b2:surfaces:area}{aire}
est $4\pi^2 R r$ (Pappus~: circonférence moyenne $2\pi R$ fois la
\omterm{def:b2:curves:length}{longueur} du cercle $2\pi r$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:surfaces:5}
Montrer que l'\omterm{def:b2:surfaces:area}{aire} du graphe de $f \in \mathcal{C}^1(K)$ est
$\iint_K \sqrt{1 + f_x^2 + f_y^2}\,\dd x\,\dd y$, et la calculer
pour le morceau de paraboloïde $z = \frac12(x^2 + y^2)$ au-dessus
du disque $x^2 + y^2 \leq 1$ (coordonnées polaires,
\cref{ch:b2:multint}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:surfaces:6}
Une courbe tracée $\gamma(t) = \sigma(u(t), v(t))$ sur la sphère de
rayon $R$ (carte sphérique) a $u = \theta(t)$, $v =
\varphi(t)$. Écrire sa \omterm{def:b2:curves:length}{longueur} comme intégrale en $\theta, \varphi$
et prouver que parmi les courbes joignant deux points d'un même
méridien $\theta = \theta_0$, l'arc de méridien est le plus court.
\emph{(Minorer l'intégrande par $R\abs{\varphi'}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:surfaces:7}
(Droites normales d'une sphère) Soit $S$ une surface de niveau
régulière $\{f = c\}$, \omterm{def:b2:metric:connected}{connexe}, dont toutes les droites normales
passent par un point fixe $\Omega$. Montrer que $S$ est contenue
dans une sphère centrée en $\Omega$. \emph{(Montrer que
$\norm{M - \Omega}^2$ a une dérivée nulle le long de toute courbe
tracée sur $S$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:surfaces:8}
(L'\omterm{def:b2:surfaces:area}{aire} est géométrique) Soit $\Phi \colon U' \to U$ un
$\mathcal{C}^1$-difféomorphisme entre \omterm{def:b2:metric:topology}{ouverts} de $\R^2$ et
$\tilde\sigma = \sigma \circ \Phi$. Montrer que
\[
\norm{\tilde\sigma_{u'} \wedge \tilde\sigma_{v'}}
= \abs{\det J_\Phi}\,
\norm{(\sigma_u \wedge \sigma_v)\circ\Phi} ,
\]
et en déduire, en utilisant la formule de changement de variables
du~\cref{ch:b2:multint}, que l'\omterm{def:b2:surfaces:area}{aire} de la
\cref{def:b2:surfaces:area} ne dépend pas du paramétrage régulier
choisi.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:surfaces:9}
(Théorème de la boîte à chapeau d'Archimède) Sur la sphère de
rayon $R$, la \emph{zone} entre les latitudes avec $z_1 \leq z \leq z_2$
($-R \leq z_1 < z_2 \leq R$) a pour \omterm{def:b2:surfaces:area}{aire} $2\pi R\,(z_2 - z_1)$~:
le prouver avec la carte sphérique, et conclure que l'\omterm{def:b2:surfaces:area}{aire} d'une
zone ne dépend que de sa hauteur --- découper une orange en
tranches d'épaisseur égale donne des quantités de peau égales.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:surfaces:10}
Montrer que toute droite normale d'une surface de révolution
$\sigma(\theta, z) = (r(z)\cos\theta,\ r(z)\sin\theta,\ z)$
($r > 0$ de classe $\mathcal C^1$) rencontre l'axe de
révolution, et localiser le point d'intersection.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:surfaces:11}
(Déroulement du cylindre) La carte $\sigma(u, v) = (\cos u,\
\sin u,\ v)$ du cylindre unité a $E = G = 1$, $F = 0$~:
le vérifier, et expliquer pourquoi toute courbe tracée $t \mapsto
\sigma(u(t), v(t))$ a la même \omterm{def:b2:curves:length}{longueur} que la courbe plane $t
\mapsto (u(t), v(t))$. En déduire que l'hélice de $(1, 0, 0)$
à $(1, 0, 2\pi c)$ faisant un tour a pour \omterm{def:b2:curves:length}{longueur}
$2\pi\sqrt{1 + c^2}$, et qu'aucune courbe tracée de mêmes
extrémités et faisant un tour \omterm{def:b2:metric:complete}{complet} n'est plus courte.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:surfaces:12}
Soit $S = \{f = c\}$ une surface de niveau régulière compacte et
$M_0 \in S$ un point à distance maximale de l'origine. Montrer
que $\nabla f(M_0)$ est colinéaire à $\vect{OM_0}$ --- la
normale au point le plus éloigné est radiale. Appliquer à
l'ellipsoïde $\frac{x^2}{a^2} + \frac{y^2}{b^2} + \frac{z^2}{c^2} = 1$ ($a >
b > c > 0$)~: trouver tous les points où la normale est radiale, et
identifier les plus éloignés.
\end{exercise}

\section{Problème~: la classification des quadriques de
$\R^3$}

\begin{problem}[{Problème du week-end --- toutes les surfaces
quadriques, classées par le théorème spectral}]\label{pb:b2:surfaces:1}
Une \emph{quadrique}\index{quadrique} est l'ensemble des zéros dans $\R^3$ d'un
polynôme de degré deux
\[
q(X) = X^{\mathsf T}\!AX + 2\,\langle b, X\rangle + c,
\qquad A \in \mathcal S_3(\R),\ A \neq 0,\ b \in \R^3,\
c \in \R .
\]
Les surfaces des figures de ce chapitre --- sphères,
ellipsoïdes, selles, cônes, cylindres --- sont toutes des
\omterm{pb:b2:surfaces:1}{quadriques}. Ce problème les classe \emph{toutes}~: le théorème
spectral (\cref{thm:b2:quadratic:spectral}) redresse la partie
quadratique, les translations \omterm{def:b2:affine:subspace}{affines} (\cref{ch:b2:affine})
absorbent la partie linéaire, et ce qui reste est une liste
courte et complète de formes normales.

\medskip
\noindent\textbf{Partie I --- La machine de réduction.}
\begin{enumerate}
  \item Soit $X = PY + t$ avec $P \in O(3)$ et $t \in \R^3$ (un
        changement rigide de coordonnées). Montrer que $q(PY + t) =
        Y^{\mathsf T}\!A'Y + 2\langle b', Y\rangle + c'$ avec
        \[
        A' = P^{\mathsf T}\!AP, \qquad
        b' = P^{\mathsf T}(At + b), \qquad
        c' = q(t) .
        \]
        En déduire que le \omterm{def:b2:reduction:eigen}{spectre} de $A$ (donc son rang et sa
        signature) est un invariant rigide de l'équation, et
        expliquer pourquoi l'équation d'une \omterm{pb:b2:surfaces:1}{quadrique} donnée n'est
        déterminée qu'à un facteur scalaire non nul près.
  \item En utilisant le théorème spectral, montrer qu'après une rotation
        l'équation devient $\sum_i \lambda_i y_i^2 +
        2\sum_i\beta_iy_i + c = 0$ avec $\lambda_1, \lambda_2,
        \lambda_3$ les \omterm{def:b2:reduction:eigen}{valeurs propres} de $A$.
  \item Pour tout $i$ avec $\lambda_i \neq 0$, absorber
        $\beta_iy_i$ par une translation ($y_i \mapsto y_i -
        \beta_i/\lambda_i$). Écrire l'équation réduite lorsque
        $\operatorname{rank} A = r$~: $\sum_{i\leq r}\lambda_i
        z_i^2 + 2\sum_{i > r}\beta_iz_i + c'' = 0$.
  \item Un \emph{centre} de la \omterm{pb:b2:surfaces:1}{quadrique} d'équation $q = 0$ est
        un point $\Omega$ avec $q(2\Omega - X) = q(X)$ pour tout
        $X$~: la symétrie centrale en $\Omega$ préserve
        l'équation, donc la surface. Montrer que $q(2\Omega - X)
        - q(X) = -4\langle A\Omega + b, X\rangle + 4\langle
        A\Omega + b, \Omega\rangle$, et en déduire~: les centres
        sont exactement les solutions de $A\Omega = -b$~; ils
        existent ssi $b \in \operatorname{im}A$, et le centre
        est unique ssi $A$ est inversible.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- \omterm{pb:b2:surfaces:1}{Quadriques} à centre
($\operatorname{rank}A = 3$).} Ici l'équation réduite est
$\lambda_1z_1^2 + \lambda_2z_2^2 + \lambda_3z_3^2 = \delta$.
\begin{enumerate}[resume]
  \item Quitte à multiplier par $-1$, supposer qu'au moins deux
        $\lambda_i > 0$. Énumérer les possibilités~:
        signature $(3, 0)$ avec $\delta > 0$, $= 0$, $< 0$,
        et signature $(2, 1)$ avec $\delta > 0$, $= 0$, $<
        0$~; nommer les six ensembles résultants (ellipsoïde,
        point, ensemble vide, hyperboloïde à une nappe, cône,
        hyperboloïde à deux nappes) et mettre chacun sous sa forme
        normale euclidienne ($\frac{x^2}{a^2} + \frac{y^2}{b^2} +
        \frac{z^2}{c^2} = 1$, etc.).
  \item Classer $x^2 + y^2 + z^2 + 4xy + 4yz + 4zx = 1$~:
        montrer que $A = 2J - I$ avec $J$ la matrice remplie de
        $1$, calculer le \omterm{def:b2:reduction:eigen}{spectre} $\{5, -1, -1\}$, et identifier un
        hyperboloïde à deux nappes de révolution autour de l'axe
        $\R(1,1,1)$.
  \item (Génératrices) Pour l'hyperboloïde à une nappe
        $\frac{x^2}{a^2} + \frac{y^2}{b^2} - \frac{z^2}{c^2}
        = 1$, factoriser
        \[
        \Bigl(\frac xa - \frac zc\Bigr)
        \Bigl(\frac xa + \frac zc\Bigr)
        = \Bigl(1 - \frac yb\Bigr)\Bigl(1 + \frac yb\Bigr)
        \]
        et produire deux familles à un paramètre de droites
        tracées sur la surface.
  \item Montrer que par chaque point de l'hyperboloïde à une
        nappe passe exactement une droite de chaque famille~:
        la surface est \emph{doublement réglée}.
  \item Le \emph{cône asymptote} de l'hyperboloïde à une
        nappe est $C : \frac{x^2}{a^2} + \frac{y^2}{b^2}
        - \frac{z^2}{c^2} = 0$. Avec $\rho^2 = \frac{x^2}{a^2}
        + \frac{y^2}{b^2}$, montrer que tout point de
        l'hyperboloïde est à distance au plus $c\bigl(\rho -
        \sqrt{\rho^2 - 1}\bigr) = \frac{c}{\rho +
        \sqrt{\rho^2 - 1}}$ de $C$, donc la surface épouse son
        cône à l'infini. Quelles sont les sections de
        l'hyperboloïde par les plans $x = \pm a$~?
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Rang $2$ et rang $1$~:
paraboloïdes, cylindres, plans.}
\begin{enumerate}[resume]
  \item Supposons $\operatorname{rank}A = 2$, disons $\lambda_1,
        \lambda_2 \neq 0 = \lambda_3$. En partant de la question
        3, séparer en deux cas selon $\beta_3 \neq 0$
        (pas de centre, par la question 4) ou $\beta_3 = 0$ (une
        droite de centres), et réduire à
        \[
        \lambda_1z_1^2 + \lambda_2z_2^2 + 2\beta_3z_3 = 0
        \qquad\text{ou}\qquad
        \lambda_1z_1^2 + \lambda_2z_2^2 + c'' = 0 :
        \]
        paraboloïdes elliptiques/hyperboliques dans le premier cas,
        cylindres sur des coniques à centre (ou paires de
        plans sécants, une droite, l'ensemble vide) dans le
        second.
  \item Montrer que la selle $z = xy$ est un paraboloïde
        hyperbolique~: tourner de $\pi/4$ dans le plan $xy$ pour
        atteindre $z = \tfrac12(u^2 - v^2)$, la surface de
        la~\cref{fig:b2:surfaces:saddle} à l'échelle près.
  \item Montrer que la selle $z = xy$ porte les deux familles de
        droites $\{x = x_0,\ z = x_0y\}$ et $\{y = y_0,\ z
        = xy_0\}$, avec exactement une droite de chaque passant par
        chaque point~: la seconde \omterm{pb:b2:surfaces:1}{quadrique} doublement réglée.
  \item Classer $x^2 + y^2 - 2x + 4y + 3 = 0$ dans $\R^3$
        (compléter les carrés~; identifier un cylindre circulaire
        droit, et donner son axe et son rayon).
  \item Maintenant soit $\operatorname{rank}A = 1$, disons $\lambda_1
        \neq 0 = \lambda_2 = \lambda_3$. En tournant dans le
        plan noyau et en translatant, réduire à
        \[
        \lambda_1z_1^2 + 2\beta z_2 = 0 \quad (\beta \neq 0)
        \qquad\text{ou}\qquad
        \lambda_1z_1^2 + c'' = 0 :
        \]
        un cylindre parabolique, ou une paire de plans parallèles, un
        plan double, ou l'ensemble vide. Classer $(x + y)^2 =
        z$ complètement (forme normale, axe d'invariance par
        translation).
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Le théorème de classification.}
\begin{enumerate}[resume]
  \item Assembler les Parties I--III en un théorème~: \emph{toute
        \omterm{pb:b2:surfaces:1}{quadrique} de $\R^3$ est envoyée par un déplacement sur
        exactement une forme normale}. Lister les dix-sept types
        \omterm{def:b2:affine:subspace}{affines} (compter les variantes vides et les ensembles
        dégénérés), et distinguer les neuf \emph{surfaces}
        \omterm{pb:b2:surfaces:1}{quadriques}~: ellipsoïde, hyperboloïdes à une et deux
        nappes, cône, paraboloïdes elliptique et hyperbolique,
        cylindres elliptique, hyperbolique et parabolique.
  \item Écrire l'\emph{algorithme} de classification~: étant
        donné $(A, b, c)$, quelles quantités calcule-t-on, dans
        quel \omterm{def:b2:structures:generated}{ordre}, et quelle branche décide de quel type~?
        Justifier que chaque étape est effective (\omterm{def:b2:reduction:eigen}{valeurs propres}
        d'une matrice \omterm{def:b2:quadratic:adjoint}{symétrique} $3\times3$, rang, résolubilité de
        $A\Omega = -b$).
  \item Exécuter l'algorithme sur $x^2 + y^2 + z^2 - 2xy - 2yz -
        2zx = 1$~: montrer que $A = 2I - J$ a pour \omterm{def:b2:reduction:eigen}{spectre} $\{2, 2,
        -1\}$ et conclure~: un hyperboloïde à une nappe de
        révolution autour de $\R(1,1,1)$.
  \item L'exécuter sur $x^2 + y^2 - z^2 - 2x + 4y + 2z + 4 = 0$~:
        trouver le centre et identifier la \omterm{pb:b2:surfaces:1}{quadrique}.
  \item L'exécuter sur $x^2 + 4xy + y^2 = 2z$~: diagonaliser le
        bloc $xy$ ($u = \frac{x+y}{\sqrt2}$, $v =
        \frac{x-y}{\sqrt2}$) et identifier la \omterm{pb:b2:surfaces:1}{quadrique}.
  \item Euclidien contre \omterm{def:b2:affine:subspace}{affine}. Montrer que deux \omterm{pb:b2:surfaces:1}{quadriques} à
        centre en forme normale sont \emph{rigidement} \omterm{def:b2:nvs:equivalent}{équivalentes}
        ssi elles ont les mêmes listes de coefficients (à
        permutation et à un scalaire positif commun sur l'équation
        près), tandis qu'\emph{affinement}, seules les données de
        signature survivent~: tout ellipsoïde est une image \omterm{def:b2:affine:subspace}{affine}
        de la sphère ronde. Quel théorème garantit que la
        signature ne peut pas changer en chemin
        (\cref{thm:b2:quadratic:sylvester})~?
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- Dividendes.}
\begin{enumerate}[resume]
  \item Montrer que toute section d'une \omterm{pb:b2:surfaces:1}{quadrique} par un plan
        \omterm{def:b2:affine:subspace}{affine} est une conique (éventuellement dégénérée) de ce plan.
        Identifier les sections de la selle $z = xy$ par les
        plans $z = c$ ($c \neq 0$ et $c = 0$).
  \item Quelles surfaces \omterm{pb:b2:surfaces:1}{quadriques} contiennent des droites~? Montrer
        que l'ellipsoïde, l'hyperboloïde à deux nappes et le
        paraboloïde elliptique n'en contiennent aucune \emph{(restreindre $q$
        à une droite et utiliser l'inégalité de Cauchy--Schwarz pour
        le cas à deux nappes)}~; que le cône et les cylindres sont
        réglés par une famille~; et conclure que les surfaces
        \omterm{pb:b2:surfaces:1}{quadriques} doublement réglées sont exactement l'hyperboloïde à
        une nappe et le paraboloïde hyperbolique.
  \item Quand seul le type \emph{\omterm{def:b2:affine:subspace}{affine}} est recherché, la
        \omterm{thm:b2:quadratic:gauss}{réduction de Gauss} (\cref{thm:b2:quadratic:gauss}) est
        moins coûteuse que la diagonalisation. Refaire la question 17
        avec l'algorithme de Gauss et vérifier la signature $(2, 1)$~;
        quelle information euclidienne Gauss perd-il~?
  \item Toutes les \omterm{def:b2:reduction:eigen}{valeurs propres} d'une matrice \omterm{def:b2:quadratic:adjoint}{symétrique} sont
        réelles~; montrer qu'en conséquence les \emph{signes}
        des \omterm{def:b2:reduction:eigen}{valeurs propres} de $A$ peuvent se lire sur le \omterm{def:b2:reduction:charpoly}{polynôme
        caractéristique} par la règle des signes de Descartes, et le
        vérifier sur la question 6~: $\chi_A(\lambda) = \lambda^3 - 3\lambda^2
        - 9\lambda - 5$ a exactement un changement de signe, donc
        signature $(1, 2)$.
  \item Synthèse. Résumer l'algorithme en quelques lignes~;
        énoncer le rôle exact joué par (i) le théorème spectral,
        (ii) l'équation du centre $A\Omega = -b$,
        (iii) le théorème d'inertie de Sylvester, (iv) la \omterm{thm:b2:quadratic:gauss}{réduction
        de Gauss}. Que donne la même machine dans $\R^2$, et
        qu'est-ce qui change dans $\R^n$~?
\end{enumerate}
\end{problem}
