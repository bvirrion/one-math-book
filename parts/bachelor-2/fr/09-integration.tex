\chapter{Intégration}\label{ch:b2:integration}

Le volume de première année a construit l'intégrale sur un segment. Ce
chapitre l'étend à des intervalles quelconques (\omterm{def:b2:integration:improper}{intégrales impropres},
avec toute la boîte à outils de comparaison), puis étudie les
intégrales \emph{dépendant d'un paramètre} --- continuité et dérivation
sous le signe intégral --- portées par le théorème de convergence
dominée, le seul résultat de ce chapitre admis. La fonction $\Gamma$
sert d'exemple fil rouge, et de porte d'entrée vers la moitié des
fonctions spéciales des mathématiques.

\section{Intégrales sur un intervalle quelconque}

\begin{definition}\label{def:b2:integration:improper}
Soit $f$ \omterm{def:b2:metric:continuity}{continue} par morceaux sur $\intco{a}{b}$ ($b \in \R$ ou
$+\infty$). L'intégrale \emph{converge}\index{intégrale impropre}
lorsque $\lim_{x \to b^-} \int_a^x f$ existe~; on écrit alors
$\int_a^b f$ pour la limite. (De même sur $\intoc{a}{b}$, et sur
$\intoo{a}{b}$ en coupant en un point intérieur --- le choix
n'importe pas, par la relation de Chasles.) L'intégrale \emph{converge
\omterm{def:b2:series:def}{absolument}} lorsque $\int_a^b \abs f$ converge~; la convergence absolue
implique la convergence, par le critère de Cauchy~:
\[
\Bigl| \int_x^{y} f \Bigr| \leq \int_x^{y} \abs f
\]
et la complétude de $\R$ (la primitive possède la propriété de
Cauchy). En détail~: posons $F(x) = \int_a^x f$ et $G(x) = \int_a^x \abs
f$. Si $\int^b\abs f$ converge, $G$ admet une limite en $b^-$, donc
pour tout $\varepsilon > 0$ il existe $c < b$ tel que $G(y) - G(x)
\leq \varepsilon$ dès que $c \leq x \leq y < b$~; la formule ci-dessus
transfère cette propriété de Cauchy à $F$. Pour toute suite $x_n
\to b^-$ les valeurs $F(x_n)$ forment alors une suite de Cauchy de
réels, convergente par complétude, et entrelacer deux telles suites
montre que la limite est la même pour toutes~: $F$ admet une limite en
$b^-$.
\end{definition}

\begin{theorem}[Boîte à outils de comparaison, cas positif]\label{thm:b2:integration:comparison}
Pour $f, g \geq 0$ \omterm{def:b2:metric:continuity}{continues} par morceaux sur $\intco{a}{b}$~:
\begin{enumerate}
  \item $\int_a^b f$ \omterm{def:b2:integration:improper}{converge} si et seulement si la primitive $x \mapsto \int_a^x
        f$ est bornée~;
  \item $f \leq g$~: la convergence de $\int g$ entraîne celle de $\int f$~;
        la divergence se transmet en sens inverse~;
  \item $f \sim g$ en $b$~: les deux intégrales sont de même nature~;
  \item les échelles de référence~: en $+\infty$, $\int^{\infty}
        \frac{\dd t}{t^\alpha}$ \omterm{def:b2:integration:improper}{converge} si et seulement si $\alpha > 1$, et
        $\int^\infty \frac{\dd t}{t(\ln t)^\beta}$ si et seulement si $\beta > 1$~;
        en une borne finie $b$, $\int^b \frac{\dd t}{(b -
        t)^\alpha}$ \omterm{def:b2:integration:improper}{converge} si et seulement si $\alpha < 1$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) La primitive $F(x) = \int_a^x f$ est croissante ($f \geq
0$). Si elle est bornée, $\ell = \sup_{x < b}F$ est fini et
$F(x) \to \ell$~: étant donné $\varepsilon > 0$, un certain $F(x_0) > \ell -
\varepsilon$, et la monotonie piège $F(x) \in
\intoc{\ell - \varepsilon}{\ell}$ pour $x_0 \leq x < b$. Si elle est
non bornée, $F \to +\infty$~: divergence.

(2) De $f \leq g$~: $\int_a^x f \leq \int_a^x g$ pour tout $x$~;
si $\int^b g$ \omterm{def:b2:integration:improper}{converge}, le membre de droite est borné, donc le
membre de gauche aussi, et (1) conclut. La contraposée transmet la
divergence en sens inverse.

(3) $f \sim g$ en $b$ fournit $c < b$ tel que
\[
\tfrac12\,g(t) \;\leq\; f(t) \;\leq\; 2\,g(t)
\qquad (c \leq t < b) :
\]
par (2) appliqué dans les deux sens sur $\intco{c}{b}$, les deux intégrales
sont de même nature~; le morceau initial $\intcc{a}{c}$ est une
intégrale propre et ne change rien.

(4) Primitives explicites~: pour $\alpha \neq 1$ et $\beta \neq 1$,
\[
\int_c^x \frac{\dd t}{t^\alpha}
= \frac{x^{1-\alpha} - c^{1-\alpha}}{1 - \alpha},
\qquad
\int_c^x \frac{\dd t}{t(\ln t)^\beta}
= \frac{(\ln x)^{1-\beta} - (\ln c)^{1-\beta}}{1 - \beta},
\]
avec des logarithmes dans les cas exclus~: bornées quand $x \to
+\infty$ exactement lorsque $\alpha > 1$, resp.\ $\beta > 1$. En une
borne finie, le changement de variable $u = b - t$ ramène à
l'échelle $\int_0 u^{-\alpha}\,\dd u$, bornée si et seulement si $\alpha < 1$.
Appliquer (1) à chaque fois.
\end{proof}

\begin{example}[Deux échauffements, menés jusqu'au bout]\label{ex:b2:integration:warmups}
\emph{(a)} $\displaystyle\int_0^1 \ln t\,\dd t$~: l'intégrande
explose en $0^+$, mais $\abs{\ln t} = o\bigl(t^{-1/2}\bigr)$
au voisinage de $0$ (les logarithmes perdent face aux puissances), et $\int_0 t^{-1/2}$
\omterm{def:b2:integration:improper}{converge}~: convergence absolue. La valeur, par parties sur
$\intcc{\varepsilon}{1}$~:
\[
\int_\varepsilon^1 \ln t\,\dd t
= \bigl[t\ln t - t\bigr]_\varepsilon^1
= -1 - \varepsilon\ln\varepsilon + \varepsilon
\xrightarrow[\varepsilon\to0^+]{} -1 .
\]
\emph{(b)} $\displaystyle\int_0^\infty \frac{\ln t}{1 +
t^2}\,\dd t$~: problème aux deux bornes, donc on coupe en $1$. Au voisinage de $0$~:
$\abs{\ln t}$ intégrable comme en (a)~; au voisinage de $\infty$~: $\frac{\ln
t}{1+t^2} = o(t^{-3/2})$~: \omterm{def:b2:series:def}{absolument} convergente. Le
changement de variable $t = \frac1u$ envoie $\intoo{0}{1}$ sur
$\intoo{1}{\infty}$ et
\[
\int_0^1 \frac{\ln t}{1+t^2}\,\dd t
= \int_1^{\infty} \frac{-\ln u}{1 + u^{-2}}\cdot
\frac{\dd u}{u^2}
= -\int_1^\infty \frac{\ln u}{1+u^2}\,\dd u :
\]
les deux moitiés se compensent, et l'intégrale vaut $0$. Éclairage
final~: la symétrie sous $t \mapsto \frac1t$ vaut une page de
calcul --- le même tour alimentait déjà
l'\cref{exo:b2:integration:3}.
\end{example}

\begin{example}[Une valeur, trois intégrales]\label{ex:b2:integration:onecos}
Étudions $I = \displaystyle\int_0^{\infty} \frac{1 - \cos
t}{t^2}\,\dd t$. En $0$~: $1 - \cos t \sim \frac{t^2}2$, donc
l'intégrande se prolonge par continuité par la valeur $\frac12$ --- aucune
singularité. En $\infty$~: $0 \leq \frac{1 - \cos t}{t^2}
\leq \frac{2}{t^2}$~: convergence absolue
(\cref{thm:b2:integration:comparison}). Valeur~: intégrons par
parties sur $\intcc{\varepsilon}{M}$ avec $u = 1 - \cos t$, $v' =
t^{-2}$~:
\[
\int_\varepsilon^M \frac{1 - \cos t}{t^2}\,\dd t
= \Bigl[-\frac{1 - \cos t}{t}\Bigr]_\varepsilon^M
+ \int_\varepsilon^M \frac{\sin t}{t}\,\dd t .
\]
Le crochet s'annule aux deux bornes ($\frac{1 -
\cos\varepsilon}{\varepsilon} \sim \frac\varepsilon2$~; numérateur
borné en $M$), et l'intégrale tend vers la valeur de Dirichlet
$\frac\pi2$ (\cref{exo:b2:integration:10})~: $I = \frac\pi2$.
Éclairage final~: avec $1 - \cos t = 2\sin^2\frac t2$ et $u =
\frac t2$,
\[
I = \int_0^\infty \frac{2\sin^2 u}{(2u)^2}\;2\,\dd u
= \int_0^\infty \Bigl(\frac{\sin u}{u}\Bigr)^{\!2}\dd u :
\]
les trois classiques $\int_0^\infty\frac{\sin t}{t}\dd t$,
$\int_0^\infty\bigl(\frac{\sin t}{t}\bigr)^2\dd t$
(\cref{exo:b2:integration:11}) et $I$ partagent tous la valeur
$\frac\pi2$, transmise de l'une à l'autre par parties et par
changement de variable --- et seule la première est
semi-convergente~: l'intégration par parties a échangé la
convergence absolue contre un intégrande plus simple.
\end{example}

\begin{example}[Une intégrale semi-convergente]\label{ex:b2:integration:sint}
$\displaystyle\int_1^{\infty} \frac{\sin t}{t}\,\dd t$ \omterm{def:b2:integration:improper}{converge}~:
intégrons par parties,
\[
\int_1^x \frac{\sin t}{t}\dd t
= \Bigl[\frac{-\cos t}{t}\Bigr]_1^x - \int_1^x \frac{\cos
t}{t^2}\dd t ,
\]
où le crochet a une limite et la dernière intégrale \omterm{def:b2:integration:improper}{converge}
\omterm{def:b2:series:def}{absolument} ($\abs{\cos t}/t^2 \leq t^{-2}$). Mais pas \omterm{def:b2:series:def}{absolument}~:
de $\abs{\sin t} \geq \sin^2 t$,
\[
\int_1^x \frac{\abs{\sin t}}{t}\,\dd t
\;\geq\; \int_1^x \frac{\sin^2t}{t}\,\dd t
= \underbrace{\int_1^x \frac{\dd t}{2t}}_{=\ \frac12\ln x
\ \to\ \infty}
\;-\; \underbrace{\int_1^x \frac{\cos 2t}{2t}\,\dd
t}_{\text{convergente}} ,
\]
où la dernière intégrale \omterm{def:b2:integration:improper}{converge} par la \emph{même}
intégration par parties que ci-dessus (avec $\sin 2t$ dans le crochet)~:
une partie divergente moins une partie convergente diverge. Ainsi
$\int_1^\infty\frac{\sin t}{t}\dd t$ \omterm{def:b2:integration:improper}{converge} sans converger
\omterm{def:b2:series:def}{absolument} --- l'analogue intégral des séries \omterm{def:b2:linalg:alternating}{alternées},
l'intégration par parties jouant le rôle du critère spécial des
séries \omterm{def:b2:linalg:alternating}{alternées}.
\end{example}

\section{Le théorème de convergence}

\begin{theorem}[Convergence dominée]\label{thm:b2:integration:dominated}
Soit $(f_n)$ \omterm{def:b2:metric:continuity}{continues} par morceaux sur un intervalle $I$, convergeant
simplement vers une fonction $f$ \omterm{def:b2:metric:continuity}{continue} par morceaux, et supposons
qu'il existe une fonction intégrable fixe $\varphi \geq 0$ ($\int_I \varphi < \infty$)
telle que\index{théorème de convergence dominée}
\[
\abs{f_n(t)} \leq \varphi(t) \qquad (t \in I,\ n \in \N).
\]
Alors toutes les $\int_I f_n$ et $\int_I f$ convergent \omterm{def:b2:series:def}{absolument}, et
\[
\int_I f_n \xrightarrow[n \to \infty]{} \int_I f .
\]
\end{theorem}
\admitted

\begin{remark}
La preuve honnête relève de la théorie de l'intégration de Lebesgue de
la troisième année~; l'énoncé, lui, sert constamment désormais.
L'hypothèse de \emph{domination} est tout l'enjeu~: la convergence
simple seule ne suffit pas ($f_n = n\,\mathbf{1}_{\intoo{0}{1/n}}$,
des bosses glissantes~: $\int f_n = 1 \not\to 0 = \int f$). Le
théorème vaut aussi pour un paramètre \omterm{def:b2:metric:continuity}{continu} ($f_\lambda$, $\lambda \to
\lambda_0$), par la caractérisation séquentielle des limites.
\end{remark}

\begin{example}[Une limite gaussienne, par domination]\label{ex:b2:integration:wallislimit}
Calculer $\displaystyle\lim_{n\to\infty} I_n$ où $I_n =
\int_0^\infty \Bigl(1 + \frac{t^2}{n}\Bigr)^{\!-n}\dd t$.
Simplement, $(1 + t^2/n)^n \to \eu^{t^2}$ (limite des intérêts
composés), donc les intégrandes tendent vers $\eu^{-t^2}$. Domination~: la
suite $n \mapsto (1 + u/n)^n$ est croissante pour $u \geq 0$
(l'inégalité arithmético-géométrique sur les $n + 1$ facteurs $1, 1 + \frac un, \dots, 1 +
\frac un$ donne $(1 + \frac u{n+1})^{n+1} \geq (1 + \frac
un)^n$), donc pour $n \geq 2$~:
\[
\Bigl(1 + \frac{t^2}{n}\Bigr)^{\!-n}
\leq \Bigl(1 + \frac{t^2}{2}\Bigr)^{\!-2},
\]
un dominant intégrable ($\sim 4t^{-4}$ à l'infini). Convergence
dominée~:
\[
I_n \xrightarrow[n\to\infty]{} \int_0^\infty \eu^{-t^2}\dd t
= \frac{\sqrt\pi}{2}
\]
(l'intégrale de Gauss de l'\cref{exo:b2:integration:8}). Vérification
finale~: le changement de variable $t = \sqrt n\tan\theta$ calcule $I_n$
exactement, $I_n = \sqrt n\int_0^{\pi/2}\cos^{2n-2}\theta\,\dd\theta
= \sqrt n\,W_{2n-2}$, et l'équivalent de Wallis $W_m \sim
\sqrt{\pi/(2m)}$ (\cref{lem:b2:comparison:wallis}) donne de nouveau $\sqrt
n\,W_{2n-2} \to \frac{\sqrt\pi}2$~: les deux piliers de ce
chapitre et du précédent concordent.
\end{example}

\begin{example}[Convergence dominée, paramètre continu]
\label{ex:b2:integration:arctanlimit}
Calculer
\[
\lim_{x\to+\infty}\int_0^\infty
\frac{\arctan(xt)}{1+t^2}\,\dd t .
\]
Pour chaque $t > 0$,
$\arctan(xt) \to \frac\pi2$ quand $x \to \infty$~; et la
domination
\[
\Bigl|\frac{\arctan(xt)}{1+t^2}\Bigr|
\leq \frac{\pi/2}{1+t^2},
\qquad\text{intégrable, indépendante de } x,
\]
vaut pour tout $x$. Par la forme à paramètre \omterm{def:b2:metric:continuity}{continu} du~\cref{thm:b2:integration:dominated} (caractérisation
séquentielle~: tester le long de tout $x_n \to \infty$),
\[
\int_0^\infty\frac{\arctan(xt)}{1+t^2}\,\dd t
\xrightarrow[x\to+\infty]{}
\frac\pi2\int_0^\infty\frac{\dd t}{1+t^2}
= \frac{\pi^2}{4} .
\]
Éclairage final~: le point isolé $t = 0$, où la limite simple
est $0$ plutôt que $\frac\pi2$, ne change rien --- la
fonction limite n'intervient que par son intégrale, l'une des
faveurs discrètes du théorème.
\end{example}

\section{Intégrales à paramètre}

\begin{theorem}[Continuité sous le signe intégral]\label{thm:b2:integration:continuity}
Soit $f \colon A \times I \to \R$ ($A$ un \omterm{def:b2:metric:def}{espace métrique}, $I$ un
intervalle) vérifiant~: $t \mapsto f(x, t)$ \omterm{def:b2:metric:continuity}{continue} par morceaux pour chaque
$x$~; $x \mapsto f(x, t)$ \omterm{def:b2:metric:continuity}{continue} pour chaque $t$~; et une
\emph{domination} $\abs{f(x,t)} \leq \varphi(t)$ ($\varphi$
intégrable sur $I$, indépendante de $x$). Alors
\[
F(x) = \int_I f(x, t)\,\dd t
\]
est définie et \omterm{def:b2:metric:continuity}{continue} sur $A$.\index{intégrale à paramètre}
\end{theorem}

\begin{proof}
Bonne définition~: la domination donne la convergence absolue. Continuité en
$x_0$~: pour toute suite $x_n \to x_0$, les fonctions $g_n(t) =
f(x_n, t)$ convergent simplement vers $f(x_0, t)$ (continuité en $x$)
sous la domination fixe $\varphi$~: la convergence dominée donne
$F(x_n) \to F(x_0)$~; on conclut par la caractérisation séquentielle de
la continuité (\cref{def:b2:metric:continuity}).
\end{proof}

\begin{theorem}[Dérivation sous le signe intégral]\label{thm:b2:integration:leibnizrule}
Soit $f \colon J \times I \to \R$ ($J$ un intervalle de paramètres)
vérifiant~: $t \mapsto f(x,t)$ intégrable sur $I$ pour chaque $x$~; $x \mapsto
f(x,t)$ de classe $C^1$ pour chaque $t$, la dérivée partielle
$\frac{\partial f}{\partial x}$ étant \omterm{def:b2:metric:continuity}{continue} par morceaux en $t$
et dominée~: $\bigl|\frac{\partial f}{\partial x}(x,t)\bigr| \leq
\psi(t)$ avec $\psi$ intégrable. Alors $F(x) = \int_I f(x,t)\dd t$
est de classe $C^1$ sur $J$ et
\[
F'(x) = \int_I \frac{\partial f}{\partial x}(x, t)\,\dd t .
\]
\end{theorem}

\begin{proof}
Fixons $x$ et $h_n \to 0$. Les taux d'accroissement
\[
\frac{F(x + h_n) - F(x)}{h_n}
= \int_I \frac{f(x + h_n, t) - f(x, t)}{h_n}\,\dd t
\]
ont des intégrandes convergeant simplement vers $\frac{\partial f}{\partial
x}(x, t)$, et dominés par $\psi(t)$~: par l'inégalité des accroissements
finis appliquée en $x$ à $t$ fixé,
\[
\Bigl|\frac{f(x + h_n, t) - f(x,t)}{h_n}\Bigr|
\leq \sup_{\xi} \Bigl|\frac{\partial f}{\partial x}(\xi, t)\Bigr|
\leq \psi(t) .
\]
La convergence dominée donne la limite $\int_I \frac{\partial
f}{\partial x}(x,t)\dd t$ des taux~: $F$ est dérivable,
de dérivée annoncée, \omterm{def:b2:metric:continuity}{continue} d'après le~\cref{thm:b2:integration:continuity}
appliqué à $\frac{\partial f}{\partial x}$.
\end{proof}

\begin{example}[Une intégrale à paramètre confrontée à une formule]\label{ex:b2:integration:arctanparam}
Soit $F(x) = \displaystyle\int_0^\infty \frac{\dd t}{t^2 + x}$
pour $x > 0$. Sur tout $\intcc{a}{b} \subset \intoo{0}{\infty}$,
l'intégrande est dominé par $\frac{1}{t^2 + a}$, intégrable
et indépendant de $x$~: $F$ est \omterm{def:b2:metric:continuity}{continue}
(\cref{thm:b2:integration:continuity}). Ici le théorème peut se
confronter à une valeur explicite~:
\[
F(x) = \Bigl[\frac{1}{\sqrt x}\arctan\frac{t}{\sqrt
x}\Bigr]_0^\infty = \frac{\pi}{2\sqrt x} ,
\]
visiblement \omterm{def:b2:metric:continuity}{continue}. Dérivons maintenant sous l'intégrale~: la
dérivée en $x$, $-\frac{1}{(t^2+x)^2}$, est dominée sur
$\intcc ab$ par $\frac{1}{(t^2+a)^2}$, intégrable~:
le~\cref{thm:b2:integration:leibnizrule} donne
\[
F'(x) = -\int_0^\infty \frac{\dd t}{(t^2 + x)^2}
\qquad\text{tandis que}\qquad
F'(x) = -\frac{\pi}{4}\,x^{-3/2} ,
\]
de sorte que nous avons \emph{calculé} une nouvelle intégrale gratuitement~:
$\int_0^\infty\frac{\dd t}{(t^2+x)^2} = \frac{\pi}{4x^{3/2}}$.
Éclairage final~: dériver une \omterm{thm:b2:integration:continuity}{intégrale à paramètre} connue est une
fabrique de nouvelles formules --- en itérant, on obtient
$\int_0^\infty\frac{\dd t}{(t^2+1)^n}$ pour tout $n$, sans aucun
changement de variable trigonométrique.
\end{example}

\begin{method}[Étudier une intégrale impropre]\label{met:b2:integration:study}
Étant donné $\int_a^b f$~:
\begin{enumerate}
  \item Localiser le problème~: recenser les bornes (ou points
        intérieurs) où $f$ est non bornée ou l'intervalle
        infini, et découper pour que chaque morceau ait exactement une
        borne problématique.
  \item Si $f$ est de signe constant près de cette borne, trouver un
        équivalent et comparer avec les échelles de référence du~\cref{thm:b2:integration:comparison}.
  \item Si $f$ oscille, tester d'abord $\abs f$ (convergence
        absolue). Si $\int\abs f$ diverge, essayer une intégration
        par parties pour échanger l'oscillation contre une décroissance, comme à
        l'\cref{ex:b2:integration:sint}~; des minorations comme
        $\abs{\sin t} \geq \sin^2t$ détectent une véritable
        semi-convergence.
  \item Pour une valeur, et pas seulement la nature~: parties, changement de variable,
        ou un paramètre (dériver une intégrale plus simple, comme à
        l'\cref{ex:b2:integration:arctanparam} et à
        l'\cref{ex:b2:integration:laplace}).
  \item Vérifications de bon sens sur toute valeur calculée~: signe et \omterm{def:b2:structures:generated}{ordre} de grandeur
        face à une borne grossière ($\int_0^\infty
        \eu^{-t^2}\dd t \in \intoo{0}{1 + \int_1^\infty
        \eu^{-t}}$, donc $\frac{\sqrt\pi}{2} \approx 0.886$ est
        plausible)~; et cohérence dimensionnelle par changement d'échelle
        ($t \mapsto \lambda t$ doit remettre les deux membres à la même
        échelle --- le détecteur le plus rapide d'un facteur perdu).
\end{enumerate}
\end{method}

\begin{remark}[Pièges fréquents]
Trois erreurs récurrentes. \emph{(i) Dominants dépendant du
paramètre~:} la domination $\abs{f(x,t)} \leq \varphi(t)$ doit
être uniforme en $x$ sur l'ensemble considéré~; elle vaut d'ordinaire sur les
segments $\intcc ab$ mais pas globalement --- pour
$\int_0^\infty\eu^{-xt}\dd t$ il n'existe aucun dominant intégrable
valable pour tout $x > 0$, mais dominer sur $x \geq a > 0$ suffit
pour travailler sur toute la demi-droite ouverte, puisque la continuité
et les dérivées sont des notions locales. \emph{(ii) Comparer des
intégrandes de signe quelconque~:} la boîte à outils de comparaison est faite pour les
fonctions \emph{positives}~; de $\abs f \leq g$ avec $\int g$ divergente on ne peut
rien conclure --- $\int_1^\infty\frac{\sin t}t\,\dd t$
\omterm{def:b2:integration:improper}{converge} bien que toute comparaison avec $\frac1t$ échoue.
\emph{(iii) Oublier la moitié du problème~:} sur
$\intoo{0}{\infty}$ toujours étudier les deux bornes séparément~;
$\int_0^\infty\frac{\dd t}{t}$ diverge aux \emph{deux}, et un
découpage d'apparence convergente peut compenser silencieusement deux infinis.
Le réflexe sûr est la liste de contrôle du~\cref{met:b2:integration:study}.
\end{remark}

\begin{example}[Un cas limite de Bertrand, au chiffre près]\label{ex:b2:integration:bertrandboundary}
L'échelle $\int^\infty\frac{\dd t}{t(\ln t)^\beta}$ du~\cref{thm:b2:integration:comparison}
se situe exactement à la frontière des échelles de puissances~; ses cas limites méritent un
calcul \omterm{def:b2:metric:complete}{complet}. Pour $\beta = 2$~:
\[
\int_\eu^{\infty}\frac{\dd t}{t(\ln t)^2}
= \Bigl[-\frac{1}{\ln t}\Bigr]_\eu^{\infty}
= 0 - (-1) = 1 ,
\]
une intégrale convergente à la valeur agréablement exacte~; tandis que pour
$\beta = 1$,
\[
\int_\eu^{x}\frac{\dd t}{t\ln t}
= \bigl[\ln\ln t\bigr]_\eu^{x} = \ln\ln x
\longrightarrow \infty ,
\]
divergente --- mais si lentement qu'atteindre $\ln\ln x = 10$
requiert $x = \eu^{\eu^{10}} \approx 10^{9566}$. Éclairage
final~: entre «~toute puissance $t^{-1-\varepsilon}$
\omterm{def:b2:integration:improper}{converge}~» et «~$t^{-1}$ diverge~» vit une échelle infinie
de niveaux logarithmiques, chacun raffinant le précédent~; le changement de variable
$u = \ln t$ ramène chaque barreau sur le précédent, ce qui
explique pourquoi les critères de Bertrand font écho à ceux de Riemann,
un cran plus haut.
\end{example}

\begin{remark}[Perspectives à l'intérieur de ce volume]
Les outils de ce chapitre vont être partout. La convergence
dominée est le moteur des approximations de l'unité du
chapitre suivant (noyaux glissants, Bernstein comme Fejér)~;
la continuité et la dérivation sous le signe intégral
produisent le calcul des coefficients de Fourier au chapitre de Fourier,
où chaque $c_n(f)$ est une \omterm{thm:b2:integration:continuity}{intégrale à paramètre} déguisée. La fonction $\Gamma$
revient deux fois~: au chapitre sur les intégrales multiples, où une
intégrale double démontre enfin la formule bêta--gamma d'Euler dans
toute sa généralité, et dans les chapitres de probabilités, où les
intégrales de type $\Gamma$ normalisent les densités usuelles et
calculent leurs moments. Et l'intégrale semi-convergente
$\int\frac{\sin t}{t}$ resurgit comme la constante de Gibbs du
chapitre de Fourier --- la même intégrale, mesurant le dépassement
des sommes partielles en un saut.
\end{remark}

\begin{definition}[La fonction $\Gamma$]\label{def:b2:integration:gamma}
Pour $x > 0$~:\index{fonction gamma}
\[
\Gamma(x) = \int_0^{\infty} t^{x-1}\,\eu^{-t}\,\dd t ,
\]
convergente aux deux bornes ($t^{x-1}$ intégrable en $0^+$ pour $x > 0$~;
décroissance exponentielle en $\infty$).
\end{definition}

\begin{theorem}\label{thm:b2:integration:gammaprops}
$\Gamma$ est \omterm{def:b2:metric:continuity}{continue} sur $\intoo{0}{+\infty}$, vérifie l'équation
fonctionnelle
\[
\Gamma(x + 1) = x\,\Gamma(x),
\qquad \Gamma(1) = 1,
\qquad\text{donc}\qquad \Gamma(n + 1) = n! ,
\]
et est de classe $C^1$ (et même $C^\infty$) avec $\Gamma'(x) =
\int_0^\infty t^{x-1}\eu^{-t}\ln t\,\dd t$.
\end{theorem}

\begin{proof}
Équation fonctionnelle~: intégrons par parties sur $\intcc{\varepsilon}{M}$
et faisons tendre les bornes~: $\int t^{x}\eu^{-t} = [-t^x\eu^{-t}] + x\int
t^{x-1}\eu^{-t}$, les termes de bord s'annulant --- en effet
$\varepsilon^x\eu^{-\varepsilon} \to 0$ quand $\varepsilon \to
0^+$ car $x > 0$, et $M^x\eu^{-M} \to 0$ quand $M \to \infty$
car l'exponentielle l'emporte sur toute puissance~; les deux
intégrales tronquées convergent vers leurs valeurs impropres par la convergence
établie dans la~\cref{def:b2:integration:gamma}. $\Gamma(1) = \int
\eu^{-t} = 1$~; la récurrence donne la factorielle.

Continuité sur $\intcc{a}{b} \subset \intoo{0}{\infty}$~: dominer
$t^{x-1}\eu^{-t}$ par $\varphi(t) = (t^{a-1} + t^{b-1})\eu^{-t}$,
intégrable et indépendante de $x \in \intcc{a}{b}$~:
le~\cref{thm:b2:integration:continuity} s'applique sur tout segment de ce type,
donc sur toute la demi-droite. Dérivabilité~: la dérivée en $x$,
$t^{x-1}\eu^{-t}\ln t$, est dominée sur $\intcc{a}{b}$ par $(t^{a-1}
+ t^{b-1})\eu^{-t}\,\abs{\ln t}$, toujours intégrable~:
le~\cref{thm:b2:integration:leibnizrule}~; en itérant on obtient toutes les
dérivées (chacune ajoute une puissance de $\ln t$, sans dommage).
\end{proof}

\begin{example}[Factorielles demi-entières]\label{ex:b2:integration:halffactorial}
L'équation fonctionnelle et $\Gamma\bigl(\frac12\bigr) =
\sqrt\pi$ (à un changement de variable de
l'\cref{exo:b2:integration:8}~: poser $t = u^2$ dans l'intégrale
de définition) engendrent toutes les valeurs demi-entières~:
\[
\Gamma\Bigl(\frac32\Bigr) = \frac12\,\Gamma\Bigl(\frac12\Bigr)
= \frac{\sqrt\pi}{2},
\qquad
\Gamma\Bigl(\frac52\Bigr) = \frac32\cdot\frac{\sqrt\pi}{2}
= \frac{3\sqrt\pi}{4},
\qquad
\Gamma\Bigl(\frac72\Bigr) = \frac{15\sqrt\pi}{8} .
\]
Puisque $\Gamma(n+1) = n!$, il est légitime d'écrire «~$\frac12! =
\frac{\sqrt\pi}{2} \approx 0.886$~»~: la factorielle a été
interpolée, et la courbe interpolante passe sous $1$
entre $0! = 1$ et $1! = 1$ (son minimum $\approx 0.8856$ en
$x \approx 1.4616$ correspond au tableau de convexité de la
partie I du problème du week-end). Éclairage final~: rien dans
l'intégrale $\int_0^\infty t^{x-1}\eu^{-t}\dd t$ ne privilégie
les entiers --- le caractère discret de la factorielle était un accident
du dénombrement, et $\sqrt\pi$ est ce qui vit entre $1$ et $1$.
\end{example}

\begin{remark}[Où va $\Gamma$ à partir d'ici]
Le problème du week-end de ce chapitre construit tout le calcul
d'Euler autour de $\Gamma$~: la \omterm{pb:b2:integration:1}{fonction bêta}, ses récurrences par
intégration par parties, les intégrales de Wallis comme valeurs de la
\omterm{pb:b2:integration:1}{fonction bêta}, et la \omterm{pb:b2:integration:1}{formule limite de Gauss}. Le chapitre sur les
intégrales multiples démontre la formule bêta--gamma d'Euler pour tous
les arguments par une intégrale double~; les chapitres de probabilités
retrouvent $\Gamma$ dans la normalisation des densités les plus courantes
et dans les moments des temps d'attente. Le volume de troisième année
reconstruit $\Gamma$ sur des fondations de Lebesgue, démontre le
théorème d'unicité de Bohr--Mollerup, et étend la formule de Stirling
des entiers à la demi-droite réelle par convergence dominée.
\end{remark}

\begin{example}[Un calcul classique par dérivation]\label{ex:b2:integration:laplace}
Pour $x \in \R$, posons $F(x) = \int_0^{\infty} \eu^{-t^2}\cos(xt)\,\dd
t$ (\omterm{def:b2:series:def}{absolument} convergente, dominée par $\eu^{-t^2}$). D'après
le~\cref{thm:b2:integration:leibnizrule} (domination de la
dérivée en $x$ par $t\,\eu^{-t^2}$, intégrable)~:
\[
F'(x) = -\int_0^\infty t\,\eu^{-t^2}\sin(xt)\,\dd t
= \Bigl[\frac{\eu^{-t^2}}{2}\sin(xt)\Bigr]_0^\infty
- \frac x2\int_0^\infty \eu^{-t^2}\cos(xt)\,\dd t
= -\frac x2\,F(x),
\]
(parties avec $u' = t\eu^{-t^2}$). L'équation différentielle $F' =
-\frac x2 F$ s'intègre en $F(x) = F(0)\,\eu^{-x^2/4}$~: l'intégrale de type gaussien
se reproduit elle-même. La constante $F(0) =
\int_0^\infty \eu^{-t^2}\dd t = \frac{\sqrt\pi}{2}$ est calculée à
l'\cref{exo:b2:integration:8} --- et de nouveau, par double intégration,
au~\cref{ch:b2:multint}.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b2:integration:1}
Nature de~:
$\displaystyle\int_0^1 \frac{\dd t}{\sqrt{t(1-t)}}$~;
$\;\displaystyle\int_1^\infty \frac{\ln t}{t^2}\dd t$~;
$\;\displaystyle\int_0^\infty \frac{\dd t}{1 + t^2\sin^2 t}$
\emph{(comparer avec le comportement de type harmonique divergent près de $t =
n\pi$)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:integration:2}
Calculer $\displaystyle\int_0^\infty t^n \eu^{-\lambda t}\,\dd t$
($\lambda > 0$) à l'aide de $\Gamma$, et
$\displaystyle\int_0^1 (\ln t)^n \dd t$ par le changement de variable $t =
\eu^{-u}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:integration:3}
Montrer que $\displaystyle\int_0^{\infty} \frac{\dd t}{(1 +
t^2)(1 + t^x)}$ est bien définie pour tout $x \in \R$ et
indépendante de $x$. \emph{(Substituer $t \mapsto \frac1t$ et
faire la moyenne des deux expressions.)} Quelle est sa valeur~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:4}
(Intégrales de Bertrand en une borne finie) Pour quels $(\alpha,
\beta)$ l'intégrale $\displaystyle\int_0^{1/2}
\frac{\dd t}{t^\alpha\,\abs{\ln t}^\beta}$ converge-t-elle~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:5}
Soit $F(x) = \displaystyle\int_0^{\infty} \frac{\eu^{-xt}}{1 +
t^2}\,\dd t$ pour $x \geq 0$. Montrer que $F$ est \omterm{def:b2:metric:continuity}{continue} sur
$\intco{0}{\infty}$, de classe $C^2$ sur $\intoo{0}{\infty}$, y vérifie $F'' +
F = \frac1x$, et que $F(x) \to 0$ quand $x \to +\infty$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:6}
(Frullani) Soit $f$ \omterm{def:b2:metric:continuity}{continue} sur $\intco{0}{+\infty}$ admettant une
limite finie $f(\infty)$ en $+\infty$. Montrer que pour $a, b > 0$~:
\[
\int_0^{\infty} \frac{f(at) - f(bt)}{t}\,\dd t
= \bigl(f(0) - f(\infty)\bigr)\,\ln\frac ba .
\]
\emph{(Sur $\intcc{\varepsilon}{M}$, substituer dans chaque morceau et
regrouper en $\int_{a\varepsilon}^{b\varepsilon} -
\int_{aM}^{bM}$ de $\frac{f(u)}u\,\dd u$~; encadrer en utilisant la
continuité en $0$ et la limite en $\infty$.)} Calculer
$\int_0^\infty \frac{\eu^{-t} - \eu^{-2t}}{t}\dd t$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:7}
Justifier et calculer $\lim_{n\to\infty} \displaystyle\int_0^n
\Bigl(1 - \frac tn\Bigr)^{\!n} t^{x-1}\,\dd t$ pour $x > 0$
\emph{(convergence dominée avec $\varphi(t) = \eu^{-t}t^{x-1}$,
en utilisant $(1 - t/n)^n \leq \eu^{-t}$~; la limite est $\Gamma(x)$)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:integration:8}
(L'intégrale de Gauss par une astuce de paramètre) Pour $x \geq 0$, posons
\[
G(x) = \Bigl(\int_0^x \eu^{-t^2}\dd t\Bigr)^{\!2},
\qquad
H(x) = \int_0^1 \frac{\eu^{-x^2(1+t^2)}}{1 + t^2}\,\dd t .
\]
Montrer que $G' + H' = 0$ (dériver $H$ sous l'intégrale et
substituer $u = xt$ dans l'intégrale obtenue), en déduire $G(x) + H(x)
= \frac\pi4$ pour tout $x$, et conclure
\[
\int_0^{\infty} \eu^{-t^2}\,\dd t = \frac{\sqrt\pi}{2} .
\]
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:integration:9}
Montrer que $\Gamma$ est \emph{log-convexe}~: $\ln\Gamma$ est convexe sur
$\intoo{0}{\infty}$. \emph{(l'inégalité de Cauchy--Schwarz pour les intégrales appliquée
à $t^{(x+y)/2 - 1}\eu^{-t} = \bigl(t^{x-1}\eu^{-t}\bigr)^{1/2}
\bigl(t^{y-1}\eu^{-t}\bigr)^{1/2}$ donne
$\Gamma\bigl(\frac{x+y}{2}\bigr)^2 \leq \Gamma(x)\Gamma(y)$~;
combiner avec la continuité et l'\cref{exo:b2:realfun:8}.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:integration:10}
(Intégrale de Dirichlet) Posons $F(x) = \displaystyle\int_0^{\infty}
\frac{\sin t}{t}\,\eu^{-xt}\,\dd t$ pour $x > 0$.
\begin{enumerate}
  \item Justifier $F'(x) = -\frac{1}{1 + x^2}$ (dériver sous
        l'intégrale~; calculer $\int_0^\infty \eu^{-xt}\sin t\,\dd
        t$ par deux intégrations par parties).
  \item Montrer que $F(x) \to 0$ quand $x \to +\infty$ et en déduire $F(x) =
        \frac\pi2 - \arctan x$.
  \item En admettant la continuité de $F$ en $0^+$ (un théorème de type Abel),
        conclure la valeur de l'intégrale semi-convergente~:
        \[
        \int_0^{\infty} \frac{\sin t}{t}\,\dd t = \frac{\pi}{2}.
        \]
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:11}
Justifier la convergence de $\displaystyle\int_0^\infty
\Bigl(\frac{\sin t}{t}\Bigr)^{\!2}\dd t$, puis la calculer par une
intégration par parties et l'\cref{exo:b2:integration:10}~:
\[
\int_0^\infty \Bigl(\frac{\sin t}{t}\Bigr)^{\!2}\dd t
= \frac{\pi}{2} .
\]
(La même valeur que $\int_0^\infty \frac{\sin t}{t}\dd t$ --- mais
cette fois la convergence est absolue.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:integration:12}
(La queue gaussienne) Pour $x > 0$, posons $T(x) = \displaystyle
\int_x^\infty \eu^{-t^2}\dd t$.
\begin{enumerate}
  \item En écrivant $\eu^{-t^2} = \frac{1}{-2t}\cdot(-2t\,\eu^{-t^2})$,
        intégrer deux fois par parties pour obtenir
        \[
        T(x) = \eu^{-x^2}\Bigl(\frac{1}{2x} -
        \frac{1}{4x^3}\Bigr) +
        \frac34\int_x^\infty \frac{\eu^{-t^2}}{t^4}\,\dd t .
        \]
  \item Majorer le reste~: $0 \leq \frac34\int_x^\infty
        t^{-4}\eu^{-t^2}\dd t \leq \frac{3}{8x^5}\,\eu^{-x^2}$,
        et en déduire l'encadrement
        \[
        \eu^{-x^2}\Bigl(\frac{1}{2x} - \frac{1}{4x^3}\Bigr)
        \leq T(x) \leq \frac{\eu^{-x^2}}{2x},
        \qquad\text{d'où}\qquad
        T(x) \sim \frac{\eu^{-x^2}}{2x} \quad (x \to +\infty).
        \]
  \item Pourquoi la série \omterm{def:b2:linalg:alternating}{alternée} complète obtenue en itérant
        les parties ne peut-elle jamais converger pour $x$ fixé~? \emph{(Comparer la
        croissance des coefficients $1\cdot3\cdots(2k-1)$ avec les
        puissances $(2x^2)^k$.)}
\end{enumerate}
\end{exercise}

\section{Problème~: les intégrales d'Euler --- bêta, gamma et la formule
limite de Gauss}

\begin{problem}\label{pb:b2:integration:1}
La fonction $\Gamma$ de la~\cref{def:b2:integration:gamma} est une
moitié du calcul des intégrales d'Euler~; l'autre moitié est la
\emph{fonction bêta}\index{fonction bêta}
\[
B(x, y) = \int_0^1 t^{x-1}(1 - t)^{y-1}\,\dd t .
\]
Ce problème développe le couple $(\Gamma, B)$ avec les seuls outils de
ce chapitre --- intégration par parties, changement de variable,
convergence dominée --- et culmine dans la \emph{formule bêta--gamma
d'Euler} $B(x,y) = \frac{\Gamma(x)\Gamma(y)}
{\Gamma(x+y)}$ sur les demi-entiers et dans la \emph{formule limite de
Gauss}\index{formule limite de Gauss} pour $\Gamma$. En chemin,
les intégrales de Wallis du~\cref{lem:b2:comparison:wallis}
réapparaissent comme valeurs de la \omterm{pb:b2:integration:1}{fonction bêta}, et la formule de
duplication de Legendre en découle.

\medskip
\noindent\textbf{Partie I --- Structure fine de $\Gamma$.}
\begin{enumerate}
  \item Rappeler pourquoi $\Gamma(x) = \int_0^\infty t^{x-1}\eu^{-t}\dd
        t$ \omterm{def:b2:integration:improper}{converge} exactement pour $x > 0$, et montrer
        \[
        \Gamma(x) \sim \frac1x \qquad (x \to 0^+)
        \]
        \emph{(équation fonctionnelle et continuité de $\Gamma$ en
        $1$)}.
  \item Montrer que $\Gamma\bigl(\tfrac12\bigr) = \sqrt\pi$
        \emph{(substituer $t = u^2$ et invoquer
        l'\cref{exo:b2:integration:8})}, et en déduire
        $\int_\R \eu^{-u^2/2}\dd u = \sqrt{2\pi}$.
  \item Montrer par récurrence, pour $n \in \N$~:
        \[
        \Gamma\Bigl(n + \frac12\Bigr)
        = \frac{(2n)!}{4^n\,n!}\,\sqrt\pi .
        \]
  \item Justifier $\Gamma''(x) = \int_0^\infty
        t^{x-1}\eu^{-t}(\ln t)^2\dd t > 0$, et en déduire que
        $\Gamma$ est strictement convexe, atteint un unique minimum en
        un certain $x_0 \in \intoo{1}{2}$ \emph{($\Gamma(1) = \Gamma(2)
        = 1$ et Rolle)}, décroît sur $\intoo{0}{x_0}$ et
        croît sur $\intoo{x_0}{\infty}$.
  \item Montrer que $\Gamma$ l'emporte sur toute puissance~: pour tout $k \in
        \N$, $x^k = o\bigl(\Gamma(x)\bigr)$ quand $x \to +\infty$
        \emph{(encadrer $x$ entre deux entiers et utiliser $\Gamma(n+1)
        = n!$ avec la monotonie de la question 4)}.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- La \omterm{pb:b2:integration:1}{fonction bêta}, par parties.}
\begin{enumerate}[resume]
  \item Montrer que $B(x,y)$ \omterm{def:b2:integration:improper}{converge} exactement pour $x > 0$ et $y >
        0$, et que $B(x,y) = B(y,x)$.
  \item Calculer $B(x, 1) = \frac1x$, et montrer par intégration par
        parties, pour $x, y > 0$~:
        \[
        B(x, y+1) = \frac{y}{x}\,B(x+1, y) .
        \]
  \item À partir de la décomposition $t^{x-1}(1-t)^{y-1} =
        t^{x}(1-t)^{y-1} + t^{x-1}(1-t)^{y}$ déduire $B(x,y) =
        B(x+1,y) + B(x,y+1)$, et combiner avec la question 7 pour obtenir
        les \emph{relations de descente}
        \[
        B(x, y+1) = \frac{y}{x+y}\,B(x,y),
        \qquad
        B(x+1, y) = \frac{x}{x+y}\,B(x,y) .
        \]
  \item En déduire, pour $m, n \geq 1$ entiers~:
        \[
        B(m, n) = \frac{(m-1)!\,(n-1)!}{(m+n-1)!}
        = \frac{1}{(m+n-1)\binom{m+n-2}{m-1}} .
        \]
  \item Montrer la \emph{formule d'Euler pour un argument entier}~:
        pour tout $x > 0$ et $n \in \N^*$,
        \[
        B(x, n) = \frac{\Gamma(x)\,\Gamma(n)}{\Gamma(x + n)}
        \]
        \emph{(récurrence sur $n$~: les deux membres valent $\frac1x$ en $n
        = 1$ et vérifient la même relation de descente)}.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Les intégrales de Wallis comme valeurs de la
\omterm{pb:b2:integration:1}{fonction bêta}.}
\begin{enumerate}[resume]
  \item Substituer $t = \sin^2\theta$ pour obtenir la forme trigonométrique
        \[
        B(x, y) = 2\int_0^{\pi/2}
        \sin^{2x-1}\theta\,\cos^{2y-1}\theta\,\dd\theta .
        \]
  \item En déduire $W_n = \frac12\,B\bigl(\frac{n+1}2, \frac12\bigr)$
        pour l'intégrale de Wallis $W_n = \int_0^{\pi/2}\sin^n
        \theta\,\dd\theta$, et retrouver la récurrence $W_n =
        \frac{n-1}{n}W_{n-2}$ du~\cref{lem:b2:comparison:wallis}
        à partir des seules relations de descente de la question 8.
  \item Calculer $B\bigl(\frac12, \frac12\bigr) = 2W_0 = \pi$ et
        le confronter à $\Gamma\bigl(\frac12\bigr)^2/\Gamma(1)$~:
        la formule d'Euler est vérifiée en $\bigl(\frac12,
        \frac12\bigr)$.
  \item Établir la forme close $W_{2n} = \frac\pi2\,
        \frac{(2n)!}{4^n(n!)^2}$ à partir de la récurrence, et vérifier
        \[
        B\Bigl(n + \frac12, \frac12\Bigr)
        = \frac{\Gamma\bigl(n + \frac12\bigr)\Gamma\bigl(
        \frac12\bigr)}{\Gamma(n+1)} .
        \]
        Conclure, par récurrence avec les relations de descente, que
        la formule d'Euler $B(x,y) =
        \frac{\Gamma(x)\Gamma(y)}{\Gamma(x+y)}$ est vérifiée dès que $2x$
        et $2y$ sont des entiers positifs.
  \item Substituer $u = \frac{t}{1-t}$ pour obtenir la troisième
        forme classique
        \[
        B(x,y) = \int_0^\infty
        \frac{u^{x-1}}{(1+u)^{x+y}}\,\dd u ,
        \]
        et vérifier directement le cas $x = y = \frac12$ ($u = v^2$
        le ramène à $\int_0^\infty\frac{2\,\dd v}{1+v^2}$).
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- La \omterm{pb:b2:integration:1}{formule limite de Gauss}.}
\begin{enumerate}[resume]
  \item Pour $x > 0$ et $n \in \N^*$, montrer par $n$ intégrations
        par parties successives~:
        \[
        \int_0^n \Bigl(1 - \frac tn\Bigr)^{\!n} t^{x-1}\,\dd t
        = \frac{n!\;n^x}{x(x+1)\cdots(x+n)} .
        \]
  \item Conclure à l'aide de l'\cref{exo:b2:integration:7} (convergence
        dominée) la \emph{\omterm{pb:b2:integration:1}{formule limite de Gauss}}~:
        \[
        \Gamma(x) = \lim_{n\to\infty}
        \frac{n!\;n^x}{x(x+1)\cdots(x+n)}
        \qquad (x > 0).
        \]
  \item En prenant les logarithmes, montrer que pour $x > 0$~:
        \[
        \ln\Gamma(x) = -\ln x - \gamma x +
        \sum_{k=1}^{\infty}\Bigl(\frac xk -
        \ln\Bigl(1 + \frac xk\Bigr)\Bigr),
        \]
        où $\gamma$ est la constante d'Euler
        (\cref{ex:b2:comparison:harmonic})~; justifier la
        convergence de la série \emph{(le terme général est
        $\sim \frac{x^2}{2k^2}$)}.
  \item Utiliser la formule de Gauss en $x = \frac12$ et l'équivalent du
        coefficient binomial central $\binom{2n}{n} \sim
        \frac{4^n}{\sqrt{\pi n}}$
        (\cref{ex:b2:comparison:centralbinomial}) pour recalculer
        $\Gamma\bigl(\frac12\bigr) = \sqrt\pi$~: la constante de
        Stirling et l'intégrale de Gauss sont le même nombre
        sous deux déguisements.
  \item Vérifier que la formule de Gauss redémontre l'équation
        fonctionnelle~: à partir de l'identité exacte
        \[
        \frac{n!\,n^{x+1}}{(x+1)\cdots(x+n+1)}
        = \frac{n!\,n^{x}}{x(x+1)\cdots(x+n)}\cdot
        \frac{n\,x}{x+n+1},
        \]
        conclure de nouveau $\Gamma(x+1) = x\,\Gamma(x)$. (La formule de Gauss
        détermine $\Gamma$ à elle seule~; le volume de troisième année
        démontre le théorème plus fin de Bohr--Mollerup~: l'équation
        fonctionnelle et la log-convexité suffisent déjà à caractériser
        $\Gamma$.)
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- Dividendes.}
\begin{enumerate}[resume]
  \item Pour $a > 0$, montrer $\int_0^\infty \eu^{-t^a}\dd t =
        \Gamma\bigl(1 + \frac1a\bigr)$, et calculer la limite quand
        $a \to +\infty$ par convergence dominée \emph{(limite simple
        $\mathbf 1_{t < 1}$~; dominer par $1$ sur
        $\intoc{0}{1}$ et par $\eu^{-t^2}$ au-delà, pour $a \geq
        2$)}. Confronter la réponse à la continuité de
        $\Gamma$.
  \item Pour $n \geq 1$, montrer
        \[
        \int_0^1 \frac{\dd t}{\sqrt{1 - t^n}}
        = \frac1n\,B\Bigl(\frac1n, \frac12\Bigr),
        \]
        et retrouver les valeurs $2$ ($n = 1$) et $\frac\pi2$ ($n
        = 2$). (Pour $n = 4$ c'est la \emph{constante de la
        lemniscate}, qui n'a pas de forme close élémentaire~; son
        histoire relève de la théorie des intégrales elliptiques.)
  \item (Moments) Pour $x > 0$ et $k \in \N$, montrer
        \[
        \frac{1}{\Gamma(x)}\int_0^\infty
        t^{k}\,t^{x-1}\eu^{-t}\,\dd t
        = \frac{\Gamma(x+k)}{\Gamma(x)}
        = x(x+1)\cdots(x+k-1),
        \]
        la factorielle croissante~; vérifier que $x = 1$ donne $k!$. (Dans
        les chapitres de probabilités, c'est le moment d'\omterm{def:b2:structures:generated}{ordre} $k$ d'une
        densité de temps d'attente standard.)
  \item Montrer l'identité de la \omterm{pb:b2:integration:1}{fonction bêta}, valable pour \emph{tout} $x > 0$~:
        \[
        B(x, x) = 2^{1-2x}\,B\Bigl(x, \frac12\Bigr)
        \]
        \emph{(substituer $t = \frac{1+s}2$, exploiter la
        symétrie en $s$, puis poser $s = \sqrt v$)}. En déduire, pour
        $2x \in \N^*$, la \emph{formule de duplication de Legendre}
        \[
        \Gamma(x)\,\Gamma\Bigl(x + \frac12\Bigr)
        = 2^{1-2x}\,\sqrt\pi\;\Gamma(2x),
        \]
        et la vérifier directement en $x = n$ via la question 3. (Pour
        $x$ général elle découle de la même identité une fois
        la formule d'Euler connue pour tous les arguments --- la preuve par
        intégrale double du chapitre sur les intégrales multiples.)
  \item Synthèse. En une phrase chacune~: (i) où l'intégration
        par parties a porté toute la partie II~; (ii) où
        la convergence dominée est intervenue dans les parties IV et V~; (iii) quels
        ingrédients asymptotiques ont été importés du chapitre de
        comparaison~; (iv) ce qui est désormais démontré de la formule d'Euler
        $B(x,y) = \Gamma(x)\Gamma(y)/\Gamma(x+y)$, et ce qu'il
        reste à établir par le calcul de l'intégrale double.
\end{enumerate}
\end{problem}
