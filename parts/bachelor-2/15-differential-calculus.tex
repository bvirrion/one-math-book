\chapter{Differential Calculus}\label{ch:b2:diffcalc}

The two-variable calculus of the Year 1 volume matures into the
differential calculus of maps between normed spaces: the
\emph{differential} as the best linear approximation, the chain rule
in full generality, Schwarz's symmetry theorem \emph{proved}, Taylor
formulas, and the complete second-order analysis of extrema. The
inverse function theorem, crown of the theory, is stated with its
proof strategy --- a Banach fixed point --- made explicit.

Throughout, $U$ is an open subset of $\R^n$ (or a normed space; the
finite-dimensional case carries all the ideas), $f \colon U \to
\R^m$.

\section{The differential}

\begin{definition}\label{def:b2:diffcalc:differential}
$f$ is \emph{differentiable at $a$}\index{differential} when there
is a (continuous) linear map $\dd f_a \colon \R^n \to \R^m$ with
\[
f(a + h) = f(a) + \dd f_a(h) + o(\norm h)
\qquad (h \to 0).
\]
The map $\dd f_a$ is unique; its matrix in the canonical bases is
the \emph{Jacobian matrix}\index{Jacobian matrix} $J_f(a) =
\bigl(\frac{\partial f_i}{\partial x_j}(a)\bigr)$.
Differentiability implies continuity and the existence of all
directional derivatives $\dd f_a(v) = \lim_{t\to0}\frac{f(a + tv)
- f(a)}{t}$; the converse fails
(\cref{exo:b2:diffcalc:2}). For $m = 1$, $\dd f_a(h) = \langle
\nabla f(a), h\rangle$: the gradient of Year 1, now understood as
the vector representing the differential.
\end{definition}

\begin{theorem}[$C^1$ criterion]\label{thm:b2:diffcalc:c1}
If all partial derivatives of $f$ exist on $U$ and are continuous
at $a$, then $f$ is differentiable at $a$. ``$C^1$ on $U$'' ---
continuous partials --- therefore implies differentiability
everywhere, with continuous differential.
\end{theorem}

\begin{proof}
Component by component ($m = 1$ suffices). The Year 1 proof for two
variables --- move one coordinate at a time, apply the one-variable
mean value theorem on each leg, and use continuity of the partials
at $a$ --- generalizes verbatim to $n$ legs:
\[
f(a + h) - f(a) = \sum_{j=1}^{n} \bigl(f(a + h^{(j)}) - f(a +
h^{(j-1)})\bigr)
= \sum_j h_j\,\frac{\partial f}{\partial x_j}(\xi_j),
\]
where $h^{(j)}$ freezes the first $j$ coordinates of $h$ and
$\xi_j$ lies on the $j$-th leg; continuity turns each
$\frac{\partial f}{\partial x_j}(\xi_j)$ into $\frac{\partial
f}{\partial x_j}(a) + o(1)$, and the error is $o(\norm h)$.
\end{proof}

\begin{theorem}[Chain rule]\label{thm:b2:diffcalc:chainrule}
If $f$ is differentiable at $a$ and $g$ at $f(a)$, then $g \circ f$
is differentiable at $a$ with
\[
\dd(g \circ f)_a = \dd g_{f(a)} \circ \dd f_a ,
\qquad
J_{g\circ f}(a) = J_g\bigl(f(a)\bigr)\,J_f(a) :
\]
Jacobians multiply.\index{chain rule}
\end{theorem}

\begin{proof}
Write $f(a + h) = f(a) + \dd f_a(h) + \norm h\,\varepsilon_1(h)$
and $g(b + k) = g(b) + \dd g_b(k) + \norm k\,\varepsilon_2(k)$ with
$b = f(a)$, $k = k(h) = \dd f_a(h) + \norm h \varepsilon_1(h)$.
Substituting,
\[
g(f(a+h)) = g(b) + \dd g_b\bigl(\dd f_a(h)\bigr)
+ \norm h\,\dd g_b(\varepsilon_1(h)) + \norm{k}\,\varepsilon_2(k),
\]
and both error terms are $o(\norm h)$: the first since $\dd g_b$ is
continuous and $\varepsilon_1 \to 0$; the second since $\norm k
\leq C\norm h$ (bounded linear map plus small term) and
$\varepsilon_2(k) \to 0$ as $h \to 0$.
\end{proof}

\begin{example}[Radial functions, once and for all]\label{ex:b2:diffcalc:radial}
Let $r(x) = \norm x_2$ on $\R^n\setminus\{0\}$ and $f = g \circ
r$ with $g$ a $C^1$ function of one variable. First, $r$ is
differentiable away from $0$: from $r^2 = \sum x_i^2$,
\[
\frac{\partial r}{\partial x_i} = \frac{x_i}{r},
\qquad\text{i.e.}\qquad
\nabla r(x) = \frac{x}{\norm x} ,
\]
the unit radial vector (differentiate $r^2$ and divide --- or
apply the chain rule to $\sqrt{\cdot}$). Then the chain rule
gives, for every radial function,
\[
\nabla f(x) = g'\bigl(\norm x\bigr)\,\frac{x}{\norm x} .
\]
Worked instance: $g(r) = \frac1r$ yields $\nabla\frac{1}{\norm
x} = -\frac{x}{\norm x^3}$, the inverse-square field of
gravitation and electrostatics --- direction radial, magnitude
$\frac{1}{\norm x^2}$. Closing insight: gradients of radial
functions are radial because the level sets are spheres and
the gradient is orthogonal to level sets; at $x = 0$, by
contrast, $r$ is \emph{not} differentiable (no candidate
linear map matches $\norm h$ from all directions) --- smooth
radial profiles need $g'(0) = 0$ to cross the origin
gracefully.
\end{example}

\begin{theorem}[Mean value inequality]\label{thm:b2:diffcalc:mvi}
Let $f$ be differentiable on $U$ and let the segment $\intcc{a}{b}
= \{a + t(b-a)\}$ lie in $U$. Then
\[
\norm{f(b) - f(a)} \leq \norm{b - a}\,
\sup_{x \in \intcc{a}{b}} \vertiii{\dd f_x} .
\]
In particular, a differentiable map with zero differential on a
\emph{connected} open set is constant.
\end{theorem}

\begin{proof}
The function $\varphi(t) = f(a + t(b-a))$ is differentiable on
$\intcc{0}{1}$ with $\varphi'(t) = \dd f_{a + t(b-a)}(b - a)$
(chain rule), of norm $\leq M\norm{b-a}$ with $M$ the displayed
sup. For $\R$-valued $f$ the one-variable mean value inequality
concludes; for vector values apply it to $t \mapsto \langle u,
\varphi(t)\rangle$ with $u$ the unit vector along $f(b) - f(a)$.
Constancy: locally constant (segments in balls) plus connectedness
(the set where $f$ equals a given value is open and closed:
\cref{ch:b2:metric}).
\end{proof}

\begin{example}[A Lipschitz constant from the MVI]\label{ex:b2:diffcalc:lipschitz}
Is $f(x, y) = \sin x\,\sin y$ Lipschitz on $\R^2$, and with
which constant? Its gradient is $\nabla f = (\cos x\sin y,\
\sin x\cos y)$, of squared norm
\[
\cos^2x\sin^2y + \sin^2x\cos^2y
\leq \sin^2 y + \cos^2y\cdot 1 = 1
\]
(bound $\cos^2x$ and $\sin^2x$ by $1$ separately), so
$\vertiii{\dd f_{(x,y)}} = \norm{\nabla f} \leq 1$ everywhere,
and \cref{thm:b2:diffcalc:mvi} on the segment between any two
points gives
\[
\abs{f(b) - f(a)} \leq \norm{b - a}_2 :
\]
$f$ is $1$-Lipschitz, and the constant is sharp (near the
origin, $f(x, \tfrac\pi2) = \sin x$ has slope $1$). Closing
insight: the mean value inequality converts a pointwise bound
on the differential into a global modulus of continuity ---
the standard route to Lipschitz estimates in every dimension,
and the engine inside \cref{exo:b2:diffcalc:12}.
\end{example}

\section{Second derivatives}

\begin{theorem}[Schwarz]\label{thm:b2:diffcalc:schwarz}
If $f$ is $C^2$ on $U$ (all second partials exist and are
continuous), then for all $i, j$:\index{Schwarz's theorem}
\[
\frac{\partial^2 f}{\partial x_i\,\partial x_j}
= \frac{\partial^2 f}{\partial x_j\,\partial x_i} .
\]
\end{theorem}

\begin{proof}
Two variables suffice ($x = x_i$, $y = x_j$, others frozen).
Consider the second difference
\[
\Delta(h) = f(a + h, b + h) - f(a + h, b) - f(a, b + h) + f(a,b) .
\]
Fix $h$ and set $\varphi(x) = f(x, b+h) - f(x, b)$: then $\Delta(h)
= \varphi(a + h) - \varphi(a)$, and two applications of the mean
value theorem give
\[
\Delta(h) = h\,\varphi'(\xi)
= h\Bigl(\frac{\partial f}{\partial x}(\xi, b+h) -
\frac{\partial f}{\partial x}(\xi, b)\Bigr)
= h^2\,\frac{\partial^2 f}{\partial y\,\partial x}(\xi, \eta),
\]
with $\xi \in \intoo{a}{a+h}$, $\eta \in \intoo{b}{b+h}$. By
continuity, $\frac{\Delta(h)}{h^2} \to \frac{\partial^2 f}{\partial
y\partial x}(a, b)$ as $h \to 0$. The same computation with the
roles of the variables exchanged (freeze the second variable first)
gives $\frac{\Delta(h)}{h^2} \to \frac{\partial^2 f}{\partial x
\partial y}(a,b)$: the two limits of the same quantity coincide.
\end{proof}

\begin{example}[Why $C^2$ is needed: Peano's counterexample]\label{ex:b2:diffcalc:peano}
Let $f(x, y) = \dfrac{xy(x^2 - y^2)}{x^2 + y^2}$, $f(0,0) = 0$.
Away from the origin $f$ is $C^\infty$; at the origin all first
and second partials exist, but the mixed ones disagree. Compute
along the axes: $f(x, 0) = f(0, y) = 0$, and for $y \neq 0$,
\[
\frac{\partial f}{\partial x}(0, y)
= \lim_{x\to0}\frac{f(x,y)}{x}
= \frac{y(0 - y^2)}{y^2} = -y ,
\qquad\text{symmetrically}\qquad
\frac{\partial f}{\partial y}(x, 0) = x .
\]
Hence
\[
\frac{\partial^2 f}{\partial y\,\partial x}(0,0)
= \frac{\dd}{\dd y}\Bigl[\frac{\partial f}{\partial
x}(0,y)\Bigr]_{y=0} = -1,
\qquad
\frac{\partial^2 f}{\partial x\,\partial y}(0,0) = +1 :
\]
the two mixed partials exist and differ. No contradiction with
\cref{thm:b2:diffcalc:schwarz}: the second partials of $f$ are
not continuous at $0$ (test along $y = tx$). Closing insight:
Schwarz's theorem is a genuine theorem about \emph{continuity},
not a formal identity --- and the hypothesis ``$C^2$'' in
Taylor--Young below is doing real work.
\end{example}

\begin{theorem}[Taylor--Young at order 2]\label{thm:b2:diffcalc:taylor}
Let $f \colon U \to \R$ be $C^2$ and $a \in U$. Then, as $h \to 0$,
\[
f(a + h) = f(a) + \langle\nabla f(a), h\rangle
+ \frac12\, \langle H_a h,\, h\rangle + o\bigl(\norm h^2\bigr),
\]
where $H_a = \bigl(\frac{\partial^2 f}{\partial x_i\partial
x_j}(a)\bigr)$ is the (symmetric, by Schwarz) \emph{Hessian
matrix}.\index{Hessian matrix}
\end{theorem}

\begin{proof}
Apply the one-variable Taylor--Young theorem (Year 1 volume) to
$\varphi(t) = f(a + th)$ on $\intcc{0}{1}$: by the chain rule,
$\varphi'(t) = \langle \nabla f(a + th), h\rangle$ and $\varphi''(t)
= \langle H_{a+th}h, h\rangle$, both continuous in $t$. Then
$\varphi(1) = \varphi(0) + \varphi'(0) + \frac12\varphi''(\theta)$
(Taylor--Lagrange) with $\theta \in \intoo{0}{1}$, and continuity
of the second partials converts $\varphi''(\theta) =
\varphi''(0) + o(1)\cdot\norm h^2$-uniformly: the displayed
expansion.
\end{proof}

\begin{example}[An expansion two ways]\label{ex:b2:diffcalc:taylortwoways}
Expand $f(x, y) = \eu^x\cos y$ at the origin to order $2$.
\emph{By composition of one-variable expansions:}
\[
\eu^x\cos y = \Bigl(1 + x + \frac{x^2}{2} +
o(x^2)\Bigr)\Bigl(1 - \frac{y^2}{2} + o(y^2)\Bigr)
= 1 + x + \frac{x^2 - y^2}{2} + o\bigl(\norm{(x,y)}^2\bigr) .
\]
\emph{By partial derivatives:} $f_x = \eu^x\cos y$, $f_y =
-\eu^x\sin y$, so $\nabla f(0) = (1, 0)$; and $f_{xx} = f$,
$f_{yy} = -f$, $f_{xy} = -\eu^x\sin y$ give $H_0 =
\operatorname{diag}(1, -1)$: \cref{thm:b2:diffcalc:taylor}
reproduces $1 + x + \frac12(x^2 - y^2)$. The two computations
agree, and the composition route was faster --- no second
partials at all. Closing insight: the origin is \emph{not} a
critical point ($\nabla f \neq 0$), so despite the indefinite
Hessian there is no saddle to declare: the linear term rules,
and the second-order test only ever speaks at critical
points.
\end{example}

\begin{theorem}[Second-order extremum test, proved]\label{thm:b2:diffcalc:extrema}
Let $f$ be $C^2$ near a critical point $a$ ($\nabla f(a) = 0$),
with Hessian $H = H_a$.
\begin{enumerate}
  \item If $H$ is positive definite, $a$ is a strict local minimum
        (negative definite: maximum).
  \item If $H$ has eigenvalues of both signs, $a$ is a saddle: no
        extremum.
  \item If $H$ is singular (and semidefinite), no conclusion.
\end{enumerate}
The Year 1 ``$rt - s^2$'' test is the case $n = 2$: $\det H = rt -
s^2$, $\operatorname{tr}$-sign read off $r$.
\end{theorem}

\begin{proof}
By the spectral theorem (\cref{thm:b2:quadratic:spectral}), $H$'s
quadratic form is squeezed between its extreme eigenvalues:
$\lambda_{\min}\norm h^2 \leq \langle Hh, h\rangle \leq
\lambda_{\max}\norm h^2$.

(1) If $\lambda_{\min} > 0$: Taylor--Young gives
\[
f(a + h) - f(a) \geq \frac{\lambda_{\min}}{2}\norm h^2 -
o(\norm h^2) > 0
\]
for small $h \neq 0$: strict local minimum.

(2) Along an eigenvector $v_+$ with $\lambda_+ > 0$: $f(a + tv_+) -
f(a) = \frac{\lambda_+}{2}t^2 + o(t^2) > 0$ for small $t$; along
$v_-$ with $\lambda_- < 0$ the difference is negative: both signs
occur in every neighborhood.

(3) $f(x,y) = x^2 + y^4$, $x^2 - y^4$, $x^2 + y^3$ share the same
singular semidefinite Hessian at $0$ with three different
behaviors.
\end{proof}

\begin{example}[A complete classification, global included]\label{ex:b2:diffcalc:quarticrun}
Classify all extrema of $f(x, y) = x^4 + y^4 - 4xy$ on $\R^2$.
\emph{Critical points:} $\nabla f = (4x^3 - 4y,\ 4y^3 - 4x) =
0$ gives $y = x^3$ and $x = y^3 = x^9$, so $x(x^8 - 1) = 0$:
the real solutions are $(0,0)$, $(1,1)$, $(-1,-1)$.
\emph{Hessians:} $H = \begin{pmatrix} 12x^2 & -4\\ -4 &
12y^2\end{pmatrix}$. At $(\pm1, \pm1)$: $\begin{pmatrix} 12 &
-4\\ -4 & 12\end{pmatrix}$, eigenvalues $8$ and $16$: positive
definite, strict local minima with $f = -2$. At $(0,0)$:
$\begin{pmatrix} 0 & -4\\ -4 & 0\end{pmatrix}$, eigenvalues
$\pm4$: a saddle. \emph{Globality:} from $2\abs{xy} \leq x^2 +
y^2$,
\[
f(x, y) \geq x^4 + y^4 - 2(x^2 + y^2)
= (x^2 - 1)^2 + (y^2 - 1)^2 + x^2 + y^2 - 2
\xrightarrow[\norm{(x,y)}\to\infty]{} +\infty :
\]
$f$ is coercive, so it attains a global minimum
(compactness of sublevel sets), necessarily at a critical
point: the value $-2$, at both $(1,1)$ and $(-1,-1)$, is the
global minimum; there is no maximum ($f$ unbounded above).
Closing insight: the local test classifies candidates, but
only a growth argument turns ``local'' into ``global'' --- the
two-step pattern of every optimization proof in this book.
\end{example}

\begin{method}[Classifying the extrema of $f \colon \R^n \to \R$]\label{met:b2:diffcalc:extrema}
\begin{enumerate}
  \item Solve $\nabla f = 0$ (all critical points; on a domain
        with boundary, treat the boundary separately as in
        \cref{exo:b2:diffcalc:7}).
  \item At each critical point, compute the Hessian and its
        eigenvalue signs --- in dimension $2$, just $\det H$
        and $\operatorname{tr} H$: $\det < 0$ saddle; $\det >
        0$ extremum, of the type given by the sign of the
        trace; $\det = 0$: the test is silent, study $f$
        along curves.
  \item For global statements, add a compactness or coercivity
        argument ($f \to +\infty$ at infinity, or a compact
        constraint set), then compare the critical values.
\end{enumerate}
\end{method}

\section{The inverse function theorem}

\begin{theorem}[Inverse function theorem]\label{thm:b2:diffcalc:inverse}
Let $f \colon U \to \R^n$ be $C^1$ and $a \in U$ with $\dd f_a$
\emph{invertible}. Then there are open neighborhoods $V \ni a$, $W
\ni f(a)$ such that $f \colon V \to W$ is a bijection with $C^1$
inverse, and\index{inverse function theorem}
\[
\dd (f^{-1})_{f(x)} = (\dd f_x)^{-1} \qquad (x \in V).
\]
\end{theorem}
\admitted

\begin{remark}[Why it is true: the fixed-point strategy]
Solving $f(x) = y$ near $a$ rewrites as the fixed-point equation
$x = x + \dd f_a^{-1}\bigl(y - f(x)\bigr) =: \Phi_y(x)$; the map
$\Phi_y$ has differential $\mathrm{id} - \dd f_a^{-1}\dd f_x$,
small near $a$ by continuity of $\dd f$, so $\Phi_y$ is a
contraction on a small closed ball and the Banach fixed point
theorem (\cref{thm:b2:metric:banach}) provides the unique local
solution $x = f^{-1}(y)$. Continuity and differentiability of the
inverse then follow from the estimates in the contraction. The full
bookkeeping is carried out in Year 3; the strategy --- and the
statement --- are used freely from now on. The companion
\emph{implicit function theorem} (solving $F(x, y) = 0$ for $y(x)$
when $\frac{\partial F}{\partial y}$ is invertible) follows by
applying the theorem to $(x, y) \mapsto (x, F(x,y))$.
\end{remark}

\begin{example}[Polar coordinates]\label{ex:b2:diffcalc:polar}
$\Phi(r, \theta) = (r\cos\theta, r\sin\theta)$ has Jacobian
\[
J_\Phi = \begin{pmatrix} \cos\theta & -r\sin\theta\\ \sin\theta &
r\cos\theta \end{pmatrix},
\qquad \det J_\Phi = r :
\]
invertible for $r \neq 0$, so $\Phi$ is a local $C^1$
diffeomorphism away from the origin --- the licence to ``change to
polar coordinates'', renewed for the multiple integrals of
\cref{ch:b2:multint}.
\end{example}

\begin{example}[Local everywhere, global nowhere]\label{ex:b2:diffcalc:localnotglobal}
Let $f(x, y) = \bigl(\eu^x\cos y,\ \eu^x\sin y\bigr)$ on
$\R^2$. Its Jacobian,
\[
J_f = \begin{pmatrix}
\eu^x\cos y & -\eu^x\sin y\\
\eu^x\sin y & \eu^x\cos y
\end{pmatrix},
\qquad
\det J_f = \eu^{2x} > 0 ,
\]
never vanishes: by \cref{thm:b2:diffcalc:inverse}, $f$ is a
local $C^1$ diffeomorphism at \emph{every} point of the plane.
Yet $f$ is far from injective: $f(x, y + 2\pi) = f(x, y)$, so
every value is taken infinitely often; and it is not surjective
either, since $\norm{f(x, y)} = \eu^x > 0$ misses the origin.
Closing insight: the inverse function theorem is irreducibly
\emph{local} --- invertibility of every $\dd f_a$ yields a
patchwork of local inverses that need not assemble into one.
(Readers who know complex numbers will recognize $z \mapsto
\eu^z$; the patchwork is the family of logarithm branches.)
Compare \cref{exo:b2:diffcalc:12}, where a quantitative global
hypothesis does force one global inverse.
\end{example}

\begin{remark}[Common pitfalls]
\emph{(i) Directional derivatives are cheap, differentials are
not:} all directional derivatives may exist --- and even fail
to depend linearly on the direction --- without
differentiability (\cref{exo:b2:diffcalc:2}); only the $C^1$
criterion
(\cref{thm:b2:diffcalc:c1}) upgrades partials to a
differential. \emph{(ii) Critical does not mean extremal:}
saddles (\cref{ex:b2:diffcalc:quarticrun}) and the silent
singular case (\cref{thm:b2:diffcalc:extrema}~(3)) both hide
behind $\nabla f = 0$. \emph{(iii) No vector-valued mean value
equality:} only the \emph{inequality} of
\cref{thm:b2:diffcalc:mvi} survives
(\cref{exo:b2:diffcalc:9}); never write $f(b) - f(a) = \dd
f_c(b-a)$ for $f$ with values in $\R^m$, $m \geq 2$.
\emph{(iv) Local invertibility is not injectivity:}
\cref{ex:b2:diffcalc:localnotglobal}. \emph{(v) The gradient
belongs to the inner product:} $\nabla f$ is the vector
representing $\dd f_a$ in a chosen inner product; change the
product (as in the weighted example of the quadratic-forms
chapter) and the gradient rotates, while the differential ---
the intrinsic object --- does not move.
\end{remark}

\begin{remark}[Where this is used]
Everything downstream of this chapter is differential calculus
applied: the differential-equations chapter linearizes flows and
uses Liouville's determinant formula (proved in this chapter's
weekend problem); the chapters on curves and surfaces study
level sets and parametrizations through the implicit function
theorem; multiple integrals change variables through Jacobians.
The weekend problem develops the calculus \emph{on the space of
matrices itself} --- differential of the determinant, of the
inverse, the matrix exponential, and the orthogonal group as a
smooth level set --- the Year 2 shadow of what the Year 3
volume formalizes as manifolds and Lie groups.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b2:diffcalc:1}
Compute the Jacobian matrices of $f(x,y) = (x^2 - y^2,\, 2xy)$ and
of $\Phi(r,\theta,z) = (r\cos\theta, r\sin\theta, z)$; where are
the differentials invertible?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:diffcalc:2}
Let $f(x,y) = \frac{x^3}{x^2 + y^2}$ ($f(0,0) = 0$). Prove that all
directional derivatives of $f$ at $0$ exist, but that $f$ is not
differentiable at $0$ \emph{(the map $v \mapsto$ directional
derivative is not linear)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:diffcalc:3}
Find and classify the critical points of $f(x, y) = x^3 + y^3 -
3xy$ using \cref{thm:b2:diffcalc:extrema}, and of $g(x,y) = x^4 +
y^4 - 2(x - y)^2$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:diffcalc:4}
Let $f \colon \R^n \to \R$ be $C^1$ and \emph{homogeneous of degree
$p$}: $f(tx) = t^pf(x)$ for $t > 0$. Prove Euler's identity
\[
\langle \nabla f(x), x\rangle = p\,f(x) ,
\]
and its converse for $C^1$ functions on $\R^n\setminus\{0\}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:diffcalc:5}
Let $A$ be symmetric and $f(x) = \frac12\langle Ax, x\rangle -
\langle b, x\rangle$. Compute $\nabla f$ and $H_f$; when is $f$
convex? Assuming $A$ positive definite, show that $f$ has a unique
global minimum at the solution of $Ax = b$ --- gradient descent's
raison d'\^etre.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:diffcalc:6}
(Lagrange multiplier, one constraint, proved by hand) Let $f, g$ be
$C^1$ on $\R^2$, and suppose $f$ attains, at $a$, a local extremum
on the level set $\{g = 0\}$, with $\nabla g(a) \neq 0$. Prove that
$\nabla f(a) = \lambda\nabla g(a)$ for some $\lambda$.
\emph{(Parametrize the level set near $a$ by the implicit function
theorem and differentiate $t \mapsto f(\gamma(t))$.)} Application:
extrema of $f(x,y) = xy$ on the circle $x^2 + y^2 = 1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:diffcalc:7}
Determine the extrema of $f(x, y) = x^2 + y^2 - xy + x - y$ on
$\R^2$, then its maximum and minimum on the closed triangle with
vertices $(0,0)$, $(1,0)$, $(0,1)$ \emph{(interior critical points,
then the three edges, then the vertices)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:diffcalc:8}
Let $f \colon \R^2 \to \R^2$, $f(x, y) = (x + y^2,\; y + x^2)$.
Show that $f$ is a local diffeomorphism near $0$, compute
$\dd(f^{-1})_{(0,0)}$, and find the largest $r$ such that $\dd f$
is invertible on the ball $\norm{(x,y)}_2 < r$ \emph{(compute
$\det J_f$)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:diffcalc:9}
(Rolle fails, mean value survives) Give $f \colon \R \to \R^2$,
$C^1$, with $f(0) = f(2\pi)$ but $f'(t) \neq 0$ for all $t$
(no vector-valued Rolle). Then verify on your example the
mean value \emph{inequality} of \cref{thm:b2:diffcalc:mvi}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:diffcalc:10}
Compute the differential and the gradient of $f(x) = \norm
x_2^2$ and of $g(x) = \langle Ax, x\rangle$ on $\R^n$ ($A$ a
square matrix, not assumed symmetric), and the Hessian of each.
For which $A$ is $g$ convex?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:diffcalc:11}
Let $f \colon \R^n \to \R$ be $C^2$. Prove that $f$ is convex if
and only if its Hessian $H_x$ is positive semidefinite at every
$x$ \emph{(reduce to one variable: $t \mapsto f(a + t(b-a))$;
use Taylor--Lagrange in one direction, and for the converse
evaluate $\varphi''$)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:diffcalc:12}
(A global inverse theorem) Let $g \colon \R^n \to \R^n$ be $C^1$
with $\vertiii{\dd g_x} \leq k < 1$ for all $x$, and $f =
\mathrm{id} + g$.
\begin{enumerate}
  \item Show $\norm{f(x) - f(y)} \geq (1 - k)\norm{x - y}$: $f$
        is injective, with continuous inverse on its image.
  \item Show that for each $y \in \R^n$ the map $x \mapsto y -
        g(x)$ is a contraction of the complete space $\R^n$,
        and conclude by the Banach fixed point theorem
        (\cref{thm:b2:metric:banach}) that $f$ is surjective.
  \item Conclude that $f$ is a bijection of $\R^n$ with
        $(1-k)^{-1}$-Lipschitz inverse --- a \emph{global}
        counterpart of \cref{thm:b2:diffcalc:inverse} (which,
        by contrast, is purely local).
\end{enumerate}
\end{exercise}

\section{Problem: The calculus of matrices --- Jacobi,
exponential, and the orthogonal group}

\begin{problem}\label{pb:b2:diffcalc:1}
The cleanest playground for differential calculus is the space
$\mathcal M_n(\R) \simeq \R^{n^2}$ itself: its most natural maps
--- product, inverse, determinant, exponential --- have
differentials of striking elegance. This problem computes them
all: the \emph{Neumann series}, the differential of the inverse,
\emph{Jacobi's formula}\index{Jacobi's formula} for the
determinant with \emph{Liouville's formula} as dividend, the
\emph{matrix exponential}\index{matrix exponential} with
$\det\eu^A = \eu^{\operatorname{tr}A}$, and finally the
orthogonal group $O_n$ as a smooth level set with the
antisymmetric matrices as tangent space --- differential
geometry in embryo. Throughout, $\vertiii\cdot$ is the operator
norm subordinate to $\norm\cdot_2$, and $\langle X, Y\rangle =
\operatorname{tr}(X^{\mathsf T}Y)$ the Frobenius inner product.

\medskip
\noindent\textbf{Part I --- The Neumann series.}
\begin{enumerate}
  \item Prove submultiplicativity, $\vertiii{AB} \leq
        \vertiii A\,\vertiii B$, and deduce that polynomial
        maps of $A$ (matrix products, determinant, trace) are
        continuous on $\mathcal M_n(\R)$.
  \item For $\vertiii X < 1$, show that $\sum_{k\geq0}X^k$
        converges absolutely in $\mathcal M_n(\R)$
        (\cref{thm:b2:nvs:absoluteconvergence}), that its sum
        is $(I - X)^{-1}$, and that
        \[
        \vertiii{(I - X)^{-1}} \leq \frac{1}{1 - \vertiii X},
        \qquad
        (I - X)^{-1} = I + X + O\bigl(\vertiii X^2\bigr) .
        \]
  \item Deduce that $GL_n(\R)$ is \emph{open}: if $A$ is
        invertible and $\vertiii H <
        \frac{1}{\vertiii{A^{-1}}}$, then $A + H$ is
        invertible. Deduce also that $GL_n(\R)$ is \emph{dense}
        in $\mathcal M_n(\R)$ \emph{(perturb $A$ by
        $\varepsilon I$: $\det(A + \varepsilon I)$ is a nonzero
        polynomial in $\varepsilon$)}.
  \item Show that the inversion map $\Phi(A) = A^{-1}$ is
        continuous on $GL_n(\R)$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- First differentials.}
\begin{enumerate}[resume]
  \item Show that the squaring map $A \mapsto A^2$ is
        differentiable with differential $H \mapsto AH + HA$,
        and more generally that $A \mapsto A^k$ has
        differential $H \mapsto \sum_{i=0}^{k-1}
        A^iHA^{k-1-i}$. Why can one \emph{not} write $kA^{k-1}H$
        in general?
  \item Prove that $\Phi(A) = A^{-1}$ is differentiable on
        $GL_n(\R)$ with
        \[
        \dd\Phi_A(H) = -A^{-1}HA^{-1}
        \]
        \emph{(write $(A + H)^{-1} = (I + A^{-1}H)^{-1}A^{-1}$
        and expand by question 2)}. Check the formula against
        the scalar case $n = 1$.
  \item For a $C^1$ curve $t \mapsto A(t) \in GL_n(\R)$, deduce
        $\bigl(A(t)^{-1}\bigr)' = -A^{-1}A'A^{-1}$, and expand
        $t \mapsto (I + tB)^{-1}$ to first order at $t = 0$.
  \item Compute the differential of $f(A) =
        \operatorname{tr}(A^k)$ and identify its gradient for
        the Frobenius inner product:
        \[
        \dd f_A(H) = k\operatorname{tr}\bigl(A^{k-1}H\bigr),
        \qquad
        \nabla f(A) = k\,\bigl(A^{k-1}\bigr)^{\mathsf T} .
        \]
  \item Same questions for $f(A) = \operatorname{tr}
        (A^{\mathsf T}A) = \norm A_F^2$: differential,
        gradient, and the (constant) Hessian; conclude that
        $\norm\cdot_F^2$ is strictly convex.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Jacobi's formula.}
\begin{enumerate}[resume]
  \item Prove
        \[
        \det(I + H) = 1 + \operatorname{tr}H +
        O\bigl(\vertiii H^2\bigr)
        \]
        \emph{(expand $\det(e_1 + h_1, \dots, e_n + h_n)$ by
        multilinearity in the columns: terms with at least two
        $h$-columns are $O(\vertiii H^2)$)}: $\dd(\det)_I =
        \operatorname{tr}$.
  \item For $A$ invertible, deduce
        \[
        \dd(\det)_A(H) = \det(A)\,
        \operatorname{tr}\bigl(A^{-1}H\bigr) .
        \]
  \item Show that for \emph{every} $A$ (invertible or not),
        $\frac{\partial\det}{\partial a_{ij}}(A) = C_{ij}$, the
        $(i,j)$ cofactor \emph{(Laplace expansion along row
        $i$)}, so that with the adjugate $\operatorname{adj}A =
        \operatorname{com}(A)^{\mathsf T}$:
        \[
        \dd(\det)_A(H) =
        \operatorname{tr}\bigl(\operatorname{adj}(A)\,H\bigr),
        \qquad
        \nabla(\det)(A) = \operatorname{com}(A) ,
        \]
        recovering question 11 when $A$ is invertible
        ($\operatorname{adj}A = \det(A)A^{-1}$). This is
        \emph{Jacobi's formula}: $\bigl(\det
        A(t)\bigr)' = \operatorname{tr}\bigl(
        \operatorname{adj}(A(t))\,A'(t)\bigr)$.
  \item (Liouville's formula) Let $A(t)$ be a $C^1$ curve of
        matrices satisfying the linear differential equation
        $A'(t) = M(t)A(t)$. Prove
        \[
        \bigl(\det A(t)\bigr)' =
        \operatorname{tr}\bigl(M(t)\bigr)\,\det A(t),
        \qquad\text{hence}\qquad
        \det A(t) = \det A(0)\,
        \exp\Bigl(\int_0^t\operatorname{tr}M\Bigr)
        \]
        \emph{(use $\operatorname{adj}(A)\,A = \det(A)I$ and
        the cyclic invariance of the trace)} --- the Wronskian
        identity that the differential-equations chapter will
        use constantly.
  \item Show that $SL_n(\R) = \{\det = 1\}$ is a smooth level
        set: at every $A \in SL_n(\R)$ the differential
        $\dd(\det)_A$ is a \emph{surjective} linear map onto
        $\R$ \emph{(evaluate it at $H = \frac1nA$)}.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The matrix exponential.}
\begin{enumerate}[resume]
  \item Show that $\eu^A = \sum_{k\geq0}\frac{A^k}{k!}$
        converges absolutely for every $A$, normally on every
        ball, with $\vertiii{\eu^A} \leq
        \eu^{\vertiii A}$; and that $\eu^A$ depends
        continuously on $A$.
  \item Prove that $AB = BA$ implies $\eu^{A+B} =
        \eu^A\eu^B$ \emph{(Cauchy product, legitimate by
        absolute convergence)}; deduce that $\eu^A$ is always
        invertible, with inverse $\eu^{-A}$: $\exp$ maps
        $\mathcal M_n(\R)$ into $GL_n(\R)$.
  \item Show that $t \mapsto \eu^{tA}$ is $C^1$ (indeed
        $C^\infty$) with
        \[
        \frac{\dd}{\dd t}\,\eu^{tA} = A\,\eu^{tA} =
        \eu^{tA}A
        \]
        \emph{(differentiate the series term by term on
        segments: the derived series converges normally)}.
  \item Prove the identity
        \[
        \det\bigl(\eu^{A}\bigr) = \eu^{\operatorname{tr}A}
        \]
        \emph{(apply Liouville's formula, question 13, to
        $A(t) = \eu^{tA}$)}. Sanity checks: $n = 1$; nilpotent
        $A$; and the trace-zero matrices land in
        $SL_n(\R)$.
  \item Show $\eu^H = I + H + O(\vertiii H^2)$, so $\exp$ is
        differentiable at $0$ with $\dd(\exp)_0 =
        \mathrm{id}$; conclude with the inverse function
        theorem (\cref{thm:b2:diffcalc:inverse}) that $\exp$
        is a $C^1$ diffeomorphism from a neighborhood of $0$
        onto a neighborhood of $I$: every matrix close to the
        identity has a logarithm.
  \item Show that $\exp$ maps symmetric matrices to symmetric
        \emph{positive definite} matrices, bijectively
        \emph{(diagonalize; the inverse is the spectral
        logarithm)}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The orthogonal group as a level
set.}
\begin{enumerate}[resume]
  \item Let $F(A) = A^{\mathsf T}A$, from $\mathcal M_n(\R)$ to
        the symmetric matrices $S_n$. Compute $\dd F_A(H) =
        A^{\mathsf T}H + H^{\mathsf T}A$ and show that at every
        $A \in O_n = F^{-1}(I)$ this differential is
        \emph{surjective} onto $S_n$ \emph{(given $S \in S_n$,
        try $H = \frac12 AS$)}: $O_n$ is a smooth level set,
        of dimension $n^2 - \frac{n(n+1)}2 = \frac{n(n-1)}2$.
  \item Show that every $C^1$ curve $A(t) \in O_n$ with $A(0) =
        I$ has an \emph{antisymmetric} velocity $A'(0)$, and
        conversely that for antisymmetric $K$ the curve
        $\eu^{tK}$ stays in $O_n$: the tangent space of $O_n$
        at $I$ is exactly the antisymmetric matrices.
  \item Show $\det\eu^{K} = 1$ for antisymmetric $K$ (question
        18): the exponential curve lives in the rotation group
        $SO_n$. Compute it fully for $n = 2$: with $J =
        \begin{pmatrix}0 & -1\\ 1 & 0\end{pmatrix}$, prove
        \[
        \eu^{\theta J} = \begin{pmatrix} \cos\theta &
        -\sin\theta\\ \sin\theta & \cos\theta\end{pmatrix} :
        \]
        the matrix exponential \emph{is} the rotation by
        $\theta$, and the series definitions of cosine and sine
        reappear inside a matrix.
  \item (The density trick) Using the density of $GL_n(\R)$
        (question 3) and continuity, extend from invertible to
        all matrices the identity
        \[
        \operatorname{adj}(AB) =
        \operatorname{adj}(B)\operatorname{adj}(A)
        \]
        \emph{(for invertible $A, B$ both sides equal
        $\det(AB)(AB)^{-1}$; both sides are polynomial in the
        entries)}.
  \item Synthesis. In one sentence each: (i) which earlier
        chapters provided the engine of each Part
        (completeness and normed algebras; the spectral
        theorem; the inverse function theorem); (ii) which
        formula of this problem the differential-equations
        chapter will lean on, and where; (iii) what
        $\dd(\det)_I = \operatorname{tr}$ and $\det\eu^A =
        \eu^{\operatorname{tr}A}$ say about trace and
        determinant as ``infinitesimal and global volume'';
        (iv) what the Year 3 volume makes of questions
        21--23 (Lie groups and their Lie algebras).
\end{enumerate}
\end{problem}
