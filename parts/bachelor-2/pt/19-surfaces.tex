\chapter{Superfícies}\label{ch:b2:surfaces}

Depois das curvas, superfícies: objetos de dois parâmetros em $\R^3$. O
cálculo \omterm{def:b2:diffcalc:differential}{diferencial} de \cref{ch:b2:diffcalc} fornece tudo
precisamos --- derivadas parciais fornecem vetores tangentes, a cruz
produto dá o normal, determinantes dar \omterm{def:b2:surfaces:area}{áreas}. Nós definimos
normal superfícies parametrizadas, deles planos tangentes, e o
\emph{\omterm{def:b2:surfaces:fff}{primeira forma fundamental}}, que codifica todos \omterm{def:b2:curves:length}{comprimento} e \omterm{def:b2:surfaces:area}{área}
medições na superfície. As superfícies também surgem como conjuntos de níveis
$ f(x, y, z) = c $; o gradiente então direciona o normal.

\section{Superfícies parametrizadas}

\begin{definition}[Superfície parametrizada regular]\label{def:b2:surfaces:param}
Deixar $ U \subseteq \R^2$ ser \omterm{def:b2:metric:topology}{abrir}. UM \emph{parametrizado
superfície}\index{parametrizado
superfície} de
aula $\mathcal{C}^k $ ($ k \geq 1$) é um mapa $\sigma \colon U \to
\R^3$, $(u, v) \mapsto \sigma(u, v)$, de classe $\mathcal{C}^k $. Um
ponto é \emph{regular} se os vetores derivados parciais
\[
\sigma_u(u, v) = \frac{\partial\sigma}{\partial u}(u,v),
\qquad
\sigma_v(u, v) = \frac{\partial\sigma}{\partial v}(u,v)
\]
são linearmente independentes, ou seja,\ $\sigma_u \wedge \sigma_v \neq 0$;
a superfície é \emph{regular} se cada ponto for.
\end{definition}

\begin{example}[As três descrições padrão]\label{ex:b2:surfaces:standard}
\leavevmode
\begin{enumerate}
\item \emph{Gráfico:} $\sigma(u, v) = (u,\ v,\ f(u, v))$ para $ f \in
\mathcal{C}^1(U)$. Sempre regular: $\sigma_u = (1, 0, f_u)$ e
$\sigma_v = (0, 1, f_v)$ são independentes.
\item \emph{Esfera (coordenadas esféricas):} para a esfera de
raio $ R $,
\[
\sigma(\theta, \varphi)
= (R\cos\theta\cos\varphi,\ R\sin\theta\cos\varphi,\
R\sin\varphi),
\qquad
(\theta, \varphi) \in \R \times
\bigl(-\tfrac\pi2, \tfrac\pi2\bigr),
\]
com $\theta $ a longitude e $\varphi $ a latitude. Um verifica
$\norm{\sigma_\theta \wedge \sigma_\varphi} = R^2\cos\varphi > 0$:
regular longe dos pólos (que este gráfico omite).
\item \emph{Conjunto de níveis:} $ S = \{(x,y,z) : f(x, y, z) = c\}$ onde
$ f $ é $\mathcal{C}^1$ e $\nabla f \neq 0$ sobre $ S $. Perto de cada
ponto, uma coordenada pode ser expressa como uma função da outra
dois pelo teorema da função implícita (\cref{ch:b2:diffcalc}), então
$ S $ é localmente um gráfico.
\end{enumerate}
\end{example}

\begin{example}[Do nível definido ao gráfico]
\label{ex:b2:surfaces:levelgraph}
O teorema da função implícita no item 3 merece uma explicação explícita
correr. Pegue a esfera $ x^2 + y^2 + z^2 = R^2$ perto de seu norte
poste $(0, 0, R)$: lá $\frac{\partial f}{\partial z} = 2z =
2R \neq 0$, e resolvendo para $ z $ dá o gráfico
\[
z = \sqrt{R^2 - x^2 - y^2},
\qquad x^2 + y^2 < R^2 ,
\]
regular em todos os lugares em seu (\omterm{def:b2:metric:topology}{abrir}) domínio --- incluindo o
pólo que o gráfico esférico perdeu. Perto de um ponto do equador
gosto $(R, 0, 0)$ o mesmo teorema resolve para $ x $ em vez de
($\frac{\partial f}{\partial x} = 2R \neq 0$). A regra de
polegar: uma superfície nivelada é um gráfico sobre o plano de coordenadas
\emph{ortogonal ao maior componente do gradiente},
e cobrindo a esfera com seis desses gráficos, um
verifica sua suavidade em todos os lugares sem trigonometria em
tudo.
\end{example}

\section{Plano tangente e normal}

\begin{definition}[Plano tangente]\label{def:b2:surfaces:tangent}
Deixar $\sigma $ seja regular em $(u_0, v_0)$, $ M_0 = \sigma(u_0, v_0)$.
O \emph{plano tangente}\index{plano tangente} $ T_{M_0}S $ o avião está passando $ M_0$
dirigido por $\operatorname{Vect}(\sigma_u, \sigma_v)$ (parciais em
$(u_0, v_0)$). O \emph{unidade normal}\index{unidade normal} é
\[
n(u_0, v_0)
= \frac{\sigma_u \wedge \sigma_v}
       {\norm{\sigma_u \wedge \sigma_v}} .
\]
\end{definition}

\begin{proposition}[Vetores tangentes são velocidade
vetores]\label{prop:b2:surfaces:velocity}
A direção de $ T_{M_0}S $ é exatamente o conjunto de vetores
$\gamma'(0)$, onde $\gamma = \sigma \circ c $ varia ao longo do
$\mathcal{C}^1$ curvas desenhadas na superfície através $ M_0$ (ou seja,\
$ c \colon (-\varepsilon, \varepsilon) \to U $ é $\mathcal{C}^1$
com $ c(0) = (u_0, v_0)$).
\end{proposition}

\begin{proof}
Se $ c(t) = (u(t), v(t))$, a regra da cadeia
(\cref{ch:b2:diffcalc}) dá
\[
\gamma'(0) = u'(0)\,\sigma_u + v'(0)\,\sigma_v
\in \operatorname{Vect}(\sigma_u, \sigma_v).
\]
Por outro lado, o vetor $ a\sigma_u + b\sigma_v $ é alcançado pelo
curva $ c(t) = (u_0 + at,\ v_0 + bt)$, que permanece no \omterm{def:b2:metric:topology}{abrir} definir
$ U $ para $\abs t $ pequeno.
\end{proof}

\begin{example}[O plano tangente do helicóide]
\label{ex:b2:surfaces:helicoidtangent}
Para o helicóide $\sigma(u, v) = (v\cos u,\ v\sin u,\ au)$,
no ponto $\sigma(0, 1) = (1, 0, 0)$:
\[
\sigma_u = (0,\ 1,\ a), \qquad \sigma_v = (1,\ 0,\ 0),
\qquad
\sigma_u \wedge \sigma_v = (0,\ a,\ -1),
\]
então o \omterm{def:b2:surfaces:tangent}{plano tangente} é $ a\,y = z $. Ele contém o todo
decisão horizontal $\{(t, 0, 0)\}$ (direção $\sigma_v $): como
para o cone de \cref{exo:b2:surfaces:1}, uma superfície governada por
linhas retas tem cada regra dentro do tangente
avião ao longo dele. A outra direção tangente $\sigma_u $ é o
velocidade da hélice $ u \mapsto \sigma(u, 1)$: um gráfico,
duas curvas desenhadas, e o todo \omterm{def:b2:surfaces:tangent}{plano tangente} é estendido ---
\cref{prop:b2:surfaces:velocity} em ação.
\end{example}

\begin{proposition}[Normal de uma superfície plana]\label{prop:b2:surfaces:gradient}
Deixar $ S = \{f = c\}$ com $ f $ de classe $\mathcal{C}^1$ e $\nabla
f(M_0) \neq 0$. Então o \omterm{def:b2:surfaces:tangent}{plano tangente} de $ S $ no $ M_0$ é o
avião através $ M_0$ ortogonal a $\nabla f(M_0)$:
\[
T_{M_0}S :\quad
\langle \nabla f(M_0),\ M - M_0\rangle = 0 .
\]
\end{proposition}

\begin{proof}
Para qualquer curva $\gamma $ desenhado em $ S $ através $ M_0$, $ f(\gamma(t)) =
c $ de forma idêntica, então a regra da cadeia dá $\langle
\nabla f(M_0), \gamma'(0)\rangle = 0$: todos os vetores de velocidade são
ortogonal ao gradiente, então a direção tangente está contida
no avião $\nabla f(M_0)^\perp $. Ambos são $2$-dimensional
subespaços --- a direção da tangente porque $ S $ é localmente um
gráfico regular (\cref{ex:b2:surfaces:standard}), o ortogonal
complementar porque $\nabla f(M_0) \neq 0$ --- portanto eles são
igual.
\end{proof}

\begin{example}\label{ex:b2:surfaces:spheretangent}
Para a esfera $ x^2 + y^2 + z^2 = R^2$: $\nabla f = 2(x, y, z)$,
então o \omterm{def:b2:surfaces:tangent}{plano tangente} no $ M_0$ é ortogonal ao raio
$\vect{OM_0}$ --- o fato clássico de que raio e \omterm{def:b2:surfaces:tangent}{plano tangente}
são perpendiculares, com equação $\langle M_0, M\rangle = R^2$.
\end{example}

\begin{example}[Plano tangente do gráfico]\label{ex:b2:surfaces:graphtangent}
Para $ z = f(x, y)$ no $(x_0, y_0)$: aplicando
\cref{prop:b2:surfaces:gradient} para $ F(x,y,z) = f(x,y) - z $,
\[
z = f(x_0, y_0)
+ f_x(x_0, y_0)(x - x_0)
+ f_y(x_0, y_0)(y - y_0) ,
\]
o \omterm{def:b2:affine:subspace}{afim} parte da expansão de Taylor de primeira ordem --- o
\omterm{def:b2:surfaces:tangent}{plano tangente} é o gráfico do \omterm{def:b2:diffcalc:differential}{diferencial}, como deve ser.
\end{example}

\begin{example}[O ponto mais próximo de um
superfície]\label{ex:b2:surfaces:closest}
Qual ponto do parabolóide $ z = x^2 + y^2$ está mais próximo de
$ P = (0, 0, 1)$? Minimize a distância quadrada ao longo do
superfície: com $\rho^2 = x^2 + y^2$,
\[
g(\rho^2) = \rho^2 + (\rho^2 - 1)^2,
\qquad
g'(\rho^2) = 1 + 2(\rho^2 - 1) = 0
\iff \rho^2 = \tfrac12 ,
\]
dando o círculo de pontos na altura $ z = \frac12$ e
distância $\sqrt{\tfrac12 + \tfrac14} = \frac{\sqrt3}2$. O
assinatura geométrica da minimalidade: nesse ponto $ M $, o
vetor $\vect{MP}$ deve ser \emph{normal} para a superfície ---
caso contrário, deslizando ao longo de uma curva desenhada com velocidade tendo um
componente em direção $ P $ diminuiria a distância. Verificar:
$\nabla(z - x^2 - y^2) = (-2x, -2y, 1)$ no $ M = (x, y,
\tfrac12)$, enquanto $\vect{MP} = (-x, -y, \tfrac12) =
\tfrac12(-2x, -2y, 1)$: paralelo, conforme previsto. O
condição de primeira ordem ``pé da perpendicular'' é o
mesmo que irá conduzir extremos em conjuntos de níveis em
\cref{exo:b2:surfaces:12}.
\end{example}

\begin{example}[Planos tangentes das quádricas: o
regra de polarização]\label{ex:b2:surfaces:polarization}
Deixar $ S : xy + yz + zx = 1$ e $ M_0 = (1, 1, 0) \in S $. Aqui
$\nabla f = (y + z,\ x + z,\ x + y)$, então $\nabla f(M_0) = (1,
1, 2)$ e o \omterm{def:b2:surfaces:tangent}{plano tangente} é
\[
(x - 1) + (y - 1) + 2z = 0,
\qquad\text{i.e.}\qquad x + y + 2z = 2 .
\]
A mesma resposta vem do \emph{\omterm{def:b2:quadratic:def}{polarização} regra} isso
generaliza \cref{ex:b2:surfaces:spheretangent} e
\cref{exo:b2:surfaces:2}: no quádricaequação de, substitua
$ x^2$ por $ x_0x $ e cada produto $ xy $ por $\frac{x_0y +
y_0x}2$ (e ciclicamente):
\[
\frac{x_0y + y_0x}2 + \frac{y_0z + z_0y}2 + \frac{z_0x +
x_0z}2 = 1
\;\xrightarrow{\ M_0 = (1,1,0)\ }\;
\frac{x + y}2 + \frac z2 + \frac z2 = 1 ,
\]
o que é $ x + y + 2z = 2$ de novo. A regra funciona porque
$\nabla $ de um forma quadrática é a forma bilinear associada
avaliado em relação ao ponto base --- tangência a um \omterm{pb:b2:surfaces:1}{quádrica} é
\omterm{def:b2:quadratic:def}{polarização}, mais uma cara \cref{ch:b2:quadratic}.
\end{example}

\section{A primeira forma fundamental}

\begin{definition}[Primeira forma fundamental]\label{def:b2:surfaces:fff}
Deixar $\sigma \colon U \to \R^3$ seja regular $\mathcal{C}^1$
superfície. Isso é \emph{primeira forma fundamental}\index{primeira forma fundamental} no $(u,v)$ é o
definido positivo forma quadrática sobre $\R^2$
\[
I(h, k) = \norm{h\,\sigma_u + k\,\sigma_v}^2
= E\,h^2 + 2F\,hk + G\,k^2,
\]
onde
\[
E = \norm{\sigma_u}^2,
\qquad
F = \langle \sigma_u, \sigma_v\rangle,
\qquad
G = \norm{\sigma_v}^2 .
\]
\end{definition}

\begin{remark}
$ I $ é a restrição do produto escalar euclidiano ambiente para
o \omterm{def:b2:surfaces:tangent}{plano tangente}, leia na base $(\sigma_u, \sigma_v)$: é
definido positivo precisamente porque $\sigma_u, \sigma_v $ são
independente (\cref{ch:b2:quadratic}). Cada quantidade métrica no
superfície --- \omterm{def:b2:curves:length}{comprimentos} de curvas desenhadas, ângulos entre elas, \omterm{def:b2:surfaces:area}{áreas} ---
é calculado a partir $ E, F, G $ sozinho. Duas superfícies com o mesmo $ E,
F, G $ em parâmetros adequados são \emph{isométrico} mesmo que eles se sentem
diferentemente no espaço: este é o ponto de partida da intrínseca
geometria.
\end{remark}

\begin{example}[Ângulos entre coordenadas
curvas]\label{ex:b2:surfaces:coordangle}
O \omterm{def:b2:surfaces:fff}{primeira forma fundamental} também mede ângulos: o
curvas coordenadas $ u \mapsto \sigma(u, v_0)$ e $ v \mapsto
\sigma(u_0, v)$ encontro no ângulo $\theta $ com
\[
\cos\theta = \frac{\langle\sigma_u,
\sigma_v\rangle}{\norm{\sigma_u}\,\norm{\sigma_v}}
= \frac{F}{\sqrt{EG}} :
\]
o coeficiente único $ F $ decide a ortogonalidade do
rede de parâmetros. Para o gráfico de esfera e o helicóide, $ F =
0$: os meridianos cortam paralelos e as hélices cortam a horizontal
decisões, em ângulos retos --- é por isso que seus \omterm{def:b2:surfaces:area}{área}
integrandos colapsaram para $\sqrt{EG}$. Para um gráfico,
$ F = f_xf_y $ desaparece apenas onde uma derivada parcial desaparece:
a rede de coordenadas de um gráfico inclinado é \emph{não}
ortogonal, embora $(x, y)$-net lá embaixo é. Quando
cálculos em uma superfície parecem pesados, o primeiro passo é
procure um gráfico com $ F = 0$.
\end{example}

\begin{example}[O gráfico de sela]\label{ex:b2:surfaces:saddlechart}
Para a sela $ z = xy $ com gráfico $\sigma(u, v) = (u, v,
uv)$:
\[
\sigma_u = (1, 0, v), \qquad \sigma_v = (0, 1, u),
\qquad
E = 1 + v^2, \quad F = uv, \quad G = 1 + u^2 ,
\]
e $ EG - F^2 = 1 + u^2 + v^2 > 0$: regular em todos os lugares. O
duas curvas coordenadas através de um ponto são \emph{direto
linhas} de $\R^3$ (consertar $ u $ ou consertar $ v $: as decisões do
sela duplamente regrada), ainda $ F \neq 0$ fora dos eixos: decisões
através de um ponto genérico não são ortogonais. Ambas as decisões mentem
no \omterm{def:b2:surfaces:tangent}{plano tangente}, que eles abrangem --- então o tangente
avião corta a superfície ao longo de duas linhas inteiras, o extremo
oposto da esfera, cujo planos tangentes toque em um
apenas ponto. O sinal do “contato de segunda ordem” entre
uma superfície e seu planos tangentes é um \omterm{thm:b2:curves:frenet2d}{curvatura} história,
retomado no volume do Ano 3.
\end{example}

\begin{proposition}[Comprimento de uma curva desenhada em um
superfície]\label{prop:b2:surfaces:curvelength}
Se $\gamma(t) = \sigma(u(t), v(t))$, $ t \in [a, b]$, é
$\mathcal{C}^1$, então
\[
L(\gamma) = \int_a^b
\sqrt{E\,u'^2 + 2F\,u'v' + G\,v'^2}\;\dd t ,
\]
com $ E, F, G $ avaliado em $(u(t), v(t))$.
\end{proposition}

\begin{proof}
$\gamma' = u'\sigma_u + v'\sigma_v $ pela regra da cadeia, então
$\norm{\gamma'}^2 = I(u', v') = Eu'^2 + 2Fu'v' + Gv'^2$; integrar
$\norm{\gamma'}$ (\cref{def:b2:curves:length}).
\end{proof}

\begin{example}[Por que os aviões sobrevoam o pólo]\label{ex:b2:surfaces:polarroute}
Dois aeroportos ficam na latitude $\varphi_0$ e oposto
longitudes: $ A = \sigma(0, \varphi_0)$ e $ B = \sigma(\pi,
\varphi_0)$ na esfera do raio $ R $. Ao longo do paralelo
($\varphi \equiv \varphi_0$), o \omterm{def:b2:curves:length}{comprimento} é
$\int_0^\pi R\cos\varphi_0\,\dd\theta = \pi R\cos\varphi_0$.
Ao longo da rota sobre o pólo (subindo o meridiano $\theta = 0$,
abaixo do meridiano $\theta = \pi $), isso é $2R(\frac\pi2 -
\varphi_0)$. Na latitude $\varphi_0 = \frac\pi3$ (sessenta
graus): rota paralela $\pi R/2 \approx 1.571\,R $, \omterm{pb:b2:quadratic:1}{polar}
rota $\pi R/3 \approx 1.047\,R $ --- um terço mais curto. Na verdade
$\pi\cos\varphi_0 \geq \pi - 2\varphi_0$ sobre
$\intcc0{\pi/2}$ (a função $\pi\cos\varphi - \pi +
2\varphi $ desaparece em ambas as extremidades e sua derivada $2 -
\pi\sin\varphi $ muda de sinal uma vez, então primeiro é crescente
então decrescente, portanto não negativo): a rota \omterm{pb:b2:quadratic:1}{polar} nunca
perde. O \omterm{def:b2:surfaces:fff}{primeira forma fundamental} virou uma questão de navegação
em duas integrais de uma linha; \cref{exo:b2:surfaces:6} empurra
a ideia de uma prova genuína de minimalidade para meridianos.
\end{example}

\begin{lemma}[Identidade Lagrange]\label{lem:b2:surfaces:lagrange}
Para todos $ a, b \in \R^3$:
$\norm{a \wedge b}^2 = \norm a^2 \norm b^2 - \langle a, b\rangle^2$.
Em particular
\[
\norm{\sigma_u \wedge \sigma_v} = \sqrt{EG - F^2}.
\]
\end{lemma}

\begin{proof}
Ambos os lados permanecem inalterados se substituirmos $ b $ por seu componente $ b_\perp
= b - \frac{\langle a, b\rangle}{\norm a^2}a $ ortogonal a $ a $
(para $ a \neq 0$; o caso $ a = 0$ é trivial): o lado esquerdo
porque $ a \wedge a = 0$, o lado direito expandindo $\norm
{b_\perp}^2 = \norm b^2 - \frac{\langle a,b\rangle^2}{\norm a^2}$
e $\langle a, b_\perp\rangle = 0$. Então basta provar a
identidade para ortogonal $ a, b $, onde se lê $\norm{a \wedge
b} = \norm a \norm b $: verdadeiro, já que para vetores ortogonais o
produto cruzado tem \omterm{def:b2:nvs:norm}{norma} $\norm a\norm b\,\abs{\sin\frac\pi2}$. O
fórmula exibida é o caso $ a = \sigma_u $, $ b = \sigma_v $.
\end{proof}

\begin{remark}
A identidade Lagrange diz $ EG - F^2 = \det\begin{psmallmatrix}
E & F\\ F & G\end{psmallmatrix}$ é o \emph{Grama
determinante} de $(\sigma_u, \sigma_v)$: o quadrado \omterm{def:b2:surfaces:area}{área} de
o paralelogramo que eles abrangem. Regularidade, positiva
certeza do primeira forma fundamentale positividade de
o grama determinante são três frases de uma condição ---
é por isso que \omterm{def:b2:surfaces:area}{área} o integrando abaixo nunca desaparece em um
gráfico normal.
\end{remark}

\begin{definition}[Área]\label{def:b2:surfaces:area}
Deixar $\sigma \colon U \to \R^3$ ser um injetivo regular
$\mathcal{C}^1$ superfície e $ K \subseteq U $ um \omterm{def:b2:metric:compact}{compactar} domínio em
quais integrais duplas fazem sentido (\cref{ch:b2:multint}). O
\emph{área}\index{área} da peça $\sigma(K)$ é
\[
\mathcal{A}
= \iint_K \norm{\sigma_u \wedge \sigma_v}\,\dd u\,\dd v
= \iint_K \sqrt{EG - F^2}\;\dd u\,\dd v .
\]
\end{definition}

\begin{remark}[Por que esta fórmula]
O retângulo $[u, u + \dd u] \times [v, v + \dd v]$ está mapeado, para
primeira ordem, no paralelogramo medido por $\sigma_u\,\dd u $
e $\sigma_v\,\dd v $, cujo \omterm{def:b2:surfaces:area}{área} é $\norm{\sigma_u \wedge
\sigma_v}\,\dd u\,\dd v $: a definição integra o local
fator de distorção de \omterm{def:b2:surfaces:area}{área}, exatamente como \omterm{def:b2:curves:length}{comprimento} do arco integra o local
velocidade. Consistência com \omterm{def:b2:curves:reparam}{mudança de parâmetros} é
\cref{exo:b2:surfaces:8}; consistência com a mudança de variáveis
fórmula para integrais duplas é discutida em
\cref{ch:b2:multint}.
\end{remark}

\begin{example}[Dois gráficos diferentes, um
área]\label{ex:b2:surfaces:samearea}
Sobre o disco unitário, compare a tigela $ z = \frac12(x^2 + y^2)$
e a sela $ z = xy $. Deles \omterm{def:b2:surfaces:area}{área} integrandos
(\cref{exo:b2:surfaces:5}) são
\[
\sqrt{1 + x^2 + y^2}
\qquad\text{and}\qquad
\sqrt{1 + y^2 + x^2} :
\]
idêntico. As duas superfícies --- uma curvando-se da mesma maneira em
todas as direções, as outras em forma de sela --- têm exatamente
igual \omterm{def:b2:surfaces:area}{áreas} em todos os domínios, $\frac{2\pi}3(2\sqrt2 - 1)$
sobre o disco da unidade. O \omterm{def:b2:surfaces:area}{área} elemento só vê o
\emph{\omterm{def:b2:curves:length}{comprimento}} do gradiente, não da disposição do
flexão; tirar a tigela da sela requer
dados de segunda ordem (a estrutura do sinal exibida em
\cref{fig:b2:surfaces:saddle}), que nenhuma quantidade de \omterm{def:b2:surfaces:area}{área}
medição detecta. \omterm{def:b2:surfaces:fff}{Primeira forma fundamental}: métrico, cego
moldar; a segunda forma de visão de forma pertence ao Ano 3.
\end{example}

\begin{example}[Área da esfera]\label{ex:b2:surfaces:spherearea}
Para a carta esférica de \cref{ex:b2:surfaces:standard}:
\[
\sigma_\theta = R(-\sin\theta\cos\varphi,\
\cos\theta\cos\varphi,\ 0),
\qquad
\sigma_\varphi = R(-\cos\theta\sin\varphi,\
-\sin\theta\sin\varphi,\ \cos\varphi),
\]
então $ E = R^2\cos^2\varphi $, $ F = 0$, $ G = R^2$ e $\sqrt{EG - F^2}
= R^2\cos\varphi $. Por isso
\[
\mathcal{A}
= \int_{-\pi/2}^{\pi/2}\!\!\int_0^{2\pi}
R^2\cos\varphi\;\dd\theta\,\dd\varphi
= 2\pi R^2\,\bigl[\sin\varphi\bigr]_{-\pi/2}^{\pi/2}
= \boxed{4\pi R^2} .
\]
\end{example}

\begin{example}[O cone, verificado contra o
fórmula escolar]\label{ex:b2:surfaces:conearea}
Para o cone $ z = \sqrt{x^2 + y^2}$ sobre o anel $ a \leq
\rho \leq b $, a fórmula gráfica de
\cref{exo:b2:surfaces:5} dá $1 + f_x^2 + f_y^2 = 2$
(calcular $ f_x = x/\rho $, $ f_y = y/\rho $), então
\[
\mathcal A = \sqrt2\,\pi\,(b^2 - a^2) .
\]
Verificação de consistência com a fórmula da altura inclinada $\pi\rho\ell $
de \cref{ex:b2:surfaces:revolution}: os cones \omterm{def:b2:metric:complete}{completos} da base
raios $ b $ e $ a $ tem lateral \omterm{def:b2:surfaces:area}{áreas} $\pi b\cdot b\sqrt2$ e
$\pi a\cdot a\sqrt2$, cuja diferença é exatamente
$\sqrt2\pi(b^2 - a^2)$. Dois gráficos, duas fórmulas, um \omterm{def:b2:surfaces:area}{área}
--- a invariância provada em \cref{exo:b2:surfaces:8}, visto
na natureza.
\end{example}

\begin{remark}[Verificações de sanidade para áreas]
Três verificações instantâneas detectam a maioria dos erros em um \omterm{def:b2:surfaces:area}{área}
computação. \emph{Dimensionamento}: dilatando uma superfície por $\lambda $
multiplica $ E, F, G $ por $\lambda^2$ e o \omterm{def:b2:surfaces:area}{área} por
$\lambda^2$ --- uma resposta cuja dependência de $ R $ não é
quadrático (como $4\pi R^2$) está errado. \emph{Positividade do
elemento}: $\sqrt{EG - F^2}$ deve ser estritamente positivo no
interior do gráfico; um valor que desaparece sinaliza uma degeneração do gráfico,
a ser extirpado como nos pólos da esfera. \emph{Simetria}: um
cálculo sobre um \omterm{def:b2:quadratic:adjoint}{simétrico} peça deve ser consistente com
somando suas partes congruentes --- o hemisfério teria melhor
dar $2\pi R^2$.
\end{remark}

\begin{example}[Superfície de revolução]\label{ex:b2:surfaces:revolution}
Girar a curva $ z \mapsto (r(z), 0, z)$, $ r > 0$ de classe
$\mathcal{C}^1$, ao redor $ z $-eixo:
\[
\sigma(\theta, z) = (r(z)\cos\theta,\ r(z)\sin\theta,\ z).
\]
Então $ E = r(z)^2$, $ F = 0$, $ G = 1 + r'(z)^2$, então
\[
\mathcal{A} = \int_a^b\!\!\int_0^{2\pi}
r(z)\sqrt{1 + r'(z)^2}\;\dd\theta\,\dd z
= 2\pi\int_a^b r(z)\sqrt{1 + r'(z)^2}\,\dd z ,
\]
a fórmula clássica (circunferência $2\pi r $ vezes a inclinação
\omterm{def:b2:curves:length}{comprimento} elemento). Para o cone $ r(z) = kz $, $ z \in [0, h]$: $\mathcal
A = 2\pi k\sqrt{1 + k^2}\,\frac{h^2}{2} = \pi \rho \ell $ com
$\rho = kh $ o raio da base e $\ell = h\sqrt{1 + k^2}$ a inclinação
altura --- a fórmula escolar, agora derivada em vez de admitida.
\end{example}

\begin{example}[O catenóide]\label{ex:b2:surfaces:catenoid}
Gire a catenária $ r(z) = \cosh z $, $ z \in \intcc{-1}{1}$,
em torno de seu eixo: o resultado \emph{catenóide} tem, pelo
fórmula de revolução e $1 + \sinh^2 z = \cosh^2 z $,
\[
\mathcal A = 2\pi\int_{-1}^1\cosh z\,\sqrt{1 + \sinh^2z}\,
\dd z = 2\pi\int_{-1}^1\cosh^2z\,\dd z
= \pi\bigl[z + \sinh z\cosh z\bigr]_{-1}^{1},
\]
aquilo é
\[
\mathcal A = 2\pi + \pi\sinh 2 \approx 17.68 .
\]
O fator de inclinação $\sqrt{1 + r'^2}$ mesclado com o raio
em um quadrado perfeito --- a mesma identidade que fez o
catenária \omterm{def:b2:curves:length}{comprimento} do arco elementar no capítulo sobre curvas. Isto
não é um acidente de álgebra: entre todas as superfícies de revolução
abrangendo os dois círculos limites, o catenóide minimiza
\omterm{def:b2:surfaces:area}{área} (é o formato de uma película de sabão entre dois anéis), e
esta propriedade variacional é precisamente o que destaca
$\cosh $; o volume do Ano 3 prova isso com o cálculo de
variações.
\end{example}

\begin{remark}[Armadilhas comuns]
(eu) \emph{Singularidades gráficas não são singularidades superficiais}:
o gráfico esférico degenera nos pólos
($\cos\varphi = 0$), mas a esfera é perfeitamente lisa ali
--- outro gráfico (troca as funções dos eixos) é regular
nos pólos. Antes de declarar um ponto singular, tente uma segunda
parametrização. (ii) \emph{Regularidade de $\sigma $ preocupações
a parametrização, não a imagem}: $\sigma(u, v) = (u^3, v,
0)$ não é regular ao longo $ u = 0$ embora a sua imagem seja uma
avião. (iii) O \omterm{def:b2:surfaces:area}{área} fórmula requer $\sigma $
\emph{injetivo} sobre $ K $: um gráfico cobrindo uma peça duas vezes
conta duas vezes ($\theta $ atropelando $\intcc0{4\pi}$
duplica a esfera \omterm{def:b2:surfaces:area}{área}). (iv) O \omterm{def:b2:surfaces:tangent}{unidade normal} está definido
até assinar na superfície, mas é \emph{escolhido} pelo gráfico
(ordem de $ u, v $); declarações envolvendo orientação devem corrigir
essa escolha. (v) Finalmente, $ EG - F^2 > 0$ não é um extra
hipótese: é exatamente regularidade, pelo Lagrange
identidade --- se desaparecer em algum lugar, o problema é o
gráfico e não \omterm{def:b2:surfaces:area}{área} ou a fórmula do \omterm{def:b2:surfaces:tangent}{plano tangente} se aplica lá.
\end{remark}

\begin{remark}[Perspectivas dentro deste volume]
Encaminhe links daqui. O \omterm{def:b2:surfaces:area}{área} elemento $\sqrt{EG -
F^2}\,\dd u\,\dd v $ é um Jacobiano bidimensional em
disfarce, e \cref{ch:b2:multint} torna a analogia exata
com o teorema da mudança de variáveis --- a superfície
integrais existem neste capítulo \omterm{def:b2:surfaces:area}{áreas} com um integrando em
placa. O \omterm{def:b2:surfaces:fff}{primeira forma fundamental} é um campo de positivo
formas quadráticas, manipulado \omterm{def:b2:funcseq:def}{pontualmente} pelas ferramentas de
\cref{ch:b2:quadratic}, cujo teorema espectral também alimenta
a classificação do fim de semana deste capítulo de \omterm{pb:b2:surfaces:1}{quádricas}. E o
linha normal gera problemas extremos em conjuntos de restrições
(\cref{ex:b2:surfaces:closest}), o germe geométrico do
Método do multiplicador de Lagrange esboçado com
\cref{thm:b2:diffcalc:extrema}.
\end{remark}

\begin{remark}[Onde isso é usado]
O \omterm{def:b2:surfaces:area}{área} elemento $\norm{\sigma_u\wedge\sigma_v}\,\dd u\,\dd v $
é a medida integral de superfície de \cref{ch:b2:multint}, onde
atende à fórmula de Green; o \omterm{def:b2:surfaces:fff}{primeira forma fundamental} é o
protótipo de um campo de formas quadráticas, estudado \omterm{def:b2:funcseq:def}{pontualmente}
com as ferramentas de \cref{ch:b2:quadratic}; e este capítulo
problema de fim de semana classifica tudo \omterm{pb:b2:surfaces:1}{quádrica} superfícies com o
teorema espectral. O volume do Ano 3 retorna às superfícies com
\omterm{def:b2:diffcalc:differential}{diferencial} formas e o teorema da divergência, e intrínseco
\omterm{thm:b2:curves:frenet2d}{curvatura} --- o que $ E, F, G $ saber sobre flexão --- é o
porta de entrada para a geometria \omterm{def:b2:diffcalc:differential}{diferencial} propriamente dita.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b2:surfaces:1}
Mostre que o planos tangentes do cone $ z = \sqrt{x^2 + y^2}$
(menos seu ápice) todos passam pelo ápice. \emph{(Parametrizar por
$\sigma(\theta, r) = (r\cos\theta, r\sin\theta, r)$, $ r > 0$.)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:surfaces:2}
Encontre o \omterm{def:b2:surfaces:tangent}{plano tangente} do elipsóide $\frac{x^2}{a^2} +
\frac{y^2}{b^2} + \frac{z^2}{c^2} = 1$ em um ponto $(x_0, y_0,
z_0)$ da superfície.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:surfaces:3}
Calcular $ E, F, G $ para o helicóide $\sigma(u, v) = (v\cos u,\
v\sin u,\ au)$, $ a > 0$, e o \omterm{def:b2:surfaces:area}{área} da peça $0 \leq u \leq
2\pi $, $0 \leq v \leq 1$, como uma integral (avalie-a usando
$\int\sqrt{v^2 + a^2}\,\dd v = \frac12\bigl(v\sqrt{v^2+a^2} +
a^2\ln(v + \sqrt{v^2 + a^2})\bigr) + C $).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:surfaces:4}
(Toro) Parametrize o toro obtido girando o círculo de
centro $(R, 0, 0)$ e raio $ r < R $ no $ xz $-avião ao redor do
$ z $-eixo:
\[
\sigma(\theta, \psi) = \bigl((R + r\cos\psi)\cos\theta,\
(R + r\cos\psi)\sin\theta,\ r\sin\psi\bigr).
\]
Calcular $ E, F, G $, verifique a regularidade e mostre que \omterm{def:b2:surfaces:area}{área} é
$4\pi^2 R r $ (Pappus: circunferência média $2\pi R $ círculo de tempos
\omterm{def:b2:curves:length}{comprimento} $2\pi r $).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:surfaces:5}
Mostre que o \omterm{def:b2:surfaces:area}{área} do gráfico de $ f \in \mathcal{C}^1(K)$ é
$\iint_K \sqrt{1 + f_x^2 + f_y^2}\,\dd x\,\dd y $, e calcule-o
para a peça parabolóide $ z = \frac12(x^2 + y^2)$ sobre o disco
$ x^2 + y^2 \leq 1$ (coordenadas \omterm{pb:b2:quadratic:1}{polares},
\cref{ch:b2:multint}).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:surfaces:6}
Uma curva desenhada $\gamma(t) = \sigma(u(t), v(t))$ na esfera de
raio $ R $ (gráfico esférico) tem $ u = \theta(t)$, $ v =
\varphi(t)$. Escreva seu \omterm{def:b2:curves:length}{comprimento} como integral em $\theta, \varphi $
e provar que entre curvas que unem dois pontos do mesmo
meridiano $\theta = \theta_0$, o arco do meridiano é o mais curto.
\emph{(Limite o integrando abaixo por $ R\abs{\varphi'}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:surfaces:7}
(Linhas normais de uma esfera) Deixe $ S $ ser uma superfície nivelada regular
$\{f = c\}$, \omterm{def:b2:metric:connected}{conectado}, cujas linhas normais passam por um
ponto fixo $\Omega $. Mostre isso $ S $ está contido em uma esfera
centrado em $\Omega $. \emph{(Mostre isso $\norm{M - \Omega}^2$ tem
derivada zero ao longo de cada curva desenhada em $ S $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:surfaces:8}
(\omterm{def:b2:surfaces:area}{Área} é geométrico) Deixe $\Phi \colon U' \to U $ seja um $\mathcal{C}^1$
difeomorfismo entre \omterm{def:b2:metric:topology}{abrir} conjuntos de $\R^2$ e $\tilde\sigma =
\sigma \circ \Phi $. Mostre isso
\[
\norm{\tilde\sigma_{u'} \wedge \tilde\sigma_{v'}}
= \abs{\det J_\Phi}\,
\norm{(\sigma_u \wedge \sigma_v)\circ\Phi} ,
\]
e deduzir, usando a fórmula de mudança de variáveis de
\cref{ch:b2:multint}, que o \omterm{def:b2:surfaces:area}{área} de
\cref{def:b2:surfaces:area} não depende do regular escolhido
parametrização.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:surfaces:9}
(Teorema da caixa de chapéu de Arquimedes) Na esfera do raio $ R $, o
\emph{zona} entre as latitudes com $ z_1 \leq z \leq z_2$
($-R \leq z_1 < z_2 \leq R $) tem \omterm{def:b2:surfaces:area}{área} $2\pi R\,(z_2 - z_1)$:
prove isso com a carta esférica e conclua que a zona
\omterm{def:b2:surfaces:area}{área} depende apenas de sua altura --- fatiar uma laranja em
fatias de espessura igual fornecem quantidades iguais de casca.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:surfaces:10}
Mostre que toda reta normal de uma superfície de revolução
$\sigma(\theta, z) = (r(z)\cos\theta,\ r(z)\sin\theta,\ z)$
($ r > 0$ de classe $\mathcal C^1$) encontra o eixo de
revolução e localize o ponto de intersecção.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:surfaces:11}
(Desenrolando o cilindro) O gráfico $\sigma(u, v) = (\cos u,\
\sin u,\ v)$ do cilindro unitário tem $ E = G = 1$, $ F = 0$:
verifique isso e explique por que cada curva desenhada $ t \mapsto
\sigma(u(t), v(t))$ tem o mesmo \omterm{def:b2:curves:length}{comprimento} como a curva plana $ t
\mapsto (u(t), v(t))$. Deduza que a hélice de $(1, 0, 0)$
para $(1, 0, 2\pi c)$ fazer uma volta tem \omterm{def:b2:curves:length}{comprimento}
$2\pi\sqrt{1 + c^2}$, e que nenhuma curva desenhada com o mesmo
pontos finais e uma volta completa é mais curta.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:surfaces:12}
Deixar $ S = \{f = c\}$ seja um \omterm{def:b2:metric:compact}{compactar} superfície nivelada regular e
$ M_0 \in S $ um ponto a uma distância máxima da origem. Mostrar
isso $\nabla f(M_0)$ é colinear com $\vect{OM_0}$ --- o
normal no ponto mais distante é radial. Aplicar ao elipsóide
$\frac{x^2}{a^2} + \frac{y^2}{b^2} + \frac{z^2}{c^2} = 1$ ($ a >
b > c > 0$): encontre todos os pontos onde a normal é radial, e
identificar os mais distantes.
\end{exercise}

\section{Problema: a classificação das quádricas de
$\R^3$}

\begin{problem}[{Problema de fim de semana --- cada superfície quádrica,
classificado pelo teorema espectral}]\label{pb:b2:surfaces:1}
UM \emph{quádrica}\index{quádrica} é o zero definido em $\R^3$ de um
polinômio de grau dois
\[
q(X) = X^{\mathsf T}\!AX + 2\,\langle b, X\rangle + c,
\qquad A \in \mathcal S_3(\R),\ A \neq 0,\ b \in \R^3,\
c \in \R .
\]
As superfícies das figuras deste capítulo --- esferas,
elipsóides, selas, cones, cilindros --- são todos \omterm{pb:b2:surfaces:1}{quádricas}.
Este problema os classifica \emph{todos}: o teorema espectral
(\cref{thm:b2:quadratic:spectral}) endireita o quadrático
parte, \omterm{def:b2:affine:subspace}{afim} traduções (\cref{ch:b2:affine}) absorver o
parte linear, e o que resta é uma lista curta e completa de
formas normais.

\medskip
\noindent\textbf{Part I --- The reduction machine.}
\begin{enumerate}
  \item Deixar $ X = PY + t $ com $ P \in O(3)$ e $ t \in \R^3$ (um
        mudança rígida de coordenadas). Mostre isso $ q(PY + t) =
        Y^{\mathsf T}\!A'Y + 2\langle b', Y\rangle + c'$ com
        \[
        A' = P^{\mathsf T}\!AP, \qquad
        b' = P^{\mathsf T}(At + b), \qquad
        c' = q(t) .
        \]
        Deduza que o \omterm{def:b2:reduction:eigen}{espectro} de $ A $ (daí a sua classificação e
        assinatura) é um invariante rígido da equação, e
        explique por que a equação de um dado \omterm{pb:b2:surfaces:1}{quádrica} é apenas
        determinado até um fator escalar diferente de zero.
  \item Usando o teorema espectral, mostre que após uma rotação
        a equação se torna $\sum_i \lambda_i y_i^2 +
        2\sum_i\beta_iy_i + c = 0$ com $\lambda_1, \lambda_2,
        \lambda_3$ o valores próprios de $ A $.
  \item Para cada $ i $ com $\lambda_i \neq 0$, absorver
        $\beta_iy_i $ por uma tradução ($ y_i \mapsto y_i -
        \beta_i/\lambda_i $). Escreva a equação reduzida quando
        $\operatorname{rank} A = r $: $\sum_{i\leq r}\lambda_i
        z_i^2 + 2\sum_{i > r}\beta_iz_i + c'' = 0$.
  \item UM \emph{centro} do \omterm{pb:b2:surfaces:1}{quádrica} da equação $ q = 0$ é
        um ponto $\Omega $ com $ q(2\Omega - X) = q(X)$ para todos
        $ X $: o ponto de reflexão em $\Omega $ preserva o
        equação, daí a superfície. Mostre isso $ q(2\Omega - X)
        - q(X) = -4\langle A\Omega + b, X\rangle + 4\langle
        A\Omega + b, \Omega\rangle $, e deduza: os centros
        são exatamente as soluções de $ A\Omega = -b $; eles
        existe se $ b \in \operatorname{im}A $, e o centro
        é único se $ A $ é invertível.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Central \omterm{pb:b2:surfaces:1}{quádricas}
($\operatorname{rank}A = 3$).} Aqui a equação reduzida é
$\lambda_1z_1^2 + \lambda_2z_2^2 + \lambda_3z_3^2 = \delta $.
\begin{enumerate}[resume]
  \item Multiplicando por $-1$ se necessário, assuma pelo menos dois
        $\lambda_i > 0$. Enumere as possibilidades:
        assinatura $(3, 0)$ com $\delta > 0$, $= 0$, $< 0$,
        e assinatura $(2, 1)$ com $\delta > 0$, $= 0$, $<
        0$; nomeie os seis conjuntos resultantes (elipsóide, ponto,
        conjunto vazio, hiperbolóide de uma folha, cone, duas folhas
        hiperbolóide) e coloque cada um em sua normal euclidiana
        formulário ($\frac{x^2}{a^2} + \frac{y^2}{b^2} +
        \frac{z^2}{c^2} = 1$, etc.).
  \item Classificar $ x^2 + y^2 + z^2 + 4xy + 4yz + 4zx = 1$:
        mostrar $ A = 2J - I $ com $ J $ a matriz de todos-uns,
        calcular o \omterm{def:b2:reduction:eigen}{espectro} $\{5, -1, -1\}$, e identificar um
        hiperbolóide de revolução de duas folhas em torno do eixo
        $\R(1,1,1)$.
  \item (Decisões) Para o hiperbolóide de uma folha
        $\frac{x^2}{a^2} + \frac{y^2}{b^2} - \frac{z^2}{c^2}
        = 1$, fator
        \[
        \Bigl(\frac xa - \frac zc\Bigr)
        \Bigl(\frac xa + \frac zc\Bigr)
        = \Bigl(1 - \frac yb\Bigr)\Bigl(1 + \frac yb\Bigr)
        \]
        e produzir duas famílias de um parâmetro de linha reta
        linhas que ficam na superfície.
  \item Mostre isso em todos os pontos da folha
        hiperbolóide passa exatamente uma linha de cada família:
        a superfície é \emph{duplamente governado}.
  \item O \emph{cone assintótico} da folha única
        hiperbolóide é $ C : \frac{x^2}{a^2} + \frac{y^2}{b^2}
        - \frac{z^2}{c^2} = 0$. Com $\rho^2 = \frac{x^2}{a^2}
        + \frac{y^2}{b^2}$, mostre que qualquer ponto do
        o hiperbolóide está no máximo distante $ c\bigl(\rho -
        \sqrt{\rho^2 - 1}\bigr) = \frac{c}{\rho +
        \sqrt{\rho^2 - 1}}$ de $ C $, então a superfície abraça seu
        cone no infinito. Quais são as seções do
        hiperbolóide pelos aviões $ x = \pm a $?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Rank $2$ and rank $1$:
paraboloids, cylinders, planes.}
\begin{enumerate}[resume]
  \item Suponha $\operatorname{rank}A = 2$, dizer $\lambda_1,
        \lambda_2 \neq 0 = \lambda_3$. Começando pela pergunta
        3, dividido em dois casos de acordo com $\beta_3 \neq 0$
        (sem centro, pela questão 4) ou $\beta_3 = 0$ (uma linha
        de centros) e reduzir para
        \[
        \lambda_1z_1^2 + \lambda_2z_2^2 + 2\beta_3z_3 = 0
        \qquad\text{or}\qquad
        \lambda_1z_1^2 + \lambda_2z_2^2 + c'' = 0 :
        \]
        parabolóides elípticos/hiperbólicos no primeiro caso,
        cilindros sobre cônicas centrais (ou pares de
        planos que se cruzam, uma linha, o conjunto vazio) no
        segundo.
  \item Mostre que a sela $ z = xy $ é um hiperbólico
        parabolóide: girar por $\pi/4$ no $ xy $-avião para
        alcançar $ z = \tfrac12(u^2 - v^2)$, a superfície de
        \cref{fig:b2:surfaces:saddle} em escala.
  \item Mostre que a sela $ z = xy $ carrega as duas linhas
        famílias $\{x = x_0,\ z = x_0y\}$ e $\{y = y_0,\ z
        = xy_0\}$, com exatamente uma linha de cada através
        cada ponto: o segundo governou duplamente \omterm{pb:b2:surfaces:1}{quádrica}.
  \item Classificar $ x^2 + y^2 - 2x + 4y + 3 = 0$ em $\R^3$
        (completar os quadrados; identificar uma circular direita
        cilindro e indique seu eixo e raio).
  \item Agora deixe $\operatorname{rank}A = 1$, dizer $\lambda_1
        \neq 0 = \lambda_2 = \lambda_3$. Girando dentro do
        plano do kernel e traduzindo, reduza para
        \[
        \lambda_1z_1^2 + 2\beta z_2 = 0 \quad (\beta \neq 0)
        \qquad\text{or}\qquad
        \lambda_1z_1^2 + c'' = 0 :
        \]
        um cilindro parabólico, ou um par de planos paralelos, um
        plano duplo ou conjunto vazio. Classificar $(x + y)^2 =
        z $ completamente (forma normal, eixo de translação
        invariância).
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The classification theorem.}
\begin{enumerate}[resume]
  \item Monte as Partes I-III em um teorema: \emph{todo
        \omterm{pb:b2:surfaces:1}{quádrica} de $\R^3$ é mapeado por um movimento rígido em
        exatamente uma forma normal}. Liste os dezessete \omterm{def:b2:affine:subspace}{afim}
        tipos (contar as variantes vazias e conjuntos degenerados),
        e destacar os nove \omterm{pb:b2:surfaces:1}{quádrica} \emph{superfícies}:
        elipsóide, hiperbolóides de uma e duas folhas, cone,
        parabolóides elípticos e hiperbólicos, elípticos,
        cilindros hiperbólicos e parabólicos.
  \item Escreva a classificação \emph{algoritmo}: dado $(A,
        b, c)$, quais quantidades você calcula, em que
        ordem e qual filial decide qual tipo? Justificar
        que cada passo é eficaz (valores próprios de um
        \omterm{def:b2:quadratic:adjoint}{simétrico} $3\times3$ matriz, classificação, solubilidade de
        $ A\Omega = -b $).
  \item Execute o algoritmo em $ x^2 + y^2 + z^2 - 2xy - 2yz -
        2zx = 1$: mostrar $ A = 2I - J $ tem \omterm{def:b2:reduction:eigen}{espectro} $\{2, 2,
        -1\}$ e concluímos: um hiperbolóide de uma folha de
        revolução sobre $\R(1,1,1)$.
  \item Execute-o $ x^2 + y^2 - z^2 - 2x + 4y + 2z + 4 = 0$:
        encontre o centro e identifique o \omterm{pb:b2:surfaces:1}{quádrica}.
  \item Execute-o $ x^2 + 4xy + y^2 = 2z $: diagonalizar o
        $ xy $-bloquear ($ u = \frac{x+y}{\sqrt2}$, $ v =
        \frac{x-y}{\sqrt2}$) e identificar o \omterm{pb:b2:surfaces:1}{quádrica}.
  \item Euclidiano versus \omterm{def:b2:affine:subspace}{afim}. Mostre que dois centros
        \omterm{pb:b2:surfaces:1}{quádricas} na forma normal são \emph{rigidamente} equivalente
        se eles tiverem as mesmas listas de coeficientes (até
        permutação e um escalar positivo comum no
        equação), enquanto \emph{\omterm{def:b2:affine:subspace}{afim}}, apenas a assinatura
        os dados sobrevivem: todo elipsóide é um \omterm{def:b2:affine:subspace}{afim} imagem de
        a esfera redonda. Qual teorema garante que
        a assinatura não pode mudar ao longo do caminho
        (\cref{thm:b2:quadratic:sylvester})?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends.}
\begin{enumerate}[resume]
  \item Mostre que cada seção de um \omterm{pb:b2:surfaces:1}{quádrica} por um \omterm{def:b2:affine:subspace}{afim}
        plano é uma cônica (possivelmente degenerada) desse plano.
        Identifique as seções da sela $ z = xy $ pelo
        aviões $ z = c $ ($ c \neq 0$ e $ c = 0$).
  \item Qual \omterm{pb:b2:surfaces:1}{quádrica} superfícies contêm linhas retas? Mostrar
        que o elipsóide, o hiperbolóide de duas folhas e o
        parabolóide elíptico não contém nenhum \emph{(restringir $ q $
        a uma reta e use a desigualdade de Cauchy-Schwarz para
        o caso de duas folhas)}; que o cone e os cilindros são
        governado por uma família; e concluir que o duplamente
        governou \omterm{pb:b2:surfaces:1}{quádrica} superfícies são exatamente de uma folha
        hiperbolóide e o parabolóide hiperbólico.
  \item Quando apenas o \emph{\omterm{def:b2:affine:subspace}{afim}} tipo é desejado, Gauss
        redução (\cref{thm:b2:quadratic:gauss}) é mais barato
        do que diagonalizar. Refaça a questão 17 com Gauss
        algoritmo e verifique a assinatura $(2, 1)$; o que
        Informação euclidiana Gauss perde?
  \item Todos valores próprios de um \omterm{def:b2:quadratic:adjoint}{simétrico} matriz são reais; mostrar
        que consequentemente o \emph{sinais} do valores próprios
        de $ A $ pode ser lido a partir do polinômio característico
        pela regra dos sinais de Descartes, e verifique isso em
        pergunta 6: $\chi_A(\lambda) = \lambda^3 - 3\lambda^2
        - 9\lambda - 5$ tem exatamente uma mudança de sinal, portanto
        assinatura $(1, 2)$.
  \item Síntese. Resuma o algoritmo em poucas linhas;
        indique o papel exato desempenhado por (i) o \omterm{def:b2:reduction:eigen}{espectro}
        teorema, (ii) a equação central $ A\Omega = -b $,
        (iii) teorema da inércia de Sylvester, (iv) teorema de Gauss
        redução. O que a mesma máquina rende em
        $\R^2$, e o que muda em $\R^n$?
\end{enumerate}
\end{problem}
