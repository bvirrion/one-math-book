\chapter{Funções de uma variável real}\label{ch:b2:realfun}

Antes que a análise passe para funções de funções (\cref{ch:b2:funcseq})
vale a pena conhecer o cenário de uma variável com mais detalhes do que
Ano 1 obrigatório: quão descontínua pode ser uma função monótona, quão
regular uma função convexa deve ser e quais propriedades especiais
desfrutar de derivados (Darboux). Esses resultados estruturais são curtos,
afiado e amado pelos examinadores.

\section{Funções monótonas}

\begin{theorem}[Regularidade de funções monótonas]\label{thm:b2:realfun:monotone}
Deixar $f \colon I \to \R $ estar aumentando em um intervalo.
\begin{enumerate}
  \item Em cada ponto interior $ a $, os limites unilaterais existem:
        \[
        f(a^-) = \sup_{x < a} f(x) \;\leq\; f(a) \;\leq\;
        f(a^+) = \inf_{x > a} f(x) ;
        \]
        toda descontinuidade é uma \emph{pular}.
  \item O conjunto de descontinuidades de $ f $ é no máximo
        \omterm{def:b2:structures:countable}{contável}.\index{função monótona!descontinuidades}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) O conjunto $\{f(x) : x < a\}$ não é vazio, limitado acima por $ f(a)$:
seu supremo $ s $ satisfaz $ f(x) \to s $ como $ x \to a^-$ (dado
$\varepsilon $, alguns $ f(x_0) > s - \varepsilon $ e monotonicidade
armadilhas $ f(x) \in \intoc{s - \varepsilon}{s}$ para $ x \in
\intoo{x_0}{a}$). Simetricamente à direita.

(2) Para cada descontinuidade $ a $ anexe o não vazio \omterm{def:b2:metric:topology}{abrir} intervalo
$ J_a = \intoo{f(a^-)}{f(a^+)}$ (um salto genuíno). Para $ a < b $
descontinuidades, $ J_a $ e $ J_b $ são disjuntos: $ f(a^+) \leq f(c)
\leq f(b^-)$ para qualquer $ c $ entre. Cada $ J_a $ contém um racional;
descontinuidades distintas obtêm racionais distintos: uma injeção do
descontinuidade definida em $\Q $, que é \omterm{def:b2:structures:countable}{contável}
(\cref{prop:b2:structures:countablestable}).
\end{proof}

\begin{example}
O limite é nítido: corrija uma enumeração $(r_n)$ de $\Q \cap
\intoo{0}{1}$ e definir $ f(x) = \sum_{n : r_n \leq x} 2^{-n}$
(uma definição de família resumida,
\cref{def:b2:series:summable}). Então $ f $ está aumentando em
$\intcc{0}{1}$ e descontínuo exatamente em todos os racionais de
$\intoo{0}{1}$ (pular $2^{-n}$ no $ r_n $): uma função monótona
\emph{pode} ser descontínuo em um denso conjunto \omterm{def:b2:structures:countable}{contável}.
\end{example}

\begin{example}[Os saltos não podem superar a ascensão]\label{ex:b2:realfun:jumpbudget}
Para $ f $ aumentando em $\intcc{a}{b}$, os saltos têm um orçamento:
se $ a < c_1 < \dots < c_m < b $ são descontinuidades com saltos
$ s_i = f(c_i^+) - f(c_i^-) > 0$, em seguida, escolhendo entrelaçamento
pontos $ a < c_1 < t_1 < c_2 < \dots $ e usando monotonicidade em
cada peça,
\[
\sum_{i=1}^{m} s_i \;\leq\; f(b) - f(a) :
\]
a subida total limita o salto total. Consequência: para
cada $ k $, no máximo $ k\,\bigl(f(b) - f(a)\bigr)$ descontinuidades
tem salto $\geq \frac1k $ --- um refinamento quantitativo de
\cref{thm:b2:realfun:monotone}~(2), já que o conjunto de descontinuidades
é o \omterm{def:b2:structures:countable}{contável} união acabou $ k $ desses conjuntos finitos. No
função de salto racional acima, o orçamento é gasto exatamente:
os saltos $2^{-n}$ soma para $1 = f(1^+) - f(0^-)$ no óbvio
sentido estendido. Funções monótonas podem saltar densamente, mas apenas
em um subsídio estrito.
\end{example}

\section{Funções convexas}

\begin{lemma}[Desigualdade de inclinação]\label{lem:b2:realfun:slopes}
Deixar $ f $ ser convexo em $ I $ e $ x < y < z $ em $ I $. Então
\[
\frac{f(y) - f(x)}{y - x}
\;\leq\; \frac{f(z) - f(x)}{z - x}
\;\leq\; \frac{f(z) - f(y)}{z - y} :
\]
as inclinações dos acordes aumentam em ambos os pontos finais.\index{convexo
função!desigualdade de inclinação}
\end{lemma}

\begin{proof}
Escrever $ y = \frac{z - y}{z - x}\,x + \frac{y - x}{z - x}\,z $: um
combinação convexa, uma vez que os dois coeficientes são positivos e
soma para $1$. Convexidade dá
\[
f(y) \;\leq\; \frac{z-y}{z-x}\,f(x) + \frac{y-x}{z-x}\,f(z).
\]
Para a desigualdade à esquerda, subtraia $ f(x)$ de ambos os lados, usando
$\frac{z-y}{z-x} - 1 = -\frac{y-x}{z-x}$:
\[
f(y) - f(x) \leq \frac{y - x}{z - x}\bigl(f(z) - f(x)\bigr),
\]
e dividir por $ y - x > 0$. Para a desigualdade correta, subtraia
em vez de $ f(z)$:
\[
f(z) - f(y) \geq f(z) - \frac{z-y}{z-x}f(x) -
\frac{y-x}{z-x}f(z)
= \frac{z - y}{z - x}\bigl(f(z) - f(x)\bigr),
\]
e dividir por $ z - y > 0$. Ambas as etapas exibidas são iguais
identidade baricêntrica lida em um endpoint diferente.
\end{proof}

\begin{theorem}[Regularidade de funções convexas]\label{thm:b2:realfun:convexreg}
Deixar $ f $ ser convexo em um intervalo $ I $.
\begin{enumerate}
  \item Em cada ponto interior, $ f $ tem unilateral finito
        derivados $ f'_g \leq f'_d $; ambos estão aumentando funções
        do ponto; em particular $ f $ é \omterm{def:b2:metric:continuity}{contínuo} no
        interior de $ I $ (mas possivelmente não nos pontos finais).
  \item $ f $ está acima de cada \emph{linha de suporte}: para $ a $ interior
        e qualquer $ m \in \intcc{f'_g(a)}{f'_d(a)}$,
        \[
        f(x) \geq f(a) + m(x - a) \qquad (x \in I).
        \]
  \item (Jensen, ponderado) Para $ x_i \in I $ e pesos $\lambda_i
        \geq 0$, $\sum\lambda_i = 1$:
        \[
        f\Bigl(\sum_i \lambda_i x_i\Bigr) \leq \sum_i \lambda_i
        f(x_i) .\index{Desigualdade de Jensen}
        \]
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Correção $ a $ interior. Por \cref{lem:b2:realfun:slopes}, a inclinação
$\tau(h) = \frac{f(a + h) - f(a)}{h}$ é uma função crescente de
$ h $ (em ambos os lados, e $\tau(h_-) \leq \tau(h_+)$ para $ h_- < 0 <
h_+$). Por isso $\tau $ tem um limite finito como $ h \to 0^-$ (aumentando,
delimitado acima por qualquer inclinação direita) --- isto é $ f'_g(a)$ --- e como
$ h \to 0^+$ ($ f'_d(a)$), com $ f'_g(a) \leq f'_d(a)$. Finito
força derivada unilateral continuidade no $ a $. Monotonicidade no
ponto: para $ a < b $ interior, $ f'_d(a) \leq \frac{f(b) - f(a)}{b -
a} \leq f'_g(b)$, novamente pela desigualdade de inclinação.

(2) Para $ x > a $: $\frac{f(x) - f(a)}{x - a} \geq f'_d(a) \geq m $;
para $ x < a $: $\frac{f(a) - f(x)}{a - x} \leq f'_g(a) \leq m $. Ambos
reorganizar para a reivindicação.

(3) Indução no número de pontos exatamente como no Ano 1
volume (o caso de dois pontos é a definição) --- ou em um golpe:
aplicar (2) em $ a = \sum\lambda_i x_i $ e média da linha de suporte
desigualdades nos pontos $ x_i $ com pesos $\lambda_i $: $\sum_i
\lambda_i f(x_i) \geq f(a) + m\sum_i\lambda_i(x_i - a) = f(a)$.
\end{proof}

\begin{example}[Descontinuidade do terminal]
Sobre $\intcc{0}{1}$, a função $ f(0) = 1$, $ f(x) = 0$ para $ x > 0$
é convexo, mas descontínuo no ponto final $0$: afirmação (1) é
afiado.
\end{example}

\begin{example}[Cantos e feixe de linhas de suporte]\label{ex:b2:realfun:corner}
Para $ f(x) = \abs x $ no $ a = 0$: as derivadas unilaterais são
$ f'_g(0) = -1$ e $ f'_d(0) = +1$, e
\cref{thm:b2:realfun:convexreg}~(2) distribui uma linha de apoio para
\emph{todo} declive $ m \in \intcc{-1}{1}$:
\[
\abs x \geq m\,x \qquad (x \in \R,\ -1 \leq m \leq 1),
\]
cada uma uma igualdade exatamente em uma meia-linha ou em $0$. Um convexo
a função é diferenciável em $ a $ precisamente quando o feixe
é recolhido em uma única linha ($ f'_g(a) = f'_d(a)$); cantos carregam
um intervalo de tangentes. Este feixe é o de dimensão finita
germe do \emph{subdiferencial} de otimização convexa ---
e a razão pela qual as funções convexas são tão robustas: mesmo onde o
derivada falha, a geometria de suporte sobrevive, o que é tudo
que a prova de Jensen usou.
\end{example}

\begin{example}[Poder significa desigualdade]\label{ex:b2:realfun:powermeans}
Para $0 < p < q $ e positivo $ x_i $ com pesos $\lambda_i $
resumindo para $1$, aplicando Jensen ao convexo $ t \mapsto t^{q/p}$
nos pontos $ x_i^p $:
\[
\Bigl(\sum \lambda_i x_i^{p}\Bigr)^{1/p}
\leq \Bigl(\sum \lambda_i x_i^{q}\Bigr)^{1/q} :
\]
potência significa aumentar com o expoente --- contendo AM--QM, e,
no limite $ p \to 0$ (\cref{exo:b2:realfun:6}), o AM-GM
desigualdade mais uma vez.
\end{example}

\begin{example}[Entropia máxima]\label{ex:b2:realfun:entropy}
Para um vetor de probabilidade $(p_1, \dots, p_n)$ (positivo,
resumindo para $1$), a entropia $ H(p) = -\sum_i p_i\ln p_i $
satisfaz
\[
H(p) \leq \ln n ,
\qquad\text{with equality iff } p_i = \frac1n \text{ for all }
i .
\]
Prova de Jensen (\cref{thm:b2:realfun:convexreg}~(3)) aplicado a
o \emph{côncavo} $\ln $ com pesos $ p_i $ nos pontos
$\frac{1}{p_i}$:
\[
H(p) = \sum_i p_i\ln\frac{1}{p_i}
\leq \ln\Bigl(\sum_i p_i\,\frac1{p_i}\Bigr) = \ln n ,
\]
igualdade forçando todos os pontos $\frac1{p_i}$ igual (estrito
concavidade), ou seja,\ $ p $ uniforme. Equivalentemente, isso é
\cref{exo:b2:realfun:7} com $ q $ uniforme. A incerteza é
maximizado pela ignorância uniformemente espalhada --- o variacional
princípio por trás da codificação, mecânica estatística e entropia
aparições de \cref{ch:b2:randomvar}.
\end{example}

\begin{method}[Encontrando a função convexa por trás de uma desigualdade]\label{met:b2:realfun:spotting}
A maioria das desigualdades clássicas são Jensen fantasiado; despir-se
um: (1) normalizar para que um \emph{média ponderada} aparece
(pesos positivos, somando $1$ --- divida pela massa total se
necessário); (2) observe qual função é aplicada internamente versus
fora da média: a afirmação ``$ f(\text{average}) \leq $
média de $ f $'' nomeia o convexo $ f $; (3) certificar a convexidade por
a segunda derivada e trata a igualdade por meio do rigor; (4)
se nenhuma média estiver visível, calcule primeiro os logaritmos --- produtos
e potências tornam-se médias, e a concavidade de $\ln $ carrega
AM--GM, Young e seus parentes (fim de semana deste capítulo
problema executa as etapas 1 a 4 em cada um deles). Se até mesmo os logaritmos
não revelar uma média, tente ler a desigualdade como
monotonicidade das encostas (\cref{lem:b2:realfun:slopes}) ---
declarações de superaditividade como \cref{exo:b2:realfun:9} viver
lá.
\end{method}

\begin{remark}[Armadilhas comuns]\label{rem:b2:realfun:pitfalls}
(i) A convexidade não é preservada pelos produtos: $ x $ e $(x - 1)^2$
são convexos em $\intcc{0}{2}$, mas seu produto $ x(x-1)^2$ tem
segunda derivada $6x - 4$, negativo em
$\intco{0}{\frac23}$ --- não convexo; nem a convexidade é preservada
por composição sem monotonicidade
(\cref{exo:b2:realfun:10}). (ii) Jensen vira para côncavo
funções: metade das desigualdades clássicas são côncavas
$\ln $-versão; aplicando a forma convexa a $\ln $ é o
maneira mais rápida de provar AM--GM \emph{para trás}. (iii) Ponto Médio
convexidade por si só não implica convexidade --- continuidade (ou
mera limitação) é necessária (\cref{exo:b2:realfun:8}); o
contra-exemplos patológicos vivem além dos axiomas deste livro. (iv)
Uma função convexa em um \emph{\omterm{def:b2:metric:topology}{abrir}} intervalo é \omterm{def:b2:metric:continuity}{contínuo},
mesmo localmente \omterm{def:b2:metric:continuity}{Lipschitz} (\cref{exo:b2:realfun:12}); em
endpoints, nada é de graça. (v) Os derivativos obedecem a Darboux, mas
não precisa ser \omterm{def:b2:metric:continuity}{contínuo} (\cref{ex:b2:realfun:oscillation}):
``$ f'$ não tem saltos'' nunca significa ``$ f'$ é \omterm{def:b2:metric:continuity}{contínuo}''.
\end{remark}

\section{A propriedade Darboux}

\begin{theorem}[Darboux]\label{thm:b2:realfun:darboux}
Deixar $ f $ ser diferenciável em um intervalo $ I $. Então $ f'$ leva cada
valor entre quaisquer dois de seus valores --- mesmo que $ f'$ não precisa ser
\omterm{def:b2:metric:continuity}{contínuo}.\index{Teorema de Darboux}
\end{theorem}

\begin{proof}
Deixar $ a < b $ em $ I $ e $ v $ estritamente entre $ f'(a)$ e $ f'(b)$,
dizer $ f'(a) < v < f'(b)$. A função $ g(x) = f(x) - vx $ é
diferenciável com $ g'(a) < 0 < g'(b)$: seu mínimo ativado
$\intcc{a}{b}$ (atingido: continuidade em um \omterm{def:b2:metric:compact}{compactar}) não está em $ a $
(logo depois $ a $, $ g $ diminui abaixo $ g(a)$) nem em $ b $ (apenas
antes $ b $, $ g $ está abaixo $ g(b)$): é interior, e há $ g'(c)
= 0$, ou seja\ $ f'(c) = v $. (Este foi um exercício com estrela do Ano 1; seu
lugar na teoria é aqui.)
\end{proof}

\begin{example}[Quais funções são derivadas?]\label{ex:b2:realfun:whichderivatives}
O teorema de Darboux é uma máquina inexistente. A função do chão
$\lfloor x\rfloor $ não é a derivada de nenhuma função em
$\R $: leva os valores $0$ e $1$ mas pula $\frac12$ sobre
$\intcc{0}{1}$, qual \cref{thm:b2:realfun:darboux} proíbe
derivados. O mesmo veredicto atinge todas as funções com um salto
--- sinal, Heaviside, todas as funções de etapa --- por mais inocentes que sejam
eles olham; suas ``antiderivadas'' ($\abs x $ para sinal, etc.)
existe apenas longe do salto e dá um nó ali com um canto.
Contraste: o extremamente descontínuo $ f'$ de
\cref{ex:b2:realfun:oscillation} \emph{é} uma derivada --- é
a descontinuidade é uma oscilação que Darboux tolera. O
O limite entre os dois comportamentos é exatamente o
corolário sem saltos abaixo.
\end{example}

\begin{corollary}\label{cor:b2:realfun:nojumps}
Uma derivada não tem descontinuidades de salto: se $ f'(a^-)$ e
$ f'(a^+)$ existem, eles são iguais $ f'(a)$. As descontinuidades de um
derivadas são sempre do tipo oscilação ($ x^2\sin\frac1x $ de
derivada em $0$, volume do ano 1).
\end{corollary}

\begin{proof}
Se $ f'(a^+) = \lim_{x\to a^+} f'(x)$ existe e difere de
$ f'(a)$, os valores estritamente entre eles seriam ignorados por $ f'$ ligado
um bairro certo --- contradizendo Darboux em intervalos
$\intcc{a}{a + h}$. (Alternativamente: o teorema do valor médio força
$ f'(a) = \lim_{h\to0^+} \frac{f(a+h)-f(a)}{h} = f'(a^+)$, o
quociente de diferença sendo um $ f'$-valor em um ponto intermediário.)
O mesmo à esquerda.
\end{proof}

\begin{example}[A derivada oscilante canônica]\label{ex:b2:realfun:oscillation}
Deixar $ f(x) = x^2\sin\frac1x $ para $ x \neq 0$ e $ f(0) = 0$. No
$0$: $\bigl|\frac{f(h) - f(0)}{h}\bigr| = \abs{h\sin\frac1h}
\leq \abs h \to 0$, então $ f'(0) = 0$ existe. Longe de $0$,
\[
f'(x) = 2x\sin\frac1x - \cos\frac1x ,
\]
cujo primeiro termo tende a $0$ enquanto $\cos\frac1x $ oscila
através $\intcc{-1}{1}$ em cada intervalo $\intoo{0}{\delta}$: o
limite $ f'(0^+)$ não existe. Então $ f'$ é definido em todos os lugares, mas
descontínuo em $0$ --- e, exatamente como
\cref{cor:b2:realfun:nojumps} prevê, a descontinuidade é uma
oscilação, não um salto: em cada $\intoo{0}{\delta}$, $ f'$ ainda
varre um intervalo \omterm{def:b2:metric:complete}{completo} ao redor $0$. Os derivativos podem ser selvagens, mas
apenas da maneira compatível com Darboux.
\end{example}

\begin{remark}[Onde este capítulo é usado]
A convexidade é o motor da indústria da desigualdade: esta
o problema do fim de semana do capítulo fabrica Young, Hölder, Minkowski
e a cadeia de potência média dele, que \cref{ch:b2:nvs}de \omterm{def:b2:nvs:norm}{norma}
teoria e \cref{ch:b2:integration}as estimativas integrais consomem;
Jensen reaparece na probabilidade como o momento das desigualdades de
\cref{ch:b2:randomvar}. A regularidade monótona retorna em
\cref{ch:b2:integration} (funções monótonas são integráveis) e,
no volume do Ano 3, como a diferenciabilidade em quase todos os lugares
de funções monótonas --- onde ``\omterm{def:b2:structures:countable}{contável} muitos saltos'' torna-se
o primeiro passo da teoria de Lebesgue.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b2:realfun:1}
Determine os conjuntos de descontinuidades e os tamanhos dos saltos: $\lfloor x
\rfloor $; $\;x - \lfloor x\rfloor $; $\;\lfloor x \rfloor + \sqrt{x
- \lfloor x\rfloor}$; a função do exemplo a seguir
\cref{thm:b2:realfun:monotone} restrito a racionais diádicos
$ r_n $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:realfun:2}
Prove que uma função crescente $ f \colon I \to \R $ com o
propriedade de valor intermediário (sua imagem de qualquer subintervalo é um
intervalo) é \omterm{def:b2:metric:continuity}{contínuo}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:realfun:3}
Quais das alternativas a seguir são convexas em seu domínio? $ x \mapsto
x\ln x $ ($ x > 0$); $\;x \mapsto \ln(1 + \eu^x)$; $\;x \mapsto
\sqrt{1 + x^2}$; $\;x \mapsto x^3$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:realfun:4}
Deixar $ f $ ser convexo em $\R $ e limitado acima. Prove isso $ f $ é
constante. \emph{(Se $ f(a) \neq f(b)$, a desigualdade de inclinação
propaga a inclinação da corda diferente de zero: além do ponto com maior
valor, $ f $ cresce pelo menos linearmente --- contradizendo a limitação.
Trate ambos os sinais da inclinação.)} Deduza que uma função convexa em
$\R $ com uma assíntota em ambas as extremidades é afim.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:realfun:5}
Deixar $ f $ ser diferenciável em $ I $ com $ f'$ monótono. Prove isso
$ f'$ é \omterm{def:b2:metric:continuity}{contínuo} \emph{(combinar
\cref{thm:b2:realfun:monotone} e
\cref{cor:b2:realfun:nojumps})}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:realfun:6}
(Média geométrica como limite) Para positivo $ x_i $ e pesos
$\lambda_i $ resumindo para $1$, provar
\[
\lim_{p \to 0^+} \Bigl(\sum_i \lambda_i x_i^p\Bigr)^{1/p}
= \prod_i x_i^{\lambda_i} ,
\]
através de $ x_i^p = \eu^{p\ln x_i} = 1 + p\ln x_i + O(p^2)$, e deduza
a desigualdade AM-GM ponderada de
\cref{ex:b2:realfun:powermeans}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:realfun:7}
(Desigualdade de entropia) Usando convexidade estrita de $ t \mapsto t\ln t $,
provar que com certeza $ p_i, q_i $ com $\sum p_i = \sum q_i = 1$:
\[
\sum_i p_i \ln\frac{p_i}{q_i} \geq 0 ,
\]
com igualdade se $ p = q $. \emph{(Escreva o lado esquerdo como $\sum q_i\,
\varphi\bigl(\frac{p_i}{q_i}\bigr)$ com $\varphi(t) = t\ln t $ e
aplicar Jensen com pesos $ q_i $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:realfun:8}
(Convexidade do ponto médio) $ f \colon I \to \R $ é \emph{ponto médio convexo}
quando $ f\bigl(\frac{x+y}{2}\bigr) \leq \frac{f(x) + f(y)}{2}$
sempre. Prove que um \emph{\omterm{def:b2:metric:continuity}{contínuo}} função convexa do ponto médio é
convexo. \emph{(Estabeleça a desigualdade de convexidade para diádicos
pesos $\frac{k}{2^m}$ por indução em $ m $, então passe para o limite
usando densidade e continuidade.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:realfun:9}
Deixar $ f $ ser convexo em $\intco{0}{+\infty}$ com $ f(0) \leq 0$. Provar
isso $ x \mapsto \frac{f(x)}{x}$ está aumentando em
$\intoo{0}{+\infty}$, e deduza isso para convexo $ f $ com $ f(0) =
0$: $ f(x + y) \geq f(x) + f(y)$ para $ x, y \geq 0$
(superaditividade).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:realfun:10}
Deixar $ f $ ser convexo em $ I $ e $ g $ convexo \emph{aumentando} em um
intervalo contendo $ f(I)$. Prove isso $ g \circ f $ é convexo e
mostre através de um contra-exemplo que a monotonicidade de $ g $ não pode ser
caiu.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:realfun:11}
(Hermite--Hadamard) Deixe $ f $ ser convexo e \omterm{def:b2:metric:continuity}{contínuo} sobre
$\intcc{a}{b}$. Provar
\[
f\Bigl(\frac{a+b}{2}\Bigr) \;\leq\; \frac{1}{b -
a}\int_a^b f(t)\,\dd t \;\leq\; \frac{f(a) + f(b)}{2} .
\]
\emph{(Esquerda: integre uma linha de suporte no ponto médio. Direita:
vinculado $ f $ pelo acorde.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:realfun:12}
Prove que uma função convexa em um \emph{\omterm{def:b2:metric:topology}{abrir}} intervalo $ I $ é
localmente \omterm{def:b2:metric:continuity}{Lipschitz}: para cada segmento $\intcc{a}{b} \subseteq I $
e margem $\delta > 0$ com $\intcc{a - \delta}{b + \delta}
\subseteq I $, a restrição de $ f $ para $\intcc{a}{b}$ é
\omterm{def:b2:metric:continuity}{Lipschitz}, com constante
$\max\Bigl(\bigl|\frac{f(a) - f(a - \delta)}{\delta}\bigr|,
\bigl|\frac{f(b + \delta) - f(b)}{\delta}\bigr|\Bigr)$
\emph{(capturar cada inclinação da corda entre esses dois pela inclinação
desigualdade)}.
\end{exercise}

\section{Problema: a caixa de ferramentas da convexidade}

Uma definição --- o acorde acima do gráfico --- gera o
toda a caixa de ferramentas das desigualdades clássicas. Este problema de fim de semana
constrói-o em ordem lógica: critérios de convexidade e Jensen estrito,
então Young, Hölder e Minkowski (as certidões de nascimento dos
$ p $-normas), a cadeia média de potência completa do mínimo ao
máximo, e dois dividendos da coroa --- desigualdade de Carleman, e
Hölder lido como uma dualidade. Tudo está provado; nada é
importado.

\begin{problem}[{Problema de fim de semana --- Young, Hölder, Minkowski,
e a cadeia de potência média}]
\label{pb:b2:realfun:1}
Por todo, $ p, q > 1$ são \emph{expoentes conjugados}: $\frac1p +
\frac1q = 1$; vetores são $ a = (a_1, \dots, a_n) \in \R^n $;
pesos $\lambda_i > 0$ satisfazer $\sum_i\lambda_i = 1$.

\medskip
\noindent\textbf{Part I --- Criteria and strict Jensen.}
\begin{enumerate}
  \item Deixar $ f $ ser diferenciável em um intervalo $ I $. Prove isso
        $ f $ é convexo se e somente se $ f'$ está aumentando
        \emph{(uma direção passando até o limite no
        desigualdade de inclinação \cref{lem:b2:realfun:slopes}; o
        outro pelo teorema do valor médio)}. Deduza o $ C^2$
        critério $ f'' \geq 0$.
  \item Suponha $ f''> 0$ sobre $ I $. Prove isso $ f $ é
        \emph{estritamente} convexo (desigualdade estrita para $ x \neq y $
        e $\lambda \in \intoo01$), e que um estritamente convexo
        função satisfaz a desigualdade de Jensen
        (\cref{thm:b2:realfun:convexreg}~(3)) com igualdade
        \emph{apenas} quando todos os $ x_i $ coincidir.
  \item Certifique as matérias-primas da caixa de ferramentas: $-\ln $ é estritamente
        convexo em $\intoo{0}{+\infty}$; $ t \mapsto t^r $ é
        estritamente convexo lá para $ r > 1$ e estritamente côncavo
        para $0 < r < 1$; $\exp $ é estritamente convexo em $\R $.
  \item (Desigualdade de Young) Para $ a, b \geq 0$, provar
        \[
        ab \;\leq\; \frac{a^p}{p} + \frac{b^q}{q},
        \]
        com igualdade se e somente se $ a^p = b^q $ \emph{(aplicar
        a concavidade de $\ln $ para os dois pontos $ a^p, b^q $
        com pesos $\frac1p, \frac1q $)}.
  \item Derive novamente AM-GM ponderado em uma linha do
        concavidade de $\ln $:
        \[
        \prod_i x_i^{\lambda_i} \leq \sum_i\lambda_ix_i \qquad
        (x_i > 0),
        \]
        com o caso da igualdade; compare com a rota limite de
        \cref{exo:b2:realfun:6}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Hölder and Minkowski.} Escrever
$\norm{a}_p = \bigl(\sum_i \abs{a_i}^p\bigr)^{1/p}$ e
$\norm{a}_\infty = \max_i\abs{a_i}$.
\begin{enumerate}[resume]
  \item (Hölder) Provar
        \[
        \sum_{i=1}^{n}\abs{a_ib_i} \;\leq\;
        \norm a_p\,\norm b_q ,
        \]
        com igualdade se os vetores $(\abs{a_i}^p)$ e
        $(\abs{b_i}^q)$ são proporcionais \emph{(normalizar
        $\norm a_p = \norm b_q = 1$ e aplicar Young termicamente)}.
  \item Identifique os casos especiais: $ p = q = 2$
        (Cauchy--Schwarz), e o par final $(p, q) = (1,
        \infty)$: declarar e provar $\sum\abs{a_ib_i} \leq
        \norm a_1\norm b_\infty $.
  \item (Minkowski) Para $ p \geq 1$, provar
        \[
        \norm{a + b}_p \leq \norm a_p + \norm b_p
        \]
        \emph{(escrever $\abs{a_i + b_i}^p \leq \abs{a_i +
        b_i}^{p-1}(\abs{a_i} + \abs{b_i})$ e aplique Hölder
        para cada produto)}. Concluir: $\norm\cdot_p $ é um \omterm{def:b2:nvs:norm}{norma}
        sobre $\R^n $ para cada $ p \in \intco{1}{+\infty}$,
        completando o quadro de \cref{ch:b2:nvs}.
  \item Versões integrais: para $ f, g $ \omterm{def:b2:metric:continuity}{contínuo} sobre
        $\intcc{a}{b}$, declare e prove Hölder e Minkowski
        para $\norm f_p = \bigl(\int_a^b\abs
        f^p\bigr)^{1/p}$ \emph{(mesmas provas, com o estrito
        positividade da integral para a igualdade
        discussão)}.
  \item Prove a monotonicidade $\norm a_q \leq \norm a_p $ para $1
        \leq p \leq q $, o limite $\norm a_p \to
        \norm a_\infty $ como $ p \to \infty $, e o inverso
        comparação com a constante nítida:
        \[
        \norm a_p \leq n^{\frac1p - \frac1q}\,\norm a_q
        \]
        \emph{(Hölder contra o vetor constante)}. Identificar
        os vetores alcançando cada igualdade.
  \item (Interpolação) Para $1 \leq p < r < q $ e $\theta \in
        \intoo01$ com $\frac1r = \frac\theta p +
        \frac{1-\theta}q $, provar
        \[
        \norm a_r \leq \norm a_p^{\theta}\,
        \norm a_q^{1-\theta}
        \]
        \emph{(aplicar Hölder com expoentes $\frac{p}{\theta
        r}$ e $\frac{q}{(1-\theta)r}$ para $\abs{a_i}^{\theta
        r}\abs{a_i}^{(1-\theta)r}$)}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The power-mean chain, complete.}
Para $ p \neq 0$ definir $ M_p = \bigl(\sum_i\lambda_i
x_i^p\bigr)^{1/p}$ ($ x_i > 0$), e $ M_0 = \prod_i
x_i^{\lambda_i}$.
\begin{enumerate}[resume]
  \item Prove isso $ p \mapsto M_p $ está aumentando em todos
        $\R^*$: tratar $ p < q < 0$ pela identidade recíproca
        $ M_{-p}(x) = M_p(1/x)^{-1}$ e fazer uma ponte através $0$ por
        mostrando $ M_p \leq M_0 \leq M_q $ para $ p < 0 < q $
        \emph{(aplique a concavidade de $\ln $ para $ x_i^q $, e o
        desigualdade invertida para expoentes negativos)}.
  \item Prove os limites $ M_p \to \max_i x_i $ como $ p \to
        +\infty $ e $ M_p \to \min_i x_i $ como $ p \to -\infty $.
  \item Escreva a cadeia $\min \leq \mathrm{HM} \leq
        \mathrm{GM} \leq \mathrm{AM} \leq \mathrm{QM} \leq
        \max $ para pesos iguais e prove o clássico
        consequência: para positivo $ a_1, \dots, a_n $,
        \[
        \Bigl(\sum_i a_i\Bigr)\Bigl(\sum_i\frac1{a_i}\Bigr)
        \geq n^2 .
        \]
  \item Relacionar significa \omterm{def:b2:nvs:norm}{normas}: para pesos iguais $\lambda_i =
        \frac1n $, $ M_p(x) = n^{-1/p}\norm x_p $. Reconcilie o
        duas monotonicidades --- significa \emph{aumentar} com $ p $
        enquanto \omterm{def:b2:nvs:norm}{normas} \emph{diminuir} (questão 10) --- em um
        frase sobre o fator $ n^{-1/p}$.
  \item Determine os casos de igualdade ao longo de toda a cadeia de
        questão 14 (pesos positivos): igualdade em qualquer lugar
        força tudo $ x_i $ igual --- convexidade estrita compensa.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Dividends.}
\begin{enumerate}[resume]
  \item (Young com um botão) Para $ a, b \geq 0$ e $\varepsilon >
        0$, provar
        \[
        ab \leq \varepsilon\,\frac{a^p}{p} +
        \varepsilon^{-q/p}\,\frac{b^q}{q},
        \]
        e o caso burro de carga $ ab \leq \varepsilon a^2 +
        \frac{b^2}{4\varepsilon}$: o truque de absorção usado
        ao longo da análise.
  \item (Em direção a Carleman) Vamos $ c_k = \frac{(k+1)^k}{k^{k-1}}$.
        Prove a identidade telescópica $\prod_{k=1}^{n}c_k =
        (n+1)^n $, e deduzir, por AM--GM aplicado aos números
        $ c_ka_k $,
        \[
        (a_1a_2\cdots a_n)^{1/n} \leq
        \frac{1}{n(n+1)}\sum_{k=1}^{n} c_k a_k
        \qquad (a_k > 0).
        \]
  \item (Desigualdade de Carleman) Soma $ n $, troque o
        ordem de soma (positiva resumido famílias,
        \cref{thm:b2:series:fubini}) e usar $\sum_{n \geq
        k}\frac{1}{n(n+1)} = \frac1k $ e $ c_k/k = \bigl(1 +
        \frac1k\bigr)^k < \eu $ para concluir: para cada
        convergente $\sum a_k $ com termos positivos,
        \[
        \sum_{n=1}^{\infty}(a_1a_2\cdots a_n)^{1/n}
        \;\leq\; \eu\sum_{k=1}^{\infty}a_k .
        \]
  \item Para $ f $ \omterm{def:b2:metric:continuity}{contínuo} e positivo em $\intcc{0}{1}$,
        provar
        \[
        \Bigl(\int_0^1 f\Bigr)\Bigl(\int_0^1\frac1f\Bigr)
        \geq 1,
        \]
        com igualdade se $ f $ é constante
        \emph{(Cauchy--Schwarz em $\sqrt f\cdot\frac1{\sqrt
        f}$)}.
  \item (Geometria das bolas) Usando o caso de igualdade de
        Minkowski, mostre que para $1 < p < \infty $ a unidade
        esfera de $\norm\cdot_p $ não contém nenhum segmento (o \omterm{def:b2:nvs:norm}{norma}
        é estritamente convexo no sentido de
        \cref{pb:b2:nvs:1}), enquanto para $ p = 1$ e $ p =
        \infty $ sim: exiba as peças planas.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Duality and synthesis.}
\begin{enumerate}[resume]
  \item (Hölder que a dualidade) Prove que para cada $ a \in
        \R^n $,
        \[
        \norm a_p = \max_{\norm b_q \leq 1}\ \sum_i a_ib_i ,
        \]
        exibindo uma maximização $ b $ explicitamente. (O $ p $-norma
        é o \omterm{def:b2:linalg:dual}{dual} do $ q $-norma --- a dimensão finita
        germe de $ L^p $ dualidade.)
  \item (Momentos) Deixe $ X $ ser uma variável aleatória tomando finitamente
        muitos valores positivos $ x_i $ com probabilidades
        $\lambda_i $. Reafirme a questão 12 como: $ r \mapsto
        \E[X^r]^{1/r}$ está aumentando --- o momento
        (Lyapunov) desigualdade, para ser reutilizada em
        \cref{ch:b2:randomvar}.
  \item Resolva com ferramentas nomeadas, em duas linhas cada: (i) para
        positivo $ a, b, c $: $ a^3 + b^3 + c^3 \geq
        \frac{(a+b+c)^3}{9}$; (ii) para positivo $ x_1, \dots,
        x_n $: $\bigl(\sum_i\sqrt{x_i}\bigr)^2 \leq
        n\sum_i x_i $.
  \item (Síntese) Desenhe a genealogia em cinco frases: acorde
        definição do lema da inclinação; encostas para apoiar linhas para
        Jensen; $\ln $ concavidade de Young para Hölder para
        Minkowski para o $ p $-normas; Jensen para o poder médio
        cadeia em momentos; AM - GM para Carleman. Nomeie os cumes
        (Hölder – Minkowski; Carleman) e indique onde
        caixa de ferramentas tem como título: o $ L^p$ espaços do Ano 3
        volume, cujos axiomas são exatamente as questões 6 e 8.
\end{enumerate}
\end{problem}
