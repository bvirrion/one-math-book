\chapter{Topologia de Espaços Métricos}\label{ch:b2:metric}

A topologia da linha real (volume Ano 1) generaliza, quase
sem mudar uma palavra, para qualquer conjunto equipado com distância. O
o ganho é enorme: sequências de funções, matrizes, curvas --- tudo
tornam-se pontos de espaços métricos, e os três pilares provados aqui
--- completude com o teorema do ponto fixo de Banach, compacidade,
conectividade --- aplique-os uniformemente. Este capítulo é o
espinha dorsal de toda a análise da metade do livro.

\section{Espaços métricos}

\begin{definition}\label{def:b2:metric:def}
UM \emph{espaço métrico}\index{espaço métrico} é um conjunto $X $ com um mapa $ d
\colon X \times X \to \R_+$ tal que, para todos $ x, y, z $:
\[
d(x,y) = 0 \iff x = y,
\qquad
d(x,y) = d(y,x),
\qquad
d(x,z) \leq d(x,y) + d(y,z).
\]
Bolas: $ B(a, r) = \{x : d(a,x) < r\}$ (\omterm{def:b2:metric:topology}{abrir}), $\overline B(a,r) =
\{x : d(a,x) \leq r\}$ (fechado). Um subconjunto $ A \subseteq X $ torna-se um
espaço métrico com a distância induzida.
\end{definition}

\begin{example}\label{ex:b2:metric:examples}
$\R $ com $\abs{x - y}$; $\R^n $ com qualquer um
\[
d_1(x,y) = \sum_i \abs{x_i - y_i},
\quad
d_2(x,y) = \Bigl(\sum_i (x_i - y_i)^2\Bigr)^{1/2},
\quad
d_\infty(x,y) = \max_i \abs{x_i - y_i};
\]
o conjunto $ C(\intcc{a}{b})$ de \omterm{def:b2:metric:continuity}{contínuo} funções com o
\emph{distância do sup} $ d_\infty(f, g) = \sup_{\intcc{a}{b}} \abs{f -
g}$ (finito: $ f - g $ é limitado); qualquer conjunto com o \emph{discreto}
distância ($ d(x,y) = 1$ para $ x \neq y $). Distâncias provenientes das normas
são objeto de \cref{ch:b2:nvs}.
\end{example}

\begin{definition}[Topologia de um espaço métrico]\label{def:b2:metric:topology}
$ U \subseteq X $ é \emph{abrir}\index{abrir}
quando cada ponto de $ U $ é o centro de uma bola contida em $ U $;
$ F $ é \emph{fechado} quando seu complemento está aberto. Bairros,
interior, fechamento, densidade e limite são definidos exatamente como no
linha real (volume do ano 1), com bolas substituindo intervalos, e o
declarações provadas lá --- uniões/interseções de conjuntos abertos,
caracterizações de interior e fechamento, fechamento como menor
superconjunto fechado --- continua com as mesmas provas. As bolas abertas são
bolas abertas e fechadas são fechadas (desigualdade triangular).
\end{definition}

\begin{example}[Interior, fechamento, limite em um conjunto]\label{ex:b2:metric:intclosure}
Em $\R $, deixar $ A = \intoc{0}{1} \cup \{2\}$. Interior:
$\intoo{0}{1}$ --- em torno de qualquer $ x \in \intoo01$ uma pequena bola
permanece em $ A $; em volta $1$, cada bola $\intoo{1-r}{1+r}$ vazamentos
fora de $ A $ à direita, então $1$ não é interior; e o
isolado $2$ também não é interior. Encerramento:
$\intcc{0}{1} \cup \{2\}$ (o ponto $0$ é um limite de $ A $,
nada mais é adicionado). Limite (fechamento menos interior):
$\{0, 1, 2\}$. Observe as assimetrias que vale a pena lembrar: uma
ponto final pode pertencer a um conjunto sem ser interior ($1$), pode
ser aderente sem pertencer ($0$), e um ponto isolado é
seu próprio limite ($2$). A mesma escrituração é executada literalmente em
qualquer \omterm{def:b2:metric:def}{espaço métrico}, com bolas no lugar dos intervalos.
\end{example}

\begin{definition}[Limites, continuidade]\label{def:b2:metric:continuity}
$ x_n \to x $ em $ X $ quando $ d(x_n, x) \to 0$. Um mapa $ f \colon X \to Y $
entre espaços métricos é \emph{contínuo}\index{contínuo} no $ a $ quando
\[
\forall \varepsilon > 0,\ \exists\delta > 0,\quad
d_X(x, a) \leq \delta \implies d_Y\bigl(f(x), f(a)\bigr) \leq
\varepsilon ;
\]
equivalentemente (mesma prova que em $\R $), $ f(x_n) \to f(a)$ para cada
sequência $ x_n \to a $. $ f $ é \emph{Lipschitz}\index{Lipschitz} com constante $ k $ quando
$ d_Y(f(x), f(y)) \leq k\, d_X(x, y)$ sempre --- então uniformemente
contínuo, portanto contínuo.
\end{definition}

\begin{theorem}[Caracterização global da continuidade]\label{thm:b2:metric:globalcontinuity}
$ f \colon X \to Y $ é \omterm{def:b2:metric:continuity}{contínuo} (em todos os pontos) se e somente se o
pré-imagem de cada \omterm{def:b2:metric:topology}{abrir} definir é \omterm{def:b2:metric:topology}{abrir} --- se a pré-imagem de cada
conjunto fechado é fechado.
\end{theorem}

\begin{proof}
($\Rightarrow $) Deixar $ V \subseteq Y $ ser \omterm{def:b2:metric:topology}{abrir} e $ a \in f^{-1}(V)$:
alguma bola $ B(f(a), \varepsilon) \subseteq V $; continuidade no $ a $
fornece $\delta $ com $ f(B(a, \delta)) \subseteq B(f(a),
\varepsilon)$, então $ B(a, \delta) \subseteq f^{-1}(V)$.

($\Leftarrow $) Dado $ a $ e $\varepsilon $: $ f^{-1}\bigl(B(f(a),
\varepsilon)\bigr)$ é \omterm{def:b2:metric:topology}{abrir} e contém $ a $, portanto contém uma bola
$ B(a, \delta)$: essa é a definição de continuidade no $ a $. Fechado
conjuntos: complementos (\cref{prop:b2:structures:images}).
\end{proof}

\section{Espaços completos}

\begin{definition}\label{def:b2:metric:complete}
Uma sequência $(x_n)$ é \emph{Cauchy} quando $\sup_{p, q \geq N} d(x_p,
x_q) \to 0$ como $ N \to \infty $. UM \omterm{def:b2:metric:def}{espaço métrico} é
\emph{completo}\index{completo} quando cada Cauchy
sequência converge. Convergente $\Rightarrow $ Cauchy sempre; fechado
subconjuntos de espaços completos são completos e subconjuntos completos de qualquer
espaço são fechados (mesmas provas como em $\R $: volume do ano 1).
\end{definition}

\begin{example}[Cauchy sem limite]\label{ex:b2:metric:cauchynolimit}
Em $ X = \Q $ com a distância usual, os truncamentos decimais de
$\sqrt2$,
\[
x_0 = 1,\quad x_1 = 1.4,\quad x_2 = 1.41,\quad x_3 = 1.414,
\quad\dots
\]
satisfazer $\abs{x_p - x_q} \leq 10^{-\min(p,q)}$: Cauchy em $\Q $.
Um limite em $\Q $ também seria o limite em $\R $, nomeadamente
$\sqrt2 \notin \Q $: não existe limite em $ X $. A incompletude é
a presença de tais “limites fantasmas”; completude de $\R $
foi projetado no volume do Ano 1 precisamente para dar a cada
Sequência de Cauchy uma casa.
\end{example}

\begin{theorem}\label{thm:b2:metric:rncomplete}
$\R^n $ (qualquer uma das três distâncias de
\cref{ex:b2:metric:examples}) e $\bigl(C(\intcc{a}{b}),
d_\infty\bigr)$ são \omterm{def:b2:metric:complete}{completo}.
\end{theorem}

\begin{proof}
$\R^n $: uma sequência de Cauchy é Cauchy em cada coordenada (cada
$\abs{x_i - y_i} \leq d(x,y)$ para todas as três distâncias), então cada
coordenadas convergem (completude de $\R $, volume do ano 1) e
convergência coordenada implica convergência para $ d_\infty $
(um número finito de coordenadas), portanto, para todos os três (os três
as distâncias dominam umas às outras dentro de fatores constantes: $ d_\infty
\leq d_2 \leq d_1 \leq n\,d_\infty $).

$ C(\intcc{a}{b})$: deixar $(f_n)$ ser $ d_\infty $-Cauchy. Para cada $ x $,
$(f_n(x))$ está Cauchy em $\R $ ($\abs{f_p(x) - f_q(x)} \leq
d_\infty(f_p, f_q)$): converge para alguns $ f(x)$. Passando para o
limite em $\abs{f_p(x) - f_q(x)} \leq \varepsilon $ (válido para $ p, q
\geq N_\varepsilon $, todos $ x $) como $ q \to \infty $: $\abs{f_p(x) -
f(x)} \leq \varepsilon $ para todos $ x $, ou seja\ $ d_\infty(f_p, f) \leq
\varepsilon $: convergência uniforme. O limite é \omterm{def:b2:metric:continuity}{contínuo}: dado
$\varepsilon $, escolha $ p $ com $\sup\abs{f_p - f} \leq \varepsilon $,
então use continuidade de $ f_p $ no $ a $ e a divisão de três mandatos
\[
\abs{f(x) - f(a)} \leq \abs{f(x) - f_p(x)} + \abs{f_p(x) - f_p(a)}
+ \abs{f_p(a) - f(a)} \leq 3\varepsilon
\]
para $ x $ perto de $ a $. (Este ``$3\varepsilon $ argumento'' retorna como
o teorema do limite uniforme de \cref{ch:b2:funcseq}.)
\end{proof}

\begin{example}[Conjuntos abertos e fechados reconhecidos pela continuidade]\label{ex:b2:metric:recognize}
A caracterização global
(\cref{thm:b2:metric:globalcontinuity}) é a ferramenta diária para
escrituração topológica. Em $\R^2$: o conjunto $\{(x, y) : x^2 + y^2
< 1,\ y > x^3\}$ é \omterm{def:b2:metric:topology}{abrir} --- isso é $ g^{-1}(\intoo{-\infty}{1})
\cap h^{-1}(\intoo{0}{+\infty})$ para o \omterm{def:b2:metric:continuity}{contínuo} $ g(x,y) =
x^2 + y^2$ e $ h(x, y) = y - x^3$, uma intersecção de dois \omterm{def:b2:metric:topology}{abrir}
pré-imagens. Em $\bigl(C(\intcc01), d_\infty\bigr)$: o conjunto de
funções com $ f(0) = f(1)$ e $\int_0^1 f = 0$ está fechado ---
a pré-imagem de $\{(0,0)\}$ sob o \omterm{def:b2:metric:continuity}{contínuo} mapa $ f \mapsto
\bigl(f(0) - f(1),\ \int_0^1 f\bigr)$ em $\R^2$ (cada
coordenada é $1$-Lipschitz, como em \cref{exo:b2:metric:3}). O
método nunca faz um desenho: exiba um \omterm{def:b2:metric:continuity}{contínuo} mapa, leia o
definido como uma pré-imagem, cite o teorema.
\end{example}

\begin{example}[Um conjunto fechado definido por infinitas condições]\label{ex:b2:metric:lipclosed}
Em $\bigl(C(\intcc01), d_\infty\bigr)$, o conjunto
\[
L = \{f : \abs{f(x) - f(y)} \leq \abs{x - y}
\ \text{for all } x, y\}
\]
de $1$-As funções \omterm{def:b2:metric:continuity}{Lipschitz} estão fechadas, embora sejam cortadas por
inúmeras condições: para cada par fixo $(x, y)$, o
mapa $ f \mapsto \abs{f(x) - f(y)} - \abs{x - y}$ é \omterm{def:b2:metric:continuity}{contínuo}
(as avaliações são $1$-Lipschitz), então cada condição
define um conjunto fechado, e $ L $ é o \emph{interseção} de
esta família --- uma intersecção arbitrária de conjuntos fechados é
fechado. O mesmo modelo certifica o fechamento para tons monótonos
funções, funções convexas, funções limitadas por um fixo $ g $:
limites uniformes herdam todas as propriedades expressáveis como uma família
de restrições pontuais fechadas. O que os limites uniformes fazem
\emph{não} herdar automaticamente --- diferenciabilidade, por
um --- é exatamente o que \cref{ch:b2:funcseq} deve trabalhar para.
\end{example}

\begin{theorem}[Teorema do ponto fixo de Banach]\label{thm:b2:metric:banach}
Deixar $ X $ ser um não-vazio \omterm{def:b2:metric:def}{espaço métrico} \omterm{def:b2:metric:complete}{completo} e $ f \colon X \to X $ um
\emph{contração}: \omterm{def:b2:metric:continuity}{Lipschitz} com constante $ k < 1$. Então $ f $ tem um
ponto fixo único $\ell $, e cada órbita $ x_{n+1} = f(x_n)$
converge para $\ell $, com\index{Teorema do ponto fixo de Banach}
\[
d(x_n, \ell) \leq \frac{k^n}{1 - k}\, d(x_1, x_0) .
\]
\end{theorem}

\begin{proof}
Singularidade: dois pontos fixos estão distantes $\leq k $ vezes em si.
Existência: $ d(x_{n+1}, x_n) \leq k^n d(x_1, x_0)$ por indução, então
para $ q > p $,
\[
d(x_q, x_p) \leq \sum_{j=p}^{q-1} d(x_{j+1}, x_j)
\leq d(x_1, x_0) \sum_{j \geq p} k^j
= \frac{k^p}{1-k}\, d(x_1, x_0) \xrightarrow[p\to\infty]{} 0 :
\]
Cauchy, portanto convergente para alguns $\ell $; continuidade de $ f $ passes
$ x_{n+1} = f(x_n)$ até o limite: $\ell = f(\ell)$. O erro vinculado
é a estimativa exibida com $ q \to \infty $.
\end{proof}

\begin{example}[Uma equação integral]\label{ex:b2:metric:integralequation}
Sobre $ X = C(\intcc{0}{1})$ (\omterm{def:b2:metric:complete}{completo},
\cref{thm:b2:metric:rncomplete}), considere $ T(f)(x) = 1 + \frac12
\int_0^x f(t)\,\dd t $. Para $ f, g \in X $:
\[
\abs{T(f)(x) - T(g)(x)} \leq \frac12 \int_0^x \abs{f - g} \leq
\frac12\, d_\infty(f, g),
\]
então $ T $ é um $\frac12$-contração: tem um único \omterm{def:b2:metric:continuity}{contínuo} fixo
ponto --- a solução de $ f' = \frac f2$, $ f(0) = 1$, nomeadamente
$\eu^{x/2}$. Este esquema, industrializado, torna-se o
Teorema de \omterm{def:b2:metric:continuity}{Cauchy-Lipschitz} de \cref{ch:b2:diffeq}.
\end{example}

\begin{example}[Um ponto fixo numérico: $ x = \cos x $]\label{ex:b2:metric:cosfix}
No \omterm{def:b2:metric:complete}{completo} $ X = \intcc{0}{1}$, o mapa $ f = \cos $ envia $ X $
em $\intcc{\cos 1}{1} \subseteq X $ e é uma contração: pelo
desigualdade de valor médio,
\[
\abs{\cos x - \cos y} \leq \bigl(\sup_{\intcc01}\abs{\sin}\bigr)
\abs{x - y} = (\sin 1)\abs{x - y},
\qquad \sin 1 \approx 0.841 < 1 .
\]
Banach: uma solução única de $ x = \cos x $ em $\intcc01$ (portanto em
$\R $: qualquer ponto fixo real está em $\intcc{-1}{1}$, então em
$\intcc{\cos 1}{1}$ após uma aplicação) e a iteração
$ x_{n+1} = \cos x_n $ converge para ele de qualquer início: $ x_\infty
\approx 0.739085$, o famoso número obtido martelando o
chave cosseno de uma calculadora. O limite de erro prevê
$(\sin1)^n/(1 - \sin1)$ decadência --- cerca de um dígito por $13$
prensas; o limite a posteriori do fim de semana deste capítulo
problema (questão 14) certifica cada etapa em tempo real.
\end{example}

\section{Compacidade}

\begin{definition}\label{def:b2:metric:compact}
UM \omterm{def:b2:metric:def}{espaço métrico} $ X $ é \emph{compactar}\index{compactar}
quando cada sequência em $ X $ tem uma subsequência convergente \emph{em}
$ X $ (propriedade Bolzano-Weierstrass). Um subconjunto é compacto quando
é assim com a distância induzida.
\end{definition}

\begin{theorem}[Primeiras propriedades]\label{thm:b2:metric:compactprops}
\begin{enumerate}
  \item UM \omterm{def:b2:metric:compact}{compactar} o subconjunto é fechado e limitado; um subconjunto fechado de um
        \omterm{def:b2:metric:compact}{compactar} o espaço é \omterm{def:b2:metric:compact}{compactar}.
  \item Em $\R^n $, o inverso é válido: \omterm{def:b2:metric:compact}{compactar} $\iff $ fechado e
        limitado.
  \item UM \omterm{def:b2:metric:continuity}{contínuo} imagem de um \omterm{def:b2:metric:compact}{compactar} o espaço é \omterm{def:b2:metric:compact}{compactar}; um
        \omterm{def:b2:metric:continuity}{contínuo} função real em um não vazio \omterm{def:b2:metric:compact}{compactar} o espaço é
        limitado e atinge seus limites.
  \item (Heine) A \omterm{def:b2:metric:continuity}{contínuo} mapa em um \omterm{def:b2:metric:compact}{compactar} o espaço é uniformemente
        \omterm{def:b2:metric:continuity}{contínuo}.
  \item Produtos: se $ X, Y $ são \omterm{def:b2:metric:compact}{compactar}, assim é $ X \times Y $ (com
        $ d\bigl((x,y),(x',y')\bigr) = d(x,x') + d(y,y')$).
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Os mesmos argumentos da linha (volume do ano 1): uma fuga para%
sequência infinita ou convergente externa não tem subsequência
convergindo para dentro; para a segunda reivindicação, extrair no ambiente
\omterm{def:b2:metric:compact}{compactar} e use o fechamento.

(2) Sequências limitadas em $\R^n $ tem convergente componente a componente
subsequências: extrair na primeira coordenada
(Bolzano--Weierstrass em $\R $), então, a partir dessa subsequência, no
segundo, e assim por diante ($ n $ extrações sucessivas); o fechamento mantém
o limite interior.

(3) Dado $(f(x_n))$, extrair $ x_{\varphi(n)} \to x \in X $;
continuidade dá $ f(x_{\varphi(n)}) \to f(x) \in f(X)$. Caso real:
compacidade de $ f(X) \subseteq \R $ torna-o fechado e delimitado, e
$\sup f(X) \in f(X)$ (o sup de um conjunto é aderente a ele, e
$ f(X)$ está fechado).

(4) A prova do Ano 1 é transferida literalmente; aqui está, em métrica
vestido. Suponha $ f \colon X \to Y $ \omterm{def:b2:metric:continuity}{contínuo} no \omterm{def:b2:metric:compact}{compactar} $ X $
mas não uniformemente \omterm{def:b2:metric:continuity}{contínuo}: alguns $\varepsilon > 0$ admite, por
cada $ n $, pontos com
\[
d_X(x_n, y_n) \leq \frac{1}{n+1}
\qquad\text{and}\qquad
d_Y\bigl(f(x_n), f(y_n)\bigr) > \varepsilon .
\]
Extrair $ x_{\varphi(n)} \to a \in X $; então $ y_{\varphi(n)} \to
a $ também (as distâncias mútuas tendem a $0$). Continuidade no $ a $
envia ambas as sequências de imagens para $ f(a)$, então
$ d_Y\bigl(f(x_{\varphi(n)}), f(y_{\varphi(n)})\bigr) \to 0$ ---
contradizendo a lacuna uniforme $> \varepsilon $. Compacidade
forneceu exatamente uma coisa: o ponto de cluster $ a $ em que
aplicar simples continuidade.

(5) Extrato no $ X $-coordenadas e, em seguida, novamente no
$ Y $-coordenadas.
\end{proof}

\begin{example}[Teorema de Heine, com e sem compacidade]\label{ex:b2:metric:heinework}
Sobre $\intcc{0}{1}$, a função $ x \mapsto x^2$ é uniformemente
\omterm{def:b2:metric:continuity}{contínuo} --- Heine diz isso sem nenhum cálculo, mas o direto
estimativa é instrutiva:
\[
\abs{x^2 - y^2} = \abs{x + y}\,\abs{x - y} \leq 2\abs{x - y},
\]
então $\delta = \varepsilon/2$ funciona \emph{para cada ponto em
uma vez}. Sobre $\R $ a mesma função não é uniformemente contínua:
com $ x_n = n $ e $ y_n = n + \frac1n $, a lacuna $\abs{x_n - y_n}
= \frac1n \to 0$ enquanto $\abs{x_n^2 - y_n^2} = 2 + \frac1{n^2}
\geq 2$: nenhum único $\delta $ serve $\varepsilon = 1$. O
mecanismo é visível: o local \omterm{def:b2:metric:continuity}{Lipschitz} constante $\abs{x + y}$
está limitado a um \omterm{def:b2:metric:compact}{compactar} e ilimitado em $\R $ ---Heine's
teorema é exatamente a afirmação de que compacidade bonés como
constantes locais uniformemente.
\end{example}

\begin{method}[Provando que um conjunto é compacto]\label{met:b2:metric:compactness}
Três rotas, em ordem de frequência. (1) \emph{Ambiente
reconhecimento:} em $\R^n $ (ou qualquer norma de dimensão finita
espaço, \cref{ch:b2:nvs}), verifique fechado --- normalmente como um
pré-imagem, \cref{ex:b2:metric:recognize} --- e limitado. (2)
\emph{Herança:} um subconjunto fechado de um conhecido \omterm{def:b2:metric:compact}{compactar} o espaço é
\omterm{def:b2:metric:compact}{compactar}; uma união finita ou um produto de compactos é \omterm{def:b2:metric:compact}{compactar}; um
\omterm{def:b2:metric:continuity}{contínuo} imagem de um \omterm{def:b2:metric:compact}{compactar} é \omterm{def:b2:metric:compact}{compactar}. (3) \emph{Nu
mãos:} extrair uma subsequência convergente de um arbitrário
sequência --- geralmente por extrações sucessivas coordenadas por
coordenar. Para provar \emph{não}-compacidade, uma testemunha
é suficiente: uma sequência sem subsequência convergente, na maioria das vezes
pontos a distância mútua $\geq \varepsilon $.
\end{method}

\begin{example}[Distâncias entre conjuntos: a compacidade ganha seu sustento]\label{ex:b2:metric:setdistance}
Deixar $ K $ ser \omterm{def:b2:metric:compact}{compactar}, $ F $ fechado, $ K \cap F = \emptyset $ em um
\omterm{def:b2:metric:def}{espaço métrico}. Então
\[
d(K, F) = \inf\,\{d(x, y) : x \in K,\ y \in F\} > 0 :
\]
a função $ x \mapsto d(x, F)$ é \omterm{def:b2:metric:continuity}{contínuo}
(\cref{exo:b2:metric:11}) e positivo em $ K $ ($ d(x, F) = 0$
colocaria $ x \in \overline F = F $), então atinge um valor positivo
mínimo no \omterm{def:b2:metric:compact}{compactar} $ K $
(\cref{thm:b2:metric:compactprops}~(3)). Compacidade não é
decorativo: para dois \emph{fechado} define o ínfimo pode desaparecer
sem ser alcançado --- em $\R^2$, a hipérbole $ F_1 = \{xy
= 1\}$ e o eixo $ F_2 = \{y = 0\}$ são conjuntos fechados disjuntos
com $ d(F_1, F_2) = 0$ (os pontos $(n, \frac1n)$ aproximar-se do
eixo). Fugir para o infinito é exatamente o que compacidade proíbe.
\end{example}

\begin{theorem}[Borel-Lebesgue]\label{thm:b2:metric:borellebesgue}
UM \omterm{def:b2:metric:def}{espaço métrico} $ X $ é \omterm{def:b2:metric:compact}{compactar} se e somente se cada capa de $ X $ por
\omterm{def:b2:metric:topology}{abrir} conjuntos tem um \emph{finito} subcapa.\index{Borel-Lebesgue
teorema}
\end{theorem}

\begin{proof}
($\Leftarrow $) Suponha $(x_n)$ não tem subsequência convergente. Nós
reivindicar cada $ x \in X $ tem uma bola $ B(x, r_x)$ contendo $ x_n $
para apenas um número finito de índices $ n $: caso contrário, cada bola
$ B(x, \frac1{k+1})$ conteria um número infinito de termos, e
escolhendo índices
\[
\varphi(0) < \varphi(1) < \varphi(2) < \cdots
\quad\text{with}\quad
x_{\varphi(k)} \in B\Bigl(x, \frac{1}{k+1}\Bigr)
\]
(possível em cada etapa precisamente porque existem infinitas
permanecem candidatos) construiria uma subsequência convergindo para $ x $.
As bolas $ B(x, r_x)$ cobrir $ X $; se um número finito deles
coberto $ X $, o conjunto de índices $\N $ seria uma união finita de
conjuntos finitos: absurdo.

($\Rightarrow $) Duas etapas. \emph{Número de Lebesgue:} por um \omterm{def:b2:metric:topology}{abrir} cobrir
$(U_i)$ de um \omterm{def:b2:metric:compact}{compactar} $ X $, há $\rho > 0$ tal que cada bola
de raio $\rho $ reside em alguns $ U_i $. Caso contrário, para cada $ n $ escolha
$ x_n $ com $ B(x_n, \frac{1}{n+1})$ em não $ U_i $; extrair
$ x_{\varphi(n)} \to x \in U_{i_0} \supseteq B(x, r)$; para grande $ n $,
$ B(x_{\varphi(n)}, \frac{1}{\varphi(n)+1}) \subseteq B(x, r)
\subseteq U_{i_0}$: contradição. \emph{Limitação total:} para
cada $\rho > 0$, um número finito de bolas de raio $\rho $ cobrir $ X $.
Caso contrário, escolha indutivamente $ x_{n+1}$ fora $ B(x_0, \rho) \cup
\dots \cup B(x_n, \rho)$: a sequência tem distâncias aos pares $\geq
\rho $, então não há Cauchy --- portanto, não há subsequência convergente ---:
contradição. Combinando: capa $ X $ por um número finito de bolas de raio
$\rho $ (o número de Lebesgue), cada um dentro de algum $ U_i $: um finito
subcapa.
\end{proof}

\begin{example}[Um $\varepsilon $-líquido, contado]\label{ex:b2:metric:epsnet}
Limitação total (da prova de
\cref{thm:b2:metric:borellebesgue}) é muito concreto
$\intcc{0}{1}$: para $\varepsilon > 0$, o $\lceil
\frac{1}{2\varepsilon}\rceil $ bolas centradas em $\varepsilon,
3\varepsilon, 5\varepsilon, \dots $ de raio $\varepsilon $
cubra-o --- sobre $\frac1{2\varepsilon}$ bolas e sem cobertura
pode fazer com menos de $\frac{1}{2\varepsilon}$ deles (cada
a bola cobre o comprimento no máximo $2\varepsilon $). Em $\intcc01^2$
a contagem de quadrados em ordem $\varepsilon^{-2}$: cobrindo números
crescer como $\varepsilon^{-d}$ em dimensão $ d $ --- um quantitativo
cara de compacidade, e a razão pela qual a dimensão infinita
bolas unitárias de \cref{ch:b2:nvs} (onde não há finito
$\frac13$-net existe) não pode ser \omterm{def:b2:metric:compact}{compactar}.
\end{example}

\begin{example}[Leitura compacta nas capas]\label{ex:b2:metric:covers}
O meio aberto intervalo $\intoc{0}{1}$ é abrangido pelo \omterm{def:b2:metric:topology}{abrir} conjuntos
$ U_n = \intoo{\frac1n}{2}$, $ n \geq 1$; qualquer subfamília finita tem um
maior índice $ N $ e sente falta $\intoc{0}{\frac1N}$: não finito
subcapa, então $\intoc{0}{1}$ não é \omterm{def:b2:metric:compact}{compactar} --- qual o
definição sequencial vê através $ x_n = \frac1n $, cujo limite
$0$ escapa. Por outro lado, adicionando o ponto único $0$
repara ambos os diagnósticos de uma só vez: em $\intcc{0}{1}$ cada uma dessas capas
deve conter um conjunto contendo $0$, que engole toda uma inicial
segmento, e um número finito de conjuntos termina o resto. As duas línguas
de \cref{thm:b2:metric:borellebesgue} sempre falha ou tem sucesso
juntos --- abrange detectar escape exatamente onde as sequências o fazem.
\end{example}

\begin{remark}[Perspectivas dentro deste volume]\label{rem:b2:metric:perspectives}
Este capítulo é a parede de suporte do volume; observe onde
cada pilar carrega peso. \emph{Completude}: o Cauchy
critério torna-se o teste de convergência para séries em Banach
espaços (\cref{ch:b2:series}), convergência uniforme em
\cref{ch:b2:funcseq} é exatamente a convergência no \omterm{def:b2:metric:complete}{completo}
$\bigl(C, d_\infty\bigr)$ e Cauchy--Lipschitz
(\cref{ch:b2:diffeq}) é o teorema do ponto fixo de Banach usando
uma equação integral. \emph{Compacidade}: isso prova o
equivalência de normas (\cref{ch:b2:nvs}), a obtenção de
extremos para a otimização de \cref{ch:b2:diffcalc}, e o
existência de melhores aproximações (\cref{ch:b2:nvs}fim de semana
problema). \emph{Conectividade}: globaliza declarações locais
--- unicidade de soluções de equações diferenciais, o
teorema do valor intermediário nas curvas (\cref{ch:b2:curves}) e
os dois componentes $ GL_n(\R)$ que a teoria da orientação
(\cref{ch:b2:multint}) se manterão separados.
\end{remark}

\begin{remark}[Armadilhas comuns]\label{rem:b2:metric:pitfalls}
(i) ``Fechado e limitado implica \omterm{def:b2:metric:compact}{compactar}'' é um teorema sobre
$\R^n $, não sobre espaços métricos: um conjunto infinito com o
métrica discreta é fechada e limitada em si mesma, mas não \omterm{def:b2:metric:compact}{compactar}
(\cref{exo:b2:metric:4}), e a bola unitária fechada de
$ C(\intcc01)$ falha também (\cref{ch:b2:nvs}). (ii) Completude é
uma propriedade do \emph{distância}, não da topologia: $\R $
com $ d(x,y) = \abs{\arctan x - \arctan y}$ tem o habitual
sequências convergentes, mas está incompleta
(\cref{exo:b2:metric:1}). (iii) Um \omterm{def:b2:metric:continuity}{contínuo} bijeção não precisa
ser um homeomorfismo --- a parametrização circular de
\cref{exo:b2:metric:7}; compacidade da fonte o repara.
(iv) O teorema de Banach precisa $ k < 1$ \emph{uniformemente}: o
condição $ d(f(x), f(y)) < d(x,y)$ por si só não garante nada em um
não compacto espaço (\cref{exo:b2:metric:5}). (v) \omterm{def:b2:metric:connected}{Conectado} faz
não implica \omterm{def:b2:metric:connected}{conectado} por caminho em geral --- mas para o \omterm{def:b2:metric:topology}{abrir}
subconjuntos de espaços normados encontrados neste livro, os dois concordam
(\cref{ch:b2:nvs}).
\end{remark}

\begin{remark}[Onde este capítulo é usado]
Em todos os lugares da metade da análise. Completude de
$ C(\intcc{a}{b})$ alimenta os teoremas de convergência de
\cref{ch:b2:funcseq} e a teoria de \omterm{def:b2:metric:continuity}{Cauchy-Lipschitz} de
\cref{ch:b2:diffeq} (o problema do fim de semana deste capítulo prova a
Picard local - Lindeldo teorema já); compacidade dá o
equivalência de normas em dimensão finita (\cref{ch:b2:nvs}) e
a existência de extremos em \cref{ch:b2:diffcalc}; conectividade
está subjacente aos argumentos de valor intermediário de \cref{ch:b2:realfun}
e a globalização da exclusividade para equações diferenciais. Em
o volume do Ano 3, compacidade em espaços funcionais (o
Arzelà - teorema de Ascoli) e o teorema da categoria de Baire
(\cref{exo:b2:metric:12} aqui) tornam-se ferramentas de uso diário.
\end{remark}

\section{Conectividade}

\begin{definition}\label{def:b2:metric:connected}
$ X $ é \emph{conectado}\index{conectado} quando admite não
partição em dois não vazios \omterm{def:b2:metric:topology}{abrir} subconjuntos --- equivalentemente, quando é
apenas subconjuntos ambos \omterm{def:b2:metric:topology}{abrir} e fechados são $\emptyset $ e $ X $. $ X $ é
\emph{conectado por caminho} quando quaisquer dois pontos são unidos por um \omterm{def:b2:metric:continuity}{contínuo}
mapa $\gamma \colon \intcc{0}{1} \to X $.
\end{definition}

\begin{theorem}\label{thm:b2:metric:connectedness}
\begin{enumerate}
  \item O \omterm{def:b2:metric:connected}{conectado} subconjuntos de $\R $ são exatamente os intervalos.
  \item UM \omterm{def:b2:metric:continuity}{contínuo} imagem de um espaço \omterm{def:b2:metric:connected}{conectado} é \omterm{def:b2:metric:connected}{conectado} ---
        daí o teorema geral do valor intermediário: a \omterm{def:b2:metric:continuity}{contínuo}
        função real em um espaço \omterm{def:b2:metric:connected}{conectado} toma todos os valores entre
        quaisquer dois de seus valores.
  \item \omterm{def:b2:metric:connected}{Conectado} por caminho $\Rightarrow $ \omterm{def:b2:metric:connected}{conectado}. (O inverso falha
        em geral; vale para \omterm{def:b2:metric:topology}{abrir} subconjuntos de espaços normados,
        \cref{ch:b2:nvs}.)
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Um não intervalo $ A $ sente falta de alguns $ z $ entre dois de seus pontos:
$ A = (A \cap \intoo{-\infty}{z}) \cup (A \cap \intoo{z}{+\infty})$
divide-o em dois não vazios \omterm{def:b2:metric:topology}{abrir} (em $ A $) peças. Por outro lado, deixe
$ I $ seja um intervalo e $ I = U \cup V $ uma partição em não vazia
relativamente \omterm{def:b2:metric:topology}{abrir} conjuntos; escolha $ a \in U $, $ b \in V $, dizer $ a < b $ e
definir $ s = \sup\,(U \cap \intcc{a}{b})$, um ponto de $\intcc{a}{b}
\subseteq I $. Se $ s \in U $: então $ s \neq b $ e abertura relativa
de $ U $ coloca um intervalo inteiro em torno $ s $ (intersectado com $ I $)
dentro $ U $ --- então pontos de $ U \cap \intcc{a}{b}$ exceder $ s $,
contradizendo o supremo. Se $ s \in V $: abertura relativa de $ V $
coloca um intervalo $\intoo{s - r}{s + r} \cap I $ dentro $ V $; mas o
supremo é aderente a $ U \cap \intcc{a}{b}$, que deve atender
intervalo --- contradição com $ U \cap V = \emptyset $. (Isso é
o argumento clopen do Ano 1 para $\R $, corra para dentro $ I $.)

(2) Se $ f(X) = U' \cup V'$ se divide em relativamente não vazio \omterm{def:b2:metric:topology}{abrir}
conjuntos, então $ X = f^{-1}(U') \cup f^{-1}(V')$ divisões $ X $
(\cref{thm:b2:metric:globalcontinuity}). TIV: $ f(X) \subseteq \R $
é \omterm{def:b2:metric:connected}{conectado}, portanto, um intervalo por (1).

(3) Suponha $ X = U \cup V $, ambos não vazios abrire junte-se $ a \in U $
para $ b \in V $ por um caminho $\gamma $: então $\gamma^{-1}(U),
\gamma^{-1}(V)$ dividir $\intcc{0}{1}$, contradizendo (1).
\end{proof}

\begin{example}\label{ex:b2:metric:glnr}
$ GL_n(\R)$ não é \omterm{def:b2:metric:connected}{conectado}: $\det $ é \omterm{def:b2:metric:continuity}{contínuo} (um polinômio em
as entradas) em $\R^*$, o que não é \omterm{def:b2:metric:connected}{conectado}; as pré-imagens de
$\R_+^*$ e $\R_-^*$ dividir $ GL_n(\R)$. (Cada peça é na verdade
\omterm{def:b2:metric:connected}{conectado} por caminho --- um exercício agradável além de nossas necessidades.) Por
contraste $ GL_n(\C)$ \emph{é} \omterm{def:b2:metric:connected}{conectado} por caminho:
\cref{exo:b2:metric:10}.
\end{example}

\begin{example}[Um ponto fixo apenas pela conexão]\label{ex:b2:metric:ivtfixed}
Todo \omterm{def:b2:metric:continuity}{contínuo} $ f \colon \intcc01 \to \intcc01$ tem um fixo
ponto --- sem hipótese de contração, sem iteração. Considerar
$ g(x) = f(x) - x $, \omterm{def:b2:metric:continuity}{contínuo} no \omterm{def:b2:metric:connected}{conectado} $\intcc01$:
\[
g(0) = f(0) \geq 0,
\qquad
g(1) = f(1) - 1 \leq 0 ,
\]
e o teorema do valor intermediário
(\cref{thm:b2:metric:connectedness}~(2)) fornece um zero de
$ g $, ou seja,\ um ponto fixo de $ f $. Contraste com Banach
(\cref{thm:b2:metric:banach}): aqui a existência é topológica
e gratuito, mas a exclusividade e o algoritmo são perdidos --- $ f =
\mathrm{id}$ tem todos os pontos fixos e a iteração de um
não contratante $ f $ pode circular para sempre. Os dois pontos fixos
teoremas deste capítulo respondem a diferentes questões com
moedas diferentes.
\end{example}

\begin{example}[$\R $ e $\R^2$ não são homeomórficos]\label{ex:b2:metric:rnotr2}
Conectividade é uma impressão digital topológica. Suponha $ h \colon
\R^2 \to \R $ eram um homeomorfismo (um \omterm{def:b2:metric:continuity}{contínuo} bijeção com
\omterm{def:b2:metric:continuity}{contínuo} inverso). Remover um ponto $ a \in \R^2$: o
restrição $ h \colon \R^2\setminus\{a\} \to
\R\setminus\{h(a)\}$ ainda é um homeomorfismo. Mas $\R^2$ menos
um ponto é \omterm{def:b2:metric:connected}{conectado} por caminho --- juntar quaisquer dois pontos por um segmento,
desviar ao longo de um segundo segmento através de um ponto auxiliar se
$ a $ bloqueia o direto --- portanto \omterm{def:b2:metric:connected}{conectado}
(\cref{thm:b2:metric:connectedness}~(3)); enquanto $\R $ menos um
ponto se divide em dois não vazios \omterm{def:b2:metric:topology}{abrir} meias linhas: não \omterm{def:b2:metric:connected}{conectado}.
Conectividade é preservado por \omterm{def:b2:metric:continuity}{contínuo} mapas: contradição.
O avião e a linha são genuinamente diferentes \emph{como
espaços topológicos} --- um fato de que apenas a cardinalidade
(\cref{exo:b2:structures:3}bijeções de estilo existem!) é muito
grosseiro de ver.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b2:metric:1}
Sobre $\R $, verifique isso $\delta(x, y) = \min(1, \abs{x - y})$ e
$ d(x,y) = \abs{\arctan x - \arctan y}$ são distâncias. Qual
sequências convergem para cada uma? É $(\R, d)$ \omterm{def:b2:metric:complete}{completo}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:metric:2}
Em um \omterm{def:b2:metric:def}{espaço métrico}, prove que uma sequência convergente é Cauchy e
limitado, e que uma sequência de Cauchy com uma subsequência convergente
converge. Deduza novamente que espaços métricos compactos são \omterm{def:b2:metric:complete}{completo}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:metric:3}
Em $\bigl(C(\intcc{0}{1}), d_\infty\bigr)$, calcule a distância
entre $ f(x) = x $ e $ g(x) = x^2$; descreva a bola fechada
$\overline B(0, 1)$; e prove que o conjunto $\{f : f(0) = 0\}$ é
fechado enquanto $\{f : f(0) > 0\}$ é \omterm{def:b2:metric:topology}{abrir}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:metric:4}
Prove que o discreto \omterm{def:b2:metric:def}{espaço métrico} $ X $ (qualquer conjunto) é completoe
que é \omterm{def:b2:metric:compact}{compactar} se e somente se $ X $ é finito. Quais subconjuntos são
\omterm{def:b2:metric:connected}{conectado}?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:metric:5}
Deixar $ X $ ser \omterm{def:b2:metric:compact}{compactar} e $ f \colon X \to X $ com
\[
d\bigl(f(x), f(y)\bigr) < d(x, y) \quad \text{for all } x \neq y .
\]
Prove isso $ f $ tem um ponto fixo único \emph{(minimizar $ x \mapsto
d(x, f(x))$)}, e dê um exemplo em $ X = \intco{1}{+\infty}$
(não \omterm{def:b2:metric:compact}{compactar}) sem ponto fixo.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:metric:6}
(Aninhado compactos) Deixar $(K_n)$ ser uma sequência decrescente de não vazia
\omterm{def:b2:metric:compact}{compactar} subconjuntos de um \omterm{def:b2:metric:def}{espaço métrico}. Prove isso $\bigcap_n K_n \neq
\emptyset $ \emph{(escolha $ x_n \in K_n $ e extrair)}. Mostrar por exemplo
aquele aninhado não vazio \emph{fechado} se instala $\R $ pode ficar vazio
interseção.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:metric:7}
Deixar $ K $ ser \omterm{def:b2:metric:compact}{compactar} e $ f \colon K \to Y $ \omterm{def:b2:metric:continuity}{contínuo} e bijetivo.
Prove isso $ f^{-1}$ é \omterm{def:b2:metric:continuity}{contínuo} \emph{(use conjuntos fechados:
\cref{thm:b2:metric:globalcontinuity} e
\cref{thm:b2:metric:compactprops})}. Dê um contraexemplo sem
compacidade ($\gamma(t) = (\cos t, \sin t)$ sobre
$\intco{0}{2\pi}$).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:metric:8}
O \emph{Conjunto de cantores} $ C $ é obtido de $\intcc{0}{1}$ por
excluindo repetidamente \omterm{def:b2:metric:topology}{abrir} terços médios. Prove isso $ C $ é \omterm{def:b2:metric:compact}{compactar},
tem interior vazio e é infinito --- de fato \omterm{def:b2:structures:countable}{equipotente} para
$\{0,1\}^{\N}$ \emph{(expansões ternárias com dígitos $0,2$;
\cref{exo:b2:structures:3})}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:metric:9}
Deixar $ X $ seja um \emph{\omterm{def:b2:metric:compact}{compactar}} \omterm{def:b2:metric:def}{espaço métrico} e $ f \colon X \to X $ um
isometria: $ d(f(x), f(y)) = d(x, y)$. Prove isso $ f $ é sobrejetivo.
\emph{Dica: se $ a \notin f(X)$, então $\varepsilon = d(a, f(X)) > 0$
(por que?); estudar a órbita $ a, f(a), f^2(a), \dots $ e mostrar seu
os pontos são pares $\geq \varepsilon $ à parte --- contradição com
compacidade.}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:metric:10}
Prove isso $ GL_n(\C)$ é \omterm{def:b2:metric:connected}{conectado} por caminho. \emph{Dica: dada $ A, B $
invertível, considere $ p(z) = \det\bigl((1 - z)A + zB\bigr)$ para $ z
\in \C $: um polinômio em $ z $, não identicamente zero, então tem
finitas raízes; juntar $0$ para $1$ em $\C $ por um caminho evitando
eles.}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:metric:11}
Para $\emptyset \neq A \subseteq X $, definir $ d(x, A) = \inf_{a \in A}
d(x, a)$. Prove isso $ x \mapsto d(x, A)$ é $1$-Lipschitz, isso
$ d(x, A) = 0$ se $ x \in \overline A $, e isso para disjunto
não vazio \emph{fechado} conjuntos $ A, B $ a função
\[
\varphi(x) = \frac{d(x, A)}{d(x, A) + d(x, B)}
\]
está bem definido, \omterm{def:b2:metric:continuity}{contínuo}, igual a $0$ exatamente em $ A $ e para
$1$ exatamente em $ B $ --- um \omterm{def:b2:metric:continuity}{contínuo} ``switch'' separando quaisquer dois
conjuntos fechados disjuntos.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:metric:12}
(Baire) Deixe $ X $ seja um \omterm{def:b2:metric:def}{espaço métrico} \omterm{def:b2:metric:complete}{completo} e $(U_n)_{n\geq1}$ um
sequência de denso \omterm{def:b2:metric:topology}{abrir} subconjuntos. Prove isso $\bigcap_n U_n $ é
denso em $ X $ \emph{(dentro de qualquer bola, construa bolas fechadas aninhadas
$\overline B(x_n, r_n) \subseteq U_n $ com $ r_n \to 0$ e usar
completude)}. Deduza isso $\R $ não é um \omterm{def:b2:structures:countable}{contável} união de
conjuntos fechados com interior vazio, e recuperar --- novamente --- isso
$\R $ é incontável.
\end{exercise}

\section{Problema: Iteração Picard}

Completude mais a contração é uma máquina de resolver: alimente-a com um
equação escrita como um problema de ponto fixo e retorna
existência, exclusividade, um algoritmo e barras de erro. Este fim de semana
problema faz a máquina funcionar com potência total na equação $ y' = f(t,
y)$: provamos o local \emph{Picard-Lindeldo teorema} (o
coração não linear de \cref{ch:b2:diffeq}Cauchy--Lipschitz
teoria), observe cada hipótese ganhar seu sustento através
contra-exemplos e coletar dividendos puramente métricos ---
\omterm{def:b2:metric:continuity}{contínuo} dependência dos dados, da equação de Kepler e do
auto-similaridade do conjunto de Cantor.

\begin{problem}[{Problema de fim de semana --- o Picard--Lindelde
teorema}]
\label{pb:b2:metric:1}
Por todo, $ t_0 \in \R $, $ y_0 \in \R $, $ a, b > 0$, e $ f $ é um
\omterm{def:b2:metric:continuity}{contínuo} função no retângulo $ R = \intcc{t_0 - a}{t_0 +
a} \times \intcc{y_0 - b}{y_0 + b}$, delimitado por $ M =
\sup_R\,\abs f $, e \emph{$ L $-Lipschitz em sua segunda variável}:
$\abs{f(t, y) - f(t, z)} \leq L\abs{y - z}$ sempre que ambos os pontos
deitar $ R $. Definir
\[
h = \min\Bigl(a, \frac bM\Bigr) \quad (\text{with } h = a
\text{ if } M = 0), \qquad I = \intcc{t_0 - h}{t_0 + h}.
\]

\medskip
\noindent\textbf{Part I --- The \omterm{def:b2:metric:complete}{completo} stage.}
\begin{enumerate}
  \item Prove as duas afirmações citadas em
        \cref{def:b2:metric:complete}: um subconjunto fechado de um
        \omterm{def:b2:metric:def}{espaço métrico} \omterm{def:b2:metric:complete}{completo} é \omterm{def:b2:metric:complete}{completo}, e um \omterm{def:b2:metric:complete}{completo} subconjunto
        de qualquer \omterm{def:b2:metric:def}{espaço métrico} está fechado. Deduza que todo fechado
        subconjunto de $\bigl(C(I), d_\infty\bigr)$ é um \omterm{def:b2:metric:complete}{completo}
        \omterm{def:b2:metric:def}{espaço métrico}.
  \item Mostre que o mapa $ C(I) \to C(I)$, $ y \mapsto \bigl(t
        \mapsto \int_{t_0}^{t}y(s)\,\dd s\bigr)$, é
        $ h $-Lipschitz para $ d_\infty $.
  \item (Os pontos fixos se movem menos que os mapas) Deixe $ g \colon X
        \to X $ seja um $ k $-contração de um \omterm{def:b2:metric:def}{espaço métrico} com fixo
        ponto $\ell_g $, e $\widetilde g \colon X \to X $
        \emph{qualquer} mapa com um ponto fixo $\ell_{\widetilde g}$.
        Provar
        \[
        d(\ell_g, \ell_{\widetilde g}) \leq
        \frac{d\bigl(g(\ell_{\widetilde g}),
        \widetilde g(\ell_{\widetilde g})\bigr)}{1 - k}
        \leq \frac{\sup_{x \in X} d\bigl(g(x), \widetilde
        g(x)\bigr)}{1 - k}.
        \]
  \item (O truque da iteração) Vamos $ X $ ser \omterm{def:b2:metric:complete}{completo} não vazio e $ g
        \colon X \to X $ um mapa --- não assumido \omterm{def:b2:metric:continuity}{contínuo} ---
        de modo que alguns iterem $ g^m $ é um $ k $-contração.
        Prove isso $ g $ tem um ponto fixo único $\ell $ e isso
        \emph{todo} órbita $ x_{n+1} = g(x_n)$ converge para
        $\ell $. \emph{(Pontos fixos de $ g $ são pontos fixos de
        $ g^m $; por outro lado $ g(\ell)$ é um ponto fixo de $ g^m $;
        dividir a órbita ao longo do mod de resíduos $ m $.)}
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The Picard--Lindelöf theorem.}
\begin{enumerate}[resume]
  \item Mostre que uma função $ y \colon I \to \intcc{y_0 - b}{y_0
        + b}$ é $ C^1$ com $ y(t_0) = y_0$ e $ y' = f(t, y)$ sobre
        $ I $ se e somente se for \omterm{def:b2:metric:continuity}{contínuo} e satisfaz o
        equação integral
        \[
        y(t) = y_0 + \int_{t_0}^{t} f\bigl(s, y(s)\bigr)\dd s
        \qquad (t \in I).
        \]
  \item Deixar $ X_h = \{y \in C(I) : \abs{y(t) - y_0} \leq b
        \text{ on } I\}$ e deixe $ T $ ser definido por $ T(y)(t) =
        y_0 + \int_{t_0}^{t}f(s, y(s))\dd s $. Mostre isso $ X_h $ é
        um subconjunto fechado não vazio de $ C(I)$, por isso completoe
        isso $ T $ mapas $ X_h $ em $ X_h $ --- é aqui $ h \leq
        b/M $ funciona.
  \item Mostre isso $ d_\infty\bigl(T(y), T(z)\bigr) \leq
        Lh\,d_\infty(y, z)$ sobre $ X_h $: se $ Lh < 1$, Banach
        teorema já conclui. Nós removemos essa pequenez
        condição a seguir.
  \item Prove por indução em $ n $:
        \[
        \abs{T^n(y)(t) - T^n(z)(t)} \leq
        \frac{\bigl(L\abs{t - t_0}\bigr)^n}{n!}\,
        d_\infty(y, z) \qquad (y, z \in X_h,\ t \in I),
        \]
        então alguma iteração de $ T $ é uma contração. Conclua com
        questão 4 (\emph{o Picard - Lindeldo teorema}): o
        Problema de Cauchy $ y' = f(t,y)$, $ y(t_0) = y_0$ tem exatamente
        uma solução em $ I = \intcc{t_0 - h}{t_0 + h}$ com
        valores em $\intcc{y_0 - b}{y_0 + b}$.
  \item Mostre que a restrição ``com valores em $\intcc{y_0 -
        b}{y_0 + b}$'' é automático: qualquer solução do Cauchy
        problema definido em $ I $ cujo gráfico começa em $ R $ fica
        em $\intcc{y_0 - b}{y_0 + b}$ \emph{(considere o primeiro
        hora de saída e limite $\abs{y(t) - y_0}$ por $ M\abs{t -
        t_0}$)}. Portanto, a exclusividade é válida entre todas as soluções em
        $ I $.
  \item Ligue a máquina $ y' = y $, $ y(0) = 1$, a partir de
        a constante $ y^{(0)} \equiv 1$: calcule o Picard
        itera $ y^{(n)}$, identifique-os e descreva os
        convergência.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Every hypothesis earns its keep.}
\begin{enumerate}[resume]
  \item (\omterm{def:b2:metric:continuity}{Lipschitz} falha, a exclusividade falha) Para $ y' =
        2\sqrt{\abs y}$, $ y(0) = 0$: verifique isso $ y \equiv 0$
        e, para cada $ c \geq 0$, a função $ y_c(t) = 0$ para
        $ t \leq c $, $ y_c(t) = (t - c)^2$ para $ t > c $, são todos
        $ C^1$ soluções em $\R $. Onde exatamente $ y \mapsto
        2\sqrt{\abs y}$ deixar de ser \omterm{def:b2:metric:continuity}{Lipschitz}?
  \item (Localidade é real) Para $ y' = y^2$, $ y(0) = 1$: resolver
        explicitamente, dê o intervalo máximo de existência, e
        calcular o melhor $ h $ o teorema pode certificar acima de tudo
        escolhas do retângulo ($ a $ grande, $ b $ grátis): mostrar
        $ h_{\max} = \sup_{b>0} \frac{b}{(1+b)^2} = \frac14$,
        enquanto a verdadeira solução vive $\intoo{-\infty}{1}$.
  \item (Completude não é decoração) Ligado $ X = \Q \cap
        \intcc{1}{2}$ com a distância habitual, deixe $ g(x) =
        \frac x2 + \frac1x $. Mostrar $ g(X) \subseteq X $, que $ g $
        é um $\frac12$-contração \emph{(valor médio
        desigualdade)}, e isso $ g $ não tem ponto fixo $ X $.
        Qual hipótese do teorema de Banach falha e o que é
        o ponto fixo na conclusão?
  \item (Barras de erro) Para um $ k $-contração $ g $ em um \omterm{def:b2:metric:complete}{completo}
        espaço, prove o \emph{a posteriori} estimativa $ d(x_n,
        \ell) \leq \frac{k}{1-k}\,d(x_n, x_{n-1})$. Para a Garça
        mapa $ g(x) = \frac x2 + \frac 1x $ sobre $\intcc{1}{2}$
        (ponto fixo $\sqrt2$), começando em $ x_0 = \frac32$:
        quantos passos o \emph{a priori} vinculado
        $\frac{k^n}{1-k}d(x_1, x_0)$ demanda por precisão
        $10^{-6}$, e quantas etapas são suficientes na realidade?
        (Calcular $ x_1, x_2, x_3$ e seus erros; o
        limite de contração é honesto, mas pessimista ---
        Heron converge quadraticamente.)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- \omterm{def:b2:metric:continuity}{Contínuo} dependence.} Neste
parte $ Lh < 1$, então $ T $ em si é uma contração $ X_h $ (pergunta
7); escrever $ y[\,y_0\,]$ para a solução com valor inicial
$ y_0$.
\begin{enumerate}[resume]
  \item (Dependência do valor inicial) Deixe $ z_0$ seja outro
        valor inicial com $\abs{z_0 - y_0}$ pequeno o suficiente para que
        ambos os problemas cabem no retângulo. Usando a questão 3,
        provar
        \[
        d_\infty\bigl(y[y_0], y[z_0]\bigr) \leq
        \frac{\abs{y_0 - z_0}}{1 - Lh}.
        \]
  \item (Longos intervalos por encadeamento) Suponha que as soluções existam
        em um longo segmento cortado em $ m $ peças consecutivas em
        cada um dos quais o limite anterior se aplica com $ Lh \leq
        \frac12$. Mostre que o desvio cresce no máximo por um fator
        $2$ por peça, portanto $ d_\infty \leq 2^m\abs{y_0 - z_0}$
        geral --- um limite de comprimento exponencial, o discreto
        sombra do $\eu^{L\abs{t - t_0}}$ do lema de Gronwall
        (\cref{ch:b2:diffeq}).
  \item (Dependência do campo) Deixe $ g $ ser outro campo em
        $ R $, também $ L $-Lipschitz em $ y $, com $\sup_R \abs{f - g}
        \leq \varepsilon $. Prove que o correspondente
        soluções satisfazem $ d_\infty \leq
        \frac{\varepsilon h}{1 - Lh}$: erro de modelagem
        se propaga linearmente.
  \item (Parâmetros) Se uma família $ f_\lambda $ de campos é
        $ L $-Lipschitz em $ y $ uniformemente e $\sup_R\abs{f_\lambda
        - f_\mu} \leq C\abs{\lambda - \mu}$, deduza que
        $\lambda \mapsto y_\lambda $ é \omterm{def:b2:metric:continuity}{Lipschitz} do
        espaço de parâmetros em $\bigl(C(I), d_\infty\bigr)$.
  \item (Os sistemas não custam nada) Explique por que as Partes I, II e IV
        segure literalmente por $ y $ com valores em $\R^n $ (e aí
        distâncias construídas em qualquer uma das distâncias de
        \cref{ex:b2:metric:examples}), então calcule todos os Picard
        itera para o sistema $ y' = Ay $, $ y(0) = (c_1,
        c_2)$, $ A = \begin{psmallmatrix} 0 & 1\\ 0 &
        0\end{psmallmatrix}$: mostra que a iteração se torna
        estacionário na solução exata após uma etapa.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Metric dividends and synthesis.}
\begin{enumerate}[resume]
  \item (Perturbação da identidade) Deixe $ X $ seja um \omterm{def:b2:metric:complete}{completo}
        estágio semelhante ao espaço normado: pegue $ X = C(I)$ ou $\R^n $. Se
        $\eta \colon X \to X $ é $ k $-Lipschitz com $ k < 1$,
        provar isso $ x \mapsto x + \eta(x)$ é uma bijeção de $ X $
        cujo inverso é $\frac1{1-k}$-Lipschitz \emph{(para cada
        $ y $, aplique Banach a $ x \mapsto y - \eta(x)$)}. Isto é
        o coração métrico do teorema da função inversa
        (\cref{ch:b2:diffcalc}).
  \item (Equação de Kepler) Para $0 \leq e < 1$ e $ m \in \R $,
        provar isso $ x = m + e\sin x $ tem exatamente uma solução,
        que a iteração $ x_{n+1} = m + e\sin x_n $ converge
        para ele desde qualquer início, e estime: para $ e = \frac12$,
        $ m = 1$, quantas iterações garantem um erro $\leq
        10^{-3}$ pelo limite a priori? (A solução é $ x
        \approx 1.4987$.)
  \item (O conjunto de Cantor é um ponto fixo) Seja $ S_1(x) = \frac
        x3$ e $ S_2(x) = \frac x3 + \frac23$ sobre $\R $, e deixe
        $ C $ seja o conjunto de Cantor de \cref{exo:b2:metric:8}. Provar
        $ C = S_1(C) \cup S_2(C)$ e explique em uma frase
        por que não \emph{outro} não vazio \omterm{def:b2:metric:compact}{compactar} conjunto satisfaz isso
        equação (o mapa $ A \mapsto S_1(A) \cup S_2(A)$ é um
        contração para uma distância entre \omterm{def:b2:metric:compact}{compactar} define --- o
        Distância de Hausdorff, tornada honesta no volume do Ano 3).
  \item (Conectividade globaliza a singularidade) Deixe $ f $ ser localmente
        \omterm{def:b2:metric:continuity}{Lipschitz} em $ y $ em um \omterm{def:b2:metric:topology}{abrir} definir e deixar $ y, z $ ser dois
        soluções de $ y' = f(t, y)$ em um intervalo comum $ J $
        com $ y(t_0) = z(t_0)$. Provar $ y = z $ sobre $ J $: mostre isso
        $\{t \in J : y(t) = z(t)\}$ não está vazio, fechado em $ J $,
        e \omterm{def:b2:metric:topology}{abrir} em $ J $ (por exclusividade local) e use o
        conectividade de intervalos
        (\cref{thm:b2:metric:connectedness}).
  \item (Sem pequenez para equações lineares) Para $ y' =
        \alpha(t)y + \beta(t)$ com $\alpha, \beta $ \omterm{def:b2:metric:continuity}{contínuo}
        em um segmento $\intcc{A}{B}$, adapte a pergunta 8 para mostrar
        o limite fatorial vale para \emph{todo} segmento,
        então a existência e a singularidade são globais lá --- o
        caso escalar de \cref{ch:b2:diffeq}Cauchy--Lipschitz
        teorema, sem restrição no comprimento $ B - A $.
  \item (Síntese) Uma frase cada: o que completude
        contribuiu; qual foi a contribuição da contração; o que
        truque iterativo comprado em comparação com Banach simples; onde
        conectividade entrou; e qual contraexemplo da Parte
        III guarda qual hipótese. Nomeie o teorema do cume,
        e diga o que substitui a contração quando $ f$ é meramente
        \omterm{def:b2:metric:continuity}{contínuo} (Teorema de Peano, via compacidade em função
        espaços --- Arzel do volume do Ano 3à--Ascoli).
\end{enumerate}
\end{problem}
