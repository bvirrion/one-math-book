\chapter{Sequências e Séries}\label{ch:b2:series}

A teoria das séries numéricas (volume do ano 1) amadurece aqui em três
direções: séries com valores em Espaços Banach, onde absoluto
a convergência faz o trabalho; os testes mais precisos para séries reais (Abel
soma); e \emph{resumido famílias} --- somatório liberado de
a ordem dos termos --- com o teorema de Fubini para somas duplas
e o Produto Cauchy. Essas ferramentas carregam todas as séries de funções
capítulos à frente.

\section{Séries em espaços normados}

\begin{definition}\label{def:b2:series:def}
Para uma sequência $(u_n)$ em um espaço normatizado $ E $, a série $\sum u_n $
converge quando suas somas parciais convergem; converge
\emph{absolutamente}\index{absolutamente} quando
$\sum \norm{u_n} < \infty $. Em um \emph{Banach} espaço, absoluto
convergência implica convergência (\cref{thm:b2:nvs:absoluteconvergence});
em um incompleto espaço isso pode falhar
(\cref{exo:b2:series:9}).
\end{definition}

\begin{example}\label{ex:b2:series:neumann}
Em $\mathcal{M}_n(K)$ (ou $\mathcal{L}_c(E)$, $ E $ Banach): para
$\vertiii A < 1$, o \emph{Série Neumann} $\sum A^k $ converge
\omterm{def:b2:series:def}{absolutamente} para $(I - A)^{-1}$ (comprovado em \cref{exo:b2:nvs:5});
$\sum \frac{A^k}{k!}$ converge \omterm{def:b2:series:def}{absolutamente} para $\eu^A $ para cada $ A $
(\cref{ex:b2:nvs:matrixexp}). Geométrico e com valor do operador
séries exponenciais se comportam como seus modelos escalares --- o todo
ponto da estrutura de Banach.
\end{example}

\section{Resumo de Abel}

\begin{theorem}[Soma e teste de Abel]\label{thm:b2:series:abel}
(Somatório por partes) Para escalares $ a_n $ e vetores $ b_n $, com $ B_n
= \sum_{k=0}^{n} b_k $:\index{Resumo de Abel}
\[
\sum_{n=0}^{N} a_n b_n
= a_N B_N - \sum_{n=0}^{N-1} (a_{n+1} - a_n) B_n .
\]
(Teste de Abel) Se $(a_n)$ é uma sequência real, decrescente para $0$ e
as somas parciais $ B_n $ são \emph{limitado} (em um \omterm{def:b2:nvs:banach}{Espaço Banach}), então
$\sum a_n b_n $ converge.
\end{theorem}

\begin{proof}
A identidade, passo a passo: com $ B_{-1} = 0$, escrever $ b_n = B_n -
B_{n-1}$ e dividir,
\[
\sum_{n=0}^{N} a_nb_n
= \sum_{n=0}^{N} a_nB_n - \sum_{n=0}^{N} a_nB_{n-1}
= \sum_{n=0}^{N} a_nB_n - \sum_{n=0}^{N-1} a_{n+1}B_{n} ,
\]
reindexando a segunda soma por $ n \mapsto n + 1$ (o $ B_{-1}$
termo desaparece); coletando o intervalo comum $0 \leq n \leq N-1$
folhas $ a_NB_N $ mais $\sum_{n\leq N-1}(a_n - a_{n+1})B_n $: o
fórmula declarada. É a integração discreta por partes, com
$(B_n)$ como antiderivada de $(b_n)$ e a diferença $ a_{n+1}
- a_n $ como derivado de $(a_n)$. Para o teste, com $\norm{B_n}
\leq M $: o
termo limite $ a_N B_N \to 0$; a série $\sum (a_n -
a_{n+1})B_n $ converge \omterm{def:b2:series:def}{absolutamente}, desde
\[
\sum_n \norm{(a_{n+1} - a_n)B_n} \leq M \sum_n (a_n - a_{n+1})
= M a_0 < \infty
\]
(telescópio, $ a_n \downarrow 0$). Ambas as peças da identidade
convergem, portanto, o mesmo acontece $\sum a_n b_n $.
\end{proof}

\begin{example}\label{ex:b2:series:sinn}
$\sum \frac{\sin n}{n}$ converge: $ a_n = \frac1n \downarrow 0$ e
$ B_n = \sum_{k=1}^{n} \sin k $ é limitado --- de fato $ B_n =
\Im\sum_{k \leq n} \eu^{\iu k} = \Im\,\frac{\eu^{\iu}(\eu^{\iu n} -
1)}{\eu^{\iu} - 1}$, de módulo $\leq \frac{2}{\abs{\eu^{\iu} - 1}}$.
Isso acontece \emph{não} convergir \omterm{def:b2:series:def}{absolutamente} ($\abs{\sin n} \geq \sin^2 n
= \frac{1 - \cos 2n}{2}$, e $\sum \frac{1 - \cos 2n}{2n}$ diverge
desde $\sum \frac{\cos 2n}{n}$ converge pelo mesmo teste de Abel enquanto
$\sum \frac{1}{2n}$ diverge). O teste da série alternada é o
caso especial $ b_n = (-1)^n $.
\end{example}

\begin{example}[Abel no círculo de convergência]\label{ex:b2:series:abelboundary}
Para qual complexo $ z $ com $\abs z = 1$ faz $\sum_{n \geq 1}
\frac{z^n}{n}$ convergem? No $ z = 1$ é a série harmônica:
divergente. Para $ z \neq 1$ no círculo, o teste de Abel se aplica
com $ a_n = \frac1n \downarrow 0$ e $ b_n = z^n $, cuja parcial
somas são limitadas independentemente de $ N $:
\[
\Bigl|\sum_{n=1}^{N} z^n\Bigr|
= \Bigl|\frac{z(z^N - 1)}{z - 1}\Bigr|
\leq \frac{2}{\abs{z - 1}} .
\]
Convergente --- embora nunca \omterm{def:b2:series:def}{absolutamente} ($\sum\frac1n $). Um
série, um círculo de comportamentos: divergência em um único ponto,
semiconvergência em todos os outros lugares. Este é o limite padrão
comportamento de séries de potências (\cref{ch:b2:powerseries}), conheci aqui
com as mãos nuas; no $ z = -1$ recupera a alternância
série harmônica, e em $ z = \eu^{\iu\theta}$ é real e
partes imaginárias são a série $\sum\frac{\cos
n\theta}{n}$ e $\sum\frac{\sin n\theta}{n}$ de
\cref{exo:b2:series:4}.
\end{example}

\begin{example}[Uma série alternada armadilhada]\label{ex:b2:series:trap}
Faz $\sum_{n\geq2} \dfrac{(-1)^n}{\sqrt n + (-1)^n}$ convergem?
Os sinais se alternam e os termos tendem a $0$ --- ainda o
teste alternado faz \emph{não} aplicar: os módulos
$\frac{1}{\sqrt n + (-1)^n}$ não estão diminuindo (eles saltam em
cada estranho $ n $). Em vez disso, expanda:
\[
\frac{(-1)^n}{\sqrt n + (-1)^n}
= \frac{(-1)^n}{\sqrt n}\cdot
\frac{1}{1 + \frac{(-1)^n}{\sqrt n}}
= \frac{(-1)^n}{\sqrt n} - \frac{1}{n}
+ O\Bigl(\frac{1}{n^{3/2}}\Bigr).
\]
A primeira parte converge (teste alternado, aplicado honestamente a
$\frac1{\sqrt n}\downarrow0$), a terceira converge \omterm{def:b2:series:def}{absolutamente}
--- mas a peça do meio é a série harmônica divergente: o
soma \emph{diverge} para $-\infty $. O insight final: quando
a monotonicidade falhar, expanda até que cada peça seja
\omterm{def:b2:series:def}{absolutamente} caso de teste convergente ou limpo; o oculto
$-\frac1n $ é invisível para a contagem de sinais.
\end{example}

\begin{remark}[Armadilhas comuns]\label{rem:b2:series:pitfalls}
(i) ``Os termos tendem a $0$'' não prova nada: a série harmônica
diverge. (ii) O teste alternado exige \emph{diminuindo}
módulos --- \cref{ex:b2:series:trap} é o canônico
contra-exemplo, e \cref{exo:b2:series:1}terceira série do
broca. (iii) Séries condicionalmente convergentes podem não ser
reorganizado (\cref{ex:b2:series:rearrange}) e seus \omterm{thm:b2:series:fubini}{Cauchy
produtos} pode divergir: para $\sum\frac{(-1)^n}{\sqrt{n+1}}$
ao quadrado, os termos diagonais satisfazem
\[
\abs{c_k} = \sum_{m=0}^{k}
\frac{1}{\sqrt{(m+1)(k-m+1)}}
\geq (k+1)\cdot\frac{2}{k+2} \longrightarrow 2 \neq 0
\]
(cada fator é no máximo $\frac{k+2}2$ por AM--GM), então $\sum c_k $
diverge --- convergência absoluta de pelo menos um fator
(\cref{exo:b2:series:8}) não é um luxo. (4) Somabilidade é
sobre \emph{absoluto} limites por definição: não existe tal
coisa como condicionalmente família resumida.
\end{remark}

\section{Famílias resumidas}

\begin{definition}\label{def:b2:series:summable}
Deixar $ I $ seja um \omterm{def:b2:structures:countable}{contável} conjunto de índices. Uma família $(u_i)_{i \in I}$ de
\emph{reais não negativos} é \emph{currículo}\index{currículo}
quando as somas parciais finitas são limitadas; sua soma é
\[
\sum_{i \in I} u_i = \sup_{F \subseteq I \text{ finite}} \sum_{i
\in F} u_i \in \intcc{0}{+\infty} .
\]
Uma família de reais ou complexos (ou vetores de Banach) é somável quando
$(\norm{u_i})$ é; sua soma é então definida dividindo-se em
partes positivas/negativas (ou reais/imaginárias) --- equivalentemente, como o
valor comum de $\sum_{n} u_{\sigma(n)}$ sobre todas as enumerações
$\sigma $ de $ I $ (Veja abaixo).
\end{definition}

\begin{method}[Escolhendo um teste]\label{met:b2:series:choosing}
Enfrentando $\sum u_n $, em ordem: (1) se $ u_n \not\to 0$, divergência,
pare. (2) Se os termos tiverem sinal constante, compare: encontre um
equivalente (\cref{ch:b2:comparison}) e coloque-o no
Mapa de Riemann-Bertrand. (3) Se os sinais alternarem com
\emph{diminuindo} módulos, o teste alternado; se os módulos
não são monótonos, expanda o termo até que cada peça seja \omterm{def:b2:series:def}{absolutamente}
convergente ou um caso de teste limpo (\cref{ex:b2:series:trap}). (4)
Se o padrão do sinal for oscilatório, mas estruturado ($\sin
n\theta $, $\eu^{\iu n\theta}$, potências de matriz), teste de Abel com
somas parciais limitadas. (5) A convergência absoluta sempre vale a pena
verificando primeiro: é mais forte, à prova de pedidos e desbloqueia \omterm{thm:b2:series:fubini}{Cauchy
produtos} e Fubini.
\end{method}

\begin{example}[Somabilidade por contagem diagonal]\label{ex:b2:series:diagonalcount}
Para qual $ s > 0$ é a família
$\bigl((m + n)^{-s}\bigr)_{m, n \geq 1}$ resumido? Agrupar o
somas parciais finitas por diagonais $ m + n = k $: a diagonal $ k $
carrega $ k - 1$ pares, cada um contribuindo $ k^{-s}$, então o
somas finitas são exatamente limitadas por (e esgotadas)
\[
\sum_{k \geq 2} \frac{k - 1}{k^{s}} ,
\]
uma série com termos positivos equivalentes a $ k^{1-s}$: resumido
se $ s - 1 > 1$, ou seja\ $ s > 2$. O índice bidimensional come
um poder \omterm{def:b2:metric:complete}{completo}: um plano de termos é “uma dimensão a mais
divergente'' do que uma linha --- a geometria de contagem do índice
conjunto, e não o tamanho dos termos individuais, decide somabilidade.
(O mesmo censo mostra $\bigl((m^2 + n^2)^{-1}\bigr)$ não é
resumido: na diagonal $ m + n = k $, cada termo é pelo menos
$ k^{-2}$, e $(k-1)\cdot k^{-2}$ somas como o harmônico
série.)
\end{example}

\begin{theorem}[Somabilidade e ordem]\label{thm:b2:series:rearrangement}
\begin{enumerate}
  \item Para famílias não negativas, a soma é invariante sob qualquer
        enumeração: $\sum_{i} u_i = \sum_{n=0}^{\infty}
        u_{\sigma(n)}$ para cada bijeção $\sigma \colon \N \to I $.
  \item Uma série real ou complexa $\sum u_n $ é \emph{comutativamente
        convergente} (todo rearranjo converge, com o mesmo
        soma) se e somente se for \omterm{def:b2:series:def}{absolutamente} convergente.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Cada soma parcial $\sum_{n \leq N} u_{\sigma(n)}$ é um finito
soma parcial da família (então $\leq $ o sup); cada finito $ F $ é
contido em alguns $\{\sigma(0), \dots, \sigma(N)\}$ (então o sup
$\leq $ limite da série). Os dois limites coincidem.

(2) Se $\sum\abs{u_n} < \infty $: para qualquer rearranjo $\sigma $ e
$\varepsilon > 0$, escolher $ N $ com $\sum_{n > N}\abs{u_n} \leq
\varepsilon $; além da classificação onde $\sigma $ esgotou
$\intint{0}{N}$, as somas parciais reorganizadas diferem das
limite original em no máximo $\varepsilon $: mesma soma. Se $\sum
\abs{u_n} = \infty $ mas $\sum u_n $ converge (caso real; complexo
segue coordenada): as partes positivas e negativas ambas
divergem, e pode-se reorganizar para atingir qualquer limite prescrito
--- Teorema de Riemann, realizado em \cref{exo:b2:series:5} ---
então a convergência comutativa falha.
\end{proof}

\begin{example}[Um rearranjo pego em flagrante]\label{ex:b2:series:rearrange}
A série harmônica alternada soma
$\sum_{n\geq1}\frac{(-1)^{n-1}}{n} = \ln 2$ (Ano 1
volume). Reorganize-o como ``um positivo, dois negativos'':
\[
1 - \frac12 - \frac14 + \frac13 - \frac16 - \frac18 + \frac15 -
\cdots
\]
Agrupando cada bloco de três,
\[
\frac{1}{2k-1} - \frac{1}{4k-2} - \frac1{4k}
= \frac{1}{4k-2} - \frac{1}{4k}
= \frac12\Bigl(\frac{1}{2k-1} - \frac1{2k}\Bigr),
\]
então a série reorganizada converge para $\frac12\ln 2$ --- metade do
soma original, exatamente com os mesmos termos. Não \omterm{def:b2:series:def}{absolutamente}
séries convergentes lembram a ordem de seus termos; resumido
as famílias são precisamente aquelas que não o fazem.
\end{example}

\begin{example}[Agrupar é seguro, desagrupar não é]\label{ex:b2:series:grouping}
Agrupando termos consecutivos de um \emph{convergente} série nunca
altera a soma: as somas parciais agrupadas formam uma subsequência de
os originais. A operação inversa é proibida:
\[
(1 - 1) + (1 - 1) + (1 - 1) + \cdots = 0 + 0 + \cdots = 0,
\]
ainda o desagrupado $1 - 1 + 1 - 1 + \cdots $ diverge (parcial
somas oscilam entre $1$ e $0$). Desagrupar é legítimo
apenas com uma hipótese compensatória --- por exemplo, termos
tendendo a $0$ com comprimentos de grupo limitados: então entre dois
somas parciais agrupadas, as originais derivam no máximo de uma soma de
$ o(1)$ termos de número limitado e transferências de convergência de volta.
Essa é exatamente a cláusula sob a qual o cálculo em bloco de
\cref{ex:b2:series:rearrange} é uma prova e não um truque de
mão.
\end{example}

\begin{theorem}[Fubini para famílias; Produtos Cauchy]\label{thm:b2:series:fubini}
Deixar $(u_{m,n})_{(m,n) \in \N^2}$ seja um resumido família dupla (ou seja,\
$\sup_F \sum_F \abs{u_{m,n}} < \infty $). Então\index{Teorema de Fubini
para série}
\[
\sum_{(m,n)} u_{m,n}
= \sum_{m=0}^{\infty}\Bigl(\sum_{n=0}^{\infty} u_{m,n}\Bigr)
= \sum_{n=0}^{\infty}\Bigl(\sum_{m=0}^{\infty} u_{m,n}\Bigr)
= \sum_{k=0}^{\infty}\Bigl(\sum_{m+n=k} u_{m,n}\Bigr),
\]
todas as séries internas convergindo (\omterm{def:b2:series:def}{absolutamente}). Em particular, se $\sum
a_m $ e $\sum b_n $ convergir \omterm{def:b2:series:def}{absolutamente}, deles \emph{Cauchy
produto}\index{Cauchy
produto} converge \omterm{def:b2:series:def}{absolutamente} com
\[
\Bigl(\sum_m a_m\Bigr)\Bigl(\sum_n b_n\Bigr)
= \sum_{k=0}^{\infty} c_k,
\qquad
c_k = \sum_{m=0}^{k} a_m b_{k-m} .
\]
\end{theorem}

\begin{proof}
\emph{Caso não negativo.} Cada agrupamento (por linhas, colunas ou
diagonais) calcula o mesmo supremo: qualquer conjunto finito de pares é
contido em um bloco finito de linhas (limitando cada soma agrupada abaixo
por somas parciais finitas e acima pelo total), e monótono
a convergência de somas parciais faz o resto --- concretamente, para linhas:
$\sum_{m \leq M}\sum_{n \leq N} u_{m,n} \leq S $ dá, deixa $ N
\to \infty $ então $ M \to \infty $, $\sum_m \sum_n u_{m,n} \leq S $;
inversamente todo finito $ F $ fica em tal retângulo, então $ S \leq
\sum_m\sum_n u_{m,n}$. Diagonais: os mesmos dois limites com triângulos
em vez de retângulos.

\emph{Caso geral.} Dividido em positivo e negativo (real e
imaginárias) partes, cada uma resumido família não negativa; os quatro
os agrupamentos concordam em cada parte e, portanto, na diferença; absoluto
a convergência da série interna vem do caso não negativo
aplicado a $\abs{u_{m,n}}$.

\emph{Produto Cauchy.} A família $ u_{m,n} = a_m b_n $ é resumido:
somas parciais finitas de $\abs{a_mb_n}$ são limitados por
$\bigl(\sum\abs{a_m}\bigr)\bigl(\sum\abs{b_n}\bigr)$. Linhas dão
$\bigl(\sum a_m\bigr)\bigl(\sum b_n\bigr)$; diagonais dão $\sum_k
c_k $.
\end{proof}

\begin{example}[A identidade exponencial, honestamente]\label{ex:b2:series:exp}
Para $ a, b \in \C $ (ou matrizes comutadas):
\[
\Bigl(\sum_m \frac{a^m}{m!}\Bigr)\Bigl(\sum_n
\frac{b^n}{n!}\Bigr)
= \sum_k \sum_{m+n=k} \frac{a^m b^n}{m!\,n!}
= \sum_k \frac{(a + b)^k}{k!},
\]
pelo teorema binomial em cada diagonal: $\eu^a \eu^b = \eu^{a+b}$
--- a equação funcional de $\exp $ derivado apenas da série.
(A comutação é usada na etapa binomial; para não comutável
matrizes a identidade falha genuinamente,
\cref{ch:b2:diffeq}.)
\end{example}

\begin{example}[Produtos Cauchy como dispositivo de computação]\label{ex:b2:series:squaresum}
Da série geométrica e \cref{exo:b2:series:2}de $\sum_{n
\geq 1} nz^n = \frac{z}{(1-z)^2}$ ($\abs z < 1$), mais um
Produto Cauchy termina o segundo momento. Multiplicar $\sum_m
mz^m $ por $\sum_n z^n $: o coeficiente diagonal é $\sum_{m=0}^k
m = \frac{k(k+1)}2$, então
\[
\frac{z}{(1-z)^3} = \sum_{k\geq0}\frac{k(k+1)}{2}\,z^k ,
\]
e a identidade $ n^2 = 2\cdot\frac{n(n+1)}2 - n $ monta
\[
\sum_{n\geq1} n^2z^n = \frac{2z}{(1-z)^3} - \frac{z}{(1-z)^2}
= \frac{z(1+z)}{(1-z)^3} .
\]
No $ z = \frac12$: $\sum_{n\geq1}\frac{n^2}{2^n} =
\frac{\frac12\cdot\frac32}{\frac18} = 6$ --- um valor fechado com
nenhuma diferenciação em lugar nenhum, apenas \omterm{def:b2:series:def}{absolutamente} série convergente
multiplicado como polinômios. A mesma telescopagem de identidades
calcula cada $\sum n^dz^n $, e os probabilistas reconhecerão
o segundo momento fatorial da lei geométrica
(\cref{ch:b2:genfun}).
\end{example}

\begin{example}[Uma avaliação de soma dupla]\label{ex:b2:series:doublesum}
Para $ s > 1$ real, deixe $\zeta(s) = \sum_{n\geq1} n^{-s}$. Contando
divisores por soma dupla --- a família $(m^{-s}n^{-s})$ sobre
$(m,n) \in (\N^*)^2$ é resumido (produto de positivo convergente
série) --- e agrupamento por produto $ q = mn $:
\[
\zeta(s)^2 = \sum_{m,n} \frac{1}{(mn)^s}
= \sum_{q=1}^{\infty} \frac{d(q)}{q^s},
\]
onde $ d(q)$ é o número de divisores de $ q $. Resumido famílias
transformar combinatória em análise.
\end{example}

\begin{example}[Uma avaliação de Fubini: $\sum_n (\zeta(n) - 1) = 1$]\label{ex:b2:series:zetasum}
Para número inteiro $ n \geq 2$, $\zeta(n) - 1 = \sum_{k \geq 2} k^{-n}$.
A família dupla $(k^{-n})_{k, n \geq 2}$ é resumido: somando
as colunas geométricas primeiro,
\[
\sum_{k\geq2}\sum_{n\geq2} \frac{1}{k^n}
= \sum_{k\geq2} \frac{1/k^2}{1 - 1/k}
= \sum_{k\geq2} \frac{1}{k(k-1)} = 1
\]
(telescópica), e todos os termos são positivos, então
\cref{thm:b2:series:fubini} autoriza a soma por linhas:
\[
\sum_{n\geq2}\bigl(\zeta(n) - 1\bigr) = 1 .
\]
Os infinitos $\zeta $-valores, cada um com aparência transcendental,
têm caudas que somam exatamente $1$. Visão final: quando um
soma dupla tem termos positivos, calcule-a em qualquer ordem
entra em colapso --- aqui as colunas são geométricas, as linhas são misteriosas,
e Fubini transfere o colapso.
\end{example}

\begin{example}[A série geométrica resolve uma equação]\label{ex:b2:series:neumannsolve}
No \omterm{def:b2:nvs:banach}{Espaço Banach} $\bigl(C(\intcc01),
\norm\cdot_\infty\bigr)$, resolver $ x - K(x) = y $ onde $ K(f)$ é
a função constante $\frac12\int_0^1 f $. O \omterm{thm:b2:nvs:continuouslinear}{norma do operador} é
$\vertiii K \leq \frac12 < 1$, então a série de Neumann se aplica
(\cref{ex:b2:series:neumann}): $ x = \sum_{n\geq0} K^n(y)$.
Calcule as iterações: $ K(y) = \frac12\int_0^1 y $ (uma constante),
e aplicando $ K $ para uma constante $ c $ dá $\frac c2$, então
$ K^n(y) = \frac{1}{2^{n-1}}\cdot\frac12\int_0^1 y $ para $ n \geq
1$. Somando as constantes geométricas:
\[
x = y + \Bigl(\int_0^1 y\Bigr)
\sum_{n\geq1}\frac{1}{2^n} = y + \int_0^1 y .
\]
Verificar: $ x - K(x) = y + \int y - \frac12\bigl(\int y + \int
y\bigr) = y $. Uma série infinita, uma resposta finita e uma
verificação unifilar --- a série geométrica é uma inversão
algoritmo, não apenas uma declaração de convergência.
\end{example}

\begin{example}[Telescópica por frações parciais]\label{ex:b2:series:telescope}
A soma exata é rara; telescópica é seu principal fornecedor.
Decompor
\[
\frac{1}{n(n+1)(n+2)}
= \frac{1}{2}\Bigl(\frac{1}{n(n+1)} -
\frac{1}{(n+1)(n+2)}\Bigr),
\]
(verifique por redução ao denominador comum), então o parcial
colapso das somas:
\[
\sum_{n=1}^{N}\frac{1}{n(n+1)(n+2)}
= \frac12\Bigl(\frac{1}{1\cdot2} -
\frac{1}{(N+1)(N+2)}\Bigr)
\longrightarrow \frac14 .
\]
O mesmo padrão --- escreva o termo como $ c(u_n - u_{n+1})$ para
um explícito $(u_n)$ --- resolvido \cref{exo:b2:series:10}
(arcotangentes) e calcula cada $\sum\frac{1}{n(n+1)\cdots(n +
k)} = \frac{1}{k\cdot k!}$. Quando existe uma soma exata neste
nível, um telescópio geralmente está escondido no termo.
\end{example}

\begin{remark}[Perspectivas dentro deste volume]\label{rem:b2:series:perspectives}
Três capítulos à frente são clientes diretos. Para
\cref{ch:b2:funcseq}: convergência normal de $\sum f_n $ é
convergência absoluta de $\sum\norm{f_n}_\infty $ no Banach
espaço $\bigl(C, \norm\cdot_\infty\bigr)$ --- deste capítulo
\cref{thm:b2:nvs:absoluteconvergence} fantasiado. Para
\cref{ch:b2:powerseries}: dentro do disco de convergência
tudo é absoluto e resumido, então Produtos Cauchy e
os rearranjos ocorrem livremente (é por isso que as séries de potências se multiplicam como
polinômios); na fronteira, o teste de Abel assume
(\cref{ex:b2:series:abelboundary}). Para \cref{ch:b2:genfun}:
funções geradoras de probabilidade são séries de potências cujas
manipulações --- produtos para somas de variáveis independentes,
somas duplas para leis compostas --- são todas licenciadas por
\cref{thm:b2:series:fubini}. Resumido famílias são legais
departamento da análise que está por vir.
\end{remark}

\begin{remark}[Onde este capítulo é usado]
Tudo que tem soma infinita passa por aqui: série de potências
(\cref{ch:b2:powerseries}) são resumido famílias disfarçadas,
Os coeficientes de Fourier são multiplicados por Produtos Cauchy e
reorganizado por Parseval (\cref{ch:b2:fourier}) e probabilidade
gerando funções (\cref{ch:b2:genfun}) são o teorema de Fubini
aplicado às expectativas. O volume do Ano 3 absorve resumido
famílias na integração de Lebesgue durante a medida de contagem ---
onde \cref{thm:b2:series:fubini} torna-se um caso especial do
Teorema de Fubini-Tonelli.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b2:series:1}
Natureza de: $\sum \dfrac{\cos n}{n}$; $\;\sum \dfrac{(-1)^n}{\ln
n}$; $\;\sum \dfrac{(-1)^n}{n^{3/4} + \cos n}$ \emph{(expandir como em
a armadilha do Ano 1: o teste alternado precisa de monotonicidade)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:series:2}
Prove isso para $\abs z < 1$: $\sum_{n\geq1} n z^{n} =
\dfrac{z}{(1-z)^2}$, através do Produto Cauchy de $\sum z^n $ com
em si.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:series:3}
(Lema do tipo Kronecker) Seja $\sum b_n $ seja uma série real convergente.
Prove, por \omterm{thm:b2:series:abel}{Resumo de Abel}, que $\dfrac{1}{n}\sum_{k=1}^{n} k\,b_k
\to 0$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:series:4}
Estude a convergência de $\sum \dfrac{\sin(n\theta)}{n^\alpha}$
($\theta \in \R $, $\alpha > 0$) --- para o qual $(\theta, \alpha)$ é
isso \omterm{def:b2:series:def}{absolutamente} convergente, semiconvergente, divergente?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:series:5}
(Rearranjo de Riemann) Deixe $\sum u_n $ ser convergente, mas não
\omterm{def:b2:series:def}{absolutamente} série real convergente, e $\ell \in \R $. Prove isso
algum rearranjo de $\sum u_n $ converge para $\ell $.
\emph{(Mostre que ambas as subséries de termos positivos e negativos divergem;
então alterne avidamente: tome termos positivos até exceder
$\ell $, depois negativo até cair abaixo e assim por diante; os termos
tendem a $0$, forçando a convergência para $\ell $.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:series:6}
Prove que a família
$\Bigl(\dfrac{x^{m+n}}{m!\,n!}\Bigr)_{(m,n)\in\N^2}$ é resumido
para cada $ x \in \R $ e derivar novamente a identidade $(\eu^x)^2 =
\eu^{2x}$ agrupando a soma dupla ao longo das diagonais $ m + n =
k $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:series:7}
Prove que a família $\bigl(\frac{1}{m^2 n^2}\bigr)_{m,n \geq 1}$
é resumido, e esse agrupamento por $\gcd $: com $ q = \gcd(m,n)$,
\[
\zeta(2)^2 = \sum_{q\geq1} \frac{1}{q^4}
\sum_{\substack{a,b \geq 1\\ \gcd(a,b)=1}} \frac{1}{a^2b^2}
= \zeta(4) \cdot S,
\]
onde $ S = \sum_{\gcd(a,b)=1} \frac{1}{a^2b^2}$: deduzir $ S =
\zeta(2)^2/\zeta(4)$. \emph{(Cada par $(m,n)$ escreve exclusivamente
$(qa, qb)$ com $\gcd(a,b) = 1$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:series:8}
(Teorema de Abel sobre produtos, versão light) Suponha $\sum a_n $
converge \omterm{def:b2:series:def}{absolutamente} e $\sum b_n $ converge. Prove que seus
Produto Cauchy $\sum c_n $ converge, com $\sum c_n = (\sum
a_n)(\sum b_n)$. \emph{(Escrever $ C_N = \sum_{k\leq N} c_k = \sum_n
a_n B_{N-n}$ com $ B $ as somas parciais de $ b $; dividir de acordo com
$ n \leq N/2$ ou não, usando limites de $(B_m)$ e o absoluto
cauda de $(a_n)$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:series:9}
No (incompleto) espaço $ E $ de sequências reais eventualmente zero
com o sup \omterm{def:b2:nvs:norm}{norma}, exibir um \omterm{def:b2:series:def}{absolutamente} série convergente que faz
não convergir em $ E $. \emph{(Tentar $ u_n = 2^{-n} e_n $ com $(e_n)$ o
sequências canônicas.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:series:10}
Verifique a identidade $\arctan(n+1) - \arctan(n) =
\arctan\dfrac{1}{n^2 + n + 1}$ e deduza o valor exato de
\[
\sum_{n=1}^{\infty} \arctan\frac{1}{n^2 + n + 1} .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:series:11}
Determine a natureza (com equivalentes) de
\[
\sum_n \bigl(\sqrt{n+1} - \sqrt n\bigr)^{\alpha}
\ (\alpha > 0), \qquad
\sum_n \Bigl(1 - \cos\frac1n\Bigr), \qquad
\sum_n \Bigl(\eu - \Bigl(1 + \frac1n\Bigr)^{\!n}\Bigr).
\]
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:series:12}
Deixar $(a_n)$ seja positivo e \emph{diminuindo} com $\sum a_n $
convergente. Prove isso $ n\,a_n \to 0$ \emph{(vinculado $ n a_{2n}$ por um
cauda)}. Mostre que a recíproca falha e que a monotonicidade é
essencial, com contra-exemplos explícitos.
\end{exercise}

\section{Problema: Euler $\zeta(2) = \pi^2/6$, por Cauchy
Soma Cotangente}

A identidade mais famosa de Euler, $1 + \frac14 + \frac19 + \cdots =
\frac{\pi^2}6$, admite uma prova completamente elementar, devido a
Cauchy: a fórmula de de Moivre produz um polinômio cujas raízes são
os números $\cot^2\frac{k\pi}{2n+1}$, Vieta soma essas raízes
exatamente, e o aperto $\cot^2\theta < \frac{1}{\theta^2} <
1 + \cot^2\theta $ esmaga as somas parciais de $\sum\frac1{k^2}$
entre dois limites racionais explícitos. Executamos a prova completa,
extrair $\zeta(4) = \frac{\pi^4}{90}$ pelo mesmo método, então
mapear toda a fronteira entre convergência e divergência com
a série de Bertrand --- e provar que a fronteira não carrega
série convergente mais lenta de todas.

\begin{problem}[{Problema de fim de semana --- $\zeta(2) = \pi^2/6$ e o
Panorama de Bertrand}]
\label{pb:b2:series:1}
Por todo, $ n \geq 1$ e $\theta_k = \dfrac{k\pi}{2n+1}$ para $ k
= 1, \dots, n $; observação $0 < \theta_k < \frac\pi2$.

\medskip
\noindent\textbf{Part I --- The cotangent identity.}
\begin{enumerate}
  \item Prove a fórmula de de Moivre $(\cos\theta +
        \iu\sin\theta)^m = \cos m\theta + \iu\sin m\theta $ ($ m
        \in \N $) e deduza, por $ m = 2n + 1$,
        \[
        \sin\bigl((2n{+}1)\theta\bigr) = \sum_{j=0}^{n}
        (-1)^j\binom{2n+1}{2j+1}\cos^{2(n-j)}\theta\,
        \sin^{2j+1}\theta .
        \]
  \item Deduza isso para $\theta \in \intoo{0}{\frac\pi2}$,
        \[
        \sin\bigl((2n{+}1)\theta\bigr) =
        \sin^{2n+1}\theta\; P_n(\cot^2\theta),
        \qquad
        P_n(x) = \sum_{j=0}^{n}
        (-1)^j\binom{2n+1}{2j+1}x^{\,n-j},
        \]
        um polinômio de grau $ n $ com coeficiente líder $2n
        + 1$.
  \item Mostre isso $ x_k = \cot^2\theta_k $, $ k = 1, \dots, n $, são
        $ n $ \emph{distinto} raízes de $ P_n $ --- daí todos
        eles.
  \item Por Vieta, prove a identidade exata
        \[
        \sum_{k=1}^{n} \cot^2\frac{k\pi}{2n+1}
        = \frac{n(2n-1)}{3}.
        \]
  \item Deduza também $\displaystyle\sum_{k=1}^{n}
        \frac{1}{\sin^2\theta_k} = \frac{2n(n+1)}{3}$.
  \item Prove o aperto: $\cot^2\theta < \dfrac1{\theta^2} <
        \dfrac{1}{\sin^2\theta}$ para $\theta \in
        \intoo{0}{\frac\pi2}$ \emph{(de $\sin\theta < \theta <
        \tan\theta $)}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The squeeze closes: Euler's
theorem.}
\begin{enumerate}[resume]
  \item Resumindo a questão 6 $ k = 1, \dots, n $ com $\theta =
        \theta_k $, estabelecer
        \[
        \frac{n(2n-1)}{3} \;<\;
        \frac{(2n+1)^2}{\pi^2}\sum_{k=1}^{n}\frac1{k^2}
        \;<\; \frac{2n(n+1)}{3}.
        \]
  \item Concluir (\emph{Teorema de Euler, pela prova de Cauchy}):
        \[
        \zeta(2) = \sum_{k=1}^{\infty}\frac{1}{k^2} =
        \frac{\pi^2}{6}.
        \]
  \item Extraia uma taxa do sanduíche: mostre
        \[
        \Bigl|\sum_{k=1}^{n}\frac1{k^2} - \frac{\pi^2}6\Bigr|
        = O\Bigl(\frac1n\Bigr),
        \]
        consistente com a cauda exata $\sum_{k>n}k^{-2} =
        \frac1n - \frac{1}{2n^2} + O(n^{-3})$ de
        \cref{exo:b2:comparison:11}.
  \item Opere a máquina um andar acima: usando o segundo Vieta
        função de $ P_n $, mostrar
        \[
        \sum_{k=1}^{n}\cot^4\theta_k =
        \Bigl(\frac{n(2n-1)}3\Bigr)^{\!2} -
        \frac{2n(2n-1)(2n-2)(2n-3)}{60}
        \;\sim\; \frac{8n^4}{45},
        \]
        e aperte com $\cot^4 < \theta^{-4} < (1 +
        \cot^2)^2$ para obter $\zeta(4) = \dfrac{\pi^4}{90}$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Dividends.}
\begin{enumerate}[resume]
  \item Deduzir de $\zeta(2) = \frac{\pi^2}6$:
        \[
        \sum_{k\geq0}\frac{1}{(2k+1)^2} = \frac{\pi^2}{8},
        \qquad
        \sum_{k\geq1}\frac{(-1)^{k-1}}{k^2} = \frac{\pi^2}{12}.
        \]
  \item Combinar com \cref{exo:b2:series:7}: calcular $ S =
        \sum_{\gcd(a,b)=1}\frac{1}{a^2b^2} =
        \frac{\zeta(2)^2}{\zeta(4)} = \frac52$ e interpretar
        $\frac{1}{\zeta(2)} = \frac{6}{\pi^2} \approx 0.608$ como
        a densidade de pares coprimos (indique a heurística
        honestamente: a contagem rigorosa é um volume do Ano 3
        importa).
  \item (Aceleração certificada) A fórmula da cauda da questão 9
        dá $\sum_{k\leq n}k^{-2} + \frac1n - \frac1{2n^2} =
        \frac{\pi^2}6 + O(n^{-3})$. Compare o trabalho necessário para
        seis dígitos de $\zeta(2)$: soma direta versus o
        soma corrigida em $ n = 100$ (onde o erro é
        $1.7\cdot10^{-7}$).
  \item Verifique a questão 4 manualmente em $ n = 1$ e $ n = 2$ (o
        valores $\cot^2\frac\pi3 = \frac13$ e
        $\cot^2\frac\pi5 + \cot^2\frac{2\pi}5 = 2$), usando
        $\cos\frac\pi5 = \frac{1+\sqrt5}4$ ou um número
        avaliação.
  \item Prove a identidade do acompanhante
        \[
        \sum_{k=1}^{n}\tan^2\frac{k\pi}{2n+1} = n(2n+1)
        \]
        \emph{(os números $\tan^2\theta_k $ são as raízes do
        polinômio invertido $ x^nP_n(1/x)$)}e verifique-o em $ n
        = 1$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The Bertrand panorama.} Para $\alpha,
\beta \in \R $, considere o \emph{Série Bertrand}
\[
\sum_{n \geq 3} \frac{1}{n^{\alpha}(\ln n)^{\beta}} .
\]
\begin{enumerate}[resume]
  \item Mostre isso para $\alpha > 1$ a série converge,
        tanto faz $\beta $ \emph{(comparar com
        $ n^{-(1+\alpha)/2}$)}.
  \item Mostre isso para $\alpha < 1$ diverge, seja qual for
        $\beta $.
  \item Para $\alpha = 1$: usando a comparação integral de série
        (\cref{thm:b2:comparison:seriesintegral}) com $ f(t) =
        \frac{1}{t(\ln t)^\beta}$, prove a convergência se $\beta
        > 1$.
  \item Iterar a fronteira: mostrar $\sum\frac{1}{n\ln n\,\ln\ln
        n}$ diverge enquanto $\sum\frac{1}{n\ln n\,(\ln\ln n)^2}$
        converge.
  \item Duas armadilhas: determinar a natureza do
        \[
        \sum_n \frac{1}{n^{1 + 1/\ln n}}
        \qquad\text{and}\qquad
        \sum_n \frac{1}{n^{1 + 1/\ln\ln n}}
        \]
        \emph{(calcular $ n^{1/\ln n}$ exatamente; comparar
        $ n^{1/\ln\ln n}$ com todo poder de $\ln n $)}.
  \item (Sem série convergente mais lenta) Deixe $\sum a_n $ seja qualquer
        série convergente com $ a_n > 0$, e $ R_n = \sum_{k
        \geq n}a_k $ suas caudas. Prove isso $\sum
        \frac{a_n}{\sqrt{R_n}}$ ainda \emph{converge}
        \emph{(compare com o telescópico $2(\sqrt{R_n} -
        \sqrt{R_{n+1}})$)}, embora $\frac{a_n/\sqrt{R_n}}{a_n}
        \to \infty $: toda série convergente é estritamente
        dominada por outra série convergente. A fronteira de
        a convergência não é uma curva, mas uma névoa.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Cross-checks and synthesis.}
\begin{enumerate}[resume]
  \item (Condensação de Cauchy) Prove: para $(a_n)$ positivo
        diminuindo, $\sum a_n $ converge se e se $\sum 2^k a_{2^k}$
        converge. Derive novamente a fronteira da questão 18 a partir dela.
  \item (O custo da lentidão) Para $\sum\frac1{n(\ln n)^2}$,
        ligue a cauda por uma integral e mostre que somando
        para $ N = 10^6$ ainda deixa um erro maior que
        $0.07$: a convergência certificada pela teoria pode ser inútil
        para números --- contraste com a pergunta 13.
  \item Classificar (com justificativas de uma linha):
        $\sum\frac1{n\ln n}$, $\sum\frac1{n^{1.01}}$,
        $\sum\frac{(\ln n)^{100}}{n^{1.001}}$,
        $\sum\frac1{n(\ln n)(\ln\ln n)^{3}}$.
  \item (Síntese) Uma frase cada: como de Moivre virou
        identidade trigonométrica em um polinômio com computável
        somas raiz; onde o aperto necessário \emph{exato}
        identidades de endpoint em vez de equivalentes; qual ferramenta
        de \cref{ch:b2:comparison} Parte IV alimentada; e o que
        a questão 21 diz sobre o sonho de uma “universidade
        teste de comparação''. Nomeie os dois cumes: Euler
        $\zeta(2) = \frac{\pi^2}6$ (e seu andar acima,
        $\zeta(4) = \frac{\pi^4}{90}$), e o Bertrand
        classificação. Observe onde $\zeta(2)$ será provado
        novamente: por Parseval em \cref{ch:b2:fourier} --- um
        teorema, duas civilizações.
\end{enumerate}
\end{problem}
