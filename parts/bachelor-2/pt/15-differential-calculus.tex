\chapter{Cálculo diferencial}\label{ch:b2:diffcalc}

O cálculo de duas variáveis do volume do Ano 1 amadurece no
cálculo \omterm{def:b2:diffcalc:differential}{diferencial} de mapas entre espaços normados: o
\emph{\omterm{def:b2:diffcalc:differential}{diferencial}} como a melhor aproximação linear, a regra da cadeia
em geral, o teorema de simetria de Schwarz \emph{comprovado},Taylor
fórmulas e a análise completa de extremos de segunda ordem. O
teorema da função inversa, coroa da teoria, é afirmado com seu
estratégia de prova --- um ponto fixo de Banach --- explicitada.

Por todo, $U $ é um \omterm{def:b2:metric:topology}{abrir} subconjunto de $\R^n $ (ou um espaço normado; o
caso de dimensão finita carrega todas as ideias), $ f \colon U \to
\R^m $.

\section{O diferencial}

\begin{definition}\label{def:b2:diffcalc:differential}
$ f $ é \emph{diferenciável}\index{diferenciável} no $ a $ quando lá
é um (\omterm{def:b2:metric:continuity}{contínuo}) mapa linear $\dd f_a \colon \R^n \to \R^m $ com
\[
f(a + h) = f(a) + \dd f_a(h) + o(\norm h)
\qquad (h \to 0).
\]
O mapa $\dd f_a $, o \emph{diferencial}\index{diferencial} de
$ f $ no $ a $, é único; sua matriz nas bases canônicas é
o \emph{Matriz Jacobiana}\index{Matriz Jacobiana} $ J_f(a) =
\bigl(\frac{\partial f_i}{\partial x_j}(a)\bigr)$.
Diferenciabilidade implica continuidade e a existência de todos
derivadas direcionais $\dd f_a(v) = \lim_{t\to0}\frac{f(a + tv)
- f(a)}{t}$; o inverso falha
(\cref{exo:b2:diffcalc:2}). Para $ m = 1$, $\dd f_a(h) = \langle
\nabla f(a), h\rangle $: o gradiente do Ano 1, agora entendido como
o vetor que representa o diferencial.
\end{definition}

\begin{theorem}[$ C^1$ critério]\label{thm:b2:diffcalc:c1}
Se todas as derivadas parciais de $ f $ existir em $ U $ e são \omterm{def:b2:metric:continuity}{contínuo}
no $ a $, então $ f $ é \omterm{def:b2:diffcalc:differential}{diferenciável} no $ a $. ``$ C^1$ sobre $ U $'' ---
\omterm{def:b2:metric:continuity}{contínuo} parciais --- portanto implica diferenciabilidade
em todos os lugares, com \omterm{def:b2:metric:continuity}{contínuo} \omterm{def:b2:diffcalc:differential}{diferencial}.
\end{theorem}

\begin{proof}
Componente por componente ($ m = 1$ é suficiente). A prova do Ano 1 para dois
variáveis --- mover uma coordenada de cada vez, aplicar a variável única
teorema do valor médio em cada perna e use continuidade das parciais
em $ a $ --- generaliza literalmente para $ n $ pernas:
\[
f(a + h) - f(a) = \sum_{j=1}^{n} \bigl(f(a + h^{(j)}) - f(a +
h^{(j-1)})\bigr)
= \sum_j h_j\,\frac{\partial f}{\partial x_j}(\xi_j),
\]
onde $ h^{(j)}$ congela o primeiro $ j $ coordenadas de $ h $ e
$\xi_j $ encontra-se no $ j $-ésima perna; continuidade gira cada
$\frac{\partial f}{\partial x_j}(\xi_j)$ em $\frac{\partial
f}{\partial x_j}(a) + o(1)$, e o erro é $ o(\norm h)$.
\end{proof}

\begin{theorem}[Regra da cadeia]\label{thm:b2:diffcalc:chainrule}
Se $ f $ é \omterm{def:b2:diffcalc:differential}{diferenciável} no $ a $ e $ g $ no $ f(a)$, então $ g \circ f $
é \omterm{def:b2:diffcalc:differential}{diferenciável} no $ a $ com
\[
\dd(g \circ f)_a = \dd g_{f(a)} \circ \dd f_a ,
\qquad
J_{g\circ f}(a) = J_g\bigl(f(a)\bigr)\,J_f(a) :
\]
Os jacobianos se multiplicam.\index{regra da cadeia}
\end{theorem}

\begin{proof}
Escrever $ f(a + h) = f(a) + \dd f_a(h) + \norm h\,\varepsilon_1(h)$
e $ g(b + k) = g(b) + \dd g_b(k) + \norm k\,\varepsilon_2(k)$ com
$ b = f(a)$, $ k = k(h) = \dd f_a(h) + \norm h \varepsilon_1(h)$.
Substituindo,
\[
g(f(a+h)) = g(b) + \dd g_b\bigl(\dd f_a(h)\bigr)
+ \norm h\,\dd g_b(\varepsilon_1(h)) + \norm{k}\,\varepsilon_2(k),
\]
e ambos os termos de erro são $ o(\norm h)$: o primeiro desde $\dd g_b $ é
\omterm{def:b2:metric:continuity}{contínuo} e $\varepsilon_1 \to 0$; o segundo desde $\norm k
\leq C\norm h $ (mapa linear limitado mais pequeno termo) e
$\varepsilon_2(k) \to 0$ como $ h \to 0$.
\end{proof}

\begin{example}[Funções radiais, de uma vez por todas]\label{ex:b2:diffcalc:radial}
Deixar $ r(x) = \norm x_2$ sobre $\R^n\setminus\{0\}$ e $ f = g \circ
r $ com $ g $ um $ C^1$ função de uma variável. Primeiro, $ r $ é
\omterm{def:b2:diffcalc:differential}{diferenciável} longe de $0$: de $ r^2 = \sum x_i^2$,
\[
\frac{\partial r}{\partial x_i} = \frac{x_i}{r},
\qquad\text{i.e.}\qquad
\nabla r(x) = \frac{x}{\norm x} ,
\]
o vetor radial unitário (diferenciar $ r^2$ e dividir --- ou
aplicar a regra da cadeia a $\sqrt{\cdot}$). Então a regra da cadeia
dá, para cada função radial,
\[
\nabla f(x) = g'\bigl(\norm x\bigr)\,\frac{x}{\norm x} .
\]
Instância trabalhada: $ g(r) = \frac1r $ rendimentos $\nabla\frac{1}{\norm
x} = -\frac{x}{\norm x^3}$, o campo inverso do quadrado de
gravitação e eletrostática --- direção radial, magnitude
$\frac{1}{\norm x^2}$. Visão final: gradientes de radial
funções são radiais porque os conjuntos de níveis são esferas e
o gradiente é ortogonal aos conjuntos de níveis; no $ x = 0$, por
contraste, $ r $ é \emph{não} \omterm{def:b2:diffcalc:differential}{diferenciável} (nenhum candidato
correspondências de mapa linear $\norm h $ de todas as direções) --- suave
perfis radiais precisam $ g'(0) = 0$ cruzar a origem
graciosamente.
\end{example}

\begin{theorem}[Desigualdade de valor médio]\label{thm:b2:diffcalc:mvi}
Deixar $ f $ ser \omterm{def:b2:diffcalc:differential}{diferenciável} sobre $ U $ e deixe o segmento $\intcc{a}{b}
= \{a + t(b-a)\}$ deitar em $ U $. Então
\[
\norm{f(b) - f(a)} \leq \norm{b - a}\,
\sup_{x \in \intcc{a}{b}} \vertiii{\dd f_x} .
\]
Em particular, um \omterm{def:b2:diffcalc:differential}{diferenciável} mapa com zero \omterm{def:b2:diffcalc:differential}{diferencial} em um
\emph{\omterm{def:b2:metric:connected}{conectado}} \omterm{def:b2:metric:topology}{abrir} conjunto é constante.
\end{theorem}

\begin{proof}
A função $\varphi(t) = f(a + t(b-a))$ é \omterm{def:b2:diffcalc:differential}{diferenciável} sobre
$\intcc{0}{1}$ com $\varphi'(t) = \dd f_{a + t(b-a)}(b - a)$
(regra da cadeia), de \omterm{def:b2:nvs:norm}{norma} $\leq M\norm{b-a}$ com $ M $ o exibido
epa. Para $\R $-valorizado $ f $ a desigualdade de valor médio de uma variável
conclui; para valores vetoriais, aplique-o a $ t \mapsto \langle u,
\varphi(t)\rangle $ com $ u $ o vetor unitário ao longo $ f(b) - f(a)$.
Constância: localmente constante (segmentos em bolas) mais conectividade
(o conjunto onde $ f $ é igual a um determinado valor é \omterm{def:b2:metric:topology}{abrir} e fechado:
\cref{ch:b2:metric}).
\end{proof}

\begin{example}[Uma constante de Lipschitz do MVI]\label{ex:b2:diffcalc:lipschitz}
É $ f(x, y) = \sin x\,\sin y $ \omterm{def:b2:metric:continuity}{Lipschitz} sobre $\R^2$, e com
qual constante? Seu gradiente é $\nabla f = (\cos x\sin y,\
\sin x\cos y)$, do quadrado \omterm{def:b2:nvs:norm}{norma}
\[
\cos^2x\sin^2y + \sin^2x\cos^2y
\leq \sin^2 y + \cos^2y\cdot 1 = 1
\]
(vinculado $\cos^2x $ e $\sin^2x $ por $1$ separadamente), então
$\vertiii{\dd f_{(x,y)}} = \norm{\nabla f} \leq 1$ em todos os lugares,
e \cref{thm:b2:diffcalc:mvi} no segmento entre quaisquer dois
pontos dá
\[
\abs{f(b) - f(a)} \leq \norm{b - a}_2 :
\]
$ f $ é $1$-Lipschitz, e a constante é nítida (perto do
origem, $ f(x, \tfrac\pi2) = \sin x $ tem inclinação $1$). Fechando
insight: a desigualdade de valor médio converte um \omterm{def:b2:funcseq:def}{pontualmente} vinculado
no \omterm{def:b2:diffcalc:differential}{diferencial} em um módulo global de continuidade ---
a rota padrão para \omterm{def:b2:metric:continuity}{Lipschitz} estimativas em todas as dimensões,
e o motor dentro \cref{exo:b2:diffcalc:12}.
\end{example}

\section{Segundas derivadas}

\begin{theorem}[Preto]\label{thm:b2:diffcalc:schwarz}
Se $ f $ é $ C^2$ sobre $ U $ (todas as segundas parciais existem e são
\omterm{def:b2:metric:continuity}{contínuo}), então para todos $ i, j $:\index{Teorema de Schwarz}
\[
\frac{\partial^2 f}{\partial x_i\,\partial x_j}
= \frac{\partial^2 f}{\partial x_j\,\partial x_i} .
\]
\end{theorem}

\begin{proof}
Duas variáveis são suficientes ($ x = x_i $, $ y = x_j $, outros congelados).
Considere a segunda diferença
\[
\Delta(h) = f(a + h, b + h) - f(a + h, b) - f(a, b + h) + f(a,b) .
\]
Consertar $ h $ e definir $\varphi(x) = f(x, b+h) - f(x, b)$: então $\Delta(h)
= \varphi(a + h) - \varphi(a)$, e duas aplicações da média
teorema do valor dar
\[
\Delta(h) = h\,\varphi'(\xi)
= h\Bigl(\frac{\partial f}{\partial x}(\xi, b+h) -
\frac{\partial f}{\partial x}(\xi, b)\Bigr)
= h^2\,\frac{\partial^2 f}{\partial y\,\partial x}(\xi, \eta),
\]
com $\xi \in \intoo{a}{a+h}$, $\eta \in \intoo{b}{b+h}$. Por
continuidade, $\frac{\Delta(h)}{h^2} \to \frac{\partial^2 f}{\partial
y\partial x}(a, b)$ como $ h \to 0$. O mesmo cálculo com o
funções das variáveis trocadas (congele a segunda variável primeiro)
dá $\frac{\Delta(h)}{h^2} \to \frac{\partial^2 f}{\partial x
\partial y}(a,b)$: os dois limites da mesma quantidade coincidem.
\end{proof}

\begin{example}[Por que $ C^2$ é necessário: contra-exemplo de Peano]\label{ex:b2:diffcalc:peano}
Deixar $ f(x, y) = \dfrac{xy(x^2 - y^2)}{x^2 + y^2}$, $ f(0,0) = 0$.
Longe da origem $ f $ é $ C^\infty $; na origem tudo primeiro
e segundas parciais existem, mas as mistas discordam. Calcular
ao longo dos eixos: $ f(x, 0) = f(0, y) = 0$, e para $ y \neq 0$,
\[
\frac{\partial f}{\partial x}(0, y)
= \lim_{x\to0}\frac{f(x,y)}{x}
= \frac{y(0 - y^2)}{y^2} = -y ,
\qquad\text{symmetrically}\qquad
\frac{\partial f}{\partial y}(x, 0) = x .
\]
Por isso
\[
\frac{\partial^2 f}{\partial y\,\partial x}(0,0)
= \frac{\dd}{\dd y}\Bigl[\frac{\partial f}{\partial
x}(0,y)\Bigr]_{y=0} = -1,
\qquad
\frac{\partial^2 f}{\partial x\,\partial y}(0,0) = +1 :
\]
as duas parciais mistas existem e diferem. Nenhuma contradição com
\cref{thm:b2:diffcalc:schwarz}: as segundas parciais de $ f $ são
não \omterm{def:b2:metric:continuity}{contínuo} no $0$ (teste junto $ y = tx $). Visão final:
O teorema de Schwarz é um teorema genuíno sobre \emph{continuidade},
não é uma identidade formal --- e a hipótese ``$ C^2$'' em
Taylor – O jovem abaixo está fazendo um trabalho de verdade.
\end{example}

\begin{theorem}[Taylor--Young na ordem 2]\label{thm:b2:diffcalc:taylor}
Deixar $ f \colon U \to \R $ ser $ C^2$ e $ a \in U $. Então, como $ h \to 0$,
\[
f(a + h) = f(a) + \langle\nabla f(a), h\rangle
+ \frac12\, \langle H_a h,\, h\rangle + o\bigl(\norm h^2\bigr),
\]
onde $ H_a = \bigl(\frac{\partial^2 f}{\partial x_i\partial
x_j}(a)\bigr)$ é o (\omterm{def:b2:quadratic:adjoint}{simétrico}, de Schwarz) \emph{Hessiano
matriz}\index{Hessiano
matriz}.
\end{theorem}

\begin{proof}
Aplique o teorema de Taylor--Young de uma variável (volume do ano 1) para
$\varphi(t) = f(a + th)$ sobre $\intcc{0}{1}$: pela regra da cadeia,
$\varphi'(t) = \langle \nabla f(a + th), h\rangle $ e $\varphi''(t)
= \langle H_{a+th}h, h\rangle $, ambos \omterm{def:b2:metric:continuity}{contínuo} em $ t $. Então
$\varphi(1) = \varphi(0) + \varphi'(0) + \frac12\varphi''(\theta)$
(Taylor-Lagrange) com $\theta \in \intoo{0}{1}$, e continuidade
das segundas parciais converte $\varphi''(\theta) =
\varphi''(0) + o(1)\cdot\norm h^2$-uniformemente: o exibido
expansão.
\end{proof}

\begin{example}[Uma expansão em dois sentidos]\label{ex:b2:diffcalc:taylortwoways}
Expandir $ f(x, y) = \eu^x\cos y $ na origem para encomendar $2$.
\emph{Por composição de expansões univariáveis:}
\[
\eu^x\cos y = \Bigl(1 + x + \frac{x^2}{2} +
o(x^2)\Bigr)\Bigl(1 - \frac{y^2}{2} + o(y^2)\Bigr)
= 1 + x + \frac{x^2 - y^2}{2} + o\bigl(\norm{(x,y)}^2\bigr) .
\]
\emph{Por derivadas parciais:} $ f_x = \eu^x\cos y $, $ f_y =
-\eu^x\sin y $, então $\nabla f(0) = (1, 0)$; e $ f_{xx} = f $,
$ f_{yy} = -f $, $ f_{xy} = -\eu^x\sin y $ dar $ H_0 =
\operatorname{diag}(1, -1)$: \cref{thm:b2:diffcalc:taylor}
reproduz $1 + x + \frac12(x^2 - y^2)$. Os dois cálculos
concordo, e a rota de composição foi mais rápida --- nenhum segundo
parciais em tudo. Visão final: a origem é \emph{não} um
ponto crítico ($\nabla f \neq 0$), portanto, apesar do tempo indefinido
Hessiano não há sela para declarar: o termo linear governa,
e o teste de segunda ordem só fala em níveis críticos
pontos.
\end{example}

\begin{theorem}[Teste extremo de segunda ordem, comprovado]\label{thm:b2:diffcalc:extrema}
Deixar $ f $ ser $ C^2$ perto de um ponto crítico $ a $ ($\nabla f(a) = 0$),
com Hessiano $ H = H_a $.
\begin{enumerate}
  \item Se $ H $ é positivo definido, $ a $ é um mínimo local estrito
        (definido negativo: máximo).
  \item Se $ H $ tem valores próprios de ambos os signos, $ a $ é uma sela: não
        extremo.
  \item Se $ H $ é singular (e semidefinido), sem conclusão.
\end{enumerate}
O Ano 1 ``$ rt - s^2$'' teste é o caso $ n = 2$: $\det H = rt -
s^2$, $\operatorname{tr}$-sinal lido $ r $.
\end{theorem}

\begin{proof}
Pelo teorema espectral (\cref{thm:b2:quadratic:spectral}), $ H $ de
forma quadrática está espremido entre seus extremos valores próprios:
$\lambda_{\min}\norm h^2 \leq \langle Hh, h\rangle \leq
\lambda_{\max}\norm h^2$.

(1) Se $\lambda_{\min} > 0$: Taylor--Young dá
\[
f(a + h) - f(a) \geq \frac{\lambda_{\min}}{2}\norm h^2 -
o(\norm h^2) > 0
\]
para pequeno $ h \neq 0$: mínimo local estrito.

(2) Ao longo de um vetor próprio $ v_+$ com $\lambda_+ > 0$: $ f(a + tv_+) -
f(a) = \frac{\lambda_+}{2}t^2 + o(t^2) > 0$ para pequeno $ t $; junto
$ v_-$ com $\lambda_- < 0$ a diferença é negativa: ambos os sinais
ocorrem em todos os bairros.

(3) $ f(x,y) = x^2 + y^4$, $ x^2 - y^4$, $ x^2 + y^3$ compartilhe o mesmo
singular semidefinido Hessiano em $0$ com três diferentes
comportamentos.
\end{proof}

\begin{example}[Uma classificação completa, incluindo global]\label{ex:b2:diffcalc:quarticrun}
Classifique todos os extremos de $ f(x, y) = x^4 + y^4 - 4xy $ sobre $\R^2$.
\emph{Pontos críticos:} $\nabla f = (4x^3 - 4y,\ 4y^3 - 4x) =
0$ dá $ y = x^3$ e $ x = y^3 = x^9$, então $ x(x^8 - 1) = 0$:
as verdadeiras soluções são $(0,0)$, $(1,1)$, $(-1,-1)$.
\emph{Hessianos:} $ H = \begin{pmatrix} 12x^2 & -4\\ -4 &
12y^2\end{pmatrix}$. No $(\pm1, \pm1)$: $\begin{pmatrix} 12 &
-4\\ -4 & 12\end{pmatrix}$, valores próprios $8$ e $16$: positivo
mínimos locais definidos e estritos com $ f = -2$. No $(0,0)$:
$\begin{pmatrix} 0 & -4\\ -4 & 0\end{pmatrix}$, valores próprios
$\pm4$: uma sela. \emph{Globalidade:} de $2\abs{xy} \leq x^2 +
y^2$,
\[
f(x, y) \geq x^4 + y^4 - 2(x^2 + y^2)
= (x^2 - 1)^2 + (y^2 - 1)^2 + x^2 + y^2 - 2
\xrightarrow[\norm{(x,y)}\to\infty]{} +\infty :
\]
$ f $ é coercitivo, por isso atinge um mínimo global
(compacidade de conjuntos de subníveis), necessariamente em um nível crítico
ponto: o valor $-2$, em ambos $(1,1)$ e $(-1,-1)$, é o
mínimo global; não há máximo ($ f $ ilimitado acima).
Conclusão: a prova local classifica os candidatos, mas
apenas um argumento de crescimento transforma o “local” em “global” --- o
padrão de duas etapas de cada prova de otimização neste livro.
\end{example}

\begin{method}[Classificando os extremos de $ f \colon \R^n \to \R $]\label{met:b2:diffcalc:extrema}
\begin{enumerate}
  \item Resolver $\nabla f = 0$ (todos os pontos críticos; em um domínio
        com limite, trate o limite separadamente como em
        \cref{exo:b2:diffcalc:7}).
  \item Em cada ponto crítico, calcule o Hessiano e seu
        \omterm{def:b2:reduction:eigen}{autovalor} sinais --- em dimensão $2$, apenas $\det H $
        e $\operatorname{tr} H $: $\det < 0$ sela; $\det >
        0$ extremo, do tipo dado pelo sinal do
        rastrear; $\det = 0$: a prova é silenciosa, estuda $ f $
        ao longo das curvas.
  \item Para declarações globais, adicione um compacidade ou coercividade
        argumento ($ f \to +\infty $ no infinito, ou um \omterm{def:b2:metric:compact}{compactar}
        conjunto de restrições) e compare os valores críticos.
\end{enumerate}
\end{method}

\section{O teorema da função inversa}

\begin{theorem}[Teorema da função inversa]\label{thm:b2:diffcalc:inverse}
Deixar $ f \colon U \to \R^n $ ser $ C^1$ e $ a \in U $ com $\dd f_a $
\emph{invertível}. Então há \omterm{def:b2:metric:topology}{abrir} bairros $ V \ni a $, $ W
\ni f(a)$ tal que $ f \colon V \to W $ é uma bijeção com $ C^1$
inverso, e\index{teorema da função inversa}
\[
\dd (f^{-1})_{f(x)} = (\dd f_x)^{-1} \qquad (x \in V).
\]
\end{theorem}
\admitted

\begin{remark}[Por que é verdade: a estratégia do ponto fixo]
Resolvendo $ f(x) = y $ aproximar $ a $ reescreve como a equação de ponto fixo
$ x = x + \dd f_a^{-1}\bigl(y - f(x)\bigr) =: \Phi_y(x)$; o mapa
$\Phi_y $ tem \omterm{def:b2:diffcalc:differential}{diferencial} $\mathrm{id} - \dd f_a^{-1}\dd f_x $,
pequeno perto $ a $ por continuidade de $\dd f $, então $\Phi_y $ é um
contração em uma pequena bola fechada e no ponto fixo de Banach
teorema (\cref{thm:b2:metric:banach}) fornece o local exclusivo
solução $ x = f^{-1}(y)$. Continuidade e diferenciabilidade do
inverso segue então das estimativas na contração. O \omterm{def:b2:metric:complete}{completo}
a escrituração contábil é realizada no Ano 3; a estratégia --- e o
declaração --- são usados livremente a partir de agora. O companheiro
\emph{teorema da função implícita} (resolvendo $ F(x, y) = 0$ para $ y(x)$
quando $\frac{\partial F}{\partial y}$ é invertível) seguido por
aplicando o teorema a $(x, y) \mapsto (x, F(x,y))$.
\end{remark}

\begin{example}[Coordenadas polares]\label{ex:b2:diffcalc:polar}
$\Phi(r, \theta) = (r\cos\theta, r\sin\theta)$ tem Jacobiano
\[
J_\Phi = \begin{pmatrix} \cos\theta & -r\sin\theta\\ \sin\theta &
r\cos\theta \end{pmatrix},
\qquad \det J_\Phi = r :
\]
invertível para $ r \neq 0$, então $\Phi $ é um local $ C^1$
difeomorfismo longe da origem --- a licença para “mudar para
coordenadas \omterm{pb:b2:quadratic:1}{polares}'', renovadas para as integrais múltiplas de
\cref{ch:b2:multint}.
\end{example}

\begin{example}[Local em qualquer lugar, global em lugar nenhum]\label{ex:b2:diffcalc:localnotglobal}
Deixar $ f(x, y) = \bigl(\eu^x\cos y,\ \eu^x\sin y\bigr)$ sobre
$\R^2$. É Jacobiano,
\[
J_f = \begin{pmatrix}
\eu^x\cos y & -\eu^x\sin y\\
\eu^x\sin y & \eu^x\cos y
\end{pmatrix},
\qquad
\det J_f = \eu^{2x} > 0 ,
\]
nunca desaparece: por \cref{thm:b2:diffcalc:inverse}, $ f $ é um
locais $ C^1$ difeomorfismo em \emph{todo} ponto do avião.
Ainda $ f $ está longe de ser injetivo: $ f(x, y + 2\pi) = f(x, y)$, então
cada valor é obtido com uma frequência infinita; e não é sobrejetivo
quer, desde $\norm{f(x, y)} = \eu^x > 0$ perde a origem.
Visão final: o teorema da função inversa é irredutivelmente
\emph{local} --- invertibilidade de cada $\dd f_a $ produz um
colcha de retalhos de inversos locais que não precisam ser reunidos em um só.
(Leitores que conhecem números complexos reconhecerão $ z \mapsto
\eu^z $; a colcha de retalhos é a família dos ramos logarítmicos.)
Comparar \cref{exo:b2:diffcalc:12}, onde um valor global quantitativo
hipótese força um inverso global.
\end{example}

\begin{remark}[Armadilhas comuns]
\emph{(i) Os derivados direcionais são baratos, diferenciais são
não:} todas as derivadas direcionais podem existir --- e até falhar
depender linearmente da direção --- sem
diferenciabilidade (\cref{exo:b2:diffcalc:2}); apenas o $ C^1$
critério
(\cref{thm:b2:diffcalc:c1}) atualiza parciais para um
\omterm{def:b2:diffcalc:differential}{diferencial}. \emph{(ii) Crítico não significa extremo:}
selas (\cref{ex:b2:diffcalc:quarticrun}) e o silencioso
caso singular (\cref{thm:b2:diffcalc:extrema}~(3)) ambos se escondem
atrás $\nabla f = 0$. \emph{(iii) Nenhum valor médio com valor vetorial
igualdade:} apenas o \emph{desigualdade} de
\cref{thm:b2:diffcalc:mvi} sobrevive
(\cref{exo:b2:diffcalc:9}); nunca escreva $ f(b) - f(a) = \dd
f_c(b-a)$ para $ f $ com valores em $\R^m $, $ m \geq 2$.
\emph{(iv) Invertibilidade local não é injetividade:}
\cref{ex:b2:diffcalc:localnotglobal}. \emph{(v) O gradiente
pertence ao produto interno:} $\nabla f $ é o vetor
representando $\dd f_a $ num produto interno escolhido; mudar o
produto (como no exemplo ponderado das formas quadráticas
capítulo) e o gradiente gira, enquanto o \omterm{def:b2:diffcalc:differential}{diferencial} ---
o objeto intrínseco --- não se move.
\end{remark}

\begin{remark}[Onde isso é usado]
Tudo o que está abaixo deste capítulo é cálculo \omterm{def:b2:diffcalc:differential}{diferencial}
aplicado: o capítulo de equações diferenciais lineariza fluxos e
usa Liouville determinante fórmula (provada neste capítulo
problema de fim de semana); os capítulos sobre estudo de curvas e superfícies
conjuntos de níveis e parametrizações através da função implícita
teorema; múltiplas integrais mudam variáveis através de Jacobianos.
O problema do fim de semana desenvolve o cálculo \emph{no espaço de
matrizes em si} --- \omterm{def:b2:diffcalc:differential}{diferencial} do determinante, do
inverso, o \omterm{ex:b2:nvs:matrixexp}{matriz exponencial}, e o grupo ortogonal como um
conjunto de nível suave --- a sombra do Ano 2 do que o Ano 3
o volume se formaliza como variedades e grupos de Lie.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b2:diffcalc:1}
Calcule as matrizes Jacobianas de $ f(x,y) = (x^2 - y^2,\, 2xy)$ e
de $\Phi(r,\theta,z) = (r\cos\theta, r\sin\theta, z)$; onde estão
o diferenciais invertível?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:diffcalc:2}
Deixar $ f(x,y) = \frac{x^3}{x^2 + y^2}$ ($ f(0,0) = 0$). Prove que tudo
derivadas direcionais de $ f $ no $0$ existe, mas isso $ f $ não é
\omterm{def:b2:diffcalc:differential}{diferenciável} no $0$ \emph{(o mapa $ v \mapsto $ direcional
derivada não é linear)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:diffcalc:3}
Encontre e classifique os pontos críticos de $ f(x, y) = x^3 + y^3 -
3xy $ usando \cref{thm:b2:diffcalc:extrema}, e de $ g(x,y) = x^4 +
y^4 - 2(x - y)^2$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffcalc:4}
Deixar $ f \colon \R^n \to \R $ ser $ C^1$ e \emph{homogêneo de grau
$ p $}: $ f(tx) = t^pf(x)$ para $ t > 0$. Prove a identidade de Euler
\[
\langle \nabla f(x), x\rangle = p\,f(x) ,
\]
e seu inverso para $ C^1$ funções ativadas $\R^n\setminus\{0\}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffcalc:5}
Deixar $ A $ ser \omterm{def:b2:quadratic:adjoint}{simétrico} e $ f(x) = \frac12\langle Ax, x\rangle -
\langle b, x\rangle $. Calcular $\nabla f $ e $ H_f $; quando é $ f $
convexo? Supondo $ A $ definido positivo, mostre que $ f $ tem um único
mínimo global na solução de $ Ax = b $ --- descida gradiente
razão de ser aqui.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffcalc:6}
(Multiplicador de Lagrange, uma restrição, provado manualmente) Deixe $ f, g $ ser
$ C^1$ sobre $\R^2$, e suponha $ f $ atinge, em $ a $, um extremo local
no nível definido $\{g = 0\}$, com $\nabla g(a) \neq 0$. Prove isso
$\nabla f(a) = \lambda\nabla g(a)$ para alguns $\lambda $.
\emph{(Parametrize o nível definido próximo $ a $ pela função implícita
teorema e diferenciar $ t \mapsto f(\gamma(t))$.)} Aplicação:
extremos de $ f(x,y) = xy $ no círculo $ x^2 + y^2 = 1$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffcalc:7}
Determine os extremos de $ f(x, y) = x^2 + y^2 - xy + x - y $ sobre
$\R^2$, então seu máximo e mínimo no triângulo fechado com
vértices $(0,0)$, $(1,0)$, $(0,1)$ \emph{(pontos críticos interiores,
depois as três arestas, depois os vértices)}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:diffcalc:8}
Deixar $ f \colon \R^2 \to \R^2$, $ f(x, y) = (x + y^2,\; y + x^2)$.
Mostre isso $ f $ é um difeomorfismo local próximo $0$, calcular
$\dd(f^{-1})_{(0,0)}$, e encontre o maior $ r $ tal que $\dd f $
é invertível na bola $\norm{(x,y)}_2 < r $ \emph{(calcular
$\det J_f $)}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:diffcalc:9}
(Rolle falha, o valor médio sobrevive) Dê $ f \colon \R \to \R^2$,
$ C^1$, com $ f(0) = f(2\pi)$ mas $ f'(t) \neq 0$ para todos $ t $
(sem Rolle com valor vetorial). Então verifique no seu exemplo o
valor médio \emph{desigualdade} de \cref{thm:b2:diffcalc:mvi}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:diffcalc:10}
Calcule o \omterm{def:b2:diffcalc:differential}{diferencial} e o gradiente de $ f(x) = \norm
x_2^2$ e de $ g(x) = \langle Ax, x\rangle $ sobre $\R^n $ ($ A $ um
matriz quadrada, não assumida \omterm{def:b2:quadratic:adjoint}{simétrico}) e o Hessiano de cada um.
Para qual $ A $ é $ g $ convexo?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffcalc:11}
Deixar $ f \colon \R^n \to \R $ ser $ C^2$. Prove isso $ f $ é convexo se
e somente se for Hessiano $ H_x $ é positivo semidefinido em cada
$ x $ \emph{(reduzir para uma variável: $ t \mapsto f(a + t(b-a))$;
use Taylor-Lagrange em uma direção, e para o inverso
avaliar $\varphi''$)}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:diffcalc:12}
(Um teorema inverso global) Deixe $ g \colon \R^n \to \R^n $ ser $ C^1$
com $\vertiii{\dd g_x} \leq k < 1$ para todos $ x $, e $ f =
\mathrm{id} + g $.
\begin{enumerate}
  \item Mostrar $\norm{f(x) - f(y)} \geq (1 - k)\norm{x - y}$: $ f $
        é injetivo, com \omterm{def:b2:metric:continuity}{contínuo} inverso em sua imagem.
  \item Mostre que para cada $ y \in \R^n $ o mapa $ x \mapsto y -
        g(x)$ é uma contração do \omterm{def:b2:metric:complete}{completo} espaço $\R^n $,
        e conclua pelo teorema do ponto fixo de Banach
        (\cref{thm:b2:metric:banach}) que $ f $ é sobrejetivo.
  \item Conclua isso $ f $ é uma bijeção de $\R^n $ com
        $(1-k)^{-1}$-Lipschitz inverso --- a \emph{global}
        contrapartida de \cref{thm:b2:diffcalc:inverse} (que,
        pelo contrário, é puramente local).
\end{enumerate}
\end{exercise}

\section{Problema: O cálculo de matrizes --- Jacobi,
exponencial e o grupo ortogonal}

\begin{problem}\label{pb:b2:diffcalc:1}
O playground mais limpo para o cálculo \omterm{def:b2:diffcalc:differential}{diferencial} é o espaço
$\mathcal M_n(\R) \simeq \R^{n^2}$ em si: seus mapas mais naturais
--- produto, inverso, determinante, exponencial --- tem
diferenciais de elegância marcante. Este problema os calcula
tudo: o \emph{Série Neumann}, o \omterm{def:b2:diffcalc:differential}{diferencial} do inverso,
\emph{Fórmula de Jacobi}\index{Fórmula de Jacobi} para o
determinante com \emph{Fórmula de Liouville} como dividendo, o
\emph{matriz exponencial}\index{matriz exponencial} com
$\det\eu^A = \eu^{\operatorname{tr}A}$, e finalmente o
grupo ortogonal $ O_n $ como um nível suave definido com o
matrizes antissimétricas como espaço tangente --- \omterm{def:b2:diffcalc:differential}{diferencial}
geometria no embrião. Por todo, $\vertiii\cdot $ é o operador
\omterm{def:b2:nvs:norm}{norma} subordinado a $\norm\cdot_2$, e $\langle X, Y\rangle =
\operatorname{tr}(X^{\mathsf T}Y)$ o produto interno de Frobenius.

\medskip
\noindent\textbf{Part I --- The Neumann series.}
\begin{enumerate}
  \item Prove a submultiplicatividade, $\vertiii{AB} \leq
        \vertiii A\,\vertiii B $, e deduza esse polinômio
        mapas de $ A $ (produtos matriciais, determinante, rastreamento) são
        \omterm{def:b2:metric:continuity}{contínuo} sobre $\mathcal M_n(\R)$.
  \item Para $\vertiii X < 1$, mostre que $\sum_{k\geq0}X^k $
        converge \omterm{def:b2:series:def}{absolutamente} em $\mathcal M_n(\R)$
        (\cref{thm:b2:nvs:absoluteconvergence}), que sua soma
        é $(I - X)^{-1}$, e isso
        \[
        \vertiii{(I - X)^{-1}} \leq \frac{1}{1 - \vertiii X},
        \qquad
        (I - X)^{-1} = I + X + O\bigl(\vertiii X^2\bigr) .
        \]
  \item Deduza isso $ GL_n(\R)$ é \emph{\omterm{def:b2:metric:topology}{abrir}}: se $ A $ é
        invertível e $\vertiii H <
        \frac{1}{\vertiii{A^{-1}}}$, então $ A + H $ é
        invertível. Deduza também que $ GL_n(\R)$ é \emph{denso}
        em $\mathcal M_n(\R)$ \emph{(perturbar $ A $ por
        $\varepsilon I $: $\det(A + \varepsilon I)$ é diferente de zero
        polinômio em $\varepsilon $)}.
  \item Mostre que o mapa de inversão $\Phi(A) = A^{-1}$ é
        \omterm{def:b2:metric:continuity}{contínuo} sobre $ GL_n(\R)$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- First diferenciais.}
\begin{enumerate}[resume]
  \item Mostre que o mapa quadratura $ A \mapsto A^2$ é
        \omterm{def:b2:diffcalc:differential}{diferenciável} com \omterm{def:b2:diffcalc:differential}{diferencial} $ H \mapsto AH + HA $,
        e mais geralmente isso $ A \mapsto A^k $ tem
        \omterm{def:b2:diffcalc:differential}{diferencial} $ H \mapsto \sum_{i=0}^{k-1}
        A^iHA^{k-1-i}$. Por que alguém pode \emph{não} escrever $ kA^{k-1}H $
        em geral?
  \item Prove isso $\Phi(A) = A^{-1}$ é \omterm{def:b2:diffcalc:differential}{diferenciável} sobre
        $ GL_n(\R)$ com
        \[
        \dd\Phi_A(H) = -A^{-1}HA^{-1}
        \]
        \emph{(escrever $(A + H)^{-1} = (I + A^{-1}H)^{-1}A^{-1}$
        e expanda pela pergunta 2)}. Verifique a fórmula contra
        o caso escalar $ n = 1$.
  \item Por um $ C^1$ curva $ t \mapsto A(t) \in GL_n(\R)$, deduzir
        $\bigl(A(t)^{-1}\bigr)' = -A^{-1}A'A^{-1}$ e expandir
        $ t \mapsto (I + tB)^{-1}$ para o primeiro pedido em $ t = 0$.
  \item Calcule o \omterm{def:b2:diffcalc:differential}{diferencial} de $ f(A) =
        \operatorname{tr}(A^k)$ e identificar seu gradiente para
        o produto interno de Frobenius:
        \[
        \dd f_A(H) = k\operatorname{tr}\bigl(A^{k-1}H\bigr),
        \qquad
        \nabla f(A) = k\,\bigl(A^{k-1}\bigr)^{\mathsf T} .
        \]
  \item As mesmas perguntas para $ f(A) = \operatorname{tr}
        (A^{\mathsf T}A) = \norm A_F^2$: \omterm{def:b2:diffcalc:differential}{diferencial},
        gradiente e o Hessiano (constante); concluir que
        $\norm\cdot_F^2$ é estritamente convexo.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Jacobi's formula.}
\begin{enumerate}[resume]
  \item Provar
        \[
        \det(I + H) = 1 + \operatorname{tr}H +
        O\bigl(\vertiii H^2\bigr)
        \]
        \emph{(expandir $\det(e_1 + h_1, \dots, e_n + h_n)$ por
        multilinearidade nas colunas: termos com pelo menos dois
        $ h $-colunas são $ O(\vertiii H^2)$)}: $\dd(\det)_I =
        \operatorname{tr}$.
  \item Para $ A $ invertível, deduzir
        \[
        \dd(\det)_A(H) = \det(A)\,
        \operatorname{tr}\bigl(A^{-1}H\bigr) .
        \]
  \item Mostre isso para \emph{todo} $ A $ (invertível ou não),
        $\frac{\partial\det}{\partial a_{ij}}(A) = C_{ij}$, o
        $(i,j)$ cofator \emph{(Expansão de Laplace ao longo da linha
        $ i $)}, de modo que com o \omterm{def:b2:quadratic:adjoint}{adjunto} $\operatorname{adj}A =
        \operatorname{com}(A)^{\mathsf T}$:
        \[
        \dd(\det)_A(H) =
        \operatorname{tr}\bigl(\operatorname{adj}(A)\,H\bigr),
        \qquad
        \nabla(\det)(A) = \operatorname{com}(A) ,
        \]
        recuperando a questão 11 quando $ A $ é invertível
        ($\operatorname{adj}A = \det(A)A^{-1}$). Isso é
        \emph{\omterm{pb:b2:diffcalc:1}{Fórmula de Jacobi}}: $\bigl(\det
        A(t)\bigr)' = \operatorname{tr}\bigl(
        \operatorname{adj}(A(t))\,A'(t)\bigr)$.
  \item (Fórmula de Liouville) Deixe $ A(t)$ seja um $ C^1$ curva de
        matrizes que satisfazem a equação \omterm{def:b2:diffcalc:differential}{diferencial} linear
        $ A'(t) = M(t)A(t)$. Provar
        \[
        \bigl(\det A(t)\bigr)' =
        \operatorname{tr}\bigl(M(t)\bigr)\,\det A(t),
        \qquad\text{hence}\qquad
        \det A(t) = \det A(0)\,
        \exp\Bigl(\int_0^t\operatorname{tr}M\Bigr)
        \]
        \emph{(usar $\operatorname{adj}(A)\,A = \det(A)I $ e
        a invariância cíclica do traço)} --- o Wronskiano
        identidade que o capítulo de equações diferenciais irá
        use constantemente.
  \item Mostre isso $ SL_n(\R) = \{\det = 1\}$ é um nível suave
        definido: a cada $ A \in SL_n(\R)$ o \omterm{def:b2:diffcalc:differential}{diferencial}
        $\dd(\det)_A $ é um \emph{sobrejetivo} mapa linear em
        $\R $ \emph{(avalie em $ H = \frac1nA $)}.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The \omterm{ex:b2:nvs:matrixexp}{matriz exponencial}.}
\begin{enumerate}[resume]
  \item Mostre isso $\eu^A = \sum_{k\geq0}\frac{A^k}{k!}$
        converge \omterm{def:b2:series:def}{absolutamente} para cada $ A $, \omterm{def:b2:funcseq:series}{normalmente} em cada
        bola, com $\vertiii{\eu^A} \leq
        \eu^{\vertiii A}$; e isso $\eu^A $ depende
        continuamente sobre $ A $.
  \item Prove isso $ AB = BA $ implica $\eu^{A+B} =
        \eu^A\eu^B $ \emph{(Produto Cauchy, legítimo por
        convergência absoluta)}; deduzir isso $\eu^A $ é sempre
        invertível, com inverso $\eu^{-A}$: $\exp $ mapas
        $\mathcal M_n(\R)$ em $ GL_n(\R)$.
  \item Mostre isso $ t \mapsto \eu^{tA}$ é $ C^1$ (de fato
        $ C^\infty $) com
        \[
        \frac{\dd}{\dd t}\,\eu^{tA} = A\,\eu^{tA} =
        \eu^{tA}A
        \]
        \emph{(diferencie a série termo por termo em
        segmentos: a série derivada converge \omterm{def:b2:funcseq:series}{normalmente})}.
  \item Prove a identidade
        \[
        \det\bigl(\eu^{A}\bigr) = \eu^{\operatorname{tr}A}
        \]
        \emph{(aplique a fórmula de Liouville, questão 13, para
        $ A(t) = \eu^{tA}$)}. Verificações de sanidade: $ n = 1$; nilpotente
        $ A $; e as matrizes de traço zero pousam em
        $ SL_n(\R)$.
  \item Mostrar $\eu^H = I + H + O(\vertiii H^2)$, então $\exp $ é
        \omterm{def:b2:diffcalc:differential}{diferenciável} no $0$ com $\dd(\exp)_0 =
        \mathrm{id}$; conclua com a função inversa
        teorema (\cref{thm:b2:diffcalc:inverse}) que $\exp $
        é um $ C^1$ difeomorfismo de uma vizinhança de $0$
        para um bairro de $ I $: toda matriz próxima ao
        identidade tem um logaritmo.
  \item Mostre isso $\exp $ mapas \omterm{def:b2:quadratic:adjoint}{simétrico} matrizes para \omterm{def:b2:quadratic:adjoint}{simétrico}
        \emph{definido positivo} matrizes, bijetivamente
        \emph{(diagonalizar; o inverso é o espectral
        logaritmo)}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The orthogonal group as a level
set.}
\begin{enumerate}[resume]
  \item Deixar $ F(A) = A^{\mathsf T}A $, de $\mathcal M_n(\R)$ para o \omterm{def:b2:quadratic:adjoint}{simétrico} matrizes $ S_n $. Calcular $\dd F_A(H) =
        A^{\mathsf T}H + H^{\mathsf T}A $ e mostrar isso a cada
        $ A \in O_n = F^{-1}(I)$ esse \omterm{def:b2:diffcalc:differential}{diferencial} é
        \emph{sobrejetivo} para $ S_n $ \emph{(dado $ S \in S_n $,
        tentar $ H = \frac12 AS $)}: $ O_n $ é um conjunto de nível suave,
        de dimensão $ n^2 - \frac{n(n+1)}2 = \frac{n(n-1)}2$.
  \item Mostre que cada $ C^1$ curva $ A(t) \in O_n $ com $ A(0) =
        I $ tem um \emph{antisimétrico} velocidade $ A'(0)$ e
        inversamente, para antisimétrico $ K $ a curva
        $\eu^{tK}$ permanece em $ O_n $: o espaço tangente de $ O_n $
        no $ I $ são exatamente as matrizes antissimétricas.
  \item Mostrar $\det\eu^{K} = 1$ para antisimétrico $ K $ (pergunta
        18): a curva exponencial reside no grupo de rotação
        $ SO_n $. Calcule-o totalmente para $ n = 2$: com $ J =
        \begin{pmatrix}0 & -1\\ 1 & 0\end{pmatrix}$, provar
        \[
        \eu^{\theta J} = \begin{pmatrix} \cos\theta &
        -\sin\theta\\ \sin\theta & \cos\theta\end{pmatrix} :
        \]
        o \omterm{ex:b2:nvs:matrixexp}{matriz exponencial} \emph{é} a rotação por
        $\theta $, e as definições de série de cosseno e seno
        reaparecer dentro de uma matriz.
  \item (O truque da densidade) Usando a densidade de $ GL_n(\R)$
        (questão 3) e continuidade, estende-se de invertível a
        todas as matrizes a identidade
        \[
        \operatorname{adj}(AB) =
        \operatorname{adj}(B)\operatorname{adj}(A)
        \]
        \emph{(para invertível $ A, B $ ambos os lados iguais
        $\det(AB)(AB)^{-1}$; ambos os lados são polinomiais no
        entradas)}.
  \item Síntese. Em uma frase cada: (i) que anteriormente
        capítulos forneceram o motor de cada parte
        (completude e álgebras normatizadas; o espectral
        teorema; o teorema da função inversa); (ii) qual
        fórmula deste problema as equações diferenciais
        capítulo se apoiará e onde; (iii) o que
        $\dd(\det)_I = \operatorname{tr}$ e $\det\eu^A =
        \eu^{\operatorname{tr}A}$ dizer sobre rastreamento e
        determinante como ``volume infinitesimal e global'';
        (iv) o que o volume do Ano 3 faz com as perguntas
        21--23 (Grupos de Lie e suas álgebras de Lie).
\end{enumerate}
\end{problem}
