\chapter{Formas Hermitianas}\label{ch:b2:hermitian}

Espaços vetoriais complexos têm sua própria geometria de produto interno, com
uma reviravolta: linearidade em uma variável, \emph{conjugado}-linearidade em
o outro. A recompensa por aceitar a reviravolta é uma teoria espectral
ainda mais limpo que o real --- Endomorfismos hermitianos tenha real
valores próprios, unitário aqueles têm unimodular valores próprios, e ambos
diagonalizar em bases ortonormais. Este curto capítulo percorre o
Programa Euclidiano de \cref{ch:b2:quadratic} sobre $\C $.

\section{Produtos internos hermitianos}

\begin{definition}\label{def:b2:hermitian:def}
UM \emph{Produto interno hermitiano}\index{Produto interno hermitiano} ligado
um espaço vetorial complexo $ E $ é um mapa $\langle\cdot,\cdot\rangle
\colon E \times E \to \C $ que é linear na segunda variável,
\omterm{def:b2:quadratic:adjoint}{conjugado-simétrico} ($\langle y, x\rangle = \conj{\langle x,
y\rangle}$ --- portanto linear conjugado na primeira variável), e
definido positivo ($\langle x, x\rangle > 0$ para $ x \neq 0$). O
exemplo padrão em $\C^n $:
\[
\langle x, y \rangle = \sum_{i=1}^{n} \conj{x_i}\, y_i ;
\]
sobre \omterm{def:b2:metric:continuity}{contínuo} funções, $\langle f, g\rangle = \int_a^b \conj
f\,g $. \omterm{def:b2:nvs:norm}{Norma}: $\norm x = \sqrt{\langle x,x\rangle}$; um
espaço complexo de dimensão finita assim equipado é um \emph{Hermitiano
espaço}\index{Hermitiano
espaço}.
\end{definition}

\begin{example}[Primeiros cálculos]\label{ex:b2:hermitian:firstcomp}
Em $\C^2$, pegar $ x = (1+\iu,\ 2-\iu)$ e $ y = (\iu,\ 1)$.
Então
\[
\norm x^2 = \abs{1+\iu}^2 + \abs{2-\iu}^2 = 2 + 5 = 7,
\qquad
\norm y^2 = 1 + 1 = 2,
\]
e, conjugando o primeiro argumento,
\[
\langle x, y\rangle
= \conj{(1+\iu)}\,\iu + \conj{(2-\iu)}\cdot1
= (1-\iu)\iu + (2+\iu)
= (\iu + 1) + (2 + \iu) = 3 + 2\iu .
\]
Cauchy-Schwarz confere: $\abs{\langle x, y\rangle}^2 = 9 +
4 = 13 \leq 14 = \norm x^2\norm y^2$ --- perto da igualdade,
porque $ x $ está próximo de um múltiplo de $ y $. Observe também
$\langle y, x\rangle = \conj{3 + 2\iu} = 3 - 2\iu $:
simetria conjugada em ação, e a razão $\langle x,
x\rangle $ é sempre real.
\end{example}

\begin{theorem}[Cauchy-Schwarz, complexo]\label{thm:b2:hermitian:cs}
$\abs{\langle x, y\rangle} \leq \norm x \norm y $, com igualdade se
$ x, y $ são linearmente dependentes; $\norm\cdot $ é um \omterm{def:b2:nvs:norm}{norma}. Além disso
existem bases ortonormais (Gram--Schmidt é executado literalmente), com
\[
x = \sum_i \langle e_i, x\rangle\, e_i,
\qquad
\norm x^2 = \sum_i \abs{\langle e_i, x\rangle}^2 .
\]
\end{theorem}

\begin{proof}
Para $ y \neq 0$ e $ t \in \C $: $0 \leq \norm{x - ty}^2 = \norm x^2
- 2\Re\bigl(\conj t\langle y, x\rangle\bigr) + \abs t^2\norm y^2$.
Escolher $ t = \frac{\langle y, x\rangle}{\norm y^2}$:
\[
0 \leq \norm x^2 - \frac{\abs{\langle y, x\rangle}^2}{\norm y^2},
\]
qual é a desigualdade; forças de igualdade $ x = ty $. Triângulo
desigualdade, na íntegra:
\[
\norm{x + y}^2 = \norm x^2 + 2\,\Re\langle x, y\rangle
+ \norm y^2
\leq \norm x^2 + 2\,\abs{\langle x, y\rangle} + \norm y^2
\leq \bigl(\norm x + \norm y\bigr)^2 ,
\]
usando $\Re z \leq \abs z $ e depois Cauchy-Schwarz; homogeneidade
e a separação são imediatas, então $\norm\cdot $ é um \omterm{def:b2:nvs:norm}{norma}.
Gram--Schmidt: como no caso real, com conjugados
colocado pela definição (observe a convenção: nossos produtos são
linear conjugado no \emph{primeiro} slot, então as coordenadas são
$\langle e_i, x\rangle $; o próximo exemplo executa o algoritmo
uma vez na íntegra).
\end{proof}

\begin{example}[Complexo Gram-Schmidt, executado na íntegra]\label{ex:b2:hermitian:gramschmidt}
Ortonormalizar a base $ v_1 = (1, \iu)$, $ v_2 = (0, 1)$ de
$\C^2$. Primeiro vetor: $\norm{v_1}^2 = \abs1^2 + \abs\iu^2 = 2$,
então $ e_1 = \frac{1}{\sqrt2}(1, \iu)$. Projeto $ v_2$ --- com o
conjugado no primeiro slot:
\[
\langle e_1, v_2\rangle
= \frac{1}{\sqrt2}\bigl(\conj{1}\cdot0 +
\conj{\iu}\cdot1\bigr)
= \frac{-\iu}{\sqrt2} ,
\qquad
v_2 - \langle e_1, v_2\rangle e_1
= (0,1) + \frac{\iu}{2}\,(1, \iu)
= \Bigl(\frac\iu2,\ \frac12\Bigr) .
\]
Isso é \omterm{def:b2:nvs:norm}{norma} é $\sqrt{\frac14 + \frac14} = \frac{1}{\sqrt2}$:
$ e_2 = \frac{1}{\sqrt2}(\iu, 1)$. Verificar: $\langle e_1,
e_2\rangle = \frac12(\conj1\cdot\iu + \conj\iu\cdot1) =
\frac12(\iu - \iu) = 0$. Coordenadas de $ v_2$ na nova base:
$ v_2 = \langle e_1, v_2\rangle e_1 + \langle e_2, v_2\rangle
e_2$ com $\langle e_2, v_2\rangle = \frac{1}{\sqrt2}$ ---
atenção à ordem: $\langle v_2, e_1\rangle $ daria o
\emph{conjugado} coeficiente. Visão final: o algoritmo é
o euclidiano literalmente; a única armadilha é onde o
a conjugação cai e a computação $\norm{v_2 -
\text{proj}}^2 > 0$ silenciosamente usa positividade --- o axioma
que faz toda a geometria funcionar.
\end{example}

\section{Endomorfismos adjuntos, hermitianos e unitários}

\begin{definition}\label{def:b2:hermitian:adjoint}
O \emph{\omterm{def:b2:quadratic:adjoint}{adjunto}} $ u^*$ de $ u \in \mathcal{L}(E)$ é definido por
$\langle u^*(x), y\rangle = \langle x, u(y)\rangle $; em um
base ortonormal, $\operatorname{Mat}(u^*) = \conj{A}^{\mathsf T} =:
A^{\dagger}$ (conjugado transpor) --- de fato, se $ B =
(b_{ij})$ é a matriz de $ u^*$ na base ortonormal
$(e_i)$, então $ b_{ij} = \langle e_i, u^*(e_j)\rangle $, e o
definir identidade dá
\[
\conj{b_{ij}}
= \langle u^*(e_j), e_i\rangle
= \langle e_j, u(e_i)\rangle = a_{ji} ,
\qquad\text{so}\qquad
B = \conj{A}^{\mathsf T} .
\] $ u $ é
\emph{Hermitiano}\index{Hermitiano} quando $ u^* = u $
($ A^\dagger = A $), \emph{unitário}\index{unitário} quando $ u^*u =
\mathrm{id}$ ($ A^\dagger A = I $: o grupo $ U(n)$), \emph{normal}
quando $ u^*u = uu^*$.
\end{definition}

\begin{proposition}\label{prop:b2:hermitian:eigenvalues}
\omterm{def:b2:reduction:eigen}{Autovalores} de um Endomorfismo hermitiano são \emph{real};
valores próprios de um unitário endomorfismo tem \emph{módulo $1$}; em
ambos os casos autoespaços para distinto valores próprios são ortogonais.
\end{proposition}

\begin{proof}
\omterm{def:b2:hermitian:adjoint}{Hermitiano}, $ u(x) = \lambda x $, $ x \neq 0$:
\[
\lambda \norm x^2 = \langle x, u(x)\rangle = \langle u(x), x\rangle
= \conj{\langle x, u(x)\rangle} = \conj\lambda\,\norm x^2 ,
\]
então $\lambda \in \R $. Unitário: $\norm{u(x)} = \norm x $ (de $ u^*u =
\mathrm{id}$), então $\abs\lambda\norm x = \norm x $. Ortogonalidade
(\omterm{def:b2:hermitian:adjoint}{Hermitiano} caso): $\lambda\langle x, y\rangle = \langle u(x),
y\rangle = \langle x, u(y)\rangle = \mu\langle x, y\rangle $ para
vetores próprios com real $\lambda \neq \mu $. Unitário caso, em
\omterm{def:b2:metric:complete}{completo}: para $ u(x) = \lambda x $, $ u(y) = \mu y $ com $\lambda \neq
\mu $ (ambos unimodulares),
\[
\langle x, y\rangle = \langle u(x), u(y)\rangle
= \conj\lambda\mu\,\langle x, y\rangle ,
\]
e $\conj\lambda\mu = \frac{\mu}{\lambda} \neq 1$: o fator
não é $1$, então $\langle x, y\rangle = 0$.
\end{proof}

\begin{example}[Uma matriz Hermitiana distorcida, diagonalizada]\label{ex:b2:hermitian:skewexample}
$ A = \begin{pmatrix} 0 & -2\\ 2 & 0\end{pmatrix}$ satisfaz
$ A^\dagger = A^{\mathsf T} = -A $: \omterm{def:b2:hermitian:adjoint}{distorcido-Hermitiano} (também real
antisimétrico --- acima $\R $ não tem valores próprios de forma alguma).
Polinômio característico $ X^2 + 4$: valores próprios $\pm2\iu $,
puramente imaginário, como \cref{exo:b2:hermitian:9} prevê em
geral. Autovetores: $(A - 2\iu I)v = 0$ dá $ v_1 =
\frac{1}{\sqrt2}(1, \iu)$, e $ v_2 = \frac{1}{\sqrt2}(1,
-\iu)$ para $-2\iu $; eles são ortogonais:
\[
\langle v_1, v_2\rangle
= \tfrac12\bigl(\conj{1}\cdot1 +
\conj{\iu}\cdot(-\iu)\bigr)
= \tfrac12(1 - 1) = 0 .
\]
Então $ A = U\operatorname{diag}(2\iu, -2\iu)\,U^\dagger $ com $ U =
(v_1\ v_2)$ unitário. Visão final: $ H = -\iu A =
\begin{pmatrix} 0 & 2\iu\\ -2\iu & 0\end{pmatrix}$ é \omterm{def:b2:hermitian:adjoint}{Hermitiano}
com o verdadeiro \omterm{def:b2:reduction:eigen}{espectro} $\{\pm2\}$ e o \emph{mesmo}
vetores próprios --- a bijeção $ u \mapsto \iu u $ entre
\omterm{def:b2:hermitian:adjoint}{Hermitiano} e \omterm{def:b2:hermitian:adjoint}{distorcido-Hermitiano} endomorfismos
(\cref{exo:b2:hermitian:9}), visto matriz por matriz; sobre $\R $
o mesmo $ A $ é uma escala de rotação sem vetores próprios em
tudo, e apenas a passagem para $\C $ revela sua forma normal.
\end{example}

\begin{theorem}[Teorema espectral hermitiano]\label{thm:b2:hermitian:spectral}
Todo Endomorfismo hermitiano de um Espaço \omterm{def:b2:hermitian:adjoint}{Hermitiano} tem uma ortonormal
base de vetores próprios (com verdadeiro valores próprios): $ A^\dagger = A $
implica $ A = U D U^{\dagger}$ com $ U \in U(n)$ e $ D $ verdadeiro
diagonal.\index{teorema espectral! Hermitiano}
\end{theorem}

\begin{proof}
Sobre $\C $, o polinômio característico divisões: um vetor próprio
$ e_1$ existe (\cref{ch:b2:reduction}) --- não compacidade argumento
necessária, uma vantagem $\C $. Normalize isso. É ortogonal
complemento $ F = e_1^\perp $ é estável: para $ x \perp e_1$,
\[
\langle e_1, u(x)\rangle = \langle u(e_1), x\rangle
= \lambda_1\langle e_1, x\rangle = 0
\]
($\lambda_1$ real). A restrição é \omterm{def:b2:hermitian:adjoint}{Hermitiano}; induzir no
dimensionar e concatenar. Em detalhe: a restrição $ u|_F $ é
um endomorfismo do Espaço \omterm{def:b2:hermitian:adjoint}{Hermitiano} $ F $ (dimensão $ n - 1$)
com $\langle u|_F(x), y\rangle = \langle x, u|_F(y)\rangle $
herdado de $ u $; a hipótese de indução fornece uma
base ortonormal $(e_2, \dots, e_n)$ de $ F $ de vetores próprios,
e $(e_1, e_2, \dots, e_n)$ é ortonormal em $ E $ ($ e_1 \perp
F $) e feito de vetores próprios de $ u $. Tradução de matriz: o
colunas de $ U $ são os $ e_i $, $ U^\dagger U = I $ expressa seu
ortonormalidade, e $ AU = UD $ coleta o \omterm{def:b2:reduction:eigen}{autovalor}
equações, de onde $ A = UDU^\dagger $ com $ D $ diagonal real
(\cref{prop:b2:hermitian:eigenvalues}).
\end{proof}

\begin{example}\label{ex:b2:hermitian:pauli}
$ A = \begin{pmatrix} 0 & -\iu\\ \iu & 0\end{pmatrix}$ é \omterm{def:b2:hermitian:adjoint}{Hermitiano}
($ A^\dagger = A $): valores próprios de $\chi_A = X^2 - 1$: $\pm 1$
(real, como prometido), com ortonormal vetores próprios
$\frac{1}{\sqrt2}(1, \iu)^{\mathsf T}$ e $\frac{1}{\sqrt2}(1,
-\iu)^{\mathsf T}$. (Os físicos sabem $ A $ como matriz de Pauli; o
realidade de \omterm{def:b2:hermitian:adjoint}{Hermitiano} \omterm{def:b2:reduction:eigen}{espectros} é o motivo pelo qual os observáveis quânticos são modelados
por \omterm{def:b2:hermitian:adjoint}{Hermitiano} operadores.)
\end{example}

\begin{example}[Uma matriz Hermitiana definida positiva, trabalhada]\label{ex:b2:hermitian:pdrun}
$ A = \begin{pmatrix} 2 & 1-\iu\\ 1+\iu & 3\end{pmatrix}$:
\omterm{def:b2:hermitian:adjoint}{Hermitiano}, já que a diagonal é real e o fora da diagonal
as entradas são conjugadas. Polinômio característico:
\[
(2-\lambda)(3-\lambda) - \abs{1-\iu}^2
= \lambda^2 - 5\lambda + 4
= (\lambda - 1)(\lambda - 4) :
\]
\omterm{def:b2:reduction:eigen}{espectro} $\{1, 4\}$, real e positivo --- $ A $ é positivo
definitivo. Autovetores: para $\lambda = 4$, o sistema $(A -
4I)v = 0$ dá $ v_4 = (1 - \iu,\ 2)$ (verifique a segunda linha:
$(1+\iu)(1-\iu) - 2 = 0$); para $\lambda = 1$, $ v_1 = (1 - \iu,\
-1)$. Ortogonalidade, com o conjugado no primeiro slot:
\[
\langle v_4, v_1\rangle
= \conj{(1-\iu)}\,(1-\iu) + \conj{2}\,(-1)
= 2 - 2 = 0 . \checkmark
\]
Normalizando ($\norm{v_4}^2 = 2 + 4 = 6$, $\norm{v_1}^2 = 2 + 1
= 3$) dá o unitário $ U = \bigl(\frac{v_4}{\sqrt6}\
\frac{v_1}{\sqrt3}\bigr)$ com $ A =
U\operatorname{diag}(4,1)U^\dagger $. Visão final: o
A leitura de Rayleigh é imediata --- na esfera unitária de
$\C^2$, $\langle x, Ax\rangle $ varia acima $\intcc{1}{4}$,
alcançado nos dois vetores próprios; este é o $ n = 2$ germe de
a teoria Courant-Fischer construída no problema do fim de semana.
Verifique se o formulário é real vetores próprios também: em
$ x = (1, \iu)$,
\[
Ax = \bigl(2 + (1-\iu)\iu,\ (1+\iu) + 3\iu\bigr)
= (3 + \iu,\ 1 + 4\iu),
\]
\[
\langle x, Ax\rangle = \conj{1}\,(3+\iu) +
\conj{\iu}\,(1+4\iu)
= (3 + \iu) + (-\iu)(1 + 4\iu) = 3 + \iu - \iu + 4 = 7 \in
\R ,
\]
como \cref{prop:b2:hermitian:eigenvalues}mecanismo de prova
(simetria conjugada contra $ A^\dagger = A $) garantias para
cada $ x $.
\end{example}

\begin{example}[Uma matriz unitária diagonalizada]\label{ex:b2:hermitian:unitary}
$ U = \frac{1}{\sqrt2}\begin{pmatrix} 1 & \iu\\ \iu & 1
\end{pmatrix}$ (unitário por \cref{exo:b2:hermitian:2}). Isso é
polinômio característico é $\bigl(X - \frac{1}{\sqrt2}\bigr)^2
+ \frac12$, com raízes
\[
\lambda_\pm = \frac{1 \pm \iu}{\sqrt2} = \eu^{\pm\iu\pi/4},
\]
unimodular como \cref{prop:b2:hermitian:eigenvalues} prometido, e
ortonormal vetores próprios $\frac{1}{\sqrt2}(1, \pm1)$. Então $ U =
V\operatorname{diag}(\eu^{\iu\pi/4},
\eu^{-\iu\pi/4})V^\dagger $: na base certa, $ U $ é um par de
rotações planas por $\pm\frac\pi4$ --- uma matriz de rotação real tem
não é real vetores próprios, mas acabou $\C $ ele se divide em dois
escalares unimodulares. Visão final: \omterm{def:b2:hermitian:adjoint}{Hermitiano} \omterm{def:b2:reduction:eigen}{espectros} ao vivo
a verdadeira linha, unitário \omterm{def:b2:reduction:eigen}{espectros} no círculo unitário; ambos são
sombras da mesma normalidade, e a transformada de Cayley de
\cref{exo:b2:hermitian:6} mapeia uma imagem para a outra.
\end{example}

\begin{example}[A transformada de Cayley, calculada]\label{ex:b2:hermitian:cayleyrun}
Correr \cref{exo:b2:hermitian:6} sobre $ H = \begin{pmatrix} 0 & 1\\
1 & 0\end{pmatrix}$ (\omterm{def:b2:hermitian:adjoint}{Hermitiano}, \omterm{def:b2:reduction:eigen}{espectro} $\{1, -1\}$,
ortonormal vetores próprios $\frac{1}{\sqrt2}(1, \pm1)$). No
eigenbasis tudo é escalar: a transformação $\lambda
\mapsto \frac{\lambda - \iu}{\lambda + \iu}$ envia
\[
1 \longmapsto \frac{1 - \iu}{1 + \iu} = -\iu,
\qquad
-1 \longmapsto \frac{-1 - \iu}{-1 + \iu} = \iu ,
\]
(multiplique pelo conjugado do denominador), então $ U = (H -
\iu I)(H + \iu I)^{-1}$ é o unitário com valores próprios
$\mp\iu $ naqueles mesmos vetores próprios:
\[
U = \frac{1}{2}\begin{pmatrix} 1 & 1\\ 1 & -1\end{pmatrix}
\begin{pmatrix} -\iu & 0\\ 0 & \iu\end{pmatrix}
\begin{pmatrix} 1 & 1\\ 1 & -1\end{pmatrix}
= \begin{pmatrix} 0 & -\iu\\ -\iu & 0\end{pmatrix} .
\]
Verificar: $ U^\dagger U = I $, e $1 \notin \operatorname{Sp}U =
\{\pm\iu\}$, como promete a teoria. Visão final: o real
a linha é mapeada no círculo unitário menos o ponto $1$ ---
\omterm{def:b2:reduction:eigen}{autovalor} por \omterm{def:b2:reduction:eigen}{autovalor}, a transformada de Cayley é a
Mmapa de Óbius $\frac{\lambda-\iu}{\lambda+\iu}$ e matrizes
basta seguir seus \omterm{def:b2:reduction:eigen}{espectros}.
\end{example}

\begin{remark}[Armadilhas comuns]
\emph{(i) Onde a barra cai:} este livro conjuga o
\emph{primeiro} slot, então as coordenadas são $\langle e_i, x\rangle $
e $\langle\lambda x, y\rangle = \conj\lambda\langle x,
y\rangle $; muitos textos conjugam o segundo slot ---
traduzir antes de comparar fórmulas ou sinais de $\iu $ vá
errado silenciosamente. \emph{(ii) Complexo \omterm{def:b2:quadratic:def}{polarização} é mais forte:}
sobre $\C $, se $\langle x, u(x)\rangle = 0$ para \emph{todos} $ x $
então $ u = 0$ (expandir $ x + y $ e $ x + \iu y $: tanto o real
e partes imaginárias de $\langle x, u(y)\rangle $ desaparecer); sobre
$\R $ isso falha --- a rotação por $\frac\pi2$ satisfaz
$\langle x, u(x)\rangle = 0$ em todos os lugares. Consequentemente, mais
$\C $ apenas, ``$\langle x, u(x)\rangle \in \R $ para todos $ x $''
já força $ u $ \omterm{def:b2:hermitian:adjoint}{Hermitiano}. \emph{(iii) O verdadeiro normal não é
\omterm{def:b2:reduction:diag}{diagonalizável}:} a matriz de
\cref{ex:b2:hermitian:skewexample} é normal, mas não tem real
\omterm{def:b2:reduction:eigen}{autovalor}; unitário diagonalização é um teorema \emph{sobre}
$\C $ e mais $\R $ só se obtém reduções de bloco.
\emph{(iv) Verificação da unitariedade:} $ U^\dagger U = I $ significa o
\emph{colunas} são ortonormais para \omterm{def:b2:hermitian:adjoint}{Hermitiano} produto ---
testando $ UU^{\mathsf T}$, ou esquecendo a conjugação no
produtos de coluna, são as duas maneiras clássicas de certificar um erro
matriz.
\end{remark}

\begin{example}[As isometrias são exatamente as unitárias]\label{ex:b2:hermitian:isometry}
\omterm{def:b2:nvs:norm}{Norma} a preservação parece mais fraca que a unitariedade, mas $\C $ isso
não é: se $\norm{u(x)} = \norm x $ para todos $ x $, então $ u^*u =
\mathrm{id}$. De fato $ v = u^*u - \mathrm{id}$ é \omterm{def:b2:hermitian:adjoint}{Hermitiano} e
satisfaz $\langle x, v(x)\rangle = \norm{u(x)}^2 - \norm x^2 =
0$ para cada $ x $; por complexo \omterm{def:b2:quadratic:def}{polarização} (armadilha (ii)
acima), um mapa com ``diagonal'' com desaparecimento idêntico é zero:
$ v = 0$. Concretamente, o \omterm{def:b2:quadratic:def}{polarização} corre
\[
0 = \langle x + y, v(x+y)\rangle
= \langle x, v(y)\rangle + \langle y, v(x)\rangle,
\qquad
0 = \langle x + \iu y, v(x + \iu y)\rangle
= \iu\langle x, v(y)\rangle - \iu\langle y, v(x)\rangle ,
\]
e as duas linhas juntas forçam $\langle x, v(y)\rangle = 0$
para todos $ x, y $. Visão final: é por isso que ``unitário'' pode
ser verificada medindo apenas os comprimentos --- rigidez que o
O capítulo de Fourier irá explorar, onde preservar a energia
$\norm f_2$ (Parseval) é o mesmo que preservar todos os
produtos de coeficientes.
\end{example}

\begin{remark}[Perspectivas dentro deste volume]
O \omterm{def:b2:hermitian:adjoint}{Hermitiano} o maquinário construído aqui é consumido quase
imediatamente. O capítulo de Fourier \emph{é} \omterm{def:b2:hermitian:adjoint}{Hermitiano} geometria
em dimensão infinita: as exponenciais $(e_n)$ são um
família ortonormal para $\langle f, g\rangle =
\frac{1}{2\pi}\int\conj fg $, a desigualdade de Bessel é a
estimativa de projeção deste capítulo
\cref{thm:b2:hermitian:cs}, e Parseval é seu limite
igualdade. A transformada finita de Fourier
(\cref{exo:b2:hermitian:10}) reaparece sempre que a convolução
deve ser diagonalizado. E o problema de fim de semana deste capítulo ---
Courant - Fischer, Weyl, entrelaçando --- fornece o
\omterm{def:b2:reduction:eigen}{autovalor} estabilidade que o capítulo de equações diferenciais
invoca quando afirma que pequenas perturbações de um sistema
mova suas frequências apenas ligeiramente. Para trás, tudo aqui
é o espelho complexo do capítulo das formas quadráticas: mantenha o
dois dicionários lado a lado ($ A^{\mathsf T} \leftrightarrow
A^\dagger $, ortogonal $\leftrightarrow $ unitário, Rayleigh
real em ambos).
\end{remark}

\begin{remark}[Endomorfismos normais]
Sobre $\C $ a afirmação definitiva é: $ u $ é \emph{unitariamente
\omterm{def:b2:reduction:diag}{diagonalizável} se e somente se for normal} ($ u^*u = uu^*$) ---
cobertura \omterm{def:b2:hermitian:adjoint}{Hermitiano}, unitário e \omterm{def:b2:hermitian:adjoint}{distorcido-Hermitiano} mapas de uma só vez. O
a prova é um agradável reforço do argumento acima
(\cref{exo:b2:hermitian:8}). Sobre $\R $, por outro lado, apenas normalidade
compra diagonalização de blocos (blocos de rotação): geometria complexa é
genuinamente mais simples.
\end{remark}

\begin{remark}[Onde isso é usado]
\omterm{def:b2:hermitian:adjoint}{Hermitiano} teoria espectral é a matemática do quantum
mecânica: os observáveis são modelados por \omterm{def:b2:hermitian:adjoint}{Hermitiano} operadores
(\omterm{def:b2:reduction:eigen}{espectros} reais = valores mensuráveis), evolução temporal por unitário
aqueles (\omterm{def:b2:nvs:norm}{norma} preservação = conservação da probabilidade). Dentro
neste livro, o capítulo de Fourier baseia-se na ortonormalidade de
as exponenciais --- a \omterm{def:b2:hermitian:adjoint}{Hermitiano} declaração do produto interno --- e
a diagonalização de matrizes circulantes
(\cref{exo:b2:hermitian:10}) é a transformada finita de Fourier.
O problema de fim de semana desenvolve o cálculo variacional de
valores próprios (Courant - Fischer, Weyl, entrelaçado), o diário
pão de análise numérica e física matemática; o ano
3 volume estende-o para \omterm{def:b2:metric:compact}{compactar} \omterm{def:b2:quadratic:adjoint}{auto-adjunto} operadores ativados
Espaços de Hilbert.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b2:hermitian:1}
Sobre $\C^2$: calcular $\langle x, y\rangle $, $\norm x $, $\norm y $ para
$ x = (1, \iu)$, $ y = (\iu, 1)$; eles são ortogonais? Dê um
base ortonormal contendo $\frac{x}{\norm x}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:hermitian:2}
Quais são \omterm{def:b2:hermitian:adjoint}{Hermitiano}? unitário? normal?
\[
\begin{pmatrix} 1 & \iu\\ -\iu & 2 \end{pmatrix},
\qquad
\frac{1}{\sqrt2}\begin{pmatrix} 1 & \iu\\ \iu & 1\end{pmatrix},
\qquad
\begin{pmatrix} 0 & 1\\ 0 & 0 \end{pmatrix}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:hermitian:3}
Prove que uma matriz $ A \in \mathcal{M}_n(\C)$ escreve exclusivamente $ A =
H + \iu K $ com $ H, K $ \omterm{def:b2:hermitian:adjoint}{Hermitiano} \emph{(o ``real e imaginário
peças'' $ H = \frac{A + A^\dagger}{2}$, $ K = \frac{A -
A^\dagger}{2\iu}$)}, e isso $ A $ é normal se $ H $ e $ K $
trajeto.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:hermitian:4}
Diagonalizar em uma base ortonormal: $ A = \begin{pmatrix} 2 & \iu\\
-\iu & 2\end{pmatrix}$ e calcular $ A^k $ para $ k \in \N $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:hermitian:5}
Prove isso $ U(n)$ é \omterm{def:b2:metric:compact}{compactar}, e que o \omterm{def:b2:reduction:eigen}{autovalor} mapa é
em: todo unimodular $\lambda $ surge para alguns unitário matriz.
Prove isso $\det U \in \mathbb{U}$ (o círculo unitário) para $ U \in
U(n)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:hermitian:6}
(transformação de Cayley) Deixe $ H $ ser \omterm{def:b2:hermitian:adjoint}{Hermitiano}. Prove isso $ H + \iu I $ é
invertível e isso $ U = (H - \iu I)(H + \iu I)^{-1}$ é unitário,
com $1 \notin \operatorname{Sp}(U)$. \emph{(Trabalhar espectralmente: em um
base própria de $ H $, tudo é escalar.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:hermitian:7}
Para $ A $ \omterm{def:b2:hermitian:adjoint}{Hermitiano} definido positivo ($\langle x, Ax\rangle > 0$ para
$ x \neq 0$), prove que $\operatorname{Sp}(A) \subseteq
\intoo{0}{\infty}$, que $ A = B^2$ por um \omterm{def:b2:hermitian:adjoint}{Hermitiano} positivo
definitivo $ B $, e isso $\det A > 0$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:hermitian:8}
(Teorema espectral para endomorfismos normais) Deixe $ u $ ser normal em um
Espaço \omterm{def:b2:hermitian:adjoint}{Hermitiano}.
\begin{enumerate}
  \item Provar $\norm{u(x)} = \norm{u^*(x)}$ para todos $ x $, e deduza
        $\ker(u - \lambda) = \ker(u^* - \conj\lambda)$.
  \item Prove isso autoespaços de $ u $ para distinto valores próprios são
        ortogonal, e que o complemento ortogonal de um
        autoespaço é $ u $-estável.
  \item Conclua por indução que $ u $ é unitariamente \omterm{def:b2:reduction:diag}{diagonalizável};
        e inversamente.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:hermitian:9}
Um endomorfismo é \emph{\omterm{def:b2:hermitian:adjoint}{distorcido-Hermitiano}} quando $ u^* = -u $. Provar
que é valores próprios são puramente imaginários, que $ u \mapsto \iu
u $ é uma bijeção de \omterm{def:b2:hermitian:adjoint}{Hermitiano} para \omterm{def:b2:hermitian:adjoint}{distorcido-Hermitiano}
endomorfismos, e que \omterm{def:b2:hermitian:adjoint}{distorcido-Hermitiano} endomorfismos são
unitariamente \omterm{def:b2:reduction:diag}{diagonalizável} (\cref{exo:b2:hermitian:8}).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:hermitian:10}
(A transformada finita de Fourier) Deixe $ S $ seja o \omterm{def:b2:structures:generated}{cíclico} mudança de
$\C^n $: $ S(x_0, x_1, \dots, x_{n-1}) = (x_{n-1}, x_0, \dots,
x_{n-2})$, e $\omega = \eu^{2\iu\pi/n}$.
\begin{enumerate}
  \item Mostre isso $ S $ é unitário, e que os vetores $ f_k =
        \frac{1}{\sqrt n}\bigl(1, \omega^k, \omega^{2k}, \dots,
        \omega^{(n-1)k}\bigr)$, $0 \leq k < n $, formar um
        base ortonormal de vetores próprios: $ Sf_k = \omega^{-k}
        f_k $.
  \item Deduza que cada \emph{circulante} matriz $ C =
        \sum_{j=0}^{n-1} c_jS^j $ é normal, diagonalizada pela
        mesma base, com valores próprios $\widehat c(k) = \sum_j
        c_j\,\omega^{-jk}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:hermitian:11}
Deixar $ P $ ser um idempotente ($ P^2 = P $) endomorfismo de um \omterm{def:b2:hermitian:def}{Hermitiano
espaço}. Prove isso $ P $ é o \emph{ortogonal} projeção em
$\operatorname{im} P $ se e somente se $ P^* = P $. Dê a matriz
da projeção ortogonal sobre $\C v $ ($\norm v = 1$) e
em um subespaço com base ortonormal $(v_1, \dots, v_k)$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:hermitian:12}
(Projetores espectrais por interpolação) Deixe $ A $ ser \omterm{def:b2:hermitian:adjoint}{Hermitiano} com
\emph{distinto} valores próprios $\lambda_1, \dots, \lambda_p $ e
autoespaço decomposição $ E = \bigoplus_i E_i $. Defina o
Polinômios de Lagrange $ L_i(X) = \prod_{j\neq i}\frac{X -
\lambda_j}{\lambda_i - \lambda_j}$. Prove isso $ P_i = L_i(A)$
é a projeção ortogonal sobre $ E_i $, que $ P_iP_j = 0$ para
$ i \neq j $, $\sum_i P_i = I $, e $ A = \sum_i \lambda_iP_i $
(o \emph{decomposição espectral}); expressar $ f(A)$ para qualquer
polinômio $ f $ em termos de $ P_i $.
\end{exercise}

\section{Problema: Courant - Fischer, Weyl e o cálculo de
valores próprios}

\begin{problem}\label{pb:b2:hermitian:1}
O valores próprios de um \omterm{def:b2:hermitian:adjoint}{Hermitiano} matriz não são apenas raízes de uma
polinômio: são soluções de \emph{problemas de otimização}.
Esse ponto de vista variacional --- Quocientes de Rayleigh e o
\emph{Courant - Teorema min-max de Fischer}\index{Courant - Teorema min-max de Fischer} --- faz valores próprios comparável, estável e
computável, e este problema colhe seus frutos clássicos:
\emph{Desigualdades de perturbação de Weyl}\index{Desigualdades de perturbação de Weyl}, \emph{Entrelaçamento de Cauchy}, o Schur e Ky Fan
traçar desigualdades, a monotonicidade da raiz quadrada da matriz,
e o \omterm{def:b2:reduction:eigen}{espectro} do Laplaciano discreto. Por todo, $ A, B,
E $ são \omterm{def:b2:hermitian:adjoint}{Hermitiano} sobre $ E = \C^n $ com valores próprios listado em
ordem decrescente $\lambda_1(A) \geq \dots \geq \lambda_n(A)$,
e $ R_A(x) = \frac{\langle x, Ax\rangle}{\langle x, x\rangle}$
para $ x \neq 0$ é o \emph{Quociente de Rayleigh}\index{Quociente de Rayleigh}.

\medskip
\noindent\textbf{Part I --- Quocientes de Rayleigh and min-max.}
Corrigir uma base própria ortonormal $(e_1, \dots, e_n)$, $ Ae_i =
\lambda_ie_i $.
\begin{enumerate}
  \item Mostre isso $ R_A(x)$ é real, e isso
        \[
        \lambda_n \leq R_A(x) \leq \lambda_1
        \qquad (x \neq 0),
        \]
        ambos os limites alcançados: $\lambda_1 = \max R_A $,
        $\lambda_n = \min R_A $.
  \item Mostre que os pontos críticos de $ R_A $ são exatamente os
        vetores próprios de $ A $ \emph{(expandir $ t \mapsto R_A(x +
        tv)$ no $ t = 0$ para $ v $ arbitrário, então substitua $ v $ por
        $\iu v $)}.
  \item Deixar $ V_k = \operatorname{Vect}(e_1, \dots, e_k)$ e
        $ W_k = \operatorname{Vect}(e_k, \dots, e_n)$. Mostrar
        \[
        \min_{x \in V_k\setminus\{0\}} R_A(x) = \lambda_k
        = \max_{x \in W_k\setminus\{0\}} R_A(x) .
        \]
  \item Prove o \emph{Teorema de Courant-Fischer}: para $1 \leq
        k \leq n $,
        \[
        \lambda_k = \max_{\dim V = k}\;\min_{x \in
        V\setminus\{0\}} R_A(x)
        = \min_{\dim W = n-k+1}\;\max_{x \in
        W\setminus\{0\}} R_A(x)
        \]
        \emph{(para qualquer $ V $ de dimensão $ k $: $ V \cap W_k \neq
        \{0\}$ por Grassmann, então $\min_V R_A \leq \lambda_k $;
        a questão 3 mostra que o limite foi alcançado)}.
  \item (Monotonicidade) Escrever $ A \leq B $ quando $ B - A $ é
        positivo semidefinido. Deduza da questão 4: $ A \leq
        B $ implica $\lambda_k(A) \leq \lambda_k(B)$ para cada
        $ k $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Weyl's inequalities.}
\begin{enumerate}[resume]
  \item Mostre que os subespaços $ V, W \subseteq \C^n $ com $\dim V
        + \dim W > n $ cruzem de forma não trivial e generalizem:
        $\dim(V_1 \cap V_2 \cap V_3) \geq \dim V_1 + \dim V_2 +
        \dim V_3 - 2n $.
  \item Provar \emph{Desigualdade de Weyl}: para $ i + j - 1 \leq n $,
        \[
        \lambda_{i+j-1}(A + B) \leq \lambda_i(A) +
        \lambda_j(B)
        \]
        \emph{(intersectar os subespaços $ W_i(A)$, $ W_j(B)$ e
        $ V_{i+j-1}(A+B)$ da questão 3 e conte
        dimensões)}.
  \item Definir $\vertiii{E}_2 = \max_{\norm x = 1}\norm{Ex}$
        e mostrar $\vertiii E_2 = \max_k\abs{\lambda_k(E)}$ para
        \omterm{def:b2:hermitian:adjoint}{Hermitiano} $ E $. Deduzir \emph{A perturbação de Weyl
        teorema}:
        \[
        \bigl|\lambda_k(A + E) - \lambda_k(A)\bigr| \leq
        \vertiii{E}_2
        \qquad (1 \leq k \leq n) :
        \]
        cada \omterm{def:b2:reduction:eigen}{autovalor} é um $1$-Função \omterm{def:b2:metric:continuity}{Lipschitz} do
        matriz.
  \item (Perturbações de nível um) Deixe $ P $ ser \omterm{def:b2:hermitian:adjoint}{Hermitiano} positivo
        semidefinido de classificação $1$. Mostrar
        \[
        \lambda_k(A) \leq \lambda_k(A + P) \leq
        \lambda_{k-1}(A) \qquad (2 \leq k \leq n),
        \]
        junto com $\lambda_1(A) \leq \lambda_1(A+P)$: o
        novo valores próprios \emph{entrelaçar} os antigos.
  \item Verifique numericamente a questão 8: $ A = \begin{pmatrix} 2 &
        \iu\\ -\iu & 2\end{pmatrix}$
        (\cref{exo:b2:hermitian:4}: \omterm{def:b2:reduction:eigen}{espectro} $\{3, 1\}$) e
        $ E = \begin{pmatrix} 0 & 1\\ 1 & 0\end{pmatrix}$
        (\omterm{def:b2:reduction:eigen}{espectro} $\{1, -1\}$): calcule o \omterm{def:b2:reduction:eigen}{espectro} de $ A +
        E $ e ambos os lados da desigualdade.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Interlacing and trace
inequalities.}
\begin{enumerate}[resume]
  \item (Entrelaçamento Cauchy) Deixe $ B $ seja o líder $(n-1)
        \times(n-1)$ submatriz principal de $ A $. Provar
        \[
        \lambda_{k+1}(A) \leq \lambda_k(B) \leq \lambda_k(A)
        \qquad (1 \leq k \leq n-1)
        \]
        \emph{(visualizar $\C^{n-1} \subseteq \C^n $; nisso, $ R_B $ é
        a restrição de $ R_A $; aplicar Courant--Fischer em
        ambos os níveis)}.
  \item Iterar: para uma submatriz principal $ B $ de tamanho $ n - m $,
        $\lambda_{k+m}(A) \leq \lambda_k(B) \leq
        \lambda_k(A)$.
  \item (Schur) Deixe $ d_1 \geq d_2 \geq \dots \geq d_n $ seja o
        entradas diagonais de $ A $, classificado. Prove, para cada $ k $:
        \[
        \sum_{i=1}^{k} d_i \leq \sum_{i=1}^{k}\lambda_i(A),
        \]
        com igualdade em $ k = n $ (o traço)
        \emph{(o $ k $ entradas diagonais escolhidas formam um
        diretor $ k\times k $ submatriz; limitou seu traço por
        questão 12)}.
  \item (Ky Fan) Prove
        \[
        \sum_{i=1}^{k}\lambda_i(A)
        = \max\Bigl\{\sum_{i=1}^{k}\langle x_i, Ax_i\rangle
        : (x_1, \dots, x_k) \text{ orthonormal}\Bigr\} .
        \]
  \item Verifique as questões 11 e 13 em
        \[
        A = \begin{pmatrix} 2 & 1 & 0\\ 1 & 2 & 1\\
        0 & 1 & 2\end{pmatrix}
        \qquad
        \bigl(\operatorname{Sp} = \{2 - \sqrt2,\ 2,\
        2+\sqrt2\}\bigr)
        \]
        contra a sua liderança $2\times2$ bloquear (\omterm{def:b2:reduction:eigen}{espectro} $\{1,
        3\}$) e sua diagonal.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The Loewner order.} $ A \leq B $
ainda significa $ B - A $ positivo semidefinido; todas as matrizes neste
parte são \omterm{def:b2:hermitian:adjoint}{Hermitiano}.
\begin{enumerate}[resume]
  \item Mostrar: $ A \leq B $ implica $ a_{ii} \leq b_{ii}$ para todos
        $ i $, $\operatorname{tr} A \leq \operatorname{tr} B $,
        e $ C^\dagger AC \leq C^\dagger BC $ para \emph{todo}
        matriz complexa $ C $.
  \item Mostre que a quadratura é \emph{não} monótono: para
        \[
        A = \begin{pmatrix} 1 & 0\\ 0 & 0\end{pmatrix},
        \qquad
        B = \begin{pmatrix} 2 & 1\\ 1 & 1\end{pmatrix},
        \]
        verificar $0 \leq A \leq B $ mas $ A^2 \not\leq B^2$.
  \item Prove que a raiz quadrada \emph{é} monótono: $0 \leq
        A \leq B $ implica $\sqrt A \leq \sqrt B $ \emph{(deixar
        $\mu $ feijão \omterm{def:b2:reduction:eigen}{autovalor} de $\sqrt B - \sqrt A $ com unidade
        vetor próprio $ v $; calcular $\langle v, (B - A)v\rangle =
        \mu\bigl(\langle v, \sqrt B\,v\rangle + \langle v,
        \sqrt A\,v\rangle\bigr)$ e discutir)}.
  \item Prove que a inversão é antítona em definido positivo
        matrizes: $0 < A \leq B $ implica $ B^{-1} \leq A^{-1}$
        \emph{(congruente por $ A^{-1/2}$ reduzir para $ I \leq M
        \Rightarrow M^{-1} \leq I $, que é escalar em um
        base espectral)}.
  \item Deixar $ A, B $ seja positivo definido. Mostre que o
        valores próprios de $ AB $ (não \omterm{def:b2:hermitian:adjoint}{Hermitiano} em geral!) são
        real e positivo, e que
        \[
        \lambda_{\max}(AB) \leq
        \lambda_{\max}(A)\,\lambda_{\max}(B)
        \]
        \emph{(conjugado por $\sqrt A $: $ AB \sim \sqrt
        A\,B\sqrt A $)}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The discrete Laplacian, worked.}
Deixar $ T_n $ seja o $ n \times n $ matriz tridiagonal com $2$ ligado
a diagonal e $-1$ nas duas diagonais adjacentes.
\begin{enumerate}[resume]
  \item Com $\theta_k = \frac{k\pi}{n+1}$, verifique se o
        vetores $ v_k = \bigl(\sin(j\theta_k)\bigr)_{1\leq j\leq
        n}$ satisfazer $ T_nv_k = (2 - 2\cos\theta_k)\,v_k $
        \emph{(identidade produto-soma; verifique as linhas limite
        $ j = 1, n $)}. Concluir:
        \[
        \operatorname{Sp}(T_n) = \Bigl\{4\sin^2
        \frac{k\pi}{2(n+1)} : 1 \leq k \leq n\Bigr\},
        \]
        tudo simples, tudo positivo: $ T_n $ é positivo definido.
  \item (Um potencial) Para uma diagonal real $ D =
        \operatorname{diag}(d_1, \dots, d_n)$, sanduíche o
        \omterm{def:b2:reduction:eigen}{espectro}: para cada $ k $,
        \[
        \lambda_k(T_n) + \min_i d_i \;\leq\;
        \lambda_k(T_n + D) \;\leq\; \lambda_k(T_n) + \max_i
        d_i .
        \]
  \item Verifique o entrelaçamento de Cauchy entre $ T_3$ e $ T_2$
        explicitamente (\omterm{def:b2:reduction:eigen}{espectros} $\{2 \pm \sqrt2, 2\}$ e $\{1,
        3\}$) e interpretar: $ T_2$ é $ T_3$ com um ponto final
        do caminho removido.
  \item Mostre o extremo valores próprios satisfazer, como $ n \to
        \infty $:
        \[
        \lambda_{\min}(T_n) = 4\sin^2\frac{\pi}{2(n+1)}
        \sim \frac{\pi^2}{(n+1)^2},
        \qquad
        \lambda_{\max}(T_n) \to 4 ,
        \]
        então o número da condição $\kappa_n =
        \lambda_{\max}/\lambda_{\min}$ cresce como
        $\frac{4(n+1)^2}{\pi^2}$: discretizando um segundo
        derivada em uma grade cada vez mais fina é
        intrinsecamente mal condicionado.
  \item Síntese. Em uma frase cada: (i) por que o
        caracterização variacional, não a \omterm{def:b2:reduction:charpoly}{característica
        polinômio}, é o que faz valores próprios estável
        (questões 8 a 9); (ii) quais perguntas usadas apenas
        $\lambda_1 = \max R_A$ e que precisava do pleno
        mínimo-máximo; (iii) o que a ordem de Loewner acrescenta ao
        história; (iv) onde essas ferramentas reaparecem (número
        análise das matrizes de rigidez da questão 24; quântico
        teoria das perturbações; e, no volume do Ano 3, o
        princípio min-max para \omterm{def:b2:metric:compact}{compactar} \omterm{def:b2:quadratic:adjoint}{auto-adjunto}
        operadores).
\end{enumerate}
\end{problem}
