\chapter{Equações Diferenciais}\label{ch:b2:diffeq}

O ano 1 resolveu as equações lineares que admitem fórmulas. Este capítulo
fornece o que as fórmulas não podem: o \emph{Teorema de \omterm{def:b2:metric:continuity}{Cauchy-Lipschitz}}
--- existência e singularidade para $y' = f(t, y)$ --- comprovado com o
Teorema do ponto fixo de Banach, exatamente como prometido em
\cref{ch:b2:metric}; então a teoria completa de \emph{linear
sistemas} $ X' = A(t)X + B(t)$, com o \omterm{ex:b2:nvs:matrixexp}{matriz exponencial} e o
wronskiano como motores computacionais.

\section{O teorema de Cauchy-Lipschitz}

\begin{theorem}[Cauchy--Lipschitz, versão global de Lipschitz]\label{thm:b2:diffeq:cauchylipschitz}
Deixar $ I $ ser um segmento, $ f \colon I \times \R^n \to \R^n $ \omterm{def:b2:metric:continuity}{contínuo}
e \emph{\omterm{def:b2:metric:continuity}{Lipschitz} na segunda variável}, \omterm{def:b2:funcseq:def}{uniformemente} no
primeiro: $\norm{f(t, y) - f(t, z)} \leq k\,\norm{y - z}$ para todos $ t
\in I $. Então para cada $(t_0, y_0) \in I \times \R^n $, o Cauchy
problema\index{Teorema de Cauchy-Lipschitz}
\[
y' = f(t, y), \qquad y(t_0) = y_0
\]
tem exatamente uma solução $ y \colon I \to \R^n $ de classe $ C^1$.
\end{theorem}

\begin{proof}
\emph{Reformulação.} UM \omterm{def:b2:metric:continuity}{contínuo} $ y $ resolve o problema se for
satisfaz a equação integral
\[
y(t) = y_0 + \int_{t_0}^{t} f\bigl(s, y(s)\bigr)\,\dd s
=: T(y)(t)
\]
(teorema fundamental do cálculo em ambos os sentidos; um \omterm{def:b2:metric:continuity}{contínuo} solução
da equação integral é automaticamente $ C^1$).

\emph{Uma contração, depois de voltar a normalizar.} No \omterm{def:b2:nvs:banach}{Espaço Banach} $ E =
C(I, \R^n)$ com o \emph{ponderado} \omterm{def:b2:nvs:norm}{norma}
\[
N(y) = \sup_{t \in I}\; \eu^{-2k\abs{t - t_0}}\,\norm{y(t)} ,
\]
(equivalente ao sup \omterm{def:b2:nvs:norm}{norma}: o peso é limitado acima e abaixo
no segmento $ I $, então $ E $ fica \omterm{def:b2:metric:complete}{completo}), estimativa para $ y, z \in
E $ e, digamos, $ t \geq t_0$:
\[
\norm{T(y)(t) - T(z)(t)}
\leq \int_{t_0}^{t} k\,\norm{y(s) - z(s)}\,\dd s
\leq k\,N(y - z)\int_{t_0}^{t} \eu^{2k(s - t_0)}\dd s
\leq \frac{N(y-z)}{2}\,\eu^{2k(t - t_0)} .
\]
Multiplicando por $\eu^{-2k(t - t_0)}$ e tomando o sup (o caso $ t
< t_0$ é \omterm{def:b2:quadratic:adjoint}{simétrico}): $ N\bigl(T(y) - T(z)\bigr) \leq \frac12 N(y -
z)$: $ T $ é um $\frac12$-contração do \omterm{def:b2:metric:complete}{completo} $(E, N)$. O
Teorema do ponto fixo de Banach (\cref{thm:b2:metric:banach}) produz um
ponto fixo único: a solução única.
\end{proof}

\begin{remark}
Para $ f $ apenas $ C^1$ (localmente \omterm{def:b2:metric:continuity}{Lipschitz}), o teorema é válido
\emph{localmente}, com uma solução máxima em um máximo \omterm{def:b2:metric:topology}{abrir} intervalo;
soluções podem explodir em tempo finito ($ y' = y^2$, $ y(0) = 1$: $ y(t)
= \frac{1}{1-t}$, foi em $ t = 1$). O global \omterm{def:b2:metric:continuity}{Lipschitz} hipótese
é o que compra todo o segmento. Duas consequências que merecem ser gravadas:
curvas de solução de uma EDO com \omterm{def:b2:metric:continuity}{Lipschitz} campo \emph{nunca cruze};
e a função zero é a única solução que desaparece em qualquer lugar de um
equação linear homogênea.
\end{remark}

\begin{example}[A singularidade é um teorema: um campo que vaza]\label{ex:b2:diffeq:leaky}
Considerar $ y' = 2\sqrt{\abs y}$ com $ y(0) = 0$. O zero
função resolve; o mesmo acontece
\[
y(t) = \begin{cases} 0 & t \leq 0,\\ t^2 & t \geq 0,
\end{cases}
\]
o que é $ C^1$ (ambas as peças têm derivada $0$ na cola
ponto) e satisfaz $ y'(t) = 2t = 2\sqrt{t^2}$ para $ t > 0$ ---
na verdade, atrasar a descolagem constitui uma solução para \emph{todo}
hora de lançamento $ c \geq 0$: infinitas soluções através do
mesmos dados iniciais. Nenhuma contradição com
\cref{thm:b2:diffeq:cauchylipschitz}: aproximar $ y = 0$,
\[
\frac{\abs{2\sqrt y - 2\sqrt z}}{\abs{y - z}}
= \frac{2}{\sqrt y + \sqrt z} \longrightarrow +\infty ,
\]
o campo não é \omterm{def:b2:metric:continuity}{Lipschitz} em $ y $, e o teorema é silencioso.
Visão final: a leitura física é um balde que está esgotando
sob a gravidade corre para trás --- do estado vazio, um
não sei dizer quando começou a encher; o determinismo das EDOs é
exatamente o \omterm{def:b2:metric:continuity}{Lipschitz} condição, não uma lei da natureza.
\end{example}

\section{Sistemas lineares}

\begin{theorem}[Estrutura de sistemas lineares]\label{thm:b2:diffeq:linear}
Deixar $ A \colon I \to \mathcal{M}_n(\R)$ e $ B \colon I \to \R^n $ ser
\omterm{def:b2:metric:continuity}{contínuo} em um intervalo $ I $. Para cada $(t_0, X_0)$ o problema
\[
X' = A(t)X + B(t), \qquad X(t_0) = X_0
\]
tem exatamente uma solução \emph{todos} de $ I $. As soluções do
sistema homogêneo ($ B = 0$) formam um espaço vetorial $\mathcal{S}_H $ de
dimensão exatamente $ n $ e avaliação $ X \mapsto X(t_0)$ é um
isomorfismo $\mathcal{S}_H \to \R^n $; solução geral $=$
especial $+$ homogêneo.
\end{theorem}

\begin{proof}
Em cada segmento $ J \subseteq I $ contendo $ t_0$: $ f(t, X) =
A(t)X + B(t)$ é \omterm{def:b2:metric:continuity}{contínuo}, e \omterm{def:b2:metric:continuity}{Lipschitz} em $ X $ com constante $ k
= \sup_J \vertiii{A(t)}$ (finito: \omterm{def:b2:metric:continuity}{contínuo} em um segmento):
\cref{thm:b2:diffeq:cauchylipschitz} aplica-se em $ J $; deixando $ J $
escape $ I $, a exclusividade cola as soluções em uma $ I $.
A linearidade do conjunto de soluções e do mapa de avaliação são claras;
a avaliação é bijetiva por existência (sobrejetiva) e unicidade
(injetivo): $\dim \mathcal{S}_H = n $. A estrutura afim é Ano
O argumento de 1 literalmente.
\end{proof}

\begin{example}[O isomorfismo de avaliação, concretamente]\label{ex:b2:diffeq:evaliso}
Para $ y'' + y = 0$, visto como o sistema $ X' =
\begin{pmatrix} 0 & 1\\ -1 & 0\end{pmatrix}X $ com $ X = (y,
y')$: o teorema diz que o espaço de solução é um plano, e que
$ X \mapsto X(0) = (y(0), y'(0))$ é um isomorfismo sobre
$\R^2$. As soluções $\cos $ e $\sin $ avaliar para $(1, 0)$
e $(0, 1)$ --- a base canônica de $\R^2$ --- então eles se formam
uma base do espaço de solução, e \emph{todo} solução é
\[
y(t) = y(0)\cos t + y'(0)\sin t ,
\]
com os coeficientes lidos diretamente dos dados iniciais, não
sistema linear para resolver. Visão final: escolhendo o
sistema fundamental cujos valores iniciais são os canônicos
base (aqui $\cos, \sin $) está exatamente escolhendo as colunas de
$\eu^{tA}$; o isomorfismo de avaliação é o motivo pelo qual inicial
condições parametrizam trajetórias --- o conteúdo geométrico
de ``dinâmica determinística'' para equações lineares.
\end{example}

\begin{definition}[Wronskiano]\label{def:b2:diffeq:wronskian}
Para soluções $ X_1, \dots, X_n $ do sistema homogêneo, o
\emph{Wronskiano}\index{Wronskiano} é $ W(t) = \det\bigl(X_1(t),
\dots, X_n(t)\bigr)$. Pelo isomorfismo acima, ou $ W $ desaparece
de forma idêntica (a família está ligada) ou nunca (um \emph{fundamental
sistema}\index{fundamental
sistema}); quantitativamente, $ W' = \operatorname{tr}\bigl(A(t)\bigr)
W $, então
\[
W(t) = W(t_0)\,\exp\Bigl(\int_{t_0}^{t}
\operatorname{tr} A(s)\,\dd s\Bigr)
\quad \text{(Liouville's formula)}.
\]
\end{definition}

\begin{example}[Liouville verificou uma equação de Euler]\label{ex:b2:diffeq:eulerwronskian}
Sobre $\intoo{0}{\infty}$, a equação $ t^2y'' + ty' - y = 0$ tem
as soluções $ y_1(t) = t $ e $ y_2(t) = \frac1t $ (substituir).
Deles wronskiano:
\[
W(t) = \det\begin{pmatrix} t & \tfrac1t\\[2pt]
1 & -\tfrac{1}{t^2}\end{pmatrix}
= -\frac1t - \frac1t = -\frac2t ,
\]
nunca zero: a sistema fundamental. Agora verifique Liouville: em
forma normalizada $ y'' + \frac1t\,y' - \frac{1}{t^2}\,y = 0$, o
matriz complementar $ A(t) = \begin{pmatrix} 0 & 1\\ \frac{1}{t^2}
& -\frac1t\end{pmatrix}$ tem rastro $-\frac1t $, então
\[
W(t) = W(1)\exp\Bigl(-\int_1^t\frac{\dd s}{s}\Bigr)
= -2\,\eu^{-\ln t} = -\frac2t . \checkmark
\]
Visão final: Liouville prevê o \emph{forma} do
wronskiano antes que qualquer solução seja conhecida --- aqui, que $ W $ deve
ser $\frac{c}{t}$; é isso que alimenta o
método de redução de pedido
(\cref{prop:b2:diffeq:secondorder}), onde saber $ y_1$ e
o wronskianoa forma de determina $ y_2$ por uma quadratura.
\end{example}

\begin{proof}[Prova da fórmula de Liouville]
$ W(t) = \det M(t)$ com $ M' = AM $. Diferenciando o determinante
como uma função multilinear das colunas,
\[
W'(t) = \sum_j \det(X_1, \dots, X_j', \dots, X_n)
= \sum_j \det(X_1, \dots, AX_j, \dots, X_n) .
\]
Agora o mapa $(C_1, \dots, C_n) \mapsto \sum_j \det(C_1, \dots,
AC_j, \dots, C_n)$ é $ n $-linear e alternado (com dois iguais
colunas $ C_i = C_k $, os termos $ j \notin \{i, k\}$ desaparecer completamente,
e os termos $ j = i $ e $ j = k $ cancelar em pares após uma coluna
swap): pelo teorema da unicidade
(\cref{thm:b2:linalg:detspace}) isso é $ c \cdot \det $, com $ c $ leia
nas colunas canônicas: $ c = \sum_j \det(e_1, \dots, Ae_j, \dots,
e_n) = \sum_j a_{jj} = \operatorname{tr} A $. Por isso $ W' =
\operatorname{tr}\bigl(A(t)\bigr)W $: uma EDO linear escalar, resolvida por
a fórmula do Ano 1.
\end{proof}

\section{Coeficientes constantes: a matriz exponencial}

\begin{theorem}\label{thm:b2:diffeq:matrixexp}
Para $ A \in \mathcal{M}_n(\R)$ (ou $\C $), o exponencial $\eu^{tA}
= \sum_k \frac{(tA)^k}{k!}$ (\cref{ex:b2:nvs:matrixexp}) satisfaz:
$ t \mapsto \eu^{tA}$ é $ C^1$ (de fato $ C^\infty $)
com\index{matriz exponencial}
\[
\frac{\dd}{\dd t}\,\eu^{tA} = A\,\eu^{tA} = \eu^{tA}A ,
\qquad
\eu^{(s+t)A} = \eu^{sA}\,\eu^{tA},
\qquad
(\eu^{A})^{-1} = \eu^{-A} ;
\]
e $\eu^{A + B} = \eu^A\eu^B $ \emph{quando} $ AB = BA $. O Cauchy
problema $ X' = AX $, $ X(0) = X_0$ tem a solução única $ X(t) =
\eu^{tA}X_0$; com uma fonte, o \emph{variação de constantes}
fórmula contém:
\[
X(t) = \eu^{(t - t_0)A}X_0 + \int_{t_0}^{t} \eu^{(t-s)A}B(s)\,\dd
s .
\]
\end{theorem}

\begin{proof}
\emph{Diferenciabilidade:} a série $\sum \frac{t^kA^k}{k!}$ e
sua série derivada de termo $\sum \frac{t^{k-1}A^k}{(k-1)!} = A\sum
\frac{(tA)^{k-1}}{(k-1)!}$ \omterm{def:b2:integration:improper}{convergir} \omterm{def:b2:funcseq:series}{normalmente} em cada segmento
(\omterm{def:b2:nvs:norm}{normas} $\leq \frac{(\abs t\,\vertiii A)^k}{k!}$): diferenciar
termo por termo (\cref{thm:b2:funcseq:seriestransfer}, com valor vetorial).
As duas ordenações $ A\eu^{tA}$ e $\eu^{tA}A $ concordo já que cada
soma parcial comuta com $ A $.

\emph{Lei do grupo:} para deslocamento $ A, B $, o Produto Cauchy do
duas séries exponenciais reorganizadas exatamente pelo teorema binomial
como em \cref{ex:b2:series:exp} (convergência absoluta no Banach
a álgebra justifica): $\eu^{A+B} = \eu^A\eu^B $; com $ B = sA $ isso
fornece a lei do grupo de um parâmetro, e $ B = -A $ o inverso.

\emph{Problema de Cauchy:} $ X(t) = \eu^{tA}X_0$ resolve (diferenciar);
exclusividade por \cref{thm:b2:diffeq:linear}. Variação de constantes:
definir $ Y(t) = \eu^{-tA}X(t)$; diferenciando, $ Y' = \eu^{-tA}(X' -
AX) = \eu^{-tA}B(t)$; integrar de $ t_0$ para $ t $ e multiplique
de volta $\eu^{tA}$.
\end{proof}

\begin{method}[Computação $\eu^{tA}$]\label{met:b2:diffeq:computeexp}
Reduzir $ A $ (\cref{ch:b2:reduction}): se $ A = PDP^{-1}$ diagonal,
$\eu^{tA} = P\,\eu^{tD}P^{-1}$ com $\eu^{tD}$ diagonal de
$\eu^{t\lambda_i}$; em uso geral Dunford $ A = D + N $
(deslocamento): $\eu^{tA} = \eu^{tD}\,\eu^{tN}$ com $\eu^{tN}$ um
\emph{polinomial} em $ t $ (nilpotência trunca a série). Complexo
valores próprios emparelhar em blocos exponenciais de tempos de rotação
(\cref{exo:b2:diffeq:5}).
\end{method}

\begin{remark}[Armadilhas comuns]
\emph{(eu) $\eu^{A+B} \neq \eu^A\eu^B $ sem comutação:}
pegar $ A = \begin{pmatrix} 0 & 1\\ 0 & 0\end{pmatrix}$, $ B =
\begin{pmatrix} 0 & 0\\ 1 & 0\end{pmatrix}$. Então $\eu^A = I +
A $, $\eu^B = I + B $ (nilpotência), então
\[
\eu^A\eu^B = \begin{pmatrix} 2 & 1\\ 1 & 1\end{pmatrix},
\qquad\text{while}\qquad
\eu^{A+B} = \cosh(1)\,I + \sinh(1)\,(A + B)
= \begin{pmatrix} \cosh 1 & \sinh 1\\ \sinh 1 & \cosh 1
\end{pmatrix},
\]
usando $(A+B)^2 = I $; e $\cosh 1 \approx 1.54 \neq 2$. O
lei do grupo de \cref{thm:b2:diffeq:matrixexp} carrega um verdadeiro
hipótese. \emph{(ii) Intuição não linear em grama linear:}
soluções de um \emph{linear} sistema com \omterm{def:b2:metric:continuity}{contínuo}
coeficientes vivem em todo o intervalo
(\cref{thm:b2:diffeq:linear}) --- se uma solução candidata
explode por dentro $ I $, a equação não era linear ou o
o cálculo está errado; inversamente, para equações não lineares
nunca prometa globalidade sem um argumento ($ y' = y^2$).
\emph{(iii) Dividindo pelo desconhecido:} separando variáveis em
$ y' = y(1-y)$ descarta silenciosamente as soluções constantes $0$ e
$1$ --- exatamente aqueles que organizam a linha de fase
(\cref{exo:b2:diffeq:3}); liste primeiro as soluções constantes.
\emph{(iv) Vetores de correção de dados iniciais, não escalares:} um $ n $-º
ordem de equação escalar precisa $ n $ condições ($ y, y', \dots $ no
$ t_0$); apenas correspondente $ y(t_0)$ deixa um
$(n-1)$-família de parâmetros, uma fonte clássica de ``perdido''
constantes.
\end{remark}

\begin{example}[UM $3\times3$ exponencial por Dunford]\label{ex:b2:diffeq:dunford3}
Resolver $ X' = AX $ para $ A = \begin{pmatrix} 2 & 1 & 0\\ 0 & 2 &
0\\ 0 & 0 & 3\end{pmatrix}$. Dunford por blocos: $ A = D + N $
com $ D = \operatorname{diag}(2, 2, 3)$ e $ N = E_{12}$, que
deslocamento diário ($ N $ vive dentro do autovalor-$2$ bloco) e $ N^2 =
0$:
\[
\eu^{tA} = \eu^{tD}\,\eu^{tN}
= \begin{pmatrix}
\eu^{2t} & t\,\eu^{2t} & 0\\
0 & \eu^{2t} & 0\\
0 & 0 & \eu^{3t}
\end{pmatrix} .
\]
A solução geral é lida coluna por coluna: $ X(t) =
\bigl(\eu^{2t}(x_0 + ty_0),\ \eu^{2t}y_0,\
\eu^{3t}z_0\bigr)$. Verificações de sanidade: em $ t = 0$ a matriz é
$ I $; isso é determinante é $\eu^{7t} =
\eu^{t\operatorname{tr}A}$, como exige Liouville; e o
fator $ t $ aparece exatamente onde o \omterm{def:b2:reduction:eigen}{autovalor} $2$ é
com defeito. Visão final: polinômios vezes exponenciais são
não é um palpite a ser memorizado --- são as séries truncadas
$\eu^{tN}$, e seu grau é limitado pela nilpotência
índice, nunca mais.
\end{example}

\begin{method}[Resolvendo $ X' = AX + B(t)$, começar a terminar]\label{met:b2:diffeq:solvelinear}
\begin{enumerate}
  \item \omterm{def:b2:reduction:eigen}{Espectro} de $ A $; então $\eu^{tA}$ por
        \cref{met:b2:diffeq:computeexp} (diagonalizar; ou
        Dunford como em \cref{ex:b2:diffeq:dunford3}; ou um
        truque polinomial como $ A^2 = -I $).
  \item Uma solução particular: variação de constantes
        $\int_{t_0}^t\eu^{(t-s)A}B(s)\dd s $ sempre funciona;
        para polinômio exponencial $ B $, um ansatz do mesmo
        forma (grau elevado em \omterm{pb:b2:diffeq:1}{ressonância},
        \cref{exo:b2:diffeq:10}) é mais rápido.
  \item Solução geral $= \eu^{(t-t_0)A}X_0 +$ especial;
        ajuste os dados iniciais \emph{durar}, no \omterm{def:b2:metric:complete}{completo}
        fórmula.
  \item Verificações de sanidade: $ X(t_0)$ correto; a parte homogênea
        o crescimento corresponde ao \omterm{def:b2:reduction:eigen}{autovalor} peças reais
        (\cref{exo:b2:diffeq:8}); e $\det $ de um fundamental
        matriz obedece a Liouville.
\end{enumerate}
\end{method}

\begin{example}[Um retrato de fase]\label{ex:b2:diffeq:phaseportrait}
$ X' = AX $ com $ A = \begin{pmatrix} 0 & 1\\ -1 & 0\end{pmatrix}$:
$ A^2 = -I $, então a série se divide em
\[
\eu^{tA} = (\cos t)\,I + (\sin t)\,A
= \begin{pmatrix} \cos t & \sin t\\ -\sin t & \cos t
\end{pmatrix} :
\]
trajetórias são círculos que correm no sentido horário --- o oscilador harmônico
$ x'' + x = 0$ em roupas de primeira ordem. \omterm{def:b2:reduction:eigen}{Autovalores} $\pm\iu $ no
eixo imaginário: a \emph{centro}. Mais geralmente, as partes reais de
o valores próprios de $ A $ decidir o crescimento ou a decadência de $\norm{X(t)}$
(\cref{exo:b2:diffeq:8}).
\end{example}

$2$: abaixo do eixo horizontal, selas; entre o eixo e

\begin{remark}[Onde isso é usado]
Os sistemas lineares são o modelo local para tudo que não é linear:
perto de um equilíbrio, um campo vetorial suave se comporta (no
casos hiperbólicos) como sua linearização, cujo retrato o
trace - o plano determinante classifica. O problema do fim de semana funciona
a história do oscilador na íntegra --- amortecimento, forçamento,
\omterm{pb:b2:diffeq:1}{ressonância}, e teoremas de comparação de Sturm para variável
coeficientes --- a matemática por trás dos amortecedores, AC
circuitos e lacunas espectrais. O volume do Ano 3 retorna
com a teoria qualitativa (fluxos, estabilidade, primeiro
integrais) em variedades.
\end{remark}

\section{Segunda ordem com coeficientes variáveis}

\begin{proposition}\label{prop:b2:diffeq:secondorder}
A equação $ y'' + a(t)y' + b(t)y = c(t)$ ($ a, b, c $ \omterm{def:b2:metric:continuity}{contínuo} sobre
$ I $) é o sistema $ X' = A(t)X + B(t)$ para $ X = (y, y')$: soluções
existem e são únicos em todos $ I $ para quaisquer dados iniciais $(y(t_0),
y'(t_0))$; soluções homogêneas formam um plano. Se alguém não desaparecer
solução homogênea $ y_1$ é conhecido, um segundo independente é
encontrado por \emph{diminuindo a ordem}: contexto $ y = y_1 z $ vira o
equação homogênea em uma equação de primeira ordem para $ z'$, resolvido
por quadraturas.
\end{proposition}

\begin{proof}
A forma do sistema e \cref{thm:b2:diffeq:linear} dê tudo
estrutural. Redução: substituição $ y = y_1z $,
\[
y_1 z'' + (2y_1' + a y_1)z' + \underbrace{(y_1'' + ay_1' +
by_1)}_{=\,0}\,z = 0 :
\]
uma equação linear de primeira ordem em $ u = z'$, solucionável até o Ano 1
fórmula; integrando $ u $ dá $ z $, por isso $ y_2 = y_1 z $,
independente de $ y_1$ em qualquer momento $ z $ é inconstante.
\end{proof}

\begin{example}\label{ex:b2:diffeq:lowering}
$ t^2y'' - 2y = 0$ sobre $\intoo{0}{\infty}$: $ y_1 = t^2$ é um
solução. Substituir $ y = t^2z $: de $ y' = t^2z' + 2tz $ e $ y''
= t^2z'' + 4tz' + 2z $,
\[
t^2y'' - 2y = t^4 z'' + 4t^3z' = 0,
\qquad\text{i.e.}\qquad \frac{z''}{z'} = -\frac4t :
\]
$ z' = t^{-4}$ (até uma constante), $ z = -\frac{1}{3t^3}$, e $ y_2
= t^2z = -\frac{1}{3t}$. Solução geral: $ y = \alpha t^2 +
\frac{\beta}{t}$.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b2:diffeq:1}
Resolver $ X' = AX $, $ X(0) = (1, 0)^{\mathsf T}$, para $ A =
\begin{pmatrix} 1 & 1\\ 0 & 2\end{pmatrix}$ (diagonalizar) e $ A =
\begin{pmatrix} 2 & 1\\ 0 & 2 \end{pmatrix}$ (Dunford).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:diffeq:2}
Quais problemas de Cauchy têm soluções globais únicas $\R $ por
\cref{thm:b2:diffeq:cauchylipschitz}? $ y' = \sin(ty)$; $\;y' =
y^2$; $\;y' = \abs y $. Por último, resolva explicitamente com $ y(0) =
0$ e $ y(0) = 1$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:diffeq:3}
Prove que duas soluções máximas distintas de $ y' = f(t,y)$ ($ f $
\omterm{def:b2:metric:continuity}{Lipschitz} em $ y $) nunca assume o mesmo valor ao mesmo tempo, e
deduzir que soluções de $ y' = y(1 - y)$ começando em
$\intoo{0}{1}$ permanecer em $\intoo{0}{1}$ para sempre.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffeq:4}
Calcular $\eu^{tA}$ para $ A = \begin{pmatrix} 3 & 1\\ -1 &
1\end{pmatrix}$ \emph{(Dunford: $(A - 2I)^2 = 0$)}e resolva $ X' =
AX + \begin{pmatrix} \eu^{2t}\\ 0\end{pmatrix}$, $ X(0) = 0$, por
variação de constantes.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffeq:5}
Para $ A = \begin{pmatrix} \alpha & -\beta\\ \beta &
\alpha\end{pmatrix}$, provar $\eu^{tA} =
\eu^{\alpha t}\begin{pmatrix} \cos\beta t & -\sin\beta t\\
\sin\beta t & \cos\beta t\end{pmatrix}$ --- trajetórias espirais ---
duas maneiras: através da série (escrever $ A = \alpha I + \beta J $, $ J^2 =
-I $), e através da identificação complexa $ z' = (\alpha +
\iu\beta)z $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffeq:6}
(Lema de Gronwall) Seja $ u $ ser \omterm{def:b2:metric:continuity}{contínuo} não negativo com $ u(t)
\leq C + k\int_{t_0}^{t} u(s)\,\dd s $ sobre $\intco{t_0}{T}$. Provar
$ u(t) \leq C\,\eu^{k(t - t_0)}$ \emph{(diferenciar $ v(t) =
\eu^{-kt}\int_{t_0}^t u $)}. Deduza novamente a singularidade em
\omterm{def:b2:metric:continuity}{Cauchy-Lipschitz} e o \omterm{def:b2:metric:continuity}{contínuo} dependência $\norm{y(t) -
z(t)} \leq \norm{y_0 - z_0}\,\eu^{k\abs{t - t_0}}$ para dois
soluções com dados iniciais diferentes.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffeq:7}
Sabendo disso $ y_1(t) = \frac{\sin t}{t}$ resolve $ ty'' + 2y' + ty =
0$ sobre $\intoo{0}{\pi}$, encontre uma segunda solução independente por
diminuindo a ordem e dando a solução geral.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:diffeq:8}
Deixar $ A \in \mathcal{M}_n(\C)$ com tudo valores próprios de (estritamente)
parte real negativa. Prove que toda solução de $ X' = AX $ tende a
$0$ como $ t \to +\infty $, com uma taxa exponencial: $\norm{X(t)}
\leq C\,\eu^{-\alpha t}$ para alguns $\alpha > 0$.
\emph{(Trigonalize; trate o sistema triangular da última linha
up ou use Dunford: $\eu^{tA} = \eu^{tD}\eu^{tN}$ com
$\norm{\eu^{tD}} \leq \eu^{-\alpha' t}$ e $\eu^{tN}$ polinômio
em $ t $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:diffeq:9}
(Não há escapatória em tempo finito para crescimento linear) Suponha $ f $ é
\omterm{def:b2:metric:continuity}{contínuo} com $\norm{f(t, y)} \leq a\norm y + b $ sobre
$\intco{0}{\infty} \times \R^n $, localmente \omterm{def:b2:metric:continuity}{Lipschitz} em $ y $. Usando
Gronwall (\cref{exo:b2:diffeq:6}) na forma integral, prove que
soluções máximas são globais (definidas em todos os
$\intco{0}{\infty}$).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:diffeq:10}
Resolver $ y'' - 3y' + 2y = \eu^{t}$: soluções homogêneas, então um
solução particular da forma $\alpha t\,\eu^{t}$ \emph{(por que
o ingênuo adivinha $\alpha\eu^t $ falhar?)}; solução geral e
a solução com $ y(0) = y'(0) = 0$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffeq:11}
Calcular $\eu^{tA}$ para o bloco da Jordânia
\[
A = \begin{pmatrix} \lambda & 1 & 0\\ 0 & \lambda & 1\\
0 & 0 & \lambda\end{pmatrix},
\]
e descreva todas as soluções de $ X' = AX $: tempos exponenciais
vetores polinomiais, com graus até $2$. Onde é que
grau polinomial vem?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:diffeq:12}
(Forçamento periódico, resposta periódica) Deixe $ A \in
\mathcal{M}_n(\R)$ e $ B \colon \R \to \R^n $ \omterm{def:b2:metric:continuity}{contínuo} e
$ T $-periódico.
\begin{enumerate}
  \item Mostre que uma solução de $ X' = AX + B(t)$ é
        $ T $-periódico se e somente se $ X(T) = X(0)$
        \emph{(comparar $ X(\cdot + T)$ e $ X $)}.
  \item Mostre isso valores próprios de $\eu^{TA}$ são os
        $\eu^{T\lambda}$, $\lambda \in \operatorname{Sp}A $
        \emph{(trigonalizar sobre $\C $)}. Deduza: se não
        \omterm{def:b2:reduction:eigen}{autovalor} de $ A $ reside em $\frac{2\iu\pi}{T}\Z $, então
        $ I - \eu^{TA}$ é invertível.
  \item Sob essa hipótese, prove que o sistema tem
        exatamente um $ T $-solução periódica, com
        \[
        X(0) = \bigl(I - \eu^{TA}\bigr)^{-1}
        \int_0^{T}\eu^{(T-s)A}B(s)\,\dd s .
        \]
        A que corresponde o caso excluído, para o
        oscilador harmônico? (O problema do fim de semana responde:
        \omterm{pb:b2:diffeq:1}{ressonância}.)
\end{enumerate}
\end{exercise}

\section{Problema: Oscilações, ressonância e Sturm
teoremas de comparação}

\begin{problem}\label{pb:b2:diffeq:1}
Uma equação rege o mundo mecânico e elétrico:
\[
x'' + 2\zeta\omega\,x' + \omega^2 x = F(t),
\qquad \omega > 0,\ \zeta \geq 0 .
\]
Este problema o estuda completamente --- através do
traço - classificação determinante de sistemas lineares planares;
os três regimes de amortecimento, a resposta em estado estacionário a
forçamento periódico com seu \emph{ressonância}\index{ressonância}
pico e o \omterm{pb:b2:diffeq:1}{ressonância} catástrofe --- então deixa constante
coeficientes para \emph{Separação e comparação de Sturm
teoremas}\index{Separação e comparação de Sturm
teoremas}, que controlam o
zeros de soluções de $ y'' + q(t)y = 0$ sem fórmula em
tudo.

\medskip
\noindent\textbf{Part I --- The trace--determinant plane.} Deixar
$ A \in \mathcal M_2(\R)$, $\tau = \operatorname{tr}A $, $\delta
= \det A $, $\Delta = \tau^2 - 4\delta $.
\begin{enumerate}
  \item Mostre que o valores próprios de $ A $ são
        $\frac{\tau\pm\sqrt\Delta}{2}$ e classifique: dois reais
        valores próprios de sinais opostos se $\delta < 0$; real
        valores próprios do mesmo sinal se $\delta > 0$, $\Delta
        \geq 0$ (sinal de $\tau $); par conjugado não real se
        $\Delta < 0$ (parte real $\frac\tau2$).
  \item (Sela, $\delta < 0$) Com valores próprios $\mu < 0 <
        \lambda $ e vetores próprios $ v_\pm $, escreva o geral
        solução e descrever as trajetórias: duas estáveis e
        dois raios instáveis, todas as outras órbitas assintóticas a
        ambos. Por que não há solução senão $0$ fique limitado a todos
        de $\R $?
  \item (Nós, $\delta > 0$, $\Delta > 0$) Para $\mu < \lambda
        < 0$: mostre que toda solução diferente de zero tende a $0$ e
        que todas as órbitas, exceto aquelas no eixo rápido, cheguem
        \emph{tangente à autodireção lenta}
        \emph{(comparar $\eu^{\mu t}$ e $\eu^{\lambda t}$)}.
  \item (Espirais e centros, $\Delta < 0$) Escrevendo o
        valores próprios $\alpha \pm \iu\beta $, usar
        \cref{exo:b2:diffeq:5} (após uma verdadeira mudança de base,
        admitido nessa generalidade ou provado para os sistemas
        da Parte II, que são os usados abaixo) para descrever
        as órbitas: espirais convergindo para $\alpha =
        \frac\tau2 < 0$, divergindo para $\tau > 0$, fechado
        curvas (centro) para $\tau = 0$.
  \item (Casos limite) Para um duplo \omterm{def:b2:reduction:eigen}{autovalor} ($\Delta =
        0$): mostrar $\eu^{tA} = \eu^{\lambda t}(I + tN)$ com $ N
        = A - \lambda I $ nilpotente, e distinguir a estrela
        ($ N = 0$) do nó impróprio ($ N \neq 0$).
        Resuma a Parte I na imagem traço-determinante de
        figura deste capítulo.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The damped oscillator.} Agora $ F =
0$: $ x'' + 2\zeta\omega x' + \omega^2x = 0$, ou seja\ $ X' = AX $
com $ A = \begin{pmatrix} 0 & 1\\ -\omega^2 &
-2\zeta\omega\end{pmatrix}$.
\begin{enumerate}[resume]
  \item Calcular $\tau, \delta, \Delta $ e coloque os três
        regimes no plano traço-determinante:
        \emph{subamortecido} $0 < \zeta < 1$ (espiral estável),
        \emph{criticamente amortecido} $\zeta = 1$ (dobro
        \omterm{def:b2:reduction:eigen}{autovalor}), \emph{superamortecido} $\zeta > 1$ (estável
        nó); $\zeta = 0$ é o centro.
  \item Resolva os três regimes explicitamente:
        \[
        \zeta < 1:\ \eu^{-\zeta\omega t}\bigl(a\cos\omega_d t
        + b\sin\omega_dt\bigr),\ \omega_d =
        \omega\sqrt{1-\zeta^2};
        \qquad
        \zeta = 1:\ (a + bt)\,\eu^{-\omega t};
        \]
        $\zeta > 1$: duas exponenciais reais. Defina o
        pseudoperíodo $\frac{2\pi}{\omega_d}$ e mostre isso
        a proporção de máximos sucessivos de $\abs x $ é o
        constante $\eu^{-2\pi\zeta/\sqrt{1-\zeta^2}}$ (o
        decremento logarítmico).
  \item (O princípio da porta mais próxima) Para $\zeta \geq 1$ o
        a taxa de decaimento é governada pelo mais lento \omterm{def:b2:reduction:eigen}{autovalor}
        $\lambda_{\mathrm{slow}} = -\omega\bigl(\zeta -
        \sqrt{\zeta^2-1}\bigr)$. Mostre isso
        $\abs{\lambda_{\mathrm{slow}}} =
        \frac{\omega}{\zeta + \sqrt{\zeta^2 - 1}}$ é um
        \emph{diminuindo} função de $\zeta \geq 1$:
        amortecimento crítico $\zeta = 1$ dá o mais rápido
        retorno não oscilante ao repouso.
  \item (Energia) Deixe $ E(t) = \frac12x'^2 +
        \frac12\omega^2x^2$. Provar $ E' = -2\zeta\omega\,x'^2
        \leq 0$, e deduza que para $\zeta > 0$ a equação
        não tem solução periódica diferente de zero \emph{(um período seria
        força $ E $ constante, portanto $ x' \equiv 0$)}.
  \item Explique em duas frases por que o centro $\zeta = 0$
        é \emph{estruturalmente frágil}: qualquer $\zeta > 0$,
        por menor que seja, destrói a periodicidade --- e onde
        que aparece no plano determinante do traço (o
        linha central tem interior vazio).
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Forced oscillations and
\omterm{pb:b2:diffeq:1}{ressonância}.} Agora $ F(t) = F\cos(\gamma t)$ com $ F, \gamma > 0$.
\begin{enumerate}[resume]
  \item ($\zeta > 0$: o estado estacionário) Procure $ x_p =
        \Re\bigl(z\,\eu^{\iu\gamma t}\bigr)$: mostrar
        \[
        z = \frac{F}{\omega^2 - \gamma^2 +
        2\iu\zeta\omega\gamma},
        \qquad
        A(\gamma) := \abs z = \frac{F}{\sqrt{(\omega^2 -
        \gamma^2)^2 + 4\zeta^2\omega^2\gamma^2}} ,
        \]
        e escrever $ x_p = A(\gamma)\cos(\gamma t - \varphi)$
        com $\tan\varphi =
        \frac{2\zeta\omega\gamma}{\omega^2-\gamma^2}$.
  \item Mostre isso \emph{todo} solução é $ x_p $ mais um
        transitório da Parte II, que tende a $0$: tanto faz
        os dados iniciais, o sistema trava no estável
        estado --- amplitude $ A(\gamma)$, atraso de fase
        $\varphi $.
  \item (O \omterm{pb:b2:diffeq:1}{ressonância} curva) Maximizar $ A $: mostre isso
        $ A(\gamma)$ tem um máximo interior se $\zeta <
        \frac{1}{\sqrt2}$, no
        \[
        \gamma_* = \omega\sqrt{1 - 2\zeta^2},
        \qquad
        A(\gamma_*) =
        \frac{F}{2\zeta\omega^2\sqrt{1-\zeta^2}} ,
        \]
        e isso para pequenos $\zeta $ o pico amplifica o
        resposta estática $ A(0) = \frac F{\omega^2}$ pelo
        fator $\approx \frac{1}{2\zeta}$.
  \item ($\zeta = 0$, desligado \omterm{pb:b2:diffeq:1}{ressonância}) Para $\gamma \neq \omega $,
        mostre que a solução com $ x(0) = x'(0) = 0$ é
        \[
        x(t) = \frac{F}{\omega^2 -
        \gamma^2}\bigl(\cos\gamma t - \cos\omega t\bigr)
        = \frac{2F}{\omega^2-\gamma^2}
        \sin\frac{(\omega-\gamma)t}{2}
        \sin\frac{(\omega+\gamma)t}{2} :
        \]
        delimitado, com \emph{batidas} --- uma oscilação rápida
        sob um envelope lento --- quando $\gamma $ está perto de
        $\omega $.
  \item ($\zeta = 0$, \omterm{pb:b2:diffeq:1}{ressonância}) Para $\gamma = \omega $, mostrar
        isso $ x_p(t) = \frac{F}{2\omega}\,t\sin(\omega t)$ é
        uma solução, e recuperá-la como o limite da questão
        14 como $\gamma \to \omega $: a amplitude aumenta
        linearmente para sempre --- o \omterm{pb:b2:diffeq:1}{ressonância} catástrofe.
  \item (Link de Fourier) Um forçamento periódico geral se decompõe
        em harmônicos (o capítulo de Fourier); por linearidade o
        estado estacionário é a soma das respostas harmônicas. Para
        um oscilador de frequência não amortecido $\omega = 3$
        forçado pela onda quadrada de
        \cref{exo:b2:fourier:1}-type (harmônicos totalmente ímpares
        inteiros), qual harmônico ressoa? Uma frase em
        por que os engenheiros temem ondas quadradas.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Sturm's theorems.} Considerar $ y''
+ q(t)\,y = 0$ em um intervalo $ I $, $ q $ \omterm{def:b2:metric:continuity}{contínuo}. (Qualquer
equação $ y'' + ay' + by = 0$ reduz a esta forma normal por
a substituição $ y = u\exp\bigl(-\frac12\int a\bigr)$;
a questão 21 mostra uma variante do truque em ação.)
\begin{enumerate}[resume]
  \item Para duas soluções $ y_1, y_2$, mostre que wronskiano
        $ W = y_1y_2' - y_1'y_2$ é \emph{constante}, zero se sef
        as soluções são proporcionais; e que um diferente de zero
        solução tem apenas \emph{simples, isolado} zeros.
  \item (Separação de tempestade) Deixe $ y_1, y_2$ ser independente
        soluções e $ a < b $ dois zeros consecutivos de $ y_1$.
        Prove isso $ y_2$ desaparece exatamente uma vez em
        $\intoo{a}{b}$ \emph{(avalie a constante $ W $ no $ a $
        e $ b $: $ W = y_1'y_2$ lá, e $ y_1'(a)$,
        $ y_1'(b)$ têm sinais opostos)}: zeros de independentes
        soluções se entrelaçam.
  \item (Comparação Sturm) Vamos $ q_1 \leq q_2$ sobre $ I $, $ y \neq
        0$ com $ y'' + q_1y = 0$, $ z \neq 0$ com $ z'' + q_2z
        = 0$, e $ a < b $ zeros consecutivos de $ y $. Mostre isso
        $ z $ desaparece em $\intcc{a}{b}$ --- estritamente dentro se
        $ q_1 < q_2$ em algum lugar $\intoo ab $ \emph{(se $ z \neq
        0$ sobre $\intoo ab $, estudar $(yz' - y'z)' = (q_1 -
        q_2)yz $ com sinais fixos para $ y, z $ e compare o
        valores limite)}.
  \item Deduza o \emph{limites de espaçamento}: se $0 < m^2 \leq
        q(t) \leq M^2$ sobre $ I $, então quaisquer dois zeros consecutivos
        $ a < b $ de uma solução diferente de zero de $ y'' + qy = 0$
        satisfazer
        \[
        \frac{\pi}{M} \;\leq\; b - a \;\leq\; \frac{\pi}{m}
        \]
        \emph{(comparar com $ u'' + M^2u = 0$ e $ u'' + m^2u
        = 0$, cujos zeros estão espaçados $\frac\pi M $ e
        $\frac\pi m $)}. Verifique o oscilador harmônico.
  \item Transformar $ ty'' + 2y' + ty = 0$
        (\cref{exo:b2:diffeq:7}) por $ u = ty $ em $ u'' + u =
        0$, recupere suas soluções $\frac{\sin t}t $,
        $\frac{\cos t}{t}$ instantaneamente e concluir que o
        zeros de cada solução diferente de zero são espaçados exatamente
        $\pi $: A visão de mundo de Sturm --- zeros são controlados por
        o coeficiente $ q $, fórmulas ou não.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Duhamel and the boundedness
frontier.}
\begin{enumerate}[resume]
  \item (Duhamel para o oscilador) Mostre que para
        \omterm{def:b2:metric:continuity}{contínuo} $ F $, a solução de $ x'' + \omega^2x =
        F(t)$ com $ x(0) = x'(0) = 0$ é
        \[
        x(t) = \frac1\omega\int_0^t\sin\bigl(\omega(t -
        s)\bigr)F(s)\,\dd s ,
        \]
        e derivar novamente a solução ressonante da questão 15
        dele com $ F(s) = F\cos(\omega s)$
        \emph{(produto-soma)}.
  \item ($\zeta > 0$: entrada limitada, saída limitada) Mostrar
        isso para $\zeta > 0$ e \emph{qualquer} limitado \omterm{def:b2:metric:continuity}{contínuo}
        $ F $, toda solução da equação amortecida é limitada
        ligado $\intco{0}{\infty}$ \emph{(variação de constantes
        mais a decadência exponencial $\vertiii{\eu^{tA}} \leq
        C\eu^{-\alpha t}$ de \cref{exo:b2:diffeq:8})}.
  \item ($\zeta = 0$) Mostre que sem amortecimento, limitado
        forçamento periódico mantém todas as soluções limitadas
        \emph{exceto} exatamente em \omterm{pb:b2:diffeq:1}{ressonância} ($\gamma =
        \omega $, questão 15 versus questão 14): o amortecimento é
        o que transforma a fronteira de limitação em uniforme
        estabilidade.
  \item Síntese. Em uma frase cada: (i) como o
        o plano traço-determinante organiza as Partes I--II e
        onde o forçamento (Parte III) o deixa; (ii) o
        significado físico de $\gamma_*$, $ A(\gamma_*)$ e
        o fator $\frac1{2\zeta}$; (iii) o que Sturm
        os teoremas dizem que as fórmulas explícitas não podem; (iv)
        quais dois resultados deste problema o resto do
        livro será reutilizado silenciosamente (Liouville-constante
        wronskianos; estabilidade de entrada limitada).
\end{enumerate}
\end{problem}
