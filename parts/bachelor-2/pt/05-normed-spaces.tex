\chapter{Espaços vetoriais normados}\label{ch:b2:nvs}

Quando o \omterm{def:b2:metric:def}{espaço métrico} é um espaço vetorial e a distância vem de
um \omterm{def:b2:nvs:norm}{norma}, topologia e álgebra linear começam a interagir: mapas lineares
são \omterm{def:b2:metric:continuity}{contínuo} exatamente quando eles estão limitados à bola unitária,
dimensão finita força tudo \omterm{def:b2:nvs:norm}{normas} concordar, e completude voltas
séries absolutamente convergentes em séries convergentes. O
divisão de dimensão finita/infinita --- cristalizada no teorema de Riesz
--- é a lição mais profunda do capítulo.

Por todo, $E, F $ são espaços vetoriais sobre $ K = \R $ ou $\C $.

\section{Normas}

\begin{definition}\label{def:b2:nvs:norm}
UM \emph{norma}\index{norma} sobre $ E $ é um mapa $\norm{\,\cdot\,} \colon E
\to \R_+$ com, para todos $ x, y \in E $, $\lambda \in K $:
\[
\norm x = 0 \iff x = 0,
\qquad
\norm{\lambda x} = \abs\lambda\,\norm x,
\qquad
\norm{x + y} \leq \norm x + \norm y .
\]
Então $ d(x, y) = \norm{x - y}$ é uma distância, e tudo
\cref{ch:b2:metric} se aplica. A desigualdade do triângulo reverso
$\bigl|\norm x - \norm y\bigr| \leq \norm{x - y}$ faz a norma
em si $1$-Lipschitz; adição e multiplicação escalar são
\omterm{def:b2:metric:continuity}{contínuo} (estimativas $\norm{(x + y) - (x' + y')} \leq \norm{x -
x'} + \norm{y - y'}$, etc.).
\end{definition}

\begin{example}\label{ex:b2:nvs:examples}
Sobre $ K^n $:
\[
\norm{x}_1 = \sum_i \abs{x_i},
\qquad
\norm{x}_2 = \Bigl(\sum_i \abs{x_i}^2\Bigr)^{1/2},
\qquad
\norm{x}_\infty = \max_i \abs{x_i}
\]
($\norm\cdot_2$ é um \omterm{def:b2:nvs:norm}{norma} por Cauchy--Schwarz, volume do ano 1). Sobre
$ C(\intcc{a}{b})$:
\[
\norm f_\infty = \sup \abs f,
\qquad
\norm f_1 = \int_a^b \abs f,
\qquad
\norm f_2 = \Bigl(\int_a^b \abs f^2\Bigr)^{1/2},
\]
sendo os dois últimos \omterm{def:b2:nvs:norm}{normas} graças à estrita positividade do
integral e integral Cauchy--Schwarz (volume do ano 1). Em matrizes:
qualquer \omterm{def:b2:nvs:norm}{norma} sobre $\mathcal M_n(K) \simeq K^{n^2}$; o \omterm{def:b2:nvs:norm}{normas} do operador
abaixo estão os estruturalmente importantes.
\end{example}

\begin{definition}[Normas equivalentes]\label{def:b2:nvs:equivalent}
Dois \omterm{def:b2:nvs:norm}{normas} $ N_1, N_2$ sobre $ E $ são \emph{equivalente}\index{equivalente} quando há constantes $ c, C > 0$ com
\[
c\,N_1 \leq N_2 \leq C\, N_1 .
\]
Equivalente \omterm{def:b2:nvs:norm}{normas} tem o mesmo \omterm{def:b2:metric:topology}{abrir} conjuntos, o mesmo convergente e
Sequências de Cauchy, as mesmas \omterm{def:b2:metric:compact}{compactar} e \omterm{def:b2:metric:complete}{completo} subconjuntos: o mesmo
análise.
\end{definition}

\begin{example}[Não equivalência em dimensão infinita]\label{ex:b2:nvs:nonequivalent}
Sobre $ C(\intcc{0}{1})$: $\norm f_1 \leq \norm f_\infty $ sempre, mas
nenhum limite reverso é válido: $ f_n(x) = x^n $ tem $\norm{f_n}_\infty = 1$
e $\norm{f_n}_1 = \frac{1}{n+1} \to 0$. Então $ f_n \to 0$ para
$\norm\cdot_1$ mas não para $\norm\cdot_\infty $: os dois \omterm{def:b2:nvs:norm}{normas}
discordar sobre a própria convergência.
\end{example}

\begin{example}[Constantes explícitas na dimensão $ n $]\label{ex:b2:nvs:constants}
Sobre $ K^n $ os três clássicos \omterm{def:b2:nvs:norm}{normas} são equivalente com afiado
constantes:
\[
\norm x_\infty \leq \norm x_2 \leq \norm x_1
\leq \sqrt n\,\norm x_2 \leq n\,\norm x_\infty ,
\]
o limite do meio $\norm x_1 \leq \sqrt n\norm x_2$ vindo de
Cauchy-Schwarz contra o vetor de todos. Vetores extremos:
$ e_1$ torna as duas primeiras desigualdades iguais, $(1, 1,
\dots, 1)$ os dois últimos. A dimensão $ n $ fica visivelmente no
constantes --- a semente quantitativa da falha no infinito
dimensão: como $ n \to \infty $ nenhuma constante uniforme sobrevive,
que é exatamente o que \cref{ex:b2:nvs:nonequivalent} exposições em
espaços funcionais.
\end{example}

\section{Mapas lineares contínuos}

\begin{theorem}[Caracterização]\label{thm:b2:nvs:continuouslinear}
Para um mapa linear $ u \colon E \to F $ entre espaços normados, o
a seguir estão equivalente:
\begin{enumerate}
  \item $ u $ é \omterm{def:b2:metric:continuity}{contínuo};
  \item $ u $ é \omterm{def:b2:metric:continuity}{contínuo} no $0$;
  \item $ u $ é limitado pela bola unitária fechada: $\sup_{\norm x \leq
        1} \norm{u(x)} < \infty $;
  \item há $ C \geq 0$ com $\norm{u(x)} \leq C \norm x $ para
        tudo $ x $;
  \item $ u $ é \omterm{def:b2:metric:continuity}{Lipschitz}.
\end{enumerate}
O menos tal $ C $ é o \emph{norma do operador}\index{norma do operador}
$\vertiii{u} = \sup_{\norm x \leq 1}\norm{u(x)} = \sup_{x \neq 0}
\frac{\norm{u(x)}}{\norm x}$; faz o espaço $\mathcal{L}_c(E,
F)$ de \omterm{def:b2:metric:continuity}{contínuo} linear mapeia um espaço normado, com
\[
\vertiii{v \circ u} \leq \vertiii v\, \vertiii u .
\]
\end{theorem}

\begin{proof}
(1 $\Rightarrow $ 2) trivial. (2 $\Rightarrow $ 3): continuidade no $0$
com $\varepsilon = 1$ dá $\delta $ com $\norm x \leq \delta
\Rightarrow \norm{u(x)} \leq 1$; homogeneidade então escala qualquer $ x $
com $\norm x \leq 1$ descendo até aquela bola e voltando:
\[
\norm{u(x)} = \frac1\delta\,\norm{u(\delta x)} \leq
\frac1\delta ,
\]
desde $\norm{\delta x} \leq \delta $. (3 $\Rightarrow $ 4):
para $ x \neq 0$, aplique o limite a $\frac{x}{\norm x}$. (4
$\Rightarrow $ 5): $\norm{u(x) - u(y)} = \norm{u(x - y)} \leq
C\norm{x - y}$. (5 $\Rightarrow $ 1) conhecido.

\omterm{def:b2:nvs:norm}{Norma} axiomas para $\vertiii\cdot $: homogeneidade e separação são
claro ($\vertiii u = 0$ forças $ u = 0$ na bola, portanto
em todos os lugares); desigualdade triangular de $\norm{(u + v)(x)} \leq
\norm{u(x)} + \norm{v(x)}$. Submultiplicatividade: $\norm{v(u(x))}
\leq \vertiii v\,\norm{u(x)} \leq \vertiii v \vertiii u \norm x $.
\end{proof}

\begin{example}\label{ex:b2:nvs:operatornorms}
Sobre $\bigl(C(\intcc{0}{1}), \norm\cdot_\infty\bigr)$: avaliação
$ f \mapsto f(0)$ tem \omterm{thm:b2:nvs:continuouslinear}{norma do operador} $1$; integração $ f \mapsto
\int_0^1 f $ tem \omterm{def:b2:nvs:norm}{norma} $1$; o mapa $ f \mapsto \int_0^1 t f(t)\dd t $
tem \omterm{def:b2:nvs:norm}{norma} $\int_0^1 t\,\dd t = \frac12$ (limite superior pelo triângulo
desigualdade para integrais; alcançado em $ f \equiv 1$). Mas
diferenciação, de $(C^1, \norm\cdot_\infty)$ para $(C^0,
\norm\cdot_\infty)$, é \emph{não} \omterm{def:b2:metric:continuity}{contínuo}: $\norm{\sin(nx)}_\infty
= 1$ enquanto a derivada tem sup \omterm{def:b2:nvs:norm}{norma} $ n $. Linear não implica
\omterm{def:b2:metric:continuity}{contínuo} em dimensão infinita.
\end{example}

\begin{example}[Duas normas, dois veredictos sobre uma sequência]\label{ex:b2:nvs:sqrtnxn}
Sobre $ C(\intcc01)$, deixar $ g_n(x) = \sqrt{n}\,x^n $. Então
\[
\norm{g_n}_1 = \frac{\sqrt n}{n + 1} \longrightarrow 0,
\qquad
\norm{g_n}_2^2 = \frac{n}{2n + 1} \longrightarrow \frac12,
\qquad
\norm{g_n}_\infty = \sqrt n \longrightarrow \infty :
\]
uma sequência, três \omterm{def:b2:nvs:norm}{normas}, três comportamentos --- convergência para
zero, sem convergência (\omterm{def:b2:nvs:norm}{normas} estabilizar em $\frac1{\sqrt2}$ mas
o limite pontual é $0$) e explosão. Concentração de massa
perto $ x = 1$ é invisível para $\norm\cdot_1$, meio visível para
$\norm\cdot_2$, dominante para $\norm\cdot_\infty $. Em infinito
dimensão, “ele converge?” não é uma questão sobre uma
sequência: é uma questão sobre uma sequência \emph{e} um \omterm{def:b2:nvs:norm}{norma}.
\end{example}

\begin{method}[Calculando uma norma de operador]\label{met:b2:nvs:opnorm}
Sempre em dois movimentos. \emph{Limite superior:} estimativa
$\norm{u(x)}$ por $ C\norm x $ usando desigualdades triangulares,
Cauchy--Schwarz, ou limites integrais --- isso prova $\vertiii u
\leq C $. \emph{Testemunha:} exibir um determinado $ x_0 \neq 0$
com $\norm{u(x_0)} = C\norm{x_0}$ (o limite é alcançado), ou um
sequência de vetores unitários $ x_n $ com $\norm{u(x_n)} \to C $ (o
o limite é aproximado). Ambos os movimentos são obrigatórios: um limite superior
sozinho dá apenas $\vertiii u \leq C $, apenas uma testemunha
$\vertiii u \geq C $. Na dimensão infinita a testemunha pode ter
ser uma sequência --- o supremo não precisa ser alcançado
(\cref{exo:b2:nvs:8}).
\end{method}

\begin{example}[Os operadores diagonais veem todas as normas da mesma forma]\label{ex:b2:nvs:diagonalnorm}
Para $ D = \operatorname{diag}(d_1, \dots, d_n)$ sobre $ K^n $ com
\emph{qualquer} do \omterm{def:b2:nvs:norm}{normas} $\norm\cdot_1, \norm\cdot_2,
\norm\cdot_\infty $: de $\abs{d_ix_i} \leq
\bigl(\max_j\abs{d_j}\bigr)\abs{x_i}$ coordenada,
$\norm{Dx} \leq \max_j\abs{d_j}\,\norm x $; e $ x = e_{j_0}$
(um índice de maximização) atinge-o. Então $\vertiii D =
\max_j\abs{d_j}$ em todos os três casos: para mapas diagonais, todos
razoável \omterm{def:b2:nvs:norm}{normas} conte a mesma história, o maior trecho
fator. Tudo que é difícil \omterm{def:b2:nvs:norm}{normas} do operador é sobre
\emph{não}-comportamento diagonal --- é por isso que o adaptado
\omterm{def:b2:nvs:norm}{normas} do problema de fim de semana deste capítulo (questão 22) trabalhado por
forçando uma matriz a se tornar diagonal primeiro.
\end{example}

\begin{example}[Somas das colunas: o $1$-norma gêmea de \cref{exo:b2:nvs:4}]\label{ex:b2:nvs:columnsums}
Sobre $(\R^n, \norm\cdot_1)$, o \omterm{thm:b2:nvs:continuouslinear}{norma do operador} de uma matriz $ A $ é
o maior absoluto \emph{coluna} soma. Execute o método: para
$\norm x_1 \leq 1$,
\[
\norm{Ax}_1 = \sum_i\Bigl|\sum_j a_{ij}x_j\Bigr|
\leq \sum_j \abs{x_j}\sum_i\abs{a_{ij}}
\leq \Bigl(\max_j\sum_i\abs{a_{ij}}\Bigr)\norm x_1 ,
\]
e o limite é alcançado em $ x = e_{j_0}$ para uma maximização
coluna $ j_0$ --- a testemunha mais organizada que se possa imaginar. Então, para $ A =
\begin{psmallmatrix}1 & -2\\ 3 & 1\end{psmallmatrix}$:
$\vertiii A_1 = \max(1 + 3,\ 2 + 1) = 4$, enquanto
$\vertiii A_\infty = 4$ também (linhas) --- uma coincidência aqui, não
uma lei: transpor as entradas da matriz assimetricamente e os dois
\omterm{def:b2:nvs:norm}{normas} parte da empresa. Linhas para $\norm\cdot_\infty $, colunas para
$\norm\cdot_1$: o mnemônico é que cada normavetores unitários de
(padrões de sinal, resp.\ vetores de base) escolha a correspondência
somas.
\end{example}

\begin{proposition}[Mapas bilineares]\label{prop:b2:nvs:bilinear}
Um mapa bilinear $ b \colon E \times F \to G $ é \omterm{def:b2:metric:continuity}{contínuo} se
$\norm{b(x,y)} \leq C\norm x\,\norm y $ para alguns $ C $; então é
\omterm{def:b2:metric:continuity}{Lipschitz} em conjuntos limitados. (Mesmo padrão de prova; o produto
$(u, v) \mapsto v \circ u $ e multiplicação de matrizes são a chave
exemplos.)
\end{proposition}

\begin{proof}
Se o limite for válido:
\[
b(x,y) - b(x_0,y_0) = b(x - x_0,\, y) + b(x_0,\, y - y_0),
\]
então $\norm{b(x,y) - b(x_0,y_0)} \leq C\norm{x - x_0}\norm y +
C\norm{x_0}\norm{y - y_0}$: continuidade no $(x_0, y_0)$, e um
\omterm{def:b2:metric:continuity}{Lipschitz} vinculado onde $\norm x, \norm y \leq R $. Por outro lado,
continuidade no $(0,0)$ dá $\delta $ com $\norm{b(x,y)} \leq 1$ sobre
$\norm x, \norm y \leq \delta $; escalar ambas as variáveis.
\end{proof}

\section{Dimensão finita}

\begin{theorem}[Equivalência de normas em dimensão finita]\label{thm:b2:nvs:finitedim}
Em um espaço de dimensão finita, \emph{todos \omterm{def:b2:nvs:norm}{normas} são
equivalente}.\index{equivalência de normas} Consequentemente, em finito
dimensão: convergência, abertura, compacidade, completude são
noções independentes de \omterm{def:b2:nvs:norm}{normas}; \omterm{def:b2:metric:compact}{compactar} $=$ fechado e delimitado;
o espaço é \omterm{def:b2:metric:complete}{completo}; e todo mapa linear (ou multilinear)
\emph{de} um espaço de dimensão finita é \omterm{def:b2:metric:continuity}{contínuo}.
\end{theorem}

\begin{proof}
Fixe uma base e identifique $ E \simeq K^n $; basta comparar qualquer
\omterm{def:b2:nvs:norm}{norma} $ N $ com $\norm\cdot_\infty $.

\emph{Uma direção é a álgebra:} $ N(x) = N(\sum x_i e_i) \leq
\sum \abs{x_i} N(e_i) \leq C \norm x_\infty $ com $ C = \sum N(e_i)$.
Isto também mostra $ N $ é \omterm{def:b2:metric:continuity}{contínuo} sobre $(K^n, \norm\cdot_\infty)$
(isso é $ C $-Lipschitz: $\abs{N(x) - N(y)} \leq N(x - y)$).

\emph{A outra é a topologia:} a esfera unitária $ S = \{x :
\norm{x}_\infty = 1\}$ é fechado e limitado em $(K^n,
\norm\cdot_\infty)$, por isso \omterm{def:b2:metric:compact}{compactar}
(\cref{thm:b2:metric:compactprops}~(2), válido para $\C^n \simeq
\R^{2n}$). O \omterm{def:b2:metric:continuity}{contínuo} função $ N $ atinge o seu mínimo $ c $ sobre
$ S $; $ c > 0$ desde $ N $ desaparece apenas em $0 \notin S $. Homogeneidade
espalha o limite: $ N(x) \geq c \norm{x}_\infty $ para todos $ x $.

Consequências: todas as declarações se reduzem a $(K^n,
\norm\cdot_\infty)$, onde são conhecidos
(\cref{thm:b2:metric:rncomplete},
\cref{thm:b2:metric:compactprops}); um linear $ u $ de
dimensão finita $ E $ satisfaz $\norm{u(x)} \leq \sum\abs{x_i}
\norm{u(e_i)} \leq C'\norm{x}_\infty $: limite (4) de
\cref{thm:b2:nvs:continuouslinear}.
\end{proof}

\begin{corollary}\label{cor:b2:nvs:closedsubspace}
Um subespaço de dimensão finita de qualquer espaço normado é fechado.
\end{corollary}

\begin{proof}
Isso é \omterm{def:b2:metric:complete}{completo} para o induzido \omterm{def:b2:nvs:norm}{norma}
(\cref{thm:b2:nvs:finitedim}), e \omterm{def:b2:metric:complete}{completo} subconjuntos são fechados
(\cref{def:b2:metric:complete}).
\end{proof}

\begin{example}[Uma melhor aproximação calculada por simetria]\label{ex:b2:nvs:bestaffine}
Em $\bigl(C(\intcc{-1}{1}), \norm\cdot_\infty\bigr)$, quão longe está
$ f(x) = \abs x $ do subespaço (fechado, bidimensional) de
funções afins $ a + bx $? Por simetria, substituindo $ a + bx $ por
$ a - bx $ folhas $\norm{f - (a \pm bx)}_\infty $ inalterado, e
o ponto médio $ a $ faz pelo menos tão bem (desigualdade triangular em
a média): basta considerar constantes. Para uma constante
$ a $:
\[
\norm{\abs x - a}_\infty = \max\,(1 - a,\ a)
\geq \frac12 ,
\]
minimizado em $ a = \frac12$: a distância é $\frac12$, alcançado
pela constante $\frac12$. Observe a curva de erro $\abs x -
\frac12$: atinge $\pm\frac12$ alternadamente em $ x = -1, 0,
1$ --- três extremos de sinal alternado para melhor
aproximação de uma família de dois parâmetros. Que
\emph{equioscilação} o padrão não é acidental; é o
assinatura de otimização que o problema de fim de semana deste capítulo
se transforma no teorema de Chebyshev.
\end{example}

\begin{example}[Subespaços fechados versus densos]\label{ex:b2:nvs:closeddense}
Em $ E = \bigl(C(\intcc{0}{1}), \norm\cdot_\infty\bigr)$: cada
$\R_n[X]$ (polinômios de grau $\leq n $, restrito a
$\intcc01$) é de dimensão finita, portanto \emph{fechado}, subespaço
--- um limite uniforme de polinômios de grau $\leq n $ é um. Mas
o sindicato $\R[X]$ de todos eles é \emph{denso} em $ E $ (o
Teorema da aproximação de Weierstrass, provado em
\cref{ch:b2:funcseq}), e subespaços próprios densos são tão
não fechado como pode ser. Moral da história: o fechamento dos subespaços é uma
privilégio de dimensão finita; empilhar pisos fechados pode construir um
arranha-céu denso.
\end{example}

\begin{theorem}[Riesz]\label{thm:b2:nvs:riesz}
A bola unitária fechada de um espaço normado $ E $ é \omterm{def:b2:metric:compact}{compactar} \emph{se e
somente se} $\dim E < \infty $.\index{Teorema de Riesz}
\end{theorem}

\begin{proof}
Dimensão finita: fechado e limitado é suficiente
(\cref{thm:b2:nvs:finitedim}).

Por outro lado, suponha $\dim E = \infty $. \emph{Lema de Riesz:} para
todo subespaço fechado adequado $ F \subsetneq E $ e $\varepsilon \in
\intoo{0}{1}$, existe um vetor unitário $ x $ com $ d(x, F) \geq 1 -
\varepsilon $. Prova: escolha $ y \notin F $, deixar $\delta = d(y, F) > 0$
($ F $ fechado), escolha $ f \in F $ com $\norm{y - f} \leq
\frac{\delta}{1 - \varepsilon}$ e definir $ x = \frac{y - f}{\norm{y -
f}}$: para qualquer $ g \in F $,
\[
\norm{x - g} = \frac{\norm{y - (f + \norm{y-f}\,g)}}{\norm{y - f}}
\geq \frac{\delta}{\norm{y-f}} \geq 1 - \varepsilon ,
\]
o numerador está a uma distância de $ y $ até um ponto de $ F $.

Agora construa vetores unitários $ x_1, x_2, \dots $ indutivamente: $ F_k =
\operatorname{Vect}(x_1, \dots, x_k)$ é de dimensão finita, portanto
fechado (\cref{cor:b2:nvs:closedsubspace}) e próprio; Lema de Riesz
com $\varepsilon = \frac12$ fornece uma unidade $ x_{k+1}$ com $ d(x_{k+1},
F_k) \geq \frac12$. A sequência satisfaz $\norm{x_p - x_q} \geq
\frac12$ para $ p \neq q $: nenhuma subsequência convergente --- a bola unitária
não é \omterm{def:b2:metric:compact}{compactar}.
\end{proof}

\begin{example}[Riesz como detector de dimensão]\label{ex:b2:nvs:rieszdetector}
É $ C(\intcc01)$ de dimensão finita? Riesz responde sem
exibindo qualquer família livre infinita explícita: a sequência
$ f_n(x) = x^n $ está na bola unitária fechada e satisfaz, por
$ m > n $, $\norm{f_n - f_m}_\infty \geq f_n(x_0) - f_m(x_0) > 0$
em pontos adequados --- quantificados de forma clara no capítulo deste capítulo
problema de fim de semana (questão 16), onde uma subsequência permanece em
distância mútua $\geq \frac14$. Nenhuma subsequência convergente, então
a bola não é \omterm{def:b2:metric:compact}{compactar}, então $\dim C(\intcc01) = \infty $ por
\cref{thm:b2:nvs:riesz}. Compacidade da bola unitária é um
dicotomia perfeita: mantém-se em dimensão finita, falha em
dimensão infinita, sem meio termo --- apenas geometria
lê o tipo de dimensão.
\end{example}

\section{Espaços Banach}

\begin{definition}\label{def:b2:nvs:banach}
UM \emph{Espaço Banach}\index{Espaço Banach} é um \omterm{def:b2:metric:complete}{completo} normatizado
espaço. Exemplos: todo espaço normado de dimensão finita
(\cref{thm:b2:nvs:finitedim}); $\bigl(C(\intcc{a}{b}),
\norm\cdot_\infty\bigr)$ (\cref{thm:b2:metric:rncomplete});
$\mathcal{L}_c(E, F)$ para $ F $ Banach (mesmo padrão de prova do
\omterm{def:b2:metric:continuity}{contínuo} funções). Não exemplo: $\bigl(C(\intcc{0}{1}),
\norm\cdot_1\bigr)$ (\cref{exo:b2:nvs:7}).
\end{definition}

\begin{example}[A norma do operador de integração]\label{ex:b2:nvs:integrationnorm}
Sobre $\bigl(C(\intcc01), \norm\cdot_\infty\bigr)$, deixar $ T(f)(x) =
\int_0^x f(t)\,\dd t $ (um endomorfismo: $ T(f)$ é \omterm{def:b2:metric:continuity}{contínuo}).
Correr \cref{met:b2:nvs:opnorm}. Limite superior:
\[
\abs{T(f)(x)} \leq \int_0^x\abs f \leq x\,\norm f_\infty \leq
\norm f_\infty ,
\]
então $\vertiii T \leq 1$. Testemunha: $ f \equiv 1$ dá $ T(f)(x) =
x $ e $\norm{T(f)}_\infty = 1 = \norm f_\infty $: alcançado,
$\vertiii T = 1$. Mas observe $\vertiii{T^2} = \frac12 \neq
\vertiii T^2$: de fato $ T^2(f)(x) = \int_0^x(x - t)f(t)\dd t $ tem
$\abs{T^2(f)(x)} \leq \frac{x^2}2\norm f_\infty $, alcançado novamente
em $ f \equiv 1$; e geralmente $\vertiii{T^n} = \frac1{n!}$ ---
o limite submultiplicativo $\vertiii T^n = 1$ está desligado por um
fatorial. Este é exatamente o fenômeno do truque iterativo de
\cref{ch:b2:metric}O problema do fim de semana se converte em problema global
solubilidade de equações diferenciais lineares.
\end{example}

\begin{theorem}[Convergência absoluta em espaços de Banach]\label{thm:b2:nvs:absoluteconvergence}
Em um \omterm{def:b2:nvs:banach}{Espaço Banach}, se $\sum \norm{u_n} < \infty $ então $\sum u_n $
converge, e $\norm{\sum u_n} \leq \sum\norm{u_n}$. (O \omterm{def:b2:metric:complete}{completo}
teoria das séries em espaços normados é \cref{ch:b2:series}.)
\end{theorem}

\begin{proof}
Somas parciais $ S_N $: para $ q > p $, $\norm{S_q - S_p} \leq
\sum_{n=p+1}^{q}\norm{u_n}$, o que tende a $0$ (critério de Cauchy
para a série real de \omterm{def:b2:nvs:norm}{normas}): $(S_N)$ é Cauchy, portanto convergente.
A desigualdade passa ao limite do triângulo finito
desigualdade.
\end{proof}

\begin{example}[Matriz exponencial, primeiro contato]\label{ex:b2:nvs:matrixexp}
O \emph{matriz exponencial}\index{matriz exponencial}:
$\mathcal{M}_n(K)$ com qualquer submultiplicativo \omterm{def:b2:nvs:norm}{norma} ($\vertiii{AB}
\leq \vertiii A \vertiii B $) é Banach (dimensão finita). Então, por
cada $ A $,
\[
\eu^A = \sum_{k=0}^{\infty} \frac{A^k}{k!}
\]
converge absolutamente ($\vertiii{A^k/k!} \leq \vertiii A^k /k!$,
resumido): bem definido. \cref{ch:b2:diffeq} explora isso
sistematicamente.
\end{example}

\begin{example}[Uma série de Neumann que termina]\label{ex:b2:nvs:neumannfinite}
Para $ A = \begin{psmallmatrix}0 & \frac12\\ 0 &
0\end{psmallmatrix}$: $\vertiii A < 1$ em qualquer \omterm{thm:b2:nvs:continuouslinear}{norma do operador}
construído em \cref{ex:b2:nvs:examples}de \omterm{def:b2:nvs:norm}{normas}, e $ A^2 = 0$, então
a série geométrica entra em colapso:
\[
(I - A)^{-1} = \sum_{k \geq 0} A^k = I + A =
\begin{pmatrix}1 & \tfrac12\\ 0 & 1\end{pmatrix},
\]
verificado por $(I - A)(I + A) = I - A^2 = I $. Nilpotência
trunca a série exatamente como truncou a exponencial em
\cref{ch:b2:reduction}; e o exemplo calibra
expectativas: o inverso de Neumann é uma série infinita em
geral, um polinômio precisamente quando a perturbação é
nilpotente, e o erro após $ N $ termos é sempre limitado por
a cauda geométrica $\vertiii A^{N+1}/(1 - \vertiii A)$.
\end{example}

\begin{example}[O exponencial de um gerador de rotação]\label{ex:b2:nvs:rotationexp}
Pegar $ A = \begin{psmallmatrix}0 & -\theta\\ \theta &
0\end{psmallmatrix}$. Então $ A^2 = -\theta^2 I $, então os poderes
\omterm{def:b2:structures:sn}{ciclo} com o período quatro, e a série se divide em par e ímpar
peças:
\[
\eu^{A} = \sum_{k}\frac{A^k}{k!}
= \Bigl(\sum_{j}\frac{(-1)^j\theta^{2j}}{(2j)!}\Bigr) I
+ \Bigl(\sum_{j}\frac{(-1)^j\theta^{2j+1}}{(2j+1)!}\Bigr)
\frac{A}{\theta}
= \begin{pmatrix}
\cos\theta & -\sin\theta\\
\sin\theta & \cos\theta
\end{pmatrix},
\]
todos os rearranjos sendo licenciados por convergência absoluta. O
exponencial de um gerador antissimétrico é uma rotação ---
computado aqui puramente a partir da série, três capítulos antes do
equação diferencial $ x' = Ax $ (\cref{ch:b2:diffeq}) explica
\emph{por que}: $\eu^{tA}$ é o movimento circular uniforme. O fechamento
insight: as identidades entre as séries de matrizes são provadas exatamente como
escalares, uma vez submultiplicativos \omterm{def:b2:nvs:norm}{norma} certifica absoluto
convergência.
\end{example}

\begin{example}[Sup norma significa uniforme: o dicionário]\label{ex:b2:nvs:uniformdictionary}
A declaração $\norm{f_n - f}_\infty \to 0$ \emph{é} uniforme
convergência: um número, $\sup_x\abs{f_n(x) - f(x)}$, limites
o erro em todos os pontos simultaneamente. O dicionário em
ação em $ f_n(x) = x^n $ sobre $\intcc{0}{1}$: pontualmente, $ f_n
\to 0$ sobre $\intco{0}{1}$ e $ f_n(1) = 1$; em \omterm{def:b2:nvs:norm}{norma},
$\norm{f_n - 0}_\infty = 1 \not\to 0$, e de fato o ponto
limite é descontínuo e, portanto, fora do alcance de um
$\norm\cdot_\infty $-limite em $ C(\intcc01)$ (que está fechado
sob limites uniformes, \cref{thm:b2:metric:rncomplete}). Sobre
$\intcc{0}{a}$, $ a < 1$: $\norm{f_n}_\infty = a^n \to 0$ ---
convergência uniforme restaurada pela redução do domínio. Cada
declaração de convergência de \cref{ch:b2:funcseq} é uma declaração
sobre este \omterm{def:b2:nvs:norm}{norma}; mantendo o dicionário em mente pelas metades
esse capítulo.
\end{example}

\begin{remark}[Armadilhas comuns]\label{rem:b2:nvs:pitfalls}
(eu) Um \omterm{thm:b2:nvs:continuouslinear}{norma do operador} depende de \emph{ambos} escolhido \omterm{def:b2:nvs:norm}{normas}: o
mesma matriz tem $\vertiii\cdot_\infty $ dado por somas de linhas
(\cref{exo:b2:nvs:4}) e um diferente $\vertiii\cdot_1$ (coluna
somas); citando ``o'' \omterm{def:b2:nvs:norm}{norma} de uma matriz sem nomear o
subjacente \omterm{def:b2:nvs:norm}{normas} não tem sentido. (ii) $\vertiii{AB} \leq
\vertiii A\,\vertiii B $ é uma desigualdade, geralmente estrita ---
poderes podem encolher muito mais rápido do que o limite $\vertiii A^k $
sugere, que é o ponto principal da adaptação \omterm{def:b2:nvs:norm}{normas} (este
problema de fim de semana do capítulo, questão 22). (iii) ``Linear implica
\omterm{def:b2:metric:continuity}{contínuo}'' é um privilégio de dimensão finita: diferenciação
em polinômios é linear e ilimitado
(\cref{ex:b2:nvs:operatornorms}). (iv) Convergência absoluta de
$\sum u_n $ só ajuda quando o espaço é \omterm{def:b2:metric:complete}{completo}
(\cref{exo:b2:series:9} constrói o contraexemplo). (v) Em
dimensão infinita, um supremo sobre a bola unitária é um genuíno
supremo: não presuma que foi alcançado
(\cref{exo:b2:nvs:8}).
\end{remark}

\begin{remark}[Perspectivas dentro deste volume]\label{rem:b2:nvs:perspectives}
Três compromissos já estão marcados. Com \cref{ch:b2:series}: em
um \omterm{def:b2:nvs:banach}{Espaço Banach}, séries absolutamente convergentes convergem, então o
séries geométricas e exponenciais de operadores tornam-se cotidianas
ferramentas --- invertendo $ I - A $, definindo $\eu^{A}$
(\cref{ex:b2:series:neumann}). Com
\cref{ch:b2:funcseq} e \cref{ch:b2:powerseries}: convergência
de sequências de funções e séries de potências é a convergência em
$\bigl(C, \norm\cdot_\infty\bigr)$
(\cref{ex:b2:nvs:uniformdictionary}) e o raio de
convergência é uma afirmação sobre quais séries geométricas
dominar. Com \cref{ch:b2:fourier}: o \omterm{def:b2:nvs:norm}{normas}
$\norm\cdot_2$ e $\norm\cdot_\infty $ discordo genuinamente em
$ C(\intcc{0}{1})$ (\cref{ex:b2:nvs:sqrtnxn}), o que é exatamente
por que a convergência quadrática média das séries de Fourier e uniforme
convergência são dois teoremas diferentes com dois teoremas diferentes
preços.
\end{remark}

\begin{remark}[Onde este capítulo é usado]
\omterm{def:b2:nvs:norm}{Normas} do operador e a série geométrica impulsiona a perturbação
argumentos de \cref{ch:b2:diffcalc} (teorema da função inversa) e
o \omterm{ex:b2:nvs:matrixexp}{matriz exponencial} de \cref{ch:b2:diffeq}; \omterm{thm:b2:nvs:finitedim}{equivalência de
normas} autoriza silenciosamente cada ``escolha seu favorito \omterm{def:b2:nvs:norm}{norma}''
argumento em \cref{ch:b2:funcseq} e além; e o
divisão finita/infinita do teorema de Riesz --- tornada quantitativa em
o problema do fim de semana deste capítulo --- é a razão pela qual o Ano 3
volume precisa de novas ferramentas (convergência fraca, Arzelà - Ascoli,
Projeções espaciais de Hilbert) onde este volume ainda poderia extrair
subsequências convergentes.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b2:nvs:1}
Sobre $\R^2$, desenhe as bolas unitárias de $\norm\cdot_1$, $\norm\cdot_2$,
$\norm\cdot_\infty $, e provar as desigualdades $\norm x_\infty
\leq \norm x_2 \leq \norm x_1 \leq 2\norm x_\infty $ com o melhor
constantes em dimensão $2$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:nvs:2}
É $ N(f) = \abs{f(0)} + \norm{f'}_\infty $ um \omterm{def:b2:nvs:norm}{norma} sobre
$ C^1(\intcc{0}{1})$? Compare com $\norm{f}_\infty $: um
a desigualdade se mantém, a outra falha (quadro).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:nvs:3}
Calcule o \omterm{thm:b2:nvs:continuouslinear}{norma do operador} de $ u(f) = \int_0^1 f(t)\,\eu^t\,\dd t $
sobre $\bigl(C(\intcc{0}{1}), \norm\cdot_\infty\bigr) \to \R $, e de
a mudança $ S(x_1, x_2, \dots, x_n) = (x_2, \dots, x_n, 0)$ sobre
$(K^n, \norm\cdot_\infty)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:nvs:4}
Sobre $(\R^n, \norm\cdot_\infty)$, prove que \omterm{thm:b2:nvs:continuouslinear}{norma do operador} de um
matriz $ A $ é $\vertiii A_\infty = \max_i \sum_j \abs{a_{ij}}$ (o
maior soma absoluta de linhas). Calcule para $\begin{pmatrix} 1 & -2\\
3 & 1\end{pmatrix}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:nvs:5}
Prove isso $ GL_n(K)$ é \omterm{def:b2:metric:topology}{abrir} em $\mathcal{M}_n(K)$ e isso $ A
\mapsto A^{-1}$ é \omterm{def:b2:metric:continuity}{contínuo} nisso. \emph{Dica: para abertura, se
$\vertiii H < \frac{1}{\vertiii{A^{-1}}}$ então $ A + H = A(I +
A^{-1}H)$ com $\vertiii{A^{-1}H} < 1$, e $ I + B $ é invertível
para $\vertiii B < 1$ pela série geométrica
(\cref{thm:b2:nvs:absoluteconvergence}); para continuidade, vinculado
$(A+H)^{-1} - A^{-1}$ usando a mesma série.}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:nvs:6}
Deixar $\varphi $ ser uma forma linear em um espaço normado $ E $. Prove isso
$\varphi $ é \omterm{def:b2:metric:continuity}{contínuo} se e somente se $\ker\varphi $ está fechado.
\emph{(Se $\ker\varphi $ está fechado e $\varphi \neq 0$, escolha $ a $
com $\varphi(a) = 1$ e $ r > 0$ com $ B(a, r) \cap \ker\varphi =
\emptyset $; deduzir $\abs{\varphi(h)} \leq \frac{1}{r}\norm h $ por um
escalando argumento em $ a - \frac{h}{\varphi(h)}$.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:nvs:7}
Prove isso $\bigl(C(\intcc{0}{1}), \norm\cdot_1\bigr)$ não é
\omterm{def:b2:metric:complete}{completo}: mostre que as funções $ f_n $, rampas afins de $0$ para
$1$ sobre $\bigl[\frac12 - \frac1n, \frac12\bigr]$ (valor $0$
antes, $1$ depois), formam uma sequência de Cauchy sem \omterm{def:b2:metric:continuity}{contínuo}
$\norm\cdot_1$-limite.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:nvs:8}
Sobre $ E = C(\intcc{0}{1})$ com $\norm\cdot_\infty $, considere
\[
\varphi(f) = \sum_{n \geq 1} (-1)^n\, 2^{-n} f\bigl(\tfrac1n\bigr).
\]
Prove isso $\varphi $ é um bem definido \omterm{def:b2:metric:continuity}{contínuo} forma linear com
$\vertiii\varphi = 1$, mas que a definição suprema
$\vertiii\varphi $ é \emph{não alcançado} na bola unitária fechada.
\emph{(Limite superior: desigualdade triangular. \omterm{def:b2:nvs:norm}{Norma} $= 1$: construir
\omterm{def:b2:metric:continuity}{contínuo} $ f_K $ com $\norm{f_K}_\infty \leq 1$ e $ f_K(\frac1n)
= (-1)^n $ para $ n \leq K $ --- os pontos $\frac1n $ estão isolados de
um ao outro. Não realização: a igualdade forçaria $ f(\frac1n) =
(-1)^n $ para cada $ n $, incompatível com continuidade de $ f $ no $0$
desde $\frac1n \to 0$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:nvs:9}
Deixar $ E $ ser um espaço normado no qual a bola unitária fechada é \omterm{def:b2:metric:compact}{compactar}.
Deriva novamente, sem citar \cref{thm:b2:nvs:riesz}, que cada
sequência limitada tem uma subsequência convergente e prove que todo
forma linear em $ E $ é \omterm{def:b2:metric:continuity}{contínuo} se e somente se $\dim E < \infty $.
\emph{(Para dimensão infinita, construa uma forma descontínua
definindo-o livremente em uma sequência normalizada linearmente independente
e estendendo --- admitindo a existência de um algébrico
complemento.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:nvs:10}
Sobre $ C(\intcc{0}{1})$, provar $\norm f_1 \leq \norm f_2 \leq \norm
f_\infty $ \emph{(Cauchy--Schwarz para o primeiro)}e mostre com
a família $ f_n(x) = x^n $ que nenhuma desigualdade pode ser revertida
até uma constante: os três \omterm{def:b2:nvs:norm}{normas} são pares não equivalente.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:nvs:11}
(Distância a um hiperplano) Deixe $\varphi $ ser diferente de zero \omterm{def:b2:metric:continuity}{contínuo}
forma linear em um espaço normado $ E $. Prove isso
\[
d\bigl(x, \ker\varphi\bigr) =
\frac{\abs{\varphi(x)}}{\vertiii\varphi}
\qquad (x \in E),
\]
e verifique \cref{exo:b2:nvs:8} que o ínfimo não precisa ser
alcançado por qualquer ponto do hiperplano.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:nvs:12}
Sobre $ E = \R[X]$ (todos os polinômios), seja $ N_1(P) =
\sup_{\intcc{0}{1}}\abs P $ e $ N_2(P) =
\sup_{\intcc{0}{2}}\abs P $. Mostre isso $ N_1 \leq N_2$ mas isso
$ N_1$ e $ N_2$ são \emph{não} equivalente; deduzir que o
identidade $(E, N_2) \to (E, N_1)$ é um \omterm{def:b2:metric:continuity}{contínuo} bijeção linear
cujo inverso é descontínuo. Mostre finalmente isso $(E, N_1)$ é
não \omterm{def:b2:metric:complete}{completo} \emph{(somas parciais de Taylor de $\eu^x $)}. Todos os três
fenômenos são impossíveis em dimensão finita --- diga por quê.
\end{exercise}

\section{Problema: Melhor Aproximação e Teorema de Chebyshev}

Quão bem uma função pode ser aproximada por polinômios de um determinado
grau, e qual polinômio funciona melhor? Do lado da existência,
a resposta pertence a este capítulo: compacidade em finito
dimensão faz com que existam melhores aproximações. Do lado explícito,
um caso não trivial pode ser resolvido completamente com as próprias mãos ---
entre todos \emph{Mônica} polinômios de grau $ n $, aquele de
menor comida \omterm{def:b2:nvs:norm}{norma} sobre $\intcc{-1}{1}$ é o (normalizado) Chebyshev
polinômio, de \omterm{def:b2:nvs:norm}{norma} $2^{1-n}$: \emph{Extrema de Chebyshev
teorema}. O problema prova ambos os lados e depois mede o quão mal
compacidade falha em dimensão infinita: a bola unitária de
$ C(\intcc{0}{1})$ contém infinitas constelações de pontos em
distância mútua $1$.

\begin{problem}[{Problema de fim de semana --- teorema extremo de Chebyshev
e a geometria da bola unitária}]
\label{pb:b2:nvs:1}
\omterm{def:b2:nvs:norm}{Normas} sem subscrito são sup \omterm{def:b2:nvs:norm}{normas} no segmento indicado.

\medskip
\noindent\textbf{Part I --- Best approximation in normed
spaces.}
\begin{enumerate}
  \item Deixar $ F $ ser um subespaço de dimensão finita de um normado
        espaço $ E $ e $ x \in E $. Prove que a distância $ d(x,
        F) = \inf_{f \in F}\norm{x - f}$ é \emph{alcançado}
        \emph{(reduzir a um subconjunto fechado e limitado de $ F $ e usar
        \cref{thm:b2:nvs:finitedim})}.
  \item UM \omterm{def:b2:nvs:norm}{norma} é \emph{estritamente convexo} quando $\norm u = \norm v
        = 1$ e $ u \neq v $ implicar $\bigl\Vert\frac{u +
        v}2\bigr\Vert < 1$. Mostre isso $\norm\cdot_2$ sobre $\R^n $
        é estritamente convexo \emph{(identidade do paralelogramo)}e
        isso $\norm\cdot_1$ e $\norm\cdot_\infty $ não são para
        $ n \geq 2$.
  \item Prove que para um estritamente convexo \omterm{def:b2:nvs:norm}{norma}, o melhor
        aproximação da questão 1 é \emph{exclusivo}.
  \item Em $(\R^2, \norm\cdot_\infty)$, calcule tudo de melhor
        aproximações de $ x = (0, 1)$ pela linha $ F =
        \operatorname{Vect}\bigl((1,0)\bigr)$: um intervalo de
        minimizadores.
  \item Em $\bigl(C(\intcc{a}{b}), \norm\cdot_\infty\bigr)$,
        Mostre que a melhor aproximação de $ f $ por
        \emph{constantes} é único, igual a $ c^* = \frac{\max f
        + \min f}{2}$, com distância $\frac{\max f - \min
        f}{2}$; calcular ambos para $ f(x) = x^2$ sobre
        $\intcc{0}{1}$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Chebyshev polynomials.}
\begin{enumerate}[resume]
  \item Mostre que existe exatamente um polinômio $ T_n $ com
        $ T_n(\cos\theta) = \cos n\theta $ para todos $\theta $
        \emph{(recorrência $ T_{n+1} = 2XT_n - T_{n-1}$ do
        fórmula de adição de cosseno)}, que $\deg T_n = n $, e isso
        seu coeficiente líder é $2^{n-1}$ para $ n \geq 1$.
  \item Mostrar $\abs{T_n} \leq 1$ sobre $\intcc{-1}{1}$, com
        $ T_n(\eta_k) = (-1)^k $ no $ n + 1$ pontos $\eta_k =
        \cos\frac{k\pi}{n}$ ($ k = 0, \dots, n $) e que o
        raízes de $ T_n $ são os $ n $ pontos
        $\cos\frac{(2k-1)\pi}{2n}$, entrelaçando o $\eta_k $.
  \item Calcular $ T_2, T_3, T_4$ e verifique a alternância de
        $ T_3$ no $\eta_0, \dots, \eta_3 = 1, \frac12, -\frac12,
        -1$ por avaliação direta.
  \item Para $\abs x \geq 1$, provar
        \[
        T_n(x) = \frac{\bigl(x + \sqrt{x^2 - 1}\bigr)^n +
        \bigl(x - \sqrt{x^2 - 1}\bigr)^n}{2},
        \]
        e deduzir $ T_n(x) \sim \frac12\bigl(x + \sqrt{x^2 -
        1}\bigr)^n \to \infty $ Geometricamente para fixo $ x > 1$.
  \item Prove a lei da composição $ T_m \circ T_n = T_{mn}$
        \emph{(verifique $\intcc{-1}{1}$ e invocar a rigidez
        de polinômios)}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Chebyshev's extremal theorem.}
Escrever $ Q_n = 2^{1-n}T_n $ (monic, pela questão 6).
\begin{enumerate}[resume]
  \item Deixar $ P $ ser mônico de grau $ n \geq 1$ com
        $\sup_{\intcc{-1}{1}}\abs P < 2^{1-n}$. Ao avaliar $ D
        = Q_n - P $ nos pontos $\eta_k $ e sinal de contagem
        mudanças, deriva uma contradição. Concluir:
        \[
        \sup_{\intcc{-1}{1}}\abs P \;\geq\; 2^{1-n}
        \qquad\text{for every monic } P \text{ of degree } n.
        \]
  \item (Caso de igualdade) Suponha $\sup_{\intcc{-1}{1}}\abs P =
        2^{1-n}$ com $ P $ mônica de grau $ n $, e deixe $ D = Q_n
        - P \neq 0$. Mostrar $(-1)^kD(\eta_k) \geq 0$ para todos $ k $;
        mostre que cada um dos $ n $ intervalos
        $\intcc{\eta_{k}}{\eta_{k-1}}$ contém um zero de $ D $,
        e que um zero compartilhado por dois intervalos consecutivos é
        um \emph{interior} apontar $\eta_k $ onde $ D' = 0$ também. Conclua isso $ D $ tem $ n $ zeros contados com
        multiplicidade, portanto $ D = 0$: o minimizador é exatamente
        $ Q_n $ --- \emph{Teorema extremo de Chebyshev}.
  \item Reafirme o teorema como uma distância: em $\intcc{-1}{1}$,
        \[
        d_\infty\bigl(X^n,\ \R_{n-1}[X]\bigr) = 2^{1-n},
        \]
        com melhor aproximação única $ X^n - Q_n $; e mostrar por
        a substituição afim $ x = \frac{1+t}2$ isso em
        $\intcc{0}{1}$ a distância se torna $2^{1-2n}$.
  \item (Nós de interpolação \omterm{def:b2:structures:ideal}{ideal}) Para $ n $ nós $ x_1, \dots,
        x_n \in \intcc{-1}{1}$, o polinômio do nó
        $\omega(x) = \prod_i(x - x_i)$ é mônico de grau $ n $.
        Deduza da questão 12 qual escolha de nós minimiza
        $\sup_{\intcc{-1}{1}}\abs\omega $, o dependente do nó
        fator do limite de erro de interpolação clássico, e
        dê o valor mínimo.
  \item Verifique o caso $ n = 2$ do teorema manualmente (encontre
        $\inf_c \sup_{\intcc{-1}{1}}\abs{x^2 - c}$ diretamente),
        e calcule numericamente a distância da questão 13 em
        $\intcc{0}{1}$ para $ n = 10$. O que seu tamanho diz
        sobre o gráfico de $ x^{10}$?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The unit ball of
$ C(\intcc{0}{1})$.}
\begin{enumerate}[resume]
  \item Deixar $ g_k(x) = x^{2^k}$. Mostrar $\norm{g_k}_\infty = 1$ e
        $\norm{g_k - g_j}_\infty \geq \frac14$ para $ j > k $
        \emph{(avaliar no ponto onde $ x^{2^k} =
        \frac12$)}: uma sequência limitada explícita sem
        subsequência convergente --- a bola unitária fechada não é
        \omterm{def:b2:metric:compact}{compactar}, com as próprias mãos.
  \item (Lema de Riesz, aprimorado) Deixe $ F $ seja um
        \emph{dimensão finita} subespaço adequado de um normado
        espaço $ E $. Usando a questão 1, produza um vetor unitário $ x $
        com $ d(x, F) = 1$ exatamente --- não apenas $\geq 1 -
        \varepsilon $ como em \cref{thm:b2:nvs:riesz}lema.
  \item Deduza: em todo espaço normado de dimensão infinita existe
        é uma sequência de vetores unitários com distâncias aos pares
        $\geq 1$, e derivar novamente o teorema de Riesz a partir dele.
  \item Em $ C(\intcc{0}{1})$, exiba tal constelação
        explicitamente: as funções da tenda $ h_n $ suportado em
        $\bigl[\frac1{n+1}, \frac1n\bigr]$ com valor de pico $1$.
        Verificar $\norm{h_n} = 1$, $\norm{h_n - h_m} = 1$ para $ n
        \neq m $, e observe que $ h_n \to 0$ pontualmente, mas não
        uniformemente.
  \item (O limite total falha) Mostre que a bola unitária fechada
        de $ C(\intcc{0}{1})$ não pode ser coberto por um número finito
        bolas de raio $\frac13$ \emph{(cada bola contém
        no máximo um $ h_n $)} --- contraste com o
        etapa de limitação total na prova de
        \cref{thm:b2:metric:borellebesgue}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- \omterm{def:b2:nvs:norm}{Normas} at work on matrices, and
synthesis.}
\begin{enumerate}[resume]
  \item Prove que cada \omterm{def:b2:reduction:eigen}{autovalor} $\lambda $ de $ A \in
        \mathcal{M}_n(\C)$ satisfaz $\abs\lambda \leq
        \vertiii A $ para cada \omterm{thm:b2:nvs:continuouslinear}{norma do operador}; aplicar
        \cref{exo:b2:nvs:4} para vincular o valores próprios de
        $\begin{psmallmatrix}1 & -2\\ 3 & 1\end{psmallmatrix}$
        e compare com seu módulo verdadeiro.
  \item (Adaptado \omterm{def:b2:nvs:norm}{normas}) Deixar $ A $ ser \omterm{def:b2:reduction:diag}{diagonalizável}, $ A =
        P\,\mathrm{diag}(\lambda_1, \dots, \lambda_n)\,P^{-1}$.
        Mostrar $ N_P(x) = \norm{P^{-1}x}_\infty $ é um \omterm{def:b2:nvs:norm}{norma} cujo
        \omterm{thm:b2:nvs:continuouslinear}{norma do operador} satisfaz $\vertiii A_{N_P} =
        \max_i\abs{\lambda_i}$.
  \item Deduzir: para \omterm{def:b2:reduction:diag}{diagonalizável} $ A $, $ A^k \to 0$ se e somente
        se tudo valores próprios satisfazer $\abs{\lambda_i} < 1$ ---
        \omterm{thm:b2:nvs:finitedim}{equivalência de normas} faz a conclusão
        independente de \omterm{def:b2:nvs:norm}{norma}. Verifique $ A =
        \frac14\begin{psmallmatrix}1 & 2\\ 2 &
        1\end{psmallmatrix}$.
  \item (As constantes de equivalência explodem) Ligado $\R_n[X]$, comparar
        $ N_c(P) = \max_k \abs{a_k}$ (coeficientes) e
        $\norm{P}_{\intcc{0}{1}}$: ambos são \omterm{def:b2:nvs:norm}{normas}, então eles são
        equivalente para cada fixo $ n $; mas mostre, usando o mônico
        minimizador da questão 13 em $\intcc{0}{1}$, que o
        melhor constante $ C_n $ em $ N_c \leq
        C_n\norm\cdot_{\intcc{0}{1}}$ satisfaz $ C_n \geq
        2^{2n-1}$. Conclua em uma frase por que “todos \omterm{def:b2:nvs:norm}{normas} são
        equivalente''morre em dimensão infinita.
  \item (Síntese) Uma frase cada: onde compacidade de
        bolas de dimensão finita trabalhadas (questões 1, 12); o que
        a convexidade estrita governa; qual é a constelação de
        as questões 18 a 19 destroem; e como a questão 24
        quantifica a falha. Nomeie o cume (Chebyshev's
        teorema extremo) e indique onde melhor aproximação
        encontra seu lar moderno (o teorema da projeção em Hilbert
        espaços, volume do Ano 3, onde completude substitui
        compacidade).
\end{enumerate}
\end{problem}
