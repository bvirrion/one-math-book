\chapter{Espaços Afins}\label{ch:b2:affine}

Os espaços vetoriais têm um ponto privilegiado --- a origem --- que
a geometria não quer. Um \emph{\omterm{def:b2:affine:def}{espaço afim}} é um espaço vetorial
que esqueceu sua origem: pontos e vetores tornam-se diferentes
espécies, relacionadas por tradução. Este breve capítulo constrói a
dicionário (pontos, \omterm{def:b2:affine:barycenter}{baricentros}, \omterm{def:b2:affine:subspace}{afim} subespaços e mapas), o
\omterm{def:b2:affine:subspace}{afim} visão da convexidade e as ferramentas de classificação usadas no
capítulos de geometria adiante.

\section{Pontos e vetores}

\begin{definition}\label{def:b2:affine:def}
Um \emph{espaço afim}\index{espaço afim} dirigido por um verdadeiro
espaço vetorial $E $ é um conjunto não vazio $\mathcal{E}$ com um mapa
$(A, B) \mapsto \vect{AB} \in E $ satisfatório
\[
\vect{AB} + \vect{BC} = \vect{AC}
\quad \text{(Chasles)},
\qquad
\text{for each } A,\ B \mapsto \vect{AB} \text{ is a bijection }
\mathcal{E} \to E .
\]
Um escreve $ B = A + u $ para o ponto único com $\vect{AB} = u $.
O \emph{dimensão} de $\mathcal{E}$ é $\dim E $. Cada vetor
o espaço é um \omterm{def:b2:affine:subspace}{afim} espaço sobre si mesmo ($\vect{AB} = B - A $); cada
escolha de origem $ O \in \mathcal{E}$ identifica $\mathcal{E}$ com
$ E $ através de $ M \mapsto \vect{OM}$.
\end{definition}

\begin{example}[Um espaço afim sem naturalidade
origem]\label{ex:b2:affine:planeexample}
O plano da solução $\mathcal E = \{(x, y, z) \in \R^3 : x + y
+ z = 1\}$ não é um subespaço vetorial ($0 \notin \mathcal E $),
mas é um \omterm{def:b2:affine:def}{espaço afim} dirigido por $ E = \{x + y + z =
0\}$: para $ A, B \in \mathcal E $ a diferença $\vect{AB} = B
- A $ pousa em $ E $ (as somas são canceladas), Chasles é herdado
de $\R^3$, e $ B \mapsto \vect{AB}$ é bijetivo em $ E $.
Nenhum ponto de $\mathcal E $ é distinto --- qualquer escolha de
``origem'' $ O \in \mathcal E $ funciona igualmente bem, e todos os
identificações $ M \mapsto \vect{OM}$ diferem pelas traduções.
Esta é a situação típica: conjuntos solução de problemas não homogêneos
problemas lineares (sistemas lineares, \omterm{def:b2:diffcalc:differential}{diferencial} linear
equações em \cref{ch:b2:diffeq}) são \omterm{def:b2:affine:subspace}{afim}, nunca linear,
e o slogan “solução particular mais kernel” é exatamente
a declaração $\mathcal F = A + F $ da próxima definição.
\end{example}

\begin{definition}[Baricentro]\label{def:b2:affine:barycenter}
Deixar $(A_i, \lambda_i)_{i \leq k}$ ser ponderado pontos com
$\sum\lambda_i \neq 0$. O \emph{baricentro}\index{baricentro} $ G
= \operatorname{bar}\bigl((A_i, \lambda_i)\bigr)$ é o único
apontar com
\[
\sum_i \lambda_i\, \vect{GA_i} = 0
\qquad\text{equivalently}\qquad
\vect{OG} = \frac{1}{\sum\lambda_i}\sum_i \lambda_i\,\vect{OA_i}
\quad (\text{any } O).
\]
Baricentros são \emph{associativo} (subgrupos de pontos podem ser
substituído pelo seu baricentro parcial com o peso somado) e
invariante sob reescalonamento de todos os pesos.
\end{definition}

\begin{proof}[Prova de existência e as fórmulas]
Correção $ O $ e escrever $ s = \sum_i\lambda_i \neq 0$. Por Chasles,
\[
\sum_i\lambda_i\,\vect{GA_i} = 0
\iff \sum_i\lambda_i\bigl(\vect{GO} + \vect{OA_i}\bigr) = 0
\iff s\,\vect{OG} = \sum_i\lambda_i\,\vect{OA_i},
\]
que determina $ G = O + \frac1s\sum_i\lambda_i\vect{OA_i}$
exclusivamente. \emph{Independência de $ O $:} para outra origem
$ O'$,
\[
\frac1s\sum_i\lambda_i\,\vect{O'A_i}
= \frac1s\sum_i\lambda_i\bigl(\vect{O'O} + \vect{OA_i}\bigr)
= \vect{O'O} + \frac1s\sum_i\lambda_i\,\vect{OA_i}
= \vect{O'G} :
\]
o mesmo ponto $ G $. \emph{Associatividade:} dividir o conjunto de índices
como $ I \sqcup J $ com $ s_I = \sum_{i\in I}\lambda_i \neq 0$,
e deixe $ G_I $ seja o \omterm{def:b2:affine:barycenter}{baricentro} de $(A_i, \lambda_i)_{i\in
I}$, para que $\sum_{i\in I}\lambda_i\vect{OA_i} =
s_I\,\vect{OG_I}$. Então
\[
s\,\vect{OG}
= \sum_{i\in I}\lambda_i\vect{OA_i} +
\sum_{j\in J}\lambda_j\vect{OA_j}
= s_I\,\vect{OG_I} + \sum_{j\in J}\lambda_j\vect{OA_j} :
\]
$ G $ é o \omterm{def:b2:affine:barycenter}{baricentro} de $(G_I, s_I)$ junto com $(A_j,
\lambda_j)_{j\in J}$, como alegado. \emph{Redimensionamento:} substituindo
cada $\lambda_i $ por $ t\lambda_i $ ($ t \neq 0$) multiplica $ s $
e a soma ponderada por $ t $, saindo $\vect{OG}$ inalterado.
\end{proof}

\begin{remark}[Armadilhas comuns]
Duas armadilhas cercam a definição. Primeiro, se a soma dos pesos
para \emph{zero}, não há \omterm{def:b2:affine:barycenter}{baricentro}: o mapa $ O \mapsto
\sum\lambda_i\vect{OA_i}$ é então independente de $ O $ e
define um \emph{vetor}, não é um ponto --- por exemplo
$(A, -1;\ B, 1)$ codifica $\vect{AB}$. Acompanhando quais
dos dois objetos que uma computação produz é metade
higiene baricêntrica. Em segundo lugar, os pesos só são significativos
para um fator comum diferente de zero; fórmulas como ``as coordenadas
de $ G $ são $\lambda_1, \dots, \lambda_k $'' pressupõe uma
normalização (geralmente $\sum\lambda_i = 1$) e esquecendo
normalizar é a fonte padrão de proporções erradas em um
figura.
\end{remark}

\begin{definition}[Subespaços afins; mapas afins]\label{def:b2:affine:subspace}
Um \emph{subespaço afim} é um conjunto $\mathcal{F} = A + F = \{A + u :
u \in F\}$ com $ F $ um subespaço vetorial (seu \emph{direção});
equivalentemente, um conjunto não vazio estável sob \omterm{def:b2:affine:barycenter}{baricentros}. Afim
subespaços de $\R^n $ são exatamente os conjuntos solução de sistemas lineares
$ MX = B $ (Ano 1: solução particular mais kernel). Um mapa $ f \colon
\mathcal{E} \to \mathcal{E}'$ é \emph{afim}\index{afim}
quando ele preservar \omterm{def:b2:affine:barycenter}{baricentros} --- equivalentemente quando
\[
f(A + u) = f(A) + \varphi(u)
\]
para um mapa linear (único) $\varphi = \vec f $, o \emph{linear
parte}. Mapas afins de $\R^n $: $ X \mapsto MX + C $. As composições são
afim com partes lineares compostas; $ f $ é bijetivo se e se $\vec f $
é.
\end{definition}

\begin{proof}[Prova de equivalência para mapas]
Se $ f(A + u) = f(A) + \varphi(u)$: para um \omterm{def:b2:affine:barycenter}{baricentro} $ G $ de $(A_i,
\lambda_i)$, expandindo cada ponto de $ A $, $ f(G) = f(A) +
\varphi(\vect{AG})$ e $\varphi(\vect{AG}) =
\frac{\sum\lambda_i\varphi(\vect{AA_i})}{\sum\lambda_i}$: $ f(G)$ é
o \omterm{def:b2:affine:barycenter}{baricentro} das imagens. Por outro lado, consertar $ A $ e definir
$\varphi(u) = \vect{f(A)\,f(A + u)}$. \emph{Homogeneidade:} $ A + tu =
\operatorname{bar}\bigl(A, 1-t;\ A + u, t\bigr)$ por cada real
$ t $, então a preservação de \omterm{def:b2:affine:barycenter}{baricentros} (com pesos reais arbitrários,
como hipotetizado) dá $\varphi(tu) = t\,\varphi(u)$ diretamente.
\emph{Aditividade:} $ A + u + v = \operatorname{bar}\bigl(A + 2u,
\tfrac12;\ A + 2v, \tfrac12\bigr)$, então $\varphi(u + v) =
\tfrac12\varphi(2u) + \tfrac12\varphi(2v) = \varphi(u) +
\varphi(v)$, usando homogeneidade. Por isso $\varphi $ é linear.
\end{proof}

\begin{remark}
A prova usada \omterm{def:b2:affine:barycenter}{baricentros} com \emph{real arbitrário}
pesos: a etapa de homogeneidade leva $ t $ fora $\intcc01$.
Se um mapa for assumido apenas para preservar \omterm{def:b2:affine:barycenter}{baricentros} com
pesos não negativos --- equivalentemente, pontos médios e segmentos
--- a linearidade do mapa vetorial não é mais gratuita:
só se consegue $\Q $-linearidade e um continuidade hipótese é
precisava concluir, exatamente como em
\cref{exo:b2:affine:5}. Distinguir ``preserva tudo
\omterm{def:b2:affine:barycenter}{baricentros}'' de `` preserva \omterm{def:b2:affine:convex}{convexo} combinações'' é um
pequena mas real sutileza do \omterm{def:b2:affine:subspace}{afim} vocabulário.
\end{remark}

\begin{example}[Geometria clássica do baricentro]\label{ex:b2:affine:median}
O centróide de um triângulo $ ABC $ é o \omterm{def:b2:affine:barycenter}{baricentro} $ G =
\operatorname{bar}(A,1; B,1; C,1)$. Associatividade com o
ponto médio $ A' = \operatorname{bar}(B, 1; C, 1)$ mostra
\[
G = \operatorname{bar}(A, 1;\ A', 2) :
\]
$ G $ encontra-se na mediana $ AA'$ em dois terços disso --- e da mesma forma
para as outras duas medianas: as três medianas são concorrentes, em
uma linha de cálculo baricêntrico.
\end{example}

\begin{example}[As bimedianas de um
quadrilátero]\label{ex:b2:affine:bimedians}
Deixar $ ABCD $ ser qualquer quadrilátero (plano ou não!) e considere
é \emph{bimedianos}: os segmentos que unem os pontos médios de
lados opostos, $ M_{AB}M_{CD}$ e $ M_{BC}M_{DA}$. Apresentar
o \omterm{def:b2:affine:barycenter}{baricentro} $ G $ de $(A,1; B,1; C,1; D,1)$ e agrupar os
pesa de duas maneiras:
\[
G = \operatorname{bar}\bigl(M_{AB}, 2;\ M_{CD}, 2\bigr)
= \operatorname{bar}\bigl(M_{BC}, 2;\ M_{DA}, 2\bigr) :
\]
$ G $ é o ponto médio de \emph{ambos} bimedianos --- então os dois
bimedianas sempre se dividem ao meio, e o quadrilátero de
os quatro pontos médios é um paralelogramo (suas diagonais são os
bimedianos). Nenhuma análise de caso, nenhuma coordenada e o
argumento sobrevive inalterado para um quadrilátero inclinado em
$\R^3$, onde uma prova baseada em imagem já seria
delicado: a associatividade não se preocupa com a dimensão.
\end{example}

\begin{example}[Classificando um mapa afim, comece a
terminar]\label{ex:b2:affine:dilation}
Deixar $ f(x, y) = (2x - 1,\ 3y - 4)$ sobre $\R^2$. Sua parte linear
é $\varphi = \operatorname{diag}(2, 3)$, cujo \omterm{def:b2:reduction:eigen}{espectro}
$\{2, 3\}$ evita $1$: pelo critério do ponto fixo comprovado
abaixo (\cref{prop:b2:affine:fixedpoint}), $ f $ tem exatamente um
ponto fixo, encontrado resolvendo
\[
x = 2x - 1, \qquad y = 3y - 4
\qquad\Longrightarrow\qquad \Omega = (1, 2).
\]
Recentralizando em $\Omega $ (definir $ x = 1 + u $, $ y = 2 + v $):
\[
f(1 + u,\ 2 + v) = (1 + 2u,\ 2 + 3v) :
\]
no quadro em $\Omega $, $ f $ \emph{é} sua parte linear, um
dilatação anisotrópica que se estende por $2$ horizontalmente e $3$
verticalmente do centro $(1, 2)$. A lição geral: uma
mapa \omterm{def:b2:affine:subspace}{afim} é ``mapa linear mais dados de localização'', e o
os dados de localização são recolhidos para uma origem bem escolhida sempre que
$1$ não é um \omterm{def:b2:reduction:eigen}{autovalor}. Por outro lado, traduzindo a origem
mal \emph{cria} os termos constantes: \omterm{def:b2:affine:subspace}{afim} geometria é
a arte de escolher onde colocar $0$.
\end{example}

\begin{remark}[Método: simultaneidade e alinhamento por
baricentros]
\cref{ex:b2:affine:median} é uma instância de um general
receita. Para provar que três cevianas de um triângulo são
concomitantemente, apresentam \emph{solteiro} sistema ponderado $(A,
\alpha; B, \beta; C, \gamma)$ e usar associatividade três
formas: agrupamento $(B, C)$ mostra o \omterm{def:b2:affine:barycenter}{baricentro} encontra-se no
ceviana de $ A $, agrupamento $(C, A)$ na ceviana de $ B $,
agrupamento $(A, B)$ no terceiro. Para as medianas, o sistema
$(A, 1; B, 1; C, 1)$ faz todo o trabalho; para corte de cevianas
os lados em proporções prescritas, os pesos são lidos
proporções. Para provar três pontos \emph{alinhado}, escreva um como um
\omterm{def:b2:affine:barycenter}{baricentro} dos outros dois (\cref{exo:b2:affine:2}) ou usar
o determinante critério de \cref{exo:b2:affine:11}. Ambos
receitas substituem a engenhosidade geométrica pela contabilidade de peso ---
é exatamente para isso que serve o cálculo bariccêntrico.
\end{remark}

\section{Convexidade, afim}

\begin{definition}\label{def:b2:affine:convex}
Um subconjunto $ C $ de um \omterm{def:b2:affine:def}{espaço afim} é \emph{convexo}\index{convexo} quando ele contém
cada \omterm{def:b2:affine:barycenter}{baricentro} com \emph{não negativo} pesos de seus pontos ---
equivalentemente, cada segmento $\intcc{A}{B} = \{\operatorname{bar}(A,
1-t; B, t) : t \in \intcc{0}{1}\}$ entre seus pontos. O
\emph{casco convexo}\index{casco convexo} $\operatorname{conv}(S)$ é o conjunto de todos
peso não negativo \omterm{def:b2:affine:barycenter}{baricentros} de pontos de $ S $ --- o menor
conjunto convexo contendo $ S $.
\end{definition}

\begin{example}[As epígrafes são convexas
conjuntos]\label{ex:b2:affine:epigraph}
A região $ C = \{(x, y) : y \geq x^2\}$ acima da parábola
é \omterm{def:b2:affine:convex}{convexo}: para $(x_1, y_1), (x_2, y_2) \in C $ e $ t \in
\intcc01$, a desigualdade de convexidade da função quadrada
dá
\[
\bigl((1-t)x_1 + tx_2\bigr)^2 \leq (1-t)x_1^2 + tx_2^2
\leq (1-t)y_1 + ty_2 ,
\]
então o \omterm{def:b2:affine:barycenter}{baricentro} permanece acima da parábola. A computação
é geral: $\{y \geq f(x)\}$ é \omterm{def:b2:affine:convex}{convexo} exatamente quando $ f $ é um
função convexa --- \omterm{def:b2:affine:convex}{convexo} \emph{conjuntos} e \omterm{def:b2:affine:convex}{convexo}
\emph{funções} (\cref{ch:b2:realfun}) são duas faces de uma
noção, sendo as epígrafes o dicionário. Esta é a geometria
razão pela qual existem linhas de suporte para funções convexas, o fato
isso provará a desigualdade de Jensen em
\cref{ch:b2:randomvar}.
\end{example}

\begin{example}[Geradores redundantes de um convexo
casco]\label{ex:b2:affine:squarehull}
Deixar $ S = \{(0,0), (2,0), (2,2), (0,2), (1,1)\}$. O quinto
ponto é o \omterm{def:b2:affine:barycenter}{baricentro}
\[
(1,1) = \operatorname{bar}\bigl((0,0), \tfrac12;\ (2,2),
\tfrac12\bigr),
\]
então já está no casco dos outros quatro:
$\operatorname{conv}(S)$ é o quadrado com os quatro cantos
como vértices. Em geral, um ponto de $ S $ isso é um
peso não negativo \omterm{def:b2:affine:barycenter}{baricentro} do \emph{outro} pontos de
$ S $ pode ser excluído sem alterar o casco; os pontos que
nunca podem ser excluídos (aqui os quatro cantos) são os
\emph{pontos extremos} do casco. Determiná-los é uma pura
\omterm{def:b2:affine:barycenter}{baricentro} computação: $(2,0)$, digamos, não pode ser escrito como
$\operatorname{bar}$ dos pontos restantes com não negativo
pesos, porque a primeira coordenada forçaria todo o peso
em pontos com $ x = 2$, e a segunda coordenada então
falha. As questões de convexidade reduzem-se, repetidas vezes, a
resolver pequenos sistemas ponderados.
\end{example}

\begin{theorem}[Carathéodory]\label{thm:b2:affine:caratheodory}
Em um \omterm{def:b2:affine:def}{espaço afim} de dimensão $ n $, cada ponto de
$\operatorname{conv}(S)$ é um \omterm{def:b2:affine:barycenter}{baricentro} de no máximo $ n + 1$ pontos
de $ S $.\index{Teorema de Carathéodory}
\end{theorem}

\begin{proof}
Deixar $ G = \operatorname{bar}(A_0, \lambda_0; \dots; A_k, \lambda_k)$
com $\lambda_i > 0$, $\sum\lambda_i = 1$, e $ k + 1 > n + 1$
pontos. O $ k $ vetores $\vect{A_0A_i}$ ($ i \geq 1$) estão vinculados
($ k > n $): $\sum_{i\geq1}\mu_i \vect{A_0A_i} = 0$ não trivialmente;
contexto $\mu_0 = -\sum_{i\geq1}\mu_i $, obtemos pesos $(\mu_i)$
com $\sum\mu_i = 0$, $\sum \mu_i\,\vect{OA_i} = 0$ (qualquer $ O $), não
tudo zero. Então para cada real $ t $ os pesos $\lambda_i - t\mu_i $
ainda soma a $1$ e, desde $\sum_i\mu_i\vect{OA_i} = 0$,
\[
\sum_i(\lambda_i - t\mu_i)\,\vect{OA_i}
= \sum_i\lambda_i\,\vect{OA_i} :
\]
eles produzem o \emph{mesmo} apontar $ G $. Agora deslize $ t $ de
$0$: alguns $\mu_i $ é positivo (eles somam zero e não são
tudo zero), então
\[
t^* = \min\Bigl\{\frac{\lambda_i}{\mu_i} : \mu_i > 0\Bigr\}
\]
é bem definido e positivo. No $ t = t^*$: para índices com
$\mu_i > 0$, $\lambda_i - t^*\mu_i \geq 0$ pela minimalidade,
com igualdade em índice minimizador; para índices com $\mu_i
\leq 0$, $\lambda_i - t^*\mu_i \geq \lambda_i > 0$. Todos
os pesos permanecem não negativos e pelo menos um morreu: $ G $ é
reescrito como um \omterm{def:b2:affine:barycenter}{baricentro} de menos pontos. Iterar enquanto mais
do que $ n + 1$ restam pontos.
\end{proof}

\begin{example}
No avião ($ n = 2$): cada ponto do \omterm{def:b2:affine:convex}{casco convexo} de um finito
conjunto está em um triângulo com vértices no conjunto --- o geométrico
conteúdo de Carathéodory, usado em otimização e probabilidade
(misturas) iguais.
\end{example}

\begin{example}[Correndo Carathéodory's
algoritmo]\label{ex:b2:affine:caratheodoryrun}
Escreva o centro do quadrado de
\cref{ex:b2:affine:squarehull} com seus quatro cantos $ A_1 =
(0,0)$, $ A_2 = (2,0)$, $ A_3 = (2,2)$, $ A_4 = (0,2)$:
\[
(1,1) = \operatorname{bar}\bigl(A_1, \tfrac14;\ A_2,
\tfrac14;\ A_3, \tfrac14;\ A_4, \tfrac14\bigr),
\]
quatro pontos de dimensão $2$ --- um a mais. A prova
receita pede pesos $(\mu_i)$ com $\sum\mu_i = 0$ e
$\sum\mu_i\vect{OA_i} = 0$: aqui $\mu = (1, -1, 1, -1)$ funciona
(as duas diagonais compartilham seu ponto médio). Deslizando $\lambda_i
\mapsto \lambda_i - t\mu_i $ mantém o \omterm{def:b2:affine:barycenter}{baricentro} fixo para
cada $ t $; o valor extremo admissível $ t = \frac14$ faz
os pesos $(0, \tfrac12, 0, \tfrac12)$, matando $ A_1$ e
$ A_3$ simultaneamente:
\[
(1,1) = \operatorname{bar}\bigl(A_2, \tfrac12;\ A_4,
\tfrac12\bigr),
\]
uma representação por dois pontos --- ainda melhor que os três
que o teorema garante, porque o centro acontece
encontra-se em um segmento entre geradores. O algoritmo é
inteiramente mecânico: encontre uma dependência, deslize até um peso
morre, repita.
\end{example}

\section{Ferramentas de classificação afim}

\begin{proposition}[Pontos fixos de mapas afins]\label{prop:b2:affine:fixedpoint}
Deixar $ f $ feijão \omterm{def:b2:affine:subspace}{afim} endomorfismo de uma dimensão finita \omterm{def:b2:affine:subspace}{afim}
espaço com parte linear $\varphi $. Se $1 \notin
\operatorname{Sp}(\varphi)$, então $ f $ tem exatamente um ponto fixo
$\Omega $, e na vetorização em $\Omega $, $ f $ \emph{é}
sua parte linear. (Traduções, com $\varphi = \mathrm{id}$ e não
ponto fixo, são a obstrução básica.)
\end{proposition}

\begin{proof}
Consertar $ O $ e escrever $ f(O + x) = f(O) + \varphi(x)$. O ponto
$ O + x $ é fixo se $ O + x = f(O) + \varphi(x)$, ou seja
\[
(\mathrm{id} - \varphi)(x) = \vect{O f(O)} .
\]
Em dimensão finita, $\mathrm{id} - \varphi $ é invertível
se $0$ não é um \omterm{def:b2:reduction:eigen}{autovalor} de $\mathrm{id} - \varphi $, se
$1 \notin \operatorname{Sp}\varphi $ --- e nesse caso o
a equação exibida tem exatamente uma solução $ x^*$, dando o
ponto fixo único $\Omega = O + x^*$. Recentralização: para qualquer
vetor $ u $,
\[
f(\Omega + u) = f(\Omega) + \varphi(u) = \Omega + \varphi(u),
\]
então no quadro com origem $\Omega $ o mapa diz $ u \mapsto
\varphi(u)$: puramente linear. Quando $1 \in
\operatorname{Sp}\varphi $, ou não existe nenhum ponto fixo (o
a equação exibida pode ser insolúvel, como para uma tradução)
ou um todo \omterm{def:b2:affine:subspace}{afim} subespaço deles faz (adicione qualquer vetor próprio
de \omterm{def:b2:reduction:eigen}{autovalor} $1$ para uma solução): a exclusividade é exatamente o
condição espectral.
\end{proof}

\begin{example}[Isometrias planas, concluídas]\label{ex:b2:affine:isometries}
Um \omterm{def:b2:affine:subspace}{afim} isometria do plano euclidiano tem parte linear em
$ O(2)$: uma rotação $ R_\theta $ ou uma reflexão (volume do ano 1). Se
$\theta \neq 0$: $1 \notin \operatorname{Sp} R_\theta $, então o mapa
é um \emph{rotação em torno de um único centro}
(\cref{prop:b2:affine:fixedpoint}). Se a parte linear for um
reflexão: uma reflexão em um eixo (existem pontos fixos) ou
um \emph{reflexão deslizante} (reflexão composta com uma tradução
ao longo do eixo, sem ponto fixo). Com traduções, este é o
\omterm{def:b2:metric:complete}{completo} classificação de isometrias planas.
\end{example}

\begin{remark}[As isometrias planas, em resumo]
Coletando os casos: identidade; traduções ($\vec f =
\mathrm{id}$, nenhum ponto fixo, a menos que seja trivial); rotações
(parte linear $ R_\theta $, $\theta \neq 0$: um centro);
reflexões (parte linear uma reflexão, uma linha de fixo
pontos); reflexões de deslizamento (mesma parte linear, sem fixo
ponto). Quatro famílias mais a identidade, cada uma reconhecida por
apenas dois dados: a parte linear e o conjunto de ponto fixo ---
o padrão de \cref{prop:b2:affine:fixedpoint} feito
exaustivo.
\end{remark}

\begin{example}[Um reflexo deslizante, capturado no
agir]\label{ex:b2:affine:glide}
Deixar $ f(x, y) = (y + 1,\ x + 1)$. A parte linear $(x, y)
\mapsto (y, x)$ é o reflexo na diagonal $ y = x $, então
$1 \in \operatorname{Sp}\vec f $ e
\cref{prop:b2:affine:fixedpoint} está em silêncio. Pontos fixos
precisaria $ x = y + 1$ e $ y = x + 1$ simultaneamente:
impossível --- não existe nenhum, então $ f $ não é um reflexo. Quadratura
estabelece a classificação:
\[
f\bigl(f(x, y)\bigr) = f(y + 1,\ x + 1) = (x + 2,\ y + 2),
\]
a tradução por $(2, 2)$: $ f $ é o \emph{deslizar
reflexão} com eixo a linha $ y = x $ (deslocado
apropriadamente: o ponto médio de $ M $ e $ f(M)$ sempre mente
$ y = x + {}$ constante, aqui $ y = x $, como se verifica $ M = (0,
0) \mapsto (1,1)$) e vetor de deslizamento $(1, 1)$, metade de $ f
\circ f $. Compare com \cref{exo:b2:affine:6}, onde o mesmo
parte linear, mas uma constante diferente produziu um resultado honesto
reflexão: com \omterm{def:b2:reduction:eigen}{autovalor} $1$ presente, o termo constante
decide tudo.
\end{example}

\begin{example}[Recursões afins são afins
dinâmica]\label{ex:b2:affine:recursion}
A recursão clássica $ u_{n+1} = au_n + b $ ($ a \neq 1$)
itera o mapa \omterm{def:b2:affine:subspace}{afim} $ f(x) = ax + b $ da linha, cujo
parte linear $ a $ evita o \omterm{def:b2:reduction:eigen}{autovalor} $1$: existe um único
ponto fixo $\omega = \frac{b}{1-a}$, e recentrando lá
(o caso unidimensional da proposição acima) vira
$ f $ na multiplicação por $ a $:
\[
u_{n+1} - \omega = a\,(u_n - \omega)
\qquad\Longrightarrow\qquad
u_n = \omega + a^n(u_0 - \omega) .
\]
Para $ u_{n+1} = \frac{u_n}2 + 3$: $\omega = 6$ e $ u_n = 6 +
(u_0 - 6)2^{-n} \to 6$. A receita ensinada para tal
recursões em \cref{ch:b2:series} --- `` subtraia o fixo
ponto'' --- é exatamente a vetorização de um mapa \omterm{def:b2:affine:subspace}{afim}
no seu ponto fixo; convergência para $\abs a < 1$ é o
fenômeno de contração que \cref{ch:b2:metric} se transformou em
o teorema do ponto fixo de Banach. Uma ideia, três capítulos.
\end{example}

\begin{example}[Encontrando o centro de uma rotação]\label{ex:b2:affine:rotationcenter}
Deixar $ f(x, y) = (-y + 2,\ x)$. A parte linear é
$\varphi(x, y) = (-y, x)$: a rotação do ângulo $\frac\pi2$,
cujo \omterm{def:b2:reduction:eigen}{espectro} $\{\iu, -\iu\}$ evita $1$. Por
\cref{prop:b2:affine:fixedpoint} há exatamente um fixo
ponto: $ x = -y + 2$ e $ y = x $ dar $ x = 1$, $ y = 1$, então
$\Omega = (1, 1)$, e $ f $ \emph{é} a rotação do centro
$(1, 1)$ e ângulo $\frac\pi2$. A lição geral: quando $1
\notin \operatorname{Sp}\vec f $, classificando $ f $ custa um
sistema linear --- a geometria está inteiramente na parte linear,
a aritmética inteiramente na localização do centro.
\end{example}

\begin{remark}[Onde a linguagem afim é usada a seguir]
\omterm{def:b2:affine:barycenter}{Baricentros} e mapas afins são a gramática da geometria
capítulos à frente: linhas tangentes e planos são \omterm{def:b2:affine:subspace}{afim} objetos
(\cref{ch:b2:curves,ch:b2:surfaces}), um \omterm{def:b2:affine:subspace}{afim} mudança de
variáveis multiplica áreas e volumes por
$\abs{\det \vec f}$ (\cref{ch:b2:multint}) e expectativa
é um \omterm{def:b2:affine:barycenter}{baricentro} com pesos dados por uma lei de probabilidade, que
é por isso que a convexidade governa a desigualdade de Jensen
(\cref{ch:b2:randomvar}). No volume do Ano 3 o mesmo
vocabulário de convexidade leva ao estudo de $ L^p $ \omterm{def:b2:nvs:norm}{normas} e de
desigualdades integrais.
\end{remark}

\begin{remark}[Perspectivas dentro deste volume]
Dois tópicos saem deste capítulo. O \emph{\omterm{def:b2:affine:subspace}{afim}} tópico:
linhas tangentes (\cref{ch:b2:curves}) e planos tangentes
(\cref{ch:b2:surfaces}) são \omterm{def:b2:affine:subspace}{afim} subespaços anexados a
objetos não lineares e a classificação das quádricas no
O capítulo de superfícies é executado na equação central deste capítulo
$ A\Omega = -b $. O \emph{\omterm{def:b2:affine:convex}{convexo}} o fio é mais longo: convexidade
de semiplanos e discos alimenta a teoria de Helly do
problema de fim de semana; convexidade de funções dá Jensen
desigualdade (\cref{ch:b2:randomvar}); e o teorema final de
o livro --- o critério de extinção para processos de ramificação
(\cref{ch:b2:genfun}) --- é decidido pela posição de um
curva convexa em relação à diagonal, uma imagem que pertence
para este capítulo tanto quanto para a probabilidade. \omterm{def:b2:affine:barycenter}{Baricentros}
volte para lá também: uma expectativa é uma \omterm{def:b2:affine:barycenter}{baricentro} com
pesos de probabilidade.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b2:affine:1}
Em $\R^3$, são os seguintes \omterm{def:b2:affine:subspace}{afim} subespaços? Dê instruções e
dimensões. $\{x + y + z = 1\}$; $\;\{x + y + z = 1,\ x - z =
3\}$; $\;\{x^2 + y^2 = 1\}$; o conjunto solução de $ MX = B $ por um
determinado sistema compatível.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:affine:2}
Prove que três pontos distintos $ A, B, C $ de um \omterm{def:b2:affine:def}{espaço afim} são
alinhado se se $ C $ é um \omterm{def:b2:affine:barycenter}{baricentro} de $ A $ e $ B $, se os vetores
$\vect{AB}, \vect{AC}$ estão vinculados. Deduzir o peso ao estilo Menelau
escrituração contábil: se $ C = \operatorname{bar}(A, 1 - t; B, t)$, localize
$ C $ para $ t = \frac12$, $ t = 2$, $ t = -1$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:affine:3}
Deixar $ f $ seja o mapa \omterm{def:b2:affine:subspace}{afim} de $\R^2$ dado por $ f(X) = MX + C $ com
$ M = \frac12\begin{pmatrix} 1 & 1\\ 1 & 1\end{pmatrix}$ e $ C =
(1, 0)^{\mathsf T}$. Determine a imagem de $ f $, seus pontos fixos
(se houver) e $ f \circ f $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:affine:4}
(Associatividade em ação) Em um triângulo $ ABC $, deixar $ I, J, K $ dividir
$ BC $, $ CA $, $ AB $ em proporções $\vect{BI} = \frac13\vect{BC}$,
$\vect{CJ} = \frac13\vect{CA}$, $\vect{AK} = \frac13\vect{AB}$.
Expressar $ I, J, K $ como \omterm{def:b2:affine:barycenter}{baricentros} e calcular o \omterm{def:b2:affine:barycenter}{baricentro} de
$(I,1;J,1;K,1)$: o que você acha e por que era previsível?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:affine:5}
Prove que um mapa $ f \colon \R^n \to \R^n $ preservando \emph{pontos médios}
($ f\bigl(\frac{A+B}{2}\bigr) = \frac{f(A) + f(B)}{2}$) e
\emph{\omterm{def:b2:metric:continuity}{contínuo}} é \omterm{def:b2:affine:subspace}{afim}. \emph{(Mostre o mapa vetorial $ u \mapsto
f(O + u) - f(O)$ é aditivo através de pontos médios, então
$\Q $-homogêneo, então $\R $-homogêneo por continuidade ---
a mesma estratégia de densidade usada para a equação funcional de Cauchy em
o volume do Ano 1; derivar novamente as etapas necessárias aqui.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:affine:6}
Classifique o mapa \omterm{def:b2:affine:subspace}{afim} $ f(x, y) = (y + 1,\; x - 1)$ do
Plano euclidiano: parte linear, pontos fixos, natureza geométrica
(reflexão? deslizamento?). Calcular $ f \circ f $ e concluir.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:affine:7}
(Radão) Deixe $ A_1, \dots, A_{n+2}$ ser pontos de um \omterm{def:b2:affine:def}{espaço afim} de
dimensão $ n $. Prove que eles podem ser divididos em dois disjuntos
grupos cujos cascos \omterm{def:b2:affine:convex}{convexos} cruzar. \emph{(Como em Carathéodory
prova, encontre pesos $\mu_i $, nem todos zero, com $\sum\mu_i = 0$ e
$\sum\mu_i\vect{OA_i} = 0$; separar pesos positivos e negativos
e normalizar ambos os lados.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:affine:8}
Deixar $ f $ feijão \omterm{def:b2:affine:subspace}{afim} endomorfismo de $\R^n $ com $ f \circ f = f $.
Prove isso $ f $ é o \omterm{def:b2:affine:subspace}{afim} projeção sobre o \omterm{def:b2:affine:subspace}{afim} subespaço
$\operatorname{Fix}(f) = \operatorname{im} f $ ao longo da direção
$\ker\vec f $, e que, inversamente, todas essas projeções são
idempotente. \emph{(Mostre primeiro que $\operatorname{im} f $ consiste
de pontos fixos.)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:affine:9}
Deixar $ G = \operatorname{bar}(A, 1;\ B, 2;\ C, 3)$ em um triângulo
$ ABC $. Usando associatividade, mostre que a reta $ AG $ conhece $ BC $
no $ M = \operatorname{bar}(B, 2;\ C, 3)$ e localize $ G $ no
segmento $\intcc AM $; localize da mesma forma a interseção de $ BG $
com $ CA $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:affine:10}
Para $\lambda \neq 0$, o \emph{homotetia} $ h_{\Omega,
\lambda}$ é o mapa \omterm{def:b2:affine:subspace}{afim} consertando $\Omega $ com parte linear
$\lambda\,\mathrm{id}$. Prove que a composição
$ h_{\Omega', \mu} \circ h_{\Omega, \lambda}$ é uma homotetia de
proporção $\lambda\mu $ quando $\lambda\mu \neq 1$, e uma tradução
quando $\lambda\mu = 1$; no caso $\lambda = \mu = -1$ (dois
reflexões pontuais), calcule o vetor de translação.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:affine:11}
(Menelau) Em um triângulo $ ABC $, deixar $ A' \in (BC)$, $ B' \in
(CA)$, $ C' \in (AB)$, todos distintos dos vértices, e
definir $\alpha, \beta, \gamma $ por $\vect{A'B} =
\alpha\,\vect{A'C}$, $\vect{B'C} = \beta\,\vect{B'A}$,
$\vect{C'A} = \gamma\,\vect{C'B}$. Prove isso $ A', B', C'$ são
alinhado se e somente se $\alpha\beta\gamma = 1$. \emph{(Escrever
cada ponto como um \omterm{def:b2:affine:barycenter}{baricentro} de dois vértices; mostre que três
os pontos são alinhados se suas linhas de coordenadas baricêntricas com
respeito a $(A, B, C)$ formar um singular $3 \times 3$ matriz.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:affine:12}
Prove que o \omterm{def:b2:affine:convex}{casco convexo} de um \omterm{def:b2:metric:compact}{compactar} subconjunto $ K $ de $\R^n $
é \omterm{def:b2:metric:compact}{compactar}. \emph{(Por \cref{thm:b2:affine:caratheodory},
$\operatorname{conv}(K)$ é a imagem de um \omterm{def:b2:metric:compact}{compactar} definido sob um
\omterm{def:b2:metric:continuity}{contínuo} mapa.)} Mostre com um exemplo em $\R^2$ que o \omterm{def:b2:affine:convex}{convexo}
casco de um \emph{fechado} o conjunto não precisa ser fechado.
\end{exercise}

\begin{problem}[{Problema de fim de semana --- teorema de Helly e dois de
seus dividendos}]\label{pb:b2:affine:1}
Lema de Radon (\cref{exo:b2:affine:7}) diz que $ n + 2$
pontos de um $ n $-dimensional \omterm{def:b2:affine:def}{espaço afim} sempre dividido em
dois grupos com interseção cascos \omterm{def:b2:affine:convex}{convexos}. Este problema se transforma
aquele fato da álgebra linear em uma cadeia de teoremas de
geometria combinatória: teorema da interseção de Helly, o
teorema do \omterm{pb:b2:affine:1}{ponto central} (uma mediana bidimensional) e o de Jung
teorema de cobertura. Durante todo o tempo, o avião é $\R^2$ com seu
estrutura euclidiana usual, e $\det $ é o determinante em
a base canônica.\index{Teorema de Helly}\index{Radônio
lema}\index{Teorema de Jung}\index{ponto central}

\medskip
\noindent\textbf{Part I --- Barycentric coordinates.} Pontos
$ A_0, \dots, A_k $ são \emph{afimmente independente} quando o
vetores $\vect{A_0A_1}, \dots, \vect{A_0A_k}$ são linearmente
independente.
\begin{enumerate}
  \item Mostre isso \omterm{def:b2:affine:subspace}{afim} a independência não depende
        escolha do ponto base $ A_0$, e que é
        equivalente a: sempre que duas famílias de pesos, cada
        resumindo para $1$, defina o mesmo \omterm{def:b2:affine:barycenter}{baricentro} de $(A_0,
        \dots, A_k)$, os pesos coincidem.
  \item Deixar $(A, B, C)$ ser afimmente independente no plano.
        Mostre que cada ponto $ M $ admite um triplo único
        $(\alpha, \beta, \gamma)$ com $\alpha + \beta +
        \gamma = 1$ e $ M = \operatorname{bar}(A, \alpha; B,
        \beta; C, \gamma)$ --- isso é \emph{baricêntrico
        coordenadas}.
  \item Prove o determinante fórmulas
        \[
        \alpha = \frac{\det(\vect{MB},
        \vect{MC})}{\det(\vect{AB}, \vect{AC})},
        \qquad
        \beta = \frac{\det(\vect{MC},
        \vect{MA})}{\det(\vect{AB}, \vect{AC})},
        \qquad
        \gamma = \frac{\det(\vect{MA},
        \vect{MB})}{\det(\vect{AB}, \vect{AC})} :
        \]
        coordenadas baricêntricas são proporções de áreas sinalizadas.
  \item As linhas $ BC $, $ CA $, $ AB $ são as linhas de coordenadas
        $\{\alpha = 0\}$, $\{\beta = 0\}$, $\{\gamma = 0\}$.
        Mostre isso $ M $ está no triângulo fechado
        $\operatorname{conv}\{A, B, C\}$ se $\alpha, \beta,
        \gamma \geq 0$, e que as três linhas cortam o plano
        em exatamente sete regiões, classificadas pelos sinais de
        $(\alpha, \beta, \gamma)$ (o padrão de sinal $(-, -,
        -)$ sendo impossível).
  \item Deixar $ u \colon \R^2 \to \R $ feijão mapa \omterm{def:b2:affine:subspace}{afim} (um
        \emph{\omterm{def:b2:affine:subspace}{afim} forma}). Mostrar $ u(M) = \alpha\,u(A) +
        \beta\,u(B) + \gamma\,u(C)$, que os conjuntos de níveis de um
        inconstante \omterm{def:b2:affine:subspace}{afim} forma são linhas, que cada linha
        surge desta forma, e que os semiplanos fechados $\{u
        \geq c\}$ são \omterm{def:b2:affine:convex}{convexo}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Radon partitions, refined.} Um
família de $ n + 2$ pontos de $\R^n $ está em \emph{geral
posição} quando cada $ n + 1$ deles são afimmente independentes.
Um \emph{\omterm{def:b2:affine:subspace}{afim} dependência} de $(A_1, \dots, A_{n+2})$ é um
família $(\mu_i)$ com $\sum_i \mu_i = 0$ e $\sum_i
\mu_i\,\vect{OA_i} = 0$ para uma (portanto, cada) origem $ O $.
\begin{enumerate}[resume]
  \item Calcule um diferente de zero \omterm{def:b2:affine:subspace}{afim} dependência dos quatro
        pontos $ A_1 = (0,0)$, $ A_2 = (3,0)$, $ A_3 = (0,3)$,
        $ A_4 = (1,1)$; dê a partição Radon e o Radon
        ponto.
  \item Mostre que para pontos na posição geral o vetor
        espaço de \omterm{def:b2:affine:subspace}{afim} dependências tem dimensão exatamente
        $1$, e que uma dependência diferente de zero tem \emph{não}
        coeficiente de fuga.
  \item Deduza que a partição Radon de $ n + 2$ pontos em
        posição geral é única (até trocar os dois
        blocos), sendo cada bloco o conjunto de índices onde
        $\mu_i $ tem um sinal fixo.
  \item Para quatro pontos do plano em posição geral,
        mostre a dicotomia: ou a partição tem tipo
        $(1, 3)$ --- um ponto interior ao triângulo de
        os outros três --- ou digite $(2, 2)$: os quatro pontos
        estão em \omterm{def:b2:affine:convex}{convexo} posição e os segmentos que unem o
        dois pares (as diagonais) se cruzam, no Radon
        ponto.
  \item Execute a questão 6 para o quadrado unitário $(0,0)$,
        $(1,0)$, $(1,1)$, $(0,1)$: dependência, partição,
        Ponto de radônio.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Helly's theorem in the plane.}
\begin{enumerate}[resume]
  \item Deixar $ C_1, C_2, C_3, C_4$ ser \omterm{def:b2:affine:convex}{convexo} subconjuntos de $\R^2$,
        quaisquer três dos quais têm um ponto comum. Escolha $ x_i \in
        \bigcap_{j \neq i} C_j $ e aplique o lema de Radon a
        $ x_1, \dots, x_4$: mostre que o ponto Radon pertence
        para todos os quatro conjuntos. \emph{(Para cada $ k $, o bloco não
        contendo $ x_k $ consiste em pontos de $ C_k $.)}
  \item (Helly) Deixe $ C_1, \dots, C_m $ ($ m \geq 3$) ser \omterm{def:b2:affine:convex}{convexo}
        subconjuntos de $\R^2$, quaisquer três dos quais se cruzam.
        Provar $\bigcap_{i=1}^m C_i \neq \emptyset $, por
        indução ligada $ m $: substituir $ C_{m-1}$ e $ C_m $ por
        $ C_{m-1} \cap C_m $ e verifique a hipótese para o
        nova família usando a questão 11.
  \item Três contra-exemplos, um por hipótese: (a) o
        três arestas fechadas de um triângulo se cruzam aos pares
        mas não têm ponto comum ($3$ não pode ser reduzido para
        $2$); (b) os quatro conjuntos $ S_i = \{x_1, \dots, x_4\}
        \setminus \{x_i\}$, para quatro pontos em geral
        posição, satisfaça a hipótese de interseção tripla
        mas não a conclusão (a convexidade é importante); (c) o
        semiplanos fechados $ H_k = \intco{k}{+\infty} \times
        \R $, $ k \in \N $, interseção de pares e triplos
        mas $\bigcap_k H_k = \emptyset $ (famílias infinitas
        preciso compacidade).
  \item (\omterm{def:b2:metric:compact}{Compactar} Helly) Deixe $(K_i)_{i \in I}$ ser um arbitrário
        família de \omterm{def:b2:metric:compact}{compactar} \omterm{def:b2:affine:convex}{convexo} subconjuntos de $\R^2$, quaisquer três
        dos quais se cruzam. Usando a questão 12 e o
        Propriedade Borel-Lebesgue
        (\cref{thm:b2:metric:borellebesgue}), mostrar
        $\bigcap_{i \in I} K_i \neq \emptyset $.
  \item (Primeiro dividendo) Deixe $ S $ ser um conjunto finito de pontos de
        o avião e $ r > 0$. Mostre: se cada três pontos de
        $ S $ encontra-se em algum disco fechado de raio $ r $, então $ S $
        encontra-se em um disco fechado de raio $ r $. \emph{(Aplicar
        Helly para os discos $\overline D(p, r)$, $ p \in S $.)}
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The centerpoint theorem.} UM
\emph{\omterm{pb:b2:affine:1}{ponto central}} de um conjunto finito $ S $ de $ n $ pontos do
avião é um ponto $ c $ (não necessariamente em $ S $) tal que cada
semiplano fechado contendo $ c $ contém pelo menos $ n/3$
pontos de $ S $.
\begin{enumerate}[resume]
  \item (Dimensão $1$) Para reais $ x_1 \leq \dots \leq x_n $,
        mostre que a mediana $ c = x_{\lceil n/2 \rceil}$
        satisfaz: toda meia linha fechada contendo $ c $
        contém pelo menos $ n/2$ do $ x_i $.
  \item (Lema de contagem) Se $ A, B, C $ são subconjuntos de $ S $ com
        $\abs A, \abs B, \abs C > \tfrac{2n}3$, mostrar $ A \cap
        B \cap C \neq \emptyset $.
  \item Deixar $ m = \floor{2n/3} + 1$ e deixe $\mathcal F $ ser
        a família (finita) do cascos \omterm{def:b2:affine:convex}{convexos}
        $\operatorname{conv}(T)$, $ T \subseteq S $, $\abs T =
        m $. Mostre que quaisquer três membros de $\mathcal F $ tem
        um ponto comum, e deduza de Helly um ponto $ c $
        comum a todos eles.
  \item Prove que isso $ c $ é um \omterm{pb:b2:affine:1}{ponto central} de $ S $: o
        \emph{teorema do \omterm{pb:b2:affine:1}{ponto central}}. \emph{(Se um fechado
        meio plano através $ c $ continha menos de $ n/3$
        pontos, seu \omterm{def:b2:metric:topology}{abrir} complemento conteria um conjunto $ T $
        de $ m $ pontos, e $\operatorname{conv}(T)$ faria
        evitar $ c $.)}
  \item Nitidez: deixe $ n = 3k $ e lugar $ k $ pontos em cada
        de três discos de pequeno raio $\varepsilon $ centrado
        nos vértices de um grande triângulo. Mostre isso para
        \emph{todo} apontar $ c $ do avião alguns fechados
        meio plano contendo $ c $ contém no máximo $ n/3$
        pontos de $ S $, então a constante $1/3$ não pode ser
        melhorou. \emph{(Entre as três direções de $ c $
        aos centros do disco, dois formam um ângulo no máximo
        $2\pi/3$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Jung's theorem and synthesis.}
\begin{enumerate}[resume]
  \item (Lema do triângulo) Deixe $ P, Q, R $ ser três pontos com
        distâncias aos pares $\leq 1$. Mostre que eles estão em um
        disco fechado de raio $1/\sqrt3$. \emph{(Se algum
        ângulo é $\geq \pi/2$, pegue o disco por mais tempo
        lado como diâmetro, usando a fórmula mediana
        $\norm{RM}^2 = \tfrac12\norm{RP}^2 +
        \tfrac12\norm{RQ}^2 - \tfrac14\norm{PQ}^2$; se o
        o triângulo é agudo, limitado pelo circunraio $ a/(2\sin
        \widehat A)$ usando seu maior ângulo, que fica em
        $\intco{\pi/3}{\pi/2}$.)}
  \item (Jung) Deduza: cada \omterm{def:b2:metric:compact}{compactar} subconjunto do plano de
        diâmetro $\leq 1$ está contido em um disco fechado de
        raio $1/\sqrt3$.
  \item Nitidez: para o triângulo equilátero $ A_1A_2A_3$
        de lado $1$ com centróide $ G $, prove o Leibniz
        identidade $\sum_i \norm{\vect{OA_i}}^2 =
        3\norm{\vect{OG}}^2 + \sum_i \norm{\vect{GA_i}}^2$
        para cada ponto $ O $, e concluímos que qualquer disco
        contendo os três vértices tem raio $\geq
        1/\sqrt3$, com igualdade apenas para o circundisco.
  \item (Helly em $\R^n $) Enuncie e prove o teorema de Helly em
        $\R^n $: se forem finitos conjuntos \omterm{def:b2:affine:convex}{convexos} são tais que
        qualquer $ n + 1$ deles se cruzam, então todos eles
        cruzar. \emph{(Lema de Radon
        \cref{exo:b2:affine:7} alças $ n + 2$ conjuntos; então
        induzir como na questão 12.)}
  \item Síntese. Monte a corrente
        \[
        \text{affine dependence} \Rightarrow \text{Radon}
        \Rightarrow \text{Helly} \Rightarrow
        \text{centerpoint and Jung},
        \]
        indicando em uma frase cada: onde álgebra linear
        entra, onde entram os sinais dos pesos, onde
        convexidade entra, e qual etapa única usou o
        dimensão do plano. O que as constantes $3$ (em
        Helly), $1/3$ (\omterm{pb:b2:affine:1}{ponto central}) e $1/\sqrt3$ (Jung)
        tornar-se em $\R^n$? (Declarar sem provas.)
\end{enumerate}
\end{problem}
