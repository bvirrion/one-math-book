\chapter{Integrais de Linha e Integrais Múltiplos}\label{ch:b2:multint}

Este capítulo estende a integração de intervalos para curvas e para
domínios do plano e do espaço. Integrais de linha integrar um
\emph{forma \omterm{def:b2:diffcalc:differential}{diferencial}} $P\,\dd x + Q\,\dd y $ ao longo de uma orientação
arco; integrais duplas e triplas integram funções sobre
domínios bidimensionais e tridimensionais. As duas teorias se encontram no
\emph{Teorema de Green-Riemann}, o fundamental bidimensional
teorema do cálculo, e a principal ferramenta computacional é
o \emph{fórmula de mudança de variáveis}, cujo fator de distorção é
o Jacobiano absoluto determinante.

\section{Integrais de linha}

\begin{definition}[Forma diferencial; integral de linha]\label{def:b2:multint:lineint}
Deixar $ U \subseteq \R^2$ ser \omterm{def:b2:metric:topology}{abrir}. UM \emph{diferencial
formulário}\index{diferencial
formulário} de
grau $1$ e classe $\mathcal{C}^0$ sobre $ U $ é uma expressão
$\omega = P\,\dd x + Q\,\dd y $ com $ P, Q \colon U \to \R $
\omterm{def:b2:metric:continuity}{contínuo} --- formalmente, um \omterm{def:b2:metric:continuity}{contínuo} mapa de $ U $ para dentro do \omterm{def:b2:linalg:dual}{dual}
de $\R^2$, $\omega(M) = P(M)\,e_1^* + Q(M)\,e_2^*$. Por um
$\mathcal{C}^1$ arco $\gamma \colon [a, b] \to U $, $\gamma(t) =
(x(t), y(t))$, o \emph{integral de linha}\index{integral de linha} de
$\omega $ junto $\gamma $
é
\[
\int_\gamma \omega
= \int_a^b \Bigl(P(\gamma(t))\,x'(t)
+ Q(\gamma(t))\,y'(t)\Bigr)\,\dd t .
\]
As definições estendem-se literalmente a $\R^3$ (formulários $ P\,\dd x +
Q\,\dd y + R\,\dd z $) e por partes $\mathcal{C}^1$ arcos (soma
sobre as peças).
\end{definition}

\begin{proposition}[Invariância e orientação]\label{prop:b2:multint:lineinv}
O \omterm{def:b2:multint:lineint}{integral de linha} permanece inalterado sob um aumento $\mathcal{C}^1$
\omterm{def:b2:curves:reparam}{mudança de parâmetro}, e muda o sinal para um decrescente. Isso
portanto, depende apenas do \emph{orientado} \omterm{def:b2:curves:reparam}{arco geométrico}.
\end{proposition}

\begin{proof}
Se $\theta \colon [c, d] \to [a, b]$ é um \omterm{def:b2:curves:reparam}{mudança de parâmetro} e
$\tilde\gamma = \gamma \circ \theta $, então pela regra da cadeia e
a mudança de variáveis de uma variável $ t = \theta(u)$,
\[
\int_{\tilde\gamma}\omega
= \int_c^d \bigl(P(\gamma(\theta(u)))\,x'(\theta(u))
+ Q(\gamma(\theta(u)))\,y'(\theta(u))\bigr)\,\theta'(u)\,\dd u
= \pm\int_a^b \bigl(Px' + Qy'\bigr)(t)\,\dd t ,
\]
com sinal $+$ se $\theta $ está aumentando ($\theta(c) = a $) e $-$
se estiver diminuindo (a troca de limites).
\end{proof}

\begin{example}[Trabalho de uma força; circulação]\label{ex:b2:multint:work}
Se $ F = (P, Q)$ é um campo de força, $\int_\gamma P\dd x + Q\dd y =
\int_a^b \langle F(\gamma(t)), \gamma'(t)\rangle\,\dd t $ é o
\emph{trabalhar} de $ F $ junto $\gamma $. Para $\omega = -y\,\dd x +
x\,\dd y $ ao longo do círculo unitário no sentido anti-horário $\gamma(t) =
(\cos t, \sin t)$:
\[
\int_\gamma \omega
= \int_0^{2\pi}\bigl((-\sin t)(-\sin t)
+ \cos t\cos t\bigr)\,\dd t = 2\pi ,
\]
o dobro do fechado \omterm{def:b2:multint:domain}{área} --- uma primeira sugestão de Green--Riemann.
\end{example}

\begin{example}[Uma integral, duas parametrizações, uma
armadilha de sinal]\label{ex:b2:multint:semicircle}
Calcular $\int_\gamma x\,\dd y $ ao longo da unidade superior
semicírculo de $(1, 0)$ para $(-1, 0)$. Com $\gamma(t) =
(\cos t, \sin t)$, $ t \in \intcc0\pi $:
\[
\int_0^\pi\cos t\cdot\cos t\,\dd t = \frac\pi2 .
\]
Com a parametrização do gráfico $ x \mapsto (x, \sqrt{1 -
x^2})$, $ x $ de $1$ para $-1$ (observe a direção!):
\[
\int_1^{-1}x\cdot\frac{-x}{\sqrt{1 - x^2}}\,\dd x
= \int_{-1}^{1}\frac{x^2}{\sqrt{1 - x^2}}\,\dd x
= \frac\pi2
\]
($ x = \sin u $ reduz-o a uma integral de Wallis). Mesmo valor,
como \cref{prop:b2:multint:lineinv} garantias --- mas apenas
porque ambas as corridas vão de $(1,0)$ para $(-1,0)$; revertendo o
viajar inverte o sinal. Fechando o caminho ao longo do $ x $-eixo
(onde $\dd y = 0$) não acrescenta nada, e o total $\frac\pi2$
é o \omterm{def:b2:multint:domain}{área} do meio-disco: a primeira instância do
\omterm{def:b2:multint:domain}{área} limite fórmulas de Green--Riemann abaixo.
\end{example}

\begin{definition}[Formulários exatos e fechados]\label{def:b2:multint:exact}
O formulário $\omega = P\,\dd x + Q\,\dd y $ de classe $\mathcal{C}^0$
é \emph{exato}\index{exato} sobre $ U $ se houver $ f \in \mathcal{C}^1(U)$
(um \emph{potencial}\index{potencial}) com $\omega = \dd f $, ou seja\ $ P = f_x $ e $ Q
= f_y $. UM $\mathcal{C}^1$ o formulário é \emph{fechado}\index{fechado} se $ P_y = Q_x $ sobre
$ U $.
\end{definition}

\begin{theorem}[Teorema fundamental para linha
integrais]\label{thm:b2:multint:ftc}
Se $\omega = \dd f $ é exato e $\gamma $ é um pedaço
$\mathcal{C}^1$ arco em $ U $ de $ A $ para $ B $, então
\[
\int_\gamma \omega = f(B) - f(A) .
\]
Em particular a integral de um forma exata ao longo de qualquer arco fechado
é zero e cada exato $\mathcal{C}^1$ formulário está fechado.
\end{theorem}

\begin{proof}
$\frac{\dd}{\dd t}f(\gamma(t)) = f_x(\gamma(t))x'(t) +
f_y(\gamma(t))y'(t)$ pela regra da cadeia
(\cref{ch:b2:diffcalc}), então o integrando em
\cref{def:b2:multint:lineint} é a derivada de $ t \mapsto
f(\gamma(t))$, e o teorema fundamental do cálculo dá o
resultado em cada peça; o telescópio de valores intermediários. Fechamento
de exato $\mathcal{C}^1$ formas é o teorema de Schwarz: $ P_y =
f_{xy} = f_{yx} = Q_x $.
\end{proof}

\begin{example}[Reconstruindo um potencial]\label{ex:b2:multint:potential}
Deixar $\omega = y\,\eu^{xy}\,\dd x + (x\,\eu^{xy} + 2y)\,\dd y $
sobre $\R^2$. Está fechado: ambas as derivadas cruzadas são iguais
$\eu^{xy}(1 + xy)$. Para encontrar um \omterm{def:b2:multint:exact}{potencial}, integrar $ P $ em
$ x $ em fixo $ y $:
\[
f(x, y) = \int y\,\eu^{xy}\,\dd x = \eu^{xy} + c(y),
\]
então ajuste $ c $ combinando $ f_y $: $ x\,\eu^{xy} + c'(y) =
x\,\eu^{xy} + 2y $ dá $ c(y) = y^2$. Então $ f(x,y) = \eu^{xy} +
y^2$, e para qualquer pedaço $\mathcal C^1$ arco de $(0,0)$
para $(1,1)$,
\[
\int_\gamma\omega = f(1,1) - f(0,0) = (\eu + 1) - 1 = \eu ,
\]
independentemente do caminho --- a receita de duas etapas (integrar
em $ x $, correto em $ y $) é o inverso prático de
\cref{thm:b2:multint:ftc} em domínios onde formulários fechados são
exato.
\end{example}

\begin{example}[Fechado não implica exato]\label{ex:b2:multint:angleform}
Sobre $ U = \R^2 \setminus \{0\}$, o \emph{forma de ângulo}
\[
\omega = \frac{-y\,\dd x + x\,\dd y}{x^2 + y^2}
\]
está fechado (cálculo direto: ambos $ P_y $ e $ Q_x $ igual
$\frac{y^2 - x^2}{(x^2+y^2)^2}$), mas é integral ao longo da unidade
círculo é $2\pi \neq 0$ (mesmo cálculo que
\cref{ex:b2:multint:work}, dividido por $1$): $\omega $ não é exato
sobre $ U $. Localmente, $\omega = \dd\theta $ para uma determinação
$\theta $ do ângulo \omterm{pb:b2:quadratic:1}{polar}; a falha é global --- o ângulo
não pode ser definido continuamente ao redor do furo. Em domínios
sem buracos a patologia desaparece: em um \emph{em forma de estrela}
\omterm{def:b2:metric:topology}{abrir} definido, cada fechado $\mathcal{C}^1$ o formulário é exato
(Lema de Poincaré, \cref{exo:b2:multint:8}).
\end{example}

\section{Integrais duplas}

Tomamos como certa a teoria da integral de Riemann em um
variável (volume do ano 1 e \cref{ch:b2:integration}) e esboço
sua versão de duas variáveis. Uma função $ f $ \omterm{def:b2:metric:continuity}{contínuo} em um retângulo
$ R = [a, b] \times [c, d]$ tem uma integral dupla $\iint_R f $,
definido por Riemann soma sobre grades exatamente como em uma variável, e
calculado por iteração:

\begin{theorem}[Fubini em um retângulo]\label{thm:b2:multint:fubini}
Para $ f $ \omterm{def:b2:metric:continuity}{contínuo} sobre $ R = [a,b] \times [c,d]$,
\[
\iint_R f
= \int_a^b \Bigl(\int_c^d f(x, y)\,\dd y\Bigr)\dd x
= \int_c^d \Bigl(\int_a^b f(x, y)\,\dd x\Bigr)\dd y .
\]
\end{theorem}

\begin{proof}
Definir $ F(x) = \int_c^d f(x, y)\,\dd y $. Uniforme continuidade de $ f $ no \omterm{def:b2:metric:compact}{compactar} $ R $ faz $ F $ \omterm{def:b2:metric:continuity}{contínuo} (estimativa dominada:
$\abs{F(x) - F(x')} \leq (d - c)\sup_y\abs{f(x,y) - f(x',y)}$).
Agora subdivida $[a,b]$ e $[c,d]$ em $ n $ partes iguais, dando uma
grade de células $ R_{ij}$ de \omterm{def:b2:multint:domain}{área} $\Delta x\,\Delta y $. Em cada célula,
$\inf_{R_{ij}} f \cdot \Delta x \Delta y \leq
\int_{x_{i-1}}^{x_i}\int_{y_{j-1}}^{y_j} f(x,y)\,\dd y\,\dd x \leq
\sup_{R_{ij}} f \cdot \Delta x \Delta y $ pela monotonicidade do
integral de uma variável (aplicada duas vezes). Somando as células, o
integral iterada $\int_a^b F $ fica espremido entre a parte inferior e
somas superiores de Riemann da grade; por uniforme continuidade ambas as somas
convergem para o valor comum que define $\iint_R f $ como $ n \to
\infty $. O mesmo argumento se aplica aos papéis de $ x $ e $ y $
trocados, então ambas as integrais iteradas são iguais $\iint_R f $.
\end{proof}

\begin{remark}
Continuidade em um \omterm{def:b2:metric:compact}{compactar} retângulo está fazendo um trabalho real em
A prova de Fubini: fornece o uniforme continuidade isso
aperta as somas de Riemann. Para integrandos mais selvagens, o
declaração falha genuinamente --- existem funções cujos dois
integrais iteradas existem e diferem. O general honesto
teorema, com integrabilidade como única hipótese, é
O teorema de Fubini para a integral de Lebesgue, provado no
Volume do ano 3; tudo neste capítulo permanece dentro do
\omterm{def:b2:metric:continuity}{contínuo} configuração onde a prova elementar acima é
\omterm{def:b2:metric:complete}{completo}.
\end{remark}

\begin{definition}[Domínios elementares]\label{def:b2:multint:domain}
Um domínio $ D \subseteq \R^2$ é \emph{$ y $-elementar} se
\[
D = \{(x, y) : a \leq x \leq b,\
\varphi_1(x) \leq y \leq \varphi_2(x)\}
\]
com $\varphi_1 \leq \varphi_2$ \omterm{def:b2:metric:continuity}{contínuo} sobre $[a,b]$
($ x $-elementar: simetricamente). Para $ f $ \omterm{def:b2:metric:continuity}{contínuo} em um
$ y $-elementar $ D $,
\[
\iint_D f
= \int_a^b\Bigl(
\int_{\varphi_1(x)}^{\varphi_2(x)} f(x,y)\,\dd y\Bigr)\dd x ,
\]
e um verifica (estendendo $ f $ por um argumento de aproximação, ou
subdividindo) que quando $ D $ é elementar em ambas as direções o
duas integrais iteradas concordam. Domínios divididos em um número finito
peças elementares são tratadas por aditividade, e as
\emph{área}\index{área} de $ D $ é $\operatorname
{Area}(D) = \iint_D 1$.
\end{definition}

\begin{example}\label{ex:b2:multint:triangle}
No triângulo $ D = \{0 \leq x \leq 1,\ 0 \leq y \leq x\}$:
\[
\iint_D xy \,\dd x\,\dd y
= \int_0^1 x\Bigl(\int_0^x y\,\dd y\Bigr)\dd x
= \int_0^1 x\cdot\frac{x^2}{2}\,\dd x = \frac18 .
\]
Trocando o pedido ($ x $ de $ y $ para $1$): $\int_0^1
y\bigl(\int_y^1 x\,\dd x\bigr)\dd y = \int_0^1
y\,\frac{1 - y^2}{2}\,\dd y = \frac18$ --- mesmo valor, diferente
computação: escolher bem a ordem de integração é metade do
artesanato.
\end{example}

\begin{example}[Quando apenas um pedido funciona]\label{ex:b2:multint:swaporder}
Calcular $ I = \displaystyle\int_0^1\!\!\int_x^1
\eu^{y^2}\,\dd y\,\dd x $. Como está escrito, a integral interna
$\int\eu^{y^2}\dd y $ não tem antiderivada elementar: o
a computação está travada. Mas o domínio é o triângulo $0 \leq
x \leq y \leq 1$, que é elementar em ambas as direções;
trocando o pedido,
\[
I = \int_0^1\!\!\int_0^y \eu^{y^2}\,\dd x\,\dd y
= \int_0^1 y\,\eu^{y^2}\,\dd y
= \Bigl[\tfrac12\eu^{y^2}\Bigr]_0^1 = \frac{\eu - 1}{2} .
\]
A variável interna $ x $ não apareceu em nenhum lugar do integrando, então
integrá-lo primeiro produziu exatamente o fator $ y $ isso
torna a integral externa imediata. Moral: Fubini não é apenas
uma licença para iterar --- é uma licença para \emph{escolher},
e a ordem correta pode transformar uma integral impossível em uma
uma linha. Sempre esboce o domínio e leia ambos
descrições antes de começar.
\end{example}

\begin{theorem}[Mudança de variáveis]\label{thm:b2:multint:changevar}
Deixar $\Phi \colon U' \to U $ seja um $\mathcal{C}^1$ difeomorfismo
entre \omterm{def:b2:metric:topology}{abrir} conjuntos de $\R^2$, deixar $ K \subseteq U $ seja um \omterm{def:b2:metric:compact}{compactar}
domínio cortado em pedaços elementares com $ K' = \Phi^{-1}(K)$ e
deixe $ f $ ser \omterm{def:b2:metric:continuity}{contínuo} sobre $ K $. Então
\[
\iint_K f(x, y)\,\dd x\,\dd y
= \iint_{K'} f\bigl(\Phi(u, v)\bigr)\,
\abs{\det J_\Phi(u, v)}\,\dd u\,\dd v .
\]
\end{theorem}
\admitted

\begin{remark}
O \omterm{def:b2:metric:complete}{completo} prova --- aproximando $\Phi $ por seu \omterm{def:b2:diffcalc:differential}{diferencial} ligado
uma grade fina e controlando as células limite --- é longo
não profundo; isso é feito na íntegra na teoria da medida do Ano 3, como um
consequência da teoria de Lebesgue. A heurística é a imagem
já usado para superfície \omterm{def:b2:multint:domain}{área}: um pequeno quadrado de lado $\dd u $ no
$(u, v)$ é mapeado, em primeira ordem, no paralelogramo estendido
por $\Phi_u\,\dd u $ e $\Phi_v\,\dd v $, cujo \omterm{def:b2:multint:domain}{área} é
$\abs{\det J_\Phi}\,\dd u\,\dd v $
(\cref{lem:b2:surfaces:lagrange}).
\end{remark}

\begin{remark}[Método: escolha da mudança de variáveis]
Três reflexos cobrem a maioria dos casos. \emph{Simetria do
integrando}: $ x^2 + y^2$ pede \omterm{pb:b2:quadratic:1}{polar}, uma estrutura de produto
apela à manutenção dos eixos cartesianos. \emph{Forma do
limite}: limites $ u(x,y) = c_1$, $ v(x,y) = c_2$ implorar por
as coordenadas $(u, v)$ eles mesmos, como no hiperbólico
exemplo de região abaixo --- o domínio se torna um retângulo,
que é toda a vitória. \emph{Estrutura linear}:
expressões em $ x + y $ e $ x - y $ convide o $45$-grau
rotação ou cisalhamento (\cref{ex:b2:multint:affine}). Em tudo
casos, três caixas para marcar antes de integrar: o mapa é um
bijeção do novo domínio no antigo; é jacobiano
é computado \emph{na direção realmente usada} (inverter em
o final é mais fácil); e o Jacobiano entra com seu
valor absoluto.
\end{remark}

\begin{example}[Coordenadas polares]\label{ex:b2:multint:polar}
$\Phi(\rho, \alpha) = (\rho\cos\alpha,\ \rho\sin\alpha)$ tem
\[
J_\Phi = \begin{pmatrix}
\cos\alpha & -\rho\sin\alpha\\
\sin\alpha & \rho\cos\alpha
\end{pmatrix},
\qquad \det J_\Phi = \rho ,
\]
então $\dd x\,\dd y = \rho\,\dd\rho\,\dd\alpha $. Para o disco $ D_R $
de raio $ R $:
\[
\iint_{D_R} e^{-(x^2 + y^2)}\,\dd x\,\dd y
= \int_0^{2\pi}\!\!\int_0^R e^{-\rho^2}\rho\,\dd\rho\,\dd\alpha
= \pi\bigl(1 - e^{-R^2}\bigr)
\xrightarrow[R\to\infty]{} \pi .
\]
Comparando com o quadrado $[-R, R]^2$ (que é espremido entre
os discos $ D_R $ e $ D_{R\sqrt2}$, todos os integrandos positivos) dá
$\bigl(\int_{-\infty}^\infty e^{-x^2}\dd x\bigr)^2 = \pi $:
\[
\boxed{\ \int_{-\infty}^{+\infty} e^{-x^2}\,\dd x = \sqrt{\pi}\ }
\]
--- a integral de Gauss novamente, agora pela sua prova mais famosa
(compare a derivação de uma variável em
\cref{ch:b2:integration}).
\end{example}

\begin{example}[Mudanças afins de variáveis]\label{ex:b2:multint:affine}
Por um mapa \omterm{def:b2:affine:subspace}{afim} $\Phi(u, v) = M(u, v)^{\mathsf T} + C $ com
$ M $ invertível, o Jacobiano é a matriz constante $ M $:
\omterm{def:b2:multint:domain}{áreas} são multiplicados pelo fator constante $\abs{\det M}$
--- a promessa feita em \cref{ch:b2:affine} agora é um teorema.
Dois usos imediatos. A elipse $\frac{x^2}{a^2} +
\frac{y^2}{b^2} \leq 1$ é a imagem do disco da unidade em
$(u, v) \mapsto (au, bv)$, então é \omterm{def:b2:multint:domain}{área} é $ ab \cdot \pi $ ---
sem cálculo. E para a integral de $ f(x + y)$ sobre o
quadrado $ K = \intcc01^2$, o cisalhamento $\Phi(u, v) = (u - v, v)$
(determinante $1$) transforma-o em uma integral de $ f(u)$ mais de um
paralelogramo, que Fubini corta em constante $ u $: com $ f =
\exp $,
\[
\iint_K \eu^{x+y}\,\dd x\,\dd y
= \Bigl(\int_0^1 \eu^x\,\dd x\Bigr)^2 = (\eu - 1)^2,
\]
como a estrutura do produto confirma. Escolhendo coordenadas
adaptado ao integrando --- não ao domínio --- é o
outra metade do ofício.
\end{example}


\begin{example}[Coordenadas adaptadas a uma curva
domínio]\label{ex:b2:multint:hyperbolicregion}
Deixar $ D $ ser a região do primeiro quadrante delimitada pelo
hipérboles $ xy = 1$ e $ xy = 3$ e as linhas $ y = x $ e
$ y = 3x $. Nas coordenadas $ u = xy $, $ v = y/x $ o domínio
torna-se o quadrado $\intcc13 \times \intcc13$; invertendo,
\[
x = \sqrt{u/v}, \qquad y = \sqrt{uv},
\qquad
\det J = x_uy_v - x_vy_u = \frac{1}{2v}
\]
(um cálculo de duas linhas com $ x = u^{1/2}v^{-1/2}$, $ y =
u^{1/2}v^{1/2}$). Por isso
\[
\operatorname{Area}(D)
= \int_1^3\!\!\int_1^3\frac{\dd u\,\dd v}{2v}
= 2\cdot\frac{\ln 3}{2} = \ln 3 \approx 1.10 .
\]
Tentando fatiar $ D $ em coordenadas cartesianas significa cortá-lo
em três partes com limites hiperbólicos e lineares ---
viável, triste e sujeito a erros. A moral se repete
\cref{ex:b2:multint:affine} com força total: leia o
equações de limite, e deixe \emph{eles} escolha o
coordenadas; o Jacobiano então converte o curvilíneo
célula da malha \omterm{def:b2:multint:domain}{área}, exatamente como $\rho $ fiz para \omterm{pb:b2:quadratic:1}{polar}
coordenadas.
\end{example}

\begin{example}[Valores médios]\label{ex:b2:multint:average}
O \emph{valor médio} de $ f $ sobre um domínio $ D $ é
$\frac1{\operatorname{Area}(D)}\iint_Df $. Amostra: a média
distância do centro para um ponto \omterm{def:b2:funcseq:def}{uniformemente} escolhido do
disco de raio $ R $ é
\[
\frac{1}{\pi R^2}\int_0^{2\pi}\!\!\int_0^R
\rho\cdot\rho\,\dd\rho\,\dd\alpha
= \frac{2\pi R^3/3}{\pi R^2} = \frac{2R}3 ,
\]
não $ R/2$: uniforme \omterm{def:b2:multint:domain}{área} coloca mais massa em raios grandes (o
anel no raio $\rho $ tem peso proporcional a $\rho $),
então a média fica além da metade. Entendendo isso
fator certo é exatamente o Jacobiano \omterm{pb:b2:quadratic:1}{polar} em ação, e o
a mesma ponderação explica o centróide $\bar z = 3R/8$ do
meia-bola calculada posteriormente no capítulo, em vez de $ R/2$.
\end{example}

\section{O teorema de Green-Riemann}

\begin{theorem}[Verde - Riemann]\label{thm:b2:multint:green}
Deixar $ K \subseteq \R^2$ seja um \omterm{def:b2:metric:compact}{compactar} domínio que é elementar em
ambas as direções (ou uma união finita de tais segmentos colados),
com limite $\partial K $ um pedaço $\mathcal{C}^1$ fechado
orientado para curva \emph{sentido anti-horário} (o domínio permanece no
esquerda). Para $ P, Q $ de classe $\mathcal{C}^1$ num bairro de
$ K $:
\[
\oint_{\partial K} P\,\dd x + Q\,\dd y
= \iint_K \Bigl(\frac{\partial Q}{\partial x}
- \frac{\partial P}{\partial y}\Bigr)\,\dd x\,\dd y .
\]
\end{theorem}

\begin{proof}
Primeiro, ambos os lados são aditivos sob corte $ K $ ao longo de um
segmentar em duas partes $ K_1, K_2$: as integrais duplas somam
por aditividade de $\iint $; quanto às integrais de fronteira, o
limites no sentido anti-horário de $ K_1$ e $ K_2$ cada travessia
o interior cortado uma vez, em \emph{oposto} direções, então em
a soma
\[
\oint_{\partial K_1} + \oint_{\partial K_2}
= \oint_{\partial K} + (\text{cut, both ways})
= \oint_{\partial K},
\]
as duas passagens ao longo do corte cancelam
(\cref{prop:b2:multint:lineinv}) e apenas o limite externo
sobrevive. Iterando um número finito de cortes, é suficiente tratar
um domínio elementar. Nós provamos $\oint P\,\dd x = -\iint_K P_y $ em um
$ y $-domínio elementar $ D = \{a \leq x \leq b,\ \varphi_1(x) \leq y
\leq \varphi_2(x)\}$; a identidade $\oint Q\,\dd y = \iint_K Q_x $
é \omterm{def:b2:quadratic:adjoint}{simétrico} ($ x $-elementar), e o teorema é a soma deles.

Calcule a integral dupla de Fubini e a de uma variável
teorema fundamental:
\[
\iint_D \frac{\partial P}{\partial y}\,\dd x\,\dd y
= \int_a^b \bigl(P(x, \varphi_2(x)) - P(x, \varphi_1(x))\bigr)
\,\dd x .
\]
Agora o limite de $ D $, no sentido anti-horário, consiste em: o inferior
gráfico $ y = \varphi_1(x)$ percorrido da esquerda para a direita, a direita
segmento vertical $ x = b $ (para cima), o gráfico superior $ y =
\varphi_2(x)$ atravessado \emph{direita para esquerda}, a vertical esquerda
segmento $ x = a $ (para baixo). Ao longo dos segmentos verticais $ x $ é
constante, então eles contribuem $0$ para $\oint P\,\dd x $; os gráficos,
parametrizado por $ x $, dar
\[
\oint_{\partial D} P\,\dd x
= \int_a^b P(x, \varphi_1(x))\,\dd x
- \int_a^b P(x, \varphi_2(x))\,\dd x
= -\iint_D \frac{\partial P}{\partial y}\,\dd x\,\dd y .
\qedhere
\]
\end{proof}

\begin{corollary}[Área perto do limite]\label{cor:b2:multint:area}
Sob as hipóteses de \cref{thm:b2:multint:green},
\[
\operatorname{Area}(K)
= \oint_{\partial K} x\,\dd y
= -\oint_{\partial K} y\,\dd x
= \frac12\oint_{\partial K} x\,\dd y - y\,\dd x .
\]
\end{corollary}

\begin{proof}
Aplicar Green--Riemann a $(P, Q) = (0, x)$, $(-y, 0)$ e
$\frac12(-y, x)$: cada vez $ Q_x - P_y = 1$.
\end{proof}

\begin{remark}[Escolhendo entre as três fórmulas de área]
As três fórmulas de limite são iguais, não intercambiáveis em
prática. Usar $\oint x\,\dd y $ quando a parametrização faz
$\dd y $ simples (gráficos sobre o $ y $-eixo), $-\oint y\,\dd x $
simetricamente e o \omterm{def:b2:quadratic:adjoint}{simétrico} meia soma quando o
parametrização trata $ x $ e $ y $ \omterm{def:b2:funcseq:def}{uniformemente} --- para a elipse
produziu um integrando constante, sem trigonometria
linearização em tudo. Em limites poligonais, a meia soma
torna-se a fórmula do cadarço de \cref{exo:b2:multint:12},
algoritmo dos topógrafos. E quando a fronteira é atravessada
no sentido horário pela parametrização dada, todas as três fórmulas
retornar \emph{menos} o \omterm{def:b2:multint:domain}{área}: um resultado negativo não é um
erro computacional, mas um relatório de orientação --- vire o
sinal, ou a parametrização.
\end{remark}

\begin{example}[Área da elipse]\label{ex:b2:multint:ellipsearea}
Para $ x = a\cos t $, $ y = b\sin t $, $ t \in [0, 2\pi]$:
\[
\operatorname{Area}
= \frac12\int_0^{2\pi}\bigl(a\cos t \cdot b\cos t
- b\sin t\cdot(-a\sin t)\bigr)\,\dd t
= \frac{ab}{2}\int_0^{2\pi}\dd t = \pi ab .
\]
\end{example}

\begin{example}[Green - Riemann como um
verificação cruzada]\label{ex:b2:multint:greencheck}
Pegar $ P = -y^3$, $ Q = x^3$ no disco unitário fechado $ D $.
Lado limite, com $\gamma(t) = (\cos t, \sin t)$:
\[
\oint_{\partial D}P\,\dd x + Q\,\dd y
= \int_0^{2\pi}\bigl(\sin^4 t + \cos^4 t\bigr)\dd t
= 2\pi\cdot\Bigl(\frac38 + \frac38\Bigr) = \frac{3\pi}2 ,
\]
por linearização ($\sin^4 + \cos^4 = \tfrac34 +
\tfrac14\cos4t $). Lado interno:
\[
\iint_D(Q_x - P_y)\,\dd x\,\dd y
= \iint_D 3(x^2 + y^2)\,\dd x\,\dd y
= 3\int_0^{2\pi}\!\!\int_0^1\rho^3\,\dd\rho\,\dd\alpha
= \frac{3\pi}2 .
\]
O mesmo número, dois cálculos muito diferentes --- e isso é
o uso prático: qualquer lado da identidade de Green é
mais fácil se torna o cálculo, o outro uma verificação. Para
circulações de campos polinomiais em torno de curvas fechadas, o
integral dupla é quase sempre o lado fácil.
\end{example}

\begin{remark}
Verde - Riemann explica \cref{ex:b2:multint:angleform}: para um
formulário fechado ($ Q_x = P_y $), a integral em torno do limite de qualquer
domínio contido em $ U $ desaparece. A forma do ângulo não é exato
apenas porque o furo na origem impede o disco delimitado
pelo círculo unitário de dentro $ U $ --- integrais de linha de
formulários fechados detectar as lacunas do domínio. (Empurrado ainda mais,
esta observação torna-se cohomologia de Rham.)
\end{remark}

\section{Integrais triplas}

A teoria se estende a três variáveis sem nenhuma ideia nova: Fubini
reduz $\iiint $ para três integrais de uma variável (seja por
\emph{fatiar}: $\iiint_K f = \int\bigl(\iint_{K_z}
f\bigr)\dd z $ sobre as fatias horizontais $ K_z $, ou por
\emph{empilhamento}: integrando em $ z $ primeiro ao longo de varas verticais),
e a fórmula de mudança de variáveis se mantém com o $3 \times 3$
Jacobiano.

\begin{example}[Coordenadas cilíndricas e esféricas]\label{ex:b2:multint:spherical}
\emph{Cilíndrico} $(x, y, z) = (\rho\cos\alpha, \rho\sin\alpha,
z)$: $\dd x\,\dd y\,\dd z = \rho\,\dd\rho\,\dd\alpha\,\dd z $.
\emph{Esférico} $(x, y, z) = (r\cos\theta\cos\varphi,\
r\sin\theta\cos\varphi,\ r\sin\varphi)$ ($\theta $ longitude,
$\varphi \in [-\frac\pi2, \frac\pi2]$ latitude): expandindo o $3
\times 3$ determinante ao longo da última linha,
\[
\det J = r^2\cos\varphi ,
\qquad
\dd x\,\dd y\,\dd z
= r^2\cos\varphi\;\dd r\,\dd\theta\,\dd\varphi .
\]
\omterm{pb:b2:multint:1}{Volume da bola} de raio $ R $:
\[
V = \int_0^R\!\!\int_0^{2\pi}\!\!\int_{-\pi/2}^{\pi/2}
r^2\cos\varphi\;\dd\varphi\,\dd\theta\,\dd r
= \frac{R^3}{3}\cdot 2\pi \cdot 2
= \boxed{\frac43\pi R^3} ,
\]
cumprindo finalmente a fórmula admitida nos capítulos do volume de
os livros anteriores.
\end{example}

\begin{example}[O tetraedro, duas vezes]
\label{ex:b2:multint:tetra}
O volume de $ T = \{x, y, z \geq 0,\ x + y + z \leq 1\}$, por
empilhamento: para fixo $(x, y)$ no triângulo $ x + y \leq 1$,
$ z $ atropela $\intcc0{1 - x - y}$, então
\[
V = \int_0^1\!\!\int_0^{1-x}(1 - x - y)\,\dd y\,\dd x
= \int_0^1\frac{(1 - x)^2}{2}\,\dd x = \frac16 .
\]
Ao fatiar: a seção em altura $ z $ é o triângulo $\{x, y
\geq 0,\ x + y \leq 1 - z\}$, de \omterm{def:b2:multint:domain}{área} $\frac{(1-z)^2}2$, e
$ V = \int_0^1\frac{(1-z)^2}2\,\dd z = \frac16$ novamente --- o
dois cálculos são as mesmas integrais em um diferente
ordem, que é tudo o que Fubini afirma. O valor $\frac16 =
\frac13\cdot\frac12\cdot1$ é a fórmula do cone
(\cref{ex:b2:multint:cone}) com base triangular, e o
$ n $ versão dimensional $1/n!$ é provado exatamente por isso
cortando o problema do fim de semana.
\end{example}

\begin{example}[Centróide de uma meia bola]\label{ex:b2:multint:centroid}
Para a meia bola superior $ H $ de raio $ R $ ($ z \geq 0$), o
altura do centróide é $\bar z = \frac1{V}\iiint_H z $, com $ V =
\frac23\pi R^3$. Em coordenadas esféricas ($ z =
r\sin\varphi $, $\varphi \in \intcc0{\pi/2}$):
\[
\iiint_H z
= \int_0^R r^3\,\dd r\int_0^{2\pi}\dd\theta
\int_0^{\pi/2}\sin\varphi\cos\varphi\,\dd\varphi
= \frac{R^4}4\cdot2\pi\cdot\frac12 = \frac{\pi R^4}4 ,
\]
então
\[
\bar z = \frac{\pi R^4/4}{2\pi R^3/3} = \frac{3R}8 :
\]
o ponto de equilíbrio de um hemisfério sólido fica a três oitavos de
o raio acima da face plana --- abaixo da meia altura
$ R/2$, como deveria, já que o sólido é mais gordo perto da base.
Cada cálculo centróide tem esta forma: um momento
integral, um volume, uma proporção e uma verificação de plausibilidade
contra a geometria.
\end{example}

\begin{example}[Volume ao fatiar: o cone]\label{ex:b2:multint:cone}
Um cone de base \omterm{def:b2:multint:domain}{área} $ A $ e altura $ h $ (ápice para cima, base em $ z = 0$):
a fatia na altura $ z $ é a base escalada pelo fator $(1 -
z/h)$, de \omterm{def:b2:multint:domain}{área} $ A(1 - z/h)^2$. Por isso
\[
V = \int_0^h A\Bigl(1 - \frac zh\Bigr)^2\dd z = \frac{Ah}{3} :
\]
um terço das fórmulas escolares, válidas para \emph{qualquer} básico
forma --- fatiar transforma-o na integral de um quadrado.
\end{example}

\begin{example}[Limites de integrabilidade no
avião]\label{ex:b2:multint:threshold}
Para qual $\alpha > 0$ faz $\iint_{D}\rho^{-\alpha}\,\dd
x\,\dd y $ convergem para o disco unitário perfurado $ D $ (limite
acima dos anulares $\varepsilon \leq \rho \leq 1$)? Em \omterm{pb:b2:quadratic:1}{polar}
coordenadas,
\[
\int_0^{2\pi}\!\!\int_\varepsilon^1\rho^{-\alpha}\,
\rho\,\dd\rho\,\dd\alpha
= 2\pi\int_\varepsilon^1\rho^{1-\alpha}\,\dd\rho ,
\]
que converge como $\varepsilon \to 0$ se $1 - \alpha > -1$,
ou seja\ $\alpha < 2$: em dimensão $2$ a singularidade crítica
expoente é a própria dimensão, o extra $\rho $ do
Jacobiano suavizando a singularidade por um poder. (Da mesma maneira
$\alpha < 3$ para uma singularidade de ponto no espaço, via $ r^2$.)
A contabilidade radial deste tipo é como a integrabilidade é
decidido de relance no quadro Lebesgue do Ano 3 ---
e é a razão $\iiint 1/r $ convergiu sem esforço em
\cref{exo:b2:multint:7}.
\end{example}

\begin{remark}[Armadilhas comuns]
(eu) \emph{Orientação}: um \omterm{def:b2:multint:lineint}{integral de linha} muda de sinal com o
direção da viagem, e Green--Riemann requer o limite
sentido anti-horário (domínio à esquerda); para um domínio com
buraco, o limite interno é atravessado \emph{sentido horário}. (ii)
\emph{O Jacobiano entra com um valor absoluto}: mudança de
variáveis nunca produz negativo \omterm{def:b2:multint:domain}{área}, e esquecendo
$\abs{\det}$ \omterm{def:b2:funcseq:series}{normalmente} inverte os sinais exatamente quando o mapa
inverte a orientação. (iii) \emph{O fator \omterm{pb:b2:quadratic:1}{polar} $\rho $}:
$\dd x\,\dd y = \rho\,\dd\rho\,\dd\alpha $, não
$\dd\rho\,\dd\alpha $ --- o erro mais comum em todo
capítulo; a análise dimensional detecta isso, já que
$\dd\rho\,\dd\alpha $ tem a dimensão de um \omterm{def:b2:curves:length}{comprimento}, não um
\omterm{def:b2:multint:domain}{área}. (4) \emph{Integrais duplas impróprias}: limites acima
discos crescentes e quadrados crescentes concordam aqui porque o
os integrandos são positivos (compressão); para mudança de sinal
integrandos o limite pode depender da exaustão, e não
afirmação é feita sem convergência absoluta. (v)
\emph{Domínios versus integrandos}: um integrando de produto em um
domínio que não é de produto faz \emph{não} fatorar a integral ---
a fatoração precisa de ambos, como em \cref{ex:b2:multint:polar}de
quadrado.
\end{remark}

\begin{remark}[Perspectivas dentro deste volume]
A integral de Gauss calculada aqui está silenciosamente em toda parte
os capítulos de probabilidade: a constante $\sqrt\pi $ dentro
Fórmula de Stirling (\cref{thm:b2:comparison:stirling}) é
integral deste capítulo, e através de Stirling ele fixa o
$1/\sqrt{\pi n}$ assintótica do retorno do passeio aleatório
probabilidades em \cref{ch:b2:proba}. O Integrais de Wallis de
o problema do fim de semana reaparece lá também, dirigindo do mesmo jeito
estimativas binomiais centrais. Na outra direção, o \omterm{def:b2:multint:domain}{área}
e os elementos de volume deste capítulo completam a geometria de
\cref{ch:b2:surfaces}, e a fórmula de Green recalcula o
\omterm{pb:b2:curves:1}{envelope} \omterm{def:b2:multint:domain}{áreas} de \cref{ch:b2:curves} (o \omterm{pb:b2:curves:1}{astroide}, em
\cref{exo:b2:multint:5}). Um capítulo, três serviços:
medida para geometria, constantes para probabilidade e o
disciplina de mudança de variáveis usada por ambos.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b2:multint:1}
Calcular $\int_\gamma y^2\,\dd x + x\,\dd y $ ao longo de: (a) o segmento
de $(0,0)$ para $(1,1)$; (b) o arco da parábola $ y = x^2$ de

Mostre isso $\omega = (2xy + y^3)\,\dd x + (x^2 + 3xy^2 + 1)\,\dd y $

\end{exercise}

Calcular $\iint_D (x + y)\,\dd x\,\dd y $ onde $ D $ é o domínio
limitado por $ y = x^2$ e $ y = x $ ($0 \leq x \leq 1$), em ambos

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:multint:4}
Usando coordenadas \omterm{pb:b2:quadratic:1}{polares}, calcule $\iint_D \frac{\dd x\,\dd y}{(1 +
x^2 + y^2)^2}$ sobre todo o plano (como um limite sobre os discos), e
$\iint_{D'} xy\,\dd x\,\dd y $ sobre o quarto do disco $ D' = \{x, y

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:multint:5}
Calcule o \omterm{def:b2:multint:domain}{área} encerrado pelo \omterm{pb:b2:curves:1}{astroide} $ x = \cos^3 t $, $ y =

\cref{cor:b2:multint:area}. \emph{(Linearizar $\sin^2 t\cos^2 t $.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:multint:6}
Calcule o volume do sólido limitado abaixo pelo parabolóide $ z
= x^2 + y^2$ e acima pelo avião $ z = 1$, por ambos os métodos:
empilhamento (integrar $1 - x^2 - y^2$ sobre o disco unitário, \omterm{pb:b2:quadratic:1}{polar}

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:multint:7}
(Atração gravitacional de uma bola --- teorema de Newton, especial
caso) Mostre que o volume da casca esférica $ a \leq r \leq
b $ é $\frac43\pi(b^3 - a^3)$ e calcular $\iiint_{B}
\frac{\dd x\,\dd y\,\dd z}{r}$ por cima da bola $ B $ de raio $ R $
($ r $ a distância até a origem). \emph{(Esférico
coordenadas.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:multint:8}
(Lema de Poincaré, caso em forma de estrela) Seja $ U $ ter formato de estrela com
respeito a $0$ (ou seja,\ $ M \in U \Rightarrow [0, M] \subseteq U $)
e $\omega = P\dd x + Q\dd y $ um fechado $\mathcal{C}^1$ formulário em
$ U $. Definir
\[
f(x, y) = \int_0^1 \bigl(x\,P(tx, ty) + y\,Q(tx, ty)\bigr)\dd t .
\]
Usando diferenciação sob o sinal integral
(\cref{ch:b2:integration}) e $ P_y = Q_x $, mostre que $ f_x = P $
e $ f_y = Q $: todo formulário fechado em forma de estrela \omterm{def:b2:metric:topology}{abrir} definir é
exato.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:multint:9}
(Dirichlet integral por dupla integração) Justifique e explore
\[
\int_0^\infty\!\!\int_0^\infty e^{-xy}\sin x\;\dd y\,\dd x
\quad\text{vs}\quad
\int_0^\infty\!\!\int_0^\infty e^{-xy}\sin x\;\dd x\,\dd y
\]
sobre $[0, A] \times [0, \infty)$: mostrar $\int_0^A \frac{\sin
x}{x}\dd x = \frac\pi2 - \int_0^\infty
e^{-Ay}\frac{y\sin A + \cos A}{1 + y^2}\dd y $ e recuperar
$\int_0^\infty \frac{\sin x}{x}\,\dd x = \frac\pi2$, comparando
com a prova integral de parâmetros de
\cref{ch:b2:integration}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:multint:10}
(Desigualdade isoperimétrica via Wirtinger) Deixe $\gamma $ seja um simples
fechado $\mathcal{C}^1$ curva de \omterm{def:b2:curves:length}{comprimento} $2\pi $, parametrizado por arco
\omterm{def:b2:curves:length}{comprimento} sobre $[0, 2\pi]$, encerrando \omterm{def:b2:multint:domain}{área} $ A $. Usando
\cref{cor:b2:multint:area}, Parseval e a desigualdade de Wirtinger
(exercícios de \cref{ch:b2:fourier}), provar $ A \leq \pi $, com
igualdade para o círculo. \emph{(Normalizar $\int_0^{2\pi} x(s)\dd s
= 0$; escrever $2A = \oint x\,\dd y - y\,\dd x $ e vinculado $2A \leq
\int (x^2 + y'^2)$ cuidadosamente através $2A = \int_0^{2\pi}(xy' - yx')\dd
s $ e $ x^2 + y'^2 \geq 2xy'$.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:multint:11}
(Momentos da bola) Para a bola $ B $ de raio $ R $ em
$\R^3$, calcular $\iiint_B z^2\,\dd x\,\dd y\,\dd z $ em
coordenadas esféricas e deduzir $\iiint_B (x^2 + y^2 +
z^2)\,\dd x\,\dd y\,\dd z $ por simetria. Verifique
último contra o cálculo do shell $\int_0^R r^2\cdot4\pi
r^2\,\dd r $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:multint:12}
(Fórmula do cadarço) Deixe $ K $ seja um polígono com vértices $(x_1,
y_1), \dots, (x_m, y_m)$ em sentido anti-horário (índices
moda $ m $). Deduzir de \cref{cor:b2:multint:area} que
\[
\operatorname{Area}(K)
= \frac12\sum_{i=1}^m
\bigl(x_iy_{i+1} - x_{i+1}y_i\bigr) ,
\]
e verifique a fórmula no triângulo $(0,0)$, $(1,0)$,
$(0,1)$.
\end{exercise}

\section{Problema: o volume da bola em dimensão $ n $}

\begin{problem}[{Problema de fim de semana --- $ V_n =
\pi^{n/2}/\Gamma(\frac n2 + 1)$, e a estranheza do alto
dimensões}]\label{pb:b2:multint:1}
O disco tem \omterm{def:b2:multint:domain}{área} $\pi $, o \omterm{pb:b2:multint:1}{volume da bola} $\frac43\pi $ --- e então? Este problema calcula o \omterm{pb:b2:multint:1}{volume da bola} unitária de
$\R^n $ para cada $ n $, duas vezes (por uma recursão de fatiamento conduzida por
o Integrais de Wallis, então através do $\Gamma $ função e o
Integral de Gauss de \cref{ex:b2:multint:polar}) e depois lê
fora da geometria: os volumes atingem o pico na dimensão cinco e correm
para zero, e quase toda uma bola de alta dimensão se esconde em um
casca fina perto de seu limite.\index{volume da bola}\index{Wallis
integrais} Por um \omterm{def:b2:metric:continuity}{contínuo} funcionar em uma bola de $\R^n $, o
integral é entendida como a $ n $-fold integral iterada
(cortando uma coordenada de cada vez, como no capítulo para $ n
\leq 3$); nós escrevemos $ B_n(R)$ para a bola fechada de raio $ R $
centrado em $0$, $ v_n(R)$ pelo seu volume e $ V_n =
v_n(1)$, com $ V_0 = 1$ por convenção.

\medskip
\noindent\textbf{Part I --- The slicing recursion.}
\begin{enumerate}
  \item Substituindo $ x_i = Ru_i $ em cada um dos $ n $ iterado
        integrais, mostre $ v_n(R) = V_nR^n $.
  \item Fatiar $ B_n(1)$ ao longo de sua última coordenada, mostre
        \[
        V_n = V_{n-1}\int_{-1}^{1}(1 -
        t^2)^{\frac{n-1}2}\,\dd t .
        \]
  \item Com $ t = \sin\theta $, identifique a integral como um
        Integral de Wallis: $\int_{-1}^1(1 -
        t^2)^{\frac{n-1}2}\dd t = 2W_n $, onde $ W_n =
        \int_0^{\pi/2}\cos^n\theta\,\dd\theta =
        \int_0^{\pi/2}\sin^n\theta\,\dd\theta $.
  \item Prove as duas identidades Wallis (integrar por partes;
        então telescópio $ nW_nW_{n-1}$):
        \[
        W_n = \frac{n-1}nW_{n-2} \quad (n \geq 2),
        \qquad
        W_nW_{n-1} = \frac{\pi}{2n} \quad (n \geq 1).
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The recursion solved.}
\begin{enumerate}[resume]
  \item Combine as questões 2 a 4 na recursão em duas etapas
        \[
        V_n = \frac{2\pi}{n}\,V_{n-2}
        \qquad (n \geq 2).
        \]
  \item Deduza o formulários fechados, para $ k \geq 0$:
        \[
        V_{2k} = \frac{\pi^k}{k!},
        \qquad
        V_{2k+1} = \frac{2^{k+1}\pi^k}{1\cdot3\cdot5\cdots
        (2k+1)} .
        \]
  \item Tabular $ V_1, \dots, V_7$ numericamente. Usando o
        proporção $ V_n/V_{n-2} = 2\pi/n $ e os valores de $2W_5$
        e $2W_6$, prove que a sequência $(V_n)$ aumenta
        até o seu máximo $ V_5 = \frac{8\pi^2}{15} \approx
        5.26$ e diminui a partir daí.
  \item Mostre isso $ V_n \to 0$ mais rápido do que qualquer geometria
        sequência, e que $\sum_{n\geq1} V_n $ converge: todos
        as bolas unitárias juntas têm volume total finito.
  \item Prove a identidade geradora
        \[
        \sum_{k\geq0} V_{2k}\,x^{2k} = \eu^{\pi x^2}
        \qquad (x \in \R),
        \]
        e deduzir $\sum_{k\geq0}V_{2k} = \eu^\pi \approx
        23.14$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Second route: $\Gamma $ and the
Gauss integral.}
\begin{enumerate}[resume]
  \item Mostre por Fubini (o integrando é um produto) que
        \[
        I_n = \int_{\R^n}\eu^{-\norm
        x^2}\dd x
        = \Bigl(\int_{-\infty}^{+\infty}
        \eu^{-t^2}\dd t\Bigr)^{\!n} = \pi^{n/2},
        \]
        o $ n $ Integral de Gauss -dimensional, entendida como uma
        limite sobre cubos $\intcc{-R}{R}^n $.
  \item Lembrar $\Gamma(s) = \int_0^\infty t^{s-1}\eu^{-t}\dd
        t $ (\cref{def:b2:integration:gamma}). De
        $\Gamma(s+1) = s\,\Gamma(s)$
        (\cref{thm:b2:integration:gammaprops}) e
        $\Gamma(\tfrac12) = \sqrt\pi $ (substituir $ t = u^2$
        e invocar a integral de Gauss), calcule
        \[
        \Gamma(k + 1) = k!,
        \qquad
        \Gamma\Bigl(k + \frac32\Bigr) =
        \frac{1\cdot3\cdots(2k+1)}{2^{k+1}}\,\sqrt\pi .
        \]
  \item Prove, por indução através da recursão da questão
        5, a fórmula única
        \[
        V_n = \frac{\pi^{n/2}}{\Gamma\bigl(\frac n2 +
        1\bigr)} \qquad (n \geq 1),
        \]
        e verifique se reproduz ambos formulários fechados da pergunta
        6.
  \item Mostrar $\int_0^\infty \eu^{-r^2}r^{n-1}\dd r =
        \tfrac12\Gamma\bigl(\tfrac n2\bigr)$ e deduzir o
        identidade
        \[
        I_n = n\,V_n\int_0^\infty \eu^{-r^2}\,r^{n-1}\,\dd r.
        \]
        Interprete: a massa gaussiana de $\R^n $ é
        coletados ao longo de conchas esféricas cujas
        ``$(n-1)$-dimensional \omterm{def:b2:multint:domain}{área}'' no raio $ r $ é
        $ nV_nr^{n-1}$ --- ambos os lados estão agora provados
        de forma independente, portanto a interpretação não custa nada.
  \item Definir $ s_{n-1} = nV_n $ (o \omterm{def:b2:multint:domain}{área} da esfera unitária
        $ S^{n-1}$, consistentemente com $ v_n(R) = \int_0^R
        s_{n-1}r^{n-1}\dd r $). Tabular $ s_0, \dots, s_3$ e
        verificar $ s_1 = 2\pi $, $ s_2 = 4\pi $, $ s_3 = 2\pi^2$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- High dimensions are strange.}
\begin{enumerate}[resume]
  \item Da fórmula de Stirling
        (\cref{thm:b2:comparison:stirling}) aplicado a $ k!$,
        mostrar por igual $ n = 2k $:
        \[
        V_n \sim \frac{1}{\sqrt{\pi n}}
        \Bigl(\frac{2\pi\eu}{n}\Bigr)^{n/2}
        \qquad (n \to \infty, \ n \text{ even}),
        \]
        e explique por que o mesmo decaimento supergeométrico se limitou
        estende-se a ímpar $ n $ através da recursão.
  \item A bola unitária está no cubo $\intcc{-1}1^n $ de
        volume $2^n $. Calcule a taxa de enchimento $ V_n/2^n $ para
        $ n = 2, 3, 10$, e mostre que tende a $0$: em alta
        dimensão, essencialmente todo o cubo está em sua
        cantos.
  \item Mostre que a fração de $ v_n(1)$ deitado dentro
        distância $\varepsilon $ da esfera limite é $1 -
        (1 - \varepsilon)^n \to 1$; numericamente, o que
        fração de um $100$ bola dimensional encontra-se no
        casca externa de espessura $1\%$?
  \item Prove o Wallis assintótico $ W_n \sim
        \sqrt{\dfrac{\pi}{2n}}$ \emph{(monotonicidade de
        $(W_n)$, a proporção $ W_n/W_{n-2} \to 1$, e $ W_nW_{n-1}
        = \frac\pi{2n}$)}, e o limite inferior $ W_n \geq
        \sqrt{\dfrac{\pi}{2(n+1)}}$ para todos $ n $.
  \item (Concentração em uma laje) A fração da unidade
        bola com a primeira coordenada além $\delta $ é
        $\int_\delta^1(1 - x^2)^{\frac{n-1}2}\dd x \,\big/\,
        (2W_n)$. Usando $1 - u \leq \eu^{-u}$ e a cauda
        vinculado $\int_\delta^\infty \eu^{-a x^2}\dd x \leq
        \frac{\eu^{-a\delta^2}}{2a\delta}$, mostre isso
        fração é no máximo
        \[
        \frac{\eu^{-(n-1)\delta^2/2}}{(n-1)\,\delta}
        \Big/ \sqrt{\frac{2\pi}{n+1}}
        \]
        e concluo: para $\delta = s/\sqrt{n-1}$, todos menos um
        fração $ O(\eu^{-s^2/2}/s)$ da bola está no
        laje $\abs{x_1} \leq s/\sqrt{n-1}$. Uma bola de alto
        dimensão é, estatisticamente, uma panqueca fina em cada
        direção de uma só vez.
  \item Reúna as questões 16 a 19 em um parágrafo: onde
        o volume de $ B_n(1)$ senta-se (perto do limite
        esfera, mas dentro $ O(1/\sqrt n)$ lajes de cada
        hiperplano através do centro), e por que esses dois
        afirmações não se contradizem.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Other bodies, and synthesis.}
\begin{enumerate}[resume]
  \item (Simplex) Deixe $\Delta_n = \{x \in \R^n : x_i \geq 0,\
        \sum x_i \leq 1\}$. Mostre por fatiamento e indução que
        $\operatorname{vol}(\Delta_n) = \frac1{n!}$.
  \item (Polítopo cruzado) Deduza que $ C_n = \{x :
        \sum\abs{x_i} \leq 1\}$ tem volume $\frac{2^n}{n!}$,
        e verifique o sanduíche $ C_n \subseteq B_n(1)
        \subseteq \intcc{-1}1^n $ ao nível dos volumes:
        $\frac{2^n}{n!} \leq V_n \leq 2^n $.
  \item Calcular $ V_4$ uma terceira maneira: fatiar $\R^4 = \R^2 \times
        \R^2$, integre o \omterm{def:b2:multint:domain}{área} do $(z, w)$-disco acabado
        o $(x, y)$-disco em coordenadas \omterm{pb:b2:quadratic:1}{polares} e recupere
        $ V_4 = \frac{\pi^2}2$.
  \item (Monte Carlo em apuros) Um ponto é desenhado \omterm{def:b2:funcseq:def}{uniformemente}
        no cubo $\intcc{-1}1^{20}$. Mostre que o
        probabilidade de cair na bola inscrita é
        $ V_{20}/2^{20} \approx 2.5\cdot10^{-8}$, então sobre
        quarenta milhões de sorteios são necessários antes do primeiro acerto
        é esperado: estimando $ V_n $ por amostragem de rejeição
        entra em colapso em alta dimensão (a maldição de
        dimensionalidade).
  \item Síntese. Duas derivações independentes se encontraram em $ V_n =
        \pi^{n/2}/\Gamma(\frac n2 + 1)$: liste qual teorema
        deste capítulo cada um usou (Fubini, mudança de
        variáveis, a integral \omterm{pb:b2:quadratic:1}{polar} de Gauss), e que
        entradas de uma variável (Wallis, $\Gamma$, Stirling).
        Onde o volume do Ano 3 refaz esse cálculo
        com a teoria de Lebesgue, e o que ela acrescenta?
\end{enumerate}
\end{problem}
