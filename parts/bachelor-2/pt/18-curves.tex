\chapter{Curvas}\label{ch:b2:curves}

A geometria agora se torna \omterm{def:b2:diffcalc:differential}{diferencial}. Uma curva é um ponto que se move através
espaço; o cálculo nos dá sua velocidade e aceleração, e a geometria
pergunta o que é intrínseco --- independente de quão rápido atravessamos o
trajetória. As respostas são \emph{\omterm{def:b2:curves:length}{comprimento} do arco}, que mede o
trajetória em si, e \emph{\omterm{thm:b2:curves:frenet2d}{curvatura}}, que mede como
curvas. Em dimensão $3$ um segundo invariante, \emph{torção},
mede como a curva sai de seu plano. A escrituração contábil
dispositivo para tudo isso é o movimento \emph{Quadro Frenet}.

\section{Arcos parametrizados}

\begin{definition}[Arco parametrizado]\label{def:b2:curves:arc}
UM \emph{arco parametrizado}\index{arco parametrizado} de classe
$\mathcal{C}^k $ ($ k \geq 1$) é
um mapa $\gamma \colon I \to \R^n $ de classe $\mathcal{C}^k $ em um
intervalo $ I $. Um ponto $\gamma(t)$ é \emph{regular} se
$\gamma'(t) \neq 0$, e o arco é \emph{regular} se tudo isso
pontos são. A linha através $\gamma(t)$ dirigido por $\gamma'(t)$
é o \emph{linha tangente}\index{linha tangente} em um ponto normal.
\end{definition}

\begin{definition}[Mudança de parâmetro]\label{def:b2:curves:reparam}
UM \emph{mudança de parâmetro}\index{mudança de parâmetro} de classe
$\mathcal{C}^k $ é um
$\mathcal{C}^k $ difeomorfismo $\theta \colon J \to I $ entre
intervalos ($\theta' \neq 0$ em todos os lugares). Os arcos $\gamma $ e
$\gamma \circ \theta $ são chamados \emph{equivalente}; um
\emph{arco geométrico}\index{arco geométrico} (ou curva) é um
classe de equivalência. Noções
invariante sob mudança de parâmetro --- a trajetória, o tangente
linha, \omterm{def:b2:curves:length}{comprimento} do arco, \omterm{thm:b2:curves:frenet2d}{curvatura} --- são chamados \emph{geométrico}.
\end{definition}

\begin{remark}
A trajetória por si só não determina a \omterm{def:b2:curves:reparam}{arco geométrico}: o
parametrizações $ t \mapsto (\cos t, \sin t)$ sobre $[0, 2\pi]$ e assim por diante
$[0, 4\pi]$ têm a mesma imagem, mas percorrem o círculo uma vez e
duas vezes. UM \omterm{def:b2:curves:reparam}{arco geométrico} lembra a multiplicidade e orientação
da travessia, não da sua velocidade.
\end{remark}

\begin{example}[A velocidade não muda nada
geométrico]\label{ex:b2:curves:speedinvariance}
Parametrize o círculo unitário por
\[
\gamma(t) = (\cos t^2,\ \sin t^2),
\qquad t \in \intcc0{\sqrt{2\pi}} .
\]
A velocidade $\norm{\gamma'(t)} = 2t $ cresce linearmente, mas
\[
L = \int_0^{\sqrt{2\pi}}2t\,\dd t = 2\pi ,
\]
o mesmo \omterm{def:b2:curves:length}{comprimento} como em velocidade constante --- como
\cref{thm:b2:curves:lengthinv} promessas, através do \omterm{def:b2:curves:reparam}{mudança de
parâmetro} $ t \mapsto t^2$. O \omterm{def:b2:curves:arc}{linha tangente}, o \omterm{thm:b2:curves:frenet2d}{curvatura}
calculado a partir de \cref{prop:b2:curves:kappaformula}, e cada
outras quantidades geométricas também concordam; apenas $ t = 0$
merece uma olhada, onde $\gamma'(0) = 0$ faz \emph{isso
parametrização} irregular, embora a trajetória seja uma
círculo perfeito. Declarações geométricas toleram coisas ruins
mal parametrizações: reparametrizar primeiro, concluir
segundo.
\end{example}

\begin{example}\label{ex:b2:curves:cusp}
O arco $\gamma(t) = (t^2, t^3)$ é $\mathcal{C}^\infty $ mas não
normal: $\gamma'(0) = (0, 0)$. Sua trajetória, a semicúbica
parábola $ y^2 = x^3$, tem um \emph{cúspide} na origem: suavidade
da parametrização não impede uma singularidade geométrica
onde a velocidade desaparece. É por isso que a hipótese de regularidade
$\gamma' \neq 0$ não é cosmético.
\end{example}

\section{Comprimento do arco}

\begin{definition}[Comprimento do arco]\label{def:b2:curves:length}
Deixar $\gamma \colon [a, b] \to \R^n $ seja um $\mathcal{C}^1$ arco. Isso é
\emph{comprimento}\index{comprimento} é
\[
L(\gamma) = \int_a^b \norm{\gamma'(t)}\, \dd t ,
\]
onde $\norm{\cdot}$ é o euclidiano \omterm{def:b2:nvs:norm}{norma}. O \emph{comprimento do arco
função}\index{comprimento do arco
função} com base em $ t_0$ é $ s(t) = \int_{t_0}^t
\norm{\gamma'(u)}\,\dd u $.
\end{definition}

\begin{theorem}[O comprimento é geométrico; poligonal
caracterização]\label{thm:b2:curves:lengthinv}
\leavevmode
\begin{enumerate}
\item Se $\theta \colon [c, d] \to [a, b]$ é um $\mathcal{C}^1$
\omterm{def:b2:curves:reparam}{mudança de parâmetro}, então $ L(\gamma \circ \theta) = L(\gamma)$.
\item $ L(\gamma)$ é o supremo do \omterm{def:b2:curves:length}{comprimentos} de inscrito
polígonos:
\[
L(\gamma)
= \sup\Bigl\{\, \sum_{i=1}^{m} \norm{\gamma(t_i) -
\gamma(t_{i-1})} \;:\; a = t_0 < t_1 < \dots < t_m = b \,\Bigr\}.
\]
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Pela mudança de variáveis $ t = \theta(u)$ (Ano 1
volume, válido desde $\theta $ é $\mathcal{C}^1$ monótono),
\[
\int_c^d \norm{(\gamma\circ\theta)'(u)}\,\dd u
= \int_c^d \norm{\gamma'(\theta(u))}\,\abs{\theta'(u)}\,\dd u
= \int_a^b \norm{\gamma'(t)}\,\dd t ,
\]
onde usamos $(\gamma\circ\theta)' = \theta'\cdot
(\gamma'\circ\theta)$ e, se $\theta $ está diminuindo, o sinal de
$\theta'$ é absorvido pela reversão dos limites.

\emph{2.} Para qualquer subdivisão, $\gamma(t_i) - \gamma(t_{i-1}) =
\int_{t_{i-1}}^{t_i}\gamma'(t)\,\dd t $, então pelo triângulo
desigualdade para integrais $\norm{\gamma(t_i) - \gamma(t_{i-1})}
\leq \int_{t_{i-1}}^{t_i}\norm{\gamma'}$: todo polígono é mais curto
do que $ L(\gamma)$, por isso $\sup \leq L(\gamma)$.

Para a desigualdade inversa, seja $\varepsilon > 0$. Desde $\gamma'$
é \omterm{def:b2:metric:continuity}{contínuo} no \omterm{def:b2:metric:compact}{compactar} $[a,b]$, é \omterm{def:b2:funcseq:def}{uniformemente} \omterm{def:b2:metric:continuity}{contínuo}:
existe $\delta > 0$ com $\norm{\gamma'(t) - \gamma'(u)} \leq
\varepsilon $ em qualquer momento $\abs{t - u} \leq \delta $. Pegue uma subdivisão
de passo $\leq \delta $. Em cada peça, por $ t \in [t_{i-1}, t_i]$,
\[
\gamma(t_i) - \gamma(t_{i-1})
= \int_{t_{i-1}}^{t_i}\gamma'(t)\,\dd t
= (t_i - t_{i-1})\,\gamma'(t_{i-1}) + R_i,
\qquad
\norm{R_i} \leq \varepsilon\,(t_i - t_{i-1}),
\]
desde $ R_i = \int_{t_{i-1}}^{t_i}(\gamma'(t) -
\gamma'(t_{i-1}))\,\dd t $. Por isso
\[
\norm{\gamma(t_i) - \gamma(t_{i-1})}
\geq (t_i - t_{i-1})\norm{\gamma'(t_{i-1})}
- \varepsilon (t_i - t_{i-1}) .
\]
Resumindo e comparando $\sum (t_i -
t_{i-1})\norm{\gamma'(t_{i-1})}$ com $\int_a^b\norm{\gamma'}$ (um
Soma de Riemann do \omterm{def:b2:metric:continuity}{contínuo} função $\norm{\gamma'}$, dentro de
$\varepsilon(b - a)$ da integral para $\delta $ pequeno o suficiente por
uniforme continuidade novamente), obtemos um polígono de \omterm{def:b2:curves:length}{comprimento} $\geq
L(\gamma) - 2\varepsilon(b - a)$. Deixando $\varepsilon \to 0$
comprova a afirmação.
\end{proof}

\begin{example}[Arquimedes e os inscritos
polígonos]\label{ex:b2:curves:archimedes}
Para o círculo unitário, o regular inscrito $ n $-gon tem \omterm{def:b2:curves:length}{comprimento}
$ L_n = 2n\sin\frac\pi n $, e a expansão $\sin x = x -
\frac{x^3}6 + O(x^5)$ dá
\[
L_n = 2\pi - \frac{\pi^3}{3n^2} + O\Bigl(\frac1{n^4}\Bigr) :
\]
o poligonal \omterm{def:b2:curves:length}{comprimentos} de \cref{thm:b2:curves:lengthinv}
\omterm{def:b2:integration:improper}{convergir} \emph{quadraticamente}. Numericamente: $ L_6 = 6$
(o hexágono, dando o bruto $\pi > 3$), enquanto $ L_{96} =
192\sin\frac{\pi}{96} \approx 6.28206$ contra $2\pi \approx
6.28319$ --- o erro $0.00113$ concorda com o previsto
$\pi^3/(3\cdot96^2) \approx 0.00112$. É por isso que Arquimedes,
duplicando o hexágono cinco vezes para $96$ lados, poderia colchetes
$\pi $ a três dígitos manualmente: cada duplicação divide o
erro por quatro. O supremo na poligonal
a caracterização não se atinge apenas no limite; é
alcançado \emph{rápido}, porque uma curva suave se separa de
seus acordes apenas de segunda ordem.
\end{example}

\begin{theorem}[Parametrização de comprimento de arco]\label{thm:b2:curves:arclength}
Deixar $\gamma \colon I \to \R^n $ seja regular $\mathcal{C}^k $ arco
($ k \geq 1$). O função \omterm{def:b2:curves:length}{comprimento} de arco $ s $ é um $\mathcal{C}^k $
difeomorfismo de $ I $ em um intervalo $ J $, e $\tilde\gamma =
\gamma \circ s^{-1}$ satisfaz $\norm{\tilde\gamma'} = 1$
em todos os lugares. Até a tradução do parâmetro e orientação,
isso \emph{\omterm{def:b2:curves:length}{comprimento} do arco} (ou \emph{velocidade unitária}) a parametrização é
único.
\end{theorem}

\begin{proof}
$ s'(t) = \norm{\gamma'(t)} > 0$ e $ s'$ é $\mathcal{C}^{k-1}$
(composição do $\mathcal{C}^{k-1}$ mapa $\gamma'$ com o \omterm{def:b2:nvs:norm}{norma},
suavizar longe de $0$), então $ s $ é $\mathcal{C}^k $, estritamente
aumentando, uma bijeção em $ J = s(I)$, e seu inverso é
$\mathcal{C}^k $ pelo teorema da função inversa na dimensão $1$
(Volume do ano 1). Então
\[
\tilde\gamma'(\sigma)
= \frac{\gamma'(t)}{s'(t)}
= \frac{\gamma'(t)}{\norm{\gamma'(t)}},
\qquad t = s^{-1}(\sigma),
\]
um vetor unitário. Se $\hat\gamma = \gamma\circ\theta $ é outro
parametrização de velocidade unitária, então $\abs{\theta'} = 1$, então $\theta'
= \pm 1$ constante (continuidade), ou seja\ $\theta(u) = \pm u + c $.
\end{proof}

\begin{remark}
\omterm{def:b2:curves:length}{Comprimento} do arco é o parâmetro que separa a geometria da
dinâmica. Uma trajetória pode ser percorrida com qualquer velocidade
perfil --- a física do movimento --- mas cada
questão invariante à parametrização (forma, \omterm{thm:b2:curves:frenet2d}{curvatura},
osculação) tem um relógio canônico, a distância percorrida.
É por isso que todos \omterm{thm:b2:curves:frenet2d}{curvatura} as fórmulas abaixo são \emph{definido}
em velocidade unitária e então \emph{traduzido} para arbitrário
parametrizações pela regra da cadeia: os fatores de tradução
são poderes de $ v = s'$, e rastreá-los corretamente é o
todo o conteúdo de \cref{prop:b2:curves:kappaformula}.
\end{remark}

\begin{example}[Círculo e hélice]\label{ex:b2:curves:helix}
Para o círculo $\gamma(t) = (R\cos t, R\sin t)$, $\norm{\gamma'} =
R $, então $ s = Rt $ e o \omterm{def:b2:curves:length}{comprimento} de uma volta completa é $2\pi R $. Para o
hélice $\gamma(t) = (a\cos t,\ a\sin t,\ bt)$ com $ a > 0$,
$\norm{\gamma'(t)} = \sqrt{a^2 + b^2}$ é constante: a hélice é
percorrido com velocidade constante e $ s = t\sqrt{a^2 + b^2}$.
\end{example}

\begin{example}[Comprimento do arco em polar
coordenadas]\label{ex:b2:curves:polarlength}
Uma curva \omterm{pb:b2:quadratic:1}{polar} $ r = r(\theta)$ é o arco $\gamma(\theta) =
(r\cos\theta,\ r\sin\theta)$, com
\[
\gamma'(\theta) = (r'\cos\theta - r\sin\theta,\
r'\sin\theta + r\cos\theta),
\qquad
\norm{\gamma'(\theta)}^2 = r'^2 + r^2
\]
(os termos cruzados são cancelados): o \omterm{pb:b2:quadratic:1}{polar} \omterm{def:b2:curves:length}{comprimento} elemento é
$\sqrt{r^2 + r'^2}\,\dd\theta $. Para o cardióide $ r = 1 +
\cos\theta $:
\[
r^2 + r'^2 = (1 + \cos\theta)^2 + \sin^2\theta = 2 +
2\cos\theta = 4\cos^2\tfrac\theta2,
\]
e assim por diante $\intcc{-\pi}{\pi}$ o cosseno do meio ângulo é
não negativo, então
\[
L = \int_{-\pi}^{\pi}2\cos\tfrac\theta2\,\dd\theta
= \Bigl[4\sin\tfrac\theta2\Bigr]_{-\pi}^{\pi}
= 4 - (-4) = 8 :
\]
como o arco ciclóide de \cref{exo:b2:curves:1}, uma curva
construído a partir de círculos tem um racional \omterm{def:b2:curves:length}{comprimento}, sem $\pi $
em qualquer lugar. A fatoração de meio ângulo é o truque padrão
para \omterm{def:b2:curves:length}{comprimentos} de curvas geradas em círculo; quando falha (o
elipse), o \omterm{def:b2:curves:length}{comprimento} é uma função genuinamente nova --- uma
integral elíptica, além das formas fechadas elementares.
\end{example}

\section{Curvatura no avião}

Ao longo desta seção, os arcos são $\mathcal{C}^2$ e regular em
o plano euclidiano orientado. Parametrizamos por \omterm{def:b2:curves:length}{comprimento} do arco e
escrever $ T(s) = \tilde\gamma'(s)$ para a tangente unitária, e $ N(s)$
para o vetor unitário diretamente ortogonal a $ T(s)$ (rotação de
$ T $ por $+\pi/2$).

\begin{theorem}[Fórmulas de avião Frenet]\label{thm:b2:curves:frenet2d}
Deixar $\tilde\gamma $ seja uma velocidade unitária $\mathcal{C}^2$ arco no
plano orientado. Existe um \omterm{def:b2:metric:continuity}{contínuo} função $\kappa $, o
(algébrico) \emph{curvatura}\index{curvatura}, tal que
\[
T'(s) = \kappa(s)\, N(s),
\qquad
N'(s) = -\kappa(s)\, T(s).
\]
\end{theorem}

\begin{proof}
Desde $\norm{T(s)}^2 = 1$ para todos $ s $, diferenciando o escalar
produto dá $2\langle T'(s), T(s)\rangle = 0$: $ T'(s)$ é ortogonal a
$ T(s)$, portanto, colinear com $ N(s)$ (dimensão $2$); escrever $ T'(s)
= \kappa(s) N(s)$ com $\kappa(s) = \langle T'(s), N(s)\rangle $,
\omterm{def:b2:metric:continuity}{contínuo}. Da mesma maneira $ N' \perp N $, então $ N' = \lambda T $; e
diferenciando $\langle T, N\rangle = 0$ dá $\langle T', N\rangle +
\langle T, N'\rangle = \kappa + \lambda = 0$.
\end{proof}

\begin{definition}\label{def:b2:curves:curvature}
Quando $\kappa(s) \neq 0$, o \emph{raio de
curvatura}\index{raio de
curvatura} é $ R(s) =
1/\abs{\kappa(s)}$ e o \emph{centro de curvatura}\index{centro de curvatura} é
$\tilde\gamma(s) + \frac{1}{\kappa(s)} N(s)$; o círculo com isso
centro e raio $ R(s)$ é o \emph{círculo osculante}\index{círculo osculante}, o
melhor aproximação circular da curva em $\tilde\gamma(s)$.
\end{definition}

\begin{example}[O círculo osculante do
exponencial]\label{ex:b2:curves:oscexp}
Para $ y = \eu^x $ no ponto $(0, 1)$: $ f'(0) = f''(0) = 1$,
então, pela fórmula gráfica abaixo,
\[
\kappa(0) = \frac{1}{(1 + 1)^{3/2}} = \frac1{2\sqrt2},
\qquad R = 2\sqrt2 .
\]
A tangente unitária é $ T = \frac{(1, 1)}{\sqrt2}$, o direto
normais $ N = \frac{(-1, 1)}{\sqrt2}$, e o \omterm{def:b2:curves:curvature}{centro de
curvatura} é
\[
(0, 1) + 2\sqrt2\cdot\frac{(-1, 1)}{\sqrt2} = (-2,\ 3) :
\]
o \omterm{def:b2:curves:curvature}{círculo osculante} tem equação $(x + 2)^2 + (y - 3)^2 =
8$. Como uma verificação da afirmação de “melhor aproximação circular”:
resolvendo a equação do círculo para $ y $ aproximar $(0,1)$ e
expansão dá $ y = 1 + x + \frac{x^2}2 + O(x^3)$ ---
exatamente a expansão de Taylor de segunda ordem de $\eu^x $. O
\omterm{def:b2:curves:curvature}{círculo osculante} corresponde ao valor, inclinação \emph{e} segundo
derivado; um círculo tangente comum corresponderia apenas ao
dois primeiros.
\end{example}

\begin{example}[A evolução de um círculo é a sua
centro]\label{ex:b2:curves:circleevolute}
Para o círculo de raio $ R $ percorrido no sentido anti-horário,
$\kappa = 1/R $ e $ N $ aponta para o centro, então o
\omterm{def:b2:curves:curvature}{centro de curvatura} $\tilde\gamma + \frac1\kappa N $ é o
centro do círculo, para cada $ s $: o \omterm{def:b2:curves:curvature}{círculo osculante}
de um círculo é o próprio círculo, e o lugar geométrico dos centros
de \omterm{thm:b2:curves:frenet2d}{curvatura} desmorona até certo ponto. Este caso degenerado
calibra \cref{exo:b2:curves:6}: aí está a evolução
velocidade é $-\frac{\kappa'}{\kappa^2}N $, que desaparece
de forma idêntica precisamente quando $\kappa $ é constante.
\end{example}

\begin{proposition}[Curvatura de forma arbitrária
parametrização]\label{prop:b2:curves:kappaformula}
Para um regular $\mathcal{C}^2$ arco plano $\gamma(t) = (x(t), y(t))$,
\[
\kappa(t)
= \frac{x'(t)\,y''(t) - y'(t)\,x''(t)}
       {\bigl(x'(t)^2 + y'(t)^2\bigr)^{3/2}} ,
\]
em particular $\kappa = \dfrac{y''}{(1 + y'^2)^{3/2}}$ para um gráfico
$ y = f(x)$.
\end{proposition}

\begin{proof}
Escrever $ v(t) = \norm{\gamma'(t)} = s'(t)$, então $\gamma' = vT $
(compondo os dados de velocidade unitária com $ s $). Diferenciando,
\[
\gamma'' = v'T + v\,T'\cdot s' = v'T + v^2\kappa N .
\]
Agora pegue o determinante (na base orientada canônica) de
$(\gamma', \gamma'')$: desde $\det(T, T) = 0$ e $\det(T, N) = 1$,
\[
\det(\gamma', \gamma'') = \det(vT,\ v'T + v^2\kappa N)
= v^3\kappa .
\]
O lado esquerdo é $ x'y'' - y'x''$, e $ v^3 = (x'^2 +
y'^2)^{3/2}$. O caso do gráfico é a parametrização $ t \mapsto (t,
f(t))$.
\end{proof}

\begin{example}[Círculo, linha, parábola]\label{ex:b2:curves:kappaexamples}
Uma linha tem $\kappa = 0$ (e inversamente: $ T' = 0$ significa $ T $
constante, então $\tilde\gamma(s) = \tilde\gamma(0) + sT $, uma linha). O
círculo de raio $ R $ percorrido no sentido anti-horário tem $\kappa = 1/R $:
com $\gamma(t) = (R\cos t, R\sin t)$, a fórmula dá $\kappa =
R^2/R^3$. Para a parábola $ y = x^2/2$: $\kappa(x) = 1/(1 +
x^2)^{3/2}$, máximo no vértice --- a parábola é mais nítida
dobrado onde ele se vira.
\end{example}

\begin{remark}[Armadilhas comuns em torno da curvatura]
(i) O algébrico \omterm{thm:b2:curves:frenet2d}{curvatura} de um arco plano muda de sinal quando
a orientação do arco ou do plano é invertida: apenas
$\abs\kappa $ e $ R = 1/\abs\kappa $ são puramente geométricos. Um
círculo percorrido no sentido horário tem $\kappa = -1/R $. (ii) O
fórmula gráfica $\kappa = f''/(1 + f'^2)^{3/2}$ silenciosamente
escolhe a parametrização por $ x $; aplicando-o a uma curva
que não é um gráfico próximo ao ponto (tangente vertical) é o
erro clássico. (iii) Em um ponto onde $\gamma' = 0$
nada está definido --- nem $ T $ nem $\kappa $ --- e o
trajetória pode realmente quebrar
(\cref{ex:b2:curves:cusp}); verifique sempre a regularidade antes
diferenciando a tangente unitária. (iv) No espaço, $\kappa =
\norm{T'} \geq 0$ por convenção: não há sinal para chegar
errado, mas também nenhum sinal para explorar --- tipo de inflexão
a informação passa para o torção. (v) Finalmente, $\kappa $ é
um derivado \emph{em relação a \omterm{def:b2:curves:length}{comprimento} do arco}: para um
parametrização de velocidade não unitária, esquecendo o fator $ v^3$
em \cref{prop:b2:curves:kappaformula} é o mais frequente
erro na prática.
\end{remark}

\begin{theorem}[A curvatura determina a curva]\label{thm:b2:curves:fundamental}
Deixar $\kappa \colon J \to \R $ ser \omterm{def:b2:metric:continuity}{contínuo}. Existe um
velocidade unitária $\mathcal{C}^2$ arco no plano com \omterm{thm:b2:curves:frenet2d}{curvatura}
$\kappa $, e é único até uma isometria direta (rotação
seguido de tradução).
\end{theorem}

\begin{proof}
\emph{Existência.} Consertar $ s_0 \in J $ e definir $\varphi(s) =
\int_{s_0}^s \kappa(u)\,\dd u $, então
\[
\tilde\gamma(s) = \Bigl(\int_{s_0}^s \cos\varphi(u)\,\dd u,\
\int_{s_0}^s \sin\varphi(u)\,\dd u\Bigr).
\]
Então $ T(s) = (\cos\varphi(s), \sin\varphi(s))$ é um vetor unitário,
$ N(s) = (-\sin\varphi, \cos\varphi)$, e
\[
T'(s) = \varphi'(s)\,(-\sin\varphi, \cos\varphi)
= \kappa(s)\,N(s) :
\]
o arco tem velocidade unitária com \omterm{thm:b2:curves:frenet2d}{curvatura} $\kappa $.

\emph{Singularidade.} Deixar $\gamma_1, \gamma_2$ ser arcos de velocidade unitária com
o mesmo \omterm{thm:b2:curves:frenet2d}{curvatura}. Cada tangente unitária eleva-se a um ângulo
função, por uma construção explícita: view $ T_j $ como o
número complexo $ z_j = a_j + \iu b_j $ de módulo $1$, escolha
$\varphi_j(0)$ com $ z_j(0) = \eu^{\iu\varphi_j(0)}$ e definir
\[
\varphi_j(s) = \varphi_j(0) + \int_0^s\det\bigl(T_j,
T_j'\bigr)(u)\,\dd u .
\]
De $\abs{z_j} = 1$: $\operatorname{Re}(\conj{z_j}\,z_j') =
0$, então $\conj{z_j}\,z_j' = \iu\det(T_j, T_j') =
\iu\,\varphi_j'$, ou seja\ $ z_j' = \iu\varphi_j'z_j $; então
\[
\bigl(z_j\,\eu^{-\iu\varphi_j}\bigr)'
= \eu^{-\iu\varphi_j}\bigl(z_j' - \iu\varphi_j'z_j\bigr) = 0,
\]
então $ z_j = \eu^{\iu\varphi_j}$ por todo: $ T_j =
(\cos\varphi_j, \sin\varphi_j)$ com $\varphi_j $ de classe
$\mathcal C^1$. Além disso $\det(T_j, T_j') = \det(T_j,
\kappa N_j) = \kappa $, então $\varphi_j' = \kappa $. Por isso $\varphi_2 = \varphi_1 + c $ por um
constante $ c $: $ T_2$ é $ T_1$ girado pelo ângulo fixo $ c $, então
integrando, $\gamma_2 = \rho(\gamma_1) + w $ onde $\rho $ é o
rotação do ângulo $ c $ e $ w $ um vetor constante.
\end{proof}

\begin{remark}
Este é o protótipo unidimensional de um \emph{fundamental
teorema da geometria}: um \omterm{def:b2:metric:complete}{completo} conjunto de invariantes locais (aqui, um
função) classifica o objeto até movimento rígido. O
a versão tridimensional abaixo precisa de dois invariantes.
\end{remark}

\begin{example}[Curvatura constante significa
círculo]\label{ex:b2:curves:constantkappa}
Pegar $\kappa \equiv \kappa_0 > 0$ na fórmula de existência:
$\varphi(s) = \kappa_0 s $ e
\[
\tilde\gamma(s) = \Bigl(\frac{\sin\kappa_0s}{\kappa_0},\
\frac{1 - \cos\kappa_0s}{\kappa_0}\Bigr) :
\]
o círculo de raio $1/\kappa_0$ centrado em $(0,
1/\kappa_0)$, percorrido em velocidade unitária. Pela singularidade
metade do teorema, \emph{todo} arco de velocidade unitária constante
\omterm{thm:b2:curves:frenet2d}{curvatura} $\kappa_0$ é um pedaço de um círculo de raio
$1/\kappa_0$ (ou uma linha se $\kappa_0 = 0$) --- o inverso
da computação em \cref{ex:b2:curves:kappaexamples}e
o caso do avião \cref{exo:b2:curves:9}.
\end{example}

\begin{example}[Reconstruindo uma curva a partir de seu
curvatura]\label{ex:b2:curves:reconstruction}
Qual curva de velocidade unitária tem \omterm{def:b2:curves:curvature}{raio de curvatura} $ R(s) = 1 +
s^2$? Seguindo a prova de existência com $\kappa(s) =
\frac1{1+s^2}$ e $ s_0 = 0$: $\varphi(s) = \arctan s $, então
\[
T(s) = (\cos\arctan s,\ \sin\arctan s)
= \Bigl(\frac{1}{\sqrt{1+s^2}},\
\frac{s}{\sqrt{1+s^2}}\Bigr),
\]
e integrando,
\[
\tilde\gamma(s) = \Bigl(\ln\bigl(s + \sqrt{1 + s^2}\bigr),\
\sqrt{1 + s^2} - 1\Bigr).
\]
Contexto $ x = \ln(s + \sqrt{1+s^2})$, ou seja\ $ s = \sinh x $, o
segunda coordenada é $\cosh x - 1$: a curva é a
\emph{catenária} $ y = \cosh x - 1$. Isso fecha o \omterm{def:b2:structures:sn}{ciclo} com
\cref{exo:b2:curves:3}, onde calculamos $ R = \cosh^2 x = 1 +
\sinh^2 x = 1 + s^2$ diretamente: o teorema fundamental
garante que a catenária é a \emph{apenas} curva com isso
\omterm{thm:b2:curves:frenet2d}{curvatura} perfil, até uma isometria direta.
\end{example}

\begin{remark}[Onde a curvatura é usada a seguir]
A decomposição $\gamma'' = v'T + v^2\kappa N $ obtido em
a prova de \cref{prop:b2:curves:kappaformula} é o
cinemática de cada movimento curvo: tangencial versus
aceleração centrípeta. \omterm{thm:b2:curves:frenet2d}{Curvatura} retornos para superfícies
(\cref{ch:b2:surfaces}) através do \omterm{thm:b2:curves:frenet2d}{curvatura} de curvas desenhadas
sobre eles, e o \omterm{pb:b2:curves:1}{envelope} cálculo do fim de semana deste capítulo
problema --- evolui, cáustico --- é a óptica geométrica de
frentes de onda. O volume do Ano 3 assume o ponto intrínseco de
visualizar novamente para subvariedades de $\R^n $.
\end{remark}

\section{Quadro Frenet no espaço}

Agora deixe $\tilde\gamma \colon J \to \R^3$ seja uma velocidade unitária
$\mathcal{C}^3$ arco que é \emph{birregular}: $ T'(s) \neq 0$ para
tudo $ s $. Então $\kappa(s) = \norm{T'(s)} > 0$ define o
\emph{\omterm{thm:b2:curves:frenet2d}{curvatura}} (sem sinal no espaço: não há normal preferido
orientação), e definimos:
\[
N(s) = \frac{T'(s)}{\kappa(s)}
\quad\text{(principal normal)},
\qquad
B(s) = T(s) \wedge N(s)
\quad\text{(binormal)} ,
\]
para que $(T, N, B)$ é um referencial ortonormal direto, o
\emph{Quadro Frenet}. O avião passou $\tilde\gamma(s)$ abrangido por
$ T, N $ é o \emph{plano osculante}.

\begin{theorem}[Fórmulas Frenet no espaço]\label{thm:b2:curves:frenet3d}
Existe um \omterm{def:b2:metric:continuity}{contínuo} função $\tau $, o \emph{tortura}\index{tortura}, com
\[
T' = \kappa N, \qquad
N' = -\kappa T + \tau B, \qquad
B' = -\tau N .
\]
\end{theorem}

\begin{proof}
A primeira fórmula é a definição de $ N $. Cada um dos vetores
$ T, N, B $ tem constante \omterm{def:b2:nvs:norm}{norma} $1$ e eles são ortogonais aos pares;
diferenciando as seis relações $\langle X, Y\rangle = \delta_{XY}$
mostra a matriz de $(T', N', B')$ na base $(T, N, B)$ é
antisimétrico: de fato $\langle X', Y\rangle + \langle X, Y'\rangle = 0$ e
$\langle X', X\rangle = 0$. Isso é $(N, T)$ a entrada é $\langle N', T\rangle =
-\langle N, T'\rangle = -\kappa $, e seu $(T, B)$-entrada de coluna
$\langle T', B\rangle = \kappa\langle N, B\rangle = 0$. Nomeando o restante
entrada gratuita $\tau = \langle N', B\rangle $ dá exatamente os três exibidos
fórmulas: antisimetria preenche $\langle B', N\rangle = -\tau $ e
$\langle B', T\rangle = 0$. Continuidade de $\tau = \langle N', B\rangle $ é claro
desde $ N'$ e $ B $ são \omterm{def:b2:metric:continuity}{contínuo} ($\tilde\gamma $ é
$\mathcal{C}^3$, então $ N = T'/\kappa $ é $\mathcal{C}^1$).
\end{proof}

\begin{example}[O vetor Darboux]\label{ex:b2:curves:darboux}
As três fórmulas de Frenet são compactadas em uma. Definir $\omega(s) =
\tau\,T + \kappa\,B $ (o \emph{Vetor Darboux}). Usando $ B
\wedge T = N $, $ T \wedge N = B $, $ N \wedge B = T $:
\[
\omega \wedge T = \kappa\,N = T', \qquad
\omega \wedge N = \tau\,B - \kappa\,T = N', \qquad
\omega \wedge B = -\tau\,N = B' :
\]
cada vetor de quadro evolui por $ X' = \omega \wedge X $, o
assinatura cinemática de um \emph{rotação instantânea} com
vetor de velocidade angular $\omega $. O quadro gira a uma taxa
$\norm\omega = \sqrt{\kappa^2 + \tau^2}$ sobre a mudança
eixo $\omega $; \omterm{thm:b2:curves:frenet2d}{curvatura} é a componente do spin sobre
o binormal, torção a componente em relação à tangente. Para
a hélice, $\omega $ é um vetor constante ao longo do cilindro
eixo --- é exatamente por isso que a estrutura da hélice entra em precessão
\omterm{def:b2:funcseq:def}{uniformemente}. A antisimetria da matriz Frenet, explorada em
\cref{exo:b2:curves:7}, é a forma matricial deste único
fato geométrico.
\end{example}

\begin{proposition}[Torção mede planaridade]\label{prop:b2:curves:torsion}
Um arco birregular está contido num plano se e somente se $\tau
\equiv 0$; nesse caso, o plano é o osculante (constante)
avião.
\end{proposition}

\begin{proof}
Se $\tau \equiv 0$, então $ B' = 0$, então $ B $ é uma unidade constante
vetor $ B_0$, e
\[
\frac{\dd}{\dd s}\langle \tilde\gamma(s), B_0\rangle
= \langle T(s), B_0\rangle = 0 :
\]
$\langle \tilde\gamma, B_0\rangle $ é constante, então o arco está em um plano
ortogonal a $ B_0$. Por outro lado, se o arco estiver em um plano $ P $,
então $ T $ e $ T'$ (por isso $ N $) são paralelos à direção de
$ P $ para todos $ s $; então $ B = T \wedge N $ é uma das duas unidades
normais de $ P $, e sendo \omterm{def:b2:metric:continuity}{contínuo} é constante; então $0 = B'
= -\tau N $ com $ N \neq 0$ forças $\tau \equiv 0$.
\end{proof}

\begin{example}[Um círculo inclinado tem torção zero]
O arco $\gamma(t) = \bigl(\cos t,\ \tfrac{\sin t}{\sqrt2},\
\tfrac{\sin t}{\sqrt2}\bigr)$ está no avião $ y = z $ e
é o círculo unitário desse plano (verifique: $\norm{\gamma(t)} =
1$ e a base ortonormal do plano $(1,0,0)$,
$(0, \tfrac1{\sqrt2}, \tfrac1{\sqrt2})$ exibe o padrão
parametrização). Sem qualquer cálculo de Frenet,
\cref{prop:b2:curves:torsion} prevê $\tau \equiv 0$ e
o binormal fixo deve ser a unidade normal do plano $\pm(0,
\tfrac1{\sqrt2}, -\tfrac1{\sqrt2})$. Torção não mede
estar “inclinado no espaço”; mede \emph{saindo} um
avião. Somente o diferente de zero da hélice $ b $ abaixo produz genuíno
torção.
\end{example}

\begin{example}[A hélice]\label{ex:b2:curves:helixfrenet}
Para a hélice $\gamma(t) = (a\cos t, a\sin t, bt)$, $ a > 0$, nós
computado $ s = ct $ com $ c = \sqrt{a^2 + b^2}$. Então
\[
T = \frac1c(-a\sin t,\ a\cos t,\ b),
\qquad
T' \cdot \frac{\dd t}{\dd s}
= \frac{1}{c^2}(-a\cos t, -a\sin t, 0),
\]
então $\kappa = a/c^2 = a/(a^2 + b^2)$ e $ N = (-\cos t, -\sin t,
0)$: a normal principal aponta horizontalmente em direção ao eixo.
Próximo $ B = T \wedge N = \frac1c(b\sin t, -b\cos t, a)$, e $ B'
\frac{\dd t}{\dd s} = \frac{b}{c^2}(\cos t, \sin t, 0) = -\tau N $
dá
\[
\boxed{\ \kappa = \frac{a}{a^2 + b^2}, \qquad
\tau = \frac{b}{a^2 + b^2}. \ }
\]
Ambos os invariantes são constantes --- e pode-se mostrar, inversamente, que
as únicas curvas birregulares com constante $\kappa > 0$ e constante
$\tau $ são hélices (círculos quando $\tau = 0$). Observe os sinais: $ b >
0$ dá uma hélice destra com positivo torção.
\end{example}

\begin{example}[Curvatura e torção sem comprimento de arco; o
cúbico torcido]\label{ex:b2:curves:twistedcubic}
Reparametrizando por \omterm{def:b2:curves:length}{comprimento} do arco geralmente é impossível em ambientes fechados
forma, então os invariantes devem ser extraídos do bruto
derivados. Escrever $ v = \norm{\gamma'} = s'$; então $\gamma' =
vT $ e, como na prova de
\cref{prop:b2:curves:kappaformula},
\[
\gamma'' = v'T + v^2\kappa N,
\qquad
\gamma' \wedge \gamma'' = v^3\kappa\,(T \wedge N) =
v^3\kappa\,B .
\]
Tirando \omterm{def:b2:nvs:norm}{normas} ($\kappa \geq 0$ no espaço):
\[
\kappa = \frac{\norm{\gamma' \wedge \gamma''}}{v^3} .
\]
Diferenciando $\gamma''$ mais uma vez e convertendo $ N' =
v(-\kappa T + \tau B)$ (regra da cadeia através $ s $), a única
$ B $-component vem do último termo:
\[
\gamma''' = \bigl(v'' - v^3\kappa^2\bigr)T +
\bigl(v'v\kappa + (v^2\kappa)'\bigr)N + v^3\kappa\tau\,B,
\]
para que, emparelhando com $\gamma' \wedge \gamma'' = v^3\kappa B $,
\[
\det(\gamma', \gamma'', \gamma''') = \langle\gamma' \wedge
\gamma'',\ \gamma'''\rangle = v^6\kappa^2\tau,
\qquad\text{i.e.}\qquad
\tau = \frac{\det(\gamma', \gamma'',
\gamma''')}{\norm{\gamma' \wedge \gamma''}^2} .
\]
Aplicação ao \emph{cúbico torcido} $\gamma(t) = (t,\
t^2,\ t^3)$ no $ t = 0$: $\gamma' = (1, 0, 0)$, $\gamma'' =
(0, 2, 0)$, $\gamma''' = (0, 0, 6)$, então $ v = 1$,
\[
\gamma' \wedge \gamma'' = (0, 0, 2), \qquad \kappa(0) = 2,
\qquad
\det(\gamma', \gamma'', \gamma''') = 12, \qquad \tau(0) =
\frac{12}{4} = 3 .
\]
Visão final: ambas as fórmulas são proporções nas quais a velocidade
$ v $ cancela no grau exato necessário --- $\kappa $ escalas
como uma segunda derivada por unidade \omterm{def:b2:curves:length}{comprimento}, $\tau $ como o
volume misto de três derivadas por área quadrada --- que
é \emph{por que} eles são geométricos enquanto $\gamma''$ em si é
não.
\end{example}

\section{Estudo local: posição em relação à tangente}

\begin{proposition}[Formato local regular
ponto]\label{prop:b2:curves:local}
Deixar $\gamma $ ser um arco plano de classe $\mathcal{C}^k $ no $ t_0$, com
$ p $ o menor índice com $\gamma^{(p)}(t_0) \neq 0$ e $ q $ o
menor índice $> p $ com $\gamma^{(q)}(t_0)$ não colinear com
$\gamma^{(p)}(t_0)$ (assumindo que ambos existem, $ q \leq k $). Na base
$(u, v) = (\gamma^{(p)}(t_0), \gamma^{(q)}(t_0))$ centrado em
$\gamma(t_0)$, Taylor--Young dá coordenadas
\[
X(t) \sim \frac{(t - t_0)^p}{p!},
\qquad
Y(t) \sim \frac{(t - t_0)^q}{q!} .
\]
O quadro local depende apenas das paridades de $ p $ e $ q $:
{\small
\begin{center}
\begin{tabular}{c c p{5.5cm}}
$ p $ ímpar, $ q $ par & \emph{ponto comum} & a curva não cruza a tangente, fica de um lado\\
$ p $ ímpar, $ q $ ímpar & \emph{ponto de inflexão} & a curva cruza a tangente\\
$ p $ par, $ q $ ímpar & \emph{cúspide do 1\textsuperscript{o} tipo} & ramos em lados opostos da tangente\\
$ p $ par, $ q $ par & \emph{cúspide do 2\textsuperscript{o} tipo} & ramos do mesmo lado da tangente\\
\end{tabular}
\end{center}
}
\end{proposition}

\begin{proof}
Taylor - Young na ordem $ q $ (a função $\gamma $ é
$\mathcal{C}^q $ aproximar $ t_0$):
\[
\gamma(t) - \gamma(t_0)
= \sum_{j=p}^{q} \frac{(t-t_0)^j}{j!}\,\gamma^{(j)}(t_0)
+ o\bigl((t-t_0)^q\bigr).
\]
Pela escolha de $ p $ e $ q $, cada $\gamma^{(j)}(t_0)$ com $ p
\leq j < q $ é colinear com $ u $; coletando componentes no
base $(u, v)$: $ X(t) = \frac{(t-t_0)^p}{p!}(1 + o(1))$ e
$ Y(t) = \frac{(t-t_0)^q}{q!}(1 + o(1))$. A tabela de sinais de $ X $ e
$ Y $ para $ t \gtrless t_0$ --- governado exatamente pelas paridades ---
dá as quatro imagens: por exemplo, se $ p $ é par, $ X > 0$ ligado
ambos os lados (ambos os ramos saem na direção $+u $: uma cúspide),
e o lado do \omterm{def:b2:curves:arc}{linha tangente} ($\operatorname{sign} Y $) vira
com $ q $ chance.
\end{proof}

\begin{example}
Para $\gamma(t) = (t^2, t^3)$ no $ t_0 = 0$
(\cref{ex:b2:curves:cusp}): $\gamma'' (0)= (2, 0)$, $\gamma'''(0) =
(0, 6)$, então $ p = 2$, $ q = 3$: uma cúspide do primeiro tipo, a
imagem familiar da parábola semicúbica. Para $\gamma(t) = (t,
t^3)$ no $0$: $ p = 1$, $ q = 3$: inflexão --- as cruzes cúbicas
sua tangente.
\end{example}

\begin{remark}[Método: execução do estudo local]
Na prática, a classificação é uma rotina de quatro etapas.
\emph{Um}, diferencie em $ t_0$ até o primeiro diferente de zero
derivada aparece: seu índice é $ p $, seu valor é o vetor
$ u $. \emph{Dois}, continue diferenciando até que uma derivada
não colinear com $ u $ aparece: índice $ q $, vetor $ v $.
\emph{Três}, leia as paridades $(p, q)$ na mesa.
\emph{Quatro}, desenhe: a curva sai junto $+u $ se $ p $ é estranho
(junto $ u $ então de volta $ u $ se $ p $ é par), do lado
de $ v $ ditado pelo sinal de $ Y $. Dois cuidados. O quadro
$(u, v)$ é geralmente \emph{não} ortonormal --- a mesa
descreve posições relativas ao \omterm{def:b2:curves:arc}{linha tangente}, não ângulos
ou distâncias, então não leia \omterm{thm:b2:curves:frenet2d}{curvatura} fora da imagem.
E derivadas intermediárias colineares com $ u $ são permitidos
entre fileiras $ p $ e $ q $ (eles apenas mudam a expansão de
$ X $); o que deve \emph{não} acontecer é parar no primeiro
derivada diferente de zero e adivinhação $ q = p + 1$: para $\gamma(t)
= (t^2, t^4 + t^5)$ o palpite ingênuo $ q = 3$ está errado, como
$\gamma^{(3)}(0)$ ainda é colinear com $\gamma''(0)$ ---
isso é precisamente \cref{exo:b2:curves:5}.
\end{remark}

\begin{exercise}[$\star\star $]\label{exo:b2:curves:4}

\in \R $.
Mostre que o ângulo entre $\gamma(t)$ e $\gamma'(t)$ é
constante, calcule o \omterm{def:b2:curves:length}{comprimento} do arco de $\gamma $ sobre $(-\infty, 0]$
(finito!), e o \omterm{thm:b2:curves:frenet2d}{curvatura}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:curves:5}
Determinar $ p $, $ q $ e a forma local (comum, inflexão,
cúspide) de $\gamma(t) = (t^2,\ t^4 + t^5)$ no $ t = 0$, e de
$\gamma(t) = (t^3,\ t^4)$ no $ t = 0$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:curves:6}
Deixar $\gamma $ ser um arco plano de velocidade unitária com $\kappa(s) > 0$ para
tudo $ s $, e deixe $ c(s) = \gamma(s) + \frac{1}{\kappa(s)}N(s)$ ser
o \omterm{def:b2:curves:curvature}{centro de curvatura} (a curva $ c $ é o \emph{evoluir}).
Supondo $\kappa $ é $\mathcal{C}^1$, mostre que $ c'(s) =
-\frac{\kappa'(s)}{\kappa(s)^2}N(s)$: a evolução é tangente a
as linhas normais de $\gamma $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:curves:7}
Deixar $ A(s)$ seja um \omterm{def:b2:metric:continuity}{contínuo} família de antisimétricos $3 \times 3$
matrizes e $ F' = F A $ uma solução matricial com $ F(s_0)$
ortogonal. Mostre isso $ F(s)$ é ortogonal para todos $ s $.
\emph{(Diferenciar $ G = F F^{\mathsf T}$ e usar exclusividade em
\omterm{def:b2:metric:continuity}{Cauchy-Lipschitz}.)} Explique a relevância para o sistema Frenet.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:curves:8}
(Total \omterm{thm:b2:curves:frenet2d}{curvatura} de uma curva convexa fechada) Deixe $\tilde\gamma $ seja um
velocidade unitária $\mathcal{C}^2$ arco plano fechado de \omterm{def:b2:curves:length}{comprimento} $ L $
(então $\tilde\gamma(s + L) = \tilde\gamma(s)$), percorrido uma vez
sentido anti-horário. Usando a função ângulo $\varphi $ com $ T =
(\cos\varphi, \sin\varphi)$ de
\cref{thm:b2:curves:fundamental}, explique por que $\varphi(L) -
\varphi(0)$ é um múltiplo de $2\pi $, e mostre que
$\int_0^L \kappa(s)\,\dd s = \varphi(L) - \varphi(0)$. (Para um
círculo de raio $ R $: $\int \kappa = \frac1R \cdot 2\pi R = 2\pi $.
O teorema da viragem de tangentes afirma o valor $2\pi $ para
cada curva fechada simples; você não é solicitado a provar isso.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:curves:9}
Mostre que uma curva espacial birregular com constante $\kappa > 0$ e
$\tau = 0$ é (um arco de) um círculo de raio $1/\kappa $.
\emph{(Usar \cref{prop:b2:curves:torsion}, então mostre o centro
$\gamma + \frac1\kappa N $ é constante.)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:curves:10}
Calcule o \omterm{def:b2:curves:length}{comprimento} do arco da parábola $ y = x^2/2$ sobre
$\intcc0a $ e mostre que é igual
\[
\tfrac12\Bigl(a\sqrt{1 + a^2} + \ln\bigl(a + \sqrt{1 +
a^2}\bigr)\Bigr).
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:curves:11}
Deixar $\gamma $ seja regular $\mathcal C^2$ arco em $\R^n $ todos
cujo linhas tangentes passar por um ponto fixo $ P $. Prove isso
a trajetória de $\gamma $ está contido em uma linha reta.
\emph{(Parametrizar por \omterm{def:b2:curves:length}{comprimento} do arco, escrever $\gamma(s) +
\lambda(s)T(s) = P $ e diferenciar.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:curves:12}
(Teorema fundamental para curvas espaciais) Deixe $\kappa > 0$ e
$\tau $ ser \omterm{def:b2:metric:continuity}{contínuo} funciona em um intervalo $ J $. Realizar
o programa da observação seguinte
\cref{ex:b2:curves:helixfrenet}: (a) mostre que o linear
sistema $ F' = FA(s)$, com $ A(s)$ o Frenet antisimétrico
matriz construída a partir de $\kappa, \tau $ e $ F(s_0)$ um direto
referencial ortonormal, tem uma solução global única, que permanece
um referencial ortonormal direto; (b) construir uma velocidade unitária
curva birregular com \omterm{thm:b2:curves:frenet2d}{curvatura} $\kappa $ e torção $\tau $;
(c) provar a unicidade até uma isometria direta de $\R^3$.
\end{exercise}

\section{Problema: envelopes --- o astroide, duas evoluções e
um cáustico}

  \draw[->] (0,-1.15) -- (0,1.25) node[left] {$ y $};
  \draw[omDef] (1,0) -- (0,0.0) (0.966,0) -- (0,0.259)

\end{tikzpicture}

{\small Uma escada de \omterm{def:b2:curves:length}{comprimento} $1$ deslizando por uma parede (azul
posições) nunca cruza o \omterm{pb:b2:curves:1}{astroide} $ x^{2/3} + y^{2/3} = 1$
(vermelho): o \omterm{pb:b2:curves:1}{astroide} é o \emph{\omterm{pb:b2:curves:1}{envelope}} da família de

\begin{problem}[{Problema de fim de semana --- a máquina de envelopes e
quatro curvas clássicas}]\label{pb:b2:curves:1}
Uma família de linhas de um parâmetro geralmente não consegue cobrir o
plano \omterm{def:b2:funcseq:def}{uniformemente}: as linhas se acumulam ao longo de uma curva tangente a todos
deles, seus \emph{envelope}\index{envelope}. Raios de luz
fazer \omterm{pb:b2:curves:1}{envelopes} visível como \emph{cáusticos} --- o brilhante
curva cúspide em uma caneca de café. Este problema constrói o
geral \omterm{pb:b2:curves:1}{envelope} máquina e, em seguida, aciona-a quatro vezes: o deslizamento
escada (\omterm{pb:b2:curves:1}{astroide}), as normais da parábola e do
ciclóide (evolui, com o pêndulo de Huygens no final), e o
cáustico de xícara de café (\omterm{pb:b2:curves:1}{nefróide}). Por todo, $ D_t $ denota o
linha de equação $ a(t)\,x + b(t)\,y = c(t)$, onde $ a, b, c $
são $\mathcal C^2$ funções com $(a(t), b(t)) \neq (0,0)$,
e $\Delta(t) = a(t)b'(t) - a'(t)b(t)$.

\medskip
\noindent\textbf{Part I --- The \omterm{pb:b2:curves:1}{envelope} machine.}
\begin{enumerate}
  \item Suponha $\Delta(t) \neq 0$. Mostre que o
        \emph{sistema característico}
        \[
        \begin{cases} a(t)\,x + b(t)\,y = c(t)\\
        a'(t)\,x + b'(t)\,y = c'(t)\end{cases}
        \]
        tem uma solução única $ E(t) = (x(t), y(t))$, dado por
        $ x = \dfrac{cb' - c'b}{\Delta}$, $ y = \dfrac{ac' -
        a'c}{\Delta}$.
  \item Suponha, além disso, que $ E $ é $\mathcal C^1$ aproximar $ t $
        com $ E'(t) \neq 0$. Diferenciando o primeiro
        equação do sistema, mostre $ a(t)\,x'(t) +
        b(t)\,y'(t) = 0$, e concluímos que a curva $ E $
        passa por um ponto de $ D_t $ \emph{com o
        direção de} $ D_t $: a família é tangente a $ E $,
        que é chamado de seu \emph{\omterm{pb:b2:curves:1}{envelope}}.
  \item Verificação de sanidade: o linhas tangentes da parábola $ y =
        x^2/2$ nos pontos $(t, t^2/2)$ são $ tx - y =
        t^2/2$. Verifique se o \omterm{pb:b2:curves:1}{envelope} máquina retorna o
        própria parábola.
  \item (\omterm{pb:b2:curves:1}{Envelope} dos normais) Deixe $\gamma $ ser de velocidade unitária
        com $\kappa(s) \neq 0$. A linha normal em
        $\gamma(s)$ é $\{M : \langle M - \gamma(s), T(s)
        \rangle = 0\}$. Mostre que seu sistema característico
        forças $\langle M - \gamma(s), N(s)\rangle =
        1/\kappa(s)$, portanto, o ponto característico é
        o \omterm{def:b2:curves:curvature}{centro de curvatura}: \emph{o \omterm{pb:b2:curves:1}{envelope} do
        normais é a evolução}, recuperando
        \cref{exo:b2:curves:6}. Verificar $\Delta(s) =
        \kappa(s)$.
  \item Duas degenerações. Para o lápis $ D_\theta : x\cos
        \theta + y\sin\theta = 0$, mostre que
        ponto característico é a origem de cada $\theta $
        (o ``\omterm{pb:b2:curves:1}{envelope}'' entra em colapso até um ponto, e $ E' = 0$:
        a questão 2 não se aplica). Para uma família paralela
        linhas ($ a, b $ constante), mostrar $\Delta \equiv 0$ e
        que o sistema característico é em geral
        inconsistente: não \omterm{pb:b2:curves:1}{envelope}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The sliding ladder and the
\omterm{pb:b2:curves:1}{astroide}.} Um segmento de \omterm{def:b2:curves:length}{comprimento} $1$ slides com uma extremidade $ P_t =
(\cos t, 0)$ no chão e o outro $ Q_t = (0, \sin t)$ ligado
a parede, $ t \in \intoo0{\pi/2}$.
\begin{enumerate}[resume]
  \item Mostre que a reta $(P_tQ_t)$ tem equação $ x\sin t +
        y\cos t = \sin t\cos t $, e que o \omterm{pb:b2:curves:1}{envelope} máquina
        dá o ponto característico
        \[
        E(t) = (\cos^3 t,\ \sin^3 t) :
        \]
        o \emph{astroide}\index{astroide}, de implícito
        equação $ x^{2/3} + y^{2/3} = 1$ (estendido ao
        outros quadrantes por simetria).
  \item Mostre isso $ E'(0) = 0$ e, usando o local
        classificação (\cref{prop:b2:curves:local}), que
        o \omterm{pb:b2:curves:1}{astroide} tem uma cúspide do primeiro tipo em $(1, 0)$
        --- e da mesma forma em seus quatro pontos do eixo.
  \item Calcular $\norm{E'(t)} = \tfrac32\abs{\sin 2t}$ e
        deduzir que o total \omterm{def:b2:curves:length}{comprimento} do \omterm{pb:b2:curves:1}{astroide} é $6$.
  \item Onde a escada toca o \omterm{pb:b2:curves:1}{astroide}? Mostrar $ E(t) =
        P_t + \sin^2 t\,(Q_t - P_t)$: o ponto de contato
        divide a escada na proporção $\sin^2 t : \cos^2
        t $, varrendo-o de uma ponta a outra como o
        corrediças de escada.
  \item Calcule a área delimitada pelo \omterm{pb:b2:curves:1}{astroide}: mostre isso
        a área do primeiro quadrante é $3\int_0^{\pi/2}\sin^4
        t\cos^2 t\,\dd t $, avalie a integral por
        linearização ($\sin^2 2t = \tfrac{1 - \cos 4t}2$),
        e concluímos que a área total é $3\pi/8$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The evolute of the parabola.}
Deixar $\gamma(t) = (t, t^2/2)$.
\begin{enumerate}[resume]
  \item Mostre que a reta normal em $\gamma(t)$ tem equação
        $ x + t\,y = t + t^3/2$.
  \item Execute o \omterm{pb:b2:curves:1}{envelope} máquina: mostre que \omterm{pb:b2:curves:1}{envelope} de
        o normal é
        \[
        E(t) = \Bigl(-t^3,\ 1 + \tfrac32 t^2\Bigr),
        \]
        com equação implícita $ x^2 = \tfrac8{27}(y - 1)^3$:
        uma parábola semicúbica.
  \item Verifique com a pergunta 4: calcule o \omterm{def:b2:curves:curvature}{centro de
        curvatura} $\gamma(t) + \frac1{\kappa(t)}N(t)$ de
        $\kappa(t) = (1 + t^2)^{-3/2}$
        (\cref{ex:b2:curves:kappaexamples}) e recuperar o
        mesmo ponto.
  \item Mostre que a evoluta tem uma cúspide de primeiro tipo em
        $(0, 1)$, o \omterm{def:b2:curves:curvature}{centro de curvatura} no vértice ---
        o ponto onde $\kappa $ é extremo, como previsto por
        a fórmula $ c' = -\frac{\kappa'}{\kappa^2}N $ de
        \cref{exo:b2:curves:6}.
  \item Quantas normais da parábola passam por um dado
        ponto $(x_0, y_0)$? Mostre que a resposta é governada por
        o cúbico $\tfrac{t^3}2 + (1 - y_0)\,t - x_0 = 0$;
        tratar o caso do eixo $ x_0 = 0$ completamente (um normal
        para $ y_0 < 1$, três para $ y_0 > 1$) e interpretar
        a evolução como a curva de transição.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The coffee-cup caustic.} Paralelo
raios de direção $(1, 0)$ golpeie o interior do espelho
círculo $ x^2 + y^2 = 1$; o raio atingindo $ P_\theta =
(\cos\theta, \sin\theta)$ reflete de acordo com a lei de
reflexão.
\begin{enumerate}[resume]
  \item Da simetria do espelho na normal (o raio),
        justificar que a direção refletida é $ v = u -
        2\langle u, n\rangle n $ com $ u = (1,0)$, $ n =
        (\cos\theta, \sin\theta)$ e calcular $ v =
        -(\cos2\theta, \sin2\theta)$.
  \item Mostre que o raio refletido está na reta
        \[
        x\sin 2\theta - y\cos 2\theta = \sin\theta .
        \]
  \item Execute o \omterm{pb:b2:curves:1}{envelope} máquina ($\Delta = 2$): mostre que
        o cáustico é
        \[
        E(\theta) = \Bigl(\tfrac{3\cos\theta -
        \cos3\theta}4,\ \tfrac{3\sin\theta -
        \sin3\theta}4\Bigr),
        \]
        o \emph{nefróide}\index{nefróide}.
  \item Calcular $ E'(\theta) = \tfrac32\sin\theta\,
        (\cos2\theta, \sin2\theta)$; verifique se a tangente
        direção é a direção do raio refletido (questão
        16), localize as duas cúspides $(\pm\tfrac12, 0)$ e
        mostre que o raio refletido cruza o eixo $ y = 0$
        no $ x = \frac1{2\cos\theta}$ --- então quase raios axiais
        foco em $ x = \tfrac12$: o foco \omterm{def:b2:curves:length}{comprimento} $ R/2$ de um
        espelho de raio $ R $.
  \item Mostre que o \omterm{pb:b2:curves:1}{nefróide} tem total \omterm{def:b2:curves:length}{comprimento} $6$ e isso
        perto $\theta = 0$,
        \[
        E(\theta) - \bigl(\tfrac12, 0\bigr) =
        \bigl(\tfrac34\theta^2 + o(\theta^2),\ \theta^3 +
        o(\theta^3)\bigr) :
        \]
        uma cúspide do primeiro tipo, apontando ao longo do eixo.
  \item Explique em um parágrafo por que o cáustico é brilhante:
        através de cada ponto fora da passagem cáustica dois
        raios refletidos, através de cada ponto dele os raios
        estão ``infinitamente concentrados'' (o mapa $(\theta,
        \text{distance along ray}) \mapsto \R^2$ tem um
        ponto crítico exatamente no \omterm{pb:b2:curves:1}{envelope}).
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Huygens: the cycloid is its own
evolute.} Deixar $\gamma(t) = (t - \sin t,\ 1 - \cos t)$, $ t \in
\intoo0{2\pi}$, um arco da ciclóide.
\begin{enumerate}[resume]
  \item Calcular $\kappa(t) = -\dfrac1{4\sin(t/2)}$ e o
        \omterm{def:b2:curves:curvature}{centro de curvatura}; mostre que a evolução é
        \[
        c(t) = (t + \sin t,\ \cos t - 1),
        \]
        e que a substituição $ t = u + \pi $ exibe isso
        como a ciclóide original traduzida por $(\pi, -2)$:
        \emph{a evolução de uma ciclóide é congruente
        ciclóide} (Huygens).
  \item Verifique se o \omterm{def:b2:curves:curvature}{raio de curvatura} no ápice $ t =
        \pi $ é igual $4$, que é metade do \omterm{def:b2:curves:length}{comprimento} $8$ de um
        arco (\cref{exo:b2:curves:1}); localize a cúspide do
        evoluem diretamente abaixo do ápice, à distância $4$.
  \item (A propriedade string) Deixe $\gamma $ ter velocidade unitária com
        $\kappa > 0$, $\kappa $ de classe $\mathcal C^1$ e
        $ R = 1/\kappa $ estritamente monótono. Usando $ c' = R'N $,
        mostrar que o \omterm{def:b2:curves:length}{comprimento} do arco da evolução entre
        $ c(s_0)$ e $ c(s_1)$ é $\abs{R(s_1) - R(s_0)}$.
        Interprete: uma corda esticada desenrolada da evolução,
        de \omterm{def:b2:curves:length}{comprimento} $ R(s_0)$ no início, tem seu final livre
        trace a curva original --- então um pêndulo balançando
        entre duas bochechas cicloidais do relógio de Huygens
        descreve uma ciclóide.
  \item Síntese. A máquina da Parte I produziu o
        \omterm{pb:b2:curves:1}{astroide}, uma parábola semcúbica, uma \omterm{pb:b2:curves:1}{nefróide} e um
        ciclóide. Para cada uma das quatro famílias, indique em uma
        frase onde as hipóteses $\Delta \neq 0$ e
        $ E' \neq 0$ realizada ou falhou, e qual evento geométrico
        (cúspide, foco, degeneração) cada falha de $ E' \neq 0$
        sinalizado. Onde devem extremos de \omterm{thm:b2:curves:frenet2d}{curvatura} aparecer em
        o \omterm{pb:b2:curves:1}{envelope} dos normais, e por quê?
\end{enumerate}
\end{problem}
