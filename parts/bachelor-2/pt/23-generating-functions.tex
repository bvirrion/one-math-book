\chapter{Funções de geração de probabilidade}\label{ch:b2:genfun}

A série de potências de \cref{ch:b2:powerseries} voltar com um
missão probabilística: para um $\N $-valorizado variável aleatória nós anexamos
a série de potências com coeficientes $\P(X = n)$. Esse
\emph{\omterm{ex:b2:powerseries:fibonacci}{função geradora}} converte somas de \omterm{def:b2:proba:independence}{independente} variáveis
em produtos, momentos em derivativos em $1$ e difícil
identidades combinatórias em multiplicações de uma linha. O
capítulo fecha o livro com duas peças de destaque: o Poisson
aproximação de raro \omterm{def:b2:proba:space}{eventos}, e o critério de extinção para
processos ramificados --- uma probabilística genuinamente infinita
cálculo resolvido inteiramente pela geometria de uma curva convexa.

\section{Definição e propriedades básicas}

\begin{definition}[Função geradora de probabilidade]\label{def:b2:genfun:pgf}
Deixar $ X $ feijão $\N $-valorizado variável aleatória, $ p_n = \P(X = n)$. O
\emph{função geradora de probabilidade}\index{função geradora de probabilidade} de $ X $ é a soma dos
série de potência
\[
G_X(t) = \E\bigl(t^X\bigr) = \sum_{n=0}^{\infty} p_n\,t^n .
\]
\end{definition}

\begin{example}[Primeiros reflexos]\label{ex:b2:genfun:firstreflexes}
Uma variável constante $ X = c $ tem $ G_X(t) = t^c $; uma mudança obedece
$ G_{X+c}(t) = t^c\,G_X(t)$; e avaliando em pontos especiais
lê informações sem qualquer expansão: $ G_X(0) = \P(X
= 0)$, $ G_X(1) = 1$, e $ G_X(-1) = \P(X\text{ even}) -
\P(X\text{ odd})$, o equilíbrio de paridade explorado em
\cref{exo:b2:genfun:10}. Esses one-liners são usados silenciosamente
em todos os lugares abaixo --- e a avaliação $ G_X(0)$ é exatamente
como as probabilidades de extinção serão extraídas de iterados
gerando funções no final do capítulo.
\end{example}

\begin{proposition}[Raio e primeiras propriedades]\label{prop:b2:genfun:radius}
A série que define $ G_X $ tem \omterm{def:b2:powerseries:radius}{raio de convergência} $\geq 1$;
$ G_X $ é definido e \omterm{def:b2:metric:continuity}{contínuo} sobre $\intcc{-1}{1}$,
$\mathcal{C}^\infty $ sobre $\intoo{-1}{1}$, com $ G_X(1) = 1$ e
$\abs{G_X(t)} \leq 1$ lá. Além disso $ G_X $ determina o \omterm{def:b2:randomvar:law}{lei} de
$ X $:
\[
p_n = \frac{G_X^{(n)}(0)}{n!} .
\]
\end{proposition}

\begin{proof}
Desde $\sum p_n = 1$ converge, os termos $ p_n\,1^n $ são limitados,
então o raio é $\geq 1$ (Lema de Abel,
\cref{ch:b2:powerseries}); no $ t = \pm1$ a série converge
\omterm{def:b2:series:def}{absolutamente} ($\sum p_n = 1$ domina); melhor, no geral
intervalo $\intcc{-1}1$,
\[
\sup_{\abs t\leq1}\,\abs{p_nt^n} = p_n
\quad\text{with}\quad \sum_np_n < \infty :
\]
a série converge \emph{\omterm{def:b2:funcseq:series}{normalmente}} sobre $\intcc{-1}1$, então
sua soma é \omterm{def:b2:metric:continuity}{contínuo} lá
(\cref{thm:b2:funcseq:weierstrass,thm:b2:funcseq:continuity}). Suavidade interna e coeficiente
fórmula são a teoria geral das séries de potências; os coeficientes
sendo recuperáveis, duas variáveis com o mesmo \omterm{ex:b2:powerseries:fibonacci}{função geradora}
tem o mesmo \omterm{def:b2:randomvar:law}{lei}.
\end{proof}

\begin{example}[As leis clássicas]\label{ex:b2:genfun:classical}
\leavevmode
\begin{itemize}
\item Bernoulli $\mathcal{B}(p)$: $ G(t) = 1 - p + pt $.
\item Binômio $\mathcal{B}(n, p)$: $ G(t) = \sum_k \binom nk
(pt)^k(1-p)^{n-k} = (1 - p + pt)^n $ (teorema binomial).
\item Geométrico $\mathcal{G}(p)$: $ G(t) =
\sum_{k\geq1}(1-p)^{k-1}p\,t^k = \dfrac{pt}{1 - (1-p)t}$ (raio
$\frac{1}{1-p} > 1$).
\item Poisson $\mathcal{P}(\lambda)$: $ G(t) =
\sum_k e^{-\lambda}\frac{(\lambda t)^k}{k!} = e^{\lambda(t - 1)}$
(raio $\infty $).
\end{itemize}
\end{example}

\begin{example}[Integrando a geração
função]\label{ex:b2:genfun:integraltrick}
Derivados de $ G_X $ no $1$ dê momentos positivos; o
\emph{integrante} dá um negativo. De $\int_0^1t^k\dd t
= \frac1{k+1}$ e integração termo a termo (normais
convergência sobre $\intcc01$):
\[
\int_0^1G_X(t)\,\dd t = \sum_{k\geq0}\frac{\P(X =
k)}{k+1} = \E\Bigl(\frac1{1+X}\Bigr).
\]
Para $ X \sim \mathcal P(\lambda)$:
\[
\E\Bigl(\frac1{1+X}\Bigr) =
\int_0^1\eu^{\lambda(t-1)}\,\dd t = \frac{1 -
\eu^{-\lambda}}{\lambda},
\]
recuperando em uma linha o cálculo da série de
\cref{ex:b2:randomvar:transferex}. O \omterm{ex:b2:powerseries:fibonacci}{função geradora}
é um instrumento de mão dupla: diferenciar em $1$ para o
momentos $\E(X)$, $\E(X(X-1))$, integrar sobre $\intcc01$
para $\E\bigl(\frac1{1+X}\bigr)$ --- um \omterm{def:b2:powerseries:analytic}{analítico} objeto,
questionado em qualquer direção que o problema precise.
\end{example}

\begin{example}[Uma lei com raio exato
um]\label{ex:b2:genfun:heavytail}
Deixar $\P(X = k) = \dfrac{6}{\pi^2k^2}$ para $ k \geq 1$ --- um
probabilidade \omterm{def:b2:randomvar:law}{lei} pela identidade de Basileia
(\cref{ex:b2:fourier:basel}). Isso é \omterm{ex:b2:powerseries:fibonacci}{função geradora}
$ G(t) = \frac6{\pi^2}\sum_{k\geq1}\frac{t^k}{k^2}$ tem \omterm{def:b2:powerseries:radius}{raio
de convergência} exatamente $1$: o limite geral ``raio $\geq
1$'' de \cref{prop:b2:genfun:radius} não pode ser melhorado. E
a média é
\[
\sum_{k\geq1}k\,\P(X = k) =
\frac6{\pi^2}\sum_{k\geq1}\frac1k = \infty :
\]
$ G $ é \omterm{def:b2:metric:continuity}{contínuo} sobre $\intcc{-1}1$, liso por dentro, mas é
derivativo explode em $1^-$ --- o gráfico chega ao
ponto $(1, 1)$ com uma tangente vertical. Caudas pesadas são
visível \emph{geometricamente} no \omterm{ex:b2:powerseries:fibonacci}{função geradora}, em
o único ponto $ t = 1$; o teorema dos momentos abaixo faz
esta correspondência exata.
\end{example}

\begin{theorem}[Momentos da geração
função]\label{thm:b2:genfun:moments}
$ X $ tem um \omterm{def:b2:randomvar:expectation}{expectativa} se e somente se $ G_X $ é \omterm{def:b2:diffcalc:differential}{diferenciável} no
$1^-$ (derivada à esquerda, finito) e então $\E(X) = G_X'(1)$.
De forma similar $ X $ tem um segundo momento se $ G_X $ é duas vezes
\omterm{def:b2:diffcalc:differential}{diferenciável} no $1^-$, e então
\[
\E\bigl(X(X - 1)\bigr) = G_X''(1),
\qquad
V(X) = G_X''(1) + G_X'(1) - G_X'(1)^2 .
\]
\end{theorem}

\begin{proof}
Para $ t \in \intoo{0}{1}$, diferenciação termo a termo dentro do
disco dá $ G_X'(t) = \sum_{n\geq1} np_n t^{n-1}$, uma série com
coeficientes não negativos: $ t \mapsto G_X'(t)$ é não decrescente em
$\intoo{0}{1}$, e por convergência monótona de somas parciais (ou
Teorema de Abel para coeficientes não negativos,
\cref{ch:b2:powerseries}),
\[
\lim_{t \to 1^-} G_X'(t)
= \sum_{n\geq1} n\,p_n \in \intcc{0}{+\infty} ,
\]
cada lado é finito exatamente quando o outro é. Quando finito, a média
teorema do valor comprime os quocientes de diferença $\frac{G_X(1) -
G_X(t)}{1 - t}$ entre valores de $ G_X'$, então $ G_X $ é
\omterm{def:b2:diffcalc:differential}{diferenciável} no $1^-$ com $ G_X'(1) = \sum np_n = \E(X)$ (por
transferência). A declaração de segunda ordem repete o argumento um nível
para cima: $ G''_X(t) = \sum_{n\geq2}n(n-1)p_nt^{n-2}$ é
não decrescente em $\intoo01$ com limite monótono
$\sum_nn(n-1)p_n = \E(X(X-1))$, finito exatamente quando $ X $ tem
um segundo momento. O \omterm{def:b2:randomvar:variance}{variação} a fórmula segue então de
König--Huygens:
\[
V(X) = \E(X^2) - \E(X)^2 = \E\bigl(X(X-1)\bigr) + \E(X) -
\E(X)^2 = G''_X(1) + G'_X(1) - G'_X(1)^2 .
\]
\end{proof}

\begin{example}
Poisson: $ G'(t) = \lambda e^{\lambda(t-1)}$, então $\E(X) = \lambda $;
$ G''(1) = \lambda^2$, então $ V(X) = \lambda^2 + \lambda - \lambda^2 =
\lambda $ --- os cálculos de \cref{ch:b2:randomvar} em um
linha cada.
\end{example}

\begin{example}[O modo de uma lei de Poisson]
\label{ex:b2:genfun:poissonmode}
Onde está $\P(X = k)$ maior para $ X \sim \mathcal
P(\lambda)$? Pesos consecutivos são comparados por meio da proporção
\[
\frac{\P(X = k+1)}{\P(X = k)} = \frac{\lambda}{k + 1} ,
\]
que excede $1$ enquanto $ k < \lambda - 1$ e cai abaixo
$1$ uma vez $ k > \lambda - 1$: os pesos sobem e depois caem, com
modo $\floor\lambda $ (e um empate entre $\lambda - 1$ e
$\lambda $ quando $\lambda $ é um número inteiro: para $\lambda = 3$,
$\P(X = 2) = \P(X = 3) = \frac92\eu^{-3} \approx 0.224$).
Os testes de proporção dos coeficientes costumam ser o caminho mais rápido
a fatos qualitativos sobre um discreto \omterm{def:b2:randomvar:law}{lei} --- não gerando
função necessário, mas os coeficientes \emph{são} o
\omterm{ex:b2:powerseries:fibonacci}{função geradora}, leia termo por termo.
\end{example}

\section{Somas de variáveis independentes}

\begin{theorem}[Multiplicatividade]\label{thm:b2:genfun:product}
Se $ X $ e $ Y $ são \omterm{def:b2:proba:independence}{independente} $\N $-valorizado variáveis aleatórias, então
\[
G_{X + Y}(t) = G_X(t)\,G_Y(t)
\qquad (\abs t \leq 1),
\]
e por indução $ G_{X_1 + \dots + X_n} = \prod_i G_{X_i}$ para
\omterm{def:b2:proba:independence}{independente} $ X_1, \dots, X_n $.
\end{theorem}

\begin{proof}
Duas provas, ambas instrutivas. \emph{Através \omterm{def:b2:randomvar:expectation}{expectativas}:} $ t^X $ e
$ t^Y $ são \omterm{def:b2:proba:independence}{independente} variáveis limitadas, então
(\cref{thm:b2:randomvar:product})
\[
G_{X+Y}(t) = \E\bigl(t^{X+Y}\bigr)
= \E\bigl(t^X t^Y\bigr)
= \E\bigl(t^X\bigr)\E\bigl(t^Y\bigr) .
\]
\emph{Através Produtos Cauchy:} o \omterm{def:b2:randomvar:law}{lei} de $ X + Y $ é a convolução
$\P(X + Y = n) = \sum_{k=0}^n \P(X = k)\P(Y = n - k)$, e o
Produto Cauchy teorema para \omterm{def:b2:series:def}{absolutamente} série convergente
(\cref{ch:b2:series}) multiplica as duas séries de potências exatamente
ao longo desta convolução.
\end{proof}

\begin{example}[Estabilidade das leis clássicas]\label{ex:b2:genfun:stability}
\omterm{def:b2:proba:independence}{Independente} binômios com o mesmo $ p $ adicionar: $(1 - p + pt)^m(1 -
p + pt)^n = (1 - p + pt)^{m+n}$, então $\mathcal{B}(m, p) +
\mathcal{B}(n, p) = \mathcal{B}(m + n, p)$ --- em particular uma soma
de $ n $ \omterm{def:b2:proba:independence}{independente} As variáveis de Bernoulli são binomiais, reprovando a
\omterm{def:b2:randomvar:law}{lei} do número de sucessos. \omterm{def:b2:proba:independence}{Independente} Poissons acrescenta:
$ e^{\lambda(t-1)}e^{\mu(t-1)} = e^{(\lambda + \mu)(t-1)}$, então
$\mathcal{P}(\lambda) + \mathcal{P}(\mu) = \mathcal{P}(\lambda +
\mu)$ --- o cálculo de convolução de
\cref{exo:b2:randomvar:2}, agora sem cálculo.
\end{example}

\begin{example}[Dois dados, um polinômio
ao quadrado]\label{ex:b2:genfun:twodice}
Por um dado justo, $ G(t) = \frac{t + t^2 + \dots + t^6}{6}$;
pela soma de dois,
\[
G(t)^2 = \frac{1}{36}\bigl(t^2 + 2t^3 + 3t^4 + 4t^5 + 5t^6 +
6t^7 + 5t^8 + 4t^9 + 3t^{10} + 2t^{11} + t^{12}\bigr) :
\]
o triangular \omterm{def:b2:randomvar:law}{lei} de somas de dados ($7$ é o modo, com
probabilidade $\frac6{36} = \frac16$), leia um polinômio
ao quadrado que um se multiplica uma vez na vida. O
fórmula de convolução exigiria onze
contando argumentos; o \omterm{ex:b2:powerseries:fibonacci}{função geradora} faz todos eles
simultaneamente, porque multiplicar polinômios \emph{é}
coeficientes convolutivos. Esta tradução mecânica ---
\omterm{def:b2:randomvar:law}{leis} para coeficientes, somas para produtos --- é o todo
modelo de negócios do capítulo, e
\cref{exo:b2:genfun:11} empurra para o surpreendente
Dados de Sicherman.
\end{example}

\begin{example}[Três dados e um coeficiente
extração]\label{ex:b2:genfun:threedice}
Para a soma $ S $ de três dados justos, $\P(S = 10)$ é o
coeficiente de $ t^{10}$ em $\bigl(\frac{t + \dots +
t^6}6\bigr)^3$. Fatore e expanda com o binômio e
série geométrica:
\[
\Bigl(\frac{t(1 - t^6)}{6(1 - t)}\Bigr)^{\!3}
= \frac{t^3}{216}\,\bigl(1 - 3t^6 + 3t^{12} -
t^{18}\bigr)\sum_{j\geq0}\binom{j+2}2t^j .
\]
O coeficiente de $ t^{10}$ requer $ t^7$ do produto:
$ j = 7$ com o termo $1$, e $ j = 1$ com o termo
$-3t^6$:
\[
\P(S = 10) = \frac{1}{216}\Bigl(\binom92 -
3\binom32\Bigr) = \frac{36 - 9}{216} = \frac{27}{216} =
\frac18 .
\]
Enumeração direta do $27$ triplos são propensos a erros; o
a álgebra é mecânica e se adapta a qualquer número de dados ---
a inclusão-exclusão visível em $(1 - t^6)^3$ está fazendo
a análise do caso automaticamente.
\end{example}

\begin{example}[Lendo uma lei sobre sua geração
função]\label{ex:b2:genfun:identify}
Qual \omterm{def:b2:randomvar:law}{lei} tem $ G(t) = \dfrac1{2 - t}$? Expanda para um poder
série:
\[
\frac{1}{2 - t} = \frac12\cdot\frac1{1 - t/2}
= \sum_{k\geq0}\frac{t^k}{2^{k+1}} :
\]
coeficientes não negativos somando a $ G(1) = 1$, então este é um
genuíno \omterm{def:b2:randomvar:law}{lei}, $\P(X = k) = 2^{-(k+1)}$ sobre $\N $ --- um
geométrico \omterm{def:b2:randomvar:law}{lei} começando em $0$. Por singularidade
(\cref{prop:b2:genfun:radius}), nenhum outro \omterm{def:b2:randomvar:law}{lei} compartilha isso
$ G $. Reconhecendo \omterm{def:b2:randomvar:law}{leis} do seu gerando funções é um
habilidade que vale a pena aprofundar: é como a ramificação crítica
iterar $ G_n(t) = \frac{n - (n-1)t}{n+1 - nt}$ do
problema do fim de semana é desmascarado como um problema geométrico \omterm{def:b2:randomvar:law}{lei} condicionado
na sobrevivência.
\end{example}

\begin{remark}
A estabilidade só vai num sentido: somas de \omterm{def:b2:proba:independence}{independente} Poissons são
Poisson, mas as diferenças não são --- $ X - Y $ leva negativo
valores, então não tem \omterm{ex:b2:powerseries:fibonacci}{função geradora} em tudo, e seu \omterm{def:b2:randomvar:law}{lei}
(o Skellam \omterm{def:b2:randomvar:law}{distribuição}) está fora deste capítulo
kit de ferramentas. Da mesma maneira $\mathcal B(m, p) + \mathcal B(n, p')$ com
$ p \neq p'$ é \emph{não} binômio: o produto $(1 - p +
pt)^m(1 - p' + p't)^n $ tem dois locais de raiz distintos, enquanto
todo pgf binomial tem uma única raiz repetida. Lendo
estabilidade dos padrões de raiz é uma pequena prévia de quanto
estruturar as codificações polinomiais.
\end{remark}

\begin{remark}[O filtro de raízes da unidade]
Avaliando em $-1$ separa par de ímpar; avaliando em tudo
$ m $-ésima raiz da unidade separa cada classe de resíduos: com
$\omega = \eu^{2\iu\pi/m}$,
\[
\P(X \equiv r \bmod m)
= \frac1m\sum_{j=0}^{m-1}\omega^{-jr}\,G_X(\omega^j),
\]
desde a média $\omega^{j(k-r)}$ sobre $ j $ rendimentos $1$ se $ k
\equiv r $ e $0$ de outra forma. Exemplo de dividendo: para a soma $ S $
de dois dados justos, cada um $ G(\omega^j) = \frac16\sum_{k=1}^6
\omega^{jk} = -\frac16$ para $ j \neq 0$ (o sétimo sétimo
raízes da unidade somam zero), então
\[
\P(7 \mid S) = \frac17\Bigl(1 +
6\cdot\frac1{36}\Bigr) = \frac16 ,
\]
confirmando a contagem de \cref{ex:b2:genfun:twodice} ---
e o método se adapta a questões em que a contagem direta
não.
\end{remark}

\begin{theorem}[Somas aleatórias: identidade de Wald para gerar
funções]\label{thm:b2:genfun:compound}
Deixar $(X_k)_{k\geq1}$ ser \omterm{def:b2:proba:independence}{independente} $\N $ variáveis com valor com o
mesmo \omterm{def:b2:randomvar:law}{lei} e \omterm{ex:b2:powerseries:fibonacci}{função geradora} $ G_X $, e deixe $ N $ feijão
$\N $ variável com valor \omterm{def:b2:proba:independence}{independente} do $ X_k $, com gerando
função $ G_N $. Então a soma aleatória $ S = X_1 + \dots + X_N $ (com
$ S = 0$ quando $ N = 0$) tem \omterm{ex:b2:powerseries:fibonacci}{função geradora}
\[
G_S = G_N \circ G_X .
\]
Em particular, se $ N $ e $ X_1$ ter \omterm{def:b2:randomvar:expectation}{expectativas}, $\E(S) =
\E(N)\,\E(X_1)$.
\end{theorem}

\begin{proof}
Condição ativada $ N $ (probabilidade total,
\cref{thm:b2:proba:bayes}): para $\abs t \leq 1$,
\[
G_S(t) = \sum_{n=0}^\infty \P(N = n)\,
\E\bigl(t^{X_1 + \dots + X_n}\bigr)
= \sum_{n=0}^\infty \P(N = n)\,G_X(t)^n
= G_N\bigl(G_X(t)\bigr),
\]
usando multiplicatividade para cada fixo $ n $ e o somabilidade de
toda a família dupla ($\abs{G_X(t)} \leq 1$). A troca de
somatórios é Fubini para resumido famílias
(\cref{ch:b2:series}). Diferenciando em $1^-$ pela regra da cadeia
e \cref{thm:b2:genfun:moments}: $\E(S) = G_N'(G_X(1))\,G_X'(1)
= G_N'(1)G_X'(1) = \E(N)\E(X_1)$.
\end{proof}

\begin{example}[Composto Poisson: seguro anual
perdas]\label{ex:b2:genfun:compoundpoisson}
Uma seguradora recebe $ N \sim \mathcal P(\lambda)$ reivindicações em um
ano, cada sinistro custando $ X_k $ (unidades inteiras, i.i.d., pgf
$ G_X $, significar $\mu $, \omterm{def:b2:proba:independence}{independente} de $ N $). Por
\cref{thm:b2:genfun:compound}, a perda total $ S $ tem
\[
G_S(t) = \eu^{\lambda(G_X(t) - 1)},
\qquad
\E(S) = \lambda\mu ,
\]
e diferenciando duas vezes em $1^-$:
\[
V(S) = \lambda\,G_X''(1) + \lambda^2\mu^2 + \lambda\mu -
(\lambda\mu)^2 = \lambda\,\E(X^2) .
\]
O \omterm{def:b2:randomvar:variance}{variação} envolve o \emph{segundo} momento de um único
reivindicação, não a sua \omterm{def:b2:randomvar:variance}{variação}: uma soma de Poisson composta sente o
grandes reivindicações ocasionais duas vezes --- uma vez até quantas, uma vez
através de quão grande. Para $\lambda = 10$ reivindicações de geometria \omterm{def:b2:randomvar:law}{lei}
com média $2$ ($\E X^2 = 6$): $\E S = 20$, $ V(S) = 60$ e
Chebyshev (\cref{ch:b2:randomvar}) já rende utilizável
margens de solvência. Este padrão de ``soma interrompida aleatoriamente'' é o
o mesmo que conduzirá a recursão ramificada de
\cref{prop:b2:genfun:branching}: composição de gerando
funções é a álgebra de populações aleatórias.
\end{example}

\begin{remark}
O independência de $ N $ dos summands não é decorativo.
Pegar $ X_k \in \{0, 2\}$ com probabilidades iguais e deixe $ N =
X_1$ (flagrantemente dependente): então $ S = X_1 + \dots + X_N $ é
$0$ quando $ X_1 = 0$, e $2 + X_2$ quando $ X_1 = 2$, então $\E(S) =
\frac12(2 + 1) = \frac32$, enquanto $\E(N)\E(X_1) = 1\cdot1 =
1$: A identidade de Wald falha. Quando o número de termos é
permitido \emph{reagir} aos próprios termos, o limpo
a estrutura do produto entra em colapso --- a teoria completa de tal
``parar'' regras é o capítulo martingale do Ano 3
volume.
\end{remark}

\section{Aproximação de Poisson}

\begin{theorem}[Lei dos eventos raros]\label{thm:b2:genfun:rareevents}
Deixar $ X_n \sim \mathcal{B}(n, p_n)$ com $ n\,p_n \to \lambda > 0$.
Então para cada $ k \in \N $:
\[
\P(X_n = k)
\xrightarrow[n\to\infty]{}
e^{-\lambda}\frac{\lambda^k}{k!} :
\]
o binômio \omterm{def:b2:randomvar:law}{lei} de muitos raros \omterm{def:b2:proba:space}{eventos} \omterm{def:b2:proba:independence}{independentes} converge para o
Poisson \omterm{def:b2:randomvar:law}{lei} de parâmetro $\lambda $.
\end{theorem}

\begin{proof}
Computação direta com $ p_n = \frac{\lambda_n}{n}$, $\lambda_n
\to \lambda $:
\[
\P(X_n = k)
= \binom nk p_n^k(1 - p_n)^{n-k}
= \frac{n(n-1)\cdots(n-k+1)}{n^k}\cdot
\frac{\lambda_n^k}{k!}\,
\bigl(1 - \tfrac{\lambda_n}{n}\bigr)^{n-k} .
\]
Como $ n \to \infty $ com $ k $ fixo: o primeiro fator tende a $1$
(produto de $ k $ fatores $\to 1$); $\lambda_n^k \to \lambda^k $;
e $\bigl(1 - \frac{\lambda_n}{n}\bigr)^{n-k} =
\exp\bigl((n-k)\ln(1 - \frac{\lambda_n}{n})\bigr) \to
e^{-\lambda}$ desde $(n - k)\ln\bigl(1 - \frac{\lambda_n}{n}\bigr)
\sim -\lambda_n \to -\lambda $
(\cref{ch:b2:comparison}). Alternativamente, ao nível
gerando funções: $ G_{X_n}(t) = \bigl(1 +
\frac{\lambda_n(t-1)}{n}\bigr)^n \to e^{\lambda(t - 1)} =
G_{\mathcal{P}(\lambda)}(t)$ para cada fixo $ t \in [0, 1]$ ---
convergência de gerando funções, que (para $\N $-valorizado
variáveis) é equivalente à convergência de cada $\P(X_n = k)$;
ver \cref{exo:b2:genfun:9}.
\end{proof}

\begin{remark}
É por isso que Poisson \omterm{def:b2:randomvar:law}{leis} contagens de modelos raros \omterm{def:b2:proba:space}{eventos} --- erros de digitação por
página, decaimentos radioativos por segundo, acidentes por dia em uma
cruzamento: cada oportunidade é quase insignificante, as oportunidades são
muitos, e apenas a taxa média $\lambda $ sobrevive no limite.
\end{remark}

\begin{example}[Observando o limite de Poisson
convergir]\label{ex:b2:genfun:poissonnumerics}
Consertar $\lambda = 2$ e deixe $ X_n \sim \mathcal B(n, 2/n)$. O
sem evento probabilidade é exatamente $\P(X_n = 0) = (1 - 2/n)^n $:
\[
n = 10:\ 0.107, \qquad
n = 20:\ 0.122, \qquad
n = 50:\ 0.130, \qquad
n = 100:\ 0.133,
\]
contra o limite $\eu^{-2} \approx 0.135$. A convergência
é monótono e de velocidade $ O(1/n)$ --- expandindo, $(1 -
2/n)^n = \eu^{-2}\bigl(1 - \tfrac2n + O(n^{-2})\bigr)$ ---
então para $ n $ na casa das centenas, o modelo de Poisson já está
com precisão até o terceiro dígito. Este é o conteúdo prático
da \omterm{def:b2:randomvar:law}{lei} dos \omterm{def:b2:proba:space}{eventos} raros: o modelador nunca sabe $ n $ e
$ p $ separadamente (quantas micro-oportunidades para um erro de digitação
uma retenção de página?), mas apenas o produto $\lambda $, e o
limite \omterm{def:b2:randomvar:law}{lei} misericordiosamente não depende de mais nada.
\end{example}

\section{Processos de ramificação}

Considere uma população partindo de um ancestral; cada indivíduo,
independentemente, tem um número aleatório de filhos com \omterm{def:b2:randomvar:law}{lei} $(p_k)_{k
\in \N}$ e \omterm{ex:b2:powerseries:fibonacci}{função geradora} $ G $ (o \emph{filhos
\omterm{def:b2:randomvar:law}{distribuição}}). Deixar $ Z_n $ ser do tamanho da geração $ n $ ($ Z_0 =
1$) e deixe $ m = G'(1) = \E(Z_1)$ seja o número médio de descendentes.

\begin{proposition}\label{prop:b2:genfun:branching}
O \omterm{ex:b2:powerseries:fibonacci}{função geradora} de $ Z_n $ é o $ n $-ésima iteração
$ G_{Z_n} = G \circ G \circ \dots \circ G $ ($ n $ vezes) e o
probabilidades de extinção $ q_n = \P(Z_n = 0)$ satisfazer
\[
q_0 = 0, \qquad q_{n+1} = G(q_n),
\]
e aumentar para a probabilidade $ q $ de eventual extinção, que
é um ponto fixo de $ G $.
\end{proposition}

\begin{proof}
Geração $ n + 1$ é a soma aleatória dos descendentes do $ Z_n $
membros da geração $ n $, com contagens \omterm{def:b2:proba:independence}{independente} um do outro
e de $ Z_n $: \cref{thm:b2:genfun:compound} dá $ G_{Z_{n+1}} =
G_{Z_n} \circ G $, e indução de $ G_{Z_0}(t) = t $ produz o
$ n $-fold iterate --- que, por associatividade de composição, pode
igualmente ser lido como $ G_{Z_{n+1}} = G \circ G_{Z_n}$. Avaliando
esta segunda forma em $0$: $ q_{n+1} = G_{Z_{n+1}}(0) =
G\bigl(G_{Z_n}(0)\bigr) = G(q_n)$. O \omterm{def:b2:proba:space}{eventos} $\{Z_n = 0\}$ aumentar (populações extintas permanecem
extinto), então $ q_n \uparrow q = \P\bigl(\bigcup_n\{Z_n = 0\}\bigr)$
por monótono continuidade (\cref{thm:b2:proba:continuity}), e
continuidade de $ G $ sobre $[0, 1]$ voltas $ q_{n+1} = G(q_n)$ em $ q =
G(q)$ no limite.
\end{proof}

\begin{example}[Assistindo a extinção convergir]
\label{ex:b2:genfun:cobwebnumerics}
Para a prole \omterm{def:b2:randomvar:law}{lei} $(p_0, p_1, p_2) = (\tfrac14,
\tfrac14, \tfrac12)$ de \cref{ex:b2:genfun:branchingexample},
$ G(t) = \tfrac14 + \tfrac14t + \tfrac12t^2$ e a iteração
$ q_{n+1} = G(q_n)$ dá
\[
q_1 = 0.25, \quad q_2 = 0.34375, \quad q_3 \approx 0.39502,
\quad q_4 \approx 0.42678, \quad q_5 \approx 0.44776,
\]
subindo em direção à probabilidade de extinção $ q = \tfrac12$.
As lacunas $ q - q_n $ são $0.25$, $0.156$, $0.105$, $0.073$,
$0.052$: cada um é aproximadamente $\tfrac34$ do anterior, e
na verdade, o teorema do valor médio dá $ q - q_{n+1} = G'(c_n)(q
- q_n)$ com $ G'(q) = \tfrac14 + q = \tfrac34$. Duas morais: uma
linhagem familiar ainda viva na geração $ n $ tem, embutido
o mesmo cálculo, probabilidade $ q - q_n $ de estar condenado
mais tarde; e a taxa de convergência da escada no
a figura abaixo é a derivada no ponto fixo --- o
O problema do fim de semana transforma ambas as observações em teoremas.
\end{example}

\begin{theorem}[Critério de extinção]\label{thm:b2:genfun:extinction}
Assumir $ p_1 \neq 1$. A probabilidade de extinção $ q $ é o
\emph{menor} ponto fixo de $ G $ em $\intcc{0}{1}$, e:
\begin{itemize}
\item se $ m \leq 1$ (subcrítico ou crítico), $ q = 1$: extinção
é certo;
\item se $ m > 1$ (supercrítico), $ q < 1$: a população sobrevive
para sempre com probabilidade positiva $1 - q $.
\end{itemize}
\end{theorem}

\begin{proof}
$ G $ é \omterm{def:b2:affine:convex}{convexo} sobre $\intcc{0}{1}$ (série de potências com não negativo
coeficientes: $ G'' \geq 0$), não decrescente, com $ G(1) = 1$.

\emph{Menor ponto fixo:} deixar $ r \in \intcc{0}{1}$ ser qualquer fixo
ponto. Então $ q_0 = 0 \leq r $, e indutivamente $ q_{n+1} = G(q_n)
\leq G(r) = r $ (monotonicidade): então $ q = \lim q_n \leq r $.

\emph{Caso $ m \leq 1$:} suponha $ r < 1$ é um ponto fixo. Pelo
teorema do valor médio em $[r, 1]$, há $ c \in \intoo{r}{1}$ com
$ G'(c) = \frac{G(1) - G(r)}{1 - r} = \frac{1 - r}{1 - r} = 1$. Mas
$ G'$ é não decrescente (convexidade) com $\lim_{t\to1^-}G'(t) = m
\leq 1$, então $ G' \leq 1$ sobre $\intoo{0}{1}$; a igualdade $ G'(c) =
1$ então forças $ G'$ ser constante igual a $1$ sobre $\intco{c}{1}$,
por isso $ G'' = \sum n(n-1)p_nt^{n-2} \equiv 0$ lá. Uma série de potências
com coeficientes não negativos desaparecendo em um intervalo tem tudo
esses coeficientes são zero: $ p_n = 0$ para $ n \geq 2$, então $ G(t) = p_0
+ p_1t $ e $1 = G'(c) = p_1$ --- contradizendo a hipótese
$ p_1 \neq 1$. Então $1$ é o único ponto fixo: $ q = 1$.

\emph{Caso $ m > 1$:} aproximar $1$, $ G(t) - t $ tem derivada $ G'(t) -
1 \to m - 1 > 0$ como $ t \to 1^-$, então $ G(t) - t < G(1) - 1 = 0$ ligado
algum intervalo $\intoo{1 - \delta}{1}$: o \omterm{def:b2:metric:continuity}{contínuo} função
$ G(t) - t $ é $\geq 0$ no $ t = 0$ ($ G(0) = p_0 \geq 0$) e $< 0$
logo abaixo $1$, então ele desaparece em algum $ r < 1$ (valor intermediário
teorema). O menor ponto fixo é então $ q \leq r < 1$.
\end{proof}

\begin{remark}[Como ler a teia de aranha]
Na figura, um movimento vertical se aplica $ G $ (de $(q_n, q_n)$
até $(q_n, G(q_n))$), um movimento horizontal para a diagonal
converte saída em entrada: a escada \emph{é} o
recursão $ q_{n+1} = G(q_n)$. Convexidade de $ G $ e $ G(1) = 1$
deixe apenas duas geometrias. Ou a curva permanece acima do
diagonal ligada $\intco01$ (significar $ m \leq 1$): a escada tem
nenhum lugar para parar antes $1$. Ou a curva cruza em algum $ q <
1$ ($ m > 1$): a escada fica presa abaixo do cruzamento
e converge para ele, à taxa geométrica $ G'(q) < 1$
quantificado em \cref{ex:b2:genfun:cobwebnumerics}. Todos os
a análise do teorema da extinção é visível neste
imagem --- por isso vale a pena desenhar antes
computação.
\end{remark}

\begin{example}\label{ex:b2:genfun:branchingexample}
Filhos \omterm{def:b2:randomvar:law}{lei}: nenhum filho, um filho, dois filhos com
probabilidades $\frac14, \frac14, \frac12$. Então $ m = \frac14 + 1 =
\frac54 > 1$ e $ G(t) = \frac14 + \frac14 t + \frac12 t^2$. Corrigido
pontos: $\frac12 t^2 - \frac34 t + \frac14 = 0$, ou seja\ $2t^2 - 3t
+ 1 = (2t - 1)(t - 1) = 0$: $ q = \frac12$. A linhagem familiar morre
fora com probabilidade $\frac12$ --- e com probabilidade $\frac12$
ele vive para sempre.
\end{example}

\end{remark}

\begin{remark}[Armadilhas comuns]
(eu) Gerando funções aplicar para $\N $ Somente variáveis com valor:
para variáveis com sinal ou não inteiras, o objeto $\E(t^X)$
perde sua estrutura de série de potências (o ano 3 a substitui por
transforma adaptada para $\R $). (ii) A primeira verificação de sanidade de
qualquer computado $ G $ é $ G(1) = 1$; a segunda é que
coeficientes não são negativos --- um coeficiente negativo
significa um deslize de álgebra, não um novo \omterm{def:b2:randomvar:law}{lei}. (iii) Em somas aleatórias,
a ordem da composição é importante: $ G_S = G_N \circ G_X $, o
\emph{exterior} função contando os termos; compondo o
outra maneira não tem sentido ($ G_X \circ G_N $ contaria itens
de itens). (iv) Necessidades de multiplicatividade independência e
fontes distintas de aleatoriedade: $ G_{2X}(t) = G_X(t^2)$, não
$ G_X(t)^2$. (v) Diferenciando em $1$ é um limite
operação: quando o raio é exatamente $1$, como em
\cref{ex:b2:genfun:heavytail}, $ G'(1^-)$ pode ser infinito,
e a formulação do limite monótono do teorema dos momentos não é
uma sutileza pedante, mas uma declaração honesta.
\end{remark}

\section*{Fechando o volume}

O \omterm{ex:b2:powerseries:fibonacci}{função geradora} é um objeto final adequado para este livro:
é simultaneamente uma série de potências (\cref{ch:b2:powerseries}), um
ferramenta de resumido famílias (\cref{ch:b2:series}), um \omterm{def:b2:randomvar:expectation}{expectativa}
(\cref{ch:b2:randomvar}), uma função convexa cuja geometria decide
extinção (\cref{ch:b2:realfun}) e uma iteração de ponto fixo
(\cref{ch:b2:metric}). A matemática do Ano 2 é uma disciplina.
O volume do Ano 3 abrirá as portas deliberadamente deixadas fechadas
aqui: Integração Lebesgue (descarregando a convergência dominada

prova completa do teorema (\cref{ch:b2:diffcalc}) na configuração de
geometria \omterm{def:b2:diffcalc:differential}{diferencial}.

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b2:genfun:1}
Calcule o \omterm{ex:b2:powerseries:fibonacci}{função geradora} do uniforme \omterm{def:b2:randomvar:law}{lei} sobre $\{1, 2,
\dots, 6\}$ (um dado justo). Mostre que a soma de dois dados justos
\emph{não pode} ser uniforme em $\{2, \dots, 12\}$: fator $ G_{X+Y}$
e contar raízes. \emph{(Uma soma uniforme forçaria $ G_X(t)G_Y(t) =
\frac{t^2}{11}\sum_{k=0}^{10}t^k $, cujas raízes diferentes de zero são
$11$-as raízes da unidade diferentes de $1$ --- nenhum deles é real ---
enquanto $ G_X/t $ e $ G_Y/t $ são polinômios reais de grau $5$, cada
possuir pelo menos uma raiz real.)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:genfun:2}
Usando gerando funções, recuperar $\E $ e $ V $ para o binômio
e geométrico \omterm{def:b2:randomvar:law}{leis} (\cref{thm:b2:genfun:moments}).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:genfun:3}
Dois dados carregados: é possível carregar dois dados (independentemente,
idênticas ou não) de modo que sua soma seja uniforme em $\{2, \dots,
12\}$? \emph{(Mesma obstrução de fatoração que em
\cref{exo:b2:genfun:1}: a resposta é não, mesmo com diferentes
cargas, porque cada fator $ G_X(t)/t $ tem grau ímpar $5$,
portanto, uma raiz real, enquanto o alvo não tem nenhuma.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:genfun:4}
Deixar $ X_1, X_2, \dots $ ser \omterm{def:b2:proba:independence}{independente} Bernoulli $\mathcal{B}(p)$
e $ N \sim \mathcal{P}(\lambda)$ \omterm{def:b2:proba:independence}{independente} deles. Mostrar,
através de \cref{thm:b2:genfun:compound}, que $ S = X_1 + \dots + X_N
\sim \mathcal{P}(\lambda p)$: um número de Poisson de itens, cada um mantido
com probabilidade $ p $, deixa um número de Poisson --- \emph{desbaste}.
Calcule também o \omterm{def:b2:randomvar:law}{lei} da contagem descartada e admire: é
$\mathcal{P}(\lambda(1-p))$, e pode-se mostrar que é \omterm{def:b2:proba:independence}{independente}
de $ S $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:genfun:5}
(Binômio negativo) Deixe $ T_r $ seja o número de lançamentos para obter
$ r $ cara (probabilidade de cabeça $ p $). Escrever $ T_r $ como uma soma de $ r $
\omterm{def:b2:proba:independence}{independente} variáveis geométricas, deduza
\[
G_{T_r}(t) = \Bigl(\frac{pt}{1 - (1-p)t}\Bigr)^{r},
\qquad
\E(T_r) = \frac rp,
\qquad
V(T_r) = \frac{r(1-p)}{p^2},
\]
e expandir $ G_{T_r}$ encontrar $\P(T_r = n) = \binom{n-1}{r-1}
p^r(1-p)^{n-r}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:genfun:6}
Para a prole \omterm{def:b2:randomvar:law}{lei} $ p_0 = \frac18$, $ p_1 = \frac38$, $ p_2 =
\frac38$, $ p_3 = \frac18$: calcular $ m $, decidir
supercriticalidade e calcular a probabilidade de extinção $ q $
exatamente. \emph{(Fatore a raiz $ t = 1$ de $ G(t) - t $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:genfun:7}
(Progênie total) Em um subcrítico \omterm{pb:b2:genfun:1}{processo de ramificação} ($ m < 1$), deixar
$ Y = \sum_{n\geq0} Z_n $ seja o número total de indivíduos já
nascido. Mostrar $\E(Y) = \sum_n m^n = \frac{1}{1 - m}$ (justifique o
troca de somatórios) e provar que o \omterm{ex:b2:powerseries:fibonacci}{função geradora}
$ H = G_Y $ satisfaz a equação funcional $ H(t) = t\,G(H(t))$.
\emph{(O ancestral, mais o total de progênies de cada um de seus
crianças, que são \omterm{def:b2:proba:independence}{independente} cópias de $ Y $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:genfun:8}
Deixar $ X $ ter \omterm{ex:b2:powerseries:fibonacci}{função geradora} $ G $ com \omterm{def:b2:powerseries:radius}{raio de convergência} $>
1$. Prove o \emph{cauda exponencial ligada}: há $ C > 0$ e
$\rho \in \intoo{0}{1}$ com $\P(X \geq n) \leq C\rho^n $.
\emph{(Markov se candidatou a $ t^X $ para um fixo $ t > 1$ dentro do
disco.)} Inversamente, mostre que se $\P(X \geq n) \leq C\rho^n $ com
$\rho < 1$, o raio de $ G $ é $\geq 1/\rho > 1$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:genfun:9}
(Continuidade teorema, caso elementar) Deixe $ X, X_1, X_2, \dots $ ser
$\N $-valorizado com $ G_{X_n}(t) \to G_X(t)$ para cada $ t \in
\intco{0}{1}$. Mostre isso $\P(X_n = k) \to \P(X = k)$ para cada
$ k $. \emph{(Indução em $ k $: para $ k = 0$ pegar $ t \to 0$ ---
com cuidado: consertar $ t $ pequeno, use $\abs{\P(X_n = 0) - G_{X_n}(t)}
\leq \frac{t}{1-t}$, válido desde a cauda $\sum_{j \geq 1}p_jt^j
\leq \frac{t}{1 - t}$; então diagonalize. Para a etapa de indução,
considere $\frac{G(t) - \P(X = 0)}{t}$, o \omterm{ex:b2:powerseries:fibonacci}{função geradora} de
um deslocado \omterm{def:b2:randomvar:law}{lei}.)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:genfun:10}
(Truque de paridade) Mostre que para um $\N $ variável com valor $ X $,
\[
\P(X \text{ even}) = \frac{1 + G_X(-1)}{2} ,
\]
e calcule essa probabilidade para $ X \sim \mathcal P(\lambda)$
e $ X \sim \mathcal B(n, p)$. O que faz $ G_X(-1) \to 0$ significa
probabilisticamente?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:genfun:11}
(Dados de Sicherman) Verifique a fatoração do dado justo
\omterm{ex:b2:powerseries:fibonacci}{função geradora}
\[
\frac{t + t^2 + \dots + t^6}{6}
= \frac{t\,(1 + t)(1 + t + t^2)(1 - t + t^2)}{6},
\]
e mostre que os dois dados com faces $\{1, 2, 2, 3, 3, 4\}$
e $\{1, 3, 4, 5, 6, 8\}$ ter gerando funções
$\frac{t(1+t)(1+t+t^2)}6$ e
$\frac{t(1+t)(1+t+t^2)(1-t+t^2)^2}6$, cujo produto é o de
dois dados padrão: esses dados exóticos produzem cada total $2,
\dots, 12$ exatamente com as probabilidades padrão.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:genfun:12}
(Esperando por duas caras seguidas) Uma moeda com probabilidade de cara
$ p $ é lançado até que apareçam duas caras consecutivas; deixar $ T $ ser
o número de lançamentos (o jogo de \cref{exo:b2:proba:6}).
Condicionado nos primeiros lançamentos, deriva um sistema linear para
o gerando funções dos estados ``sem chefe atual''
e ``uma cabeça atual'', e concluir
\[
G_T(t) = \frac{p^2t^2}{1 - qt - pqt^2}
\qquad (q = 1 - p);
\]
verificar $ G_T(1) = 1$ e $\E(T) = \dfrac{1 + p}{p^2}$ ($= 6$
por uma moeda justa).
\end{exercise}

\section{Problema: o processo Galton-Watson, resolvido}

\begin{problem}[{Problema de fim de semana --- taxas de crescimento, exatas
soluções, descendência total e a crítica de Kolmogorov
estimativa}]\label{pb:b2:genfun:1}
O critério de extinção (\cref{thm:b2:genfun:extinction})
divide processos de ramificação\index{processo de ramificação} em
subcrítico, crítico e supercrítico --- mas diz
nada sobre \emph{taxas}: quão rápido uma linha condenada morre, quão rápido
grande, um sobrevivente cresce. Este problema os calcula. Nós mantemos
a notação do capítulo: descendência \omterm{def:b2:randomvar:law}{lei} $(p_k)$ com pgf $ G $,
significar $ m = G'(1)$, tamanhos de geração $ Z_n $ ($ Z_0 = 1$), itera
$ G_n = G_{Z_n}$, probabilidades de extinção $ q_n = \P(Z_n = 0)
\uparrow q $; nós sempre assumimos $ p_1 \neq 1$ e, onde segundo
momentos aparecem, $ G''(1) < \infty $, e escrevemos $\sigma^2 =
V(Z_1)$.

\medskip
\noindent\textbf{Part I --- Moments of the generations.}
\begin{enumerate}
  \item Mostrar $\E(Z_n) = m^n $ \emph{(regra da cadeia em $ G_n = G
        \circ G_{n-1}$ no $1^-$, usando $ G_{n-1}(1) = 1$ e
        \cref{thm:b2:genfun:moments})}.
  \item Estabeleça a recursão $ G_n''(1) =
        G''(1)\,m^{2(n-1)} + m\,G_{n-1}''(1)$ e resolva:
        $ G_n''(1) = G''(1)\,m^{n-1}\dfrac{m^n - 1}{m - 1}$
        para $ m \neq 1$, e $ G_n''(1) = n\,G''(1)$ para $ m =
        1$.
  \item Deduzir
        \[
        V(Z_n) = \sigma^2m^{n-1}\,\frac{m^n - 1}{m - 1}
        \quad (m \neq 1),
        \qquad
        V(Z_n) = n\,\sigma^2 \quad (m = 1).
        \]
  \item (Taxa subcrítica, limite superior) Para $ m < 1$, mostrar
        $\P(Z_n > 0) \leq m^n $ \emph{(Markov no
        com valor inteiro $ Z_n $)}: a extinção é certa com um
        taxa geométrica --- um refinamento quantitativo da
        critério do capítulo.
  \item (Taxa subcrítica, limite inferior) Usando
        Cauchy-Schwarz em $ Z_n\mathbf 1_{Z_n > 0}$, mostrar
        \[
        \P(Z_n > 0) \geq \frac{\E(Z_n)^2}{\E(Z_n^2)}
        \geq c\,m^{n}
        \quad\text{with}\quad
        c = \Bigl(\frac{\sigma^2}{m(1-m)} + 1\Bigr)^{-1} :
        \]
        a taxa geométrica $ m^n $ é exato até constantes.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The geometric family, solved
exactly.} Deixe a prole \omterm{def:b2:randomvar:law}{lei} ser geométrico em $\N $: $ p_k =
qp^k $ ($ k \geq 0$), com $0 < p < 1$, $ q = 1 - p $.
\begin{enumerate}[resume]
  \item Calcular $ G(t) = \dfrac{q}{1 - pt}$ e $ m = \dfrac
        pq $; localizar os três regimes em termos de $ p $.
  \item Resolver $ G(t) = t $: mostra que os pontos fixos são $1$ e
        $ q/p = 1/m $, e recuperar a probabilidade de extinção
        $ q_{\mathrm{ext}} = \min(1, 1/m)$.
  \item Prove por indução as formas fechadas
        \[
        q_n = \frac{m^n - 1}{m^{n+1} - 1} \quad (m \neq 1),
        \qquad
        q_n = \frac{n}{n+1} \quad (m = 1).
        \]
  \item Deduza as taxas exatas: $1 - q_n \sim (1 - m)\,m^n $
        no caso subcrítico, e $ q_{\mathrm{ext}} - q_n
        \sim \dfrac{m - 1}{m^{2}}\cdot m^{-n}$ no
        caso supercrítico; verifique se o supercrítico
        razão de contração é $ G'(q_{\mathrm{ext}}) = 1/m $.
  \item Caso crítico ($ p = \tfrac12$): calcular $\sigma^2 =
        2$ e observe $1 - q_n = \frac1{n+1}$: a sobrevivência decai
        gosto $\frac1n $ --- nem geométrico nem resumido.
  \item Ainda crítico: prove por indução a iteração completa
        \[
        G_n(t) = \frac{n - (n-1)t}{n + 1 - nt},
        \]
        e deduzir isso condicionado à sobrevivência, $ Z_n $ é
        geométrico ligado $\N^*$ com parâmetro $\frac1{n+1}$:
        \[
        \P(Z_n = k \mid Z_n > 0) = \frac1{n+1}
        \Bigl(\frac{n}{n+1}\Bigr)^{k-1},
        \qquad
        \E(Z_n \mid Z_n > 0) = n + 1 .
        \]
        A linha média morre, mas as linhas sobreviventes
        tamanho do pedido $ n $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Total progeny.} Deixar $ Y =
\sum_{n\geq0}Z_n \in \N^* \cup \{\infty\}$ seja o número total
de indivíduos já nascidos, e $ H(t) =
\sum_{k\geq1}\P(Y = k)t^k $.
\begin{enumerate}[resume]
  \item Justificar $\P(Y < \infty) = q_{\mathrm{ext}}$ e
        lembrar de \cref{exo:b2:genfun:7} o funcional
        equação $ H(t) = t\,G(H(t))$ (cuja derivação fez
        não usar $ m < 1$).
  \item (Ramificação binária) Para $ p_0 = p_2 = \frac12$
        (crítico), resolva a equação funcional:
        \[
        H(t) = \frac{1 - \sqrt{1 - t^2}}{t},
        \]
        e expandir com \cref{ex:b2:powerseries:catalan} obter
        \[
        \P(Y = 2k + 1) = \frac{C_k}{2^{2k+1}},
        \qquad C_k = \frac1{k+1}\binom{2k}k ;
        \]
        confira os valores $\P(Y = 1) = \frac12$ e $\P(Y = 3)
        = \frac18$ por contagem direta.
  \item Diferenciando a equação funcional em $1^-$,
        mostre isso $\E(Y) = \frac{1}{1-m}$ para $ m < 1$, enquanto
        forças de criticidade $\E(Y) = \infty $: o crítico
        a progênie total é finita quase certamente com infinito
        significa.
  \item Com o binômio central assintótico
        (\cref{ex:b2:comparison:centralbinomial}), mostrar
        \[
        \P(Y = 2k+1) \sim \frac{1}{2\sqrt\pi\,k^{3/2}},
        \]
        um pesado $ k^{-3/2}$ cauda e deduzir $\P(Y > n)
        \asymp n^{-1/2}$ (limites superior e inferior deste
        ordem é suficiente).
  \item Compare com a feira passeio aleatório (o fim de semana
        problema de \cref{ch:b2:proba}): certo, mas
        tempos de retorno médios infinitos lá, certos, mas
        progênie total de média infinita aqui, ambos com
        $ n^{-3/2}$ \omterm{def:b2:randomvar:law}{leis} locais. Um parágrafo sobre o porquê
        a criticidade produz essa assinatura.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Kolmogorov's estimate at
criticality.} Assumir $ m = 1$, $0 < \sigma^2 = G''(1) <
\infty $.
\begin{enumerate}[resume]
  \item Mostre isso $ G''$ estende continuamente para $\intcc01$
        \emph{(aumentando não negativo com limite finito)} e
        deduzir a expansão de Taylor em $1$:
        \[
        G(t) = t + b\,(1-t)^2 + o\bigl((1-t)^2\bigr),
        \qquad b = \frac{G''(1)}2 = \frac{\sigma^2}2 .
        \]
  \item Para $ t \in \intco01$, definir $ h(t) = \dfrac1{1 - G(t)} -
        \dfrac1{1 - t}$. Mostrar
        \[
        h(t) = \frac{G(t) - t}{(1 - G(t))(1 - t)}
        \xrightarrow[t\to1^-]{} b .
        \]
  \item Telescópio ao longo da iteração $ q_{j+1} = G(q_j)$:
        \[
        \frac1{1 - q_n} = 1 + \sum_{j=0}^{n-1}h(q_j),
        \]
        e concluir com um argumento de Cesaro que
        \[
        \P(Z_n > 0) = 1 - q_n \sim \frac{2}{\sigma^2\,n}
        \]
        --- \emph{Estimativa de Kolmogorov}: cada crítico
        \omterm{pb:b2:genfun:1}{processo de ramificação} morre à taxa universal $1/n $,
        com apenas a lembrança constante da prole \omterm{def:b2:randomvar:law}{lei}.
  \item Verifique a estimativa em relação à geometria crítica
        caso da questão 10.
  \item Deduzir $\E(Z_n \mid Z_n > 0) = \dfrac{1}{1 - q_n}
        \sim \dfrac{\sigma^2 n}{2}$ \emph{(observação $\E(Z_n
        \mathbf 1_{Z_n>0}) = \E(Z_n) = 1$)}e verifique
        contra a questão 11: condicionada à sobrevivência, o
        população cresce \emph{linearmente} --- o crítico
        corda bamba entre a morte e a explosão.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Applications and synthesis.}
\begin{enumerate}[resume]
  \item (Epidemias, reações em cadeia) Para uma prole de Poisson
        \omterm{def:b2:randomvar:law}{lei} $\mathcal P(\lambda)$ --- cada caso infecta
        $\mathcal P(\lambda)$ novos casos --- escreva o
        equação de extinção $ q = \eu^{\lambda(q-1)}$ e
        resolva numericamente para $\lambda = 1.5$ ($ q \approx
        0.417$) e $\lambda = 2$ ($ q \approx 0.203$):
        a partir de um caso, um grande surto é
        \emph{não} certo mesmo quando $\lambda > 1$. Explique
        por que a iteração $ q_{n+1} = \eu^{\lambda(q_n - 1)}$
        de $ q_0 = 0$ converge para a raiz direita.
  \item A partir de $ k $ ancestrais em vez de um, mostre que
        a probabilidade de extinção é $ q^k $. Aplicação:
        com $\lambda = 1.5$, quantos casos iniciais formam um
        surto pelo menos $99\%$ provável?
  \item (Condicionando um processo supercrítico na extinção)
        Para $ m > 1$ com probabilidade de extinção $ q \in
        \intoo01$: primeiro prove por convexidade que $ G'(q) < 1$
        no menor ponto fixo e deduza
        $ q_{\mathrm{ext}} - q_n = O\bigl(G'(q)^n\bigr)$
        (convergência geométrica, conforme exemplificou a questão 9).
        Então mostre isso $\widehat G(t) = G(qt)/q $ é o pgf
        de uma prole genuína \omterm{def:b2:randomvar:law}{lei}, com média $\widehat m =
        G'(q) < 1$: um processo companheiro subcrítico. Verifique
        na família geométrica: condicionando o
        supercrítico $(p, q)$-processo em swaps de extinção
        $ p $ e $ q $. (A declaração completa --- o condicionado
        processo \emph{é} o processo companheiro --- está provado
        no volume do Ano 3; aqui você verificou seu
        sombra da \omterm{ex:b2:powerseries:fibonacci}{função geradora}.)
  \item Síntese: elaborar a tabela de tricotomia --- para $ m <
        1$, $ m = 1$, $ m > 1$: valor de $ q $; taxa de $\P(Z_n >
        0)$ ou de $ q - q_n $; $\E(Y)$; tamanho de um sobrevivente
        geração. Indique em uma frase por ferramenta como
        composição de pgfs, convexidade, Taylor em $1^-$ e
        A média de Cesaro carregava todo o problema, e o que
        o volume do Ano 3 acrescenta (o martingale $ Z_n/m^n$ e
        Limite exponencial de Yaglom \omterm{def:b2:randomvar:law}{lei}).
\end{enumerate}
\end{problem}
