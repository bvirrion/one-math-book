\chapter{Sequences and Series of Functions}\label{ch:b2:funcseq}

When functions converge to a function, which properties survive the
passage to the limit? Pointwise convergence preserves almost
nothing; \emph{uniform} convergence --- convergence in the sup norm
--- preserves continuity, integrals on segments and, with one twist,
derivatives. This chapter proves the three transfer theorems,
their series versions, and crowns them with the Weierstrass
approximation theorem, proved by Bernstein's beautiful probabilistic
polynomials.

\section{Pointwise and uniform convergence}

\begin{definition}\label{def:b2:funcseq:def}
Let $f_n, f \colon X \to \R$ (or $\C$, or a normed space), $X$ any
set. $(f_n)$ converges to $f$
\emph{pointwise}\index{pointwise convergence} when $f_n(x) \to
f(x)$ for every $x$; \emph{uniformly}\index{uniform convergence}
when
\[
\norm{f_n - f}_\infty = \sup_{x \in X}\, \abs{f_n(x) - f(x)}
\xrightarrow[n \to \infty]{} 0 .
\]
Uniform implies pointwise; on $C(\intcc{a}{b})$, uniform convergence
is exactly convergence in the Banach space
$\bigl(C(\intcc{a}{b}), \norm\cdot_\infty\bigr)$ of
\cref{ch:b2:nvs}.
\end{definition}

\begin{example}\label{ex:b2:funcseq:xn}
On $\intcc{0}{1}$, $f_n(x) = x^n$ converges pointwise to the
\emph{discontinuous} limit $f = \mathbf{1}_{\{1\}}$; the convergence
is not uniform: $\norm{f_n - f}_\infty \geq f_n\bigl(1 -
\tfrac1n\bigr) = (1 - \tfrac1n)^n \to \eu^{-1} \neq 0$. On
$\intcc{0}{a}$ with $a < 1$ it \emph{is} uniform ($\sup = a^n \to
0$): uniformity is a property of the domain as much as of the
sequence.
\end{example}

\begin{example}[Two limits that refuse to commute]\label{ex:b2:funcseq:doublelimit}
The whole chapter is about interchanging limits, so here is the
smallest possible failure. Let $a_{n,m} = \dfrac{n}{n+m}$ for
$n, m \geq 1$. Then
\[
\lim_{m\to\infty}\Bigl(\lim_{n\to\infty}a_{n,m}\Bigr)
= \lim_{m\to\infty} 1 = 1,
\qquad
\lim_{n\to\infty}\Bigl(\lim_{m\to\infty}a_{n,m}\Bigr)
= \lim_{n\to\infty} 0 = 0 :
\]
both iterated limits exist and they differ. Every transfer
theorem of this chapter is a licence to commute two limits ---
$\lim_n$ with $\lim_{x\to a}$ (continuity), with $\int$
(integration), with $\frac{\dd}{\dd x}$ (differentiation) ---
and uniform convergence is precisely the fee that makes the
commutation legal. Closing insight: whenever a ``proof''
silently swaps two limit operations, this two-line array is
the counterexample to hold against it; the sliding bumps of
\cref{exo:b2:funcseq:2} are the same phenomenon wearing an
integral sign.
\end{example}

\section{The three transfer theorems}

\begin{theorem}[Continuity]\label{thm:b2:funcseq:continuity}
If each $f_n$ is continuous at $a$ and $f_n \to f$ uniformly on a
neighborhood of $a$, then $f$ is continuous at $a$. A uniform limit
of continuous functions is continuous.
\end{theorem}

\begin{proof}
The $3\varepsilon$ argument already used in
\cref{thm:b2:metric:rncomplete}: pick $n$ with $\norm{f_n -
f}_\infty \leq \varepsilon$, then $\delta$ from continuity of
$f_n$ at $a$; for $\abs{x - a} \leq \delta$,
\[
\abs{f(x) - f(a)} \leq \abs{f(x) - f_n(x)} + \abs{f_n(x) - f_n(a)}
+ \abs{f_n(a) - f(a)} \leq 3\varepsilon . \qedhere
\]
\end{proof}

\begin{example}[Uniformity fails exactly where the limit breaks]\label{ex:b2:funcseq:threshold}
On $\intcc{0}{2}$, let $f_n(x) = \dfrac{x^n}{1 + x^n}$. The
pointwise limit is a three-piece function:
\[
f(x) = \begin{cases} 0 & 0 \leq x < 1,\\[2pt]
\tfrac12 & x = 1,\\[2pt]
1 & 1 < x \leq 2, \end{cases}
\]
discontinuous at $1$, so by \cref{thm:b2:funcseq:continuity}
the convergence cannot be uniform on $\intcc{0}{2}$. On the
closed pieces avoiding the threshold it is: for $0 \leq x \leq
a < 1$,
\[
\sup_{\intcc{0}{a}}\abs{f_n - 0}
= \frac{a^n}{1 + a^n} \leq a^n \to 0 ,
\]
and for $1 < b \leq x \leq 2$,
\[
\sup_{\intcc{b}{2}}\abs{f_n - 1}
= \frac{1}{1 + b^n} \leq b^{-n} \to 0 ,
\]
both suprema computed by monotonicity of $u \mapsto
\frac{u}{1+u}$ and of $x \mapsto x^n$. Closing insight: the
failure of uniformity is localized at the discontinuity of the
limit --- the same geometry as \cref{ex:b2:funcseq:xn}, and the
reason the ``uniform on every segment inside'' discipline
recurs all chapter long.
\end{example}

\begin{theorem}[Integration on a segment]\label{thm:b2:funcseq:integration}
If $f_n \to f$ uniformly on $\intcc{a}{b}$, with $f_n$ piecewise
continuous ($f$ likewise), then
\[
\int_a^b f_n \longrightarrow \int_a^b f .
\]
\end{theorem}

\begin{proof}
Linearity and the triangle inequality for integrals give
\[
\Bigl|\int_a^b f_n - \int_a^b f\Bigr|
= \Bigl|\int_a^b (f_n - f)\Bigr|
\leq \int_a^b\abs{f_n - f}
\leq (b - a)\,\norm{f_n - f}_\infty \longrightarrow 0 .
\]
The length factor $(b - a)$ is where compactness of the segment
enters: on non-compact intervals the same estimate produces the
useless bound $\infty\cdot0$, and the conclusion genuinely
fails without domination --- the flat bumps $f_n =
\frac1n\mathbf 1_{\intcc{0}{n}}$ converge uniformly to $0$ on
$\intco{0}{\infty}$ yet keep $\int f_n = 1$ (pitfalls remark
below), and the sliding bumps of \cref{ch:b2:integration}'s
remark do the same with pointwise convergence; uniform
convergence controls heights, never widths.
\end{proof}

\begin{theorem}[Differentiation]\label{thm:b2:funcseq:differentiation}
Let $f_n$ be $C^1$ on an interval $I$, with: $(f_n')$ converging
\emph{uniformly} on $I$ (or on every segment of $I$) to some $g$,
and $(f_n(x_0))$ converging at one point $x_0$. Then $(f_n)$
converges (uniformly on segments) to a $C^1$ function $f$, and $f'
= g$: one may differentiate the limit.
\end{theorem}

\begin{proof}
Define $f(x) = \lim f_n(x_0) + \int_{x_0}^x g$: legitimate, $g$
being continuous --- indeed $g$ is the \emph{uniform} limit on
segments of the continuous $f_n'$, so
\cref{thm:b2:funcseq:continuity} applies, and the integral of
a continuous function is well defined with, by the fundamental
theorem of calculus,
\[
f'(x) = g(x) \qquad (x \in I) :
\]
the candidate limit is $C^1$ with the right derivative
\emph{by construction}, before any convergence is proved. By
the fundamental theorem again, $f_n(x) = f_n(x_0) + \int_{x_0}^x
f_n'$; subtracting,
\[
\abs{f_n(x) - f(x)} \leq \abs{f_n(x_0) - \lim f_n(x_0)}
+ \abs{x - x_0}\,\norm{f_n' - g}_{\infty} ,
\]
which tends to $0$ uniformly on every segment. And $f$ is $C^1$
with $f' = g$ by construction.
\end{proof}

\begin{example}[Why the hypothesis sits on the derivatives]\label{ex:b2:funcseq:sqrtabs}
Let $F_n(x) = \sqrt{x^2 + \frac1n}$ on $\R$. Each $F_n$ is
$C^1$ (indeed $C^\infty$), and the convergence to $\abs x$ is
uniform on all of $\R$:
\[
0 \leq F_n(x) - \abs x
= \frac{(x^2 + \frac1n) - x^2}{\sqrt{x^2+\frac1n} + \abs x}
= \frac{1/n}{\sqrt{x^2 + \frac1n} + \abs x}
\leq \frac{1/n}{1/\sqrt n} = \frac{1}{\sqrt n} .
\]
Yet the limit $\abs x$ is not differentiable at $0$: uniform
convergence of the \emph{functions}, however fast, transfers no
differentiability. The failure is visible on the derivatives:
\[
F_n'(x) = \frac{x}{\sqrt{x^2 + \frac1n}}
\longrightarrow \begin{cases} 1 & x > 0,\\ 0 & x = 0,\\
-1 & x < 0, \end{cases}
\]
a discontinuous pointwise limit, so $(F_n')$ cannot converge
uniformly near $0$ (\cref{thm:b2:funcseq:continuity} again).
Closing insight:
\cref{thm:b2:funcseq:differentiation} deliberately assumes
uniform convergence of the $f_n'$, not of the $f_n$ ---
this example is the reason.
\end{example}

\section{Series of functions}

\begin{definition}\label{def:b2:funcseq:series}
A series of functions $\sum u_n$ converges pointwise/uniformly when
its partial sums do. It converges
\emph{normally}\index{normal convergence} (on $X$) when $\sum
\norm{u_n}_\infty < \infty$. Normal convergence implies uniform
convergence (in the Banach space of bounded functions:
\cref{thm:b2:nvs:absoluteconvergence}), which implies pointwise;
both implications are strict.
\end{definition}

\begin{example}[One series, three verdicts]\label{ex:b2:funcseq:threemodes}
Take $u_n(x) = \frac{x^n}{n}$ on $\intco{0}{1}$. \emph{Pointwise:}
converges for every $x \in \intco01$ (comparison with the
geometric series). \emph{Normal on $\intcc{0}{a}$, $a < 1$:}
$\norm{u_n}_{\infty,\intcc0a} = \frac{a^n}{n}$, summable.
\emph{Not normal on $\intco{0}{1}$:}
$\norm{u_n}_{\infty,\intco01} = \frac1n$, and $\sum\frac1n$
diverges. \emph{Not even uniform on $\intco{0}{1}$:} the
remainder resists near $1$,
\[
R_N(x) = \sum_{n>N}\frac{x^n}{n}
\geq \sum_{n=N+1}^{2N}\frac{x^n}{n}
\geq \frac{N\,x^{2N}}{2N} = \frac{x^{2N}}{2}
\xrightarrow[x\to1^-]{} \frac12 ,
\]
so $\sup_{\intco01}\abs{R_N} \geq \frac12$ for every $N$.
Closing insight: all four verdicts coexist peacefully --- the
sum $-\ln(1-x)$ is continuous on $\intco{0}{1}$ because
continuity only needs uniformity \emph{near each point}, i.e.\
on the segments $\intcc0a$; blowing up at the edge is the
sum's right.
\end{example}

\begin{theorem}[Transfer for series]\label{thm:b2:funcseq:seriestransfer}
If $\sum u_n$ converges uniformly (e.g.\ normally) on the relevant
set: continuity of all $u_n$ at $a$ passes to the sum; integration
on a segment can be done term by term; and if $\sum u_n(x_0)$
converges while $\sum u_n'$ converges uniformly on segments, the
sum is $C^1$ with derivative $\sum u_n'$.
\end{theorem}

\begin{proof}
Everything is the corresponding theorem applied to the partial
sums $S_N = \sum_{n\leq N}u_n$, which are finite sums of
functions with the relevant regularity. Continuity: each $S_N$
is continuous at $a$ and $S_N \to \sum u_n$ uniformly:
\cref{thm:b2:funcseq:continuity}. Integration: on the segment,
\[
\int_a^b \sum_{n\geq0} u_n
= \lim_N \int_a^b S_N
= \lim_N \sum_{n=0}^{N}\int_a^b u_n
= \sum_{n\geq0}\int_a^b u_n
\]
by \cref{thm:b2:funcseq:integration} (first equality) and
linearity of the integral (second). Differentiation: the
$S_N$ are $C^1$, $S_N(x_0)$ converges, and $S_N' =
\sum_{n\leq N}u_n'$ converges uniformly on segments:
\cref{thm:b2:funcseq:differentiation} gives that the sum is
$C^1$ with derivative $\lim S_N' = \sum u_n'$.
\end{proof}

\begin{remark}[Common pitfalls]
Four traps, all seen in examination copies. \emph{(i)
Half-checked suprema:} evaluating $f_n$ along a well-chosen
sequence $x_n$ only bounds $\norm{f_n - f}_\infty$ from
\emph{below} --- enough to disprove uniformity (as in
\cref{ex:b2:funcseq:xn}), never to prove it; to prove it, bound
the sup by a computation valid for \emph{all} $x$. \emph{(ii)
Uniformity on the wrong set:} normal or uniform convergence
often holds on every $\intcc{-a}{a}$ or $\intco\delta\infty$
but fails on the open union; that is no obstacle --- continuity
and differentiability are local, so the segment-by-segment
discipline of \cref{ex:b2:funcseq:zeta} gives them on the whole
open set. \emph{(iii) Integrating over non-segments:}
\cref{thm:b2:funcseq:integration} is a statement about
segments; on $\intco0\infty$, uniform convergence does not
prevent mass from escaping to infinity ($f_n =
\frac1n\mathbf 1_{\intcc{0}{n}}$ converges uniformly to $0$,
with $\int f_n = 1$) --- use dominated convergence there.
\emph{(iv) Differentiating the limit:}
\cref{ex:b2:funcseq:sqrtabs}; the derivative hypothesis is on
$(f_n')$, and no rate of convergence of $(f_n)$ can replace
it.
\end{remark}

\begin{example}[The Riemann $\zeta$ function]\label{ex:b2:funcseq:zeta}
$\zeta(s) = \sum_{n\geq1} n^{-s}$ converges normally on every
half-line $\intco{a}{+\infty}$, $a > 1$ ($\norm{n^{-s}}_\infty =
n^{-a}$, summable): $\zeta$ is continuous on $\intoo{1}{+\infty}$;
differentiating termwise (the derived series $\sum -\ln n\; n^{-s}$
also converges normally on $\intco{a}{\infty}$), $\zeta$ is $C^1$
--- and, iterating, $C^\infty$ --- with $\zeta'(s) = -\sum \frac{\ln
n}{n^s}$. Note the discipline: normal convergence is checked on
\emph{sub}-half-lines, never on the open $\intoo{1}{\infty}$
itself, where it fails.
\end{example}

\begin{example}[A logarithmic series, worked to the end]\label{ex:b2:funcseq:logseries}
Let $F(x) = \sum_{n\geq1} \frac{\eu^{-nx}}{n}$ on
$\intoo{0}{\infty}$. Each term is bounded on $\intco{\delta}
\infty$ by $\frac{\eu^{-n\delta}}{n} \leq \eu^{-n\delta}$, a
convergent geometric series: normal convergence on every
$\intco\delta\infty$, so $F$ is continuous on
$\intoo{0}{\infty}$. The derived series $\sum -\eu^{-nx}$ is
likewise normally convergent on $\intco\delta\infty$
($\norm{\eu^{-nx}}_{\infty,\intco\delta\infty} =
\eu^{-n\delta}$), so $F$ is $C^1$ with a geometric derivative:
\[
F'(x) = -\sum_{n\geq1}\eu^{-nx}
= \frac{-\eu^{-x}}{1 - \eu^{-x}}
= \frac{-1}{\eu^{x} - 1} .
\]
Iterating, $F$ is $C^\infty$. Integrating $F'$ (both $F$ and $x
\mapsto -\ln(1 - \eu^{-x})$ vanish at $+\infty$ and have the
same derivative on $\intoo0\infty$):
\[
F(x) = -\ln\bigl(1 - \eu^{-x}\bigr),
\]
the logarithmic series at $t = \eu^{-x}$. Closing insight: as $x
\to 0^+$, $F(x) = -\ln(x + O(x^2)) = \ln\frac1x + O(x)$ --- the
series diverges logarithmically at the boundary, exactly like
the harmonic series it becomes at $x = 0$; normal convergence on
$\intco\delta\infty$ but not on $\intoo0\infty$ is the
symptom.
\end{example}

\begin{method}[Proving or disproving uniform convergence]\label{met:b2:funcseq:uniform}
For $f_n \to f$ pointwise on $X$:
\begin{enumerate}
  \item Compute or bound $\norm{f_n - f}_\infty$: study the
        function $x \mapsto \abs{f_n(x) - f(x)}$ (derivative,
        monotonicity) to locate its maximum; a bound valid for
        all $x$ that tends to $0$ proves uniformity.
  \item To \emph{disprove}: exhibit points $x_n$ with
        $\abs{f_n(x_n) - f(x_n)} \not\to 0$ (often $x_n$ tracks
        the moving bump, as in \cref{exo:b2:funcseq:1}); or
        invoke a transfer theorem in contrapositive --- a
        discontinuous limit of continuous functions
        (\cref{ex:b2:funcseq:threshold}), or $\int f_n
        \not\to \int f$ on a segment.
  \item For series, try normal convergence first
        ($\sum\sup\abs{u_n} < \infty$); if it fails globally,
        test it on the sub-segments that matter
        (\cref{ex:b2:funcseq:threemodes}); if it fails
        everywhere, uniform convergence may still hold via the
        alternating remainder bound
        (\cref{exo:b2:funcseq:4}) or summation by parts.
\end{enumerate}
\end{method}

\section{The Weierstrass approximation theorem}

\begin{theorem}[Weierstrass, via Bernstein]\label{thm:b2:funcseq:weierstrass}
Every continuous $f \colon \intcc{0}{1} \to \R$ is a uniform limit
of polynomials --- explicitly, of its \emph{Bernstein
polynomials}\index{Bernstein polynomials}\index{Weierstrass
approximation theorem}
\[
B_n(f)(x) = \sum_{k=0}^{n} f\Bigl(\frac kn\Bigr)\binom nk x^k
(1-x)^{n-k} .
\]
\end{theorem}

\begin{proof}
Fix $x \in \intcc{0}{1}$ and set $p_k(x) = \binom nk x^k(1 -
x)^{n-k}$. Three binomial identities, obtained by evaluating
$(x + y)^n$ and its two $x$-derivatives at $y = 1 - x$:
\[
\sum_k p_k = 1,
\qquad
\sum_k k\,p_k = nx,
\qquad
\sum_k k(k-1) p_k = n(n-1)x^2 .
\]
In detail: $(x+y)^n = \sum_k\binom nk x^ky^{n-k}$ at $y = 1-x$
is the first; differentiating in $x$,
\[
n(x+y)^{n-1} = \sum_k k\binom nk x^{k-1}y^{n-k} ,
\]
then multiplying by $x$ and setting $y = 1 - x$ gives the
second; differentiating twice and multiplying by $x^2$ gives
the third. Expanding $(k - nx)^2 = k(k-1) + k(1 - 2nx) +
n^2x^2$ and combining the three:
\[
\sum_k (k - nx)^2 p_k
= n(n-1)x^2 + nx(1 - 2nx) + n^2x^2
= nx(1 - x) \leq \frac n4 ,
\]
the \emph{variance identity}.

Now estimate, using $\sum p_k = 1$:
\[
\abs{B_n(f)(x) - f(x)}
\leq \sum_{k} \Bigl| f\Bigl(\frac kn\Bigr) - f(x)\Bigr|\, p_k(x)
= \Sigma_{\text{near}} + \Sigma_{\text{far}} ,
\]
splitting according to $\abs{\frac kn - x} \leq \delta$ or not.
Given $\varepsilon > 0$, uniform continuity of $f$ (Heine) provides
$\delta$ with $\Sigma_{\text{near}} \leq \varepsilon$. For the far
sum, with $M = \norm f_\infty$: by the variance identity and
Chebyshev's counting trick,
\[
\Sigma_{\text{far}} \leq 2M \sum_{\abs{k - nx} > n\delta} p_k
\leq 2M\,\frac{\sum_k (k - nx)^2 p_k}{n^2\delta^2}
\leq \frac{2M}{4 n \delta^2} \cdot 1
= \frac{M}{2n\delta^2}
\xrightarrow[n\to\infty]{} 0 ,
\]
uniformly in $x$. So $\norm{B_n(f) - f}_\infty \leq \varepsilon +
\frac{M}{2n\delta^2} \leq 2\varepsilon$ for large $n$.
\end{proof}

\begin{remark}
By affine substitution the theorem holds on any segment
$\intcc{a}{b}$. It fails on $\R$ (a uniform limit of polynomials on
$\R$ is a polynomial: \cref{exo:b2:funcseq:8}). The probabilistic
reading --- $B_n(f)(x)$ is the expected value of $f$ at a binomial
average, and the variance bound is Chebyshev's inequality --- is
made honest in \cref{ch:b2:genfun}.
\end{remark}

\begin{remark}[Where this is used]
Weierstrass approximation is the density theorem of classical
analysis: it makes $C(\intcc ab)$ separable, lets one check
integral identities on polynomials only (moment problems), and
underlies the trigonometric version proved in the Fourier
chapter via Fej\'er's kernel. The weekend problem of this
chapter extracts the quantitative content of Bernstein's proof
--- convergence rates governed by the modulus of continuity ---
and then isolates what really made it work, in Korovkin's
theorem: positivity plus three test functions. The Year 3
volume generalizes the density statement to arbitrary
subalgebras (Stone--Weierstrass) and to compact spaces.
\end{remark}

\begin{example}[Polygonal approximation, with a rate]\label{ex:b2:funcseq:polygonal}
For $L$-Lipschitz $f$ on $\intcc{0}{1}$, let $I_nf$ be the
piecewise-affine interpolant at the nodes $\frac kn$. On a
cell $\intcc{\frac kn}{\frac{k+1}n}$, both $f(x)$ and
$I_nf(x)$ lie between the extreme values that an $L$-Lipschitz
function can take given the two nodal values, so for $x$ in
the cell, writing $x_k = \frac kn$:
\[
\abs{I_nf(x) - f(x)}
\leq \abs{I_nf(x) - f(x_k)} + \abs{f(x_k) - f(x)}
\leq L\,\abs{x - x_k} + L\,\abs{x - x_k}
\leq \frac{2L}{n}
\]
(the interpolant is itself $L$-Lipschitz on the cell: its
slope is a difference quotient of $f$). Hence $\norm{I_nf -
f}_\infty \leq \frac{2L}{n}$: polygonal approximation of
Lipschitz functions converges at speed $\frac1n$ ---
\emph{faster} than Bernstein's $\frac{1}{\sqrt n}$ for the
same class (weekend problem, Part II). Closing insight: the
polygon interpolates but is not smooth, Bernstein is smooth
but slow; there is no free lunch between regularity of the
approximant and speed --- a trade-off made precise by the
saturation results of the weekend problem.
\end{example}

\begin{remark}[Perspectives within this volume]
Uniform convergence is this book's workhorse from here on. The
power-series chapter runs entirely on normal convergence on
compact subdisks --- every term-by-term theorem there is a
special case of this chapter's transfer theorems. The Fourier
chapter lives one floor up: its partial sums $S_N$ fail
exactly where this chapter warns they might (pointwise but not
uniform at jumps), and its Fej\'er means succeed by the same
$3\varepsilon$ mechanics that proved
\cref{thm:b2:funcseq:continuity}. The differential-equations
chapter defines $\eu^{tA}$ by a normally convergent series and
differentiates it termwise --- literally
\cref{thm:b2:funcseq:seriestransfer} applied to matrix
entries. When in doubt later in the book about ``why can we
do this'', the answer is usually a theorem of this chapter.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b2:funcseq:1}
Study the pointwise and uniform convergence on $\intcc{0}{1}$, then
on $\intcc{0}{a}$ ($a < 1$) or $\intco{\delta}{1}$ as relevant, of:
\[
f_n(x) = \frac{x}{1 + nx},
\qquad
g_n(x) = n x\,\eu^{-n x^2},
\qquad
h_n(x) = x^n(1 - x^n).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:funcseq:2}
Prove that $\displaystyle\int_0^1 g_n \not\to \int_0^1 \lim g_n$
for the $g_n$ of \cref{exo:b2:funcseq:1}, and reconcile with
\cref{thm:b2:funcseq:integration}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:funcseq:3}
Prove that $S(x) = \sum_{n\geq1} \dfrac{x^n}{n^2}$ is continuous on
$\intcc{-1}{1}$, and that $S$ is $C^1$ on $\intoo{-1}{1}$ with
$S'(x) = -\frac{\ln(1-x)}{x}$ for $0 < \abs x < 1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:funcseq:4}
Let $F(x) = \sum_{n \geq 0} \dfrac{(-1)^n}{n + x}$ on
$\intoo{0}{+\infty}$. Prove uniform (not normal) convergence on
$\intco{\delta}{\infty}$ via the alternating series remainder
bound, continuity, and the functional equation $F(x) + F(x + 1) =
\frac1x$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:funcseq:5}
(Dini) Let $f_n \colon K \to \R$ be continuous on a \emph{compact}
metric space, with $f_n \to f$ \emph{pointwise}, $f$ continuous,
and $(f_n(x))$ \emph{decreasing} in $n$ for each $x$. Prove that
the convergence is uniform. \emph{(Given $\varepsilon$, the open
sets $U_n = \{x : f_n(x) - f(x) < \varepsilon\}$ increase and cover
$K$; extract a finite subcover ---
\cref{thm:b2:metric:borellebesgue}.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:funcseq:6}
Prove that $\displaystyle\lim_{n\to\infty} \int_0^1
\frac{n\,f(x)}{1 + n^2x^2}\,\dd x = \frac{\pi}{2} f(0)$ for every
continuous $f$ on $\intcc{0}{1}$. \emph{(Substitute $u = nx$; split
off $f(0)$; dominate.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:funcseq:7}
Compute the Bernstein polynomials of $f(x) = x^2$ explicitly and
verify the uniform error $\norm{B_n f - f}_\infty =
O\bigl(\frac1n\bigr)$ predicted by the proof of
\cref{thm:b2:funcseq:weierstrass} --- here exactly $\frac{x(1 -
x)}{n}$ at each point.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:funcseq:8}
Prove that if polynomials $P_n$ converge uniformly \emph{on all of
$\R$} to $f$, then $f$ is a polynomial. \emph{(For large $m, n$,
$P_n - P_m$ is a bounded polynomial on $\R$, hence constant; so the
sequence stabilizes modulo constants.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:funcseq:9}
(A continuous, nowhere differentiable function --- guided) Let
$\varphi$ be the distance to the nearest integer ($1$-periodic,
$\norm{\varphi}_\infty = \frac12$, $1$-Lipschitz) and
\[
W(x) = \sum_{n=0}^{\infty} \Bigl(\frac{3}{4}\Bigr)^{\!n}
\varphi(4^n x) .
\]
Prove: (a) $W$ is continuous on $\R$ (normal convergence); (b) for
every $x$ and every $m$, choosing $h_m = \pm\frac12\cdot 4^{-m}$
with the sign making $\varphi$ affine on the segment from $4^m x$
to $4^m(x + h_m)$, the difference quotient satisfies
\[
\Bigl|\frac{W(x + h_m) - W(x)}{h_m}\Bigr| \geq 3^m -
\sum_{n<m} 3^n \geq \frac{3^m + 1}{2} \xrightarrow[m\to\infty]{}
\infty
\]
(terms $n > m$ vanish by periodicity; term $n = m$ contributes
exactly $3^m$; terms $n < m$ are bounded by the Lipschitz property).
Conclude that $W$ is differentiable nowhere.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:funcseq:10}
Let $u_n(x) = (-1)^n x^n(1 - x)$ on $\intcc{0}{1}$. Show that
$\sum u_n$ converges pointwise on $\intcc{0}{1}$ and compute its
sum; show that the convergence is uniform on $\intcc{0}{1}$
\emph{(bound the remainder $R_N(x) = \sum_{n > N} u_n(x)$, a
geometric tail, by its first term and maximize $x^{N+1}(1-x)$)}
but \emph{not} normal \emph{(compute $\norm{u_n}_\infty$)}:
uniform convergence is strictly weaker than normal convergence.
Contrast with $\sum x^n(1-x)$, whose sum is discontinuous at
$1$: there even uniformity fails.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:funcseq:11}
Let $f_n \to f$ uniformly on a metric space $X$, each $f_n$
continuous, and let $x_n \to x$ in $X$. Prove $f_n(x_n) \to
f(x)$. Show by an example on $X = \intcc{0}{1}$ that pointwise
convergence does not suffice, even with $f$ continuous
\emph{(use the bumps $g_n$ of \cref{exo:b2:funcseq:1} and $x_n =
\frac{1}{\sqrt{2n}}$)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:funcseq:12}
(A Volterra integral equation by series) For $f \in
C(\intcc{0}{1})$ define $Tf(x) = \int_0^x f(t)\,\dd t$.
\begin{enumerate}
  \item Show by induction that for $n \geq 1$:
        \[
        T^n f(x) = \int_0^x \frac{(x -
        t)^{n-1}}{(n-1)!}\,f(t)\,\dd t,
        \qquad
        \norm{T^n f}_\infty \leq \frac{\norm f_\infty}{n!} .
        \]
  \item Deduce that $S = \sum_{n\geq0} T^n f$ converges normally
        on $\intcc{0}{1}$ and solves the integral equation $S = f
        + TS$.
  \item Verify that $S(x) = f(x) + \int_0^x \eu^{x-t}f(t)\,\dd
        t$ solves the same equation, and prove uniqueness of
        continuous solutions \emph{(if $S = TS$ then
        $\norm{S}_\infty \leq \norm{T^nS}_\infty \to 0$)}:
        conclude the closed form of the sum.
\end{enumerate}
\end{exercise}

\section{Problem: Rates of approximation and Korovkin's theorem}

\begin{problem}\label{pb:b2:funcseq:1}
The Bernstein proof of \cref{thm:b2:funcseq:weierstrass} hides
two treasures. First, it is \emph{quantitative}: how fast $B_nf
\to f$ is governed by the modulus of continuity of $f$, with the
sharp rate attained by $\abs{x - \frac12}$. Second, it is
\emph{structural}: all that mattered was that $B_n$ is a
positive linear operator behaving well on $1$, $x$, $x^2$ ---
that observation, isolated, is \emph{Korovkin's
theorem}\index{Korovkin's theorem}. This problem proves both,
and closes with Voronovskaya's exact asymptotic. Throughout,
$f \in C(\intcc{0}{1})$, $M = \norm f_\infty$, $p_k(x) = \binom
nk x^k(1-x)^{n-k}$, and $e_j$ denotes $x \mapsto x^j$.

\medskip
\noindent\textbf{Part I --- The Bernstein operator.}
\begin{enumerate}
  \item Show that $B_n$ is linear, \emph{positive} ($f \geq 0
        \Rightarrow B_nf \geq 0$), hence monotone ($f \leq g
        \Rightarrow B_nf \leq B_ng$), with
        $\norm{B_nf}_\infty \leq \norm f_\infty$, and that
        $B_nf$ interpolates $f$ at both endpoints.
  \item Rederive the identities $B_n e_0 = e_0$, $B_n e_1 =
        e_1$ and $B_n e_2 = e_2 + \frac{e_1 - e_2}{n}$
        \emph{(differentiate $(x + y)^n$ twice and set $y = 1 -
        x$)}.
  \item Deduce the variance identity $\sum_k \bigl(\frac kn -
        x\bigr)^2 p_k(x) = \frac{x(1-x)}{n}$ and, by
        Cauchy--Schwarz, the first-moment bound
        \[
        \sum_{k=0}^{n}\Bigl|\frac kn - x\Bigr|\,p_k(x)
        \leq \sqrt{\frac{x(1-x)}{n}} \leq \frac{1}{2\sqrt n} .
        \]
  \item Show that if $f$ is convex then $B_nf \geq f$ on
        $\intcc{0}{1}$ \emph{(finite Jensen inequality for the
        weights $p_k(x)$)}.
  \item (Chebyshev's counting bound, restated) For $\delta > 0$
        show
        \[
        \sum_{\abs{k/n - x} > \delta} p_k(x)
        \leq \frac{x(1-x)}{n\delta^2}
        \leq \frac{1}{4n\delta^2} ,
        \]
        and give the probabilistic reading: $B_nf(x)$ averages
        $f$ over a binomial sample mean concentrating at $x$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Rates: the modulus of continuity.}
For $\delta > 0$ set $\omega(\delta) = \sup\{\abs{f(s) - f(t)}
: s, t \in \intcc{0}{1},\ \abs{s - t} \leq \delta\}$.
\begin{enumerate}[resume]
  \item Show: $\omega$ is finite, nondecreasing,
        $\omega(\delta) \to 0$ as $\delta \to 0^+$ (Heine),
        subadditive ($\omega(\delta_1 + \delta_2) \leq
        \omega(\delta_1) + \omega(\delta_2)$), and
        $\omega(\lambda\delta) \leq (1 +
        \lambda)\,\omega(\delta)$ for all $\lambda > 0$.
  \item Prove the master estimate, for every $\delta > 0$:
        \[
        \abs{B_nf(x) - f(x)}
        \leq \sum_k \omega\Bigl(\Bigl|\frac kn -
        x\Bigr|\Bigr)p_k(x)
        \leq \Bigl(1 + \frac1\delta\sum_k\Bigl|\frac kn -
        x\Bigr|p_k(x)\Bigr)\,\omega(\delta) .
        \]
  \item Choose $\delta = n^{-1/2}$ and conclude the
        \emph{quantitative Weierstrass theorem}:
        \[
        \norm{B_nf - f}_\infty \leq
        \frac32\,\omega\Bigl(\frac{1}{\sqrt n}\Bigr)
        \xrightarrow[n\to\infty]{} 0 .
        \]
  \item Deduce the rates: $\norm{B_nf - f}_\infty \leq
        \frac{3L}{2\sqrt n}$ for $L$-Lipschitz $f$, and $\leq
        \frac32 C n^{-\alpha/2}$ for $\alpha$-H\"older $f$
        ($\abs{f(s) - f(t)} \leq C\abs{s-t}^\alpha$).
  \item (The sharp example --- a binomial identity) For $m
        \geq 1$ prove
        \[
        \sum_{k=m+1}^{2m} (k - m)\binom{2m}{k}
        = \frac{m}{2}\binom{2m}{m},
        \qquad\text{hence}\qquad
        \sum_{k=0}^{2m}\abs{k - m}\binom{2m}{k}
        = m\binom{2m}{m}
        \]
        \emph{(use $k\binom{2m}k = 2m\binom{2m-1}{k-1}$ and the
        symmetry of the binomial line, which gives
        $\sum_{j=m}^{2m-1}\binom{2m-1}{j} = 2^{2m-2}$)}.
  \item For $f(t) = \abs{t - \frac12}$ deduce the exact value
        and its asymptotics (central binomial,
        \cref{ex:b2:comparison:centralbinomial}):
        \[
        B_{2m}f\Bigl(\frac12\Bigr) - f\Bigl(\frac12\Bigr)
        = \frac{\binom{2m}{m}}{2\cdot4^{m}}
        \;\sim\; \frac{1}{2\sqrt{\pi m}} :
        \]
        the rate $\omega(n^{-1/2})$ of question 8 is attained
        (up to a constant) --- for merely continuous $f$,
        Bernstein's $n^{-1/2}$ is honest.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Korovkin's theorem.} Let $(L_n)$
be a sequence of \emph{positive linear} operators from
$C(\intcc{0}{1})$ to itself such that $L_ne_j \to e_j$
uniformly for $j = 0, 1, 2$.
\begin{enumerate}[resume]
  \item Show that a positive linear $L$ is monotone and
        satisfies $\abs{Lf} \leq L\abs f$ pointwise.
  \item Show: for every $\varepsilon > 0$ there is $\delta > 0$
        such that for \emph{all} $s, x \in \intcc{0}{1}$:
        \[
        \abs{f(s) - f(x)} \leq \varepsilon +
        \frac{2M}{\delta^2}(s - x)^2
        \]
        \emph{(treat $\abs{s - x} \leq \delta$ by Heine and
        $\abs{s-x} > \delta$ by the crude bound $2M$)}.
  \item Fix $x$, apply $L_n$ to the inequality of question 13
        in the variable $s$, and derive
        \[
        \abs{L_nf(x) - f(x)\,L_ne_0(x)}
        \leq \varepsilon\,L_ne_0(x) + \frac{2M}{\delta^2}
        \bigl(L_ne_2(x) - 2x\,L_ne_1(x) + x^2 L_ne_0(x)\bigr).
        \]
  \item Show that $\sup_x \bigl(L_ne_2(x) - 2x\,L_ne_1(x) + x^2
        L_ne_0(x)\bigr) \to 0$, then assemble
        \emph{Korovkin's theorem}: $L_nf \to f$ uniformly for
        \emph{every} $f \in C(\intcc{0}{1})$.
  \item Check that $(B_n)$ satisfies Korovkin's hypotheses:
        Weierstrass a third time, from three monomials.
  \item Let $I_n$ be the piecewise-affine interpolation
        operator at the nodes $\frac kn$. Show $I_n$ is
        positive linear, $I_ne_0 = e_0$, $I_ne_1 = e_1$, and
        $\norm{I_ne_2 - e_2}_\infty = \frac{1}{4n^2}$
        \emph{(on each cell the error of affine interpolation
        of $t^2$ is $(t - a)(b - t)$)}. Conclude by Korovkin:
        polygonal interpolants converge uniformly for every
        continuous $f$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Dividends: density, moments,
derivatives.}
\begin{enumerate}[resume]
  \item Show that polynomials with \emph{rational} coefficients
        are dense in $\bigl(C(\intcc{0}{1}),
        \norm\cdot_\infty\bigr)$: this Banach space is
        separable.
  \item (Moments determine the function) Let $f \in
        C(\intcc{0}{1})$ with $\int_0^1 f(t)\,t^n \dd t = 0$
        for every $n \in \N$. Show $\int_0^1 f P = 0$ for every
        polynomial, then $\int_0^1 f^2 = 0$, then $f = 0$.
  \item Prove the derivative identity
        \[
        (B_nf)'(x) = n\sum_{k=0}^{n-1}\Bigl(
        f\Bigl(\frac{k+1}{n}\Bigr) -
        f\Bigl(\frac kn\Bigr)\Bigr)\,
        \binom{n-1}{k}x^k(1-x)^{n-1-k}
        \]
        \emph{(differentiate $p_k$ and reindex --- an Abel
        summation)}.
  \item Suppose $f$ is $C^1$. Using the mean value theorem in
        each increment and comparing with $B_{n-1}(f')$, show
        $(B_nf)' \to f'$ uniformly on $\intcc{0}{1}$. Deduce:
        for $f \in C^1$ there are polynomials converging to $f$
        \emph{together with} their derivatives.
  \item Suppose $f$ is $C^2$. By Taylor--Lagrange at $x$ show
        \[
        \abs{B_nf(x) - f(x)} \leq
        \frac{\norm{f''}_\infty}{2}\cdot\frac{x(1-x)}{n}
        \leq \frac{\norm{f''}_\infty}{8n} :
        \]
        smoothness upgrades the rate from $n^{-1/2}$ to
        $n^{-1}$.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Saturation: Voronovskaya's
theorem.}
\begin{enumerate}[resume]
  \item Prove the fourth-moment identity
        \[
        \sum_k (k - nx)^4 p_k(x)
        = nx(1-x)\bigl(1 + 3(n-2)x(1-x)\bigr) \leq n^2
        \quad (n \geq 1)
        \]
        \emph{(expand $k^4$ in falling factorials $k(k-1)\cdots$
        and use the differentiation trick of question 2 twice
        more)}.
  \item (Voronovskaya) Let $f$ be $C^2$ and $x \in
        \intcc{0}{1}$. Writing $f(t) = f(x) + f'(x)(t-x) +
        \frac{f''(x)}2(t-x)^2 + \eta(t)(t-x)^2$ with $\eta$
        bounded and $\eta(t) \to 0$ as $t \to x$, prove
        \[
        n\bigl(B_nf(x) - f(x)\bigr)
        \xrightarrow[n\to\infty]{}
        \frac{x(1-x)}{2}\,f''(x)
        \]
        \emph{(split the $\eta$ sum at $\abs{t - x} \leq
        \delta$; control the far part with question 23)}. So
        the error of question 22 is exact in order \emph{and}
        constant: $B_n$ \emph{saturates} at $\frac1n$, however
        smooth $f$ may be --- compare
        \cref{exo:b2:funcseq:7}.
  \item Synthesis. In one sentence each: (i) what positivity
        alone bought (Parts I and III); (ii) where compactness
        of $\intcc{0}{1}$ entered every part; (iii) why three
        test functions suffice in Korovkin's theorem; (iv) the
        trade-off Bernstein makes (robust $n^{-1/2}$ for rough
        $f$, but a $\frac1n$ ceiling for smooth $f$), and which
        chapter of this book will play the same game with
        trigonometric polynomials.
\end{enumerate}
\end{problem}
