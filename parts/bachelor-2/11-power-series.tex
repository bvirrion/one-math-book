\chapter{Power Series}\label{ch:b2:powerseries}

Power series are the best-behaved series of functions in
mathematics: inside their disk of convergence they converge
normally on compacts, may be differentiated and integrated term by
term without a second thought, and their sums --- \emph{analytic
functions} --- are determined by their coefficients. This chapter
proves that whole package and reclaims, honestly, every Taylor
series of the Year 1 volume; generating functions close it with
algebraic dividends.

\section{Radius of convergence}

\begin{lemma}[Abel]\label{lem:b2:powerseries:abel}
If the sequence $(a_n z_0^n)$ is bounded for some $z_0 \neq 0$,
then $\sum a_n z^n$ converges absolutely for every $\abs z < \abs
{z_0}$, and normally on every disk $\abs z \leq r < \abs{z_0}$.
\end{lemma}

\begin{proof}
With $\abs{a_n z_0^n} \leq M$ and $\abs z \leq r$:
\[
\abs{a_n z^n} = \abs{a_n z_0^n}\,\Bigl|\frac{z}{z_0}\Bigr|^n
\leq M\Bigl(\frac{r}{\abs{z_0}}\Bigr)^{\!n},
\]
a convergent geometric bound, uniform on the disk.
\end{proof}

\begin{definition}[Radius of convergence]\label{def:b2:powerseries:radius}
The \emph{radius of convergence}\index{radius of convergence} of
$\sum a_n z^n$ is
\[
R = \sup\{r \geq 0 : (a_n r^n) \text{ bounded}\} \in
\intcc{0}{+\infty} .
\]
By \cref{lem:b2:powerseries:abel}: absolute convergence for $\abs
z < R$ (normal on compact subdisks), divergence --- indeed unbounded
terms --- for $\abs z > R$. On the boundary circle, anything can
happen (\cref{exo:b2:powerseries:2}). In practice $R$ is computed
by d'Alembert's ratio test on $\abs{a_n}\abs z^n$ or by comparison.
\end{definition}

\begin{proposition}[Operations]\label{prop:b2:powerseries:operations}
Let $\sum a_nz^n$, $\sum b_nz^n$ have radii $R_a, R_b$. Then, for
$\abs z < \min(R_a, R_b)$:
\[
\sum (a_n + b_n)z^n = \sum a_nz^n + \sum b_nz^n,
\qquad
\Bigl(\sum a_nz^n\Bigr)\Bigl(\sum b_nz^n\Bigr) = \sum c_n z^n,
\quad c_n = \sum_{k=0}^{n} a_kb_{n-k},
\]
both series having radius $\geq \min(R_a, R_b)$. (The product is
the Cauchy product, legitimate by absolute convergence and
\cref{thm:b2:series:fubini}.)
\end{proposition}

\begin{proof}
Sums of convergent series; for the product, both factors converge
absolutely inside the smaller disk and
\cref{thm:b2:series:fubini} applies.
\end{proof}

\section{Regularity of the sum}

\begin{theorem}[Term-by-term calculus]\label{thm:b2:powerseries:calculus}
Let $f(x) = \sum_{n\geq0} a_n x^n$ have radius $R > 0$ (real
variable $x \in \intoo{-R}{R}$).
\begin{enumerate}
  \item The derived series $\sum n\,a_n x^{n-1}$ has the \emph{same}
        radius $R$, and $f$ is $C^1$ with $f'(x) = \sum_{n \geq 1}
        n a_n x^{n-1}$. Iterating, $f$ is $C^\infty$ and
        \[
        a_n = \frac{f^{(n)}(0)}{n!} :
        \]
        the coefficients of a power series are unique (two series
        with the same sum near $0$ have equal coefficients).
  \item Term-by-term primitive: $\sum \frac{a_n}{n+1}x^{n+1}$ has
        radius $R$ and derivative $f$.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{Same radius:} if $(a_nr^n)$ is bounded and $r' < r$, then
$n\abs{a_n} r'^{\,n-1} = \frac{n}{r'}\abs{a_nr^n}
\bigl(\frac{r'}{r}\bigr)^n$ is bounded (indeed $\to 0$: geometric
beats $n$), so $R' \geq R$; conversely $\abs{a_n x^n} \leq \abs x
\cdot n\abs{a_n}\abs x^{n-1}$ gives $R \geq R'$.

\emph{Differentiation:} on $\intcc{-r}{r}$, $r < R$, the derived
series converges normally ($n\abs{a_n}r^{n-1}$ summable by the
radius computation); the original converges at $x = 0$: the
differentiation theorem for series
(\cref{thm:b2:funcseq:seriestransfer}) applies on every such
segment, hence on $\intoo{-R}{R}$. Iterating $k$ times and
evaluating at $0$: $f^{(k)}(0) = k!\,a_k$ (only the constant term
of the $k$-th derived series survives). Primitives: same radius by
the same computation, and differentiate term by term back.
\end{proof}

\begin{example}[The classics, this time honestly]\label{ex:b2:powerseries:classics}
$\displaystyle\frac{1}{1 - x} = \sum x^n$ ($R = 1$); integrating
term by term (\cref{thm:b2:powerseries:calculus}~(2)):
\[
-\ln(1 - x) = \sum_{n\geq1} \frac{x^n}{n},
\qquad
\arctan x = \sum_{n \geq 0} \frac{(-1)^n x^{2n+1}}{2n+1}
\quad (\abs x < 1),
\]
the second from $\frac{1}{1+x^2} = \sum (-1)^n x^{2n}$. And $\exp$:
the series $E(x) = \sum \frac{x^n}{n!}$ ($R = \infty$) satisfies
$E' = E$, $E(0) = 1$ by termwise differentiation, so $E = \exp$ by
the Year 1 uniqueness. Every ``standard expansion'' of the Year 1
volume is now a theorem about its full power series.
\end{example}

\begin{definition}[Analytic functions]\label{def:b2:powerseries:analytic}
$f$ is \emph{analytic}\index{analytic function} at $x_0$ when it is
the sum of a power series in $(x - x_0)$ on a neighborhood; on an
interval, when at every point. Sums of power series are analytic
inside their disk (rearrangement of the expansion --- admitted at
this level for re-centering, the case $x_0 = 0$ being
\cref{thm:b2:powerseries:calculus}). Analytic implies $C^\infty$;
the converse \emph{fails}: the flat function $\eu^{-1/x^2}$
(\cref{exo:b2:powerseries:7}).
\end{definition}

\begin{method}[Expanding via a differential equation]\label{met:b2:powerseries:ode}
To expand a function $f$ in a power series: find a linear ODE with
polynomial coefficients satisfied by $f$; inject $\sum a_nx^n$;
identify coefficients to get a recurrence for $(a_n)$; solve, and
check radius and initial conditions. Example --- the binomial
series: $f(x) = (1+x)^\alpha$ satisfies $(1+x)f' = \alpha f$, $f(0)
= 1$; injecting gives $(n+1)a_{n+1} = (\alpha - n)a_n$, so $a_n =
\binom{\alpha}{n}$, radius $1$ (ratio test), and the sum, satisfying
the same ODE with the same initial value, equals $(1 + x)^\alpha$
by the uniqueness theorem for linear ODEs (Year 1 volume).
\end{method}

\section{Generating functions}

\begin{example}[Fibonacci]\label{ex:b2:powerseries:fibonacci}
Let $F(x) = \sum_{n\geq0} F_n x^n$ (Fibonacci numbers, $F_0 = 0$,
$F_1 = 1$). The recurrence $F_{n+2} = F_{n+1} + F_n$ translates,
multiplying by $x^{n+2}$ and summing, into
\[
F(x) - x = x\,F(x) + x^2 F(x)
\quad\Longrightarrow\quad
F(x) = \frac{x}{1 - x - x^2} ,
\]
valid where the series converges ($R = \frac{1}{\varphi}$, from
$F_n \sim \frac{\varphi^n}{\sqrt5}$). Partial fractions on
$\frac{x}{1 - x - x^2}$ and the geometric series re-derive Binet's
formula --- generating functions industrialize linear
recurrences.\index{generating function}
\end{example}

\begin{example}[Catalan numbers]\label{ex:b2:powerseries:catalan}
The Catalan numbers $C_n$ (number of triangulations, of bracketings,
of Dyck paths, \dots) satisfy $C_0 = 1$ and $C_{n+1} = \sum_{k=0}^n
C_kC_{n-k}$. The generating function $C(x) = \sum C_nx^n$ then
satisfies (Cauchy product!)
\[
C(x) = 1 + x\,C(x)^2
\quad\Longrightarrow\quad
C(x) = \frac{1 - \sqrt{1 - 4x}}{2x} ,
\]
choosing the root with $C(0) = 1$. Expanding $\sqrt{1 - 4x}$ by the
binomial series gives the closed form
\[
C_n = \frac{1}{n+1}\binom{2n}{n} ,
\]
carried out in \cref{exo:b2:powerseries:8}.\index{Catalan numbers}
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b2:powerseries:1}
Radii of convergence:
$\sum \dfrac{n^2}{2^n}z^n$; $\;\sum \dfrac{z^n}{\binom{2n}{n}}$;
$\;\sum z^{n!}$; $\;\sum \bigl(2 + (-1)^n\bigr)^n z^n$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:powerseries:2}
Show that $\sum z^n$, $\sum \frac{z^n}{n}$, $\sum \frac{z^n}{n^2}$
all have radius $1$ but behave differently at $z = 1$ and $z = -1$:
divergence/divergence, divergence/convergence,
convergence/convergence.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:powerseries:3}
Compute the sums, for $\abs x < 1$:
\[
\sum_{n\geq0} n x^n,
\qquad
\sum_{n\geq0} n^2 x^n,
\qquad
\sum_{n\geq0} \frac{x^{2n+1}}{2n+1} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:powerseries:4}
Expand in power series at $0$, with radius: $\dfrac{1}{(1-x)(2-x)}$
(partial fractions); $\;\ln(1 + x + x^2)$ \emph{(write $1 + x + x^2
= \frac{1 - x^3}{1 - x}$)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:powerseries:5}
Prove that $f(x) = \sum_{n\geq1} H_n x^n = -\dfrac{\ln(1 -
x)}{1 - x}$ for $\abs x < 1$, where $H_n$ is the harmonic number
\emph{(Cauchy product of $\sum x^n$ and $\sum \frac{x^n}{n}$)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:powerseries:6}
Solve by generating function the recurrence $u_0 = 1$, $u_{n+1} =
2u_n + n$: compute $U(x) = \sum u_nx^n$ in closed form, decompose,
and read off $u_n = 2^{n+1} - n - 1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:powerseries:7}
Let $f(x) = \eu^{-1/x^2}$ for $x \neq 0$, $f(0) = 0$. Prove that
$f$ is $C^\infty$ on $\R$ with $f^{(n)}(0) = 0$ for all $n$
\emph{(show by induction that $f^{(n)}(x) = P_n\bigl(\frac1x\bigr)
\eu^{-1/x^2}$ for polynomials $P_n$, and use the growth
comparison)}. Conclude that $f$ is not analytic at $0$: its Taylor
series at $0$ converges --- to the wrong function.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:powerseries:8}
Complete \cref{ex:b2:powerseries:catalan}: expand $\sqrt{1 - 4x}$
with the binomial series, showing
\[
\binom{1/2}{n+1}(-4)^{n+1} = -\frac{2}{n+1}\binom{2n}{n},
\]
and deduce $C_n = \frac{1}{n+1}\binom{2n}{n}$; determine the radius
of convergence of $C(x)$ and the asymptotics of $C_n$ via
Stirling.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:powerseries:9}
(Abel's radial limit theorem, special case) Suppose $\sum a_n$
converges. Prove that $\lim_{x \to 1^-} \sum_{n} a_n x^n = \sum_n
a_n$. \emph{(Abel summation: with $A_n$ the partial sums and $A =
\lim A_n$, write $\sum a_nx^n = (1 - x)\sum A_n x^n$; then $\sum
a_nx^n - A = (1-x)\sum (A_n - A)x^n$, split the sum at a large
$N$.)} Application: $\sum \frac{(-1)^{n-1}}{n} = \ln 2$ and $\sum
\frac{(-1)^n}{2n+1} = \frac\pi4$, re-proved from the power series.
\end{exercise}
