\chapter{Discrete Random Variables}\label{ch:b2:randomvar}

\omterm{def:b2:randomvar:law}{Random variables} organize probability computations around
functions rather than \omterm{def:b2:proba:space}{events}. On \omterm{def:b2:structures:countable}{countable} spaces the theory is
powered by the \omterm{def:b2:series:summable}{summable} families of \cref{ch:b2:series}:
\omterm{def:b2:randomvar:expectation}{expectation} is the sum of a family indexed by the \omterm{def:b2:proba:space}{sample space},
and all its properties --- linearity, transfer, product formula
for \omterm{def:b2:proba:independence}{independent} variables --- are theorems about \omterm{def:b2:series:summable}{summable}
families. The chapter proves the key inequalities of Markov,
Chebyshev, Cauchy--Schwarz and Jensen, and ends with the classical
\omterm{def:b2:randomvar:law}{laws} and the weak law of large numbers, whose proof is two lines
once Chebyshev is available.

\section{Random variables and their laws}

\begin{definition}[Discrete random variable; law]\label{def:b2:randomvar:law}
Let $(\Omega, \P)$ be a \omterm{def:b2:structures:countable}{countable} \omterm{def:b2:proba:space}{probability space}. A
\emph{random variable}\index{random variable} is a map $X \colon \Omega \to E$ ($E$ any
set; \emph{real} random variable when $E = \R$). Its \emph{law}\index{law!of a random variable}
(or \emph{distribution}\index{distribution}) is the \omterm{def:b2:proba:space}{probability measure} $\P_X$ on the
\omterm{def:b2:structures:countable}{countable set} $X(\Omega)$ defined by
\[
\P_X(\{x\}) = \P(X = x)
= \P\bigl(\{\omega : X(\omega) = x\}\bigr) .
\]
\end{definition}

\begin{example}[The classical laws]\label{ex:b2:randomvar:laws}
\leavevmode
\begin{itemize}
\item \emph{Bernoulli} $\mathcal{B}(p)$: $X \in \{0, 1\}$, $\P(X =
1) = p$. Indicator of an \omterm{def:b2:proba:space}{event}.
\item \emph{Binomial} $\mathcal{B}(n, p)$: $\P(X = k) =
\binom nk p^k(1-p)^{n-k}$, $0 \leq k \leq n$: number of successes
in $n$ \omterm{def:b2:proba:independence}{independent} Bernoulli trials (High School volume; re-proved
below via sums of \omterm{def:b2:proba:independence}{independent} variables).
\item \emph{Geometric} $\mathcal{G}(p)$: $\P(X = k) =
(1-p)^{k-1}p$, $k \in \N^*$: rank of the first success
(\cref{ex:b2:proba:geometric}).
\item \emph{Poisson} $\mathcal{P}(\lambda)$: $\P(X = k) =
e^{-\lambda}\frac{\lambda^k}{k!}$, $k \in \N$ --- a \omterm{def:b2:proba:space}{probability
measure} by the exponential series. The law of rare events
(\cref{ch:b2:genfun}).
\end{itemize}
\end{example}

\begin{remark}[Which law models what]
The four \omterm{def:b2:randomvar:law}{laws} answer four primitive questions: Bernoulli,
``did it happen?''; binomial, ``how many times in $n$
tries?''; geometric, ``how long until the first time?'';
Poisson, ``how many \omterm{def:b2:proba:space}{events} at a given rate, when tries are
many and individually unlikely?''. Recognizing the question
is nine tenths of the modelling: sums of indicators point to
the binomial, waiting times to the geometric, \omterm{def:b2:proba:space}{rare-event}
counts to the Poisson --- with the passage from binomial to
Poisson made precise by the law of rare events in
\cref{ch:b2:genfun}.
\end{remark}

\begin{proposition}[Memorylessness of the geometric
law]\label{prop:b2:randomvar:memoryless}
If $X \sim \mathcal{G}(p)$, then for all $m, n \in \N$:
\[
\P(X > m + n \mid X > m) = \P(X > n) ,
\]
and the geometric \omterm{def:b2:randomvar:law}{laws} are the only \omterm{def:b2:randomvar:law}{laws} on $\N^*$ with this
property.
\end{proposition}

\begin{proof}
Summing the geometric weights, $\P(X > n) = (1-p)^n$. Hence
\[
\P(X > m + n \mid X > m)
= \frac{\P(X > m + n)}{\P(X > m)}
= \frac{(1-p)^{m+n}}{(1-p)^m} = (1-p)^n = \P(X > n).
\]
Conversely, if $G(n) = \P(X > n)$ satisfies $G(m + n) =
G(m)G(n)$ with $G(0) = 1$, then $G(n) = G(1)^n$ by induction; $q =
G(1) \in \intco{0}{1}$, and $q = 0$ or the \omterm{def:b2:randomvar:law}{law} is
$\mathcal{G}(1 - q)$: $\P(X = k) = G(k-1) - G(k) =
q^{k-1}(1 - q)$.
\end{proof}

\begin{example}[No number is ever ``due'']
Roll a die, waiting for a six: the waiting time is $X \sim
\mathcal G(1/6)$. Memorylessness says that after $10$
fruitless rolls, the \emph{remaining} wait $X - 10$, given
$X > 10$, is again $\mathcal G(1/6)$: the conditional
expected wait is still $6$ rolls, exactly as at the start.
The die does not remember, and no six is ever ``due'' --- the
gambler's fallacy is the belief that the conditional \omterm{def:b2:randomvar:law}{law}
should have shifted. Conversely, the proposition's uniqueness
half says this indifference \emph{characterizes} geometric
waiting times: any waiting time whose forecast never updates
is geometric. Real queues and lifetimes usually do update,
which is precisely how one detects that they are not
geometric.
\end{example}

\section{Expectation}

\begin{definition}[Expectation]\label{def:b2:randomvar:expectation}
A real \omterm{def:b2:randomvar:law}{random variable} $X$ on $(\Omega, \P)$ \emph{has an
expectation} if the family
$\bigl(X(\omega)\,\P(\{\omega\})\bigr)_{\omega \in \Omega}$ is
\omterm{def:b2:series:summable}{summable} (\cref{ch:b2:series}); its \emph{expectation}\index{expectation} is then
\[
\E(X) = \sum_{\omega \in \Omega} X(\omega)\,\P(\{\omega\}) .
\]
\end{definition}

\begin{theorem}[Transfer theorem]\label{thm:b2:randomvar:transfer}
$X$ has an \omterm{def:b2:randomvar:expectation}{expectation} if and only if the family $\bigl(x\,\P(X =
x)\bigr)_{x \in X(\Omega)}$ is \omterm{def:b2:series:summable}{summable}, and then
\[
\E(X) = \sum_{x \in X(\Omega)} x\,\P(X = x) .
\]
More generally, for $f \colon X(\Omega) \to \R$, the variable
$f(X)$ has an \omterm{def:b2:randomvar:expectation}{expectation} iff $\sum_x \abs{f(x)}\,\P(X = x) <
\infty$, and then $\E(f(X)) = \sum_x f(x)\,\P(X = x)$.
\end{theorem}

\begin{proof}
Partition $\Omega$ into the level sets $\Omega_x = \{X = x\}$, $x
\in X(\Omega)$. By the summation-by-packets theorem for \omterm{def:b2:series:summable}{summable}
families (\cref{ch:b2:series}), the family
$(X(\omega)\P(\{\omega\}))_\omega$ is \omterm{def:b2:series:summable}{summable} iff each packet is
(automatic: $\sum_{\omega \in \Omega_x}\abs{x}\P(\{\omega\}) =
\abs x\,\P(X = x)$) \emph{and} the family of packet sums
$\bigl(x\,\P(X = x)\bigr)_x$ is \omterm{def:b2:series:summable}{summable} --- and then the total
sums agree. For $f(X)$: apply the proven statement to the variable $Y =
f \circ X$, whose level sets are $\{Y = y\} =
\bigsqcup_{x : f(x) = y}\{X = x\}$; a second summation by
packets converts $\sum_y y\,\P(Y = y)$ into $\sum_x
f(x)\,\P(X = x)$, the packets now grouping the values $x$
by their image $f(x)$, with absolute \omterm{def:b2:series:summable}{summability} of one
family equivalent to that of the other.
\end{proof}

\begin{theorem}[Properties of expectation]\label{thm:b2:randomvar:linearity}
On the set of \omterm{def:b2:randomvar:law}{random variables} with \omterm{def:b2:randomvar:expectation}{expectation}:
\begin{enumerate}
\item (Linearity) $\E(aX + bY) = a\,\E(X) + b\,\E(Y)$.
\item (Positivity and monotonicity) $X \geq 0 \Rightarrow \E(X)
\geq 0$; $X \leq Y \Rightarrow \E(X) \leq \E(Y)$; and
$\abs{\E(X)} \leq \E(\abs X)$.
\item (Domination) If $\abs X \leq Z$ and $Z$ has an \omterm{def:b2:randomvar:expectation}{expectation},
so does $X$.
\end{enumerate}
\end{theorem}

\begin{proof}
All are properties of sums of \omterm{def:b2:series:summable}{summable} families
(\cref{ch:b2:series}): linearity of the sum, positivity term by
term, and the domination criterion for \omterm{def:b2:series:summable}{summability}. (Note that
linearity is immediate on the \emph{definition} over $\Omega$,
while it would be awkward on the transfer formula --- one benefit
of defining $\E$ upstream.)
\end{proof}

\begin{example}\label{ex:b2:randomvar:expectations}
$X \sim \mathcal{B}(n, p)$: writing $X = X_1 + \dots + X_n$ as a
sum of Bernoulli indicators and using linearity, $\E(X) = np$ ---
no binomial coefficients needed. $X \sim \mathcal{G}(p)$: $\E(X) =
\sum_{k\geq1}k(1-p)^{k-1}p = p\cdot\frac{1}{(1 - (1-p))^2} =
\frac1p$, by differentiating the geometric series inside its disk
(\cref{ch:b2:powerseries}). $X \sim \mathcal{P}(\lambda)$: $\E(X)
= \sum_{k\geq1}k e^{-\lambda}\frac{\lambda^k}{k!} = \lambda
e^{-\lambda}\sum_{j\geq0}\frac{\lambda^j}{j!} = \lambda$.
\end{example}

\begin{example}[Transfer in action]\label{ex:b2:randomvar:transferex}
For $X \sim \mathcal P(\lambda)$, compute
$\E\bigl(\frac1{1+X}\bigr)$ --- the \omterm{def:b2:randomvar:law}{law} of $\frac1{1+X}$
itself is awkward, but transfer never asks for it:
\[
\E\Bigl(\frac1{1+X}\Bigr)
= \sum_{k\geq0}\frac{1}{k+1}\,\eu^{-\lambda}
\frac{\lambda^k}{k!}
= \frac{\eu^{-\lambda}}{\lambda}\sum_{k\geq0}
\frac{\lambda^{k+1}}{(k+1)!}
= \frac{\eu^{-\lambda}}{\lambda}\bigl(\eu^\lambda - 1\bigr)
= \frac{1 - \eu^{-\lambda}}{\lambda} .
\]
Two lessons. Computationally: recognizing a shifted
exponential series is the whole work --- transfer reduces
\omterm{def:b2:randomvar:expectation}{expectations} of $f(X)$ to series manipulation. Structurally:
the naive plug-in value would be $\frac1{1 + \E X} =
\frac1{1 + \lambda}$, and the true answer is larger,
\[
\frac{1 - \eu^{-\lambda}}{\lambda} \geq
\frac{1}{1 + \lambda},
\]
exactly as Jensen's inequality for the convex function $t
\mapsto \frac1{1+t}$ demands. \omterm{def:b2:randomvar:expectation}{Expectations} of
\omterm{def:b2:affine:convex}{convex} images sit above the naive plug-in value, and transfer
plus a series check makes the abstract inequality concrete.
\end{example}

\begin{theorem}[Independence and products]\label{thm:b2:randomvar:product}
\omterm{def:b2:randomvar:law}{Random variables} $X, Y$ are \emph{\omterm{def:b2:proba:independence}{independent}} if $\P(X = x, Y =
y) = \P(X = x)\P(Y = y)$ for all $x, y$ --- equivalently, the
\omterm{def:b2:proba:space}{events} $\{X \in A\}$ and $\{Y \in B\}$ are \omterm{def:b2:proba:independence}{independent} for all $A,
B$. If $X$ and $Y$ are \omterm{def:b2:proba:independence}{independent} real variables with
\omterm{def:b2:randomvar:expectation}{expectations}, then $XY$ has an \omterm{def:b2:randomvar:expectation}{expectation} and
\[
\E(XY) = \E(X)\,\E(Y) .
\]
\end{theorem}

\begin{proof}
The equivalence of the two formulations follows by summing the
\omterm{def:b2:funcseq:def}{pointwise} identity over $(x, y) \in A \times B$
($\sigma$-additivity twice). For the product: the double family
$\bigl(xy\,\P(X = x)\P(Y = y)\bigr)_{(x,y)}$ is \omterm{def:b2:series:summable}{summable}, since by
Fubini for families (\cref{ch:b2:series})
\[
\sum_{x, y}\abs x \abs y\,\P(X{=}x)\P(Y{=}y)
= \Bigl(\sum_x \abs x \P(X{=}x)\Bigr)
\Bigl(\sum_y \abs y \P(Y{=}y)\Bigr) < \infty ;
\]
by \omterm{def:b2:proba:independence}{independence} this family is exactly $\bigl(xy\,\P(X = x, Y =
y)\bigr)$, whose sum is $\E(XY)$ by transfer applied to the
variable $(X, Y) \mapsto xy$; Fubini again evaluates the unsigned
sum as the product $\E(X)\E(Y)$.
\end{proof}

\begin{example}[Products, with and without
independence]\label{ex:b2:randomvar:productcontrast}
Roll two fair dice. If $Y$ is the second die (\omterm{def:b2:proba:independence}{independent} of
the first), $\E(XY) = \E(X)\E(Y) = 3.5^2 = 12.25$. If instead
$Y = X$ (the ``product'' of a die with itself),
\[
\E(X^2) = \frac{1 + 4 + 9 + 16 + 25 + 36}{6} = \frac{91}{6}
\approx 15.17 \neq 12.25 :
\]
same marginal \omterm{def:b2:randomvar:law}{laws} in both scenarios, different joint \omterm{def:b2:randomvar:law}{laws},
different product \omterm{def:b2:randomvar:expectation}{expectations}. The moral, worth engraving:
$\E(XY)$ is a functional of the \emph{pair}, not of the two
marginals --- and the gap $\E(X^2) - \E(X)^2 \approx 2.92$
is, by K\"onig--Huygens, precisely the \omterm{def:b2:randomvar:variance}{variance}
$\frac{35}{12}$ of the die.
\end{example}

\section{Variance, covariance, and the classical inequalities}

\begin{definition}[Moments, variance]\label{def:b2:randomvar:variance}
$X$ has a \emph{moment of order 2} if $X^2$ has an \omterm{def:b2:randomvar:expectation}{expectation}
(then so does $X$, by domination: $\abs X \leq \frac{1 +
X^2}{2}$). Its \emph{variance}\index{variance} and \emph{standard
deviation}\index{standard deviation} are
then
\[
V(X) = \E\bigl((X - \E(X))^2\bigr)
= \E(X^2) - \E(X)^2 ,
\qquad
\sigma(X) = \sqrt{V(X)} ,
\]
(the second form --- the \emph{K\"onig--Huygens} formula --- by
expanding the square and using linearity:
\[
\E\bigl((X - \E X)^2\bigr)
= \E\bigl(X^2 - 2X\,\E X + \E(X)^2\bigr)
= \E(X^2) - 2\,\E(X)^2 + \E(X)^2 ,
\]
the middle term using that $\E X$ is a constant). For $X, Y$ with second moments,
the \emph{covariance}\index{covariance} is
\[
\operatorname{Cov}(X, Y)
= \E\bigl((X - \E X)(Y - \E Y)\bigr)
= \E(XY) - \E(X)\E(Y) .
\]
\end{definition}

\begin{theorem}[Variance toolkit]\label{thm:b2:randomvar:variancerules}
For variables with second moments:
\begin{enumerate}
\item $V(aX + b) = a^2\,V(X)$;
\item $V(X + Y) = V(X) + V(Y) + 2\operatorname{Cov}(X, Y)$, and
more generally
\[
V\Bigl(\sum_{i=1}^n X_i\Bigr)
= \sum_{i=1}^n V(X_i)
+ 2\sum_{i < j}\operatorname{Cov}(X_i, X_j) ;
\]
\item if $X, Y$ are \omterm{def:b2:proba:independence}{independent}, $\operatorname{Cov}(X, Y) = 0$
(the converse is false), so \omterm{def:b2:randomvar:variance}{variances} of \omterm{def:b2:proba:independence}{independent} variables
add.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1} and \emph{2} are expansions of squares plus linearity;
the products $X_iX_j$ have \omterm{def:b2:randomvar:expectation}{expectations} by Cauchy--Schwarz below
(or by $\abs{X_iX_j} \leq \frac{X_i^2 + X_j^2}{2}$). \emph{3} is
\cref{thm:b2:randomvar:product} applied to the centered
variables. A standard counterexample to the converse: $X$ uniform
on $\{-1, 0, 1\}$ and $Y = X^2$ are uncorrelated
($\E(XY) = \E(X^3) = 0 = \E X \cdot \E Y$) but clearly dependent.
\end{proof}

\begin{theorem}[Markov and Chebyshev
inequalities]\label{thm:b2:randomvar:markov}
\leavevmode
\begin{enumerate}
\item (Markov) If $X \geq 0$ has an \omterm{def:b2:randomvar:expectation}{expectation}, then for every $a
> 0$:
\[
\P(X \geq a) \leq \frac{\E(X)}{a} .
\]
\item (Chebyshev) If $X$ has a second moment, then for every
$\varepsilon > 0$:
\[
\P\bigl(\abs{X - \E(X)} \geq \varepsilon\bigr)
\leq \frac{V(X)}{\varepsilon^2} .
\]
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} \omterm{def:b2:funcseq:def}{Pointwise}, $a\,\mathbf{1}_{X \geq a} \leq X$ (on the
\omterm{def:b2:proba:space}{event} the left side is $a \leq X$; off it, $0 \leq X$). Take
\omterm{def:b2:randomvar:expectation}{expectations}: $a\,\P(X \geq a) \leq \E(X)$ by monotonicity and
$\E(\mathbf{1}_A) = \P(A)$.
\emph{2.} Apply Markov to the nonnegative variable $(X - \E
X)^2$ at level $a = \varepsilon^2$: the \omterm{def:b2:proba:space}{event} $\{(X - \E X)^2
\geq \varepsilon^2\}$ is exactly $\{\abs{X - \E X} \geq
\varepsilon\}$.
\end{proof}

\begin{example}[Uncorrelated but glued
together]\label{ex:b2:randomvar:sumdiff}
Roll two fair dice, $X$ and $Y$ \omterm{def:b2:proba:independence}{independent}, and set $S = X +
Y$, $D = X - Y$. By bilinearity of the \omterm{def:b2:randomvar:variance}{covariance},
\[
\operatorname{Cov}(S, D) = V(X) - V(Y) +
\operatorname{Cov}(Y, X) - \operatorname{Cov}(X, Y) = V(X) -
V(Y) = 0 :
\]
sum and difference are uncorrelated. \omterm{def:b2:proba:independence}{Independent}? Certainly
not: $S = 12$ forces $D = 0$, while $\P(D = 0) =
\frac16$ unconditionally. Correlation only tests the
\emph{linear} part of a dependence; here the dependence is
carried by the constraint that $S$ and $D$ have the same
parity, invisible to \omterm{def:b2:randomvar:variance}{covariance}. (For this pair, zero
\omterm{def:b2:randomvar:variance}{covariance} needed $V(X) = V(Y)$: identical \omterm{def:b2:randomvar:law}{distributions},
not \omterm{def:b2:proba:independence}{independence}, did the work.)
\end{example}

\begin{example}[When Markov is exact]\label{ex:b2:randomvar:markovsharp}
Markov's inequality is an equality precisely when nothing is
wasted in the bound $a\,\mathbf 1_{X\geq a} \leq X$: the
variable must take only the values $0$ and $a$. Concretely,
if $\P(X = a) = \pi$ and $\P(X = 0) = 1 - \pi$, then $\E(X)
= a\pi$ and
\[
\P(X \geq a) = \pi = \frac{\E(X)}{a} .
\]
A realistic reading: in a population where average wealth is
$100$ and wealth is either $0$ or $10^6$, the proportion of
millionaires is exactly $10^{-4}$ --- Markov's bound, hit
exactly by maximal inequality. Whenever $X$ spreads over
intermediate values the bound is strict, often wildly so;
but as the extreme case shows, no better inequality can be
extracted from the mean alone.
\end{example}

\begin{example}[Chebyshev is sharp --- without further
hypotheses]\label{ex:b2:randomvar:chebsharp}
Fix $\varepsilon > 0$, $q \in \intoc01$, and let $X$ take the
values $\pm\varepsilon$ with probability $\frac q2$ each and
$0$ with probability $1 - q$. Then $\E(X) = 0$, $V(X) =
q\varepsilon^2$, and
\[
\P\bigl(\abs{X - \E X} \geq \varepsilon\bigr) = q
= \frac{V(X)}{\varepsilon^2} :
\]
equality in Chebyshev. So the inequality cannot be improved
using only the \omterm{def:b2:randomvar:variance}{variance} --- the $1/\varepsilon^2$ decay is
the exact price of second-moment information. Faster decay
requires stronger hypotheses: boundedness of the variable
buys \emph{exponential} concentration, as
\cref{exo:b2:randomvar:7} previews and this chapter's weekend
problem develops systematically.
\end{example}

\begin{theorem}[Cauchy--Schwarz and Jensen]\label{thm:b2:randomvar:jensen}
\leavevmode
\begin{enumerate}
\item (Cauchy--Schwarz) If $X, Y$ have second moments, $XY$ has an
\omterm{def:b2:randomvar:expectation}{expectation} and $\E(XY)^2 \leq \E(X^2)\,\E(Y^2)$; consequently
$\operatorname{Cov}(X,Y)^2 \leq V(X)V(Y)$.
\item (Jensen) If $\varphi \colon I \to \R$ is \omterm{def:b2:affine:convex}{convex} on an
interval containing $X(\Omega)$, and $X$, $\varphi(X)$ have
\omterm{def:b2:randomvar:expectation}{expectations}, then
\[
\varphi\bigl(\E(X)\bigr) \leq \E\bigl(\varphi(X)\bigr) .
\]
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} \omterm{def:b2:series:summable}{Summability} of $XY$: $\abs{XY} \leq \frac{X^2 + Y^2}2$.
The map $(X, Y) \mapsto \E(XY)$ is a positive \omterm{def:b2:quadratic:def}{symmetric bilinear
form} on the space of variables with second moments, so the abstract
Cauchy--Schwarz inequality of \cref{ch:b2:quadratic} applies
(positive \emph{semi}-definite suffices for the inequality).
Applying it to the centered variables gives the \omterm{def:b2:randomvar:variance}{covariance} bound.

\emph{2.} First, $m = \E(X)$ lies in $I$: $I$ is an interval
containing all values of $X$, and \omterm{def:b2:randomvar:expectation}{expectation} is monotone, so $m$
is between $\inf X(\Omega)$ and $\sup X(\Omega)$. By the
support-line theorem for convex functions
(\cref{ch:b2:realfun}), there are $\alpha, \beta$ with
$\varphi(t) \geq \alpha t + \beta$ for all $t \in I$ and
$\varphi(m) = \alpha m + \beta$. Then, \omterm{def:b2:funcseq:def}{pointwise} on $\Omega$,
$\varphi(X) \geq \alpha X + \beta$; taking \omterm{def:b2:randomvar:expectation}{expectations},
\[
\E\bigl(\varphi(X)\bigr) \geq \alpha\,\E(X) + \beta
= \varphi\bigl(\E(X)\bigr). \qedhere
\]
\end{proof}

\begin{example}
Jensen with $\varphi(t) = t^2$ gives $\E(X)^2 \leq \E(X^2)$ ---
positivity of the \omterm{def:b2:randomvar:variance}{variance}; with $\varphi(t) = 1/t$ on
$\intoo{0}{\infty}$: $\frac{1}{\E X} \leq \E\bigl(\frac1X\bigr)$
--- the harmonic mean is below the arithmetic mean, now in random
form.
\end{example}

\begin{remark}[Common pitfalls]
(i) $\E(XY) = \E(X)\E(Y)$ \emph{requires} \omterm{def:b2:proba:independence}{independence} (or at
least zero \omterm{def:b2:randomvar:variance}{covariance}): taking $Y = X$ gives $\E(X^2) \neq
\E(X)^2$ whenever $V(X) > 0$. (ii) Likewise $V(X + X) =
4V(X)$, not $2V(X)$: \omterm{def:b2:randomvar:variance}{variances} add only across \omterm{def:b2:proba:independence}{independent}
(or uncorrelated) summands. (iii) $\E(f(X))$ is not
$f(\E(X))$; for \omterm{def:b2:affine:convex}{convex} $f$ Jensen even tells you the
direction of the error, as in
\cref{ex:b2:randomvar:transferex}. (iv) Existence is a real
hypothesis: for the St Petersburg variable $X = 2^K$ with
$\P(K = k) = 2^{-k}$ ($k \geq 1$),
\[
\sum_{k\geq1}2^k\cdot2^{-k} = \sum_{k\geq1}1 = \infty :
\]
$X$ is finite almost surely yet has no \omterm{def:b2:randomvar:expectation}{expectation}, and no
fair entry price for the game exists. \omterm{def:b2:series:summable}{Summability} in the
definition of $\E$ is not bookkeeping pedantry --- it is
where heavy tails are detected. (v) Finally, the transfer
theorem needs \emph{absolute} \omterm{def:b2:series:summable}{summability} before any
rearrangement of the sum over values is legitimate
(\cref{ch:b2:series}).
\end{remark}


\begin{example}[Chebyshev on one hundred
tosses]\label{ex:b2:randomvar:hundredtosses}
For $X \sim \mathcal B(100, \frac12)$: $\E X = 50$, $V(X) =
25$. Chebyshev with $\varepsilon = 6$:
\[
\P(45 \leq X \leq 55) = \P(\abs{X - 50} < 6)
\geq 1 - \frac{25}{36} \approx 0.31 ,
\]
while the exact binomial sum gives $\approx 0.73$. The
guaranteed $31\%$ is far from the truth, but it required
\emph{only} the mean and \omterm{def:b2:randomvar:variance}{variance} --- the same certificate
applies verbatim to any variable with $\E = 50$, $V = 25$,
however exotic, and \cref{ex:b2:randomvar:chebsharp} shows
some such variable saturates it. Universality has a price;
when the \omterm{def:b2:randomvar:law}{distribution} is genuinely binomial, the exponential
tools of the weekend problem close most of the gap.
\end{example}

\begin{example}[The correlation of a part with its
whole]\label{ex:b2:randomvar:partwhole}
For \omterm{def:b2:proba:independence}{independent} identically distributed $X, Y$ with \omterm{def:b2:randomvar:variance}{variance}
$\sigma^2 > 0$, how correlated is one summand with the sum
$S = X + Y$? Compute
\[
\operatorname{Cov}(X, S) = \operatorname{Cov}(X, X) +
\operatorname{Cov}(X, Y) = \sigma^2 + 0 = \sigma^2,
\qquad V(S) = 2\sigma^2,
\]
so the correlation coefficient is
\[
\rho(X, S) = \frac{\operatorname{Cov}(X,
S)}{\sigma(X)\,\sigma(S)}
= \frac{\sigma^2}{\sigma\cdot\sigma\sqrt2}
= \frac{1}{\sqrt2} \approx 0.707 ,
\]
whatever the common \omterm{def:b2:randomvar:law}{law} --- dice, coins, Poisson counts. With
$n$ summands the same computation gives $\rho(X_1, S_n) =
1/\sqrt n$: each individual term's influence on the total
dilutes like a square root, which is the correlational shadow
of the $\sqrt n$ scale of fluctuations. Cauchy--Schwarz
guarantees $\abs\rho \leq 1$ always; here the bound is met
exactly in the degenerate case $n = 1$ and decays predictably
after.
\end{example}

\begin{example}[Weighted AM--GM from
Jensen]\label{ex:b2:randomvar:amgm}
Let $Y$ take the positive values $a_1, \dots, a_k$ with
probabilities $\lambda_1, \dots, \lambda_k$. The function
$-\ln$ is \omterm{def:b2:affine:convex}{convex} on $\intoo0\infty$, so Jensen gives
$-\ln\E(Y) \leq \E(-\ln Y)$, i.e.
\[
a_1^{\lambda_1}a_2^{\lambda_2}\cdots a_k^{\lambda_k}
\;\leq\; \lambda_1a_1 + \lambda_2a_2 + \dots + \lambda_ka_k :
\]
the weighted arithmetic--geometric inequality, with equality
iff $Y$ is constant. Equal weights $\lambda_i = \frac1k$
recover the classical AM--GM. Probability has quietly proved
a purely algebraic theorem: choosing a probability \omterm{def:b2:randomvar:law}{law} is
just a bookkeeping device for \omterm{def:b2:affine:convex}{convex} combinations --- the
barycentric viewpoint of \cref{ch:b2:affine} once more, now
with Jensen as the engine.
\end{example}

\section{The weak law of large numbers}

\begin{theorem}[Weak law of large numbers]\label{thm:b2:randomvar:lln}
Let $(X_k)_{k \geq 1}$ be pairwise \omterm{def:b2:proba:independence}{independent} \omterm{def:b2:randomvar:law}{random variables}
with the same \omterm{def:b2:randomvar:law}{law}, admitting a second moment; write $m = \E(X_1)$
and $S_n = X_1 + \dots + X_n$. Then for every $\varepsilon > 0$:
\[
\P\Bigl(\,\Bigl|\frac{S_n}{n} - m\Bigr| \geq \varepsilon\Bigr)
\;\leq\; \frac{V(X_1)}{n\,\varepsilon^2}
\xrightarrow[n \to \infty]{} 0 .
\]
\end{theorem}

\begin{proof}
By linearity $\E(S_n/n) = m$; by
\cref{thm:b2:randomvar:variancerules} (pairwise \omterm{def:b2:proba:independence}{independence}
kills the \omterm{def:b2:randomvar:variance}{covariances}) $V(S_n) = n\,V(X_1)$, so $V(S_n/n) =
V(X_1)/n$. Chebyshev's inequality applied to $S_n/n$ gives the
bound.
\end{proof}

\begin{remark}
This is the theorem that connects probability to frequency: for
$X_k$ the indicator of an \omterm{def:b2:proba:space}{event} $A$ in \omterm{def:b2:proba:independence}{independent} repetitions,
$S_n/n$ is the observed frequency of $A$, and the law of large
numbers says it concentrates around $\P(A)$ at rate
$\frac{p(1-p)}{n\varepsilon^2}$. The \emph{strong} law ($S_n/n \to
m$ almost surely) is a Year 3 theorem --- its proof for fourth
moments is within reach, though: see \cref{exo:b2:randomvar:9},
which runs Borel--Cantelli on the Chebyshev-type bound. The same
Chebyshev estimate powered the Bernstein-polynomial proof of the
Weierstrass approximation theorem in \cref{ch:b2:funcseq} --- the
counting lemma there \emph{was} the weak law of large numbers in
disguise.
\end{remark}

\begin{example}[Collecting fifty coupons]
\label{ex:b2:randomvar:couponnumeric}
The coupon collector of \cref{exo:b2:randomvar:3} with $n =
50$ distinct toys: the expected total is
\[
\E(T_{50}) = 50\,H_{50} = 50\sum_{k=1}^{50}\frac1k
\approx 50 \times 4.499 \approx 225
\]
boxes --- four and a half times the naive guess $50$. The
harmonic growth is the whole story: the first $25$ toys
arrive in about $50\ln2 \approx 35$ boxes, while the
\emph{last} toy alone costs $50$ boxes on average (a
geometric wait of parameter $\frac1{50}$). Completion
problems are dominated by their endgame, which is why
\cref{exo:b2:randomvar:12} finds fluctuations of order $n$
--- the size of that final geometric wait --- around the
mean $n\ln n$.
\end{example}

\begin{example}[How large must $n$ be?]\label{ex:b2:randomvar:llnnumeric}
To pin the observed frequency within $\varepsilon = 0.01$ of
$\P(A)$ with confidence $95\%$, Chebyshev's bound demands
\[
\frac{p(1-p)}{n\varepsilon^2} \leq \frac{1}{4n\varepsilon^2}
\leq 0.05,
\qquad\text{i.e.}\qquad
n \geq \frac{1}{4\cdot0.05\cdot(0.01)^2} = 50\,000 .
\]
The dependence is brutal in $\varepsilon$ (quadratic) and
mild in the confidence (linear in $1/\alpha$). Both features
are properties of the \emph{bound}, not of the truth: the
weekend problem's exponential inequalities lower the price of
confidence from $1/\alpha$ to $\ln(1/\alpha)$ --- the same
specification will cost about $18\,500$ samples there ---
while the $1/\varepsilon^2$ scale is genuine and
unimprovable. Knowing which part of a bound is loose is as
useful as the bound itself.
\end{example}

\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b2:randomvar:1}
Compute $\E(X)$ and $V(X)$ for $X \sim \mathcal{B}(n, p)$ (via
indicators), $X \sim \mathcal{P}(\lambda)$ (show $V(X) =
\lambda$), and $X \sim \mathcal{G}(p)$ (show $V(X) =
\frac{1-p}{p^2}$; use $\E(X(X-1))$ and the second derivative of
the geometric series).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:randomvar:2}
Let $X \sim \mathcal{P}(\lambda)$ and $Y \sim \mathcal{P}(\mu)$ be
\omterm{def:b2:proba:independence}{independent}. Show that $X + Y \sim \mathcal{P}(\lambda + \mu)$
(convolution of the weights; binomial theorem), and that the
conditional \omterm{def:b2:randomvar:law}{law} of $X$ given $X + Y = n$ is binomial
$\mathcal{B}\bigl(n, \frac{\lambda}{\lambda + \mu}\bigr)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:randomvar:3}
(Coupon collector, \omterm{def:b2:randomvar:expectation}{expectation}) A cereal brand hides one of $n$
distinct toys, uniformly, in each box. Let $T_n$ be the number of
boxes needed to collect all $n$ toys. Writing $T_n$ as a sum of
\omterm{def:b2:proba:independence}{independent} geometric variables (time to see a \emph{new} toy when
$k$ are still missing), show
\[
\E(T_n) = n\sum_{k=1}^{n}\frac{1}{k} \sim n\ln n
\]
(equivalent by the series--integral comparison of
\cref{ch:b2:comparison}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:randomvar:4}
Let $X \geq 0$ be integer-valued. Prove the \emph{tail formula}
\[
\E(X) = \sum_{n=1}^{\infty} \P(X \geq n)
\]
(when either side is finite), by writing $X =
\sum_{n\geq1}\mathbf{1}_{X \geq n}$ and exchanging summations
(Fubini for nonnegative families). Recover $\E(X) = \frac1p$ for
the geometric \omterm{def:b2:randomvar:law}{law}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:randomvar:5}
(Sampling without replacement is more concentrated) An urn has $N$
balls, $M$ of them white. Draw $n \leq N$ without replacement and
let $X$ count the whites (\emph{hypergeometric} \omterm{def:b2:randomvar:law}{law}). Using
indicators $X = \sum_{i=1}^n Y_i$ with $Y_i$ the $i$-th draw:
show each $Y_i$ is Bernoulli of parameter $p = M/N$ (symmetry!),
conclude $\E(X) = np$ exactly as with replacement, and show
$\operatorname{Cov}(Y_i, Y_j) = -\frac{p(1-p)}{N-1} < 0$ for $i
\neq j$, hence $V(X) = np(1-p)\frac{N - n}{N - 1} \leq np(1-p)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:randomvar:6}
Let $X$ have a second moment. Show that $c \mapsto \E\bigl((X -
c)^2\bigr)$ is minimal exactly at $c = \E(X)$, with minimum
$V(X)$. Then show that $\P(X = \E(X)) = 1$ if and only if $V(X) =
0$. \emph{(For the second point: if $V(X) = 0$, use Chebyshev
with $\varepsilon = 1/n$ and monotone \omterm{def:b2:metric:continuity}{continuity},
\cref{thm:b2:proba:continuity}.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:randomvar:7}
(Concentration beats Markov) Let $S_n \sim \mathcal{B}(n,
\frac12)$ (number of heads in $n$ fair tosses). Compare the
bounds given by Markov ($\P(S_n \geq \frac{3n}{4})$), by Chebyshev,
and by the exponential (Chernoff) method:
\[
\P\Bigl(S_n \geq \frac{3n}4\Bigr)
\leq \E\bigl(e^{tS_n}\bigr)e^{-3nt/4}
= \Bigl(\frac{1 + e^t}{2}\Bigr)^n e^{-3nt/4}
\quad (t > 0),
\]
and optimize $t$ to obtain an exponentially small bound. \emph{(At
$t = \ln 3$: bound $\bigl(2\cdot 3^{-3/4}\bigr)^n \approx
(0.877)^n$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:randomvar:8}
(Weierstrass again, probabilistically) Let $f \colon [0,1] \to \R$
be \omterm{def:b2:metric:continuity}{continuous} and $S_n \sim \mathcal{B}(n, x)$. Show that the
Bernstein polynomial $B_nf(x) = \sum_{k=0}^n f\bigl(\frac
kn\bigr)\binom nk x^k(1-x)^{n-k}$ equals
$\E\bigl[f\bigl(\frac{S_n}{n}\bigr)\bigr]$, and re-derive the
estimate $\abs{B_nf(x) - f(x)} \leq \omega_f(\delta) +
\frac{2\norm f_\infty}{4n\delta^2}$ of \cref{ch:b2:funcseq} in
this probabilistic language (split on $\bigl|\frac{S_n}{n} -
x\bigr| \geq \delta$ and use Chebyshev).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:randomvar:9}
(Strong \omterm{def:b2:randomvar:law}{law} under fourth moments) Let $(X_k)$ be \omterm{def:b2:proba:independence}{independent},
identically distributed, centered ($\E X_1 = 0$), with
$\E(X_1^4) < \infty$. Expanding $\E(S_n^4)$ and counting the
surviving terms (only $\E(X_i^4)$ and $\E(X_i^2X_j^2)$ terms, $i
\neq j$), show $\E(S_n^4) \leq C n^2$ for a constant $C$. Deduce
$\sum_n \P\bigl(\abs{S_n/n} \geq \varepsilon\bigr) < \infty$
for each $\varepsilon > 0$ (Markov at order 4) and conclude with
Borel--Cantelli (\cref{thm:b2:proba:borelcantelli}) that
$S_n/n \to 0$ almost surely along a suitable formulation: the
\omterm{def:b2:proba:space}{event} $\bigcap_{j}\bigcup_N\bigcap_{n \geq N}\{\abs{S_n/n} <
\frac1j\}$ has probability $1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:randomvar:10}
Two fair dice are thrown; let $M$ be the larger of the two
results. Using the tail formula of \cref{exo:b2:randomvar:4}
(finite version), show
\[
\E(M) = \sum_{k=1}^{6}\P(M \geq k)
= 6 - \sum_{j=0}^5\Bigl(\frac j6\Bigr)^2 = \frac{161}{36}
\approx 4.47 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:randomvar:11}
Let $F_n$ be the number of fixed points of a uniformly random
permutation of $\{1, \dots, n\}$ ($n \geq 2$). Writing $F_n =
\sum_i\mathbf 1_{\sigma(i) = i}$, compute $\E(F_n) = 1$,
$\operatorname{Cov}(\mathbf 1_{\sigma(i)=i}, \mathbf
1_{\sigma(j)=j}) = \frac1{n^2(n-1)}$ for $i \neq j$, and
conclude $V(F_n) = 1$: on average one letter is fixed, with
\omterm{def:b2:randomvar:variance}{variance} exactly $1$, whatever $n$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:randomvar:12}
(Coupon collector, concentration) In the setting of
\cref{exo:b2:randomvar:3}, show
\[
V(T_n) = \sum_{k=1}^n\frac{1 - k/n}{(k/n)^2}
\leq n^2\sum_{k=1}^n\frac{1}{k^2} \leq \frac{\pi^2}{6}n^2,
\]
using \omterm{def:b2:proba:independence}{independence} of the geometric stages and $V(\mathcal
G(p)) = \frac{1-p}{p^2}$ (\cref{exo:b2:randomvar:1}; the value
$\pi^2/6$ is \cref{ex:b2:fourier:basel}). Deduce with
Chebyshev that $\dfrac{T_n}{n\ln n} \to 1$ \emph{in
probability}: the collector's total time is $n\ln n$ up to
fluctuations of order $n$.
\end{exercise}

\section{Problem: the concentration toolbox, from Markov to
Hoeffding}

\begin{problem}[{Weekend problem --- exponential concentration
by hand, and how many people a poll must ask}]
\label{pb:b2:randomvar:1}
Markov's inequality costs one moment and buys a $1/a$ decay;
Chebyshev costs two moments and buys $1/\varepsilon^2$ ---
and \cref{ex:b2:randomvar:chebsharp} shows that is all those
moments can buy. This problem climbs the rest of the ladder:
the exponential (Chernoff) method with its \emph{exact} rate
for coin flips, Hoeffding's inequality for all bounded
variables, and the payoff --- explicit, honest sample sizes
for polls, election calls and coin testing.
\index{concentration inequalities}\index{Hoeffding's
inequality}\index{Chernoff bound} Throughout, $S_n \sim
\mathcal B(n, p)$ is a sum of $n$ \omterm{def:b2:proba:independence}{independent} Bernoulli
variables and $\widehat p_n = S_n/n$ the empirical frequency.

\medskip
\noindent\textbf{Part I --- Calibration on the fair coin.}
Here $p = \frac12$ and $a \in \intoo{\frac12}{1}$.
\begin{enumerate}
  \item Markov at level $an$: show $\P(S_n \geq an) \leq
        \frac1{2a}$, a bound that does not even tend to $0$.
        Where does Markov lose so much?
  \item Chebyshev: using the symmetry of the fair binomial
        about $n/2$, show
        \[
        \P(S_n \geq an) = \tfrac12\,
        \P\bigl(\abs{S_n - \tfrac n2} \geq n(a -
        \tfrac12)\bigr)
        \leq \frac{1}{8n(a - 1/2)^2},
        \]
        i.e.\ $\frac2n$ at $a = \frac34$: polynomial decay at
        last.
  \item (Chernoff, general level) Compute
        $\E(\eu^{tS_n}) = \bigl(\frac{1 + \eu^t}2\bigr)^n$ and
        optimize $\P(S_n \geq an) \leq
        \E(\eu^{tS_n})\eu^{-tan}$ over $t > 0$: show the
        optimal $t$ is $\ln\frac{a}{1-a}$ and
        \[
        \P(S_n \geq an) \leq \eu^{-n\,I(a)},
        \qquad
        I(a) = \ln 2 + a\ln a + (1-a)\ln(1-a) > 0 .
        \]
        Check that $a = \frac34$ recovers the bound
        $\bigl(2\cdot3^{-3/4}\bigr)^n$ of
        \cref{exo:b2:randomvar:7}.
  \item (The exponent is exact) Let $k = an$ be an integer.
        From the fact that $\binom nk a^k(1-a)^{n-k}$ is the
        largest of the $n + 1$ terms of a probability
        \omterm{def:b2:randomvar:law}{distribution}, prove $\binom nk \geq
        \frac{\eu^{nH(a)}}{n+1}$ with $H(a) = -a\ln a -
        (1-a)\ln(1-a)$, and deduce the matching lower bound
        \[
        \P(S_n \geq an) \geq \binom{n}{an}2^{-n}
        \geq \frac{\eu^{-n\,I(a)}}{n + 1} .
        \]
  \item Tabulate the three bounds at $n = 100$, $a =
        \frac34$: Markov $\frac23$, Chebyshev $0.02$,
        Chernoff $\approx 2.1\cdot10^{-6}$ (the true value is
        $\approx 2.8\cdot10^{-7}$). Moral, in one sentence?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Hoeffding's inequality.}
\begin{enumerate}[resume]
  \item (Rademacher case) For $\varepsilon = \pm1$ with
        probability $\frac12$ each, prove
        \[
        \E(\eu^{t\varepsilon}) = \cosh t \leq \eu^{t^2/2}
        \qquad (t \in \R)
        \]
        by comparing the two series term by term ($(2k)! \geq
        2^kk!$).
  \item Deduce, for \omterm{def:b2:proba:independence}{independent} Rademacher variables
        $\varepsilon_1, \dots, \varepsilon_n$ and every $s >
        0$:
        \[
        \P\Bigl(\sum_{i=1}^n\varepsilon_i \geq s\Bigr)
        \leq \eu^{-s^2/(2n)} .
        \]
  \item Translate to fair coins ($X_i =
        \frac{1+\varepsilon_i}2$): $\P\bigl(\widehat p_n -
        \tfrac12 \geq \delta\bigr) \leq \eu^{-2n\delta^2}$,
        and the two-sided version with a factor $2$.
  \item (Hoeffding's lemma) Let $X \in \intcc01$ with $\E X =
        p$, and $\psi(t) = \ln\E(\eu^{tX})$. Justify that
        $\psi$ is twice \omterm{def:b2:diffcalc:differential}{differentiable} with
        \[
        \psi''(t) = \E_t(X^2) - \E_t(X)^2, \qquad
        \E_t(Y) := \frac{\E(Y\eu^{tX})}{\E(\eu^{tX})},
        \]
        a \emph{\omterm{def:b2:randomvar:variance}{variance}} of a reweighted variable still
        taking values in $\intcc01$; bound it by $\frac14$
        (\cref{exo:b2:randomvar:6}'s minimality argument) and
        conclude by Taylor:
        \[
        \E\bigl(\eu^{t(X - p)}\bigr) \leq \eu^{t^2/8} .
        \]
  \item (Hoeffding's inequality) For \omterm{def:b2:proba:independence}{independent} $X_i \in
        \intcc01$ with common mean $p$, deduce
        \[
        \P\bigl(\abs{\widehat p_n - p} \geq \delta\bigr)
        \leq 2\,\eu^{-2n\delta^2}
        \qquad (\delta > 0).
        \]
  \item Compare the Chebyshev rate
        $\frac{p(1-p)}{n\delta^2}$ with Hoeffding's
        $2\eu^{-2n\delta^2}$: which hypothesis does each
        require, and from which $n$ (roughly) does the
        exponential bound win at $\delta = 0.03$, $p =
        \frac12$?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- How many people must a poll
ask?} A poll asks $n$ \omterm{def:b2:proba:independence}{independent}, uniformly chosen voters;
each answers honestly; $p$ is the true score, $\widehat p_n$
the poll's figure.
\begin{enumerate}[resume]
  \item Show that the poll is accurate to $\pm\delta$ with
        confidence $1 - \alpha$ (i.e.\
        $\P(\abs{\widehat p_n - p} \geq \delta) \leq \alpha$)
        as soon as
        \[
        n \;\geq\; \frac{\ln(2/\alpha)}{2\,\delta^2} .
        \]
  \item Compute the required $n$ for the standard ``three
        points, ninety-five percent'' specification ($\delta
        = 0.03$, $\alpha = 0.05$): $n \geq 2050$; and for one
        point: $n \geq 18\,445$. Observe --- and explain ---
        the striking fact that the answer does not involve
        the population size.
  \item Redo question 13 with Chebyshev ($V(X_1) = p(1-p)
        \leq \frac14$): $n \geq \frac1{4\alpha\delta^2} =
        5556$ at three points. Note that sampling
        \emph{without} replacement only helps
        (\cref{exo:b2:randomvar:5}: the \omterm{def:b2:randomvar:variance}{variance} shrinks by
        $\frac{N-n}{N-1}$).
  \item (Calling an election) A candidate's true score is $p
        = 0.52$. How many voters must be polled so that
        $\P(\widehat p_n \leq \tfrac12) \leq 0.01$? Show $n
        \geq \frac{\ln 100}{2\cdot(0.02)^2} \approx 5757$ ---
        calling a close race costs far more than estimating a
        score.
  \item What the mathematics does \emph{not} cover: list the
        modelling assumptions used (\omterm{def:b2:proba:independence}{independent} uniform
        sampling, honest answers, fixed $p$), and explain in
        a short paragraph why real polling errors are
        dominated by \emph{bias} (non-uniform sampling,
        non-response), which no increase of $n$ reduces.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Sharper and cheaper.}
\begin{enumerate}[resume]
  \item (Median of means: exponential decay from two moments)
        Split a budget of $km$ samples into $k$ \omterm{def:b2:proba:independence}{independent}
        groups of $m$; let $\widehat p^{(1)}, \dots, \widehat
        p^{(k)}$ be the group means and $M$ their median.
        Choose $m$ so that each group satisfies
        $\P(\abs{\widehat p^{(i)} - p} \geq \delta) \leq
        \frac18$ (Chebyshev: $m \geq \frac2{\delta^2}$
        suffices). Show that if $\abs{M - p} \geq \delta$
        then at least $k/2$ groups err, and deduce
        \[
        \P(\abs{M - p} \geq \delta)
        \leq \binom{k}{\lceil k/2\rceil}\Bigl(\frac18
        \Bigr)^{k/2}
        \leq 2^k\cdot 8^{-k/2} = 2^{-k/2} :
        \]
        exponential concentration using nothing beyond
        \omterm{def:b2:randomvar:variance}{variances}.
  \item (Paley--Zygmund) For $X \geq 0$ with a second moment,
        prove $\P(X > 0) \geq \dfrac{\E(X)^2}{\E(X^2)}$
        \emph{(Cauchy--Schwarz on $X\mathbf 1_{X>0}$)}: the
        reverse-direction tool --- moments can also force
        \omterm{def:b2:proba:space}{events} to happen.
  \item (Pinsker-lite) Show $I(a) \geq 2\bigl(a -
        \tfrac12\bigr)^2$ on $\intoo{\frac12}1$ \emph{(the
        difference vanishes to second order at $\frac12$ and
        its second derivative is $\frac1{a(1-a)} - 4 \geq
        0$)}: Chernoff's exact exponent always beats
        Hoeffding's quadratic one.
  \item Expand $I\bigl(\tfrac12 + \delta\bigr) = 2\delta^2 +
        O(\delta^4)$ and combine with question 4: for small
        deviations the Hoeffding exponent $2n\delta^2$ is
        asymptotically \emph{exact} --- no method can beat it
        by more than polynomial factors.
  \item Draw up the toolbox table: for Markov, Chebyshev, the
        fourth-moment bound of \cref{exo:b2:randomvar:9},
        Hoeffding, and Chernoff with exponent $I$, state in
        one line each: hypothesis required, decay obtained,
        and the question in this problem where it was
        sharpest.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends.}
\begin{enumerate}[resume]
  \item (Testing a coin) A coin is either fair or biased with
        $p = 0.55$. You flip it $n$ times and declare
        ``biased'' when $\widehat p_n > 0.525$. Show that
        both error probabilities are at most
        $\eu^{-2n(0.025)^2}$, and that $n \geq 3685$ flips
        guarantee both below $1\%$.
  \item (Rare \omterm{def:b2:proba:space}{events} need a variance-aware bound) Let $p =
        0.01$ and take the relative specification $\delta =
        p/2 = 0.005$, $\alpha = 0.05$. Compare the sample
        sizes demanded by Hoeffding ($n \approx 74\,000$) and
        by Chebyshev with the true \omterm{def:b2:randomvar:variance}{variance} $p(1-p)$ ($n
        \approx 7920$): the variance-blind exponential bound
        loses to the humble second moment. State the moral,
        and where the missing tool (a variance-aware
        exponential bound; the Poisson approximation of
        \cref{ch:b2:genfun}) will come from.
  \item (Strong \omterm{def:b2:randomvar:law}{law} for coins) From $\sum_n
        2\eu^{-2n\delta^2} < \infty$ and Borel--Cantelli
        (\cref{thm:b2:proba:borelcantelli}), prove that
        $\widehat p_n \to p$ almost surely for \omterm{def:b2:proba:independence}{independent}
        coin flips: formulate the almost-sure \omterm{def:b2:proba:space}{event} as
        $\bigcap_j\bigcup_N\bigcap_{n\geq N}
        \{\abs{\widehat p_n - p} < \tfrac1j\}$ as in
        \cref{exo:b2:randomvar:9}, and conclude. (Boundedness
        replaces the fourth moment used there.)
  \item Synthesis. In five sentences: what each rung of the
        ladder (moments one, two, four; bounded exponential;
        exact exponent) costs and buys; why polling $2050$
        people suffices for a country of any size; and which
        of these bounds the Year 3 volume will sharpen into
        the central limit theorem's exact constants.
\end{enumerate}
\end{problem}
