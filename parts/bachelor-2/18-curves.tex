\chapter{Curves}\label{ch:b2:curves}

Geometry now turns differential. A curve is a point moving through
space; calculus gives us its velocity and acceleration, and geometry
asks what is intrinsic --- independent of how fast we traverse the
trajectory. The answers are \emph{\omterm{def:b2:curves:length}{arc length}}, which measures the
trajectory itself, and \emph{\omterm{thm:b2:curves:frenet2d}{curvature}}, which measures how it
bends. In dimension $3$ a second invariant, \emph{\omterm{thm:b2:curves:frenet3d}{torsion}},
measures how the curve twists out of its plane. The bookkeeping
device for all of this is the moving \emph{Frenet frame}.

\section{Parametrized arcs}

\begin{definition}[Parametrized arc]\label{def:b2:curves:arc}
A \emph{parametrized arc}\index{parametrized arc} of class
$\mathcal{C}^k$ ($k \geq 1$) is
a map $\gamma \colon I \to \R^n$ of class $\mathcal{C}^k$ on an
interval $I$. A point $\gamma(t)$ is \emph{regular} if
$\gamma'(t) \neq 0$, and the arc is \emph{regular} if all its
points are. The line through $\gamma(t)$ directed by $\gamma'(t)$
is the \emph{tangent line}\index{tangent line} at a regular point.
\end{definition}

\begin{definition}[Change of parameter]\label{def:b2:curves:reparam}
A \emph{change of parameter}\index{change of parameter} of class
$\mathcal{C}^k$ is a
$\mathcal{C}^k$ diffeomorphism $\theta \colon J \to I$ between
intervals ($\theta' \neq 0$ everywhere). The arcs $\gamma$ and
$\gamma \circ \theta$ are called \emph{equivalent}; a
\emph{geometric arc}\index{geometric arc} (or curve) is an
equivalence class. Notions
invariant under change of parameter --- the trajectory, the \omterm{def:b2:curves:arc}{tangent
line}, \omterm{def:b2:curves:length}{arc length}, \omterm{thm:b2:curves:frenet2d}{curvature} --- are called \emph{geometric}.
\end{definition}

\begin{remark}
The trajectory alone does not determine the \omterm{def:b2:curves:reparam}{geometric arc}: the
parametrizations $t \mapsto (\cos t, \sin t)$ on $[0, 2\pi]$ and on
$[0, 4\pi]$ have the same image but travel the circle once and
twice. A \omterm{def:b2:curves:reparam}{geometric arc} remembers the multiplicity and orientation
of the traversal, not its speed.
\end{remark}

\begin{example}[Speed changes nothing
geometric]\label{ex:b2:curves:speedinvariance}
Parametrize the unit circle by
\[
\gamma(t) = (\cos t^2,\ \sin t^2),
\qquad t \in \intcc0{\sqrt{2\pi}} .
\]
The speed $\norm{\gamma'(t)} = 2t$ grows linearly, yet
\[
L = \int_0^{\sqrt{2\pi}}2t\,\dd t = 2\pi ,
\]
the same \omterm{def:b2:curves:length}{length} as at constant speed --- as
\cref{thm:b2:curves:lengthinv} promises, via the \omterm{def:b2:curves:reparam}{change of
parameter} $t \mapsto t^2$. The \omterm{def:b2:curves:arc}{tangent line}, the \omterm{thm:b2:curves:frenet2d}{curvature}
computed from \cref{prop:b2:curves:kappaformula}, and every
other geometric quantity agree as well; only $t = 0$
deserves a glance, where $\gamma'(0) = 0$ makes \emph{this
parametrization} irregular although the trajectory is a
perfect circle. Geometric statements tolerate bad
parametrizations badly: reparametrize first, conclude
second.
\end{example}

\begin{example}\label{ex:b2:curves:cusp}
The arc $\gamma(t) = (t^2, t^3)$ is $\mathcal{C}^\infty$ but not
regular: $\gamma'(0) = (0, 0)$. Its trajectory, the semicubical
parabola $y^2 = x^3$, has a \emph{cusp} at the origin: smoothness
of the parametrization does not prevent a geometric singularity
where the velocity vanishes. This is why the regularity hypothesis
$\gamma' \neq 0$ is not cosmetic.
\end{example}

\section{Arc length}

\begin{definition}[Arc length]\label{def:b2:curves:length}
Let $\gamma \colon [a, b] \to \R^n$ be a $\mathcal{C}^1$ arc. Its
\emph{length}\index{arc length} is
\[
L(\gamma) = \int_a^b \norm{\gamma'(t)}\, \dd t ,
\]
where $\norm{\cdot}$ is the Euclidean \omterm{def:b2:nvs:norm}{norm}. The \emph{arc-length
function}\index{arc-length function} based at $t_0$ is $s(t) = \int_{t_0}^t
\norm{\gamma'(u)}\,\dd u$.
\end{definition}

\begin{theorem}[Length is geometric; polygonal
characterization]\label{thm:b2:curves:lengthinv}
\leavevmode
\begin{enumerate}
\item If $\theta \colon [c, d] \to [a, b]$ is a $\mathcal{C}^1$
\omterm{def:b2:curves:reparam}{change of parameter}, then $L(\gamma \circ \theta) = L(\gamma)$.
\item $L(\gamma)$ is the supremum of the \omterm{def:b2:curves:length}{lengths} of inscribed
polygons:
\[
L(\gamma)
= \sup\Bigl\{\, \sum_{i=1}^{m} \norm{\gamma(t_i) -
\gamma(t_{i-1})} \;:\; a = t_0 < t_1 < \dots < t_m = b \,\Bigr\}.
\]
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} By the change of variables $t = \theta(u)$ (Year 1
volume, valid since $\theta$ is $\mathcal{C}^1$ monotone),
\[
\int_c^d \norm{(\gamma\circ\theta)'(u)}\,\dd u
= \int_c^d \norm{\gamma'(\theta(u))}\,\abs{\theta'(u)}\,\dd u
= \int_a^b \norm{\gamma'(t)}\,\dd t ,
\]
where we used $(\gamma\circ\theta)' = \theta'\cdot
(\gamma'\circ\theta)$ and, if $\theta$ is decreasing, the sign of
$\theta'$ is absorbed by the reversal of the bounds.

\emph{2.} For any subdivision, $\gamma(t_i) - \gamma(t_{i-1}) =
\int_{t_{i-1}}^{t_i}\gamma'(t)\,\dd t$, so by the triangle
inequality for integrals $\norm{\gamma(t_i) - \gamma(t_{i-1})}
\leq \int_{t_{i-1}}^{t_i}\norm{\gamma'}$: every polygon is shorter
than $L(\gamma)$, hence $\sup \leq L(\gamma)$.

For the reverse inequality, let $\varepsilon > 0$. Since $\gamma'$
is \omterm{def:b2:metric:continuity}{continuous} on the \omterm{def:b2:metric:compact}{compact} $[a,b]$, it is uniformly continuous:
there is $\delta > 0$ with $\norm{\gamma'(t) - \gamma'(u)} \leq
\varepsilon$ whenever $\abs{t - u} \leq \delta$. Take a subdivision
of step $\leq \delta$. On each piece, for $t \in [t_{i-1}, t_i]$,
\[
\gamma(t_i) - \gamma(t_{i-1})
= \int_{t_{i-1}}^{t_i}\gamma'(t)\,\dd t
= (t_i - t_{i-1})\,\gamma'(t_{i-1}) + R_i,
\qquad
\norm{R_i} \leq \varepsilon\,(t_i - t_{i-1}),
\]
since $R_i = \int_{t_{i-1}}^{t_i}(\gamma'(t) -
\gamma'(t_{i-1}))\,\dd t$. Hence
\[
\norm{\gamma(t_i) - \gamma(t_{i-1})}
\geq (t_i - t_{i-1})\norm{\gamma'(t_{i-1})}
- \varepsilon (t_i - t_{i-1}) .
\]
Summing, and comparing $\sum (t_i -
t_{i-1})\norm{\gamma'(t_{i-1})}$ with $\int_a^b\norm{\gamma'}$ (a
Riemann sum of the \omterm{def:b2:metric:continuity}{continuous} function $\norm{\gamma'}$, within
$\varepsilon(b - a)$ of the integral for $\delta$ small enough by
uniform \omterm{def:b2:metric:continuity}{continuity} again), we get a polygon of \omterm{def:b2:curves:length}{length} $\geq
L(\gamma) - 2\varepsilon(b - a)$. Letting $\varepsilon \to 0$
proves the claim.
\end{proof}

\begin{example}[Archimedes and the inscribed
polygons]\label{ex:b2:curves:archimedes}
For the unit circle, the inscribed regular $n$-gon has \omterm{def:b2:curves:length}{length}
$L_n = 2n\sin\frac\pi n$, and the expansion $\sin x = x -
\frac{x^3}6 + O(x^5)$ gives
\[
L_n = 2\pi - \frac{\pi^3}{3n^2} + O\Bigl(\frac1{n^4}\Bigr) :
\]
the polygonal \omterm{def:b2:curves:length}{lengths} of \cref{thm:b2:curves:lengthinv}
converge \emph{quadratically}. Numerically: $L_6 = 6$
(the hexagon, giving the crude $\pi > 3$), while $L_{96} =
192\sin\frac{\pi}{96} \approx 6.28206$ against $2\pi \approx
6.28319$ --- the error $0.00113$ agrees with the predicted
$\pi^3/(3\cdot96^2) \approx 0.00112$. This is why Archimedes,
doubling the hexagon five times to $96$ sides, could bracket
$\pi$ to three digits by hand: each doubling divides the
error by four. The supremum in the polygonal
characterization is not just attained in the limit; it is
attained \emph{fast}, because a smooth curve separates from
its chords only at second order.
\end{example}

\begin{theorem}[Arc-length parametrization]\label{thm:b2:curves:arclength}
Let $\gamma \colon I \to \R^n$ be a regular $\mathcal{C}^k$ arc
($k \geq 1$). The \omterm{def:b2:curves:length}{arc-length function} $s$ is a $\mathcal{C}^k$
diffeomorphism from $I$ onto an interval $J$, and $\tilde\gamma =
\gamma \circ s^{-1}$ satisfies $\norm{\tilde\gamma'} = 1$
everywhere. Up to translation of the parameter and orientation,
this \emph{\omterm{def:b2:curves:length}{arc-length}} (or \emph{unit-speed}) parametrization is
unique.
\end{theorem}

\begin{proof}
$s'(t) = \norm{\gamma'(t)} > 0$ and $s'$ is $\mathcal{C}^{k-1}$
(composition of the $\mathcal{C}^{k-1}$ map $\gamma'$ with the \omterm{def:b2:nvs:norm}{norm},
smooth away from $0$), so $s$ is $\mathcal{C}^k$, strictly
increasing, a bijection onto $J = s(I)$, and its inverse is
$\mathcal{C}^k$ by the inverse function theorem in dimension $1$
(Year 1 volume). Then
\[
\tilde\gamma'(\sigma)
= \frac{\gamma'(t)}{s'(t)}
= \frac{\gamma'(t)}{\norm{\gamma'(t)}},
\qquad t = s^{-1}(\sigma),
\]
a unit vector. If $\hat\gamma = \gamma\circ\theta$ is another
unit-speed parametrization, then $\abs{\theta'} = 1$, so $\theta'
= \pm 1$ constant (\omterm{def:b2:metric:continuity}{continuity}), i.e.\ $\theta(u) = \pm u + c$.
\end{proof}

\begin{remark}
\omterm{def:b2:curves:length}{Arc length} is the parameter that separates geometry from
dynamics. A trajectory can be traversed with any speed
profile --- the physics of the motion --- but every
parametrization-invariant question (shape, bending,
osculation) has a canonical clock, the distance travelled.
This is why all \omterm{thm:b2:curves:frenet2d}{curvature} formulas below are \emph{defined}
at unit speed and then \emph{translated} to arbitrary
parametrizations by the chain rule: the translation factors
are powers of $v = s'$, and tracking them correctly is the
entire content of \cref{prop:b2:curves:kappaformula}.
\end{remark}

\begin{example}[Circle and helix]\label{ex:b2:curves:helix}
For the circle $\gamma(t) = (R\cos t, R\sin t)$, $\norm{\gamma'} =
R$, so $s = Rt$ and the \omterm{def:b2:curves:length}{length} of a full turn is $2\pi R$. For the
helix $\gamma(t) = (a\cos t,\ a\sin t,\ bt)$ with $a > 0$,
$\norm{\gamma'(t)} = \sqrt{a^2 + b^2}$ is constant: the helix is
traversed at constant speed, and $s = t\sqrt{a^2 + b^2}$.
\end{example}

\begin{example}[Arc length in polar
coordinates]\label{ex:b2:curves:polarlength}
A polar curve $r = r(\theta)$ is the arc $\gamma(\theta) =
(r\cos\theta,\ r\sin\theta)$, with
\[
\gamma'(\theta) = (r'\cos\theta - r\sin\theta,\
r'\sin\theta + r\cos\theta),
\qquad
\norm{\gamma'(\theta)}^2 = r'^2 + r^2
\]
(the cross terms cancel): the polar \omterm{def:b2:curves:length}{length} element is
$\sqrt{r^2 + r'^2}\,\dd\theta$. For the cardioid $r = 1 +
\cos\theta$:
\[
r^2 + r'^2 = (1 + \cos\theta)^2 + \sin^2\theta = 2 +
2\cos\theta = 4\cos^2\tfrac\theta2,
\]
and on $\intcc{-\pi}{\pi}$ the half-angle cosine is
nonnegative, so
\[
L = \int_{-\pi}^{\pi}2\cos\tfrac\theta2\,\dd\theta
= \Bigl[4\sin\tfrac\theta2\Bigr]_{-\pi}^{\pi}
= 4 - (-4) = 8 :
\]
like the cycloid arch of \cref{exo:b2:curves:1}, a curve
built from circles has a rational \omterm{def:b2:curves:length}{length}, with no $\pi$
anywhere. The half-angle factorization is the standard trick
for \omterm{def:b2:curves:length}{lengths} of circle-generated curves; when it fails (the
ellipse), the \omterm{def:b2:curves:length}{length} is a genuinely new function --- an
elliptic integral, beyond elementary closed forms.
\end{example}

\section{Curvature in the plane}

Throughout this section, arcs are $\mathcal{C}^2$ and regular in
the oriented Euclidean plane. We parametrize by \omterm{def:b2:curves:length}{arc length} and
write $T(s) = \tilde\gamma'(s)$ for the unit tangent, and $N(s)$
for the unit vector directly orthogonal to $T(s)$ (rotation of
$T$ by $+\pi/2$).

\begin{theorem}[Plane Frenet formulas]\label{thm:b2:curves:frenet2d}
Let $\tilde\gamma$ be a unit-speed $\mathcal{C}^2$ arc in the
oriented plane. There is a \omterm{def:b2:metric:continuity}{continuous} function $\kappa$, the
(algebraic) \emph{curvature}\index{curvature}, such that
\[
T'(s) = \kappa(s)\, N(s),
\qquad
N'(s) = -\kappa(s)\, T(s).
\]
\end{theorem}

\begin{proof}
Since $\norm{T(s)}^2 = 1$ for all $s$, differentiating the scalar
product gives $2\langle T'(s), T(s)\rangle = 0$: $T'(s)$ is orthogonal to
$T(s)$, hence collinear with $N(s)$ (dimension $2$); write $T'(s)
= \kappa(s) N(s)$ with $\kappa(s) = \langle T'(s), N(s)\rangle$,
\omterm{def:b2:metric:continuity}{continuous}. Likewise $N' \perp N$, so $N' = \lambda T$; and
differentiating $\langle T, N\rangle = 0$ gives $\langle T', N\rangle +
\langle T, N'\rangle = \kappa + \lambda = 0$.
\end{proof}

\begin{definition}\label{def:b2:curves:curvature}
When $\kappa(s) \neq 0$, the \emph{radius of
curvature}\index{radius of curvature} is $R(s) =
1/\abs{\kappa(s)}$ and the \emph{center of curvature}\index{center of
curvature} is
$\tilde\gamma(s) + \frac{1}{\kappa(s)} N(s)$; the circle with that
center and radius $R(s)$ is the \emph{osculating circle}\index{osculating circle}, the
best circular approximation of the curve at $\tilde\gamma(s)$.
\end{definition}

\begin{example}[The osculating circle of the
exponential]\label{ex:b2:curves:oscexp}
For $y = \eu^x$ at the point $(0, 1)$: $f'(0) = f''(0) = 1$,
so by the graph formula below,
\[
\kappa(0) = \frac{1}{(1 + 1)^{3/2}} = \frac1{2\sqrt2},
\qquad R = 2\sqrt2 .
\]
The unit tangent is $T = \frac{(1, 1)}{\sqrt2}$, the direct
normal $N = \frac{(-1, 1)}{\sqrt2}$, and the \omterm{def:b2:curves:curvature}{center of
curvature} is
\[
(0, 1) + 2\sqrt2\cdot\frac{(-1, 1)}{\sqrt2} = (-2,\ 3) :
\]
the \omterm{def:b2:curves:curvature}{osculating circle} has equation $(x + 2)^2 + (y - 3)^2 =
8$. As a check of the ``best circular approximation'' claim:
solving the circle's equation for $y$ near $(0,1)$ and
expanding gives $y = 1 + x + \frac{x^2}2 + O(x^3)$ ---
exactly the second-order Taylor expansion of $\eu^x$. The
\omterm{def:b2:curves:curvature}{osculating circle} matches value, slope \emph{and} second
derivative; an ordinary tangent circle would match only the
first two.
\end{example}

\begin{example}[The evolute of a circle is its
center]\label{ex:b2:curves:circleevolute}
For the circle of radius $R$ traversed counterclockwise,
$\kappa = 1/R$ and $N$ points toward the center, so the
\omterm{def:b2:curves:curvature}{center of curvature} $\tilde\gamma + \frac1\kappa N$ is the
center of the circle, for every $s$: the \omterm{def:b2:curves:curvature}{osculating circle}
of a circle is the circle itself, and the locus of centers
of \omterm{thm:b2:curves:frenet2d}{curvature} collapses to a point. This degenerate case
calibrates \cref{exo:b2:curves:6}: there the evolute's
velocity is $-\frac{\kappa'}{\kappa^2}N$, which vanishes
identically precisely when $\kappa$ is constant.
\end{example}

\begin{proposition}[Curvature in an arbitrary
parametrization]\label{prop:b2:curves:kappaformula}
For a regular $\mathcal{C}^2$ plane arc $\gamma(t) = (x(t), y(t))$,
\[
\kappa(t)
= \frac{x'(t)\,y''(t) - y'(t)\,x''(t)}
       {\bigl(x'(t)^2 + y'(t)^2\bigr)^{3/2}} ,
\]
in particular $\kappa = \dfrac{y''}{(1 + y'^2)^{3/2}}$ for a graph
$y = f(x)$.
\end{proposition}

\begin{proof}
Write $v(t) = \norm{\gamma'(t)} = s'(t)$, so $\gamma' = vT$
(composing the unit-speed data with $s$). Differentiating,
\[
\gamma'' = v'T + v\,T'\cdot s' = v'T + v^2\kappa N .
\]
Now take the \omterm{def:b2:linalg:det}{determinant} (in the canonical oriented basis) of
$(\gamma', \gamma'')$: since $\det(T, T) = 0$ and $\det(T, N) = 1$,
\[
\det(\gamma', \gamma'') = \det(vT,\ v'T + v^2\kappa N)
= v^3\kappa .
\]
The left side is $x'y'' - y'x''$, and $v^3 = (x'^2 +
y'^2)^{3/2}$. The graph case is the parametrization $t \mapsto (t,
f(t))$.
\end{proof}

\begin{example}[Circle, line, parabola]\label{ex:b2:curves:kappaexamples}
A line has $\kappa = 0$ (and conversely: $T' = 0$ means $T$
constant, so $\tilde\gamma(s) = \tilde\gamma(0) + sT$, a line). The
circle of radius $R$ traversed counterclockwise has $\kappa = 1/R$:
with $\gamma(t) = (R\cos t, R\sin t)$, the formula gives $\kappa =
R^2/R^3$. For the parabola $y = x^2/2$: $\kappa(x) = 1/(1 +
x^2)^{3/2}$, maximal at the vertex --- the parabola is most sharply
bent where it turns around.
\end{example}

\begin{remark}[Common pitfalls around curvature]
(i) The algebraic \omterm{thm:b2:curves:frenet2d}{curvature} of a plane arc changes sign when
the orientation of the arc or of the plane is reversed: only
$\abs\kappa$ and $R = 1/\abs\kappa$ are purely geometric. A
circle traversed clockwise has $\kappa = -1/R$. (ii) The
graph formula $\kappa = f''/(1 + f'^2)^{3/2}$ silently
chooses the parametrization by $x$; applying it to a curve
that is not a graph near the point (vertical tangent) is the
classical blunder. (iii) At a point where $\gamma' = 0$
nothing is defined --- neither $T$ nor $\kappa$ --- and the
trajectory may genuinely break
(\cref{ex:b2:curves:cusp}); always check regularity before
differentiating the unit tangent. (iv) In space, $\kappa =
\norm{T'} \geq 0$ by convention: there is no sign to get
wrong, but also no sign to exploit --- inflection-type
information moves into the \omterm{thm:b2:curves:frenet3d}{torsion}. (v) Finally, $\kappa$ is
a derivative \emph{with respect to \omterm{def:b2:curves:length}{arc length}}: for a
non-unit-speed parametrization, forgetting the factor $v^3$
in \cref{prop:b2:curves:kappaformula} is the most frequent
error in practice.
\end{remark}

\begin{theorem}[Curvature determines the curve]\label{thm:b2:curves:fundamental}
Let $\kappa \colon J \to \R$ be \omterm{def:b2:metric:continuity}{continuous}. There exists a
unit-speed $\mathcal{C}^2$ arc in the plane with \omterm{thm:b2:curves:frenet2d}{curvature}
$\kappa$, and it is unique up to a direct isometry (rotation
followed by translation).
\end{theorem}

\begin{proof}
\emph{Existence.} Fix $s_0 \in J$ and set $\varphi(s) =
\int_{s_0}^s \kappa(u)\,\dd u$, then
\[
\tilde\gamma(s) = \Bigl(\int_{s_0}^s \cos\varphi(u)\,\dd u,\
\int_{s_0}^s \sin\varphi(u)\,\dd u\Bigr).
\]
Then $T(s) = (\cos\varphi(s), \sin\varphi(s))$ is a unit vector,
$N(s) = (-\sin\varphi, \cos\varphi)$, and
\[
T'(s) = \varphi'(s)\,(-\sin\varphi, \cos\varphi)
= \kappa(s)\,N(s) :
\]
the arc is unit-speed with \omterm{thm:b2:curves:frenet2d}{curvature} $\kappa$.

\emph{Uniqueness.} Let $\gamma_1, \gamma_2$ be unit-speed arcs with
the same \omterm{thm:b2:curves:frenet2d}{curvature}. Each unit tangent lifts to an angle
function, by an explicit construction: view $T_j$ as the
complex number $z_j = a_j + \iu b_j$ of modulus $1$, pick
$\varphi_j(0)$ with $z_j(0) = \eu^{\iu\varphi_j(0)}$, and set
\[
\varphi_j(s) = \varphi_j(0) + \int_0^s\det\bigl(T_j,
T_j'\bigr)(u)\,\dd u .
\]
From $\abs{z_j} = 1$: $\operatorname{Re}(\conj{z_j}\,z_j') =
0$, so $\conj{z_j}\,z_j' = \iu\det(T_j, T_j') =
\iu\,\varphi_j'$, i.e.\ $z_j' = \iu\varphi_j'z_j$; then
\[
\bigl(z_j\,\eu^{-\iu\varphi_j}\bigr)'
= \eu^{-\iu\varphi_j}\bigl(z_j' - \iu\varphi_j'z_j\bigr) = 0,
\]
so $z_j = \eu^{\iu\varphi_j}$ throughout: $T_j =
(\cos\varphi_j, \sin\varphi_j)$ with $\varphi_j$ of class
$\mathcal C^1$. Moreover $\det(T_j, T_j') = \det(T_j,
\kappa N_j) = \kappa$, so $\varphi_j' = \kappa$. Hence $\varphi_2 = \varphi_1 + c$ for a
constant $c$: $T_2$ is $T_1$ rotated by the fixed angle $c$, so
integrating, $\gamma_2 = \rho(\gamma_1) + w$ where $\rho$ is the
rotation of angle $c$ and $w$ a constant vector.
\end{proof}

\begin{remark}
This is the one-dimensional prototype of a \emph{fundamental
theorem of geometry}: a \omterm{def:b2:metric:complete}{complete} set of local invariants (here, one
function) classifies the object up to rigid motion. The
three-dimensional version below needs two invariants.
\end{remark}

\begin{example}[Constant curvature means
circle]\label{ex:b2:curves:constantkappa}
Take $\kappa \equiv \kappa_0 > 0$ in the existence formula:
$\varphi(s) = \kappa_0 s$ and
\[
\tilde\gamma(s) = \Bigl(\frac{\sin\kappa_0s}{\kappa_0},\
\frac{1 - \cos\kappa_0s}{\kappa_0}\Bigr) :
\]
the circle of radius $1/\kappa_0$ centered at $(0,
1/\kappa_0)$, traversed at unit speed. By the uniqueness
half of the theorem, \emph{every} unit-speed arc of constant
\omterm{thm:b2:curves:frenet2d}{curvature} $\kappa_0$ is a piece of a circle of radius
$1/\kappa_0$ (or a line if $\kappa_0 = 0$) --- the converse
of the computation in \cref{ex:b2:curves:kappaexamples}, and
the plane case of \cref{exo:b2:curves:9}.
\end{example}

\begin{example}[Reconstructing a curve from its
curvature]\label{ex:b2:curves:reconstruction}
Which unit-speed curve has \omterm{def:b2:curves:curvature}{radius of curvature} $R(s) = 1 +
s^2$? Following the existence proof with $\kappa(s) =
\frac1{1+s^2}$ and $s_0 = 0$: $\varphi(s) = \arctan s$, so
\[
T(s) = (\cos\arctan s,\ \sin\arctan s)
= \Bigl(\frac{1}{\sqrt{1+s^2}},\
\frac{s}{\sqrt{1+s^2}}\Bigr),
\]
and integrating,
\[
\tilde\gamma(s) = \Bigl(\ln\bigl(s + \sqrt{1 + s^2}\bigr),\
\sqrt{1 + s^2} - 1\Bigr).
\]
Setting $x = \ln(s + \sqrt{1+s^2})$, i.e.\ $s = \sinh x$, the
second coordinate is $\cosh x - 1$: the curve is the
\emph{catenary} $y = \cosh x - 1$. This closes the loop with
\cref{exo:b2:curves:3}, where we computed $R = \cosh^2 x = 1 +
\sinh^2 x = 1 + s^2$ directly: the fundamental theorem
guarantees the catenary is the \emph{only} curve with this
\omterm{thm:b2:curves:frenet2d}{curvature} profile, up to a direct isometry.
\end{example}

\begin{remark}[Where curvature is used next]
The decomposition $\gamma'' = v'T + v^2\kappa N$ obtained in
the proof of \cref{prop:b2:curves:kappaformula} is the
kinematics of every curved motion: tangential versus
centripetal acceleration. \omterm{thm:b2:curves:frenet2d}{Curvature} returns for surfaces
(\cref{ch:b2:surfaces}) through the \omterm{thm:b2:curves:frenet2d}{curvature} of curves drawn
on them, and the \omterm{pb:b2:curves:1}{envelope} calculus of this chapter's weekend
problem --- evolutes, caustics --- is the geometric optics of
wavefronts. The Year 3 volume takes the intrinsic point of
view up again for submanifolds of $\R^n$.
\end{remark}

\section{Frenet frame in space}

Now let $\tilde\gamma \colon J \to \R^3$ be a unit-speed
$\mathcal{C}^3$ arc that is \emph{biregular}: $T'(s) \neq 0$ for
all $s$. Then $\kappa(s) = \norm{T'(s)} > 0$ defines the
\emph{\omterm{thm:b2:curves:frenet2d}{curvature}} (no sign in space: there is no preferred normal
orientation), and we set:
\[
N(s) = \frac{T'(s)}{\kappa(s)}
\quad\text{(principal normal)},
\qquad
B(s) = T(s) \wedge N(s)
\quad\text{(binormal)} ,
\]
so that $(T, N, B)$ is a direct orthonormal frame, the
\emph{Frenet frame}. The plane through $\tilde\gamma(s)$ spanned by
$T, N$ is the \emph{osculating plane}.

\begin{theorem}[Frenet formulas in space]\label{thm:b2:curves:frenet3d}
There is a \omterm{def:b2:metric:continuity}{continuous} function $\tau$, the \emph{torsion}\index{torsion}, with
\[
T' = \kappa N, \qquad
N' = -\kappa T + \tau B, \qquad
B' = -\tau N .
\]
\end{theorem}

\begin{proof}
The first formula is the definition of $N$. Each of the vectors
$T, N, B$ has constant \omterm{def:b2:nvs:norm}{norm} $1$ and they are pairwise orthogonal;
differentiating the six relations $\langle X, Y\rangle = \delta_{XY}$
shows the matrix of $(T', N', B')$ in the basis $(T, N, B)$ is
antisymmetric: indeed $\langle X', Y\rangle + \langle X, Y'\rangle = 0$ and
$\langle X', X\rangle = 0$. Its $(N, T)$ entry is $\langle N', T\rangle =
-\langle N, T'\rangle = -\kappa$, and its $(T, B)$-column entry
$\langle T', B\rangle = \kappa\langle N, B\rangle = 0$. Naming the remaining
free entry $\tau = \langle N', B\rangle$ gives exactly the three displayed
formulas: antisymmetry fills in $\langle B', N\rangle = -\tau$ and
$\langle B', T\rangle = 0$. \omterm{def:b2:metric:continuity}{Continuity} of $\tau = \langle N', B\rangle$ is clear
since $N'$ and $B$ are \omterm{def:b2:metric:continuity}{continuous} ($\tilde\gamma$ is
$\mathcal{C}^3$, so $N = T'/\kappa$ is $\mathcal{C}^1$).
\end{proof}

\begin{example}[The Darboux vector]\label{ex:b2:curves:darboux}
The three Frenet formulas compress into one. Set $\omega(s) =
\tau\,T + \kappa\,B$ (the \emph{Darboux vector}). Using $B
\wedge T = N$, $T \wedge N = B$, $N \wedge B = T$:
\[
\omega \wedge T = \kappa\,N = T', \qquad
\omega \wedge N = \tau\,B - \kappa\,T = N', \qquad
\omega \wedge B = -\tau\,N = B' :
\]
each frame vector evolves by $X' = \omega \wedge X$, the
kinematic signature of an \emph{instantaneous rotation} with
angular velocity vector $\omega$. The frame spins at rate
$\norm\omega = \sqrt{\kappa^2 + \tau^2}$ about the moving
axis $\omega$; \omterm{thm:b2:curves:frenet2d}{curvature} is the component of the spin about
the binormal, \omterm{thm:b2:curves:frenet3d}{torsion} the component about the tangent. For
the helix, $\omega$ is a constant vector along the cylinder's
axis --- which is exactly why the helix's frame precesses
evenly. The antisymmetry of the Frenet matrix, exploited in
\cref{exo:b2:curves:7}, is the matrix form of this single
geometric fact.
\end{example}

\begin{proposition}[Torsion measures planarity]\label{prop:b2:curves:torsion}
A biregular arc is contained in a plane if and only if $\tau
\equiv 0$; in that case the plane is the (constant) osculating
plane.
\end{proposition}

\begin{proof}
If $\tau \equiv 0$, then $B' = 0$, so $B$ is a constant unit
vector $B_0$, and
\[
\frac{\dd}{\dd s}\langle \tilde\gamma(s), B_0\rangle
= \langle T(s), B_0\rangle = 0 :
\]
$\langle \tilde\gamma, B_0\rangle$ is constant, so the arc lies in a plane
orthogonal to $B_0$. Conversely, if the arc lies in a plane $P$,
then $T$ and $T'$ (hence $N$) are parallel to the direction of
$P$ for all $s$; so $B = T \wedge N$ is one of the two unit
normals of $P$, and being \omterm{def:b2:metric:continuity}{continuous} it is constant; then $0 = B'
= -\tau N$ with $N \neq 0$ forces $\tau \equiv 0$.
\end{proof}

\begin{example}[A tilted circle has zero torsion]
The arc $\gamma(t) = \bigl(\cos t,\ \tfrac{\sin t}{\sqrt2},\
\tfrac{\sin t}{\sqrt2}\bigr)$ lies in the plane $y = z$, and
is the unit circle of that plane (check: $\norm{\gamma(t)} =
1$ and the plane's orthonormal basis $(1,0,0)$,
$(0, \tfrac1{\sqrt2}, \tfrac1{\sqrt2})$ exhibits the standard
parametrization). Without any Frenet computation,
\cref{prop:b2:curves:torsion} predicts $\tau \equiv 0$, and
the fixed binormal must be the plane's unit normal $\pm(0,
\tfrac1{\sqrt2}, -\tfrac1{\sqrt2})$. \omterm{thm:b2:curves:frenet3d}{Torsion} does not measure
being ``tilted in space''; it measures \emph{leaving} a
plane. Only the helix's nonzero $b$ below produces genuine
\omterm{thm:b2:curves:frenet3d}{torsion}.
\end{example}

\begin{example}[The helix]\label{ex:b2:curves:helixfrenet}
For the helix $\gamma(t) = (a\cos t, a\sin t, bt)$, $a > 0$, we
computed $s = ct$ with $c = \sqrt{a^2 + b^2}$. Then
\[
T = \frac1c(-a\sin t,\ a\cos t,\ b),
\qquad
T' \cdot \frac{\dd t}{\dd s}
= \frac{1}{c^2}(-a\cos t, -a\sin t, 0),
\]
so $\kappa = a/c^2 = a/(a^2 + b^2)$ and $N = (-\cos t, -\sin t,
0)$: the principal normal points horizontally toward the axis.
Next $B = T \wedge N = \frac1c(b\sin t, -b\cos t, a)$, and $B'
\frac{\dd t}{\dd s} = \frac{b}{c^2}(\cos t, \sin t, 0) = -\tau N$
gives
\[
\boxed{\ \kappa = \frac{a}{a^2 + b^2}, \qquad
\tau = \frac{b}{a^2 + b^2}. \ }
\]
Both invariants are constant --- and one can show, conversely, that
the only biregular curves with constant $\kappa > 0$ and constant
$\tau$ are helices (circles when $\tau = 0$). Note the signs: $b >
0$ gives a right-handed helix with positive \omterm{thm:b2:curves:frenet3d}{torsion}.
\end{example}

\begin{example}[Curvature and torsion without arc length; the
twisted cubic]\label{ex:b2:curves:twistedcubic}
Reparametrizing by \omterm{def:b2:curves:length}{arc length} is usually impossible in closed
form, so the invariants must be extracted from the raw
derivatives. Write $v = \norm{\gamma'} = s'$; then $\gamma' =
vT$ and, as in the proof of
\cref{prop:b2:curves:kappaformula},
\[
\gamma'' = v'T + v^2\kappa N,
\qquad
\gamma' \wedge \gamma'' = v^3\kappa\,(T \wedge N) =
v^3\kappa\,B .
\]
Taking \omterm{def:b2:nvs:norm}{norms} ($\kappa \geq 0$ in space):
\[
\kappa = \frac{\norm{\gamma' \wedge \gamma''}}{v^3} .
\]
Differentiating $\gamma''$ once more and converting $N' =
v(-\kappa T + \tau B)$ (chain rule through $s$), the only
$B$-component comes from the last term:
\[
\gamma''' = \bigl(v'' - v^3\kappa^2\bigr)T +
\bigl(v'v\kappa + (v^2\kappa)'\bigr)N + v^3\kappa\tau\,B,
\]
so that, pairing with $\gamma' \wedge \gamma'' = v^3\kappa B$,
\[
\det(\gamma', \gamma'', \gamma''') = \langle\gamma' \wedge
\gamma'',\ \gamma'''\rangle = v^6\kappa^2\tau,
\qquad\text{i.e.}\qquad
\tau = \frac{\det(\gamma', \gamma'',
\gamma''')}{\norm{\gamma' \wedge \gamma''}^2} .
\]
Application to the \emph{twisted cubic} $\gamma(t) = (t,\
t^2,\ t^3)$ at $t = 0$: $\gamma' = (1, 0, 0)$, $\gamma'' =
(0, 2, 0)$, $\gamma''' = (0, 0, 6)$, so $v = 1$,
\[
\gamma' \wedge \gamma'' = (0, 0, 2), \qquad \kappa(0) = 2,
\qquad
\det(\gamma', \gamma'', \gamma''') = 12, \qquad \tau(0) =
\frac{12}{4} = 3 .
\]
Closing insight: both formulas are ratios in which the speed
$v$ cancels to the exact degree needed --- $\kappa$ scales
like a second derivative per unit \omterm{def:b2:curves:length}{length}, $\tau$ like the
mixed volume of three derivatives per squared area --- which
is \emph{why} they are geometric while $\gamma''$ itself is
not.
\end{example}

\begin{remark}[Fundamental theorem for space curves]
As in the plane, the pair $(\kappa, \tau)$ with $\kappa > 0$
determines a biregular arc up to direct isometry of $\R^3$: the
Frenet formulas form a linear differential system for the frame
$(T, N, B)$, to which the Cauchy--Lipschitz theory of
\cref{ch:b2:diffeq} applies; orthonormality of the solution frame
is preserved because the coefficient matrix is antisymmetric (same
Gram-matrix argument as in \cref{exo:b2:curves:7}), and the curve
is recovered by integrating $T$. We leave the details to the
reader as a substantial but instructive exercise.
\end{remark}

\section{Local study: position with respect to the tangent}

\begin{proposition}[Local shape at a regular
point]\label{prop:b2:curves:local}
Let $\gamma$ be a plane arc of class $\mathcal{C}^k$ at $t_0$, with
$p$ the smallest index with $\gamma^{(p)}(t_0) \neq 0$ and $q$ the
smallest index $> p$ with $\gamma^{(q)}(t_0)$ not collinear with
$\gamma^{(p)}(t_0)$ (assuming both exist, $q \leq k$). In the basis
$(u, v) = (\gamma^{(p)}(t_0), \gamma^{(q)}(t_0))$ centered at
$\gamma(t_0)$, Taylor--Young gives coordinates
\[
X(t) \sim \frac{(t - t_0)^p}{p!},
\qquad
Y(t) \sim \frac{(t - t_0)^q}{q!} .
\]
The local picture depends only on the parities of $p$ and $q$:
\begin{center}
\begin{tabular}{c c l}
$p$ odd, $q$ even & \emph{ordinary point} & curve crosses no line,
stays on one side of tangent\\
$p$ odd, $q$ odd & \emph{inflection point} & curve crosses its
tangent\\
$p$ even, $q$ odd & \emph{cusp of the first kind} & both branches
on opposite sides of tangent\\
$p$ even, $q$ even & \emph{cusp of the second kind} & both
branches on the same side\\
\end{tabular}
\end{center}
\end{proposition}

\begin{proof}
Taylor--Young at order $q$ (the function $\gamma$ is
$\mathcal{C}^q$ near $t_0$):
\[
\gamma(t) - \gamma(t_0)
= \sum_{j=p}^{q} \frac{(t-t_0)^j}{j!}\,\gamma^{(j)}(t_0)
+ o\bigl((t-t_0)^q\bigr).
\]
By the choice of $p$ and $q$, each $\gamma^{(j)}(t_0)$ with $p
\leq j < q$ is collinear with $u$; collecting components in the
basis $(u, v)$: $X(t) = \frac{(t-t_0)^p}{p!}(1 + o(1))$ and
$Y(t) = \frac{(t-t_0)^q}{q!}(1 + o(1))$. The sign table of $X$ and
$Y$ for $t \gtrless t_0$ --- governed exactly by the parities ---
gives the four pictures: for instance if $p$ is even, $X > 0$ on
both sides (both branches leave in the direction $+u$: a cusp),
and the side of the \omterm{def:b2:curves:arc}{tangent line} ($\operatorname{sign} Y$) flips
with $q$ odd.
\end{proof}

\begin{example}
For $\gamma(t) = (t^2, t^3)$ at $t_0 = 0$
(\cref{ex:b2:curves:cusp}): $\gamma'' (0)= (2, 0)$, $\gamma'''(0) =
(0, 6)$, so $p = 2$, $q = 3$: a cusp of the first kind, the
familiar picture of the semicubical parabola. For $\gamma(t) = (t,
t^3)$ at $0$: $p = 1$, $q = 3$: inflection --- the cubic crosses
its tangent.
\end{example}

\begin{remark}[Perspectives within this volume]
Curves feed the next chapters in three ways. Drawn on a
surface, they define its tangent planes and its first
fundamental form (\cref{ch:b2:surfaces}), and their \omterm{def:b2:curves:length}{lengths}
are computed by restricting the ambient metric --- the
chapter ahead is largely this chapter relativized. The
\omterm{pb:b2:curves:1}{envelope} calculus of the weekend problem meets double
integrals in \cref{ch:b2:multint}, where the \omterm{pb:b2:curves:1}{astroid}'s area
is recomputed by Green's formula
(\cref{exo:b2:multint:5}) --- one curve, two theories,
matching answers. And the Frenet system already used the
linear differential equations of \cref{ch:b2:diffeq}
(existence, uniqueness, and the orthogonality-preservation
argument of \cref{exo:b2:curves:7}): the fundamental theorem
of curves is a differential equations theorem wearing
geometric clothes.
\end{remark}

\begin{remark}[Method: running the local study]
In practice the classification is a four-step routine.
\emph{One}, differentiate at $t_0$ until the first nonzero
derivative appears: its index is $p$, its value the vector
$u$. \emph{Two}, keep differentiating until a derivative
not collinear with $u$ appears: index $q$, vector $v$.
\emph{Three}, read the parities $(p, q)$ in the table.
\emph{Four}, draw: the curve leaves along $+u$ if $p$ is odd
(along $u$ then back along $u$ if $p$ is even), on the side
of $v$ dictated by the sign of $Y$. Two cautions. The frame
$(u, v)$ is generally \emph{not} orthonormal --- the table
describes positions relative to the \omterm{def:b2:curves:arc}{tangent line}, not angles
or distances, so do not read \omterm{thm:b2:curves:frenet2d}{curvature} off the picture.
And intermediate derivatives collinear with $u$ are allowed
between ranks $p$ and $q$ (they only shift the expansion of
$X$); what must \emph{not} happen is stopping at the first
nonzero derivative and guessing $q = p + 1$: for $\gamma(t)
= (t^2, t^4 + t^5)$ the naive guess $q = 3$ is wrong, as
$\gamma^{(3)}(0)$ is still collinear with $\gamma''(0)$ ---
this is precisely \cref{exo:b2:curves:5}.
\end{remark}

\begin{problem}[{Weekend problem --- the envelope machine and
four classical curves}]\label{pb:b2:curves:1}
A one-parameter family of lines usually fails to cover the
plane evenly: the lines pile up along a curve tangent to all
of them, their \emph{envelope}\index{envelope}. Light rays
make \omterm{pb:b2:curves:1}{envelopes} visible as \emph{caustics} --- the bright
cusped curve in a mug of coffee. This problem builds the
general \omterm{pb:b2:curves:1}{envelope} machine, then runs it four times: the sliding
ladder (\omterm{pb:b2:curves:1}{astroid}), the normals of the parabola and of the
cycloid (evolutes, with Huygens' pendulum at the end), and the
coffee-cup caustic (\omterm{pb:b2:curves:1}{nephroid}). Throughout, $D_t$ denotes the
line of equation $a(t)\,x + b(t)\,y = c(t)$, where $a, b, c$
are $\mathcal C^2$ functions with $(a(t), b(t)) \neq (0,0)$,
and $\Delta(t) = a(t)b'(t) - a'(t)b(t)$.

\medskip
\noindent\textbf{Part I --- The \omterm{pb:b2:curves:1}{envelope} machine.}
\begin{enumerate}
  \item Suppose $\Delta(t) \neq 0$. Show that the
        \emph{characteristic system}
        \[
        \begin{cases} a(t)\,x + b(t)\,y = c(t)\\
        a'(t)\,x + b'(t)\,y = c'(t)\end{cases}
        \]
        has a unique solution $E(t) = (x(t), y(t))$, given by
        $x = \dfrac{cb' - c'b}{\Delta}$, $y = \dfrac{ac' -
        a'c}{\Delta}$.
  \item Assume moreover that $E$ is $\mathcal C^1$ near $t$
        with $E'(t) \neq 0$. Differentiating the first
        equation of the system, show $a(t)\,x'(t) +
        b(t)\,y'(t) = 0$, and conclude that the curve $E$
        passes through a point of $D_t$ \emph{with the
        direction of} $D_t$: the family is tangent to $E$,
        which is called its \emph{\omterm{pb:b2:curves:1}{envelope}}.
  \item Sanity check: the \omterm{def:b2:curves:arc}{tangent lines} of the parabola $y =
        x^2/2$ at the points $(t, t^2/2)$ are $tx - y =
        t^2/2$. Verify that the \omterm{pb:b2:curves:1}{envelope} machine returns the
        parabola itself.
  \item (\omterm{pb:b2:curves:1}{Envelope} of the normals) Let $\gamma$ be unit-speed
        with $\kappa(s) \neq 0$. The normal line at
        $\gamma(s)$ is $\{M : \langle M - \gamma(s), T(s)
        \rangle = 0\}$. Show that its characteristic system
        forces $\langle M - \gamma(s), N(s)\rangle =
        1/\kappa(s)$, hence that the characteristic point is
        the \omterm{def:b2:curves:curvature}{center of curvature}: \emph{the \omterm{pb:b2:curves:1}{envelope} of the
        normals is the evolute}, recovering
        \cref{exo:b2:curves:6}. Check $\Delta(s) =
        \kappa(s)$.
  \item Two degenerations. For the pencil $D_\theta : x\cos
        \theta + y\sin\theta = 0$, show that the
        characteristic point is the origin for every $\theta$
        (the ``\omterm{pb:b2:curves:1}{envelope}'' collapses to a point, and $E' = 0$:
        question 2 does not apply). For a family of parallel
        lines ($a, b$ constant), show $\Delta \equiv 0$ and
        that the characteristic system is in general
        inconsistent: no \omterm{pb:b2:curves:1}{envelope}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The sliding ladder and the
\omterm{pb:b2:curves:1}{astroid}.} A segment of \omterm{def:b2:curves:length}{length} $1$ slides with one end $P_t =
(\cos t, 0)$ on the floor and the other $Q_t = (0, \sin t)$ on
the wall, $t \in \intoo0{\pi/2}$.
\begin{enumerate}[resume]
  \item Show that the line $(P_tQ_t)$ has equation $x\sin t +
        y\cos t = \sin t\cos t$, and that the \omterm{pb:b2:curves:1}{envelope} machine
        gives the characteristic point
        \[
        E(t) = (\cos^3 t,\ \sin^3 t) :
        \]
        the \emph{astroid}\index{astroid}, of implicit
        equation $x^{2/3} + y^{2/3} = 1$ (extended to the
        other quadrants by symmetry).
  \item Show that $E'(0) = 0$ and, using the local
        classification (\cref{prop:b2:curves:local}), that
        the \omterm{pb:b2:curves:1}{astroid} has a cusp of the first kind at $(1, 0)$
        --- and likewise at its four axis points.
  \item Compute $\norm{E'(t)} = \tfrac32\abs{\sin 2t}$ and
        deduce that the total \omterm{def:b2:curves:length}{length} of the \omterm{pb:b2:curves:1}{astroid} is $6$.
  \item Where does the ladder touch the \omterm{pb:b2:curves:1}{astroid}? Show $E(t) =
        P_t + \sin^2 t\,(Q_t - P_t)$: the contact point
        divides the ladder in the ratio $\sin^2 t : \cos^2
        t$, sweeping it from one end to the other as the
        ladder slides.
  \item Compute the area enclosed by the \omterm{pb:b2:curves:1}{astroid}: show that
        the first-quadrant area is $3\int_0^{\pi/2}\sin^4
        t\cos^2 t\,\dd t$, evaluate the integral by
        linearization ($\sin^2 2t = \tfrac{1 - \cos 4t}2$),
        and conclude that the total area is $3\pi/8$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The evolute of the parabola.}
Let $\gamma(t) = (t, t^2/2)$.
\begin{enumerate}[resume]
  \item Show that the normal line at $\gamma(t)$ has equation
        $x + t\,y = t + t^3/2$.
  \item Run the \omterm{pb:b2:curves:1}{envelope} machine: show that the \omterm{pb:b2:curves:1}{envelope} of
        the normals is
        \[
        E(t) = \Bigl(-t^3,\ 1 + \tfrac32 t^2\Bigr),
        \]
        with implicit equation $x^2 = \tfrac8{27}(y - 1)^3$:
        a semicubical parabola.
  \item Cross-check with question 4: compute the \omterm{def:b2:curves:curvature}{center of
        curvature} $\gamma(t) + \frac1{\kappa(t)}N(t)$ from
        $\kappa(t) = (1 + t^2)^{-3/2}$
        (\cref{ex:b2:curves:kappaexamples}) and recover the
        same point.
  \item Show that the evolute has a cusp of the first kind at
        $(0, 1)$, the \omterm{def:b2:curves:curvature}{center of curvature} at the vertex ---
        the point where $\kappa$ is extremal, as predicted by
        the formula $c' = -\frac{\kappa'}{\kappa^2}N$ of
        \cref{exo:b2:curves:6}.
  \item How many normals of the parabola pass through a given
        point $(x_0, y_0)$? Show the answer is governed by
        the cubic $\tfrac{t^3}2 + (1 - y_0)\,t - x_0 = 0$;
        treat the axis case $x_0 = 0$ completely (one normal
        for $y_0 < 1$, three for $y_0 > 1$), and interpret
        the evolute as the transition curve.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The coffee-cup caustic.} Parallel
rays of direction $(1, 0)$ strike the inside of the mirror
circle $x^2 + y^2 = 1$; the ray hitting $P_\theta =
(\cos\theta, \sin\theta)$ reflects according to the law of
reflection.
\begin{enumerate}[resume]
  \item From the mirror symmetry in the normal (the radius),
        justify that the reflected direction is $v = u -
        2\langle u, n\rangle n$ with $u = (1,0)$, $n =
        (\cos\theta, \sin\theta)$, and compute $v =
        -(\cos2\theta, \sin2\theta)$.
  \item Show that the reflected ray lies on the line
        \[
        x\sin 2\theta - y\cos 2\theta = \sin\theta .
        \]
  \item Run the \omterm{pb:b2:curves:1}{envelope} machine ($\Delta = 2$): show that
        the caustic is
        \[
        E(\theta) = \Bigl(\tfrac{3\cos\theta -
        \cos3\theta}4,\ \tfrac{3\sin\theta -
        \sin3\theta}4\Bigr),
        \]
        the \emph{nephroid}\index{nephroid}.
  \item Compute $E'(\theta) = \tfrac32\sin\theta\,
        (\cos2\theta, \sin2\theta)$; check that the tangent
        direction is the reflected-ray direction (question
        16), locate the two cusps $(\pm\tfrac12, 0)$, and
        show that the reflected ray crosses the axis $y = 0$
        at $x = \frac1{2\cos\theta}$ --- so nearly axial rays
        focus at $x = \tfrac12$: the focal \omterm{def:b2:curves:length}{length} $R/2$ of a
        mirror of radius $R$.
  \item Show that the \omterm{pb:b2:curves:1}{nephroid} has total \omterm{def:b2:curves:length}{length} $6$ and that
        near $\theta = 0$,
        \[
        E(\theta) - \bigl(\tfrac12, 0\bigr) =
        \bigl(\tfrac34\theta^2 + o(\theta^2),\ \theta^3 +
        o(\theta^3)\bigr) :
        \]
        a cusp of the first kind, pointing along the axis.
  \item Explain in one paragraph why the caustic is bright:
        through every point just outside the caustic pass two
        reflected rays, through every point on it the rays
        are ``infinitely concentrated'' (the map $(\theta,
        \text{distance along ray}) \mapsto \R^2$ has a
        critical point exactly on the \omterm{pb:b2:curves:1}{envelope}).
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Huygens: the cycloid is its own
evolute.} Let $\gamma(t) = (t - \sin t,\ 1 - \cos t)$, $t \in
\intoo0{2\pi}$, one arch of the cycloid.
\begin{enumerate}[resume]
  \item Compute $\kappa(t) = -\dfrac1{4\sin(t/2)}$ and the
        \omterm{def:b2:curves:curvature}{center of curvature}; show that the evolute is
        \[
        c(t) = (t + \sin t,\ \cos t - 1),
        \]
        and that the substitution $t = u + \pi$ exhibits it
        as the original cycloid translated by $(\pi, -2)$:
        \emph{the evolute of a cycloid is a congruent
        cycloid} (Huygens).
  \item Verify that the \omterm{def:b2:curves:curvature}{radius of curvature} at the apex $t =
        \pi$ equals $4$, which is half the \omterm{def:b2:curves:length}{length} $8$ of one
        arch (\cref{exo:b2:curves:1}); locate the cusp of the
        evolute directly below the apex, at distance $4$.
  \item (The string property) Let $\gamma$ be unit-speed with
        $\kappa > 0$, $\kappa$ of class $\mathcal C^1$ and
        $R = 1/\kappa$ strictly monotone. Using $c' = R'N$,
        show that the \omterm{def:b2:curves:length}{arc length} of the evolute between
        $c(s_0)$ and $c(s_1)$ is $\abs{R(s_1) - R(s_0)}$.
        Interpret: a taut string unwound from the evolute,
        of \omterm{def:b2:curves:length}{length} $R(s_0)$ at the start, has its free end
        trace the original curve --- so a pendulum swinging
        between two cycloidal cheeks of Huygens' clock
        describes a cycloid.
  \item Synthesis. The machine of Part I produced the
        \omterm{pb:b2:curves:1}{astroid}, a semicubical parabola, a \omterm{pb:b2:curves:1}{nephroid} and a
        cycloid. For each of the four families, state in one
        sentence where the hypotheses $\Delta \neq 0$ and
        $E' \neq 0$ held or failed, and what geometric event
        (cusp, focus, degeneracy) each failure of $E' \neq 0$
        signalled. Where must extrema of \omterm{thm:b2:curves:frenet2d}{curvature} appear on
        the \omterm{pb:b2:curves:1}{envelope} of the normals, and why?
\end{enumerate}
\end{problem}
