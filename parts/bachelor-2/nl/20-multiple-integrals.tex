\chapter{Lijnintegralen en meervoudige integralen}\label{ch:b2:multint}

Dit hoofdstuk breidt integratie uit van intervallen naar krommen
en naar domeinen van het vlak en de ruimte. Lijnintegralen
integreren een \emph{\omterm{def:b2:multint:lineint}{differentiaalvorm}} $P\,\dd x + Q\,\dd y$
langs een georiënteerde boog; dubbel- en driedubbelintegralen
integreren functies over twee- en driedimensionale domeinen. De
twee theorieën ontmoeten elkaar in de \emph{stelling van
Green--Riemann}, de tweedimensionale fundamentele stelling van de
analyse, en het hoofdcomputatiegereedschap doorheen is de
\emph{variabelensubstitutieformule}, waarvan de vervormingsfactor
de absolute Jacobiaandeterminant is.

\section{Lijnintegralen}

\begin{definition}[Differentiaalvorm;
lijnintegraal]\label{def:b2:multint:lineint}
Zij $U \subseteq \R^2$ open. Een
\emph{differentiaalvorm}\index{differentiaalvorm} van graad $1$
en klasse $\mathcal{C}^0$ op $U$ is een uitdrukking $\omega =
P\,\dd x + Q\,\dd y$ met $P, Q \colon U \to \R$ continu ---
formeel, een continue afbeelding van $U$ in de duale van
$\R^2$, $\omega(M) = P(M)\,e_1^* + Q(M)\,e_2^*$. Voor een
$\mathcal{C}^1$-boog $\gamma \colon [a, b] \to U$, $\gamma(t)
= (x(t), y(t))$, is de
\emph{lijnintegraal}\index{lijnintegraal} van $\omega$ langs
$\gamma$
\[
\int_\gamma \omega
= \int_a^b \Bigl(P(\gamma(t))\,x'(t)
+ Q(\gamma(t))\,y'(t)\Bigr)\,\dd t .
\]
De definities strekken zich letterlijk uit tot $\R^3$ (vormen
$P\,\dd x + Q\,\dd y + R\,\dd z$) en tot stuksgewijs
$\mathcal{C}^1$-bogen (som over de stukken).
\end{definition}

\begin{proposition}[Invariantie en
oriëntatie]\label{prop:b2:multint:lineinv}
De \omterm{def:b2:multint:lineint}{lijnintegraal} is onveranderd onder een stijgende
$\mathcal{C}^1$-parameterwisseling, en wisselt van teken onder
een dalende. Ze hangt dus alleen af van de
\emph{georiënteerde} \omterm{def:b2:curves:reparam}{meetkundige boog}.
\end{proposition}

\begin{proof}
Als $\theta \colon [c, d] \to [a, b]$ een \omterm{def:b2:curves:reparam}{parameterwisseling} is
en $\tilde\gamma = \gamma \circ \theta$, dan door de
kettingregel en de éénvariabele variabelensubstitutie $t =
\theta(u)$,
\[
\int_{\tilde\gamma}\omega
= \int_c^d \bigl(P(\gamma(\theta(u)))\,x'(\theta(u))
+ Q(\gamma(\theta(u)))\,y'(\theta(u))\bigr)\,\theta'(u)\,\dd u
= \pm\int_a^b \bigl(Px' + Qy'\bigr)(t)\,\dd t ,
\]
met teken $+$ als $\theta$ stijgend is ($\theta(c) = a$) en $-$
als dalend (de grenzen wisselen).
\end{proof}

\begin{example}[Arbeid van een kracht;
circulatie]\label{ex:b2:multint:work}
Als $F = (P, Q)$ een krachtenveld is, is $\int_\gamma P\dd x +
Q\dd y = \int_a^b \langle F(\gamma(t)), \gamma'(t)\rangle\,\dd
t$ de \emph{arbeid} van $F$ langs $\gamma$. Voor $\omega =
-y\,\dd x + x\,\dd y$ langs de linkse eenheidscirkel
$\gamma(t) = (\cos t, \sin t)$:
\[
\int_\gamma \omega
= \int_0^{2\pi}\bigl((-\sin t)(-\sin t)
+ \cos t\cos t\bigr)\,\dd t = 2\pi ,
\]
tweemaal de ingesloten \omterm{def:b2:multint:domain}{oppervlakte} --- een eerste hint van
Green--Riemann.
\end{example}

\begin{example}[Eén integraal, twee parametrisaties, één
tekensval]\label{ex:b2:multint:semicircle}
Bereken $\int_\gamma x\,\dd y$ langs de bovenste
eenheidshalfcirkel van $(1, 0)$ naar $(-1, 0)$. Met $\gamma(t)
= (\cos t, \sin t)$, $t \in \intcc0\pi$:
\[
\int_0^\pi\cos t\cdot\cos t\,\dd t = \frac\pi2 .
\]
Met de grafiekparametrisatie $x \mapsto (x, \sqrt{1 -
x^2})$, $x$ van $1$ naar $-1$ (let op de richting!):
\[
\int_1^{-1}x\cdot\frac{-x}{\sqrt{1 - x^2}}\,\dd x
= \int_{-1}^{1}\frac{x^2}{\sqrt{1 - x^2}}\,\dd x
= \frac\pi2
\]
($x = \sin u$ herleidt het tot een Wallis-integraal).
Dezelfde waarde, zoals \cref{prop:b2:multint:lineinv}
garandeert --- maar alleen omdat beide runs van $(1,0)$ naar
$(-1,0)$ gaan; de reis omkeren keert het teken om. De baan
langs de $x$-as sluiten (waar $\dd y = 0$) voegt niets toe, en
het totaal $\frac\pi2$ is de \omterm{def:b2:multint:domain}{oppervlakte} van de halfschijf: de
eerste instantie van de rand-oppervlakteformules van
Green--Riemann hieronder.
\end{example}

\begin{definition}[Exacte en gesloten
vormen]\label{def:b2:multint:exact}
De vorm $\omega = P\,\dd x + Q\,\dd y$ van klasse
$\mathcal{C}^0$ is \emph{exact}\index{exacte vorm} op $U$ als
er $f \in \mathcal{C}^1(U)$ is (een
\emph{potentiaal}\index{potentiaal}) met $\omega = \dd f$,
d.w.z.\ $P = f_x$ en $Q = f_y$. Een $\mathcal{C}^1$-vorm is
\emph{gesloten}\index{gesloten vorm} als $P_y = Q_x$ op $U$.
\end{definition}

\begin{theorem}[Fundamentele stelling voor
lijnintegralen]\label{thm:b2:multint:ftc}
Als $\omega = \dd f$ \omterm{def:b2:multint:exact}{exact} is en $\gamma$ een stuksgewijs
$\mathcal{C}^1$-boog in $U$ van $A$ naar $B$, dan
\[
\int_\gamma \omega = f(B) - f(A) .
\]
In het bijzonder is de integraal van een \omterm{def:b2:multint:exact}{exacte vorm} langs
elke \omterm{def:b2:multint:exact}{gesloten} boog nul, en is elke exacte $\mathcal{C}^1$-vorm
\omterm{def:b2:multint:exact}{gesloten}.
\end{theorem}

\begin{proof}
$\frac{\dd}{\dd t}f(\gamma(t)) = f_x(\gamma(t))x'(t) +
f_y(\gamma(t))y'(t)$ door de kettingregel
(\cref{ch:b2:diffcalc}), dus is de integrand in
\cref{def:b2:multint:lineint} de afgeleide van $t \mapsto
f(\gamma(t))$, en de fundamentele stelling van de analyse
geeft het resultaat op elk stuk; de intermediaire waarden
telescoperen. Geslotenheid van exacte $\mathcal{C}^1$-vormen
is de stelling van Schwarz: $P_y = f_{xy} = f_{yx} = Q_x$.
\end{proof}

\begin{example}[Een potentiaal
reconstrueren]\label{ex:b2:multint:potential}
Zij $\omega = y\,\eu^{xy}\,\dd x + (x\,\eu^{xy} + 2y)\,\dd y$
op $\R^2$. Ze is \omterm{def:b2:multint:exact}{gesloten}: beide kruisafgeleiden zijn gelijk
aan $\eu^{xy}(1 + xy)$. Om een \omterm{def:b2:multint:exact}{potentiaal} te vinden, integreer
$P$ in $x$ bij vaste $y$:
\[
f(x, y) = \int y\,\eu^{xy}\,\dd x = \eu^{xy} + c(y),
\]
pas daarna $c$ aan door $f_y$ te matchen: $x\,\eu^{xy} +
c'(y) = x\,\eu^{xy} + 2y$ geeft $c(y) = y^2$. Dus $f(x,y) =
\eu^{xy} + y^2$, en voor elke stuksgewijs $\mathcal C^1$-boog
van $(0,0)$ naar $(1,1)$,
\[
\int_\gamma\omega = f(1,1) - f(0,0) = (\eu + 1) - 1 = \eu ,
\]
onafhankelijk van de baan --- het tweestapsrecept (integreer
in $x$, corrigeer in $y$) is de praktische omkering van
\cref{thm:b2:multint:ftc} op domeinen waar gesloten vormen
\omterm{def:b2:multint:exact}{exact} zijn.
\end{example}

\begin{example}[Gesloten impliceert niet
exact]\label{ex:b2:multint:angleform}
Op $U = \R^2 \setminus \{0\}$ is de \emph{hoekvorm}
\[
\omega = \frac{-y\,\dd x + x\,\dd y}{x^2 + y^2}
\]
\omterm{def:b2:multint:exact}{gesloten} (directe berekening: zowel $P_y$ als $Q_x$ gelijk
aan $\frac{y^2 - x^2}{(x^2+y^2)^2}$), maar haar integraal
langs de eenheidscirkel is $2\pi \neq 0$ (dezelfde berekening
als \cref{ex:b2:multint:work}, gedeeld door $1$): $\omega$ is
niet \omterm{def:b2:multint:exact}{exact} op $U$. Lokaal is $\omega = \dd\theta$ voor een
bepaling $\theta$ van de poolhoek; het falen is globaal ---
de hoek kan niet continu rond de punctie worden
gedefinieerd. Op domeinen zonder gaten verdwijnt de
pathologie: op een \emph{sterdomein} open verzameling is elke
\omterm{def:b2:multint:exact}{gesloten} $\mathcal{C}^1$-vorm \omterm{def:b2:multint:exact}{exact} (lemma van Poincaré,
\cref{exo:b2:multint:8}).
\end{example}

\section{Dubbelintegralen}

We nemen de theorie van de Riemann-integraal in één
variabele (volume Jaar~1, en \cref{ch:b2:integration}) als
gegeven en schetsen haar tweevariabele versie. Een functie
$f$ continu op een rechthoek $R = [a, b] \times [c, d]$
heeft een dubbelintegraal $\iint_R f$, gedefinieerd door
Riemannsommen over roosters precies als in één variabele, en
berekend door iteratie:

\begin{theorem}[Fubini op een
rechthoek]\label{thm:b2:multint:fubini}
Voor $f$ continu op $R = [a,b] \times [c,d]$,
\[
\iint_R f
= \int_a^b \Bigl(\int_c^d f(x, y)\,\dd y\Bigr)\dd x
= \int_c^d \Bigl(\int_a^b f(x, y)\,\dd x\Bigr)\dd y .
\]
\end{theorem}

\begin{proof}
Stel $F(x) = \int_c^d f(x, y)\,\dd y$. Uniforme continuïteit
van $f$ op het compacte $R$ maakt $F$ continu (gedomineerde
schatting: $\abs{F(x) - F(x')} \leq (d -
c)\sup_y\abs{f(x,y) - f(x',y)}$). Onderverdeel nu $[a,b]$ en
$[c,d]$ in $n$ gelijke delen, wat een rooster van cellen
$R_{ij}$ van \omterm{def:b2:multint:domain}{oppervlakte} $\Delta x\,\Delta y$ geeft. Op elke
cel, $\inf_{R_{ij}} f \cdot \Delta x \Delta y \leq
\int_{x_{i-1}}^{x_i}\int_{y_{j-1}}^{y_j} f(x,y)\,\dd y\,\dd
x \leq \sup_{R_{ij}} f \cdot \Delta x \Delta y$ door
monotoniciteit van de éénvariabele-integraal (tweemaal
toegepast). Optellen over cellen, de geitereerde integraal
$\int_a^b F$ wordt ingeklemd tussen de onder- en
boven-Riemannsommen van het rooster; door uniforme
continuïteit convergeren beide sommen naar de gemeenschappelijke
waarde die $\iint_R f$ definieert als $n \to \infty$.
Hetzelfde argument geldt met de rollen van $x$ en $y$
verwisseld, dus zijn beide geitereerde integralen gelijk aan
$\iint_R f$.
\end{proof}

\begin{remark}
Continuïteit op een compacte rechthoek doet echt werk in het
bewijs van Fubini: ze levert de uniforme continuïteit die de
Riemannsommen inklamt. Voor wildere integranden faalt de
uitspraak echt --- er zijn functies waarvan de twee
geitereerde integralen bestaan en verschillen. De eerlijke
algemene stelling, met integreerbaarheid als enige
hypothese, is de stelling van Fubini voor de
Lebesgue-integraal, bewezen in het volume van Jaar~3; alles
in dit hoofdstuk blijft binnen de continue setting waar het
elementaire bewijs hierboven compleet is.
\end{remark}

\begin{definition}[Elementaire
domeinen]\label{def:b2:multint:domain}
Een domein $D \subseteq \R^2$ is \emph{$y$-elementair} als
\[
D = \{(x, y) : a \leq x \leq b,\
\varphi_1(x) \leq y \leq \varphi_2(x)\}
\]
met $\varphi_1 \leq \varphi_2$ continu op $[a,b]$
($x$-elementair: \omterm{def:b2:quadratic:adjoint}{symmetrisch}). Voor $f$ continu op een
$y$-elementair $D$,
\[
\iint_D f
= \int_a^b\Bigl(
\int_{\varphi_1(x)}^{\varphi_2(x)} f(x,y)\,\dd y\Bigr)\dd x ,
\]
en men controleert (door $f$ met een
benaderingsargument uit te breiden, of door onderverdelen)
dat wanneer $D$ elementair is in beide richtingen de twee
geitereerde integralen overeenkomen. Domeinen gesneden in
eindig veel elementaire stukken worden door additiviteit
behandeld, en de
\emph{oppervlakte}\index{oppervlakte!van een vlak domein}
van $D$ is $\operatorname{Area}(D) = \iint_D 1$.
\end{definition}

\begin{example}\label{ex:b2:multint:triangle}
Op de driehoek $D = \{0 \leq x \leq 1,\ 0 \leq y \leq x\}$:
\[
\iint_D xy \,\dd x\,\dd y
= \int_0^1 x\Bigl(\int_0^x y\,\dd y\Bigr)\dd x
= \int_0^1 x\cdot\frac{x^2}{2}\,\dd x = \frac18 .
\]
De volgorde verwisselen ($x$ van $y$ tot $1$): $\int_0^1
y\bigl(\int_y^1 x\,\dd x\bigr)\dd y = \int_0^1
y\,\frac{1 - y^2}{2}\,\dd y = \frac18$ --- dezelfde waarde,
andere berekening: de volgorde van integratie goed kiezen is
de helft van het vakmanschap.
\end{example}

\begin{example}[Wanneer slechts één volgorde
werkt]\label{ex:b2:multint:swaporder}
Bereken $I = \displaystyle\int_0^1\!\!\int_x^1
\eu^{y^2}\,\dd y\,\dd x$. Zoals geschreven heeft de
binnenintegraal $\int\eu^{y^2}\dd y$ geen elementaire
primitieve: de berekening zit vast. Maar het domein is de
driehoek $0 \leq x \leq y \leq 1$, elementair in beide
richtingen; de volgorde verwisselen,
\[
I = \int_0^1\!\!\int_0^y \eu^{y^2}\,\dd x\,\dd y
= \int_0^1 y\,\eu^{y^2}\,\dd y
= \Bigl[\tfrac12\eu^{y^2}\Bigr]_0^1 = \frac{\eu - 1}{2} .
\]
De binnenvariabele $x$ verscheen nergens in de integrand, dus
haar eerst integreren produceerde precies de factor $y$ die
de buitenintegraal onmiddellijk maakt. Moraal: Fubini is niet
alleen een licentie om te itereren --- het is een licentie om
te \emph{kiezen}, en de juiste volgorde kan een onmogelijke
integraal in een éénregel veranderen. Schets altijd het
domein en lees beide beschrijvingen af voor je begint.
\end{example}

\begin{theorem}[Variabelensubstitutie]\label{thm:b2:multint:changevar}
Zij $\Phi \colon U' \to U$ een $\mathcal{C}^1$-diffeomorfisme
tussen open verzamelingen van $\R^2$, zij $K \subseteq U$ een
\omterm{def:b2:metric:compact}{compact} domein gesneden in elementaire stukken met $K' =
\Phi^{-1}(K)$, en zij $f$ continu op $K$. Dan
\[
\iint_K f(x, y)\,\dd x\,\dd y
= \iint_{K'} f\bigl(\Phi(u, v)\bigr)\,
\abs{\det J_\Phi(u, v)}\,\dd u\,\dd v .
\]
\end{theorem}
\admitted

\begin{remark}
Het complete bewijs --- $\Phi$ benaderen door haar
\omterm{def:b2:diffcalc:differential}{differentiaal} op een fijn rooster en de randcellen
beheersen --- is lang maar niet diep; het wordt \omterm{def:b2:metric:complete}{volledig}
gedaan in de maattheorie van Jaar~3, als gevolg van de
Lebesgue-theorie. De heuristiek is het beeld al gebruikt voor
\omterm{def:b2:multint:domain}{oppervlakte}: een klein vierkant van zijde $\dd u$ in $(u,
v)$ wordt, tot eerste orde, afgebeeld op het parallellogram
opgespannen door $\Phi_u\,\dd u$ en $\Phi_v\,\dd v$, waarvan
de \omterm{def:b2:multint:domain}{oppervlakte} $\abs{\det J_\Phi}\,\dd u\,\dd v$ is
(\cref{lem:b2:surfaces:lagrange}).
\end{remark}

\begin{remark}[Methode: de variabelensubstitutie
kiezen]
Drie reflexen dekken de meeste gevallen.
\emph{Symmetrie van de integrand}: $x^2 + y^2$ roept om
pool, een productstructuur roept om Cartesische assen te
houden. \emph{Vorm van de rand}: randen $u(x,y) = c_1$,
$v(x,y) = c_2$ smeken om de coördinaten $(u, v)$ zelf, zoals
in het hyperbolische-gebiedvoorbeeld hieronder --- het
domein wordt een rechthoek, wat de hele overwinning is.
\emph{Lineaire structuur}: uitdrukkingen in $x + y$ en $x -
y$ nodigen de $45$-gradenrotatie of een shear uit
(\cref{ex:b2:multint:affine}). In alle gevallen drie
vakjes af te vinken voor integreren: de afbeelding is een
bijectie van het nieuwe domein op het oude; haar Jacobiaan
wordt berekend \emph{in de echt gebruikte richting}
(inverteer aan het eind als makkelijker); en de Jacobiaan
treedt binnen met haar absolute waarde.
\end{remark}

\begin{example}[Poolcoördinaten]\label{ex:b2:multint:polar}
$\Phi(\rho, \alpha) = (\rho\cos\alpha,\ \rho\sin\alpha)$
heeft
\[
J_\Phi = \begin{pmatrix}
\cos\alpha & -\rho\sin\alpha\\
\sin\alpha & \rho\cos\alpha
\end{pmatrix},
\qquad \det J_\Phi = \rho ,
\]
dus $\dd x\,\dd y = \rho\,\dd\rho\,\dd\alpha$. Voor de
schijf $D_R$ van straal $R$:
\[
\iint_{D_R} e^{-(x^2 + y^2)}\,\dd x\,\dd y
= \int_0^{2\pi}\!\!\int_0^R e^{-\rho^2}\rho\,\dd\rho\,\dd\alpha
= \pi\bigl(1 - e^{-R^2}\bigr)
\xrightarrow[R\to\infty]{} \pi .
\]
Vergelijken met het vierkant $[-R, R]^2$ (dat ingeklemd is
tussen de schijven $D_R$ en $D_{R\sqrt2}$, alle integranden
positief) geeft $\bigl(\int_{-\infty}^\infty
e^{-x^2}\dd x\bigr)^2 = \pi$:
\[
\boxed{\ \int_{-\infty}^{+\infty} e^{-x^2}\,\dd x = \sqrt{\pi}\ }
\]
--- de Gauss-integraal opnieuw, nu door haar beroemdste
bewijs (vergelijk de éénvariabele-afleiding in
\cref{ch:b2:integration}).
\end{example}

\begin{example}[Affiene
variabelensubstituties]\label{ex:b2:multint:affine}
Voor een \omterm{def:b2:affine:subspace}{affiene afbeelding} $\Phi(u, v) = M(u, v)^{\mathsf
T} + C$ met $M$ inverteerbaar is de Jacobiaan de constante
matrix $M$: \omterm{def:b2:multint:domain}{oppervlakten} worden vermenigvuldigd met de
constante factor $\abs{\det M}$ --- de belofte van
\cref{ch:b2:affine} is nu een stelling. Twee onmiddellijke
gebruiken. De ellips $\frac{x^2}{a^2} + \frac{y^2}{b^2}
\leq 1$ is het beeld van de eenheidsschijf onder $(u, v)
\mapsto (au, bv)$, dus is haar \omterm{def:b2:multint:domain}{oppervlakte} $ab \cdot \pi$
--- geen berekening. En voor de integraal van $f(x + y)$
over het vierkant $K = \intcc01^2$ maakt de shear $\Phi(u,
v) = (u - v, v)$ (\omterm{def:b2:linalg:det}{determinant} $1$) er een integraal van
$f(u)$ over een parallellogram van, die Fubini snijdt bij
constante $u$: met $f = \exp$,
\[
\iint_K \eu^{x+y}\,\dd x\,\dd y
= \Bigl(\int_0^1 \eu^x\,\dd x\Bigr)^2 = (\eu - 1)^2,
\]
zoals de productstructuur bevestigt. Coördinaten kiezen
aangepast aan de integrand --- niet aan het domein --- is
de andere helft van het vakmanschap.
\end{example}

\begin{example}[Coördinaten aangepast aan een
kromlijnig domein]\label{ex:b2:multint:hyperbolicregion}
Zij $D$ het gebied van het eerste kwadrant begrensd door de
hyperbolen $xy = 1$ en $xy = 3$ en de lijnen $y = x$ en $y =
3x$. In de coördinaten $u = xy$, $v = y/x$ wordt het domein
het vierkant $\intcc13 \times \intcc13$; inverteren,
\[
x = \sqrt{u/v}, \qquad y = \sqrt{uv},
\qquad
\det J = x_uy_v - x_vy_u = \frac{1}{2v}
\]
(een tweeregels-berekening met $x = u^{1/2}v^{-1/2}$, $y =
u^{1/2}v^{1/2}$). Vandaar
\[
\operatorname{Area}(D)
= \int_1^3\!\!\int_1^3\frac{\dd u\,\dd v}{2v}
= 2\cdot\frac{\ln 3}{2} = \ln 3 \approx 1.10 .
\]
Proberen $D$ in Cartesische coördinaten te snijden betekent
haar in drie stukken snijden met hyperbolische en lineaire
randen --- haalbaar, vreugdeloos, en foutgevoelig. De moraal
herhaalt \cref{ex:b2:multint:affine} op volle kracht: lees
de randvergelijkingen, en laat \emph{hen} de coördinaten
kiezen; de Jacobiaan converteert dan de celoppervlakte van
het kromlijnige net, precies zoals $\rho$ dat deed voor
poolcoördinaten.
\end{example}

\begin{example}[Gemiddelde
waarden]\label{ex:b2:multint:average}
De \emph{gemiddelde waarde} van $f$ over een domein $D$ is
$\frac1{\operatorname{Area}(D)}\iint_Df$. Voorbeeld: de
gemiddelde afstand tot het centrum voor een uniform gekozen
punt van de schijf van straal $R$ is
\[
\frac{1}{\pi R^2}\int_0^{2\pi}\!\!\int_0^R
\rho\cdot\rho\,\dd\rho\,\dd\alpha
= \frac{2\pi R^3/3}{\pi R^2} = \frac{2R}3 ,
\]
niet $R/2$: uniforme \omterm{def:b2:multint:domain}{oppervlakte} legt meer massa op grote
stralen (de annulus op straal $\rho$ heeft gewicht
evenredig met $\rho$), dus zit het gemiddelde voorbij het
halverwege-merk. Deze factor goed krijgen is precies de
pool-Jacobiaan aan het werk, en dezelfde weging verklaart
het zwaartepunt $\bar z = 3R/8$ van de halfbol later in het
hoofdstuk in plaats van $R/2$.
\end{example}

\section{De stelling van Green--Riemann}

\begin{theorem}[Green--Riemann]\label{thm:b2:multint:green}
Zij $K \subseteq \R^2$ een \omterm{def:b2:metric:compact}{compact} domein dat elementair is in
beide richtingen (of een eindige vereniging van zulke,
geplakt langs segmenten), met rand $\partial K$ een
stuksgewijs $\mathcal{C}^1$ \omterm{def:b2:multint:exact}{gesloten} kromme georiënteerd
\emph{linksom} (het domein blijft links). Voor $P, Q$ van
klasse $\mathcal{C}^1$ op een omgeving van $K$:
\[
\oint_{\partial K} P\,\dd x + Q\,\dd y
= \iint_K \Bigl(\frac{\partial Q}{\partial x}
- \frac{\partial P}{\partial y}\Bigr)\,\dd x\,\dd y .
\]
\end{theorem}

\begin{proof}
Eerst zijn beide zijden additief onder snijden van $K$ langs
een segment in twee stukken $K_1, K_2$: de
dubbelintegralen tellen op door additiviteit van $\iint$;
wat de randintegralen betreft, doorlopen de linkse randen
van $K_1$ en $K_2$ de binnenste snede elk éénmaal, in
\emph{overstaande} richtingen, dus in de som
\[
\oint_{\partial K_1} + \oint_{\partial K_2}
= \oint_{\partial K} + (\text{snede, beide kanten})
= \oint_{\partial K},
\]
heffen de twee passages langs de snede op
(\cref{prop:b2:multint:lineinv}) en overleeft alleen de
buitenrand. Eindig veel sneden itererend, volstaat een
elementair domein. We bewijzen $\oint P\,\dd x = -\iint_K
P_y$ op een $y$-elementair domein $D = \{a \leq x \leq b,\
\varphi_1(x) \leq y \leq \varphi_2(x)\}$; de identiteit
$\oint Q\,\dd y = \iint_K Q_x$ is \omterm{def:b2:quadratic:adjoint}{symmetrisch}
($x$-elementair), en de stelling is hun som.

Bereken de dubbelintegraal door Fubini en de
éénvariabele fundamentele stelling:
\[
\iint_D \frac{\partial P}{\partial y}\,\dd x\,\dd y
= \int_a^b \bigl(P(x, \varphi_2(x)) - P(x, \varphi_1(x))\bigr)
\,\dd x .
\]
Nu bestaat de rand van $D$, linksom, uit: de onderste
grafiek $y = \varphi_1(x)$ van links naar rechts doorlopen,
het rechter verticale segment $x = b$ (omhoog), de bovenste
grafiek $y = \varphi_2(x)$ van \emph{rechts naar links}
doorlopen, het linker verticale segment $x = a$ (omlaag).
Langs de verticale segmenten is $x$ constant, dus dragen ze
$0$ bij aan $\oint P\,\dd x$; de grafieken, geparametriseerd
door $x$, geven
\[
\oint_{\partial D} P\,\dd x
= \int_a^b P(x, \varphi_1(x))\,\dd x
- \int_a^b P(x, \varphi_2(x))\,\dd x
= -\iint_D \frac{\partial P}{\partial y}\,\dd x\,\dd y .
\qedhere
\]
\end{proof}

\begin{corollary}[Oppervlakte via de
rand]\label{cor:b2:multint:area}
Onder de hypotheses van \cref{thm:b2:multint:green},
\[
\operatorname{Area}(K)
= \oint_{\partial K} x\,\dd y
= -\oint_{\partial K} y\,\dd x
= \frac12\oint_{\partial K} x\,\dd y - y\,\dd x .
\]
\end{corollary}

\begin{proof}
Pas Green--Riemann toe op $(P, Q) = (0, x)$, $(-y, 0)$ en
$\frac12(-y, x)$: elke keer $Q_x - P_y = 1$.
\end{proof}

\begin{remark}[Kiezen onder de drie
oppervlakteformules]
De drie randformules zijn gelijk, niet
uitwisselbaar in de praktijk. Gebruik $\oint x\,\dd y$ wanneer
de parametrisatie $\dd y$ eenvoudig maakt (grafieken over de
$y$-as), $-\oint y\,\dd x$ \omterm{def:b2:quadratic:adjoint}{symmetrisch}, en de symmetrische
halfsom wanneer de parametrisatie $x$ en $y$ gelijk behandelt
--- voor de ellips produceerde ze een constante integrand,
zonder enige trigonometrische linearisatie. Op polygonale
randen wordt de halfsom de vetersluitformule van
\cref{exo:b2:multint:12}, het algoritme van landmeters. En
wanneer de rand rechtsom wordt doorlopen door de gegeven
parametrisatie, geven alle drie de formules \emph{min} de
\omterm{def:b2:multint:domain}{oppervlakte} terug: een negatief resultaat is geen
berekeningsfout maar een oriëntatierapport --- keer het
teken om, of de parametrisatie.
\end{remark}

\begin{example}[Oppervlakte van de
ellips]\label{ex:b2:multint:ellipsearea}
Voor $x = a\cos t$, $y = b\sin t$, $t \in [0, 2\pi]$:
\[
\operatorname{Area}
= \frac12\int_0^{2\pi}\bigl(a\cos t \cdot b\cos t
- b\sin t\cdot(-a\sin t)\bigr)\,\dd t
= \frac{ab}{2}\int_0^{2\pi}\dd t = \pi ab .
\]
\end{example}

\begin{example}[Green--Riemann als
kruiscontrole]\label{ex:b2:multint:greencheck}
Neem $P = -y^3$, $Q = x^3$ op de \omterm{def:b2:multint:exact}{gesloten} eenheidsschijf
$D$. Randzijde, met $\gamma(t) = (\cos t, \sin t)$:
\[
\oint_{\partial D}P\,\dd x + Q\,\dd y
= \int_0^{2\pi}\bigl(\sin^4 t + \cos^4 t\bigr)\dd t
= 2\pi\cdot\Bigl(\frac38 + \frac38\Bigr) = \frac{3\pi}2 ,
\]
door linearisatie ($\sin^4 + \cos^4 = \tfrac34 +
\tfrac14\cos4t$). Binnenzijde:
\[
\iint_D(Q_x - P_y)\,\dd x\,\dd y
= \iint_D 3(x^2 + y^2)\,\dd x\,\dd y
= 3\int_0^{2\pi}\!\!\int_0^1\rho^3\,\dd\rho\,\dd\alpha
= \frac{3\pi}2 .
\]
Hetzelfde getal, twee zeer verschillende berekeningen ---
en dat is het praktische nut: welke kant van de
Green-identiteit makkelijker is wordt de berekening, de
andere een verificatie. Voor circulaties van polynomiale
velden rond \omterm{def:b2:multint:exact}{gesloten} krommen is de dubbelintegraal bijna
altijd de makkelijke kant.
\end{example}

\begin{remark}
Green--Riemann verklaart \cref{ex:b2:multint:angleform}:
voor een gesloten vorm ($Q_x = P_y$) verdwijnt de integraal
rond de rand van elk domein bevat in $U$. De hoekvorm faalt
\omterm{def:b2:multint:exact}{exact} te zijn alleen omdat de punctie in de oorsprong
verhindert dat de schijf begrensd door de eenheidscirkel
binnen $U$ ligt --- lijnintegralen van gesloten vormen
detecteren de gaten van het domein. (Verder geduwd wordt
deze observatie de de Rham-cohomologie.)
\end{remark}

\section{Driedubbelintegralen}

De theorie strekt zich tot drie variabelen uit zonder nieuw
idee: Fubini herleidt $\iiint$ tot drie
éénvariabele-integralen (ofwel door \emph{snijden}:
$\iiint_K f = \int\bigl(\iint_{K_z} f\bigr)\dd z$ over de
horizontale sneden $K_z$, ofwel door \emph{stapelen}: eerst
in $z$ integreren langs verticale stokjes), en de
variabelensubstitutieformule geldt met de $3 \times
3$-Jacobiaan.

\begin{example}[Cilindrische en sferische
coördinaten]\label{ex:b2:multint:spherical}
\emph{Cilindrisch} $(x, y, z) = (\rho\cos\alpha,
\rho\sin\alpha, z)$: $\dd x\,\dd y\,\dd z =
\rho\,\dd\rho\,\dd\alpha\,\dd z$. \emph{Sferisch} $(x, y,
z) = (r\cos\theta\cos\varphi,\ r\sin\theta\cos\varphi,\
r\sin\varphi)$ ($\theta$ lengte, $\varphi \in
[-\frac\pi2, \frac\pi2]$ breedte): de $3 \times
3$-determinant langs de laatste rij ontwikkelen,
\[
\det J = r^2\cos\varphi ,
\qquad
\dd x\,\dd y\,\dd z
= r^2\cos\varphi\;\dd r\,\dd\theta\,\dd\varphi .
\]
Volume van de bal van straal $R$:
\[
V = \int_0^R\!\!\int_0^{2\pi}\!\!\int_{-\pi/2}^{\pi/2}
r^2\cos\varphi\;\dd\varphi\,\dd\theta\,\dd r
= \frac{R^3}{3}\cdot 2\pi \cdot 2
= \boxed{\frac43\pi R^3} ,
\]
eindelijk de formule aangenomen in de volumehoofdstukken van
de eerdere boeken aflossend.
\end{example}

\begin{example}[Het tetraëder,
tweemaal]\label{ex:b2:multint:tetra}
Het volume van $T = \{x, y, z \geq 0,\ x + y + z \leq 1\}$,
door stapelen: voor vaste $(x, y)$ in de driehoek $x + y
\leq 1$ loopt $z$ over $\intcc0{1 - x - y}$, dus
\[
V = \int_0^1\!\!\int_0^{1-x}(1 - x - y)\,\dd y\,\dd x
= \int_0^1\frac{(1 - x)^2}{2}\,\dd x = \frac16 .
\]
Door snijden: de snede op hoogte $z$ is de driehoek $\{x, y
\geq 0,\ x + y \leq 1 - z\}$, van \omterm{def:b2:multint:domain}{oppervlakte}
$\frac{(1-z)^2}2$, en $V = \int_0^1\frac{(1-z)^2}2\,\dd z
= \frac16$ opnieuw --- de twee berekeningen zijn dezelfde
integralen in een andere volgorde, wat alles is wat Fubini
claimt. De waarde $\frac16 = \frac13\cdot\frac12\cdot1$ is
de kegelformule (\cref{ex:b2:multint:cone}) met driehoekige
basis, en de $n$-dimensionale versie $1/n!$ wordt door
precies dit snijden bewezen in het weekendprobleem.
\end{example}

\begin{example}[Zwaartepunt van een
halfbol]\label{ex:b2:multint:centroid}
Voor de bovenste halfbol $H$ van straal $R$ ($z \geq 0$) is
de zwaartepunthoogte $\bar z = \frac1{V}\iiint_H z$, met $V
= \frac23\pi R^3$. In sferische coördinaten ($z =
r\sin\varphi$, $\varphi \in \intcc0{\pi/2}$):
\[
\iiint_H z
= \int_0^R r^3\,\dd r\int_0^{2\pi}\dd\theta
\int_0^{\pi/2}\sin\varphi\cos\varphi\,\dd\varphi
= \frac{R^4}4\cdot2\pi\cdot\frac12 = \frac{\pi R^4}4 ,
\]
dus
\[
\bar z = \frac{\pi R^4/4}{2\pi R^3/3} = \frac{3R}8 :
\]
het evenwichtspunt van een massieve hemisfeer zit op drie
achtsten van de straal boven het platte vlak --- onder de
halve hoogte $R/2$, zoals het moet, omdat de massa dikker
is nabij de basis. Elke zwaartepuntberekening heeft deze
vorm: één momentintegraal, één volume, één verhouding, en
een aannemelijkheidscontrole tegen de meetkunde.
\end{example}

\begin{example}[Volume door snijden: de
kegel]\label{ex:b2:multint:cone}
Een kegel van basisoppervlakte $A$ en hoogte $h$ (top omhoog,
basis in $z = 0$): de snede op hoogte $z$ is de basis
geschaald met de factor $(1 - z/h)$, van \omterm{def:b2:multint:domain}{oppervlakte} $A(1 -
z/h)^2$. Vandaar
\[
V = \int_0^h A\Bigl(1 - \frac zh\Bigr)^2\dd z = \frac{Ah}{3} :
\]
het éénderde van de schoolformules, geldig voor \emph{elke}
basisvorm --- snijden maakt er de integraal van een
kwadraat van.
\end{example}

\begin{example}[Integreerbaarheidsdrempels in het
vlak]\label{ex:b2:multint:threshold}
Voor welke $\alpha > 0$ convergeert $\iint_{D}\rho^{-\alpha}\,
\dd x\,\dd y$ op de gepuncteerde eenheidsschijf $D$ (limiet
over annuli $\varepsilon \leq \rho \leq 1$)? In
poolcoördinaten,
\[
\int_0^{2\pi}\!\!\int_\varepsilon^1\rho^{-\alpha}\,
\rho\,\dd\rho\,\dd\alpha
= 2\pi\int_\varepsilon^1\rho^{1-\alpha}\,\dd\rho ,
\]
wat convergeert als $\varepsilon \to 0$ dan en slechts dan
als $1 - \alpha > -1$, d.w.z.\ $\alpha < 2$: in dimensie $2$
is de kritische singulariteitsexponent de dimensie zelf, de
extra $\rho$ van de Jacobiaan verzacht de singulariteit met
één macht. (Evenzo $\alpha < 3$ voor een puntsingulariteit
in de ruimte, via $r^2$.) Radiale boekhouding van dit type
is hoe integreerbaarheid in één oogopslag wordt beslist in
het Lebesgue-kader van Jaar~3 --- en het is de reden dat
$\iiint 1/r$ moeiteloos convergeerde in
\cref{exo:b2:multint:7}.
\end{example}

\begin{remark}[Klassieke valkuilen]
(i) \emph{Oriëntatie}: een \omterm{def:b2:multint:lineint}{lijnintegraal} wisselt van teken
met de reisrichting, en Green--Riemann eist de rand linksom
(domein links); voor een domein met een gat wordt de
binnenrand \emph{rechtsom} doorlopen. (ii) \emph{De
Jacobiaan treedt binnen met een absolute waarde}:
variabelensubstitutie produceert nooit negatieve
\omterm{def:b2:multint:domain}{oppervlakte}, en $\abs{\det}$ vergeten keert typisch precies
teken om wanneer de afbeelding de oriëntatie omkeert. (iii)
\emph{De poolfactor $\rho$}: $\dd x\,\dd y =
\rho\,\dd\rho\,\dd\alpha$, niet $\dd\rho\,\dd\alpha$ --- de
frequentste fout in het hele hoofdstuk; dimensionale
analyse vangt haar, omdat $\dd\rho\,\dd\alpha$ de dimensie
van een lengte heeft, geen \omterm{def:b2:multint:domain}{oppervlakte}. (iv) \emph{Oneigenlijke
dubbelintegralen}: limieten over groeiende schijven en
groeiende vierkanten komen hier overeen omdat de integranden
positief zijn (inklemmen); voor tekenwisselende integranden
kan de limiet van de uitputting afhangen, en er wordt geen
claim gemaakt zonder absolute convergentie. (v)
\emph{Domeinen versus integranden}: een productintegrand op
een niet-productdomein \emph{factoriseert} de integraal
niet --- factorisatie heeft beide nodig, zoals in
\cref{ex:b2:multint:polar}'s vierkant.
\end{remark}

\begin{remark}[Perspectieven binnen dit volume]
De Gauss-integraal hier berekend is stilzwijgend overal in
de waarschijnlijkheidshoofdstukken: de constante $\sqrt\pi$
binnen de formule van Stirling
(\cref{thm:b2:comparison:stirling}) is de integraal van dit
hoofdstuk, en via Stirling fixteert ze de
$1/\sqrt{\pi n}$-asymptotiek van de
terugkeerwaarschijnlijkheden van de random walk in
\cref{ch:b2:proba}. De Wallis-integralen van het
weekendprobleem keren daar ook terug, dezelfde
centrale-binomiaalschattingen sturend. In de andere
richting voltooien de oppervlakte- en volume-elementen van
dit hoofdstuk de meetkunde van \cref{ch:b2:surfaces}, en de
\omterm{pb:b2:multint:1}{formule van Green} herberekent de \omterm{def:b2:multint:domain}{omhullende-oppervlakten}
van \cref{ch:b2:curves} (de \omterm{pb:b2:curves:1}{astroïde}, in
\cref{exo:b2:multint:5}). Eén hoofdstuk, drie diensten:
maat voor meetkunde, constanten voor waarschijnlijkheid, en
de variabelensubstitutiediscipline door beide gebruikt.
\end{remark}

\begin{exercise}[$\star\star$]\label{exo:b2:multint:7}

\begin{exercise}[$\star\star\star$]\label{exo:b2:multint:8}
(Lemma van Poincaré, sterdomeingeval) Zij $U$ stervormig ten
opzichte van $0$ (d.w.z.\ $M \in U \Rightarrow [0, M]
\subseteq U$) en $\omega = P\dd x + Q\dd y$ een \omterm{def:b2:multint:exact}{gesloten}
$\mathcal{C}^1$-vorm op $U$. Definieer
\[
f(x, y) = \int_0^1 \bigl(x\,P(tx, ty) + y\,Q(tx, ty)\bigr)\dd t .
\]
Met differentiatie onder het integraalteken
(\cref{ch:b2:integration}) en $P_y = Q_x$, toon dat $f_x =
P$ en $f_y = Q$: elke gesloten vorm op een stervormig open
domein is \omterm{def:b2:multint:exact}{exact}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:multint:9}
(Dirichlet-integraal door dubbelintegratie) Rechtvaardig en
exploiteer
\[
\int_0^\infty\!\!\int_0^\infty e^{-xy}\sin x\;\dd y\,\dd x
\quad\text{vs}\quad
\int_0^\infty\!\!\int_0^\infty e^{-xy}\sin x\;\dd x\,\dd y
\]
op $[0, A] \times [0, \infty)$: toon $\int_0^A \frac{\sin
x}{x}\dd x = \frac\pi2 - \int_0^\infty
e^{-Ay}\frac{y\sin A + \cos A}{1 + y^2}\dd y$ en herwin
$\int_0^\infty \frac{\sin x}{x}\,\dd x = \frac\pi2$,
vergelijkend met het parameter-integraalbewijs van
\cref{ch:b2:integration}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:multint:10}
(Isoperimetrische ongelijkheid via Wirtinger) Zij $\gamma$
een eenvoudige \omterm{def:b2:multint:exact}{gesloten} $\mathcal{C}^1$-kromme van lengte
$2\pi$, geparametriseerd op \omterm{def:b2:curves:length}{booglengte} op $[0, 2\pi]$, die
\omterm{def:b2:multint:domain}{oppervlakte} $A$ insluit. Met \cref{cor:b2:multint:area},
Parseval en de Wirtinger-ongelijkheid (oefeningen van
\cref{ch:b2:fourier}), bewijs $A \leq \pi$, met gelijkheid
voor de cirkel. \emph{(Normaliseer $\int_0^{2\pi} x(s)\dd s
= 0$; schrijf $2A = \oint x\,\dd y - y\,\dd x$ en grens $2A
\leq \int (x^2 + y'^2)$ zorgvuldig via $2A =
\int_0^{2\pi}(xy' - yx')\dd s$ en $x^2 + y'^2 \geq
2xy'$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:multint:11}
(Momenten van de bal) Voor de bal $B$ van straal $R$ in

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:multint:12}
(Vetersluitformule) Zij $K$ een polygoon met hoekpunten

\operatorname{Area}(K)
= \frac12\sum_{i=1}^m

\end{exercise}

\section{Probleem: het volume van de bal in dimensie $n$}

\begin{problem}[{Weekendprobleem --- $V_n =
\pi^{n/2}/\Gamma(\frac n2 + 1)$, en de vreemdheid van hoge
dimensies}]\label{pb:b2:multint:1}
De schijf heeft \omterm{def:b2:multint:domain}{oppervlakte} $\pi$, de bal volume
$\frac43\pi$ --- en dan? Dit probleem berekent het volume
van de eenheidsbal van $\R^n$ voor elke $n$, tweemaal (door
een snij-recursie gestuurd door de Wallis-integralen, daarna
via de $\Gamma$-functie en de Gauss-integraal van
\cref{ex:b2:multint:polar}), en leest dan de meetkunde af:
de volumes pieken in dimensie vijf en snellen naar nul, en
bijna de hele hoogdimensionale bal verbergt zich in een
dunne schil nabij haar rand.\index{balvolume}\index{Wallis-integralen}
Voor een continue functie op een bal van $\R^n$ wordt de
integraal begrepen als de $n$-voudig geitereerde integraal
(één coördinaat tegelijk snijden, zoals in het hoofdstuk
voor $n \leq 3$); we schrijven $B_n(R)$ voor de \omterm{def:b2:multint:exact}{gesloten} bal
van straal $R$ gecentreerd in $0$, $v_n(R)$ voor haar
volume, en $V_n = v_n(1)$, met $V_0 = 1$ bij conventie.

\medskip
\noindent\textbf{Deel I --- De snij-recursie.}
\begin{enumerate}
  \item Substitueren $x_i = Ru_i$ in elk van de $n$
        geitereerde integralen, toon $v_n(R) = V_nR^n$.
  \item Snijden van $B_n(1)$ langs haar laatste coördinaat,
        toon
        \[
        V_n = V_{n-1}\int_{-1}^{1}(1 -
        t^2)^{\frac{n-1}2}\,\dd t .
        \]
  \item Met $t = \sin\theta$, identificeer de integraal als
        een Wallis-integraal: $\int_{-1}^1(1 -
        t^2)^{\frac{n-1}2}\dd t = 2W_n$, waar $W_n =
        \int_0^{\pi/2}\cos^n\theta\,\dd\theta =
        \int_0^{\pi/2}\sin^n\theta\,\dd\theta$.
  \item Bewijs de twee Wallis-identiteiten (partieel
        integreren; daarna $nW_nW_{n-1}$ telescoperen):
        \[
        W_n = \frac{n-1}nW_{n-2} \quad (n \geq 2),
        \qquad
        W_nW_{n-1} = \frac{\pi}{2n} \quad (n \geq 1).
        \]
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De recursie opgelost.}
\begin{enumerate}[resume]
  \item Combineer vragen 2--4 tot de tweestapsrecursie
        \[
        V_n = \frac{2\pi}{n}\,V_{n-2}
        \qquad (n \geq 2).
        \]
  \item Leid de gesloten vormen af, voor $k \geq 0$:
        \[
        V_{2k} = \frac{\pi^k}{k!},
        \qquad
        V_{2k+1} = \frac{2^{k+1}\pi^k}{1\cdot3\cdot5\cdots
        (2k+1)} .
        \]
  \item Tabuleer $V_1, \dots, V_7$ numeriek. Met de
        verhouding $V_n/V_{n-2} = 2\pi/n$ en de waarden van
        $2W_5$ en $2W_6$, bewijs dat de rij $(V_n)$ stijgt
        tot haar maximum $V_5 = \frac{8\pi^2}{15} \approx
        5.26$ en daarna daalt.
  \item Toon dat $V_n \to 0$ sneller dan elke meetkundige
        rij, en dat $\sum_{n\geq1} V_n$ convergeert: alle
        eenheidsballen samen hebben eindig totaal volume.
  \item Bewijs de genererende identiteit
        \[
        \sum_{k\geq0} V_{2k}\,x^{2k} = \eu^{\pi x^2}
        \qquad (x \in \R),
        \]
        en leid $\sum_{k\geq0}V_{2k} = \eu^\pi \approx
        23.14$ af.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Tweede route: $\Gamma$ en de
Gauss-integraal.}
\begin{enumerate}[resume]
  \item Toon door Fubini (de integrand is een product) dat
        \[
        I_n = \int_{\R^n}\eu^{-\norm
        x^2}\dd x
        = \Bigl(\int_{-\infty}^{+\infty}
        \eu^{-t^2}\dd t\Bigr)^{\!n} = \pi^{n/2},
        \]
        de $n$-dimensionale Gauss-integraal, begrepen als
        limiet over kubussen $\intcc{-R}{R}^n$.
  \item Herinner $\Gamma(s) = \int_0^\infty
        t^{s-1}\eu^{-t}\dd t$
        (\cref{def:b2:integration:gamma}). Uit $\Gamma(s+1)
        = s\,\Gamma(s)$
        (\cref{thm:b2:integration:gammaprops}) en
        $\Gamma(\tfrac12) = \sqrt\pi$ (substitueer $t =
        u^2$ en roep de Gauss-integraal in), bereken
        \[
        \Gamma(k + 1) = k!,
        \qquad
        \Gamma\Bigl(k + \frac32\Bigr) =
        \frac{1\cdot3\cdots(2k+1)}{2^{k+1}}\,\sqrt\pi .
        \]
  \item Bewijs, door inductie via de recursie van vraag 5,
        de enkele formule
        \[
        V_n = \frac{\pi^{n/2}}{\Gamma\bigl(\frac n2 +
        1\bigr)} \qquad (n \geq 1),
        \]
        en controleer dat ze beide gesloten vormen van
        vraag 6 reproduceert.
  \item Toon $\int_0^\infty \eu^{-r^2}r^{n-1}\dd r =
        \tfrac12\Gamma\bigl(\tfrac n2\bigr)$ en leid de
        identiteit
        \[
        I_n = n\,V_n\int_0^\infty \eu^{-r^2}\,r^{n-1}\,\dd r.
        \]
        af. Interpreteer: de Gauss-massa van $\R^n$ wordt
        verzameld langs sferische schillen waarvan de
        ``$(n-1)$-dimensionale \omterm{def:b2:multint:domain}{oppervlakte}'' op straal $r$
        $nV_nr^{n-1}$ is --- beide zijden zijn nu
        onafhankelijk bewezen, dus kost de interpretatie
        niets.
  \item Stel $s_{n-1} = nV_n$ (de \omterm{def:b2:multint:domain}{oppervlakte} van de
        eenheidssfeer $S^{n-1}$, consistent met $v_n(R) =
        \int_0^R s_{n-1}r^{n-1}\dd r$). Tabuleer $s_0,
        \dots, s_3$ en controleer $s_1 = 2\pi$, $s_2 =
        4\pi$, $s_3 = 2\pi^2$.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Hoge dimensies zijn vreemd.}
\begin{enumerate}[resume]
  \item Uit de formule van Stirling
        (\cref{thm:b2:comparison:stirling}) toegepast op
        $k!$, toon voor even $n = 2k$:
        \[
        V_n \sim \frac{1}{\sqrt{\pi n}}
        \Bigl(\frac{2\pi\eu}{n}\Bigr)^{n/2}
        \qquad (n \to \infty, \ n \text{ even}),
        \]
        en verklaar waarom dezelfde super-meetkundige
        vervalgrens zich tot oneven $n$ uitstrekt via de
        recursie.
  \item De eenheidsbal zit in de kubus $\intcc{-1}1^n$ van
        volume $2^n$. Bereken de vullingsverhouding
        $V_n/2^n$ voor $n = 2, 3, 10$, en toon dat ze naar
        $0$ streeft: in hoge dimensie ligt in wezen de hele
        kubus in haar hoeken.
  \item Toon dat de fractie van $v_n(1)$ liggend binnen
        afstand $\varepsilon$ van de randsfeer $1 - (1 -
        \varepsilon)^n \to 1$ is; numeriek, welke fractie
        van een $100$-dimensionale bal ligt in de
        buitenschil van dikte $1\%$?
  \item Bewijs de Wallis-asymptotiek $W_n \sim
        \sqrt{\dfrac{\pi}{2n}}$ \emph{(monotoniciteit van
        $(W_n)$, de verhouding $W_n/W_{n-2} \to 1$, en
        $W_nW_{n-1} = \frac\pi{2n}$)}, en de ondergrens
        $W_n \geq \sqrt{\dfrac{\pi}{2(n+1)}}$ voor alle
        $n$.
  \item (Concentratie op een slab) De fractie van de
        eenheidsbal met eerste coördinaat voorbij $\delta$
        is $\int_\delta^1(1 - x^2)^{\frac{n-1}2}\dd x
        \,\big/\, (2W_n)$. Met $1 - u \leq \eu^{-u}$ en de
        staartgrens $\int_\delta^\infty \eu^{-a x^2}\dd x
        \leq \frac{\eu^{-a\delta^2}}{2a\delta}$, toon dat
        deze fractie ten hoogste
        \[
        \frac{\eu^{-(n-1)\delta^2/2}}{(n-1)\,\delta}
        \Big/ \sqrt{\frac{2\pi}{n+1}}
        \]
        is en concludeer: voor $\delta = s/\sqrt{n-1}$ ligt
        alles behalve een fractie $O(\eu^{-s^2/2}/s)$ van
        de bal in de slab $\abs{x_1} \leq s/\sqrt{n-1}$.
        Een bal van hoge dimensie is, statistisch, een
        dunne pannekoek in elke richting tegelijk.
  \item Assembleer vragen 16--19 in één alinea: waar het
        volume van $B_n(1)$ zit (nabij de randsfeer, maar
        toch binnen $O(1/\sqrt n)$-slabs van elk hypervlak
        door het centrum), en waarom deze twee uitspraken
        elkaar niet tegenspreken.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Andere lichamen, en synthese.}
\begin{enumerate}[resume]
  \item (Simplex) Zij $\Delta_n = \{x \in \R^n : x_i \geq
        0,\ \sum x_i \leq 1\}$. Toon door snijden en
        inductie dat $\operatorname{vol}(\Delta_n) =
        \frac1{n!}$.
  \item (Kruispolytoop) Leid af dat $C_n = \{x :
        \sum\abs{x_i} \leq 1\}$ volume $\frac{2^n}{n!}$
        heeft, en verifieer de sandwich $C_n \subseteq
        B_n(1) \subseteq \intcc{-1}1^n$ op het niveau van
        volumes: $\frac{2^n}{n!} \leq V_n \leq 2^n$.
  \item Bereken $V_4$ op een derde manier: snijd $\R^4 =
        \R^2 \times \R^2$, integreer de \omterm{def:b2:multint:domain}{oppervlakte} van de
        $(z, w)$-schijf over de $(x, y)$-schijf in
        poolcoördinaten, en herwin $V_4 = \frac{\pi^2}2$.
  \item (Monte Carlo in de knel) Een punt wordt uniform
        getrokken in de kubus $\intcc{-1}1^{20}$. Toon dat
        de waarschijnlijkheid dat ze in de ingeschreven bal
        landt $V_{20}/2^{20} \approx 2.5\cdot10^{-8}$ is,
        zodat ongeveer veertig miljoen trekkingen nodig
        zijn voor de eerste treffer wordt verwacht:
        $V_n$ schatten door rejection sampling stort in in
        hoge dimensie (de curse of dimensionality).
  \item Synthese. Twee onafhankelijke afleidingen ontmoetten
        elkaar in $V_n = \pi^{n/2}/\Gamma(\frac n2 + 1)$:
        lijst welke stelling van dit hoofdstuk elk gebruikte
        (Fubini, variabelensubstitutie, de pool-Gauss-integraal),
        en welke éénvariabele-inputs (Wallis, $\Gamma$,
        Stirling). Waar doet het volume van Jaar~3 deze
        berekening over met de Lebesgue-theorie, en wat
        voegt ze toe?
\end{enumerate}
\end{problem}
