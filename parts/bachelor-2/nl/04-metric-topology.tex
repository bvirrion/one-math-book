\chapter{Topologie van metrische ruimten}\label{ch:b2:metric}

De topologie van de reële lijn (volume van Jaar~1) veralgemeent,
bijna zonder een woord te wijzigen, tot elke verzameling voorzien van
een afstand. De winst is enorm: rijen van functies, matrices, krommen
--- allemaal worden punten van \omterm{def:b2:metric:def}{metrische ruimten}, en de drie
pijlers die hier bewezen worden --- volledigheid met de
vastepuntsstelling van Banach, compactheid, samenhang ---
zijn er uniform op van toepassing. Dit hoofdstuk is de ruggengraat
van de hele analytische helft van het boek.

\section{Metrische ruimten}

\begin{definition}\label{def:b2:metric:def}
Een \emph{metrische ruimte}\index{metrische ruimte} is een
verzameling $X$ met een afbeelding $d \colon X \times X \to \R_+$
zodat, voor alle $x, y, z$:
\[
d(x,y) = 0 \iff x = y,
\qquad
d(x,y) = d(y,x),
\qquad
d(x,z) \leq d(x,y) + d(y,z).
\]
Bollen: $B(a, r) = \{x : d(a,x) < r\}$ (\omterm{def:b2:metric:topology}{open}), $\overline B(a,r) =
\{x : d(a,x) \leq r\}$ (gesloten). Een deelverzameling $A \subseteq X$
wordt een metrische ruimte met de geïnduceerde afstand.
\end{definition}

\begin{example}\label{ex:b2:metric:examples}
$\R$ met $\abs{x - y}$; $\R^n$ met een van
\[
d_1(x,y) = \sum_i \abs{x_i - y_i},
\quad
d_2(x,y) = \Bigl(\sum_i (x_i - y_i)^2\Bigr)^{1/2},
\quad
d_\infty(x,y) = \max_i \abs{x_i - y_i};
\]
de verzameling $C(\intcc{a}{b})$ van continue functies met de
\emph{sup-afstand} $d_\infty(f, g) = \sup_{\intcc{a}{b}} \abs{f -
g}$ (eindig: $f - g$ is begrensd); elke verzameling met de
\emph{discrete} afstand ($d(x,y) = 1$ voor $x \neq y$). Afstanden
afkomstig van normen zijn het onderwerp van \cref{ch:b2:nvs}.
\end{example}

\begin{definition}[Topologie van een metrische ruimte]\label{def:b2:metric:topology}
$U \subseteq X$ is \emph{open}\index{open verzameling!in een metrische ruimte}
wanneer elk punt van $U$ middelpunt is van een bol die in $U$ ligt;
$F$ is \emph{gesloten} wanneer haar complement open is. Omgevingen,
inwendige, afsluiting, dichtheid, rand worden precies gedefinieerd
als op de reële lijn (volume van Jaar~1), met bollen in plaats van
intervallen, en de daar bewezen uitspraken --- unies/doorsneden van
open verzamelingen, karakteriseringen van inwendige en afsluiting,
afsluiting als kleinste gesloten bovverzameling --- gaan over met
dezelfde bewijzen. Open bollen zijn open, gesloten bollen zijn
gesloten (driehoeksongelijkheid).
\end{definition}

\begin{example}[Inwendige, afsluiting, rand op één verzameling]\label{ex:b2:metric:intclosure}
In $\R$, zij $A = \intoc{0}{1} \cup \{2\}$. Inwendige:
$\intoo{0}{1}$ --- rond elk $x \in \intoo01$ blijft een kleine bol
in $A$; rond $1$ loopt elke bol $\intoo{1-r}{1+r}$ rechts uit $A$
weg, dus $1$ is niet inwendig; en het geïsoleerde punt $2$ is dat
evenmin. Afsluiting: $\intcc{0}{1} \cup \{2\}$ (het punt $0$ is een
limiet van $A$, er wordt niets anders toegevoegd). Rand
(afsluiting minus inwendige): $\{0, 1, 2\}$. Merk de asymmetrieën
op die het onthouden waard zijn: een eindpunt kan tot een
verzameling behoren zonder inwendig te zijn ($1$), aanklevend zijn
zonder te behoren ($0$), en een geïsoleerd punt is zijn eigen rand
($2$). Dezelfde boekhouding loopt letterlijk in elke \omterm{def:b2:metric:def}{metrische
ruimte}, met bollen in plaats van intervallen.
\end{example}

\begin{definition}[Limieten, continuïteit]\label{def:b2:metric:continuity}
$x_n \to x$ in $X$ wanneer $d(x_n, x) \to 0$. Een afbeelding $f
\colon X \to Y$ tussen \omterm{def:b2:metric:def}{metrische ruimten} is
\emph{continu}\index{continu} in $a$ wanneer
\[
\forall \varepsilon > 0,\ \exists\delta > 0,\quad
d_X(x, a) \leq \delta \implies d_Y\bigl(f(x), f(a)\bigr) \leq
\varepsilon ;
\]
equivalent (zelfde bewijs als op $\R$), $f(x_n) \to f(a)$ voor elke
rij $x_n \to a$. $f$ is \emph{Lipschitz}\index{Lipschitz} met
constante $k$ wanneer $d_Y(f(x), f(y)) \leq k\, d_X(x, y)$ altijd
--- dan uniform continu, dus continu.
\end{definition}

\begin{theorem}[Globale karakterisering van continuïteit]\label{thm:b2:metric:globalcontinuity}
$f \colon X \to Y$ is \omterm{def:b2:metric:continuity}{continu} (in elk punt) als en slechts als het
origineel van elke \omterm{def:b2:metric:topology}{open} verzameling \omterm{def:b2:metric:topology}{open} is --- iff het origineel
van elke gesloten verzameling gesloten is.
\end{theorem}

\begin{proof}
($\Rightarrow$) Zij $V \subseteq Y$ \omterm{def:b2:metric:topology}{open} en $a \in f^{-1}(V)$: er
is een bol $B(f(a), \varepsilon) \subseteq V$; continuïteit in $a$
levert $\delta$ met $f(B(a, \delta)) \subseteq B(f(a),
\varepsilon)$, dus $B(a, \delta) \subseteq f^{-1}(V)$.

($\Leftarrow$) Gegeven $a$ en $\varepsilon$: $f^{-1}\bigl(B(f(a),
\varepsilon)\bigr)$ is \omterm{def:b2:metric:topology}{open} en bevat $a$, bevat dus een bol
$B(a, \delta)$: dat is de definitie van continuïteit in $a$.
Gesloten verzamelingen: complementen (\cref{prop:b2:structures:images}).
\end{proof}

\section{Volledige ruimten}

\begin{definition}\label{def:b2:metric:complete}
Een rij $(x_n)$ is \emph{Cauchy} wanneer $\sup_{p, q \geq N} d(x_p,
x_q) \to 0$ als $N \to \infty$. Een \omterm{def:b2:metric:def}{metrische ruimte} is
\emph{volledig}\index{volledige metrische ruimte} wanneer elke
Cauchyrij convergeert. Convergent $\Rightarrow$ Cauchy altijd;
gesloten deelverzamelingen van volledige ruimten zijn volledig, en
volledige deelverzamelingen van elke ruimte zijn gesloten (zelfde
bewijzen als op $\R$: volume van Jaar~1).
\end{definition}

\begin{example}[Cauchy zonder limiet]\label{ex:b2:metric:cauchynolimit}
In $X = \Q$ met de gewone afstand voldoen de decimale afknottingen
van $\sqrt2$,
\[
x_0 = 1,\quad x_1 = 1.4,\quad x_2 = 1.41,\quad x_3 = 1.414,
\quad\dots
\]
aan $\abs{x_p - x_q} \leq 10^{-\min(p,q)}$: Cauchy in $\Q$. Een
limiet in $\Q$ zou ook de limiet in $\R$ zijn, namelijk
$\sqrt2 \notin \Q$: er bestaat geen limiet in $X$. Onvolledigheid
is de aanwezigheid van zulke ``fantoomlimieten''; de volledigheid
van $\R$ is in het volume van Jaar~1 precies geconstrueerd om elke
Cauchyrij een thuis te geven.
\end{example}

\begin{theorem}\label{thm:b2:metric:rncomplete}
$\R^n$ (een van de drie afstanden van
\cref{ex:b2:metric:examples}) en $\bigl(C(\intcc{a}{b}),
d_\infty\bigr)$ zijn \omterm{def:b2:metric:complete}{volledig}.
\end{theorem}

\begin{proof}
$\R^n$: een Cauchyrij is Cauchy in elke coördinaat (elke
$\abs{x_i - y_i} \leq d(x,y)$ voor alle drie de afstanden), dus
elke coördinaat convergeert (volledigheid van $\R$, volume van
Jaar~1), en coördinaatsgewijze convergentie impliceert convergentie
voor $d_\infty$ (eindig veel coördinaten), dus voor alle drie (de
drie afstanden domineren elkaar tot op constante factoren:
$d_\infty \leq d_2 \leq d_1 \leq n\,d_\infty$).

$C(\intcc{a}{b})$: zij $(f_n)$ $d_\infty$-Cauchy. Voor elke $x$ is
$(f_n(x))$ Cauchy in $\R$ ($\abs{f_p(x) - f_q(x)} \leq
d_\infty(f_p, f_q)$): convergeert naar zekere $f(x)$. Overgang naar
de limiet in $\abs{f_p(x) - f_q(x)} \leq \varepsilon$ (geldig voor
$p, q \geq N_\varepsilon$, alle $x$) als $q \to \infty$:
$\abs{f_p(x) - f(x)} \leq \varepsilon$ voor alle $x$, d.w.z.\
$d_\infty(f_p, f) \leq \varepsilon$: uniforme convergentie. De
limiet is \omterm{def:b2:metric:continuity}{continu}: gegeven $\varepsilon$, kies $p$ met
$\sup\abs{f_p - f} \leq \varepsilon$, gebruik dan continuïteit van
$f_p$ in $a$ en de drietermssplitsing
\[
\abs{f(x) - f(a)} \leq \abs{f(x) - f_p(x)} + \abs{f_p(x) - f_p(a)}
+ \abs{f_p(a) - f(a)} \leq 3\varepsilon
\]
voor $x$ dicht bij $a$. (Dit ``$3\varepsilon$-argument'' keert terug
als de uniforme-limietstelling van \cref{ch:b2:funcseq}.)
\end{proof}

\begin{example}[Open en gesloten verzamelingen herkend via continuïteit]\label{ex:b2:metric:recognize}
De globale karakterisering
(\cref{thm:b2:metric:globalcontinuity}) is het alledaagse
instrument voor topologische boekhouding. In $\R^2$: de verzameling
$\{(x, y) : x^2 + y^2 < 1,\ y > x^3\}$ is \omterm{def:b2:metric:topology}{open} --- zij is
$g^{-1}(\intoo{-\infty}{1}) \cap h^{-1}(\intoo{0}{+\infty})$ voor
de continue $g(x,y) = x^2 + y^2$ en $h(x, y) = y - x^3$, een
doorsnede van twee \omterm{def:b2:metric:topology}{open} originelen. In $\bigl(C(\intcc01),
d_\infty\bigr)$: de verzameling van functies met $f(0) = f(1)$ en
$\int_0^1 f = 0$ is gesloten --- het origineel van $\{(0,0)\}$
onder de continue afbeelding $f \mapsto \bigl(f(0) - f(1),\
\int_0^1 f\bigr)$ naar $\R^2$ (elke coördinaat is $1$-Lipschitz,
zoals in \cref{exo:b2:metric:3}). De methode tekent nooit een
plaatje: toon een continue afbeelding, lees de verzameling als
origineel, citeer de stelling.
\end{example}

\begin{example}[Een gesloten verzameling gedefinieerd door oneindig veel voorwaarden]\label{ex:b2:metric:lipclosed}
In $\bigl(C(\intcc01), d_\infty\bigr)$ is de verzameling
\[
L = \{f : \abs{f(x) - f(y)} \leq \abs{x - y}
\ \text{voor alle } x, y\}
\]
van $1$-Lipschitzfuncties gesloten, hoewel ze door overaftelbaar
veel voorwaarden wordt uitgesneden: voor elk vast paar $(x, y)$ is
de afbeelding $f \mapsto \abs{f(x) - f(y)} - \abs{x - y}$ \omterm{def:b2:metric:continuity}{continu}
(evaluaties zijn $1$-Lipschitz), dus elke enkele voorwaarde
definieert een gesloten verzameling, en $L$ is de
\emph{doorsnede} van deze familie --- een willekeurige doorsnede
van gesloten verzamelingen is gesloten. Hetzelfde sjabloon
certificeert geslotenheid voor monotone functies, convexe
functies, functies begrensd door een vaste $g$: uniforme limieten
erven elke eigenschap die als familie van gesloten puntgewijze
beperkingen uitdrukbaar is. Wat uniforme limieten \emph{niet}
automatisch erven --- differentieerbaarheid, bijvoorbeeld --- is
precies waar \cref{ch:b2:funcseq} hard voor moet werken.
\end{example}

\begin{theorem}[Vastepuntsstelling van Banach]\label{thm:b2:metric:banach}
Zij $X$ een niet-lege \omterm{def:b2:metric:complete}{volledige metrische ruimte} en $f \colon X \to
X$ een \emph{contractie}: \omterm{def:b2:metric:continuity}{Lipschitz} met constante $k < 1$. Dan
heeft $f$ een uniek vast punt $\ell$, en elke baan $x_{n+1} =
f(x_n)$ convergeert naar $\ell$, met\index{vastepuntsstelling van Banach}
\[
d(x_n, \ell) \leq \frac{k^n}{1 - k}\, d(x_1, x_0) .
\]
\end{theorem}

\begin{proof}
Uniciteit: twee vaste punten liggen op afstand $\leq k$ maal
zichzelf. Bestaan: $d(x_{n+1}, x_n) \leq k^n d(x_1, x_0)$ door
inductie, dus voor $q > p$,
\[
d(x_q, x_p) \leq \sum_{j=p}^{q-1} d(x_{j+1}, x_j)
\leq d(x_1, x_0) \sum_{j \geq p} k^j
= \frac{k^p}{1-k}\, d(x_1, x_0) \xrightarrow[p\to\infty]{} 0 :
\]
Cauchy, dus convergent naar zekere $\ell$; continuïteit van $f$
brengt $x_{n+1} = f(x_n)$ over naar de limiet: $\ell = f(\ell)$.
De foutenschatting is de weergegeven schatting met $q \to \infty$.
\end{proof}

\begin{example}[Een integraalvergelijking]\label{ex:b2:metric:integralequation}
Op $X = C(\intcc{0}{1})$ (\omterm{def:b2:metric:complete}{volledig},
\cref{thm:b2:metric:rncomplete}), beschouw $T(f)(x) = 1 + \frac12
\int_0^x f(t)\,\dd t$. Voor $f, g \in X$:
\[
\abs{T(f)(x) - T(g)(x)} \leq \frac12 \int_0^x \abs{f - g} \leq
\frac12\, d_\infty(f, g),
\]
dus $T$ is een $\frac12$-contractie: zij heeft een uniek \omterm{def:b2:metric:continuity}{continu}
vast punt --- de oplossing van $f' = \frac f2$, $f(0) = 1$, namelijk
$\eu^{x/2}$. Dit schema, geïndustrialiseerd, wordt de
stelling van Cauchy--Lipschitz van \cref{ch:b2:diffeq}.
\end{example}

\begin{example}[Een numeriek vast punt: $x = \cos x$]\label{ex:b2:metric:cosfix}
Op de volledige $X = \intcc{0}{1}$ stuurt de afbeelding $f = \cos$
$X$ naar $\intcc{\cos 1}{1} \subseteq X$ en is een contractie: door
de middelwaardeongelijkheid,
\[
\abs{\cos x - \cos y} \leq \bigl(\sup_{\intcc01}\abs{\sin}\bigr)
\abs{x - y} = (\sin 1)\abs{x - y},
\qquad \sin 1 \approx 0.841 < 1 .
\]
Banach: een unieke oplossing van $x = \cos x$ in $\intcc01$ (dus in
$\R$: elk reëel vast punt ligt in $\intcc{-1}{1}$, dan in
$\intcc{\cos 1}{1}$ na één toepassing), en de iteratie $x_{n+1} =
\cos x_n$ convergeert ernaartoe vanaf elke start: $x_\infty
\approx 0.739085$, het beroemde getal verkregen door op de
cosinustoets van een rekenmachine te hameren. De foutenschatting
voorspelt verval met $(\sin1)^n/(1 - \sin1)$ --- ongeveer één
cijfer per $13$ drukken; de a-posteriori-grens van het
weekendvraagstuk van dit hoofdstuk (vraag 14) certificeert elke
stap ter plaatse.
\end{example}

\section{Compactheid}

\begin{definition}\label{def:b2:metric:compact}
Een \omterm{def:b2:metric:def}{metrische ruimte} $X$ is \emph{compact}\index{compacte metrische ruimte}
wanneer elke rij in $X$ een deelrij heeft die convergeert \emph{in}
$X$ (de Bolzano--Weierstrass-eigenschap). Een deelverzameling is
compact wanneer zij dat is met de geïnduceerde afstand.
\end{definition}

\begin{theorem}[Eerste eigenschappen]\label{thm:b2:metric:compactprops}
\begin{enumerate}
  \item Een compacte deelverzameling is gesloten en begrensd; een
        gesloten deelverzameling van een compacte ruimte is
        \omterm{def:b2:metric:compact}{compact}.
  \item In $\R^n$ geldt het omgekeerde: \omterm{def:b2:metric:compact}{compact} $\iff$ gesloten en
        begrensd.
  \item Een \omterm{def:b2:metric:continuity}{continu} beeld van een compacte ruimte is \omterm{def:b2:metric:compact}{compact}; een
        continue reële functie op een niet-lege compacte ruimte is
        begrensd en bereikt haar grenzen.
  \item (Heine) Een continue afbeelding op een compacte ruimte is
        uniform continu.
  \item Producten: als $X, Y$ \omterm{def:b2:metric:compact}{compact} zijn, is $X \times Y$ dat
        ook (met $d\bigl((x,y),(x',y')\bigr) = d(x,x') +
        d(y,y')$).
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Zelfde argumenten als op de lijn (volume van Jaar~1): een naar
oneindig ontsnappende of buiten convergente rij heeft geen deelrij
die binnen convergeert; voor de tweede claim, extraheer in de
omgevende compacte en gebruik geslotenheid.

(2) Begrensde rijen in $\R^n$ hebben componentgewijs convergente
deelrijen: extraheer op de eerste coördinaat
(Bolzano--Weierstrass op $\R$), dan, uit die deelrij, op de
tweede, enzovoort ($n$ opeenvolgende extracties); geslotenheid
houdt de limiet binnen.

(3) Gegeven $(f(x_n))$, extraheer $x_{\varphi(n)} \to x \in X$;
continuïteit geeft $f(x_{\varphi(n)}) \to f(x) \in f(X)$. Reëel
geval: compactheid van $f(X) \subseteq \R$ maakt haar gesloten en
begrensd, en $\sup f(X) \in f(X)$ (het sup van een verzameling is
aanklevend, en $f(X)$ is gesloten).

(4) Het bewijs van Jaar~1 gaat letterlijk over; hier is het, in
metrisch gewaad. Veronderstel $f \colon X \to Y$ \omterm{def:b2:metric:continuity}{continu} op de
compacte $X$ maar niet uniform continu: zekere $\varepsilon > 0$
laat, voor elke $n$, punten toe met
\[
d_X(x_n, y_n) \leq \frac{1}{n+1}
\qquad\text{en}\qquad
d_Y\bigl(f(x_n), f(y_n)\bigr) > \varepsilon .
\]
Extraheer $x_{\varphi(n)} \to a \in X$; dan $y_{\varphi(n)} \to a$
ook (de onderlinge afstanden gaan naar $0$). Continuïteit in $a$
stuurt beide beeldrijen naar $f(a)$, dus
$d_Y\bigl(f(x_{\varphi(n)}), f(y_{\varphi(n)})\bigr) \to 0$ ---
in tegenspraak met de uniforme kloof $> \varepsilon$. Compactheid
leverde precies één ding: het ophopingspunt $a$ waarop gewone
continuïteit toe te passen is.

(5) Extraheer op de $X$-coördinaten, dan opnieuw op de
$Y$-coördinaten.
\end{proof}

\begin{example}[Stelling van Heine, met en zonder compactheid]\label{ex:b2:metric:heinework}
Op $\intcc{0}{1}$ is de functie $x \mapsto x^2$ uniform continu ---
Heine zegt dat zonder rekenwerk, maar de directe schatting is
instructief:
\[
\abs{x^2 - y^2} = \abs{x + y}\,\abs{x - y} \leq 2\abs{x - y},
\]
dus $\delta = \varepsilon/2$ werkt \emph{voor elk punt tegelijk}.
Op $\R$ is dezelfde functie niet uniform continu: met $x_n = n$ en
$y_n = n + \frac1n$ is de kloof $\abs{x_n - y_n} = \frac1n \to 0$
terwijl $\abs{x_n^2 - y_n^2} = 2 + \frac1{n^2} \geq 2$: geen
enkele $\delta$ bedient $\varepsilon = 1$. Het mechanisme is
zichtbaar: de lokale Lipschitzconstante $\abs{x + y}$ is begrensd
op een \omterm{def:b2:metric:compact}{compact} en onbegrensd op $\R$ --- de stelling van Heine is
precies de uitspraak dat compactheid zulke lokale constanten
uniform kapot.
\end{example}

\begin{method}[Bewijzen dat een verzameling compact is]\label{met:b2:metric:compactness}
Drie routes, in volgorde van frequentie. (1) \emph{Herkenning in
de omgeving:} in $\R^n$ (of elke eindigdimensionale genormeerde
ruimte, \cref{ch:b2:nvs}), verifieer gesloten --- typisch als
origineel, \cref{ex:b2:metric:recognize} --- en begrensd. (2)
\emph{Overerving:} een gesloten deelverzameling van een bekende
compacte ruimte is \omterm{def:b2:metric:compact}{compact}; een eindige unie of een product van
compacta is \omterm{def:b2:metric:compact}{compact}; een \omterm{def:b2:metric:continuity}{continu} beeld van een \omterm{def:b2:metric:compact}{compact} is \omterm{def:b2:metric:compact}{compact}.
(3) \emph{Met de blote hand:} extraheer een convergente deelrij
uit een willekeurige rij --- gewoonlijk door opeenvolgende
extracties coördinaat voor coördinaat. Om
\emph{niet}-compactheid te bewijzen volstaat één getuige: een rij
zonder convergente deelrij, meestal punten met onderlinge afstand
$\geq \varepsilon$.
\end{method}

\begin{example}[Afstanden tussen verzamelingen: compactheid verdient haar loon]\label{ex:b2:metric:setdistance}
Zij $K$ \omterm{def:b2:metric:compact}{compact}, $F$ gesloten, $K \cap F = \emptyset$ in een
\omterm{def:b2:metric:def}{metrische ruimte}. Dan
\[
d(K, F) = \inf\,\{d(x, y) : x \in K,\ y \in F\} > 0 :
\]
de functie $x \mapsto d(x, F)$ is \omterm{def:b2:metric:continuity}{continu}
(\cref{exo:b2:metric:11}) en positief op $K$ ($d(x, F) = 0$ zou
$x \in \overline F = F$ plaatsen), dus bereikt zij een positief
minimum op de compacte $K$
(\cref{thm:b2:metric:compactprops}~(3)). Compactheid is geen
decoratie: voor twee \emph{gesloten} verzamelingen kan het
infimum verdwijnen zonder bereikt te worden --- in $\R^2$ zijn de
hyperbool $F_1 = \{xy = 1\}$ en de as $F_2 = \{y = 0\}$ disjuncte
gesloten verzamelingen met $d(F_1, F_2) = 0$ (de punten $(n,
\frac1n)$ naderen de as). Ontsnapping naar oneindig is precies wat
compactheid verbiedt.
\end{example}

\begin{theorem}[Borel--Lebesgue]\label{thm:b2:metric:borellebesgue}
Een \omterm{def:b2:metric:def}{metrische ruimte} $X$ is \omterm{def:b2:metric:compact}{compact} als en slechts als elke
overdekking van $X$ door \omterm{def:b2:metric:topology}{open} verzamelingen een \emph{eindige}
deeloverdekking heeft.\index{stelling van Borel--Lebesgue}
\end{theorem}

\begin{proof}
($\Leftarrow$) Veronderstel dat $(x_n)$ geen convergente deelrij
heeft. We claimen dat elk $x \in X$ een bol $B(x, r_x)$ heeft die
$x_n$ voor slechts eindig veel indices $n$ bevat: anders zou elke
bol $B(x, \frac1{k+1})$ oneindig veel termen bevatten, en het
kiezen van indices
\[
\varphi(0) < \varphi(1) < \varphi(2) < \cdots
\quad\text{met}\quad
x_{\varphi(k)} \in B\Bigl(x, \frac{1}{k+1}\Bigr)
\]
(mogelijk bij elke stap precies omdat er oneindig veel kandidaten
overblijven) zou een naar $x$ convergente deelrij bouwen. De bollen
$B(x, r_x)$ overdekken $X$; als eindig veel ervan $X$ overdekten,
zou de indexverzameling $\N$ een eindige unie van eindige
verzamelingen zijn: absurd.

($\Rightarrow$) Twee stappen. \emph{Lebesguegetal:} voor een \omterm{def:b2:metric:topology}{open}
overdekking $(U_i)$ van een compacte $X$ bestaat $\rho > 0$ zodat
elke bol van straal $\rho$ in zekere $U_i$ ligt. Anders kies voor
elke $n$ een $x_n$ met $B(x_n, \frac{1}{n+1})$ in geen enkele
$U_i$; extraheer $x_{\varphi(n)} \to x \in U_{i_0} \supseteq
B(x, r)$; voor grote $n$, $B(x_{\varphi(n)},
\frac{1}{\varphi(n)+1}) \subseteq B(x, r) \subseteq U_{i_0}$:
tegenspraak. \emph{Totale begrensdheid:} voor elke $\rho > 0$
overdekken eindig veel bollen van straal $\rho$ $X$. Anders kies
inductief $x_{n+1}$ buiten $B(x_0, \rho) \cup \dots \cup B(x_n,
\rho)$: de rij heeft paarsgewijze afstanden $\geq \rho$, dus geen
Cauchy- --- dus geen convergente --- deelrij: tegenspraak.
Combineren: overdek $X$ met eindig veel bollen van straal $\rho$
(het Lebesguegetal), elk in zekere $U_i$: een eindige
deeloverdekking.
\end{proof}

\begin{example}[Een $\varepsilon$-net, geteld]\label{ex:b2:metric:epsnet}
Totale begrensdheid (uit het bewijs van
\cref{thm:b2:metric:borellebesgue}) is heel concreet op
$\intcc{0}{1}$: voor $\varepsilon > 0$ overdekken de $\lceil
\frac{1}{2\varepsilon}\rceil$ bollen met middelpunten
$\varepsilon, 3\varepsilon, 5\varepsilon, \dots$ van straal
$\varepsilon$ haar --- ongeveer $\frac1{2\varepsilon}$ bollen, en
geen overdekking kan met minder dan $\frac{1}{2\varepsilon}$ ervan
toekomen (elke bol dekt lengte ten hoogste $2\varepsilon$). In
$\intcc01^2$ kwadrateert de telling tot orde $\varepsilon^{-2}$:
overdekkingsaantallen groeien als $\varepsilon^{-d}$ in dimensie
$d$ --- een kwantitatief gezicht van compactheid, en de reden
waarom de oneindigdimensionale eenheidsbollen van
\cref{ch:b2:nvs} (waar helemaal geen eindig $\frac13$-net bestaat)
niet \omterm{def:b2:metric:compact}{compact} kunnen zijn.
\end{example}

\begin{example}[Compactheid lezen op overdekkingen]\label{ex:b2:metric:covers}
Het halfopen interval $\intoc{0}{1}$ wordt overdekt door de \omterm{def:b2:metric:topology}{open}
verzamelingen $U_n = \intoo{\frac1n}{2}$, $n \geq 1$; elke eindige
deelfamilie heeft een grootste index $N$ en mist
$\intoc{0}{\frac1N}$: geen eindige deeloverdekking, dus
$\intoc{0}{1}$ is niet \omterm{def:b2:metric:compact}{compact} --- wat de sequentiële definitie
ziet via $x_n = \frac1n$, waarvan de limiet $0$ ontsnapt. Anderzijds
herstelt het toevoegen van het enkele punt $0$ beide diagnoses
tegelijk: op $\intcc{0}{1}$ moet elke zulke overdekking een
verzameling bevatten die $0$ bevat, die een heel initieel segment
opslokt, en eindig veel verzamelingen voltooien de rest. De twee
talen van \cref{thm:b2:metric:borellebesgue} falen of slagen altijd
samen --- overdekkingen detecteren ontsnapping precies waar rijen
dat doen.
\end{example}

\begin{remark}[Perspectieven binnen dit volume]\label{rem:b2:metric:perspectives}
Dit hoofdstuk is de dragende muur van het volume; let op waar
elke pijler gewicht draagt. \emph{Volledigheid}: het Cauchy-criterium
wordt de convergentietest voor reeksen in Banachruimten
(\cref{ch:b2:series}), uniforme convergentie in
\cref{ch:b2:funcseq} is precies convergentie in de volledige
$\bigl(C, d_\infty\bigr)$, en Cauchy--Lipschitz
(\cref{ch:b2:diffeq}) is de vastepuntsstelling van Banach in
integraalvergelijkingskleding. \emph{Compactheid}: zij bewijst de
equivalentie van normen (\cref{ch:b2:nvs}), het bereiken van
extrema voor de optimalisatie van \cref{ch:b2:diffcalc}, en het
bestaan van beste benaderingen (weekendvraagstuk van
\cref{ch:b2:nvs}). \emph{Samenhang}: zij globaliseert lokale
uitspraken --- uniciteit van oplossingen van differentiaalvergelijkingen,
de tussenwaardestelling op krommen (\cref{ch:b2:curves}), en de
twee componenten van $GL_n(\R)$ die de oriëntatietheorie
(\cref{ch:b2:multint}) uit elkaar zal houden.
\end{remark}

\begin{remark}[Veelgemaakte valkuilen]\label{rem:b2:metric:pitfalls}
(i) ``Gesloten en begrensd impliceert \omterm{def:b2:metric:compact}{compact}'' is een stelling over
$\R^n$, niet over \omterm{def:b2:metric:def}{metrische ruimten}: een oneindige verzameling met
de discrete metriek is in zichzelf gesloten en begrensd maar niet
\omterm{def:b2:metric:compact}{compact} (\cref{exo:b2:metric:4}), en de gesloten eenheidsbol van
$C(\intcc01)$ faalt ook (\cref{ch:b2:nvs}). (ii) Volledigheid is
een eigenschap van de \emph{afstand}, niet van de topologie: $\R$
met $d(x,y) = \abs{\arctan x - \arctan y}$ heeft de gewone
convergente rijen maar is onvolledig
(\cref{exo:b2:metric:1}). (iii) Een continue bijectie hoeft geen
homeomorfisme te zijn --- de cirkelparametrisering van
\cref{exo:b2:metric:7}; compactheid van de bron herstelt het. (iv)
De stelling van Banach eist $k < 1$ \emph{uniform}: de voorwaarde
$d(f(x), f(y)) < d(x,y)$ alleen garandeert niets op een
niet-compacte ruimte (\cref{exo:b2:metric:5}). (v) \omterm{def:b2:metric:connected}{Samenhangend}
impliceert niet boogsamenhangend in het algemeen --- maar voor de
\omterm{def:b2:metric:topology}{open} deelverzamelingen van genormeerde ruimten die in dit boek
voorkomen, vallen de twee samen (\cref{ch:b2:nvs}).
\end{remark}

\begin{remark}[Waar dit hoofdstuk gebruikt wordt]
Overal in de analytische helft. Volledigheid van
$C(\intcc{a}{b})$ voedt de convergentiestellingen van
\cref{ch:b2:funcseq} en de Cauchy--Lipschitz-theorie van
\cref{ch:b2:diffeq} (het weekendvraagstuk van dit hoofdstuk
bewijst al de lokale stelling van Picard--Lindelöf); compactheid
geeft de equivalentie van normen in eindige dimensie
(\cref{ch:b2:nvs}) en het bestaan van extrema in
\cref{ch:b2:diffcalc}; samenhang ligt ten grondslag aan de
tussenwaardeargumenten van \cref{ch:b2:realfun} en de
uniciteitsglobalisering voor differentiaalvergelijkingen. In het
volume van Jaar~3 worden compactheid in functieruimten (de stelling
van Arzelà--Ascoli) en de categoriestelling van Baire
(\cref{exo:b2:metric:12} hier) dagelijkse instrumenten.
\end{remark}

\section{Samenhang}

\begin{definition}\label{def:b2:metric:connected}
$X$ is \emph{samenhangend}\index{samenhangende ruimte} wanneer zij
geen partitie in twee niet-lege \omterm{def:b2:metric:topology}{open} deelverzamelingen toelaat ---
equivalent, wanneer haar enige deelverzamelingen die zowel \omterm{def:b2:metric:topology}{open} als
gesloten zijn $\emptyset$ en $X$ zijn. $X$ is
\emph{boogsamenhangend} wanneer twee willekeurige punten verbonden
worden door een continue afbeelding $\gamma \colon \intcc{0}{1} \to
X$.
\end{definition}

\begin{theorem}\label{thm:b2:metric:connectedness}
\begin{enumerate}
  \item De samenhangende deelverzamelingen van $\R$ zijn precies de
        intervallen.
  \item Een \omterm{def:b2:metric:continuity}{continu} beeld van een \omterm{def:b2:metric:connected}{samenhangende ruimte} is
        \omterm{def:b2:metric:connected}{samenhangend} --- vandaar de algemene tussenwaardestelling:
        een continue reële functie op een \omterm{def:b2:metric:connected}{samenhangende ruimte}
        neemt elke waarde aan tussen twee van haar waarden.
  \item Boogsamenhangend $\Rightarrow$ \omterm{def:b2:metric:connected}{samenhangend}. (Het omgekeerde
        faalt in het algemeen; het geldt voor \omterm{def:b2:metric:topology}{open}
        deelverzamelingen van genormeerde ruimten,
        \cref{ch:b2:nvs}.)
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Een niet-interval $A$ mist zekere $z$ tussen twee van haar
punten: $A = (A \cap \intoo{-\infty}{z}) \cup (A \cap
\intoo{z}{+\infty})$ splitst haar in twee niet-lege (in $A$) \omterm{def:b2:metric:topology}{open}
stukken. Omgekeerd, zij $I$ een interval en $I = U \cup V$ een
partitie in niet-lege relatief \omterm{def:b2:metric:topology}{open} verzamelingen; kies $a \in U$,
$b \in V$, zeg $a < b$, en stel $s = \sup\,(U \cap \intcc{a}{b})$,
een punt van $\intcc{a}{b} \subseteq I$. Als $s \in U$: dan $s \neq
b$, en relatieve openheid van $U$ plaatst een heel interval rond $s$
(doorsneden met $I$) in $U$ --- dus punten van $U \cap
\intcc{a}{b}$ overschrijden $s$, in tegenspraak met het supremum.
Als $s \in V$: relatieve openheid van $V$ plaatst een interval
$\intoo{s - r}{s + r} \cap I$ in $V$; maar het supremum is
aanklevend aan $U \cap \intcc{a}{b}$, dat dat interval moet
ontmoeten --- tegenspraak met $U \cap V = \emptyset$. (Dit is het
clopen-argument van Jaar~1 voor $\R$, uitgevoerd binnen $I$.)

(2) Als $f(X) = U' \cup V'$ splitst in niet-lege relatief \omterm{def:b2:metric:topology}{open}
verzamelingen, dan splitst $X = f^{-1}(U') \cup f^{-1}(V')$ de $X$
(\cref{thm:b2:metric:globalcontinuity}). Tussenwaarde: $f(X)
\subseteq \R$ is \omterm{def:b2:metric:connected}{samenhangend}, dus een interval door (1).

(3) Veronderstel $X = U \cup V$, beide niet-lege \omterm{def:b2:metric:topology}{open}, en verbind
$a \in U$ met $b \in V$ door een pad $\gamma$: dan splitsen
$\gamma^{-1}(U), \gamma^{-1}(V)$ het $\intcc{0}{1}$, in tegenspraak
met (1).
\end{proof}

\begin{example}\label{ex:b2:metric:glnr}
$GL_n(\R)$ is niet \omterm{def:b2:metric:connected}{samenhangend}: $\det$ is \omterm{def:b2:metric:continuity}{continu} (een polynoom in
de entries) op $\R^*$, die niet \omterm{def:b2:metric:connected}{samenhangend} is; de originelen van
$\R_+^*$ en $\R_-^*$ splitsen $GL_n(\R)$. (Elk stuk is in feite
boogsamenhangend --- een aangename oefening voorbij onze
behoeften.) Daarentegen \emph{is} $GL_n(\C)$ boogsamenhangend:
\cref{exo:b2:metric:10}.
\end{example}

\begin{example}[Een vast punt uit alleen samenhang]\label{ex:b2:metric:ivtfixed}
Elke continue $f \colon \intcc01 \to \intcc01$ heeft een vast punt
--- geen contractiehypothese, geen iteratie. Beschouw $g(x) =
f(x) - x$, \omterm{def:b2:metric:continuity}{continu} op de samenhangende $\intcc01$:
\[
g(0) = f(0) \geq 0,
\qquad
g(1) = f(1) - 1 \leq 0 ,
\]
en de tussenwaardestelling
(\cref{thm:b2:metric:connectedness}~(2)) levert een nulpunt van
$g$, d.w.z.\ een vast punt van $f$. Contrast met Banach
(\cref{thm:b2:metric:banach}): hier is bestaan topologisch en
gratis, maar uniciteit en het algoritme gaan verloren --- $f =
\mathrm{id}$ heeft elk punt vast, en iteratie van een
niet-contracterende $f$ kan eeuwig cycleren. De twee
vastepuntsstellingen van dit hoofdstuk beantwoorden verschillende
vragen met verschillende munten.
\end{example}

\begin{example}[$\R$ en $\R^2$ zijn niet homeomorf]\label{ex:b2:metric:rnotr2}
Samenhang is een topologische vingerafdruk. Veronderstel dat $h
\colon \R^2 \to \R$ een homeomorfisme was (een continue bijectie met
continue inverse). Verwijder één punt $a \in \R^2$: de restrictie $h
\colon \R^2\setminus\{a\} \to \R\setminus\{h(a)\}$ is nog een
homeomorfisme. Maar $\R^2$ minus een punt is boogsamenhangend ---
verbind twee willekeurige punten door een segment, omweg langs een
tweede segment via een hulppunt als $a$ het directe blokkeert ---
dus \omterm{def:b2:metric:connected}{samenhangend} (\cref{thm:b2:metric:connectedness}~(3)); terwijl
$\R$ minus een punt splitst in twee niet-lege \omterm{def:b2:metric:topology}{open} halfrechten:
niet \omterm{def:b2:metric:connected}{samenhangend}. Samenhang wordt bewaard door continue
afbeeldingen: tegenspraak. Het vlak en de lijn zijn werkelijk
verschillend \emph{als topologische ruimten} --- een feit dat
cardinaliteit alleen (bijecties in de stijl van
\cref{exo:b2:structures:3} bestaan!) te grof is om te zien.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b2:metric:1}
Controleer op $\R$ dat $\delta(x, y) = \min(1, \abs{x - y})$ en
$d(x,y) = \abs{\arctan x - \arctan y}$ afstanden zijn. Welke rijen
convergeren voor elk? Is $(\R, d)$ \omterm{def:b2:metric:complete}{volledig}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:metric:2}
Bewijs in een \omterm{def:b2:metric:def}{metrische ruimte} dat een convergente rij Cauchy en
begrensd is, en dat een Cauchyrij met een convergente deelrij
convergeert. Leid opnieuw af dat \omterm{def:b2:metric:compact}{compacte metrische ruimten}
\omterm{def:b2:metric:complete}{volledig} zijn.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:metric:3}
Bereken in $\bigl(C(\intcc{0}{1}), d_\infty\bigr)$ de afstand
tussen $f(x) = x$ en $g(x) = x^2$; beschrijf de gesloten bol
$\overline B(0, 1)$; en bewijs dat de verzameling $\{f : f(0) =
0\}$ gesloten is terwijl $\{f : f(0) > 0\}$ \omterm{def:b2:metric:topology}{open} is.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:metric:4}
Bewijs dat de discrete \omterm{def:b2:metric:def}{metrische ruimte} $X$ (elke verzameling)
\omterm{def:b2:metric:complete}{volledig} is, en dat zij \omterm{def:b2:metric:compact}{compact} is als en slechts als $X$ eindig
is. Welke deelverzamelingen zijn \omterm{def:b2:metric:connected}{samenhangend}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:metric:5}
Zij $X$ \omterm{def:b2:metric:compact}{compact} en $f \colon X \to X$ met
\[
d\bigl(f(x), f(y)\bigr) < d(x, y) \quad \text{voor alle } x \neq y .
\]
Bewijs dat $f$ een uniek vast punt heeft \emph{(minimaliseer $x
\mapsto d(x, f(x))$}), en geef een voorbeeld op $X =
\intco{1}{+\infty}$ (niet \omterm{def:b2:metric:compact}{compact}) zonder vast punt.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:metric:6}
(Geneste compacta) Zij $(K_n)$ een dalende rij van niet-lege
compacte deelverzamelingen van een \omterm{def:b2:metric:def}{metrische ruimte}. Bewijs dat
$\bigcap_n K_n \neq \emptyset$ \emph{(kies $x_n \in K_n$ en
extraheer)}. Toon met een voorbeeld dat niet-lege geneste
\emph{gesloten} verzamelingen in $\R$ lege doorsnede kunnen
hebben.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:metric:7}
Zij $K$ \omterm{def:b2:metric:compact}{compact} en $f \colon K \to Y$ \omterm{def:b2:metric:continuity}{continu} en bijectief. Bewijs
dat $f^{-1}$ \omterm{def:b2:metric:continuity}{continu} is \emph{(gebruik gesloten verzamelingen:
\cref{thm:b2:metric:globalcontinuity} en
\cref{thm:b2:metric:compactprops})}. Geef een tegenvoorbeeld zonder
compactheid ($\gamma(t) = (\cos t, \sin t)$ op
$\intco{0}{2\pi}$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:metric:8}
De \emph{Cantorverzameling} $C$ ontstaat uit $\intcc{0}{1}$ door
herhaaldelijk \omterm{def:b2:metric:topology}{open} middelste derden te verwijderen. Bewijs dat $C$
\omterm{def:b2:metric:compact}{compact} is, leeg inwendige heeft, en oneindig is --- ja, equipotent
met $\{0,1\}^{\N}$ \emph{(ternaire ontwikkelingen met cijfers $0,2$;
\cref{exo:b2:structures:3})}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:metric:9}
Zij $X$ een \emph{compacte} \omterm{def:b2:metric:def}{metrische ruimte} en $f \colon X \to X$
een isometrie: $d(f(x), f(y)) = d(x, y)$. Bewijs dat $f$ surjectief
is. \emph{Hint: als $a \notin f(X)$, dan $\varepsilon = d(a, f(X))
> 0$ (waarom?); bestudeer de baan $a, f(a), f^2(a), \dots$ en toon
dat haar punten paarsgewijs $\geq \varepsilon$ uit elkaar liggen ---
tegenspraak met compactheid.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:metric:10}
Bewijs dat $GL_n(\C)$ boogsamenhangend is. \emph{Hint: gegeven $A,
B$ inverteerbaar, beschouw $p(z) = \det\bigl((1 - z)A + zB\bigr)$
voor $z \in \C$: een polynoom in $z$, niet identiek nul, dus met
eindig veel wortels; verbind $0$ met $1$ in $\C$ door een pad dat
ze vermijdt.}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:metric:11}
Voor $\emptyset \neq A \subseteq X$, stel $d(x, A) = \inf_{a \in A}
d(x, a)$. Bewijs dat $x \mapsto d(x, A)$ $1$-Lipschitz is, dat
$d(x, A) = 0$ iff $x \in \overline A$, en dat voor disjuncte
niet-lege \emph{gesloten} verzamelingen $A, B$ de functie
\[
\varphi(x) = \frac{d(x, A)}{d(x, A) + d(x, B)}
\]
goed gedefinieerd is, \omterm{def:b2:metric:continuity}{continu}, gelijk aan $0$ precies op $A$ en aan
$1$ precies op $B$ --- een continue ``schakelaar'' die twee
disjuncte gesloten verzamelingen scheidt.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:metric:12}
(Baire) Zij $X$ een \omterm{def:b2:metric:complete}{volledige metrische ruimte} en $(U_n)_{n\geq1}$
een rij dichte \omterm{def:b2:metric:topology}{open} deelverzamelingen. Bewijs dat $\bigcap_n U_n$
dicht is in $X$ \emph{(bouw binnen elke bol geneste gesloten bollen
$\overline B(x_n, r_n) \subseteq U_n$ met $r_n \to 0$ en gebruik
volledigheid)}. Leid af dat $\R$ geen aftelbare unie is van
gesloten verzamelingen met leeg inwendige, en herwin --- opnieuw ---
dat $\R$ overaftelbaar is.
\end{exercise}

\section{Vraagstuk: Picard-iteratie}

Volledigheid plus contractie is een oplossingsmachine: voer haar een
vergelijking in, geschreven als vastpuntprobleem, en zij geeft
bestaan, uniciteit, een algoritme, en foutenbalken. Dit
weekendvraagstuk draait de machine op volle kracht op de
vergelijking $y' = f(t, y)$: we bewijzen de lokale
\emph{stelling van Picard--Lindelöf} (het niet-lineaire hart van de
Cauchy--Lipschitz-theorie van \cref{ch:b2:diffeq}), zien elke
hypothese haar loon verdienen via tegenvoorbeelden, en oogsten
zuiver metrische dividenden --- continue afhankelijkheid van de
gegevens, de vergelijking van Kepler, en de zelfgelijkenis van de
Cantorverzameling.

\begin{problem}[{Weekendvraagstuk --- de stelling van Picard--Lindelöf}]
\label{pb:b2:metric:1}
Doorheen, $t_0 \in \R$, $y_0 \in \R$, $a, b > 0$, en $f$ is een
continue functie op de rechthoek $R = \intcc{t_0 - a}{t_0 +
a} \times \intcc{y_0 - b}{y_0 + b}$, begrensd door $M =
\sup_R\,\abs f$, en \emph{$L$-Lipschitz in haar tweede variabele}:
$\abs{f(t, y) - f(t, z)} \leq L\abs{y - z}$ wanneer beide punten
in $R$ liggen. Stel
\[
h = \min\Bigl(a, \frac bM\Bigr) \quad (\text{met } h = a
\text{ als } M = 0), \qquad I = \intcc{t_0 - h}{t_0 + h}.
\]

\medskip
\noindent\textbf{Deel I --- Het volledige podium.}
\begin{enumerate}
  \item Bewijs de twee uitspraken geciteerd in
        \cref{def:b2:metric:complete}: een gesloten
        deelverzameling van een \omterm{def:b2:metric:complete}{volledige metrische ruimte} is
        \omterm{def:b2:metric:complete}{volledig}, en een volledige deelverzameling van elke
        \omterm{def:b2:metric:def}{metrische ruimte} is gesloten. Leid af dat elke gesloten
        deelverzameling van $\bigl(C(I), d_\infty\bigr)$ een
        \omterm{def:b2:metric:complete}{volledige metrische ruimte} is.
  \item Toon dat de afbeelding $C(I) \to C(I)$, $y \mapsto \bigl(t
        \mapsto \int_{t_0}^{t}y(s)\,\dd s\bigr)$, $h$-Lipschitz is
        voor $d_\infty$.
  \item (Vaste punten bewegen minder dan de afbeeldingen) Zij $g
        \colon X \to X$ een $k$-contractie van een \omterm{def:b2:metric:def}{metrische
        ruimte} met vast punt $\ell_g$, en $\widetilde g \colon X
        \to X$ \emph{elke} afbeelding met een vast punt
        $\ell_{\widetilde g}$. Bewijs
        \[
        d(\ell_g, \ell_{\widetilde g}) \leq
        \frac{d\bigl(g(\ell_{\widetilde g}),
        \widetilde g(\ell_{\widetilde g})\bigr)}{1 - k}
        \leq \frac{\sup_{x \in X} d\bigl(g(x), \widetilde
        g(x)\bigr)}{1 - k}.
        \]
  \item (De iterateertruc) Zij $X$ \omterm{def:b2:metric:complete}{volledig} niet-leeg en $g
        \colon X \to X$ een afbeelding --- niet verondersteld
        \omterm{def:b2:metric:continuity}{continu} --- zodat zekere iteraat $g^m$ een $k$-contractie
        is. Bewijs dat $g$ een uniek vast punt $\ell$ heeft en dat
        \emph{elke} baan $x_{n+1} = g(x_n)$ naar $\ell$
        convergeert. \emph{(Vaste punten van $g$ zijn vaste punten
        van $g^m$; omgekeerd is $g(\ell)$ een vast punt van $g^m$;
        splits de baan langs resten mod $m$.)}
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De stelling van Picard--Lindelöf.}
\begin{enumerate}[resume]
  \item Toon dat een functie $y \colon I \to \intcc{y_0 - b}{y_0
        + b}$ $C^1$ is met $y(t_0) = y_0$ en $y' = f(t, y)$ op $I$
        als en slechts als zij \omterm{def:b2:metric:continuity}{continu} is en voldoet aan de
        integraalvergelijking
        \[
        y(t) = y_0 + \int_{t_0}^{t} f\bigl(s, y(s)\bigr)\dd s
        \qquad (t \in I).
        \]
  \item Zij $X_h = \{y \in C(I) : \abs{y(t) - y_0} \leq b
        \text{ op } I\}$ en zij $T$ gedefinieerd door $T(y)(t) =
        y_0 + \int_{t_0}^{t}f(s, y(s))\dd s$. Toon dat $X_h$ een
        niet-lege gesloten deelverzameling van $C(I)$ is, dus
        \omterm{def:b2:metric:complete}{volledig}, en dat $T$ $X_h$ in $X_h$ afbeeldt --- hier
        werkt $h \leq b/M$.
  \item Toon dat $d_\infty\bigl(T(y), T(z)\bigr) \leq
        Lh\,d_\infty(y, z)$ op $X_h$: als $Lh < 1$, besluit de
        stelling van Banach al. We verwijderen deze
        kleinheidsvoorwaarde hierna.
  \item Bewijs door inductie op $n$:
        \[
        \abs{T^n(y)(t) - T^n(z)(t)} \leq
        \frac{\bigl(L\abs{t - t_0}\bigr)^n}{n!}\,
        d_\infty(y, z) \qquad (y, z \in X_h,\ t \in I),
        \]
        zodat zekere iteraat van $T$ een contractie is. Besluit
        met vraag 4 (\emph{de stelling van Picard--Lindelöf}): het
        Cauchyprobleem $y' = f(t,y)$, $y(t_0) = y_0$ heeft precies
        één oplossing op $I = \intcc{t_0 - h}{t_0 + h}$ met
        waarden in $\intcc{y_0 - b}{y_0 + b}$.
  \item Toon dat de restrictie ``met waarden in $\intcc{y_0 -
        b}{y_0 + b}$'' automatisch is: elke oplossing van het
        Cauchyprobleem gedefinieerd op $I$ waarvan de grafiek in
        $R$ start, blijft in $\intcc{y_0 - b}{y_0 + b}$
        \emph{(beschouw de eerste uittredingstijd en begrens
        $\abs{y(t) - y_0}$ door $M\abs{t - t_0}$)}. Dus geldt
        uniciteit onder alle oplossingen op $I$.
  \item Draai de machine op $y' = y$, $y(0) = 1$, startend van de
        constante $y^{(0)} \equiv 1$: bereken de Picard-iteraten
        $y^{(n)}$, identificeer ze, en beschrijf de convergentie.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Elke hypothese verdient haar loon.}
\begin{enumerate}[resume]
  \item (\omterm{def:b2:metric:continuity}{Lipschitz} faalt, uniciteit faalt) Voor $y' =
        2\sqrt{\abs y}$, $y(0) = 0$: controleer dat $y \equiv 0$
        en, voor elke $c \geq 0$, de functie $y_c(t) = 0$ voor $t
        \leq c$, $y_c(t) = (t - c)^2$ voor $t > c$, alle $C^1$
        oplossingen op $\R$ zijn. Waar precies faalt $y \mapsto
        2\sqrt{\abs y}$ om \omterm{def:b2:metric:continuity}{Lipschitz} te zijn?
  \item (Lokaliteit is echt) Voor $y' = y^2$, $y(0) = 1$: los
        expliciet op, geef het maximale bestaansinterval, en
        bereken de beste $h$ die de stelling over alle keuzes van
        de rechthoek kan certificeren ($a$ groot, $b$ vrij): toon
        $h_{\max} = \sup_{b>0} \frac{b}{(1+b)^2} = \frac14$,
        terwijl de ware oplossing op $\intoo{-\infty}{1}$ leeft.
  \item (Volledigheid is geen decor) Op $X = \Q \cap
        \intcc{1}{2}$ met de gewone afstand, zij $g(x) =
        \frac x2 + \frac1x$. Toon $g(X) \subseteq X$, dat $g$ een
        $\frac12$-contractie is \emph{(middelwaardeongelijkheid)},
        en dat $g$ geen vast punt in $X$ heeft. Welke hypothese van
        de stelling van Banach faalt, en wat is het vaste punt in
        de voltooiing?
  \item (Foutenbalken) Voor een $k$-contractie $g$ op een volledige
        ruimte, bewijs de \emph{a-posteriori}-schatting $d(x_n,
        \ell) \leq \frac{k}{1-k}\,d(x_n, x_{n-1})$. Voor de
        afbeelding van Heron $g(x) = \frac x2 + \frac 1x$ op
        $\intcc{1}{2}$ (vast punt $\sqrt2$), startend bij $x_0 =
        \frac32$: hoeveel stappen eist de \emph{a-priori}-grens
        $\frac{k^n}{1-k}d(x_1, x_0)$ voor nauwkeurigheid
        $10^{-6}$, en hoeveel stappen volstaan in werkelijkheid?
        (Bereken $x_1, x_2, x_3$ en hun fouten; de
        contractiegrens is eerlijk maar pessimistisch --- Heron
        convergeert kwadratisch.)
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Continue afhankelijkheid.} In dit
deel is $Lh < 1$, dus $T$ zelf is een contractie op $X_h$ (vraag
7); schrijf $y[\,y_0\,]$ voor de oplossing met beginwaarde $y_0$.
\begin{enumerate}[resume]
  \item (Afhankelijkheid van de beginwaarde) Zij $z_0$ een andere
        beginwaarde met $\abs{z_0 - y_0}$ klein genoeg dat beide
        problemen in de rechthoek passen. Bewijs met vraag 3
        \[
        d_\infty\bigl(y[y_0], y[z_0]\bigr) \leq
        \frac{\abs{y_0 - z_0}}{1 - Lh}.
        \]
  \item (Lange intervallen door aaneenschakeling) Veronderstel dat
        de oplossingen bestaan op een lang segment, gesneden in $m$
        opeenvolgende stukken waarop de vorige grens geldt met $Lh
        \leq \frac12$. Toon dat de afwijking per stuk met een
        factor ten hoogste $2$ groeit, dus $d_\infty \leq
        2^m\abs{y_0 - z_0}$ overall --- een exponentieel-in-lengte
        grens, de discrete schaduw van de $\eu^{L\abs{t - t_0}}$
        van het lemma van Gronwall (\cref{ch:b2:diffeq}).
  \item (Afhankelijkheid van het veld) Zij $g$ een ander veld op
        $R$, ook $L$-Lipschitz in $y$, met $\sup_R \abs{f - g}
        \leq \varepsilon$. Bewijs dat de overeenkomstige
        oplossingen $d_\infty \leq \frac{\varepsilon h}{1 - Lh}$
        voldoen: modelleringsfout plant zich lineair voort.
  \item (Parameters) Als een familie $f_\lambda$ van velden
        $L$-Lipschitz in $y$ uniform is en $\sup_R\abs{f_\lambda
        - f_\mu} \leq C\abs{\lambda - \mu}$, leid af dat $\lambda
        \mapsto y_\lambda$ \omterm{def:b2:metric:continuity}{Lipschitz} is van de parameterruimte naar
        $\bigl(C(I), d_\infty\bigr)$.
  \item (Systemen kosten niets) Leg uit waarom Delen I, II en IV
        letterlijk gelden voor $y$ met waarden in $\R^n$
        (sup-afstanden gebouwd op een van de afstanden van
        \cref{ex:b2:metric:examples}), en bereken dan alle
        Picard-iteraten voor het systeem $y' = Ay$, $y(0) =
        (c_1, c_2)$, $A = \begin{psmallmatrix} 0 & 1\\ 0 &
        0\end{psmallmatrix}$: toon dat de iteratie stationair
        wordt bij de exacte oplossing na één stap.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Metrische dividenden en synthese.}
\begin{enumerate}[resume]
  \item (Perturbatie van de identiteit) Zij $X$ een
        genormeerde-ruimte-achtig podium: neem $X = C(I)$ of
        $\R^n$. Als $\eta \colon X \to X$ $k$-Lipschitz is met $k
        < 1$, bewijs dat $x \mapsto x + \eta(x)$ een bijectie van
        $X$ is waarvan de inverse $\frac1{1-k}$-Lipschitz is
        \emph{(pas voor elke $y$ Banach toe op $x \mapsto y -
        \eta(x)$)}. Dit is het metrische hart van de
        inversefunctiestelling (\cref{ch:b2:diffcalc}).
  \item (Vergelijking van Kepler) Voor $0 \leq e < 1$ en $m \in
        \R$, bewijs dat $x = m + e\sin x$ precies één oplossing
        heeft, dat de iteratie $x_{n+1} = m + e\sin x_n$ ernaar
        convergeert vanaf elke start, en schat: voor $e =
        \frac12$, $m = 1$, hoeveel iteraties garanderen een fout
        $\leq 10^{-3}$ via de a-priori-grens? (De oplossing is $x
        \approx 1.4987$.)
  \item (De Cantorverzameling is een vast punt) Zij $S_1(x) =
        \frac x3$ en $S_2(x) = \frac x3 + \frac23$ op $\R$, en
        zij $C$ de Cantorverzameling van \cref{exo:b2:metric:8}.
        Bewijs $C = S_1(C) \cup S_2(C)$, en leg in één zin uit
        waarom geen \emph{andere} niet-lege compacte verzameling
        aan deze vergelijking voldoet (de afbeelding $A \mapsto
        S_1(A) \cup S_2(A)$ is een contractie voor een afstand
        tussen compacte verzamelingen --- de Hausdorff-afstand,
        eerlijk gemaakt in het volume van Jaar~3).
  \item (Samenhang globaliseert uniciteit) Zij $f$ lokaal \omterm{def:b2:metric:continuity}{Lipschitz}
        in $y$ op een \omterm{def:b2:metric:topology}{open} verzameling, en zij $y, z$ twee
        oplossingen van $y' = f(t, y)$ op een gemeenschappelijk
        interval $J$ met $y(t_0) = z(t_0)$. Bewijs $y = z$ op $J$:
        toon dat $\{t \in J : y(t) = z(t)\}$ niet-leeg, gesloten
        in $J$, en \omterm{def:b2:metric:topology}{open} in $J$ is (door lokale uniciteit), en
        gebruik de samenhang van intervallen
        (\cref{thm:b2:metric:connectedness}).
  \item (Geen kleinheid voor lineaire vergelijkingen) Voor $y' =
        \alpha(t)y + \beta(t)$ met $\alpha, \beta$ \omterm{def:b2:metric:continuity}{continu} op een
        segment $\intcc{A}{B}$, pas vraag 8 aan om te tonen dat de
        faculteitsgrens op het \emph{hele} segment geldt, dus
        bestaan en uniciteit daar globaal zijn --- het scalaire
        geval van de stelling van Cauchy--Lipschitz van
        \cref{ch:b2:diffeq}, zonder restrictie op de lengte $B -
        A$.
  \item (Synthese) Eén zin elk: wat volledigheid bijdroeg; wat de
        contractie bijdroeg; wat de iterateertruc kocht vergeleken
        met gewone Banach; waar samenhang binnenkwam; en welk
        tegenvoorbeeld van Deel III welke hypothese bewaakt. Noem
        de topstelling, en zeg wat contractie vervangt wanneer $f$
        slechts \omterm{def:b2:metric:continuity}{continu} is (stelling van Peano, via compactheid
        in functieruimten --- de stelling van Arzelà--Ascoli van
        het volume van Jaar~3).
\end{enumerate}
\end{problem}
