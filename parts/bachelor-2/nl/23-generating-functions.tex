\chapter{Kansgenererende functies}\label{ch:b2:genfun}

De machtreeksen van \cref{ch:b2:powerseries} keren terug met een
probabilistische missie: aan een $\N$-waardige \omterm{def:b2:randomvar:law}{stochastische
variabele} koppelen we de machtreeks met coëfficiënten $\P(X =
n)$. Deze \emph{\omterm{ex:b2:powerseries:fibonacci}{genererende functie}} zet sommen van
onafhankelijke variabelen om in producten, momenten in
afgeleiden in $1$, en harde combinatorische identiteiten in
één-regel-vermenigvuldigingen. Het hoofdstuk sluit het boek af
met twee pronkstukken: de Poisson-benadering van zeldzame
\omterm{def:b2:proba:space}{gebeurtenissen}, en het uitstervingscriterium voor
vertakkingsprocessen --- een werkelijk oneindige
probabilistische berekening, geheel opgelost door de
meetkunde van een convexe curve.

\section{Definitie en basiseigenschappen}

\begin{definition}[Kansgenererende
functie]\label{def:b2:genfun:pgf}
Zij $X$ een $\N$-waardige \omterm{def:b2:randomvar:law}{stochastische variabele}, $p_n = \P(X
= n)$. De \emph{kansgenererende
functie}\index{kansgenererende functie} van $X$ is de som van
de machtreeks
\[
G_X(t) = \E\bigl(t^X\bigr) = \sum_{n=0}^{\infty} p_n\,t^n .
\]
\end{definition}

\begin{example}[Eerste reflexen]\label{ex:b2:genfun:firstreflexes}
Een constante variabele $X = c$ heeft $G_X(t) = t^c$; een
verschuiving gehoorzaamt $G_{X+c}(t) = t^c\,G_X(t)$; en
evalueren in speciale punten leest informatie af zonder enige
ontwikkeling: $G_X(0) = \P(X = 0)$, $G_X(1) = 1$, en $G_X(-1)
= \P(X\text{ even}) - \P(X\text{ oneven})$, de
pariteitsbalans uitgebuit in \cref{exo:b2:genfun:10}. Deze
één-regels worden overal hieronder stilzwijgend gebruikt ---
en de evaluatie $G_X(0)$ is precies hoe uitsterfkansen uit
geïtereerde \omterm{ex:b2:powerseries:fibonacci}{genererende functies} aan het eind van het
hoofdstuk geëxtraheerd worden.
\end{example}

\begin{proposition}[Straal en eerste
eigenschappen]\label{prop:b2:genfun:radius}
De reeks die $G_X$ definieert heeft \omterm{def:b2:powerseries:radius}{convergentiestraal} $\geq
1$; $G_X$ is gedefinieerd en continu op $\intcc{-1}{1}$,
$\mathcal{C}^\infty$ op $\intoo{-1}{1}$, met $G_X(1) = 1$ en
$\abs{G_X(t)} \leq 1$ daar. Bovendien bepaalt $G_X$ de wet van
$X$:
\[
p_n = \frac{G_X^{(n)}(0)}{n!} .
\]
\end{proposition}

\begin{proof}
Omdat $\sum p_n = 1$ convergeert, zijn de termen $p_n\,1^n$
begrensd, dus de straal is $\geq 1$ (lemma van Abel,
\cref{ch:b2:powerseries}); in $t = \pm1$ convergeert de reeks
absoluut ($\sum p_n = 1$ domineert); beter nog, op het hele
interval $\intcc{-1}1$,
\[
\sup_{\abs t\leq1}\,\abs{p_nt^n} = p_n
\quad\text{met}\quad \sum_np_n < \infty :
\]
de reeks convergeert \emph{normaal} op $\intcc{-1}1$, dus haar
som is daar continu
(\cref{thm:b2:funcseq:weierstrass,thm:b2:funcseq:continuity}).
Gladheid binnenin en de coëfficiëntenformule zijn de algemene
theorie van machtreeksen; omdat de coëfficiënten
terugwinbaar zijn, hebben twee variabelen met dezelfde
\omterm{ex:b2:powerseries:fibonacci}{genererende functie} dezelfde wet.
\end{proof}

\begin{example}[De klassieke wetten]\label{ex:b2:genfun:classical}
\leavevmode
\begin{itemize}
\item Bernoulli $\mathcal{B}(p)$: $G(t) = 1 - p + pt$.
\item Binomiaal $\mathcal{B}(n, p)$: $G(t) = \sum_k \binom nk
(pt)^k(1-p)^{n-k} = (1 - p + pt)^n$ (binomium van Newton).
\item Geometrisch $\mathcal{G}(p)$: $G(t) =
\sum_{k\geq1}(1-p)^{k-1}p\,t^k = \dfrac{pt}{1 - (1-p)t}$
(straal $\frac{1}{1-p} > 1$).
\item Poisson $\mathcal{P}(\lambda)$: $G(t) =
\sum_k e^{-\lambda}\frac{(\lambda t)^k}{k!} = e^{\lambda(t -
1)}$ (straal $\infty$).
\end{itemize}
\end{example}

\begin{example}[De genererende functie
integreren]\label{ex:b2:genfun:integraltrick}
Afgeleiden van $G_X$ in $1$ geven positieve momenten; de
\emph{integraal} geeft een negatief. Uit $\int_0^1t^k\dd t =
\frac1{k+1}$ en term-voor-term-integratie (\omterm{def:b2:funcseq:series}{normale
convergentie} op $\intcc01$):
\[
\int_0^1G_X(t)\,\dd t = \sum_{k\geq0}\frac{\P(X =
k)}{k+1} = \E\Bigl(\frac1{1+X}\Bigr).
\]
Voor $X \sim \mathcal P(\lambda)$:
\[
\E\Bigl(\frac1{1+X}\Bigr) =
\int_0^1\eu^{\lambda(t-1)}\,\dd t = \frac{1 -
\eu^{-\lambda}}{\lambda},
\]
in één regel de reeksberekening van
\cref{ex:b2:randomvar:transferex} herstellend. De \omterm{ex:b2:powerseries:fibonacci}{genererende
functie} is een tweerichtingsinstrument: differentieer in $1$
voor de momenten $\E(X)$, $\E(X(X-1))$, integreer over
$\intcc01$ voor $\E\bigl(\frac1{1+X}\bigr)$ --- één
\omterm{def:b2:powerseries:analytic}{analytisch} object, bevraagd in welke richting het probleem
nodig heeft.
\end{example}

\begin{example}[Een wet met straal precies
één]\label{ex:b2:genfun:heavytail}
Laat $\P(X = k) = \dfrac{6}{\pi^2k^2}$ voor $k \geq 1$ --- een
waarschijnlijkheidswet door de identiteit van Basel
(\cref{ex:b2:fourier:basel}). Haar \omterm{ex:b2:powerseries:fibonacci}{genererende functie} $G(t)
= \frac6{\pi^2}\sum_{k\geq1}\frac{t^k}{k^2}$ heeft
\omterm{def:b2:powerseries:radius}{convergentiestraal} precies $1$: de algemene grens ``straal
$\geq 1$'' van \cref{prop:b2:genfun:radius} kan niet verbeterd
worden. En het gemiddelde is
\[
\sum_{k\geq1}k\,\P(X = k) =
\frac6{\pi^2}\sum_{k\geq1}\frac1k = \infty :
\]
$G$ is continu op $\intcc{-1}1$, glad binnenin, maar haar
afgeleide blaast op in $1^-$ --- de grafiek arriveert in het
punt $(1, 1)$ met een verticale \omterm{def:b2:curves:arc}{raaklijn}. Zware staarten
zijn \emph{meetkundig} zichtbaar op de \omterm{ex:b2:powerseries:fibonacci}{genererende functie},
in het enige punt $t = 1$; de momentenstelling hieronder
maakt deze correspondentie exact.
\end{example}

\begin{theorem}[Momenten uit de genererende
functie]\label{thm:b2:genfun:moments}
$X$ heeft een \omterm{def:b2:randomvar:expectation}{verwachting} dan en slechts dan als $G_X$
\omterm{def:b2:diffcalc:differential}{differentieerbaar} is in $1^-$ (linkerafgeleide, eindig), en
dan $\E(X) = G_X'(1)$. Evenzo heeft $X$ een tweede moment iff
$G_X$ tweemaal \omterm{def:b2:diffcalc:differential}{differentieerbaar} is in $1^-$, en dan
\[
\E\bigl(X(X - 1)\bigr) = G_X''(1),
\qquad
V(X) = G_X''(1) + G_X'(1) - G_X'(1)^2 .
\]
\end{theorem}

\begin{proof}
Voor $t \in \intoo{0}{1}$ geeft term-voor-term-differentiatie
binnen de schijf $G_X'(t) = \sum_{n\geq1} np_n t^{n-1}$, een
reeks met niet-negatieve coëfficiënten: $t \mapsto G_X'(t)$ is
niet-dalend op $\intoo{0}{1}$, en door monotone convergentie
van deelsommen (of de stelling van Abel voor niet-negatieve
coëfficiënten, \cref{ch:b2:powerseries}),
\[
\lim_{t \to 1^-} G_X'(t)
= \sum_{n\geq1} n\,p_n \in \intcc{0}{+\infty} ,
\]
elke zijde eindig precies wanneer de andere is. Wanneer
eindig, knijpt de middelwaardestelling de
differentiequotiënten $\frac{G_X(1) - G_X(t)}{1 - t}$ tussen
waarden van $G_X'$, zodat $G_X$ \omterm{def:b2:diffcalc:differential}{differentieerbaar} is in $1^-$
met $G_X'(1) = \sum np_n = \E(X)$ (door overdracht). De
tweede-orde-uitspraak herhaalt het argument één niveau hoger:
$G''_X(t) = \sum_{n\geq2}n(n-1)p_nt^{n-2}$ is niet-dalend op
$\intoo01$ met monotone limiet $\sum_nn(n-1)p_n =
\E(X(X-1))$, eindig precies wanneer $X$ een tweede moment
heeft. De variantieformule volgt dan uit König--Huygens:
\[
V(X) = \E(X^2) - \E(X)^2 = \E\bigl(X(X-1)\bigr) + \E(X) -
\E(X)^2 = G''_X(1) + G'_X(1) - G'_X(1)^2 .
\]
\end{proof}

\begin{example}
Poisson: $G'(t) = \lambda e^{\lambda(t-1)}$, dus $\E(X) =
\lambda$; $G''(1) = \lambda^2$, dus $V(X) = \lambda^2 +
\lambda - \lambda^2 = \lambda$ --- de berekeningen van
\cref{ch:b2:randomvar} in één regel elk.
\end{example}

\begin{example}[De modus van een Poisson-wet]
\label{ex:b2:genfun:poissonmode}
Waar is $\P(X = k)$ het grootst voor $X \sim \mathcal
P(\lambda)$? Opeenvolgende gewichten vergelijken via de
verhouding
\[
\frac{\P(X = k+1)}{\P(X = k)} = \frac{\lambda}{k + 1} ,
\]
die $1$ overschrijdt zolang $k < \lambda - 1$ en onder $1$
zakt zodra $k > \lambda - 1$: de gewichten stijgen dan dalen,
met modus $\floor\lambda$ (en een gelijkspel tussen $\lambda
- 1$ en $\lambda$ wanneer $\lambda$ geheel is: voor $\lambda
= 3$, $\P(X = 2) = \P(X = 3) = \frac92\eu^{-3} \approx
0.224$). Verhoudingstests op de coëfficiënten zijn vaak de
snelste route naar kwalitatieve feiten over een discrete wet
--- geen \omterm{ex:b2:powerseries:fibonacci}{genererende functie} nodig, maar de coëfficiënten
\emph{zijn} de \omterm{ex:b2:powerseries:fibonacci}{genererende functie}, term voor term gelezen.
\end{example}

\section{Sommen van onafhankelijke variabelen}

\begin{theorem}[Multiplicativiteit]\label{thm:b2:genfun:product}
Als $X$ en $Y$ onafhankelijke $\N$-waardige \omterm{def:b2:randomvar:law}{stochastische
variabelen} zijn, dan
\[
G_{X + Y}(t) = G_X(t)\,G_Y(t)
\qquad (\abs t \leq 1),
\]
en door inductie $G_{X_1 + \dots + X_n} = \prod_i G_{X_i}$
voor onafhankelijke $X_1, \dots, X_n$.
\end{theorem}

\begin{proof}
Twee bewijzen, beide leerzaam. \emph{Via \omterm{def:b2:randomvar:expectation}{verwachtingen}:}
$t^X$ en $t^Y$ zijn onafhankelijke begrensde variabelen, dus
(\cref{thm:b2:randomvar:product})
\[
G_{X+Y}(t) = \E\bigl(t^{X+Y}\bigr)
= \E\bigl(t^X t^Y\bigr)
= \E\bigl(t^X\bigr)\E\bigl(t^Y\bigr) .
\]
\emph{Via \omterm{thm:b2:series:fubini}{Cauchy-producten}:} de wet van $X + Y$ is de
convolutie $\P(X + Y = n) = \sum_{k=0}^n \P(X = k)\P(Y = n -
k)$, en de Cauchy-productstelling voor absoluut convergente
reeksen (\cref{ch:b2:series}) vermenigvuldigt de twee
machtreeksen precies langs deze convolutie.
\end{proof}

\begin{example}[Stabiliteit van de klassieke
wetten]\label{ex:b2:genfun:stability}
Onafhankelijke binomialen met dezelfde $p$ tellen op: $(1 - p
+ pt)^m(1 - p + pt)^n = (1 - p + pt)^{m+n}$, dus
$\mathcal{B}(m, p) + \mathcal{B}(n, p) = \mathcal{B}(m + n,
p)$ --- in het bijzonder is een som van $n$ onafhankelijke
Bernoulli-variabelen binomiaal, wat de wet van het aantal
successen opnieuw bewijst. Onafhankelijke Poissons tellen op:
$e^{\lambda(t-1)}e^{\mu(t-1)} = e^{(\lambda + \mu)(t-1)}$,
dus $\mathcal{P}(\lambda) + \mathcal{P}(\mu) =
\mathcal{P}(\lambda + \mu)$ --- de convolutieberekening van
\cref{exo:b2:randomvar:2}, nu zonder berekening.
\end{example}

\begin{example}[Twee dobbelstenen, één polynoom
gekwadrateerd]\label{ex:b2:genfun:twodice}
Voor één eerlijke dobbelsteen, $G(t) = \frac{t + t^2 + \dots
+ t^6}{6}$; voor de som van twee,
\[
G(t)^2 = \frac{1}{36}\bigl(t^2 + 2t^3 + 3t^4 + 4t^5 + 5t^6 +
6t^7 + 5t^8 + 4t^9 + 3t^{10} + 2t^{11} + t^{12}\bigr) :
\]
de driehoekige wet van dobbelsteensommen ($7$ is de modus,
met waarschijnlijkheid $\frac6{36} = \frac16$), afgelezen van
een polynoomkwadraat dat men één keer in een leven
uitvermenigvuldigt. De convolutieformule zou elf aparte
telargumenten geëist hebben; de \omterm{ex:b2:powerseries:fibonacci}{genererende functie} doet ze
allemaal tegelijk, omdat polynomen vermenigvuldigen
\emph{is} coëfficiënten convolueren. Deze mechanische
vertaling --- wetten naar coëfficiënten, sommen naar
producten --- is het hele bedrijfsmodel van het hoofdstuk, en
\cref{exo:b2:genfun:11} duwt haar tot de verrassende
Sicherman-dobbelstenen.
\end{example}

\begin{example}[Drie dobbelstenen en een
coëfficiëntenextractie]\label{ex:b2:genfun:threedice}
Voor de som $S$ van drie eerlijke dobbelstenen is $\P(S =
10)$ de coëfficiënt van $t^{10}$ in $\bigl(\frac{t + \dots +
t^6}6\bigr)^3$. Factoriseer en ontwikkel met de binomiale en
geometrische reeks:
\[
\Bigl(\frac{t(1 - t^6)}{6(1 - t)}\Bigr)^{\!3}
= \frac{t^3}{216}\,\bigl(1 - 3t^6 + 3t^{12} -
t^{18}\bigr)\sum_{j\geq0}\binom{j+2}2t^j .
\]
De coëfficiënt van $t^{10}$ eist $t^7$ uit het product: $j =
7$ met de term $1$, en $j = 1$ met de term $-3t^6$:
\[
\P(S = 10) = \frac{1}{216}\Bigl(\binom92 -
3\binom32\Bigr) = \frac{36 - 9}{216} = \frac{27}{216} =
\frac18 .
\]
Directe enumeratie van de $27$ triples is foutgevoelig; de
algebra is mechanisch en schaalt naar elk aantal stenen ---
de inclusie--exclusie zichtbaar in $(1 - t^6)^3$ doet de
gevalsanalyse automatisch.
\end{example}

\begin{example}[Een wet aflezen van haar genererende
functie]\label{ex:b2:genfun:identify}
Welke wet heeft $G(t) = \dfrac1{2 - t}$? Ontwikkel tot een
machtreeks:
\[
\frac{1}{2 - t} = \frac12\cdot\frac1{1 - t/2}
= \sum_{k\geq0}\frac{t^k}{2^{k+1}} :
\]
niet-negatieve coëfficiënten sommerend tot $G(1) = 1$, dus
dit is een echte wet, $\P(X = k) = 2^{-(k+1)}$ op $\N$ ---
een geometrische wet startend in $0$. Door uniciteit
(\cref{prop:b2:genfun:radius}) deelt geen andere wet deze
$G$. Wetten herkennen aan hun \omterm{ex:b2:powerseries:fibonacci}{genererende functies} is een
vaardigheid die oefening verdient: zo wordt de kritieke
vertakkingsiteratie $G_n(t) = \frac{n - (n-1)t}{n+1 - nt}$
van het weekendprobleem ontmaskerd als een geometrische wet
geconditioneerd op overleving.
\end{example}

\begin{remark}
Stabiliteit gaat maar één kant op: sommen van onafhankelijke
Poissons zijn Poisson, maar verschillen niet --- $X - Y$
neemt negatieve waarden, dus heeft helemaal geen \omterm{ex:b2:powerseries:fibonacci}{genererende
functie}, en haar wet (de Skellam-verdeling) ligt buiten de
gereedschapskist van dit hoofdstuk. Evenzo is $\mathcal B(m,
p) + \mathcal B(n, p')$ met $p \neq p'$ \emph{niet}
binomiaal: het product $(1 - p + pt)^m(1 - p' + p't)^n$
heeft twee verschillende wortellocaties, terwijl elke
binomiale pgf één herhaalde wortel heeft. Stabiliteit
aflezen van wortelpatronen is een kleine vooruitblik op
hoeveel structuur het polynoom codeert.
\end{remark}

\begin{remark}[Het wortels-van-eenheid-filter]
Evalueren in $-1$ scheidt even van oneven; evalueren in alle
$m$-de eenheidswortels scheidt elke restklasse: met $\omega =
\eu^{2\iu\pi/m}$,
\[
\P(X \equiv r \bmod m)
= \frac1m\sum_{j=0}^{m-1}\omega^{-jr}\,G_X(\omega^j),
\]
omdat middelen van $\omega^{j(k-r)}$ over $j$ $1$ oplevert
als $k \equiv r$ en $0$ anders. Voorbeelddividend: voor de
som $S$ van twee eerlijke dobbelstenen, elke $G(\omega^j) =
\frac16\sum_{k=1}^6 \omega^{jk} = -\frac16$ voor $j \neq 0$
(de zeven zevende eenheidswortels sommeren tot nul), dus
\[
\P(7 \mid S) = \frac17\Bigl(1 +
6\cdot\frac1{36}\Bigr) = \frac16 ,
\]
wat de telling van \cref{ex:b2:genfun:twodice} bevestigt ---
en de methode schaalt naar vragen waar direct tellen dat
niet doet.
\end{remark}

\begin{theorem}[Stochastische sommen: de identiteit van Wald
voor genererende
functies]\label{thm:b2:genfun:compound}
Zij $(X_k)_{k\geq1}$ onafhankelijke $\N$-waardige variabelen
met dezelfde wet en \omterm{ex:b2:powerseries:fibonacci}{genererende functie} $G_X$, en zij $N$ een
$\N$-waardige variabele \omterm{def:b2:proba:independence}{onafhankelijk} van de $X_k$, met
\omterm{ex:b2:powerseries:fibonacci}{genererende functie} $G_N$. Dan heeft de stochastische som $S
= X_1 + \dots + X_N$ (met $S = 0$ wanneer $N = 0$)
\omterm{ex:b2:powerseries:fibonacci}{genererende functie}
\[
G_S = G_N \circ G_X .
\]
In het bijzonder, als $N$ en $X_1$ \omterm{def:b2:randomvar:expectation}{verwachtingen} hebben,
$\E(S) = \E(N)\,\E(X_1)$.
\end{theorem}

\begin{proof}
Conditioneer op $N$ (totale waarschijnlijkheid,
\cref{thm:b2:proba:bayes}): voor $\abs t \leq 1$,
\[
G_S(t) = \sum_{n=0}^\infty \P(N = n)\,
\E\bigl(t^{X_1 + \dots + X_n}\bigr)
= \sum_{n=0}^\infty \P(N = n)\,G_X(t)^n
= G_N\bigl(G_X(t)\bigr),
\]
met multiplicativiteit voor elke vaste $n$ en de
sommeerbaarheid van de hele dubbele familie ($\abs{G_X(t)}
\leq 1$). De verwisseling van sommaties is Fubini voor
\omterm{def:b2:series:summable}{sommeerbare families} (\cref{ch:b2:series}). Differentiëren
in $1^-$ door de kettingregel en
\cref{thm:b2:genfun:moments}: $\E(S) =
G_N'(G_X(1))\,G_X'(1) = G_N'(1)G_X'(1) = \E(N)\E(X_1)$.
\end{proof}

\begin{example}[Samengestelde Poisson: jaarlijkse
verzekeringsverliezen]\label{ex:b2:genfun:compoundpoisson}
Een verzekeraar ontvangt $N \sim \mathcal P(\lambda)$ claims
in een jaar, elke claim kostend $X_k$ (gehele eenheden,
i.i.d., pgf $G_X$, gemiddelde $\mu$, onafhankelijk van $N$).
Door \cref{thm:b2:genfun:compound} heeft het totale verlies
$S$
\[
G_S(t) = \eu^{\lambda(G_X(t) - 1)},
\qquad
\E(S) = \lambda\mu ,
\]
en tweemaal differentiëren in $1^-$:
\[
V(S) = \lambda\,G_X''(1) + \lambda^2\mu^2 + \lambda\mu -
(\lambda\mu)^2 = \lambda\,\E(X^2) .
\]
De \omterm{def:b2:randomvar:variance}{variantie} betrekt het \emph{tweede} moment van een enkele
claim, niet haar \omterm{def:b2:randomvar:variance}{variantie}: een samengestelde Poissonsom
voelt de occasionele grote claim tweemaal --- één keer via
hoeveel, één keer via hoe groot. Voor $\lambda = 10$ claims
van geometrische wet met gemiddelde $2$ ($\E X^2 = 6$): $\E
S = 20$, $V(S) = 60$, en Chebyshev (\cref{ch:b2:randomvar})
levert al bruikbare solvabiliteitsmarges. Dit
``willekeurig gestopte som''-patroon is hetzelfde dat de
vertakkingsrecursie van \cref{prop:b2:genfun:branching} zal
drijven: compositie van \omterm{ex:b2:powerseries:fibonacci}{genererende functies} is de algebra
van stochastische populaties.
\end{example}

\begin{remark}
De onafhankelijkheid van $N$ van de sommanden is niet
decoratief. Neem $X_k \in \{0, 2\}$ met gelijke
waarschijnlijkheden en laat $N = X_1$ (flagrant afhankelijk):
dan is $S = X_1 + \dots + X_N$ gelijk aan $0$ wanneer $X_1 =
0$, en $2 + X_2$ wanneer $X_1 = 2$, dus $\E(S) =
\frac12(2 + 1) = \frac32$, terwijl $\E(N)\E(X_1) = 1\cdot1 =
1$: de identiteit van Wald faalt. Wanneer het aantal termen
mag \emph{reageren} op de termen zelf, stort de schone
productstructuur in --- de volle theorie van zulke
``stop''-regels is het martingaalhoofdstuk van het Bachelor
jaar~3-volume.
\end{remark}

\section{Poisson-benadering}

\begin{theorem}[Wet van zeldzame
gebeurtenissen]\label{thm:b2:genfun:rareevents}
Zij $X_n \sim \mathcal{B}(n, p_n)$ met $n\,p_n \to \lambda >
0$. Dan voor elke $k \in \N$:
\[
\P(X_n = k)
\xrightarrow[n\to\infty]{}
e^{-\lambda}\frac{\lambda^k}{k!} :
\]
de binomiale wet van vele zeldzame \omterm{def:b2:proba:independence}{onafhankelijke
gebeurtenissen} convergeert naar de Poisson-wet van parameter
$\lambda$.
\end{theorem}

\begin{proof}
Directe berekening met $p_n = \frac{\lambda_n}{n}$,
$\lambda_n \to \lambda$:
\[
\P(X_n = k)
= \binom nk p_n^k(1 - p_n)^{n-k}
= \frac{n(n-1)\cdots(n-k+1)}{n^k}\cdot
\frac{\lambda_n^k}{k!}\,
\bigl(1 - \tfrac{\lambda_n}{n}\bigr)^{n-k} .
\]
Als $n \to \infty$ met $k$ vast: de eerste factor neigt naar
$1$ (product van $k$ factoren $\to 1$); $\lambda_n^k \to
\lambda^k$; en $\bigl(1 - \frac{\lambda_n}{n}\bigr)^{n-k} =
\exp\bigl((n-k)\ln(1 - \frac{\lambda_n}{n})\bigr) \to
e^{-\lambda}$ omdat $(n - k)\ln\bigl(1 -
\frac{\lambda_n}{n}\bigr) \sim -\lambda_n \to -\lambda$
(\cref{ch:b2:comparison}). Alternatief, op het niveau van
\omterm{ex:b2:powerseries:fibonacci}{genererende functies}: $G_{X_n}(t) = \bigl(1 +
\frac{\lambda_n(t-1)}{n}\bigr)^n \to e^{\lambda(t - 1)} =
G_{\mathcal{P}(\lambda)}(t)$ voor elke vaste $t \in [0, 1]$
--- convergentie van \omterm{ex:b2:powerseries:fibonacci}{genererende functies}, die (voor
$\N$-waardige variabelen) equivalent is aan convergentie van
elke $\P(X_n = k)$; zie \cref{exo:b2:genfun:9}.
\end{proof}

\begin{remark}
Dit is waarom Poisson-wetten tellingen van zeldzame
\omterm{def:b2:proba:space}{gebeurtenissen} modelleren --- typefouten per pagina,
radioactieve vervallen per seconde, ongelukken per dag op
een kruising: elke kans is bijna verwaarloosbaar, kansen zijn
talrijk, en alleen het gemiddelde tempo $\lambda$ overleeft
in de limiet.
\end{remark}

\begin{example}[De Poisson-limiet zien
convergeren]\label{ex:b2:genfun:poissonnumerics}
Fixeer $\lambda = 2$ en laat $X_n \sim \mathcal B(n, 2/n)$.
De geen-gebeurtenis-waarschijnlijkheid is precies $\P(X_n =
0) = (1 - 2/n)^n$:
\[
n = 10:\ 0.107, \qquad
n = 20:\ 0.122, \qquad
n = 50:\ 0.130, \qquad
n = 100:\ 0.133,
\]
tegenover de limiet $\eu^{-2} \approx 0.135$. De
convergentie is monotoon en van snelheid $O(1/n)$ ---
ontwikkelend, $(1 - 2/n)^n = \eu^{-2}\bigl(1 - \tfrac2n +
O(n^{-2})\bigr)$ --- dus voor $n$ in de honderden is het
Poisson-model al accuraat tot op de derde cijfer. Dit is de
praktische inhoud van de wet van zeldzame gebeurtenissen: de
modelleerder kent $n$ en $p$ nooit apart (hoeveel
micro-kansen voor een typefout houdt een pagina?), maar
alleen hun product $\lambda$, en de limietwet hangt
genadig van niets anders af.
\end{example}

\section{Vertakkingsprocessen}

Beschouw een populatie startend van één voorouder; elk
individu heeft, \omterm{def:b2:proba:independence}{onafhankelijk}, een willekeurig aantal
kinderen met wet $(p_k)_{k \in \N}$ en \omterm{ex:b2:powerseries:fibonacci}{genererende functie}
$G$ (de \emph{nakomelingenverdeling}). Laat $Z_n$ de grootte
van generatie $n$ zijn ($Z_0 = 1$), en laat $m = G'(1) =
\E(Z_1)$ het gemiddelde nakomelingenaantal zijn.

\begin{proposition}\label{prop:b2:genfun:branching}
De \omterm{ex:b2:powerseries:fibonacci}{genererende functie} van $Z_n$ is de $n$-de iterate
$G_{Z_n} = G \circ G \circ \dots \circ G$ ($n$ keer), en de
uitsterfkansen $q_n = \P(Z_n = 0)$ voldoen aan
\[
q_0 = 0, \qquad q_{n+1} = G(q_n),
\]
en stijgen naar de waarschijnlijkheid $q$ van uiteindelijk
uitsterven, die een vast punt van $G$ is.
\end{proposition}

\begin{proof}
Generatie $n + 1$ is de stochastische som van de
nakomelingen van de $Z_n$ leden van generatie $n$, met
tellingen \omterm{def:b2:proba:independence}{onafhankelijk} van elkaar en van $Z_n$:
\cref{thm:b2:genfun:compound} geeft $G_{Z_{n+1}} =
G_{Z_n} \circ G$, en inductie vanaf $G_{Z_0}(t) = t$ levert
de $n$-voudige iterate --- die, door associativiteit van
compositie, even goed gelezen kan worden als $G_{Z_{n+1}} =
G \circ G_{Z_n}$. Evalueren van deze tweede vorm in $0$:
$q_{n+1} = G_{Z_{n+1}}(0) = G\bigl(G_{Z_n}(0)\bigr) =
G(q_n)$. De \omterm{def:b2:proba:space}{gebeurtenissen} $\{Z_n = 0\}$ stijgen
(uitgestorven populaties blijven uitgestorven), dus $q_n
\uparrow q = \P\bigl(\bigcup_n\{Z_n = 0\}\bigr)$ door
monotone continuïteit (\cref{thm:b2:proba:continuity}), en
continuïteit van $G$ op $[0, 1]$ zet $q_{n+1} = G(q_n)$ om
in $q = G(q)$ in de limiet.
\end{proof}

\begin{example}[Uitsterven zien
convergeren]\label{ex:b2:genfun:cobwebnumerics}
Voor de nakomelingenwet $(p_0, p_1, p_2) = (\tfrac14,
\tfrac14, \tfrac12)$ van \cref{ex:b2:genfun:branchingexample},
$G(t) = \tfrac14 + \tfrac14t + \tfrac12t^2$ en de iteratie
$q_{n+1} = G(q_n)$ geeft
\[
q_1 = 0.25, \quad q_2 = 0.34375, \quad q_3 \approx 0.39502,
\quad q_4 \approx 0.42678, \quad q_5 \approx 0.44776,
\]
klimmend naar de uitsterfkans $q = \tfrac12$. De gaten $q -
q_n$ zijn $0.25$, $0.156$, $0.105$, $0.073$, $0.052$: elk is
ongeveer $\tfrac34$ van de vorige, en inderdaad geeft de
middelwaardestelling $q - q_{n+1} = G'(c_n)(q - q_n)$ met
$G'(q) = \tfrac14 + q = \tfrac34$. Twee moralen: een
familielijn nog levend in generatie $n$ heeft, ingebouwd in
dezelfde berekening, waarschijnlijkheid $q - q_n$ later
gedoemd te zijn; en de convergentiesnelheid van de trap in
de figuur hieronder is de afgeleide in het vast punt --- het
weekendprobleem zet beide observaties om in stellingen.
\end{example}

\begin{theorem}[Uitstervingscriterium]\label{thm:b2:genfun:extinction}
Neem aan $p_1 \neq 1$. De uitsterfkans $q$ is het
\emph{kleinste} vast punt van $G$ in $\intcc{0}{1}$, en:
\begin{itemize}
\item als $m \leq 1$ (subkritisch of kritiek), $q = 1$:
uitsterven is zeker;
\item als $m > 1$ (superkritisch), $q < 1$: de populatie
overleeft voor altijd met positieve waarschijnlijkheid $1 -
q$.
\end{itemize}
\end{theorem}

\begin{proof}
$G$ is \omterm{def:b2:affine:convex}{convex} op $\intcc{0}{1}$ (machtreeks met
niet-negatieve coëfficiënten: $G'' \geq 0$), niet-dalend,
met $G(1) = 1$.

\emph{Kleinste vast punt:} laat $r \in \intcc{0}{1}$ enig
vast punt zijn. Dan $q_0 = 0 \leq r$, en inductief $q_{n+1}
= G(q_n) \leq G(r) = r$ (monotoniciteit): dus $q = \lim
q_n \leq r$.

\emph{Geval $m \leq 1$:} veronderstel $r < 1$ is een vast
punt. Door de middelwaardestelling op $[r, 1]$ bestaat $c
\in \intoo{r}{1}$ met $G'(c) = \frac{G(1) - G(r)}{1 - r} =
\frac{1 - r}{1 - r} = 1$. Maar $G'$ is niet-dalend
(convexiteit) met $\lim_{t\to1^-}G'(t) = m \leq 1$, dus $G'
\leq 1$ op $\intoo{0}{1}$; de gelijkheid $G'(c) = 1$ dwingt
dan $G'$ constant gelijk aan $1$ op $\intco{c}{1}$, vandaar
$G'' = \sum n(n-1)p_nt^{n-2} \equiv 0$ daar. Een
machtreeks met niet-negatieve coëfficiënten die verdwijnt
op een interval heeft al die coëfficiënten nul: $p_n = 0$
voor $n \geq 2$, dus $G(t) = p_0 + p_1t$ en $1 = G'(c) =
p_1$ --- tegensprekend de hypothese $p_1 \neq 1$. Dus $1$
is het enige vast punt: $q = 1$.

\emph{Geval $m > 1$:} nabij $1$ heeft $G(t) - t$ afgeleide
$G'(t) - 1 \to m - 1 > 0$ als $t \to 1^-$, dus $G(t) - t <
G(1) - 1 = 0$ op enig interval $\intoo{1 - \delta}{1}$: de
continue functie $G(t) - t$ is $\geq 0$ in $t = 0$ ($G(0) =
p_0 \geq 0$) en $< 0$ net onder $1$, dus verdwijnt zij in
enig $r < 1$ (tussenwaardestelling). Het kleinste vast punt
is dan $q \leq r < 1$.
\end{proof}

\begin{remark}[Hoe de cobweb te lezen]
In de figuur past een verticale beweging $G$ toe (van $(q_n,
q_n)$ omhoog naar $(q_n, G(q_n))$), een horizontale
beweging naar de diagonaal zet output om in input: de trap
\emph{is} de recursie $q_{n+1} = G(q_n)$. Convexiteit van
$G$ en $G(1) = 1$ laten slechts twee meetkunden over. Ofwel
blijft de curve boven de diagonaal op $\intco01$
(gemiddelde $m \leq 1$): de trap heeft nergens te stoppen
vóór $1$. Of de curve kruist in enig $q < 1$ ($m > 1$): de
trap is gevangen onder de kruising en convergeert ernaar,
met de geometrische snelheid $G'(q) < 1$ gekwantificeerd in
\cref{ex:b2:genfun:cobwebnumerics}. Alle analyse van de
uitstervingsstelling is zichtbaar in dit ene plaatje ---
waarom het waard is te tekenen vóór te rekenen.
\end{remark}

\begin{example}\label{ex:b2:genfun:branchingexample}
Nakomelingenwet: geen kind, één kind, twee kinderen met
waarschijnlijkheden $\frac14, \frac14, \frac12$. Dan $m =
\frac14 + 1 = \frac54 > 1$ en $G(t) = \frac14 + \frac14 t +
\frac12 t^2$. Vastpunten: $\frac12 t^2 - \frac34 t +
\frac14 = 0$, d.w.z.\ $2t^2 - 3t + 1 = (2t - 1)(t - 1) =
0$: $q = \frac12$. De familielijn sterft uit met
waarschijnlijkheid $\frac12$ --- en met
waarschijnlijkheid $\frac12$ leeft zij voor altijd.
\end{example}

\begin{remark}[Gangbare valkuilen]
(i) \omterm{ex:b2:powerseries:fibonacci}{Genererende functies} gelden alleen voor $\N$-waardige
variabelen: voor getekende of niet-gehele variabelen verliest
het object $\E(t^X)$ haar machtreeksstructuur (Bachelor
jaar~3 vervangt haar door transformen aangepast aan $\R$).
(ii) De eerste sanity-check van elke berekende $G$ is $G(1)
= 1$; de tweede is dat de coëfficiënten niet-negatief zijn
--- een negatieve coëfficiënt betekent een algebrafout, niet
een nieuwe wet. (iii) Bij stochastische sommen telt de
compositievolgorde: $G_S = G_N \circ G_X$, de
\emph{buitenste} functie tellend de termen; de andere kant
om is zinloos ($G_X \circ G_N$ zou items van items tellen).
(iv) Multiplicativiteit eist onafhankelijkheid en
verschillende bronnen van toeval: $G_{2X}(t) = G_X(t^2)$,
niet $G_X(t)^2$. (v) Differentiëren in $1$ is een
randoperatie: wanneer de straal precies $1$ is, zoals in
\cref{ex:b2:genfun:heavytail}, kan $G'(1^-)$ oneindig zijn,
en de monotone-limietformulering van de momentenstelling is
geen pedante fijnheid maar de eerlijke uitspraak.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b2:genfun:1}
Bereken de \omterm{ex:b2:powerseries:fibonacci}{genererende functie} van de uniforme wet op $\{1,
2, \dots, 6\}$ (een eerlijke dobbelsteen). Toon dat de som
van twee eerlijke dobbelstenen \emph{niet} uniform op $\{2,
\dots, 12\}$ kan zijn: factoriseer $G_{X+Y}$ en tel
wortels. \emph{(Een uniforme som zou $G_X(t)G_Y(t) =
\frac{t^2}{11}\sum_{k=0}^{10}t^k$ forceren, waarvan de
niet-nul-wortels de $11$-de eenheidswortels anders dan $1$
zijn --- geen van hen reëel --- terwijl $G_X/t$ en $G_Y/t$
reële polynomen van graad $5$ zijn, elk minstens één reële
wortel bezittend.)}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:genfun:2}
Herstel met \omterm{ex:b2:powerseries:fibonacci}{genererende functies} $\E$ en $V$ voor de
binomiale en geometrische wetten
(\cref{thm:b2:genfun:moments}).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:genfun:3}
Twee valse dobbelstenen: is het mogelijk twee dobbelstenen
te beladen (\omterm{def:b2:proba:independence}{onafhankelijk}, identiek of niet) zodat hun som
uniform is op $\{2, \dots, 12\}$? \emph{(Zelfde
factorisatie-obstructie als in \cref{exo:b2:genfun:1}: het
antwoord is nee zelfs met verschillende beladingen, omdat
elke factor $G_X(t)/t$ oneven graad $5$ heeft, dus een reële
wortel, terwijl het doel er geen heeft.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:genfun:4}
Zij $X_1, X_2, \dots$ onafhankelijke Bernoulli
$\mathcal{B}(p)$ en $N \sim \mathcal{P}(\lambda)$
\omterm{def:b2:proba:independence}{onafhankelijk} van hen. Toon, via
\cref{thm:b2:genfun:compound}, dat $S = X_1 + \dots + X_N
\sim \mathcal{P}(\lambda p)$: een Poisson-aantal items, elk
gehouden met waarschijnlijkheid $p$, laat een Poisson-aantal
over --- \emph{verdunning}. Bereken ook de wet van de
verworpen telling en bewonder: zij is
$\mathcal{P}(\lambda(1-p))$, en men kan tonen dat zij
\omterm{def:b2:proba:independence}{onafhankelijk} is van $S$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:genfun:5}
(Negatief binomiaal) Laat $T_r$ het aantal worpen zijn om
$r$ kop te verkrijgen (kopkans $p$). Schrijf $T_r$ als som
van $r$ onafhankelijke geometrische variabelen, leid af
\[
G_{T_r}(t) = \Bigl(\frac{pt}{1 - (1-p)t}\Bigr)^{r},
\qquad
\E(T_r) = \frac rp,
\qquad
V(T_r) = \frac{r(1-p)}{p^2},
\]
en ontwikkel $G_{T_r}$ om $\P(T_r = n) =
\binom{n-1}{r-1}p^r(1-p)^{n-r}$ te vinden.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:genfun:6}
Voor de nakomelingenwet $p_0 = \frac18$, $p_1 = \frac38$,
$p_2 = \frac38$, $p_3 = \frac18$: bereken $m$, beslis
superkriticiteit, en bereken de uitsterfkans $q$ exact.
\emph{(Factoriseer de wortel $t = 1$ van $G(t) - t$ uit.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:genfun:7}
(Totale nakomelingschap) In een subkritisch
\omterm{pb:b2:genfun:1}{vertakkingsproces} ($m < 1$), laat $Y = \sum_{n\geq0} Z_n$
het totale aantal ooit geboren individuen zijn. Toon $\E(Y)
= \sum_n m^n = \frac{1}{1 - m}$ (rechtvaardig de
verwisseling van sommaties), en bewijs dat de \omterm{ex:b2:powerseries:fibonacci}{genererende
functie} $H = G_Y$ voldoet aan de functionaalvergelijking
$H(t) = t\,G(H(t))$. \emph{(De voorouder, plus de totale
nakomelingschappen van elk van zijn kinderen, die
onafhankelijke kopieën van $Y$ zijn.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:genfun:8}
Zij $X$ \omterm{ex:b2:powerseries:fibonacci}{genererende functie} $G$ hebben met
\omterm{def:b2:powerseries:radius}{convergentiestraal} $> 1$. Bewijs de \emph{exponentiële
staartgrens}: er bestaan $C > 0$ en $\rho \in \intoo{0}{1}$
met $\P(X \geq n) \leq C\rho^n$. \emph{(Markov toegepast op
$t^X$ voor een vaste $t > 1$ binnen de schijf.)} Omgekeerd,
toon dat als $\P(X \geq n) \leq C\rho^n$ met $\rho < 1$, de
straal van $G$ $\geq 1/\rho > 1$ is.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:genfun:9}
(Continuïteitsstelling, elementair geval) Zij $X, X_1, X_2,
\dots$ $\N$-waardig met $G_{X_n}(t) \to G_X(t)$ voor elke $t
\in \intco{0}{1}$. Toon dat $\P(X_n = k) \to \P(X = k)$ voor
elke $k$. \emph{(Inductie op $k$: voor $k = 0$ neem $t \to 0$
--- zorgvuldig: fixeer $t$ klein, gebruik $\abs{\P(X_n = 0)
- G_{X_n}(t)} \leq \frac{t}{1-t}$, geldig omdat de staart
$\sum_{j \geq 1}p_jt^j \leq \frac{t}{1 - t}$; diagonaliseer
dan. Voor de inductiestap, beschouw $\frac{G(t) - \P(X =
0)}{t}$, de \omterm{ex:b2:powerseries:fibonacci}{genererende functie} van een verschoven wet.)}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:genfun:10}
(Pariteitstruc) Toon dat voor een $\N$-waardige variabele
$X$,
\[
\P(X \text{ even}) = \frac{1 + G_X(-1)}{2} ,
\]
en bereken deze waarschijnlijkheid voor $X \sim \mathcal
P(\lambda)$ en $X \sim \mathcal B(n, p)$. Wat betekent
$G_X(-1) \to 0$ probabilistisch?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:genfun:11}
(Sicherman-dobbelstenen) Verifieer de factorisatie van de
\omterm{ex:b2:powerseries:fibonacci}{genererende functie} van de eerlijke dobbelsteen
\[
\frac{t + t^2 + \dots + t^6}{6}
= \frac{t\,(1 + t)(1 + t + t^2)(1 - t + t^2)}{6},
\]
en toon dat de twee dobbelstenen met vlakken $\{1, 2, 2, 3,
3, 4\}$ en $\{1, 3, 4, 5, 6, 8\}$ \omterm{ex:b2:powerseries:fibonacci}{genererende functies}
$\frac{t(1+t)(1+t+t^2)}6$ en
$\frac{t(1+t)(1+t+t^2)(1-t+t^2)^2}6$ hebben, waarvan het
product dat van twee standaarddobbelstenen is: deze
exotische dobbelstenen produceren elk totaal $2, \dots, 12$
met precies de standaardkansen.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:genfun:12}
(Wachten op twee kop op een rij) Een munt met kopkans $p$
wordt gegooid tot twee opeenvolgende koppen verschijnen;
laat $T$ het aantal worpen zijn (het spel van
\cref{exo:b2:proba:6}). Conditionerend op de eerste worpen,
leid een lineair stelsel af voor de \omterm{ex:b2:powerseries:fibonacci}{genererende functies}
vanuit de toestanden ``geen huidige kop'' en ``één huidige
kop'', en concludeer
\[
G_T(t) = \frac{p^2t^2}{1 - qt - pqt^2}
\qquad (q = 1 - p);
\]
controleer $G_T(1) = 1$ en $\E(T) = \dfrac{1 + p}{p^2}$ ($=
6$ voor een eerlijke munt).
\end{exercise}

\section{Probleem: het Galton--Watson-proces, opgelost}

\begin{problem}[{Weekendprobleem --- groeitempi, exacte
oplossingen, totale nakomelingschap, en de kritieke schatting
van Kolmogorov}]\label{pb:b2:genfun:1}
Het uitstervingscriterium (\cref{thm:b2:genfun:extinction})
splitst vertakkingsprocessen\index{vertakkingsproces} in
subkritisch, kritiek en superkritisch --- maar zegt niets
over \emph{tempi}: hoe snel een gedoemde lijn sterft, hoe
groot een overlevende groeit. Dit probleem berekent ze. We
houden de notatie van het hoofdstuk: nakomelingenwet
$(p_k)$ met pgf $G$, gemiddelde $m = G'(1)$,
generatiegroottes $Z_n$ ($Z_0 = 1$), iteraten $G_n =
G_{Z_n}$, uitsterfkansen $q_n = \P(Z_n = 0) \uparrow q$; we
nemen altijd aan $p_1 \neq 1$ en, waar tweede momenten
verschijnen, $G''(1) < \infty$, en we schrijven $\sigma^2 =
V(Z_1)$.

\medskip
\noindent\textbf{Deel I --- Momenten van de generaties.}
\begin{enumerate}
  \item Toon $\E(Z_n) = m^n$ \emph{(kettingregel op $G_n = G
        \circ G_{n-1}$ in $1^-$, met $G_{n-1}(1) = 1$ en
        \cref{thm:b2:genfun:moments})}.
  \item Stel de recursie $G_n''(1) =
        G''(1)\,m^{2(n-1)} + m\,G_{n-1}''(1)$ op en los haar
        op: $G_n''(1) = G''(1)\,m^{n-1}\dfrac{m^n - 1}{m -
        1}$ voor $m \neq 1$, en $G_n''(1) = n\,G''(1)$ voor
        $m = 1$.
  \item Leid af
        \[
        V(Z_n) = \sigma^2m^{n-1}\,\frac{m^n - 1}{m - 1}
        \quad (m \neq 1),
        \qquad
        V(Z_n) = n\,\sigma^2 \quad (m = 1).
        \]
  \item (Subkritisch tempo, bovengrens) Voor $m < 1$, toon
        $\P(Z_n > 0) \leq m^n$ \emph{(Markov op de
        geheelwaardige $Z_n$)}: uitsterven is zeker met een
        geometrisch tempo --- een kwantitatieve verfijning
        van het criterium van het hoofdstuk.
  \item (Subkritisch tempo, ondergrens) Gebruik makend van
        Cauchy--Schwarz op $Z_n\mathbf 1_{Z_n > 0}$, toon
        \[
        \P(Z_n > 0) \geq \frac{\E(Z_n)^2}{\E(Z_n^2)}
        \geq c\,m^{n}
        \quad\text{met}\quad
        c = \Bigl(\frac{\sigma^2}{m(1-m)} + 1\Bigr)^{-1} :
        \]
        het geometrische tempo $m^n$ is exact tot op
        constanten.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De geometrische familie,
exact opgelost.} Laat de nakomelingenwet geometrisch op $\N$
zijn: $p_k = qp^k$ ($k \geq 0$), met $0 < p < 1$, $q = 1 -
p$.
\begin{enumerate}[resume]
  \item Bereken $G(t) = \dfrac{q}{1 - pt}$ en $m = \dfrac
        pq$; lokaliseer de drie regimes in termen van $p$.
  \item Los $G(t) = t$ op: toon dat de vastpunten $1$ en
        $q/p = 1/m$ zijn, en herstel de uitsterfkans
        $q_{\mathrm{ext}} = \min(1, 1/m)$.
  \item Bewijs door inductie de gesloten vormen
        \[
        q_n = \frac{m^n - 1}{m^{n+1} - 1} \quad (m \neq 1),
        \qquad
        q_n = \frac{n}{n+1} \quad (m = 1).
        \]
  \item Leid de exacte tempi af: $1 - q_n \sim (1 -
        m)\,m^n$ in het subkritische geval, en
        $q_{\mathrm{ext}} - q_n \sim \dfrac{m -
        1}{m^{2}}\cdot m^{-n}$ in het superkritische geval;
        controleer dat de superkritische contractieverhouding
        $G'(q_{\mathrm{ext}}) = 1/m$ is.
  \item Kritiek geval ($p = \tfrac12$): bereken $\sigma^2 =
        2$ en noteer $1 - q_n = \frac1{n+1}$: overleving
        daalt als $\frac1n$ --- noch geometrisch noch
        \omterm{def:b2:series:summable}{sommeerbaar}.
  \item Nog steeds kritiek: bewijs door inductie de volle
        iterate
        \[
        G_n(t) = \frac{n - (n-1)t}{n + 1 - nt},
        \]
        en leid af dat geconditioneerd op overleving, $Z_n$
        geometrisch is op $\N^*$ met parameter $\frac1{n+1}$:
        \[
        \P(Z_n = k \mid Z_n > 0) = \frac1{n+1}
        \Bigl(\frac{n}{n+1}\Bigr)^{k-1},
        \qquad
        \E(Z_n \mid Z_n > 0) = n + 1 .
        \]
        De gemiddelde lijn sterft, maar de overlevende
        lijnen hebben grootte van orde $n$.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Totale nakomelingschap.} Laat
$Y = \sum_{n\geq0}Z_n \in \N^* \cup \{\infty\}$ het totale
aantal ooit geboren individuen zijn, en $H(t) =
\sum_{k\geq1}\P(Y = k)t^k$.
\begin{enumerate}[resume]
  \item Rechtvaardig $\P(Y < \infty) = q_{\mathrm{ext}}$, en
        herinner uit \cref{exo:b2:genfun:7} de
        functionaalvergelijking $H(t) = t\,G(H(t))$ (wier
        afleiding $m < 1$ niet gebruikte).
  \item (Binaire vertakking) Voor $p_0 = p_2 = \frac12$
        (kritiek), los de functionaalvergelijking op:
        \[
        H(t) = \frac{1 - \sqrt{1 - t^2}}{t},
        \]
        en ontwikkel met \cref{ex:b2:powerseries:catalan}
        om te krijgen
        \[
        \P(Y = 2k + 1) = \frac{C_k}{2^{2k+1}},
        \qquad C_k = \frac1{k+1}\binom{2k}k ;
        \]
        controleer de waarden $\P(Y = 1) = \frac12$ en
        $\P(Y = 3) = \frac18$ door direct tellen.
  \item Differentiëren van de functionaalvergelijking in
        $1^-$ toont dat $\E(Y) = \frac{1}{1-m}$ voor $m <
        1$, terwijl kriticiteit $\E(Y) = \infty$ forceert:
        de kritieke totale nakomelingschap is bijna zeker
        eindig met oneindig gemiddelde.
  \item Met de centrale binomiale asymptotiek
        (\cref{ex:b2:comparison:centralbinomial}), toon
        \[
        \P(Y = 2k+1) \sim \frac{1}{2\sqrt\pi\,k^{3/2}},
        \]
        een zware $k^{-3/2}$-staart, en leid af $\P(Y > n)
        \asymp n^{-1/2}$ (boven- en ondergrenzen van deze
        orde volstaan).
  \item Vergelijk met de eerlijke \omterm{pb:b2:proba:1}{toevalswandeling} (het
        weekendprobleem van \cref{ch:b2:proba}): zekere maar
        oneindig-gemiddelde terugkeertijden daar, zekere
        maar oneindig-gemiddelde totale nakomelingschap
        hier, beide met $n^{-3/2}$-lokale wetten. Eén alinea
        over waarom kriticiteit deze signatuur produceert.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De schatting van Kolmogorov in
kriticiteit.} Neem aan $m = 1$, $0 < \sigma^2 = G''(1) <
\infty$.
\begin{enumerate}[resume]
  \item Toon dat $G''$ continu uitbreidt tot $\intcc01$
        \emph{(niet-negatief stijgend met eindige limiet)}
        en leid de Taylor-ontwikkeling in $1$ af:
        \[
        G(t) = t + b\,(1-t)^2 + o\bigl((1-t)^2\bigr),
        \qquad b = \frac{G''(1)}2 = \frac{\sigma^2}2 .
        \]
  \item Voor $t \in \intco01$, stel $h(t) = \dfrac1{1 -
        G(t)} - \dfrac1{1 - t}$. Toon
        \[
        h(t) = \frac{G(t) - t}{(1 - G(t))(1 - t)}
        \xrightarrow[t\to1^-]{} b .
        \]
  \item Telescoop langs de iteratie $q_{j+1} = G(q_j)$:
        \[
        \frac1{1 - q_n} = 1 + \sum_{j=0}^{n-1}h(q_j),
        \]
        en besluit met een Cesaro-argument dat
        \[
        \P(Z_n > 0) = 1 - q_n \sim \frac{2}{\sigma^2\,n}
        \]
        --- \emph{de schatting van Kolmogorov}: elk kritiek
        \omterm{pb:b2:genfun:1}{vertakkingsproces} sterft met het universele tempo
        $1/n$, met alleen de constante die de
        nakomelingenwet onthoudt.
  \item Controleer de schatting tegen het kritieke
        geometrische geval van vraag 10.
  \item Leid af $\E(Z_n \mid Z_n > 0) = \dfrac{1}{1 - q_n}
        \sim \dfrac{\sigma^2 n}{2}$ \emph{(noteer $\E(Z_n
        \mathbf 1_{Z_n>0}) = \E(Z_n) = 1$)}, en controleer
        tegen vraag 11: geconditioneerd op overleving
        groeit de populatie \emph{lineair} --- het kritieke
        slappe koord tussen dood en explosie.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Toepassingen en synthese.}
\begin{enumerate}[resume]
  \item (Epidemieën, kettingreacties) Voor een Poisson-
        nakomelingenwet $\mathcal P(\lambda)$ --- elk
        geval infecteert $\mathcal P(\lambda)$ nieuwe
        gevallen --- schrijf de uitstervingsvergelijking $q
        = \eu^{\lambda(q-1)}$ en los haar numeriek op voor
        $\lambda = 1.5$ ($q \approx 0.417$) en $\lambda =
        2$ ($q \approx 0.203$): startend van één geval is
        een grote uitbraak \emph{niet} zeker zelfs wanneer
        $\lambda > 1$. Leg uit waarom de iteratie $q_{n+1}
        = \eu^{\lambda(q_n - 1)}$ vanaf $q_0 = 0$ naar de
        juiste wortel convergeert.
  \item Startend van $k$ voorouders in plaats van één, toon
        dat de uitsterfkans $q^k$ is. Toepassing: met
        $\lambda = 1.5$, hoeveel begingevallen maken een
        uitbraak minstens $99\%$ waarschijnlijk?
  \item (Een superkritisch proces conditioneren op
        uitsterven) Voor $m > 1$ met uitsterfkans $q \in
        \intoo01$: bewijs eerst door convexiteit dat $G'(q)
        < 1$ in het kleinste vast punt, en leid af
        $q_{\mathrm{ext}} - q_n = O\bigl(G'(q)^n\bigr)$
        (geometrische convergentie, zoals vraag 9
        illustreerde). Toon dan dat $\widehat G(t) =
        G(qt)/q$ de pgf is van een bona fide
        nakomelingenwet, met gemiddelde $\widehat m =
        G'(q) < 1$: een subkritisch metgezelproces.
        Verifieer op de geometrische familie: conditioneren
        van het superkritische $(p, q)$-proces op
        uitsterven verwisselt $p$ en $q$. (De volle
        uitspraak --- het geconditioneerde proces \emph{is}
        het metgezelproces --- wordt bewezen in het
        Bachelor jaar~3-volume; hier hebt u haar
        genererende-functieschaduw geverifieerd.)
  \item Synthese: stel de trichotomietabel op --- voor $m <
        1$, $m = 1$, $m > 1$: waarde van $q$; tempo van
        $\P(Z_n > 0)$ of van $q - q_n$; $\E(Y)$; grootte
        van een overlevende generatie. Stel in één zin per
        gereedschap hoe compositie van pgf's, convexiteit,
        Taylor in $1^-$, en Cesaro-middeling het hele
        probleem droegen, en wat het Bachelor
        jaar~3-volume toevoegt (de martingaal $Z_n/m^n$ en
        de exponentiële limietwet van Yaglom).
\end{enumerate}
\end{problem}
