\chapter{Differentiaalrekening}\label{ch:b2:diffcalc}

De calculus in twee variabelen van het volume Jaar~1 rijpt hier tot de
differentiaalrekening van afbeeldingen tussen genormeerde ruimten: de
\emph{\omterm{def:b2:diffcalc:differential}{differentiaal}} als beste lineaire benadering, de kettingregel in
volle algemeenheid, de symmetriestelling van Schwarz \emph{bewezen},
Taylorformules, en de volledige tweede-orde analyse van extrema. De
stelling van de inverse functie, kroon van de theorie, wordt
geformuleerd met haar bewijsstrategie --- een Banach-vastpunt ---
expliciet gemaakt.

Overal is $U$ een open deelverzameling van $\R^n$ (of van een genormeerde
ruimte; het eindigdimensionale geval draagt alle ideeën), $f \colon U \to
\R^m$.

\section{De differentiaal}

\begin{definition}\label{def:b2:diffcalc:differential}
$f$ is \emph{differentieerbaar}\index{differentieerbaar} in $a$ wanneer er
een (continue) lineaire afbeelding $\dd f_a \colon \R^n \to \R^m$ bestaat
met
\[
f(a + h) = f(a) + \dd f_a(h) + o(\norm h)
\qquad (h \to 0).
\]
De afbeelding $\dd f_a$, de \emph{differentiaal}\index{differentiaal} van
$f$ in $a$, is uniek; haar matrix in de canonieke basissen is de
\emph{Jacobimatrix}\index{Jacobimatrix} $J_f(a) =
\bigl(\frac{\partial f_i}{\partial x_j}(a)\bigr)$.
Differentieerbaarheid impliceert continuïteit en het bestaan van alle
richtingsafgeleiden $\dd f_a(v) = \lim_{t\to0}\frac{f(a + tv)
- f(a)}{t}$; het omgekeerde faalt
(\cref{exo:b2:diffcalc:2}). Voor $m = 1$ is $\dd f_a(h) = \langle
\nabla f(a), h\rangle$: de gradiënt van Jaar~1, nu begrepen als de
vector die de differentiaal voorstelt.
\end{definition}

\begin{theorem}[$C^1$-criterium]\label{thm:b2:diffcalc:c1}
Als alle partiële afgeleiden van $f$ op $U$ bestaan en continu zijn in
$a$, dan is $f$ \omterm{def:b2:diffcalc:differential}{differentieerbaar} in $a$. ``$C^1$ op $U$'' --- continue
partiële afgeleiden --- impliceert dus differentieerbaarheid overal, met
continue \omterm{def:b2:diffcalc:differential}{differentiaal}.
\end{theorem}

\begin{proof}
Componentgewijs ($m = 1$ volstaat). Het bewijs van Jaar~1 voor twee
variabelen --- één coördinaat tegelijk verplaatsen, de
éénvariabele middelwaardestelling op elk been toepassen, en de
continuïteit van de partiële afgeleiden in $a$ gebruiken --- generaliseert
woordelijk naar $n$ benen:
\[
f(a + h) - f(a) = \sum_{j=1}^{n} \bigl(f(a + h^{(j)}) - f(a +
h^{(j-1)})\bigr)
= \sum_j h_j\,\frac{\partial f}{\partial x_j}(\xi_j),
\]
waar $h^{(j)}$ de eerste $j$ coördinaten van $h$ bevriest en $\xi_j$
op het $j$-de been ligt; continuïteit maakt elke
$\frac{\partial f}{\partial x_j}(\xi_j)$ tot $\frac{\partial
f}{\partial x_j}(a) + o(1)$, en de fout is $o(\norm h)$.
\end{proof}

\begin{theorem}[Kettingregel]\label{thm:b2:diffcalc:chainrule}
Als $f$ \omterm{def:b2:diffcalc:differential}{differentieerbaar} is in $a$ en $g$ in $f(a)$, dan is $g \circ f$
\omterm{def:b2:diffcalc:differential}{differentieerbaar} in $a$ met
\[
\dd(g \circ f)_a = \dd g_{f(a)} \circ \dd f_a ,
\qquad
J_{g\circ f}(a) = J_g\bigl(f(a)\bigr)\,J_f(a) :
\]
Jacobimatrices vermenigvuldigen.\index{kettingregel}
\end{theorem}

\begin{proof}
Schrijf $f(a + h) = f(a) + \dd f_a(h) + \norm h\,\varepsilon_1(h)$
en $g(b + k) = g(b) + \dd g_b(k) + \norm k\,\varepsilon_2(k)$ met
$b = f(a)$, $k = k(h) = \dd f_a(h) + \norm h \varepsilon_1(h)$.
Substitueren levert
\[
g(f(a+h)) = g(b) + \dd g_b\bigl(\dd f_a(h)\bigr)
+ \norm h\,\dd g_b(\varepsilon_1(h)) + \norm{k}\,\varepsilon_2(k),
\]
en beide fouttermen zijn $o(\norm h)$: de eerste omdat $\dd g_b$
continu is en $\varepsilon_1 \to 0$; de tweede omdat $\norm k
\leq C\norm h$ (begrensde lineaire afbeelding plus kleine term) en
$\varepsilon_2(k) \to 0$ als $h \to 0$.
\end{proof}

\begin{example}[Radiale functies, eens en voor altijd]\label{ex:b2:diffcalc:radial}
Zij $r(x) = \norm x_2$ op $\R^n\setminus\{0\}$ en $f = g \circ
r$ met $g$ een $C^1$-functie van één variabele. Ten eerste is $r$
\omterm{def:b2:diffcalc:differential}{differentieerbaar} buiten $0$: uit $r^2 = \sum x_i^2$ volgt
\[
\frac{\partial r}{\partial x_i} = \frac{x_i}{r},
\qquad\text{d.w.z.}\qquad
\nabla r(x) = \frac{x}{\norm x} ,
\]
de eenheidsradiale vector (differentieer $r^2$ en deel --- of
pas de kettingregel toe op $\sqrt{\cdot}$). De kettingregel geeft
dan, voor elke radiale functie,
\[
\nabla f(x) = g'\bigl(\norm x\bigr)\,\frac{x}{\norm x} .
\]
Uitgewerkt voorbeeld: $g(r) = \frac1r$ levert $\nabla\frac{1}{\norm
x} = -\frac{x}{\norm x^3}$, het omgekeerd-kwadratische veld van
gravitatie en elektrostatica --- richting radiaal, grootte
$\frac{1}{\norm x^2}$. Slotinzicht: gradiënten van radiale
functies zijn radiaal omdat de niveauverzamelingen bollen zijn en
de gradiënt orthogonaal is op niveauverzamelingen; in $x = 0$ is
$r$ daarentegen \emph{niet} \omterm{def:b2:diffcalc:differential}{differentieerbaar} (geen kandidaat
lineaire afbeelding past bij $\norm h$ vanuit alle richtingen) ---
gladde radiale profielen hebben $g'(0) = 0$ nodig om de oorsprong
netjes te kruisen.
\end{example}

\begin{theorem}[Middelwaardeongelijkheid]\label{thm:b2:diffcalc:mvi}
Zij $f$ \omterm{def:b2:diffcalc:differential}{differentieerbaar} op $U$ en het segment $\intcc{a}{b}
= \{a + t(b-a)\}$ bevat in $U$. Dan is
\[
\norm{f(b) - f(a)} \leq \norm{b - a}\,
\sup_{x \in \intcc{a}{b}} \vertiii{\dd f_x} .
\]
In het bijzonder is een differentieerbare afbeelding met
nuldifferentiaal op een \emph{samenhangende} open verzameling
constant.
\end{theorem}

\begin{proof}
De functie $\varphi(t) = f(a + t(b-a))$ is \omterm{def:b2:diffcalc:differential}{differentieerbaar} op
$\intcc{0}{1}$ met $\varphi'(t) = \dd f_{a + t(b-a)}(b - a)$
(kettingregel), van \omterm{def:b2:nvs:norm}{norm} $\leq M\norm{b-a}$ met $M$ de
weergegeven sup. Voor $\R$-waardige $f$ besluit de
éénvariabele middelwaardeongelijkheid; voor vectorwaarden pas
haar toe op $t \mapsto \langle u, \varphi(t)\rangle$ met $u$ de
eenheidsvector langs $f(b) - f(a)$.
Constantheid: lokaal constant (segmenten in ballen) plus
samenhang (de verzameling waar $f$ een gegeven waarde aanneemt is
open en gesloten: \cref{ch:b2:metric}).
\end{proof}

\begin{example}[Een Lipschitzconstante uit de MVI]\label{ex:b2:diffcalc:lipschitz}
Is $f(x, y) = \sin x\,\sin y$ \omterm{def:b2:metric:continuity}{Lipschitz} op $\R^2$, en met
welke constante? Haar gradiënt is $\nabla f = (\cos x\sin y,\
\sin x\cos y)$, van gekwadrateerde \omterm{def:b2:nvs:norm}{norm}
\[
\cos^2x\sin^2y + \sin^2x\cos^2y
\leq \sin^2 y + \cos^2y\cdot 1 = 1
\]
(bind $\cos^2x$ en $\sin^2x$ afzonderlijk door $1$), dus
$\vertiii{\dd f_{(x,y)}} = \norm{\nabla f} \leq 1$ overal,
en \cref{thm:b2:diffcalc:mvi} op het segment tussen twee
willekeurige punten geeft
\[
\abs{f(b) - f(a)} \leq \norm{b - a}_2 :
\]
$f$ is $1$-Lipschitz, en de constante is scherp (nabij de
oorsprong heeft $f(x, \tfrac\pi2) = \sin x$ helling $1$).
Slotinzicht: de middelwaardeongelijkheid zet een puntsgewijze
grens op de \omterm{def:b2:diffcalc:differential}{differentiaal} om in een globale continuïteitsmodulus
--- de standaardroute naar Lipschitzschattingen in elke dimensie,
en de motor in \cref{exo:b2:diffcalc:12}.
\end{example}

\section{Tweede afgeleiden}

\begin{theorem}[Schwarz]\label{thm:b2:diffcalc:schwarz}
Als $f$ van klasse $C^2$ is op $U$ (alle tweede partiële afgeleiden
bestaan en zijn continu), dan geldt voor alle $i, j$:\index{stelling van Schwarz}
\[
\frac{\partial^2 f}{\partial x_i\,\partial x_j}
= \frac{\partial^2 f}{\partial x_j\,\partial x_i} .
\]
\end{theorem}

\begin{proof}
Twee variabelen volstaan ($x = x_i$, $y = x_j$, de rest bevroren).
Beschouw het tweede verschil
\[
\Delta(h) = f(a + h, b + h) - f(a + h, b) - f(a, b + h) + f(a,b) .
\]
Fixeer $h$ en stel $\varphi(x) = f(x, b+h) - f(x, b)$: dan is $\Delta(h)
= \varphi(a + h) - \varphi(a)$, en twee toepassingen van de
middelwaardestelling geven
\[
\Delta(h) = h\,\varphi'(\xi)
= h\Bigl(\frac{\partial f}{\partial x}(\xi, b+h) -
\frac{\partial f}{\partial x}(\xi, b)\Bigr)
= h^2\,\frac{\partial^2 f}{\partial y\,\partial x}(\xi, \eta),
\]
met $\xi \in \intoo{a}{a+h}$, $\eta \in \intoo{b}{b+h}$. Door
continuïteit geldt $\frac{\Delta(h)}{h^2} \to \frac{\partial^2 f}{\partial
y\partial x}(a, b)$ als $h \to 0$. Dezelfde berekening met de
rollen van de variabelen omgewisseld (eerst de tweede variabele
bevriezen) geeft $\frac{\Delta(h)}{h^2} \to \frac{\partial^2 f}{\partial x
\partial y}(a,b)$: de twee limieten van dezelfde grootheid vallen
samen.
\end{proof}

\begin{example}[Waarom $C^2$ nodig is: het tegenvoorbeeld van Peano]\label{ex:b2:diffcalc:peano}
Zij $f(x, y) = \dfrac{xy(x^2 - y^2)}{x^2 + y^2}$, $f(0,0) = 0$.
Buiten de oorsprong is $f$ van klasse $C^\infty$; in de oorsprong
bestaan alle eerste en tweede partiële afgeleiden, maar de gemengde
zijn het oneens. Bereken langs de assen: $f(x, 0) = f(0, y) = 0$, en
voor $y \neq 0$,
\[
\frac{\partial f}{\partial x}(0, y)
= \lim_{x\to0}\frac{f(x,y)}{x}
= \frac{y(0 - y^2)}{y^2} = -y ,
\qquad\text{symmetrisch}\qquad
\frac{\partial f}{\partial y}(x, 0) = x .
\]
Dus
\[
\frac{\partial^2 f}{\partial y\,\partial x}(0,0)
= \frac{\dd}{\dd y}\Bigl[\frac{\partial f}{\partial
x}(0,y)\Bigr]_{y=0} = -1,
\qquad
\frac{\partial^2 f}{\partial x\,\partial y}(0,0) = +1 :
\]
de twee gemengde partiële afgeleiden bestaan en verschillen. Geen
tegenspraak met \cref{thm:b2:diffcalc:schwarz}: de tweede partiële
afgeleiden van $f$ zijn niet continu in $0$ (test langs $y = tx$).
Slotinzicht: de stelling van Schwarz is een echte stelling over
\emph{continuïteit}, geen formele identiteit --- en de hypothese
``$C^2$'' in Taylor--Young hieronder doet echt werk.
\end{example}

\begin{theorem}[Taylor--Young van orde 2]\label{thm:b2:diffcalc:taylor}
Zij $f \colon U \to \R$ van klasse $C^2$ en $a \in U$. Dan, als $h \to 0$,
\[
f(a + h) = f(a) + \langle\nabla f(a), h\rangle
+ \frac12\, \langle H_a h,\, h\rangle + o\bigl(\norm h^2\bigr),
\]
waar $H_a = \bigl(\frac{\partial^2 f}{\partial x_i\partial
x_j}(a)\bigr)$ de (symmetrische, door Schwarz) \emph{Hessematrice}\index{Hessematrice}
is.
\end{theorem}

\begin{proof}
Pas de éénvariabele stelling van Taylor--Young (volume Jaar~1) toe op
$\varphi(t) = f(a + th)$ op $\intcc{0}{1}$: door de kettingregel is
$\varphi'(t) = \langle \nabla f(a + th), h\rangle$ en $\varphi''(t)
= \langle H_{a+th}h, h\rangle$, beide continu in $t$. Dan is
$\varphi(1) = \varphi(0) + \varphi'(0) + \frac12\varphi''(\theta)$
(Taylor--Lagrange) met $\theta \in \intoo{0}{1}$, en de continuïteit
van de tweede partiële afgeleiden maakt $\varphi''(\theta) =
\varphi''(0) + o(1)\cdot\norm h^2$ uniform: de weergegeven
ontwikkeling.
\end{proof}

\begin{example}[Een ontwikkeling op twee manieren]\label{ex:b2:diffcalc:taylortwoways}
Ontwikkel $f(x, y) = \eu^x\cos y$ in de oorsprong tot orde $2$.
\emph{Via samenstelling van éénvariabele ontwikkelingen:}
\[
\eu^x\cos y = \Bigl(1 + x + \frac{x^2}{2} +
o(x^2)\Bigr)\Bigl(1 - \frac{y^2}{2} + o(y^2)\Bigr)
= 1 + x + \frac{x^2 - y^2}{2} + o\bigl(\norm{(x,y)}^2\bigr) .
\]
\emph{Via partiële afgeleiden:} $f_x = \eu^x\cos y$, $f_y =
-\eu^x\sin y$, dus $\nabla f(0) = (1, 0)$; en $f_{xx} = f$,
$f_{yy} = -f$, $f_{xy} = -\eu^x\sin y$ geven $H_0 =
\operatorname{diag}(1, -1)$: \cref{thm:b2:diffcalc:taylor}
reproduceert $1 + x + \frac12(x^2 - y^2)$. De twee berekeningen
stemmen overeen, en de samenstellingsroute was sneller --- geen
tweede partiële afgeleiden nodig. Slotinzicht: de oorsprong is
\emph{geen} kritiek punt ($\nabla f \neq 0$), dus ondanks de
onbepaalde Hessematrice is er geen zadel te verklaren: de
lineaire term regelt, en de tweede-ordetest spreekt alleen in
kritieke punten.
\end{example}

\begin{theorem}[Tweede-orde extremumtest, bewezen]\label{thm:b2:diffcalc:extrema}
Zij $f$ van klasse $C^2$ nabij een kritiek punt $a$ ($\nabla f(a) = 0$),
met Hessematrice $H = H_a$.
\begin{enumerate}
  \item Als $H$ positief definiet is, is $a$ een strikt lokaal minimum
        (negatief definiet: maximum).
  \item Als $H$ \omterm{def:b2:reduction:eigen}{eigenwaarden} van beide tekens heeft, is $a$ een zadel:
        geen extremum.
  \item Als $H$ singulier is (en semidefiniet), geen conclusie.
\end{enumerate}
De ``$rt - s^2$''-test van Jaar~1 is het geval $n = 2$: $\det H = rt -
s^2$, het teken van $\operatorname{tr}$ afgelezen uit $r$.
\end{theorem}

\begin{proof}
Door de spectraalstelling (\cref{thm:b2:quadratic:spectral}) wordt de
\omterm{def:b2:quadratic:def}{kwadratische vorm} van $H$ ingeklemd tussen haar extreme \omterm{def:b2:reduction:eigen}{eigenwaarden}:
$\lambda_{\min}\norm h^2 \leq \langle Hh, h\rangle \leq
\lambda_{\max}\norm h^2$.

(1) Als $\lambda_{\min} > 0$: Taylor--Young geeft
\[
f(a + h) - f(a) \geq \frac{\lambda_{\min}}{2}\norm h^2 -
o(\norm h^2) > 0
\]
voor kleine $h \neq 0$: strikt lokaal minimum.

(2) Langs een \omterm{def:b2:reduction:eigen}{eigenvector} $v_+$ met $\lambda_+ > 0$: $f(a + tv_+) -
f(a) = \frac{\lambda_+}{2}t^2 + o(t^2) > 0$ voor kleine $t$; langs
$v_-$ met $\lambda_- < 0$ is het verschil negatief: beide tekens
komen in elke omgeving voor.

(3) $f(x,y) = x^2 + y^4$, $x^2 - y^4$, $x^2 + y^3$ delen dezelfde
singuliere semidefinite Hessematrice in $0$ met drie verschillende
gedragingen.
\end{proof}

\begin{example}[Een volledige classificatie, globaal inbegrepen]\label{ex:b2:diffcalc:quarticrun}
Classificeer alle extrema van $f(x, y) = x^4 + y^4 - 4xy$ op $\R^2$.
\emph{Kritieke punten:} $\nabla f = (4x^3 - 4y,\ 4y^3 - 4x) =
0$ geeft $y = x^3$ en $x = y^3 = x^9$, dus $x(x^8 - 1) = 0$:
de reële oplossingen zijn $(0,0)$, $(1,1)$, $(-1,-1)$.
\emph{Hessemmatrices:} $H = \begin{pmatrix} 12x^2 & -4\\ -4 &
12y^2\end{pmatrix}$. In $(\pm1, \pm1)$: $\begin{pmatrix} 12 &
-4\\ -4 & 12\end{pmatrix}$, \omterm{def:b2:reduction:eigen}{eigenwaarden} $8$ en $16$: positief
definiet, strikte lokale minima met $f = -2$. In $(0,0)$:
$\begin{pmatrix} 0 & -4\\ -4 & 0\end{pmatrix}$, \omterm{def:b2:reduction:eigen}{eigenwaarden}
$\pm4$: een zadel. \emph{Globaliteit:} uit $2\abs{xy} \leq x^2 +
y^2$,
\[
f(x, y) \geq x^4 + y^4 - 2(x^2 + y^2)
= (x^2 - 1)^2 + (y^2 - 1)^2 + x^2 + y^2 - 2
\xrightarrow[\norm{(x,y)}\to\infty]{} +\infty :
\]
$f$ is coërcief, dus bereikt een globaal minimum
(compactheid van subniveauverzamelingen), noodzakelijk in een
kritiek punt: de waarde $-2$, in zowel $(1,1)$ als $(-1,-1)$, is
het globale minimum; er is geen maximum ($f$ onbegrensd naar
boven). Slotinzicht: de lokale test classificeert kandidaten, maar
alleen een groeianalyse maakt ``lokaal'' tot ``globaal'' --- het
tweestaps-patroon van elk optimaliseringsbewijs in dit boek.
\end{example}

\begin{method}[De extrema van $f \colon \R^n \to \R$ classificeren]\label{met:b2:diffcalc:extrema}
\begin{enumerate}
  \item Los $\nabla f = 0$ op (alle kritieke punten; op een domein
        met rand, behandel de rand afzonderlijk zoals in
        \cref{exo:b2:diffcalc:7}).
  \item In elk kritiek punt, bereken de Hessematrice en de tekens
        van haar \omterm{def:b2:reduction:eigen}{eigenwaarden} --- in dimensie $2$ volstaan $\det H$
        en $\operatorname{tr} H$: $\det < 0$ zadel; $\det >
        0$ extremum, van het type gegeven door het teken van het
        spoor; $\det = 0$: de test zwijgt, bestudeer $f$
        langs krommen.
  \item Voor globale uitspraken, voeg een compactheids- of
        coërciviteitsargument toe ($f \to +\infty$ op oneindig, of
        een compacte restrictieverzameling), en vergelijk dan de
        kritieke waarden.
\end{enumerate}
\end{method}

\section{De stelling van de inverse functie}

\begin{theorem}[Stelling van de inverse functie]\label{thm:b2:diffcalc:inverse}
Zij $f \colon U \to \R^n$ van klasse $C^1$ en $a \in U$ met $\dd f_a$
\emph{inverteerbaar}. Dan bestaan open omgevingen $V \ni a$, $W
\ni f(a)$ zodat $f \colon V \to W$ een bijectie is met $C^1$-inverse,
en\index{stelling van de inverse functie}
\[
\dd (f^{-1})_{f(x)} = (\dd f_x)^{-1} \qquad (x \in V).
\]
\end{theorem}
\admitted

\begin{remark}[Waarom het waar is: de vastpuntstrategie]
$f(x) = y$ oplossen nabij $a$ herschrijft zich als de vastpuntvergelijking
$x = x + \dd f_a^{-1}\bigl(y - f(x)\bigr) =: \Phi_y(x)$; de afbeelding
$\Phi_y$ heeft \omterm{def:b2:diffcalc:differential}{differentiaal} $\mathrm{id} - \dd f_a^{-1}\dd f_x$,
klein nabij $a$ door continuïteit van $\dd f$, dus $\Phi_y$ is een
contractie op een kleine gesloten bal en de vastpuntstelling van
Banach (\cref{thm:b2:metric:banach}) levert de unieke lokale
oplossing $x = f^{-1}(y)$. Continuïteit en differentieerbaarheid van
de inverse volgen dan uit de schattingen in de contractie. De volledige
boekhouding wordt in Jaar~3 uitgevoerd; de strategie --- en de
uitspraak --- worden vanaf nu vrij gebruikt. De begeleidende
\emph{stelling van de impliciete functie} ($F(x, y) = 0$ oplossen naar
$y(x)$ wanneer $\frac{\partial F}{\partial y}$ inverteerbaar is) volgt
door de stelling toe te passen op $(x, y) \mapsto (x, F(x,y))$.
\end{remark}

\begin{example}[Poolcoördinaten]\label{ex:b2:diffcalc:polar}
$\Phi(r, \theta) = (r\cos\theta, r\sin\theta)$ heeft \omterm{def:b2:diffcalc:differential}{Jacobimatrix}
\[
J_\Phi = \begin{pmatrix} \cos\theta & -r\sin\theta\\ \sin\theta &
r\cos\theta \end{pmatrix},
\qquad \det J_\Phi = r :
\]
inverteerbaar voor $r \neq 0$, dus $\Phi$ is een lokaal $C^1$-
diffeomorfisme buiten de oorsprong --- de vergunning om ``over te
gaan op poolcoördinaten'', vernieuwd voor de meervoudige integralen van
\cref{ch:b2:multint}.
\end{example}

\begin{example}[Lokaal overal, globaal nergens]\label{ex:b2:diffcalc:localnotglobal}
Zij $f(x, y) = \bigl(\eu^x\cos y,\ \eu^x\sin y\bigr)$ op
$\R^2$. Haar \omterm{def:b2:diffcalc:differential}{Jacobimatrix},
\[
J_f = \begin{pmatrix}
\eu^x\cos y & -\eu^x\sin y\\
\eu^x\sin y & \eu^x\cos y
\end{pmatrix},
\qquad
\det J_f = \eu^{2x} > 0 ,
\]
verdwijnt nooit: door \cref{thm:b2:diffcalc:inverse} is $f$ een
lokaal $C^1$-diffeomorfisme in \emph{elk} punt van het vlak.
Toch is $f$ verre van injectief: $f(x, y + 2\pi) = f(x, y)$, dus
elke waarde wordt oneindig vaak aangenomen; en ze is ook niet
surjectief, want $\norm{f(x, y)} = \eu^x > 0$ mist de oorsprong.
Slotinzicht: de stelling van de inverse functie is onherleidbaar
\emph{lokaal} --- inverteerbaarheid van elke $\dd f_a$ levert een
mozaïek van lokale inversen die niet tot één hoeven samen te
voegen. (Lezers die complexe getallen kennen herkennen $z \mapsto
\eu^z$; het mozaïek is de familie van logaritmetakken.)
Vergelijk \cref{exo:b2:diffcalc:12}, waar een kwantitatieve globale
hypothese wel één globale inverse afdwingt.
\end{example}

\begin{remark}[Veelvoorkomende valkuilen]
\emph{(i) Richtingsafgeleiden zijn goedkoop, differentialen niet:}
alle richtingsafgeleiden kunnen bestaan --- en zelfs niet lineair
van de richting afhangen --- zonder differentieerbaarheid
(\cref{exo:b2:diffcalc:2}); alleen het $C^1$-criterium
(\cref{thm:b2:diffcalc:c1}) promoveert partiële afgeleiden tot een
\omterm{def:b2:diffcalc:differential}{differentiaal}. \emph{(ii) Kritiek betekent niet extremal:}
zadels (\cref{ex:b2:diffcalc:quarticrun}) en het stille
singuliere geval (\cref{thm:b2:diffcalc:extrema}~(3)) schuilen
beide achter $\nabla f = 0$. \emph{(iii) Geen vectorwaardige
middelwaardegelijkheid:} alleen de \emph{ongelijkheid} van
\cref{thm:b2:diffcalc:mvi} overleeft
(\cref{exo:b2:diffcalc:9}); schrijf nooit $f(b) - f(a) = \dd
f_c(b-a)$ voor $f$ met waarden in $\R^m$, $m \geq 2$.
\emph{(iv) Lokale inverteerbaarheid is geen injectiviteit:}
\cref{ex:b2:diffcalc:localnotglobal}. \emph{(v) De gradiënt
hoort bij het inproduct:} $\nabla f$ is de vector die $\dd f_a$
voorstelt in een gekozen inproduct; verander het product (zoals
in het gewogen voorbeeld van het hoofdstuk over \omterm{def:b2:quadratic:def}{kwadratische
vormen}) en de gradiënt roteert, terwijl de \omterm{def:b2:diffcalc:differential}{differentiaal} ---
het intrinsieke object --- niet beweegt.
\end{remark}

\begin{remark}[Waar dit wordt gebruikt]
Alles stroomafwaarts van dit hoofdstuk is toegepaste
differentiaalrekening: het hoofdstuk over differentiaalvergelijkingen
lineariseert stromen en gebruikt de determinantformule van Liouville
(bewezen in het weekendprobleem van dit hoofdstuk); de hoofdstukken
over krommen en oppervlakken bestuderen niveauverzamelingen en
parametriseringen via de stelling van de impliciete functie;
meervoudige integralen veranderen van variabelen via Jacobimatrices.
Het weekendprobleem ontwikkelt de calculus \emph{op de ruimte van
matrices zelf} --- \omterm{def:b2:diffcalc:differential}{differentiaal} van de \omterm{def:b2:linalg:det}{determinant}, van de inverse,
de \omterm{ex:b2:nvs:matrixexp}{matrixexponentiaal}, en de orthogonale groep als gladde
niveauverzameling --- de schaduw van Jaar~2 van wat het volume Jaar~3
formaliseert als variëteiten en Liegroepen.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b2:diffcalc:1}
Bereken de Jacobimatrices van $f(x,y) = (x^2 - y^2,\, 2xy)$ en
van $\Phi(r,\theta,z) = (r\cos\theta, r\sin\theta, z)$; waar zijn
de differentialen inverteerbaar?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:diffcalc:2}
Zij $f(x,y) = \frac{x^3}{x^2 + y^2}$ ($f(0,0) = 0$). Bewijs dat alle
richtingsafgeleiden van $f$ in $0$ bestaan, maar dat $f$ niet
\omterm{def:b2:diffcalc:differential}{differentieerbaar} is in $0$ \emph{(de afbeelding $v \mapsto$
richtingsafgeleide is niet lineair)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:diffcalc:3}
Vind en classificeer de kritieke punten van $f(x, y) = x^3 + y^3 -
3xy$ met \cref{thm:b2:diffcalc:extrema}, en van $g(x,y) = x^4 +
y^4 - 2(x - y)^2$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:diffcalc:4}
Zij $f \colon \R^n \to \R$ van klasse $C^1$ en \emph{homogeen van graad
$p$}: $f(tx) = t^pf(x)$ voor $t > 0$. Bewijs de identiteit van Euler
\[
\langle \nabla f(x), x\rangle = p\,f(x) ,
\]
en haar omkering voor $C^1$-functies op $\R^n\setminus\{0\}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:diffcalc:5}
Zij $A$ \omterm{def:b2:quadratic:adjoint}{symmetrisch} en $f(x) = \frac12\langle Ax, x\rangle -
\langle b, x\rangle$. Bereken $\nabla f$ en $H_f$; wanneer is $f$
convex? Onder de aanname dat $A$ positief definiet is, toon aan dat
$f$ een uniek globaal minimum heeft in de oplossing van $Ax = b$ ---
de bestaansreden van gradiëntdaling.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:diffcalc:6}
(Lagrange-multiplicator, één nevenvoorwaarde, met de hand bewezen)
Zij $f, g$ van klasse $C^1$ op $\R^2$, en veronderstel dat $f$ in $a$
een lokaal extremum bereikt op de niveauverzameling $\{g = 0\}$, met
$\nabla g(a) \neq 0$. Bewijs dat $\nabla f(a) = \lambda\nabla g(a)$
voor zekere $\lambda$.
\emph{(Parametriseer de niveauverzameling nabij $a$ met de stelling
van de impliciete functie en differentieer $t \mapsto f(\gamma(t))$.)}
Toepassing: extrema van $f(x,y) = xy$ op de cirkel $x^2 + y^2 = 1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:diffcalc:7}
Bepaal de extrema van $f(x, y) = x^2 + y^2 - xy + x - y$ op
$\R^2$, daarna haar maximum en minimum op de gesloten driehoek met
hoekpunten $(0,0)$, $(1,0)$, $(0,1)$ \emph{(inwendige kritieke
punten, dan de drie randen, dan de hoekpunten)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:diffcalc:8}
Zij $f \colon \R^2 \to \R^2$, $f(x, y) = (x + y^2,\; y + x^2)$.
Toon aan dat $f$ een lokaal diffeomorfisme is nabij $0$, bereken
$\dd(f^{-1})_{(0,0)}$, en vind de grootste $r$ zodat $\dd f$
inverteerbaar is op de bal $\norm{(x,y)}_2 < r$ \emph{(bereken
$\det J_f$)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:diffcalc:9}
(Rolle faalt, middelwaarde overleeft) Geef $f \colon \R \to \R^2$,
van klasse $C^1$, met $f(0) = f(2\pi)$ maar $f'(t) \neq 0$ voor alle
$t$ (geen vectorwaardige Rolle). Verifieer daarna op jouw voorbeeld de
middelwaarde\emph{ongelijkheid} van \cref{thm:b2:diffcalc:mvi}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:diffcalc:10}
Bereken de \omterm{def:b2:diffcalc:differential}{differentiaal} en de gradiënt van $f(x) = \norm
x_2^2$ en van $g(x) = \langle Ax, x\rangle$ op $\R^n$ ($A$ een
vierkante matrix, niet als \omterm{def:b2:quadratic:adjoint}{symmetrisch} verondersteld), en de
Hessematrice van elk. Voor welke $A$ is $g$ convex?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:diffcalc:11}
Zij $f \colon \R^n \to \R$ van klasse $C^2$. Bewijs dat $f$ convex is
dan en slechts dan als haar Hessematrice $H_x$ positief semidefiniet is
in elke $x$ \emph{(reduceer tot één variabele: $t \mapsto f(a +
t(b-a))$; gebruik Taylor--Lagrange in één richting, en voor de
omkering evalueer $\varphi''$)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:diffcalc:12}
(Een globale inversestelling) Zij $g \colon \R^n \to \R^n$ van klasse
$C^1$ met $\vertiii{\dd g_x} \leq k < 1$ voor alle $x$, en $f =
\mathrm{id} + g$.
\begin{enumerate}
  \item Toon aan $\norm{f(x) - f(y)} \geq (1 - k)\norm{x - y}$: $f$
        is injectief, met continue inverse op haar beeld.
  \item Toon aan dat voor elke $y \in \R^n$ de afbeelding $x \mapsto y -
        g(x)$ een contractie is van de complete ruimte $\R^n$,
        en concludeer met de vastpuntstelling van Banach
        (\cref{thm:b2:metric:banach}) dat $f$ surjectief is.
  \item Concludeer dat $f$ een bijectie van $\R^n$ is met
        $(1-k)^{-1}$-Lipschitz-inverse --- een \emph{globaal}
        tegenhanger van \cref{thm:b2:diffcalc:inverse} (die
        daarentegen puur lokaal is).
\end{enumerate}
\end{exercise}

\section{Probleem: De calculus van matrices --- Jacobi,
exponentiaal, en de orthogonale groep}

\begin{problem}\label{pb:b2:diffcalc:1}
De schoonste speeltuin voor differentiaalrekening is de ruimte
$\mathcal M_n(\R) \simeq \R^{n^2}$ zelf: haar natuurlijkste afbeeldingen
--- product, inverse, \omterm{def:b2:linalg:det}{determinant}, exponentiaal --- hebben
differentialen van treffende elegantie. Dit probleem berekent ze
allemaal: de \emph{Neumannreeks}, de \omterm{def:b2:diffcalc:differential}{differentiaal} van de inverse,
de \emph{formule van Jacobi}\index{formule van Jacobi} voor de
\omterm{def:b2:linalg:det}{determinant} met de \emph{formule van Liouville} als dividend, de
\emph{matrixexponentiaal}\index{matrixexponentiaal} met
$\det\eu^A = \eu^{\operatorname{tr}A}$, en ten slotte de
orthogonale groep $O_n$ als gladde niveauverzameling met de
antisymmetrische matrices als raakruimte --- differentiaalmeetkunde
in wording. Overal is $\vertiii\cdot$ de \omterm{thm:b2:nvs:continuouslinear}{operatornorm} ondergeschikt
aan $\norm\cdot_2$, en $\langle X, Y\rangle =
\operatorname{tr}(X^{\mathsf T}Y)$ het Frobenius-inproduct.

\medskip
\noindent\textbf{Deel I --- De Neumannreeks.}
\begin{enumerate}
  \item Bewijs submultiplicativiteit, $\vertiii{AB} \leq
        \vertiii A\,\vertiii B$, en leid af dat polynoomafbeeldingen
        van $A$ (matrixproducten, \omterm{def:b2:linalg:det}{determinant}, spoor) continu zijn
        op $\mathcal M_n(\R)$.
  \item Voor $\vertiii X < 1$, toon aan dat $\sum_{k\geq0}X^k$
        absoluut convergeert in $\mathcal M_n(\R)$
        (\cref{thm:b2:nvs:absoluteconvergence}), dat haar som
        $(I - X)^{-1}$ is, en dat
        \[
        \vertiii{(I - X)^{-1}} \leq \frac{1}{1 - \vertiii X},
        \qquad
        (I - X)^{-1} = I + X + O\bigl(\vertiii X^2\bigr) .
        \]
  \item Leid af dat $GL_n(\R)$ \emph{open} is: als $A$
        inverteerbaar is en $\vertiii H <
        \frac{1}{\vertiii{A^{-1}}}$, dan is $A + H$
        inverteerbaar. Leid ook af dat $GL_n(\R)$ \emph{dicht}
        is in $\mathcal M_n(\R)$ \emph{(verstoor $A$ met
        $\varepsilon I$: $\det(A + \varepsilon I)$ is een niet-nul
        polynoom in $\varepsilon$)}.
  \item Toon aan dat de inversieafbeelding $\Phi(A) = A^{-1}$
        continu is op $GL_n(\R)$.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Eerste differentialen.}
\begin{enumerate}[resume]
  \item Toon aan dat de kwadrateringsafbeelding $A \mapsto A^2$
        \omterm{def:b2:diffcalc:differential}{differentieerbaar} is met \omterm{def:b2:diffcalc:differential}{differentiaal} $H \mapsto AH + HA$,
        en algemener dat $A \mapsto A^k$ \omterm{def:b2:diffcalc:differential}{differentiaal} $H \mapsto
        \sum_{i=0}^{k-1} A^iHA^{k-1-i}$ heeft. Waarom kan men
        \emph{niet} $kA^{k-1}H$ schrijven in het algemeen?
  \item Bewijs dat $\Phi(A) = A^{-1}$ \omterm{def:b2:diffcalc:differential}{differentieerbaar} is op
        $GL_n(\R)$ met
        \[
        \dd\Phi_A(H) = -A^{-1}HA^{-1}
        \]
        \emph{(schrijf $(A + H)^{-1} = (I + A^{-1}H)^{-1}A^{-1}$
        en ontwikkel met vraag 2)}. Controleer de formule tegen
        het scalaire geval $n = 1$.
  \item Voor een $C^1$-kromme $t \mapsto A(t) \in GL_n(\R)$, leid af
        $\bigl(A(t)^{-1}\bigr)' = -A^{-1}A'A^{-1}$, en ontwikkel
        $t \mapsto (I + tB)^{-1}$ tot eerste orde in $t = 0$.
  \item Bereken de \omterm{def:b2:diffcalc:differential}{differentiaal} van $f(A) =
        \operatorname{tr}(A^k)$ en identificeer haar gradiënt voor
        het Frobenius-inproduct:
        \[
        \dd f_A(H) = k\operatorname{tr}\bigl(A^{k-1}H\bigr),
        \qquad
        \nabla f(A) = k\,\bigl(A^{k-1}\bigr)^{\mathsf T} .
        \]
  \item Dezelfde vragen voor $f(A) = \operatorname{tr}
        (A^{\mathsf T}A) = \norm A_F^2$: \omterm{def:b2:diffcalc:differential}{differentiaal},
        gradiënt, en de (constante) Hessematrice; concludeer dat
        $\norm\cdot_F^2$ strikt convex is.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De formule van Jacobi.}
\begin{enumerate}[resume]
  \item Bewijs
        \[
        \det(I + H) = 1 + \operatorname{tr}H +
        O\bigl(\vertiii H^2\bigr)
        \]
        \emph{(ontwikkel $\det(e_1 + h_1, \dots, e_n + h_n)$ door
        multilineaire in de kolommen: termen met minstens twee
        $h$-kolommen zijn $O(\vertiii H^2)$)}: $\dd(\det)_I =
        \operatorname{tr}$.
  \item Voor $A$ inverteerbaar, leid af
        \[
        \dd(\det)_A(H) = \det(A)\,
        \operatorname{tr}\bigl(A^{-1}H\bigr) .
        \]
  \item Toon aan dat voor \emph{elke} $A$ (inverteerbaar of niet),
        $\frac{\partial\det}{\partial a_{ij}}(A) = C_{ij}$, de
        $(i,j)$-cofactor \emph{(Laplace-ontwikkeling langs rij
        $i$)}, zodat met de geadjungeerde $\operatorname{adj}A =
        \operatorname{com}(A)^{\mathsf T}$:
        \[
        \dd(\det)_A(H) =
        \operatorname{tr}\bigl(\operatorname{adj}(A)\,H\bigr),
        \qquad
        \nabla(\det)(A) = \operatorname{com}(A) ,
        \]
        wat vraag 11 terugvindt wanneer $A$ inverteerbaar is
        ($\operatorname{adj}A = \det(A)A^{-1}$). Dit is de
        \emph{formule van Jacobi}: $\bigl(\det
        A(t)\bigr)' = \operatorname{tr}\bigl(
        \operatorname{adj}(A(t))\,A'(t)\bigr)$.
  \item (Formule van Liouville) Zij $A(t)$ een $C^1$-kromme van
        matrices die voldoet aan de lineaire differentiaalvergelijking
        $A'(t) = M(t)A(t)$. Bewijs
        \[
        \bigl(\det A(t)\bigr)' =
        \operatorname{tr}\bigl(M(t)\bigr)\,\det A(t),
        \qquad\text{dus}\qquad
        \det A(t) = \det A(0)\,
        \exp\Bigl(\int_0^t\operatorname{tr}M\Bigr)
        \]
        \emph{(gebruik $\operatorname{adj}(A)\,A = \det(A)I$ en
        de cyclische invariantie van het spoor)} --- de
        Wronskiaan-identiteit die het hoofdstuk over
        differentiaalvergelijkingen voortdurend zal gebruiken.
  \item Toon aan dat $SL_n(\R) = \{\det = 1\}$ een gladde
        niveauverzameling is: in elke $A \in SL_n(\R)$ is de
        \omterm{def:b2:diffcalc:differential}{differentiaal} $\dd(\det)_A$ een \emph{surjectieve} lineaire
        afbeelding op $\R$ \emph{(evalueer haar in $H = \frac1nA$)}.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De \omterm{ex:b2:nvs:matrixexp}{matrixexponentiaal}.}
\begin{enumerate}[resume]
  \item Toon aan dat $\eu^A = \sum_{k\geq0}\frac{A^k}{k!}$
        absoluut convergeert voor elke $A$, normaal op elke
        bal, met $\vertiii{\eu^A} \leq
        \eu^{\vertiii A}$; en dat $\eu^A$ continu van $A$
        afhangt.
  \item Bewijs dat $AB = BA$ impliceert $\eu^{A+B} =
        \eu^A\eu^B$ \emph{(\omterm{thm:b2:series:fubini}{Cauchy-product}, legitiem door
        absolute convergentie)}; leid af dat $\eu^A$ altijd
        inverteerbaar is, met inverse $\eu^{-A}$: $\exp$ beeldt
        $\mathcal M_n(\R)$ af in $GL_n(\R)$.
  \item Toon aan dat $t \mapsto \eu^{tA}$ van klasse $C^1$ is
        (inderdaad $C^\infty$) met
        \[
        \frac{\dd}{\dd t}\,\eu^{tA} = A\,\eu^{tA} =
        \eu^{tA}A
        \]
        \emph{(differentieer de reeks term voor term op
        segmenten: de afgeleide reeks convergeert normaal)}.
  \item Bewijs de identiteit
        \[
        \det\bigl(\eu^{A}\bigr) = \eu^{\operatorname{tr}A}
        \]
        \emph{(pas de formule van Liouville, vraag 13, toe op
        $A(t) = \eu^{tA}$)}. Controles: $n = 1$; nilpotente
        $A$; en de spoorloze matrices landen in
        $SL_n(\R)$.
  \item Toon aan $\eu^H = I + H + O(\vertiii H^2)$, dus $\exp$ is
        \omterm{def:b2:diffcalc:differential}{differentieerbaar} in $0$ met $\dd(\exp)_0 =
        \mathrm{id}$; concludeer met de stelling van de inverse
        functie (\cref{thm:b2:diffcalc:inverse}) dat $\exp$
        een $C^1$-diffeomorfisme is van een omgeving van $0$
        op een omgeving van $I$: elke matrix dicht bij de
        identiteit heeft een logaritme.
  \item Toon aan dat $\exp$ symmetrische matrices afbeeldt op
        symmetrische \emph{positief definiete} matrices, bijectief
        \emph{(diagonaliseer; de inverse is de spectrale
        logaritme)}.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- De orthogonale groep als niveauverzameling.}
\begin{enumerate}[resume]
  \item Zij $F(A) = A^{\mathsf T}A$, van $\mathcal M_n(\R)$ naar
        de symmetrische matrices $S_n$. Bereken $\dd F_A(H) =
        A^{\mathsf T}H + H^{\mathsf T}A$ en toon aan dat in elke
        $A \in O_n = F^{-1}(I)$ deze \omterm{def:b2:diffcalc:differential}{differentiaal}
        \emph{surjectief} is op $S_n$ \emph{(gegeven $S \in S_n$,
        probeer $H = \frac12 AS$)}: $O_n$ is een gladde
        niveauverzameling, van dimensie $n^2 - \frac{n(n+1)}2 =
        \frac{n(n-1)}2$.
  \item Toon aan dat elke $C^1$-kromme $A(t) \in O_n$ met $A(0) =
        I$ een \emph{antisymmetrische} snelheid $A'(0)$ heeft, en
        omgekeerd dat voor antisymmetrische $K$ de kromme
        $\eu^{tK}$ in $O_n$ blijft: de raakruimte van $O_n$
        in $I$ is precies de antisymmetrische matrices.
  \item Toon aan $\det\eu^{K} = 1$ voor antisymmetrische $K$ (vraag
        18): de exponentiaalkromme leeft in de rotatiegroep
        $SO_n$. Bereken haar \omterm{def:b2:metric:complete}{volledig} voor $n = 2$: met $J =
        \begin{pmatrix}0 & -1\\ 1 & 0\end{pmatrix}$, bewijs
        \[
        \eu^{\theta J} = \begin{pmatrix} \cos\theta &
        -\sin\theta\\ \sin\theta & \cos\theta\end{pmatrix} :
        \]
        de \omterm{ex:b2:nvs:matrixexp}{matrixexponentiaal} \emph{is} de rotatie over
        $\theta$, en de reeksdefinities van cosinus en sinus
        verschijnen opnieuw binnen een matrix.
  \item (De dichtheidstruc) Met de dichtheid van $GL_n(\R)$
        (vraag 3) en continuïteit, breid van inverteerbaar naar
        alle matrices de identiteit uit
        \[
        \operatorname{adj}(AB) =
        \operatorname{adj}(B)\operatorname{adj}(A)
        \]
        \emph{(voor inverteerbare $A, B$ zijn beide zijden gelijk
        aan $\det(AB)(AB)^{-1}$; beide zijden zijn polynomiaal in
        de entries)}.
  \item Synthese. In één zin elk: (i) welke eerdere
        hoofdstukken de motor van elk Deel leverden
        (volledigheid en genormeerde algebra's; de
        spectraalstelling; de stelling van de inverse functie);
        (ii) op welke formule van dit probleem het hoofdstuk over
        differentiaalvergelijkingen leunt, en waar; (iii) wat
        $\dd(\det)_I = \operatorname{tr}$ en $\det\eu^A =
        \eu^{\operatorname{tr}A}$ zeggen over spoor en
        \omterm{def:b2:linalg:det}{determinant} als ``infinitesimaal en globaal volume'';
        (iv) wat het volume Jaar~3 maakt van vragen
        21--23 (Liegroepen en hun Liealgebra's).
\end{enumerate}
\end{problem}
