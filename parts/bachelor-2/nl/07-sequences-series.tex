\chapter{Rijen en reeksen}\label{ch:b2:series}

De theorie van de numerieke reeksen (boekdeel van Jaar~1) rijpt hier
in drie richtingen: reeksen met waarden in \omterm{def:b2:nvs:banach}{Banachruimten}, waar absolute
convergentie het werk doet; de fijnere criteria voor reële reeksen
(Abelsommatie); en \emph{\omterm{def:b2:series:summable}{sommeerbare families}} --- sommatie bevrijd
van de volgorde van de termen --- met de stelling van Fubini voor
dubbele sommen en het \omterm{thm:b2:series:fubini}{Cauchy-product}. Deze gereedschappen dragen alle
hoofdstukken over functiereeksen die volgen.

\section{Reeksen in genormeerde ruimten}

\begin{definition}\label{def:b2:series:def}
Voor een rij $(u_n)$ in een genormeerde ruimte $E$ convergeert de
reeks $\sum u_n$ wanneer haar partiële sommen dat doen; zij
convergeert \emph{absoluut}\index{absolute convergentie!in een
Banachruimte} wanneer $\sum \norm{u_n} < \infty$. In een
\emph{Banach}ruimte impliceert absolute convergentie convergentie
(\cref{thm:b2:nvs:absoluteconvergence}); in een niet-volledige
ruimte kan dit mislukken (\cref{exo:b2:series:9}).
\end{definition}

\begin{example}\label{ex:b2:series:neumann}
In $\mathcal{M}_n(K)$ (of $\mathcal{L}_c(E)$, $E$ Banach): voor
$\vertiii A < 1$ convergeert de \emph{Neumannreeks} $\sum A^k$
\omterm{def:b2:series:def}{absoluut} naar $(I - A)^{-1}$ (bewezen in \cref{exo:b2:nvs:5});
$\sum \frac{A^k}{k!}$ convergeert \omterm{def:b2:series:def}{absoluut} naar $\eu^A$ voor elke
$A$ (\cref{ex:b2:nvs:matrixexp}). Operatorwaardige meetkundige en
exponentiële reeksen gedragen zich als hun scalaire modellen ---
precies de punt van het Banach-kader.
\end{example}

\section{Abelsommatie}

\begin{theorem}[Abelsommatie en -criterium]\label{thm:b2:series:abel}
(Partiële sommatie) Voor scalaren $a_n$ en vectoren $b_n$, met
$B_n = \sum_{k=0}^{n} b_k$:\index{Abelsommatie}
\[
\sum_{n=0}^{N} a_n b_n
= a_N B_N - \sum_{n=0}^{N-1} (a_{n+1} - a_n) B_n .
\]
(Criterium van Abel) Als $(a_n)$ een reële rij is, dalend naar $0$,
en de partiële sommen $B_n$ \emph{begrensd} zijn (in een
\omterm{def:b2:nvs:banach}{Banachruimte}), dan convergeert $\sum a_n b_n$.
\end{theorem}

\begin{proof}
De identiteit, stap voor stap: met $B_{-1} = 0$, schrijf $b_n = B_n
- B_{n-1}$ en splits,
\[
\sum_{n=0}^{N} a_nb_n
= \sum_{n=0}^{N} a_nB_n - \sum_{n=0}^{N} a_nB_{n-1}
= \sum_{n=0}^{N} a_nB_n - \sum_{n=0}^{N-1} a_{n+1}B_{n} ,
\]
herindexeer de tweede som via $n \mapsto n + 1$ (de $B_{-1}$-term
verdwijnt); verzamel het gemeenschappelijke bereik $0 \leq n \leq
N-1$ en houd $a_NB_N$ plus $\sum_{n\leq N-1}(a_n - a_{n+1})B_n$
over: de gestelde formule. Het is de discrete partiële integratie,
met $(B_n)$ als primitieve van $(b_n)$ en het verschil $a_{n+1} -
a_n$ als afgeleide van $(a_n)$. Voor het criterium, met
$\norm{B_n} \leq M$: de randterm $a_N B_N \to 0$; de reeks
$\sum (a_n - a_{n+1})B_n$ convergeert \omterm{def:b2:series:def}{absoluut}, want
\[
\sum_n \norm{(a_{n+1} - a_n)B_n} \leq M \sum_n (a_n - a_{n+1})
= M a_0 < \infty
\]
(telescoperend, $a_n \downarrow 0$). Beide stukken van de
identiteit convergeren, dus ook $\sum a_n b_n$.
\end{proof}

\begin{example}\label{ex:b2:series:sinn}
$\sum \frac{\sin n}{n}$ convergeert: $a_n = \frac1n \downarrow 0$ en
$B_n = \sum_{k=1}^{n} \sin k$ is begrensd --- inderdaad $B_n =
\Im\sum_{k \leq n} \eu^{\iu k} = \Im\,\frac{\eu^{\iu}(\eu^{\iu n} -
1)}{\eu^{\iu} - 1}$, van modulus $\leq \frac{2}{\abs{\eu^{\iu} - 1}}$.
Zij convergeert \emph{niet} \omterm{def:b2:series:def}{absoluut} ($\abs{\sin n} \geq \sin^2 n
= \frac{1 - \cos 2n}{2}$, en $\sum \frac{1 - \cos 2n}{2n}$
divergeert omdat $\sum \frac{\cos 2n}{n}$ convergeert door hetzelfde
Abelcriterium terwijl $\sum \frac{1}{2n}$ divergeert). Het
criterium voor alternerende reeksen is het speciale geval $b_n =
(-1)^n$.
\end{example}

\begin{example}[Abel op de convergentiecirkel]\label{ex:b2:series:abelboundary}
Voor welke complexe $z$ met $\abs z = 1$ convergeert
$\sum_{n \geq 1} \frac{z^n}{n}$? In $z = 1$ is het de harmonische
reeks: divergent. Voor $z \neq 1$ op de cirkel past het
Abelcriterium met $a_n = \frac1n \downarrow 0$ en $b_n = z^n$,
waarvan de partiële sommen begrensd zijn onafhankelijk van $N$:
\[
\Bigl|\sum_{n=1}^{N} z^n\Bigr|
= \Bigl|\frac{z(z^N - 1)}{z - 1}\Bigr|
\leq \frac{2}{\abs{z - 1}} .
\]
Convergent --- al nooit \omterm{def:b2:series:def}{absoluut} ($\sum\frac1n$). Eén reeks, een
cirkel van gedragingen: divergentie in één punt, semi-convergentie
overal elders. Dit is het standaard randgedrag van machtreeksen
(\cref{ch:b2:powerseries}), hier met blote handen ontmoet; in $z =
-1$ herstelt men de alternerende harmonische reeks, en in $z =
\eu^{\iu\theta}$ zijn reële en imaginaire delen de reeksen
$\sum\frac{\cos n\theta}{n}$ en $\sum\frac{\sin n\theta}{n}$ van
\cref{exo:b2:series:4}.
\end{example}

\begin{example}[Een verstrikte alternerende reeks]\label{ex:b2:series:trap}
Convergeert $\sum_{n\geq2} \dfrac{(-1)^n}{\sqrt n + (-1)^n}$?
De tekens wisselen en de termen streven naar $0$ --- toch past
het alternerend criterium \emph{niet}: de moduli
$\frac{1}{\sqrt n + (-1)^n}$ zijn niet dalend (ze springen omhoog
bij elke oneven $n$). Ontwikkel in plaats daarvan:
\[
\frac{(-1)^n}{\sqrt n + (-1)^n}
= \frac{(-1)^n}{\sqrt n}\cdot
\frac{1}{1 + \frac{(-1)^n}{\sqrt n}}
= \frac{(-1)^n}{\sqrt n} - \frac{1}{n}
+ O\Bigl(\frac{1}{n^{3/2}}\Bigr).
\]
Het eerste stuk convergeert (alternerend criterium, eerlijk
toegepast op $\frac1{\sqrt n}\downarrow0$), het derde convergeert
\omterm{def:b2:series:def}{absoluut} --- maar het middelste is de divergente harmonische
reeks: de som \emph{divergeert} naar $-\infty$. Het afsluitende
inzicht: wanneer monotonie faalt, ontwikkel tot elk stuk ofwel
\omterm{def:b2:series:def}{absoluut} convergent ofwel een schoon testgeval is; de verborgen
$-\frac1n$ is onzichtbaar voor teken-tellen.
\end{example}

\begin{remark}[Veelgemaakte valkuilen]\label{rem:b2:series:pitfalls}
(i) ``Termen streven naar $0$'' bewijst niets: de harmonische reeks
divergeert. (ii) Het alternerend criterium eist \emph{dalende}
moduli --- \cref{ex:b2:series:trap} is het canonieke
tegenvoorbeeld, en de derde reeks van \cref{exo:b2:series:1} de
oefening. (iii) Semi-convergente reeksen mogen niet herschikt
worden (\cref{ex:b2:series:rearrange}), en hun \omterm{thm:b2:series:fubini}{Cauchy-producten}
kunnen divergeren: voor het kwadraat van
$\sum\frac{(-1)^n}{\sqrt{n+1}}$ voldoen de diagonaaltermen aan
\[
\abs{c_k} = \sum_{m=0}^{k}
\frac{1}{\sqrt{(m+1)(k-m+1)}}
\geq (k+1)\cdot\frac{2}{k+2} \longrightarrow 2 \neq 0
\]
(elke factor is ten hoogste $\frac{k+2}2$ door AM--GM), dus
$\sum c_k$ divergeert --- absolute convergentie van minstens één
factor (\cref{exo:b2:series:8}) is geen luxe. (iv) Sommeerbaarheid
gaat over \emph{absolute} grenzen per definitie: er bestaat geen
semi-sommeerbare familie.
\end{remark}

\section{Sommeerbare families}

\begin{definition}\label{def:b2:series:summable}
Zij $I$ een aftelbare indexverzameling. Een familie $(u_i)_{i \in
I}$ van \emph{niet-negatieve reëlen} is
\emph{sommeerbaar}\index{sommeerbare familie} wanneer de eindige
partiële sommen begrensd zijn; haar som is
\[
\sum_{i \in I} u_i = \sup_{F \subseteq I \text{ eindig}} \sum_{i
\in F} u_i \in \intcc{0}{+\infty} .
\]
Een familie van reëlen of complexen (of Banachvectoren) is
sommeerbaar wanneer $(\norm{u_i})$ dat is; haar som wordt dan
gedefinieerd door splitsing in positieve/negatieve (of
reële/imaginaire) delen --- equivalent, als de gemeenschappelijke
waarde van $\sum_{n} u_{\sigma(n)}$ over alle enumeraties $\sigma$
van $I$ (zie hieronder).
\end{definition}

\begin{method}[Een criterium kiezen]\label{met:b2:series:choosing}
Tegenover $\sum u_n$, in volgorde: (1) als $u_n \not\to 0$,
divergentie, stop. (2) Als de termen constant teken hebben,
vergelijk: zoek een equivalent (\cref{ch:b2:comparison}) en plaats
het op de Riemann--Bertrand-kaart. (3) Als de tekens alterneren met
\emph{dalende} moduli, het alternerend criterium; als de moduli
niet monotoon zijn, ontwikkel de term tot elk stuk \omterm{def:b2:series:def}{absoluut}
convergent of een schoon testgeval is (\cref{ex:b2:series:trap}).
(4) Als het tekenpatroon oscillerend maar gestructureerd is
($\sin n\theta$, $\eu^{\iu n\theta}$, matrixmachten),
Abelcriterium met begrensde partiële sommen. (5) Absolute
convergentie is altijd de moeite van een eerste check waard: zij
is sterker, volgorde-bestendig, en ontgrendelt \omterm{thm:b2:series:fubini}{Cauchy-producten} en
Fubini.
\end{method}

\begin{example}[Sommeerbaarheid door diagonaaltelling]\label{ex:b2:series:diagonalcount}
Voor welke $s > 0$ is de familie
$\bigl((m + n)^{-s}\bigr)_{m, n \geq 1}$ \omterm{def:b2:series:summable}{sommeerbaar}? Groepeer de
eindige partiële sommen per diagonalen $m + n = k$: de diagonaal
$k$ draagt $k - 1$ paren, elk bijdragend $k^{-s}$, zodat de
eindige sommen precies begrensd worden door (en uitputten)
\[
\sum_{k \geq 2} \frac{k - 1}{k^{s}} ,
\]
een reeks met positieve termen equivalent met $k^{1-s}$:
\omterm{def:b2:series:summable}{sommeerbaar} d.e.s.d.a.\ $s - 1 > 1$, d.w.z.\ $s > 2$. De
twee-dimensionale index eet één volle macht: een vlak van termen
is ``één dimensie divergenter'' dan een lijn --- de
tellingsmeetkunde van de indexverzameling, niet de grootte van
individuele termen, beslist over sommeerbaarheid. (Dezelfde
telling toont dat $\bigl((m^2 + n^2)^{-1}\bigr)$ niet
\omterm{def:b2:series:summable}{sommeerbaar} is: op de diagonaal $m + n = k$ is elke term minstens
$k^{-2}$, en $(k-1)\cdot k^{-2}$ sommeert als de harmonische
reeks.)
\end{example}

\begin{theorem}[Sommeerbaarheid en volgorde]\label{thm:b2:series:rearrangement}
\begin{enumerate}
  \item Voor niet-negatieve families is de som invariant onder
        elke enumeratie: $\sum_{i} u_i = \sum_{n=0}^{\infty}
        u_{\sigma(n)}$ voor elke bijectie $\sigma \colon \N \to I$.
  \item Een reële of complexe reeks $\sum u_n$ is
        \emph{commutatief convergent} (elke herschikking
        convergeert, met dezelfde som) dan en slechts dan als zij
        \omterm{def:b2:series:def}{absoluut} convergent is.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Elke partiële som $\sum_{n \leq N} u_{\sigma(n)}$ is een
eindige partiële som van de familie (dus $\leq$ het sup); elke
eindige $F$ zit in een $\{\sigma(0), \dots, \sigma(N)\}$ (dus het
sup $\leq$ de limiet van de reeks). De twee grenzen vallen samen.

(2) Als $\sum\abs{u_n} < \infty$: voor elke herschikking $\sigma$
en $\varepsilon > 0$, kies $N$ met $\sum_{n > N}\abs{u_n} \leq
\varepsilon$; voorbij de rang waar $\sigma$ het interval
$\intint{0}{N}$ heeft uitgeput, verschillen de herschikte
partiële sommen van de oorspronkelijke limiet met ten hoogste
$\varepsilon$: dezelfde som. Als $\sum \abs{u_n} = \infty$ maar
$\sum u_n$ convergeert (reëel geval; complex volgt
coördinaatsgewijs): de positieve en negatieve delen divergeren
beide, en men kan herschikken om elke voorgeschreven limiet te
bereiken --- de stelling van Riemann, uitgewerkt in
\cref{exo:b2:series:5} --- dus faalt commutatieve convergentie.
\end{proof}

\begin{example}[Een herschikking op heterdaad]\label{ex:b2:series:rearrange}
De alternerende harmonische reeks sommeert tot
$\sum_{n\geq1}\frac{(-1)^{n-1}}{n} = \ln 2$ (boekdeel van
Jaar~1). Herschik haar als ``één positief, twee negatieven'':
\[
1 - \frac12 - \frac14 + \frac13 - \frac16 - \frac18 + \frac15 -
\cdots
\]
Groepeer elk blok van drie,
\[
\frac{1}{2k-1} - \frac{1}{4k-2} - \frac1{4k}
= \frac{1}{4k-2} - \frac{1}{4k}
= \frac12\Bigl(\frac{1}{2k-1} - \frac1{2k}\Bigr),
\]
zodat de herschikte reeks convergeert naar $\frac12\ln 2$ ---
de helft van de oorspronkelijke som, met precies dezelfde termen.
\omterm{def:b2:series:def}{Niet-absoluut} convergente reeksen onthouden de volgorde van hun
termen; \omterm{def:b2:series:summable}{sommeerbare families} zijn precies die welke dat niet
doen.
\end{example}

\begin{example}[Groeperen is veilig, ontgroeperen niet]\label{ex:b2:series:grouping}
Groeperen van opeenvolgende termen van een \emph{convergente}
reeks verandert de som nooit: de gegroepeerde partiële sommen
vormen een deelrij van de oorspronkelijke. De omgekeerde
operatie is verboden:
\[
(1 - 1) + (1 - 1) + (1 - 1) + \cdots = 0 + 0 + \cdots = 0,
\]
maar de ontgroepeerde $1 - 1 + 1 - 1 + \cdots$ divergeert
(partiële sommen oscilleren tussen $1$ en $0$). Ontgroeperen is
enkel legitiem met een compenserende hypothese --- bijvoorbeeld
termen die naar $0$ streven met begrensde groeplengten: dan
drijven tussen twee gegroepeerde partiële sommen de
oorspronkelijke er met ten hoogste een som van $o(1)$-termen van
begrensd aantal van af, en de convergentie draagt over. Dat is
precies de clausule waaronder de blokberekening van
\cref{ex:b2:series:rearrange} een bewijs is en geen goocheltruc.
\end{example}

\begin{theorem}[Fubini voor families; Cauchy-producten]\label{thm:b2:series:fubini}
Zij $(u_{m,n})_{(m,n) \in \N^2}$ een sommeerbare dubbele familie
(d.w.z.\ $\sup_F \sum_F \abs{u_{m,n}} < \infty$). Dan\index{stelling
van Fubini!voor reeksen}
\[
\sum_{(m,n)} u_{m,n}
= \sum_{m=0}^{\infty}\Bigl(\sum_{n=0}^{\infty} u_{m,n}\Bigr)
= \sum_{n=0}^{\infty}\Bigl(\sum_{m=0}^{\infty} u_{m,n}\Bigr)
= \sum_{k=0}^{\infty}\Bigl(\sum_{m+n=k} u_{m,n}\Bigr),
\]
alle inwendige reeksen convergent (\omterm{def:b2:series:def}{absoluut}). In het bijzonder,
als $\sum a_m$ en $\sum b_n$ \omterm{def:b2:series:def}{absoluut} convergeren, convergeert
hun \emph{Cauchy-product}\index{Cauchy-product} \omterm{def:b2:series:def}{absoluut} met
\[
\Bigl(\sum_m a_m\Bigr)\Bigl(\sum_n b_n\Bigr)
= \sum_{k=0}^{\infty} c_k,
\qquad
c_k = \sum_{m=0}^{k} a_m b_{k-m} .
\]
\end{theorem}

\begin{proof}
\emph{Niet-negatief geval.} Elke groepering (per rijen, kolommen
of diagonalen) berekent hetzelfde supremum: elke eindige
verzameling paren zit in een eindig blok rijen (wat elke
gegroepeerde som van onder begrensd door eindige partiële sommen
en van boven door de totaalsom), en monotone convergentie van
partiële sommen doet de rest --- concreet, voor rijen:
$\sum_{m \leq M}\sum_{n \leq N} u_{m,n} \leq S$ geeft, $N \to
\infty$ dan $M \to \infty$, $\sum_m \sum_n u_{m,n} \leq S$;
omgekeerd zit elke eindige $F$ in zo'n rechthoek, dus $S \leq
\sum_m\sum_n u_{m,n}$. Diagonalen: dezelfde twee grenzen met
driehoeken in plaats van rechthoeken.

\emph{Algemeen geval.} Splits in positieve en negatieve (reële
en imaginaire) delen, elk een sommeerbare niet-negatieve
familie; de vier groeperingen komen overeen op elk deel, dus op
het verschil; absolute convergentie van de inwendige reeksen
komt van het niet-negatieve geval toegepast op $\abs{u_{m,n}}$.

\emph{\omterm{thm:b2:series:fubini}{Cauchy-product}.} De familie $u_{m,n} = a_m b_n$ is
\omterm{def:b2:series:summable}{sommeerbaar}: eindige partiële sommen van $\abs{a_mb_n}$ zijn
begrensd door
$\bigl(\sum\abs{a_m}\bigr)\bigl(\sum\abs{b_n}\bigr)$. Rijen geven
$\bigl(\sum a_m\bigr)\bigl(\sum b_n\bigr)$; diagonalen geven
$\sum_k c_k$.
\end{proof}

\begin{example}[De exponentiële identiteit, eerlijk]\label{ex:b2:series:exp}
Voor $a, b \in \C$ (of commuteren de matrices):
\[
\Bigl(\sum_m \frac{a^m}{m!}\Bigr)\Bigl(\sum_n
\frac{b^n}{n!}\Bigr)
= \sum_k \sum_{m+n=k} \frac{a^m b^n}{m!\,n!}
= \sum_k \frac{(a + b)^k}{k!},
\]
via het binomium op elke diagonaal: $\eu^a \eu^b = \eu^{a+b}$
--- de functionaalvergelijking van $\exp$ afgeleid uit de reeks
alleen. (Commutatie wordt gebruikt in de binomiale stap; voor
niet-commuterende matrices faalt de identiteit echt,
\cref{ch:b2:diffeq}.)
\end{example}

\begin{example}[Cauchy-producten als rekenmachine]\label{ex:b2:series:squaresum}
Uit de meetkundige reeks en $\sum_{n \geq 1} nz^n =
\frac{z}{(1-z)^2}$ van \cref{exo:b2:series:2} ($\abs z < 1$)
rondt één extra \omterm{thm:b2:series:fubini}{Cauchy-product} het tweede moment af. Vermenigvuldig
$\sum_m mz^m$ met $\sum_n z^n$: de diagonaalcoëfficiënt is
$\sum_{m=0}^k m = \frac{k(k+1)}2$, dus
\[
\frac{z}{(1-z)^3} = \sum_{k\geq0}\frac{k(k+1)}{2}\,z^k ,
\]
en de identiteit $n^2 = 2\cdot\frac{n(n+1)}2 - n$ assembleert
\[
\sum_{n\geq1} n^2z^n = \frac{2z}{(1-z)^3} - \frac{z}{(1-z)^2}
= \frac{z(1+z)}{(1-z)^3} .
\]
In $z = \frac12$: $\sum_{n\geq1}\frac{n^2}{2^n} =
\frac{\frac12\cdot\frac32}{\frac18} = 6$ --- een gesloten waarde
zonder enige afleiding, enkel \omterm{def:b2:series:def}{absoluut} convergente reeksen
vermenigvuldigd als veeltermen. Dezelfde telescopering van
identiteiten berekent elke $\sum n^dz^n$, en probabilisten
herkennen het tweede factoriële moment van de meetkundige wet
(\cref{ch:b2:genfun}).
\end{example}

\begin{example}[Een dubbelsomevaluatie]\label{ex:b2:series:doublesum}
Voor $s > 1$ reëel, stel $\zeta(s) = \sum_{n\geq1} n^{-s}$.
Delers tellen door dubbele sommatie --- de familie
$(m^{-s}n^{-s})$ over $(m,n) \in (\N^*)^2$ is \omterm{def:b2:series:summable}{sommeerbaar}
(product van convergente positieve reeksen) --- en groeperen
volgens het product $q = mn$:
\[
\zeta(s)^2 = \sum_{m,n} \frac{1}{(mn)^s}
= \sum_{q=1}^{\infty} \frac{d(q)}{q^s},
\]
waar $d(q)$ het aantal delers van $q$ is. \omterm{def:b2:series:summable}{Sommeerbare families}
zetten combinatoriek om in analyse.
\end{example}

\begin{example}[Een Fubini-evaluatie: $\sum_n (\zeta(n) - 1) = 1$]\label{ex:b2:series:zetasum}
Voor gehele $n \geq 2$ is $\zeta(n) - 1 = \sum_{k \geq 2}
k^{-n}$. De dubbele familie $(k^{-n})_{k, n \geq 2}$ is
\omterm{def:b2:series:summable}{sommeerbaar}: sommeer eerst de meetkundige kolommen,
\[
\sum_{k\geq2}\sum_{n\geq2} \frac{1}{k^n}
= \sum_{k\geq2} \frac{1/k^2}{1 - 1/k}
= \sum_{k\geq2} \frac{1}{k(k-1)} = 1
\]
(telescoperend), en alle termen zijn positief, zodat
\cref{thm:b2:series:fubini} rijen toestaat in plaats daarvan:
\[
\sum_{n\geq2}\bigl(\zeta(n) - 1\bigr) = 1 .
\]
De oneindig vele $\zeta$-waarden, elk transcendent-ogend, hebben
staarten die precies tot $1$ optellen. Afsluitend inzicht: wanneer
een dubbelsom positieve termen heeft, bereken haar in de volgorde
die instort --- hier zijn kolommen meetkundig, rijen mysterieus,
en Fubini draagt de instorting over.
\end{example}

\begin{example}[De meetkundige reeks lost een vergelijking op]\label{ex:b2:series:neumannsolve}
In de \omterm{def:b2:nvs:banach}{Banachruimte} $\bigl(C(\intcc01),
\norm\cdot_\infty\bigr)$, los $x - K(x) = y$ op waar $K(f)$ de
constante functie $\frac12\int_0^1 f$ is. De \omterm{thm:b2:nvs:continuouslinear}{operatornorm} is
$\vertiii K \leq \frac12 < 1$, dus past de Neumannreeks
(\cref{ex:b2:series:neumann}): $x = \sum_{n\geq0} K^n(y)$.
Bereken de iteraten: $K(y) = \frac12\int_0^1 y$ (een constante),
en $K$ op een constante $c$ geeft $\frac c2$, dus $K^n(y) =
\frac{1}{2^{n-1}}\cdot\frac12\int_0^1 y$ voor $n \geq 1$.
Sommeer de meetkundige constanten:
\[
x = y + \Bigl(\int_0^1 y\Bigr)
\sum_{n\geq1}\frac{1}{2^n} = y + \int_0^1 y .
\]
Controle: $x - K(x) = y + \int y - \frac12\bigl(\int y + \int
y\bigr) = y$. Een oneindige reeks, een eindig antwoord, en een
éénregelig verificatie --- de meetkundige reeks is een
inversiealgoritme, niet enkel een convergentieverklaring.
\end{example}

\begin{example}[Telescoperen via partiële breuken]\label{ex:b2:series:telescope}
Exacte sommatie is zeldzaam; telescoperen is haar
hoofdleverancier. Ontbind
\[
\frac{1}{n(n+1)(n+2)}
= \frac{1}{2}\Bigl(\frac{1}{n(n+1)} -
\frac{1}{(n+1)(n+2)}\Bigr),
\]
(controleer door reductie tot de gemeenschappelijke noemer),
zodat de partiële sommen instorten:
\[
\sum_{n=1}^{N}\frac{1}{n(n+1)(n+2)}
= \frac12\Bigl(\frac{1}{1\cdot2} -
\frac{1}{(N+1)(N+2)}\Bigr)
\longrightarrow \frac14 .
\]
Hetzelfde patroon --- schrijf de term als $c(u_n - u_{n+1})$
voor een expliciete $(u_n)$ --- loste \cref{exo:b2:series:10}
op (arctangensen) en berekent elke
$\sum\frac{1}{n(n+1)\cdots(n + k)} = \frac{1}{k\cdot k!}$.
Wanneer een exacte som op dit niveau bestaat, schuilt er
meestal een telescoop in de term.
\end{example}

\begin{remark}[Perspectieven binnen dit boekdeel]\label{rem:b2:series:perspectives}
Drie hoofdstukken verderop zijn directe cliënten. Voor
\cref{ch:b2:funcseq}: normale convergentie van $\sum f_n$ is
absolute convergentie van $\sum\norm{f_n}_\infty$ in de
\omterm{def:b2:nvs:banach}{Banachruimte} $\bigl(C, \norm\cdot_\infty\bigr)$ ---
\cref{thm:b2:nvs:absoluteconvergence} van dit hoofdstuk in
kostuum. Voor \cref{ch:b2:powerseries}: binnen de
convergentieschijf is alles \omterm{def:b2:series:def}{absoluut} en \omterm{def:b2:series:summable}{sommeerbaar}, zodat
\omterm{thm:b2:series:fubini}{Cauchy-producten} en herschikkingen vrij lopen (daarom
vermenigvuldigen machtreeksen als veeltermen); op de rand
neemt het Abelcriterium over
(\cref{ex:b2:series:abelboundary}). Voor \cref{ch:b2:genfun}:
waarschijnlijkheidsgenererende functies zijn machtreeksen
waarvan de manipulaties --- producten voor sommen van
onafhankelijke variabelen, dubbelsommen voor samengestelde
wetten --- alle gelicentieerd worden door
\cref{thm:b2:series:fubini}. \omterm{def:b2:series:summable}{Sommeerbare families} zijn de
juridische dienst van de analyse die volgt.
\end{remark}

\begin{remark}[Waar dit hoofdstuk gebruikt wordt]
Alles met een oneindige som passeert hier: machtreeksen
(\cref{ch:b2:powerseries}) zijn \omterm{def:b2:series:summable}{sommeerbare families} in
vermoming, Fouriercoëfficiënten worden vermenigvuldigd door
\omterm{thm:b2:series:fubini}{Cauchy-producten} en herschikt door Parseval
(\cref{ch:b2:fourier}), en waarschijnlijkheidsgenererende
functies (\cref{ch:b2:genfun}) zijn de stelling van Fubini
toegepast op verwachtingen. Het boekdeel van Jaar~3 absorbeert
\omterm{def:b2:series:summable}{sommeerbare families} in Lebesgue-integratie over de
telmaat --- waar \cref{thm:b2:series:fubini} een speciaal
geval wordt van de stelling van Fubini--Tonelli.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b2:series:1}
Aard van: $\sum \dfrac{\cos n}{n}$; $\;\sum \dfrac{(-1)^n}{\ln
n}$; $\;\sum \dfrac{(-1)^n}{n^{3/4} + \cos n}$ \emph{(ontwikkel
zoals in de val van Jaar~1: het alternerend criterium eist
monotonie)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:series:2}
Bewijs dat voor $\abs z < 1$: $\sum_{n\geq1} n z^{n} =
\dfrac{z}{(1-z)^2}$, via het \omterm{thm:b2:series:fubini}{Cauchy-product} van $\sum z^n$ met
zichzelf.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:series:3}
(Lemma van Kronecker-type) Zij $\sum b_n$ een convergente reële
reeks. Bewijs, via Abelsommatie, dat $\dfrac{1}{n}\sum_{k=1}^{n}
k\,b_k \to 0$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:series:4}
Bestudeer de convergentie van $\sum \dfrac{\sin(n\theta)}{n^\alpha}$
($\theta \in \R$, $\alpha > 0$) --- voor welke $(\theta, \alpha)$
is zij \omterm{def:b2:series:def}{absoluut} convergent, semi-convergent, divergent?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:series:5}
(Herschikking van Riemann) Zij $\sum u_n$ een convergente maar
niet \omterm{def:b2:series:def}{absoluut} convergente reële reeks, en $\ell \in \R$. Bewijs
dat enige herschikking van $\sum u_n$ naar $\ell$ convergeert.
\emph{(Toon dat beide deelreeksen van positieve en negatieve
termen divergeren; wissel dan gretig af: neem positieve termen
tot $\ell$ overschreden is, dan negatieve tot eronder, enzovoort;
de termen streven naar $0$, wat convergentie naar $\ell$
afdwingt.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:series:6}
Bewijs dat de familie
$\Bigl(\dfrac{x^{m+n}}{m!\,n!}\Bigr)_{(m,n)\in\N^2}$
\omterm{def:b2:series:summable}{sommeerbaar} is voor elke $x \in \R$, en herleid de identiteit
$(\eu^x)^2 = \eu^{2x}$ door de dubbelsom te groeperen langs de
diagonalen $m + n = k$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:series:7}
Bewijs dat de familie $\bigl(\frac{1}{m^2 n^2}\bigr)_{m,n \geq 1}$
\omterm{def:b2:series:summable}{sommeerbaar} is, en dat groeperen via $\gcd$: met $q = \gcd(m,n)$,
\[
\zeta(2)^2 = \sum_{q\geq1} \frac{1}{q^4}
\sum_{\substack{a,b \geq 1\\ \gcd(a,b)=1}} \frac{1}{a^2b^2}
= \zeta(4) \cdot S,
\]
waar $S = \sum_{\gcd(a,b)=1} \frac{1}{a^2b^2}$: leid af $S =
\zeta(2)^2/\zeta(4)$. \emph{(Elk paar $(m,n)$ schrijft uniek
$(qa, qb)$ met $\gcd(a,b) = 1$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:series:8}
(Stelling van Abel over producten, lichte versie) Stel dat
$\sum a_n$ \omterm{def:b2:series:def}{absoluut} convergeert en $\sum b_n$ convergeert. Bewijs
dat hun \omterm{thm:b2:series:fubini}{Cauchy-product} $\sum c_n$ convergeert, met $\sum c_n =
(\sum a_n)(\sum b_n)$. \emph{(Schrijf $C_N = \sum_{k\leq N} c_k
= \sum_n a_n B_{N-n}$ met $B$ de partiële sommen van $b$; splits
volgens $n \leq N/2$ of niet, gebruik begrensdheid van $(B_m)$
en de absolute staart van $(a_n)$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:series:9}
In de (niet-volledige) ruimte $E$ van reële rijen die uiteindelijk
nul zijn met de \omterm{def:b2:nvs:norm}{sup-norm}, vertoon een \omterm{def:b2:series:def}{absoluut} convergente reeks
die niet convergeert in $E$. \emph{(Probeer $u_n = 2^{-n} e_n$
met $(e_n)$ de canonieke rijen.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:series:10}
Verifieer de identiteit $\arctan(n+1) - \arctan(n) =
\arctan\dfrac{1}{n^2 + n + 1}$, en leid de exacte waarde af van
\[
\sum_{n=1}^{\infty} \arctan\frac{1}{n^2 + n + 1} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:series:11}
Bepaal de aard (met equivalenten) van
\[
\sum_n \bigl(\sqrt{n+1} - \sqrt n\bigr)^{\alpha}
\ (\alpha > 0), \qquad
\sum_n \Bigl(1 - \cos\frac1n\Bigr), \qquad
\sum_n \Bigl(\eu - \Bigl(1 + \frac1n\Bigr)^{\!n}\Bigr).
\]
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:series:12}
Zij $(a_n)$ positief en \emph{dalend} met $\sum a_n$ convergent.
Bewijs dat $n\,a_n \to 0$ \emph{(begrens $n a_{2n}$ door een
staart)}. Toon dat de omgekeerde faalt, en dat monotonie
essentieel is, met expliciete tegenvoorbeelden.
\end{exercise}

\section{Probleem: Eulers $\zeta(2) = \pi^2/6$, via Cauchys
cotangenssom}

Eulers beroemdste identiteit, $1 + \frac14 + \frac19 + \cdots =
\frac{\pi^2}6$, laat een \omterm{def:b2:metric:complete}{volledig} elementair bewijs toe, van
Cauchy: de formule van de Moivre produceert een veelterm waarvan
de wortels de getallen $\cot^2\frac{k\pi}{2n+1}$ zijn, Vieta
sommeert die wortels exact, en de insluiting $\cot^2\theta <
\frac{1}{\theta^2} < 1 + \cot^2\theta$ verplettert de partiële
sommen van $\sum\frac1{k^2}$ tussen twee expliciete rationale
grenzen. We voeren het bewijs \omterm{def:b2:metric:complete}{volledig} uit, extraheren $\zeta(4)
= \frac{\pi^4}{90}$ met dezelfde methode, en mappen dan de hele
grens tussen convergentie en divergentie met de Bertrandreeksen
--- en bewijzen dat die grens helemaal geen traagste convergente
reeks draagt.

\begin{problem}[{Weekendopgave --- $\zeta(2) = \pi^2/6$ en het
Bertrand-panorama}]
\label{pb:b2:series:1}
Doorheen, $n \geq 1$ en $\theta_k = \dfrac{k\pi}{2n+1}$ voor $k
= 1, \dots, n$; noteer $0 < \theta_k < \frac\pi2$.

\medskip
\noindent\textbf{Deel I --- De cotangensidentiteit.}
\begin{enumerate}
  \item Bewijs de formule van de Moivre $(\cos\theta +
        \iu\sin\theta)^m = \cos m\theta + \iu\sin m\theta$ ($m
        \in \N$), en leid af, voor $m = 2n + 1$,
        \[
        \sin\bigl((2n{+}1)\theta\bigr) = \sum_{j=0}^{n}
        (-1)^j\binom{2n+1}{2j+1}\cos^{2(n-j)}\theta\,
        \sin^{2j+1}\theta .
        \]
  \item Leid af dat voor $\theta \in \intoo{0}{\frac\pi2}$,
        \[
        \sin\bigl((2n{+}1)\theta\bigr) =
        \sin^{2n+1}\theta\; P_n(\cot^2\theta),
        \qquad
        P_n(x) = \sum_{j=0}^{n}
        (-1)^j\binom{2n+1}{2j+1}x^{\,n-j},
        \]
        een veelterm van graad $n$ met kopcoëfficiënt $2n
        + 1$.
  \item Toon dat $x_k = \cot^2\theta_k$, $k = 1, \dots, n$, $n$
        \emph{onderscheiden} wortels van $P_n$ zijn --- dus
        alle.
  \item Via Vieta, bewijs de exacte identiteit
        \[
        \sum_{k=1}^{n} \cot^2\frac{k\pi}{2n+1}
        = \frac{n(2n-1)}{3}.
        \]
  \item Leid ook af $\displaystyle\sum_{k=1}^{n}
        \frac{1}{\sin^2\theta_k} = \frac{2n(n+1)}{3}$.
  \item Bewijs de insluiting: $\cot^2\theta < \dfrac1{\theta^2} <
        \dfrac{1}{\sin^2\theta}$ voor $\theta \in
        \intoo{0}{\frac\pi2}$ \emph{(uit $\sin\theta < \theta <
        \tan\theta$)}.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De insluiting sluit: de stelling van
Euler.}
\begin{enumerate}[resume]
  \item Sommeer vraag 6 over $k = 1, \dots, n$ met $\theta =
        \theta_k$, en vestig
        \[
        \frac{n(2n-1)}{3} \;<\;
        \frac{(2n+1)^2}{\pi^2}\sum_{k=1}^{n}\frac1{k^2}
        \;<\; \frac{2n(n+1)}{3}.
        \]
  \item Concludeer (\emph{stelling van Euler, via Cauchys
        bewijs}):
        \[
        \zeta(2) = \sum_{k=1}^{\infty}\frac{1}{k^2} =
        \frac{\pi^2}{6}.
        \]
  \item Extraheer een snelheid uit de sandwich: toon
        \[
        \Bigl|\sum_{k=1}^{n}\frac1{k^2} - \frac{\pi^2}6\Bigr|
        = O\Bigl(\frac1n\Bigr),
        \]
        consistent met de exacte staart $\sum_{k>n}k^{-2} =
        \frac1n - \frac{1}{2n^2} + O(n^{-3})$ van
        \cref{exo:b2:comparison:11}.
  \item Draai de machine één verdieping hoger: met de tweede
        Vieta-functie van $P_n$, toon
        \[
        \sum_{k=1}^{n}\cot^4\theta_k =
        \Bigl(\frac{n(2n-1)}3\Bigr)^{\!2} -
        \frac{2n(2n-1)(2n-2)(2n-3)}{60}
        \;\sim\; \frac{8n^4}{45},
        \]
        en sluit in met $\cot^4 < \theta^{-4} < (1 +
        \cot^2)^2$ om $\zeta(4) = \dfrac{\pi^4}{90}$ te
        verkrijgen.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Dividenden.}
\begin{enumerate}[resume]
  \item Leid uit $\zeta(2) = \frac{\pi^2}6$ af:
        \[
        \sum_{k\geq0}\frac{1}{(2k+1)^2} = \frac{\pi^2}{8},
        \qquad
        \sum_{k\geq1}\frac{(-1)^{k-1}}{k^2} = \frac{\pi^2}{12}.
        \]
  \item Combineer met \cref{exo:b2:series:7}: bereken $S =
        \sum_{\gcd(a,b)=1}\frac{1}{a^2b^2} =
        \frac{\zeta(2)^2}{\zeta(4)} = \frac52$, en interpreteer
        $\frac{1}{\zeta(2)} = \frac{6}{\pi^2} \approx 0.608$
        als de dichtheid van coprieme paren (formuleer de
        heuristiek eerlijk: de rigoureuze telling is een zaak
        van het boekdeel van Jaar~3).
  \item (Gecertificeerde versnelling) De staartformule van vraag
        9 geeft $\sum_{k\leq n}k^{-2} + \frac1n - \frac1{2n^2} =
        \frac{\pi^2}6 + O(n^{-3})$. Vergelijk het werk nodig
        voor zes decimalen van $\zeta(2)$: directe sommatie
        versus de gecorrigeerde som bij $n = 100$ (waar de fout
        $1.7\cdot10^{-7}$ is).
  \item Controleer vraag 4 met de hand bij $n = 1$ en $n = 2$
        (de waarden $\cot^2\frac\pi3 = \frac13$ en
        $\cot^2\frac\pi5 + \cot^2\frac{2\pi}5 = 2$), met
        $\cos\frac\pi5 = \frac{1+\sqrt5}4$ of een numerieke
        evaluatie.
  \item Bewijs de companion-identiteit
        \[
        \sum_{k=1}^{n}\tan^2\frac{k\pi}{2n+1} = n(2n+1)
        \]
        \emph{(de getallen $\tan^2\theta_k$ zijn de wortels van
        de omgekeerde veelterm $x^nP_n(1/x)$)}, en verifieer
        haar bij $n = 1$.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Het Bertrand-panorama.} Voor $\alpha,
\beta \in \R$, beschouw de \emph{Bertrandreeks}
\[
\sum_{n \geq 3} \frac{1}{n^{\alpha}(\ln n)^{\beta}} .
\]
\begin{enumerate}[resume]
  \item Toon dat voor $\alpha > 1$ de reeks convergeert,
        ongeacht $\beta$ \emph{(vergelijk met
        $n^{-(1+\alpha)/2}$)}.
  \item Toon dat voor $\alpha < 1$ zij divergeert, ongeacht
        $\beta$.
  \item Voor $\alpha = 1$: met de reeks--integraalvergelijking
        (\cref{thm:b2:comparison:seriesintegral}) met $f(t) =
        \frac{1}{t(\ln t)^\beta}$, bewijs convergentie d.e.s.d.a.\
        $\beta > 1$.
  \item Itereer de grens: toon $\sum\frac{1}{n\ln n\,\ln\ln
        n}$ divergeert terwijl $\sum\frac{1}{n\ln n\,(\ln\ln
        n)^2}$ convergeert.
  \item Twee vallen: bepaal de aard van
        \[
        \sum_n \frac{1}{n^{1 + 1/\ln n}}
        \qquad\text{en}\qquad
        \sum_n \frac{1}{n^{1 + 1/\ln\ln n}}
        \]
        \emph{(bereken $n^{1/\ln n}$ exact; vergelijk
        $n^{1/\ln\ln n}$ met elke macht van $\ln n$)}.
  \item (Geen traagste convergente reeks) Zij $\sum a_n$ enige
        convergente reeks met $a_n > 0$, en $R_n = \sum_{k
        \geq n}a_k$ haar staarten. Bewijs dat $\sum
        \frac{a_n}{\sqrt{R_n}}$ nog \emph{convergeert}
        \emph{(vergelijk met de telescoperende $2(\sqrt{R_n} -
        \sqrt{R_{n+1}})$)}, hoewel $\frac{a_n/\sqrt{R_n}}{a_n}
        \to \infty$: elke convergente reeks wordt strikt
        gedomineerd door een andere convergente reeks. De grens
        van convergentie is geen kromme maar een mist.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Controles en synthese.}
\begin{enumerate}[resume]
  \item (Cauchy-condensatie) Bewijs: voor $(a_n)$ positief
        dalend, convergeert $\sum a_n$ d.e.s.d.a.\ $\sum 2^k
        a_{2^k}$ convergeert. Herleid de grens van vraag 18
        hieruit.
  \item (De prijs van traagheid) Voor $\sum\frac1{n(\ln n)^2}$,
        begrens de staart door een integraal en toon dat
        sommeren tot $N = 10^6$ nog een fout groter dan $0.07$
        laat: door de theorie gecertificeerde convergentie kan
        nutteloos zijn voor numeriek werk --- contrasteer met
        vraag 13.
  \item Classificeer (met éénregelige rechtvaardigingen):
        $\sum\frac1{n\ln n}$, $\sum\frac1{n^{1.01}}$,
        $\sum\frac{(\ln n)^{100}}{n^{1.001}}$,
        $\sum\frac1{n(\ln n)(\ln\ln n)^{3}}$.
  \item (Synthese) Eén zin elk: hoe de Moivre een
        goniometrische identiteit in een veelterm met
        berekenbare wortelsommen omzette; waar de insluiting
        \emph{exacte} eindpuntidentiteiten nodig had eerder dan
        equivalenten; welk gereedschap van
        \cref{ch:b2:comparison} Deel IV aandreef; en wat vraag
        21 zegt over de droom van een ``universeel
        vergelijkingscriterium''. Noem de twee toppen: Eulers
        $\zeta(2) = \frac{\pi^2}6$ (en haar verdieping erboven,
        $\zeta(4) = \frac{\pi^4}{90}$), en de
        Bertrand-classificatie. Noteer waar $\zeta(2)$ opnieuw
        bewezen wordt: door Parseval in \cref{ch:b2:fourier}
        --- één stelling, twee beschavingen.
\end{enumerate}
\end{problem}
