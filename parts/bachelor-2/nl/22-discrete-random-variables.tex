\chapter{Discrete stochastische variabelen}\label{ch:b2:randomvar}

\omterm{def:b2:randomvar:law}{Stochastische variabelen} organiseren kansberekeningen rond
functies in plaats van \omterm{def:b2:proba:space}{gebeurtenissen}. Op aftelbare ruimten
wordt de theorie aangedreven door de \omterm{def:b2:series:summable}{sommeerbare families} van
\cref{ch:b2:series}: \omterm{def:b2:randomvar:expectation}{verwachting} is de som van een familie
geïndexeerd door de \omterm{def:b2:proba:space}{uitkomstenruimte}, en al haar eigenschappen
--- lineariteit, overdracht, productformule voor
onafhankelijke variabelen --- zijn stellingen over
\omterm{def:b2:series:summable}{sommeerbare families}. Het hoofdstuk bewijst de sleutelongelijkheden
van Markov, Chebyshev, Cauchy--Schwarz en Jensen, en eindigt
met de klassieke wetten en de zwakke wet van de grote
aantallen, waarvan het bewijs twee regels is zodra Chebyshev
beschikbaar is.

\section{Stochastische variabelen en hun wetten}

\begin{definition}[Discrete stochastische variabele;
wet]\label{def:b2:randomvar:law}
Zij $(\Omega, \P)$ een aftelbare \omterm{def:b2:proba:space}{waarschijnlijkheidsruimte}. Een
\emph{stochastische variabele}\index{stochastische variabele} is
een afbeelding $X \colon \Omega \to E$ ($E$ willekeurige
verzameling; \emph{reële} stochastische variabele wanneer $E =
\R$). Haar \emph{wet}\index{wet!van een stochastische
variabele} (of \emph{verdeling}\index{verdeling}) is de
\omterm{def:b2:proba:space}{waarschijnlijkheidsmaat} $\P_X$ op de \omterm{def:b2:structures:countable}{aftelbare verzameling}
$X(\Omega)$ gedefinieerd door
\[
\P_X(\{x\}) = \P(X = x)
= \P\bigl(\{\omega : X(\omega) = x\}\bigr) .
\]
\end{definition}

\begin{example}[De klassieke wetten]\label{ex:b2:randomvar:laws}
\leavevmode
\begin{itemize}
\item \emph{Bernoulli} $\mathcal{B}(p)$: $X \in \{0, 1\}$, $\P(X =
1) = p$. Indicator van een \omterm{def:b2:proba:space}{gebeurtenis}.
\item \emph{Binomiaal} $\mathcal{B}(n, p)$: $\P(X = k) =
\binom nk p^k(1-p)^{n-k}$, $0 \leq k \leq n$: aantal successen
in $n$ onafhankelijke Bernoulli-proeven
(middelbareschoolvolume; hieronder opnieuw bewezen via sommen
van onafhankelijke variabelen).
\item \emph{Geometrisch} $\mathcal{G}(p)$: $\P(X = k) =
(1-p)^{k-1}p$, $k \in \N^*$: rang van het eerste succes
(\cref{ex:b2:proba:geometric}).
\item \emph{Poisson} $\mathcal{P}(\lambda)$: $\P(X = k) =
e^{-\lambda}\frac{\lambda^k}{k!}$, $k \in \N$ --- een
\omterm{def:b2:proba:space}{waarschijnlijkheidsmaat} door de exponentiële reeks. De wet van
zeldzame gebeurtenissen (\cref{ch:b2:genfun}).
\end{itemize}
\end{example}

\begin{remark}[Welke wet modelleert wat]
De vier wetten beantwoorden vier primitieve vragen: Bernoulli,
``is het gebeurd?''; binomiaal, ``hoe vaak in $n$ pogingen?'';
geometrisch, ``hoe lang tot de eerste keer?''; Poisson, ``hoeveel
\omterm{def:b2:proba:space}{gebeurtenissen} bij een gegeven tempo, wanneer pogingen talrijk
en individueel onwaarschijnlijk zijn?''. De vraag herkennen is
negen tiende van het modelleren: sommen van indicatoren wijzen
naar de binomiaal, wachttijden naar de geometrische,
zeldzame-gebeurtenistellingen naar Poisson --- met de overgang
van binomiaal naar Poisson precies gemaakt door de wet van
zeldzame gebeurtenissen in \cref{ch:b2:genfun}.
\end{remark}

\begin{proposition}[Geheugenloosheid van de geometrische
wet]\label{prop:b2:randomvar:memoryless}
Als $X \sim \mathcal{G}(p)$, dan voor alle $m, n \in \N$:
\[
\P(X > m + n \mid X > m) = \P(X > n) ,
\]
en de geometrische wetten zijn de enige wetten op $\N^*$ met
deze eigenschap.
\end{proposition}

\begin{proof}
Sommeren van de geometrische gewichten geeft $\P(X > n) =
(1-p)^n$. Dus
\[
\P(X > m + n \mid X > m)
= \frac{\P(X > m + n)}{\P(X > m)}
= \frac{(1-p)^{m+n}}{(1-p)^m} = (1-p)^n = \P(X > n).
\]
Omgekeerd, als $G(n) = \P(X > n)$ voldoet aan $G(m + n) =
G(m)G(n)$ met $G(0) = 1$, dan $G(n) = G(1)^n$ door inductie; $q =
G(1) \in \intco{0}{1}$, en $q = 0$ of de \omterm{def:b2:randomvar:law}{wet} is
$\mathcal{G}(1 - q)$: $\P(X = k) = G(k-1) - G(k) =
q^{k-1}(1 - q)$.
\end{proof}

\begin{example}[Geen getal is ooit ``verschuldigd'']
Gooi een dobbelsteen, wachtend op een zes: de wachttijd is $X
\sim \mathcal G(1/6)$. Geheugenloosheid zegt dat na $10$
vruchteloze worpen de \emph{resterende} wacht $X - 10$, gegeven
$X > 10$, opnieuw $\mathcal G(1/6)$ is: de voorwaardelijke
verwachte wacht is nog steeds $6$ worpen, precies als bij de
start. De dobbelsteen herinnert zich niets, en geen zes is ooit
verschuldigd: de gokkersdrogreden is precies het geloof dat de
voorwaardelijke \omterm{def:b2:randomvar:law}{wet} had moeten
verschuiven. Omgekeerd zegt de uniciteitshelft van de propositie
dat deze onverschilligheid geometrische wachttijden
\emph{karakteriseert}: elke wachttijd waarvan de voorspelling
nooit bijwerkt is geometrisch. Echte wachtrijen en levensduren
werken hun voorspelling meestal wél bij, wat precies is hoe men
detecteert dat ze niet geometrisch zijn.
\end{example}

\section{Verwachting}

\begin{definition}[Verwachting]\label{def:b2:randomvar:expectation}
Een reële \omterm{def:b2:randomvar:law}{stochastische variabele} $X$ op $(\Omega, \P)$
\emph{heeft een verwachting} als de familie
$\bigl(X(\omega)\,\P(\{\omega\})\bigr)_{\omega \in \Omega}$
\omterm{def:b2:series:summable}{sommeerbaar} is (\cref{ch:b2:series}); haar
\emph{verwachting}\index{verwachting} is dan
\[
\E(X) = \sum_{\omega \in \Omega} X(\omega)\,\P(\{\omega\}) .
\]
\end{definition}

\begin{theorem}[Overdrachtsstelling]\label{thm:b2:randomvar:transfer}
$X$ heeft een \omterm{def:b2:randomvar:expectation}{verwachting} dan en slechts dan als de familie
$\bigl(x\,\P(X = x)\bigr)_{x \in X(\Omega)}$ \omterm{def:b2:series:summable}{sommeerbaar} is, en
dan
\[
\E(X) = \sum_{x \in X(\Omega)} x\,\P(X = x) .
\]
Algemener, voor $f \colon X(\Omega) \to \R$ heeft de variabele
$f(X)$ een \omterm{def:b2:randomvar:expectation}{verwachting} iff $\sum_x \abs{f(x)}\,\P(X = x) <
\infty$, en dan $\E(f(X)) = \sum_x f(x)\,\P(X = x)$.
\end{theorem}

\begin{proof}
Partitioneer $\Omega$ in de niveausets $\Omega_x = \{X = x\}$, $x
\in X(\Omega)$. Door de sommatie-per-pakketten-stelling voor
\omterm{def:b2:series:summable}{sommeerbare families} (\cref{ch:b2:series}) is de familie
$(X(\omega)\P(\{\omega\}))_\omega$ \omterm{def:b2:series:summable}{sommeerbaar} iff elk pakket
dat is (automatisch: $\sum_{\omega \in \Omega_x}\abs{x}\P(\{\omega\})
= \abs x\,\P(X = x)$) \emph{en} de familie van paksommen
$\bigl(x\,\P(X = x)\bigr)_x$ \omterm{def:b2:series:summable}{sommeerbaar} is --- en dan stemmen
de totalsommen overeen. Voor $f(X)$: pas de bewezen uitspraak
toe op de variabele $Y = f \circ X$, wier niveausets $\{Y = y\}
= \bigsqcup_{x : f(x) = y}\{X = x\}$ zijn; een tweede sommatie
per pakketten zet $\sum_y y\,\P(Y = y)$ om in $\sum_x
f(x)\,\P(X = x)$, waarbij de pakketten nu de waarden $x$ groeperen
naar hun beeld $f(x)$, met absolute sommeerbaarheid van de ene
familie equivalent aan die van de andere.
\end{proof}

\begin{theorem}[Eigenschappen van de
verwachting]\label{thm:b2:randomvar:linearity}
Op de verzameling \omterm{def:b2:randomvar:law}{stochastische variabelen} met \omterm{def:b2:randomvar:expectation}{verwachting}:
\begin{enumerate}
\item (Lineariteit) $\E(aX + bY) = a\,\E(X) + b\,\E(Y)$.
\item (Positiviteit en monotoniciteit) $X \geq 0 \Rightarrow
\E(X) \geq 0$; $X \leq Y \Rightarrow \E(X) \leq \E(Y)$; en
$\abs{\E(X)} \leq \E(\abs X)$.
\item (Dominatie) Als $\abs X \leq Z$ en $Z$ een \omterm{def:b2:randomvar:expectation}{verwachting}
heeft, dan ook $X$.
\end{enumerate}
\end{theorem}

\begin{proof}
Alles zijn eigenschappen van sommen van \omterm{def:b2:series:summable}{sommeerbare families}
(\cref{ch:b2:series}): lineariteit van de som, positiviteit
term voor term, en het dominatiecriterium voor sommeerbaarheid.
(Merk op dat lineariteit onmiddellijk is op de
\emph{definitie} over $\Omega$, terwijl zij onhandig zou zijn
op de overdrachtsformule --- één voordeel van $\E$ stroomopwaarts
definiëren.)
\end{proof}

\begin{example}\label{ex:b2:randomvar:expectations}
$X \sim \mathcal{B}(n, p)$: schrijvend $X = X_1 + \dots + X_n$
als som van Bernoulli-indicatoren en lineariteit gebruikend,
$\E(X) = np$ --- geen binomiaalcoëfficiënten nodig. $X \sim
\mathcal{G}(p)$: $\E(X) = \sum_{k\geq1}k(1-p)^{k-1}p =
p\cdot\frac{1}{(1 - (1-p))^2} = \frac1p$, door differentiatie
van de geometrische reeks binnen haar schijf
(\cref{ch:b2:powerseries}). $X \sim \mathcal{P}(\lambda)$:
$\E(X) = \sum_{k\geq1}k e^{-\lambda}\frac{\lambda^k}{k!} =
\lambda e^{-\lambda}\sum_{j\geq0}\frac{\lambda^j}{j!} =
\lambda$.
\end{example}

\begin{example}[Overdracht in actie]\label{ex:b2:randomvar:transferex}
Voor $X \sim \mathcal P(\lambda)$, bereken
$\E\bigl(\frac1{1+X}\bigr)$ --- de \omterm{def:b2:randomvar:law}{wet} van $\frac1{1+X}$ zelf
is onhandig, maar overdracht vraagt er nooit om:
\[
\E\Bigl(\frac1{1+X}\Bigr)
= \sum_{k\geq0}\frac{1}{k+1}\,\eu^{-\lambda}
\frac{\lambda^k}{k!}
= \frac{\eu^{-\lambda}}{\lambda}\sum_{k\geq0}
\frac{\lambda^{k+1}}{(k+1)!}
= \frac{\eu^{-\lambda}}{\lambda}\bigl(\eu^\lambda - 1\bigr)
= \frac{1 - \eu^{-\lambda}}{\lambda} .
\]
Twee lessen. Computationeel: een verschoven exponentiële reeks
herkennen is het hele werk --- overdracht reduceert \omterm{def:b2:randomvar:expectation}{verwachtingen}
van $f(X)$ tot reeksmanipulatie. Structureel: de naïeve
inplugwaarde zou $\frac1{1 + \E X} = \frac1{1 + \lambda}$
zijn, en het ware antwoord is groter,
\[
\frac{1 - \eu^{-\lambda}}{\lambda} \geq
\frac{1}{1 + \lambda},
\]
precies zoals de ongelijkheid van Jensen voor de convexe functie
$t \mapsto \frac1{1+t}$ eist. \omterm{def:b2:randomvar:expectation}{Verwachtingen} van convexe beelden
zitten boven de naïeve inplugwaarde, en overdracht plus een
reekscontrole maakt de abstracte ongelijkheid concreet.
\end{example}

\begin{theorem}[Onafhankelijkheid en
producten]\label{thm:b2:randomvar:product}
\omterm{def:b2:randomvar:law}{Stochastische variabelen} $X, Y$ zijn \emph{\omterm{def:b2:proba:independence}{onafhankelijk}} als
$\P(X = x, Y = y) = \P(X = x)\P(Y = y)$ voor alle $x, y$ ---
equivalent, de \omterm{def:b2:proba:space}{gebeurtenissen} $\{X \in A\}$ en $\{Y \in B\}$
zijn \omterm{def:b2:proba:independence}{onafhankelijk} voor alle $A, B$. Als $X$ en $Y$
onafhankelijke reële variabelen met \omterm{def:b2:randomvar:expectation}{verwachting} zijn, dan heeft
$XY$ een \omterm{def:b2:randomvar:expectation}{verwachting} en
\[
\E(XY) = \E(X)\,\E(Y) .
\]
\end{theorem}

\begin{proof}
De equivalentie van de twee formuleringen volgt door de
puntsgewijze identiteit te sommeren over $(x, y) \in A \times
B$ ($\sigma$-additiviteit tweemaal). Voor het product: de
dubbele familie $\bigl(xy\,\P(X = x)\P(Y = y)\bigr)_{(x,y)}$ is
\omterm{def:b2:series:summable}{sommeerbaar}, omdat door Fubini voor families
(\cref{ch:b2:series})
\[
\sum_{x, y}\abs x \abs y\,\P(X{=}x)\P(Y{=}y)
= \Bigl(\sum_x \abs x \P(X{=}x)\Bigr)
\Bigl(\sum_y \abs y \P(Y{=}y)\Bigr) < \infty ;
\]
door onafhankelijkheid is deze familie precies $\bigl(xy\,\P(X
= x, Y = y)\bigr)$, waarvan de som $\E(XY)$ is door overdracht
toegepast op de variabele $(X, Y) \mapsto xy$; Fubini evalueert
opnieuw de ongetekende som als het product $\E(X)\E(Y)$.
\end{proof}

\begin{example}[Producten, met en zonder
onafhankelijkheid]\label{ex:b2:randomvar:productcontrast}
Gooi twee eerlijke dobbelstenen. Als $Y$ de tweede steen is
(\omterm{def:b2:proba:independence}{onafhankelijk} van de eerste), $\E(XY) = \E(X)\E(Y) = 3.5^2 =
12.25$. Als in plaats daarvan $Y = X$ (het ``product'' van een
steen met zichzelf),
\[
\E(X^2) = \frac{1 + 4 + 9 + 16 + 25 + 36}{6} = \frac{91}{6}
\approx 15.17 \neq 12.25 :
\]
zelfde marginale wetten in beide scenario's, verschillende
gezamenlijke wetten, verschillende productverwachtingen. De
moraal, het graveren waard: $\E(XY)$ is een functionaal van het
\emph{paar}, niet van de twee marginalen --- en de kloof
$\E(X^2) - \E(X)^2 \approx 2.92$ is, door König--Huygens,
precies de \omterm{def:b2:randomvar:variance}{variantie} $\frac{35}{12}$ van de steen.
\end{example}

\section{Variantie, covariantie en de klassieke
ongelijkheden}

\begin{definition}[Momenten, variantie]\label{def:b2:randomvar:variance}
$X$ heeft een \emph{moment van orde 2} als $X^2$ een
\omterm{def:b2:randomvar:expectation}{verwachting} heeft (dan ook $X$, door dominatie: $\abs X \leq
\frac{1 + X^2}{2}$). Haar \emph{variantie}\index{variantie} en
\emph{standaardafwijking}\index{standaardafwijking} zijn dan
\[
V(X) = \E\bigl((X - \E(X))^2\bigr)
= \E(X^2) - \E(X)^2 ,
\qquad
\sigma(X) = \sqrt{V(X)} ,
\]
(de tweede vorm --- de \emph{König--Huygens}-formule --- door
het kwadraat te ontwikkelen en lineariteit te gebruiken:
\[
\E\bigl((X - \E X)^2\bigr)
= \E\bigl(X^2 - 2X\,\E X + \E(X)^2\bigr)
= \E(X^2) - 2\,\E(X)^2 + \E(X)^2 ,
\]
de middelste term gebruikmakend van dat $\E X$ een constante
is). Voor $X, Y$ met tweede momenten is de
\emph{covariantie}\index{covariantie}
\[
\operatorname{Cov}(X, Y)
= \E\bigl((X - \E X)(Y - \E Y)\bigr)
= \E(XY) - \E(X)\E(Y) .
\]
\end{definition}

\begin{theorem}[Variantiegereedschap]\label{thm:b2:randomvar:variancerules}
Voor variabelen met tweede momenten:
\begin{enumerate}
\item $V(aX + b) = a^2\,V(X)$;
\item $V(X + Y) = V(X) + V(Y) + 2\operatorname{Cov}(X, Y)$, en
algemener
\[
V\Bigl(\sum_{i=1}^n X_i\Bigr)
= \sum_{i=1}^n V(X_i)
+ 2\sum_{i < j}\operatorname{Cov}(X_i, X_j) ;
\]
\item als $X, Y$ \omterm{def:b2:proba:independence}{onafhankelijk} zijn, $\operatorname{Cov}(X, Y)
= 0$ (het omgekeerde is onwaar), dus \omterm{def:b2:randomvar:variance}{varianties} van
onafhankelijke variabelen tellen op.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1} en \emph{2} zijn ontwikkelingen van kwadraten plus
lineariteit; de producten $X_iX_j$ hebben \omterm{def:b2:randomvar:expectation}{verwachtingen} door
Cauchy--Schwarz hieronder (of door $\abs{X_iX_j} \leq
\frac{X_i^2 + X_j^2}{2}$). \emph{3} is
\cref{thm:b2:randomvar:product} toegepast op de gecentreerde
variabelen. Een standaardtegenvoorbeeld van het omgekeerde: $X$
uniform op $\{-1, 0, 1\}$ en $Y = X^2$ zijn ongecorreleerd
($\E(XY) = \E(X^3) = 0 = \E X \cdot \E Y$) maar duidelijk
afhankelijk.
\end{proof}

\begin{theorem}[Ongelijkheden van Markov en
Chebyshev]\label{thm:b2:randomvar:markov}
\leavevmode
\begin{enumerate}
\item (Markov) Als $X \geq 0$ een \omterm{def:b2:randomvar:expectation}{verwachting} heeft, dan voor
elke $a > 0$:
\[
\P(X \geq a) \leq \frac{\E(X)}{a} .
\]
\item (Chebyshev) Als $X$ een tweede moment heeft, dan voor
elke $\varepsilon > 0$:
\[
\P\bigl(\abs{X - \E(X)} \geq \varepsilon\bigr)
\leq \frac{V(X)}{\varepsilon^2} .
\]
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Puntsgewijs, $a\,\mathbf{1}_{X \geq a} \leq X$ (op de
\omterm{def:b2:proba:space}{gebeurtenis} is de linkerzijde $a \leq X$; erbuiten, $0 \leq X$).
Neem \omterm{def:b2:randomvar:expectation}{verwachtingen}: $a\,\P(X \geq a) \leq \E(X)$ door
monotoniciteit en $\E(\mathbf{1}_A) = \P(A)$.
\emph{2.} Pas Markov toe op de niet-negatieve variabele $(X -
\E X)^2$ op niveau $a = \varepsilon^2$: de \omterm{def:b2:proba:space}{gebeurtenis} $\{(X -
\E X)^2 \geq \varepsilon^2\}$ is precies $\{\abs{X - \E X} \geq
\varepsilon\}$.
\end{proof}

\begin{example}[Ongecorreleerd maar aan elkaar
vastgeplakt]\label{ex:b2:randomvar:sumdiff}
Gooi twee eerlijke dobbelstenen, $X$ en $Y$ \omterm{def:b2:proba:independence}{onafhankelijk}, en
stel $S = X + Y$, $D = X - Y$. Door bilineariteit van de
\omterm{def:b2:randomvar:variance}{covariantie},
\[
\operatorname{Cov}(S, D) = V(X) - V(Y) +
\operatorname{Cov}(Y, X) - \operatorname{Cov}(X, Y) = V(X) -
V(Y) = 0 :
\]
som en verschil zijn ongecorreleerd. \omterm{def:b2:proba:independence}{Onafhankelijk}? Zeker
niet: $S = 12$ dwingt $D = 0$, terwijl $\P(D = 0) =
\frac16$ onvoorwaardelijk. Correlatie test alleen het
\emph{lineaire} deel van een afhankelijkheid; hier wordt de
afhankelijkheid gedragen door de constraint dat $S$ en $D$
dezelfde pariteit hebben, onzichtbaar voor \omterm{def:b2:randomvar:variance}{covariantie}. (Voor
dit paar had \omterm{def:b2:randomvar:variance}{nul-covariantie} $V(X) = V(Y)$ nodig: identieke
\omterm{def:b2:randomvar:law}{verdelingen}, niet onafhankelijkheid, deden het werk.)
\end{example}

\begin{example}[Wanneer Markov exact is]\label{ex:b2:randomvar:markovsharp}
De ongelijkheid van Markov is een gelijkheid precies wanneer
niets verspild wordt in de grens $a\,\mathbf 1_{X\geq a} \leq
X$: de variabele mag alleen de waarden $0$ en $a$ aannemen.
Concreet, als $\P(X = a) = \pi$ en $\P(X = 0) = 1 - \pi$, dan
$\E(X) = a\pi$ en
\[
\P(X \geq a) = \pi = \frac{\E(X)}{a} .
\]
Een realistische lezing: in een populatie waar gemiddelde
rijkdom $100$ is en rijkdom ofwel $0$ ofwel $10^6$, is het
aandeel miljonairs precies $10^{-4}$ --- de grens van Markov,
exact geraakt door maximale ongelijkheid. Wanneer $X$ over
tussenliggende waarden spreidt is de grens strikt, vaak wild;
maar zoals het extreme geval toont, kan geen betere
ongelijkheid uit het gemiddelde alleen gehaald worden.
\end{example}

\begin{example}[Chebyshev is scherp --- zonder verdere
hypothesen]\label{ex:b2:randomvar:chebsharp}
Fixeer $\varepsilon > 0$, $q \in \intoc01$, en laat $X$ de
waarden $\pm\varepsilon$ aannemen met waarschijnlijkheid
$\frac q2$ elk en $0$ met waarschijnlijkheid $1 - q$. Dan
$\E(X) = 0$, $V(X) = q\varepsilon^2$, en
\[
\P\bigl(\abs{X - \E X} \geq \varepsilon\bigr) = q
= \frac{V(X)}{\varepsilon^2} :
\]
gelijkheid in Chebyshev. Dus de ongelijkheid kan niet
verbeterd worden met alleen de \omterm{def:b2:randomvar:variance}{variantie} --- het
$1/\varepsilon^2$-verval is de exacte prijs van
tweede-momentinformatie. Sneller verval eist sterkere
hypothesen: begrensdheid van de variabele koopt
\emph{exponentiële} concentratie, zoals
\cref{exo:b2:randomvar:7} vooruitkijkt en het
weekendprobleem van dit hoofdstuk systematisch ontwikkelt.
\end{example}

\begin{theorem}[Cauchy--Schwarz en
Jensen]\label{thm:b2:randomvar:jensen}
\leavevmode
\begin{enumerate}
\item (Cauchy--Schwarz) Als $X, Y$ tweede momenten hebben, heeft
$XY$ een \omterm{def:b2:randomvar:expectation}{verwachting} en $\E(XY)^2 \leq \E(X^2)\,\E(Y^2)$;
dienvolgens $\operatorname{Cov}(X,Y)^2 \leq V(X)V(Y)$.
\item (Jensen) Als $\varphi \colon I \to \R$ \omterm{def:b2:affine:convex}{convex} is op een
interval dat $X(\Omega)$ bevat, en $X$, $\varphi(X)$
\omterm{def:b2:randomvar:expectation}{verwachtingen} hebben, dan
\[
\varphi\bigl(\E(X)\bigr) \leq \E\bigl(\varphi(X)\bigr) .
\]
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Sommeerbaarheid van $XY$: $\abs{XY} \leq \frac{X^2 +
Y^2}2$. De afbeelding $(X, Y) \mapsto \E(XY)$ is een positieve
\omterm{def:b2:quadratic:def}{symmetrische bilineaire vorm} op de ruimte van variabelen met
tweede momenten, dus de abstracte Cauchy--Schwarz-ongelijkheid
van \cref{ch:b2:quadratic} past toe (positief
\emph{semi}-definiet volstaat voor de ongelijkheid). Toepassing
op de gecentreerde variabelen geeft de covariantiegrens.

\emph{2.} Eerst ligt $m = \E(X)$ in $I$: $I$ is een interval
dat alle waarden van $X$ bevat, en \omterm{def:b2:randomvar:expectation}{verwachting} is monotoon, dus
$m$ ligt tussen $\inf X(\Omega)$ en $\sup X(\Omega)$. Door de
steunlijnstelling voor convexe functies
(\cref{ch:b2:realfun}) bestaan $\alpha, \beta$ met
$\varphi(t) \geq \alpha t + \beta$ voor alle $t \in I$ en
$\varphi(m) = \alpha m + \beta$. Dan, puntsgewijs op $\Omega$,
$\varphi(X) \geq \alpha X + \beta$; \omterm{def:b2:randomvar:expectation}{verwachtingen} nemend,
\[
\E\bigl(\varphi(X)\bigr) \geq \alpha\,\E(X) + \beta
= \varphi\bigl(\E(X)\bigr). \qedhere
\]
\end{proof}

\begin{example}
Jensen met $\varphi(t) = t^2$ geeft $\E(X)^2 \leq \E(X^2)$ ---
positiviteit van de \omterm{def:b2:randomvar:variance}{variantie}; met $\varphi(t) = 1/t$ op
$\intoo{0}{\infty}$: $\frac{1}{\E X} \leq
\E\bigl(\frac1X\bigr)$ --- het harmonisch gemiddelde ligt onder
het rekenkundig gemiddelde, nu in stochastische vorm.
\end{example}

\begin{remark}[Gangbare valkuilen]
(i) $\E(XY) = \E(X)\E(Y)$ \emph{eist} onafhankelijkheid (of
minstens \omterm{def:b2:randomvar:variance}{nul-covariantie}): $Y = X$ nemen geeft $\E(X^2) \neq
\E(X)^2$ wanneer $V(X) > 0$. (ii) Evenzo $V(X + X) = 4V(X)$,
niet $2V(X)$: \omterm{def:b2:randomvar:variance}{varianties} tellen alleen op over onafhankelijke
(of ongecorreleerde) sommanden. (iii) $\E(f(X))$ is niet
$f(\E(X))$; voor convexe $f$ vertelt Jensen u zelfs de richting
van de fout, zoals in \cref{ex:b2:randomvar:transferex}. (iv)
Bestaan is een echte hypothese: voor de Sint-Petersburgvariabele
$X = 2^K$ met $\P(K = k) = 2^{-k}$ ($k \geq 1$),
\[
\sum_{k\geq1}2^k\cdot2^{-k} = \sum_{k\geq1}1 = \infty :
\]
$X$ is bijna zeker eindig maar heeft geen \omterm{def:b2:randomvar:expectation}{verwachting}, en er
bestaat geen eerlijke instapprijs voor het spel.
Sommeerbaarheid in de definitie van $\E$ is geen
boekhoudkundige pedanterie --- het is waar zware staarten
gedetecteerd worden. (v) Tenslotte eist de
overdrachtsstelling \emph{absolute} sommeerbaarheid voordat
enige herschikking van de som over waarden legitiem is
(\cref{ch:b2:series}).
\end{remark}


\begin{example}[Chebyshev op honderd
worpen]\label{ex:b2:randomvar:hundredtosses}
Voor $X \sim \mathcal B(100, \frac12)$: $\E X = 50$, $V(X) =
25$. Chebyshev met $\varepsilon = 6$:
\[
\P(45 \leq X \leq 55) = \P(\abs{X - 50} < 6)
\geq 1 - \frac{25}{36} \approx 0.31 ,
\]
terwijl de exacte binomiale som $\approx 0.73$ geeft. De
gegarandeerde $31\%$ ligt ver van de waarheid, maar eiste
\emph{alleen} het gemiddelde en de \omterm{def:b2:randomvar:variance}{variantie} --- hetzelfde
certificaat past letterlijk toe op elke variabele met $\E =
50$, $V = 25$, hoe exotisch ook, en
\cref{ex:b2:randomvar:chebsharp} toont dat enige zulke
variabele haar verzadigt. Universaliteit heeft een prijs;
wanneer de \omterm{def:b2:randomvar:law}{verdeling} werkelijk binomiaal is, dichten de
exponentiële gereedschappen van het weekendprobleem het
grootste deel van de kloof.
\end{example}

\begin{example}[De correlatie van een deel met zijn
geheel]\label{ex:b2:randomvar:partwhole}
Voor onafhankelijke identiek verdeelde $X, Y$ met \omterm{def:b2:randomvar:variance}{variantie}
$\sigma^2 > 0$, hoe gecorreleerd is één sommand met de som $S
= X + Y$? Bereken
\[
\operatorname{Cov}(X, S) = \operatorname{Cov}(X, X) +
\operatorname{Cov}(X, Y) = \sigma^2 + 0 = \sigma^2,
\qquad V(S) = 2\sigma^2,
\]
dus de correlatiecoëfficiënt is
\[
\rho(X, S) = \frac{\operatorname{Cov}(X,
S)}{\sigma(X)\,\sigma(S)}
= \frac{\sigma^2}{\sigma\cdot\sigma\sqrt2}
= \frac{1}{\sqrt2} \approx 0.707 ,
\]
ongeacht de gemeenschappelijke \omterm{def:b2:randomvar:law}{wet} --- dobbelstenen, munten,
Poisson-tellingen. Met $n$ sommanden geeft dezelfde
berekening $\rho(X_1, S_n) = 1/\sqrt n$: de invloed van elke
individuele term op het totaal verdunt als een
vierkantswortel, wat de correlationele schaduw is van de
$\sqrt n$-schaal van fluctuaties. Cauchy--Schwarz garandeert
$\abs\rho \leq 1$ altijd; hier wordt de grens exact geraakt in
het ontaarde geval $n = 1$ en daalt voorspelbaar erna.
\end{example}

\begin{example}[Gewogen AM--GM uit
Jensen]\label{ex:b2:randomvar:amgm}
Laat $Y$ de positieve waarden $a_1, \dots, a_k$ aannemen met
waarschijnlijkheden $\lambda_1, \dots, \lambda_k$. De functie
$-\ln$ is \omterm{def:b2:affine:convex}{convex} op $\intoo0\infty$, dus Jensen geeft
$-\ln\E(Y) \leq \E(-\ln Y)$, d.w.z.
\[
a_1^{\lambda_1}a_2^{\lambda_2}\cdots a_k^{\lambda_k}
\;\leq\; \lambda_1a_1 + \lambda_2a_2 + \dots + \lambda_ka_k :
\]
de gewogen rekenkundig--meetkundige ongelijkheid, met
gelijkheid iff $Y$ constant is. Gelijke gewichten $\lambda_i =
\frac1k$ herstellen de klassieke AM--GM. Kansrekening heeft
stilzwijgend een zuiver algebraïsche stelling bewezen: een
waarschijnlijkheidswet kiezen is slechts een boekhoudkundig
hulpmiddel voor convexe combinaties --- het barycentrische
standpunt van \cref{ch:b2:affine} opnieuw, nu met Jensen als
motor.
\end{example}

\section{De zwakke wet van de grote aantallen}

\begin{theorem}[Zwakke wet van de grote
aantallen]\label{thm:b2:randomvar:lln}
Zij $(X_k)_{k \geq 1}$ paarsgewijs onafhankelijke
\omterm{def:b2:randomvar:law}{stochastische variabelen} met dezelfde \omterm{def:b2:randomvar:law}{wet}, met een tweede
moment; schrijf $m = \E(X_1)$ en $S_n = X_1 + \dots + X_n$.
Dan voor elke $\varepsilon > 0$:
\[
\P\Bigl(\,\Bigl|\frac{S_n}{n} - m\Bigr| \geq \varepsilon\Bigr)
\;\leq\; \frac{V(X_1)}{n\,\varepsilon^2}
\xrightarrow[n \to \infty]{} 0 .
\]
\end{theorem}

\begin{proof}
Door lineariteit $\E(S_n/n) = m$; door
\cref{thm:b2:randomvar:variancerules} (paarsgewijze
onafhankelijkheid doodt de \omterm{def:b2:randomvar:variance}{covarianties}) $V(S_n) =
n\,V(X_1)$, dus $V(S_n/n) = V(X_1)/n$. De ongelijkheid van
Chebyshev toegepast op $S_n/n$ geeft de grens.
\end{proof}

\begin{remark}
Dit is de stelling die waarschijnlijkheid met frequentie
verbindt: voor $X_k$ de indicator van een \omterm{def:b2:proba:space}{gebeurtenis} $A$ in
onafhankelijke herhalingen is $S_n/n$ de waargenomen
frequentie van $A$, en de wet van de grote aantallen zegt dat
zij rond $\P(A)$ concentreert met tempo
$\frac{p(1-p)}{n\varepsilon^2}$. De \emph{sterke} \omterm{def:b2:randomvar:law}{wet} ($S_n/n
\to m$ bijna zeker) is een Bachelor jaar~3-stelling --- haar
bewijs voor vierde momenten is echter binnen bereik: zie
\cref{exo:b2:randomvar:9}, die Borel--Cantelli op de
Chebyshev-achtige grens draait. Dezelfde Chebyshev-schatting
dreef het Bernstein-polynoom-bewijs van de
Weierstrass-benaderingsstelling in \cref{ch:b2:funcseq} --- het
tellemma daar \emph{was} de zwakke wet van de grote aantallen
in vermomming.
\end{remark}

\begin{example}[Vijftig coupons verzamelen]
\label{ex:b2:randomvar:couponnumeric}
De couponverzamelaar van \cref{exo:b2:randomvar:3} met $n =
50$ verschillende speeltjes: de verwachte total is
\[
\E(T_{50}) = 50\,H_{50} = 50\sum_{k=1}^{50}\frac1k
\approx 50 \times 4.499 \approx 225
\]
dozen --- vier en een half keer de naïeve gok $50$. De
harmonische groei is het hele verhaal: de eerste $25$
speeltjes arriveren in ongeveer $50\ln2 \approx 35$ dozen,
terwijl het \emph{laatste} speeltje alleen al gemiddeld $50$
dozen kost (een geometrische wacht van parameter $\frac1{50}$).
Voltooiingsproblemen worden gedomineerd door hun eindspel,
waarom \cref{exo:b2:randomvar:12} fluctuaties van orde $n$
vindt --- de grootte van die finale geometrische wacht ---
rond het gemiddelde $n\ln n$.
\end{example}

\begin{example}[Hoe groot moet $n$
zijn?]\label{ex:b2:randomvar:llnnumeric}
Om de waargenomen frequentie binnen $\varepsilon = 0.01$ van
$\P(A)$ te pinnen met vertrouwen $95\%$, eist de grens van
Chebyshev
\[
\frac{p(1-p)}{n\varepsilon^2} \leq \frac{1}{4n\varepsilon^2}
\leq 0.05,
\qquad\text{d.w.z.}\qquad
n \geq \frac{1}{4\cdot0.05\cdot(0.01)^2} = 50\,000 .
\]
De afhankelijkheid is brutaal in $\varepsilon$ (kwadratisch) en
mild in het vertrouwen (lineair in $1/\alpha$). Beide kenmerken
zijn eigenschappen van de \emph{grens}, niet van de waarheid:
de exponentiële ongelijkheden van het weekendprobleem verlagen
de prijs van vertrouwen van $1/\alpha$ naar $\ln(1/\alpha)$ ---
dezelfde specificatie zal daar ongeveer $18\,500$ steekproeven
kosten --- terwijl de $1/\varepsilon^2$-schaal echt en
onverbeterbaar is. \omterm{def:b2:randomvar:law}{Weten} welk deel van een grens los zit is
even nuttig als de grens zelf.
\end{example}

Voorwaarts voedt alles hier \cref{ch:b2:genfun}: de
\omterm{def:b2:randomvar:expectation}{verwachting} $\E(t^X)$ van één listige functie van $X$ pakt de
hele \omterm{def:b2:randomvar:law}{wet} in een machtreeks, momenten worden afgeleiden in $1$,
en Wald-achtige identiteiten voor stochastische sommen dragen
de vertakkingsprocentheorie; de productstelling voor
onafhankelijke variabelen wordt multiplicativiteit van
\omterm{ex:b2:powerseries:fibonacci}{genererende functies}. Achterwaarts is \omterm{def:b2:randomvar:expectation}{verwachting} een

$\eu^{tX}$ --- één ongelijkheid, opgewaardeerd door één goede
variabelesubstitutie, over drie hoofdstukken.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b2:randomvar:1}
Bereken $\E(X)$ en $V(X)$ voor $X \sim \mathcal{B}(n, p)$ (via
indicatoren), $X \sim \mathcal{P}(\lambda)$ (toon $V(X) =
\lambda$), en $X \sim \mathcal{G}(p)$ (toon $V(X) =
\frac{1-p}{p^2}$; gebruik $\E(X(X-1))$ en de tweede afgeleide
van de geometrische reeks).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:randomvar:2}
Zij $X \sim \mathcal{P}(\lambda)$ en $Y \sim \mathcal{P}(\mu)$
\omterm{def:b2:proba:independence}{onafhankelijk}. Toon dat $X + Y \sim \mathcal{P}(\lambda +
\mu)$ (convolutie van de gewichten; binomium van Newton), en
dat de voorwaardelijke \omterm{def:b2:randomvar:law}{wet} van $X$ gegeven $X + Y = n$
binomiaal $\mathcal{B}\bigl(n,
\frac{\lambda}{\lambda + \mu}\bigr)$ is.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:randomvar:3}
(Couponverzamelaar, \omterm{def:b2:randomvar:expectation}{verwachting}) Een ontbijtgranenmerk verbergt
één van $n$ verschillende speeltjes, uniform, in elke doos.
Laat $T_n$ het aantal dozen zijn dat nodig is om alle $n$
speeltjes te verzamelen. Schrijvend $T_n$ als som van
onafhankelijke geometrische variabelen (tijd tot een
\emph{nieuw} speeltje wanneer er nog $k$ ontbreken), toon
\[
\E(T_n) = n\sum_{k=1}^{n}\frac{1}{k} \sim n\ln n
\]
(equivalent via de reeks--integraalvergelijking van
\cref{ch:b2:comparison}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:randomvar:4}
Zij $X \geq 0$ geheelwaardig. Bewijs de \emph{staartformule}
\[
\E(X) = \sum_{n=1}^{\infty} \P(X \geq n)
\]
(wanneer één van beide zijden eindig is), door $X =
\sum_{n\geq1}\mathbf{1}_{X \geq n}$ te schrijven en sommaties
te verwisselen (Fubini voor niet-negatieve families). Herstel
$\E(X) = \frac1p$ voor de geometrische \omterm{def:b2:randomvar:law}{wet}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:randomvar:5}
(Steekproef zonder teruglegging is meer geconcentreerd) Een
urn heeft $N$ ballen, waarvan $M$ wit. Trek $n \leq N$ zonder
teruglegging en laat $X$ de witten tellen
(\emph{hypergeometrische} \omterm{def:b2:randomvar:law}{wet}). Gebruik makend van
indicatoren $X = \sum_{i=1}^n Y_i$ met $Y_i$ de $i$-de
trekking: toon dat elke $Y_i$ Bernoulli is van parameter $p =
M/N$ (symmetrie!), concludeer $\E(X) = np$ precies als met
teruglegging, en toon $\operatorname{Cov}(Y_i, Y_j) =
-\frac{p(1-p)}{N-1} < 0$ voor $i \neq j$, dus $V(X) =
np(1-p)\frac{N - n}{N - 1} \leq np(1-p)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:randomvar:6}
Zij $X$ een tweede moment hebben. Toon dat $c \mapsto
\E\bigl((X - c)^2\bigr)$ minimaal is precies in $c = \E(X)$,
met minimum $V(X)$. Toon dan dat $\P(X = \E(X)) = 1$ dan en
slechts dan als $V(X) = 0$. \emph{(Voor het tweede punt: als
$V(X) = 0$, gebruik Chebyshev met $\varepsilon = 1/n$ en
monotone continuïteit, \cref{thm:b2:proba:continuity}.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:randomvar:7}
(Concentratie verslaat Markov) Zij $S_n \sim \mathcal{B}(n,
\frac12)$ (aantal kop in $n$ eerlijke worpen). Vergelijk de
grenzen gegeven door Markov ($\P(S_n \geq \frac{3n}{4})$),
door Chebyshev, en door de exponentiële (Chernoff-)methode:
\[
\P\Bigl(S_n \geq \frac{3n}4\Bigr)
\leq \E\bigl(e^{tS_n}\bigr)e^{-3nt/4}
= \Bigl(\frac{1 + e^t}{2}\Bigr)^n e^{-3nt/4}
\quad (t > 0),
\]
en optimaliseer $t$ om een exponentieel kleine grens te
verkrijgen. \emph{(Bij $t = \ln 3$: grens
$\bigl(2\cdot 3^{-3/4}\bigr)^n \approx (0.877)^n$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:randomvar:8}
(Weierstrass opnieuw, probabilistisch) Zij $f \colon [0,1]
\to \R$ continu en $S_n \sim \mathcal{B}(n, x)$. Toon dat het
Bernstein-polynoom $B_nf(x) = \sum_{k=0}^n f\bigl(\frac
kn\bigr)\binom nk x^k(1-x)^{n-k}$ gelijk is aan
$\E\bigl[f\bigl(\frac{S_n}{n}\bigr)\bigr]$, en herleid de
schatting $\abs{B_nf(x) - f(x)} \leq \omega_f(\delta) +
\frac{2\norm f_\infty}{4n\delta^2}$ van \cref{ch:b2:funcseq}
in deze probabilistische taal (splits op
$\bigl|\frac{S_n}{n} - x\bigr| \geq \delta$ en gebruik
Chebyshev).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:randomvar:9}
(Sterke \omterm{def:b2:randomvar:law}{wet} onder vierde momenten) Zij $(X_k)$ \omterm{def:b2:proba:independence}{onafhankelijk},
identiek verdeeld, gecentreerd ($\E X_1 = 0$), met $\E(X_1^4)
< \infty$. Ontwikkel $\E(S_n^4)$ en tel de overlevende termen
(alleen $\E(X_i^4)$- en $\E(X_i^2X_j^2)$-termen, $i \neq j$),
toon $\E(S_n^4) \leq C n^2$ voor een constante $C$. Leid af
$\sum_n \P\bigl(\abs{S_n/n} \geq \varepsilon\bigr) < \infty$
voor elke $\varepsilon > 0$ (Markov van orde 4) en concludeer
met Borel--Cantelli (\cref{thm:b2:proba:borelcantelli}) dat
$S_n/n \to 0$ bijna zeker langs een geschikte formulering: de
\omterm{def:b2:proba:space}{gebeurtenis} $\bigcap_{j}\bigcup_N\bigcap_{n \geq
N}\{\abs{S_n/n} < \frac1j\}$ heeft waarschijnlijkheid $1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:randomvar:10}
Twee eerlijke dobbelstenen worden gegooid; laat $M$ de grotere
van de twee resultaten zijn. Gebruik de staartformule van
\cref{exo:b2:randomvar:4} (eindige versie) om te tonen
\[
\E(M) = \sum_{k=1}^{6}\P(M \geq k)
= 6 - \sum_{j=0}^5\Bigl(\frac j6\Bigr)^2 = \frac{161}{36}
\approx 4.47 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:randomvar:11}
Laat $F_n$ het aantal vaste punten zijn van een uniform
willekeurige permutatie van $\{1, \dots, n\}$ ($n \geq 2$).
Schrijvend $F_n = \sum_i\mathbf 1_{\sigma(i) = i}$, bereken
$\E(F_n) = 1$, $\operatorname{Cov}(\mathbf 1_{\sigma(i)=i},
\mathbf 1_{\sigma(j)=j}) = \frac1{n^2(n-1)}$ voor $i \neq j$,
en concludeer $V(F_n) = 1$: gemiddeld is één letter vast, met
\omterm{def:b2:randomvar:variance}{variantie} precies $1$, ongeacht $n$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:randomvar:12}
(Couponverzamelaar, concentratie) In de setting van
\cref{exo:b2:randomvar:3}, toon
\[
V(T_n) = \sum_{k=1}^n\frac{1 - k/n}{(k/n)^2}
\leq n^2\sum_{k=1}^n\frac{1}{k^2} \leq \frac{\pi^2}{6}n^2,
\]
met onafhankelijkheid van de geometrische stadia en
$V(\mathcal G(p)) = \frac{1-p}{p^2}$
(\cref{exo:b2:randomvar:1}; de waarde $\pi^2/6$ is
\cref{ex:b2:fourier:basel}). Leid met Chebyshev af dat
$\dfrac{T_n}{n\ln n} \to 1$ \emph{in waarschijnlijkheid}: de
totale tijd van de verzamelaar is $n\ln n$ op
fluctuaties van orde $n$ na.
\end{exercise}

\section{Probleem: de concentratiegereedschapskist, van
Markov tot Hoeffding}

\begin{problem}[{Weekendprobleem --- exponentiële
concentratie met de hand, en hoeveel mensen een peiling moet
vragen}]
\label{pb:b2:randomvar:1}
De ongelijkheid van Markov kost één moment en koopt een
$1/a$-verval; Chebyshev kost twee momenten en koopt
$1/\varepsilon^2$ --- en \cref{ex:b2:randomvar:chebsharp}
toont dat dat alles is wat die momenten kunnen kopen. Dit
probleem klimt de rest van de ladder: de exponentiële
(Chernoff-)methode met haar \emph{exacte} tempo voor
muntworpen, de ongelijkheid van Hoeffding voor alle begrensde
variabelen, en de beloning --- expliciete, eerlijke
steekproefgroottes voor peilingen, verkiezingsuitspraken en
muntentesten.
\index{concentratieongelijkheden}\index{ongelijkheid van
Hoeffding}\index{Chernoff-grens} Doorheen is $S_n \sim
\mathcal B(n, p)$ een som van $n$ onafhankelijke
Bernoulli-variabelen en $\widehat p_n = S_n/n$ de empirische
frequentie.

\medskip
\noindent\textbf{Deel I --- Calibratie op de eerlijke munt.}
Hier $p = \frac12$ en $a \in \intoo{\frac12}{1}$.
\begin{enumerate}
  \item Markov op niveau $an$: toon $\P(S_n \geq an) \leq
        \frac1{2a}$, een grens die niet eens naar $0$ neigt.
        Waar verliest Markov zoveel?
  \item Chebyshev: gebruik makend van de symmetrie van de
        eerlijke binomiaal om $n/2$, toon
        \[
        \P(S_n \geq an) = \tfrac12\,
        \P\bigl(\abs{S_n - \tfrac n2} \geq n(a -
        \tfrac12)\bigr)
        \leq \frac{1}{8n(a - 1/2)^2},
        \]
        d.w.z.\ $\frac2n$ bij $a = \frac34$: eindelijk
        polynomiaal verval.
  \item (Chernoff, algemeen niveau) Bereken $\E(\eu^{tS_n}) =
        \bigl(\frac{1 + \eu^t}2\bigr)^n$ en optimaliseer
        $\P(S_n \geq an) \leq \E(\eu^{tS_n})\eu^{-tan}$ over
        $t > 0$: toon dat de optimale $t$ $\ln\frac{a}{1-a}$
        is en
        \[
        \P(S_n \geq an) \leq \eu^{-n\,I(a)},
        \qquad
        I(a) = \ln 2 + a\ln a + (1-a)\ln(1-a) > 0 .
        \]
        Controleer dat $a = \frac34$ de grens
        $\bigl(2\cdot3^{-3/4}\bigr)^n$ van
        \cref{exo:b2:randomvar:7} herstelt.
  \item (De exponent is exact) Laat $k = an$ een geheel getal
        zijn. Uit het feit dat $\binom nk a^k(1-a)^{n-k}$ de
        grootste is van de $n + 1$ termen van een
        waarschijnlijkheidsverdeling, bewijs $\binom nk \geq
        \frac{\eu^{nH(a)}}{n+1}$ met $H(a) = -a\ln a -
        (1-a)\ln(1-a)$, en leid de passende ondergrens af
        \[
        \P(S_n \geq an) \geq \binom{n}{an}2^{-n}
        \geq \frac{\eu^{-n\,I(a)}}{n + 1} .
        \]
  \item Tabuleer de drie grenzen bij $n = 100$, $a =
        \frac34$: Markov $\frac23$, Chebyshev $0.02$,
        Chernoff $\approx 2.1\cdot10^{-6}$ (de ware waarde is
        $\approx 2.8\cdot10^{-7}$). Moraal, in één zin?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De ongelijkheid van Hoeffding.}
\begin{enumerate}[resume]
  \item (Rademacher-geval) Voor $\varepsilon = \pm1$ met
        waarschijnlijkheid $\frac12$ elk, bewijs
        \[
        \E(\eu^{t\varepsilon}) = \cosh t \leq \eu^{t^2/2}
        \qquad (t \in \R)
        \]
        door de twee reeksen term voor term te vergelijken
        ($(2k)! \geq 2^kk!$).
  \item Leid af, voor onafhankelijke Rademacher-variabelen
        $\varepsilon_1, \dots, \varepsilon_n$ en elke $s >
        0$:
        \[
        \P\Bigl(\sum_{i=1}^n\varepsilon_i \geq s\Bigr)
        \leq \eu^{-s^2/(2n)} .
        \]
  \item Vertaal naar eerlijke munten ($X_i =
        \frac{1+\varepsilon_i}2$): $\P\bigl(\widehat p_n -
        \tfrac12 \geq \delta\bigr) \leq \eu^{-2n\delta^2}$,
        en de tweezijdige versie met een factor $2$.
  \item (Lemma van Hoeffding) Zij $X \in \intcc01$ met $\E X
        = p$, en $\psi(t) = \ln\E(\eu^{tX})$. Rechtvaardig
        dat $\psi$ tweemaal \omterm{def:b2:diffcalc:differential}{differentieerbaar} is met
        \[
        \psi''(t) = \E_t(X^2) - \E_t(X)^2, \qquad
        \E_t(Y) := \frac{\E(Y\eu^{tX})}{\E(\eu^{tX})},
        \]
        een \emph{\omterm{def:b2:randomvar:variance}{variantie}} van een herwogen variabele die
        nog waarden in $\intcc01$ aanneemt; begrens haar
        door $\frac14$ (minimaliteitsargument van
        \cref{exo:b2:randomvar:6}) en besluit door Taylor:
        \[
        \E\bigl(\eu^{t(X - p)}\bigr) \leq \eu^{t^2/8} .
        \]
  \item (Ongelijkheid van Hoeffding) Voor onafhankelijke $X_i
        \in \intcc01$ met gemeenschappelijk gemiddelde $p$,
        leid af
        \[
        \P\bigl(\abs{\widehat p_n - p} \geq \delta\bigr)
        \leq 2\,\eu^{-2n\delta^2}
        \qquad (\delta > 0).
        \]
  \item Vergelijk het Chebyshev-tempo
        $\frac{p(1-p)}{n\delta^2}$ met dat van Hoeffding
        $2\eu^{-2n\delta^2}$: welke hypothese eist elk, en
        vanaf welke $n$ (ongeveer) wint de exponentiële
        grens bij $\delta = 0.03$, $p = \frac12$?
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Hoeveel mensen moet een
peiling vragen?} Een peiling vraagt $n$ onafhankelijke,
uniform gekozen kiezers; elk antwoord is eerlijk; $p$ is de
ware score, $\widehat p_n$ het peilingscijfer.
\begin{enumerate}[resume]
  \item Toon dat de peiling accuraat is tot $\pm\delta$ met
        vertrouwen $1 - \alpha$ (d.w.z.\
        $\P(\abs{\widehat p_n - p} \geq \delta) \leq
        \alpha$) zodra
        \[
        n \;\geq\; \frac{\ln(2/\alpha)}{2\,\delta^2} .
        \]
  \item Bereken de vereiste $n$ voor de standaard
        ``drie punten, vijfennegentig procent''-specificatie
        ($\delta = 0.03$, $\alpha = 0.05$): $n \geq 2050$;
        en voor één punt: $n \geq 18\,445$. Observeer --- en
        leg uit --- het opvallende feit dat het antwoord de
        populatiegrootte niet bevat.
  \item Doe vraag 13 over met Chebyshev ($V(X_1) = p(1-p)
        \leq \frac14$): $n \geq \frac1{4\alpha\delta^2} =
        5556$ bij drie punten. Merk op dat steekproef
        \emph{zonder} teruglegging alleen helpt
        (\cref{exo:b2:randomvar:5}: de \omterm{def:b2:randomvar:variance}{variantie} krimpt met
        $\frac{N-n}{N-1}$).
  \item (Een verkiezing uitroepen) De ware score van een
        kandidaat is $p = 0.52$. Hoeveel kiezers moeten
        gepolld worden zodat $\P(\widehat p_n \leq
        \tfrac12) \leq 0.01$? Toon $n \geq
        \frac{\ln 100}{2\cdot(0.02)^2} \approx 5757$ ---
        een nipte race uitroepen kost veel meer dan een
        score schatten.
  \item Wat de wiskunde \emph{niet} dekt: noem de
        modelleringsaannames gebruikt (onafhankelijke
        uniforme steekproef, eerlijke antwoorden, vaste
        $p$), en leg in een korte alinea uit waarom echte
        peilingsfouten gedomineerd worden door
        \emph{vertekening} (niet-uniforme steekproef,
        non-respons), die geen verhoging van $n$ reduceert.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Scherper en goedkoper.}
\begin{enumerate}[resume]
  \item (Mediaan van gemiddelden: exponentieel verval uit
        twee momenten) Splits een budget van $km$
        steekproeven in $k$ onafhankelijke groepen van $m$;
        laat $\widehat p^{(1)}, \dots, \widehat p^{(k)}$ de
        groepsgemiddelden zijn en $M$ hun mediaan. Kies $m$
        zodat elke groep voldoet aan
        $\P(\abs{\widehat p^{(i)} - p} \geq \delta) \leq
        \frac18$ (Chebyshev: $m \geq \frac2{\delta^2}$
        volstaat). Toon dat als $\abs{M - p} \geq \delta$
        dan minstens $k/2$ groepen fout gaan, en leid af
        \[
        \P(\abs{M - p} \geq \delta)
        \leq \binom{k}{\lceil k/2\rceil}\Bigl(\frac18
        \Bigr)^{k/2}
        \leq 2^k\cdot 8^{-k/2} = 2^{-k/2} :
        \]
        exponentiële concentratie met niets buiten
        \omterm{def:b2:randomvar:variance}{varianties}.
  \item (Paley--Zygmund) Voor $X \geq 0$ met een tweede
        moment, bewijs $\P(X > 0) \geq
        \dfrac{\E(X)^2}{\E(X^2)}$ \emph{(Cauchy--Schwarz op
        $X\mathbf 1_{X>0}$)}: het
        omgekeerde-richtingsgereedschap --- momenten kunnen
        ook \omterm{def:b2:proba:space}{gebeurtenissen} forceren te gebeuren.
  \item (Pinsker-lite) Toon $I(a) \geq 2\bigl(a -
        \tfrac12\bigr)^2$ op $\intoo{\frac12}1$ \emph{(het
        verschil verdwijnt tot tweede orde in $\frac12$ en
        zijn tweede afgeleide is $\frac1{a(1-a)} - 4 \geq
        0$)}: de exacte exponent van Chernoff verslaat
        altijd de kwadratische van Hoeffding.
  \item Ontwikkel $I\bigl(\tfrac12 + \delta\bigr) =
        2\delta^2 + O(\delta^4)$ en combineer met vraag 4:
        voor kleine afwijkingen is de Hoeffding-exponent
        $2n\delta^2$ asymptotisch \emph{exact} --- geen
        methode kan haar met meer dan polynomiale factoren
        verslaan.
  \item Stel de gereedschapstabel op: voor Markov, Chebyshev,
        de vierde-momentgrens van \cref{exo:b2:randomvar:9},
        Hoeffding, en Chernoff met exponent $I$, noem in
        één regel elk: vereiste hypothese, verkregen verval,
        en de vraag in dit probleem waar zij het scherpst
        was.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Dividenden.}
\begin{enumerate}[resume]
  \item (Een munt testen) Een munt is ofwel eerlijk ofwel
        vertekend met $p = 0.55$. U gooit haar $n$ keer en
        verklaart ``vertekend'' wanneer $\widehat p_n >
        0.525$. Toon dat beide foutkansen ten hoogste
        $\eu^{-2n(0.025)^2}$ zijn, en dat $n \geq 3685$
        worpen beide onder $1\%$ garanderen.
  \item (Zeldzame \omterm{def:b2:proba:space}{gebeurtenissen} hebben een
        variantie-bewuste grens nodig) Laat $p = 0.01$ en
        neem de relatieve specificatie $\delta = p/2 =
        0.005$, $\alpha = 0.05$. Vergelijk de
        steekproefgroottes geëist door Hoeffding ($n
        \approx 74\,000$) en door Chebyshev met de ware
        \omterm{def:b2:randomvar:variance}{variantie} $p(1-p)$ ($n \approx 7920$): de
        variantie-blinde exponentiële grens verliest van
        het nederige tweede moment. Stel de moraal, en waar
        het ontbrekende gereedschap (een variantie-bewuste
        exponentiële grens; de Poisson-benadering van
        \cref{ch:b2:genfun}) vandaan zal komen.
  \item (Sterke \omterm{def:b2:randomvar:law}{wet} voor munten) Uit $\sum_n
        2\eu^{-2n\delta^2} < \infty$ en Borel--Cantelli
        (\cref{thm:b2:proba:borelcantelli}), bewijs dat
        $\widehat p_n \to p$ bijna zeker voor
        onafhankelijke muntworpen: formuleer de
        bijna-zekere \omterm{def:b2:proba:space}{gebeurtenis} als
        $\bigcap_j\bigcup_N\bigcap_{n\geq N}
        \{\abs{\widehat p_n - p} < \tfrac1j\}$ zoals in
        \cref{exo:b2:randomvar:9}, en concludeer.
        (Begrensdheid vervangt het vierde moment daar
        gebruikt.)
  \item Synthese. In vijf zinnen: wat elke sport van de
        ladder (momenten één, twee, vier; begrensde
        exponentieel; exacte exponent) kost en koopt;
        waarom peilen van $2050$ mensen volstaat voor een
        land van elke grootte; en welke van deze grenzen
        het Bachelor jaar~3-volume zal aanscherpen tot de
        exacte constanten van de centrale limietstelling.
\end{enumerate}
\end{problem}
