\chapter{Affiene ruimten}\label{ch:b2:affine}

Vectorruimten hebben een bevoorrecht punt --- de oorsprong --- dat
de meetkunde niet wil. Een \emph{\omterm{def:b2:affine:def}{affiene ruimte}} is een vectorruimte
die haar oorsprong is vergeten: punten en vectoren worden
verschillende soorten, verbonden door translatie. Dit korte
hoofdstuk bouwt het woordenboek (punten, barycentra, affiene
deelruimten en afbeeldingen), de affiene kijk op convexiteit, en de
classificatietools die in de volgende meetkundige hoofdstukken
worden gebruikt.

\section{Punten en vectoren}

\begin{definition}\label{def:b2:affine:def}
Een \emph{affiene ruimte}\index{affiene ruimte} gericht door een reële
vectorruimte $E$ is een niet-lege verzameling $\mathcal{E}$ met een
afbeelding $(A, B) \mapsto \vect{AB} \in E$ die voldoet aan
\[
\vect{AB} + \vect{BC} = \vect{AC}
\quad \text{(Chasles)},
\qquad
\text{voor elke } A,\ B \mapsto \vect{AB} \text{ is een bijectie }
\mathcal{E} \to E .
\]
Men schrijft $B = A + u$ voor het unieke punt met $\vect{AB} = u$.
De \emph{dimensie} van $\mathcal{E}$ is $\dim E$. Elke
vectorruimte is een affiene ruimte over zichzelf ($\vect{AB} = B -
A$); elke keuze van oorsprong $O \in \mathcal{E}$ identificeert
$\mathcal{E}$ met $E$ via $M \mapsto \vect{OM}$.
\end{definition}

\begin{example}[Een affiene ruimte zonder natuurlijke
oorsprong]\label{ex:b2:affine:planeexample}
Het oplossingsvlak $\mathcal E = \{(x, y, z) \in \R^3 : x + y
+ z = 1\}$ is geen vectordeelruimte ($0 \notin \mathcal E$),
maar wel een \omterm{def:b2:affine:def}{affiene ruimte} gericht door $E = \{x + y + z =
0\}$: voor $A, B \in \mathcal E$ landt het verschil $\vect{AB} = B
- A$ in $E$ (de sommen heffen elkaar op), Chasles is geërfd
van $\R^3$, en $B \mapsto \vect{AB}$ is bijectief op $E$.
Geen punt van $\mathcal E$ is onderscheiden --- elke keuze van
``oorsprong'' $O \in \mathcal E$ werkt even goed, en alle
identificaties $M \mapsto \vect{OM}$ verschillen door
translaties. Dit is de typische situatie: oplossingsverzamelingen
van inhomogene lineaire problemen (lineaire stelsels, lineaire
differentiaalvergelijkingen in \cref{ch:b2:diffeq}) zijn \omterm{def:b2:affine:subspace}{affien},
nooit lineair, en de slogan ``particuliere oplossing plus kern''
is precies de uitspraak $\mathcal F = A + F$ van de volgende
definitie.
\end{example}

\begin{definition}[Barycentrum]\label{def:b2:affine:barycenter}
Zij $(A_i, \lambda_i)_{i \leq k}$ gewogen punten met
$\sum\lambda_i \neq 0$. Het \emph{barycentrum}\index{barycentrum} $G
= \operatorname{bar}\bigl((A_i, \lambda_i)\bigr)$ is het unieke
punt met
\[
\sum_i \lambda_i\, \vect{GA_i} = 0
\qquad\text{equivalent}\qquad
\vect{OG} = \frac{1}{\sum\lambda_i}\sum_i \lambda_i\,\vect{OA_i}
\quad (\text{elke } O).
\]
Barycentra zijn \emph{associatief} (deelgroepen van punten mogen
worden vervangen door hun partiële barycentrum met het
samengenomen gewicht) en invariant onder herschaling van alle
gewichten.
\end{definition}

\begin{proof}[Bewijs van bestaan en de formules]
Fixeer $O$ en schrijf $s = \sum_i\lambda_i \neq 0$. Via Chasles,
\[
\sum_i\lambda_i\,\vect{GA_i} = 0
\iff \sum_i\lambda_i\bigl(\vect{GO} + \vect{OA_i}\bigr) = 0
\iff s\,\vect{OG} = \sum_i\lambda_i\,\vect{OA_i},
\]
wat $G = O + \frac1s\sum_i\lambda_i\vect{OA_i}$ uniek
bepaalt. \emph{Onafhankelijkheid van $O$:} voor een andere
oorsprong $O'$,
\[
\frac1s\sum_i\lambda_i\,\vect{O'A_i}
= \frac1s\sum_i\lambda_i\bigl(\vect{O'O} + \vect{OA_i}\bigr)
= \vect{O'O} + \frac1s\sum_i\lambda_i\,\vect{OA_i}
= \vect{O'G} :
\]
hetzelfde punt $G$. \emph{Associativiteit:} splits de
indexverzameling als $I \sqcup J$ met $s_I = \sum_{i\in
I}\lambda_i \neq 0$, en zij $G_I$ het \omterm{def:b2:affine:barycenter}{barycentrum} van $(A_i,
\lambda_i)_{i\in I}$, zodat $\sum_{i\in I}\lambda_i\vect{OA_i} =
s_I\,\vect{OG_I}$. Dan
\[
s\,\vect{OG}
= \sum_{i\in I}\lambda_i\vect{OA_i} +
\sum_{j\in J}\lambda_j\vect{OA_j}
= s_I\,\vect{OG_I} + \sum_{j\in J}\lambda_j\vect{OA_j} :
\]
$G$ is het \omterm{def:b2:affine:barycenter}{barycentrum} van $(G_I, s_I)$ samen met $(A_j,
\lambda_j)_{j\in J}$, zoals beweerd. \emph{Herschaling:}
vervanging van elke $\lambda_i$ door $t\lambda_i$ ($t \neq 0$)
vermenigvuldigt $s$ en de gewogen som met $t$, en laat
$\vect{OG}$ onveranderd.
\end{proof}

\begin{remark}[Klassieke valkuilen]
Twee vallen omringen de definitie. Ten eerste, als de gewichten
tot \emph{nul} sommeren, is er geen \omterm{def:b2:affine:barycenter}{barycentrum}: de afbeelding
$O \mapsto \sum\lambda_i\vect{OA_i}$ is dan onafhankelijk van
$O$ en definieert een \emph{vector}, geen punt --- bijvoorbeeld
encodeert $(A, -1;\ B, 1)$ de vector $\vect{AB}$. Bijhouden
welk van de twee objecten een berekening oplevert is de helft
van de barycentrische hygiëne. Ten tweede zijn gewichten alleen
betekenisvol op een gemeenschappelijke niet-nul factor na;
formules als ``de coördinaten van $G$ zijn $\lambda_1, \dots,
\lambda_k$'' veronderstellen een normalisatie (meestal
$\sum\lambda_i = 1$), en vergeten te normaliseren is de
standaardbron van foute verhoudingen op een figuur.
\end{remark}

\begin{definition}[Affiene deelruimten; affiene
afbeeldingen]\label{def:b2:affine:subspace}
Een \emph{affiene deelruimte} is een verzameling $\mathcal{F} =
A + F = \{A + u : u \in F\}$ met $F$ een vectordeelruimte (haar
\emph{richting}); equivalent, een niet-lege verzameling die
stabiel is onder barycentra. Affiene deelruimten van $\R^n$ zijn
precies de oplossingsverzamelingen van lineaire stelsels $MX =
B$ (Jaar~1: particuliere oplossing plus kern). Een afbeelding
$f \colon \mathcal{E} \to \mathcal{E}'$ is
\emph{affien}\index{affiene afbeelding} wanneer ze barycentra
behoudt --- equivalent wanneer
\[
f(A + u) = f(A) + \varphi(u)
\]
voor een (unieke) lineaire afbeelding $\varphi = \vec f$, het
\emph{lineaire deel}. Affiene afbeeldingen van $\R^n$: $X
\mapsto MX + C$. Samenstellingen zijn affien met
samengestelde lineaire delen; $f$ is bijectief dan en slechts
dan als $\vec f$ dat is.
\end{definition}

\begin{proof}[Bewijs van de equivalentie voor afbeeldingen]
Als $f(A + u) = f(A) + \varphi(u)$: voor een \omterm{def:b2:affine:barycenter}{barycentrum} $G$
van $(A_i, \lambda_i)$, door elk punt uit $A$ te ontwikkelen,
$f(G) = f(A) + \varphi(\vect{AG})$ en $\varphi(\vect{AG}) =
\frac{\sum\lambda_i\varphi(\vect{AA_i})}{\sum\lambda_i}$: $f(G)$
is het \omterm{def:b2:affine:barycenter}{barycentrum} van de beelden. Omgekeerd, fixeer $A$ en
definieer $\varphi(u) = \vect{f(A)\,f(A + u)}$.
\emph{Homogeniteit:} $A + tu = \operatorname{bar}\bigl(A,
1-t;\ A + u, t\bigr)$ voor elke reële $t$, dus behoud van
barycentra (met willekeurige reële gewichten, zoals
verondersteld) geeft direct $\varphi(tu) = t\,\varphi(u)$.
\emph{Additiviteit:} $A + u + v = \operatorname{bar}\bigl(A +
2u, \tfrac12;\ A + 2v, \tfrac12\bigr)$, dus $\varphi(u + v) =
\tfrac12\varphi(2u) + \tfrac12\varphi(2v) = \varphi(u) +
\varphi(v)$, via homogeniteit. Dus is $\varphi$ lineair.
\end{proof}

\begin{remark}
Het bewijs gebruikte barycentra met \emph{willekeurige reële}
gewichten: de homogeniteitsstap neemt $t$ buiten $\intcc01$.
Als een afbeelding slechts barycentra met niet-negatieve
gewichten behoudt --- equivalent midpunten en segmenten ---
komt lineariteit van de vectorafbeelding niet gratis: men
verkrijgt alleen $\Q$-lineariteit, en een continuïteitshypothese
is nodig om te concluderen, precies als in
\cref{exo:b2:affine:5}. Het onderscheid tussen ``behoudt alle
barycentra'' en ``behoudt convexe combinaties'' is een klein
maar reëel subtiliteit van de affiene woordenschat.
\end{remark}

\begin{example}[Klassieke barycentrische
meetkunde]\label{ex:b2:affine:median}
Het zwaartepunt van een driehoek $ABC$ is het \omterm{def:b2:affine:barycenter}{barycentrum} $G =
\operatorname{bar}(A,1; B,1; C,1)$. Associativiteit met het
midden $A' = \operatorname{bar}(B, 1; C, 1)$ toont
\[
G = \operatorname{bar}(A, 1;\ A', 2) :
\]
$G$ ligt op de zwaartelijn $AA'$ op twee derden ervan --- en
evenzo voor de andere twee zwaartelijnen: de drie
zwaartelijnen zijn concurrent, in één regel barycentrische
rekenkunde.
\end{example}

\begin{example}[De bimedianen van een
vierhoek]\label{ex:b2:affine:bimedians}
Zij $ABCD$ een willekeurige vierhoek (vlak of niet!) en
beschouw haar \emph{bimedianen}: de segmenten die de midpunten
van overstaande zijden verbinden, $M_{AB}M_{CD}$ en
$M_{BC}M_{DA}$. Introduceer het \omterm{def:b2:affine:barycenter}{barycentrum} $G$ van $(A,1;
B,1; C,1; D,1)$ en groepeer de gewichten op twee manieren:
\[
G = \operatorname{bar}\bigl(M_{AB}, 2;\ M_{CD}, 2\bigr)
= \operatorname{bar}\bigl(M_{BC}, 2;\ M_{DA}, 2\bigr) :
\]
$G$ is het midden van \emph{beide} bimedianen --- dus bisecteren
de twee bimedianen elkaar altijd, en de vierhoek van de vier
midpunten is een parallellogram (haar diagonalen zijn de
bimedianen). Geen gevalonderscheid, geen coördinaten, en het
argument overleeft onveranderd voor een scheve vierhoek in
$\R^3$, waar een bewijs op basis van een tekening al delicaat
zou zijn: associativiteit geeft niet om dimensie.
\end{example}

\begin{example}[Een affiene afbeelding classificeren, van begin
tot eind]\label{ex:b2:affine:dilation}
Zij $f(x, y) = (2x - 1,\ 3y - 4)$ op $\R^2$. Haar lineaire deel
is $\varphi = \operatorname{diag}(2, 3)$, waarvan het spectrum
$\{2, 3\}$ de $1$ vermijdt: door het criterium van vaste punten
hieronder (\cref{prop:b2:affine:fixedpoint}) heeft $f$ precies
één vast punt, gevonden door op te lossen
\[
x = 2x - 1, \qquad y = 3y - 4
\qquad\Longrightarrow\qquad \Omega = (1, 2).
\]
Hercentreren in $\Omega$ (stel $x = 1 + u$, $y = 2 + v$):
\[
f(1 + u,\ 2 + v) = (1 + 2u,\ 2 + 3v) :
\]
in het raam in $\Omega$ \emph{is} $f$ haar lineaire deel, een
anisotrope dilatie die met factor $2$ horizontaal en $3$
verticaal uitrekt vanuit het centrum $(1, 2)$. De algemene les:
een \omterm{def:b2:affine:subspace}{affiene afbeelding} is ``lineaire afbeelding plus
locatiegegevens'', en de locatiegegevens storten in tot één
goed gekozen oorsprong wanneer $1$ geen \omterm{def:b2:reduction:eigen}{eigenwaarde} is.
Omgekeerd \emph{creëert} een slechte keuze van oorsprong de
constante termen: affiene meetkunde is de kunst van waar je $0$
zet.
\end{example}

\begin{remark}[Methode: concurrentie en collineariteit via
barycentra]
\cref{ex:b2:affine:median} is een instantie van een algemeen
recept. Om te bewijzen dat drie cevianen van een driehoek
concurrent zijn, vertoon één gewogen stelsel $(A, \alpha; B,
\beta; C, \gamma)$ en gebruik associativiteit op drie manieren:
groeperen van $(B, C)$ toont dat het \omterm{def:b2:affine:barycenter}{barycentrum} op de ceviaan
vanuit $A$ ligt, groeperen van $(C, A)$ op de ceviaan vanuit
$B$, groeperen van $(A, B)$ op de derde. Voor de zwaartelijnen
doet het stelsel $(A, 1; B, 1; C, 1)$ al het werk; voor
cevianen die de zijden in voorgeschreven verhoudingen snijden,
worden de gewichten van de verhoudingen afgelezen. Om te
bewijzen dat drie punten \emph{collineair} zijn, schrijf er één
als \omterm{def:b2:affine:barycenter}{barycentrum} van de andere twee (\cref{exo:b2:affine:2}),
of gebruik het determinantcriterium van
\cref{exo:b2:affine:11}. Beide recepten vervangen meetkundige
vindingrijkheid door gewichtenboekhouding --- dat is precies
waarvoor barycentrische rekenkunde dient.
\end{remark}

\section{Convexiteit, affien}

\begin{definition}\label{def:b2:affine:convex}
Een deelverzameling $C$ van een \omterm{def:b2:affine:def}{affiene ruimte} is
\emph{convex}\index{convexe verzameling} wanneer ze elk
\omterm{def:b2:affine:barycenter}{barycentrum} met \emph{niet-negatieve} gewichten van haar punten
bevat --- equivalent, elk segment $\intcc{A}{B} =
\{\operatorname{bar}(A, 1-t; B, t) : t \in \intcc{0}{1}\}$
tussen haar punten. De \emph{convexe
omhulling}\index{convexe omhulling} $\operatorname{conv}(S)$ is
de verzameling van alle barycentra met niet-negatieve gewichten
van punten van $S$ --- de kleinste convexe verzameling die $S$
bevat.
\end{definition}

\begin{example}[Epigrafen zijn convexe
verzamelingen]\label{ex:b2:affine:epigraph}
Het gebied $C = \{(x, y) : y \geq x^2\}$ boven de parabool is
\omterm{def:b2:affine:convex}{convex}: voor $(x_1, y_1), (x_2, y_2) \in C$ en $t \in
\intcc01$ geeft de convexiteitsongelijkheid van de
kwadraatfunctie
\[
\bigl((1-t)x_1 + tx_2\bigr)^2 \leq (1-t)x_1^2 + tx_2^2
\leq (1-t)y_1 + ty_2 ,
\]
zodat het \omterm{def:b2:affine:barycenter}{barycentrum} boven de parabool blijft. De berekening
is algemeen: $\{y \geq f(x)\}$ is \omterm{def:b2:affine:convex}{convex} precies wanneer $f$
een convexe functie is --- convexe \emph{verzamelingen} en
convexe \emph{functies} (\cref{ch:b2:realfun}) zijn twee
gezichten van één begrip, met epigrafen als woordenboek. Dit
is de meetkundige reden dat steunlijnen bestaan voor convexe
functies, het feit dat Jensens ongelijkheid zal bewijzen in
\cref{ch:b2:randomvar}.
\end{example}

\begin{example}[{Redundante generatoren van een convexe
omhulling}]\label{ex:b2:affine:squarehull}
Zij
\[
S = \bigl\{(0,0),\ (2,0),\ (2,2),\ (0,2),\ (1,1)\bigr\}.
\]
Het vijfde punt is het \omterm{def:b2:affine:barycenter}{barycentrum}
\[
(1,1) = \operatorname{bar}\bigl((0,0), \tfrac12;\ (2,2),
\tfrac12\bigr),
\]
dus het ligt al in de omhulling van de andere vier:
$\operatorname{conv}(S)$ is het vierkant met de vier hoeken
als hoekpunten. In het algemeen kan een punt van $S$ dat een
\omterm{def:b2:affine:barycenter}{barycentrum} met niet-negatieve gewichten is van de
\emph{andere} punten van $S$ worden weggelaten zonder de
omhulling te veranderen; de punten die nooit weggelaten kunnen
worden (hier de vier hoeken) zijn de \emph{extreme punten}
van de omhulling. Die bepalen is een pure
barycentrumberekening: $(2,0)$ bijvoorbeeld kan niet worden
geschreven als $\operatorname{bar}$ van de resterende punten
met niet-negatieve gewichten, omdat de eerste coördinaat al
het gewicht op punten met $x = 2$ zou dwingen, en de tweede
coördinaat dan faalt. Convexiteitsvragen herleiden, keer op
keer, tot het oplossen van kleine gewogen stelsels.
\end{example}

\begin{theorem}[Carathéodory]\label{thm:b2:affine:caratheodory}
In een \omterm{def:b2:affine:def}{affiene ruimte} van dimensie $n$ is elk punt van
$\operatorname{conv}(S)$ een \omterm{def:b2:affine:barycenter}{barycentrum} van ten hoogste $n +
1$ punten van $S$.\index{stelling van Carathéodory}
\end{theorem}

\begin{proof}
Zij $G = \operatorname{bar}(A_0, \lambda_0; \dots; A_k,
\lambda_k)$ met $\lambda_i > 0$, $\sum\lambda_i = 1$, en $k + 1
> n + 1$ punten. De $k$ vectoren $\vect{A_0A_i}$ ($i \geq 1$)
zijn lineair afhankelijk ($k > n$): $\sum_{i\geq1}\mu_i
\vect{A_0A_i} = 0$ niet-triviaal; met $\mu_0 =
-\sum_{i\geq1}\mu_i$ krijgen we gewichten $(\mu_i)$ met
$\sum\mu_i = 0$, $\sum \mu_i\,\vect{OA_i} = 0$ (elke $O$),
niet alle nul. Dan produceren voor elke reële $t$ de gewichten
$\lambda_i - t\mu_i$ nog steeds som $1$ en, omdat
$\sum_i\mu_i\vect{OA_i} = 0$,
\[
\sum_i(\lambda_i - t\mu_i)\,\vect{OA_i}
= \sum_i\lambda_i\,\vect{OA_i} :
\]
ze leveren het \emph{zelfde} punt $G$. Schuif nu $t$ vanaf
$0$: enige $\mu_i$ is positief (ze sommeren tot nul en zijn
niet alle nul), dus
\[
t^* = \min\Bigl\{\frac{\lambda_i}{\mu_i} : \mu_i > 0\Bigr\}
\]
is welgedefinieerd en positief. Bij $t = t^*$: voor indices
met $\mu_i > 0$ geldt $\lambda_i - t^*\mu_i \geq 0$ door
minimaliteit, met gelijkheid bij een minimerende index; voor
indices met $\mu_i \leq 0$, $\lambda_i - t^*\mu_i \geq
\lambda_i > 0$. Alle gewichten blijven niet-negatief en ten
minste één is gestorven: $G$ is herschreven als \omterm{def:b2:affine:barycenter}{barycentrum}
van minder punten. Itereer zolang er meer dan $n + 1$ punten
overblijven.
\end{proof}

\begin{example}
In het vlak ($n = 2$): elk punt van de \omterm{def:b2:affine:convex}{convexe omhulling} van
een eindige verzameling ligt in een driehoek met hoekpunten
in de verzameling --- de meetkundige inhoud van Carathéodory,
gebruikt in optimalisatie en waarschijnlijkheid (mengsels)
gelijkelijk.
\end{example}

\begin{example}[Het algoritme van Carathéodory
uitvoeren]\label{ex:b2:affine:caratheodoryrun}
Schrijf het centrum van het vierkant van
\cref{ex:b2:affine:squarehull} met de vier hoeken $A_1 =
(0,0)$, $A_2 = (2,0)$, $A_3 = (2,2)$, $A_4 = (0,2)$:
\[
(1,1) = \operatorname{bar}\bigl(A_1, \tfrac14;\ A_2,
\tfrac14;\ A_3, \tfrac14;\ A_4, \tfrac14\bigr),
\]
vier punten in dimensie $2$ --- één te veel. Het recept van
het bewijs vraagt om gewichten $(\mu_i)$ met $\sum\mu_i = 0$
en $\sum\mu_i\vect{OA_i} = 0$: hier werkt $\mu = (1, -1, 1,
-1)$ (de twee diagonalen delen hun midden). Schuiven van
$\lambda_i \mapsto \lambda_i - t\mu_i$ houdt het \omterm{def:b2:affine:barycenter}{barycentrum}
vast voor elke $t$; de extreme toelaatbare waarde $t =
\frac14$ maakt de gewichten $(0, \tfrac12, 0, \tfrac12)$, en
doodt $A_1$ en $A_3$ tegelijk:
\[
(1,1) = \operatorname{bar}\bigl(A_2, \tfrac12;\ A_4,
\tfrac12\bigr),
\]
een voorstelling door twee punten --- zelfs beter dan de
drie die de stelling garandeert, omdat het centrum toevallig
op een segment tussen generatoren ligt. Het algoritme is
geheel mechanisch: vind een afhankelijkheidsrelatie, schuif
tot een gewicht sterft, herhaal.
\end{example}

\section{Affiene classificatietools}

\begin{proposition}[Vaste punten van affiene
afbeeldingen]\label{prop:b2:affine:fixedpoint}
Zij $f$ een \omterm{def:b2:affine:subspace}{affien} endomorfisme van een eindigdimensionale
\omterm{def:b2:affine:def}{affiene ruimte} met lineair deel $\varphi$. Als $1 \notin
\operatorname{Sp}(\varphi)$, dan heeft $f$ precies één vast
punt $\Omega$, en in de vectorialisatie in $\Omega$
\emph{is} $f$ haar lineaire deel. (Translaties, met $\varphi
= \mathrm{id}$ en geen vast punt, zijn de basale obstructie.)
\end{proposition}

\begin{proof}
Fixeer $O$ en schrijf $f(O + x) = f(O) + \varphi(x)$. Het punt
$O + x$ is vast dan en slechts dan als $O + x = f(O) +
\varphi(x)$, d.w.z.
\[
(\mathrm{id} - \varphi)(x) = \vect{O f(O)} .
\]
In eindige dimensie is $\mathrm{id} - \varphi$ inverteerbaar
dan en slechts dan als $0$ geen \omterm{def:b2:reduction:eigen}{eigenwaarde} is van
$\mathrm{id} - \varphi$, d.w.z.\ $1 \notin
\operatorname{Sp}\varphi$ --- en in dat geval heeft de
weergegeven vergelijking precies één oplossing $x^*$, die het
unieke vaste punt $\Omega = O + x^*$ geeft. Hercentreren: voor
elke vector $u$,
\[
f(\Omega + u) = f(\Omega) + \varphi(u) = \Omega + \varphi(u),
\]
dus in het raam met oorsprong $\Omega$ leest de afbeelding $u
\mapsto \varphi(u)$: puur lineair. Wanneer $1 \in
\operatorname{Sp}\varphi$, bestaat er ofwel geen vast punt
(de weergegeven vergelijking kan onoplosbaar zijn, zoals bij
een translatie) ofwel een hele affiene deelruimte ervan (voeg
elke \omterm{def:b2:reduction:eigen}{eigenvector} van \omterm{def:b2:reduction:eigen}{eigenwaarde} $1$ toe aan een oplossing):
uniciteit is precies de spectrale voorwaarde.
\end{proof}

\begin{example}[Isometrieën van het vlak,
voltooid]\label{ex:b2:affine:isometries}
Een affiene isometrie van het euclidische vlak heeft lineair
deel in $O(2)$: een rotatie $R_\theta$ of een spiegeling
(volume Jaar~1). Als $\theta \neq 0$: $1 \notin
\operatorname{Sp} R_\theta$, dus de afbeelding is een
\emph{rotatie om een uniek centrum}
(\cref{prop:b2:affine:fixedpoint}). Als het lineaire deel een
spiegeling is: ofwel een spiegeling in een as (vaste punten
bestaan) ofwel een \emph{glijspiegeling} (spiegeling
samengesteld met een translatie langs de as, geen vast punt).
Met de translaties is dit de volledige classificatie van
vlakke isometrieën.
\end{example}

\begin{remark}[De vlakke isometrieën, in één oogopslag]
De gevallen verzamelend: identiteit; translaties ($\vec f =
\mathrm{id}$, geen vast punt tenzij triviaal); rotaties
(lineair deel $R_\theta$, $\theta \neq 0$: één centrum);
spiegelingen (lineair deel een spiegeling, een lijn van vaste
punten); glijspiegelingen (zelfde lineaire deel, geen vast
punt). Vier families plus de identiteit, elk herkend door
slechts twee gegevens: het lineaire deel en de
verzameling vaste punten --- het patroon van
\cref{prop:b2:affine:fixedpoint} exhaustief gemaakt.
\end{remark}

\begin{example}[Een glijspiegeling, op heterdaad
betrapt]\label{ex:b2:affine:glide}
Zij $f(x, y) = (y + 1,\ x + 1)$. Het lineaire deel $(x, y)
\mapsto (y, x)$ is de spiegeling in de diagonaal $y = x$, dus
$1 \in \operatorname{Sp}\vec f$ en
\cref{prop:b2:affine:fixedpoint} zwijgt. Vaste punten zouden
tegelijk $x = y + 1$ en $y = x + 1$ nodig hebben: onmogelijk
--- er bestaan er geen, dus $f$ is geen spiegeling. Kwadrateren
beslist de classificatie:
\[
f\bigl(f(x, y)\bigr) = f(y + 1,\ x + 1) = (x + 2,\ y + 2),
\]
de translatie over $(2, 2)$: $f$ is de \emph{glijspiegeling}
met as de lijn $y = x$ (passend verschoven: het midden van $M$
en $f(M)$ ligt altijd op $y = x + {}$constante, hier $y = x$,
zoals men controleert op $M = (0, 0) \mapsto (1,1)$) en
glijvector $(1, 1)$, de helft van $f \circ f$. Vergelijk met
\cref{exo:b2:affine:6}, waar hetzelfde lineaire deel maar een
andere constante een eerlijke spiegeling opleverde: met
\omterm{def:b2:reduction:eigen}{eigenwaarde} $1$ aanwezig beslist de constante term alles.
\end{example}

\begin{example}[Affiene recursies zijn affiene
dynamica]\label{ex:b2:affine:recursion}
De klassieke recursie $u_{n+1} = au_n + b$ ($a \neq 1$)
itereert de \omterm{def:b2:affine:subspace}{affiene afbeelding} $f(x) = ax + b$ van de lijn,
waarvan het lineaire deel $a$ de \omterm{def:b2:reduction:eigen}{eigenwaarde} $1$ vermijdt: er
is een uniek vast punt $\omega = \frac{b}{1-a}$, en
hercentreren daar (het eendimensionale geval van de
propositie hierboven) maakt van $f$ vermenigvuldiging met $a$:
\[
u_{n+1} - \omega = a\,(u_n - \omega)
\qquad\Longrightarrow\qquad
u_n = \omega + a^n(u_0 - \omega) .
\]
Voor $u_{n+1} = \frac{u_n}2 + 3$: $\omega = 6$ en $u_n = 6 +
(u_0 - 6)2^{-n} \to 6$. Het recept dat voor zulke recursies
in \cref{ch:b2:series} wordt geleerd --- ``trek het vaste
punt af'' --- is precies de vectorialisatie van een \omterm{def:b2:affine:subspace}{affiene
afbeelding} in haar vaste punt; convergentie voor $\abs a < 1$
is het contractieverschijnsel dat \cref{ch:b2:metric} tot de
vaste-puntstelling van Banach maakte. Eén idee, drie
hoofdstukken.
\end{example}

\begin{example}[Het centrum van een rotatie
vinden]\label{ex:b2:affine:rotationcenter}
Zij $f(x, y) = (-y + 2,\ x)$. Het lineaire deel is
$\varphi(x, y) = (-y, x)$: de rotatie over hoek $\frac\pi2$,
waarvan het spectrum $\{\iu, -\iu\}$ de $1$ vermijdt. Door
\cref{prop:b2:affine:fixedpoint} is er precies één vast punt:
$x = -y + 2$ en $y = x$ geven $x = 1$, $y = 1$, dus $\Omega =
(1, 1)$, en $f$ \emph{is} de rotatie met centrum $(1, 1)$ en
hoek $\frac\pi2$. De algemene les: wanneer $1 \notin
\operatorname{Sp}\vec f$, kost het classificeren van $f$ één
lineair stelsel --- de meetkunde zit geheel in het lineaire
deel, de rekenkunde geheel in het lokaliseren van het
centrum.
\end{example}

\begin{remark}[Waar affiene taal hierna wordt gebruikt]
Barycentra en \omterm{def:b2:affine:subspace}{affiene afbeeldingen} zijn de grammatica van de
volgende meetkundige hoofdstukken: raaklijnen en raakvlakken
zijn affiene objecten
(\cref{ch:b2:curves,ch:b2:surfaces}), een affiene
variabelensubstitutie vermenigvuldigt oppervlakten en
inhouden met $\abs{\det \vec f}$ (\cref{ch:b2:multint}), en
de verwachting is een \omterm{def:b2:affine:barycenter}{barycentrum} met gewichten gegeven door
een waarschijnlijkheidswet, waarom convexiteit Jensens
ongelijkheid bestuurt (\cref{ch:b2:randomvar}). In het volume
van Jaar~3 draagt dezelfde convexiteitswoordenschat de studie
van $L^p$-normen en integraalongelijkheden.
\end{remark}

\begin{remark}[Perspectieven binnen dit volume]
Twee draden verlaten dit hoofdstuk. De \emph{affiene} draad:
raaklijnen (\cref{ch:b2:curves}) en raakvlakken
(\cref{ch:b2:surfaces}) zijn affiene deelruimten gehecht aan
niet-lineaire objecten, en de classificatie van kwadrieken in
het oppervlakkenhoofdstuk loopt op de centrumvergelijking
$A\Omega = -b$ van dit hoofdstuk. De \emph{convexe} draad is
langer: convexiteit van halfvlakken en schijven voedt de
Helly-theorie van het weekendprobleem; convexiteit van
functies geeft Jensens ongelijkheid (\cref{ch:b2:randomvar});
en de slotstelling van het boek --- het uitdovingscriterium
voor vertakkingsprocessen (\cref{ch:b2:genfun}) --- wordt
beslist door de ligging van een convexe kromme ten opzichte
van de diagonaal, een beeld dat evenzeer bij dit hoofdstuk
hoort als bij de waarschijnlijkheid. Barycentra keren daar
ook terug: een verwachting is een \omterm{def:b2:affine:barycenter}{barycentrum} met
waarschijnlijkheidsgewichten.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b2:affine:1}
In $\R^3$, zijn de volgende affiene deelruimten? Geef
richtingen en dimensies. $\{x + y + z = 1\}$; $\;\{x + y + z
= 1,\ x - z = 3\}$; $\;\{x^2 + y^2 = 1\}$; de
oplossingsverzameling van $MX = B$ voor een gegeven
compatibel stelsel.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:affine:2}
Bewijs dat drie onderscheiden punten $A, B, C$ van een
\omterm{def:b2:affine:def}{affiene ruimte} collineair zijn dan en slechts dan als $C$ een
\omterm{def:b2:affine:barycenter}{barycentrum} is van $A$ en $B$, dan en slechts dan als de
vectoren $\vect{AB}, \vect{AC}$ lineair afhankelijk zijn.
Leid Menelaos-achtige gewichtenboekhouding af: als $C =
\operatorname{bar}(A, 1 - t; B, t)$, lokaliseer $C$ voor $t
= \frac12$, $t = 2$, $t = -1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:affine:3}
Zij $f$ de \omterm{def:b2:affine:subspace}{affiene afbeelding} van $\R^2$ gegeven door $f(X)
= MX + C$ met $M = \frac12\begin{pmatrix} 1 & 1\\ 1 &
1\end{pmatrix}$ en $C = (1, 0)^{\mathsf T}$. Bepaal het
beeld van $f$, haar vaste punten (indien aanwezig), en $f
\circ f$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:affine:4}
(Associativiteit in actie) In een driehoek $ABC$, zij $I, J,
K$ die $BC$, $CA$, $AB$ verdelen in de verhoudingen
$\vect{BI} = \frac13\vect{BC}$, $\vect{CJ} =
\frac13\vect{CA}$, $\vect{AK} = \frac13\vect{AB}$. Druk $I,
J, K$ uit als barycentra en bereken het \omterm{def:b2:affine:barycenter}{barycentrum} van
$(I,1;J,1;K,1)$: wat vind je, en waarom was het
voorspelbaar?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:affine:5}
Bewijs dat een afbeelding $f \colon \R^n \to \R^n$ die
\emph{midpunten} behoudt
($f\bigl(\frac{A+B}{2}\bigr) = \frac{f(A) + f(B)}{2}$) en
\emph{continu} is, \omterm{def:b2:affine:subspace}{affien} is. \emph{(Toon dat de
vectorafbeelding $u \mapsto f(O + u) - f(O)$ additief is via
midpunten, daarna $\Q$-homogeen, daarna $\R$-homogeen door
continuïteit --- dezelfde densiteitsstrategie als voor de
functionaalvergelijking van Cauchy in het volume van Jaar~1;
leid de nodige stappen hier opnieuw af.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:affine:6}
Classificeer de \omterm{def:b2:affine:subspace}{affiene afbeelding} $f(x, y) = (y + 1,\; x -
1)$ van het euclidische vlak: lineair deel, vaste punten,
meetkundige aard (spiegeling? glij?). Bereken $f \circ f$ en
concludeer.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:affine:7}
(Radon) Zij $A_1, \dots, A_{n+2}$ punten van een \omterm{def:b2:affine:def}{affiene
ruimte} van dimensie $n$. Bewijs dat ze gesplitst kunnen
worden in twee disjuncte groepen waarvan de \omterm{def:b2:affine:convex}{convexe
omhullingen} snijden. \emph{(Zoals in het bewijs van
Carathéodory, vind gewichten $\mu_i$, niet alle nul, met
$\sum\mu_i = 0$ en $\sum\mu_i\vect{OA_i} = 0$; scheid
positieve en negatieve gewichten en normaliseer beide
zijden.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:affine:8}
Zij $f$ een \omterm{def:b2:affine:subspace}{affien} endomorfisme van $\R^n$ met $f \circ f =
f$. Bewijs dat $f$ de affiene projectie is op de affiene
deelruimte $\operatorname{Fix}(f) = \operatorname{im} f$
langs de richting $\ker\vec f$, en dat omgekeerd alle
dergelijke projecties idempotent zijn. \emph{(Toon eerst dat
$\operatorname{im} f$ uit vaste punten bestaat.)}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:affine:9}
Zij $G = \operatorname{bar}(A, 1;\ B, 2;\ C, 3)$ in een
driehoek $ABC$. Toon met associativiteit dat de lijn $AG$
de lijn $BC$ snijdt in $M = \operatorname{bar}(B, 2;\ C,
3)$, en lokaliseer $G$ op het segment $\intcc AM$;
lokaliseer evenzo het snijpunt van $BG$ met $CA$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:affine:10}
Voor $\lambda \neq 0$ is de \emph{homothetie} $h_{\Omega,
\lambda}$ de \omterm{def:b2:affine:subspace}{affiene afbeelding} die $\Omega$ vast laat met
lineair deel $\lambda\,\mathrm{id}$. Bewijs dat de
samenstelling $h_{\Omega', \mu} \circ h_{\Omega, \lambda}$
een homothetie is van verhouding $\lambda\mu$ wanneer
$\lambda\mu \neq 1$, en een translatie wanneer $\lambda\mu =
1$; in het geval $\lambda = \mu = -1$ (twee
puntsymmetrieën), bereken de translatievector.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:affine:11}
(Menelaos) In een driehoek $ABC$, zij $A' \in (BC)$, $B'
\in (CA)$, $C' \in (AB)$, alle verschillend van de
hoekpunten, en definieer $\alpha, \beta, \gamma$ door
$\vect{A'B} = \alpha\,\vect{A'C}$, $\vect{B'C} =
\beta\,\vect{B'A}$, $\vect{C'A} = \gamma\,\vect{C'B}$. Bewijs
dat $A', B', C'$ collineair zijn dan en slechts dan als
$\alpha\beta\gamma = 1$. \emph{(Schrijf elk punt als
\omterm{def:b2:affine:barycenter}{barycentrum} van twee hoekpunten; toon dat drie punten
collineair zijn dan en slechts dan als hun barycentrische
coördinatenrijen ten opzichte van $(A, B, C)$ een singuliere
$3 \times 3$-matrix vormen.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:affine:12}
Bewijs dat de \omterm{def:b2:affine:convex}{convexe omhulling} van een compacte
deelverzameling $K$ van $\R^n$ \omterm{def:b2:metric:compact}{compact} is. \emph{(Door
\cref{thm:b2:affine:caratheodory} is
$\operatorname{conv}(K)$ het beeld van een compacte
verzameling onder een continue afbeelding.)} Toon met een
voorbeeld in $\R^2$ dat de \omterm{def:b2:affine:convex}{convexe omhulling} van een
\emph{gesloten} verzameling niet noodzakelijk gesloten is.
\end{exercise}

\begin{problem}[{Weekendprobleem --- de stelling van Helly en
twee van haar dividenden}]\label{pb:b2:affine:1}
Het lemma van Radon (\cref{exo:b2:affine:7}) zegt dat $n + 2$
punten van een $n$-dimensionale \omterm{def:b2:affine:def}{affiene ruimte} zich altijd
splitsen in twee groepen met snijdende \omterm{def:b2:affine:convex}{convexe omhullingen}.
Dit probleem maakt van dat ene lineaire-algebrafeit een keten
van stellingen van combinatorische meetkunde: de
doorsnedestelling van Helly, de centrumspuntstelling (een
tweedimensionale mediaan), en de overdekkingsstelling van
Jung. Doorheen is het vlak $\R^2$ met zijn gebruikelijke
euclidische structuur, en $\det$ is de \omterm{def:b2:linalg:det}{determinant} in de
canonieke basis.\index{stelling van Helly}\index{lemma van
Radon}\index{stelling van Jung}\index{centrumspunt}

\medskip
\noindent\textbf{Deel I --- Barycentrische coördinaten.}
Punten $A_0, \dots, A_k$ zijn \emph{\omterm{def:b2:affine:subspace}{affien} onafhankelijk}
wanneer de vectoren $\vect{A_0A_1}, \dots, \vect{A_0A_k}$
lineair onafhankelijk zijn.
\begin{enumerate}
  \item Toon dat affiene onafhankelijkheid niet afhangt van
        de keuze van het baspunt $A_0$, en dat ze equivalent
        is aan: wanneer twee families van gewichten, elk met
        som $1$, hetzelfde \omterm{def:b2:affine:barycenter}{barycentrum} van $(A_0, \dots,
        A_k)$ definiëren, vallen de gewichten samen.
  \item Zij $(A, B, C)$ \omterm{def:b2:affine:subspace}{affien} onafhankelijk in het vlak.
        Toon dat elk punt $M$ een unieke triple $(\alpha,
        \beta, \gamma)$ toelaat met $\alpha + \beta + \gamma
        = 1$ en $M = \operatorname{bar}(A, \alpha; B,
        \beta; C, \gamma)$ --- haar \emph{barycentrische
        coördinaten}.
  \item Bewijs de determinantformules
        \[
        \alpha = \frac{\det(\vect{MB},
        \vect{MC})}{\det(\vect{AB}, \vect{AC})},
        \qquad
        \beta = \frac{\det(\vect{MC},
        \vect{MA})}{\det(\vect{AB}, \vect{AC})},
        \qquad
        \gamma = \frac{\det(\vect{MA},
        \vect{MB})}{\det(\vect{AB}, \vect{AC})} :
        \]
        barycentrische coördinaten zijn verhoudingen van
        georiënteerde oppervlakten.
  \item De lijnen $BC$, $CA$, $AB$ zijn de coördinaatlijnen
        $\{\alpha = 0\}$, $\{\beta = 0\}$, $\{\gamma = 0\}$.
        Toon dat $M$ in de gesloten driehoek
        $\operatorname{conv}\{A, B, C\}$ ligt dan en slechts
        dan als $\alpha, \beta, \gamma \geq 0$, en dat de
        drie lijnen het vlak in precies zeven gebieden
        snijden, geclassificeerd door de tekens van
        $(\alpha, \beta, \gamma)$ (het tekenpatroon $(-,-,-)$
        is onmogelijk).
  \item Zij $u \colon \R^2 \to \R$ een \omterm{def:b2:affine:subspace}{affiene afbeelding}
        (een \emph{affiene vorm}). Toon $u(M) =
        \alpha\,u(A) + \beta\,u(B) + \gamma\,u(C)$, dat de
        niveauverzamelingen van een niet-constante affiene
        vorm lijnen zijn, dat elke lijn zo ontstaat, en dat
        de gesloten halfvlakken $\{u \geq c\}$ \omterm{def:b2:affine:convex}{convex} zijn.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Radon-partities, verfijnd.} Een
familie van $n + 2$ punten van $\R^n$ is in \emph{algemene
ligging} wanneer elke $n + 1$ ervan \omterm{def:b2:affine:subspace}{affien} onafhankelijk
zijn. Een \emph{affiene afhankelijkheidsrelatie} van $(A_1,
\dots, A_{n+2})$ is een familie $(\mu_i)$ met $\sum_i \mu_i
= 0$ en $\sum_i \mu_i\,\vect{OA_i} = 0$ voor één (dus elke)
oorsprong $O$.
\begin{enumerate}[resume]
  \item Bereken een niet-nulle affiene
        afhankelijkheidsrelatie van de vier punten $A_1 =
        (0,0)$, $A_2 = (3,0)$, $A_3 = (0,3)$, $A_4 = (1,1)$;
        geef de Radon-partitie en het Radon-punt.
  \item Toon dat voor punten in algemene ligging de
        vectorruimte van affiene afhankelijkheidsrelaties
        dimensie precies $1$ heeft, en dat een niet-nulle
        relatie \emph{geen} verdwijnende coëfficiënt heeft.
  \item Leid af dat de Radon-partitie van $n + 2$ punten in
        algemene ligging uniek is (op verwisseling van de
        twee blokken na), waarbij elk blok de verzameling
        indices is waar $\mu_i$ één vast teken heeft.
  \item Voor vier punten van het vlak in algemene ligging,
        toon de dichotomie: ofwel heeft de partitie type
        $(1, 3)$ --- één punt inwendig in de driehoek van
        de andere drie --- of type $(2, 2)$: de vier punten
        liggen \omterm{def:b2:affine:convex}{convex} en de segmenten die de twee paren
        verbinden (de diagonalen) snijden, in het
        Radon-punt.
  \item Voer vraag 6 uit voor het eenheidsvierkant $(0,0)$,
        $(1,0)$, $(1,1)$, $(0,1)$: relatie, partitie,
        Radon-punt.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De stelling van Helly in het
vlak.}
\begin{enumerate}[resume]
  \item Zij $C_1, C_2, C_3, C_4$ convexe deelverzamelingen
        van $\R^2$, waarvan elke drie een gemeenschappelijk
        punt hebben. Kies $x_i \in \bigcap_{j \neq i} C_j$
        en pas het lemma van Radon toe op $x_1, \dots,
        x_4$: toon dat het Radon-punt tot alle vier
        verzamelingen behoort. \emph{(Voor elke $k$ bestaat
        het blok dat $x_k$ niet bevat uit punten van
        $C_k$.)}
  \item (Helly) Zij $C_1, \dots, C_m$ ($m \geq 3$) convexe
        deelverzamelingen van $\R^2$, waarvan elke drie
        snijden. Bewijs $\bigcap_{i=1}^m C_i \neq
        \emptyset$, door inductie op $m$: vervang $C_{m-1}$
        en $C_m$ door $C_{m-1} \cap C_m$ en controleer de
        hypothese voor de nieuwe familie met vraag 11.
  \item Drie tegenvoorbeelden, één per hypothese: (a) de
        drie gesloten randen van een driehoek snijden
        paarsgewijs maar hebben geen gemeenschappelijk punt
        ($3$ kan niet tot $2$ worden verlaagd); (b) de vier
        verzamelingen $S_i = \{x_1, \dots, x_4\} \setminus
        \{x_i\}$, voor vier punten in algemene ligging,
        voldoen aan de driedoorsnedehypothese maar niet aan
        de conclusie (convexiteit telt); (c) de gesloten
        halfvlakken $H_k = \intco{k}{+\infty} \times \R$,
        $k \in \N$, snijden paarsgewijs en
        drievoudig maar $\bigcap_k H_k = \emptyset$
        (oneindige families hebben compactheid nodig).
  \item (Compacte Helly) Zij $(K_i)_{i \in I}$ een
        willekeurige familie van compacte convexe
        deelverzamelingen van $\R^2$, waarvan elke drie
        snijden. Met vraag 12 en de eigenschap van
        Borel--Lebesgue (\cref{thm:b2:metric:borellebesgue}),
        toon $\bigcap_{i \in I} K_i \neq \emptyset$.
  \item (Eerste dividend) Zij $S$ een eindige verzameling
        punten van het vlak en $r > 0$. Toon: als elke drie
        punten van $S$ in een gesloten schijf van straal $r$
        liggen, dan ligt $S$ in één gesloten schijf van
        straal $r$. \emph{(Pas Helly toe op de schijven
        $\overline D(p, r)$, $p \in S$.)}
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De centrumspuntstelling.} Een
\emph{centrumspunt} van een eindige verzameling $S$ van $n$
punten van het vlak is een punt $c$ (niet noodzakelijk in
$S$) zodat elk gesloten halfvlak dat $c$ bevat ten minste
$n/3$ punten van $S$ bevat.
\begin{enumerate}[resume]
  \item (Dimensie $1$) Voor reële $x_1 \leq \dots \leq
        x_n$, toon dat de mediaan $c = x_{\lceil n/2
        \rceil}$ voldoet: elke gesloten halfrechte die $c$
        bevat bevat ten minste $n/2$ van de $x_i$.
  \item (Tellemma) Als $A, B, C$ deelverzamelingen van $S$
        zijn met $\abs A, \abs B, \abs C > \tfrac{2n}3$,
        toon $A \cap B \cap C \neq \emptyset$.
  \item Zij $m = \floor{2n/3} + 1$ en zij $\mathcal F$ de
        (eindige) familie van de \omterm{def:b2:affine:convex}{convexe omhullingen}
        $\operatorname{conv}(T)$, $T \subseteq S$, $\abs T =
        m$. Toon dat elke drie leden van $\mathcal F$ een
        gemeenschappelijk punt hebben, en leid uit Helly
        een punt $c$ af dat ze allemaal gemeen hebben.
  \item Bewijs dat dit $c$ een centrumspunt van $S$ is: de
        \emph{centrumspuntstelling}. \emph{(Als een gesloten
        halfvlak door $c$ minder dan $n/3$ punten bevatte,
        zou zijn open complement een verzameling $T$ van
        $m$ punten bevatten, en $\operatorname{conv}(T)$
        zou $c$ mijden.)}
  \item Scherpte: zij $n = 3k$ en plaats $k$ punten in elk
        van drie schijven van kleine straal $\varepsilon$
        gecentreerd in de hoekpunten van een grote
        driehoek. Toon dat voor \emph{elk} punt $c$ van het
        vlak enig gesloten halfvlak dat $c$ bevat ten
        hoogste $n/3$ punten van $S$ bevat, zodat de
        constante $1/3$ niet verbeterd kan worden.
        \emph{(Onder de drie richtingen van $c$ naar de
        schijfcentra maken er twee een hoek van ten hoogste
        $2\pi/3$.)}
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- De stelling van Jung en
synthese.}
\begin{enumerate}[resume]
  \item (Driehoekslemma) Zij $P, Q, R$ drie punten met
        paarsgewijze afstanden $\leq 1$. Toon dat ze in een
        gesloten schijf van straal $1/\sqrt3$ liggen.
        \emph{(Als enige hoek $\geq \pi/2$ is, neem de
        schijf op de langste zijde als diameter, met de
        mediaanformule $\norm{RM}^2 =
        \tfrac12\norm{RP}^2 + \tfrac12\norm{RQ}^2 -
        \tfrac14\norm{PQ}^2$; als de driehoek scherphoekig
        is, begrens de omgeschreven straal $a/(2\sin
        \widehat A)$ met haar grootste hoek, die in
        $\intco{\pi/3}{\pi/2}$ ligt.)}
  \item (Jung) Leid af: elke compacte deelverzameling van
        het vlak van diameter $\leq 1$ is bevat in een
        gesloten schijf van straal $1/\sqrt3$.
  \item Scherpte: voor de gelijkzijdige driehoek $A_1A_2A_3$
        van zijde $1$ met zwaartepunt $G$, bewijs de
        identiteit van Leibniz $\sum_i \norm{\vect{OA_i}}^2
        = 3\norm{\vect{OG}}^2 + \sum_i
        \norm{\vect{GA_i}}^2$ voor elk punt $O$, en
        concludeer dat elke schijf die de drie hoekpunten
        bevat straal $\geq 1/\sqrt3$ heeft, met gelijkheid
        alleen voor de omgeschreven schijf.
  \item (Helly in $\R^n$) Formuleer en bewijs de stelling
        van Helly in $\R^n$: als eindig veel \omterm{def:b2:affine:convex}{convexe
        verzamelingen} zo zijn dat elke $n + 1$ ervan
        snijden, dan snijden ze allemaal. \emph{(Het lemma
        van Radon \cref{exo:b2:affine:7} behandelt $n + 2$
        verzamelingen; induceer daarna zoals in vraag 12.)}
  \item Synthese. Assembleer de keten
        \[
        \text{affiene afhankelijkheid} \Rightarrow \text{Radon}
        \Rightarrow \text{Helly} \Rightarrow
        \text{centrumspunt en Jung},
        \]
        met in één zin elk: waar lineaire algebra binnentreedt,
        waar de tekens van de gewichten binnentreden, waar
        convexiteit binnentreedt, en welke ene stap de
        dimensie van het vlak gebruikte. Wat worden de
        constanten $3$ (in Helly), $1/3$ (centrumspunt) en
        $1/\sqrt3$ (Jung) in $\R^n$? (Formuleer zonder
        bewijs.)
\end{enumerate}
\end{problem}
