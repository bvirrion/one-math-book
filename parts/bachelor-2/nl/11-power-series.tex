\chapter{Machtreeksen}\label{ch:b2:powerseries}

Machtreeksen zijn de best-gedragen reeksen van functies in de
wiskunde: binnen hun convergentieschijf convergeren zij normaal
op compacta, mogen term voor term gedifferentieerd en
geïntegreerd worden zonder tweede gedachte, en hun sommen ---
\emph{\omterm{def:b2:powerseries:analytic}{analytische functies}} --- worden door hun coëfficiënten
bepaald. Dit hoofdstuk bewijst dat hele pakket en herwint,
eerlijk, elke Taylorreeks van het boekdeel van Jaar~1;
\omterm{ex:b2:powerseries:fibonacci}{genererende functies} sluiten het met algebraïsche dividenden.

\section{Convergentiestraal}

\begin{lemma}[Abel]\label{lem:b2:powerseries:abel}
Als de rij $(a_n z_0^n)$ begrensd is voor enige $z_0 \neq 0$,
dan convergeert $\sum a_n z^n$ absoluut voor elke $\abs z < \abs
{z_0}$, en normaal op elke schijf $\abs z \leq r < \abs{z_0}$.
\end{lemma}

\begin{proof}
Met $\abs{a_n z_0^n} \leq M$ en $\abs z \leq r$:
\[
\abs{a_n z^n} = \abs{a_n z_0^n}\,\Bigl|\frac{z}{z_0}\Bigr|^n
\leq M\Bigl(\frac{r}{\abs{z_0}}\Bigr)^{\!n},
\]
een convergente meetkundige grens, uniform op de schijf.
\end{proof}

\begin{definition}[Convergentiestraal]\label{def:b2:powerseries:radius}
De \emph{convergentiestraal}\index{convergentiestraal} van
$\sum a_n z^n$ is
\[
R = \sup\{r \geq 0 : (a_n r^n) \text{ begrensd}\} \in
\intcc{0}{+\infty} .
\]
Door \cref{lem:b2:powerseries:abel}: absolute convergentie voor
$\abs z < R$ (normaal op compacte subschijven), divergentie ---
inderdaad onbegrensde termen --- voor $\abs z > R$. Op de
randcirkel kan van alles gebeuren (\cref{exo:b2:powerseries:2}).
In de praktijk wordt $R$ berekend via het quotiëntcriterium van
d'Alembert op $\abs{a_n}\abs z^n$ of via vergelijking.
\end{definition}

\begin{example}[Een straal zonder quotiëntcriterium]\label{ex:b2:powerseries:sinn}
Wat is de straal van $\sum \sin(n)\,z^n$? Het quotiënt
$\abs{\sin(n+1)/\sin n}$ heeft geen limiet, maar de definitie
werkt direct. \emph{$R \geq 1$:} $\abs{\sin n} \leq 1$, dus
$(\sin n\cdot r^n)$ is begrensd voor elke $r < 1$ --- zelfs
voor $r = 1$. \emph{$R \leq 1$:} het volstaat dat $\sin n
\not\to 0$. Stel $\sin n \to 0$; de optelformule
\[
\sin(n+1) = \sin n\cos 1 + \cos n\sin 1
\]
zou $\cos n \to 0$ forceren (los op naar $\cos n$: $\sin 1 \neq
0$), in tegenspraak met $\sin^2 n + \cos^2 n = 1$. Dus de
termen $\sin(n)\,1^n$ gaan niet naar $0$: de reeks divergeert
in $z = 1$, en $R \leq 1$. Conclusie: $R = 1$. Slotinzicht: de
straal is een uitspraak over $\abs{a_n}r^n$ begrensd zijn ---
geen limiet van quotiënten is ooit vereist, en
begrensdheidsargumenten regelen gevallen die het
quotiëntcriterium niet kan raken (vergelijk de oscillerende
coëfficiënten van \cref{exo:b2:powerseries:1}).
\end{example}

\begin{proposition}[Operaties]\label{prop:b2:powerseries:operations}
Zij $\sum a_nz^n$, $\sum b_nz^n$ met stralen $R_a, R_b$. Dan,
voor $\abs z < \min(R_a, R_b)$:
\[
\sum (a_n + b_n)z^n = \sum a_nz^n + \sum b_nz^n,
\qquad
\Bigl(\sum a_nz^n\Bigr)\Bigl(\sum b_nz^n\Bigr) = \sum c_n z^n,
\quad c_n = \sum_{k=0}^{n} a_kb_{n-k},
\]
beide reeksen met straal $\geq \min(R_a, R_b)$. (Het product is
het Cauchyproduct, legitiem door absolute convergentie en
\cref{thm:b2:series:fubini}.)
\end{proposition}

\begin{proof}
De somformule is de lineariteit van convergente reeksen, en
$(a_n + b_n)r^n$ is begrensd wanneer zowel $a_nr^n$ als
$b_nr^n$ dat zijn: straal $\geq \min(R_a, R_b)$. Voor het
product, fixeer $\abs z < \min(R_a, R_b)$: beide reeksen
convergeren daar \emph{absoluut}
(\cref{lem:b2:powerseries:abel}), dus de dubbel geïndexeerde
familie $(a_kz^k\,b_lz^l)_{k,l}$ is \omterm{def:b2:series:summable}{sommeerbaar}, en
\cref{thm:b2:series:fubini} laat elke groepering toe.
Groeperen op $k + l = n$:
\[
\Bigl(\sum_k a_kz^k\Bigr)\Bigl(\sum_l b_lz^l\Bigr)
= \sum_{n\geq0}\Bigl(\sum_{k+l=n}a_kb_l\Bigr)z^n
= \sum_{n\geq0}c_nz^n ,
\]
absoluut convergent voor elke zulke $z$: de productreeks heeft
eveneens straal $\geq \min(R_a, R_b)$.
\end{proof}

\begin{example}[Een Cauchy-kwadraat, dubbel gecontroleerd]\label{ex:b2:powerseries:cauchysquare}
Kwadrateer de meetkundige reeks: voor $\abs x < 1$ is de
coëfficiënt van $x^n$ in $\bigl(\sum x^k\bigr)^2$ gelijk aan
$c_n = \sum_{k+l=n} 1\cdot1 = n + 1$, dus
\[
\frac{1}{(1-x)^2} = \sum_{n\geq0}(n+1)\,x^n .
\]
Controleer term-voor-term differentiatie
(\cref{thm:b2:powerseries:calculus} hieronder): differentiëren
van $\frac{1}{1-x} = \sum x^n$ geeft $\frac{1}{(1-x)^2} = \sum
nx^{n-1} = \sum(n+1)x^n$ --- dezelfde reeks via twee
onverwante mechanismen. Slotinzicht: wanneer een
coëfficiëntidentiteit mysterieus lijkt, produceert een van deze
twee motoren (convolutie of differentiatie) haar meestal in één
regel; de vraag van het weekendprobleem over
$\sum\binom{2k}k\binom{2n-2k}{n-k} = 4^n$ draait de
convolutiemotor op volle kracht.
\end{example}

\begin{example}[Vermenigvuldigen met $\frac{1}{1-x}$ sommeert de coëfficiënten]\label{ex:b2:powerseries:partialsums}
Een Cauchyproduct tegen de meetkundige reeks heeft een
memorabele betekenis: voor elke $\sum a_nx^n$ van straal $R > 0$
en $\abs x < \min(R, 1)$,
\[
\frac{1}{1-x}\sum_{n\geq0}a_nx^n
= \sum_{n\geq0}\Bigl(\sum_{k=0}^{n}a_k\Bigr)x^n :
\]
vermenigvuldigen met $\frac{1}{1-x}$ vervangt coëfficiënten door
hun partiële sommen (convolueren met de rij van enen).
Voorbeeld: $\dfrac{\eu^x}{1-x} = \sum_n s_n x^n$ met $s_n =
\sum_{k\leq n}\frac{1}{k!}$, de partiële sommen van $\eu$ ---
vergelijk \cref{exo:b2:powerseries:11}, waar hetzelfde product
met $\eu^{-x}$ de derangementtellingen codeert. Slotinzicht:
operaties op machtreeksen zijn operaties op
coëfficiëntenrijen in vermomming (vermenigvuldig met
$\frac1{1-x}$: sommeer; vermenigvuldig met $x$: verschuif;
differentieer: vermenigvuldig met $n$ en verschuif) --- een
woordenboek dat het genererende-functies-hoofdstuk vloeiend zal
lezen.
\end{example}

\section{Regulariteit van de som}

\begin{theorem}[Term-voor-term calculus]\label{thm:b2:powerseries:calculus}
Zij $f(x) = \sum_{n\geq0} a_n x^n$ met straal $R > 0$ (reële
variabele $x \in \intoo{-R}{R}$).
\begin{enumerate}
  \item De afgeleide reeks $\sum n\,a_n x^{n-1}$ heeft dezelfde
        \emph{straal} $R$, en $f$ is $C^1$ met $f'(x) = \sum_{n
        \geq 1} n a_n x^{n-1}$. Itererend is $f$ van klasse
        $C^\infty$ en
        \[
        a_n = \frac{f^{(n)}(0)}{n!} :
        \]
        de coëfficiënten van een machtreeks zijn uniek (twee
        reeksen met dezelfde som bij $0$ hebben gelijke
        coëfficiënten).
  \item Term-voor-term primitieve: $\sum \frac{a_n}{n+1}x^{n+1}$
        heeft straal $R$ en afgeleide $f$.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{Zelfde straal:} als $(a_nr^n)$ begrensd is en $r' < r$,
dan is $n\abs{a_n} r'^{\,n-1} = \frac{n}{r'}\abs{a_nr^n}
\bigl(\frac{r'}{r}\bigr)^n$ begrensd (zelfs $\to 0$: meetkundig
verslaat $n$), dus $R' \geq R$; omgekeerd geeft $\abs{a_n x^n}
\leq \abs x \cdot n\abs{a_n}\abs x^{n-1}$ dat $R \geq R'$.

\emph{Differentiatie:} op $\intcc{-r}{r}$, $r < R$, convergeert
de afgeleide reeks normaal ($n\abs{a_n}r^{n-1}$ \omterm{def:b2:series:summable}{sommeerbaar} door
de straalberekening); de oorspronkelijke convergeert in $x =
0$: de differentiatiestelling voor reeksen
(\cref{thm:b2:funcseq:seriestransfer}) is van toepassing op elk
zulk segment, dus op $\intoo{-R}{R}$. $k$ maal itereren en
evalueren in $0$: expliciet is de $k$-de afgeleide reeks
\[
f^{(k)}(x) = \sum_{n\geq k} n(n-1)\cdots(n-k+1)\,a_n\,x^{n-k},
\]
en in $x = 0$ verdwijnt elke term met $n > k$, zodat alleen de
constante term $k(k-1)\cdots1\cdot a_k$ overblijft:
$f^{(k)}(0) = k!\,a_k$. Uniciteit van coëfficiënten volgt: twee
machtreeksen met dezelfde som bij $0$ hebben dezelfde
afgeleiden in $0$, dus dezelfde $a_k$. Primitieven: dezelfde
straal via dezelfde berekening, en term-voor-term terug
differentiëren.
\end{proof}

\begin{example}[Een reeks evalueren in een punt]\label{ex:b2:powerseries:evaluate}
Wat is $\sum_{n\geq1}\dfrac{n^2}{2^n}$? Het is de som
$\sum n^2x^n$ van \cref{exo:b2:powerseries:3} geëvalueerd
\emph{binnen} de schijf, in $x = \frac12 < 1 = R$, waar elke
manipulatie gebruikt om de gesloten vorm af te leiden legitiem
was:
\[
\sum_{n\geq1} n^2x^n = \frac{x(1+x)}{(1-x)^3}
\quad\Longrightarrow\quad
\sum_{n\geq1}\frac{n^2}{2^n}
= \frac{\frac12\cdot\frac32}{(\frac12)^3}
= \frac{3/4}{1/8} = 6 .
\]
Zelfde motor, andere knoppen: $x = \frac13$ geeft
$\sum\frac{n^2}{3^n} = \frac{\frac13\cdot\frac43}{(2/3)^3} =
\frac32$. Slotinzicht: een machtreeksidentiteit is een machine,
geen enkele formule --- één afleiding prijst elke numerieke
reeks $\sum n^2q^n$ in één keer, voor alle $\abs q < 1$; zo zal
het genererende-functies-hoofdstuk verwachtingen en
varianties en gros berekenen.
\end{example}

\begin{example}[De klassiekers, ditmaal eerlijk]\label{ex:b2:powerseries:classics}
$\displaystyle\frac{1}{1 - x} = \sum x^n$ ($R = 1$); term-voor-term
integreren (\cref{thm:b2:powerseries:calculus}~(2)):
\[
-\ln(1 - x) = \sum_{n\geq1} \frac{x^n}{n},
\qquad
\arctan x = \sum_{n \geq 0} \frac{(-1)^n x^{2n+1}}{2n+1}
\quad (\abs x < 1),
\]
de tweede in twee stappen: substitueer $-x^2$ in de meetkundige
reeks om $\frac{1}{1+x^2} = \sum(-1)^nx^{2n}$ te krijgen
(straal $1$, want $\abs{x^2} < 1 \iff \abs x < 1$), neem dan
de term-voor-term primitieve die in $0$ verdwijnt; beide zijden
zijn primitieven van dezelfde functie met dezelfde waarde in
$0$, dus gelijk op $\intoo{-1}{1}$. En $\exp$: de reeks $E(x)
= \sum \frac{x^n}{n!}$ ($R = \infty$) voldoet aan $E' = E$,
$E(0) = 1$ door termgewijze differentiatie, dus $E = \exp$ via
de uniciteit uit Jaar~1. Elke ``standaardontwikkeling'' van het
boekdeel van Jaar~1 is nu een stelling over haar volledige
machtreeks.
\end{example}

\begin{example}[Een logaritme berekend vanuit de schijf]\label{ex:b2:powerseries:lntwo}
Evalueer $-\ln(1-x) = \sum\frac{x^n}{n}$ in het inwendige punt
$x = \frac12$:
\[
\sum_{n\geq1}\frac{1}{n\,2^n} = \ln 2 ,
\]
een snel convergerende voorstelling van $\ln 2$ (tien termen
geven al $0.69306\ldots$ tegen $\ln 2 = 0.69314\ldots$), ver
beter dan de alternerende reeks $1 - \frac12 + \frac13 -
\dots$ die alleen op de rand beschikbaar is. Slotinzicht:
wanneer een constante zowel aan de rand als strikt binnen de
schijf bereikbaar is, wint het inwendige numeriek ---
meetkundig verval tegen harmonisch verval.
\end{example}

\begin{example}[Differentiatie bewaart de straal, niet de rand]\label{ex:b2:powerseries:boundaryloss}
De reeks $\sum_{n\geq1}\frac{x^n}{n^2}$ heeft straal $1$ en
convergeert in \emph{beide} eindpunten ($\sum\frac1{n^2}$ en
haar alternerende tweeling). Haar afgeleide reeks,
\[
\sum_{n\geq1}\frac{x^{n-1}}{n} ,
\]
heeft dezelfde straal $1$ --- zoals
\cref{thm:b2:powerseries:calculus} garandeert --- maar divergeert
nu in $x = 1$ (harmonische reeks) terwijl zij nog convergeert in
$x = -1$ (alternerend). Nog een differentiatie levert
$\sum_{n\geq2}\frac{n-1}{n}x^{n-2}$, divergent in beide
uiteinden (termen gaan niet naar $0$). Slotinzicht: elke
differentiatie vermenigvuldigt de coëfficiënten met $n$, wat de
straal nooit verplaatst (meetkundig verslaat polynomiaal) maar
één orde van randverval opeet; term-voor-term calculus is een
inwendige sport, en wat er op de rand gebeurt moet opnieuw
onderzocht worden --- de Abel--Taubertheorie van het
weekendprobleem is precies dat heronderzoek.
\end{example}

\begin{example}[Een reeks splitsen via residuen --- tot het eind]\label{ex:b2:powerseries:quarter}
Bereken $f(x) = \sum_{n\geq0} \dfrac{x^{4n}}{(4n)!}$ in gesloten
vorm. Zowel $\cosh x = \sum \frac{x^{2m}}{(2m)!}$ als $\cos x =
\sum \frac{(-1)^m x^{2m}}{(2m)!}$ hebben straal $\infty$, dus
hun gemiddelde mag term voor term berekend worden:
\[
\frac{\cosh x + \cos x}{2}
= \sum_{m\geq0}\frac{1 + (-1)^m}{2}\,\frac{x^{2m}}{(2m)!}
= \sum_{m \text{ even}}\frac{x^{2m}}{(2m)!}
= \sum_{n\geq0}\frac{x^{4n}}{(4n)!} = f(x) .
\]
Het filter $\frac{1+(-1)^m}{2}$ houdt precies de even $m$: dit
is de reële avatar van het wortels-van-éénheid-filter (de
complexe versie, met $\iu^n$, extraheert residuen modulo $4$ in
één slag). Slotcontrole: $f$ lost $f'''' = f$ op met $f(0) = 1$, $f'(0) = f''(0) = f'''(0) = 0$ --- differentieer de reeks
viermaal (\cref{thm:b2:powerseries:calculus}) en zie hoe zij
zichzelf reproduceert; $\frac{\cosh + \cos}{2}$ voldoet aan
dezelfde gegevens.
\end{example}

\begin{definition}[Analytische functies]\label{def:b2:powerseries:analytic}
$f$ is \emph{analytisch}\index{analytische functie} in $x_0$
wanneer zij de som is van een machtreeks in $(x - x_0)$ op een
omgeving; op een interval, wanneer in elk punt. Sommen van
machtreeksen zijn analytisch binnen hun schijf (herschikking
van de ontwikkeling --- toegegeven op dit niveau voor
hercentrering, het geval $x_0 = 0$ zijnde
\cref{thm:b2:powerseries:calculus}). Analytisch impliceert
$C^\infty$; het omgekeerde \emph{faalt}: de platte functie
$\eu^{-1/x^2}$ (\cref{exo:b2:powerseries:7}).
\end{definition}

\begin{example}[Hercentreren, en de straal als afstand]\label{ex:b2:powerseries:recenter}
Ontwikkel $f(x) = \frac{1}{1-x}$ rond $x_0 = \frac12$: schrijf
$x = \frac12 + h$,
\[
\frac{1}{1 - x} = \frac{1}{\frac12 - h}
= \frac{2}{1 - 2h}
= \sum_{n\geq0} 2^{n+1}\,h^n
= \sum_{n\geq0} 2^{n+1}\Bigl(x - \frac12\Bigr)^{\!n},
\]
geldig voor $\abs{2h} < 1$, d.w.z.\ $\abs{x - \frac12} <
\frac12$. De nieuwe straal is precies de afstand van het nieuwe
centrum tot de singulariteit $x = 1$: hercentreren krimpt (of
groeit) de schijf om de dichtstbijzijnde obstructie te passen.
Slotinzicht: dit is het beeld achter de definitie van
analyticiteit --- één functie, vele lokale machtreeksen, elk
levend op de grootste schijf die de problemen vermijdt; het
boekdeel van Jaar~3 maakt de heuristiek ``straal $=$ afstand tot
de dichtstbijzijnde complexe singulariteit'' tot een stelling.
\end{example}

\begin{remark}[Veelgemaakte valkuilen]
\emph{(i) Het quotiëntcriterium is voldoende, niet nodig:}
wanneer $\abs{a_{n+1}/a_n}$ geen limiet heeft
(\cref{ex:b2:powerseries:sinn}, \cref{exo:b2:powerseries:1}),
ga terug naar de definitie: $R = \sup\{r : (a_nr^n)$
begrensd$\}$. \emph{(ii) Niets steekt de rand gratis over:}
term-voor-term differentiatie en integratie zijn stellingen
\emph{binnen} de open schijf; in $\abs x = R$ moet elke reeks
opnieuw onderzocht worden (dat is het hele onderwerp van het
weekendprobleem). \emph{(iii) Straal van een som:}
$\min(R_a, R_b)$ is slechts een ondergrens --- cancellaties
kunnen haar vergroten ($a_n = 1$, $b_n = -1$: som identiek $0$,
straal $\infty$). \emph{(iv) $C^\infty$ is niet \omterm{def:b2:powerseries:analytic}{analytisch}:}
een convergente Taylorreeks kan naar de \emph{verkeerde}
functie convergeren (\cref{exo:b2:powerseries:7}); vóór men
$f(x) = \sum \frac{f^{(n)}(0)}{n!}x^n$ schrijft, bewijs het ---
via een D.V.\ (\cref{met:b2:powerseries:ode}), een
resttermschatting, of een integraalformule.
\end{remark}

\begin{remark}[Waar dit gebruikt wordt]
Machtreeksen zijn het werkpaard van drie latere hoofdstukken:
het differentiaalvergelijkingenhoofdstuk lost lineaire D.V.'s
op door $\sum a_nx^n$ te injecteren (de methodedoos hieronder,
geïndustrialiseerd); het genererende-functies-hoofdstuk zet
identiteiten over waarschijnlijkheden om in identiteiten over
machtreeksen en terug; en het boekdeel van Jaar~3 maakt de
complexe variabele officieel, waar analyticiteit equivalent
wordt aan complexe differentieerbaarheid en de ``toegegeven
hercentrering'' hierboven haar eerlijke bewijs krijgt. Het
weekendprobleem verkent de ene plek waar de stellingen van dit
hoofdstuk zwijgen: de rand $\abs x = R$ zelf.
\end{remark}

\begin{method}[Ontwikkelen via een differentiaalvergelijking]\label{met:b2:powerseries:ode}
Om een functie $f$ in een machtreeks te ontwikkelen: vind een
lineaire D.V.\ met polynomiale coëfficiënten waaraan $f$
voldoet; injecteer $\sum a_nx^n$; identificeer coëfficiënten om
een recurrentie voor $(a_n)$ te krijgen; los op, en controleer
straal en beginvoorwaarden. Voorbeeld --- de binomiaalreeks:
$f(x) = (1+x)^\alpha$ voldoet aan $(1+x)f' = \alpha f$, $f(0)
= 1$; injecteren geeft $(n+1)a_{n+1} = (\alpha - n)a_n$, dus
$a_n = \binom{\alpha}{n}$, straal $1$ (quotiëntcriterium), en
de som, die aan dezelfde D.V.\ met dezelfde beginwaarde
voldoet, is gelijk aan $(1 + x)^\alpha$ door de
uniciteitsstelling voor lineaire D.V.'s (boekdeel van Jaar~1).
\end{method}

\begin{example}[De methode op een gedwongen vergelijking]\label{ex:b2:powerseries:odeforced}
Los $y' = y + x$, $y(0) = 0$, op via machtreeksen. Injecteer $y
= \sum a_nx^n$:
\[
\sum_{n\geq0}(n+1)a_{n+1}x^n
= \sum_{n\geq0}a_nx^n + x ,
\]
en coëfficiënten identificeren: $a_1 = a_0 = 0$, $2a_2 = a_1 +
1 = 1$, en $(n+1)a_{n+1} = a_n$ voor $n \geq 2$. Dus $a_2 =
\frac{1}{2!}$ en, door inductie, $a_n = \frac{1}{n!}$ voor elke
$n \geq 2$: straal $\infty$, en
\[
y(x) = \sum_{n\geq2}\frac{x^n}{n!} = \eu^x - 1 - x .
\]
Controle: $y' = \eu^x - 1 = y + x$ en $y(0) = 0$. Slotinzicht:
de recurrentie \emph{is} de vergelijking, coëfficiënt voor
coëfficiënt; de forceringsterm verstoort slechts eindig veel
begincoëfficiënten, waarna het homogene patroon overneemt ---
een discrete schaduw van ``particuliere oplossing plus
homogene oplossing''.
\end{example}

\section{Genererende functies}

\begin{example}[Fibonacci]\label{ex:b2:powerseries:fibonacci}
Zij $F(x) = \sum_{n\geq0} F_n x^n$ (Fibonaccigetallen, $F_0 =
0$, $F_1 = 1$). De recurrentie $F_{n+2} = F_{n+1} + F_n$
vertaalt zich, na vermenigvuldigen met $x^{n+2}$ en sommeren,
tot
\[
F(x) - x = x\,F(x) + x^2 F(x)
\quad\Longrightarrow\quad
F(x) = \frac{x}{1 - x - x^2} ,
\]
geldig waar de reeks convergeert. De straal is
$\frac{1}{\varphi}$: uit $F_n \sim
\frac{\varphi^n}{\sqrt5}$ (Binet, volgend voorbeeld --- of de
ruwe inductie $F_n \leq 2^n$ plus de recurrentie) geeft het
quotiëntcriterium
\[
\frac{F_{n+1}\abs x^{n+1}}{F_n\abs x^n}
\longrightarrow \varphi\abs x ,
\qquad\text{convergentie precies wanneer } \abs x < \frac1\varphi
\approx 0.618 .
\]
Partiële breuken op $\frac{x}{1 - x - x^2}$ en de meetkundige
reeks herleiden de formule van Binet --- \omterm{ex:b2:powerseries:fibonacci}{genererende functies}
industrialiseren lineaire recurrenties.\index{genererende
functie}
\end{example}

\begin{example}[De formule van Binet, uitgevoerd]\label{ex:b2:powerseries:binet}
Zij $\varphi = \frac{1+\sqrt5}{2}$ en $\psi =
\frac{1-\sqrt5}{2}$, de wortels van $X^2 = X + 1$; aangezien
$\varphi + \psi = 1$ en $\varphi\psi = -1$,
\[
1 - x - x^2 = (1 - \varphi x)(1 - \psi x) .
\]
Partiële breuken: zoekend $\frac{x}{(1-\varphi x)(1-\psi x)} =
\frac{A}{1 - \varphi x} + \frac{B}{1 - \psi x}$, de constante
term geeft $A + B = 0$ en de $x$-coëfficiënt $-A\psi -
B\varphi = 1$, dus $A(\varphi - \psi) = 1$: $A =
\frac{1}{\sqrt5} = -B$. Twee meetkundige reeksen later,
\[
F(x) = \frac{1}{\sqrt5}\sum_{n\geq0}
\bigl(\varphi^n - \psi^n\bigr)x^n
\quad\Longrightarrow\quad
F_n = \frac{\varphi^n - \psi^n}{\sqrt5}
\]
door uniciteit van coëfficiënten
(\cref{thm:b2:powerseries:calculus}). Aangezien $\abs\psi <
1$, heeft de term $\frac{\psi^n}{\sqrt5}$ absolute waarde $<
\frac12$: $F_n$ is het \emph{dichtstbijzijnde gehele getal} bij
$\frac{\varphi^n}{\sqrt5}$. Slotinzicht: de straal
$\frac1\varphi$ van $F$ is het omgekeerde van de dominante
wortel --- groei van coëfficiënten en \omterm{def:b2:powerseries:radius}{convergentiestraal} zijn
dezelfde informatie in tegengestelde richtingen gelezen.
\end{example}

\begin{example}[Catalangetallen]\label{ex:b2:powerseries:catalan}
De \omterm{ex:b2:powerseries:catalan}{Catalangetallen} $C_n$ (aantal triangulaties, van
haakjeszettingen, van Dyckpaden, \dots) voldoen aan $C_0 = 1$
en $C_{n+1} = \sum_{k=0}^n C_kC_{n-k}$. De \omterm{ex:b2:powerseries:fibonacci}{genererende functie}
$C(x) = \sum C_nx^n$ voldoet dan (Cauchyproduct!)
\[
C(x) = 1 + x\,C(x)^2
\quad\Longrightarrow\quad
C(x) = \frac{1 - \sqrt{1 - 4x}}{2x} ,
\]
kieszend de wortel met $C(0) = 1$: oplossen van de
kwadratische $xC^2 - C + 1 = 0$ geeft de twee kandidaten
$\frac{1 \pm \sqrt{1-4x}}{2x}$, en als $x \to 0$ explodeert de
``$+$''-wortel als $\frac1x$ terwijl de ``$-$''-wortel naar $1$
gaat (ontwikkel $\sqrt{1-4x} = 1 - 2x + O(x^2)$) --- alleen het
minteken kan een machtreeks met $C_0 = 1$ dragen. Ontwikkelen
van $\sqrt{1 - 4x}$ via de binomiaalreeks geeft de gesloten
vorm
\[
C_n = \frac{1}{n+1}\binom{2n}{n} ,
\]
uitgevoerd in \cref{exo:b2:powerseries:8}.\index{Catalangetallen}
\end{example}

\begin{remark}[Formele reeksen versus convergente reeksen]
Elke genererende-functie-berekening hierboven eindigt met een
beroep op de uniciteit van coëfficiënten, en die stelling leeft
\emph{binnen} een schijf van positieve straal: vóór men $F_n$
of $C_n$ ``afleest'', moet men $R > 0$ kennen. Een ruwe a
priori grens volstaat --- $F_n \leq 2^n$ (directe inductie)
geeft $R \geq \frac12$ voor Fibonacci; $C_n \leq 4^n$ (elk
Catalangetal telt deelverzamelingen van paden) geeft $R \geq
\frac14$. Pas op voor het ontaarde uiteinde van de schaal:
$\sum n!\,x^n$ heeft straal $0$, en haar als functie
manipuleren is zinloos --- identiteiten met zulke reeksen
behoren tot de \emph{formele} calculus van coëfficiënten, een
puur algebraïsch spel met eigen (andere) regels. Op dit
niveau: verzeker eerst een positieve straal, reken dan vrij
binnen haar.
\end{remark}

\begin{remark}[Perspectieven binnen dit boekdeel]
Machtreeksen zijn een van de twee grote
ontwikkelingsmachines van dit boek; de andere is de
Fourierreeks van de harmonische hoofdstukken, en ze vergelijken
is leerzaam. Een machtreeks is star: haar coëfficiënten zijn
gedwongen ($a_n = f^{(n)}(0)/n!$), haar convergentie is
meedogenloos (normaal binnen, hopeloos buiten), en haar som is
\omterm{def:b2:powerseries:analytic}{analytisch} --- oneindig star
(\cref{def:b2:powerseries:analytic}). Een Fourierreeks is
soepel: zij stelt slechts stuksgewijs-gladde signalen voor, ten
prijze van delicate convergentievragen aan de rand van
gladheid. De twee theorieën ontmoeten elkaar in het
weekendprobleem van dit hoofdstuk: Cesàro- en Abelsommatie,
hier ontwikkeld voor de randcirkel, keren terug in het
Fourierhoofdstuk als de kernen van Fejér en Poisson. Intussen
consumeert het differentiaalvergelijkingenhoofdstuk
machtreeksen direct ($\eu^{tA}$, reeksoplossingen), en het
genererende-functies-hoofdstuk maakt de truc van
\cref{ex:b2:powerseries:fibonacci} tot een systematische
calculus voor waarschijnlijkheden.
\end{remark}

Convergentiestralen:
$\sum \dfrac{n^2}{2^n}z^n$; $\;\sum \dfrac{z^n}{\binom{2n}{n}}$;
$\;\sum z^{n!}$; $\;\sum \bigl(2 + (-1)^n\bigr)^n z^n$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:powerseries:2}
Toon dat $\sum z^n$, $\sum \frac{z^n}{n}$, $\sum \frac{z^n}{n^2}$
alle straal $1$ hebben maar zich verschillend gedragen in $z =
1$ en $z = -1$: divergentie/divergentie,
divergentie/convergentie, convergentie/convergentie.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:powerseries:3}
Bereken de sommen, voor $\abs x < 1$:
\[
\sum_{n\geq0} n x^n,
\qquad
\sum_{n\geq0} n^2 x^n,
\qquad
\sum_{n\geq0} \frac{x^{2n+1}}{2n+1} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:powerseries:4}
Ontwikkel in machtreeks in $0$, met straal:
$\dfrac{1}{(1-x)(2-x)}$ (partiële breuken); $\;\ln(1 + x +
x^2)$ \emph{(schrijf $1 + x + x^2 = \frac{1 - x^3}{1 - x}$)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:powerseries:5}
Bewijs dat $f(x) = \sum_{n\geq1} H_n x^n = -\dfrac{\ln(1 -
x)}{1 - x}$ voor $\abs x < 1$, waar $H_n$ het harmonische getal
is \emph{(Cauchyproduct van $\sum x^n$ en $\sum
\frac{x^n}{n}$)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:powerseries:6}
Los via \omterm{ex:b2:powerseries:fibonacci}{genererende functie} de recurrentie $u_0 = 1$, $u_{n+1}
= 2u_n + n$ op: bereken $U(x) = \sum u_nx^n$ in gesloten vorm,
ontbind, en lees $u_n = 2^{n+1} - n - 1$ af.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:powerseries:7}
Zij $f(x) = \eu^{-1/x^2}$ voor $x \neq 0$, $f(0) = 0$. Bewijs
dat $f$ van klasse $C^\infty$ is op $\R$ met $f^{(n)}(0) = 0$
voor alle $n$ \emph{(toon door inductie dat $f^{(n)}(x) =
P_n\bigl(\frac1x\bigr) \eu^{-1/x^2}$ voor veeltermen $P_n$, en
gebruik de groeivergelijking)}. Concludeer dat $f$ niet
\omterm{def:b2:powerseries:analytic}{analytisch} is in $0$: haar Taylorreeks in $0$ convergeert ---
naar de verkeerde functie.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:powerseries:8}
Voltooi \cref{ex:b2:powerseries:catalan}: ontwikkel $\sqrt{1 -
4x}$ met de binomiaalreeks, toon
\[
\binom{1/2}{n+1}(-4)^{n+1} = -\frac{2}{n+1}\binom{2n}{n},
\]
en leid $C_n = \frac{1}{n+1}\binom{2n}{n}$ af; bepaal de
\omterm{def:b2:powerseries:radius}{convergentiestraal} van $C(x)$ en de asymptotiek van $C_n$ via
Stirling.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:powerseries:9}
(Abels radiale limietstelling, speciaal geval) Stel $\sum a_n$
convergeert. Bewijs dat $\lim_{x \to 1^-} \sum_{n} a_n x^n =
\sum_n a_n$. \emph{(Abelsommatie: met $A_n$ de partiële sommen
en $A = \lim A_n$, schrijf $\sum a_nx^n = (1 - x)\sum A_n x^n$;
dan $\sum a_nx^n - A = (1-x)\sum (A_n - A)x^n$, splits de som
bij een grote $N$.)} Toepassing: $\sum \frac{(-1)^{n-1}}{n} =
\ln 2$ en $\sum \frac{(-1)^n}{2n+1} = \frac\pi4$, herbewezen
vanuit de machtreeksen.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:powerseries:10}
Toon dat $\displaystyle\sum_{n\geq1}\frac{x^n}{n(n+1)} = 1 +
\frac{1-x}{x}\,\ln(1-x)$ voor $0 < \abs x < 1$, bepaal de
straal, en controleer dat de convergentie normaal is op
$\intcc{-1}{1}$; verifieer dat de waarde in $x = 1$ voorspeld
door continuïteit overeenkomt met de telescoperende som $\sum
\frac{1}{n(n+1)} = 1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:powerseries:11}
(Derangements) Zij $D_n$ het aantal permutaties van $n$
objecten zonder vast punt ($D_0 = 1$). Sorteren van de
permutaties van $\{1, \dots, n\}$ naar hun vaste-puntenverzameling
geeft $n! = \sum_{k=0}^{n}\binom nk D_{n-k}$. Vermenigvuldig
met $\frac{x^n}{n!}$, sommeer, en herken een Cauchyproduct om
de exponentiële \omterm{ex:b2:powerseries:fibonacci}{genererende functie}
\[
\sum_{n\geq0} D_n\,\frac{x^n}{n!} = \frac{\eu^{-x}}{1-x}
\qquad (\abs x < 1),
\]
te verkrijgen, lees dan de gesloten vorm $\dfrac{D_n}{n!} =
\sum_{k=0}^{n}\dfrac{(-1)^k}{k!}$ af en de limiet
$\dfrac{D_n}{n!} \to \eu^{-1}$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:powerseries:12}
Bewijs, met de binomiaalreeks van
\cref{met:b2:powerseries:ode}, dat
\[
\frac{1}{\sqrt{1 - 4x}} = \sum_{n\geq0}\binom{2n}{n}x^n
\qquad \Bigl(\abs x < \frac14\Bigr),
\]
en leid, door kwadrateren (Cauchyproduct tegen
$\frac{1}{1-4x} = \sum 4^nx^n$), de convolutie-identiteit
\[
\sum_{k=0}^{n}\binom{2k}{k}\binom{2n-2k}{n-k} = 4^n .
\]
\end{exercise}

\section{Probleem: Abel, Tauber, en de rand van convergentie}

\begin{problem}\label{pb:b2:powerseries:1}
Binnen de convergentieschijf is alles gemakkelijk; al het
drama van machtreeksen speelt zich \emph{op} de rand af. Dit
probleem bouwt de randtheorie in de reële variabele: de stelling
van Abel in haar uniforme vorm, haar omkering onder de
voorwaarde van Tauber, de Cesàro--Abel-hiërarchie van
sommatiemethoden (met de stelling van Frobenius), term-voor-term
integratie tot de rand met klassieke constanten als
dividenden, en ten slotte de starheid van \omterm{def:b2:powerseries:analytic}{analytische functies}
--- de identiteitsstelling.\index{stelling van
Abel}\index{stelling van Tauber} Doorheen is $(a_n)$ een reële
rij, $f(x) = \sum_{n\geq0} a_nx^n$, en $A_n = a_0 + \dots +
a_n$.

\medskip
\noindent\textbf{Deel I --- De stelling van Abel, uniform.}
Neem in dit deel aan dat $\sum a_n$ convergeert, en zet $r_n =
\sum_{k\geq n} a_k$ (dus $r_n \to 0$ en $a_n = r_n -
r_{n+1}$).
\begin{enumerate}
  \item Bewijs, via partiële sommatie, dat voor alle $0 \leq x
        \leq 1$ en $N \leq M$:
        \[
        \Bigl|\sum_{n=N}^{M} a_n x^n\Bigr|
        \leq 2\sup_{n \geq N}\,\abs{r_n} .
        \]
  \item Leid af dat $\sum a_nx^n$ \emph{uniform} convergeert op
        $\intcc{0}{1}$, dat haar som daar continu is, en herwin
        de radiale limiet van \cref{exo:b2:powerseries:9}:
        $f(x) \to \sum a_n$ als $x \to 1^-$.
  \item (Stelling van Abel voor Cauchyproducten) Zij $\sum a_n
        = A$, $\sum b_n = B$ en stel dat het Cauchyproduct
        $\sum c_n$, $c_n = \sum_{k} a_kb_{n-k}$,
        \emph{convergeert}, met som $C$. Bewijs $C = AB$
        \emph{(binnen de schijf geldt de productidentiteit door
        \cref{prop:b2:powerseries:operations}; laat $x \to
        1^-$)}.
  \item Toon dat de hypothese ertoe doet: voor $a_n = b_n =
        \frac{(-1)^n}{\sqrt{n+1}}$ convergeren beide reeksen,
        toch $\abs{c_n} \geq \frac{2(n+1)}{n+2} \geq 1$
        \emph{(boundeer elke factor
        $\sqrt{(k+1)(n-k+1)}$ via AM--GM)}: het Cauchyproduct
        van twee convergente reeksen kan divergeren.
  \item (Een dividend van \cref{exo:b2:powerseries:5}) Toon dat
        $\bigl(\ln(1-x)\bigr)^2 = 2\sum_{n\geq1}
        \frac{H_n}{n+1}x^{n+1}$ op $\intoo{-1}{1}$, controleer
        dat $\bigl(\frac{H_n}{n+1}\bigr)_{n\geq1}$ daalt naar
        $0$, en concludeer met Abel:
        \[
        \sum_{n\geq1} (-1)^{n+1}\,\frac{H_n}{n+1}
        = \frac{(\ln 2)^2}{2} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De omkering van Tauber.} Noem
$\sum a_n$ \emph{\omterm{thm:b2:series:abel}{Abel-sommeerbaar}} tot $L$ wanneer $f(x) \to L$
als $x \to 1^-$.
\begin{enumerate}[resume]
  \item Toon dat $\sum (-1)^n$ \omterm{thm:b2:series:abel}{Abel-sommeerbaar} is tot
        $\frac12$ doch divergent: de stelling van Abel heeft
        geen onvoorwaardelijke omkering.
  \item (Lemma van Cesàro) Als $u_n \to 0$ dan $\frac{u_1 +
        \dots + u_N}{N} \to 0$ \emph{(splits de som bij een
        vaste $m$)}.
  \item Neem nu $n\,a_n \to 0$ en $f(x) \to L$ aan. Met $x_N =
        1 - \frac1N$, bewijs de twee schattingen
        \[
        \Bigl|\sum_{n=0}^{N} a_n\bigl(1 - x_N^n\bigr)\Bigr|
        \leq \frac{1}{N}\sum_{n=1}^{N} n\,\abs{a_n},
        \qquad
        \Bigl|\sum_{n>N} a_n x_N^n\Bigr|
        \leq \sup_{n>N}\bigl(n\abs{a_n}\bigr)
        \]
        \emph{(voor de eerste, $1 - x^n \leq n(1-x)$; voor de
        tweede, $\abs{a_n} \leq \frac{1}{N}\sup_{m>N}
        m\abs{a_m}$ en $\sum x_N^n \leq N$)}.
  \item Concludeer de \emph{stelling van Tauber}: als $n\,a_n
        \to 0$ en $\sum a_n$ \omterm{thm:b2:series:abel}{Abel-sommeerbaar} is tot $L$, dan
        convergeert $\sum a_n$ naar $L$.
  \item (De gemakkelijke Tauberiaanse voor positieve
        coëfficiënten) Als $a_n \geq 0$ en $f$ begrensd is op
        $\intco{0}{1}$, toon dat $\sum a_n$ convergeert en
        $\sum a_n = \lim_{x\to1^-} f(x)$ \emph{(boundeer
        $\sum_{n\leq N}a_nx^n \leq f(x)$ en laat $x \to 1^-$,
        gebruik dan Abel)}.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Cesàromiddelden en de stelling van
Frobenius.}
Noem $\sum a_n$ \emph{\omterm{thm:b2:series:abel}{Cesàro-sommeerbaar}} tot $L$ wanneer
$\sigma_N = \frac{A_0 + \dots + A_{N-1}}{N} \to L$.
\begin{enumerate}[resume]
  \item Toon dat een convergente reeks \omterm{thm:b2:series:abel}{Cesàro-sommeerbaar} is tot
        haar som \emph{(vraag~7 toegepast op $A_n - L$)}.
  \item Bereken de Cesàro-waarde van $\sum(-1)^n$ en controleer
        dat zij overeenkomt met de Abel-waarde $\frac12$ van
        vraag~6.
  \item Met $S_n = A_0 + \dots + A_n = (n+1)\,\sigma_{n+1}$,
        bewijs de twee identiteiten, voor $0 \leq x < 1$:
        \[
        f(x) = (1-x)^2\sum_{n\geq0}(n+1)\,\sigma_{n+1}x^n,
        \qquad
        (1-x)^2\sum_{n\geq0}(n+1)x^n = 1 .
        \]
  \item (Frobenius) Leid af: als $\sigma_N \to L$ dan $f(x)
        \to L$ als $x \to 1^-$ --- \omterm{thm:b2:series:abel}{Cesàro-sommeerbaar}
        impliceert \omterm{thm:b2:series:abel}{Abel-sommeerbaar}, tot dezelfde waarde
        \emph{(trek de twee identiteiten af en splits de som bij
        een grote $N$, als in
        \cref{exo:b2:powerseries:9})}.
  \item Toon dat de hiërarchie
        \[
        \text{convergent} \;\Longrightarrow\;
        \text{Cesàro-sommeerbaar} \;\Longrightarrow\;
        \text{Abel-sommeerbaar}
        \]
        strikt is bij beide pijlen: vraag~6 voor de eerste; voor
        de tweede, toon dat $\sum(-1)^n(n+1)$ \omterm{thm:b2:series:abel}{Abel-sommeerbaar}
        is tot $\frac14$ (bereken $f$) maar niet
        \omterm{thm:b2:series:abel}{Cesàro-sommeerbaar} (bereken $\sigma_N$ apart voor even
        en oneven $N$).
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Integreren tot de rand.}
\begin{enumerate}[resume]
  \item Stel $\sum a_nx^n$ convergeert op $\intco{0}{1}$ en
        $\sum \frac{a_n}{n+1}$ convergeert. Bewijs dat de
        \omterm{def:b2:integration:improper}{oneigenlijke integraal} $\int_0^1 f$ bestaat en
        \[
        \int_0^1 \Bigl(\sum_{n\geq0} a_nx^n\Bigr)\dd x
        = \sum_{n\geq0}\frac{a_n}{n+1}
        \]
        \emph{(de primitieve $F(x) = \sum\frac{a_n}{n+1}x^{n+1}$
        is continu in $1$ door Deel~I)}.
  \item Zij $\eta = \sum_{n\geq1}\frac{(-1)^{n-1}}{n^2}$. Toon
        $\int_0^1 \frac{\ln(1+x)}{x}\dd x = \eta$ en, door even
        en oneven indices te splitsen in de absoluut convergente
        $\sum \frac1{n^2}$, dat $\eta =
        \frac12\sum_{n\geq1}\frac{1}{n^2}$. (Het
        weekendprobleem van het Fourierhoofdstuk evalueert
        $\sum\frac1{n^2} = \frac{\pi^2}{6}$.)
  \item Bewijs
        \[
        \sum_{n\geq0}\frac{(-1)^n}{3n+1}
        = \int_0^1\frac{\dd x}{1+x^3}
        = \frac13\Bigl(\ln 2 + \frac{\pi}{\sqrt3}\Bigr)
        \]
        \emph{(de reeks convergeert door Leibniz; integreer de
        meetkundige reeks $\sum(-1)^nx^{3n}$ met vraag~16; dan
        partiële breuken: $\frac{1}{1+x^3} =
        \frac{1/3}{1+x} + \frac{(2-x)/3}{x^2-x+1}$)}.
  \item Uit de binomiaalreeks voor $(1-t)^{-1/2}$
        (\cref{exo:b2:powerseries:12}) leid af
        \[
        \arcsin x = \sum_{n\geq0}
        \frac{\binom{2n}{n}}{4^n(2n+1)}\,x^{2n+1}
        \quad(\abs x < 1),
        \qquad\text{dan}\qquad
        \sum_{n\geq0}\frac{\binom{2n}{n}}{4^n(2n+1)}
        = \frac\pi2 ,
        \]
        de randwaarde rechtvaardigend door \emph{normale}
        convergentie op $\intcc{-1}{1}$ (gebruik
        $\binom{2n}n4^{-n} \sim \frac{1}{\sqrt{\pi n}}$,
        \cref{ex:b2:comparison:centralbinomial}) --- hier is
        zelfs Abel niet nodig.
  \item (Catalan op de rand) Toon dat $\sum C_n 4^{-n} = 2$: de
        Catalanreeks van \cref{ex:b2:powerseries:catalan}
        convergeert \emph{in} haar straal $\frac14$
        (asymptotiek van \cref{exo:b2:powerseries:8}), haar som
        is continu op $\intcc{0}{\frac14}$, en de gesloten vorm
        heeft limiet $2$ daar.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Starheid: de
identiteitsstelling.}
\begin{enumerate}[resume]
  \item (Geïsoleerde nulpunten) Zij $f = \sum a_nx^n$ met straal
        $R > 0$ en niet alle $a_n = 0$; zij $m$ de kleinste
        index met $a_m \neq 0$. Toon $f(x) = x^m g(x)$ met $g$
        een machtreeks van straal $R$, $g(0) = a_m \neq 0$, en
        leid af dat $f$ geen nulpunt heeft in enige gepuncteerde
        omgeving van $0$.
  \item (Identiteitsstelling) Zij $f, h$ sommen van
        machtreeksen bij $0$ en $(x_k)$ een rij van
        \emph{niet-nule} punten met $x_k \to 0$ en $f(x_k) =
        h(x_k)$. Bewijs dat $f$ en $h$ dezelfde coëfficiënten
        hebben, dus samenvallen bij $0$.
  \item Vind \emph{alle} functies $f$ \omterm{def:b2:powerseries:analytic}{analytisch} bij $0$ met
        \[
        f\Bigl(\frac1k\Bigr) = \frac{k^2}{k^2+1}
        \qquad\text{voor alle grote gehele } k .
        \]
  \item Toon dat een functie \omterm{def:b2:powerseries:analytic}{analytisch} op een open interval
        $I$ die verdwijnt op een subinterval identiek verdwijnt
        op $I$ \emph{(de verzameling van punten waaromheen $f$
        identiek verdwijnt is open en, door de
        identiteitsstelling toegepast in ophopingspunten,
        gesloten in $I$)}. Concludeer dat geen niet-nule
        \omterm{def:b2:powerseries:analytic}{analytische functie} op $\R$ compacte drager heeft ---
        terwijl $C^\infty$-bultfuncties bestaan
        (\cref{exo:b2:powerseries:7} levert de bouwsteen):
        analyticiteit is star, gladheid is slap.
  \item Synthese. In één zin elk: (i) wat de stelling van Abel
        toevoegt aan het normale-convergentiepakket van
        \cref{lem:b2:powerseries:abel}; (ii) de exacte
