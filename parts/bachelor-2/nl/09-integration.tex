\chapter{Integratie}\label{ch:b2:integration}

Het boekdeel van Jaar~1 bouwde de integraal op een segment. Dit
hoofdstuk breidt haar uit tot willekeurige intervallen (oneigenlijke
integralen, met de volledige vergelijkingsgereedschapskist), en
bestudeert dan integralen \emph{afhankelijk van een parameter} ---
continuïteit en differentiatie onder het integraalteken --- aangedreven
door de stelling van de gedomineerde convergentie, het ene resultaat van
dit hoofdstuk dat op vertrouwen wordt genomen. De $\Gamma$-functie
dient als lopend voorbeeld, en als de poort naar de helft van de
speciale functies van de wiskunde.

\section{Integralen op een willekeurig interval}

\begin{definition}\label{def:b2:integration:improper}
Zij $f$ stuksgewijs continu op $\intco{a}{b}$ ($b \in \R$ of
$+\infty$). De integraal \emph{convergeert}\index{oneigenlijke
integraal} wanneer $\lim_{x \to b^-} \int_a^x f$ bestaat; men
schrijft dan $\int_a^b f$ voor de limiet. (Evenzo op $\intoc{a}{b}$,
en op $\intoo{a}{b}$ door splitsing in een inwendig punt --- de
keuze doet er niet toe, door Chasles.) De integraal \emph{convergeert
absoluut} wanneer $\int_a^b \abs f$ convergeert; absolute
convergentie impliceert convergentie, door het Cauchycriterium:
\[
\Bigl| \int_x^{y} f \Bigr| \leq \int_x^{y} \abs f
\]
en de volledigheid van $\R$ (de primitieve heeft de
Cauchy-eigenschap). In detail: zij $F(x) = \int_a^x f$ en $G(x) =
\int_a^x \abs f$. Als $\int^b\abs f$ convergeert, heeft $G$ een
limiet in $b^-$, dus voor elke $\varepsilon > 0$ bestaat $c < b$ met
$G(y) - G(x) \leq \varepsilon$ wanneer $c \leq x \leq y < b$; de
weergave draagt deze Cauchy-eigenschap over op $F$. Voor elke rij
$x_n \to b^-$ vormen de waarden $F(x_n)$ dan een Cauchyrij van
reëlen, convergent door volledigheid, en ineenvlechten van twee
zulke rijen toont dat de limiet voor alle hetzelfde is: $F$ heeft
een limiet in $b^-$.
\end{definition}

\begin{theorem}[Positieve vergelijkingsgereedschapskist]\label{thm:b2:integration:comparison}
Voor $f, g \geq 0$ stuksgewijs continu op $\intco{a}{b}$:
\begin{enumerate}
  \item $\int_a^b f$ \omterm{def:b2:integration:improper}{convergeert} d.e.s.d.a.\ de primitieve $x
        \mapsto \int_a^x f$ begrensd is;
  \item $f \leq g$: convergentie van $\int g$ dwingt die van
        $\int f$; divergentie draagt de andere kant op over;
  \item $f \sim g$ in $b$: de twee integralen hebben dezelfde
        aard;
  \item de referentieschalen: in $+\infty$ \omterm{def:b2:integration:improper}{convergeert}
        $\int^{\infty} \frac{\dd t}{t^\alpha}$ d.e.s.d.a.\ $\alpha
        > 1$, en $\int^\infty \frac{\dd t}{t(\ln t)^\beta}$
        d.e.s.d.a.\ $\beta > 1$; in een eindig eindpunt $b$
        \omterm{def:b2:integration:improper}{convergeert} $\int^b \frac{\dd t}{(b - t)^\alpha}$
        d.e.s.d.a.\ $\alpha < 1$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) De primitieve $F(x) = \int_a^x f$ is niet-dalend ($f \geq 0$).
Als zij begrensd is, is $\ell = \sup_{x < b}F$ eindig en $F(x) \to
\ell$: gegeven $\varepsilon > 0$, enige $F(x_0) > \ell -
\varepsilon$, en monotonie vangt $F(x) \in \intoc{\ell -
\varepsilon}{\ell}$ voor $x_0 \leq x < b$. Als zij onbegrensd is,
$F \to +\infty$: divergentie.

(2) Uit $f \leq g$: $\int_a^x f \leq \int_a^x g$ voor alle $x$;
als $\int^b g$ \omterm{def:b2:integration:improper}{convergeert}, is de rechterkant begrensd, dus ook
de linkerkant, en (1) concludeert. Contrapositie draagt
divergentie de andere kant op.

(3) $f \sim g$ in $b$ levert $c < b$ met
\[
\tfrac12\,g(t) \;\leq\; f(t) \;\leq\; 2\,g(t)
\qquad (c \leq t < b) :
\]
door (2) beide kanten op $\intco{c}{b}$ hebben de twee integralen
dezelfde aard; het beginstuk $\intcc{a}{c}$ is een eigenlijke
integraal en verandert niets.

(4) Expliciete primitieven: voor $\alpha \neq 1$ en $\beta \neq 1$,
\[
\int_c^x \frac{\dd t}{t^\alpha}
= \frac{x^{1-\alpha} - c^{1-\alpha}}{1 - \alpha},
\qquad
\int_c^x \frac{\dd t}{t(\ln t)^\beta}
= \frac{(\ln x)^{1-\beta} - (\ln c)^{1-\beta}}{1 - \beta},
\]
met logaritmen in de uitgesloten gevallen: begrensd als $x \to
+\infty$ precies wanneer $\alpha > 1$, resp.\ $\beta > 1$. In
een eindig eindpunt reduceert de substitutie $u = b - t$ tot de
schaal $\int_0 u^{-\alpha}\,\dd u$, begrensd d.e.s.d.a.\ $\alpha
< 1$. Pas (1) telkens toe.
\end{proof}

\begin{example}[Twee opwarmers, tot het eind uitgewerkt]\label{ex:b2:integration:warmups}
\emph{(a)} $\displaystyle\int_0^1 \ln t\,\dd t$: de integrand
blaast op in $0^+$, maar $\abs{\ln t} = o\bigl(t^{-1/2}\bigr)$
daar (logaritmen verliezen van machten), en $\int_0 t^{-1/2}$
\omterm{def:b2:integration:improper}{convergeert}: absolute convergentie. De waarde, via partiële
integratie op $\intcc{\varepsilon}{1}$:
\[
\int_\varepsilon^1 \ln t\,\dd t
= \bigl[t\ln t - t\bigr]_\varepsilon^1
= -1 - \varepsilon\ln\varepsilon + \varepsilon
\xrightarrow[\varepsilon\to0^+]{} -1 .
\]
\emph{(b)} $\displaystyle\int_0^\infty \frac{\ln t}{1 +
t^2}\,\dd t$: problemen aan beide einden, dus splits in $1$.
Nabij $0$: $\abs{\ln t}$ integreerbaar als in (a); nabij
$\infty$: $\frac{\ln t}{1+t^2} = o(t^{-3/2})$: absoluut
convergent. De substitutie $t = \frac1u$ beeldt $\intoo{0}{1}$
af op $\intoo{1}{\infty}$ en
\[
\int_0^1 \frac{\ln t}{1+t^2}\,\dd t
= \int_1^{\infty} \frac{-\ln u}{1 + u^{-2}}\cdot
\frac{\dd u}{u^2}
= -\int_1^\infty \frac{\ln u}{1+u^2}\,\dd u :
\]
de twee helften cancelen, en de integraal is $0$. Afsluitend
inzicht: symmetrie onder $t \mapsto \frac1t$ is een pagina
berekening waard --- dezelfde truc dreef al
\cref{exo:b2:integration:3} aan.
\end{example}

\begin{example}[Eén waarde, drie integralen]\label{ex:b2:integration:onecos}
Bestudeer $I = \displaystyle\int_0^{\infty} \frac{1 - \cos
t}{t^2}\,\dd t$. In $0$: $1 - \cos t \sim \frac{t^2}2$, dus de
integrand zet zich continu voort met waarde $\frac12$ --- geen
singulariteit. In $\infty$: $0 \leq \frac{1 - \cos t}{t^2}
\leq \frac{2}{t^2}$: absolute convergentie
(\cref{thm:b2:integration:comparison}). Waarde: partiële
integratie op $\intcc{\varepsilon}{M}$ met $u = 1 - \cos t$, $v'
= t^{-2}$:
\[
\int_\varepsilon^M \frac{1 - \cos t}{t^2}\,\dd t
= \Bigl[-\frac{1 - \cos t}{t}\Bigr]_\varepsilon^M
+ \int_\varepsilon^M \frac{\sin t}{t}\,\dd t .
\]
De haak verdwijnt aan beide einden ($\frac{1 -
\cos\varepsilon}{\varepsilon} \sim \frac\varepsilon2$;
begrensde teller in $M$), en de integraal streeft naar de
Dirichlet-waarde $\frac\pi2$ (\cref{exo:b2:integration:10}): $I
= \frac\pi2$. Afsluitend inzicht: met $1 - \cos t =
2\sin^2\frac t2$ en $u = \frac t2$,
\[
I = \int_0^\infty \frac{2\sin^2 u}{(2u)^2}\;2\,\dd u
= \int_0^\infty \Bigl(\frac{\sin u}{u}\Bigr)^{\!2}\dd u :
\]
de drie klassiekers $\int_0^\infty\frac{\sin t}{t}\dd t$,
$\int_0^\infty\bigl(\frac{\sin t}{t}\bigr)^2\dd t$
(\cref{exo:b2:integration:11}) en $I$ delen allen de waarde
$\frac\pi2$, doorgegeven via partiële integratie en substitutie
--- en enkel de eerste is semi-convergent: partiële integratie
ruilde de absolute convergentie in voor een eenvoudiger
integrand.
\end{example}

\begin{example}[Een semi-convergente integraal]\label{ex:b2:integration:sint}
$\displaystyle\int_1^{\infty} \frac{\sin t}{t}\,\dd t$
\omterm{def:b2:integration:improper}{convergeert}: partiële integratie,
\[
\int_1^x \frac{\sin t}{t}\dd t
= \Bigl[\frac{-\cos t}{t}\Bigr]_1^x - \int_1^x \frac{\cos
t}{t^2}\dd t ,
\]
waar de haak een limiet heeft en de laatste integraal absoluut
\omterm{def:b2:integration:improper}{convergeert} ($\abs{\cos t}/t^2 \leq t^{-2}$). Maar niet absoluut:
uit $\abs{\sin t} \geq \sin^2 t$,
\[
\int_1^x \frac{\abs{\sin t}}{t}\,\dd t
\;\geq\; \int_1^x \frac{\sin^2t}{t}\,\dd t
= \underbrace{\int_1^x \frac{\dd t}{2t}}_{=\ \frac12\ln x
\ \to\ \infty}
\;-\; \underbrace{\int_1^x \frac{\cos 2t}{2t}\,\dd
t}_{\text{convergent}} ,
\]
waar de laatste integraal \omterm{def:b2:integration:improper}{convergeert} door \emph{dezelfde}
partiële integratie als hierboven (met $\sin 2t$ in de haak):
een divergent stuk minus een convergent divergeert. Dus
\omterm{def:b2:integration:improper}{convergeert} $\int_1^\infty\frac{\sin t}{t}\dd t$ zonder
absoluut te convergeren --- de integraalanaloog van de
alternerende reeks, met partiële integratie in de rol van het
alternerend criterium.
\end{example}

\section{De convergentie\-stelling}

\begin{theorem}[Gedomineerde convergentie]\label{thm:b2:integration:dominated}
Zij $(f_n)$ stuksgewijs continu op een interval $I$, puntsgewijs
convergent naar een stuksgewijs continue $f$, en veronderstel dat
er een vaste integreerbare $\varphi \geq 0$ is ($\int_I \varphi <
\infty$) met\index{stelling van de gedomineerde convergentie}
\[
\abs{f_n(t)} \leq \varphi(t) \qquad (t \in I,\ n \in \N).
\]
Dan convergeren alle $\int_I f_n$ en $\int_I f$ absoluut, en
\[
\int_I f_n \xrightarrow[n \to \infty]{} \int_I f .
\]
\end{theorem}
\admitted

\begin{remark}
Het eerlijke bewijs behoort tot de Lebesgue-integratietheorie van
Jaar~3; de uitspraak wordt echter voortdurend vanaf nu gebruikt.
De \emph{dominantie}hypothese is het hele punt: puntsgewijze
convergentie alleen volstaat niet ($f_n = n\,\mathbf{1}_{\intoo{0}{1/n}}$,
glijdende bulten: $\int f_n = 1 \not\to 0 = \int f$). De
stelling geldt ook voor een continue parameter ($f_\lambda$,
$\lambda \to \lambda_0$), via de sequentiële karakterisering van
limieten.
\end{remark}

\begin{example}[Een Gaussische limiet, via dominantie]\label{ex:b2:integration:wallislimit}
Bereken $\displaystyle\lim_{n\to\infty} I_n$ waar $I_n =
\int_0^\infty \Bigl(1 + \frac{t^2}{n}\Bigr)^{\!-n}\dd t$.
Puntsgewijs, $(1 + t^2/n)^n \to \eu^{t^2}$
(samengestelde-interestlimiet), dus de integranden streven naar
$\eu^{-t^2}$. Dominantie: de rij $n \mapsto (1 + u/n)^n$ is
niet-dalend voor $u \geq 0$ (AM--GM op de $n + 1$ factoren $1, 1
+ \frac un, \dots, 1 + \frac un$ geeft $(1 + \frac
u{n+1})^{n+1} \geq (1 + \frac un)^n$), dus voor $n \geq 2$:
\[
\Bigl(1 + \frac{t^2}{n}\Bigr)^{\!-n}
\leq \Bigl(1 + \frac{t^2}{2}\Bigr)^{\!-2},
\]
een integreerbare dominator ($\sim 4t^{-4}$ in oneindig).
Gedomineerde convergentie:
\[
I_n \xrightarrow[n\to\infty]{} \int_0^\infty \eu^{-t^2}\dd t
= \frac{\sqrt\pi}{2}
\]
(de Gaussische integraal van \cref{exo:b2:integration:8}).
Afsluitende controle: de substitutie $t = \sqrt n\tan\theta$
berekent $I_n$ exact, $I_n =
\sqrt n\int_0^{\pi/2}\cos^{2n-2}\theta\,\dd\theta =
\sqrt n\,W_{2n-2}$, en de Wallis-asymptotiek $W_m \sim
\sqrt{\pi/(2m)}$ (\cref{lem:b2:comparison:wallis}) geeft $\sqrt
n\,W_{2n-2} \to \frac{\sqrt\pi}2$ opnieuw: de twee pijlers van
dit hoofdstuk en het vorige komen overeen.
\end{example}

\begin{example}[Gedomineerde convergentie, continue parameter]
\label{ex:b2:integration:arctanlimit}
Bereken
\[
\lim_{x\to+\infty}\int_0^\infty
\frac{\arctan(xt)}{1+t^2}\,\dd t .
\]
Voor elke $t > 0$ is $\arctan(xt) \to \frac\pi2$ als $x \to
\infty$; en de dominantie
\[
\Bigl|\frac{\arctan(xt)}{1+t^2}\Bigr|
\leq \frac{\pi/2}{1+t^2},
\qquad\text{integreerbaar, onafhankelijk van } x,
\]
geldt voor alle $x$. Door de continue-parametervorm van
\cref{thm:b2:integration:dominated} (sequentiële
karakterisering: test langs elke $x_n \to \infty$),
\[
\int_0^\infty\frac{\arctan(xt)}{1+t^2}\,\dd t
\xrightarrow[x\to+\infty]{}
\frac\pi2\int_0^\infty\frac{\dd t}{1+t^2}
= \frac{\pi^2}{4} .
\]
Afsluitend inzicht: het enkele punt $t = 0$, waar de
puntsgewijze limiet $0$ is eerder dan $\frac\pi2$, verandert
niets --- de limietfunctie treedt enkel via haar integraal
binnen, een van de stille genaden van de stelling.
\end{example}

\section{Integralen met een parameter}

\begin{theorem}[Continuïteit onder het integraalteken]\label{thm:b2:integration:continuity}
Zij $f \colon A \times I \to \R$ ($A$ een \omterm{def:b2:metric:def}{metrische ruimte}, $I$
een interval) met: $t \mapsto f(x, t)$ stuksgewijs continu voor
elke $x$; $x \mapsto f(x, t)$ continu voor elke $t$; en een
\emph{dominantie} $\abs{f(x,t)} \leq \varphi(t)$ ($\varphi$
integreerbaar op $I$, onafhankelijk van $x$). Dan is
\[
F(x) = \int_I f(x, t)\,\dd t
\]
gedefinieerd en continu op $A$.\index{parameterintegraal}
\end{theorem}

\begin{proof}
Gedefinieerdheid: dominantie geeft absolute convergentie.
Continuïteit in $x_0$: voor elke rij $x_n \to x_0$ convergeren
de functies $g_n(t) = f(x_n, t)$ puntsgewijs naar $f(x_0, t)$
(continuïteit in $x$) onder de vaste dominantie $\varphi$:
gedomineerde convergentie geeft $F(x_n) \to F(x_0)$;
concludeer via de sequentiële karakterisering van continuïteit
(\cref{def:b2:metric:continuity}).
\end{proof}

\begin{theorem}[Differentiatie onder het integraalteken]\label{thm:b2:integration:leibnizrule}
Zij $f \colon J \times I \to \R$ ($J$ een parameterinterval)
met: $t \mapsto f(x,t)$ integreerbaar op $I$ voor elke $x$; $x
\mapsto f(x,t)$ van klasse $C^1$ voor elke $t$, de partiële
afgeleide $\frac{\partial f}{\partial x}$ stuksgewijs continu in
$t$ en gedomineerd: $\bigl|\frac{\partial f}{\partial
x}(x,t)\bigr| \leq \psi(t)$ met $\psi$ integreerbaar. Dan is
$F(x) = \int_I f(x,t)\dd t$ van klasse $C^1$ op $J$ en
\[
F'(x) = \int_I \frac{\partial f}{\partial x}(x, t)\,\dd t .
\]
\end{theorem}

\begin{proof}
Fixeer $x$ en $h_n \to 0$. De differentiequotiënten
\[
\frac{F(x + h_n) - F(x)}{h_n}
= \int_I \frac{f(x + h_n, t) - f(x, t)}{h_n}\,\dd t
\]
hebben integranden die puntsgewijs naar $\frac{\partial
f}{\partial x}(x, t)$ convergeren, en gedomineerd door
$\psi(t)$: door de middelwaarde-ongelijkheid in $x$ bij vaste
$t$,
\[
\Bigl|\frac{f(x + h_n, t) - f(x,t)}{h_n}\Bigr|
\leq \sup_{\xi} \Bigl|\frac{\partial f}{\partial x}(\xi, t)\Bigr|
\leq \psi(t) .
\]
Gedomineerde convergentie geeft de limiet $\int_I
\frac{\partial f}{\partial x}(x,t)\dd t$ van de quotiënten: $F$
is differentieerbaar met de aangekondigde afgeleide, die
continu is door \cref{thm:b2:integration:continuity} toegepast
op $\frac{\partial f}{\partial x}$.
\end{proof}

\begin{example}[Een parameterintegraal gecontroleerd tegen een formule]\label{ex:b2:integration:arctanparam}
Zij $F(x) = \displaystyle\int_0^\infty \frac{\dd t}{t^2 + x}$
voor $x > 0$. Op elk $\intcc{a}{b} \subset \intoo{0}{\infty}$
wordt de integrand gedomineerd door $\frac{1}{t^2 + a}$,
integreerbaar en onafhankelijk van $x$: $F$ is continu
(\cref{thm:b2:integration:continuity}). Hier kan de stelling
tegen een expliciete waarde gecontroleerd worden:
\[
F(x) = \Bigl[\frac{1}{\sqrt x}\arctan\frac{t}{\sqrt
x}\Bigr]_0^\infty = \frac{\pi}{2\sqrt x} ,
\]
zichtbaar continu. Differentieer nu onder de integraal: de
$x$-afgeleide $-\frac{1}{(t^2+x)^2}$ wordt op $\intcc ab$
gedomineerd door $\frac{1}{(t^2+a)^2}$, integreerbaar:
\cref{thm:b2:integration:leibnizrule} geeft
\[
F'(x) = -\int_0^\infty \frac{\dd t}{(t^2 + x)^2}
\qquad\text{terwijl}\qquad
F'(x) = -\frac{\pi}{4}\,x^{-3/2} ,
\]
dus hebben we \emph{gratis} een nieuwe integraal berekend:
$\int_0^\infty\frac{\dd t}{(t^2+x)^2} = \frac{\pi}{4x^{3/2}}$.
Afsluitend inzicht: differentiëren van een bekende
\omterm{thm:b2:integration:continuity}{parameterintegraal} is een fabriek van nieuwe formules ---
itereren geeft $\int_0^\infty\frac{\dd t}{(t^2+1)^n}$ voor
elke $n$, zonder goniometrische substituties.
\end{example}

\begin{method}[Een oneigenlijke integraal bestuderen]\label{met:b2:integration:study}
Gegeven $\int_a^b f$:
\begin{enumerate}
  \item Lokaliseer het probleem: som de eindpunten (of inwendige
        punten) op waar $f$ onbegrensd is of het interval
        oneindig, en splits zodat elk stuk precies één
        problematisch einde heeft.
  \item Als $f$ constant teken heeft nabij dat einde, zoek een
        equivalent en vergelijk met de referentieschalen van
        \cref{thm:b2:integration:comparison}.
  \item Als $f$ oscilleert, test eerst $\abs f$ (absolute
        convergentie). Als $\int\abs f$ divergeert, probeer
        partiële integratie om de oscillatie te ruilen voor
        verval, zoals in \cref{ex:b2:integration:sint};
        minoraties zoals $\abs{\sin t} \geq \sin^2t$ detecteren
        echte semi-convergentie.
  \item Voor een waarde, niet enkel de aard: partiële
        integratie, substitutie, of een parameter
        (differentieer een eenvoudiger integraal, zoals in
        \cref{ex:b2:integration:arctanparam} en
        \cref{ex:b2:integration:laplace}).
  \item Gezondheidscontroles op elke berekende waarde: teken en
        ruwe grootte tegen een grove grens ($\int_0^\infty
        \eu^{-t^2}\dd t \in \intoo{0}{1 + \int_1^\infty
        \eu^{-t}}$, dus $\frac{\sqrt\pi}{2} \approx 0.886$ is
        aannemelijk); en dimensionale consistentie onder
        schaling ($t \mapsto \lambda t$ moet beide kanten
        dezelfde manier herschalen --- de snelste detector van
        een verloren factor).
\end{enumerate}
\end{method}

\begin{remark}[Veelgemaakte valkuilen]
Drie terugkerende fouten. \emph{(i) Parameterafhankelijke
dominatoren:} de dominantie $\abs{f(x,t)} \leq \varphi(t)$ moet
uniform zijn in $x$ op de beschouwde verzameling; zij geldt
meestal op segmenten $\intcc ab$ maar niet globaal --- voor
$\int_0^\infty\eu^{-xt}\dd t$ is er geen integreerbare
dominator geldig voor alle $x > 0$, maar domineren op $x \geq a
> 0$ volstaat om op de hele open halfrechte te werken, want
continuïteit en afgeleiden zijn lokale begrippen. \emph{(ii)
Vergelijken van getekende integranden:} de
vergelijkingsgereedschapskist is voor \emph{niet-negatieve}
functies; uit $\abs f \leq g$ met $\int g$ divergent mag men
niets concluderen --- $\int_1^\infty\frac{\sin t}t\,\dd t$
\omterm{def:b2:integration:improper}{convergeert} hoewel elke vergelijking met $\frac1t$ faalt.
\emph{(iii) De helft van het probleem vergeten:} op
$\intoo{0}{\infty}$ bestudeer steeds beide einden apart;
$\int_0^\infty\frac{\dd t}{t}$ divergeert aan \emph{beide}, en
een convergent-ogende splitsing kan stilzwijgend twee
oneindigheden cancelen. De veilige reflex is de checklist van
\cref{met:b2:integration:study}.
\end{remark}

\begin{example}[Een Bertrand-grensgeval, tot op de cijfer]\label{ex:b2:integration:bertrandboundary}
De schaal $\int^\infty\frac{\dd t}{t(\ln t)^\beta}$ van
\cref{thm:b2:integration:comparison} zit precies op de rand van
de machtschalen; haar grensgevallen verdienen één volle
berekening. Voor $\beta = 2$:
\[
\int_\eu^{\infty}\frac{\dd t}{t(\ln t)^2}
= \Bigl[-\frac{1}{\ln t}\Bigr]_\eu^{\infty}
= 0 - (-1) = 1 ,
\]
een convergente integraal met een prettig exacte waarde; terwijl
voor $\beta = 1$,
\[
\int_\eu^{x}\frac{\dd t}{t\ln t}
= \bigl[\ln\ln t\bigr]_\eu^{x} = \ln\ln x
\longrightarrow \infty ,
\]
divergent --- maar zo traag dat $\ln\ln x = 10$ bereiken $x =
\eu^{\eu^{10}} \approx 10^{9566}$ eist. Afsluitend inzicht:
tussen ``elke macht $t^{-1-\varepsilon}$ \omterm{def:b2:integration:improper}{convergeert}'' en
``$t^{-1}$ divergeert'' leeft een oneindige ladder van
logaritmische schalen, elk de vorige verfijnend; de substitutie
$u = \ln t$ stort elke sport in op de vorige, wat is waarom de
Bertrand-criteria de Riemann-criteria één niveau hoger
echoën.
\end{example}

\begin{remark}[Perspectieven binnen dit boekdeel]
De gereedschappen van dit hoofdstuk staan overal op het punt te
verschijnen. Gedomineerde convergentie is de motor achter de
benaderingsidentiteiten van het volgende hoofdstuk (glijdende
kernen, Bernstein en Fejér gelijkelijk); continuïteit en
differentiatie onder het integraalteken produceren de calculus
van Fouriercoëfficiënten in het Fourier-hoofdstuk, waar elke
$c_n(f)$ een \omterm{thm:b2:integration:continuity}{parameterintegraal} in vermoming is. De
$\Gamma$-functie keert tweemaal terug: in het hoofdstuk over
meervoudige integralen, waar een dubbelintegraal eindelijk de
Beta--Gamma-formule van Euler \omterm{def:b2:metric:complete}{volledig} bewijst, en in de
kansrekeningshoofdstukken, waar $\Gamma$-type integralen de
standaarddichtheden normaliseren en hun momenten berekenen. En
de semi-convergente $\int\frac{\sin t}{t}$ herrijst als de
Gibbs-constante van het Fourier-hoofdstuk --- dezelfde
integraal, die de overshoot van partiële sommen bij een sprong
meet.
\end{remark}

\begin{definition}[De $\Gamma$-functie]\label{def:b2:integration:gamma}
Voor $x > 0$:\index{Gamma-functie}
\[
\Gamma(x) = \int_0^{\infty} t^{x-1}\,\eu^{-t}\,\dd t ,
\]
convergent aan beide einden ($t^{x-1}$ integreerbaar in $0^+$
voor $x > 0$; exponentieel verval in $\infty$).
\end{definition}

\begin{theorem}\label{thm:b2:integration:gammaprops}
$\Gamma$ is continu op $\intoo{0}{+\infty}$, voldoet aan de
functionaalvergelijking
\[
\Gamma(x + 1) = x\,\Gamma(x),
\qquad \Gamma(1) = 1,
\qquad\text{dus}\qquad \Gamma(n + 1) = n! ,
\]
en is van klasse $C^1$ (inderdaad $C^\infty$) met $\Gamma'(x) =
\int_0^\infty t^{x-1}\eu^{-t}\ln t\,\dd t$.
\end{theorem}

\begin{proof}
Functionaalvergelijking: partiële integratie op
$\intcc{\varepsilon}{M}$ en laat de einden gaan: $\int
t^{x}\eu^{-t} = [-t^x\eu^{-t}] + x\int t^{x-1}\eu^{-t}$,
randtermen verdwijnend --- inderdaad $\varepsilon^x\eu^{-\varepsilon}
\to 0$ als $\varepsilon \to 0^+$ omdat $x > 0$, en
$M^x\eu^{-M} \to 0$ als $M \to \infty$ omdat de exponentiële
elke macht verslaat; beide afgeknotte integralen convergeren
naar hun oneigenlijke waarden door de convergentie vestigd in
\cref{def:b2:integration:gamma}. $\Gamma(1) = \int \eu^{-t} =
1$; inductie geeft de faculteit.

Continuïteit op $\intcc{a}{b} \subset \intoo{0}{\infty}$:
domineer $t^{x-1}\eu^{-t}$ door $\varphi(t) = (t^{a-1} +
t^{b-1})\eu^{-t}$, integreerbaar en onafhankelijk van $x \in
\intcc{a}{b}$: \cref{thm:b2:integration:continuity} past op elk
dergelijk segment, dus op de hele halfrechte. Differentieerbaarheid:
de $x$-afgeleide $t^{x-1}\eu^{-t}\ln t$ wordt op $\intcc{a}{b}$
gedomineerd door $(t^{a-1} + t^{b-1})\eu^{-t}\,\abs{\ln t}$,
nog steeds integreerbaar:
\cref{thm:b2:integration:leibnizrule}; itereren geeft alle
afgeleiden (elk voegt een macht van $\ln t$ toe, onschuldig).
\end{proof}

\begin{example}[Halfgehele faculteiten]\label{ex:b2:integration:halffactorial}
De functionaalvergelijking en $\Gamma\bigl(\frac12\bigr) =
\sqrt\pi$ (één substitutie verwijderd van
\cref{exo:b2:integration:8}: stel $t = u^2$ in de
definiërende integraal) genereren alle halfgehele waarden:
\[
\Gamma\Bigl(\frac32\Bigr) = \frac12\,\Gamma\Bigl(\frac12\Bigr)
= \frac{\sqrt\pi}{2},
\qquad
\Gamma\Bigl(\frac52\Bigr) = \frac32\cdot\frac{\sqrt\pi}{2}
= \frac{3\sqrt\pi}{4},
\qquad
\Gamma\Bigl(\frac72\Bigr) = \frac{15\sqrt\pi}{8} .
\]
Aangezien $\Gamma(n+1) = n!$, mag men zeggen ``$\frac12! =
\frac{\sqrt\pi}{2} \approx 0.886$'': de faculteit is
geïnterpoleerd, en de interpolerende kromme duikt onder $1$
tussen $0! = 1$ en $1! = 1$ (haar minimum $\approx 0.8856$ in
$x \approx 1.4616$ past bij het convexiteitsbeeld van Deel I
van de weekendopgave). Afsluitend inzicht: niets in de
integraal $\int_0^\infty t^{x-1}\eu^{-t}\dd t$ bevoorrecht
gehele getallen --- de discrete aard van de faculteit was een
ongeval van tellen, en $\sqrt\pi$ is wat tussen $1$ en $1$
leeft.
\end{example}

\begin{remark}[Waar $\Gamma$ van hierheen gaat]
De weekendopgave van dit hoofdstuk bouwt de hele Euler-calculus
rond $\Gamma$: de \omterm{pb:b2:integration:1}{Beta-functie}, haar partiële-integratie-
recursies, de Wallis-integralen als Beta-waarden, en de
limietformule van Gauss. Het hoofdstuk over meervoudige
integralen bewijst de Beta--Gamma-formule van Euler voor alle
argumenten via een dubbelintegraal; de
kansrekeningshoofdstukken ontmoeten $\Gamma$ opnieuw in de
normalisatie van de meest voorkomende dichtheden en in de
momenten van wachttijden. Het boekdeel van Jaar~3 herbouwt
$\Gamma$ op Lebesgue-fundamenten, bewijst de
uniciteitsstelling van Bohr--Mollerup, en breidt de formule
van Stirling van gehele getallen tot de reële halfrechte uit
via gedomineerde convergentie.
\end{remark}

\begin{example}[Een klassieke berekening via differentiatie]\label{ex:b2:integration:laplace}
Voor $x \in \R$, zij $F(x) = \int_0^{\infty}
\eu^{-t^2}\cos(xt)\,\dd t$ (absoluut convergent, gedomineerd
door $\eu^{-t^2}$). Door
\cref{thm:b2:integration:leibnizrule} (dominantie van de
$x$-afgeleide door $t\,\eu^{-t^2}$, integreerbaar):
\[
F'(x) = -\int_0^\infty t\,\eu^{-t^2}\sin(xt)\,\dd t
= \Bigl[\frac{\eu^{-t^2}}{2}\sin(xt)\Bigr]_0^\infty
- \frac x2\int_0^\infty \eu^{-t^2}\cos(xt)\,\dd t
= -\frac x2\,F(x),
\]
(partiële integratie met $u' = t\eu^{-t^2}$). De
differentiaalvergelijking $F' = -\frac x2 F$ integreert tot
$F(x) = F(0)\,\eu^{-x^2/4}$: de Gauss-type integraal
reproduceert zichzelf. De constante $F(0) = \int_0^\infty
\eu^{-t^2}\dd t = \frac{\sqrt\pi}{2}$ wordt berekend in
\cref{exo:b2:integration:8} --- en opnieuw, via
dubbelintegratie, in \cref{ch:b2:multint}.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b2:integration:1}
Aard van:
$\displaystyle\int_0^1 \frac{\dd t}{\sqrt{t(1-t)}}$;
$\;\displaystyle\int_1^\infty \frac{\ln t}{t^2}\dd t$;
$\;\displaystyle\int_0^\infty \frac{\dd t}{1 + t^2\sin^2 t}$
\emph{(vergelijk met het divergente harmonische-type gedrag
nabij $t = n\pi$)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:integration:2}
Bereken $\displaystyle\int_0^\infty t^n \eu^{-\lambda t}\,\dd t$
($\lambda > 0$) via $\Gamma$, en
$\displaystyle\int_0^1 (\ln t)^n \dd t$ via de substitutie $t =
\eu^{-u}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:integration:3}
Bewijs dat $\displaystyle\int_0^{\infty} \frac{\dd t}{(1 +
t^2)(1 + t^x)}$ welgedefinieerd is voor elke $x \in \R$ en
onafhankelijk van $x$. \emph{(Substitueer $t \mapsto \frac1t$
en gemiddeld de twee uitdrukkingen.)} Wat is haar waarde?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:4}
(Bertrand-integralen in een eindig eindpunt) Voor welke
$(\alpha, \beta)$ \omterm{def:b2:integration:improper}{convergeert} $\displaystyle\int_0^{1/2}
\frac{\dd t}{t^\alpha\,\abs{\ln t}^\beta}$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:5}
Zij $F(x) = \displaystyle\int_0^{\infty} \frac{\eu^{-xt}}{1 +
t^2}\,\dd t$ voor $x \geq 0$. Bewijs dat $F$ continu is op
$\intco{0}{\infty}$, $C^2$ op $\intoo{0}{\infty}$, voldoet aan
$F'' + F = \frac1x$ daar, en dat $F(x) \to 0$ als $x \to
+\infty$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:6}
(Frullani) Zij $f$ continu op $\intco{0}{+\infty}$ met een
eindige limiet $f(\infty)$ in $+\infty$. Bewijs dat voor $a, b
> 0$:
\[
\int_0^{\infty} \frac{f(at) - f(bt)}{t}\,\dd t
= \bigl(f(0) - f(\infty)\bigr)\,\ln\frac ba .
\]
\emph{(Op $\intcc{\varepsilon}{M}$, substitueer in elk stuk en
hergroepeer tot $\int_{a\varepsilon}^{b\varepsilon} -
\int_{aM}^{bM}$ van $\frac{f(u)}u\,\dd u$; sluit in met de
continuïteit in $0$ en de limiet in $\infty$.)} Bereken
$\int_0^\infty \frac{\eu^{-t} - \eu^{-2t}}{t}\dd t$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:7}
Rechtvaardig en bereken $\lim_{n\to\infty} \displaystyle\int_0^n
\Bigl(1 - \frac tn\Bigr)^{\!n} t^{x-1}\,\dd t$ voor $x > 0$
\emph{(gedomineerde convergentie met $\varphi(t) =
\eu^{-t}t^{x-1}$, gebruik $(1 - t/n)^n \leq \eu^{-t}$; de
limiet is $\Gamma(x)$)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:integration:8}
(De Gaussische integraal via een parametertruc) Voor $x \geq 0$
stel
\[
G(x) = \Bigl(\int_0^x \eu^{-t^2}\dd t\Bigr)^{\!2},
\qquad
H(x) = \int_0^1 \frac{\eu^{-x^2(1+t^2)}}{1 + t^2}\,\dd t .
\]
Bewijs dat $G' + H' = 0$ (differentieer $H$ onder de integraal
en substitueer $u = xt$ in de resulterende integraal), leid $G(x)
+ H(x) = \frac\pi4$ af voor alle $x$, en concludeer
\[
\int_0^{\infty} \eu^{-t^2}\,\dd t = \frac{\sqrt\pi}{2} .
\]
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:integration:9}
Bewijs dat $\Gamma$ \emph{log-convex} is: $\ln\Gamma$ is convex
op $\intoo{0}{\infty}$. \emph{(Cauchy--Schwarz voor integralen
toegepast op $t^{(x+y)/2 - 1}\eu^{-t} =
\bigl(t^{x-1}\eu^{-t}\bigr)^{1/2}
\bigl(t^{y-1}\eu^{-t}\bigr)^{1/2}$ geeft
$\Gamma\bigl(\frac{x+y}{2}\bigr)^2 \leq \Gamma(x)\Gamma(y)$;
combineer met continuïteit en \cref{exo:b2:realfun:8}.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:integration:10}
(Dirichlet-integraal) Stel $F(x) = \displaystyle\int_0^{\infty}
\frac{\sin t}{t}\,\eu^{-xt}\,\dd t$ voor $x > 0$.
\begin{enumerate}
  \item Rechtvaardig $F'(x) = -\frac{1}{1 + x^2}$ (differentieer
        onder de integraal; bereken $\int_0^\infty
        \eu^{-xt}\sin t\,\dd t$ via twee partiële integraties).
  \item Bewijs $F(x) \to 0$ als $x \to +\infty$ en leid $F(x) =
        \frac\pi2 - \arctan x$ af.
  \item De continuïteit van $F$ in $0^+$ aannemend (een
        Abel-type stelling), concludeer de waarde van de
        semi-convergente integraal:
        \[
        \int_0^{\infty} \frac{\sin t}{t}\,\dd t = \frac{\pi}{2}.
        \]
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:integration:11}
Rechtvaardig de convergentie van $\displaystyle\int_0^\infty
\Bigl(\frac{\sin t}{t}\Bigr)^{\!2}\dd t$, bereken haar dan via
één partiële integratie en \cref{exo:b2:integration:10}:
\[
\int_0^\infty \Bigl(\frac{\sin t}{t}\Bigr)^{\!2}\dd t
= \frac{\pi}{2} .
\]
(Dezelfde waarde als $\int_0^\infty \frac{\sin t}{t}\dd t$ ---
maar deze keer is de convergentie absoluut.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:integration:12}
(De Gaussische staart) Voor $x > 0$ stel $T(x) = \displaystyle
\int_x^\infty \eu^{-t^2}\dd t$.
\begin{enumerate}
  \item Schrijvend $\eu^{-t^2} = \frac{1}{-2t}\cdot(-2t\,\eu^{-t^2})$,
        integreer partieel tweemaal om
        \[
        T(x) = \eu^{-x^2}\Bigl(\frac{1}{2x} -
        \frac{1}{4x^3}\Bigr) +
        \frac34\int_x^\infty \frac{\eu^{-t^2}}{t^4}\,\dd t .
        \]
        te verkrijgen.
  \item Begrens de rest: $0 \leq \frac34\int_x^\infty
        t^{-4}\eu^{-t^2}\dd t \leq \frac{3}{8x^5}\,\eu^{-x^2}$,
        en leid de insluiting af
        \[
        \eu^{-x^2}\Bigl(\frac{1}{2x} - \frac{1}{4x^3}\Bigr)
        \leq T(x) \leq \frac{\eu^{-x^2}}{2x},
        \qquad\text{dus}\qquad
        T(x) \sim \frac{\eu^{-x^2}}{2x} \quad (x \to +\infty).
        \]
  \item Waarom kan de volle alternerende reeks verkregen door
        de partiële integratie te itereren nooit convergeren
        voor vaste $x$? \emph{(Vergelijk de groei van de
        coëfficiënten $1\cdot3\cdots(2k-1)$ met de machten
        $(2x^2)^k$.)}
\end{enumerate}
\end{exercise}

\section{Probleem: Eulers integralen --- Beta, Gamma, en de
limietformule van Gauss}

\begin{problem}\label{pb:b2:integration:1}
De $\Gamma$-functie van \cref{def:b2:integration:gamma} is de
ene helft van Eulers integraalcalculus; de andere helft is de
\emph{Beta-functie}\index{Beta-functie}
\[
B(x, y) = \int_0^1 t^{x-1}(1 - t)^{y-1}\,\dd t .
\]
Dit probleem ontwikkelt het paar $(\Gamma, B)$ met de
gereedschappen van dit hoofdstuk alleen --- partiële
integratie, substitutie, gedomineerde convergentie --- en
culmineert in de \emph{Beta--Gamma-formule van Euler} $B(x,y) =
\frac{\Gamma(x)\Gamma(y)}{\Gamma(x+y)}$ op de halfgehele en in
de \emph{limietformule van Gauss}\index{limietformule van Gauss}
voor $\Gamma$. Onderweg herrijzen de Wallis-integralen van
\cref{lem:b2:comparison:wallis} als Beta-waarden, en valt de
duplicatieformule van Legendre eruit.

\medskip
\noindent\textbf{Deel I --- Fijne structuur van $\Gamma$.}
\begin{enumerate}
  \item Herinner waarom $\Gamma(x) = \int_0^\infty
        t^{x-1}\eu^{-t}\dd t$ precies voor $x > 0$ \omterm{def:b2:integration:improper}{convergeert},
        en toon
        \[
        \Gamma(x) \sim \frac1x \qquad (x \to 0^+)
        \]
        \emph{(functionaalvergelijking plus continuïteit van
        $\Gamma$ in $1$)}.
  \item Bewijs $\Gamma\bigl(\tfrac12\bigr) = \sqrt\pi$
        \emph{(substitueer $t = u^2$ en roep
        \cref{exo:b2:integration:8} aan)}, en leid
        $\int_\R \eu^{-u^2/2}\dd u = \sqrt{2\pi}$ af.
  \item Toon door inductie, voor $n \in \N$:
        \[
        \Gamma\Bigl(n + \frac12\Bigr)
        = \frac{(2n)!}{4^n\,n!}\,\sqrt\pi .
        \]
  \item Rechtvaardig $\Gamma''(x) = \int_0^\infty
        t^{x-1}\eu^{-t}(\ln t)^2\dd t > 0$, en leid af dat
        $\Gamma$ strikt convex is, een uniek minimum bereikt in
        enige $x_0 \in \intoo{1}{2}$ \emph{($\Gamma(1) =
        \Gamma(2) = 1$ en Rolle)}, daalt op $\intoo{0}{x_0}$ en
        stijgt op $\intoo{x_0}{\infty}$.
  \item Toon dat $\Gamma$ elke macht verslaat: voor elke $k \in
        \N$ is $x^k = o\bigl(\Gamma(x)\bigr)$ als $x \to
        +\infty$ \emph{(sluit $x$ in tussen gehele getallen en
        gebruik $\Gamma(n+1) = n!$ met de monotonie van vraag
        4)}.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De \omterm{pb:b2:integration:1}{Beta-functie}, via partiële
integratie.}
\begin{enumerate}[resume]
  \item Toon dat $B(x,y)$ precies voor $x > 0$ en $y > 0$
        \omterm{def:b2:integration:improper}{convergeert}, en dat $B(x,y) = B(y,x)$.
  \item Bereken $B(x, 1) = \frac1x$, en bewijs via partiële
        integratie, voor $x, y > 0$:
        \[
        B(x, y+1) = \frac{y}{x}\,B(x+1, y) .
        \]
  \item Uit de splitsing $t^{x-1}(1-t)^{y-1} =
        t^{x}(1-t)^{y-1} + t^{x-1}(1-t)^{y}$ leid $B(x,y) =
        B(x+1,y) + B(x,y+1)$ af, en combineer met vraag 7 tot
        de \emph{afdaalrelaties}
        \[
        B(x, y+1) = \frac{y}{x+y}\,B(x,y),
        \qquad
        B(x+1, y) = \frac{x}{x+y}\,B(x,y) .
        \]
  \item Leid af, voor $m, n \geq 1$ geheel:
        \[
        B(m, n) = \frac{(m-1)!\,(n-1)!}{(m+n-1)!}
        = \frac{1}{(m+n-1)\binom{m+n-2}{m-1}} .
        \]
  \item Bewijs de \emph{formule van Euler voor één geheel
        argument}: voor elke $x > 0$ en $n \in \N^*$,
        \[
        B(x, n) = \frac{\Gamma(x)\,\Gamma(n)}{\Gamma(x + n)}
        \]
        \emph{(inductie op $n$: beide kanten gelijk $\frac1x$
        in $n = 1$ en gehoorzamen dezelfde
        afdaalrelatie)}.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Wallis-integralen als
Beta-waarden.}
\begin{enumerate}[resume]
  \item Substitueer $t = \sin^2\theta$ om de goniometrische
        vorm te verkrijgen
        \[
        B(x, y) = 2\int_0^{\pi/2}
        \sin^{2x-1}\theta\,\cos^{2y-1}\theta\,\dd\theta .
        \]
  \item Leid $W_n = \frac12\,B\bigl(\frac{n+1}2, \frac12\bigr)$
        af voor de Wallis-integraal $W_n =
        \int_0^{\pi/2}\sin^n \theta\,\dd\theta$, en herstel de
        recurrente $W_n = \frac{n-1}{n}W_{n-2}$ van
        \cref{lem:b2:comparison:wallis} uit de afdaalrelaties
        van vraag 8 alleen.
  \item Bereken $B\bigl(\frac12, \frac12\bigr) = 2W_0 = \pi$ en
        controleer haar tegen
        $\Gamma\bigl(\frac12\bigr)^2/\Gamma(1)$: de formule van
        Euler geldt in $\bigl(\frac12, \frac12\bigr)$.
  \item Leid de gesloten vorm $W_{2n} = \frac\pi2\,
        \frac{(2n)!}{4^n(n!)^2}$ af uit de recurrente, en
        verifieer
        \[
        B\Bigl(n + \frac12, \frac12\Bigr)
        = \frac{\Gamma\bigl(n + \frac12\bigr)\Gamma\bigl(
        \frac12\bigr)}{\Gamma(n+1)} .
        \]
        Concludeer, via inductie met de afdaalrelaties, dat de
        formule van Euler $B(x,y) =
        \frac{\Gamma(x)\Gamma(y)}{\Gamma(x+y)}$ geldt wanneer
        $2x$ en $2y$ positieve gehele getallen zijn.
  \item Substitueer $u = \frac{t}{1-t}$ om de derde klassieke
        vorm te verkrijgen
        \[
        B(x,y) = \int_0^\infty
        \frac{u^{x-1}}{(1+u)^{x+y}}\,\dd u ,
        \]
        en controleer het geval $x = y = \frac12$ rechtstreeks
        ($u = v^2$ reduceert tot
        $\int_0^\infty\frac{2\,\dd v}{1+v^2}$).
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De limietformule van Gauss.}
\begin{enumerate}[resume]
  \item Voor $x > 0$ en $n \in \N^*$, bewijs via $n$
        opeenvolgende partiële integraties:
        \[
        \int_0^n \Bigl(1 - \frac tn\Bigr)^{\!n} t^{x-1}\,\dd t
        = \frac{n!\;n^x}{x(x+1)\cdots(x+n)} .
        \]
  \item Concludeer met \cref{exo:b2:integration:7}
        (gedomineerde convergentie) de \emph{limietformule van
        Gauss}:
        \[
        \Gamma(x) = \lim_{n\to\infty}
        \frac{n!\;n^x}{x(x+1)\cdots(x+n)}
        \qquad (x > 0).
        \]
  \item Logaritmen nemend, toon dat voor $x > 0$:
        \[
        \ln\Gamma(x) = -\ln x - \gamma x +
        \sum_{k=1}^{\infty}\Bigl(\frac xk -
        \ln\Bigl(1 + \frac xk\Bigr)\Bigr),
        \]
        waar $\gamma$ de constante van Euler is
        (\cref{ex:b2:comparison:harmonic}); rechtvaardig de
        convergentie van de reeks \emph{(de algemene term is
        $\sim \frac{x^2}{2k^2}$)}.
  \item Gebruik de formule van Gauss in $x = \frac12$ en de
        centrale binomiale asymptotiek $\binom{2n}{n} \sim
        \frac{4^n}{\sqrt{\pi n}}$
        (\cref{ex:b2:comparison:centralbinomial}) om
        $\Gamma\bigl(\frac12\bigr) = \sqrt\pi$ opnieuw te
        berekenen: de constante van Stirling en de Gaussische
        integraal zijn hetzelfde getal in twee vermommingen.
  \item Controleer dat de formule van Gauss de
        functionaalvergelijking herbewijst: uit de exacte
        identiteit
        \[
        \frac{n!\,n^{x+1}}{(x+1)\cdots(x+n+1)}
        = \frac{n!\,n^{x}}{x(x+1)\cdots(x+n)}\cdot
        \frac{n\,x}{x+n+1},
        \]
        concludeer opnieuw $\Gamma(x+1) = x\,\Gamma(x)$. (De
        formule van Gauss bepaalt $\Gamma$ outright; het
        boekdeel van Jaar~3 bewijst de scherpere stelling van
        Bohr--Mollerup: de functionaalvergelijking plus
        log-convexiteit spijkeren $\Gamma$ al vast.)
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Dividenden.}
\begin{enumerate}[resume]
  \item Voor $a > 0$ toon $\int_0^\infty \eu^{-t^a}\dd t =
        \Gamma\bigl(1 + \frac1a\bigr)$, en bereken de limiet
        als $a \to +\infty$ via gedomineerde convergentie
        \emph{(puntsgewijze limiet $\mathbf 1_{t < 1}$;
        domineer door $1$ op $\intoc{0}{1}$ en door
        $\eu^{-t^2}$ daarbuiten, voor $a \geq 2$)}. Controleer
        het antwoord tegen de continuïteit van $\Gamma$.
  \item Voor $n \geq 1$ toon
        \[
        \int_0^1 \frac{\dd t}{\sqrt{1 - t^n}}
        = \frac1n\,B\Bigl(\frac1n, \frac12\Bigr),
        \]
        en herstel de waarden $2$ ($n = 1$) en $\frac\pi2$ ($n
        = 2$). (Voor $n = 4$ is dit de \emph{lemniscaatconstante},
        die geen elementaire gesloten vorm heeft; haar verhaal
        behoort tot de theorie van elliptische integralen.)
  \item (Momenten) Voor $x > 0$ en $k \in \N$, toon
        \[
        \frac{1}{\Gamma(x)}\int_0^\infty
        t^{k}\,t^{x-1}\eu^{-t}\,\dd t
        = \frac{\Gamma(x+k)}{\Gamma(x)}
        = x(x+1)\cdots(x+k-1),
        \]
        de stijgende faculteit; controleer dat $x = 1$ $k!$
        geeft. (In de kansrekeningshoofdstukken is dit het
        $k$-de moment van een standaard wachttijddichtheid.)
  \item Bewijs de Beta-identiteit, geldig voor \emph{alle} $x >
        0$:
        \[
        B(x, x) = 2^{1-2x}\,B\Bigl(x, \frac12\Bigr)
        \]
        \emph{(substitueer $t = \frac{1+s}2$, benut de
        symmetrie in $s$, stel dan $s = \sqrt v$)}. Leid af,
        voor $2x \in \N^*$, de \emph{duplicatieformule van
        Legendre}
        \[
        \Gamma(x)\,\Gamma\Bigl(x + \frac12\Bigr)
        = 2^{1-2x}\,\sqrt\pi\;\Gamma(2x),
        \]
        en verifieer haar rechtstreeks in $x = n$ via vraag 3.
        (Voor algemene $x$ volgt zij uit dezelfde identiteit
        zodra de formule van Euler voor alle argumenten bekend
        is --- het dubbelintegraalbewijs in het hoofdstuk over
        meervoudige integralen.)
  \item Synthese. In één zin elk: (i) waar partiële integratie
        heel Deel II droeg; (ii) waar gedomineerde convergentie
        Delen IV en V binnentrad; (iii) welke asymptotische
        invoer uit het vergelijkingshoofdstuk werd ingevoerd;
        (iv) wat nu bewezen is van de formule van Euler
        $B(x,y) = \Gamma(x)\Gamma(y)/\Gamma(x+y)$, en wat
        resteert voor de dubbelintegraal om te regelen.
\end{enumerate}
\end{problem}
