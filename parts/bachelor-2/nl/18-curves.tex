\chapter{Krommen}\label{ch:b2:curves}

De meetkunde wordt nu \omterm{def:b2:diffcalc:differential}{differentiaal}. Een kromme is een punt dat
door de ruimte beweegt; de analyse levert snelheid en versnelling,
en de meetkunde vraagt wat intrinsiek is --- onafhankelijk van hoe
snel we de baan doorlopen. De antwoorden zijn de \emph{\omterm{def:b2:curves:length}{booglengte}},
die de baan zelf meet, en de \emph{\omterm{thm:b2:curves:frenet2d}{kromming}}, die meet hoe ze
buigt. In dimensie $3$ meet een tweede invariant, de
\emph{\omterm{thm:b2:curves:frenet3d}{torsie}}, hoe de kromme uit haar vlak verdraait. Het
boekhoudapparaat voor dit alles is het bewegende
\emph{Frenet-raam}.

\section{Geparametriseerde bogen}

\begin{definition}[Geparametriseerde boog]\label{def:b2:curves:arc}
Een \emph{geparametriseerde boog}\index{geparametriseerde boog} van
klasse $\mathcal{C}^k$ ($k \geq 1$) is een afbeelding $\gamma
\colon I \to \R^n$ van klasse $\mathcal{C}^k$ op een interval $I$.
Een punt $\gamma(t)$ is \emph{regulier} als $\gamma'(t) \neq 0$,
en de boog is \emph{regulier} als al haar punten dat zijn. De
rechte door $\gamma(t)$ gericht door $\gamma'(t)$ is de
\emph{raaklijn}\index{raaklijn} in een regulier punt.
\end{definition}

\begin{definition}[Parameterwisseling]\label{def:b2:curves:reparam}
Een \emph{parameterwisseling}\index{parameterwisseling} van
klasse $\mathcal{C}^k$ is een $\mathcal{C}^k$-diffeomorfisme
$\theta \colon J \to I$ tussen intervallen ($\theta' \neq 0$
overal). De bogen $\gamma$ en $\gamma \circ \theta$ heten
\emph{equivalent}; een \emph{meetkundige
boog}\index{meetkundige boog} (of kromme) is een
equivalentieklasse. Begrippen die invariant zijn onder
parameterwisseling --- de baan, de \omterm{def:b2:curves:arc}{raaklijn}, de \omterm{def:b2:curves:length}{booglengte}, de
\omterm{thm:b2:curves:frenet2d}{kromming} --- heten \emph{meetkundig}.
\end{definition}

\begin{remark}
De baan alleen bepaalt de \omterm{def:b2:curves:reparam}{meetkundige boog} niet: de
parametrisaties $t \mapsto (\cos t, \sin t)$ op $[0, 2\pi]$ en
op $[0, 4\pi]$ hebben hetzelfde beeld maar doorlopen de cirkel
één en twee keer. Een \omterm{def:b2:curves:reparam}{meetkundige boog} onthoudt de
multipliciteit en oriëntatie van de doorloping, niet haar
snelheid.
\end{remark}

\begin{example}[Snelheid verandert niets
meetkundigs]\label{ex:b2:curves:speedinvariance}
Parametriseer de eenheidscirkel door
\[
\gamma(t) = (\cos t^2,\ \sin t^2),
\qquad t \in \intcc0{\sqrt{2\pi}} .
\]
De snelheid $\norm{\gamma'(t)} = 2t$ groeit lineair, maar
\[
L = \int_0^{\sqrt{2\pi}}2t\,\dd t = 2\pi ,
\]
dezelfde \omterm{def:b2:curves:length}{lengte} als bij constante snelheid --- zoals
\cref{thm:b2:curves:lengthinv} belooft, via de
\omterm{def:b2:curves:reparam}{parameterwisseling} $t \mapsto t^2$. De \omterm{def:b2:curves:arc}{raaklijn}, de \omterm{thm:b2:curves:frenet2d}{kromming}
berekend uit \cref{prop:b2:curves:kappaformula}, en elke
andere meetkundige grootheid komen eveneens overeen; alleen $t
= 0$ verdient een blik, waar $\gamma'(0) = 0$ \emph{deze
parametrisatie} irregulier maakt hoewel de baan een perfecte
cirkel is. Meetkundige uitspraken verdragen slechte
parametrisaties slecht: herparametriseer eerst, concludeer
daarna.
\end{example}

\begin{example}\label{ex:b2:curves:cusp}
De boog $\gamma(t) = (t^2, t^3)$ is $\mathcal{C}^\infty$ maar
niet regulier: $\gamma'(0) = (0, 0)$. Haar baan, de
semicubische parabool $y^2 = x^3$, heeft een
\emph{keerpunt} in de oorsprong: gladheid van de
parametrisatie verhindert geen meetkundige singulariteit waar
de snelheid verdwijnt. Daarom is de regulariteitshypothese
$\gamma' \neq 0$ geen cosmetica.
\end{example}

\section{Booglengte}

\begin{definition}[Booglengte]\label{def:b2:curves:length}
Zij $\gamma \colon [a, b] \to \R^n$ een $\mathcal{C}^1$-boog.
Haar \emph{lengte}\index{booglengte} is
\[
L(\gamma) = \int_a^b \norm{\gamma'(t)}\, \dd t ,
\]
waar $\norm{\cdot}$ de euclidische \omterm{def:b2:nvs:norm}{norm} is. De
\emph{booglengtefunctie}\index{booglengtefunctie} gebaseerd in
$t_0$ is $s(t) = \int_{t_0}^t \norm{\gamma'(u)}\,\dd u$.
\end{definition}

\begin{theorem}[Lengte is meetkundig; polygonale
karakterisering]\label{thm:b2:curves:lengthinv}
\leavevmode
\begin{enumerate}
\item Als $\theta \colon [c, d] \to [a, b]$ een
$\mathcal{C}^1$-parameterwisseling is, dan $L(\gamma \circ
\theta) = L(\gamma)$.
\item $L(\gamma)$ is het supremum van de \omterm{def:b2:curves:length}{lengten} van
ingeschreven polygonen:
\[
L(\gamma)
= \sup\Bigl\{\, \sum_{i=1}^{m} \norm{\gamma(t_i) -
\gamma(t_{i-1})} \;:\; a = t_0 < t_1 < \dots < t_m = b \,\Bigr\}.
\]
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Door de variabelensubstitutie $t = \theta(u)$ (volume
Jaar~1, geldig omdat $\theta$ $\mathcal{C}^1$ monotoon is),
\[
\int_c^d \norm{(\gamma\circ\theta)'(u)}\,\dd u
= \int_c^d \norm{\gamma'(\theta(u))}\,\abs{\theta'(u)}\,\dd u
= \int_a^b \norm{\gamma'(t)}\,\dd t ,
\]
waar we $(\gamma\circ\theta)' = \theta'\cdot
(\gamma'\circ\theta)$ gebruikten en, als $\theta$ dalend is, het
teken van $\theta'$ door de omkering van de grenzen wordt
opgenomen.

\emph{2.} Voor elke onderverdeling, $\gamma(t_i) -
\gamma(t_{i-1}) = \int_{t_{i-1}}^{t_i}\gamma'(t)\,\dd t$, dus
door de driehoeksongelijkheid voor integralen $\norm{\gamma(t_i)
- \gamma(t_{i-1})} \leq \int_{t_{i-1}}^{t_i}\norm{\gamma'}$:
elke polygoon is korter dan $L(\gamma)$, dus $\sup \leq
L(\gamma)$.

Voor de omgekeerde ongelijkheid, zij $\varepsilon > 0$. Omdat
$\gamma'$ continu is op het compacte $[a,b]$, is ze uniform
continu: er is $\delta > 0$ met $\norm{\gamma'(t) - \gamma'(u)}
\leq \varepsilon$ wanneer $\abs{t - u} \leq \delta$. Neem een
onderverdeling van stap $\leq \delta$. Op elk stuk, voor $t \in
[t_{i-1}, t_i]$,
\[
\gamma(t_i) - \gamma(t_{i-1})
= \int_{t_{i-1}}^{t_i}\gamma'(t)\,\dd t
= (t_i - t_{i-1})\,\gamma'(t_{i-1}) + R_i,
\qquad
\norm{R_i} \leq \varepsilon\,(t_i - t_{i-1}),
\]
omdat $R_i = \int_{t_{i-1}}^{t_i}(\gamma'(t) -
\gamma'(t_{i-1}))\,\dd t$. Dus
\[
\norm{\gamma(t_i) - \gamma(t_{i-1})}
\geq (t_i - t_{i-1})\norm{\gamma'(t_{i-1})}
- \varepsilon (t_i - t_{i-1}) .
\]
Optellen, en $\sum (t_i - t_{i-1})\norm{\gamma'(t_{i-1})}$
vergelijken met $\int_a^b\norm{\gamma'}$ (een Riemannsom van de
continue functie $\norm{\gamma'}$, binnen $\varepsilon(b - a)$
van de integraal voor $\delta$ klein genoeg door opnieuw
uniforme continuïteit), levert een polygoon van \omterm{def:b2:curves:length}{lengte} $\geq
L(\gamma) - 2\varepsilon(b - a)$. Laten $\varepsilon \to 0$
bewijst de bewering.
\end{proof}

\begin{example}[Archimedes en de ingeschreven
polygonen]\label{ex:b2:curves:archimedes}
Voor de eenheidscirkel heeft de ingeschreven regelmatige
$n$-hoek \omterm{def:b2:curves:length}{lengte} $L_n = 2n\sin\frac\pi n$, en de ontwikkeling
$\sin x = x - \frac{x^3}6 + O(x^5)$ geeft
\[
L_n = 2\pi - \frac{\pi^3}{3n^2} + O\Bigl(\frac1{n^4}\Bigr) :
\]
de polygonale \omterm{def:b2:curves:length}{lengten} van \cref{thm:b2:curves:lengthinv}
convergeren \emph{kwadratisch}. Numeriek: $L_6 = 6$ (de
zeshoek, die de ruwe $\pi > 3$ geeft), terwijl $L_{96} =
192\sin\frac{\pi}{96} \approx 6.28206$ tegen $2\pi \approx
6.28319$ --- de fout $0.00113$ stemt overeen met de
voorspelde $\pi^3/(3\cdot96^2) \approx 0.00112$. Daarom kon
Archimedes, door de zeshoek vijfmaal te verdubbelen tot $96$
zijden, $\pi$ met de hand tot drie cijfers inklemmen: elke
verdubbeling deelt de fout door vier. Het supremum in de
polygonale karakterisering wordt niet alleen in de limiet
bereikt; het wordt \emph{snel} bereikt, omdat een gladde
kromme van haar koorden slechts op tweede orde scheidt.
\end{example}

\begin{theorem}[Booglengteparametrisatie]\label{thm:b2:curves:arclength}
Zij $\gamma \colon I \to \R^n$ een reguliere $\mathcal{C}^k$-boog
($k \geq 1$). De \omterm{def:b2:curves:length}{booglengtefunctie} $s$ is een
$\mathcal{C}^k$-diffeomorfisme van $I$ op een interval $J$, en
$\tilde\gamma = \gamma \circ s^{-1}$ voldoet aan
$\norm{\tilde\gamma'} = 1$ overal. Op een translatie van de
parameter en de oriëntatie na is deze \emph{\omterm{def:b2:curves:length}{booglengte}}- (of
\emph{eenheidssnelheids-})parametrisatie uniek.
\end{theorem}

\begin{proof}
$s'(t) = \norm{\gamma'(t)} > 0$ en $s'$ is $\mathcal{C}^{k-1}$
(samenstelling van de $\mathcal{C}^{k-1}$-afbeelding $\gamma'$
met de \omterm{def:b2:nvs:norm}{norm}, glad weg van $0$), dus is $s$ $\mathcal{C}^k$,
strikt stijgend, een bijectie op $J = s(I)$, en haar inverse is
$\mathcal{C}^k$ door de inversefunctiestelling in dimensie $1$
(volume Jaar~1). Dan
\[
\tilde\gamma'(\sigma)
= \frac{\gamma'(t)}{s'(t)}
= \frac{\gamma'(t)}{\norm{\gamma'(t)}},
\qquad t = s^{-1}(\sigma),
\]
een eenheidsvector. Als $\hat\gamma = \gamma\circ\theta$ een
andere eenheidssnelheidsparametrisatie is, dan $\abs{\theta'} =
1$, dus $\theta' = \pm 1$ constant (continuïteit), d.w.z.\
$\theta(u) = \pm u + c$.
\end{proof}

\begin{remark}
\omterm{def:b2:curves:length}{Booglengte} is de parameter die meetkunde van dynamica scheidt.
Een baan kan met elk snelheidsprofiel worden doorlopen --- de
fysica van de beweging --- maar elke
parametrisatie-invariante vraag (vorm, buiging, osculatie)
heeft een canonieke klok, de afgelegde afstand. Daarom worden
alle krommingsformules hieronder \emph{gedefinieerd} bij
eenheidssnelheid en daarna naar willekeurige parametrisaties
\emph{vertaald} door de kettingregel: de vertaalfactoren zijn
machten van $v = s'$, en die correct bijhouden is de gehele
inhoud van \cref{prop:b2:curves:kappaformula}.
\end{remark}

\begin{example}[Cirkel en schroeflijn]\label{ex:b2:curves:helix}
Voor de cirkel $\gamma(t) = (R\cos t, R\sin t)$, $\norm{\gamma'}
= R$, dus $s = Rt$ en de \omterm{def:b2:curves:length}{lengte} van een volle omwenteling is
$2\pi R$. Voor de schroeflijn $\gamma(t) = (a\cos t,\ a\sin t,\
bt)$ met $a > 0$, is $\norm{\gamma'(t)} = \sqrt{a^2 + b^2}$
constant: de schroeflijn wordt met constante snelheid
doorlopen, en $s = t\sqrt{a^2 + b^2}$.
\end{example}

\begin{example}[Booglengte in poolcoördinaten]\label{ex:b2:curves:polarlength}
Een poolkromme $r = r(\theta)$ is de boog $\gamma(\theta) =
(r\cos\theta,\ r\sin\theta)$, met
\[
\gamma'(\theta) = (r'\cos\theta - r\sin\theta,\
r'\sin\theta + r\cos\theta),
\qquad
\norm{\gamma'(\theta)}^2 = r'^2 + r^2
\]
(de kruistermen heffen op): het poolse lengte-element is
$\sqrt{r^2 + r'^2}\,\dd\theta$. Voor de cardioïde $r = 1 +
\cos\theta$:
\[
r^2 + r'^2 = (1 + \cos\theta)^2 + \sin^2\theta = 2 +
2\cos\theta = 4\cos^2\tfrac\theta2,
\]
en op $\intcc{-\pi}{\pi}$ is de halve-hoekcosinus
niet-negatief, dus
\[
L = \int_{-\pi}^{\pi}2\cos\tfrac\theta2\,\dd\theta
= \Bigl[4\sin\tfrac\theta2\Bigr]_{-\pi}^{\pi}
= 4 - (-4) = 8 :
\]
zoals de cycloïdeboog van \cref{exo:b2:curves:1} heeft een
kromme gebouwd uit cirkels een rationale \omterm{def:b2:curves:length}{lengte}, zonder $\pi$
ergens. De halve-hoekfactorisatie is de standaardtruc voor
\omterm{def:b2:curves:length}{lengten} van cirkel-gegenereerde krommen; wanneer ze faalt (de
ellips), is de \omterm{def:b2:curves:length}{lengte} een echt nieuwe functie --- een
elliptische integraal, voorbij elementaire gesloten vormen.
\end{example}

\section{Kromming in het vlak}

Doorheen deze paragraaf zijn bogen $\mathcal{C}^2$ en regulier
in het georiënteerde euclidische vlak. We parametriseren op
\omterm{def:b2:curves:length}{booglengte} en schrijven $T(s) = \tilde\gamma'(s)$ voor de
eenheidstangens, en $N(s)$ voor de eenheidsvector direct
orthogonaal op $T(s)$ (rotatie van $T$ over $+\pi/2$).

\begin{theorem}[Vlakke Frenet-formules]\label{thm:b2:curves:frenet2d}
Zij $\tilde\gamma$ een eenheidssnelheids-$\mathcal{C}^2$-boog in
het georiënteerde vlak. Er is een continue functie $\kappa$, de
(algebraïsche) \emph{kromming}\index{kromming}, zodat
\[
T'(s) = \kappa(s)\, N(s),
\qquad
N'(s) = -\kappa(s)\, T(s).
\]
\end{theorem}

\begin{proof}
Omdat $\norm{T(s)}^2 = 1$ voor alle $s$, geeft differentiëren
van het inwendig product $2\langle T'(s), T(s)\rangle = 0$:
$T'(s)$ is orthogonaal op $T(s)$, dus collineair met $N(s)$
(dimensie $2$); schrijf $T'(s) = \kappa(s) N(s)$ met $\kappa(s)
= \langle T'(s), N(s)\rangle$, continu. Evenzo $N' \perp N$,
dus $N' = \lambda T$; en differentiëren van $\langle T,
N\rangle = 0$ geeft $\langle T', N\rangle + \langle T,
N'\rangle = \kappa + \lambda = 0$.
\end{proof}

\begin{definition}\label{def:b2:curves:curvature}
Wanneer $\kappa(s) \neq 0$, is de
\emph{kromtestraal}\index{kromtestraal} $R(s) =
1/\abs{\kappa(s)}$ en het
\emph{kromtemiddelpunt}\index{kromtemiddelpunt} is
$\tilde\gamma(s) + \frac{1}{\kappa(s)} N(s)$; de cirkel met dat
centrum en straal $R(s)$ is de
\emph{osculatiecirkel}\index{osculatiecirkel}, de beste
cirkelvormige benadering van de kromme in $\tilde\gamma(s)$.
\end{definition}

\begin{example}[De osculatiecirkel van de
exponentiële]\label{ex:b2:curves:oscexp}
Voor $y = \eu^x$ in het punt $(0, 1)$: $f'(0) = f''(0) = 1$,
dus door de grafiekformule hieronder,
\[
\kappa(0) = \frac{1}{(1 + 1)^{3/2}} = \frac1{2\sqrt2},
\qquad R = 2\sqrt2 .
\]
De eenheidstangens is $T = \frac{(1, 1)}{\sqrt2}$, de directe
normaal $N = \frac{(-1, 1)}{\sqrt2}$, en het \omterm{def:b2:curves:curvature}{kromtemiddelpunt} is
\[
(0, 1) + 2\sqrt2\cdot\frac{(-1, 1)}{\sqrt2} = (-2,\ 3) :
\]
de \omterm{def:b2:curves:curvature}{osculatiecirkel} heeft vergelijking $(x + 2)^2 + (y - 3)^2 =
8$. Als controle van de claim ``beste cirkelvormige
benadering'': de cirkelvergelijking naar $y$ oplossen nabij
$(0,1)$ en ontwikkelen geeft $y = 1 + x + \frac{x^2}2 + O(x^3)$
--- precies de tweede-orde Taylorontwikkeling van $\eu^x$. De
\omterm{def:b2:curves:curvature}{osculatiecirkel} past waarde, richtingscoëfficiënt \emph{en}
tweede afgeleide; een gewone raakcirkel zou alleen de eerste
twee passen.
\end{example}

\begin{example}[De evolute van een cirkel is haar
centrum]\label{ex:b2:curves:circleevolute}
Voor de cirkel van straal $R$ linksom doorlopen, $\kappa = 1/R$
en $N$ wijst naar het centrum, dus is het \omterm{def:b2:curves:curvature}{kromtemiddelpunt}
$\tilde\gamma + \frac1\kappa N$ het centrum van de cirkel, voor
elke $s$: de \omterm{def:b2:curves:curvature}{osculatiecirkel} van een cirkel is de cirkel zelf,
en de meetkundige plaats van \omterm{def:b2:curves:curvature}{kromtemiddelpunten} stort in tot
een punt. Dit ontaarde geval kalibreert
\cref{exo:b2:curves:6}: daar is de snelheid van de evolute
$-\frac{\kappa'}{\kappa^2}N$, die identiek verdwijnt precies
wanneer $\kappa$ constant is.
\end{example}

\begin{proposition}[Kromming in een willekeurige
parametrisatie]\label{prop:b2:curves:kappaformula}
Voor een reguliere $\mathcal{C}^2$ vlakke boog $\gamma(t) =
(x(t), y(t))$,
\[
\kappa(t)
= \frac{x'(t)\,y''(t) - y'(t)\,x''(t)}
       {\bigl(x'(t)^2 + y'(t)^2\bigr)^{3/2}} ,
\]
in het bijzonder $\kappa = \dfrac{y''}{(1 + y'^2)^{3/2}}$ voor
een grafiek $y = f(x)$.
\end{proposition}

\begin{proof}
Schrijf $v(t) = \norm{\gamma'(t)} = s'(t)$, dus $\gamma' = vT$
(samenstellen van de eenheidssnelheidsgegevens met $s$).
Differentiëren,
\[
\gamma'' = v'T + v\,T'\cdot s' = v'T + v^2\kappa N .
\]
Neem nu de \omterm{def:b2:linalg:det}{determinant} (in de canonieke georiënteerde basis)
van $(\gamma', \gamma'')$: omdat $\det(T, T) = 0$ en $\det(T,
N) = 1$,
\[
\det(\gamma', \gamma'') = \det(vT,\ v'T + v^2\kappa N)
= v^3\kappa .
\]
De linkerkant is $x'y'' - y'x''$, en $v^3 = (x'^2 +
y'^2)^{3/2}$. Het grafiekgeval is de parametrisatie $t \mapsto
(t, f(t))$.
\end{proof}

\begin{example}[Cirkel, rechte, parabool]\label{ex:b2:curves:kappaexamples}
Een rechte heeft $\kappa = 0$ (en omgekeerd: $T' = 0$ betekent
$T$ constant, dus $\tilde\gamma(s) = \tilde\gamma(0) + sT$, een
rechte). De cirkel van straal $R$ linksom doorlopen heeft
$\kappa = 1/R$: met $\gamma(t) = (R\cos t, R\sin t)$ geeft de
formule $\kappa = R^2/R^3$. Voor de parabool $y = x^2/2$:
$\kappa(x) = 1/(1 + x^2)^{3/2}$, maximaal in de top --- de
parabool is het scherpst gebogen waar ze omkeert.
\end{example}

\begin{remark}[Klassieke valkuilen rond kromming]
(i) De algebraïsche \omterm{thm:b2:curves:frenet2d}{kromming} van een vlakke boog wisselt van
teken wanneer de oriëntatie van de boog of van het vlak wordt
omgekeerd: alleen $\abs\kappa$ en $R = 1/\abs\kappa$ zijn puur
meetkundig. Een cirkel rechtsom doorlopen heeft $\kappa =
-1/R$. (ii) De grafiekformule $\kappa = f''/(1 + f'^2)^{3/2}$
kiest stilzwijgend de parametrisatie op $x$; haar toepassen op
een kromme die nabij het punt geen grafiek is (verticale
\omterm{def:b2:curves:arc}{raaklijn}) is de klassieke blunder. (iii) In een punt waar
$\gamma' = 0$ is niets gedefinieerd --- noch $T$ noch $\kappa$
--- en de baan kan echt breken (\cref{ex:b2:curves:cusp});
controleer altijd regulariteit voor je de eenheidstangens
differentieert. (iv) In de ruimte is $\kappa = \norm{T'} \geq
0$ bij conventie: er is geen teken om fout te gaan, maar ook
geen teken om uit te buiten --- buigpuntachtige informatie
verhuist naar de \omterm{thm:b2:curves:frenet3d}{torsie}. (v) Ten slotte is $\kappa$ een
afgeleide \emph{naar \omterm{def:b2:curves:length}{booglengte}}: voor een
niet-eenheidssnelheidsparametrisatie is vergeten van de factor
$v^3$ in \cref{prop:b2:curves:kappaformula} de frequentste
fout in de praktijk.
\end{remark}

\begin{theorem}[Kromming bepaalt de
kromme]\label{thm:b2:curves:fundamental}
Zij $\kappa \colon J \to \R$ continu. Er bestaat een
eenheidssnelheids-$\mathcal{C}^2$-boog in het vlak met \omterm{thm:b2:curves:frenet2d}{kromming}
$\kappa$, en ze is uniek op een directe isometrie na (rotatie
gevolgd door translatie).
\end{theorem}

\begin{proof}
\emph{Bestaan.} Fixeer $s_0 \in J$ en stel $\varphi(s) =
\int_{s_0}^s \kappa(u)\,\dd u$, daarna
\[
\tilde\gamma(s) = \Bigl(\int_{s_0}^s \cos\varphi(u)\,\dd u,\
\int_{s_0}^s \sin\varphi(u)\,\dd u\Bigr).
\]
Dan is $T(s) = (\cos\varphi(s), \sin\varphi(s))$ een
eenheidsvector, $N(s) = (-\sin\varphi, \cos\varphi)$, en
\[
T'(s) = \varphi'(s)\,(-\sin\varphi, \cos\varphi)
= \kappa(s)\,N(s) :
\]
de boog is eenheidssnelheid met \omterm{thm:b2:curves:frenet2d}{kromming} $\kappa$.

\emph{Uniciteit.} Zij $\gamma_1, \gamma_2$
eenheidssnelheidsbogen met dezelfde \omterm{thm:b2:curves:frenet2d}{kromming}. Elke
eenheidstangens lift tot een hoekfunctie, door een expliciete
constructie: beschouw $T_j$ als het complexe getal $z_j = a_j
+ \iu b_j$ van modulus $1$, kies $\varphi_j(0)$ met $z_j(0) =
\eu^{\iu\varphi_j(0)}$, en stel
\[
\varphi_j(s) = \varphi_j(0) + \int_0^s\det\bigl(T_j,
T_j'\bigr)(u)\,\dd u .
\]
Uit $\abs{z_j} = 1$: $\operatorname{Re}(\conj{z_j}\,z_j') =
0$, dus $\conj{z_j}\,z_j' = \iu\det(T_j, T_j') =
\iu\,\varphi_j'$, d.w.z.\ $z_j' = \iu\varphi_j'z_j$; daarna
\[
\bigl(z_j\,\eu^{-\iu\varphi_j}\bigr)'
= \eu^{-\iu\varphi_j}\bigl(z_j' - \iu\varphi_j'z_j\bigr) = 0,
\]
dus $z_j = \eu^{\iu\varphi_j}$ doorheen: $T_j =
(\cos\varphi_j, \sin\varphi_j)$ met $\varphi_j$ van klasse
$\mathcal C^1$. Bovendien $\det(T_j, T_j') = \det(T_j, \kappa
N_j) = \kappa$, dus $\varphi_j' = \kappa$. Vandaar $\varphi_2
= \varphi_1 + c$ voor een constante $c$: $T_2$ is $T_1$
geroteerd over de vaste hoek $c$, dus door integreren
$\gamma_2 = \rho(\gamma_1) + w$ waar $\rho$ de rotatie over
hoek $c$ is en $w$ een constante vector.
\end{proof}

\begin{remark}
Dit is het eendimensionale prototype van een
\emph{fundamentele stelling van de meetkunde}: een complete
verzameling lokale invarianten (hier één functie) classificeert
het object op een starre beweging na. De driedimensionale
versie hieronder heeft twee invarianten nodig.
\end{remark}

\begin{example}[Constante kromming betekent
cirkel]\label{ex:b2:curves:constantkappa}
Neem $\kappa \equiv \kappa_0 > 0$ in de bestaansformule:
$\varphi(s) = \kappa_0 s$ en
\[
\tilde\gamma(s) = \Bigl(\frac{\sin\kappa_0s}{\kappa_0},\
\frac{1 - \cos\kappa_0s}{\kappa_0}\Bigr) :
\]
de cirkel van straal $1/\kappa_0$ gecentreerd in $(0,
1/\kappa_0)$, doorlopen met eenheidssnelheid. Door de
uniciteitshelft van de stelling is \emph{elke}
eenheidssnelheidsboog van constante \omterm{thm:b2:curves:frenet2d}{kromming} $\kappa_0$ een
stuk van een cirkel van straal $1/\kappa_0$ (of een rechte als
$\kappa_0 = 0$) --- het omgekeerde van de berekening in
\cref{ex:b2:curves:kappaexamples}, en het vlakke geval van
\cref{exo:b2:curves:9}.
\end{example}

\begin{example}[Een kromme reconstrueren uit haar
kromming]\label{ex:b2:curves:reconstruction}
Welke eenheidssnelheidskromme heeft \omterm{def:b2:curves:curvature}{kromtestraal} $R(s) = 1 +
s^2$? Volgend op het bestaansbewijs met $\kappa(s) =
\frac1{1+s^2}$ en $s_0 = 0$: $\varphi(s) = \arctan s$, dus
\[
T(s) = (\cos\arctan s,\ \sin\arctan s)
= \Bigl(\frac{1}{\sqrt{1+s^2}},\
\frac{s}{\sqrt{1+s^2}}\Bigr),
\]
en integreren,
\[
\tilde\gamma(s) = \Bigl(\ln\bigl(s + \sqrt{1 + s^2}\bigr),\
\sqrt{1 + s^2} - 1\Bigr).
\]
Met $x = \ln(s + \sqrt{1+s^2})$, d.w.z.\ $s = \sinh x$, is de
tweede coördinaat $\cosh x - 1$: de kromme is de
\emph{kettinglijn} $y = \cosh x - 1$. Dit sluit de lus met
\cref{exo:b2:curves:3}, waar we $R = \cosh^2 x = 1 +
\sinh^2 x = 1 + s^2$ direct berekenden: de fundamentele
stelling garandeert dat de kettinglijn de \emph{enige}
kromme is met dit krommingsprofiel, op een directe isometrie
na.
\end{example}

\begin{remark}[Waar kromming hierna wordt gebruikt]
De ontbinding $\gamma'' = v'T + v^2\kappa N$ verkregen in het
bewijs van \cref{prop:b2:curves:kappaformula} is de kinematica
van elke gekromde beweging: tangentiële versus centripetale
versnelling. \omterm{thm:b2:curves:frenet2d}{Kromming} keert terug voor oppervlakken
(\cref{ch:b2:surfaces}) via de \omterm{thm:b2:curves:frenet2d}{kromming} van erop getekende
krommen, en de omhullendencalculus van het weekendprobleem van
dit hoofdstuk --- evoluten, caustieken --- is de meetkundige
optica van golfvlakken. Het volume van Jaar~3 neemt het
intrinsieke standpunt opnieuw op voor deelvariëteiten van
$\R^n$.
\end{remark}

\section{Frenet-raam in de ruimte}

Zij nu $\tilde\gamma \colon J \to \R^3$ een
eenheidssnelheids-$\mathcal{C}^3$-boog die \emph{biregulair}
is: $T'(s) \neq 0$ voor alle $s$. Dan definieert $\kappa(s) =
\norm{T'(s)} > 0$ de \emph{\omterm{thm:b2:curves:frenet2d}{kromming}} (geen teken in de ruimte:
er is geen voorkeursoriëntatie van de normaal), en we stellen:
\[
N(s) = \frac{T'(s)}{\kappa(s)}
\quad\text{(hoofdnormaal)},
\qquad
B(s) = T(s) \wedge N(s)
\quad\text{(binormaal)} ,
\]
zodat $(T, N, B)$ een direct orthonormaal raam is, het
\emph{Frenet-raam}. Het vlak door $\tilde\gamma(s)$ opgespannen
door $T, N$ is het \emph{osculatievlak}.

\begin{theorem}[Frenet-formules in de
ruimte]\label{thm:b2:curves:frenet3d}
Er is een continue functie $\tau$, de
\emph{torsie}\index{torsie}, met
\[
T' = \kappa N, \qquad
N' = -\kappa T + \tau B, \qquad
B' = -\tau N .
\]
\end{theorem}

\begin{proof}
De eerste formule is de definitie van $N$. Elk van de vectoren
$T, N, B$ heeft constante \omterm{def:b2:nvs:norm}{norm} $1$ en ze zijn paarsgewijs
orthogonaal; differentiëren van de zes relaties $\langle X,
Y\rangle = \delta_{XY}$ toont dat de matrix van $(T', N', B')$
in de basis $(T, N, B)$ antisymmetrisch is: immers $\langle X',
Y\rangle + \langle X, Y'\rangle = 0$ en $\langle X', X\rangle =
0$. Haar $(N, T)$-entry is $\langle N', T\rangle = -\langle N,
T'\rangle = -\kappa$, en haar $(T, B)$-kolomentry $\langle T',
B\rangle = \kappa\langle N, B\rangle = 0$. De resterende vrije
entry $\tau = \langle N', B\rangle$ noemen geeft precies de
drie weergegeven formules: antisymmetrie vult $\langle B',
N\rangle = -\tau$ en $\langle B', T\rangle = 0$ in. Continuïteit
van $\tau = \langle N', B\rangle$ is duidelijk omdat $N'$ en
$B$ continu zijn ($\tilde\gamma$ is $\mathcal{C}^3$, dus $N =
T'/\kappa$ is $\mathcal{C}^1$).
\end{proof}

\begin{example}[De Darboux-vector]\label{ex:b2:curves:darboux}
De drie Frenet-formules comprimeren tot één. Stel $\omega(s) =
\tau\,T + \kappa\,B$ (de \emph{Darboux-vector}). Met $B \wedge
T = N$, $T \wedge N = B$, $N \wedge B = T$:
\[
\omega \wedge T = \kappa\,N = T', \qquad
\omega \wedge N = \tau\,B - \kappa\,T = N', \qquad
\omega \wedge B = -\tau\,N = B' :
\]
elke raamvector evolueert door $X' = \omega \wedge X$, de
kinematische handtekening van een \emph{momentane rotatie} met
hoek­snelheidsvector $\omega$. Het raam draait met snelheid
$\norm\omega = \sqrt{\kappa^2 + \tau^2}$ om de bewegende as
$\omega$; \omterm{thm:b2:curves:frenet2d}{kromming} is de component van de draaiing om de
binormaal, \omterm{thm:b2:curves:frenet3d}{torsie} de component om de tangens. Voor de
schroeflijn is $\omega$ een constante vector langs de as van de
cilinder --- precies daarom precesseert het raam van de
schroeflijn gelijkmatig. De antisymmetrie van de
Frenet-matrix, uitgebuit in \cref{exo:b2:curves:7}, is de
matrixvorm van dit ene meetkundige feit.
\end{example}

\begin{proposition}[Torsie meet
planariteit]\label{prop:b2:curves:torsion}
Een biregulaire boog ligt in een vlak dan en slechts dan als
$\tau \equiv 0$; in dat geval is het vlak het (constante)
osculatievlak.
\end{proposition}

\begin{proof}
Als $\tau \equiv 0$, dan $B' = 0$, dus is $B$ een constante
eenheidsvector $B_0$, en
\[
\frac{\dd}{\dd s}\langle \tilde\gamma(s), B_0\rangle
= \langle T(s), B_0\rangle = 0 :
\]
$\langle \tilde\gamma, B_0\rangle$ is constant, dus ligt de
boog in een vlak orthogonaal op $B_0$. Omgekeerd, als de boog
in een vlak $P$ ligt, dan zijn $T$ en $T'$ (dus $N$) evenwijdig
aan de richting van $P$ voor alle $s$; dus is $B = T \wedge N$
één van de twee eenheidsnormalen van $P$, en omdat ze continu
is is ze constant; dan $0 = B' = -\tau N$ met $N \neq 0$
dwingt $\tau \equiv 0$.
\end{proof}

\begin{example}[Een gekantelde cirkel heeft nultorsie]
De boog $\gamma(t) = \bigl(\cos t,\ \tfrac{\sin t}{\sqrt2},\
\tfrac{\sin t}{\sqrt2}\bigr)$ ligt in het vlak $y = z$, en is
de eenheidscirkel van dat vlak (controle: $\norm{\gamma(t)} =
1$ en de orthonormale basis $(1,0,0)$, $(0, \tfrac1{\sqrt2},
\tfrac1{\sqrt2})$ van het vlak vertoont de standaardparametrisatie).
Zonder enige Frenet-berekening voorspelt
\cref{prop:b2:curves:torsion} $\tau \equiv 0$, en de vaste
binormaal moet de eenheidsnormaal $\pm(0, \tfrac1{\sqrt2},
-\tfrac1{\sqrt2})$ van het vlak zijn. \omterm{thm:b2:curves:frenet3d}{Torsie} meet niet
``gekanteld in de ruimte zijn''; ze meet \emph{een vlak
verlaten}. Alleen de niet-nulle $b$ van de schroeflijn hieronder
produceert echte \omterm{thm:b2:curves:frenet3d}{torsie}.
\end{example}

\begin{example}[De schroeflijn]\label{ex:b2:curves:helixfrenet}
Voor de schroeflijn $\gamma(t) = (a\cos t, a\sin t, bt)$, $a >
0$, berekenden we $s = ct$ met $c = \sqrt{a^2 + b^2}$. Dan
\[
T = \frac1c(-a\sin t,\ a\cos t,\ b),
\qquad
T' \cdot \frac{\dd t}{\dd s}
= \frac{1}{c^2}(-a\cos t, -a\sin t, 0),
\]
dus $\kappa = a/c^2 = a/(a^2 + b^2)$ en $N = (-\cos t, -\sin t,
0)$: de hoofdnormaal wijst horizontaal naar de as. Vervolgens
$B = T \wedge N = \frac1c(b\sin t, -b\cos t, a)$, en $B'
\frac{\dd t}{\dd s} = \frac{b}{c^2}(\cos t, \sin t, 0) =
-\tau N$ geeft
\[
\boxed{\ \kappa = \frac{a}{a^2 + b^2}, \qquad
\tau = \frac{b}{a^2 + b^2}. \ }
\]
Beide invarianten zijn constant --- en men kan omgekeerd
tonen dat de enige biregulaire krommen met constante $\kappa >
0$ en constante $\tau$ schroeflijnen zijn (cirkels wanneer
$\tau = 0$). Let op de tekens: $b > 0$ geeft een
rechtshandige schroeflijn met positieve \omterm{thm:b2:curves:frenet3d}{torsie}.
\end{example}

\begin{example}[Kromming en torsie zonder booglengte; de
verdraaide kubische]\label{ex:b2:curves:twistedcubic}
Herparametriseren op \omterm{def:b2:curves:length}{booglengte} is meestal in gesloten vorm
onmogelijk, dus moeten de invarianten uit de ruwe afgeleiden
worden gehaald. Schrijf $v = \norm{\gamma'} = s'$; dan
$\gamma' = vT$ en, zoals in het bewijs van
\cref{prop:b2:curves:kappaformula},
\[
\gamma'' = v'T + v^2\kappa N,
\qquad
\gamma' \wedge \gamma'' = v^3\kappa\,(T \wedge N) =
v^3\kappa\,B .
\]
\omterm{def:b2:nvs:norm}{Normen} nemen ($\kappa \geq 0$ in de ruimte):
\[
\kappa = \frac{\norm{\gamma' \wedge \gamma''}}{v^3} .
\]
Nogmaals differentiëren van $\gamma''$ en $N' =
v(-\kappa T + \tau B)$ omzetten (kettingregel door $s$), de
enige $B$-component komt van de laatste term:
\[
\gamma''' = \bigl(v'' - v^3\kappa^2\bigr)T +
\bigl(v'v\kappa + (v^2\kappa)'\bigr)N + v^3\kappa\tau\,B,
\]
zodat, gepaard met $\gamma' \wedge \gamma'' = v^3\kappa B$,
\[
\det(\gamma', \gamma'', \gamma''') = \langle\gamma' \wedge
\gamma'',\ \gamma'''\rangle = v^6\kappa^2\tau,
\qquad\text{d.w.z.}\qquad
\tau = \frac{\det(\gamma', \gamma'',
\gamma''')}{\norm{\gamma' \wedge \gamma''}^2} .
\]
Toepassing op de \emph{verdraaide kubische} $\gamma(t) = (t,\
t^2,\ t^3)$ in $t = 0$: $\gamma' = (1, 0, 0)$, $\gamma'' =
(0, 2, 0)$, $\gamma''' = (0, 0, 6)$, dus $v = 1$,
\[
\gamma' \wedge \gamma'' = (0, 0, 2), \qquad \kappa(0) = 2,
\qquad
\det(\gamma', \gamma'', \gamma''') = 12, \qquad \tau(0) =
\frac{12}{4} = 3 .
\]
Slotinzicht: beide formules zijn verhoudingen waarin de
snelheid $v$ tot precies de nodige graad wegvalt --- $\kappa$
schaalt als een tweede afgeleide per lengte-eenheid, $\tau$
als het gemengde volume van drie afgeleiden per
kwadraatoppervlak --- wat is \emph{waarom} ze meetkundig zijn
terwijl $\gamma''$ zelf dat niet is.
\end{example}

\begin{remark}[Fundamentele stelling voor ruimtekrommen]
Zoals in het vlak bepaalt het paar $(\kappa, \tau)$ met
$\kappa > 0$ een biregulaire boog op een directe isometrie van
$\R^3$ na: de Frenet-formules vormen een lineair
differentiaalstelsel voor het raam $(T, N, B)$, waarop de
Cauchy--Lipschitz-theorie van \cref{ch:b2:diffeq} van
toepassing is; orthonormaliteit van het oplossingsraam blijft
bewaard omdat de coëfficiëntenmatrix antisymmetrisch is
(zelfde Gram-matrixargument als in \cref{exo:b2:curves:7}),
en de kromme wordt herwonnen door $T$ te integreren. We laten
de details aan de lezer als een stevige maar leerzame
oefening.
\end{remark}

\section{Lokale studie: ligging ten opzichte van de raaklijn}

\begin{proposition}[Lokale vorm in een regulier
punt]\label{prop:b2:curves:local}
Zij $\gamma$ een vlakke boog van klasse $\mathcal{C}^k$ in
$t_0$, met $p$ de kleinste index met $\gamma^{(p)}(t_0) \neq
0$ en $q$ de kleinste index $> p$ met $\gamma^{(q)}(t_0)$ niet
collineair met $\gamma^{(p)}(t_0)$ (beide bestaan verondersteld,
$q \leq k$). In de basis $(u, v) = (\gamma^{(p)}(t_0),
\gamma^{(q)}(t_0))$ gecentreerd in $\gamma(t_0)$ geeft
Taylor--Young coördinaten
\[
X(t) \sim \frac{(t - t_0)^p}{p!},
\qquad
Y(t) \sim \frac{(t - t_0)^q}{q!} .
\]
Het lokale beeld hangt alleen af van de pariteiten van $p$ en
$q$:
\begin{center}
\small
\begin{tabular}{@{}c c p{0.42\textwidth}@{}}
$p$ oneven, $q$ even & \emph{ordinair punt} &
kromme blijft aan één kant van de \omterm{def:b2:curves:arc}{raaklijn}\\
$p$ oneven, $q$ oneven & \emph{buigpunt} &
kromme kruist haar \omterm{def:b2:curves:arc}{raaklijn}\\
$p$ even, $q$ oneven & \emph{keerpunt 1e soort} &
takken aan overstaande kanten van de \omterm{def:b2:curves:arc}{raaklijn}\\
$p$ even, $q$ even & \emph{keerpunt 2e soort} &
beide takken aan dezelfde kant\\
\end{tabular}
\end{center}
\end{proposition}

\begin{proof}
Taylor--Young van orde $q$ (de functie $\gamma$ is
$\mathcal{C}^q$ nabij $t_0$):
\[
\gamma(t) - \gamma(t_0)
= \sum_{j=p}^{q} \frac{(t-t_0)^j}{j!}\,\gamma^{(j)}(t_0)
+ o\bigl((t-t_0)^q\bigr).
\]
Door de keuze van $p$ en $q$ is elke $\gamma^{(j)}(t_0)$ met
$p \leq j < q$ collineair met $u$; componenten verzamelen in
de basis $(u, v)$: $X(t) = \frac{(t-t_0)^p}{p!}(1 + o(1))$ en
$Y(t) = \frac{(t-t_0)^q}{q!}(1 + o(1))$. De tekentabel van $X$
en $Y$ voor $t \gtrless t_0$ --- precies bestuurd door de
pariteiten --- geeft de vier beelden: bijvoorbeeld als $p$
even is, $X > 0$ aan beide kanten (beide takken vertrekken in
de richting $+u$: een keerpunt), en de kant van de \omterm{def:b2:curves:arc}{raaklijn}
($\operatorname{sign} Y$) wisselt met $q$ oneven.
\end{proof}

\begin{example}
Voor $\gamma(t) = (t^2, t^3)$ in $t_0 = 0$
(\cref{ex:b2:curves:cusp}): $\gamma'' (0)= (2, 0)$,
$\gamma'''(0) = (0, 6)$, dus $p = 2$, $q = 3$: een keerpunt
van de eerste soort, het vertrouwde beeld van de semicubische
parabool. Voor $\gamma(t) = (t, t^3)$ in $0$: $p = 1$, $q =
3$: buigpunt --- de kubische kruist haar \omterm{def:b2:curves:arc}{raaklijn}.
\end{example}

\begin{remark}[Perspectieven binnen dit volume]
Krommen voeden de volgende hoofdstukken op drie manieren. Op
een oppervlak getekend, definiëren ze haar raakvlakken en
haar eerste fundamentaalvorm (\cref{ch:b2:surfaces}), en hun
\omterm{def:b2:curves:length}{lengten} worden berekend door de omgevende metriek te
beperken --- het komende hoofdstuk is grotendeels dit
hoofdstuk gerelativeerd. De omhullendencalculus van het
weekendprobleem ontmoet dubbelintegralen in
\cref{ch:b2:multint}, waar de oppervlakte van de \omterm{pb:b2:curves:1}{astroïde}
opnieuw wordt berekend met de formule van Green
(\cref{exo:b2:multint:5}) --- één kromme, twee theorieën,
overeenkomende antwoorden. En het Frenet-stelsel gebruikte al
de lineaire differentiaalvergelijkingen van
\cref{ch:b2:diffeq} (bestaan, uniciteit, en het
orthogonaliteitsbehoudargument van \cref{exo:b2:curves:7}):
de fundamentele stelling van krommen is een
differentiaalvergelijkingenstelling in meetkundige kleren.
\end{remark}

\begin{remark}[Methode: de lokale studie uitvoeren]
In de praktijk is de classificatie een routine van vier
stappen. \emph{Eén}, differentieer in $t_0$ tot de eerste
niet-nulle afgeleide verschijnt: haar index is $p$, haar
waarde de vector $u$. \emph{Twee}, blijf differentiëren tot
een afgeleide niet collineair met $u$ verschijnt: index $q$,
vector $v$. \emph{Drie}, lees de pariteiten $(p, q)$ in de
tabel. \emph{Vier}, teken: de kromme vertrekt langs $+u$ als
$p$ oneven is (langs $u$ dan terug langs $u$ als $p$ even
is), aan de kant van $v$ voorgeschreven door het teken van
$Y$. Twee voorzorgen. Het raam $(u, v)$ is in het algemeen
\emph{niet} orthonormaal --- de tabel beschrijft posities ten
opzichte van de \omterm{def:b2:curves:arc}{raaklijn}, geen hoeken of afstanden, dus lees
geen \omterm{thm:b2:curves:frenet2d}{kromming} van de tekening. En intermediaire afgeleiden
collineair met $u$ zijn toegestaan tussen rangen $p$ en $q$
(ze verschuiven alleen de ontwikkeling van $X$); wat
\emph{niet} mag gebeuren is stoppen bij de eerste niet-nulle
afgeleide en $q = p + 1$ raden: voor $\gamma(t) = (t^2, t^4
+ t^5)$ is de naïeve gok $q = 3$ fout, omdat
$\gamma^{(3)}(0)$ nog collineair is met $\gamma''(0)$ ---
dat is precies \cref{exo:b2:curves:5}.
\end{remark}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:curves:4}
(Logaritmische spiraal) Zij $\gamma(t) = e^{t}(\cos t,\ \sin
t)$, $t \in \R$. Toon dat de hoek tussen $\gamma(t)$ en
$\gamma'(t)$ constant is, bereken de \omterm{def:b2:curves:length}{booglengte} van $\gamma$
op $(-\infty, 0]$ (eindig!), en de \omterm{thm:b2:curves:frenet2d}{kromming}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:curves:5}
Bepaal $p$, $q$ en de lokale vorm (ordinair, buigpunt,
keerpunt) van $\gamma(t) = (t^2,\ t^4 + t^5)$ in $t = 0$, en
van $\gamma(t) = (t^3,\ t^4)$ in $t = 0$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:curves:6}
Zij $\gamma$ een eenheidssnelheids vlakke boog met $\kappa(s)
> 0$ voor alle $s$, en zij $c(s) = \gamma(s) +
\frac{1}{\kappa(s)}N(s)$ het \omterm{def:b2:curves:curvature}{kromtemiddelpunt} (de kromme $c$
is de \emph{evolute}). Onder de aanname dat $\kappa$
$\mathcal{C}^1$ is, toon $c'(s) =
-\frac{\kappa'(s)}{\kappa(s)^2}N(s)$: de evolute is raak aan
de normaallijnen van $\gamma$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:curves:7}
Zij $A(s)$ een continue familie van antisymmetrische $3
\times 3$-matrices en $F' = F A$ een matrixoplossing met
$F(s_0)$ orthogonaal. Toon dat $F(s)$ orthogonaal is voor
alle $s$. \emph{(Differentiëer $G = F F^{\mathsf T}$ en
gebruik uniciteit in Cauchy--Lipschitz.)} Verklaar de
relevantie voor het Frenet-stelsel.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:curves:8}
(Totale \omterm{thm:b2:curves:frenet2d}{kromming} van een gesloten convexe kromme) Zij
$\tilde\gamma$ een eenheidssnelheids-$\mathcal{C}^2$ gesloten
vlakke boog van \omterm{def:b2:curves:length}{lengte} $L$ (dus $\tilde\gamma(s + L) =
\tilde\gamma(s)$), éénmaal linksom doorlopen. Met de
hoekfunctie $\varphi$ met $T = (\cos\varphi, \sin\varphi)$
uit \cref{thm:b2:curves:fundamental}, verklaar waarom
$\varphi(L) - \varphi(0)$ een veelvoud van $2\pi$ is, en
toon dat $\int_0^L \kappa(s)\,\dd s = \varphi(L) -
\varphi(0)$. (Voor een cirkel van straal $R$: $\int \kappa =
\frac1R \cdot 2\pi R = 2\pi$. De stelling van de draaiende
tangens beweert de waarde $2\pi$ voor elke eenvoudige
gesloten kromme; u wordt niet gevraagd dat te bewijzen.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:curves:9}
Toon dat een biregulaire ruimtekromme met constante $\kappa >
0$ en $\tau = 0$ (een boog van) een cirkel van straal
$1/\kappa$ is. \emph{(Gebruik \cref{prop:b2:curves:torsion},
toon daarna dat het centrum $\gamma + \frac1\kappa N$
constant is.)}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:curves:10}
Bereken de \omterm{def:b2:curves:length}{booglengte} van de parabool $y = x^2/2$ over
$\intcc0a$ en toon dat ze gelijk is aan
\[
\tfrac12\Bigl(a\sqrt{1 + a^2} + \ln\bigl(a + \sqrt{1 +
a^2}\bigr)\Bigr).
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:curves:11}
Zij $\gamma$ een reguliere $\mathcal C^2$-boog in $\R^n$
waarvan alle \omterm{def:b2:curves:arc}{raaklijnen} door een vast punt $P$ gaan. Bewijs
dat de baan van $\gamma$ bevat is in een rechte lijn.
\emph{(Parametriseer op \omterm{def:b2:curves:length}{booglengte}, schrijf $\gamma(s) +
\lambda(s)T(s) = P$ en differentieer.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:curves:12}
(Fundamentele stelling voor ruimtekrommen) Zij $\kappa > 0$
en $\tau$ continue functies op een interval $J$. Voer het
programma uit van de opmerking na
\cref{ex:b2:curves:helixfrenet}: (a) toon dat het lineaire
stelsel $F' = FA(s)$, met $A(s)$ de antisymmetrische
Frenet-matrix gebouwd uit $\kappa, \tau$ en $F(s_0)$ een
direct orthonormaal raam, een unieke globale oplossing
heeft, die een direct orthonormaal raam blijft; (b)
construeer een eenheidssnelheids biregulaire kromme met
\omterm{thm:b2:curves:frenet2d}{kromming} $\kappa$ en \omterm{thm:b2:curves:frenet3d}{torsie} $\tau$; (c) bewijs uniciteit op
een directe isometrie van $\R^3$ na.
\end{exercise}

\section{Probleem: omhullenden --- de astroïde, twee evoluten,
en een caustiek}

  \draw[->] (0,-1.15) -- (0,1.25) node[left] {$y$};
  \draw[omDef] (1,0) -- (0,0.0) (0.966,0) -- (0,0.259)

{\small Een ladder van \omterm{def:b2:curves:length}{lengte} $1$ die langs een muur
afglijdt (blauwe posities) kruist nooit de \omterm{pb:b2:curves:1}{astroïde}
$x^{2/3} + y^{2/3} = 1$ (rood): de \omterm{pb:b2:curves:1}{astroïde} is de
\emph{\omterm{pb:b2:curves:1}{omhullende}} van de familie segmenten, raak aan elk

\begin{problem}[{Weekendprobleem --- de omhullendemachine en
vier klassieke krommen}]\label{pb:b2:curves:1}
Een éénparameterfamilie van lijnen dekt het vlak meestal niet
gelijkmatig: de lijnen stapelen zich op langs een kromme raak
aan ze allemaal, hun \emph{omhullende}\index{omhullende}.
Lichtstralen maken \omterm{pb:b2:curves:1}{omhullenden} zichtbaar als
\emph{caustieken} --- de heldere gekeerde kromme in een mok
koffie. Dit probleem bouwt de algemene omhullendemachine, en
draait haar dan vier keer: de glijdende ladder (\omterm{pb:b2:curves:1}{astroïde}), de
normalen van de parabool en van de cycloïde (evoluten, met
Huygens' slinger aan het eind), en de koffiekop-caustiek
(\omterm{pb:b2:curves:1}{nefroïde}). Doorheen duidt $D_t$ de lijn van vergelijking
$a(t)\,x + b(t)\,y = c(t)$ aan, waar $a, b, c$
$\mathcal C^2$-functies zijn met $(a(t), b(t)) \neq (0,0)$,
en $\Delta(t) = a(t)b'(t) - a'(t)b(t)$.

\medskip
\noindent\textbf{Deel I --- De omhullendemachine.}
\begin{enumerate}
  \item Stel $\Delta(t) \neq 0$. Toon dat het
        \emph{karakteristieke stelsel}
        \[
        \begin{cases} a(t)\,x + b(t)\,y = c(t)\\
        a'(t)\,x + b'(t)\,y = c'(t)\end{cases}
        \]
        een unieke oplossing $E(t) = (x(t), y(t))$ heeft,
        gegeven door $x = \dfrac{cb' - c'b}{\Delta}$, $y =
        \dfrac{ac' - a'c}{\Delta}$.
  \item Neem bovendien aan dat $E$ $\mathcal C^1$ nabij $t$
        is met $E'(t) \neq 0$. Door de eerste vergelijking
        van het stelsel te differentiëren, toon $a(t)\,x'(t)
        + b(t)\,y'(t) = 0$, en concludeer dat de kromme $E$
        door een punt van $D_t$ gaat \emph{met de richting
        van} $D_t$: de familie is raak aan $E$, die haar
        \emph{\omterm{pb:b2:curves:1}{omhullende}} heet.
  \item Gezondheidscontrole: de \omterm{def:b2:curves:arc}{raaklijnen} van de parabool
        $y = x^2/2$ in de punten $(t, t^2/2)$ zijn $tx - y =
        t^2/2$. Verifieer dat de omhullendemachine de
        parabool zelf teruggeeft.
  \item (\omterm{pb:b2:curves:1}{Omhullende} van de normalen) Zij $\gamma$
        eenheidssnelheid met $\kappa(s) \neq 0$. De
        normaallijn in $\gamma(s)$ is $\{M : \langle M -
        \gamma(s), T(s) \rangle = 0\}$. Toon dat haar
        karakteristieke stelsel $\langle M - \gamma(s),
        N(s)\rangle = 1/\kappa(s)$ dwingt, dus dat het
        karakteristieke punt het \omterm{def:b2:curves:curvature}{kromtemiddelpunt} is:
        \emph{de \omterm{pb:b2:curves:1}{omhullende} van de normalen is de evolute},
        wat \cref{exo:b2:curves:6} herwint. Controleer
        $\Delta(s) = \kappa(s)$.
  \item Twee ontaardingen. Voor de bundel $D_\theta :
        x\cos\theta + y\sin\theta = 0$, toon dat het
        karakteristieke punt de oorsprong is voor elke
        $\theta$ (de ``\omterm{pb:b2:curves:1}{omhullende}'' stort in tot een punt,
        en $E' = 0$: vraag 2 is niet van toepassing). Voor
        een familie evenwijdige lijnen ($a, b$ constant),
        toon $\Delta \equiv 0$ en dat het karakteristieke
        stelsel in het algemeen inconsistent is: geen
        \omterm{pb:b2:curves:1}{omhullende}.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De glijdende ladder en de
\omterm{pb:b2:curves:1}{astroïde}.} Een segment van \omterm{def:b2:curves:length}{lengte} $1$ glijdt met één
uiteinde $P_t = (\cos t, 0)$ op de vloer en het andere $Q_t
= (0, \sin t)$ op de muur, $t \in \intoo0{\pi/2}$.
\begin{enumerate}[resume]
  \item Toon dat de lijn $(P_tQ_t)$ vergelijking $x\sin t +
        y\cos t = \sin t\cos t$ heeft, en dat de
        omhullendemachine het karakteristieke punt
        \[
        E(t) = (\cos^3 t,\ \sin^3 t) :
        \]
        geeft: de \emph{astroïde}\index{astroïde}, van
        impliciete vergelijking $x^{2/3} + y^{2/3} = 1$
        (uitgebreid tot de andere kwadranten door
        symmetrie).
  \item Toon dat $E'(0) = 0$ en, met de lokale classificatie
        (\cref{prop:b2:curves:local}), dat de \omterm{pb:b2:curves:1}{astroïde} een
        keerpunt van de eerste soort heeft in $(1, 0)$ ---
        en evenzo in haar vier as-punten.
  \item Bereken $\norm{E'(t)} = \tfrac32\abs{\sin 2t}$ en
        leid af dat de totale \omterm{def:b2:curves:length}{lengte} van de \omterm{pb:b2:curves:1}{astroïde} $6$ is.
  \item Waar raakt de ladder de \omterm{pb:b2:curves:1}{astroïde}? Toon $E(t) = P_t +
        \sin^2 t\,(Q_t - P_t)$: het raakpunt deelt de ladder
        in de verhouding $\sin^2 t : \cos^2 t$, en veegt
        haar van het ene uiteinde naar het andere terwijl de
        ladder glijdt.
  \item Bereken de door de \omterm{pb:b2:curves:1}{astroïde} ingesloten oppervlakte:
        toon dat de eerste-kwadrantoppervlakte
        $3\int_0^{\pi/2}\sin^4 t\cos^2 t\,\dd t$ is,
        evalueer de integraal door linearisatie ($\sin^2 2t
        = \tfrac{1 - \cos 4t}2$), en concludeer dat de
        totale oppervlakte $3\pi/8$ is.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De evolute van de parabool.}
Zij $\gamma(t) = (t, t^2/2)$.
\begin{enumerate}[resume]
  \item Toon dat de normaallijn in $\gamma(t)$ vergelijking
        $x + t\,y = t + t^3/2$ heeft.
  \item Draai de omhullendemachine: toon dat de \omterm{pb:b2:curves:1}{omhullende}
        van de normalen
        \[
        E(t) = \Bigl(-t^3,\ 1 + \tfrac32 t^2\Bigr),
        \]
        is, met impliciete vergelijking $x^2 =
        \tfrac8{27}(y - 1)^3$: een semicubische parabool.
  \item Controleer met vraag 4: bereken het
        \omterm{def:b2:curves:curvature}{kromtemiddelpunt} $\gamma(t) +
        \frac1{\kappa(t)}N(t)$ uit $\kappa(t) = (1 +
        t^2)^{-3/2}$ (\cref{ex:b2:curves:kappaexamples}) en
        herwin hetzelfde punt.
  \item Toon dat de evolute een keerpunt van de eerste soort
        heeft in $(0, 1)$, het \omterm{def:b2:curves:curvature}{kromtemiddelpunt} in de top
        --- het punt waar $\kappa$ extreem is, zoals
        voorspeld door de formule $c' =
        -\frac{\kappa'}{\kappa^2}N$ van
        \cref{exo:b2:curves:6}.
  \item Hoeveel normalen van de parabool gaan door een
        gegeven punt $(x_0, y_0)$? Toon dat het antwoord
        bestuurd wordt door de kubische $\tfrac{t^3}2 + (1
        - y_0)\,t - x_0 = 0$; behandel het as-geval $x_0 =
        0$ \omterm{def:b2:metric:complete}{volledig} (één normaal voor $y_0 < 1$, drie voor
        $y_0 > 1$), en interpreteer de evolute als de
        overgangskromme.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De koffiekop-caustiek.}
Evenwijdige stralen van richting $(1, 0)$ treffen de
binnenkant van de spiegelcirkel $x^2 + y^2 = 1$; de straal
die $P_\theta = (\cos\theta, \sin\theta)$ treft, kaatst
volgens de reflectiewet.
\begin{enumerate}[resume]
  \item Uit de spiegelsymmetrie in de normaal (de straal),
        rechtvaardig dat de gereflecteerde richting $v = u -
        2\langle u, n\rangle n$ is met $u = (1,0)$, $n =
        (\cos\theta, \sin\theta)$, en bereken $v =
        -(\cos2\theta, \sin2\theta)$.
  \item Toon dat de gereflecteerde straal op de lijn
        \[
        x\sin 2\theta - y\cos 2\theta = \sin\theta .
        \]
        ligt.
  \item Draai de omhullendemachine ($\Delta = 2$): toon dat
        de caustiek
        \[
        E(\theta) = \Bigl(\tfrac{3\cos\theta -
        \cos3\theta}4,\ \tfrac{3\sin\theta -
        \sin3\theta}4\Bigr),
        \]
        is, de \emph{nefroïde}\index{nefroïde}.
  \item Bereken $E'(\theta) = \tfrac32\sin\theta\,
        (\cos2\theta, \sin2\theta)$; controleer dat de
        raakrichting de gereflecteerde-straalrichting is
        (vraag 16), lokaliseer de twee keerpunten
        $(\pm\tfrac12, 0)$, en toon dat de gereflecteerde
        straal de as $y = 0$ snijdt in $x =
        \frac1{2\cos\theta}$ --- dus focussen bijna-axiale
        stralen in $x = \tfrac12$: de brandpuntsafstand
        $R/2$ van een spiegel van straal $R$.
  \item Toon dat de \omterm{pb:b2:curves:1}{nefroïde} totale \omterm{def:b2:curves:length}{lengte} $6$ heeft en dat
        nabij $\theta = 0$,
        \[
        E(\theta) - \bigl(\tfrac12, 0\bigr) =
        \bigl(\tfrac34\theta^2 + o(\theta^2),\ \theta^3 +
        o(\theta^3)\bigr) :
        \]
        een keerpunt van de eerste soort, wijzend langs de
        as.
  \item Verklaar in één alinea waarom de caustiek helder is:
        door elk punt net buiten de caustiek gaan twee
        gereflecteerde stralen, door elk punt erop zijn de
        stralen ``oneindig geconcentreerd'' (de afbeelding
        $(\theta, \text{afstand langs straal}) \mapsto
        \R^2$ heeft precies op de \omterm{pb:b2:curves:1}{omhullende} een
        kritiek punt).
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Huygens: de cycloïde is haar
eigen evolute.} Zij $\gamma(t) = (t - \sin t,\ 1 - \cos t)$,
$t \in \intoo0{2\pi}$, één boog van de cycloïde.
\begin{enumerate}[resume]
  \item Bereken $\kappa(t) = -\dfrac1{4\sin(t/2)}$ en het
        \omterm{def:b2:curves:curvature}{kromtemiddelpunt}; toon dat de evolute
        \[
        c(t) = (t + \sin t,\ \cos t - 1),
        \]
        is, en dat de substitutie $t = u + \pi$ haar als de
        oorspronkelijke cycloïde getransleerd over $(\pi,
        -2)$ vertoont: \emph{de evolute van een cycloïde is
        een congruente cycloïde} (Huygens).
  \item Verifieer dat de \omterm{def:b2:curves:curvature}{kromtestraal} in de top $t = \pi$
        gelijk is aan $4$, wat de helft is van de \omterm{def:b2:curves:length}{lengte} $8$
        van één boog (\cref{exo:b2:curves:1}); lokaliseer
        het keerpunt van de evolute direct onder de top, op
        afstand $4$.
  \item (De snaareigenschap) Zij $\gamma$ eenheidssnelheid
        met $\kappa > 0$, $\kappa$ van klasse $\mathcal
        C^1$ en $R = 1/\kappa$ strikt monotoon. Met $c' =
        R'N$, toon dat de \omterm{def:b2:curves:length}{booglengte} van de evolute tussen
        $c(s_0)$ en $c(s_1)$ gelijk is aan $\abs{R(s_1) -
        R(s_0)}$. Interpreteer: een strakke snaar
        afgewikkeld van de evolute, van \omterm{def:b2:curves:length}{lengte} $R(s_0)$ bij
        de start, heeft haar vrije uiteinde de
        oorspronkelijke kromme beschrijven --- dus
        beschrijft een slinger die tussen twee cycloïdale
        wangen van Huygens' klok zwaait een cycloïde.
  \item Synthese. De machine van Deel I produceerde de
        \omterm{pb:b2:curves:1}{astroïde}, een semicubische parabool, een \omterm{pb:b2:curves:1}{nefroïde}
        en een cycloïde. Voor elk van de vier families,
        formuleer in één zin waar de hypotheses $\Delta
        \neq 0$ en $E' \neq 0$ golden of faalden, en welk
        meetkundig gebeuren (keerpunt, brandpunt,
        ontaarding) elk falen van $E' \neq 0$ signaleerde.
        Waar moeten extrema van \omterm{thm:b2:curves:frenet2d}{kromming} op de \omterm{pb:b2:curves:1}{omhullende}
        van de normalen verschijnen, en waarom?
\end{enumerate}
\end{problem}
