\chapter{Oppervlakken}\label{ch:b2:surfaces}

Na de krommen, de oppervlakken: tweeparameterobjecten in $\R^3$. De
differentiaalrekening van \cref{ch:b2:diffcalc} levert alles wat we
nodig hebben --- partiële afgeleiden geven raakvectoren, het
kruisproduct de normaal, \omterm{def:b2:linalg:det}{determinanten} de \omterm{def:b2:surfaces:area}{oppervlakten}. We
definiëren reguliere \omterm{def:b2:surfaces:param}{geparametriseerde oppervlakken}, hun
raakvlakken, en de \emph{\omterm{def:b2:surfaces:fff}{eerste fundamentaalvorm}}, die alle
lengte- en oppervlaktemetingen op het oppervlak codeert.
Oppervlakken ontstaan ook als niveauverzamelingen $f(x, y, z) =
c$; de gradiënt richt dan de normaal.

\section{Geparametriseerde oppervlakken}

\begin{definition}[Regulier geparametriseerd
oppervlak]\label{def:b2:surfaces:param}
Zij $U \subseteq \R^2$ open. Een \emph{geparametriseerd
oppervlak}\index{geparametriseerd oppervlak} van klasse
$\mathcal{C}^k$ ($k \geq 1$) is een afbeelding $\sigma \colon U
\to \R^3$, $(u, v) \mapsto \sigma(u, v)$, van klasse
$\mathcal{C}^k$. Een punt is \emph{regulier} als de partiële
afgeleidevectoren
\[
\sigma_u(u, v) = \frac{\partial\sigma}{\partial u}(u,v),
\qquad
\sigma_v(u, v) = \frac{\partial\sigma}{\partial v}(u,v)
\]
lineair onafhankelijk zijn, d.w.z.\ $\sigma_u \wedge \sigma_v
\neq 0$; het oppervlak is \emph{regulier} als elk punt dat is.
\end{definition}

\begin{example}[De drie standaardbeschrijvingen]\label{ex:b2:surfaces:standard}
\leavevmode
\begin{enumerate}
\item \emph{Grafiek:} $\sigma(u, v) = (u,\ v,\ f(u, v))$ voor $f
\in \mathcal{C}^1(U)$. Altijd regulier: $\sigma_u = (1, 0, f_u)$
en $\sigma_v = (0, 1, f_v)$ zijn onafhankelijk.
\item \emph{Bol (sferische coördinaten):} voor de bol van
straal $R$,
\[
\sigma(\theta, \varphi)
= (R\cos\theta\cos\varphi,\ R\sin\theta\cos\varphi,\
R\sin\varphi),
\qquad
(\theta, \varphi) \in \R \times
\bigl(-\tfrac\pi2, \tfrac\pi2\bigr),
\]
met $\theta$ de lengte en $\varphi$ de breedte. Men
controleert $\norm{\sigma_\theta \wedge \sigma_\varphi} =
R^2\cos\varphi > 0$: regulier weg van de polen (die deze
kaart weglaat).
\item \emph{Niveauverzameling:} $S = \{(x,y,z) : f(x, y, z) =
c\}$ waar $f$ $\mathcal{C}^1$ is en $\nabla f \neq 0$ op $S$.
Nabij elk punt kan één coördinaat als functie van de andere
twee worden uitgedrukt door de impliciete-functiestelling
(\cref{ch:b2:diffcalc}), dus is $S$ lokaal een grafiek.
\end{enumerate}
\end{example}

\begin{example}[Van niveauverzameling naar
grafiek]\label{ex:b2:surfaces:levelgraph}
De impliciete-functiestelling in item 3 verdient één
expliciete run. Neem de bol $x^2 + y^2 + z^2 = R^2$ nabij haar
noordpool $(0, 0, R)$: daar $\frac{\partial f}{\partial z} =
2z = 2R \neq 0$, en oplossen naar $z$ geeft de grafiekkaart
\[
z = \sqrt{R^2 - x^2 - y^2},
\qquad x^2 + y^2 < R^2 ,
\]
regulier overal op haar (open) domein --- inclusief de pool
die de sferische kaart miste. Nabij een equatoriaalpunt zoals
$(R, 0, 0)$ lost dezelfde stelling naar $x$ op
($\frac{\partial f}{\partial x} = 2R \neq 0$). De
vuistregel: een niveauoppervlak is een grafiek over het
coördinaatvlak \emph{orthogonaal op de grootste component van
de gradiënt}, en door de bol met zes zulke grafiekkaarten te
bedekken controleert men haar gladheid overal zonder enige
trigonometrie.
\end{example}

\section{Raakvlak en normaal}

\begin{definition}[Raakvlak]\label{def:b2:surfaces:tangent}
Zij $\sigma$ regulier in $(u_0, v_0)$, $M_0 = \sigma(u_0,
v_0)$. Het \emph{raakvlak}\index{raakvlak} $T_{M_0}S$ is het
vlak door $M_0$ gericht door $\operatorname{Vect}(\sigma_u,
\sigma_v)$ (partiëlen in $(u_0, v_0)$). De
\emph{eenheidsnormaal}\index{eenheidsnormaal} is
\[
n(u_0, v_0)
= \frac{\sigma_u \wedge \sigma_v}
       {\norm{\sigma_u \wedge \sigma_v}} .
\]
\end{definition}

\begin{proposition}[Raakvectoren zijn
snelheidsvectoren]\label{prop:b2:surfaces:velocity}
De richting van $T_{M_0}S$ is precies de verzameling vectoren
$\gamma'(0)$, waar $\gamma = \sigma \circ c$ loopt over de
$\mathcal{C}^1$-krommen op het oppervlak getekend door $M_0$
(d.w.z.\ $c \colon (-\varepsilon, \varepsilon) \to U$ is
$\mathcal{C}^1$ met $c(0) = (u_0, v_0)$).
\end{proposition}

\begin{proof}
Als $c(t) = (u(t), v(t))$, geeft de kettingregel
(\cref{ch:b2:diffcalc})
\[
\gamma'(0) = u'(0)\,\sigma_u + v'(0)\,\sigma_v
\in \operatorname{Vect}(\sigma_u, \sigma_v).
\]
Omgekeerd wordt de vector $a\sigma_u + b\sigma_v$ bereikt door
de kromme $c(t) = (u_0 + at,\ v_0 + bt)$, die voor $\abs t$
klein in de open verzameling $U$ blijft.
\end{proof}

\begin{example}[Het raakvlak van de
helicoïde]\label{ex:b2:surfaces:helicoidtangent}
Voor de helicoïde $\sigma(u, v) = (v\cos u,\ v\sin u,\ au)$,
in het punt $\sigma(0, 1) = (1, 0, 0)$:
\[
\sigma_u = (0,\ 1,\ a), \qquad \sigma_v = (1,\ 0,\ 0),
\qquad
\sigma_u \wedge \sigma_v = (0,\ a,\ -1),
\]
dus is het \omterm{def:b2:surfaces:tangent}{raakvlak} $a\,y = z$. Het bevat de hele
horizontale regellijn $\{(t, 0, 0)\}$ (richting $\sigma_v$):
zoals voor de kegel van \cref{exo:b2:surfaces:1} heeft een
oppervlak geregeerd door rechte lijnen elke regellijn binnen
het \omterm{def:b2:surfaces:tangent}{raakvlak} erlangs. De andere raakrichting $\sigma_u$ is de
snelheid van de schroeflijn $u \mapsto \sigma(u, 1)$: één
kaart, twee getekende krommen, en het hele \omterm{def:b2:surfaces:tangent}{raakvlak} is
opgespannen --- \cref{prop:b2:surfaces:velocity} in actie.
\end{example}

\begin{proposition}[Normaal van een
niveauoppervlak]\label{prop:b2:surfaces:gradient}
Zij $S = \{f = c\}$ met $f$ van klasse $\mathcal{C}^1$ en
$\nabla f(M_0) \neq 0$. Dan is het \omterm{def:b2:surfaces:tangent}{raakvlak} van $S$ in $M_0$
het vlak door $M_0$ orthogonaal op $\nabla f(M_0)$:
\[
T_{M_0}S :\quad
\langle \nabla f(M_0),\ M - M_0\rangle = 0 .
\]
\end{proposition}

\begin{proof}
Voor elke kromme $\gamma$ op $S$ getekend door $M_0$,
$f(\gamma(t)) = c$ identiek, dus geeft de kettingregel
$\langle \nabla f(M_0), \gamma'(0)\rangle = 0$: alle
snelheidsvectoren zijn orthogonaal op de gradiënt, dus is de
raakrichting bevat in het vlak $\nabla f(M_0)^\perp$. Beide
zijn $2$-dimensionale deelruimten --- de raakrichting omdat
$S$ lokaal een reguliere grafiek is
(\cref{ex:b2:surfaces:standard}), het orthogonaal complement
omdat $\nabla f(M_0) \neq 0$ --- dus zijn ze gelijk.
\end{proof}

\begin{example}\label{ex:b2:surfaces:spheretangent}
Voor de bol $x^2 + y^2 + z^2 = R^2$: $\nabla f = 2(x, y, z)$,
dus is het \omterm{def:b2:surfaces:tangent}{raakvlak} in $M_0$ orthogonaal op de straal
$\vect{OM_0}$ --- het klassieke feit dat straal en \omterm{def:b2:surfaces:tangent}{raakvlak}
loodrecht staan, met vergelijking $\langle M_0, M\rangle =
R^2$.
\end{example}

\begin{example}[Raakvlak van een
grafiek]\label{ex:b2:surfaces:graphtangent}
Voor $z = f(x, y)$ in $(x_0, y_0)$: toepassen van
\cref{prop:b2:surfaces:gradient} op $F(x,y,z) = f(x,y) - z$,
\[
z = f(x_0, y_0)
+ f_x(x_0, y_0)(x - x_0)
+ f_y(x_0, y_0)(y - y_0) ,
\]
het affiene deel van de eerste-orde Taylorontwikkeling ---
het \omterm{def:b2:surfaces:tangent}{raakvlak} is de grafiek van de \omterm{def:b2:diffcalc:differential}{differentiaal}, zoals het
moet.
\end{example}

\begin{example}[Het dichtstbijzijnde punt van een
oppervlak]\label{ex:b2:surfaces:closest}
Welk punt van de paraboloïde $z = x^2 + y^2$ is het dichtst
bij $P = (0, 0, 1)$? Minimaliseer de gekwadrateerde afstand
langs het oppervlak: met $\rho^2 = x^2 + y^2$,
\[
g(\rho^2) = \rho^2 + (\rho^2 - 1)^2,
\qquad
g'(\rho^2) = 1 + 2(\rho^2 - 1) = 0
\iff \rho^2 = \tfrac12 ,
\]
wat de cirkel van punten op hoogte $z = \frac12$ en afstand
$\sqrt{\tfrac12 + \tfrac14} = \frac{\sqrt3}2$ geeft. De
meetkundige handtekening van minimaliteit: in zo'n punt $M$
moet de vector $\vect{MP}$ \emph{normaal} op het oppervlak
zijn --- anders zou glijden langs een getekende kromme met
snelheid die een component naar $P$ heeft de afstand
verkleinen. Controle: $\nabla(z - x^2 - y^2) = (-2x, -2y,
1)$ in $M = (x, y, \tfrac12)$, terwijl $\vect{MP} = (-x, -y,
\tfrac12) = \tfrac12(-2x, -2y, 1)$: evenwijdig, zoals
voorspeld. De eerste-ordevoorwaarde ``voet van de loodlijn''
is dezelfde die extrema op niveauverzamelingen zal sturen in
\cref{exo:b2:surfaces:12}.
\end{example}

\begin{example}[Raakvlakken van kwadrieken: de
polarisatieregel]\label{ex:b2:surfaces:polarization}
Zij $S : xy + yz + zx = 1$ en $M_0 = (1, 1, 0) \in S$. Hier
$\nabla f = (y + z,\ x + z,\ x + y)$, dus $\nabla f(M_0) =
(1, 1, 2)$ en het \omterm{def:b2:surfaces:tangent}{raakvlak} is
\[
(x - 1) + (y - 1) + 2z = 0,
\qquad\text{d.w.z.}\qquad x + y + 2z = 2 .
\]
Hetzelfde antwoord komt uit de \emph{polarisatieregel} die
\cref{ex:b2:surfaces:spheretangent} en
\cref{exo:b2:surfaces:2} generaliseert: in de
kwadriekvergelijking, vervang $x^2$ door $x_0x$, en elk
product $xy$ door $\frac{x_0y + y_0x}2$ (en cyclisch):
\[
\frac{x_0y + y_0x}2 + \frac{y_0z + z_0y}2 + \frac{z_0x +
x_0z}2 = 1
\;\xrightarrow{\ M_0 = (1,1,0)\ }\;
\frac{x + y}2 + \frac z2 + \frac z2 = 1 ,
\]
wat opnieuw $x + y + 2z = 2$ is. De regel werkt omdat
$\nabla$ van een \omterm{def:b2:quadratic:def}{kwadratische vorm} de geassocieerde
bilineaire vorm geëvalueerd tegen het baspunt is ---
raakheid aan een \omterm{pb:b2:surfaces:1}{kwadriek} is polarisatie, nog een gezicht
van \cref{ch:b2:quadratic}.
\end{example}

\section{De eerste fundamentaalvorm}

\begin{definition}[Eerste fundamentaalvorm]\label{def:b2:surfaces:fff}
Zij $\sigma \colon U \to \R^3$ een regulier
$\mathcal{C}^1$-oppervlak. Haar \emph{eerste
fundamentaalvorm}\index{eerste fundamentaalvorm} in $(u,v)$
is de positief definiete \omterm{def:b2:quadratic:def}{kwadratische vorm} op $\R^2$
\[
I(h, k) = \norm{h\,\sigma_u + k\,\sigma_v}^2
= E\,h^2 + 2F\,hk + G\,k^2,
\]
waar
\[
E = \norm{\sigma_u}^2,
\qquad
F = \langle \sigma_u, \sigma_v\rangle,
\qquad
G = \norm{\sigma_v}^2 .
\]
\end{definition}

\begin{remark}
$I$ is de beperking van het omgevende euclidische inwendig
product tot het \omterm{def:b2:surfaces:tangent}{raakvlak}, gelezen in de basis $(\sigma_u,
\sigma_v)$: ze is positief definiet precies omdat $\sigma_u,
\sigma_v$ onafhankelijk zijn (\cref{ch:b2:quadratic}). Elke
metrische grootheid op het oppervlak --- lengten van
getekende krommen, hoeken ertussen, \omterm{def:b2:surfaces:area}{oppervlakten} --- wordt
alleen uit $E, F, G$ berekend. Twee oppervlakken met dezelfde
$E, F, G$ in geschikte parameters zijn \emph{isometrisch}
zelfs als ze verschillend in de ruimte zitten: dit is het
startpunt van intrinsieke meetkunde.
\end{remark}

\begin{example}[Hoeken tussen
coördinaatkrommen]\label{ex:b2:surfaces:coordangle}
De \omterm{def:b2:surfaces:fff}{eerste fundamentaalvorm} meet ook hoeken: de
coördinaatkrommen $u \mapsto \sigma(u, v_0)$ en $v \mapsto
\sigma(u_0, v)$ snijden onder de hoek $\theta$ met
\[
\cos\theta = \frac{\langle\sigma_u,
\sigma_v\rangle}{\norm{\sigma_u}\,\norm{\sigma_v}}
= \frac{F}{\sqrt{EG}} :
\]
de enige coëfficiënt $F$ beslist orthogonaliteit van het
parameternet. Voor de bolkaart en de helicoïde, $F = 0$:
meridianen snijden parallelcirkels, en schroeflijnen snijden
de horizontale regellijnen, loodrecht --- daarom stortten
hun oppervlakte-integranden in tot $\sqrt{EG}$. Voor een
grafiekkaart verdwijnt $F = f_xf_y$ alleen waar een partiële
afgeleide dat doet: het coördinaatnet van een gekantelde
grafiek is \emph{niet} orthogonaal, hoewel het $(x,
y)$-net beneden dat is. Wanneer berekeningen op een
oppervlak zwaar lijken, is de eerste zet een kaart met $F =
0$ te zoeken.
\end{example}

\begin{example}[De zadelkaart]\label{ex:b2:surfaces:saddlechart}
Voor het zadel $z = xy$ met kaart $\sigma(u, v) = (u, v,
uv)$:
\[
\sigma_u = (1, 0, v), \qquad \sigma_v = (0, 1, u),
\qquad
E = 1 + v^2, \quad F = uv, \quad G = 1 + u^2 ,
\]
en $EG - F^2 = 1 + u^2 + v^2 > 0$: regulier overal. De twee
coördinaatkrommen door een punt zijn \emph{rechte lijnen}
van $\R^3$ (fixeer $u$ of fixeer $v$: de regellijnen van
het dubbel geregeerde zadel), maar $F \neq 0$ buiten de
assen: regellijnen door een generiek punt zijn niet
orthogonaal. Beide regellijnen liggen in het \omterm{def:b2:surfaces:tangent}{raakvlak}, dat
ze opspannen --- dus snijdt het \omterm{def:b2:surfaces:tangent}{raakvlak} het oppervlak langs
twee hele lijnen, het extreme tegendeel van de bol, waarvan
de raakvlakken slechts in één punt raken. Het teken van het
``tweede-ordecontact'' tussen een oppervlak en haar
raakvlakken is een krommingsverhaal, opgepakt in het volume
van Jaar~3.
\end{example}

\begin{proposition}[Lengte van een kromme op een
oppervlak]\label{prop:b2:surfaces:curvelength}
Als $\gamma(t) = \sigma(u(t), v(t))$, $t \in [a, b]$,
$\mathcal{C}^1$ is, dan
\[
L(\gamma) = \int_a^b
\sqrt{E\,u'^2 + 2F\,u'v' + G\,v'^2}\;\dd t ,
\]
met $E, F, G$ geëvalueerd in $(u(t), v(t))$.
\end{proposition}

\begin{proof}
$\gamma' = u'\sigma_u + v'\sigma_v$ door de kettingregel, dus
$\norm{\gamma'}^2 = I(u', v') = Eu'^2 + 2Fu'v' + Gv'^2$;
integreer $\norm{\gamma'}$ (\cref{def:b2:curves:length}).
\end{proof}

\begin{example}[Waarom vliegtuigen over de pool
vliegen]\label{ex:b2:surfaces:polarroute}
Twee luchthavens liggen op breedte $\varphi_0$ en
overstaande lengten: $A = \sigma(0, \varphi_0)$ en $B =
\sigma(\pi, \varphi_0)$ op de bol van straal $R$. Langs de
parallel ($\varphi \equiv \varphi_0$) is de lengte
$\int_0^\pi R\cos\varphi_0\,\dd\theta = \pi R\cos\varphi_0$.
Langs de route over de pool (omhoog de meridiaan $\theta =
0$, omlaag de meridiaan $\theta = \pi$) is ze $2R(\frac\pi2
- \varphi_0)$. Op breedte $\varphi_0 = \frac\pi3$ (zestig
graden): parallelroute $\pi R/2 \approx 1.571\,R$,
poolroute $\pi R/3 \approx 1.047\,R$ --- een derde korter. In
feite $\pi\cos\varphi_0 \geq \pi - 2\varphi_0$ op
$\intcc0{\pi/2}$ (de functie $\pi\cos\varphi - \pi +
2\varphi$ verdwijnt in beide einden en haar afgeleide $2 -
\pi\sin\varphi$ wisselt één keer van teken, dus is ze eerst
stijgend dan dalend, dus niet-negatief): de poolroute
verliest nooit. De \omterm{def:b2:surfaces:fff}{eerste fundamentaalvorm} maakte van een
navigatievraag twee éénregels-integralen;
\cref{exo:b2:surfaces:6} duwt het idee door naar een echt
minimaliteitsbewijs voor meridianen.
\end{example}

\begin{lemma}[Identiteit van
Lagrange]\label{lem:b2:surfaces:lagrange}
Voor alle $a, b \in \R^3$:
$\norm{a \wedge b}^2 = \norm a^2 \norm b^2 - \langle a,
b\rangle^2$. In het bijzonder
\[
\norm{\sigma_u \wedge \sigma_v} = \sqrt{EG - F^2}.
\]
\end{lemma}

\begin{proof}
Beide zijden blijven onveranderd als we $b$ vervangen door
haar component $b_\perp = b -
\frac{\langle a, b\rangle}{\norm a^2}a$ orthogonaal op $a$
(voor $a \neq 0$; het geval $a = 0$ is triviaal): de
linkerkant omdat $a \wedge a = 0$, de rechterkant door
ontwikkelen van $\norm{b_\perp}^2 = \norm b^2 -
\frac{\langle a,b\rangle^2}{\norm a^2}$ en $\langle a,
b_\perp\rangle = 0$. Dus volstaat de identiteit voor
orthogonale $a, b$, waar ze leest $\norm{a \wedge b} =
\norm a \norm b$: waar, omdat voor orthogonale vectoren het
kruisproduct \omterm{def:b2:nvs:norm}{norm} $\norm a\norm b\,\abs{\sin\frac\pi2}$
heeft. De weergegeven formule is het geval $a = \sigma_u$,
$b = \sigma_v$.
\end{proof}

\begin{remark}
De Lagrange-identiteit zegt dat $EG - F^2 =
\det\begin{psmallmatrix} E & F\\ F & G\end{psmallmatrix}$ de
\emph{\omterm{def:b2:linalg:det}{Gram-determinant}} van $(\sigma_u, \sigma_v)$ is: de
gekadrateerde \omterm{def:b2:surfaces:area}{oppervlakte} van het parallellogram dat ze
opspannen. Regulariteit, positieve definietheid van de
\omterm{def:b2:surfaces:fff}{eerste fundamentaalvorm}, en positiviteit van de
\omterm{def:b2:linalg:det}{Gram-determinant} zijn drie formuleringen van één voorwaarde
--- daarom verdwijnt de oppervlakte-integrand hieronder
nooit op een reguliere kaart.
\end{remark}

\begin{definition}[Oppervlakte]\label{def:b2:surfaces:area}
Zij $\sigma \colon U \to \R^3$ een regulier injectief
$\mathcal{C}^1$-oppervlak en $K \subseteq U$ een \omterm{def:b2:metric:compact}{compact}
domein waarop dubbelintegralen zin hebben
(\cref{ch:b2:multint}). De
\emph{oppervlakte}\index{oppervlakte!van een oppervlak} van
het stuk $\sigma(K)$ is
\[
\mathcal{A}
= \iint_K \norm{\sigma_u \wedge \sigma_v}\,\dd u\,\dd v
= \iint_K \sqrt{EG - F^2}\;\dd u\,\dd v .
\]
\end{definition}

\begin{remark}[Waarom deze formule]
Het rechthoek $[u, u + \dd u] \times [v, v + \dd v]$ wordt,
tot eerste orde, afgebeeld op het parallellogram opgespannen
door $\sigma_u\,\dd u$ en $\sigma_v\,\dd v$, waarvan de
\omterm{def:b2:surfaces:area}{oppervlakte} $\norm{\sigma_u \wedge \sigma_v}\,\dd u\,\dd v$
is: de definitie integreert de lokale
oppervlakte-vervormingsfactor, precies zoals \omterm{def:b2:curves:length}{booglengte} de
lokale snelheid integreert. Consistentie met
\omterm{def:b2:curves:reparam}{parameterwisseling} is \cref{exo:b2:surfaces:8}; consistentie
met de variabelensubstitutieformule voor dubbelintegralen
wordt besproken in \cref{ch:b2:multint}.
\end{remark}

\begin{example}[Twee verschillende grafieken, één
oppervlakte]\label{ex:b2:surfaces:samearea}
Over de eenheidsschijf, vergelijk de kom $z = \frac12(x^2 +
y^2)$ en het zadel $z = xy$. Hun
oppervlakte-integranden (\cref{exo:b2:surfaces:5}) zijn
\[
\sqrt{1 + x^2 + y^2}
\qquad\text{en}\qquad
\sqrt{1 + y^2 + x^2} :
\]
identiek. De twee oppervlakken --- één die in alle
richtingen dezelfde kant op buigt, de andere zadelvormig ---
hebben precies gelijke \omterm{def:b2:surfaces:area}{oppervlakten} over elk domein,
$\frac{2\pi}3(2\sqrt2 - 1)$ over de eenheidsschijf. Het
oppervlakte-element ziet alleen de \emph{lengte} van de
gradiënt, niet de schikking van de buiging; de kom van het
zadel onderscheiden vereist tweede-ordegegevens (de
tekenstructuur vertoond in \cref{fig:b2:surfaces:saddle}),
die geen enkele oppervlaktemeting detecteert. \omterm{def:b2:surfaces:fff}{Eerste
fundamentaalvorm}: metrisch, blind voor vorm; de
vorm-ziende tweede vorm hoort bij Jaar~3.
\end{example}

\begin{example}[Oppervlakte van de
bol]\label{ex:b2:surfaces:spherearea}
Voor de sferische kaart van \cref{ex:b2:surfaces:standard}:
\[
\sigma_\theta = R(-\sin\theta\cos\varphi,\
\cos\theta\cos\varphi,\ 0),
\qquad
\sigma_\varphi = R(-\cos\theta\sin\varphi,\
-\sin\theta\sin\varphi,\ \cos\varphi),
\]
dus $E = R^2\cos^2\varphi$, $F = 0$, $G = R^2$ en $\sqrt{EG -
F^2} = R^2\cos\varphi$. Vandaar
\[
\mathcal{A}
= \int_{-\pi/2}^{\pi/2}\!\!\int_0^{2\pi}
R^2\cos\varphi\;\dd\theta\,\dd\varphi
= 2\pi R^2\,\bigl[\sin\varphi\bigr]_{-\pi/2}^{\pi/2}
= \boxed{4\pi R^2} .
\]
\end{example}

\begin{example}[De kegel, gecontroleerd tegen de
schoolformule]\label{ex:b2:surfaces:conearea}
Voor de kegel $z = \sqrt{x^2 + y^2}$ over de annulus $a
\leq \rho \leq b$ geeft de grafiekformule van
\cref{exo:b2:surfaces:5} $1 + f_x^2 + f_y^2 = 2$
(bereken $f_x = x/\rho$, $f_y = y/\rho$), dus
\[
\mathcal A = \sqrt2\,\pi\,(b^2 - a^2) .
\]
Consistentiecontrole met de schuine-hoogteformule
$\pi\rho\ell$ van \cref{ex:b2:surfaces:revolution}: de volle
kegels van basisstralen $b$ en $a$ hebben laterale
\omterm{def:b2:surfaces:area}{oppervlakten} $\pi b\cdot b\sqrt2$ en $\pi a\cdot a\sqrt2$,
waarvan het verschil precies $\sqrt2\pi(b^2 - a^2)$ is. Twee
kaarten, twee formules, één \omterm{def:b2:surfaces:area}{oppervlakte} --- de invariantie
bewijzen in \cref{exo:b2:surfaces:8}, gezien in het wild.
\end{example}

\begin{remark}[Gezondheidscontroles voor oppervlakten]
Drie instantcontroles vangen de meeste fouten in een
oppervlakteberekening. \emph{Schaling}: dilateren van een
oppervlak met $\lambda$ vermenigvuldigt $E, F, G$ met
$\lambda^2$ en de \omterm{def:b2:surfaces:area}{oppervlakte} met $\lambda^2$ --- een
antwoord waarvan de afhankelijkheid van $R$ niet
kwadratisch is (zoals $4\pi R^2$) is fout. \emph{Positiviteit
van het element}: $\sqrt{EG - F^2}$ moet strikt positief zijn
op het inwendige van de kaart; een verdwijnende waarde
signaleert een kaartdegeneratie, te exciseren zoals bij de
polen van de bol. \emph{Symmetrie}: een berekening over een
\omterm{def:b2:quadratic:adjoint}{symmetrisch} stuk moet consistent zijn met het sommeren van
haar congruente delen --- het halfrond moet $2\pi R^2$
geven.
\end{remark}

\begin{example}[Omwentelingsoppervlak]\label{ex:b2:surfaces:revolution}
Roteer de kromme $z \mapsto (r(z), 0, z)$, $r > 0$ van
klasse $\mathcal{C}^1$, om de $z$-as:
\[
\sigma(\theta, z) = (r(z)\cos\theta,\ r(z)\sin\theta,\ z).
\]
Dan $E = r(z)^2$, $F = 0$, $G = 1 + r'(z)^2$, dus
\[
\mathcal{A} = \int_a^b\!\!\int_0^{2\pi}
r(z)\sqrt{1 + r'(z)^2}\;\dd\theta\,\dd z
= 2\pi\int_a^b r(z)\sqrt{1 + r'(z)^2}\,\dd z ,
\]
de klassieke formule (omtrek $2\pi r$ maal het schuine
lengte-element). Voor de kegel $r(z) = kz$, $z \in [0, h]$:
$\mathcal A = 2\pi k\sqrt{1 + k^2}\,\frac{h^2}{2} = \pi \rho
\ell$ met $\rho = kh$ de basisstraal en $\ell = h\sqrt{1 +
k^2}$ de schuine hoogte --- de schoolformule, nu afgeleid
in plaats van aangenomen.
\end{example}

\begin{example}[De catenoïde]\label{ex:b2:surfaces:catenoid}
Roteer de kettinglijn $r(z) = \cosh z$, $z \in
\intcc{-1}{1}$, om haar as: de resulterende
\emph{catenoïde} heeft, door de omwentelingsformule en $1 +
\sinh^2 z = \cosh^2 z$,
\[
\mathcal A = 2\pi\int_{-1}^1\cosh z\,\sqrt{1 + \sinh^2z}\,
\dd z = 2\pi\int_{-1}^1\cosh^2z\,\dd z
= \pi\bigl[z + \sinh z\cosh z\bigr]_{-1}^{1},
\]
dat is
\[
\mathcal A = 2\pi + \pi\sinh 2 \approx 17.68 .
\]
De schuine factor $\sqrt{1 + r'^2}$ smolt met de straal tot
een perfect kwadraat --- dezelfde identiteit die de
\omterm{def:b2:curves:length}{booglengte} van de kettinglijn elementair maakte in het
krommenhoofdstuk. Dit is geen toeval van algebra: onder alle
omwentelingsoppervlakken die de twee randcirkels overspannen,
minimaliseert de catenoïde de \omterm{def:b2:surfaces:area}{oppervlakte} (het is de vorm
van een zeepvlies tussen twee ringen), en deze
variatie-eigenschap is precies wat $\cosh$ uitkiest; het
volume van Jaar~3 bewijst het met de variatierekening.
\end{example}

\begin{remark}[Klassieke valkuilen]
(i) \emph{Kaartsingulariteiten zijn geen
oppervlaksingulariteiten}: de sferische kaart ontaardt in de
polen ($\cos\varphi = 0$), maar de bol is daar perfect glad
--- een andere kaart (verwissel de rollen van de assen) is
regulier in de polen. Voor je een punt singulier verklaart,
probeer een tweede parametrisatie. (ii) \emph{Regulariteit
van $\sigma$ betreft de parametrisatie, niet het beeld}:
$\sigma(u, v) = (u^3, v, 0)$ is niet-regulier langs $u = 0$
hoewel haar beeld een vlak is. (iii) De
oppervlakteformule vereist dat $\sigma$ \emph{injectief} is
op $K$: een kaart die een stuk tweemaal bedekt telt het
tweemaal ($\theta$ lopend over $\intcc0{4\pi}$ verdubbelt de
\omterm{def:b2:surfaces:area}{oppervlakte} van de bol). (iv) De \omterm{def:b2:surfaces:tangent}{eenheidsnormaal} is op teken
na door het oppervlak gedefinieerd maar wordt door de kaart
\emph{gekozen} (volgorde van $u, v$); uitspraken over
oriëntatie moeten die keuze vastleggen. (v) Ten slotte is
$EG - F^2 > 0$ geen extra hypothese: het is precies
regulariteit, door de Lagrange-identiteit --- als ze ergens
verdwijnt, is het probleem de kaart, en geen oppervlakte-
of raakvlakformule is daar van toepassing.
\end{remark}

\begin{remark}[Perspectieven binnen dit volume]
Voorwaartse schakels vanaf hier. Het
oppervlakte-element $\sqrt{EG - F^2}\,\dd u\,\dd v$ is een
tweedimensionale Jacobiaan in vermomming, en
\cref{ch:b2:multint} maakt de analogie exact met de
variabelensubstitutiestelling --- de oppervlakintegralen
daar zijn de \omterm{def:b2:surfaces:area}{oppervlakten} van dit hoofdstuk met een
integrand aan boord. De \omterm{def:b2:surfaces:fff}{eerste fundamentaalvorm} is een veld
van positieve \omterm{def:b2:quadratic:def}{kwadratische vormen}, puntsgewijs behandeld met
de tools van \cref{ch:b2:quadratic}, waarvan de spectrale
stelling ook de weekendclassificatie van \omterm{pb:b2:surfaces:1}{kwadrieken} van dit
hoofdstuk voedt. En de normaallijn stuurt extremumproblemen
op constraintverzamelingen (\cref{ex:b2:surfaces:closest}),
de meetkundige kiem van de Lagrange-multipliermethode
geschetst met \cref{thm:b2:diffcalc:extrema}.
\end{remark}

\begin{remark}[Waar dit wordt gebruikt]
Het oppervlakte-element $\norm{\sigma_u\wedge\sigma_v}\,\dd
u\,\dd v$ is de oppervlakintegraalmaat van
\cref{ch:b2:multint}, waar ze de formule van Green
ontmoet; de \omterm{def:b2:surfaces:fff}{eerste fundamentaalvorm} is het prototype van een
veld van \omterm{def:b2:quadratic:def}{kwadratische vormen}, puntsgewijs bestudeerd met de
tools van \cref{ch:b2:quadratic}; en het weekendprobleem van
dit hoofdstuk classificeert alle kwadriekoppervlakken met de
spectrale stelling. Het volume van Jaar~3 keert terug naar
oppervlakken met differentiaalvormen en de
divergentstelling, en intrinsieke \omterm{thm:b2:curves:frenet2d}{kromming} --- wat $E, F, G$
weten over buiging --- is de toegangspoort tot de
differentiaalmeetkunde proper.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b2:surfaces:1}
Toon dat de raakvlakken van de kegel $z = \sqrt{x^2 + y^2}$
(min haar top) allemaal door de top gaan.
\emph{(Parametriseer door $\sigma(\theta, r) = (r\cos\theta,
r\sin\theta, r)$, $r > 0$.)}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:surfaces:2}
Vind het \omterm{def:b2:surfaces:tangent}{raakvlak} van de ellipsoïde $\frac{x^2}{a^2} +
\frac{y^2}{b^2} + \frac{z^2}{c^2} = 1$ in een punt $(x_0,
y_0, z_0)$ van het oppervlak.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:surfaces:3}
Bereken $E, F, G$ voor de helicoïde $\sigma(u, v) =
(v\cos u,\ v\sin u,\ au)$, $a > 0$, en de \omterm{def:b2:surfaces:area}{oppervlakte} van
het stuk $0 \leq u \leq 2\pi$, $0 \leq v \leq 1$, als
integraal (evalueer haar met $\int\sqrt{v^2 + a^2}\,\dd v =
\frac12\bigl(v\sqrt{v^2+a^2} + a^2\ln(v + \sqrt{v^2 +
a^2})\bigr) + C$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:surfaces:4}
(Torus) Parametriseer de torus verkregen door de cirkel van
centrum $(R, 0, 0)$ en straal $r < R$ in het $xz$-vlak om de
$z$-as te roteren:
\[
\sigma(\theta, \psi) = \bigl((R + r\cos\psi)\cos\theta,\
(R + r\cos\psi)\sin\theta,\ r\sin\psi\bigr).
\]
Bereken $E, F, G$, controleer regulariteit, en toon dat de
\omterm{def:b2:surfaces:area}{oppervlakte} $4\pi^2 R r$ is (Pappus: gemiddelde omtrek $2\pi
R$ maal cirkellengte $2\pi r$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:surfaces:5}
Toon dat de \omterm{def:b2:surfaces:area}{oppervlakte} van de grafiek van $f \in
\mathcal{C}^1(K)$ gelijk is aan $\iint_K \sqrt{1 + f_x^2 +
f_y^2}\,\dd x\,\dd y$, en bereken haar voor het
paraboloïdestuk $z = \frac12(x^2 + y^2)$ over de schijf
$x^2 + y^2 \leq 1$ (poolcoördinaten,
\cref{ch:b2:multint}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:surfaces:6}
Een getekende kromme $\gamma(t) = \sigma(u(t), v(t))$ op de
bol van straal $R$ (sferische kaart) heeft $u = \theta(t)$,
$v = \varphi(t)$. Schrijf haar lengte als integraal in
$\theta, \varphi$ en bewijs dat onder krommen die twee
punten van dezelfde meridiaan $\theta = \theta_0$
verbinden, de meridiaanboog de kortste is. \emph{(Grenz de
integrand van onder door $R\abs{\varphi'}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:surfaces:7}
(Normaallijnen van een bol) Zij $S$ een regulier
niveauoppervlak $\{f = c\}$, \omterm{def:b2:metric:connected}{samenhangend}, waarvan alle
normaallijnen door een vast punt $\Omega$ gaan. Toon dat $S$
bevat is in een bol gecentreerd in $\Omega$. \emph{(Toon dat
$\norm{M - \Omega}^2$ afgeleide nul heeft langs elke op $S$
getekende kromme.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:surfaces:8}
(\omterm{def:b2:surfaces:area}{Oppervlakte} is meetkundig) Zij $\Phi \colon U' \to U$ een
$\mathcal{C}^1$-diffeomorfisme tussen open verzamelingen van
$\R^2$ en $\tilde\sigma = \sigma \circ \Phi$. Toon dat
\[
\norm{\tilde\sigma_{u'} \wedge \tilde\sigma_{v'}}
= \abs{\det J_\Phi}\,
\norm{(\sigma_u \wedge \sigma_v)\circ\Phi} ,
\]
en leid, met de variabelensubstitutieformule van
\cref{ch:b2:multint}, af dat de \omterm{def:b2:surfaces:area}{oppervlakte} van
\cref{def:b2:surfaces:area} niet van de gekozen reguliere
parametrisatie afhangt.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:surfaces:9}
(De hoedendoosstelling van Archimedes) Op de bol van straal
$R$ heeft de \emph{zone} tussen de breedten met $z_1 \leq z
\leq z_2$ ($-R \leq z_1 < z_2 \leq R$) \omterm{def:b2:surfaces:area}{oppervlakte} $2\pi
R\,(z_2 - z_1)$: bewijs het met de sferische kaart, en
concludeer dat de \omterm{def:b2:surfaces:area}{oppervlakte} van een zone alleen van haar
hoogte afhangt --- een sinaasappel in schijven van gelijke
dikte snijden geeft gelijke hoeveelheden schil.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:surfaces:10}
Toon dat elke normaallijn van een omwentelingsoppervlak
$\sigma(\theta, z) = (r(z)\cos\theta,\ r(z)\sin\theta,\ z)$
($r > 0$ van klasse $\mathcal C^1$) de omwentelingsas
snijdt, en lokaliseer het snijpunt.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:surfaces:11}
(De cilinder afrollen) De kaart $\sigma(u, v) = (\cos u,\
\sin u,\ v)$ van de eenheidscilinder heeft $E = G = 1$, $F =
0$: verifieer dit, en verklaar waarom elke getekende kromme
$t \mapsto \sigma(u(t), v(t))$ dezelfde lengte heeft als de
vlakke kromme $t \mapsto (u(t), v(t))$. Leid af dat de
schroeflijn van $(1, 0, 0)$ naar $(1, 0, 2\pi c)$ met één
omwenteling lengte $2\pi\sqrt{1 + c^2}$ heeft, en dat geen
getekende kromme met dezelfde eindpunten en één volle
omwenteling korter is.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:surfaces:12}
Zij $S = \{f = c\}$ een \omterm{def:b2:metric:compact}{compact} regulier niveauoppervlak en
$M_0 \in S$ een punt op maximale afstand van de oorsprong.
Toon dat $\nabla f(M_0)$ collineair is met $\vect{OM_0}$ ---
de normaal in het verste punt is radiaal. Pas toe op de
ellipsoïde $\frac{x^2}{a^2} + \frac{y^2}{b^2} +
\frac{z^2}{c^2} = 1$ ($a > b > c > 0$): vind alle punten
waar de normaal radiaal is, en identificeer de verste.
\end{exercise}

\section{Probleem: de classificatie van kwadrieken van
$\R^3$}

\begin{problem}[{Weekendprobleem --- elk kwadriekoppervlak,
gesorteerd door de spectrale stelling}]\label{pb:b2:surfaces:1}
Een \emph{kwadriek}\index{kwadriek} is de nulverzameling in
$\R^3$ van een tweedegraadspolynoom
\[
q(X) = X^{\mathsf T}\!AX + 2\,\langle b, X\rangle + c,
\qquad A \in \mathcal S_3(\R),\ A \neq 0,\ b \in \R^3,\
c \in \R .
\]
De oppervlakken van de figuren van dit hoofdstuk --- bollen,
ellipsoïden, zadels, kegels, cilinders --- zijn allemaal
\omterm{pb:b2:surfaces:1}{kwadrieken}. Dit probleem classificeert ze \emph{alle}: de
spectrale stelling (\cref{thm:b2:quadratic:spectral})
recht de kwadratische deel, affiene translaties
(\cref{ch:b2:affine}) absorberen het lineaire deel, en wat
overblijft is een korte, complete lijst van normaalvormen.

\medskip
\noindent\textbf{Deel I --- De reductiemachine.}
\begin{enumerate}
  \item Zij $X = PY + t$ met $P \in O(3)$ en $t \in \R^3$
        (een starre coördinatenwisseling). Toon dat $q(PY +
        t) = Y^{\mathsf T}\!A'Y + 2\langle b', Y\rangle +
        c'$ met
        \[
        A' = P^{\mathsf T}\!AP, \qquad
        b' = P^{\mathsf T}(At + b), \qquad
        c' = q(t) .
        \]
        Leid af dat het spectrum van $A$ (dus haar rang en
        signatuur) een starre invariant van de vergelijking
        is, en verklaar waarom de vergelijking van een
        gegeven \omterm{pb:b2:surfaces:1}{kwadriek} alleen op een niet-nulle
        scalaire factor na bepaald is.
  \item Met de spectrale stelling, toon dat na een rotatie
        de vergelijking $\sum_i \lambda_i y_i^2 +
        2\sum_i\beta_iy_i + c = 0$ wordt met $\lambda_1,
        \lambda_2, \lambda_3$ de \omterm{def:b2:reduction:eigen}{eigenwaarden} van $A$.
  \item Voor elke $i$ met $\lambda_i \neq 0$, absorbeer
        $\beta_iy_i$ door een translatie ($y_i \mapsto y_i -
        \beta_i/\lambda_i$). Schrijf de gereduceerde
        vergelijking wanneer $\operatorname{rank} A = r$:
        $\sum_{i\leq r}\lambda_i z_i^2 + 2\sum_{i >
        r}\beta_iz_i + c'' = 0$.
  \item Een \emph{centrum} van de \omterm{pb:b2:surfaces:1}{kwadriek} van vergelijking
        $q = 0$ is een punt $\Omega$ met $q(2\Omega - X) =
        q(X)$ voor alle $X$: de puntsymmetrie in $\Omega$
        behoudt de vergelijking, dus het oppervlak. Toon dat
        $q(2\Omega - X) - q(X) = -4\langle A\Omega + b,
        X\rangle + 4\langle A\Omega + b, \Omega\rangle$, en
        leid af: de centra zijn precies de oplossingen van
        $A\Omega = -b$; ze bestaan dan en slechts dan als
        $b \in \operatorname{im}A$, en het centrum is uniek
        dan en slechts dan als $A$ inverteerbaar is.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Centrale \omterm{pb:b2:surfaces:1}{kwadrieken}
($\operatorname{rank}A = 3$).} Hier is de gereduceerde
vergelijking $\lambda_1z_1^2 + \lambda_2z_2^2 +
\lambda_3z_3^2 = \delta$.
\begin{enumerate}[resume]
  \item Vermenigvuldigend met $-1$ indien nodig, neem aan
        dat ten minste twee $\lambda_i > 0$ zijn. Enumereer
        de mogelijkheden: signatuur $(3, 0)$ met $\delta >
        0$, $= 0$, $< 0$, en signatuur $(2, 1)$ met $\delta
        > 0$, $= 0$, $< 0$; noem de zes resulterende
        verzamelingen (ellipsoïde, punt, lege verzameling,
        éénbladige hyperboloïde, kegel, tweeladige
        hyperboloïde) en breng elk in haar euclidische
        normaalvorm ($\frac{x^2}{a^2} + \frac{y^2}{b^2} +
        \frac{z^2}{c^2} = 1$, enz.).
  \item Classificeer $x^2 + y^2 + z^2 + 4xy + 4yz + 4zx =
        1$: toon $A = 2J - I$ met $J$ de all-ones-matrix,
        bereken het spectrum $\{5, -1, -1\}$, en
        identificeer een tweeladige hyperboloïde van
        omwenteling om de as $\R(1,1,1)$.
  \item (Regellijnen) Voor de éénbladige hyperboloïde
        $\frac{x^2}{a^2} + \frac{y^2}{b^2} - \frac{z^2}{c^2}
        = 1$, factoriseer
        \[
        \Bigl(\frac xa - \frac zc\Bigr)
        \Bigl(\frac xa + \frac zc\Bigr)
        = \Bigl(1 - \frac yb\Bigr)\Bigl(1 + \frac yb\Bigr)
        \]
        en produceer twee éénparameterfamilies van rechte
        lijnen liggend op het oppervlak.
  \item Toon dat door elk punt van de éénbladige
        hyperboloïde precies één lijn van elke familie gaat:
        het oppervlak is \emph{dubbel geregeerd}.
  \item De \emph{asymptotische kegel} van de éénbladige
        hyperboloïde is $C : \frac{x^2}{a^2} +
        \frac{y^2}{b^2} - \frac{z^2}{c^2} = 0$. Met $\rho^2
        = \frac{x^2}{a^2} + \frac{y^2}{b^2}$, toon dat elk
        punt van de hyperboloïde op afstand ten hoogste
        $c\bigl(\rho - \sqrt{\rho^2 - 1}\bigr) =
        \frac{c}{\rho + \sqrt{\rho^2 - 1}}$ van $C$ ligt,
        zodat het oppervlak haar kegel op oneindig omhelst.
        Wat zijn de sneden van de hyperboloïde met de
        vlakken $x = \pm a$?
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Rang $2$ en rang $1$:
paraboloïden, cilinders, vlakken.}
\begin{enumerate}[resume]
  \item Stel $\operatorname{rank}A = 2$, zeg $\lambda_1,
        \lambda_2 \neq 0 = \lambda_3$. Startend van vraag
        3, splits in twee gevallen naargelang $\beta_3 \neq
        0$ (geen centrum, door vraag 4) of $\beta_3 = 0$
        (een lijn van centra), en reduceer tot
        \[
        \lambda_1z_1^2 + \lambda_2z_2^2 + 2\beta_3z_3 = 0
        \qquad\text{of}\qquad
        \lambda_1z_1^2 + \lambda_2z_2^2 + c'' = 0 :
        \]
        elliptische/hyperbolische paraboloïden in het
        eerste geval, cilinders over centrale kegelsneden
        (of paren snijdende vlakken, een lijn, de lege
        verzameling) in het tweede.
  \item Toon dat het zadel $z = xy$ een hyperbolische
        paraboloïde is: roteer over $\pi/4$ in het
        $xy$-vlak om $z = \tfrac12(u^2 - v^2)$ te bereiken,
        het oppervlak van \cref{fig:b2:surfaces:saddle} op
        schaal na.
  \item Toon dat het zadel $z = xy$ de twee lijnfamilies
        $\{x = x_0,\ z = x_0y\}$ en $\{y = y_0,\ z =
        xy_0\}$ draagt, met precies één lijn van elke door
        elk punt: de tweede dubbel geregeerde \omterm{pb:b2:surfaces:1}{kwadriek}.
  \item Classificeer $x^2 + y^2 - 2x + 4y + 3 = 0$ in
        $\R^3$ (maak de kwadraten af; identificeer een
        rechte circulaire cilinder, en geef haar as en
        straal).
  \item Zij nu $\operatorname{rank}A = 1$, zeg $\lambda_1
        \neq 0 = \lambda_2 = \lambda_3$. Roteren binnen het
        kernvlak en transleren, reduceer tot
        \[
        \lambda_1z_1^2 + 2\beta z_2 = 0 \quad (\beta \neq 0)
        \qquad\text{of}\qquad
        \lambda_1z_1^2 + c'' = 0 :
        \]
        een parabolische cilinder, of een paar evenwijdige
        vlakken, een dubbel vlak, of de lege verzameling.
        Classificeer $(x + y)^2 = z$ \omterm{def:b2:metric:complete}{volledig}
        (normaalvorm, as van translatie-invariantie).
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De classificatiestelling.}
\begin{enumerate}[resume]
  \item Assembleer Delen I--III tot een stelling:
        \emph{elke \omterm{pb:b2:surfaces:1}{kwadriek} van $\R^3$ wordt door een
        starre beweging op precies één normaalvorm
        afgebeeld}. Lijst de zeventien affiene types (tel
        de lege varianten en ontaarde verzamelingen), en
        licht de negen kwadriek-\emph{oppervlakken} eruit:
        ellipsoïde, één- en tweeladige hyperboloïden,
        kegel, elliptische en hyperbolische paraboloïden,
        elliptische, hyperbolische en parabolische
        cilinders.
  \item Schrijf het classificatie-\emph{algoritme}: gegeven
        $(A, b, c)$, welke grootheden bereken je, in welke
        volgorde, en welke tak beslist welk type?
        Rechtvaardig dat elke stap effectief is
        (\omterm{def:b2:reduction:eigen}{eigenwaarden} van een symmetrische
        $3\times3$-matrix, rang, oplosbaarheid van
        $A\Omega = -b$).
  \item Draai het algoritme op $x^2 + y^2 + z^2 - 2xy -
        2yz - 2zx = 1$: toon dat $A = 2I - J$ spectrum
        $\{2, 2, -1\}$ heeft en concludeer: een éénbladige
        hyperboloïde van omwenteling om $\R(1,1,1)$.
  \item Draai het op $x^2 + y^2 - z^2 - 2x + 4y + 2z + 4 =
        0$: vind het centrum en identificeer de \omterm{pb:b2:surfaces:1}{kwadriek}.
  \item Draai het op $x^2 + 4xy + y^2 = 2z$: diagonaliseer
        het $xy$-blok ($u = \frac{x+y}{\sqrt2}$, $v =
        \frac{x-y}{\sqrt2}$) en identificeer de \omterm{pb:b2:surfaces:1}{kwadriek}.
  \item Euclidisch versus affien. Toon dat twee centrale
        \omterm{pb:b2:surfaces:1}{kwadrieken} in normaalvorm \emph{star} equivalent
        zijn dan en slechts dan als ze dezelfde
        coëfficiëntenlijsten hebben (op permutatie en een
        gemeenschappelijke positieve scalar op de
        vergelijking na), terwijl \emph{affien} alleen de
        signatuurgegevens overleven: elke ellipsoïde is een
        affien beeld van de ronde bol. Welke stelling
        garandeert dat de signatuur onderweg niet kan
        veranderen (\cref{thm:b2:quadratic:sylvester})?
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Dividenden.}
\begin{enumerate}[resume]
  \item Toon dat elke snede van een \omterm{pb:b2:surfaces:1}{kwadriek} met een affien
        vlak een kegelsnede is (mogelijk ontaard) van dat
        vlak. Identificeer de sneden van het zadel $z = xy$
        met de vlakken $z = c$ ($c \neq 0$ en $c = 0$).
  \item Welke kwadriekoppervlakken bevatten rechte lijnen?
        Toon dat de ellipsoïde, de tweeladige hyperboloïde
        en de elliptische paraboloïde er geen bevatten
        \emph{(beperk $q$ tot een lijn en gebruik de
        Cauchy--Schwarz-ongelijkheid voor het
        tweeladige geval)}; dat kegel en cilinders door
        één familie geregeerd worden; en concludeer dat de
        dubbel geregeerde kwadriekoppervlakken precies de
        éénbladige hyperboloïde en de hyperbolische
        paraboloïde zijn.
  \item Wanneer alleen het \emph{affiene} type gewenst is,
        is de \omterm{thm:b2:quadratic:gauss}{Gauss-reductie} (\cref{thm:b2:quadratic:gauss})
        goedkoper dan diagonaliseren. Doe vraag 17 opnieuw
        met het Gauss-algoritme en controleer de signatuur
        $(2, 1)$; welke euclidische informatie verliest
        Gauss?
  \item Alle \omterm{def:b2:reduction:eigen}{eigenwaarden} van een symmetrische matrix zijn
        reëel; toon dat bijgevolg de \emph{tekens} van de
        \omterm{def:b2:reduction:eigen}{eigenwaarden} van $A$ van het karakteristieke
        polynoom kunnen worden afgelezen met de
        tekenregel van Descartes, en verifieer het op
        vraag 6: $\chi_A(\lambda) = \lambda^3 - 3\lambda^2
        - 9\lambda - 5$ heeft precies één tekenwisseling,
        dus signatuur $(1, 2)$.
  \item Synthese. Vat het algoritme in enkele regels samen;
        formuleer de exacte rol gespeeld door (i) de
        spectrale stelling, (ii) de centrumvergelijking
        $A\Omega = -b$, (iii) de inertiestelling van
        Sylvester, (iv) de \omterm{thm:b2:quadratic:gauss}{Gauss-reductie}. Wat levert
        dezelfde machine in $\R^2$, en wat verandert in
        $\R^n$?
\end{enumerate}
\end{problem}
