\chapter{Functies van een reële veranderlijke}\label{ch:b2:realfun}

Voor de analyse overgaat naar functies van functies
(\cref{ch:b2:funcseq}) loont het de eenvariabele landschap fijner te
kennen dan Jaar~1 eiste: hoe discontinu een monotone functie kan zijn,
hoe regulier een convexe functie moet zijn, en welke speciale
eigenschappen afgeleiden genieten (Darboux). Deze structurele
resultaten zijn kort, scherp, en geliefd bij examinatoren.

\section{Monotone functies}

\begin{theorem}[Regulariteit van monotone functies]\label{thm:b2:realfun:monotone}
Zij $f \colon I \to \R$ stijgend op een interval.
\begin{enumerate}
  \item In elk inwendig punt $a$ bestaan de eenzijdige limieten:
        \[
        f(a^-) = \sup_{x < a} f(x) \;\leq\; f(a) \;\leq\;
        f(a^+) = \inf_{x > a} f(x) ;
        \]
        elke discontinuïteit is een \emph{sprong}.
  \item De verzameling discontinuïteiten van $f$ is ten hoogste
        \omterm{def:b2:structures:countable}{aftelbaar}.\index{monotone functie!discontinuïteiten}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) De verzameling $\{f(x) : x < a\}$ is niet-leeg, van boven
begrensd door $f(a)$: haar supremum $s$ voldoet aan $f(x) \to s$
als $x \to a^-$ (gegeven $\varepsilon$, enige $f(x_0) > s -
\varepsilon$, en monotonie vangt $f(x) \in \intoc{s -
\varepsilon}{s}$ voor $x \in \intoo{x_0}{a}$). Symmetrisch rechts.

(2) Koppel aan elke discontinuïteit $a$ het niet-lege open interval
$J_a = \intoo{f(a^-)}{f(a^+)}$ (een echte sprong). Voor $a < b$
discontinuïteiten zijn $J_a$ en $J_b$ disjunct: $f(a^+) \leq f(c)
\leq f(b^-)$ voor elke $c$ ertussen. Elk $J_a$ bevat een rationaal;
onderscheiden discontinuïteiten krijgen onderscheiden rationalen:
een injectie van de discontinuïteitsverzameling in $\Q$, die
\omterm{def:b2:structures:countable}{aftelbaar} is (\cref{prop:b2:structures:countablestable}).
\end{proof}

\begin{example}
De grens is scherp: fixeer een enumeratie $(r_n)$ van $\Q \cap
\intoo{0}{1}$ en stel $f(x) = \sum_{n : r_n \leq x} 2^{-n}$
(een definitie via \omterm{def:b2:series:summable}{sommeerbare families},
\cref{def:b2:series:summable}). Dan is $f$ stijgend op
$\intcc{0}{1}$ en discontinu precies in elke rationaal van
$\intoo{0}{1}$ (sprong $2^{-n}$ in $r_n$): een monotone functie
\emph{kan} discontinu zijn op een dichte \omterm{def:b2:structures:countable}{aftelbare verzameling}.
\end{example}

\begin{example}[De sprongen kunnen de stijging niet overtreffen]\label{ex:b2:realfun:jumpbudget}
Voor $f$ stijgend op $\intcc{a}{b}$ hebben de sprongen een budget:
als $a < c_1 < \dots < c_m < b$ discontinuïteiten zijn met sprongen
$s_i = f(c_i^+) - f(c_i^-) > 0$, kies dan ineengevlochten punten
$a < c_1 < t_1 < c_2 < \dots$ en gebruik monotonie op elk stuk,
\[
\sum_{i=1}^{m} s_i \;\leq\; f(b) - f(a) :
\]
de totale stijging begrensd de totale sprong. Gevolg: voor elke
$k$ hebben ten hoogste $k\,\bigl(f(b) - f(a)\bigr)$
discontinuïteiten sprong $\geq \frac1k$ --- een kwantitatieve
verfijning van \cref{thm:b2:realfun:monotone}~(2), want de
discontinuïteitsverzameling is de aftelbare unie over $k$ van
deze eindige verzamelingen. Op de rationale-sprongfunctie hierboven
wordt het budget exact besteed: de sprongen $2^{-n}$ sommen tot
$1 = f(1^+) - f(0^-)$ in de voor de hand liggende uitgebreide zin.
Monotone functies mogen dicht springen, maar slechts op een strikte
toelage.
\end{example}

\section{Convexe functies}

\begin{lemma}[Hellingongelijkheid]\label{lem:b2:realfun:slopes}
Zij $f$ convex op $I$ en $x < y < z$ in $I$. Dan
\[
\frac{f(y) - f(x)}{y - x}
\;\leq\; \frac{f(z) - f(x)}{z - x}
\;\leq\; \frac{f(z) - f(y)}{z - y} :
\]
hellingen van koorden stijgen in beide eindpunten.\index{convexe
functie!hellingongelijkheid}
\end{lemma}

\begin{proof}
Schrijf $y = \frac{z - y}{z - x}\,x + \frac{y - x}{z - x}\,z$: een
convexe combinatie, want de twee coëfficiënten zijn positief en
sommeren tot $1$. Convexiteit geeft
\[
f(y) \;\leq\; \frac{z-y}{z-x}\,f(x) + \frac{y-x}{z-x}\,f(z).
\]
Voor de linkerongelijkheid, trek $f(x)$ van beide kanten af, met
$\frac{z-y}{z-x} - 1 = -\frac{y-x}{z-x}$:
\[
f(y) - f(x) \leq \frac{y - x}{z - x}\bigl(f(z) - f(x)\bigr),
\]
en deel door $y - x > 0$. Voor de rechterongelijkheid, trek
in plaats daarvan af van $f(z)$:
\[
f(z) - f(y) \geq f(z) - \frac{z-y}{z-x}f(x) -
\frac{y-x}{z-x}f(z)
= \frac{z - y}{z - x}\bigl(f(z) - f(x)\bigr),
\]
en deel door $z - y > 0$. Beide getoonde stappen zijn dezelfde
barycentrische identiteit, gelezen tegen een ander eindpunt.
\end{proof}

\begin{theorem}[Regulariteit van convexe functies]\label{thm:b2:realfun:convexreg}
Zij $f$ convex op een interval $I$.
\begin{enumerate}
  \item In elk inwendig punt heeft $f$ eindige eenzijdige
        afgeleiden $f'_g \leq f'_d$; beide zijn stijgende
        functies van het punt; in het bijzonder is $f$ continu
        op het inwendige van $I$ (maar mogelijk niet in de
        eindpunten).
  \item $f$ ligt boven elke \emph{steunlijn}: voor $a$ inwendig
        en enige $m \in \intcc{f'_g(a)}{f'_d(a)}$,
        \[
        f(x) \geq f(a) + m(x - a) \qquad (x \in I).
        \]
  \item (Jensen, gewogen) Voor $x_i \in I$ en gewichten
        $\lambda_i \geq 0$, $\sum\lambda_i = 1$:
        \[
        f\Bigl(\sum_i \lambda_i x_i\Bigr) \leq \sum_i \lambda_i
        f(x_i) .\index{ongelijkheid van Jensen}
        \]
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Fixeer $a$ inwendig. Door \cref{lem:b2:realfun:slopes} is de
helling $\tau(h) = \frac{f(a + h) - f(a)}{h}$ een stijgende
functie van $h$ (aan beide kanten, en $\tau(h_-) \leq
\tau(h_+)$ voor $h_- < 0 < h_+$). Dus heeft $\tau$ een eindige
limiet als $h \to 0^-$ (stijgend, van boven begrensd door enige
rechterhelling) --- dit is $f'_g(a)$ --- en als $h \to 0^+$
($f'_d(a)$), met $f'_g(a) \leq f'_d(a)$. Eindige eenzijdige
afgeleiden dwingen continuïteit in $a$. Monotonie in het punt:
voor $a < b$ inwendig, $f'_d(a) \leq \frac{f(b) - f(a)}{b - a}
\leq f'_g(b)$, opnieuw door de hellingongelijkheid.

(2) Voor $x > a$: $\frac{f(x) - f(a)}{x - a} \geq f'_d(a) \geq m$;
voor $x < a$: $\frac{f(a) - f(x)}{a - x} \leq f'_g(a) \leq m$.
Beide herschikken tot de bewering.

(3) Inductie op het aantal punten precies als in het boekdeel van
Jaar~1 (het tweepuntsgeval is de definitie) --- of in één slag:
pas (2) toe in $a = \sum\lambda_i x_i$ en gemiddeld de
steunlijnongelijkheden in de punten $x_i$ met gewichten
$\lambda_i$: $\sum_i \lambda_i f(x_i) \geq f(a) +
m\sum_i\lambda_i(x_i - a) = f(a)$.
\end{proof}

\begin{example}[Discontinuïteit in een eindpunt]
Op $\intcc{0}{1}$ is de functie $f(0) = 1$, $f(x) = 0$ voor $x >
0$ convex maar discontinu in het eindpunt $0$: bewering (1) is
scherp.
\end{example}

\begin{example}[Hoeken en de bundel steunlijnen]\label{ex:b2:realfun:corner}
Voor $f(x) = \abs x$ in $a = 0$: de eenzijdige afgeleiden zijn
$f'_g(0) = -1$ en $f'_d(0) = +1$, en
\cref{thm:b2:realfun:convexreg}~(2) reikt een steunlijn uit voor
\emph{elke} helling $m \in \intcc{-1}{1}$:
\[
\abs x \geq m\,x \qquad (x \in \R,\ -1 \leq m \leq 1),
\]
elk een gelijkheid precies op een halfrechte of in $0$. Een
convexe functie is differentieerbaar in $a$ precies wanneer de
bundel tot één lijn instort ($f'_g(a) = f'_d(a)$); hoeken dragen
een interval van raaklijnen. Deze bundel is de
eindig-dimensionale kiem van het \emph{subdifferentiaal} van
convexe optimalisatie --- en de reden dat convexe functies zo
robust zijn: zelfs waar de afgeleide faalt, overleeft de
steungeometrie, en dat is alles wat Jensens bewijs gebruikte.
\end{example}

\begin{example}[Machtsgemiddelde-ongelijkheid]\label{ex:b2:realfun:powermeans}
Voor $0 < p < q$ en positieve $x_i$ met gewichten $\lambda_i$ die
tot $1$ sommen, Jensen toepassen op de convexe $t \mapsto
t^{q/p}$ in de punten $x_i^p$:
\[
\Bigl(\sum \lambda_i x_i^{p}\Bigr)^{1/p}
\leq \Bigl(\sum \lambda_i x_i^{q}\Bigr)^{1/q} :
\]
machtsgemiddelden stijgen met de exponent --- met AM--QM, en, in
de limiet $p \to 0$ (\cref{exo:b2:realfun:6}), opnieuw de
AM--GM-ongelijkheid.
\end{example}

\begin{example}[Maximale entropie]\label{ex:b2:realfun:entropy}
Voor een waarschijnlijkheidsvector $(p_1, \dots, p_n)$
(positief, sommerend tot $1$) voldoet de entropie $H(p) =
-\sum_i p_i\ln p_i$ aan
\[
H(p) \leq \ln n ,
\qquad\text{met gelijkheid d.e.s.d.a.\ } p_i = \frac1n
\text{ voor alle } i .
\]
Bewijs via Jensen (\cref{thm:b2:realfun:convexreg}~(3))
toegepast op de \emph{concave} $\ln$ met gewichten $p_i$ in de
punten $\frac{1}{p_i}$:
\[
H(p) = \sum_i p_i\ln\frac{1}{p_i}
\leq \ln\Bigl(\sum_i p_i\,\frac1{p_i}\Bigr) = \ln n ,
\]
gelijkheid dwingt alle punten $\frac1{p_i}$ gelijk (strikte
concaviteit), d.w.z.\ $p$ uniform. Equivalent is dit
\cref{exo:b2:realfun:7} met $q$ uniform. Onzekerheid wordt
gemaximaliseerd door gelijkmatig gespreide onwetendheid --- het
variatieprincipe achter codering, statistische mechanica, en de
entropieverschijningen van \cref{ch:b2:randomvar}.
\end{example}

\begin{method}[De convexe functie achter een ongelijkheid vinden]\label{met:b2:realfun:spotting}
De meeste klassieke ongelijkheden zijn Jensen in kostuum; om er
één uit te kleden: (1) normaliseer zodat een \emph{gewogen
gemiddelde} verschijnt (gewichten positief, sommerend tot $1$
--- deel door een totale massa indien nodig); (2) kijk welke
functie binnen versus buiten het gemiddelde wordt toegepast: de
bewering ``$f(\text{gemiddelde}) \leq$ gemiddelde van $f$'' noemt
de convexe $f$; (3) certificeer convexiteit via de tweede
afgeleide, en behandel gelijkheid via striktheid; (4) als geen
gemiddelde zichtbaar is, neem eerst logaritmen --- producten en
machten worden gemiddelden, en de concaviteit van $\ln$ draagt
AM--GM, Young en hun verwanten (de weekendopgave van dit
hoofdstuk voert stappen 1--4 op elk van hen uit). Als zelfs
logaritmen geen gemiddelde onthullen, probeer de ongelijkheid te
lezen als monotonie van hellingen (\cref{lem:b2:realfun:slopes})
--- superadditiviteitsuitspraken zoals \cref{exo:b2:realfun:9}
wonen daar.
\end{method}

\begin{remark}[Veelgemaakte valkuilen]\label{rem:b2:realfun:pitfalls}
(i) Convexiteit blijft niet bewaard onder producten: $x$ en $(x
- 1)^2$ zijn convex op $\intcc{0}{2}$, maar hun product $x(x-1)^2$
heeft tweede afgeleide $6x - 4$, negatief op
$\intco{0}{\frac23}$ --- niet convex; noch blijft convexiteit
bewaard onder samenstelling zonder monotonie
(\cref{exo:b2:realfun:10}). (ii) Jensen keert om voor concave
functies: de helft van de klassieke ongelijkheden is de concave
$\ln$-versie; de convexe vorm op $\ln$ toepassen is de snelste
manier om AM--GM \emph{achterstevoren} te bewijzen. (iii)
Midpuntconvexiteit alleen impliceert geen convexiteit ---
continuïteit (of louter begrensdheid) is nodig
(\cref{exo:b2:realfun:8}); de pathologische tegenvoorbeelden
leven voorbij de axioma's van dit boek. (iv) Een convexe functie
op een \emph{open} interval is continu, zelfs lokaal \omterm{def:b2:metric:continuity}{Lipschitz}
(\cref{exo:b2:realfun:12}); in eindpunten is niets gratis. (v)
Afgeleiden gehoorzamen Darboux maar hoeven niet continu te zijn
(\cref{ex:b2:realfun:oscillation}): ``$f'$ heeft geen sprongen''
betekent nooit ``$f'$ is continu''.
\end{remark}

\section{De Darboux-eigenschap}

\begin{theorem}[Darboux]\label{thm:b2:realfun:darboux}
Zij $f$ differentieerbaar op een interval $I$. Dan neemt $f'$
elke waarde tussen enige twee van haar waarden --- ook al hoeft
$f'$ niet continu te zijn.\index{stelling van Darboux}
\end{theorem}

\begin{proof}
Zij $a < b$ in $I$ en $v$ strikt tussen $f'(a)$ en $f'(b)$, zeg
$f'(a) < v < f'(b)$. De functie $g(x) = f(x) - vx$ is
differentieerbaar met $g'(a) < 0 < g'(b)$: haar minimum op
$\intcc{a}{b}$ (bereikt: continuïteit op een compactum) ligt niet
in $a$ (vlak na $a$ daalt $g$ onder $g(a)$) noch in $b$ (vlak
vóór $b$ is $g$ onder $g(b)$): het is inwendig, en daar $g'(c) =
0$, d.w.z.\ $f'(c) = v$. (Dit was een met ster gemarkeerde
oefening van Jaar~1; haar plaats in de theorie is hier.)
\end{proof}

\begin{example}[Welke functies zijn afgeleiden?]\label{ex:b2:realfun:whichderivatives}
De stelling van Darboux is een niet-bestaansmachine. De
vloerfunctie $\lfloor x\rfloor$ is niet de afgeleide van enige
functie op $\R$: zij neemt de waarden $0$ en $1$ maar slaat
$\frac12$ over op $\intcc{0}{1}$, wat
\cref{thm:b2:realfun:darboux} voor afgeleiden verbiedt. Hetzelfde
oordeel treft elke functie met een sprong --- sign, Heaviside,
alle trapfuncties --- hoe onschuldig zij er ook uitzien; hun
``primitieven'' ($\abs x$ voor sign, enz.) bestaan enkel weg van
de sprong en knopen daar met een hoek. Contrasteer: de wild
discontinue $f'$ van \cref{ex:b2:realfun:oscillation} \emph{is}
een afgeleide --- haar discontinuïteit is een oscillatie, die
Darboux verdraagt. De grens tussen de twee gedragingen is precies
het geen-sprongen-gevolg hieronder.
\end{example}

\begin{corollary}\label{cor:b2:realfun:nojumps}
Een afgeleide heeft geen sprongdiscontinuïteiten: als $f'(a^-)$
en $f'(a^+)$ bestaan, zijn zij gelijk aan $f'(a)$. De
discontinuïteiten van een afgeleide zijn altijd van
oscillatietype (de afgeleide van $x^2\sin\frac1x$ in $0$,
boekdeel van Jaar~1).
\end{corollary}

\begin{proof}
Als $f'(a^+) = \lim_{x\to a^+} f'(x)$ bestaat en verschilt van
$f'(a)$, zouden waarden strikt ertussen door $f'$ overgeslagen
worden op een rechteromgeving --- in tegenspraak met Darboux op
intervallen $\intcc{a}{a + h}$. (Alternatief: de
middelwaardestelling dwingt $f'(a) = \lim_{h\to0^+}
\frac{f(a+h)-f(a)}{h} = f'(a^+)$, het
differentiequotiënt zijnde een $f'$-waarde in een
tussenpunt.) Hetzelfde links.
\end{proof}

\begin{example}[De canonieke oscillerende afgeleide]\label{ex:b2:realfun:oscillation}
Zij $f(x) = x^2\sin\frac1x$ voor $x \neq 0$ en $f(0) = 0$. In
$0$: $\bigl|\frac{f(h) - f(0)}{h}\bigr| = \abs{h\sin\frac1h}
\leq \abs h \to 0$, dus $f'(0) = 0$ bestaat. Weg van $0$,
\[
f'(x) = 2x\sin\frac1x - \cos\frac1x ,
\]
waarvan de eerste term naar $0$ streeft terwijl $\cos\frac1x$
oscilleert door $\intcc{-1}{1}$ op elk interval
$\intoo{0}{\delta}$: de limiet $f'(0^+)$ bestaat niet. Dus is
$f'$ overal gedefinieerd maar discontinu in $0$ --- en, precies
zoals \cref{cor:b2:realfun:nojumps} voorspelt, is de
discontinuïteit een oscillatie, geen sprong: op elk
$\intoo{0}{\delta}$ veegt $f'$ nog een vol interval rond $0$.
Afgeleiden kunnen wild zijn, maar enkel op de
Darboux-compatibele manier.
\end{example}

\begin{remark}[Waar dit hoofdstuk gebruikt wordt]
Convexiteit is de motor van de ongelijkheidsindustrie: de
weekendopgave van dit hoofdstuk fabriceert Young, Hölder,
Minkowski en de machtsgemiddelde-keten eruit, die de
normtheorie van \cref{ch:b2:nvs} en de integraalschattingen van
\cref{ch:b2:integration} consumeren; Jensen keert terug in de
kansrekening als de momentongelijkheden van
\cref{ch:b2:randomvar}. Monotone regulariteit keert terug in
\cref{ch:b2:integration} (monotone functies zijn integreerbaar)
en, in het boekdeel van Jaar~3, als de bijna-overal
differentieerbaarheid van monotone functies --- waar
``\omterm{def:b2:structures:countable}{aftelbaar} vele sprongen'' de eerste stap van Lebesgues
theorie wordt.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b2:realfun:1}
Bepaal de discontinuïteitsverzamelingen en de spronggroottes:
$\lfloor x \rfloor$; $\;x - \lfloor x\rfloor$; $\;\lfloor x
\rfloor + \sqrt{x - \lfloor x\rfloor}$; de functie van het
voorbeeld na \cref{thm:b2:realfun:monotone} beperkt tot dyadische
rationalen $r_n$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:realfun:2}
Bewijs dat een stijgende functie $f \colon I \to \R$ met de
tussenwaarde-eigenschap (haar beeld van elk deelinterval is een
interval) continu is.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:realfun:3}
Welke van de volgende zijn convex op hun domein? $x \mapsto
x\ln x$ ($x > 0$); $\;x \mapsto \ln(1 + \eu^x)$; $\;x \mapsto
\sqrt{1 + x^2}$; $\;x \mapsto x^3$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:realfun:4}
Zij $f$ convex op $\R$ en van boven begrensd. Bewijs dat $f$
constant is. \emph{(Als $f(a) \neq f(b)$, plant de
hellingongelijkheid de niet-nul koordehelling voort: voorbij het
punt met de grotere waarde groeit $f$ minstens lineair --- in
tegenspraak met begrensdheid. Behandel beide tekens van de
helling.)} Leid af dat een convexe functie op $\R$ met een
asymptoot aan beide einden affien is.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:realfun:5}
Zij $f$ differentieerbaar op $I$ met $f'$ monotoon. Bewijs dat
$f'$ continu is \emph{(combineer
\cref{thm:b2:realfun:monotone} en
\cref{cor:b2:realfun:nojumps})}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:realfun:6}
(Meetkundig gemiddelde als limiet) Voor positieve $x_i$ en
gewichten $\lambda_i$ die tot $1$ sommen, bewijs
\[
\lim_{p \to 0^+} \Bigl(\sum_i \lambda_i x_i^p\Bigr)^{1/p}
= \prod_i x_i^{\lambda_i} ,
\]
via $x_i^p = \eu^{p\ln x_i} = 1 + p\ln x_i + O(p^2)$, en leid
de gewogen AM--GM-ongelijkheid af uit
\cref{ex:b2:realfun:powermeans}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:realfun:7}
(Entropieongelijkheid) Met strikte convexiteit van $t \mapsto
t\ln t$, bewijs dat voor positieve $p_i, q_i$ met $\sum p_i =
\sum q_i = 1$:
\[
\sum_i p_i \ln\frac{p_i}{q_i} \geq 0 ,
\]
met gelijkheid d.e.s.d.a.\ $p = q$. \emph{(Schrijf de linkerkant
als $\sum q_i\, \varphi\bigl(\frac{p_i}{q_i}\bigr)$ met
$\varphi(t) = t\ln t$ en pas Jensen toe met gewichten $q_i$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:realfun:8}
(Midpuntconvexiteit) $f \colon I \to \R$ is \emph{midpuntconvex}
wanneer $f\bigl(\frac{x+y}{2}\bigr) \leq \frac{f(x) + f(y)}{2}$
steeds. Bewijs dat een \emph{continue} midpuntconvexe functie
convex is. \emph{(Vestig de convexiteitsongelijkheid voor
dyadische gewichten $\frac{k}{2^m}$ door inductie op $m$, ga dan
over tot de limiet met dichtheid en continuïteit.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:realfun:9}
Zij $f$ convex op $\intco{0}{+\infty}$ met $f(0) \leq 0$. Bewijs
dat $x \mapsto \frac{f(x)}{x}$ stijgend is op
$\intoo{0}{+\infty}$, en leid af dat voor convexe $f$ met $f(0)
= 0$: $f(x + y) \geq f(x) + f(y)$ voor $x, y \geq 0$
(superadditiviteit).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:realfun:10}
Zij $f$ convex op $I$ en $g$ convex \emph{stijgend} op een
interval dat $f(I)$ bevat. Bewijs dat $g \circ f$ convex is, en
toon met een tegenvoorbeeld dat monotonie van $g$ niet
weggelaten kan worden.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:realfun:11}
(Hermite--Hadamard) Zij $f$ convex en continu op
$\intcc{a}{b}$. Bewijs
\[
f\Bigl(\frac{a+b}{2}\Bigr) \;\leq\; \frac{1}{b -
a}\int_a^b f(t)\,\dd t \;\leq\; \frac{f(a) + f(b)}{2} .
\]
\emph{(Links: integreer een steunlijn in het middenpunt. Rechts:
begrens $f$ door de koorde.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:realfun:12}
Bewijs dat een convexe functie op een \emph{open} interval $I$
lokaal \omterm{def:b2:metric:continuity}{Lipschitz} is: voor elk segment $\intcc{a}{b} \subseteq I$
en marge $\delta > 0$ met $\intcc{a - \delta}{b + \delta}
\subseteq I$ is de restrictie van $f$ tot $\intcc{a}{b}$
\omterm{def:b2:metric:continuity}{Lipschitz}, met constante
$\max\Bigl(\bigl|\frac{f(a) - f(a - \delta)}{\delta}\bigr|,
\bigl|\frac{f(b + \delta) - f(b)}{\delta}\bigr|\Bigr)$
\emph{(vang elke koordehelling tussen deze twee via de
hellingongelijkheid)}.
\end{exercise}

\section{Probleem: De convexiteitsgereedschapskist}

Eén definitie --- de koorde boven de grafiek --- genereert de
hele gereedschapskist van klassieke ongelijkheden. Deze
weekendopgave bouwt haar in logische volgorde: convexiteitscriteria
en strikt Jensen, dan Young, Hölder en Minkowski (de
geboortebewijzen van de $p$-normen), de volledige
machtsgemiddelde-keten van minimum tot maximum, en twee
kroondividenden --- de ongelijkheid van Carleman, en Hölder
gelezen als dualiteit. Alles wordt bewezen; niets wordt
ingevoerd.

\begin{problem}[{Weekendopgave --- Young, Hölder, Minkowski,
en de machtsgemiddelde-keten}]
\label{pb:b2:realfun:1}
Doorheen zijn $p, q > 1$ \emph{geconjugeerde exponenten}:
$\frac1p + \frac1q = 1$; vectoren zijn $a = (a_1, \dots, a_n)
\in \R^n$; gewichten $\lambda_i > 0$ voldoen aan
$\sum_i\lambda_i = 1$.

\medskip
\noindent\textbf{Deel I --- Criteria en strikt Jensen.}
\begin{enumerate}
  \item Zij $f$ differentieerbaar op een interval $I$. Bewijs dat
        $f$ convex is dan en slechts dan als $f'$ stijgend is
        \emph{(één richting door naar de limiet te gaan in de
        hellingongelijkheid \cref{lem:b2:realfun:slopes}; de
        andere door de middelwaardestelling)}. Leid het
        $C^2$-criterium $f'' \geq 0$ af.
  \item Stel $f''> 0$ op $I$. Bewijs dat $f$ \emph{strikt}
        convex is (strikte ongelijkheid voor $x \neq y$ en
        $\lambda \in \intoo01$), en dat een strikt convexe
        functie de ongelijkheid van Jensen
        (\cref{thm:b2:realfun:convexreg}~(3)) voldoet met
        gelijkheid \emph{enkel} wanneer alle $x_i$ samenvallen.
  \item Certificeer de grondstoffen van de gereedschapskist:
        $-\ln$ is strikt convex op $\intoo{0}{+\infty}$; $t
        \mapsto t^r$ is daar strikt convex voor $r > 1$ en
        strikt concaaf voor $0 < r < 1$; $\exp$ is strikt convex
        op $\R$.
  \item (Ongelijkheid van Young) Voor $a, b \geq 0$, bewijs
        \[
        ab \;\leq\; \frac{a^p}{p} + \frac{b^q}{q},
        \]
        met gelijkheid d.e.s.d.a.\ $a^p = b^q$ \emph{(pas de
        concaviteit van $\ln$ toe op de twee punten $a^p, b^q$
        met gewichten $\frac1p, \frac1q$)}.
  \item Herleid gewogen AM--GM in één regel uit de concaviteit
        van $\ln$:
        \[
        \prod_i x_i^{\lambda_i} \leq \sum_i\lambda_ix_i \qquad
        (x_i > 0),
        \]
        met het gelijkheidsgeval; vergelijk met de limietroute
        van \cref{exo:b2:realfun:6}.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Hölder en Minkowski.} Schrijf
$\norm{a}_p = \bigl(\sum_i \abs{a_i}^p\bigr)^{1/p}$ en
$\norm{a}_\infty = \max_i\abs{a_i}$.
\begin{enumerate}[resume]
  \item (Hölder) Bewijs
        \[
        \sum_{i=1}^{n}\abs{a_ib_i} \;\leq\;
        \norm a_p\,\norm b_q ,
        \]
        met gelijkheid d.e.s.d.a.\ de vectoren $(\abs{a_i}^p)$
        en $(\abs{b_i}^q)$ evenredig zijn \emph{(normaliseer
        $\norm a_p = \norm b_q = 1$ en pas Young termsgewijs
        toe)}.
  \item Identificeer de speciale gevallen: $p = q = 2$
        (Cauchy--Schwarz), en het eindpuntpaar $(p, q) = (1,
        \infty)$: formuleer en bewijs $\sum\abs{a_ib_i} \leq
        \norm a_1\norm b_\infty$.
  \item (Minkowski) Voor $p \geq 1$, bewijs
        \[
        \norm{a + b}_p \leq \norm a_p + \norm b_p
        \]
        \emph{(schrijf $\abs{a_i + b_i}^p \leq \abs{a_i +
        b_i}^{p-1}(\abs{a_i} + \abs{b_i})$ en pas Hölder toe op
        elk product)}. Concludeer: $\norm\cdot_p$ is een \omterm{def:b2:nvs:norm}{norm}
        op $\R^n$ voor elke $p \in \intco{1}{+\infty}$, wat het
        beeld van \cref{ch:b2:nvs} compleet maakt.
  \item Integraalversies: voor $f, g$ continu op
        $\intcc{a}{b}$, formuleer en bewijs Hölder en Minkowski
        voor $\norm f_p = \bigl(\int_a^b\abs
        f^p\bigr)^{1/p}$ \emph{(dezelfde bewijzen, met de
        strikte positiviteit van de integraal voor de
        gelijkheidsdiscussie)}.
  \item Bewijs de monotonie $\norm a_q \leq \norm a_p$ voor $1
        \leq p \leq q$, de limiet $\norm a_p \to
        \norm a_\infty$ als $p \to \infty$, en de omgekeerde
        vergelijking met de scherpe constante:
        \[
        \norm a_p \leq n^{\frac1p - \frac1q}\,\norm a_q
        \]
        \emph{(Hölder tegen de constante vector)}. Identificeer
        de vectoren die elke gelijkheid bereiken.
  \item (Interpolatie) Voor $1 \leq p < r < q$ en $\theta \in
        \intoo01$ met $\frac1r = \frac\theta p +
        \frac{1-\theta}q$, bewijs
        \[
        \norm a_r \leq \norm a_p^{\theta}\,
        \norm a_q^{1-\theta}
        \]
        \emph{(pas Hölder toe met exponenten $\frac{p}{\theta
        r}$ en $\frac{q}{(1-\theta)r}$ op
        $\abs{a_i}^{\theta r}\abs{a_i}^{(1-\theta)r}$)}.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De machtsgemiddelde-keten,
compleet.} Voor $p \neq 0$ stel $M_p = \bigl(\sum_i\lambda_i
x_i^p\bigr)^{1/p}$ ($x_i > 0$), en $M_0 = \prod_i
x_i^{\lambda_i}$.
\begin{enumerate}[resume]
  \item Bewijs dat $p \mapsto M_p$ stijgend is op heel $\R^*$:
        behandel $p < q < 0$ via de reciproke identiteit
        $M_{-p}(x) = M_p(1/x)^{-1}$, en overbrug door $0$ door
        $M_p \leq M_0 \leq M_q$ te tonen voor $p < 0 < q$
        \emph{(pas de concaviteit van $\ln$ toe op $x_i^q$, en
        de omgekeerde ongelijkheid voor negatieve exponenten)}.
  \item Bewijs de limieten $M_p \to \max_i x_i$ als $p \to
        +\infty$ en $M_p \to \min_i x_i$ als $p \to -\infty$.
  \item Schrijf de keten $\min \leq \mathrm{HM} \leq
        \mathrm{GM} \leq \mathrm{AM} \leq \mathrm{QM} \leq
        \max$ uit voor gelijke gewichten, en bewijs het
        klassieke gevolg: voor positieve $a_1, \dots, a_n$,
        \[
        \Bigl(\sum_i a_i\Bigr)\Bigl(\sum_i\frac1{a_i}\Bigr)
        \geq n^2 .
        \]
  \item Relateer gemiddelden aan \omterm{def:b2:nvs:norm}{normen}: voor gelijke gewichten
        $\lambda_i = \frac1n$, $M_p(x) = n^{-1/p}\norm x_p$.
        Verzoen de twee monotonieën --- gemiddelden
        \emph{stijgen} met $p$ terwijl \omterm{def:b2:nvs:norm}{normen} \emph{dalen}
        (vraag 10) --- in één zin over de factor $n^{-1/p}$.
  \item Bepaal de gelijkheidsgevallen langs de hele keten van
        vraag 14 (positieve gewichten): gelijkheid ergens
        dwingt alle $x_i$ gelijk --- strikte convexiteit loont.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Dividenden.}
\begin{enumerate}[resume]
  \item (Young met een knop) Voor $a, b \geq 0$ en $\varepsilon >
        0$, bewijs
        \[
        ab \leq \varepsilon\,\frac{a^p}{p} +
        \varepsilon^{-q/p}\,\frac{b^q}{q},
        \]
        en het werkpaardgeval $ab \leq \varepsilon a^2 +
        \frac{b^2}{4\varepsilon}$: de absorptietruc die overal
        in de analyse gebruikt wordt.
  \item (Naar Carleman) Zij $c_k = \frac{(k+1)^k}{k^{k-1}}$.
        Bewijs de telescoperende identiteit $\prod_{k=1}^{n}c_k
        = (n+1)^n$, en leid af, via AM--GM toegepast op de
        getallen $c_ka_k$,
        \[
        (a_1a_2\cdots a_n)^{1/n} \leq
        \frac{1}{n(n+1)}\sum_{k=1}^{n} c_k a_k
        \qquad (a_k > 0).
        \]
  \item (Ongelijkheid van Carleman) Sommeer over $n$, wissel de
        sommatievolgorde (positieve \omterm{def:b2:series:summable}{sommeerbare families},
        \cref{thm:b2:series:fubini}), en gebruik $\sum_{n \geq
        k}\frac{1}{n(n+1)} = \frac1k$ en $c_k/k = \bigl(1 +
        \frac1k\bigr)^k < \eu$ om te concluderen: voor elke
        convergente $\sum a_k$ met positieve termen,
        \[
        \sum_{n=1}^{\infty}(a_1a_2\cdots a_n)^{1/n}
        \;\leq\; \eu\sum_{k=1}^{\infty}a_k .
        \]
  \item Voor $f$ continu en positief op $\intcc{0}{1}$, bewijs
        \[
        \Bigl(\int_0^1 f\Bigr)\Bigl(\int_0^1\frac1f\Bigr)
        \geq 1,
        \]
        met gelijkheid d.e.s.d.a.\ $f$ constant is
        \emph{(Cauchy--Schwarz op $\sqrt f\cdot\frac1{\sqrt
        f}$)}.
  \item (Meetkunde van de ballen) Met het gelijkheidsgeval van
        Minkowski, toon dat voor $1 < p < \infty$ de
        eenheidssfeer van $\norm\cdot_p$ geen segment bevat (de
        \omterm{def:b2:nvs:norm}{norm} is strikt convex in de zin van
        \cref{pb:b2:nvs:1}), terwijl voor $p = 1$ en $p =
        \infty$ wel: vertoon de platte stukken.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Dualiteit en synthese.}
\begin{enumerate}[resume]
  \item (Hölder als dualiteit) Bewijs dat voor elke $a \in
        \R^n$,
        \[
        \norm a_p = \max_{\norm b_q \leq 1}\ \sum_i a_ib_i ,
        \]
        met een maximaliserende $b$ expliciet. (De $p$-norm is
        de duale van de $q$-norm --- de eindig-dimensionale
        kiem van $L^p$-dualiteit.)
  \item (Momenten) Zij $X$ een stochastische veranderlijke die
        eindig veel positieve waarden $x_i$ aanneemt met
        kansen $\lambda_i$. Herformuleer vraag 12 als: $r
        \mapsto \E[X^r]^{1/r}$ is stijgend --- de
        momentongelijkheid van Lyapunov, te hergebruiken in
        \cref{ch:b2:randomvar}.
  \item Los op met benoemde gereedschappen, in twee regels elk:
        (i) voor positieve $a, b, c$: $a^3 + b^3 + c^3 \geq
        \frac{(a+b+c)^3}{9}$; (ii) voor positieve $x_1, \dots,
        x_n$: $\bigl(\sum_i\sqrt{x_i}\bigr)^2 \leq
        n\sum_i x_i$.
  \item (Synthese) Teken de genealogie in vijf zinnen:
        koordedefinitie naar hellingslemma; hellingen naar
        steunlijnen naar Jensen; concaviteit van $\ln$ naar
        Young naar Hölder naar Minkowski naar de $p$-normen;
        Jensen naar de machtsgemiddelde-keten naar momenten;
        AM--GM naar Carleman. Noem de toppen
        (Hölder--Minkowski; Carleman), en zeg waar de
        gereedschapskist naartoe gaat: de $L^p$-ruimten van
        het boekdeel van Jaar~3, waarvan de axioma's precies
        vragen 6 en 8 zijn.
\end{enumerate}
\end{problem}
