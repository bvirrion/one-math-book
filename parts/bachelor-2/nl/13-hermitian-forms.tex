\chapter{Hermitische vormen}\label{ch:b2:hermitian}

Complexe vectorruimten hebben hun eigen inwendig-productmeetkunde, met
één twist: lineariteit in de ene variabele, \emph{geconjugeerde}
lineariteit in de andere. De beloning voor het accepteren van de twist
is een spectraaltheorie die nog schoner is dan de reële --- hermitische
endomorfismen hebben reële \omterm{def:b2:reduction:eigen}{eigenwaarden}, unitaire hebben unimodulaire
\omterm{def:b2:reduction:eigen}{eigenwaarden}, en beide diagonaliseren in orthonormale basissen. Dit
korte hoofdstuk voert het euclidische programma van \cref{ch:b2:quadratic}
uit over $\C$.

\section{Hermitische inwendige producten}

\begin{definition}\label{def:b2:hermitian:def}
Een \emph{hermitisch inwendig product}\index{hermitisch inwendig product} op
een complexe vectorruimte $E$ is een afbeelding $\langle\cdot,\cdot\rangle
\colon E \times E \to \C$ die lineair is in de tweede variabele,
\omterm{def:b2:quadratic:adjoint}{geconjugeerd-symmetrisch} ($\langle y, x\rangle = \conj{\langle x,
y\rangle}$ --- dus geconjugeerd-lineair in de eerste variabele), en
positief definiet ($\langle x, x\rangle > 0$ voor $x \neq 0$). Het
standaardvoorbeeld op $\C^n$:
\[
\langle x, y \rangle = \sum_{i=1}^{n} \conj{x_i}\, y_i ;
\]
op continue functies, $\langle f, g\rangle = \int_a^b \conj
f\,g$. \omterm{def:b2:nvs:norm}{Norm}: $\norm x = \sqrt{\langle x,x\rangle}$; een
eindigdimensionale complexe ruimte zo uitgerust is een \emph{hermitische
ruimte}\index{hermitische ruimte}.
\end{definition}

\begin{example}[Eerste berekeningen]\label{ex:b2:hermitian:firstcomp}
In $\C^2$, neem $x = (1+\iu,\ 2-\iu)$ en $y = (\iu,\ 1)$.
Dan
\[
\norm x^2 = \abs{1+\iu}^2 + \abs{2-\iu}^2 = 2 + 5 = 7,
\qquad
\norm y^2 = 1 + 1 = 2,
\]
en, met conjugatie op het eerste argument,
\[
\langle x, y\rangle
= \conj{(1+\iu)}\,\iu + \conj{(2-\iu)}\cdot1
= (1-\iu)\iu + (2+\iu)
= (\iu + 1) + (2 + \iu) = 3 + 2\iu .
\]
Cauchy--Schwarz klopt: $\abs{\langle x, y\rangle}^2 = 9 +
4 = 13 \leq 14 = \norm x^2\norm y^2$ --- dicht bij gelijkheid,
omdat $x$ dicht bij een veelvoud van $y$ ligt. Merk ook op
$\langle y, x\rangle = \conj{3 + 2\iu} = 3 - 2\iu$:
geconjugeerde symmetrie in actie, en de reden dat $\langle x,
x\rangle$ altijd reëel is.
\end{example}

\begin{theorem}[Cauchy--Schwarz, complex]\label{thm:b2:hermitian:cs}
$\abs{\langle x, y\rangle} \leq \norm x \norm y$, met gelijkheid d.e.s.d.a.\
$x, y$ lineair afhankelijk zijn; $\norm\cdot$ is een \omterm{def:b2:nvs:norm}{norm}. Bovendien
bestaan orthonormale basissen (Gram--Schmidt loopt letterlijk), met
\[
x = \sum_i \langle e_i, x\rangle\, e_i,
\qquad
\norm x^2 = \sum_i \abs{\langle e_i, x\rangle}^2 .
\]
\end{theorem}

\begin{proof}
Voor $y \neq 0$ en $t \in \C$: $0 \leq \norm{x - ty}^2 = \norm x^2
- 2\Re\bigl(\conj t\langle y, x\rangle\bigr) + \abs t^2\norm y^2$.
Kies $t = \frac{\langle y, x\rangle}{\norm y^2}$:
\[
0 \leq \norm x^2 - \frac{\abs{\langle y, x\rangle}^2}{\norm y^2},
\]
wat de ongelijkheid is; gelijkheid dwingt $x = ty$. Driehoeks-
ongelijkheid, \omterm{def:b2:metric:complete}{volledig}:
\[
\norm{x + y}^2 = \norm x^2 + 2\,\Re\langle x, y\rangle
+ \norm y^2
\leq \norm x^2 + 2\,\abs{\langle x, y\rangle} + \norm y^2
\leq \bigl(\norm x + \norm y\bigr)^2 ,
\]
via $\Re z \leq \abs z$ en dan Cauchy--Schwarz; homogeniteit
en scheiding zijn onmiddellijk, dus $\norm\cdot$ is een \omterm{def:b2:nvs:norm}{norm}.
Gram--Schmidt: zoals in het reële geval, met conjugaten
geplaatst door de definitie (let op de conventie: onze producten zijn
geconjugeerd-lineair in de \emph{eerste} slot, dus coördinaten zijn
$\langle e_i, x\rangle$; het volgende voorbeeld voert het algoritme
één keer \omterm{def:b2:metric:complete}{volledig} uit).
\end{proof}

\begin{example}[Complexe Gram--Schmidt, volledig uitgevoerd]\label{ex:b2:hermitian:gramschmidt}
Orthonormaliseer de basis $v_1 = (1, \iu)$, $v_2 = (0, 1)$ van
$\C^2$. Eerste vector: $\norm{v_1}^2 = \abs1^2 + \abs\iu^2 = 2$,
dus $e_1 = \frac{1}{\sqrt2}(1, \iu)$. Projecteer $v_2$ --- met de
conjugatie in de eerste slot:
\[
\langle e_1, v_2\rangle
= \frac{1}{\sqrt2}\bigl(\conj{1}\cdot0 +
\conj{\iu}\cdot1\bigr)
= \frac{-\iu}{\sqrt2} ,
\qquad
v_2 - \langle e_1, v_2\rangle e_1
= (0,1) + \frac{\iu}{2}\,(1, \iu)
= \Bigl(\frac\iu2,\ \frac12\Bigr) .
\]
Haar \omterm{def:b2:nvs:norm}{norm} is $\sqrt{\frac14 + \frac14} = \frac{1}{\sqrt2}$:
$e_2 = \frac{1}{\sqrt2}(\iu, 1)$. Controle: $\langle e_1,
e_2\rangle = \frac12(\conj1\cdot\iu + \conj\iu\cdot1) =
\frac12(\iu - \iu) = 0$. Coördinaten van $v_2$ in de nieuwe basis:
$v_2 = \langle e_1, v_2\rangle e_1 + \langle e_2, v_2\rangle
e_2$ met $\langle e_2, v_2\rangle = \frac{1}{\sqrt2}$ ---
let op de volgorde: $\langle v_2, e_1\rangle$ zou de
\emph{geconjugeerde} coëfficiënt geven. Slotinzicht: het algoritme is
letterlijk het euclidische; de enige val is waar de
conjugatie valt, en het berekenen van $\norm{v_2 -
\text{proj}}^2 > 0$ gebruikt stilzwijgend positiviteit --- het axioma
dat de hele meetkunde doet werken.
\end{example}

\section{Adjunct, hermitische en unitaire endomorfismen}

\begin{definition}\label{def:b2:hermitian:adjoint}
De \emph{\omterm{def:b2:quadratic:adjoint}{adjunct}} $u^*$ van $u \in \mathcal{L}(E)$ is gedefinieerd door
$\langle u^*(x), y\rangle = \langle x, u(y)\rangle$; in een
orthonormale basis is $\operatorname{Mat}(u^*) = \conj{A}^{\mathsf T} =:
A^{\dagger}$ (geconjugeerde \omterm{def:b2:linalg:transpose}{getransponeerde}) --- inderdaad, als $B =
(b_{ij})$ de matrix van $u^*$ is in de orthonormale basis
$(e_i)$, dan is $b_{ij} = \langle e_i, u^*(e_j)\rangle$, en de
definerende identiteit geeft
\[
\conj{b_{ij}}
= \langle u^*(e_j), e_i\rangle
= \langle e_j, u(e_i)\rangle = a_{ji} ,
\qquad\text{dus}\qquad
B = \conj{A}^{\mathsf T} .
\] $u$ is
\emph{hermitisch}\index{hermitisch endomorfisme} wanneer $u^* = u$
($A^\dagger = A$), \emph{unitair}\index{unitaire groep} wanneer $u^*u =
\mathrm{id}$ ($A^\dagger A = I$: de groep $U(n)$), \emph{normaal}
wanneer $u^*u = uu^*$.
\end{definition}

\begin{proposition}\label{prop:b2:hermitian:eigenvalues}
\omterm{def:b2:reduction:eigen}{Eigenwaarden} van een \omterm{def:b2:hermitian:adjoint}{hermitisch endomorfisme} zijn \emph{reëel};
\omterm{def:b2:reduction:eigen}{eigenwaarden} van een \omterm{def:b2:hermitian:adjoint}{unitair} endomorfisme hebben \emph{modulus $1$}; in
beide gevallen zijn \omterm{def:b2:reduction:eigen}{eigenruimten} bij verschillende \omterm{def:b2:reduction:eigen}{eigenwaarden}
orthogonaal.
\end{proposition}

\begin{proof}
\omterm{def:b2:hermitian:adjoint}{Hermitisch}, $u(x) = \lambda x$, $x \neq 0$:
\[
\lambda \norm x^2 = \langle x, u(x)\rangle = \langle u(x), x\rangle
= \conj{\langle x, u(x)\rangle} = \conj\lambda\,\norm x^2 ,
\]
dus $\lambda \in \R$. \omterm{def:b2:hermitian:adjoint}{Unitair}: $\norm{u(x)} = \norm x$ (uit $u^*u =
\mathrm{id}$), dus $\abs\lambda\norm x = \norm x$. Orthogonaliteit
(\omterm{def:b2:hermitian:adjoint}{hermitisch} geval): $\lambda\langle x, y\rangle = \langle u(x),
y\rangle = \langle x, u(y)\rangle = \mu\langle x, y\rangle$ voor
\omterm{def:b2:reduction:eigen}{eigenvectoren} met reële $\lambda \neq \mu$. \omterm{def:b2:hermitian:adjoint}{Unitair} geval,
\omterm{def:b2:metric:complete}{volledig}: voor $u(x) = \lambda x$, $u(y) = \mu y$ met $\lambda \neq
\mu$ (beide unimodulair),
\[
\langle x, y\rangle = \langle u(x), u(y)\rangle
= \conj\lambda\mu\,\langle x, y\rangle ,
\]
en $\conj\lambda\mu = \frac{\mu}{\lambda} \neq 1$: de factor
is niet $1$, dus $\langle x, y\rangle = 0$.
\end{proof}

\begin{example}[Een scheef-hermitische matrix, gediagonaliseerd]\label{ex:b2:hermitian:skewexample}
$A = \begin{pmatrix} 0 & -2\\ 2 & 0\end{pmatrix}$ voldoet aan
$A^\dagger = A^{\mathsf T} = -A$: \omterm{def:b2:hermitian:adjoint}{scheef-hermitisch} (ook reëel
antisymmetrisch --- over $\R$ heeft zij überhaupt geen \omterm{def:b2:reduction:eigen}{eigenwaarden}).
\omterm{def:b2:reduction:charpoly}{Karakteristieke veelterm} $X^2 + 4$: \omterm{def:b2:reduction:eigen}{eigenwaarden} $\pm2\iu$,
zuiver imaginair, zoals \cref{exo:b2:hermitian:9} in het
algemeen voorspelt. \omterm{def:b2:reduction:eigen}{Eigenvectoren}: $(A - 2\iu I)v = 0$ geeft $v_1 =
\frac{1}{\sqrt2}(1, \iu)$, en $v_2 = \frac{1}{\sqrt2}(1,
-\iu)$ voor $-2\iu$; zij zijn orthogonaal:
\[
\langle v_1, v_2\rangle
= \tfrac12\bigl(\conj{1}\cdot1 +
\conj{\iu}\cdot(-\iu)\bigr)
= \tfrac12(1 - 1) = 0 .
\]
Dus $A = U\operatorname{diag}(2\iu, -2\iu)\,U^\dagger$ met $U =
(v_1\ v_2)$ \omterm{def:b2:hermitian:adjoint}{unitair}. Slotinzicht: $H = -\iu A =
\begin{pmatrix} 0 & 2\iu\\ -2\iu & 0\end{pmatrix}$ is \omterm{def:b2:hermitian:adjoint}{hermitisch}
met het reële spectrum $\{\pm2\}$ en de \emph{zelfde}
\omterm{def:b2:reduction:eigen}{eigenvectoren} --- de bijectie $u \mapsto \iu u$ tussen
hermitische en scheef-hermitische endomorfismen
(\cref{exo:b2:hermitian:9}), matrix voor matrix gezien; over $\R$
is dezelfde $A$ een rotatie-schaling zonder \omterm{def:b2:reduction:eigen}{eigenvectoren}
helemaal, en alleen de overgang naar $\C$ onthult haar normale vorm.
\end{example}

\begin{theorem}[Hermitische spectraalstelling]\label{thm:b2:hermitian:spectral}
Elk \omterm{def:b2:hermitian:adjoint}{hermitisch endomorfisme} van een \omterm{def:b2:hermitian:def}{hermitische ruimte} heeft een orthonormale
basis van \omterm{def:b2:reduction:eigen}{eigenvectoren} (met reële \omterm{def:b2:reduction:eigen}{eigenwaarden}): $A^\dagger = A$
impliceert $A = U D U^{\dagger}$ met $U \in U(n)$ en $D$ reëel
diagonaal.\index{spectraalstelling!hermitisch}
\end{theorem}

\begin{proof}
Over $\C$ splitst de \omterm{def:b2:reduction:charpoly}{karakteristieke veelterm}: er bestaat een \omterm{def:b2:reduction:eigen}{eigenvector}
$e_1$ (\cref{ch:b2:reduction}) --- geen compactheidsargument
nodig, een voordeel van $\C$. Normaliseer hem. Zijn orthogonaal
complement $F = e_1^\perp$ is stabiel: voor $x \perp e_1$,
\[
\langle e_1, u(x)\rangle = \langle u(e_1), x\rangle
= \lambda_1\langle e_1, x\rangle = 0
\]
($\lambda_1$ reëel). De restrictie is \omterm{def:b2:hermitian:adjoint}{hermitisch}; inducteer naar de
dimensie en concateneer. In detail: de restrictie $u|_F$ is
een endomorfisme van de \omterm{def:b2:hermitian:def}{hermitische ruimte} $F$ (dimensie $n - 1$)
met $\langle u|_F(x), y\rangle = \langle x, u|_F(y)\rangle$
geërfd van $u$; de inductiehypothese levert een
orthonormale basis $(e_2, \dots, e_n)$ van $F$ van \omterm{def:b2:reduction:eigen}{eigenvectoren},
en $(e_1, e_2, \dots, e_n)$ is orthonormaal in $E$ ($e_1 \perp
F$) en bestaat uit \omterm{def:b2:reduction:eigen}{eigenvectoren} van $u$. Matrixvertaling: de
kolommen van $U$ zijn de $e_i$, $U^\dagger U = I$ drukt hun
orthonormaliteit uit, en $AU = UD$ verzamelt de eigenwaarde-
vergelijkingen, vandaar $A = UDU^\dagger$ met $D$ reëel diagonaal
(\cref{prop:b2:hermitian:eigenvalues}).
\end{proof}

\begin{example}\label{ex:b2:hermitian:pauli}
$A = \begin{pmatrix} 0 & -\iu\\ \iu & 0\end{pmatrix}$ is \omterm{def:b2:hermitian:adjoint}{hermitisch}
($A^\dagger = A$): \omterm{def:b2:reduction:eigen}{eigenwaarden} uit $\chi_A = X^2 - 1$: $\pm 1$
(reëel, zoals beloofd), met orthonormale \omterm{def:b2:reduction:eigen}{eigenvectoren}
$\frac{1}{\sqrt2}(1, \iu)^{\mathsf T}$ en $\frac{1}{\sqrt2}(1,
-\iu)^{\mathsf T}$. (Natuurkundigen kennen $A$ als een Pauli-matrix; de
realiteit van hermitische spectra is waarom kwantumobservabelen
gemodelleerd worden door hermitische operatoren.)
\end{example}

\begin{example}[Een positief definiete hermitische matrix, uitgewerkt]\label{ex:b2:hermitian:pdrun}
$A = \begin{pmatrix} 2 & 1-\iu\\ 1+\iu & 3\end{pmatrix}$:
\omterm{def:b2:hermitian:adjoint}{hermitisch}, want de diagonaal is reëel en de off-diagonale
ingangen zijn conjugaten. \omterm{def:b2:reduction:charpoly}{Karakteristieke veelterm}:
\[
(2-\lambda)(3-\lambda) - \abs{1-\iu}^2
= \lambda^2 - 5\lambda + 4
= (\lambda - 1)(\lambda - 4) :
\]
spectrum $\{1, 4\}$, reëel en positief --- $A$ is positief
definiet. \omterm{def:b2:reduction:eigen}{Eigenvectoren}: voor $\lambda = 4$ geeft het stelsel $(A -
4I)v = 0$ $v_4 = (1 - \iu,\ 2)$ (controleer de tweede rij:
$(1+\iu)(1-\iu) - 2 = 0$); voor $\lambda = 1$, $v_1 = (1 - \iu,\
-1)$. Orthogonaliteit, met de conjugatie in de eerste slot:
\[
\langle v_4, v_1\rangle
= \conj{(1-\iu)}\,(1-\iu) + \conj{2}\,(-1)
= 2 - 2 = 0 . \checkmark
\]
Normaliseren ($\norm{v_4}^2 = 2 + 4 = 6$, $\norm{v_1}^2 = 2 + 1
= 3$) geeft de unitaire $U = \bigl(\frac{v_4}{\sqrt6}\
\frac{v_1}{\sqrt3}\bigr)$ met $A =
U\operatorname{diag}(4,1)U^\dagger$. Slotinzicht: de
Rayleigh-lezing is onmiddellijk --- op de eenheidsbol van
$\C^2$ doorloopt $\langle x, Ax\rangle$ $\intcc{1}{4}$,
bereikt in de twee \omterm{def:b2:reduction:eigen}{eigenvectoren}; dit is de $n = 2$-kiem van
de Courant--Fischer-theorie gebouwd in het weekeindprobleem.
Steekproef dat de vorm reëel is buiten de \omterm{def:b2:reduction:eigen}{eigenvectoren}: in
$x = (1, \iu)$,
\[
Ax = \bigl(2 + (1-\iu)\iu,\ (1+\iu) + 3\iu\bigr)
= (3 + \iu,\ 1 + 4\iu),
\]
\[
\langle x, Ax\rangle = \conj{1}\,(3+\iu) +
\conj{\iu}\,(1+4\iu)
= (3 + \iu) + (-\iu)(1 + 4\iu) = 3 + \iu - \iu + 4 = 7 \in
\R ,
\]
zoals het bewijsmechanisme van \cref{prop:b2:hermitian:eigenvalues}
(geconjugeerde symmetrie tegen $A^\dagger = A$) garandeert voor
elke $x$.
\end{example}

\begin{example}[Een unitaire matrix gediagonaliseerd]\label{ex:b2:hermitian:unitary}
$U = \frac{1}{\sqrt2}\begin{pmatrix} 1 & \iu\\ \iu & 1
\end{pmatrix}$ (\omterm{def:b2:hermitian:adjoint}{unitair} via \cref{exo:b2:hermitian:2}). Haar
\omterm{def:b2:reduction:charpoly}{karakteristieke veelterm} is $\bigl(X - \frac{1}{\sqrt2}\bigr)^2
+ \frac12$, met wortels
\[
\lambda_\pm = \frac{1 \pm \iu}{\sqrt2} = \eu^{\pm\iu\pi/4},
\]
unimodulair zoals \cref{prop:b2:hermitian:eigenvalues} beloofde, en
orthonormale \omterm{def:b2:reduction:eigen}{eigenvectoren} $\frac{1}{\sqrt2}(1, \pm1)$. Dus $U =
V\operatorname{diag}(\eu^{\iu\pi/4},
\eu^{-\iu\pi/4})V^\dagger$: in de juiste basis is $U$ een paar
vlakke rotaties over $\pm\frac\pi4$ --- een reële rotatiematrix heeft
geen reële \omterm{def:b2:reduction:eigen}{eigenvectoren}, maar over $\C$ splitst zij in twee
unimodulaire scalaren. Slotinzicht: hermitische spectra leven op
de reële lijn, unitaire spectra op de eenheidscirkel; beide zijn
schaduwen van dezelfde normaliteit, en de Cayley-transformatie van
\cref{exo:b2:hermitian:6} beeldt het ene plaatje op het andere af.
\end{example}

\begin{example}[De Cayley-transformatie, berekend]\label{ex:b2:hermitian:cayleyrun}
Voer \cref{exo:b2:hermitian:6} uit op $H = \begin{pmatrix} 0 & 1\\
1 & 0\end{pmatrix}$ (\omterm{def:b2:hermitian:adjoint}{hermitisch}, spectrum $\{1, -1\}$,
orthonormale \omterm{def:b2:reduction:eigen}{eigenvectoren} $\frac{1}{\sqrt2}(1, \pm1)$). In de
eigenbasis is alles scalair: de transformatie $\lambda
\mapsto \frac{\lambda - \iu}{\lambda + \iu}$ stuurt
\[
1 \longmapsto \frac{1 - \iu}{1 + \iu} = -\iu,
\qquad
-1 \longmapsto \frac{-1 - \iu}{-1 + \iu} = \iu ,
\]
(vermenigvuldig met de geconjugeerde van de noemer), dus $U = (H -
\iu I)(H + \iu I)^{-1}$ is de unitaire met \omterm{def:b2:reduction:eigen}{eigenwaarden}
$\mp\iu$ op diezelfde \omterm{def:b2:reduction:eigen}{eigenvectoren}:
\[
U = \frac{1}{2}\begin{pmatrix} 1 & 1\\ 1 & -1\end{pmatrix}
\begin{pmatrix} -\iu & 0\\ 0 & \iu\end{pmatrix}
\begin{pmatrix} 1 & 1\\ 1 & -1\end{pmatrix}
= \begin{pmatrix} 0 & -\iu\\ -\iu & 0\end{pmatrix} .
\]
Controle: $U^\dagger U = I$, en $1 \notin \operatorname{Sp}U =
\{\pm\iu\}$, zoals de theorie belooft. Slotinzicht: de reële
lijn beeldt af op de eenheidscirkel minus het punt $1$ ---
\omterm{def:b2:reduction:eigen}{eigenwaarde} voor \omterm{def:b2:reduction:eigen}{eigenwaarde} is de Cayley-transformatie de
Möbius-afbeelding $\frac{\lambda-\iu}{\lambda+\iu}$, en matrices
volgen gewoon hun spectra.
\end{example}

\begin{remark}[Veelgemaakte valkuilen]
\emph{(i) Waar de streep valt:} dit boek conjugeert de
\emph{eerste} slot, dus coördinaten zijn $\langle e_i, x\rangle$
en $\langle\lambda x, y\rangle = \conj\lambda\langle x,
y\rangle$; veel teksten conjugeren in plaats daarvan de tweede slot ---
vertaal vóór formules vergeleken worden, anders gaan tekens van $\iu$
stilzwijgend fout. \emph{(ii) Complexe polarisatie is sterker:}
over $\C$, als $\langle x, u(x)\rangle = 0$ voor \emph{alle} $x$
dan $u = 0$ (werk $x + y$ en $x + \iu y$ uit: zowel het reële
als het imaginaire deel van $\langle x, u(y)\rangle$ verdwijnt); over
$\R$ faalt dit --- de rotatie over $\frac\pi2$ voldoet aan
$\langle x, u(x)\rangle = 0$ overal. Bijgevolg, alleen over
$\C$, dwingt ``$\langle x, u(x)\rangle \in \R$ voor alle $x$''
reeds dat $u$ \omterm{def:b2:hermitian:adjoint}{hermitisch} is. \emph{(iii) Reëel normaal is niet
\omterm{def:b2:reduction:diag}{diagonaliseerbaar}:} de matrix van
\cref{ex:b2:hermitian:skewexample} is normaal maar heeft geen reële
\omterm{def:b2:reduction:eigen}{eigenwaarde}; unitaire diagonalisatie is een stelling \emph{over}
$\C$, en over $\R$ krijgt men alleen blokreducties.
\emph{(iv) Unitariteit controleren:} $U^\dagger U = I$ betekent dat de
\emph{kolommen} orthonormaal zijn voor het hermitische product ---
$UU^{\mathsf T}$ testen, of de conjugatie in de
kolomproducten vergeten, zijn de twee klassieke manieren om een
verkeerde matrix te certificeren.
\end{remark}

\begin{example}[Isometrieën zijn precies de unitairen]\label{ex:b2:hermitian:isometry}
Normbehoud lijkt zwakker dan unitariteit, maar over $\C$ is dat
niet zo: als $\norm{u(x)} = \norm x$ voor alle $x$, dan $u^*u =
\mathrm{id}$. Inderdaad is $v = u^*u - \mathrm{id}$ \omterm{def:b2:hermitian:adjoint}{hermitisch} en
voldoet $\langle x, v(x)\rangle = \norm{u(x)}^2 - \norm x^2 =
0$ voor elke $x$; door complexe polarisatie (valkuil (ii)
hierboven) is een afbeelding met identiek verdwijnende ``diagonaal''
nul: $v = 0$. Concreet loopt de polarisatie
\[
0 = \langle x + y, v(x+y)\rangle
= \langle x, v(y)\rangle + \langle y, v(x)\rangle,
\qquad
0 = \langle x + \iu y, v(x + \iu y)\rangle
= \iu\langle x, v(y)\rangle - \iu\langle y, v(x)\rangle ,
\]
en de twee regels samen dwingen $\langle x, v(y)\rangle = 0$
voor alle $x, y$. Slotinzicht: daarom kan ``\omterm{def:b2:hermitian:adjoint}{unitair}''
gecontroleerd worden door alleen lengtes te meten --- starheid die het
Fourier-hoofdstuk zal uitbuiten, waar behoud van de energie
$\norm f_2$ (Parseval) hetzelfde is als behoud van alle inwendige
producten van coëfficiënten.
\end{example}

\begin{remark}[Perspectieven binnen dit volume]
De hermitische machinerie die hier gebouwd is, wordt bijna
onmiddellijk geconsumeerd. Het Fourier-hoofdstuk \emph{is} hermitische
meetkunde in oneindige dimensie: de exponentialen $(e_n)$ zijn een
orthonormale familie voor $\langle f, g\rangle =
\frac{1}{2\pi}\int\conj fg$, de ongelijkheid van Bessel is de
projectieschatting van
\cref{thm:b2:hermitian:cs} van dit hoofdstuk, en Parseval is haar
limietgelijkheid. De eindige Fourier-transformatie
(\cref{exo:b2:hermitian:10}) keert terug wanneer convolutie
gediagonaliseerd moet worden. En het weekeindprobleem van dit
hoofdstuk --- Courant--Fischer, Weyl, interlacing --- levert de
eigenwaarde-stabiliteit die het differentiaalvergelijkingen-
hoofdstuk inroept wanneer het beweert dat kleine storingen van een
systeem zijn frequenties slechts licht verschuiven. Achterwaarts is
alles hier de complexe spiegel van het kwadratische-vormenhoofdstuk:
houd de twee woordenboeken naast elkaar ($A^{\mathsf T} \leftrightarrow
A^\dagger$, orthogonaal $\leftrightarrow$ \omterm{def:b2:hermitian:adjoint}{unitair}, Rayleigh
reëel in beide).
\end{remark}

\begin{remark}[Normale endomorfismen]
Over $\C$ is de definitieve uitspraak: $u$ is \emph{\omterm{def:b2:hermitian:adjoint}{unitair}
\omterm{def:b2:reduction:diag}{diagonaliseerbaar} d.e.s.d.a.\ hij normaal is} ($u^*u = uu^*$) ---
hermitische, unitaire en scheef-hermitische afbeeldingen in één keer
dekkend. Het bewijs is een aangename versterking van het argument hierboven
(\cref{exo:b2:hermitian:8}). Over $\R$, daarentegen, koopt normaliteit
alleen blokdiagonalisatie (rotatieblokken): complexe meetkunde is
oprecht eenvoudiger.
\end{remark}

\begin{remark}[Waar dit gebruikt wordt]
Hermitische spectraaltheorie is de wiskunde van de kwantum-
mechanica: observabelen worden gemodelleerd door hermitische operatoren
(reële spectra = meetbare waarden), tijdevolutie door unitaire
(normbehoud = behoud van waarschijnlijkheid). Binnen
dit boek steunt het Fourier-hoofdstuk op de orthonormaliteit van
de exponentialen --- een hermitische inwendig-productuitspraak --- en
is de diagonalisatie van circulante matrices
(\cref{exo:b2:hermitian:10}) de eindige Fourier-transformatie.
Het weekeindprobleem ontwikkelt de variationele calculus van
\omterm{def:b2:reduction:eigen}{eigenwaarden} (Courant--Fischer, Weyl, interlacing), het dagelijks
brood van numerieke analyse en mathematische fysica; het jaar-
3-volume breidt het uit tot compacte zelftoegevoegde operatoren op
Hilbert-ruimten.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b2:hermitian:1}
Op $\C^2$: bereken $\langle x, y\rangle$, $\norm x$, $\norm y$ voor
$x = (1, \iu)$, $y = (\iu, 1)$; zijn zij orthogonaal? Geef een
orthonormale basis die $\frac{x}{\norm x}$ bevat.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:hermitian:2}
Welke zijn \omterm{def:b2:hermitian:adjoint}{hermitisch}? \omterm{def:b2:hermitian:adjoint}{unitair}? normaal?
\[
\begin{pmatrix} 1 & \iu\\ -\iu & 2 \end{pmatrix},
\qquad
\frac{1}{\sqrt2}\begin{pmatrix} 1 & \iu\\ \iu & 1\end{pmatrix},
\qquad
\begin{pmatrix} 0 & 1\\ 0 & 0 \end{pmatrix}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:hermitian:3}
Bewijs dat een matrix $A \in \mathcal{M}_n(\C)$ uniek schrijft $A =
H + \iu K$ met $H, K$ \omterm{def:b2:hermitian:adjoint}{hermitisch} \emph{(de ``reële en imaginaire
delen'' $H = \frac{A + A^\dagger}{2}$, $K = \frac{A -
A^\dagger}{2\iu}$)}, en dat $A$ normaal is d.e.s.d.a.\ $H$ en $K$
commutereren.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:hermitian:4}
Diagonaliseer in een orthonormale basis: $A = \begin{pmatrix} 2 & \iu\\
-\iu & 2\end{pmatrix}$, en bereken $A^k$ voor $k \in \N$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:hermitian:5}
Bewijs dat $U(n)$ \omterm{def:b2:metric:compact}{compact} is, en dat de eigenwaarde-afbeelding
surjectief is: elke unimodulaire $\lambda$ ontstaat voor enige unitaire
matrix. Bewijs dat $\det U \in \mathbb{U}$ (de eenheidscirkel) voor $U \in
U(n)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:hermitian:6}
(Cayley-transformatie) Laat $H$ \omterm{def:b2:hermitian:adjoint}{hermitisch} zijn. Bewijs dat $H + \iu I$
inverteerbaar is en dat $U = (H - \iu I)(H + \iu I)^{-1}$ \omterm{def:b2:hermitian:adjoint}{unitair} is,
met $1 \notin \operatorname{Sp}(U)$. \emph{(Werk spectraal: op een
eigenbasis van $H$ is alles scalair.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:hermitian:7}
Voor $A$ \omterm{def:b2:hermitian:adjoint}{hermitisch} positief definiet ($\langle x, Ax\rangle > 0$ voor
$x \neq 0$), bewijs dat $\operatorname{Sp}(A) \subseteq
\intoo{0}{\infty}$, dat $A = B^2$ voor een hermitische positief
definiete $B$, en dat $\det A > 0$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:hermitian:8}
(Spectraalstelling voor normale endomorfismen) Laat $u$ normaal zijn op een
\omterm{def:b2:hermitian:def}{hermitische ruimte}.
\begin{enumerate}
  \item Bewijs $\norm{u(x)} = \norm{u^*(x)}$ voor alle $x$, en leid af
        $\ker(u - \lambda) = \ker(u^* - \conj\lambda)$.
  \item Bewijs dat \omterm{def:b2:reduction:eigen}{eigenruimten} van $u$ bij verschillende \omterm{def:b2:reduction:eigen}{eigenwaarden}
        orthogonaal zijn, en dat het orthogonaal complement van een
        \omterm{def:b2:reduction:eigen}{eigenruimte} $u$-stabiel is.
  \item Concludeer inductief dat $u$ \omterm{def:b2:hermitian:adjoint}{unitair} \omterm{def:b2:reduction:diag}{diagonaliseerbaar} is;
        en omgekeerd.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:hermitian:9}
Een endomorfisme is \emph{\omterm{def:b2:hermitian:adjoint}{scheef-hermitisch}} wanneer $u^* = -u$. Bewijs
dat zijn \omterm{def:b2:reduction:eigen}{eigenwaarden} zuiver imaginair zijn, dat $u \mapsto \iu
u$ een bijectie is van hermitische naar scheef-hermitische
endomorfismen, en dat scheef-hermitische endomorfismen
\omterm{def:b2:hermitian:adjoint}{unitair} \omterm{def:b2:reduction:diag}{diagonaliseerbaar} zijn (\cref{exo:b2:hermitian:8}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:hermitian:10}
(De eindige Fourier-transformatie) Laat $S$ de cyclische verschuiving van
$\C^n$ zijn: $S(x_0, x_1, \dots, x_{n-1}) = (x_{n-1}, x_0, \dots,
x_{n-2})$, en $\omega = \eu^{2\iu\pi/n}$.
\begin{enumerate}
  \item Toon dat $S$ \omterm{def:b2:hermitian:adjoint}{unitair} is, en dat de vectoren $f_k =
        \frac{1}{\sqrt n}\bigl(1, \omega^k, \omega^{2k}, \dots,
        \omega^{(n-1)k}\bigr)$, $0 \leq k < n$, een
        orthonormale basis van \omterm{def:b2:reduction:eigen}{eigenvectoren} vormen: $Sf_k = \omega^{-k}
        f_k$.
  \item Leid af dat elke \emph{circulante} matrix $C =
        \sum_{j=0}^{n-1} c_jS^j$ normaal is, gediagonaliseerd door
        dezelfde basis, met \omterm{def:b2:reduction:eigen}{eigenwaarden} $\widehat c(k) = \sum_j
        c_j\,\omega^{-jk}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:hermitian:11}
Laat $P$ een idempotent ($P^2 = P$) endomorfisme van een \omterm{def:b2:hermitian:def}{hermitische
ruimte} zijn. Bewijs dat $P$ de \emph{orthogonale} projectie is op
$\operatorname{im} P$ d.e.s.d.a.\ $P^* = P$. Geef de matrix
van de orthogonale projectie op $\C v$ ($\norm v = 1$), en
op een deelruimte met orthonormale basis $(v_1, \dots, v_k)$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:hermitian:12}
(Spectrale projectoren via interpolatie) Laat $A$ \omterm{def:b2:hermitian:adjoint}{hermitisch} zijn met
\emph{onderscheiden} \omterm{def:b2:reduction:eigen}{eigenwaarden} $\lambda_1, \dots, \lambda_p$ en
eigenruimte-ontbinding $E = \bigoplus_i E_i$. Definieer de
Lagrange-veeltermen $L_i(X) = \prod_{j\neq i}\frac{X -
\lambda_j}{\lambda_i - \lambda_j}$. Bewijs dat $P_i = L_i(A)$
de orthogonale projectie is op $E_i$, dat $P_iP_j = 0$ voor
$i \neq j$, $\sum_i P_i = I$, en $A = \sum_i \lambda_iP_i$
(de \emph{spectrale ontbinding}); druk $f(A)$ voor willekeurige
veelterm $f$ uit in termen van de $P_i$.
\end{exercise}

\section{Probleem: Courant--Fischer, Weyl en de calculus van
eigenwaarden}

\begin{problem}\label{pb:b2:hermitian:1}
De \omterm{def:b2:reduction:eigen}{eigenwaarden} van een hermitische matrix zijn niet alleen wortels van een
veelterm: zij zijn oplossingen van \emph{optimalisatieproblemen}.
Dat variationele gezichtspunt --- \omterm{pb:b2:hermitian:1}{Rayleigh-quotiënten} en de
\emph{min-max-stelling van Courant--Fischer}\index{stelling van Courant--Fischer}
--- maakt \omterm{def:b2:reduction:eigen}{eigenwaarden} vergelijkbaar, stabiel en
berekenbaar, en dit probleem oogst zijn klassieke vruchten:
\emph{storingsongelijkheden van Weyl}\index{ongelijkheid van Weyl},
\emph{Cauchy-interlacing}, de spoorongelijkheden van Schur en Ky Fan,
de monotonie van de matrixvierkantswortel,
en het spectrum van de discrete Laplace-operator. Doorheen is $A, B,
E$ \omterm{def:b2:hermitian:adjoint}{hermitisch} op $E = \C^n$ met \omterm{def:b2:reduction:eigen}{eigenwaarden} genoteerd in
dalende volgorde $\lambda_1(A) \geq \dots \geq \lambda_n(A)$,
en $R_A(x) = \frac{\langle x, Ax\rangle}{\langle x, x\rangle}$
voor $x \neq 0$ is het \emph{Rayleigh-quotiënt}\index{Rayleigh-quotiënt}.

\medskip
\noindent\textbf{Deel I --- \omterm{pb:b2:hermitian:1}{Rayleigh-quotiënten} en min-max.}
Fixeer een orthonormale eigenbasis $(e_1, \dots, e_n)$, $Ae_i =
\lambda_ie_i$.
\begin{enumerate}
  \item Toon dat $R_A(x)$ reëel is, en dat
        \[
        \lambda_n \leq R_A(x) \leq \lambda_1
        \qquad (x \neq 0),
        \]
        beide grenzen bereikt: $\lambda_1 = \max R_A$,
        $\lambda_n = \min R_A$.
  \item Toon dat de kritieke punten van $R_A$ precies de
        \omterm{def:b2:reduction:eigen}{eigenvectoren} van $A$ zijn \emph{(werk $t \mapsto R_A(x +
        tv)$ uit in $t = 0$ voor willekeurige $v$, vervang dan $v$
        door $\iu v$)}.
  \item Laat $V_k = \operatorname{Vect}(e_1, \dots, e_k)$ en
        $W_k = \operatorname{Vect}(e_k, \dots, e_n)$. Toon
        \[
        \min_{x \in V_k\setminus\{0\}} R_A(x) = \lambda_k
        = \max_{x \in W_k\setminus\{0\}} R_A(x) .
        \]
  \item Bewijs de \emph{stelling van Courant--Fischer}: voor $1 \leq
        k \leq n$,
        \[
        \lambda_k = \max_{\dim V = k}\;\min_{x \in
        V\setminus\{0\}} R_A(x)
        = \min_{\dim W = n-k+1}\;\max_{x \in
        W\setminus\{0\}} R_A(x)
        \]
        \emph{(voor elke $V$ van dimensie $k$: $V \cap W_k \neq
        \{0\}$ via Grassmann, dus $\min_V R_A \leq \lambda_k$;
        vraag 3 toont dat de grens bereikt wordt)}.
  \item (Monotonie) Schrijf $A \leq B$ wanneer $B - A$
        positief semi-definiet is. Leid uit vraag 4 af: $A \leq
        B$ impliceert $\lambda_k(A) \leq \lambda_k(B)$ voor elke
        $k$.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De ongelijkheden van Weyl.}
\begin{enumerate}[resume]
  \item Toon dat deelruimten $V, W \subseteq \C^n$ met $\dim V
        + \dim W > n$ niet-triviaal snijden, en veralgemeen:
        $\dim(V_1 \cap V_2 \cap V_3) \geq \dim V_1 + \dim V_2 +
        \dim V_3 - 2n$.
  \item Bewijs de \emph{ongelijkheid van Weyl}: voor $i + j - 1 \leq n$,
        \[
        \lambda_{i+j-1}(A + B) \leq \lambda_i(A) +
        \lambda_j(B)
        \]
        \emph{(snijd de deelruimten $W_i(A)$, $W_j(B)$ en
        $V_{i+j-1}(A+B)$ van vraag 3 en tel
        dimensies)}.
  \item Definieer $\vertiii{E}_2 = \max_{\norm x = 1}\norm{Ex}$
        en toon $\vertiii E_2 = \max_k\abs{\lambda_k(E)}$ voor
        hermitische $E$. Leid de \emph{storingsstelling van Weyl}
        af:
        \[
        \bigl|\lambda_k(A + E) - \lambda_k(A)\bigr| \leq
        \vertiii{E}_2
        \qquad (1 \leq k \leq n) :
        \]
        elke \omterm{def:b2:reduction:eigen}{eigenwaarde} is een $1$-Lipschitz-functie van de
        matrix.
  \item (Rang-één-storingen) Laat $P$ \omterm{def:b2:hermitian:adjoint}{hermitisch} positief
        semi-definiet van rang $1$ zijn. Toon
        \[
        \lambda_k(A) \leq \lambda_k(A + P) \leq
        \lambda_{k-1}(A) \qquad (2 \leq k \leq n),
        \]
        samen met $\lambda_1(A) \leq \lambda_1(A+P)$: de
        nieuwe \omterm{def:b2:reduction:eigen}{eigenwaarden} \emph{interlacen} de oude.
  \item Controleer vraag 8 numeriek: $A = \begin{pmatrix} 2 &
        \iu\\ -\iu & 2\end{pmatrix}$
        (\cref{exo:b2:hermitian:4}: spectrum $\{3, 1\}$) en
        $E = \begin{pmatrix} 0 & 1\\ 1 & 0\end{pmatrix}$
        (spectrum $\{1, -1\}$): bereken het spectrum van $A +
        E$ en beide kanten van de ongelijkheid.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Interlacing en spoor-
ongelijkheden.}
\begin{enumerate}[resume]
  \item (Cauchy-interlacing) Laat $B$ de leidende $(n-1)
        \times(n-1)$ principale submatrix van $A$ zijn. Bewijs
        \[
        \lambda_{k+1}(A) \leq \lambda_k(B) \leq \lambda_k(A)
        \qquad (1 \leq k \leq n-1)
        \]
        \emph{(bekijk $\C^{n-1} \subseteq \C^n$; daarop is $R_B$
        de restrictie van $R_A$; pas Courant--Fischer toe op
        beide niveaus)}.
  \item Itereer: voor een principale submatrix $B$ van grootte $n - m$,
        $\lambda_{k+m}(A) \leq \lambda_k(B) \leq
        \lambda_k(A)$.
  \item (Schur) Laat $d_1 \geq d_2 \geq \dots \geq d_n$ de
        diagonale ingangen van $A$ zijn, gesorteerd. Bewijs, voor elke $k$:
        \[
        \sum_{i=1}^{k} d_i \leq \sum_{i=1}^{k}\lambda_i(A),
        \]
        met gelijkheid in $k = n$ (het spoor)
        \emph{(de $k$ gekozen diagonale ingangen vormen een
        principale $k\times k$-submatrix; begrens haar spoor via
        vraag 12)}.
  \item (Ky Fan) Bewijs
        \[
        \sum_{i=1}^{k}\lambda_i(A)
        = \max\Bigl\{\sum_{i=1}^{k}\langle x_i, Ax_i\rangle
        : (x_1, \dots, x_k) \text{ orthonormaal}\Bigr\} .
        \]
  \item Verifieer vragen 11 en 13 op
        \[
        A = \begin{pmatrix} 2 & 1 & 0\\ 1 & 2 & 1\\
        0 & 1 & 2\end{pmatrix}
        \qquad
        \bigl(\operatorname{Sp} = \{2 - \sqrt2,\ 2,\
        2+\sqrt2\}\bigr)
        \]
        tegen haar leidende $2\times2$-blok (spectrum $\{1,
        3\}$) en haar diagonaal.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De Loewner-orde.} $A \leq B$
betekent nog steeds $B - A$ positief semi-definiet; alle matrices in dit
deel zijn \omterm{def:b2:hermitian:adjoint}{hermitisch}.
\begin{enumerate}[resume]
  \item Toon: $A \leq B$ impliceert $a_{ii} \leq b_{ii}$ voor alle
        $i$, $\operatorname{tr} A \leq \operatorname{tr} B$,
        en $C^\dagger AC \leq C^\dagger BC$ voor \emph{elke}
        complexe matrix $C$.
  \item Toon dat kwadrateren \emph{niet} monotoon is: voor
        \[
        A = \begin{pmatrix} 1 & 0\\ 0 & 0\end{pmatrix},
        \qquad
        B = \begin{pmatrix} 2 & 1\\ 1 & 1\end{pmatrix},
        \]
        controleer $0 \leq A \leq B$ maar $A^2 \not\leq B^2$.
  \item Bewijs dat de vierkantswortel \emph{wel} monotoon is: $0 \leq
        A \leq B$ impliceert $\sqrt A \leq \sqrt B$ \emph{(laat
        $\mu$ een \omterm{def:b2:reduction:eigen}{eigenwaarde} van $\sqrt B - \sqrt A$ zijn met eenheids-
        \omterm{def:b2:reduction:eigen}{eigenvector} $v$; bereken $\langle v, (B - A)v\rangle =
        \mu\bigl(\langle v, \sqrt B\,v\rangle + \langle v,
        \sqrt A\,v\rangle\bigr)$ en discussieer)}.
  \item Bewijs dat inverteren antitonoon is op positief definiete
        matrices: $0 < A \leq B$ impliceert $B^{-1} \leq A^{-1}$
        \emph{(congruentie via $A^{-1/2}$ om te herleiden tot $I \leq M
        \Rightarrow M^{-1} \leq I$, wat scalair is in een
        spectraalbasis)}.
  \item Laat $A, B$ positief definiet zijn. Toon dat de
        \omterm{def:b2:reduction:eigen}{eigenwaarden} van $AB$ (niet \omterm{def:b2:hermitian:adjoint}{hermitisch} in het algemeen!)
        reëel en positief zijn, en dat
        \[
        \lambda_{\max}(AB) \leq
        \lambda_{\max}(A)\,\lambda_{\max}(B)
        \]
        \emph{(conjugeren met $\sqrt A$: $AB \sim \sqrt
        A\,B\sqrt A$)}.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- De discrete Laplace-operator, uitgewerkt.}
Laat $T_n$ de $n \times n$ tridiagonale matrix zijn met $2$ op
de diagonaal en $-1$ op de twee aangrenzende diagonalen.
\begin{enumerate}[resume]
  \item Met $\theta_k = \frac{k\pi}{n+1}$, verifieer dat de
        vectoren $v_k = \bigl(\sin(j\theta_k)\bigr)_{1\leq j\leq
        n}$ voldoen aan $T_nv_k = (2 - 2\cos\theta_k)\,v_k$
        \emph{(product-tot-som-identiteit; controleer de randrijen
        $j = 1, n$)}. Concludeer:
        \[
        \operatorname{Sp}(T_n) = \Bigl\{4\sin^2
        \frac{k\pi}{2(n+1)} : 1 \leq k \leq n\Bigr\},
        \]
        alle enkelvoudig, alle positief: $T_n$ is positief definiet.
  \item (Een potentiaal) Voor een reële diagonaal $D =
        \operatorname{diag}(d_1, \dots, d_n)$, sandwich het
        spectrum: voor elke $k$,
        \[
        \lambda_k(T_n) + \min_i d_i \;\leq\;
        \lambda_k(T_n + D) \;\leq\; \lambda_k(T_n) + \max_i
        d_i .
        \]
  \item Controleer Cauchy-interlacing tussen $T_3$ en $T_2$
        expliciet (spectra $\{2 \pm \sqrt2, 2\}$ en $\{1,
        3\}$), en interpreteer: $T_2$ is $T_3$ met één eindpunt
        van het pad verwijderd.
  \item Toon dat de extreme \omterm{def:b2:reduction:eigen}{eigenwaarden}, als $n \to
        \infty$, voldoen aan:
        \[
        \lambda_{\min}(T_n) = 4\sin^2\frac{\pi}{2(n+1)}
        \sim \frac{\pi^2}{(n+1)^2},
        \qquad
        \lambda_{\max}(T_n) \to 4 ,
        \]
        dus het conditioneringsgetal $\kappa_n =
        \lambda_{\max}/\lambda_{\min}$ groeit als
        $\frac{4(n+1)^2}{\pi^2}$: een tweede
        afgeleide discretiseren op een steeds fijnere grid is
        intrinsiek slecht geconditioneerd.
  \item Synthese. In één zin elk: (i) waarom de
        variationele karakterisering, niet de \omterm{def:b2:reduction:charpoly}{karakteristieke
        veelterm}, \omterm{def:b2:reduction:eigen}{eigenwaarden} stabiel maakt
        (vragen 8--9); (ii) welke vragen alleen
        $\lambda_1 = \max R_A$ gebruikten en welke de volledige
        min-max nodig hadden; (iii) wat de Loewner-orde toevoegt aan
        het verhaal; (iv) waar deze gereedschappen terugkeren (numerieke
        analyse van de stijfheidsmatrices van vraag 24; kwantum-
        storingstheorie; en, in het jaar-3-volume, het
        min-max-principe voor compacte zelftoegevoegde
        operatoren).
\end{enumerate}
\end{problem}
