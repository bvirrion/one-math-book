\chapter{Genormeerde vectorruimten}\label{ch:b2:nvs}

Wanneer de \omterm{def:b2:metric:def}{metrische ruimte} een vectorruimte is en de afstand van
een \omterm{def:b2:nvs:norm}{norm} komt, beginnen topologie en lineaire algebra te
wisselenwerken: lineaire afbeeldingen zijn precies continu wanneer
zij begrensd zijn op de eenheidsbol, eindige dimensie dwingt alle
\omterm{def:b2:nvs:norm}{normen} om overeen te stemmen, en volledigheid maakt absoluut
convergente reeksen tot convergente. De kloof eindig/oneindig
dimensionaal --- gekristalliseerd in de stelling van Riesz --- is
de diepste les van het hoofdstuk.

Doorheen zijn $E, F$ vectorruimten over $K = \R$ of $\C$.

\section{Normen}

\begin{definition}\label{def:b2:nvs:norm}
Een \emph{norm}\index{norm} op $E$ is een afbeelding $\norm{\,\cdot\,}
\colon E \to \R_+$ met, voor alle $x, y \in E$, $\lambda \in K$:
\[
\norm x = 0 \iff x = 0,
\qquad
\norm{\lambda x} = \abs\lambda\,\norm x,
\qquad
\norm{x + y} \leq \norm x + \norm y .
\]
Dan is $d(x, y) = \norm{x - y}$ een afstand, en heel
\cref{ch:b2:metric} is van toepassing. De omgekeerde
driehoeksongelijkheid $\bigl|\norm x - \norm y\bigr| \leq
\norm{x - y}$ maakt de norm zelf $1$-Lipschitz; optelling en
scalaire vermenigvuldiging zijn continu (schattingen $\norm{(x +
y) - (x' + y')} \leq \norm{x - x'} + \norm{y - y'}$, enz.).
\end{definition}

\begin{example}\label{ex:b2:nvs:examples}
Op $K^n$:
\[
\norm{x}_1 = \sum_i \abs{x_i},
\qquad
\norm{x}_2 = \Bigl(\sum_i \abs{x_i}^2\Bigr)^{1/2},
\qquad
\norm{x}_\infty = \max_i \abs{x_i}
\]
($\norm\cdot_2$ is een \omterm{def:b2:nvs:norm}{norm} door Cauchy--Schwarz, volume van
Jaar~1). Op $C(\intcc{a}{b})$:
\[
\norm f_\infty = \sup \abs f,
\qquad
\norm f_1 = \int_a^b \abs f,
\qquad
\norm f_2 = \Bigl(\int_a^b \abs f^2\Bigr)^{1/2},
\]
de laatste twee zijn \omterm{def:b2:nvs:norm}{normen} dankzij strikte positiviteit van de
integraal en integrale Cauchy--Schwarz (volume van Jaar~1). Op
matrices: elke \omterm{def:b2:nvs:norm}{norm} op $\mathcal M_n(K) \simeq K^{n^2}$; de
\omterm{thm:b2:nvs:continuouslinear}{operatornormen} hieronder zijn de structureel belangrijke.
\end{example}

\begin{definition}[Equivalente normen]\label{def:b2:nvs:equivalent}
Twee \omterm{def:b2:nvs:norm}{normen} $N_1, N_2$ op $E$ zijn \emph{equivalent}\index{equivalente
normen} wanneer er constanten $c, C > 0$ zijn met
\[
c\,N_1 \leq N_2 \leq C\, N_1 .
\]
Equivalente \omterm{def:b2:nvs:norm}{normen} hebben dezelfde open verzamelingen, dezelfde
convergente en Cauchyrijen, dezelfde compacte en volledige
deelverzamelingen: dezelfde analyse.
\end{definition}

\begin{example}[Niet-equivalentie in oneindige dimensie]\label{ex:b2:nvs:nonequivalent}
Op $C(\intcc{0}{1})$: $\norm f_1 \leq \norm f_\infty$ altijd, maar
geen omgekeerde grens geldt: $f_n(x) = x^n$ heeft $\norm{f_n}_\infty
= 1$ en $\norm{f_n}_1 = \frac{1}{n+1} \to 0$. Dus $f_n \to 0$ voor
$\norm\cdot_1$ maar niet voor $\norm\cdot_\infty$: de twee \omterm{def:b2:nvs:norm}{normen}
zijn het oneens over convergentie zelf.
\end{example}

\begin{example}[Expliciete constanten in dimensie $n$]\label{ex:b2:nvs:constants}
Op $K^n$ zijn de drie klassieke \omterm{def:b2:nvs:norm}{normen} \omterm{def:b2:nvs:equivalent}{equivalent} met scherpe
constanten:
\[
\norm x_\infty \leq \norm x_2 \leq \norm x_1
\leq \sqrt n\,\norm x_2 \leq n\,\norm x_\infty ,
\]
de middelste grens $\norm x_1 \leq \sqrt n\norm x_2$ komt van
Cauchy--Schwarz tegen de all-ones-vector. Extremale vectoren: $e_1$
maakt de eerste twee ongelijkheden tot gelijkheden, $(1, 1, \dots,
1)$ de laatste twee. De dimensie $n$ zit zichtbaar in de
constanten --- het kwantitatieve zaad van het falen in oneindige
dimensie: als $n \to \infty$ overleeft geen uniforme constante, precies
wat \cref{ex:b2:nvs:nonequivalent} op functieruimten tentoonspreidt.
\end{example}

\section{Continue lineaire afbeeldingen}

\begin{theorem}[Karakterisering]\label{thm:b2:nvs:continuouslinear}
Voor een lineaire afbeelding $u \colon E \to F$ tussen genormeerde
ruimten zijn de volgende \omterm{def:b2:nvs:equivalent}{equivalent}:
\begin{enumerate}
  \item $u$ is continu;
  \item $u$ is continu in $0$;
  \item $u$ is begrensd op de gesloten eenheidsbol:
        $\sup_{\norm x \leq 1} \norm{u(x)} < \infty$;
  \item er is $C \geq 0$ met $\norm{u(x)} \leq C \norm x$ voor alle
        $x$;
  \item $u$ is \omterm{def:b2:metric:continuity}{Lipschitz}.
\end{enumerate}
De kleinste zulke $C$ is de \emph{operatornorm}\index{operatornorm}
$\vertiii{u} = \sup_{\norm x \leq 1}\norm{u(x)} = \sup_{x \neq 0}
\frac{\norm{u(x)}}{\norm x}$; zij maakt de ruimte $\mathcal{L}_c(E,
F)$ van continue lineaire afbeeldingen tot een genormeerde ruimte,
met
\[
\vertiii{v \circ u} \leq \vertiii v\, \vertiii u .
\]
\end{theorem}

\begin{proof}
(1 $\Rightarrow$ 2) triviaal. (2 $\Rightarrow$ 3): continuïteit in
$0$ met $\varepsilon = 1$ geeft $\delta$ met $\norm x \leq \delta
\Rightarrow \norm{u(x)} \leq 1$; homogeniteit schaalt dan elke $x$
met $\norm x \leq 1$ omlaag in die bol en terug:
\[
\norm{u(x)} = \frac1\delta\,\norm{u(\delta x)} \leq
\frac1\delta ,
\]
aangezien $\norm{\delta x} \leq \delta$. (3 $\Rightarrow$ 4): voor
$x \neq 0$, pas de grens toe op $\frac{x}{\norm x}$. (4
$\Rightarrow$ 5): $\norm{u(x) - u(y)} = \norm{u(x - y)} \leq
C\norm{x - y}$. (5 $\Rightarrow$ 1) bekend.

Normaxioma's voor $\vertiii\cdot$: homogeniteit en scheiding zijn
duidelijk ($\vertiii u = 0$ forceert $u = 0$ op de bol, dus overal);
driehoeksongelijkheid uit $\norm{(u + v)(x)} \leq \norm{u(x)} +
\norm{v(x)}$. Submultiplicativiteit: $\norm{v(u(x))} \leq
\vertiii v\,\norm{u(x)} \leq \vertiii v \vertiii u \norm x$.
\end{proof}

\begin{example}\label{ex:b2:nvs:operatornorms}
Op $\bigl(C(\intcc{0}{1}), \norm\cdot_\infty\bigr)$: evaluatie $f
\mapsto f(0)$ heeft \omterm{thm:b2:nvs:continuouslinear}{operatornorm} $1$; integratie $f \mapsto
\int_0^1 f$ heeft \omterm{def:b2:nvs:norm}{norm} $1$; de afbeelding $f \mapsto \int_0^1 t
f(t)\dd t$ heeft \omterm{def:b2:nvs:norm}{norm} $\int_0^1 t\,\dd t = \frac12$ (bovengrens door
de driehoeksongelijkheid voor integralen; bereikt in $f \equiv 1$).
Maar differentiatie, van $(C^1, \norm\cdot_\infty)$ naar $(C^0,
\norm\cdot_\infty)$, is \emph{niet} continu: $\norm{\sin(nx)}_\infty
= 1$ terwijl de afgeleide \omterm{def:b2:nvs:norm}{sup-norm} $n$ heeft. Lineair impliceert
niet continu in oneindige dimensie.
\end{example}

\begin{example}[Twee normen, twee oordelen over één rij]\label{ex:b2:nvs:sqrtnxn}
Op $C(\intcc01)$, zij $g_n(x) = \sqrt{n}\,x^n$. Dan
\[
\norm{g_n}_1 = \frac{\sqrt n}{n + 1} \longrightarrow 0,
\qquad
\norm{g_n}_2^2 = \frac{n}{2n + 1} \longrightarrow \frac12,
\qquad
\norm{g_n}_\infty = \sqrt n \longrightarrow \infty :
\]
één rij, drie \omterm{def:b2:nvs:norm}{normen}, drie gedragingen --- convergentie naar
nul, geen convergentie (\omterm{def:b2:nvs:norm}{normen} stabiliseren op $\frac1{\sqrt2}$ maar
de puntgewijze limiet is $0$), en explosie. Massa die concentreert
nabij $x = 1$ is onzichtbaar voor $\norm\cdot_1$, half zichtbaar
voor $\norm\cdot_2$, dominant voor $\norm\cdot_\infty$. In
oneindige dimensie is ``convergeert het?'' geen vraag over een rij:
het is een vraag over een rij \emph{en} een \omterm{def:b2:nvs:norm}{norm}.
\end{example}

\begin{method}[Een operatornorm berekenen]\label{met:b2:nvs:opnorm}
Altijd in twee zetten. \emph{Bovengrens:} schat $\norm{u(x)}$ door
$C\norm x$ met driehoeksongelijkheden, Cauchy--Schwarz, of
integraalschattingen --- dit bewijst $\vertiii u \leq C$.
\emph{Getuige:} toon ofwel een specifiek $x_0 \neq 0$ met
$\norm{u(x_0)} = C\norm{x_0}$ (de grens wordt bereikt), of een rij
eenheidsvectoren $x_n$ met $\norm{u(x_n)} \to C$ (de grens wordt
benaderd). Beide zetten zijn verplicht: een bovengrens alleen geeft
slechts $\vertiii u \leq C$, een getuige alleen slechts $\vertiii u
\geq C$. In oneindige dimensie mag de getuige een rij moeten zijn
--- het supremum hoeft niet bereikt te worden
(\cref{exo:b2:nvs:8}).
\end{method}

\begin{example}[Diagonale operatoren zien elke norm hetzelfde]\label{ex:b2:nvs:diagonalnorm}
Voor $D = \operatorname{diag}(d_1, \dots, d_n)$ op $K^n$ met
\emph{elke} van de \omterm{def:b2:nvs:norm}{normen} $\norm\cdot_1, \norm\cdot_2,
\norm\cdot_\infty$: uit $\abs{d_ix_i} \leq
\bigl(\max_j\abs{d_j}\bigr)\abs{x_i}$ coördinaatsgewijs,
$\norm{Dx} \leq \max_j\abs{d_j}\,\norm x$; en $x = e_{j_0}$ (een
maximaliserende index) bereikt het. Dus $\vertiii D =
\max_j\abs{d_j}$ in alle drie de gevallen: voor diagonale
afbeeldingen vertellen alle redelijke \omterm{def:b2:nvs:norm}{normen} hetzelfde verhaal, de
grootste rekkingsfactor. Alles moeilijks over \omterm{thm:b2:nvs:continuouslinear}{operatornormen} gaat
over \emph{niet}-diagonaal gedrag --- daarom werken de aangepaste
\omterm{def:b2:nvs:norm}{normen} van het weekendvraagstuk van dit hoofdstuk (vraag 22) door
eerst een matrix diagonaal te maken.
\end{example}

\begin{example}[Kolomsommen: de $1$-norm-tweeling van \cref{exo:b2:nvs:4}]\label{ex:b2:nvs:columnsums}
Op $(\R^n, \norm\cdot_1)$ is de \omterm{thm:b2:nvs:continuouslinear}{operatornorm} van een matrix $A$ de
grootste absolute \emph{kolom}som. Voer de methode uit: voor $\norm
x_1 \leq 1$,
\[
\norm{Ax}_1 = \sum_i\Bigl|\sum_j a_{ij}x_j\Bigr|
\leq \sum_j \abs{x_j}\sum_i\abs{a_{ij}}
\leq \Bigl(\max_j\sum_i\abs{a_{ij}}\Bigr)\norm x_1 ,
\]
en de grens wordt bereikt in $x = e_{j_0}$ voor een maximaliserende
kolom $j_0$ --- de netste denkbare getuige. Dus voor $A =
\begin{psmallmatrix}1 & -2\\ 3 & 1\end{psmallmatrix}$:
$\vertiii A_1 = \max(1 + 3,\ 2 + 1) = 4$, terwijl $\vertiii
A_\infty = 4$ ook (rijen) --- hier een toeval, geen wet:
transponeer de matrixentries asymmetrisch en de twee \omterm{def:b2:nvs:norm}{normen}
scheiden. Rijen voor $\norm\cdot_\infty$, kolommen voor
$\norm\cdot_1$: de ezelsbrug is dat de eenheidsvectoren van elke
\omterm{def:b2:nvs:norm}{norm} (tekenpatronen, resp.\ basisvectoren) de bijpassende sommen
eruit pikken.
\end{example}

\begin{proposition}[Bilineaire afbeeldingen]\label{prop:b2:nvs:bilinear}
Een bilineaire afbeelding $b \colon E \times F \to G$ is continu
iff $\norm{b(x,y)} \leq C\norm x\,\norm y$ voor zekere $C$; dan is
zij \omterm{def:b2:metric:continuity}{Lipschitz} op begrensde verzamelingen. (Zelfde bewijspatroon; het
product $(u, v) \mapsto v \circ u$ en matrixvermenigvuldiging zijn
de sleutelvoorbeelden.)
\end{proposition}

\begin{proof}
Als de grens geldt:
\[
b(x,y) - b(x_0,y_0) = b(x - x_0,\, y) + b(x_0,\, y - y_0),
\]
dus $\norm{b(x,y) - b(x_0,y_0)} \leq C\norm{x - x_0}\norm y +
C\norm{x_0}\norm{y - y_0}$: continuïteit in $(x_0, y_0)$, en een
Lipschitzgrens waar $\norm x, \norm y \leq R$. Omgekeerd geeft
continuïteit in $(0,0)$ $\delta$ met $\norm{b(x,y)} \leq 1$ op
$\norm x, \norm y \leq \delta$; schaal beide variabelen.
\end{proof}

\section{Eindige dimensie}

\begin{theorem}[Equivalentie van normen in eindige dimensie]\label{thm:b2:nvs:finitedim}
Op een eindigdimensionale ruimte zijn \emph{alle \omterm{def:b2:nvs:norm}{normen}
\omterm{def:b2:nvs:equivalent}{equivalent}}.\index{equivalentie van normen} Bijgevolg, in eindige
dimensie: convergentie, openheid, compactheid, volledigheid zijn
normonafhankelijke begrippen; \omterm{def:b2:metric:compact}{compact} $=$ gesloten en begrensd; de
ruimte is \omterm{def:b2:metric:complete}{volledig}; en elke lineaire (of multilinaire) afbeelding
\emph{vanuit} een eindigdimensionale ruimte is continu.
\end{theorem}

\begin{proof}
Fixeer een basis en identificeer $E \simeq K^n$; het volstaat om
elke \omterm{def:b2:nvs:norm}{norm} $N$ te vergelijken met $\norm\cdot_\infty$.

\emph{Eén richting is algebra:} $N(x) = N(\sum x_i e_i) \leq \sum
\abs{x_i} N(e_i) \leq C \norm x_\infty$ met $C = \sum N(e_i)$. Dit
toont ook dat $N$ continu is op $(K^n, \norm\cdot_\infty)$ (zij is
$C$-Lipschitz: $\abs{N(x) - N(y)} \leq N(x - y)$).

\emph{De andere is topologie:} de eenheidssfeer $S = \{x :
\norm{x}_\infty = 1\}$ is gesloten en begrensd in $(K^n,
\norm\cdot_\infty)$, dus \omterm{def:b2:metric:compact}{compact}
(\cref{thm:b2:metric:compactprops}~(2), geldig voor $\C^n \simeq
\R^{2n}$). De continue functie $N$ bereikt haar minimum $c$ op $S$;
$c > 0$ aangezien $N$ alleen in $0 \notin S$ verdwijnt.
Homogeniteit spreidt de grens: $N(x) \geq c \norm{x}_\infty$ voor
alle $x$.

Gevolgen: alle uitspraken herleiden tot $(K^n,
\norm\cdot_\infty)$, waar ze bekend zijn
(\cref{thm:b2:metric:rncomplete},
\cref{thm:b2:metric:compactprops}); een lineaire $u$ vanuit
eindigdimensionale $E$ voldoet aan $\norm{u(x)} \leq \sum\abs{x_i}
\norm{u(e_i)} \leq C'\norm{x}_\infty$: grens (4) van
\cref{thm:b2:nvs:continuouslinear}.
\end{proof}

\begin{corollary}\label{cor:b2:nvs:closedsubspace}
Een eindigdimensionale deelruimte van elke genormeerde ruimte is
gesloten.
\end{corollary}

\begin{proof}
Zij is \omterm{def:b2:metric:complete}{volledig} voor de geïnduceerde \omterm{def:b2:nvs:norm}{norm}
(\cref{thm:b2:nvs:finitedim}), en volledige deelverzamelingen zijn
gesloten (\cref{def:b2:metric:complete}).
\end{proof}

\begin{example}[Een beste benadering berekend door symmetrie]\label{ex:b2:nvs:bestaffine}
In $\bigl(C(\intcc{-1}{1}), \norm\cdot_\infty\bigr)$, hoe ver is
$f(x) = \abs x$ van de (gesloten, tweedimensionale) deelruimte van
affiene functies $a + bx$? Door symmetrie laat vervangen van $a +
bx$ door $a - bx$ $\norm{f - (a \pm bx)}_\infty$ onveranderd, en
het middenpunt $a$ doet minstens zo goed (driehoeksongelijkheid op
het gemiddelde): het volstaat constanten te beschouwen. Voor een
constante $a$:
\[
\norm{\abs x - a}_\infty = \max\,(1 - a,\ a)
\geq \frac12 ,
\]
geminimaliseerd in $a = \frac12$: de afstand is $\frac12$, bereikt
door de constante $\frac12$. Merk de foutcurve $\abs x - \frac12$
op: zij bereikt $\pm\frac12$ afwisselend in $x = -1, 0, 1$ --- drie
extrema van wisselend teken voor een beste benadering uit een
tweeparameterfamilie. Dat \emph{equioscillatie}patroon is geen
toeval; het is de handtekening van optimaliteit die het
weekendvraagstuk van dit hoofdstuk tot de stelling van Chebyshev
maakt.
\end{example}

\begin{example}[Gesloten versus dichte deelruimten]\label{ex:b2:nvs:closeddense}
In $E = \bigl(C(\intcc{0}{1}), \norm\cdot_\infty\bigr)$: elke
$\R_n[X]$ (polynomen van graad $\leq n$, beperkt tot $\intcc01$) is
een eindigdimensionale, dus \emph{gesloten}, deelruimte --- een
uniforme limiet van polynomen van graad $\leq n$ is er een. Maar de
unie $\R[X]$ van alle is \emph{dicht} in $E$ (de
approximatiestelling van Weierstrass, bewezen in
\cref{ch:b2:funcseq}), en dichte echte deelruimten zijn zo
niet-gesloten als maar kan. De moraal: geslotenheid van deelruimten
is een eindigdimensionaal privilege; gestapelde gesloten verdiepingen
kunnen een dichte wolkenkrabber bouwen.
\end{example}

\begin{theorem}[Riesz]\label{thm:b2:nvs:riesz}
De gesloten eenheidsbol van een genormeerde ruimte $E$ is \omterm{def:b2:metric:compact}{compact}
\emph{als en slechts als} $\dim E < \infty$.\index{stelling van Riesz}
\end{theorem}

\begin{proof}
Eindige dimensie: gesloten en begrensd volstaat
(\cref{thm:b2:nvs:finitedim}).

Omgekeerd, veronderstel $\dim E = \infty$. \emph{Lemma van Riesz:}
voor elke echte gesloten deelruimte $F \subsetneq E$ en
$\varepsilon \in \intoo{0}{1}$ bestaat een eenheidsvector $x$ met
$d(x, F) \geq 1 - \varepsilon$. Bewijs: kies $y \notin F$, zij
$\delta = d(y, F) > 0$ ($F$ gesloten), kies $f \in F$ met $\norm{y
- f} \leq \frac{\delta}{1 - \varepsilon}$, en stel $x =
\frac{y - f}{\norm{y - f}}$: voor elke $g \in F$,
\[
\norm{x - g} = \frac{\norm{y - (f + \norm{y-f}\,g)}}{\norm{y - f}}
\geq \frac{\delta}{\norm{y-f}} \geq 1 - \varepsilon ,
\]
de teller is een afstand van $y$ tot een punt van $F$.

Bouw nu eenheidsvectoren $x_1, x_2, \dots$ inductief: $F_k =
\operatorname{Vect}(x_1, \dots, x_k)$ is eindigdimensionaal, dus
gesloten (\cref{cor:b2:nvs:closedsubspace}) en echt; het lemma van
Riesz met $\varepsilon = \frac12$ levert een eenheid $x_{k+1}$ met
$d(x_{k+1}, F_k) \geq \frac12$. De rij voldoet aan $\norm{x_p -
x_q} \geq \frac12$ voor $p \neq q$: geen convergente deelrij --- de
eenheidsbol is niet \omterm{def:b2:metric:compact}{compact}.
\end{proof}

\begin{example}[Riesz als dimensiedetector]\label{ex:b2:nvs:rieszdetector}
Is $C(\intcc01)$ eindigdimensionaal? Riesz antwoordt zonder een
expliciete oneindige vrije familie te tonen: de rij $f_n(x) = x^n$
ligt in de gesloten eenheidsbol en voldoet, voor $m > n$,
$\norm{f_n - f_m}_\infty \geq f_n(x_0) - f_m(x_0) > 0$ in geschikte
punten --- netjes gekwantificeerd in het weekendvraagstuk van dit
hoofdstuk (vraag 16), waar een deelrij onderlinge afstand $\geq
\frac14$ houdt. Geen convergente deelrij, dus de bol is niet
\omterm{def:b2:metric:compact}{compact}, dus $\dim C(\intcc01) = \infty$ door
\cref{thm:b2:nvs:riesz}. Compactheid van de eenheidsbol is een
perfecte dichotomie: zij geldt in eindige dimensie, faalt in
oneindige dimensie, zonder middenweg --- meetkunde alleen leest het
dimensietype af.
\end{example}

\section{Banachruimten}

\begin{definition}\label{def:b2:nvs:banach}
Een \emph{Banachruimte}\index{Banachruimte} is een volledige
genormeerde ruimte. Voorbeelden: elke eindigdimensionale genormeerde
ruimte (\cref{thm:b2:nvs:finitedim}); $\bigl(C(\intcc{a}{b}),
\norm\cdot_\infty\bigr)$ (\cref{thm:b2:metric:rncomplete});
$\mathcal{L}_c(E, F)$ voor $F$ Banach (zelfde bewijspatroon als voor
continue functies). Niet-voorbeeld: $\bigl(C(\intcc{0}{1}),
\norm\cdot_1\bigr)$ (\cref{exo:b2:nvs:7}).
\end{definition}

\begin{example}[De operatornorm van integratie]\label{ex:b2:nvs:integrationnorm}
Op $\bigl(C(\intcc01), \norm\cdot_\infty\bigr)$, zij $T(f)(x) =
\int_0^x f(t)\,\dd t$ (een endomorfisme: $T(f)$ is continu). Voer
\cref{met:b2:nvs:opnorm} uit. Bovengrens:
\[
\abs{T(f)(x)} \leq \int_0^x\abs f \leq x\,\norm f_\infty \leq
\norm f_\infty ,
\]
dus $\vertiii T \leq 1$. Getuige: $f \equiv 1$ geeft $T(f)(x) = x$
en $\norm{T(f)}_\infty = 1 = \norm f_\infty$: bereikt, $\vertiii T
= 1$. Maar merk $\vertiii{T^2} = \frac12 \neq \vertiii T^2$:
inderdaad heeft $T^2(f)(x) = \int_0^x(x - t)f(t)\dd t$
$\abs{T^2(f)(x)} \leq \frac{x^2}2\norm f_\infty$, opnieuw bereikt
in $f \equiv 1$; en algemeen $\vertiii{T^n} = \frac1{n!}$ --- de
submultiplicatieve grens $\vertiii T^n = 1$ zit een faculteit ernaast.
Dit is precies het fenomeen dat de iterateertruc van het
weekendvraagstuk van \cref{ch:b2:metric} omzet in globale
oplosbaarheid van lineaire differentiaalvergelijkingen.
\end{example}

\begin{theorem}[Absolute convergentie in Banachruimten]\label{thm:b2:nvs:absoluteconvergence}
In een \omterm{def:b2:nvs:banach}{Banachruimte}, als $\sum \norm{u_n} < \infty$ dan convergeert
$\sum u_n$, en $\norm{\sum u_n} \leq \sum\norm{u_n}$. (De volledige
theorie van reeksen in genormeerde ruimten is \cref{ch:b2:series}.)
\end{theorem}

\begin{proof}
Partiële sommen $S_N$: voor $q > p$, $\norm{S_q - S_p} \leq
\sum_{n=p+1}^{q}\norm{u_n}$, wat naar $0$ gaat (Cauchy-criterium
voor de reële reeks van \omterm{def:b2:nvs:norm}{normen}): $(S_N)$ is Cauchy, dus convergent.
De ongelijkheid gaat over naar de limiet vanuit de eindige
driehoeksongelijkheid.
\end{proof}

\begin{example}[Matrixexponentieel, eerste contact]\label{ex:b2:nvs:matrixexp}
Het \emph{matrixexponentieel}\index{matrixexponentieel}:
$\mathcal{M}_n(K)$ met elke submultiplicatieve \omterm{def:b2:nvs:norm}{norm} ($\vertiii{AB}
\leq \vertiii A \vertiii B$) is Banach (eindige dimensie). Dan, voor
elke $A$,
\[
\eu^A = \sum_{k=0}^{\infty} \frac{A^k}{k!}
\]
convergeert absoluut ($\vertiii{A^k/k!} \leq \vertiii A^k /k!$,
sommeerbaar): welgedefinieerd. \cref{ch:b2:diffeq} buit het
systematisch uit.
\end{example}

\begin{example}[Een Neumannreeks die afbreekt]\label{ex:b2:nvs:neumannfinite}
Voor $A = \begin{psmallmatrix}0 & \frac12\\ 0 &
0\end{psmallmatrix}$: $\vertiii A < 1$ in elke \omterm{thm:b2:nvs:continuouslinear}{operatornorm} gebouwd
op de \omterm{def:b2:nvs:norm}{normen} van \cref{ex:b2:nvs:examples}, en $A^2 = 0$, dus
storrt de meetkundige reeks in:
\[
(I - A)^{-1} = \sum_{k \geq 0} A^k = I + A =
\begin{pmatrix}1 & \tfrac12\\ 0 & 1\end{pmatrix},
\]
geverifieerd door $(I - A)(I + A) = I - A^2 = I$. Nilpotentie kapt
de reeks af precies zoals zij het exponentieel kapte in
\cref{ch:b2:reduction}; en het voorbeeld kalibreert verwachtingen:
de Neumann-inverse is in het algemeen een oneindige reeks, een
polynoom precies wanneer de perturbatie nilpotent is, en de fout na
$N$ termen is altijd begrensd door de meetkundige staart $\vertiii
A^{N+1}/(1 - \vertiii A)$.
\end{example}

\begin{example}[Het exponentieel van een rotatiegenerator]\label{ex:b2:nvs:rotationexp}
Neem $A = \begin{psmallmatrix}0 & -\theta\\ \theta &
0\end{psmallmatrix}$. Dan $A^2 = -\theta^2 I$, dus de machten cyclen
met periode vier, en de reeks splitst in even en oneven delen:
\[
\eu^{A} = \sum_{k}\frac{A^k}{k!}
= \Bigl(\sum_{j}\frac{(-1)^j\theta^{2j}}{(2j)!}\Bigr) I
+ \Bigl(\sum_{j}\frac{(-1)^j\theta^{2j+1}}{(2j+1)!}\Bigr)
\frac{A}{\theta}
= \begin{pmatrix}
\cos\theta & -\sin\theta\\
\sin\theta & \cos\theta
\end{pmatrix},
\]
alle herschikkingen gelicentieerd door absolute convergentie. Het
exponentieel van een antisymmetrische generator is een rotatie ---
hier puur uit de reeks berekend, drie hoofdstukken vóór de
differentiaalvergelijking $x' = Ax$ (\cref{ch:b2:diffeq}) uitlegt
\emph{waarom}: $\eu^{tA}$ is uniforme circulaire beweging. Het
sluitende inzicht: identiteiten tussen matrixreeksen worden bewezen
precies als scalaire, zodra een submultiplicatieve \omterm{def:b2:nvs:norm}{norm} absolute
convergentie certificeert.
\end{example}

\begin{example}[Sup-norm betekent uniform: het woordenboek]\label{ex:b2:nvs:uniformdictionary}
De uitspraak $\norm{f_n - f}_\infty \to 0$ \emph{is} uniforme
convergentie: één getal, $\sup_x\abs{f_n(x) - f(x)}$, begrens de
fout in elk punt tegelijk. Het woordenboek in actie op $f_n(x) =
x^n$ over $\intcc{0}{1}$: puntgewijs, $f_n \to 0$ op $\intco{0}{1}$
en $f_n(1) = 1$; in \omterm{def:b2:nvs:norm}{norm}, $\norm{f_n - 0}_\infty = 1 \not\to 0$, en
inderdaad is de puntgewijze limiet discontinu, dus buiten bereik van
een $\norm\cdot_\infty$-limiet in $C(\intcc01)$ (die gesloten is
onder uniforme limieten, \cref{thm:b2:metric:rncomplete}). Op
$\intcc{0}{a}$, $a < 1$: $\norm{f_n}_\infty = a^n \to 0$ ---
uniforme convergentie hersteld door het domein te verkleinen. Elke
convergentie-uitspraak van \cref{ch:b2:funcseq} is een uitspraak
over deze ene \omterm{def:b2:nvs:norm}{norm}; het woordenboek in gedachten houden halveert
dat hoofdstuk.
\end{example}

\begin{remark}[Veelgemaakte valkuilen]\label{rem:b2:nvs:pitfalls}
(i) Een \omterm{thm:b2:nvs:continuouslinear}{operatornorm} hangt af van \emph{beide} gekozen \omterm{def:b2:nvs:norm}{normen}:
dezelfde matrix heeft $\vertiii\cdot_\infty$ gegeven door rijsommen
(\cref{exo:b2:nvs:4}) en een andere $\vertiii\cdot_1$
(kolomsommen); ``de'' \omterm{def:b2:nvs:norm}{norm} van een matrix citeren zonder de
onderliggende \omterm{def:b2:nvs:norm}{normen} te noemen is betekenisloos. (ii) $\vertiii{AB}
\leq \vertiii A\,\vertiii B$ is een ongelijkheid, meestal strikt ---
machten kunnen veel sneller krimpen dan de grens $\vertiii A^k$
suggereert, wat precies het punt van aangepaste \omterm{def:b2:nvs:norm}{normen} is
(weekendvraagstuk van dit hoofdstuk, vraag 22). (iii) ``Lineair
impliceert continu'' is een eindigdimensionaal privilege:
differentiatie op polynomen is lineair en onbegrensd
(\cref{ex:b2:nvs:operatornorms}). (iv) Absolute convergentie van
$\sum u_n$ helpt alleen wanneer de ruimte \omterm{def:b2:metric:complete}{volledig} is
(\cref{exo:b2:series:9} bouwt het tegenvoorbeeld). (v) In
oneindige dimensie is een supremum over de eenheidsbol een echt
supremum: neem niet aan dat het bereikt wordt
(\cref{exo:b2:nvs:8}).
\end{remark}

\begin{remark}[Perspectieven binnen dit volume]\label{rem:b2:nvs:perspectives}
Drie afspraken liggen nu vast. Met \cref{ch:b2:series}: in een
\omterm{def:b2:nvs:banach}{Banachruimte} convergeren absoluut convergente reeksen, dus worden
de meetkundige en exponentiële reeksen van operatoren alledaagse
instrumenten --- $I - A$ inverteren, $\eu^{A}$ definiëren
(\cref{ex:b2:series:neumann}). Met \cref{ch:b2:funcseq} en
\cref{ch:b2:powerseries}: convergentie van functierijen en
machtreeksen is convergentie in $\bigl(C, \norm\cdot_\infty\bigr)$
(\cref{ex:b2:nvs:uniformdictionary}), en de convergentiestraal is
een uitspraak over welke meetkundige reeksen domineren. Met
\cref{ch:b2:fourier}: de \omterm{def:b2:nvs:norm}{normen} $\norm\cdot_2$ en
$\norm\cdot_\infty$ zijn het oprecht oneens op $C(\intcc{0}{1})$
(\cref{ex:b2:nvs:sqrtnxn}), wat precies is waarom
kwadratische-middelconvergentie van Fourierreeksen en uniforme
convergentie twee verschillende stellingen met twee verschillende
prijzen zijn.
\end{remark}

\begin{remark}[Waar dit hoofdstuk gebruikt wordt]
\omterm{thm:b2:nvs:continuouslinear}{Operatornormen} en de meetkundige reeks drijven de
perturbatieargumenten van \cref{ch:b2:diffcalc}
(inversefunctiestelling) en het \omterm{ex:b2:nvs:matrixexp}{matrixexponentieel} van
\cref{ch:b2:diffeq}; \omterm{thm:b2:nvs:finitedim}{equivalentie van normen} machtigt stilzwijgend
elk ``kies je favoriete \omterm{def:b2:nvs:norm}{norm}''-argument in \cref{ch:b2:funcseq} en
verder; en de kloof eindig/oneindig van de stelling van Riesz ---
kwantitatief gemaakt in het weekendvraagstuk van dit hoofdstuk ---
is de reden waarom het volume van Jaar~3 nieuwe instrumenten nodig
heeft (zwakke convergentie, Arzelà--Ascoli, Hilbertprojecties)
waar dit volume nog convergente deelrijen kon extraheren.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b2:nvs:1}
Teken op $\R^2$ de eenheidsbollen van $\norm\cdot_1$,
$\norm\cdot_2$, $\norm\cdot_\infty$, en bewijs de ongelijkheden
$\norm x_\infty \leq \norm x_2 \leq \norm x_1 \leq 2\norm
x_\infty$ met de beste constanten in dimensie $2$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:nvs:2}
Is $N(f) = \abs{f(0)} + \norm{f'}_\infty$ een \omterm{def:b2:nvs:norm}{norm} op
$C^1(\intcc{0}{1})$? Vergelijk haar met $\norm{f}_\infty$: één
ongelijkheid geldt, de andere faalt (toon).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:nvs:3}
Bereken de \omterm{thm:b2:nvs:continuouslinear}{operatornorm} van $u(f) = \int_0^1 f(t)\,\eu^t\,\dd t$ op
$\bigl(C(\intcc{0}{1}), \norm\cdot_\infty\bigr) \to \R$, en van de
verschuiving $S(x_1, x_2, \dots, x_n) = (x_2, \dots, x_n, 0)$ op
$(K^n, \norm\cdot_\infty)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:nvs:4}
Bewijs op $(\R^n, \norm\cdot_\infty)$ dat de \omterm{thm:b2:nvs:continuouslinear}{operatornorm} van een
matrix $A$ is $\vertiii A_\infty = \max_i \sum_j \abs{a_{ij}}$ (de
grootste absolute rijsom). Bereken haar voor $\begin{pmatrix} 1 &
-2\\ 3 & 1\end{pmatrix}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:nvs:5}
Bewijs dat $GL_n(K)$ open is in $\mathcal{M}_n(K)$ en dat $A
\mapsto A^{-1}$ continu is erop. \emph{Hint: voor openheid, als
$\vertiii H < \frac{1}{\vertiii{A^{-1}}}$ dan $A + H = A(I +
A^{-1}H)$ met $\vertiii{A^{-1}H} < 1$, en $I + B$ is inverteerbaar
voor $\vertiii B < 1$ door de meetkundige reeks
(\cref{thm:b2:nvs:absoluteconvergence}); voor continuïteit, begrens
$(A+H)^{-1} - A^{-1}$ met dezelfde reeks.}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:nvs:6}
Zij $\varphi$ een lineaire vorm op een genormeerde ruimte $E$.
Bewijs dat $\varphi$ continu is als en slechts als $\ker\varphi$
gesloten is. \emph{(Als $\ker\varphi$ gesloten is en $\varphi \neq
0$, kies $a$ met $\varphi(a) = 1$ en $r > 0$ met $B(a, r) \cap
\ker\varphi = \emptyset$; leid $\abs{\varphi(h)} \leq
\frac{1}{r}\norm h$ af door een schalingsargument op $a -
\frac{h}{\varphi(h)}$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:nvs:7}
Bewijs dat $\bigl(C(\intcc{0}{1}), \norm\cdot_1\bigr)$ niet \omterm{def:b2:metric:complete}{volledig}
is: toon dat de functies $f_n$, affiene hellingen van $0$ naar $1$
over $\bigl[\frac12 - \frac1n, \frac12\bigr]$ (waarde $0$ ervoor,
$1$ erna), een Cauchyrij vormen zonder continue
$\norm\cdot_1$-limiet.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:nvs:8}
Op $E = C(\intcc{0}{1})$ met $\norm\cdot_\infty$, beschouw
\[
\varphi(f) = \sum_{n \geq 1} (-1)^n\, 2^{-n} f\bigl(\tfrac1n\bigr).
\]
Bewijs dat $\varphi$ een welgedefinieerde continue lineaire vorm is
met $\vertiii\varphi = 1$, maar dat het supremum dat
$\vertiii\varphi$ definieert \emph{niet bereikt} wordt op de
gesloten eenheidsbol. \emph{(Bovengrens: driehoeksongelijkheid.
\omterm{def:b2:nvs:norm}{Norm} $= 1$: bouw continue $f_K$ met $\norm{f_K}_\infty \leq 1$ en
$f_K(\frac1n) = (-1)^n$ voor $n \leq K$ --- de punten $\frac1n$
zijn van elkaar geïsoleerd. Niet-bereiking: gelijkheid zou
$f(\frac1n) = (-1)^n$ forceren voor elke $n$, onverenigbaar met
continuïteit van $f$ in $0$ aangezien $\frac1n \to 0$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:nvs:9}
Zij $E$ een genormeerde ruimte waarin de gesloten eenheidsbol
\omterm{def:b2:metric:compact}{compact} is. Leid opnieuw, zonder \cref{thm:b2:nvs:riesz} te
citeren, af dat elke begrensde rij een convergente deelrij heeft, en
bewijs dat elke lineaire vorm op $E$ continu is als en slechts als
$\dim E < \infty$. \emph{(Voor oneindige dimensie, bouw een
discontinue vorm door haar vrij te definiëren op een lineair
onafhankelijke genormaliseerde rij en uit te breiden --- met
toegeving van het bestaan van een algebraïsch complement.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:nvs:10}
Bewijs op $C(\intcc{0}{1})$ dat $\norm f_1 \leq \norm f_2 \leq
\norm f_\infty$ \emph{(Cauchy--Schwarz voor de eerste)}, en toon met
de familie $f_n(x) = x^n$ dat geen van beide ongelijkheden tot op een
constante omkeerbaar is: de drie \omterm{def:b2:nvs:norm}{normen} zijn paarsgewijs
\omterm{def:b2:nvs:equivalent}{niet-equivalent}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:nvs:11}
(Afstand tot een hypervlak) Zij $\varphi$ een niet-nulle continue
lineaire vorm op een genormeerde ruimte $E$. Bewijs dat
\[
d\bigl(x, \ker\varphi\bigr) =
\frac{\abs{\varphi(x)}}{\vertiii\varphi}
\qquad (x \in E),
\]
en controleer op \cref{exo:b2:nvs:8} dat het infimum niet door enig
punt van het hypervlak hoeft te worden bereikt.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:nvs:12}
Op $E = \R[X]$ (alle polynomen), zij $N_1(P) =
\sup_{\intcc{0}{1}}\abs P$ en $N_2(P) =
\sup_{\intcc{0}{2}}\abs P$. Toon dat $N_1 \leq N_2$ maar dat $N_1$
en $N_2$ \emph{niet} \omterm{def:b2:nvs:equivalent}{equivalent} zijn; leid af dat de identiteit
$(E, N_2) \to (E, N_1)$ een continue lineaire bijectie is waarvan
de inverse discontinu is. Toon ten slotte dat $(E, N_1)$ niet
\omterm{def:b2:metric:complete}{volledig} is \emph{(Taylor-partiële sommen van $\eu^x$)}. Alle drie
de fenomenen zijn onmogelijk in eindige dimensie --- zeg waarom.
\end{exercise}

\section{Vraagstuk: Beste benadering en de stelling van Chebyshev}

Hoe goed kan een functie benaderd worden door polynomen van gegeven
graad, en welk polynoom doet het het best? Aan de bestaanskant
hoort het antwoord bij dit hoofdstuk: compactheid in eindige
dimensie doet beste benaderingen bestaan. Aan de expliciete kant
kan één niet-triviaal geval met de blote hand \omterm{def:b2:metric:complete}{volledig} opgelost
worden --- onder alle \emph{monische} polynomen van graad $n$ is
dat van kleinste \omterm{def:b2:nvs:norm}{sup-norm} op $\intcc{-1}{1}$ het
(genormaliseerde) Chebyshevpolynoom, van \omterm{def:b2:nvs:norm}{norm} $2^{1-n}$: de
\emph{extremale stelling van Chebyshev}. Het vraagstuk bewijst beide
kanten, en meet dan hoe erg compactheid faalt in oneindige
dimensie: de eenheidsbol van $C(\intcc{0}{1})$ bevat oneindige
constellaties van punten op onderlinge afstand $1$.

\begin{problem}[{Weekendvraagstuk --- de extremale stelling van Chebyshev
en de meetkunde van de eenheidsbol}]
\label{pb:b2:nvs:1}
\omterm{def:b2:nvs:norm}{Normen} zonder subscript zijn \omterm{def:b2:nvs:norm}{sup-normen} op het aangegeven segment.

\medskip
\noindent\textbf{Deel I --- Beste benadering in genormeerde
ruimten.}
\begin{enumerate}
  \item Zij $F$ een eindigdimensionale deelruimte van een
        genormeerde ruimte $E$ en $x \in E$. Bewijs dat de afstand
        $d(x, F) = \inf_{f \in F}\norm{x - f}$ \emph{bereikt}
        wordt \emph{(herleid tot een gesloten begrensde
        deelverzameling van $F$ en gebruik
        \cref{thm:b2:nvs:finitedim})}.
  \item Een \omterm{def:b2:nvs:norm}{norm} is \emph{strikt convex} wanneer $\norm u = \norm
        v = 1$ en $u \neq v$ impliceren
        $\bigl\Vert\frac{u + v}2\bigr\Vert < 1$. Toon dat
        $\norm\cdot_2$ op $\R^n$ strikt convex is
        \emph{(parallellogramidentiteit)}, en dat $\norm\cdot_1$
        en $\norm\cdot_\infty$ dat niet zijn voor $n \geq 2$.
  \item Bewijs dat voor een strikt convexe \omterm{def:b2:nvs:norm}{norm} de beste
        benadering van vraag 1 \emph{uniek} is.
  \item Bereken in $(\R^2, \norm\cdot_\infty)$ alle beste
        benaderingen van $x = (0, 1)$ door de lijn $F =
        \operatorname{Vect}\bigl((1,0)\bigr)$: een interval van
        minimizers.
  \item Toon in $\bigl(C(\intcc{a}{b}), \norm\cdot_\infty\bigr)$
        dat de beste benadering van $f$ door \emph{constanten}
        uniek is, gelijk aan $c^* = \frac{\max f + \min f}{2}$,
        met afstand $\frac{\max f - \min f}{2}$; bereken beide
        voor $f(x) = x^2$ op $\intcc{0}{1}$.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Chebyshevpolynomen.}
\begin{enumerate}[resume]
  \item Toon dat er precies één polynoom $T_n$ is met
        $T_n(\cos\theta) = \cos n\theta$ voor alle $\theta$
        \emph{(recurrentie $T_{n+1} = 2XT_n - T_{n-1}$ uit de
        cosinusoptelformule)}, dat $\deg T_n = n$, en dat haar
        leidende coëfficiënt $2^{n-1}$ is voor $n \geq 1$.
  \item Toon $\abs{T_n} \leq 1$ op $\intcc{-1}{1}$, met
        $T_n(\eta_k) = (-1)^k$ in de $n + 1$ punten $\eta_k =
        \cos\frac{k\pi}{n}$ ($k = 0, \dots, n$), en dat de
        wortels van $T_n$ de $n$ punten
        $\cos\frac{(2k-1)\pi}{2n}$ zijn, die de $\eta_k$
        interliniëren.
  \item Bereken $T_2, T_3, T_4$, en verifieer de alternatie van
        $T_3$ in $\eta_0, \dots, \eta_3 = 1, \frac12, -\frac12,
        -1$ door directe evaluatie.
  \item Bewijs voor $\abs x \geq 1$
        \[
        T_n(x) = \frac{\bigl(x + \sqrt{x^2 - 1}\bigr)^n +
        \bigl(x - \sqrt{x^2 - 1}\bigr)^n}{2},
        \]
        en leid $T_n(x) \sim \frac12\bigl(x + \sqrt{x^2 -
        1}\bigr)^n \to \infty$ meetkundig af voor vaste $x > 1$.
  \item Bewijs de compositiewet $T_m \circ T_n = T_{mn}$
        \emph{(controleer op $\intcc{-1}{1}$ en roep de
        rigiditeit van polynomen in)}.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De extremale stelling van Chebyshev.}
Schrijf $Q_n = 2^{1-n}T_n$ (monisch, door vraag 6).
\begin{enumerate}[resume]
  \item Zij $P$ monisch van graad $n \geq 1$ met
        $\sup_{\intcc{-1}{1}}\abs P < 2^{1-n}$. Door $D = Q_n -
        P$ te evalueren in de punten $\eta_k$ en tekenwisselingen
        te tellen, leid een tegenspraak af. Besluit:
        \[
        \sup_{\intcc{-1}{1}}\abs P \;\geq\; 2^{1-n}
        \qquad\text{voor elke monische } P \text{ van graad } n.
        \]
  \item (Gelijkheidsgeval) Veronderstel $\sup_{\intcc{-1}{1}}\abs
        P = 2^{1-n}$ met $P$ monisch van graad $n$, en zij $D =
        Q_n - P \neq 0$. Toon $(-1)^kD(\eta_k) \geq 0$ voor alle
        $k$; toon dat elk van de $n$ intervallen
        $\intcc{\eta_{k}}{\eta_{k-1}}$ een nulpunt van $D$ bevat,
        en dat een nulpunt gedeeld door twee opeenvolgende
        intervallen een \emph{inwendig} punt $\eta_k$ is waar ook
        $D' = 0$. Besluit dat $D$ $n$ nulpunten heeft geteld met
        multipliciteit, dus $D = 0$: de minimizer is precies $Q_n$
        --- de \emph{extremale stelling van Chebyshev}.
  \item Herformuleer de stelling als een afstand: op
        $\intcc{-1}{1}$,
        \[
        d_\infty\bigl(X^n,\ \R_{n-1}[X]\bigr) = 2^{1-n},
        \]
        met unieke beste benadering $X^n - Q_n$; en toon door de
        affiene substitutie $x = \frac{1+t}2$ dat op $\intcc{0}{1}$
        de afstand $2^{1-2n}$ wordt.
  \item (Optimale interpolatieknopen) Voor $n$ knopen $x_1, \dots,
        x_n \in \intcc{-1}{1}$ is het knooppolynoom $\omega(x) =
        \prod_i(x - x_i)$ monisch van graad $n$. Leid uit vraag 12
        af welke knopenkeuze $\sup_{\intcc{-1}{1}}\abs\omega$
        minimaliseert, de knoopafhankelijke factor van de
        klassieke interpolatiefoutgrens, en geef de minimale
        waarde.
  \item Verifieer het geval $n = 2$ van de stelling met de hand
        (vind $\inf_c \sup_{\intcc{-1}{1}}\abs{x^2 - c}$ direct),
        en bereken numeriek de afstand van vraag 13 op
        $\intcc{0}{1}$ voor $n = 10$. Wat zegt haar grootte over
        de grafiek van $x^{10}$?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De eenheidsbol van
$C(\intcc{0}{1})$.}
\begin{enumerate}[resume]
  \item Zij $g_k(x) = x^{2^k}$. Toon $\norm{g_k}_\infty = 1$ en
        $\norm{g_k - g_j}_\infty \geq \frac14$ voor $j > k$
        \emph{(evalueer in het punt waar $x^{2^k} = \frac12$)}:
        een expliciete begrensde rij zonder convergente deelrij
        --- de gesloten eenheidsbol is niet \omterm{def:b2:metric:compact}{compact}, met de blote
        hand.
  \item (Lemma van Riesz, aangescherpt) Zij $F$ een
        \emph{eindigdimensionale} echte deelruimte van een
        genormeerde ruimte $E$. Produceer met vraag 1 een
        eenheidsvector $x$ met $d(x, F) = 1$ precies --- niet
        slechts $\geq 1 - \varepsilon$ zoals in het lemma van
        \cref{thm:b2:nvs:riesz}.
  \item Leid af: in elke oneindigdimensionale genormeerde ruimte
        bestaat een rij eenheidsvectoren met paarsgewijze
        afstanden $\geq 1$, en herleid de stelling van Riesz
        eruit.
  \item Exhibeer in $C(\intcc{0}{1})$ zo'n constellatie expliciet:
        de tentfuncties $h_n$ met steun op $\bigl[\frac1{n+1},
        \frac1n\bigr]$ en piekwaarde $1$. Verifieer $\norm{h_n} =
        1$, $\norm{h_n - h_m} = 1$ voor $n \neq m$, en merk op dat
        $h_n \to 0$ puntgewijs maar niet uniform.
  \item (Totale begrensdheid faalt) Toon dat de gesloten
        eenheidsbol van $C(\intcc{0}{1})$ niet overdekt kan worden
        door eindig veel bollen van straal $\frac13$ \emph{(elke
        zulke bol bevat ten hoogste één $h_n$)} --- contrasteer
        met de totale-begrensdheidsstap in het bewijs van
        \cref{thm:b2:metric:borellebesgue}.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- \omterm{def:b2:nvs:norm}{Normen} aan het werk op matrices, en
synthese.}
\begin{enumerate}[resume]
  \item Bewijs dat elke \omterm{def:b2:reduction:eigen}{eigenwaarde} $\lambda$ van $A \in
        \mathcal{M}_n(\C)$ voldoet aan $\abs\lambda \leq
        \vertiii A$ voor elke \omterm{thm:b2:nvs:continuouslinear}{operatornorm}; pas
        \cref{exo:b2:nvs:4} toe om de \omterm{def:b2:reduction:eigen}{eigenwaarden} van
        $\begin{psmallmatrix}1 & -2\\ 3 & 1\end{psmallmatrix}$ te
        begrenzen en vergelijk met hun ware modulus.
  \item (Aangepaste \omterm{def:b2:nvs:norm}{normen}) Zij $A$ \omterm{def:b2:reduction:diag}{diagonaliseerbaar}, $A =
        P\,\mathrm{diag}(\lambda_1, \dots, \lambda_n)\,P^{-1}$.
        Toon dat $N_P(x) = \norm{P^{-1}x}_\infty$ een \omterm{def:b2:nvs:norm}{norm} is
        waarvan de \omterm{thm:b2:nvs:continuouslinear}{operatornorm} voldoet aan $\vertiii A_{N_P} =
        \max_i\abs{\lambda_i}$.
  \item Leid af: voor diagonaliseerbare $A$, $A^k \to 0$ als en
        slechts als alle \omterm{def:b2:reduction:eigen}{eigenwaarden} $\abs{\lambda_i} < 1$
        voldoen --- \omterm{thm:b2:nvs:finitedim}{equivalentie van normen} maakt de conclusie
        normonafhankelijk. Controleer op $A =
        \frac14\begin{psmallmatrix}1 & 2\\ 2 &
        1\end{psmallmatrix}$.
  \item (Equivalentieconstanten blazen op) Vergelijk op $\R_n[X]$
        $N_c(P) = \max_k \abs{a_k}$ (coëfficiënten) en
        $\norm{P}_{\intcc{0}{1}}$: beide zijn \omterm{def:b2:nvs:norm}{normen}, dus
        \omterm{def:b2:nvs:equivalent}{equivalent} voor elke vaste $n$; maar toon, met de monische
        minimizer van vraag 13 op $\intcc{0}{1}$, dat de beste
        constante $C_n$ in $N_c \leq
        C_n\norm\cdot_{\intcc{0}{1}}$ voldoet aan $C_n \geq
        2^{2n-1}$. Besluit in één zin waarom ``alle \omterm{def:b2:nvs:norm}{normen} zijn
        \omterm{def:b2:nvs:equivalent}{equivalent}'' sterft in oneindige dimensie.
  \item (Synthese) Eén zin elk: waar compactheid van
        eindigdimensionale bollen werkte (vragen 1, 12); wat
        strikte convexiteit bestuurt; wat de constellatie van
        vragen 18--19 vernietigt; en hoe vraag 24 het falen
        kwantificeert. Noem de top (de extremale stelling van
        Chebyshev) en zeg waar beste benadering haar moderne
        thuis vindt (de projectiestelling op Hilbertruimten,
        volume van Jaar~3, waar volledigheid compactheid
        vervangt).
\end{enumerate}
\end{problem}
