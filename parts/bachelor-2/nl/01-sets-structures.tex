\chapter{Verzamelingen en structuren}\label{ch:b2:structures}

Dit openingshoofdstuk slijpt de fundamenten uit het Bachelor
jaar~1-volume tot werkinstrumenten van het vak: de calculus van
verzamelingen en quotiënten, de vergelijking van oneindige
verzamelingen (aftelbaarheid, Cantor--Bernstein), en de
structurele theorie van groepen en ringen --- de stelling van
Lagrange, de symmetrische groep met haar signatuur, idealen en
de Chinese reststelling. Alles hier wordt onophoudelijk gebruikt
in de rest van het boek: de signatuur bouwt de determinant
(\cref{ch:b2:linalg}), \omterm{def:b2:structures:quotientring}{quotiëntringen} drijven de rekenkunde, en
aftelbaarheid ligt ten grondslag aan zowel topologie als
waarschijnlijkheidsrekening.

\section{Verzamelingen, afbeeldingen, quotiënten}

We gebruiken vrijelijk de taal van verzamelingen, afbeeldingen en
equivalentie- en orderelaties die in het Bachelor jaar~1-volume
is opgezet. Twee verfijningen verdienen een ordentelijke uitspraak.

\begin{proposition}[Beelden en originelen van families]\label{prop:b2:structures:images}
Zij $f \colon E \to F$ en zij $(A_i)_{i \in I}$, $(B_j)_{j \in J}$
families van deelverzamelingen van $E$, resp.\ $F$. Dan
\[
f^{-1}\Bigl(\bigcup_j B_j\Bigr) = \bigcup_j f^{-1}(B_j),
\qquad
f^{-1}\Bigl(\bigcap_j B_j\Bigr) = \bigcap_j f^{-1}(B_j),
\qquad
f^{-1}(F \setminus B) = E \setminus f^{-1}(B),
\]
\[
f\Bigl(\bigcup_i A_i\Bigr) = \bigcup_i f(A_i),
\qquad
f\Bigl(\bigcap_i A_i\Bigr) \subseteq \bigcap_i f(A_i)
\quad (\text{gelijkheid voor injectieve } f).
\]
\end{proposition}

\begin{proof}
Elke identiteit is een uitwerking van de definities; bijvoorbeeld
$x \in f^{-1}(\bigcap B_j) \iff f(x) \in B_j$ voor alle $j$
$\iff x \in f^{-1}(B_j)$ voor alle $j$. De beeldidentiteiten en
het falen van de gelijkheid in het doorsnedengeval (met de
injectiviteitscorrectie) zijn in het Bachelor jaar~1-volume
bewezen voor twee verzamelingen; de argumenten zijn identiek
voor families.
\end{proof}

\begin{example}[Waar de beeldinclusie strikt is]\label{ex:b2:structures:imagestrict}
Neem $f \colon \R \to \R$, $f(x) = x^2$, met $A_1 =
\intcc{-1}{0}$ en $A_2 = \intcc{0}{1}$. Dan
\[
f(A_1 \cap A_2) = f(\{0\}) = \{0\},
\qquad
f(A_1) \cap f(A_2) = \intcc{0}{1} \cap \intcc{0}{1} =
\intcc{0}{1} :
\]
de inclusie van \cref{prop:b2:structures:images} is zo strikt
als maar kan --- de twee originelen $\pm x$ van een gemeenschappelijke
waarde leven in verschillende $A_i$. Injectiviteit is precies wat
deze splitsing verbiedt, en daarom voldoen originelen (die nooit
punten samenvoegen) onvoorwaardelijk aan alle vier de
identiteiten, terwijl beelden die over doorsneden verliezen.
Vuistregel voor het hele boek: duw \emph{originelen} vrijelijk
door verzamelingsoperaties; behandel beelden met zorg.
\end{example}

\begin{definition}[Quotiëntverzameling]\label{def:b2:structures:quotient}
Zij $\mathcal{R}$ een equivalentierelatie op $E$. De
\emph{quotiëntverzameling}\index{quotiëntverzameling} $E/\mathcal{R}$ is de verzameling van
equivalentieklassen; de surjectie $\pi \colon E \to E/\mathcal{R}$,
$x \mapsto \mathrm{cl}(x)$, is de \emph{canonieke
projectie}\index{canonieke projectie}.

\emph{Universele eigenschap (factorisatie):} als $f \colon E \to F$
\emph{compatibel} is met $\mathcal{R}$ (d.w.z.\ $x \mathbin{\mathcal{R}}
y \implies f(x) = f(y)$), bestaat er precies één afbeelding $\overline f
\colon E/\mathcal{R} \to F$ met $f = \overline f \circ \pi$.
\end{definition}

\begin{proof}[Bewijs van de universele eigenschap]
Uniciteit: de eis $f = \overline f \circ \pi$ luidt
\[
\overline f\bigl(\mathrm{cl}(x)\bigr) = f(x)
\qquad (x \in E),
\]
en omdat $\pi$ surjectief is, is elk element van
$E/\mathcal{R}$ van de vorm $\mathrm{cl}(x)$: de waarden van
$\overline f$ zijn alle gedwongen. Bestaan: neem de weergave als
de \emph{definitie} van $\overline f$; ze is eenduidig precies
door compatibiliteit --- als $\mathrm{cl}(x) =
\mathrm{cl}(y)$, dan $x \mathbin{\mathcal{R}} y$, dus $f(x) =
f(y)$ en de twee kandidaatwaarden stemmen overeen --- en ze
factoriseert $f$ per constructie. Merk de arbeidsverdeling op:
surjectiviteit van $\pi$ geeft uniciteit, compatibiliteit geeft
bestaan.
\end{proof}

\begin{example}
$\Z/n\Z$ is het quotiënt van $\Z$ door congruentie modulo $n$; de
welgedefinieerdheidscontroles van het Bachelor jaar~1-volume
waren gevallen van de universele eigenschap. Quotiënten maken van
``compatibele constructies op representanten'' eerlijke
afbeeldingen --- we gebruiken dit voortdurend hieronder.
\end{example}

\section{Aftelbaarheid en kardinaliteit}

\begin{definition}[Gelijkmachtigheid, aftelbaarheid]\label{def:b2:structures:countable}
Twee verzamelingen zijn \emph{gelijkmachtig}\index{gelijkmachtigheid} wanneer een bijectie
ze verbindt. Een verzameling is \emph{aftelbaar}\index{aftelbare verzameling} wanneer ze
gelijkmachtig is met $\N$ (sommige auteurs nemen eindige
verzamelingen mee; wij zeggen
\emph{hoogstens aftelbaar} voor ``eindig of aftelbaar'').
\end{definition}

\begin{proposition}[Stabiliteitseigenschappen]\label{prop:b2:structures:countablestable}
\begin{enumerate}
  \item Elke oneindige deelverzameling van $\N$ is \omterm{def:b2:structures:countable}{aftelbaar}; een
        verzameling is hoogstens \omterm{def:b2:structures:countable}{aftelbaar} desda ze injecteert in
        $\N$ desda ze leeg is of een surjectief beeld van $\N$.
  \item $\N \times \N$ is \omterm{def:b2:structures:countable}{aftelbaar}; een product van twee hoogstens
        \omterm{def:b2:structures:countable}{aftelbare verzamelingen} is hoogstens \omterm{def:b2:structures:countable}{aftelbaar}.
  \item Een hoogstens aftelbare unie van hoogstens \omterm{def:b2:structures:countable}{aftelbare
        verzamelingen} is hoogstens \omterm{def:b2:structures:countable}{aftelbaar}.
  \item $\Z$ en $\Q$ zijn \omterm{def:b2:structures:countable}{aftelbaar}.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) Sorteer een oneindige $A \subseteq \N$ door herhaalde minima:
$a_0 = \min A$, $a_{k+1} = \min\,(A \setminus \{a_0, \dots, a_k\})$
(niet-leeg omdat $A$ oneindig is); de afbeelding $k \mapsto a_k$ is
strikt stijgend, injectief, en surjectief op $A$ (elke $a \in A$
overtreft slechts eindig veel elementen van $A$, dus wordt
bereikt). Als $E$ injecteert in $\N$ via $\varphi$, is $E$
\omterm{def:b2:structures:countable}{gelijkmachtig} met $\varphi(E) \subseteq \N$: eindig of \omterm{def:b2:structures:countable}{aftelbaar}.
Als $s \colon \N \to E$ surjectief is, injecteert $x \mapsto \min
s^{-1}(\{x\})$ $E$ in $\N$.

(2) De afbeelding $(p, q) \mapsto 2^p(2q + 1) - 1$ is een bijectie
$\N^2 \to \N$ (elk positief geheel getal heeft een unieke
oneven--even-splitsing $2^p m$ met $m$ oneven, door unieke
factorisatie). Producten: stel injecties samen.

(3) Gegeven verzamelingen $E_n$ met surjecties $s_n \colon \N \to
E_n$ (onschadelijk als sommige $E_n$ eindig is: herhaal waarden),
is de afbeelding $(n, k) \mapsto s_n(k)$ een surjectie van de
aftelbare $\N^2$ op $\bigcup E_n$.

(4) $\Z = \N \cup (-\N^*)$: aftelbare unie. $\Q$ is een surjectief
beeld van $\Z \times \N^*$ (de breukafbeelding), dus hoogstens
\omterm{def:b2:structures:countable}{aftelbaar}, en oneindig.
\end{proof}

\begin{example}[Een paarfunctie, uitgewerkt]\label{ex:b2:structures:pairing}
De bijectie $(p, q) \mapsto 2^p(2q + 1) - 1$ uit het bewijs
verdient het om aan het werk gezien te worden. Haar eerste waarden:
\[
\begin{array}{c|ccccc}
 & q = 0 & q = 1 & q = 2 & q = 3 & q = 4\\
\hline
p = 0 & 0 & 2 & 4 & 6 & 8\\
p = 1 & 1 & 5 & 9 & 13 & 17\\
p = 2 & 3 & 11 & 19 & 27 & 35\\
p = 3 & 7 & 23 & 39 & 55 & 71
\end{array}
\]
Rij $p$ verzamelt de gehele getallen $n$ waarvoor $n + 1$ precies
deelbaar is door $2^p$: elk natuurlijk getal verschijnt precies
één keer. Decodering is even expliciet als codering: voor $n =
43$, factoriseer $n + 1 = 44 = 2^2\cdot 11 = 2^2(2\cdot5 + 1)$,
dus $(p, q) = (2, 5)$. Het sluitende inzicht: bewijzen van
aftelbaarheid zijn vaak \emph{algoritmen} in vermomming --- hier,
``factoriseer de tweeën uit''.
\end{example}

\begin{example}[De algebraïsche getallen zijn aftelbaar]\label{ex:b2:structures:algebraic}
Een complex getal is \emph{algebraïsch} wanneer het een
niet-nulle veelterm met rationale coëfficiënten annuleert. De
verzameling $\overline\Q$ van algebraïsche getallen is \omterm{def:b2:structures:countable}{aftelbaar}:
veeltermen van graad $\leq d$ over $\Q$ injecteren in $\Q^{d+1}$,
een eindig product van \omterm{def:b2:structures:countable}{aftelbare verzamelingen}
(\cref{prop:b2:structures:countablestable}~(2)); de unie over
$d$ enumereert de niet-nulle rationale veeltermen als $P_0, P_1,
P_2, \dots$; elke $P_k$ heeft eindig veel wortels; en
\[
\overline\Q = \bigcup_{k \in \N}\ \{\text{wortels van } P_k\}
\]
is een aftelbare unie van eindige verzamelingen
(\cref{prop:b2:structures:countablestable}~(3)), oneindig
aangezien ze $\Q$ bevat. Gecombineerd met de overaftelbaarheid
van $\R$ (\cref{thm:b2:structures:cantor} hieronder) bewijst dit
--- zonder er een enkele te tonen --- dat transcendentale
getallen bestaan en een overaftelbare meerderheid vormen:
Cantors telargument van 1874, bestaan louter door kardinaliteit.
\end{example}

\begin{theorem}[Cantor; overaftelbaarheid van $\R$]\label{thm:b2:structures:cantor}
\begin{enumerate}
  \item Voor elke verzameling $E$ bestaat er geen surjectie $E \to
        \mathcal{P}(E)$.
  \item $\R$ is \emph{niet} \omterm{def:b2:structures:countable}{aftelbaar}.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) is bewezen in het Bachelor jaar~1-volume (de
diagonaalverzameling $D = \{x : x \notin f(x)\}$).

(2) Stel dat $(x_n)_{n \in \N}$ $\R$ enumereert. Bouw geneste
segmenten $I_0 \supseteq I_1 \supseteq \dots$ met $\abs{I_n} =
3^{-n}$ en $x_n \notin I_n$: splits het huidige segment in drie
gesloten derden; minstens één derde vermijdt $x_n$ (een punt
ontmoet hoogstens twee van de drie). De geneste-segmentenstelling
(aangrenzende eindpunten) levert $\ell \in \bigcap_n I_n$; maar
$\ell = x_N$ voor zekere $N$, en $x_N \notin I_N$: contradictie.
\end{proof}

\begin{theorem}[Cantor--Bernstein]\label{thm:b2:structures:cantorbernstein}
Als $E$ injecteert in $F$ en $F$ injecteert in $E$, dan zijn $E$ en
$F$ \omterm{def:b2:structures:countable}{gelijkmachtig}.\index{stelling van Cantor--Bernstein}
\end{theorem}

\begin{proof}
Zij $f \colon E \to F$ en $g \colon F \to E$ injecties. Voor elk
punt (van $E$ of $F$), volg zijn \emph{voorouderketen} van
opeenvolgende originelen, $x \mapsto g^{-1}(x) \mapsto
f^{-1}(g^{-1}(x)) \mapsto \dots$ --- elke stap is gedefinieerd
zolang het huidige punt in het beeld van de relevante injectie
ligt, en is dan uniek door injectiviteit. Drie wederzijds
uitsluitende lotgevallen: de keten stopt bij een punt van $E
\setminus g(F)$ (\emph{oorsprong in $E$}), stopt bij een punt van
$F \setminus f(E)$ (\emph{oorsprong in $F$}), of stopt nooit. Dit
partitioneert $E = E_E \cup E_F \cup E_\infty$ en $F = F_E \cup
F_F \cup F_\infty$ naar oorsprong.

Merk nu op: $f$ beeldt $E_E$ \emph{op} $F_E$ af --- de keten van
$f(x)$ is de keten van $x$ voorafgegaan door één stap, dus
oorsprongen matchen; en elke $y \in F_E$ heeft een keten met
minstens één stap (haar oorsprong ligt in $E$), dus $y = f(x)$
met $x \in E_E$. Hetzelfde argument levert bijecties $f \colon
E_\infty \to F_\infty$ en $g \colon F_F \to E_F$. Samenplakken,
\[
h(x) =
\begin{cases}
f(x) & \text{als } x \in E_E \cup E_\infty,\\
g^{-1}(x) & \text{als } x \in E_F,
\end{cases}
\]
is een bijectie van $E$ op $F = F_E \cup F_\infty \cup F_F$: ze is
stuksgewijs bijectief, en de drie doelen zijn disjunct.
\end{proof}

\begin{example}\label{ex:b2:structures:cbexample}
$\intoo{0}{1}$ en $\intcc{0}{1}$ zijn \omterm{def:b2:structures:countable}{gelijkmachtig}: de
identiteit injecteert de ene kant op, $x \mapsto \frac{x + 1}{3}$
de andere; de stelling fabriceert de (noodzakelijk discontinue)
bijectie. Evenzo zijn $\R$, $\intoo{0}{1}$ (via $\tanh$-achtige
bijecties) en $\mathcal{P}(\N)$ (binaire ontwikkelingen,
\cref{exo:b2:structures:3}) alle \omterm{def:b2:structures:countable}{gelijkmachtig}: ``de
kardinaliteit van het continuüm''.
\end{example}

\begin{example}[Het segment en het vierkant]\label{ex:b2:structures:squareseg}
$\intcc{0}{1}$ en $\intcc{0}{1}^2$ zijn \omterm{def:b2:structures:countable}{gelijkmachtig} --- dimensie
is onzichtbaar voor kardinaliteit. Eén injectie is triviaal: $x
\mapsto (x, 0)$. Voor de andere, stuur $(x, y)$ naar het reële
getal waarvan de decimale cijfers die van $x$ en $y$ verweven,
\[
(0.x_1x_2x_3\dots,\ 0.y_1y_2y_3\dots)
\;\longmapsto\; 0.x_1y_1x_2y_2x_3y_3\dots,
\]
kies voor elke coördinaat de ontwikkeling die niet eindigt op
louter $9$'s: met die conventie bepalen de cijfers van het beeld
die van $x$ en $y$, dus is de afbeelding injectief (ze hoeft niet
surjectief te zijn --- beelden hebben bijvoorbeeld nooit oneven
posities die uiteindelijk $9$ zijn --- en dat is in \omterm{def:b2:structures:generated}{orde}).
Cantor--Bernstein (\cref{thm:b2:structures:cantorbernstein})
stelt een echte bijectie samen. Continuïteit is natuurlijk
hopeloos: een continue bijectie ertussen is onmogelijk --- de
metrische hoofdstukken leggen uit waarom (samenhang
onderscheidt de lijn van het vlak, \cref{ch:b2:metric}).
\end{example}

\section{Groepen}

\begin{definition}[Voortgebrachte deelgroep; orde]\label{def:b2:structures:generated}
Zij $G$ een groep en $A \subseteq G$. De deelgroep
\emph{voortgebracht} door $A$, genoteerd $\langle A \rangle$, is
de kleinste deelgroep die $A$ bevat --- concreet, alle eindige
producten van elementen van $A$ en hun inversen. Een groep is
\emph{cyclisch}\index{cyclische groep} wanneer ze door één element wordt voortgebracht:
$\langle a\rangle = \{a^k : k \in \Z\}$. De
\emph{orde}\index{orde!van een element} van $a \in G$ is
$\operatorname{ord}(a) = \abs{\langle a \rangle}$ (mogelijk
oneindig); wanneer eindig, is ze de kleinste $n \geq 1$ met $a^n
= e$, en $a^k = e \iff \operatorname{ord}(a) \mid k$.
\end{definition}

\begin{proof}[Bewijs van de ordekarakterisering]
Als $a^m = e$ voor zekere $m \geq 1$, zij $n \geq 1$ minimaal met
$a^n = e$. De elementen $e, a, \dots, a^{n-1}$ zijn paarsgewijs
verschillend ($a^{i} = a^{j}$ met $0 \leq i < j < n$ geeft
$a^{j-i} = e$, in strijd met minimaliteit), en elke $a^k$
reduceert tot een daarvan door Euclidische deling $k = nq + r$:
$\langle a\rangle$ heeft precies $n$ elementen, en $a^k = a^r =
e \iff r = 0 \iff n \mid k$. Als geen macht triviaal is, zijn
alle $a^k$ ($k \in \Z$) verschillend (zelfde delingsargument) en
is de \omterm{def:b2:structures:generated}{orde} oneindig.
\end{proof}

\begin{theorem}[Lagrange]\label{thm:b2:structures:lagrange}
Zij $G$ een eindige groep en $H$ een deelgroep. Dan deelt $\abs H$
$\abs G$.\index{stelling van Lagrange} In het bijzonder deelt de
\omterm{def:b2:structures:generated}{orde} van elk element $\abs G$, en $a^{\abs G} = e$ voor alle $a
\in G$.
\end{theorem}

\begin{proof}
De relatie $x \sim y \iff x^{-1}y \in H$ is een equivalentie
(reflexief: $e \in H$; symmetrisch: inversen; transitief:
producten). De klasse van $x$ is de \emph{linker nevenklasse} $xH
= \{xh : h \in H\}$, en $h \mapsto xh$ is een bijectie $H \to xH$
(inverse $y \mapsto x^{-1}y$): alle klassen hebben $\abs H$
elementen. Klassen partitioneren $G$ (de algemene
partitiestelling van het Bachelor jaar~1-volume), dus $\abs G =
\abs H \times (\text{aantal nevenklassen})$. Voor een element:
pas dit toe op $H = \langle a\rangle$; dan $a^{\abs G} =
(a^{\operatorname{ord} a})^{\abs G / \operatorname{ord} a} = e$.
\end{proof}

\begin{example}[Nevenklassen in actie: $A_3$ in $\mathfrak{S}_3$]\label{ex:b2:structures:cosets}
Neem $G = \mathfrak{S}_3$ (\omterm{def:b2:structures:generated}{orde} $6$) en $H = A_3 =
\{\mathrm{id},\ (1\,2\,3),\ (1\,3\,2)\}$. De linker nevenklassen
zijn
\[
H = \{\mathrm{id},\ (1\,2\,3),\ (1\,3\,2)\},
\qquad
(1\,2)H = \{(1\,2),\ (2\,3),\ (1\,3)\} :
\]
twee klassen van drie elementen die $G$ partitioneren, precies
zoals de telling $\abs G = \abs H \times (\text{aantal
nevenklassen})$ eist --- en zichtbaar de partitie in even en
oneven permutaties. Merk op $(1\,3)H = (1\,2)H$ hoewel $(1\,3)
\neq (1\,2)$: nevenklassen zijn \emph{klassen}, niet gelabeld
door hun representanten, en $x^{-1}y \in H$ is de enige
legitieme vergelijking. Dit twee-klassenbeeld is het algemene
voor de signatuur: $A_n$ en haar eenzame begeleidende
nevenklasse splitsen $\mathfrak{S}_n$ in tweeën, en zo telt het
weekendprobleem bereikbare puzzelposities.
\end{example}

\begin{example}\label{ex:b2:structures:lagrangeapp}
Twee onmiddellijke dividenden. \emph{Groepen van priemorde zijn
\omterm{def:b2:structures:generated}{cyclisch}:} als $\abs G = p$ priem is en $a \neq e$, dan deelt
$\operatorname{ord}(a)$ $p$ en is niet $1$, dus is ze $p$:
$\langle a\rangle = G$. \emph{Het deelgroepenrooster van
$\Z/12\Z$:} door \cref{prop:b2:structures:cyclic} hieronder
bestaat er precies één deelgroep per deler van $12$ --- \omterm{def:b2:structures:generated}{ordes} $1,
2, 3, 4, 6, 12$, voortgebracht respectievelijk door $\overline
0$, $\overline 6$, $\overline 4$, $\overline 3$, $\overline 2$,
$\overline 1$. De sluitende waarschuwing: de \emph{omgekeerde}
van Lagrange faalt in het algemeen --- $A_4$ heeft \omterm{def:b2:structures:generated}{orde} $12$
maar geen deelgroep van \omterm{def:b2:structures:generated}{orde} $6$, zoals we bewijzen in het
weekendprobleem van dit hoofdstuk (\cref{pb:b2:structures:1},
vraag 14). Lagrange beperkt de mogelijke \omterm{def:b2:structures:generated}{ordes}; ze belooft ze
niet.
\end{example}

\begin{proposition}[Cyclische groepen]\label{prop:b2:structures:cyclic}
Zij $G = \langle a \rangle$ \omterm{def:b2:structures:generated}{cyclisch} van \omterm{def:b2:structures:generated}{orde} $n$.
\begin{enumerate}
  \item $G$ is isomorf met $(\Z/n\Z, +)$, via $\overline k \mapsto
        a^k$.
  \item Elke deelgroep van $G$ is \omterm{def:b2:structures:generated}{cyclisch}; voor elke deler $d \mid n$
        bestaat er precies één deelgroep van \omterm{def:b2:structures:generated}{orde} $d$, namelijk
        $\langle a^{n/d}\rangle$.
  \item $a^k$ brengt $G$ voort desda $\gcd(k, n) = 1$: $G$ heeft
        $\varphi(n)$ voortbrengers (Eulers
        totientfunctie\index{totientfunctie van Euler}).
\end{enumerate}
\end{proposition}

\begin{proof}
(1) De afbeelding $k \mapsto a^k$ van $\Z$ op $G$ is compatibel
met congruentie mod $n$ ($a^{k} = a^{k'} \iff n \mid k - k'$,
door de ordekarakterisering); de universele eigenschap
(\cref{def:b2:structures:quotient}) levert een welgedefinieerd
bijectief morfisme van $\Z/n\Z$.

(2) Zij $H \leq G$ niet-triviaal en $m$ minimaal $\geq 1$ met
$a^m \in H$. Euclidische deling toont $H = \langle a^m\rangle$
(voor $a^k \in H$: $k = mq + r$ dwingt $a^r \in H$, dus $r =
0$), en $m \mid n$ (deel $n$ door $m$: $a^{n \bmod m} \in H$).
Dan $\abs H = n/m$; met $m = n/d$ realiseert men elke deler $d$.
Uniciteit: elke deelgroep van \omterm{def:b2:structures:generated}{orde} $d$ is, door het voorgaande,
van de vorm $\langle a^m \rangle$ met $n/m = d$ --- dus $m =
n/d$ is gedwongen en de deelgroep is bepaald.

(3) We claimen $\operatorname{ord}(a^k) = \frac{n}{\gcd(k, n)}$.
Schrijf $d = \gcd(k, n)$. Voor elke $m \geq 1$ geeft de
ordekarakterisering van \cref{def:b2:structures:generated} de
keten van equivalenties
\[
(a^k)^m = e
\iff n \mid km
\iff \frac{n}{d} \,\Big|\, \frac{k}{d}\,m
\iff \frac{n}{d} \,\Big|\, m ,
\]
de laatste stap door het lemma van Gauss, omdat $\frac nd$ en
$\frac kd$ relatief priem zijn. De kleinste zulke $m$ is $\frac
nd$: $\operatorname{ord}(a^k) = \frac n{\gcd(k,n)}$, die $n$
gelijk is desda $\gcd(k, n) = 1$. Er zijn $\varphi(n)$ zulke
klassen $k$ modulo $n$.
\end{proof}

\section{De symmetrische groep}

\begin{definition}\label{def:b2:structures:sn}
$\mathfrak{S}_n$ is de groep van permutaties van $\intint{1}{n}$
(\omterm{def:b2:structures:generated}{orde} $n!$). Een \emph{cykel}\index{cykel} $(a_1\,a_2\,\cdots\,a_k)$
stuurt $a_1 \mapsto a_2 \mapsto \dots \mapsto a_k \mapsto a_1$ en
fixteert al het overige; $k$ is haar \emph{lengte}, een $2$-cykel is
een \emph{transpositie}\index{transpositie}. Twee cykels zijn
\emph{disjunct} wanneer hun steunen (niet-gefixeerde punten) dat zijn.
\end{definition}

\begin{theorem}[Cykelontbinding]\label{thm:b2:structures:cycles}
Elke permutatie $\sigma \neq \mathrm{id}$ is een product van
paarsgewijs disjuncte \omterm{def:b2:structures:sn}{cykels}, uniek op de volgorde van de factoren na.
Disjuncte \omterm{def:b2:structures:sn}{cykels} commuteren, en $\operatorname{ord}(\sigma)$ is de
kgv van de lengtes.
\end{theorem}

\begin{proof}
Beschouw de ``baan''-relatie op de steun van $\sigma$: $x \sim y$ desda
$y = \sigma^k(x)$ voor zekere $k \in \Z$ --- een equivalentierelatie.
Elke klasse $\{x, \sigma(x), \dots, \sigma^{k-1}(x)\}$ (eindig, dus
de iteraten keren terug --- de eerste herhaling moet naar $x$
terugkeren door injectiviteit) draagt de \omterm{def:b2:structures:sn}{cykel} $(x\ \sigma(x)\
\cdots\ \sigma^{k-1}(x))$, en $\sigma$ is het product van deze
\omterm{def:b2:structures:sn}{cykels}: op elke baan werkt alleen de corresponderende \omterm{def:b2:structures:sn}{cykel}.
Uniciteit: elke disjuncte-cykel-factorisatie reproduceert precies de
banen (de \omterm{def:b2:structures:sn}{cykel} door $x$ moet $(x\ \sigma(x)\ \cdots)$ zijn).
Disjuncte \omterm{def:b2:structures:sn}{cykels} commuteren omdat ze disjuncte punten bewegen; de
orde-uitspraak volgt omdat $\sigma^m = \mathrm{id}$ desda de
$m$-de macht van elke \omterm{def:b2:structures:sn}{cykel} dat is (disjunctheid), desda elke
lengte $m$ deelt.
\end{proof}

\begin{example}[Cykeltype als census]\label{ex:b2:structures:cycletype}
Hoeveel permutaties van $\mathfrak{S}_9$ hebben het cykeltype
$(4, 3, 2)$ --- één $4$-cykel, één $3$-cykel, één \omterm{def:b2:structures:sn}{transpositie}?
Kies de steunen en de cyclische \omterm{def:b2:structures:generated}{ordes}:
\[
\frac{9!}{4\cdot 3\cdot 2}
= \frac{362\,880}{24} = 15\,120 :
\]
rangschik de negen symbolen in een rij ($9!$ manieren), plaats de
eerste vier, volgende drie, laatste twee in \omterm{def:b2:structures:sn}{cykels}, en deel door
de rotaties binnen elke haak ($4$, $3$ en $2$ ervan) die dezelfde
permutatie geven. (Hier verschillende \omterm{def:b2:structures:sn}{cykel}\emph{lengtes}, dus
geen verdere deling; gelijke lengtes zouden ook delen door de
permutaties van de gelijke haken vereisen.) Elke zulke permutatie
heeft \omterm{def:b2:structures:generated}{orde} $\operatorname{lcm}(4,3,2) = 12$ en signatuur
$(-1)^3(-1)^2(-1)^1 = +1$
(\cref{thm:b2:structures:cycles} en de signatuurstelling
hieronder). Eén partitie van $9$, één conjugatieklasse, één
census --- de combinatoriek van $\mathfrak{S}_n$ is de rekenkunde
van partities.
\end{example}

\begin{theorem}[Signatuur]\label{thm:b2:structures:signature}
Er is precies één groepsmorfisme $\varepsilon \colon
\mathfrak{S}_n \to \{\pm 1\}$ (voor $n \geq 2$) dat de waarde $-1$
aanneemt op \omterm{def:b2:structures:sn}{transposities}: de \emph{signatuur}\index{signatuur}.
Bovendien is $\varepsilon(\sigma) = (-1)^{I(\sigma)}$ waarbij
$I(\sigma)$ het aantal \emph{inversies} is (paren $i < j$ met
$\sigma(i) > \sigma(j)$), heeft een $k$-cykel signatuur
$(-1)^{k-1}$, en heeft de \emph{alternerende groep} $A_n =
\ker\varepsilon$ \omterm{def:b2:structures:generated}{orde} $\frac{n!}{2}$.
\end{theorem}

\begin{proof}
\emph{Bestaan.} Voor $\sigma \in \mathfrak{S}_n$ stel
\[
\varepsilon(\sigma)
= \prod_{1 \leq i < j \leq n}
\frac{\sigma(j) - \sigma(i)}{j - i} .
\]
De absolute waarden van de factoren vermenigvuldigen tot $1$ (de
ongeordende paren $\{\sigma(i), \sigma(j)\}$ lopen over alle paren),
dus $\varepsilon(\sigma) = (-1)^{I(\sigma)} \in \{\pm1\}$. Morfisme:
voor $\sigma, \tau$,
\[
\varepsilon(\sigma\tau)
= \prod_{i<j} \frac{\sigma(\tau(j)) - \sigma(\tau(i))}{j - i}
= \prod_{i<j} \frac{\sigma(\tau(j)) - \sigma(\tau(i))}{\tau(j) -
\tau(i)} \cdot \prod_{i<j} \frac{\tau(j) - \tau(i)}{j - i}
= \varepsilon(\sigma)\,\varepsilon(\tau),
\]
het middelste product is $\varepsilon(\sigma)$ na herindexering
door de paren $\{\tau(i), \tau(j)\}$ (elk ongeordend paar
verschijnt één keer, en teller en noemer wisselen van teken
samen). Een \omterm{def:b2:structures:sn}{transpositie} $\tau = (a\,b)$ met $a < b$ heeft een
oneven aantal inversies; precies geteld: de geïnverteerde paren
$(i, j)$, $i < j$, met $\tau(i) > \tau(j)$ zijn
\[
(a, j) \ \text{voor } a < j < b, \qquad
(i, b) \ \text{voor } a < i < b, \qquad
(a, b) \ \text{zelf},
\]
dat is $(b - a - 1) + (b - a - 1) + 1 = 2(b - a) - 1$ ervan,
oneven. (Alternatief: controleer $(1\,2)$ direct, met één
inversie, en conjugeer --- conjugaten hebben gelijke signatuur
omdat $\varepsilon$ een morfisme is naar een abelse groep.)
Dus $\varepsilon((a\,b)) = (-1)^{2(b-a)-1} = -1$.

\emph{Uniciteit.} \omterm{def:b2:structures:sn}{Transposities} brengen $\mathfrak{S}_n$ voort
(elke \omterm{def:b2:structures:sn}{cykel} $(a_1\cdots a_k) = (a_1\,a_k)(a_1\,a_{k-1})\cdots(a_1\,a_2)$,
en \cref{thm:b2:structures:cycles} voltooit); een morfisme naar
$\{\pm1\}$ is bepaald door zijn waarden op voortbrengers.

\emph{Gevolgen.} De cykelidentiteit hierboven schrijft een
$k$-cykel als $k - 1$ \omterm{def:b2:structures:sn}{transposities}: signatuur $(-1)^{k-1}$.
$A_n$: het morfisme $\varepsilon$ is surjectief (\omterm{def:b2:structures:sn}{transposities}
bestaan voor $n \geq 2$), en de twee ``nevenklassen'' $A_n$ en
$(1\,2)A_n$ zijn \omterm{def:b2:structures:countable}{gelijkmachtig} en partitioneren $\mathfrak{S}_n$
(argument van Lagrange): $\abs{A_n} = \frac{n!}{2}$.
\end{proof}

\begin{example}\label{ex:b2:structures:snexample}
$\sigma = \begin{pmatrix} 1&2&3&4&5&6\\ 3&6&5&4&1&2 \end{pmatrix}
= (1\,3\,5)(2\,6)$: \omterm{def:b2:structures:generated}{orde} $\operatorname{lcm}(3,2) = 6$, signatuur
$(-1)^{2}\cdot(-1)^{1} = -1$. De signatuur is de snelste
pariteitscontrole op schudden --- en de motor van de determinant
in \cref{ch:b2:linalg}.
\end{example}

\begin{example}[Drie wegen naar één teken]\label{ex:b2:structures:threeroads}
Zij $\sigma \in \mathfrak{S}_5$ de afbeelding die $1, 2, 3, 4, 5$
stuurt naar $3, 5, 4, 1, 2$. \emph{Via \omterm{def:b2:structures:sn}{cykels}:} $1 \mapsto 3
\mapsto 4 \mapsto 1$ en $2 \mapsto 5 \mapsto 2$, dus $\sigma =
(1\,3\,4)(2\,5)$ en $\varepsilon(\sigma) = (-1)^{2}(-1)^{1} =
-1$. \emph{Via inversies:} in de waardenlijst $3, 5, 4, 1, 2$
zijn de uit-volgorde-paren $(3,1)$, $(3,2)$, $(5,4)$, $(5,1)$,
$(5,2)$, $(4,1)$, $(4,2)$: zeven ervan, en $(-1)^7 = -1$.
\emph{Via \omterm{def:b2:structures:sn}{transposities}:} $\sigma = (1\,4)(1\,3)(2\,5)$, drie
factoren, $(-1)^3 = -1$. Drie berekeningen, één pariteit: de
uniciteit in \cref{thm:b2:structures:signature} garandeert dat
geen boekhoudschema ze ooit oneens kan maken --- precies wat
$\varepsilon$ bruikbaar maakt als invariant (zie het
weekendprobleem).
\end{example}

\begin{remark}[Waar de signatuur hierheen gaat]
De signatuur is het zaad van drie latere oogsten: ze bouwt de
determinant en haar productregel in \cref{ch:b2:linalg}; ze
drijft pariteitsinvarianten voor combinatorische puzzels (het
weekendprobleem van dit hoofdstuk lost de vijftienpuzzel ermee
op); en de alternerende groepen $A_n$ die ze definieert worden
centraal in het Bachelor jaar~3-volume, waar hun eenvoudigheid
voor $n \geq 5$ verklaart waarom vergelijkingen van graad~$5$
geen oplossing in radicalen hebben.
\end{remark}

\section{Ringen, idealen, quotiënten}

\begin{definition}[Ideaal]\label{def:b2:structures:ideal}
Zij $A$ een commutatieve ring. Een \emph{ideaal}\index{ideaal} $I
\subseteq A$ is een additieve deelgroep zodanig dat $a x \in I$
voor alle $a \in A$, $x \in I$. Kernen van ringmorfismen zijn
idealen; $I = A$ desda $1 \in I$ desda $I$ een eenheid bevat. Het
ideaal \emph{voortgebracht} door $x$ is $xA = \{xa\}$ (een
\emph{hoofdideaal}).
\end{definition}

\begin{theorem}[{Idealen van $\Z$ en van $K[X]$}]\label{thm:b2:structures:principal}
Elk \omterm{def:b2:structures:ideal}{ideaal} van $\Z$ is $n\Z$ voor unieke $n \in \N$; elk \omterm{def:b2:structures:ideal}{ideaal} van
$K[X]$ ($K$ een lichaam) is $P\,K[X]$ voor unieke monische (of
nulle) $P$. Bijgevolg bestaan ggd's in beide ringen met
Bézout-relaties: $x\Z + y\Z = \gcd(x,y)\Z$, en evenzo voor
veeltermen.
\end{theorem}

\begin{proof}
Voor $\Z$ was dit de deelgroepstelling van het Bachelor
jaar~1-volume (een \omterm{def:b2:structures:ideal}{ideaal} is in het bijzonder een deelgroep, en
$n\Z$ is een \omterm{def:b2:structures:ideal}{ideaal}). Voor $K[X]$: zij $I \neq \{0\}$ een \omterm{def:b2:structures:ideal}{ideaal}
en $P \in I$ niet-nul van minimale graad, genormaliseerd monisch.
Voor $F \in I$ geeft Euclidische deling $F = PQ + R$ dat $R = F -
PQ \in I$ met $\deg R < \deg P$: minimaliteit dwingt $R = 0$, dus
$I = P\,K[X]$. Uniciteit: twee monische voortbrengers delen
elkaar. De Bézout-uitspraken zijn de gelijkheid van het \omterm{def:b2:structures:ideal}{ideaal}
$x\Z + y\Z$ (resp.\ zijn veeltermanaloog) met het hoofdideaal van
de ggd --- de definitie zelf van ggd uit Jaar~1, nu herkend als
een uitspraak over idealen.
\end{proof}

\begin{example}[Een veelterm-ggd, op twee manieren]\label{ex:b2:structures:polygcd}
Bereken $\gcd(X^3 - 1,\ X^2 - 1)$ in $\Q[X]$. \emph{Via Euclid:}
\[
X^3 - 1 = X\,(X^2 - 1) + (X - 1),
\qquad
X^2 - 1 = (X + 1)(X - 1) + 0 ,
\]
dus de ggd is $X - 1$, en terugsubstitutie geeft de Bézout-relatie
\[
X - 1 = 1\cdot(X^3 - 1) - X\cdot(X^2 - 1).
\]
\emph{Via idealen:} het \omterm{def:b2:structures:ideal}{ideaal} $(X^3 - 1)\Q[X] + (X^2 - 1)\Q[X]$ is
principaal (\cref{thm:b2:structures:principal}); het bevat $X -
1$ (de weergave) en is bevat in $(X - 1)\Q[X]$ (beide
voortbrengers verdwijnen in $1$, dus zijn veelvouden van $X -
1$): de monische voortbrenger is $X - 1$. Het sluitende inzicht:
het ideaalstandpunt identificeert de ggd \emph{zonder te delen}
--- gemeenschappelijke wortels lokaliseren het \omterm{def:b2:structures:ideal}{ideaal}, en Euclid
certificeert het slechts.
\end{example}

\begin{definition}[Quotiëntring $\Z/n\Z$, hernomen]\label{def:b2:structures:quotientring}
Voor een \omterm{def:b2:structures:ideal}{ideaal} $I$ van $A$ is de relatie $x \sim y \iff x - y \in
I$ een equivalentie compatibel met $+$ en $\times$; de
\omterm{def:b2:structures:quotient}{quotiëntverzameling} $A/I$ erft een ringstructuur --- de
\emph{quotiëntring}\index{quotiëntring} --- die $\pi \colon A \to
A/I$ tot een morfisme maakt met kern $I$. Voor $A = \Z$, $I =
n\Z$ is dit de $\Z/n\Z$ van het Bachelor jaar~1-volume, nu met
haar universele eigenschap: elk morfisme dat $I$ doodt,
factoriseert door $A/I$.
\end{definition}

\begin{theorem}[Chinese reststelling, ringvorm]\label{thm:b2:structures:crt}
Als $\gcd(m, n) = 1$, is de afbeelding
\[
\Z/mn\Z \longrightarrow \Z/m\Z \times \Z/n\Z,
\qquad
\overline{x} \longmapsto (x \bmod m,\; x \bmod n)
\]
een ringisomorfisme.\index{Chinese reststelling} Bijgevolg is
$\varphi(mn) = \varphi(m)\varphi(n)$ voor relatief prieme $m, n$,
en
\[
\varphi(n) = n \prod_{p \mid n} \Bigl(1 - \frac 1p\Bigr)
\quad (p \text{ priem}).
\]
\end{theorem}

\begin{proof}
De afbeelding is een welgedefinieerd ringmorfisme
(compatibiliteiten zijn onmiddellijk). Injectiviteit: $x \equiv
0$ mod $m$ en mod $n$ met $\gcd(m,n) = 1$ dwingt $mn \mid x$
(Gauss). Surjectiviteit: beide kanten hebben $mn$ elementen, dus
injectiviteit volstaat (eindige gelijke kardinaliteiten) --- of
expliciet: uit een Bézout-relatie $um + vn = 1$ beeldt de klasse
van
\[
x = b\,um + a\,vn
\]
af op $(a \bmod m,\ b \bmod n)$, omdat $vn = 1 - um \equiv 1
\pmod m$ $x \equiv a \pmod m$ maakt, en symmetrisch mod $n$ ---
het recept numeriek gebruikt in
\cref{ex:b2:structures:crtinverse}. Eenheden corresponderen met
paren van eenheden (de eenheden van een productring zijn de paren
van eenheden), dus $\varphi(mn) = \varphi(m)\varphi(n)$. Voor een
priemmacht is $\varphi(p^k) = p^k - p^{k-1}$ (de niet-eenheden
mod $p^k$ zijn de veelvouden van $p$); multiplicativiteit stelt
de productformule samen.
\end{proof}

\begin{example}[Het Chinese isomorfisme inverteren]\label{ex:b2:structures:crtinverse}
Neem $m = 8$, $n = 9$. De inverse van het isomorfisme wordt
expliciet gemaakt door de twee \emph{idempotenten}: zoek $u \equiv
1 \pmod 8$, $u \equiv 0 \pmod 9$ en $v \equiv 0 \pmod 8$, $v
\equiv 1 \pmod 9$. Uit $u = 9k \equiv 1 \pmod 8$: $k \equiv 1$,
dus $u = 9$; uit $v = 8k \equiv 1 \pmod 9$: $-k \equiv 1$, $k
\equiv 8$, dus $v = 64$. Dan is de klasse van $x = 9a + 64b$
modulo $72$ de unieke oplossing van $x \equiv a \pmod 8$, $x
\equiv b \pmod 9$: voor $a = 3$, $b = 5$ krijgt men $27 + 320 =
347 \equiv 59 \pmod{72}$ --- precies de tussenwaarde gevonden
door substitutie in \cref{exo:b2:structures:8}. Het sluitende
inzicht: $u$ en $v$ voldoen aan $u + v \equiv 1$, $uv \equiv 0$,
$u^2 \equiv u$, $v^2 \equiv v$ modulo $72$; ze zijn de beelden
van $(1, 0)$ en $(0, 1)$, en elke Chinese ontbinding is in de
kern een ontbinding van $1$ in orthogonale idempotenten.
\end{example}

\begin{theorem}[Euler; Fermat hernomen]\label{thm:b2:structures:euler}
De eenheden van $\Z/n\Z$ vormen een groep van \omterm{def:b2:structures:generated}{orde} $\varphi(n)$;
dus voor $\gcd(a, n) = 1$:
\[
a^{\varphi(n)} \equiv 1 \pmod n
\qquad (\text{stelling van Euler\index{stelling van Euler}}),
\]
en de kleine stelling van Fermat is het geval $n = p$ priem, nu
één regel uit Lagrange.
\end{theorem}

\begin{proof}
De inverteerbare klassen zijn precies die van gehele getallen
relatief priem met $n$ (Bachelor jaar~1-volume): $\varphi(n)$
ervan, die een groep vormen onder vermenigvuldiging. Lagrange
(\cref{thm:b2:structures:lagrange}): elk element tot de macht
van de groepsorde is het neutrale element.
\end{proof}

\begin{example}[Een eenhedengroep zonder voortbrenger]\label{ex:b2:structures:units15}
De groep $(\Z/15\Z)^*$ heeft $\varphi(15) = \varphi(3)\varphi(5)
= 8$ elementen. Is ze \omterm{def:b2:structures:generated}{cyclisch}? Bereken \omterm{def:b2:structures:generated}{ordes} via het Chinese
isomorfisme $(\Z/15\Z)^* \simeq (\Z/3\Z)^* \times (\Z/5\Z)^*$
(een eenheid mod $15$ is een paar eenheden): de factoren hebben
\omterm{def:b2:structures:generated}{ordes} $2$ en $4$, dus deelt de \omterm{def:b2:structures:generated}{orde} van elk element
$\operatorname{lcm}(2, 4) = 4 < 8$ --- geen element brengt voort.
Concreet:
\[
2^4 = 16 \equiv 1, \qquad
4^2 = 16 \equiv 1, \qquad
7^4 \equiv 1, \qquad
11^2 = 121 \equiv 1, \qquad
14^2 \equiv 1 \pmod{15} :
\]
\omterm{def:b2:structures:generated}{ordes} $4, 2, 4, 2, 2$ en nooit $8$. Contrast met
\cref{exo:b2:structures:10}: $(\Z/p\Z)^*$ \emph{is} \omterm{def:b2:structures:generated}{cyclisch}
voor $p$ priem, omdat daar de eenhedengroep in een lichaam zit.
De stelling van Euler geldt nog met exponent $\varphi(15) = 8$,
maar de echte universele exponent hier is $4$ --- Euler is een
bovengrens, niet altijd de scherpe.
\end{example}

\begin{definition}[Algebra]\label{def:b2:structures:algebra}
Een \emph{$K$-algebra}\index{algebra} is een $K$-vectorruimte $A$
met een ringstructuur waarvan de vermenigvuldiging $K$-bilineair
is. Voorbeelden: $K[X]$, $\mathcal{M}_n(K)$, $\mathcal{L}(E)$,
functieruimten $\mathcal{F}(X, K)$, $\C$ als $\R$-algebra.
Morfismen van algebras zijn lineaire ringmorfismen; de
\emph{evaluatie} $P \mapsto P(u)$ van $K[X]$ naar
$\mathcal{L}(E)$ (of $\mathcal{M}_n(K)$) is het centrale
voorbeeld, dat \cref{ch:b2:reduction} drijft.
\end{definition}

\begin{example}[Een evaluatiemorfisme en zijn kern]\label{ex:b2:structures:evaluation}
Neem $A = \begin{pmatrix}0 & 1\\ 0 & 0\end{pmatrix}$ en de
evaluatie $\varepsilon_A \colon \R[X] \to \mathcal{M}_2(\R)$,
$P \mapsto P(A)$. Omdat $A^2 = 0$,
\[
P(A) = P(0)\,I + P'(0)\,A =
\begin{pmatrix} P(0) & P'(0)\\ 0 & P(0)\end{pmatrix},
\]
(alleen de constante en lineaire termen van $P$ overleven). Dus
$\ker\varepsilon_A = \{P : P(0) = P'(0) = 0\} = X^2\,\R[X]$: een
hoofdideaal, precies zoals \cref{thm:b2:structures:principal}
voorspelt, voortgebracht door de monische $X^2$ van minimale
graad in de kern --- de \emph{minimale veelterm} van $A$, ster
van \cref{ch:b2:reduction}. Het beeld is de
tweedimensionale commutatieve \omterm{def:b2:structures:algebra}{algebra} $\{aI + bA\}$:
evaluatiemorfismen krimpen de oneindigdimensionale $\R[X]$ tot
kleine, berekenbare \omterm{def:b2:structures:algebra}{algebras}.
\end{example}

\begin{remark}[Perspectieven: drie melodieën om te beluisteren]\label{rem:b2:structures:perspectives}
Drie structurele ideeën uit dit hoofdstuk keren doorheen het
volume terug, telkens in zwaardere orkestratie. \emph{Factorisatie
door een quotiënt} (\cref{def:b2:structures:quotient}): ze bouwt
$\Z/n\Z$ hier, definieert afbeeldingen op oplossingsverzamelingen
van lineaire stelsels in \cref{ch:b2:linalg}, en ligt stilzwijgend
ten grondslag aan elk ``welgedefinieerd op klassen''-argument.
\emph{Invarianten}: de signatuur is een morfisme naar $\{\pm1\}$
dat geen legale zet kan ontwijken --- dezelfde logica geeft de
productregel van de determinant (\cref{ch:b2:linalg}), de
gelijkvormigheidsinvariantie van het spoor, en de behouden
grootheden van \cref{ch:b2:diffeq}. \emph{Tellen tegen een
structuur}: Lagrange telt via nevenklassen, dimensie telt via
bases (\cref{ch:b2:linalg}), multipliciteit telt via
veeltermgraden (\cref{ch:b2:reduction}); wanneer een grens
wonderbaarlijk lijkt, doet een partitie of graadstructuur het
tellen.
\end{remark}

\begin{remark}[Veelvoorkomende valkuilen]\label{rem:b2:structures:pitfalls}
Vier klassiekers. (i) Een afbeelding op een quotiënt moet
\emph{welgedefinieerd} worden gecontroleerd: ``$\overline x
\mapsto$ (formule op $x$)'' is legitiem alleen als de formule
constant is op klassen --- de compatibiliteit van
\cref{def:b2:structures:quotient}, geen formaliteit. (ii)
$\operatorname{ord}(ab) =
\operatorname{lcm}(\operatorname{ord}a, \operatorname{ord}b)$ is
\emph{vals} in het algemeen, zelfs voor commuterende elementen
($a$ en $a^{-1}$); \cref{exo:b2:structures:4} geeft de juiste
relatief-priem-en-commuterende uitspraak, en disjuncte \omterm{def:b2:structures:sn}{cykels} de
juiste permutatieversie. (iii) Aftelbaarheid overleeft aftelbare
\emph{unies} en eindige \emph{producten}, maar niet aftelbare
producten: $\{0,1\}^{\N}$ is overaftelbaar
(\cref{exo:b2:structures:3}) hoewel elke factor twee elementen
heeft. (iv) Cantor--Bernstein heeft alleen injecties beide kanten
nodig, maar de bijectie die ze bouwt is meestal discontinu en
niet-expliciet --- verwacht geen formule
(\cref{ex:b2:structures:cbexample}).
\end{remark}

\begin{remark}[Waar dit hoofdstuk wordt gebruikt]
Bijna overal. De signatuur bouwt determinanten
(\cref{ch:b2:linalg}); het evaluatiemorfisme $P \mapsto P(u)$
en de hoofdidealen van $K[X]$ produceren minimale veeltermen en
de kernontbindingen van \cref{ch:b2:reduction}; aftelbaarheid is
het podium waarop \cref{ch:b2:proba} optreedt
(waarschijnlijkheid op aftelbare ruimten) en de reden dat
topologie aftelbare dichte verzamelingen blijft produceren
(\cref{ch:b2:metric}). De quotiëntconstructie $A/I$ wordt
heringezet in het Bachelor jaar~3-volume om lichamen $K[X]/(P)$
te bouwen en, daaruit, Galoistheorie: de hier bewezen
universele eigenschap wordt daar woord voor woord gebruikt.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b2:structures:1}
Welke van de volgende verzamelingen zijn \omterm{def:b2:structures:countable}{aftelbaar}? De verzameling
van eindige deelverzamelingen van $\N$; de verzameling van
\emph{alle} deelverzamelingen van $\N$; $\R \setminus \Q$; de
verzameling van veeltermen met rationale coëfficiënten; de
verzameling van rijen van $0$'en en $1$'en die uiteindelijk nul
zijn.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:structures:2}
In $\mathfrak{S}_7$, zij $\sigma = (1\,4\,2\,6)(3\,5)$ en $\tau =
(2\,3\,7)$. Bereken $\sigma\tau$ en $\tau\sigma$ in
disjuncte-cykel-vorm, de \omterm{def:b2:structures:generated}{ordes} en signaturen van alle vier de
permutaties, en $\sigma^{2026}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:structures:3}
Construeer expliciete injecties die tonen dat $\mathcal{P}(\N)$,
$\intcc{0}{1}$ en de verzameling $\{0,1\}^{\N}$ van binaire rijen
paarsgewijs \omterm{def:b2:structures:countable}{gelijkmachtig} zijn \emph{(binaire ontwikkelingen beide
kanten op; Cantor--Bernstein absorbeert de dubbele-representatie-hinder)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:structures:4}
Zij $G$ een groep en $a, b \in G$ commuterende elementen van
eindige relatief prieme \omterm{def:b2:structures:generated}{ordes} $m$ en $n$. Bewijs dat
$\operatorname{ord}(ab) = mn$. Toon met een voorbeeld in
$\mathfrak{S}_3$ dat commutatie essentieel is.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:structures:5}
Zij $G$ een eindige groep van even \omterm{def:b2:structures:generated}{orde}. Bewijs dat $G$ een
element van \omterm{def:b2:structures:generated}{orde} $2$ bevat. \emph{(Paar elk element met zijn
inverse; tel de zelf-gepaarde.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:structures:6}
Bewijs dat $A_n$ ($n \geq 3$) wordt voortgebracht door de
$3$-cykels. \emph{(Een product van twee \omterm{def:b2:structures:sn}{transposities} is een
$3$-cykel of een product van twee $3$-cykels.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:structures:7}
Bepaal alle groepsmorfismen: van $(\Q, +)$ naar $(\Z, +)$; van
$(\Z/n\Z, +)$ naar $(\Z/m\Z, +)$ \emph{(tel ze: $\gcd(m,n)$)}; van
$(\Q, +)$ naar $(\Q_+^*, \times)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:structures:8}
Met de Chinese reststelling, bereken $\varphi(360)$, vind alle $x$
met $x \equiv 3 \pmod 8$, $x \equiv 5 \pmod 9$ en $x \equiv 2
\pmod 5$, en bereken de laatste twee cijfers van $3^{2026}$
\emph{(Euler mod $100$; let op: werk mod $4$ en mod $25$)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:structures:9}
Bewijs dat een eindig integriteitsdomein een lichaam is. Leid af
dat $\Z/n\Z$ een lichaam is desda $n$ priem is (opnieuw).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:structures:10}
(Een klassieker) Zij $K$ een lichaam en $G$ een \emph{eindige}
deelgroep van $(K^*, \times)$. Bewijs dat $G$ \omterm{def:b2:structures:generated}{cyclisch} is.
\emph{Hint: zij $m$ de maximale \omterm{def:b2:structures:generated}{orde} onder de elementen van $G$;
toon dat de \omterm{def:b2:structures:generated}{orde} van elk element $m$ deelt (met
\cref{exo:b2:structures:4} op geschikte relatief prieme delen),
dus voldoet heel $G$ aan $x^m = 1$; tel wortels van $X^m - 1$.}
In het bijzonder is $(\Z/p\Z)^*$ \omterm{def:b2:structures:generated}{cyclisch}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:structures:11}
Bewijs dat de groep $(\Q, +)$ niet \omterm{def:b2:structures:generated}{cyclisch} is, en erger: ze is
niet eens eindig voortgebracht. Bewijs anderzijds dat elke eindig
voortgebrachte deelgroep van $(\Q, +)$ \omterm{def:b2:structures:generated}{cyclisch} is.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:structures:12}
(Criterium van Dedekind) Bewijs dat elke oneindige verzameling een
aftelbare deelverzameling bevat, en leid af dat een verzameling
$E$ oneindig is desda ze \omterm{def:b2:structures:countable}{gelijkmachtig} is met een echte
deelverzameling van zichzelf. \emph{(Voor de directe implicatie,
verschuif een aftelbare deelverzameling met één stap; voor de
omgekeerde, herinner het duivenhokprincipe.)}
\end{exercise}

\section{Probleem: De vijftienpuzzel}

De vijftienpuzzel is een $4 \times 4$-tray met vijftien schuifende
tegels genummerd $1$ tot $15$ en één lege cel; een zet schuift een
van de tegels die aan de lege cel grenzen erin. In de jaren 1890
populariseerde Sam Loyd de puzzel door \$1000 te bieden aan
iedereen die de tegels $14$ en $15$ kon verwisselen en elke andere
tegel op zijn plaats terugbrengen. Niemand heeft ooit geïncasseerd,
en dit weekendprobleem bewijst beide helften van de reden: de
signatuur van \cref{thm:b2:structures:signature} verbiedt Loyds
verwisseling, en --- de zwaardere, constructieve helft --- is
\emph{alles} wat de signatuur toelaat echt oplosbaar. De volledige
uitspraak is de stelling van Johnson--Story (1879).

\begin{problem}[{Weekendprobleem --- de oplosbaarheidsstelling van
Johnson--Story}]
\label{pb:b2:structures:1}
Nummer de cellen $1$ tot $16$ in leesvolgorde (links naar rechts,
boven naar onder), zodat cel $k$ in rij $i$ en kolom $j$ zit met $k
= 4(i - 1) + j$. Cel $16$ (rechtsonder) is het \emph{thuis} van de
lege cel; we behandelen de lege cel als een zestiende tegel,
genoteerd $b$ en geïdentificeerd met het getal $16$. Een
\emph{configuratie} is een bijectie $\sigma \colon \intint1{16}
\to \intint1{16}$, cel $\mapsto$ inhoud; de \emph{opgeloste}
configuratie is $\sigma = \mathrm{id}$. Overal is $\varepsilon$ de
signatuur van \cref{thm:b2:structures:signature} en zijn twee
cellen \emph{aangrenzend} wanneer ze een rand van de tray delen.

\medskip
\noindent\textbf{Deel I --- Configuraties, zetten, signaturen.}
\begin{enumerate}
  \item Rechtvaardig dat de configuraties precies de elementen
        van $\mathfrak{S}_{16}$ zijn, zodat er $16! =
        20\,922\,789\,888\,000$ van zijn, en dat het aantal legale
        zetten vanuit een gegeven configuratie $2$, $3$ of $4$ is,
        al naargelang de lege cel in een hoek, op een rand, of in
        het inwendige ligt.
  \item Zij $\sigma$ een configuratie, $p = \sigma^{-1}(16)$ de
        cel van de leegte, en $c$ een cel aangrenzend aan $p$.
        Toon dat het schuiven van de tegel van $c$ in $p$ de
        configuratie $\sigma' = \sigma \circ \tau$ produceert met
        $\tau = (p\ c)$, en leid af dat elke zet de signatuur
        omkeert: $\varepsilon(\sigma') = -\varepsilon(\sigma)$.
  \item Schaakbord de tray: $\chi(k) = (-1)^{i+j}$ voor de cel
        $k$ in rij $i$, kolom $j$. Toon dat elke zet
        $\chi(\text{cel van de leegte})$ omkeert, en leid af dat
        een rij zetten die de leegte terugbrengt naar haar
        startcel even lengte heeft.
  \item Toon dat
        \[
        I(\sigma) = \varepsilon(\sigma)\,
        \chi\bigl(\sigma^{-1}(16)\bigr)
        \]
        invariant is onder elke legale zet, en bereken
        $I(\mathrm{id})$.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Loyds premie: de invariant aan het
werk.}
\begin{enumerate}[resume]
  \item Loyds configuratie $\sigma_L$ stemt overeen met de
        opgeloste behalve dat cellen $14$ en $15$ de tegels $15$
        en $14$ houden. Bereken $I(\sigma_L)$ en concludeer dat
        geen rij zetten $\sigma_L$ met de opgeloste configuratie
        verbindt: Loyds \$1000 was nooit in gevaar.
  \item Toon dat precies de helft van alle configuraties $I =
        +1$ voldoet: $\abs{\{\sigma : I(\sigma) = +1\}} =
        16!/2$. \emph{(Voor een vaste lege cel, paar
        configuraties door samenstellen met één vaste
        \omterm{def:b2:structures:sn}{transpositie} van twee andere cellen.)}
  \item Toon dat elke zet ongedaan wordt gemaakt door een legale
        zet, dat ``$\sigma'$ is bereikbaar vanuit $\sigma$ door
        legale zetten'' een equivalentierelatie is, en dat de
        klasse $R$ van de opgeloste configuratie $R \subseteq \{I
        = +1\}$ voldoet. Concludeer dat er minstens twee klassen
        zijn.
  \item Stel de leegte is thuis: $\sigma(16) = 16$. Toon dat
        $I(\sigma) = \varepsilon(\rho)$ waarbij $\rho \in
        \mathfrak{S}_{15}$ de restrictie van $\sigma$ tot de
        cellen $1, \dots, 15$ is, en dat elke configuratie door
        legale zetten naar een met leegte thuis kan worden
        gevoerd. Concludeer: om $R = \{I = +1\}$ te bewijzen
        volstaat het elke \emph{even} permutatie van de vijftien
        niet-thuis-cellen te realiseren door een rij zetten die
        start en eindigt met de leegte thuis.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Leegterondes en de programmagroep.}
Een \emph{programma} is een eindige rij legale zetten, gestart
vanuit een configuratie met de leegte thuis, waarvan de eindconfiguratie
weer de leegte thuis heeft. Haar \emph{effect} is de permutatie
$\pi$ van de cellen gedefinieerd door: de inhoud van cel $x$
eindigt in cel $\pi(x)$.
\begin{enumerate}[resume]
  \item Toon dat een programma uitgevoerd vanuit $\sigma$ eindigt
        in $\sigma \circ \pi^{-1}$; dat twee programma's na
        elkaar uitvoeren hun effecten samenstelt; en dat de
        verzameling $H$ van alle effecten een deelgroep is van
        $\mathfrak{S}_{15}$ (permutaties van de cellen $1,
        \dots, 15$) bevat in de alternerende groep $A_{15}$.
  \item (De elementaire ronde) Vanuit de leegte thuis, schuif de
        leegte rond het $2 \times 2$-blok rechtsonder: cellen
        $16 \to 12 \to 11 \to 15 \to 16$. Toon dat het effect de
        $3$-cykel $(11\ 12\ 15)$ is, en dat de omgekeerde ronde
        $(11\ 15\ 12)$ geeft. Beide liggen in $H$.
  \item (De grote ronde) Verifieer dat
        \[
        16 \to 15 \to 14 \to 13 \to 9 \to 5 \to 1 \to 2 \to 3
        \to 4 \to 8 \to 7 \to 6 \to 10 \to 11 \to 12 \to 16
        \]
        een gesloten wandeling is door alle zestien cellen
        (alleen aangrenzende stappen), en dat haar effect de
        $15$-cykel is
        \[
        \zeta = (15\ 12\ 11\ 10\ 6\ 7\ 8\ 4\ 3\ 2\ 1\ 5\ 9\ 13\
        14) .
        \]
        Schrijvend $x_0 = 15$, $x_1 = 12$, $x_2 = 11$, \dots,
        $x_{14} = 14$ voor haar cykelorde, controleer dat de
        omgekeerde elementaire ronde van vraag 10 precies $(x_0\
        x_1\ x_2)$ is.
  \item Bewijs de conjugatieformule in elke $\mathfrak{S}_n$:
        voor een permutatie $g$ en een $3$-cykel,
        \[
        g\,(a\ b\ c)\,g^{-1} = \bigl(g(a)\ g(b)\ g(c)\bigr),
        \]
        en merk op dat $H$, als groep, gesloten is onder
        conjugatie door haar eigen elementen.
  \item Leid af dat $H$ alle vijftien \emph{opeenvolgende}
        $3$-cykels van de grote ronde bevat:
        \[
        s_t = (x_t\ x_{t+1}\ x_{t+2}) \qquad (t \in \Z/15\Z,
        \text{ indices mod } 15).
        \]
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De alternerende groep voortbrengen.}
\begin{enumerate}[resume]
  \item (Lemma A) Zij $s$ en $t$ $3$-cykels waarvan de steunen
        precies twee punten delen, zeg steunen $\{a, b, c\}$ en
        $\{b, c, d\}$. Toon dat, na eventuele vervanging van $s$
        of $t$ door haar inverse (wat de voortgebrachte deelgroep
        niet verandert), het product $st$ een dubbele
        \omterm{def:b2:structures:sn}{transpositie} is; toon dat $A_4$ geen deelgroep van \omterm{def:b2:structures:generated}{orde}
        $6$ bevat \emph{(een deelgroep van index $2$ bevat elk
        kwadraat; tel de $3$-cykels onder de kwadraten)}; en
        concludeer dat $\langle s, t\rangle$ de hele alternerende
        groep van de vier letters $\{a, b, c, d\}$ is.
  \item (Lemma B) Zij $X$ een verzameling van $k \geq 4$ letters,
        $w \notin X$, en zij $G$ een deelgroep van zekere
        $\mathfrak{S}_n$ die elke even permutatie van $X$ bevat
        en één $3$-cykel $(u\ v\ w)$ met $u, v \in X$. Toon dat
        voor alle verschillende $a, b \in X$ er een \emph{even}
        permutatie $g$ van $X$ bestaat met $g(u) = a$, $g(v) =
        b$, en leid af $(a\ b\ w) \in G$.
  \item Leid af dat de groep $G$ van lemma B elke even permutatie
        van $X \cup \{w\}$ bevat \emph{(gebruik
        \cref{exo:b2:structures:6}: de $3$-cykels brengen voort)}.
        Dan, lemma's A en B aaneenrijgend langs de opeenvolgende
        $3$-cykels $s_0, s_1, \dots, s_{12}$ van vraag 13, bewijs
        dat $\langle s_0, \dots, s_{12}\rangle = A_{15}$.
  \item Concludeer dat $H = A_{15}$: \emph{elke even herschikking
        van de vijftien tegels is haalbaar door een programma},
        en $H$ heeft $15!/2 = 653\,837\,184\,000$ elementen.
  \item (De stelling van Johnson--Story, 1879) Stel vragen 6, 7,
        8 en 17 samen: de configuraties bereikbaar vanuit de
        opgeloste zijn \emph{precies} de $16!/2 =
        10\,461\,394\,944\,000$ configuraties met $I = +1$; en
        bereikbaarheid heeft precies \emph{twee} klassen, de
        klasse van de opgeloste configuratie en de klasse van
        Loyds $\sigma_L$. \emph{(Voor het tweede punt, herlabel
        de tegels $14$ en $15$: toon $\sigma \mapsto (14\ 15)
        \circ \sigma$ stuurt zetrijen naar zetrijen en
        verwisselt $\{I = +1\}$ met $\{I = -1\}$.)}
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Criteria, varianten, en het zicht van
boven.}
\begin{enumerate}[resume]
  \item (Het praktische criterium) Lees de vijftien tegels in
        leesvolgorde van hun cellen, de leegte overslaand, en zij
        $N$ het aantal inversies van deze lijst; zij $r$ de rij
        van de leegte geteld vanonder. Toon dat $I(\sigma) =
        (-1)^{N + r + 1}$, zodat $\sigma$ oplosbaar is desda $N
        + r$ oneven is.
  \item (Groepsacties) Een \emph{actie} van een groep $G$ op een
        verzameling $X$ is een afbeelding $G \times X \to X$,
        $(g, x) \mapsto g \cdot x$, met $e \cdot x = x$ en $g
        \cdot (h \cdot x) = (gh) \cdot x$; de \emph{baan} van $x$
        is $G \cdot x$, en de actie is \emph{vrij} wanneer $g
        \cdot x = x$ dwingt $g = e$. Toon dat $h \cdot \sigma =
        \sigma \circ h^{-1}$ een vrije actie van $H$ op de
        verzameling van leegte-thuis-configuraties definieert,
        dat haar banen precies de klassen van wederzijdse
        bereikbaarheid door programma's zijn, en herwin uit de
        baantelling dat deze configuraties in precies
        $15!\,/\,\abs H = 2$ klassen splijten.
  \item (De $3 \times 3$-obstructie) Toon dat het $3 \times
        3$-bord \emph{geen} gesloten wandeling toelaat die elke
        cel precies één keer bezoekt: de
        grote-ronde-strategie van Deel III faalt voor de
        achtpuzzel. \emph{(Schaakbord de negen cellen.)}
  \item (De reparatie) Op het $3 \times 3$-bord met cellen $1$
        tot $9$ in leesvolgorde en thuis $9$: bereken de effecten
        van de omtreksronde $9 \to 8 \to 7 \to 4 \to 1 \to 2 \to
        3 \to 6 \to 9$ (een $7$-cykel $\zeta'$ die het centrum
        $5$ fixeert) en van de hoekronde $9 \to 6 \to 5 \to 8 \to
        9$ (een $3$-cykel door het centrum). Conjugatie van de
        laatste door de machten van $\zeta'$ en aaneenrijgen van
        lemma's A en B, bewijs dat de programmagroep van de
        achtpuzzel heel $A_8$ is, en dus dat precies $9!/2 =
        181\,440$ van de $9! = 362\,880$ configuraties oplosbaar
        zijn.
  \item (Een arm bord) Zij nu het bord een enkele \omterm{def:b2:structures:sn}{cykel} van $n
        \geq 4$ cellen met $n - 1$ tegels. Toon dat de cyclische
        \omterm{def:b2:structures:generated}{orde} van de tegels invariant is, dat elke
        bereikbaarheidsklasse precies $n(n - 1)$ configuraties
        heeft \emph{(de klassen zijn de banen van een \omterm{def:b2:structures:generated}{cyclische
        groep} van \omterm{def:b2:structures:generated}{orde} $\operatorname{lcm}(n, n-1) =
        n(n-1)$)}, en dat er $(n - 2)!$ klassen zijn --- voor $n
        \geq 5$ veel meer dan $2$: op een dun bord vangt de
        pariteitsinvariant bijna niets, en regeert de meetkunde.
  \item Twee vonnissen door het criterium van vraag 19: de
        volledig omgekeerde tray (tegels $15, 14, \dots, 1$ in
        cellen $1$ tot $15$, leegte thuis) en de tray met de
        leegte in cel $1$ gevolgd door de tegels $15, 14,
        \dots, 1$ in cellen $2$ tot $16$. Welke is oplosbaar?
  \item (Synthese) Het bewijs heeft twee onafhankelijke pijlers:
        een \emph{invariant} ($I$, gebouwd uit het
        signatuurmorfisme) die toont dat hoogstens de helft van
        de configuraties bereikbaar is, en een \emph{expliciete
        voortbrengings}stelling ($H = A_{15}$) die toont dat
        minstens de helft dat is. In één zin elk, zeg waar het
        volgende binnentrad: de morfisme-eigenschap van
        $\varepsilon$; de stelling van Lagrange; de
        voortbrenging van $A_n$ door $3$-cykels; conjugatie.
        Formuleer het metaprincipe in één regel.
\end{enumerate}
\end{problem}
