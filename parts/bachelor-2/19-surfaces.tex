\chapter{Surfaces}\label{ch:b2:surfaces}

After curves, surfaces: two-parameter objects in $\R^3$. The
differential calculus of \cref{ch:b2:diffcalc} provides everything
we need --- partial derivatives give tangent vectors, the cross
product gives the normal, \omterm{def:b2:linalg:det}{determinants} give \omterm{def:b2:surfaces:area}{areas}. We define
regular \omterm{def:b2:surfaces:param}{parametrized surfaces}, their \omterm{def:b2:surfaces:tangent}{tangent planes}, and the
\emph{\omterm{def:b2:surfaces:fff}{first fundamental form}}, which encodes all \omterm{def:b2:curves:length}{length} and \omterm{def:b2:surfaces:area}{area}
measurements on the surface. Surfaces also arise as level sets
$f(x, y, z) = c$; the gradient then directs the normal.

\section{Parametrized surfaces}

\begin{definition}[Regular parametrized surface]\label{def:b2:surfaces:param}
Let $U \subseteq \R^2$ be \omterm{def:b2:metric:topology}{open}. A \emph{parametrized
surface}\index{parametrized surface} of
class $\mathcal{C}^k$ ($k \geq 1$) is a map $\sigma \colon U \to
\R^3$, $(u, v) \mapsto \sigma(u, v)$, of class $\mathcal{C}^k$. A
point is \emph{regular} if the partial derivative vectors
\[
\sigma_u(u, v) = \frac{\partial\sigma}{\partial u}(u,v),
\qquad
\sigma_v(u, v) = \frac{\partial\sigma}{\partial v}(u,v)
\]
are linearly independent, i.e.\ $\sigma_u \wedge \sigma_v \neq 0$;
the surface is \emph{regular} if every point is.
\end{definition}

\begin{example}[The three standard descriptions]\label{ex:b2:surfaces:standard}
\leavevmode
\begin{enumerate}
\item \emph{Graph:} $\sigma(u, v) = (u,\ v,\ f(u, v))$ for $f \in
\mathcal{C}^1(U)$. Always regular: $\sigma_u = (1, 0, f_u)$ and
$\sigma_v = (0, 1, f_v)$ are independent.
\item \emph{Sphere (spherical coordinates):} for the sphere of
radius $R$,
\[
\sigma(\theta, \varphi)
= (R\cos\theta\cos\varphi,\ R\sin\theta\cos\varphi,\
R\sin\varphi),
\qquad
(\theta, \varphi) \in \R \times
\bigl(-\tfrac\pi2, \tfrac\pi2\bigr),
\]
with $\theta$ the longitude and $\varphi$ the latitude. One checks
$\norm{\sigma_\theta \wedge \sigma_\varphi} = R^2\cos\varphi > 0$:
regular away from the poles (which this chart omits).
\item \emph{Level set:} $S = \{(x,y,z) : f(x, y, z) = c\}$ where
$f$ is $\mathcal{C}^1$ and $\nabla f \neq 0$ on $S$. Near each
point, one coordinate can be expressed as a function of the other
two by the implicit function theorem (\cref{ch:b2:diffcalc}), so
$S$ is locally a graph.
\end{enumerate}
\end{example}

\begin{example}[From level set to graph]
\label{ex:b2:surfaces:levelgraph}
The implicit function theorem in item 3 deserves one explicit
run. Take the sphere $x^2 + y^2 + z^2 = R^2$ near its north
pole $(0, 0, R)$: there $\frac{\partial f}{\partial z} = 2z =
2R \neq 0$, and solving for $z$ gives the graph chart
\[
z = \sqrt{R^2 - x^2 - y^2},
\qquad x^2 + y^2 < R^2 ,
\]
regular everywhere on its (\omterm{def:b2:metric:topology}{open}) domain --- including the
pole that the spherical chart missed. Near an equator point
like $(R, 0, 0)$ the same theorem solves for $x$ instead
($\frac{\partial f}{\partial x} = 2R \neq 0$). The rule of
thumb: a level surface is a graph over the coordinate plane
\emph{orthogonal to the largest component of the gradient},
and by covering the sphere with six such graph charts one
checks its smoothness everywhere with no trigonometry at
all.
\end{example}

\section{Tangent plane and normal}

\begin{definition}[Tangent plane]\label{def:b2:surfaces:tangent}
Let $\sigma$ be regular at $(u_0, v_0)$, $M_0 = \sigma(u_0, v_0)$.
The \emph{tangent plane}\index{tangent plane} $T_{M_0}S$ is the plane through $M_0$
directed by $\operatorname{Vect}(\sigma_u, \sigma_v)$ (partials at
$(u_0, v_0)$). The \emph{unit normal}\index{unit normal} is
\[
n(u_0, v_0)
= \frac{\sigma_u \wedge \sigma_v}
       {\norm{\sigma_u \wedge \sigma_v}} .
\]
\end{definition}

\begin{proposition}[Tangent vectors are velocity
vectors]\label{prop:b2:surfaces:velocity}
The direction of $T_{M_0}S$ is exactly the set of vectors
$\gamma'(0)$, where $\gamma = \sigma \circ c$ ranges over the
$\mathcal{C}^1$ curves drawn on the surface through $M_0$ (i.e.\
$c \colon (-\varepsilon, \varepsilon) \to U$ is $\mathcal{C}^1$
with $c(0) = (u_0, v_0)$).
\end{proposition}

\begin{proof}
If $c(t) = (u(t), v(t))$, the chain rule
(\cref{ch:b2:diffcalc}) gives
\[
\gamma'(0) = u'(0)\,\sigma_u + v'(0)\,\sigma_v
\in \operatorname{Vect}(\sigma_u, \sigma_v).
\]
Conversely, the vector $a\sigma_u + b\sigma_v$ is attained by the
curve $c(t) = (u_0 + at,\ v_0 + bt)$, which stays in the \omterm{def:b2:metric:topology}{open} set
$U$ for $\abs t$ small.
\end{proof}

\begin{example}[The helicoid's tangent plane]
\label{ex:b2:surfaces:helicoidtangent}
For the helicoid $\sigma(u, v) = (v\cos u,\ v\sin u,\ au)$,
at the point $\sigma(0, 1) = (1, 0, 0)$:
\[
\sigma_u = (0,\ 1,\ a), \qquad \sigma_v = (1,\ 0,\ 0),
\qquad
\sigma_u \wedge \sigma_v = (0,\ a,\ -1),
\]
so the \omterm{def:b2:surfaces:tangent}{tangent plane} is $a\,y = z$. It contains the whole
horizontal ruling $\{(t, 0, 0)\}$ (direction $\sigma_v$): as
for the cone of \cref{exo:b2:surfaces:1}, a surface ruled by
straight lines has each ruling lying inside the \omterm{def:b2:surfaces:tangent}{tangent
plane} along it. The other tangent direction $\sigma_u$ is the
velocity of the helix $u \mapsto \sigma(u, 1)$: one chart,
two drawn curves, and the whole \omterm{def:b2:surfaces:tangent}{tangent plane} is spanned ---
\cref{prop:b2:surfaces:velocity} in action.
\end{example}

\begin{proposition}[Normal of a level surface]\label{prop:b2:surfaces:gradient}
Let $S = \{f = c\}$ with $f$ of class $\mathcal{C}^1$ and $\nabla
f(M_0) \neq 0$. Then the \omterm{def:b2:surfaces:tangent}{tangent plane} of $S$ at $M_0$ is the
plane through $M_0$ orthogonal to $\nabla f(M_0)$:
\[
T_{M_0}S :\quad
\langle \nabla f(M_0),\ M - M_0\rangle = 0 .
\]
\end{proposition}

\begin{proof}
For any curve $\gamma$ drawn on $S$ through $M_0$, $f(\gamma(t)) =
c$ identically, so the chain rule gives $\langle
\nabla f(M_0), \gamma'(0)\rangle = 0$: all velocity vectors are
orthogonal to the gradient, so the tangent direction is contained
in the plane $\nabla f(M_0)^\perp$. Both are $2$-dimensional
subspaces --- the tangent direction because $S$ is locally a
regular graph (\cref{ex:b2:surfaces:standard}), the orthogonal
complement because $\nabla f(M_0) \neq 0$ --- hence they are
equal.
\end{proof}

\begin{example}\label{ex:b2:surfaces:spheretangent}
For the sphere $x^2 + y^2 + z^2 = R^2$: $\nabla f = 2(x, y, z)$,
so the \omterm{def:b2:surfaces:tangent}{tangent plane} at $M_0$ is orthogonal to the radius
$\vect{OM_0}$ --- the classical fact that radius and \omterm{def:b2:surfaces:tangent}{tangent plane}
are perpendicular, with equation $\langle M_0, M\rangle = R^2$.
\end{example}

\begin{example}[Graph tangent plane]\label{ex:b2:surfaces:graphtangent}
For $z = f(x, y)$ at $(x_0, y_0)$: applying
\cref{prop:b2:surfaces:gradient} to $F(x,y,z) = f(x,y) - z$,
\[
z = f(x_0, y_0)
+ f_x(x_0, y_0)(x - x_0)
+ f_y(x_0, y_0)(y - y_0) ,
\]
the \omterm{def:b2:affine:subspace}{affine} part of the first-order Taylor expansion --- the
\omterm{def:b2:surfaces:tangent}{tangent plane} is the graph of the \omterm{def:b2:diffcalc:differential}{differential}, as it must be.
\end{example}

\begin{example}[The closest point of a
surface]\label{ex:b2:surfaces:closest}
Which point of the paraboloid $z = x^2 + y^2$ is nearest to
$P = (0, 0, 1)$? Minimize the squared distance along the
surface: with $\rho^2 = x^2 + y^2$,
\[
g(\rho^2) = \rho^2 + (\rho^2 - 1)^2,
\qquad
g'(\rho^2) = 1 + 2(\rho^2 - 1) = 0
\iff \rho^2 = \tfrac12 ,
\]
giving the circle of points at height $z = \frac12$ and
distance $\sqrt{\tfrac12 + \tfrac14} = \frac{\sqrt3}2$. The
geometric signature of minimality: at such a point $M$, the
vector $\vect{MP}$ must be \emph{normal} to the surface ---
otherwise sliding along a drawn curve with velocity having a
component toward $P$ would decrease the distance. Check:
$\nabla(z - x^2 - y^2) = (-2x, -2y, 1)$ at $M = (x, y,
\tfrac12)$, while $\vect{MP} = (-x, -y, \tfrac12) =
\tfrac12(-2x, -2y, 1)$: parallel, as predicted. The
first-order condition ``foot of the perpendicular'' is the
same one that will drive extrema on level sets in
\cref{exo:b2:surfaces:12}.
\end{example}

\begin{example}[Tangent planes of quadrics: the
polarization rule]\label{ex:b2:surfaces:polarization}
Let $S : xy + yz + zx = 1$ and $M_0 = (1, 1, 0) \in S$. Here
$\nabla f = (y + z,\ x + z,\ x + y)$, so $\nabla f(M_0) = (1,
1, 2)$ and the \omterm{def:b2:surfaces:tangent}{tangent plane} is
\[
(x - 1) + (y - 1) + 2z = 0,
\qquad\text{i.e.}\qquad x + y + 2z = 2 .
\]
The same answer comes from the \emph{\omterm{def:b2:quadratic:def}{polarization} rule} that
generalizes \cref{ex:b2:surfaces:spheretangent} and
\cref{exo:b2:surfaces:2}: in the \omterm{pb:b2:surfaces:1}{quadric}'s equation, replace
$x^2$ by $x_0x$, and each product $xy$ by $\frac{x_0y +
y_0x}2$ (and cyclically):
\[
\frac{x_0y + y_0x}2 + \frac{y_0z + z_0y}2 + \frac{z_0x +
x_0z}2 = 1
\;\xrightarrow{\ M_0 = (1,1,0)\ }\;
\frac{x + y}2 + \frac z2 + \frac z2 = 1 ,
\]
which is $x + y + 2z = 2$ again. The rule works because
$\nabla$ of a \omterm{def:b2:quadratic:def}{quadratic form} is the associated bilinear form
evaluated against the base point --- tangency to a \omterm{pb:b2:surfaces:1}{quadric} is
\omterm{def:b2:quadratic:def}{polarization}, one more face of \cref{ch:b2:quadratic}.
\end{example}

\section{The first fundamental form}

\begin{definition}[First fundamental form]\label{def:b2:surfaces:fff}
Let $\sigma \colon U \to \R^3$ be a regular $\mathcal{C}^1$
surface. Its \emph{first fundamental form}\index{first fundamental
form} at $(u,v)$ is the
positive definite \omterm{def:b2:quadratic:def}{quadratic form} on $\R^2$
\[
I(h, k) = \norm{h\,\sigma_u + k\,\sigma_v}^2
= E\,h^2 + 2F\,hk + G\,k^2,
\]
where
\[
E = \norm{\sigma_u}^2,
\qquad
F = \langle \sigma_u, \sigma_v\rangle,
\qquad
G = \norm{\sigma_v}^2 .
\]
\end{definition}

\begin{remark}
$I$ is the restriction of the ambient Euclidean scalar product to
the \omterm{def:b2:surfaces:tangent}{tangent plane}, read in the basis $(\sigma_u, \sigma_v)$: it is
positive definite precisely because $\sigma_u, \sigma_v$ are
independent (\cref{ch:b2:quadratic}). Every metric quantity on the
surface --- \omterm{def:b2:curves:length}{lengths} of drawn curves, angles between them, \omterm{def:b2:surfaces:area}{areas} ---
is computed from $E, F, G$ alone. Two surfaces with the same $E,
F, G$ in suitable parameters are \emph{isometric} even if they sit
differently in space: this is the starting point of intrinsic
geometry.
\end{remark}

\begin{example}[Angles between coordinate
curves]\label{ex:b2:surfaces:coordangle}
The \omterm{def:b2:surfaces:fff}{first fundamental form} also measures angles: the
coordinate curves $u \mapsto \sigma(u, v_0)$ and $v \mapsto
\sigma(u_0, v)$ meet at the angle $\theta$ with
\[
\cos\theta = \frac{\langle\sigma_u,
\sigma_v\rangle}{\norm{\sigma_u}\,\norm{\sigma_v}}
= \frac{F}{\sqrt{EG}} :
\]
the single coefficient $F$ decides orthogonality of the
parameter net. For the sphere chart and the helicoid, $F =
0$: meridians cut parallels, and helices cut the horizontal
rulings, at right angles --- which is why their \omterm{def:b2:surfaces:area}{area}
integrands collapsed to $\sqrt{EG}$. For a graph chart,
$F = f_xf_y$ vanishes only where a partial derivative does:
the coordinate net of a tilted graph is \emph{not}
orthogonal, even though the $(x, y)$-net downstairs is. When
computations on a surface look heavy, the first move is to
seek a chart with $F = 0$.
\end{example}

\begin{example}[The saddle chart]\label{ex:b2:surfaces:saddlechart}
For the saddle $z = xy$ with chart $\sigma(u, v) = (u, v,
uv)$:
\[
\sigma_u = (1, 0, v), \qquad \sigma_v = (0, 1, u),
\qquad
E = 1 + v^2, \quad F = uv, \quad G = 1 + u^2 ,
\]
and $EG - F^2 = 1 + u^2 + v^2 > 0$: regular everywhere. The
two coordinate curves through a point are \emph{straight
lines} of $\R^3$ (fix $u$ or fix $v$: the rulings of the
doubly ruled saddle), yet $F \neq 0$ off the axes: rulings
through a generic point are not orthogonal. Both rulings lie
in the \omterm{def:b2:surfaces:tangent}{tangent plane}, which they span --- so the \omterm{def:b2:surfaces:tangent}{tangent
plane} cuts the surface along two whole lines, the extreme
opposite of the sphere, whose \omterm{def:b2:surfaces:tangent}{tangent planes} touch at one
point only. The sign of the ``second-order contact'' between
a surface and its \omterm{def:b2:surfaces:tangent}{tangent planes} is a \omterm{thm:b2:curves:frenet2d}{curvature} story,
taken up in the Year 3 volume.
\end{example}

\begin{proposition}[Length of a curve drawn on a
surface]\label{prop:b2:surfaces:curvelength}
If $\gamma(t) = \sigma(u(t), v(t))$, $t \in [a, b]$, is
$\mathcal{C}^1$, then
\[
L(\gamma) = \int_a^b
\sqrt{E\,u'^2 + 2F\,u'v' + G\,v'^2}\;\dd t ,
\]
with $E, F, G$ evaluated at $(u(t), v(t))$.
\end{proposition}

\begin{proof}
$\gamma' = u'\sigma_u + v'\sigma_v$ by the chain rule, so
$\norm{\gamma'}^2 = I(u', v') = Eu'^2 + 2Fu'v' + Gv'^2$; integrate
$\norm{\gamma'}$ (\cref{def:b2:curves:length}).
\end{proof}

\begin{example}[Why airliners fly over the pole]\label{ex:b2:surfaces:polarroute}
Two airports sit at latitude $\varphi_0$ and opposite
longitudes: $A = \sigma(0, \varphi_0)$ and $B = \sigma(\pi,
\varphi_0)$ on the sphere of radius $R$. Along the parallel
($\varphi \equiv \varphi_0$), the \omterm{def:b2:curves:length}{length} is
$\int_0^\pi R\cos\varphi_0\,\dd\theta = \pi R\cos\varphi_0$.
Along the route over the pole (up the meridian $\theta = 0$,
down the meridian $\theta = \pi$), it is $2R(\frac\pi2 -
\varphi_0)$. At latitude $\varphi_0 = \frac\pi3$ (sixty
degrees): parallel route $\pi R/2 \approx 1.571\,R$, polar
route $\pi R/3 \approx 1.047\,R$ --- a third shorter. In fact
$\pi\cos\varphi_0 \geq \pi - 2\varphi_0$ on
$\intcc0{\pi/2}$ (the function $\pi\cos\varphi - \pi +
2\varphi$ vanishes at both ends and its derivative $2 -
\pi\sin\varphi$ changes sign once, so it is first increasing
then decreasing, hence nonnegative): the polar route never
loses. The \omterm{def:b2:surfaces:fff}{first fundamental form} turned a navigation question
into two one-line integrals; \cref{exo:b2:surfaces:6} pushes
the idea to a genuine minimality proof for meridians.
\end{example}

\begin{lemma}[Lagrange identity]\label{lem:b2:surfaces:lagrange}
For all $a, b \in \R^3$:
$\norm{a \wedge b}^2 = \norm a^2 \norm b^2 - \langle a, b\rangle^2$.
In particular
\[
\norm{\sigma_u \wedge \sigma_v} = \sqrt{EG - F^2}.
\]
\end{lemma}

\begin{proof}
Both sides are unchanged if we replace $b$ by its component $b_\perp
= b - \frac{\langle a, b\rangle}{\norm a^2}a$ orthogonal to $a$
(for $a \neq 0$; the case $a = 0$ is trivial): the left side
because $a \wedge a = 0$, the right side by expanding $\norm
{b_\perp}^2 = \norm b^2 - \frac{\langle a,b\rangle^2}{\norm a^2}$
and $\langle a, b_\perp\rangle = 0$. So it suffices to prove the
identity for orthogonal $a, b$, where it reads $\norm{a \wedge
b} = \norm a \norm b$: true, since for orthogonal vectors the
cross product has \omterm{def:b2:nvs:norm}{norm} $\norm a\norm b\,\abs{\sin\frac\pi2}$. The
displayed formula is the case $a = \sigma_u$, $b = \sigma_v$.
\end{proof}

\begin{remark}
The Lagrange identity says $EG - F^2 = \det\begin{psmallmatrix}
E & F\\ F & G\end{psmallmatrix}$ is the \emph{Gram
\omterm{def:b2:linalg:det}{determinant}} of $(\sigma_u, \sigma_v)$: the squared \omterm{def:b2:surfaces:area}{area} of
the parallelogram they span. Regularity, positive
definiteness of the \omterm{def:b2:surfaces:fff}{first fundamental form}, and positivity of
the Gram \omterm{def:b2:linalg:det}{determinant} are three phrasings of one condition ---
which is why the \omterm{def:b2:surfaces:area}{area} integrand below never vanishes on a
regular chart.
\end{remark}

\begin{definition}[Area]\label{def:b2:surfaces:area}
Let $\sigma \colon U \to \R^3$ be a regular injective
$\mathcal{C}^1$ surface and $K \subseteq U$ a \omterm{def:b2:metric:compact}{compact} domain on
which double integrals make sense (\cref{ch:b2:multint}). The
\emph{area}\index{area!of a surface} of the piece $\sigma(K)$ is
\[
\mathcal{A}
= \iint_K \norm{\sigma_u \wedge \sigma_v}\,\dd u\,\dd v
= \iint_K \sqrt{EG - F^2}\;\dd u\,\dd v .
\]
\end{definition}

\begin{remark}[Why this formula]
The rectangle $[u, u + \dd u] \times [v, v + \dd v]$ is mapped, to
first order, onto the parallelogram spanned by $\sigma_u\,\dd u$
and $\sigma_v\,\dd v$, whose \omterm{def:b2:surfaces:area}{area} is $\norm{\sigma_u \wedge
\sigma_v}\,\dd u\,\dd v$: the definition integrates the local
area-distortion factor, exactly as \omterm{def:b2:curves:length}{arc length} integrates the local
speed. Consistency with \omterm{def:b2:curves:reparam}{change of parameters} is
\cref{exo:b2:surfaces:8}; consistency with the change-of-variables
formula for double integrals is discussed in
\cref{ch:b2:multint}.
\end{remark}

\begin{example}[Two different graphs, one
area]\label{ex:b2:surfaces:samearea}
Over the unit disk, compare the bowl $z = \frac12(x^2 + y^2)$
and the saddle $z = xy$. Their \omterm{def:b2:surfaces:area}{area} integrands
(\cref{exo:b2:surfaces:5}) are
\[
\sqrt{1 + x^2 + y^2}
\qquad\text{and}\qquad
\sqrt{1 + y^2 + x^2} :
\]
identical. The two surfaces --- one curving the same way in
all directions, the other saddle-shaped --- have exactly
equal \omterm{def:b2:surfaces:area}{areas} over every domain, $\frac{2\pi}3(2\sqrt2 - 1)$
over the unit disk. The \omterm{def:b2:surfaces:area}{area} element only sees the
\emph{\omterm{def:b2:curves:length}{length}} of the gradient, not the arrangement of the
bending; telling the bowl from the saddle requires
second-order data (the sign structure exhibited in
\cref{fig:b2:surfaces:saddle}), which no amount of \omterm{def:b2:surfaces:area}{area}
measurement detects. \omterm{def:b2:surfaces:fff}{First fundamental form}: metric, blind
to shape; the shape-seeing second form belongs to Year 3.
\end{example}

\begin{example}[Area of the sphere]\label{ex:b2:surfaces:spherearea}
For the spherical chart of \cref{ex:b2:surfaces:standard}:
\[
\sigma_\theta = R(-\sin\theta\cos\varphi,\
\cos\theta\cos\varphi,\ 0),
\qquad
\sigma_\varphi = R(-\cos\theta\sin\varphi,\
-\sin\theta\sin\varphi,\ \cos\varphi),
\]
so $E = R^2\cos^2\varphi$, $F = 0$, $G = R^2$ and $\sqrt{EG - F^2}
= R^2\cos\varphi$. Hence
\[
\mathcal{A}
= \int_{-\pi/2}^{\pi/2}\!\!\int_0^{2\pi}
R^2\cos\varphi\;\dd\theta\,\dd\varphi
= 2\pi R^2\,\bigl[\sin\varphi\bigr]_{-\pi/2}^{\pi/2}
= \boxed{4\pi R^2} .
\]
\end{example}

\begin{example}[The cone, checked against the
school formula]\label{ex:b2:surfaces:conearea}
For the cone $z = \sqrt{x^2 + y^2}$ over the annulus $a \leq
\rho \leq b$, the graph formula of
\cref{exo:b2:surfaces:5} gives $1 + f_x^2 + f_y^2 = 2$
(compute $f_x = x/\rho$, $f_y = y/\rho$), so
\[
\mathcal A = \sqrt2\,\pi\,(b^2 - a^2) .
\]
Consistency check with the slant-height formula $\pi\rho\ell$
of \cref{ex:b2:surfaces:revolution}: the full cones of base
radii $b$ and $a$ have lateral \omterm{def:b2:surfaces:area}{areas} $\pi b\cdot b\sqrt2$ and
$\pi a\cdot a\sqrt2$, whose difference is exactly
$\sqrt2\pi(b^2 - a^2)$. Two charts, two formulas, one \omterm{def:b2:surfaces:area}{area}
--- the invariance proved in \cref{exo:b2:surfaces:8}, seen
in the wild.
\end{example}

\begin{remark}[Sanity checks for areas]
Three instant checks catch most errors in an \omterm{def:b2:surfaces:area}{area}
computation. \emph{Scaling}: dilating a surface by $\lambda$
multiplies $E, F, G$ by $\lambda^2$ and the \omterm{def:b2:surfaces:area}{area} by
$\lambda^2$ --- an answer whose dependence on $R$ is not
quadratic (like $4\pi R^2$) is wrong. \emph{Positivity of the
element}: $\sqrt{EG - F^2}$ must be strictly positive on the
chart's interior; a vanishing value flags a chart degeneracy,
to be excised as in the sphere's poles. \emph{Symmetry}: a
computation over a \omterm{def:b2:quadratic:adjoint}{symmetric} piece must be consistent with
summing its congruent parts --- the hemisphere had better
give $2\pi R^2$.
\end{remark}

\begin{example}[Surface of revolution]\label{ex:b2:surfaces:revolution}
Rotate the curve $z \mapsto (r(z), 0, z)$, $r > 0$ of class
$\mathcal{C}^1$, around the $z$-axis:
\[
\sigma(\theta, z) = (r(z)\cos\theta,\ r(z)\sin\theta,\ z).
\]
Then $E = r(z)^2$, $F = 0$, $G = 1 + r'(z)^2$, so
\[
\mathcal{A} = \int_a^b\!\!\int_0^{2\pi}
r(z)\sqrt{1 + r'(z)^2}\;\dd\theta\,\dd z
= 2\pi\int_a^b r(z)\sqrt{1 + r'(z)^2}\,\dd z ,
\]
the classical formula (circumference $2\pi r$ times the slant
\omterm{def:b2:curves:length}{length} element). For the cone $r(z) = kz$, $z \in [0, h]$: $\mathcal
A = 2\pi k\sqrt{1 + k^2}\,\frac{h^2}{2} = \pi \rho \ell$ with
$\rho = kh$ the base radius and $\ell = h\sqrt{1 + k^2}$ the slant
height --- the school formula, now derived rather than admitted.
\end{example}

\begin{example}[The catenoid]\label{ex:b2:surfaces:catenoid}
Rotate the catenary $r(z) = \cosh z$, $z \in \intcc{-1}{1}$,
around its axis: the resulting \emph{catenoid} has, by the
revolution formula and $1 + \sinh^2 z = \cosh^2 z$,
\[
\mathcal A = 2\pi\int_{-1}^1\cosh z\,\sqrt{1 + \sinh^2z}\,
\dd z = 2\pi\int_{-1}^1\cosh^2z\,\dd z
= \pi\bigl[z + \sinh z\cosh z\bigr]_{-1}^{1},
\]
that is
\[
\mathcal A = 2\pi + \pi\sinh 2 \approx 17.68 .
\]
The slant factor $\sqrt{1 + r'^2}$ merged with the radius
into a perfect square --- the same identity that made the
catenary's \omterm{def:b2:curves:length}{arc length} elementary in the curves chapter. This
is no accident of algebra: among all surfaces of revolution
spanning the two boundary circles, the catenoid minimizes
\omterm{def:b2:surfaces:area}{area} (it is the shape of a soap film between two rings), and
this variational property is precisely what singles out
$\cosh$; the Year 3 volume proves it with the calculus of
variations.
\end{example}

\begin{remark}[Common pitfalls]
(i) \emph{Chart singularities are not surface singularities}:
the spherical chart degenerates at the poles
($\cos\varphi = 0$), but the sphere is perfectly smooth there
--- another chart (exchange the roles of the axes) is regular
at the poles. Before declaring a point singular, try a second
parametrization. (ii) \emph{Regularity of $\sigma$ concerns
the parametrization, not the image}: $\sigma(u, v) = (u^3, v,
0)$ is non-regular along $u = 0$ although its image is a
plane. (iii) The \omterm{def:b2:surfaces:area}{area} formula requires $\sigma$
\emph{injective} on $K$: a chart covering a piece twice
counts it twice ($\theta$ running over $\intcc0{4\pi}$
doubles the sphere's \omterm{def:b2:surfaces:area}{area}). (iv) The \omterm{def:b2:surfaces:tangent}{unit normal} is defined
up to sign by the surface but is \emph{chosen} by the chart
(order of $u, v$); statements involving orientation must fix
that choice. (v) Finally, $EG - F^2 > 0$ is not an extra
hypothesis: it is exactly regularity, by the Lagrange
identity --- if it vanishes somewhere, the problem is the
chart, and no \omterm{def:b2:surfaces:area}{area} or tangent-plane formula applies there.
\end{remark}

\begin{remark}[Perspectives within this volume]
Forward links from here. The \omterm{def:b2:surfaces:area}{area} element $\sqrt{EG -
F^2}\,\dd u\,\dd v$ is a two-dimensional Jacobian in
disguise, and \cref{ch:b2:multint} makes the analogy exact
with the change-of-variables theorem --- the surface
integrals there are this chapter's \omterm{def:b2:surfaces:area}{areas} with an integrand on
board. The \omterm{def:b2:surfaces:fff}{first fundamental form} is a field of positive
\omterm{def:b2:quadratic:def}{quadratic forms}, handled \omterm{def:b2:funcseq:def}{pointwise} by the tools of
\cref{ch:b2:quadratic}, whose spectral theorem also powers
this chapter's weekend classification of \omterm{pb:b2:surfaces:1}{quadrics}. And the
normal line drives extremum problems on constraint sets
(\cref{ex:b2:surfaces:closest}), the geometric germ of the
Lagrange multiplier method sketched with
\cref{thm:b2:diffcalc:extrema}.
\end{remark}

\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b2:surfaces:1}
Show that the \omterm{def:b2:surfaces:tangent}{tangent planes} of the cone $z = \sqrt{x^2 + y^2}$
(minus its apex) all pass through the apex. \emph{(Parametrize by
$\sigma(\theta, r) = (r\cos\theta, r\sin\theta, r)$, $r > 0$.)}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:surfaces:2}
Find the \omterm{def:b2:surfaces:tangent}{tangent plane} of the ellipsoid $\frac{x^2}{a^2} +
\frac{y^2}{b^2} + \frac{z^2}{c^2} = 1$ at a point $(x_0, y_0,
z_0)$ of the surface.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:surfaces:3}
Compute $E, F, G$ for the helicoid $\sigma(u, v) = (v\cos u,\
v\sin u,\ au)$, $a > 0$, and the \omterm{def:b2:surfaces:area}{area} of the piece $0 \leq u \leq
2\pi$, $0 \leq v \leq 1$, as an integral (evaluate it using
$\int\sqrt{v^2 + a^2}\,\dd v = \frac12\bigl(v\sqrt{v^2+a^2} +
a^2\ln(v + \sqrt{v^2 + a^2})\bigr) + C$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:surfaces:4}
(Torus) Parametrize the torus obtained by rotating the circle of
center $(R, 0, 0)$ and radius $r < R$ in the $xz$-plane around the
$z$-axis:
\[
\sigma(\theta, \psi) = \bigl((R + r\cos\psi)\cos\theta,\
(R + r\cos\psi)\sin\theta,\ r\sin\psi\bigr).
\]
Compute $E, F, G$, check regularity, and show that the \omterm{def:b2:surfaces:area}{area} is
$4\pi^2 R r$ (Pappus: mean circumference $2\pi R$ times circle
\omterm{def:b2:curves:length}{length} $2\pi r$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:surfaces:5}
Show that the \omterm{def:b2:surfaces:area}{area} of the graph of $f \in \mathcal{C}^1(K)$ is
$\iint_K \sqrt{1 + f_x^2 + f_y^2}\,\dd x\,\dd y$, and compute it
for the paraboloid piece $z = \frac12(x^2 + y^2)$ over the disk
$x^2 + y^2 \leq 1$ (polar coordinates,
\cref{ch:b2:multint}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:surfaces:6}
A drawn curve $\gamma(t) = \sigma(u(t), v(t))$ on the sphere of
radius $R$ (spherical chart) has $u = \theta(t)$, $v =
\varphi(t)$. Write its \omterm{def:b2:curves:length}{length} as an integral in $\theta, \varphi$
and prove that among curves joining two points of the same
meridian $\theta = \theta_0$, the meridian arc is the shortest.
\emph{(Bound the integrand below by $R\abs{\varphi'}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:surfaces:7}
(Normal lines of a sphere) Let $S$ be a regular level surface
$\{f = c\}$, \omterm{def:b2:metric:connected}{connected}, all of whose normal lines pass through a
fixed point $\Omega$. Show that $S$ is contained in a sphere
centered at $\Omega$. \emph{(Show that $\norm{M - \Omega}^2$ has
zero derivative along every curve drawn on $S$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:surfaces:8}
(\omterm{def:b2:surfaces:area}{Area} is geometric) Let $\Phi \colon U' \to U$ be a $\mathcal{C}^1$
diffeomorphism between \omterm{def:b2:metric:topology}{open} sets of $\R^2$ and $\tilde\sigma =
\sigma \circ \Phi$. Show that
\[
\norm{\tilde\sigma_{u'} \wedge \tilde\sigma_{v'}}
= \abs{\det J_\Phi}\,
\norm{(\sigma_u \wedge \sigma_v)\circ\Phi} ,
\]
and deduce, using the change-of-variables formula of
\cref{ch:b2:multint}, that the \omterm{def:b2:surfaces:area}{area} of
\cref{def:b2:surfaces:area} does not depend on the chosen regular
parametrization.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:surfaces:9}
(Archimedes' hat-box theorem) On the sphere of radius $R$, the
\emph{zone} between the latitudes with $z_1 \leq z \leq z_2$
($-R \leq z_1 < z_2 \leq R$) has \omterm{def:b2:surfaces:area}{area} $2\pi R\,(z_2 - z_1)$:
prove it with the spherical chart, and conclude that a zone's
\omterm{def:b2:surfaces:area}{area} depends only on its height --- slicing an orange into
equal-thickness slices gives equal amounts of peel.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:surfaces:10}
Show that every normal line of a surface of revolution
$\sigma(\theta, z) = (r(z)\cos\theta,\ r(z)\sin\theta,\ z)$
($r > 0$ of class $\mathcal C^1$) meets the axis of
revolution, and locate the intersection point.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:surfaces:11}
(Unrolling the cylinder) The chart $\sigma(u, v) = (\cos u,\
\sin u,\ v)$ of the unit cylinder has $E = G = 1$, $F = 0$:
verify this, and explain why every drawn curve $t \mapsto
\sigma(u(t), v(t))$ has the same \omterm{def:b2:curves:length}{length} as the plane curve $t
\mapsto (u(t), v(t))$. Deduce that the helix from $(1, 0, 0)$
to $(1, 0, 2\pi c)$ making one turn has \omterm{def:b2:curves:length}{length}
$2\pi\sqrt{1 + c^2}$, and that no drawn curve with the same
endpoints and one full turn is shorter.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:surfaces:12}
Let $S = \{f = c\}$ be a \omterm{def:b2:metric:compact}{compact} regular level surface and
$M_0 \in S$ a point at maximal distance from the origin. Show
that $\nabla f(M_0)$ is collinear with $\vect{OM_0}$ --- the
normal at the farthest point is radial. Apply to the ellipsoid
$\frac{x^2}{a^2} + \frac{y^2}{b^2} + \frac{z^2}{c^2} = 1$ ($a >
b > c > 0$): find all points where the normal is radial, and
identify the farthest ones.
\end{exercise}

\section{Problem: the classification of quadrics of
$\R^3$}

\begin{problem}[{Weekend problem --- every quadric surface,
sorted by the spectral theorem}]\label{pb:b2:surfaces:1}
A \emph{quadric}\index{quadric} is the zero set in $\R^3$ of a
degree-two polynomial
\[
q(X) = X^{\mathsf T}\!AX + 2\,\langle b, X\rangle + c,
\qquad A \in \mathcal S_3(\R),\ A \neq 0,\ b \in \R^3,\
c \in \R .
\]
The surfaces of this chapter's figures --- spheres,
ellipsoids, saddles, cones, cylinders --- are all \omterm{pb:b2:surfaces:1}{quadrics}.
This problem classifies them \emph{all}: the spectral theorem
(\cref{thm:b2:quadratic:spectral}) straightens the quadratic
part, \omterm{def:b2:affine:subspace}{affine} translations (\cref{ch:b2:affine}) absorb the
linear part, and what remains is a short, complete list of
normal forms.

\medskip
\noindent\textbf{Part I --- The reduction machine.}
\begin{enumerate}
  \item Let $X = PY + t$ with $P \in O(3)$ and $t \in \R^3$ (a
        rigid change of coordinates). Show that $q(PY + t) =
        Y^{\mathsf T}\!A'Y + 2\langle b', Y\rangle + c'$ with
        \[
        A' = P^{\mathsf T}\!AP, \qquad
        b' = P^{\mathsf T}(At + b), \qquad
        c' = q(t) .
        \]
        Deduce that the \omterm{def:b2:reduction:eigen}{spectrum} of $A$ (hence its rank and
        signature) is a rigid invariant of the equation, and
        explain why the equation of a given \omterm{pb:b2:surfaces:1}{quadric} is only
        determined up to a nonzero scalar factor.
  \item Using the spectral theorem, show that after a rotation
        the equation becomes $\sum_i \lambda_i y_i^2 +
        2\sum_i\beta_iy_i + c = 0$ with $\lambda_1, \lambda_2,
        \lambda_3$ the \omterm{def:b2:reduction:eigen}{eigenvalues} of $A$.
  \item For every $i$ with $\lambda_i \neq 0$, absorb
        $\beta_iy_i$ by a translation ($y_i \mapsto y_i -
        \beta_i/\lambda_i$). Write the reduced equation when
        $\operatorname{rank} A = r$: $\sum_{i\leq r}\lambda_i
        z_i^2 + 2\sum_{i > r}\beta_iz_i + c'' = 0$.
  \item A \emph{center} of the \omterm{pb:b2:surfaces:1}{quadric} of equation $q = 0$ is
        a point $\Omega$ with $q(2\Omega - X) = q(X)$ for all
        $X$: the point reflection in $\Omega$ preserves the
        equation, hence the surface. Show that $q(2\Omega - X)
        - q(X) = -4\langle A\Omega + b, X\rangle + 4\langle
        A\Omega + b, \Omega\rangle$, and deduce: the centers
        are exactly the solutions of $A\Omega = -b$; they
        exist iff $b \in \operatorname{im}A$, and the center
        is unique iff $A$ is invertible.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Central \omterm{pb:b2:surfaces:1}{quadrics}
($\operatorname{rank}A = 3$).} Here the reduced equation is
$\lambda_1z_1^2 + \lambda_2z_2^2 + \lambda_3z_3^2 = \delta$.
\begin{enumerate}[resume]
  \item Multiplying by $-1$ if needed, assume at least two
        $\lambda_i > 0$. Enumerate the possibilities:
        signature $(3, 0)$ with $\delta > 0$, $= 0$, $< 0$,
        and signature $(2, 1)$ with $\delta > 0$, $= 0$, $<
        0$; name the six resulting sets (ellipsoid, point,
        empty set, one-sheet hyperboloid, cone, two-sheet
        hyperboloid) and put each in its Euclidean normal
        form ($\frac{x^2}{a^2} + \frac{y^2}{b^2} +
        \frac{z^2}{c^2} = 1$, etc.).
  \item Classify $x^2 + y^2 + z^2 + 4xy + 4yz + 4zx = 1$:
        show $A = 2J - I$ with $J$ the all-ones matrix,
        compute the \omterm{def:b2:reduction:eigen}{spectrum} $\{5, -1, -1\}$, and identify a
        two-sheet hyperboloid of revolution about the axis
        $\R(1,1,1)$.
  \item (Rulings) For the one-sheet hyperboloid
        $\frac{x^2}{a^2} + \frac{y^2}{b^2} - \frac{z^2}{c^2}
        = 1$, factor
        \[
        \Bigl(\frac xa - \frac zc\Bigr)
        \Bigl(\frac xa + \frac zc\Bigr)
        = \Bigl(1 - \frac yb\Bigr)\Bigl(1 + \frac yb\Bigr)
        \]
        and produce two one-parameter families of straight
        lines lying on the surface.
  \item Show that through every point of the one-sheet
        hyperboloid passes exactly one line of each family:
        the surface is \emph{doubly ruled}.
  \item The \emph{asymptotic cone} of the one-sheet
        hyperboloid is $C : \frac{x^2}{a^2} + \frac{y^2}{b^2}
        - \frac{z^2}{c^2} = 0$. With $\rho^2 = \frac{x^2}{a^2}
        + \frac{y^2}{b^2}$, show that any point of the
        hyperboloid is at distance at most $c\bigl(\rho -
        \sqrt{\rho^2 - 1}\bigr) = \frac{c}{\rho +
        \sqrt{\rho^2 - 1}}$ from $C$, so the surface hugs its
        cone at infinity. What are the sections of the
        hyperboloid by the planes $x = \pm a$?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Rank $2$ and rank $1$:
paraboloids, cylinders, planes.}
\begin{enumerate}[resume]
  \item Suppose $\operatorname{rank}A = 2$, say $\lambda_1,
        \lambda_2 \neq 0 = \lambda_3$. Starting from question
        3, split into two cases according to $\beta_3 \neq 0$
        (no center, by question 4) or $\beta_3 = 0$ (a line
        of centers), and reduce to
        \[
        \lambda_1z_1^2 + \lambda_2z_2^2 + 2\beta_3z_3 = 0
        \qquad\text{or}\qquad
        \lambda_1z_1^2 + \lambda_2z_2^2 + c'' = 0 :
        \]
        elliptic/hyperbolic paraboloids in the first case,
        cylinders over central conics (or pairs of
        intersecting planes, a line, the empty set) in the
        second.
  \item Show that the saddle $z = xy$ is a hyperbolic
        paraboloid: rotate by $\pi/4$ in the $xy$-plane to
        reach $z = \tfrac12(u^2 - v^2)$, the surface of
        \cref{fig:b2:surfaces:saddle} up to scale.
  \item Show that the saddle $z = xy$ carries the two line
        families $\{x = x_0,\ z = x_0y\}$ and $\{y = y_0,\ z
        = xy_0\}$, with exactly one line of each through
        every point: the second doubly ruled \omterm{pb:b2:surfaces:1}{quadric}.
  \item Classify $x^2 + y^2 - 2x + 4y + 3 = 0$ in $\R^3$
        (complete the squares; identify a right circular
        cylinder, and give its axis and radius).
  \item Now let $\operatorname{rank}A = 1$, say $\lambda_1
        \neq 0 = \lambda_2 = \lambda_3$. Rotating within the
        kernel plane and translating, reduce to
        \[
        \lambda_1z_1^2 + 2\beta z_2 = 0 \quad (\beta \neq 0)
        \qquad\text{or}\qquad
        \lambda_1z_1^2 + c'' = 0 :
        \]
        a parabolic cylinder, or a pair of parallel planes, a
        double plane, or the empty set. Classify $(x + y)^2 =
        z$ completely (normal form, axis of translation
        invariance).
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The classification theorem.}
\begin{enumerate}[resume]
  \item Assemble Parts I--III into a theorem: \emph{every
        \omterm{pb:b2:surfaces:1}{quadric} of $\R^3$ is mapped by a rigid motion onto
        exactly one normal form}. List the seventeen \omterm{def:b2:affine:subspace}{affine}
        types (count the empty variants and degenerate sets),
        and single out the nine \omterm{pb:b2:surfaces:1}{quadric} \emph{surfaces}:
        ellipsoid, one- and two-sheet hyperboloids, cone,
        elliptic and hyperbolic paraboloids, elliptic,
        hyperbolic and parabolic cylinders.
  \item Write the classification \emph{algorithm}: given $(A,
        b, c)$, which quantities do you compute, in which
        order, and which branch decides which type? Justify
        that each step is effective (\omterm{def:b2:reduction:eigen}{eigenvalues} of a
        \omterm{def:b2:quadratic:adjoint}{symmetric} $3\times3$ matrix, rank, solvability of
        $A\Omega = -b$).
  \item Run the algorithm on $x^2 + y^2 + z^2 - 2xy - 2yz -
        2zx = 1$: show $A = 2I - J$ has \omterm{def:b2:reduction:eigen}{spectrum} $\{2, 2,
        -1\}$ and conclude: a one-sheet hyperboloid of
        revolution about $\R(1,1,1)$.
  \item Run it on $x^2 + y^2 - z^2 - 2x + 4y + 2z + 4 = 0$:
        find the center and identify the \omterm{pb:b2:surfaces:1}{quadric}.
  \item Run it on $x^2 + 4xy + y^2 = 2z$: diagonalize the
        $xy$-block ($u = \frac{x+y}{\sqrt2}$, $v =
        \frac{x-y}{\sqrt2}$) and identify the \omterm{pb:b2:surfaces:1}{quadric}.
  \item Euclidean versus \omterm{def:b2:affine:subspace}{affine}. Show that two central
        \omterm{pb:b2:surfaces:1}{quadrics} in normal form are \emph{rigidly} equivalent
        iff they have the same coefficient lists (up to
        permutation and a common positive scalar on the
        equation), while \emph{affinely}, only the signature
        data survives: every ellipsoid is an \omterm{def:b2:affine:subspace}{affine} image of
        the round sphere. Which theorem guarantees that the
        signature cannot change along the way
        (\cref{thm:b2:quadratic:sylvester})?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends.}
\begin{enumerate}[resume]
  \item Show that every section of a \omterm{pb:b2:surfaces:1}{quadric} by an \omterm{def:b2:affine:subspace}{affine}
        plane is a conic (possibly degenerate) of that plane.
        Identify the sections of the saddle $z = xy$ by the
        planes $z = c$ ($c \neq 0$ and $c = 0$).
  \item Which \omterm{pb:b2:surfaces:1}{quadric} surfaces contain straight lines? Show
        that the ellipsoid, the two-sheet hyperboloid and the
        elliptic paraboloid contain none \emph{(restrict $q$
        to a line and use the Cauchy--Schwarz inequality for
        the two-sheet case)}; that cone and cylinders are
        ruled by one family; and conclude that the doubly
        ruled \omterm{pb:b2:surfaces:1}{quadric} surfaces are exactly the one-sheet
        hyperboloid and the hyperbolic paraboloid.
  \item When only the \emph{\omterm{def:b2:affine:subspace}{affine}} type is wanted, Gauss's
        reduction (\cref{thm:b2:quadratic:gauss}) is cheaper
        than diagonalizing. Redo question 17 with Gauss's
        algorithm and check the signature $(2, 1)$; what
        Euclidean information does Gauss lose?
  \item All \omterm{def:b2:reduction:eigen}{eigenvalues} of a \omterm{def:b2:quadratic:adjoint}{symmetric} matrix are real; show
        that consequently the \emph{signs} of the \omterm{def:b2:reduction:eigen}{eigenvalues}
        of $A$ can be read from the \omterm{def:b2:reduction:charpoly}{characteristic polynomial}
        by Descartes' rule of signs, and verify it on
        question 6: $\chi_A(\lambda) = \lambda^3 - 3\lambda^2
        - 9\lambda - 5$ has exactly one sign change, hence
        signature $(1, 2)$.
  \item Synthesis. Summarize the algorithm in a few lines;
        state the exact role played by (i) the spectral
        theorem, (ii) the center equation $A\Omega = -b$,
        (iii) Sylvester's inertia theorem, (iv) Gauss's
        reduction. What does the same machine yield in
        $\R^2$, and what changes in $\R^n$?
\end{enumerate}
\end{problem}
