\begin{solution}{exo:b2:funcseq:1}
$f_n(x) = \frac{x}{1 + nx}$: \omterm{def:b2:funcseq:def}{pontualmente} limite $0$ sobre $\intcc{0}{1}$.
\omterm{def:b2:funcseq:def}{Uniformemente}: $ f_n $ aumenta em $\intcc{0}{1}$ (derivado
$\frac{1}{(1+nx)^2} > 0$), então $\norm{f_n}_\infty = f_n(1) =
\frac{1}{1+n} \to 0$: \emph{uniforme} sobre $\intcc{0}{1}$.

$ g_n(x) = nx\,\eu^{-nx^2}$: \omterm{def:b2:funcseq:def}{pontualmente} limite $0$ (exponencial
batidas). E aí: $ g_n' = n\eu^{-nx^2}(1 - 2nx^2)$ desaparece em $ x_n =
\frac{1}{\sqrt{2n}}$, onde $ g_n(x_n) = \sqrt{\frac n2}\,\eu^{-1/2}
\to \infty $: não uniforme em $\intcc{0}{1}$ --- mas uniforme
$\intco{\delta}{1}$, já que há $ g_n(x) \leq n\,\eu^{-n\delta^2}
\to 0$.

$ h_n(x) = x^n(1 - x^n)$: \omterm{def:b2:funcseq:def}{pontualmente} limite $0$ sobre $\intcc{0}{1}$
(ambos os fatores; em $ x = 1$, $ h_n = 0$). Sup: com $ u = x^n \in
\intcc{0}{1}$, $ u(1-u) \leq \frac14$ alcançado em $ u = \frac12$,
ou seja\ $ x = 2^{-1/n} \in \intoo{0}{1}$: $\norm{h_n}_\infty =
\frac14 \not\to 0$: não uniforme em $\intcc{0}{1}$; uniforme em
$\intcc{0}{a}$ ($\sup \leq a^n \to 0$).
\end{solution}

\begin{solution}{exo:b2:funcseq:2}
$\int_0^1 nx\,\eu^{-nx^2}\dd x = \bigl[-\tfrac12
\eu^{-nx^2}\bigr]_0^1 = \frac{1 - \eu^{-n}}{2} \to \frac12$, enquanto
$\int_0^1 \lim g_n = 0$. Nenhuma contradição:
\cref{thm:b2:funcseq:integration} requer \emph{uniforme}
convergência no segmento, que falha aqui (o aumento de altura
$\sim\sqrt n $ desliza em direção $0$).
\end{solution}

\begin{solution}{exo:b2:funcseq:3}
Convergência normal sobre $\intcc{-1}{1}$: $\norm{x^n/n^2}_\infty =
\frac{1}{n^2}$, resumido: $ S $ é \omterm{def:b2:metric:continuity}{contínuo} lá
(\cref{thm:b2:funcseq:seriestransfer}).

Derivada: a série derivada $\sum \frac{x^{n-1}}{n}$ converge
\omterm{def:b2:funcseq:series}{normalmente} em cada $\intcc{-a}{a}$, $ a < 1$ ($\sup =
\frac{a^{n-1}}{n}$): $ S $ é $ C^1$ sobre $\intoo{-1}{1}$ com
\[
S'(x) = \sum_{n\geq1} \frac{x^{n-1}}{n} = \frac1x \sum_{n\geq1}
\frac{x^n}{n} = -\frac{\ln(1 - x)}{x}
\qquad (0 < \abs x < 1),
\]
a última identidade sendo a série logarítmica do Ano 1 (rederivada
honestamente em \cref{ch:b2:powerseries}).
\end{solution}

\begin{solution}{exo:b2:funcseq:4}
Para fixo $ x > 0$ a série está \omterm{def:b2:linalg:alternating}{alternando} com $\frac{1}{n + x}
\downarrow 0$: convergência pontual, e o restante vinculado
$\abs{R_N(x)} \leq \frac{1}{N + 1 + x} \leq \frac{1}{N+1}$ é
\emph{uniforme} sobre $\intco{\delta}{\infty}$ (na verdade em
$\intoo{0}{\infty}$): convergência uniforme. (Não é normal:
$\norm{\frac{(-1)^n}{n+x}}_\infty = \frac{1}{n + \delta}$,
divergente.) Continuidade segue de
\cref{thm:b2:funcseq:seriestransfer}.

Equação funcional: reindexação $ F(x + 1)$ com $ m = n + 1$:
\[
F(x+1) = \sum_{n\geq0} \frac{(-1)^n}{n + 1 + x}
= \sum_{m\geq1}\frac{(-1)^{m-1}}{m+x} ,
\]
então, isolando o $ m = 0$ prazo de $ F(x)$,
\[
F(x) + F(x+1)
= \frac{1}{x} + \sum_{m\geq1}
\frac{(-1)^m + (-1)^{m-1}}{m+x} = \frac1x .
\]
\end{solution}

\begin{solution}{exo:b2:funcseq:5}
Definir $ g_n = f_n - f \geq 0$ (diminuindo em $ n $, por hipótese; o
limite é $0$ \omterm{def:b2:funcseq:def}{pontualmente}); cada $ g_n $ é \omterm{def:b2:metric:continuity}{contínuo}. Consertar $\varepsilon
> 0$ e deixe $ U_n = \{x : g_n(x) < \varepsilon\}$: \omterm{def:b2:metric:topology}{abrir} (pré-imagem
de um \omterm{def:b2:metric:topology}{abrir} definido), aumentando ($ g_{n+1} \leq g_n $) e cobrindo $ K $
(convergência pontual). Por Borel-Lebesgue
(\cref{thm:b2:metric:borellebesgue}), finitos $ U_{n_1}
\subseteq \dots \subseteq U_{n_k}$ cobrir $ K $: então $ K = U_{n_k}$,
ou seja\ $\norm{g_{n_k}}_\infty \leq \varepsilon $, e por monotonicidade
$\norm{g_n}_\infty \leq \varepsilon $ para todos $ n \geq n_k $: uniforme
convergência. (A monotonicidade é essencial: os solavancos deslizantes
\cref{exo:b2:funcseq:2} \omterm{def:b2:integration:improper}{convergir} \omterm{def:b2:funcseq:def}{pontualmente} em um \omterm{def:b2:metric:compact}{compactar} sem
uniformidade.)
\end{solution}

\begin{solution}{exo:b2:funcseq:6}
Substituir $ u = nx $:
\[
\int_0^1 \frac{n f(x)}{1 + n^2x^2}\dd x
= \int_0^n \frac{f(u/n)}{1 + u^2}\,\dd u .
\]
Os integrandos $ h_n(u) = \frac{f(u/n)}{1+u^2}\mathbf{1}_{u \leq n}$
\omterm{def:b2:integration:improper}{convergir} \omterm{def:b2:funcseq:def}{pontualmente} para $\frac{f(0)}{1+u^2}$ (continuidade de $ f $ no
$0$) e são dominados por $\frac{\norm f_\infty}{1 + u^2}$,
integrável em $\intco{0}{\infty}$: convergência dominada
(\cref{thm:b2:integration:dominated}) dá o limite
\[
\int_0^\infty \frac{f(0)}{1 + u^2}\dd u = \frac{\pi}{2} f(0) .
\]
(Os grãos concentram-se em $0$: uma identidade aproximada.)
\end{solution}

\begin{solution}{exo:b2:funcseq:7}
Para $ f(x) = x^2$, use a segunda família de identidade binomial do
prova: $\sum_k k^2 p_k = n(n-1)x^2 + nx $. Por isso
\[
B_n(f)(x) = \sum_k \frac{k^2}{n^2}\,p_k
= \frac{n(n-1)x^2 + nx}{n^2}
= x^2 + \frac{x(1 - x)}{n} :
\]
$ B_n(f) - f = \frac{x(1-x)}{n}$, de sup \omterm{def:b2:nvs:norm}{norma} $\frac{1}{4n} =
O\bigl(\frac1n\bigr)$, como previsto.
\end{solution}

\begin{solution}{exo:b2:funcseq:8}
Para $\varepsilon = 1$ há $ N $ com $\norm{P_n - P_m}_{\infty,
\R} \leq 1$ para $ m, n \geq N $. Um polinômio limitado $\R $ é
constante (um não constante tende a $\pm\infty $): $ P_n - P_m =
c_{n,m}$, constantes. Então, para $ n \geq N $: $ P_n = P_N + c_n $ com
$ c_n = P_n(0) - P_N(0)$ convergente (convergência pontual no $0$).
Por isso $ f = \lim P_n = P_N + \lim c_n $: um polinômio.
\end{solution}

\begin{solution}{exo:b2:funcseq:9}
(um) $\norm{(3/4)^n\varphi(4^n\cdot)}_\infty = \frac12
(3/4)^n $: convergência normal, então $ W $ é \omterm{def:b2:metric:continuity}{contínuo}
(\cref{thm:b2:funcseq:seriestransfer}).

(b) Correção $ x $, $ m $; escolha o sinal de $ h_m = \pm\frac12 4^{-m}$ então
que o segmento $\intcc{4^mx}{4^m(x + h_m)}$ (de comprimento $\frac12$)
não contém meio inteiro, fazendo $\varphi $ afim de inclinação $\pm1$
nele (possível: um intervalo de comprimento $\frac12$ atende no máximo um
ponto meio inteiro; escolha o lado que evita).

Para $ n > m $: $4^n h_m = \pm\frac12 4^{n-m}$ é um número inteiro e
$\varphi $ é $1$-periódico: o $ n $-ésimo termo da diferença
desaparece.

Para $ n = m $: $\abs{\varphi(4^m x + 4^m h_m) - \varphi(4^m x)} =
\abs{4^m h_m} = \frac12$ ($\varphi $ afim de inclinação $\pm 1$ no
segmento), então o termo contribui exatamente $(3/4)^m \cdot
\frac{1/2}{\abs{h_m}} = (3/4)^m\,4^m = 3^m $ no quociente.

Para $ n < m $: o $1$-Lipschitz $\varphi $ dá
$\bigl|(3/4)^n\bigl(\varphi(4^nx + 4^nh_m) -
\varphi(4^nx)\bigr)\bigr| \leq (3/4)^n 4^n\abs{h_m} =
3^n\abs{h_m}$: cada um contribui no máximo $3^n $ para o quociente.

Por isso
\[
\Bigl|\frac{W(x + h_m) - W(x)}{h_m}\Bigr|
\geq 3^m - \sum_{n=0}^{m-1} 3^n
= 3^m - \frac{3^m - 1}{2} = \frac{3^m + 1}{2}
\longrightarrow \infty .
\]
Se $ W $ eram diferenciáveis em $ x $, cada quociente de diferença ao longo
$ h_m \to 0$ convergiria para $ W'(x)$: contradição. $ W $ é
\omterm{def:b2:metric:continuity}{contínuo} em todos os lugares, diferenciável em lugar nenhum.
\end{solution}

\begin{solution}{exo:b2:funcseq:10}
\omterm{def:b2:funcseq:def}{Pontualmente}: para $ x \in \intco{0}{1}$ a série é geométrica de
proporção $-x $,
\[
\sum_{n\geq0}(-1)^n x^n(1-x) = \frac{1-x}{1+x},
\]
e em $ x = 1$ todo termo desaparece: soma $0 = \frac{1-1}{2}$,
consistente --- a soma é \omterm{def:b2:metric:continuity}{contínuo} sobre $\intcc{0}{1}$.
Uniformidade: o restante é uma cauda geométrica,
\[
\abs{R_N(x)} = \frac{x^{N+1}(1-x)}{1+x} \leq x^{N+1}(1 - x)
\leq \max_{\intcc01} t^{N+1}(1-t)
= \frac{1}{N+2}\Bigl(\frac{N+1}{N+2}\Bigr)^{\!N+1}
\leq \frac{1}{N+2} \to 0 ,
\]
\omterm{def:b2:funcseq:def}{uniformemente} em $ x $. Não é normal: $\norm{u_n}_\infty =
\max x^n(1-x) = \frac{1}{n+1}\bigl(\frac{n}{n+1}\bigr)^n \sim
\frac{1}{\eu\,n}$, e $\sum \frac1{\eu n}$ diverge. Contraste:
$\sum x^n(1-x)$ tem somas parciais $1 - x^{N+1}$, convergindo
\omterm{def:b2:funcseq:def}{pontualmente} para o \emph{descontínuo} $\mathbf 1_{\intco01}$: por
\cref{thm:b2:funcseq:continuity}, essa convergência não pode ser
uniforme em $\intcc{0}{1}$.
\end{solution}

\begin{solution}{exo:b2:funcseq:11}
O limite $ f $ é \omterm{def:b2:metric:continuity}{contínuo} (\cref{thm:b2:funcseq:continuity}).
Então
\[
\abs{f_n(x_n) - f(x)}
\leq \abs{f_n(x_n) - f(x_n)} + \abs{f(x_n) - f(x)}
\leq \norm{f_n - f}_\infty + \abs{f(x_n) - f(x)} ,
\]
e ambos os termos tendem a $0$ (convergência uniforme; continuidade de
$ f $ no $ x $). Contra-exemplo sob mera convergência pontual:
$ g_n(x) = nx\,\eu^{-nx^2} \to 0$ \omterm{def:b2:funcseq:def}{pontualmente} sobre $\intcc{0}{1}$
com $ g_n $ e o limite \omterm{def:b2:metric:continuity}{contínuo}, ainda em $ x_n =
\frac{1}{\sqrt{2n}} \to 0$:
\[
g_n(x_n) = \sqrt{\frac n2}\,\eu^{-1/2} \longrightarrow +\infty
\neq 0 = f(0) .
\]
\end{solution}

\begin{solution}{exo:b2:funcseq:12}
\begin{enumerate}
  \item Indução. $ n = 1$ é a definição. Suponha a fórmula
        para $ n $ e definir $ g(x) = \int_0^x \frac{(x-t)^n}{n!}
        f(t)\dd t $. Para um integrando \omterm{def:b2:metric:continuity}{contínuo} em $(x,t)$ e
        $ C^1$ em $ x $, o limite da variável \omterm{thm:b2:integration:continuity}{integral do parâmetro}
        diferencia como
        \[
        g'(x) = \frac{(x-x)^n}{n!}f(x)
        + \int_0^x \frac{(x-t)^{n-1}}{(n-1)!}f(t)\dd t
        = T^nf(x)
        \]
        (dividir $ g(x+h) - g(x)$ na faixa
        $\int_x^{x+h}$, que é $ O(h\cdot\sup)$ com o
        integrando desaparecendo em $ t = x $ como $ h^n $, e o fixo
        integral do $ x $-incremento, tratado pelo valor médio
        desigualdade e continuidade). Também $(T^{n+1}f)' = T^nf $
        (teorema fundamental do cálculo) e $ g(0) =
        T^{n+1}f(0) = 0$: duas primitivas de $ T^nf $ desaparecendo em
        $0$ coincidem, então $ T^{n+1}f = g $. O limite:
        \[
        \abs{T^nf(x)} \leq \norm f_\infty
        \int_0^x \frac{(x-t)^{n-1}}{(n-1)!}\dd t
        = \norm f_\infty\,\frac{x^n}{n!}
        \leq \frac{\norm f_\infty}{n!} .
        \]
  \item $\sum_n \norm{T^nf}_\infty \leq \eu\,\norm f_\infty $:
        convergência normal e, portanto, uniforme; $ S $ é \omterm{def:b2:metric:continuity}{contínuo}.
        As somas parciais satisfazem $ S_N = f + T S_{N-1}$, e $ T $
        é $1$-Lipschitz para $\norm\cdot_\infty $
        ($\abs{Tg(x)} \leq x\norm g_\infty $): deixar $ N \to
        \infty $ em ambos os lados dá $ S = f + TS $.
  \item Definir $ V(x) = f(x) + \eu^x\int_0^x \eu^{-t}f(t)\dd t $.
        Então $ V - f $ é $ C^1$ com $(V-f)'(x) =
        \eu^x\int_0^x\eu^{-t}f + f(x) = V(x)$, e $(TV)' = V $
        com $(V - f)(0) = TV(0) = 0$: então $ V - f = TV $, ou seja\
        $ V $ resolve a equação. Singularidade: se $ S_1, S_2$ são
        \omterm{def:b2:metric:continuity}{contínuo} soluções, $ D = S_1 - S_2$ satisfaz $ D =
        TD $, por isso $ D = T^nD $ para todos $ n $ e $\norm D_\infty
        \leq \frac{\norm D_\infty}{n!} \to 0$: $ D = 0$.
        Portanto
        \[
        \sum_{n\geq0} T^nf(x) = f(x) +
        \int_0^x \eu^{x-t}f(t)\,\dd t .
        \]
        (A série $\sum T^n $ é uma série geométrica de
        operadores: uma primeira amostra do resolvente
        $(\mathrm{Id} - T)^{-1}$, desenvolvido no Ano 3
        volume.)
\end{enumerate}
\end{solution}

\begin{solution}{pb:b2:funcseq:1}
\textbf{1.} A linearidade fica clara na fórmula. Positividade:
os pesos $ p_k(x) \geq 0$, então $ f \geq 0$ forças $ B_nf \geq
0$; a monotonicidade segue aplicada a $ g - f $. Vinculado: $\pm f \leq
\norm f_\infty $ dá $\pm B_nf \leq \norm f_\infty B_ne_0 =
\norm f_\infty $. Pontos finais: $ p_k(0) = \mathbf 1_{k=0}$ e
$ p_k(1) = \mathbf 1_{k=n}$, então $ B_nf(0) = f(0)$, $ B_nf(1) =
f(1)$.

\textbf{2.} Diferenciar $(x+y)^n = \sum_k\binom nk x^ky^{n-k}$
em $ x $, multiplique por $ x $ e definir $ y = 1 - x $:
\[
nx = \sum_k k\,p_k(x) ;
\]
duas vezes, multiplicando por $ x^2$: $ n(n-1)x^2 = \sum_k k(k-1)p_k(x)$.
Por isso $ B_ne_0 = 1$ (teorema binomial), $ B_ne_1(x) =
\frac{nx}{n} = x $, e
\[
B_ne_2(x) = \frac{\sum_k k^2p_k}{n^2}
= \frac{n(n-1)x^2 + nx}{n^2}
= x^2 + \frac{x(1-x)}{n} .
\]

\textbf{3.} Expandir:
\[
\sum_k\Bigl(\frac kn - x\Bigr)^{\!2} p_k
= B_ne_2(x) - 2x\,B_ne_1(x) + x^2
= \frac{x(1-x)}{n} .
\]
Cauchy--Schwarz com a divisão $\abs{k/n - x}\sqrt{p_k}
\cdot \sqrt{p_k}$:
\[
\sum_k\Bigl|\frac kn - x\Bigr| p_k
\leq \Bigl(\sum_k\Bigl(\frac kn - x\Bigr)^2
p_k\Bigr)^{\!1/2}
= \sqrt{\frac{x(1-x)}{n}} \leq \frac{1}{2\sqrt n},
\]
usando $ x(1-x) \leq \frac14$.

\textbf{4.} Os pesos $ p_k(x)$ são não negativos com soma $1$
e baricentro $\sum_k \frac kn p_k(x) = x $ (questão 2). O
desigualdade de Jensen finita para o convexo $ f $ (indução do
definição de dois pontos, volume do Ano 1) dá
\[
f(x) = f\Bigl(\sum_k \frac kn\,p_k\Bigr)
\leq \sum_k f\Bigl(\frac kn\Bigr)p_k = B_nf(x) .
\]

\textbf{5.} Sobre $\{k : \abs{k/n - x} > \delta\}$ um tem
$\bigl(\frac{k/n - x}{\delta}\bigr)^2 > 1$, então
\[
\sum_{\abs{k/n-x}>\delta} p_k
\leq \frac{1}{\delta^2}\sum_k\Bigl(\frac kn -
x\Bigr)^2p_k
= \frac{x(1-x)}{n\delta^2} \leq \frac{1}{4n\delta^2} .
\]
Leitura: $ p_k(x)$ é a lei de uma frequência de amostra $ S_n/n $ de
$ n $ lançamentos de moeda de viés $ x $; sua média é $ x $, sua variação
$\frac{x(1-x)}n \to 0$, e a tela é de Chebyshev
desigualdade: a massa se concentra em $ x $, então calculando a média $ f $
contra ele se reproduz $ f(x)$ no limite
(\cref{ch:b2:genfun} oficializa o vocabulário).

\textbf{6.} $\omega \leq 2M < \infty $; a monotonicidade é clara
(sobre um conjunto maior). Heine: $ f $ \omterm{def:b2:metric:continuity}{contínuo} em um \omterm{def:b2:metric:compact}{compactar} é
\omterm{def:b2:funcseq:def}{uniformemente} \omterm{def:b2:metric:continuity}{contínuo}, o que diz exatamente $\omega(\delta) \to 0$
como $\delta \to 0^+$. Subaditividade: se $\abs{s - t} \leq
\delta_1 + \delta_2$, o ponto $ u $ no segmento $\intcc st $
à distância $\min(\delta_1, \abs{s-t})$ de $ s $ satisfaz
$\abs{s-u} \leq \delta_1$, $\abs{u-t} \leq \delta_2$, e
$\abs{f(s)-f(t)} \leq \abs{f(s)-f(u)} + \abs{f(u)-f(t)}$.
Iterando, $\omega(p\delta) \leq p\,\omega(\delta)$ para $ p \in
\N^*$; para $\lambda > 0$, com $ p = \lceil\lambda\rceil \leq 1
+ \lambda $: $\omega(\lambda\delta) \leq \omega(p\delta) \leq
p\,\omega(\delta) \leq (1+\lambda)\omega(\delta)$.

\textbf{7.} Desde $\sum p_k = 1$:
\[
\abs{B_nf(x) - f(x)}
= \Bigl|\sum_k\bigl(f(k/n) - f(x)\bigr)p_k\Bigr|
\leq \sum_k\omega\bigl(\abs{k/n - x}\bigr)p_k .
\]
Para cada $ k $, questão 6 com $\lambda = \abs{k/n - x}/\delta $
dá $\omega(\abs{k/n-x}) \leq \bigl(1 +
\frac{\abs{k/n-x}}\delta\bigr)\omega(\delta)$; somando contra
o $ p_k $ produz a estimativa mestre.

\textbf{8.} Insira o limite da questão 3:
\[
\abs{B_nf(x) - f(x)} \leq \Bigl(1 +
\frac{1}{2\delta\sqrt n}\Bigr)\omega(\delta),
\]
\omterm{def:b2:funcseq:def}{uniformemente} em $ x $; com $\delta = n^{-1/2}$ o parêntese é
$\frac32$: $\norm{B_nf - f}_\infty \leq
\frac32\omega(n^{-1/2}) \to 0$ pela pergunta 6 (Heine). Isto
reprova \cref{thm:b2:funcseq:weierstrass} com uma taxa.

\textbf{9.} $ L $-Lipschitz significa $\omega(\delta) \leq L\delta $:
avaliar $\frac{3L}{2\sqrt n}$. $\alpha $-Hmeios mais antigos
$\omega(\delta) \leq C\delta^\alpha $: avaliar $\frac{3C}{2}
n^{-\alpha/2}$.

\textbf{10.} Usando $ k\binom{2m}k = 2m\binom{2m-1}{k-1}$:
\[
\sum_{k=m+1}^{2m}k\binom{2m}k
= 2m\sum_{j=m}^{2m-1}\binom{2m-1}{j}
= 2m\cdot 2^{2m-2},
\]
porque $ j \mapsto 2m-1-j $ bijetos $\{m,\dots,2m-1\}$ para
$\{0,\dots,m-1\}$, então a soma é metade de $2^{2m-1}$. Também
$\sum_{k=m+1}^{2m}\binom{2m}k = \frac{2^{2m} -
\binom{2m}m}{2}$ (mesma simetria). Por isso
\[
\sum_{k=m+1}^{2m}(k-m)\binom{2m}k
= m\,2^{2m-1} - m\,\frac{2^{2m} - \binom{2m}m}{2}
= \frac m2\binom{2m}m .
\]
A substituição $ k \mapsto 2m-k $ mapeia os termos com $ k < m $
para aqueles com $ k > m $ (binômios iguais, iguais $\abs{k-m}$):
a soma absoluta é o dobro da soma unilateral, $ m\binom{2m}m $.

\textbf{11.} No $ x = \frac12$, $ p_k(\tfrac12) =
\binom{2m}k2^{-2m}$ e $ f(\tfrac12) = 0$:
\[
B_{2m}f\Bigl(\frac12\Bigr)
= \sum_k\Bigl|\frac{k}{2m} - \frac12\Bigr|
\binom{2m}k 2^{-2m}
= \frac{2^{-2m}}{2m}\,m\binom{2m}m
= \frac{\binom{2m}m}{2\cdot4^m}
\sim \frac{1}{2\sqrt{\pi m}}
\]
por \cref{ex:b2:comparison:centralbinomial}. Desde $\omega_f
(\delta) = \delta $ aqui (a função é $1$-Lipschitz e o
limite é alcançado), a questão 8 prevê no máximo
$\frac32(2m)^{-1/2}$: o verdadeiro erro $\frac{1}{2\sqrt{\pi m}}$
tem exatamente a ordem $ n^{-1/2}$ --- a taxa é acentuada até
a constante.

\textbf{12.} $ f \leq g $ dá $ g - f \geq 0$, então $ L(g-f) \geq
0$, ou seja\ $ Lf \leq Lg $. De $-\abs f \leq f \leq \abs f $:
$-L\abs f \leq Lf \leq L\abs f $, ou seja\ $\abs{Lf} \leq
L\abs f $.

\textbf{13.} Por Heine escolha $\delta $ com $\abs{f(s)-f(x)}
\leq \varepsilon $ em qualquer momento $\abs{s-x} \leq \delta $. Se
$\abs{s - x} > \delta $, então $\frac{(s-x)^2}{\delta^2} > 1$ e
$\abs{f(s)-f(x)} \leq 2M \leq \frac{2M}{\delta^2}(s-x)^2$. Em
ambos os casos, o limite reivindicado é válido.

\textbf{14.} Consertar $ x $; a questão 13 diz, como funções de $ s $:
\[
-\varepsilon e_0 - \frac{2M}{\delta^2}q_x
\;\leq\; f - f(x)e_0
\;\leq\; \varepsilon e_0 + \frac{2M}{\delta^2}q_x,
\qquad q_x = e_2 - 2x\,e_1 + x^2e_0 .
\]
Aplique o linear monótono $ L_n $ (questão 12) e avaliar em
$ x $:
\[
\abs{L_nf(x) - f(x)L_ne_0(x)}
\leq \varepsilon L_ne_0(x)
+ \frac{2M}{\delta^2}\bigl(L_ne_2(x) - 2xL_ne_1(x)
+ x^2L_ne_0(x)\bigr) .
\]

\textbf{15.} Escrever $\alpha_j = L_ne_j - e_j $, então
$\norm{\alpha_j}_\infty \to 0$. Desde $ e_2(x) - 2xe_1(x) +
x^2e_0(x) = 0$:
\[
L_ne_2(x) - 2xL_ne_1(x) + x^2L_ne_0(x)
= \alpha_2(x) - 2x\,\alpha_1(x) + x^2\alpha_0(x),
\]
de sup \omterm{def:b2:nvs:norm}{norma} no máximo $\norm{\alpha_2} + 2\norm{\alpha_1} +
\norm{\alpha_0} \to 0$. Também $ L_ne_0 \to e_0$ \omterm{def:b2:funcseq:def}{uniformemente}, então
$ L_ne_0 \leq 2$ para grande $ n $, e $\abs{f(x)}\abs{L_ne_0(x) -
1} \leq M\norm{\alpha_0} \to 0$. Montando com a questão 14:
para $ n $ grande, \omterm{def:b2:funcseq:def}{uniformemente} em $ x $,
\[
\abs{L_nf(x) - f(x)} \leq 2\varepsilon +
\frac{2M}{\delta^2}\,o(1) + M\,o(1) \leq 3\varepsilon :
\]
$ L_nf \to f $ \omterm{def:b2:funcseq:def}{uniformemente} --- Teorema de Korovkin.

\textbf{16.} $ B_ne_0 = e_0$ e $ B_ne_1 = e_1$ exatamente, e
$\norm{B_ne_2 - e_2}_\infty = \max_x\frac{x(1-x)}{n} =
\frac{1}{4n} \to 0$ (questão 2): Korovkin se aplica, e
Weierstrass segue pela terceira vez.

\textbf{17.} $ I_nf $ é linear em $ f $ (os valores nodais são) e em
cada célula, o interpolante afim de valores nodais não negativos é
não negativo: positivo. $ I_ne_0 = e_0$ e $ I_ne_1 = e_1$
porque uma função afim é igual ao seu próprio interpolante. Em um
célula $\intcc ab $ ($ b - a = \frac1n $), o interpolante afim de
$ e_2$ é $ L(t) = (a+b)t - ab $, e
\[
L(t) - t^2 = (t-a)(b-t) \in
\intcc{0}{\tfrac{(b-a)^2}{4}} ,
\]
com o máximo no ponto médio: $\norm{I_ne_2 - e_2}_\infty =
\frac{1}{4n^2} \to 0$. Korovkin: $ I_nf \to f $ \omterm{def:b2:funcseq:def}{uniformemente} para
cada \omterm{def:b2:metric:continuity}{contínuo} $ f $ --- aproximação poligonal, sem
estimativa adicional necessária.

\textbf{18.} Dado $ f $ e $\varepsilon $: Weierstrass fornece
um polinômio $ P = \sum_{j=0}^d a_jx^j $ com $\norm{f -
P}_\infty \leq \frac\varepsilon2$; substituindo cada $ a_j $ por um
racional $ b_j $ com $\abs{a_j - b_j} \leq
\frac{\varepsilon}{2(d+1)}$ move o sup \omterm{def:b2:nvs:norm}{norma} sobre
$\intcc{0}{1}$ por no máximo $\frac\varepsilon2$. O conjunto de
polinômios de coeficiente racional é um \omterm{def:b2:structures:countable}{contável} união (acima
$ d $) de conjuntos contáveis, por isso contávele denso:
$ C(\intcc{0}{1})$ é separável.

\textbf{19.} Por linearidade $\int_0^1 fP = 0$ para cada
polinômio $ P $. Escolha polinômios $ P_n \to f $ \omterm{def:b2:funcseq:def}{uniformemente}
(Weierstrass):
\[
\Bigl|\int_0^1 f^2\Bigr|
= \Bigl|\int_0^1 f\,(f - P_n)\Bigr|
\leq \norm f_\infty\,\norm{f - P_n}_\infty
\longrightarrow 0 ,
\]
então $\int_0^1 f^2 = 0$. Se $ f(x_0) \neq 0$, continuidade dá
$ f^2 \geq c > 0$ em um subintervalo, contradizendo o desaparecimento
integral: $ f = 0$. Consequentemente dois \omterm{def:b2:metric:continuity}{contínuo} funções com
os mesmos momentos $\int f t^n $ coincidir.

\textbf{20.} Com $ p_{n,k}(x) = \binom nk x^k(1-x)^{n-k}$ e
as convenções $ p_{n-1,-1} = p_{n-1,n} = 0$, a regra do produto
e $ k\binom nk = n\binom{n-1}{k-1}$, $(n-k)\binom nk =
n\binom{n-1}{k}$ dar
\[
p_{n,k}'(x) = n\bigl(p_{n-1,k-1}(x) - p_{n-1,k}(x)\bigr) .
\]
Somando contra $ f(k/n)$ e mudando o índice no primeiro
soma (\omterm{thm:b2:series:abel}{Resumo de Abel}):
\[
(B_nf)'(x) = n\sum_{j=0}^{n-1}\Bigl(f\Bigl(\frac{j+1}n\Bigr)
- f\Bigl(\frac jn\Bigr)\Bigr)p_{n-1,j}(x) .
\]

\textbf{21.} Pelo teorema do valor médio, $ f(\frac{j+1}n) -
f(\frac jn) = \frac1n f'(\xi_j)$ com $\xi_j \in
\intoo{j/n}{(j+1)/n}$, então $(B_nf)'(x) = \sum_j
f'(\xi_j)\,p_{n-1,j}(x)$. O nó $\frac{j}{n-1}$ também reside em
$\intcc{j/n}{(j+1)/n}$ (ambas as desigualdades se reduzem a $ j \leq
n-1$), por isso $\abs{\xi_j - \frac j{n-1}} \leq \frac1n $ e
\[
\bigl|(B_nf)'(x) - B_{n-1}(f')(x)\bigr|
\leq \sum_j\Bigl|f'(\xi_j) -
f'\Bigl(\frac{j}{n-1}\Bigr)\Bigr| p_{n-1,j}(x)
\leq \omega_{f'}\Bigl(\frac1n\Bigr) \longrightarrow 0
\]
\omterm{def:b2:funcseq:def}{uniformemente}. Desde $ B_{n-1}(f') \to f'$ \omterm{def:b2:funcseq:def}{uniformemente}
(\cref{thm:b2:funcseq:weierstrass} aplicado ao \omterm{def:b2:metric:continuity}{contínuo}
$ f'$), a desigualdade triangular dá $(B_nf)' \to f'$
\omterm{def:b2:funcseq:def}{uniformemente}. Os polinômios $ P_n = B_nf $ então convirja para $ f $
no $ C^1$ senso.

\textbf{22.} Taylor-Lagrange em $ x $: $ f(\frac kn) - f(x) =
f'(x)(\frac kn - x) + \frac{f''(\xi_k)}2(\frac kn - x)^2$.
Somando contra $ p_k $, o termo linear morre (questão 2):
\[
\abs{B_nf(x) - f(x)}
\leq \frac{\norm{f''}_\infty}{2}\sum_k\Bigl(\frac kn -
x\Bigr)^2p_k
= \frac{\norm{f''}_\infty}{2}\cdot\frac{x(1-x)}{n}
\leq \frac{\norm{f''}_\infty}{8n} .
\]

\textbf{23.} Mais duas diferenciações de $(x+y)^n $ dê o
momentos fatoriais, com $ n_{(j)} = n(n-1)\cdots(n-j+1)$:
\[
\sum_k k_{(j)}\,p_k = n_{(j)}\,x^j \qquad (j = 3, 4),
\]
e $ k^3 = k_{(3)} + 3k_{(2)} + k $, $ k^4 = k_{(4)} + 6k_{(3)} +
7k_{(2)} + k $ converta-os em momentos de poder:
\[
\sum_k k^3p_k = n_{(3)}x^3 + 3n_{(2)}x^2 + nx,
\qquad
\sum_k k^4p_k = n_{(4)}x^4 + 6n_{(3)}x^3 + 7n_{(2)}x^2 + nx .
\]
Expandindo $(k - nx)^4$ e coletando (um paciente, mas puramente
cálculo mecânico com os quatro momentos de potência):
\[
\sum_k(k-nx)^4p_k = nx(1-x)\bigl(1 + 3(n-2)x(1-x)\bigr) .
\]
Com $ x(1-x) \leq \frac14$: o lado direito está no máximo
$\frac n4\bigl(1 + \frac{3n}4\bigr) = \frac{3n^2}{16} +
\frac n4 \leq n^2$ para $ n \geq 1$.

\textbf{24.} A forma Peano de Taylor em $ x $ define $\eta(t) =
\frac{f(t) - f(x) - f'(x)(t-x) - \frac12f''(x)(t-x)^2}
{(t-x)^2}$ para $ t \neq x $, $\eta(x) = 0$: por Taylor--Lagrange
$\eta(t) = \frac12\bigl(f''(\xi) - f''(x)\bigr)$ para alguns $\xi $
entre $ t $ e $ x $, então $\abs\eta \leq \norm{f''}_\infty $ e
$\eta(t) \to 0$ como $ t \to x $ (continuidade de $ f''$). Resumindo
a expansão contra $ p_k $ e usando as perguntas 2 a 3:
\[
n\bigl(B_nf(x) - f(x)\bigr)
= \frac{x(1-x)}{2}f''(x)
+ n\sum_k\eta\Bigl(\frac kn\Bigr)\Bigl(\frac kn -
x\Bigr)^2p_k .
\]
Dado $\varepsilon $, escolher $\delta $ com $\abs\eta \leq
\varepsilon $ sobre $\abs{t - x}\leq\delta $. Parte próxima: no máximo
$\varepsilon\,n\cdot\frac{x(1-x)}n \leq \varepsilon $. Parte distante:
com $ C = \norm{f''}_\infty $ e a questão 23,
\[
n\,C\sum_{\abs{k/n-x}>\delta}\Bigl(\frac kn -
x\Bigr)^2p_k
\leq \frac{nC}{\delta^2}\sum_k\Bigl(\frac kn -
x\Bigr)^4p_k
= \frac{nC}{\delta^2 n^4}\sum_k(k-nx)^4p_k
\leq \frac{C}{\delta^2 n} \longrightarrow 0 .
\]
Por isso $ n(B_nf(x) - f(x)) \to \frac{x(1-x)}2f''(x)$ ---
Teorema de Voronovskaya. Para $ f = e_2$ isso é exato em todos
$ n $ (\cref{exo:b2:funcseq:7}): o $\frac1n $ teto é real.

\textbf{25.} (i) Positividade transformada \omterm{def:b2:funcseq:def}{pontualmente} desigualdades em
desigualdades de operador: deu o \omterm{def:b2:nvs:norm}{norma} obrigado, Jensen,
Chebyshev e todo Korovkin --- a linearidade por si só prova
nada aqui. (ii) Compacidade inserido através de Heine (perguntas
6, 13), através da limitação de $ f $, e através do próprio \omterm{def:b2:nvs:norm}{norma}
$\norm\cdot_\infty $ sendo finito. (iii) Três funções de teste
basta porque a positividade reduz tudo ao controle
$ L_n $ na única família $(s-x)^2 = e_2 - 2xe_1 + x^2e_0$,
cuja extensão é a de $ e_0, e_1, e_2$. (iv) Bernstein converge
na taxa honesta $\omega(n^{-1/2})$ para cada \omterm{def:b2:metric:continuity}{contínuo} $ f $
(nítido, questão 11), mas satura em $\frac1n $ para suave $ f$
(questão 24); o capítulo Fourier executa o mesmo programa para
funções periódicas com Fejḱernel de er --- outro positivo
operador com as mesmas virtudes e a mesma modéstia.
\end{solution}
