\section*{Capítulo \ref{ch:b2:randomvar} --- Variáveis aleatórias discretas}

\begin{solution}{exo:b2:randomvar:1}
\emph{Binômio:} $X = \sum_{i=1}^n X_i $ com \omterm{def:b2:proba:independence}{independente} Bernoulli
$ X_i $; $ V(X_i) = \E(X_i^2) - \E(X_i)^2 = p - p^2$, e variações
de \omterm{def:b2:proba:independence}{independente} variáveis adicionar
(\cref{thm:b2:randomvar:variancerules}):
\[
\E(X) = np, \qquad V(X) = np(1-p) .
\]
\emph{Poisson:} $\E\bigl(X(X-1)\bigr) =
\sum_{k\geq2}k(k-1)e^{-\lambda}\frac{\lambda^k}{k!} = \lambda^2
e^{-\lambda}\sum_{j\geq0}\frac{\lambda^j}{j!} = \lambda^2$, então
\[
V(X) = \E(X^2) - \E(X)^2
= \lambda^2 + \lambda - \lambda^2 = \lambda .
\]
\emph{Geométrico} ($ q = 1 - p $): diferenciando
$\sum_{k\geq0}q^k = \frac{1}{1-q}$ duas vezes dentro do disco
(\cref{ch:b2:powerseries}), $\sum_{k\geq2}k(k-1)q^{k-2} =
\frac{2}{(1-q)^3}$, então
\[
\E\bigl(X(X-1)\bigr) = pq\sum_{k\geq2}k(k-1)q^{k-2}
= \frac{2q}{p^2},
\qquad
V(X) = \frac{2q}{p^2} + \frac1p - \frac{1}{p^2}
= \frac{q}{p^2} = \frac{1-p}{p^2} .
\]
\end{solution}

\begin{solution}{exo:b2:randomvar:2}
\emph{Soma:} para $ n \in \N $, por disjunção e independência,
\[
\P(X + Y = n)
= \sum_{k=0}^n \P(X = k)\P(Y = n - k)
= e^{-(\lambda + \mu)}\frac{1}{n!}
\sum_{k=0}^n \binom nk \lambda^k\mu^{n-k}
= e^{-(\lambda+\mu)}\frac{(\lambda + \mu)^n}{n!}
\]
pelo teorema binomial: $ X + Y \sim \mathcal{P}(\lambda + \mu)$.
\emph{Condicional \omterm{def:b2:randomvar:law}{lei}:} para $0 \leq k \leq n $,
\[
\P(X = k \mid X + Y = n)
= \frac{\P(X = k)\P(Y = n - k)}{\P(X + Y = n)}
= \binom nk
\Bigl(\frac{\lambda}{\lambda+\mu}\Bigr)^{k}
\Bigl(\frac{\mu}{\lambda+\mu}\Bigr)^{n-k} ,
\]
o binômio \omterm{def:b2:randomvar:law}{lei} $\mathcal{B}\bigl(n,
\frac{\lambda}{\lambda+\mu}\bigr)$: dada a contagem total, cada
evento ``escolhe'' independentemente a primeira fonte com probabilidade
proporcional à sua taxa.
\end{solution}

\begin{solution}{exo:b2:randomvar:3}
Quando $ k $ ainda faltam brinquedos, cada caixa nova traz um brinquedo novo
com probabilidade $\frac kn $, independentemente do passado: o
tempo de espera $ W_k $ para o próximo novo brinquedo é geométrico
$\mathcal{G}\bigl(\frac kn\bigr)$, com $\E(W_k) = \frac nk $, e
$ T_n = W_n + W_{n-1} + \dots + W_1$ (a primeira caixa sempre dá um novo
brinquedo: $ W_n = 1$, consistente com $\E = n/n $). Por linearidade,
\[
\E(T_n) = \sum_{k=1}^n \frac nk = n\sum_{k=1}^n\frac1k
\sim n\ln n ,
\]
usando $\sum_{k\leq n}\frac1k = \ln n + \gamma + o(1)$
(\cref{ch:b2:comparison}). Coletar os últimos brinquedos é o que
custos: metade das caixas vai para o punhado final.
\end{solution}

\begin{solution}{exo:b2:randomvar:4}
\omterm{def:b2:funcseq:def}{Pontualmente}, $ X(\omega) = \#\{n \geq 1 : X(\omega) \geq n\} =
\sum_{n\geq1}\mathbf{1}_{X \geq n}(\omega)$. A família dupla
$\bigl(\mathbf{1}_{X \geq n}(\omega)\,\P(\{\omega\})\bigr)_{n,
\omega}$ não é negativo, então Fubini para famílias
(\cref{ch:b2:series}) aplica-se incondicionalmente: somando primeiro em
$ n $ dá $\E(X)$, somando primeiro em $\omega $ dá $\sum_n \P(X
\geq n)$; os dois são simultaneamente finitos e iguais. Para $ X \sim
\mathcal{G}(p)$: $\P(X \geq n) = q^{n-1}$ ($ q = 1-p $), então $\E(X) =
\sum_{n\geq1}q^{n-1} = \frac{1}{1 - q} = \frac1p $.
\end{solution}

\begin{solution}{exo:b2:randomvar:5}
\emph{Simetria:} o $ i $-a bola sorteada é uma bola \omterm{def:b2:funcseq:def}{uniformemente} aleatória
da urna (qualquer um dos $ N $ bolas têm a mesma probabilidade de cair
posição $ i $ da ordem de saque), então $\P(Y_i = 1) = \frac MN =
p $ e $\E(X) = np $ por linearidade --- não independência necessário.

\emph{\omterm{def:b2:randomvar:variance}{Covariância}:} para $ i \neq j $, $\E(Y_iY_j) = \P(\text{draws }
i, j \text{ both white}) = \frac{M(M-1)}{N(N-1)}$ (pares ordenados
de posições distintas obtém um par ordenado de bolas distintas,
\omterm{def:b2:funcseq:def}{uniformemente}). Por isso
\[
\operatorname{Cov}(Y_i, Y_j)
= \frac{M(M-1)}{N(N-1)} - \frac{M^2}{N^2}
= \frac{M(N - M)}{N^2}\cdot\frac{-1}{N-1}
= -\frac{p(1-p)}{N-1} < 0 :
\]
tirar uma bola branca torna as brancas mais escassas para os outros sorteios.
Por \cref{thm:b2:randomvar:variancerules},
\[
V(X) = np(1-p) + n(n-1)\Bigl(-\frac{p(1-p)}{N-1}\Bigr)
= np(1-p)\,\frac{N - n}{N - 1} \leq np(1-p) :
\]
amostragem sem reposição tem a mesma média, mas \emph{menor}
\omterm{def:b2:randomvar:variance}{variação} do que com substituição (igualdade apenas para $ n = 1$), o
correlações negativas atuando como estabilizador. Para $ n = N $ o
\omterm{def:b2:randomvar:variance}{variação} desaparece: a contagem é então determinística.
\end{solution}

\begin{solution}{exo:b2:randomvar:6}
Expandindo ao redor $ m = \E(X)$:
\[
\E\bigl((X - c)^2\bigr)
= \E\bigl((X - m)^2\bigr) + 2(m - c)\,\E(X - m) + (m - c)^2
= V(X) + (m - c)^2 ,
\]
mínimo exatamente em $ c = m $ com valor $ V(X)$ --- \omterm{def:b2:randomvar:expectation}{expectativa} é
o melhor preditor constante no quadrado médio.

Se $\P(X = m) = 1$ então $(X - m)^2$ desaparece com probabilidade
$1$, então $ V(X) = 0$ (a família definidora tem termos nulos, exceto em um
conjunto nulo). Por outro lado, se $ V(X) = 0$, Chebyshev
(\cref{thm:b2:randomvar:markov}) dá $\P\bigl(\abs{X - m} \geq
\frac1n\bigr) \leq n^2\,V(X) = 0$ para cada $ n $; o \omterm{def:b2:proba:space}{eventos}
$\bigl\{\abs{X - m} \geq \frac1n\bigr\}$ aumentar para $\{X \neq
m\}$, tão monótono continuidade (\cref{thm:b2:proba:continuity})
rendimentos $\P(X \neq m) = 0$.
\end{solution}

\begin{solution}{exo:b2:randomvar:7}
$\E(S_n) = \frac n2$ e $ V(S_n) = \frac n4$.
\emph{Markov:} $\P\bigl(S_n \geq \frac{3n}4\bigr) \leq
\frac{n/2}{3n/4} = \frac23$ --- um limite constante, inútil para
grande $ n $.
\emph{Chebyshev:} o evento implica $\abs{S_n - \frac n2} \geq
\frac n4$, então a probabilidade é $\leq \frac{n/4}{(n/4)^2} =
\frac4n $ --- decai, mas apenas polinomialmente.
\emph{Chernoff:} por independência, $\E(e^{tS_n}) =
\prod_{i=1}^n\E(e^{tX_i}) = \bigl(\frac{1 + e^t}{2}\bigr)^n $ e
Markov se candidatou a $ e^{tS_n} \geq e^{3nt/4}$ dá, para cada $ t >
0$,
\[
\P\Bigl(S_n \geq \frac{3n}4\Bigr)
\leq \Bigl(\frac{1 + e^t}{2}\Bigr)^n e^{-3nt/4}
= \exp\Bigl(n\bigl(\ln\tfrac{1 + e^t}{2} - \tfrac{3t}4\bigr)\Bigr).
\]
Minimize o expoente: $\frac{\dd}{\dd t}\ln\frac{1+e^t}{2} =
\frac{e^t}{1 + e^t} = \frac34$ no $ e^t = 3$, ou seja\ $ t = \ln 3$,
dando
\[
\P\Bigl(S_n \geq \frac{3n}4\Bigr)
\leq \Bigl(\frac{4}{2}\Bigr)^n 3^{-3n/4}
= \bigl(2 \cdot 3^{-3/4}\bigr)^n \approx (0.877)^n ,
\]
exponencialmente pequeno. A hierarquia Markov $\to $ Tchebyshev $\to $
Chernoff é a escada padrão: cada degrau aplica Markov a um
função de crescimento mais rápido da variável.
\end{solution}

\begin{solution}{exo:b2:randomvar:8}
Pelo teorema da transferência (\cref{thm:b2:randomvar:transfer})
aplicado a $ f\bigl(\frac{S_n}{n}\bigr)$ com $ S_n \sim
\mathcal{B}(n, x)$:
\[
\E\Bigl[f\Bigl(\frac{S_n}{n}\Bigr)\Bigr]
= \sum_{k=0}^n f\Bigl(\frac kn\Bigr)\binom nk x^k(1-x)^{n-k}
= B_nf(x) .
\]
Consertar $\delta > 0$ e dividir $\abs{f(S_n/n) - f(x)}$ no evento $ D
= \bigl\{\abs{\frac{S_n}{n} - x} \geq \delta\bigr\}$:
desligado $ D $, a diferença é no máximo o módulo de continuidade
$\omega_f(\delta) = \sup_{\abs{s - t}\leq\delta}\abs{f(s) -
f(t)}$; sobre $ D $, no máximo $2\norm f_\infty $. Tirando \omterm{def:b2:randomvar:expectation}{expectativas} e
usando Chebyshev com $ V\bigl(\frac{S_n}{n}\bigr) =
\frac{x(1-x)}{n} \leq \frac{1}{4n}$:
\[
\abs{B_nf(x) - f(x)}
\leq \E\,\abs{f(S_n/n) - f(x)}
\leq \omega_f(\delta)
+ 2\norm f_\infty\,\P(D)
\leq \omega_f(\delta) + \frac{2\norm f_\infty}{4n\delta^2} .
\]
Uniforme continuidade de $ f $ sobre $[0, 1]$ faz $\omega_f(\delta) \to
0$: escolher $\delta $ então $ n $, e $ B_nf \to f $ \omterm{def:b2:funcseq:def}{uniformemente} --- o
Teorema de aproximação de Weierstrass de \cref{ch:b2:funcseq}, cujo
O ``lema da contagem'' agora é reconhecível como a desigualdade de Chebyshev
para o binômio \omterm{def:b2:randomvar:law}{lei}.
\end{solution}

\begin{solution}{exo:b2:randomvar:9}
Expandir $ S_n^4 = \sum_{i,j,k,l}X_iX_jX_kX_l $ e pegue
\omterm{def:b2:randomvar:expectation}{expectativas}. Por independência e centralização, qualquer termo contendo
um índice que aparece exatamente uma vez desaparece ($\E(X_i) = 0$ fatores
fora). Termos sobreviventes: o $ n $ termos diagonais $\E(X_i^4)$, e o
termos emparelhando dois pares de índices iguais; $\E(X_i^2X_j^2) =
\E(X_1^2)^2$ para $ i \neq j $, aparecendo $3n(n-1)$ vezes: escolha
o par não ordenado de valores ($\binom n2$ maneiras), então o
$\frac{4!}{2!\,2!} = 6$ maneiras de colocá-los nos quatro slots ---
$6\binom n2 = 3n(n-1)$. Portanto, com $\E(X_1^2)^2
\leq \E(X_1^4)$ (Jensen ou Cauchy--Schwarz),
\[
\E(S_n^4) = n\,\E(X_1^4) + 3n(n-1)\,\E(X_1^2)^2
\leq C n^2,
\qquad C = 4\,\E(X_1^4) .
\]
Markov na ordem 4:
\[
\P\Bigl(\Bigl|\frac{S_n}{n}\Bigr| \geq \varepsilon\Bigr)
= \P\bigl(S_n^4 \geq n^4\varepsilon^4\bigr)
\leq \frac{Cn^2}{n^4\varepsilon^4}
= \frac{C}{n^2\varepsilon^4} ,
\]
um resumido série. Por Borel--Cantelli~1
(\cref{thm:b2:proba:borelcantelli}), para cada $ j $ o evento
$ B_j = \limsup_n\bigl\{\abs{S_n/n} \geq \frac1j\bigr\}$ tem
probabilidade $0$, então $\P\bigl(\bigcup_j B_j\bigr) = 0$ por
\omterm{def:b2:structures:countable}{contável} subaditividade. No complemento --- de probabilidade $1$
--- para cada $ j $ há $ N $ com $\abs{S_n/n} < \frac1j $ para
tudo $ n \geq N $: precisamente $ S_n/n \to 0$. A forte \omterm{def:b2:randomvar:law}{lei} dos grandes
os números são válidos em um quarto momento; removendo essa hipótese
(Teorema de Kolmogorov) é o trabalho do Ano 3.
\end{solution}

\begin{solution}{exo:b2:randomvar:10}
$\P(M \leq k) = \bigl(\frac k6\bigr)^2$ (ambos os dados no máximo
$ k $, independentemente), então $\P(M \geq k) = 1 -
\bigl(\frac{k-1}6\bigr)^2$ e
\[
\E(M) = \sum_{k=1}^6\P(M \geq k)
= 6 - \frac{0 + 1 + 4 + 9 + 16 + 25}{36}
= 6 - \frac{55}{36} = \frac{161}{36} \approx 4.47 ,
\]
confortavelmente acima da média $3.5$ de um único dado, como
máximo deveria ser.
\end{solution}

\begin{solution}{exo:b2:randomvar:11}
Com $ I_i = \mathbf 1_{\sigma(i) = i}$: $\P(\sigma(i) = i) =
\frac{(n-1)!}{n!} = \frac1n $, então $\E(F_n) = n\cdot\frac1n =
1$. Para $ i \neq j $: $\P(\sigma(i) = i, \sigma(j) = j) =
\frac{(n-2)!}{n!} = \frac1{n(n-1)}$, por isso
\[
\operatorname{Cov}(I_i, I_j) = \frac1{n(n-1)} - \frac1{n^2}
= \frac{1}{n^2(n-1)} .
\]
Pelo \omterm{def:b2:randomvar:variance}{variação} kit de ferramentas
(\cref{thm:b2:randomvar:variancerules}),
\[
V(F_n) = n\cdot\frac1n\Bigl(1 - \frac1n\Bigr)
+ n(n-1)\cdot\frac1{n^2(n-1)}
= 1 - \frac1n + \frac1n = 1 .
\]
Significar $1$, \omterm{def:b2:randomvar:variance}{variação} $1$, \omterm{def:b2:proba:independence}{independente} de $ n $ --- consistente
com o limite de Poisson do problema de correspondência
(\cref{exo:b2:proba:5}).
\end{solution}

\begin{solution}{exo:b2:randomvar:12}
$ T_n = \sum_{k=1}^nG_k $ onde $ G_k \sim \mathcal G(k/n)$ é
a hora de ver um brinquedo novo quando $ k $ faltam, as etapas
sendo \omterm{def:b2:proba:independence}{independente}. Por isso
\[
V(T_n) = \sum_{k=1}^n\frac{1 - k/n}{(k/n)^2}
\leq \sum_{k=1}^n\frac{n^2}{k^2}
\leq \frac{\pi^2}6\,n^2 ,
\]
por \cref{ex:b2:fourier:basel}. Com $\E(T_n) = nH_n $, $ H_n =
\sum_1^n\frac1k $ (\cref{exo:b2:randomvar:3}), Chebyshev dá,
para $\varepsilon > 0$,
\[
\P\bigl(\abs{T_n - nH_n} \geq \varepsilon\,n\ln n\bigr)
\leq \frac{\pi^2n^2/6}{\varepsilon^2n^2\ln^2 n}
= \frac{\pi^2}{6\,\varepsilon^2\ln^2n}
\xrightarrow[n\to\infty]{} 0 .
\]
Desde $ H_n \sim \ln n $, dividindo por $ n\ln n $ mostra
$ T_n/(n\ln n) \to 1$ em probabilidade: as flutuações de
$ T_n $ são de ordem $ n $, insignificante em relação à média $ n\ln n $.
\end{solution}

\begin{solution}{pb:b2:randomvar:1}
\textbf{1.} $\E(S_n) = \frac n2$ e Markov
(\cref{thm:b2:randomvar:markov}) dar $\P(S_n \geq an) \leq
\frac{n/2}{an} = \frac1{2a}$. Markov só conhece o significado: é
não consegue distinguir uma variável concentrada em $ n/2$ de um
espalhado entre $0$ e $ n $, então precifica a cauda como se tudo
a massa poderia ficar ali.

\textbf{2.} O binômio justo é \omterm{def:b2:quadratic:adjoint}{simétrico} sobre $ n/2$
($ S_n $ e $ n - S_n $ tem o mesmo \omterm{def:b2:randomvar:law}{lei}), então com $ x = n(a -
\frac12) > 0$ os dois \omterm{def:b2:proba:space}{eventos} $\{S_n - \frac n2 \geq x\}$ e
$\{S_n - \frac n2 \leq -x\}$ são disjuntos e equiprováveis:
$\P(S_n \geq an) = \frac12\P(\abs{S_n - \frac n2} \geq x)$.
Chebyshev com $ V(S_n) = \frac n4$:
\[
\P(S_n \geq an)
\leq \frac12\cdot\frac{n/4}{n^2(a - 1/2)^2}
= \frac1{8n(a - 1/2)^2},
\]
o que é $\frac2n $ no $ a = \frac34$.

\textbf{3.} Por independência e o teorema do produto,
$\E(\eu^{tS_n}) = \bigl(\E \eu^{tX_1}\bigr)^n = \bigl(\frac{1
+ \eu^t}2\bigr)^n $. Markov se candidatou a $\eu^{tS_n}$:
\[
\P(S_n \geq an) \leq \eu^{-tan}\Bigl(\frac{1 +
\eu^t}2\Bigr)^{\!n} = \exp\Bigl(n\bigl(\ln\tfrac{1 +
\eu^t}2 - ta\bigr)\Bigr).
\]
A derivada do expoente em $ t $ é $\frac{\eu^t}{1 + \eu^t}
- a $, desaparecendo em $\eu^t = \frac a{1-a}$, ou seja\ $ t^* =
\ln\frac a{1-a} > 0$; lá $\frac{1 + \eu^{t^*}}2 =
\frac1{2(1-a)}$ e o expoente é igual
\[
n\Bigl(-\ln 2 - \ln(1-a) - a\ln\frac a{1-a}\Bigr)
= -n\bigl(\ln2 + a\ln a + (1-a)\ln(1-a)\bigr) = -n\,I(a),
\]
com $ I(\frac12) = 0$ e $ I'(a) = \ln\frac a{1-a} > 0$ sobre
$\intoo{\frac12}1$: $ I(a) > 0$. No $ a = \frac34$:
$\eu^{-I(3/4)} = \frac12(\tfrac34)^{-3/4}(\tfrac14)^{-1/4} =
2\cdot3^{-3/4}$, o limite de \cref{exo:b2:randomvar:7}.

\textbf{4.} O $ n + 1$ números $\binom nja^j(1-a)^{n-j}$ soma
para $1$, e o maior é aquele em $ j = k = an $ (o modo
de $\mathcal B(n, a)$ é $\floor{(n+1)a} = k $ aqui). Um
máximo de $ n + 1$ números somando a $1$ é pelo menos
$\frac1{n+1}$:
\[
\binom nk a^k(1-a)^{n-k} \geq \frac1{n+1}
\quad\Longrightarrow\quad
\binom nk \geq \frac{a^{-an}(1-a)^{-n(1-a)}}{n+1}
= \frac{\eu^{nH(a)}}{n+1}.
\]
Por isso $\P(S_n \geq an) \geq \binom{n}{an}2^{-n} \geq
\eu^{n(H(a) - \ln2)}/(n+1) = \eu^{-nI(a)}/(n+1)$: até o
fator polinomial $ n + 1$, o expoente de Chernoff é a verdade.

\textbf{5.} $ n = 100$, $ a = \frac34$: Markov $\frac23$;
Tchebyshev $\frac2{100} = 0.02$; Chernoff
$(2\cdot3^{-3/4})^{100} = \eu^{-100\,I(3/4)} \approx
2.1\cdot10^{-6}$, contra o exato $2.8\cdot10^{-7}$. Moral:
cada momento de informação divide o limite polinomialmente;
o momento exponencial muda seu \emph{natureza}.

\textbf{6.} $\cosh t = \sum_{k\geq0}\frac{t^{2k}}{(2k)!}$ e
$\eu^{t^2/2} = \sum_{k\geq0}\frac{t^{2k}}{2^kk!}$; a reivindicação
segue termo por termo de $(2k)! \geq 2^kk!$, que vale por
indução: $(2k)! = 2k(2k-1)\cdot(2k-2)! \geq 2k\cdot
2^{k-1}(k-1)! = 2^kk!\cdot(2k-1) \geq 2^kk!$.

\textbf{7.} Por independência, $\E\bigl(\eu^{t\sum\varepsilon_i}
\bigr) = (\cosh t)^n \leq \eu^{nt^2/2}$, então Markov dá
$\P(\sum\varepsilon_i \geq s) \leq \eu^{nt^2/2 - ts}$;
minimizando em $ t = s/n $ rendimentos $\eu^{-s^2/(2n)}$.

\textbf{8.} Com $ X_i = \frac{1 + \varepsilon_i}2$, $\widehat
p_n - \frac12 = \frac1{2n}\sum\varepsilon_i $, então
$\{\widehat p_n - \frac12 \geq \delta\} =
\{\sum\varepsilon_i \geq 2n\delta\}$ e a questão 7 dá a
vinculado $\eu^{-(2n\delta)^2/(2n)} = \eu^{-2n\delta^2}$. O
evento \omterm{def:b2:quadratic:adjoint}{simétrico} tem o mesmo limite, de onde o fator $2$ para
$\abs{\widehat p_n - \frac12} \geq \delta $.

\textbf{9.} $\E(\eu^{tX}) = \sum_x\eu^{tx}\P(X = x)$ é um
série de funções suaves de $ t $ cujo termo a termo
os derivativos são dominados, em todos \omterm{def:b2:metric:compact}{compactar} $ t $-intervalo, por
$\eu^{\abs t}\P(X = x)$ (como $0 \leq x \leq 1$): pelo
teorema de diferenciação para \omterm{def:b2:funcseq:series}{normalmente} série convergente
(\cref{thm:b2:funcseq:differentiation}) é duas vezes
\omterm{def:b2:diffcalc:differential}{diferenciável}, e a regra do quociente dá $\psi' = \E_t(X)$
e $\psi'' = \E_t(X^2) - \E_t(X)^2$, onde $\E_t $ é o
\omterm{def:b2:randomvar:expectation}{expectativa} para os pesos reponderados
$\eu^{tx}\P(X{=}x)/\E(\eu^{tX})$ --- não negativo, somando
$1$, transportado pelos mesmos valores $ x \in \intcc01$. UM \omterm{def:b2:randomvar:variance}{variação}
de um $\intcc01$-variável com valor é no máximo $\frac14$: por
\cref{exo:b2:randomvar:6}, isso é $\min_c\E_t((X - c)^2) \leq
\E_t\bigl((X - \tfrac12)^2\bigr) \leq \tfrac14$. Taylor com
resto integral, usando $\psi(0) = 0$, $\psi'(0) = p $:
\[
\psi(t) = tp + \int_0^t(t - s)\,\psi''(s)\,\dd s
\leq tp + \frac{t^2}2\cdot\frac14,
\]
ou seja\ $\E(\eu^{t(X - p)}) \leq \eu^{t^2/8}$ para tudo de verdade $ t $.

\textbf{10.} Por independência, $\E\bigl(\eu^{t(S_n -
np)}\bigr) \leq \eu^{nt^2/8}$; Markov e a otimização $ t
= 4\delta $ dar
\[
\P(\widehat p_n - p \geq \delta)
\leq \eu^{nt^2/8 - tn\delta}\Big|_{t = 4\delta}
= \eu^{-2n\delta^2};
\]
aplicando isso às variáveis $1 - X_i $ (também em
$\intcc01$) limita a outra cauda, de onde os dois lados
$2\eu^{-2n\delta^2}$.

\textbf{11.} Chebyshev precisa apenas de um segundo momento e dá
$\frac{p(1-p)}{n\delta^2}$; Necessidades de Hoefding
\emph{limitação} e dá $2\eu^{-2n\delta^2}$. No $ p =
\frac12$, $\delta = 0.03$: os limites são
$\frac{278}{n}$ (aproximadamente) versus
$2\eu^{-0.0018n}$; eles cruzam perto $ n \approx 1200$, depois
qual o limite exponencial vence, e de forma selvagem ($ n = 5000$:
$0.056$ contra $2.5\cdot10^{-4}$).

\textbf{12.} Por Hoeffding (pergunta 10), $\P(\abs{\widehat
p_n - p} \geq \delta) \leq 2\eu^{-2n\delta^2} \leq \alpha $ assim que $2n\delta^2 \geq \ln\frac2\alpha $, ou seja\ $ n \geq
\frac{\ln(2/\alpha)}{2\delta^2}$.

\textbf{13.} $\delta = 0.03$, $\alpha = 0.05$: $ n \geq
\frac{\ln 40}{2\cdot0.0009} \approx 2049.4$: $2050$ pessoas.
Para $\delta = 0.01$: $ n \geq \frac{\ln40}{0.0002} \approx
18\,445$. O tamanho da população nunca aparece porque cada
eleitor amostrado é modelado como um novo Bernoulli $(p)$ sorteio: o
a dificuldade da enquete é a \omterm{def:b2:randomvar:variance}{variação} de uma moeda, não do tamanho de
o país. Reduzir a margem pela metade custa quatro vezes a amostra
--- o $1/\delta^2$ \omterm{def:b2:randomvar:law}{lei}.

\textbf{14.} Chebyshev: $\P(\abs{\widehat p_n - p} \geq
\delta) \leq \frac{p(1-p)}{n\delta^2} \leq
\frac1{4n\delta^2} \leq \alpha $ para $ n \geq
\frac1{4\alpha\delta^2}$, ou seja\ $5556$ em três pontos ---
sobre $2.7$ vezes o requisito de Hoeffding. Sem
substituição, o \omterm{def:b2:randomvar:variance}{variação} é multiplicado por $\frac{N -
n}{N-1} < 1$ (\cref{exo:b2:randomvar:5}), então o mesmo $ n $ pode
só fazer melhor: o cálculo com substituição é o
conservador.

\textbf{15.} $\{\widehat p_n \leq \frac12\} \subseteq
\{\widehat p_n - 0.52 \leq -0.02\}$, então pelo unilateral
Hoeffding com destino $\P(\widehat p_n \leq \tfrac12) \leq
\eu^{-2n(0.02)^2} \leq 0.01$ assim que $ n \geq \frac{\ln
100}{2\cdot0.0004} \approx 5756.5$: $5757$ eleitores. O custo
escalas como o inverso do quadrado do \emph{liderar}, não do
precisão desejada: corridas acirradas são caras.

\textbf{16.} Usado: a amostra é retirada \omterm{def:b2:funcseq:def}{uniformemente} e
independentemente do eleitorado; cada pessoa amostrada
responde, honestamente, e $ p $ não se move durante a votação. Verdadeiro
pesquisas violam todos os três: entrevistados acessíveis e dispostos
não são uma amostra uniforme (viés de seleção e não resposta),
e as respostas podem ser falsas ou instáveis. Estes são
\emph{viés} erros: eles mudam $\E(\widehat p_n)$ longe de
$ p $ por um montante \omterm{def:b2:proba:independence}{independente} $ n $, então não há tamanho de amostra
os reduz --- a matemática desta parte controla apenas
o prazo de flutuação.

\textbf{17.} Chebyshev para um grupo de tamanho $ m $: $\P(\abs{
\widehat p^{(i)} - p} \geq \delta) \leq \frac{1}{4m\delta^2}
\leq \frac18$ para $ m \geq \frac2{\delta^2}$. Se menos que
$ k/2$ grupos erram, então mais de $ k/2$ dos valores
$\widehat p^{(i)}$ deitar no \omterm{def:b2:metric:topology}{abrir} intervalo $\intoo{p -
\delta}{p + \delta}$, e o mesmo acontece com sua mediana; por isso
$\{\abs{M - p} \geq \delta\}$ forças pelo menos $\lceil
k/2\rceil $ erros entre $ k $ \omterm{def:b2:proba:independence}{independente} grupos. A união
ligado sobre o $\binom k{\lceil k/2\rceil}$ possíveis conjuntos de
grupos errantes dão
\[
\P(\abs{M - p} \geq \delta)
\leq \binom{k}{\lceil k/2\rceil}
\Bigl(\frac18\Bigr)^{k/2}
\leq 2^k\,8^{-k/2} = 2^{-k/2} :
\]
decadência exponencial no número de grupos, comprado com
nada além de variações --- útil precisamente quando os pedidos
são ilimitados e Hoeffding não está disponível.

\textbf{18.} Cauchy-Schwarz
(\cref{thm:b2:randomvar:jensen}):
\[
\E(X) = \E(X\,\mathbf 1_{X>0})
\leq \sqrt{\E(X^2)}\sqrt{\E(\mathbf 1_{X>0}^2)}
= \sqrt{\E(X^2)\,\P(X > 0)} ;
\]
quadrado e dividido.

\textbf{19.} Deixar $ h(a) = I(a) - 2(a - \tfrac12)^2$. Então
$ h(\tfrac12) = 0$, $ h'(a) = \ln\frac a{1-a} - 4(a -
\tfrac12)$ desaparece em $\tfrac12$, e
\[
h''(a) = \frac1a + \frac1{1-a} - 4 = \frac{1}{a(1-a)} - 4
\geq 0
\]
desde $ a(1-a) \leq \frac14$. Então $ h'$ aumenta de $0$ sobre
$\intco{\frac12}1$, por isso $ h' \geq 0$ e $ h \geq 0$: $ I(a)
\geq 2(a - \tfrac12)^2$.

\textbf{20.} $ I(\tfrac12) = I'(\tfrac12) = 0$, $ I''(a) =
\frac1{a(1-a)}$ dá $ I''(\tfrac12) = 4$, e $ I'''(\tfrac12)
= 0$ (a função é \omterm{def:b2:quadratic:adjoint}{simétrico} sobre $\tfrac12$), então
$ I(\tfrac12 + \delta) = 2\delta^2 + O(\delta^4)$. Pergunta 4
então limita a cauda verdadeira \emph{abaixo} por
$\eu^{-n(2\delta^2 + O(\delta^4))}/(n+1)$: para pequenos $\delta $
o expoente de Hoeffding $2n\delta^2$ é assintoticamente exato
--- apenas polinômio em-$ n $ melhorias são possíveis.

\textbf{21.} Markov: um momento, decadência $1/a $, útil apenas como
o motor atrás dos demais (a questão 1 mostra que está vazio).
Chebyshev: dois momentos, decadência $\frac{V}{n\delta^2}$, afiado
sem mais hipóteses
(\cref{ex:b2:randomvar:chebsharp}) e a melhor ferramenta em
questão 23. Quarto momento (\cref{exo:b2:randomvar:9}):
decadência $ C/n^2$, apenas o suficiente somabilidade por uma \omterm{def:b2:randomvar:law}{lei} forte.
Hoeffding: variáveis limitadas, decaimento $2\eu^{-2n\delta^2}$,
o carro-chefe da Parte III. Chernoff com a taxa exata
$ I(a)$: momentos exponenciais \omterm{def:b2:metric:complete}{completos}, expoente imbatível
(questões 4, 20), ponto de referência para todo o resto.

\textbf{22.} Se a moeda for honesta: $\P(\widehat p_n > 0.525)
\leq \P(\widehat p_n - \tfrac12 \geq 0.025) \leq
\eu^{-2n(0.025)^2}$. Se $ p = 0.55$: $\P(\widehat p_n \leq
0.525) \leq \P(\widehat p_n - 0.55 \leq -0.025) \leq
\eu^{-2n(0.025)^2}$. Ambos os erros estão abaixo $0.01$ quando
$2n(0.025)^2 \geq \ln 100$, ou seja\ $ n \geq 3684.2$: $3685$
vira. (Distinguindo hipóteses $2.5$ custos de pontos separados
que estimativa para $\pm2.5$ custos de pontos.)

\textbf{23.} Hoefding: $ n \geq \frac{\ln 40}{2(0.005)^2}
\approx 73\,778$. Chebyshev com o verdadeiro \omterm{def:b2:randomvar:variance}{variação} $ p(1-p)
= 0.0099$: $ n \geq \frac{0.0099}{0.05\cdot(0.005)^2} =
7920$ --- nove vezes mais barato. Expoente de Hoeffding
$2n\delta^2$ preços o \omterm{def:b2:randomvar:variance}{variação} no seu pior caso $\frac14$,
absurdamente pessimista quando $ p = 0.01$; o humilde segundo
momento sabe melhor. A ferramenta que falta é ciente da \omterm{def:b2:randomvar:variance}{variação}
limite exponencial (desigualdade de Bernstein, ano 3) --- ou,
para raro \omterm{def:b2:proba:space}{eventos}, a aproximação de Poisson provou em
\cref{ch:b2:genfun}, que funciona no relativo natural
escala.

\textbf{24.} Consertar $\delta > 0$: $\sum_n 2\eu^{-2n\delta^2} <
\infty $ (série do tipo geométrico), então Borel--Cantelli ~ 1
(\cref{thm:b2:proba:borelcantelli}) dá
$\P(\abs{\widehat p_n - p} \geq \delta \text{ infinitely
often}) = 0$, ou seja\ o evento $ E_j =
\bigcup_N\bigcap_{n\geq N}\{\abs{\widehat p_n - p} <
\tfrac1j\}$ tem probabilidade $1$ para cada $ j $. O \omterm{def:b2:structures:countable}{contável}
interseção $\bigcap_jE_j $ ainda tem probabilidade $1$
(subaditividade nos complementos), e nele $\widehat p_n
\to p $: a \omterm{def:b2:randomvar:law}{lei} forte dos grandes números para lançamentos de moeda, com
limitação desempenhando o papel que o quarto momento desempenhou em
\cref{exo:b2:randomvar:9}.

\textbf{25.} Um momento compra um limite fixo; dois comprar
$1/(n\delta^2)$, e nada mais (o exemplo de nitidez); quatro
comprar $1/n^2$, o suficiente para transformar-se em uma \omterm{def:b2:randomvar:law}{lei} quase certa;
compras limitadas $\eu^{-2n\delta^2}$; e o \omterm{def:b2:metric:complete}{completo}
momento exponencial compra a taxa exata $ I $, que nenhum método
batidas. Votação $2050$ pessoas é suficiente para qualquer país porque
a flutuação da amostra é governada pela moeda \omterm{def:b2:randomvar:variance}{variação},
não o tamanho da população --- o $1/\delta^2$ e
$\ln(1/\alpha)$ as etiquetas de preços são universais. O volume do Ano 3
o teorema do limite central substitui essas desigualdades, pelo
$\sqrt n$ escala, por um limite exato \omterm{def:b2:randomvar:law}{lei} com explícito
constantes --- transformando cada limite deste problema em um
igualdade assintótica.
\end{solution}
