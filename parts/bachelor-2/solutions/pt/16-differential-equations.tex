\section*{Capítulo \ref{ch:b2:diffeq} --- Equações Diferenciais}

\begin{solution}{exo:b2:diffeq:1}
Primeira matriz: valores próprios $1, 2$, vetores próprios $(1,0)$ e $(1,1)$.
Decompor $ X_0 = (1,0) = 1\cdot(1,0) + 0\cdot(1,1)$: a solução
é
\[
X(t) = \eu^{t}\begin{pmatrix}1\\ 0\end{pmatrix} .
\]
(O vetor inicial é ele próprio um vetor próprio.)

Segundo: $ A = 2I + N $, $ N = E_{12}$, $ N^2 = 0$: $\eu^{tA} =
\eu^{2t}(I + tN)$, então
\[
X(t) = \eu^{2t}\begin{pmatrix} 1 & t\\ 0 & 1\end{pmatrix}
\begin{pmatrix}1\\ 0\end{pmatrix}
= \eu^{2t}\begin{pmatrix}1\\ 0\end{pmatrix}.
\]
\end{solution}

\begin{solution}{exo:b2:diffeq:2}
$ y' = \sin(ty)$: $\abs{\sin(ty) - \sin(tz)} \leq \abs t\,\abs{y -
z}$ --- \omterm{def:b2:metric:continuity}{Lipschitz} em $ y $ \omterm{def:b2:funcseq:def}{uniformemente} em cada segmento de tempo:
soluções globais únicas em $\R $ (aplique o teorema em cada
segmento).

$ y' = y^2$: localmente \omterm{def:b2:metric:continuity}{Lipschitz} apenas; nenhum teorema global, e de fato
$ y(0) = 1$ explode em $ t = 1$.

$ y' = \abs y $: $\abs\cdot $ é $1$-Lipschitz: existência global e
singularidade. Com $ y(0) = 0$: $ y \equiv 0$ (singularidade!). Com $ y(0)
= 1$: $ y $ permanece positivo (não pode cruzar a solução zero), então $ y' =
y $: $ y = \eu^t $.
\end{solution}

\begin{solution}{exo:b2:diffeq:3}
Se $ y(t_1) = z(t_1)$ em algum momento, então $ y $ e $ z $ resolver o mesmo
Problema de Cauchy em $ t_1$: forças de singularidade $ y = z $ em seu comum
intervalo --- soluções distintas nunca se encontram.

Para $ y' = y(1-y)$: as constantes $0$ e $1$ são soluções. Um
solução começando em $\intoo{0}{1}$ nunca pode alcançar $0$ ou $1$
(colideria com uma solução constante): permanece em
$\intoo{0}{1}$, portanto, é global (limitado: sem explosão --- por exemplo\
por \cref{exo:b2:diffeq:9}critério de, ou porque o campo vetorial
está delimitado na faixa presa).
\end{solution}

\begin{solution}{exo:b2:diffeq:4}
$(A - 2I)^2 = \begin{pmatrix}1 & 1\\ -1 & -1\end{pmatrix}^2 = 0$:
Dunford com $ D = 2I $, $ N = A - 2I $:
\[
\eu^{tA} = \eu^{2t}\,(I + tN)
= \eu^{2t}\begin{pmatrix} 1 + t & t\\ -t & 1 - t\end{pmatrix}.
\]
Variação de constantes com $ B(s) = (\eu^{2s}, 0)^{\mathsf T}$:
\[
X(t) = \int_0^t \eu^{(t-s)A}B(s)\,\dd s
= \eu^{2t}\int_0^t \begin{pmatrix} 1 + (t-s)\\ -(t-s)
\end{pmatrix}\dd s
= \eu^{2t}\begin{pmatrix} t + \frac{t^2}{2}\\[2pt] -\frac{t^2}{2}
\end{pmatrix},
\]
usando $\eu^{(t-s)A}B(s) = \eu^{2(t-s)}(I + (t-s)N)\,(\eu^{2s},
0)^{\mathsf T} = \eu^{2t}\bigl(1 + (t-s),\, -(t-s)\bigr)^{\mathsf
T}$. (Verificar: $ X(0) = 0$; $ X' - AX = (\eu^{2t}, 0)^{\mathsf T}$
por diferenciação.)
\end{solution}

\begin{solution}{exo:b2:diffeq:5}
\emph{Série:} $ A = \alpha I + \beta J $ com $ J =
\begin{pmatrix}0 & -1\\ 1 & 0\end{pmatrix}$, $ J^2 = -I $; os dois
as convocações comutam, então $\eu^{tA} = \eu^{\alpha t}\,\eu^{\beta tJ}$,
e a série de $\eu^{\beta t J}$ divide em potências pares/ímpares
em $\cos(\beta t)I + \sin(\beta t)J $: a escala de rotação declarada
matriz.

\emph{Complexo:} identificar $(x, y) \in \R^2$ com $ z = x + \iu y $;
o sistema $ X' = AX $ lê $ z' = (\alpha + \iu\beta)z $, cujo
solução $ z(t) = \eu^{\alpha t}\eu^{\iu\beta t}z_0$ é exatamente o
espiral: módulo $\eu^{\alpha t}$, argumento avançando em velocidade
$\beta $.
\end{solution}

\begin{solution}{exo:b2:diffeq:6}
Deixar $ v(t) = \eu^{-k(t - t_0)}\int_{t_0}^t u $. Então
\[
v'(t) = \eu^{-k(t-t_0)}\Bigl(u(t) - k\int_{t_0}^t u\Bigr)
\leq C\,\eu^{-k(t-t_0)},
\]
pela hipótese. Integrando de $ t_0$ para $ t $ ($ v(t_0) = 0$):
$ v(t) \leq \frac{C}{k}\bigl(1 - \eu^{-k(t - t_0)}\bigr)$, ou seja\
$ k\int_{t_0}^t u \leq C\bigl(\eu^{k(t-t_0)} - 1\bigr)$; alimentação
isso de volta à hipótese: $ u(t) \leq C\eu^{k(t-t_0)}$.

Singularidade/dependência: duas soluções $ y, z $ da integral
equação satisfazer
\[
\norm{y(t) - z(t)} \leq \norm{y_0 - z_0} +
k\int_{t_0}^{t}\norm{y - z},
\]
e Gronwall com $ C = \norm{y_0 - z_0}$ dá o exponencial
vinculado; $ C = 0$ dá exclusividade.
\end{solution}

\begin{solution}{exo:b2:diffeq:7}
Substituir $ y = y_1 z $ com $ y_1 = \frac{\sin t}{t}$: o geral
fórmula de redução (\cref{prop:b2:diffeq:secondorder}) dá, por $ u
= z'$,
\[
y_1 u' + \Bigl(2y_1' + \frac{2}{t}\,y_1\Bigr)u = 0
\]
(a equação normalizada como $ y'' + \frac2t y' + y = 0$). Calcular
$2y_1' + \frac2t y_1 = 2\,\frac{t\cos t - \sin t}{t^2} +
\frac{2\sin t}{t^2} = \frac{2\cos t}{t}$: então
\[
\frac{u'}{u} = -\frac{2\cos t}{t}\cdot\frac{t}{\sin t}
= -2\cot t
\quad\Longrightarrow\quad
u = \frac{1}{\sin^2 t}
\quad (\text{up to a constant}),
\]
e $ z = -\cot t $, dando $ y_2 = y_1 z = -\frac{\cos t}{t}$.
Solução geral em $\intoo{0}{\pi}$:
\[
y(t) = \alpha\,\frac{\sin t}{t} + \beta\,\frac{\cos t}{t} .
\]
(Estas são as funções esféricas de Bessel de ordem zero.)
\end{solution}

\begin{solution}{exo:b2:diffeq:8}
Dunford: $ A = D + N $ deslocamento, $ D $ \omterm{def:b2:reduction:diag}{diagonalizável} com o mesmo
valores próprios, $ N $ nilpotente, então
\[
\eu^{tA} = \eu^{tD}\,\eu^{tN},
\qquad
\eu^{tN} = \sum_{k < n} \frac{t^kN^k}{k!}
\ \text{(a matrix polynomial in } t).
\]
Deixar $-2\alpha = \max_i \Re\lambda_i < 0$. Numa base diagonalizando
$ D $, $\vertiii{\eu^{tD}} \leq \eu^{-2\alpha t}$ (entradas
$\eu^{t\lambda_i}$ de módulo $\eu^{t\Re\lambda_i}$); \omterm{def:b2:nvs:norm}{normas} em
bases diferentes diferem por constantes. Por isso
\[
\norm{X(t)} \leq \vertiii{\eu^{tA}}\,\norm{X_0}
\leq C'\,\eu^{-2\alpha t}\,(1 + t)^{n-1}\,\norm{X_0}
\leq C\,\eu^{-\alpha t}\norm{X_0} ,
\]
absorvendo o polinômio em um fator exponencial
($\eu^{-\alpha t}(1+t)^{n-1} \to 0$, portanto limitado).
\end{solution}

\begin{solution}{exo:b2:diffeq:9}
Deixar $ y $ seja uma solução máxima em $\intco{0}{T}$, $ T \leq \infty $,
e suponha $ T < \infty $. A forma integral dá, por $ t < T $,
\[
\norm{y(t)} \leq \norm{y_0} + \int_0^t \bigl(a\norm{y(s)} +
b\bigr)\dd s
\leq \bigl(\norm{y_0} + bT\bigr) + a\int_0^t\norm{y(s)}\,\dd s ,
\]
e limites de Gronwall $\norm{y(t)} \leq (\norm{y_0} +
bT)\,\eu^{aT} =: M $ sobre $\intco{0}{T}$: a solução permanece em um
\omterm{def:b2:metric:compact}{compactar} bola. Então $ y' = f(t, y)$ é delimitado perto $ T $, então $ y $ é
\omterm{def:b2:metric:continuity}{Lipschitz} aproximar $ T $ e estende continuamente para $ T $ (Cauchy
critério); resolvendo o problema de Cauchy em $(T, y(T))$ prolonga $ y $
além $ T $, contradizendo a maximidade. Por isso $ T = \infty $: não
fuga em tempo finito sob crescimento linear.
\end{solution}

\begin{solution}{exo:b2:diffeq:10}
Raízes características de $ r^2 - 3r + 2$: $1$ e $2$, então o
soluções homogêneas são $ a\eu^t + b\eu^{2t}$. O palpite
$\alpha\eu^t $ falha porque $\eu^t $ já resolve o
equação homogênea (a raiz $ r = 1$ ``ressoa'' com o
lado direito). Com $ y = \alpha t\,\eu^t $: $ y' = \alpha(1 +
t)\eu^t $, $ y'' = \alpha(2 + t)\eu^t $, e
\[
y'' - 3y' + 2y = \alpha\eu^t\bigl(2 + t - 3 - 3t + 2t\bigr)
= -\alpha\,\eu^t :
\]
$\alpha = -1$, $ y_p = -t\,\eu^t $. Solução geral: $ y = a\eu^t
+ b\eu^{2t} - t\eu^t $. Dados iniciais $ y(0) = y'(0) = 0$: $ a + b
= 0$ e $ a + 2b - 1 = 0$: $ b = 1$, $ a = -1$:
\[
y(t) = \eu^{2t} - (1 + t)\,\eu^{t} .
\]

\end{solution}

\begin{solution}{exo:b2:diffeq:11}
$ A = \lambda I + N $ com $ N = E_{12} + E_{23}$: $ N^2 = E_{13}$,
$ N^3 = 0$, e $\lambda I $ viaja com $ N $:
\[
\eu^{tA} = \eu^{\lambda t}\Bigl(I + tN +
\frac{t^2}{2}N^2\Bigr)
= \eu^{\lambda t}\begin{pmatrix}
1 & t & \frac{t^2}{2}\\
0 & 1 & t\\
0 & 0 & 1
\end{pmatrix}.
\]
Soluções: $ X(t) = \eu^{\lambda t}\bigl(X_0 + tNX_0 +
\frac{t^2}2N^2X_0\bigr)$ --- cada componente é $\eu^{\lambda
t}$ vezes um polinômio de grau $\leq 2$. O limite do diploma é
o índice de nilpotência menos um: a série de $\eu^{tN}$
trunca em $ N^2$.
\end{solution}

\begin{solution}{exo:b2:diffeq:12}
\begin{enumerate}
  \item Se $ X(T) = X(0)$, então $ Y(t) = X(t + T)$ resolve $ Y' =
        AY + B(t + T) = AY + B(t)$ com $ Y(0) = X(0)$:
        singularidade (\cref{thm:b2:diffeq:linear}) dá $ Y = X $,
        ou seja\ $ X $ é $ T $-periódico. O inverso é trivial.
  \item Trigonalizar $\C $: $ A = PT'P^{-1}$ com $ T'$ superior
        triangular, diagonal $(\lambda_i)$. Cada poder de um
        matriz triangular é triangular com diagonal
        $(\lambda_i^k)$, então $\eu^{TA} = P\eu^{TT'}P^{-1}$ é
        triangular na mesma base com diagonal
        $(\eu^{T\lambda_i})$: esses são os valores próprios. Então
        $ I - \eu^{TA}$ é invertível se $\eu^{T\lambda} \neq
        1$ para cada \omterm{def:b2:reduction:eigen}{autovalor}, ou seja,\ se $ T\lambda \notin
        2\iu\pi\Z $, que é a hipótese declarada.
  \item Variação de constantes: $ X(T) = \eu^{TA}X(0) +
        \int_0^T\eu^{(T-s)A}B(s)\dd s $, então $ X(T) = X(0)$ lê
        \[
        \bigl(I - \eu^{TA}\bigr)X(0) =
        \int_0^{T}\eu^{(T-s)A}B(s)\,\dd s ,
        \]
        que tem uma solução única $ X(0)$ sob o
        hipótese de invertibilidade: exatamente uma $ T $-periódico
        solução. Para o oscilador harmônico ($\lambda =
        \pm\iu\omega $), o caso excluído é $\omega T \in
        2\pi\Z $: forçando cujo período é um múltiplo do
        período natural --- \omterm{pb:b2:diffeq:1}{ressonância}, como o problema do fim de semana
        quantifica.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b2:diffeq:1}
\textbf{1.} O polinômio característico é $\lambda^2 -
\tau\lambda + \delta $, com raízes
$\frac{\tau\pm\sqrt\Delta}{2}$. Se $\delta < 0$ então $\Delta =
\tau^2 - 4\delta > 0$ e as duas raízes reais têm produto
$\delta < 0$: sinais opostos. Se $\delta > 0$ e $\Delta \geq
0$: raízes reais do produto $> 0$ e soma $\tau $: ambos os
sinal de $\tau $. Se $\Delta < 0$: par conjugado $\alpha \pm
\iu\beta $ com $\alpha = \frac\tau2$, $\beta =
\frac{\sqrt{-\Delta}}2$.

\textbf{2.} $ X(t) = a\,\eu^{\mu t}v_- + b\,\eu^{\lambda t}v_+$.
Órbitas com $ a = 0$ (resp.\ $ b = 0$) correr ao longo do instável
(resp.\ estável) linha própria; todos os outros têm $\norm X \to \infty $
em ambas as direções do tempo, assintótico para $\R v_+$ como $ t \to
+\infty $ e para $\R v_-$ como $ t \to -\infty $: a sela
foto. Limite em todos $\R $ forças $ b = 0$ (outro
explosão em $+\infty $) e $ a = 0$ (no $-\infty $): apenas o
origem.

\textbf{3.} Com $\mu < \lambda < 0$, ambas as exponenciais decaem:
$ X(t) \to 0$. Se $ b \neq 0$, fator $\eu^{\lambda t}$:
\[
X(t) = \eu^{\lambda t}\bigl(b\,v_\lambda + a\,\eu^{(\mu -
\lambda)t}v_\mu\bigr),
\qquad \eu^{(\mu-\lambda)t} \to 0 :
\]
a direção de $ X(t)$ tende a $\R v_\lambda $, o
\emph{lento} autodireção --- todas as órbitas, exceto o eixo rápido
chegar tangente a ele (o painel direito da fase do capítulo
retratos).

\textbf{4.} Na base onde $ A = \begin{pmatrix} \alpha &
-\beta\\ \beta & \alpha\end{pmatrix}$
(\cref{exo:b2:diffeq:5}; para os sistemas osciladores da Parte II
esta forma é alcançada por uma mudança real explícita de base), o
solução é $\eu^{\alpha t}$ vezes uma rotação do ângulo $\beta
t $: espirais logarítmicas, contraindo quando $\alpha = \frac\tau2
< 0$, expandindo quando $\tau > 0$ e curvas fechadas (elipses
nas coordenadas originais) quando $\tau = 0$: o centro.

\textbf{5.} $\Delta = 0$ dá o dobro \omterm{def:b2:reduction:eigen}{autovalor} $\lambda =
\frac\tau2$; por Cayley-Hamilton
(\cref{thm:b2:reduction:cayleyhamilton}), $(A - \lambda I)^2 =
0$, então $ N = A - \lambda I $ é nilpotente, comuta com
$\lambda I $, e $\eu^{tA} = \eu^{\lambda t}(I + tN)$. Se $ N =
0$: $ A = \lambda I $, todos os raios são órbitas (nó estrela). Se $ N
\neq 0$: $ X(t) = \eu^{\lambda t}(X_0 + tNX_0)$, e para $ NX_0
\neq 0$ a direção converge para a única direção própria
$\operatorname{im}N $: nó impróprio. Isso completa o
traço - imagem determinante.

\textbf{6.} $\tau = -2\zeta\omega $, $\delta = \omega^2 > 0$,
$\Delta = 4\omega^2(\zeta^2 - 1)$. Então: $0 < \zeta < 1$ dá
$\Delta < 0$, $\tau < 0$: espiral estável; $\zeta = 1$: $\Delta =
0$: nó estável degenerado; $\zeta > 1$: $\Delta > 0$, $\tau <
0$, $\delta > 0$: nó estável; $\zeta = 0$: $\tau = 0$,
$\delta > 0$: centro. Uma viagem vertical no plano em
$\delta = \omega^2$.

\textbf{7.} Raízes $ r = -\zeta\omega \pm
\omega\sqrt{\zeta^2-1}$. Para $\zeta < 1$: $ r = -\zeta\omega
\pm \iu\omega_d $, $\omega_d = \omega\sqrt{1-\zeta^2}$:
\[
x(t) = \eu^{-\zeta\omega t}\bigl(a\cos\omega_dt +
b\sin\omega_dt\bigr)
= R\,\eu^{-\zeta\omega t}\cos(\omega_dt - \varphi) .
\]
Para $\zeta = 1$: $ x = (a + bt)\eu^{-\omega t}$. Para $\zeta >
1$: $ x = a\eu^{r_-t} + b\eu^{r_+t}$, ambas as taxas negativas.
Máximos sucessivos de $\abs x $ no caso subamortecido ocorre em
tempos separados pelo pseudoperíodo $\frac{2\pi}{\omega_d}$
(mesma fase do cosseno), e sua razão é
$\eu^{-\zeta\omega\cdot2\pi/\omega_d} =
\eu^{-2\pi\zeta/\sqrt{1-\zeta^2}}$: o decremento logarítmico,
um medidor de amortecimento legível em um osciloscópio.

\textbf{8.} Racionalizando,
\[
\abs{\lambda_{\mathrm{slow}}} = \omega\bigl(\zeta -
\sqrt{\zeta^2-1}\bigr)
= \frac{\omega}{\zeta + \sqrt{\zeta^2 - 1}} ,
\]
cujo denominador aumenta com $\zeta \geq 1$: a decadência
taxa é maior em $\zeta = 1$, onde é igual $\omega $. Um
porta superamortecida fecha sem bater, mas \emph{devagar};
o amortecimento crítico é o \omterm{def:b2:structures:ideal}{ideal} do engenheiro.

\textbf{9.} $ E' = x'x'' + \omega^2xx' =
x'\bigl(-2\zeta\omega x' - \omega^2x\bigr) + \omega^2xx' =
-2\zeta\omega\,x'^2 \leq 0$. Se $ x $ eram periódicos e
inconstante, $ E $ seria periódico e não crescente, portanto
constante, forçando $ x' \equiv 0$: $ x $ constante e então
$\omega^2x = 0$: $ x \equiv 0$. Então, para $\zeta > 0$ o único
solução periódica é o descanso: o amortecimento mata todos \omterm{def:b2:structures:sn}{ciclo}.

\textbf{10.} O centro $\zeta = 0$ vive na linha $\tau =
0$ do plano traço-determinante --- um conjunto com vazio
interior: uma perturbação arbitrariamente pequena da matriz
(qualquer amortecimento físico) se move $\tau $ zero e desliga o
órbitas fechadas em espirais. Periodicidade do não amortecido
oscilador é, portanto, um fenômeno de ponta de navalha, não um robusto
um.

\textbf{11.} Substituindo $ x_p = \Re(z\eu^{\iu\gamma t})$ em
a equação:
\[
\bigl(-\gamma^2 + 2\iu\zeta\omega\gamma +
\omega^2\bigr)z = F
\quad\Longrightarrow\quad
z = \frac{F}{\omega^2 - \gamma^2 + 2\iu\zeta\omega\gamma},
\]
então $ x_p = \abs z\cos(\gamma t - \varphi)$ com $\varphi =
\arg(\omega^2 - \gamma^2 + 2\iu\zeta\omega\gamma)$, ou seja\
$\tan\varphi = \frac{2\zeta\omega\gamma}{\omega^2 -
\gamma^2}$, e o declarado $ A(\gamma) = \abs z $.

\textbf{12.} A diferença de duas soluções resolve o problema
equação homogênea, que para $\zeta > 0$ decai para $0$
(questão 7): toda solução é igual $ x_p $ mais um transitório
desaparecendo no infinito. O estado estacionário é um atrator global:
as condições iniciais são esquecidas, apenas $ A(\gamma)$ e o
atraso de fase $\varphi $ permanecer.

\textbf{13.} Minimizar $ g(u) = (\omega^2 - u)^2 +
4\zeta^2\omega^2u $ sobre $ u = \gamma^2 \geq 0$: $ g'(u) =
-2(\omega^2 - u) + 4\zeta^2\omega^2 = 0$ no $ u = \omega^2(1 -
2\zeta^2)$, interior se $\zeta < \frac{1}{\sqrt2}$. Lá
\[
g(u_*) = 4\zeta^4\omega^4 + 4\zeta^2\omega^4(1 - 2\zeta^2)
= 4\zeta^2\omega^4(1 - \zeta^2),
\qquad
A(\gamma_*) = \frac{F}{2\zeta\omega^2\sqrt{1 - \zeta^2}} .
\]
Contra a resposta estática $ A(0) = \frac{F}{\omega^2}$:
amplificação $\frac{1}{2\zeta\sqrt{1-\zeta^2}} \approx
\frac{1}{2\zeta}$ para pequeno $\zeta $ --- um levemente amortecido
sistema próximo $\gamma_* \approx \omega $ multiplica a entrada a
cem vezes mais quando $\zeta = 0.005$.

\textbf{14.} O declarado $ x $ satisfaz $ x(0) = x'(0) = 0$ e
\[
x'' + \omega^2 x = \frac{F(\omega^2 -
\gamma^2)\cos\gamma t}{\omega^2 - \gamma^2} = F\cos\gamma t
\]
(o $\cos\omega t $ peças canceladas). A forma do produto segue
de $\cos p - \cos q = 2\sin\frac{q+p}{2}\sin\frac{q-p}2$
com $ p = \gamma t $, $ q = \omega t $. Para $\gamma $ perto de
$\omega $, o fator $\sin\frac{(\omega-\gamma)t}2$ é um lento
envelope modulando a oscilação rápida
$\sin\frac{(\omega+\gamma)t}2$: batidas, com amplitude
$\frac{2F}{\abs{\omega^2-\gamma^2}}$ --- grande, mas limitado.

\textbf{15.} Para $ x_p = \frac{F}{2\omega}t\sin\omega t $:
\[
x_p'' = \frac{F}{2\omega}\bigl(2\omega\cos\omega t -
\omega^2t\sin\omega t\bigr)
= F\cos\omega t - \omega^2x_p :
\]
uma solução. E em fixo $ t $, deixando $\gamma \to \omega $ em
questão 14:
\[
\frac{2F\sin\frac{(\omega-\gamma)t}2
\sin\frac{(\omega+\gamma)t}{2}}
{(\omega-\gamma)(\omega+\gamma)}
\longrightarrow
\frac{2F\cdot\frac{(\omega-\gamma)t}2\big/(\omega-\gamma)
\cdot\sin\omega t}{2\omega}
= \frac{F\,t\sin\omega t}{2\omega} .
\]
A amplitude cresce linearmente sem limites: o \omterm{pb:b2:diffeq:1}{ressonância}
catástrofe --- a razão pela qual os soldados quebram os degraus nas pontes.

\textbf{16.} A onda quadrada carrega harmônicos em todos os ímpares
frequência $ n = 1, 3, 5, \dots $; por linearidade, cada harmônico
$ n $ é amplificado pela resposta do oscilador em $\gamma = n $.
Para $\omega = 3$ os hits do terceiro harmônico \omterm{pb:b2:diffeq:1}{ressonância} exatamente.
Os engenheiros temem entradas quadradas (e dente de serra) porque
excitar \emph{todos} harmônicos estranhos ao mesmo tempo: qualquer que seja o natural
frequência da estrutura, algum harmônico está esperando por isso.

\textbf{17.} $ W' = y_1y_2'' - y_1''y_2 = -qy_1y_2 + qy_1y_2 =
0$: $ W $ é constante (Liouville com um companheiro traço zero
matriz). $ W = 0$ em um ponto torna os dados iniciais de $ y_2$
proporcionais aos de $ y_1$, por isso $ y_2$ proporcional a
$ y_1$ (singularidade); $ W \neq 0$ se for independente. Se $ y(t_0) =
y'(t_0) = 0$ então $ y \equiv 0$ (singularidade): um diferente de zero
a solução tem zeros simples e um zero simples é isolado
($ y'$ de sinal fixo próximo).

\textbf{18.} Entre zeros consecutivos $ a < b $, $ y_1$ mantém
um sinal, digamos $ y_1 > 0$ sobre $\intoo ab $: então $ y_1'(a) > 0$ e
$ y_1'(b) < 0$ (zeros simples). Avaliando a constante $ W =
y_1y_2' - y_1'y_2$ no $ a $ e $ b $:
\[
W = -y_1'(a)\,y_2(a) = -y_1'(b)\,y_2(b) ,
\]
então $ y_2(a)$ e $ y_2(b)$ têm sinais opostos ($ W \neq 0$
proíbe desaparecer): $ y_2$ desaparece em $\intoo ab $
(valores intermediários). Não pode desaparecer duas vezes ali: dois zeros
de $ y_2$ flanquearia um zero de $ y_1$ pelo mesmo argumento com
os papéis trocados, contradizendo a consecutividade: exatamente
um zero --- entrelaçamento.

\textbf{19.} Suponha $ z $ não tem zero em $\intoo ab $; substituindo
$ y, z $ por seus negativos, assuma $ y > 0$ e $ z > 0$ sobre
$\intoo ab $. Definir $\varphi = yz' - y'z $: $\varphi' = yz'' -
y''z = (q_1 - q_2)\,yz \leq 0$ sobre $\intoo ab $: $\varphi $ é
não crescente. Mas $\varphi(a) = -y'(a)z(a) \leq 0$ (como
$ y'(a) > 0$, $ z(a) \geq 0$) e $\varphi(b) = -y'(b)z(b) \geq
0$ (como $ y'(b) < 0$, $ z(b) \geq 0$): uma função não crescente
fugindo de $\leq 0$ para $\geq 0$ desaparece de forma idêntica, então
$(q_1 - q_2)yz \equiv 0$ sobre $\intoo ab $. Se $ q_1 < q_2$
em algum lugar $\intoo ab $, isso é um absurdo ($ y, z > 0$ lá):
$ z $ deve desaparecer estritamente por dentro. Em geral ($ q_1 \leq q_2$),
qualquer $ z $ desaparece em $\intoo ab $, ou $\varphi \equiv 0$
forças $ z $ proporcional a $ y $, que desaparece em $ a $ e $ b $:
em todos os casos $ z $ tem um zero em $\intcc ab $.

\textbf{20.} Limite superior: comparar $ y $ (coeficiente $ q \geq
m^2$) com $ u(t) = \sin(m(t - a))$ (coeficiente $ m^2 \leq q $,
então $ y $ desempenha o papel de $ z $ na questão 19): se $ y $ não tinha
zero em $\intoc{a}{a + \pi/m}$, os zeros $ a $ e $ a +
\frac\pi m $ de $ u $ seria consecutivo com $ y \neq 0$
entre eles, contrariando a questão 19: zeros consecutivos de
$ y $ estão à distância $\leq \frac\pi m $. Limite inferior: se dois
zeros consecutivos $ a < b $ de $ y $ tive $ b - a < \frac\pi M $,
então $ z(t) = \sin(M(t-a))$ (coeficiente $ M^2 \geq q $) iria
tem que desaparecer em $\intcc ab \subset \intoo{a}{a + \pi/M}
\cup\{a\}$, onde seu único zero é $ a $ em si --- mas
questão 19 aplicada em $\intoo{a}{b}$ com rigor no
pontos finais dão um zero em $\intcc ab $, e $ z > 0$ sobre
$\intoc ab $: contradição. Por isso $\frac\pi M \leq b - a \leq
\frac\pi m $; para $ q \equiv \omega^2$ ambos os limites colapsam para
o espaçamento exato $\frac\pi\omega $ da harmônica
oscilador.

\textbf{21.} Com $ u = ty $: $ u'' = ty'' + 2y'$, então $ ty'' + 2y'
+ ty = u'' + u = 0$: $ u = A\sin t + B\cos t = R\sin(t +
\varphi)$, e $ y = \frac{u}{t}$ recupera $\frac{\sin t}t $ e
$\frac{\cos t}t $ (\cref{exo:b2:diffeq:7}) sem redução de
ordem. Os zeros de qualquer solução diferente de zero são aqueles de
$ R\sin(t + \varphi)$: espaçado exatamente $\pi $ ---Sturm's
filosofia em ação: o coeficiente $ q \equiv 1$ dita
os zeros, com fórmula ou sem fórmula.

\textbf{22.} Definir $ x(t) =
\frac1\omega\int_0^t\sin(\omega(t-s))F(s)\dd s $. Então $ x(0) =
0$;
\[
x'(t) = \frac1\omega\sin(0)F(t) +
\int_0^t\cos(\omega(t-s))F(s)\dd s
= \int_0^t\cos(\omega(t-s))F(s)\dd s ,
\]
então $ x'(0) = 0$; e $ x''(t) = F(t) -
\omega\int_0^t\sin(\omega(t-s))F(s)\dd s = F(t) - \omega^2x(t)$
(diferenciação de um \omterm{thm:b2:integration:continuity}{integral do parâmetro} com limite variável,
como no capítulo de integração). Com $ F(s) = F\cos\omega s $,
produto-soma dá
\[
\int_0^t\sin(\omega(t-s))\cos(\omega s)\dd s
= \frac12\int_0^t\bigl(\sin\omega t + \sin(\omega t -
2\omega s)\bigr)\dd s
= \frac{t}{2}\sin\omega t ,
\]
(a segunda parte integra a zero), então $ x =
\frac{F}{2\omega}t\sin\omega t $: pergunta 15 novamente, de
Duhamel.

\textbf{23.} No formato do sistema $ X' = AX + (0, F(t))^{\mathsf T}$
com $\operatorname{Sp}A $ de partes reais negativas ($\zeta >
0$): variação de constantes e \cref{exo:b2:diffeq:8}
($\vertiii{\eu^{tA}} \leq C\eu^{-\alpha t}$) dar
\[
\norm{X(t)} \leq C\eu^{-\alpha t}\norm{X_0}
+ \int_0^t C\eu^{-\alpha(t-s)}\norm{F}_\infty\dd s
\leq C\norm{X_0} + \frac{C\norm F_\infty}{\alpha} :
\]
entrada limitada, saída limitada --- \omterm{def:b2:funcseq:def}{uniformemente} no inicial
dados após o transitório.

\textbf{24.} Para $\zeta = 0$ e $\gamma \neq \omega $,
a solução da questão 14 é limitada, e adicionando qualquer homogêneo
solução (limitada: as órbitas do centro são círculos) mantém
limitado; no $\gamma = \omega $, a questão 15 cresce linearmente. Então
para o oscilador não amortecido, limitação sob periodicidade
forçar falha em exatamente uma frequência --- \omterm{pb:b2:diffeq:1}{ressonância} ---
enquanto a pergunta 23 mostra quaisquer restaurações de amortecimento positivas
limite para \emph{todos} entradas limitadas.

\textbf{25.} (i) O plano determinante do traço classifica todos
dinâmica linear planar autônoma e oscilador da Parte II
caminha sobre uma linha vertical dela; forçando sai do avião
(não autônomo), e Duhamel assume. (ii) $\gamma_*$ é
a frequência que o sistema prefere, $ A(\gamma_*)$ o preço de
emocionante, e $\frac{1}{2\zeta}$ o fator de amplificação
--- o \omterm{pb:b2:diffeq:1}{ressonância} engenheiros de nitidez chamam a qualidade
fator. (iii) Os teoremas de Sturm leem oscilação fora do sinal
e tamanho de $ q$ sozinhos: eles governam equações (Bessel,
Schrödinger) cujas soluções não possuem fórmulas elementares.
(iv) Constante wronskianos (pergunta 17, via Liouville) e
estabilidade de entrada limitada (pergunta 23) são reutilizados silenciosamente
sempre que o livro atende equações de coeficiente variável ou
sistemas perturbados.
\end{solution}
