\section*{Capítulo \ref{ch:b2:integration} --- Integração}

\begin{solution}{exo:b2:integration:1}
$\int_0^1 \frac{\dd t}{\sqrt{t(1-t)}}$: aproximar $0$, $\sim t^{-1/2}$
($\alpha = \frac12 < 1$: converge); aproximar $1$, $\sim
(1-t)^{-1/2}$: converge. Convergente (seu valor é $\pi $, pelo
substituição $ t = \sin^2\theta $).

$\int_1^\infty \frac{\ln t}{t^2}$: $\frac{\ln t}{t^2} = o(t^{-3/2})$:
convergente (valor $1$ por partes).

$\int_0^\infty \frac{\dd t}{1 + t^2\sin^2 t}$: divergente. Aproximar $ t =
n\pi $, escrever $ t = n\pi + u $: $\sin^2 t = \sin^2 u \leq u^2$, breve
$\abs u \leq \frac{1}{n}$, $1 + t^2\sin^2 t \leq 1 + (n\pi +
1)^2u^2 \leq C n^2 u^2 + 1$; por isso
\[
\int_{n\pi - 1/n}^{n\pi + 1/n} \frac{\dd t}{1 + t^2\sin^2 t}
\geq \int_{-1/n}^{1/n} \frac{\dd u}{1 + Cn^2u^2}
= \frac{2\arctan\sqrt C}{\sqrt C}\cdot\frac{1}{n} ,
\]
um termo de uma série divergente do tipo harmônico: somando $ n $, o
primitivo é ilimitado.
\end{solution}

\begin{solution}{exo:b2:integration:2}
Substituir $ u = \lambda t $:
\[
\int_0^\infty t^n \eu^{-\lambda t}\dd t
= \frac{1}{\lambda^{n+1}}\int_0^\infty u^n\eu^{-u}\dd u
= \frac{\Gamma(n+1)}{\lambda^{n+1}} = \frac{n!}{\lambda^{n+1}} .
\]
Com $ t = \eu^{-u}$ ($\dd t = -\eu^{-u}\dd u $):
\[
\int_0^1 (\ln t)^n \dd t = \int_0^{\infty} (-u)^n \eu^{-u}\,\dd u
= (-1)^n\, n! .
\]
\end{solution}

\begin{solution}{exo:b2:integration:3}
Convergência: o integrando é $\leq \frac{1}{1+t^2}$ aproximar $\infty $
e delimitado perto $0$ (ambos os fatores limitados abaixo de $0$):
\omterm{def:b2:series:def}{absolutamente} convergente, para cada $ x $. Substituindo $ t = \frac1u $
($\dd t = -\frac{\dd u}{u^2}$):
\[
I(x) = \int_0^\infty \frac{1}{\bigl(1 +
\frac1{u^2}\bigr)\bigl(1 + u^{-x}\bigr)}\cdot\frac{\dd u}{u^2}
= \int_0^\infty \frac{u^x}{(1 + u^2)(1 + u^x)}\,\dd u .
\]
Somando as duas expressões de $ I(x)$:
\[
2I(x) = \int_0^\infty \frac{1 + t^x}{(1+t^2)(1+t^x)}\dd t
= \int_0^\infty \frac{\dd t}{1 + t^2} = \frac{\pi}{2} :
\]
$ I(x) = \frac\pi4$, independente de $ x $.
\end{solution}

\begin{solution}{exo:b2:integration:4}
Aproximar $0^+$, com $ u = \abs{\ln t} \to \infty $. Se $\alpha < 1$:
convergência independentemente de $\beta $ (comparar com $ t^{-\alpha'}$ para
$\alpha < \alpha' < 1$: o fator log é batido). Se $\alpha > 1$:
divergência, independentemente de $\beta $ (comparar com $ t^{-\alpha''}$,
$1 < \alpha'' < \alpha $). Se $\alpha = 1$: substituto $ t =
\eu^{-u}$:
\[
\int_0^{1/2} \frac{\dd t}{t\,\abs{\ln t}^\beta}
= \int_{\ln 2}^{\infty} \frac{\dd u}{u^\beta},
\]
convergente se sef $\beta > 1$. Resumo: convergência sse $\alpha < 1$,
ou ($\alpha = 1$ e $\beta > 1$) --- o espelho do Bertrand
série.
\end{solution}

\begin{solution}{exo:b2:integration:5}
Continuidade sobre $\intco{0}{\infty}$: dominação
$\bigl|\frac{\eu^{-xt}}{1+t^2}\bigr| \leq \frac{1}{1+t^2}$,
integrável, uniforme em $ x \geq 0$:
\cref{thm:b2:integration:continuity}.

$ C^2$ sobre $\intoo{0}{\infty}$: sobre $ x \geq a > 0$, os dois primeiros
$ x $-derivados $\frac{-t\,\eu^{-xt}}{1+t^2}$ e
$\frac{t^2\eu^{-xt}}{1+t^2}$ são dominados por $ t\,\eu^{-at}$ e
$\eu^{-at}$: duas aplicações de
\cref{thm:b2:integration:leibnizrule}. Então
\[
F''(x) + F(x) = \int_0^\infty \frac{t^2 + 1}{1 + t^2}\,\eu^{-xt}\dd
t = \int_0^\infty \eu^{-xt}\dd t = \frac1x .
\]
Limite: $0 \leq F(x) \leq \int_0^\infty \eu^{-xt}\dd t = \frac1x \to
0$.
\end{solution}

\begin{solution}{exo:b2:integration:6}
Sobre $\intcc{\varepsilon}{M}$, substitua $ u = at $ e $ u = bt $ em
as duas metades:
\[
\int_\varepsilon^M \frac{f(at) - f(bt)}{t}\dd t
= \int_{a\varepsilon}^{aM}\frac{f(u)}{u}\dd u
- \int_{b\varepsilon}^{bM}\frac{f(u)}{u}\dd u
= \int_{a\varepsilon}^{b\varepsilon} \frac{f(u)}{u}\dd u
- \int_{aM}^{bM} \frac{f(u)}{u}\dd u .
\]
Primeira peça: $ f(u) = f(0) + o(1)$ aproximar $0$, e $\int_{a\varepsilon}
^{b\varepsilon} \frac{\dd u}{u} = \ln\frac ba $: a peça tende a
$ f(0)\ln\frac ba $. Segunda peça: $ f(u) \to f(\infty)$, mesmo
computação: tende a $ f(\infty)\ln\frac ba $. Daí o impróprio
integrante converge para $\bigl(f(0) - f(\infty)\bigr)\ln\frac ba $.

Com $ f(t) = \eu^{-t}$ ($ f(0) = 1$, $ f(\infty) = 0$), $ a = 1$, $ b
= 2$:
\[
\int_0^\infty \frac{\eu^{-t} - \eu^{-2t}}{t}\dd t = \ln 2 .
\]
\end{solution}

\begin{solution}{exo:b2:integration:7}
Estenda o integrando por $0$ além $ t = n $: $ g_n(t) = (1 -
\frac tn)^n t^{x-1}\mathbf{1}_{t \leq n}$. Pontualmente, $ g_n(t) \to
\eu^{-t}t^{x-1}$ (o limite de juros compostos, volume do Ano 1).
Dominação: $\ln(1 - u) \leq -u $ dá $(1 - \frac tn)^n \leq
\eu^{-t}$ sobre $\intcc{0}{n}$, então $\abs{g_n(t)} \leq
\eu^{-t}t^{x-1} = \varphi(t)$, integrável. Convergência dominada:
\[
\int_0^n \Bigl(1 - \frac tn\Bigr)^n t^{x-1}\dd t
\xrightarrow[n\to\infty]{} \int_0^\infty \eu^{-t}t^{x-1}\dd t
= \Gamma(x) .
\]
(Calcular o lado esquerdo por partes repetidas dá o produto de Euler
formulário $\Gamma(x) = \lim \frac{n!\,n^x}{x(x+1)\cdots(x+n)}$.)
\end{solution}

\begin{solution}{exo:b2:integration:8}
$ H $ é diferenciável em $ x $ (integrando $ C^1$ em $ x $, derivada
$-2x(1+t^2)\cdot\frac{\eu^{-x^2(1+t^2)}}{1+t^2} =
-2x\,\eu^{-x^2}\eu^{-x^2t^2}$, \omterm{def:b2:metric:continuity}{contínuo} e delimitado compactos
de $ x $, dominação sobre $ t \in \intcc{0}{1}$ trivial):
\[
H'(x) = -2x\,\eu^{-x^2}\int_0^1 \eu^{-x^2t^2}\,\dd t
\overset{u = xt}{=} -2\,\eu^{-x^2}\int_0^x \eu^{-u^2}\,\dd u
= -G'(x),
\]
desde $ G'(x) = 2\eu^{-x^2}\int_0^x \eu^{-t^2}\dd t $ (regra da cadeia em
o quadrado, teorema fundamental do cálculo). Então $ G + H $ é
constante, igual a $ G(0) + H(0) = 0 + \int_0^1 \frac{\dd t}{1+t^2}
= \frac\pi4$.

Como $ x \to \infty $: $0 \leq H(x) \leq \eu^{-x^2}\int_0^1 \dd t \to
0$, então $ G(x) \to \frac\pi4$:
\[
\int_0^\infty \eu^{-t^2}\dd t = \sqrt{\frac\pi4} =
\frac{\sqrt\pi}{2} .
\]
(Consequentemente $\Gamma\bigl(\frac12\bigr) = 2\int_0^\infty
\eu^{-t^2}\dd t = \sqrt\pi $, pela substituição $ t = \sqrt u $.)
\end{solution}

\begin{solution}{exo:b2:integration:9}
Cauchy--Schwarz (volume do ano 1, válido em $\intcc{\varepsilon}{M}$
e passado ao limite) aplicado à fatoração
$ t^{\frac{x+y}{2}-1}\eu^{-t} = \bigl(t^{x-1}\eu^{-t}\bigr)^{1/2}
\bigl(t^{y-1}\eu^{-t}\bigr)^{1/2}$:
\[
\Gamma\Bigl(\frac{x+y}{2}\Bigr) \leq
\Gamma(x)^{1/2}\,\Gamma(y)^{1/2}
\quad\Longrightarrow\quad
\ln\Gamma\Bigl(\frac{x+y}{2}\Bigr) \leq \frac{\ln\Gamma(x) +
\ln\Gamma(y)}{2} :
\]
$\ln\Gamma $ é ponto médio convexo; ser \omterm{def:b2:metric:continuity}{contínuo}
(\cref{thm:b2:integration:gammaprops}), é convexo
(\cref{exo:b2:realfun:8}). (Pinos de log-convexidade $\Gamma $ para baixo
exclusivamente entre interpolações do fatorial --- o
Teorema de Bohr-Mollerup, uma pérola do Ano 3.)
\end{solution}

\begin{solution}{exo:b2:integration:10}
\begin{enumerate}
  \item Sobre $ x \geq a > 0$: o $ x $-derivada do integrando é
        $-\sin t\,\eu^{-xt}$, dominado por $\eu^{-at}$:
        \cref{thm:b2:integration:leibnizrule} dá $ F'(x) =
        -\int_0^\infty \eu^{-xt}\sin t\,\dd t $. Duas integrações por
        partes (ou o exponencial complexo):
        \[
        \int_0^\infty \eu^{-xt}\sin t\,\dd t
        = \Im \int_0^\infty \eu^{(-x+\iu)t}\dd t
        = \Im\frac{1}{x - \iu} = \frac{1}{1 + x^2} .
        \]
  \item $\abs{F(x)} \leq \int_0^\infty \eu^{-xt}\dd t = \frac1x \to
        0$. Integrando $ F' = -\frac{1}{1+x^2}$ de $ x $ para
        $\infty $: $0 - F(x) = -\bigl(\frac\pi2 - \arctan x\bigr)$,
        então $ F(x) = \frac\pi2 - \arctan x $.
  \item Deixando $ x \to 0^+$ com o admitido continuidade: $ F(0^+) =
        \frac\pi2$, e $ F(0) = \int_0^\infty \frac{\sin t}{t}\dd
        t $ (a integral semiconvergente de Dirichlet,
        \cref{ex:b2:integration:sint}): seu valor é
        $\frac\pi2$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b2:integration:11}
Convergência: perto $0$ o integrando estende continuamente pelo
valor $1$ ($\sin t \sim t $); no infinito é $\leq t^{-2}$:
convergência absoluta. Sobre $\intcc{\varepsilon}{M}$, integrar por
peças com $ u = \sin^2 t $, $ v' = t^{-2}$:
\[
\int_\varepsilon^M \frac{\sin^2 t}{t^2}\dd t
= \Bigl[-\frac{\sin^2 t}{t}\Bigr]_\varepsilon^M
+ \int_\varepsilon^M \frac{2\sin t\cos t}{t}\dd t
= \Bigl[-\frac{\sin^2 t}{t}\Bigr]_\varepsilon^M
+ \int_{2\varepsilon}^{2M} \frac{\sin u}{u}\dd u
\]
($ u = 2t $ na última integral). O colchete tende a $0$ em ambos
termina ($\sin^2\varepsilon/\varepsilon \leq \varepsilon $;
$\sin^2 M/M \leq 1/M $), e a última integral tende a
$\int_0^\infty \frac{\sin u}{u}\dd u = \frac\pi2$
(\cref{exo:b2:integration:10}). Por isso
\[
\int_0^\infty \Bigl(\frac{\sin t}{t}\Bigr)^{\!2}\dd t
= \frac\pi2 .
\]
\end{solution}

\begin{solution}{exo:b2:integration:12}
\begin{enumerate}
  \item Peças com $ u = \frac{-1}{2t}$, $ v' = -2t\,\eu^{-t^2}$
        (então $ v = \eu^{-t^2}$):
        \[
        T(x) = \Bigl[\frac{-\eu^{-t^2}}{2t}\Bigr]_x^\infty
        - \int_x^\infty \frac{\eu^{-t^2}}{2t^2}\dd t
        = \frac{\eu^{-x^2}}{2x}
        - \int_x^\infty \frac{\eu^{-t^2}}{2t^2}\dd t .
        \]
        Mesmo dispositivo na nova integral ($ u = \frac{-1}{4t^3}$,
        $ v' = -2t\,\eu^{-t^2}$):
        \[
        \int_x^\infty \frac{\eu^{-t^2}}{2t^2}\dd t
        = \frac{\eu^{-x^2}}{4x^3}
        - \frac34\int_x^\infty \frac{\eu^{-t^2}}{t^4}\dd t ,
        \]
        daí a identidade anunciada.
  \item Mais uma integração por partes limita o restante:
        \[
        \int_x^\infty \frac{\eu^{-t^2}}{t^4}\dd t
        = \frac{\eu^{-x^2}}{2x^5}
        - \frac52\int_x^\infty\frac{\eu^{-t^2}}{t^6}\dd t
        \leq \frac{\eu^{-x^2}}{2x^5},
        \]
        então $0 \leq \frac34\int_x^\infty t^{-4}\eu^{-t^2}\dd t
        \leq \frac{3}{8x^5}\eu^{-x^2}$. Abandonando o (positivo)
        resto na identidade da questão 1 dá o menor
        vinculado; eliminando o segundo termo (negativo) do primeiro
        peças dá $ T(x) \leq \frac{\eu^{-x^2}}{2x}$. Dividindo
        o escalonamento por $\frac{\eu^{-x^2}}{2x}$: a proporção é
        espremido entre $1 - \frac{1}{2x^2}$ e $1$, então $ T(x)
        \sim \frac{\eu^{-x^2}}{2x}$.
  \item Iterar as partes produz a série formal
        \[
        T(x) \approx \frac{\eu^{-x^2}}{2x}\Bigl(1 -
        \frac{1}{2x^2} + \frac{1\cdot3}{(2x^2)^2} -
        \frac{1\cdot3\cdot5}{(2x^2)^3} + \cdots\Bigr),
        \]
        cujo $ k $-ésimo coeficiente $1\cdot3\cdots(2k-1) =
        \frac{(2k)!}{2^k k!}$ cresce mais rápido do que qualquer geometria
        sequência: para fixo $ x $ os termos
        $\frac{1\cdot3\cdots(2k-1)}{(2x^2)^k}$ tende ao infinito
        (sua proporção é $\frac{2k+1}{2x^2} \to \infty $), então o
        série diverge para cada $ x $. É um
        \emph{assintótico} expansão: truncado em qualquer fixo
        ordem, o erro é da ordem do primeiro omitido
        termo como $ x \to \infty $ --- mas nunca convergente
        série. (Esta estimativa de cauda é o Gaussiano padrão
        limite final dos capítulos de probabilidade.)
\end{enumerate}
\end{solution}

\begin{solution}{pb:b2:integration:1}
\textbf{1.} No $0^+$ o integrando é $\sim t^{x-1}$: o
escala de ponto final finito converge se $1 - x < 1$, ou seja\ $ x > 0$
(e para $ x \leq 0$, $ t^{x-1} \geq t^{-1}$ diverge); no
$+\infty $, $ t^{x-1}\eu^{-t} = o(t^{-2})$ converge para cada
$ x $. Então $\Gamma(x) = \frac{\Gamma(x+1)}{x}$ e $\Gamma(x+1)
\to \Gamma(1) = 1$ como $ x \to 0^+$ (continuidade,
\cref{thm:b2:integration:gammaprops}): $\Gamma(x) \sim
\frac1x $.

\textbf{2.} Com $ t = u^2$, $\dd t = 2u\,\dd u $:
\[
\Gamma\Bigl(\frac12\Bigr) = \int_0^\infty
t^{-1/2}\eu^{-t}\dd t
= \int_0^\infty \frac{\eu^{-u^2}}{u}\,2u\,\dd u
= 2\int_0^\infty \eu^{-u^2}\dd u = \sqrt\pi
\]
por \cref{exo:b2:integration:8}. Com $ u = v/\sqrt2$:
\[
\int_\R \eu^{-v^2/2}\dd v
= 2\sqrt2\int_0^\infty \eu^{-u^2}\dd u
= \sqrt2\,\sqrt\pi = \sqrt{2\pi} .
\]

\textbf{3.} Verdadeiro para $ n = 0$ (ambos os lados $\sqrt\pi $). Se
$\Gamma(n + \frac12) = \frac{(2n)!}{4^n n!}\sqrt\pi $, o
equação funcional dá
\[
\Gamma\Bigl(n + 1 + \frac12\Bigr)
= \Bigl(n + \frac12\Bigr)\Gamma\Bigl(n + \frac12\Bigr)
= \frac{2n+1}{2}\cdot\frac{(2n)!}{4^n n!}\sqrt\pi
= \frac{(2n+2)!}{4^{n+1}(n+1)!}\sqrt\pi ,
\]
o último passo porque $\frac{(2n+2)!}{(2n)!} = (2n+2)(2n+1)$
e $\frac{2n+1}{2} = \frac{(2n+2)(2n+1)}{4(n+1)}$.

\textbf{4.} \cref{thm:b2:integration:gammaprops} dá
$\Gamma''(x) = \int_0^\infty t^{x-1}\eu^{-t}(\ln t)^2\dd t $
(duas aplicações da regra de Leibniz, dominações como no
prova do teorema); o integrando é $\geq 0$ e não de forma idêntica
zero, então $\Gamma'' > 0$: $\Gamma $ é estritamente convexo e
$\Gamma'$ está aumentando estritamente. Desde $\Gamma(1) = \Gamma(2)
= 1$, Rolle fornece $ x_0 \in \intoo12$ com $\Gamma'(x_0) =
0$; estrita monotonicidade de $\Gamma'$ faz $ x_0$ é único
zero, com $\Gamma' < 0$ antes e $\Gamma' > 0$ depois:
$\Gamma $ diminui em $\intoo0{x_0}$, aumenta em
$\intoo{x_0}\infty $, e $ x_0$ é o mínimo único.

\textbf{5.} Deixar $ k \in \N $ e $ x \geq 3$; escolha o número inteiro
$ n $ com $ n + 1 \leq x < n + 2$ (então $ n \geq 1$). Pelo
monotonicidade da questão 4 (válida a partir de $ x_0 < 2$ sobre): $\Gamma(x)
\geq \Gamma(n + 1) = n!$, enquanto $ x^k \leq (n+2)^k $. Desde
$\frac{n!}{(n+2)^k} \to \infty $ (fatoriais vencem potências, Ano 1
volume), $\frac{\Gamma(x)}{x^k} \geq \frac{n!}{(n+2)^k} \to
\infty $ como $ x \to \infty $: $ x^k = o(\Gamma(x))$.

\textbf{6.} Aproximar $0$ o integrando é $\sim t^{x-1}$
(convergente se sef $ x > 0$), aproximar $1$ isso é $\sim (1-t)^{y-1}$
(se $ y > 0$); ambas as comparações são entre funções positivas,
então $ B(x,y)$ converge exatamente para $ x, y > 0$. A substituição
$ t \mapsto 1 - t $ troca os dois fatores: $ B(x,y) = B(y,x)$.

\textbf{7.} $ B(x,1) = \int_0^1 t^{x-1}\dd t = \frac1x $. Peças
ligado $\intcc\varepsilon{1-\varepsilon}$ com $ u = (1-t)^y $, $ v =
\frac{t^x}{x}$:
\[
\int t^{x-1}(1-t)^{y}\dd t
= \Bigl[\frac{t^x(1-t)^y}{x}\Bigr]
+ \frac{y}{x}\int t^{x}(1-t)^{y-1}\dd t ;
\]
o colchete desaparece em ambas as extremidades como $\varepsilon \to 0$ ($ x >
0$ no $0$, $ y > 0$ no $1$), saindo $ B(x, y+1) = \frac
yx\,B(x+1, y)$.

\textbf{8.} Desde $ t + (1-t) = 1$:
\[
t^{x-1}(1-t)^{y-1} = t^{x}(1-t)^{y-1} + t^{x-1}(1-t)^{y},
\]
então $ B(x,y) = B(x+1,y) + B(x,y+1)$. A pergunta 7 diz $ B(x+1,y) =
\frac xy B(x,y+1)$; substituindo,
\[
B(x,y) = \Bigl(\frac xy + 1\Bigr)B(x,y+1)
= \frac{x+y}{y}\,B(x,y+1),
\]
ou seja\ $ B(x,y+1) = \frac{y}{x+y}B(x,y)$; a relação gêmea
segue pela simetria da questão 6.

\textbf{9.} Indução ativada $ n $ em fixo $ m $: $ B(m,1) = \frac1m =
\frac{(m-1)!\,0!}{m!}$, e se a fórmula for válida em $ n $,
\[
B(m, n+1) = \frac{n}{m+n}\,B(m,n)
= \frac{n}{m+n}\cdot\frac{(m-1)!(n-1)!}{(m+n-1)!}
= \frac{(m-1)!\,n!}{(m+n)!} .
\]
Reescrevendo: $ B(m,n) = \frac{(m-1)!(n-1)!}{(m+n-1)!} =
\bigl[(m+n-1)\binom{m+n-2}{m-1}\bigr]^{-1}$.

\textbf{10.} Ambos os lados $ B(x,n) =
\frac{\Gamma(x)\Gamma(n)}{\Gamma(x+n)}$ igual $\frac1x $ no $ n =
1$ ($\Gamma(1) = 1$, $\Gamma(x+1) = x\Gamma(x)$). Se eles concordarem
em $ n $, então pela relação de descendência e pelo funcional
equação:
\[
B(x, n+1) = \frac{n}{x+n}\,B(x,n), \qquad
\frac{\Gamma(x)\Gamma(n+1)}{\Gamma(x+n+1)}
= \frac{n}{x+n}\cdot
\frac{\Gamma(x)\Gamma(n)}{\Gamma(x+n)} :
\]
as duas sequências obedecem à mesma recursão da mesma semente,
portanto, concordo para todos $ n \in \N^*$ e tudo $ x > 0$.

\textbf{11.} Com $ t = \sin^2\theta $ ($\theta \in
\intoo0{\pi/2}$, $\dd t = 2\sin\theta\cos\theta\,\dd\theta $),
$ t^{x-1} = \sin^{2x-2}\theta $ e $(1-t)^{y-1} =
\cos^{2y-2}\theta $:
\[
B(x,y) = \int_0^{\pi/2}\sin^{2x-2}\theta\,\cos^{2y-2}\theta
\cdot 2\sin\theta\cos\theta\,\dd\theta
= 2\int_0^{\pi/2}\sin^{2x-1}\theta\,\cos^{2y-1}\theta\,
\dd\theta .
\]

\textbf{12.} Pegar $ y = \frac12$ (eliminando o fator cosseno) e
$2x - 1 = n $: $ B\bigl(\frac{n+1}2, \frac12\bigr) = 2W_n $, ou seja\
$ W_n = \frac12 B\bigl(\frac{n+1}2,\frac12\bigr)$. A descida
relação na primeira variável dá
\[
\frac{W_n}{W_{n-2}}
= \frac{B\bigl(\frac{n-1}2 + 1, \frac12\bigr)}
{B\bigl(\frac{n-1}2, \frac12\bigr)}
= \frac{\frac{n-1}2}{\frac{n-1}2 + \frac12}
= \frac{n-1}{n} :
\]
a recorrência de Wallis, desta vez sem integração por partes
em senos --- A Parte II fez o trabalho de uma vez por todas.

\textbf{13.} $ B\bigl(\frac12,\frac12\bigr) = 2W_0 =
2\cdot\frac\pi2 = \pi $, enquanto
$\Gamma\bigl(\frac12\bigr)^2/\Gamma(1) = (\sqrt\pi)^2 = \pi $:
A fórmula de Euler é válida em $\bigl(\frac12,\frac12\bigr)$.

\textbf{14.} Iterando $ W_{2n} = \frac{2n-1}{2n}W_{2n-2}$ de
$ W_0 = \frac\pi2$:
\[
W_{2n} = \frac\pi2\prod_{k=1}^{n}\frac{2k-1}{2k}
= \frac\pi2\cdot\frac{(2n)!}{4^n(n!)^2},
\]
desde $\prod(2k-1) = \frac{(2n)!}{2^n n!}$ e $\prod 2k = 2^n
n!$. Portanto, usando a questão 3:
\[
B\Bigl(n+\frac12, \frac12\Bigr) = 2W_{2n}
= \pi\,\frac{(2n)!}{4^n(n!)^2}
= \frac{(2n)!\sqrt\pi}{4^n n!}\cdot\frac{\sqrt\pi}{n!}
= \frac{\Gamma\bigl(n+\frac12\bigr)\Gamma\bigl(\frac12\bigr)}
{\Gamma(n+1)} .
\]
Agora conserte $ x \in \frac12\N^*$. A fórmula de Euler é válida em $(x,
\frac12)$: para $ x $ número inteiro esta é a questão 10 (com simetria),
para $ x = n + \frac12$ é a tela acima. Ambos os lados
A fórmula de Euler obedece à recursão descendente $ y \mapsto y + 1$
(questão 8 à esquerda, a equação funcional à direita,
como na questão 10): a indução propaga a fórmula de $ y =
\frac12$ e $ y = 1$ para cada $ y \in \frac12\N^*$. Euler
a fórmula, portanto, é válida sempre que $2x, 2y \in \N^*$.

\textbf{15.} Com $ u = \frac{t}{1-t}$, ou seja\ $ t =
\frac{u}{1+u}$, $1 - t = \frac{1}{1+u}$, $\dd t =
\frac{\dd u}{(1+u)^2}$:
\[
B(x,y) = \int_0^\infty
\Bigl(\frac{u}{1+u}\Bigr)^{x-1}
\Bigl(\frac{1}{1+u}\Bigr)^{y-1}
\frac{\dd u}{(1+u)^2}
= \int_0^\infty \frac{u^{x-1}}{(1+u)^{x+y}}\,\dd u .
\]
No $ x = y = \frac12$, com $ u = v^2$:
\[
\int_0^\infty \frac{u^{-1/2}}{1+u}\dd u
= \int_0^\infty \frac{2\,\dd v}{1+v^2} = \pi
= B\Bigl(\frac12,\frac12\Bigr) . \checkmark
\]

\textbf{16.} Uma integração por partes, por $1 \leq k \leq n $
e $ s > 0$ ($ u = (1 - t/n)^k $, $ v = t^s/s $; os termos limite
desaparecer):
\[
\int_0^n \Bigl(1-\frac tn\Bigr)^{\!k} t^{s-1}\dd t
= \frac{k}{ns}\int_0^n \Bigl(1-\frac tn\Bigr)^{\!k-1}
t^{s}\dd t .
\]
A partir de $ k = n $, $ s = x $ e iterando $ n $ vezes:
\[
\int_0^n \Bigl(1-\frac tn\Bigr)^{\!n} t^{x-1}\dd t
= \frac{n(n-1)\cdots1}{n^n\,x(x+1)\cdots(x+n-1)}
\int_0^n t^{x+n-1}\dd t
= \frac{n!}{n^n}\cdot
\frac{n^{x+n}}{x(x+1)\cdots(x+n)} ,
\]
o que é $\dfrac{n!\,n^x}{x(x+1)\cdots(x+n)}$.

\textbf{17.} Por \cref{exo:b2:integration:7} o lado esquerdo tende
para $\Gamma(x)$ (convergência dominada com dominador
$ t^{x-1}\eu^{-t}$); o lado direito é o quociente de Gauss:
\[
\Gamma(x) = \lim_{n\to\infty}
\frac{n!\,n^x}{x(x+1)\cdots(x+n)} .
\]

\textbf{18.} Tomando logaritmos no quociente da questão 16
$ G_n(x)$ e divisão $\ln(x+k) = \ln k + \ln(1 + x/k)$ para $ k
\geq 1$:
\[
\ln G_n(x) = x\ln n - \ln x - \sum_{k=1}^n
\ln\Bigl(1+\frac xk\Bigr)
= -\ln x + x(\ln n - H_n)
+ \sum_{k=1}^n\Bigl(\frac xk -
\ln\Bigl(1+\frac xk\Bigr)\Bigr).
\]
Para $ u \geq 0$, $ u - \frac{u^2}2 \leq \ln(1+u) \leq u $, então o
termo geral reside em $\intcc{0}{x^2/(2k^2)}$: a série
converge (comparação com $\sum k^{-2}$). Desde $\ln n - H_n
\to -\gamma $ (\cref{ex:b2:comparison:harmonic}) e $\ln G_n(x)
\to \ln\Gamma(x)$ (questão 17 e continuidade de $\ln $):
\[
\ln\Gamma(x) = -\ln x - \gamma x
+ \sum_{k=1}^\infty\Bigl(\frac xk -
\ln\Bigl(1+\frac xk\Bigr)\Bigr) .
\]

\textbf{19.} No $ x = \frac12$, o denominador é
$\prod_{k=0}^n\bigl(k+\frac12\bigr) =
\frac{(2n+1)!}{2^{2n+1}n!}$ (multiplique as metades), então
\[
G_n\Bigl(\frac12\Bigr)
= \frac{n!\,\sqrt n\;2^{2n+1}n!}{(2n+1)!}
= \frac{2\sqrt n\;4^n}{(2n+1)\binom{2n}{n}} .
\]
Com $\binom{2n}{n} \sim \frac{4^n}{\sqrt{\pi n}}$
(\cref{ex:b2:comparison:centralbinomial}):
\[
G_n\Bigl(\frac12\Bigr)
\sim \frac{2\sqrt n\,\sqrt{\pi n}}{2n+1}
\longrightarrow \sqrt\pi = \Gamma\Bigl(\frac12\Bigr) .
\]
O coeficiente binomial central $\sqrt\pi $ (que veio de
Wallis, portanto da constante de Stirling) e a Gaussiana
integrais $\sqrt\pi $ são o mesmo número.

\textbf{20.} A identidade é álgebra direta: multiplique
$\frac{n!\,n^x}{x(x+1)\cdots(x+n)}$ por $\frac{nx}{x+n+1}$ e
absorver $ x $ no produto, $ n $ em $ n^x $. Deixando $ n \to
\infty $: o lado esquerdo tende a $\Gamma(x+1)$ (Gauss em $ x+1$),
o lado direito para $\Gamma(x)\cdot x\cdot 1$ desde
$\frac{n}{x+n+1} \to 1$: $\Gamma(x+1) = x\Gamma(x)$ ---
recuperado sem uma única integração por partes.

\textbf{21.} Com $ u = t^a $, $ t = u^{1/a}$, $\dd t = \frac1a
u^{1/a - 1}\dd u $:
\[
\int_0^\infty \eu^{-t^a}\dd t
= \frac1a\int_0^\infty u^{\frac1a - 1}\eu^{-u}\dd u
= \frac1a\,\Gamma\Bigl(\frac1a\Bigr)
= \Gamma\Bigl(1 + \frac1a\Bigr) .
\]
Como $ a \to +\infty $ (ao longo de qualquer sequência): $\eu^{-t^a} \to 1$ para
$0 < t < 1$, $\to \eu^{-1}$ no $ t = 1$, $\to 0$ para $ t > 1$;
para $ a \geq 2$ dominar por $\mathbf 1_{t \leq 1} +
\eu^{-t^2}\mathbf 1_{t > 1}$ ($ t^a \geq t^2$ para $ t \geq 1$),
integrável. Convergência dominada: a integral tende a
$\int_0^1 1\,\dd t = 1$ --- como deveria, já que $\Gamma(1 +
\frac1a) \to \Gamma(1) = 1$ por continuidade.

\textbf{22.} Com $ u = t^n $, $\dd t = \frac1n u^{1/n - 1}\dd
u $:
\[
\int_0^1 \frac{\dd t}{\sqrt{1-t^n}}
= \frac1n\int_0^1 u^{\frac1n-1}(1-u)^{-1/2}\dd u
= \frac1n\,B\Bigl(\frac1n, \frac12\Bigr) .
\]
$ n = 1$: $ B\bigl(1,\frac12\bigr) = B\bigl(\frac12,1\bigr) = 2$,
correspondência $\int_0^1\frac{\dd t}{\sqrt{1-t}} = 2$. $ n = 2$:
$\frac12 B\bigl(\frac12,\frac12\bigr) = \frac\pi2 = \arcsin 1$.
Para $ n = 4$ o valor $\frac14 B\bigl(\frac14,\frac12\bigr)$ é
a constante de lemniscata: nenhuma forma fechada elementar.

\textbf{23.} Iterando a equação funcional:
\[
\frac{1}{\Gamma(x)}\int_0^\infty t^{x+k-1}\eu^{-t}\dd t
= \frac{\Gamma(x+k)}{\Gamma(x)}
= (x+k-1)(x+k-2)\cdots x ,
\]
o fatorial crescente com $ k $ fatores. No $ x = 1$:
$\Gamma(1+k)/\Gamma(1) = k!$, os momentos de $\eu^{-t}$ de
\cref{exo:b2:integration:2}.

\textbf{24.} Substituir $ t = \frac{1+s}2$ ($ s \in
\intoo{-1}1$, $\dd t = \frac{\dd s}2$, $ t(1-t) =
\frac{1-s^2}4$):
\[
B(x,x) = \int_{-1}^{1}\Bigl(\frac{1-s^2}{4}\Bigr)^{x-1}
\frac{\dd s}{2}
= 4^{1-x}\int_0^1 (1-s^2)^{x-1}\dd s
\]
(o integrando é par). Então $ s = \sqrt v $ ($\dd s =
\frac{\dd v}{2\sqrt v}$):
\[
B(x,x) = \frac{4^{1-x}}{2}\int_0^1
v^{-1/2}(1-v)^{x-1}\dd v
= 2^{1-2x}\,B\Bigl(\frac12, x\Bigr) .
\]
Para $2x \in \N^*$ cada argumento à vista reside em
$\frac12\N^*$, então a fórmula de Euler (questão 14) se aplica a ambos
lados:
\[
\frac{\Gamma(x)^2}{\Gamma(2x)}
= 2^{1-2x}\,
\frac{\Gamma\bigl(\frac12\bigr)\Gamma(x)}
{\Gamma\bigl(x+\frac12\bigr)}
\quad\Longleftrightarrow\quad
\Gamma(x)\,\Gamma\Bigl(x+\frac12\Bigr)
= 2^{1-2x}\sqrt\pi\;\Gamma(2x) .
\]
Verificação direta em $ x = n $: o lado esquerdo é $(n-1)!\cdot
\frac{(2n)!\sqrt\pi}{4^n n!} = \frac{(2n)!\sqrt\pi}{4^n n}$,
o lado direito $2\cdot4^{-n}\sqrt\pi\,(2n-1)! =
\frac{(2n)!\sqrt\pi}{4^n n}$: igual.

\textbf{25.} (i) Integração por peças produzidas $ B(x,y+1) =
\frac yx B(x+1,y)$, a identidade única a partir da qual cada
relação de descendência, os valores inteiros e meio inteiros e o
Recorrência de Wallis em todo o fluxo. (ii) Convergência dominada transformada
as integrais elementares $\int_0^n(1-t/n)^n t^{x-1}$ em
$\Gamma(x)$ (Fórmula de Gauss, questão 17) e calculou o
limite $ a \to \infty $ na questão 21. (iii) Da comparação
capítulo importamos a constante de Euler ($\ln n - H_n \to
-\gamma $, questão 18) e o binômio central assintótico
(questão 19) --- ou seja, a fórmula de Stirling disfarçada. (iv)
Fórmula de Euler $ B(x,y) = \Gamma(x)\Gamma(y)/\Gamma(x+y)$ é
agora comprovado para $ y \in \N^*$ com $ x > 0$ arbitrário (pergunta
10) e para todos os pares meio inteiros (questão 14); o geral
caso $ x, y > 0$ aguarda o cálculo da integral dupla do
capítulo sobre integrais múltiplas, que fatora
$\Gamma(x)\Gamma(y)$ sobre um quarto de plano.
\end{solution}
