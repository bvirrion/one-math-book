\section*{Capítulo \ref{ch:b2:multint} --- Integrais de Linha e Integrais Múltiplos}

\begin{solution}{exo:b2:multint:1}
(a) Segmento $\gamma(t) = (t, t)$, $ t \in [0,1]$:
\[
\int_\gamma y^2\dd x + x\dd y
= \int_0^1 (t^2 + t)\,\dd t = \frac13 + \frac12 = \frac56 .
\]
(b) Parábola $\gamma(t) = (t, t^2)$:
\[
\int_0^1 \bigl(t^4\cdot 1 + t\cdot 2t\bigr)\dd t
= \frac15 + \frac23 = \frac{13}{15} .
\]
Os dois valores diferem, então a integral depende do caminho: a forma
é \emph{não} exato --- consistentemente, $ P_y = 2y \neq 1 = Q_x $, então
nem está fechado.
\end{solution}

\begin{solution}{exo:b2:multint:2}
$ P = 2xy + y^3$, $ Q = x^2 + 3xy^2 + 1$: $ P_y = 2x + 3y^2 = Q_x $,
fechado em $\R^2$. Procurar $ f $ com $ f_x = P $: $ f = x^2y + xy^3
+ g(y)$; então $ f_y = x^2 + 3xy^2 + g'(y) = Q $ forças $ g'(y) = 1$,
dizer $ g(y) = y $. Então
\[
f(x, y) = x^2y + xy^3 + y
\]
é um \omterm{def:b2:multint:exact}{potencial} ($\R^2$ tem formato de estrela, então \omterm{def:b2:multint:exact}{potencial} teve que
existe pelo lema de Poincaré --- mas exibi-lo é mais rápido). Por
\cref{thm:b2:multint:ftc}, para qualquer arco de $(0,0)$ para $(1,2)$:
\[
\int_\gamma\omega = f(1, 2) - f(0, 0) = 2 + 8 + 2 = 12 .
\]
\end{solution}

\begin{solution}{exo:b2:multint:3}
O domínio é $\{0 \leq x \leq 1,\ x^2 \leq y \leq x\}$.
$ y $-primeiro:
\[
\int_0^1\!\int_{x^2}^{x}(x + y)\,\dd y\,\dd x
= \int_0^1\Bigl(x(x - x^2)
+ \frac{x^2 - x^4}{2}\Bigr)\dd x
= \int_0^1\Bigl(\frac{3x^2}{2} - x^3 - \frac{x^4}{2}\Bigr)\dd x
= \frac12 - \frac14 - \frac1{10} = \frac{3}{20} .
\]
$ x $-primeiro: a fatia na altura $ y \in [0, 1]$ é $ y \leq x \leq
\sqrt y $, então
\[
\int_0^1\!\int_{y}^{\sqrt y}(x + y)\,\dd x\,\dd y
= \int_0^1\Bigl(\frac{y - y^2}{2}
+ y(\sqrt y - y)\Bigr)\dd y
= \frac14 - \frac16 + \frac25 - \frac13 = \frac{3}{20} .
\]
\end{solution}

\begin{solution}{exo:b2:multint:4}
\emph{Primeira integral.} No disco $ D_R $, em coordenadas \omterm{pb:b2:quadratic:1}{polares}:
\[
\iint_{D_R}\frac{\dd x\,\dd y}{(1 + x^2 + y^2)^2}
= \int_0^{2\pi}\!\!\int_0^R
\frac{\rho\,\dd\rho\,\dd\alpha}{(1 + \rho^2)^2}
= 2\pi\Bigl[-\frac{1}{2(1 + \rho^2)}\Bigr]_0^R
= \pi\Bigl(1 - \frac{1}{1 + R^2}\Bigr)
\xrightarrow[R \to \infty]{} \pi .
\]
\emph{Segunda integral.} O quarto do disco é $0 \leq \alpha \leq
\frac\pi2$, $0 \leq \rho \leq 1$, e $ xy =
\rho^2\cos\alpha\sin\alpha $:
\[
\iint_{D'}xy\,\dd x\,\dd y
= \int_0^{\pi/2}\!\!\cos\alpha\sin\alpha\,\dd\alpha
\int_0^1 \rho^3\,\dd\rho
= \frac12\cdot\frac14 = \frac18 .
\]
\end{solution}

\begin{solution}{exo:b2:multint:5}
Por \cref{cor:b2:multint:area} com $ x = \cos^3 t $, $ y = \sin^3 t $:
$ x' = -3\cos^2 t\sin t $, $ y' = 3\sin^2 t\cos t $, então
\[
xy' - yx' = 3\cos^4 t\sin^2 t + 3\sin^4 t\cos^2 t
= 3\sin^2 t\cos^2 t = \frac{3}{4}\sin^2 2t
= \frac{3}{8}(1 - \cos 4t) .
\]
Por isso
\[
A = \frac12\int_0^{2\pi}\frac38(1 - \cos 4t)\,\dd t
= \frac{3}{16}\cdot 2\pi = \frac{3\pi}{8} .
\]
(O \omterm{pb:b2:curves:1}{astroide} cabe no disco unitário de \omterm{def:b2:multint:domain}{área} $\pi $; três oitavos de
$\pi $ é plausível por sua forma de estrela de quatro cúspides.)
\end{solution}

\begin{solution}{exo:b2:multint:6}
\emph{Empilhamento:} acima de cada $(x, y)$ do disco unitário $ D $, $ z $
corre de $ x^2 + y^2$ para $1$:
\[
V = \iint_D \bigl(1 - x^2 - y^2\bigr)\dd x\,\dd y
= \int_0^{2\pi}\!\!\int_0^1 (1 - \rho^2)\rho\,\dd\rho\,\dd\alpha
= 2\pi\Bigl(\frac12 - \frac14\Bigr) = \frac\pi2 .
\]
\emph{Fatiamento:} a fatia na altura $ z \in [0, 1]$ é o disco
$ x^2 + y^2 \leq z $, de \omterm{def:b2:multint:domain}{área} $\pi z $:
\[
V = \int_0^1 \pi z\,\dd z = \frac\pi2 .
\]
\end{solution}

\begin{solution}{exo:b2:multint:7}
Em coordenadas esféricas o elemento de volume é
$ r^2\cos\varphi\,\dd r\,\dd\theta\,\dd\varphi $
(\cref{ex:b2:multint:spherical}), e a parte angular integra
para $4\pi $ ($2\pi $ de $\theta $, $\int_{-\pi/2}^{\pi/2}\cos = 2$).
O volume da casca é
\[
\int_a^b 4\pi r^2\,\dd r = \frac43\pi\bigl(b^3 - a^3\bigr).
\]
Para a segunda integral, o integrando $1/r $ depende apenas de $ r $:
\[
\iiint_B \frac{\dd x\,\dd y\,\dd z}{r}
= \int_0^R 4\pi r^2\cdot\frac1r\,\dd r
= 4\pi\,\frac{R^2}{2} = 2\pi R^2 .
\]
(O integrando explode na origem, mas de forma inofensiva: $ r^2/r = r $
é \omterm{def:b2:metric:continuity}{contínuo} --- a integral sobre as cascas $\varepsilon \leq r
\leq R $ converge como $\varepsilon \to 0$, que é o exato
sentido da afirmação. Este tipo de cálculo é o primeiro
passo em direção ao teorema de Newton de que uma bola homogênea atrai
uma massa pontual em seu centro.)
\end{solution}

\begin{solution}{exo:b2:multint:8}
O integrando $ g(t; x, y) = xP(tx, ty) + yQ(tx, ty)$ é
$\mathcal{C}^1$ em $(x, y)$, \omterm{def:b2:metric:continuity}{contínuo} em $ t $, com parcial
derivados \omterm{def:b2:metric:continuity}{contínuo} sobre $[0,1] \times U $; diferenciação sob
o sinal integral (\cref{ch:b2:integration}, aplicado no
\omterm{def:b2:metric:compact}{compactar} $ t $-intervalo $[0,1]$, a dominação é automática lá)
dá
\[
f_x(x, y)
= \int_0^1 \bigl(P(tx, ty) + tx\,P_x(tx, ty)
+ ty\,Q_x(tx, ty)\bigr)\dd t .
\]
Usando o fechamento $ Q_x = P_y $:
\[
tx\,P_x(tx, ty) + ty\,P_y(tx, ty)
= t\,\frac{\dd}{\dd t}\bigl[P(tx, ty)\bigr] ,
\]
então o integrando é $ P(tx, ty) + t\frac{\dd}{\dd t}P(tx, ty) =
\frac{\dd}{\dd t}\bigl[t\,P(tx, ty)\bigr]$ e
\[
f_x(x, y) = \Bigl[t\,P(tx, ty)\Bigr]_0^1 = P(x, y) .
\]
Simetricamente $ f_y = Q $ (mesmo cálculo com $ P_y = Q_x $ usado
para o outro lado). Observe onde entra a hipótese: $ f $ está definido
integrando ao longo do segmento $[0, M]$, que reside em $ U $
precisamente porque $ U $ tem formato de estrela.
\end{solution}

\begin{solution}{exo:b2:multint:9}
Na faixa $[0, A] \times [0, \infty)$ a função $(x, y)
\mapsto e^{-xy}\sin x $ não é \omterm{def:b2:series:def}{absolutamente} integrável até $ y =
\infty $ \omterm{def:b2:funcseq:def}{uniformemente} em um sentido ingênuo, mas cada integral iterada
converge e sua igualdade segue de Fubini em $[0, A]
\times [0, B]$ mais um limite $ B \to \infty $ (a cauda
$\int_0^A\int_B^\infty e^{-xy}\abs{\sin x}\,\dd y\,\dd x \leq
\int_0^A \frac{e^{-Bx}\abs{\sin x}}{x}\dd x \leq \int_0^A e^{-Bx}
\dd x\to 0$, usando $\abs{\sin x} \leq x $).

\emph{$ y $ primeiro:} $\int_0^\infty e^{-xy}\,\dd y = \frac1x $ para $ x
> 0$, então a primeira integral é $\int_0^A \frac{\sin x}{x}\,\dd x $.

\emph{$ x $ primeiro:} duas integrações por partes (ou tomando o
parte imaginária de $\int_0^A e^{(i - y)x}\dd x $) dar
\[
\int_0^A e^{-xy}\sin x\,\dd x
= \frac{1 - e^{-Ay}(y\sin A + \cos A)}{1 + y^2} .
\]
Integrando em $ y $ sobre $[0, \infty)$, O termo $\int_0^\infty
\frac{\dd y}{1 + y^2} = \frac\pi2$ se separa:
\[
\int_0^A \frac{\sin x}{x}\,\dd x
= \frac{\pi}{2}
- \int_0^\infty e^{-Ay}\,\frac{y\sin A + \cos A}{1 + y^2}\,\dd y .
\]
O restante é limitado por $\int_0^\infty e^{-Ay}\frac{y +
1}{1 + y^2}\dd y \leq \int_0^\infty e^{-Ay}\cdot\frac{1+y}{1+y^2}
\,\dd y \to 0$ como $ A \to \infty $ (convergência dominada, ou
limite bruto $\frac{1 + y}{1 + y^2} \leq \frac32$ dando
$\frac{3}{2A}$). Por isso $\int_0^\infty\frac{\sin x}{x}\dd x =
\frac\pi2$ --- o mesmo valor obtido em
\cref{ch:b2:integration} ao diferenciar um \omterm{thm:b2:integration:continuity}{integral do parâmetro};
aqui Fubini faz o trabalho.
\end{solution}

\begin{solution}{exo:b2:multint:10}
Parametrizar por \omterm{def:b2:curves:length}{comprimento} do arco $ s \in [0, 2\pi]$, então $ x'^2 + y'^2 = 1$,
e traduza para que $\int_0^{2\pi} x(s)\,\dd s = 0$. Por
\cref{cor:b2:multint:area},
\[
2A = \oint x\,\dd y - y\,\dd x
= \int_0^{2\pi}\bigl(xy' - yx'\bigr)\dd s .
\]
Integrando $\oint y\,\dd x $ por partes ao longo do período (limite
termos cancelados por periodicidade), $-\int yx' = \int y'x $, então na verdade
$2A = 2\int_0^{2\pi}xy'\,\dd s $. Então $2xy' \leq x^2 + y'^2$
dá
\[
2A \leq \int_0^{2\pi}\bigl(x^2 + y'^2\bigr)\dd s
= \int_0^{2\pi} x^2 + \int_0^{2\pi}\bigl(1 - x'^2\bigr)
= 2\pi - \int_0^{2\pi}\bigl(x'^2 - x^2\bigr)\dd s .
\]
Desigualdade de Wirtinger (\cref{ch:b2:fourier} exercícios: para um
$2\pi $-periódico $\mathcal{C}^1$ função com média zero,
$\int x^2 \leq \int x'^2$) torna a última integral não negativa:
$ A \leq \pi $. Igualdade requer igualdade em Wirtinger ($ x(s) =
a\cos s + b\sin s $) e em $2xy' \leq x^2 + y'^2$ ($ y' = x $
\omterm{def:b2:funcseq:def}{pontualmente}), que força $ y = a\sin s - b\cos s + c $: a curva é
o círculo unitário (adequadamente centralizado). Como uma curva de \omterm{def:b2:curves:length}{comprimento} $ L $
redimensiona para \omterm{def:b2:curves:length}{comprimento} $2\pi $, a afirmação geral é $ A \leq
\frac{L^2}{4\pi}$: entre todas as curvas fechadas de determinado perímetro, o
círculo envolve o maior \omterm{def:b2:multint:domain}{área}.
\end{solution}

\begin{solution}{exo:b2:multint:11}
Em coordenadas esféricas $ z = r\sin\varphi $ e $\dd x\,\dd
y\,\dd z = r^2\cos\varphi\,\dd r\,\dd\theta\,\dd\varphi $:
\[
\iiint_B z^2
= \int_0^R r^4\,\dd r\int_0^{2\pi}\dd\theta
\int_{-\pi/2}^{\pi/2}\sin^2\varphi\cos\varphi\,\dd\varphi
= \frac{R^5}5\cdot2\pi\cdot
\Bigl[\frac{\sin^3\varphi}3\Bigr]_{-\pi/2}^{\pi/2}
= \frac{4\pi R^5}{15}.
\]
Pela simetria da bola sob permutação de coordenadas,
$\iiint_B x^2 = \iiint_B y^2 = \iiint_B z^2$, então $\iiint_B(x^2
+ y^2 + z^2) = 3\cdot\frac{4\pi R^5}{15} = \frac{4\pi R^5}5$.
Verificação de casca: $\int_0^R r^2\cdot 4\pi r^2\,\dd r = \frac{4\pi
R^5}5$ --- o integrando $ r^2$ é constante na esfera de
raio $ r $, de \omterm{def:b2:multint:domain}{área} $4\pi r^2$.
\end{solution}

\begin{solution}{exo:b2:multint:12}
Parametrize a aresta de $(x_i, y_i)$ para $(x_{i+1}, y_{i+1})$
por $\gamma(t) = \bigl((1-t)x_i + tx_{i+1},\ (1-t)y_i +
ty_{i+1}\bigr)$. Sua contribuição para
$\frac12\oint(x\,\dd y - y\,\dd x)$ é
\[
\frac12\int_0^1\Bigl(\bigl((1-t)x_i +
tx_{i+1}\bigr)(y_{i+1} - y_i) - \bigl((1-t)y_i +
ty_{i+1}\bigr)(x_{i+1} - x_i)\Bigr)\dd t ,
\]
e desde então $\int_0^1\bigl((1-t)u + tv\bigr)\dd t = \frac{u +
v}2$, isso é igual
\[
\frac14\Bigl((x_i + x_{i+1})(y_{i+1} - y_i) - (y_i +
y_{i+1})(x_{i+1} - x_i)\Bigr)
= \frac12\bigl(x_iy_{i+1} - x_{i+1}y_i\bigr),
\]
o cancelamento dos termos cruzados. Resumindo o $ m $ bordas dá
a fórmula do cadarço, por \cref{cor:b2:multint:area}. Triângulo
$(0,0), (1,0), (0,1)$: $\frac12\bigl((0\cdot0 - 1\cdot0) +
(1\cdot1 - 0\cdot0) + (0\cdot0 - 0\cdot1)\bigr) = \frac12$,
o correto \omterm{def:b2:multint:domain}{área}.
\end{solution}

\begin{solution}{pb:b2:multint:1}
\textbf{1.} A bola $ B_n(R)$ é descrito por limites iterados
$-R \leq x_n \leq R $, então $\abs{x_{n-1}} \leq \sqrt{R^2 -
x_n^2}$, e assim por diante; substituindo $ x_i = Ru_i $ em cada um dos
$ n $ integrais de uma variável multiplica cada uma por $ R $ e mapas
os limites daqueles de $ B_n(1)$: $ v_n(R) = R^n\,v_n(1) =
V_nR^n $.

\textbf{2.} Cortando junto $ x_n = t $: a fatia de $ B_n(1)$ é
a bola $ B_{n-1}\bigl(\sqrt{1 - t^2}\bigr)$, então, por pergunta
1,
\[
V_n = \int_{-1}^1 v_{n-1}\bigl(\sqrt{1 - t^2}\bigr)\,\dd t
= V_{n-1}\int_{-1}^{1}(1 - t^2)^{\frac{n-1}2}\,\dd t .
\]

\textbf{3.} Com $ t = \sin\theta $, $\dd t =
\cos\theta\,\dd\theta $ e $(1 - t^2)^{\frac{n-1}2} =
\cos^{n-1}\theta $ sobre $\intcc{-\pi/2}{\pi/2}$:
\[
\int_{-1}^1(1 - t^2)^{\frac{n-1}2}\dd t
= \int_{-\pi/2}^{\pi/2}\cos^n\theta\,\dd\theta
= 2\int_0^{\pi/2}\cos^n\theta\,\dd\theta = 2W_n,
\]
e $\theta \mapsto \frac\pi2 - \theta $ troca o seno
e formas de cosseno de $ W_n $.

\textbf{4.} Escrever $\sin^n = \sin^{n-2}(1 - \cos^2)$ e
integrar $\int\sin^{n-2}\cos\cdot\cos $ por partes ($ v =
\frac{\sin^{n-1}}{n-1}$):
\[
W_n = W_{n-2} - \frac{W_n}{n-1}
\quad\Longrightarrow\quad
W_n = \frac{n-1}{n}W_{n-2}.
\]
Por isso $ nW_nW_{n-1} = (n-1)W_{n-1}W_{n-2}$: a sequência
$(nW_nW_{n-1})$ é constante, igual a $1\cdot W_1W_0 =
1\cdot\frac\pi2$, então $ W_nW_{n-1} = \frac{\pi}{2n}$.

\textbf{5.} As perguntas 2 a 3 dão $ V_n = 2W_nV_{n-1}$, duas vezes:
\[
V_n = 2W_n\cdot 2W_{n-1}\,V_{n-2}
= 4\,\frac{\pi}{2n}\,V_{n-2} = \frac{2\pi}n\,V_{n-2}.
\]

\textbf{6.} De $ V_0 = 1$: $ V_{2k} = \frac{2\pi}{2k}V_{2k-2}
= \frac\pi kV_{2k-2}$, então $ V_{2k} = \frac{\pi^k}{k!}$ por
indução. De $ V_1 = 2$: $ V_{2k+1} =
\frac{2\pi}{2k+1}V_{2k-1}$, então
\[
V_{2k+1} = 2\prod_{j=1}^k\frac{2\pi}{2j+1}
= \frac{2^{k+1}\pi^k}{1\cdot3\cdot5\cdots(2k+1)} .
\]

\textbf{7.} $ V_1 = 2$, $ V_2 = \pi \approx 3.142$, $ V_3 =
\frac{4\pi}3 \approx 4.189$, $ V_4 = \frac{\pi^2}2 \approx
4.935$, $ V_5 = \frac{8\pi^2}{15} \approx 5.264$, $ V_6 =
\frac{\pi^3}6 \approx 5.168$, $ V_7 = \frac{16\pi^3}{105}
\approx 4.725$. A proporção de um passo é $ V_n/V_{n-1} = 2W_n $,
e $(W_n)$ está diminuindo ($\sin^n \leq \sin^{n-1}$
\omterm{def:b2:funcseq:def}{pontualmente}). Agora $2W_5 = 2\cdot\frac45\cdot\frac23 =
\frac{16}{15} > 1$ enquanto $2W_6 =
2\cdot\frac56\cdot\frac34\cdot\frac12\cdot\frac\pi2 =
\frac{5\pi}{16} < 1$: as proporções excedem $1$ até $ n = 5$ e
estão abaixo $1$ de $ n = 6$ sobre --- $(V_n)$ aumenta para o seu
máximo $ V_5$ e depois diminui.

\textbf{8.} Para $ n \geq 13 > 4\pi $: $ V_n/V_{n-2} = 2\pi/n <
\tfrac12$, então $ V_{n} \leq C\cdot 2^{-n/2}$ com um fixo
constante; melhor, para qualquer $ q > 0$, $2\pi/n < q^2$ para grande
$ n $, então $ V_n/q^n \to 0$: a decadência supera todas as geometrias
sequência. Convergência de $\sum V_n $ segue da proporção
$ V_n/V_{n-2} \to 0$ (compare com uma série geométrica de
alguma classificação).

\textbf{9.} $\sum_{k\geq0}V_{2k}x^{2k} =
\sum_{k\geq0}\frac{(\pi x^2)^k}{k!} = \eu^{\pi x^2}$, o
série exponencial (\cref{ch:b2:powerseries}), convergente para
cada $ x $. No $ x = 1$: $\sum_kV_{2k} = \eu^\pi \approx
23.14$.

\textbf{10.} No cubo $\intcc{-R}R^n $ o integrando é o
produto $\prod_i\eu^{-x_i^2}$, então a integral iterada
fatora: $\bigl(\int_{-R}^R\eu^{-t^2}\dd t\bigr)^n $.
Deixando $ R \to \infty $ e usando
$\int_\R\eu^{-t^2}\dd t = \sqrt\pi $
(\cref{ex:b2:multint:polar}): $ I_n = \pi^{n/2}$.

\textbf{11.} $ t = u^2$ dá $\Gamma(\tfrac12) =
\int_0^\infty t^{-1/2}\eu^{-t}\dd t =
2\int_0^\infty\eu^{-u^2}\dd u = \sqrt\pi $. Iterando
$\Gamma(s+1) = s\Gamma(s)$: $\Gamma(k+1) = k!\,\Gamma(1) =
k!$, e
\[
\Gamma\Bigl(k + \frac32\Bigr)
= \Bigl(k + \frac12\Bigr)\Bigl(k - \frac12\Bigr)\cdots
\frac12\cdot\Gamma\Bigl(\frac12\Bigr)
= \frac{(2k+1)(2k-1)\cdots1}{2^{k+1}}\,\sqrt\pi .
\]

\textbf{12.} Definir $ F_n = \pi^{n/2}/\Gamma(\frac n2 + 1)$.
Desde $\Gamma(\frac n2 + 1) = \frac n2\,\Gamma(\frac n2) =
\frac n2\,\Gamma(\frac{n-2}2 + 1)$, obtemos $ F_n =
\frac{2\pi}nF_{n-2}$: a mesma recursão que $ V_n $ (pergunta
5). Bases: $ F_1 = \sqrt\pi/\Gamma(\frac32) =
\sqrt\pi/(\frac{\sqrt\pi}2) = 2 = V_1$ e $ F_2 =
\pi/\Gamma(2) = \pi = V_2$. Por indução $ V_n = F_n $ para todos
$ n $; a questão 11 transforma isso de volta nos dois formulários fechados de
questão 6.

\textbf{13.} Com $ r = \sqrt t $: $\int_0^\infty
\eu^{-r^2}r^{n-1}\dd r = \frac12\int_0^\infty
t^{\frac n2 - 1}\eu^{-t}\dd t = \frac12\Gamma(\frac n2)$.
Por isso
\[
n\,V_n\int_0^\infty\eu^{-r^2}r^{n-1}\dd r
= V_n\cdot\frac n2\,\Gamma\Bigl(\frac n2\Bigr)
= V_n\,\Gamma\Bigl(\frac n2 + 1\Bigr) = \pi^{n/2} = I_n .
\]
Provados ambos os lados, a identidade pode ser \emph{ler} como
a decomposição em casca da integral gaussiana: a esfera
de raio $ r $ carrega \omterm{def:b2:multint:domain}{área} $ nV_nr^{n-1}$, e o gaussiano
peso $\eu^{-r^2}$ está integrado sobre as cascas.

\textbf{14.} $ s_0 = V_1 = 2$ (o $0$-esfera tem dois pontos),
$ s_1 = 2V_2 = 2\pi $, $ s_2 = 3V_3 = 4\pi $, $ s_3 = 4V_4 =
2\pi^2$; e $\int_0^R s_{n-1}r^{n-1}\dd r = V_nR^n =
v_n(R)$: \omterm{def:b2:multint:domain}{área} é a derivada radial do volume.

\textbf{15.} Para $ n = 2k $, Stirling
(\cref{thm:b2:comparison:stirling}) dá $ k! \sim
\sqrt{2\pi k}\,(k/\eu)^k $, então
\[
V_{2k} = \frac{\pi^k}{k!}
\sim \frac{(\pi\eu/k)^k}{\sqrt{2\pi k}}
= \frac1{\sqrt{\pi n}}\Bigl(\frac{2\pi\eu}n\Bigr)^{n/2}
\qquad (n = 2k).
\]
Por estranho $ n $: $ V_{2k+1} = 2W_{2k+1}V_{2k} \leq 2V_{2k}$, então
os mesmos limites de decaimento supergeométrico são mantidos (até um fator
$2$ e uma mudança de um no expoente) --- para cada $ q >
0$, $ V_n = o(q^n)$.

\textbf{16.} $ V_2/4 = \pi/4 \approx 0.785$; $ V_3/8 = \pi/6
\approx 0.524$; $ V_{10}/2^{10} = \frac{\pi^5}{120\cdot1024}
\approx 0.0025$. Em geral $\frac{V_n/2^n}{V_{n-2}/2^{n-2}}
= \frac{2\pi}{4n} = \frac{\pi}{2n} \to 0$: a proporção tende a
$0$ (supergeometricamente). A bola inscrita ocupa um
fração de fuga: o volume do cubo migra para seu
cantos.

\textbf{17.} Pela questão 1, a bola interna de raio $1 -
\varepsilon $ tem volume $ V_n(1-\varepsilon)^n $, então o exterior
shell carrega a fração $1 - (1 - \varepsilon)^n \to 1$.
Para $ n = 100$, $\varepsilon = 0.01$: $(0.99)^{100} =
\eu^{100\ln0.99} \approx \eu^{-1.005} \approx 0.366$: sobre
$63\%$ da bola está dentro $1\%$ de sua superfície.

\textbf{18.} $(W_n)$ diminui, então $ W_n \leq W_{n-1} \leq
W_{n-2} = \frac{n}{n-1}W_n $: apertando, $ W_{n-1}/W_n \to 1$.
Multiplicando por $ W_nW_{n-1} = \frac\pi{2n}$: $ W_n^2 \sim
\frac\pi{2n}$, ou seja\ $ W_n \sim \sqrt{\pi/(2n)}$. Limite inferior:
$ W_n^2 \geq W_nW_{n+1} = \frac{\pi}{2(n+1)}$, então $ W_n \geq
\sqrt{\pi/(2(n+1))}$ para cada $ n $.

\textbf{19.} Numerador: $1 - x^2 \leq \eu^{-x^2}$ dá $(1 -
x^2)^{\frac{n-1}2} \leq \eu^{-(n-1)x^2/2}$, e com $ a =
\frac{n-1}2$,
\[
\int_\delta^1(1 - x^2)^{\frac{n-1}2}\dd x
\leq \int_\delta^\infty\eu^{-ax^2}\dd x
\leq \int_\delta^\infty\frac x\delta\,\eu^{-ax^2}\dd x
= \frac{\eu^{-a\delta^2}}{2a\delta}
= \frac{\eu^{-(n-1)\delta^2/2}}{(n-1)\delta}.
\]
Denominador: $2W_n \geq \sqrt{2\pi/(n+1)}$ pela pergunta 18.
A divisão fornece o limite exibido. Para $\delta =
s/\sqrt{n-1}$ torna-se
$\sqrt{\tfrac{n+1}{2\pi(n-1)}}\;\eu^{-s^2/2}/s = O\bigl(
\eu^{-s^2/2}/s\bigr)$, \omterm{def:b2:funcseq:def}{uniformemente} em $ n $: fora da laje
$\abs{x_1} \leq s/\sqrt{n-1}$ quase não há volume, por
$ s $ moderadamente grande --- e por simetria o mesmo vale para
todas as direções.

\textbf{20.} As duas afirmações coexistem porque descrevem
coordenadas diferentes do mesmo ponto. Quase todos os pontos
de $ B_n(1)$ tem \omterm{def:b2:nvs:norm}{norma} perto de $1$ (questão 17: radial
concentração perto da esfera), mas cada um de seus $ n $
coordenadas são pequenas, de ordem $1/\sqrt n $ (questão 19),
o que é consistente desde $ n $ coordenadas de tamanho $1/\sqrt n $
ter \omterm{def:b2:nvs:norm}{norma} de ordem $1$. Concentrados de volume de alta dimensão
onde todas as coordenadas compartilham o \omterm{def:b2:nvs:norm}{norma} orçamento igualmente ---
perto da esfera, mas longe de todos os pólos do eixo de coordenadas.

\textbf{21.} Fatiar $\Delta_n $ no $ x_n = t \in \intcc01$: o
fatia é $\{x' \in \R^{n-1} : x_i \geq 0,\ \sum x_i \leq 1 -
t\} = (1-t)\Delta_{n-1}$, de volume
$(1-t)^{n-1}\operatorname{vol}(\Delta_{n-1})$ pela homogeneidade.
Então
\[
\operatorname{vol}(\Delta_n) =
\operatorname{vol}(\Delta_{n-1})\int_0^1(1 - t)^{n-1}\dd t =
\frac{\operatorname{vol}(\Delta_{n-1})}{n}
\quad\Longrightarrow\quad
\operatorname{vol}(\Delta_n) = \frac1{n!}\,.
\]

\textbf{22.} O $2^n $ corte de signos-ortantes $ C_n $ em $2^n $
cópias de $\Delta_n $ (com sobreposições insignificantes na coordenada
hiperplanos): $\operatorname{vol}(C_n) = \frac{2^n}{n!}$. Se
$\sum\abs{x_i} \leq 1$ então $\sum x_i^2 \leq
\bigl(\sum\abs{x_i}\bigr)^2 \leq 1$: $ C_n \subseteq B_n(1)$;
e $ B_n(1) \subseteq \intcc{-1}1^n $ desde $\abs{x_i} \leq
\norm x $. Por isso $\frac{2^n}{n!} \leq V_n \leq 2^n $ ---
consistente com a questão 15, que coloca $ V_n $ entre o
escalas fatoriais e geométricas.

\textbf{23.} Para $(x, y)$ no disco unitário, a fatia de
$ B_4(1)$ é o disco de raio $\sqrt{1 - x^2 - y^2}$ no
$(z, w)$-avião, de \omterm{def:b2:multint:domain}{área} $\pi(1 - x^2 - y^2)$. Em \omterm{pb:b2:quadratic:1}{polar}
coordenadas:
\[
V_4 = \iint_{x^2+y^2\leq1}\pi(1 - x^2 - y^2)\,\dd x\,\dd y
= \pi\int_0^{2\pi}\!\!\int_0^1(1 - \rho^2)\rho\,
\dd\rho\,\dd\alpha
= \pi\cdot2\pi\cdot\frac14 = \frac{\pi^2}2 ,
\]
concordando com a questão 6.

\textbf{24.} A probabilidade é a razão do volume
$\dfrac{V_{20}}{2^{20}} = \dfrac{\pi^{10}}{10!\cdot2^{20}}
\approx \dfrac{0.0258}{1\,048\,576} \approx
2.5\cdot10^{-8}$. O número de empates até o primeiro acerto é
de ordem inversa, cerca de $4\cdot10^7$: um amostrador de rejeição
que funcionou perfeitamente para o disco ($\pi/4$ de acessos) é
inútil em dimensão $20$ --- a maldição da dimensionalidade em
uma linha.

\textbf{25.} Rota um (Partes I--II) usada: tipo Fubini
fatiamento da integral iterada, substituição de uma variável
em cada coordenada (homogeneidade), e o Integrais de Wallis
--- cálculo puro de uma variável mais indução. Rota dois
(Parte III) utilizado: Fubini para a estrutura do produto de $ I_n $,
a mudança \omterm{pb:b2:quadratic:1}{polar} das variáveis através da integral de Gauss de
\cref{ex:b2:multint:polar}, e o $\Gamma $ função
equação funcional. Eles se encontram em $ V_n =
\pi^{n/2}/\Gamma(\frac n2 + 1)$, com Stirling
(\cref{thm:b2:comparison:stirling}) convertendo a fórmula
em assintóticos. O volume do Ano 3 reconstrói tudo isso em
Integral de Lebesgue: aí Fubini e a mudança de variáveis
são teoremas para funções gerais integráveis, esféricas
existem coordenadas em todas as dimensões, e as mesmas bola
volumes reaparecem como os dividendos trabalhados do
medida de produto e problemas de Stirling --- com dominado
convergência substituindo nossos apertos feitos à mão.
\end{solution}
