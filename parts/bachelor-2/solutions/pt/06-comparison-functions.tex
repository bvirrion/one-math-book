\begin{solution}{exo:b2:comparison:1}
$\sqrt{x^2 + x + 1} = x\sqrt{1 + \tfrac1x + \tfrac{1}{x^2}} = x +
\frac12 + \frac38\cdot\frac1x + o\bigl(\frac1x\bigr)$ (binomial

\frac{1}{8x^2} = \frac{1}{2x} + \frac{3}{8x^2}$, entĂŁo multiplique por

\[

= \Bigl(1 + \frac{\sin x}{x}\Bigr)
\Bigl(1 + \frac{\ln x}{x} + \frac{(\ln x)^2}{x^2} +

\[
\frac{x + \sin x}{x - \ln x}
= 1 + \frac{\ln x}{x} + \frac{\sin x}{x}

\]
\end{solution}

\begin{solution}{exo:b2:comparison:2}
$ f(t) = t^\alpha $ ($ t \geq 1$).

$\alpha > -1$: divergĂŞncia, e por
\cref{thm:b2:comparison:seriesintegral}~(2), $\sum_{k\leq n}

$\alpha < -1$: convergĂŞncia, com resto $\sum_{k > n} k^\alpha

\end{solution}

\begin{solution}{exo:b2:comparison:3}
Deixar $ v_n = H_n - \ln n - \gamma \to 0$. EntĂŁo
\[
v_n - v_{n+1} = \ln\frac{n+1}{n} - \frac{1}{n+1}
= \Bigl(\frac1n - \frac{1}{2n^2}\Bigr) - \Bigl(\frac1n -

\end{solution}

\begin{solution}{exo:b2:comparison:5}
$ g(x) = x^n + x - 1$ aumenta estritamente em $\intcc{0}{1}$ de $-1$
para $1$: raiz Ăşnica $ x_n $. Desde $ x_n^n = 1 - x_n \in

\quad\Longrightarrow\quad

\]
EntĂŁo $ n\varepsilon_n = -\ln\varepsilon_n\,(1 + o(1)) \to +\infty $ e

\end{solution}

\begin{solution}{exo:b2:comparison:7}
Divida os dois maiores termos:
\[
\sum_{k=0}^{n} k! = n! + (n-1)! + \sum_{k \leq n-2} k! ,
\qquad
\sum_{k\leq n-2} k! \leq (n-1)\,(n-2)! = (n-1)! .
\]
EntĂŁo $1 \leq \frac{1}{n!}\sum k! \leq 1 + \frac{2}{n}$: o limite Ă©
$1$. Refino: $\frac{(n-1)!}{n!} = \frac1n $ e o limite bruto
$\sum_{k \leq n-2}k! \leq (n-1)!$ pode ser afiado da mesma maneira:
$\sum_{k\leq n-2} k! = (n-2)!\,(1 + O(\frac1n)) = O\bigl(\frac{n!}{n^2}\bigr)$.
Por isso
\[
\sum_{k=0}^{n} k! = n!\Bigl(1 + \frac1n + O\Bigl(\frac{1}{n^2}\Bigr)\Bigr).
\]
\end{solution}

\begin{solution}{exo:b2:comparison:10}

\begin{solution}{exo:b2:comparison:11}

\begin{solution}{exo:b2:comparison:12}

\begin{solution}{pb:b2:comparison:1}
\textbf{1.} Subtraindo as duas expansĂµes: $\sum_i (c_i -

$ u = \frac{\ln x}x \to 0$,
\[
\frac{1}{x - \ln x} = \frac1x\cdot\frac{1}{1 - u}
= \frac1x\bigl(1 + u + u^2 + O(u^3)\bigr)
= \frac1x + \frac{\ln x}{x^2} + \frac{(\ln x)^2}{x^3} +
o\Bigl(\frac{(\ln x)^2}{x^3}\Bigr).
\]
NĂŁo $\frac c{x^2}$ termo aparece porque a expansĂŁo Ă© uma
sĂ©rie geomĂ©trica em $ u = \frac{\ln x}{x}$: cada termo carrega como

\textbf{2.} $ f(x) = \eu^x + x $ Ă© \omterm{def:b2:metric:continuity}{contĂ­nuo}, estritamente

$ x_n/n \to 0$ e $ x_n = \ln n + \ln(1 - x_n/n) = \ln n + o(1)

\textbf{3.} Escrever $ u_n = x_n/n $. Segunda passagem: $ u_n = \frac{\ln n
+ o(1)}{n}$, entĂŁo
\[
