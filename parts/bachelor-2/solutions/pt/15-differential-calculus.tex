\begin{solution}{exo:b2:diffcalc:1}
$J_f = \begin{pmatrix} 2x & -2y\\ 2y & 2x\end{pmatrix}$, $\det J_f
= 4(x^2 + y^2)$: invertível longe da origem. (Esse $ f $ é $ z
\mapsto z^2$ em disfarce complexo.)

$ J_\Phi = \begin{pmatrix} \cos\theta & -r\sin\theta & 0\\
\sin\theta & r\cos\theta & 0\\ 0 & 0 & 1\end{pmatrix}$, $\det = r $:
invertível para $ r \neq 0$ (coordenadas cilíndricas).
\end{solution}

\begin{solution}{exo:b2:diffcalc:2}
Para $ v = (a, b) \neq 0$: $\frac{f(tv) - 0}{t} = \frac{t^3a^3}{t\cdot
t^2(a^2+b^2)} = \frac{a^3}{a^2 + b^2}$: cada direcional
existe derivada, com valor $ D_v = \frac{a^3}{a^2+b^2}$. Mas $ v
\mapsto D_v $ não é linear ($ D_{(1,0)} = 1$, $ D_{(0,1)} = 0$,
$ D_{(1,1)} = \frac12 \neq 1$): nenhum mapa linear pode produzir estes
valores, então $ f $ não é \omterm{def:b2:diffcalc:differential}{diferenciável} no $0$ (o \omterm{def:b2:diffcalc:differential}{diferencial} faria
tem que ser $ v \mapsto D_v $).
\end{solution}

\begin{solution}{exo:b2:diffcalc:3}
$ f = x^3 + y^3 - 3xy $: pontos críticos $(0,0)$ e $(1,1)$ (Ano 1
cálculo). Hessianos: $ H = \begin{pmatrix} 6x & -3\\ -3 &
6y\end{pmatrix}$. No $(0,0)$: sinais próprios misturados ($\det = -9 < 0$):
sela. No $(1,1)$: $\det = 27 > 0$, rastrear $> 0$: positivo
mínimo local definido e estrito --- agora justificado por
\cref{thm:b2:diffcalc:extrema} em vez de decretado.

$ g = x^4 + y^4 - 2(x-y)^2$: $\nabla g = (4x^3 - 4(x - y),\; 4y^3 +
4(x-y))$; pontos críticos $(0,0)$, $(\sqrt2, -\sqrt2)$,
$(-\sqrt2, \sqrt2)$ (Ano 1). No $(\pm\sqrt2, \mp\sqrt2)$: $ H =
\begin{pmatrix} 12\cdot2 - 4 & 4\\ 4 & 20\end{pmatrix} =
\begin{pmatrix} 20 & 4\\ 4 & 20\end{pmatrix}$: definido positivo
(diagonalmente dominante; valores próprios $24, 16$): mínimos locais estritos.
No $(0,0)$: $ H = \begin{pmatrix} -4 & 4\\ 4 & -4\end{pmatrix}$,
singular negativo semidefinido: o teste é silencioso; o direcional
estudo ($ g(x,x) = 2x^4 > 0$, $ g(x,-x) = 2x^4 - 8x^2 < 0$ pequeno)
mostra um ponto semelhante a uma sela --- sem extremo.
\end{solution}

\begin{solution}{exo:b2:diffcalc:4}
Diferenciar $ t \mapsto f(tx)$ no $ t = 1$: pela regra da cadeia,
$\langle \nabla f(x), x\rangle $; por homogeneidade a mesma função
é $ t^pf(x)$, de derivada $ pf(x)$ no $ t = 1$: Identidade de Euler.

Converse: consertar $ x \neq 0$ e deixe $\varphi(t) = f(tx) - t^p f(x)$
sobre $ t > 0$. Então $\varphi'(t) = \langle\nabla f(tx), x\rangle -
pt^{p-1}f(x) = \frac1t\bigl(\langle \nabla f(tx), tx\rangle -
p\,t^pf(x)\bigr)$. A hipótese --- a identidade de Euler no ponto
$ tx $ --- avalia o colchete como $ p\,f(tx) - p\,t^pf(x) =
p\,\varphi(t)$. Então $\varphi' = \frac{p}{t}\varphi $ com $\varphi(1)
= 0$: a solução única da EDO linear é $\varphi \equiv 0$ (Ano
1 exclusividade), ou seja,\ $ f(tx) = t^pf(x)$.
\end{solution}

\begin{solution}{exo:b2:diffcalc:5}
Expandindo $ f(x + h) - f(x) = \langle Ax - b, h\rangle +
\frac12\langle Ah, h\rangle $ (simetria de $ A $): $\nabla f(x) = Ax -
b $ e $ H_f = A $ em todos os lugares. $ f $ é convexo se $ A $ é positivo
semidefinido (o teste de Hessian, global aqui desde $ H $ é constante:
a fórmula de Taylor de segunda ordem é exata). Se $ A $ é positivo
definido: o único ponto crítico é $ x^* = A^{-1}b $, e $ f(x^*
+ h) - f(x^*) = \frac12\langle Ah, h\rangle \geq
\frac{\lambda_{\min}}{2}\norm h^2 > 0$ para $ h \neq 0$: rigoroso
mínimo global.
\end{solution}

\begin{solution}{exo:b2:diffcalc:6}
Desde $\nabla g(a) \neq 0$, um parcial, digamos $\frac{\partial
g}{\partial y}(a) \neq 0$: o teorema da função implícita
(\cref{thm:b2:diffcalc:inverse}companheiro) parametrizações $\{g =
0\}$ aproximar $ a = (a_1, a_2)$ como $\gamma(t) = (t, y(t))$ com $ y $
$ C^1$, $ y'(t) = -\frac{\partial_x g}{\partial_y g}(\gamma(t))$
(diferenciar $ g(t, y(t)) = 0$). A função de uma variável $ t
\mapsto f(\gamma(t))$ tem um extremo local em $ t = a_1$:
\[
0 = \frac{\dd}{\dd t}f(\gamma(t))\Big|_{a_1}
= \partial_x f(a) + \partial_y f(a)\,y'(a_1)
= \partial_x f(a) - \partial_yf(a)\frac{\partial_x
g(a)}{\partial_y g(a)} :
\]
os vetores $\nabla f(a)$ e $\nabla g(a)$ tem proporcional
coordenadas: $\nabla f(a) = \lambda \nabla g(a)$ com $\lambda =
\frac{\partial_y f(a)}{\partial_y g(a)}$.

Aplicação: no círculo, $\nabla(xy) = (y, x)$ paralelo a
$(2x, 2y)$ forças $ y^2 = x^2$; com a restrição, os candidatos
são $\pm\bigl(\tfrac{1}{\sqrt2}, \tfrac{1}{\sqrt2}\bigr)$ (valor
$\frac12$) e $\pm\bigl(\tfrac{1}{\sqrt2},
-\tfrac{1}{\sqrt2}\bigr)$ (valor $-\frac12$): máx. $\frac12$, mínimo
$-\frac12$ (atingido: o círculo é \omterm{def:b2:metric:compact}{compactar}).
\end{solution}

\begin{solution}{exo:b2:diffcalc:7}
$\nabla f = (2x - y + 1,\; 2y - x - 1) = 0$: resolver, $ x =
-\frac13$, $ y = \frac13$. Hessiano $\begin{pmatrix} 2 & -1\\ -1 &
2\end{pmatrix}$, definido positivo: mínimo global do quadrático
$ f $, valor $ f\bigl(-\frac13, \frac13\bigr) = -\frac13$.

No triângulo $ T $: o ponto crítico interior $(-\frac13,
\frac13) \notin T $ (negativo $ x $). Bordas: ligadas $ y = 0$, $ x \in
\intcc{0}{1}$: $ f = x^2 + x $, aumentando: extremos $0$ e $2$. Sobre
$ x = 0$: $ f = y^2 - y $, mínimo $-\frac14$ no $ y = \frac12$, valores
$0$ e $0$ nas extremidades. Sobre $ x + y = 1$: substituto $ y = 1 - x $,
$ f = x^2 + (1-x)^2 - x(1-x) + x - (1-x) = 3x^2 - x $; sobre
$\intcc{0}{1}$: mínimo $-\frac{1}{12}$ no $ x = \frac16$, valores
$0$ (no $ x=0$) e $2$ (no $ x=1$). Vértices: $ f(0,0) = 0$, $ f(1,0)
= 2$, $ f(0,1) = 0$. Global ativado $ T $: mínimo $-\frac14$ no $(0,
\frac12)$, máximo $2$ no $(1, 0)$.
\end{solution}

\begin{solution}{exo:b2:diffcalc:8}
$ J_f = \begin{pmatrix} 1 & 2y\\ 2x & 1\end{pmatrix}$, $\det J_f =
1 - 4xy $. No $0$: $\det = 1 \neq 0$: difeomorfismo local
(\cref{thm:b2:diffcalc:inverse}), com
\[
\dd(f^{-1})_{(0,0)} = (J_f(0))^{-1} = I_2 .
\]
Invertibilidade em uma bola: necessidade $4\abs{xy} < 1$ por todo; sobre
$\norm{(x,y)}_2 < r $, $\abs{xy} \leq \frac{x^2 + y^2}{2} <
\frac{r^2}{2}$, então $ r = \frac{1}{\sqrt2}$ funciona; e é o maior:
em $(x, y) = \bigl(\tfrac12, \tfrac12\bigr)$, de \omterm{def:b2:nvs:norm}{norma}
$\frac{1}{\sqrt2}$, $\det J_f = 0$.
\end{solution}

\begin{solution}{exo:b2:diffcalc:9}
$ f(t) = (\cos t, \sin t)$: $ f(0) = f(2\pi) = (1, 0)$, ainda $ f'(t) =
(-\sin t, \cos t)$ tem \omterm{def:b2:nvs:norm}{norma} $1$, nunca zero: nenhum ponto onde o
a derivada desaparece --- Rolle não tem análogo vetorial. A média
a desigualdade de valor se mantém confortavelmente: $\norm{f(2\pi) - f(0)} = 0 \leq
2\pi \cdot \sup\norm{f'} = 2\pi $.
\end{solution}

\begin{solution}{exo:b2:diffcalc:10}
$ f(x+h) - f(x) = 2\langle x, h\rangle + \norm h^2$: $\dd f_x =
2\langle x, \cdot\rangle $, $\nabla f(x) = 2x $, Hessiano $2I $
(constante). Para $ g $:
\[
g(x + h) - g(x) = \langle Ax, h\rangle + \langle Ah, x\rangle
+ \langle Ah, h\rangle
= \bigl\langle (A + A^{\mathsf T})x,\ h\bigr\rangle +
O(\norm h^2),
\]
então $\nabla g(x) = (A + A^{\mathsf T})x $ e $ H_g = A +
A^{\mathsf T}$, constante. Por \cref{exo:b2:diffcalc:11}, $ g $ é
convexo se sef $ A + A^{\mathsf T}$ é positivo semidefinido ---
apenas o \omterm{def:b2:quadratic:adjoint}{simétrico} parte de $ A $ importa, como de fato $ g(x) =
\langle \frac{A + A^{\mathsf T}}2 x, x\rangle $.
\end{solution}

\begin{solution}{exo:b2:diffcalc:11}
$ f $ é convexo se sua restrição a cada segmento for convexa,
ou seja, se cada $\varphi(t) = f(a + tv)$ é convexo. Pela cadeia
regra $\varphi''(t) = \langle H_{a+tv}\,v,\ v\rangle $.

Se todos os Hessianos forem semidefinidos positivos: $\varphi'' \geq 0$,
então cada um $\varphi $ é convexo (volume do ano 1) e $ f $ é convexo.
Inversamente se $ f $ é convexo, cada $\varphi $ é convexo, então
$\varphi''(0) \geq 0$: $\langle H_a v, v\rangle \geq 0$ para
cada $ a $ e todas as direções $ v $: todos os hessianos são positivos
semidefinido.
\end{solution}

\begin{solution}{exo:b2:diffcalc:12}
\begin{enumerate}
  \item Pela desigualdade de valor médio
        (\cref{thm:b2:diffcalc:mvi}) aplicado a $ g $:
        $\norm{g(x) - g(y)} \leq k\norm{x-y}$, então
        \[
        \norm{f(x) - f(y)} \geq \norm{x - y} - \norm{g(x) -
        g(y)} \geq (1 - k)\norm{x - y} :
        \]
        $ f $ é injetivo e $ f^{-1}$ (definido na imagem) é
        $\frac{1}{1-k}$-Lipschitz.
  \item Consertar $ y $; $ T(x) = y - g(x)$ satisfaz $\norm{T(x) -
        T(x')} = \norm{g(x') - g(x)} \leq k\norm{x - x'}$: um
        contração do \omterm{def:b2:metric:complete}{completo} espaço $\R^n $. Banach
        (\cref{thm:b2:metric:banach}) fornece um ponto fixo $ x^*
        = y - g(x^*)$, ou seja\ $ f(x^*) = y $: $ f $ é sobrejetivo.
  \item $ f $ é, portanto, uma bijeção de $\R^n $ com
        $\frac{1}{1-k}$-Inverso de \omterm{def:b2:metric:continuity}{Lipschitz}: um inverso global
        teorema, onde a pequenez de $\dd g $ em todos os lugares
        substitui a hipótese de invertibilidade local de
        \cref{thm:b2:diffcalc:inverse}.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b2:diffcalc:1}
\textbf{1.} $\norm{ABx} \leq \vertiii A\norm{Bx} \leq
\vertiii A\vertiii B\norm x $: assumir o controle $\norm x = 1$.
Produtos matriciais, $\det $ e $\operatorname{tr}$ são polinomiais
funções das entradas, portanto \omterm{def:b2:metric:continuity}{contínuo} ($\mathcal M_n(\R)
\simeq \R^{n^2}$, todos \omterm{def:b2:nvs:norm}{normas} equivalente:
\cref{thm:b2:nvs:finitedim}).

\textbf{2.} $\sum\vertiii{X^k} \leq \sum\vertiii X^k <
\infty $: a série converge \omterm{def:b2:series:def}{absolutamente}, portanto converge
(\cref{thm:b2:nvs:absoluteconvergence}). De $(I -
X)\sum_{k\leq N}X^k = I - X^{N+1} \to I $: a soma é $(I -
X)^{-1}$. \omterm{def:b2:nvs:norm}{Norma}: $\leq \sum\vertiii X^k = \frac{1}{1 -
\vertiii X}$; e
\[
(I - X)^{-1} - I - X = \sum_{k\geq2}X^k = X^2(I - X)^{-1},
\qquad
\vertiii{X^2(I-X)^{-1}} \leq
\frac{\vertiii X^2}{1 - \vertiii X} = O(\vertiii X^2).
\]

\textbf{3.} $ A + H = A(I + A^{-1}H)$ com $\vertiii{A^{-1}H}
\leq \vertiii{A^{-1}}\vertiii H < 1$: invertível pela questão 2:
o \omterm{def:b2:metric:topology}{abrir} bola de raio $\vertiii{A^{-1}}^{-1}$ em volta $ A $ mentiras
em $ GL_n(\R)$. Densidade: $\det(A + \varepsilon I)$ é um diploma-$ n $
polinômio em $\varepsilon $ com coeficiente líder $1$: isso
tem um número finito de raízes, então há $\varepsilon_k \to 0$
com $ A + \varepsilon_kI $ invertível, convergindo para $ A $.

\textbf{4.} Para $\vertiii H < \frac{1}{2\vertiii{A^{-1}}}$:
\[
(A + H)^{-1} - A^{-1}
= \bigl[(I + A^{-1}H)^{-1} - I\bigr]A^{-1},
\]
de \omterm{def:b2:nvs:norm}{norma} no máximo $\frac{\vertiii{A^{-1}H}}{1 -
\vertiii{A^{-1}H}}\,\vertiii{A^{-1}} \leq
2\vertiii{A^{-1}}^2\vertiii H \to 0$: $\Phi $ é \omterm{def:b2:metric:continuity}{contínuo} em
cada $ A \in GL_n(\R)$.

\textbf{5.} $(A+H)^2 = A^2 + AH + HA + H^2$: o mapa $ H
\mapsto AH + HA $ é linear e o erro $ H^2$ é
$ O(\vertiii H^2)$. Expandindo $(A + H)^k $ e classificando por
número de $ H $-fatores: a parte linear é
$\sum_{i=0}^{k-1}A^iHA^{k-1-i}$, e os termos com $\geq 2$
fatores $ H $ são limitados por $\binom k2$-muitos produtos de \omterm{def:b2:nvs:norm}{norma}
$\leq \vertiii A^{k-2}\vertiii H^2$-escala: $ O(\vertiii H^2)$.
Não se pode reduzir a soma para $ kA^{k-1}H $ porque $ H $ e $ A $
não precisa se deslocar --- a soma é o não comutativo correto
derivado.

\textbf{6.} Para pequenos $ H $:
\[
(A+H)^{-1} = (I + A^{-1}H)^{-1}A^{-1}
= \bigl(I - A^{-1}H + O(\vertiii H^2)\bigr)A^{-1}
= A^{-1} - A^{-1}HA^{-1} + O(\vertiii H^2) :
\]
$\dd\Phi_A(H) = -A^{-1}HA^{-1}$, linear em $ H $. Para $ n = 1$:
$\dd(1/a)(h) = -h/a^2$, a derivada familiar.

\textbf{7.} Regra da cadeia ao longo da curva:
$\bigl(A(t)^{-1}\bigr)' = \dd\Phi_{A(t)}(A'(t)) =
-A(t)^{-1}A'(t)A(t)^{-1}$. No $ A(t) = I + tB $, $ t = 0$: $(I +
tB)^{-1} = I - tB + O(t^2)$.

\textbf{8.} Pela questão 5 e invariância cíclica do traço:
\[
\dd f_A(H) = \operatorname{tr}\Bigl(\sum_{i=0}^{k-1}
A^iHA^{k-1-i}\Bigr) = k\operatorname{tr}\bigl(A^{k-1}H\bigr) .
\]
Contra o produto Frobenius, $\dd f_A(H) =
\operatorname{tr}\bigl((\nabla f)^{\mathsf T}H\bigr)$ requer
$(\nabla f)^{\mathsf T} = kA^{k-1}$: $\nabla f(A) =
k\,(A^{k-1})^{\mathsf T}$.

\textbf{9.} $ f(A + H) - f(A) = 2\operatorname{tr}
(A^{\mathsf T}H) + \operatorname{tr}(H^{\mathsf T}H)$: o
\omterm{def:b2:diffcalc:differential}{diferencial} é $ H \mapsto 2\operatorname{tr}(A^{\mathsf T}H) =
2\langle A, H\rangle $, então $\nabla f(A) = 2A $; a segunda ordem
termo é exatamente $\norm H_F^2$: o Hessian é o dobro do
identidade forma quadrática, positivo definido e constante, então
$\norm\cdot_F^2$ é estritamente convexo (a fórmula de Taylor é
exato aqui).

\textbf{10.} Por multilinearidade nas colunas, $\det(I + H) =
\sum_{S\subseteq\{1,\dots,n\}}\det(M_S)$ onde $ M_S $ tem coluna
$ h_j $ para $ j \in S $ e $ e_j $ de outra forma. $ S = \varnothing $
dá $1$; $ S = \{j\}$ dá o determinante de $ I $ com coluna
$ j $ substituído por $ h_j $, nomeadamente a sua $ j $-ésima entrada $ h_{jj}$,
resumindo para $\operatorname{tr}H $; cada termo com $\abs S \geq 2$
é um determinante com pelo menos duas colunas de tamanho
$ O(\vertiii H)$, por isso $ O(\vertiii H^2)$ (mapas multilineares em um
espaço de dimensão finita são limitados), e existem finitamente
muitos. Então $\det(I + H) = 1 + \operatorname{tr}H +
O(\vertiii H^2)$: $\dd(\det)_I = \operatorname{tr}$.

\textbf{11.} $\det(A + H) = \det A\,\det(I + A^{-1}H) =
\det A\,\bigl(1 + \operatorname{tr}(A^{-1}H) +
O(\vertiii H^2)\bigr)$: o \omterm{def:b2:diffcalc:differential}{diferencial} é $ H \mapsto
\det(A)\operatorname{tr}(A^{-1}H)$.

\textbf{12.} Expansão de Laplace ao longo da linha $ i $: $\det A =
\sum_j a_{ij}C_{ij}$, e os cofatores $ C_{ij}$ não envolva
linha $ i $: $\frac{\partial\det}{\partial a_{ij}} = C_{ij}$. Por isso
\[
\dd(\det)_A(H) = \sum_{i,j}C_{ij}h_{ij}
= \operatorname{tr}\bigl(\operatorname{com}(A)^{\mathsf T}
H\bigr)
= \operatorname{tr}\bigl(\operatorname{adj}(A)H\bigr),
\qquad
\nabla(\det)(A) = \operatorname{com}(A) .
\]
Para invertível $ A $, $\operatorname{adj}A = \det(A)A^{-1}$
recupera a questão 11. Ao longo de um $ C^1$ curva, a regra da cadeia diz
$(\det A(t))' = \operatorname{tr}(\operatorname{adj}
(A(t))\,A'(t))$: \omterm{pb:b2:diffcalc:1}{Fórmula de Jacobi}.

\textbf{13.} Com $ A' = MA $ e
$\operatorname{adj}(A)A = \det(A)I $:
\[
(\det A)' = \operatorname{tr}\bigl(\operatorname{adj}(A)MA
\bigr)
= \operatorname{tr}\bigl(A\operatorname{adj}(A)M\bigr)
= \det A\;\operatorname{tr}M
\]
(ciclicidade; $ A\operatorname{adj}A = \det(A) I $ também). O
EDO linear escalar $ y' = \operatorname{tr}(M(t))\,y $ tem o
solução única $ y(t) = y(0)\exp\bigl(\int_0^t
\operatorname{tr}M\bigr)$ (Ano 1): Fórmula de Liouville.

\textbf{14.} No $ A \in SL_n(\R)$, pegar $ H = \frac1nA $:
$\dd(\det)_A\bigl(\tfrac1nA\bigr) = \frac1n\det(A)
\operatorname{tr}(A^{-1}A) = \frac1n\cdot1\cdot n = 1 \neq 0$:
o \omterm{def:b2:diffcalc:differential}{diferencial} é uma forma linear diferente de zero, portanto sobrejetiva
para $\R $ em todos os pontos do nível definido: $ SL_n(\R)$ é um
conjunto de nível suave (de dimensão $ n^2 - 1$).

\textbf{15.} $\sum_k\vertiii{A^k/k!} \leq
\sum\vertiii A^k/k! = \eu^{\vertiii A}$: convergência absoluta
(pergunta 2 completude argumento), com convergência normal
em cada bola $\vertiii A \leq R $ (vinculado $ R^k/k!$ independente
de $ A $). Cada soma parcial é \omterm{def:b2:metric:continuity}{contínuo} (polinomial); o
limite uniforme de bolas é \omterm{def:b2:metric:continuity}{contínuo}: $ A \mapsto \eu^A $ é
\omterm{def:b2:metric:continuity}{contínuo}, com $\vertiii{\eu^A} \leq \eu^{\vertiii A}$.

\textbf{16.} Ambas as séries convergem \omterm{def:b2:series:def}{absolutamente}, então o \omterm{thm:b2:series:fubini}{Cauchy
produto} é legítimo (\cref{thm:b2:series:fubini}):
\[
\eu^A\eu^B = \sum_{n\geq0}\frac{1}{n!}\sum_{k=0}^n\binom
nkA^kB^{n-k}
= \sum_{n\geq0}\frac{(A+B)^n}{n!} = \eu^{A+B},
\]
a identidade binomial que requer $ AB = BA $. Com $ B = -A $:
$\eu^A\eu^{-A} = \eu^0 = I $: todo $\eu^A \in GL_n(\R)$.

\textbf{17.} A série $\sum t^kA^k/k!$ e sua série derivada
$\sum t^{k-1}A^k/(k-1)! = A\sum t^{k-1}A^{k-1}/(k-1)!$ \omterm{def:b2:integration:improper}{convergir}
\omterm{def:b2:funcseq:series}{normalmente} em cada segmento $\abs t \leq T $ (limites
$ T^k\vertiii A^k/k!$): o teorema da diferenciação para séries
(\cref{thm:b2:funcseq:seriestransfer}, entrada aplicada por entrada)
dá $\frac{\dd}{\dd t}\eu^{tA} = A\eu^{tA}$; factoring $ A $
à direita, em vez disso, dá $\eu^{tA}A $. Iterando: $ C^\infty $.

\textbf{18.} $ A(t) = \eu^{tA}$ satisfaz $ A'(t) = A\,A(t)$:
Fórmula de Liouville (questão 13) com constante $ M = A $ dá
$\det\eu^{tA} = \eu^{t\operatorname{tr}A}$ (valor $1$ no $ t =
0$); no $ t = 1$, $\det\eu^A = \eu^{\operatorname{tr}A}$.
Verificações: $ n = 1$ é o próprio exponencial; um nilpotente $ A $ tem
$\operatorname{tr}A = 0$ e $\eu^A $ unipotente de determinante
$1$; $\operatorname{tr}A = 0$ dá $\det\eu^A = 1$: o
matrizes de traço zero são enviadas para $ SL_n(\R)$.

\textbf{19.} $\eu^H - I - H = \sum_{k\geq2}H^k/k!$, de \omterm{def:b2:nvs:norm}{norma}
$\leq \vertiii H^2\eu^{\vertiii H} = O(\vertiii H^2)$:
$\dd(\exp)_0 = \mathrm{id}$, invertível. Além disso $\exp $ é
$ C^1$: pela questão 5, o \omterm{def:b2:diffcalc:differential}{diferencial} candidato $ H \mapsto
\sum_k\frac1{k!}\sum_iA^iHA^{k-1-i}$ é um \omterm{def:b2:funcseq:series}{normalmente} convergente
série de mapas lineares dependendo continuamente sobre $ A $ (limites
$\vertiii A^{k-1}/(k-1)!$ em bolas), então as parciais existem e
são \omterm{def:b2:metric:continuity}{contínuo} (\cref{thm:b2:diffcalc:c1} e a série
teorema da transferência). O teorema da função inversa
(\cref{thm:b2:diffcalc:inverse}) aplica-se em $0$: $\exp $ é um
$ C^1$ difeomorfismo de uma vizinhança de $0$ em um
bairro de $ I $ --- matrizes próximas $ I $ tem logaritmos.

\textbf{20.} Para \omterm{def:b2:quadratic:adjoint}{simétrico} $ S = PDP^{\mathsf T}$ (espectral
teorema): $\eu^S = P\eu^DP^{\mathsf T}$ é \omterm{def:b2:quadratic:adjoint}{simétrico} com
valores próprios $\eu^{\lambda_i} > 0$: definido positivo.
Surjetividade: um definido positivo $ Q =
P\operatorname{diag}(\mu_i)P^{\mathsf T}$ ($\mu_i > 0$) é
$\eu^S $ para $ S = P\operatorname{diag}(\ln\mu_i)P^{\mathsf T}$.
Injetividade: $\eu^S $ determina seu autoespaços, que são
exatamente aqueles de $ S $ (em cada autoespaço de $ S $ para $\lambda $,
$\eu^S $ atua como $\eu^\lambda $; distinto $\lambda $ dar
distinto $\eu^\lambda $) e tomando $\ln $ do valores próprios
recupera $ S $. Então $\exp $ é uma bijeção de \omterm{def:b2:quadratic:adjoint}{simétrico} matrizes
em definidos positivos.

\textbf{21.} $ F(A + H) = A^{\mathsf T}A + A^{\mathsf T}H +
H^{\mathsf T}A + H^{\mathsf T}H $: $\dd F_A(H) =
A^{\mathsf T}H + H^{\mathsf T}A $ (valores em $ S_n $; erro
$ O(\vertiii H^2)$). No $ A \in O_n $ e para $ S \in S_n $, o
escolha $ H = \frac12AS $ dá
\[
A^{\mathsf T}\cdot\tfrac12AS + \tfrac12(AS)^{\mathsf T}A
= \tfrac12 S + \tfrac12 S^{\mathsf T} = S :
\]
sobrejetivo. $ O_n = F^{-1}(I)$ é um conjunto de níveis suaves de
dimensão $ n^2 - \dim S_n = \frac{n(n-1)}2$.

\textbf{22.} Diferenciando $ A(t)^{\mathsf T}A(t) = I $ no $ t
= 0$ (com $ A(0) = I $): $ A'(0)^{\mathsf T} + A'(0) = 0$:
antisimétrico. Por outro lado, para $ K^{\mathsf T} = -K $:
$(\eu^{tK})^{\mathsf T}\eu^{tK} = \eu^{tK^{\mathsf T}}
\eu^{tK} = \eu^{-tK}\eu^{tK} = I $ (transpor a série
termicamente; os expoentes comutam): a curva permanece em $ O_n $,
com velocidade $ K $ no $ t = 0$. Espaço tangente em $ I $ $=$ o
matrizes antissimétricas, da dimensão esperada
$\frac{n(n-1)}2$.

\textbf{23.} $\operatorname{tr}K = 0$ para antisimétrico $ K $,
então $\det\eu^K = \eu^0 = 1$ (questão 18): o exponencial
pousa em $ SO_n $. Para $ n = 2$: $ J^2 = -I $, então $ J^{2m} =
(-1)^mI $, $ J^{2m+1} = (-1)^mJ $, e
\[
\eu^{\theta J}
= \Bigl(\sum_m\frac{(-1)^m\theta^{2m}}{(2m)!}\Bigr)I
+ \Bigl(\sum_m\frac{(-1)^m\theta^{2m+1}}{(2m+1)!}\Bigr)J
= \cos\theta\,I + \sin\theta\,J ,
\]
a rotação por $\theta $: a série seno e cosseno ao vivo
dentro do \omterm{ex:b2:nvs:matrixexp}{matriz exponencial}.

\textbf{24.} Para invertível $ A, B $: $\operatorname{adj}(AB) =
\det(AB)(AB)^{-1} = \det(B)\det(A)B^{-1}A^{-1} =
\operatorname{adj}(B)\operatorname{adj}(A)$. Ambos os lados do
identidade são polinomiais (portanto \omterm{def:b2:metric:continuity}{contínuo}) mapas das entradas
de $(A, B)$; eles concordam com o subconjunto denso $ GL_n\times GL_n $
de $\mathcal M_n\times\mathcal M_n $ (pergunta 3: aproximado
cada fator), então eles concordam em todos os lugares.
