\begin{solution}{exo:b2:genfun:2}
\emph{BinĂŽmio:} $ G(t) = (1 - p + pt)^n $, $ G'(t) = np(1 - p +
pt)^{n-1}$, $ G''(t) = n(n-1)p^2(1 - p + pt)^{n-2}$, entĂŁo
\[
\E(X) = G'(1) = np,
\qquad
V(X) = G''(1) + G'(1) - G'(1)^2
= n(n-1)p^2 + np - n^2p^2 = np(1-p).
\]
\emph{GeomĂ©trico} ($ q = 1 - p $): $ G(t) = \frac{pt}{1 - qt}$, entĂŁo
$ G'(t) = \frac{p}{(1 - qt)^2}$ e $ G''(t) = \frac{2pq}{(1 -
qt)^3}$; no $ t = 1$ (usando $1 - q = p $):
\[
\E(X) = \frac{p}{p^2} = \frac1p,
\qquad
V(X) = \frac{2q}{p^2} + \frac1p - \frac{1}{p^2}
= \frac{2q + p - 1}{p^2}
= \frac{q}{p^2} ,
\]
correspondĂȘncia \cref{exo:b2:randomvar:1} com menos trabalho.
\end{solution}

\begin{solution}{exo:b2:genfun:4}
Por \cref{thm:b2:genfun:compound} com $ G_N(s) = e^{\lambda(s-1)}$
e $ G_X(t) = 1 - p + pt $:
\[
G_S(t) = e^{\lambda(1 - p + pt - 1)} = e^{\lambda p(t - 1)} :
\]
$ S \sim \mathcal{P}(\lambda p)$. A contagem descartada $ D = N - S $
conta os mesmos itens mantidos com probabilidade $1 - p $, entĂŁo pelo
mesmo cĂĄlculo $ D \sim \mathcal{P}(\lambda(1 - p))$.
IndependĂȘncia, diretamente: para $ j, k \in \N $,
\begin{align*}
\P(S = j,\ D = k)
&= \P(N = j + k)\,\binom{j+k}{j}p^jq^k
= e^{-\lambda}\frac{\lambda^{j+k}}{(j+k)!}\,
\frac{(j+k)!}{j!\,k!}\,p^jq^k\\
&= \Bigl(e^{-\lambda p}\frac{(\lambda p)^j}{j!}\Bigr)
\Bigl(e^{-\lambda q}\frac{(\lambda q)^k}{k!}\Bigr)
\end{align*}
com $ q = 1 - p $: a junta \omterm{def:b2:randomvar:law}{lei} fatores como
$\mathcal{P}(\lambda p) \otimes \mathcal{P}(\lambda q)$. Um Poisson
fluxo dividido em rendimentos aleatĂłrios \emph{\omterm{def:b2:proba:independence}{independente}} Fluxos de Poisson ---
um pequeno milagre constantemente usado na teoria das filas.
\end{solution}

\begin{solution}{exo:b2:genfun:5}

\begin{solution}{exo:b2:genfun:7}
\emph{\omterm{def:b2:randomvar:expectation}{Expectativa}.} Primeiro $\E(Z_n) = m^n $: por
\cref{thm:b2:genfun:compound}, $\E(Z_{n+1}) = \E(Z_n)\,m $, e

a descendĂȘncia total Ă© $ Y = 1 + Y_1 + \dots + Y_k $, onde $ Y_i $ Ă© o

\]

\omterm{def:b2:randomvar:law}{lei} $ p_0 = 1 - p $, $ p_2 = p $ de ramificaĂ§ĂŁo binĂĄria, esta quadrĂĄtica

\end{solution}

\begin{solution}{exo:b2:genfun:8}
Deixar $ R > 1$ seja o raio e fixe $ t \in \intoo{1}{R}$. EntĂŁo
$\E(t^X) = G(t) < \infty $, e a desigualdade de Markov

\end{solution}

\begin{solution}{exo:b2:genfun:9}
Escrever $ p_k^{(n)} = \P(X_n = k)$, $ p_k = \P(X = k)$.

\emph{Caso $ k = 0$.} Para $ t \in \intoo{0}{1}$ e qualquer \omterm{def:b2:randomvar:law}{lei}
$(q_j)$ com $\sum_j q_j \leq 1$:
\[
\Bigl|\,q_0 - \sum_j q_jt^j\Bigr|
= \sum_{j \geq 1} q_j t^j
\leq \sum_{j\geq1}t^j = \frac{t}{1 - t} .
\]
Por isso
\[
\abs{p_0^{(n)} - p_0}
\leq \frac{2t}{1 - t}
+ \abs{G_{X_n}(t) - G_X(t)} .
\]
Dado $\varepsilon > 0$, escolher $ t $ com $\frac{2t}{1-t} <
\frac\varepsilon2$, entĂŁo $ n_0$ tal que o Ășltimo termo seja $<
\frac\varepsilon2$ para $ n \geq n_0$: entĂŁo $ p_0^{(n)} \to p_0$.

\emph{Etapa de induĂ§ĂŁo.} Assumir $ p_j^{(n)} \to p_j $ para $ j < k $.
Considere o \emph{mudou} funĂ§Ă”es
\[
g_n(t) = \frac{G_{X_n}(t) - p^{(n)}_0}{t}
= \sum_{j\geq0} p^{(n)}_{j+1}t^j,
\qquad
g(t) = \frac{G_X(t) - p_0}{t} ,
\]
gerando funĂ§Ă”es das sequĂȘncias de subprobabilidade
$(p^{(n)}_{j+1})_j $ (massa total $\leq 1$, que Ă© tudo
$ k = 0$ argumento usado). Para fixo $ t \in \intoo{0}{1}$, $ g_n(t)
\to g(t)$ por hipĂłtese e caso $ k = 0$. Aplicando o $ k =
0$ argumento para $ g_n $ dĂĄ $ p_1^{(n)} \to p_1$; iterando o
mudanĂ§a $ k $ vezes dĂĄ $ p_k^{(n)} \to p_k $ para cada $ k $. (Isso Ă©
a instĂąncia discreta e elementar de LĂ©vy continuidade teorema,
cuja forma geral --- para funĂ§Ă”es caracterĂ­sticas --- Ă© um ano
3 marco.)
\end{solution}

\begin{solution}{pb:b2:genfun:1}
\textbf{1.} Para $ t \in \intoo01$, a regra da cadeia em $ G_n = G
\circ G_{n-1}$ dĂĄ $ G_n'(t) =
G'\bigl(G_{n-1}(t)\bigr)G_{n-1}'(t)$. Como $ t \to 1^-$,
$ G_{n-1}(t) \uparrow 1$, e $ G'$ nĂŁo Ă© decrescente com
limite esquerdo $ m $ no $1$, entĂŁo o primeiro fator tende a $ m $; por
induĂ§ĂŁo o segundo tende a $ m^{n-1}$. Por
\cref{thm:b2:genfun:moments}, $\E(Z_n) = G_n'(1^-) = m^n $.

\textbf{2.} Diferenciando mais uma vez,
\[
G_n'' = G''(G_{n-1})\,(G_{n-1}')^2 +
G'(G_{n-1})\,G_{n-1}'',
\]
e deixando $ t \to 1^-$: $ a_n = G''(1)m^{2(n-1)} + m\,
a_{n-1}$ com $ a_n = G_n''(1)$, $ a_1 = G''(1)$. Para $ m \neq
1$ verifica-se por induĂ§ĂŁo que $ a_n = G''(1)\,m^{n-1}
\frac{m^n - 1}{m - 1}$ (a recursĂŁo adiciona $ G''(1)m^{2n-2}$
para $ m\cdot G''(1)m^{n-2}\frac{m^{n-1}-1}{m-1}$, e
$ m^{n-1} + \frac{m^{n-1}-1}{m-1} = \frac{m^n - 1}{m-1}$);
para $ m = 1$, $ a_n = a_{n-1} + G''(1) = n\,G''(1)$.

\textbf{3.} $ V(Z_n) = a_n + m^n - m^{2n}$ e $ G''(1) =
\sigma^2 + m^2 - m $. Para $ m \neq 1$, a peĂ§a $(m^2 -
m)m^{n-1}\frac{m^n-1}{m-1} = m^n(m^n - 1)$ cancela $ m^n -
m^{2n}$ exatamente, deixando $ V(Z_n) =
\sigma^2m^{n-1}\frac{m^n-1}{m-1}$. Para $ m = 1$: $ V(Z_n) =
nG''(1) = n\sigma^2$.

\textbf{4.} $ Z_n $ Ă© uma variĂĄvel inteira nĂŁo negativa, entĂŁo
$\P(Z_n > 0) = \P(Z_n \geq 1) \leq \E(Z_n) = m^n $ por Markov
(\cref{thm:b2:randomvar:markov}). Para $ m < 1$ isso decai
geometricamente --- e sumariamente, entĂŁo Borel-Cantelli atĂ© dĂĄ
que apenas um nĂșmero finito de geraĂ§Ă”es nĂŁo Ă© vazia, o que Ă©
extinĂ§ĂŁo novamente.

\textbf{5.} Cauchy-Schwarz: $\E(Z_n)^2 = \E(Z_n\mathbf
1_{Z_n>0})^2 \leq \E(Z_n^2)\,\P(Z_n > 0)$. Com a pergunta 3
e $ m < 1$:
\[
\E(Z_n^2) = V(Z_n) + m^{2n}
\leq \frac{\sigma^2m^{n-1}}{1-m} + m^{2n},
\]
entĂŁo, dividindo $ m^{2n}$ por este limite e simplificando por
$ m^n $,
\[
\P(Z_n > 0) \geq \frac{m^n}{\frac{\sigma^2}{m(1-m)} + m^n}
\geq \Bigl(\frac{\sigma^2}{m(1-m)} + 1\Bigr)^{-1}m^n ,
\]
usando $ m^n \leq 1$ no denominador. Com a pergunta 4:
$\P(Z_n > 0) \asymp m^n $.

\textbf{6.} $ G(t) = q\sum_k(pt)^k = \frac{q}{1 - pt}$, e $ m
= G'(1) = \frac{pq}{(1-p)^2} = \frac pq $. SubcrĂ­tico para $ p
< \frac12$, crĂ­tico para $ p = \frac12$, supercrĂ­tico para $ p
> \frac12$.

\textbf{7.} $ G(t) = t $ lĂȘ $ pt^2 - t + q = 0$, com raĂ­zes
$\frac{1 \pm \abs{p - q}}{2p}$, ou seja\ $1$ e $\frac qp =
\frac1m $. A probabilidade de extinĂ§ĂŁo Ă© a menor fixa
apontar $\intcc01$ (\cref{thm:b2:genfun:extinction}):
$ q_{\mathrm{ext}} = 1$ se $ m \leq 1$, e $\frac1m $ se $ m >
1$.

\textbf{8.} Para $ m \neq 1$, com $ p = \frac m{m+1}$, $ q =
\frac1{m+1}$: se $ q_n = \frac{m^n - 1}{m^{n+1} - 1}$, entĂŁo
\[
1 - p\,q_n = \frac{(m+1)(m^{n+1} - 1) - m(m^n - 1)}
{(m+1)(m^{n+1} - 1)} = \frac{m^{n+2} - 1}{(m+1)(m^{n+1} -
1)},
\]
entĂŁo $ q_{n+1} = \frac{q}{1 - pq_n} = \frac{m^{n+1} -
1}{m^{n+2} - 1}$; o caso base $ q_0 = 0$ segura. Para $ m = 1$:
$ G(t) = \frac1{2 - t}$ e $ q_{n+1} = \frac1{2 -
\frac{n}{n+1}} = \frac{n+1}{n+2}$, com $ q_0 = 0$.

\textbf{9.} $1 - q_n = \frac{m^n(m - 1)}{m^{n+1} - 1}$. Para
$ m < 1$ o denominador tende a $-1$: $1 - q_n \sim (1 -
m)\,m^n $. Para $ m > 1$:
\[
q_{\mathrm{ext}} - q_n = \frac1m - \frac{m^n - 1}{m^{n+1} -
1} = \frac{m - 1}{m\,(m^{n+1} - 1)} \sim \frac{m -
1}{m^{2}}\;m^{-n} .
\]
E $ G'(t) = \frac{pq}{(1 - pt)^2}$ avaliado em $ t = \frac
qp $ (onde $1 - pt = 1 - q = p $) dĂĄ $ G'(q_{\mathrm{ext}})
= \frac qp = \frac1m $: a proporĂ§ĂŁo observada $ m^{-1}$ Ă© exatamente
a derivada no ponto fixo de atraĂ§ĂŁo.

\textbf{10.} Para $ p = \frac12$: $ G''(t) = \frac{1/4}{(1 -
t/2)^3}$, entĂŁo $ G''(1) = 2$ e $\sigma^2 = G''(1) + m - m^2 =
