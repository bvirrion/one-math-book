\section*{Capítulo \ref{ch:b2:series} --- Sequências e Séries}

\begin{solution}{exo:b2:series:1}
$\sum\frac{\cos n}{n}$: Teste de Abel com $ a_n = \frac1n $ e $ b_n =
\cos n $, cujas somas parciais são limitadas (parte real de uma geometria
soma, como em \cref{ex:b2:series:sinn}): convergente (não \omterm{def:b2:series:def}{absolutamente},
pelo mesmo $\cos^2$ truque).

$\sum \frac{(-1)^n}{\ln n}$ ($ n \geq 2$): teste alternado,
$\frac{1}{\ln n} \downarrow 0$: convergente; não \omterm{def:b2:series:def}{absolutamente} ($\ln n
\leq n $).

$\sum \frac{(-1)^n}{n^{3/4} + \cos n}$: expandir,
\[
\frac{(-1)^n}{n^{3/4} + \cos n}
= \frac{(-1)^n}{n^{3/4}}\cdot
\frac{1}{1 + \frac{\cos n}{n^{3/4}}}
= \frac{(-1)^n}{n^{3/4}}
- \frac{(-1)^n\cos n}{n^{3/2}} + O\Bigl(\frac{1}{n^{9/4}}\Bigr).
\]
Primeira série: alternada, convergente. Segundo: \omterm{def:b2:series:def}{absolutamente}
convergente ($\frac{1}{n^{3/2}}$ escala). Terceiro: \omterm{def:b2:series:def}{absolutamente}
convergente. Total: convergente.
\end{solution}

\begin{solution}{exo:b2:series:2}
Produto Cauchy de $\sum_{m\geq0} z^m $ consigo mesmo (ambos \omterm{def:b2:series:def}{absolutamente}
convergente para $\abs z < 1$): o coeficiente diagonal é $ c_k =
\sum_{m=0}^{k} 1 = k + 1$, então
\[
\frac{1}{(1-z)^2} = \sum_{k\geq0} (k+1)z^k .
\]
Multiplicando por $ z $ e reindexação: $\sum_{n \geq 1} n z^n =
\frac{z}{(1-z)^2}$.
\end{solution}

\begin{solution}{exo:b2:series:3}
Deixar $ B_n = \sum_{k \leq n} b_k \to B $. \omterm{thm:b2:series:abel}{Resumo de Abel} com $ a_k = k $:
\[
\sum_{k=1}^{n} k\,b_k = n B_n - \sum_{k=1}^{n-1} B_k
\quad\Longrightarrow\quad
\frac1n \sum_{k=1}^{n} k b_k = B_n - \frac{1}{n}\sum_{k=1}^{n-1}
B_k .
\]
O Cesàro meio do convergente $(B_k)$ tende ao seu limite $ B $
(volume do ano 1), então o lado direito tende a $ B - B = 0$.
\end{solution}

\begin{solution}{exo:b2:series:4}
Se $\theta \in \pi\Z $: todos os termos desaparecem --- convergem trivialmente.
Assumir $\theta \notin \pi\Z $.

$\alpha > 1$: \omterm{def:b2:series:def}{absolutamente} convergente (dominação por $ n^{-\alpha}$).

$0 < \alpha \leq 1$: O teste de Abel se aplica ($ a_n = n^{-\alpha}
\downarrow 0$; somas parciais de $\sin n\theta $ limitado por
$\frac{1}{\abs{\sin(\theta/2)}}$, soma geométrica): convergente. Não
\omterm{def:b2:series:def}{absolutamente}: $\abs{\sin n\theta} \geq \sin^2 n\theta = \frac{1 -
\cos 2n\theta}{2}$, e $\sum \frac{1 - \cos 2n\theta}{2n^\alpha}$
diverge ($\sum n^{-\alpha}$ diverge; $\sum
\frac{\cos 2n\theta}{n^\alpha}$ converge por Abel quando $2\theta
\notin 2\pi\Z $; o caso excluído $2\theta \in 2\pi\Z $ significa
$\theta \in \pi\Z $, já tratado). Semi-convergente.
\end{solution}

\begin{solution}{exo:b2:series:5}
Deixar $ p_1, p_2, \dots $ sejam os termos não negativos de $(u_n)$ em ordem,
$ q_1, q_2, \dots $ os negativos. Ambos $\sum p_k $ e $\sum
q_k $ divergem: se um deles convergisse, o outro seria igual ao
convergente $\sum u_n $ menos isso, portanto converge também --- e então
$\sum \abs{u_n} = \sum p_k - \sum q_k $ convergiria, contradizendo
a hipótese. Também $ u_n \to 0$ ($\sum u_n $ converge).

Rearranjo ganancioso: aceite termos positivos $ p_1, p_2, \dots $ até
o total acumulado primeiro excede $\ell $ (possível: $\sum p_k =
+\infty $); depois termos negativos até que o total primeiro caia abaixo
$\ell $ (possível: $\sum q_k = -\infty $); repetir para sempre (cada
fase é finita e cada termo é usado exatamente uma vez: um verdadeiro
rearranjo). Após cada mudança, a distância do corredor
total para $\ell $ é no máximo o último termo utilizado; já que os termos usados
no $ m $-th switch tem índice $\to \infty $, e $ u_n \to 0$,
os totais correntes convergem para $\ell $.
\end{solution}

\begin{solution}{exo:b2:series:6}
Somabilidade: as somas parciais finitas de
$\frac{\abs x^{m+n}}{m!n!}$ são limitados por $\bigl(\sum_m
\frac{\abs x^m}{m!}\bigr)^2 = \eu^{2\abs x}$. Agrupamento diagonal
(\cref{thm:b2:series:fubini}):
\[
(\eu^{x})^2 = \sum_{m,n} \frac{x^{m+n}}{m!\,n!}
= \sum_{k=0}^{\infty} x^k \sum_{m+n=k} \frac{1}{m!\,n!}
= \sum_k \frac{x^k}{k!}\sum_{m=0}^{k}\binom km
= \sum_k \frac{(2x)^k}{k!} = \eu^{2x} .
\]
\end{solution}

\begin{solution}{exo:b2:series:7}
Somabilidade: delimitado por $\zeta(2)^2$ como uma família de produtos
(\cref{thm:b2:series:fubini}argumento do produto de Cauchy). O mapa
$(q, a, b) \mapsto (qa, qb)$, de triplos com $\gcd(a, b) = 1$
para pares $(m, n)$, é uma bijeção (conjunto $ q = \gcd(m,n)$). Agrupamento
o família resumida correspondentemente (uma partição do conjunto de índices ---
legítimo para resumido famílias por
\cref{thm:b2:series:rearrangement}/\cref{thm:b2:series:fubini}
aplicado à partição em \omterm{def:b2:structures:countable}{contável} muitas aulas):
\[
\zeta(2)^2 = \sum_{q} \sum_{\gcd(a,b)=1} \frac{1}{q^4 a^2 b^2}
= \zeta(4)\, S,
\qquad\text{so}\qquad
S = \frac{\zeta(2)^2}{\zeta(4)} .
\]
(Com os valores $\zeta(2) = \frac{\pi^2}{6}$,
$\zeta(4) = \frac{\pi^4}{90}$ de
\cref{ch:b2:fourier}: $ S = \frac{5}{2}$.)
\end{solution}

\begin{solution}{exo:b2:series:8}
Deixar $ A = \sum a_n $ (absoluto), $ B_m = \sum_{k\leq m} b_k \to B $,
limitado por $ M $. Então
\[
C_N = \sum_{k=0}^{N} c_k = \sum_{n=0}^{N} a_n B_{N-n}
\]
(coletar pelo índice de $ a $). Escrever
\[
C_N - AB = \sum_{n=0}^{N} a_n (B_{N-n} - B) - B\sum_{n > N} a_n .
\]
O último termo tende a $0$. Divida a soma em $ n = \lfloor N/2
\rfloor $: para $ n \leq N/2$, $ N - n \geq N/2$, então $\abs{B_{N-n} - B}
\leq \varepsilon_N := \sup_{m \geq N/2}\abs{B_m - B} \to 0$ e
esta parte é $\leq \varepsilon_N \sum\abs{a_n}$; para $ n > N/2$,
$\abs{B_{N-n} - B} \leq 2M $, e esta parte é $\leq 2M \sum_{n >
N/2} \abs{a_n} \to 0$. Por isso $ C_N \to AB $.
\end{solution}

\begin{solution}{exo:b2:series:9}
Pegar $ u_n = 2^{-n} e_n $ ($ e_n $ a sequência com um único $1$ em
posição $ n $). Então $\sum \norm{u_n}_\infty = \sum 2^{-n} <
\infty $: \omterm{def:b2:series:def}{absolutamente} convergente. Mas as somas parciais $ S_N =
(1, \tfrac12, \dots, 2^{-N}, 0, \dots)$ teria que convergir para
a sequência $(2^{-n})_n $, que é \emph{não} eventualmente zero:
fora $ E $. Dentro $ E $, $(S_N)$ é Cauchy sem limite ($\norm{S_N
- x}_\infty \geq 2^{-N-1}$ falha em ajudar qualquer eventual-zero $ x $:
para qualquer $ x \in E $ desaparecendo além da classificação $ K $, $\norm{S_N - x} \geq
2^{-K-1}$ para $ N > K $): a série não converge em $ E $.
Completude é exatamente o que \cref{thm:b2:nvs:absoluteconvergence}
precisa.
\end{solution}

\begin{solution}{exo:b2:series:10}
Ambos $\arctan(n+1) - \arctan n $ e
$\arctan\frac{1}{n^2+n+1}$ deitar em $\intoo{0}{\frac\pi2}$, e o
fórmula de adição tangente dá
\[
\tan\bigl(\arctan(n{+}1) - \arctan n\bigr)
= \frac{(n+1) - n}{1 + n(n+1)} = \frac{1}{n^2 + n + 1} :
\]
tangentes iguais em um intervalo onde $\tan $ é injetivo, então o
identidade mantém. Telescópico,
\[
\sum_{n=1}^{N}\arctan\frac{1}{n^2+n+1}
= \arctan(N{+}1) - \arctan 1 \xrightarrow[N\to\infty]{}
\frac\pi2 - \frac\pi4 = \frac\pi4 .
\]

\end{solution}

\begin{solution}{exo:b2:series:11}
Primeiro: $\sqrt{n+1} - \sqrt n = \frac{1}{\sqrt{n+1} + \sqrt n}
\sim \frac{1}{2\sqrt n}$, então os termos são $\sim
2^{-\alpha}n^{-\alpha/2}$: convergência se sef $\frac\alpha2 > 1$,
ou seja\ $\alpha > 2$. Segundo: $1 - \cos\frac1n \sim
\frac{1}{2n^2}$: converge. Terceiro: $\bigl(1 + \frac1n\bigr)^n =
\eu^{\,n\ln(1 + 1/n)} = \eu^{\,1 - \frac1{2n} + O(n^{-2})} =
\eu\bigl(1 - \frac{1}{2n} + O(n^{-2})\bigr)$, então
\[
\eu - \Bigl(1 + \frac1n\Bigr)^{\!n} \sim \frac{\eu}{2n} :
\]
termos positivos equivalentes a um múltiplo harmônico: diverge.
\end{solution}

\begin{solution}{exo:b2:series:12}
Pela monotonicidade, $ n\,a_{2n} \leq a_{n+1} + a_{n+2} + \dots +
a_{2n} = S_{2n} - S_n \to 0$ (Critério de Cauchy para o
série convergente). Por isso $2n\,a_{2n} \to 0$, e $(2n{+}1)\,
a_{2n+1} \leq (2n{+}1)a_{2n} = \frac{2n+1}{2n}\,(2n\,a_{2n}) \to
0$: ambas as subsequências de $(na_n)$ tendem a $0$, então $ na_n \to 0$.

\emph{A conversa falha:} $ a_n = \frac1{n\ln n}$ é positivo
diminuindo com $ na_n = \frac1{\ln n} \to 0$, ainda $\sum a_n $
diverge (fronteira de Bertrand, \cref{pb:b2:series:1}, pergunta
18). \emph{Monotonicidade essencial:} deixar $ a_n = \frac1n $ quando $ n $
é um poder de $2$ e $ a_n = 2^{-n}$ de outra forma: $\sum a_n \leq
\sum_k 2^{-k} + \sum_n 2^{-n} < \infty $, mas $ na_n = 1$ junto
os poderes de $2$: $ na_n \not\to 0$.
\end{solution}

\begin{solution}{pb:b2:series:1}
\textbf{1.} Indução ativada $ m $: para $ m = 0$ ambos os lados estão $1$;
o passo multiplica por $\cos\theta + \iu\sin\theta $ e usa o
fórmulas de adição $\cos(m\theta + \theta) =
\cos m\theta\cos\theta - \sin m\theta\sin\theta $, $\sin(m\theta
+ \theta) = \sin m\theta\cos\theta + \cos m\theta\sin\theta $.
Expandindo em vez disso pelo teorema binomial com $ m = 2n+1$ e
coletando a parte imaginária (os poderes ímpares de
$\iu\sin\theta $, com $\iu^{2j+1} = (-1)^j\iu $):
\[
\sin\bigl((2n{+}1)\theta\bigr) = \sum_{j=0}^{n}(-1)^j
\binom{2n+1}{2j+1}\cos^{2(n-j)}\theta\,\sin^{2j+1}\theta .
\]

\textbf{2.} Sobre $\intoo0{\frac\pi2}$, $\sin\theta \neq 0$: fator
$\sin^{2n+1}\theta $ de cada termo, deixando
$\bigl(\frac{\cos^2\theta}{\sin^2\theta}\bigr)^{n-j} =
(\cot^2\theta)^{n-j}$: a identidade exibida com $ P_n(x) =
\sum_j(-1)^j\binom{2n+1}{2j+1}x^{n-j}$. Seu grau-$ n $
coeficiente é $ j = 0$ de $\binom{2n+1}{1} = 2n + 1 \neq 0$.

\textbf{3.} No $\theta_k = \frac{k\pi}{2n+1}$:
$\sin\bigl((2n{+}1)\theta_k\bigr) = \sin k\pi = 0$ enquanto
$\sin^{2n+1}\theta_k \neq 0$, então $ P_n(\cot^2\theta_k) = 0$. O
$\theta_k $ aumentar estritamente em $\intoo0{\frac\pi2}$, onde
$\cot^2$ é estritamente decrescente: os valores $ x_k =
\cot^2\theta_k $ são pares distintos --- $ n $ raízes distintas de
um diploma-$ n $ polinômio, portanto, todos eles.

\textbf{4.} Vieta: a soma das raízes é menos a razão de
o $ x^{n-1}$- e $ x^n $-coeficientes:
\[
\sum_{k=1}^{n}\cot^2\theta_k =
\frac{\binom{2n+1}{3}}{\binom{2n+1}{1}}
= \frac{(2n+1)(2n)(2n-1)/6}{2n+1} = \frac{n(2n-1)}{3}.
\]

\textbf{5.} $\frac{1}{\sin^2\theta} = 1 + \cot^2\theta $:
resumindo, $ n + \frac{n(2n-1)}3 = \frac{3n + 2n^2 - n}{3} =
\frac{2n(n+1)}{3}$.

\textbf{6.} Sobre $\intoo{0}{\frac\pi2}$: $\sin\theta < \theta <
\tan\theta $ (Volume do ano 1). Tomando inversões recíprocas:
$\cot\theta < \frac1\theta < \frac1{\sin\theta}$ e quadratura
(todos positivos) dá $\cot^2\theta < \frac1{\theta^2} <
\frac1{\sin^2\theta}$.

\textbf{7.} Soma a questão 6 em $\theta = \theta_k $ sobre $ k \leq
n $, usando as questões 4 e 5, e $\frac1{\theta_k^2} =
\frac{(2n+1)^2}{k^2\pi^2}$:
\[
\frac{n(2n-1)}3 < \frac{(2n+1)^2}{\pi^2}\sum_{k=1}^n\frac1{k^2}
< \frac{2n(n+1)}3 .
\]

\textbf{8.} Multiplicar por $\frac{\pi^2}{(2n+1)^2}$:
\[
\frac{\pi^2}{3}\cdot\frac{n(2n-1)}{(2n+1)^2}
< \sum_{k=1}^{n}\frac1{k^2} <
\frac{\pi^2}{3}\cdot\frac{2n(n+1)}{(2n+1)^2}.
\]
Ambos os limites tendem a $\frac{\pi^2}3\cdot\frac12 = \frac{\pi^2}6$
(as frações racionais tendem a $\frac12$). As somas parciais
aumentam, então eles convergem, e o aperto dá $\zeta(2) =
\frac{\pi^2}6$: Teorema de Euler, pela prova de Cauchy.

\textbf{9.} As somas parciais aumentam para $\zeta(2) =
\frac{\pi^2}6$, então $0 \leq \frac{\pi^2}6 - \sum_{k\leq n}k^{-2}$;
e o limite inferior da questão 8 dá
\[
\frac{\pi^2}6 - \sum_{k\leq n}\frac1{k^2}
\leq \frac{\pi^2}6 - \frac{\pi^2}3\cdot\frac{n(2n-1)}{(2n+1)^2}
= \frac{\pi^2}6\cdot\frac{(2n+1)^2 - (4n^2 -
2n)}{(2n+1)^2}
= \frac{\pi^2}6\cdot\frac{6n + 1}{(2n+1)^2}
= O\Bigl(\frac1n\Bigr),
\]
combinando com a cauda exata $\frac1n - \frac1{2n^2} + O(n^{-3})$ de
\cref{exo:b2:comparison:11}.

\textbf{10.} A segunda função simétrica elementar do
raízes é $\sigma_2 = \frac{\binom{2n+1}5}{\binom{2n+1}1} =
\frac{(2n)(2n-1)(2n-2)(2n-3)}{120}$, então
\[
\sum_k\cot^4\theta_k = \sigma_1^2 - 2\sigma_2
= \Bigl(\frac{n(2n-1)}3\Bigr)^2 -
\frac{2n(2n-1)(2n-2)(2n-3)}{60}
\sim \frac{4n^4}9 - \frac{4n^4}{15} = \frac{8n^4}{45}.
\]
Apertando $\cot^4\theta < \theta^{-4} < (1 + \cot^2\theta)^2 =
1 + 2\cot^2\theta + \cot^4\theta $ e soma: ambas as somas externas
são $\frac{8n^4}{45}(1 + o(1))$ (o adicionado $ n + 2\sigma_1 =
O(n^2)$ é insignificante), enquanto o meio é
$\frac{(2n+1)^4}{\pi^4}\sum_{k\leq n}k^{-4}$. Por isso
\[
\sum_{k\leq n}\frac1{k^4} \longrightarrow
\pi^4\cdot\frac{8/45}{16} = \frac{\pi^4}{90}.
\]

\textbf{11.} Divisão $\zeta(2)$ sobre paridades: $\sum_{\text{even}}
= \sum_j\frac{1}{(2j)^2} = \frac14\zeta(2) = \frac{\pi^2}{24}$,
então $\sum_{\text{odd}} = \zeta(2) - \frac{\pi^2}{24} =
\frac{\pi^2}8$. \omterm{def:b2:linalg:alternating}{Alternando}: $\sum_k\frac{(-1)^{k-1}}{k^2} =
\sum_{\text{odd}} - \sum_{\text{even}} = \frac{\pi^2}8 -
\frac{\pi^2}{24} = \frac{\pi^2}{12}$ (convergência absoluta
justifica o reagrupamento,
\cref{thm:b2:series:rearrangement}).

\textbf{12.} $ S = \frac{\zeta(2)^2}{\zeta(4)} =
\frac{(\pi^2/6)^2}{\pi^4/90} = \frac{90}{36} = \frac52$.
Heurística: a identidade $\zeta(2)^2 = \zeta(4)S $ de
\cref{exo:b2:series:7} diz que retirar o $\gcd $
renormaliza pares em pares coprimos; o recíproco
$\frac1{\zeta(2)} = \frac6{\pi^2} \approx 0.608$ é o natural
candidato para a densidade de pares coprimos entre todos os pares ---
uma declaração sobre $\lim_N \frac{1}{N^2}\#\{(m,n) \leq N :
\gcd = 1\}$ cuja prova honesta (com termos de erro) pertence a
o volume do Ano 3.

\textbf{13.} A soma direta apresenta erro $\sim \frac1n $: seis
dígitos exigem cerca de $10^6$ termos. A soma corrigida
$\sum_{k\leq n}k^{-2} + \frac1n - \frac1{2n^2}$ tem erro
$ O(n^{-3})$: no $ n = 100$ é igual $1.6449339\dots $ contra
$\frac{\pi^2}6 = 1.6449341\dots $ --- erro $1.7\cdot10^{-7}$,
sete dígitos de cem termos. Batidas de correções assintóticas
paciência bruta em quatro ordens de magnitude.

\textbf{14.} $ n = 1$: $ P_1(x) = 3x - 1$, raiz $\frac13$ e
de fato $\cot^2\frac\pi3 = \bigl(\frac1{\sqrt3}\bigr)^2 =
\frac13 = \frac{1\cdot1}3$. $ n = 2$: a fórmula prevê
$\frac{2\cdot3}3 = 2$; com $\cos\frac\pi5 = \frac{1 +
\sqrt5}{4}$, calcula-se $\cot^2 36^\circ \approx 1.894$ e
$\cot^2 72^\circ \approx 0.106$: soma $2.000$.

\textbf{15.} Os números $\tan^2\theta_k = \frac1{x_k}$ são os
raízes de $ Q(x) = x^nP_n\bigl(\frac1x\bigr) =
\sum_{j=0}^n(-1)^j\binom{2n+1}{2j+1}x^j $ (o $ x_k $ são
diferente de zero). Vieta em $ Q $: o coeficiente líder é $(-1)^n $
(prazo $ j = n $), o próximo é $(-1)^{n-1}\binom{2n+1}{2n-1} =
(-1)^{n-1}\binom{2n+1}{2}$, então
\[
\sum_{k=1}^n\tan^2\theta_k =
-\frac{(-1)^{n-1}\binom{2n+1}2}{(-1)^n} = \binom{2n+1}2\cdot
\frac{2}{2n+1}\cdot\frac{2n+1}{2}
= n(2n+1).
\]
Verificar $ n = 1$: $\tan^2\frac\pi3 = 3 = 1\cdot3$.

\textbf{16.} Deixar $\gamma = \frac{1+\alpha}2 \in
\intoo{1}{\alpha}$. Então $\frac{n^{-\alpha}(\ln
n)^{-\beta}}{n^{-\gamma}} = n^{\gamma - \alpha}(\ln n)^{-\beta}
\to 0$ (uma potência negativa de $ n $ supera qualquer poder de $\ln n $), então
eventualmente os termos são $\leq n^{-\gamma}$ com $\gamma > 1$:
convergência por comparação com uma série de Riemann.

\textbf{17.} Deixar $\gamma = \frac{1+\alpha}2 \in
\intoo{\alpha}{1}$: agora $\frac{n^{-\gamma}}{n^{-\alpha}(\ln
n)^{-\beta}} = n^{\alpha-\gamma}(\ln n)^{\beta} \to 0$, então
eventualmente os termos são $\geq n^{-\gamma}$ com $\gamma < 1$:
divergência.

\textbf{18.} $ f(t) = \frac1{t(\ln t)^\beta}$ é positivo,
\omterm{def:b2:metric:continuity}{contínuo}, e diminuindo para grandes $ t $ (seu logaritmo tem
derivada $-\frac1t\bigl(1 + \frac{\beta}{\ln t}\bigr) < 0$
eventualmente). Antiderivadas: para $\beta \neq 1$,
$\int^x f = \frac{(\ln x)^{1-\beta}}{1-\beta} + \text{const}$,
que tem um limite finito se $\beta > 1$; para $\beta = 1$,
$\int^x f = \ln\ln x \to \infty $. Por
\cref{thm:b2:comparison:seriesintegral}, a série e o
integral compartilham sua natureza: convergência se $\beta > 1$.

\textbf{19.} Mesmo teste: $\frac{\dd}{\dd t}\ln\ln\ln t =
\frac{1}{t\ln t\,\ln\ln t}$, e $\ln\ln\ln t \to \infty $:
divergência. E $\frac{\dd}{\dd t}\Bigl(-\frac1{\ln\ln t}\Bigr)
= \frac{1}{t\ln t\,(\ln\ln t)^2}$ com $-\frac1{\ln\ln t} \to
0$: convergência.

\textbf{20.} Primeiro: $ n^{1/\ln n} = \eu^{\ln n/\ln n} = \eu $, então
os termos são exatamente $\frac{1}{\eu\,n}$: um múltiplo do
série harmônica, \emph{divergente} --- o expoente $1 +
\frac1{\ln n}$ rasteja para $1$ muito rápido. Segundo: $ n^{1/\ln\ln n}
= \eu^{\ln n/\ln\ln n}$, e $\frac{\ln n}{\ln\ln n} \geq
2\ln\ln n $ eventualmente, então $ n^{1/\ln\ln n} \geq (\ln n)^2$: o
termos são $\leq \frac1{n(\ln n)^2}$, um Bertrand convergente
série (questão 18): \emph{convergente}. A fronteira passa
estritamente entre esses dois expoentes.

\textbf{21.} $ R_n \downarrow 0$ e
\[
\sqrt{R_n} - \sqrt{R_{n+1}} = \frac{R_n -
R_{n+1}}{\sqrt{R_n} + \sqrt{R_{n+1}}} =
\frac{a_n}{\sqrt{R_n} + \sqrt{R_{n+1}}} \geq
\frac{a_n}{2\sqrt{R_n}},
\]
então $\sum_n \frac{a_n}{\sqrt{R_n}} \leq 2\sum_n(\sqrt{R_n} -
\sqrt{R_{n+1}}) = 2\sqrt{R_1} < \infty $ (telescópio). Ainda
$\frac{a_n/\sqrt{R_n}}{a_n} = \frac1{\sqrt{R_n}} \to \infty $:
a nova série converge embora seja infinitamente maior. Não
a série convergente é mais lenta; testes de comparação com qualquer
a família fixa nunca pode ser completa.

\textbf{22.} Para diminuir positivo $(a_n)$, agrupe os termos
entre poderes consecutivos de $2$:
\[
2^{k}a_{2^{k+1}} \leq \sum_{n=2^k}^{2^{k+1}-1} a_n \leq
2^ka_{2^k} .
\]
Resumindo $ k $: se $\sum 2^ka_{2^k}$ converge, a parcial
somas de $\sum a_n $ são limitados (converge); se $\sum a_n $
converge, então $\sum_k 2^{k+1}a_{2^{k+1}} \leq 2\sum_n a_n <
\infty $. Para $ a_n = \frac1{n(\ln n)^\beta}$: $2^ka_{2^k} =
\frac{1}{(k\ln 2)^\beta}$, e $\sum k^{-\beta}$ converge se e se
$\beta > 1$: a fronteira da questão 18 novamente, sem
integrais.

\textbf{23.} Pela comparação integral,
\[
\sum_{n > N}\frac{1}{n(\ln n)^2} \geq
\int_{N+1}^{\infty}\frac{\dd t}{t(\ln t)^2} =
\frac{1}{\ln(N+1)},
\]
que em $ N = 10^6$ é $\approx 0.0724$: depois de um milhão de termos
a cauda ainda excede $0.07$ --- a série converge, mas não
a soma direta sempre exibirá sua soma. Compare com
questão 13, onde uma correção assintótica comprou sete
dígitos de cem termos: saber \emph{como} uma série
converge vale mais do que saber que converge.

\textbf{24.} $\sum\frac1{n\ln n}$: diverge ($\alpha = 1$,
$\beta = 1$, questão 18). $\sum\frac1{n^{1.01}}$: converge
(Riemann, $\alpha > 1$). $\sum\frac{(\ln n)^{100}}{n^{1.001}}$:
converge ($\alpha = 1.001 > 1$, $\beta = -100$, questão 16).
$\sum\frac1{n\ln n(\ln\ln n)^3}$: converge (questão 19
padrão: antiderivada $-\frac12(\ln\ln t)^{-2}$, finito
limite).

\textbf{25.} De Moivre converte o desaparecimento de $\sin(2n{+}1)
\theta_k $ no desaparecimento de um polinômio em
$\cot^2\theta_k $, e Vieta lê as raízes exatas que
a análise por si só poderia apenas estimar (questões 1 a 5). O
apertar precisava do \emph{exato} valores $\frac{n(2n-1)}3$ e
$\frac{2n(n+1)}3$ em ambos os lados --- equivalentes teriam
imploramos a pergunta, já que a questão toda é a constante
$\frac{\pi^2}6$ (questões 7 a 8). A Parte IV foi executada inteiramente em
\cref{ch:b2:comparison}série de - comparação integral, o
antiderivadas logarítmicas fazendo a classificação (questões
18-19). A questão 21 destrói o sonho de um universal
teste de comparação: abaixo de cada série convergente encontra-se outra,
infinitamente mais lento --- escalas como o mapa de Bertrand da fronteira
cada vez mais finamente, mas nunca alcançá-lo. Cúpulas: Euler $\zeta(2)
= \frac{\pi^2}6$ com seu andar superior $\zeta(4) =
\frac{\pi^4}{90}$ (questões 8, 10), e o Bertrand
classificação (questões 16 a 18); $\zeta(2)$ retorna em
\cref{ch:b2:fourier}, onde a identidade de Parseval o prova novamente
em uma linha da série de Fourier do dente de serra --- um
constante, duas civilizações.
\end{solution}
