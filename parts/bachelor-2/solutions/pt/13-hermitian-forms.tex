\section*{Capítulo \ref{ch:b2:hermitian} --- Formas Hermitianas}

\begin{solution}{exo:b2:hermitian:1}
$\langle x, y\rangle = \conj{1}\cdot\iu + \conj{\iu}\cdot 1 = \iu -
\iu = 0$: ortogonal. $\norm x = \norm y = \sqrt{1 + 1} = \sqrt2$.
Base ortonormal: $\bigl(\frac{1}{\sqrt2}(1, \iu),\;
\frac{1}{\sqrt2}(\iu, 1)\bigr)$ --- o próprio par normalizado.
\end{solution}

\begin{solution}{exo:b2:hermitian:2}
Primeiro: é igual ao seu conjugado transpor (diagonal real, $\conj{\iu} =
-\iu $ trocado): \omterm{def:b2:hermitian:adjoint}{Hermitiano} (portanto normal); não unitário ($ A^\dagger A
\neq I $: colunas não unidade).

Segundo: $ A^\dagger A = \frac12\begin{pmatrix} 1 & -\iu\\ -\iu &
1\end{pmatrix}\begin{pmatrix} 1 & \iu\\ \iu & 1\end{pmatrix} =
\frac12\begin{pmatrix} 2 & 0\\ 0 & 2\end{pmatrix} = I $: unitário
(portanto normal); não \omterm{def:b2:hermitian:adjoint}{Hermitiano}.

Terceiro: $ A^\dagger A = E_{22} \neq E_{11} = AA^\dagger $: não é normal
(então nem \omterm{def:b2:hermitian:adjoint}{Hermitiano} nem unitário) --- o nilpotente padrão
contra-exemplo.
\end{solution}

\begin{solution}{exo:b2:hermitian:3}
Singularidade: $ A = H + \iu K $ com $ H^\dagger = H $, $ K^\dagger = K $
forças $ A^\dagger = H - \iu K $, então $ H = \frac{A + A^\dagger}{2}$,
$ K = \frac{A - A^\dagger}{2\iu}$; essas fórmulas são \omterm{def:b2:hermitian:adjoint}{Hermitiano}
(verificar: $\bigl(\frac{A - A^\dagger}{2\iu}\bigr)^\dagger = \frac{
A^\dagger - A}{-2\iu} = K $) e reconstruir $ A $: existência.

Normalidade: $ A^\dagger A - AA^\dagger = (H - \iu K)(H + \iu K) - (H
+ \iu K)(H - \iu K) = 2\iu(HK - KH)$: desaparece se $ HK = KH $.
\end{solution}

\begin{solution}{exo:b2:hermitian:4}
$\chi_A = (X-2)^2 - 1$: valores próprios $3$ e $1$. Autovetores, por
cálculo direto:
\[
A\begin{pmatrix}1\\ -\iu\end{pmatrix}
= \begin{pmatrix} 2 + \iu(-\iu)\\ -\iu + 2(-\iu)\end{pmatrix}
= \begin{pmatrix} 3\\ -3\iu \end{pmatrix}
= 3\begin{pmatrix}1\\ -\iu\end{pmatrix},
\qquad
A\begin{pmatrix}1\\ \iu\end{pmatrix}
= \begin{pmatrix} 2 + \iu\cdot\iu\\ -\iu + 2\iu\end{pmatrix}
= \begin{pmatrix}1\\ \iu\end{pmatrix}.
\]
Base própria ortonormal: $ u_1 = \frac{1}{\sqrt2}(1, -\iu)$
(\omterm{def:b2:reduction:eigen}{autovalor} $3$), $ u_2 = \frac{1}{\sqrt2}(1, \iu)$ (\omterm{def:b2:reduction:eigen}{autovalor}
$1$); ortogonalidade como em \cref{exo:b2:hermitian:1}. Poderes, através
as projeções espectrais $ A^k = 3^k u_1u_1^\dagger + 1^k\,
u_2u_2^\dagger $:
\[
A^k = U\begin{pmatrix} 3^k & 0\\ 0 & 1\end{pmatrix}U^\dagger
= \frac{3^k}{2}\begin{pmatrix} 1 & \iu\\ -\iu & 1\end{pmatrix}
+ \frac{1}{2}\begin{pmatrix} 1 & -\iu\\ \iu & 1\end{pmatrix}
= \frac12\begin{pmatrix}
3^k + 1 & (3^k - 1)\iu\\
-(3^k-1)\iu & 3^k + 1
\end{pmatrix}.
\]
(Verificar $ k = 1$: recupera $ A $.)
\end{solution}

\begin{solution}{exo:b2:hermitian:5}
\omterm{def:b2:metric:compact}{Compactar}: fechado (pré-imagem de $ I $ sob o \omterm{def:b2:metric:continuity}{contínuo} $ U \mapsto
U^\dagger U $) e limitado (colunas são vetores unitários: entradas de
módulo $\leq 1$) em $\mathcal{M}_n(\C) \simeq \R^{2n^2}$.

\omterm{def:b2:reduction:eigen}{Autovalores}: $\operatorname{diag}(\lambda, 1, \dots, 1)$ é unitário
para qualquer $\abs\lambda = 1$. Determinante: $\abs{\det U}^2 = \det
U^\dagger \det U = \det(U^\dagger U) = 1$ (usando $\det A^\dagger =
\conj{\det A}$): $\det U $ está no círculo unitário.
\end{solution}

\begin{solution}{exo:b2:hermitian:6}
Pelo teorema espectral, trabalhe em uma autobase ortonormal de $ H $:
tudo se reduz a escalares $\lambda \in \R $ (o valores próprios).
$ H + \iu I $ tem valores próprios $\lambda + \iu \neq 0$: invertível. $ U $
tem valores próprios $\mu = \frac{\lambda - \iu}{\lambda + \iu}$, de
módulo $1$ ($\abs{\lambda - \iu} = \abs{\lambda + \iu}$ sério
$\lambda $): $ U^\dagger U = I $ mantém desde $ U $ é unitariamente
\omterm{def:b2:reduction:diag}{diagonalizável} com unimodular valores próprios (é diagonal no
base ortonormal escolhida). E $\mu = 1$ forçaria $-\iu = \iu $:
impossível, então $1 \notin \operatorname{Sp} U $. (O Cayley
transformar mapas \omterm{def:b2:hermitian:adjoint}{Hermitiano} para unitário menos um ponto --- a matriz
versão do mapa de $\R $ para o círculo.)
\end{solution}

\begin{solution}{exo:b2:hermitian:7}
Para um par próprio $ Ax = \lambda x $ ($ x \neq 0$): $\lambda\norm x^2 =
\langle x, Ax\rangle > 0$, então $\lambda > 0$ (já real,
\cref{prop:b2:hermitian:eigenvalues}). Raiz quadrada: em um espectral
base, $ B = U\operatorname{diag}(\sqrt{\lambda_i})U^\dagger $:
\omterm{def:b2:hermitian:adjoint}{Hermitiano}, definido positivo, $ B^2 = A $. Determinante: produto de
o positivo valores próprios.
\end{solution}

\begin{solution}{exo:b2:hermitian:8}
\begin{enumerate}
  \item $\norm{u(x)}^2 = \langle u(x), u(x)\rangle = \langle x,
        u^*u(x)\rangle = \langle x, uu^*(x)\rangle =
        \norm{u^*(x)}^2$. Aplicando isso ao normal $ u -
        \lambda\,\mathrm{id}$ (isso é \omterm{def:b2:quadratic:adjoint}{adjunto} é $ u^* -
        \conj\lambda $, e a normalidade é herdada): $\norm{(u -
        \lambda)x} = \norm{(u^* - \conj\lambda)x}$, então os grãos
        concordo.
  \item Para vetores próprios $ u(x) = \lambda x $, $ u(y) = \mu y $
        ($\lambda \neq \mu $): usando (1), $ u^*(x) = \conj\lambda x $;
        então
        \[
        \lambda\langle y, x\rangle = \langle y, u(x)\rangle
        = \langle u^*(y), x\rangle = \langle \conj\mu\, y, x\rangle
        = \mu \langle y, x\rangle ,
        \]
        então $\langle y, x\rangle = 0$. Estabilidade de $ E_\lambda^\perp $:
        para $ x \perp E_\lambda $ e $ z \in E_\lambda $, $\langle z,
        u(x)\rangle = \langle u^*(z), x\rangle =
        \langle\conj\lambda z, x\rangle = 0$.
  \item Indução na dimensão: acima $\C $, $ u $ tem um vetor próprio
        $ e_1$ (normalizar); seu complemento ortogonal é estável
        sob $ u $ (por (2)) \emph{e} sob $ u^*$ (mesmo argumento
        com funções trocadas), então a restrição é normal: induzir
        e concatenar autobases ortonormais. Por outro lado, um
        unitariamente \omterm{def:b2:reduction:diag}{diagonalizável} $ u = UDU^\dagger $ satisfaz $ u^*u
        = U\conj D D U^\dagger = U D\conj D U^\dagger = uu^*$:
        normal.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b2:hermitian:9}
\omterm{def:b2:reduction:eigen}{Autovalores}: para $ u(x) = \lambda x $, $ x \neq 0$:
\[
\lambda\norm x^2 = \langle x, u(x)\rangle
= \langle u^*(x), x\rangle = -\langle u(x), x\rangle
= -\conj{\langle x, u(x)\rangle} = -\conj\lambda\,\norm x^2 ,
\]
então $\lambda = -\conj\lambda $: puramente imaginário. Desde $(\iu
u)^* = -\iu\,u^*$ (o \omterm{def:b2:quadratic:adjoint}{adjunto} é conjugado-linear em escalares),
$ u^* = u $ dá $(\iu u)^* = -\iu u $: o mapa $ u \mapsto \iu u $
envia \omterm{def:b2:hermitian:adjoint}{Hermitiano} para \omterm{def:b2:hermitian:adjoint}{distorcido-Hermitiano}, com inverso $ w \mapsto -\iu
w $: uma bijeção. UM \omterm{def:b2:hermitian:adjoint}{distorcido-Hermitiano} $ u $ satisfaz $ u^*u = -u^2 =
uu^*$: normal, portanto unitariamente \omterm{def:b2:reduction:diag}{diagonalizável} por
\cref{exo:b2:hermitian:8}.
\end{solution}

\begin{solution}{exo:b2:hermitian:10}
\begin{enumerate}
  \item $ S $ permuta uma base ortonormal: $\norm{Sx} = \norm
        x $, então $ S $ é unitário. Coordenadas de indexação por $ j = 0,
        \dots, n-1$ módulo $ n $: $(Sx)_j = x_{j-1}$, então para
        $(f_k)_j = \frac{\omega^{jk}}{\sqrt n}$:
        \[
        (Sf_k)_j = \frac{\omega^{(j-1)k}}{\sqrt n}
        = \omega^{-k}\,(f_k)_j :
        \qquad Sf_k = \omega^{-k}f_k .
        \]
        Ortonormalidade: $\langle f_k, f_l\rangle = \frac1n
        \sum_j \omega^{j(l-k)} = \delta_{kl}$ (soma geométrica
        de uma raiz não trivial de unidade desaparece).
  \item $ Cf_k = \sum_j c_j S^jf_k = \bigl(\sum_j
        c_j\omega^{-jk}\bigr)f_k = \widehat c(k)\,f_k $: cada
        circulante é diagonal na base ortonormal de Fourier,
        portanto normal, com \omterm{def:b2:reduction:eigen}{espectro} $\{\widehat c(k)\}$. (O
        mudança de base é a transformada discreta de Fourier:
        convolução se torna multiplicação.)
\end{enumerate}
\end{solution}

\begin{solution}{exo:b2:hermitian:11}
($\Leftarrow $) Deixar $ P^2 = P = P^*$. Todo $ v $ divide como $ v = Pv
+ (v - Pv)$ com $ Pv \in \operatorname{im} P $ e $ P(v - Pv) =
0$. As duas peças são ortogonais: para qualquer $ x, y $,
\[
\langle Px, (I - P)y\rangle = \langle x, P(I-P)y\rangle
= \langle x, (P - P^2)y\rangle = 0 :
\]
$\ker P \perp \operatorname{im} P $, então $ P $ é o ortogonal
projeção em sua imagem. ($\Rightarrow $) Se $ P $ é o
projeção ortogonal em $ F = \operatorname{im}P $: para todos
$ x, y $, $\langle Px, y\rangle = \langle Px, Py\rangle $ (o
componente $ y - Py \perp F $ gotas) e simetricamente $\langle
x, Py\rangle = \langle Px, Py\rangle $: $\langle Px, y\rangle =
\langle x, Py\rangle $, ou seja\ $ P^* = P $. Matrizes: em $\C v $
($\norm v = 1$): $ Px = v\,\langle v, x\rangle $, ou seja\ $ P =
vv^\dagger $; para $\operatorname{Vect}(v_1, \dots, v_k)$
ortonormal: $ P = \sum_i v_iv_i^\dagger $.
\end{solution}

\begin{solution}{exo:b2:hermitian:12}
Diagonalizar $ A = U D U^\dagger $ (teorema espectral), $ D $ diagonal
com entradas entre os $\lambda_i $. Então $ P_i = L_i(A) = U
L_i(D)U^\dagger $, e $ L_i(D)$ é diagonal com entradas
$ L_i(\lambda_j) = \delta_{ij}$: aqueles exatamente nos slots de
$ E_i $. Então $ P_i $ é \omterm{def:b2:hermitian:adjoint}{Hermitiano} ($ L_i $ real, $ D $ real), idempotente,
com imagem $ E_i $ e núcleo $\bigoplus_{j\neq i}E_j = E_i^\perp $
(ortogonalidade de autoespaços): a projeção ortogonal em
$ E_i $ (\cref{exo:b2:hermitian:11}). Padrões diagonais disjuntos
dar $ P_iP_j = 0$ ($ i \neq j $); $\sum_i L_i = 1$ (grau $< p $,
valor $1$ no $ p $ pontos), então $\sum P_i = I $; e $\sum_i
\lambda_iL_i(\lambda_j) = \lambda_j $ dá $ A = \sum
\lambda_iP_i $. Para qualquer polinômio $ f $: $ f(D)$ tem diagonal
$ f(\lambda_j)$, então
\[
f(A) = \sum_{i=1}^{p} f(\lambda_i)\,P_i :
\]
funções de $ A $ são calculados espectralmente --- o cálculo que
o volume do Ano 3 se estende a \omterm{def:b2:metric:continuity}{contínuo} $ f $ e além.
\end{solution}

\begin{solution}{pb:b2:hermitian:1}
\textbf{1.} $\conj{\langle x, Ax\rangle} = \langle Ax, x\rangle
= \langle x, A^*x\rangle = \langle x, Ax\rangle $: real. Escrita
$ x = \sum c_ie_i $:
\[
R_A(x) = \frac{\sum_i\lambda_i\abs{c_i}^2}
{\sum_i\abs{c_i}^2} ,
\]
uma média ponderada do valores próprios: está em
$\intcc{\lambda_n}{\lambda_1}$, com os limites atingidos em
$ e_1$ e $ e_n $.

\textbf{2.} Sério $ t $ e qualquer $ v $, expandir $ R_A(x + tv) =
\frac{N(t)}{D(t)}$ com
\[
N(t) = \langle x, Ax\rangle + 2t\Re\langle v, Ax\rangle +
t^2\langle v, Av\rangle,
\quad
D(t) = \norm x^2 + 2t\Re\langle v, x\rangle + t^2\norm v^2 .
\]
A derivada em $ t = 0$ é
\[
\frac{2}{\norm x^2}\,
\Re\bigl\langle v,\ Ax - R_A(x)\,x\bigr\rangle .
\]
Ele desaparece para todos $ v $ se $\Re\langle v, w\rangle = 0$ para
tudo $ v $, onde $ w = Ax - R_A(x)x $; substituindo $ v $ por $\iu v $
mata a parte imaginária também: $ w = 0$, ou seja\ $ Ax = R_A(x)x $.
Pontos críticos de $ R_A $ são exatamente vetores próprios, com
valor crítico o \omterm{def:b2:reduction:eigen}{autovalor}.

\textbf{3.} Para $ x = \sum_{i\leq k}c_ie_i \in V_k $: $ R_A(x)$ é
uma média ponderada de $\lambda_1, \dots, \lambda_k $, por isso
$\geq \lambda_k $, com igualdade em $ e_k $: $\min_{V_k} R_A =
\lambda_k $. Simetricamente ligado $ W_k $ a média envolve
$\lambda_k, \dots, \lambda_n $: $\max_{W_k}R_A = \lambda_k $.

\textbf{4.} Deixar $\dim V = k $. Então $\dim V + \dim W_k = n + 1 >
n $, então existe uma unidade $ x \in V \cap W_k $, e $ R_A(x) \leq
\lambda_k $ (pergunta 3): $\min_{V}R_A \leq \lambda_k $ para cada
tal $ V $. Desde $ V_k $ atinge $\lambda_k $, o máximo-mínimo é igual
$\lambda_k $. A fórmula min-max é o mesmo argumento com o
papéis invertidos ($\dim W = n - k + 1$ forças $ W \cap V_k \neq
\{0\}$, então $\max_W R_A \geq \lambda_k $, alcançado em $ W_k $).

\textbf{5.} $ R_B(x) = R_A(x) + \frac{\langle x,
(B-A)x\rangle}{\norm x^2} \geq R_A(x)$ \omterm{def:b2:funcseq:def}{pontualmente}. Tirando
$\min $ sobre qualquer $ k $-dimensional $ V $ e então $\max $ sobre $ V $:
$\lambda_k(B) \geq \lambda_k(A)$ pela pergunta 4.

\textbf{6.} Grassman: $\dim(V\cap W) = \dim V + \dim W -
\dim(V + W) \geq \dim V + \dim W - n > 0$. Aplicando isso duas vezes:
\[
\dim(V_1\cap V_2\cap V_3)
\geq \dim(V_1\cap V_2) + \dim V_3 - n
\geq \dim V_1 + \dim V_2 + \dim V_3 - 2n .
\]

\textbf{7.} Os subespaços $ W_i(A)$, $ W_j(B)$ (pergunta 3, para
$ A $ e $ B $) e $ V_{i+j-1}(A+B)$ tem dimensões $(n-i+1) +
(n-j+1) + (i+j-1) = 2n + 1 > 2n $: pela questão 6 existe uma unidade
vetor $ x $ em todos os três. Então
\[
\lambda_{i+j-1}(A+B) \leq R_{A+B}(x)
= R_A(x) + R_B(x) \leq \lambda_i(A) + \lambda_j(B),
\]
a desigualdade à esquerda porque $ x \in V_{i+j-1}(A+B)$ (pergunta
3), o direito pelos dois $ W $'s.

\textbf{8.} Em uma base espectral de $ E $: $\norm{Ex}^2 = \sum
\lambda_k(E)^2\abs{c_k}^2 \leq \max_k\lambda_k(E)^2\,\norm
x^2$, alcançado no correspondente vetor próprio: $\vertiii E_2
= \max_k\abs{\lambda_k(E)}$. Weyl com $ j = 1$: $\lambda_k(A+E)
\leq \lambda_k(A) + \lambda_1(E) \leq \lambda_k(A) + \vertiii
E_2$; aplicando isso a $(A+E) + (-E)$: $\lambda_k(A) \leq
\lambda_k(A+E) + \vertiii E_2$. Junto:
$\abs{\lambda_k(A+E) - \lambda_k(A)} \leq \vertiii E_2$.

\textbf{9.} Limites inferiores: $ P \geq 0$ e questão 5. Superior: $ P $
tem classificação $1$, então $\lambda_2(P) = 0$; Weyl com $ i = k-1$, $ j =
2$:
\[
\lambda_k(A + P) \leq \lambda_{k-1}(A) + \lambda_2(P)
= \lambda_{k-1}(A) .
\]

\textbf{10.} $ A + E = \begin{pmatrix} 2 & 1+\iu\\ 1-\iu &
2\end{pmatrix}$: polinômio característico $(2-\lambda)^2 -
\abs{1+\iu}^2 = (2-\lambda)^2 - 2$, \omterm{def:b2:reduction:eigen}{espectro} $\{2 + \sqrt2,\ 2
- \sqrt2\}$. Contra $\operatorname{Sp}A = \{3, 1\}$:
\[
\abs{(2+\sqrt2) - 3} = \abs{(2-\sqrt2) - 1} = \sqrt2 - 1
\approx 0.414 \leq 1 = \vertiii E_2 . \checkmark
\]

\textbf{11.} Visualizar $\C^{n-1} = \operatorname{Vect}(e_1, \dots,
e_{n-1})$ dentro $\C^n $ (base padrão): para $ x $ lá,
$\langle x, Bx\rangle = \langle x, Ax\rangle $, então $ R_B $ é o
restrição de $ R_A $. Limite superior: o máximo-mínimo para $\lambda_k
(B)$ varia acima $ k $ subespaços dimensionais \emph{de}
$\C^{n-1}$, uma subfamília daqueles de $\C^n $: $\lambda_k(B) \leq
\lambda_k(A)$. Limite inferior: o min-max para $\lambda_k(B)$
varia em subespaços de $\C^{n-1}$ de dimensão $(n-1)-k+1 =
n-k $; cada um também é um subespaço de $\C^n $ de dimensão $ n -
(k+1) + 1$, então seu máximo é $\geq \lambda_{k+1}(A)$:
$\lambda_k(B) \geq \lambda_{k+1}(A)$.

\textbf{12.} Remova as linhas/colunas uma de cada vez e encadeie
questão 11: cada remoção desloca o índice inferior em um, dando
$\lambda_{k+m}(A) \leq \lambda_k(B) \leq \lambda_k(A)$.

\textbf{13.} Conjugando $ A $ por uma matriz de permutação (unitário)
nem altera o \omterm{def:b2:reduction:eigen}{espectro} nem o multiconjunto da diagonal
entradas: assumir $ d_1, \dots, d_k $ ocupar as posições de liderança.
O líder $ k\times k $ submatriz principal $ B $ então tem
$\operatorname{tr} B = \sum_{i\leq k}d_i $, e seu valores próprios
satisfazer $\mu_i(B) \leq \lambda_i(A)$ (questão 12): somando,
$\sum_{i\leq k}d_i \leq \sum_{i\leq k}\lambda_i(A)$. No $ k =
n $ ambos os lados estão $\operatorname{tr} A $.

\textbf{14.} Tirando $ x_i = e_i $ dá o valor
$\sum_{i\leq k}\lambda_i $: o máximo é $\geq $. Por outro lado, um
família ortonormal $(x_1, \dots, x_k)$ estende-se a um
base ortonormal, ou seja,\ para uma unitário $ U $ com primeiras colunas
$ x_i $; então $\sum_i\langle x_i, Ax_i\rangle $ é a soma dos
primeiro $ k $ entradas diagonais de $ U^\dagger AU $, que por questão
13 é no máximo a soma de seus $ k $ maior valores próprios ---
a saber $\sum_{i\leq k}\lambda_i(A)$. Máximo de Ky Fan
segue o princípio.

\textbf{15.} Entrelaçamento ($\operatorname{Sp}A = \{2+\sqrt2, 2,
2-\sqrt2\}$, $\operatorname{Sp}B = \{3, 1\}$):
\[
2 \leq 3 \leq 2 + \sqrt2,
\qquad
2 - \sqrt2 \leq 1 \leq 2 . \checkmark
\]
Schur com diagonal $(2,2,2)$: $2 \leq 2+\sqrt2$; $4 \leq 4 +
\sqrt2$; $6 = 6$ (traço). \checkmark

\textbf{16.} $ b_{ii} - a_{ii} = \langle e_i, (B-A)e_i\rangle
\geq 0$; a soma fornece os traços. Para qualquer $ C $: $\langle x,
C^\dagger(B - A)Cx\rangle = \langle Cx, (B-A)(Cx)\rangle \geq
0$: $ C^\dagger AC \leq C^\dagger BC $.

\textbf{17.} $ A \geq 0$ claro; $ B - A = \begin{pmatrix} 1 & 1\\
1 & 1\end{pmatrix}$ é positivo semidefinido (valores próprios $2,
0$): $ A \leq B $. Mas
\[
B^2 = \begin{pmatrix} 5 & 3\\ 3 & 2\end{pmatrix},
\qquad
B^2 - A^2 = \begin{pmatrix} 4 & 3\\ 3 & 2\end{pmatrix},
\qquad \det(B^2 - A^2) = -1 < 0 :
\]
não positivo semidefinido. A quadratura não respeita o
Ordem de Loewner.

\textbf{18.} Deixar $ S = \sqrt A $, $ T = \sqrt B $ (\omterm{def:b2:hermitian:adjoint}{Hermitiano}
positivo semidefinido, \cref{exo:b2:hermitian:7} estendido para
semidefinido pela mesma fórmula espectral). $ T - S $ é
\omterm{def:b2:hermitian:adjoint}{Hermitiano}; deixar $\mu $ seja qualquer \omterm{def:b2:reduction:eigen}{autovalor}, $ v $ uma unidade vetor próprio.
De $ T^2 - S^2 = T(T - S) + (T - S)S $:
\[
0 \leq \langle v, (B - A)v\rangle
= \langle Tv, (T-S)v\rangle + \langle (T-S)v, Sv\rangle
= \mu\bigl(\langle v, Tv\rangle + \langle v,
Sv\rangle\bigr)
\]
($\mu $ é real). Se $\langle v, Tv\rangle + \langle v,
Sv\rangle > 0$, então $\mu \geq 0$. Se desaparecer, ambos
os termos não negativos desaparecem; $\langle v, Tv\rangle =
\norm{T^{1/2}v}^2 = 0$ forças $ Tv = 0$, da mesma maneira $ Sv = 0$, então
$\mu v = (T - S)v = 0$ e $\mu = 0$. Todos valores próprios de $ T -
S $ são $\geq 0$: $\sqrt A \leq \sqrt B $.

\textbf{19.} \omterm{def:b2:quadratic:def}{Congruência} por $ A^{-1/2}$ (questão 16): $ I \leq M
:= A^{-1/2}BA^{-1/2}$. Então tudo valores próprios de $ M $ são $\geq 1$,
daí aqueles de $ M^{-1}$ deitar em $\intoc{0}{1}$: $ M^{-1} \leq I $.
Mas $ M^{-1} = A^{1/2}B^{-1}A^{1/2}$; congruente $ M^{-1} \leq
I $ por $ A^{-1/2}$ dá $ B^{-1} \leq A^{-1}$.

\textbf{20.} $\sqrt A^{-1}(AB)\sqrt A = \sqrt A\,B\sqrt A $: então
$ AB $ é semelhante ao \omterm{def:b2:hermitian:adjoint}{Hermitiano} definido positivo $\sqrt
A\,B\sqrt A $ (definido: $\langle x, \sqrt AB\sqrt Ax\rangle =
\langle \sqrt Ax, B\sqrt Ax\rangle > 0$): isso é valores próprios são
real e positivo. Além disso
\[
\langle x, \sqrt AB\sqrt Ax\rangle
\leq \lambda_{\max}(B)\,\norm{\sqrt Ax}^2
= \lambda_{\max}(B)\,\langle x, Ax\rangle
\leq \lambda_{\max}(A)\lambda_{\max}(B)\norm x^2 ,
\]
então $\lambda_{\max}(AB) = \max R_{\sqrt AB\sqrt A} \leq
\lambda_{\max}(A)\lambda_{\max}(B)$.

\textbf{21.} Com $ v_k = (\sin j\theta_k)_j $ e a identidade

2\sin(j\theta)\cos\theta $: para $2 \leq j \leq n-1$,
\[
(T_nv_k)_j = -\sin((j{-}1)\theta_k) + 2\sin(j\theta_k) -
\sin((j{+}1)\theta_k)
= (2 - 2\cos\theta_k)\sin(j\theta_k) .
\]
Linha $1$ funciona porque $\sin(0\cdot\theta_k) = 0$, linha $ n $

condições selecione exatamente $\theta_k = \frac{k\pi}{n+1}$. Então
$ T_nv_k = 4\sin^2\bigl(\frac{k\pi}{2(n+1)}\bigr)v_k $; o $ n $
os valores são distintos em $\intoo04$ e o $ v_k \neq 0$: isto é
o todo \omterm{def:b2:reduction:eigen}{espectro}, positivo, então $ T_n $ é positivo definido.
