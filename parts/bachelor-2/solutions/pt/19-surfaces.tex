\section*{Capítulo \ref{ch:b2:surfaces} --- Superfícies}

\begin{solution}{exo:b2:surfaces:1}
Com $\sigma(\theta, r) = (r\cos\theta,\ r\sin\theta,\ r)$:
\[
\sigma_\theta = (-r\sin\theta,\ r\cos\theta,\ 0),
\qquad
\sigma_r = (\cos\theta,\ \sin\theta,\ 1),
\]
independente para $ r > 0$. O \omterm{def:b2:surfaces:tangent}{plano tangente} no $ M_0 =
\sigma(\theta_0, r_0)$ passa $ M_0$ com instruções
$\sigma_\theta, \sigma_r $. Agora $ M_0 - O = r_0(\cos\theta_0,
\sin\theta_0, 1) = r_0\,\sigma_r(\theta_0, r_0)$ é em si um
direção tangente: o ápice $ O $ encontra-se no \omterm{def:b2:surfaces:tangent}{plano tangente}. (Este
é o comportamento geral dos cones: eles são regidos por linhas que passam
o ápice e um \omterm{def:b2:surfaces:tangent}{plano tangente} contém a linha dominante através do
ponto de tangência.)
\end{solution}

\begin{solution}{exo:b2:surfaces:2}
Aplicar \cref{prop:b2:surfaces:gradient} para $ f(x,y,z) =
\frac{x^2}{a^2} + \frac{y^2}{b^2} + \frac{z^2}{c^2}$: $\nabla
f(x_0,y_0,z_0) = 2\bigl(\frac{x_0}{a^2}, \frac{y_0}{b^2},
\frac{z_0}{c^2}\bigr) \neq 0$ na superfície. O \omterm{def:b2:surfaces:tangent}{plano tangente} é
\[
\frac{x_0}{a^2}(x - x_0) + \frac{y_0}{b^2}(y - y_0)
+ \frac{z_0}{c^2}(z - z_0) = 0,
\qquad\text{i.e.}\qquad
\frac{x_0\,x}{a^2} + \frac{y_0\,y}{b^2} + \frac{z_0\,z}{c^2} = 1,
\]
usando isso $(x_0, y_0, z_0)$ satisfaz a equação do elipsóide ---
a regra de ``dividir os quadrados'' generalizando a esfera $\langle
M_0, M \rangle = R^2$.
\end{solution}

\begin{solution}{exo:b2:surfaces:3}
$\sigma_u = (-v\sin u,\ v\cos u,\ a)$ e $\sigma_v = (\cos u,\
\sin u,\ 0)$, então
\[
E = v^2 + a^2, \qquad F = 0, \qquad G = 1,
\qquad
\sqrt{EG - F^2} = \sqrt{v^2 + a^2} > 0
\]
(o helicóide é regular em todos os lugares, inclusive em seu eixo $ v =
0$). \omterm{def:b2:surfaces:area}{Área} da peça:
\[
\mathcal{A} = \int_0^{2\pi}\!\!\int_0^1
\sqrt{v^2 + a^2}\;\dd v\,\dd u
= 2\pi\cdot\frac12\Bigl[v\sqrt{v^2 + a^2}
+ a^2\ln\bigl(v + \sqrt{v^2 + a^2}\bigr)\Bigr]_0^1 ,
\]
ou seja\ $\mathcal{A} = \pi\Bigl(\sqrt{1 + a^2} +
a^2\ln\frac{1 + \sqrt{1 + a^2}}{a}\Bigr)$.
\end{solution}

\begin{solution}{exo:b2:surfaces:4}
Derivados:
\[
\sigma_\theta = \bigl(-(R + r\cos\psi)\sin\theta,\
(R + r\cos\psi)\cos\theta,\ 0\bigr),
\qquad
\sigma_\psi = \bigl(-r\sin\psi\cos\theta,\
-r\sin\psi\sin\theta,\ r\cos\psi\bigr).
\]
Então
\[
E = (R + r\cos\psi)^2,
\qquad
F = r\sin\psi\cos\psi\,(R + r\cos\psi)
\bigl(\sin\theta\cos\theta - \sin\theta\cos\theta\bigr) = 0,
\qquad
G = r^2 ,
\]
então $\sqrt{EG - F^2} = r(R + r\cos\psi) \geq r(R - r) > 0$:
regular em todos os lugares. \omterm{def:b2:surfaces:area}{Área}:
\[
\mathcal{A}
= \int_0^{2\pi}\!\!\int_0^{2\pi} r(R + r\cos\psi)
\,\dd\theta\,\dd\psi
= 2\pi r \Bigl(2\pi R + r\int_0^{2\pi}\cos\psi\,\dd\psi\Bigr)
= 4\pi^2 R r ,
\]
o $\cos\psi $ termo integrando a zero --- teorema de Pappus: \omterm{def:b2:surfaces:area}{área}
$=$ (\omterm{def:b2:curves:length}{comprimento} do círculo girado) $\times $ (distância percorrida por
seu centro).
\end{solution}

\begin{solution}{exo:b2:surfaces:5}
Para $\sigma(x, y) = (x, y, f(x,y))$: $\sigma_x = (1, 0, f_x)$,
$\sigma_y = (0, 1, f_y)$, então $ E = 1 + f_x^2$, $ F = f_xf_y $, $ G = 1
+ f_y^2$ e
\[
EG - F^2 = (1 + f_x^2)(1 + f_y^2) - f_x^2f_y^2
= 1 + f_x^2 + f_y^2 ,
\]
dando o declarado \omterm{def:b2:surfaces:area}{área} fórmula. Para $ f = \frac12(x^2 + y^2)$ ligado
o disco da unidade: $1 + f_x^2 + f_y^2 = 1 + x^2 + y^2$, e em \omterm{pb:b2:quadratic:1}{polares}
coordenadas ($ x = \rho\cos\alpha $, $ y = \rho\sin\alpha $, Jacobiano
$\rho $, \cref{ch:b2:multint}):
\[
\mathcal{A}
= \int_0^{2\pi}\!\!\int_0^1 \sqrt{1 + \rho^2}\;\rho
\,\dd\rho\,\dd\alpha
= 2\pi\Bigl[\tfrac13(1 + \rho^2)^{3/2}\Bigr]_0^1
= \frac{2\pi}{3}\bigl(2\sqrt2 - 1\bigr).
\]
\end{solution}

\begin{solution}{exo:b2:surfaces:6}
De \cref{ex:b2:surfaces:spherearea}cálculo, $ E =
R^2\cos^2\varphi $, $ F = 0$, $ G = R^2$, então por
\cref{prop:b2:surfaces:curvelength}
\[
L = \int_a^b R\sqrt{\cos^2\varphi(t)\,\theta'(t)^2
+ \varphi'(t)^2}\;\dd t .
\]
Sejam os pontos finais $(\theta_0, \varphi_1)$ e $(\theta_0,
\varphi_2)$, $\varphi_1 < \varphi_2$. Para qualquer curva de união,
\[
L \geq \int_a^b R\,\abs{\varphi'(t)}\,\dd t
\geq R\,\Bigl|\int_a^b \varphi'(t)\,\dd t\Bigr|
= R\,(\varphi_2 - \varphi_1),
\]
descartando o não negativo $\cos^2\varphi\,\theta'^2$ termo e usando
a desigualdade triangular para integrais. O arco meridiano
$\theta \equiv \theta_0$, $\varphi $ aumentando de $\varphi_1$ para
$\varphi_2$, tem \omterm{def:b2:curves:length}{comprimento} exatamente $ R(\varphi_2 - \varphi_1)$: é
mais curto. (Os meridianos são círculos máximos; este é o primeiro,
caso elementar do fato de que as geodésicas da esfera são ótimas
círculos.)
\end{solution}

\begin{solution}{exo:b2:surfaces:7}
Corrigir uma curva $\gamma $ desenhado em $ S $ e deixe $ g(t) = \norm{\gamma(t)
- \Omega}^2$. Então $ g'(t) = 2\langle \gamma'(t),\ \gamma(t) -
\Omega\rangle $. A linha normal em $ M = \gamma(t)$ passa
$\Omega $ por hipótese, então $\gamma(t) - \Omega $ é um \emph{normal}
vetor, ortogonal ao \omterm{def:b2:surfaces:tangent}{plano tangente}, em especial ao
velocidade $\gamma'(t)$ (\cref{prop:b2:surfaces:velocity}): $ g' =
0$, e $ g $ é constante ao longo de cada curva desenhada.

Agora o conjunto $ S_c = \{M \in S : \norm{M - \Omega}^2 = c\}$ é
fechado em $ S $; também é \omterm{def:b2:metric:topology}{abrir} em $ S $: perto de qualquer um de seus pontos, $ S $
é um gráfico regular, então qualquer ponto próximo de $ S $ está unido a ele por
uma curva desenhada (um segmento elevado), ao longo da qual $ g $ é constante. Como
$ S $ é \omterm{def:b2:metric:connected}{conectado} e $ S_c $ não vazio para o direito $ c $, $ S = S_c
\subseteq $ a esfera do centro $\Omega $ e raio $\sqrt c $
(\cref{ch:b2:metric}: conectividade argumento).
\end{solution}

\begin{solution}{exo:b2:surfaces:8}
Escrever $\Phi(u', v') = (u, v)$. Pela regra da cadeia,
\[
\tilde\sigma_{u'} = \frac{\partial u}{\partial u'}\sigma_u
+ \frac{\partial v}{\partial u'}\sigma_v,
\qquad
\tilde\sigma_{v'} = \frac{\partial u}{\partial v'}\sigma_u
+ \frac{\partial v}{\partial v'}\sigma_v ,
\]
parciais de $\sigma $ avaliado em $\Phi(u',v')$. Expandindo o
produto cruzado bilinearmente e usando $\sigma_u \wedge \sigma_u =
\sigma_v \wedge \sigma_v = 0$, $\sigma_v \wedge \sigma_u =
-\sigma_u \wedge \sigma_v $:
\[
\tilde\sigma_{u'} \wedge \tilde\sigma_{v'}
= \Bigl(\frac{\partial u}{\partial u'}
\frac{\partial v}{\partial v'}
- \frac{\partial v}{\partial u'}
\frac{\partial u}{\partial v'}\Bigr)\,
\sigma_u \wedge \sigma_v
= \det J_\Phi \cdot (\sigma_u \wedge \sigma_v)\circ\Phi .
\]
Tirando \omterm{def:b2:nvs:norm}{normas} dá a identidade. Então, pela mudança de variáveis
fórmula (\cref{ch:b2:multint}) aplicado ao mapa $\Phi $ sobre $ K' =
\Phi^{-1}(K)$:
\[
\iint_{K'} \norm{\tilde\sigma_{u'} \wedge \tilde\sigma_{v'}}
\,\dd u'\dd v'
= \iint_{K'} \norm{\sigma_u \wedge \sigma_v}\circ\Phi\;
\abs{\det J_\Phi}\,\dd u'\dd v'
= \iint_K \norm{\sigma_u \wedge \sigma_v}\,\dd u\,\dd v :
\]
as duas parametrizações atribuem o mesmo \omterm{def:b2:surfaces:area}{área} para o mesmo pedaço de
superfície.
\end{solution}

\begin{solution}{exo:b2:surfaces:9}
No gráfico esférico, $ z = R\sin\varphi $, e a zona
corresponde a $\varphi_1 \leq \varphi \leq \varphi_2$ com
$ z_i = R\sin\varphi_i $. Com o \omterm{def:b2:surfaces:area}{área} elemento
$ R^2\cos\varphi\,\dd\theta\,\dd\varphi $
(\cref{ex:b2:surfaces:spherearea}):
\[
\mathcal A = \int_{\varphi_1}^{\varphi_2}\!\!\int_0^{2\pi}
R^2\cos\varphi\,\dd\theta\,\dd\varphi
= 2\pi R^2(\sin\varphi_2 - \sin\varphi_1)
= 2\pi R\,(z_2 - z_1) .
\]
O resultado depende apenas da altura $ z_2 - z_1$: fatias de
espessura igual carrega igual \omterm{def:b2:surfaces:area}{áreas}, seja cortado no equador
ou no pólo --- teorema da caixa de chapéu de Arquimedes, e a razão
a lateral \omterm{def:b2:surfaces:area}{área} do cilindro circunscrito ($2\pi R \cdot
2R = 4\pi R^2$) é igual a \omterm{def:b2:surfaces:area}{área} da esfera.
\end{solution}

\begin{solution}{exo:b2:surfaces:10}
$\sigma_\theta = (-r\sin\theta,\ r\cos\theta,\ 0)$ e
$\sigma_z = (r'\cos\theta,\ r'\sin\theta,\ 1)$, então
\[
\sigma_\theta \wedge \sigma_z
= (r\cos\theta,\ r\sin\theta,\ -r\,r') ,
\]
um vetor normal em $ M = (r\cos\theta, r\sin\theta, z)$. O
linha normal é
\[
t \mapsto \bigl(r(1 + t)\cos\theta,\ r(1 + t)\sin\theta,\
z - t\,r\,r'\bigr),
\]
que em $ t = -1$ alcança $(0,\ 0,\ z + r(z)\,r'(z))$: cada
a linha normal encontra o eixo, na altura $ z + rr'$. (Isso é
a razão tridimensional pela qual a simetria rotacional sobrevive em
o campo normal.)
\end{solution}

\begin{solution}{exo:b2:surfaces:11}
$\sigma_u = (-\sin u, \cos u, 0)$, $\sigma_v = (0, 0, 1)$: $ E
= 1$, $ F = 0$, $ G = 1$. Por
\cref{prop:b2:surfaces:curvelength} o \omterm{def:b2:curves:length}{comprimento} de um desenhado
curva é $\int\sqrt{u'^2 + v'^2}\,\dd t $ --- o \omterm{def:b2:curves:length}{comprimento} do seu
sombra de parâmetro $(u(t), v(t))$ no plano: o gráfico é um
isometria local (o desenrolar do cilindro). A hélice
$ t \mapsto (\cos t, \sin t, ct)$, $ t \in \intcc0{2\pi}$, tem
sombreie o segmento de $(0,0)$ para $(2\pi, 2\pi c)$, de \omterm{def:b2:curves:length}{comprimento}
$2\pi\sqrt{1 + c^2}$. Qualquer curva desenhada de $(1,0,0)$ para $(1,
0, 2\pi c)$ dar uma volta completa tem um \omterm{def:b2:metric:continuity}{contínuo} sombra
juntando-se $(0, 0)$ para $(2\pi, 2\pi c)$, de avião \omterm{def:b2:curves:length}{comprimento} $\geq $
o segmento reto; desde \omterm{def:b2:curves:length}{comprimentos} concordo, a hélice é
mais curto.
\end{solution}

\begin{solution}{exo:b2:surfaces:12}
A função $ g(M) = \norm{M}^2$ é \omterm{def:b2:metric:continuity}{contínuo} no \omterm{def:b2:metric:compact}{compactar}
$ S $, então atinge seu máximo em algum $ M_0$. Para cada curva
$\gamma $ desenhado em $ S $ com $\gamma(0) = M_0$, a função $ t
\mapsto \norm{\gamma(t)}^2$ tem um máximo em $ t = 0$, então é
derivada $2\langle\gamma'(0), M_0\rangle $ desaparece: $ M_0$ é
ortogonal a cada vetor tangente, ou seja, \ normal a $ S $ no
$ M_0$. Como $\nabla f(M_0) \neq 0$ também direciona a linha normal
(\cref{prop:b2:surfaces:gradient}), $\nabla f(M_0)$ e
$\vect{OM_0}$ são colineares. Para o elipsóide, radialidade
significa
\[
\Bigl(\frac{x_0}{a^2}, \frac{y_0}{b^2},
\frac{z_0}{c^2}\Bigr) = \mu\,(x_0, y_0, z_0) :
\]
cada coordenada satisfaz $ x_0(\frac1{a^2} - \mu) = 0$, etc.;
desde $ a^{-2}, b^{-2}, c^{-2}$ são distintos, no máximo um
coordenada é diferente de zero e as soluções na superfície são
os pontos finais dos seis eixos $(\pm a, 0, 0)$, $(0, \pm b, 0)$, $(0,
0, \pm c)$. Os pontos mais distantes são $(\pm a, 0, 0)$, em
distância $ a = \max(a,b,c)$.
\end{solution}

\begin{solution}{pb:b2:surfaces:1}
\textbf{1.} Expandindo e usando a simetria de $ A $
($ t^{\mathsf T}\!APY = (At)^{\mathsf T}PY $):
\[
q(PY + t) = Y^{\mathsf T}P^{\mathsf T}\!APY +
2\,(At + b)^{\mathsf T}PY + \bigl(t^{\mathsf T}\!At +
2b^{\mathsf T}t + c\bigr),
\]
que é o triplo exibido. $ A' = P^{\mathsf T}\!AP =
P^{-1}AP $ é semelhante a $ A $: mesmo polinômio característico,
\omterm{def:b2:reduction:eigen}{espectro}, classificação, assinatura. Finalmente $\{q = 0\} = \{sq = 0\}$
para $ s \neq 0$, então apenas a equação \emph{até um escalar} é
anexado ao conjunto; dimensionando por $ s $ multiplica tudo valores próprios
por $ s $.

\textbf{2.} O teorema espectral fornece $ P \in O(3)$ com
$ P^{\mathsf T}\!AP = \operatorname{diag}(\lambda_1, \lambda_2,
\lambda_3)$; pergunta 1 com $ t = 0$ transforma a equação em
$\sum\lambda_iy_i^2 + 2\sum\beta_iy_i + c = 0$, onde $\beta =
P^{\mathsf T}b $.

\textbf{3.} Para $\lambda_i \neq 0$: $\lambda_iy_i^2 +
2\beta_iy_i = \lambda_i\bigl(y_i +
\frac{\beta_i}{\lambda_i}\bigr)^2 -
\frac{\beta_i^2}{\lambda_i}$; a tradução $ z_i = y_i +
\beta_i/\lambda_i $ (e $ z_i = y_i $ para $ i > r $) dá
\[
\sum_{i \leq r}\lambda_iz_i^2 + 2\sum_{i > r}\beta_iz_i +
c'' = 0, \qquad
c'' = c - \sum_{i\leq r}\frac{\beta_i^2}{\lambda_i}.
\]

\textbf{4.} $ q(2\Omega - X) = (2\Omega - X)^{\mathsf T}
A(2\Omega - X) + 2b^{\mathsf T}(2\Omega - X) + c $; expandindo
e subtraindo $ q(X)$, os termos quadráticos se cancelam e
\[
q(2\Omega - X) - q(X) = -4\,\langle A\Omega + b,\ X\rangle +
4\,\langle A\Omega + b,\ \Omega\rangle .
\]
Se $ A\Omega + b = 0$ isso desaparece de forma idêntica: o ponto
reflexão preserva $ q $, daí o \omterm{pb:b2:surfaces:1}{quádrica}. Por outro lado,
``$ q(2\Omega - X) = q(X)$ para todos $ X $'' diz que o \omterm{def:b2:affine:subspace}{afim}
a função acima desaparece em todos $\R^3$, o que força a sua
parte linear $ A\Omega + b $ ser zero. Então centros $=$ soluções de
$ A\Omega = -b $: um conjunto não vazio se $ b \in \operatorname{im}A $
(um \omterm{def:b2:affine:subspace}{afim} subespaço dirigido por $\ker A $) e um único ponto
se $ A $ é invertível.

\textbf{5.} Assinatura $(3,0)$ (todos $\lambda_i > 0$): $\delta >
0$ dá $\frac{x^2}{a^2} + \frac{y^2}{b^2} + \frac{z^2}{c^2}
= 1$ com $ a = \sqrt{\delta/\lambda_1}$, etc.\ --- um
elipsóide; $\delta = 0$: o ponto único $ O $; $\delta < 0$:
vazio. Assinatura $(2,1)$ ($\lambda_1, \lambda_2 > 0 >
\lambda_3$): $\delta > 0$: $\frac{x^2}{a^2} + \frac{y^2}{b^2}
- \frac{z^2}{c^2} = 1$, o hiperbolóide de uma folha; $\delta =
0$: o cone $\frac{x^2}{a^2} + \frac{y^2}{b^2} =
\frac{z^2}{c^2}$; $\delta < 0$: $\frac{z^2}{c^2} -
\frac{x^2}{a^2} - \frac{y^2}{b^2} = 1$, as duas folhas
hiperbolóide ($\abs z \geq c $: dois componentes).

\textbf{6.} A parte quadrática tem matriz $ A $ com diagonal
$1$ e fora da diagonal $2$: $ A = 2J - I $. Desde $ J $ tem \omterm{def:b2:reduction:eigen}{espectro}
$\{3, 0, 0\}$ (vetor próprio $(1,1,1)$ para $3$), $ A $ tem
\omterm{def:b2:reduction:eigen}{espectro} $\{5, -1, -1\}$, o \omterm{def:b2:reduction:eigen}{autovalor} $5$ carregado por
$\R(1,1,1)$. Nas coordenadas giradas: $5u^2 - v^2 - w^2 =
1$, ou seja\ $\frac{u^2}{1/5} - v^2 - w^2 = 1$: uma folha dupla
hiperbolóide, de revolução (igual valores próprios $-1$) sobre o
eixo $\R(1,1,1)$.

\textbf{7.} A superfície é $\bigl(\frac xa - \frac
zc\bigr)\bigl(\frac xa + \frac zc\bigr) = \bigl(1 - \frac
yb\bigr)\bigl(1 + \frac yb\bigr)$. Para $(\lambda : \mu) \neq
(0:0)$ definir a linha
\[
D_{\lambda:\mu} :\quad
\lambda\Bigl(\frac xa - \frac zc\Bigr) = \mu\Bigl(1 - \frac
yb\Bigr), \qquad
\mu\Bigl(\frac xa + \frac zc\Bigr) = \lambda\Bigl(1 + \frac
yb\Bigr)
\]
(dois independentes \omterm{def:b2:affine:subspace}{afim} equações: uma reta). Multiplicando o
duas equações mostram cada ponto de $ D_{\lambda:\mu}$ mente
a superfície quando $\lambda\mu \neq 0$; os casos $\lambda = 0$
ou $\mu = 0$ são verificados diretamente (por exemplo,\ $\lambda = 0$: $ y =
b $, $\frac xa = -\frac zc $, o que satisfaz a equação). O
segunda família $ D'_{\lambda:\mu}$ troca os dois da direita
fatores.

\textbf{8.} Consertar $ M $ na superfície. As condições para $ M \in
D_{\lambda:\mu}$ formar um $2\times2$ sistema linear homogêneo
em $(\lambda, \mu)$ cujo determinante é
\[
\Bigl(\frac{x^2}{a^2} - \frac{z^2}{c^2}\Bigr) - \Bigl(1 -
\frac{y^2}{b^2}\Bigr) = 0
\]
precisamente porque $ M $ encontra-se no \omterm{pb:b2:surfaces:1}{quádrica}: um não trivial
solução $(\lambda : \mu)$ existe. A matriz de coeficientes é
nunca zero (isso forçaria $1 - \frac yb = 1 + \frac yb =
0$), então sua classificação é $1$ e a solução é única até
escala: exatamente uma linhagem da família passa $ M $.
O mesmo vale para a segunda família, e as duas linhas são
distinto (em $(a, 0, 0)$ eles são $\{x = a,\ \frac zc = \frac
yb\}$ e $\{x = a,\ \frac zc = -\frac yb\}$): a folha única
o hiperbolóide é duplamente governado.

\textbf{9.} Deixar $ M = (x, y, z)$ no hiperbolóide, $\rho^2 =
\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1 + \frac{z^2}{c^2} \geq
1$. O ponto $ N = (x,\ y,\ \varepsilon c\rho)$ com
$\varepsilon $ o sinal de $ z $ satisfaz $\frac{x^2}{a^2} +
\frac{y^2}{b^2} - \frac{(c\rho)^2}{c^2} = 0$: $ N \in C $, e
\[
d(M, C) \leq \abs{z - \varepsilon c\rho}
= c\bigl(\rho - \sqrt{\rho^2 - 1}\bigr)
= \frac{c}{\rho + \sqrt{\rho^2 - 1}} .
\]
Se $\norm M \to \infty $ então $\rho \to \infty $ (todos os três
coordenadas são limitadas por múltiplos de $\rho $), então $ d(M, C)
\to 0$. A seção $ x = a $: $\frac{y^2}{b^2} -
\frac{z^2}{c^2} = 0$, o par de linhas cruzadas de questão
8 --- e da mesma forma em $ x = -a $.

\textbf{10.} Com $\lambda_3 = 0$, pergunta 3 sai
$\lambda_1z_1^2 + \lambda_2z_2^2 + 2\beta_3z_3 + c'' = 0$. Em
a base própria, $\operatorname{im}A = \operatorname{Vect}(e_1,
e_2)$, então pela questão 4 centros existem se $\beta_3 = 0$. Se
$\beta_3 \neq 0$: a tradução $ z_3 \mapsto z_3 -
c''/(2\beta_3)$ remove a constante, deixando $\lambda_1z_1^2
+ \lambda_2z_2^2 + 2\beta_3z_3 = 0$, ou seja\ $ z_3 = px^2 +
qy^2$ após renomear: um \emph{parabolóide elíptico} se
$\lambda_1\lambda_2 > 0$, um \emph{parabolóide hiperbólico} se
$\lambda_1\lambda_2 < 0$ --- e de fato nenhum centro. Se
$\beta_3 = 0$: a equação $\lambda_1z_1^2 + \lambda_2z_2^2 +
c'' = 0$ não envolve $ z_3$: o \omterm{pb:b2:surfaces:1}{quádrica} é um cilindro
sobre o plano cônico correspondente --- cilindro elíptico,
linha ou conjunto vazio quando $\lambda_1\lambda_2 > 0$; hiperbólico
cilindro ou par de planos que se cruzam quando
$\lambda_1\lambda_2 < 0$ --- com toda uma linha de centros
$\{(z_1^*, z_2^*)\} \times \R $.

\textbf{11.} $ xy - z = 0$. Substituindo $ x = \frac{u +
v}{\sqrt2}$, $ y = \frac{u - v}{\sqrt2}$ (rotação por $\pi/4$):
$ xy = \frac{u^2 - v^2}2$, então a equação se torna $ z =
\frac12(u^2 - v^2)$: um parabolóide hiperbólico --- a sela
da figura, até o fator $\frac12$.

\textbf{12.} A linha $\{x = x_0,\ z = x_0y\}$ (parametrizado
por $ y $) claramente reside $ z = xy $, assim como $\{y = y_0,\ z =
xy_0\}$; através $(x_0, y_0, x_0y_0)$ passe os dois.
Singularidade: se $ t \mapsto (x_0 + tv_1, y_0 + tv_2, z_0 +
tv_3)$ permanece na superfície, o coeficiente de $ t^2$ em $(x_0
+ tv_1)(y_0 + tv_2) - z_0 - tv_3$ dá $ v_1v_2 = 0$, então
$ v_1 = 0$ ou $ v_2 = 0$, pousando em uma das duas famílias:
uma linha de cada através de cada ponto --- a segunda duplamente
governou \omterm{pb:b2:surfaces:1}{quádrica}.

\textbf{13.} Completando quadrados: $(x - 1)^2 + (y + 2)^2 = 2$,
sem nenhuma condição $ z $: um cilindro circular reto de raio
$\sqrt2$ e eixo a linha vertical $\{(1, -2, z) : z \in
\R\}$ --- uma linha de centros, como prevê a questão 10.

\textbf{14.} Com $\lambda_2 = \lambda_3 = 0$ o reduzido
equação é $\lambda_1z_1^2 + 2\beta_2z_2 + 2\beta_3z_3 + c''
= 0$. Uma rotação do plano do kernel $(z_2, z_3)$ alinha o
forma linear: $2\beta_2z_2 + 2\beta_3z_3 = 2\beta w $ com
$\beta = \sqrt{\beta_2^2 + \beta_3^2}$. Se $\beta \neq 0$,
traduzir $ w $ absorver $ c''$: $\lambda_1z_1^2 + 2\beta w =
0$, um \emph{cilindro parabólico}; se $\beta = 0$:
$\lambda_1z_1^2 = -c''$ dá dois planos paralelos ($ c''
\lambda_1 < 0$), um plano duplo ($ c'' = 0$), ou o vazio
definido. Para $(x + y)^2 = z $: com $ u = \frac{x + y}{\sqrt2}$ o
leituras de equações $ z = 2u^2$: um cilindro parabólico, invariante
sob traduções junto $(1, -1, 0)$.

\textbf{15.} Todo \omterm{pb:b2:surfaces:1}{quádrica} é transportado por uma rotação mais
traduções para um dos seguintes: (classificação 3) elipsóide, ponto, vazio
conjunto, hiperbolóide de uma folha, cone, hiperbolóide de duas folhas;
(classificação 2) parabolóide elíptico, parabolóide hiperbólico, elíptico
cilindro, linha, conjunto vazio, cilindro hiperbólico, par de
planos que se cruzam; (classificação 1) cilindro parabólico, par de
planos paralelos, plano duplo, conjunto vazio. Identificando o
três variantes vazias como distintas \emph{\omterm{def:b2:affine:subspace}{afim} tipos} de
equações, a contagem é dezessete; entre eles nove são honestos
superfícies: elipsóide, os dois hiperbolóides, o cone, os dois
parabolóides e os três cilindros.

\textbf{16.} Algoritmo. (i) Leia $(A, b, c)$; calcular o
polinômio característico de $ A $, isso é valores próprios (real, por
o teorema espectral) e $ r = \operatorname{rank}A $. (ii)
Resolver $ A\Omega = -b $ (Eliminação Gaussiana): solucionável ou não
--- centros ou não. (iii) Se for solucionável, transfira para um centro:
a equação se torna $\sum\lambda_iz_i^2 + c'' = 0$ com $ c''
= c + \langle b, \Omega\rangle $; ordenar por $ r $, a assinatura,
e o sinal de $ c''$ usando as questões 5, 10, 14. (iv) Se não
solucionável ($ r \leq 2$), gire e reduza como nas questões 10
e 14: parabolóide ($ r = 2$) ou cilindro parabólico ($ r = 1$),
elíptico/hiperbólico de acordo com o sinal de
$\lambda_1\lambda_2$. Cada etapa é um cálculo finito:
raízes de uma cúbica com raízes reais, postos, sistemas lineares.

\textbf{17.} $ A $ tem diagonal $1$, fora da diagonal $-1$: $ A = 2I
- J $, \omterm{def:b2:reduction:eigen}{espectro} $\{2 - 3,\ 2,\ 2\} = \{-1, 2, 2\}$ com $-1$
sobre $\R(1,1,1)$. Assinatura $(2,1)$, $ b = 0$, lado direito
$\delta = 1 > 0$: $2u^2 + 2v^2 - w^2 = 1$, uma folha
hiperbolóide de revolução em torno do eixo $\R(1,1,1)$.

\textbf{18.} Quadrados \omterm{def:b2:metric:complete}{completos}: $(x-1)^2 + (y+2)^2 - (z-1)^2 +
(-1 - 4 + 1 + 4) = 0$, ou seja
\[
(x-1)^2 + (y+2)^2 = (z-1)^2 :
\]
a constante desapareceu --- uma circular reta \emph{cone} com
vértice (e centro único) $(1, -2, 1)$ e eixo paralelo a
$ Oz $.

\textbf{19.} A parte quadrática $ x^2 + 4xy + y^2$ tem matriz
$\begin{psmallmatrix}1 & 2\\ 2 & 1\end{psmallmatrix}$ (no
$ xy $-avião), valores próprios $3$ (sobre $(1,1)$) e $-1$ (sobre
$(1,-1)$): com $ u = \frac{x+y}{\sqrt2}$, $ v =
\frac{x-y}{\sqrt2}$ é igual $3u^2 - v^2$, e o \omterm{pb:b2:surfaces:1}{quádrica}
é
\[
z = \tfrac32u^2 - \tfrac12v^2 :
\]
um parabolóide hiperbólico ($ A $ tem classificação $2$ e $ b $ tem um
componente ao longo $\ker A = \R e_z $: sem centro).

\textbf{20.} Um movimento rígido transforma os dados da equação por
$ A \mapsto P^{\mathsf T}\!AP $ (mesmo valores próprios) e o
formas normais não têm liberdade residual, exceto permutar
coordenadas e multiplicando toda a equação por um escalar
($> 0$ para preservar a escrita): duas formas normais centrais
coincidem até a isometria se as listas de coeficientes concordarem até
permutação e fator positivo comum --- para o elipsóide,
se os semieixos $(a, b, c)$ concordar. \omterm{def:b2:affine:subspace}{Afim}, pode-se também
dimensionar cada coordenada separadamente ($ z_i \mapsto z_i\sqrt{\abs
{\lambda_i}}$), que apaga o valores próprios e sai apenas
seus sinais: todo elipsóide se torna $ u^2 + v^2 + w^2 = 1$,
a esfera. Teorema da inércia de Sylvester
(\cref{thm:b2:quadratic:sylvester}) garante a assinatura
sobrevive a qualquer mudança linear invertível: o \omterm{def:b2:affine:subspace}{afim} tipos de
questão 15 são genuinamente distintas.

\textbf{21.} Parametrize o plano \omterm{def:b2:affine:subspace}{afim}: $ M = P + su +
tv $. Então $ q(P + su + tv)$ é um polinômio de grau $\leq 2$
em $(s, t)$ (expandir o forma quadrática bilinearmente), então o
seção $\{q = 0\}$ é uma cônica do plano, possivelmente
degenerar. Para $ z = xy $ e o avião $ z = c $: $ xy = c $, um
hipérbole para $ c \neq 0$, e para $ c = 0$ as duas coordenadas
linhas --- o par de decisões através da origem.

\textbf{22.} \emph{Elipsóide}: limitado, não contém linha.
\emph{Hiperbolóide de duas folhas} $\frac{z^2}{c^2} -
\frac{x^2}{a^2} - \frac{y^2}{b^2} = 1$: restringir a $ p + tv $;
o $ t^2$ coeficiente $\frac{v_3^2}{c^2} - \frac{v_1^2}{a^2}
- \frac{v_2^2}{b^2}$ deve desaparecer, então $ v_3 \neq 0$ (outro $ v =
0$); o $ t $ coeficiente dá $\frac{p_3v_3}{c^2} =
\frac{p_1v_1}{a^2} + \frac{p_2v_2}{b^2}$ e Cauchy-Schwarz
rendimentos
\[
\frac{p_3^2}{c^2}
= \frac{c^2}{v_3^2}\Bigl(\frac{p_1v_1}{a^2} +
\frac{p_2v_2}{b^2}\Bigr)^2
\leq \frac{c^2}{v_3^2}\Bigl(\frac{p_1^2}{a^2} +
\frac{p_2^2}{b^2}\Bigr)\frac{v_3^2}{c^2}
= \frac{p_1^2}{a^2} + \frac{p_2^2}{b^2},
\]
então o termo constante é $\leq 0 \neq 1$: sem linha.
\emph{Parabolóide elíptico} $ z = \frac{x^2}{a^2} +
\frac{y^2}{b^2}$: o $ t^2$ forças de coeficiente $ v_1 = v_2 =
0$, então a equação é linear não constante em $ t $: sem linha.
\emph{Cone} $ x^2 + y^2 = z^2$: uma linha nele satisfaz $ q(p) =
q(v) = B(p, v) = 0$ para o formulário Lorentz; igualdade no
avião Cauchy-Schwarz $\abs{p_1v_1 + p_2v_2} = \abs{p_3v_3} =
\sqrt{(p_1^2 + p_2^2)(v_1^2 + v_2^2)}$ forças $(p_1, p_2)
\parallel (v_1, v_2)$ e então $ p = kv $: todas as linhas passam
através do vértice --- uma família. \emph{Cilindros}: para o
cilindros elípticos e parabólicos $ t^2$ forças de coeficiente
$ v_1 = v_2 = 0$ (somente decisões); para o cilindro hiperbólico
$\frac{x^2}{a^2} - \frac{y^2}{b^2} = 1$, $\frac{v_1}a = \pm
\frac{v_2}b $ com $ v_2 \neq 0$ leva através do $ t $ coeficiente
para $\frac{p_1^2}{a^2} = \frac{p_2^2}{b^2}$, contradizendo o
termo constante $1$: novamente apenas as decisões verticais. Então o
duplamente governado \omterm{pb:b2:surfaces:1}{quádrica} superfícies são exatamente de uma folha
hiperbolóide e o parabolóide hiperbólico.

\textbf{23.} Gauss: $ x^2 - 2xy - 2zx = (x - y - z)^2 - y^2 -
z^2 - 2yz $, então
\[
q = (x - y - z)^2 - 4yz = (x - y - z)^2 + (y - z)^2 - (y +
z)^2 :
\]
três quadrados independentes com sinais $(+, +, -)$ --- assinatura
$(2, 1)$, correspondendo à questão 17, sem \omterm{def:b2:reduction:eigen}{autovalor}
computação. Gauss perde os dados métricos: as novas coordenadas
não são ortonormais, então valores próprios (a forma do
hiperbolóide, seus eixos e suas \omterm{def:b2:curves:length}{comprimentos}) desapareceram; apenas o
\omterm{def:b2:affine:subspace}{afim} tipo permanece.

\textbf{24.} Deixar $ p $, $ n $, $ z $ sejam os números positivos,
negativo e zero valores próprios, $ p + n + z = 3$. Descartes'
limites de regra $ p $ pelo número $ V $ de mudanças de sinal de
$\chi_A $, e $ n $ pelo número $ V'$ de mudanças de sinal de
$\chi_A(-\lambda)$; além disso, cada par de números consecutivos diferentes de zero
coeficientes produz uma mudança em exatamente um dos dois
polinômios, então $ V + V' \leq 3 - z $ (zero raízes são visíveis
como coeficientes finais nulos). Então $ p + n = 3 - z \geq
V + V' \geq p + n $: igualdade, então $ p = V $ exatamente --- o
sinais do valores próprios pode ser lido. Para $\chi_A(\lambda)
= \lambda^3 - 3\lambda^2 - 9\lambda - 5$: sinais $+,-,-,-$
dar $ V = 1$, e $\chi_A(-\lambda) = -\lambda^3 - 3\lambda^2
+ 9\lambda - 5$ tem sinais $-,-,+,-$: $ V' = 2$. Assinatura $(1,
2)$ --- consistente com a fatoração exata $\chi_A =
(\lambda - 5)(\lambda + 1)^2$ da questão 6.

\textbf{25.} Algoritmo: diagonalizar a parte quadrática
ortonormalmente (teorema espectral: este é o único
analiticamente passo profundo, e é o que faz o
classificação \emph{Euclidiano}); resolver $ A\Omega = -b $ para
decidir tipos centrais versus parabólicos e traduzir
a parte linear sempre que possível (\omterm{def:b2:affine:subspace}{afim} geometria); leia o
tipo de classificação, assinatura e constante (Sylvester
garante que estes são invariantes); quando apenas o \omterm{def:b2:affine:subspace}{afim} tipo
importa, a redução de Gauss substitui o teorema espectral em
o custo das informações métricas. Em $\R^2$ o mesmo
máquina classifica cônicas: elipse, hipérbole, parábola, mais
pares de retas, uma reta, um ponto e conjuntos vazios. Em $\R^n $
nada muda a não ser a escrituração contábil: os tipos são indexados pelo
assinatura de $ A $, a posição de $ b $ em relação a
$\operatorname{im}A $, e uma constante --- com o
$(n{+}2)$ matriz delimitada dimensional de $(A, b, c)$
fornecendo um \omterm{def:b2:metric:compact}{compactar} invariante.
\end{solution}
