\section*{अध्याय \ref{ch:b2:randomvar} --- असतत यादृच्छिक चर}

\begin{solution}{exo:b2:randomvar:1}
\emph{द्विपद:} $X = \sum_{i=1}^n X_i $ साथ \omterm{def:b2:proba:independence}{स्वतंत्र} Bernoulli
$ X_i $; $ V(X_i) = \E(X_i^2) - \E(X_i)^2 = p - p^2$, और प्रसरण
का \omterm{def:b2:proba:independence}{स्वतंत्र} चर \omterm{def:b2:quadratic:adjoint}{जोड़}ें
(\cref{thm:b2:randomvar:variancerules}):
\[
\E(X) = np, \qquad V(X) = np(1-p) .
\]
\emph{पॉइसन:} $\E\bigl(X(X-1)\bigr) =
\sum_{k\geq2}k(k-1)e^{-\lambda}\frac{\lambda^k}{k!} = \lambda^2
e^{-\lambda}\sum_{j\geq0}\frac{\lambda^j}{j!} = \lambda^2$, इसलिए
\[
V(X) = \E(X^2) - \E(X)^2
= \lambda^2 + \lambda - \lambda^2 = \lambda .
\]
\emph{ज्यामितिक} ($ q = 1 - p $): विभेद करना
$\sum_{k\geq0}q^k = \frac{1}{1-q}$ डिस्क के अंदर दो बार
(\cref{ch:b2:powerseries}), $\sum_{k\geq2}k(k-1)q^{k-2} =
\frac{2}{(1-q)^3}$, इसलिए
\[
\E\bigl(X(X-1)\bigr) = pq\sum_{k\geq2}k(k-1)q^{k-2}
= \frac{2q}{p^2},
\qquad
V(X) = \frac{2q}{p^2} + \frac1p - \frac{1}{p^2}
= \frac{q}{p^2} = \frac{1-p}{p^2} .
\]
\end{solution}

\begin{solution}{exo:b2:randomvar:2}
\emph{\omterm{def:b2:quadratic:adjoint}{जोड़}:} के लिए $ n \in \N $, असम्बद्धता से और स्वतंत्रता,
\[
\P(X + Y = n)
= \sum_{k=0}^n \P(X = k)\P(Y = n - k)
= e^{-(\lambda + \mu)}\frac{1}{n!}
\sum_{k=0}^n \binom nk \lambda^k\mu^{n-k}
= e^{-(\lambda+\mu)}\frac{(\lambda + \mu)^n}{n!}
\]
द्विपद प्रमेय द्वारा: $ X + Y \sim \mathcal{P}(\lambda + \mu)$.
\emph{सशर्त \omterm{def:b2:randomvar:law}{कानून}:} के लिए $0 \leq k \leq n $,
\[
\P(X = k \mid X + Y = n)
= \frac{\P(X = k)\P(Y = n - k)}{\P(X + Y = n)}
= \binom nk
\Bigl(\frac{\lambda}{\lambda+\mu}\Bigr)^{k}
\Bigl(\frac{\mu}{\lambda+\mu}\Bigr)^{n-k} ,
\]
द्विपद \omterm{def:b2:randomvar:law}{कानून} $\mathcal{B}\bigl(n,
\frac{\lambda}{\lambda+\mu}\bigr)$: प्रत्येक की कुल संख्या दी गई है
आयोजन \omterm{def:b2:proba:independence}{स्वतंत्र} रूप से संभाव्यता वाला पहला स्रोत ``चुनता'' है
इसकी दर के समानुपाती.
\end{solution}

\begin{solution}{exo:b2:randomvar:3}
कब $ k $ खिलौने अभी भी गायब हैं, हर नया डिब्बा एक नया खिलौना लाता है
संभाव्यता के साथ $\frac kn $, अतीत से \omterm{def:b2:proba:independence}{स्वतंत्र}:
प्रतीक्षा का समय $ W_k $ अगले नए खिलौने के लिए ज्यामितीय है
$\mathcal{G}\bigl(\frac kn\bigr)$, साथ $\E(W_k) = \frac nk $, और
$ T_n = W_n + W_{n-1} + \dots + W_1$ (पहला डिब्बा हमेशा नया देता है
खिलौना: $ W_n = 1$, के साथ संगत $\E = n/n $). रैखिकता से,
\[
\E(T_n) = \sum_{k=1}^n \frac nk = n\sum_{k=1}^n\frac1k
\sim n\ln n ,
\]
का उपयोग करते हुए $\sum_{k\leq n}\frac1k = \ln n + \gamma + o(1)$
(\cref{ch:b2:comparison}). आखिरी कुछ खिलौने इकट्ठा करना ही क्या है
लागत: आधे डिब्बे अंतिम मुट्ठी में चले जाते हैं।
\end{solution}

\begin{solution}{exo:b2:randomvar:4}
\omterm{def:b2:funcseq:def}{बिंदुवार}, $ X(\omega) = \#\{n \geq 1 : X(\omega) \geq n\} =
\sum_{n\geq1}\mathbf{1}_{X \geq n}(\omega)$. दोहरा परिवार
$\bigl(\mathbf{1}_{X \geq n}(\omega)\,\P(\{\omega\})\bigr)_{n,
\omega}$ गैर-नकारात्मक है, इसलिए परिवारों के लिए फ़ुबिनी
(\cref{ch:b2:series}) बिना शर्त लागू होता है: पहले संक्षेपण करें
$ n $ देता है $\E(X)$, पहले संक्षेप करें $\omega $ देता है $\sum_n \P(X
\geq n)$; दोनों एक साथ परिमित और समान हैं। के लिए $ X \sim
\mathcal{G}(p)$: $\P(X \geq n) = q^{n-1}$ ($ q = 1-p $), इसलिए $\E(X) =
\sum_{n\geq1}q^{n-1} = \frac{1}{1 - q} = \frac1p $.
\end{solution}

\begin{solution}{exo:b2:randomvar:5}
\emph{समरूपता:}  $ i $- निकाली गई गेंद एक \omterm{def:b2:funcseq:def}{समान रूप से} यादृच्छिक गेंद है
कलश का (कोई भी.) $ N $ गेंदों के अंदर गिरने की समान \omterm{def:b2:multint:exact}{संभावना} है
स्थिति $ i $ ड्राइंग क्रम का), तो $\P(Y_i = 1) = \frac MN =
p $ और $\E(X) = np $ रैखिकता से --- नहीं स्वतंत्रता आवश्यकता है।

\emph{\omterm{def:b2:randomvar:variance}{सहप्रसरण}:} के लिए $ i \neq j $, $\E(Y_iY_j) = \P(\text{draws }
i, j \text{ both white}) = \frac{M(M-1)}{N(N-1)}$ (\omterm{def:b2:quadratic:adjoint}{जोड़}ियों का \omterm{def:b2:structures:generated}{आदेश} दिया
अलग-अलग स्थिति में अलग-अलग गेंदों की एक क्रमबद्ध \omterm{def:b2:quadratic:adjoint}{जोड़}ी मिलती है,
\omterm{def:b2:funcseq:def}{समान रूप से})। इस तरह
\[
\operatorname{Cov}(Y_i, Y_j)
= \frac{M(M-1)}{N(N-1)} - \frac{M^2}{N^2}
= \frac{M(N - M)}{N^2}\cdot\frac{-1}{N-1}
= -\frac{p(1-p)}{N-1} < 0 :
\]
एक सफेद गेंद खींचने से अन्य ड्रॉ के लिए सफेद गेंदें दुर्लभ हो जाती हैं।
द्वारा \cref{thm:b2:randomvar:variancerules},
\[
V(X) = np(1-p) + n(n-1)\Bigl(-\frac{p(1-p)}{N-1}\Bigr)
= np(1-p)\,\frac{N - n}{N - 1} \leq np(1-p) :
\]
प्रतिस्थापन के बिना प्रतिचयन का माध्य समान होता है \emph{छोटे}
\omterm{def:b2:randomvar:variance}{झगड़ा} प्रतिस्थापन की तुलना में (केवल समानता के लिए)। $ n = 1$), द
नकारात्मक सहसंबंध एक स्थिरक के रूप में कार्य करते हैं। के लिए $ n = N $ 
\omterm{def:b2:randomvar:variance}{झगड़ा} गायब हो जाता है: तब गिनती नियतात्मक होती है।
\end{solution}

\begin{solution}{exo:b2:randomvar:6}
चारों ओर विस्तार हो रहा है $ m = \E(X)$:
\[
\E\bigl((X - c)^2\bigr)
= \E\bigl((X - m)^2\bigr) + 2(m - c)\,\E(X - m) + (m - c)^2
= V(X) + (m - c)^2 ,
\]
बिलकुल न्यूनतम $ c = m $ मूल्य के साथ $ V(X)$ --- \omterm{def:b2:randomvar:expectation}{अपेक्षा} है
माध्य वर्ग में सर्वोत्तम स्थिरांक भविष्यवक्ता।

अगर $\P(X = m) = 1$ तब $(X - m)^2$ \omterm{def:b2:multint:exact}{संभावना} के साथ गायब हो जाता है
$1$, इसलिए $ V(X) = 0$ (परिभाषित परिवार में a को छोड़कर शून्य पद हैं
शून्य सेट)। इसके विपरीत, यदि $ V(X) = 0$, चेबीशेव
(\cref{thm:b2:randomvar:markov}) देता है $\P\bigl(\abs{X - m} \geq
\frac1n\bigr) \leq n^2\,V(X) = 0$ हरएक के लिए $ n $; \omterm{def:b2:proba:space}{घटनाएँ}
$\bigl\{\abs{X - m} \geq \frac1n\bigr\}$ करने के लिए वृद्धि $\{X \neq
m\}$, इतना नीरस निरंतरता (\cref{thm:b2:proba:continuity})
पैदावार $\P(X \neq m) = 0$.
\end{solution}

\begin{solution}{exo:b2:randomvar:7}
$\E(S_n) = \frac n2$ और $ V(S_n) = \frac n4$.
\emph{मार्कोव:} $\P\bigl(S_n \geq \frac{3n}4\bigr) \leq
\frac{n/2}{3n/4} = \frac23$ --- एक \omterm{def:b2:metric:continuity}{निरंतर} बाध्य, के लिए बेकार
बड़ा $ n $.
\emph{चेबीशेव:}  आयोजन तात्पर्य $\abs{S_n - \frac n2} \geq
\frac n4$, तो \omterm{def:b2:multint:exact}{संभावना} है $\leq \frac{n/4}{(n/4)^2} =
\frac4n $ --- क्षय होता है, लेकिन केवल बहुपद रूप से।
\emph{चेर्नॉफ़:} द्वारा स्वतंत्रता, $\E(e^{tS_n}) =
\prod_{i=1}^n\E(e^{tX_i}) = \bigl(\frac{1 + e^t}{2}\bigr)^n $, और
मार्कोव ने आवेदन किया $ e^{tS_n} \geq e^{3nt/4}$ देता है, हर एक के लिए $ t >
0$,
\[
\P\Bigl(S_n \geq \frac{3n}4\Bigr)
\leq \Bigl(\frac{1 + e^t}{2}\Bigr)^n e^{-3nt/4}
= \exp\Bigl(n\bigl(\ln\tfrac{1 + e^t}{2} - \tfrac{3t}4\bigr)\Bigr).
\]
घातांक को छोटा करें: $\frac{\dd}{\dd t}\ln\frac{1+e^t}{2} =
\frac{e^t}{1 + e^t} = \frac34$ पर $ e^t = 3$, यानी\ $ t = \ln 3$,
देना
\[
\P\Bigl(S_n \geq \frac{3n}4\Bigr)
\leq \Bigl(\frac{4}{2}\Bigr)^n 3^{-3n/4}
= \bigl(2 \cdot 3^{-3/4}\bigr)^n \approx (0.877)^n ,
\]
तेजी से छोटा. पदानुक्रम मार्कोव $\to $ Chebyshev $\to $
चेर्नॉफ़ मानक सीढ़ी है: प्रत्येक सीढ़ी मार्कोव को लागू करती है
वेरिएबल का तेजी से बढ़ने वाला कार्य।
\end{solution}

\begin{solution}{exo:b2:randomvar:8}
\omterm{def:b2:structures:sn}{स्थानांतरण} प्रमेय द्वारा (\cref{thm:b2:randomvar:transfer})
के लिए आवेदन किया $ f\bigl(\frac{S_n}{n}\bigr)$ साथ $ S_n \sim
\mathcal{B}(n, x)$:
\[
\E\Bigl[f\Bigl(\frac{S_n}{n}\Bigr)\Bigr]
= \sum_{k=0}^n f\Bigl(\frac kn\Bigr)\binom nk x^k(1-x)^{n-k}
= B_nf(x) .
\]
हल करना $\delta > 0$ और विभाजित $\abs{f(S_n/n) - f(x)}$ पर आयोजन $ D
= \bigl\{\abs{\frac{S_n}{n} - x} \geq \delta\bigr\}$:
बंद $ D $, \omterm{def:b2:diffcalc:differential}{अंतर} अधिकतम मापांक का है निरंतरता
$\omega_f(\delta) = \sup_{\abs{s - t}\leq\delta}\abs{f(s) -
f(t)}$; पर $ D $, अधिक से अधिक $2\norm f_\infty $. लेना अपेक्षाएं और
के साथ चेबीशेव का उपयोग करना $ V\bigl(\frac{S_n}{n}\bigr) =
\frac{x(1-x)}{n} \leq \frac{1}{4n}$:
\[
\abs{B_nf(x) - f(x)}
\leq \E\,\abs{f(S_n/n) - f(x)}
\leq \omega_f(\delta)
+ 2\norm f_\infty\,\P(D)
\leq \omega_f(\delta) + \frac{2\norm f_\infty}{4n\delta^2} .
\]
वर्दी निरंतरता का $ f $ पर $[0, 1]$ बनाता है $\omega_f(\delta) \to
0$: चुनना $\delta $ तब $ n $, और $ B_nf \to f $ \omterm{def:b2:funcseq:def}{समान रूप से} --- द
वीयरस्ट्रैस सन्निकटन प्रमेय \cref{ch:b2:funcseq}, किसका
``काउंटिंग लेम्मा'' अब चेबीशेव की असमानता के रूप में पहचानी जाने योग्य है
द्विपद के लिए \omterm{def:b2:randomvar:law}{कानून}.
\end{solution}

\begin{solution}{exo:b2:randomvar:9}
बढ़ाना $ S_n^4 = \sum_{i,j,k,l}X_iX_jX_kX_l $ और ले लो
अपेक्षाएं. द्वारा स्वतंत्रता और केन्द्रित करना, कोई भी शब्द युक्त
ठीक एक बार दिखने वाला सूचकांक गायब हो जाता है ($\E(X_i) = 0$ कारक
बाहर). जीवित शर्तें: $ n $ विकर्ण पद $\E(X_i^4)$, और
समान सूचकांकों के दो युग्मों को जोड़ने वाले पद, $\E(X_i^2X_j^2) =
\E(X_1^2)^2$ के लिए $ i \neq j $, उपस्थिति $3n(n-1)$ समय: चुनें
मूल्यों की अव्यवस्थित \omterm{def:b2:quadratic:adjoint}{जोड़}ी ($\binom n2$ तरीके), फिर
$\frac{4!}{2!\,2!} = 6$ उन्हें चार खांचों में रखने के तरीके ---
$6\binom n2 = 3n(n-1)$. इसलिए, साथ $\E(X_1^2)^2
\leq \E(X_1^4)$ (जेन्सेन या कॉची-श्वार्ज़),
\[
\E(S_n^4) = n\,\E(X_1^4) + 3n(n-1)\,\E(X_1^2)^2
\leq C n^2,
\qquad C = 4\,\E(X_1^4) .
\]
मार्कोव क्रम 4 पर:
\[
\P\Bigl(\Bigl|\frac{S_n}{n}\Bigr| \geq \varepsilon\Bigr)
= \P\bigl(S_n^4 \geq n^4\varepsilon^4\bigr)
\leq \frac{Cn^2}{n^4\varepsilon^4}
= \frac{C}{n^2\varepsilon^4} ,
\]
ए \omterm{def:b2:series:summable}{संक्षेपणीय} शृंखला। बोरेल द्वारा--कैंटेली~1
(\cref{thm:b2:proba:borelcantelli}), प्रत्येक के लिए $ j $  आयोजन
$ B_j = \limsup_n\bigl\{\abs{S_n/n} \geq \frac1j\bigr\}$ है
संभाव्यता $0$, इसलिए $\P\bigl(\bigcup_j B_j\bigr) = 0$ द्वारा
\omterm{def:b2:structures:countable}{गणनीय} उपादेयता. संभाव्यता के पूरक --- पर $1$
--- हरएक के लिए $ j $ वहाँ है $ N $ साथ $\abs{S_n/n} < \frac1j $ सभी के लिए $ n \geq N $: \omterm{def:b2:multint:exact}{एकदम सही} $ S_n/n \to 0$. बड़े का मजबूत \omterm{def:b2:randomvar:law}{कानून}
संख्याएँ चौथे क्षण के \omterm{def:b2:diffcalc:differential}{अंतर}्गत रहती हैं; उस परिकल्पना को दूर करना
(कोलमोगोरोव का प्रमेय) वर्ष 3 का कार्य है।
\end{solution}

\begin{solution}{exo:b2:randomvar:11}
साथ $ I_i = \mathbf 1_{\sigma(i) = i}$: $\P(\sigma(i) = i) =
\frac{(n-1)!}{n!} = \frac1n $, इसलिए $\E(F_n) = n\cdot\frac1n =
1$. के लिए $ i \neq j $: $\P(\sigma(i) = i, \sigma(j) = j) =
\frac{(n-2)!}{n!} = \frac1{n(n-1)}$, इस तरह
\[
\operatorname{Cov}(I_i, I_j) = \frac1{n(n-1)} - \frac1{n^2}
= \frac{1}{n^2(n-1)} .
\]
से \omterm{def:b2:randomvar:variance}{झगड़ा} टूलकिट
(\cref{thm:b2:randomvar:variancerules}),
\[
V(F_n) = n\cdot\frac1n\Bigl(1 - \frac1n\Bigr)
+ n(n-1)\cdot\frac1{n^2(n-1)}
= 1 - \frac1n + \frac1n = 1 .
\]
अर्थ $1$, \omterm{def:b2:randomvar:variance}{झगड़ा} $1$, से \omterm{def:b2:proba:independence}{स्वतंत्र} $ n $ --- सुसंगत
मिलान समस्या की पॉइसन सीमा के साथ
(\cref{exo:b2:proba:5}).
\end{solution}

\begin{solution}{exo:b2:randomvar:12}
$ T_n = \sum_{k=1}^nG_k $ कहाँ $ G_k \sim \mathcal G(k/n)$ है
एक नया खिलौना देखने का समय कब है $ k $ गायब हैं, चरण
किया जा रहा है \omterm{def:b2:proba:independence}{स्वतंत्र}. इस तरह
\[
V(T_n) = \sum_{k=1}^n\frac{1 - k/n}{(k/n)^2}
\leq \sum_{k=1}^n\frac{n^2}{k^2}
\leq \frac{\pi^2}6\,n^2 ,
\]
द्वारा \cref{ex:b2:fourier:basel}. साथ $\E(T_n) = nH_n $, $ H_n =
\sum_1^n\frac1k $ (\cref{exo:b2:randomvar:3}), चेबीशेव देता है,
के लिए $\varepsilon > 0$,
\[
\P\bigl(\abs{T_n - nH_n} \geq \varepsilon\,n\ln n\bigr)
\leq \frac{\pi^2n^2/6}{\varepsilon^2n^2\ln^2 n}
= \frac{\pi^2}{6\,\varepsilon^2\ln^2n}
\xrightarrow[n\to\infty]{} 0 .
\]
तब से $ H_n \sim \ln n $, से विभाजित करना $ n\ln n $ दिखाता है
$ T_n/(n\ln n) \to 1$ संभाव्यता में: के उतार-चढ़ाव
$ T_n $ व्यवस्थित हैं $ n $, माध्य के विरुद्ध नगण्य $ n\ln n $.
\end{solution}

\begin{solution}{pb:b2:randomvar:1}
\textbf{1.} $\E(S_n) = \frac n2$ और मार्कोव
(\cref{thm:b2:randomvar:markov}) देना $\P(S_n \geq an) \leq
\frac{n/2}{an} = \frac1{2a}$. मार्कोव केवल माध्य जानता है: यह
पर केंद्रित एक चर को अलग नहीं कर सकता $ n/2$ एक से
के बीच फैल गया $0$ और $ n $, इसलिए यह पूंछ की कीमत सभी की तरह है
जनसमूह वहां बैठ सकता था।

\textbf{2.} उचित द्विपद है \omterm{def:b2:quadratic:adjoint}{सममित} के बारे में $ n/2$
($ S_n $ और $ n - S_n $ एक ही लें \omterm{def:b2:randomvar:law}{कानून}), के साथ $ x = n(a -
\frac12) > 0$ दो \omterm{def:b2:proba:space}{घटनाएँ} $\{S_n - \frac n2 \geq x\}$ और
$\{S_n - \frac n2 \leq -x\}$ असंयुक्त और समसंभाव्य हैं:
$\P(S_n \geq an) = \frac12\P(\abs{S_n - \frac n2} \geq x)$.
चेबीशेव के साथ $ V(S_n) = \frac n4$:
\[
\P(S_n \geq an)
\leq \frac12\cdot\frac{n/4}{n^2(a - 1/2)^2}
= \frac1{8n(a - 1/2)^2},
\]
जो है $\frac2n $ पर $ a = \frac34$.

\textbf{3.} द्वारा स्वतंत्रता और उत्पाद प्रमेय,
$\E(\eu^{tS_n}) = \bigl(\E \eu^{tX_1}\bigr)^n = \bigl(\frac{1
+ \eu^t}2\bigr)^n $. मार्कोव ने आवेदन किया $\eu^{tS_n}$:
\[
\P(S_n \geq an) \leq \eu^{-tan}\Bigl(\frac{1 +
\eu^t}2\Bigr)^{\!n} = \exp\Bigl(n\bigl(\ln\tfrac{1 +
\eu^t}2 - ta\bigr)\Bigr).
\]
प्रतिपादक की व्युत्पत्ति $ t $ है $\frac{\eu^t}{1 + \eu^t}
- a $, गायब हो रहा है $\eu^t = \frac a{1-a}$, यानी\ $ t^* =
\ln\frac a{1-a} > 0$; वहाँ $\frac{1 + \eu^{t^*}}2 =
\frac1{2(1-a)}$ और घातांक बराबर है
\[
n\Bigl(-\ln 2 - \ln(1-a) - a\ln\frac a{1-a}\Bigr)
= -n\bigl(\ln2 + a\ln a + (1-a)\ln(1-a)\bigr) = -n\,I(a),
\]
साथ $ I(\frac12) = 0$ और $ I'(a) = \ln\frac a{1-a} > 0$ पर
$\intoo{\frac12}1$: $ I(a) > 0$. पर $ a = \frac34$:
$\eu^{-I(3/4)} = \frac12(\tfrac34)^{-3/4}(\tfrac14)^{-1/4} =
2\cdot3^{-3/4}$, की सीमा \cref{exo:b2:randomvar:7}.

\textbf{4.}  $ n + 1$ नंबर $\binom nja^j(1-a)^{n-j}$ योग
को $1$, और सबसे बड़ा वह है जो पर है $ j = k = an $ (मोड
का $\mathcal B(n, a)$ है $\floor{(n+1)a} = k $ यहाँ)। ए
अधिकतम $ n + 1$ संख्याओं का योग $1$ कम से कम है
$\frac1{n+1}$:
\[
\binom nk a^k(1-a)^{n-k} \geq \frac1{n+1}
\quad\Longrightarrow\quad
\binom nk \geq \frac{a^{-an}(1-a)^{-n(1-a)}}{n+1}
= \frac{\eu^{nH(a)}}{n+1}.
\]
इस तरह $\P(S_n \geq an) \geq \binom{n}{an}2^{-n} \geq
\eu^{n(H(a) - \ln2)}/(n+1) = \eu^{-nI(a)}/(n+1)$: तक
बहुपद कारक $ n + 1$, चेर्नॉफ़ का प्रतिपादक सत्य है।

\textbf{5.} $ n = 100$, $ a = \frac34$: मार्कोव $\frac23$;
Chebyshev $\frac2{100} = 0.02$; चेर्नॉफ़
$(2\cdot3^{-3/4})^{100} = \eu^{-100\,I(3/4)} \approx
2.1\cdot10^{-6}$, सटीक के विरुद्ध $2.8\cdot10^{-7}$. नैतिक:
सूचना का प्रत्येक क्षण बाउंड को बहुपद रूप से विभाजित करता है;
घातांकीय क्षण इसे बदल देता है \emph{प्रकृति}.

\textbf{6.} $\cosh t = \sum_{k\geq0}\frac{t^{2k}}{(2k)!}$ और
$\eu^{t^2/2} = \sum_{k\geq0}\frac{t^{2k}}{2^kk!}$; दावा
से पद दर पद अनुसरण करता है $(2k)! \geq 2^kk!$, जो धारण करता है
प्रेरण: $(2k)! = 2k(2k-1)\cdot(2k-2)! \geq 2k\cdot
2^{k-1}(k-1)! = 2^kk!\cdot(2k-1) \geq 2^kk!$.

\textbf{7.} द्वारा स्वतंत्रता, $\E\bigl(\eu^{t\sum\varepsilon_i}
\bigr) = (\cosh t)^n \leq \eu^{nt^2/2}$, तो मार्कोव देता है
$\P(\sum\varepsilon_i \geq s) \leq \eu^{nt^2/2 - ts}$;
पर न्यूनतम करना $ t = s/n $ पैदावार $\eu^{-s^2/(2n)}$.

\textbf{8.} साथ $ X_i = \frac{1 + \varepsilon_i}2$, $\widehat
p_n - \frac12 = \frac1{2n}\sum\varepsilon_i $, इसलिए
$\{\widehat p_n - \frac12 \geq \delta\} =
\{\sum\varepsilon_i \geq 2n\delta\}$ और प्रश्न 7 देता है
बाध्य $\eu^{-(2n\delta)^2/(2n)} = \eu^{-2n\delta^2}$. द
\omterm{def:b2:quadratic:adjoint}{सममित} घटना की वही सीमा होती है, जहां से कारक निकलता है $2$ के लिए
$\abs{\widehat p_n - \frac12} \geq \delta $.

\textbf{9.} $\E(\eu^{tX}) = \sum_x\eu^{tx}\P(X = x)$ एक है
के सुचारु कार्यों की श्रृंखला $ t $ जिसका टर्म-दर-टर्म
हर एक पर डेरिवेटिव का बोलबाला है \omterm{def:b2:metric:compact}{कॉम्पैक्ट} $ t $-अंतराल, द्वारा
$\eu^{\abs t}\P(X = x)$ (जैसा $0 \leq x \leq 1$): द्वारा
के लिए विभेदन प्रमेय \omterm{def:b2:funcseq:series}{सामान्यतः} \omterm{def:b2:integration:improper}{अभिसरण} श्रृंखला
(\cref{thm:b2:funcseq:differentiation}) यह दो बार है
\omterm{def:b2:diffcalc:differential}{विभेदक}, और भागफल नियम देता है $\psi' = \E_t(X)$
और $\psi'' = \E_t(X^2) - \E_t(X)^2$, कहाँ $\E_t $ है
\omterm{def:b2:randomvar:expectation}{अपेक्षा} पुनर्भारित वजन के लिए
$\eu^{tx}\P(X{=}x)/\E(\eu^{tX})$ --- गैर-नकारात्मक, संक्षेप में
$1$, समान मूल्यों द्वारा किया जाता है $ x \in \intcc01$. ए \omterm{def:b2:randomvar:variance}{झगड़ा}
एक का $\intcc01$-वैल्यूड वैरिएबल अधिकतम है $\frac14$: द्वारा
\cref{exo:b2:randomvar:6}, यह है $\min_c\E_t((X - c)^2) \leq
\E_t\bigl((X - \tfrac12)^2\bigr) \leq \tfrac14$. टेलर के साथ
अभिन्न शेष, का उपयोग कर $\psi(0) = 0$, $\psi'(0) = p $:
\[
\psi(t) = tp + \int_0^t(t - s)\,\psi''(s)\,\dd s
\leq tp + \frac{t^2}2\cdot\frac14,
\]
यानी\ $\E(\eu^{t(X - p)}) \leq \eu^{t^2/8}$ सभी वास्तविक के लिए $ t $.

\textbf{10.} द्वारा स्वतंत्रता, $\E\bigl(\eu^{t(S_n -
np)}\bigr) \leq \eu^{nt^2/8}$; मार्कोव और अनुकूलन $ t
= 4\delta $ देना
\[
\P(\widehat p_n - p \geq \delta)
\leq \eu^{nt^2/8 - tn\delta}\Big|_{t = 4\delta}
= \eu^{-2n\delta^2};
\]
इसे वेरिएबल्स पर लागू करना $1 - X_i $ (में भी
$\intcc01$) दूसरी पूंछ को बांधता है, जहां से दो-तरफा
$2\eu^{-2n\delta^2}$.

\textbf{11.} चेबीशेव को केवल एक दूसरे क्षण की आवश्यकता है और वह देता है
$\frac{p(1-p)}{n\delta^2}$; होफ़डिंग की ज़रूरतें
\emph{सीमाबद्धता} और देता है $2\eu^{-2n\delta^2}$. पर $ p =
\frac12$, $\delta = 0.03$: सीमाएं हैं
$\frac{278}{n}$ (लगभग) बनाम
$2\eu^{-0.0018n}$; वे पास से गुजरते हैं $ n \approx 1200$, के बाद
जो घातांकीय बाउंड जीतता है, और बेतहाशा ऐसा ($ n = 5000$:
$0.056$ बनाम $2.5\cdot10^{-4}$).

\textbf{12.} होफ़डिंग द्वारा (प्रश्न 10), $\P(\abs{\widehat
p_n - p} \geq \delta) \leq 2\eu^{-2n\delta^2} \leq \alpha $ जैसे ही $2n\delta^2 \geq \ln\frac2\alpha $, यानी\ $ n \geq
\frac{\ln(2/\alpha)}{2\delta^2}$.

\textbf{13.} $\delta = 0.03$, $\alpha = 0.05$: $ n \geq
\frac{\ln 40}{2\cdot0.0009} \approx 2049.4$: $2050$ लोग।
के लिए $\delta = 0.01$: $ n \geq \frac{\ln40}{0.0002} \approx
18\,445$. जनसंख्या का आकार कभी प्रकट नहीं होता क्योंकि प्रत्येक
नमूना मतदाता को ताज़ा बर्नौली के रूप में तैयार किया गया है $(p)$ ड्रा: द
पोल की कठिनाई है \omterm{def:b2:randomvar:variance}{झगड़ा} एक सिक्के का, आकार का नहीं
देश. मार्जिन आधा करने पर नमूने की लागत चार गुना हो जाती है
--- द $1/\delta^2$ \omterm{def:b2:randomvar:law}{कानून}।

\textbf{14.} चेबीशेव: $\P(\abs{\widehat p_n - p} \geq
\delta) \leq \frac{p(1-p)}{n\delta^2} \leq
\frac1{4n\delta^2} \leq \alpha $ के लिए $ n \geq
\frac1{4\alpha\delta^2}$, यानी\ $5556$ तीन बिंदुओं पर ---
के बारे में $2.7$ होफ़डिंग आवश्यकता से कई गुना अधिक। बिना
प्रतिस्थापन, \omterm{def:b2:randomvar:variance}{झगड़ा} से गुणा किया जाता है $\frac{N -
n}{N-1} < 1$ (\cref{exo:b2:randomvar:5}), तो वही $ n $ कर सकते हैं
केवल बेहतर करें: प्रतिस्थापन के साथ गणना है
रूढ़िवादी.

\textbf{15.} $\{\widehat p_n \leq \frac12\} \subseteq
\{\widehat p_n - 0.52 \leq -0.02\}$, तो एक तरफा से
होफ़डिंग बाध्य $\P(\widehat p_n \leq \tfrac12) \leq
\eu^{-2n(0.02)^2} \leq 0.01$ जैसे ही $ n \geq \frac{\ln
100}{2\cdot0.0004} \approx 5756.5$: $5757$ मतदाता. लागत
तराजू के व्युत्क्रम वर्ग की तरह \emph{नेतृत्व करना}, का नहीं
वांछित परिशुद्धता: नजदीकी दौड़ महंगी होती है।

\textbf{16.} प्रयुक्त: नमूना \omterm{def:b2:funcseq:def}{समान रूप से} खींचा जाता है और
निर्वाचक मंडल से \omterm{def:b2:proba:independence}{स्वतंत्र} रूप से; प्रत्येक नमूना व्यक्ति
उत्तर, ईमानदारी से, और $ p $ मतदान के दौरान हिलता नहीं है. असली
सर्वेक्षण तीनों का उल्लंघन करते हैं: पहुंच योग्य और इच्छुक उत्तरदाता
एक समान नमूना नहीं हैं (चयन और गैर-प्रतिक्रिया पूर्वाग्रह),
और उत्तर असत्य या अस्थिर हो सकते हैं। ये हैं
\emph{पक्षपात} त्रुटियाँ: वे स्थानांतरित हो जाते हैं $\E(\widehat p_n)$ से दूर
$ p $ से \omterm{def:b2:proba:independence}{स्वतंत्र} राशि से $ n $, इसलिए कोई नमूना आकार नहीं
उन्हें कम करता है --- इस भाग का गणित ही नियंत्रित करता है
उतार-चढ़ाव अवधि.

\textbf{17.} आकार के एक समूह के लिए चेबीशेव $ m $: $\P(\abs{
\widehat p^{(i)} - p} \geq \delta) \leq \frac{1}{4m\delta^2}
\leq \frac18$ के लिए $ m \geq \frac2{\delta^2}$. यदि इससे कम हो
$ k/2$ समूह गलती करते हैं, फिर इससे भी अधिक $ k/2$ मूल्यों का
$\widehat p^{(i)}$ में झूठ बोलना \omterm{def:b2:metric:topology}{खुला} \omterm{def:b2:diffcalc:differential}{अंतर}ाल $\intoo{p -
\delta}{p + \delta}$, और उनका माध्यिका भी ऐसा ही है; इस तरह
$\{\abs{M - p} \geq \delta\}$ कम से कम बल $\lceil
k/2\rceil $ के बीच त्रुटियाँ $ k $ \omterm{def:b2:proba:independence}{स्वतंत्र} समूह. संघ
के ऊपर बंधा हुआ $\binom k{\lceil k/2\rceil}$ के संभावित सेट
गलती करने वाले समूह देते हैं
\[
\P(\abs{M - p} \geq \delta)
\leq \binom{k}{\lceil k/2\rceil}
\Bigl(\frac18\Bigr)^{k/2}
\leq 2^k\,8^{-k/2} = 2^{-k/2} :
\]
खरीदे गए समूहों की संख्या में तेजी से गिरावट
इसके अलावा कुछ नहीं प्रसरण --- सटीक रूप से उपयोगी जब सारांश
असीमित हैं और होफ़डिंग अनुपलब्ध है।

\textbf{18.} कॉची-श्वार्ज़
(\cref{thm:b2:randomvar:jensen}):
\[
\E(X) = \E(X\,\mathbf 1_{X>0})
\leq \sqrt{\E(X^2)}\sqrt{\E(\mathbf 1_{X>0}^2)}
= \sqrt{\E(X^2)\,\P(X > 0)} ;
\]
वर्गाकार करें और विभाजित करें।

\textbf{19.} होने देना $ h(a) = I(a) - 2(a - \tfrac12)^2$. तब
$ h(\tfrac12) = 0$, $ h'(a) = \ln\frac a{1-a} - 4(a -
\tfrac12)$ पर गायब हो जाता है $\tfrac12$, और
\[
h''(a) = \frac1a + \frac1{1-a} - 4 = \frac{1}{a(1-a)} - 4
\geq 0
\]
तब से $ a(1-a) \leq \frac14$. इसलिए $ h'$ से बढ़ता है $0$ पर
$\intco{\frac12}1$, इस तरह $ h' \geq 0$ और $ h \geq 0$: $ I(a)
\geq 2(a - \tfrac12)^2$.

\textbf{20.} $ I(\tfrac12) = I'(\tfrac12) = 0$, $ I''(a) =
\frac1{a(1-a)}$ देता है $ I''(\tfrac12) = 4$, और $ I'''(\tfrac12)
= 0$ (फ़ंक्शन है \omterm{def:b2:quadratic:adjoint}{सममित} के बारे में $\tfrac12$), इसलिए
$ I(\tfrac12 + \delta) = 2\delta^2 + O(\delta^4)$. प्रश्न 4
फिर असली पूंछ को बांधता है \emph{नीचे} द्वारा
$\eu^{-n(2\delta^2 + O(\delta^4))}/(n+1)$: छोटे के लिए $\delta $
होफ़डिंग प्रतिपादक $2n\delta^2$ स्पर्शोन्मुख रूप से सटीक है
--- केवल बहुपद-में-$ n $ सुधार संभव है.

\textbf{21.} मार्कोव: एक क्षण, क्षय $1/a $, केवल के रूप में उपयोगी है
दूसरों के पीछे इंजन (प्रश्न 1 इसे सपाट दिखाता है)।
चेबीशेव: दो क्षण, क्षय $\frac{V}{n\delta^2}$, तेज़
बिना किसी और परिकल्पना के
(\cref{ex:b2:randomvar:chebsharp}), और सबसे अच्छा उपकरण
प्रश्न 23. चौथा क्षण (\cref{exo:b2:randomvar:9}):
क्षय $ C/n^2$, अभी काफी संक्षेपण एक मजबूत \omterm{def:b2:randomvar:law}{कानून} के लिए.
होफ़डिंग: परिबद्ध चर, क्षय $2\eu^{-2n\delta^2}$,
भाग III का कार्यकर्ता. सटीक दर के साथ चेर्नॉफ़
$ I(a)$: पूर्ण घातीय क्षण, अपराजेय घातांक
(प्रश्न 4,20), बाकी सभी चीजों के लिए संदर्भ बिंदु।

\textbf{22.} यदि सिक्का उचित है: $\P(\widehat p_n > 0.525)
\leq \P(\widehat p_n - \tfrac12 \geq 0.025) \leq
\eu^{-2n(0.025)^2}$. अगर $ p = 0.55$: $\P(\widehat p_n \leq
0.525) \leq \P(\widehat p_n - 0.55 \leq -0.025) \leq
\eu^{-2n(0.025)^2}$. दोनों त्रुटियाँ नीचे हैं $0.01$ कब
$2n(0.025)^2 \geq \ln 100$, यानी\ $ n \geq 3684.2$: $3685$
फ़्लिप. (परिकल्पनाओं को अलग करना $2.5$ अलग-अलग लागतों को इंगित करता है
क्या आकलन करना है $\pm2.5$ अंक लागत.)

\textbf{23.} होफ़डिंग: $ n \geq \frac{\ln 40}{2(0.005)^2}
\approx 73\,778$. सच के साथ चेबीशेव \omterm{def:b2:randomvar:variance}{झगड़ा} $ p(1-p)
= 0.0099$: $ n \geq \frac{0.0099}{0.05\cdot(0.005)^2} =
7920$ --- नौ गुना सस्ता। होफ़डिंग के प्रतिपादक
$2n\delta^2$ कीमतें \omterm{def:b2:randomvar:variance}{झगड़ा} सबसे खराब स्थिति में $\frac14$,
बेतुका निराशावादी जब $ p = 0.01$; विनम्र दूसरा
क्षण बेहतर जानता है. गुम उपकरण एक विचरण-जागरूक है
घातीय सीमा (बर्नस्टीन की असमानता, वर्ष 3) --- या,
दुर्लभ के लिए \omterm{def:b2:proba:space}{घटनाएँ}, पॉइसन सन्निकटन सिद्ध हुआ
\cref{ch:b2:genfun}, जो प्राकृतिक सापेक्ष पर कार्य करता है
स्केल.

\textbf{24.} हल करना $\delta > 0$: $\sum_n 2\eu^{-2n\delta^2} <
\infty $ (ज्यामितीय-प्रकार श्रृंखला), इसलिए बोरेल--कैंटेली~1
(\cref{thm:b2:proba:borelcantelli}) देता है
$\P(\abs{\widehat p_n - p} \geq \delta \text{ infinitely
often}) = 0$, यानी\  आयोजन $ E_j =
\bigcup_N\bigcap_{n\geq N}\{\abs{\widehat p_n - p} <
\tfrac1j\}$ \omterm{def:b2:multint:exact}{संभावना} है $1$ प्रत्येक के लिए $ j $. \omterm{def:b2:structures:countable}{गणनीय}
चौराहा $\bigcap_jE_j $ अभी भी \omterm{def:b2:multint:exact}{संभावना} है $1$
(पूरकों पर उप-अध्ययन), और उस पर $\widehat p_n
\to p $: सिक्का उछालने के लिए बड़ी संख्या का मजबूत नियम
सीमाबद्धता वह भूमिका निभा रही है जो चौथे क्षण ने निभाई
\cref{exo:b2:randomvar:9}.

\textbf{25.} एक पल एक फ्लैट बाउंड खरीदता है; दो खरीदें
$1/(n\delta^2)$, और नहीं (तीक्ष्णता उदाहरण); चार
खरीदो $1/n^2$, लगभग निश्चित \omterm{def:b2:randomvar:law}{कानून} की दूरबीन से जांच करने के लिए पर्याप्त;
सीमाबद्धता खरीदती है $\eu^{-2n\delta^2}$; और पूर्ण
घातीय क्षण सटीक दर खरीदता है $ I $, जिसकी कोई विधि नहीं
धड़कता है. मतदान $2050$ किसी भी देश के लिए लोग ही पर्याप्त होते हैं क्योंकि
नमूने का उतार-चढ़ाव सिक्के द्वारा नियंत्रित होता है \omterm{def:b2:randomvar:variance}{झगड़ा},
जनसंख्या का आकार नहीं --- $1/\delta^2$ और
$\ln(1/\alpha)$ मूल्य टैग सार्वभौमिक हैं. वर्ष 3 खंड
केंद्रीय सीमा प्रमेय इन असमानताओं को प्रतिस्थापित करता है
$\sqrt n$ पैमाने, एक सटीक सीमा से \omterm{def:b2:randomvar:law}{कानून} स्पष्ट के साथ
स्थिरांक --- इस समस्या की प्रत्येक सीमा को एक में बदलना
स्पर्शोन्मुख समानता.
\end{solution}
