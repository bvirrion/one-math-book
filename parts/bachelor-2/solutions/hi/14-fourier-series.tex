\section*{अध्याय \ref{ch:b2:fourier} --- फूरियर श्रृंखला}

\begin{solution}{exo:b2:fourier:1}
विषमता मार डालती है $a_n $. के लिए $ n \geq 1$:
\[
b_n = \frac{2}{\pi}\int_0^{\pi} \sin nt\,\dd t
= \frac{2}{\pi}\cdot\frac{1 - (-1)^n}{n}
= \begin{cases} \frac{4}{\pi n} & n \text{ odd},\\ 0 & n
\text{ even}. \end{cases}
\]
इसलिए $ S(f)(t) = \frac{4}{\pi}\sum_{k\geq0} \frac{\sin\bigl((2k+1)t
\bigr)}{2k+1}$. डिरिचलेट पर $ t = \frac\pi2$ (ए निरंतरता बिंदु मान $1$): $\sin\bigl((2k+1)\frac\pi2\bigr) = (-1)^k $, देना
\[
1 = \frac4\pi \sum_{k\geq0}\frac{(-1)^k}{2k+1}
\quad\Longrightarrow\quad
\sum_{k\geq0}\frac{(-1)^k}{2k+1} = \frac{\pi}{4}
\quad\text{(Leibniz)}.
\]
छलांग पर $ t = 0$: श्रृंखला का सार है $0 = \frac{f(0^+) +
f(0^-)}{2}$, जैसा कि डिरिचलेट निर्धारित करता है (प्रत्येक शब्द गायब हो जाता है:
सुसंगत)।
\end{solution}

\begin{solution}{exo:b2:fourier:2}
समता मार देती है $ b_n $; $ a_0 = \frac{1}{\pi}\int_{-\pi}^\pi\abs
t\,\dd t = \pi $, और के लिए $ n \geq 1$:
\[
a_n = \frac{2}{\pi}\int_0^\pi t\cos nt\,\dd t
= \frac{2}{\pi}\cdot\frac{(-1)^n - 1}{n^2}
= \begin{cases} -\frac{4}{\pi n^2} & n \text{ odd},\\ 0 & n
\text{ even}, \end{cases}
\]
(भागों द्वारा एक एकीकरण)। इस तरह
\[
\abs t = \frac{\pi}{2} - \frac{4}{\pi}\sum_{k\geq0}
\frac{\cos\bigl((2k+1)t\bigr)}{(2k+1)^2} ,
\]
साथ \emph{सामान्य} \omterm{def:b2:integration:improper}{अभिसरण} ($\sum (2k+1)^{-2} < \infty $) --- जैसा
\cref{thm:b2:fourier:parseval}~(1) इसके लिए भविष्यवाणी करता है \omterm{def:b2:metric:continuity}{निरंतर}
टुकड़े-टुकड़े-$ C^1$ समारोह। पर $ t = 0$:
\[
0 = \frac\pi2 - \frac4\pi\sum_{k\geq0}\frac{1}{(2k+1)^2}
\quad\Longrightarrow\quad
\sum_{k\geq0}\frac{1}{(2k+1)^2} = \frac{\pi^2}{8} .
\]
विभाजन $\sum \frac{1}{n^2}$ विषम और सम भागों में: $ S =
\frac{\pi^2}{8} + \frac S4$, इसलिए $ S = \frac{\pi^2}{6}$: बेसल फिर से.
\end{solution}

\begin{solution}{exo:b2:fourier:3}
समता: $ b_n = 0$; $ a_0 = \frac{1}{\pi}\int_{-\pi}^{\pi} t^2 =
\frac{2\pi^2}{3}$; भागों द्वारा दो एकीकरण देते हैं $ a_n =
\frac{4(-1)^n}{n^2}$ ($ n \geq 1$). इस तरह
\[
t^2 = \frac{\pi^2}{3} + 4\sum_{n\geq1}
\frac{(-1)^n}{n^2}\cos nt
\qquad (\abs t \leq \pi),
\]
\omterm{def:b2:funcseq:series}{सामान्यतः} अभिसारी। पर $ t = 0$: $0 = \frac{\pi^2}{3} +
4\sum\frac{(-1)^n}{n^2}$, यानी\ $\sum_{n\geq1}
\frac{(-1)^{n+1}}{n^2} = \frac{\pi^2}{12}$. पारसेवल:
\[
\frac{1}{2\pi}\int_{-\pi}^{\pi} t^4\,\dd t = \frac{\pi^4}{5}
= \frac{a_0^2}{4} + \frac12\sum_{n\geq1} a_n^2
= \frac{\pi^4}{9} + 8\sum_{n\geq1}\frac{1}{n^4} ,
\]
इसलिए $\sum \frac{1}{n^4} = \frac18\bigl(\frac{\pi^4}{5} -
\frac{\pi^4}{9}\bigr) = \frac{\pi^4}{90}$.
\end{solution}

\begin{solution}{exo:b2:fourier:4}
यदि इसके अलावा $ f $ टुकड़े-टुकड़े में है $ C^1$: पारसेवल (सिद्ध) देता है
$\norm f_2^2 = \sum\abs{c_n}^2 = 0$, और की सख्त सकारात्मकता
का अभिन्न अंग \omterm{def:b2:metric:continuity}{निरंतर} $\abs f^2$ बल $ f = 0$.

महज़ के लिए \omterm{def:b2:metric:continuity}{निरंतर} $ f $, सामान्य पार्सेवल स्वीकार करें: वही एक-पंक्ति
प्रमाण. (घनत्व मार्ग की रूपरेखा: Fej\'एर/वीयरस्ट्रैस-प्रकार
त्रिकोणमितीय सन्निकटन से पता चलता है कि त्रिकोणमितीय बहुपद हैं
$\norm\cdot_2$-के बीच घना \omterm{def:b2:metric:continuity}{निरंतर} आवधिक कार्य; तब से $ f
\perp $ उन सभी को, $\norm f_2^2 = \langle f, f - P\rangle \leq
\norm f_2\norm{f - P}_2$ सन्निकटन के लिए $ P $, मजबूर करना $\norm f_2 =
0$.)
\end{solution}

\begin{solution}{exo:b2:fourier:5}
समता: $ b_n = 0$;
\[
a_n = \frac{2}{\pi}\int_0^\pi \cos(\alpha t)\cos(nt)\,\dd t
= \frac{2}{\pi}\cdot
\frac{(-1)^n\,\alpha\sin(\pi\alpha)}{\alpha^2 - n^2}
\]
(उत्पाद-से-योग, फिर एकीकृत करें; $ a_0 =
\frac{2\sin(\pi\alpha)}{\pi\alpha}$). डिरिचलेट पर $ t = \pi $ (ए
निरंतरता आवधिक विस्तार का बिंदु, जिसका एकतरफ़ा मान
समता से सहमत):
\[
\cos(\pi\alpha)
= \frac{\sin(\pi\alpha)}{\pi\alpha}
+ \sum_{n\geq1} \frac{2\alpha\sin(\pi\alpha)}{\pi(\alpha^2 -
n^2)}\,(-1)^n\cos(n\pi)
= \frac{\sin(\pi\alpha)}{\pi}\Bigl(\frac{1}{\alpha} +
\sum_{n\geq1}\frac{2\alpha}{\alpha^2 - n^2}\Bigr),
\]
का उपयोग करते हुए $(-1)^n\cos n\pi = 1$. द्वारा विभाजित करना $\sin(\pi\alpha)/\pi $
कोटैंजेंट विस्तार देता है।
\end{solution}

\begin{solution}{exo:b2:fourier:6}
बार-बार दोहराना $ c_n(f') = \iu n\,c_n(f)$ ($ k $ समय, द्वारा एकीकरण
के पार भाग $ C^{k}$ मिलान सीमा मान वाले टुकड़े):
$ c_n(f^{(k)}) = (\iu n)^k c_n(f)$. के गुणांक
टुकड़े-टुकड़े में \omterm{def:b2:metric:continuity}{निरंतर} $ f^{(k)}$ बंधे हुए हैं (वास्तव में $\to 0$, बेसेल):
\[
\abs{c_n(f)} = \frac{\abs{c_n(f^{(k)})}}{\abs n^k}
= O\bigl(\abs n^{-k}\bigr) .
\]
\end{solution}

\begin{solution}{exo:b2:fourier:7}
पारसेवल के लिए $ f $ और के लिए $ f'$ (दोनों वैध: $ f $ है $ C^1$, $ f'$
टुकड़े-टुकड़े में \omterm{def:b2:metric:continuity}{निरंतर} --- वास्तव में \omterm{def:b2:metric:continuity}{निरंतर}):
\[
\frac{1}{2\pi}\int \abs f^2 = \sum_{n\neq0} \abs{c_n}^2
\quad (c_0 = 0 \text{ by the mean-zero hypothesis}),
\qquad
\frac{1}{2\pi}\int \abs{f'}^2 = \sum_{n\neq0} n^2\abs{c_n}^2 .
\]
अवधिवार, $ n^2\abs{c_n}^2 \geq \abs{c_n}^2$ के लिए $\abs n \geq 1$:
असमानता इस प्रकार है। समानता बल $(n^2 - 1)\abs{c_n}^2 = 0$
सभी के लिए $ n $, यानी\ $ c_n = 0$ के लिए $\abs n \geq 2$: $ f(t) =
c_1\eu^{\iu t} + c_{-1}\eu^{-\iu t} = a\cos t + b\sin t $ (वास्तविक
फॉर्म); इसके विपरीत ऐसे $ f $ समानता दो.
\end{solution}

\begin{solution}{exo:b2:fourier:8}
से \cref{exo:b2:fourier:1}, $ S_{2N-1}(x) =
\frac4\pi\sum_{k=0}^{N-1}\frac{\sin((2k+1)x)}{2k+1}$. पर $ x_N =
\frac{\pi}{2N}$, साथ $ u_k = (2k+1)x_N = \frac{(2k+1)\pi}{2N}$:
\[
S_{2N-1}(x_N) = \frac{4}{\pi}\sum_{k=0}^{N-1}\frac{\sin u_k}{2k+1}
= \frac{4}{\pi}\sum_{k=0}^{N-1}\frac{\sin u_k}{u_k}\cdot
\frac{u_k}{2k+1}
= \frac{2}{\pi}\sum_{k=0}^{N-1}\frac{\sin u_k}{u_k}\cdot
\frac{\pi}{N},
\]
तब से $\frac{u_k}{2k+1} = \frac{\pi}{2N}$. बिन्दु $ u_k $ हैं
के मध्यबिंदु $ N $ के उपअंतराल $\intcc{0}{\pi}$ लम्बाई का
$\frac{\pi}{N}$: योग एक मध्यबिंदु रीमैन योग है
\omterm{def:b2:metric:continuity}{निरंतर} $ u \mapsto \frac{\sin u}{u}$, इसलिए एकाग्र हो जाता है
\[
\frac{2}{\pi}\int_0^\pi \frac{\sin u}{u}\,\dd u
\approx \frac{2}{\pi}\times 1.8519 \approx 1.179 .
\]
छलांग के निकट आंशिक रकम मूल्य से अधिक हो जाती है $1$ द्वारा $\approx
18\%$ हमेशा के लिए आधी छलांग: गिब्स की घटना, परिमाणित।
\end{solution}

\begin{solution}{exo:b2:fourier:9}
से $\cos 3t = 4\cos^3t - 3\cos t $:
\[
\cos^3 t = \frac{3\cos t + \cos 3t}{4},
\qquad
\sin^2t\,\cos t = \cos t - \cos^3 t
= \frac{\cos t - \cos 3t}{4} .
\]
प्रत्येक एक त्रिकोणमितीय बहुपद है, इसलिए अपने स्वयं के बराबर है
फूरियर श्रृंखला (गुणांकों की विशिष्टता: दो विस्तार
सभी के साथ एक त्रिकोणमितीय बहुपद से भिन्न होगा
गुणांक शून्य). के लिए $\cos^3t $: $ a_1 = \frac34$, $ a_3 =
\frac14$, अन्य सभी $ a_n $ और सब $ b_n $ शून्य; $ c_{\pm1} =
\frac38$, $ c_{\pm3} = \frac18$. के लिए $\sin^2t\cos t $: $ a_1 =
\frac14$, $ a_3 = -\frac14$; $ c_{\pm1} = \frac18$, $ c_{\pm3} =
-\frac18$.
\end{solution}

\begin{solution}{exo:b2:fourier:10}
प्रत्यक्ष गणना:
\[
c_n = \frac{1}{2\pi}\int_{-\pi}^{\pi}\eu^{(a - \iu n)t}\dd t
= \frac{\eu^{(a-\iu n)\pi} - \eu^{-(a - \iu n)\pi}}
{2\pi(a - \iu n)}
= \frac{(-1)^n\sinh(a\pi)}{\pi(a - \iu n)} ,
\]
का उपयोग करते हुए $\eu^{\pm\iu n\pi} = (-1)^n $. पर $ t = \pi $ आवधिक
एक्सटेंशन से कूदता है $\eu^{a\pi}$ को $\eu^{-a\pi}$; डिरिचलेट
(\omterm{def:b2:quadratic:adjoint}{सममित} आंशिक रकम) देता है
\[
\cosh(a\pi) = \sum_{n\in\Z}(-1)^n c_n\,
= \frac{\sinh(a\pi)}{\pi}\Bigl(\frac1a +
\sum_{n\geq1}\Bigl(\frac{1}{a - \iu n} +
\frac{1}{a + \iu n}\Bigr)\Bigr)
= \frac{\sinh(a\pi)}{\pi}\Bigl(\frac1a +
\sum_{n\geq1}\frac{2a}{a^2 + n^2}\Bigr) ,
\]
युग्मित पदों के काल्पनिक भागों को रद्द करना। द्वारा विभाजित करें
$\sinh(a\pi)$:
\[
\coth(\pi a) = \frac{1}{\pi a} +
\sum_{n\geq1}\frac{2a}{\pi(a^2 + n^2)} ,
\]
का अतिपरवलयिक जुड़वां \cref{exo:b2:fourier:5}.
\end{solution}

\begin{solution}{exo:b2:fourier:11}
क्रमविनिमेयता: स्थानापन्न $ s = x - t $ और की आवधिकता का उपयोग करें
इंटीग्रैंड. के लिए $ c_n(f * g)$, इंटीग्रैंड $(x, t) \mapsto
f(x-t)g(t)\eu^{-\iu nx}$ है \omterm{def:b2:metric:continuity}{निरंतर}; दोनों ने पुनरावृति की
इंटीग्रल सहमत हैं (दोनों, की ऊपरी सीमा के कार्यों के रूप में
बाहरी चर, बाएं समापन बिंदु पर गायब हो जाता है और वही रहता है
व्युत्पन्न --- निरंतरता प्लस
\cref{thm:b2:integration:continuity} विभेदीकरण को उचित ठहराएँ
पुनरावृत्त अभिन्न)। इस तरह
\[
c_n(f*g) = \frac{1}{2\pi}\int_{-\pi}^{\pi} g(t)\,\eu^{-\iu nt}
\Bigl(\frac{1}{2\pi}\int_{-\pi}^{\pi}
f(x-t)\,\eu^{-\iu n(x-t)}\dd x\Bigr)\dd t
= c_n(f)\,c_n(g),
\]
आंतरिक अभिन्न अस्तित्व $ c_n(f)$ हरएक के लिए $ t $ (प्रतिस्थापन
और आवधिकता)। अंत में \cref{lem:b2:fourier:kernel} कहते हैं
$ S_N(f)(x) = \frac{1}{2\pi}\int f(x+u)D_N(u)\dd u $; द
प्रतिस्थापन $ u \mapsto -t $ और की समता $ D_N $ इसे चालू करो
में $(f * D_N)(x)$.
\end{solution}

\begin{solution}{exo:b2:fourier:12}
\begin{enumerate}
  \item साथ $ w = r\eu^{\iu t}$:
        \[
        P_r(t) = 1 + 2\,\Re\frac{w}{1 - w}
        = \Re\frac{1 + w}{1 - w}
        = \frac{1 - \abs w^2}{\abs{1 - w}^2}
        = \frac{1 - r^2}{1 - 2r\cos t + r^2} > 0 .
        \]
        अर्थ $1$: का अवधि-दर-अवधि एकीकरण \omterm{def:b2:funcseq:series}{सामान्यतः}
        \omterm{def:b2:integration:improper}{अभिसरण} श्रृंखला ही रखती है $ n = 0$.
  \item के लिए $\delta \leq \abs t \leq \pi $: $\cos t \leq
        \cos\delta $, इसलिए $ P_r(t) \leq \frac{1 - r^2}{1 -
        2r\cos\delta + r^2}$, जिसका हर प्रवृत्त होता है $2 -
        2\cos\delta > 0$ जबकि अंश की प्रवृत्ति होती है $0$:
        एकसमान \omterm{def:b2:integration:improper}{अभिसरण} को $0$ के किसी भी पड़ोस से दूर
        $0$.
  \item अवधि-दर-अवधि एकीकरण (सामान्य \omterm{def:b2:integration:improper}{अभिसरण} में $ t $)
        देता है $(f * P_r)(x) = \sum_n r^{\abs
        n}c_n(f)\eu^{\iu nx}$. अनुमानित-पहचान
        तर्क: मतलब के साथ $1$ और सकारात्मकता,
        \[
        \abs{(f*P_r)(x) - f(x)}
        \leq \frac{1}{2\pi}\int_{-\pi}^{\pi}
        \abs{f(x-t) - f(x)}\,P_r(t)\,\dd t ,
        \]
        पर विभाजित $\abs t = \delta $: अधिक से अधिक $\varepsilon $
        (हेन) प्लस $2\norm f_\infty\sup_{\delta\leq\abs
        t\leq\pi}P_r \to \varepsilon $: एकसमान \omterm{def:b2:integration:improper}{अभिसरण} जैसा
        $ r \to 1^-$. यह है \omterm{thm:b2:series:abel}{हाबिल सारांश} फूरियर का
        शृंखला --- शक्ति-श्रृंखला का फूरियर जुड़वां
        अध्याय का सीमा सिद्धांत.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b2:fourier:1}
\textbf{1.} औसत \cref{lem:b2:fourier:kernel} ऊपर $ n = 0,
\dots, N-1$ (अभिन्न की रैखिकता) देता है $\sigma_N(f)(x) =
\frac{1}{2\pi}\int f(x+u)F_N(u)\dd u $; प्रत्येक $ D_n $ मतलब है
$1$, इसलिए $ F_N $ मतलब है $1$.

\textbf{2.} साथ $\eu^{\iu u} - 1 = 2\iu\,\eu^{\iu
u/2}\sin\frac u2$:
\[
\sum_{n=0}^{N-1}\sin\Bigl(\Bigl(n + \frac12\Bigr)u\Bigr)
= \Im\Bigl[\eu^{\iu u/2}\,\frac{\eu^{\iu Nu} - 1}{\eu^{\iu u}
- 1}\Bigr]
= \Re\,\frac{1 - \eu^{\iu Nu}}{2\sin\frac u2}
= \frac{1 - \cos Nu}{2\sin\frac u2}
= \frac{\sin^2\frac{Nu}2}{\sin\frac u2} .
\]
द्वारा विभाजित करना $ N\sin\frac u2$:
\[
F_N(u) = \frac1N\sum_{n=0}^{N-1}
\frac{\sin\bigl((n+\frac12)u\bigr)}{\sin\frac u2}
= \frac{1}{N}\,
\frac{\sin^2\frac{Nu}{2}}{\sin^2\frac u2} \geq 0 .
\]

\textbf{3.} पर $\delta \leq \abs u \leq \pi $: $\sin^2\frac u2
\geq \sin^2\frac\delta2$ और $\sin^2\frac{Nu}2 \leq 1$: $ F_N
\leq \frac{1}{N\sin^2(\delta/2)} \to 0$, \omterm{def:b2:funcseq:def}{समान रूप से} वहाँ।

\textbf{4.} इकाई से मतलब है, $\sigma_Nf(x) - f(x) =
\frac{1}{2\pi}\int\bigl(f(x+u) - f(x)\bigr)F_N(u)\dd u $. दिया गया
$\varepsilon $, हेन की पैदावार $\delta $ साथ $\abs{f(x+u) - f(x)}
\leq \varepsilon $ के लिए $\abs u \leq \delta $, \omterm{def:b2:funcseq:def}{समान रूप से} में $ x $.
फिर, का उपयोग कर $ F_N \geq 0$ और इसकी इकाई का मतलब है,
\[
\abs{\sigma_Nf(x) - f(x)}
\leq \varepsilon +
2\norm f_\infty\cdot\frac{1}{2\pi}
\int_{\delta\leq\abs u\leq\pi}F_N
\leq \varepsilon + \frac{2\norm
f_\infty}{N\sin^2\frac\delta2}
\leq 2\varepsilon
\]
बड़े के लिए $ N $, \omterm{def:b2:funcseq:def}{समान रूप से} में $ x $: फ़ेज़\'एर का प्रमेय.

\textbf{5.} $ F_N $ मतलब के साथ भी है $1$: प्रत्येक आधा
$\intcc{0}{\pi}$, $\intcc{-\pi}{0}$ मतलब रखता है $\frac12$.
तब
\[
\sigma_Nf(x) - \frac{f(x^+)+f(x^-)}{2}
= \frac{1}{2\pi}\int_0^\pi\bigl(f(x+u) -
f(x^+)\bigr)F_N\,\dd u
+ \frac{1}{2\pi}\int_{-\pi}^0\bigl(f(x+u) -
f(x^-)\bigr)F_N\,\dd u ;
\]
प्रत्येक आधे पर, विभाजित करें $\abs u = \delta $ जहां एक तरफा
सीमा है $\varepsilon $-बंद करें, और प्रश्न 3 को दूर तक मारने दें
भाग: दोनों अभिन्न अंग होते हैं $0$. बाध्य: $ F_N \geq 0$
देता है $\abs{\sigma_Nf(x)} \leq
\frac{1}{2\pi}\int\abs{f(x+u)}F_N \leq \norm f_\infty $.

\textbf{6.} प्रत्येक $\sigma_N(f)$ एक त्रिकोणमितीय बहुपद है
(औसतन $ S_n(f)$, $ n < N $), और $\sigma_N(f) \to f $
\omterm{def:b2:funcseq:def}{समान रूप से}: त्रिकोणमितीय वीयरस्ट्रैस प्रमेय।

\textbf{7.} $ c_n(f) = 0$ सभी के लिए $ n $ हर बनाता है $ S_n(f) = 0$,
इसलिए हर $\sigma_N(f) = 0$; फेज द्वारा\'एर, $ f = \lim\sigma_Nf =
0$. इसे किसी अंतर पर लागू करना: \omterm{def:b2:metric:continuity}{निरंतर} आवधिक
कार्य उनके द्वारा निर्धारित होते हैं \omterm{def:b2:fourier:coefficients}{फूरियर गुणांक}.

\textbf{8.} $ S_Nf $ का ओर्थोगोनल प्रक्षेपण है $ f $ पर
$\mathcal T_N $ (\cref{prop:b2:fourier:bessel}), इसलिए यह न्यूनतम हो जाता है
$\norm{f - P}_2$ ऊपर $ P \in \mathcal T_N $; तब से $\sigma_Nf
\in \mathcal T_{N-1} \subseteq \mathcal T_N $:
\[
\norm{f - S_Nf}_2 \leq \norm{f - \sigma_Nf}_2
\leq \norm{f - \sigma_Nf}_\infty \to 0
\]
(मध्यम असमानता क्योंकि माध्य $\abs\cdot^2$ पर है
सबसे सुपर चुकता)। पाइथागोरस $\norm f_2^2 = \sum_{\abs
n\leq N}\abs{c_n}^2 + \norm{f - S_Nf}_2^2$ फिर पास हो जाता है
सीमा: प्रत्येक के लिए पारसेवल \omterm{def:b2:metric:continuity}{निरंतर} आवधिक $ f $.

\textbf{9.} होने देना $ t_1, \dots, t_p $ की छलांग हो $ f $ एक में
अवधि, $ M = \norm f_\infty $. छोटे के लिए $\eta $, परिभाषित करना $ g = f $
अंतराल के बाहर $\intoo{t_j - \eta}{t_j + \eta}$ और द्वारा
ऐसे प्रत्येक अंतराल में एफ़िन कॉर्ड: $ g $ है \omterm{def:b2:metric:continuity}{निरंतर},
आवधिक, $\norm g_\infty \leq M $, और
\[
\norm{f - g}_2^2 \leq \frac{1}{2\pi}\,p\cdot(2M)^2\cdot2\eta
\leq \varepsilon^2
\]
के लिए $\eta $ छोटा। बेसेल बनाता है $ S_N $ के लिए एक संकुचन
$\norm\cdot_2$, इसलिए
\[
\norm{f - S_Nf}_2
\leq \norm{f - g}_2 + \norm{g - S_Ng}_2 + \norm{S_N(g -
f)}_2
\leq 2\varepsilon + \norm{g - S_Ng}_2 ,
\]
और प्रश्न 8 देता है $\limsup_N\norm{f - S_Nf}_2 \leq
2\varepsilon $ हरएक के लिए $\varepsilon $: $\norm{f - S_Nf}_2 \to
0$, और पाइथागोरस प्रत्येक टुकड़े के लिए पार्सेवल उत्पन्न करता है
\omterm{def:b2:metric:continuity}{निरंतर} $ f $: ``भर्ती''
\cref{thm:b2:fourier:parseval} अब एक प्रमेय है.

\textbf{10.} चौकोर लहर है $\norm f_\infty = 1$, तो
प्रश्न 5 देता है $\abs{\sigma_N(f)} \leq 1$ हर जगह और के लिए
प्रत्येक $ N $ --- कोई ओवरशूट नहीं, कभी भी --- जबकि
\cref{exo:b2:fourier:8} दिखाता है $\sup_xS_{2N-1}(f)(x) \to
\approx 1.179$. एक जिम्मेदार संपत्ति: $ F_N \geq 0$, इसलिए
$\sigma_Nf(x)$ एक भारित है \emph{औसत} के मूल्यों का $ f $
और कभी नहीं छोड़ सकते $\intcc{\min f}{\max f}$; $ D_N $ लेता है
नकारात्मक मान, इसलिए $ S_N $ कर सकना।

\textbf{11.} $\abs{\sin N\theta} \leq N\abs{\sin\theta}$ द्वारा
प्रेरण ($\abs{\sin(N{+}1)\theta} \leq
\abs{\sin N\theta}\abs{\cos\theta} +
\abs{\cos N\theta}\abs{\sin\theta} \leq
(N+1)\abs{\sin\theta}$): साथ $\theta = \frac u2$,
\[
F_N(u) = \frac{\sin^2\frac{Nu}2}{N\sin^2\frac u2}
\leq \frac{N^2\sin^2\frac u2}{N\sin^2\frac u2} = N .
\]
की अवतलता $\sin $ पर $\intcc{0}{\frac\pi2}$ देता है
$\sin\frac u2 \geq \frac{u}{\pi}$ के लिए $0 \leq u \leq \pi $, इसलिए
$ F_N(u) \leq \frac{1}{N(u/\pi)^2} = \frac{\pi^2}{Nu^2}$.

\textbf{12.} समता और विभाजन से $\frac1N $:
\[
\frac{1}{2\pi}\int_{-\pi}^{\pi}\abs uF_N
= \frac1\pi\int_0^\pi uF_N
\leq \frac1\pi\Bigl(\int_0^{1/N}uN\,\dd u +
\int_{1/N}^{\pi}\frac{\pi^2}{Nu}\,\dd u\Bigr)
= \frac{1}{2\pi N} + \frac{\pi\ln(\pi N)}{N} .
\]
के लिए $ N \geq 2$: $\ln(\pi N) \leq \bigl(1 +
\frac{\ln\pi}{\ln2}\bigr)\ln N \leq 2.66\ln N $ और
$\frac{1}{2\pi N} \leq \frac{\ln N}{N}$, तो क्षण है
$\leq \frac{9\ln N}{N}$.

\textbf{13.} के लिए $ L $-लिप्सचिट्ज़ $ f $:
\[
\abs{\sigma_Nf(x) - f(x)}
\leq \frac{1}{2\pi}\int\abs{f(x+u) - f(x)}F_N(u)\dd u
\leq L\cdot\frac{1}{2\pi}\int\abs uF_N
\leq \frac{9L\ln N}{N},
\]
\omterm{def:b2:funcseq:def}{समान रूप से} में $ x $.

\textbf{14.} $ c_0(e_1) = 0$ देता है $ S_0(e_1) = 0$, जबकि
$ S_n(e_1) = e_1$ के लिए $ n \geq 1$: $\sigma_N(e_1) =
\frac{N-1}{N}e_1$, इसलिए $\norm{\sigma_Ne_1 - e_1}_\infty =
\frac1N $. यहां तक कि इस संपूर्ण, बैंड-सीमित सिग्नल के लिए भी दर है
$\frac1N $: फ़ेज़\'यह \omterm{def:b2:series:def}{बिल्कुल} बर्नस्टीन के संचालक की तरह संतृप्त होता है
पर संतृप्त होता है $\frac1n $ (वोरोनोव्स्काया, में
फ़ंक्शन-अनुक्रम अध्याय की सप्ताहांत समस्या)।

\textbf{15.} अगर $ f $ पर गायब हो जाता है $\intoo{x-\delta}{x+\delta}$,
तब $\sigma_Nf(x) = \frac{1}{2\pi}\int_{\delta \leq \abs u
\leq \pi}f(x+u)F_N(u)\dd u $, अधिकतम पूर्ण मूल्य का
$\frac{\norm f_\infty}{N\sin^2(\delta/2)} \to 0$: सेस\`एआरओ
मतलब पर $ x $ केवल देखें $ f $ पास में $ x $.

\textbf{16.} दिया गया $\intcc ab $ और $\varepsilon $, चुनना
\omterm{def:b2:metric:continuity}{निरंतर} $1$-आवधिक टुकड़े-टुकड़े-एफ़िन $\varphi^\pm $ साथ
$\varphi^- \leq \mathbf 1_{\intcc ab} \leq \varphi^+$ और
$\int_0^1(\varphi^+ - \varphi^-) \leq \varepsilon $ (ट्रेपेज़ॉइड्स
कुल लंबाई के अंतराल पर ढलान के साथ $\varepsilon $).
तब
\[
\limsup_N\frac{\#\{n \leq N : x_n \in \intcc ab\}}{N}
\leq \lim_N\frac1N\sum_{n\leq N}\varphi^+(x_n)
= \int_0^1\varphi^+ \leq b - a + \varepsilon ,
\]
और \omterm{def:b2:quadratic:adjoint}{सममित} रूप से $\liminf \geq b - a - \varepsilon $: द
अनुपात की ओर प्रवृत्त होता है $ b - a $.

\textbf{17.} के लिए $ f = \eu^{2\iu\pi k\cdot}$ साथ $ k \neq 0$ द
परिकल्पना सीमा देती है $0 = \int_0^1f $; के लिए $ k = 0$ दोनों
पक्ष हैं $1$; रैखिकता प्रत्येक त्रिकोणमिति को संभालती है
बहुपद के लिए \omterm{def:b2:metric:continuity}{निरंतर} $1$-आवधिक $ f $ और $\varepsilon >
0$, प्रश्न 6 (द्वारा परिवहन किया गया $ t = 2\pi x $) एक प्रदान करता है
त्रिकोणमितीय बहुपद $ P $ साथ $\norm{f - P}_\infty \leq
\varepsilon $:
\[
\Bigl|\frac1N\sum_{n\leq N}f(x_n) - \int_0^1f\Bigr|
\leq 2\varepsilon +
\Bigl|\frac1N\sum_{n\leq N}P(x_n) - \int_0^1P\Bigr|
\longrightarrow 2\varepsilon .
\]
प्रश्न 16 के साथ: $(x_n)$ \omterm{def:b2:funcseq:def}{समान रूप से} वितरित है.

\textbf{18.} के लिए $ k \neq 0$ और तर्कहीन $\alpha $, $ w =
\eu^{2\iu\pi k\alpha} \neq 1$:
\[
\Bigl|\sum_{n=1}^{N}w^n\Bigr|
= \Bigl|\frac{w(w^N - 1)}{w - 1}\Bigr|
\leq \frac{2}{\abs{1 - w}} ,
\]
एक बाध्य स्वतंत्र $ N $; द्वारा विभाजित करना $ N $ वेइल्स देता है
मानदंड, और प्रश्न 17 का निष्कर्ष: $(\{n\alpha\})$ है
\omterm{def:b2:funcseq:def}{समान रूप से} वितरित

\textbf{19.} प्रत्येक उपअंतराल को स्पर्शोन्मुख अनुपात प्राप्त होता है
इसकी लंबाई के बराबर, विशेष रूप से अनंत रूप से कई बिंदु:
$(\{n\alpha\})$ घना है. \omterm{pb:b2:fourier:1}{समान वितरण} मजबूत है: a
अपना लगभग सारा समय एक में बिताते हुए अनुक्रम सघन हो सकता है
कोने (घनत्व कहता है कि अनुक्रम कहां है \emph{जाता है},
समवितरण कहता है \emph{कितनी बार}).

\textbf{20.} $2^n $ अग्रणी अंक है $1$ आईएफएफ $10^m \leq 2^n <
2\cdot10^m $ कुछ के लिए $ m $, यानी\ आईएफएफ $\{n\log_{10}2\} \in
\intco{0}{\log_{10}2}$. अतार्किकता: $\log_{10}2 =
\frac pq $ देगा $2^q = 10^p = 2^p5^p $, के लिए असंभव $ p
\geq 1$ अद्वितीय गुणनखंडन द्वारा. वेइल का प्रमेय (प्रश्न 18
के साथ $\alpha = \log_{10}2$; आधा \omterm{def:b2:metric:topology}{खुला} अंतराल निचोड़ा हुआ है
पास की लंबाई के बंद लोगों के बीच) अनुपात देता है
$\log_{10}2 \approx 0.301$: के पहले अंक $2^n $ अनुसरण करें
बेनफोर्ड का नियम.

\textbf{21.} $ z $ है $ C^1$, इसलिए इसकी फूरियर श्रृंखला \omterm{def:b2:integration:improper}{अभिसरण} करती है
\omterm{def:b2:funcseq:series}{सामान्यतः} योग के साथ $ z $ (\cref{thm:b2:fourier:parseval}~(1)),
और $ c_n(z') = \iu n\,c_n $. तब से $\abs{z'} = \frac{L}{2\pi}$
स्थिर है,
\[
\int_0^{2\pi}\abs{z'}^2\dd t
= 2\pi\Bigl(\frac{L}{2\pi}\Bigr)^{\!2}
= \frac{L^2}{2\pi} ,
\]
और पार्सेवल ने आवेदन किया \omterm{def:b2:metric:continuity}{निरंतर} $ z'$ देता है
$\frac{1}{2\pi}\int_0^{2\pi}\abs{z'}^2 =
\sum_n\abs{\iu nc_n}^2$, यानी\ $\frac{L^2}{2\pi} = 2\pi\sum_n
n^2\abs{c_n}^2$.

\textbf{22.} ध्रुवीकृत पारसेवल $\frac{1}{2\pi}\int\conj
fg = \sum\conj{c_n(f)}c_n(g)$ पार्सेवल से इस प्रकार लागू होता है
$ f + g $ और $ f + \iu g $ (\omterm{def:b2:quadratic:def}{ध्रुवीकरण} पहचान), दोनों
\omterm{def:b2:metric:continuity}{निरंतर}. साथ $ f = z $, $ g = z'$:
\[
A = \frac12\,\Im\int_0^{2\pi}\conj z\,z'
= \pi\,\Im\sum_n\conj{c_n}(\iu n c_n)
= \pi\sum_n n\abs{c_n}^2 .
\]

\textbf{23.} संयुक्त प्रश्न 21-22:
\[
L^2 - 4\pi A
= 4\pi^2\sum_n n^2\abs{c_n}^2 -
4\pi^2\sum_n n\abs{c_n}^2
= 4\pi^2\sum_{n\in\Z}(n^2 - n)\abs{c_n}^2 \geq 0 ,
\]
तब से $ n^2 - n = n(n-1) \geq 0$ प्रत्येक पूर्णांक के लिए. समानता
बल $ c_n = 0$ सभी के लिए $ n \notin \{0, 1\}$: $ z(t) = c_0 +
c_1\eu^{\iu t}$, केंद्र का एक वृत्त $ c_0$ और त्रिज्या
$\abs{c_1} = \frac{L}{2\pi}$ (\omterm{def:b2:metric:continuity}{निरंतर} गति). हर्विट्ज़ का प्रमाण
आइसोपेरिमेट्रिक असमानता का: $ A \leq \frac{L^2}{4\pi}$,
केवल घेरा.

\textbf{24.} त्रिज्या का वृत्त $ R $: $ L = 2\pi R $, $ A = \pi
R^2$: $ L^2 = 4\pi^2R^2 = 4\pi A $: समानता. भुजा का वर्ग
$ a $: $ L^2 = 16a^2 > 4\pi a^2 = 4\pi A $ (जैसा $16 > 4\pi \approx
12.57$). नकारात्मक के लिए $ n $, $ n^2 - n = n(n - 1)$ एक उत्पाद है
दो ऋणात्मक पूर्णांकों का: धनात्मक --- इतना पिछड़ा-घुमावदार
मोड लागत क्षेत्र दो बार। प्रश्न 21 में प्रविष्ट स्थिर गति,
परिवर्तित करना $\int\abs{z'}^2$ में $\frac{L^2}{2\pi}$; एक के लिए
गैर स्थिर गति, कॉची-श्वार्ज़ देता है $\int\abs{z'}^2 \geq
\frac{(\int\abs{z'})^2}{2\pi} = \frac{L^2}{2\pi}$, तो
असमानता कायम है, वृत्त ही एकमात्र समानता है
मामला.

\textbf{25.} (i) हर चीज़ बहती है $ F_N \geq 0$ (यूनिट के साथ
माध्य और एकाग्रता); $ D_N $ इकाई माध्य और एकाग्रता है
दोलन का लेकिन सकारात्मकता का नहीं, और गिब्स कीमत है।
(ii) सेस\`एआरओ संक्षेपण फूरियर श्रृंखला का तात्पर्य इसका है
हाबिल संक्षेपण उसी राशि के साथ (फ्रोबेनियस, में साबित हुआ
पावर-सीरीज़ चैप्टर की सप्ताहांत समस्या) --- पॉइसन-कर्नेल
का मार्ग \cref{exo:b2:fourier:12} \omterm{def:b2:series:def}{बिल्कुल} हाबिल की पद्धति है.
(iii) वेइल के प्रमेय को केवल एकसमान सन्निकटन की आवश्यकता थी
(प्रश्न 6); आइसोपरिमेट्रिक असमानता के लिए पार्सेवल की आवश्यकता थी
स्वयं (प्रश्न 8,21--22). (iv) वर्ष 3 खंड सिद्ध करता है
पूर्णता: घातांक एक हिल्बर्ट आधार बनाते हैं $ L^2$,
पार्सेवल हिल्बर्ट रिक्त स्थान और फ़ेज़ की एक आइसोमेट्री बन जाता है\'एर का
प्रमेय यह कथन बन जाता है कि यह आइसोमेट्री गणना योग्य है
सकारात्मक औसत से.
\end{solution}
