\section*{अध्याय \ref{ch:b2:curves} --- घटता}

\begin{solution}{exo:b2:curves:1}
$\gamma'(t) = (1 - \cos t,\ \sin t)$, इसलिए
\[
\norm{\gamma'(t)}^2 = (1 - \cos t)^2 + \sin^2 t
= 2 - 2\cos t = 4\sin^2\tfrac t2 ,
\]
और $\norm{\gamma'(t)} = 2\sin\frac t2$ (पर गैरनकारात्मक $[0,
2\pi]$). इस तरह
\[
L = \int_0^{2\pi} 2\sin\tfrac t2\,\dd t
= \Bigl[-4\cos\tfrac t2\Bigr]_0^{2\pi} = 8 :
\]
साइक्लोइड का एक आर्क है \omterm{def:b2:curves:length}{लंबाई} $8$ (त्रिज्या के एक पहिये के लिए
$1$) --- व्रेन का एक प्रसिद्ध परिणाम, नं. के साथ $\pi $ \omterm{def:b2:diffcalc:differential}{अंतर}्दृष्टि।
\end{solution}

\begin{solution}{exo:b2:curves:2}
साथ $ x = a\cos t $, $ y = b\sin t $: $ x' = -a\sin t $, $ y' = b\cos
t $, $ x'' = -a\cos t $, $ y'' = -b\sin t $, तो द्वारा
\cref{prop:b2:curves:kappaformula}
\[
\kappa(t)
= \frac{x'y'' - y'x''}{(x'^2 + y'^2)^{3/2}}
= \frac{ab\sin^2 t + ab\cos^2 t}
       {(a^2\sin^2 t + b^2\cos^2 t)^{3/2}}
= \frac{ab}{(a^2\sin^2 t + b^2\cos^2 t)^{3/2}} .
\]
जब हर न्यूनतम होता है $\sin t = 0$ (कीमत $ b^3$, अंक
$(\pm a, 0)$) और अधिकतम कब $\cos t = 0$ (कीमत $ a^3$, अंक
$(0, \pm b)$), तब से $ a > b $. इस तरह $\kappa $ पर अधिकतम है
प्रमुख अक्ष के सिरे, $\kappa_{\max} = a/b^2$, और न्यूनतम पर
लघु अक्ष के सिरे, $\kappa_{\min} = b/a^2$: दीर्घवृत्त
अपनी लंबी धुरी के सिरों पर सबसे तेजी से झुकता है।
\end{solution}

\begin{solution}{exo:b2:curves:3}
ग्राफ़ के लिए $\gamma(x) = (x, \cosh x)$: $\norm{\gamma'(x)} =
\sqrt{1 + \sinh^2 x} = \cosh x $, इतना वक्राकार \omterm{def:b2:curves:length}{लंबाई} से $0$ को $ x $
है $\int_0^x \cosh u\,\dd u = \sinh x $. \omterm{thm:b2:curves:frenet2d}{वक्रता} एक ग्राफ का
(\cref{prop:b2:curves:kappaformula}):
\[
\kappa(x) = \frac{f''(x)}{(1 + f'(x)^2)^{3/2}}
= \frac{\cosh x}{\cosh^3 x} = \frac{1}{\cosh^2 x} ,
\]
इसलिए $ R(x) = \cosh^2 x $, जैसा कि घोषणा की गई थी। स्पष्ट संयोग पर ध्यान दें
$ R(x) = 1 + s(x)^2$ साथ $ s = \sinh x $  वक्राकार \omterm{def:b2:curves:length}{लंबाई}: द
कैटेनरी का \omterm{def:b2:curves:curvature}{वक्रता त्रिज्या} के वर्ग के साथ बढ़ता है वक्राकार \omterm{def:b2:curves:length}{लंबाई} शिखर से.
\end{solution}

\begin{solution}{exo:b2:curves:4}
$\gamma'(t) = e^t(\cos t - \sin t,\ \sin t + \cos t)$, इसलिए
\[
\langle \gamma(t), \gamma'(t)\rangle = e^{2t}
\bigl(\cos t(\cos t - \sin t) + \sin t(\sin t + \cos t)\bigr)
= e^{2t},
\]
जबकि $\norm{\gamma(t)} = e^t $ और $\norm{\gamma'(t)} =
e^t\sqrt 2$. इस तरह
\[
\cos\angle\bigl(\gamma, \gamma'\bigr)
= \frac{e^{2t}}{e^t \cdot e^t\sqrt2} = \frac{1}{\sqrt2} :
\]
\omterm{def:b2:curves:arc}{स्पर्श रेखा} सदैव कोण बनाती है $\pi/4$ त्रिज्या के साथ---
\emph{समकोणेवाला} लघुगणक सर्पिल की संपत्ति. वक्राकार \omterm{def:b2:curves:length}{लंबाई}
पर $(-\infty, 0]$:
\[
\int_{-\infty}^0 \norm{\gamma'(t)}\,\dd t
= \sqrt2\int_{-\infty}^0 e^t\,\dd t = \sqrt 2 ,
\]
परिमित यद्यपि सर्पिल हवाएँ अनन्त रूप से अनेक बार चारों ओर घूमती हैं
उत्पत्ति. \omterm{thm:b2:curves:frenet2d}{वक्रता}: साथ $ x'y'' - y'x''$ से गणना की गई $\gamma'' =
e^t(-2\sin t,\ 2\cos t)$,
\[
x'y'' - y'x'' = e^{2t}\bigl(2\cos t(\cos t - \sin t)
+ 2\sin t(\sin t + \cos t)\bigr) = 2e^{2t},
\]
इसलिए $\kappa(t) = \dfrac{2e^{2t}}{(e^t\sqrt2)^3} =
\dfrac{1}{e^t\sqrt2}$: द \omterm{thm:b2:curves:frenet2d}{वक्रता} है $1/(\sqrt2\,
\norm{\gamma})$, जैसे-जैसे सर्पिल बढ़ता है, क्षय होता जाता है।
\end{solution}

\begin{solution}{exo:b2:curves:5}
\emph{पहला चाप:} $\gamma(t) = (t^2,\ t^4 + t^5)$. डेरिवेटिव पर
$0$: $\gamma'' = (2, 0) \neq 0$, इसलिए $ p = 2$. तब $\gamma^{(3)}(0)
= (0, 0)$, $\gamma^{(4)}(0) = (0, 24)$, के साथ संरेख नहीं $(2,
0)$: $ q = 4$. दोनों सम: \emph{दूसरे प्रकार का पुच्छल} --- दोनों
शाखाएँ दिशा में निकलती हैं $+u = (1,0)$ और वैसे ही रहो
स्पर्शरेखा का पक्ष. (वास्तव में $ y = x^2 \pm x^{5/2}$ दो पर
शाखाएँ: छोटी के लिए समान चिह्न $ x $.)

\emph{दूसरा चाप:} $\gamma(t) = (t^3, t^4)$. $\gamma'(0) =
\gamma''(0) = 0$, $\gamma^{(3)}(0) = (6, 0)$: $ p = 3$, विषम। अगला
$\gamma^{(4)}(0) = (0, 24)$: $ q = 4$, यहां तक की। विषम--सम:
\emph{साधारण बिंदु} --- लुप्त वेग के बावजूद,
प्रक्षेपवक्र $ y = x^{4/3}$ रहकर, मूल को सहजता से पार कर जाता है
इसके स्पर्शरेखा के ऊपर $ y = 0$.
\end{solution}

\begin{solution}{exo:b2:curves:6}
\omterm{def:b2:diffcalc:differential}{अंतर} $ c(s) = \gamma(s) + \dfrac{1}{\kappa(s)}N(s)$ का उपयोग कर रहे हैं
समतल फ़्रेनेट सूत्र
(\cref{thm:b2:curves:frenet2d}):
\[
c'(s) = T(s) - \frac{\kappa'(s)}{\kappa(s)^2}N(s)
+ \frac{1}{\kappa(s)}\,\bigl(-\kappa(s)T(s)\bigr)
= -\frac{\kappa'(s)}{\kappa(s)^2}\,N(s) ,
\]
स्पर्शरेखा पद \omterm{def:b2:series:def}{बिल्कुल} रद्द हो रहे हैं। तो का वेग
evolute द्वारा किया जाता है $ N(s)$, जो की सामान्य रेखा को निर्देशित करता है
$\gamma $ पर $\gamma(s)$ --- और मुद्दा $ c(s)$ उसी पर झूठ बोलता है
सामान्य रेखा: उत्क्रांति है \emph{\omterm{pb:b2:curves:1}{लिफ़ाफ़ा} सामान्य का}.
(कहाँ $\kappa' = 0$ विकसित में एक विलक्षण बिंदु है; यह है
एक दीर्घवृत्त के विकास के क्यूप्स का निर्माण क्या होता है।)
\end{solution}

\begin{solution}{exo:b2:curves:7}
होने देना $ G(s) = F(s)F(s)^{\mathsf T}$. फिर, का उपयोग कर $ F' = FA $ और
$(F^{\mathsf T})' = (F')^{\mathsf T} = A^{\mathsf T}F^{\mathsf T}$,
\[
G' = F'F^{\mathsf T} + F(F^{\mathsf T})'
= FAF^{\mathsf T} + FA^{\mathsf T}F^{\mathsf T}
= F(A + A^{\mathsf T})F^{\mathsf T} = 0
\]
प्रति\omterm{def:b2:quadratic:adjoint}{सममित}ि द्वारा. इसलिए $ G $ \omterm{def:b2:diffcalc:differential}{अंतर}ाल पर स्थिर है, के बराबर
$ G(s_0) = F(s_0)F(s_0)^{\mathsf T} = I $: $ F(s)$ के लिए ओर्थोगोनल है
प्रत्येक $ s $. (वैकल्पिक रूप से, गणना के बिना $ G'$ शून्य से: दोनों
$ G $ और स्थिरांक $ I $ रैखिक प्रणाली को हल करें $ Y' = YA +
A^{\mathsf T}Y $ समान प्रारंभिक मूल्य के साथ, और कॉची--लिप्सचिट्ज़
रैखिक प्रणालियों के लिए विशिष्टता, \cref{ch:b2:diffeq}, बल $ G
\equiv I $.)

\emph{प्रासंगिकता:} फ़्रेनेट प्रणाली $(T, N, B)' = (T, N, B)\,A(s)$
एंटीसिमेट्रिक गुणांक मैट्रिक्स है
\[
A = \begin{pmatrix} 0 & -\kappa & 0\\ \kappa & 0 & -\tau\\
0 & \tau & 0\end{pmatrix}
\]
(स्तंभ व्यक्त कर रहे हैं $ T', N', B'$). उपरोक्त गणना से पता चलता है
वह एक समाधान फ़्रेम जो ऑर्थोनॉर्मल प्रारंभ होता है \emph{रहता है}
ऑर्थोनॉर्मल --- मौलिक प्रमेय में मुख्य चरण
से एक वक्र का पुनर्निर्माण $(\kappa, \tau)$.
\end{solution}

\begin{solution}{exo:b2:curves:8}
द्वारा \cref{thm:b2:curves:fundamental} (विशिष्टता भाग), वहाँ एक है
$\mathcal{C}^1$ कोण समारोह $\varphi $ साथ $ T(s) =
(\cos\varphi(s), \sin\varphi(s))$ और $\varphi' = \kappa $. इस तरह
\[
\int_0^L \kappa(s)\,\dd s = \varphi(L) - \varphi(0) .
\]
चूँकि चाप आवर्त का बंद है $ L $, $ T(L) = T(0)$:
$(\cos\varphi(L), \sin\varphi(L)) = (\cos\varphi(0),
\sin\varphi(0))$, इसलिए $\varphi(L) - \varphi(0) \in 2\pi\Z $. संपूर्ण \omterm{thm:b2:curves:frenet2d}{वक्रता} इसलिए एक बंद वक्र हमेशा एक पूर्णांक होता है
का एकाधिक $2\pi $ --- पूर्णांक है \emph{घुमावदार संख्या}
स्पर्शरेखा की (पूर्ण घुमावों की संख्या $ T $ बनाता है)। के लिए
त्रिज्या का वृत्त $ R $: $\kappa = 1/R $ और $ L = 2\pi R $, कुल
\omterm{thm:b2:curves:frenet2d}{वक्रता} $2\pi $, घुमावदार संख्या $1$; घूमने का प्रमेय
स्पर्शरेखा बताती है कि यह मान प्रत्येक साधारण बंद वक्र के लिए है।
\end{solution}

\begin{solution}{exo:b2:curves:9}
तब से $\tau \equiv 0$, वक्र एक समतल में स्थित है
(\cref{prop:b2:curves:torsion}); उस विमान में काम करो. पर विचार करें
उम्मीदवार केंद्र
\[
c(s) = \gamma(s) + \frac{1}{\kappa}N(s)
\qquad (\kappa \text{ constant}).
\]
फ़्रेनेट फ़ार्मुलों के साथ \omterm{def:b2:diffcalc:differential}{अंतर} करना ($ N' = -\kappa T + \tau B
= -\kappa T $ यहाँ):
\[
c'(s) = T + \frac1\kappa(-\kappa T) = 0 ,
\]
इसलिए $ c $ एक स्थिर बिंदु है $\Omega $. तब $\norm{\gamma(s) -
\Omega} = \norm{-\frac1\kappa N(s)} = \frac1\kappa $ सभी के लिए $ s $:
वक्र केंद्र के वृत्त पर स्थित है $\Omega $ और त्रिज्या
$1/\kappa $ (इसके तल में), और इसका एक अचर चाप होने के नाते, यह
उस वृत्त का एक चाप है.
\end{solution}

\begin{solution}{exo:b2:curves:10}
ग्राफ़ के लिए $\gamma(x) = (x, x^2/2)$, $\norm{\gamma'(x)} =
\sqrt{1 + x^2}$, इसलिए $ L = \int_0^a\sqrt{1+x^2}\,\dd x $.
स्थानापन्न $ x = \sinh u $ ($\dd x = \cosh u\,\dd u $, $ u $ से
$0$ को $ u_a = \ln(a + \sqrt{1+a^2})$):
\[
L = \int_0^{u_a}\cosh^2 u\,\dd u
= \frac12\bigl[u + \sinh u\cosh u\bigr]_0^{u_a}
= \frac12\Bigl(\ln\bigl(a + \sqrt{1+a^2}\bigr) +
a\sqrt{1+a^2}\Bigr),
\]
का उपयोग करते हुए $\cosh^2 u = \frac{1 + \cosh 2u}2$ और $\sinh u_a = a $,
$\cosh u_a = \sqrt{1 + a^2}$.
\end{solution}

\begin{solution}{exo:b2:curves:11}
द्वारा पैरामीट्रिज वक्राकार \omterm{def:b2:curves:length}{लंबाई}
(\cref{thm:b2:curves:arclength}) और सेट करें $\lambda(s) =
\langle P - \gamma(s), T(s)\rangle $, ए $\mathcal C^1$
समारोह; तब से $ P $ पर स्थित है \omterm{def:b2:curves:arc}{स्पर्श रेखा} पर $\gamma(s)$,
सदिश $ P - \gamma(s)$ के साथ संरेख है $ T(s)$, इसलिए $ P =
\gamma(s) + \lambda(s)T(s)$. विभेद करना,
\[
0 = T(s) + \lambda'(s)T(s) + \lambda(s)T'(s)
= \bigl(1 + \lambda'(s)\bigr)T(s) + \lambda(s)T'(s),
\]
और $ T'(s) \perp T(s)$ (\omterm{def:b2:diffcalc:differential}{अंतर} करें $\norm T^2 = 1$), तो
दोनों घटक गायब हो जाते हैं: $\lambda' = -1$ और $\lambda T' = 0$.
तब $\lambda(s) = c - s $ अधिक से अधिक एक बार गायब हो जाता है, इसलिए $ T' = 0$
घने सेट पर, इसलिए हर जगह से निरंतरता: $ T $ एक है
\omterm{def:b2:metric:continuity}{निरंतर} इकाई वेक्टर और $\gamma(s) = \gamma(s_0) + (s -
s_0)T $: एक सीधी रेखा (के माध्यम से) $ P $, जैसा कि यह होना चाहिए)।
\end{solution}

\begin{solution}{exo:b2:curves:12}
(ए) फ्रेनेट मैट्रिक्स
\[
A(s) = \begin{pmatrix} 0 & -\kappa & 0\\ \kappa & 0 & -\tau\\
0 & \tau & 0\end{pmatrix}
\]
है \omterm{def:b2:metric:continuity}{निरंतर} प्रविष्टियाँ, इसलिए रैखिक प्रणाली $ F' = FA(s)$,
$ F(s_0) = F_0$ (एक सीधा ऑर्थोनॉर्मल मैट्रिक्स) में एक अद्वितीय है
सभी पर समाधान $ J $ (\cref{thm:b2:diffeq:linear}). द्वारा
\cref{exo:b2:curves:7}, $ F(s)$ प्रत्येक के लिए ओर्थोगोनल है $ s $;
$\det F $ है \omterm{def:b2:metric:continuity}{निरंतर} मूल्यों के साथ $\{\pm1\}$ और बराबर है
$1$ पर $ s_0$, इसलिए $ F(s)$ सभी के लिए प्रत्यक्ष है $ s $.

(बी) पंक्तियों को पढ़ें $ T, N, B $ का $ F $ (ताकि $ T' = \kappa
N $, $ N' = -\kappa T + \tau B $, $ B' = -\tau N $) और सेट करें
$\gamma(s) = \gamma_0 + \int_{s_0}^s T(u)\,\dd u $. तब
$\gamma' = T $ एक इकाई वेक्टर है: इकाई गति; $ T' = \kappa N $
साथ $\kappa > 0$ और $ N $ इकाई ओर्थोगोनल से $ T $, इसलिए $\gamma $
के साथ द्विनियमित है \omterm{thm:b2:curves:frenet2d}{वक्रता} $\norm{T'} = \kappa $ और
प्रमुख सामान्य $ N $; द्विअसामान्य है $ T \wedge N = B $
(प्रत्यक्ष ऑर्थोनॉर्मल फ्रेम), और $ B' = -\tau N $ की पहचान करता है
\omterm{thm:b2:curves:frenet3d}{टोशन} जैसा $\tau $.

(सी) चलो $\gamma_1, \gamma_2$ इकाई-गति द्विनियमित वक्र बनें
उसी के साथ $(\kappa, \tau)$. एक अनोखा प्रत्यक्ष है
आइसोमेट्री $\Phi = \rho + w $ ($\rho \in SO(3)$) भेजना
$\gamma_1(s_0)$ को $\gamma_2(s_0)$ और फ्रेनेट फ़्रेम
$\gamma_1$ पर $ s_0$ उस के लिए $\gamma_2$ पर $ s_0$. वक्र
$\Phi\circ\gamma_1$ समान अपरिवर्तनीयों के साथ इकाई-गति है
(इसका फ्रेम है $\rho $ उस पर लागू किया गया $\gamma_1$, और
$\rho $ प्रत्यक्ष होने के कारण क्रॉस उत्पादों को संरक्षित करता है)। अब
के फ्रेम $\Phi\circ\gamma_1$ और $\gamma_2$ दोनों सुलझा लेते हैं $ F' =
FA(s)$ समान प्रारंभिक मान के साथ, इसलिए वे मेल खाते हैं
विशिष्टता; विशेष रूप से स्पर्शरेखाएं सहमत होती हैं, और एकीकृत होती हैं
सामान्य बिंदु से $ s_0$: $\Phi\circ\gamma_1 = \gamma_2$.
\end{solution}

\begin{solution}{pb:b2:curves:1}
\textbf{1.} प्रणाली रैखिक है $(x, y)$ साथ सिद्ध
$\Delta(t) = a b' - a'b \neq 0$: क्रैमर का नियम देता है
अनोखा समाधान
\[
x = \frac{c b' - c' b}{\Delta}, \qquad
y = \frac{a c' - a' c}{\Delta}.
\]

\textbf{2.} तब से $ a(t)x(t) + b(t)y(t) = c(t)$ \omterm{def:b2:funcseq:def}{समान रूप से},
विभेदीकरण देता है $ a'x + b'y + ax' + by' = c'$; द
दूसरा चारित्रिक समीकरण मारता है $ a'x + b'y - c'$, इसलिए
$ a(t)x'(t) + b(t)y'(t) = 0$: $ E'(t)$ के लिए ओर्थोगोनल है $(a,
b)$, इसलिए समानांतर $(-b, a)$, इसकी दिशा $ D_t $. जैसा
$ E(t) \in D_t $ (पहला समीकरण) और $ E'(t) \neq 0$, रेखा
$ D_t $ बिलकुल यही है \omterm{def:b2:curves:arc}{स्पर्श रेखा} वक्र का $ E $ पर $ E(t)$.

\textbf{3.} यहाँ $(a, b, c) = (t, -1, t^2/2)$, इसलिए $\Delta =
t\cdot0 - 1\cdot(-1) = 1$ और
\[
x = \frac{c b' - c' b}{\Delta} = \tfrac{t^2}2\cdot 0 +
t = t, \qquad
y = \frac{a c' - a' c}{\Delta} = t\cdot t - \tfrac{t^2}2 =
\tfrac{t^2}2 :
\]
 \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} की स्पर्शरेखा रेखाएँ परवलय का है
परवलय, जैसा कि होना चाहिए।

\textbf{4.} सामान्य रेखा है $\langle M, T(s)\rangle =
\langle\gamma(s), T(s)\rangle $: गुणांक $ a = T_1$, $ b =
T_2$, $ c = \langle\gamma, T\rangle $. के साथ भेद करना
फ़्रेनेट ($ T' = \kappa N $): $ a' = \kappa N_1$, $ b' = \kappa
N_2$, और $ c' = \langle T, T\rangle + \langle\gamma, \kappa
N\rangle = 1 + \kappa\langle\gamma, N\rangle $. दूसरा
विशेषता समीकरण $\kappa\langle M, N\rangle = 1 +
\kappa\langle\gamma, N\rangle $ पढ़ता $\kappa\langle M -
\gamma, N\rangle = 1$. पहला कहता है $ M - \gamma \perp T $,
इसलिए $ M - \gamma = \mu N $ साथ $\mu = 1/\kappa $: द
विशेषता बिंदु है $\gamma + \frac1\kappa N $, द केंद्र
\omterm{thm:b2:curves:frenet2d}{वक्रता} का, और यह \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} सामान्यों में से एक उत्क्रांति है
का \cref{exo:b2:curves:6}. अंत में $\Delta = T_1\kappa N_2 -
\kappa N_1 T_2 = \kappa\det(T, N) = \kappa \neq 0$.

\textbf{5.} पेंसिल: प्रणाली $ x\cos\theta + y\sin\theta =
0$, $-x\sin\theta + y\cos\theta = 0$ है सिद्ध $1$ और
समाधान $(0,0)$ हरएक के लिए $\theta $: $ E \equiv (0,0)$, $ E'
\equiv 0$, और कोई वक्र नहीं है --- केवल उभयनिष्ठ बिंदु है
सभी पंक्तियों में से. समानांतर परिवार: $ a' = b' = 0$ देता है
$\Delta \equiv 0$ और दूसरा समीकरण $0 = c'(t)$, जो
जैसे ही परिवार वास्तव में आगे बढ़ता है विफल हो जाता है: कोई विशेषता नहीं
बिंदु, और वास्तव में समानांतर रेखाओं का एक परिवार संख्या को स्पर्श करता है
अपने सभी सदस्यों के अनुदिश वक्र।

\textbf{6.} के माध्यम से लाइन $(\cos t, 0)$ और $(0, \sin t)$
है $\frac x{\cos t} + \frac y{\sin t} = 1$, यानी\ $ x\sin t +
y\cos t = \sin t\cos t $. साथ $(a, b, c) = (\sin t, \cos t,
\sin t\cos t)$: $ a' = \cos t $, $ b' = -\sin t $, $ c' = \cos
2t $, $\Delta = -\sin^2 t - \cos^2 t = -1$. क्रैमर:
\begin{align*}
x &= \frac{cb' - c'b}{-1} = \sin^2 t\cos t + \cos 2t\cos t
= \cos t\,(\sin^2 t + \cos^2 t - \sin^2 t) = \cos^3 t,\\
y &= \frac{ac' - a'c}{-1} = \sin t\cos^2 t - \sin t\cos 2t
= \sin t\,(\cos^2 t - \cos^2 t + \sin^2 t) = \sin^3 t .
\end{align*}
और $(\cos^3t)^{2/3} + (\sin^3t)^{2/3} = 1$: द \omterm{pb:b2:curves:1}{astroid}.

\textbf{7.} $ E'(t) = 3(-\cos^2 t\sin t,\ \sin^2 t\cos t)$
पर गायब हो जाता है $ t = 0$. वहाँ, $ E''(0) = (-3, 0) \neq 0$ देता है
$ p = 2$; $ x $-का घटक $ E $ में भी है $ t $, इसलिए
$ E'''(0) = (0, 6)$, संरेख नहीं: $ q = 3$. सम--विषम: शिखर
पहली तरह का $(1, 0)$
(\cref{prop:b2:curves:local}), स्पर्शरेखा के साथ
$ x $-अक्ष. समरूपता $ x \mapsto -x $, $ y \mapsto -y $, $(x,
y) \mapsto (y, x)$ की \omterm{pb:b2:curves:1}{astroid} पुच्छल को परिवहन करें $(-1,
0)$ और $(0, \pm1)$.

\textbf{8.} $ E'(t) = 3\sin t\cos t\,(-\cos t, \sin t)$, इसलिए
$\norm{E'(t)} = 3\abs{\sin t\cos t} = \tfrac32\abs{\sin 2t}$.
एक चतुर्थांश: $\int_0^{\pi/2}\tfrac32\sin 2t\,\dd t =
\tfrac32$, और समरूपता द्वारा कुल \omterm{def:b2:curves:length}{लंबाई} है $4 \cdot
\tfrac32 = 6$.

\textbf{9.} $ E(t) - P_t = (\cos^3 t - \cos t,\ \sin^3 t) =
\sin^2 t\,(-\cos t,\ \sin t) = \sin^2 t\,(Q_t - P_t)$. तो
संपर्क बिंदु है \omterm{def:b2:affine:barycenter}{केन्द्रक} का $(P_t, \cos^2 t)$ और
$(Q_t, \sin^2 t)$: जैसा $ t $ से चलती है $0$ को $\pi/2$ यह स्लाइड करता है
फर्श के सिरे से सीढ़ी के दीवार के सिरे तक।

\textbf{10.} प्रथम चतुर्थांश में क्षेत्र के \omterm{def:b2:diffcalc:differential}{अंतर}्गत
\omterm{pb:b2:curves:1}{astroid} क्षेत्रफल है $\int_0^1 y\,\dd x $ साथ $ x = \cos^3 t $
से घट रहा है $1$ को $0$ जैसा $ t $ से चला जाता है $0$ को $\pi/2$:
\[
\int_0^1 y\,\dd x
= \int_{\pi/2}^{0}\sin^3 t\,(-3\cos^2 t\sin t)\,\dd t
= 3\int_0^{\pi/2}\sin^4 t\cos^2 t\,\dd t .
\]
रैखिक बनाना: $\sin^4 t\cos^2 t = (\sin t\cos t)^2\sin^2 t =
\tfrac18\bigl(\sin^2 2t - \sin^2 2t\cos 2t\bigr)$, और
$\int_0^{\pi/2}\sin^2 2t\,\dd t = \tfrac\pi4$ जबकि
$\int_0^{\pi/2}\sin^2 2t\cos 2t\,\dd t =
\bigl[\tfrac{\sin^3 2t}6\bigr]_0^{\pi/2} = 0$. तो
अभिन्न है $\tfrac\pi{32}$, चतुर्भुज क्षेत्र
$\tfrac{3\pi}{32}$, और संलग्न क्षेत्र $4 \cdot
\tfrac{3\pi}{32} = \tfrac{3\pi}8$.

\textbf{11.} पर स्पर्शरेखा $\gamma(t) = (t, t^2/2)$ है
द्वारा निर्देशित $(1, t)$, तो सामान्य रेखा है $\{(x, y) : (x -
t) + t\,(y - t^2/2) = 0\}$, यानी\ $ x + t\,y = t +
\tfrac{t^3}2$.

\textbf{12.} $(a, b, c) = (1, t, t + t^3/2)$: $ a' = 0$, $ b' =
1$, $ c' = 1 + \tfrac32 t^2$, $\Delta = 1$. तब $ y = ac' -
a'c = 1 + \tfrac32t^2$ और
\[
x = cb' - c'b = t + \tfrac{t^3}2 - t\Bigl(1 +
\tfrac32t^2\Bigr) = -t^3 .
\]
खत्म करना $ t $: $ t^2 = \tfrac23(y - 1)$ और $ x^2 = t^6 =
\tfrac8{27}(y - 1)^3$: शीर्ष के साथ एक अर्धघनाकार परवलय
$(0, 1)$.

\textbf{13.} $ T = (1, t)/\sqrt{1+t^2}$, $ N = (-t,
1)/\sqrt{1+t^2}$, और $\kappa = (1+t^2)^{-3/2}$, इसलिए
\[
\gamma + \frac1\kappa N = (t, \tfrac{t^2}2) +
(1+t^2)\,(-t, 1) = \Bigl(-t^3,\ 1 + \tfrac32t^2\Bigr),
\]
वही वक्र: \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} सामान्य का स्थान है
के केंद्र \omterm{thm:b2:curves:frenet2d}{वक्रता}, जैसा कि प्रश्न 4 में वादा किया गया था।

\textbf{14.} $ E'(t) = (-3t^2, 3t)$ पर गायब हो जाता है $ t = 0$;
$ E''(0) = (0, 3) \neq 0$ देता है $ p = 2$ और $ E'''(0) = (-6,
0)$ देता है $ q = 3$: पहली तरह का एक पुच्छल $(0, 1)$. द
शिखर कहाँ है $\kappa = (1+t^2)^{-3/2}$ अधिकतम है, इसलिए
$\kappa'(0) = 0$ और उत्क्रांति का वेग
$-\frac{\kappa'}{\kappa^2}N $ \omterm{def:b2:series:def}{बिल्कुल} वहीं गायब हो जाता है: क्यूप्स ऑफ
विकसित व्यक्ति चरम सीमा पर बैठता है \omterm{thm:b2:curves:frenet2d}{वक्रता}.

\textbf{15.} पैरामीटर पर सामान्य $ t $ से गुजरता है
$(x_0, y_0)$ आईएफएफ $ x_0 + t\,y_0 = t + \tfrac{t^3}2$, यानी
\[
\frac{t^3}2 + (1 - y_0)\,t - x_0 = 0 ,
\]
एक घन इंच $ t $: एक या तीन वास्तविक जड़ें (बिना गिनती के)।
बहुलता, सामान्य बिंदुओं के लिए)। धुरी पर $ x_0 = 0$ यह
जैसे कारक $ t\bigl(\tfrac{t^2}2 + 1 - y_0\bigr) = 0$: द
जड़ $ t = 0$ (शीर्ष पर अक्ष सामान्य है), प्लस
$ t = \pm\sqrt{2(y_0 - 1)}$ कब $ y_0 > 1$. तो: एक सामान्य
के लिए $ y_0 < 1$, तीन के लिए $ y_0 > 1$, और कम से $ y_0 = 1$ द
ट्रिपल रूट विकसित के शिखर को चिह्नित करता है। सामान्य तौर पर ए
घन के दोहरे मूल का अर्थ है कि बिंदु दोनों को संतुष्ट करता है
रेखा समीकरण और उसका $ t $-व्युत्पन्न --- यह झूठ है \emph{पर
\omterm{pb:b2:curves:1}{लिफ़ाफ़ा}}: evolute ठीक बीच की सीमा है
एक-सामान्य और तीन-सामान्य क्षेत्र।

\textbf{16.} दर्पण में प्रतिबिंब सामान्य को उलट देता है
दिशा का घटक और स्पर्शरेखा को बनाए रखता है:
लेखन $ u = \langle u, n\rangle n + u_{\mathrm{tan}}$, द
प्रतिबिंबित दिशा है $ u_{\mathrm{tan}} - \langle u,
n\rangle n = u - 2\langle u, n\rangle n $. यहाँ $\langle u,
n\rangle = \cos\theta $, इसलिए
\[
v = (1, 0) - 2\cos\theta\,(\cos\theta, \sin\theta)
= (1 - 2\cos^2\theta,\ -2\sin\theta\cos\theta)
= -(\cos2\theta,\ \sin2\theta).
\]

\textbf{17.} परावर्तित किरण गुजरती है $ P_\theta =
(\cos\theta, \sin\theta)$ दिशा के साथ $(\cos2\theta,
\sin2\theta)$; एक सामान्य वेक्टर है $(-\sin2\theta,
\cos2\theta)$, तो लाइन है
\[
-\sin2\theta\,(x - \cos\theta) + \cos2\theta\,(y -
\sin\theta) = 0,
\]
और स्थिरांक है $-\sin2\theta\cos\theta +
\cos2\theta\sin\theta = -\sin\theta $: से गुणा करना $-1$,
$ x\sin2\theta - y\cos2\theta = \sin\theta $.

\textbf{18.} $(a, b, c) = (\sin2\theta, -\cos2\theta,
\sin\theta)$: $ a' = 2\cos2\theta $, $ b' = 2\sin2\theta $, $ c' =
\cos\theta $, $\Delta = 2\sin^22\theta + 2\cos^22\theta = 2$.
क्रैमर, फिर उत्पाद-से-योग सूत्र:
\begin{align*}
x &= \frac{2\sin\theta\sin2\theta +
\cos\theta\cos2\theta}{2}
= \frac{(\cos\theta - \cos3\theta) + \frac12(\cos\theta +
\cos3\theta)}{2} = \frac{3\cos\theta - \cos3\theta}4,\\
y &= \frac{\sin2\theta\cos\theta - 2\cos2\theta\sin\theta}2
= \frac{\frac12(\sin3\theta + \sin\theta) - (\sin3\theta -
\sin\theta)}2 = \frac{3\sin\theta - \sin3\theta}4 :
\end{align*}
 \omterm{pb:b2:curves:1}{नेफ्रोइड}, दो क्यूप्स वाला एक बंद वक्र।

\textbf{19.} विभेद करना और गुणनखंडन करना $\sin3\theta
- \sin\theta = 2\cos2\theta\sin\theta $, $\cos\theta -
\cos3\theta = 2\sin2\theta\sin\theta $:
\[
E'(\theta) = \tfrac34\bigl(\sin3\theta - \sin\theta,\
\cos\theta - \cos3\theta\bigr)
= \tfrac32\sin\theta\,(\cos2\theta, \sin2\theta),
\]
प्रश्न 16 की प्रतिबिंबित दिशा के समानांतर: प्रत्येक
परावर्तित किरण कास्टिक की स्पर्शरेखा होती है, जैसे कि \omterm{pb:b2:curves:1}{लिफ़ाफ़ा}
संपत्ति की मांग. $ E' = 0$ बिलकुल पर $\sin\theta = 0$:
$ E(0) = (\tfrac12, 0)$ और $ E(\pi) = (-\tfrac12, 0)$, दो
cusps. सेटिंग $ y = 0$ पंक्ति समीकरण में: $ x\sin2\theta =
\sin\theta $, इसलिए $ x = \frac1{2\cos\theta} \to \frac12$ जैसा
$\theta \to 0$: पैराएक्सियल किरणें दूरी पर केंद्रित होती हैं $ R/2$ से
केंद्र---फोकल \omterm{def:b2:curves:length}{लंबाई} गोलाकार दर्पण का.

\textbf{20.} $\norm{E'(\theta)} = \tfrac32\abs{\sin\theta}$,
इतना \omterm{def:b2:curves:length}{लंबाई} है $\tfrac32\int_0^{2\pi}\abs{\sin\theta}\,
\dd\theta = \tfrac32\cdot4 = 6$. पास में $\theta = 0$, साथ
$\cos k\theta = 1 - \tfrac{k^2\theta^2}2 + O(\theta^4)$ और
$\sin k\theta = k\theta - \tfrac{k^3\theta^3}6 +
O(\theta^5)$:
\[
x - \tfrac12 = \frac{3\theta^2 + O(\theta^4)}{4}
= \tfrac34\theta^2 + O(\theta^4),
\qquad
y = \frac{4\theta^3 + O(\theta^5)}{4}
= \theta^3 + O(\theta^5) :
\]
$ p = 2$, $ q = 3$, की ओर इशारा करते हुए पहली तरह का एक पुच्छल
अक्ष --- कॉफ़ी-कप कास्टिक का चमकीला बिंदु।

\textbf{21.} द्वारा प्रकाशित बिंदुओं को पैरामीट्रिज करें
$\Phi(\theta, r) = P_\theta + r\,v_\theta $ (साथ में स्थिति
प्रत्येक परावर्तित किरण)। जैकोबियन सिद्ध
$\det(\partial_\theta\Phi, \partial_r\Phi) =
\det(P_\theta' + rv_\theta', v_\theta)$ है \omterm{def:b2:affine:subspace}{एफ़िन} में $ r $ और
\omterm{def:b2:series:def}{बिल्कुल} एक के लिए गायब हो जाता है $ r = r_*(\theta)$ --- और
$\Phi(\theta, r_*(\theta))$ चूँकि, विशेषता बिंदु है
वहां किरण की दिशा और परिवार की विविधता बन जाती है
आश्रित. से परे \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} मानचित्र एक स्थानीय भिन्नता है,
और कास्टिक के ठीक अंदर एक बिंदु पर पास की दो किरणें पड़ती हैं
(दो समाधान $\theta $), उससे बाहर एक बिंदु
परिवार का हिस्सा; कास्टिक पर दोनों विलीन हो जाते हैं। रोशनी
तीव्रता जैकोबियन के निरपेक्षता के विपरीत आनुपातिक है
मूल्य, इसलिए यह साथ-साथ उड़ता है \omterm{pb:b2:curves:1}{लिफ़ाफ़ा}: कास्टिक है
चमकदार वक्र, शिखर पर सबसे चमकीला, जहां
पतन सबसे खराब है.

\textbf{22.} $ x' = 1 - \cos t $, $ y' = \sin t $, $ x'' = \sin
t $, $ y'' = \cos t $, इसलिए $ x'y'' - y'x'' = \cos t - 1$ और
$\norm{\gamma'}^2 = 2(1 - \cos t) = 4\sin^2\tfrac t2$;
द्वारा \cref{prop:b2:curves:kappaformula},
\[
\kappa(t) = \frac{-(1 - \cos t)}{8\sin^3\tfrac t2}
= \frac{-2\sin^2\tfrac t2}{8\sin^3\tfrac t2}
= -\frac1{4\sin\tfrac t2}
\qquad (0 < t < 2\pi).
\]
साथ $ T = (\sin\tfrac t2, \cos\tfrac t2)$ (विभाजित करना $\gamma'$
द्वारा $2\sin\tfrac t2$) और $ N = (-\cos\tfrac t2, \sin\tfrac
t2)$:
\[
\gamma + \frac1\kappa N
= \gamma - 4\sin\tfrac t2\,\Bigl(-\cos\tfrac t2,\
\sin\tfrac t2\Bigr)
= \bigl(t - \sin t + 2\sin t,\ 1 - \cos t - 2(1 - \cos
t)\bigr),
\]
यानी\ $ c(t) = (t + \sin t,\ \cos t - 1)$. स्थानापन्न $ t = u
+ \pi $:
\[
c = \bigl(u + \pi - \sin u,\ -\cos u - 1\bigr)
= \bigl((u - \sin u) + \pi,\ (1 - \cos u) - 2\bigr) :
\]
चक्रवात $\gamma(u)$ द्वारा अनुवादित $(\pi, -2)$. द
एक चक्रवात का विकास एक सर्वांगसम चक्रवात है, जो लटका हुआ है
नीचे का स्तर.

\textbf{23.} $ R(t) = 1/\abs{\kappa(t)} = 4\sin\tfrac t2$, इसलिए
$ R(\pi) = 4$: मेहराब का आधा भाग \omterm{def:b2:curves:length}{लंबाई} $8$ में गणना की गई
\cref{exo:b2:curves:1}. उत्क्रांति का वेग $ c'(t) = (1 +
\cos t, -\sin t)$ पर गायब हो जाता है $ t = \pi $: शिखर है $ c(\pi)
= (\pi, -2)$, सीधे शीर्ष के नीचे $\gamma(\pi) = (\pi,
2)$, दूरी पर $4 = R(\pi)$, बिलकुल \omterm{def:b2:curves:length}{लंबाई} का
वहाँ osculated त्रिज्या.

\textbf{24.} से \cref{exo:b2:curves:6}, $ c'(s) =
-\frac{\kappa'(s)}{\kappa(s)^2}N(s) = R'(s)\,N(s)$, इसलिए
$\norm{c'(s)} = \abs{R'(s)}$ और, एकरसता के लिए $ R $,
\[
\int_{s_0}^{s_1}\norm{c'(s)}\,\dd s =
\Bigl|\int_{s_0}^{s_1}R'(s)\,\dd s\Bigr| = \abs{R(s_1) -
R(s_0)} .
\]
कहना $ R $ घट जाती है. परे उत्क्रांति के साथ रखी एक डोरी
$ c(s_0)$ और खंड द्वारा लंबे समय तक $ c(s_0)$ को
$\gamma(s_0)$ (जो प्रश्न 4 के अनुसार, वृत्तांत की स्पर्शरेखा है)
है, जब छील दिया जाता है $ c(s)$ और तना हुआ, सीधा
का हिस्सा \omterm{def:b2:curves:length}{लंबाई} $ R(s_0) - \bigl(R(s_0) - R(s)\bigr) = R(s)$
से इशारा करते हुए $ c(s)$ सामान्य के साथ--बिल्कुल लैंडिंग
$\gamma(s)$: मुक्त सिरा मूल वक्र का पता लगाता है
(``इनवॉल्व'')। ह्यूजेंस ने दो साइक्लोइडल के बीच एक पेंडुलम लटकाया
गाल: नाल विकसित पर लपेटता है, इसलिए बॉब एक का वर्णन करता है
साइक्लोइड --- टॉटोक्रोन, जिसका दोलन काल होता है
आयाम पर निर्भर नहीं है.

\textbf{25.} परवलय की स्पर्श रेखाएँ: $\Delta = 1$ और $ E' =
(1, t) \neq 0$ हर जगह --- सहज \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} (परवलय
स्वयं)। फिसलने वाली सीढ़ी: $\Delta = -1$, लेकिन $ E' =
\tfrac32\sin 2t\,(-\cos t, \sin t)$ चतुर्थांश पर लुप्त हो जाता है
समाप्त होता है --- के चार सिरे \omterm{pb:b2:curves:1}{astroid}. के सामान्य
परवलय और चक्रवात का: $\Delta = \kappa \neq 0$, और
$ E' = R'N $ \omterm{def:b2:series:def}{बिल्कुल} वहीं गायब हो जाता है जहां \omterm{thm:b2:curves:frenet2d}{वक्रता} चरम है
--- उत्क्रांति के पुच्छल $(0,1)$ और $(\pi, -2)$. कास्टिक:
$\Delta = 2$, और $ E' = \tfrac32\sin\theta\,(\cos2\theta,
\sin2\theta)$ पर गायब हो जाता है $\theta = 0, \pi $ --- दो पुच्छल
का \omterm{pb:b2:curves:1}{नेफ्रोइड}, दर्पण के केन्द्र बिन्दु। की चरम सीमा
\omterm{thm:b2:curves:frenet2d}{वक्रता} \emph{अवश्य} पर क्यूस्प्स का उत्पादन करें \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} का
सामान्य, क्योंकि उत्क्रांति का वेग है $ R'N$: इसीलिए
दीर्घवृत्त के परिभ्रमण में चार पुच्छल (चार शीर्ष) होते हैं,
और पतित परिवार (पेंसिल, समानताएं) मामले हैं
जहां मशीन एक बिंदु या कुछ भी आउटपुट नहीं देती है।
\end{solution}
