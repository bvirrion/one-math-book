\section*{अध्याय \ref{ch:b2:structures} --- सेट और संरचनाएँ}

\begin{solution}{exo:b2:structures:1}
\emph{के परिमित उपसमुच्चय $\N $:} \omterm{def:b2:structures:countable}{गणनीय} --- के उपसमुच्चय का समुच्चय
$\intint{0}{n}$ परिमित है, और परिमित उपसमुच्चय बनाते हैं \omterm{def:b2:structures:countable}{गणनीय}
संघ ख़त्म $ n $ यहाँ इन
(\cref{prop:b2:structures:countablestable}~(3)); इसके बाद से अनंत
सभी सिंगलटन शामिल हैं।

\emph{के सभी उपसमुच्चय $\N $:} नहीं \omterm{def:b2:structures:countable}{गणनीय}, कैंटर के प्रमेय द्वारा
(\cref{thm:b2:structures:cantor}~(1)साथ $ E = \N $).

\emph{$\R \setminus \Q $:} नहीं \omterm{def:b2:structures:countable}{गणनीय} --- अन्यथा $\R = \Q \cup
(\R\setminus\Q)$ दो का मिलन होगा \omterm{def:b2:structures:countable}{गणनीय} समुच्चय,
विरोधाभासी \cref{thm:b2:structures:cantor}~(2).

\emph{बहुपद ख़त्म $\Q $:} \omterm{def:b2:structures:countable}{गणनीय} --- डिग्री के बहुपद
$\leq n $ में इंजेक्ट करें $\Q^{n+1}$ (के परिमित उत्पाद \omterm{def:b2:structures:countable}{गणनीय} समुच्चय),
और संघ को अपने अधिकार में ले लो $ n $.

\emph{अंततः-शून्य बाइनरी अनुक्रम:} \omterm{def:b2:structures:countable}{गणनीय} --- वे आपत्ति जताते हैं
के परिमित उपसमुच्चय $\N $ (सहारा)।
\end{solution}

\begin{solution}{exo:b2:structures:2}
पहले सही कारक लागू करके तत्व दर तत्व की गणना करें।
$\sigma\tau $ भेजता है $1 \mapsto \sigma(1) = 4$, $\;2 \mapsto \sigma(3)
= 5$, $\;3 \mapsto \sigma(7) = 7$, $\;4 \mapsto \sigma(4) = 2$, $\;5
\mapsto \sigma(5) = 3$, $\;6 \mapsto \sigma(6) = 1$, $\;7 \mapsto
\sigma(2) = 6$:
\[
\sigma\tau = (1\,4\,2\,5\,3\,7\,6),
\]
ए $7$-चक्र। वैसे ही $\tau\sigma $ भेजता है $1 \mapsto \tau(4) = 4$,
$\;2 \mapsto \tau(6) = 6$, $\;3 \mapsto \tau(5) = 5$, $\;4 \mapsto
\tau(2) = 3$, $\;5 \mapsto \tau(3) = 7$, $\;6 \mapsto \tau(1) = 1$,
$\;7 \mapsto \tau(7) = 2$:
\[
\tau\sigma = (1\,4\,3\,5\,7\,2\,6),
\]
यह भी एक $7$-चक्र (जैसा कि अपेक्षित था: $\sigma\tau $ और $\tau\sigma $ हैं
संयुग्मित करें, इसलिए उनका साझा करें \omterm{def:b2:structures:sn}{चक्र} प्रकार)।

\omterm{def:b2:structures:generated}{आदेश} और \omterm{thm:b2:structures:signature}{हस्ताक्षर}: $\sigma $ है \omterm{def:b2:structures:sn}{चक्र} प्रकार $(4,2)$: \omterm{def:b2:structures:generated}{आदेश}
$\operatorname{lcm}(4,2) = 4$, \omterm{thm:b2:structures:signature}{हस्ताक्षर} $(-1)^3(-1)^1 = +1$; $\tau $
एक है $3$-चक्र: \omterm{def:b2:structures:generated}{आदेश} $3$, \omterm{thm:b2:structures:signature}{हस्ताक्षर} $+1$; दोनों उत्पाद हैं
$7$-चक्र: \omterm{def:b2:structures:generated}{आदेश} $7$, \omterm{thm:b2:structures:signature}{हस्ताक्षर} $(-1)^6 = +1$.

$\sigma^{2026}$: $2026 = 4 \times 506 + 2$, इसलिए $\sigma^{2026} =
\sigma^2 = (1\,2)(4\,6)$ (वर्ग करें $4$-चक्र; \omterm{def:b2:structures:sn}{स्थानांतरण}
वर्ग दूर)।
\end{solution}

\begin{solution}{exo:b2:structures:3}
$\{0,1\}^{\N} \to \mathcal{P}(\N)$: इसके समर्थन के लिए एक अनुक्रम मानचित्र
--- एक आक्षेप (संकेतक कार्य), किसी प्रमेय की आवश्यकता नहीं।

$\{0,1\}^{\N} \to \intcc{0}{1}$: आधार-$3$ नक्शा $(a_n) \mapsto
\sum 2a_n 3^{-n-1}$ इंजेक्शन है (दो अलग-अलग अनुक्रम भिन्न हैं
प्रथम रैंक पर $ N $; पूँछें किसी अंतर की भरपाई नहीं कर सकतीं $2\cdot
3^{-N-1}$, तब से $\sum_{n > N} 2\cdot 3^{-n-1} = 3^{-N-1} <
2\cdot3^{-N-1}$).

$\intcc{0}{1} \to \{0,1\}^{\N}$: द्विआधारी विस्तार, चुनना (कहना)
विस्तार बिल्कुल ख़त्म नहीं हो रहा है $1$'s: इंजेक्शन.

कैंटर--बर्नस्टीन द्वारा (\cref{thm:b2:structures:cantorbernstein})
पिछले दो इंजेक्शनों पर लागू, $\intcc{0}{1}$ और
$\{0,1\}^{\N}$ हैं \omterm{def:b2:structures:countable}{सुसज्जित}, इसलिए सभी तीन सेट हैं।
\end{solution}

\begin{solution}{exo:b2:structures:4}
होने देना $ c = ab = ba $ और $ d = \operatorname{ord}(c)$. पहला $ c^{mn} =
a^{mn} b^{mn} = e $ (कम्यूटेशन शक्ति को विभाजित करने की अनुमति देता है), इसलिए $ d
\mid mn $. इसके विपरीत $ c^d = e $ देता है $ a^d = b^{-d}$; यह तत्व
में निहित है $\langle a\rangle \cap \langle b\rangle $, एक उपसमूह जिसका
\omterm{def:b2:structures:generated}{आदेश} दोनों को विभाजित करता है $ m $ और $ n $ (प्रत्येक में लैग्रेंज \omterm{def:b2:structures:generated}{चक्रीय} समूह),
इसलिए यह तुच्छ है: $ a^d = b^d = e $, इसलिए $ m \mid d $ और $ n \mid d $, और
सहप्रधानता द्वारा $ mn \mid d $. इस तरह $ d = mn $.

में $\mathfrak{S}_3$: लेना $ a = (1\,2)$ (\omterm{def:b2:structures:generated}{आदेश} $2$) और $ b =
(1\,2\,3)$ (\omterm{def:b2:structures:generated}{आदेश} $3$), सह अभाज्य \omterm{def:b2:structures:generated}{आदेश}, जो आवागमन नहीं करते: $ ab =
(2\,3)$ है \omterm{def:b2:structures:generated}{आदेश} $2 \neq 6$ --- वास्तव में $\mathfrak{S}_3$ नहीं है
का तत्व \omterm{def:b2:structures:generated}{आदेश} $6$. आवागमन आवश्यक है.
\end{solution}

\begin{solution}{exo:b2:structures:5}
हर जोड़ी $ x \in G $ साथ $ x^{-1}$. जोड़े $\{x, x^{-1}\}$ साथ
$ x \neq x^{-1}$ दो तत्व हैं और उनका संघ विभाजित है; द
शेष तत्व बिल्कुल वही हैं जिनके साथ हैं $ x = x^{-1}$, यानी\ $ x^2 =
e $. तब से $\abs G $ सम है और दो-तत्व जोड़े एक सम को कवर करते हैं
तत्वों की संख्या, सेट $\{x : x^2 = e\}$ यहां तक कि प्रमुखता भी है;
इसमें है $ e $, इसलिए इसमें कम से कम एक अन्य तत्व शामिल है $ x \neq
e $ --- का एक तत्व \omterm{def:b2:structures:generated}{आदेश} $2$.
\end{solution}

\begin{solution}{exo:b2:structures:6}
का प्रत्येक तत्व $ A_n $ की सम संख्या का गुणनफल है
प्रतिस्थापन (\cref{thm:b2:structures:signature}: में विघटित करें
प्रतिस्थापन; \omterm{thm:b2:structures:signature}{हस्ताक्षर} होने के बाद से गिनती सम है $+1$). यह
दो का प्रत्येक गुणनफल लिखना पर्याप्त है प्रतिस्थापन साथ
$3$-चक्र:
\[
(a\,b)(a\,c) = (a\,c\,b),
\qquad
(a\,b)(c\,d) = (a\,c\,b)(a\,c\,d) \quad (\text{distinct } a,b,c,d),
\]
(मूल्यांकन द्वारा जांचें), और $(a\,b)(a\,b) = \mathrm{id}$. इतना
$3$-चक्र उत्पन्न होते हैं $ A_n $.
\end{solution}

\begin{solution}{exo:b2:structures:7}
\emph{$(\Q,+) \to (\Z,+)$:} केवल शून्य रूपवाद। किसी के लिए $ x $ और
प्रत्येक $ n \geq 1$, $ f(x) = n f\bigl(\frac xn\bigr)$ से विभाज्य है
$ n $ में $\Z $; प्रत्येक से विभाज्य एकमात्र पूर्णांक $ n $ है $0$, इसलिए
$ f(x) = 0$ सभी के लिए $ x $.

\emph{$(\Z/n\Z, +) \to (\Z/m\Z, +)$:} एक रूपवाद द्वारा निर्धारित किया जाता है
$ c = f(\overline 1)$, जिसे संतुष्ट करना होगा $ n c \equiv 0 \pmod m $,
यानी\ $ c $ का गुणज है $\frac{m}{\gcd(m,n)}$; वहाँ हैं
$\gcd(m,n)$ ऐसी कक्षाएं, और प्रत्येक विकल्प एक रूपवाद को परिभाषित करता है
(कारक $ k \mapsto kc $ के माध्यम से $\Z/n\Z $ सार्वभौमिक संपत्ति द्वारा)।

\emph{$(\Q, +) \to (\Q_+^*, \times)$:} केवल तुच्छ। अगर
$ f(x) = y $, फिर प्रत्येक के लिए $ n $, $ y = f(n \cdot \frac xn) =
f(\frac xn)^n $ एक $ n $-वें शक्ति में $\Q_+^*$. लेकिन एक तर्कसंगत
$ y \neq 1$ एक नहीं हो सकता $ n $-सभी के लिए शक्ति $ n $: कुछ अभाज्य प्रकट होता है
में $ y $ एक शून्येतर घातांक के साथ $ v $, और $ n \nmid v $ के लिए $ n > \abs
v $ (के प्रतिपादक $ n $-th शक्तियां गुणज हैं $ n $, अद्वितीय द्वारा
गुणनखंडन)। इस तरह $ f \equiv 1$.
\end{solution}

\begin{solution}{exo:b2:structures:8}
$360 = 2^3 \cdot 3^2 \cdot 5$:
$\varphi(360) = 360\bigl(1 - \tfrac12\bigr)\bigl(1 -
\tfrac13\bigr)\bigl(1 - \tfrac15\bigr) = 360 \cdot \tfrac12 \cdot
\tfrac23 \cdot \tfrac45 = 96$.

सिस्टम: मॉड्यूलि $8, 9, 5$ जोड़ीवार सहअभाज्य, कुल $360$. से $ x
\equiv 3 \pmod 8$ और $ x \equiv 5 \pmod 9$: $ x = 3 + 8k $ साथ $3 +
8k \equiv 5 \pmod 9$, यानी\ $-k \equiv 2$, $ k \equiv -2 \equiv 7
\pmod 9$: $ x \equiv 3 + 56 = 59 \pmod{72}$. तब $59 + 72\ell \equiv
2 \pmod 5$: $4 + 2\ell \equiv 2$, $2\ell \equiv 3 \equiv 8$, $\ell
\equiv 4 \pmod 5$: $ x \equiv 59 + 288 = 347 \pmod{360}$.

के अंतिम दो अंक $3^{2026}$: मॉड $4$, $3^{2026} = 9^{1013} \equiv
1$. आधुनिक $25$: $\varphi(25) = 20$ और $2026 = 20\cdot101 + 6$, इसलिए
$3^{2026} \equiv 3^6 = 729 \equiv 4 \pmod{25}$. हल करना $ x \equiv 1
\pmod 4$, $ x \equiv 4 \pmod{25}$: $ x = 4 + 25k \equiv 1 \pmod 4$
देता है $ k \equiv 1 \pmod 4$: $ x \equiv 29 \pmod{100}$. आखिरी दो
अंक हैं $29$.
\end{solution}

\begin{solution}{exo:b2:structures:9}
होने देना $ A $ एक परिमित अभिन्न डोमेन बनें और $ a \in A $, $ a \neq 0$. वो नक्शा $ x \mapsto ax $ इंजेक्शन है ($ ax = ay \implies a(x - y) = 0
\implies x = y $, कोई शून्य भाजक नहीं); एक परिमित समुच्चय का एक विशेषण मानचित्र
स्वयं विशेषण है (वर्ष 1 खंड, पिजनहोल तुल्यता)।
इसलिए $1 = ab $ कुछ के लिए $ b $: प्रत्येक अशून्य तत्व व्युत्क्रमणीय है, $ A $
एक मैदान है.

$\Z/n\Z $: अगर $ n $ अभाज्य है यह एक अभिन्न डोमेन है ($ n \mid ab
\implies n \mid a $ या $ n \mid b $, यूक्लिड की लेम्मा), परिमित, इसलिए ए
क्षेत्र; अगर $ n = rs $ समग्र है, $\overline r\,\overline s =
\overline 0$ शून्य भाजक प्रदर्शित करता है।
\end{solution}

\begin{solution}{exo:b2:structures:10}
होने देना $ m = \max\{\operatorname{ord}(x) : x \in G\}$, पर प्राप्त किया गया $ a $.

\emph{दावा: हर $ x \in G $ है \omterm{def:b2:structures:generated}{आदेश} डिवाइडिंग $ m $.} मान लीजिए कुछ
$ x $ है \omterm{def:b2:structures:generated}{आदेश} $ q $ साथ $ q \nmid m $: फिर कुछ प्रधान शक्ति $ p^k $
विभाजित $ q $ लेकिन नहीं $ m $. लिखना $ m = p^j m'$ साथ $ p \nmid m'$ और $ j
< k $. तत्व $ a^{p^j}$ है \omterm{def:b2:structures:generated}{आदेश} $ m'$; तत्व
$ x^{q/p^k}$ है \omterm{def:b2:structures:generated}{आदेश} $ p^k $; इन \omterm{def:b2:structures:generated}{आदेश} सहअभाज्य और दो हैं
तत्व आवागमन करते हैं ($ G \subseteq K^*$ एबेलियन है), इसलिए द्वारा
\cref{exo:b2:structures:4} उनके उत्पाद है \omterm{def:b2:structures:generated}{आदेश} $ p^k m' > p^j m'
= m $: अधिकतमता का खंडन।

तो सभी $ x \in G $ संतुष्ट $ x^m = 1$: बहुपद $ X^m - 1$ पर है
कम से कम $\abs G $ क्षेत्र में जड़ें $ K $, कहाँ से $\abs G \leq m $
(डिग्री का एक शून्येतर बहुपद $ m $ अधिकतम है $ m $ जड़ें, वर्ष 1
वॉल्यूम). लेकिन $ m = \operatorname{ord}(a) \leq \abs G $ लैग्रेंज द्वारा.
इस तरह $ m = \abs G $ और $\langle a \rangle $, कार्डिनैलिटी का $ m =
\abs G $, सब कुछ है $ G $: \omterm{def:b2:structures:generated}{चक्रीय}.

के लिए $ K = \Z/p\Z $: $(\Z/p\Z)^*$ का एक परिमित उपसमूह है $ K^*$, इस तरह
\omterm{def:b2:structures:generated}{चक्रीय} (का \omterm{def:b2:structures:generated}{आदेश} $ p - 1$).
\end{solution}

\begin{solution}{exo:b2:structures:11}
\emph{नहीं \omterm{def:b2:structures:generated}{चक्रीय}:} उपसमूह $\langle \frac pq\rangle $ शामिल हैं
के पूर्णांक गुणजों का $\frac pq $, जिनमें से सभी के पास है
हर को विभाजित करना $ q $ (न्यूनतम शब्दों में); इसलिए यह चूक जाता है
$\frac{1}{2q}$. कोई भी अकेला जनरेटर असीमित तक नहीं पहुंच सकता
के भाजक $\Q $.

\emph{अंतिम रूप से नहीं आप जेनरेट हुई:} उपसमूह आप जेनरेट हुई द्वारा
$\frac{p_1}{q_1}, \dots, \frac{p_k}{q_k}$ तर्कसंगत से मिलकर बनता है
जिनके हर विभाजित होते हैं $ Q = q_1 \cdots q_k $ (पूर्णांक संयोजन
हर को विभाजित करना है $ Q $): यह चूक जाता है $\frac{1}{2Q}$.

\emph{परिमित आप जेनरेट हुई उपसमूह हैं \omterm{def:b2:structures:generated}{चक्रीय}:} साथ $ Q $ उपरोक्तानुसार,
उपसमूह $ H = \langle \frac{p_1}{q_1}, \dots,
\frac{p_k}{q_k}\rangle $ में निहित है $\frac{1}{Q}\Z $. वो नक्शा $ x
\mapsto Qx $ से एक समरूपता है $\frac1Q\Z $ पर $\Z $ भार उठाते
$ H $ के एक उपसमूह को $\Z $, जो है $ n\Z $ कुछ के लिए $ n $ (वर्ष 1
वॉल्यूम): तो $ H = \frac{n}{Q}\Z $ है \omterm{def:b2:structures:generated}{चक्रीय}, आप जेनरेट हुई द्वारा
$\frac nQ $.
\end{solution}

\begin{solution}{exo:b2:structures:12}
\emph{ए \omterm{def:b2:structures:countable}{गणनीय} उपसमुच्चय} होने देना $ E $ अनंत हो. निर्माण $ a_0,
a_1, a_2, \dots $ आगमनात्मक रूप से: $ E $ गैर-रिक्त है, चुनें $ a_0 \in E $;
अगर $ a_0, \dots, a_n $ चुने गए हैं, $ E \setminus \{a_0, \dots,
a_n\}$ गैर-रिक्त है ($ E $ परिमित नहीं है), चुनें $ a_{n+1}$ वहाँ।
$ a_n $ निर्माण के आधार पर जोड़ीवार अलग-अलग हैं, इसलिए $ A = \{a_n : n
\in \N\}$ एक है \omterm{def:b2:structures:countable}{गणनीय} का उपसमुच्चय $ E $.

\emph{अनंत $\implies $ \omterm{def:b2:structures:countable}{सुसज्जित} एक उचित उपसमुच्चय के लिए.} परिभाषित करना
$ f \colon E \to E \setminus \{a_0\}$ द्वारा $ f(a_n) = a_{n+1}$ और
$ f(x) = x $ के लिए $ x \notin A $. यह इंजेक्शन है (दो टुकड़े हैं
असंयुक्त छवियों के साथ विशेषण) और पर विशेषण $ E \setminus
\{a_0\}$: प्रत्येक $ a_{n+1}$ हिट है, हर $ x \notin A $ हिट है. इसलिए

= 20\,922\,789\,888\,000$ उनमें से. एक कानूनी कदम एक टाइल को खिसका देता है
रिक्त स्थान के निकट, इसलिए चालों की संख्या की संख्या है
रिक्त कक्ष के पड़ोसी: $2$ चार कोने वाली कोशिकाओं के लिए,
$3$ आठ किनारे वाली कोशिकाओं के लिए, $4$ चार आंतरिक कोशिकाओं के लिए.

\textbf{2.} स्लाइड के बाद, सेल $ p $ की पूर्व सामग्री रखती है

$\varepsilon\bigl((p\ c)\bigr) = -1$:

\textbf{3.} निकटवर्ती कोशिकाएँ बिल्कुल एक चरण में भिन्न होती हैं

रिक्त स्थान की एक बंद सैर के साथ, $\chi $ प्रति चाल एक बार फ़्लिप किया जाता है

\textbf{8.} अगर $\sigma(16) = 16$, तब $\sigma $ को क्रमपरिवर्तित करता है

\textbf{9.} \emph{एकल चाल:} की सामग्री $ c $ में समाप्त होता है $ p $

\emph{प्रेरण:} यदि किसी क्रम का प्रभाव पड़ता है $\pi_1$ और ले जाता है

प्रभाव का $\pi_2 = (p'\ c')$ पैदावार $(\sigma \circ \pi_1^{-1})
\circ \pi_2^{-1} = \sigma \circ (\pi_2\pi_1)^{-1}$, और सामग्री
आगे बढ़ें $\pi_2 \circ \pi_1$ (पहला $\pi_1$, तब $\pi_2$). तो
प्रभाव बनाते हैं, और एक प्रोग्राम चलाते हैं $\sigma $ पर समाप्त होता है $\sigma
\circ \pi^{-1}$. \emph{उपसमूह:} खाली प्रोग्राम का प्रभाव होता है
$\mathrm{id}$; संयोजन उत्पाद देता है; किसी प्रोग्राम को उलटना
(प्रश्न 7) व्युत्क्रम देता है। एक प्रोग्राम का प्रभाव सेल को ठीक करता है $16$

\emph{समता:} का एक कार्यक्रम $ k $ चाल है $ k $ यहां तक कि (प्रश्न
3), और $\varepsilon(\sigma \circ \pi^{-1}) =

\mapsto 1$, $3 \mapsto 2$, $4 \mapsto 3$, $8 \mapsto 4$, $7

जो बिल्कुल सही है $(11\ 15\ 12)$, रिवर्स प्राथमिक दौरा।

\textbf{12.} होने देना $\gamma = (a\ b\ c)$ और $ x \in \intint1n $. अगर

द्वारा तय किया गया $\gamma $, इसलिए $ x $ तय है. इस तरह $ g\gamma g^{-1} =
(g(a)\ g(b)\ g(c))$. और के लिए $ g, h \in H $, $ ghg^{-1} \in H $ द्वारा
उपसमूह स्वयंसिद्ध.

(b\ c\ d)$ (ए $3$-साइकिल चालू $\{a,b,c\}$ है $(a\ b\ c)$ या उसके
उलटा; इसी तरह चालू $\{b,c,d\}$; इसके द्वारा एक जनरेटर को प्रतिस्थापित करना
उलटे पत्ते $\langle s, t\rangle $ अपरिवर्तित)। फिर, आवेदन करना

\[
st \colon a \mapsto b,\quad b \mapsto a,\quad c \mapsto d,\quad
d \mapsto c, \qquad\text{i.e.}\quad st = (a\ b)(c\ d),

एक डबल \omterm{def:b2:structures:sn}{स्थानांतरण}. उपसमूह $ G = \langle s, t\rangle $
इसमें चार अक्षरों का सम क्रमपरिवर्तन शामिल है, इसलिए $ G \leq
A_4$ और $\abs G \mid 12$; इसमें एक तत्व शामिल है \omterm{def:b2:structures:generated}{आदेश} $3$
और एक \omterm{def:b2:structures:generated}{आदेश} $2$, इसलिए $6 \mid \abs G $ (लैग्रेंज,
\cref{thm:b2:structures:lagrange}, दोनों पर लागू \omterm{def:b2:structures:generated}{चक्रीय}
उपसमूह)। अगर $ A_4$ एक उपसमूह था $ K $ का \omterm{def:b2:structures:generated}{आदेश} $6$, यह होगा
सूचकांक है $2$, और तब $ g^2 \in K $ हरएक के लिए $ g \in A_4$: के लिए
$ g \in K $ यह स्पष्ट है; के लिए $ g \notin K $ एकमात्र सहसमुच्चय हैं
$ K $ और $ gK $, तो कोसेट $ g^2K $ है $ K $ या $ gK $, और $ g^2K =
gK $ जबरदस्ती करेगा $ g \in K $. तो हर वर्ग में निहित है $ K $. लेकिन हर
$3$-चक्र $\gamma $ एक वर्ग है, $\gamma = (\gamma^2)^2$, और
$ A_4$ आठ शामिल हैं $3$-चक्र: $8 > 6$, विरोधाभास। इस तरह
$\abs G = 12$: $ G = A_4$.

के सभी समान क्रमपरिवर्तन $ X_4 = \{x_0, x_1, x_2, x_3\}$. अगर $ G $
के सभी सम क्रमपरिवर्तन शामिल हैं $ X_m = \{x_0, \dots,
x_{m-1}\}$ ($4 \leq m \leq 14$), तब $ s_{m-2} = (x_{m-2}\
x_{m-1}\ x_m)$ है $ u = x_{m-2}, v = x_{m-1} \in X_m $ और नया
पत्र $ w = x_m $: लेम्मा बी और पहला भाग सभी को सम देता है
का क्रमपरिवर्तन $ X_{m+1}$. तक प्रेरण $ m = 14$: $ G \supseteq
A_{15}$ (यहां तक कि सभी पंद्रह कोशिकाओं के क्रमपरिवर्तन), और $ G
\subseteq A_{15}$ प्रत्येक के बाद से $ s_t $ सम है: $\langle s_0, \dots,
s_{12}\rangle = A_{15}$.

\textbf{17.} प्रश्न 13 और 16: $ A_{15} = \langle s_0, \dots,
s_{12}\rangle \subseteq H $; प्रश्न 9: $ H \subseteq A_{15}$. इसलिए
$ H = A_{15}$, का \omterm{def:b2:structures:generated}{आदेश} $15!/2 = 653\,837\,184\,000$: हर सम
पन्द्रह टाइलों की पुनर्व्यवस्था एक कार्यक्रम का प्रभाव है।

\textbf{18.} प्रश्न 8 घटाया गया $ R = \{I = +1\}$ साकार करने के लिए
प्रत्येक सम $\rho \in \mathfrak{S}_{15}$ एक प्रोग्राम द्वारा: द्वारा किया गया
प्रश्न 17. प्रश्न 6 के साथ, $\abs R = 16!/2 =
10\,461\,394\,944\,000$. \emph{दो कक्षाएं:} होने देना $ t_0 = (14\
15)$ पर कार्रवाई \emph{अंतर्वस्तु}: $\varphi(\sigma) = t_0 \circ
\sigma $. की ओर से एक कानूनी कदम $\sigma $ की ओर से एक कानूनी कदम है
$\varphi(\sigma)$ (रिक्त कक्ष अपरिवर्तित है:
$(t_0\sigma)^{-1}(16) = \sigma^{-1}(t_0(16)) = \sigma^{-1}(16)$,
और स्थानांतरित सेल वही है), और $\varphi(\sigma \circ \tau)
= \varphi(\sigma) \circ \tau $: $\varphi $ मानचित्र अनुक्रमों को स्थानांतरित करते हैं
अनुक्रमों को विशेषात्मक रूप से स्थानांतरित करें (यह एक समावेशन है)। यह पलट जाता है $ I $:
$\varepsilon(t_0\sigma) = -\varepsilon(\sigma)$, वही खाली
सेल. इस तरह $\varphi $ कक्षा को मैप करता है $ R = \{I = +1\}$ का
$\mathrm{id}$ विशेष रूप से के वर्ग पर $\varphi(\mathrm{id})
= \sigma_L $, जो इसलिए सब कुछ है $\{I = -1\}$: बिल्कुल दो
कक्षाएं. यह जॉनसन--स्टोरी प्रमेय है।

\textbf{19.} पढ़ने के क्रम में कोशिकाओं को अनुक्रमित करें और दें $ k = 4(i -
1) + j $ रिक्त कक्ष बनें. के व्युत्क्रम गिनें $\sigma $
(कोशिकाओं के जोड़े $ x < y $ साथ $\sigma(x) > \sigma(y)$): के जोड़े
दो टाइल सेल योगदान करते हैं $ N $; रिक्त स्थान वाले जोड़े: कोशिकाएँ
खाली होने के बाद सभी टाइलें पकड़ें $< 16$, प्रत्येक उलटा ($16 - k $
जोड़े), इसके पहले की कोशिकाएँ कभी भी उलटी नहीं होतीं। इसलिए
$\varepsilon(\sigma) = (-1)^{N + 16 - k} = (-1)^{N + k}$. तब से
$ k = 4(i-1) + j \equiv j \pmod 2$,
\[
I(\sigma) = (-1)^{N + j}\,(-1)^{i + j} = (-1)^{N + i}
= (-1)^{N + r + 1}
\]
का उपयोग करते हुए $ i = 5 - r $. प्रश्न 18 द्वारा, $\sigma $ यदि हल करने योग्य है
$ I(\sigma) = +1$ आईएफएफ $ N + r $ अजीब है. जांचें: हल हो गया, $ N = 0$, $ r
= 1$: अजीब, हल करने योग्य; लॉयड, $ N = 1$, $ r = 1$: सम, अघुलनशील।

\textbf{20.} \emph{कार्रवाई:} $ e \cdot \sigma = \sigma \circ
\mathrm{id} = \sigma $ और $ g \cdot (h \cdot \sigma) = \sigma
\circ h^{-1} \circ g^{-1} = \sigma \circ (gh)^{-1} = (gh) \cdot
\sigma $; और $\sigma \circ h^{-1}$ फिर से एक खाली घर है
कॉन्फ़िगरेशन ($ h $ सेल को ठीक करता है $16$). \emph{मुक्त:} $\sigma \circ
h^{-1} = \sigma $ देता है $ h^{-1} = \mathrm{id}$ (साथ लिखें
$\sigma^{-1}$). \emph{कक्षाएँ = कार्यक्रम कक्षाएं:} प्रश्न 9
से पहुंच योग्य कॉन्फ़िगरेशन कहता है $\sigma $ कार्यक्रमों द्वारा हैं
बिलकुल $\sigma \circ \pi^{-1}$, $\pi \in H $: कक्षा $ H
\cdot \sigma $. \emph{गिनती करना:} उन्मुक्तता बनाती है $ h \mapsto h \cdot
\sigma $ इंजेक्शन, तो हर कक्षा में है $\abs H = 15!/2$ तत्व;
 $15!$ इसलिए ब्लैंक-होम कॉन्फ़िगरेशन को विभाजित किया गया है
$15!\,/\,(15!/2) = 2$ कक्षाएँ --- की रिक्त-गृह छाया
दो जॉनसन--कहानी कक्षाएँ।

\textbf{22.} \emph{परिधि भ्रमण} $9 \to 8 \to 7 \to 4 \to 1
\to 2 \to 3 \to 6 \to 9$ (सभी चरण आसन्न; लंबाई $8$, सम):
प्रश्न 11 के बहीखाता द्वारा $ c_1 = 8, c_2 = 7, c_3 =
4, c_4 = 1, c_5 = 2, c_6 = 3, c_7 = 6$, प्रभाव है
\[
\zeta' = (8\ 6\ 3\ 2\ 1\ 4\ 7),
\]
ए $7$-केंद्र को ठीक करने वाली साइकिल $5$ (की सामग्री $7$ की ओर ले जाता है $8$,
का $4$ को $7$, का $1$ को $4$, का $2$ को $1$, का $3$ को $2$, का
$6$ को $3$, और का $8$ को $6$). \emph{कोने का दौरा} $9 \to 6 \to
5 \to 8 \to 9$: प्रभाव $(6\ 8\ 5)$ (की सामग्री $5$ की ओर ले जाता है
$6$, का $8$ को $5$, का $6$ --- पर पार्क किया गया $9$ --- को $8$). तय करना
$ y_t = \zeta'^{\,t}(8)$: $ y_0 = 8, y_1 = 6, y_2 = 3, y_3 = 2,
y_4 = 1, y_5 = 4, y_6 = 7$. संयुग्मन (प्रश्न 12):
\[
\zeta'^{\,t}\,(6\ 8\ 5)\,\zeta'^{-t}
= (y_{t+1}\ y_t\ 5) =: T_t \in H_{3\times3},
\]
तब से $\zeta'$ ठीक करता है $5$. का समर्थन करता है $ T_0 = (y_1\ y_0\ 5)$
और $ T_1 = (y_2\ y_1\ 5)$ बिल्कुल साझा करें $\{y_1, 5\}$: लेम्मा ए
सभी को समान क्रमपरिवर्तन देता है $\{y_0, y_1, y_2, 5\}$. तब
$ T_2 = (y_3\ y_2\ 5)$ साथ संलग्न है $ y_3$ लेम्मा बी द्वारा (इसके अक्षर
$ y_2, 5$ वर्तमान सेट में झूठ बोलो, $ k = 4$), और $ T_3, T_4, T_5$
अगला होना $ y_4, y_5, y_6$ बदले में: सभी समान क्रमपरिवर्तन
कार्यक्रम समूह में आठ गैर-घरेलू सेल भी शामिल हैं
इसमें सम क्रमपरिवर्तन शामिल हैं (प्रश्न 9 का तर्क है
बोर्ड-स्वतंत्र)। इसलिए $ H_{3\times3} = A_8$, और तर्क
प्रश्न 6, 8, 18 --- भी बोर्ड-स्वतंत्र --- दर्शाता है
पहुंच योग्य कॉन्फ़िगरेशन बिल्कुल वही हैं जिनके साथ $ I = +1$: का आधा $9!$, यानी\ $181\,440$.

\textbf{23.} कोशिकाओं को लेबल करें $0, \dots, n-1$ चारों ओर \omterm{def:b2:structures:sn}{चक्र}.
एक चाल अपने दो पड़ोसियों में से एक के साथ रिक्त स्थान की अदला-बदली करती है। पढ़ें
रिक्त स्थान के ठीक बाद शुरू होने वाले \omterm{def:b2:structures:generated}{चक्रीय} क्रम में टाइलें: एक शब्द $ w $
सूचीबद्ध करना $ n - 1$ टाइल्स. रिक्त स्थान को एक कदम आगे बढ़ाना
प्रतिस्थापित करता है $(p, w)$ द्वारा $(p + 1, \rho w)$, कहाँ $ p $ रिक्त है
सेल और $\rho $ \omterm{def:b2:structures:generated}{चक्रीय} रूप से शब्द को एक-एक करके घुमाता है; पिछड़ा
चाल उलटी है. \emph{\omterm{def:b2:structures:generated}{चक्रीय}} टाइल्स का क्रम (
घूर्णन तक शब्द) इस प्रकार अपरिवर्तनीय है। का पहुंच योग्य वर्ग
$(p, w)$ मानचित्र की कक्षा है $ g \colon (p, w) \mapsto (p+1,
\rho w)$, का एक तत्व \omterm{def:b2:structures:generated}{आदेश} $\operatorname{lcm}(n, n-1) =
n(n-1)$ दोनों के उत्पाद में \omterm{def:b2:structures:generated}{चक्रीय} समूह (का अनुवाद
$\Z/n\Z $ और के घूर्णन $ n-1$ शब्द स्थिति), एलसीएम
किया जा रहा है $ n(n-1)$ क्योंकि $\gcd(n, n-1) = 1$: प्रत्येक वर्ग के पास है
बिलकुल $ n(n-1)$ विन्यास, सभी एक ही हार के साथ।
कक्षाएं: $ n!\,/\,\bigl(n(n-1)\bigr) = (n-2)!$. के लिए $ n \geq 5$,
$(n-2)! > 2$: समता अपरिवर्तनीय (सर्वोत्तम दो वर्ग) है
लगभग सभी रुकावटों के प्रति अंधा; का धन $4
\times 4$ बोर्ड --- जहां समता है \emph{केवल} रुकावट
--- वास्तव में एक ज्यामितीय तथ्य है, औपचारिक नहीं।

\textbf{24.} दोनों ट्रे में टाइलें पूरी तरह से उलटे क्रम में हैं,
तो $ N = \binom{15}{2} = 105$ दोनों ही मामलों में (टाइल्स की प्रत्येक जोड़ी
उलटा है)। \emph{खाली घर:} $ r = 1$, $ N + r = 106$ सम:
न सुलझाया जा सकने वाला \emph{कक्ष में रिक्त $1$:} रिक्त स्थान शीर्ष पर है
पंक्ति, $ r = 4$, $ N + r = 109$ विषम: हल करने योग्य। दो ट्रे जो भिन्न हैं
केवल तभी जब छेद दीवार के विपरीत दिशा में पड़ता है।

\textbf{25.} \emph{रूपवाद संपत्ति:} यह ``एक चाल ='' को रूपांतरित करता है
एक \omterm{def:b2:structures:sn}{स्थानांतरण}'' में ``एक चाल = एक संकेत फ्लिप'' (प्रश्न
2, 4), बनाना $ I $ चाल दर चाल गणना योग्य। \emph{लैग्रेंज:} यह
मजबूर $6 \mid \abs{\langle s, t\rangle}$ लेम्मा ए और आकार में
क्रम में सहसमुच्चय-$6$ बहिष्करण (प्रश्न 14).
\emph{पीढ़ी द्वारा $3$-चक्र:} इसने `` परिवर्तित कर दिया $ H$ शामिल है
पर्याप्त $3$-चक्र'' को `` में $ H$ सभी शामिल हैं $ A_{15}$''
(प्रश्न 16). \emph{संयुग्मन:} इसने पन्द्रह का निर्माण किया
लगातार $3$-एकल से साइकिल $2 \times 2$ भ्रमण
भव्य दौरे द्वारा ले जाया गया (प्रश्न 12-13), और
$3$-चक्र $(a\ b\ w)$ लेम्मा बी में \emph{मेटा-सिद्धांत:} एक
अपरिवर्तनीय असंभवता साबित करता है, एक स्पष्ट निर्माण साबित करता है
संभावना, और एक समस्या पूरी तरह से तभी हल होती है जब ये दोनों हों
सीमाएँ मिलती हैं --- यहाँ, एक आधे पर।
