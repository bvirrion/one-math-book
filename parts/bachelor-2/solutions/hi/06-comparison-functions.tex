\section*{अध्याय \ref{ch:b2:comparison} --- कार्यों की तुलना}

\begin{solution}{exo:b2:comparison:1}
$\sqrt{x^2 + x + 1} = x\sqrt{1 + \tfrac1x + \tfrac{1}{x^2}} = x +
\frac12 + \frac38\cdot\frac1x + o\bigl(\frac1x\bigr)$ (द्विपद
विस्तार: $\frac12 u - \frac18 u^2$ साथ $ u = \frac1x +
\frac{1}{x^2}$ देता है $\frac{1}{2x} + \frac{1}{2x^2} -
\frac{1}{8x^2} = \frac{1}{2x} + \frac{3}{8x^2}$, फिर से गुणा करें
$ x $).

$\ln(x^2 + x) - 2\ln x = \ln\bigl(1 + \tfrac1x\bigr) = \frac1x -
\frac{1}{2x^2} + o\bigl(\frac{1}{x^2}\bigr)$.

तीसरा कार्य: प्रत्येक कारक का विस्तार करें,
\[
\frac{x + \sin x}{x - \ln x}
= \Bigl(1 + \frac{\sin x}{x}\Bigr)
\Bigl(1 + \frac{\ln x}{x} + \frac{(\ln x)^2}{x^2} +
O\Bigl(\frac{(\ln x)^3}{x^3}\Bigr)\Bigr).
\]
अंशदान को पैमाने पर क्रमबद्ध करें $+\infty $: $\frac{\ln x}{x}
\gg \frac{1}{x} \geq \bigl|\frac{\sin x}{x}\bigr| \gg \frac{(\ln
x)^2}{x^2}$. अग्रणी के बाद दो पद $1$ इसलिए हैं
$\frac{\ln x}{x}$, फिर परिबद्ध-दोलन पद $\frac{\sin
x}{x}$:
\[
\frac{x + \sin x}{x - \ln x}
= 1 + \frac{\ln x}{x} + \frac{\sin x}{x}
+ O\Bigl(\frac{(\ln x)^2}{x^2}\Bigr).
\]
\end{solution}

\begin{solution}{exo:b2:comparison:2}
$ f(t) = t^\alpha $ ($ t \geq 1$).

$\alpha > -1$: विचलन, और द्वारा
\cref{thm:b2:comparison:seriesintegral}~(2), $\sum_{k\leq n}
k^\alpha = \frac{n^{\alpha+1}}{\alpha+1} + C + o(1)$ अगर $\alpha <
0$ (कहाँ $ f $ घट जाती है); के लिए $\alpha \geq 0$ ($ f $ बढ़ रहा है)
उलटी असमानताओं के साथ समान कोष्ठक देता है $\sum_{k \leq n}
k^\alpha \sim \frac{n^{\alpha + 1}}{\alpha + 1}$.

$\alpha = -1$: $ H_n = \ln n + \gamma + o(1)$
(\cref{ex:b2:comparison:harmonic}).

$\alpha < -1$: अभिसरण, शेषफल के साथ $\sum_{k > n} k^\alpha
\sim \frac{n^{\alpha+1}}{-(\alpha+1)}$ ब्रैकेटिंग द्वारा (1) (दोनों
अभिन्न सीमाएँ उस मान के समतुल्य हैं)।
\end{solution}

\begin{solution}{exo:b2:comparison:3}
होने देना $ v_n = H_n - \ln n - \gamma \to 0$. तब
\[
v_n - v_{n+1} = \ln\frac{n+1}{n} - \frac{1}{n+1}
= \Bigl(\frac1n - \frac{1}{2n^2}\Bigr) - \Bigl(\frac1n -
\frac{1}{n^2}\Bigr) + O\Bigl(\frac{1}{n^3}\Bigr)
= \frac{1}{2n^2} + O\Bigl(\frac{1}{n^3}\Bigr),
\]
का उपयोग करते हुए $\frac{1}{n+1} = \frac1n - \frac{1}{n^2} + O(n^{-3})$. तब से
$ v_n \to 0$, पूंछ को दूरबीन से देखना:
\[
v_n = \sum_{k \geq n} (v_k - v_{k+1})
= \sum_{k\geq n} \Bigl(\frac{1}{2k^2} + O(k^{-3})\Bigr)
= \frac{1}{2n} + O\Bigl(\frac{1}{n^2}\Bigr),
\]
द्वारा \cref{thm:b2:comparison:seriesintegral}~(1) पर लागू किया गया $ t^{-2}$
(शेष $\sim \frac1n $, आधा) और को $ t^{-3}$. इस तरह $ H_n =
\ln n + \gamma + \frac{1}{2n} + o(\frac1n)$.
\end{solution}

\begin{solution}{exo:b2:comparison:4}
तीन बार स्टर्लिंग:
\[
\frac{(3n)!}{(n!)^3}
\sim \frac{\sqrt{6\pi n}\,(3n/\eu)^{3n}}
{(2\pi n)^{3/2}\,(n/\eu)^{3n}}
= \frac{\sqrt{6}\; 27^{\,n}}{2\pi n} \cdot
\frac{1}{\sqrt{2\pi n}}\cdot\sqrt{2\pi n}\;
= \frac{\sqrt3\,27^n}{2\pi n} .
\]
(सावधानी से: $\frac{\sqrt{6\pi n}}{(2\pi n)^{3/2}} =
\frac{\sqrt6}{(2\pi n)\sqrt{2\pi n}}\sqrt{\pi n} =
\frac{\sqrt3}{2\pi n}$.)

$\dfrac{n!}{n^n} \sim \sqrt{2\pi n}\,\eu^{-n}$.

$\sqrt[n]{n!} = \exp\bigl(\frac{\ln n!}{n}\bigr)$ साथ $\ln n! = n\ln
n - n + \frac12\ln(2\pi n) + o(1)$:
\[
\sqrt[n]{n!} = \exp\Bigl(\ln n - 1 + \frac{\ln(2\pi n)}{2n} +
o\Bigl(\frac{\ln n}{n}\Bigr)\Bigr)
= \frac{n}{\eu}\Bigl(1 + \frac{\ln(2\pi n)}{2n} +
o\Bigl(\frac{\ln n}{n}\Bigr)\Bigr).
\]
\end{solution}

\begin{solution}{exo:b2:comparison:5}
$ g(x) = x^n + x - 1$ पर सख्ती बढ़ती है $\intcc{0}{1}$ से $-1$
को $1$: अद्वितीय जड़ $ x_n $. तब से $ x_n^n = 1 - x_n \in
\intoo{0}{1}$: अगर $ x_n \leq c < 1$ फिर, एक क्रम के साथ $ x_n^n
\leq c^n \to 0$, इसलिए $1 - x_n \to 0$: के साथ विरोधाभास $ x_n \leq
c $. इस तरह $ x_n \to 1$.

लिखना $ x_n = 1 - \varepsilon_n $, $\varepsilon_n \to 0^+$. द
समीकरण पढ़ता है $(1 - \varepsilon_n)^n = \varepsilon_n $, यानी
\[
n\ln(1 - \varepsilon_n) = \ln \varepsilon_n
\quad\Longrightarrow\quad
-n\varepsilon_n\bigl(1 + o(1)\bigr) = \ln\varepsilon_n .
\]
इसलिए $ n\varepsilon_n = -\ln\varepsilon_n\,(1 + o(1)) \to +\infty $, और
फिर से लघुगणक लेना: $\ln n + \ln\varepsilon_n =
\ln(-\ln\varepsilon_n) + o(1)$. तब से $\ln(-\ln \varepsilon_n) =
o(\ln(1/\varepsilon_n))$, यह देता है $\ln\varepsilon_n \sim -\ln
n $, कहाँ से $\varepsilon_n = \frac{-\ln\varepsilon_n}{n}(1 + o(1))
\sim \frac{\ln n}{n}$:
\[
x_n = 1 - \frac{\ln n}{n} + o\Bigl(\frac{\ln n}{n}\Bigr) .
\]
\end{solution}

\begin{solution}{exo:b2:comparison:6}
सटीक संबंध: $\cot y_n = x_n = n\pi + \frac\pi2 - y_n $, साथ $ y_n
\sim \frac{1}{n\pi}$ (\cref{ex:b2:comparison:tan}). बढ़ाना $\cot y
= \frac1y - \frac y3 + O(y^3)$:
\[
\frac{1}{y_n} - \frac{y_n}{3} + O(y_n^3) = n\pi + \frac\pi2 - y_n
\quad\Longrightarrow\quad
\frac{1}{y_n} = n\pi + \frac\pi2 + O\Bigl(\frac1n\Bigr),
\]
(शर्तें $-y_n $ और $-\frac{y_n}{3}$ हैं $ O(\frac1n)$). उलटा:
\[
y_n = \frac{1}{n\pi}\cdot\frac{1}{1 + \frac{1}{2n} + O(n^{-2})}
= \frac{1}{n\pi}\Bigl(1 - \frac{1}{2n} + O\Bigl(\frac{1}{n^2}\Bigr)\Bigr)
= \frac{1}{n\pi} - \frac{1}{2n^2\pi} + O\Bigl(\frac{1}{n^3}\Bigr).
\]
इस तरह
\[
x_n = n\pi + \frac{\pi}{2} - y_n
= n\pi + \frac\pi2 - \frac{1}{n\pi} + \frac{1}{2n^2\pi} +
o\Bigl(\frac{1}{n^2}\Bigr).
\]
\end{solution}

\begin{solution}{exo:b2:comparison:7}
दो सबसे बड़े शब्दों को विभाजित करें:
\[
\sum_{k=0}^{n} k! = n! + (n-1)! + \sum_{k \leq n-2} k! ,
\qquad
\sum_{k\leq n-2} k! \leq (n-1)\,(n-2)! = (n-1)! .
\]
इसलिए $1 \leq \frac{1}{n!}\sum k! \leq 1 + \frac{2}{n}$: हद है
$1$. शोधन: $\frac{(n-1)!}{n!} = \frac1n $ और क्रूड बाउंड
$\sum_{k \leq n-2}k! \leq (n-1)!$ उसी प्रकार तेज़ किया जा सकता है:
$\sum_{k\leq n-2} k! = (n-2)!\,(1 + O(\frac1n)) = O\bigl(\frac{n!}{n^2}\bigr)$.
इस तरह
\[
\sum_{k=0}^{n} k! = n!\Bigl(1 + \frac1n + O\Bigl(\frac{1}{n^2}\Bigr)\Bigr).
\]
\end{solution}

\begin{solution}{exo:b2:comparison:8}
$(u_n)$ बढ़ता है; यदि इसे सीमित कर दिया जाए तो यह एकत्रित हो जाएगा $\ell $ साथ
$\ell = \ell + \frac1\ell $: बेतुका। इसलिए $ u_n \to \infty $.

वर्ग: $ u_{n+1}^2 = u_n^2 + 2 + u_n^{-2}$, इसलिए
\[
u_n^2 = u_0^2 + 2n + \sum_{k=0}^{n-1} \frac{1}{u_k^2} .
\]
योग है $ o(n)$ (शर्तें प्रवृत्त होती हैं $0$, सेस\`एआरओ), तो $ u_n^2 \sim 2n $
और $ u_n \sim \sqrt{2n}$.

शोधन: $\frac{1}{u_k^2} \sim \frac{1}{2k}$, तो तुलना से
(\cref{thm:b2:comparison:seriesintegral}, या आंशिक के \omterm{def:b2:nvs:equivalent}{समकक्ष}
सकारात्मक श्रृंखला का योग) $\sum_{k<n} u_k^{-2} \sim \frac12 \ln n $.
इस तरह
\[
u_n^2 = 2n + \frac{\ln n}{2}\,(1 + o(1)) + O(1)
\quad\Longrightarrow\quad
u_n = \sqrt{2n}\sqrt{1 + \frac{\ln n}{4n} + o\Bigl(\frac{\ln
n}{n}\Bigr)}
= \sqrt{2n}\Bigl(1 + \frac{\ln n}{8n} + o\Bigl(\frac{\ln
n}{n}\Bigr)\Bigr).
\]
\end{solution}

\begin{solution}{exo:b2:comparison:9}
कारक $ n $ और सेट करें $ t = \ln n $:
\[
S_n = \frac1n \sum_{k=1}^{n} \frac{1}{1 + t\,\frac kn} .
\]
तय के लिए $ t $, योग एक रीमैन योग है $ u \mapsto \frac{1}{1 +
tu}$ पर $\intcc{0}{1}$; कार्य एकरस है $ u $, तो
रीमैन योग को एक जाल द्वारा स्थानांतरित अभिन्न अंग द्वारा ब्रैकेट किया गया है:
\[
\int_0^1 \frac{\dd u}{1 + tu} - \frac1n
\leq S_n \leq \int_0^1 \frac{\dd u}{1 + tu} + \frac1n
\]
(एक मोनोटोन फ़ंक्शन के रीमैन की तुलना इसके साथ करती है
अभिन्न, प्रत्येक के लिए मान्य $ n $ अपने साथ $ t = \ln n $). अब
$\int_0^1 \frac{\dd u}{1 + tu} = \frac{\ln(1 + t)}{t}$, और
$\frac1n = o\bigl(\frac{\ln t}{t}\bigr)$: इस तरह
\[
S_n = \frac{\ln(1 + \ln n)}{\ln n} + O\Bigl(\frac 1n\Bigr)
\;\sim\; \frac{\ln\ln n}{\ln n} .
\]
\end{solution}

\begin{solution}{exo:b2:comparison:10}
पहचान: $(\ln n)^{\ln n} = \eu^{\ln n\,\ln\ln n} =
\bigl(\eu^{\ln n}\bigr)^{\ln\ln n} = n^{\ln\ln n}$. रैंकिंग:
लघुगणक की तुलना करें. $\ln(n^2) = 2\ln n $; $\ln\bigl((\ln
n)^{\ln n}\bigr) = \ln n\ln\ln n $; $\ln(2^n) = n\ln2$;
$\ln(n!) = n\ln n - n + O(\ln n)$ (स्टर्लिंग, या क्रूडर
ब्रैकेटिंग $\ln n! \sim n\ln n $); $\ln(n^n) = n\ln n $. तब से
$2\ln n = o(\ln n\ln\ln n)$, $\ln n\ln\ln n = o(n)$, $ n\ln 2 =
o(n\ln n - n)$, और $ n \ln n - n \sim n\ln n $ लेकिन $ n! / n^n \to
0$ (लॉग का अंतर है $-n + O(\ln n) \to -\infty $):
\[
n^2 = o\bigl((\ln n)^{\ln n}\bigr),\quad
(\ln n)^{\ln n} = o(2^n),\quad
2^n = o(n!),\quad
n! = o(n^n).
\]
(प्रत्येक चरण के लिए: लघुगणक का अंतर होता है
$+\infty $, इसलिए अनुपात बढ़ता है $0$.)
\end{solution}

\begin{solution}{exo:b2:comparison:11}
विघटित $\frac1{k^2} = \frac1{k(k+1)} + \frac1{k^2(k+1)}$ और
के लिए योग $ k > n $:
\[
\sum_{k>n}\frac1{k^2} = \frac1{n+1} +
\sum_{k>n}\frac{1}{k^2(k+1)} ,
\]
पहला योग टेलीस्कोपिंग बिल्कुल ($\frac1{k(k+1)} = \frac1k -
\frac1{k+1}$). दूसरे के लिए: $\frac{1}{k^2(k+1)} = \frac1{k^3}
+ O\bigl(\frac1{k^4}\bigr)$ (तब से $\frac{1}{k^2(k+1)} -
\frac1{k^3} = \frac{-1}{k^3(k+1)}$), और द्वारा
अभिन्न तुलना $\sum_{k>n}\frac1{k^3} = \frac1{2n^2} +
O\bigl(\frac1{n^3}\bigr)$, $\sum_{k>n}\frac1{k^4} =
O\bigl(\frac1{n^3}\bigr)$. इस तरह
\[
\sum_{k>n}\frac1{k^2}
= \frac1{n+1} + \frac{1}{2n^2} + O\Bigl(\frac1{n^3}\Bigr)
= \frac1n - \frac1{n^2} + \frac{1}{2n^2} +
O\Bigl(\frac1{n^3}\Bigr)
= \frac1n - \frac{1}{2n^2} + O\Bigl(\frac1{n^3}\Bigr),
\]
का उपयोग करते हुए $\frac1{n+1} = \frac1n - \frac1{n^2} +
O\bigl(\frac1{n^3}\bigr)$.
\end{solution}

\begin{solution}{exo:b2:comparison:12}
$(u_n)$ बढ़ता है; यदि इसे परिबद्ध किया जाता तो यह एक में परिवर्तित हो जाता
परिमित $\ell $ साथ $\ell = \ell + \eu^{-\ell}$: असंभव। इसलिए
$ u_n \to \infty $. होने देना $ v_n = \eu^{u_n} \to \infty $: तब
\[
v_{n+1} = \eu^{u_n + \eu^{-u_n}} = v_n\,\eu^{1/v_n}
= v_n\Bigl(1 + \frac1{v_n} + \frac1{2v_n^2} +
O\bigl(v_n^{-3}\bigr)\Bigr)
= v_n + 1 + \frac{1}{2v_n} + O\bigl(v_n^{-2}\bigr).
\]
जोड़ने पर $ v_{k+1} - v_k = 1 + O(1)$ पहले देता है $ v_n = n +
O(n)$, इस तरह $ v_n \geq cn $ अंततः; के साथ पुनः सारांशित करना
$\frac1{2v_k} = O(\frac1k)$ देता है $ v_n = n + O(\ln n)$. एक और
उत्तीर्ण करना: $\frac{1}{2v_k} = \frac{1}{2k}\bigl(1 +
O\bigl(\tfrac{\ln k}k\bigr)\bigr)$, इसलिए
\[
v_n = n + \sum_{k<n}\frac1{2k} + O(1) = n + \frac{\ln n}2 +
O(1).
\]
अंत में $ u_n = \ln v_n = \ln n + \ln\Bigl(1 + \frac{\ln n}{2n} +
O\bigl(\tfrac1n\bigr)\Bigr) = \ln n + \frac{\ln n}{2n} +
O\bigl(\tfrac1n\bigr)$.
\end{solution}

\begin{solution}{pb:b2:comparison:1}
\textbf{1.} दो विस्तारों को घटाने पर: $\sum_i (c_i -
c_i')\varphi_i = o(\varphi_k)$. यदि कुछ गुणांक भिन्न है, तो चलो
$ i_0$ पहले बनें: से विभाजित करना $\varphi_{i_0}$ और उपयोग कर रहे हैं
$\varphi_j = o(\varphi_{i_0})$ के लिए $ j > i_0$ देता है $ c_{i_0} -
c_{i_0}' = o(1)$: शून्य, विरोधाभास. विस्तार के लिए: साथ
$ u = \frac{\ln x}x \to 0$,
\[
\frac{1}{x - \ln x} = \frac1x\cdot\frac{1}{1 - u}
= \frac1x\bigl(1 + u + u^2 + O(u^3)\bigr)
= \frac1x + \frac{\ln x}{x^2} + \frac{(\ln x)^2}{x^3} +
o\Bigl(\frac{(\ln x)^2}{x^3}\Bigr).
\]
नहीं $\frac c{x^2}$ शब्द प्रकट होता है क्योंकि विस्तार एक है
ज्यामितीय श्रृंखला में $ u = \frac{\ln x}{x}$: प्रत्येक पद इस प्रकार है
की अनेक शक्तियाँ $\ln x $ के रूप में $\frac1x $ पहले से परे; द
स्केल पायदान $\frac1{x^2}$ (का गुणांक $(\ln x)^0$) बस है
अनुपस्थित, गुणांक के साथ $0$.

\textbf{2.} $ f(x) = \eu^x + x $ है \omterm{def:b2:metric:continuity}{निरंतर}, सख्ती से
सीमा के साथ बढ़ रहा है $-\infty $ और $+\infty $: एक आक्षेप
$\R \to \R $, इसलिए $ x_n = f^{-1}(n)$ मौजूद है और अद्वितीय है, और
$ x_n \to +\infty $ ($ f^{-1}$ तक बढ़ जाता है $+\infty $). से
$\eu^{x_n} = n - x_n $: $ x_n = \ln(n - x_n) \leq \ln n $, इसलिए
$ x_n/n \to 0$ और $ x_n = \ln n + \ln(1 - x_n/n) = \ln n + o(1)
\sim \ln n $.

\textbf{3.} लिखना $ u_n = x_n/n $. दूसरा पास: $ u_n = \frac{\ln n
+ o(1)}{n}$, इसलिए
\[
x_n = \ln n + \ln(1 - u_n) = \ln n - u_n + O(u_n^2)
= \ln n - \frac{\ln n}{n} + o\Bigl(\frac{\ln n}n\Bigr).
\]
तीसरा पास: अभी $ u_n = \frac{\ln n}{n} - \frac{\ln n}{n^2} +
o\bigl(\frac{\ln n}{n^2}\bigr)$, और $\ln(1 - u_n) = -u_n -
\frac{u_n^2}2 + O(u_n^3)$:
\[
x_n = \ln n - \frac{\ln n}n + \frac{\ln n}{n^2}
- \frac{(\ln n)^2}{2n^2} + o\Bigl(\frac{(\ln
n)^2}{n^2}\Bigr)
= \ln n - \frac{\ln n}{n} - \frac{(\ln n)^2}{2n^2} +
o\Bigl(\frac{(\ln n)^2}{n^2}\Bigr),
\]
शब्द $\frac{\ln n}{n^2}$ में समाहित किया जा रहा है
$ o\bigl(\frac{(\ln n)^2}{n^2}\bigr)$.

\textbf{4.} पर $ n = 1000$: $\ln 1000 \approx 6.90776$ (गलती
$7\cdot10^{-3}$); दो शर्तें: $6.90085$ (गलती
$2\cdot10^{-5}$); तीन शर्तें: $6.90082$ (नीचे त्रुटि
$10^{-5}$), ख़िलाफ़ $ x_{1000} \approx 6.90083$. प्रत्येक पास खरीदता है
मोटे तौर पर अनुमानित कारक $\frac{\ln n}{n}$.

\textbf{5.} दाईं ओर से प्रारंभ करते हुए, भागों द्वारा दो एकीकरण:
के साथ $\frac{\dd}{\dd t}\bigl[\tfrac12t(1-t)\bigr] = \tfrac12 -
t $ और $ t(1-t)$ दोनों सिरों पर लुप्त हो जाना,
\[
\frac12\int_0^1 t(1-t)g''(t)\dd t
= -\int_0^1\Bigl(\frac12 - t\Bigr)g'(t)\dd t
= -\Bigl[\Bigl(\frac12 - t\Bigr)g\Bigr]_0^1 - \int_0^1 g
= \frac{g(0) + g(1)}2 - \int_0^1 g .
\]
पुनर्व्यवस्थित, यह बताई गई पहचान है।

\textbf{6.} वेतन वृद्धि की गणना करें, फिर प्रश्न 5 को लागू करें
$ g(t) = f(n + t)$:
\begin{align*}
E_{n+1} - E_n
&= f(n{+}1) - \int_n^{n+1}\!f - \frac{f(n{+}1) - f(n)}2 \\
&= \frac{f(n) + f(n{+}1)}2 - \int_n^{n+1}\!f
= \frac12\int_0^1 t(1-t)f''(n+t)\dd t .
\end{align*}
तब से $0 \leq t(1-t) \leq \frac14$: $\abs{E_{n+1} - E_n} \leq
\frac18\int_n^{n+1}\abs{f''}$, जिसका योग खत्म हो गया $ n $ द्वारा अभिसरित होता है
परिकल्पना: $(E_n)$ अभिसरण (बिल्कुल योगयोग्य वृद्धि)
कुछ को $ E $, साथ
\[
\abs{E - E_n} \leq \sum_{k\geq n}\abs{E_{k+1} - E_k} \leq
\frac18\int_n^\infty\abs{f''} .
\]

\textbf{7.} $ f(t) = \frac1t $: $ f''(t) = \frac2{t^3}$,
$\int_1^\infty\abs{f''} = 1 < \infty $. प्रश्न 6:
\[
H_n = \ln n + \frac{1 + \frac1n}{2} + E + (E_n - E)
= \ln n + \Bigl(E + \frac12\Bigr) + \frac1{2n} +
\varepsilon_n,
\]
साथ $\abs{\varepsilon_n} = \abs{E_n - E} \leq
\frac18\int_n^\infty\frac{2\dd t}{t^3} = \frac1{8n^2}$.
के साथ तुलना करना $ H_n = \ln n + \gamma + o(1)$
(\cref{ex:b2:comparison:harmonic}) पहचानता है $ E + \frac12 =
\gamma $.

\textbf{8.} प्रश्न 6 के वेतन वृद्धि सूत्र से,
\[
\varepsilon_n = E_n - E = -\sum_{k\geq n}\frac12\int_0^1
t(1-t)\,\frac{2\,\dd t}{(k+t)^3}
= -\sum_{k \geq n}\Bigl(\frac1{k^3}\int_0^1t(1-t)\dd t +
O\Bigl(\frac1{k^4}\Bigr)\Bigr),
\]
का उपयोग करते हुए $\frac{1}{(k+t)^3} = \frac1{k^3} +
O\bigl(\frac1{k^4}\bigr)$ के लिए समान रूप से $ t \in \intcc01$. साथ
$\int_0^1 t(1-t) = \frac16$ और
$\sum_{k\geq n}\frac1{k^3} \sim \frac{1}{2n^2}$
(\cref{thm:b2:comparison:seriesintegral}):
\[
\varepsilon_n = -\frac16\cdot\frac{1}{2n^2} +
o\Bigl(\frac1{n^2}\Bigr) = -\frac{1}{12n^2} +
o\Bigl(\frac{1}{n^2}\Bigr).
\]

\textbf{9.} $ f = \ln $: $ f''(t) = -\frac1{t^2}$, बिल्कुल
अभिन्न. प्रश्न 6 देता है
\[
\ln n! = \int_1^n\ln t\,\dd t + \frac{\ln n}2 + E + O\Bigl(
\frac1{8}\int_n^\infty\frac{\dd t}{t^2}\Bigr)
= \Bigl(n + \frac12\Bigr)\ln n - n + 1 + E +
O\Bigl(\frac1n\Bigr),
\]
इसलिए $ d_n = 1 + E + O\bigl(\frac1n\bigr)$: का अभिसरण $(d_n)$
--- चरण 1 \cref{thm:b2:comparison:stirling} --- प्लस
दर $ O(1/n)$. (स्टर्लिंग की सीमा का मान देता है $ E =
\ln\sqrt{2\pi} - 1$.)

\textbf{10.} $ f(t) = t^{-1/2}$: $ f''(t) = \frac34 t^{-5/2}$,
बिल्कुल अभिन्न. प्रश्न 6:
\[
\sum_{k=1}^n \frac1{\sqrt k}
= 2\sqrt n - 2 + \frac{1 + \frac1{\sqrt n}}2 + E +
O\bigl(n^{-3/2}\bigr)
= 2\sqrt n + c + \frac{1}{2\sqrt n} +
O\bigl(n^{-3/2}\bigr),
\]
साथ $ c = E - \frac32$. पर $ n = 10^4$: $2\sqrt n = 200$, $ c
\approx -1.46035$, $\frac1{2\sqrt n} = 0.005$: भविष्यवाणी की
$198.54465$, सचमुच $\sum_{k\leq10^4}k^{-1/2} =
198.544645\dots $ --- तीन पद, सात अंक।

\textbf{11.} $ t \mapsto t\ln t $ है \omterm{def:b2:metric:continuity}{निरंतर} और सख्ती से
बढ़ता जा रहा है $\intco1\infty $ (व्युत्पन्न $\ln t + 1 \geq 1$),
से $0$ को $+\infty $: एक अनोखा $ x_n $ मौजूद है, और $ x_n \to
\infty $ (अन्य $ x_n\ln x_n $ बंधे रहेंगे)। लेना
लघुगणक में $ x_n\ln x_n = n $: $\ln x_n + \ln\ln x_n = \ln n $;
तब से $\ln\ln x_n = o(\ln x_n)$, से विभाजित करना $\ln x_n $ देता है
$\frac{\ln n}{\ln x_n} \to 1$: $\ln x_n \sim \ln n $.

\textbf{12.} से $ x_n = \frac{n}{\ln x_n}$ और $\ln x_n \sim
\ln n $: $ x_n \sim \frac{n}{\ln n}$. अगला पास: $\ln\ln x_n =
\ln\bigl(\ln n\,(1 + o(1))\bigr) = \ln\ln n + o(1)$, इसलिए $\ln
x_n = \ln n - \ln\ln n + o(1)$ और
\[
x_n = \frac{n}{\ln n - \ln\ln n + o(1)}
= \frac{n}{\ln n}\cdot\frac{1}{1 - \frac{\ln\ln n +
o(1)}{\ln n}}
= \frac{n}{\ln n}\Bigl(1 + \frac{\ln\ln n}{\ln n} +
o\Bigl(\frac{\ln\ln n}{\ln n}\Bigr)\Bigr).
\]

\textbf{13.} पर $ n = 10^6$: $\frac{n}{\ln n} \approx 72\,382$
(द्वारा बंद $18\%$), दो पद देते हैं $\approx 86\,140$ (द्वारा बंद
$1.9\%$), सत्य के विरुद्ध $ x \approx 87\,848$. लाभ प्रति
पास ही एकमात्र कारक है $\frac{\ln\ln n}{\ln n} \approx
\frac{2.63}{13.8} \approx 0.19$: लघुगणकीय पैमाने अभिसरित होते हैं
पागल कर देने वाली धीमी गति के साथ --- जीवन का एक तथ्य जहां भी अभाज्य हों
सम्मिलित।

\textbf{14.} बिलकुल हैं $ n $ अभाज्य $\leq p_n $ (अर्थात्
$ p_1, \dots, p_n $): $\pi(p_n) = n $. अभाज्य संख्या प्रमेय
(स्वीकृत; वर्ष 3 खंड) देता है $ n = \pi(p_n) \sim
\frac{p_n}{\ln p_n}$, यानी\ $ p_n \sim n\ln p_n $: यह है
समीकरण $ x\ln x \approx n $ पीछे की ओर पढ़ें. लघुगणक लेना:
$\ln p_n = \ln n + \ln\ln p_n + o(1)$, और $\ln\ln p_n =
o(\ln p_n)$ बल $\ln p_n \sim \ln n $ जैसा कि प्रश्न 11 में है।
वापस प्रतिस्थापित करना:
\[
p_n \sim n\ln p_n = n\,\ln n\,\frac{\ln p_n}{\ln n} \sim n\ln
n .
\]

\textbf{15.} (ए) ठीक करें $\varepsilon > 0$; के लिए $ k $ बड़ा, $(1 -
\varepsilon)k\ln k \leq p_k \leq (1 + \varepsilon)k\ln k $. द्वारा
वृद्धि के साथ तुलना $ t\ln t $
(\cref{thm:b2:comparison:seriesintegral}-प्रकार ब्रैकेटिंग),
$\sum_{k\leq n}k\ln k = \int_1^n t\ln t\,\dd t + O(n\ln n) =
\frac{n^2\ln n}2 - \frac{n^2}4 + O(n\ln n) \sim
\frac{n^2\ln n}2$. इस तरह $\sum_{k\leq n}p_k = \frac{n^2\ln
n}{2}(1 + O(\varepsilon) + o(1))$ हरएक के लिए $\varepsilon $:
$\sum_{k\leq n}p_k \sim \frac{n^2\ln n}2$. (बी) प्रधान द्वारा
संख्या प्रमेय, तक के पूर्णांकों के बीच $10^{100}$ ए
अनुपात $\sim \frac{1}{\ln 10^{100}} = \frac1{230.26\dots}$
अभाज्य हैं: एक समान रूप से यादृच्छिक $100$-अंकीय पूर्णांक अभाज्य है
के बारे में संभावना के साथ $\frac1{230}$.

\textbf{16.} पर $\intoo{n\pi}{n\pi + \frac\pi2}$, $ g(x) = \tan
x - \frac1x $ है \omterm{def:b2:metric:continuity}{निरंतर} और सख्ती से बढ़ रहा है ($ g' = 1 +
\tan^2x + \frac1{x^2} > 0$), साथ $ g \to -\frac1{n\pi} < 0$ पर
बायां छोर और $ g \to +\infty $ दाईं ओर: बिल्कुल एक
जड़ $ x_n $. तब से $\tan z_n = \tan x_n = \frac1{x_n} \to 0$
साथ $ z_n \in \intoo{0}{\frac\pi2}$: $ z_n \to 0^+$.

\textbf{17.} $\tan z_n \sim z_n $ और $\frac1{x_n} \sim
\frac1{n\pi}$: $ z_n \sim \frac1{n\pi}$.

\textbf{18.} $ z_n = \arctan\frac1{x_n}$ और $\arctan u = u +
O(u^3)$. साथ $ z_n = O(\frac1n)$:
\[
\frac1{x_n} = \frac{1}{n\pi}\cdot\frac1{1 + \frac{z_n}{n\pi}}
= \frac1{n\pi} - \frac{z_n}{n^2\pi^2} +
O\Bigl(\frac1{n^4}\Bigr)
= \frac1{n\pi} + O\Bigl(\frac1{n^3}\Bigr),
\]
इसलिए $ z_n = \frac1{n\pi} + O\bigl(\frac1{n^3}\bigr)$: द
$\frac{c}{n^2}$ पायदान गुणांक वहन करता है $0$, क्योंकि
करने के लिए पहला सुधार $\frac1{x_n}$ स्वयं आकार का है
$\frac{z_n}{n^2} = O(n^{-3})$.

\textbf{19.} डालना $ z_n = \frac1{n\pi} + O(n^{-3})$ में
पिछला प्रदर्शन:
\[
\frac{1}{x_n} = \frac{1}{n\pi} - \frac{1}{n^3\pi^3} +
O\Bigl(\frac1{n^5}\Bigr),
\]
तब $ z_n = \arctan\frac1{x_n} = \frac1{x_n} -
\frac{1}{3}\Bigl(\frac1{x_n}\Bigr)^3 + O\Bigl(\frac1{n^5}\Bigr)
= \frac1{n\pi} - \frac{1}{n^3\pi^3} - \frac{1}{3n^3\pi^3} +
O\Bigl(\frac1{n^5}\Bigr)$:
\[
x_n = n\pi + \frac{1}{n\pi} - \frac{4}{3\pi^3n^3} +
O\Bigl(\frac1{n^5}\Bigr).
\]

\textbf{20.} पर $ n = 3$: एक कार्यकाल $9.53088$, तीन शर्तें
$9.52929$, सच्ची जड़ $9.52933$:त्रुटियाँ $1.5\cdot10^{-3}$ और
$5\cdot10^{-5}$. कंट्रास्ट: के लिए $\tan x = x $ जड़ को अवश्य बनाना चाहिए
$\tan $ विशाल, इसलिए यह गले लगाता है \emph{सही} अंत $ n\pi +
\frac\pi2$ खिड़की से, दूरी पर $\sim\frac1{n\pi}$
स्पर्शोन्मुख से पहले; के लिए $ x\tan x = 1$ जड़ को अवश्य बनाना चाहिए
$\tan $ छोटा, इसलिए यह इसके ठीक पीछे बैठता है \emph{बाएं} अंत $ n\pi $,
दूरी पर $\sim\frac1{n\pi}$ शून्य के बाद. वही तरीका,
दर्पण भूगोल.

\textbf{21.} $\sin u < u $ पर $\intoo0\pi $ और $\sin $ एमएपीएस
$\intoo0\pi $ में $\intoc01 \subseteq \intoo0\pi $: एक के बाद एक
कदम $ u_1 \in \intoc{0}{1}$, तब $(u_n)$ घटता है और है
से नीचे घिरा हुआ है $0$: यह एक निश्चित बिंदु पर एकत्रित होता है $\sin $,
यानी\ को $0$. विस्तार: $\sin u = u(1 - \frac{u^2}6 +
o(u^2))$, इसलिए
\[
\frac{1}{u_{n+1}^2} - \frac1{u_n^2}
= \frac{1}{u_n^2}\Bigl(\bigl(1 - \tfrac{u_n^2}6 +
o(u_n^2)\bigr)^{-2} - 1\Bigr)
= \frac{1}{u_n^2}\Bigl(\frac{u_n^2}{3} + o(u_n^2)\Bigr)
\longrightarrow \frac13 .
\]

\textbf{22.} सेस द्वारा\`एआरओ (वर्ष 1 खंड), का माध्य
वेतन वृद्धि समान सीमा तक परिवर्तित होती है:
\[
\frac{1}{n}\cdot\frac{1}{u_n^2}
= \frac1n\Bigl(\frac1{u_0^2} + \sum_{k=0}^{n-1}
\Bigl(\frac1{u_{k+1}^2} - \frac1{u_k^2}\Bigr)\Bigr)
\longrightarrow \frac13 ,
\]
इसलिए $ u_n^2 \sim \frac3n $ और, सभी शर्तें सकारात्मक हैं, $ u_n
\sim \sqrt{3/n}$.

\textbf{23.} प्रश्न 7 द्वारा, $\abs{H_n - \ln n - \gamma -
\frac1{2n}} \leq \frac1{8n^2}$. पर $ n = 10^6$ यह बाध्य है
$\frac{1}{8\cdot10^{12}} = 1.25\cdot10^{-13}$: तीन की गणना की गई
शब्द मिलियन-टर्म हार्मोनिक योग को तेरह तक पहुंचाते हैं
अंक, पूरी तरह से कठोर त्रुटि प्रमाणपत्र के साथ --- संपूर्ण
स्पष्ट शेषफल के साथ एक स्पर्शोन्मुख सूत्र का बिंदु।

\textbf{24.} (सच्चा: $\ln u_n - \ln v_n = \ln\frac{u_n}{v_n}
\to 0$ जबकि $\ln v_n \to +\infty $, तो लघुगणक का अनुपात
की ओर प्रवृत्त होता है $1$. (बी) गलत: $ u_n = n + 1 \sim v_n = n $, लेकिन
$\eu^{u_n}/\eu^{v_n} = \eu \neq 1$. समतुल्यता सहन करती है
योगात्मक त्रुटियाँ $ o(1)$ प्रतिपादक में, नहीं $ O(1)$. (सी) गलत:
$ f(x) = x + \sin(x^2) \sim g(x) = x $ पर $+\infty $, लेकिन $ f'(x)
= 1 + 2x\cos(x^2)$ जबकि असीमित रूप से दोलन करता है $ g' = 1$:
समतुल्य कार्यों के व्युत्पन्नों की तुलना करने की आवश्यकता नहीं है
सब.

\textbf{25.} का पाश \cref{met:b2:comparison:implicit} भागा
समान रूप से तीन बार: जड़ का स्थानीयकरण करें, कच्चा तेल निकालें
अवधि, इसे अगले ऑर्डर के लिए वापस फ़ीड करें --- चालू $\eu^x + x = n $
(भाग I), चालू $ x\ln x = n $ (भाग III), चालू $ x\tan x = 1$ (भाग
चतुर्थ). ट्रैपेज़ॉइड सुधार श्रृंखला को उन्नत करता है--अभिन्न
``अंतर अभिसरण'' से स्पष्ट रूप से तुलना करें
$\frac{f(1) + f(n)}2$ एक प्रमाणित के साथ शब्द $ O(\int_n^\infty
\abs{f''})$ शेष --- के स्थान पर स्थिरांक और त्रुटि पट्टियाँ
मात्र अभिसरण. अभाज्य संख्याओं का सेतु शुद्ध व्युत्क्रम है: द
अभाज्य संख्या प्रमेय कहता है $\pi(x)\ln x \approx x $, इसलिए $ p_n $,
द्वारा परिभाषित $\pi(p_n) = n $, एक हल करता है $ x\ln x = n $ समीकरण ---
और इसकी स्पर्शोन्मुखता विरासत में मिली है। प्रश्न 24 का नियम (ए).
से प्रत्येक मार्ग को वैध बनाया $ u_n \sim v_n $ को $\ln u_n \sim
\ln v_n $ (प्रश्न 11,14); (बी) की मिथ्याता के कारण हम हैं
कभी भी तुल्यताएँ घातांकित न करें। शिखर सम्मेलन: यूलर-मैकलॉरिन
प्रथम क्रम का सूत्र (प्रश्न 6), और स्पर्शोन्मुख नियम
$ p_n \sim n\ln n $ की $ n$-वाँ अभाज्य (प्रश्न 14)।
\end{solution}
