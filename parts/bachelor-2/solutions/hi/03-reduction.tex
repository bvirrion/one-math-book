\section*{अध्याय \ref{ch:b2:reduction} --- एंडोमोर्फिज्म में कमी}

\begin{solution}{exo:b2:reduction:1}
$A $: $\chi_A = X^2 - 2X - 3 = (X - 3)(X + 1)$. आइगेनवेक्टर: के लिए
$3$: $(1,1)$; के लिए $-1$: $(1,-1)$. इसलिए $ P = \begin{pmatrix} 1 & 1\\ 1
& -1\end{pmatrix}$ देता है $ P^{-1}AP = \operatorname{diag}(3, -1)$.

$ B = J - I $ कहाँ $ J $ ऑल-वन्स मैट्रिक्स है। $ J $ रैंक है $1$ साथ
$ Jv = 3v $ के लिए $ v = (1,1,1)$ और $ Jw = 0$ विमान पर $ x + y + z =
0$: \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} का $ B $ है $\{2, -1\}$ साथ eigenspaces
$\operatorname{Vect}(1,1,1)$ (आयाम $1$) और $\{x + y + z = 0\}$
(आयाम $2$, आधार $(1,-1,0), (1,0,-1)$). $ P $ इन तीनों के साथ
कॉलम देता है $ P^{-1}BP = \operatorname{diag}(2, -1, -1)$.
\end{solution}

\begin{solution}{exo:b2:reduction:2}
\emph{ईजेनस्पेसेस:} $\chi_C = (X-1)^2$, अकेला \omterm{def:b2:reduction:eigen}{eigenvalue} $1$;
$\ker(C - I) = \ker\begin{pmatrix} 0&1\\ 0&0\end{pmatrix}$ है
लाइन $\operatorname{Vect}(e_1)$: आयाम $1 < 2 = m_1$, तो नहीं
\omterm{def:b2:reduction:diag}{विकर्णीय} (\cref{thm:b2:reduction:diagcrit}).

\emph{\omterm{def:b2:reduction:polyu}{न्यूनतम बहुपद}:} $\mu_C $ विभाजित $(X-1)^2$ और $ C \neq I $,
इसलिए $\mu_C = (X-1)^2$: एक दोहरा मूल, इसलिए नहीं \omterm{def:b2:reduction:diag}{विकर्णीय}
(\cref{cor:b2:reduction:minpolycrit}).
\end{solution}

\begin{solution}{exo:b2:reduction:3}
$ X^2 - 5X + 6 = (X-2)(X-3)$: सरल जड़ों से विभाजित, इसलिए $ u $ है
\omterm{def:b2:reduction:diag}{विकर्णीय} (\cref{cor:b2:reduction:minpolycrit}), साथ
$\operatorname{Sp}(u) \subseteq \{2, 3\}$. संभावित स्पेक्ट्रा:
$\{2\}$ ($ u = 2\,\mathrm{id}$), $\{3\}$ ($ u = 3\,\mathrm{id}$), या
$\{2, 3\}$.

शक्तियाँ: तलाश $ u^k = a_k\,\mathrm{id} + b_k\,u $. पर eigenspaces,
यह पढ़ता है $2^k = a_k + 2b_k $ और $3^k = a_k + 3b_k $: सुलझाना, $ b_k
= 3^k - 2^k $, $ a_k = 3\cdot2^k - 2\cdot 3^k $:
\[
u^k = (3\cdot 2^k - 2\cdot 3^k)\,\mathrm{id} + (3^k - 2^k)\, u .
\]
(तीनों स्पेक्ट्रा के लिए मान्य: पहचानें आइगेनवैल्यू-वार हैं।)
\end{solution}

\begin{solution}{exo:b2:reduction:4}
$ u $ \omterm{def:b2:reduction:diag}{विकर्णीय}: $ P = \prod_{\lambda}(X - \lambda)$ ऊपर
\omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} सत्यानाश कर देता है $ u $, विभाजन, सरल जड़ें। तब $ P(u|_F) =
P(u)|_F = 0$: प्रतिबंध एक विभाजित बहुपद द्वारा नष्ट हो जाता है
सरल जड़ों के साथ, इसलिए \omterm{def:b2:reduction:diag}{विकर्णीय}
(\cref{cor:b2:reduction:minpolycrit}).
\end{solution}

\begin{solution}{exo:b2:reduction:5}
$\chi_A = X^2 - X - 1$, जड़ें $\varphi $ और $\psi $ (विशिष्ट):
\omterm{def:b2:reduction:diag}{विकर्णीय}, साथ eigenvectors $(\varphi, 1)$ और $(\psi, 1)$.
पुनरावृत्ति देता है $\begin{pmatrix} F_{n+1}\\ F_n \end{pmatrix} =
A^n \begin{pmatrix}1\\ 0\end{pmatrix}$. विघटित $(1, 0)$ पर
eigenvectors: $(1,0) = \frac{1}{\varphi - \psi}\bigl((\varphi, 1) -
(\psi, 1)\bigr)$ साथ $\varphi - \psi = \sqrt5$. को लागू करने $ A^n $
प्रत्येक eigencomponent को उसके द्वारा गुणा करता है \omterm{def:b2:reduction:eigen}{eigenvalue}'एस $ n $-थ शक्ति;
दूसरा निर्देशांक पढ़ना:
\[
F_n = \frac{\varphi^n - \psi^n}{\sqrt 5} .
\]
(जाँच करना: $ n = 1$ देता है $\frac{\varphi - \psi}{\sqrt5} = 1$.)
\end{solution}

\begin{solution}{exo:b2:reduction:6}
होने देना $ P = \prod_i (X - \mu_i)$ संहार करना $ u^2$, सरल से विभाजित
जड़ें $\mu_i $ (द \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} का $ u^2$). तब से $ u $ उलटा है, $0$
एक नहीं है \omterm{def:b2:reduction:eigen}{eigenvalue} का $ u^2$ ($\det u^2 = (\det u)^2 \neq 0$), तो सभी $\mu_i \neq 0$. तब
\[
Q(X) = \prod_i (X^2 - \mu_i) = \prod_i (X - \sqrt{\mu_i})(X +
\sqrt{\mu_i})
\]
सत्यानाश कर देता है $ u $: $\;Q(u) = \prod_i (u^2 - \mu_i\,\mathrm{id}) =
P(u^2) = 0$. इसकी जड़ें $\pm
\sqrt{\mu_i}$ (जटिल वर्गमूल) जोड़ीवार भिन्न हैं क्योंकि
द $\mu_i $ विशिष्ट और शून्येतर हैं ($\sqrt{\mu_i} = -\sqrt{\mu_j}$
देगा $\mu_i = \mu_j $). विभाजित + सरल जड़ें: $ u $ है
\omterm{def:b2:reduction:diag}{विकर्णीय}.

उलटापन के बिना प्रति उदाहरण: $ u = \begin{pmatrix} 0 & 1\\
0 & 0\end{pmatrix}$: $ u^2 = 0$ है \omterm{def:b2:reduction:diag}{विकर्णीय}, $ u $ क्या नहीं है।
\end{solution}

\begin{solution}{exo:b2:reduction:7}
$ A = 2I + N $ साथ $ N $ स्थानान्तरित करना ($ N e_2 = e_1$, $ Ne_3 = e_2$),
$ N^3 = 0$, $ N^2 = E_{13}$: यह \emph{है}  \omterm{thm:b2:reduction:dunford}{डनफोर्ड अपघटन}
($2I $ विकर्ण, $ N $ निलपोटेंट, वे आवागमन करते हैं; विशिष्टता इसे बनाती है
एक). आने-जाने की शर्तों के साथ द्विपद:
\[
A^k = 2^k I + k 2^{k-1} N + \binom k2 2^{k-2} N^2
= \begin{pmatrix}
2^k & k2^{k-1} & \binom k2 2^{k-2}\\
0 & 2^k & k2^{k-1}\\
0 & 0 & 2^k
\end{pmatrix},
\]
\[
\eu^{tA} = \eu^{2t}\Bigl(I + tN + \frac{t^2}{2}N^2\Bigr)
= \eu^{2t}\begin{pmatrix}
1 & t & t^2/2\\
0 & 1 & t\\
0 & 0 & 1
\end{pmatrix}.
\]
\end{solution}

\begin{solution}{exo:b2:reduction:8}
$ X^k - 1$ सत्यानाश कर देता है $ A $ और बंट जाता है $\C $ साथ $ k $ विशिष्ट
जड़ें $\eu^{2\iu\pi j/k}$: $ A $ है \omterm{def:b2:reduction:diag}{विकर्णीय}
(\cref{cor:b2:reduction:minpolycrit}) और इसके eigenvalues, की जड़ें
$ X^k - 1$, हैं $ k $- एकता की जड़ें.

यदि इसके अलावा $ A $ इकाई के साथ \omterm{def:b2:reduction:diag}{त्रिकोणीय} मैट्रिक्स के समान है
विकर्ण: सभी eigenvalues बराबर $1$, और $ A $, \omterm{def:b2:reduction:diag}{विकर्णीय} के साथ
एकमात्र \omterm{def:b2:reduction:eigen}{eigenvalue} $1$, है $ P\,I\,P^{-1} = I $.
\end{solution}

\begin{solution}{exo:b2:reduction:9}
लिखना $ E = \bigoplus_\lambda E_\lambda(u)$
(\cref{thm:b2:reduction:diagcrit}). प्रत्येक $ E_\lambda(u)$ है
$ v $-स्थिर: के लिए $ x \in E_\lambda $, $ u(v(x)) = v(u(x)) = \lambda
v(x)$. का प्रतिबंध $ v $ को $ E_\lambda(u)$ है \omterm{def:b2:reduction:diag}{विकर्णीय}
(\cref{exo:b2:reduction:4}): का एक आधार चुनें $ E_\lambda(u)$ बना होना $ v $-ईजेनवेक्टर। इन आधारों को समग्र रूप से संयोजित करना $\lambda $
का आधार देता है $ E $ जिनके वेक्टर हैं eigenvectors का \emph{दोनों}
$ u $ (सदस्यता द्वारा $ E_\lambda(u)$) और $ v $ (निर्माण द्वारा)।
\end{solution}

\begin{solution}{exo:b2:reduction:10}
($\Rightarrow $) होने देना $ u $ होना \omterm{def:b2:reduction:diag}{विकर्णीय} और $ F $ स्थिर। तब
$ u|_F $ है \omterm{def:b2:reduction:diag}{विकर्णीय} (\cref{exo:b2:reduction:4}): $ F $ एक है
का आधार eigenvectors, जो प्रत्येक वैश्विक के अंदर विस्तारित होता है \omterm{def:b2:reduction:eigen}{eigenspace}
$ E_\lambda $, के आधार पर $ E_\lambda $ (अपूर्ण आधार प्रमेय
भीतर $ E_\lambda $, के भाग से शुरू होता है $ F $का आधार झूठ बोल रहा है
वहाँ --- ध्यान दें $ F = \bigoplus_\lambda (F \cap E_\lambda)$ तब से
$ u|_F $ है \omterm{def:b2:reduction:diag}{विकर्णीय}). जोड़े गए वैक्टर एक स्थिर विस्तार करते हैं
पूरक (प्रत्येक कुछ में निहित है $ E_\lambda $, तो उनकी अवधि है
$ u $-स्थिर)।

($\Leftarrow $) होने देना $ F = \sum_\lambda E_\lambda(u)$ (एक स्थिर
उपस्थान) और $ G $ एक स्थिर पूरक. अगर $ G \neq \{0\}$: $\chi_{u|_G}$
विभाजित हो जाता है $\C $, इसलिए $ u|_G $ एक है आइजन्वेक्टर $ x \in G $
(\cref{thm:b2:reduction:trigonalization} या सीधे तौर पर अस्तित्व
एक जड़ का); लेकिन हर आइजन्वेक्टर का $ u $ में निहित है $ F $, इसलिए $ x \in F
\cap G = \{0\}$: विरोधाभास। इस तरह $ G = \{0\}$ और $ E = F $: द
eigenspaces भरना $ E $, यानी\ $ u $ है \omterm{def:b2:reduction:diag}{विकर्णीय}.
\end{solution}

\begin{solution}{exo:b2:reduction:11}
त्रिकोणीकरण: $ A = PTP^{-1}$, $ T = \begin{pmatrix} \lambda & c\\ 0 &
\mu\end{pmatrix}$, $\abs\lambda, \abs\mu < 1$. तब $ A^k =
PT^kP^{-1}$, और यह पर्याप्त है $ T^k \to 0$.

\emph{विशिष्ट eigenvalues:} प्रेरण देता है
\[
T^k = \begin{pmatrix}
\lambda^k & c\,\dfrac{\lambda^k - \mu^k}{\lambda - \mu}\\[4pt]
0 & \mu^k
\end{pmatrix},
\]
और प्रत्येक प्रविष्टि की ओर रुझान होता है $0$ ($\abs{\lambda}^k, \abs\mu^k \to 0$).

\emph{बराबर eigenvalues ($\mu = \lambda $):} $ T = \lambda I + cE_{12}$
और $ T^k = \lambda^k I + k\lambda^{k-1}cE_{12}$; प्रविष्टि
$ k\lambda^{k-1} \to 0$ तब से $\abs\lambda < 1$ (ज्यामितीय धड़कन
बहुपद)। दोनों ही मामलों में $ T^k \to 0$ प्रवेशवार, इसलिए $ A^k =
PT^kP^{-1} \to 0$ (मैट्रिक्स गुणन निश्चित द्वारा $ P, P^{-1}$ है
प्रविष्टियों में निरंतर --- उत्पाद की प्रत्येक प्रविष्टि निश्चित है
रैखिक संयोजन)।
\end{solution}

\begin{solution}{exo:b2:reduction:12}
$\ker u $ आयाम है $ n - 1$ (रैंक--शून्यता), इसलिए $0$ एक
\omterm{def:b2:reduction:eigen}{eigenvalue} का \omterm{def:b2:reduction:charpoly}{ज्यामितीय बहुलता} $ n - 1$, और $\chi_u $ है
से विभाज्य $ X^{n-1}$
(\cref{def:b2:reduction:charpoly}: ज्यामितीय $\leq $ बीजगणितीय)।
लिखना $\chi_u = X^{n-1}(X - \alpha)$; का गुणांक
$ X^{n-1}$ प्राणी $-\operatorname{tr} u $, हम पाते हैं $\alpha =
\operatorname{tr} u $: $\chi_u = X^{n-1}(X - \operatorname{tr}
u)$.

अगर $\operatorname{tr} u \neq 0$: द \omterm{def:b2:reduction:eigen}{eigenvalue}
$\operatorname{tr} u $ की जड़ है $\chi_u $, तो यह एक वहन करता है
आइजन्वेक्टर; eigenspaces के लिए $0$ और $\operatorname{tr} u $
आयाम हैं $ n - 1$ और $\geq 1$, संक्षेप में $\geq n $: वे
भरें $ E $, और $ u $ है \omterm{def:b2:reduction:diag}{विकर्णीय}
(\cref{thm:b2:reduction:diagcrit}). अगर $\operatorname{tr} u = 0$:
द्वारा \cref{exo:b2:linalg:5}, $ u^2 = (\operatorname{tr} u)u = 0$
साथ $ u \neq 0$: $ u $ एक शून्येतर शून्यशक्तिमान है, और ए \omterm{def:b2:reduction:diag}{विकर्णीय}
निलपोटेंट शून्य है (\cref{prop:b2:reduction:nilpotent}): नहीं
\omterm{def:b2:reduction:diag}{विकर्णीय}.
\end{solution}

\begin{solution}{pb:b2:reduction:1}
\textbf{1.} पहला $ k - 1$ के निर्देशांक $ Cv_n $ हैं $ u_{n+1},
\dots, u_{n+k-1}$ (सुपरडायगोनल शिफ्ट्स), और आखिरी है
$ a_0 u_n + \dots + a_{k-1}u_{n+k-1}$. इसलिए $ v_{n+1} = Cv_n $ धारण करता है
सबके लिए $ n $ यदि अंतिम निर्देशांक सभी के लिए मेल खाते हैं $ n $, यानी\ आईएफएफ
$(\mathcal R)$ धारण करता है. पुनरावृत्त, $ v_n = C^nv_0$.

\textbf{2.} बढ़ाना $ D_k(X) = \det(XI_k - C)$ पहले के साथ
कॉलम: दो गैर-शून्य प्रविष्टियाँ हैं $ X $ (पद $(1,1)$) और
$-a_0$ (पद $(k,1)$). पहला माइनर है $ D_{k-1}$-आकार का
गुणांकों के लिए $ a_1, \dots, a_{k-1}$; दूसरा नाबालिग है
विकर्ण के साथ ऊपरी \omterm{def:b2:reduction:diag}{त्रिकोणीय} $-1$: सिद्ध $(-1)^{k-1}$,
संकेत के साथ $(-1)^{k+1}$ पद से. प्रेरण चालू $ k $
(आधार $ k = 1$: $ X - a_0$) देता है
\[
D_k(X) = X\bigl(X^{k-1} - a_{k-1}X^{k-2} - \dots - a_1\bigr) -
a_0 = P(X).
\]
के लिए $\mu_C $: तब से $ Q(C^{\mathsf T}) = Q(C)^{\mathsf T}$ किसी के लिए
बहुपद, $ C $ और $ C^{\mathsf T}$ एक ही लें विनाशक,
इसलिए वही है \omterm{def:b2:reduction:polyu}{न्यूनतम बहुपद}. के लिए $ C^{\mathsf T}$: द
कॉलम पढ़ें $ C^{\mathsf T}e_1 = e_2$, \dots, $ C^{\mathsf
T}e_{k-1} = e_k $, इसलिए $(e_1, C^{\mathsf T}e_1, \dots, (C^{\mathsf
T})^{k-1}e_1)$ विहित आधार है: मुफ़्त। एक बहुपद $ Q
\neq 0$ डिग्री का $< k $ फिर है $ Q(C^{\mathsf T})e_1 \neq 0$
(यह आधार वैक्टर का एक गैर-तुच्छ संयोजन है): $\deg\mu \geq
k $. जैसा $\mu \mid \chi = P $ साथ $\deg P = k $: $\mu_C = P $.

\textbf{3.} के लिए $ v = (1, \lambda, \dots, \lambda^{k-1})^{\mathsf
T}$: पंक्तियाँ $1$ को $ k-1$ का $ Cv $ देना $\lambda, \lambda^2, \dots,
\lambda^{k-1}$, यानी\ $\lambda $ पहले बार $ k - 1$ प्रविष्टियाँ
का $ v $; अंतिम पंक्ति देती है $\sum_m a_m\lambda^m = \lambda^k -
P(\lambda) = \lambda^k = \lambda\cdot\lambda^{k-1}$. इसलिए $ Cv =
\lambda v $. इसके विपरीत, समीकरण $(Cx)_i = \lambda x_i $ के लिए
$ i < k $ पढ़ना $ x_{i+1} = \lambda x_i $: कोई आइजन्वेक्टर के लिए आनुपातिक है $ v $ --- प्रत्येक \omterm{def:b2:reduction:eigen}{eigenspace} सटीक आयाम है
$1$. \omterm{def:b2:reduction:diag}{विकर्णीय} यदि \omterm{def:b2:reduction:eigen}{eigenspace} आयामों का योग $ k $
(\cref{thm:b2:reduction:diagcrit}) यदि वहाँ हैं $ k $ विशिष्ट
eigenvalues आईएफएफ $ P $ है $ k $ विशिष्ट जड़ें ( eigenvalues हैं
की जड़ें $\chi_C = P $).

\textbf{4.} प्रत्येक $(\lambda_i^n)_n $ हल करती है $(\mathcal R)$:
$\lambda_i^{n+k} = \lambda_i^n\,\lambda_i^k =
\lambda_i^n\sum_m a_m\lambda_i^m $. उन्मुक्तता: एक लुप्त हो जाना
संयोजन $\sum_i c_i\lambda_i^n = 0$ के लिए $ n = 0, \dots, k-1$
एक वेंडरमोंडे प्रणाली है (\cref{exo:b2:linalg:11}) में $ c_i $:
सभी $ c_i = 0$. समाधान स्थान का आयाम है $ k $ (प्रश्न 6,
जिसका प्रमाण प्राथमिक और स्वतंत्र है): $ k $ निःशुल्क समाधान
एक आधार बनाते हैं, और निर्देशांक अद्वितीय होते हैं।

\textbf{5.} $ P = X^2 - X - 6 = (X - 3)(X + 2)$: सामान्य समाधान
$ u_n = A\,3^n + B(-2)^n $. प्रारंभिक शर्तें: $ A + B = 1$, $3A -
2B = 8$: $ A = 2$, $ B = -1$:
\[
u_n = 2\cdot 3^n - (-2)^n .
\]
(जाँच करना: $ u_2 = u_1 + 6u_0 = 14$ और $2\cdot9 - 4 = 14$.)

\textbf{6.} $ P(S)\bigl((u_n)\bigr)$ अनुक्रम है $ n \mapsto
u_{n+k} - a_{k-1}u_{n+k-1} - \dots - a_0u_n $: यदि यह गायब हो जाता है
$(\mathcal R)$ धारण करता है, इसलिए समाधान सेट है $\ker P(S)$, ए
उपस्थान. वो नक्शा $\ker P(S) \to \C^k $, $ u \mapsto (u_0, \dots,
u_{k-1})$, रैखिक है, इंजेक्टिव (पुनरावृत्ति निर्धारित करती है
$ u_{k}, u_{k+1}, \dots $ पहले से $ k $ मान, प्रेरण द्वारा)
और विशेषण (परिभाषित करें) $ u_n $ किसी भी प्रारंभिक डेटा से पुनरावर्ती रूप से):
आयाम $ k $.

\textbf{7.} का प्रमाण \cref{thm:b2:reduction:kernels} उपयोग करता है
केवल: बेज़आउट पहचान में $\C[X]$, और तथ्य यह है कि
एक निश्चित एंडोमोर्फिज्म में बहुपद आवागमन करते हैं। न ही उल्लेख है
परिवेशीय स्थान का आयाम: लेम्मा शब्दशः रखता है
$ S \in \mathcal{L}(\mathcal{S})$. इस तरह
\[
\ker P(S) = \bigoplus_{i=1}^{r}
\ker\,(S - \lambda_i\,\mathrm{id})^{m_i}.
\]

\textbf{8.} के लिए $ Q \in \C[X]$: $(S -
\lambda)\bigl(Q(n)\lambda^n\bigr)_n $ है $ n $-वाँ पद
$ Q(n{+}1)\lambda^{n+1} - \lambda Q(n)\lambda^n =
\lambda^{n+1}(\Delta Q)(n)$, साथ $\Delta Q = Q(X{+}1) - Q(X)$
डिग्री का $\deg Q - 1$ (प्रमुख शर्तें रद्द करें)। पुनरावृत्त,
$(S - \lambda)^m\bigl(Q(n)\lambda^n\bigr) =
\bigl(\lambda^{n+m}(\Delta^m Q)(n)\bigr)_n $, और $\Delta^m Q = 0$
कब $\deg Q \leq m - 1$: दाहिनी ओर का सेट इसमें समाहित है
गिरी. यह आयाम का एक उपस्थान है $ m $: क्रम
$(n^j\lambda^n)_n $, $0 \leq j < m $, तब से स्वतंत्र हैं
$\sum_j c_j n^j\lambda^n = 0$ सभी के लिए $ n $ बलों (से विभाजित करना)
$\lambda^n \neq 0$) बहुपद $\sum_j c_jX^j $ पर गायब हो जाना
प्रत्येक $ n \in \N $, इसलिए शून्य होना। इसके विपरीत $\dim\ker(S -
\lambda)^m \leq m $: विस्तार $(S - \lambda)^m = \sum_j
\binom mj(-\lambda)^{m-j}S^j $, समीकरण $(S - \lambda)^m u =
0$ क्रम की एक रैखिक पुनरावृत्ति है $ m $ (नेतृत्व गुणांक
$1$), इसलिए $ u $ इसके द्वारा निर्धारित किया जाता है $ u_0, \dots, u_{m-1}$ जैसा कि
प्रश्न 6. आयामों की समानता समाप्त होती है।

\textbf{9.} प्रश्न 7 और 8 को संयोजित करें: प्रत्येक समाधान विघटित हो जाता है
के तत्वों के योग के रूप में विशिष्ट रूप से $\ker(S -
\lambda_i)^{m_i}$, यानी\ $ u_n = \sum_i Q_i(n)\lambda_i^n $ साथ
$\deg Q_i \leq m_i - 1$; $ Q_i $ अद्वितीय हैं क्योंकि
अपघटन प्रत्यक्ष है और, प्रत्येक सारांश के भीतर,
के गुणांक $ Q_i $ आधार में निर्देशांक हैं
$(n^j\lambda_i^n)_j $ (प्रश्न 8). आयामों पर स्वच्छता जांच:
$\sum_i m_i = k $.

\textbf{10.} $ P = X^2 - 4X + 4 = (X - 2)^2$: समाधान $(a +
bn)2^n $. आरंभिक डेटा: $ a = 1$, $2(a + b) = 0$, इसलिए $ b = -1$:
\[
u_n = (1 - n)\,2^n .
\]
जाँच करना: $ u_2 = 4u_1 - 4u_0 = -4$, और $(1 - 2)\cdot4 = -4$.

\textbf{11.} लिखना $ u_n = \lambda_1^n\bigl(c_1 + \sum_{i\geq2}
c_i(\lambda_i/\lambda_1)^n\bigr)$; प्रत्येक अनुपात का मापांक होता है $< 1$,
तो ब्रैकेट की प्रवृत्ति होती है $ c_1 \neq 0$: $ u_n \sim c_1\lambda_1^n $.
विशेष रूप से $ u_n \neq 0$ बड़े के लिए $ n $, और
\[
\frac{u_{n+1}}{u_n} =
\lambda_1\,\frac{c_1 + o(1)}{c_1 + o(1)} \longrightarrow
\lambda_1 .
\]

\textbf{12.} गणना करें:
\[
q(a_{n+1}, b_{n+1}) = (a_n + 2b_n)^2 - 2(a_n + b_n)^2
= -a_n^2 + 2b_n^2 = -q(a_n, b_n).
\]
साथ $ q(a_0, b_0) = 1 - 2 = -1$: $ a_n^2 - 2b_n^2 = (-1)^{n+1}$.
संरचनात्मक रूप से: $ q(a, b) = (a - \sqrt2\,b)(a + \sqrt2\,b)$ और
रेखीय मानचित्र $ M $ कारक को गुणा करता है $ a + \sqrt2 b $ द्वारा $1 +
\sqrt2$ और कारक $ a - \sqrt2 b $ द्वारा $1 - \sqrt2$ (गणना करें:
$ a_{n+1} + \sqrt2 b_{n+1} = (1 + \sqrt2)(a_n + \sqrt2 b_n)$);
उत्पाद को गुणा किया जाता है $(1 + \sqrt2)(1 - \sqrt2) = -1 =
\det M $ हर कदम.

\textbf{13.} तब से $ a_n^2 - 2b_n^2 = (a_n - \sqrt2 b_n)(a_n +
\sqrt2 b_n) = (-1)^{n+1}$,
\[
\Bigl|\frac{a_n}{b_n} - \sqrt2\Bigr|
= \frac{\abs{a_n^2 - 2b_n^2}}{b_n(a_n + \sqrt2 b_n)}
= \frac{1}{b_n(a_n + \sqrt2 b_n)} \leq \frac1{2b_n^2},
\]
का उपयोग करते हुए $ a_n \geq b_n \geq 1$ (प्रेरण: दोनों बढ़ते हैं) तो $ a_n +
\sqrt2 b_n \geq (1 + \sqrt2)b_n \geq 2b_n $. eigenvalues का $ M $:
$\chi_M = X^2 - 2X - 1$, जड़ें $1 \pm \sqrt2$; तब से $(a_0, b_0)$
प्रमुख पर एक गैर-शून्य घटक है आइजन्वेक्टर (सभी प्रविष्टियाँ
सकारात्मक), $ b_n \sim c(1 + \sqrt2)^n $ साथ $ c > 0$ (प्रश्न
11). इसलिए त्रुटि है $\asymp (1 + \sqrt2)^{-2n} = (3 +
2\sqrt2)^{-n}$: अनुपात के साथ ज्यामितीय क्षय $1/(3 + 2\sqrt2) = 3
- 2\sqrt2 \approx 0.172$.

\textbf{14.} (ए) प्रश्न 9 से: $\abs{u_n} \leq \sum_i
\abs{Q_i(n)}\abs{\lambda_i}^n \leq \bigl(\sum_i
\abs{Q_i(n)}\bigr)\rho^n $, और प्रत्येक $\abs{Q_i(n)} \leq C_i
n^{m_i - 1} \leq C_i n^{m-1}$ के लिए $ n \geq 1$: स्थिरांकों का योग करें.
(बी) चलो $\rho' = \max_{i \geq 2}\abs{\lambda_i} <
\abs{\lambda_1}$ और $ d = \deg Q_1$, नेतृत्व गुणांक $ c
\neq 0$. तब $ u_n = Q_1(n)\lambda_1^n + R_n $ साथ $\abs{R_n}
\leq Cn^{m-1}\rho'^n $, और
\[
\frac{R_n}{Q_1(n)\lambda_1^n} = O\Bigl(n^{m-1-d}
\bigl(\rho'/\abs{\lambda_1}\bigr)^n\Bigr) \longrightarrow 0
\]
(ज्यामितीय बहुपद को हराता है)। इसलिए
\[
u_n \sim Q_1(n)\,\lambda_1^n \sim c\,n^d\lambda_1^n,
\qquad
\frac{u_{n+1}}{u_n} \longrightarrow \lambda_1
\quad\text{(since } Q_1(n{+}1)/Q_1(n) \to 1\text{)}.
\]
प्रश्न 10 पर जाँच करें: के लिए $ u_n = (1-n)2^n $ अनुपात है
\[
\frac{(-n)2^{n+1}}{(1-n)2^n} = 2\,\frac{-n}{1-n}
\longrightarrow 2 = \lambda_1 .
\]

\textbf{15.} प्रेरण चालू $ n $. के लिए $ n = 1$, $ A_{ij}$ मायने रखता है
लंबाई की सैर $1$. कदम: लम्बाई की सैर $ n + 1$ से $ i $ को
$ j $ लम्बी सैर है $ n $ से $ i $ किसी शिखर तक $\ell $
उसके बाद एक बढ़त $\ell j $:
\[
\#\{\text{walks}\} = \sum_{\ell} (A^n)_{i\ell}A_{\ell j} =
(A^{n+1})_{ij}.
\]

\textbf{16.} $ J = 3\Pi $ कहाँ $\Pi = J/3$ पर प्रक्षेपण है
$\operatorname{Vect}(1,1,1)$ विमान के साथ $ x + y + z = 0$
($\Pi^2 = \Pi $ तब से $ J^2 = 3J $). तब $ A = J - I = 2\Pi - (I -
\Pi)$, और तबसे $\Pi $ और $ I - \Pi $ पूरक हैं
अनुमान,
\[
A^n = 2^n\,\Pi + (-1)^n (I - \Pi),
\qquad\text{i.e.}\qquad
(A^n)_{ij} = \frac{2^n}3 + (-1)^n\Bigl(\delta_{ij} -
\frac13\Bigr),
\]
जो दो प्रदर्शित सूत्र देता है। पर $ n = 2$: विकर्ण
$(4 + 2)/3 = 2$ (चलता है $ i \to \ell \to i $ दो पड़ोसियों के लिए
$\ell $); बंद विकर्ण $(4 - 1)/3 = 1$ (एकल चलना $ i \to
\ell \to j $ तीसरे शीर्ष के माध्यम से)।

\textbf{17.} होने देना $ w_n^{(0)}, w_n^{(1)}$ के स्वीकार्य शब्द गिनें
लंबाई $ n $ में समाप्त हो रहा है $0$, सम्मान।\ $1$. एक पत्र संलग्न करना: ए $0$
किसी भी चीज़ का अनुसरण कर सकते हैं, a $1$ केवल एक $0$:
\[
\begin{pmatrix} w_{n+1}^{(0)}\\ w_{n+1}^{(1)}\end{pmatrix}
= \begin{pmatrix} 1 & 1\\ 1 & 0\end{pmatrix}
\begin{pmatrix} w_n^{(0)}\\ w_n^{(1)}\end{pmatrix}.
\]
संक्षेप में, $ w_{n+2} = w_{n+1} + w_n $ (या: पहले पर शर्त
पत्र). साथ $ w_1 = 2$, $ w_2 = 3$: $ w_n = F_{n+2}$ प्रेरण द्वारा
($ F_3 = 2$, $ F_4 = 3$, वही पुनरावृत्ति)। विकास: की जड़ें
$ X^2 - X - 1$ हैं $\varphi > \abs\psi $
(\cref{exo:b2:reduction:5}), और यह $\varphi $-घटक है
शून्येतर (द $ w_n $ सकारात्मक हैं और $\psi^n \to 0$), तो सवाल
11 देता है $ w_{n+1}/w_n \to \varphi = \frac{1 + \sqrt5}2$.

\textbf{18.} $ A = \begin{psmallmatrix} 0&1&0\\ 1&0&1\\
0&1&0\end{psmallmatrix}$. जाँच करना:
\[
A(1, \pm\sqrt2, 1)^{\mathsf T}
= (\pm\sqrt2, 2, \pm\sqrt2)^{\mathsf T}
= \pm\sqrt2\,(1, \pm\sqrt2, 1)^{\mathsf T},
\qquad
A(1, 0, -1)^{\mathsf T} = 0 :
\]
eigenvalues $\sqrt2, -\sqrt2, 0$ ($= 2\cos\frac\pi4,
2\cos\frac{3\pi}4, 2\cos\frac\pi2$). विघटित $ e_1$ पर
eigenbasis और तीसरा निर्देशांक पढ़ें, या समरूपता का उपयोग करें: साथ
$ v_\pm = (1, \pm\sqrt2, 1)$, $ v_0 = (1, 0, -1)$, एक जाँच करता है $ e_1
= \frac14 v_+ + \frac14 v_- + \frac12 v_0$, अभीतक के लिए तो $ n \geq 1$
\[
(A^n)_{13} = \Bigl(\tfrac14(\sqrt2)^n v_+ +
\tfrac14(-\sqrt2)^n v_- + 0\Bigr)_{\!3}
= \frac{(\sqrt2)^n + (-\sqrt2)^n}{4},
\]
विषम के लिए शून्य $ n $ (द्विपक्षीय ग्राफ़: सिरे सम दूरी पर हैं),
और $2\cdot 2^{n/2}/4 = 2^{n/2 - 1}$ एक जैसे के लिए $ n $. पर $ n = 4$:
$2^{1} = 2$, दो चालों का मिलान $1\,2\,1\,2\,3$ और
$1\,2\,3\,2\,3$.

\textbf{19.} लंबाई के बंद रास्ते $ n $ से $ i $ हैं
$(A^n)_{ii}$; संक्षेप में $ i $ देता है $\operatorname{tr}(A^n)$.
त्रिकोणीकरण $ A $ (ऊपर $\C $), $ A^n $ विकर्ण के साथ \omterm{def:b2:reduction:diag}{त्रिकोणीय} है

त्रिभुज: $\operatorname{tr}(A^n) = 3\,\frac{2^n + 2(-1)^n}3 =
2^n + 2(-1)^n = 2^n + (-1)^n + (-1)^n $: द \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} $\{2, -1,
-1\}$, प्रश्न 16 के अनुरूप।

\textbf{20.} के कॉलम $ W^{\mathsf T}$: $ W^{\mathsf T}e_i =
e_{i-1}$ के लिए $ i \geq 1$ और $ W^{\mathsf T}e_0 = e_{k-1}$;
क्रम में पुनः लेबल करना $ e_0, e_1, \dots $ यह बिलकुल यही है
का साथी मैट्रिक्स $ X^k - 1$ ($ a_0 = 1$, अन्य $ a_m = 0$).
प्रश्न 2: $\chi_{W} = \chi_{W^{\mathsf T}} = X^k - 1 =
\mu_{W}$. जड़ें $\omega^j $ ($ j = 0, \dots, k-1$) हैं $ k $
विशिष्ट $ k $- एकता की जड़ें: $ W $ है \omterm{def:b2:reduction:diag}{विकर्णीय} (प्रश्न
3, या \cref{exo:b2:reduction:8}: $ W^k = I $). आइगेनवेक्टर: $ Wf_j
= \sum_m \omega^{-jm}e_{m+1} = \sum_{m'}\omega^{-j(m'-1)}e_{m'}
= \omega^j f_j $.

\textbf{21.} सर्कुलेटेंट बहुपद हैं $ W $, और बहुपद
एक निश्चित मैट्रिक्स में एक दूसरे के साथ आवागमन करते हैं। प्रत्येक $ f_j $ एक
आइजन्वेक्टर हर शक्ति का: $ W^m f_j = \omega^{jm}f_j $, इसलिए
\[
Cf_j = \sum_m c_m\omega^{jm} f_j = \widehat c(\omega^j)\,f_j :
\]
बुनियाद $(f_0, \dots, f_{k-1})$ (निःशुल्क: वेंडरमोंडे में
विशिष्ट $\omega^{-j}$, \cref{exo:b2:linalg:11}) विकर्णित करता है
प्रत्येक सर्कुलेटर एक बार में, कहा गया के साथ eigenvalues.

\textbf{22.}  सिद्ध का उत्पाद है eigenvalues
(विकर्ण): $\det C = \prod_{j}\widehat c(\omega^j)$. के लिए $ k
= 3$, $ c_0 = a $, $ c_1 = b $, $ c_2 = c $ और $\omega = j =

\[
\det C = (a + b + c)(a + bj + cj^2)(a + bj^2 + cj^4),
\]
और $ j^4 = j $: बिल्कुल का गुणनखंडन

\textbf{23.} $ M = \frac12(W + W^{-1})$ एक परिचालितकर्ता है ($ W^{-1} =
W^{k-1}$), साथ eigenvalues $\frac12(\omega^j + \omega^{-j}) =
\cos\frac{2\pi j}k $ उसी आधार पर $ f_j $. निर्देशांक: लिखें
$ x^{(0)} = \sum_j \alpha_j f_j $. के निर्देशांक $ f_j $ का योग
$\sum_m \omega^{-jm}$, जो है $ k $ के लिए $ j = 0$ और $0$
अन्यथा (अनुपात के साथ ज्यामितीय योग $\omega^{-j} \neq 1$).
के निर्देशांकों का योग करना $ x^{(0)}$: $\sum_m x^{(0)}_m =
\alpha_0\,k $, इसलिए $\alpha_0 = \frac1k\sum_m x^{(0)}_m $, मतलब।

\textbf{24.} $ x^{(n)} = M^nx^{(0)} = \sum_j
\alpha_j\cos^n\bigl(\tfrac{2\pi j}k\bigr)f_j $. विषम के लिए $ k $,
$\abs{\cos(2\pi j/k)} < 1$ हरएक के लिए $ j \neq 0$ (कोण है
कभी नहीं $0$ या $\pi $), तो सभी शर्तें लेकिन $ j = 0$ प्रवृत्त $0$:
$ x^{(n)} \to \alpha_0 f_0$, स्थिर वेक्टर के बराबर
माध्य --- एक विषम रिंग पर औसत बराबर होता है। के लिए $ k = 4$ 
eigenvalues हैं $1, 0, -1, 0$: द $ j = 2$ अवधि
$\alpha_2(-1)^nf_2$ साथ $ f_2 = (1, -1, 1, -1)^{\mathsf T}$
सदैव दोलन करता रहता है। बाधा प्रत्यावर्ती माध्य है:
के निर्देशांकों को गुणा करना $ x^{(0)}$ द्वारा $(-1)^m $ और
योग, वही ज्यामितीय-योग गणना देता है $\sum_m
(-1)^mx^{(0)}_m = 4\alpha_2$: प्रक्रिया अभिसरण होती है यदि
$ x^{(0)}_0 - x^{(0)}_1 + x^{(0)}_2 - x^{(0)}_3 = 0$, और फिर
माध्य में परिवर्तित हो जाता है।

\textbf{25.} साथी मैट्रिक्स एक अदिश पुनरावृत्ति को परिवर्तित करता है
क्रम का $ k $ प्रथम-क्रम वेक्टर पुनरावृत्ति में, ताकि
बंद सूत्र कथन बन जाते हैं $ C^n $ --- कमी की
घरेलू मैदान (प्रश्न 1-5). \omterm{thm:b2:reduction:kernels}{कर्नेल अपघटन लेम्मा} है
शुद्ध बहुपद बीजगणित (बेज़ाउट प्लस कम्यूटेशन), इसलिए यह विभाजित हो जाता है
$\ker P(S)$ चाहे $\mathcal{S}$ अनंत-आयामी है
(प्रश्न 7--9). प्रमुख eigenvalues विकास को नियंत्रित करें क्योंकि
इसके बाद अन्य सभी योगदान ज्यामितीय रूप से नगण्य हैं
सामान्यीकरण --- यही कारण है कि पेल त्रुटि कम हो जाती है
प्रमुख मूल का वर्ग (प्रश्न 11--14)। की शक्तियां
आसन्न मैट्रिक्स गिनती चलती है क्योंकि मैट्रिक्स गुणन योग होता है
मध्यवर्ती शीर्षों पर, इसलिए स्पेक्ट्रा गिनती बंद चलती है
(प्रश्न 15--19). कम्यूटिंग मैट्रिसेस एक ईजेनबेसिस साझा करते हैं, और
एक फूरियर आधार फिर संपूर्ण परिपत्र बीजगणित को विकर्णित करता है
एक ही झटके में (प्रश्न 20--24). शिखर सम्मेलन: मौलिक
रैखिक पुनरावृत्तियों का प्रमेय (प्रश्न 9); और गैर-नकारात्मक के लिए
मैट्रिक्स, प्रमुख जड़ों की तरह कारण $\varphi $ या $1 +
\sqrt2$ स्वतः ही वास्तविक, सकारात्मक और सरल है
पेरोन--फ्रोबेनियस प्रमेय, वर्ष 3 खंड में सिद्ध हुआ।
\end{solution}
