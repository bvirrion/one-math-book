\begin{solution}{exo:b2:funcseq:1}
$f_n(x) = \frac{x}{1 + nx}$: \omterm{def:b2:funcseq:def}{बिंदुवार} आप LIMIT $0$ पर $\intcc{0}{1}$.
\omterm{def:b2:funcseq:def}{समान रूप से}: $ f_n $ पर बढ़ता है $\intcc{0}{1}$ (व्युत्पन्न
$\frac{1}{(1+nx)^2} > 0$), इसलिए $\norm{f_n}_\infty = f_n(1) =
\frac{1}{1+n} \to 0$: \emph{वर्दी} पर $\intcc{0}{1}$.

$ g_n(x) = nx\,\eu^{-nx^2}$: \omterm{def:b2:funcseq:def}{बिंदुवार} आप LIMIT $0$ (घातीय
धड़कता है)। समर्थन: $ g_n' = n\eu^{-nx^2}(1 - 2nx^2)$ पर गायब हो जाता है $ x_n =
\frac{1}{\sqrt{2n}}$, कहाँ $ g_n(x_n) = \sqrt{\frac n2}\,\eu^{-1/2}
\to \infty $: पर एक समान नहीं $\intcc{0}{1}$ --- लेकिन एक समान पर
$\intco{\delta}{1}$, वहां से $ g_n(x) \leq n\,\eu^{-n\delta^2}
\to 0$.

$ h_n(x) = x^n(1 - x^n)$: \omterm{def:b2:funcseq:def}{बिंदुवार} आप LIMIT $0$ पर $\intcc{0}{1}$
(दोनों कारक; पर $ x = 1$, $ h_n = 0$). सुपर: साथ $ u = x^n \in
\intcc{0}{1}$, $ u(1-u) \leq \frac14$ पर प्राप्त किया $ u = \frac12$,
यानी\ $ x = 2^{-1/n} \in \intoo{0}{1}$: $\norm{h_n}_\infty =
\frac14 \not\to 0$: पर एक समान नहीं $\intcc{0}{1}$; वर्दी पर
$\intcc{0}{a}$ ($\sup \leq a^n \to 0$).
\end{solution}

\begin{solution}{exo:b2:funcseq:4}
तय के लिए $ x > 0$ श्रृंखला बारी-बारी से चल रही है $\frac{1}{n + x}
\downarrow 0$: \omterm{def:b2:funcseq:def}{बिंदुवार} \omterm{def:b2:integration:improper}{अभिसरण}, और शेष बाध्य है
$\abs{R_N(x)} \leq \frac{1}{N + 1 + x} \leq \frac{1}{N+1}$ है
\emph{वर्दी} पर $\intco{\delta}{\infty}$ (वास्तव में चालू
$\intoo{0}{\infty}$): एकसमान \omterm{def:b2:integration:improper}{अभिसरण}. (सामान्य नहीं हैं:
$\norm{\frac{(-1)^n}{n+x}}_\infty = \frac{1}{n + \delta}$,
भिन्न।) निरंतरता से अनुसरण करता है
\cref{thm:b2:funcseq:seriestransfer}.

कार्यात्मक समीकरण: पुनःसूचकांक $ F(x + 1)$ साथ $ m = n + 1$:
\[
F(x+1) = \sum_{n\geq0} \frac{(-1)^n}{n + 1 + x}
= \sum_{m\geq1}\frac{(-1)^{m-1}}{m+x} ,
\]
तो, अलग कर रहा हूँ $ m = 0$ की अवधि $ F(x)$,
\[
F(x) + F(x+1)
= \frac{1}{x} + \sum_{m\geq1}
\frac{(-1)^m + (-1)^{m-1}}{m+x} = \frac1x .
\]
\end{solution}

\begin{solution}{exo:b2:funcseq:6}

\[
\int_0^1 \frac{n f(x)}{1 + n^2x^2}\dd x
= \int_0^n \frac{f(u/n)}{1 + u^2}\,\dd u .
\]

\]

\end{solution}

\begin{solution}{exo:b2:funcseq:7}
के लिए $ f(x) = x^2$, से दूसरे द्विपद पहचान परिवार का उपयोग करें
प्रमाण: $\sum_k k^2 p_k = n(n-1)x^2 + nx $. इस तरह
\[
B_n(f)(x) = \sum_k \frac{k^2}{n^2}\,p_k
= \frac{n(n-1)x^2 + nx}{n^2}
= x^2 + \frac{x(1 - x)}{n} :
\]
$ B_n(f) - f = \frac{x(1-x)}{n}$, सुप का आदर्श $\frac{1}{4n} =
O\bigl(\frac1n\bigr)$, जैसी कि भविष्यवाणी की गई थी।
\end{solution}

\begin{solution}{exo:b2:funcseq:9}
(ए) $\norm{(3/4)^n\varphi(4^n\cdot)}_\infty = \frac12
(3/4)^n $: सामान्य \omterm{def:b2:integration:improper}{अभिसरण}, इसलिए $ W $ है \omterm{def:b2:metric:continuity}{निरंतर}
(\cref{thm:b2:funcseq:seriestransfer}).

(बी) ठीक करें $ x $, $ m $; का चिह्न चुनें $ h_m = \pm\frac12 4^{-m}$ तो
वह खंड $\intcc{4^mx}{4^m(x + h_m)}$ (लंबाई का $\frac12$)
इसमें कोई अर्ध-पूर्णांक नहीं है, जिससे $\varphi $ ढलान का बंधन $\pm1$
इस पर (संभवतः: लंबाई का एक अंतराल $\frac12$ ज्यादा से ज्यादा एक मिलता है
अर्ध-पूर्णांक बिंदु; इससे बचते हुए पक्ष चुनें)।

के लिए $ n > m $: $4^n h_m = \pm\frac12 4^{n-m}$ एक पूर्णांक है, और
$\varphi $ है $1$-आवधिक: द $ n $अंतर का -वाँ पद
गायब हो जाता है.

के लिए $ n = m $: $\abs{\varphi(4^m x + 4^m h_m) - \varphi(4^m x)} =
\abs{4^m h_m} = \frac12$ ($\varphi $ ढलान का बंधन $\pm 1$ पर
खंड), इसलिए यह शब्द सटीक रूप से योगदान देता है $(3/4)^m \cdot
\frac{1/2}{\abs{h_m}} = (3/4)^m\,4^m = 3^m $ भागफल में.

के लिए $ n < m $: द $1$-लिप्सचिट्ज़ $\varphi $ देता है
$\bigl|(3/4)^n\bigl(\varphi(4^nx + 4^nh_m) -
\varphi(4^nx)\bigr)\bigr| \leq (3/4)^n 4^n\abs{h_m} =
3^n\abs{h_m}$: प्रत्येक अधिकतम योगदान देता है $3^n $ भागफल को.

इस तरह
\[
\Bigl|\frac{W(x + h_m) - W(x)}{h_m}\Bigr|
\geq 3^m - \sum_{n=0}^{m-1} 3^n
= 3^m - \frac{3^m - 1}{2} = \frac{3^m + 1}{2}
\longrightarrow \infty .
\]
अगर $ W $ पर भिन्न थे $ x $, हर अंतर भागफल के साथ
$ h_m \to 0$ में जुट जायेंगे $ W'(x)$: विरोधाभास। $ W $ है
\omterm{def:b2:metric:continuity}{निरंतर} हर जगह, कहीं भी भिन्न नहीं।
\end{solution}

\begin{solution}{exo:b2:funcseq:10}
\omterm{def:b2:funcseq:def}{बिंदुवार}: के लिए $ x \in \intco{0}{1}$ श्रृंखला ज्यामितीय है
अनुपात $-x $,
\[
\sum_{n\geq0}(-1)^n x^n(1-x) = \frac{1-x}{1+x},
\]
और कम से $ x = 1$ प्रत्येक पद लुप्त हो जाता है: योग $0 = \frac{1-1}{2}$,
सुसंगत---योग है \omterm{def:b2:metric:continuity}{निरंतर} पर $\intcc{0}{1}$.
एकरूपता: शेष एक ज्यामितीय पूंछ है,
\[
\abs{R_N(x)} = \frac{x^{N+1}(1-x)}{1+x} \leq x^{N+1}(1 - x)
\leq \max_{\intcc01} t^{N+1}(1-t)
= \frac{1}{N+2}\Bigl(\frac{N+1}{N+2}\Bigr)^{\!N+1}
\leq \frac{1}{N+2} \to 0 ,
\]
\omterm{def:b2:funcseq:def}{समान रूप से} में $ x $. सामान्य नहीं हैं: $\norm{u_n}_\infty =
\max x^n(1-x) = \frac{1}{n+1}\bigl(\frac{n}{n+1}\bigr)^n \sim
\frac{1}{\eu\,n}$, और $\sum \frac1{\eu n}$ विचलन अंतर:
$\sum x^n(1-x)$ आंशिक रकम है $1 - x^{N+1}$, \omterm{def:b2:integration:improper}{अभिसरण}
\omterm{def:b2:funcseq:def}{बिंदुवार} तक \emph{टूटनेवाला} $\mathbf 1_{\intco01}$: द्वारा
\cref{thm:b2:funcseq:continuity}, वह \omterm{def:b2:integration:improper}{अभिसरण} नहीं हो सकता
वर्दी पर $\intcc{0}{1}$.
\end{solution}

\begin{solution}{exo:b2:funcseq:11}
सीमा $ f $ है \omterm{def:b2:metric:continuity}{निरंतर} (\cref{thm:b2:funcseq:continuity}).
तब
\[
\abs{f_n(x_n) - f(x)}
\leq \abs{f_n(x_n) - f(x_n)} + \abs{f(x_n) - f(x)}
\leq \norm{f_n - f}_\infty + \abs{f(x_n) - f(x)} ,
\]
और दोनों शर्तें प्रवृत्त होती हैं $0$ (एकसमान \omterm{def:b2:integration:improper}{अभिसरण}; निरंतरता का

\frac{1}{\sqrt{2n}} \to 0$:
\[
g_n(x_n) = \sqrt{\frac n2}\,\eu^{-1/2} \longrightarrow +\infty
\neq 0 = f(0) .
\]
\end{solution}

\begin{solution}{exo:b2:funcseq:12}
\begin{enumerate}
  \item प्रेरण। $ n = 1$ परिभाषा है. सूत्र मान लीजिए
        के लिए $ n $ और सेट करें $ g(x) = \int_0^x \frac{(x-t)^n}{n!}
        f(t)\dd t $. एक एकीकृत के लिए \omterm{def:b2:metric:continuity}{निरंतर} में $(x,t)$ और
        $ C^1$ में $ x $, परिवर्तनीय-सीमा \omterm{thm:b2:integration:continuity}{पैरामीटर अभिन्न}
        के रूप में विभेदित करता है
        \[
        g'(x) = \frac{(x-x)^n}{n!}f(x)
        + \int_0^x \frac{(x-t)^{n-1}}{(n-1)!}f(t)\dd t
        = T^nf(x)
        \]
        (विभाजित करना $ g(x+h) - g(x)$ पट्टी में
        $\int_x^{x+h}$, जो है $ O(h\cdot\sup)$ के साथ
        इंटीग्रैंड गायब हो रहा है $ t = x $ पसंद $ h^n $, और निश्चित
        का अभिन्न अंग $ x $-वृद्धि, माध्य मान द्वारा नियंत्रित
        असमानता और निरंतरता). भी $(T^{n+1}f)' = T^nf $
        (कैलकुलस का मौलिक प्रमेय) और $ g(0) =
        T^{n+1}f(0) = 0$: के दो आदिम $ T^nf $ पर लुप्त हो रहा है
        $0$ संयोग, तो $ T^{n+1}f = g $. बाध्य:
        \[
        \abs{T^nf(x)} \leq \norm f_\infty
        \int_0^x \frac{(x-t)^{n-1}}{(n-1)!}\dd t
        = \norm f_\infty\,\frac{x^n}{n!}
        \leq \frac{\norm f_\infty}{n!} .
        \]
  \item $\sum_n \norm{T^nf}_\infty \leq \eu\,\norm f_\infty $:
        सामान्य, इसलिए एकसमान, \omterm{def:b2:integration:improper}{अभिसरण}; $ S $ है \omterm{def:b2:metric:continuity}{निरंतर}.
        आंशिक रकम संतुष्ट करती है $ S_N = f + T S_{N-1}$, और $ T $
        है $1$-लिप्सचिट्ज़ के लिए $\norm\cdot_\infty $
        ($\abs{Tg(x)} \leq x\norm g_\infty $: देना $ N \to
        \infty $ दोनों पक्षों में देता है $ S = f + TS $.
  \item तय करना $ V(x) = f(x) + \eu^x\int_0^x \eu^{-t}f(t)\dd t $.
        तब $ V - f $ है $ C^1$ साथ $(V-f)'(x) =
        \eu^x\int_0^x\eu^{-t}f + f(x) = V(x)$, और $(TV)' = V $
        साथ $(V - f)(0) = TV(0) = 0$: इसलिए $ V - f = TV $, अर्थात\
        $ V $ समीकरण हल करता है. विशिष्टता: यदि $ S_1, S_2$ हैं
        \omterm{def:b2:metric:continuity}{निरंतर} समाधान, $ D = S_1 - S_2$ संतुष्ट $ D =
        TD $, इस तरह $ D = T^nD $ सभी के लिए $ n $ और $\norm D_\infty
        \leq \frac{\norm D_\infty}{n!} \to 0$: $ D = 0$.
        इसलिए
        \[
        \sum_{n\geq0} T^nf(x) = f(x) +
        \int_0^x \eu^{x-t}f(t)\,\dd t .
        \]
        (श्रृंखला $\sum T^n $ की एक ज्यामितीय श्रृंखला है
        ऑपरेटर्स: रिसॉल्वेंट का पहला स्वाद
        $(\mathrm{Id} - T)^{-1}$, वर्ष 3 में विकसित किया गया
        आयतन.)
\end{enumerate}
\end{solution}

\begin{solution}{pb:b2:funcseq:1}
\textbf{1.} सूत्र से रैखिकता स्पष्ट है। सकारात्मकता:
वजन $ p_k(x) \geq 0$, इसलिए $ f \geq 0$ बल $ B_nf \geq
0$; एकरसता इस प्रकार लागू होती है $ g - f $. अवश्यंभावी: $\pm f \leq
\norm f_\infty $ देता है $\pm B_nf \leq \norm f_\infty B_ne_0 =
\norm f_\infty $. समापन बिंदु: $ p_k(0) = \mathbf 1_{k=0}$ और
$ p_k(1) = \mathbf 1_{k=n}$, इसलिए $ B_nf(0) = f(0)$, $ B_nf(1) =
f(1)$.

\textbf{2.} अंतर $(x+y)^n = \sum_k\binom nk x^ky^{n-k}$
में $ x $, गुणा करके $ x $, और सेट करें $ y = 1 - x $:
\[
nx = \sum_k k\,p_k(x) ;
\]
दो बार, से गुणा करना $ x^2$: $ n(n-1)x^2 = \sum_k k(k-1)p_k(x)$.
इस तरह $ B_ne_0 = 1$ (द्विपद प्रमेय), $ B_ne_1(x) =
\frac{nx}{n} = x $, और
\[
B_ne_2(x) = \frac{\sum_k k^2p_k}{n^2}
= \frac{n(n-1)x^2 + nx}{n^2}
= x^2 + \frac{x(1-x)}{n} .
\]

\textbf{3.} बढ़ाना:
\[
\sum_k\Bigl(\frac kn - x\Bigr)^{\!2} p_k
= B_ne_2(x) - 2x\,B_ne_1(x) + x^2
= \frac{x(1-x)}{n} .
\]
कॉची--बंटवारे के साथ श्वार्ज़ $\abs{k/n - x}\sqrt{p_k}
\cdot \sqrt{p_k}$:
\[
\sum_k\Bigl|\frac kn - x\Bigr| p_k
\leq \Bigl(\sum_k\Bigl(\frac kn - x\Bigr)^2
p_k\Bigr)^{\!1/2}
= \sqrt{\frac{x(1-x)}{n}} \leq \frac{1}{2\sqrt n},
\]
का उपयोग करते हुए $ x(1-x) \leq \frac14$.

\textbf{4.} वज़न $ p_k(x)$ योग के साथ अऋणात्मक हैं $1$
और बैरीसेंटर $\sum_k \frac kn p_k(x) = x $ (प्रश्न 2). द
उत्तल के लिए परिमित जेन्सेन असमानता $ f $ (से प्रेरण
दो-बिंदु परिभाषा, वर्ष 1 खंड) देता है
\[
f(x) = f\Bigl(\sum_k \frac kn\,p_k\Bigr)
\leq \sum_k f\Bigl(\frac kn\Bigr)p_k = B_nf(x) .
\]

\textbf{5.} पर $\{k : \abs{k/n - x} > \delta\}$ एक के पास है
$\bigl(\frac{k/n - x}{\delta}\bigr)^2 > 1$, इसलिए
\[
\sum_{\abs{k/n-x}>\delta} p_k
\leq \frac{1}{\delta^2}\sum_k\Bigl(\frac kn -
x\Bigr)^2p_k
= \frac{x(1-x)}{n\delta^2} \leq \frac{1}{4n\delta^2} .
\]
पढ़ना: $ p_k(x)$ नमूना आवृत्ति का नियम है $ S_n/n $ का
$ n $ पूर्वाग्रह का सिक्का उछाल $ x $; इसका माध्य है $ x $, इसका विचरण
$\frac{x(1-x)}n \to 0$, और डिस्प्ले चेबीशेव का है
असमानता: द्रव्यमान पर ध्यान केंद्रित करता है $ x $, बहुत औसत $ f $
इसके विरुद्ध पुनरुत्पादन करता है $ f(x)$ सीमा में
(\cref{ch:b2:genfun} शब्दावली को आधिकारिक बनाता है)।

\textbf{6.} $\omega \leq 2M < \infty $; एकरसता स्पष्ट है
(एक बड़े सेट पर समर्थन)। हेन: $ f $ \omterm{def:b2:metric:continuity}{निरंतर} एक पर \omterm{def:b2:metric:compact}{कॉम्पैक्ट} है
\omterm{def:b2:funcseq:def}{समान रूप से} \omterm{def:b2:metric:continuity}{निरंतर}, जो सटीक रूप से कहता है $\omega(\delta) \to 0$
जैसा $\delta \to 0^+$. सबएडिटिविटी: यदि $\abs{s - t} \leq
\delta_1 + \delta_2$, बात $ u $ खंड पर $\intcc st $
दूरी पर $\min(\delta_1, \abs{s-t})$ से $ s $ संतुष्ट
$\abs{s-u} \leq \delta_1$, $\abs{u-t} \leq \delta_2$, और
$\abs{f(s)-f(t)} \leq \abs{f(s)-f(u)} + \abs{f(u)-f(t)}$.
पुनरावृत्त, $\omega(p\delta) \leq p\,\omega(\delta)$ के लिए $ p \in
\N^*$; के लिए $\lambda > 0$, साथ $ p = \lceil\lambda\rceil \leq 1
+ \lambda $: $\omega(\lambda\delta) \leq \omega(p\delta) \leq
p\,\omega(\delta) \leq (1+\lambda)\omega(\delta)$.

\textbf{7.} तब से $\sum p_k = 1$:
\[
\abs{B_nf(x) - f(x)}
= \Bigl|\sum_k\bigl(f(k/n) - f(x)\bigr)p_k\Bigr|
\leq \sum_k\omega\bigl(\abs{k/n - x}\bigr)p_k .
\]
प्रत्येक के लिए $ k $, प्रश्न 6 के साथ $\lambda = \abs{k/n - x}/\delta $
देता है $\omega(\abs{k/n-x}) \leq \bigl(1 +
\frac{\abs{k/n-x}}\delta\bigr)\omega(\delta)$; विरुद्ध योग
द $ p_k $ मास्टर अनुमान प्राप्त होता है।

\textbf{8.} प्रश्न 3 की सीमा डालें:
\[
\abs{B_nf(x) - f(x)} \leq \Bigl(1 +
\frac{1}{2\delta\sqrt n}\Bigr)\omega(\delta),
\]
\omterm{def:b2:funcseq:def}{समान रूप से} में $ x $; साथ $\delta = n^{-1/2}$ कोष्ठक है
$\frac32$: $\norm{B_nf - f}_\infty \leq
\frac32\omega(n^{-1/2}) \to 0$ प्रश्न 6 (हेन) द्वारा। यह
डाँटता है \cref{thm:b2:funcseq:weierstrass} एक दर के साथ.

\textbf{9.} $ L $-लिप्सचिट्ज़ का मतलब है $\omega(\delta) \leq L\delta $:
दर $\frac{3L}{2\sqrt n}$. $\alpha $-एच\"पुराने साधन
$\omega(\delta) \leq C\delta^\alpha $: दर $\frac{3C}{2}
n^{-\alpha/2}$.

\textbf{10.} का उपयोग करते हुए $ k\binom{2m}k = 2m\binom{2m-1}{k-1}$:
\[
\sum_{k=m+1}^{2m}k\binom{2m}k
= 2m\sum_{j=m}^{2m-1}\binom{2m-1}{j}
= 2m\cdot 2^{2m-2},
\]
क्योंकि $ j \mapsto 2m-1-j $ बिज़ेक्ट्स $\{m,\dots,2m-1\}$ पर
$\{0,\dots,m-1\}$, तो योग का आधा है $2^{2m-1}$. भी
$\sum_{k=m+1}^{2m}\binom{2m}k = \frac{2^{2m} -
\binom{2m}m}{2}$ (समान समरूपता)। इस तरह
\[
\sum_{k=m+1}^{2m}(k-m)\binom{2m}k
= m\,2^{2m-1} - m\,\frac{2^{2m} - \binom{2m}m}{2}
= \frac m2\binom{2m}m .
\]
प्रतिस्थापन $ k \mapsto 2m-k $ के साथ शर्तों को मैप करता है $ k < m $
उन पर $ k > m $ (समान द्विपद, समान $\abs{k-m}$):
पूर्ण योग एक तरफा योग का दोगुना है, $ m\binom{2m}m $.

\textbf{11.} पर $ x = \frac12$, $ p_k(\tfrac12) =
\binom{2m}k2^{-2m}$ और $ f(\tfrac12) = 0$:
\[
B_{2m}f\Bigl(\frac12\Bigr)
= \sum_k\Bigl|\frac{k}{2m} - \frac12\Bigr|
\binom{2m}k 2^{-2m}
= \frac{2^{-2m}}{2m}\,m\binom{2m}m
= \frac{\binom{2m}m}{2\cdot4^m}
\sim \frac{1}{2\sqrt{\pi m}}
\]
द्वारा \cref{ex:b2:comparison:centralbinomial}. तब से $\omega_f
(\delta) = \delta $ यहाँ (फ़ंक्शन है $1$-लिप्सचिट्ज़ और
सीमा प्राप्त हो गई है), प्रश्न 8 अधिकतम भविष्यवाणी करता है
$\frac32(2m)^{-1/2}$: सच्ची त्रुटि $\frac{1}{2\sqrt{\pi m}}$
\omterm{def:b2:series:def}{बिल्कुल} सही \omterm{def:b2:structures:generated}{आदेश} है $ n^{-1/2}$ --- तक की दर तेज है
स्थिरांक.

\textbf{12.} $ f \leq g $ देता है $ g - f \geq 0$, इसलिए $ L(g-f) \geq
0$, यानी\ $ Lf \leq Lg $. से $-\abs f \leq f \leq \abs f $:
$-L\abs f \leq Lf \leq L\abs f $, यानी\ $\abs{Lf} \leq
L\abs f $.

\textbf{13.} हेन द्वारा चुनें $\delta $ साथ $\abs{f(s)-f(x)}
\leq \varepsilon $ जब कभी भी $\abs{s-x} \leq \delta $. अगर
$\abs{s - x} > \delta $, तब $\frac{(s-x)^2}{\delta^2} > 1$ और
$\abs{f(s)-f(x)} \leq 2M \leq \frac{2M}{\delta^2}(s-x)^2$. में
दोनों मामलों में दावा किया गया बाध्यता कायम है।

\textbf{14.} हल करना $ x $; प्रश्न 13 कहता है, के कार्यों के रूप में $ s $:
\[
-\varepsilon e_0 - \frac{2M}{\delta^2}q_x
\;\leq\; f - f(x)e_0
\;\leq\; \varepsilon e_0 + \frac{2M}{\delta^2}q_x,
\qquad q_x = e_2 - 2x\,e_1 + x^2e_0 .
\]
मोनोटोन रैखिक लागू करें $ L_n $ (प्रश्न 12) और मूल्यांकन करें
$ x $:
\[
\abs{L_nf(x) - f(x)L_ne_0(x)}
\leq \varepsilon L_ne_0(x)
+ \frac{2M}{\delta^2}\bigl(L_ne_2(x) - 2xL_ne_1(x)
+ x^2L_ne_0(x)\bigr) .
\]

\textbf{15.} लिखना $\alpha_j = L_ne_j - e_j $, इसलिए
$\norm{\alpha_j}_\infty \to 0$. तब से $ e_2(x) - 2xe_1(x) +
x^2e_0(x) = 0$:
\[
L_ne_2(x) - 2xL_ne_1(x) + x^2L_ne_0(x)
= \alpha_2(x) - 2x\,\alpha_1(x) + x^2\alpha_0(x),
\]
सुपर का आदर्श अधिक से अधिक $\norm{\alpha_2} + 2\norm{\alpha_1} +
\norm{\alpha_0} \to 0$. भी $ L_ne_0 \to e_0$ \omterm{def:b2:funcseq:def}{समान रूप से}, इसलिए
$ L_ne_0 \leq 2$ बड़े के लिए $ n $, और $\abs{f(x)}\abs{L_ne_0(x) -
1} \leq M\norm{\alpha_0} \to 0$. प्रश्न 14 के साथ संयोजन:
के लिए $ n $ बड़ा, \omterm{def:b2:funcseq:def}{समान रूप से} $ x $,
\[
\abs{L_nf(x) - f(x)} \leq 2\varepsilon +
\frac{2M}{\delta^2}\,o(1) + M\,o(1) \leq 3\varepsilon :
\]
$ L_nf \to f $ \omterm{def:b2:funcseq:def}{समान रूप से} ---कोरोवकिन का प्रमेय.

\textbf{16.} $ B_ne_0 = e_0$ और $ B_ne_1 = e_1$ \omterm{def:b2:series:def}{बिल्कुल}, और
$\norm{B_ne_2 - e_2}_\infty = \max_x\frac{x(1-x)}{n} =
\frac{1}{4n} \to 0$ (प्रश्न 2): कोरोवकिन लागू होता है, और
वीयरस्ट्रैस तीसरी बार अनुसरण करता है।

\textbf{17.} $ I_nf $ में रैखिक है $ f $ (नोडल मान हैं), और आगे
प्रत्येक कोशिका गैर-नकारात्मक नोडल मानों का एफ़िन इंटरपोलेंट है
गैरनकारात्मक: सकारात्मक. $ I_ne_0 = e_0$ और $ I_ne_1 = e_1$
क्योंकि एक एफ़िन फ़ंक्शन अपने स्वयं के इंटरपोलेंट के बराबर होता है। ए पर
सेल $\intcc ab $ ($ b - a = \frac1n $), का एफ़िन इंटरपोलेंट
$ e_2$ है $ L(t) = (a+b)t - ab $, और
\[
L(t) - t^2 = (t-a)(b-t) \in
\intcc{0}{\tfrac{(b-a)^2}{4}} ,
\]
मध्यबिंदु पर अधिकतम के साथ: $\norm{I_ne_2 - e_2}_\infty =
\frac{1}{4n^2} \to 0$. कोरोवकिन: $ I_nf \to f $ \omterm{def:b2:funcseq:def}{समान रूप से} हरएक के लिए \omterm{def:b2:metric:continuity}{निरंतर} $ f $ --- बहुभुज सन्निकटन, संख्या के साथ
आगे अनुमान की जरूरत है.

\textbf{18.} दिया गया $ f $ और $\varepsilon $: वीयरस्ट्रैस प्रदान करता है
एक बहुपद $ P = \sum_{j=0}^d a_jx^j $ साथ $\norm{f -
P}_\infty \leq \frac\varepsilon2$; प्रत्येक को प्रतिस्थापित करना $ a_j $ ए द्वारा
तर्कसंगत $ b_j $ साथ $\abs{a_j - b_j} \leq
\frac{\varepsilon}{2(d+1)}$ सुपर को हिलाता है आदर्श पर
$\intcc{0}{1}$ अधिक से अधिक द्वारा $\frac\varepsilon2$. का सेट
तर्कसंगत-गुणांक बहुपद एक है \omterm{def:b2:structures:countable}{गणनीय} संघ (पर.)
$ d $) का \omterm{def:b2:structures:countable}{गणनीय} समुच्चय, इस तरह \omterm{def:b2:structures:countable}{गणनीय}, और सघन:
$ C(\intcc{0}{1})$ वियोज्य है.

\textbf{19.} रैखिकता से $\int_0^1 fP = 0$ प्रत्येक के लिए
बहुपद $ P $. बहुपद चुनें $ P_n \to f $ \omterm{def:b2:funcseq:def}{समान रूप से}
(वीयरस्ट्रैस):
\[
\Bigl|\int_0^1 f^2\Bigr|
= \Bigl|\int_0^1 f\,(f - P_n)\Bigr|
\leq \norm f_\infty\,\norm{f - P_n}_\infty
\longrightarrow 0 ,
\]
इसलिए $\int_0^1 f^2 = 0$. अगर $ f(x_0) \neq 0$, निरंतरता देता है
$ f^2 \geq c > 0$ एक उपअंतराल पर, लुप्त होने का खंडन
अभिन्न: $ f = 0$. फलस्वरूप दो \omterm{def:b2:metric:continuity}{निरंतर} के साथ कार्य करता है
वही क्षण $\int f t^n $ संयोग.

\textbf{20.} साथ $ p_{n,k}(x) = \binom nk x^k(1-x)^{n-k}$ और
सम्मेलन $ p_{n-1,-1} = p_{n-1,n} = 0$, उत्पाद नियम
और $ k\binom nk = n\binom{n-1}{k-1}$, $(n-k)\binom nk =
n\binom{n-1}{k}$ देना
\[
p_{n,k}'(x) = n\bigl(p_{n-1,k-1}(x) - p_{n-1,k}(x)\bigr) .
\]
विरुद्ध योग $ f(k/n)$ और पहले सूचकांक को स्थानांतरित करना
योग (\omterm{thm:b2:series:abel}{हाबिल सारांश}):
\[
(B_nf)'(x) = n\sum_{j=0}^{n-1}\Bigl(f\Bigl(\frac{j+1}n\Bigr)
- f\Bigl(\frac jn\Bigr)\Bigr)p_{n-1,j}(x) .
\]

\textbf{21.} माध्य मान प्रमेय द्वारा, $ f(\frac{j+1}n) -
f(\frac jn) = \frac1n f'(\xi_j)$ साथ $\xi_j \in
\intoo{j/n}{(j+1)/n}$, इसलिए $(B_nf)'(x) = \sum_j
f'(\xi_j)\,p_{n-1,j}(x)$. नोड $\frac{j}{n-1}$ में भी निहित है
$\intcc{j/n}{(j+1)/n}$ (दोनों असमानताएं कम हो जाती हैं $ j \leq
n-1$), इस तरह $\abs{\xi_j - \frac j{n-1}} \leq \frac1n $ और
\[
\bigl|(B_nf)'(x) - B_{n-1}(f')(x)\bigr|
\leq \sum_j\Bigl|f'(\xi_j) -
f'\Bigl(\frac{j}{n-1}\Bigr)\Bigr| p_{n-1,j}(x)
\leq \omega_{f'}\Bigl(\frac1n\Bigr) \longrightarrow 0
\]
\omterm{def:b2:funcseq:def}{समान रूप से}. तब से $ B_{n-1}(f') \to f'$ \omterm{def:b2:funcseq:def}{समान रूप से}
(\cref{thm:b2:funcseq:weierstrass} पर लागू किया गया \omterm{def:b2:metric:continuity}{निरंतर}
$ f'$), त्रिभुज असमानता देता है $(B_nf)' \to f'$
\omterm{def:b2:funcseq:def}{समान रूप से}. बहुपद $ P_n = B_nf $ फिर जुट जाओ $ f $
में $ C^1$ समझ।

\textbf{22.} टेलर--लैग्रेंज एट $ x $: $ f(\frac kn) - f(x) =
f'(x)(\frac kn - x) + \frac{f''(\xi_k)}2(\frac kn - x)^2$.
विरुद्ध योग $ p_k $, रैखिक पद मर जाता है (प्रश्न 2):
\[
\abs{B_nf(x) - f(x)}
\leq \frac{\norm{f''}_\infty}{2}\sum_k\Bigl(\frac kn -
x\Bigr)^2p_k
= \frac{\norm{f''}_\infty}{2}\cdot\frac{x(1-x)}{n}
\leq \frac{\norm{f''}_\infty}{8n} .
\]

\textbf{23.} के दो और भेद $(x+y)^n $ दे दो
तथ्यात्मक क्षण, साथ $ n_{(j)} = n(n-1)\cdots(n-j+1)$:
\[
\sum_k k_{(j)}\,p_k = n_{(j)}\,x^j \qquad (j = 3, 4),
\]
और $ k^3 = k_{(3)} + 3k_{(2)} + k $, $ k^4 = k_{(4)} + 6k_{(3)} +
7k_{(2)} + k $ उन्हें शक्ति क्षणों में परिवर्तित करें:
\[
\sum_k k^3p_k = n_{(3)}x^3 + 3n_{(2)}x^2 + nx,
\qquad
\sum_k k^4p_k = n_{(4)}x^4 + 6n_{(3)}x^3 + 7n_{(2)}x^2 + nx .
\]
विस्तार $(k - nx)^4$ और संग्रह करना (एक धैर्यवान लेकिन विशुद्ध रूप से)।
चार शक्ति क्षणों के साथ यांत्रिक गणना):
\[
\sum_k(k-nx)^4p_k = nx(1-x)\bigl(1 + 3(n-2)x(1-x)\bigr) .
\]
साथ $ x(1-x) \leq \frac14$: दाहिना भाग अधिक से अधिक है
$\frac n4\bigl(1 + \frac{3n}4\bigr) = \frac{3n^2}{16} +
\frac n4 \leq n^2$ के लिए $ n \geq 1$.

\textbf{24.} टेलर का पीनो रूप $ x $ को परिभाषित करता है $\eta(t) =
\frac{f(t) - f(x) - f'(x)(t-x) - \frac12f''(x)(t-x)^2}
{(t-x)^2}$ के लिए $ t \neq x $, $\eta(x) = 0$: टेलर--लैग्रेंज द्वारा
$\eta(t) = \frac12\bigl(f''(\xi) - f''(x)\bigr)$ कुछ के लिए $\xi $
बीच में $ t $ और $ x $, इसलिए $\abs\eta \leq \norm{f''}_\infty $ और
$\eta(t) \to 0$ जैसा $ t \to x $ (निरंतरता का $ f''$). संक्षेप
के विरुद्ध विस्तार $ p_k $ और प्रश्न 2-3 का उपयोग करते हुए:
\[
n\bigl(B_nf(x) - f(x)\bigr)
= \frac{x(1-x)}{2}f''(x)
+ n\sum_k\eta\Bigl(\frac kn\Bigr)\Bigl(\frac kn -
x\Bigr)^2p_k .
\]
दिया गया $\varepsilon $, चुनना $\delta $ साथ $\abs\eta \leq
\varepsilon $ पर $\abs{t - x}\leq\delta $. निकट भाग: अधिक से अधिक
$\varepsilon\,n\cdot\frac{x(1-x)}n \leq \varepsilon $. सुदूर भाग:
के साथ $ C = \norm{f''}_\infty $ और प्रश्न 23,
\[
n\,C\sum_{\abs{k/n-x}>\delta}\Bigl(\frac kn -
x\Bigr)^2p_k
\leq \frac{nC}{\delta^2}\sum_k\Bigl(\frac kn -
x\Bigr)^4p_k
= \frac{nC}{\delta^2 n^4}\sum_k(k-nx)^4p_k
\leq \frac{C}{\delta^2 n} \longrightarrow 0 .
\]
इस तरह $ n(B_nf(x) - f(x)) \to \frac{x(1-x)}2f''(x)$ ---
वोरोनोव्स्काया का प्रमेय। के लिए $ f = e_2$ यह हर मामले में सटीक है
$ n $ (\cref{exo:b2:funcseq:7}): द $\frac1n $ छत असली है.

\textbf{25.} (i) सकारात्मकता आ गई \omterm{def:b2:funcseq:def}{बिंदुवार} में असमानताएँ
ऑपरेटर असमानताएँ: इसने दिया आदर्श बाध्य, जेन्सेन,
चेबीशेव, और संपूर्ण कोरोवकिन --- केवल रैखिकता ही सिद्ध होती है
यहाँ कुछ भी नहीं. (ii) सघनता हेइन के माध्यम से प्रवेश किया (प्रश्न
6, 13), की सीमा के माध्यम से $ f $, और बहुत के माध्यम से आदर्श
$\norm\cdot_\infty $ परिमित होना. (iii) तीन परीक्षण कार्य
पर्याप्त है क्योंकि सकारात्मकता हर चीज़ को नियंत्रित करने तक सीमित कर देती है
$ L_n $ एकल परिवार पर $(s-x)^2 = e_2 - 2xe_1 + x^2e_0$,
जिसका विस्तार है $ e_0, e_1, e_2$. (iv) बर्नस्टीन \omterm{def:b2:integration:improper}{अभिसरण} करता है
ईमानदार दर पर $\omega(n^{-1/2})$ हरएक के लिए \omterm{def:b2:metric:continuity}{निरंतर} $ f $
(तेज, प्रश्न 11) लेकिन संतृप्त होता है $\frac1n $ चिकनी के लिए $ f$
(प्रश्न 24); फूरियर अध्याय के लिए वही कार्यक्रम चलाता है
Fej के साथ आवधिक कार्य\'एर का कर्नेल --- एक और सकारात्मक
समान गुणों और समान शील वाला संचालक।
\end{solution}
