\section*{अध्याय \ref{ch:b2:metric} --- मीट्रिक रिक्त स्थान की टोपोलॉजी}

\begin{solution}{exo:b2:metric:1}
$\delta $: समरूपता और पृथक्करण स्पष्ट हैं; त्रिभुज असमानता:
$\min(1, u + v) \leq \min(1,u) + \min(1,v)$ के लिए $ u, v \geq 0$ (यदि
या तो मिनट है $1$, दाहिनी ओर है $\geq 1$; अन्यथा यह है $ u + v $).
$ d $: यह का पुलबैक है $\abs{\cdot}$ इंजेक्शन द्वारा $\arctan $:
तीन स्वयंसिद्ध \omterm{def:b2:structures:sn}{स्थानांतरण}।

अभिसरण: के लिए $\delta $, $\delta(x_n, x) \to 0 \iff \abs{x_n - x}
\to 0$ (छोटे मानों के लिए दो दूरियाँ सहमत हैं): समान अभिसरण
क्रम हमेशा की तरह. के लिए $ d $: $ d(x_n, x) \to 0 \iff \arctan x_n \to
\arctan x \iff x_n \to x $ (निरंतरता और सख्त एकरसता
$\arctan $ और प्रासंगिक श्रेणियों पर इसका उलटा): फिर से
सामान्य अभिसरण.

$(\R, d)$ है \emph{नहीं} \omterm{def:b2:metric:complete}{पूरा}: $ x_n = n $ संतुष्ट $ d(x_p, x_q)
= \abs{\arctan p - \arctan q} \to 0$ (दोनों की प्रवृत्ति होती है $\frac\pi2$):
कॉची; लेकिन $(x_n)$ के लिए एकत्रित नहीं होता है $ d $ (इसका $ d $-सीमा होगी
एक सामान्य सीमा हो)। संपूर्णता की एक संपत्ति है
\emph{दूरी}, केवल अभिसारी अनुक्रमों का नहीं।
\end{solution}

\begin{solution}{exo:b2:metric:2}
संमिलित $\Rightarrow $ कॉची: $ d(x_p, x_q) \leq d(x_p, \ell) +
d(\ell, x_q)$. परिबद्ध: परे $ N $, $ d(x_n, \ell) \leq 1$; द
निश्चित रूप से अनेक प्रथम पद भी किसी न किसी दायरे में हैं।

कॉची $+$ अभिसारी परिणाम $ x_{\varphi(n)} \to \ell $: दिया गया
$\varepsilon $, बड़े के लिए $ n $, $ d(x_n, \ell) \leq d(x_n,
x_{\varphi(n)}) + d(x_{\varphi(n)}, \ell) \leq 2\varepsilon $ (द
कॉची द्वारा पहला कार्यकाल, तब से $\varphi(n) \geq n $).

सघन $\Rightarrow $ \omterm{def:b2:metric:complete}{पूरा}: कॉची अनुक्रम में एक अभिसरण होता है
परिणाम (सघनता), इसलिए अभिसरण होता है।
\end{solution}

\begin{solution}{exo:b2:metric:3}
$ d_\infty(f, g) = \sup_{\intcc{0}{1}} \abs{x - x^2} = \frac14$
(अधिकतम $ x - x^2$ पर $ x = \frac12$).

$\overline B(0, 1) = \{f : \sup\abs f \leq 1\}$: द \omterm{def:b2:metric:continuity}{निरंतर}
में मानों के साथ कार्य करता है $\intcc{-1}{1}$.

$\{f : f(0) = 0\}$ की पूर्व छवि है $\{0\}$ नीचे
\emph{मूल्यांकन} $ f \mapsto f(0)$, जो है $1$-लिप्सचिट्ज़
($\abs{f(0) - g(0)} \leq d_\infty(f,g)$), इस तरह \omterm{def:b2:metric:continuity}{निरंतर}: सेट
बंद है (\cref{thm:b2:metric:globalcontinuity}). वैसे ही $\{f :
f(0) > 0\}$ की पूर्व छवि है \omterm{def:b2:metric:topology}{खुला} $\intoo{0}{+\infty}$: \omterm{def:b2:metric:topology}{खुला}.
\end{solution}

\begin{solution}{exo:b2:metric:4}

अंततः स्थिर, इसलिए अभिसरण।

\emph{सघन यदि परिमित:} अगर $ X $ परिमित है, किसी भी अनुक्रम में कुछ न कुछ लगता है
मान अनंत बार (\omterm{def:b2:metric:continuity}{निरंतर} अनुवर्ती)। अगर $ X $ अनंत है,
जोड़ीवार अलग-अलग बिंदुओं के अनुक्रम में सभी पारस्परिक दूरियाँ होती हैं $1$:
कोई कॉची अनुवर्ती नहीं, इसलिए कोई अभिसरण नहीं।

\emph{\omterm{def:b2:metric:connected}{जुड़े हुए} उपसमुच्चय:} सिंगलटन (और $\emptyset $). कोई $ A $
दो बिंदुओं के साथ $ x \neq y $ के रूप में विभाजित होता है $\{x\} \cup (A
\setminus\{x\})$, दोनों \omterm{def:b2:metric:topology}{खुला} में $ A $ (एक असतत का प्रत्येक उपसमुच्चय
अंतरिक्ष है \omterm{def:b2:metric:topology}{खुला} --- त्रिज्या की गेंदें $\frac12$ सिंगलटन हैं)।
\end{solution}

\begin{solution}{exo:b2:metric:5}
समारोह $ g(x) = d(x, f(x))$ है \omterm{def:b2:metric:continuity}{निरंतर} पर \omterm{def:b2:metric:compact}{कॉम्पैक्ट} $ X $
($\abs{g(x) - g(y)} \leq 2d(x,y)$ दो त्रिभुज असमानताओं द्वारा), इसलिए
यह कुछ पर अपना न्यूनतम स्तर प्राप्त कर लेता है $ a $
(\cref{thm:b2:metric:compactprops}). अगर $ f(a) \neq a $:
\[
g\bigl(f(a)\bigr) = d\bigl(f(a), f(f(a))\bigr) < d(a, f(a)) = g(a),
\]
न्यूनतमता का खंडन. इसलिए $ f(a) = a $; हमेशा की तरह विशिष्टता (दो
निश्चित बिंदु $ a \neq b $ देना $ d(a,b) = d(f(a), f(b)) < d(a,b)$).

गैर \omterm{def:b2:metric:compact}{कॉम्पैक्ट} उदाहरण: $ f(x) = x + \frac1x $ पर $\intco{1}{+\infty}$:
$\abs{f(x) - f(y)} = \abs{x - y}\,\abs{1 - \frac{1}{xy}} < \abs{x -
y}$ के लिए $ x \neq y $ (जैसा $ xy > 1$), अभी तक $ f(x) > x $ हर जगह.
\end{solution}

\begin{solution}{exo:b2:metric:6}

\begin{solution}{exo:b2:metric:8}
$ C = \bigcap_n C_n $ जहां प्रत्येक $ C_n $ (संघ का $2^n $ बंद
लंबाई का अंतराल $3^{-n}$) बन्द है: $ C $ बंद और घिरा हुआ है
में $\R $, इस तरह \omterm{def:b2:metric:compact}{कॉम्पैक्ट} (\cref{thm:b2:metric:compactprops}~(2)).

खाली आंतरिक भाग: $ C $ इसमें लंबाई का कोई अंतराल नहीं है $> 3^{-n}$ (यह
अंदर बैठता है $ C_n $, जिनके घटकों की लंबाई वह है), प्रत्येक के लिए
$ n $.

कार्डिनैलिटी: के अंक $ C $ बिल्कुल असली हैं $\sum_{n\geq1}
a_n 3^{-n}$ अंकों के साथ $ a_n \in \{0, 2\}$ (प्रत्येक चरण में,
हटाया गया मध्य तीसरा अंक हटा देता है $1$); वो नक्शा $(a_n) \mapsto
\sum a_n 3^{-n}$ से एक आक्षेप है $\{0,2\}^{\N^*}$ पर $ C $
(इंजेक्शन के रूप में \cref{exo:b2:structures:3}). इसलिए $ C $ है \omterm{def:b2:structures:countable}{सुसज्जित}
को $\{0,1\}^{\N}$: बेशुमार, यद्यपि ``लंबाई शून्य''।
\end{solution}

\begin{solution}{exo:b2:metric:9}
कल्पना करना $ a \notin f(X)$. छवि $ f(X)$ है \omterm{def:b2:metric:compact}{कॉम्पैक्ट} (\omterm{def:b2:metric:continuity}{निरंतर}
छवि), इसलिए बंद; इसलिए
\[
\varepsilon = d\bigl(a, f(X)\bigr)
= \inf_{y \in f(X)} d(a, y) > 0
\]
(ए की अधिकतम सीमा) \omterm{def:b2:metric:continuity}{निरंतर} ए पर कार्य करें \omterm{def:b2:metric:compact}{कॉम्पैक्ट} सेट प्राप्त हो गया है;
यदि ऐसा होता $0$, $ a $ बंद का पालन करेंगे $ f(X)$, इसलिए में
यह).

कक्षा पर विचार करें $ x_n = f^n(a)$ ($ x_0 = a $). के लिए $ p < q $:
\[
d(x_p, x_q) = d\bigl(f^p(a), f^p(f^{q-p}(a))\bigr)
= d\bigl(a, f^{q-p}(a)\bigr) \geq \varepsilon,
\]
(आइसोमेट्री पुनरावृत्त $ p $ समय; और $ f^{q-p}(a) \in f(X)$ तब से $ q - p
\geq 1$). आपसी दूरियों वाला एक क्रम $\geq \varepsilon $ नहीं है
अभिसारी परिणाम --- विरोधाभासी सघनता. इस तरह $ f(X) =
X $.
\end{solution}

\begin{solution}{exo:b2:metric:10}
होने देना $ A, B \in GL_n(\C)$ और $ p(z) = \det\bigl((1-z)A + zB\bigr)$: ए
बहुपद में $ z $ (प्रत्येक प्रविष्टि को सम्मिलित किया गया है $ z $; सिद्ध एक है
प्रविष्टियों में बहुपद)। $ p(0) = \det A \neq 0$: $ p $ नहीं है
समान रूप से शून्य, इसलिए इसकी अनंत रूप से कई जड़ें हैं $ z_1, \dots, z_m $
(किसी के बराबर नहीं) $0$ या $1$: $ p(1) = \det B \neq 0$). विमान $\C $
शून्य से बहुत सारे अंक हैं पथ से जुड़ा हुआ: से एक पथ $0$ को $1$
से बचना $ z_i $ मौजूद है (दूर तक एक बिंदु से होकर एक टूटी हुई रेखा लें
सभी जड़ों से, या एक गोलाकार चाप से; केवल सीमित रूप से अनेक बाधाएँ)।
ऐसे पथ पर $\gamma $, $ t \mapsto (1 - \gamma(t))A +
\gamma(t)B $ एक है \omterm{def:b2:metric:continuity}{निरंतर} पथ \emph{अंदर} $ GL_n(\C)$ से $ A $
को $ B $ (द सिद्ध इस पर कभी गायब नहीं होता). इस तरह $ GL_n(\C)$ है
पथ से जुड़ा हुआ --- अपने असली चचेरे भाई के विपरीत
(\cref{ex:b2:metric:glnr}): जटिल विमान में चलने के लिए जगह है
बाधाओं के आसपास.
\end{solution}

\begin{solution}{exo:b2:metric:11}
\emph{\omterm{def:b2:metric:continuity}{लिप्सचिट्ज़}:} के लिए $ a \in A $, $ d(x, a) \leq d(x, y) + d(y,
a)$; अधिकतम ले लो $ a $: $ d(x, A) \leq d(x, y) + d(y,
A)$, और स्वैप $ x, y $: $\abs{d(x,A) - d(y,A)} \leq d(x,y)$.

\emph{लुप्त होना:} $ d(x, A) = 0$ अगर वहाँ हैं $ a_n \in A $ साथ
$ d(x, a_n) \to 0$ आईएफएफ $ x $ के अंकों की एक सीमा है $ A $ आईएफएफ $ x \in
\overline A $.

\emph{बटन:} असंयुक्त बंद के लिए $ A, B $: हर
$ d(x,A) + d(x,B)$ कभी गायब नहीं होता (यह मजबूर करेगा $ x \in \overline
A \cap \overline B = A \cap B = \emptyset $), इसलिए $\varphi $ ठीक है
परिभाषित, और \omterm{def:b2:metric:continuity}{निरंतर} के भागफल के रूप में \omterm{def:b2:metric:continuity}{निरंतर} कार्य
गैर-लुप्त न होने वाले हर के साथ। $\varphi(x) = 0$ आईएफएफ $ d(x, A) = 0$
आईएफएफ $ x \in A $; $\varphi(x) = 1$ आईएफएफ $ d(x, B) = 0$ आईएफएफ $ x \in B $;
और $0 \leq \varphi \leq 1$ हर जगह.
\end{solution}

\begin{solution}{exo:b2:metric:12}
होने देना $ B(x_0, r_0)$ कोई भी गेंद हो; हमें एक बिंदु मिलता है $\bigcap U_n $
इस में। तब से $ U_1$ घना है और \omterm{def:b2:metric:topology}{खुला}, $ U_1 \cap B(x_0, r_0)$ है
गैररिक्त और \omterm{def:b2:metric:topology}{खुला}: इसमें एक बंद गेंद है $\overline B(x_1,
r_1)$ साथ $0 < r_1 \leq \frac{r_0}2$ (त्रिज्या को सिकोड़ें)।
आगमनात्मक रूप से, $ U_{n+1} \cap B(x_n, r_n)$ गैर-रिक्त है \omterm{def:b2:metric:topology}{खुला}: चुनना
$\overline B(x_{n+1}, r_{n+1}) \subseteq U_{n+1} \cap B(x_n,
r_n)$ साथ $ r_{n+1} \leq \frac{r_n}2$. के लिए $ p, q \geq n $, दोनों
$ x_p, x_q $ रिहायश $ B(x_n, r_n)$ साथ $ r_n \leq 2^{-n}r_0$: द
अनुक्रम कॉची है, और कुछ में परिवर्तित होता है $\ell $ द्वारा
पूर्णता. प्रत्येक के लिए $ n $, अनुक्रम की पूंछ में निहित है
\emph{बंद किया हुआ} गेंद $\overline B(x_{n+1}, r_{n+1}) \subseteq U_{n+1}
\cap B(x_0, r_0)$, इसलिए $\ell \in U_{n+1}$ हरएक के लिए $ n $ और $\ell
\in \overline B(x_1, r_1) \subseteq B(x_0, r_0)$. इस तरह
$\bigcap_n U_n $ हर गेंद से मिलता है: सघन.

\emph{आवेदन पत्र:} अगर $\R = \bigcup_n F_n $ साथ $ F_n $ बंद
खाली आंतरिक भाग के साथ, फिर $ U_n = \R \setminus F_n $ सघन हैं
($\overline{U_n} = \R $ आईएफएफ $ F_n $ खाली आंतरिक भाग है) और \omterm{def:b2:metric:topology}{खुला},
और बेयर एक बिंदु देता है $\bigcap U_n = \R \setminus \bigcup
F_n $: विरोधाभास। विशेष रूप से $\R \neq \bigcup_{x \in D}
\{x\}$ एक के लिए \omterm{def:b2:structures:countable}{गणनीय} $ D $ (सिंगलटन खाली के साथ बंद हैं
आंतरिक): $\R $ बेशुमार है ---
\cref{thm:b2:structures:cantor} दूसरे मार्ग से.
\end{solution}

\begin{solution}{pb:b2:metric:1}

का $ A $ में एकत्रित हो जाता है $ x $; यह कॉची है, इसलिए यह अभिसरण करता है \emph{में}
$ A $; सीमाएँ अद्वितीय हैं, इसलिए $ x \in A $: $ A $ बंद किया हुआ। तब से
$\bigl(C(I), d_\infty\bigr)$ है \omterm{def:b2:metric:complete}{पूरा}
(\cref{thm:b2:metric:rncomplete}), इसके बंद उपसमुच्चय हैं
\omterm{def:b2:metric:complete}{पूरा}.

\textbf{2.} के लिए $ y, z \in C(I)$ और $ t \in I $:
\[
\Bigl|\int_{t_0}^{t}y - \int_{t_0}^{t}z\Bigr| \leq
\abs{t - t_0}\,\sup_I\abs{y - z} \leq h\,d_\infty(y, z),
\]
और सूप ले लो $ t $.

\textbf{3.} दो निश्चित बिंदु समीकरणों और त्रिभुज का उपयोग करना
असमानता:
\[
d(\ell_g, \ell_{\widetilde g})
= d\bigl(g(\ell_g), \widetilde g(\ell_{\widetilde g})\bigr)
\leq d\bigl(g(\ell_g), g(\ell_{\widetilde g})\bigr) +
d\bigl(g(\ell_{\widetilde g}), \widetilde g(\ell_{\widetilde
g})\bigr)
\leq k\,d(\ell_g, \ell_{\widetilde g}) +
d\bigl(g(\ell_{\widetilde g}), \widetilde g(\ell_{\widetilde
g})\bigr),
\]
और के लिए हल करें $ d(\ell_g, \ell_{\widetilde g})$ (गुणांक
$1 - k $ सकारात्मक है)। दूसरी असमानता मूल्यांकन को बाध्य करती है
वर्दी वाले द्वारा अंतर.

\textbf{4.} $ g^m $ एक अरिक्त पर संकुचन है \omterm{def:b2:metric:complete}{पूरा} स्थान:
इसका एक अद्वितीय निश्चित बिंदु है $\ell $
(\cref{thm:b2:metric:banach}). तब $ g^m(g(\ell)) =
g(g^m(\ell)) = g(\ell)$: $ g(\ell)$ का एक निश्चित बिंदु है $ g^m $,
इसलिए $ g(\ell) = \ell $ विशिष्टता से. का कोई निश्चित बिंदु $ g $ में से एक है $ g^m $: के लिए विशिष्टता $ g $. कक्षाएँ: ठीक करें $ r \in \{0, \dots,
m-1\}$; परिणाम $(x_{qm + r})_q $ है $ g^m $-कक्षा
पर शुरू हुआ $ x_r $, तो यह अभिसरण होता है $\ell $ जैसा $ q \to \infty $
(फिर से बनच)। सभी $ m $ परिणाम उसी में परिवर्तित हो जाते हैं
$\ell $, इस तरह $ x_n \to \ell $: दिया गया $\varepsilon $, प्रत्येक अवशेष
कक्षा अंततः भीतर है $\varepsilon $, और वहाँ निश्चित रूप से हैं
कई वर्ग.

\textbf{5.} अगर $ y $ है \omterm{def:b2:metric:continuity}{निरंतर} मूल्यों के साथ $\intcc{y_0 -
b}{y_0 + b}$, इंटीग्रैंड $ s \mapsto f(s, y(s))$ है \omterm{def:b2:metric:continuity}{निरंतर}
पर $ I $ (रचना), तो दाहिनी ओर है $ C^1$ के साथ
व्युत्पन्न $ f(t, y(t))$ (कैलकुलस का मौलिक प्रमेय, वर्ष 1
वॉल्यूम). अगर $ y $ अभिन्न समीकरण को संतुष्ट करता है, यह वह है
$ C^1$ समारोह, $ y(t_0) = y_0$, और $ y' = f(t, y)$. इसके विपरीत,
एकीकरण $ y' = f(s, y(s))$ से $ t_0$ को $ t $ देता है
अभिन्न समीकरण.

\textbf{6.} $ X_h $ स्थिरांक शामिल है $ y_0$; के रूप में यह बंद है
की पूर्वछवि $\intcc{0}{b}$ नीचे \omterm{def:b2:metric:continuity}{निरंतर} नक्शा $ y
\mapsto d_\infty(y, y_0)$ (दूरियां हैं $1$-लिप्सचिट्ज़), इसलिए
\omterm{def:b2:metric:complete}{पूरा} प्रश्न 1 द्वारा. स्थिरता: के लिए $ y \in X_h $ और $ t \in
I $,
\[
\abs{T(y)(t) - y_0} = \Bigl|\int_{t_0}^{t}f(s, y(s))\dd s\Bigr|
\leq M\abs{t - t_0} \leq Mh \leq b ,
\]
द्वारा अंतिम चरण $ h \leq b/M $ (या $ M = 0$, मामूली)। और $ T(y)$
है \omterm{def:b2:metric:continuity}{निरंतर} ($ C^1$ सम, प्रश्न 5): $ T(y) \in X_h $.

\textbf{7.} के लिए $ t \in I $:
\[
\abs{T(y)(t) - T(z)(t)} \leq \int_{t_0}^{t}\abs{f(s, y(s)) -
f(s, z(s))}\,\abs{\dd s} \leq L\abs{t - t_0}\,d_\infty(y,z)
\leq Lh\,d_\infty(y,z).
\]
अगर $ Lh < 1$: $ T $ अरिक्त का संकुचन है \omterm{def:b2:metric:complete}{पूरा}
$ X_h $, और बानाच एक अद्वितीय निश्चित बिंदु देता है --- प्रश्न 5 द्वारा,
अनोखा समाधान.

\textbf{8.} प्रेरण; मामला $ n = 1$ प्रश्न 7 का मध्य है
असमानता. के लिए बाध्य मानते हुए $ n $, के लिए $ t \geq t_0$ (मामला $ t \leq t_0$ सममित है):
\[
\abs{T^{n+1}(y)(t) - T^{n+1}(z)(t)}
\leq L\int_{t_0}^{t}\abs{T^n(y)(s) - T^n(z)(s)}\dd s
\leq L\int_{t_0}^{t}\frac{L^n(s - t_0)^n}{n!}\dd s\;
d_\infty(y,z),
\]
और अभिन्न का मूल्यांकन करता है $\frac{L^n(t -
t_0)^{n+1}}{(n+1)!}$: के साथ बंधा हुआ $ n + 1$. इस तरह
$ d_\infty(T^n y, T^n z) \leq \frac{(Lh)^n}{n!}d_\infty(y, z)$,
और $\frac{(Lh)^n}{n!} \to 0$ (घातांकीय श्रृंखला
अभिसरण): कुछ $ T^m $ एक संकुचन है. प्रश्न 4 पर लागू होता है
द \omterm{def:b2:metric:complete}{पूरा} $ X_h $: $ T $ एक अद्वितीय निश्चित बिंदु है, अर्थात\ द
कॉची समस्या का बिल्कुल एक ही समाधान है $ I $ मूल्यों के साथ
$\intcc{y_0 - b}{y_0 + b}$.

\textbf{9.} होने देना $ y $ पर समस्या का समाधान करें $ I $ और मान लीजिए सेट
$ E = \{t \in I, t > t_0 : \abs{y(t) - y_0} > b\}$ गैर-रिक्त है
(पक्ष $ t < t_0$ सममित है); होने देना $\tau = \inf E $. द्वारा
निरंतरता, $\abs{y(s) - y_0} \leq b $ के लिए $ s \in
\intcc{t_0}{\tau}$, तो ग्राफ में निहित है $ R $ वहाँ, अभिन्न
समीकरण कायम है $\tau $, और
\[
\abs{y(\tau) - y_0} = \Bigl|\int_{t_0}^{\tau}f\bigl(s,
y(s)\bigr)\dd s\Bigr| \leq M(\tau - t_0) \leq Mh \leq b .
\]
अगर $\tau < t_0 + h $, के अंक $ E $ मनमाने ढंग से करीब $\tau $
दाईं ओर से, द्वारा निरंतरता, $\abs{y(\tau) - y_0} \geq
b $, इस तरह $= b $; लेकिन फिर प्रदर्शन बल देता है $ M(\tau - t_0) =
Mh $, यानी\ $\tau = t_0 + h $: विरोधाभास। इसलिए $\tau = t_0 + h $,
$ E \subseteq \{t_0 + h\}$, और डिस्प्ले (at $\tau = t_0 + h $)
देता है $\abs{y(t_0 + h) - y_0} \leq b $, सदस्यता का खंडन
में $ E $. इस तरह $ E = \emptyset $: हर समाधान चालू $ I $ में रहता है
बैंड, का एक निश्चित बिंदु है $ T $ में $ X_h $, और विशिष्टता है
बिना शर्त.

\textbf{10.} $ T(y)(t) = 1 + \int_0^t y $. से $ y^{(0)} \equiv
1$:
\[
y^{(1)}(t) = 1 + t,\quad
y^{(2)}(t) = 1 + t + \frac{t^2}2,\quad\dots\quad
y^{(n)}(t) = \sum_{k=0}^{n}\frac{t^k}{k!}
\]
(प्रेरण: आंशिक योग को एकीकृत करने से अगला पद जुड़ जाता है)।
ये टेलर के आंशिक योग हैं $\eu^{\,t}$; किसी भी सीमा पर
$ I $ वे समान रूप से एकत्रित होते हैं $\eu^{\,t}$ (पूंछ है
अभिसरण संख्यात्मक श्रृंखला का प्रभुत्व $\sum h^k/k!$),
जो वास्तव में अनोखा समाधान है।

\textbf{11.} $ y \equiv 0$ एक समाधान है. के लिए $ y_c $: यह है $ C^1$
(दोनों टुकड़े हैं, और पर $ t = c $ डेरिवेटिव मैच: $0$ और
$2(t - c) \to 0$), और के लिए $ t > c $: $ y_c' = 2(t - c) =
2\sqrt{(t-c)^2} = 2\sqrt{\abs{y_c}}$; के लिए $ t \leq c $ दोनों तरफ
गायब हो जाना. तो कॉची समस्या $ y(0) = 0$ अपरिमित रूप से अनेक हैं
समाधान ($ c \geq 0$ मनमाना, और $ y \equiv 0$). फील्ड
$\varphi(y) = 2\sqrt{\abs y}$ क्या नहीं है \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} पास में $0$:
$\frac{\varphi(y) - \varphi(0)}{y - 0} = \frac{2}{\sqrt y} \to
+\infty $ जैसा $ y \to 0^+$: कोई स्थिरांक नहीं $ L $ किसी पर काम करता है
का पड़ोस $0$ --- बिल्कुल कहाँ सभी समाधान
शाखा.

\textbf{12.} चरों को अलग करना (या सीधे जाँच करना),
अद्वितीय स्थानीय समाधान है $ y(t) = \frac1{1 - t}$, पर परिभाषित किया गया
$\intoo{-\infty}{1}$ और पर उड़ रहा है $ t = 1$. के लिए
आयत $\intcc{-a}{a} \times \intcc{1 - b}{1 + b}$: $ M =
\sup y^2 = (1 + b)^2$, तो प्रमाणित आधी-चौड़ाई है $ h =
\min\bigl(a, \frac{b}{(1+b)^2}\bigr)$. अधिकतम
$\frac{b}{(1+b)^2}$: व्युत्पन्न शून्य पर $ b = 1$, कीमत
$\frac14$. अतः प्रमेय केवल जीवन की गारंटी देता है
$\intcc{-\frac14}{\frac14}$ --- सही से कम सही
जीवनकाल $1$ आगे, और असीम रूप से कम पीछे: प्रमेय
स्वभाव से स्थानीय है, और ब्लो-अप से पता चलता है कि यह अन्यथा नहीं हो सकता।

\textbf{13.} $ g $ एमएपीएस $\intcc12$ अपने आप में: $ g $ पर घट जाती है
$\intcc{1}{\sqrt2}$ और (अध्ययन के बाद) बढ़ जाता है $ g'(x) = \frac12
- \frac1{x^2}$), साथ $ g(1) = g(2) = \frac32$ और न्यूनतम
$ g(\sqrt2) = \sqrt2 > 1$: $ g(\intcc12) \subseteq
\intcc{\sqrt2}{\frac32} \subseteq \intcc12$; और $ g $ मानचित्र
तर्कसंगत से तर्कसंगत। सिकुड़न: $\abs{g'(x)} = \abs{\frac12
- \frac1{x^2}} \leq \frac12$ पर $\intcc12$ ($\frac1{x^2} \in
\intcc{\frac14}{1}$), तो औसत मूल्य असमानता देता है
$\abs{g(x) - g(y)} \leq \frac12\abs{x - y}$. एक निश्चित बिंदु
संतुष्ट करता है $\frac x2 = \frac1x $, यानी\ $ x^2 = 2$: असंभव में
$\Q $. असफल परिकल्पना है पूर्णता का $ X $ ($\Q \cap
\intcc12$ क्या नहीं है \omterm{def:b2:metric:complete}{पूरा}); पूर्णता में $\intcc12$ द
निश्चित बिंदु है $\sqrt2$ --- बानाच प्रमेय पर चलते हैं
तर्कसंगत \emph{बनाता है} तर्कहीन.

\textbf{14.} वापस: $ d(x_n, \ell) \leq d(x_n, x_{n+1}) +
d(x_{n+1}, \ell) \leq k\,d(x_{n-1}, x_n) + k\,d(x_n, \ell)$,
कहाँ से $ d(x_n, \ell) \leq \frac{k}{1-k}d(x_n, x_{n-1})$. बगुला
से $ x_0 = \frac32$: $ x_1 = \frac{17}{12}$, $ d(x_1, x_0) =
\frac1{12}$, $ k = \frac12$: एक प्राथमिकता बाध्य
$\frac{k^n}{1-k}d(x_1,x_0) = \frac{2^{-n+1}}{12}$ नीचे गिरता है
$10^{-6}$ पहले पर $ n = 18$. यथार्थ में $ x_1 = \frac{17}{12}
\approx 1.41667$ (गलती $2.5\cdot10^{-3}$), $ x_2 =
\frac{577}{408} \approx 1.4142157$ (गलती $2.1\cdot10^{-6}$),
$ x_3 \approx 1.41421356237469$ (गलती $1.6\cdot10^{-12}$):
तीन चरण पर्याप्त हैं. प्रत्येक बगुला मोटे तौर पर कदम रखता है \emph{चौकों} द
त्रुटि (द्विघात अभिसरण, एक न्यूटन घटना:
\cref{ch:b2:realfun}); संकुचन का अनुमान, जो केवल
इसे आधा कर देता है, सबसे खराब स्थिति के लिए ईमानदार है लेकिन यहां निराशावादी है।

\textbf{15.} प्रश्न 3 को साथ में लागू करें $ g = T_{y_0}$ (एक
$ Lh $-सिकुड़न, $ Lh < 1$) और $\widetilde g = T_{z_0}$, किसका
निश्चित बिंदु है $ y[z_0]$. किसी के लिए $ y $,
\[
\abs{T_{y_0}(y)(t) - T_{z_0}(y)(t)} = \abs{y_0 - z_0},
\]
(अभिन्नांक समान हैं), इसलिए $\sup_y

d_\infty(y[y_0], y[z_0]) \leq \frac{\abs{y_0 - z_0}}{1 - Lh} .

\textbf{16.} प्रत्येक टुकड़े पर, प्रश्न 15 लागू होता है
प्रारंभिक डेटा के रूप में बाएं-अंतबिंदु मान विचलन को सीमित करता है
सही समापन बिंदु:
\[
d_\infty \leq \frac{1}{1 - 1/2}\,(\text{left deviation})
= 2\,(\text{left deviation}) .
\] द्वारा
के ऊपर प्रेरण $ m $ टुकड़े, अंतिम विचलन अधिकतम है
$2^m\abs{y_0 - z_0}$, और संपूर्ण पर एकसमान विचलन
खंड समान सीमा का पालन करता है (प्रत्येक टुकड़े का समर्थन नियंत्रित होता है
इसका चरण)। लंबाई के टुकड़ों के साथ $ h \asymp \frac1{2L}$, द
कारक है $2^m = 2^{\,\text{length}\cdot 2L}$: घातीय में
अंतराल की लंबाई, बिल्कुल ग्रोनवाल की सीमा के समान
$\eu^{L\abs{t-t_0}}$ बेहतर स्थिरांक के साथ भविष्यवाणी करता है।

\textbf{17.} वही योजना: के लिए $ y \in X_h $,

\abs{T_f(y)(t) - T_g(y)(t)} \leq \int_{t_0}^t \abs{f(s,y(s)) -

\textbf{19.} प्रत्येक तर्क का केवल उपयोग किया जाता है: मीट्रिक स्वयंसिद्ध,
पूर्णता मंच का, बाध्य $\abs{\int} \leq
\int\abs{\cdot}$, और यह \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} की संपत्ति $ f $ --- सभी
के लिए उपलब्ध है $\R^n $-मूल्यवान कार्यों के साथ $ d_\infty $ पर बनाया गया
की समतुल्य दूरियों में से कोई भी \cref{ex:b2:metric:examples}
(\omterm{def:b2:metric:complete}{पूरा} द्वारा \cref{thm:b2:metric:rncomplete}). के लिए $ y' = Ay $
निलपोटेंट के साथ $ A $: $ y^{(0)} \equiv (c_1, c_2)$,
\[
y^{(1)}(t) = (c_1, c_2) + \int_0^t (c_2, 0)\,\dd s
= (c_1 + tc_2,\; c_2),
\]
और $ Ay^{(1)}(s) = (c_2, 0)$ दोबारा: $ y^{(2)} = y^{(1)}$. द
पुनरावृत्ति से स्थिर है $ n = 1$ पर, सटीक समाधान पर
$ y(t) = (c_1 + tc_2, c_2) = \eu^{tA}y(0)$ --- शून्यशक्ति
घातीय श्रृंखला को छोटा करता है, और पिकार्ड नोटिस करता है।

\textbf{20.} हल करना $ y \in X $ और चलो $ g_y(x) = y - \eta(x)$: ए
$ k $- का संकुचन \omterm{def:b2:metric:complete}{पूरा} $ X $, तो बिल्कुल एक है $ x $
साथ $ x + \eta(x) = y $: वो नक्शा $\Phi = \mathrm{id} + \eta $ है
विशेषण। \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} उलटा: यदि $\Phi(x) = y $ और $\Phi(x') =
y'$,
\[
d(x, x') \leq d(y, y') + d\bigl(\eta(x), \eta(x')\bigr)
\leq d(y, y') + k\,d(x, x'),
\]
इसलिए $ d(x, x') \leq \frac{1}{1-k}d(y, y')$. (यहाँ दूरियाँ आती हैं
मानक संरचना से, इसलिए $ d(a - c, b - c) = d(a, b)$, जो
पहली असमानता का प्रयोग किया गया।)

\textbf{21.} $ g(x) = m + e\sin x $ है $ e $-लिप्सचिट्ज़ पर
\omterm{def:b2:metric:complete}{पूरा} $\R $ (माध्य मूल्य असमानता, $\abs{g'} = \abs{e\cos x}
\leq e < 1$): बनच एक अनोखा और वैश्विक समाधान देता है
पुनरावृत्ति का अभिसरण. के लिए $ e = \frac12$, $ m = 1$, $ x_0
= 1$: $ x_1 = 1 + \frac12\sin 1 \approx 1.42074$, $ d(x_1, x_0)
\approx 0.4207$, और एक प्राथमिकता बाध्य
$\frac{(1/2)^n}{1/2}\cdot 0.4207 \leq 10^{-3}$ पहले पर रखता है
$ n = 10$: दस पुनरावृत्तियाँ प्रमाणित (सही मूल्य $ x \approx
1.4987$ वास्तव में पहुँच गया है $10^{-3}$ पहले से ही आसपास $ n =
5$).

\textbf{22.} अंक विवरण का प्रयोग करें
(\cref{exo:b2:metric:8}): $ C $ योगों का समुच्चय है
$\sum_{n\geq1}a_n3^{-n}$, $ a_n \in \{0, 2\}$. तब $ S_1(C) =
\{x/3 : x \in C\}$ के साथ उपसमुच्चय है $ a_1 = 0$, और $ S_2(C) =
\{x/3 + 2/3\}$ के साथ उपसमुच्चय $ a_1 = 2$: उनका मिलन, खत्म
का निःशुल्क चयन $ a_1$, बिलकुल है $ C $. एक में विशिष्टता
वाक्य: अरिक्त के स्थान पर \omterm{def:b2:metric:compact}{कॉम्पैक्ट} के उपसमुच्चय
$\intcc01$ हॉसडॉर्फ़ दूरी द्वारा मापित, $ A \mapsto
S_1(A) \cup S_2(A)$ एक है $\frac13$-ए का संकुचन \omterm{def:b2:metric:complete}{पूरा}
स्थान, इसलिए बानाच केवल एक निश्चित सेट की अनुमति देता है --- वर्ष 3
वॉल्यूम हॉसडॉर्फ़ मीट्रिक और इस तर्क को कठोर बनाता है।

\textbf{23.} होने देना $ Z = \{t \in J : y(t) = z(t)\}$: गैररिक्त
($ t_0 \in Z $), बंद किया $ J $ (दो का तुल्यकारक \omterm{def:b2:metric:continuity}{निरंतर} मानचित्र:
की पूर्वछवि $\{0\}$ अंतर्गत $ y - z $). \omterm{def:b2:metric:topology}{खुला}: अगर $ t_1 \in Z $, लागू करें
बिंदु पर स्थानीय प्रमेय (प्रश्न 8)। $(t_1, y(t_1))$,
एक आयत में जहां $ f $ है \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़}: थोड़े-थोड़े अंतराल पर
चारों ओर $ t_1$ दोनों $ y $ और $ z $ उसी कॉची समस्या को हल करें, इसलिए
वे वहां मेल खाते हैं (प्रश्न 9 की बिना शर्त विशिष्टता):
का एक पड़ोस $ t_1$ में निहित है $ Z $. का एक गैररिक्त उपसमुच्चय
अंतराल $ J $ वह दोनों है \omterm{def:b2:metric:topology}{खुला} और अंदर बंद हो गया $ J $ सब कुछ है $ J $
(अंतराल हैं \omterm{def:b2:metric:connected}{जुड़े हुए},
\cref{thm:b2:metric:connectedness}): $ y = z $ पर $ J $.

\textbf{24.} यहाँ $ f(t, y) = \alpha(t)y + \beta(t)$ है
$ L $-लिप्सचिट्ज़ में $ y $ सब पर $\intcc AB \times \R $ साथ $ L =
\sup\abs\alpha $ (परिमित: $\alpha $ \omterm{def:b2:metric:continuity}{निरंतर} एक खंड पर), और
कोई बैंड नहीं $\intcc{y_0 - b}{y_0 + b}$ आवश्यक है: ले लो $ X = C(\intcc
AB)$ संपूर्ण, जिस पर $ T $ अच्छी तरह से परिभाषित है. का प्रेरण
प्रश्न 8 शब्दशः चलता है और देता है $ d_\infty(T^ny, T^nz) \leq
\frac{(L(B - A))^n}{n!}d_\infty(y, z)$: एक पुनरावृत्त एक है
लंबाई चाहे जो भी हो, संकुचन $ B - A $, और प्रश्न 4
निष्कर्ष: पूरे खंड पर एक और केवल एक ही समाधान।
रैखिकता बिल्कुल एक बार प्रवेश करती है: यह बनाती है \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} बाध्य
वैश्विक में $ y $, आयत और उसके को हटा रहा है $ h \leq b/M $.

\textbf{25.} \emph{संपूर्णता} का कॉची अनुक्रम बदल दिया
एक वास्तविक समाधान (प्रश्न 1, 6, 8), और उसके में पुनरावृत्त होता है
अनुपस्थिति चलो $\sqrt2$ से भागना $\Q $ (प्रश्न 13).
\emph{सिकुड़न} विशिष्टता, एल्गोरिथ्म, और दिया
त्रुटि पट्टियाँ (प्रश्न 7, 14)। \emph{पुनरावृत्त चाल} हटा दिया गया
लघुता की स्थिति $ Lh < 1$, तो प्रमाणित अंतराल
पर ही निर्भर करता है $ M $, पर नहीं $ L $ --- और रैखिक समीकरण बनाये
वैश्विक (प्रश्न 8,24)। \emph{संयुक्तता} स्थानीय को बढ़ावा दिया गया
वैश्विक विशिष्टता की विशिष्टता (प्रश्न 23)। द
प्रतिउदाहरण: $2\sqrt{\abs y}$ गार्ड \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} (प्रश्न
11), $ y^2$ गार्ड लोकैलिटी (प्रश्न 12), $\Q $ गार्ड
पूर्णता (प्रश्न 13). शिखर है
पिकार्ड--लिंडेल\"प्रमेय का (प्रश्न 8); कब $ f$ मात्र है
\omterm{def:b2:metric:continuity}{निरंतर}, अस्तित्व तो बच जाता है लेकिन विशिष्टता मर जाती है, और
प्रमाण संकुचन का व्यापार करता है सघनता फ़ंक्शन सेट का
--- अर्ज़ेल के माध्यम से पीनो का प्रमेय\`ए--अस्कोली, वर्ष 3 में
आयतन.
