\begin{solution}{exo:b2:diffcalc:2}
के लिए $ v = (a, b) \neq 0$: $\frac{f(tv) - 0}{t} = \frac{t^3a^3}{t\cdot

\end{solution}

\begin{solution}{exo:b2:diffcalc:3}
$ f = x^3 + y^3 - 3xy $: महत्वपूर्ण बिंदु $(0,0)$ और $(1,1)$ (वर्ष 1
गणना). हेस्सियन: $ H = \begin{pmatrix} 6x & -3\\ -3 &
6y\end{pmatrix}$. पर $(0,0)$: ईजिन-चिह्न मिश्रित ($\det = -9 < 0$):
काठी. पर $(1,1)$: $\det = 27 > 0$, पता लगाना $> 0$: सकारात्मक
निश्चित, सख्त स्थानीय न्यूनतम --- अब उचित है
\cref{thm:b2:diffcalc:extrema} \omterm{def:b2:structures:generated}{आदेश} के बजाय.

$ g = x^4 + y^4 - 2(x-y)^2$: $\nabla g = (4x^3 - 4(x - y),\; 4y^3 +
4(x-y))$; महत्वपूर्ण बिंदु $(0,0)$, $(\sqrt2, -\sqrt2)$,
$(-\sqrt2, \sqrt2)$ (वर्ष 1). पर $(\pm\sqrt2, \mp\sqrt2)$: $ H =
\begin{pmatrix} 12\cdot2 - 4 & 4\\ 4 & 20\end{pmatrix} =
\begin{pmatrix} 20 & 4\\ 4 & 20\end{pmatrix}$: सकारात्मक निश्चित
(तिरछे प्रमुख; eigenvalues $24, 16$): सख्त स्थानीय न्यूनतम।
पर $(0,0)$: $ H = \begin{pmatrix} -4 & 4\\ 4 & -4\end{pmatrix}$,
एकवचन नकारात्मक अर्धनिश्चित: परीक्षण मौन है; दिशात्मक
अध्ययन ($ g(x,x) = 2x^4 > 0$, $ g(x,-x) = 2x^4 - 8x^2 < 0$ छोटा)
एक काठी जैसा बिंदु दिखाता है --- कोई चरम सीमा नहीं।
\end{solution}

\begin{solution}{exo:b2:diffcalc:4}
\omterm{def:b2:diffcalc:differential}{अंतर} $ t \mapsto f(tx)$ पर $ t = 1$: शृंखला नियम द्वारा,
$\langle \nabla f(x), x\rangle $; एकरूपता से समान कार्य
है $ t^pf(x)$, व्युत्पन्न का $ pf(x)$ पर $ t = 1$: यूलर की पहचान.

बातचीत: ठीक करो $ x \neq 0$ और चलो $\varphi(t) = f(tx) - t^p f(x)$
पर $ t > 0$. तब $\varphi'(t) = \langle\nabla f(tx), x\rangle -
pt^{p-1}f(x) = \frac1t\bigl(\langle \nabla f(tx), tx\rangle -
p\,t^pf(x)\bigr)$. परिकल्पना --- बिंदु पर यूलर की पहचान
$ tx $ --- ब्रैकेट का मूल्यांकन इस प्रकार करता है $ p\,f(tx) - p\,t^pf(x) =
p\,\varphi(t)$. इसलिए $\varphi' = \frac{p}{t}\varphi $ साथ $\varphi(1)
= 0$: रैखिक ODE का अद्वितीय समाधान है $\varphi \equiv 0$ (वर्ष
1 विशिष्टता), अर्थात्।\ $ f(tx) = t^pf(x)$.
\end{solution}

\begin{solution}{exo:b2:diffcalc:5}
विस्तार $ f(x + h) - f(x) = \langle Ax - b, h\rangle +
\frac12\langle Ah, h\rangle $ (की समरूपता $ A $): $\nabla f(x) = Ax -
b $ और $ H_f = A $ हर जगह. $ f $ उत्तल है यदि $ A $ सकारात्मक है
अर्धनिश्चित (हेस्सियन परीक्षण, तब से वैश्विक)। $ H $ स्थिर है:
दूसरे क्रम का टेलर फॉर्मूला सटीक है)। अगर $ A $ सकारात्मक है
निश्चित: अद्वितीय महत्वपूर्ण बिंदु है $ x^* = A^{-1}b $, और $ f(x^*
+ h) - f(x^*) = \frac12\langle Ah, h\rangle \geq
\frac{\lambda_{\min}}{2}\norm h^2 > 0$ के लिए $ h \neq 0$: सख्त
वैश्विक न्यूनतम.
\end{solution}

\begin{solution}{exo:b2:diffcalc:6}
तब से $\nabla g(a) \neq 0$, एक आंशिक, कहते हैं $\frac{\partial
g}{\partial y}(a) \neq 0$: \omterm{def:b2:diffcalc:differential}{अंतर}्निहित कार्य प्रमेय
(\cref{thm:b2:diffcalc:inverse}का साथी) पैरामीटराइज़ करता है $\{g =
0\}$ पास में $ a = (a_1, a_2)$ जैसा $\gamma(t) = (t, y(t))$ साथ $ y $
$ C^1$, $ y'(t) = -\frac{\partial_x g}{\partial_y g}(\gamma(t))$
(\omterm{def:b2:diffcalc:differential}{अंतर} करें $ g(t, y(t)) = 0$). एक-चर फ़ंक्शन $ t
\mapsto f(\gamma(t))$ पर एक स्थानीय चरम सीमा है $ t = a_1$:
\[
0 = \frac{\dd}{\dd t}f(\gamma(t))\Big|_{a_1}
= \partial_x f(a) + \partial_y f(a)\,y'(a_1)
= \partial_x f(a) - \partial_yf(a)\frac{\partial_x
g(a)}{\partial_y g(a)} :
\]
सदिश $\nabla f(a)$ और $\nabla g(a)$ आनुपातिक है
निर्देशांक: $\nabla f(a) = \lambda \nabla g(a)$ साथ $\lambda =
\frac{\partial_y f(a)}{\partial_y g(a)}$.

आवेदन: वृत्त पर, $\nabla(xy) = (y, x)$ के समानांतर
$(2x, 2y)$ बल $ y^2 = x^2$; बाधा के साथ, उम्मीदवार
हैं $\pm\bigl(\tfrac{1}{\sqrt2}, \tfrac{1}{\sqrt2}\bigr)$ (कीमत
$\frac12$) और $\pm\bigl(\tfrac{1}{\sqrt2},
-\tfrac{1}{\sqrt2}\bigr)$ (कीमत $-\frac12$): अधिकतम $\frac12$, मि
$-\frac12$ (प्राप्त: वृत्त है \omterm{def:b2:metric:compact}{कॉम्पैक्ट}).
\end{solution}

\begin{solution}{exo:b2:diffcalc:7}
$\nabla f = (2x - y + 1,\; 2y - x - 1) = 0$: सुलझाना, $ x =
-\frac13$, $ y = \frac13$. हेस्सियन $\begin{pmatrix} 2 & -1\\ -1 &
2\end{pmatrix}$, सकारात्मक निश्चित: द्विघात का वैश्विक न्यूनतम
$ f $, कीमत $ f\bigl(-\frac13, \frac13\bigr) = -\frac13$.

त्रिकोण पर $ T $: आंतरिक महत्वपूर्ण बिंदु $(-\frac13,
\frac13) \notin T $ (नकारात्मक $ x $). किनारे: चालू $ y = 0$, $ x \in
\intcc{0}{1}$: $ f = x^2 + x $, बढ़ रहा है: चरम $0$ और $2$. पर
$ x = 0$: $ f = y^2 - y $, न्यूनतम $-\frac14$ पर $ y = \frac12$, मूल्य
$0$ और $0$ अंत में. पर $ x + y = 1$: विकल्प $ y = 1 - x $,
$ f = x^2 + (1-x)^2 - x(1-x) + x - (1-x) = 3x^2 - x $; पर
$\intcc{0}{1}$: न्यूनतम $-\frac{1}{12}$ पर $ x = \frac16$, मूल्य
$0$ (पर $ x=0$) और $2$ (पर $ x=1$). शीर्ष: $ f(0,0) = 0$, $ f(1,0)
= 2$, $ f(0,1) = 0$. वैश्विक पर $ T $: न्यूनतम $-\frac14$ पर $(0,
\frac12)$, अधिकतम $2$ पर $(1, 0)$.
\end{solution}

\begin{solution}{exo:b2:diffcalc:10}
$ f(x+h) - f(x) = 2\langle x, h\rangle + \norm h^2$: $\dd f_x =
2\langle x, \cdot\rangle $, $\nabla f(x) = 2x $, हेसियन $2I $
(स्थिर)। के लिए $ g $:
\[
g(x + h) - g(x) = \langle Ax, h\rangle + \langle Ah, x\rangle
+ \langle Ah, h\rangle
= \bigl\langle (A + A^{\mathsf T})x,\ h\bigr\rangle +
O(\norm h^2),
\]
इसलिए $\nabla g(x) = (A + A^{\mathsf T})x $ और $ H_g = A +
A^{\mathsf T}$, स्थिर। द्वारा \cref{exo:b2:diffcalc:11}, $ g $ है
उत्तल आईएफएफ $ A + A^{\mathsf T}$ सकारात्मक अर्धनिश्चित है ---
केवल \omterm{def:b2:quadratic:adjoint}{सममित} का हिस्सा $ A $ वास्तव में मायने रखता है $ g(x) =
\langle \frac{A + A^{\mathsf T}}2 x, x\rangle $.
\end{solution}

\begin{solution}{exo:b2:diffcalc:11}
$ f $ उत्तल है यदि प्रत्येक खंड पर इसका प्रतिबंध उत्तल है,
यानी\ यदि हर $\varphi(t) = f(a + tv)$ उत्तल है. जंजीर से
नियम $\varphi''(t) = \langle H_{a+tv}\,v,\ v\rangle $.

यदि सभी हेसियन सकारात्मक अर्धनिश्चित हैं: $\varphi'' \geq 0$,
तो प्रत्येक $\varphi $ उत्तल है (वर्ष 1 खंड) और $ f $ उत्तल है.
इसके विपरीत यदि $ f $ उत्तल है, प्रत्येक $\varphi $ उत्तल है, इसलिए
$\varphi''(0) \geq 0$: $\langle H_a v, v\rangle \geq 0$ हरएक के लिए $ a $ और हर दिशा $ v $: सभी हेस्सियन सकारात्मक हैं
अर्धनिश्चित.
\end{solution}

\begin{solution}{exo:b2:diffcalc:12}
\begin{enumerate}
  \item माध्य मूल्य असमानता से
        (\cref{thm:b2:diffcalc:mvi}) के लिए आवेदन किया $ g $:
        $\norm{g(x) - g(y)} \leq k\norm{x-y}$, इसलिए
        \[
        \norm{f(x) - f(y)} \geq \norm{x - y} - \norm{g(x) -
        g(y)} \geq (1 - k)\norm{x - y} :
        \]
        $ f $ इंजेक्शन है और $ f^{-1}$ (छवि पर परिभाषित) है
        $\frac{1}{1-k}$-लिप्सचिट्ज़.
  \item हल करना $ y $; $ T(x) = y - g(x)$ संतुष्ट $\norm{T(x) -
        T(x')} = \norm{g(x') - g(x)} \leq k\norm{x - x'}$: ए
        का संकुचन \omterm{def:b2:metric:complete}{पूरा} \omterm{def:b2:diffcalc:differential}{अंतर}िक्ष $\R^n $. बनच
        (\cref{thm:b2:metric:banach}) एक निश्चित बिंदु देता है $ x^*
        = y - g(x^*)$, यानी\ $ f(x^*) = y $: $ f $ विशेषण है.
  \item $ f $ इस प्रकार यह एक आक्षेप है $\R^n $ साथ
        $\frac{1}{1-k}$-लिप्सचिट्ज़ व्युत्क्रम: एक वैश्विक व्युत्क्रम
        प्रमेय, जहां की लघुता $\dd g $ हर जगह
        की स्थानीय व्युत्क्रमणीयता परिकल्पना को प्रतिस्थापित करता है
        \cref{thm:b2:diffcalc:inverse}.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b2:diffcalc:1}
\textbf{1.} $\norm{ABx} \leq \vertiii A\norm{Bx} \leq
\vertiii A\vertiii B\norm x $:सुपर ले लो $\norm x = 1$.
मैट्रिक्स उत्पाद, $\det $ और $\operatorname{tr}$ बहुपद हैं
प्रविष्टियों के कार्य, इसलिए \omterm{def:b2:metric:continuity}{निरंतर} ($\mathcal M_n(\R)
\simeq \R^{n^2}$, सभी मानदंड \omterm{def:b2:nvs:equivalent}{समकक्ष}:
\cref{thm:b2:nvs:finitedim}).

\textbf{2.} $\sum\vertiii{X^k} \leq \sum\vertiii X^k <
\infty $: शृंखला अभिसरित होती है \omterm{def:b2:series:def}{बिल्कुल}, इसलिए \omterm{def:b2:integration:improper}{अभिसरण} होता है
(\cref{thm:b2:nvs:absoluteconvergence}). से $(I -
X)\sum_{k\leq N}X^k = I - X^{N+1} \to I $: योग है $(I -
X)^{-1}$. आदर्श: $\leq \sum\vertiii X^k = \frac{1}{1 -
\vertiii X}$; और
\[
(I - X)^{-1} - I - X = \sum_{k\geq2}X^k = X^2(I - X)^{-1},
\qquad
\vertiii{X^2(I-X)^{-1}} \leq
\frac{\vertiii X^2}{1 - \vertiii X} = O(\vertiii X^2).
\]

\textbf{3.} $ A + H = A(I + A^{-1}H)$ साथ $\vertiii{A^{-1}H}
\leq \vertiii{A^{-1}}\vertiii H < 1$: प्रश्न 2 से उलटा:
द \omterm{def:b2:metric:topology}{खुला} त्रिज्या की गेंद $\vertiii{A^{-1}}^{-1}$ आस-पास $ A $ में निहित है $ GL_n(\R)$. घनत्व: $\det(A + \varepsilon I)$ एक डिग्री है-$ n $
बहुपद में $\varepsilon $ अग्रणी गुणांक के साथ $1$: यह
इसकी जड़ें बहुत सी हैं, इसलिए हैं $\varepsilon_k \to 0$
साथ $ A + \varepsilon_kI $ उलटा, अभिसरित $ A $.

\textbf{4.} के लिए $\vertiii H < \frac{1}{2\vertiii{A^{-1}}}$:
\[
(A + H)^{-1} - A^{-1}
= \bigl[(I + A^{-1}H)^{-1} - I\bigr]A^{-1},
\]
का आदर्श अधिक से अधिक $\frac{\vertiii{A^{-1}H}}{1 -
\vertiii{A^{-1}H}}\,\vertiii{A^{-1}} \leq
2\vertiii{A^{-1}}^2\vertiii H \to 0$: $\Phi $ है \omterm{def:b2:metric:continuity}{निरंतर} पर
प्रत्येक $ A \in GL_n(\R)$.

\textbf{5.} $(A+H)^2 = A^2 + AH + HA + H^2$: वो नक्शा $ H
\mapsto AH + HA $ रैखिक है और त्रुटि $ H^2$ है
$ O(\vertiii H^2)$. विस्तार $(A + H)^k $ और के अनुसार क्रमबद्ध करें
की संख्या $ H $-कारक: रैखिक भाग है
$\sum_{i=0}^{k-1}A^iHA^{k-1-i}$, और शर्तों के साथ $\geq 2$
कारकों $ H $ से बंधे हैं $\binom k2$-के कई उत्पाद आदर्श
$\leq \vertiii A^{k-2}\vertiii H^2$-पैमाना: $ O(\vertiii H^2)$.
कोई भी योग को संक्षिप्त नहीं कर सकता $ kA^{k-1}H $ क्योंकि $ H $ और $ A $
लघुकरण की आवश्यकता नहीं है --- योग सही गैर-अनुक्रमणात्मक है
व्युत्पन्न.

\textbf{6.} छोटे के लिए $ H $:
\[
(A+H)^{-1} = (I + A^{-1}H)^{-1}A^{-1}
= \bigl(I - A^{-1}H + O(\vertiii H^2)\bigr)A^{-1}
= A^{-1} - A^{-1}HA^{-1} + O(\vertiii H^2) :
\]
$\dd\Phi_A(H) = -A^{-1}HA^{-1}$, रैखिक में $ H $. के लिए $ n = 1$:
$\dd(1/a)(h) = -h/a^2$, परिचित व्युत्पन्न।

\textbf{7.} वक्र के साथ श्रृंखला नियम:
$\bigl(A(t)^{-1}\bigr)' = \dd\Phi_{A(t)}(A'(t)) =
-A(t)^{-1}A'(t)A(t)^{-1}$. पर $ A(t) = I + tB $, $ t = 0$: $(I +
tB)^{-1} = I - tB + O(t^2)$.

\textbf{8.} प्रश्न 5 और ट्रेस के \omterm{def:b2:structures:generated}{चक्रीय} अपरिवर्तनीयता से:
\[
\dd f_A(H) = \operatorname{tr}\Bigl(\sum_{i=0}^{k-1}
A^iHA^{k-1-i}\Bigr) = k\operatorname{tr}\bigl(A^{k-1}H\bigr) .
\]
फ्रोबेनियस उत्पाद के विरुद्ध, $\dd f_A(H) =
\operatorname{tr}\bigl((\nabla f)^{\mathsf T}H\bigr)$ आवश्यक है
$(\nabla f)^{\mathsf T} = kA^{k-1}$: $\nabla f(A) =
k\,(A^{k-1})^{\mathsf T}$.

\textbf{9.} $ f(A + H) - f(A) = 2\operatorname{tr}
(A^{\mathsf T}H) + \operatorname{tr}(H^{\mathsf T}H)$: द
\omterm{def:b2:diffcalc:differential}{अंतर} है $ H \mapsto 2\operatorname{tr}(A^{\mathsf T}H) =
2\langle A, H\rangle $, इसलिए $\nabla f(A) = 2A $; दूसरे क्रम का
शब्द \omterm{def:b2:series:def}{बिल्कुल} सही है $\norm H_F^2$: हेस्सियन दोगुना है
पहचान द्विघात रूप, सकारात्मक निश्चित और स्थिर, इसलिए
$\norm\cdot_F^2$ सख्ती से उत्तल है (टेलर सूत्र है
\omterm{def:b2:series:def}{बिल्कुल} यहीं)।

\textbf{10.} स्तंभों में बहुरेखीयता द्वारा, $\det(I + H) =
\sum_{S\subseteq\{1,\dots,n\}}\det(M_S)$ कहाँ $ M_S $ कॉलम है
$ h_j $ के लिए $ j \in S $ और $ e_j $ अन्यथा। $ S = \varnothing $
देता है $1$; $ S = \{j\}$ देता है सिद्ध का $ I $ कॉलम के साथ
$ j $ द्वारा प्रतिस्थापित $ h_j $, अर्थात् इसका $ j $-वीं प्रविष्टि $ h_{jj}$,
संक्षेप में $\operatorname{tr}H $; प्रत्येक पद के साथ $\abs S \geq 2$
एक है सिद्ध आकार के कम से कम दो स्तंभों के साथ
$ O(\vertiii H)$, इस तरह $ O(\vertiii H^2)$ (ए पर बहुरेखीय मानचित्र
परिमित-आयामी स्थान सीमित हैं), और परिमित रूप से हैं
अनेक. इसलिए $\det(I + H) = 1 + \operatorname{tr}H +
O(\vertiii H^2)$: $\dd(\det)_I = \operatorname{tr}$.

\textbf{11.} $\det(A + H) = \det A\,\det(I + A^{-1}H) =
\det A\,\bigl(1 + \operatorname{tr}(A^{-1}H) +
O(\vertiii H^2)\bigr)$: द \omterm{def:b2:diffcalc:differential}{अंतर} है $ H \mapsto
\det(A)\operatorname{tr}(A^{-1}H)$.

\textbf{12.} पंक्ति के साथ लाप्लास विस्तार $ i $: $\det A =
\sum_j a_{ij}C_{ij}$, और सहकारक $ C_{ij}$ शामिल न हों
पंक्ति $ i $: $\frac{\partial\det}{\partial a_{ij}} = C_{ij}$. इस तरह
\[
\dd(\det)_A(H) = \sum_{i,j}C_{ij}h_{ij}
= \operatorname{tr}\bigl(\operatorname{com}(A)^{\mathsf T}
H\bigr)
= \operatorname{tr}\bigl(\operatorname{adj}(A)H\bigr),
\qquad
\nabla(\det)(A) = \operatorname{com}(A) .
\]
उलटे के लिए $ A $, $\operatorname{adj}A = \det(A)A^{-1}$
प्रश्न 11 को पुनर्प्राप्त करता है। ए के साथ $ C^1$ वक्र, श्रृंखला नियम पढ़ता है
$(\det A(t))' = \operatorname{tr}(\operatorname{adj}
(A(t))\,A'(t))$: \omterm{pb:b2:diffcalc:1}{जैकोबी का सूत्र}.

\textbf{13.} साथ $ A' = MA $ और
$\operatorname{adj}(A)A = \det(A)I $:
\[
(\det A)' = \operatorname{tr}\bigl(\operatorname{adj}(A)MA
\bigr)
= \operatorname{tr}\bigl(A\operatorname{adj}(A)M\bigr)
= \det A\;\operatorname{tr}M
\]
(\omterm{def:b2:structures:sn}{चक्र}ीयता; $ A\operatorname{adj}A = \det(A) I $ भी)। द
अदिश रैखिक ODE $ y' = \operatorname{tr}(M(t))\,y $ के पास है
अनोखा समाधान $ y(t) = y(0)\exp\bigl(\int_0^t
\operatorname{tr}M\bigr)$ (वर्ष 1): लिउविले का सूत्र।

\textbf{14.} पर $ A \in SL_n(\R)$, लेना $ H = \frac1nA $:
$\dd(\det)_A\bigl(\tfrac1nA\bigr) = \frac1n\det(A)
\operatorname{tr}(A^{-1}A) = \frac1n\cdot1\cdot n = 1 \neq 0$:
द \omterm{def:b2:diffcalc:differential}{अंतर} एक अशून्य रैखिक रूप है, इसलिए विशेषण
पर $\R $ स्तर सेट के प्रत्येक बिंदु पर: $ SL_n(\R)$ एक है
सुचारू स्तर सेट (आयाम का)। $ n^2 - 1$).

\textbf{15.} $\sum_k\vertiii{A^k/k!} \leq
\sum\vertiii A^k/k! = \eu^{\vertiii A}$: पूर्ण \omterm{def:b2:integration:improper}{अभिसरण}
(प्रश्न 2 पूर्णता तर्क), साथ सामान्य \omterm{def:b2:integration:improper}{अभिसरण}
हर गेंद पर $\vertiii A \leq R $ (अवश्यंभावी $ R^k/k!$ स्वतंत्र
का $ A $). प्रत्येक आंशिक योग है \omterm{def:b2:metric:continuity}{निरंतर} (बहुपद); द
गेंदों पर एकसमान सीमा है \omterm{def:b2:metric:continuity}{निरंतर}: $ A \mapsto \eu^A $ है
\omterm{def:b2:metric:continuity}{निरंतर}, साथ $\vertiii{\eu^A} \leq \eu^{\vertiii A}$.

\textbf{16.} दोनों शृंखलाएँ मिलती हैं \omterm{def:b2:series:def}{बिल्कुल}, इतना \omterm{thm:b2:series:fubini}{कॉची
उत्पाद} वैध है (\cref{thm:b2:series:fubini}):
\[
\eu^A\eu^B = \sum_{n\geq0}\frac{1}{n!}\sum_{k=0}^n\binom
nkA^kB^{n-k}
= \sum_{n\geq0}\frac{(A+B)^n}{n!} = \eu^{A+B},
\]
द्विपद पहचान की आवश्यकता है $ AB = BA $. साथ $ B = -A $:
$\eu^A\eu^{-A} = \eu^0 = I $: प्रत्येक $\eu^A \in GL_n(\R)$.

\textbf{17.} शृंखला $\sum t^kA^k/k!$ और इसकी व्युत्पन्न श्रृंखला
$\sum t^{k-1}A^k/(k-1)! = A\sum t^{k-1}A^{k-1}/(k-1)!$ एकाग्र
\omterm{def:b2:funcseq:series}{सामान्यतः} हर खंड पर $\abs t \leq T $ (सीमा
$ T^k\vertiii A^k/k!$): श्रृंखला के लिए विभेदन प्रमेय
(\cref{thm:b2:funcseq:seriestransfer}, प्रविष्टि द्वारा लागू प्रविष्टि)
देता है $\frac{\dd}{\dd t}\eu^{tA} = A\eu^{tA}$; फैक्टरिंग $ A $
इसके बजाय दाईं ओर देता है $\eu^{tA}A $. पुनरावृत्त: $ C^\infty $.

\textbf{18.} $ A(t) = \eu^{tA}$ संतुष्ट $ A'(t) = A\,A(t)$:
लिउविले का सूत्र (प्रश्न 13) स्थिरांक के साथ $ M = A $ देता है
$\det\eu^{tA} = \eu^{t\operatorname{tr}A}$ (कीमत $1$ पर $ t =
0$); पर $ t = 1$, $\det\eu^A = \eu^{\operatorname{tr}A}$.
जाँच: $ n = 1$ स्वयं घातांक है; एक शून्यशक्तिशाली $ A $ है
$\operatorname{tr}A = 0$ और $\eu^A $ का एकरूपता सिद्ध
$1$; $\operatorname{tr}A = 0$ देता है $\det\eu^A = 1$: द
ट्रेस-शून्य मैट्रिक्स भेजे जाते हैं $ SL_n(\R)$.

\textbf{19.} $\eu^H - I - H = \sum_{k\geq2}H^k/k!$, का आदर्श
$\leq \vertiii H^2\eu^{\vertiii H} = O(\vertiii H^2)$:
$\dd(\exp)_0 = \mathrm{id}$, उलटा। इसके अतिरिक्त $\exp $ है
$ C^1$: प्रश्न 5 द्वारा, द \omterm{def:b2:diffcalc:differential}{अंतर} उम्मीदवार $ H \mapsto
\sum_k\frac1{k!}\sum_iA^iHA^{k-1-i}$ एक है \omterm{def:b2:funcseq:series}{सामान्यतः} अभिसारी
रैखिक मानचित्रों की श्रृंखला निर्भर करती है लगातार पर $ A $ (सीमा
$\vertiii A^{k-1}/(k-1)!$ गेंदों पर), तो आंशिक मौजूद हैं और
हैं \omterm{def:b2:metric:continuity}{निरंतर} (\cref{thm:b2:diffcalc:c1} और श्रृंखला
\omterm{def:b2:structures:sn}{स्थानांतरण} प्रमेय)। व्युत्क्रम फलन प्रमेय
(\cref{thm:b2:diffcalc:inverse}) पर लागू होता है $0$: $\exp $ एक है
$ C^1$ के एक पड़ोस की भिन्नता $0$ एक पर
का पड़ोस $ I $ ---मैट्रिसेस निकट $ I $ लघुगणक हैं.

\textbf{20.} के लिए \omterm{def:b2:quadratic:adjoint}{सममित} $ S = PDP^{\mathsf T}$ (स्पेक्ट्रल
प्रमेय): $\eu^S = P\eu^DP^{\mathsf T}$ है \omterm{def:b2:quadratic:adjoint}{सममित} साथ
eigenvalues $\eu^{\lambda_i} > 0$: सकारात्मक निश्चित.
विशेषणात्मकता: एक सकारात्मक निश्चितता $ Q =
P\operatorname{diag}(\mu_i)P^{\mathsf T}$ ($\mu_i > 0$) है
$\eu^S $ के लिए $ S = P\operatorname{diag}(\ln\mu_i)P^{\mathsf T}$.
इंजेक्शन: $\eu^S $ इसका निर्धारण करता है eigenspaces, जो हैं
\omterm{def:b2:series:def}{बिल्कुल} वही $ S $ (सभी के ऊपर \omterm{def:b2:reduction:eigen}{eigenspace} का $ S $ के लिए $\lambda $,
$\eu^S $ के समान एक्ट करें $\eu^\lambda $; विशिष्ट $\lambda $ देना
विशिष्ट $\eu^\lambda $), और ले रहा है $\ln $ की eigenvalues
ठीक $ S $. इसलिए $\exp $ से एक आक्षेप है \omterm{def:b2:quadratic:adjoint}{सममित} matrices
सकारात्मक निश्चित लोगों पर।

\textbf{21.} $ F(A + H) = A^{\mathsf T}A + A^{\mathsf T}H +
H^{\mathsf T}A + H^{\mathsf T}H $: $\dd F_A(H) =
A^{\mathsf T}H + H^{\mathsf T}A $ (में मान $ S_n $; गलती
$ O(\vertiii H^2)$). पर $ A \in O_n $ और के लिए $ S \in S_n $, विकल्प $ H = \frac12AS $ देता है
\[
A^{\mathsf T}\cdot\tfrac12AS + \tfrac12(AS)^{\mathsf T}A
= \tfrac12 S + \tfrac12 S^{\mathsf T} = S :
\]
विशेषण. $ O_n = F^{-1}(I)$ का एक सहज स्तर सेट है
आयाम $ n^2 - \dim S_n = \frac{n(n-1)}2$.

\textbf{22.} फर्क $ A(t)^{\mathsf T}A(t) = I $ पर $ t
= 0$ (साथ $ A(0) = I $): $ A'(0)^{\mathsf T} + A'(0) = 0$:
असिमेट्रिक इसके विपरीत, के लिए $ K^{\mathsf T} = -K $:
$(\eu^{tK})^{\mathsf T}\eu^{tK} = \eu^{tK^{\mathsf T}}
\eu^{tK} = \eu^{-tK}\eu^{tK} = I $ (\omterm{def:b2:linalg:transpose}{खिसकाना} शृंखला
अवधिवार; घातांक कम्यूट करते हैं): वक्र अंदर रहता है $ O_n $,
वेग के साथ $ K $ पर $ t = 0$. पर स्पर्शरेखा स्थान $ I $ $=$ द
अपेक्षित आयाम के एंटीसिमेट्रिक मैट्रिक्स
$\frac{n(n-1)}2$.

\textbf{23.} $\operatorname{tr}K = 0$ एंटीसिमेट्रिक के लिए $ K $,
इसलिए $\det\eu^K = \eu^0 = 1$ (प्रश्न 18): घातांक
में उतरता है $ SO_n $. के लिए $ n = 2$: $ J^2 = -I $, इसलिए $ J^{2m} =
(-1)^mI $, $ J^{2m+1} = (-1)^mJ $, और
\[
\eu^{\theta J}
= \Bigl(\sum_m\frac{(-1)^m\theta^{2m}}{(2m)!}\Bigr)I
+ \Bigl(\sum_m\frac{(-1)^m\theta^{2m+1}}{(2m+1)!}\Bigr)J
= \cos\theta\,I + \sin\theta\,J ,
\]
द्वारा घूर्णन $\theta $: साइन और कोसाइन श्रृंखला लाइव
के अंदर \omterm{ex:b2:nvs:matrixexp}{मैट्रिक्स घातांक}.

\textbf{24.} उलटे के लिए $ A, B $: $\operatorname{adj}(AB) =
\det(AB)(AB)^{-1} = \det(B)\det(A)B^{-1}A^{-1} =
\operatorname{adj}(B)\operatorname{adj}(A)$. के दोनों तरफ
पहचान बहुपद हैं (इसलिए \omterm{def:b2:metric:continuity}{निरंतर}) प्रविष्टियों के मानचित्र
का $(A, B)$; वे सघन उपसमुच्चय पर सहमत हैं $ GL_n\times GL_n $
का $\mathcal M_n\times\mathcal M_n $ (प्रश्न 3: अनुमानित
प्रत्येक कारक), इसलिए वे हर जगह सहमत हैं।

\textbf{25.} (i) जिस भाग पर मैं भागा पूर्णता परिमित-आयामी का
मानक स्थान (\omterm{def:b2:series:def}{बिल्कुल} \omterm{def:b2:integration:improper}{अभिसरण} श्रृंखला \omterm{def:b2:integration:improper}{अभिसरण}), भाग IV
प्रश्न 20, भाग V के लिए उसी प्लस वर्णक्रमीय प्रमेय पर
व्युत्क्रम फलन प्रमेय पर स्थानीय लघुगणक। (ii) द
अंतर-समीकरण अध्याय लिउविले के सूत्र पर निर्भर करता है
(प्रश्न 13) रैखिक प्रणालियों के व्रोनस्कियन के लिए, और आगे
$\frac{\dd}{\dd t}\eu^{tA} = A\eu^{tA}$ (प्रश्न 17), जो है
वह कथन $\eu^{tA}$ हल करती है $ X' = AX $. (iii)
$\dd(\det)_I = \operatorname{tr}$ कहते हैं निशान है
आयतन परिवर्तन की असीम दर, और $\det\eu^A =
\eu^{\operatorname{tr}A}$ उस कथन को विश्व स्तर पर एकीकृत करता है।
(iv) वर्ष 3 खंड में संरचनाओं के नाम बताए गए हैं: $ O_n $ और
$ SL_n(\R)$ झूठ समूह हैं, उनके स्पर्शरेखा स्थान हैं $ I $
(एंटीसिमेट्रिक और ट्रेस-शून्य मैट्रिक्स) झूठ बीजगणित हैं, और
$\exp$ उनके बीच का पुल है.
\end{solution}
