\section*{अध्याय \ref{ch:b2:series} --- अनुक्रम और शृंखला}

\begin{solution}{exo:b2:series:1}
$\sum\frac{\cos n}{n}$: हाबिल का परीक्षण $ a_n = \frac1n $ और $ b_n =
\cos n $, जिसका आंशिक योग परिबद्ध है (ज्यामितीय का वास्तविक भाग)।
योग, जैसा कि \cref{ex:b2:series:sinn}): अभिसारी (नहीं \omterm{def:b2:series:def}{बिल्कुल},
समान द्वारा $\cos^2$ चाल)।

$\sum \frac{(-1)^n}{\ln n}$ ($ n \geq 2$): वैकल्पिक परीक्षण,
$\frac{1}{\ln n} \downarrow 0$: अभिसरण; नहीं \omterm{def:b2:series:def}{बिल्कुल} ($\ln n
\leq n $).

$\sum \frac{(-1)^n}{n^{3/4} + \cos n}$: बढ़ाना,
\[
\frac{(-1)^n}{n^{3/4} + \cos n}
= \frac{(-1)^n}{n^{3/4}}\cdot
\frac{1}{1 + \frac{\cos n}{n^{3/4}}}
= \frac{(-1)^n}{n^{3/4}}
- \frac{(-1)^n\cos n}{n^{3/2}} + O\Bigl(\frac{1}{n^{9/4}}\Bigr).
\]
पहली श्रृंखला: प्रत्यावर्ती, अभिसरण। दूसरा: \omterm{def:b2:series:def}{बिल्कुल}
अभिसरण ($\frac{1}{n^{3/2}}$ पैमाना)। तीसरा: \omterm{def:b2:series:def}{बिल्कुल}
अभिसारी। कुल: अभिसरण.
\end{solution}

\begin{solution}{exo:b2:series:2}
\omterm{thm:b2:series:fubini}{कॉची उत्पाद} का $\sum_{m\geq0} z^m $ स्वयं के साथ (दोनों \omterm{def:b2:series:def}{बिल्कुल}
के लिए अभिसरण $\abs z < 1$): विकर्ण गुणांक है $ c_k =
\sum_{m=0}^{k} 1 = k + 1$, इसलिए
\[
\frac{1}{(1-z)^2} = \sum_{k\geq0} (k+1)z^k .
\]
से गुणा करना $ z $ और पुनः अनुक्रमणिका: $\sum_{n \geq 1} n z^n =
\frac{z}{(1-z)^2}$.
\end{solution}

\begin{solution}{exo:b2:series:3}
होने देना $ B_n = \sum_{k \leq n} b_k \to B $. \omterm{thm:b2:series:abel}{हाबिल सारांश} साथ $ a_k = k $:
\[
\sum_{k=1}^{n} k\,b_k = n B_n - \sum_{k=1}^{n-1} B_k
\quad\Longrightarrow\quad
\frac1n \sum_{k=1}^{n} k b_k = B_n - \frac{1}{n}\sum_{k=1}^{n-1}
B_k .
\]
सेस\`एआरओ अभिसरण का मतलब है $(B_k)$ अपनी सीमा तक प्रवृत्त होते हैं $ B $
(वर्ष 1 खंड), तो दाहिनी ओर झुकता है $ B - B = 0$.
\end{solution}

\begin{solution}{exo:b2:series:4}
अगर $\theta \in \pi\Z $: सभी शब्द लुप्त हो जाते हैं --- तुच्छ रूप से अभिसरण।
मान लीजिए $\theta \notin \pi\Z $.

$\alpha > 1$: \omterm{def:b2:series:def}{बिल्कुल} अभिसरण (द्वारा वर्चस्व) $ n^{-\alpha}$).

$0 < \alpha \leq 1$: हाबिल का परीक्षण लागू होता है ($ a_n = n^{-\alpha}
\downarrow 0$; का आंशिक योग $\sin n\theta $ से घिरा
$\frac{1}{\abs{\sin(\theta/2)}}$, ज्यामितीय योग): अभिसरण। नहीं
\omterm{def:b2:series:def}{बिल्कुल}: $\abs{\sin n\theta} \geq \sin^2 n\theta = \frac{1 -
\cos 2n\theta}{2}$, और $\sum \frac{1 - \cos 2n\theta}{2n^\alpha}$
विचलन ($\sum n^{-\alpha}$ विचलन; $\sum
\frac{\cos 2n\theta}{n^\alpha}$ जब हाबिल द्वारा अभिसरण होता है $2\theta
\notin 2\pi\Z $; बहिष्कृत मामला $2\theta \in 2\pi\Z $ मतलब
$\theta \in \pi\Z $, पहले से ही संभाला हुआ)। अर्ध-अभिसारी।
\end{solution}

\begin{solution}{exo:b2:series:5}
होने देना $ p_1, p_2, \dots $ के अऋणात्मक पद हों $(u_n)$ क्रम में,
$ q_1, q_2, \dots $ नकारात्मक वाले. दोनों $\sum p_k $ और $\sum
q_k $ विचलन: यदि उनमें से एक अभिसरण हो जाता है, तो दूसरा बराबर हो जाएगा
अभिसारी $\sum u_n $ इसे घटाएं, इसलिए अभिसरण भी करें --- और फिर
$\sum \abs{u_n} = \sum p_k - \sum q_k $ एकाग्र हो जायेंगे, विरोधाभासी
परिकल्पना. भी $ u_n \to 0$ ($\sum u_n $ अभिसरण)।

लालची पुनर्व्यवस्था: सकारात्मक शर्तें लें $ p_1, p_2, \dots $ जब तक
रनिंग टोटल पहले से अधिक है $\ell $ (संभव: $\sum p_k =
+\infty $); तब तक नकारात्मक पद जब तक कि कुल पहले नीचे न आ जाए
$\ell $ (संभव: $\sum q_k = -\infty $); हमेशा के लिए दोहराएँ (प्रत्येक
चरण परिमित है, और प्रत्येक शब्द का प्रयोग \omterm{def:b2:series:def}{बिल्कुल} एक बार किया जाता है: एक वास्तविक
पुनर्व्यवस्था)। प्रत्येक स्विच के बाद, रनिंग से दूरी
कुल करने के लिए $\ell $ अधिक से अधिक अंतिम शब्द का प्रयोग किया गया है; चूँकि जिन शब्दों का प्रयोग किया गया है
पर $ m $-वें स्विच में इंडेक्स है $\to \infty $, और $ u_n \to 0$,
चल रहे योग एकत्रित होते हैं $\ell $.
\end{solution}

\begin{solution}{exo:b2:series:6}
संक्षेपण: का परिमित आंशिक योग
$\frac{\abs x^{m+n}}{m!n!}$ से बंधे हैं $\bigl(\sum_m
\frac{\abs x^m}{m!}\bigr)^2 = \eu^{2\abs x}$. विकर्ण समूहन
(\cref{thm:b2:series:fubini}):
\[
(\eu^{x})^2 = \sum_{m,n} \frac{x^{m+n}}{m!\,n!}
= \sum_{k=0}^{\infty} x^k \sum_{m+n=k} \frac{1}{m!\,n!}
= \sum_k \frac{x^k}{k!}\sum_{m=0}^{k}\binom km
= \sum_k \frac{(2x)^k}{k!} = \eu^{2x} .
\]
\end{solution}

\begin{solution}{exo:b2:series:7}
संक्षेपण: से घिरा $\zeta(2)^2$ एक उत्पाद परिवार के रूप में
(\cref{thm:b2:series:fubini}कॉची-उत्पाद तर्क)। वो नक्शा
$(q, a, b) \mapsto (qa, qb)$, त्रिगुणों से $\gcd(a, b) = 1$
जोड़े को $(m, n)$, एक आक्षेप (सेट) है $ q = \gcd(m,n)$). समूहीकरण
द सारांशित परिवार तदनुसार (सूचकांक सेट का एक विभाजन ---
के लिए वैध \omterm{def:b2:series:summable}{संक्षेपणीय} परिवारों द्वारा
\cref{thm:b2:series:rearrangement}/\cref{thm:b2:series:fubini}
में विभाजन के लिए लागू किया गया \omterm{def:b2:structures:countable}{गणनीय} कई वर्ग):
\[
\zeta(2)^2 = \sum_{q} \sum_{\gcd(a,b)=1} \frac{1}{q^4 a^2 b^2}
= \zeta(4)\, S,
\qquad\text{so}\qquad
S = \frac{\zeta(2)^2}{\zeta(4)} .
\]
(मूल्यों के साथ $\zeta(2) = \frac{\pi^2}{6}$,
$\zeta(4) = \frac{\pi^4}{90}$ से
\cref{ch:b2:fourier}: $ S = \frac{5}{2}$.)
\end{solution}

\begin{solution}{exo:b2:series:8}
होने देना $ A = \sum a_n $ (निरपेक्ष), $ B_m = \sum_{k\leq m} b_k \to B $,
से घिरा $ M $. तब
\[
C_N = \sum_{k=0}^{N} c_k = \sum_{n=0}^{N} a_n B_{N-n}
\]
(के सूचकांक द्वारा एकत्र करें $ a $). लिखना
\[
C_N - AB = \sum_{n=0}^{N} a_n (B_{N-n} - B) - B\sum_{n > N} a_n .
\]
अंतिम पद की प्रवृत्ति होती है $0$. योग को विभाजित करें $ n = \lfloor N/2
\rfloor $: के लिए $ n \leq N/2$, $ N - n \geq N/2$, इसलिए $\abs{B_{N-n} - B}
\leq \varepsilon_N := \sup_{m \geq N/2}\abs{B_m - B} \to 0$, और
यह भाग है $\leq \varepsilon_N \sum\abs{a_n}$; के लिए $ n > N/2$,
$\abs{B_{N-n} - B} \leq 2M $, और यह हिस्सा है $\leq 2M \sum_{n >
N/2} \abs{a_n} \to 0$. इस तरह $ C_N \to AB $.
\end{solution}

\begin{solution}{exo:b2:series:9}
लेना $ u_n = 2^{-n} e_n $ ($ e_n $ एकल के साथ अनुक्रम $1$ स्थिति में $ n $). तब $\sum \norm{u_n}_\infty = \sum 2^{-n} <
\infty $: \omterm{def:b2:series:def}{बिल्कुल} अभिसारी। लेकिन आंशिक रकम $ S_N =
(1, \tfrac12, \dots, 2^{-N}, 0, \dots)$ में जुटना होगा
क्रम $(2^{-n})_n $, जो है \emph{नहीं} अंततः शून्य:
बाहर $ E $. अंदर $ E $, $(S_N)$ बिना किसी सीमा के कॉची है ($\norm{S_N
- x}_\infty \geq 2^{-N-1}$ किसी भी अंततः-शून्य की मदद करने में विफल रहता है $ x $:
किसी के लिए $ x \in E $ पद से परे लुप्त हो जाना $ K $, $\norm{S_N - x} \geq
2^{-K-1}$ के लिए $ N > K $): श्रृंखला में अभिसरण नहीं होता है $ E $.
संपूर्णता बिलकुल वही है \cref{thm:b2:nvs:absoluteconvergence}
जरूरत है.
\end{solution}

\begin{solution}{exo:b2:series:10}
दोनों $\arctan(n+1) - \arctan n $ और
$\arctan\frac{1}{n^2+n+1}$ रिहायश $\intoo{0}{\frac\pi2}$, और
स्पर्शरेखा जोड़ सूत्र देता है
\[
\tan\bigl(\arctan(n{+}1) - \arctan n\bigr)
= \frac{(n+1) - n}{1 + n(n+1)} = \frac{1}{n^2 + n + 1} :
\]
एक अंतराल में समान स्पर्शरेखाएँ जहाँ $\tan $ इंजेक्शन है, इसलिए
पहचान रखती है. टेलीस्कोपिंग,
\[
\sum_{n=1}^{N}\arctan\frac{1}{n^2+n+1}
= \arctan(N{+}1) - \arctan 1 \xrightarrow[N\to\infty]{}
\frac\pi2 - \frac\pi4 = \frac\pi4 .
\]

\end{solution}

\begin{solution}{exo:b2:series:11}
पहला: $\sqrt{n+1} - \sqrt n = \frac{1}{\sqrt{n+1} + \sqrt n}
\sim \frac{1}{2\sqrt n}$, तो शर्तें हैं $\sim
2^{-\alpha}n^{-\alpha/2}$: अभिसरण यदि $\frac\alpha2 > 1$,
यानी\ $\alpha > 2$. दूसरा: $1 - \cos\frac1n \sim
\frac{1}{2n^2}$: अभिसरण। तीसरा: $\bigl(1 + \frac1n\bigr)^n =
\eu^{\,n\ln(1 + 1/n)} = \eu^{\,1 - \frac1{2n} + O(n^{-2})} =
\eu\bigl(1 - \frac{1}{2n} + O(n^{-2})\bigr)$, इसलिए
\[
\eu - \Bigl(1 + \frac1n\Bigr)^{\!n} \sim \frac{\eu}{2n} :
\]
एक हार्मोनिक गुणक के समतुल्य सकारात्मक शब्द: विचलन।
\end{solution}

\begin{solution}{exo:b2:series:12}
एकरसता से, $ n\,a_{2n} \leq a_{n+1} + a_{n+2} + \dots +
a_{2n} = S_{2n} - S_n \to 0$ (के लिए कॉची मानदंड
अभिसरण श्रृंखला)। इस तरह $2n\,a_{2n} \to 0$, और $(2n{+}1)\,
a_{2n+1} \leq (2n{+}1)a_{2n} = \frac{2n+1}{2n}\,(2n\,a_{2n}) \to
0$: के दोनों अनुवर्ती $(na_n)$ प्रवृत्त $0$, इसलिए $ na_n \to 0$.

\emph{बातचीत विफल:} $ a_n = \frac1{n\ln n}$ सकारात्मक है
के साथ घट रहा है $ na_n = \frac1{\ln n} \to 0$, अभी तक $\sum a_n $
विचलन (बर्ट्रेंड फ्रंटियर, \cref{pb:b2:series:1}, प्रश्न
18). \emph{एकरसता आवश्यक:} होने देना $ a_n = \frac1n $ कब $ n $
की एक शक्ति है $2$ और $ a_n = 2^{-n}$ अन्यथा: $\sum a_n \leq
\sum_k 2^{-k} + \sum_n 2^{-n} < \infty $, लेकिन $ na_n = 1$ साथ में
की शक्तियां $2$: $ na_n \not\to 0$.
\end{solution}

\begin{solution}{pb:b2:series:1}
\textbf{1.} प्रेरण चालू $ m $: के लिए $ m = 0$ दोनों पक्ष हैं $1$;
कदम कई गुना बढ़ जाता है $\cos\theta + \iu\sin\theta $ और का उपयोग करता है
अतिरिक्त सूत्र $\cos(m\theta + \theta) =
\cos m\theta\cos\theta - \sin m\theta\sin\theta $, $\sin(m\theta
+ \theta) = \sin m\theta\cos\theta + \cos m\theta\sin\theta $.
इसके बजाय द्विपद प्रमेय द्वारा विस्तार $ m = 2n+1$ और
काल्पनिक भाग (की विषम शक्तियाँ) एकत्रित करना
$\iu\sin\theta $, साथ $\iu^{2j+1} = (-1)^j\iu $):
\[
\sin\bigl((2n{+}1)\theta\bigr) = \sum_{j=0}^{n}(-1)^j
\binom{2n+1}{2j+1}\cos^{2(n-j)}\theta\,\sin^{2j+1}\theta .
\]

\textbf{2.} पर $\intoo0{\frac\pi2}$, $\sin\theta \neq 0$: कारक
$\sin^{2n+1}\theta $ प्रत्येक पद से, छोड़कर
$\bigl(\frac{\cos^2\theta}{\sin^2\theta}\bigr)^{n-j} =
(\cot^2\theta)^{n-j}$: के साथ प्रदर्शित पहचान $ P_n(x) =
\sum_j(-1)^j\binom{2n+1}{2j+1}x^{n-j}$. इसकी डिग्री-$ n $
गुणांक है $ j = 0$'एस $\binom{2n+1}{1} = 2n + 1 \neq 0$.

\textbf{3.} पर $\theta_k = \frac{k\pi}{2n+1}$:
$\sin\bigl((2n{+}1)\theta_k\bigr) = \sin k\pi = 0$ जबकि
$\sin^{2n+1}\theta_k \neq 0$, इसलिए $ P_n(\cot^2\theta_k) = 0$.
$\theta_k $ में सख्ती से वृद्धि करें $\intoo0{\frac\pi2}$, कहाँ
$\cot^2$ सख्ती से घट रहा है: मूल्य $ x_k =
\cot^2\theta_k $ जोड़ीवार भिन्न हैं --- $ n $ की विशिष्ट जड़ें
एक डिग्री-$ n $ बहुपद, इसलिए ये सभी।

\textbf{4.} विएटा: जड़ों का योग शून्य से अनुपात है
द $ x^{n-1}$- और $ x^n $-गुणांक:
\[
\sum_{k=1}^{n}\cot^2\theta_k =
\frac{\binom{2n+1}{3}}{\binom{2n+1}{1}}
= \frac{(2n+1)(2n)(2n-1)/6}{2n+1} = \frac{n(2n-1)}{3}.
\]

\textbf{5.} $\frac{1}{\sin^2\theta} = 1 + \cot^2\theta $:
संक्षेप, $ n + \frac{n(2n-1)}3 = \frac{3n + 2n^2 - n}{3} =
\frac{2n(n+1)}{3}$.

\textbf{6.} पर $\intoo{0}{\frac\pi2}$: $\sin\theta < \theta <
\tan\theta $ (वर्ष 1 खंड). व्युत्क्रम लेना:
$\cot\theta < \frac1\theta < \frac1{\sin\theta}$, और चुकता करना
(सभी सकारात्मक) देता है $\cot^2\theta < \frac1{\theta^2} <
\frac1{\sin^2\theta}$.

\textbf{7.} योग प्रश्न 6 पर $\theta = \theta_k $ ऊपर $ k \leq
n $, प्रश्न 4 और 5 का उपयोग करते हुए, और $\frac1{\theta_k^2} =
\frac{(2n+1)^2}{k^2\pi^2}$:
\[
\frac{n(2n-1)}3 < \frac{(2n+1)^2}{\pi^2}\sum_{k=1}^n\frac1{k^2}
< \frac{2n(n+1)}3 .
\]

\textbf{8.} गुणा करके $\frac{\pi^2}{(2n+1)^2}$:
\[
\frac{\pi^2}{3}\cdot\frac{n(2n-1)}{(2n+1)^2}
< \sum_{k=1}^{n}\frac1{k^2} <
\frac{\pi^2}{3}\cdot\frac{2n(n+1)}{(2n+1)^2}.
\]
दोनों सीमाएँ प्रवृत्त होती हैं $\frac{\pi^2}3\cdot\frac12 = \frac{\pi^2}6$
(तर्कसंगत भिन्नों की प्रवृत्ति होती है $\frac12$). आंशिक रकम
वृद्धि, इसलिए वे एकत्रित होते हैं, और निचोड़ देते हैं $\zeta(2) =
\frac{\pi^2}6$: यूलर का प्रमेय, कॉची के प्रमाण द्वारा।

\textbf{9.} आंशिक रकम बढ़ जाती है $\zeta(2) =
\frac{\pi^2}6$, इसलिए $0 \leq \frac{\pi^2}6 - \sum_{k\leq n}k^{-2}$;
और प्रश्न 8 की निचली सीमा देती है
\[
\frac{\pi^2}6 - \sum_{k\leq n}\frac1{k^2}
\leq \frac{\pi^2}6 - \frac{\pi^2}3\cdot\frac{n(2n-1)}{(2n+1)^2}
= \frac{\pi^2}6\cdot\frac{(2n+1)^2 - (4n^2 -
2n)}{(2n+1)^2}
= \frac{\pi^2}6\cdot\frac{6n + 1}{(2n+1)^2}
= O\Bigl(\frac1n\Bigr),
\]
सटीक पूँछ से मेल खाता हुआ $\frac1n - \frac1{2n^2} + O(n^{-3})$ का
\cref{exo:b2:comparison:11}.

\textbf{10.} का दूसरा प्राथमिक सममितीय कार्य
जड़ें है $\sigma_2 = \frac{\binom{2n+1}5}{\binom{2n+1}1} =
\frac{(2n)(2n-1)(2n-2)(2n-3)}{120}$, इसलिए
\[
\sum_k\cot^4\theta_k = \sigma_1^2 - 2\sigma_2
= \Bigl(\frac{n(2n-1)}3\Bigr)^2 -
\frac{2n(2n-1)(2n-2)(2n-3)}{60}
\sim \frac{4n^4}9 - \frac{4n^4}{15} = \frac{8n^4}{45}.
\]
फैलाएंगे $\cot^4\theta < \theta^{-4} < (1 + \cot^2\theta)^2 =
1 + 2\cot^2\theta + \cot^4\theta $ और योग: दोनों बाहरी योग
हैं $\frac{8n^4}{45}(1 + o(1))$ (जोड़ा गया $ n + 2\sigma_1 =
O(n^2)$ नगण्य है), जबकि मध्य है
$\frac{(2n+1)^4}{\pi^4}\sum_{k\leq n}k^{-4}$. इस तरह
\[
\sum_{k\leq n}\frac1{k^4} \longrightarrow
\pi^4\cdot\frac{8/45}{16} = \frac{\pi^4}{90}.
\]

\textbf{11.} विभाजन $\zeta(2)$ समता पर: $\sum_{\text{even}}
= \sum_j\frac{1}{(2j)^2} = \frac14\zeta(2) = \frac{\pi^2}{24}$,
इसलिए $\sum_{\text{odd}} = \zeta(2) - \frac{\pi^2}{24} =
\frac{\pi^2}8$. वैकल्पिक: $\sum_k\frac{(-1)^{k-1}}{k^2} =
\sum_{\text{odd}} - \sum_{\text{even}} = \frac{\pi^2}8 -
\frac{\pi^2}{24} = \frac{\pi^2}{12}$ (पूर्ण अभिसरण
पुनर्समूहन को उचित ठहराता है,
\cref{thm:b2:series:rearrangement}).

\textbf{12.} $ S = \frac{\zeta(2)^2}{\zeta(4)} =
\frac{(\pi^2/6)^2}{\pi^4/90} = \frac{90}{36} = \frac52$.
अनुमानी: पहचान $\zeta(2)^2 = \zeta(4)S $ का
\cref{exo:b2:series:7} कहते हैं कि बाहर खींच रहा हूँ $\gcd $
जोड़ियों को सहअभाज्य जोड़ियों में पुनर्सामान्यीकृत करता है; पारस्परिक
$\frac1{\zeta(2)} = \frac6{\pi^2} \approx 0.608$ प्राकृतिक है
सभी जोड़ियों के बीच सहअभाज्य जोड़े के घनत्व के लिए उम्मीदवार ---
के बारे में एक बयान $\lim_N \frac{1}{N^2}\#\{(m,n) \leq N :
\gcd = 1\}$ जिसका ईमानदार प्रमाण (त्रुटि शर्तों के साथ) है
वर्ष 3 खंड.

\textbf{13.} प्रत्यक्ष योग में त्रुटि है $\sim \frac1n $: छह
अंकों की आवश्यकता है $10^6$ शर्तें। सही किया गया योग
$\sum_{k\leq n}k^{-2} + \frac1n - \frac1{2n^2}$ त्रुटि है
$ O(n^{-3})$: पर $ n = 100$ यह बराबर है $1.6449339\dots $ ख़िलाफ़
$\frac{\pi^2}6 = 1.6449341\dots $ --- गलती $1.7\cdot10^{-7}$,
सौ पदों में से सात अंक। स्पर्शोन्मुख सुधार हरा
परिमाण के चार क्रमों से कच्चा धैर्य।

\textbf{14.} $ n = 1$: $ P_1(x) = 3x - 1$, जड़ $\frac13$, सचमुच $\cot^2\frac\pi3 = \bigl(\frac1{\sqrt3}\bigr)^2 =
\frac13 = \frac{1\cdot1}3$. $ n = 2$: सूत्र भविष्यवाणी करता है
$\frac{2\cdot3}3 = 2$; साथ $\cos\frac\pi5 = \frac{1 +
\sqrt5}{4}$, कोई गणना करता है $\cot^2 36^\circ \approx 1.894$ और
$\cot^2 72^\circ \approx 0.106$: जोड़ $2.000$.

\textbf{15.} संख्या $\tan^2\theta_k = \frac1{x_k}$ हैं
की जड़ें $ Q(x) = x^nP_n\bigl(\frac1x\bigr) =
\sum_{j=0}^n(-1)^j\binom{2n+1}{2j+1}x^j $ (द $ x_k $ हैं
शून्येतर)। विएटा चालू $ Q $: अग्रणी गुणांक है $(-1)^n $
(अवधि $ j = n $), अगला है $(-1)^{n-1}\binom{2n+1}{2n-1} =
(-1)^{n-1}\binom{2n+1}{2}$, इसलिए
\[
\sum_{k=1}^n\tan^2\theta_k =
-\frac{(-1)^{n-1}\binom{2n+1}2}{(-1)^n} = \binom{2n+1}2\cdot
\frac{2}{2n+1}\cdot\frac{2n+1}{2}
= n(2n+1).
\]
जाँच करना $ n = 1$: $\tan^2\frac\pi3 = 3 = 1\cdot3$.

\textbf{16.} होने देना $\gamma = \frac{1+\alpha}2 \in
\intoo{1}{\alpha}$. तब $\frac{n^{-\alpha}(\ln
n)^{-\beta}}{n^{-\gamma}} = n^{\gamma - \alpha}(\ln n)^{-\beta}
\to 0$ (की एक नकारात्मक शक्ति $ n $ की किसी भी शक्ति को मात देता है $\ln n $), तो
अंततः शर्तें हैं $\leq n^{-\gamma}$ साथ $\gamma > 1$:
रीमैन श्रृंखला के साथ तुलना करके अभिसरण।

\textbf{17.} होने देना $\gamma = \frac{1+\alpha}2 \in
\intoo{\alpha}{1}$: अब $\frac{n^{-\gamma}}{n^{-\alpha}(\ln
n)^{-\beta}} = n^{\alpha-\gamma}(\ln n)^{\beta} \to 0$, तो
अंततः शर्तें हैं $\geq n^{-\gamma}$ साथ $\gamma < 1$:
विचलन.

\textbf{18.} $ f(t) = \frac1{t(\ln t)^\beta}$ सकारात्मक है,
\omterm{def:b2:metric:continuity}{निरंतर}, और बड़े पैमाने पर घट रहा है $ t $ (इसका लघुगणक है
व्युत्पन्न $-\frac1t\bigl(1 + \frac{\beta}{\ln t}\bigr) < 0$
अंततः)। प्रतिव्युत्पन्न: के लिए $\beta \neq 1$,
$\int^x f = \frac{(\ln x)^{1-\beta}}{1-\beta} + \text{const}$,
जिसकी एक सीमित सीमा है iff $\beta > 1$; के लिए $\beta = 1$,
$\int^x f = \ln\ln x \to \infty $. द्वारा
\cref{thm:b2:comparison:seriesintegral}, श्रृंखला और
अभिन्न अपनी प्रकृति साझा करें: अभिसरण आईएफएफ $\beta > 1$.

\textbf{19.} वही परीक्षण: $\frac{\dd}{\dd t}\ln\ln\ln t =
\frac{1}{t\ln t\,\ln\ln t}$, और $\ln\ln\ln t \to \infty $:
विचलन. और $\frac{\dd}{\dd t}\Bigl(-\frac1{\ln\ln t}\Bigr)
= \frac{1}{t\ln t\,(\ln\ln t)^2}$ साथ $-\frac1{\ln\ln t} \to
0$: अभिसरण.

\textbf{20.} पहला: $ n^{1/\ln n} = \eu^{\ln n/\ln n} = \eu $, तो
शर्तें बिलकुल वैसी हैं $\frac{1}{\eu\,n}$: का एक गुणज
हार्मोनिक श्रृंखला, \emph{विभिन्न} --- प्रतिपादक $1 +
\frac1{\ln n}$ रेंगता है $1$ बहुत तेज़। दूसरा: $ n^{1/\ln\ln n}
= \eu^{\ln n/\ln\ln n}$, और $\frac{\ln n}{\ln\ln n} \geq
2\ln\ln n $ अंततः, ऐसा $ n^{1/\ln\ln n} \geq (\ln n)^2$: द
शर्तें हैं $\leq \frac1{n(\ln n)^2}$, एक अभिसरण बर्ट्रेंड
शृंखला (प्रश्न 18): \emph{संमिलित}. सीमा गुजरती है
इन दोनों प्रतिपादकों के बीच सख्ती से।

\textbf{21.} $ R_n \downarrow 0$ और
\[
\sqrt{R_n} - \sqrt{R_{n+1}} = \frac{R_n -
R_{n+1}}{\sqrt{R_n} + \sqrt{R_{n+1}}} =
\frac{a_n}{\sqrt{R_n} + \sqrt{R_{n+1}}} \geq
\frac{a_n}{2\sqrt{R_n}},
\]
इसलिए $\sum_n \frac{a_n}{\sqrt{R_n}} \leq 2\sum_n(\sqrt{R_n} -
\sqrt{R_{n+1}}) = 2\sqrt{R_1} < \infty $ (दूरबीन)। अभी तक
$\frac{a_n/\sqrt{R_n}}{a_n} = \frac1{\sqrt{R_n}} \to \infty $:
नई श्रृंखला असीम रूप से बड़ी होते हुए भी एकत्रित होती है। नहीं
अभिसरण श्रृंखला सबसे धीमी है; किसी के विरुद्ध तुलना परीक्षण
निश्चित परिवार कभी पूर्ण नहीं हो सकता।

\textbf{22.} सकारात्मकता कम करने के लिए $(a_n)$, शर्तों को समूहित करें
की लगातार शक्तियों के बीच $2$:
\[
2^{k}a_{2^{k+1}} \leq \sum_{n=2^k}^{2^{k+1}-1} a_n \leq
2^ka_{2^k} .
\]
संक्षेप में कहना $ k $: अगर $\sum 2^ka_{2^k}$ अभिसरण, आंशिक
का योग $\sum a_n $ बंधे हुए हैं (अभिसरण); अगर $\sum a_n $
फिर एकत्रित हो जाता है $\sum_k 2^{k+1}a_{2^{k+1}} \leq 2\sum_n a_n <
\infty $. के लिए $ a_n = \frac1{n(\ln n)^\beta}$: $2^ka_{2^k} =
\frac{1}{(k\ln 2)^\beta}$, और $\sum k^{-\beta}$ यदि अभिसरण करता है
$\beta > 1$: प्रश्न 18 की सीमा फिर से, बिना
अभिन्न।

\textbf{23.} अभिन्न तुलना द्वारा,
\[
\sum_{n > N}\frac{1}{n(\ln n)^2} \geq
\int_{N+1}^{\infty}\frac{\dd t}{t(\ln t)^2} =
\frac{1}{\ln(N+1)},
\]
जो पर $ N = 10^6$ है $\approx 0.0724$: एक लाख शर्तों के बाद
पूंछ अभी भी अधिक है $0.07$ --- शृंखला अभिसरित होती है, लेकिन नहीं
प्रत्यक्ष योग कभी भी अपना योग प्रदर्शित करेगा। के साथ तुलना करें
प्रश्न 13, जहां एक स्पर्शोन्मुख सुधार ने सात खरीदे
सौ पदों से अंक: जानना \emph{कैसे} एक शृंखला
अभिसरण यह जानने से अधिक मूल्यवान है कि यह होता है।

\textbf{24.} $\sum\frac1{n\ln n}$: विचलन ($\alpha = 1$,
$\beta = 1$, प्रश्न 18). $\sum\frac1{n^{1.01}}$: अभिसरण
(रीमैन, $\alpha > 1$). $\sum\frac{(\ln n)^{100}}{n^{1.001}}$:
अभिसरण ($\alpha = 1.001 > 1$, $\beta = -100$, प्रश्न 16).
$\sum\frac1{n\ln n(\ln\ln n)^3}$: अभिसरण (प्रश्न 19)
पैटर्न: प्रतिव्युत्पन्न $-\frac12(\ln\ln t)^{-2}$, परिमित
सीमा).

\textbf{25.} डी मोइवरे लुप्त हो जाने को परिवर्तित करता है $\sin(2n{+}1)
\theta_k $ एक बहुपद के लुप्त होने में
$\cot^2\theta_k $, और विएटा ने सटीक मूल योगों को पढ़ा
अकेले विश्लेषण से केवल अनुमान लगाया जा सकता है (प्रश्न 1--5)। द
निचोड़ की जरूरत है \emph{एकदम सही} मान $\frac{n(2n-1)}3$ और
$\frac{2n(n+1)}3$ दोनों तरफ --- \omterm{def:b2:nvs:equivalent}{समकक्ष} होंगे
प्रश्न पूछा, क्योंकि संपूर्ण बिंदु स्थिर है
$\frac{\pi^2}6$ (प्रश्न 7-8). भाग IV पूरी तरह से चला
\cref{ch:b2:comparison}की शृंखला--अभिन्न तुलना, द
लॉगरिदमिक एंटीडेरिवेटिव्स वर्गीकरण कर रहे हैं (प्रश्न)।
18--19). प्रश्न 21 सार्वभौम के स्वप्न को नष्ट कर देता है
तुलना परीक्षण: प्रत्येक अभिसारी श्रृंखला के नीचे एक और श्रृंखला होती है,
असीम रूप से धीमे --- बर्ट्रेंड के मानचित्र जैसे पैमाने
कभी भी अधिक सूक्ष्मता से लेकिन उस तक कभी नहीं पहुँचना। शिखर सम्मेलन: यूलर का $\zeta(2)
= \frac{\pi^2}6$ इसकी ऊपरी मंजिल के साथ $\zeta(4) =
\frac{\pi^4}{90}$ (प्रश्न 8,10), और बर्ट्रेंड
वर्गीकरण (प्रश्न 16--18); $\zeta(2)$ में लौटता है
\cref{ch:b2:fourier}, जहां पारसेवल की पहचान इसे फिर से साबित करती है
सॉटूथ की फूरियर श्रृंखला से एक पंक्ति में --- एक
स्थिर, दो सभ्यताएँ।
\end{solution}
