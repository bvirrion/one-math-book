\section*{अध्याय \ref{ch:b2:multint} --- लाइन इंटीग्रल और मल्टीपल इंटीग्रल}

\begin{solution}{exo:b2:multint:1}
(ए) खंड $\gamma(t) = (t, t)$, $ t \in [0,1]$:
\[
\int_\gamma y^2\dd x + x\dd y
= \int_0^1 (t^2 + t)\,\dd t = \frac13 + \frac12 = \frac56 .
\]
(बी) परबोला $\gamma(t) = (t, t^2)$:
\[
\int_0^1 \bigl(t^4\cdot 1 + t\cdot 2t\bigr)\dd t
= \frac15 + \frac23 = \frac{13}{15} .
\]
दोनों मान भिन्न हैं, इसलिए अभिन्न पथ-निर्भर है: रूप
है \emph{नहीं} \omterm{def:b2:multint:exact}{एकदम सही} --- लगातार, $ P_y = 2y \neq 1 = Q_x $, तो
यह बंद भी नहीं है.
\end{solution}

\begin{solution}{exo:b2:multint:2}
$ P = 2xy + y^3$, $ Q = x^2 + 3xy^2 + 1$: $ P_y = 2x + 3y^2 = Q_x $,
पर बंद $\R^2$. निम्न को खोजें $ f $ साथ $ f_x = P $: $ f = x^2y + xy^3
+ g(y)$; तब $ f_y = x^2 + 3xy^2 + g'(y) = Q $ बल $ g'(y) = 1$,
कहना $ g(y) = y $. इसलिए
\[
f(x, y) = x^2y + xy^3 + y
\]
एक है \omterm{def:b2:multint:exact}{संभावना} ($\R^2$ तारे के आकार का है, इसलिए a \omterm{def:b2:multint:exact}{संभावना} करना पड़ा
पोंकारे की लेम्मा द्वारा मौजूद --- लेकिन इसे प्रदर्शित करना तेज़ है)। द्वारा
\cref{thm:b2:multint:ftc}, से किसी भी चाप के लिए $(0,0)$ को $(1,2)$:
\[
\int_\gamma\omega = f(1, 2) - f(0, 0) = 2 + 8 + 2 = 12 .
\]
\end{solution}

\begin{solution}{exo:b2:multint:3}
डोमेन है $\{0 \leq x \leq 1,\ x^2 \leq y \leq x\}$.
$ y $-पहला:
\[
\int_0^1\!\int_{x^2}^{x}(x + y)\,\dd y\,\dd x
= \int_0^1\Bigl(x(x - x^2)
+ \frac{x^2 - x^4}{2}\Bigr)\dd x
= \int_0^1\Bigl(\frac{3x^2}{2} - x^3 - \frac{x^4}{2}\Bigr)\dd x
= \frac12 - \frac14 - \frac1{10} = \frac{3}{20} .
\]
$ x $-पहला: ऊंचाई पर टुकड़ा $ y \in [0, 1]$ है $ y \leq x \leq
\sqrt y $, इसलिए
\[
\int_0^1\!\int_{y}^{\sqrt y}(x + y)\,\dd x\,\dd y
= \int_0^1\Bigl(\frac{y - y^2}{2}
+ y(\sqrt y - y)\Bigr)\dd y
= \frac14 - \frac16 + \frac25 - \frac13 = \frac{3}{20} .
\]
\end{solution}

\begin{solution}{exo:b2:multint:4}
\emph{पहला अभिन्न.} डिस्क पर $ D_R $, ध्रुवीय निर्देशांक में:
\[
\iint_{D_R}\frac{\dd x\,\dd y}{(1 + x^2 + y^2)^2}
= \int_0^{2\pi}\!\!\int_0^R
\frac{\rho\,\dd\rho\,\dd\alpha}{(1 + \rho^2)^2}
= 2\pi\Bigl[-\frac{1}{2(1 + \rho^2)}\Bigr]_0^R
= \pi\Bigl(1 - \frac{1}{1 + R^2}\Bigr)
\xrightarrow[R \to \infty]{} \pi .
\]
\emph{दूसरा अभिन्न.} क्वार्टर डिस्क है $0 \leq \alpha \leq
\frac\pi2$, $0 \leq \rho \leq 1$, और $ xy =
\rho^2\cos\alpha\sin\alpha $:
\[
\iint_{D'}xy\,\dd x\,\dd y
= \int_0^{\pi/2}\!\!\cos\alpha\sin\alpha\,\dd\alpha
\int_0^1 \rho^3\,\dd\rho
= \frac12\cdot\frac14 = \frac18 .
\]
\end{solution}

\begin{solution}{exo:b2:multint:5}
द्वारा \cref{cor:b2:multint:area} साथ $ x = \cos^3 t $, $ y = \sin^3 t $:
$ x' = -3\cos^2 t\sin t $, $ y' = 3\sin^2 t\cos t $, इसलिए
\[
xy' - yx' = 3\cos^4 t\sin^2 t + 3\sin^4 t\cos^2 t
= 3\sin^2 t\cos^2 t = \frac{3}{4}\sin^2 2t
= \frac{3}{8}(1 - \cos 4t) .
\]
इस तरह
\[
A = \frac12\int_0^{2\pi}\frac38(1 - \cos 4t)\,\dd t
= \frac{3}{16}\cdot 2\pi = \frac{3\pi}{8} .
\]
(द \omterm{pb:b2:curves:1}{astroid} की यूनिट डिस्क में फिट बैठता है \omterm{def:b2:multint:domain}{क्षेत्र} $\pi $; का तीन आठवां हिस्सा
$\pi $ यह अपने चार-पुच्छल तारे के आकार के लिए प्रशंसनीय है।)
\end{solution}

\begin{solution}{exo:b2:multint:6}
\emph{स्टैकिंग:} प्रत्येक के ऊपर $(x, y)$ यूनिट डिस्क का $ D $, $ z $
से चलती है $ x^2 + y^2$ को $1$:
\[
V = \iint_D \bigl(1 - x^2 - y^2\bigr)\dd x\,\dd y
= \int_0^{2\pi}\!\!\int_0^1 (1 - \rho^2)\rho\,\dd\rho\,\dd\alpha
= 2\pi\Bigl(\frac12 - \frac14\Bigr) = \frac\pi2 .
\]
\emph{टुकड़ा करना:} ऊंचाई पर टुकड़ा $ z \in [0, 1]$ डिस्क है
$ x^2 + y^2 \leq z $, का \omterm{def:b2:multint:domain}{क्षेत्र} $\pi z $:
\[
V = \int_0^1 \pi z\,\dd z = \frac\pi2 .
\]
\end{solution}

\begin{solution}{exo:b2:multint:7}
गोलाकार निर्देशांक में आयतन तत्व है
$ r^2\cos\varphi\,\dd r\,\dd\theta\,\dd\varphi $
(\cref{ex:b2:multint:spherical}), और कोणीय भाग एकीकृत हो जाता है
को $4\pi $ ($2\pi $ से $\theta $, $\int_{-\pi/2}^{\pi/2}\cos = 2$).
शैल आयतन है
\[
\int_a^b 4\pi r^2\,\dd r = \frac43\pi\bigl(b^3 - a^3\bigr).
\]
दूसरे अभिन्न के लिए, इंटीग्रैंड $1/r $ पर ही निर्भर करता है $ r $:
\[
\iiint_B \frac{\dd x\,\dd y\,\dd z}{r}
= \int_0^R 4\pi r^2\cdot\frac1r\,\dd r
= 4\pi\,\frac{R^2}{2} = 2\pi R^2 .
\]
(इंटीग्रैंड मूल पर विस्फोट करता है, लेकिन हानिरहित रूप से: $ r^2/r = r $
है \omterm{def:b2:metric:continuity}{निरंतर} --- गोले पर अभिन्न $\varepsilon \leq r
\leq R $ के रूप में \omterm{def:b2:integration:improper}{अभिसरण} होता है $\varepsilon \to 0$, जो सटीक है
कथन का अर्थ. इस तरह की गणना पहली है
न्यूटन के प्रमेय की ओर कदम बढ़ाएं कि एक सजातीय गेंद आकर्षित होती है
इसके केंद्र पर एक बिंदु द्रव्यमान।)
\end{solution}

\begin{solution}{exo:b2:multint:8}
एकात्म $ g(t; x, y) = xP(tx, ty) + yQ(tx, ty)$ है
$\mathcal{C}^1$ में $(x, y)$, \omterm{def:b2:metric:continuity}{निरंतर} में $ t $, आंशिक के साथ
व्युत्पन्न \omterm{def:b2:metric:continuity}{निरंतर} पर $[0,1] \times U $; के तहत भेदभाव
अभिन्न चिह्न (\cref{ch:b2:integration}, पर लागू किया गया
\omterm{def:b2:metric:compact}{कॉम्पैक्ट} $ t $-अंतराल $[0,1]$, वहां प्रभुत्व स्वत: होता है)
देता है
\[
f_x(x, y)
= \int_0^1 \bigl(P(tx, ty) + tx\,P_x(tx, ty)
+ ty\,Q_x(tx, ty)\bigr)\dd t .
\]
बंदता का उपयोग करना $ Q_x = P_y $:
\[
tx\,P_x(tx, ty) + ty\,P_y(tx, ty)
= t\,\frac{\dd}{\dd t}\bigl[P(tx, ty)\bigr] ,
\]
तो इंटीग्रैंड है $ P(tx, ty) + t\frac{\dd}{\dd t}P(tx, ty) =
\frac{\dd}{\dd t}\bigl[t\,P(tx, ty)\bigr]$ और
\[
f_x(x, y) = \Bigl[t\,P(tx, ty)\Bigr]_0^1 = P(x, y) .
\]
संतुलित $ f_y = Q $ (समान गणना के साथ $ P_y = Q_x $ प्रयोग किया गया
दूसरे तरीके से)। ध्यान दें कि परिकल्पना कहाँ प्रवेश करती है: $ f $ परिभाषित किया गया है
खंड के साथ एकीकरण करके $[0, M]$, जिसमें निहित है $ U $
ठीक इसलिए क्योंकि $ U $ तारे के आकार का है.
\end{solution}

\begin{solution}{exo:b2:multint:9}
पट्टी पर $[0, A] \times [0, \infty)$ समारोह $(x, y)
\mapsto e^{-xy}\sin x $ क्या नहीं है \omterm{def:b2:series:def}{बिल्कुल} तक एकीकृत $ y =
\infty $ \omterm{def:b2:funcseq:def}{समान रूप से} एक भोले अर्थ में, लेकिन प्रत्येक पुनरावृत्त अभिन्न
\omterm{def:b2:integration:improper}{अभिसरण} होता है और उनकी समानता फ़ुबिनी से आगे बढ़ती है $[0, A]
\times [0, B]$ प्लस एक सीमा $ B \to \infty $ (पूंछ
$\int_0^A\int_B^\infty e^{-xy}\abs{\sin x}\,\dd y\,\dd x \leq
\int_0^A \frac{e^{-Bx}\abs{\sin x}}{x}\dd x \leq \int_0^A e^{-Bx}
\dd x\to 0$, का उपयोग करना $\abs{\sin x} \leq x $).

\emph{$ y $ पहला:} $\int_0^\infty e^{-xy}\,\dd y = \frac1x $ के लिए $ x
> 0$, तो पहला अभिन्न है $\int_0^A \frac{\sin x}{x}\,\dd x $.

\emph{$ x $ पहला:} भागों द्वारा दो एकीकरण (या लेना
का काल्पनिक भाग $\int_0^A e^{(i - y)x}\dd x $) देना
\[
\int_0^A e^{-xy}\sin x\,\dd x
= \frac{1 - e^{-Ay}(y\sin A + \cos A)}{1 + y^2} .
\]
में एकीकृत करना $ y $ ऊपर $[0, \infty)$, शब्द $\int_0^\infty
\frac{\dd y}{1 + y^2} = \frac\pi2$ अलग हो जाता है:
\[
\int_0^A \frac{\sin x}{x}\,\dd x
= \frac{\pi}{2}
- \int_0^\infty e^{-Ay}\,\frac{y\sin A + \cos A}{1 + y^2}\,\dd y .
\]
शेषफल से परिबद्ध है $\int_0^\infty e^{-Ay}\frac{y +
1}{1 + y^2}\dd y \leq \int_0^\infty e^{-Ay}\cdot\frac{1+y}{1+y^2}
\,\dd y \to 0$ जैसा $ A \to \infty $ (प्रभुत्व \omterm{def:b2:integration:improper}{अभिसरण}, या
अपरिष्कृत बाध्य $\frac{1 + y}{1 + y^2} \leq \frac32$ दे रही है
$\frac{3}{2A}$). इस तरह $\int_0^\infty\frac{\sin x}{x}\dd x =
\frac\pi2$ --- में समान मान प्राप्त होता है
\cref{ch:b2:integration} a को विभेदित करके \omterm{thm:b2:integration:continuity}{पैरामीटर अभिन्न};
यहाँ फ़ुबिनी इसके स्थान पर कार्य करती है।
\end{solution}

\begin{solution}{exo:b2:multint:10}
द्वारा पैरामीट्रिज वक्राकार \omterm{def:b2:curves:length}{लंबाई} $ s \in [0, 2\pi]$, इसलिए $ x'^2 + y'^2 = 1$,
और अनुवाद करें ताकि $\int_0^{2\pi} x(s)\,\dd s = 0$. द्वारा
\cref{cor:b2:multint:area},
\[
2A = \oint x\,\dd y - y\,\dd x
= \int_0^{2\pi}\bigl(xy' - yx'\bigr)\dd s .
\]
घालमेल $\oint y\,\dd x $ अवधि के दौरान भागों द्वारा (सीमा)।
शर्तें आवधिकता द्वारा रद्द), $-\int yx' = \int y'x $, तो वास्तव में
$2A = 2\int_0^{2\pi}xy'\,\dd s $. तब $2xy' \leq x^2 + y'^2$
देता है
\[
2A \leq \int_0^{2\pi}\bigl(x^2 + y'^2\bigr)\dd s
= \int_0^{2\pi} x^2 + \int_0^{2\pi}\bigl(1 - x'^2\bigr)
= 2\pi - \int_0^{2\pi}\bigl(x'^2 - x^2\bigr)\dd s .
\]
विर्टिंगर की असमानता (\cref{ch:b2:fourier} अभ्यास: ए के लिए
$2\pi $-आवधिक $\mathcal{C}^1$ शून्य माध्य के साथ कार्य करें,
$\int x^2 \leq \int x'^2$) अंतिम पूर्णांक को अऋणात्मक बनाता है:
$ A \leq \pi $. विर्टिंगर में समानता के लिए समानता की आवश्यकता है ($ x(s) =
a\cos s + b\sin s $) और में $2xy' \leq x^2 + y'^2$ ($ y' = x $
\omterm{def:b2:funcseq:def}{बिंदुवार}), जो मजबूर करता है $ y = a\sin s - b\cos s + c $: वक्र है
इकाई वृत्त (उपयुक्त रूप से केन्द्रित)। के एक वक्र के बाद से \omterm{def:b2:curves:length}{लंबाई} $ L $
को पुनः मापता है \omterm{def:b2:curves:length}{लंबाई} $2\pi $, सामान्य कथन है $ A \leq
\frac{L^2}{4\pi}$: दी गई परिधि के सभी बंद वक्रों के बीच,
वृत्त सबसे बड़े को घेरता है \omterm{def:b2:multint:domain}{क्षेत्र}.
\end{solution}

\begin{solution}{exo:b2:multint:11}
गोलाकार निर्देशांक में $ z = r\sin\varphi $ और $\dd x\,\dd
y\,\dd z = r^2\cos\varphi\,\dd r\,\dd\theta\,\dd\varphi $:
\[
\iiint_B z^2
= \int_0^R r^4\,\dd r\int_0^{2\pi}\dd\theta
\int_{-\pi/2}^{\pi/2}\sin^2\varphi\cos\varphi\,\dd\varphi
= \frac{R^5}5\cdot2\pi\cdot
\Bigl[\frac{\sin^3\varphi}3\Bigr]_{-\pi/2}^{\pi/2}
= \frac{4\pi R^5}{15}.
\]
क्रमपरिवर्तन निर्देशांक के \omterm{def:b2:diffcalc:differential}{अंतर}्गत गेंद की समरूपता द्वारा,
$\iiint_B x^2 = \iiint_B y^2 = \iiint_B z^2$, इसलिए $\iiint_B(x^2
+ y^2 + z^2) = 3\cdot\frac{4\pi R^5}{15} = \frac{4\pi R^5}5$.
शैल जाँच: $\int_0^R r^2\cdot 4\pi r^2\,\dd r = \frac{4\pi
R^5}5$ --- एकात्म $ r^2$ के \omterm{def:b2:multint:domain}{क्षेत्र} पर स्थिर है
त्रिज्या $ r $, का \omterm{def:b2:multint:domain}{क्षेत्र} $4\pi r^2$.
\end{solution}

\begin{solution}{exo:b2:multint:12}
किनारे को पैरामीट्रिज करें $(x_i, y_i)$ को $(x_{i+1}, y_{i+1})$
द्वारा $\gamma(t) = \bigl((1-t)x_i + tx_{i+1},\ (1-t)y_i +
ty_{i+1}\bigr)$. इसका योगदान है
$\frac12\oint(x\,\dd y - y\,\dd x)$ है
\[
\frac12\int_0^1\Bigl(\bigl((1-t)x_i +
tx_{i+1}\bigr)(y_{i+1} - y_i) - \bigl((1-t)y_i +
ty_{i+1}\bigr)(x_{i+1} - x_i)\Bigr)\dd t ,
\]
और तबसे $\int_0^1\bigl((1-t)u + tv\bigr)\dd t = \frac{u +
v}2$, यह बराबर है
\[
\frac14\Bigl((x_i + x_{i+1})(y_{i+1} - y_i) - (y_i +
y_{i+1})(x_{i+1} - x_i)\Bigr)
= \frac12\bigl(x_iy_{i+1} - x_{i+1}y_i\bigr),
\]
क्रॉस शर्तें रद्द करना। संक्षेप में कहें तो $ m $ किनारे देता है
जूते का फीता फार्मूला, द्वारा \cref{cor:b2:multint:area}. त्रिकोण
$(0,0), (1,0), (0,1)$: $\frac12\bigl((0\cdot0 - 1\cdot0) +
(1\cdot1 - 0\cdot0) + (0\cdot0 - 0\cdot1)\bigr) = \frac12$,
सही \omterm{def:b2:multint:domain}{क्षेत्र}.
\end{solution}

\begin{solution}{pb:b2:multint:1}
\textbf{1.} गेंद $ B_n(R)$ पुनरावृत्त सीमाओं द्वारा वर्णित है
$-R \leq x_n \leq R $, तब $\abs{x_{n-1}} \leq \sqrt{R^2 -
x_n^2}$, और इसी तरह; प्रतिस्थापन $ x_i = Ru_i $ प्रत्येक में
$ n $ एक-चर इंटीग्रल को प्रत्येक से गुणा किया जाता है $ R $ और मानचित्र
उन पर सीमाएं $ B_n(1)$: $ v_n(R) = R^n\,v_n(1) =
V_nR^n $.

\textbf{2.} साथ-साथ काटना $ x_n = t $: का टुकड़ा $ B_n(1)$ है
गेंद $ B_{n-1}\bigl(\sqrt{1 - t^2}\bigr)$, तो, प्रश्न से
1,
\[
V_n = \int_{-1}^1 v_{n-1}\bigl(\sqrt{1 - t^2}\bigr)\,\dd t
= V_{n-1}\int_{-1}^{1}(1 - t^2)^{\frac{n-1}2}\,\dd t .
\]

\textbf{3.} साथ $ t = \sin\theta $, $\dd t =
\cos\theta\,\dd\theta $ और $(1 - t^2)^{\frac{n-1}2} =
\cos^{n-1}\theta $ पर $\intcc{-\pi/2}{\pi/2}$:
\[
\int_{-1}^1(1 - t^2)^{\frac{n-1}2}\dd t
= \int_{-\pi/2}^{\pi/2}\cos^n\theta\,\dd\theta
= 2\int_0^{\pi/2}\cos^n\theta\,\dd\theta = 2W_n,
\]
और $\theta \mapsto \frac\pi2 - \theta $ साइन को आपस में बदल देता है
और के कोसाइन रूप $ W_n $.

\textbf{4.} लिखना $\sin^n = \sin^{n-2}(1 - \cos^2)$ और
एकीकृत करें $\int\sin^{n-2}\cos\cdot\cos $ भागों द्वारा ($ v =
\frac{\sin^{n-1}}{n-1}$):
\[
W_n = W_{n-2} - \frac{W_n}{n-1}
\quad\Longrightarrow\quad
W_n = \frac{n-1}{n}W_{n-2}.
\]
इस तरह $ nW_nW_{n-1} = (n-1)W_{n-1}W_{n-2}$: क्रम
$(nW_nW_{n-1})$ स्थिर है, बराबर है $1\cdot W_1W_0 =
1\cdot\frac\pi2$, इसलिए $ W_nW_{n-1} = \frac{\pi}{2n}$.

\textbf{5.} प्रश्न 2--3 दीजिए $ V_n = 2W_nV_{n-1}$, दो बार:
\[
V_n = 2W_n\cdot 2W_{n-1}\,V_{n-2}
= 4\,\frac{\pi}{2n}\,V_{n-2} = \frac{2\pi}n\,V_{n-2}.
\]

\textbf{6.} से $ V_0 = 1$: $ V_{2k} = \frac{2\pi}{2k}V_{2k-2}
= \frac\pi kV_{2k-2}$, इसलिए $ V_{2k} = \frac{\pi^k}{k!}$ द्वारा
प्रेरण. से $ V_1 = 2$: $ V_{2k+1} =
\frac{2\pi}{2k+1}V_{2k-1}$, इसलिए
\[
V_{2k+1} = 2\prod_{j=1}^k\frac{2\pi}{2j+1}
= \frac{2^{k+1}\pi^k}{1\cdot3\cdot5\cdots(2k+1)} .
\]

\textbf{7.} $ V_1 = 2$, $ V_2 = \pi \approx 3.142$, $ V_3 =
\frac{4\pi}3 \approx 4.189$, $ V_4 = \frac{\pi^2}2 \approx
4.935$, $ V_5 = \frac{8\pi^2}{15} \approx 5.264$, $ V_6 =
\frac{\pi^3}6 \approx 5.168$, $ V_7 = \frac{16\pi^3}{105}
\approx 4.725$. एक कदम अनुपात है $ V_n/V_{n-1} = 2W_n $,
और $(W_n)$ गिरते हुए ($\sin^n \leq \sin^{n-1}$
\omterm{def:b2:funcseq:def}{बिंदुवार}). अब $2W_5 = 2\cdot\frac45\cdot\frac23 =
\frac{16}{15} > 1$ जबकि $2W_6 =
2\cdot\frac56\cdot\frac34\cdot\frac12\cdot\frac\pi2 =
\frac{5\pi}{16} < 1$: अनुपात अधिक है $1$ तक $ n = 5$ और
नीचे हैं $1$ से $ n = 6$ पर --- $(V_n)$ उसकी ओर बढ़ जाता है
अधिकतम $ V_5$ और फिर घट जाती है.

\textbf{8.} के लिए $ n \geq 13 > 4\pi $: $ V_n/V_{n-2} = 2\pi/n <
\tfrac12$, इसलिए $ V_{n} \leq C\cdot 2^{-n/2}$ एक निश्चित के साथ
\omterm{def:b2:metric:continuity}{निरंतर}; किसी के लिए भी बेहतर $ q > 0$, $2\pi/n < q^2$ बड़े के लिए
$ n $, इसलिए $ V_n/q^n \to 0$: क्षय हर ज्यामितीय को मात देता है
क्रम. का \omterm{def:b2:integration:improper}{अभिसरण} $\sum V_n $ अनुपात से चलता है
$ V_n/V_{n-2} \to 0$ (से एक ज्यामितीय श्रृंखला के साथ तुलना करें
कुछ रैंक पर)।

\textbf{9.} $\sum_{k\geq0}V_{2k}x^{2k} =
\sum_{k\geq0}\frac{(\pi x^2)^k}{k!} = \eu^{\pi x^2}$, द
घातीय श्रृंखला (\cref{ch:b2:powerseries}), के लिए \omterm{def:b2:integration:improper}{अभिसरण}
प्रत्येक $ x $. पर $ x = 1$: $\sum_kV_{2k} = \eu^\pi \approx
23.14$.

\textbf{10.} घन पर $\intcc{-R}R^n $ इंटीग्रैंड है
उत्पाद $\prod_i\eu^{-x_i^2}$, इसलिए पुनरावृत्त अभिन्न
कारक: $\bigl(\int_{-R}^R\eu^{-t^2}\dd t\bigr)^n $.
दे $ R \to \infty $ और उपयोग कर रहे हैं
$\int_\R\eu^{-t^2}\dd t = \sqrt\pi $
(\cref{ex:b2:multint:polar}): $ I_n = \pi^{n/2}$.

\textbf{11.} $ t = u^2$ देता है $\Gamma(\tfrac12) =
\int_0^\infty t^{-1/2}\eu^{-t}\dd t =
2\int_0^\infty\eu^{-u^2}\dd u = \sqrt\pi $. बार-बार दोहराना
$\Gamma(s+1) = s\Gamma(s)$: $\Gamma(k+1) = k!\,\Gamma(1) =
k!$, और
\[
\Gamma\Bigl(k + \frac32\Bigr)
= \Bigl(k + \frac12\Bigr)\Bigl(k - \frac12\Bigr)\cdots
\frac12\cdot\Gamma\Bigl(\frac12\Bigr)
= \frac{(2k+1)(2k-1)\cdots1}{2^{k+1}}\,\sqrt\pi .
\]

\textbf{12.} तय करना $ F_n = \pi^{n/2}/\Gamma(\frac n2 + 1)$.
तब से $\Gamma(\frac n2 + 1) = \frac n2\,\Gamma(\frac n2) =
\frac n2\,\Gamma(\frac{n-2}2 + 1)$, हम पाते हैं $ F_n =
\frac{2\pi}nF_{n-2}$: जैसा ही पुनरावर्तन $ V_n $ (प्रश्न
5). आधार: $ F_1 = \sqrt\pi/\Gamma(\frac32) =
\sqrt\pi/(\frac{\sqrt\pi}2) = 2 = V_1$ और $ F_2 =
\pi/\Gamma(2) = \pi = V_2$. प्रेरण द्वारा $ V_n = F_n $ सभी के लिए
$ n $; प्रश्न 11 इसे वापस दो में बदल देता है बंद प्रपत्र का
प्रश्न 6.

\textbf{13.} साथ $ r = \sqrt t $: $\int_0^\infty
\eu^{-r^2}r^{n-1}\dd r = \frac12\int_0^\infty
t^{\frac n2 - 1}\eu^{-t}\dd t = \frac12\Gamma(\frac n2)$.
इस तरह
\[
n\,V_n\int_0^\infty\eu^{-r^2}r^{n-1}\dd r
= V_n\cdot\frac n2\,\Gamma\Bigl(\frac n2\Bigr)
= V_n\,\Gamma\Bigl(\frac n2 + 1\Bigr) = \pi^{n/2} = I_n .
\]
दोनों पक्षों के सिद्ध होने से हो सकती है पहचान \emph{पढ़ना} जैसे
गॉसियन इंटीग्रल का शैल अपघटन: गोला
त्रिज्या का $ r $ किया जाता है \omterm{def:b2:multint:domain}{क्षेत्र} $ nV_nr^{n-1}$, और गाऊसी
वजन $\eu^{-r^2}$ कोशों पर एकीकृत किया गया है।

\textbf{14.} $ s_0 = V_1 = 2$ (द $0$-गोला दो बिंदु है),
$ s_1 = 2V_2 = 2\pi $, $ s_2 = 3V_3 = 4\pi $, $ s_3 = 4V_4 =
2\pi^2$; और $\int_0^R s_{n-1}r^{n-1}\dd r = V_nR^n =
v_n(R)$: \omterm{def:b2:multint:domain}{क्षेत्र} आयतन का रेडियल व्युत्पन्न है।

\textbf{15.} के लिए $ n = 2k $, स्टर्लिंग
(\cref{thm:b2:comparison:stirling}) देता है $ k! \sim
\sqrt{2\pi k}\,(k/\eu)^k $, इसलिए
\[
V_{2k} = \frac{\pi^k}{k!}
\sim \frac{(\pi\eu/k)^k}{\sqrt{2\pi k}}
= \frac1{\sqrt{\pi n}}\Bigl(\frac{2\pi\eu}n\Bigr)^{n/2}
\qquad (n = 2k).
\]
विषम के लिए $ n $: $ V_{2k+1} = 2W_{2k+1}V_{2k} \leq 2V_{2k}$, तो
वही सुपर-ज्यामितीय क्षय सीमाएं (एक कारक तक) रखती हैं
$2$ और घातांक में एक का बदलाव) --- प्रत्येक के लिए $ q >
0$, $ V_n = o(q^n)$.

\textbf{16.} $ V_2/4 = \pi/4 \approx 0.785$; $ V_3/8 = \pi/6
\approx 0.524$; $ V_{10}/2^{10} = \frac{\pi^5}{120\cdot1024}
\approx 0.0025$. सामान्य तौर पर $\frac{V_n/2^n}{V_{n-2}/2^{n-2}}
= \frac{2\pi}{4n} = \frac{\pi}{2n} \to 0$: अनुपात की ओर रुझान होता है
$0$ (सुपर-ज्यामितीय रूप से)। अंकित गेंद पर कब्जा है
लुप्त अंश: घन का आयतन उसके पास स्थानांतरित हो जाता है
कोने.

\textbf{17.} प्रश्न 1 द्वारा त्रिज्या की भीतरी गेंद $1 -
\varepsilon $ मात्रा है $ V_n(1-\varepsilon)^n $, तो बाहरी
शेल अंश को वहन करता है $1 - (1 - \varepsilon)^n \to 1$.
के लिए $ n = 100$, $\varepsilon = 0.01$: $(0.99)^{100} =
\eu^{100\ln0.99} \approx \eu^{-1.005} \approx 0.366$: के बारे में
$63\%$ गेंद के भीतर निहित है $1\%$ इसकी सतह का.

\textbf{18.} $(W_n)$ घट जाती है, इसलिए $ W_n \leq W_{n-1} \leq
W_{n-2} = \frac{n}{n-1}W_n $: निचोड़ना, $ W_{n-1}/W_n \to 1$.
से गुणा करना $ W_nW_{n-1} = \frac\pi{2n}$: $ W_n^2 \sim
\frac\pi{2n}$, यानी\ $ W_n \sim \sqrt{\pi/(2n)}$. निचली सीमा:
$ W_n^2 \geq W_nW_{n+1} = \frac{\pi}{2(n+1)}$, इसलिए $ W_n \geq
\sqrt{\pi/(2(n+1))}$ हरएक के लिए $ n $.

\textbf{19.} अंश: $1 - x^2 \leq \eu^{-x^2}$ देता है $(1 -
x^2)^{\frac{n-1}2} \leq \eu^{-(n-1)x^2/2}$, और साथ में $ a =
\frac{n-1}2$,
\[
\int_\delta^1(1 - x^2)^{\frac{n-1}2}\dd x
\leq \int_\delta^\infty\eu^{-ax^2}\dd x
\leq \int_\delta^\infty\frac x\delta\,\eu^{-ax^2}\dd x
= \frac{\eu^{-a\delta^2}}{2a\delta}
= \frac{\eu^{-(n-1)\delta^2/2}}{(n-1)\delta}.
\]
भाजक: $2W_n \geq \sqrt{2\pi/(n+1)}$ प्रश्न 18 द्वारा.
विभाजित करने से प्रदर्शित सीमा प्राप्त होती है। के लिए $\delta =
s/\sqrt{n-1}$ यह बन जाता है
$\sqrt{\tfrac{n+1}{2\pi(n-1)}}\;\eu^{-s^2/2}/s = O\bigl(
\eu^{-s^2/2}/s\bigr)$, \omterm{def:b2:funcseq:def}{समान रूप से} में $ n $: स्लैब के बाहर
$\abs{x_1} \leq s/\sqrt{n-1}$ लगभग कोई मात्रा नहीं है, के लिए
$ s $ मध्यम रूप से बड़ा --- और समरूपता से भी यही बात लागू होती है
हर दिशा.

\textbf{20.} दोनों कथन सह-अस्तित्व में हैं क्योंकि वे वर्णन करते हैं
एक ही बिंदु के विभिन्न निर्देशांक. लगभग हर बिंदु
का $ B_n(1)$ है आदर्श के करीब $1$ (प्रश्न 17: रेडियल
गोले के पास एकाग्रता), फिर भी प्रत्येक $ n $
निर्देशांक छोटा है, क्रम का $1/\sqrt n $ (प्रश्न 19),
जो तब से लगातार जारी है $ n $ आकार के निर्देशांक $1/\sqrt n $
पास होना आदर्श \omterm{def:b2:structures:generated}{आदेश} की $1$. उच्च-आयामी आयतन संकेंद्रित होता है
जहां सभी निर्देशांक साझा करते हैं आदर्श बजट \omterm{def:b2:funcseq:def}{समान रूप से} ---
गोले के निकट, लेकिन प्रत्येक निर्देशांक अक्ष ध्रुव से दूर।

\textbf{21.} टुकड़ा $\Delta_n $ पर $ x_n = t \in \intcc01$: द
टुकड़ा है $\{x' \in \R^{n-1} : x_i \geq 0,\ \sum x_i \leq 1 -
t\} = (1-t)\Delta_{n-1}$, आयतन का
$(1-t)^{n-1}\operatorname{vol}(\Delta_{n-1})$ एकरूपता से.
इसलिए
\[
\operatorname{vol}(\Delta_n) =
\operatorname{vol}(\Delta_{n-1})\int_0^1(1 - t)^{n-1}\dd t =
\frac{\operatorname{vol}(\Delta_{n-1})}{n}
\quad\Longrightarrow\quad
\operatorname{vol}(\Delta_n) = \frac1{n!}\,.
\]

\textbf{22.}  $2^n $ साइन-ऑर्थेंट्स कट $ C_n $ में $2^n $
की प्रतियाँ $\Delta_n $ (समन्वय पर नगण्य ओवरलैप के साथ
हाइपरप्लेन): $\operatorname{vol}(C_n) = \frac{2^n}{n!}$. अगर
$\sum\abs{x_i} \leq 1$ तब $\sum x_i^2 \leq
\bigl(\sum\abs{x_i}\bigr)^2 \leq 1$: $ C_n \subseteq B_n(1)$;
और $ B_n(1) \subseteq \intcc{-1}1^n $ तब से $\abs{x_i} \leq
\norm x $. इस तरह $\frac{2^n}{n!} \leq V_n \leq 2^n $ ---
प्रश्न 15 के अनुरूप कौन-कौन से स्थान हैं $ V_n $ के बीच
भाज्य और ज्यामितीय पैमाने।

\textbf{23.} के लिए $(x, y)$ यूनिट डिस्क में, का टुकड़ा
$ B_4(1)$ त्रिज्या की डिस्क है $\sqrt{1 - x^2 - y^2}$ में
$(z, w)$-विमान, का \omterm{def:b2:multint:domain}{क्षेत्र} $\pi(1 - x^2 - y^2)$. ध्रुवीय में
निर्देशांक:
\[
V_4 = \iint_{x^2+y^2\leq1}\pi(1 - x^2 - y^2)\,\dd x\,\dd y
= \pi\int_0^{2\pi}\!\!\int_0^1(1 - \rho^2)\rho\,
\dd\rho\,\dd\alpha
= \pi\cdot2\pi\cdot\frac14 = \frac{\pi^2}2 ,
\]
प्रश्न 6 से सहमत.

\textbf{24.} संभाव्यता आयतन अनुपात है
$\dfrac{V_{20}}{2^{20}} = \dfrac{\pi^{10}}{10!\cdot2^{20}}
\approx \dfrac{0.0258}{1\,048\,576} \approx
2.5\cdot10^{-8}$. पहली हिट तक ड्रॉ की संख्या है
क्रम के विपरीत, के बारे में $4\cdot10^7$: एक अस्वीकृति नमूना
इसने डिस्क के लिए खूबसूरती से काम किया ($\pi/4$ हिट्स का) है
आयाम में बेकार $20$ --- आयामीता का अभिशाप
एक पंक्ति.

\textbf{25.} रूट एक (भाग I--II) का उपयोग किया गया: फ़ुबिनी-प्रकार
पुनरावृत्त अभिन्न, एक-चर प्रतिस्थापन का टुकड़ा करना
प्रत्येक समन्वय (एकरूपता) में, और वालिस इंटीग्रल्स
--- शुद्ध एक-चर कैलकुलस प्लस इंडक्शन। मार्ग दो
(भाग III) का उपयोग किया गया: फ़ुबिनी की उत्पाद संरचना के लिए $ I_n $,
गॉस इंटीग्रल के माध्यम से चरों का ध्रुवीय परिवर्तन
\cref{ex:b2:multint:polar}, और यह $\Gamma $ समारोह का
कार्यात्मक समीकरण. वे अंदर मिलते हैं $ V_n =
\pi^{n/2}/\Gamma(\frac n2 + 1)$, स्टर्लिंग के साथ
(\cref{thm:b2:comparison:stirling}) सूत्र को परिवर्तित करना
स्पर्शोन्मुखता में। वर्ष 3 का खंड इन सबका पुनर्निर्माण करता है
लेबेस्ग्यू का अभिन्न अंग: वहां फ़ुबिनी और चर का परिवर्तन
सामान्य पूर्णांकीय फलनों के लिए प्रमेय गोलाकार हैं
निर्देशांक हर आयाम में मौजूद हैं, और वही गेंद
वॉल्यूम के कार्यित लाभांश के रूप में पुनः प्रकट होते हैं
उत्पाद-माप और स्टर्लिंग समस्याएं --- हावी हैं
\omterm{def:b2:integration:improper}{अभिसरण} हमारे हाथ से बने निचोड़ों की जगह ले रहा है।
\end{solution}
