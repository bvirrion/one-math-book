\section*{अध्याय \ref{ch:b2:integration} --- एकीकरण}

\begin{solution}{exo:b2:integration:1}
$\int_0^1 \frac{\dd t}{\sqrt{t(1-t)}}$: पास में $0$, $\sim t^{-1/2}$
($\alpha = \frac12 < 1$: \omterm{def:b2:integration:improper}{अभिसरण}); पास में $1$, $\sim
(1-t)^{-1/2}$: \omterm{def:b2:integration:improper}{अभिसरण}. \omterm{def:b2:integration:improper}{अभिसरण} (इसका मान है $\pi $, द्वारा
प्रतिस्थापन $ t = \sin^2\theta $).

$\int_1^\infty \frac{\ln t}{t^2}$: $\frac{\ln t}{t^2} = o(t^{-3/2})$:
\omterm{def:b2:integration:improper}{अभिसरण} (मान $1$ भागों द्वारा)।

$\int_0^\infty \frac{\dd t}{1 + t^2\sin^2 t}$: भिन्न। पास में $ t =
n\pi $, लिखना $ t = n\pi + u $: $\sin^2 t = \sin^2 u \leq u^2$, जल्द ही
$\abs u \leq \frac{1}{n}$, $1 + t^2\sin^2 t \leq 1 + (n\pi +
1)^2u^2 \leq C n^2 u^2 + 1$; इस तरह
\[
\int_{n\pi - 1/n}^{n\pi + 1/n} \frac{\dd t}{1 + t^2\sin^2 t}
\geq \int_{-1/n}^{1/n} \frac{\dd u}{1 + Cn^2u^2}
= \frac{2\arctan\sqrt C}{\sqrt C}\cdot\frac{1}{n} ,
\]
अपसारी हार्मोनिक-प्रकार की श्रृंखला का एक पद: सारांश $ n $, द
आदिम असीमित है.
\end{solution}

\begin{solution}{exo:b2:integration:2}
विकल्प $ u = \lambda t $:
\[
\int_0^\infty t^n \eu^{-\lambda t}\dd t
= \frac{1}{\lambda^{n+1}}\int_0^\infty u^n\eu^{-u}\dd u
= \frac{\Gamma(n+1)}{\lambda^{n+1}} = \frac{n!}{\lambda^{n+1}} .
\]
साथ $ t = \eu^{-u}$ ($\dd t = -\eu^{-u}\dd u $):
\[
\int_0^1 (\ln t)^n \dd t = \int_0^{\infty} (-u)^n \eu^{-u}\,\dd u
= (-1)^n\, n! .
\]
\end{solution}

\begin{solution}{exo:b2:integration:3}
\omterm{def:b2:integration:improper}{अभिसरण}: एकीकरण है $\leq \frac{1}{1+t^2}$ पास में $\infty $
और निकट ही घिरा हुआ है $0$ (दोनों कारक नीचे से दूर बंधे हैं $0$):
\omterm{def:b2:series:def}{बिल्कुल} \omterm{def:b2:integration:improper}{अभिसरण}, प्रत्येक के लिए $ x $. स्थानापन्न $ t = \frac1u $
($\dd t = -\frac{\dd u}{u^2}$):
\[
I(x) = \int_0^\infty \frac{1}{\bigl(1 +
\frac1{u^2}\bigr)\bigl(1 + u^{-x}\bigr)}\cdot\frac{\dd u}{u^2}
= \int_0^\infty \frac{u^x}{(1 + u^2)(1 + u^x)}\,\dd u .
\]
के दो भाव जोड़ना $ I(x)$:
\[
2I(x) = \int_0^\infty \frac{1 + t^x}{(1+t^2)(1+t^x)}\dd t
= \int_0^\infty \frac{\dd t}{1 + t^2} = \frac{\pi}{2} :
\]
$ I(x) = \frac\pi4$, से स्वतंत्र $ x $.
\end{solution}

\begin{solution}{exo:b2:integration:4}
पास में $0^+$, साथ $ u = \abs{\ln t} \to \infty $. अगर $\alpha < 1$:
की परवाह किए बिना \omterm{def:b2:integration:improper}{अभिसरण} $\beta $ (साथ तुलना करें $ t^{-\alpha'}$ के लिए
$\alpha < \alpha' < 1$: लॉग फैक्टर पीटा गया है)। अगर $\alpha > 1$:
विचलन की परवाह किए बिना $\beta $ (साथ तुलना करें $ t^{-\alpha''}$,
$1 < \alpha'' < \alpha $). अगर $\alpha = 1$: विकल्प $ t =
\eu^{-u}$:
\[
\int_0^{1/2} \frac{\dd t}{t\,\abs{\ln t}^\beta}
= \int_{\ln 2}^{\infty} \frac{\dd u}{u^\beta},
\]
\omterm{def:b2:integration:improper}{अभिसरण} आईएफएफ $\beta > 1$. सारांश: \omterm{def:b2:integration:improper}{अभिसरण} आईएफएफ $\alpha < 1$,
या ($\alpha = 1$ और $\beta > 1$) --- बर्ट्रेंड का दर्पण
शृंखला.
\end{solution}

\begin{solution}{exo:b2:integration:5}
निरंतरता पर $\intco{0}{\infty}$: प्रभुत्व
$\bigl|\frac{\eu^{-xt}}{1+t^2}\bigr| \leq \frac{1}{1+t^2}$,
अभिन्न, एकसमान $ x \geq 0$:
\cref{thm:b2:integration:continuity}.

$ C^2$ पर $\intoo{0}{\infty}$: पर $ x \geq a > 0$, पहले दो
$ x $-व्युत्पन्न $\frac{-t\,\eu^{-xt}}{1+t^2}$ और
$\frac{t^2\eu^{-xt}}{1+t^2}$ का बोलबाला है $ t\,\eu^{-at}$ और
$\eu^{-at}$: के दो अनुप्रयोग
\cref{thm:b2:integration:leibnizrule}. तब
\[
F''(x) + F(x) = \int_0^\infty \frac{t^2 + 1}{1 + t^2}\,\eu^{-xt}\dd
t = \int_0^\infty \eu^{-xt}\dd t = \frac1x .
\]
सीमा: $0 \leq F(x) \leq \int_0^\infty \eu^{-xt}\dd t = \frac1x \to
0$.
\end{solution}

\begin{solution}{exo:b2:integration:6}
पर $\intcc{\varepsilon}{M}$, विकल्प $ u = at $ और $ u = bt $ में
दो हिस्से:
\[
\int_\varepsilon^M \frac{f(at) - f(bt)}{t}\dd t
= \int_{a\varepsilon}^{aM}\frac{f(u)}{u}\dd u
- \int_{b\varepsilon}^{bM}\frac{f(u)}{u}\dd u
= \int_{a\varepsilon}^{b\varepsilon} \frac{f(u)}{u}\dd u
- \int_{aM}^{bM} \frac{f(u)}{u}\dd u .
\]
पहला टुकड़ा: $ f(u) = f(0) + o(1)$ पास में $0$, और $\int_{a\varepsilon}
^{b\varepsilon} \frac{\dd u}{u} = \ln\frac ba $: टुकड़ा प्रवृत्त होता है
$ f(0)\ln\frac ba $. दूसरा टुकड़ा: $ f(u) \to f(\infty)$, वही
गणना: की ओर प्रवृत्त होता है $ f(\infty)\ln\frac ba $. इसलिए अभिन्न अनुचित \omterm{def:b2:integration:improper}{अभिसरण} को $\bigl(f(0) - f(\infty)\bigr)\ln\frac ba $.

साथ $ f(t) = \eu^{-t}$ ($ f(0) = 1$, $ f(\infty) = 0$), $ a = 1$, $ b
= 2$:
\[
\int_0^\infty \frac{\eu^{-t} - \eu^{-2t}}{t}\dd t = \ln 2 .
\]
\end{solution}

\begin{solution}{exo:b2:integration:7}
द्वारा इंटीग्रैंड का विस्तार करें $0$ आगे $ t = n $: $ g_n(t) = (1 -
\frac tn)^n t^{x-1}\mathbf{1}_{t \leq n}$. बिंदुवार, $ g_n(t) \to
\eu^{-t}t^{x-1}$ (चक्रवृद्धि-ब्याज सीमा, वर्ष 1 खंड)।
प्रभुत्व: $\ln(1 - u) \leq -u $ देता है $(1 - \frac tn)^n \leq
\eu^{-t}$ पर $\intcc{0}{n}$, इसलिए $\abs{g_n(t)} \leq
\eu^{-t}t^{x-1} = \varphi(t)$, अभिन्न. प्रभुत्व \omterm{def:b2:integration:improper}{अभिसरण}:
\[
\int_0^n \Bigl(1 - \frac tn\Bigr)^n t^{x-1}\dd t
\xrightarrow[n\to\infty]{} \int_0^\infty \eu^{-t}t^{x-1}\dd t
= \Gamma(x) .
\]
(बार-बार भागों द्वारा बाईं ओर की गणना करने पर यूलर का उत्पाद मिलता है
प्रपत्र $\Gamma(x) = \lim \frac{n!\,n^x}{x(x+1)\cdots(x+n)}$.)
\end{solution}

\begin{solution}{exo:b2:integration:8}
$ H $ में भिन्न है $ x $ (एकीकृत $ C^1$ में $ x $, व्युत्पन्न
$-2x(1+t^2)\cdot\frac{\eu^{-x^2(1+t^2)}}{1+t^2} =
-2x\,\eu^{-x^2}\eu^{-x^2t^2}$, \omterm{def:b2:metric:continuity}{निरंतर} और सीमाबद्ध है \omterm{def:b2:metric:compact}{कॉम्पैक्ट} कैमरों
का $ x $, प्रभुत्व खत्म $ t \in \intcc{0}{1}$ मामूली):
\[
H'(x) = -2x\,\eu^{-x^2}\int_0^1 \eu^{-x^2t^2}\,\dd t
\overset{u = xt}{=} -2\,\eu^{-x^2}\int_0^x \eu^{-u^2}\,\dd u
= -G'(x),
\]
तब से $ G'(x) = 2\eu^{-x^2}\int_0^x \eu^{-t^2}\dd t $ (श्रृंखला नियम चालू
वर्ग, कलन का मौलिक प्रमेय)। इसलिए $ G + H $ है
स्थिर, बराबर $ G(0) + H(0) = 0 + \int_0^1 \frac{\dd t}{1+t^2}
= \frac\pi4$.

जैसा $ x \to \infty $: $0 \leq H(x) \leq \eu^{-x^2}\int_0^1 \dd t \to
0$, इसलिए $ G(x) \to \frac\pi4$:
\[
\int_0^\infty \eu^{-t^2}\dd t = \sqrt{\frac\pi4} =
\frac{\sqrt\pi}{2} .
\]
(फलस्वरूप $\Gamma\bigl(\frac12\bigr) = 2\int_0^\infty
\eu^{-t^2}\dd t = \sqrt\pi $, प्रतिस्थापन द्वारा $ t = \sqrt u $.)
\end{solution}

\begin{solution}{exo:b2:integration:9}

और सीमा तक पारित) गुणनखंडन पर लागू होता है

\Gamma\Bigl(\frac{x+y}{2}\Bigr) \leq
\Gamma(x)^{1/2}\,\Gamma(y)^{1/2}
\quad\Longrightarrow\quad
\ln\Gamma\Bigl(\frac{x+y}{2}\Bigr) \leq \frac{\ln\Gamma(x) +
\ln\Gamma(y)}{2} :
\]
$\ln\Gamma $ मध्यबिंदु उत्तल है; प्राणी \omterm{def:b2:metric:continuity}{निरंतर}

\end{solution}

\begin{solution}{exo:b2:integration:10}
\begin{enumerate}
  \item पर $ x \geq a > 0$: द $ x $-इंटीग्रैंड का व्युत्पन्न है
        $-\sin t\,\eu^{-xt}$, का प्रभुत्व है $\eu^{-at}$:
        \cref{thm:b2:integration:leibnizrule} देता है $ F'(x) =
        -\int_0^\infty \eu^{-xt}\sin t\,\dd t $. द्वारा दो एकीकरण
        भाग (या जटिल घातांक):
        \[
        \int_0^\infty \eu^{-xt}\sin t\,\dd t
        = \Im \int_0^\infty \eu^{(-x+\iu)t}\dd t
        = \Im\frac{1}{x - \iu} = \frac{1}{1 + x^2} .
        \]
  \item $\abs{F(x)} \leq \int_0^\infty \eu^{-xt}\dd t = \frac1x \to
        0$. घालमेल $ F' = -\frac{1}{1+x^2}$ से $ x $ को
        $\infty $: $0 - F(x) = -\bigl(\frac\pi2 - \arctan x\bigr)$,
        इसलिए $ F(x) = \frac\pi2 - \arctan x $.
  \item दे $ x \to 0^+$ भर्ती के साथ निरंतरता: $ F(0^+) =
        \frac\pi2$, और $ F(0) = \int_0^\infty \frac{\sin t}{t}\dd
        t $ (अर्ध-अभिसारी डिरिचलेट इंटीग्रल,
        \cref{ex:b2:integration:sint}): इसका मान है
        $\frac\pi2$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b2:integration:11}
\omterm{def:b2:integration:improper}{अभिसरण}: निकट $0$ इंटीग्रैंड का विस्तार होता है लगातार द्वारा
मूल्य $1$ ($\sin t \sim t $); अनंत पर यह है $\leq t^{-2}$:
पूर्ण \omterm{def:b2:integration:improper}{अभिसरण}. पर $\intcc{\varepsilon}{M}$, द्वारा एकीकृत करें
भागों के साथ $ u = \sin^2 t $, $ v' = t^{-2}$:
\[
\int_\varepsilon^M \frac{\sin^2 t}{t^2}\dd t
= \Bigl[-\frac{\sin^2 t}{t}\Bigr]_\varepsilon^M
+ \int_\varepsilon^M \frac{2\sin t\cos t}{t}\dd t
= \Bigl[-\frac{\sin^2 t}{t}\Bigr]_\varepsilon^M
+ \int_{2\varepsilon}^{2M} \frac{\sin u}{u}\dd u
\]
($ u = 2t $ अंतिम अभिन्न में)। ब्रैकेट की प्रवृत्ति होती है $0$ दोनों पर
समाप्त ($\sin^2\varepsilon/\varepsilon \leq \varepsilon $;
$\sin^2 M/M \leq 1/M $), और अंतिम अभिन्न अंग की ओर प्रवृत्त होता है
$\int_0^\infty \frac{\sin u}{u}\dd u = \frac\pi2$
(\cref{exo:b2:integration:10}). इस तरह
\[
\int_0^\infty \Bigl(\frac{\sin t}{t}\Bigr)^{\!2}\dd t
= \frac\pi2 .
\]
\end{solution}

\begin{solution}{exo:b2:integration:12}
\begin{enumerate}
  \item भागों के साथ $ u = \frac{-1}{2t}$, $ v' = -2t\,\eu^{-t^2}$
        (इसलिए $ v = \eu^{-t^2}$):
        \[
        T(x) = \Bigl[\frac{-\eu^{-t^2}}{2t}\Bigr]_x^\infty
        - \int_x^\infty \frac{\eu^{-t^2}}{2t^2}\dd t
        = \frac{\eu^{-x^2}}{2x}
        - \int_x^\infty \frac{\eu^{-t^2}}{2t^2}\dd t .
        \]
        नए इंटीग्रल पर वही डिवाइस ($ u = \frac{-1}{4t^3}$,
        $ v' = -2t\,\eu^{-t^2}$):
        \[
        \int_x^\infty \frac{\eu^{-t^2}}{2t^2}\dd t
        = \frac{\eu^{-x^2}}{4x^3}
        - \frac34\int_x^\infty \frac{\eu^{-t^2}}{t^4}\dd t ,
        \]
        घोषित पहचान कहां से.
  \item भागों द्वारा एक और एकीकरण शेष को सीमित करता है:
        \[
        \int_x^\infty \frac{\eu^{-t^2}}{t^4}\dd t
        = \frac{\eu^{-x^2}}{2x^5}
        - \frac52\int_x^\infty\frac{\eu^{-t^2}}{t^6}\dd t
        \leq \frac{\eu^{-x^2}}{2x^5},
        \]
        इसलिए $0 \leq \frac34\int_x^\infty t^{-4}\eu^{-t^2}\dd t
        \leq \frac{3}{8x^5}\eu^{-x^2}$. (सकारात्मक) गिराना
        प्रश्न 1 की पहचान में शेष निम्न देता है
        बाध्य; पहले के (नकारात्मक) दूसरे पद को छोड़ना
        भाग देता है $ T(x) \leq \frac{\eu^{-x^2}}{2x}$. बाँटना
        द्वारा ब्रैकेटिंग $\frac{\eu^{-x^2}}{2x}$: अनुपात है
        बीच में निचोड़ा हुआ $1 - \frac{1}{2x^2}$ और $1$, इसलिए $ T(x)
        \sim \frac{\eu^{-x^2}}{2x}$.
  \item भागों को पुनरावृत्त करने से औपचारिक श्रृंखला उत्पन्न होती है
        \[
        T(x) \approx \frac{\eu^{-x^2}}{2x}\Bigl(1 -
        \frac{1}{2x^2} + \frac{1\cdot3}{(2x^2)^2} -
        \frac{1\cdot3\cdot5}{(2x^2)^3} + \cdots\Bigr),
        \]
        किसका $ k $-वें गुणांक $1\cdot3\cdots(2k-1) =
        \frac{(2k)!}{2^k k!}$ किसी भी ज्यामितीय की तुलना में तेजी से बढ़ता है
        अनुक्रम: निश्चित के लिए $ x $ शर्तें
        $\frac{1\cdot3\cdots(2k-1)}{(2x^2)^k}$ अनन्त की ओर प्रवृत्त होते हैं
        (उनका अनुपात है $\frac{2k+1}{2x^2} \to \infty $), तो
        श्रृंखला प्रत्येक के लिए भिन्न होती है $ x $. यह है एक
        \emph{asymptotic} विस्तार: किसी निश्चित पर छोटा किया गया
        क्रम, त्रुटि पहले छोड़े गए क्रम की है
        शब्द के रूप में $ x \to \infty $ --- लेकिन कभी भी \omterm{def:b2:integration:improper}{अभिसरण} नहीं
        शृंखला. (यह पूंछ अनुमान मानक गाऊसी है
        संभाव्यता अध्यायों की पूंछ बाउंड।)
\end{enumerate}
\end{solution}

\begin{solution}{pb:b2:integration:1}
\textbf{1.} पर $0^+$ इंटीग्रैंड है $\sim t^{x-1}$: द
परिमित-अंतबिंदु पैमाना \omterm{def:b2:integration:improper}{अभिसरण} आईएफएफ $1 - x < 1$, यानी\ $ x > 0$
(और के लिए $ x \leq 0$, $ t^{x-1} \geq t^{-1}$ विचलन); पर
$+\infty $, $ t^{x-1}\eu^{-t} = o(t^{-2})$ \omterm{def:b2:integration:improper}{अभिसरण} हरएक के लिए
$ x $. तब $\Gamma(x) = \frac{\Gamma(x+1)}{x}$ और $\Gamma(x+1)
\to \Gamma(1) = 1$ जैसा $ x \to 0^+$ (निरंतरता,
\cref{thm:b2:integration:gammaprops}): $\Gamma(x) \sim
\frac1x $.

\textbf{2.} साथ $ t = u^2$, $\dd t = 2u\,\dd u $:
\[
\Gamma\Bigl(\frac12\Bigr) = \int_0^\infty
t^{-1/2}\eu^{-t}\dd t
= \int_0^\infty \frac{\eu^{-u^2}}{u}\,2u\,\dd u
= 2\int_0^\infty \eu^{-u^2}\dd u = \sqrt\pi
\]
द्वारा \cref{exo:b2:integration:8}. साथ $ u = v/\sqrt2$:
\[
\int_\R \eu^{-v^2/2}\dd v
= 2\sqrt2\int_0^\infty \eu^{-u^2}\dd u
= \sqrt2\,\sqrt\pi = \sqrt{2\pi} .
\]

\textbf{3.} के लिए सच है $ n = 0$ (दोनों पक्षों $\sqrt\pi $). अगर
$\Gamma(n + \frac12) = \frac{(2n)!}{4^n n!}\sqrt\pi $, द
कार्यात्मक समीकरण देता है
\[
\Gamma\Bigl(n + 1 + \frac12\Bigr)
= \Bigl(n + \frac12\Bigr)\Gamma\Bigl(n + \frac12\Bigr)
= \frac{2n+1}{2}\cdot\frac{(2n)!}{4^n n!}\sqrt\pi
= \frac{(2n+2)!}{4^{n+1}(n+1)!}\sqrt\pi ,
\]
अंतिम चरण क्योंकि $\frac{(2n+2)!}{(2n)!} = (2n+2)(2n+1)$
और $\frac{2n+1}{2} = \frac{(2n+2)(2n+1)}{4(n+1)}$.

\textbf{4.} \cref{thm:b2:integration:gammaprops} देता है
$\Gamma''(x) = \int_0^\infty t^{x-1}\eu^{-t}(\ln t)^2\dd t $
(लीबनिज़ नियम के दो अनुप्रयोग, प्रभुत्व जैसे
प्रमेय का प्रमाण); इंटीग्रैंड है $\geq 0$ और समान रूप से नहीं
शून्य, तो $\Gamma'' > 0$: $\Gamma $ सख्ती से उत्तल है और
$\Gamma'$ सख्ती से बढ़ रहा है. तब से $\Gamma(1) = \Gamma(2)
= 1$, रोले प्रदान करता है $ x_0 \in \intoo12$ साथ $\Gamma'(x_0) =
0$; की सख्त एकरसता $\Gamma'$ बनाता है $ x_0$ यह अद्वितीय है
शून्य, साथ $\Gamma' < 0$ पहले और $\Gamma' > 0$ बाद में:
$\Gamma $ पर घट जाती है $\intoo0{x_0}$, बढ़ता जाता है
$\intoo{x_0}\infty $, और $ x_0$ अद्वितीय न्यूनतम है.

\textbf{5.} होने देना $ k \in \N $ और $ x \geq 3$; पूर्णांक चुनें
$ n $ साथ $ n + 1 \leq x < n + 2$ (इसलिए $ n \geq 1$). द्वारा
प्रश्न 4 की एकरसता (से मान्य) $ x_0 < 2$ पर): $\Gamma(x)
\geq \Gamma(n + 1) = n!$, जबकि $ x^k \leq (n+2)^k $. तब से
$\frac{n!}{(n+2)^k} \to \infty $ (फैक्टोरियल्स ने शक्तियों को हराया, वर्ष 1
आयतन), $\frac{\Gamma(x)}{x^k} \geq \frac{n!}{(n+2)^k} \to
\infty $ जैसा $ x \to \infty $: $ x^k = o(\Gamma(x))$.

\textbf{6.} पास में $0$ इंटीग्रैंड है $\sim t^{x-1}$
(\omterm{def:b2:integration:improper}{अभिसरण} आईएफएफ $ x > 0$), पास में $1$ यह है $\sim (1-t)^{y-1}$
(आईएफएफ $ y > 0$); दोनों तुलनाएँ सकारात्मक कार्यों के बीच हैं,
तो $ B(x,y)$ \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:series:def}{बिल्कुल} के लिए $ x, y > 0$. प्रतिस्थापन
$ t \mapsto 1 - t $ दो कारकों की अदला-बदली करता है: $ B(x,y) = B(y,x)$.

\textbf{7.} $ B(x,1) = \int_0^1 t^{x-1}\dd t = \frac1x $. पार्ट्स
पर $\intcc\varepsilon{1-\varepsilon}$ साथ $ u = (1-t)^y $, $ v =
\frac{t^x}{x}$:
\[
\int t^{x-1}(1-t)^{y}\dd t
= \Bigl[\frac{t^x(1-t)^y}{x}\Bigr]
+ \frac{y}{x}\int t^{x}(1-t)^{y-1}\dd t ;
\]
ब्रैकेट दोनों सिरों पर गायब हो जाता है $\varepsilon \to 0$ ($ x >
0$ पर $0$, $ y > 0$ पर $1$), जा रहा है $ B(x, y+1) = \frac
yx\,B(x+1, y)$.

\textbf{8.} तब से $ t + (1-t) = 1$:
\[
t^{x-1}(1-t)^{y-1} = t^{x}(1-t)^{y-1} + t^{x-1}(1-t)^{y},
\]
इसलिए $ B(x,y) = B(x+1,y) + B(x,y+1)$. प्रश्न 7 पढ़ता है $ B(x+1,y) =
\frac xy B(x,y+1)$; प्रतिस्थापित करना,
\[
B(x,y) = \Bigl(\frac xy + 1\Bigr)B(x,y+1)
= \frac{x+y}{y}\,B(x,y+1),
\]
यानी\ $ B(x,y+1) = \frac{y}{x+y}B(x,y)$; जुड़वाँ रिश्ता
प्रश्न 6 की समरूपता का अनुसरण करता है।

\textbf{9.} प्रेरण चालू $ n $ निश्चित पर $ m $: $ B(m,1) = \frac1m =
\frac{(m-1)!\,0!}{m!}$, और यदि सूत्र कायम रहता है $ n $,
\[
B(m, n+1) = \frac{n}{m+n}\,B(m,n)
= \frac{n}{m+n}\cdot\frac{(m-1)!(n-1)!}{(m+n-1)!}
= \frac{(m-1)!\,n!}{(m+n)!} .
\]
पुनर्लेखन: $ B(m,n) = \frac{(m-1)!(n-1)!}{(m+n-1)!} =
\bigl[(m+n-1)\binom{m+n-2}{m-1}\bigr]^{-1}$.

\textbf{10.} के दोनों तरफ $ B(x,n) =
\frac{\Gamma(x)\Gamma(n)}{\Gamma(x+n)}$ बराबर $\frac1x $ पर $ n =
1$ ($\Gamma(1) = 1$, $\Gamma(x+1) = x\Gamma(x)$). अगर वे सहमत हैं
पर $ n $, फिर वंश संबंध और कार्यात्मक द्वारा
समीकरण:
\[
B(x, n+1) = \frac{n}{x+n}\,B(x,n), \qquad
\frac{\Gamma(x)\Gamma(n+1)}{\Gamma(x+n+1)}
= \frac{n}{x+n}\cdot
\frac{\Gamma(x)\Gamma(n)}{\Gamma(x+n)} :
\]
दोनों अनुक्रम एक ही बीज से समान प्रत्यावर्तन का पालन करते हैं,
इसलिए सभी के लिए सहमत हैं $ n \in \N^*$ और सब $ x > 0$.

\textbf{11.} साथ $ t = \sin^2\theta $ ($\theta \in
\intoo0{\pi/2}$, $\dd t = 2\sin\theta\cos\theta\,\dd\theta $),
$ t^{x-1} = \sin^{2x-2}\theta $ और $(1-t)^{y-1} =
\cos^{2y-2}\theta $:
\[
B(x,y) = \int_0^{\pi/2}\sin^{2x-2}\theta\,\cos^{2y-2}\theta
\cdot 2\sin\theta\cos\theta\,\dd\theta
= 2\int_0^{\pi/2}\sin^{2x-1}\theta\,\cos^{2y-1}\theta\,
\dd\theta .
\]

\textbf{12.} लेना $ y = \frac12$ (कोसाइन कारक को मारना) और
$2x - 1 = n $: $ B\bigl(\frac{n+1}2, \frac12\bigr) = 2W_n $, अर्थात\
$ W_n = \frac12 B\bigl(\frac{n+1}2,\frac12\bigr)$. अवतरण
पहले वेरिएबल में संबंध देता है
\[
\frac{W_n}{W_{n-2}}
= \frac{B\bigl(\frac{n-1}2 + 1, \frac12\bigr)}
{B\bigl(\frac{n-1}2, \frac12\bigr)}
= \frac{\frac{n-1}2}{\frac{n-1}2 + \frac12}
= \frac{n-1}{n} :
\]
वालिस की पुनरावृत्ति, इस बार भागों द्वारा कोई एकीकरण नहीं
साइन्स पर --- भाग II ने एक बार और हमेशा के लिए काम किया।

\textbf{13.} $ B\bigl(\frac12,\frac12\bigr) = 2W_0 =
2\cdot\frac\pi2 = \pi $, जबकि
$\Gamma\bigl(\frac12\bigr)^2/\Gamma(1) = (\sqrt\pi)^2 = \pi $:
यूलर का सूत्र कायम है $\bigl(\frac12,\frac12\bigr)$.

\textbf{14.} बार-बार दोहराना $ W_{2n} = \frac{2n-1}{2n}W_{2n-2}$ से
$ W_0 = \frac\pi2$:
\[
W_{2n} = \frac\pi2\prod_{k=1}^{n}\frac{2k-1}{2k}
= \frac\pi2\cdot\frac{(2n)!}{4^n(n!)^2},
\]
तब से $\prod(2k-1) = \frac{(2n)!}{2^n n!}$ और $\prod 2k = 2^n
n!$. इसलिए, प्रश्न 3 का उपयोग करते हुए:
\[
B\Bigl(n+\frac12, \frac12\Bigr) = 2W_{2n}
= \pi\,\frac{(2n)!}{4^n(n!)^2}
= \frac{(2n)!\sqrt\pi}{4^n n!}\cdot\frac{\sqrt\pi}{n!}
= \frac{\Gamma\bigl(n+\frac12\bigr)\Gamma\bigl(\frac12\bigr)}
{\Gamma(n+1)} .
\]
अब ठीक करो $ x \in \frac12\N^*$. यूलर का सूत्र कायम है $(x,
\frac12)$: के लिए $ x $ पूर्णांक यह प्रश्न 10 है (समरूपता के साथ),
के लिए $ x = n + \frac12$ यह ऊपर का डिस्प्ले है। के दोनों तरफ
यूलर का सूत्र डीसेंट रिकर्सन का पालन करता है $ y \mapsto y + 1$
(बाईं ओर प्रश्न 8, दाईं ओर कार्यात्मक समीकरण,
जैसा कि प्रश्न 10 में है): प्रेरण सूत्र को प्रसारित करता है $ y =
\frac12$ और $ y = 1$ प्रत्येक को $ y \in \frac12\N^*$. यूलर का
इसलिए सूत्र जब भी लागू होता है $2x, 2y \in \N^*$.

\textbf{15.} साथ $ u = \frac{t}{1-t}$, यानी\ $ t =
\frac{u}{1+u}$, $1 - t = \frac{1}{1+u}$, $\dd t =
\frac{\dd u}{(1+u)^2}$:
\[
B(x,y) = \int_0^\infty
\Bigl(\frac{u}{1+u}\Bigr)^{x-1}
\Bigl(\frac{1}{1+u}\Bigr)^{y-1}
\frac{\dd u}{(1+u)^2}
= \int_0^\infty \frac{u^{x-1}}{(1+u)^{x+y}}\,\dd u .
\]
पर $ x = y = \frac12$, साथ $ u = v^2$:
\[
\int_0^\infty \frac{u^{-1/2}}{1+u}\dd u
= \int_0^\infty \frac{2\,\dd v}{1+v^2} = \pi
= B\Bigl(\frac12,\frac12\Bigr) . \checkmark
\]

\textbf{16.} भागों द्वारा एक एकीकरण, के लिए $1 \leq k \leq n $
और $ s > 0$ ($ u = (1 - t/n)^k $, $ v = t^s/s $; सीमा शर्तें
गायब हो जाना):
\[
\int_0^n \Bigl(1-\frac tn\Bigr)^{\!k} t^{s-1}\dd t
= \frac{k}{ns}\int_0^n \Bigl(1-\frac tn\Bigr)^{\!k-1}
t^{s}\dd t .
\]
से शुरू $ k = n $, $ s = x $ और पुनरावृत्त $ n $ समय:
\[
\int_0^n \Bigl(1-\frac tn\Bigr)^{\!n} t^{x-1}\dd t
= \frac{n(n-1)\cdots1}{n^n\,x(x+1)\cdots(x+n-1)}
\int_0^n t^{x+n-1}\dd t
= \frac{n!}{n^n}\cdot
\frac{n^{x+n}}{x(x+1)\cdots(x+n)} ,
\]
जो है $\dfrac{n!\,n^x}{x(x+1)\cdots(x+n)}$.

\textbf{17.} द्वारा \cref{exo:b2:integration:7} बायां भाग झुकता है
को $\Gamma(x)$ (प्रमुख के साथ प्रभुत्व \omterm{def:b2:integration:improper}{अभिसरण}
$ t^{x-1}\eu^{-t}$); दाईं ओर गॉस का भागफल है:
\[
\Gamma(x) = \lim_{n\to\infty}
\frac{n!\,n^x}{x(x+1)\cdots(x+n)} .
\]

\textbf{18.} प्रश्न 16 के भागफल में लघुगणक लेना
$ G_n(x)$ और बंटवारा $\ln(x+k) = \ln k + \ln(1 + x/k)$ के लिए $ k
\geq 1$:
\[
\ln G_n(x) = x\ln n - \ln x - \sum_{k=1}^n
\ln\Bigl(1+\frac xk\Bigr)
= -\ln x + x(\ln n - H_n)
+ \sum_{k=1}^n\Bigl(\frac xk -
\ln\Bigl(1+\frac xk\Bigr)\Bigr).
\]
के लिए $ u \geq 0$, $ u - \frac{u^2}2 \leq \ln(1+u) \leq u $, तो
सामान्य शब्द में निहित है $\intcc{0}{x^2/(2k^2)}$: शृंखला
\omterm{def:b2:integration:improper}{अभिसरण} (इसके साथ तुलना $\sum k^{-2}$). तब से $\ln n - H_n
\to -\gamma $ (\cref{ex:b2:comparison:harmonic}) और $\ln G_n(x)
\to \ln\Gamma(x)$ (प्रश्न 17 और निरंतरता का $\ln $):
\[
\ln\Gamma(x) = -\ln x - \gamma x
+ \sum_{k=1}^\infty\Bigl(\frac xk -
\ln\Bigl(1+\frac xk\Bigr)\Bigr) .
\]

\textbf{19.} पर $ x = \frac12$, हर है
$\prod_{k=0}^n\bigl(k+\frac12\bigr) =
\frac{(2n+1)!}{2^{2n+1}n!}$ (आधे हिस्सों को गुणा करें), इसलिए
\[
G_n\Bigl(\frac12\Bigr)
= \frac{n!\,\sqrt n\;2^{2n+1}n!}{(2n+1)!}
= \frac{2\sqrt n\;4^n}{(2n+1)\binom{2n}{n}} .
\]
साथ $\binom{2n}{n} \sim \frac{4^n}{\sqrt{\pi n}}$
(\cref{ex:b2:comparison:centralbinomial}):
\[
G_n\Bigl(\frac12\Bigr)
\sim \frac{2\sqrt n\,\sqrt{\pi n}}{2n+1}
\longrightarrow \sqrt\pi = \Gamma\Bigl(\frac12\Bigr) .
\]
केंद्रीय द्विपद गुणांक $\sqrt\pi $ (जो से आया है
वालिस, इसलिए स्टर्लिंग स्थिरांक से) और गाऊसी
अभिन्न का $\sqrt\pi $ एक ही संख्या हैं.

\textbf{20.} पहचान प्रत्यक्ष बीजगणित है: गुणा
$\frac{n!\,n^x}{x(x+1)\cdots(x+n)}$ द्वारा $\frac{nx}{x+n+1}$ और
अवशोषित $ x $ उत्पाद में, $ n $ में $ n^x $. दे $ n \to
\infty $: बाईं ओर की ओर झुकाव होता है $\Gamma(x+1)$ (गॉस एट $ x+1$),
दाहिनी ओर $\Gamma(x)\cdot x\cdot 1$ तब से
$\frac{n}{x+n+1} \to 1$: $\Gamma(x+1) = x\Gamma(x)$ ---
भागों द्वारा एक भी एकीकरण के बिना पुनर्प्राप्त किया गया।

\textbf{21.} साथ $ u = t^a $, $ t = u^{1/a}$, $\dd t = \frac1a
u^{1/a - 1}\dd u $:
\[
\int_0^\infty \eu^{-t^a}\dd t
= \frac1a\int_0^\infty u^{\frac1a - 1}\eu^{-u}\dd u
= \frac1a\,\Gamma\Bigl(\frac1a\Bigr)
= \Gamma\Bigl(1 + \frac1a\Bigr) .
\]
जैसा $ a \to +\infty $ (किसी भी क्रम के साथ): $\eu^{-t^a} \to 1$ के लिए
$0 < t < 1$, $\to \eu^{-1}$ पर $ t = 1$, $\to 0$ के लिए $ t > 1$;
के लिए $ a \geq 2$ का प्रभुत्व $\mathbf 1_{t \leq 1} +
\eu^{-t^2}\mathbf 1_{t > 1}$ ($ t^a \geq t^2$ के लिए $ t \geq 1$),
अभिन्न. प्रभुत्व \omterm{def:b2:integration:improper}{अभिसरण}: अभिन्न की ओर रुझान होता है
$\int_0^1 1\,\dd t = 1$ --- जैसा कि होना चाहिए, चूँकि $\Gamma(1 +
\frac1a) \to \Gamma(1) = 1$ द्वारा निरंतरता.

\textbf{22.} साथ $ u = t^n $, $\dd t = \frac1n u^{1/n - 1}\dd
u $:
\[
\int_0^1 \frac{\dd t}{\sqrt{1-t^n}}
= \frac1n\int_0^1 u^{\frac1n-1}(1-u)^{-1/2}\dd u
= \frac1n\,B\Bigl(\frac1n, \frac12\Bigr) .
\]
$ n = 1$: $ B\bigl(1,\frac12\bigr) = B\bigl(\frac12,1\bigr) = 2$,
मेल मिलाना $\int_0^1\frac{\dd t}{\sqrt{1-t}} = 2$. $ n = 2$:
$\frac12 B\bigl(\frac12,\frac12\bigr) = \frac\pi2 = \arcsin 1$.
के लिए $ n = 4$ मूल्य $\frac14 B\bigl(\frac14,\frac12\bigr)$ है
लेम्निस्केट स्थिरांक: कोई प्राथमिक बंद रूप नहीं।

\textbf{23.} कार्यात्मक समीकरण को दोहराना:
\[
\frac{1}{\Gamma(x)}\int_0^\infty t^{x+k-1}\eu^{-t}\dd t
= \frac{\Gamma(x+k)}{\Gamma(x)}
= (x+k-1)(x+k-2)\cdots x ,
\]
के साथ बढ़ती फैक्टोरियल $ k $ कारक. पर $ x = 1$:
$\Gamma(1+k)/\Gamma(1) = k!$, के क्षण $\eu^{-t}$ से
\cref{exo:b2:integration:2}.

\textbf{24.} विकल्प $ t = \frac{1+s}2$ ($ s \in
\intoo{-1}1$, $\dd t = \frac{\dd s}2$, $ t(1-t) =
\frac{1-s^2}4$):
\[
B(x,x) = \int_{-1}^{1}\Bigl(\frac{1-s^2}{4}\Bigr)^{x-1}
\frac{\dd s}{2}
= 4^{1-x}\int_0^1 (1-s^2)^{x-1}\dd s
\]
(इंटीग्रैंड सम है)। तब $ s = \sqrt v $ ($\dd s =
\frac{\dd v}{2\sqrt v}$):
\[
B(x,x) = \frac{4^{1-x}}{2}\int_0^1
v^{-1/2}(1-v)^{x-1}\dd v
= 2^{1-2x}\,B\Bigl(\frac12, x\Bigr) .
\]
के लिए $2x \in \N^*$ देखने में हर तर्क निहित है
$\frac12\N^*$, इसलिए यूलर का सूत्र (प्रश्न 14) दोनों पर लागू होता है
पक्ष:
\[
\frac{\Gamma(x)^2}{\Gamma(2x)}
= 2^{1-2x}\,
\frac{\Gamma\bigl(\frac12\bigr)\Gamma(x)}
{\Gamma\bigl(x+\frac12\bigr)}
\quad\Longleftrightarrow\quad
\Gamma(x)\,\Gamma\Bigl(x+\frac12\Bigr)
= 2^{1-2x}\sqrt\pi\;\Gamma(2x) .
\]
पर सीधे जांच करें $ x = n $: बाईं ओर है $(n-1)!\cdot
\frac{(2n)!\sqrt\pi}{4^n n!} = \frac{(2n)!\sqrt\pi}{4^n n}$,
दाहिनी ओर $2\cdot4^{-n}\sqrt\pi\,(2n-1)! =
\frac{(2n)!\sqrt\pi}{4^n n}$: बराबर।

\textbf{25.} (i) उत्पादित भागों द्वारा एकीकरण $ B(x,y+1) =
\frac yx B(x+1,y)$, एक ही पहचान जिससे हर एक
वंश संबंध, पूर्णांक और अर्ध-पूर्णांक मान, और
वालिस सभी प्रवाह की पुनरावृत्ति करता है। (ii) प्रभुत्वपूर्ण \omterm{def:b2:integration:improper}{अभिसरण} बदल गया
प्राथमिक अभिन्न अंग $\int_0^n(1-t/n)^n t^{x-1}$ में
$\Gamma(x)$ (गॉस का सूत्र, प्रश्न 17) और गणना की
सीमा $ a \to \infty $ प्रश्न 21 में। (iii) तुलना से
अध्याय हमने यूलर का स्थिरांक आयात किया ($\ln n - H_n \to
-\gamma $, प्रश्न 18) और केंद्रीय द्विपद स्पर्शोन्मुखता
(प्रश्न 19) --- अर्थात\ भेष में स्टर्लिंग का फार्मूला। (iv)
यूलर का सूत्र $ B(x,y) = \Gamma(x)\Gamma(y)/\Gamma(x+y)$ है
अब के लिए सिद्ध हो गया $ y \in \N^*$ साथ $ x > 0$ मनमाना (प्रश्न
10) और सभी अर्ध-पूर्णांक युग्मों के लिए (प्रश्न 14); सामान्य
मामला $ x, y > 0$ की दोहरी-अभिन्न गणना की प्रतीक्षा कर रहा है
अनेक समाकलन पर अध्याय, जो गुणनखंडन करता है
$\Gamma(x)\Gamma(y)$ एक चौथाई-तल पर।
\end{solution}
