\begin{solution}{exo:b2:surfaces:1}
साथ $\sigma(\theta, r) = (r\cos\theta,\ r\sin\theta,\ r)$:
\[
\sigma_\theta = (-r\sin\theta,\ r\cos\theta,\ 0),
\qquad
\sigma_r = (\cos\theta,\ \sin\theta,\ 1),
\]
के लिए स्वतंत्र $ r > 0$. \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल} पर $ M_0 =
\sigma(\theta_0, r_0)$ से गुजरता है $ M_0$ निर्देशों के साथ
$\sigma_\theta, \sigma_r $. अब $ M_0 - O = r_0(\cos\theta_0,
\sin\theta_0, 1) = r_0\,\sigma_r(\theta_0, r_0)$ स्वयं एक है
स्पर्शरेखा दिशा: शीर्ष $ O $ पर स्थित है \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल}. (यह
यह शंकु का सामान्य व्यवहार है: वे रेखाओं द्वारा शासित होते हैं
शीर्ष, और ए \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल} के माध्यम से सत्तारूढ़ रेखा शामिल है
स्पर्शरेखा बिंदु.)
\end{solution}

\begin{solution}{exo:b2:surfaces:2}
आवेदन करना \cref{prop:b2:surfaces:gradient} को $ f(x,y,z) =
\frac{x^2}{a^2} + \frac{y^2}{b^2} + \frac{z^2}{c^2}$: $\nabla
f(x_0,y_0,z_0) = 2\bigl(\frac{x_0}{a^2}, \frac{y_0}{b^2},
\frac{z_0}{c^2}\bigr) \neq 0$ सतह पर। \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल} है
\[
\frac{x_0}{a^2}(x - x_0) + \frac{y_0}{b^2}(y - y_0)
+ \frac{z_0}{c^2}(z - z_0) = 0,
\qquad\text{i.e.}\qquad
\frac{x_0\,x}{a^2} + \frac{y_0\,y}{b^2} + \frac{z_0\,z}{c^2} = 1,
\]
उसका उपयोग करना $(x_0, y_0, z_0)$ दीर्घवृत्ताकार समीकरण को संतुष्ट करता है ---
गोले को सामान्यीकृत करने वाला ``वर्गों को विभाजित करें'' नियम $\langle
M_0, M \rangle = R^2$.
\end{solution}

\begin{solution}{exo:b2:surfaces:3}
$\sigma_u = (-v\sin u,\ v\cos u,\ a)$ और $\sigma_v = (\cos u,\
\sin u,\ 0)$, इसलिए
\[
E = v^2 + a^2, \qquad F = 0, \qquad G = 1,
\qquad
\sqrt{EG - F^2} = \sqrt{v^2 + a^2} > 0
\]
(हेलिकॉइड अपनी धुरी सहित हर जगह नियमित है $ v =
0$). \omterm{def:b2:surfaces:area}{क्षेत्र} टुकड़े का:
\[
\mathcal{A} = \int_0^{2\pi}\!\!\int_0^1
\sqrt{v^2 + a^2}\;\dd v\,\dd u
= 2\pi\cdot\frac12\Bigl[v\sqrt{v^2 + a^2}
+ a^2\ln\bigl(v + \sqrt{v^2 + a^2}\bigr)\Bigr]_0^1 ,
\]
यानी\ $\mathcal{A} = \pi\Bigl(\sqrt{1 + a^2} +
a^2\ln\frac{1 + \sqrt{1 + a^2}}{a}\Bigr)$.
\end{solution}

\begin{solution}{exo:b2:surfaces:4}
डेरिवेटिव:
\[
\sigma_\theta = \bigl(-(R + r\cos\psi)\sin\theta,\
(R + r\cos\psi)\cos\theta,\ 0\bigr),
\qquad
\sigma_\psi = \bigl(-r\sin\psi\cos\theta,\
-r\sin\psi\sin\theta,\ r\cos\psi\bigr).
\]
तब
\[
E = (R + r\cos\psi)^2,
\qquad
F = r\sin\psi\cos\psi\,(R + r\cos\psi)
\bigl(\sin\theta\cos\theta - \sin\theta\cos\theta\bigr) = 0,
\qquad
G = r^2 ,
\]
इसलिए $\sqrt{EG - F^2} = r(R + r\cos\psi) \geq r(R - r) > 0$:
हर जगह नियमित. \omterm{def:b2:surfaces:area}{क्षेत्र}:
\[
\mathcal{A}
= \int_0^{2\pi}\!\!\int_0^{2\pi} r(R + r\cos\psi)
\,\dd\theta\,\dd\psi
= 2\pi r \Bigl(2\pi R + r\int_0^{2\pi}\cos\psi\,\dd\psi\Bigr)
= 4\pi^2 R r ,
\]
 $\cos\psi $ शून्य से एकीकृत होने वाला पद --- पप्पस प्रमेय: \omterm{def:b2:surfaces:area}{क्षेत्र}
$=$ (\omterm{def:b2:curves:length}{लंबाई} घुमाए गए वृत्त का) $\times $ (दूरी तय की गई
इसका केंद्र)।
\end{solution}

\begin{solution}{exo:b2:surfaces:5}
के लिए $\sigma(x, y) = (x, y, f(x,y))$: $\sigma_x = (1, 0, f_x)$,
$\sigma_y = (0, 1, f_y)$, इसलिए $ E = 1 + f_x^2$, $ F = f_xf_y $, $ G = 1
+ f_y^2$ और
\[
EG - F^2 = (1 + f_x^2)(1 + f_y^2) - f_x^2f_y^2
= 1 + f_x^2 + f_y^2 ,
\]
कहा गया है \omterm{def:b2:surfaces:area}{क्षेत्र} सूत्र. के लिए $ f = \frac12(x^2 + y^2)$ पर
यूनिट डिस्क: $1 + f_x^2 + f_y^2 = 1 + x^2 + y^2$, और ध्रुवीय में
निर्देशांक ($ x = \rho\cos\alpha $, $ y = \rho\sin\alpha $, जैकोबियन
$\rho $, \cref{ch:b2:multint}):
\[
\mathcal{A}
= \int_0^{2\pi}\!\!\int_0^1 \sqrt{1 + \rho^2}\;\rho
\,\dd\rho\,\dd\alpha
= 2\pi\Bigl[\tfrac13(1 + \rho^2)^{3/2}\Bigr]_0^1
= \frac{2\pi}{3}\bigl(2\sqrt2 - 1\bigr).
\]
\end{solution}

\begin{solution}{exo:b2:surfaces:6}
से \cref{ex:b2:surfaces:spherearea}की गणना, $ E =
R^2\cos^2\varphi $, $ F = 0$, $ G = R^2$, तो द्वारा
\cref{prop:b2:surfaces:curvelength}
\[
L = \int_a^b R\sqrt{\cos^2\varphi(t)\,\theta'(t)^2
+ \varphi'(t)^2}\;\dd t .
\]
अंतिमबिंदु होने दो $(\theta_0, \varphi_1)$ और $(\theta_0,
\varphi_2)$, $\varphi_1 < \varphi_2$. किसी भी जुड़ने वाले वक्र के लिए,
\[
L \geq \int_a^b R\,\abs{\varphi'(t)}\,\dd t
\geq R\,\Bigl|\int_a^b \varphi'(t)\,\dd t\Bigr|
= R\,(\varphi_2 - \varphi_1),
\]
गैर-नकारात्मक को छोड़ना $\cos^2\varphi\,\theta'^2$ शब्द और प्रयोग
अभिन्नों के लिए त्रिभुज असमानता। मेरिडियन आर्क
$\theta \equiv \theta_0$, $\varphi $ से बढ़ रहा है $\varphi_1$ को
$\varphi_2$, है \omterm{def:b2:curves:length}{लंबाई} \omterm{def:b2:series:def}{बिल्कुल} $ R(\varphi_2 - \varphi_1)$: यह है
सबसे छोटा. (मेरिडियन महान वृत्त हैं; यह पहला है,
इस तथ्य का प्राथमिक मामला कि \omterm{def:b2:surfaces:area}{क्षेत्र} की भू-भौतिकी महान है
वृत्त.)
\end{solution}

\begin{solution}{exo:b2:surfaces:7}
एक वक्र ठीक करें $\gamma $ के नाम आहरित $ S $ और चलो $ g(t) = \norm{\gamma(t)
- \Omega}^2$. तब $ g'(t) = 2\langle \gamma'(t),\ \gamma(t) -
\Omega\rangle $. सामान्य रेखा पर $ M = \gamma(t)$ से गुजरता है
$\Omega $ परिकल्पना द्वारा, इसलिए $\gamma(t) - \Omega $ एक है \emph{सामान्य}
वेक्टर, ओर्थोगोनल से \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल}, विशेष रूप से को
वेग $\gamma'(t)$ (\cref{prop:b2:surfaces:velocity}): $ g' =
0$, और $ g $ प्रत्येक खींचे गए वक्र के साथ स्थिर है।

अब सेट $ S_c = \{M \in S : \norm{M - \Omega}^2 = c\}$ है
में बंद $ S $; ये भी \omterm{def:b2:metric:topology}{खुला} में $ S $: इसके किसी भी बिंदु के निकट, $ S $
एक नियमित ग्राफ़ है, इसलिए कोई भी निकटवर्ती बिंदु $ S $ इसके साथ जुड़ गया है
एक खींचा हुआ वक्र (एक उठा हुआ खंड), जिसके अनुदिश $ g $ स्थिर है. जैसा
$ S $ है \omterm{def:b2:metric:connected}{जुड़े हुए} और $ S_c $ अधिकार के लिए गैररिक्त $ c $, $ S = S_c
\subseteq $ केंद्र का \omterm{def:b2:surfaces:area}{क्षेत्र} $\Omega $ और त्रिज्या $\sqrt c $
(\cref{ch:b2:metric}: संयुक्तता तर्क)।
\end{solution}

\begin{solution}{exo:b2:surfaces:8}
लिखना $\Phi(u', v') = (u, v)$. शृंखला नियम से,
\[
\tilde\sigma_{u'} = \frac{\partial u}{\partial u'}\sigma_u
+ \frac{\partial v}{\partial u'}\sigma_v,
\qquad
\tilde\sigma_{v'} = \frac{\partial u}{\partial v'}\sigma_u
+ \frac{\partial v}{\partial v'}\sigma_v ,
\]
के आंशिक $\sigma $ पर मूल्यांकन किया गया $\Phi(u',v')$. का विस्तार करना
क्रॉस उत्पाद द्विरेखीय रूप से और उपयोग करना $\sigma_u \wedge \sigma_u =
\sigma_v \wedge \sigma_v = 0$, $\sigma_v \wedge \sigma_u =
-\sigma_u \wedge \sigma_v $:
\[
\tilde\sigma_{u'} \wedge \tilde\sigma_{v'}
= \Bigl(\frac{\partial u}{\partial u'}
\frac{\partial v}{\partial v'}
- \frac{\partial v}{\partial u'}
\frac{\partial u}{\partial v'}\Bigr)\,
\sigma_u \wedge \sigma_v
= \det J_\Phi \cdot (\sigma_u \wedge \sigma_v)\circ\Phi .
\]
लेना मानदंड पहचान देता है. फिर, चर-परिवर्तन द्वारा
सूत्र (\cref{ch:b2:multint}) मानचित्र पर लागू किया गया $\Phi $ पर $ K' =
\Phi^{-1}(K)$:
\[
\iint_{K'} \norm{\tilde\sigma_{u'} \wedge \tilde\sigma_{v'}}
\,\dd u'\dd v'
= \iint_{K'} \norm{\sigma_u \wedge \sigma_v}\circ\Phi\;
\abs{\det J_\Phi}\,\dd u'\dd v'
= \iint_K \norm{\sigma_u \wedge \sigma_v}\,\dd u\,\dd v :
\]
दोनों पैरामीट्रिज़ेशन समान निर्दिष्ट करते हैं \omterm{def:b2:surfaces:area}{क्षेत्र} के एक ही टुकड़े के लिए
सतह.
\end{solution}

\begin{solution}{exo:b2:surfaces:9}
गोलाकार चार्ट में, $ z = R\sin\varphi $, और \omterm{def:b2:surfaces:area}{क्षेत्र}
से मेल खाता है $\varphi_1 \leq \varphi \leq \varphi_2$ साथ
$ z_i = R\sin\varphi_i $. साथ \omterm{def:b2:surfaces:area}{क्षेत्र} तत्व
$ R^2\cos\varphi\,\dd\theta\,\dd\varphi $
(\cref{ex:b2:surfaces:spherearea}):
\[
\mathcal A = \int_{\varphi_1}^{\varphi_2}\!\!\int_0^{2\pi}
R^2\cos\varphi\,\dd\theta\,\dd\varphi
= 2\pi R^2(\sin\varphi_2 - \sin\varphi_1)
= 2\pi R\,(z_2 - z_1) .
\]
परिणाम केवल ऊंचाई पर निर्भर करता है $ z_2 - z_1$: के टुकड़े
समान मोटाई समान ले जाती है \omterm{def:b2:surfaces:area}{क्षेत्र}ों, चाहे भूमध्य रेखा पर काटा जाए
या ध्रुव पर --- आर्किमिडीज़ का हैट-बॉक्स प्रमेय, और कारण
पार्श्व \omterm{def:b2:surfaces:area}{क्षेत्र} परिचालित सिलेंडर का ($2\pi R \cdot
2R = 4\pi R^2$) के बराबर है \omterm{def:b2:surfaces:area}{क्षेत्र} गोले का.
\end{solution}

\begin{solution}{exo:b2:surfaces:11}
$\sigma_u = (-\sin u, \cos u, 0)$, $\sigma_v = (0, 0, 1)$: $ E
= 1$, $ F = 0$, $ G = 1$. द्वारा
\cref{prop:b2:surfaces:curvelength}  \omterm{def:b2:curves:length}{लंबाई} एक खींचा का
वक्र है $\int\sqrt{u'^2 + v'^2}\,\dd t $ --- द \omterm{def:b2:curves:length}{लंबाई} इसके
पैरामीटर छाया $(u(t), v(t))$ विमान में: चार्ट एक है
स्थानीय आइसोमेट्री (सिलेंडर का अनियंत्रित होना)। हेलिक्स
$ t \mapsto (\cos t, \sin t, ct)$, $ t \in \intcc0{2\pi}$, है
से खंड को छायांकित करें $(0,0)$ को $(2\pi, 2\pi c)$, का \omterm{def:b2:curves:length}{लंबाई}
$2\pi\sqrt{1 + c^2}$. से कोई भी खींचा गया वक्र $(1,0,0)$ को $(1,
0, 2\pi c)$ एक \omterm{def:b2:metric:complete}{पूरा} चक्कर लगाना एक है \omterm{def:b2:metric:continuity}{निरंतर} छाया
शामिल होना $(0, 0)$ को $(2\pi, 2\pi c)$, हवाई जहाज़ का \omterm{def:b2:curves:length}{लंबाई} $\geq $
सीधा खंड; तब से \omterm{def:b2:curves:length}{लंबाई} सहमत हूँ, हेलिक्स है
सबसे छोटा.
\end{solution}

\begin{solution}{exo:b2:surfaces:12}
समारोह $ g(M) = \norm{M}^2$ है \omterm{def:b2:metric:continuity}{निरंतर} पर \omterm{def:b2:metric:compact}{कॉम्पैक्ट}
$ S $, इसलिए कुछ स्थानों पर यह अपनी अधिकतम सीमा प्राप्त कर लेता है $ M_0$. प्रत्येक वक्र के लिए
$\gamma $ के नाम आहरित $ S $ साथ $\gamma(0) = M_0$, समारोह $ t
\mapsto \norm{\gamma(t)}^2$ पर अधिकतम है $ t = 0$, तो यह है
व्युत्पन्न $2\langle\gamma'(0), M_0\rangle $ गायब हो जाता है: $ M_0$ है
प्रत्येक स्पर्शरेखा वेक्टर के लिए ऑर्थोगोनल, यानी\ सामान्य $ S $ पर
$ M_0$. जैसा $\nabla f(M_0) \neq 0$ सामान्य रेखा को भी निर्देशित करता है
(\cref{prop:b2:surfaces:gradient}), $\nabla f(M_0)$ और
$\vect{OM_0}$ संरेख हैं. दीर्घवृत्ताकार, रेडियलिटी के लिए
मतलब
\[
\Bigl(\frac{x_0}{a^2}, \frac{y_0}{b^2},
\frac{z_0}{c^2}\Bigr) = \mu\,(x_0, y_0, z_0) :
\]
प्रत्येक समन्वय संतुष्ट करता है $ x_0(\frac1{a^2} - \mu) = 0$, वगैरह।;
तब से $ a^{-2}, b^{-2}, c^{-2}$ अलग-अलग हैं, ज़्यादा से ज़्यादा एक
निर्देशांक अशून्य है, और सतह पर समाधान हैं
छह अक्ष समापनबिंदु $(\pm a, 0, 0)$, $(0, \pm b, 0)$, $(0,
0, \pm c)$. सबसे दूर बिंदु हैं $(\pm a, 0, 0)$, पर
दूरी $ a = \max(a,b,c)$.
\end{solution}

\begin{solution}{pb:b2:surfaces:1}
\textbf{1.} का विस्तार करना, और समरूपता का उपयोग करना $ A $
($ t^{\mathsf T}\!APY = (At)^{\mathsf T}PY $):
\[
q(PY + t) = Y^{\mathsf T}P^{\mathsf T}\!APY +
2\,(At + b)^{\mathsf T}PY + \bigl(t^{\mathsf T}\!At +
2b^{\mathsf T}t + c\bigr),
\]
जो प्रदर्शित त्रिगुण है। $ A' = P^{\mathsf T}\!AP =
P^{-1}AP $ के समान है $ A $: वही \omterm{def:b2:reduction:charpoly}{विशेषता बहुपद},
\omterm{def:b2:reduction:eigen}{स्पेक्ट्रम}, रैंक, \omterm{thm:b2:structures:signature}{हस्ताक्षर}। अंत में $\{q = 0\} = \{sq = 0\}$
के लिए $ s \neq 0$, तो केवल समीकरण \emph{एक अदिश राशि तक} है
सेट से जुड़ा हुआ; द्वारा स्केलिंग $ s $ सभी को गुणा करता है eigenvalues
द्वारा $ s $.

\textbf{2.} वर्णक्रमीय प्रमेय प्रदान करता है $ P \in O(3)$ साथ
$ P^{\mathsf T}\!AP = \operatorname{diag}(\lambda_1, \lambda_2,
\lambda_3)$; प्रश्न 1 के साथ $ t = 0$ समीकरण को बदल देता है
$\sum\lambda_iy_i^2 + 2\sum\beta_iy_i + c = 0$, कहाँ $\beta =
P^{\mathsf T}b $.

\textbf{3.} के लिए $\lambda_i \neq 0$: $\lambda_iy_i^2 +
2\beta_iy_i = \lambda_i\bigl(y_i +
\frac{\beta_i}{\lambda_i}\bigr)^2 -
\frac{\beta_i^2}{\lambda_i}$; अनुवाद $ z_i = y_i +
\beta_i/\lambda_i $ (और $ z_i = y_i $ के लिए $ i > r $) देता है
\[
\sum_{i \leq r}\lambda_iz_i^2 + 2\sum_{i > r}\beta_iz_i +
c'' = 0, \qquad
c'' = c - \sum_{i\leq r}\frac{\beta_i^2}{\lambda_i}.
\]

\textbf{4.} $ q(2\Omega - X) = (2\Omega - X)^{\mathsf T}
A(2\Omega - X) + 2b^{\mathsf T}(2\Omega - X) + c $; विस्तार
और घटाना $ q(X)$, \omterm{pb:b2:surfaces:1}{द्विघात} पद रद्द करें और
\[
q(2\Omega - X) - q(X) = -4\,\langle A\Omega + b,\ X\rangle +
4\,\langle A\Omega + b,\ \Omega\rangle .
\]
अगर $ A\Omega + b = 0$ यह \omterm{def:b2:funcseq:def}{समान रूप से} गायब हो जाता है: बिंदु
प्रतिबिंब सुरक्षित रखता है $ q $, इसलिए \omterm{pb:b2:surfaces:1}{द्विघात}. इसके विपरीत,
``$ q(2\Omega - X) = q(X)$ सभी के लिए $ X $'' कहते हैं कि \omterm{def:b2:affine:subspace}{एफ़िन}
उपरोक्त फ़ंक्शन सभी पर गायब हो जाता है $\R^3$, जो इसे मजबूर करता है
रैखिक भाग $ A\Omega + b $ शून्य होना. तो केन्द्र $=$ के समाधान
$ A\Omega = -b $: एक गैररिक्त सेट यदि $ b \in \operatorname{im}A $
(एक \omterm{def:b2:affine:subspace}{एफ़िन} उपस्थान द्वारा निर्देशित $\ker A $), और एक बिंदु
यदि $ A $ उलटा है.

\textbf{5.} \omterm{thm:b2:structures:signature}{हस्ताक्षर} $(3,0)$ (सभी $\lambda_i > 0$): $\delta >
0$ देता है $\frac{x^2}{a^2} + \frac{y^2}{b^2} + \frac{z^2}{c^2}
= 1$ साथ $ a = \sqrt{\delta/\lambda_1}$, वगैरह।\ --- एक
दीर्घवृत्ताकार; $\delta = 0$: एकल बिंदु $ O $; $\delta < 0$:
खाली। \omterm{thm:b2:structures:signature}{हस्ताक्षर} $(2,1)$ ($\lambda_1, \lambda_2 > 0 >
\lambda_3$): $\delta > 0$: $\frac{x^2}{a^2} + \frac{y^2}{b^2}
- \frac{z^2}{c^2} = 1$, एक-शीट हाइपरबोलॉइड; $\delta =
0$: शंकु $\frac{x^2}{a^2} + \frac{y^2}{b^2} =
\frac{z^2}{c^2}$; $\delta < 0$: $\frac{z^2}{c^2} -
\frac{x^2}{a^2} - \frac{y^2}{b^2} = 1$, दो-पत्र
हाइपरबोलॉइड ($\abs z \geq c $: दो घटक)।

\textbf{6.} \omterm{pb:b2:surfaces:1}{द्विघात} भाग में मैट्रिक्स होता है $ A $ विकर्ण के साथ
$1$ और ऑफ-विकर्ण $2$: $ A = 2J - I $. तब से $ J $ है \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम}
$\{3, 0, 0\}$ (आइजन्वेक्टर $(1,1,1)$ के लिए $3$), $ A $ है
\omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} $\{5, -1, -1\}$, द \omterm{def:b2:reduction:eigen}{eigenvalue} $5$ द्वारा किया गया
$\R(1,1,1)$. घुमाए गए निर्देशांक में: $5u^2 - v^2 - w^2 =
1$, यानी\ $\frac{u^2}{1/5} - v^2 - w^2 = 1$: एक दो शीट
अतिपरवलयज, क्रांति का (बराबर)। eigenvalues $-1$) के बारे में
अक्ष $\R(1,1,1)$.

\textbf{7.} सतह है $\bigl(\frac xa - \frac
zc\bigr)\bigl(\frac xa + \frac zc\bigr) = \bigl(1 - \frac
yb\bigr)\bigl(1 + \frac yb\bigr)$. के लिए $(\lambda : \mu) \neq
(0:0)$ रेखा को परिभाषित करें
\[
D_{\lambda:\mu} :\quad
\lambda\Bigl(\frac xa - \frac zc\Bigr) = \mu\Bigl(1 - \frac
yb\Bigr), \qquad
\mu\Bigl(\frac xa + \frac zc\Bigr) = \lambda\Bigl(1 + \frac
yb\Bigr)
\]
(दो स्वतंत्र \omterm{def:b2:affine:subspace}{एफ़िन} समीकरण: एक पंक्ति)। का गुणा करना
दो समीकरण प्रत्येक बिंदु को दर्शाते हैं $ D_{\lambda:\mu}$ पर झूठ बोलता है
सतह जब $\lambda\mu \neq 0$; मामले $\lambda = 0$
या $\mu = 0$ सीधे जाँच की जाती है (उदा.\ $\lambda = 0$: $ y =
b $, $\frac xa = -\frac zc $, जो समीकरण को संतुष्ट करता है)। द
दूसरा परिवार $ D'_{\lambda:\mu}$ दो दाहिने हाथ की अदला-बदली करता है
कारक.

\textbf{8.} हल करना $ M $ सतह पर। के लिए शर्तें $ M \in
D_{\lambda:\mu}$ फॉर्म ए $2\times2$ सजातीय रैखिक प्रणाली
में $(\lambda, \mu)$ किसका सिद्ध है
\[
\Bigl(\frac{x^2}{a^2} - \frac{z^2}{c^2}\Bigr) - \Bigl(1 -
\frac{y^2}{b^2}\Bigr) = 0
\]
ठीक इसलिए क्योंकि $ M $ पर स्थित है \omterm{pb:b2:surfaces:1}{द्विघात}: एक गैरतुच्छ
समाधान $(\lambda : \mu)$ मौजूद है. गुणांक मैट्रिक्स है
कभी शून्य नहीं (जो मजबूर करेगा $1 - \frac yb = 1 + \frac yb =
0$), तो इसकी रैंक है $1$ और समाधान तक अद्वितीय है
पैमाना: परिवार की \omterm{def:b2:series:def}{बिल्कुल} एक पंक्ति गुजरती है $ M $.
दूसरे परिवार के लिए भी यही बात लागू होती है, और दो पंक्तियाँ हैं
विशिष्ट (पर $(a, 0, 0)$ वे हैं $\{x = a,\ \frac zc = \frac
yb\}$ और $\{x = a,\ \frac zc = -\frac yb\}$): एक-पत्रक
हाइपरबोलॉइड दोगुना शासित है।

\textbf{9.} होने देना $ M = (x, y, z)$ हाइपरबोलॉइड पर, $\rho^2 =
\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1 + \frac{z^2}{c^2} \geq
1$. बात $ N = (x,\ y,\ \varepsilon c\rho)$ साथ
$\varepsilon $ का संकेत $ z $ संतुष्ट $\frac{x^2}{a^2} +
\frac{y^2}{b^2} - \frac{(c\rho)^2}{c^2} = 0$: $ N \in C $, और
\[
d(M, C) \leq \abs{z - \varepsilon c\rho}
= c\bigl(\rho - \sqrt{\rho^2 - 1}\bigr)
= \frac{c}{\rho + \sqrt{\rho^2 - 1}} .
\]
अगर $\norm M \to \infty $ तब $\rho \to \infty $ (तीनों
निर्देशांक के गुणजों से घिरे होते हैं $\rho $), इसलिए $ d(M, C)
\to 0$. खंड $ x = a $: $\frac{y^2}{b^2} -
\frac{z^2}{c^2} = 0$, प्रश्न की क्रॉसिंग लाइनों की \omterm{def:b2:quadratic:adjoint}{जोड़}ी
8 --- और इसी तरह पर $ x = -a $.

\textbf{10.} साथ $\lambda_3 = 0$, प्रश्न 3 के पत्ते
$\lambda_1z_1^2 + \lambda_2z_2^2 + 2\beta_3z_3 + c'' = 0$. में
ईजेनबासिस, $\operatorname{im}A = \operatorname{Vect}(e_1,
e_2)$, तो प्रश्न के अनुसार 4 केंद्र मौजूद हैं यदि $\beta_3 = 0$. अगर
$\beta_3 \neq 0$: अनुवाद $ z_3 \mapsto z_3 -
c''/(2\beta_3)$ स्थिरांक को हटाता है, छोड़ता है $\lambda_1z_1^2
+ \lambda_2z_2^2 + 2\beta_3z_3 = 0$, यानी\ $ z_3 = px^2 +
qy^2$ नाम बदलने के बाद: ए \emph{अण्डाकार परवलयिक} अगर
$\lambda_1\lambda_2 > 0$, ए \emph{हाइपरबोलिक पैराबोलॉइड} अगर
$\lambda_1\lambda_2 < 0$ --- और वास्तव में कोई केंद्र नहीं। अगर
$\beta_3 = 0$: समीकरण $\lambda_1z_1^2 + \lambda_2z_2^2 +
c'' = 0$ शामिल नहीं है $ z_3$: द \omterm{pb:b2:surfaces:1}{द्विघात} एक सिलेंडर है
संगत समतल शांकव --- अण्डाकार बेलन पर,
लाइन, या खाली सेट जब $\lambda_1\lambda_2 > 0$; अतिशयोक्तिपूर्ण
सिलेंडर या प्रतिच्छेदी विमानों की \omterm{def:b2:quadratic:adjoint}{जोड़}ी जब
$\lambda_1\lambda_2 < 0$ --- केंद्रों की एक पूरी श्रृंखला के साथ
$\{(z_1^*, z_2^*)\} \times \R $.

\textbf{11.} $ xy - z = 0$. स्थानापन्न $ x = \frac{u +
v}{\sqrt2}$, $ y = \frac{u - v}{\sqrt2}$ (द्वारा घूर्णन $\pi/4$):
$ xy = \frac{u^2 - v^2}2$, तो समीकरण बन जाता है $ z =
\frac12(u^2 - v^2)$: एक अतिशयोक्तिपूर्ण परवलयज --- काठी
आकृति का, कारक तक $\frac12$.

\textbf{12.} रेखा $\{x = x_0,\ z = x_0y\}$ (पैरामीट्रिज्ड
द्वारा $ y $) स्पष्ट रूप से झूठ बोलता है $ z = xy $, के रूप में करता है $\{y = y_0,\ z =
xy_0\}$; के माध्यम से $(x_0, y_0, x_0y_0)$ उन दोनों को पास करो.
विशिष्टता: यदि $ t \mapsto (x_0 + tv_1, y_0 + tv_2, z_0 +
tv_3)$ सतह पर रहता है, का गुणांक $ t^2$ में $(x_0
+ tv_1)(y_0 + tv_2) - z_0 - tv_3$ देता है $ v_1v_2 = 0$, इसलिए
$ v_1 = 0$ या $ v_2 = 0$, दो परिवारों में से एक में उतरना:
प्रत्येक बिंदु के माध्यम से प्रत्येक की एक पंक्ति --- दूसरी दोगुनी
शासन किया \omterm{pb:b2:surfaces:1}{द्विघात}.

\textbf{13.} वर्गों को \omterm{def:b2:metric:complete}{पूरा} करना: $(x - 1)^2 + (y + 2)^2 = 2$,
बिना किसी शर्त के $ z $: त्रिज्या का एक लंब वृत्तीय बेलन
$\sqrt2$ और अक्ष ऊर्ध्वाधर रेखा $\{(1, -2, z) : z \in
\R\}$ --- केंद्रों की एक पंक्ति, जैसा कि प्रश्न 10 भविष्यवाणी करता है।

\textbf{14.} साथ $\lambda_2 = \lambda_3 = 0$ कम हो गया
समीकरण है $\lambda_1z_1^2 + 2\beta_2z_2 + 2\beta_3z_3 + c''
= 0$. कर्नेल तल का एक घूर्णन $(z_2, z_3)$ को संरेखित करता है
रैखिक रूप: $2\beta_2z_2 + 2\beta_3z_3 = 2\beta w $ साथ
$\beta = \sqrt{\beta_2^2 + \beta_3^2}$. अगर $\beta \neq 0$,
अनुवाद $ w $ अवशोषित करना $ c''$: $\lambda_1z_1^2 + 2\beta w =
0$, ए \emph{परवलयिक सिलेंडर}; अगर $\beta = 0$:
$\lambda_1z_1^2 = -c''$ दो समांतर तल देता है ($ c''
\lambda_1 < 0$), एक दोहरा विमान ($ c'' = 0$), या खाली
सेट. के लिए $(x + y)^2 = z $: साथ $ u = \frac{x + y}{\sqrt2}$ द
समीकरण पढ़ता है $ z = 2u^2$: एक परवलयिक सिलेंडर, अपरिवर्तनीय
साथ में अनुवाद के \omterm{def:b2:diffcalc:differential}{अंतर}्गत $(1, -1, 0)$.

\textbf{15.} प्रत्येक \omterm{pb:b2:surfaces:1}{द्विघात} एक रोटेशन प्लस द्वारा किया जाता है

(रैंक 2) अण्डाकार पैराबोलॉइड, हाइपरबोलिक पैराबोलॉइड, अण्डाकार
सिलेंडर, लाइन, खाली सेट, अतिशयोक्तिपूर्ण सिलेंडर, \omterm{def:b2:quadratic:adjoint}{जोड़}ी
प्रतिच्छेदी तल; (रैंक 1) परवलयिक सिलेंडर, की \omterm{def:b2:quadratic:adjoint}{जोड़}ी

तीन खाली वेरिएंट अलग-अलग हैं \emph{\omterm{def:b2:affine:subspace}{एफ़िन} प्रकार} का

\textbf{20.} एक कठोर गति समीकरण के डेटा को बदल देती है
$ A \mapsto P^{\mathsf T}\!AP $ (वही eigenvalues) और

दीर्घवृत्ताकार के लिए क्रमपरिवर्तन और सामान्य सकारात्मक कारक ---
यदि अर्धअक्ष $(a, b, c)$ सहमत होना। स्नेहपूर्वक, कोई भी कर सकता है

{\lambda_i}}$), जो मिटा देता है eigenvalues और केवल छोड़ देता है
उनके संकेत: प्रत्येक दीर्घवृत्ताभ बन जाता है $ u^2 + v^2 + w^2 = 1$,
गोला. सिल्वेस्टर की जड़ता प्रमेय
(\cref{thm:b2:quadratic:sylvester}) \omterm{thm:b2:structures:signature}{हस्ताक्षर} की गारंटी देता है
किसी भी उलटे रैखिक परिवर्तन से बच जाता है: \omterm{def:b2:affine:subspace}{एफ़िन} के प्रकार
प्रश्न 15 वास्तव में भिन्न हैं।

में $(s, t)$ (इसका विस्तार करें \omterm{pb:b2:surfaces:1}{द्विघात} रूप द्विरेखीय रूप से), इसलिए
अनुभाग $\{q = 0\}$ संभवतः समतल का एक शंकु है
पतित. के लिए $ z = xy $ और विमान $ z = c $: $ xy = c $, ए
के लिए अतिपरवलय $ c \neq 0$, और के लिए $ c = 0$ दोनों समन्वय करते हैं
पंक्तियाँ --- मूल के माध्यम से निर्णयों की \omterm{def:b2:quadratic:adjoint}{जोड़}ी।

\textbf{22.} \emph{दीर्घवृत्ताभ}: घिरा हुआ, इसमें कोई रेखा नहीं है।
\emph{दो-शीट हाइपरबोलाइड} $\frac{z^2}{c^2} -
\frac{x^2}{a^2} - \frac{y^2}{b^2} = 1$: तक सीमित $ p + tv $;
 $ t^2$ गुणक $\frac{v_3^2}{c^2} - \frac{v_1^2}{a^2}
- \frac{v_2^2}{b^2}$ गायब हो जाना चाहिए, इसलिए $ v_3 \neq 0$ (अन्य $ v =
0$); $ t $ गुणांक देता है $\frac{p_3v_3}{c^2} =
\frac{p_1v_1}{a^2} + \frac{p_2v_2}{b^2}$, और कॉची-श्वार्ज़
पैदावार
\[
\frac{p_3^2}{c^2}
= \frac{c^2}{v_3^2}\Bigl(\frac{p_1v_1}{a^2} +
\frac{p_2v_2}{b^2}\Bigr)^2
\leq \frac{c^2}{v_3^2}\Bigl(\frac{p_1^2}{a^2} +
\frac{p_2^2}{b^2}\Bigr)\frac{v_3^2}{c^2}
= \frac{p_1^2}{a^2} + \frac{p_2^2}{b^2},
\]
अतः अचर पद है $\leq 0 \neq 1$: कोई पंक्ति नहीं।
\emph{अण्डाकार परवलयिक} $ z = \frac{x^2}{a^2} +
\frac{y^2}{b^2}$: द $ t^2$ गुणांक बल $ v_1 = v_2 =
0$, तो समीकरण रैखिक अचर है $ t $: कोई पंक्ति नहीं।
\emph{कोन} $ x^2 + y^2 = z^2$: इस पर एक पंक्ति संतुष्ट करती है $ q(p) =
q(v) = B(p, v) = 0$ लोरेंत्ज़ फॉर्म के लिए; में समानता
विमान कॉची--श्वार्ज़ $\abs{p_1v_1 + p_2v_2} = \abs{p_3v_3} =
\sqrt{(p_1^2 + p_2^2)(v_1^2 + v_2^2)}$ बल $(p_1, p_2)
\parallel (v_1, v_2)$ और तब $ p = kv $: सभी पंक्तियाँ गुजरती हैं
शीर्ष के माध्यम से --- एक परिवार। \emph{सिलेंडर}: के लिए
अण्डाकार और परवलयिक सिलेंडर $ t^2$ गुणांक बल
$ v_1 = v_2 = 0$ (केवल निर्णय); अतिशयोक्तिपूर्ण सिलेंडर के लिए
$\frac{x^2}{a^2} - \frac{y^2}{b^2} = 1$, $\frac{v_1}a = \pm
\frac{v_2}b $ साथ $ v_2 \neq 0$ के माध्यम से नेतृत्व करता है $ t $ गुणांक
को $\frac{p_1^2}{a^2} = \frac{p_2^2}{b^2}$, का खंडन करते हुए
स्थिर अवधि $1$: पुनः केवल ऊर्ध्वाधर निर्णय। तो
दोगुना शासन किया \omterm{pb:b2:surfaces:1}{द्विघात} सतहें \omterm{def:b2:series:def}{बिल्कुल} एक-शीट वाली हैं
हाइपरबोलॉइड और हाइपरबोलिक पैराबोलॉइड।

\textbf{23.} गॉस: $ x^2 - 2xy - 2zx = (x - y - z)^2 - y^2 -
z^2 - 2yz $, इसलिए
\[
q = (x - y - z)^2 - 4yz = (x - y - z)^2 + (y - z)^2 - (y +
z)^2 :
\]
चिन्हों के साथ तीन स्वतंत्र वर्ग $(+, +, -)$ --- \omterm{thm:b2:structures:signature}{हस्ताक्षर}
$(2, 1)$, मिलान प्रश्न 17, संख्या के साथ \omterm{def:b2:reduction:eigen}{eigenvalue}
गणना. गॉस ने मीट्रिक डेटा खो दिया: नए निर्देशांक
ऑर्थोनॉर्मल नहीं हैं, इसलिए eigenvalues (का आकार
हाइपरबोलॉइड, इसके अक्ष और उनके \omterm{def:b2:curves:length}{लंबाई}) चले गए; केवल
\omterm{def:b2:affine:subspace}{एफ़िन} प्रकार रहता है.

\textbf{24.} होने देना $ p $, $ n $, $ z $ सकारात्मक की संख्या हो,
नकारात्मक और शून्य eigenvalues, $ p + n + z = 3$. डेसकार्टेस'
नियम सीमा $ p $ संख्या से $ V $ के संकेत परिवर्तन
$\chi_A $, और $ n $ संख्या से $ V'$ के संकेत परिवर्तन
$\chi_A(-\lambda)$; इसके अलावा क्रमागत अशून्य की प्रत्येक \omterm{def:b2:quadratic:adjoint}{जोड़}ी
गुणांक दोनों में से \omterm{def:b2:series:def}{बिल्कुल} एक में परिवर्तन उत्पन्न करता है
बहुपद, तो $ V + V' \leq 3 - z $ (शून्य जड़ें दिखाई दे रही हैं
लुप्त हो रहे अनुगामी गुणांक के रूप में)। तब $ p + n = 3 - z \geq
V + V' \geq p + n $: समानता, तो $ p = V $ बिलकुल ---
के लक्षण eigenvalues पढ़ा जा सकता है. के लिए $\chi_A(\lambda)
= \lambda^3 - 3\lambda^2 - 9\lambda - 5$: संकेत $+,-,-,-$
देना $ V = 1$, और $\chi_A(-\lambda) = -\lambda^3 - 3\lambda^2
+ 9\lambda - 5$ संकेत हैं $-,-,+,-$: $ V' = 2$. \omterm{thm:b2:structures:signature}{हस्ताक्षर} $(1,
2)$ --- सटीक गुणनखंडन के अनुरूप $\chi_A =
(\lambda - 5)(\lambda + 1)^2$ प्रश्न 6 का.

\textbf{25.} एल्गोरिथम: \omterm{pb:b2:surfaces:1}{द्विघात} भाग को विकर्णित करें
ऑर्थोनॉर्मलली (वर्णक्रमीय प्रमेय: यह एकमात्र है
\omterm{def:b2:powerseries:analytic}{विश्लेषणात्मक} गहरा कदम, और यही बनाता है
वर्गीकरण \emph{इयूक्लिडियन}); हल करना $ A\Omega = -b $ को
केंद्रीय बनाम परवलयिक प्रकार तय करें और दूर अनुवाद करें
जहां संभव हो रैखिक भाग (\omterm{def:b2:affine:subspace}{एफ़िन} ज्यामिति); पढ़ें
रैंक, \omterm{thm:b2:structures:signature}{हस्ताक्षर} और स्थिरांक से टाइप करें (सिल्वेस्टर
गारंटी देता है कि ये अपरिवर्तनीय हैं); जब केवल \omterm{def:b2:affine:subspace}{एफ़िन} प्रकार
मायने रखता है, गॉस की कमी वर्णक्रमीय प्रमेय को प्रतिस्थापित करती है
मीट्रिक जानकारी की लागत. में $\R^2$ वही
मशीन शंकुओं को वर्गीकृत करती है: दीर्घवृत्त, अतिपरवलय, परवलय, प्लस
रेखाओं के \omterm{def:b2:quadratic:adjoint}{जोड़}े, एक रेखा, एक बिंदु और रिक्त समुच्चय। में $\R^n $
बहीखाता के अलावा कुछ भी नहीं बदलता है: प्रकारों को अनुक्रमित किया जाता है
के \omterm{thm:b2:structures:signature}{हस्ताक्षर} $ A $, की स्थिति $ b $ के सापेक्ष
$\operatorname{im}A $, और एक स्थिरांक --- के साथ
$(n{+}2)$-आयामी सीमाबद्ध मैट्रिक्स $(A, b, c)$
एक प्रदान करना \omterm{def:b2:metric:compact}{कॉम्पैक्ट} अपरिवर्तनीय.
