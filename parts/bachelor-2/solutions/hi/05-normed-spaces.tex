\section*{अध्याय \ref{ch:b2:nvs} --- सामान्य वेक्टर रिक्त स्थान}

\begin{solution}{exo:b2:nvs:1}
यूनिट बॉल्स: एक हीरा ($\norm\cdot_1$), एक डिस्क ($\norm\cdot_2$), ए
वर्ग ($\norm\cdot_\infty $), उस क्रम में नेस्टेड। असमानताएँ:
$\norm x_\infty \leq \norm x_2$ (एक वर्ग अधिकतम योग के बराबर होता है);
$\norm x_2 \leq \norm x_1$ (वर्गीकरण: $ x_1^2 + x_2^2 \leq (\abs{x_1}
+ \abs{x_2})^2$); $\norm x_1 \leq 2\norm x_\infty $ (दो पद, प्रत्येक
$\leq \max $). कुशाग्रता: $(1, 0)$ पहले दो समानताएँ बनाता है;
$(1, 1)$ बनाता है $\norm x_1 = 2\norm x_\infty $ और दिखाता भी है
$\norm x_2 = \sqrt2 \norm x_\infty $ और $\norm x_1 = \sqrt2 \norm
x_2$ दूसरी दिशा में चरम अनुपात हैं।
\end{solution}

\begin{solution}{exo:b2:nvs:2}
\omterm{def:b2:nvs:norm}{आदर्श} अभिगृहीत: समरूपता और त्रिभुज असमानता विरासत में मिली है
अवधिवार; पृथक्करण: $ N(f) = 0$ बल $ f' = 0$ (इसलिए $ f $ स्थिर)
और $ f(0) = 0$: $ f = 0$. यह एक है \omterm{def:b2:nvs:norm}{आदर्श}.

तुलना: $\norm f_\infty \leq N(f)$, तब से $\abs{f(x)} \leq
\abs{f(0)} + \abs{\int_0^x f'} \leq \abs{f(0)} +
\norm{f'}_\infty $. बातचीत विफल हो जाती है: ले लो $ f_n(x) = \frac1n
\sin(nx)$: तब $\norm{f_n}_\infty \leq \frac1n \to 0$ जबकि
$ N(f_n) = 0 + \norm{\cos(nx)}_\infty = 1$. कोई स्थिरांक नहीं $ C $ देता है
$ N \leq C\norm\cdot_\infty $.
\end{solution}

\begin{solution}{exo:b2:nvs:3}
$\abs{u(f)} \leq \norm f_\infty \int_0^1 \eu^t\,\dd t = (\eu -
1)\norm f_\infty $, के लिए समानता के साथ $ f \equiv 1$: $\vertiii u =

बदलाव: $\norm{S(x)}_\infty = \max(\abs{x_2}, \dots, \abs{x_n}) \leq

\end{solution}

\begin{solution}{exo:b2:nvs:4}
ऊपरी सीमा: के लिए $\norm x_\infty \leq 1$,
\[
\abs{(Ax)_i} = \Bigl|\sum_j a_{ij}x_j\Bigr| \leq \sum_j
\abs{a_{ij}},
\]
इसलिए $\norm{Ax}_\infty \leq \max_i \sum_j \abs{a_{ij}}$. प्राप्त: चलो
$ i_0$ अधिकतम का एहसास करें और लें $ x_j = \operatorname{sign}(a_{i_0
j})$ (मापांक $1$ प्रविष्टियाँ): फिर $(Ax)_{i_0} = \sum_j \abs{a_{i_0
j}}$. इसलिए सूत्र. दिए गए मैट्रिक्स के लिए: पंक्ति योग $3$ और
$4$: $\vertiii A_\infty = 4$.
\end{solution}

\begin{solution}{exo:b2:nvs:5}
\emph{जियोमीट्रिक श्रंखला:} के लिए $\vertiii B < 1$, श्रृंखला $\sum
B^k $ बनच में बिल्कुल अभिसरण होता है $\mathcal{M}_n(K)$
(\cref{thm:b2:nvs:absoluteconvergence}, $\vertiii{B^k} \leq
\vertiii B^k $), और
\[
(I - B)\sum_{k=0}^{K} B^k = I - B^{K+1} \longrightarrow I :
\]
द्वारा निरंतरता उत्पाद का
(\cref{prop:b2:nvs:bilinear}), $(I - B)\sum_{k\geq0} B^k = I $, इसलिए
$ I - B $ योग के व्युत्क्रम के साथ व्युत्क्रमणीय है (और
$\vertiii{(I-B)^{-1}} \leq \frac{1}{1 - \vertiii B}$).

\emph{खुलापन:} के लिए $ A $ उलटा और $\vertiii H <
\frac{1}{\vertiii{A^{-1}}}$: $ A + H = A(I + A^{-1}H)$ साथ
$\vertiii{A^{-1}H} \leq \vertiii{A^{-1}}\vertiii H < 1$: उलटा.
तो चारों ओर एक गेंद $ A $ में रहता है $ GL_n $.

\emph{निरंतरता उलटा का:} साथ $ B = -A^{-1}H $,
\[
(A + H)^{-1} - A^{-1} = \bigl((I - B)^{-1} - I\bigr)A^{-1}
= \Bigl(\sum_{k \geq 1} B^k\Bigr) A^{-1},
\]
का \omterm{def:b2:nvs:norm}{आदर्श} $\leq \frac{\vertiii B}{1 - \vertiii B}\vertiii{A^{-1}}
\to 0$ जैसा $ H \to 0$.
\end{solution}

\begin{solution}{exo:b2:nvs:6}
\omterm{def:b2:metric:continuity}{निरंतर} $\Rightarrow $ बंद कर्नेल: बंद की पूर्वछवि
$\{0\}$ (\cref{thm:b2:metric:globalcontinuity}).

इसके विपरीत, मान लीजिए $\ker\varphi $ बंद और $\varphi \neq 0$. चुनना
$ a $ साथ $\varphi(a) = 1$; तब से $ a \notin \ker\varphi $ और
कर्नेल बंद है, कुछ गेंद $ B(a, r)$ इसकी याद आती है. अब चलो $ h \in E $
साथ $\varphi(h) \neq 0$: वेक्टर $ a - \frac{h}{\varphi(h)}$
में निहित है $\ker\varphi $, इसलिए बाहर $ B(a, r)$:
\[
\Bigl\Vert \frac{h}{\varphi(h)} \Bigr\Vert \geq r
\quad\Longrightarrow\quad
\abs{\varphi(h)} \leq \frac{\norm h}{r},
\]
एक असमानता जब तुच्छ रूप से सत्य होती है $\varphi(h) = 0$ साथ ही: बाध्य
(4) का \cref{thm:b2:nvs:continuouslinear}: \omterm{def:b2:metric:continuity}{निरंतर}.
\end{solution}

\begin{solution}{exo:b2:nvs:7}
होने देना $ f_n $ होना $0$ पर $\intcc{0}{\frac12 - \frac1n}$, तक संबद्ध करें
मूल्य $1$ पर $\frac12$, तब $1$ पर $\intcc{\frac12}{1}$. के लिए $ m
\geq n $, $ f_m - f_n $ लंबाई के अंतराल पर समर्थित है
$\frac1n $ मूल्यों के साथ $\intcc{-1}{1}$: $\norm{f_m - f_n}_1 \leq
\frac1n $: कॉची.

कल्पना करना $ f_n \to f $ में $\norm\cdot_1$ साथ $ f $ \omterm{def:b2:metric:continuity}{निरंतर}. पर
$\intcc{0}{\frac12 - \delta}$ (तय $\delta $): $\int \abs{f} =
\int\abs{f - f_n} \leq \norm{f - f_n}_1 \to 0$ के लिए $ n >
\frac1\delta $, इसलिए $\int_0^{1/2 - \delta}\abs f = 0$, और सख्ती से
सकारात्मकता $ f = 0$ वहाँ --- प्रत्येक के लिए $\delta $: $ f = 0$ पर
$\intoo{0}{\frac12}$. उसी प्रकार $ f = 1$ पर $\intcc{\frac12}{1}$
(द $ f_n $ सभी समान $1$ वहाँ)। द्वारा निरंतरता पर $\frac12$: $0 =
1$, बेतुका। कोई सीमा मौजूद नहीं है: स्थान नहीं है \omterm{def:b2:metric:complete}{पूरा}.
\end{solution}

\begin{solution}{exo:b2:nvs:8}
अच्छी तरह से परिभाषित और \omterm{def:b2:metric:continuity}{निरंतर}: $\abs{\varphi(f)} \leq \sum 2^{-n}
\norm f_\infty = \norm f_\infty $, इसलिए $\varphi $ एक रैखिक रूप है
के साथ $\vertiii\varphi \leq 1$ (श्रृंखला बिल्कुल एकाग्र होती है
प्रत्येक $ f $).

\omterm{def:b2:nvs:norm}{आदर्श} $1$: हल करना $ K $; बिन्दु $1, \frac12, \dots, \frac1K $ हैं
जोड़ीवार अलग, तो वहाँ एक है \omterm{def:b2:metric:continuity}{निरंतर} $ f_K $, $\norm{f_K}_\infty
\leq 1$, साथ $ f_K(\frac1n) = (-1)^n $ के लिए $ n \leq K $ (टुकड़े-टुकड़े में
एफ़िन इंटरपोलेशन, \omterm{def:b2:metric:continuity}{निरंतर} निकट $0$). तब
\[
\varphi(f_K) \geq \sum_{n=1}^{K} 2^{-n} - \sum_{n > K} 2^{-n}
= 1 - 2^{-K+1} \xrightarrow[K \to \infty]{} 1 .
\]

प्राप्त नहीं: यदि $\norm f_\infty \leq 1$ और $\varphi(f) = 1$, प्रत्येक
अवधि को अपना अधिकतम योगदान देना चाहिए: $(-1)^n f(\frac1n) = 1$ हरएक के लिए
$ n $ (अन्यथा एक कार्यकाल के सख्त घाटे की भरपाई नहीं की जा सकती,
सभी शर्तें हैं $\leq 2^{-n}$). इसलिए $ f(\frac1n) = (-1)^n $; लेकिन
$\frac1n \to 0$ और $ f $ है \omterm{def:b2:metric:continuity}{निरंतर} पर $0$, मजबूर कर रहा है
का विरोधाभासी अभिसरण $(-1)^n $. इसलिए सर्वोच्च एक नहीं है
अधिकतम --- सीमित आयाम में असंभव, जहां बंद इकाई
गेंद है \omterm{def:b2:metric:compact}{कॉम्पैक्ट}.
\end{solution}

\begin{solution}{exo:b2:nvs:9}
\emph{बंधे हुए क्रम:} एक बंधा हुआ क्रम कुछ बंद में निहित है
गेंद $\overline B(0, R) = R\,\overline B(0,1)$, \omterm{def:b2:metric:compact}{कॉम्पैक्ट} (की छवि
द \omterm{def:b2:metric:compact}{कॉम्पैक्ट} होमियोमोर्फिज्म के तहत यूनिट बॉल $ x \mapsto Rx $):
वहां से निकालें.

\emph{रैखिक रूप:} अगर $\dim E < \infty $, प्रत्येक रेखीय मानचित्र से
$ E $ है \omterm{def:b2:metric:continuity}{निरंतर} (\cref{thm:b2:nvs:finitedim}). इसके विपरीत,
मान लीजिये $\dim E = \infty $ (जो, द्वारा \cref{thm:b2:nvs:riesz},
वास्तव में द्वारा बाहर रखा गया है सघनता परिकल्पना --- का बिंदु
यह प्रश्न दो गुणों के बीच निहितार्थ है
सामान्य मानकीकृत स्थान): एक सामान्यीकृत रैखिक रूप से स्वतंत्र चुनें
क्रम $(e_n)$, इसे बीजगणितीय आधार पर \omterm{def:b2:metric:complete}{पूरा} करें (स्वीकृत),
और परिभाषित करें $\varphi(e_n) = n $, $\varphi = 0$ दूसरे आधार पर
वेक्टर, रैखिक रूप से विस्तारित। तब $\abs{\varphi(e_n)} = n $ साथ
$\norm{e_n} = 1$: यूनिट बॉल पर अनबाउंड, असंतत। तो
``सभी रूप \omterm{def:b2:metric:continuity}{निरंतर}'' सीमित आयाम को बल देता है।
\end{solution}

\begin{solution}{exo:b2:nvs:10}
कॉची--श्वार्ज़ \omterm{def:b2:metric:continuity}{निरंतर} कार्य के साथ $1$: $\norm f_1 =
\int_0^1 \abs f\cdot 1 \leq \bigl(\int_0^1
f^2\bigr)^{1/2}\bigl(\int_0^1 1\bigr)^{1/2} = \norm f_2$. और
$\norm f_2^2 = \int f^2 \leq \norm f_\infty^2$. के लिए $ f_n(x) =
x^n $:
\[
\norm{f_n}_1 = \frac1{n+1}, \qquad
\norm{f_n}_2 = \frac1{\sqrt{2n+1}}, \qquad
\norm{f_n}_\infty = 1 .
\]
तब $\norm{f_n}_2/\norm{f_n}_1 = \frac{n+1}{\sqrt{2n+1}} \to
\infty $ और $\norm{f_n}_\infty/\norm{f_n}_2 = \sqrt{2n+1} \to
\infty $: कोई विपरीत असमानताएं नहीं, कोई जोड़ा नहीं है \omterm{def:b2:nvs:equivalent}{समकक्ष}.
\end{solution}

\begin{solution}{exo:b2:nvs:11}
\emph{दूरी के लिए निचली सीमा:} के लिए $ h \in \ker\varphi $,
$\abs{\varphi(x)} = \abs{\varphi(x - h)} \leq
\vertiii\varphi\,\norm{x - h}$; अधिकतम ले लो $ h $:
$ d(x, \ker\varphi) \geq \abs{\varphi(x)}/\vertiii\varphi $.

\emph{ऊपरी सीमा:} हम मान सकते हैं $\varphi(x) \neq 0$. दिया गया
$\varepsilon > 0$, एक इकाई चुनें $ u $ साथ $\abs{\varphi(u)} \geq
\vertiii\varphi - \varepsilon > 0$ और सेट करें $ h = x -
\frac{\varphi(x)}{\varphi(u)}\,u $: तब $\varphi(h) = 0$ और
\[
\norm{x - h} = \frac{\abs{\varphi(x)}}{\abs{\varphi(u)}}
\leq \frac{\abs{\varphi(x)}}{\vertiii\varphi - \varepsilon}.
\]
होने देना $\varepsilon \to 0$: $ d(x, \ker\varphi) \leq
\abs{\varphi(x)}/\vertiii\varphi $; समानता.

\emph{अप्राप्ति:} लेना $\varphi $ से \cref{exo:b2:nvs:8}
($\vertiii\varphi = 1$, प्राप्त नहीं) और कोई भी $ x $ साथ
$\varphi(x) \neq 0$. यदि कुछ $ h \in \ker\varphi $ समझना
$\norm{x - h} = \abs{\varphi(x)}$, यूनिट वेक्टर $ v = (x -
h)/\norm{x - h}$ संतुष्ट करेगा $\abs{\varphi(v)} =
\abs{\varphi(x)}/\norm{x - h} = 1 = \vertiii\varphi $: द
\omterm{thm:b2:nvs:continuouslinear}{ऑपरेटर मानदंड} प्राप्त होगा ---विरोधाभास.
\end{solution}

\begin{solution}{exo:b2:nvs:12}
$ N_1 \leq N_2$ डोमेन में सुपर की एकरसता है, इसलिए
पहचान $(E, N_2) \to (E, N_1)$ है $1$-लिप्सचिट्ज़. के लिए $ P_n(x) =
(x/2)^n $: $ N_2(P_n) = 1$ (पर प्राप्त किया गया $ x = 2$) जबकि $ N_1(P_n) =
2^{-n}$: एक बाध्य $ N_2 \leq CN_1$ देगा $1 \leq C2^{-n}$ सभी के लिए $ n $: असंभव। इतना मानदंड नहीं हैं \omterm{def:b2:nvs:equivalent}{समकक्ष} और
व्युत्क्रम पहचान एक असंतत रैखिक आक्षेप है।

\emph{अपूर्णता:} होने देना $ S_n = \sum_{k=0}^{n}\frac{X^k}{k!}$.
के लिए $ m > n $, $ N_1(S_m - S_n) \leq \sum_{k>n}\frac1{k!} \to 0$:
के लिए कॉची $ N_1$. अगर $ S_n \to P $ में $(E, N_1)$, फिर बिंदुवार
$ P(x) = \lim S_n(x) = \eu^x $ पर $\intcc{0}{1}$; लेकिन एक बहुपद
डिग्री का $ d $ बराबरी नहीं कर सकता $\eu^x $ एक अंतराल पर (अंतर करें)
$ d + 1$ समय: बायां भाग मर जाता है, $\eu^x $ नहीं) में कोई सीमा नहीं
$ E $: नहीं \omterm{def:b2:metric:complete}{पूरा}.

सीमित आयाम में तीनों घटनाएँ असंभव हैं: सभी मानदंड
हैं \omterm{def:b2:nvs:equivalent}{समकक्ष}, हर मानक स्थान है \omterm{def:b2:metric:complete}{पूरा}, और उलटा
एक रेखीय आक्षेप एक परिमित-आयामी स्थान से रैखिक होता है,
इसलिए \omterm{def:b2:metric:continuity}{निरंतर} (\cref{thm:b2:nvs:finitedim}).
\end{solution}

\begin{solution}{pb:b2:nvs:1}
\textbf{1.} उम्मीदवारों के फॉर्म पर विचार करने लायक है $ K = \{f \in F
: \norm{x - f} \leq \norm x\}$: गैररिक्त ($0 \in K $), बंद
(के अंतर्गत एक बंद अंतराल की पूर्वछवि \omterm{def:b2:metric:continuity}{निरंतर} $ f \mapsto
\norm{x - f}$, बंद के साथ प्रतिच्छेदित $ F $,
\cref{cor:b2:nvs:closedsubspace}), घिरा हुआ ($\norm f \leq
\norm{f - x} + \norm x \leq 2\norm x $). में
परिमित-आयामी $ F $, बंद और परिबद्ध साधन \omterm{def:b2:metric:compact}{कॉम्पैक्ट}
(\cref{thm:b2:nvs:finitedim}); \omterm{def:b2:metric:continuity}{निरंतर} समारोह $ f
\mapsto \norm{x - f}$ पर प्राप्त होता है $ K $ यह न्यूनतम है, जो बराबर है
सबके ऊपर असीम $ F $ (कोई $ f \notin K $ देता है $\norm{x -
f} > \norm x \geq \inf $).

\textbf{2.} समांतर चतुर्भुज पहचान में $(\R^n,
\norm\cdot_2)$: $\norm{u + v}^2 + \norm{u - v}^2 = 2\norm u^2 +
2\norm v^2$ (निर्देशांकों के योग के वर्गों का विस्तार करें)। के लिए
इकाई $ u \neq v $:
\[
\Bigl\Vert\frac{u+v}2\Bigr\Vert^2 = 1 - \frac{\norm{u -
v}^2}{4} < 1 .
\]
सख्ती से उत्तल नहीं: के लिए $\norm\cdot_\infty $, लेना $ u = (1, 1, 0,
\dots)$, $ v = (1, -1, 0, \dots)$: मध्यबिंदु के साथ इकाई सदिश
$(1, 0, \dots)$ का \omterm{def:b2:nvs:norm}{आदर्श} $1$; के लिए $\norm\cdot_1$, लेना $ u = (1,
0, \dots)$, $ v = (0, 1, 0, \dots)$: मध्यबिंदु $(\frac12,
\frac12, 0, \dots)$ का \omterm{def:b2:nvs:norm}{आदर्श} $1$.

\textbf{3.} होने देना $ d = d(x, F)$. अगर $ d = 0$: $ x \in \overline F =
F $ और एकमात्र मिनिमाइज़र है $ x $. अगर $ d > 0$ और $ f_1 \neq f_2$
दोनों न्यूनतम करें: $ u = \frac{x - f_1}{d}$ और $ v = \frac{x -
f_2}{d}$ अलग-अलग इकाई सदिश हैं, इसलिए
\[
\Bigl\Vert x - \frac{f_1 + f_2}2\Bigr\Vert
= d\,\Bigl\Vert\frac{u + v}2\Bigr\Vert < d ,
\]
साथ $\frac{f_1 + f_2}2 \in F $: की परिभाषा का खंडन करता है
$ d $. अनोखा मिनिमाइज़र.

\textbf{4.} $\norm{(0,1) - t(1,0)}_\infty = \max(\abs t, 1)
\geq 1$, समानता के साथ यदि $\abs t \leq 1$: मिनिमाइज़र फॉर्म
खंड $\{t(1, 0) : t \in \intcc{-1}{1}\}$, सभी दूरी पर
$1$ --- विशिष्टता बिल्कुल विफल हो जाती है क्योंकि वर्गाकार गेंद होती है
समतल भुजाएँ (प्रश्न 2)।

\textbf{5.} होने देना $ M = \max f $, $ m = \min f $ (प्राप्त:
सघनता). किसी भी स्थिरांक के लिए $ c $: $\sup\abs{f - c} \geq
\max(M - c,\, c - m) \geq \frac{M - m}2$, आखिरी असमानता
क्योंकि दोनों मात्राएँ औसत हैं $\frac{M-m}2$; में समानता
दोनों ताकतें $ M - c = c - m $, यानी\ $ c = c^* = \frac{M + m}2$.
इसके विपरीत $\sup\abs{f - c^*} = \max(M - c^*, c^* - m) =
\frac{M - m}2$. अद्वितीय सर्वोत्तम स्थिरांक. के लिए $ f(x) = x^2$ पर
$\intcc01$: $ c^* = \frac12$, दूरी $\frac12$.

\textbf{6.} से $\cos(n{+}1)\theta + \cos(n{-}1)\theta =
2\cos\theta\cos n\theta $: द्वारा परिभाषित बहुपद $ T_0 = 1$,
$ T_1 = X $, $ T_{n+1} = 2XT_n - T_{n-1}$ संतुष्ट $ T_n(\cos\theta)
= \cos n\theta $ प्रेरण द्वारा. विशिष्टता: दो बहुपद
पर सहमति $\intcc{-1}{1}$ (अनंत अनेक बिंदु) बराबर हैं।
पुनः प्रेरण: $\deg T_n = n $ अग्रणी गुणांक के साथ
$2^{n-1}$ के लिए $ n \geq 1$ ($ T_1$: गुणांक $1 = 2^0$; द
पुनरावृत्ति इसे दोगुना कर देती है)।

\textbf{7.} प्रत्येक $ x \in \intcc{-1}{1}$ है $\cos\theta $, और
$\abs{\cos n\theta} \leq 1$. पर $\eta_k = \cos\frac{k\pi}n $:
$ T_n(\eta_k) = \cos k\pi = (-1)^k $, और $1 = \eta_0 > \eta_1 >
\dots > \eta_n = -1$. जड़ें: $\cos n\theta = 0$ आईएफएफ $\theta =
\frac{(2k-1)\pi}{2n}$: द $ n $ विशिष्ट बिंदु
$\cos\frac{(2k-1)\pi}{2n}$, और तबसे $\frac{(k-1)\pi}n <
\frac{(2k-1)\pi}{2n} < \frac{k\pi}n $, प्रत्येक जड़ सख्ती से निहित है
दो लगातार चरम सीमाओं के बीच.

\textbf{8.} $ T_2 = 2X^2 - 1$, $ T_3 = 4X^3 - 3X $, $ T_4 = 8X^4 -
8X^2 + 1$. के लिए $ T_3$: $ T_3(1) = 1$, $ T_3(\tfrac12) = \tfrac12 -
\tfrac32 = -1$, $ T_3(-\tfrac12) = 1$, $ T_3(-1) = -1$: उत्तम
प्रत्यावर्तन.

\textbf{9.} होने देना $ u_\pm = x \pm \sqrt{x^2 - 1}$ के लिए $ x \geq 1$:
की जड़ें $ z^2 - 2xz + 1$, साथ $ u_+u_- = 1$. क्रम
$ s_n = \frac{u_+^n + u_-^n}2$ संतुष्ट $ s_{n+1} = 2x\,s_n -
s_{n-1}$ (द्विघात से न्यूटन-प्रकार की पुनरावृत्ति), $ s_0 = 1$,
$ s_1 = x $: समान पुनरावृत्ति और प्रारंभिक मान $ n \mapsto
T_n(x)$, इसलिए $ s_n = T_n(x)$ सभी के लिए $ n $. तब से $0 < u_- \leq 1
\leq u_+$ साथ $ u_+ > 1$ के लिए $ x > 1$: $ T_n(x) \geq
\frac{u_+^n}2 \to \infty $ और $ T_n(x) \sim \frac12\bigl(x +
\sqrt{x^2-1}\bigr)^n $. (के लिए $ x \leq -1$ समता का उपयोग करें $ T_n(-x)
= (-1)^nT_n(x)$, पुनरावृत्ति से स्पष्ट।)

\textbf{10.} हरएक के लिए $\theta $: $ T_m\bigl(T_n(\cos\theta)\bigr)
= T_m(\cos n\theta) = \cos mn\theta =
T_{mn}(\cos\theta)$. बहुपद $ T_m \circ T_n $ और
$ T_{mn}$ पर सहमत $\intcc{-1}{1}$, इसलिए बराबर हैं।

\textbf{11.} $ D = Q_n - P $ की डिग्री है $\leq n - 1$ (मोनिक
प्रमुख शर्तें रद्द करें)। चरम सीमा पर: $(-1)^kD(\eta_k) =
2^{1-n} - (-1)^kP(\eta_k) \geq 2^{1-n} - \abs{P(\eta_k)} > 0$ द्वारा
परिकल्पना. इसलिए $ D $ पर प्रत्यावर्ती चिह्नों का शून्येतर मान लेता है
द $ n + 1$ घटते अंक $\eta_0 > \dots > \eta_n $: द्वारा
मध्यवर्ती मूल्य प्रमेय यह कम से कम है $ n $ विशिष्ट जड़ें,
प्रत्येक में एक \omterm{def:b2:metric:topology}{खुला} अंतराल $\intoo{\eta_{k}}{\eta_{k-1}}$. ए
डिग्री का शून्येतर बहुपद $\leq n - 1$ क्या नहीं हो सकता $ n $ जड़ें;
और $ D = 0$ सख्त संकेतों का खंडन करता है। विरोधाभास:
$\sup\abs P \geq 2^{1-n}$ प्रत्येक राक्षसी के लिए $ P $ डिग्री का $ n $.

\textbf{12.} अब $(-1)^kD(\eta_k) = 2^{1-n} - (-1)^kP(\eta_k)
\geq 0$ तब से $\abs{P} \leq 2^{1-n}$. सभी के ऊपर
$\intcc{\eta_k}{\eta_{k-1}}$ ($ k = 1, \dots, n $) समापन बिंदु
के मान $ D $ विपरीत कमजोर संकेत हैं: मध्यवर्ती मूल्य
प्रमेय से शून्य प्राप्त होता है $ z_k $ बंद अंतराल में. यदि $ z_k $
जोड़ीवार अलग चुना जा सकता है, $ D \neq 0$ डिग्री का $\leq n-1$
है $ n $ जड़ें: विरोधाभास. केवल दो लगातार अंतराल ही हो सकते हैं
शून्य साझा करें $ z_k = z_{k+1} = \eta_k $ साथ $0 < k < n $
(आंतरिक भाग)। वहाँ, $ D(\eta_k) = 0$ मतलब $ P(\eta_k) =
(-1)^k2^{1-n}$, का एक चरम मूल्य $ P $ पर $\intcc{-1}{1}$
एक पर प्राप्त किया \emph{आंतरिक भाग} बिंदु: $ P'(\eta_k) = 0$; और
$\eta_k $ का आंतरिक चरम है $ T_n $ बहुत: $ Q_n'(\eta_k) =
0$. इस तरह $ D'(\eta_k) = 0$: $\eta_k $ बहुलता का मूल है
$\geq 2$, साझा अंतराल की भरपाई करना। सभी मामलों में $ D $ है
कम से कम $ n $ जड़ों को बहुलता, डिग्री के साथ गिना जाता है $\leq n -
1$, इसलिए $ D = 0$: $ P = Q_n $. चेबीशेव का चरम प्रमेय है
सिद्ध: अद्वितीय मोनिक मिनिमाइज़र है $2^{1-n}T_n $, सुप का \omterm{def:b2:nvs:norm}{आदर्श}
$2^{1-n}$.

\textbf{13.} डिग्री के मोनिक बहुपद $ n $ बिल्कुल वही हैं
$ X^n - R $ साथ $ R \in \R_{n-1}[X]$, इसलिए
\[
d_\infty\bigl(X^n, \R_{n-1}[X]\bigr) = \min_{P \text{
monic}}\ \sup_{\intcc{-1}{1}}\abs P = 2^{1-n},
\]
विशिष्ट रूप से $ R^* = X^n - Q_n $. प्रतिस्थापन $ x = \frac{1+t}2$:
अगर $ P $ डिग्री का राक्षसी है $ n $ पर $\intcc01$, तब $ t \mapsto
2^nP\bigl(\frac{1+t}2\bigr)$ मोनिक चालू है $\intcc{-1}{1}$ के साथ
सुपर के बराबर $2^n\sup_{\intcc01}\abs P $: इस तरह
$\sup_{\intcc01}\abs P \geq 2^{-n}\cdot2^{1-n} = 2^{1-2n}$, के साथ
समानता बिल्कुल के लिए $ P^*(x) = 2^{-n}Q_n(2x - 1)$: पर
$\intcc{0}{1}$ दूरी है $2^{1-2n}$.

\textbf{14.} $\omega $ डिग्री का राक्षसी है $ n $, इसलिए
$\sup_{\intcc{-1}{1}}\abs\omega \geq 2^{1-n}$ समानता के साथ यदि
$\omega = Q_n = 2^{1-n}T_n $, यानी\ यदि नोड्स हैं $ n $
की जड़ें $ T_n $: $ x_k = \cos\frac{(2k-1)\pi}{2n}$ --- द
\emph{चेबीशेव नोड्स}. न्यूनतम मूल्य: $2^{1-n}$. समदूरस्थ
नोड्स सख्ती से बदतर हैं; प्रक्षेप त्रुटि कारक है
अंतिम बिंदुओं के निकट क्लस्टरिंग नोड्स द्वारा न्यूनतम किया गया।

\textbf{15.} $ n = 2$ हाथ से: $ x^2$ तक फैली हुई है $\intcc01$, इसलिए
$\sup_{\intcc{-1}{1}}\abs{x^2 - c} = \max(\abs c, \abs{1 - c})
\geq \frac12$, पर न्यूनतम किया गया $ c = \frac12$: न्यूनतम मोनिक
द्विघात $ x^2 - \frac12 = \frac12(2x^2 - 1) = Q_2$, कीमत
$\frac12 = 2^{1-2}$. के लिए $ n = 10$ पर $\intcc{0}{1}$: $2^{1-20} =
2^{-19} \approx 1.9\cdot10^{-6}$. डिग्री के कुछ बहुपद $9$
के दो करोड़वें हिस्से के भीतर रहता है $ x^{10}$ कुल मिलाकर
$\intcc01$: इस पैमाने पर दोनों ग्राफ़ अप्रभेद्य हैं
--- की समतलता $ x^{10}$ पास में $0$ निचली डिग्री को सब कुछ करने दें
काम.

\textbf{16.} $\norm{g_k}_\infty = g_k(1) = 1$. के लिए $ j > k $ तय करना
$ a = 2^k $, $ b = 2^j \geq 2a $, और मूल्यांकन करें $ x_0 = 2^{-1/a}$
(इसलिए $ x_0^a = \frac12$):
\[
g_k(x_0) - g_j(x_0) = \frac12 - \Bigl(\frac12\Bigr)^{b/a}
\geq \frac12 - \frac14 = \frac14 .
\]
इसलिए $\norm{g_k - g_j}_\infty \geq \frac14$ सभी के लिए $ j \neq k $: नहीं
अनुवर्ती कॉची है, कोई भी अभिसरण नहीं करता है। की बंद इकाई गेंद
$\bigl(C(\intcc01), \norm\cdot_\infty\bigr)$ क्या नहीं है \omterm{def:b2:metric:compact}{कॉम्पैक्ट}.

\textbf{17.} $ F $ बन्द है (\cref{cor:b2:nvs:closedsubspace})
और उचित: चुनें $ y \notin F $, इसलिए $\delta = d(y, F) > 0$. द्वारा
प्रश्न 1 दूरी कुछ पर प्राप्त होती है $ f^* \in F $. तय करना $ x
= \frac{y - f^*}{\delta}$, एक इकाई वेक्टर ($\norm{y - f^*} =
\delta $). हरएक के लिए $ g \in F $:
\[
\norm{x - g} = \frac{\norm{y - (f^* + \delta g)}}{\delta}
\geq \frac{\delta}{\delta} = 1 ,
\]
तब से $ f^* + \delta g \in F $. इस तरह $ d(x, F) \geq 1$; और $ d(x,
F) \leq \norm{x - 0} = 1$: बिल्कुल $1$.

\textbf{18.} अनंत-आयामी में $ E $, यूनिट वैक्टर बनाएं
आगमनात्मक रूप से: $ x_1$ मनमाना; दिया गया $ x_1, \dots, x_k $, द
उपस्थान $ F_k = \operatorname{Vect}(x_1, \dots, x_k)$ है
परिमित-आयामी, इसलिए उचित, और प्रश्न 17 एक इकाई देता है
$ x_{k+1}$ साथ $ d(x_{k+1}, F_k) = 1$: विशेष रूप से
$\norm{x_{k+1} - x_i} \geq 1$ के लिए $ i \leq k $. अनुक्रम है
जोड़ीवार दूरियाँ $\geq 1$: यूनिट बॉल में एक अनुक्रम होता है
बिना किसी अभिसरण परिणाम के, इसलिए नहीं है \omterm{def:b2:metric:compact}{कॉम्पैक्ट} ---
रिज़्ज़ का प्रमेय, तीव्र स्थिरांक के साथ $1$.

\textbf{19.} होने देना $ h_n $ प्रत्येक आधे भाग पर स्नेह रखें
$\bigl[\frac1{n+1}, \frac1n\bigr]$, से $0$ तक $1$ पर
मध्यबिंदु और वापस $0$, और अन्यत्र शून्य: \omterm{def:b2:metric:continuity}{निरंतर},
$\norm{h_n}_\infty = 1$. के लिए $ n \neq m $ समर्थन मिलते हैं
अधिकांश एक सामान्य समापन बिंदु पर, जहां दोनों गायब हो जाते हैं; के चरम पर
$ h_n $, $ h_m = 0$: $\norm{h_n - h_m}_\infty = 1$ बिल्कुल। के लिए
निश्चित $ x > 0$: $ h_n(x) = 0$ जैसे ही $\frac1n < x $, और
$ h_n(0) = 0$ हमेशा: $ h_n \to 0$ बिंदुवार; लेकिन
$\norm{h_n - 0}_\infty = 1$: समान रूप से नहीं. एक स्पष्ट
नक्षत्र परस्पर दूरी पर $1$ यूनिट बॉल के अंदर.

\textbf{20.} त्रिज्या की एक गेंद $\frac13$ व्यास है $\leq
\frac23 < 1$, इसलिए इसमें अधिक से अधिक एक शामिल है $ h_n $ (दो हैं
दूरी पर $1$). ऐसी गेंदों में परिमित रूप से कई गेंदें होती हैं
अनंत अनेक में से $ h_n $: वे यूनिट बॉल को कवर नहीं कर सकते।
कुल सीमा --- जो \omterm{def:b2:metric:compact}{कॉम्पैक्ट} \omterm{def:b2:metric:def}{मीट्रिक स्थान} आनंद लें, द्वारा
का प्रमाण \cref{thm:b2:metric:borellebesgue} --- उतनी ही बुरी तरह विफल रहता है
यथासंभव.

\textbf{21.} अगर $ Ax = \lambda x $ साथ $ x \neq 0$:
$\abs\lambda\,\norm x = \norm{Ax} \leq \vertiii A\,\norm x $,
इसलिए $\abs\lambda \leq \vertiii A $. के लिए
$\begin{psmallmatrix}1 & -2\\ 3 & 1\end{psmallmatrix}$:
$\vertiii A_\infty = \max(1 + 2,\ 3 + 1) = 4$
(\cref{exo:b2:nvs:4}), तो हर \omterm{def:b2:reduction:eigen}{eigenvalue} मापांक है $\leq
4$; वास्तव में $\chi_A = X^2 - 2X + 7$ देता है $\lambda = 1 \pm
\iu\sqrt6$ मापांक का $\sqrt7 \approx 2.65$: सीमा है
वैध, तीव्र नहीं.

\textbf{22.} $ N_P $ एक है \omterm{def:b2:nvs:norm}{आदर्श}: $ N_P(x) = 0$ बल $ P^{-1}x = 0$,
इसलिए $ x = 0$; समरूपता और त्रिकोण असमानता विरासत में मिली है
से $\norm\cdot_\infty $ रैखिक के माध्यम से $ P^{-1}$. संचालिका
\omterm{def:b2:nvs:norm}{आदर्श}: साथ $ y = P^{-1}x $ और $ D = \mathrm{diag}(\lambda_i)$,
\[
N_P(Ax) = \norm{P^{-1}AP\,y}_\infty = \norm{Dy}_\infty,
\]
इसलिए $\vertiii A_{N_P}$ है $\norm\cdot_\infty $-संचालक \omterm{def:b2:nvs:norm}{आदर्श}
का $ D $, जो इसकी सबसे बड़ी पूर्ण पंक्ति राशि है
(\cref{exo:b2:nvs:4}): $\max_i\abs{\lambda_i}$.

\textbf{23.} यदि सभी $\abs{\lambda_i} < 1$: $ N_P(A^kx) \leq
\rho^kN_P(x)$ साथ $\rho = \max\abs{\lambda_i} < 1$, इसलिए $ A^kx
\to 0$ हरएक के लिए $ x $, और $ A^k \to 0$ मेँ कोई \omterm{def:b2:nvs:norm}{आदर्श} पर
$\mathcal{M}_n $ (सभी \omterm{def:b2:nvs:equivalent}{समकक्ष} सीमित आयाम में,
\cref{thm:b2:nvs:finitedim}; का अभिसरण $ A^ke_j $ प्रत्येक के लिए
$ j $ प्रवेशवार अभिसरण है)। यदि कुछ $\abs{\lambda} \geq 1$
साथ आइजन्वेक्टर $ x $: $\norm{A^kx} = \abs\lambda^k\norm x
\not\to 0$. के लिए $ A = \frac14\begin{psmallmatrix}1 & 2\\ 2 &
1\end{psmallmatrix}$: eigenvalues $\frac14(1 \pm 2) =
\frac34, -\frac14$, दोनों मापांक के $< 1$: $ A^k \to 0$.

\textbf{24.} दोनों हैं मानदंड परिमित-आयामी पर
$\R_n[X]$, इस तरह \omterm{def:b2:nvs:equivalent}{समकक्ष} प्रत्येक के लिए $ n $. न्यूनतम लें
राक्षसी $ P^*_n $ प्रश्न 13 पर $\intcc01$: इसका गुणांक
$ X^n $ है $1$, इसलिए $ N_c(P^*_n) \geq 1$, जबकि
$\norm{P^*_n}_{\intcc01} = 2^{1-2n}$. इस तरह
\[
C_n \geq \frac{N_c(P^*_n)}{\norm{P^*_n}_{\intcc01}}
\geq 2^{2n-1} .
\]
तुल्यता स्थिरांक आयाम के साथ विस्फोटित होते हैं: पर
संघ $\R[X]$ कोई भी स्थिरांक कार्य नहीं करता है, जो वास्तव में है
गैर-समतुल्यता देखी गई \cref{exo:b2:nvs:12} --- ``सभी मानदंड
हैं \omterm{def:b2:nvs:equivalent}{समकक्ष}'' एक समय में एक आयाम के बारे में एक प्रमेय है,
और अनंत आयाम वह है जहां यह मर जाता है।

\textbf{25.} सघनता परिमित में बंद परिबद्ध समुच्चयों का
आयाम ने सर्वोत्तम सन्निकटन का अस्तित्व उत्पन्न किया (प्रश्न)।
1) और शून्य-गिनती को संचालित किया $ n + 1$ चरम बिंदु
(प्रश्न 11-12, प्राप्त समर्थन के माध्यम से)। सख्त उत्तलता नियंत्रित करती है
सर्वोत्तम सन्निकटन की विशिष्टता --- गोल गेंदें एक देती हैं
मिनिमाइजर, फ्लैट-साइड वाली गेंदें उनके खंड देती हैं (प्रश्न
2--4). परस्पर दूरी पर इकाई सदिशों का तारामंडल
$1$ (प्रश्न 17--19) नष्ट कर देता है सघनता यूनिट बॉल का
और इसके साथ पूर्ण सीमाबद्धता (प्रश्न 20)। प्रश्न 24
पतन की मात्रा निर्धारित करता है: दो से संबंधित स्थिरांक मानदंड पर
$\R_n[X]$ जैसे बढ़ो $4^n $, इसलिए कोई समान तुलना नहीं बचती
के लिए मार्ग $\R[X]$. शिखर चेबीशेव का चरम है
प्रमेय (प्रश्न 11-12): अद्वितीय मोनिक मिनिमाइज़र
$2^{1-n}T_n$. सर्वोत्तम सन्निकटन में इसका आधुनिक घर मिलता है
हिल्बर्ट रिक्त स्थान, जहां प्रक्षेपण प्रमेय प्रतिस्थापित होता है
सघनता द्वारा पूर्णता प्लस समांतर चतुर्भुज पहचान ---
वर्ष 3 खंड में ईमानदारी से साबित हुआ।
\end{solution}
