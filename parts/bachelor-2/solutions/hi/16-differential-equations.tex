\section*{अध्याय \ref{ch:b2:diffeq} --- विभेदक समीकरण}

\begin{solution}{exo:b2:diffeq:1}
पहला मैट्रिक्स: eigenvalues $1, 2$, eigenvectors $(1,0)$ और $(1,1)$.
विघटित $ X_0 = (1,0) = 1\cdot(1,0) + 0\cdot(1,1)$: समाधान
है
\[
X(t) = \eu^{t}\begin{pmatrix}1\\ 0\end{pmatrix} .
\]
(प्रारंभिक वेक्टर स्वयं एक है आइजन्वेक्टर.)

दूसरा: $ A = 2I + N $, $ N = E_{12}$, $ N^2 = 0$: $\eu^{tA} =
\eu^{2t}(I + tN)$, इसलिए
\[
X(t) = \eu^{2t}\begin{pmatrix} 1 & t\\ 0 & 1\end{pmatrix}
\begin{pmatrix}1\\ 0\end{pmatrix}
= \eu^{2t}\begin{pmatrix}1\\ 0\end{pmatrix}.
\]
\end{solution}

\begin{solution}{exo:b2:diffeq:2}
$ y' = \sin(ty)$: $\abs{\sin(ty) - \sin(tz)} \leq \abs t\,\abs{y -
z}$ --- \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} में $ y $ \omterm{def:b2:funcseq:def}{समान रूप से} समय के हर खंड पर:
अद्वितीय वैश्विक समाधान $\R $ (प्रत्येक पर प्रमेय लागू करें
खंड)।

$ y' = y^2$: स्थानीय स्तर पर \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} केवल; कोई वैश्विक प्रमेय नहीं, और वास्तव में
$ y(0) = 1$ पर विस्फोट होता है $ t = 1$.

$ y' = \abs y $: $\abs\cdot $ है $1$-लिप्सचिट्ज़: वैश्विक अस्तित्व और
विशिष्टता. साथ $ y(0) = 0$: $ y \equiv 0$ (विशिष्टता!) साथ $ y(0)
= 1$: $ y $ सकारात्मक रहता है (शून्य समाधान को पार नहीं कर सकता), इसलिए $ y' =
y $: $ y = \eu^t $.
\end{solution}

\begin{solution}{exo:b2:diffeq:3}
अगर $ y(t_1) = z(t_1)$ फिर किसी समय $ y $ और $ z $ वही हल करें
कॉची समस्या पर $ t_1$: विशिष्टता बल $ y = z $ उनके आम पर
\omterm{def:b2:diffcalc:differential}{अंतर}ाल --- अलग-अलग समाधान कभी नहीं मिलते।

के लिए $ y' = y(1-y)$: स्थिरांक $0$ और $1$ समाधान हैं. ए
समाधान शुरू हो रहा है $\intoo{0}{1}$ कभी नहीं पहुंच सकते $0$ या $1$
(यह एक \omterm{def:b2:metric:continuity}{निरंतर} समाधान से टकराएगा): यह अंदर रहता है
$\intoo{0}{1}$, इसलिए वैश्विक है (परिबद्ध: कोई विस्फोट नहीं --- जैसे\
द्वारा \cref{exo:b2:diffeq:9}की कसौटी, या क्योंकि वेक्टर क्षेत्र
फंसी हुई पट्टी पर बंधा हुआ है)।
\end{solution}

\begin{solution}{exo:b2:diffeq:4}
$(A - 2I)^2 = \begin{pmatrix}1 & 1\\ -1 & -1\end{pmatrix}^2 = 0$:
डनफोर्ड के साथ $ D = 2I $, $ N = A - 2I $:
\[
\eu^{tA} = \eu^{2t}\,(I + tN)
= \eu^{2t}\begin{pmatrix} 1 + t & t\\ -t & 1 - t\end{pmatrix}.
\]
के साथ स्थिरांक का परिवर्तन $ B(s) = (\eu^{2s}, 0)^{\mathsf T}$:
\[
X(t) = \int_0^t \eu^{(t-s)A}B(s)\,\dd s
= \eu^{2t}\int_0^t \begin{pmatrix} 1 + (t-s)\\ -(t-s)
\end{pmatrix}\dd s
= \eu^{2t}\begin{pmatrix} t + \frac{t^2}{2}\\[2pt] -\frac{t^2}{2}
\end{pmatrix},
\]
का उपयोग करते हुए $\eu^{(t-s)A}B(s) = \eu^{2(t-s)}(I + (t-s)N)\,(\eu^{2s},
0)^{\mathsf T} = \eu^{2t}\bigl(1 + (t-s),\, -(t-s)\bigr)^{\mathsf
T}$. (जाँच करना: $ X(0) = 0$; $ X' - AX = (\eu^{2t}, 0)^{\mathsf T}$
विभेदन द्वारा।)
\end{solution}

\begin{solution}{exo:b2:diffeq:5}
\emph{शृंखला:} $ A = \alpha I + \beta J $ साथ $ J =
\begin{pmatrix}0 & -1\\ 1 & 0\end{pmatrix}$, $ J^2 = -I $; दो
सारांश आवागमन, तो $\eu^{tA} = \eu^{\alpha t}\,\eu^{\beta tJ}$,
और की श्रृंखला $\eu^{\beta t J}$ सम/विषम शक्तियों के साथ विभाजित होता है
में $\cos(\beta t)I + \sin(\beta t)J $: बताई गई रोटेशन-स्केलिंग
मैट्रिक्स.

\emph{जटिल:} पहचान करना $(x, y) \in \R^2$ साथ $ z = x + \iu y $;
प्रणाली $ X' = AX $ पढ़ता $ z' = (\alpha + \iu\beta)z $, किसका
समाधान $ z(t) = \eu^{\alpha t}\eu^{\iu\beta t}z_0$ बिलकुल यही है
सर्पिल: मापांक $\eu^{\alpha t}$, तर्क गति से आगे बढ़ रहा है
$\beta $.
\end{solution}

\begin{solution}{exo:b2:diffeq:6}
होने देना $ v(t) = \eu^{-k(t - t_0)}\int_{t_0}^t u $. तब
\[
v'(t) = \eu^{-k(t-t_0)}\Bigl(u(t) - k\int_{t_0}^t u\Bigr)
\leq C\,\eu^{-k(t-t_0)},
\]
परिकल्पना द्वारा. से एकीकृत किया जा रहा है $ t_0$ को $ t $ ($ v(t_0) = 0$):
$ v(t) \leq \frac{C}{k}\bigl(1 - \eu^{-k(t - t_0)}\bigr)$, अर्थात\
$ k\int_{t_0}^t u \leq C\bigl(\eu^{k(t-t_0)} - 1\bigr)$; खिलाना
यह वापस परिकल्पना में: $ u(t) \leq C\eu^{k(t-t_0)}$.

विशिष्टता/निर्भरता: दो समाधान $ y, z $ अभिन्न का
समीकरण संतुष्ट
\[
\norm{y(t) - z(t)} \leq \norm{y_0 - z_0} +
k\int_{t_0}^{t}\norm{y - z},
\]
और ग्रोनवाल के साथ $ C = \norm{y_0 - z_0}$ घातांक देता है
बाध्य; $ C = 0$ विशिष्टता देता है.
\end{solution}

\begin{solution}{exo:b2:diffeq:7}
विकल्प $ y = y_1 z $ साथ $ y_1 = \frac{\sin t}{t}$: सामान्य
कम करने का फार्मूला (\cref{prop:b2:diffeq:secondorder}) देता है, के लिए $ u
= z'$,
\[
y_1 u' + \Bigl(2y_1' + \frac{2}{t}\,y_1\Bigr)u = 0
\]
(समीकरण को सामान्यीकृत किया गया $ y'' + \frac2t y' + y = 0$). गणना करना
$2y_1' + \frac2t y_1 = 2\,\frac{t\cos t - \sin t}{t^2} +
\frac{2\sin t}{t^2} = \frac{2\cos t}{t}$: इसलिए
\[
\frac{u'}{u} = -\frac{2\cos t}{t}\cdot\frac{t}{\sin t}
= -2\cot t
\quad\Longrightarrow\quad
u = \frac{1}{\sin^2 t}
\quad (\text{up to a constant}),
\]
और $ z = -\cot t $, देना $ y_2 = y_1 z = -\frac{\cos t}{t}$.
सामान्य समाधान चालू $\intoo{0}{\pi}$:
\[
y(t) = \alpha\,\frac{\sin t}{t} + \beta\,\frac{\cos t}{t} .
\]
(ये क्रम शून्य के गोलाकार बेसेल फलन हैं।)
\end{solution}

\begin{solution}{exo:b2:diffeq:8}
डनफोर्ड: $ A = D + N $ आवागमन, $ D $ \omterm{def:b2:reduction:diag}{विकर्णीय} उसी के साथ
eigenvalues, $ N $ निलपोटेंट, इसलिए
\[
\eu^{tA} = \eu^{tD}\,\eu^{tN},
\qquad
\eu^{tN} = \sum_{k < n} \frac{t^kN^k}{k!}
\ \text{(a matrix polynomial in } t).
\]
होने देना $-2\alpha = \max_i \Re\lambda_i < 0$. एक आधार में विकर्णीकरण
$ D $, $\vertiii{\eu^{tD}} \leq \eu^{-2\alpha t}$ (प्रविष्टियां
$\eu^{t\lambda_i}$ मापांक का $\eu^{t\Re\lambda_i}$); मानदंड में
अलग-अलग आधार स्थिरांक के अनुसार भिन्न-भिन्न होते हैं। इस तरह
\[
\norm{X(t)} \leq \vertiii{\eu^{tA}}\,\norm{X_0}
\leq C'\,\eu^{-2\alpha t}\,(1 + t)^{n-1}\,\norm{X_0}
\leq C\,\eu^{-\alpha t}\norm{X_0} ,
\]
बहुपद को एक घातीय गुणनखंड में समाहित करना
($\eu^{-\alpha t}(1+t)^{n-1} \to 0$, इसलिए घिरा हुआ)।
\end{solution}

\begin{solution}{exo:b2:diffeq:9}
होने देना $ y $ पर एक अधिकतम समाधान हो $\intco{0}{T}$, $ T \leq \infty $,
और मान लीजिए $ T < \infty $. अभिन्न रूप देता है, के लिए $ t < T $,
\[
\norm{y(t)} \leq \norm{y_0} + \int_0^t \bigl(a\norm{y(s)} +
b\bigr)\dd s
\leq \bigl(\norm{y_0} + bT\bigr) + a\int_0^t\norm{y(s)}\,\dd s ,
\]
और ग्रोनवाल सीमाएँ $\norm{y(t)} \leq (\norm{y_0} +
bT)\,\eu^{aT} =: M $ पर $\intco{0}{T}$: समाधान a में रहता है
\omterm{def:b2:metric:compact}{कॉम्पैक्ट} गेंद। तब $ y' = f(t, y)$ के पास घिरा हुआ है $ T $, इसलिए $ y $ है
\omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} पास में $ T $ और फैलता है लगातार को $ T $ (कॉची
कसौटी); कॉची समस्या का समाधान $(T, y(T))$ को बढाता है $ y $
आगे $ T $, अधिकतमता का खंडन। इस तरह $ T = \infty $: नहीं
रैखिक विकास के \omterm{def:b2:diffcalc:differential}{अंतर}्गत परिमित समय से बचना।
\end{solution}

\begin{solution}{exo:b2:diffeq:10}
की विशेषता जड़ें $ r^2 - 3r + 2$: $1$ और $2$, तो
सजातीय समाधान हैं $ a\eu^t + b\eu^{2t}$. अनुमान
$\alpha\eu^t $ विफल रहता है क्योंकि $\eu^t $ पहले से ही हल करता है
सजातीय समीकरण (मूल $ r = 1$ के साथ प्रतिध्वनित होता है
दाहिनी ओर)। साथ $ y = \alpha t\,\eu^t $: $ y' = \alpha(1 +
t)\eu^t $, $ y'' = \alpha(2 + t)\eu^t $, और
\[
y'' - 3y' + 2y = \alpha\eu^t\bigl(2 + t - 3 - 3t + 2t\bigr)
= -\alpha\,\eu^t :
\]
$\alpha = -1$, $ y_p = -t\,\eu^t $. सामान्य समाधान: $ y = a\eu^t
+ b\eu^{2t} - t\eu^t $. आरंभिक डेटा $ y(0) = y'(0) = 0$: $ a + b
= 0$ और $ a + 2b - 1 = 0$: $ b = 1$, $ a = -1$:
\[
y(t) = \eu^{2t} - (1 + t)\,\eu^{t} .
\]

\end{solution}

\begin{solution}{exo:b2:diffeq:11}
$ A = \lambda I + N $ साथ $ N = E_{12} + E_{23}$: $ N^2 = E_{13}$,
$ N^3 = 0$, और $\lambda I $ के साथ आवागमन करता है $ N $:
\[
\eu^{tA} = \eu^{\lambda t}\Bigl(I + tN +
\frac{t^2}{2}N^2\Bigr)
= \eu^{\lambda t}\begin{pmatrix}
1 & t & \frac{t^2}{2}\\
0 & 1 & t\\
0 & 0 & 1
\end{pmatrix}.
\]
समाधान: $ X(t) = \eu^{\lambda t}\bigl(X_0 + tNX_0 +
\frac{t^2}2N^2X_0\bigr)$ --- प्रत्येक घटक है $\eu^{\lambda
t}$ डिग्री के बहुपद का गुना $\leq 2$. डिग्री की बाध्यता है
निलपोटेंसी इंडेक्स शून्य से एक: की श्रृंखला $\eu^{tN}$
पर काट-छाँट करता है $ N^2$.
\end{solution}

\begin{solution}{exo:b2:diffeq:12}
\begin{enumerate}
  \item अगर $ X(T) = X(0)$, तब $ Y(t) = X(t + T)$ हल करती है $ Y' =
        AY + B(t + T) = AY + B(t)$ साथ $ Y(0) = X(0)$:
        विशिष्टता (\cref{thm:b2:diffeq:linear}) देता है $ Y = X $,
        यानी\ $ X $ है $ T $-आवधिक. बातचीत तुच्छ है.
  \item ऊपर त्रिकोणीकरण करें $\C $: $ A = PT'P^{-1}$ साथ $ T'$ ऊपरी
        \omterm{def:b2:reduction:diag}{त्रिकोणीय}, विकर्ण $(\lambda_i)$. ए की हर शक्ति
        \omterm{def:b2:reduction:diag}{त्रिकोणीय} मैट्रिक्स विकर्ण के साथ \omterm{def:b2:reduction:diag}{त्रिकोणीय} है
        $(\lambda_i^k)$, इसलिए $\eu^{TA} = P\eu^{TT'}P^{-1}$ है
        विकर्ण के साथ समान आधार पर \omterm{def:b2:reduction:diag}{त्रिकोणीय}
        $(\eu^{T\lambda_i})$: वो हैं eigenvalues. तब
        $ I - \eu^{TA}$ उलटा है iff $\eu^{T\lambda} \neq
        1$ हरएक के लिए \omterm{def:b2:reduction:eigen}{eigenvalue}, यानी\ आईएफएफ $ T\lambda \notin
        2\iu\pi\Z $, जो कि बताई गई परिकल्पना है।
  \item स्थिरांक का परिवर्तन: $ X(T) = \eu^{TA}X(0) +
        \int_0^T\eu^{(T-s)A}B(s)\dd s $, इसलिए $ X(T) = X(0)$ पढ़ता
        \[
        \bigl(I - \eu^{TA}\bigr)X(0) =
        \int_0^{T}\eu^{(T-s)A}B(s)\,\dd s ,
        \]
        जिसका एक अनोखा समाधान है $ X(0)$ के \omterm{def:b2:diffcalc:differential}{अंतर}्गत
        व्युत्क्रमणीयता परिकल्पना: \omterm{def:b2:series:def}{बिल्कुल} एक $ T $-आवधिक
        समाधान. हार्मोनिक ऑसिलेटर के लिए ($\lambda =
        \pm\iu\omega $), बहिष्कृत मामला है $\omega T \in
        2\pi\Z $: बाध्य करना जिसका आवर्त एक गुणक है
        प्राकृतिक काल --- \omterm{pb:b2:diffeq:1}{गूंज}, सप्ताहांत की समस्या के रूप में
        परिमाणित करता है।
\end{enumerate}
\end{solution}

\begin{solution}{pb:b2:diffeq:1}
\textbf{1.}  \omterm{def:b2:reduction:charpoly}{विशेषता बहुपद} है $\lambda^2 -
\tau\lambda + \delta $, जड़ों के साथ
$\frac{\tau\pm\sqrt\Delta}{2}$. अगर $\delta < 0$ तब $\Delta =
\tau^2 - 4\delta > 0$ और दो वास्तविक जड़ों का उत्पाद है
$\delta < 0$:विपरीत चिह्न. अगर $\delta > 0$ और $\Delta \geq
0$: उत्पाद की वास्तविक जड़ें $> 0$ और योग $\tau $: दोनों
का संकेत $\tau $. अगर $\Delta < 0$: संयुग्मी युग्म $\alpha \pm
\iu\beta $ साथ $\alpha = \frac\tau2$, $\beta =
\frac{\sqrt{-\Delta}}2$.

\textbf{2.} $ X(t) = a\,\eu^{\mu t}v_- + b\,\eu^{\lambda t}v_+$.
के साथ परिक्रमा करता है $ a = 0$ (सम्मान.\ $ b = 0$) अस्थिर के साथ दौड़ें
(सम्मान.\ स्थिर) आइजेनलाइन; अन्य सभी के पास है $\norm X \to \infty $
दोनों समय दिशाओं में, स्पर्शोन्मुख $\R v_+$ जैसा $ t \to
+\infty $ और को $\R v_-$ जैसा $ t \to -\infty $: काठी
चित्र. सब पर सीमाबद्धता $\R $ बल $ b = 0$ (अन्यथा
पर विस्फोट $+\infty $) और $ a = 0$ (पर $-\infty $): केवल
उत्पत्ति.

\textbf{3.} साथ $\mu < \lambda < 0$, दोनों घातीय क्षय:
$ X(t) \to 0$. अगर $ b \neq 0$, कारक $\eu^{\lambda t}$:
\[
X(t) = \eu^{\lambda t}\bigl(b\,v_\lambda + a\,\eu^{(\mu -
\lambda)t}v_\mu\bigr),
\qquad \eu^{(\mu-\lambda)t} \to 0 :
\]
इसकी दिशा $ X(t)$ आदत है $\R v_\lambda $, द
\emph{धीमा} eigendirection --- तेज़ अक्ष को छोड़कर सभी कक्षाएँ
इसके स्पर्शरेखा पर पहुंचें (अध्याय के चरण का दायां पैनल)।
पोर्ट्रेट्स)।

\textbf{4.} आधार में जहां $ A = \begin{pmatrix} \alpha &
-\beta\\ \beta & \alpha\end{pmatrix}$
(\cref{exo:b2:diffeq:5}; भाग II के ऑसिलेटर सिस्टम के लिए
यह रूप आधार के स्पष्ट वास्तविक परिवर्तन द्वारा प्राप्त किया जाता है), द
समाधान है $\eu^{\alpha t}$ कोण के एक घूर्णन का समय $\beta
t $: लघुगणक सर्पिल, संकुचन कब $\alpha = \frac\tau2
< 0$, जब विस्तार हो रहा है $\tau > 0$, और बंद वक्र (दीर्घवृत्त)।
मूल निर्देशांक में) कब $\tau = 0$: केंद्र.

\textbf{5.} $\Delta = 0$ दोगुना देता है \omterm{def:b2:reduction:eigen}{eigenvalue} $\lambda =
\frac\tau2$; केली-हैमिल्टन द्वारा
(\cref{thm:b2:reduction:cayleyhamilton}), $(A - \lambda I)^2 =
0$, इसलिए $ N = A - \lambda I $ शून्यशक्तिमान है, साथ आवागमन करता है
$\lambda I $, और $\eu^{tA} = \eu^{\lambda t}(I + tN)$. अगर $ N =
0$: $ A = \lambda I $, सभी किरणें कक्षाएँ (स्टार नोड) हैं। अगर $ N
\neq 0$: $ X(t) = \eu^{\lambda t}(X_0 + tNX_0)$, और के लिए $ NX_0
\neq 0$ दिशा एकल eigendirection में परिवर्तित हो जाती है
$\operatorname{im}N $: अनुचित नोड. यह \omterm{def:b2:metric:complete}{पूरा} होता है
ट्रेस--निर्धारक चित्र।

\textbf{6.} $\tau = -2\zeta\omega $, $\delta = \omega^2 > 0$,
$\Delta = 4\omega^2(\zeta^2 - 1)$. इसलिए: $0 < \zeta < 1$ देता है
$\Delta < 0$, $\tau < 0$: स्थिर सर्पिल; $\zeta = 1$: $\Delta =
0$: पतित स्थिर नोड; $\zeta > 1$: $\Delta > 0$, $\tau <
0$, $\delta > 0$: स्थिर नोड; $\zeta = 0$: $\tau = 0$,
$\delta > 0$: केंद्र। विमान में एक ऊर्ध्वाधर यात्रा
$\delta = \omega^2$.

\textbf{7.} जड़ों $ r = -\zeta\omega \pm
\omega\sqrt{\zeta^2-1}$. के लिए $\zeta < 1$: $ r = -\zeta\omega
\pm \iu\omega_d $, $\omega_d = \omega\sqrt{1-\zeta^2}$:
\[
x(t) = \eu^{-\zeta\omega t}\bigl(a\cos\omega_dt +
b\sin\omega_dt\bigr)
= R\,\eu^{-\zeta\omega t}\cos(\omega_dt - \varphi) .
\]
के लिए $\zeta = 1$: $ x = (a + bt)\eu^{-\omega t}$. के लिए $\zeta >
1$: $ x = a\eu^{r_-t} + b\eu^{r_+t}$, दोनों दरें नकारात्मक।
की क्रमिक अधिकतमा $\abs x $ अंडरडैम्प्ड मामले में घटित होता है
समय को छद्म काल द्वारा अलग किया गया $\frac{2\pi}{\omega_d}$
(कोसाइन का समान चरण), और उनका अनुपात है
$\eu^{-\zeta\omega\cdot2\pi/\omega_d} =
\eu^{-2\pi\zeta/\sqrt{1-\zeta^2}}$: लघुगणकीय कमी,
आस्टसीलस्कप पर पढ़ने योग्य अवमंदन मीटर।

\textbf{8.} युक्तिसंगत बनाना,
\[
\abs{\lambda_{\mathrm{slow}}} = \omega\bigl(\zeta -
\sqrt{\zeta^2-1}\bigr)
= \frac{\omega}{\zeta + \sqrt{\zeta^2 - 1}} ,
\]
जिसका हर बढ़ता है $\zeta \geq 1$: क्षय
दर सबसे अधिक है $\zeta = 1$, जहां यह बराबर है $\omega $. एक
अत्यधिक नमी वाला दरवाज़ा बिना पटके बंद हो जाता है लेकिन \emph{धीरे से};
क्रिटिकल डैम्पिंग इंजीनियर के लिए इष्टतम है।

\textbf{9.} $ E' = x'x'' + \omega^2xx' =
x'\bigl(-2\zeta\omega x' - \omega^2x\bigr) + \omega^2xx' =
-2\zeta\omega\,x'^2 \leq 0$. अगर $ x $ आवधिक थे और
अचर, $ E $ इसलिए, आवधिक और गैर-वृद्धिशील होगा
स्थिर, जबरदस्ती $ x' \equiv 0$: $ x $ स्थिर, और फिर
$\omega^2x = 0$: $ x \equiv 0$. अभीतक के लिए तो $\zeta > 0$ एकमात्र
आवधिक समाधान आराम है: भीगना हर किसी को मारता है \omterm{def:b2:structures:sn}{चक्र}.

\textbf{10.} केंद्र $\zeta = 0$ लाइन पर रहता है $\tau =
0$ ट्रेस का - निर्धारक विमान - खाली के साथ एक सेट
आंतरिक: मैट्रिक्स का एक मनमाना छोटा गड़बड़ी
(कोई भी भौतिक अवमंदन) चलता है $\tau $ शून्य से बाहर और बदल जाता है
सर्पिलों में बंद कक्षाएँ। अवमंदित की आवधिकता
इस प्रकार थरथरानवाला एक रेजर-किनारे वाली घटना है, मजबूत नहीं
एक.

\textbf{11.} स्थानापन्न $ x_p = \Re(z\eu^{\iu\gamma t})$ में
समीकरण:
\[
\bigl(-\gamma^2 + 2\iu\zeta\omega\gamma +
\omega^2\bigr)z = F
\quad\Longrightarrow\quad
z = \frac{F}{\omega^2 - \gamma^2 + 2\iu\zeta\omega\gamma},
\]
इसलिए $ x_p = \abs z\cos(\gamma t - \varphi)$ साथ $\varphi =
\arg(\omega^2 - \gamma^2 + 2\iu\zeta\omega\gamma)$, अर्थात\
$\tan\varphi = \frac{2\zeta\omega\gamma}{\omega^2 -
\gamma^2}$, और कहा गया $ A(\gamma) = \abs z $.

\textbf{12.} दो समाधानों का \omterm{def:b2:diffcalc:differential}{अंतर} हल करता है
सजातीय समीकरण, जिसके लिए $\zeta > 0$ का क्षय होता है $0$
(प्रश्न 7): प्रत्येक समाधान बराबर होता है $ x_p $ प्लस एक क्षणिक
अनंत पर लुप्त हो जाना. स्थिर अवस्था एक वैश्विक आकर्षण है:
केवल प्रारंभिक स्थितियाँ ही भुला दी जाती हैं $ A(\gamma)$ और
चरण \omterm{def:b2:diffcalc:differential}{अंतर}ाल $\varphi $ अवशेष।

\textbf{13.} छोटा करना $ g(u) = (\omega^2 - u)^2 +
4\zeta^2\omega^2u $ ऊपर $ u = \gamma^2 \geq 0$: $ g'(u) =
-2(\omega^2 - u) + 4\zeta^2\omega^2 = 0$ पर $ u = \omega^2(1 -
2\zeta^2)$, आंतरिक आईएफएफ $\zeta < \frac{1}{\sqrt2}$. वहाँ
\[
g(u_*) = 4\zeta^4\omega^4 + 4\zeta^2\omega^4(1 - 2\zeta^2)
= 4\zeta^2\omega^4(1 - \zeta^2),
\qquad
A(\gamma_*) = \frac{F}{2\zeta\omega^2\sqrt{1 - \zeta^2}} .
\]
स्थैतिक प्रतिक्रिया के विरुद्ध $ A(0) = \frac{F}{\omega^2}$:
प्रवर्धन $\frac{1}{2\zeta\sqrt{1-\zeta^2}} \approx
\frac{1}{2\zeta}$ छोटे के लिए $\zeta $ --- हल्का गीला
सिस्टम के पास $\gamma_* \approx \omega $ इनपुट को गुणा करता है a
सौ गुना जब $\zeta = 0.005$.

\textbf{14.} कहा गया $ x $ संतुष्ट $ x(0) = x'(0) = 0$ और
\[
x'' + \omega^2 x = \frac{F(\omega^2 -
\gamma^2)\cos\gamma t}{\omega^2 - \gamma^2} = F\cos\gamma t
\]
(द $\cos\omega t $ भाग रद्द)। उत्पाद प्रपत्र इस प्रकार है
से $\cos p - \cos q = 2\sin\frac{q+p}{2}\sin\frac{q-p}2$
साथ $ p = \gamma t $, $ q = \omega t $. के लिए $\gamma $ के करीब
$\omega $, कारक $\sin\frac{(\omega-\gamma)t}2$ एक धीमा है
तेज दोलन को नियंत्रित करने वाला लिफाफा
$\sin\frac{(\omega+\gamma)t}2$: धड़कता है, आयाम के साथ
$\frac{2F}{\abs{\omega^2-\gamma^2}}$ --- बड़ा, लेकिन घिरा हुआ।

\textbf{15.} के लिए $ x_p = \frac{F}{2\omega}t\sin\omega t $:
\[
x_p'' = \frac{F}{2\omega}\bigl(2\omega\cos\omega t -
\omega^2t\sin\omega t\bigr)
= F\cos\omega t - \omega^2x_p :
\]
एक समाधान। और निश्चित पर $ t $, देना $\gamma \to \omega $ में
प्रश्न 14:
\[
\frac{2F\sin\frac{(\omega-\gamma)t}2
\sin\frac{(\omega+\gamma)t}{2}}
{(\omega-\gamma)(\omega+\gamma)}
\longrightarrow
\frac{2F\cdot\frac{(\omega-\gamma)t}2\big/(\omega-\gamma)
\cdot\sin\omega t}{2\omega}
= \frac{F\,t\sin\omega t}{2\omega} .
\]
आयाम बिना किसी सीमा के रैखिक रूप से बढ़ता है: \omterm{pb:b2:diffeq:1}{गूंज}
आपदा---सैनिकों के पुलों पर कदम तोड़ने का कारण।

\textbf{16.} वर्गाकार तरंग हर विषम परिस्थिति में हार्मोनिक्स वहन करती है
आवृत्ति $ n = 1, 3, 5, \dots $; रैखिकता द्वारा, प्रत्येक हार्मोनिक
$ n $ पर थरथरानवाला की प्रतिक्रिया द्वारा बढ़ाया जाता है $\gamma = n $.
के लिए $\omega = 3$ तीसरा हार्मोनिक हिट \omterm{pb:b2:diffeq:1}{गूंज} \omterm{def:b2:series:def}{बिल्कुल}।
इंजीनियर वर्गाकार (और सॉटूथ) इनपुट से डरते हैं क्योंकि वे
उत्तेजित करना \emph{सभी} एक ही बार में अजीब हार्मोनिक्स: जो भी प्राकृतिक हो
संरचना की आवृत्ति, कुछ हार्मोनिक इसकी प्रतीक्षा कर रहे हैं।

\textbf{17.} $ W' = y_1y_2'' - y_1''y_2 = -qy_1y_2 + qy_1y_2 =
0$: $ W $ स्थिर है (लिउविले एक ट्रेस-शून्य साथी के साथ
मैट्रिक्स)। $ W = 0$ एक बिंदु पर प्रारंभिक डेटा बनाता है $ y_2$
उन लोगों के लिए आनुपातिक $ y_1$, इस तरह $ y_2$ के लिए आनुपातिक
$ y_1$ (विशिष्टता); $ W \neq 0$ यदि स्वतंत्र. अगर $ y(t_0) =
y'(t_0) = 0$ तब $ y \equiv 0$ (विशिष्टता): एक शून्येतर
समाधान में सरल शून्य होते हैं, और एक साधारण शून्य पृथक होता है
($ y'$ पास में निश्चित चिन्ह का)।

\textbf{18.} लगातार शून्य के बीच $ a < b $, $ y_1$ रखता है
एक संकेत, कहो $ y_1 > 0$ पर $\intoo ab $: तब $ y_1'(a) > 0$ और
$ y_1'(b) < 0$ (सरल शून्य). स्थिरांक का मूल्यांकन $ W =
y_1y_2' - y_1'y_2$ पर $ a $ और $ b $:
\[
W = -y_1'(a)\,y_2(a) = -y_1'(b)\,y_2(b) ,
\]
इसलिए $ y_2(a)$ और $ y_2(b)$ विपरीत संकेत हैं ($ W \neq 0$
या तो गायब होने से मना करता है): $ y_2$ में गायब हो जाता है $\intoo ab $
(मध्यवर्ती मूल्य)। यह वहां दो बार गायब नहीं हो सकता: दो शून्य
का $ y_2$ का एक शून्य फ़्लैंक होगा $ y_1$ उसी तर्क से
भूमिकाओं का आदान-प्रदान हुआ, निरंतरता का खंडन: \omterm{def:b2:series:def}{बिल्कुल}
एक शून्य---इंटरलेसिंग।

\textbf{19.} कल्पना करना $ z $ इसमें कोई शून्य नहीं है $\intoo ab $; की जगह
$ y, z $ उनकी नकारात्मकताओं से, मान लीजिए $ y > 0$ और $ z > 0$ पर
$\intoo ab $. तय करना $\varphi = yz' - y'z $: $\varphi' = yz'' -
y''z = (q_1 - q_2)\,yz \leq 0$ पर $\intoo ab $: $\varphi $ है
न बढ़ने वाला. लेकिन $\varphi(a) = -y'(a)z(a) \leq 0$ (जैसा
$ y'(a) > 0$, $ z(a) \geq 0$) और $\varphi(b) = -y'(b)z(b) \geq
0$ (जैसा $ y'(b) < 0$, $ z(b) \geq 0$): एक गैर-बढ़ने वाला कार्य
से चल रहा है $\leq 0$ को $\geq 0$ \omterm{def:b2:funcseq:def}{समान रूप से} गायब हो जाता है, इसलिए
$(q_1 - q_2)yz \equiv 0$ पर $\intoo ab $. अगर $ q_1 < q_2$
कहीं $\intoo ab $, यह बेतुका है ($ y, z > 0$ वहाँ):
$ z $ भीतर सख्ती से गायब हो जाना चाहिए। सामान्य तौर पर ($ q_1 \leq q_2$),
दोनों में से एक $ z $ में गायब हो जाता है $\intoo ab $, या $\varphi \equiv 0$
बल $ z $ के लिए आनुपातिक $ y $, जो गायब हो जाता है $ a $ और $ b $:
सभी मामलों में $ z $ में शून्य है $\intcc ab $.

\textbf{20.} ऊपरी सीमा: तुलना करें $ y $ (गुणांक $ q \geq
m^2$) साथ $ u(t) = \sin(m(t - a))$ (गुणांक $ m^2 \leq q $,
इसलिए $ y $ की भूमिका निभाता है $ z $ प्रश्न 19 में): यदि $ y $ नहीं था
शून्य में $\intoc{a}{a + \pi/m}$, शून्य $ a $ और $ a +
\frac\pi m $ का $ u $ के साथ लगातार होगा $ y \neq 0$
उनके बीच, प्रश्न 19 का खंडन: लगातार शून्य
$ y $ दूरी पर हैं $\leq \frac\pi m $. निचली सीमा: यदि दो
लगातार शून्य $ a < b $ का $ y $ था $ b - a < \frac\pi M $,
तब $ z(t) = \sin(M(t-a))$ (गुणांक $ M^2 \geq q $) होगा
में लुप्त हो जाना है $\intcc ab \subset \intoo{a}{a + \pi/M}
\cup\{a\}$, जहां इसका केवल शून्य है $ a $ स्वयं --- लेकिन
प्रश्न 19 पर लागू $\intoo{a}{b}$ पर सख्ती के साथ
समापनबिंदु शून्य देता है $\intcc ab $, और $ z > 0$ पर
$\intoc ab $: विरोधाभास। इस तरह $\frac\pi M \leq b - a \leq
\frac\pi m $; के लिए $ q \equiv \omega^2$ दोनों सीमाएँ ढह जाती हैं
सटीक \omterm{def:b2:diffcalc:differential}{अंतर} $\frac\pi\omega $ हार्मोनिक का
थरथरानवाला.

\textbf{21.} साथ $ u = ty $: $ u'' = ty'' + 2y'$, इसलिए $ ty'' + 2y'
+ ty = u'' + u = 0$: $ u = A\sin t + B\cos t = R\sin(t +
\varphi)$, और $ y = \frac{u}{t}$ ठीक $\frac{\sin t}t $ और
$\frac{\cos t}t $ (\cref{exo:b2:diffeq:7}) बिना किसी कमी के

क्रिया में दर्शन: गुणांक $ q \equiv 1$ निर्देश देता है

\textbf{22.} तय करना $ x(t) =
\frac1\omega\int_0^t\sin(\omega(t-s))F(s)\dd s $. तब $ x(0) =
0$;
\[
x'(t) = \frac1\omega\sin(0)F(t) +
\int_0^t\cos(\omega(t-s))F(s)\dd s
= \int_0^t\cos(\omega(t-s))F(s)\dd s ,
\]
इसलिए $ x'(0) = 0$; और $ x''(t) = F(t) -
\omega\int_0^t\sin(\omega(t-s))F(s)\dd s = F(t) - \omega^2x(t)$
(ए का विभेदन) \omterm{thm:b2:integration:continuity}{पैरामीटर अभिन्न} परिवर्तनीय सीमा के साथ,
जैसा कि एकीकरण अध्याय में है)। साथ $ F(s) = F\cos\omega s $,
उत्पाद-से-योग देता है
\[
\int_0^t\sin(\omega(t-s))\cos(\omega s)\dd s
= \frac12\int_0^t\bigl(\sin\omega t + \sin(\omega t -
2\omega s)\bigr)\dd s
= \frac{t}{2}\sin\omega t ,
\]
(दूसरा टुकड़ा शून्य पर एकीकृत होता है), इसलिए $ x =
\frac{F}{2\omega}t\sin\omega t $: प्रश्न 15 पुनः, से
डुहामेल.

\textbf{23.} सिस्टम रूप में $ X' = AX + (0, F(t))^{\mathsf T}$

\textbf{25.} (i) ट्रेस-निर्धारक तल सभी को वर्गीकृत करता है
