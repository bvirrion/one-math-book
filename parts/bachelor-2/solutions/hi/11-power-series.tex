\section*{अध्याय \ref{ch:b2:powerseries} --- बिजली की श्रृंखला}

\begin{solution}{exo:b2:powerseries:1}
$\sum \frac{n^2}{2^n}z^n $: अनुपात $\frac{(n+1)^2}{2^{n+1}}\cdot
\frac{2^n}{n^2} \to \frac12$: $ R = 2$.

$\sum \frac{z^n}{\binom{2n}{n}}$: $\binom{2n}{n} \sim
\frac{4^n}{\sqrt{\pi n}}$ (\cref{ex:b2:comparison:centralbinomial}),
इसलिए $\abs{a_n}^{-1} \approx 4^n $ बहुपद गुणनखंड तक: $ R = 4$
(अनुपात परीक्षण: $\frac{\binom{2n}{n}}{\binom{2n+2}{n+1}} =
\frac{(n+1)^2}{(2n+1)(2n+2)} \to \frac14$).

$\sum z^{n!}$: गुणांक $ a_k = 1$ अगर $ k = n!$, अन्य $0$. के लिए
$\abs z < 1$, $\sum \abs z^{n!}$ \omterm{def:b2:integration:improper}{अभिसरण} (ज्यामितीय द्वारा प्रभुत्व);
के लिए $\abs z \geq 1$ शर्तें अनुकूल नहीं हैं $0$: $ R = 1$.

$\sum (2 + (-1)^n)^n z^n $: गुणांक $3^n $ (यहां तक की $ n $) और $1$
(विषम $ n $). की सीमा $ a_nr^n $ आवश्यक है $3r \leq 1$; और $ r <
\frac13$ काम करता है: $ R = \frac13$.
\end{solution}

\begin{solution}{exo:b2:powerseries:2}
तीनों की त्रिज्या है $1$ (अनुपात परीक्षण). पर $ z = 1$: $\sum 1$
विचलन; $\sum\frac1n $ विचलन; $\sum\frac{1}{n^2}$ जुटता है. पर
$ z = -1$: $\sum(-1)^n $ विचलन; $\sum\frac{(-1)^n}{n}$ जुटता है
(वैकल्पिक); $\sum\frac{(-1)^n}{n^2}$ \omterm{def:b2:integration:improper}{अभिसरण} (\omterm{def:b2:series:def}{बिल्कुल}).
सीमा व्यवहार त्रिज्या के लिए अदृश्य है।
\end{solution}

\begin{solution}{exo:b2:powerseries:3}
से $\frac{1}{1-x} = \sum x^n $, अंतर करें और गुणा करें $ x $
(\cref{thm:b2:powerseries:calculus}):
\[
\sum n x^n = \frac{x}{(1-x)^2} .
\]
एक बार फिर अंतर करें, गुणा करें $ x $ दोबारा:
\[
\sum n^2 x^n = x\,\frac{\dd}{\dd x}\Bigl(\frac{x}{(1-x)^2}\Bigr)
= \frac{x(1 + x)}{(1-x)^3} .
\]
तीसरा योग: यह का विषम भाग है $-\ln(1 - x)$:
\[
\sum_{n\geq0} \frac{x^{2n+1}}{2n+1}
= \frac{-\ln(1-x) + \ln(1+x)}{2}
= \frac12 \ln\frac{1+x}{1-x}
= \operatorname{artanh} x .
\]
\end{solution}

\begin{solution}{exo:b2:powerseries:4}
आंशिक अंश: $\frac{1}{(1-x)(2-x)} = \frac{1}{1-x} -
\frac{1}{2 - x} = \sum x^n - \frac12\sum \bigl(\frac x2\bigr)^n $:
\[
\frac{1}{(1-x)(2-x)} = \sum_{n\geq0} \Bigl(1 -
\frac{1}{2^{n+1}}\Bigr)x^n,
\qquad R = 1 .
\]

$\ln(1 + x + x^2) = \ln\frac{1 - x^3}{1 - x} = \ln(1 - x^3) -
\ln(1 - x) = \sum_{n\geq1}\frac{x^n}{n} -
\sum_{m\geq1}\frac{x^{3m}}{m}$: का गुणांक $ x^n $ है
$\frac1n $ अगर $3 \nmid n $, और $\frac1n - \frac{3}{n} = -\frac2n $ अगर
$3 \mid n $. RADIUS $1$ (निकटतम बाधा: के लिए श्रृंखला
$\ln(1-x^3)$).
\end{solution}

\begin{solution}{exo:b2:powerseries:5}
\omterm{thm:b2:series:fubini}{कॉची उत्पाद} का $\sum_{m \geq 0} x^m $ (गुणांक $1$) और
$\sum_{k\geq1} \frac{x^k}{k}$ (गुणांक $\frac1k $, $ k \geq
1$), दोनों \omterm{def:b2:series:def}{बिल्कुल} के लिए \omterm{def:b2:integration:improper}{अभिसरण} $\abs x < 1$: गुणांक
का $ x^n $ उत्पाद में है $\sum_{k=1}^{n} \frac1k \cdot 1 = H_n $.
इस तरह
\[
\Bigl(\sum x^m\Bigr)\Bigl(\sum \frac{x^k}{k}\Bigr)
= \frac{1}{1-x}\cdot\bigl(-\ln(1-x)\bigr)
= \sum_{n\geq1} H_n x^n .
\]
\end{solution}

\begin{solution}{exo:b2:powerseries:6}
पुनरावृत्ति को इससे गुणा करें $ x^{n+1}$ और योग ($\abs x <
\frac12$):
\[
U(x) - 1 = 2x\,U(x) + \sum_{n\geq0} n\,x^{n+1}
= 2x\,U(x) + \frac{x^2}{(1-x)^2} ,
\]
का उपयोग करते हुए \cref{exo:b2:powerseries:3}. इस तरह
\[
U(x) = \frac{1}{1 - 2x}\Bigl(1 + \frac{x^2}{(1-x)^2}\Bigr)
= \frac{1 - 2x + 2x^2}{(1-2x)(1-x)^2} .
\]
आंशिक भिन्न (कवर-अप)। $ x = \frac12$ गुणांक देता है $2$;
दोहरे ध्रुव पर $ x = 1$, गुणांक $-1$; मध्य
पर मूल्यांकन करने पर गुणांक लुप्त हो जाता है $ x = 0$):
\[
U(x) = \frac{2}{1-2x} - \frac{1}{(1 - x)^2} .
\]
दोनों का विस्तार:
\[
u_n = 2\cdot 2^n - (n + 1) = 2^{n+1} - n - 1 .
\]
(जाँच करना: $ u_0 = 1$, $ u_1 = 2u_0 + 0 = 2 = 4 - 2$.)
\end{solution}

\begin{solution}{exo:b2:powerseries:7}
प्रेरण: $ f'(x) = \frac{2}{x^3}\eu^{-1/x^2}$, और यदि $ f^{(n)}(x)
= P_n(\tfrac1x)\eu^{-1/x^2}$ तब
\[
f^{(n+1)}(x) =
\Bigl(-\frac{1}{x^2}\,P_n'\Bigl(\frac1x\Bigr) +
\frac{2}{x^3}\,P_n\Bigl(\frac1x\Bigr)\Bigr)\eu^{-1/x^2} :
\]
फिर से बताए गए फॉर्म का. पर $0$: अंतर भागफल $\frac{f^{(n)}(h)}{h} =
\frac1h P_n(\frac1h)\eu^{-1/h^2} \to 0$ जैसा $ h \to 0$, तब से
$ Q(u)\,\eu^{-u^2} \to 0$ जैसा $ u \to \pm\infty $ किसी भी बहुपद के लिए
$ Q $ (एक्सपोनेंशियल बीट्स पॉवर्स): प्रेरण द्वारा सभी $ f^{(n)}(0)$ मौजूद
और गायब हो जाओ, और प्रत्येक $ f^{(n)}$ है \omterm{def:b2:metric:continuity}{निरंतर} पर $0$ उसी के द्वारा
सीमा. इसलिए $ f \in C^\infty $ शून्य टेलर श्रृंखला के साथ $0$; द
टेलर श्रृंखला का सार है $0 \neq f $: नहीं \omterm{def:b2:powerseries:analytic}{विश्लेषणात्मक} पर $0$.
\end{solution}

\begin{solution}{exo:b2:powerseries:8}
द्विपद श्रृंखला: $\sqrt{1-4x} = \sum_{k\geq0}
\binom{1/2}{k}(-4x)^k $. के लिए $ k = n + 1 \geq 1$:
\begin{align*}
\binom{1/2}{n+1}(-4)^{n+1}
&= \frac{\frac12\bigl(\frac12 - 1\bigr)\cdots\bigl(\frac12 -
n\bigr)}{(n+1)!}\,(-4)^{n+1}\\
&= \frac{(-1)^n\,1\cdot3\cdots(2n-1)}{2^{n+1}(n+1)!}\,(-4)^{n+1}
= -\frac{2}{n+1}\cdot\frac{(2n)!}{n!\,n!} ,
\end{align*}
का उपयोग करते हुए $1\cdot3\cdots(2n-1) = \frac{(2n)!}{2^n n!}$. इस तरह
\[
C(x) = \frac{1 - \sqrt{1-4x}}{2x}
= \frac{1}{2x}\sum_{n\geq0}\frac{2}{n+1}\binom{2n}{n}x^{n+1}
= \sum_{n\geq0} \frac{1}{n+1}\binom{2n}{n}\,x^n :
\]
$ C_n = \frac{1}{n+1}\binom{2n}{n}$. त्रिज्या: $\frac14$ (द्विपद
श्रृंखला में $4x $). स्पर्शोन्मुखता के माध्यम से
\cref{ex:b2:comparison:centralbinomial}:
\[
C_n \sim \frac{4^n}{\sqrt{\pi}\; n^{3/2}} .
\]
\end{solution}

\begin{solution}{exo:b2:powerseries:9}
साथ $ A_n = \sum_{k \leq n} a_k \to A $: \omterm{thm:b2:series:abel}{हाबिल सारांश} देता है, के लिए
$0 \leq x < 1$,
\[
\sum_{n=0}^{\infty} a_n x^n = (1 - x)\sum_{n=0}^{\infty} A_n x^n
\]
(दोनों पक्ष जुटे: $(A_n)$ घिरा है; और पहचान अनुसरण करती है
से $ a_n = A_n - A_{n-1}$ और पुनः अनुक्रमणिका)। तब से $(1 -
x)\sum x^n = 1$:
\[
\sum_n a_nx^n - A = (1-x)\sum_{n} (A_n - A)x^n .
\]
दिया गया $\varepsilon $, चुनना $ N $ साथ $\abs{A_n - A} \leq
\varepsilon $ के लिए $ n > N $; तब
\[
\Bigl|\sum a_nx^n - A\Bigr|
\leq (1-x)\sum_{n \leq N}\abs{A_n - A} + \varepsilon(1 -
x)\sum_{n > N}x^n
\leq (1-x)\,C_N + \varepsilon ,
\]
और देना $ x \to 1^-$: लिमसुप $\leq \varepsilon $ सभी के लिए
$\varepsilon $. अतः रेडियल सीमा है $ A $.

अनुप्रयोग: $\sum \frac{(-1)^{n-1}}{n}$ \omterm{def:b2:integration:improper}{अभिसरण} (वैकल्पिक),
और के लिए $ x < 1$ इसकी शक्ति शृंखला का योग है $\ln(1 + x)$: हाबिल द्वारा,
योग है $\ln 2$. वैसे ही $\sum\frac{(-1)^n}{2n+1}x^{2n+1} =
\arctan x $ देता है $\frac\pi4$ पर $ x = 1$ --- वर्ष 1 अभिन्न
सबूत, अब संरचनात्मक।
\end{solution}

\begin{solution}{exo:b2:powerseries:10}
दोनों $\sum\frac{x^n}{n}$ और $\sum\frac{x^n}{n+1}$ त्रिज्या है
$1$, और $\frac{1}{n(n+1)} = \frac1n - \frac1{n+1}$, अभीतक के लिए तो $0
< \abs x < 1$:
\[
\sum_{n\geq1}\frac{x^n}{n(n+1)}
= -\ln(1-x) - \frac1x\sum_{n\geq1}\frac{x^{n+1}}{n+1}
= -\ln(1-x) - \frac{-\ln(1-x) - x}{x}
= 1 + \frac{1-x}{x}\ln(1-x) .
\]
RADIUS $1$; और $\norm{x^n/(n(n+1))}_{\infty,\intcc{-1}1} =
\frac{1}{n(n+1)}$ है \omterm{def:b2:series:summable}{संक्षेपणीय}: सामान्य \omterm{def:b2:integration:improper}{अभिसरण} पर
$\intcc{-1}{1}$, तो योग है \omterm{def:b2:metric:continuity}{निरंतर} वहाँ। जैसा $ x \to
1^-$, $(1-x)\ln(1-x) \to 0$ और बंद रूप की प्रवृत्ति होती है $1$ ---
टेलीस्कोपिंग मूल्य के अनुरूप $\sum\frac{1}{n(n+1)} =
\lim_N\bigl(1 - \frac{1}{N+1}\bigr) = 1$ पर $ x = 1$.
\end{solution}

\begin{solution}{exo:b2:powerseries:11}
क्रमबद्ध करना $ n!$ उनके निश्चित-बिंदु सेट द्वारा क्रमपरिवर्तन: चुनना
द $ k $ निश्चित बिंदु ($\binom nk $ तरीके) और दूसरे को परेशान करना
$ n - k $ वस्तुएं देता है $ n! = \sum_{k=0}^n\binom nk D_{n-k}$.
द्वारा विभाजित करना $ n!$:
\[
1 = \sum_{k=0}^{n}\frac{1}{k!}\cdot\frac{D_{n-k}}{(n-k)!} ,
\]
जो बिलकुल यही कहता है \omterm{thm:b2:series:fubini}{कॉची उत्पाद} का $\eu^x =
\sum\frac{x^k}{k!}$ और $ D(x) = \sum D_n\frac{x^n}{n!}$ है
$\sum x^n = \frac{1}{1-x}$. दोनों कारक मिलते हैं \omterm{def:b2:series:def}{बिल्कुल}
के लिए $\abs x < 1$ ($ D_n \leq n!$, इसलिए $ D $ का प्रभुत्व है
ज्यामितीय श्रृंखला): उत्पाद पहचान वैध है
(\cref{prop:b2:powerseries:operations}), और
\[
D(x) = \frac{\eu^{-x}}{1-x} .
\]
\omterm{thm:b2:series:fubini}{कॉची उत्पाद} का $\eu^{-x} = \sum\frac{(-1)^kx^k}{k!}$ और
$\sum x^m $: का गुणांक $ x^n $ है
$\sum_{k=0}^{n}\frac{(-1)^k}{k!}$, और शक्ति की विशिष्टता से
श्रृंखला गुणांक (\cref{thm:b2:powerseries:calculus}):
\[
\frac{D_n}{n!} = \sum_{k=0}^{n}\frac{(-1)^k}{k!}
\xrightarrow[n\to\infty]{} \eu^{-1} :
\]
के बारे में $37\%$ सभी क्रमपरिवर्तन, चाहे जो भी हों, विक्षिप्तताएँ हैं
$ n $.
\end{solution}

\begin{solution}{exo:b2:powerseries:12}
द्विपद श्रृंखला के साथ $\alpha = -\frac12$ पर $-4x $:
\[
\binom{-1/2}{n}(-4)^n
= \frac{\bigl(-\frac12\bigr)\bigl(-\frac32\bigr)\cdots
\bigl(-\frac{2n-1}2\bigr)}{n!}(-4)^n
= \frac{1\cdot3\cdots(2n-1)}{2^n\,n!}\,4^n
= \frac{(2n)!}{2^n n!}\cdot\frac{2^n}{n!}
= \binom{2n}{n},
\]
का उपयोग करते हुए $1\cdot3\cdots(2n-1) = \frac{(2n)!}{2^nn!}$. इस तरह
$(1-4x)^{-1/2} = \sum\binom{2n}nx^n $ के लिए $\abs{4x} < 1$.
चुकता करना (\omterm{thm:b2:series:fubini}{कॉची उत्पाद}, पूर्ण \omterm{def:b2:integration:improper}{अभिसरण} द्वारा वैध)
और तुलना कर रहे हैं $\frac{1}{1-4x} = \sum 4^nx^n $: द
का गुणांक $ x^n $ वर्ग में है
$\sum_{k=0}^n\binom{2k}k\binom{2n-2k}{n-k}$, और की विशिष्टता
गुणांक देता है
\[
\sum_{k=0}^{n}\binom{2k}{k}\binom{2n-2k}{n-k} = 4^n .
\]
\end{solution}

\begin{solution}{pb:b2:powerseries:1}
\textbf{1.} साथ $ a_n = r_n - r_{n+1}$, भागों द्वारा योग:
\[
\sum_{n=N}^{M} a_nx^n
= r_Nx^N + \sum_{n=N+1}^{M} r_n\bigl(x^n - x^{n-1}\bigr)
- r_{M+1}x^M .
\]
के लिए $0 \leq x \leq 1$ वेतन वृद्धि $ x^{n-1} - x^n $ हैं
गैर-नकारात्मक और दूरबीन $ x^N - x^M $; साथ $ s = \sup_{n\geq
N}\abs{r_n}$:
\[
\Bigl|\sum_{n=N}^{M}a_nx^n\Bigr|
\leq s\bigl(x^N + (x^N - x^M) + x^M\bigr) = 2s\,x^N \leq 2s .
\]

\textbf{2.} तब से $ r_n \to 0$, $\sup_{n\geq N}\abs{r_n} \to 0$:
प्रश्न 1 \omterm{def:b2:series:def}{बिल्कुल} एक समान कॉची मानदंड है
$\intcc{0}{1}$, इसलिए $\sum a_nx^n $ \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:funcseq:def}{समान रूप से} वहाँ और
इसका योग है \omterm{def:b2:metric:continuity}{निरंतर} (\cref{thm:b2:funcseq:seriestransfer}).
पर मूल्य $1$ प्राणी $\sum a_n $, निरंतरता पर $1$ है
की रेडियल सीमा \cref{exo:b2:powerseries:9}.

\textbf{3.} के लिए $\abs x < 1$ सभी तीन शक्ति श्रृंखलाएँ अभिसरित होती हैं
\omterm{def:b2:series:def}{बिल्कुल} और $\bigl(\sum a_nx^n\bigr)\bigl(\sum b_nx^n\bigr)
= \sum c_nx^n $ (\cref{prop:b2:powerseries:operations}). द्वारा
प्रश्न 2 प्रत्येक कारक और उत्पाद पक्ष हैं \omterm{def:b2:metric:continuity}{निरंतर} पर
$\intcc{0}{1}$ (उनकी गुणांक श्रृंखला \omterm{def:b2:integration:improper}{अभिसरण} करती है
परिकल्पना); दे $ x \to 1^-$ पहचान में: $ AB = C $.

\textbf{4.} यहाँ
\[
\abs{c_n} = \sum_{k=0}^{n}
\frac{1}{\sqrt{(k+1)(n-k+1)}}
\geq \sum_{k=0}^{n}\frac{2}{n+2}
= \frac{2(n+1)}{n+2} \geq 1,
\]
एएम--जीएम द्वारा: $\sqrt{(k+1)(n-k+1)} \leq \frac{(k+1) +
(n-k+1)}{2} = \frac{n+2}{2}$. का सामान्य शब्द $\sum c_n $
प्रवृत्ति नहीं है $0$: द \omterm{thm:b2:series:fubini}{कॉची उत्पाद} हालाँकि, विचलन होता है
दोनों कारक \omterm{def:b2:integration:improper}{अभिसरण} (वैकल्पिक श्रृंखला) करते हैं।

\textbf{5.} \cref{exo:b2:powerseries:5} देता है
$\frac{-\ln(1-x)}{1-x} = \sum H_nx^n $ ($\abs x < 1$).
शब्द-दर-अवधि आदिम
(\cref{thm:b2:powerseries:calculus}~(2)), दोनों पक्ष लुप्त हो रहे हैं
पर $0$:
\[
\frac{\bigl(\ln(1-x)\bigr)^2}{2}
= \sum_{n\geq1}\frac{H_n}{n+1}\,x^{n+1} .
\]
घटाना: $(n+2)H_n \geq (n+1)H_{n+1}$ के बराबर है $ H_n \geq
1$, के लिए सच है $ n \geq 1$; और $\frac{H_n}{n+1} \sim \frac{\ln
n}{n} \to 0$: पर $ x = -1$ श्रृंखला द्वारा \omterm{def:b2:integration:improper}{अभिसरण} होता है
वैकल्पिक परीक्षण. स्थानापन्न $ x \mapsto -x $ और आवेदन कर रहे हैं
प्रश्न 2 पर $ x = 1$:
\[
\frac{(\ln 2)^2}{2}
= \sum_{n\geq1}\frac{H_n}{n+1}(-1)^{n+1},
\]
घोषित मूल्य.

\textbf{6.} $ f(x) = \sum(-1)^nx^n = \frac{1}{1+x} \to \frac12$
जैसा $ x \to 1^-$: हाबिल-सारांश करने योग्य $\frac12$. लेकिन आंशिक
रकम हैं $1, 0, 1, 0, \dots $: भिन्न।

\textbf{7.} दिया गया $\varepsilon > 0$ चुनना $ m $ साथ $\abs{u_n}
\leq \varepsilon $ के लिए $ n > m $; के लिए $ N \geq m $:
\[
\Bigl|\frac{u_1 + \dots + u_N}{N}\Bigr|
\leq \frac{\abs{u_1} + \dots + \abs{u_m}}{N} +
\varepsilon\,\frac{N - m}{N}
\leq \frac{C_m}{N} + \varepsilon,
\]
इसलिए $\limsup \leq \varepsilon $ हरएक के लिए $\varepsilon $: द
मतलब प्रवृत्त होते हैं $0$.

\textbf{8.} के लिए $0 \leq x \leq 1$: $1 - x^n = (1-x)(1 + x +
\dots + x^{n-1}) \leq n(1-x)$, इसलिए
\[
\Bigl|\sum_{n=0}^N a_n(1 - x_N^n)\Bigr|
\leq (1 - x_N)\sum_{n=1}^N n\abs{a_n}
= \frac1N\sum_{n=1}^{N}n\abs{a_n} .
\]
के लिए $ n > N $: $\abs{a_n} = \frac{n\abs{a_n}}{n} \leq
\frac{1}{N}\sup_{m>N}m\abs{a_m}$, और $\sum_{n>N}x_N^n \leq
\frac{1}{1 - x_N} = N $:
\[
\Bigl|\sum_{n>N}a_nx_N^n\Bigr|
\leq \frac{\sup_{m>N}m\abs{a_m}}{N}\cdot N
= \sup_{m>N}\,m\abs{a_m} .
\]

\textbf{9.} विघटित
\[
A_N - L = \sum_{n=0}^{N}a_n\bigl(1 - x_N^n\bigr)
- \sum_{n>N}a_nx_N^n + \bigl(f(x_N) - L\bigr) .
\]
पहला पद प्रवृत्त होता है $0$ प्रश्न 7 द्वारा (का साधन)
$ n\abs{a_n} \to 0$), प्रश्न 8 द्वारा दूसरा (सुपर की प्रवृत्ति होती है
$0$), तीसरा क्योंकि $ x_N \to 1^-$ और $ f(x) \to L $. इस तरह
$ A_N \to L $: टाउबर का प्रमेय.

\textbf{10.} के लिए $ x \in \intco{0}{1}$ और कोई भी $ N $:
$\sum_{n\leq N}a_nx^n \leq f(x) \leq M $ (गैर-नकारात्मक शब्द)।
होने देना $ x \to 1^-$ परिमित योग में: $\sum_{n\leq N}a_n \leq M $.
आंशिक रकम बढ़ रही है और सीमित है: $\sum a_n $
\omterm{def:b2:integration:improper}{अभिसरण} करता है, और फिर प्रश्न 2 देता है $\lim_{x\to1^-}f(x) =
\sum a_n $.

\textbf{11.} $\sigma_N - L $ का माध्य है $ N $ नंबर
$ A_n - L $ ($0 \leq n < N $), जिसकी प्रवृत्ति होती है $0$: प्रश्न 7.

\textbf{12.} $ A_n = 1$ एक जैसे के लिए $ n $, $0$ विषम के लिए: $ A_0 + \dots
+ A_{N-1} = \lceil N/2\rceil $, इसलिए $\sigma_N =
\frac{\lceil N/2\rceil}{N} \to \frac12$, का हाबिल मूल्य
प्रश्न 6.

\textbf{13.} अंतर्गत $\sigma_N \to L $ एक के पास है $ S_n = O(n)$, इस तरह
$ A_n = S_n - S_{n-1} = O(n)$ और $ a_n = O(n)$: सारी श्रृंखला
नीचे त्रिज्या है $\geq 1$. के लिए $\abs x < 1$, से $ a_n = A_n -
A_{n-1}$ और $ A_nx^n \to 0$:
\[
(1-x)\sum_n A_nx^n = \sum_n A_nx^n - \sum_n A_nx^{n+1}
= \sum_n a_nx^n = f(x),
\]
और \omterm{def:b2:funcseq:def}{समान रूप से} $(1-x)\sum S_nx^n = \sum A_nx^n $, इसलिए $ f(x) =
(1-x)^2\sum_n S_nx^n = (1-x)^2\sum_n(n+1)\sigma_{n+1}x^n $.
अंत में $\sum(n+1)x^n = \frac{1}{(1-x)^2}$
(\cref{exo:b2:powerseries:3}), जो दूसरी पहचान है।

\textbf{14.} घटाने $ L $ से दूसरी पहचान का समय
प्रथम:
\[
f(x) - L = (1-x)^2\sum_{n\geq0}(n+1)
\bigl(\sigma_{n+1} - L\bigr)x^n .
\]
दिया गया $\varepsilon $, चुनना $ N $ साथ $\abs{\sigma_{n+1} - L}
\leq \varepsilon $ के लिए $ n \geq N $; तब
\[
\abs{f(x) - L} \leq (1-x)^2 C_N +
\varepsilon(1-x)^2\sum_{n}(n+1)x^n
= (1-x)^2C_N + \varepsilon ,
\]
और देना $ x \to 1^-$: $\limsup \leq \varepsilon $. इस तरह
$ f(x) \to L $: फ्रोबेनियस प्रमेय.

\textbf{15.} $ f(x) = \sum(-1)^n(n+1)x^n = \frac{1}{(1+x)^2}$
(ज्यामितीय श्रृंखला को अलग करें $-x $): हाबिल मूल्य
$\frac14$. आंशिक योग: $ A_{2k} = k+1$, $ A_{2k+1} = -(k+1)$
(तत्काल प्रेरण). तब $ S_{2m-1} = 0$ (लगातार जोड़े
रद्द करें) और $ S_{2m} = m + 1$, इसलिए
\[
\sigma_{2m} = \frac{S_{2m-1}}{2m} = 0,
\qquad
\sigma_{2m+1} = \frac{m+1}{2m+1} \to \frac12 :
\]
$(\sigma_N)$ दो अलग-अलग क्लस्टर मान हैं: नहीं
सेस\`एआरओ-योग्य। प्रश्न 6 और 11-14 के साथ, पदानुक्रम
अभिसारी $\Rightarrow $ सीईएस\`एआरओ $\Rightarrow $ हाबिल सख्त है
दोनों तीरों पर.

\textbf{16.} आदिम शृंखला $ F(x) =
\sum\frac{a_n}{n+1}x^{n+1}$ समान त्रिज्या है और $ F' = f $ पर
$\intco{0}{1}$ (\cref{thm:b2:powerseries:calculus}); तब से
$\sum\frac{a_n}{n+1}$ \omterm{def:b2:integration:improper}{अभिसरण}, भाग I (प्रश्न 2) बनाता है $ F $
\omterm{def:b2:metric:continuity}{निरंतर} पर $\intcc{0}{1}$. जैसा $\int_0^x f = F(x)$ (बराबर
व्युत्पन्न, समान मूल्य $0$ पर $0$),
\[
\int_0^x f \xrightarrow[x\to1^-]{} F(1)
= \sum_{n\geq0}\frac{a_n}{n+1} :
\]
 अभिन्न अनुचित बताए गए मूल्य के साथ मौजूद है।

\textbf{17.} $\frac{\ln(1+x)}{x} =
\sum_{n\geq1}\frac{(-1)^{n-1}}{n}x^{n-1}$ (त्रिज्या $1$;
\omterm{def:b2:metric:continuity}{निरंतर} पर $0$). की शृंखला $\frac{a_m}{m+1}$ है
$\sum_{n\geq1}\frac{(-1)^{n-1}}{n^2}$, \omterm{def:b2:series:def}{बिल्कुल} अभिसारी:
प्रश्न 16 देता है $\int_0^1\frac{\ln(1+x)}{x}\dd x = \eta $. में
द \omterm{def:b2:series:def}{बिल्कुल} संमिलित $\sum\frac1{n^2}$, फिर से इकट्ठा करें और भी
अजीब:
\[
\eta = \sum_{\text{odd}}\frac1{n^2} -
\sum_{\text{even}}\frac1{n^2}
= \sum_{n}\frac1{n^2} - 2\sum_{k}\frac1{(2k)^2}
= \Bigl(1 - \frac12\Bigr)\sum_n\frac1{n^2}
= \frac12\sum_{n\geq1}\frac1{n^2} .
\]

\textbf{18.} लीबनिज़: $\frac{1}{3n+1}\downarrow0$, श्रृंखला
जुटता है. पर $\intco{0}{1}$, $\sum(-1)^nx^{3n} =
\frac{1}{1+x^3}$, और $\sum\frac{(-1)^n}{3n+1}$ \omterm{def:b2:integration:improper}{अभिसरण}:
प्रश्न 16 देता है $\sum\frac{(-1)^n}{3n+1} =
\int_0^1\frac{\dd x}{1+x^3}$. आंशिक भिन्न (जांचें:
$\frac13(x^2-x+1) + \frac{2-x}{3}(1+x) = 1$):
\[
\int_0^1\frac{\dd x}{1+x^3}
= \frac13\ln2 + \frac13\int_0^1\frac{2-x}{x^2-x+1}\dd x .
\]
लिखना $2 - x = -\frac12(2x-1) + \frac32$: द $\ln(x^2-x+1)$
भाग दोनों सिरों पर गायब हो जाता है, और
\[
\frac32\int_0^1\frac{\dd x}{(x-\frac12)^2 + \frac34}
= \frac32\cdot\frac{2}{\sqrt3}
\Bigl[\arctan\frac{2x-1}{\sqrt3}\Bigr]_0^1
= \sqrt3\cdot\frac{\pi}{3} = \frac{\pi}{\sqrt3} .
\]
कुल: $\frac13\bigl(\ln2 + \frac{\pi}{\sqrt3}\bigr)$.

\textbf{19.} स्थानापन्न $ t = x^2$ की शृंखला में
\cref{exo:b2:powerseries:12} और द्वारा शब्द को एकीकृत करना
शब्द (का आदिम) $(1-x^2)^{-1/2}$ पर लुप्त हो रहा है $0$ है
$\arcsin $):
\[
\arcsin x = \sum_{n\geq0}
\frac{\binom{2n}n}{4^n(2n+1)}x^{2n+1}
\qquad(\abs x < 1) .
\]
गुणांक हैं $\sim \frac{1}{2\sqrt\pi\,n^{3/2}}$
(\cref{ex:b2:comparison:centralbinomial}), \omterm{def:b2:series:summable}{संक्षेपणीय}: शृंखला
जुटता है \emph{\omterm{def:b2:funcseq:series}{सामान्यतः}} पर $\intcc{-1}{1}$, इसका योग है
\omterm{def:b2:metric:continuity}{निरंतर} वहाँ, और के साथ मेल खाता है \omterm{def:b2:metric:continuity}{निरंतर} $\arcsin $
पर $\intoo{-1}{1}$, इसलिए पर $ x = 1$ बहुत:
\[
\sum_{n\geq0}\frac{\binom{2n}n}{4^n(2n+1)}
= \arcsin 1 = \frac\pi2 .
\]

\textbf{20.} $ C_n4^{-n} \sim \frac{1}{\sqrt\pi\,n^{3/2}}$
(\cref{exo:b2:powerseries:8}): सामान्य \omterm{def:b2:integration:improper}{अभिसरण} का $\sum
C_nx^n $ पर $\intcc{0}{\frac14}$, तो इसका योग है \omterm{def:b2:metric:continuity}{निरंतर}
वहाँ; पर $\intoo{0}{\frac14}$ यह बराबर है
$\frac{1-\sqrt{1-4x}}{2x}$
(\cref{ex:b2:powerseries:catalan}), जिसकी सीमा पर
$\frac14^-$ है $\frac{1-0}{1/2} = 2$. इस तरह $\sum_{n\geq0}
C_n4^{-n} = 2$.

\textbf{21.} $ f(x) = \sum_{n\geq m}a_nx^n = x^m g(x)$ साथ
$ g(x) = \sum_{k\geq0}a_{m+k}x^k $; अगर $(a_nr^n)$ तब परिबद्ध है
$(a_{m+k}r^k)$ बहुत है (से विभाजित करें)। $ r^m $): $ g $ त्रिज्या है $\geq
R $. $ g $ है \omterm{def:b2:metric:continuity}{निरंतर} साथ $ g(0) = a_m \neq 0$, इसलिए $ g \neq 0$
कुछ पर $\intcc{-\delta}{\delta}$, और $ f(x) = x^mg(x) \neq 0$
के लिए $0 < \abs x \leq \delta $.

\textbf{22.} $ d = f - h $ निकट घात श्रृंखला का योग है $0$
शून्येतर बिंदुओं पर लुप्त हो जाना $ x_k \to 0$. यदि कुछ
का गुणांक $ d $ शून्येतर थे, प्रश्न 21 एक देगा
का पड़ोस पंचर हो गया $0$ के शून्य से मुक्त $ d $ ---
विरोधाभासी $ d(x_k) = 0$. तो सभी गुणांक $ d $ गायब होना:
$ f $ और $ h $ समान गुणांक रखते हैं और निकट संपाती होते हैं $0$.

\textbf{23.} समारोह $ h(x) = \frac{1}{1+x^2} =
\sum(-1)^nx^{2n}$ (त्रिज्या $1$) संतुष्ट करता है $ h(\frac1k) =
\frac{1}{1 + 1/k^2} = \frac{k^2}{k^2+1}$. कोई \omterm{def:b2:powerseries:analytic}{विश्लेषणात्मक} $ f $ के साथ
उन्हीं मूल्यों से सहमत हैं $ h $ बिंदुओं पर $\frac1k \to 0$:
पहचान प्रमेय द्वारा (प्रश्न 22), $ f = \frac{1}{1+x^2}$
पास में $0$ --- अनोखा समाधान.

\textbf{24.} होने देना $ Z $ के बिंदुओं का समुच्चय हो $ I $ होना
पड़ोस जिस पर $ f $ \omterm{def:b2:funcseq:def}{समान रूप से} गायब हो जाता है: \omterm{def:b2:metric:topology}{खुला} द्वारा
परिभाषा, गैररिक्त (उपअंतराल)। बंद किया $ I $: अगर $ y
\in I $ के अंकों की एक सीमा है $ Z $, तब $ y $ एक संचय है
के शून्य का बिंदु $ f $; का विस्तार $ f $ एक शक्ति श्रृंखला में $ y $
(विश्लेषणात्मकता) और प्रश्न 21-22 को लागू करने पर पुनः केन्द्रित $ y $,
सभी गुणांक पर $ y $ गायब हो जाओ, तो $ f \equiv 0$ पास में $ y $: $ y
\in Z $. एक अंतराल है \omterm{def:b2:metric:connected}{जुड़े हुए}, इसलिए $ Z = I $: $ f \equiv 0$ पर
$ I $. विशेष रूप से ए \omterm{def:b2:powerseries:analytic}{विश्लेषणात्मक} कार्य पर $\R $ लुप्त हो जाना
बाहर ए \omterm{def:b2:metric:compact}{कॉम्पैक्ट} इसलिए, अंतराल पर सेट गायब हो जाता है
हर जगह: कोई शून्येतर नहीं \omterm{def:b2:powerseries:analytic}{विश्लेषणात्मक} उभार। $ C^\infty $ दुनिया है
अलग: के फ्लैट फ़ंक्शन को ग्लूइंग करना
\cref{exo:b2:powerseries:7} (उदा.\ $ x \mapsto
\eu^{-1/x^2}\mathbf 1_{x>0}$ और इसका दर्पण) चिकना पैदा करता है
के साथ टकराता है \omterm{def:b2:metric:compact}{कॉम्पैक्ट} सहायता।

\textbf{25.} (मैं) सामान्य \omterm{def:b2:integration:improper}{अभिसरण} रहता है \omterm{def:b2:metric:compact}{कॉम्पैक्ट} उपडिस्क
डिस्क के अंदर सख्ती से; हाबिल का प्रमेय विस्तारित होता है निरंतरता को
एक सीमा बिंदु, एकमात्र परिकल्पना के तहत कि
गुणांक श्रृंखला वहां एकत्रित होती है। (ii) इसका विपरीत सत्य है
टाउबर की शर्त के तहत $ na_n \to 0$ (प्रश्न 9), और
सेस\`एआरओ संक्षेपण \omterm{def:b2:integration:improper}{अभिसरण} और के बीच सख्ती से बैठता है
हाबिल संक्षेपण (फ्रोबेनियस, प्रश्न 14--15)। (iii) ए के लिए
दोस्त: $\sum\frac{(-1)^n}{3n+1} = \int_0^1\frac{\dd x}{1+x^3}$
फिर, ज्यामितीय श्रृंखला को सीमा में एकीकृत करके
आंशिक अंश. (iv) सेस\`एआरओ का मतलब फूरियर में वापसी है
फ़ेज़ के रूप में अध्याय\'एर का प्रमेय, जहां आंशिक योगों का औसत निकाला जाता है
की विफलता को सुधारता है \omterm{def:b2:funcseq:def}{बिंदुवार} \omterm{def:b2:integration:improper}{अभिसरण} --- वही दवा,
नया रोगी.
\end{solution}
