\section*{अध्याय \ref{ch:b2:genfun} --- संभाव्यता उत्पन्न करने वाले कार्य}

\begin{solution}{exo:b2:genfun:1}
सही छाप: $G(t) = \frac16(t + t^2 + \dots + t^6) = \frac t6(1 + t
+ \dots + t^5)$. यदि दो निष्पक्ष पासों का योग एकसमान हो
$\{2, \dots, 12\}$, तब
\[
G(t)^2 = \frac{t^2}{36}\,h(t)^2
= \frac{t^2}{11}\sum_{k=0}^{10}t^k ,
\qquad h(t) = 1 + t + \dots + t^5 .
\]
अब $ h $ विषम डिग्री का एक वास्तविक बहुपद है $5$, तो यह एक वास्तविक है
जड़ (मध्यवर्ती मूल्य प्रमेय; ठोस रूप से $ h(-1) = 0$), इस तरह
$ h^2$ एक वास्तविक जड़ है. लेकिन $\sum_{k=0}^{10}t^k $ कोई नहीं है: यह है
के लिए सकारात्मक $ t \geq 0$, और के लिए $ t < 0$ यह बराबर है
$\frac{t^{11} - 1}{t - 1}$, दो ऋणात्मक संख्याओं का भागफल।
विरोधाभास --- दो निष्पक्ष पासों का योग कभी भी एक समान नहीं होता (जैसा कि)
परिचित \omterm{def:b2:reduction:diag}{त्रिकोणीय} \omterm{def:b2:randomvar:law}{वितरण} पासों का योग पुष्टि करता है)।
\end{solution}

\begin{solution}{exo:b2:genfun:2}
\emph{द्विपद:} $ G(t) = (1 - p + pt)^n $, $ G'(t) = np(1 - p +
pt)^{n-1}$, $ G''(t) = n(n-1)p^2(1 - p + pt)^{n-2}$, इसलिए
\[
\E(X) = G'(1) = np,
\qquad
V(X) = G''(1) + G'(1) - G'(1)^2
= n(n-1)p^2 + np - n^2p^2 = np(1-p).
\]
\emph{ज्यामितिक} ($ q = 1 - p $): $ G(t) = \frac{pt}{1 - qt}$, इसलिए
$ G'(t) = \frac{p}{(1 - qt)^2}$ और $ G''(t) = \frac{2pq}{(1 -
qt)^3}$; पर $ t = 1$ (का उपयोग करते हुए $1 - q = p $):
\[
\E(X) = \frac{p}{p^2} = \frac1p,
\qquad
V(X) = \frac{2q}{p^2} + \frac1p - \frac{1}{p^2}
= \frac{2q + p - 1}{p^2}
= \frac{q}{p^2} ,
\]
मेल मिलाना \cref{exo:b2:randomvar:1} कम काम के साथ.
\end{solution}

\begin{solution}{exo:b2:genfun:3}
नहीं, अलग-अलग लोडिंग के साथ भी। कल्पना करना $ X, Y $ हैं \omterm{def:b2:randomvar:law}{कानून} पर
$\{1, \dots, 6\}$ एक समान राशि के साथ. तब $ G_X(t) = t\,a(t)$ और
$ G_Y(t) = t\,b(t)$ साथ $ a, b $ डिग्री के वास्तविक बहुपद \emph{अधिक से अधिक} $5$ --- और उनकी डिग्रियों का योग अवश्य होना चाहिए $10$ (योग पहुँचता है
$12$ सकारात्मक \omterm{def:b2:multint:exact}{संभावना} के साथ), तो $\deg a = \deg b = 5$, दोनों
अजीब. के रूप में \cref{exo:b2:genfun:1},
\[
a(t)\,b(t) = \frac{1}{11}\sum_{k=0}^{10}t^k
\]
बायीं ओर एक वास्तविक जड़ को बाध्य करेगा (प्रत्येक विषम-डिग्री वास्तविक
बहुपद में दाहिनी ओर एक) और कोई नहीं है। तो दो की कोई लोडिंग नहीं
\omterm{def:b2:proba:independence}{स्वतंत्र} पासा --- बराबर या नहीं --- एक समान योग उत्पन्न करता है।
\end{solution}

\begin{solution}{exo:b2:genfun:4}
द्वारा \cref{thm:b2:genfun:compound} साथ $ G_N(s) = e^{\lambda(s-1)}$
और $ G_X(t) = 1 - p + pt $:
\[
G_S(t) = e^{\lambda(1 - p + pt - 1)} = e^{\lambda p(t - 1)} :
\]
$ S \sim \mathcal{P}(\lambda p)$. छोड़ी गई गिनती $ D = N - S $
संभाव्यता के साथ रखी गई समान वस्तुओं को गिनता है $1 - p $, तो द्वारा
वही गणना $ D \sim \mathcal{P}(\lambda(1 - p))$.
आज़ादी, सीधे: के लिए $ j, k \in \N $,
\begin{align*}
\P(S = j,\ D = k)
&= \P(N = j + k)\,\binom{j+k}{j}p^jq^k
= e^{-\lambda}\frac{\lambda^{j+k}}{(j+k)!}\,
\frac{(j+k)!}{j!\,k!}\,p^jq^k\\
&= \Bigl(e^{-\lambda p}\frac{(\lambda p)^j}{j!}\Bigr)
\Bigl(e^{-\lambda q}\frac{(\lambda q)^k}{k!}\Bigr)
\end{align*}
साथ $ q = 1 - p $: \omterm{def:b2:quadratic:adjoint}{जोड़} \omterm{def:b2:randomvar:law}{कानून} जैसे कारक
$\mathcal{P}(\lambda p) \otimes \mathcal{P}(\lambda q)$. एक पॉइसन
यादृच्छिक उपज पर प्रवाह विभाजन \emph{\omterm{def:b2:proba:independence}{स्वतंत्र}} पॉइसन प्रवाह ---
कतार सिद्धांत में लगातार उपयोग किया जाने वाला एक छोटा सा चमत्कार।
\end{solution}

\begin{solution}{exo:b2:genfun:5}
लगातार शीर्षों के बीच प्रतीक्षा समय है \omterm{def:b2:proba:independence}{स्वतंत्र}
ज्यामितिक $\mathcal{G}(p)$ चर (स्मृतिहीनता: प्रत्येक के बाद
हेड गेम रीस्टार्ट होता है), तो $ T_r = W_1 + \dots + W_r $ और
गुणात्मकता (\cref{thm:b2:genfun:product}) देता है
\[
G_{T_r}(t) = \Bigl(\frac{pt}{1 - qt}\Bigr)^{r},
\qquad
\E(T_r) = r\,\E(W_1) = \frac rp,
\qquad
V(T_r) = r\,V(W_1) = \frac{rq}{p^2}
\]
($ q = 1 - p $; प्रसरण द्वारा \omterm{def:b2:quadratic:adjoint}{जोड़}ें स्वतंत्रता). विस्तार: द्वारा
सामान्यीकृत द्विपद श्रृंखला
(\cref{ch:b2:powerseries}), $(1 - qt)^{-r} = \sum_{m\geq0}
\binom{m + r - 1}{r - 1}q^mt^m $, तो का गुणांक $ t^n $ में
$ p^rt^r(1 - qt)^{-r}$ (के साथ) है $ m = n - r $)
\[
\P(T_r = n) = \binom{n-1}{r-1}p^r(1-p)^{n-r},
\qquad n \geq r ,
\]
 \emph{नकारात्मक द्विपद} \omterm{def:b2:randomvar:law}{कानून} --- संयुक्त रूप से: द $ r $-वें
सिर टॉस पर गिर जाता है $ n $ यदि पिछला $ r - 1$ प्रमुख अपना चयन करते हैं
प्रथम स्थान पर है $ n - 1$ उछालता है।
\end{solution}

\begin{solution}{exo:b2:genfun:6}
$ m = 1\cdot\frac38 + 2\cdot\frac38 + 3\cdot\frac18 = \frac{3 + 6 +
3}{8} = \frac32 > 1$: सुपरक्रिटिकल. \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन} है
\[
G(t) = \frac{1 + 3t + 3t^2 + t^3}{8} = \frac{(1 + t)^3}{8} ,
\]
तो निश्चित बिंदु हल करते हैं $(1 + t)^3 = 8t $, यानी\ $ t^3 + 3t^2 - 5t + 1
= 0$. गारंटीशुदा रूट को फैक्टरिंग करना $ t = 1$:
\[
t^3 + 3t^2 - 5t + 1 = (t - 1)\bigl(t^2 + 4t - 1\bigr),
\]
और $ t^2 + 4t - 1 = 0$ देता है $ t = -2 \pm \sqrt5$. जड़ में
$\intco{0}{1}$ है $\sqrt5 - 2 \approx 0.236$: द्वारा
\cref{thm:b2:genfun:extinction},
\[
q = \sqrt 5 - 2 .
\]
(एक सुखद जाँच: संतान \omterm{def:b2:randomvar:law}{कानून} क्या यह $3$ \omterm{def:b2:proba:independence}{स्वतंत्र}
उचित सिक्के, $ Z_1 \sim \mathcal{B}(3, \frac12)$.)
\end{solution}

\begin{solution}{exo:b2:genfun:7}
\emph{\omterm{def:b2:randomvar:expectation}{अपेक्षा}.} पहला $\E(Z_n) = m^n $: द्वारा
\cref{thm:b2:genfun:compound}, $\E(Z_{n+1}) = \E(Z_n)\,m $, और
$\E(Z_0) = 1$. परिवार $\bigl(Z_n(\omega)\P(\{\omega\})
\bigr)_{n, \omega}$ गैर-नकारात्मक है, इसलिए परिवारों के लिए फ़ुबिनी लागू होता है
बिना शर्त:
\[
\E(Y) = \sum_{n=0}^{\infty}\E(Z_n)
= \sum_{n=0}^\infty m^n = \frac{1}{1 - m} < \infty
\]
(विशेष रूप से $ Y $ लगभग निश्चित रूप से परिमित है: के अनुरूप
सबक्रिटिकल मामले में निश्चित विलुप्ति)।

\emph{कार्यात्मक समीकरण.} द्वारा जनसंख्या का विघटन करें
पूर्वज के बच्चे: यदि पूर्वज के पास है $ Z_1 = k $ बच्चे,
कुल संतान है $ Y = 1 + Y_1 + \dots + Y_k $, कहाँ $ Y_i $ है
की कुल संतान $ i $-वें बच्चे की पंक्ति --- और $ Y_i $ हैं
\omterm{def:b2:proba:independence}{स्वतंत्र} की प्रतियाँ $ Y $, से \omterm{def:b2:proba:independence}{स्वतंत्र} $ Z_1$ (विशिष्ट पंक्तियाँ
असंयुक्त का उपयोग करें, \omterm{def:b2:proba:independence}{स्वतंत्र} प्रजनन \omterm{def:b2:proba:space}{घटनाएँ}). कंडीशनिंग चालू
$ Z_1$ के रूप में \cref{thm:b2:genfun:compound}:
\[
H(t) = \E\bigl(t^Y\bigr)
= t\sum_{k=0}^\infty \P(Z_1 = k)\,H(t)^k
= t\,G\bigl(H(t)\bigr),
\]
कारक $ t $ पूर्वज का ही हिसाब। (के लिए
\omterm{def:b2:randomvar:law}{कानून} $ p_0 = 1 - p $, $ p_2 = p $ द्विआधारी शाखा का, यह \omterm{pb:b2:surfaces:1}{द्विघात}
में समीकरण $ H $ स्पष्ट रूप से हल किया जा सकता है और विस्तारित किया जा सकता है ---
\omterm{ex:b2:powerseries:catalan}{कैटलन संख्या} का \cref{ch:b2:powerseries} परिवार की गिनती करो
पेड़.)
\end{solution}

\begin{solution}{exo:b2:genfun:8}
होने देना $ R > 1$ त्रिज्या बनें और ठीक करें $ t \in \intoo{1}{R}$. तब
$\E(t^X) = G(t) < \infty $, और मार्कोव की असमानता
(\cref{thm:b2:randomvar:markov}) गैर-नकारात्मक पर लागू होता है
परिवर्तनशील $ t^X $ स्तर पर $ t^n $:
\[
\P(X \geq n) = \P\bigl(t^X \geq t^n\bigr)
\leq \frac{G(t)}{t^n} = C\rho^n,
\qquad C = G(t),\quad \rho = \frac1t \in \intoo{0}{1}.
\]
\emph{बातचीत:} अगर $\P(X \geq n) \leq C\rho^n $, तब $ p_n \leq
\P(X \geq n) \leq C\rho^n $, अभीतक के लिए तो $\abs t < \frac1\rho $ द
शृंखला $\sum p_n\abs t^n $ \omterm{def:b2:integration:improper}{अभिसरण} ज्यामितीय का प्रभुत्व है
शृंखला $ C\sum(\rho\abs t)^n $: त्रिज्या कम से कम है
$\frac1\rho > 1$. की त्रिज्या \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन} और ज्यामितीय
पूँछ का क्षय एक ही गुण के दो पहलू हैं।
\end{solution}

\begin{solution}{exo:b2:genfun:9}
लिखना $ p_k^{(n)} = \P(X_n = k)$, $ p_k = \P(X = k)$.

\emph{मामला $ k = 0$.} के लिए $ t \in \intoo{0}{1}$ और कोई भी \omterm{def:b2:randomvar:law}{कानून}
$(q_j)$ साथ $\sum_j q_j \leq 1$:
\[
\Bigl|\,q_0 - \sum_j q_jt^j\Bigr|
= \sum_{j \geq 1} q_j t^j
\leq \sum_{j\geq1}t^j = \frac{t}{1 - t} .
\]
इस तरह
\[
\abs{p_0^{(n)} - p_0}
\leq \frac{2t}{1 - t}
+ \abs{G_{X_n}(t) - G_X(t)} .
\]
दिया गया $\varepsilon > 0$, चुनना $ t $ साथ $\frac{2t}{1-t} <
\frac\varepsilon2$, तब $ n_0$ ऐसा कि अंतिम पद है $<
\frac\varepsilon2$ के लिए $ n \geq n_0$: इसलिए $ p_0^{(n)} \to p_0$.

\emph{प्रेरण चरण.} मान लीजिए $ p_j^{(n)} \to p_j $ के लिए $ j < k $.
पर विचार करें \emph{स्थानांतरित कर दिया} कार्य
\[
g_n(t) = \frac{G_{X_n}(t) - p^{(n)}_0}{t}
= \sum_{j\geq0} p^{(n)}_{j+1}t^j,
\qquad
g(t) = \frac{G_X(t) - p_0}{t} ,
\]
कार्य उत्पन्न करना \omterm{def:b2:multint:exact}{उप-संभावना} अनुक्रमों का
$(p^{(n)}_{j+1})_j $ (कुल द्रव्यमान $\leq 1$, जो सब कुछ है
$ k = 0$ तर्क का प्रयोग किया गया)। तय के लिए $ t \in \intoo{0}{1}$, $ g_n(t)
\to g(t)$ परिकल्पना और मामले द्वारा $ k = 0$. को लागू करना $ k =
0$ को तर्क $ g_n $ देता है $ p_1^{(n)} \to p_1$; की पुनरावृति
बदलाव $ k $ समय देता है $ p_k^{(n)} \to p_k $ हरएक के लिए $ k $. (यह है
लेवी का असतत, प्राथमिक उदाहरण निरंतरता प्रमेय,
जिसका सामान्य रूप---विशेष कार्यों के लिए---एक वर्ष है
3 मील का पत्थर.)
\end{solution}

\begin{solution}{exo:b2:genfun:10}
\omterm{def:b2:funcseq:def}{बिंदुवार}, $\frac{1 + (-1)^X}{2}$ के बराबर होती है $1$ कब $ X $ सम है
और $0$ जब विषम, तो ले रहा हूँ अपेक्षाएं (\omterm{def:b2:structures:sn}{स्थानांतरण} करना),
\[
\P(X \text{ even}) = \frac{1 + \E\bigl((-1)^X\bigr)}2 =
\frac{1 + G_X(-1)}2 .
\]
पॉइसन: $\frac{1 + \eu^{-2\lambda}}2 \to \frac12$ जैसा
$\lambda $ बढ़ता है. द्विपद: $\frac{1 + (1 - 2p)^n}2$. दोनों ही मामलों में $ G_X(-1) \to 0$ की समता कहती है $ X $ मेला बन जाता है
सिक्का: \omterm{def:b2:randomvar:law}{कानून} कई पूर्णांकों में फैल जाता है और उसे भूल जाता है
समता.
\end{solution}

\begin{solution}{exo:b2:genfun:11}
$ t + \dots + t^6 = t\,\frac{1 - t^6}{1 - t}$ और $1 - t^6 =
(1 - t)(1 + t)(1 + t + t^2)(1 - t + t^2)$, कहा गया है
गुणनखंडीकरण. पहली मौत के लिए, $(1 + t)(1 + t + t^2) = 1 +
2t + 2t^2 + t^3$, इसलिए $\frac{t(1+t)(1+t+t^2)}6 = \frac{t +
2t^2 + 2t^3 + t^4}6$: चेहरे $\{1, 2, 2, 3, 3, 4\}$. के लिए
दूसरा, विस्तार
\[
(1 + 2t + 2t^2 + t^3)(1 - t + t^2)^2 = 1 + t^2 + t^3 + t^4 +
t^5 + t^7,
\]
इसलिए $\frac{t(1+t)(1+t+t^2)(1-t+t^2)^2}6 = \frac{t + t^3 + t^4
+ t^5 + t^6 + t^8}6$: चेहरे $\{1, 3, 4, 5, 6, 8\}$. द
दोनों का उत्पाद कार्य उत्पन्न करना छह को पुनः एकत्रित करता है
में कारक $\bigl(\frac{t(1+t)(1+t+t^2)(1-t+t^2)}6
\bigr)^2$, मानक पासे के कार्य का वर्ग:
सिचरमन \omterm{def:b2:quadratic:adjoint}{जोड़}ी \omterm{def:b2:series:def}{बिल्कुल} मानक है \omterm{def:b2:randomvar:law}{कानून} कुल के लिए ---
कार्य उत्पन्न करना ऐसे सभी पुनर्समूहन को वर्गीकृत करें।
\end{solution}

\begin{solution}{exo:b2:genfun:12}
होने देना $ A $ और $ B $ हो कार्य उत्पन्न करना शेष का
अवधि ``नो करंट हेड'' और ``वन करंट'' से शुरू हुई
सिर'' एक टॉस खर्च होता है, तो: राज्य से $0$, पूँछ
राज्य में वापस आ जाता है $0$, सिर राज्य की ओर बढ़ता है $1$; राज्य से
$1$, चित खेल समाप्त करता है, पट अपनी स्थिति में लौट आता है $0$:
\[
A(t) = t\bigl(q\,A(t) + p\,B(t)\bigr),
\qquad
B(t) = t\bigl(p + q\,A(t)\bigr).
\]
प्रतिस्थापन: $ A(1 - qt) = pt\,B = pt(pt + qtA)$, इसलिए
\[
G_T(t) = A(t) = \frac{p^2t^2}{1 - qt - pq\,t^2} .
\]
पर $ t = 1$ हर है $1 - q - pq = p(1 - q) = p^2$:
$ G_T(1) = 1$, खेल लगभग निश्चित रूप से समाप्त होता है (जैसे
\cref{exo:b2:proba:6} प्रत्यावर्तन द्वारा दिखाया गया)। लघुगणक
भेदभाव पर $1$: $\E(T) = 2 - \frac{D'(1)}{D(1)}$ साथ
$ D(t) = 1 - qt - pqt^2$, $ D'(1) = -q - 2pq $:
\[
\E(T) = 2 + \frac{q + 2pq}{p^2} = \frac{2p^2 + q + 2pq}{p^2}
= \frac{1 + p}{p^2},
\]
जो है $6$ के लिए $ p = \frac12$.
\end{solution}

\begin{solution}{pb:b2:genfun:1}
\textbf{1.} के लिए $ t \in \intoo01$, श्रृंखला नियम चालू $ G_n = G
\circ G_{n-1}$ देता है $ G_n'(t) =
G'\bigl(G_{n-1}(t)\bigr)G_{n-1}'(t)$. जैसा $ t \to 1^-$,
$ G_{n-1}(t) \uparrow 1$, और $ G'$ के साथ घट नहीं रहा है
बाईं सीमा $ m $ पर $1$, इसलिए पहला कारक प्रवृत्त होता है $ m $; द्वारा
प्रेरण दूसरे की ओर जाता है $ m^{n-1}$. द्वारा
\cref{thm:b2:genfun:moments}, $\E(Z_n) = G_n'(1^-) = m^n $.

\textbf{2.} एक बार फिर \omterm{def:b2:diffcalc:differential}{अंतर} करना,
\[
G_n'' = G''(G_{n-1})\,(G_{n-1}')^2 +
G'(G_{n-1})\,G_{n-1}'',
\]
और देना $ t \to 1^-$: $ a_n = G''(1)m^{2(n-1)} + m\,
a_{n-1}$ साथ $ a_n = G_n''(1)$, $ a_1 = G''(1)$. के लिए $ m \neq
1$ एक प्रेरण द्वारा जाँच करता है कि $ a_n = G''(1)\,m^{n-1}
\frac{m^n - 1}{m - 1}$ (पुनरावर्तन जोड़ता है $ G''(1)m^{2n-2}$
को $ m\cdot G''(1)m^{n-2}\frac{m^{n-1}-1}{m-1}$, और
$ m^{n-1} + \frac{m^{n-1}-1}{m-1} = \frac{m^n - 1}{m-1}$);
के लिए $ m = 1$, $ a_n = a_{n-1} + G''(1) = n\,G''(1)$.

\textbf{3.} $ V(Z_n) = a_n + m^n - m^{2n}$ और $ G''(1) =
\sigma^2 + m^2 - m $. के लिए $ m \neq 1$, टुकड़ा $(m^2 -
m)m^{n-1}\frac{m^n-1}{m-1} = m^n(m^n - 1)$ रद्द $ m^n -
m^{2n}$ बिलकुल, जा रहा हूँ $ V(Z_n) =
\sigma^2m^{n-1}\frac{m^n-1}{m-1}$. के लिए $ m = 1$: $ V(Z_n) =
nG''(1) = n\sigma^2$.

\textbf{4.} $ Z_n $ एक अऋणात्मक पूर्णांक चर है, इसलिए
$\P(Z_n > 0) = \P(Z_n \geq 1) \leq \E(Z_n) = m^n $ मार्कोव द्वारा
(\cref{thm:b2:randomvar:markov}). के लिए $ m < 1$ यह क्षय हो जाता है
ज्यामितीय रूप से --- और संक्षेप में, इसलिए बोरेल--कैंटेली भी देता है
केवल सीमित रूप से कई पीढ़ियाँ गैर-रिक्त हैं, जो कि है
फिर से विलुप्ति.

\textbf{5.} कॉची-श्वार्ज़: $\E(Z_n)^2 = \E(Z_n\mathbf
1_{Z_n>0})^2 \leq \E(Z_n^2)\,\P(Z_n > 0)$. प्रश्न 3 के साथ
और $ m < 1$:
\[
\E(Z_n^2) = V(Z_n) + m^{2n}
\leq \frac{\sigma^2m^{n-1}}{1-m} + m^{2n},
\]
तो, विभाजन $ m^{2n}$ इसके द्वारा बाध्य और सरलीकरण
$ m^n $,
\[
\P(Z_n > 0) \geq \frac{m^n}{\frac{\sigma^2}{m(1-m)} + m^n}
\geq \Bigl(\frac{\sigma^2}{m(1-m)} + 1\Bigr)^{-1}m^n ,
\]
का उपयोग करते हुए $ m^n \leq 1$ हर में. प्रश्न 4 के साथ:
$\P(Z_n > 0) \asymp m^n $.

\textbf{6.} $ G(t) = q\sum_k(pt)^k = \frac{q}{1 - pt}$, और $ m
= G'(1) = \frac{pq}{(1-p)^2} = \frac pq $. के लिए सबक्रिटिकल $ p
< \frac12$, के लिए महत्वपूर्ण $ p = \frac12$, के लिए अतिमहत्वपूर्ण $ p
> \frac12$.

\textbf{7.} $ G(t) = t $ पढ़ता $ pt^2 - t + q = 0$, जड़ों के साथ
$\frac{1 \pm \abs{p - q}}{2p}$, यानी\ $1$ और $\frac qp =
\frac1m $. विलुप्त होने की \omterm{def:b2:multint:exact}{संभावना} सबसे कम निश्चित है
इंगित करें $\intcc01$ (\cref{thm:b2:genfun:extinction}):
$ q_{\mathrm{ext}} = 1$ अगर $ m \leq 1$, और $\frac1m $ अगर $ m >
1$.

\textbf{8.} के लिए $ m \neq 1$, साथ $ p = \frac m{m+1}$, $ q =
\frac1{m+1}$: अगर $ q_n = \frac{m^n - 1}{m^{n+1} - 1}$, तब
\[
1 - p\,q_n = \frac{(m+1)(m^{n+1} - 1) - m(m^n - 1)}
{(m+1)(m^{n+1} - 1)} = \frac{m^{n+2} - 1}{(m+1)(m^{n+1} -
1)},
\]
इसलिए $ q_{n+1} = \frac{q}{1 - pq_n} = \frac{m^{n+1} -
1}{m^{n+2} - 1}$; आधार मामला $ q_0 = 0$ धारण करता है. के लिए $ m = 1$:
$ G(t) = \frac1{2 - t}$ और $ q_{n+1} = \frac1{2 -
\frac{n}{n+1}} = \frac{n+1}{n+2}$, साथ $ q_0 = 0$.

\textbf{9.} $1 - q_n = \frac{m^n(m - 1)}{m^{n+1} - 1}$. के लिए
$ m < 1$ हर की प्रवृत्ति होती है $-1$: $1 - q_n \sim (1 -
m)\,m^n $. के लिए $ m > 1$:
\[
q_{\mathrm{ext}} - q_n = \frac1m - \frac{m^n - 1}{m^{n+1} -
1} = \frac{m - 1}{m\,(m^{n+1} - 1)} \sim \frac{m -
1}{m^{2}}\;m^{-n} .
\]
और $ G'(t) = \frac{pq}{(1 - pt)^2}$ पर मूल्यांकन किया गया $ t = \frac
qp $ (कहाँ $1 - pt = 1 - q = p $) देता है $ G'(q_{\mathrm{ext}})
= \frac qp = \frac1m $: प्रेक्षित अनुपात $ m^{-1}$ बिलकुल है
आकर्षित करने वाले निश्चित बिंदु पर व्युत्पन्न।

\textbf{10.} के लिए $ p = \frac12$: $ G''(t) = \frac{1/4}{(1 -
t/2)^3}$, इसलिए $ G''(1) = 2$ और $\sigma^2 = G''(1) + m - m^2 =
2$. बंद फॉर्म देता है $1 - q_n = \frac1{n+1}$: द
जीवित रहने की \omterm{def:b2:multint:exact}{संभावना} जैसे कम हो जाती है $1/n $ --- बहुत धीरे-धीरे होना
\omterm{def:b2:series:summable}{संक्षेपणीय}, किसी भी सबक्रिटिकल दर के विपरीत।

\textbf{11.} प्रेरण: $ G_1(t) = \frac1{2-t}$ से मेल खाता है
के लिए सूत्र $ n = 1$, और
\[
G(G_n(t)) = \cfrac{1}{2 - \cfrac{n - (n-1)t}{n+1 - nt}}
= \frac{n + 1 - nt}{2(n+1) - 2nt - n + (n-1)t}
= \frac{n+1 - nt}{n + 2 - (n+1)t} .
\]
तब
\[
\frac{G_n(t) - q_n}{1 - q_n}
= (n+1)\,\Bigl(\frac{n - (n-1)t}{n+1 - nt} -
\frac{n}{n+1}\Bigr)
= \frac{t}{n + 1 - nt}
= \frac{\frac{t}{n+1}}{1 - \frac{n}{n+1}t} ,
\]
ज्यामितीय का पीजीएफ \omterm{def:b2:randomvar:law}{कानून} $\mathcal G\bigl(\frac1{n+1}
\bigr)$ पर $\N^*$ (\cref{ex:b2:genfun:classical}): दिया गया
अस्तित्व, $\P(Z_n = k \mid Z_n > 0) =
\frac1{n+1}\bigl(\frac n{n+1}\bigr)^{k-1}$, सशर्त के साथ
मतलब $ n + 1$. बिना शर्त मतलब $1 = \E(Z_n)$ है
लुप्त होती उत्तरजीविता संभाव्यता और रैखिक रूप से उत्पाद
सशर्त आकार बढ़ रहा है।

\textbf{12.} यदि पीढ़ी पीढ़ी में रेखा विलुप्त हो जाती है $ n $, तब
$ Y = Z_0 + \dots + Z_{n-1}$ परिमित है; अगर यह कभी नहीं जाता
विलुप्त, $ Y \geq \sum_n 1 = \infty $. इसलिए $\{Y < \infty\}$ है
विलुप्ति आयोजन और $\P(Y < \infty) = q_{\mathrm{ext}}$.
की व्युत्पत्ति $ H(t) = tG(H(t))$ में
\cref{exo:b2:genfun:7} --- पूर्वज का योगदान होता है
कारक $ t $, इसके बच्चे मिल गए \omterm{def:b2:proba:independence}{स्वतंत्र} की प्रतियाँ $ Y $
के माध्यम से गिना गया $ G $ --- केवल प्रयोग किया जाता है \cref
{thm:b2:genfun:compound}, हर व्यवस्था में मान्य।

\textbf{13.} साथ $ G(s) = \frac{1 + s^2}2$ समीकरण पढ़ता है
$ tH^2 - 2H + t = 0$, इसलिए $ H = \frac{1 - \sqrt{1 - t^2}}{t}$
(जड़ के साथ $ H(0) = 0$). कैटलन श्रृंखला के साथ तुलना
$ C(x) = \frac{1 - \sqrt{1 - 4x}}{2x}$
(\cref{ex:b2:powerseries:catalan}): $ H(t) = \frac
t2\,C\bigl(\frac{t^2}4\bigr) =
\sum_{k\geq0}C_k\,\frac{t^{2k+1}}{2^{2k+1}}$, यानी\ $\P(Y =
2k+1) = C_k2^{-2k-1}$. जाँच: $\P(Y = 1) = C_0/2 = \frac12$
(पूर्वज की कोई संतान नहीं है); $\P(Y = 3) = C_1/8 = \frac18$
(दो बच्चे, दोनों निःसंतान: $\frac12\cdot\frac12\cdot
\frac12$).

\textbf{14.} फर्क $ H = tG(H)$ पर $\intoo01$ और
देना $ t \to 1^-$ (मोनोटोन सीमाएं जैसे कि
\cref{thm:b2:genfun:moments}): $ H'(1)\bigl(1 - G'(H(1))\bigr)
= G(H(1))$. सबक्रिटिकल मामले में $ H(1) = 1$ और $\E(Y) =
H'(1) = \frac1{1 - m}$. गंभीर मामले में $ G'(1) = 1$
बायां कारक गायब हो जाता है जबकि दायां पक्ष गायब हो जाता है $1$: नहीं
परिमित $ H'(1)$ मौजूद हो सकता है, इसलिए $\E(Y) = \infty $ --- अभी तक $\P(Y <
\infty) = q = 1$.

\textbf{15.} $ C_k = \frac1{k+1}\binom{2k}k \sim
\frac{4^k}{\sqrt\pi\,k^{3/2}}$ द्वारा
\cref{ex:b2:comparison:centralbinomial}, इसलिए
\[
\P(Y = 2k+1) = \frac{C_k}{2\cdot4^{k}} \sim
\frac1{2\sqrt\pi\,k^{3/2}} .
\]
पूँछ का योग (तुलना के साथ)। $\int_K^\infty
k^{-3/2}\dd k = 2K^{-1/2}$, ऊपर और नीचे): $\P(Y > 2K)
\asymp K^{-1/2}$, यानी\ $\P(Y > n) \asymp n^{-1/2}$ --- ए
अनंत माध्य के साथ भारी पूँछ, परिमाणीकरण प्रश्न 14।

\textbf{16.} दोनों महत्वपूर्ण वस्तुएँ --- फेयर वॉक की वापसी
समय (सप्ताहांत की समस्या) \cref{ch:b2:proba}) और
महत्वपूर्ण कुल संतानें --- लगभग निश्चित रूप से सीमित हैं
अनंत माध्य, घातांक के स्थानीय नियमों के साथ $-3/2$ और पूँछ
प्रतिपादक का $-1/2$. यह कोई संयोग नहीं है: ए की खोज
वंश-वृक्ष बच्चा-बच्चा पैदा करता है $\pm1$ पथ (एक कदम
प्रति जन्म ऊपर, प्रति मृत्यु एक नीचे) जो \omterm{def:b2:series:def}{बिल्कुल} उचित है
चलना, और $ Y $ प्रथम पारित होने का समय बन जाता है। आलोचनात्मकता का अर्थ है
शून्य बहाव: प्रक्रिया हमेशा दोनों के कगार पर होती है
विलुप्ति और विस्फोट, और $\sqrt{}$-स्केल
शून्य-बहाव यादृच्छिकता के उतार-चढ़ाव ठीक यही उत्पन्न करते हैं
प्रतिपादक.

\textbf{17.} $ G''(t) = \sum_{n\geq2}n(n-1)p_nt^{n-2}$ है
गैर-नकारात्मक शब्द, इसलिए इसमें कोई कमी नहीं हो रही है $\intco01$ के साथ
परिमित सीमा $ G''(1) = \sigma^2$ (महत्वपूर्णता बनाता है
$\E Z_1(Z_1 - 1) = \sigma^2$); के साथ एक गैर-घटता हुआ कार्य
सीमा सीमा मान के बराबर है \omterm{def:b2:metric:continuity}{निरंतर} पर $1$.
टेलर बिंदु पर अभिन्न शेष के साथ $1$:
\[
G(t) = 1 + (t - 1) + \int_1^t(t - s)G''(s)\,\dd s
= t + \frac{G''(1)}2(1-t)^2 + o\bigl((1-t)^2\bigr),
\]
तब से $ G''(s) = G''(1) + o(1)$ जैसा $ s \to 1^-$.

\textbf{18.} एक सामान्य भाजक को कम करना, $ h(t) =
\frac{G(t) - t}{(1 - G(t))(1 - t)}$. प्रश्न 17 द्वारा
अंश है $ b(1-t)^2 + o((1-t)^2)$ और $1 - G(t) = (1 -
t)\bigl(1 - b(1-t) + o(1-t)\bigr)$, इसलिए $ h(t) \to b $.

\textbf{19.} की परिभाषा के अनुसार $ h $ पर $ t = q_j $ और $ G(q_j) =
q_{j+1}$: $\frac1{1 - q_{j+1}} - \frac1{1-q_j} = h(q_j)$;
से संक्षेप $ j = 0$ ($ q_0 = 0$) डिस्प्ले देता है. के बाद से
महत्वपूर्ण प्रक्रिया समाप्त हो जाती है, $ q_j \uparrow 1$, इसलिए $ h(q_j) \to
b $ और सेसरो का मतलब है $\frac1n\sum_{j<n}h(q_j) \to b $:
$\frac1{1-q_n} \sim bn $, यानी
\[
\P(Z_n > 0) \sim \frac1{bn} = \frac{2}{\sigma^2 n} .
\]

\textbf{20.} ज्यामितीय महत्वपूर्ण मामला: $\sigma^2 = 2$
(प्रश्न 10), इसलिए कोलमोगोरोव भविष्यवाणी करता है $1 - q_n \sim \frac1n $
--- और सटीक मान है $\frac1{n+1}$.

\textbf{21.} तब से $ Z_n\mathbf 1_{Z_n > 0} = Z_n $, $\E(Z_n
\mid Z_n > 0) = \frac{\E(Z_n)}{\P(Z_n > 0)} = \frac1{1 -
q_n} \sim \frac{\sigma^2n}2$. ज्यामितीय मामले में यह है
$ n + 1$, प्रश्न 11 से \omterm{def:b2:series:def}{बिल्कुल} मेल खाता हुआ ($\sigma^2 = 2$). द
गंभीर चित्र: विलुप्ति निश्चित है, औसत आकार है
पर जमे हुए $1$, और दुर्लभ जीवित रेखाओं का आकार बढ़ रहा है
रैखिक रूप से --- प्रत्येक कारक दूसरे को संतुलित करता है।

\textbf{22.} के लिए $\mathcal P(\lambda)$ संतान, $ G(t) =
\eu^{\lambda(t-1)}$ और विलुप्त होने की \omterm{def:b2:multint:exact}{संभावना} है
की सबसे छोटी जड़ $ q = \eu^{\lambda(q-1)}$. संख्यात्मक रूप से:
$\lambda = 1.5$ देता है $ q \approx 0.417$ (पुनरावृत्ति $ q \mapsto
\eu^{1.5(q-1)}$: $0, 0.223, 0.312, 0.356, \dots \to 0.4172$);
$\lambda = 2$ देता है $ q \approx 0.203$. तो एक सूचकांक मामला
\omterm{def:b2:multint:exact}{संभावना} के साथ एक बड़े प्रकोप को भड़काता है $58\%$
($\lambda = 1.5$) या $80\%$ ($\lambda = 2$) --- \omterm{def:b2:multint:exact}{संभावना} है, नहीं
निश्चित. से पुनरावृत्ति $ q_0 = 0$ में परिवर्तित हो जाता है
\emph{सबसे छोटा} जड़ क्योंकि $ G $ घट नहीं रहा है: द्वारा
प्रेरण $ q_n \leq r $ किसी निश्चित बिंदु के लिए $ r $, और $(q_n)$
बढ़ता है (है)। $\P(Z_n = 0)$), इसलिए इसकी सीमा निश्चित है
अन्य सभी के नीचे बिंदु।

\textbf{23.}  $ k $ पूर्वजों को मिला \omterm{def:b2:proba:independence}{स्वतंत्र} परिवार
पेड़, और पूर्ण विलुप्ति का प्रतिच्छेदन है $ k $
\omterm{def:b2:proba:independence}{स्वतंत्र} विलुप्त होने \omterm{def:b2:proba:space}{घटनाएँ}: \omterm{def:b2:multint:exact}{संभावना} $ q^k $. के लिए
$\lambda = 1.5$: प्रकोप की \omterm{def:b2:multint:exact}{संभावना} $1 - q^k \geq 0.99$
आवश्यक है $ q^k \leq 0.01$, यानी\ $ k \geq
\frac{\ln 0.01}{\ln 0.417} \approx 5.3$: छह प्रारंभिक मामले
प्रकोप बनाओ $99\%$ निश्चित।

\textbf{24.} \emph{$ G'(q) < 1$:} $ G - \mathrm{id}$ है \omterm{def:b2:affine:convex}{उत्तल}
और गायब हो जाता है $ q $ और $1$, सो है $\leq 0$ पर $\intcc
q1$; अगर $ G'(q) = 1$, पर स्पर्शरेखा $ q $ (कौन सी उत्तलता
नीचे के स्थान $ G $) मजबूर करेगा $ G(t) \geq t $ पर $\intcc q1$,
इस तरह $ G \equiv \mathrm{id}$ वहाँ, सभी गुणांकों को ख़त्म करना
$ p_n $ ($ n \geq 2$) और विरोधाभासी $ m > 1$. \emph{ज्यामितीय
\omterm{def:b2:integration:improper}{अभिसरण}:} $ q_n < q $ सभी के लिए $ n $ (प्रेरण, $ G $
बढ़ रहा है), और माध्य मान प्रमेय देता है $ q - q_{n+1} =
G'(c_n)(q - q_n)$ साथ $ c_n \in \intoo{q_n}q $, इसलिए $ G'(c_n)
\leq G'(q) < 1$ और $ q - q_n \leq q\,G'(q)^n $.
\emph{सहयोगी प्रक्रिया:} $\widehat G(t) = G(qt)/q =
\sum_kp_kq^{k-1}t^k $ गैरऋणात्मक गुणांक है और
$\widehat G(1) = G(q)/q = 1$: एक पीजीएफ; इसका माध्य है $\widehat
G'(1) = G'(q) < 1$: सबक्रिटिकल. ज्यामितीय परिवार: $ G(t) =
\frac{q}{1-pt}$, $ q_{\mathrm{ext}} = \frac qp $, और
\[
\widehat G(t) = \frac pq\cdot\frac{q}{1 - p\frac qp t}
= \frac{p}{1 - qt} :
\]
ज्यामितीय संतान \omterm{def:b2:randomvar:law}{कानून} साथ $ p $ और $ q $ आदान-प्रदान ---
इसके विलुप्त होने पर देखी जाने वाली सुपरक्रिटिकल प्रक्रिया आयोजन है
मिरर सबक्रिटिकल एक।

\textbf{25.} टेबल: $ m < 1$: $ q = 1$, $\P(Z_n > 0)
\asymp m^n $ (प्रश्न 4--5), $\E(Y) = \frac1{1-m}$,
परिबद्ध सशर्त माध्य की पीढ़ियों तक जीवित रहना। $ m = 1$:
$ q = 1$, $\P(Z_n > 0) \sim \frac2{\sigma^2n}$ (कोलमोगोरोव),
$\E(Y) = \infty $ साथ $\P(Y > n) \asymp n^{-1/2}$, बचे हुए लोग
आकार का $\sim \frac{\sigma^2n}2$. $ m > 1$: $ q < 1$ है
सबसे छोटा निश्चित बिंदु, $ q - q_n = O(G'(q)^n)$, $\E(Z_n) = m^n $
विकास, और वातानुकूलित होने पर मरने की प्रक्रिया है
सबक्रिटिकल साथी (प्रश्न 24)। उपकरण: रचना
पीजीएफएस ने जनसंख्या पुनरावृत्ति को फ़ंक्शन पुनरावृत्ति में बदल दिया;
उत्तलता ने निश्चित बिंदुओं की ज्यामिति तय की; टेलर पर
$1^-$ क्षणिक परिकल्पनाओं को स्थानीय विस्तारों में परिवर्तित किया गया; और
सेसरो ने कोलमोगोरोव का औसत निकाला $1/n $ ए से
दूरबीन योग. वर्ष 3 का खंड मार्टिंगेल जोड़ता है $ Z_n/
m^n $ ---जिसकी लगभग-निश्चित सीमा परिष्कृत होती है $\E(Z_n) = m^n$ में
एक प्रक्षेपवक्र-दर-प्रक्षेपवक्र विकास दर --- और याग्लोम
प्रमेय, सीमा \omterm{def:b2:randomvar:law}{कानून} सशर्त ज्यामिति के पीछे
प्रश्न 11 में देखा गया।
\end{solution}
