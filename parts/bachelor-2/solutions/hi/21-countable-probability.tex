\section*{अध्याय \ref{ch:b2:proba} --- गणनीय स्थानों पर संभाव्यता}

\begin{solution}{exo:b2:proba:1}
समरूपता द्वारा: ड्राइंग क्रम एक \omterm{def:b2:funcseq:def}{समान रूप से} यादृच्छिक सापेक्ष को प्रेरित करता है
गेंदों पर ऑर्डर करें $1$ और $2$, इसलिए $\P(1 \text{ before } 2) =
\frac12$. औपचारिक रूप से: गेंदों की स्थिति का आदान-प्रदान $1$ और $2$
एक ड्राइंग अनुक्रम में (समसंभाव्य) का एक आक्षेप है
परिणाम जो स्वैप करते हैं आयोजन इसके पूरक के साथ. गेंदों के बीच
$1, \dots, k $: इनका सापेक्ष क्रम $ k $ गेंदें एक समान हैं
के बीच $ k!$ \omterm{def:b2:structures:generated}{आदेश}, और गेंद $1$ में प्रथम है $(k-1)!$ का
उन्हें: संभाव्यता $\frac{(k-1)!}{k!} = \frac1k $.
\end{solution}

\begin{solution}{exo:b2:proba:2}
$\frac{1}{k(k+1)} = \frac1k - \frac1{k+1}$, इसलिए $\sum_{k\geq1} p_k $
दूरबीनों को $1$: ए संभाव्यता माप. सम परिणाम:
\[
\P(2\N^*) = \sum_{j=1}^{\infty}\frac{1}{2j(2j+1)}
= \sum_{j=1}^{\infty}\Bigl(\frac{1}{2j} - \frac{1}{2j+1}\Bigr)
= \frac12 - \frac13 + \frac14 - \frac15 + \cdots
\]
यह अपने पहले पद के साथ एक प्रत्यावर्ती हार्मोनिक श्रृंखला है
हटा दिया गया और संकेत फ़्लिप कर दिए गए: चूँकि $\ln 2 = 1 - \frac12 + \frac13 -
\frac14 + \cdots $ (\cref{ch:b2:series}),
\[
\P(2\N^*) = -\bigl(\ln 2 - 1\bigr) = 1 - \ln 2 \approx 0.307 .
\]
\end{solution}

\begin{solution}{exo:b2:proba:3}
होने देना $ S $ = बीमार, $+$ = सकारात्मक परीक्षण. बेयस
(\cref{thm:b2:proba:bayes}) विभाजन के साथ $\{S, S^c\}$:
\[
\P(S \mid +)
= \frac{0.99 \times 10^{-4}}
       {0.99 \times 10^{-4} + 0.01 \times 0.9999}
= \frac{0.000099}{0.000099 + 0.009999}
\approx 0.0098 ,
\]
नीचे $1\%$. हालाँकि परीक्षण ``99 है\% सटीक'', एक सकारात्मक
परिणाम आपके बारे में छोड़ देता है $99\%$ स्वस्थ होने की \omterm{def:b2:multint:exact}{संभावना}: मिथ्या
विशाल स्वस्थ्य बहुसंख्यकों के बीच सकारात्मकताएँ सत्य को झुठलाती हैं
छोटे बीमार अल्पसंख्यक से सकारात्मकता। दुर्लभ के लिए स्क्रीनिंग परीक्षण
शर्तों को हमेशा इस आधार-दर के माध्यम से पढ़ा जाना चाहिए
गणना.
\end{solution}

\begin{solution}{exo:b2:proba:4}
\omterm{def:b2:funcseq:def}{बिंदुवार} पर $\Omega $: $\omega \in \bigcup A_i $ अगर कुछ कारक
$1 - \mathbf{1}_{A_i}(\omega)$ गायब हो जाता है, इसलिए
\[
\mathbf{1}_{\bigcup A_i}
= 1 - \prod_{i=1}^n\bigl(1 - \mathbf{1}_{A_i}\bigr)
= \sum_{\emptyset \neq J \subseteq \{1,\dots,n\}}
(-1)^{\abs J + 1}\prod_{i \in J}\mathbf{1}_{A_i} ,
\]
उत्पाद का विस्तार करना और आगे बढ़ाना $1$ आर-पार। अब
$\prod_{i\in J}\mathbf{1}_{A_i} = \mathbf{1}_{\bigcap_{i \in J}
A_i}$, और वज़न के विरुद्ध योग $\P(\{\omega\})$ ---
वैध: प्रत्येक परिवार के लिए सीमित संख्या में कई शर्तें \omterm{def:b2:series:summable}{संक्षेपणीय} ---
प्रत्येक सूचक को उसकी संभाव्यता में बदल देता है आयोजन, देना
सूत्र.
\end{solution}

\begin{solution}{exo:b2:proba:5}
होने देना $ A_i $ = ``पत्र $ i $ सही लिफाफे में है'' के लिए $ J $ का
आकार $ k $, $\P\bigl(\bigcap_{i\in J}A_i\bigr) =
\frac{(n-k)!}{n!}$ (हल करना $ k $ पत्र, बाकी को क्रमबद्ध करें)। द्वारा
समावेश--बहिष्करण,
\[
\P\Bigl(\bigcup A_i\Bigr)
= \sum_{k=1}^n (-1)^{k+1}\binom nk \frac{(n-k)!}{n!}
= \sum_{k=1}^n \frac{(-1)^{k+1}}{k!} ,
\]
इसलिए
\[
\P(\text{no match})
= 1 - \P\Bigl(\bigcup A_i\Bigr)
= \sum_{k=0}^{n}\frac{(-1)^k}{k!}
\xrightarrow[n\to\infty]{} e^{-1} \approx 0.368 .
\]
\omterm{def:b2:series:def}{बिल्कुल} एक मिलान: \omterm{def:b2:series:def}{बिल्कुल} एक निश्चित बिंदु के साथ एक क्रमपरिवर्तन है
निश्चित अक्षर के चयन द्वारा निर्धारित ($ n $ तरीके) और ए
\emph{गड़बड़ी} (नो-मैच व्यवस्था) दूसरे की $ n - 1$;
लिखना $ D_{n-1} = (n-1)!\sum_{k=0}^{n-1}\frac{(-1)^k}{k!}$ के लिए
विक्षोभों की संख्या (पहला भाग, द्वारा बढ़ाया गया)।
$(n-1)!$),
\[
\P(\text{exactly one match})
= \frac{n\,D_{n-1}}{n!}
= \frac{D_{n-1}}{(n-1)!}
= \sum_{k=0}^{n-1}\frac{(-1)^k}{k!}
\xrightarrow[n\to\infty]{} e^{-1} :
\]
सीमा में, ``कोई मेल नहीं'' और ``\omterm{def:b2:series:def}{बिल्कुल} एक मैच'' \omterm{def:b2:funcseq:def}{समान रूप से} हैं
\omterm{def:b2:multint:exact}{संभावना}, प्रत्येक \omterm{def:b2:multint:exact}{संभावना} के साथ $ e^{-1}$.
\end{solution}

\begin{solution}{exo:b2:proba:6}
प्रारंभ पर स्थिति (श्रृंखला नियम/
\cref{thm:b2:proba:bayes}):
\begin{itemize}
\item पहले टॉस टी (\omterm{def:b2:multint:exact}{संभावना}) $1 - p $): खेल पुनः आरंभ होता है
नये सिरे से; से अधिक समय तक चलने वाला $ n $ का अर्थ है अधिक समय तक चलने वाला $ n - 1$ से
वहाँ: योगदान $(1-p)\,q_{n-1}$;
\item पहले HT (\omterm{def:b2:multint:exact}{संभावना}) उछालता है $ p(1-p)$): दो के बाद पुनः आरंभ करें
टॉस: योगदान $ p(1-p)\,q_{n-2}$;
\item पहला टॉस एचएच: खेल समाप्त हो गया है (भीतर)। $ n $ उछालना, $ n
\geq 2$): योगदान देता है $0$.
\end{itemize}
इस तरह $ q_n = (1-p)q_{n-1} + p(1-p)q_{n-2}$. विशेषता
समीकरण $ r^2 = (1-p)r + p(1-p)$ जड़ें हैं
\[
r_\pm = \frac{(1-p) \pm \sqrt{(1-p)^2 + 4p(1-p)}}{2},
\]
साथ $\abs{r_\pm} < 1$: वास्तव में बहुपद $\chi(r) = r^2 -
(1-p)r - p(1-p)$ संतुष्ट $\chi(1) = 1 - (1-p) - p(1-p) = p^2 >
0$ और $\chi(-1) = 1 + (1-p) - p(1-p) > 0$, जबकि $\chi(0) =
-p(1-p) < 0$: एक जड़ में $\intoo{-1}{0}$, एक में $\intoo{0}{1}$.
इसलिए $ q_n = \alpha r_+^n + \beta r_-^n \to 0$. \omterm{def:b2:proba:space}{घटनाएँ} ``खेल
से अधिक रहता है $ n $'' ''खेल कभी ख़त्म नहीं होता'' तक घटा; एक लय
निरंतरता (\cref{thm:b2:proba:continuity}) देता है
$\P(\text{never ends}) = \lim q_n = 0$: खेल लगभग समाप्त हो जाता है
निश्चित रूप से.
\end{solution}

\begin{solution}{exo:b2:proba:7}
\emph{$\P(R_n) = 1/n $:} सबसे पहले के बीच $ n $ प्रत्येक को खींचता है $ n $
अंतिम ड्रा की सापेक्ष स्थिति \omterm{def:b2:funcseq:def}{समान रूप से} संभावित है (एकरूपता)।
सापेक्ष क्रम का), और $ R_n $ है आयोजन कि यह है
सबसे बड़ी: \omterm{def:b2:multint:exact}{संभावना} $1/n $.

\emph{असीमित अनेक रिकार्ड:} $\sum_n \P(R_n) = \sum 1/n =
\infty $ और यह $ R_n $ हैं \omterm{def:b2:proba:independence}{स्वतंत्र} (स्वीकार किया), तो
बोरेल--कैंटेली~2 (\cref{thm:b2:proba:borelcantelli}) देता है
$\P(\limsup R_n) = 1$: रिकॉर्ड कभी नहीं रुकते, लगभग निश्चित रूप से --- लेकिन
वे लघुगणकीय रूप से पतले होते हैं।

\emph{लगातार रिकॉर्ड:} द्वारा स्वतंत्रता,
\[
\sum_n \P(R_n \cap R_{n+1})
= \sum_n \frac{1}{n(n+1)} < \infty ,
\]
इसलिए बोरेल--कैंटेली~1 लागू होता है: लगभग निश्चित रूप से, केवल सीमित रूप से अनेक
टाइम्स एक रिकॉर्ड है जिसके तुरंत बाद दूसरा रिकॉर्ड आता है। दो
लेम्मा के आधे भाग एक साथ काम करते हैं: अनंत रूप से कई रिकॉर्ड, लेकिन
(ए.एस.) अंततः कभी भी एक पंक्ति में दो नहीं होते।
\end{solution}

\begin{solution}{exo:b2:proba:8}
\emph{पहला भाग:} $\sum \P(A_n) = \sum\frac{1}{n+1} = \infty $
साथ स्वतंत्रता: बोरेल--कैंटेली~2 देता है $\P(\limsup A_n) = 1$.
प्रत्येक व्यक्ति $ A_n $ इसकी \omterm{def:b2:multint:exact}{संभावना} कम होती जा रही है, फिर भी लगभग हर
$\omega $ उनमें से अनन्त अनेकों का है।

\emph{बिना प्रति उदाहरण स्वतंत्रता:} लेना $\Omega = \N^*$
वज़न के साथ $ p_k = \frac{1}{k(k+1)}$ का
\cref{exo:b2:proba:2}, और $ A_n = \{k \in \N^* : k \geq n\}$.
तब
\[
\P(A_n) = \sum_{k \geq n}\Bigl(\frac1k - \frac1{k+1}\Bigr)
= \frac1n ,
\qquad
\sum_n \P(A_n) = \infty ,
\]
लेकिन $ A_n $ कम हो रहे हैं, इसलिए $\limsup_n A_n = \bigcap_n A_n =
\emptyset $: $\P(\limsup A_n) = 0$. का विचलन $\sum\P(A_n)$
अकेले कुछ भी गारंटी नहीं देता जब \omterm{def:b2:proba:space}{घटनाएँ} सिकुड़ते हुए ढेर लगना
\omterm{def:b2:diffcalc:differential}{अंतर}िक्ष का हिस्सा --- स्वतंत्रता वही उसे मना करता है
साजिश.
\end{solution}

\begin{solution}{exo:b2:proba:9}
साथ $ q = 1 - p $, पहला सिर रैंक पर पड़ता है $2j + 1$ के साथ
संभाव्यता $ q^{2j}p $, इसलिए
\[
\P(\text{odd rank}) = \sum_{j\geq0}q^{2j}p
= \frac{p}{1 - q^2} = \frac{1}{1 + q} .
\]
एक उचित सिक्के के लिए: $\frac1{1 + 1/2} = \frac23$. (स्वच्छता जांच:
रैंक के बाद से विषम रैंक की \omterm{def:b2:multint:exact}{संभावना} अधिक होनी चाहिए $1$ पहले आता है ---
और वास्तव में $\frac1{1+q} > \frac12$ हमेशा।)
\end{solution}

\begin{solution}{exo:b2:proba:10}
 \omterm{def:b2:proba:space}{घटनाएँ} $ B_N = \bigcap_{n=1}^N A_n^c $ को कम करें
$\bigcap_nA_n^c $, और तक स्वतंत्रता पूरकों का
$\P(B_N) = \prod_{n=1}^N(1 - p_n)$; एक लय निरंतरता
(\cref{thm:b2:proba:continuity}) प्रदर्शित सीमा देता है।
लघुगणक लेते हुए, $\prod(1 - p_n) > 0$ आईएफएफ $\sum-\ln(1 -
p_n) < \infty $. अगर $\sum p_n < \infty $ तब $ p_n \to 0$ और
$-\ln(1 - p_n) \sim p_n $: लॉग-श्रृंखला \omterm{def:b2:integration:improper}{अभिसरण} करती है। अगर
$\sum p_n = \infty $, तब $-\ln(1 - p_n) \geq p_n $ बल
विचलन, तो उत्पाद है $0$. ये मेल खाता है
बोरेल--कैंटेली~2: के लिए $\sum p_n = \infty $, न केवल है
$\P(\text{no }A_n\text{ occurs}) = 0$, लेकिन लगभग निश्चित रूप से
असीम रूप से अनेक $ A_n $ घटित होना।
\end{solution}

\begin{solution}{exo:b2:proba:11}
बक्सा कहो $ A $ वह जो सबसे पहले खाली खोजा गया है, दूसरे के साथ
बॉक्स होल्डिंग $ k $. इसका मतलब है: सबसे पहले में से $2n - k $
पहुंचता है, \omterm{def:b2:series:def}{बिल्कुल} $ n $ को गया $ A $ और $ n - k $ को $ B $ (कुछ में
ऑर्डर), और पहुंच संख्या $2n - k + 1$ को गया $ A $ फिर से,
यह खाली मिल रहा है. पहुँच हैं \omterm{def:b2:proba:independence}{स्वतंत्र} उचित विकल्प,
तो यह आयोजन \omterm{def:b2:multint:exact}{संभावना} है
$\binom{2n-k}{n}2^{-(2n-k)}\cdot\frac12$; दोहरीकरण (खाली
बॉक्स कोई भी हो सकता है) देता है
\[
\P(\text{other box has }k) = \binom{2n-k}{n}\,2^{-(2n-k)} .
\]
के लिए $ n = 1$: $ k = 1$ देता है $\binom11 2^{-1} = \frac12$ और $ k
= 0$ देता है $\binom21 2^{-2} = \frac12$: कुल $1$, जैसा कि होना चाहिए
हो.
\end{solution}

\begin{solution}{exo:b2:proba:12}
(ए) यदि $\P(\{n\}) = c $ सभी के लिए $ n $, $\sigma $-addivity
बल $1 = \sum_nc $: असंभव, चाहे $ c = 0$ (\omterm{def:b2:quadratic:adjoint}{जोड़} $0$)
या $ c > 0$ (योग अनंत). पर कोई एक समान \omterm{def:b2:multint:exact}{संभावना} नहीं है
$\N $.

(बी) यदि $ A \cap B = \emptyset $ और $ d(A)$, $ d(B)$ अस्तित्व में है, तो
$\abs{(A \sqcup B)\cap\intint1n} = \abs{A\cap\intint1n} +
\abs{B\cap\intint1n}$, इसलिए $ d(A \sqcup B) = d(A) + d(B)$:
ऐसे युग्मों पर परिमित संयोजकता। प्रत्येक सिंगलटन की गिनती होती है
कार्य अंततः स्थिर है, इसलिए घनत्व $0$, जबकि
$ d(\N^*) = 1$. थे $ d $ $\sigma $-योजक, $\N^* =
\bigsqcup_k\{k\}$ देगा $1 = \sum_k 0 = 0$: घनत्व है
अंतिम रूप से योगात्मक लेकिन नहीं $\sigma $-योगात्मक --- अभिगृहीत है
सामग्री.

(सी) चलो $ A = \bigcup_{k\geq0}\intint{4^k}{2\cdot4^k - 1}$
(से ब्लॉक करता है $4^k $ को $2\cdot4^k - 1$). पर $ n = 2\cdot4^K -
1$ गिनती है $\sum_{k\leq K}4^k \sim \frac43 4^K $, देना
अनुपात $\to \frac23$; पर $ n = 4^{K+1} - 1$ गिनती है
अपरिवर्तित, अनुपात दे रहा है $\to \frac13$. अनुपात में उतार-चढ़ाव होता है
सीमाओं के बीच $\frac13$ और $\frac23$: कोई घनत्व नहीं.
\end{solution}

\begin{solution}{pb:b2:proba:1}
\textbf{1.} \omterm{def:b2:curves:length}{लंबाई} का एक पथ $ n $ के सेट द्वारा निर्धारित किया जाता है
इसके ऊपर की सीढ़ियाँ; पर समाप्त हो रहा है $ k $ मतलब $ u $ ऊपर-कदम और $ n - u $
के साथ नीचे कदम $ u - (n - u) = k $, यानी\ $ u = \frac{n+k}2$:
संभव है अगर $ n + k $ सम है और $\abs k \leq n $, में
$\binom{n}{(n+k)/2}$ तौर तरीकों। प्रत्येक विशिष्ट पथ का एक बिंदु है
उचित उत्पाद माप पर $ n $ टॉस: संभाव्यता $2^{-n}$.
इस तरह $\P(S_n = k) = N_n(k)2^{-n}$.

\textbf{2.} $ S_n $ की समता है $ n $, इसलिए $ S_{2n+1} \neq
0$; और $ u_n = N_{2n}(0)4^{-n} = \binom{2n}n4^{-n}$. मान:
$ u_1 = \frac12$, $ u_2 = \frac6{16} = \frac38$, $ u_3 =
\frac{20}{64} = \frac5{16}$.

\textbf{3.} $\dfrac{u_n}{u_{n-1}} =
\dfrac{\binom{2n}n}{4\binom{2n-2}{n-1}} =
\dfrac{(2n)(2n-1)}{4n^2} = \dfrac{2n-1}{2n} < 1$: घट रहा है.
द्वारा \cref{ex:b2:comparison:centralbinomial}, $\binom{2n}n \sim
\frac{4^n}{\sqrt{\pi n}}$, इसलिए $ u_n \sim \frac1{\sqrt{\pi n}}
\to 0$, और $\sum u_n $ तुलना से भिन्न होता है $\sum
n^{-1/2}$.

\textbf{4.} से एक रास्ता दिया गया है $1$ को $ k $ छू $0$,
इसके प्रारंभिक खंड को प्रतिबिंबित करें (तक)। \emph{पहला} का दौरा करें
$0$) क्षैतिज अक्ष के माध्यम से: परिणाम एक पथ है
$-1$ को $ k $, और ऑपरेशन एक इनवोल्यूशन है --- प्रत्येक
से पथ $-1$ को $ k \geq 1$ पार करना होगा $0$, और प्रतिबिंबित
इसका प्रारंभिक खंड मूल को पुनः प्राप्त करता है। इसलिए
स्पर्श पथ संख्या $ N_{n-1}(k + 1)$ (से $-1$ को $ k $ द
विस्थापन है $ k + 1$). से एक रास्ता $0$ को $ k $ प्रवास के $>
0$ समय हो चुका $0$ एक कदम ऊपर से शुरू होता है और फिर आगे बढ़ता है
$1$ को $ k $ में $ n - 1$ बिना छुए कदम $0$: वहाँ हैं
$ N_{n-1}(k-1) - N_{n-1}(k+1)$ उनमें से.

\textbf{5.} साथ $ m = \frac{n+k}2$, का उपयोग करना
$\binom{n-1}{m-1} = \frac mn\binom nm $ और $\binom{n-1}{m} =
\frac{n-m}n\binom nm $:
\[
\frac{N_{n-1}(k-1) - N_{n-1}(k+1)}{N_n(k)}
= \frac{\binom{n-1}{m-1} - \binom{n-1}{m}}{\binom nm}
= \frac{m - (n - m)}{n} = \frac kn .
\]
के लिए $ n = 3$, $ k = 1$: $ N_3(1) = 3$ पथ ($++-$, $+-+$,
$-++$), जिनमें से केवल $++-$ सकारात्मक रहता है ($+-+$ को लौटता है
$0$ समय पर $2$): तीन में से एक, और $\frac kn = \frac13$.

\textbf{6.} समरूपता से \omterm{def:b2:multint:exact}{संभावना} है $2\P(S_i > 0\ \forall
i \leq 2n)$. समापनबिंदु पर संक्षेपण $2k $ और उपयोग कर रहे हैं
प्रश्न 4 (साथ) $ n $ द्वारा प्रतिस्थापित $2n $):
\[
\P(S_i > 0\ \forall i) = 2^{-2n}\sum_{k\geq1}
\bigl(N_{2n-1}(2k-1) - N_{2n-1}(2k+1)\bigr)
= 2^{-2n}\,N_{2n-1}(1),
\]
एक दूरबीन योग. अब $ N_{2n-1}(1) = \binom{2n-1}{n}$ और
$2\binom{2n-1}n = \binom{2n}n $ (पास्कल), तो प्रदर्शित
\omterm{def:b2:multint:exact}{संभावना} है $2\cdot2^{-2n}\binom{2n-1}n =
\binom{2n}n4^{-n} = u_n $.

\textbf{7.}  \omterm{def:b2:proba:space}{घटनाएँ} $ D_n = \{S_i \neq 0,\ i \leq 2n\}$
कमी, प्रतिच्छेदन के साथ ``कभी वापसी नहीं''; मोनोटोन द्वारा
निरंतरता और प्रश्न 6, $\P(\text{no return}) = \lim u_n =
0$: चलना लगभग निश्चित रूप से लौटता है। इसके अतिरिक्त $ f_n = \P(D_{n-1})
- \P(D_n) = u_{n-1} - u_n $, और प्रश्न 3 से
\[
u_{n-1} - u_n = u_n\Bigl(\frac{2n}{2n-1} - 1\Bigr) =
\frac{u_n}{2n-1};
\qquad
\sum_{n\geq1}f_n = u_0 - \lim u_n = 1 .
\]

\textbf{8.} $2n\,f_n = \frac{2n}{2n-1}u_n \geq u_n $, और
$\sum u_n = \infty $ (प्रश्न 3): शृंखला $\sum 2nf_n $
विचलन पहला रिटर्न निश्चित है लेकिन इसका कोई सीमित मतलब नहीं है
प्रतीक्षा का समय---चलना है \emph{शून्य आवर्तक}, में
शब्दावली वह \cref{ch:b2:randomvar} प्रदान करेगा।

\textbf{9.}  आयोजन ``कम से कम $ k $ रिटर्न'' है
विच्छेद \omterm{def:b2:structures:countable}{गणनीय} संघ, खत्म $0 < n_1 < \dots < n_k $, की \omterm{def:b2:proba:space}{घटनाएँ} ``पहला $ k $ रिटर्न ठीक समय पर होता है
$2n_1, \dots, 2n_k $''. ऐसा आयोजन का चौराहा है
$ k $ \omterm{def:b2:proba:space}{घटनाएँ} असंयुक्त टॉस ब्लॉकों पर निर्भर करता है
$\intint1{2n_1}$, $\intint{2n_1+1}{2n_2}$, \dots, प्रत्येक ब्लॉक
इसके बाद पहली बार वापसी के लिए नए सिरे से चलने की आवश्यकता है
चरणों की \omterm{def:b2:series:def}{बिल्कुल} आवंटित संख्या; द्वारा स्वतंत्रता का
ब्लॉक इसकी \omterm{def:b2:multint:exact}{संभावना} है $ f_{n_1}f_{n_2-n_1}\cdots
f_{n_k-n_{k-1}}$. पैकेट द्वारा योग (\cref{ch:b2:series},
सभी पद अऋणात्मक):
\[
\P(\text{at least }k\text{ returns})
= \Bigl(\sum_{n\geq1}f_n\Bigr)^{\!k} = 1^k = 1 .
\]
 \omterm{def:b2:proba:space}{घटनाएँ} में कमी $ k $, तो मोनोटोन द्वारा निरंतरता
$\P(\text{infinitely many returns}) = 1$: पुनरावृत्ति.

\textbf{10.} प्रश्न 9 के अनुसार चलना अनन्त रूप से अनेक बनाता है
से दूर भ्रमण $0$. प्रत्येक भ्रमण का पहला चरण है
एक ताजा सिक्का, \omterm{def:b2:proba:independence}{स्वतंत्र} पहले की हर चीज़ का: the
\omterm{def:b2:multint:exact}{संभावना} है कि प्रथम $ m $ सभी भ्रमण नीचे की ओर शुरू होते हैं
है $2^{-m}$. तक पहुँचने $1$ चलने के लिए केवल एक ऊपर की ओर जाने की आवश्यकता है
भ्रमण प्रारंभ (से.) $<0$ इसे गुजरना ही होगा $0$ पहले
पहुँचना $1$, कदम हो रहे हैं $\pm1$), इसलिए $\P(\text{never hit }1)
\leq 2^{-m}$ हरएक के लिए $ m $: वॉक हिट $1$ लगभग निश्चित रूप से.
(लगभग निश्चित रूप से सीमित) हिटिंग समय पर विघटित होना
वॉक फिर से शुरू हो गई है, वहां एक नई वॉक शुरू हो गई है $1$: द्वारा
प्रेरण यह हर हिट करता है $ k \geq 1$ लगभग निश्चित रूप से, और द्वारा
समरूपता हर $ k \leq -1$. अंत में, सबसे पहले पुनः आरंभ करना
का दौरा करें $ k $, प्रश्न 9 ताज़ा सैर पर लागू होता है: प्रत्येक
साइट पर अनगिनत बार, लगभग निश्चित रूप से, विजिट किया जाता है।

\textbf{11.}  \omterm{def:b2:proba:space}{घटनाएँ} $ A_n = \{S_{2n} = 0\}$ से दूर हैं
\omterm{def:b2:proba:independence}{स्वतंत्र} (पर होना $0$ समय पर $2n $ पर होना बनाता है $0$ पर
समय $2n + 2$ की तुलना में कहीं अधिक \omterm{def:b2:multint:exact}{संभावना} है $ u_{n+1}$), तो
बोरेल--कैंटेली~2 अनुपलब्ध है, और वास्तव में संपूर्ण कार्य
भाग II का इसे प्रतिस्थापित करना था। दूसरी दिशा की जरूरत नहीं है
स्वतंत्रता: \emph{अगर} $\sum\P(A_n)$ जुटता है,
बोरेल--कैंटेली~1 लगभग निश्चित रूप से सीमित मात्रा में रिटर्न देता है।
वह निहितार्थ प्रत्येक क्षणभंगुर प्रमाण का इंजन है
नीचे.

\textbf{12.} समय पर वापसी $2n $ आवश्यक है $ n $ ऊपर- और $ n $
नीचे-चरण: $\P(S_{2n} = 0) = \binom{2n}np^nq^n =
u_n(4pq)^n $, और $4pq = 1 - (p - q)^2 < 1$ के लिए $ p \neq
\frac12$. तब से $ u_n \leq 1$, श्रृंखला $\sum\P(S_{2n} = 0)$
ज्यामितीय का प्रभुत्व है $\sum(4pq)^n $: \omterm{def:b2:integration:improper}{अभिसरण}. द्वारा
बोरेल--कैंटेली~1, $\P(S_{2n} = 0 \text{ infinitely often}) =
0$: बहुत सारे रिटर्न, लगभग निश्चित रूप से।

\textbf{13.} के लिए $ n + k $ यहां तक की, $\P(S_n = k) =
\binom{n}{\frac{n+k}2}p^{\frac{n+k}2}q^{\frac{n-k}2}$; द
द्विपद गुणांक अधिक से अधिक केंद्रीय है, और
$ p^{\frac{n+k}2}q^{\frac{n-k}2} = (pq)^{n/2}(p/q)^{k/2}$,
बताई गई बाध्यता दे रहे हैं $\leq
2^n(pq)^{n/2}(p/q)^{k/2} = (4pq)^{n/2}(p/q)^{k/2}$, \omterm{def:b2:series:summable}{संक्षेपणीय}
में $ n $ तब से $\sqrt{4pq} < 1$. बोरेल--कैंटेली~1: साइट $ k $ है
लगभग निश्चित रूप से अक्सर दौरा किया; संघ ख़त्म $ k \in
\Z $ असाधारण शून्य का \omterm{def:b2:proba:space}{घटनाएँ} अभी भी शून्य है (\omterm{def:b2:structures:countable}{गणनीय}
सबएडिटिविटी)। लगभग निश्चित रूप से प्रत्येक साइट पर अंतिम बार विजिट किया जाता है
अक्सर, तो पूर्णांक अनुक्रम $(S_n)$ प्रत्येक सीमा को छोड़ देता है
अच्छे के लिए खिड़की: $\abs{S_n} \to \infty $.

\textbf{14.} $\P(S_n \neq 0,\ 1 \leq n \leq 200) = u_{100} =
\binom{200}{100}4^{-100} \approx \frac1{\sqrt{100\pi}}
\approx 0.056$: इसकी \omterm{def:b2:multint:exact}{संभावना} बीस में से एक से भी अधिक है $200$
निष्पक्ष टॉस कभी नहीं बंधते। क्षय $1/\sqrt{\pi n}$ है
अत्यंत धीमी गति से: टाई की निश्चितता (प्रश्न 7) है
बहुत लंबे टाई-फ्री स्ट्रेच के साथ संगत --- पहली बार
भाग IV की आर्कसीन घटना का स्वाद।

\textbf{15.} पहले चरण पर शर्त. अगर $ X_1 = +1$ तब
$ T_1 = 1$, और $ f_1 = \frac12$ मेल खाता है. अगर $ X_1 = -1$, द
चलना चाहिए से चढ़ना चाहिए $-1$ को $1$; ब्लॉक अपघटन द्वारा,
पर लौट रहा हूँ $0$ समय पर पहली बार $2n $ इस प्रकार विभाजित होता है:
एक कदम नीचे, फिर एक नई सैर शुरू हुई $-1$ प्रथम
पहुँचना $0$ --- \omterm{def:b2:funcseq:def}{समान रूप से} एक ताज़ा सैर, पहली बार पहुँचना
$+1$ --- में $2n - 1$ कदम, या \omterm{def:b2:quadratic:adjoint}{सममित} घटना ऊपर की ओर।
दोनों संकेत \omterm{def:b2:funcseq:def}{समान रूप से} योगदान करते हैं:
\[
f_n = 2\cdot\tfrac12\,\P(T_1 = 2n - 1) = \P(T_1 = 2n-1) .
\]
इस तरह $\P(T_1 < \infty) = \sum f_n = 1$, जबकि $\sum_n(2n -
1)f_n = \sum_n u_n = \infty $ प्रश्न 7 द्वारा: चलना पहुँचता है
$1$ लगभग निश्चित रूप से, अनंत माध्य समय में।

\textbf{16.} PARTITION $\{M_n \geq k\}$ टर्मिनल मान द्वारा
$ S_n = m $. के लिए $ m \geq k $ स्थिति $ M_n \geq k $ है
स्वचालित. के लिए $ m < k $, इसके बाद के पथ को प्रतिबिंबित करें
\emph{पहला} स्तर पर जाएँ $ k $: यह बीच का आक्षेप है
$\{M_n \geq k, S_n = m\}$ और $\{S_n = 2k - m\}$ (हर पथ
पर समाप्त हो रहा है $2k - m > k $ दौरा $ k $; पीछे प्रतिबिंबित करना है
उलटा)। इस तरह
\[
\P(M_n \geq k)
= \sum_{m > k}\P(S_n = m) + \P(S_n = k)
+ \sum_{m < k}\P(S_n = 2k - m)
= 2\P(S_n > k) + \P(S_n = k).
\]

\textbf{17.} सम समय पर $2n $ साथ $ k = 1$: $\P(S_{2n} = 1)
= 0$ और $\P(S_{2n} > 1) = \P(S_{2n} \geq 2)$, इसलिए
\[
\P(M_{2n} \geq 1) = 2\P(S_{2n} \geq 2)
= \P(S_{2n} \geq 2) + \P(S_{2n} \leq -2)
= 1 - u_n .
\]
इस प्रकार $\P(S_i \leq 0\ \forall i \leq 2n) = u_n $: चलना
पहले कभी नेतृत्व नहीं करता $2n $ ठीक उतनी ही बार कदम उठाएं
कभी नहीं जुड़ता (प्रश्न 6) --- दो \omterm{def:b2:series:def}{बिल्कुल} अलग \omterm{def:b2:proba:space}{घटनाएँ},
उसी द्वारा ले जाया गया $ u_n $.

\textbf{18.} $\{L_{2n} = 2k\} = \{S_{2k} = 0\} \cap
\{\text{the walk of the tosses } 2k+1, \dots, 2n \text{ has
no zero}\}$. दो \omterm{def:b2:proba:space}{घटनाएँ} असंयुक्त टॉस ब्लॉकों पर निर्भर,
तो वे हैं \omterm{def:b2:proba:independence}{स्वतंत्र}; पहले की \omterm{def:b2:multint:exact}{संभावना} है $ u_k $, दूसरा $ u_{n-k}$ प्रश्न 6 द्वारा नये सिरे से लागू किया गया
$(2n-2k)$-कदम चलना. इस तरह $\P(L_{2n} = 2k) = u_ku_{n-k}$.
तब से $ L_{2n}$ \omterm{def:b2:series:def}{बिल्कुल} मान लेता है $0, 2, \dots, 2n $,
इन संभावनाओं का योग है $1$:
$\sum_{k=0}^nu_ku_{n-k} = 1$, एक द्विपद पहचान प्रदान की गई
संभाव्य विभाजन द्वारा.

\textbf{19.} समरूपता तत्काल है: $ u_ku_{n-k} =
u_{n-k}u_k $. जैसा $ u_j $ में घट जाती है $ j $, उत्पाद
$ u_ku_{n-k}$ केंद्रीय के लिए सबसे छोटा है $ k $ और सबसे बड़ा
चरम $ k \in \{0, n\}$, जहां यह बराबर है $ u_n $;
मात्रात्मक $ u_ku_{n-k} \approx \frac1{\pi\sqrt{k(n-k)}}$
थोक में, विरुद्ध $ u_n \approx \frac1{\sqrt{\pi n}}$ पर
किनारों. के लिए $ n = 5$: $\P(L_{10} = 0) = \P(L_{10} = 10) =
u_5 = \frac{63}{256} \approx 0.246$, जबकि $\P(L_{10} = 4) =
u_2u_3 = \frac38\cdot\frac5{16} = \frac{15}{128} \approx
0.117$. एक लंबे निष्पक्ष खेल में अंतिम बराबरी सबसे अधिक होती है
संभवतः शुरुआत या अंत के करीब: एक खिलाड़ी
आम तौर पर बिना किसी पूर्वाग्रह के, विशाल विस्तार के लिए नेतृत्व करता है
सिक्का.

एक की टाई $2n $-स्टेप गेम अत्यधिक मूल्यों के साथ फैला हुआ है
सबसे अधिक \omterm{def:b2:multint:exact}{संभावना} है (प्रश्न 18--19), और कभी भी नेतृत्व न करने की क्षमता है
वही धीरे-धीरे ख़त्म होने वाली \omterm{def:b2:multint:exact}{संभावना} $ u_n $ कभी न बांधने वाले के रूप में
(प्रश्न 17). सीमा में निश्चितता, हर जगह दृढ़ता
परिमित क्षितिज: यही उचित चलना है।

\textbf{21.} PARTITION $\{S_{2n} = 0\}$ ($ n \geq 1$) द्वारा
पहली वापसी का समय $2k $, $1 \leq k \leq n $: का पहला ब्लॉक
$2k $ टॉसेस को पहले रिटर्न का एहसास होता है, बाकी को $2n - 2k $
टॉस से एक नई चाल की वापसी का एहसास होता है, और ब्लॉक होते हैं
\omterm{def:b2:proba:independence}{स्वतंत्र}: $ u_n = \sum_{k=1}^nf_ku_{n-k}$. दोनों श्रृंखला
$ U(x) = \sum u_nx^n $, $ F(x) = \sum f_nx^n $ त्रिज्या है $\geq
1$ (गुणांक में $\intcc01$), और यह \omterm{thm:b2:series:fubini}{कॉची उत्पाद}
(\cref{ch:b2:powerseries}) देता है, के लिए $0 \leq x < 1$,
\[
U(x) - 1 = \sum_{n\geq1}\Bigl(\sum_{k=1}^n
f_ku_{n-k}\Bigr)x^n = F(x)\,U(x),
\qquad\text{i.e.}\qquad
U(x)\bigl(1 - F(x)\bigr) = 1 .
\]

\textbf{22.} जैसा $ x \uparrow 1$, $ U(x)$ और $ F(x)$ वृद्धि
(गैरनकारात्मक गुणांक); प्रत्येक आंशिक योग $\sum_{n\leq
N}u_n $ की एक सीमा है $\sum_{n\leq N}u_nx^n \leq U(x)$, इसलिए
$ U(x) \uparrow \sum u_n \in \intoc0{+\infty}$, और इसी तरह
$ F(x) \uparrow f = \sum f_n $. अगर $\sum u_n = \infty $: $1 -
F(x) = 1/U(x) \to 0$, इसलिए $ f = 1$. अगर $\sum u_n = S < \infty $:
$1 - f = 1/S > 0$, इसलिए $ f < 1$. जाँच: निष्पक्ष चलना, $\sum u_n =
\infty $ और $ f = 1$ (प्रश्न 3, 7); पक्षपातपूर्ण चाल, $\sum
u_n(4pq)^n < \infty $ और तदनुसार $ f = 1 -
1/\sum_{n\geq0}u_n(4pq)^n < 1$, के अनुरूप
रिटर्न की संख्या की लगभग निश्चित परिमितता (प्रश्न)।
12).

\textbf{23.} चार चरणों के लिए $(\pm1, 0), (0, \pm1)$ की
$\Z^2$ चलना, की वृद्धि $ U = X + Y $ और $ V = X - Y $
हैं: $(+,+)$ के लिए $(1,0)$, $(+,-)$ के लिए $(0,1)$, $(-,+)$ के लिए
$(0,-1)$, $(-,-)$ के लिए $(-1,0)$ --- चिह्नों की प्रत्येक \omterm{def:b2:quadratic:adjoint}{जोड़}ी के साथ
संभाव्यता $\frac14 = \frac12\cdot\frac12$: दो
समन्वित चालें $(U_n)$ और $(V_n)$ हैं \omterm{def:b2:proba:independence}{स्वतंत्र} निष्पक्ष
चलता रहता है $\Z $. तब से $ S^{(2)}_{2n} = (0,0)$ आईएफएफ $ U_{2n} = 0$
और $ V_{2n} = 0$,
\[
\P\bigl(S^{(2)}_{2n} = (0,0)\bigr) = u_n^2 \sim \frac1{\pi
n}, \qquad \sum_nu_n^2 = \infty .
\]
प्रश्न 21 की नवीनीकरण पहचान और द्वंद्व
प्रश्न 22 में किसी भी एक-आयामी (केवल) का उपयोग नहीं किया गया
पहले रिटर्न और डिसजॉइंट-ब्लॉक पर विघटन
स्वतंत्रता), इसलिए $\sum u_n^{(2)} = \infty $ देता है $ f^{(2)} =
1$, और प्रश्न 9 का तर्क इसे उन्नत करता है: आगे बढ़ें
$\Z^2$ मूल की ओर अनंत बार, लगभग निश्चित रूप से लौटता है।

\textbf{24.} भर्ती हुए बाध्य के साथ $\P(S^{(3)}_{2n} = 0)
\leq Cn^{-3/2}$, श्रृंखला \omterm{def:b2:integration:improper}{अभिसरण} करती है, और बोरेल--कैंटेली~1
लगभग निश्चित रूप से बहुत सारे रिटर्न देता है: चलते रहो
$\Z^3$ क्षणिक है (और वही घातांक के साथ बंधा हुआ है $-d/2$
हर संभालता है $ d \geq 3$). कुल मिलाकर: \emph{पी\'ओलेया का
प्रमेय} --- द \omterm{pb:b2:proba:1}{सरल यादृच्छिक चलना} पर आवर्ती है $\Z $ और
$\Z^2$, क्षणिक पर $\Z^d $ के लिए $ d \geq 3$. एक शराबी आदमी पाता है
उसका घर का रास्ता; एक नशे में धुत्त पक्षी ऐसा नहीं कर सकता।

\textbf{25.} पथ गणना और प्रतिबिंब ने सटीक उत्पादन किया
कानून ($ u_n $, मतपत्र प्रमेय, $ f_n $, अधिकतम, अंतिम
शून्य); एक लय निरंतरता हर सीमित को परिवर्तित किया
कथन (``कम से कम एक बार लौटाता है'', ``अनंत बार'')
परिमित-क्षितिज संभावनाओं की सीमा में;
असंयुक्त-ब्लॉक स्वतंत्रता नवीनीकरण को संचालित किया
विघटन (प्रश्न 9, 18, 21) --- यह है \omterm{def:b2:structures:countable}{गणनीय}
मार्कोव संपत्ति का कंकाल; बोरेल--कैंटेली~1 था
क्षणभंगुर हथियार (प्रश्न 12-13, 24), आवश्यकता नहीं
स्वतंत्रता; नवीनीकरण पहचान ने सब कुछ व्यवस्थित कर दिया
द्विभाजन $\sum u_n = \infty \iff $ पुनरावृत्ति. द सिंगल
\omterm{def:b2:powerseries:analytic}{विश्लेषणात्मक} इनपुट स्थानीय अनुमान है $ u_n \sim 1/\sqrt{\pi
n}$: यह वर्गाकार है $1/(\pi n)$ अभी भी विचलन (आयाम) $2$,
आवर्ती), जबकि $ n^{-3/2}$ \omterm{def:b2:integration:improper}{अभिसरण} (आयाम) $3$,
क्षणिक) --- पी\'ओलेया का प्रमेय, अंततः, एक कथन है
के विचलन के बारे में $\sum n^{-d/2}$.
\end{solution}
