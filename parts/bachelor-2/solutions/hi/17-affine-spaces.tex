\section*{अध्याय \ref{ch:b2:affine} --- एफ़िन स्पेस}

\begin{solution}{exo:b2:affine:1}
$\{x + y + z = 1\}$: \omterm{def:b2:affine:subspace}{एफ़िन} समतल, दिशा सदिश तल $\{x
+ y + z = 0\}$, आयाम $2$. \omterm{def:b2:quadratic:adjoint}{जोड़}ा जा रहा है $ x - z = 3$: एक \omterm{def:b2:affine:subspace}{एफ़िन} लाइन
(दो स्वतंत्र समीकरण), दिशा $\{x + y + z = 0,\ x = z\}
= \operatorname{Vect}\bigl((1, -2, 1)\bigr)$, आयाम $1$. $\{x^2
+ y^2 = 1\}$: एक सिलेंडर --- नीचे स्थिर नहीं है बैरीसेंटर (द
का मध्यबिंदु $(1,0,0)$ और $(-1,0,0)$ मूल है, से बाहर
सिलेंडर): नहीं \omterm{def:b2:affine:subspace}{एफ़िन}. एक सुसंगत प्रणाली $ MX = B $: \omterm{def:b2:affine:subspace}{एफ़िन}
उपस्पेस $ X_0 + \ker M $ आयाम का $\dim\ker M $, जैसा कि याद किया गया
\cref{def:b2:affine:subspace}.
\end{solution}

\begin{solution}{exo:b2:affine:3}
$ M $ पर प्रक्षेपण मैट्रिक्स है $\operatorname{Vect}(1,1)$ साथ में
$(1,-1)$ (जाँच करना $ M^2 = M $). की छवि $ f $: $\{MX + C\} = C +
\operatorname{im} M $: द \omterm{def:b2:affine:subspace}{एफ़िन} के माध्यम से रेखा $(1,0)$ निर्देशक
$(1,1)$. निश्चित बिंदु: $ X = MX + C $, यानी\ $(I - M)X = C $; लेकिन $ C =
(1, 0)^{\mathsf T}$ और $\operatorname{im}(I - M) =
\operatorname{Vect}(1,-1)$; है $(1,0)$ इस में? $(1, 0) =
\alpha(1,-1)$ बल $\alpha = 1$ और $0 = -1$: नहीं। कोई निश्चित नहीं
अंक. और
\[
f(f(X)) = M(MX + C) + C = MX + MC + C = f(X) + MC,
\qquad MC = \tfrac12(1,1)^{\mathsf T} :
\]
$ f\circ f $ है $ f $ इसके बाद छवि रेखा के साथ अनुवाद किया जाएगा
--- $ f $ एक ``ग्लाइड प्रोजेक्शन'' है: लाइन पर प्रक्षेपण
एक स्लाइड के साथ बनाया गया।
\end{solution}

\begin{solution}{exo:b2:affine:5}
तय करना $ g(u) = f(O + u) - f(O)$ (इसमें काम कर रहे हैं $\R^n $ पर वेक्टरीकृत
$ O $), $ g(0) = 0$.

\emph{संयोजकता:} $\frac{(O + u) + (O + v)}{2} = O + \frac{u +
v}{2}$, इसलिए मध्यबिंदु संरक्षण देता है $ g\bigl(\frac{u+v}{2}\bigr)
= \frac{g(u) + g(v)}{2}$; साथ $ v = 0$: $ g(u/2) = g(u)/2$;
संयोजन, $ g(u + v) = 2g\bigl(\frac{u+v}{2}\bigr) = g(u) + g(v)$.

\emph{$\Q $-एकरूपता:} एडिटिविटी देता है $ g(nu) = ng(u)$ ($ n \in
\N $, प्रेरण), फिर $ g(-u) = -g(u)$ (\omterm{def:b2:quadratic:adjoint}{जोड़}ें), फिर $ g(\frac pq u) =
\frac pq g(u)$ (आवेदन करना $ q $, स्केलिंग की इंजेक्टिविटी का उपयोग करें)।

\emph{$\R $-एकरूपता:} के लिए $ t \in \R $, तर्कसंगत लें $ t_n \to
t $: $ g(t_nu) = t_ng(u)$, और निरंतरता का $ g $ (विरासत में मिला $ f $)
सीमा तक गुजरता है: $ g(tu) = tg(u)$. इस तरह $ g $ रैखिक है और $ f =
f(O) + g $: \omterm{def:b2:affine:subspace}{एफ़िन}.
\end{solution}

\begin{solution}{exo:b2:affine:6}

$ y = x $ (ऑर्थोगोनल, सिद्ध $-1$). निश्चित बिंदु: $(x, y) = (y

\end{solution}

\begin{solution}{exo:b2:affine:7}
 $ n + 1$ वैक्टर $\vect{A_1A_i}$ ($ i \geq 2$) में \omterm{def:b2:metric:connected}{जुड़े हुए} हैं
आयाम $ n $: वहाँ हैं $\mu_i $, सभी शून्य नहीं, साथ $\sum_{i\geq2}
\mu_i\vect{A_1A_i} = 0$; तय करना $\mu_1 = -\sum_{i \geq 2}\mu_i $, इसलिए
$\sum_{i}\mu_i = 0$ और $\sum_i \mu_i\,\vect{OA_i} = 0$ हरएक के लिए
$ O $, सबके साथ नहीं $\mu_i $ शून्य। विभाजित सूचकांक: $ P = \{i : \mu_i >
0\}$, $ N = \{i : \mu_i < 0\}$, दोनों गैर-रिक्त ( $\mu_i $ का योग
शून्य और सभी शून्य नहीं हैं)। साथ $ s = \sum_{i\in P}\mu_i =
-\sum_{i \in N}\mu_i > 0$:
\[
\operatorname{bar}\bigl(A_i, \tfrac{\mu_i}{s}\bigr)_{i \in P}
= \operatorname{bar}\bigl(A_i, \tfrac{-\mu_i}{s}\bigr)_{i \in N},
\]
(दोनों पक्षों की बात बराबर है $ X $ साथ $\vect{OX} = \frac1s\sum_{i\in
P}\mu_i\vect{OA_i}$, संबंध द्वारा): दोनों का एक सामान्य बिंदु
\omterm{def:b2:affine:convex}{उत्तल} पतवारें, असम्बद्ध सूचकांक समूहों के साथ।
\end{solution}

\begin{solution}{exo:b2:affine:8}
\emph{छवि = निश्चित बिंदु:} के लिए $ Y = f(X)$, $ f(Y) = f(f(X)) =
f(X) = Y $: प्रत्येक छवि बिंदु निश्चित है; इसके विपरीत निश्चित बिंदु हैं
छवियाँ। इसलिए $\mathcal{F} = \operatorname{im} f =
\operatorname{Fix}(f)$ गैर-रिक्त है, और यह एक है \omterm{def:b2:affine:subspace}{एफ़िन} उपस्थान
(एक की छवि \omterm{def:b2:affine:subspace}{एफ़िन} मानचित्र), दिशा के साथ $\operatorname{im}\vec f $.

\emph{प्रक्षेपण संरचना:} $\vec f $ निष्क्रिय है ($\vec{f\circ
f} = \vec f^{\,2} = \vec f $), इसलिए $ E = \operatorname{im}\vec f
\oplus \ker \vec f $ (\cref{ex:b2:reduction:projections}). किसी के लिए
बिंदु $ X $, वेक्टर पर विचार करें $\vect{f(X)\,X}$; आवेदन $\vec f $:
\[
\vec f\bigl(\vect{f(X)\,X}\bigr) = \vect{f(f(X))\,f(X)} = 0
\qquad (f \circ f = f),
\]
इसलिए $\vect{f(X)\,X} \in \ker\vec f $. इस तरह $ X = f(X) +
\vect{f(X)X}$ प्रदर्शित करता है $ X $ के एक बिंदु के रूप में $\mathcal{F}$ अनुवादित
के एक वेक्टर द्वारा $\ker\vec f $: $ f $ वास्तव में प्रक्षेपण है
$\mathcal{F}$ साथ में $\ker\vec f $. इसके विपरीत ऐसे अनुमान
स्पष्ट रूप से संतुष्ट करें $ f \circ f = f $.
\end{solution}

\begin{solution}{exo:b2:affine:10}
किसी मूल पर सदिशीकरण करें $ O $ और बिंदुओं को सदिश के रूप में लिखें:
$ h_{\Omega, \lambda}(x) = \omega + \lambda(x - \omega)$ साथ
$\omega = \vect{O\Omega}$. रचना $ g = h_{\Omega',
\mu} \circ h_{\Omega, \lambda}$ है \omterm{def:b2:affine:subspace}{एफ़िन} रैखिक भाग के साथ
$\mu\lambda\,\mathrm{id}$. अगर $\lambda\mu \neq 1$: $1 \notin
\operatorname{Sp}(\lambda\mu\,\mathrm{id})$, इसलिए
\cref{prop:b2:affine:fixedpoint} एक अद्वितीय निश्चित बिंदु उत्पन्न करता है
$\Omega''$ और, वहां वेक्टरकृत, $ g =
\lambda\mu\,\mathrm{id}$: समरूपता $ h_{\Omega'',
\lambda\mu}$. अगर $\lambda\mu = 1$ रैखिक भाग है
पहचान, तो $ g $ एक अनुवाद है; विस्तार,
\[
g(x) = \omega' + \mu\bigl(\omega + \lambda(x - \omega) -
\omega'\bigr) = x + (1 - \mu)\,\omega' + \mu(1 -
\lambda)\,\omega .
\]
के लिए $\lambda = \mu = -1$ (बिंदु प्रतिबिंब) वेक्टर है
$2\omega' - 2\omega = 2\,\vect{\Omega\Omega'}$: द
में बिंदु प्रतिबिंब की संरचना $\Omega $ तब
$\Omega'$ द्वारा अनुवाद है $2\,\vect{\Omega\Omega'}$.
\end{solution}

\begin{solution}{exo:b2:affine:11}
$\vect{A'B} = \alpha\,\vect{A'C}$ बिलकुल कहता है $1\cdot
\vect{A'B} - \alpha\,\vect{A'C} = 0$, यानी\ $ A' =
\operatorname{bar}(B, 1;\ C, -\alpha)$ (कुल वजन $1 -
\alpha \neq 0$ तब से $ B \neq C $); वैसे ही $ B' =
\operatorname{bar}(C, 1;\ A, -\beta)$ और $ C' =
\operatorname{bar}(A, 1;\ B, -\gamma)$.

\emph{संरेखण मानदंड.} प्रत्येक बिंदु दीजिए $ P $ इसका
सामान्यीकृत बैरीसेंट्रिक पंक्ति $ p = (p_A, p_B, p_C)$, $ p_A + p_B +
p_C = 1$, इसके संबंध में $(A, B, C)$. अगर $\sum_i c_i p_i = 0$
साथ $(c_1, c_2, c_3) \neq 0$ तीन अंक के लिए $ P_1, P_2,
P_3$, फिर प्रविष्टियों का योग देता है $\sum c_i = 0$, और
$\sum_i c_i \vect{OP_i} = \sum_j \bigl(\sum_i
c_ip_{ij}\bigr)\vect{OV_j} = 0$: द $ P_i $ आत्मीयता से हैं
निर्भर, यानी\ संरेखित. इसके विपरीत ए \omterm{def:b2:affine:subspace}{एफ़िन} निर्भरता
$(t_i)$ देता है $ w = \sum t_ip_i $ प्रविष्टियों के योग के साथ $0$
और $\sum_j w_j\vect{OV_j} = 0$; से विस्तार हो रहा है $ A $, $ w_B
\vect{AB} + w_C\vect{AC} = 0$, इसलिए $ w = 0$ द्वारा \omterm{def:b2:affine:subspace}{एफ़िन}
की स्वतंत्रता $(A, B, C)$: पंक्तियाँ रैखिक रूप से निर्भर हैं।
इसलिए संरेखण लुप्त होने के समान है $3 \times 3$ सिद्ध,
और गैर-शून्य कारकों द्वारा पंक्तियों को स्केल करना $1 - \alpha $, $1 -
\beta $, $1 - \gamma $ कुछ भी नहीं बदलता:
\[
\det\begin{pmatrix} 0 & 1 & -\alpha\\ -\beta & 0 & 1\\ 1 &
-\gamma & 0\end{pmatrix}
= 1 - \alpha\beta\gamma .
\]
इस तरह $ A', B', C'$ संरेखित हैं यदि $\alpha\beta\gamma = 1$:
मेनेलौस का प्रमेय.
\end{solution}

\begin{solution}{exo:b2:affine:12}
होने देना $\Delta = \{\lambda \in \R^{n+1} : \lambda_i \geq 0,\
\sum\lambda_i = 1\}$: बंद और अंदर घिरा हुआ $\R^{n+1}$, इस तरह
\omterm{def:b2:metric:compact}{कॉम्पैक्ट}, और $ K^{n+1}$ है \omterm{def:b2:metric:compact}{कॉम्पैक्ट} एक सीमित उत्पाद के रूप में. वो नक्शा
\[
\Phi \colon \Delta \times K^{n+1} \to \R^n, \qquad
\Phi(\lambda, x_0, \dots, x_n) = \sum_{i=0}^n \lambda_i x_i
\]
है \omterm{def:b2:metric:continuity}{निरंतर}, और \cref{thm:b2:affine:caratheodory} कहते हैं
\omterm{def:b2:series:def}{बिल्कुल} वही $\operatorname{conv}(K) = \Phi(\Delta \times
K^{n+1})$: ए \omterm{def:b2:metric:continuity}{निरंतर} एक की छवि \omterm{def:b2:metric:compact}{कॉम्पैक्ट} तय करना
(\cref{thm:b2:metric:compactprops}), इस तरह \omterm{def:b2:metric:compact}{कॉम्पैक्ट}.

एक बंद सेट के लिए: ले लो $ S = (\R \times \{0\}) \cup \{(0,
1)\}$, बंद किया $\R^2$. ए \omterm{def:b2:affine:convex}{उत्तल} संयोजन वजन डाल रहा है
$ t $ पर $(0,1)$ और $1 - t $ अक्ष बिंदुओं पर दूसरा है
समन्वय करें $ t $, इसलिए
\[
\operatorname{conv}(S) = \bigl(\R \times \intco01\bigr) \cup
\{(0,1)\}
\]
(के लिए $0 \leq t < 1$, $(x, t) = t\,(0,1) + (1-t)\,\bigl(\tfrac
x{1-t}, 0\bigr)$). बात $(1, 1) = \lim_{t \to 1}(1, t)$
अनुवर्ती है लेकिन पतवार में नहीं: बंद नहीं है।
\end{solution}

\begin{solution}{pb:b2:affine:1}
\textbf{1.} पर पुनः आधार बनाएं $ A_j $: के लिए $ i \neq j $, $\vect{A_jA_i} =
\vect{A_0A_i} - \vect{A_0A_j}$. अगर $\sum_{i \neq j}
c_i\vect{A_jA_i} = 0$, विस्तार देता है $\sum_{i \notin \{0,
j\}} c_i\,\vect{A_0A_i} - \bigl(\sum_{i\neq j}
c_i\bigr)\vect{A_0A_j} = 0$; की स्वतंत्रता
$\vect{A_0A_i}$ बल $ c_i = 0$ के लिए $ i \notin \{0, j\}$, तब
$ c_0 = 0$: स्वतंत्रता पर $ A_j $. समतुल्यता के लिए: दो
वजन परिवार $(\lambda_i)$, $(\lambda_i')$ संक्षेप में $1$
उसी के साथ \omterm{def:b2:affine:barycenter}{केन्द्रक} देना, साथ $\nu = \lambda -
\lambda'$: $\sum\nu_i = 0$ और (उत्पत्ति $ A_0$) $\sum_{i \geq
1}\nu_i\,\vect{A_0A_i} = 0$, इसलिए $\nu = 0$ स्वतंत्रता के तहत.
इसके विपरीत एक गैरतुच्छ संबंध $\sum_{i\geq1}\mu_i
\vect{A_0A_i} = 0$, द्वारा पूर्ण की गयी $\mu_0 = -\sum_{i\geq1}
\mu_i $, आइए एक \omterm{def:b2:quadratic:adjoint}{जोड़}ें $ t(\mu_i)$ बिना किसी वजन वाले परिवार के लिए
को हिलाना \omterm{def:b2:affine:barycenter}{केन्द्रक}: गैर-विशिष्टता.

\textbf{2.} $(\vect{AB}, \vect{AC})$ का आधार है $\R^2$:
लिखना $\vect{AM} = \beta\,\vect{AB} + \gamma\,\vect{AC}$
(अद्वितीय) और सेट $\alpha = 1 - \beta - \gamma $; \omterm{def:b2:affine:barycenter}{केन्द्रक}
मूल स्थिति $ A $ \omterm{def:b2:series:def}{बिल्कुल} पढ़ता है $\vect{AM} =
\beta\,\vect{AB} + \gamma\,\vect{AC}$. विशिष्टता प्रश्न है
1.

\textbf{3.} से $\alpha\vect{MA} + \beta\vect{MB} +
\gamma\vect{MC} = 0$ और चैसल्स, $\vect{MA} =
-\beta\,\vect{AB} - \gamma\,\vect{AC}$. साथ $ D =
\det(\vect{AB}, \vect{AC})$:
\begin{align*}
\det(\vect{MB}, \vect{MC})
&= \det(\vect{MA} + \vect{AB},\ \vect{MA} + \vect{AC})\\
&= \det(\vect{MA}, \vect{AC}) + \det(\vect{AB}, \vect{MA}) +
D = -\beta D - \gamma D + D = \alpha D,
\end{align*}
द्विरेखीयता का उपयोग करना और $\det(\vect{MA}, \vect{AC}) = -\beta D $,
$\det(\vect{AB}, \vect{MA}) = -\gamma D $. बाकी दो
सूत्र भूमिकाओं के साथ समान गणना का अनुसरण करते हैं
\omterm{def:b2:structures:generated}{चक्रीय} रूप से क्रमबद्ध।

\textbf{4.} परिभाषा से $\operatorname{conv}\{A, B, C\}$ है
का सेट बैरीसेंटर अऋणात्मक भार के साथ; सामान्यीकरण
योग करने के लिए वजन $1$ और विशिष्टता का आह्वान (प्रश्न 2),
$ M \in \operatorname{conv}\{A,B,C\}$ आईएफएफ $\alpha, \beta,
\gamma \geq 0$. प्रत्येक निर्देशांक एक है \omterm{def:b2:affine:subspace}{एफ़िन} का कार्य $ M $
(प्रश्न 3: ए $2\times2$ सिद्ध एक कॉलम के साथ \omterm{def:b2:affine:subspace}{एफ़िन}
में $ M $), तो प्रत्येक \omterm{def:b2:metric:topology}{खुला} साइन कंडीशन एक को परिभाषित करती है \omterm{def:b2:metric:topology}{खुला}
आधा समतल. नमूना $(-,-,-)$ के विपरीत है $\alpha + \beta
+ \gamma = 1$; शेष सात पैटर्न में से प्रत्येक है
एहसास हुआ: कम से कम एक के साथ एक संकेत-सम्मान करने वाले ट्रिपल को स्केल करें
$+$ प्रविष्टि ताकि (सकारात्मक) योग हो $1$ --- जैसे\
$(-1, 1, 1)$, $(3, -1, -1)$, $(\frac13, \frac13, \frac13)$,
और क्रमपरिवर्तन.

\textbf{5.} एक \omterm{def:b2:affine:subspace}{एफ़िन} मानचित्र बरकरार रखता है बैरीसेंटर
(\cref{def:b2:affine:subspace}), इसलिए $ u(M) = \alpha u(A) +
\beta u(B) + \gamma u(C)$. लिखना $ u(x, y) = ax + by + c $
साथ $(a, b) \neq (0,0)$: $\{u = c'\}$ एक पंक्ति है, और प्रत्येक
लाइन $ ax + by = c'$ एक ऐसा लेवल सेट है. अगर $ u(M), u(N) \geq
c $ और $ t \in \intcc01$, तब $ u\bigl(\operatorname{bar}(M,
1-t; N, t)\bigr) = (1-t)u(M) + tu(N) \geq c $: अर्ध-तल हैं
\omterm{def:b2:affine:convex}{उत्तल}.

\textbf{6.} शर्तें $\sum\mu_i = 0$, $3\mu_2 + \mu_4 =
0$, $3\mu_3 + \mu_4 = 0$ देना (लेना) $\mu_4 = 3$) द
निर्भरता $(\mu_1, \mu_2, \mu_3, \mu_4) = (-1, -1, -1, 3)$.
चिन्ह इस प्रकार विभाजित हैं $\{A_4\} \mid \{A_1, A_2, A_3\}$, और
प्रत्येक पक्ष को सामान्य बनाना $3$:
\[
A_4 = \operatorname{bar}\bigl(A_1, \tfrac13;\ A_2, \tfrac13;\
A_3, \tfrac13\bigr) = (1,1),
\]
त्रिभुज का \omterm{def:b2:affine:barycenter}{केन्द्रक}: रेडॉन बिंदु है $ A_4$
स्वयं, जो वास्तव में त्रिभुज के अंदर स्थित है $ A_1A_2A_3$.

\textbf{7.} रेखीय मानचित्र $\Phi \colon \R^{n+2} \to \R
\times \R^n $, $\mu \mapsto (\sum\mu_i,\
\sum\mu_i\vect{OA_i})$, रैंक है $\leq n + 1$, इसलिए $\dim\ker
\Phi \geq 1$. यदि दो स्वतंत्र निर्भरताएँ $\mu, \mu'$
अस्तित्व में, एक उपयुक्त संयोजन $\nu = \mu'_{n+2}\mu -
\mu_{n+2}\mu'$ (या $\mu $ स्वयं यदि दोनों अंतिम गुणांक हों
गायब हो जाना) एक होगा \emph{अशून्य} के साथ निर्भरता $\nu_{n+2}
= 0$; तक सीमित करना $ A_1, \dots, A_{n+1}$ और पुनः आधार बना रहा हूँ
$ A_1$, कुछ $\nu_i \neq 0$ साथ $ i \geq 2$ (एक एकल अशून्य
वजन का योग शून्य नहीं हो सकता), एक गैर-तुच्छ संबंध देता है
$\sum_{i\geq2}\nu_i\vect{A_1A_i} = 0$: द $ n+1$ अंक होंगे
सामान्य स्थिति के विपरीत, आत्मीयता से आश्रित रहें। इसलिए $\dim\ker
\Phi = 1$. वही प्रतिबंध तर्क एक गैर-शून्य दिखाता है
निर्भरता का कोई लुप्त होने वाला गुणांक नहीं है।

\textbf{8.} होने देना $\mu \neq 0$ एक निर्भरता हो, $ P = \{i :
\mu_i > 0\}$ और $ N = \{i : \mu_i < 0\}$: दोनों गैर-रिक्त
($\sum\mu_i = 0$, $\mu \neq 0$) और संपूर्ण (कोई शून्य नहीं
गुणांक)। रेडॉन का निर्माण (\cref{exo:b2:affine:7})
ठीक इसी विभाजन से सामान्य पतवार बिंदु उत्पन्न होता है।
चूँकि निर्भरता शून्येतर अदिश तक अद्वितीय होती है
(प्रश्न 7), अव्यवस्थित युग्म $\{P, N\}$ --- इसलिए
रैडॉन विभाजन --- अद्वितीय है.

\textbf{9.} ब्लॉक गैर-रिक्त हैं, इसलिए प्रकार है $(1,3)$ या
$(2,2)$. प्रकार $(1,3)$, अवरोध पैदा करना $\{j\}$: रेडॉन बिंदु स्थित है
$\operatorname{conv}\{A_j\} = \{A_j\}$, इसलिए $ A_j \in
\operatorname{conv}$ अन्य तीन में से; यह एक पर झूठ नहीं बोल सकता
बढ़त (तीन बिंदुओं को सामान्य के विपरीत संरेखित किया जाएगा
स्थिति), तो $ A_j $ त्रिभुज का आंतरिक भाग है. प्रकार
$(2,2)$, ब्लॉक $\{i,j\} \mid \{k,l\}$: रेडॉन बिंदु $ z $
पर स्थित है $\intcc{A_i}{A_j} \cap \intcc{A_k}{A_l}$, और $ z $ है
कोई समापन बिंदु नहीं (जो तीन बिंदुओं को संरेखित करेगा): दो
खंड एक आंतरिक बिंदु पर प्रतिच्छेद करते हैं। इसके अलावा कोई मतलब झूठ नहीं है
दूसरों के पतवार में: ऐसी रोकथाम $ A_l =
\operatorname{bar}(A_i, \lambda_i)_{i \neq l}$ साथ
$\lambda_i \geq 0$ एक \omterm{def:b2:affine:subspace}{एफ़िन} साइन पैटर्न के साथ निर्भरता
$(+,+,+,-)$, जो विशिष्टता से (प्रश्न 8) बना देगा
विभाजन $(1,3)$. तो में $(2,2)$ मामले में चार बिंदु हैं
में \omterm{def:b2:affine:convex}{उत्तल} स्थिति और क्रॉसिंग खंड हैं
विकर्ण.

\textbf{10.} समीकरण $\mu_2 + \mu_3 = 0$, $\mu_3 +
\mu_4 = 0$, $\sum\mu_i = 0$ निर्भरता दो $(1, -1, 1,
-1)$: विभाजन $\{(0,0), (1,1)\} \mid \{(1,0), (0,1)\}$, और
\[
\operatorname{bar}\bigl((0,0), \tfrac12;\ (1,1),
\tfrac12\bigr) = \bigl(\tfrac12, \tfrac12\bigr) =
\operatorname{bar}\bigl((1,0), \tfrac12;\ (0,1),
\tfrac12\bigr) :
\]
रेडॉन बिंदु वर्ग का केंद्र है, जहां दोनों हैं

\textbf{11.} रैडॉन ने आवेदन किया $ x_1, \dots, x_4$ ब्लॉक देता है
$ I \mid J $ और एक बिंदु $ z \in \operatorname{conv}\{x_i : i
\in I\} \cap \operatorname{conv}\{x_j : j \in J\}$. हल करना $ k
\in \{1, \dots, 4\}$, कहना $ k \in I $. प्रत्येक $ j \in J $
संतुष्ट $ j \neq k $, इसलिए $ x_j \in C_k $ पसंद से $ x_j \in
\bigcap_{l \neq j}C_l $; तब से $ C_k $ है \omterm{def:b2:affine:convex}{उत्तल}, $ z \in
\operatorname{conv}\{x_j : j \in J\} \subseteq C_k $. जैसा $ k $
मनमाना था, $ z \in C_1 \cap C_2 \cap C_3 \cap C_4$.

\textbf{12.} प्रेरण चालू $ m $. के लिए $ m = 3$ परिकल्पना है
निष्कर्ष; $ m = 4$ प्रश्न 11 है. चलो $ m \geq 4$,
के लिए कथन मानें $ m $ सेट, और ले लो $ C_1, \dots,
C_{m+1}$ ट्रिपल-इंटरसेक्शन संपत्ति के साथ। तय करना $ C_m' =
C_m \cap C_{m+1}$, \omterm{def:b2:affine:convex}{उत्तल}. परिवार $ C_1, \dots, C_{m-1},
C_m'$ है $ m $ सदस्य; एक तिगुना परहेज $ C_m'$ द्वारा प्रतिच्छेद करता है
परिकल्पना, और एक त्रिगुण $\{C_i, C_j, C_m'\}$ है
चौराहा $ C_i \cap C_j \cap C_m \cap C_{m+1}$, गैररिक्त
प्रश्न 11 द्वारा लागू किया गया $ C_i, C_j, C_m, C_{m+1}$ (कोई तीन
इनमें से परिकल्पना द्वारा मिलते हैं)। अब प्रेरण परिकल्पना
नए परिवार का एक सामान्य बिंदु उत्पन्न होता है, अर्थात।\ के सभी $ m+1$
सेट.

\textbf{13.} (ए) बंद किनारे $\intcc PQ $, $\intcc QR $,
$\intcc RP $ एक अविक्षिप्त त्रिभुज का: कोई भी दो शेयर a
शीर्ष, लेकिन तीनों का एक उभयनिष्ठ बिंदु स्थित होगा $\intcc
PQ \cap \intcc RP = \{P\}$ और में $\intcc QR $, जो शामिल नहीं है
$ P $. (बी) कोई तीन सेट $ S_i = \{x_1, \dots, x_4\}
\setminus \{x_i\}$ चार बिंदुओं में से तीन को छोड़ते हुए छोड़ें
\omterm{def:b2:series:def}{बिल्कुल} एक सामान्य बिंदु; कुल प्रतिच्छेदन प्रत्येक को छोड़ देता है
बिंदु. $ S_i $ परिमित हैं, नहीं \omterm{def:b2:affine:convex}{उत्तल}: उत्तलता है
आवश्यक. (सी) बहुत सारे $ H_k = \intco{k}{+\infty} \times
\R $ में प्रतिच्छेद करें $\intco{k_{\max}}{+\infty} \times \R \neq
\emptyset $, अभी तक कोई मतलब नहीं है $ x \geq k $ सभी के लिए $ k \in \N $:
अनंत परिवारों के लिए, सघनता जरूरी है।

\textbf{14.} कल्पना करना $\bigcap_{i \in I}K_i = \emptyset $ और
ठीक करो $ i_0$. प्रत्येक $ x \in K_{i_0}$ कुछ याद आता है $ K_i $, इसलिए
$ K_{i_0} \subseteq \bigcup_{i \in I}(\R^2 \setminus K_i)$, ए
द्वारा कवर करें \omterm{def:b2:metric:topology}{खुला} सेट ($ K_i $ है \omterm{def:b2:metric:compact}{कॉम्पैक्ट}, इसलिए बंद)। द्वारा
बोरेल--लेब्सग्यू (\cref{thm:b2:metric:borellebesgue}) अंतिम रूप से
बहुत सारे पर्याप्त: $ K_{i_0} \cap K_{i_1} \cap \dots \cap K_{i_N} =
\emptyset $. लेकिन इस सीमित परिवार के कोई भी तीन सदस्य
\omterm{def:b2:affine:convex}{उत्तल} सेट प्रतिच्छेद, इसलिए प्रश्न 12 प्रतिच्छेद बनाता है
गैर-रिक्त: विरोधाभास.

\textbf{15.} तय करना $ D_p = \overline D(p, r)$ के लिए $ p \in S $:
\omterm{def:b2:metric:compact}{कॉम्पैक्ट} \omterm{def:b2:affine:convex}{उत्तल} सेट. के लिए $ p, q, s \in S $, परिकल्पना
एक बंद डिस्क देता है $\overline D(z, r)$ युक्त $ p, q, s $;
तब $\norm{\vect{zp}}, \norm{\vect{zq}}, \norm{\vect{zs}}
\leq r $, यानी\ $ z \in D_p \cap D_q \cap D_s $. हेली द्वारा
(प्रश्न 12; परिवार सीमित है) है $ c \in
\bigcap_{p \in S}D_p $: प्रत्येक $ p \in S $ संतुष्ट
$\norm{\vect{cp}} \leq r $, इसलिए $ S \subseteq \overline D(c,
r)$.

\textbf{16.} होने देना $ c = x_{\lceil n/2\rceil}$. एक बंद
अर्ध-पंक्ति युक्त $ c $ है $\intoc{-\infty}{t}$ साथ $ t \geq
c $ या $\intco{t}{+\infty}$ साथ $ t \leq c $. पहला
शामिल है $ x_1, \dots, x_{\lceil n/2\rceil}$: कम से कम
$\lceil n/2\rceil \geq n/2$ अंक. दूसरे में शामिल है
$ x_{\lceil n/2\rceil}, \dots, x_n $: \omterm{def:b2:series:def}{बिल्कुल} $ n - \lceil
n/2\rceil + 1 = \floor{n/2} + 1 > n/2$ अंक.

\textbf{17.} $\abs{A \cap B} = \abs A + \abs B - \abs{A \cup
B} \geq \abs A + \abs B - n > \tfrac{4n}3 - n = \tfrac n3$,
तब
\[
\abs{A \cap B \cap C} \geq \abs{A \cap B} + \abs C - n >
\tfrac n3 + \tfrac{2n}3 - n = 0 .
\]

\textbf{18.} टिप्पणी $ m > 2n/3$. के लिए $ T_1, T_2, T_3 \subseteq S $
प्रमुखता का $ m $, प्रश्न 17 एक बिंदु प्रदान करता है $ x \in T_1
\cap T_2 \cap T_3$; तब $ x \in \operatorname{conv}(T_i)$ प्रत्येक के लिए $ i $: के कोई तीन सदस्य $\mathcal F $ मिलो। परिवार
परिमित है (अंतिम रूप से अनेक उपसमुच्चय)। $ S $) और इसमें शामिल हैं
\omterm{def:b2:affine:convex}{उत्तल} सेट, तो हेली (प्रश्न 12) देती है $ c \in
\bigcap_{\abs T = m}\operatorname{conv}(T)$.

\textbf{19.} मान लीजिए कोई बंद आधा तल $ H \ni c $
से कम शामिल है $ n/3$ के अंक $ S $. यह उसका पूरक है $ U $
एक \omterm{def:b2:metric:topology}{खुला} आधा समतल, \omterm{def:b2:affine:convex}{उत्तल}, साथ $\abs{S \cap U} > 2n/3$,
इस तरह $\abs{S \cap U} \geq \floor{2n/3} + 1 = m $; चुनना $ T
\subseteq S \cap U $ साथ $\abs T = m $. तब
$\operatorname{conv}(T) \subseteq U $ की उत्तलता से $ U $, इसलिए
$ c \in \operatorname{conv}(T) \subseteq U $:विरोधाभास
के साथ $ c \in H $. इसलिए प्रत्येक बंद अर्ध-तल युक्त $ c $
कम से कम शामिल है $ n/3$ अंक: $ c $ एक केंद्रबिंदु है.

\textbf{20.} त्रिभुज की भुजा समबाहु लीजिए $ L $ और
$\varepsilon = L/100$. होने देना $ c $ कोई भी बिंदु हो; हम एक प्रदर्शन करते हैं
बंद आधा विमान युक्त $ c $ और अधिक से अधिक $ k $ अंक.

\emph{केस 1: $ c $ भीतर है $ L/10$ एक शिखर का, कहो $ B $.}
से दिशानिर्देश $ c $ को $ A $ और को $ C $ से विचलन
दिशानिर्देश $ B \to A $, $ B \to C $ अधिक से अधिक द्वारा
$\arcsin\bigl(\tfrac{L/10}{9L/10}\bigr) = \arcsin\tfrac19$,
तो वे एक कोण बनाते हैं $\leq \tfrac\pi3 + 2\arcsin\tfrac19 <
\tfrac{2\pi}3$. \emph{केस 2: $ c $ की दूरी पर है $> L/10$
सभी शीर्षों से.} अगर $ c $ त्रिकोण में है, तीन
दिशाओं के बीच कोणीय \omterm{def:b2:diffcalc:differential}{अंतर}ाल $ u_A, u_B, u_C $ से $ c $
शीर्षों को योग करने के लिए $2\pi $, तो कुछ गैप है $\leq 2\pi/3$;
अगर $ c $ बाहर है, तीनों दिशाएं एक में हैं \omterm{def:b2:metric:topology}{खुला}
दिशाओं का आधा तल और उनमें से दो एक कोण बनाते हैं $<
\pi/2$. प्रत्येक स्थिति में दो दिशाएँ, मान लीजिए की ओर $ X $ और $ Y $,
एक कोण बनाओ $\leq 2\pi/3$; होने देना $ w $ उनकी इकाई समद्विभाजक हो,
तो $\langle u_X, w\rangle, \langle u_Y, w\rangle \geq
\cos\tfrac\pi3 = \tfrac12$. किसी भी बिंदु के लिए $ b $ डिस्क का
चारों ओर $ X $:
\[
\langle \vect{cb}, w\rangle \geq
\tfrac12\norm{\vect{cX}} - \varepsilon > 0,
\]
तब से $\norm{\vect{cX}} \geq L/10 > 2\varepsilon $ (और
इसी तरह के लिए $ Y $): द \omterm{def:b2:metric:topology}{खुला} आधा समतल $\{\langle \vect{cx},
w\rangle > 0\}$ दोनों समूहों को निगल जाता है। यह बंद है
पूरक में शामिल है $ c $ और अधिक से अधिक $ k $ के अंक
तीसरा क्लस्टर. तो विमान के धड़कने का कोई मतलब नहीं $ n/3$: के साथ
प्रश्न 19, केंद्रबिंदु स्थिरांक \omterm{def:b2:series:def}{बिल्कुल} है $1/3$.

\textbf{21.} कोणों को व्यवस्थित करें; सबसे वृहद, $\theta $,
संतुष्ट करता है $\theta \geq \pi/3$ (तीनों का योग है $\pi $). अगर
$\theta \geq \pi/2$, पर कहो $ R $, होने देना $ M $ का मध्यबिंदु हो
विपरीत पक्ष $\intcc PQ $. माध्यिका सूत्र ($\vect{RM}
= \tfrac12(\vect{RP} + \vect{RQ})$, विस्तार करें और समाप्त करें
$\langle\vect{RP}, \vect{RQ}\rangle $ के कानून के साथ
कोसाइन) देता है
\[
\norm{\vect{RM}}^2 = \tfrac12\norm{\vect{RP}}^2 +
\tfrac12\norm{\vect{RQ}}^2 - \tfrac14\norm{\vect{PQ}}^2 \leq
\tfrac12\norm{\vect{PQ}}^2 - \tfrac14\norm{\vect{PQ}}^2 =
\tfrac14\norm{\vect{PQ}}^2,
\]
का उपयोग करते हुए $\norm{\vect{PQ}}^2 = \norm{\vect{RP}}^2 +
\norm{\vect{RQ}}^2 - 2\langle\vect{RP}, \vect{RQ}\rangle \geq
\norm{\vect{RP}}^2 + \norm{\vect{RQ}}^2$ (आंतरिक उत्पाद
है $\leq 0$). तो व्यास की डिस्क $\intcc PQ $, त्रिज्या का
$\leq \tfrac12 < \tfrac1{\sqrt3}$, सभी तीन बिंदु शामिल हैं
(विकृत संरेखित त्रिक नीचे आते हैं $\theta = \pi $). अगर
$\theta < \pi/2$ त्रिभुज तीव्र है; ज्या के नियम द्वारा
परिधि है $ R_c = a/(2\sin\theta)$ साथ $ a \leq 1$
विपरीत पक्ष $\theta $, और $\theta \in
\intco{\pi/3}{\pi/2}$ देता है $\sin\theta \geq \sqrt3/2$, इसलिए
$ R_c \leq 1/\sqrt3$: सर्कमडिस्क काम करता है।

\textbf{22.} के लिए $ p \in S $ होने देना $ K_p = \overline D(p,
1/\sqrt3)$: \omterm{def:b2:metric:compact}{कॉम्पैक्ट} \omterm{def:b2:affine:convex}{उत्तल}. कोई तीन बिंदु $ p, q, s $ का $ S $
\omterm{def:b2:quadratic:adjoint}{जोड़}ीवार दूरी पर हैं $\leq 1$, तो प्रश्न 21 एक देता है
त्रिज्या की डिस्क $1/\sqrt3$ उन्हें समाहित करते हुए: इसका केंद्र स्थित है
$ K_p \cap K_q \cap K_s $. द्वारा \omterm{def:b2:metric:compact}{कॉम्पैक्ट} हेली (प्रश्न 14,
मनमाने ढंग से परिवारों की अनुमति है) वहाँ है $ c \in \bigcap_{p\in
S}K_p $: प्रत्येक $ p \in S $ भीतर है $1/\sqrt3$ का $ c $, यानी\ $ S
\subseteq \overline D(c, 1/\sqrt3)$. यह जंग का प्रमेय है
विमान में.

\textbf{23.} साथ $ G $ \omterm{def:b2:affine:barycenter}{केन्द्रक}, $\sum_i\vect{GA_i} = 0$,
इसलिए
\[
\sum_i\norm{\vect{OA_i}}^2 = \sum_i\norm{\vect{OG} +
\vect{GA_i}}^2 = 3\norm{\vect{OG}}^2 + 2\Bigl\langle
\vect{OG}, \sum_i\vect{GA_i}\Bigr\rangle +
\sum_i\norm{\vect{GA_i}}^2,
\]
और मध्य पद लुप्त हो जाता है: लीबनिज पहचान। के लिए
भुजा का समबाहु त्रिभुज $1$, $\norm{\vect{GA_i}} =
1/\sqrt3$ (ऊंचाई का दो तिहाई $\sqrt3/2$), इसलिए
$\sum_i\norm{\vect{GA_i}}^2 = 1$. अगर $\overline D(O, r)$
फिर शीर्ष शामिल हैं $3r^2 \geq
\sum_i\norm{\vect{OA_i}}^2 = 3\norm{\vect{OG}}^2 + 1 \geq 1$:
$ r \geq 1/\sqrt3$, समानता के बल के साथ $ O = G $ और सब
तीन दूरियाँ बराबर $ r $ --- सर्कमडिस्क. जंग का
स्थिर $1/\sqrt3$ तेज़ है.

\textbf{24.} \emph{हेली इन $\R^n $: अगर $ C_1, \dots, C_m $
($ m \geq n + 1$) हैं \omterm{def:b2:affine:convex}{उत्तल} के उपसमुच्चय $\R^n $ और कोई भी $ n + 1$
उनमें से एक दूसरे को काटते हैं, फिर वे सभी एक दूसरे को काटते हैं।} बेस केस $ m = n +
2$: चुनना $ x_i \in \bigcap_{j\neq i}C_j $; रैडॉन की लेम्मा
(\cref{exo:b2:affine:7}) विभाजन $ x_1, \dots, x_{n+2}$ में
ब्लॉक $ I \mid J $ एक सामान्य पतवार बिंदु के साथ $ z $, और प्रत्येक के लिए
$ k $, ब्लॉक में शामिल नहीं है $ k $ के बिंदुओं से मिलकर बनता है
$ C_k $, इसलिए $ z \in C_k $ उत्तलता से, \omterm{def:b2:series:def}{बिल्कुल} जैसा प्रश्न में है
11. प्रेरण कदम के लिए $ m \geq n + 2$: प्रतिस्थापित करें $ C_m,
C_{m+1}$ द्वारा $ C_m \cap C_{m+1}$; एक $(n+1)$-नए का टुपल
प्रतिच्छेदित सदस्य वाले परिवार की राशि होती है $ n + 2$
पुराने सेटों का, बेस केस द्वारा संभाला गया, और अन्य
टुपल्स परिकल्पना द्वारा कवर किए गए हैं। प्रेरण द्वारा समापन करें
परिकल्पना.

\textbf{25.} रैखिक बीजगणित एक बार प्रवेश करता है: $ n + 2$ वेक्टर में
द $(n+1)$ \omterm{def:b2:quadratic:adjoint}{जोड़}े का आयामी स्थान (कुल वजन,
भारित स्थिति) पर निर्भर होना चाहिए --- यही है \omterm{def:b2:affine:subspace}{एफ़िन}
निर्भरता. इसके गुणांकों के चिह्न बिंदुओं को विभाजित करते हैं
दो रेडॉन ब्लॉकों में और एक रैखिक संबंध में बदल दें
दो गैर-नकारात्मक की समानता बैरीसेंटर. उत्तलता का प्रयोग किया जाता है
ठीक दो बार: हेली के चरण में (अंकों का पतवार)। $C_k $
में रहता है $ C_k $) और अनुप्रयोगों में (आधा-तल और
डिस्क हैं \omterm{def:b2:affine:convex}{उत्तल}). विमान का आयाम ही दर्ज किया गया
संख्या के माध्यम से $4 = 2 + 2$ रैडॉन को खिलाए गए अंकों की संख्या, यानी\
``$3 = 2 + 1$ हेली की परिकल्पना में ''; बाकी सब कुछ
आयाम-मुक्त था, जैसा कि प्रश्न 24 पुष्टि करता है। में $\R^n $ द
स्थिरांक बन जाते हैं: हेली संख्या $ n + 1$; केंद्रबिंदु स्थिरांक
$\frac1{n+1}$ (प्रत्येक परिमित समुच्चय में प्रत्येक बंद बिंदु पर एक बिंदु होता है
आधा स्थान जिसके माध्यम से एक अंश होता है $\geq
\frac1{n+1}$ इसका); जंग त्रिज्या
$\sqrt{\frac{n}{2(n+1)}}$ व्यास के सेट के लिए $1$ --- के बराबर $1/\sqrt3$ कब $ n = 2$.
\end{solution}
