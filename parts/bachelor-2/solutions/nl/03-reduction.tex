\section*{Hoofdstuk \ref{ch:b2:reduction} --- Reductie van endomorfismen}

\begin{solution}{exo:b2:reduction:1}
$A$: $\chi_A = X^2 - 2X - 3 = (X - 3)(X + 1)$. \omterm{def:b2:reduction:eigen}{Eigenvectoren}: voor
$3$: $(1,1)$; voor $-1$: $(1,-1)$. Dus $P = \begin{pmatrix} 1 & 1\\ 1
& -1\end{pmatrix}$ geeft $P^{-1}AP = \operatorname{diag}(3, -1)$.

$B = J - I$ waarbij $J$ de all-eens-matrix is. $J$ heeft rang $1$
met $Jv = 3v$ voor $v = (1,1,1)$ en $Jw = 0$ op het vlak $x + y +
z = 0$: \omterm{def:b2:reduction:eigen}{spectrum} van $B$ is $\{2, -1\}$ met \omterm{def:b2:reduction:eigen}{eigenruimten}
$\operatorname{Vect}(1,1,1)$ (dimensie $1$) en $\{x + y + z =
0\}$ (dimensie $2$, basis $(1,-1,0), (1,0,-1)$). $P$ met deze
drie kolommen geeft $P^{-1}BP = \operatorname{diag}(2, -1, -1)$.
\end{solution}

\begin{solution}{exo:b2:reduction:2}
\emph{\omterm{def:b2:reduction:eigen}{Eigenruimten}:} $\chi_C = (X-1)^2$, enkele \omterm{def:b2:reduction:eigen}{eigenwaarde} $1$;
$\ker(C - I) = \ker\begin{pmatrix} 0&1\\ 0&0\end{pmatrix}$ is de
lijn $\operatorname{Vect}(e_1)$: dimensie $1 < 2 = m_1$, dus niet
\omterm{def:b2:reduction:diag}{diagonaliseerbaar} (\cref{thm:b2:reduction:diagcrit}).

\emph{\omterm{def:b2:reduction:polyu}{Minimale veelterm}:} $\mu_C$ deelt $(X-1)^2$ en $C \neq I$,
dus $\mu_C = (X-1)^2$: een dubbele wortel, dus niet
\omterm{def:b2:reduction:diag}{diagonaliseerbaar} (\cref{cor:b2:reduction:minpolycrit}).
\end{solution}

\begin{solution}{exo:b2:reduction:3}
$X^2 - 5X + 6 = (X-2)(X-3)$: gesplitst met enkelvoudige wortels,
dus is $u$ \omterm{def:b2:reduction:diag}{diagonaliseerbaar}
(\cref{cor:b2:reduction:minpolycrit}), met
$\operatorname{Sp}(u) \subseteq \{2, 3\}$. Mogelijke spectra:
$\{2\}$ ($u = 2\,\mathrm{id}$), $\{3\}$ ($u = 3\,\mathrm{id}$),
of $\{2, 3\}$.

Machten: zoek $u^k = a_k\,\mathrm{id} + b_k\,u$. Op de
\omterm{def:b2:reduction:eigen}{eigenruimten} luidt dit $2^k = a_k + 2b_k$ en $3^k = a_k +
3b_k$: oplossend, $b_k = 3^k - 2^k$, $a_k = 3\cdot2^k - 2\cdot
3^k$:
\[
u^k = (3\cdot 2^k - 2\cdot 3^k)\,\mathrm{id} + (3^k - 2^k)\, u .
\]
(Geldig voor alle drie de spectra: de identiteiten gelden
eigenwaarde-gewijs.)
\end{solution}

\begin{solution}{exo:b2:reduction:4}
$u$ \omterm{def:b2:reduction:diag}{diagonaliseerbaar}: $P = \prod_{\lambda}(X - \lambda)$ over
het \omterm{def:b2:reduction:eigen}{spectrum} annuleert $u$, splitst, enkelvoudige wortels. Dan
$P(u|_F) = P(u)|_F = 0$: de restrictie wordt geannuleerd door
een gesplitste veelterm met enkelvoudige wortels, dus
\omterm{def:b2:reduction:diag}{diagonaliseerbaar} (\cref{cor:b2:reduction:minpolycrit}).
\end{solution}

\begin{solution}{exo:b2:reduction:5}
$\chi_A = X^2 - X - 1$, wortels $\varphi$ en $\psi$
(verschillend): \omterm{def:b2:reduction:diag}{diagonaliseerbaar}, met \omterm{def:b2:reduction:eigen}{eigenvectoren} $(\varphi,
1)$ en $(\psi, 1)$. De recurrentie geeft $\begin{pmatrix}
F_{n+1}\\ F_n \end{pmatrix} = A^n \begin{pmatrix}1\\
0\end{pmatrix}$. Ontbind $(1, 0)$ op de \omterm{def:b2:reduction:eigen}{eigenvectoren}: $(1,0) =
\frac{1}{\varphi - \psi}\bigl((\varphi, 1) - (\psi, 1)\bigr)$
met $\varphi - \psi = \sqrt5$. $A^n$ toepassen vermenigvuldigt
elke eigencomponent met de $n$-de macht van haar \omterm{def:b2:reduction:eigen}{eigenwaarde};
de tweede coördinaat lezend:
\[
F_n = \frac{\varphi^n - \psi^n}{\sqrt 5} .
\]
(Controle: $n = 1$ geeft $\frac{\varphi - \psi}{\sqrt5} = 1$.)
\end{solution}

\begin{solution}{exo:b2:reduction:6}
Zij $P = \prod_i (X - \mu_i)$ $u^2$ annuleert, gesplitst met
enkelvoudige wortels $\mu_i$ (het \omterm{def:b2:reduction:eigen}{spectrum} van $u^2$). Omdat $u$
inverteerbaar is, is $0$ geen \omterm{def:b2:reduction:eigen}{eigenwaarde} van $u^2$ ($\det u^2 =
(\det u)^2 \neq 0$), dus alle $\mu_i \neq 0$. Dan
\[
Q(X) = \prod_i (X^2 - \mu_i) = \prod_i (X - \sqrt{\mu_i})(X +
\sqrt{\mu_i})
\]
annuleert $u$: $\;Q(u) = \prod_i (u^2 - \mu_i\,\mathrm{id}) =
P(u^2) = 0$. Haar wortels $\pm \sqrt{\mu_i}$ (complexe
vierkantswortels) zijn paarsgewijs verschillend omdat de $\mu_i$
verschillend en niet-nul zijn ($\sqrt{\mu_i} = -\sqrt{\mu_j}$ zou
$\mu_i = \mu_j$ geven). Gesplitst + enkelvoudige wortels: $u$ is
\omterm{def:b2:reduction:diag}{diagonaliseerbaar}.

Tegenvoorbeeld zonder inverteerbaarheid: $u = \begin{pmatrix} 0
& 1\\ 0 & 0\end{pmatrix}$: $u^2 = 0$ is \omterm{def:b2:reduction:diag}{diagonaliseerbaar}, $u$
is het niet.
\end{solution}

\begin{solution}{exo:b2:reduction:7}
$A = 2I + N$ met $N$ de shift ($N e_2 = e_1$, $Ne_3 = e_2$),
$N^3 = 0$, $N^2 = E_{13}$: dit \emph{is} de \omterm{thm:b2:reduction:dunford}{Dunford-ontbinding}
($2I$ diagonaal, $N$ nilpotent, ze commuten; uniciteit maakt
haar de unieke). Binomiaal met commuterende termen:
\[
A^k = 2^k I + k 2^{k-1} N + \binom k2 2^{k-2} N^2
= \begin{pmatrix}
2^k & k2^{k-1} & \binom k2 2^{k-2}\\
0 & 2^k & k2^{k-1}\\
0 & 0 & 2^k
\end{pmatrix},
\]
\[
\eu^{tA} = \eu^{2t}\Bigl(I + tN + \frac{t^2}{2}N^2\Bigr)
= \eu^{2t}\begin{pmatrix}
1 & t & t^2/2\\
0 & 1 & t\\
0 & 0 & 1
\end{pmatrix}.
\]
\end{solution}

\begin{solution}{exo:b2:reduction:8}
$X^k - 1$ annuleert $A$ en splitst over $\C$ met de $k$
verschillende wortels $\eu^{2\iu\pi j/k}$: $A$ is
\omterm{def:b2:reduction:diag}{diagonaliseerbaar} (\cref{cor:b2:reduction:minpolycrit}) en haar
\omterm{def:b2:reduction:eigen}{eigenwaarden}, wortels van $X^k - 1$, zijn $k$-de
eenheidswortels.

Als bovendien $A$ gelijkvormig is met een driehoekige matrix met
eenheidsdiagonaal: alle \omterm{def:b2:reduction:eigen}{eigenwaarden} gelijk $1$, en $A$,
\omterm{def:b2:reduction:diag}{diagonaliseerbaar} met enige \omterm{def:b2:reduction:eigen}{eigenwaarde} $1$, is $P\,I\,P^{-1} =
I$.
\end{solution}

\begin{solution}{exo:b2:reduction:9}
Schrijf $E = \bigoplus_\lambda E_\lambda(u)$
(\cref{thm:b2:reduction:diagcrit}). Elke $E_\lambda(u)$ is
$v$-stabiel: voor $x \in E_\lambda$, $u(v(x)) = v(u(x)) =
\lambda v(x)$. De restrictie van $v$ tot $E_\lambda(u)$ is
\omterm{def:b2:reduction:diag}{diagonaliseerbaar} (\cref{exo:b2:reduction:4}): kies een basis
van $E_\lambda(u)$ bestaande uit $v$-eigenvectoren. Concatenatie
van deze bases over alle $\lambda$ geeft een basis van $E$
waarvan de vectoren \omterm{def:b2:reduction:eigen}{eigenvectoren} zijn van \emph{beide} $u$
(door lidmaatschap van $E_\lambda(u)$) en $v$ (per constructie).
\end{solution}

\begin{solution}{exo:b2:reduction:10}
($\Rightarrow$) Zij $u$ \omterm{def:b2:reduction:diag}{diagonaliseerbaar} en $F$ stabiel. Dan is
$u|_F$ \omterm{def:b2:reduction:diag}{diagonaliseerbaar} (\cref{exo:b2:reduction:4}): $F$ heeft
een basis van \omterm{def:b2:reduction:eigen}{eigenvectoren}, die zich, binnen elke globale
\omterm{def:b2:reduction:eigen}{eigenruimte} $E_\lambda$, uitbreidt tot een basis van
$E_\lambda$ (onvolledige-basisstelling binnen $E_\lambda$,
startend vanuit het deel van de basis van $F$ dat daar ligt ---
merk op $F = \bigoplus_\lambda (F \cap E_\lambda)$ omdat
$u|_F$ \omterm{def:b2:reduction:diag}{diagonaliseerbaar} is). De toegevoegde vectoren spannen
een stabiel complement op (elk ligt in zekere $E_\lambda$, dus
is hun opspansel $u$-stabiel).

($\Leftarrow$) Zij $F = \sum_\lambda E_\lambda(u)$ (een stabiele
deelruimte) en $G$ een stabiel complement. Als $G \neq \{0\}$:
$\chi_{u|_G}$ splitst over $\C$, dus heeft $u|_G$ een
\omterm{def:b2:reduction:eigen}{eigenvector} $x \in G$
(\cref{thm:b2:reduction:trigonalization} of direct bestaan van
een wortel); maar elke \omterm{def:b2:reduction:eigen}{eigenvector} van $u$ ligt in $F$, dus $x
\in F \cap G = \{0\}$: contradictie. Dus $G = \{0\}$ en $E =
F$: de \omterm{def:b2:reduction:eigen}{eigenruimten} vullen $E$, d.w.z.\ $u$ is
\omterm{def:b2:reduction:diag}{diagonaliseerbaar}.
\end{solution}

\begin{solution}{exo:b2:reduction:11}
Trigonaliseer: $A = PTP^{-1}$, $T = \begin{pmatrix} \lambda &
c\\ 0 & \mu\end{pmatrix}$, $\abs\lambda, \abs\mu < 1$. Dan $A^k
= PT^kP^{-1}$, en het volstaat dat $T^k \to 0$.

\emph{Verschillende \omterm{def:b2:reduction:eigen}{eigenwaarden}:} inductie geeft
\[
T^k = \begin{pmatrix}
\lambda^k & c\,\dfrac{\lambda^k - \mu^k}{\lambda - \mu}\\[4pt]
0 & \mu^k
\end{pmatrix},
\]
en elke entry neigt naar $0$ ($\abs{\lambda}^k, \abs\mu^k \to
0$).

\emph{Gelijke \omterm{def:b2:reduction:eigen}{eigenwaarden} ($\mu = \lambda$):} $T = \lambda I +
cE_{12}$ en $T^k = \lambda^k I + k\lambda^{k-1}cE_{12}$; de
entry $k\lambda^{k-1} \to 0$ omdat $\abs\lambda < 1$
(geometrisch verslaat polynomiaal). In beide gevallen $T^k \to
0$ entrygewijs, dus $A^k = PT^kP^{-1} \to 0$
(matrixvermenigvuldiging met vaste $P, P^{-1}$ is continu in de
entries --- elke entry van het product is een vaste lineaire
combinatie).
\end{solution}

\begin{solution}{exo:b2:reduction:12}
$\ker u$ heeft dimensie $n - 1$ (rang--nulheid), dus is $0$ een
\omterm{def:b2:reduction:eigen}{eigenwaarde} van \omterm{def:b2:reduction:charpoly}{meetkundige multipliciteit} $n - 1$, en is
$\chi_u$ deelbaar door $X^{n-1}$
(\cref{def:b2:reduction:charpoly}: meetkundig $\leq$
algebraïsch). Schrijf $\chi_u = X^{n-1}(X - \alpha)$; de
coëfficiënt van $X^{n-1}$ is $-\operatorname{tr} u$, dus $\alpha
= \operatorname{tr} u$: $\chi_u = X^{n-1}(X - \operatorname{tr}
u)$.

Als $\operatorname{tr} u \neq 0$: de \omterm{def:b2:reduction:eigen}{eigenwaarde}
$\operatorname{tr} u$ is een wortel van $\chi_u$, dus draagt ze
een \omterm{def:b2:reduction:eigen}{eigenvector}; de \omterm{def:b2:reduction:eigen}{eigenruimten} voor $0$ en $\operatorname{tr}
u$ hebben dimensies $n - 1$ en $\geq 1$, sommerend tot $\geq n$:
ze vullen $E$, en $u$ is \omterm{def:b2:reduction:diag}{diagonaliseerbaar}
(\cref{thm:b2:reduction:diagcrit}). Als $\operatorname{tr} u =
0$: door \cref{exo:b2:linalg:5}, $u^2 = (\operatorname{tr}
u)u = 0$ met $u \neq 0$: $u$ is een niet-nulle nilpotente, en
een diagonaliseerbare nilpotente is nul
(\cref{prop:b2:reduction:nilpotent}): niet \omterm{def:b2:reduction:diag}{diagonaliseerbaar}.
\end{solution}

\begin{solution}{pb:b2:reduction:1}
\textbf{1.} De eerste $k - 1$ coördinaten van $Cv_n$ zijn
$u_{n+1}, \dots, u_{n+k-1}$ (de superdiagonaalverschuivingen),
en de laatste is $a_0 u_n + \dots + a_{k-1}u_{n+k-1}$. Dus
$v_{n+1} = Cv_n$ geldt voor alle $n$ desda de laatste
coördinaten voor alle $n$ matchen, d.w.z.\ desda $(\mathcal R)$
geldt. Itererend, $v_n = C^nv_0$.

\textbf{2.} Ontwikkel $D_k(X) = \det(XI_k - C)$ langs de eerste
kolom: de twee niet-nulle entries zijn $X$ (positie $(1,1)$) en
$-a_0$ (positie $(k,1)$). De eerste minor is $D_{k-1}$-vormig
voor de coëfficiënten $a_1, \dots, a_{k-1}$; de tweede minor is
bovendriehoekig met diagonaal $-1$: \omterm{def:b2:linalg:det}{determinant} $(-1)^{k-1}$,
met teken $(-1)^{k+1}$ van de positie. Inductie op $k$ (basis $k
= 1$: $X - a_0$) geeft
\[
D_k(X) = X\bigl(X^{k-1} - a_{k-1}X^{k-2} - \dots - a_1\bigr) -
a_0 = P(X).
\]
Voor $\mu_C$: omdat $Q(C^{\mathsf T}) = Q(C)^{\mathsf T}$ voor
elke veelterm, hebben $C$ en $C^{\mathsf T}$ dezelfde
\omterm{def:b2:linalg:annihilator}{annihilatoren}, dus dezelfde \omterm{def:b2:reduction:polyu}{minimale veelterm}. Voor $C^{\mathsf
T}$: de kolommen lezen $C^{\mathsf T}e_1 = e_2$, \dots,
$C^{\mathsf T}e_{k-1} = e_k$, dus is $(e_1, C^{\mathsf T}e_1,
\dots, (C^{\mathsf T})^{k-1}e_1)$ de canonieke basis: vrij. Een
veelterm $Q \neq 0$ van graad $< k$ heeft dan $Q(C^{\mathsf
T})e_1 \neq 0$ (ze is een niet-triviale combinatie van
basisvectoren): $\deg\mu \geq k$. Omdat $\mu \mid \chi = P$ met
$\deg P = k$: $\mu_C = P$.

\textbf{3.} Voor $v = (1, \lambda, \dots, \lambda^{k-1})^{\mathsf
T}$: rijen $1$ tot $k-1$ van $Cv$ geven $\lambda, \lambda^2,
\dots, \lambda^{k-1}$, d.w.z.\ $\lambda$ maal de eerste $k - 1$
entries van $v$; de laatste rij geeft $\sum_m a_m\lambda^m =
\lambda^k - P(\lambda) = \lambda^k =
\lambda\cdot\lambda^{k-1}$. Dus $Cv = \lambda v$. Omgekeerd
luiden de vergelijkingen $(Cx)_i = \lambda x_i$ voor $i < k$
$x_{i+1} = \lambda x_i$: elke \omterm{def:b2:reduction:eigen}{eigenvector} is proportioneel met
$v$ --- elke \omterm{def:b2:reduction:eigen}{eigenruimte} heeft precies dimensie $1$.
\omterm{def:b2:reduction:diag}{Diagonaliseerbaar} desda de eigenruimtedimensies tot $k$
sommeren (\cref{thm:b2:reduction:diagcrit}) desda er $k$
verschillende \omterm{def:b2:reduction:eigen}{eigenwaarden} zijn desda $P$ $k$ verschillende
wortels heeft (de \omterm{def:b2:reduction:eigen}{eigenwaarden} zijn de wortels van $\chi_C =
P$).

\textbf{4.} Elke $(\lambda_i^n)_n$ lost $(\mathcal R)$ op:
$\lambda_i^{n+k} = \lambda_i^n\,\lambda_i^k =
\lambda_i^n\sum_m a_m\lambda_i^m$. Vrijheid: een verdwijnende
combinatie $\sum_i c_i\lambda_i^n = 0$ voor $n = 0, \dots, k-1$
is een Vandermondestelsel (\cref{exo:b2:linalg:11}) in de $c_i$:
alle $c_i = 0$. De oplossingsruimte heeft dimensie $k$ (vraag 6,
waarvan het bewijs elementair en onafhankelijk is): $k$ vrije
oplossingen vormen een basis, en coördinaten zijn uniek.

\textbf{5.} $P = X^2 - X - 6 = (X - 3)(X + 2)$: algemene
oplossing $u_n = A\,3^n + B(-2)^n$. Beginvoorwaarden: $A + B =
1$, $3A - 2B = 8$: $A = 2$, $B = -1$:
\[
u_n = 2\cdot 3^n - (-2)^n .
\]
(Controle: $u_2 = u_1 + 6u_0 = 14$ en $2\cdot9 - 4 = 14$.)

\textbf{6.} $P(S)\bigl((u_n)\bigr)$ is de rij $n \mapsto
u_{n+k} - a_{k-1}u_{n+k-1} - \dots - a_0u_n$: ze verdwijnt desda
$(\mathcal R)$ geldt, dus is de oplossingsverzameling $\ker
P(S)$, een deelruimte. De afbeelding $\ker P(S) \to \C^k$, $u
\mapsto (u_0, \dots, u_{k-1})$, is lineair, injectief (de
recurrentie bepaalt $u_{k}, u_{k+1}, \dots$ uit de eerste $k$
waarden, door inductie) en surjectief (definieer $u_n$
recursief uit willekeurige beginwaarden): dimensie $k$.

\textbf{7.} Het bewijs van \cref{thm:b2:reduction:kernels}
gebruikt alleen: de Bézout-identiteit in $\C[X]$, en het feit
dat veeltermen in een vast endomorfisme commuten. Geen van
beide noemt de dimensie van de omgevingsruimte: het lemma geldt
letterlijk voor $S \in \mathcal{L}(\mathcal{S})$. Dus
\[
\ker P(S) = \bigoplus_{i=1}^{r}
\ker\,(S - \lambda_i\,\mathrm{id})^{m_i}.
\]

\textbf{8.} Voor $Q \in \C[X]$: $(S -
\lambda)\bigl(Q(n)\lambda^n\bigr)_n$ heeft $n$-de term
$Q(n{+}1)\lambda^{n+1} - \lambda Q(n)\lambda^n =
\lambda^{n+1}(\Delta Q)(n)$, met $\Delta Q = Q(X{+}1) - Q(X)$
van graad $\deg Q - 1$ (de leidende termen cancelleren).
Itererend, $(S - \lambda)^m\bigl(Q(n)\lambda^n\bigr) =
\bigl(\lambda^{n+m}(\Delta^m Q)(n)\bigr)_n$, en $\Delta^m Q =
0$ wanneer $\deg Q \leq m - 1$: de rechterverzameling is bevat
in de kern. Ze is een deelruimte van dimensie $m$: de rijen
$(n^j\lambda^n)_n$, $0 \leq j < m$, zijn vrij, omdat $\sum_j
c_j n^j\lambda^n = 0$ voor alle $n$ (delen door $\lambda^n \neq
0$) de veelterm $\sum_j c_jX^j$ dwingt te verdwijnen in elke $n
\in \N$, dus nul te zijn. Omgekeerd $\dim\ker(S - \lambda)^m
\leq m$: ontwikkelend $(S - \lambda)^m = \sum_j
\binom mj(-\lambda)^{m-j}S^j$, is de vergelijking $(S -
\lambda)^m u = 0$ een lineaire recurrentie van orde $m$
(leidende coëfficiënt $1$), dus is $u$ bepaald door $u_0,
\dots, u_{m-1}$ zoals in vraag 6. Dimensiegelijkheid besluit.

\textbf{9.} Combineer vragen 7 en 8: elke oplossing ontbindt
uniek als som van elementen van de $\ker(S -
\lambda_i)^{m_i}$, d.w.z.\ $u_n = \sum_i Q_i(n)\lambda_i^n$ met
$\deg Q_i \leq m_i - 1$; de $Q_i$ zijn uniek omdat de
ontbinding direct is en, binnen elke sommand, de coëfficiënten
van $Q_i$ coördinaten zijn in de basis $(n^j\lambda_i^n)_j$
(vraag 8). Saniteitscontrole op dimensies: $\sum_i m_i = k$.

\textbf{10.} $P = X^2 - 4X + 4 = (X - 2)^2$: oplossingen $(a +
bn)2^n$. Beginwaarden: $a = 1$, $2(a + b) = 0$, dus $b = -1$:
\[
u_n = (1 - n)\,2^n .
\]
Controle: $u_2 = 4u_1 - 4u_0 = -4$, en $(1 - 2)\cdot4 = -4$.

\textbf{11.} Schrijf $u_n = \lambda_1^n\bigl(c_1 + \sum_{i\geq2}
c_i(\lambda_i/\lambda_1)^n\bigr)$; elke ratio heeft modulus $<
1$, dus neigt de haak naar $c_1 \neq 0$: $u_n \sim
c_1\lambda_1^n$. In het bijzonder $u_n \neq 0$ voor grote $n$,
en
\[
\frac{u_{n+1}}{u_n} =
\lambda_1\,\frac{c_1 + o(1)}{c_1 + o(1)} \longrightarrow
\lambda_1 .
\]

\textbf{12.} Bereken:
\[
q(a_{n+1}, b_{n+1}) = (a_n + 2b_n)^2 - 2(a_n + b_n)^2
= -a_n^2 + 2b_n^2 = -q(a_n, b_n).
\]
Met $q(a_0, b_0) = 1 - 2 = -1$: $a_n^2 - 2b_n^2 = (-1)^{n+1}$.
Structureel: $q(a, b) = (a - \sqrt2\,b)(a + \sqrt2\,b)$ en de
lineaire afbeelding $M$ vermenigvuldigt de factor $a + \sqrt2 b$
met $1 + \sqrt2$ en de factor $a - \sqrt2 b$ met $1 - \sqrt2$
(bereken: $a_{n+1} + \sqrt2 b_{n+1} = (1 + \sqrt2)(a_n +
\sqrt2 b_n)$); het product wordt vermenigvuldigd met $(1 +
\sqrt2)(1 - \sqrt2) = -1 = \det M$ bij elke stap.

\textbf{13.} Omdat $a_n^2 - 2b_n^2 = (a_n - \sqrt2 b_n)(a_n +
\sqrt2 b_n) = (-1)^{n+1}$,
\[
\Bigl|\frac{a_n}{b_n} - \sqrt2\Bigr|
= \frac{\abs{a_n^2 - 2b_n^2}}{b_n(a_n + \sqrt2 b_n)}
= \frac{1}{b_n(a_n + \sqrt2 b_n)} \leq \frac1{2b_n^2},
\]
gebruikend $a_n \geq b_n \geq 1$ (inductie: beide stijgen) dus
$a_n + \sqrt2 b_n \geq (1 + \sqrt2)b_n \geq 2b_n$. \omterm{def:b2:reduction:eigen}{Eigenwaarden}
van $M$: $\chi_M = X^2 - 2X - 1$, wortels $1 \pm \sqrt2$;
omdat $(a_0, b_0)$ een niet-nulle component op de dominante
\omterm{def:b2:reduction:eigen}{eigenvector} heeft (alle entries positief), $b_n \sim c(1 +
\sqrt2)^n$ met $c > 0$ (vraag 11). Dus is de fout $\asymp (1 +
\sqrt2)^{-2n} = (3 + 2\sqrt2)^{-n}$: geometrisch verval met
ratio $1/(3 + 2\sqrt2) = 3 - 2\sqrt2 \approx 0.172$.

\textbf{14.} (a) Uit vraag 9: $\abs{u_n} \leq \sum_i
\abs{Q_i(n)}\abs{\lambda_i}^n \leq \bigl(\sum_i
\abs{Q_i(n)}\bigr)\rho^n$, en elke $\abs{Q_i(n)} \leq C_i
n^{m_i - 1} \leq C_i n^{m-1}$ voor $n \geq 1$: sommeer de
constanten. (b) Zij $\rho' = \max_{i \geq 2}\abs{\lambda_i} <
\abs{\lambda_1}$ en $d = \deg Q_1$, leidende coëfficiënt $c
\neq 0$. Dan $u_n = Q_1(n)\lambda_1^n + R_n$ met $\abs{R_n}
\leq Cn^{m-1}\rho'^n$, en
\[
\frac{R_n}{Q_1(n)\lambda_1^n} = O\Bigl(n^{m-1-d}
\bigl(\rho'/\abs{\lambda_1}\bigr)^n\Bigr) \longrightarrow 0
\]
(geometrisch verslaat polynomiaal). Dus
\[
u_n \sim Q_1(n)\,\lambda_1^n \sim c\,n^d\lambda_1^n,
\qquad
\frac{u_{n+1}}{u_n} \longrightarrow \lambda_1
\quad\text{(omdat } Q_1(n{+}1)/Q_1(n) \to 1\text{)}.
\]
Controle op vraag 10: voor $u_n = (1-n)2^n$ is de ratio
\[
\frac{(-n)2^{n+1}}{(1-n)2^n} = 2\,\frac{-n}{1-n}
\longrightarrow 2 = \lambda_1 .
\]

\textbf{15.} Inductie op $n$. Voor $n = 1$ telt $A_{ij}$
wandelingen van lengte $1$. Stap: een wandeling van lengte $n +
1$ van $i$ naar $j$ is een wandeling van lengte $n$ van $i$
naar zekere knoop $\ell$ gevolgd door een rand $\ell j$:
\[
\#\{\text{wandelingen}\} = \sum_{\ell} (A^n)_{i\ell}A_{\ell j} =
(A^{n+1})_{ij}.
\]

\textbf{16.} $J = 3\Pi$ waarbij $\Pi = J/3$ de projectie is op
$\operatorname{Vect}(1,1,1)$ langs het vlak $x + y + z = 0$
($\Pi^2 = \Pi$ omdat $J^2 = 3J$). Dan $A = J - I = 2\Pi - (I -
\Pi)$, en omdat $\Pi$ en $I - \Pi$ complementaire projecties
zijn,
\[
A^n = 2^n\,\Pi + (-1)^n (I - \Pi),
\qquad\text{d.w.z.}\qquad
(A^n)_{ij} = \frac{2^n}3 + (-1)^n\Bigl(\delta_{ij} -
\frac13\Bigr),
\]
wat de twee weergegeven formules geeft. Bij $n = 2$: diagonaal
$(4 + 2)/3 = 2$ (wandelingen $i \to \ell \to i$ voor de twee
buren $\ell$); buitendiagonaal $(4 - 1)/3 = 1$ (de enkele
wandeling $i \to \ell \to j$ door de derde knoop).

\textbf{17.} Zij $w_n^{(0)}, w_n^{(1)}$ toegelaten woorden van
lengte $n$ eindigend op $0$, resp.\ $1$ tellen. Een letter
toevoegen: een $0$ mag op alles volgen, een $1$ alleen op een
$0$:
\[
\begin{pmatrix} w_{n+1}^{(0)}\\ w_{n+1}^{(1)}\end{pmatrix}
= \begin{pmatrix} 1 & 1\\ 1 & 0\end{pmatrix}
\begin{pmatrix} w_n^{(0)}\\ w_n^{(1)}\end{pmatrix}.
\]
Sommerend, $w_{n+2} = w_{n+1} + w_n$ (of: conditioneer op de
eerste letter). Met $w_1 = 2$, $w_2 = 3$: $w_n = F_{n+2}$ door
inductie ($F_3 = 2$, $F_4 = 3$, zelfde recurrentie). Groei: de
wortels van $X^2 - X - 1$ zijn $\varphi > \abs\psi$
(\cref{exo:b2:reduction:5}), en de $\varphi$-component is
niet-nul (de $w_n$ zijn positief en $\psi^n \to 0$), dus geeft
vraag 11 $w_{n+1}/w_n \to \varphi = \frac{1 + \sqrt5}2$.

\textbf{18.} $A = \begin{psmallmatrix} 0&1&0\\ 1&0&1\\
0&1&0\end{psmallmatrix}$. Controle:
\[
A(1, \pm\sqrt2, 1)^{\mathsf T}
= (\pm\sqrt2, 2, \pm\sqrt2)^{\mathsf T}
= \pm\sqrt2\,(1, \pm\sqrt2, 1)^{\mathsf T},
\qquad
A(1, 0, -1)^{\mathsf T} = 0 :
\]
\omterm{def:b2:reduction:eigen}{eigenwaarden} $\sqrt2, -\sqrt2, 0$ ($= 2\cos\frac\pi4,
2\cos\frac{3\pi}4, 2\cos\frac\pi2$). Ontbind $e_1$ op de
eigenbasis en lees de derde coördinaat, of gebruik symmetrie:
met $v_\pm = (1, \pm\sqrt2, 1)$, $v_0 = (1, 0, -1)$, controleert
men $e_1 = \frac14 v_+ + \frac14 v_- + \frac12 v_0$, dus voor
$n \geq 1$
\[
(A^n)_{13} = \Bigl(\tfrac14(\sqrt2)^n v_+ +
\tfrac14(-\sqrt2)^n v_- + 0\Bigr)_{\!3}
= \frac{(\sqrt2)^n + (-\sqrt2)^n}{4},
\]
nul voor oneven $n$ (bipartiete graaf: einden liggen op even
afstand), en $2\cdot 2^{n/2}/4 = 2^{n/2 - 1}$ voor even $n$. Bij
$n = 4$: $2^{1} = 2$, matchend de twee wandelingen $1\,2\,1\,2\,3$
en $1\,2\,3\,2\,3$.

\textbf{19.} Gesloten wandelingen van lengte $n$ vanuit $i$ zijn
$(A^n)_{ii}$; sommeren over $i$ geeft $\operatorname{tr}(A^n)$.
Trigonaliserend $A$ (over $\C$), is $A^n$ driehoekig met
diagonaal $\lambda_i^n$: $\operatorname{tr}(A^n) =
\sum_i\lambda_i^n$. Driehoek: $\operatorname{tr}(A^n) =
3\,\frac{2^n + 2(-1)^n}3 = 2^n + 2(-1)^n = 2^n + (-1)^n +
(-1)^n$: het \omterm{def:b2:reduction:eigen}{spectrum} $\{2, -1, -1\}$, consistent met vraag 16.

\textbf{20.} De kolommen van $W^{\mathsf T}$: $W^{\mathsf T}e_i
= e_{i-1}$ voor $i \geq 1$ en $W^{\mathsf T}e_0 = e_{k-1}$;
herlabelen in de volgorde $e_0, e_1, \dots$ is dit precies de
begeleidende matrix van $X^k - 1$ ($a_0 = 1$, andere $a_m =
0$). Vraag 2: $\chi_{W} = \chi_{W^{\mathsf T}} = X^k - 1 =
\mu_{W}$. De wortels $\omega^j$ ($j = 0, \dots, k-1$) zijn de
$k$ verschillende $k$-de eenheidswortels: $W$ is
\omterm{def:b2:reduction:diag}{diagonaliseerbaar} (vraag 3, of \cref{exo:b2:reduction:8}: $W^k
= I$). \omterm{def:b2:reduction:eigen}{Eigenvectoren}: $Wf_j = \sum_m \omega^{-jm}e_{m+1} =
\sum_{m'}\omega^{-j(m'-1)}e_{m'} = \omega^j f_j$.

\textbf{21.} Circulanten zijn veeltermen in $W$, en veeltermen
in een vaste matrix commuten met elkaar. Elke $f_j$ is een
\omterm{def:b2:reduction:eigen}{eigenvector} van elke macht: $W^m f_j = \omega^{jm}f_j$, dus
\[
Cf_j = \sum_m c_m\omega^{jm} f_j = \widehat c(\omega^j)\,f_j :
\]
de basis $(f_0, \dots, f_{k-1})$ (vrij: Vandermonde in de
verschillende $\omega^{-j}$, \cref{exo:b2:linalg:11})
diagonaliseert elke circulant tegelijk, met de gestelde
\omterm{def:b2:reduction:eigen}{eigenwaarden}.

\textbf{22.} De \omterm{def:b2:linalg:det}{determinant} is het product van de \omterm{def:b2:reduction:eigen}{eigenwaarden}
(diagonaliseer): $\det C = \prod_{j}\widehat c(\omega^j)$. Voor
$k = 3$, $c_0 = a$, $c_1 = b$, $c_2 = c$ en $\omega = j =
\eu^{2\iu\pi/3}$:
\[
\det C = (a + b + c)(a + bj + cj^2)(a + bj^2 + cj^4),
\]
en $j^4 = j$: precies de factorisatie van
\cref{exo:b2:linalg:8}.

\textbf{23.} $M = \frac12(W + W^{-1})$ is een circulant
($W^{-1} = W^{k-1}$), met \omterm{def:b2:reduction:eigen}{eigenwaarden} $\frac12(\omega^j +
\omega^{-j}) = \cos\frac{2\pi j}k$ op dezelfde basis $f_j$.
Coördinaten: schrijf $x^{(0)} = \sum_j \alpha_j f_j$. De
coördinaten van $f_j$ sommeren tot $\sum_m \omega^{-jm}$, wat
$k$ is voor $j = 0$ en $0$ anders (geometrische som met ratio
$\omega^{-j} \neq 1$). Sommerend de coördinaten van $x^{(0)}$:
$\sum_m x^{(0)}_m = \alpha_0\,k$, dus $\alpha_0 =
\frac1k\sum_m x^{(0)}_m$, het gemiddelde.

\textbf{24.} $x^{(n)} = M^nx^{(0)} = \sum_j
\alpha_j\cos^n\bigl(\tfrac{2\pi j}k\bigr)f_j$. Voor oneven $k$,
$\abs{\cos(2\pi j/k)} < 1$ voor elke $j \neq 0$ (de hoek is
nooit $0$ of $\pi$), dus neigen alle termen behalve $j = 0$
naar $0$: $x^{(n)} \to \alpha_0 f_0$, de constante vector
gelijk aan het gemiddelde --- middelen op een oneven ring
egaliseert. Voor $k = 4$ zijn de \omterm{def:b2:reduction:eigen}{eigenwaarden} $1, 0, -1, 0$:
de $j = 2$-term $\alpha_2(-1)^nf_2$ met $f_2 = (1, -1, 1,
-1)^{\mathsf T}$ oscilleert eeuwig. De obstructie is het
alternerende gemiddelde: de coördinaten van $x^{(0)}$ met
$(-1)^m$ vermenigvuldigen en sommeren, dezelfde
geometrische-som-berekening geeft $\sum_m (-1)^mx^{(0)}_m =
4\alpha_2$: het proces convergeert desda $x^{(0)}_0 -
x^{(0)}_1 + x^{(0)}_2 - x^{(0)}_3 = 0$, en convergeert dan
naar het gemiddelde.

\textbf{25.} De begeleidende matrix zet een scalaire
recurrentie van orde $k$ om in een eerste-orde
vectorrecurrentie, zodat gesloten formules uitspraken over
$C^n$ worden --- het thuisveld van reductie (vragen 1--5). Het
kernontbindingslemma is pure veeltermalgebra (Bézout plus
commutatie), dus splijt ze $\ker P(S)$ hoewel $\mathcal{S}$
oneindigdimensionaal is (vragen 7--9). Dominante \omterm{def:b2:reduction:eigen}{eigenwaarden}
regeren groei omdat elke andere bijdrage na normalisatie
geometrisch verwaarloosbaar is --- wat ook is waarom de
Pell-fout vervalt met het kwadraat van de dominante wortel
(vragen 11--14). Machten van de adjacentiematrix tellen
wandelingen omdat matrixvermenigvuldiging over tussenliggende
knopen sommeert, zodat spectra gesloten wandelingen tellen
(vragen 15--19). Commuterende matrices delen een eigenbasis, en
één Fourierbasis diagonaliseert dan heel de circulante algebra
in één slag (vragen 20--24). Toppen: de fundamentele stelling
van lineaire recurrenties (vraag 9); en voor niet-negatieve
matrices is de reden dat dominante wortels zoals $\varphi$ of
$1 + \sqrt2$ automatisch reëel, positief en enkelvoudig zijn de
stelling van Perron--Frobenius, bewezen in het Bachelor
jaar~3-volume.
\end{solution}
