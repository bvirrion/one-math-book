\section*{Hoofdstuk \ref{ch:b2:diffeq} --- Differentiaalvergelijkingen}

\begin{solution}{exo:b2:diffeq:1}
Eerste matrix: \omterm{def:b2:reduction:eigen}{eigenwaarden} $1, 2$, \omterm{def:b2:reduction:eigen}{eigenvectoren} $(1,0)$ en $(1,1)$.
Ontbind $X_0 = (1,0) = 1\cdot(1,0) + 0\cdot(1,1)$: de oplossing
is
\[
X(t) = \eu^{t}\begin{pmatrix}1\\ 0\end{pmatrix} .
\]
(De beginvector is zelf een \omterm{def:b2:reduction:eigen}{eigenvector}.)

Tweede: $A = 2I + N$, $N = E_{12}$, $N^2 = 0$: $\eu^{tA} =
\eu^{2t}(I + tN)$, dus
\[
X(t) = \eu^{2t}\begin{pmatrix} 1 & t\\ 0 & 1\end{pmatrix}
\begin{pmatrix}1\\ 0\end{pmatrix}
= \eu^{2t}\begin{pmatrix}1\\ 0\end{pmatrix}.
\]
\end{solution}

\begin{solution}{exo:b2:diffeq:2}
$y' = \sin(ty)$: $\abs{\sin(ty) - \sin(tz)} \leq \abs t\,\abs{y -
z}$ --- \omterm{def:b2:metric:continuity}{Lipschitz} in $y$ uniform op elk tijdssegment:
unieke globale oplossingen op $\R$ (pas de stelling toe op elk
segment).

$y' = y^2$: alleen lokaal \omterm{def:b2:metric:continuity}{Lipschitz}; geen globale stelling, en
inderdaad explodeert $y(0) = 1$ bij $t = 1$.

$y' = \abs y$: $\abs\cdot$ is $1$-Lipschitz: globaal bestaan en
eenduidigheid. Met $y(0) = 0$: $y \equiv 0$ (eenduidigheid!). Met
$y(0) = 1$: $y$ blijft positief (kan de nuloplossing niet kruisen),
dus $y' = y$: $y = \eu^t$.
\end{solution}

\begin{solution}{exo:b2:diffeq:3}
Als $y(t_1) = z(t_1)$ op enig tijdstip, dan lossen $y$ en $z$
hetzelfde Cauchy-probleem in $t_1$ op: eenduidigheid dwingt $y =
z$ op hun gemeenschappelijk interval --- verschillende oplossingen
ontmoeten elkaar nooit.

Voor $y' = y(1-y)$: de constanten $0$ en $1$ zijn oplossingen. Een
oplossing die in $\intoo{0}{1}$ start kan $0$ of $1$ nooit
bereiken (ze zou met een constante oplossing botsen): ze blijft in
$\intoo{0}{1}$, en is dus globaal (begrensd: geen explosie --- bijv.\
via het criterium van \cref{exo:b2:diffeq:9}, of omdat het
vectorveld begrensd is op de gevangen strook).
\end{solution}

\begin{solution}{exo:b2:diffeq:4}
$(A - 2I)^2 = \begin{pmatrix}1 & 1\\ -1 & -1\end{pmatrix}^2 = 0$:
Dunford met $D = 2I$, $N = A - 2I$:
\[
\eu^{tA} = \eu^{2t}\,(I + tN)
= \eu^{2t}\begin{pmatrix} 1 + t & t\\ -t & 1 - t\end{pmatrix}.
\]
Variatie van constanten met $B(s) = (\eu^{2s}, 0)^{\mathsf T}$:
\[
X(t) = \int_0^t \eu^{(t-s)A}B(s)\,\dd s
= \eu^{2t}\int_0^t \begin{pmatrix} 1 + (t-s)\\ -(t-s)
\end{pmatrix}\dd s
= \eu^{2t}\begin{pmatrix} t + \frac{t^2}{2}\\[2pt] -\frac{t^2}{2}
\end{pmatrix},
\]
met $\eu^{(t-s)A}B(s) = \eu^{2(t-s)}(I + (t-s)N)\,(\eu^{2s},
0)^{\mathsf T} = \eu^{2t}\bigl(1 + (t-s),\, -(t-s)\bigr)^{\mathsf
T}$. (Controle: $X(0) = 0$; $X' - AX = (\eu^{2t}, 0)^{\mathsf T}$
door differentiatie.)
\end{solution}

\begin{solution}{exo:b2:diffeq:5}
\emph{Reeks:} $A = \alpha I + \beta J$ met $J =
\begin{pmatrix}0 & -1\\ 1 & 0\end{pmatrix}$, $J^2 = -I$; de twee
sommanden commuten, dus $\eu^{tA} = \eu^{\alpha t}\,\eu^{\beta tJ}$,
en de reeks van $\eu^{\beta t J}$ splitst langs even/oneven machten
in $\cos(\beta t)I + \sin(\beta t)J$: de gestelde
rotatie-schaalmatrix.

\emph{Complex:} identificeer $(x, y) \in \R^2$ met $z = x + \iu y$;
het stelsel $X' = AX$ leest $z' = (\alpha + \iu\beta)z$, waarvan de
oplossing $z(t) = \eu^{\alpha t}\eu^{\iu\beta t}z_0$ precies de
spiraal is: modulus $\eu^{\alpha t}$, argument dat met snelheid
$\beta$ vooruitgaat.
\end{solution}

\begin{solution}{exo:b2:diffeq:6}
Zij $v(t) = \eu^{-k(t - t_0)}\int_{t_0}^t u$. Dan
\[
v'(t) = \eu^{-k(t-t_0)}\Bigl(u(t) - k\int_{t_0}^t u\Bigr)
\leq C\,\eu^{-k(t-t_0)},
\]
door de hypothese. Integreren van $t_0$ tot $t$ ($v(t_0) = 0$):
$v(t) \leq \frac{C}{k}\bigl(1 - \eu^{-k(t - t_0)}\bigr)$, d.w.z.\
$k\int_{t_0}^t u \leq C\bigl(\eu^{k(t-t_0)} - 1\bigr)$; dit
terugvoeren in de hypothese: $u(t) \leq C\eu^{k(t-t_0)}$.

Eenduidigheid/afhankelijkheid: twee oplossingen $y, z$ van de
integraalvergelijking voldoen aan
\[
\norm{y(t) - z(t)} \leq \norm{y_0 - z_0} +
k\int_{t_0}^{t}\norm{y - z},
\]
en Gronwall met $C = \norm{y_0 - z_0}$ geeft de exponentiële
grens; $C = 0$ geeft eenduidigheid.
\end{solution}

\begin{solution}{exo:b2:diffeq:7}
Substitueer $y = y_1 z$ met $y_1 = \frac{\sin t}{t}$: de algemene
formule van ordeverlaging (\cref{prop:b2:diffeq:secondorder}) geeft,
voor $u = z'$,
\[
y_1 u' + \Bigl(2y_1' + \frac{2}{t}\,y_1\Bigr)u = 0
\]
(de vergelijking genormaliseerd als $y'' + \frac2t y' + y = 0$).
Bereken $2y_1' + \frac2t y_1 = 2\,\frac{t\cos t - \sin t}{t^2} +
\frac{2\sin t}{t^2} = \frac{2\cos t}{t}$: dus
\[
\frac{u'}{u} = -\frac{2\cos t}{t}\cdot\frac{t}{\sin t}
= -2\cot t
\quad\Longrightarrow\quad
u = \frac{1}{\sin^2 t}
\quad (\text{tot op een constante}),
\]
en $z = -\cot t$, wat $y_2 = y_1 z = -\frac{\cos t}{t}$ geeft.
Algemene oplossing op $\intoo{0}{\pi}$:
\[
y(t) = \alpha\,\frac{\sin t}{t} + \beta\,\frac{\cos t}{t} .
\]
(Dit zijn de sferische Besselfuncties van orde nul.)
\end{solution}

\begin{solution}{exo:b2:diffeq:8}
Dunford: $A = D + N$ commuterend, $D$ \omterm{def:b2:reduction:diag}{diagonaliseerbaar} met dezelfde
\omterm{def:b2:reduction:eigen}{eigenwaarden}, $N$ nilpotent, dus
\[
\eu^{tA} = \eu^{tD}\,\eu^{tN},
\qquad
\eu^{tN} = \sum_{k < n} \frac{t^kN^k}{k!}
\ \text{(een matrixpolynoom in } t).
\]
Zij $-2\alpha = \max_i \Re\lambda_i < 0$. In een basis die $D$
diagonaliseert, $\vertiii{\eu^{tD}} \leq \eu^{-2\alpha t}$
(entries $\eu^{t\lambda_i}$ van modulus $\eu^{t\Re\lambda_i}$);
\omterm{def:b2:nvs:norm}{normen} in verschillende basissen verschillen met constanten. Dus
\[
\norm{X(t)} \leq \vertiii{\eu^{tA}}\,\norm{X_0}
\leq C'\,\eu^{-2\alpha t}\,(1 + t)^{n-1}\,\norm{X_0}
\leq C\,\eu^{-\alpha t}\norm{X_0} ,
\]
waarbij het polynoom in één exponentiële factor wordt opgenomen
($\eu^{-\alpha t}(1+t)^{n-1} \to 0$, dus begrensd).
\end{solution}

\begin{solution}{exo:b2:diffeq:9}
Zij $y$ een maximale oplossing op $\intco{0}{T}$, $T \leq \infty$,
en veronderstel $T < \infty$. De integraalvorm geeft, voor $t < T$,
\[
\norm{y(t)} \leq \norm{y_0} + \int_0^t \bigl(a\norm{y(s)} +
b\bigr)\dd s
\leq \bigl(\norm{y_0} + bT\bigr) + a\int_0^t\norm{y(s)}\,\dd s ,
\]
en Gronwall bindt $\norm{y(t)} \leq (\norm{y_0} +
bT)\,\eu^{aT} =: M$ op $\intco{0}{T}$: de oplossing blijft in een
compacte bal. Dan is $y' = f(t, y)$ begrensd nabij $T$, dus $y$ is
\omterm{def:b2:metric:continuity}{Lipschitz} nabij $T$ en zet zich continu voort tot $T$
(Cauchy-criterium); het Cauchy-probleem in $(T, y(T))$ oplossen
verlengt $y$ voorbij $T$, in tegenspraak met maximaliteit. Dus $T =
\infty$: geen ontsnapping in eindige tijd onder lineaire groei.
\end{solution}

\begin{solution}{exo:b2:diffeq:10}
Karakteristieke wortels van $r^2 - 3r + 2$: $1$ en $2$, dus de
homogene oplossingen zijn $a\eu^t + b\eu^{2t}$. De gissing
$\alpha\eu^t$ faalt omdat $\eu^t$ al de homogene vergelijking
oplost (de wortel $r = 1$ ``resoneert'' met het rechterlid). Met
$y = \alpha t\,\eu^t$: $y' = \alpha(1 + t)\eu^t$, $y'' =
\alpha(2 + t)\eu^t$, en
\[
y'' - 3y' + 2y = \alpha\eu^t\bigl(2 + t - 3 - 3t + 2t\bigr)
= -\alpha\,\eu^t :
\]
$\alpha = -1$, $y_p = -t\,\eu^t$. Algemene oplossing: $y = a\eu^t
+ b\eu^{2t} - t\eu^t$. Beginwaarden $y(0) = y'(0) = 0$: $a + b
= 0$ en $a + 2b - 1 = 0$: $b = 1$, $a = -1$:
\[
y(t) = \eu^{2t} - (1 + t)\,\eu^{t} .
\]

\end{solution}

\begin{solution}{exo:b2:diffeq:11}
$A = \lambda I + N$ met $N = E_{12} + E_{23}$: $N^2 = E_{13}$,
$N^3 = 0$, en $\lambda I$ commuteert met $N$:
\[
\eu^{tA} = \eu^{\lambda t}\Bigl(I + tN +
\frac{t^2}{2}N^2\Bigr)
= \eu^{\lambda t}\begin{pmatrix}
1 & t & \frac{t^2}{2}\\
0 & 1 & t\\
0 & 0 & 1
\end{pmatrix}.
\]
Oplossingen: $X(t) = \eu^{\lambda t}\bigl(X_0 + tNX_0 +
\frac{t^2}2N^2X_0\bigr)$ --- elke component is $\eu^{\lambda
t}$ maal een polynoom van graad $\leq 2$. De graadgrens is de
nilpotentie-index minus één: de reeks van $\eu^{tN}$ knipt af
bij $N^2$.
\end{solution}

\begin{solution}{exo:b2:diffeq:12}
\begin{enumerate}
  \item Als $X(T) = X(0)$, dan lost $Y(t) = X(t + T)$ op $Y' =
        AY + B(t + T) = AY + B(t)$ met $Y(0) = X(0)$:
        eenduidigheid (\cref{thm:b2:diffeq:linear}) geeft $Y = X$,
        d.w.z.\ $X$ is $T$-periodiek. Het omgekeerde is triviaal.
  \item Trigonaliseer over $\C$: $A = PT'P^{-1}$ met $T'$ boven-
        driehoekig, diagonaal $(\lambda_i)$. Elke macht van een
        driehoekige matrix is driehoekig met diagonaal
        $(\lambda_i^k)$, dus $\eu^{TA} = P\eu^{TT'}P^{-1}$ is
        driehoekig in dezelfde basis met diagonaal
        $(\eu^{T\lambda_i})$: dat zijn de \omterm{def:b2:reduction:eigen}{eigenwaarden}. Dan is
        $I - \eu^{TA}$ inverteerbaar desda $\eu^{T\lambda} \neq
        1$ voor elke \omterm{def:b2:reduction:eigen}{eigenwaarde}, d.w.z.\ desda $T\lambda \notin
        2\iu\pi\Z$, wat de gestelde hypothese is.
  \item Variatie van constanten: $X(T) = \eu^{TA}X(0) +
        \int_0^T\eu^{(T-s)A}B(s)\dd s$, dus $X(T) = X(0)$ leest
        \[
        \bigl(I - \eu^{TA}\bigr)X(0) =
        \int_0^{T}\eu^{(T-s)A}B(s)\,\dd s ,
        \]
        wat een unieke oplossing $X(0)$ heeft onder de
        inverteerbaarheidshypothese: precies één $T$-periodieke
        oplossing. Voor de harmonische oscillator ($\lambda =
        \pm\iu\omega$) is het uitgesloten geval $\omega T \in
        2\pi\Z$: forcering waarvan de periode een veelvoud is van
        de natuurlijke periode --- \omterm{pb:b2:diffeq:1}{resonantie}, zoals het
        weekendprobleem kwantificeert.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b2:diffeq:1}
\textbf{1.} Het karakteristieke polynoom is $\lambda^2 -
\tau\lambda + \delta$, met wortels
$\frac{\tau\pm\sqrt\Delta}{2}$. Als $\delta < 0$ dan $\Delta =
\tau^2 - 4\delta > 0$ en de twee reële wortels hebben product
$\delta < 0$: tegengestelde tekens. Als $\delta > 0$ en $\Delta \geq
0$: reële wortels van product $> 0$ en som $\tau$: beide van het
teken van $\tau$. Als $\Delta < 0$: geconjugeerd paar $\alpha \pm
\iu\beta$ met $\alpha = \frac\tau2$, $\beta =
\frac{\sqrt{-\Delta}}2$.

\textbf{2.} $X(t) = a\,\eu^{\mu t}v_- + b\,\eu^{\lambda t}v_+$.
Banen met $a = 0$ (resp.\ $b = 0$) lopen langs de instabiele
(resp.\ stabiele) eigenlijn; alle andere hebben $\norm X \to \infty$
in beide tijdrichtingen, asymptotisch aan $\R v_+$ als $t \to
+\infty$ en aan $\R v_-$ als $t \to -\infty$: het zadelbeeld.
Begrensdheid op heel $\R$ dwingt $b = 0$ (anders explosie bij
$+\infty$) en $a = 0$ (bij $-\infty$): alleen de oorsprong.

\textbf{3.} Met $\mu < \lambda < 0$ vervallen beide exponentiëlen:
$X(t) \to 0$. Als $b \neq 0$, factoriseer $\eu^{\lambda t}$:
\[
X(t) = \eu^{\lambda t}\bigl(b\,v_\lambda + a\,\eu^{(\mu -
\lambda)t}v_\mu\bigr),
\qquad \eu^{(\mu-\lambda)t} \to 0 :
\]
de richting van $X(t)$ neigt naar $\R v_\lambda$, de
\emph{trage} eigenrichting --- alle banen behalve de snelle as
komen er tangentieel aan (het rechterpaneel van de faseportretten
van het hoofdstuk).

\textbf{4.} In de basis waar $A = \begin{pmatrix} \alpha &
-\beta\\ \beta & \alpha\end{pmatrix}$
(\cref{exo:b2:diffeq:5}; voor de oscillatorstelsels van Deel II
wordt deze vorm bereikt door een expliciete reële basiswissel), is
de oplossing $\eu^{\alpha t}$ maal een rotatie over hoek $\beta
t$: logaritmische spiralen, samentrekkend wanneer $\alpha =
\frac\tau2 < 0$, uitzettend wanneer $\tau > 0$, en gesloten
krommen (ellipsen in de oorspronkelijke coördinaten) wanneer $\tau
= 0$: het centrum.

\textbf{5.} $\Delta = 0$ geeft de dubbele \omterm{def:b2:reduction:eigen}{eigenwaarde} $\lambda =
\frac\tau2$; door Cayley--Hamilton
(\cref{thm:b2:reduction:cayleyhamilton}), $(A - \lambda I)^2 =
0$, dus $N = A - \lambda I$ is nilpotent, commuteert met
$\lambda I$, en $\eu^{tA} = \eu^{\lambda t}(I + tN)$. Als $N =
0$: $A = \lambda I$, alle stralen zijn banen (sterknoop). Als $N
\neq 0$: $X(t) = \eu^{\lambda t}(X_0 + tNX_0)$, en voor $NX_0
\neq 0$ convergeert de richting naar de enige eigenrichting
$\operatorname{im}N$: oneigenlijke knoop. Dit voltooit het
spoor--determinant-beeld.

\textbf{6.} $\tau = -2\zeta\omega$, $\delta = \omega^2 > 0$,
$\Delta = 4\omega^2(\zeta^2 - 1)$. Dus: $0 < \zeta < 1$ geeft
$\Delta < 0$, $\tau < 0$: stabiele spiraal; $\zeta = 1$: $\Delta =
0$: ontaarde stabiele knoop; $\zeta > 1$: $\Delta > 0$, $\tau <
0$, $\delta > 0$: stabiele knoop; $\zeta = 0$: $\tau = 0$,
$\delta > 0$: centrum. Een verticale reis in het vlak bij
$\delta = \omega^2$.

\textbf{7.} Wortels $r = -\zeta\omega \pm
\omega\sqrt{\zeta^2-1}$. Voor $\zeta < 1$: $r = -\zeta\omega
\pm \iu\omega_d$, $\omega_d = \omega\sqrt{1-\zeta^2}$:
\[
x(t) = \eu^{-\zeta\omega t}\bigl(a\cos\omega_dt +
b\sin\omega_dt\bigr)
= R\,\eu^{-\zeta\omega t}\cos(\omega_dt - \varphi) .
\]
Voor $\zeta = 1$: $x = (a + bt)\eu^{-\omega t}$. Voor $\zeta >
1$: $x = a\eu^{r_-t} + b\eu^{r_+t}$, beide snelheden negatief.
Opeenvolgende maxima van $\abs x$ in het ondergedempte geval
vinden plaats op tijden gescheiden door de pseudo-periode
$\frac{2\pi}{\omega_d}$ (zelfde fase van de cosinus), en hun
verhouding is $\eu^{-\zeta\omega\cdot2\pi/\omega_d} =
\eu^{-2\pi\zeta/\sqrt{1-\zeta^2}}$: het logaritmische decrement,
een dempingsmeter afleesbaar op een oscilloscoop.

\textbf{8.} Rationaliseren,
\[
\abs{\lambda_{\mathrm{slow}}} = \omega\bigl(\zeta -
\sqrt{\zeta^2-1}\bigr)
= \frac{\omega}{\zeta + \sqrt{\zeta^2 - 1}} ,
\]
waarvan de noemer toeneemt met $\zeta \geq 1$: de
vervalsnelheid is het grootst bij $\zeta = 1$, waar ze gelijk is
aan $\omega$. Een overgedempte deur sluit zonder te slaan maar
\emph{langzaam}; kritische demping is het optimum van de
ingenieur.

\textbf{9.} $E' = x'x'' + \omega^2xx' =
x'\bigl(-2\zeta\omega x' - \omega^2x\bigr) + \omega^2xx' =
-2\zeta\omega\,x'^2 \leq 0$. Als $x$ periodiek en niet-constant
was, zou $E$ periodiek en niet-stijgend zijn, dus constant,
wat $x' \equiv 0$ dwingt: $x$ constant, en dan $\omega^2x = 0$:
$x \equiv 0$. Dus voor $\zeta > 0$ is de enige periodieke
oplossing rust: demping doodt elke cyclus.

\textbf{10.} Het centrum $\zeta = 0$ leeft op de lijn $\tau =
0$ van het spoor--determinant-vlak --- een verzameling met leeg
inwendige: een willekeurig kleine verstoring van de matrix
(elke fysische demping) verplaatst $\tau$ van nul en maakt van de
gesloten banen spiralen. Periodiciteit van de ongedempte
oscillator is dus een scheermesrand-fenomeen, geen robuust
verschijnsel.

\textbf{11.} Substitueren van $x_p = \Re(z\eu^{\iu\gamma t})$ in
de vergelijking:
\[
\bigl(-\gamma^2 + 2\iu\zeta\omega\gamma +
\omega^2\bigr)z = F
\quad\Longrightarrow\quad
z = \frac{F}{\omega^2 - \gamma^2 + 2\iu\zeta\omega\gamma},
\]
dus $x_p = \abs z\cos(\gamma t - \varphi)$ met $\varphi =
\arg(\omega^2 - \gamma^2 + 2\iu\zeta\omega\gamma)$, d.w.z.\
$\tan\varphi = \frac{2\zeta\omega\gamma}{\omega^2 -
\gamma^2}$, en de gestelde $A(\gamma) = \abs z$.

\textbf{12.} Het verschil van twee oplossingen lost de homogene
vergelijking op, die voor $\zeta > 0$ naar $0$ vervalt
(vraag 7): elke oplossing is $x_p$ plus een transient die op
oneindig verdwijnt. De stationaire toestand is een globale
attractor: beginvoorwaarden worden vergeten, alleen $A(\gamma)$
en de fasevertraging $\varphi$ blijven.

\textbf{13.} Minimaliseer $g(u) = (\omega^2 - u)^2 +
4\zeta^2\omega^2u$ over $u = \gamma^2 \geq 0$: $g'(u) =
-2(\omega^2 - u) + 4\zeta^2\omega^2 = 0$ in $u = \omega^2(1 -
2\zeta^2)$, inwendig desda $\zeta < \frac{1}{\sqrt2}$. Daar
\[
g(u_*) = 4\zeta^4\omega^4 + 4\zeta^2\omega^4(1 - 2\zeta^2)
= 4\zeta^2\omega^4(1 - \zeta^2),
\qquad
A(\gamma_*) = \frac{F}{2\zeta\omega^2\sqrt{1 - \zeta^2}} .
\]
Tegen de statische respons $A(0) = \frac{F}{\omega^2}$:
versterking $\frac{1}{2\zeta\sqrt{1-\zeta^2}} \approx
\frac{1}{2\zeta}$ voor kleine $\zeta$ --- een licht gedempt
systeem nabij $\gamma_* \approx \omega$ vermenigvuldigt de input
honderdvoudig wanneer $\zeta = 0{,}005$.

\textbf{14.} De gestelde $x$ voldoet aan $x(0) = x'(0) = 0$ en
\[
x'' + \omega^2 x = \frac{F(\omega^2 -
\gamma^2)\cos\gamma t}{\omega^2 - \gamma^2} = F\cos\gamma t
\]
(de $\cos\omega t$-delen heffen elkaar op). De productvorm volgt
uit $\cos p - \cos q = 2\sin\frac{q+p}{2}\sin\frac{q-p}2$
met $p = \gamma t$, $q = \omega t$. Voor $\gamma$ dicht bij
$\omega$ is de factor $\sin\frac{(\omega-\gamma)t}2$ een trage
envelop die de snelle oscillatie
$\sin\frac{(\omega+\gamma)t}2$ moduleert: zwevingen, met
amplitude $\frac{2F}{\abs{\omega^2-\gamma^2}}$ --- groot, maar
begrensd.

\textbf{15.} Voor $x_p = \frac{F}{2\omega}t\sin\omega t$:
\[
x_p'' = \frac{F}{2\omega}\bigl(2\omega\cos\omega t -
\omega^2t\sin\omega t\bigr)
= F\cos\omega t - \omega^2x_p :
\]
een oplossing. En bij vaste $t$, $\gamma \to \omega$ laten in
vraag 14:
\[
\frac{2F\sin\frac{(\omega-\gamma)t}2
\sin\frac{(\omega+\gamma)t}{2}}
{(\omega-\gamma)(\omega+\gamma)}
\longrightarrow
\frac{2F\cdot\frac{(\omega-\gamma)t}2\big/(\omega-\gamma)
\cdot\sin\omega t}{2\omega}
= \frac{F\,t\sin\omega t}{2\omega} .
\]
De amplitude groeit lineair zonder grens: de resonantieramp ---
de reden dat soldaten op bruggen de pas breken.

\textbf{16.} De blokgolf draagt harmonischen op elke oneven
frequentie $n = 1, 3, 5, \dots$; door lineariteit wordt elke
harmonische $n$ versterkt door de oscillatorrespons bij $\gamma =
n$. Voor $\omega = 3$ raakt de derde harmonische \omterm{pb:b2:diffeq:1}{resonantie}
precies. Ingenieurs vrezen blok- (en zaagtand)inputs omdat ze
\emph{alle} oneven harmonischen tegelijk opwekken: wat de
natuurlijke frequentie van de structuur ook is, er wacht altijd
een harmonische.

\textbf{17.} $W' = y_1y_2'' - y_1''y_2 = -qy_1y_2 + qy_1y_2 =
0$: $W$ is constant (Liouville met een spoorloze begeleidende
matrix). $W = 0$ in één punt maakt de beginwaarden van $y_2$
evenredig met die van $y_1$, dus $y_2$ evenredig met $y_1$
(eenduidigheid); $W \neq 0$ desda onafhankelijk. Als $y(t_0) =
y'(t_0) = 0$ dan $y \equiv 0$ (eenduidigheid): een
niet-nuloplossing heeft enkelvoudige nulpunten, en een
enkelvoudig nulpunt is geïsoleerd ($y'$ van vast teken in de
buurt).

\textbf{18.} Tussen opeenvolgende nulpunten $a < b$ houdt $y_1$
één teken, zeg $y_1 > 0$ op $\intoo ab$: dan $y_1'(a) > 0$ en
$y_1'(b) < 0$ (enkelvoudige nulpunten). Evalueren van de
constante $W = y_1y_2' - y_1'y_2$ in $a$ en $b$:
\[
W = -y_1'(a)\,y_2(a) = -y_1'(b)\,y_2(b) ,
\]
dus $y_2(a)$ en $y_2(b)$ hebben tegengestelde tekens ($W \neq 0$
verbiedt dat een van beide verdwijnt): $y_2$ verdwijnt in
$\intoo ab$ (tussenwaarden). Ze kan daar niet tweemaal
verdwijnen: twee nulpunten van $y_2$ zouden een nulpunt van
$y_1$ flankeren door hetzelfde argument met omgewisselde rollen,
in tegenspraak met opeenvolgendheid: precies één nulpunt ---
afwisseling.

\textbf{19.} Veronderstel dat $z$ geen nulpunt heeft in $\intoo
ab$; door $y, z$ door hun tegenovergestelden te vervangen,
neem $y > 0$ en $z > 0$ op $\intoo ab$ aan. Stel $\varphi =
yz' - y'z$: $\varphi' = yz'' - y''z = (q_1 - q_2)\,yz \leq 0$
op $\intoo ab$: $\varphi$ is niet-stijgend. Maar $\varphi(a) =
-y'(a)z(a) \leq 0$ (want $y'(a) > 0$, $z(a) \geq 0$) en
$\varphi(b) = -y'(b)z(b) \geq 0$ (want $y'(b) < 0$, $z(b) \geq
0$): een niet-stijgende functie die van $\leq 0$ naar $\geq 0$
loopt verdwijnt identiek, dus $(q_1 - q_2)yz \equiv 0$ op
$\intoo ab$. Als $q_1 < q_2$ ergens in $\intoo ab$, is dit
absurd ($y, z > 0$ daar): $z$ moet strikt binnenin verdwijnen. In
het algemeen ($q_1 \leq q_2$), ofwel verdwijnt $z$ in $\intoo
ab$, ofwel dwingt $\varphi \equiv 0$ dat $z$ evenredig is met
$y$, die in $a$ en $b$ verdwijnt: in alle gevallen heeft $z$ een
nulpunt in $\intcc ab$.

\textbf{20.} Bovengrens: vergelijk $y$ (coëfficiënt $q \geq
m^2$) met $u(t) = \sin(m(t - a))$ (coëfficiënt $m^2 \leq q$,
dus $y$ speelt de rol van $z$ in vraag 19): als $y$ geen nulpunt
had in $\intoc{a}{a + \pi/m}$, zouden de nulpunten $a$ en $a +
\frac\pi m$ van $u$ opeenvolgend zijn met $y \neq 0$ ertussen,
in tegenspraak met vraag 19: opeenvolgende nulpunten van $y$
liggen op afstand $\leq \frac\pi m$. Ondergrens: als twee
opeenvolgende nulpunten $a < b$ van $y$ $b - a < \frac\pi M$
hadden, dan zou $z(t) = \sin(M(t-a))$ (coëfficiënt $M^2 \geq
q$) moeten verdwijnen in $\intcc ab \subset \intoo{a}{a +
\pi/M} \cup\{a\}$, waar haar enige nulpunt $a$ zelf is --- maar
vraag 19 toegepast op $\intoo{a}{b}$ met strengheid in de
eindpunten geeft een nulpunt in $\intcc ab$, en $z > 0$ op
$\intoc ab$: tegenspraak. Dus $\frac\pi M \leq b - a \leq
\frac\pi m$; voor $q \equiv \omega^2$ storten beide grenzen in
tot de exacte afstand $\frac\pi\omega$ van de harmonische
oscillator.

\textbf{21.} Met $u = ty$: $u'' = ty'' + 2y'$, dus $ty'' + 2y'
+ ty = u'' + u = 0$: $u = A\sin t + B\cos t = R\sin(t +
\varphi)$, en $y = \frac{u}{t}$ herwint $\frac{\sin t}t$ en
$\frac{\cos t}t$ (\cref{exo:b2:diffeq:7}) zonder
ordeverlaging. De nulpunten van elke niet-nuloplossing zijn die
van $R\sin(t + \varphi)$: precies $\pi$ uit elkaar --- de
filosofie van Sturm in actie: de coëfficiënt $q \equiv 1$
dicteert de nulpunten, formule of geen formule.

\textbf{22.} Stel $x(t) =
\frac1\omega\int_0^t\sin(\omega(t-s))F(s)\dd s$. Dan $x(0) =
0$;
\[
x'(t) = \frac1\omega\sin(0)F(t) +
\int_0^t\cos(\omega(t-s))F(s)\dd s
= \int_0^t\cos(\omega(t-s))F(s)\dd s ,
\]
dus $x'(0) = 0$; en $x''(t) = F(t) -
\omega\int_0^t\sin(\omega(t-s))F(s)\dd s = F(t) - \omega^2x(t)
$ (differentiatie van een \omterm{thm:b2:integration:continuity}{parameterintegraal} met variabele grens,
zoals in het integratiehoofdstuk). Met $F(s) = F\cos\omega s$
geeft product-naar-som
\[
\int_0^t\sin(\omega(t-s))\cos(\omega s)\dd s
= \frac12\int_0^t\bigl(\sin\omega t + \sin(\omega t -
2\omega s)\bigr)\dd s
= \frac{t}{2}\sin\omega t ,
\]
(het tweede stuk integreert tot nul), dus $x =
\frac{F}{2\omega}t\sin\omega t$: vraag 15 opnieuw, via
Duhamel.

\textbf{23.} In stelselvorm $X' = AX + (0, F(t))^{\mathsf T}$
met $\operatorname{Sp}A$ van negatieve reële delen ($\zeta >
0$): variatie van constanten en \cref{exo:b2:diffeq:8}
($\vertiii{\eu^{tA}} \leq C\eu^{-\alpha t}$) geven
\[
\norm{X(t)} \leq C\eu^{-\alpha t}\norm{X_0}
+ \int_0^t C\eu^{-\alpha(t-s)}\norm{F}_\infty\dd s
\leq C\norm{X_0} + \frac{C\norm F_\infty}{\alpha} :
\]
begrensde input, begrensde output --- uniform in de
beginwaarden na de transient.

\textbf{24.} Voor $\zeta = 0$ en $\gamma \neq \omega$ is de
oplossing van vraag 14 begrensd, en elke homogene oplossing
toevoegen (begrensd: de banen van het centrum zijn cirkels)
houdt haar begrensd; bij $\gamma = \omega$ groeit vraag 15
lineair. Dus faalt begrensdheid onder periodieke forcering voor
de ongedempte oscillator bij precies één frequentie ---
\omterm{pb:b2:diffeq:1}{resonantie} --- terwijl vraag 23 toont dat elke positieve demping
begrensdheid herstelt voor \emph{alle} begrensde inputs.

\textbf{25.} (i) Het spoor--determinant-vlak classificeert alle
autonome vlakke lineaire dynamica, en de oscillator van Deel II
loopt een verticale lijn ervan; forcering verlaat het vlak
(niet-autonoom), en Duhamel neemt het over. (ii) $\gamma_*$ is
de frequentie die het systeem prefereert, $A(\gamma_*)$ de prijs
van haar opwekking, en $\frac{1}{2\zeta}$ de versterkingsfactor
--- de resonantiescherpte die ingenieurs de kwaliteitsfactor
noemen. (iii) De stellingen van Sturm lezen oscillatie af van
teken en grootte van $q$ alleen: ze beheersen vergelijkingen
(Bessel, Schrödinger) waarvan de oplossingen geen elementaire
formules hebben. (iv) Constante Wronskianen (vraag 17, via
Liouville) en begrensde-input-stabiliteit (vraag 23) worden
stilzwijgend hergebruikt wanneer het boek
variabele-coëfficiëntvergelijkingen of verstoorde stelsels
ontmoet.
\end{solution}
