\section*{Hoofdstuk \ref{ch:b2:series} --- Rijen en reeksen}

\begin{solution}{exo:b2:series:1}
$\sum\frac{\cos n}{n}$: Abelcriterium met $a_n = \frac1n$ en $b_n =
\cos n$, waarvan de partiële sommen begrensd zijn (reëel deel van
een meetkundige som, zoals in \cref{ex:b2:series:sinn}): convergent
(niet \omterm{def:b2:series:def}{absoluut}, via dezelfde $\cos^2$-truc).

$\sum \frac{(-1)^n}{\ln n}$ ($n \geq 2$): alternerend criterium,
$\frac{1}{\ln n} \downarrow 0$: convergent; niet \omterm{def:b2:series:def}{absoluut} ($\ln n
\leq n$).

$\sum \frac{(-1)^n}{n^{3/4} + \cos n}$: ontwikkel,
\[
\frac{(-1)^n}{n^{3/4} + \cos n}
= \frac{(-1)^n}{n^{3/4}}\cdot
\frac{1}{1 + \frac{\cos n}{n^{3/4}}}
= \frac{(-1)^n}{n^{3/4}}
- \frac{(-1)^n\cos n}{n^{3/2}} + O\Bigl(\frac{1}{n^{9/4}}\Bigr).
\]
Eerste reeks: alternerend, convergent. Tweede: \omterm{def:b2:series:def}{absoluut} convergent
(schaal $\frac{1}{n^{3/2}}$). Derde: \omterm{def:b2:series:def}{absoluut} convergent. Totaal:
convergent.
\end{solution}

\begin{solution}{exo:b2:series:2}
\omterm{thm:b2:series:fubini}{Cauchy-product} van $\sum_{m\geq0} z^m$ met zichzelf (beide \omterm{def:b2:series:def}{absoluut}
convergent voor $\abs z < 1$): de diagonaalcoëfficiënt is $c_k =
\sum_{m=0}^{k} 1 = k + 1$, dus
\[
\frac{1}{(1-z)^2} = \sum_{k\geq0} (k+1)z^k .
\]
Vermenigvuldig met $z$ en herindexeer: $\sum_{n \geq 1} n z^n =
\frac{z}{(1-z)^2}$.
\end{solution}

\begin{solution}{exo:b2:series:3}
Zij $B_n = \sum_{k \leq n} b_k \to B$. Abelsommatie met $a_k = k$:
\[
\sum_{k=1}^{n} k\,b_k = n B_n - \sum_{k=1}^{n-1} B_k
\quad\Longrightarrow\quad
\frac1n \sum_{k=1}^{n} k b_k = B_n - \frac{1}{n}\sum_{k=1}^{n-1}
B_k .
\]
De Cesàro-gemiddelden van de convergente $(B_k)$ streven naar haar
limiet $B$ (boekdeel van Jaar~1), dus de rechterkant streef naar
$B - B = 0$.
\end{solution}

\begin{solution}{exo:b2:series:4}
Als $\theta \in \pi\Z$: alle termen verdwijnen --- triviaal
convergent. Neem $\theta \notin \pi\Z$.

$\alpha > 1$: \omterm{def:b2:series:def}{absoluut} convergent (dominantie door $n^{-\alpha}$).

$0 < \alpha \leq 1$: Abelcriterium past ($a_n = n^{-\alpha}
\downarrow 0$; partiële sommen van $\sin n\theta$ begrensd door
$\frac{1}{\abs{\sin(\theta/2)}}$, meetkundige som): convergent.
Niet \omterm{def:b2:series:def}{absoluut}: $\abs{\sin n\theta} \geq \sin^2 n\theta = \frac{1 -
\cos 2n\theta}{2}$, en $\sum \frac{1 - \cos 2n\theta}{2n^\alpha}$
divergeert ($\sum n^{-\alpha}$ divergeert; $\sum
\frac{\cos 2n\theta}{n^\alpha}$ convergeert door Abel wanneer
$2\theta \notin 2\pi\Z$; het uitgesloten geval $2\theta \in
2\pi\Z$ betekent $\theta \in \pi\Z$, reeds behandeld).
Semi-convergent.
\end{solution}

\begin{solution}{exo:b2:series:5}
Zij $p_1, p_2, \dots$ de niet-negatieve termen van $(u_n)$ in
volgorde, $q_1, q_2, \dots$ de negatieve. Zowel $\sum p_k$ als
$\sum q_k$ divergeren: als een van beide convergeerde, zou de
andere gelijk zijn aan de convergente $\sum u_n$ min die, dus ook
convergeren --- en dan zou $\sum \abs{u_n} = \sum p_k - \sum q_k$
convergeren, in tegenspraak met de hypothese. Ook $u_n \to 0$
($\sum u_n$ convergeert).

Gretige herschikking: neem positieve termen $p_1, p_2, \dots$ tot
het lopende totaal $\ell$ voor het eerst overschrijdt (mogelijk:
$\sum p_k = +\infty$); dan negatieve termen tot het totaal voor
het eerst onder $\ell$ daalt (mogelijk: $\sum q_k = -\infty$);
herhaal eeuwig (elke fase is eindig, en elke term wordt precies
één keer gebruikt: een echte herschikking). Na elke wissel is de
afstand van het lopende totaal tot $\ell$ ten hoogste de
laatstgebruikte term; aangezien de termen bij de $m$-de wissel
index $\to \infty$ hebben, en $u_n \to 0$, convergeren de
lopende totalen naar $\ell$.
\end{solution}

\begin{solution}{exo:b2:series:6}
Sommeerbaarheid: de eindige partiële sommen van
$\frac{\abs x^{m+n}}{m!n!}$ zijn begrensd door $\bigl(\sum_m
\frac{\abs x^m}{m!}\bigr)^2 = \eu^{2\abs x}$. Diagonaalgroepering
(\cref{thm:b2:series:fubini}):
\[
(\eu^{x})^2 = \sum_{m,n} \frac{x^{m+n}}{m!\,n!}
= \sum_{k=0}^{\infty} x^k \sum_{m+n=k} \frac{1}{m!\,n!}
= \sum_k \frac{x^k}{k!}\sum_{m=0}^{k}\binom km
= \sum_k \frac{(2x)^k}{k!} = \eu^{2x} .
\]
\end{solution}

\begin{solution}{exo:b2:series:7}
Sommeerbaarheid: begrensd door $\zeta(2)^2$ als productfamilie
(Cauchy-productargument van \cref{thm:b2:series:fubini}). De
afbeelding $(q, a, b) \mapsto (qa, qb)$, van drietallen met
$\gcd(a, b) = 1$ naar paren $(m, n)$, is een bijectie (stel $q =
\gcd(m,n)$). Groepeer de \omterm{def:b2:series:summable}{sommeerbare familie} dienovereenkomstig
(een partitie van de indexverzameling --- legitiem voor
\omterm{def:b2:series:summable}{sommeerbare families} door
\cref{thm:b2:series:rearrangement}/\cref{thm:b2:series:fubini}
toegepast op de partitie in \omterm{def:b2:structures:countable}{aftelbaar} vele klassen):
\[
\zeta(2)^2 = \sum_{q} \sum_{\gcd(a,b)=1} \frac{1}{q^4 a^2 b^2}
= \zeta(4)\, S,
\qquad\text{dus}\qquad
S = \frac{\zeta(2)^2}{\zeta(4)} .
\]
(Met de waarden $\zeta(2) = \frac{\pi^2}{6}$,
$\zeta(4) = \frac{\pi^4}{90}$ uit
\cref{ch:b2:fourier}: $S = \frac{5}{2}$.)
\end{solution}

\begin{solution}{exo:b2:series:8}
Zij $A = \sum a_n$ (\omterm{def:b2:series:def}{absoluut}), $B_m = \sum_{k\leq m} b_k \to B$,
begrensd door $M$. Dan
\[
C_N = \sum_{k=0}^{N} c_k = \sum_{n=0}^{N} a_n B_{N-n}
\]
(verzamel op de index van $a$). Schrijf
\[
C_N - AB = \sum_{n=0}^{N} a_n (B_{N-n} - B) - B\sum_{n > N} a_n .
\]
De laatste term streeft naar $0$. Splits de som bij $n = \lfloor
N/2 \rfloor$: voor $n \leq N/2$ is $N - n \geq N/2$, dus
$\abs{B_{N-n} - B} \leq \varepsilon_N := \sup_{m \geq
N/2}\abs{B_m - B} \to 0$, en dit deel is $\leq \varepsilon_N
\sum\abs{a_n}$; voor $n > N/2$ is $\abs{B_{N-n} - B} \leq 2M$, en
dit deel is $\leq 2M \sum_{n > N/2} \abs{a_n} \to 0$. Dus $C_N
\to AB$.
\end{solution}

\begin{solution}{exo:b2:series:9}
Neem $u_n = 2^{-n} e_n$ ($e_n$ de rij met een enkele $1$ op
positie $n$). Dan $\sum \norm{u_n}_\infty = \sum 2^{-n} <
\infty$: \omterm{def:b2:series:def}{absoluut} convergent. Maar de partiële sommen $S_N =
(1, \tfrac12, \dots, 2^{-N}, 0, \dots)$ zouden moeten
convergeren naar de rij $(2^{-n})_n$, die \emph{niet} uiteindelijk
nul is: buiten $E$. Binnen $E$ is $(S_N)$ Cauchy zonder limiet
($\norm{S_N - x}_\infty \geq 2^{-N-1}$ helpt geen enkele
uiteindelijk-nul $x$: voor elke $x \in E$ die verdwijnt voorbij
rang $K$ is $\norm{S_N - x} \geq 2^{-K-1}$ voor $N > K$): de reeks
convergeert niet in $E$. Volledigheid is precies wat
\cref{thm:b2:nvs:absoluteconvergence} nodig heeft.
\end{solution}

\begin{solution}{exo:b2:series:10}
Zowel $\arctan(n+1) - \arctan n$ als
$\arctan\frac{1}{n^2+n+1}$ liggen in $\intoo{0}{\frac\pi2}$, en
de tangens-optelformule geeft
\[
\tan\bigl(\arctan(n{+}1) - \arctan n\bigr)
= \frac{(n+1) - n}{1 + n(n+1)} = \frac{1}{n^2 + n + 1} :
\]
gelijke tangensen in een interval waar $\tan$ injectief is, dus
de identiteit geldt. Telescoperend,
\[
\sum_{n=1}^{N}\arctan\frac{1}{n^2+n+1}
= \arctan(N{+}1) - \arctan 1 \xrightarrow[N\to\infty]{}
\frac\pi2 - \frac\pi4 = \frac\pi4 .
\]

\end{solution}

\begin{solution}{exo:b2:series:11}
Eerste: $\sqrt{n+1} - \sqrt n = \frac{1}{\sqrt{n+1} + \sqrt n}
\sim \frac{1}{2\sqrt n}$, dus de termen zijn $\sim
2^{-\alpha}n^{-\alpha/2}$: convergentie d.e.s.d.a.\
$\frac\alpha2 > 1$, d.w.z.\ $\alpha > 2$. Tweede: $1 -
\cos\frac1n \sim \frac{1}{2n^2}$: convergeert. Derde:
$\bigl(1 + \frac1n\bigr)^n = \eu^{\,n\ln(1 + 1/n)} =
\eu^{\,1 - \frac1{2n} + O(n^{-2})} =
\eu\bigl(1 - \frac{1}{2n} + O(n^{-2})\bigr)$, dus
\[
\eu - \Bigl(1 + \frac1n\Bigr)^{\!n} \sim \frac{\eu}{2n} :
\]
positieve termen equivalent met een harmonisch veelvoud:
divergeert.
\end{solution}

\begin{solution}{exo:b2:series:12}
Door monotonie is $n\,a_{2n} \leq a_{n+1} + a_{n+2} + \dots +
a_{2n} = S_{2n} - S_n \to 0$ (Cauchycriterium voor de
convergente reeks). Dus $2n\,a_{2n} \to 0$, en $(2n{+}1)\,
a_{2n+1} \leq (2n{+}1)a_{2n} = \frac{2n+1}{2n}\,(2n\,a_{2n}) \to
0$: beide deelrijen van $(na_n)$ streven naar $0$, dus $na_n \to
0$.

\emph{Omgekeerde faalt:} $a_n = \frac1{n\ln n}$ is positief
dalend met $na_n = \frac1{\ln n} \to 0$, maar $\sum a_n$
divergeert (Bertrand-grens, \cref{pb:b2:series:1}, vraag 18).
\emph{Monotonie essentieel:} laat $a_n = \frac1n$ wanneer $n$
een macht van $2$ is en $a_n = 2^{-n}$ anders: $\sum a_n \leq
\sum_k 2^{-k} + \sum_n 2^{-n} < \infty$, maar $na_n = 1$ langs
de machten van $2$: $na_n \not\to 0$.
\end{solution}

\begin{solution}{pb:b2:series:1}
\textbf{1.} Inductie op $m$: voor $m = 0$ zijn beide kanten $1$;
de stap vermenigvuldigt met $\cos\theta + \iu\sin\theta$ en
gebruikt de optelformules $\cos(m\theta + \theta) =
\cos m\theta\cos\theta - \sin m\theta\sin\theta$, $\sin(m\theta
+ \theta) = \sin m\theta\cos\theta + \cos m\theta\sin\theta$.
Uitwerken via het binomium met $m = 2n+1$ en verzamelen van het
imaginaire deel (de oneven machten van $\iu\sin\theta$, met
$\iu^{2j+1} = (-1)^j\iu$):
\[
\sin\bigl((2n{+}1)\theta\bigr) = \sum_{j=0}^{n}(-1)^j
\binom{2n+1}{2j+1}\cos^{2(n-j)}\theta\,\sin^{2j+1}\theta .
\]

\textbf{2.} Op $\intoo0{\frac\pi2}$ is $\sin\theta \neq 0$:
factoriseer $\sin^{2n+1}\theta$ uit elke term, en houd
$\bigl(\frac{\cos^2\theta}{\sin^2\theta}\bigr)^{n-j} =
(\cot^2\theta)^{n-j}$ over: de getoonde identiteit met $P_n(x)
= \sum_j(-1)^j\binom{2n+1}{2j+1}x^{n-j}$. Haar
graad-$n$-coëfficiënt is die van $j = 0$: $\binom{2n+1}{1} =
2n + 1 \neq 0$.

\textbf{3.} In $\theta_k = \frac{k\pi}{2n+1}$:
$\sin\bigl((2n{+}1)\theta_k\bigr) = \sin k\pi = 0$ terwijl
$\sin^{2n+1}\theta_k \neq 0$, dus $P_n(\cot^2\theta_k) = 0$. De
$\theta_k$ stijgen strikt in $\intoo0{\frac\pi2}$, waar
$\cot^2$ strikt daalt: de waarden $x_k = \cot^2\theta_k$ zijn
paarsgewijs onderscheiden --- $n$ onderscheiden wortels van een
veelterm van graad $n$, dus alle.

\textbf{4.} Vieta: de som van de wortels is min de verhouding
van de $x^{n-1}$- en $x^n$-coëfficiënten:
\[
\sum_{k=1}^{n}\cot^2\theta_k =
\frac{\binom{2n+1}{3}}{\binom{2n+1}{1}}
= \frac{(2n+1)(2n)(2n-1)/6}{2n+1} = \frac{n(2n-1)}{3}.
\]

\textbf{5.} $\frac{1}{\sin^2\theta} = 1 + \cot^2\theta$:
sommeren, $n + \frac{n(2n-1)}3 = \frac{3n + 2n^2 - n}{3} =
\frac{2n(n+1)}{3}$.

\textbf{6.} Op $\intoo{0}{\frac\pi2}$: $\sin\theta < \theta <
\tan\theta$ (boekdeel van Jaar~1). Reciproke keren om:
$\cot\theta < \frac1\theta < \frac1{\sin\theta}$, en kwadrateren
(alles positief) geeft $\cot^2\theta < \frac1{\theta^2} <
\frac1{\sin^2\theta}$.

\textbf{7.} Sommeer vraag 6 in $\theta = \theta_k$ over $k \leq
n$, met vragen 4 en 5, en $\frac1{\theta_k^2} =
\frac{(2n+1)^2}{k^2\pi^2}$:
\[
\frac{n(2n-1)}3 < \frac{(2n+1)^2}{\pi^2}\sum_{k=1}^n\frac1{k^2}
< \frac{2n(n+1)}3 .
\]

\textbf{8.} Vermenigvuldig met $\frac{\pi^2}{(2n+1)^2}$:
\[
\frac{\pi^2}{3}\cdot\frac{n(2n-1)}{(2n+1)^2}
< \sum_{k=1}^{n}\frac1{k^2} <
\frac{\pi^2}{3}\cdot\frac{2n(n+1)}{(2n+1)^2}.
\]
Beide grenzen streven naar $\frac{\pi^2}3\cdot\frac12 =
\frac{\pi^2}6$ (de rationale breuken streven naar $\frac12$).
De partiële sommen stijgen, dus convergeren zij, en de
insluiting geeft $\zeta(2) = \frac{\pi^2}6$: de stelling van
Euler, via Cauchys bewijs.

\textbf{9.} De partiële sommen stijgen naar $\zeta(2) =
\frac{\pi^2}6$, dus $0 \leq \frac{\pi^2}6 - \sum_{k\leq
n}k^{-2}$; en de ondergrens van vraag 8 geeft
\[
\frac{\pi^2}6 - \sum_{k\leq n}\frac1{k^2}
\leq \frac{\pi^2}6 - \frac{\pi^2}3\cdot\frac{n(2n-1)}{(2n+1)^2}
= \frac{\pi^2}6\cdot\frac{(2n+1)^2 - (4n^2 -
2n)}{(2n+1)^2}
= \frac{\pi^2}6\cdot\frac{6n + 1}{(2n+1)^2}
= O\Bigl(\frac1n\Bigr),
\]
in overeenstemming met de exacte staart $\frac1n -
\frac1{2n^2} + O(n^{-3})$ van \cref{exo:b2:comparison:11}.

\textbf{10.} De tweede elementaire symmetrische functie van de
wortels is $\sigma_2 = \frac{\binom{2n+1}5}{\binom{2n+1}1} =
\frac{(2n)(2n-1)(2n-2)(2n-3)}{120}$, dus
\[
\sum_k\cot^4\theta_k = \sigma_1^2 - 2\sigma_2
= \Bigl(\frac{n(2n-1)}3\Bigr)^2 -
\frac{2n(2n-1)(2n-2)(2n-3)}{60}
\sim \frac{4n^4}9 - \frac{4n^4}{15} = \frac{8n^4}{45}.
\]
Insluiten $\cot^4\theta < \theta^{-4} < (1 + \cot^2\theta)^2 =
1 + 2\cot^2\theta + \cot^4\theta$ en sommeren: beide
buitensommen zijn $\frac{8n^4}{45}(1 + o(1))$ (de toegevoegde
$n + 2\sigma_1 = O(n^2)$ is verwaarloosbaar), terwijl de
middelste $\frac{(2n+1)^4}{\pi^4}\sum_{k\leq n}k^{-4}$ is. Dus
\[
\sum_{k\leq n}\frac1{k^4} \longrightarrow
\pi^4\cdot\frac{8/45}{16} = \frac{\pi^4}{90}.
\]

\textbf{11.} Splits $\zeta(2)$ over pariteiten: $\sum_{\text{even}}
= \sum_j\frac{1}{(2j)^2} = \frac14\zeta(2) = \frac{\pi^2}{24}$,
dus $\sum_{\text{odd}} = \zeta(2) - \frac{\pi^2}{24} =
\frac{\pi^2}8$. Alternerend: $\sum_k\frac{(-1)^{k-1}}{k^2} =
\sum_{\text{odd}} - \sum_{\text{even}} = \frac{\pi^2}8 -
\frac{\pi^2}{24} = \frac{\pi^2}{12}$ (absolute convergentie
rechtvaardigt de hergroepering,
\cref{thm:b2:series:rearrangement}).

\textbf{12.} $S = \frac{\zeta(2)^2}{\zeta(4)} =
\frac{(\pi^2/6)^2}{\pi^4/90} = \frac{90}{36} = \frac52$.
Heuristiek: de identiteit $\zeta(2)^2 = \zeta(4)S$ van
\cref{exo:b2:series:7} zegt dat het uittrekken van de $\gcd$
paren renormaliseert tot coprieme paren; de reciproke
$\frac1{\zeta(2)} = \frac6{\pi^2} \approx 0.608$ is de
natuurlijke kandidaat voor de dichtheid van coprieme paren
onder alle paren --- een uitspraak over $\lim_N
\frac{1}{N^2}\#\{(m,n) \leq N : \gcd = 1\}$ waarvan het
eerlijke bewijs (met fouttermen) tot het boekdeel van Jaar~3
behoort.

\textbf{13.} Directe sommatie heeft fout $\sim \frac1n$: zes
decimalen vereisen ongeveer $10^6$ termen. De gecorrigeerde som
$\sum_{k\leq n}k^{-2} + \frac1n - \frac1{2n^2}$ heeft fout
$O(n^{-3})$: bij $n = 100$ is zij $1.6449339\dots$ tegenover
$\frac{\pi^2}6 = 1.6449341\dots$ --- fout $1.7\cdot10^{-7}$,
zeven decimalen uit honderd termen. Asymptotische correcties
verslaan ruwe geduld met vier ordes van grootte.

\textbf{14.} $n = 1$: $P_1(x) = 3x - 1$, wortel $\frac13$, en
inderdaad $\cot^2\frac\pi3 = \bigl(\frac1{\sqrt3}\bigr)^2 =
\frac13 = \frac{1\cdot1}3$. $n = 2$: de formule voorspelt
$\frac{2\cdot3}3 = 2$; met $\cos\frac\pi5 = \frac{1 +
\sqrt5}{4}$ berekent men $\cot^2 36^\circ \approx 1.894$ en
$\cot^2 72^\circ \approx 0.106$: som $2.000$.

\textbf{15.} De getallen $\tan^2\theta_k = \frac1{x_k}$ zijn de
wortels van $Q(x) = x^nP_n\bigl(\frac1x\bigr) =
\sum_{j=0}^n(-1)^j\binom{2n+1}{2j+1}x^j$ (de $x_k$ zijn
niet-nul). Vieta op $Q$: de kopcoëfficiënt is $(-1)^n$ (term $j
= n$), de volgende is $(-1)^{n-1}\binom{2n+1}{2n-1} =
(-1)^{n-1}\binom{2n+1}{2}$, dus
\[
\sum_{k=1}^n\tan^2\theta_k =
-\frac{(-1)^{n-1}\binom{2n+1}2}{(-1)^n} = \binom{2n+1}2\cdot
\frac{2}{2n+1}\cdot\frac{2n+1}{2}
= n(2n+1).
\]
Controle $n = 1$: $\tan^2\frac\pi3 = 3 = 1\cdot3$.

\textbf{16.} Zij $\gamma = \frac{1+\alpha}2 \in
\intoo{1}{\alpha}$. Dan $\frac{n^{-\alpha}(\ln
n)^{-\beta}}{n^{-\gamma}} = n^{\gamma - \alpha}(\ln n)^{-\beta}
\to 0$ (een negatieve macht van $n$ verslaat elke macht van
$\ln n$), dus uiteindelijk zijn de termen $\leq n^{-\gamma}$
met $\gamma > 1$: convergentie door vergelijking met een
Riemann-reeks.

\textbf{17.} Zij $\gamma = \frac{1+\alpha}2 \in
\intoo{\alpha}{1}$: nu $\frac{n^{-\gamma}}{n^{-\alpha}(\ln
n)^{-\beta}} = n^{\alpha-\gamma}(\ln n)^{\beta} \to 0$, dus
uiteindelijk zijn de termen $\geq n^{-\gamma}$ met $\gamma <
1$: divergentie.

\textbf{18.} $f(t) = \frac1{t(\ln t)^\beta}$ is positief,
continu en dalend voor grote $t$ (haar logaritme heeft
afgeleide $-\frac1t\bigl(1 + \frac{\beta}{\ln t}\bigr) < 0$
uiteindelijk). Primitieven: voor $\beta \neq 1$,
$\int^x f = \frac{(\ln x)^{1-\beta}}{1-\beta} + \text{const}$,
die een eindige limiet heeft d.e.s.d.a.\ $\beta > 1$; voor
$\beta = 1$, $\int^x f = \ln\ln x \to \infty$. Door
\cref{thm:b2:comparison:seriesintegral} delen de reeks en de
integraal hun aard: convergentie d.e.s.d.a.\ $\beta > 1$.

\textbf{19.} Zelfde test: $\frac{\dd}{\dd t}\ln\ln\ln t =
\frac{1}{t\ln t\,\ln\ln t}$, en $\ln\ln\ln t \to \infty$:
divergentie. En $\frac{\dd}{\dd t}\Bigl(-\frac1{\ln\ln t}\Bigr)
= \frac{1}{t\ln t\,(\ln\ln t)^2}$ met $-\frac1{\ln\ln t} \to
0$: convergentie.

\textbf{20.} Eerste: $n^{1/\ln n} = \eu^{\ln n/\ln n} = \eu$,
dus de termen zijn precies $\frac{1}{\eu\,n}$: een veelvoud van
de harmonische reeks, \emph{divergent} --- de exponent $1 +
\frac1{\ln n}$ kruipt te snel naar $1$. Tweede: $n^{1/\ln\ln n}
= \eu^{\ln n/\ln\ln n}$, en $\frac{\ln n}{\ln\ln n} \geq
2\ln\ln n$ uiteindelijk, dus $n^{1/\ln\ln n} \geq (\ln n)^2$:
de termen zijn $\leq \frac1{n(\ln n)^2}$, een convergente
Bertrandreeks (vraag 18): \emph{convergent}. De grens loopt
strikt tussen deze twee exponenten.

\textbf{21.} $R_n \downarrow 0$ en
\[
\sqrt{R_n} - \sqrt{R_{n+1}} = \frac{R_n -
R_{n+1}}{\sqrt{R_n} + \sqrt{R_{n+1}}} =
\frac{a_n}{\sqrt{R_n} + \sqrt{R_{n+1}}} \geq
\frac{a_n}{2\sqrt{R_n}},
\]
dus $\sum_n \frac{a_n}{\sqrt{R_n}} \leq 2\sum_n(\sqrt{R_n} -
\sqrt{R_{n+1}}) = 2\sqrt{R_1} < \infty$ (telescoperend). Toch
$\frac{a_n/\sqrt{R_n}}{a_n} = \frac1{\sqrt{R_n}} \to \infty$:
de nieuwe reeks convergeert terwijl zij oneindig groter is.
Geen convergente reeks is de traagste; vergelijkingscriteria
tegen enige vaste familie kunnen nooit \omterm{def:b2:metric:complete}{volledig} zijn.

\textbf{22.} Voor dalende positieve $(a_n)$, groepeer de termen
tussen opeenvolgende machten van $2$:
\[
2^{k}a_{2^{k+1}} \leq \sum_{n=2^k}^{2^{k+1}-1} a_n \leq
2^ka_{2^k} .
\]
Sommeren over $k$: als $\sum 2^ka_{2^k}$ convergeert, zijn de
partiële sommen van $\sum a_n$ begrensd (convergeert); als
$\sum a_n$ convergeert, dan $\sum_k 2^{k+1}a_{2^{k+1}} \leq
2\sum_n a_n < \infty$. Voor $a_n = \frac1{n(\ln n)^\beta}$:
$2^ka_{2^k} = \frac{1}{(k\ln 2)^\beta}$, en $\sum k^{-\beta}$
convergeert d.e.s.d.a.\ $\beta > 1$: opnieuw de grens van
vraag 18, zonder integralen.

\textbf{23.} Door de integraalvergelijking,
\[
\sum_{n > N}\frac{1}{n(\ln n)^2} \geq
\int_{N+1}^{\infty}\frac{\dd t}{t(\ln t)^2} =
\frac{1}{\ln(N+1)},
\]
die bij $N = 10^6$ ongeveer $0.0724$ is: na een miljoen termen
overschrijdt de staart nog $0.07$ --- de reeks convergeert, maar
geen directe sommatie zal ooit haar som tentoonspreiden.
Contrasteer met vraag 13, waar één asymptotische correctie
zeven decimalen uit honderd termen kocht: weten \emph{hoe} een
reeks convergeert is meer waard dan weten \emph{dat} zij dat
doet.

\textbf{24.} $\sum\frac1{n\ln n}$: divergeert ($\alpha = 1$,
$\beta = 1$, vraag 18). $\sum\frac1{n^{1.01}}$: convergeert
(Riemann, $\alpha > 1$). $\sum\frac{(\ln n)^{100}}{n^{1.001}}$:
convergeert ($\alpha = 1.001 > 1$, $\beta = -100$, vraag 16).
$\sum\frac1{n\ln n(\ln\ln n)^3}$: convergeert (patroon van
vraag 19: primitieve $-\frac12(\ln\ln t)^{-2}$, eindige
limiet).

\textbf{25.} De Moivre zet het verdwijnen van $\sin(2n{+}1)
\theta_k$ om in het verdwijnen van een veelterm in
$\cot^2\theta_k$, en Vieta leest de exacte wortelsommen af die
analyse alleen maar had kunnen schatten (vragen 1--5). De
insluiting had de \emph{exacte} waarden $\frac{n(2n-1)}3$ en
$\frac{2n(n+1)}3$ aan beide kanten nodig --- equivalenten
zouden de vraag bedelen, want het hele punt is de constante
$\frac{\pi^2}6$ (vragen 7--8). Deel IV draaide \omterm{def:b2:metric:complete}{volledig} op de
reeks--integraalvergelijking van \cref{ch:b2:comparison}, de
logaritmische primitieven die classificeerden (vragen 18--19).
Vraag 21 vernietigt de droom van een universeel
vergelijkingscriterium: onder elke convergente reeks ligt een
andere, oneindig tragere --- schalen zoals die van Bertrand
mappen de grens steeds fijner maar bereiken haar nooit. Toppen:
Eulers $\zeta(2) = \frac{\pi^2}6$ met haar bovenverdieping
$\zeta(4) = \frac{\pi^4}{90}$ (vragen 8, 10), en de
Bertrand-classificatie (vragen 16--18); $\zeta(2)$ keert terug
in \cref{ch:b2:fourier}, waar de identiteit van Parseval haar
in één regel herbewijst uit de Fourierreeks van de zaagtand ---
één constante, twee beschavingen.
\end{solution}
