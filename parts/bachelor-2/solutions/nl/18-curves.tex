\section*{Hoofdstuk \ref{ch:b2:curves} --- Krommen}

\begin{solution}{exo:b2:curves:1}
$\gamma'(t) = (1 - \cos t,\ \sin t)$, dus
\[
\norm{\gamma'(t)}^2 = (1 - \cos t)^2 + \sin^2 t
= 2 - 2\cos t = 4\sin^2\tfrac t2 ,
\]
en $\norm{\gamma'(t)} = 2\sin\frac t2$ (niet-negatief op $[0,
2\pi]$). Vandaar
\[
L = \int_0^{2\pi} 2\sin\tfrac t2\,\dd t
= \Bigl[-4\cos\tfrac t2\Bigr]_0^{2\pi} = 8 :
\]
één boog van de cycloïde heeft \omterm{def:b2:curves:length}{lengte} $8$ (voor een wiel van
straal $1$) --- een beroemd resultaat van Wren, zonder $\pi$ in
zicht.
\end{solution}

\begin{solution}{exo:b2:curves:2}
Met $x = a\cos t$, $y = b\sin t$: $x' = -a\sin t$, $y' = b\cos
t$, $x'' = -a\cos t$, $y'' = -b\sin t$, dus door
\cref{prop:b2:curves:kappaformula}
\[
\kappa(t)
= \frac{x'y'' - y'x''}{(x'^2 + y'^2)^{3/2}}
= \frac{ab\sin^2 t + ab\cos^2 t}
       {(a^2\sin^2 t + b^2\cos^2 t)^{3/2}}
= \frac{ab}{(a^2\sin^2 t + b^2\cos^2 t)^{3/2}} .
\]
De noemer is minimaal wanneer $\sin t = 0$ (waarde $b^3$,
punten $(\pm a, 0)$) en maximaal wanneer $\cos t = 0$ (waarde
$a^3$, punten $(0, \pm b)$), omdat $a > b$. Dus is $\kappa$
maximaal in de einden van de grote as, $\kappa_{\max} =
a/b^2$, en minimaal in de einden van de kleine as,
$\kappa_{\min} = b/a^2$: de ellips buigt het scherpst in de
uiteinden van haar lange as.
\end{solution}

\begin{solution}{exo:b2:curves:3}
Voor de grafiek $\gamma(x) = (x, \cosh x)$: $\norm{\gamma'(x)}
= \sqrt{1 + \sinh^2 x} = \cosh x$, dus is de \omterm{def:b2:curves:length}{booglengte} van
$0$ tot $x$ gelijk aan $\int_0^x \cosh u\,\dd u = \sinh x$.
\omterm{thm:b2:curves:frenet2d}{Kromming} van een grafiek (\cref{prop:b2:curves:kappaformula}):
\[
\kappa(x) = \frac{f''(x)}{(1 + f'(x)^2)^{3/2}}
= \frac{\cosh x}{\cosh^3 x} = \frac{1}{\cosh^2 x} ,
\]
dus $R(x) = \cosh^2 x$, zoals aangekondigd. Merk de nette
toeval op $R(x) = 1 + s(x)^2$ met $s = \sinh x$ de
\omterm{def:b2:curves:length}{booglengte}: de \omterm{def:b2:curves:curvature}{kromtestraal} van de kettinglijn groeit met het
kwadraat van de \omterm{def:b2:curves:length}{booglengte} vanaf de top.
\end{solution}

\begin{solution}{exo:b2:curves:4}
$\gamma'(t) = e^t(\cos t - \sin t,\ \sin t + \cos t)$, dus
\[
\langle \gamma(t), \gamma'(t)\rangle = e^{2t}
\bigl(\cos t(\cos t - \sin t) + \sin t(\sin t + \cos t)\bigr)
= e^{2t},
\]
terwijl $\norm{\gamma(t)} = e^t$ en $\norm{\gamma'(t)} =
e^t\sqrt 2$. Vandaar
\[
\cos\angle\bigl(\gamma, \gamma'\bigr)
= \frac{e^{2t}}{e^t \cdot e^t\sqrt2} = \frac{1}{\sqrt2} :
\]
de \omterm{def:b2:curves:arc}{raaklijn} maakt altijd de hoek $\pi/4$ met de straal --- de
\emph{equiangular}e eigenschap van de logaritmische spiraal.
\omterm{def:b2:curves:length}{Booglengte} op $(-\infty, 0]$:
\[
\int_{-\infty}^0 \norm{\gamma'(t)}\,\dd t
= \sqrt2\int_{-\infty}^0 e^t\,\dd t = \sqrt 2 ,
\]
eindig hoewel de spiraal oneindig vaak om de oorsprong
draait. \omterm{thm:b2:curves:frenet2d}{Kromming}: met $x'y'' - y'x''$ berekend uit $\gamma''
= e^t(-2\sin t,\ 2\cos t)$,
\[
x'y'' - y'x'' = e^{2t}\bigl(2\cos t(\cos t - \sin t)
+ 2\sin t(\sin t + \cos t)\bigr) = 2e^{2t},
\]
dus $\kappa(t) = \dfrac{2e^{2t}}{(e^t\sqrt2)^3} =
\dfrac{1}{e^t\sqrt2}$: de \omterm{thm:b2:curves:frenet2d}{kromming} is $1/(\sqrt2\,
\norm{\gamma})$, afnemend naarmate de spiraal groeit.
\end{solution}

\begin{solution}{exo:b2:curves:5}
\emph{Eerste boog:} $\gamma(t) = (t^2,\ t^4 + t^5)$.
Afgeleiden in $0$: $\gamma'' = (2, 0) \neq 0$, dus $p = 2$.
Dan $\gamma^{(3)}(0) = (0, 0)$, $\gamma^{(4)}(0) = (0, 24)$,
niet collineair met $(2, 0)$: $q = 4$. Beide even:
\emph{keerpunt van de tweede soort} --- beide takken
vertrekken in de richting $+u = (1,0)$ en blijven aan
dezelfde kant van de \omterm{def:b2:curves:arc}{raaklijn}. (Inderdaad $y = x^2 \pm
x^{5/2}$ op de twee takken: zelfde teken voor kleine $x$.)

\emph{Tweede boog:} $\gamma(t) = (t^3, t^4)$. $\gamma'(0) =
\gamma''(0) = 0$, $\gamma^{(3)}(0) = (6, 0)$: $p = 3$,
oneven. Volgende $\gamma^{(4)}(0) = (0, 24)$: $q = 4$, even.
Oneven--even: \emph{ordinair punt} --- ondanks de
verdwijnende snelheid kruist de baan $y = x^{4/3}$ de
oorsprong glad, boven haar \omterm{def:b2:curves:arc}{raaklijn} $y = 0$ blijvend.
\end{solution}

\begin{solution}{exo:b2:curves:6}
Differentiëer $c(s) = \gamma(s) + \dfrac{1}{\kappa(s)}N(s)$
met de vlakke Frenet-formules
(\cref{thm:b2:curves:frenet2d}):
\[
c'(s) = T(s) - \frac{\kappa'(s)}{\kappa(s)^2}N(s)
+ \frac{1}{\kappa(s)}\,\bigl(-\kappa(s)T(s)\bigr)
= -\frac{\kappa'(s)}{\kappa(s)^2}\,N(s) ,
\]
de raaktermen heffen precies op. Dus wordt de snelheid van de
evolute gedragen door $N(s)$, die de normaallijn van $\gamma$
in $\gamma(s)$ richt --- en het punt $c(s)$ ligt op precies
die normaallijn: de evolute is de \emph{\omterm{pb:b2:curves:1}{omhullende} van de
normalen}. (Waar $\kappa' = 0$ heeft de evolute een
singulier punt; dat produceert de keerpunten van de evolute
van een ellips.)
\end{solution}

\begin{solution}{exo:b2:curves:7}
Zij $G(s) = F(s)F(s)^{\mathsf T}$. Dan, met $F' = FA$ en
$(F^{\mathsf T})' = (F')^{\mathsf T} = A^{\mathsf T}F^{\mathsf T}$,
\[
G' = F'F^{\mathsf T} + F(F^{\mathsf T})'
= FAF^{\mathsf T} + FA^{\mathsf T}F^{\mathsf T}
= F(A + A^{\mathsf T})F^{\mathsf T} = 0
\]
door antisymmetrie. Dus is $G$ constant op het interval,
gelijk aan $G(s_0) = F(s_0)F(s_0)^{\mathsf T} = I$: $F(s)$ is
orthogonaal voor elke $s$. (Alternatief, zonder $G'$ tot nul
te rekenen: zowel $G$ als de constante $I$ lossen het
lineaire stelsel $Y' = YA + A^{\mathsf T}Y$ op met dezelfde
beginwaarde, en Cauchy--Lipschitz-uniciteit voor lineaire
stelsels, \cref{ch:b2:diffeq}, dwingt $G \equiv I$.)

\emph{Relevantie:} het Frenet-stelsel $(T, N, B)' = (T, N,
B)\,A(s)$ heeft de antisymmetrische coëfficiëntenmatrix
\[
A = \begin{pmatrix} 0 & -\kappa & 0\\ \kappa & 0 & -\tau\\
0 & \tau & 0\end{pmatrix}
\]
(kolommen die $T', N', B'$ uitdrukken). De berekening
hierboven toont dat een oplossingsraam dat orthonormaal
start orthonormaal \emph{blijft} --- de sleutelstap in de
fundamentele stelling die een kromme uit $(\kappa, \tau)$
reconstrueert.
\end{solution}

\begin{solution}{exo:b2:curves:8}
Door \cref{thm:b2:curves:fundamental} (uniciteitsdeel) is er
een $\mathcal{C}^1$-hoekfunctie $\varphi$ met $T(s) =
(\cos\varphi(s), \sin\varphi(s))$ en $\varphi' = \kappa$.
Vandaar
\[
\int_0^L \kappa(s)\,\dd s = \varphi(L) - \varphi(0) .
\]
Omdat de boog gesloten is van periode $L$, $T(L) = T(0)$:
$(\cos\varphi(L), \sin\varphi(L)) = (\cos\varphi(0),
\sin\varphi(0))$, dus $\varphi(L) - \varphi(0) \in 2\pi\Z$.
De totale \omterm{thm:b2:curves:frenet2d}{kromming} van een gesloten kromme is dus altijd een
geheel veelvoud van $2\pi$ --- het gehele getal is het
\emph{windingsgetal} van de tangens (het aantal volle
omwentelingen dat $T$ maakt). Voor de cirkel van straal $R$:
$\kappa = 1/R$ en $L = 2\pi R$, totale \omterm{thm:b2:curves:frenet2d}{kromming} $2\pi$,
windingsgetal $1$; de stelling van de draaiende tangens
beweert dat deze waarde voor elke eenvoudige gesloten kromme
geldt.
\end{solution}

\begin{solution}{exo:b2:curves:9}
Omdat $\tau \equiv 0$ ligt de kromme in een vlak
(\cref{prop:b2:curves:torsion}); werk in dat vlak. Beschouw
het kandidaat-centrum
\[
c(s) = \gamma(s) + \frac{1}{\kappa}N(s)
\qquad (\kappa \text{ constant}).
\]
Differentiëren met de Frenet-formules ($N' = -\kappa T + \tau B
= -\kappa T$ hier):
\[
c'(s) = T + \frac1\kappa(-\kappa T) = 0 ,
\]
dus is $c$ een constant punt $\Omega$. Dan $\norm{\gamma(s) -
\Omega} = \norm{-\frac1\kappa N(s)} = \frac1\kappa$ voor alle
$s$: de kromme ligt op de cirkel met centrum $\Omega$ en
straal $1/\kappa$ (in haar vlak), en is als niet-constante
boog ervan een boog van die cirkel.
\end{solution}

\begin{solution}{exo:b2:curves:10}
Voor de grafiek $\gamma(x) = (x, x^2/2)$, $\norm{\gamma'(x)}
= \sqrt{1 + x^2}$, dus $L = \int_0^a\sqrt{1+x^2}\,\dd x$.
Substitueren $x = \sinh u$ ($\dd x = \cosh u\,\dd u$, $u$ van
$0$ tot $u_a = \ln(a + \sqrt{1+a^2})$):
\[
L = \int_0^{u_a}\cosh^2 u\,\dd u
= \frac12\bigl[u + \sinh u\cosh u\bigr]_0^{u_a}
= \frac12\Bigl(\ln\bigl(a + \sqrt{1+a^2}\bigr) +
a\sqrt{1+a^2}\Bigr),
\]
met $\cosh^2 u = \frac{1 + \cosh 2u}2$ en $\sinh u_a = a$,
$\cosh u_a = \sqrt{1 + a^2}$.
\end{solution}

\begin{solution}{exo:b2:curves:11}
Parametriseer op \omterm{def:b2:curves:length}{booglengte}
(\cref{thm:b2:curves:arclength}) en stel $\lambda(s) =
\langle P - \gamma(s), T(s)\rangle$, een $\mathcal
C^1$-functie; omdat $P$ op de \omterm{def:b2:curves:arc}{raaklijn} in $\gamma(s)$ ligt,
is de vector $P - \gamma(s)$ collineair met $T(s)$, dus $P =
\gamma(s) + \lambda(s)T(s)$. Differentiëren,
\[
0 = T(s) + \lambda'(s)T(s) + \lambda(s)T'(s)
= \bigl(1 + \lambda'(s)\bigr)T(s) + \lambda(s)T'(s),
\]
en $T'(s) \perp T(s)$ (differentiëer $\norm T^2 = 1$), dus
verdwijnen beide componenten: $\lambda' = -1$ en $\lambda T'
= 0$. Dan verdwijnt $\lambda(s) = c - s$ ten hoogste één
keer, dus $T' = 0$ op een dichte verzameling, dus overal door
continuïteit: $T$ is een constante eenheidsvector en
$\gamma(s) = \gamma(s_0) + (s - s_0)T$: een rechte lijn
(door $P$, zoals het moet).
\end{solution}

\begin{solution}{exo:b2:curves:12}
(a) De Frenet-matrix
\[
A(s) = \begin{pmatrix} 0 & -\kappa & 0\\ \kappa & 0 & -\tau\\
0 & \tau & 0\end{pmatrix}
\]
heeft continue entries, dus heeft het lineaire stelsel $F' =
FA(s)$, $F(s_0) = F_0$ (een directe orthonormale matrix) een
unieke oplossing op heel $J$ (\cref{thm:b2:diffeq:linear}).
Door \cref{exo:b2:curves:7} is $F(s)$ orthogonaal voor elke
$s$; $\det F$ is continu met waarden in $\{\pm1\}$ en gelijk
aan $1$ in $s_0$, dus is $F(s)$ direct voor alle $s$.

(b) Lees de rijen $T, N, B$ van $F$ af (zodat $T' = \kappa
N$, $N' = -\kappa T + \tau B$, $B' = -\tau N$) en stel
$\gamma(s) = \gamma_0 + \int_{s_0}^s T(u)\,\dd u$. Dan is
$\gamma' = T$ een eenheidsvector: eenheidssnelheid; $T' =
\kappa N$ met $\kappa > 0$ en $N$ eenheid orthogonaal op $T$,
dus is $\gamma$ biregulair met \omterm{thm:b2:curves:frenet2d}{kromming} $\norm{T'} = \kappa$
en hoofdnormaal $N$; de binormaal is $T \wedge N = B$
(direct orthonormaal raam), en $B' = -\tau N$ identificeert
de \omterm{thm:b2:curves:frenet3d}{torsie} als $\tau$.

(c) Zij $\gamma_1, \gamma_2$ eenheidssnelheids biregulaire
krommen met dezelfde $(\kappa, \tau)$. Er is een unieke
directe isometrie $\Phi = \rho + w$ ($\rho \in SO(3)$) die
$\gamma_1(s_0)$ naar $\gamma_2(s_0)$ stuurt en het
Frenet-raam van $\gamma_1$ in $s_0$ naar dat van $\gamma_2$
in $s_0$. De kromme $\Phi\circ\gamma_1$ is eenheidssnelheid
met dezelfde invarianten (haar raam is $\rho$ toegepast op
dat van $\gamma_1$, en $\rho$ behoudt kruisproducten, omdat
ze direct is). Nu lossen de ramen van $\Phi\circ\gamma_1$ en
$\gamma_2$ beide $F' = FA(s)$ op met dezelfde beginwaarde,
dus vallen ze samen door uniciteit; in het bijzonder komen
de tangenten overeen, en integreren vanaf het gemeenschappelijke
punt $s_0$: $\Phi\circ\gamma_1 = \gamma_2$.
\end{solution}

\begin{solution}{pb:b2:curves:1}
\textbf{1.} Het stelsel is lineair in $(x, y)$ met \omterm{def:b2:linalg:det}{determinant}
$\Delta(t) = a b' - a'b \neq 0$: de regel van Cramer geeft de
unieke oplossing
\[
x = \frac{c b' - c' b}{\Delta}, \qquad
y = \frac{a c' - a' c}{\Delta}.
\]

\textbf{2.} Omdat $a(t)x(t) + b(t)y(t) = c(t)$ identiek,
geeft differentiëren $a'x + b'y + ax' + by' = c'$; de tweede
karakteristieke vergelijking doodt $a'x + b'y - c'$, dus
$a(t)x'(t) + b(t)y'(t) = 0$: $E'(t)$ is orthogonaal op $(a,
b)$, dus evenwijdig aan $(-b, a)$, de richting van $D_t$.
Omdat $E(t) \in D_t$ (eerste vergelijking) en $E'(t) \neq 0$,
is de lijn $D_t$ precies de \omterm{def:b2:curves:arc}{raaklijn} van de kromme $E$ in
$E(t)$.

\textbf{3.} Hier $(a, b, c) = (t, -1, t^2/2)$, dus $\Delta =
t\cdot0 - 1\cdot(-1) = 1$ en
\[
x = \frac{c b' - c' b}{\Delta} = \tfrac{t^2}2\cdot 0 +
t = t, \qquad
y = \frac{a c' - a' c}{\Delta} = t\cdot t - \tfrac{t^2}2 =
\tfrac{t^2}2 :
\]
de \omterm{pb:b2:curves:1}{omhullende} van de \omterm{def:b2:curves:arc}{raaklijnen} van de parabool is de
parabool, zoals het hoort.

\textbf{4.} De normaallijn is $\langle M, T(s)\rangle =
\langle\gamma(s), T(s)\rangle$: coëfficiënten $a = T_1$, $b =
T_2$, $c = \langle\gamma, T\rangle$. Differentiëren met
Frenet ($T' = \kappa N$): $a' = \kappa N_1$, $b' = \kappa
N_2$, en $c' = \langle T, T\rangle + \langle\gamma, \kappa
N\rangle = 1 + \kappa\langle\gamma, N\rangle$. De tweede
karakteristieke vergelijking $\kappa\langle M, N\rangle = 1 +
\kappa\langle\gamma, N\rangle$ leest $\kappa\langle M -
\gamma, N\rangle = 1$. De eerste zegt $M - \gamma \perp T$,
dus $M - \gamma = \mu N$ met $\mu = 1/\kappa$: het
karakteristieke punt is $\gamma + \frac1\kappa N$, het
\omterm{def:b2:curves:curvature}{kromtemiddelpunt}, en de \omterm{pb:b2:curves:1}{omhullende} van de normalen is de
evolute van \cref{exo:b2:curves:6}. Ten slotte $\Delta =
T_1\kappa N_2 - \kappa N_1 T_2 = \kappa\det(T, N) = \kappa
\neq 0$.

\textbf{5.} Bundel: het stelsel $x\cos\theta + y\sin\theta =
0$, $-x\sin\theta + y\cos\theta = 0$ heeft \omterm{def:b2:linalg:det}{determinant} $1$ en
oplossing $(0,0)$ voor elke $\theta$: $E \equiv (0,0)$, $E'
\equiv 0$, en er is geen kromme --- slechts het
gemeenschappelijke punt van alle lijnen. Evenwijdige
familie: $a' = b' = 0$ geeft $\Delta \equiv 0$ en de tweede
vergelijking $0 = c'(t)$, die faalt zodra de familie echt
beweegt: geen karakteristiek punt, en inderdaad raakt een
familie evenwijdige lijnen geen kromme langs al haar leden.

\textbf{6.} De lijn door $(\cos t, 0)$ en $(0, \sin t)$ is
$\frac x{\cos t} + \frac y{\sin t} = 1$, d.w.z.\ $x\sin t +
y\cos t = \sin t\cos t$. Met $(a, b, c) = (\sin t, \cos t,
\sin t\cos t)$: $a' = \cos t$, $b' = -\sin t$, $c' = \cos
2t$, $\Delta = -\sin^2 t - \cos^2 t = -1$. Cramer:
\begin{align*}
x &= \frac{cb' - c'b}{-1} = \sin^2 t\cos t + \cos 2t\cos t
= \cos t\,(\sin^2 t + \cos^2 t - \sin^2 t) = \cos^3 t,\\
y &= \frac{ac' - a'c}{-1} = \sin t\cos^2 t - \sin t\cos 2t
= \sin t\,(\cos^2 t - \cos^2 t + \sin^2 t) = \sin^3 t .
\end{align*}
En $(\cos^3t)^{2/3} + (\sin^3t)^{2/3} = 1$: de \omterm{pb:b2:curves:1}{astroïde}.

\textbf{7.} $E'(t) = 3(-\cos^2 t\sin t,\ \sin^2 t\cos t)$
verdwijnt in $t = 0$. Daar geeft $E''(0) = (-3, 0) \neq 0$
$p = 2$; de $x$-component van $E$ is even in $t$, dus
$E'''(0) = (0, 6)$, niet collineair: $q = 3$. Even--oneven:
keerpunt van de eerste soort in $(1, 0)$
(\cref{prop:b2:curves:local}), met \omterm{def:b2:curves:arc}{raaklijn} langs de
$x$-as. De symmetrieën $x \mapsto -x$, $y \mapsto -y$, $(x,
y) \mapsto (y, x)$ van de \omterm{pb:b2:curves:1}{astroïde} transporteren het
keerpunt naar $(-1, 0)$ en $(0, \pm1)$.

\textbf{8.} $E'(t) = 3\sin t\cos t\,(-\cos t, \sin t)$, dus
$\norm{E'(t)} = 3\abs{\sin t\cos t} = \tfrac32\abs{\sin 2t}$.
Eén kwadrant: $\int_0^{\pi/2}\tfrac32\sin 2t\,\dd t =
\tfrac32$, en door symmetrie is de totale \omterm{def:b2:curves:length}{lengte} $4 \cdot
\tfrac32 = 6$.

\textbf{9.} $E(t) - P_t = (\cos^3 t - \cos t,\ \sin^3 t) =
\sin^2 t\,(-\cos t,\ \sin t) = \sin^2 t\,(Q_t - P_t)$. Dus is
het raakpunt het \omterm{def:b2:affine:barycenter}{barycentrum} van $(P_t, \cos^2 t)$ en $(Q_t,
\sin^2 t)$: terwijl $t$ van $0$ tot $\pi/2$ loopt glijdt ze
van het vloereinde naar het muureinde van de ladder.

\textbf{10.} In het eerste kwadrant heeft het gebied onder
de \omterm{pb:b2:curves:1}{astroïde} oppervlakte $\int_0^1 y\,\dd x$ met $x = \cos^3
t$ dalend van $1$ tot $0$ terwijl $t$ van $0$ tot $\pi/2$
loopt:
\[
\int_0^1 y\,\dd x
= \int_{\pi/2}^{0}\sin^3 t\,(-3\cos^2 t\sin t)\,\dd t
= 3\int_0^{\pi/2}\sin^4 t\cos^2 t\,\dd t .
\]
Lineariseren: $\sin^4 t\cos^2 t = (\sin t\cos t)^2\sin^2 t =
\tfrac18\bigl(\sin^2 2t - \sin^2 2t\cos 2t\bigr)$, en
$\int_0^{\pi/2}\sin^2 2t\,\dd t = \tfrac\pi4$ terwijl
$\int_0^{\pi/2}\sin^2 2t\cos 2t\,\dd t =
\bigl[\tfrac{\sin^3 2t}6\bigr]_0^{\pi/2} = 0$. Dus is de
integraal $\tfrac\pi{32}$, de kwadrantoppervlakte
$\tfrac{3\pi}{32}$, en de ingesloten oppervlakte $4 \cdot
\tfrac{3\pi}{32} = \tfrac{3\pi}8$.

\textbf{11.} De \omterm{def:b2:curves:arc}{raaklijn} in $\gamma(t) = (t, t^2/2)$ is
gericht door $(1, t)$, dus is de normaallijn $\{(x, y) : (x -
t) + t\,(y - t^2/2) = 0\}$, d.w.z.\ $x + t\,y = t +
\tfrac{t^3}2$.

\textbf{12.} $(a, b, c) = (1, t, t + t^3/2)$: $a' = 0$, $b' =
1$, $c' = 1 + \tfrac32 t^2$, $\Delta = 1$. Dan $y = ac' -
a'c = 1 + \tfrac32t^2$ en
\[
x = cb' - c'b = t + \tfrac{t^3}2 - t\Bigl(1 +
\tfrac32t^2\Bigr) = -t^3 .
\]
Elimineren van $t$: $t^2 = \tfrac23(y - 1)$ en $x^2 = t^6 =
\tfrac8{27}(y - 1)^3$: een semicubische parabool met top
$(0, 1)$.

\textbf{13.} $T = (1, t)/\sqrt{1+t^2}$, $N = (-t,
1)/\sqrt{1+t^2}$, en $\kappa = (1+t^2)^{-3/2}$, dus
\[
\gamma + \frac1\kappa N = (t, \tfrac{t^2}2) +
(1+t^2)\,(-t, 1) = \Bigl(-t^3,\ 1 + \tfrac32t^2\Bigr),
\]
dezelfde kromme: de \omterm{pb:b2:curves:1}{omhullende} van de normalen is de
meetkundige plaats van de \omterm{def:b2:curves:curvature}{kromtemiddelpunten}, zoals vraag 4
beloofde.

\textbf{14.} $E'(t) = (-3t^2, 3t)$ verdwijnt in $t = 0$;
$E''(0) = (0, 3) \neq 0$ geeft $p = 2$ en $E'''(0) = (-6,
0)$ geeft $q = 3$: een keerpunt van de eerste soort in $(0,
1)$. De top is waar $\kappa = (1+t^2)^{-3/2}$ maximaal is,
dus $\kappa'(0) = 0$ en de snelheid van de evolute
$-\frac{\kappa'}{\kappa^2}N$ verdwijnt precies daar:
keerpunten van de evolute zitten in de extrema van
\omterm{thm:b2:curves:frenet2d}{kromming}.

\textbf{15.} De normaal bij parameter $t$ gaat door $(x_0,
y_0)$ dan en slechts dan als $x_0 + t\,y_0 = t +
\tfrac{t^3}2$, d.w.z.
\[
\frac{t^3}2 + (1 - y_0)\,t - x_0 = 0 ,
\]
een kubische in $t$: één of drie reële wortels (geteld
zonder multipliciteit, voor generieke punten). Op de as
$x_0 = 0$ factoriseert ze als $t\bigl(\tfrac{t^2}2 + 1 -
y_0\bigr) = 0$: de wortel $t = 0$ (de as is de normaal in de
top), plus $t = \pm\sqrt{2(y_0 - 1)}$ wanneer $y_0 > 1$.
Dus: één normaal voor $y_0 < 1$, drie voor $y_0 > 1$, en in
$y_0 = 1$ markeert de drievoudige wortel het keerpunt van de
evolute. In het algemeen betekent een dubbele wortel van de
kubische dat het punt zowel aan de lijnvergelijking als aan
haar $t$-afgeleide voldoet --- ze ligt \emph{op de
\omterm{pb:b2:curves:1}{omhullende}}: de evolute is precies de grens tussen de
één-normaal- en drie-normaalregio's.

\textbf{16.} Reflectie in de spiegel keert de
normaalcomponent van de richting om en houdt de tangentiële:
schrijvend $u = \langle u, n\rangle n + u_{\mathrm{tan}}$,
is de gereflecteerde richting $u_{\mathrm{tan}} - \langle u,
n\rangle n = u - 2\langle u, n\rangle n$. Hier $\langle u,
n\rangle = \cos\theta$, dus
\[
v = (1, 0) - 2\cos\theta\,(\cos\theta, \sin\theta)
= (1 - 2\cos^2\theta,\ -2\sin\theta\cos\theta)
= -(\cos2\theta,\ \sin2\theta).
\]

\textbf{17.} De gereflecteerde straal gaat door $P_\theta =
(\cos\theta, \sin\theta)$ met richting $(\cos2\theta,
\sin2\theta)$; een normaalvector is $(-\sin2\theta,
\cos2\theta)$, dus is de lijn
\[
-\sin2\theta\,(x - \cos\theta) + \cos2\theta\,(y -
\sin\theta) = 0,
\]
en de constante is $-\sin2\theta\cos\theta +
\cos2\theta\sin\theta = -\sin\theta$: vermenigvuldigen met
$-1$, $x\sin2\theta - y\cos2\theta = \sin\theta$.

\textbf{18.} $(a, b, c) = (\sin2\theta, -\cos2\theta,
\sin\theta)$: $a' = 2\cos2\theta$, $b' = 2\sin2\theta$, $c' =
\cos\theta$, $\Delta = 2\sin^22\theta + 2\cos^22\theta = 2$.
Cramer, daarna product-naar-somformules:
\begin{align*}
x &= \frac{2\sin\theta\sin2\theta +
\cos\theta\cos2\theta}{2}
= \frac{(\cos\theta - \cos3\theta) + \frac12(\cos\theta +
\cos3\theta)}{2} = \frac{3\cos\theta - \cos3\theta}4,\\
y &= \frac{\sin2\theta\cos\theta - 2\cos2\theta\sin\theta}2
= \frac{\frac12(\sin3\theta + \sin\theta) - (\sin3\theta -
\sin\theta)}2 = \frac{3\sin\theta - \sin3\theta}4 :
\end{align*}
de \omterm{pb:b2:curves:1}{nefroïde}, een gesloten kromme met twee keerpunten.

\textbf{19.} Differentiëren en factoriseren met $\sin3\theta
- \sin\theta = 2\cos2\theta\sin\theta$, $\cos\theta -
\cos3\theta = 2\sin2\theta\sin\theta$:
\[
E'(\theta) = \tfrac34\bigl(\sin3\theta - \sin\theta,\
\cos\theta - \cos3\theta\bigr)
= \tfrac32\sin\theta\,(\cos2\theta, \sin2\theta),
\]
evenwijdig aan de gereflecteerde richting van vraag 16: elke
gereflecteerde straal is raak aan de caustiek, zoals de
omhullende-eigenschap eist. $E' = 0$ precies wanneer
$\sin\theta = 0$: $E(0) = (\tfrac12, 0)$ en $E(\pi) =
(-\tfrac12, 0)$, de twee keerpunten. Stellen $y = 0$ in de
lijnvergelijking: $x\sin2\theta = \sin\theta$, dus $x =
\frac1{2\cos\theta} \to \frac12$ als $\theta \to 0$:
paraxiale stralen focussen op afstand $R/2$ van het centrum
--- de brandpuntsafstand van de bolvormige spiegel.

\textbf{20.} $\norm{E'(\theta)} = \tfrac32\abs{\sin\theta}$,
dus is de \omterm{def:b2:curves:length}{lengte} $\tfrac32\int_0^{2\pi}\abs{\sin\theta}\,
\dd\theta = \tfrac32\cdot4 = 6$. Nabij $\theta = 0$, met
$\cos k\theta = 1 - \tfrac{k^2\theta^2}2 + O(\theta^4)$ en
$\sin k\theta = k\theta - \tfrac{k^3\theta^3}6 +
O(\theta^5)$:
\[
x - \tfrac12 = \frac{3\theta^2 + O(\theta^4)}{4}
= \tfrac34\theta^2 + O(\theta^4),
\qquad
y = \frac{4\theta^3 + O(\theta^5)}{4}
= \theta^3 + O(\theta^5) :
\]
$p = 2$, $q = 3$, een keerpunt van de eerste soort wijzend
langs de as --- het heldere punt van de koffiekop-caustiek.

\textbf{21.} Parametriseer de belichte punten door
$\Phi(\theta, r) = P_\theta + r\,v_\theta$ (positie langs
elke gereflecteerde straal). De Jacobiaandeterminant
$\det(\partial_\theta\Phi, \partial_r\Phi) =
\det(P_\theta' + rv_\theta', v_\theta)$ is affien in $r$ en
verdwijnt voor precies één $r = r_*(\theta)$ --- en
$\Phi(\theta, r_*(\theta))$ is het karakteristieke punt,
omdat daar de straalrichting en de variatie van de familie
afhankelijk worden. Buiten de \omterm{pb:b2:curves:1}{omhullende} is de afbeelding
een lokaal diffeomorfisme, en een punt net binnen de
caustiek wordt door twee nabije stralen getroffen (twee
oplossingen $\theta$), een punt erbuiten door geen van dat
deel van de familie; op de caustiek smelten de twee samen.
Lichtintensiteit is omgekeerd evenredig met de absolute
waarde van de Jacobiaan, dus blaast ze op langs de
\omterm{pb:b2:curves:1}{omhullende}: de caustiek is de heldere kromme, het helderst
van al in het keerpunt, waar de ontaarding het ergst is.

\textbf{22.} $x' = 1 - \cos t$, $y' = \sin t$, $x'' = \sin
t$, $y'' = \cos t$, dus $x'y'' - y'x'' = \cos t - 1$ en
$\norm{\gamma'}^2 = 2(1 - \cos t) = 4\sin^2\tfrac t2$;
door \cref{prop:b2:curves:kappaformula},
\[
\kappa(t) = \frac{-(1 - \cos t)}{8\sin^3\tfrac t2}
= \frac{-2\sin^2\tfrac t2}{8\sin^3\tfrac t2}
= -\frac1{4\sin\tfrac t2}
\qquad (0 < t < 2\pi).
\]
Met $T = (\sin\tfrac t2, \cos\tfrac t2)$ (deel $\gamma'$
door $2\sin\tfrac t2$) en $N = (-\cos\tfrac t2, \sin\tfrac
t2)$:
\[
\gamma + \frac1\kappa N
= \gamma - 4\sin\tfrac t2\,\Bigl(-\cos\tfrac t2,\
\sin\tfrac t2\Bigr)
= \bigl(t - \sin t + 2\sin t,\ 1 - \cos t - 2(1 - \cos
t)\bigr),
\]
d.w.z.\ $c(t) = (t + \sin t,\ \cos t - 1)$. Substitueren $t
= u + \pi$:
\[
c = \bigl(u + \pi - \sin u,\ -\cos u - 1\bigr)
= \bigl((u - \sin u) + \pi,\ (1 - \cos u) - 2\bigr) :
\]
de cycloïde $\gamma(u)$ getransleerd over $(\pi, -2)$. De
evolute van een cycloïde is een congruente cycloïde, hangend
één niveau lager.

\textbf{23.} $R(t) = 1/\abs{\kappa(t)} = 4\sin\tfrac t2$, dus
$R(\pi) = 4$: de helft van de \omterm{def:b2:curves:length}{booglengte} $8$ berekend in
\cref{exo:b2:curves:1}. De snelheid van de evolute $c'(t) =
(1 + \cos t, -\sin t)$ verdwijnt in $t = \pi$: het keerpunt
is $c(\pi) = (\pi, -2)$, direct onder de top $\gamma(\pi) =
(\pi, 2)$, op afstand $4 = R(\pi)$, precies de \omterm{def:b2:curves:length}{lengte} van de
osculatiestraal daar.

\textbf{24.} Uit \cref{exo:b2:curves:6}, $c'(s) =
-\frac{\kappa'(s)}{\kappa(s)^2}N(s) = R'(s)\,N(s)$, dus
$\norm{c'(s)} = \abs{R'(s)}$ en, voor monotone $R$,
\[
\int_{s_0}^{s_1}\norm{c'(s)}\,\dd s =
\Bigl|\int_{s_0}^{s_1}R'(s)\,\dd s\Bigr| = \abs{R(s_1) -
R(s_0)} .
\]
Zeg $R$ daalt. Een snaar gelegd langs de evolute voorbij
$c(s_0)$ en verlengd met het segment van $c(s_0)$ naar
$\gamma(s_0)$ (dat raak is aan de evolute, door vraag 4)
heeft, wanneer afgepeld tot $c(s)$ en strak gespannen,
recht deel van \omterm{def:b2:curves:length}{lengte} $R(s_0) - \bigl(R(s_0) - R(s)\bigr) =
R(s)$ wijzend van $c(s)$ langs de normaal --- precies
landend op $\gamma(s)$: het vrije uiteinde beschrijft de
oorspronkelijke kromme (``involute''). Huygens hing een
slinger tussen twee cycloïdale wangen: het koord wikkelt op
de evolute, dus beschrijft de bob een cycloïde --- de
tautochroon, waarvan de oscillatieperiode niet van de
amplitude afhangt.

\textbf{25.} \omterm{def:b2:curves:arc}{Raaklijnen} van de parabool: $\Delta = 1$ en $E'
= (1, t) \neq 0$ overal --- gladde \omterm{pb:b2:curves:1}{omhullende} (de parabool
zelf). Glijdende ladder: $\Delta = -1$, maar $E' =
\tfrac32\sin 2t\,(-\cos t, \sin t)$ verdwijnt in de
kwadrant-einden --- de vier keerpunten van de \omterm{pb:b2:curves:1}{astroïde}.
Normalen van de parabool en van de cycloïde: $\Delta =
\kappa \neq 0$, en $E' = R'N$ verdwijnt precies waar de
\omterm{thm:b2:curves:frenet2d}{kromming} extreem is --- keerpunten van de evoluten in
$(0,1)$ en $(\pi, -2)$. Caustiek: $\Delta = 2$, en $E' =
\tfrac32\sin\theta\,(\cos2\theta, \sin2\theta)$ verdwijnt in
$\theta = 0, \pi$ --- de twee keerpunten van de \omterm{pb:b2:curves:1}{nefroïde},
de brandpunten van de spiegel. Extrema van \omterm{thm:b2:curves:frenet2d}{kromming}
\emph{moeten} keerpunten op de \omterm{pb:b2:curves:1}{omhullende} van de normalen
produceren, omdat de snelheid van de evolute $R'N$ is:
daarom heeft de evolute van de ellips vier keerpunten (vier
toppen), en zijn de ontaarde families (bundel, evenwijdigen)
de gevallen waar de machine een punt of niets produceert.
\end{solution}
