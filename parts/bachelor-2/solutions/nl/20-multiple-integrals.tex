\section*{Hoofdstuk \ref{ch:b2:multint} --- Lijnintegralen en meervoudige integralen}

\begin{solution}{exo:b2:multint:1}
(a) Segment $\gamma(t) = (t, t)$, $t \in [0,1]$:
\[
\int_\gamma y^2\dd x + x\dd y
= \int_0^1 (t^2 + t)\,\dd t = \frac13 + \frac12 = \frac56 .
\]
(b) Parabool $\gamma(t) = (t, t^2)$:
\[
\int_0^1 \bigl(t^4\cdot 1 + t\cdot 2t\bigr)\dd t
= \frac15 + \frac23 = \frac{13}{15} .
\]
De twee waarden verschillen, dus is de integraal
pad-afhankelijk: de vorm is \emph{niet} \omterm{def:b2:multint:exact}{exact} --- consistent,
$P_y = 2y \neq 1 = Q_x$, dus is ze niet eens \omterm{def:b2:multint:exact}{gesloten}.
\end{solution}

\begin{solution}{exo:b2:multint:2}
$P = 2xy + y^3$, $Q = x^2 + 3xy^2 + 1$: $P_y = 2x + 3y^2 =
Q_x$, \omterm{def:b2:multint:exact}{gesloten} op $\R^2$. Zoek $f$ met $f_x = P$: $f = x^2y
+ xy^3 + g(y)$; dan dwingt $f_y = x^2 + 3xy^2 + g'(y) = Q$
$g'(y) = 1$, zeg $g(y) = y$. Dus
\[
f(x, y) = x^2y + xy^3 + y
\]
is een \omterm{def:b2:multint:exact}{potentiaal} ($\R^2$ is stervormig, dus moest er een
\omterm{def:b2:multint:exact}{potentiaal} bestaan door het lemma van Poincaré --- maar haar
vertonen is sneller). Door \cref{thm:b2:multint:ftc}, voor
elke boog van $(0,0)$ naar $(1,2)$:
\[
\int_\gamma\omega = f(1, 2) - f(0, 0) = 2 + 8 + 2 = 12 .
\]
\end{solution}

\begin{solution}{exo:b2:multint:3}
Het domein is $\{0 \leq x \leq 1,\ x^2 \leq y \leq x\}$.
$y$-eerst:
\[
\int_0^1\!\int_{x^2}^{x}(x + y)\,\dd y\,\dd x
= \int_0^1\Bigl(x(x - x^2)
+ \frac{x^2 - x^4}{2}\Bigr)\dd x
= \int_0^1\Bigl(\frac{3x^2}{2} - x^3 - \frac{x^4}{2}\Bigr)\dd x
= \frac12 - \frac14 - \frac1{10} = \frac{3}{20} .
\]
$x$-eerst: de snede op hoogte $y \in [0, 1]$ is $y \leq x
\leq \sqrt y$, dus
\[
\int_0^1\!\int_{y}^{\sqrt y}(x + y)\,\dd x\,\dd y
= \int_0^1\Bigl(\frac{y - y^2}{2}
+ y(\sqrt y - y)\Bigr)\dd y
= \frac14 - \frac16 + \frac25 - \frac13 = \frac{3}{20} .
\]
\end{solution}

\begin{solution}{exo:b2:multint:4}
\emph{Eerste integraal.} Op de schijf $D_R$, in
poolcoördinaten:
\[
\iint_{D_R}\frac{\dd x\,\dd y}{(1 + x^2 + y^2)^2}
= \int_0^{2\pi}\!\!\int_0^R
\frac{\rho\,\dd\rho\,\dd\alpha}{(1 + \rho^2)^2}
= 2\pi\Bigl[-\frac{1}{2(1 + \rho^2)}\Bigr]_0^R
= \pi\Bigl(1 - \frac{1}{1 + R^2}\Bigr)
\xrightarrow[R \to \infty]{} \pi .
\]
\emph{Tweede integraal.} De kwartschijf is $0 \leq \alpha
\leq \frac\pi2$, $0 \leq \rho \leq 1$, en $xy =
\rho^2\cos\alpha\sin\alpha$:
\[
\iint_{D'}xy\,\dd x\,\dd y
= \int_0^{\pi/2}\!\!\cos\alpha\sin\alpha\,\dd\alpha
\int_0^1 \rho^3\,\dd\rho
= \frac12\cdot\frac14 = \frac18 .
\]
\end{solution}

\begin{solution}{exo:b2:multint:5}
Door \cref{cor:b2:multint:area} met $x = \cos^3 t$, $y =
\sin^3 t$: $x' = -3\cos^2 t\sin t$, $y' = 3\sin^2 t\cos t$,
\[
\operatorname{Area}
= \frac12\int_0^{2\pi}\bigl(x y' - y x'\bigr)\,\dd t
= \frac12\int_0^{2\pi} 3\cos^2 t\sin^2 t\,(\cos^2 t +
\sin^2 t)\,\dd t
= \frac32\int_0^{2\pi}\sin^2 t\cos^2 t\,\dd t .
\]
Lineariseren: $\sin^2 t\cos^2 t = \frac14\sin^2 2t =
\frac18(1 - \cos 4t)$, dus
\[
\operatorname{Area}
= \frac32\cdot\frac18\int_0^{2\pi}(1 - \cos 4t)\,\dd t
= \frac{3\pi}8 ,
\]
overeenkomend met \cref{pb:b2:curves:1}.
\end{solution}

\begin{solution}{exo:b2:multint:6}
\emph{Stapelen:} de projectie is de eenheidsschijf, $z$ van
$x^2 + y^2$ tot $1$:
\[
V = \iint_{x^2+y^2\leq 1}(1 - x^2 - y^2)\,\dd x\,\dd y
= \int_0^{2\pi}\!\!\int_0^1(1 - \rho^2)\rho\,\dd\rho\,\dd\alpha
= 2\pi\Bigl[\frac{\rho^2}2 - \frac{\rho^4}4\Bigr]_0^1
= \frac\pi2 .
\]
\emph{Snijden:} op hoogte $z \in [0,1]$ is de snede de
schijf van straal $\sqrt z$, \omterm{def:b2:multint:domain}{oppervlakte} $\pi z$:
\[
V = \int_0^1 \pi z\,\dd z = \frac\pi2 .
\]
\end{solution}

\begin{solution}{exo:b2:multint:7}
In sferische coördinaten, $\dd x\,\dd y\,\dd z =
r^2\cos\varphi\,\dd r\,\dd\theta\,\dd\varphi$:
\[
\operatorname{vol}\{a \leq r \leq b\}
= \int_a^b r^2\,\dd r\int_0^{2\pi}\dd\theta
\int_{-\pi/2}^{\pi/2}\cos\varphi\,\dd\varphi
= \frac{b^3 - a^3}{3}\cdot 2\pi \cdot 2
= \frac43\pi(b^3 - a^3) .
\]
Voor $\iiint_B 1/r$: de integrand is $1/r$, dus
\[
\iiint_B\frac1r
= \int_0^R r\,\dd r\int_0^{2\pi}\dd\theta
\int_{-\pi/2}^{\pi/2}\cos\varphi\,\dd\varphi
= \frac{R^2}2\cdot 2\pi \cdot 2 = 2\pi R^2 .
\]
\end{solution}

\begin{solution}{exo:b2:multint:8}
Differentiëren onder het integraalteken
(\cref{ch:b2:integration}):
\[
f_x = \int_0^1 \bigl(P(tx,ty) + x\,t P_x(tx,ty) +
y\,t Q_x(tx,ty)\bigr)\dd t .
\]
Met $P_y = Q_x$ en de kettingregel,
$\frac{\dd}{\dd t}\bigl(t P(tx,ty)\bigr) = P(tx,ty) +
t\bigl(x P_x + y P_y\bigr)(tx,ty) = P + t(xP_x + yQ_x)$,
dus
\[
f_x = \int_0^1 \frac{\dd}{\dd t}\bigl(t P(tx,ty)\bigr)\,\dd t
= \bigl[t P(tx,ty)\bigr]_0^1 = P(x,y) .
\]
Evenzo $f_y = Q$. Dus $\omega = \dd f$: \omterm{def:b2:multint:exact}{exact}.
\end{solution}

\begin{solution}{exo:b2:multint:9}
Op $[0,A]\times[0,\infty)$ is $e^{-xy}\abs{\sin x} \leq
e^{-xy}$ integreerbaar (Fubini: $\int_0^A\abs{\sin
x}/x\,\dd x < \infty$ na integreren in $y$), dus is
verwisselen legitiem. $y$-eerst:
\[
\int_0^A\sin x\Bigl(\int_0^\infty e^{-xy}\,\dd y\Bigr)\dd x
= \int_0^A\frac{\sin x}{x}\,\dd x .
\]
$x$-eerst: $\int_0^\infty e^{-xy}\sin x\,\dd x =
\frac{1}{1+y^2}$ (imaginair deel van $\int e^{-(y-\iu)x}$),
maar met grens $A$ is de onbepaalde integraal
$\int_0^A e^{-xy}\sin x\,\dd x =
\frac{1 - e^{-Ay}(y\sin A + \cos A)}{1+y^2}$ (partieel of
complex). Integreren in $y$:
\[
\int_0^A\frac{\sin x}{x}\,\dd x
= \int_0^\infty\frac{1 - e^{-Ay}(y\sin A + \cos
A)}{1+y^2}\,\dd y
= \frac\pi2 - \int_0^\infty
e^{-Ay}\frac{y\sin A + \cos A}{1+y^2}\,\dd y .
\]
Als $A \to \infty$ verdwijnt de rest (gedomineerd door
$1/(1+y^2)$), dus $\int_0^\infty \frac{\sin x}{x}\,\dd x =
\frac\pi2$.
\end{solution}

\begin{solution}{exo:b2:multint:10}
Met booglengteparametrisatie, $x'^2 + y'^2 = 1$. Door
\cref{cor:b2:multint:area},
\[
2A = \oint x\,\dd y - y\,\dd x
= \int_0^{2\pi}\bigl(xy' - yx'\bigr)\dd s .
\]
Partieel integreren van $\oint y\,\dd x$ over de periode
(randtermen heffen op door periodiciteit), $-\int yx' =
\int y'x$, dus in feite $2A = 2\int_0^{2\pi}xy'\,\dd s$. Dan
$2xy' \leq x^2 + y'^2$ geeft
\[
2A \leq \int_0^{2\pi}\bigl(x^2 + y'^2\bigr)\dd s
= \int_0^{2\pi} x^2 + \int_0^{2\pi}\bigl(1 - x'^2\bigr)
= 2\pi - \int_0^{2\pi}\bigl(x'^2 - x^2\bigr)\dd s .
\]
De Wirtinger-ongelijkheid (\cref{ch:b2:fourier}-oefeningen:
voor een $2\pi$-periodieke $\mathcal{C}^1$-functie met
gemiddelde nul, $\int x^2 \leq \int x'^2$) maakt de laatste
integraal niet-negatief: $A \leq \pi$. Gelijkheid eist
gelijkheid in Wirtinger ($x(s) = a\cos s + b\sin s$) en in
$2xy' \leq x^2 + y'^2$ ($y' = x$ puntsgewijs), wat $y =
a\sin s - b\cos s + c$ dwingt: de kromme is de
eenheidscirkel (passend gecentreerd). Omdat een kromme van
lengte $L$ herschaalt tot lengte $2\pi$, is de algemene
uitspraak $A \leq \frac{L^2}{4\pi}$: onder alle \omterm{def:b2:multint:exact}{gesloten}
krommen van gegeven omtrek sluit de cirkel de grootste
\omterm{def:b2:multint:domain}{oppervlakte} in.
\end{solution}

\begin{solution}{exo:b2:multint:11}
In sferische coördinaten, $z = r\sin\varphi$:
\[
\iiint_B z^2
= \int_0^R r^4\,\dd r\int_0^{2\pi}\dd\theta
\int_{-\pi/2}^{\pi/2}\sin^2\varphi\cos\varphi\,\dd\varphi
= \frac{R^5}5\cdot 2\pi \cdot \frac23
= \frac{4\pi R^5}{15} .
\]
Door symmetrie $\iiint x^2 = \iiint y^2 = \iiint z^2$, dus
$\iiint_B(x^2+y^2+z^2) = 3\cdot\frac{4\pi R^5}{15} =
\frac{4\pi R^5}{5}$. Schilcontrole: $\int_0^R r^2\cdot
4\pi r^2\,\dd r = 4\pi\frac{R^5}{5}$, hetzelfde.
\end{solution}

\begin{solution}{exo:b2:multint:12}
De rand bestaat uit de segmenten van $(x_i, y_i)$ naar
$(x_{i+1}, y_{i+1})$. Op het $i$-de segment, $\gamma(t) =
(1-t)(x_i,y_i) + t(x_{i+1},y_{i+1})$, $t \in [0,1]$:
\[
\int_{\gamma_i} x\,\dd y - y\,\dd x
= \int_0^1\bigl(x(t)(y_{i+1}-y_i) - y(t)(x_{i+1}-x_i)\bigr)
\,\dd t
= x_iy_{i+1} - x_{i+1}y_i
\]
(de lineaire termen heffen op). Dus $\operatorname{Area}(K)
= \frac12\sum_i(x_iy_{i+1} - x_{i+1}y_i)$. Voor de driehoek
$(0,0)$, $(1,0)$, $(0,1)$: $\frac12\bigl(0\cdot0 - 1\cdot0
+ 1\cdot1 - 0\cdot0 + 0\cdot0 - 0\cdot1\bigr) = \frac12$.
\end{solution}

\begin{solution}{pb:b2:multint:1}
\textbf{1.} De bal $B_n(R)$ wordt beschreven door geitereerde
grenzen $-R \leq x_n \leq R$, daarna $\abs{x_{n-1}} \leq
\sqrt{R^2 - x_n^2}$, enzovoort; substitueren $x_i = Ru_i$ in
elk van de $n$ éénvariabele-integralen vermenigvuldigt elk
met $R$ en beeldt de grenzen af op die van $B_n(1)$: $v_n(R)
= R^n\,v_n(1) = V_nR^n$.

\textbf{2.} Snijden langs $x_n = t$: de snede van $B_n(1)$ is
de bal $B_{n-1}\bigl(\sqrt{1 - t^2}\bigr)$, dus, door vraag
1,
\[
V_n = \int_{-1}^1 v_{n-1}\bigl(\sqrt{1 - t^2}\bigr)\,\dd t
= V_{n-1}\int_{-1}^{1}(1 - t^2)^{\frac{n-1}2}\,\dd t .
\]

\textbf{3.} Met $t = \sin\theta$, $\dd t =
\cos\theta\,\dd\theta$ en $(1 - t^2)^{\frac{n-1}2} =
\cos^{n-1}\theta$ op $\intcc{-\pi/2}{\pi/2}$:
\[
\int_{-1}^1(1 - t^2)^{\frac{n-1}2}\dd t
= \int_{-\pi/2}^{\pi/2}\cos^n\theta\,\dd\theta
= 2\int_0^{\pi/2}\cos^n\theta\,\dd\theta = 2W_n,
\]
en $\theta \mapsto \frac\pi2 - \theta$ verwisselt de sinus-
en cosinusvormen van $W_n$.

\textbf{4.} Schrijf $\sin^n = \sin^{n-2}(1 - \cos^2)$ en
integreer $\int\sin^{n-2}\cos\cdot\cos$ partieel ($v =
\frac{\sin^{n-1}}{n-1}$):
\[
W_n = W_{n-2} - \frac{W_n}{n-1}
\quad\Longrightarrow\quad
W_n = \frac{n-1}{n}W_{n-2}.
\]
Vandaar $nW_nW_{n-1} = (n-1)W_{n-1}W_{n-2}$: de rij
$(nW_nW_{n-1})$ is constant, gelijk aan $1\cdot W_1W_0 =
1\cdot\frac\pi2$, dus $W_nW_{n-1} = \frac{\pi}{2n}$.

\textbf{5.} Vragen 2--3 geven $V_n = 2W_nV_{n-1}$, tweemaal:
\[
V_n = 2W_n\cdot 2W_{n-1}\,V_{n-2}
= 4\,\frac{\pi}{2n}\,V_{n-2} = \frac{2\pi}n\,V_{n-2}.
\]

\textbf{6.} Uit $V_0 = 1$: $V_{2k} = \frac{2\pi}{2k}V_{2k-2}
= \frac\pi kV_{2k-2}$, dus $V_{2k} = \frac{\pi^k}{k!}$ door
inductie. Uit $V_1 = 2$: $V_{2k+1} =
\frac{2\pi}{2k+1}V_{2k-1}$, dus
\[
V_{2k+1} = 2\prod_{j=1}^k\frac{2\pi}{2j+1}
= \frac{2^{k+1}\pi^k}{1\cdot3\cdot5\cdots(2k+1)} .
\]

\textbf{7.} $V_1 = 2$, $V_2 = \pi \approx 3.142$, $V_3 =
\frac{4\pi}3 \approx 4.189$, $V_4 = \frac{\pi^2}2 \approx
4.935$, $V_5 = \frac{8\pi^2}{15} \approx 5.264$, $V_6 =
\frac{\pi^3}6 \approx 5.168$, $V_7 = \frac{16\pi^3}{105}
\approx 4.725$. De éénstapsverhouding is $V_n/V_{n-1} =
2W_n$, en $(W_n)$ is dalend ($\sin^n \leq \sin^{n-1}$
puntsgewijs). Nu $2W_5 = 2\cdot\frac45\cdot\frac23 =
\frac{16}{15} > 1$ terwijl $2W_6 =
2\cdot\frac56\cdot\frac34\cdot\frac12\cdot\frac\pi2 =
\frac{5\pi}{16} < 1$: de verhoudingen overschrijden $1$ tot
$n = 5$ en liggen onder $1$ vanaf $n = 6$ --- $(V_n)$ stijgt
tot haar maximum $V_5$ en daalt daarna.

\textbf{8.} Voor $n \geq 13 > 4\pi$: $V_n/V_{n-2} = 2\pi/n <
\tfrac12$, dus $V_{n} \leq C\cdot 2^{-n/2}$ met een vaste
constante; beter, voor elke $q > 0$, $2\pi/n < q^2$ voor
grote $n$, dus $V_n/q^n \to 0$: het verval verslaat elke
meetkundige rij. Convergentie van $\sum V_n$ volgt uit de
verhouding $V_n/V_{n-2} \to 0$ (vergelijk met een
meetkundige reeks vanaf zekere rang).

\textbf{9.} $\sum_{k\geq0}V_{2k}x^{2k} =
\sum_{k\geq0}\frac{(\pi x^2)^k}{k!} = \eu^{\pi x^2}$, de
exponentiële reeks (\cref{ch:b2:powerseries}), convergent
voor elke $x$. In $x = 1$: $\sum_kV_{2k} = \eu^\pi \approx
23.14$.

\textbf{10.} Op de kubus $\intcc{-R}R^n$ is de integrand het
product $\prod_i\eu^{-x_i^2}$, dus factoriseert de
geitereerde integraal: $\bigl(\int_{-R}^R\eu^{-t^2}\dd
t\bigr)^n$. Laten $R \to \infty$ en
$\int_\R\eu^{-t^2}\dd t = \sqrt\pi$
(\cref{ex:b2:multint:polar}) gebruiken: $I_n = \pi^{n/2}$.

\textbf{11.} $t = u^2$ geeft $\Gamma(\tfrac12) =
\int_0^\infty t^{-1/2}\eu^{-t}\dd t =
2\int_0^\infty\eu^{-u^2}\dd u = \sqrt\pi$. Itereren van
$\Gamma(s+1) = s\Gamma(s)$: $\Gamma(k+1) = k!\,\Gamma(1) =
k!$, en
\[
\Gamma\Bigl(k + \frac32\Bigr)
= \Bigl(k + \frac12\Bigr)\Bigl(k - \frac12\Bigr)\cdots
\frac12\cdot\Gamma\Bigl(\frac12\Bigr)
= \frac{(2k+1)(2k-1)\cdots1}{2^{k+1}}\,\sqrt\pi .
\]

\textbf{12.} Stel $F_n = \pi^{n/2}/\Gamma(\frac n2 + 1)$.
Omdat $\Gamma(\frac n2 + 1) = \frac n2\,\Gamma(\frac n2) =
\frac n2\,\Gamma(\frac{n-2}2 + 1)$, krijgen we $F_n =
\frac{2\pi}nF_{n-2}$: dezelfde recursie als $V_n$ (vraag 5).
Bases: $F_1 = \sqrt\pi/\Gamma(\frac32) =
\sqrt\pi/(\frac{\sqrt\pi}2) = 2 = V_1$ en $F_2 =
\pi/\Gamma(2) = \pi = V_2$. Door inductie $V_n = F_n$ voor
alle $n$; vraag 11 zet dit terug in de twee gesloten vormen
van vraag 6.

\textbf{13.} Met $r = \sqrt t$: $\int_0^\infty
\eu^{-r^2}r^{n-1}\dd r = \frac12\int_0^\infty
t^{\frac n2 - 1}\eu^{-t}\dd t = \frac12\Gamma(\frac n2)$.
Vandaar
\[
n\,V_n\int_0^\infty\eu^{-r^2}r^{n-1}\dd r
= V_n\cdot\frac n2\,\Gamma\Bigl(\frac n2\Bigr)
= V_n\,\Gamma\Bigl(\frac n2 + 1\Bigr) = \pi^{n/2} = I_n .
\]
Beide zijden bewezen, kan de identiteit worden
\emph{gelezen} als de schilontbinding van de
Gauss-integraal: de sfeer van straal $r$ draagt \omterm{def:b2:multint:domain}{oppervlakte}
$nV_nr^{n-1}$, en het Gauss-gewicht $\eu^{-r^2}$ wordt over
de schillen geïntegreerd.

\textbf{14.} $s_0 = V_1 = 2$ (de $0$-sfeer is twee punten),
$s_1 = 2V_2 = 2\pi$, $s_2 = 3V_3 = 4\pi$, $s_3 = 4V_4 =
2\pi^2$; en $\int_0^R s_{n-1}r^{n-1}\dd r = V_nR^n =
v_n(R)$: \omterm{def:b2:multint:domain}{oppervlakte} is de radiale afgeleide van volume.

\textbf{15.} Voor $n = 2k$ geeft Stirling
(\cref{thm:b2:comparison:stirling}) $k! \sim
\sqrt{2\pi k}\,(k/\eu)^k$, dus
\[
V_{2k} = \frac{\pi^k}{k!}
\sim \frac{(\pi\eu/k)^k}{\sqrt{2\pi k}}
= \frac1{\sqrt{\pi n}}\Bigl(\frac{2\pi\eu}n\Bigr)^{n/2}
\qquad (n = 2k).
\]
Voor oneven $n$: $V_{2k+1} = 2W_{2k+1}V_{2k} \leq 2V_{2k}$,
dus gelden dezelfde super-meetkundige vervalgrenzen (op een
factor $2$ en een verschuiving van één in de exponent na)
--- voor elke $q > 0$, $V_n = o(q^n)$.

\textbf{16.} $V_2/4 = \pi/4 \approx 0.785$; $V_3/8 = \pi/6
\approx 0.524$; $V_{10}/2^{10} = \frac{\pi^5}{120\cdot1024}
\approx 0.0025$. In het algemeen
$\frac{V_n/2^n}{V_{n-2}/2^{n-2}} = \frac{2\pi}{4n} =
\frac{\pi}{2n} \to 0$: de verhouding streeft naar $0$
(super-meetkundig). De ingeschreven bal neemt een
verdwijnende fractie in: het volume van de kubus migreert
naar haar hoeken.

\textbf{17.} Door vraag 1 heeft de binnenbal van straal $1
- \varepsilon$ volume $V_n(1-\varepsilon)^n$, dus draagt de
buitenschil de fractie $1 - (1 - \varepsilon)^n \to 1$. Voor
$n = 100$, $\varepsilon = 0.01$: $(0.99)^{100} =
\eu^{100\ln0.99} \approx \eu^{-1.005} \approx 0.366$:
ongeveer $63\%$ van de bal ligt binnen $1\%$ van haar
oppervlak.

\textbf{18.} $(W_n)$ daalt, dus $W_n \leq W_{n-1} \leq
W_{n-2} = \frac{n}{n-1}W_n$: inklammen, $W_{n-1}/W_n \to
1$. Vermenigvuldigen met $W_nW_{n-1} = \frac\pi{2n}$:
$W_n^2 \sim \frac\pi{2n}$, d.w.z.\ $W_n \sim
\sqrt{\pi/(2n)}$. Ondergrens: $W_n^2 \geq W_nW_{n+1} =
\frac{\pi}{2(n+1)}$, dus $W_n \geq \sqrt{\pi/(2(n+1))}$
voor elke $n$.

\textbf{19.} Teller: $1 - x^2 \leq \eu^{-x^2}$ geeft $(1 -
x^2)^{\frac{n-1}2} \leq \eu^{-(n-1)x^2/2}$, en met $a =
\frac{n-1}2$,
\[
\int_\delta^1(1 - x^2)^{\frac{n-1}2}\dd x
\leq \int_\delta^\infty\eu^{-ax^2}\dd x
\leq \int_\delta^\infty\frac x\delta\,\eu^{-ax^2}\dd x
= \frac{\eu^{-a\delta^2}}{2a\delta}
= \frac{\eu^{-(n-1)\delta^2/2}}{(n-1)\delta}.
\]
Noemer: $2W_n \geq \sqrt{2\pi/(n+1)}$ door vraag 18. Delen
geeft de weergegeven grens. Voor $\delta = s/\sqrt{n-1}$
wordt ze
$\sqrt{\tfrac{n+1}{2\pi(n-1)}}\;\eu^{-s^2/2}/s = O\bigl(
\eu^{-s^2/2}/s\bigr)$, uniform in $n$: buiten de slab
$\abs{x_1} \leq s/\sqrt{n-1}$ is er bijna geen volume, voor
$s$ matig groot --- en door symmetrie geldt hetzelfde voor
elke richting.

\textbf{20.} De twee uitspraken bestaan naast elkaar omdat
ze verschillende coördinaten van hetzelfde punt beschrijven.
Bijna elk punt van $B_n(1)$ heeft \omterm{def:b2:nvs:norm}{norm} dicht bij $1$ (vraag
17: radiale concentratie nabij de sfeer), maar elk van haar
$n$ coördinaten is klein, van orde $1/\sqrt n$ (vraag 19),
wat consistent is omdat $n$ coördinaten van grootte
$1/\sqrt n$ \omterm{def:b2:nvs:norm}{norm} van orde $1$ hebben. Hoogdimensionaal
volume concentreert zich waar alle coördinaten het
normbudget gelijk delen --- nabij de sfeer, maar ver van
elke coördinaat-aspool.

\textbf{21.} Snijd $\Delta_n$ in $x_n = t \in \intcc01$: de
snede is $\{x' \in \R^{n-1} : x_i \geq 0,\ \sum x_i \leq 1 -
t\} = (1-t)\Delta_{n-1}$, van volume
$(1-t)^{n-1}\operatorname{vol}(\Delta_{n-1})$ door
homogeniteit. Dus
\[
\operatorname{vol}(\Delta_n) =
\operatorname{vol}(\Delta_{n-1})\int_0^1(1 - t)^{n-1}\dd t =
\frac{\operatorname{vol}(\Delta_{n-1})}{n}
\quad\Longrightarrow\quad
\operatorname{vol}(\Delta_n) = \frac1{n!}\,.
\]

\textbf{22.} De $2^n$ teken-orthanten snijden $C_n$ in $2^n$
kopieën van $\Delta_n$ (met verwaarloosbare overlappingen op
coördinaathypervlakken): $\operatorname{vol}(C_n) =
\frac{2^n}{n!}$. Als $\sum\abs{x_i} \leq 1$ dan $\sum x_i^2
\leq \bigl(\sum\abs{x_i}\bigr)^2 \leq 1$: $C_n \subseteq
B_n(1)$; en $B_n(1) \subseteq \intcc{-1}1^n$ omdat
$\abs{x_i} \leq \norm x$. Vandaar $\frac{2^n}{n!} \leq V_n
\leq 2^n$ --- consistent met vraag 15, die $V_n$ tussen de
factoriële en meetkundige schalen plaatst.

\textbf{23.} Voor $(x, y)$ in de eenheidsschijf is de snede
van $B_4(1)$ de schijf van straal $\sqrt{1 - x^2 - y^2}$ in
het $(z, w)$-vlak, van \omterm{def:b2:multint:domain}{oppervlakte} $\pi(1 - x^2 - y^2)$. In
poolcoördinaten:
\[
V_4 = \iint_{x^2+y^2\leq1}\pi(1 - x^2 - y^2)\,\dd x\,\dd y
= \pi\int_0^{2\pi}\!\!\int_0^1(1 - \rho^2)\rho\,
\dd\rho\,\dd\alpha
= \pi\cdot2\pi\cdot\frac14 = \frac{\pi^2}2 ,
\]
overeenkomend met vraag 6.

\textbf{24.} De waarschijnlijkheid is de volumeverhouding
$\dfrac{V_{20}}{2^{20}} = \dfrac{\pi^{10}}{10!\cdot2^{20}}
\approx \dfrac{0.0258}{1\,048\,576} \approx
2.5\cdot10^{-8}$. Het aantal trekkingen tot de eerste
treffer is van orde de inverse, ongeveer $4\cdot10^7$: een
rejection sampler die prachtig werkte voor de schijf
($\pi/4$ van de treffers) is nutteloos in dimensie $20$ ---
de curse of dimensionality in één regel.

\textbf{25.} Route één (Delen I--II) gebruikte:
Fubini-achtig snijden van de geitereerde integraal,
éénvariabele substitutie in elke coördinaat
(homogeniteit), en de Wallis-integralen --- pure
éénvariabele analyse plus inductie. Route twee (Deel III)
gebruikte: Fubini voor de productstructuur van $I_n$, de
pool-variabelensubstitutie via de Gauss-integraal van
\cref{ex:b2:multint:polar}, en de
functionaalvergelijking van de $\Gamma$-functie. Ze
ontmoeten elkaar in $V_n = \pi^{n/2}/\Gamma(\frac n2 + 1)$,
met Stirling (\cref{thm:b2:comparison:stirling}) die de
formule in asymptotiek omzet. Het volume van Jaar~3
herbouwt dit alles op de Lebesgue-integraal: daar zijn
Fubini en de variabelensubstitutie stellingen voor
algemene integreerbare functies, bestaan sferische
coördinaten in elke dimensie, en keren dezelfde balvolumes
terug als de uitgewerkte dividenden van de productmaat- en
Stirling-problemen --- met gedomineerde convergentie die
onze handgemaakte inklammingen vervangt.
