\section*{Hoofdstuk \ref{ch:b2:surfaces} --- Oppervlakken}

\begin{solution}{exo:b2:surfaces:1}
Met $\sigma(\theta, r) = (r\cos\theta,\ r\sin\theta,\ r)$:
\[
\sigma_\theta = (-r\sin\theta,\ r\cos\theta,\ 0),
\qquad
\sigma_r = (\cos\theta,\ \sin\theta,\ 1),
\]
onafhankelijk voor $r > 0$. Het \omterm{def:b2:surfaces:tangent}{raakvlak} in $M_0 =
\sigma(\theta_0, r_0)$ gaat door $M_0$ met richtingen
$\sigma_\theta, \sigma_r$. Nu is $M_0 - O = r_0(\cos\theta_0,
\sin\theta_0, 1) = r_0\,\sigma_r(\theta_0, r_0)$ zelf een
raakrichting: de top $O$ ligt in het \omterm{def:b2:surfaces:tangent}{raakvlak}. (Dit is het
algemene gedrag van kegels: ze worden geregeerd door lijnen
door de top, en een \omterm{def:b2:surfaces:tangent}{raakvlak} bevat de regellijn door het
raakpunt.)
\end{solution}

\begin{solution}{exo:b2:surfaces:2}
Pas \cref{prop:b2:surfaces:gradient} toe op $f(x,y,z) =
\frac{x^2}{a^2} + \frac{y^2}{b^2} + \frac{z^2}{c^2}$: $\nabla
f(x_0,y_0,z_0) = 2\bigl(\frac{x_0}{a^2}, \frac{y_0}{b^2},
\frac{z_0}{c^2}\bigr) \neq 0$ op het oppervlak. Het \omterm{def:b2:surfaces:tangent}{raakvlak}
is
\[
\frac{x_0}{a^2}(x - x_0) + \frac{y_0}{b^2}(y - y_0)
+ \frac{z_0}{c^2}(z - z_0) = 0,
\qquad\text{d.w.z.}\qquad
\frac{x_0\,x}{a^2} + \frac{y_0\,y}{b^2} + \frac{z_0\,z}{c^2} = 1,
\]
gebruikend dat $(x_0, y_0, z_0)$ aan de ellipsoïdevergelijking
voldoet --- de ``splits de kwadraten''-regel die $\langle
M_0, M \rangle = R^2$ van de bol generaliseert.
\end{solution}

\begin{solution}{exo:b2:surfaces:3}
$\sigma_u = (-v\sin u,\ v\cos u,\ a)$ en $\sigma_v = (\cos u,\
\sin u,\ 0)$, dus
\[
E = v^2 + a^2, \qquad F = 0, \qquad G = 1,
\qquad
\sqrt{EG - F^2} = \sqrt{v^2 + a^2} > 0
\]
(de helicoïde is overal regulier, ook op haar as $v = 0$).
\omterm{def:b2:surfaces:area}{Oppervlakte} van het stuk:
\[
\mathcal{A} = \int_0^{2\pi}\!\!\int_0^1
\sqrt{v^2 + a^2}\;\dd v\,\dd u
= 2\pi\cdot\frac12\Bigl[v\sqrt{v^2 + a^2}
+ a^2\ln\bigl(v + \sqrt{v^2 + a^2}\bigr)\Bigr]_0^1 ,
\]
d.w.z.\ $\mathcal{A} = \pi\Bigl(\sqrt{1 + a^2} +
a^2\ln\frac{1 + \sqrt{1 + a^2}}{a}\Bigr)$.
\end{solution}

\begin{solution}{exo:b2:surfaces:4}
Afgeleiden:
\[
\sigma_\theta = \bigl(-(R + r\cos\psi)\sin\theta,\
(R + r\cos\psi)\cos\theta,\ 0\bigr),
\qquad
\sigma_\psi = \bigl(-r\sin\psi\cos\theta,\
-r\sin\psi\sin\theta,\ r\cos\psi\bigr).
\]
Dan
\[
E = (R + r\cos\psi)^2,
\qquad
F = r\sin\psi\cos\psi\,(R + r\cos\psi)
\bigl(\sin\theta\cos\theta - \sin\theta\cos\theta\bigr) = 0,
\qquad
G = r^2 ,
\]
dus $\sqrt{EG - F^2} = r(R + r\cos\psi) \geq r(R - r) > 0$:
regulier overal. \omterm{def:b2:surfaces:area}{Oppervlakte}:
\[
\mathcal{A}
= \int_0^{2\pi}\!\!\int_0^{2\pi} r(R + r\cos\psi)
\,\dd\theta\,\dd\psi
= 2\pi r \Bigl(2\pi R + r\int_0^{2\pi}\cos\psi\,\dd\psi\Bigr)
= 4\pi^2 R r ,
\]
de $\cos\psi$-term integreert tot nul --- de stelling van
Pappus: \omterm{def:b2:surfaces:area}{oppervlakte} $=$ (lengte van de geroteerde cirkel)
$\times$ (afstand afgelegd door haar centrum).
\end{solution}

\begin{solution}{exo:b2:surfaces:5}
Voor $\sigma(x, y) = (x, y, f(x,y))$: $\sigma_x = (1, 0,
f_x)$, $\sigma_y = (0, 1, f_y)$, dus $E = 1 + f_x^2$, $F =
f_xf_y$, $G = 1 + f_y^2$ en
\[
EG - F^2 = (1 + f_x^2)(1 + f_y^2) - f_x^2f_y^2
= 1 + f_x^2 + f_y^2 ,
\]
wat de geclaimde formule geeft. Voor $f = \frac12(x^2 +
y^2)$: $f_x = x$, $f_y = y$, dus
\[
\mathcal{A} = \iint_{x^2+y^2\leq 1}\sqrt{1 + x^2 + y^2}\,
\dd x\,\dd y
= \int_0^{2\pi}\!\!\int_0^1 \rho\sqrt{1 + \rho^2}\,\dd\rho\,
\dd\theta
= 2\pi\cdot\frac13\bigl[(1+\rho^2)^{3/2}\bigr]_0^1
= \frac{2\pi}3(2\sqrt2 - 1) .
\]
\end{solution}

\begin{solution}{exo:b2:surfaces:6}
Met $E = R^2\cos^2\varphi$, $F = 0$, $G = R^2$
(\cref{ex:b2:surfaces:spherearea}):
\[
L(\gamma) = \int_a^b R\sqrt{\cos^2\varphi\,\theta'^2 +
\varphi'^2}\,\dd t
\geq \int_a^b R\abs{\varphi'}\,\dd t
\geq R\bigl|\varphi(b) - \varphi(a)\bigr|,
\]
met gelijkheid dan en slechts dan als $\theta' \equiv 0$ en
$\varphi'$ van constant teken: de kromme loopt monotoon
langs een meridiaan. Voor twee punten op $\theta =
\theta_0$ is de meridiaanboog dus strikt korter dan elke
niet-meridionale concurrent.
\end{solution}

\begin{solution}{exo:b2:surfaces:7}
Voor elke op $S$ getekende kromme $\gamma$ is $\gamma'(t)
\perp \nabla f(\gamma(t))$
(\cref{prop:b2:surfaces:gradient}). De hypothese zegt dat de
normaal, dus $\nabla f(M)$, evenwijdig is aan $M - \Omega$
voor elke $M \in S$: $\nabla f(M) = \lambda(M)\,(M -
\Omega)$. Dus $\langle \gamma', \gamma - \Omega\rangle = 0$
langs elke getekende kromme, d.w.z.\ $\frac{\dd}{\dd
t}\norm{\gamma - \Omega}^2 = 0$. Omdat $S$ \omterm{def:b2:metric:connected}{samenhangend} is,
is $\norm{M - \Omega}$ constant op $S$: $S$ ligt op een bol
met centrum $\Omega$.
\end{solution}

\begin{solution}{exo:b2:surfaces:8}
Schrijf $\Phi(u', v') = (u(u',v'), v(u',v'))$. De kettingregel
geeft $\tilde\sigma_{u'} = u_{u'}\sigma_u + v_{u'}\sigma_v$
en $\tilde\sigma_{v'} = u_{v'}\sigma_u + v_{v'}\sigma_v$,
dus
\[
\tilde\sigma_{u'} \wedge \tilde\sigma_{v'}
= (u_{u'}v_{v'} - u_{v'}v_{u'})\,(\sigma_u \wedge
\sigma_v)\circ\Phi
= (\det J_\Phi)\,(\sigma_u \wedge \sigma_v)\circ\Phi .
\]
\omterm{def:b2:nvs:norm}{Normen} nemen levert de claim. Voor de \omterm{def:b2:surfaces:area}{oppervlakte}: als $K' =
\Phi^{-1}(K)$, dan
\[
\iint_{K'}\norm{\tilde\sigma_{u'}\wedge\tilde\sigma_{v'}}\,
\dd u'\,\dd v'
= \iint_{K'}\abs{\det J_\Phi}\,\norm{\sigma_u\wedge
\sigma_v}\circ\Phi\,\dd u'\,\dd v'
= \iint_K\norm{\sigma_u\wedge\sigma_v}\,\dd u\,\dd v
\]
door de variabelensubstitutieformule
(\cref{thm:b2:multint:changevar}): de \omterm{def:b2:surfaces:area}{oppervlakte} hangt niet
van de kaart af.
\end{solution}

\begin{solution}{exo:b2:surfaces:9}
Met de sferische kaart, $z = R\sin\varphi$, dus $\varphi =
\arcsin(z/R)$ en $\dd\varphi = \dd z/(R\cos\varphi)$. Het
oppervlakte-element $R^2\cos\varphi\,\dd\theta\,\dd\varphi$
wordt $R\,\dd\theta\,\dd z$:
\[
\mathcal{A} = \int_{z_1}^{z_2}\!\!\int_0^{2\pi}
R\,\dd\theta\,\dd z = 2\pi R(z_2 - z_1) .
\]
De breedten zelf verdwijnen: alleen de hoogte telt. Gelijke
hoogteschijven van een sinaasappel geven dus gelijke
schiloppervlakten.
\end{solution}

\begin{solution}{exo:b2:surfaces:10}
$\sigma_\theta = (-r\sin\theta,\ r\cos\theta,\ 0)$,
$\sigma_z = (r'\cos\theta,\ r'\sin\theta,\ 1)$, dus
\[
\sigma_\theta \wedge \sigma_z = (r\cos\theta,\ r\sin\theta,\
-rr') ,
\]
en de normaallijn in $M = \sigma(\theta, z)$ is $M +
t\,(\sigma_\theta \wedge \sigma_z)$. Haar snijpunt met de
$z$-as ($x = y = 0$) eist
\[
r\cos\theta + t\,r\cos\theta = 0, \qquad
r\sin\theta + t\,r\sin\theta = 0 ,
\]
d.w.z.\ $t = -1$ (omdat $r > 0$); de $z$-coördinaat is dan $z
- rr'(-1)\cdot\frac{-rr'}{\norm{\cdots}}$ wacht --- eenvoudiger:
met $n = (r\cos\theta,\ r\sin\theta,\ -rr')$, $M + t n$ heeft
$x = r\cos\theta(1+t)$, $y = r\sin\theta(1+t)$, nul wanneer
$t = -1$, en $z$-coördinaat $z + (-1)(-rr') = z + rr'$. Het
snijpunt is $(0,0, z + r(z)r'(z))$.
\end{solution}

\begin{solution}{exo:b2:surfaces:11}
$\sigma_u = (-\sin u,\ \cos u,\ 0)$, $\sigma_v = (0,0,1)$:
$E = 1$, $F = 0$, $G = 1$. Voor $\gamma(t) =
\sigma(u(t),v(t))$ is $\norm{\gamma'}^2 = Eu'^2 + 2Fu'v' +
Gv'^2 = u'^2 + v'^2$, precies de snelheid van de vlakke
kromme $(u,v)$: lengten komen overeen (de cilinder
``afrollen'' is een isometrie). De schroeflijn van één
omwenteling met stijging $2\pi c$ is $u(t) = t$, $v(t) =
ct$ op $[0, 2\pi]$: lengte $\int_0^{2\pi}\sqrt{1+c^2}\,\dd t
= 2\pi\sqrt{1+c^2}$. Elke andere kromme met dezelfde
eindpunten en $\Delta u = 2\pi$ voldoet aan
\[
L = \int\sqrt{u'^2 + v'^2}\,\dd t \geq \int\abs{u'}\,\dd t
\geq 2\pi ,
\]
maar strenger: door Cauchy--Schwarz of door $v' = c u'$ voor
de schroeflijn te vergelijken, is de rechte lijn in het
$(u,v)$-vlak van $(0,0)$ naar $(2\pi, 2\pi c)$ de kortste,
van lengte $2\pi\sqrt{1+c^2}$, en elke afwijking is strikt
langer.
\end{solution}

\begin{solution}{exo:b2:surfaces:12}
Maximaliseer $g(M) = \norm{M}^2$ op $S = \{f = c\}$. In een
maximum $M_0$ is $\nabla g(M_0) = 2M_0$ orthogonaal op
$T_{M_0}S$, dus evenwijdig aan $\nabla f(M_0)$
(\cref{prop:b2:surfaces:gradient}): $\nabla f(M_0)$ is
collineair met $\vect{OM_0}$. Voor de ellipsoïde, $\nabla f
= 2\bigl(\frac{x}{a^2}, \frac{y}{b^2}, \frac{z}{c^2}\bigr)$
evenwijdig aan $(x,y,z)$ dwingt, componentgewijs, ofwel de
coördinaat nul ofwel $\frac1{a^2} = \frac1{b^2} = \frac1{c^2}$
(onmogelijk omdat $a > b > c$). Dus liggen de kandidaten op
de assen: $(\pm a, 0, 0)$, $(0, \pm b, 0)$, $(0, 0, \pm c)$.
De verste zijn $(\pm a, 0, 0)$, van afstand $a$.
\end{solution}
\begin{solution}{pb:b2:surfaces:1}
\textbf{1.} Ontwikkelen, en met de symmetrie van $A$
($t^{\mathsf T}\!APY = (At)^{\mathsf T}PY$):
\[
q(PY + t) = Y^{\mathsf T}P^{\mathsf T}\!APY +
2\,(At + b)^{\mathsf T}PY + \bigl(t^{\mathsf T}\!At +
2b^{\mathsf T}t + c\bigr),
\]
wat het weergegeven triple is. $A' = P^{\mathsf T}\!AP =
P^{-1}AP$ is gelijksoortig met $A$: zelfde karakteristieke
polynoom, spectrum, rang, signatuur. Ten slotte $\{q = 0\} =
\{sq = 0\}$ voor $s \neq 0$, dus is alleen de vergelijking
\emph{op een scalar na} aan de verzameling gehecht; schalen
met $s$ vermenigvuldigt alle \omterm{def:b2:reduction:eigen}{eigenwaarden} met $s$.

\textbf{2.} De spectrale stelling levert $P \in O(3)$ met
$P^{\mathsf T}\!AP = \operatorname{diag}(\lambda_1,
\lambda_2, \lambda_3)$; vraag 1 met $t = 0$ maakt van de
vergelijking $\sum\lambda_iy_i^2 + 2\sum\beta_iy_i + c = 0$,
waar $\beta = P^{\mathsf T}b$.

\textbf{3.} Voor $\lambda_i \neq 0$: $\lambda_iy_i^2 +
2\beta_iy_i = \lambda_i\bigl(y_i +
\frac{\beta_i}{\lambda_i}\bigr)^2 -
\frac{\beta_i^2}{\lambda_i}$; de translatie $z_i = y_i +
\beta_i/\lambda_i$ (en $z_i = y_i$ voor $i > r$) geeft
\[
\sum_{i \leq r}\lambda_iz_i^2 + 2\sum_{i > r}\beta_iz_i +
c'' = 0, \qquad
c'' = c - \sum_{i\leq r}\frac{\beta_i^2}{\lambda_i}.
\]

\textbf{4.} $q(2\Omega - X) = (2\Omega - X)^{\mathsf T}
A(2\Omega - X) + 2b^{\mathsf T}(2\Omega - X) + c$;
ontwikkelen en $q(X)$ aftrekken, de kwadratische termen
heffen op en
\[
q(2\Omega - X) - q(X) = -4\,\langle A\Omega + b,\ X\rangle +
4\,\langle A\Omega + b,\ \Omega\rangle .
\]
Als $A\Omega + b = 0$ verdwijnt dit identiek: de
puntsymmetrie behoudt $q$, dus de \omterm{pb:b2:surfaces:1}{kwadriek}. Omgekeerd zegt
``$q(2\Omega - X) = q(X)$ voor alle $X$'' dat de affiene
functie hierboven op heel $\R^3$ verdwijnt, wat haar
lineaire deel $A\Omega + b$ tot nul dwingt. Dus centra $=$
oplossingen van $A\Omega = -b$: een niet-lege verzameling
dan en slechts dan als $b \in \operatorname{im}A$ (een
affiene deelruimte gericht door $\ker A$), en een enkel
punt dan en slechts dan als $A$ inverteerbaar is.

\textbf{5.} Signatuur $(3,0)$ (alle $\lambda_i > 0$):
$\delta > 0$ geeft $\frac{x^2}{a^2} + \frac{y^2}{b^2} +
\frac{z^2}{c^2} = 1$ met $a = \sqrt{\delta/\lambda_1}$, enz.\
--- een ellipsoïde; $\delta = 0$: het enkele punt $O$;
$\delta < 0$: leeg. Signatuur $(2,1)$ ($\lambda_1, \lambda_2
> 0 > \lambda_3$): $\delta > 0$: $\frac{x^2}{a^2} +
\frac{y^2}{b^2} - \frac{z^2}{c^2} = 1$, de éénbladige
hyperboloïde; $\delta = 0$: de kegel $\frac{x^2}{a^2} +
\frac{y^2}{b^2} = \frac{z^2}{c^2}$; $\delta < 0$:
$\frac{z^2}{c^2} - \frac{x^2}{a^2} - \frac{y^2}{b^2} = 1$,
de tweeladige hyperboloïde ($\abs z \geq c$: twee
componenten).

\textbf{6.} Het kwadratische deel heeft matrix $A$ met
diagonaal $1$ en buiten-diagonaal $2$: $A = 2J - I$. Omdat
$J$ spectrum $\{3, 0, 0\}$ heeft (\omterm{def:b2:reduction:eigen}{eigenvector} $(1,1,1)$ voor
$3$), heeft $A$ spectrum $\{5, -1, -1\}$, de \omterm{def:b2:reduction:eigen}{eigenwaarde} $5$
gedragen door $\R(1,1,1)$. In de geroteerde coördinaten:
$5u^2 - v^2 - w^2 = 1$, d.w.z.\ $\frac{u^2}{1/5} - v^2 -
w^2 = 1$: een tweeladige hyperboloïde, van omwenteling
(gelijke \omterm{def:b2:reduction:eigen}{eigenwaarden} $-1$) om de as $\R(1,1,1)$.

\textbf{7.} Het oppervlak is $\bigl(\frac xa - \frac
zc\bigr)\bigl(\frac xa + \frac zc\bigr) = \bigl(1 - \frac
yb\bigr)\bigl(1 + \frac yb\bigr)$. Voor $(\lambda : \mu)
\neq (0:0)$ definieer de lijn
\[
D_{\lambda:\mu} :\quad
\lambda\Bigl(\frac xa - \frac zc\Bigr) = \mu\Bigl(1 - \frac
yb\Bigr), \qquad
\mu\Bigl(\frac xa + \frac zc\Bigr) = \lambda\Bigl(1 + \frac
yb\Bigr)
\]
(twee onafhankelijke affiene vergelijkingen: een lijn).
Vermenigvuldigen van de twee vergelijkingen toont dat elk
punt van $D_{\lambda:\mu}$ op het oppervlak ligt wanneer
$\lambda\mu \neq 0$; de gevallen $\lambda = 0$ of $\mu = 0$
worden direct gecontroleerd (bijv.\ $\lambda = 0$: $y = b$,
$\frac xa = -\frac zc$, wat aan de vergelijking voldoet). De
tweede familie $D'_{\lambda:\mu}$ verwisselt de twee
rechterfactoren.

\textbf{8.} Fixeer $M$ op het oppervlak. De voorwaarden voor
$M \in D_{\lambda:\mu}$ vormen een $2\times2$ homogeen
lineair stelsel in $(\lambda, \mu)$ waarvan de \omterm{def:b2:linalg:det}{determinant}
\[
\Bigl(\frac{x^2}{a^2} - \frac{z^2}{c^2}\Bigr) - \Bigl(1 -
\frac{y^2}{b^2}\Bigr) = 0
\]
is precies omdat $M$ op de \omterm{pb:b2:surfaces:1}{kwadriek} ligt: een niet-triviale
oplossing $(\lambda : \mu)$ bestaat. De
coëfficiëntenmatrix is nooit nul (dat zou $1 - \frac yb = 1
+ \frac yb = 0$ dwingen), dus is haar rang $1$ en de
oplossing uniek op schaal na: precies één lijn van de
familie gaat door $M$. Hetzelfde geldt voor de tweede
familie, en de twee lijnen zijn onderscheiden (in $(a, 0,
0)$ zijn ze $\{x = a,\ \frac zc = \frac yb\}$ en $\{x = a,\
\frac zc = -\frac yb\}$): de éénbladige hyperboloïde is
dubbel geregeerd.

\textbf{9.} Zij $M = (x, y, z)$ op de hyperboloïde, $\rho^2
= \frac{x^2}{a^2} + \frac{y^2}{b^2} = 1 + \frac{z^2}{c^2}
\geq 1$. Het punt $N = (x,\ y,\ \varepsilon c\rho)$ met
$\varepsilon$ het teken van $z$ voldoet aan $\frac{x^2}{a^2}
+ \frac{y^2}{b^2} - \frac{(c\rho)^2}{c^2} = 0$: $N \in C$,
en
\[
d(M, C) \leq \abs{z - \varepsilon c\rho}
= c\bigl(\rho - \sqrt{\rho^2 - 1}\bigr)
= \frac{c}{\rho + \sqrt{\rho^2 - 1}} .
\]
Als $\norm M \to \infty$ dan $\rho \to \infty$ (alle drie
de coördinaten zijn begrensd door veelvouden van $\rho$),
dus $d(M, C) \to 0$. De snede $x = a$: $\frac{y^2}{b^2} -
\frac{z^2}{c^2} = 0$, het paar kruisende lijnen van vraag 8
--- en evenzo in $x = -a$.

\textbf{10.} Met $\lambda_3 = 0$ laat vraag 3
$\lambda_1z_1^2 + \lambda_2z_2^2 + 2\beta_3z_3 + c'' = 0$
over. In de eigenbasis, $\operatorname{im}A =
\operatorname{Vect}(e_1, e_2)$, dus bestaan centra door
vraag 4 dan en slechts dan als $\beta_3 = 0$. Als $\beta_3
\neq 0$: de translatie $z_3 \mapsto z_3 - c''/(2\beta_3)$
verwijdert de constante, en laat $\lambda_1z_1^2 +
\lambda_2z_2^2 + 2\beta_3z_3 = 0$ over, d.w.z.\ $z_3 = px^2
+ qy^2$ na hernoemen: een \emph{elliptische paraboloïde} als
$\lambda_1\lambda_2 > 0$, een \emph{hyperbolische
paraboloïde} als $\lambda_1\lambda_2 < 0$ --- en inderdaad
geen centrum. Als $\beta_3 = 0$: de vergelijking
$\lambda_1z_1^2 + \lambda_2z_2^2 + c'' = 0$ betrekt $z_3$
niet: de \omterm{pb:b2:surfaces:1}{kwadriek} is een cilinder over de corresponderende
vlakke kegelsnede --- elliptische cilinder, lijn, of lege
verzameling wanneer $\lambda_1\lambda_2 > 0$; hyperbolische
cilinder of paar snijdende vlakken wanneer
$\lambda_1\lambda_2 < 0$ --- met een hele lijn van centra
$\{(z_1^*, z_2^*)\} \times \R$.

\textbf{11.} $xy - z = 0$. Substitueren $x = \frac{u +
v}{\sqrt2}$, $y = \frac{u - v}{\sqrt2}$ (rotatie over
$\pi/4$): $xy = \frac{u^2 - v^2}2$, dus wordt de
vergelijking $z = \frac12(u^2 - v^2)$: een hyperbolische
paraboloïde --- het zadel van de figuur, op de factor
$\frac12$ na.

\textbf{12.} De lijn $\{x = x_0,\ z = x_0y\}$ (geparametriseerd
door $y$) ligt duidelijk op $z = xy$, evenals $\{y = y_0,\
z = xy_0\}$; door $(x_0, y_0, x_0y_0)$ gaan de twee ervan.
Uniciteit: als $t \mapsto (x_0 + tv_1, y_0 + tv_2, z_0 +
tv_3)$ op het oppervlak blijft, geeft de coëfficiënt van
$t^2$ in $(x_0 + tv_1)(y_0 + tv_2) - z_0 - tv_3$ $v_1v_2 =
0$, dus $v_1 = 0$ of $v_2 = 0$, landend in één van de twee
families: één lijn van elke door elk punt --- de tweede
dubbel geregeerde \omterm{pb:b2:surfaces:1}{kwadriek}.

\textbf{13.} Kwadraten afmaken: $(x - 1)^2 + (y + 2)^2 = 2$,
zonder voorwaarde op $z$: een rechte circulaire cilinder
van straal $\sqrt2$ en as de verticale lijn $\{(1, -2, z) :
z \in \R\}$ --- een lijn van centra, zoals vraag 10
voorspelt.

\textbf{14.} Met $\lambda_2 = \lambda_3 = 0$ is de
gereduceerde vergelijking $\lambda_1z_1^2 + 2\beta_2z_2 +
2\beta_3z_3 + c'' = 0$. Een rotatie van het kernvlak
$(z_2, z_3)$ richt de lineaire vorm: $2\beta_2z_2 +
2\beta_3z_3 = 2\beta w$ met $\beta = \sqrt{\beta_2^2 +
\beta_3^2}$. Als $\beta \neq 0$, translateren van $w$ om
$c''$ te absorberen: $\lambda_1z_1^2 + 2\beta w = 0$, een
\emph{parabolische cilinder}; als $\beta = 0$:
$\lambda_1z_1^2 = -c''$ geeft twee evenwijdige vlakken
($c'' \lambda_1 < 0$), een dubbel vlak ($c'' = 0$), of de
lege verzameling. Voor $(x + y)^2 = z$: met $u = \frac{x +
y}{\sqrt2}$ leest de vergelijking $z = 2u^2$: een
parabolische cilinder, invariant onder translaties langs
$(1, -1, 0)$.

\textbf{15.} Elke \omterm{pb:b2:surfaces:1}{kwadriek} wordt door een rotatie plus
translaties op één van de volgende afgebeeld: (rang 3)
ellipsoïde, punt, lege verzameling, éénbladige
hyperboloïde, kegel, tweeladige hyperboloïde; (rang 2)
elliptische paraboloïde, hyperbolische paraboloïde,
elliptische cilinder, lijn, lege verzameling, hyperbolische
cilinder, paar snijdende vlakken; (rang 1) parabolische
cilinder, paar evenwijdige vlakken, dubbel vlak, lege
verzameling. De drie lege varianten als onderscheiden
\emph{affiene types} van vergelijkingen identificerend, is
het aantal zeventien; onder hen zijn er negen eerlijke
oppervlakken: ellipsoïde, de twee hyperboloïden, de kegel,
de twee paraboloïden, en de drie cilinders.

\textbf{16.} Algoritme. (i) Lees $(A, b, c)$ af; bereken het
karakteristieke polynoom van $A$, haar \omterm{def:b2:reduction:eigen}{eigenwaarden} (reëel,
door de spectrale stelling) en $r = \operatorname{rank}A$.
(ii) Los $A\Omega = -b$ op (Gauss-eliminatie): oplosbaar of
niet --- centra of niet. (iii) Als oplosbaar, translateren
naar een centrum: de vergelijking wordt $\sum\lambda_iz_i^2
+ c'' = 0$ met $c'' = c + \langle b, \Omega\rangle$; sorteer
op $r$, de signatuur, en het teken van $c''$ met vragen 5,
10, 14. (iv) Als niet oplosbaar ($r \leq 2$), roteer en
reduceer zoals in vragen 10 en 14: paraboloïde ($r = 2$) of
parabolische cilinder ($r = 1$), elliptisch/hyperbolisch
naargelang het teken van $\lambda_1\lambda_2$. Elke stap is
een eindige berekening: wortels van een kubische met reële
wortels, rangen, lineaire stelsels.

\textbf{17.} $A$ heeft diagonaal $1$, buiten-diagonaal $-1$:
$A = 2I - J$, spectrum $\{2 - 3,\ 2,\ 2\} = \{-1, 2, 2\}$
met $-1$ op $\R(1,1,1)$. Signatuur $(2,1)$, $b = 0$,
rechterlid $\delta = 1 > 0$: $2u^2 + 2v^2 - w^2 = 1$, een
éénbladige hyperboloïde van omwenteling om de as
$\R(1,1,1)$.

\textbf{18.} Kwadraten afmaken: $(x-1)^2 + (y+2)^2 -
(z-1)^2 + (-1 - 4 + 1 + 4) = 0$, d.w.z.
\[
(x-1)^2 + (y+2)^2 = (z-1)^2 :
\]
de constante verdween --- een rechte circulaire
\emph{kegel} met top (en uniek centrum) $(1, -2, 1)$ en as
evenwijdig aan $Oz$.

\textbf{19.} Het kwadratische deel $x^2 + 4xy + y^2$ heeft
matrix $\begin{psmallmatrix}1 & 2\\ 2 & 1\end{psmallmatrix}$
(in het $xy$-vlak), \omterm{def:b2:reduction:eigen}{eigenwaarden} $3$ (op $(1,1)$) en $-1$
(op $(1,-1)$): met $u = \frac{x+y}{\sqrt2}$, $v =
\frac{x-y}{\sqrt2}$ is ze $3u^2 - v^2$, en de \omterm{pb:b2:surfaces:1}{kwadriek} is
\[
z = \tfrac32u^2 - \tfrac12v^2 :
\]
een hyperbolische paraboloïde ($A$ heeft rang $2$ en $b$
heeft een component langs $\ker A = \R e_z$: geen centrum).

\textbf{20.} Een starre beweging transformeert de gegevens
van de vergelijking door $A \mapsto P^{\mathsf T}\!AP$
(zelfde \omterm{def:b2:reduction:eigen}{eigenwaarden}) en de normaalvormen hebben geen
restvrijheid behalve permuteren van coördinaten en
vermenigvuldigen van de hele vergelijking met een scalar
($> 0$ om de schrijfwijze te behouden): twee centrale
normaalvormen vallen op isometrie na samen dan en slechts
dan als de coëfficiëntenlijsten op permutatie en
gemeenschappelijke positieve factor na overeenkomen ---
voor de ellipsoïde, dan en slechts dan als de halfassen
$(a, b, c)$ overeenkomen. Affien mag men ook elke
coördinaat apart schalen ($z_i \mapsto
z_i\sqrt{\abs{\lambda_i}}$), wat de \omterm{def:b2:reduction:eigen}{eigenwaarden} wist en
alleen hun tekens overlaat: elke ellipsoïde wordt $u^2 +
v^2 + w^2 = 1$, de bol. De inertiestelling van Sylvester
(\cref{thm:b2:quadratic:sylvester}) garandeert dat de
signatuur elke inverteerbare lineaire wisseling overleeft:
de affiene types van vraag 15 zijn echt onderscheiden.

\textbf{21.} Parametriseer het vlak affien: $M = P + su +
tv$. Dan is $q(P + su + tv)$ een polynoom van graad $\leq 2$
in $(s, t)$ (ontwikkel de \omterm{def:b2:quadratic:def}{kwadratische vorm} bilineair), dus
is de snede $\{q = 0\}$ een kegelsnede van het vlak,
mogelijk ontaard. Voor $z = xy$ en het vlak $z = c$: $xy =
c$, een hyperbool voor $c \neq 0$, en voor $c = 0$ de twee
coördinaatlijnen --- het paar regellijnen door de
oorsprong.

\textbf{22.} \emph{Ellipsoïde}: begrensd, bevat geen lijn.
\emph{Tweeladige hyperboloïde} $\frac{z^2}{c^2} -
\frac{x^2}{a^2} - \frac{y^2}{b^2} = 1$: beperk tot $p +
tv$; de $t^2$-coëfficiënt $\frac{v_3^2}{c^2} -
\frac{v_1^2}{a^2} - \frac{v_2^2}{b^2}$ moet verdwijnen, dus
$v_3 \neq 0$ (anders $v = 0$); de $t$-coëfficiënt geeft
$\frac{p_3v_3}{c^2} = \frac{p_1v_1}{a^2} +
\frac{p_2v_2}{b^2}$, en Cauchy--Schwarz levert
\[
\frac{p_3^2}{c^2}
= \frac{c^2}{v_3^2}\Bigl(\frac{p_1v_1}{a^2} +
\frac{p_2v_2}{b^2}\Bigr)^2
\leq \frac{c^2}{v_3^2}\Bigl(\frac{p_1^2}{a^2} +
\frac{p_2^2}{b^2}\Bigr)\frac{v_3^2}{c^2}
= \frac{p_1^2}{a^2} + \frac{p_2^2}{b^2},
\]
dus is de constante term $\leq 0 \neq 1$: geen lijn.
\emph{Elliptische paraboloïde} $z = \frac{x^2}{a^2} +
\frac{y^2}{b^2}$: de $t^2$-coëfficiënt dwingt $v_1 = v_2 =
0$, daarna is de vergelijking lineair niet-constant in $t$:
geen lijn. \emph{Kegel} $x^2 + y^2 = z^2$: een lijn erop
voldoet aan $q(p) = q(v) = B(p, v) = 0$ voor de
Lorentz-vorm; gelijkheid in de vlakke Cauchy--Schwarz
$\abs{p_1v_1 + p_2v_2} = \abs{p_3v_3} =
\sqrt{(p_1^2 + p_2^2)(v_1^2 + v_2^2)}$ dwingt $(p_1, p_2)
\parallel (v_1, v_2)$ en daarna $p = kv$: alle lijnen gaan
door de top --- één familie. \emph{Cilinders}: voor de
elliptische en parabolische cilinders dwingt de
$t^2$-coëfficiënt $v_1 = v_2 = 0$ (alleen regellijnen); voor
de hyperbolische cilinder $\frac{x^2}{a^2} - \frac{y^2}{b^2}
= 1$ leidt $\frac{v_1}a = \pm \frac{v_2}b$ met $v_2 \neq 0$
via de $t$-coëfficiënt tot $\frac{p_1^2}{a^2} =
\frac{p_2^2}{b^2}$, in tegenspraak met de constante term
$1$: opnieuw alleen de verticale regellijnen. Dus zijn de
dubbel geregeerde kwadriekoppervlakken precies de
éénbladige hyperboloïde en de hyperbolische paraboloïde.

\textbf{23.} Gauss: $x^2 - 2xy - 2zx = (x - y - z)^2 - y^2 -
z^2 - 2yz$, dus
\[
q = (x - y - z)^2 - 4yz = (x - y - z)^2 + (y - z)^2 - (y +
z)^2 :
\]
drie onafhankelijke kwadraten met tekens $(+, +, -)$ ---
signatuur $(2, 1)$, overeenkomend met vraag 17, zonder
eigenwaardeberekening. Gauss verliest de metrische
gegevens: de nieuwe coördinaten zijn niet orthonormaal, dus
zijn de \omterm{def:b2:reduction:eigen}{eigenwaarden} (de vorm van de hyperboloïde, haar
assen en hun lengten) weg; alleen het affiene type blijft.

\textbf{24.} Zij $p$, $n$, $z$ de aantallen positieve,
negatieve en nuleigenwaarden, $p + n + z = 3$. De
tekenregel van Descartes begrenst $p$ door het aantal $V$
van tekenwisselingen van $\chi_A$, en $n$ door het aantal
$V'$ van tekenwisselingen van $\chi_A(-\lambda)$; bovendien
produceert elk paar opeenvolgende niet-nulle coëfficiënten
een wisseling in precies één van de twee polynomen, dus $V
+ V' \leq 3 - z$ (nulwortels zijn zichtbaar als verdwijnende
trailing coëfficiënten). Dan $p + n = 3 - z \geq V + V'
\geq p + n$: gelijkheid, dus $p = V$ precies --- de tekens
van de \omterm{def:b2:reduction:eigen}{eigenwaarden} kunnen worden afgelezen. Voor
$\chi_A(\lambda) = \lambda^3 - 3\lambda^2 - 9\lambda - 5$:
tekens $+,-,-,-$ geven $V = 1$, en $\chi_A(-\lambda) =
-\lambda^3 - 3\lambda^2 + 9\lambda - 5$ heeft tekens
$-,-,+,-$: $V' = 2$. Signatuur $(1, 2)$ --- consistent met
de exacte factorisatie $\chi_A = (\lambda - 5)(\lambda +
1)^2$ van vraag 6.

\textbf{25.} Algoritme: diagonaliseer het kwadratische deel
orthonormaal (spectrale stelling: dit is de enige
\omterm{def:b2:powerseries:analytic}{analytisch} diepe stap, en het is wat de classificatie
\emph{euclidisch} maakt); los $A\Omega = -b$ op om centrale
versus parabolische types te beslissen en het lineaire deel
weg te translateren waar mogelijk (affiene meetkunde); lees
het type van rang, signatuur en de constante (Sylvester
garandeert dat dit invarianten zijn); wanneer alleen het
affiene type telt, vervangt de \omterm{thm:b2:quadratic:gauss}{Gauss-reductie} de spectrale
stelling ten koste van de metrische informatie. In $\R^2$
classificeert dezelfde machine kegelsneden: ellips,
hyperbool, parabool, plus paren lijnen, een lijn, een
punt, en lege verzamelingen. In $\R^n$ verandert niets
behalve de boekhouding: de types worden geïndexeerd door de
signatuur van $A$, de ligging van $b$ ten opzichte van
$\operatorname{im}A$, en één constante --- met de
$(n{+}2)$-dimensionale gerande matrix van $(A, b, c)$ als
compacte invariant.
\end{solution}
