\section*{Hoofdstuk \ref{ch:b2:funcseq} --- Rijen en reeksen van
functies}

\begin{solution}{exo:b2:funcseq:1}
$f_n(x) = \frac{x}{1 + nx}$: puntsgewijze limiet $0$ op $\intcc{0}{1}$.
\omterm{def:b2:funcseq:def}{Uniform}: $f_n$ stijgt op $\intcc{0}{1}$ (afgeleide
$\frac{1}{(1+nx)^2} > 0$), dus $\norm{f_n}_\infty = f_n(1) =
\frac{1}{1+n} \to 0$: \emph{\omterm{def:b2:funcseq:def}{uniform}} op $\intcc{0}{1}$.

$g_n(x) = nx\,\eu^{-nx^2}$: puntsgewijze limiet $0$ (exponentieel
wint). Sup: $g_n' = n\eu^{-nx^2}(1 - 2nx^2)$ verdwijnt in $x_n =
\frac{1}{\sqrt{2n}}$, waar $g_n(x_n) = \sqrt{\frac n2}\,\eu^{-1/2}
\to \infty$: niet \omterm{def:b2:funcseq:def}{uniform} op $\intcc{0}{1}$ --- maar \omterm{def:b2:funcseq:def}{uniform} op
$\intco{\delta}{1}$, want daar $g_n(x) \leq n\,\eu^{-n\delta^2}
\to 0$.

$h_n(x) = x^n(1 - x^n)$: puntsgewijze limiet $0$ op $\intcc{0}{1}$
(beide factoren; in $x = 1$, $h_n = 0$). Sup: met $u = x^n \in
\intcc{0}{1}$, $u(1-u) \leq \frac14$ bereikt in $u = \frac12$,
d.w.z.\ $x = 2^{-1/n} \in \intoo{0}{1}$: $\norm{h_n}_\infty =
\frac14 \not\to 0$: niet \omterm{def:b2:funcseq:def}{uniform} op $\intcc{0}{1}$; \omterm{def:b2:funcseq:def}{uniform} op
$\intcc{0}{a}$ ($\sup \leq a^n \to 0$).
\end{solution}

\begin{solution}{exo:b2:funcseq:2}
$\int_0^1 nx\,\eu^{-nx^2}\dd x = \bigl[-\tfrac12
\eu^{-nx^2}\bigr]_0^1 = \frac{1 - \eu^{-n}}{2} \to \frac12$, terwijl
$\int_0^1 \lim g_n = 0$. Geen tegenspraak:
\cref{thm:b2:funcseq:integration} eist \emph{uniforme}
convergentie op het segment, die hier faalt (de bult van hoogte
$\sim\sqrt n$ glijdt naar $0$).
\end{solution}

\begin{solution}{exo:b2:funcseq:3}
\omterm{def:b2:funcseq:series}{Normale convergentie} op $\intcc{-1}{1}$: $\norm{x^n/n^2}_\infty =
\frac{1}{n^2}$, \omterm{def:b2:series:summable}{sommeerbaar}: $S$ is daar continu
(\cref{thm:b2:funcseq:seriestransfer}).

Afgeleide: de afgeleide reeks $\sum \frac{x^{n-1}}{n}$ convergeert
\omterm{def:b2:funcseq:series}{normaal} op elke $\intcc{-a}{a}$, $a < 1$ ($\sup =
\frac{a^{n-1}}{n}$): $S$ is $C^1$ op $\intoo{-1}{1}$ met
\[
S'(x) = \sum_{n\geq1} \frac{x^{n-1}}{n} = \frac1x \sum_{n\geq1}
\frac{x^n}{n} = -\frac{\ln(1 - x)}{x}
\qquad (0 < \abs x < 1),
\]
de laatste identiteit zijnde de logaritmische reeks van Jaar~1
(eerlijk herleid in \cref{ch:b2:powerseries}).
\end{solution}

\begin{solution}{exo:b2:funcseq:4}
Voor vaste $x > 0$ is de reeks alternerend met $\frac{1}{n + x}
\downarrow 0$: \omterm{def:b2:funcseq:def}{puntsgewijze convergentie}, en de resttermgrens
$\abs{R_N(x)} \leq \frac{1}{N + 1 + x} \leq \frac{1}{N+1}$ is
\emph{\omterm{def:b2:funcseq:def}{uniform}} op $\intco{\delta}{\infty}$ (zelfs op
$\intoo{0}{\infty}$): \omterm{def:b2:funcseq:def}{uniforme convergentie}. (Niet \omterm{def:b2:funcseq:series}{normaal}:
$\norm{\frac{(-1)^n}{n+x}}_\infty = \frac{1}{n + \delta}$,
divergent.) ContinuĂŻteit volgt uit
\cref{thm:b2:funcseq:seriestransfer}.

Functionaalvergelijking: herindexeer $F(x + 1)$ met $m = n + 1$:
\[
F(x+1) = \sum_{n\geq0} \frac{(-1)^n}{n + 1 + x}
= \sum_{m\geq1}\frac{(-1)^{m-1}}{m+x} ,
\]
dus, isolerend de $m = 0$-term van $F(x)$,
\[
F(x) + F(x+1)
= \frac{1}{x} + \sum_{m\geq1}
\frac{(-1)^m + (-1)^{m-1}}{m+x} = \frac1x .
\]
\end{solution}

\begin{solution}{exo:b2:funcseq:5}
Zet $g_n = f_n - f \geq 0$ (dalend in $n$, per hypothese; de
limiet is $0$ puntsgewijs); elke $g_n$ is continu. Fixeer
$\varepsilon > 0$ en zij $U_n = \{x : g_n(x) < \varepsilon\}$:
open (origineel van een open verzameling), stijgend ($g_{n+1}
\leq g_n$), en overdekkend $K$ (\omterm{def:b2:funcseq:def}{puntsgewijze convergentie}). Via
Borel--Lebesgue (\cref{thm:b2:metric:borellebesgue}) overdekken
eindig veel $U_{n_1} \subseteq \dots \subseteq U_{n_k}$ de $K$:
dus $K = U_{n_k}$, d.w.z.\ $\norm{g_{n_k}}_\infty \leq
\varepsilon$, en door monotonie $\norm{g_n}_\infty \leq
\varepsilon$ voor alle $n \geq n_k$: \omterm{def:b2:funcseq:def}{uniforme convergentie}.
(Monotonie is essentieel: de glijdende bulten van
\cref{exo:b2:funcseq:2} convergeren puntsgewijs op een \omterm{def:b2:metric:compact}{compact}
zonder uniformiteit.)
\end{solution}

\begin{solution}{exo:b2:funcseq:6}
Substitueer $u = nx$:
\[
\int_0^1 \frac{n f(x)}{1 + n^2x^2}\dd x
= \int_0^n \frac{f(u/n)}{1 + u^2}\,\dd u .
\]
De integranden $h_n(u) = \frac{f(u/n)}{1+u^2}\mathbf{1}_{u \leq n}$
convergeren puntsgewijs naar $\frac{f(0)}{1+u^2}$ (continuĂŻteit
van $f$ in $0$) en worden overheerst door
$\frac{\norm f_\infty}{1 + u^2}$, integreerbaar op
$\intco{0}{\infty}$: gedomineerde convergentie
(\cref{thm:b2:integration:dominated}) geeft de limiet
\[
\int_0^\infty \frac{f(0)}{1 + u^2}\dd u = \frac{\pi}{2} f(0) .
\]
(De kernen concentreren in $0$: een benaderde identiteit.)
\end{solution}

\begin{solution}{exo:b2:funcseq:7}
Voor $f(x) = x^2$, gebruik de tweede binomiale-identiteitenfamilie
uit het bewijs: $\sum_k k^2 p_k = n(n-1)x^2 + nx$. Dus
\[
B_n(f)(x) = \sum_k \frac{k^2}{n^2}\,p_k
= \frac{n(n-1)x^2 + nx}{n^2}
= x^2 + \frac{x(1 - x)}{n} :
\]
$B_n(f) - f = \frac{x(1-x)}{n}$, van \omterm{def:b2:nvs:norm}{sup-norm} $\frac{1}{4n} =
O\bigl(\frac1n\bigr)$, zoals voorspeld.
\end{solution}

\begin{solution}{exo:b2:funcseq:8}
Voor $\varepsilon = 1$ is er $N$ met $\norm{P_n - P_m}_{\infty,
\R} \leq 1$ voor $m, n \geq N$. Een begrensde veelterm op $\R$ is
constant (een niet-constante gaat naar $\pm\infty$): $P_n - P_m =
c_{n,m}$, constanten. Dus voor $n \geq N$: $P_n = P_N + c_n$ met
$c_n = P_n(0) - P_N(0)$ convergent (\omterm{def:b2:funcseq:def}{puntsgewijze convergentie} in
$0$). Dus $f = \lim P_n = P_N + \lim c_n$: een veelterm.
\end{solution}

\begin{solution}{exo:b2:funcseq:9}
(a) $\norm{(3/4)^n\varphi(4^n\cdot)}_\infty = \frac12
(3/4)^n$: \omterm{def:b2:funcseq:series}{normale convergentie}, dus $W$ is continu
(\cref{thm:b2:funcseq:seriestransfer}).

(b) Fixeer $x$, $m$; kies het teken van $h_m = \pm\frac12 4^{-m}$
zodat het segment $\intcc{4^mx}{4^m(x + h_m)}$ (van lengte
$\frac12$) geen halfgeheel getal bevat, waardoor $\varphi$ affien
van helling $\pm1$ is erop (mogelijk: een interval van lengte
$\frac12$ ontmoet hoogstens Ă©Ă©n halfgeheel punt; kies de kant die
het vermijdt).

Voor $n > m$: $4^n h_m = \pm\frac12 4^{n-m}$ is geheel, en
$\varphi$ is $1$-periodiek: de $n$-de term van het verschil
verdwijnt.

Voor $n = m$: $\abs{\varphi(4^m x + 4^m h_m) - \varphi(4^m x)} =
\abs{4^m h_m} = \frac12$ ($\varphi$ affien van helling $\pm 1$ op
het segment), dus de term draagt precies $(3/4)^m \cdot
\frac{1/2}{\abs{h_m}} = (3/4)^m\,4^m = 3^m$ bij in het quotiĂ«nt.

Voor $n < m$: de $1$-Lipschitz $\varphi$ geeft
$\bigl|(3/4)^n\bigl(\varphi(4^nx + 4^nh_m) -
\varphi(4^nx)\bigr)\bigr| \leq (3/4)^n 4^n\abs{h_m} =
3^n\abs{h_m}$: elk draagt hoogstens $3^n$ bij aan het quotiĂ«nt.

Dus
\[
\Bigl|\frac{W(x + h_m) - W(x)}{h_m}\Bigr|
\geq 3^m - \sum_{n=0}^{m-1} 3^n
= 3^m - \frac{3^m - 1}{2} = \frac{3^m + 1}{2}
\longrightarrow \infty .
\]
Als $W$ differentieerbaar was in $x$, zou elk
differentiequotiĂ«nt langs $h_m \to 0$ naar $W'(x)$ convergeren:
tegenspraak. $W$ is overal continu, nergens differentieerbaar.
\end{solution}

\begin{solution}{exo:b2:funcseq:10}
Puntsgewijs: voor $x \in \intco{0}{1}$ is de reeks meetkundig van
verhouding $-x$,
\[
\sum_{n\geq0}(-1)^n x^n(1-x) = \frac{1-x}{1+x},
\]
en in $x = 1$ verdwijnt elke term: som $0 = \frac{1-1}{2}$,
consistent --- de som is continu op $\intcc{0}{1}$.
Uniformiteit: de restterm is een meetkundige staart,
\[
\abs{R_N(x)} = \frac{x^{N+1}(1-x)}{1+x} \leq x^{N+1}(1 - x)
\leq \max_{\intcc01} t^{N+1}(1-t)
= \frac{1}{N+2}\Bigl(\frac{N+1}{N+2}\Bigr)^{\!N+1}
\leq \frac{1}{N+2} \to 0 ,
\]
\omterm{def:b2:funcseq:def}{uniform} in $x$. Niet \omterm{def:b2:funcseq:series}{normaal}: $\norm{u_n}_\infty =
\max x^n(1-x) = \frac{1}{n+1}\bigl(\frac{n}{n+1}\bigr)^n \sim
\frac{1}{\eu\,n}$, en $\sum \frac1{\eu n}$ divergeert. Contrast:
$\sum x^n(1-x)$ heeft partiĂ«le sommen $1 - x^{N+1}$, convergerend
puntsgewijs naar de \emph{discontinue} $\mathbf 1_{\intco01}$:
door \cref{thm:b2:funcseq:continuity} kan die convergentie niet
\omterm{def:b2:funcseq:def}{uniform} zijn op $\intcc{0}{1}$.
\end{solution}

\begin{solution}{exo:b2:funcseq:11}
De limiet $f$ is continu (\cref{thm:b2:funcseq:continuity}).
Dan
\[
\abs{f_n(x_n) - f(x)}
\leq \abs{f_n(x_n) - f(x_n)} + \abs{f(x_n) - f(x)}
\leq \norm{f_n - f}_\infty + \abs{f(x_n) - f(x)} ,
\]
en beide termen gaan naar $0$ (\omterm{def:b2:funcseq:def}{uniforme convergentie};
continuĂŻteit van $f$ in $x$). Tegenvoorbeeld onder louter
\omterm{def:b2:funcseq:def}{puntsgewijze convergentie}: $g_n(x) = nx\,\eu^{-nx^2} \to 0$
puntsgewijs op $\intcc{0}{1}$ met $g_n$ en de limiet continu,
toch in $x_n = \frac{1}{\sqrt{2n}} \to 0$:
\[
g_n(x_n) = \sqrt{\frac n2}\,\eu^{-1/2} \longrightarrow +\infty
\neq 0 = f(0) .
\]
\end{solution}

\begin{solution}{exo:b2:funcseq:12}
\begin{enumerate}
  \item Inductie. $n = 1$ is de definitie. Neem de formule aan
        voor $n$ en zet $g(x) = \int_0^x \frac{(x-t)^n}{n!}
        f(t)\dd t$. Voor een integrand continu in $(x,t)$ en
        $C^1$ in $x$ differentieert de \omterm{thm:b2:integration:continuity}{parameterintegraal} met
        variabele grens als
        \[
        g'(x) = \frac{(x-x)^n}{n!}f(x)
        + \int_0^x \frac{(x-t)^{n-1}}{(n-1)!}f(t)\dd t
        = T^nf(x)
        \]
        (splits $g(x+h) - g(x)$ in de strook $\int_x^{x+h}$, die
        $O(h\cdot\sup)$ is met de integrand die in $t = x$ als
        $h^n$ verdwijnt, en de vaste integraal van de
        $x$-toename, behandeld via de middelwaarde-ongelijkheid
        en continuĂŻteit). Ook $(T^{n+1}f)' = T^nf$
        (hoofdstelling) en $g(0) = T^{n+1}f(0) = 0$: twee
        primitieven van $T^nf$ die in $0$ verdwijnen vallen
        samen, dus $T^{n+1}f = g$. De grens:
        \[
        \abs{T^nf(x)} \leq \norm f_\infty
        \int_0^x \frac{(x-t)^{n-1}}{(n-1)!}\dd t
        = \norm f_\infty\,\frac{x^n}{n!}
        \leq \frac{\norm f_\infty}{n!} .
        \]
  \item $\sum_n \norm{T^nf}_\infty \leq \eu\,\norm f_\infty$:
        normale, dus uniforme, convergentie; $S$ is continu. De
        partiĂ«le sommen voldoen aan $S_N = f + T S_{N-1}$, en $T$
        is $1$-Lipschitz voor $\norm\cdot_\infty$
        ($\abs{Tg(x)} \leq x\norm g_\infty$): $N \to \infty$ in
        beide zijden geeft $S = f + TS$.
  \item Zet $V(x) = f(x) + \eu^x\int_0^x \eu^{-t}f(t)\dd t$.
        Dan is $V - f$ van klasse $C^1$ met $(V-f)'(x) =
        \eu^x\int_0^x\eu^{-t}f + f(x) = V(x)$, en $(TV)' = V$
        met $(V - f)(0) = TV(0) = 0$: dus $V - f = TV$, d.w.z.\
        $V$ lost de vergelijking op. Uniciteit: als $S_1, S_2$
        continue oplossingen zijn, voldoet $D = S_1 - S_2$ aan
        $D = TD$, dus $D = T^nD$ voor alle $n$ en $\norm D_\infty
        \leq \frac{\norm D_\infty}{n!} \to 0$: $D = 0$.
        Daarom
        \[
        \sum_{n\geq0} T^nf(x) = f(x) +
        \int_0^x \eu^{x-t}f(t)\,\dd t .
        \]
        (De reeks $\sum T^n$ is een meetkundige reeks van
        operatoren: een eerste voorproef van de resolvente
        $(\mathrm{Id} - T)^{-1}$, ontwikkeld in het boekdeel
        van Jaar~3.)
\end{enumerate}
\end{solution}

\begin{solution}{pb:b2:funcseq:1}
\textbf{1.} Lineariteit is duidelijk uit de formule.
Positiviteit: de gewichten $p_k(x) \geq 0$, dus $f \geq 0$
forceert $B_nf \geq 0$; monotonie volgt toegepast op $g - f$.
Grens: $\pm f \leq \norm f_\infty$ geeft $\pm B_nf \leq \norm
f_\infty B_ne_0 = \norm f_\infty$. Eindpunten: $p_k(0) =
\mathbf 1_{k=0}$ en $p_k(1) = \mathbf 1_{k=n}$, dus $B_nf(0) =
f(0)$, $B_nf(1) = f(1)$.

\textbf{2.} Differentieer $(x+y)^n = \sum_k\binom nk x^ky^{n-k}$
naar $x$, vermenigvuldig met $x$, en zet $y = 1 - x$:
\[
nx = \sum_k k\,p_k(x) ;
\]
tweemaal, vermenigvuldigend met $x^2$: $n(n-1)x^2 = \sum_k
k(k-1)p_k(x)$. Dus $B_ne_0 = 1$ (binomiaalstelling), $B_ne_1(x)
= \frac{nx}{n} = x$, en
\[
B_ne_2(x) = \frac{\sum_k k^2p_k}{n^2}
= \frac{n(n-1)x^2 + nx}{n^2}
= x^2 + \frac{x(1-x)}{n} .
\]

\textbf{3.} Ontwikkel:
\[
\sum_k\Bigl(\frac kn - x\Bigr)^{\!2} p_k
= B_ne_2(x) - 2x\,B_ne_1(x) + x^2
= \frac{x(1-x)}{n} .
\]
Cauchy--Schwarz met de splitsing $\abs{k/n - x}\sqrt{p_k}
\cdot \sqrt{p_k}$:
\[
\sum_k\Bigl|\frac kn - x\Bigr| p_k
\leq \Bigl(\sum_k\Bigl(\frac kn - x\Bigr)^2
p_k\Bigr)^{\!1/2}
= \sqrt{\frac{x(1-x)}{n}} \leq \frac{1}{2\sqrt n},
\]
gebruikend $x(1-x) \leq \frac14$.

\textbf{4.} De gewichten $p_k(x)$ zijn niet-negatief met som $1$
en barycentrum $\sum_k \frac kn p_k(x) = x$ (vraag~2). De
eindige ongelijkheid van Jensen voor de convexe $f$ (inductie
vanuit de twee-puntsdefinitie, boekdeel van Jaar~1) geeft
\[
f(x) = f\Bigl(\sum_k \frac kn\,p_k\Bigr)
\leq \sum_k f\Bigl(\frac kn\Bigr)p_k = B_nf(x) .
\]

\textbf{5.} Op $\{k : \abs{k/n - x} > \delta\}$ heeft men
$\bigl(\frac{k/n - x}{\delta}\bigr)^2 > 1$, dus
\[
\sum_{\abs{k/n-x}>\delta} p_k
\leq \frac{1}{\delta^2}\sum_k\Bigl(\frac kn -
x\Bigr)^2p_k
= \frac{x(1-x)}{n\delta^2} \leq \frac{1}{4n\delta^2} .
\]
Lezing: $p_k(x)$ is de wet van een steekproeffrequentie $S_n/n$
van $n$ muntworpen van bias $x$; haar gemiddelde is $x$, haar
variantie $\frac{x(1-x)}n \to 0$, en de display is de
ongelijkheid van Chebyshev: de massa concentreert in $x$, dus
middelen van $f$ ertegen reproduceert $f(x)$ in de limiet
(\cref{ch:b2:genfun} maakt de vocabulaire officieel).

\textbf{6.} $\omega \leq 2M < \infty$; monotonie is duidelijk
(sup over een grotere verzameling). Heine: $f$ continu op een
\omterm{def:b2:metric:compact}{compact} is uniform continu, wat precies $\omega(\delta) \to 0$
als $\delta \to 0^+$ zegt. Subadditiviteit: als $\abs{s - t}
\leq \delta_1 + \delta_2$, voldoet het punt $u$ op het segment
$\intcc st$ op afstand $\min(\delta_1, \abs{s-t})$ van $s$ aan
$\abs{s-u} \leq \delta_1$, $\abs{u-t} \leq \delta_2$, en
$\abs{f(s)-f(t)} \leq \abs{f(s)-f(u)} + \abs{f(u)-f(t)}$.
Itererend, $\omega(p\delta) \leq p\,\omega(\delta)$ voor $p \in
\N^*$; voor $\lambda > 0$, met $p = \lceil\lambda\rceil \leq 1
+ \lambda$: $\omega(\lambda\delta) \leq \omega(p\delta) \leq
p\,\omega(\delta) \leq (1+\lambda)\omega(\delta)$.

\textbf{7.} Aangezien $\sum p_k = 1$:
\[
\abs{B_nf(x) - f(x)}
= \Bigl|\sum_k\bigl(f(k/n) - f(x)\bigr)p_k\Bigr|
\leq \sum_k\omega\bigl(\abs{k/n - x}\bigr)p_k .
\]
Voor elke $k$ geeft vraag~6 met $\lambda = \abs{k/n - x}/\delta$
dat $\omega(\abs{k/n-x}) \leq \bigl(1 +
\frac{\abs{k/n-x}}\delta\bigr)\omega(\delta)$; sommeren tegen
de $p_k$ levert de meesterschatting.

\textbf{8.} Voeg de grens van vraag~3 in:
\[
\abs{B_nf(x) - f(x)} \leq \Bigl(1 +
\frac{1}{2\delta\sqrt n}\Bigr)\omega(\delta),
\]
\omterm{def:b2:funcseq:def}{uniform} in $x$; met $\delta = n^{-1/2}$ is de haak $\frac32$:
$\norm{B_nf - f}_\infty \leq \frac32\omega(n^{-1/2}) \to 0$
door vraag~6 (Heine). Dit herbewijst
\cref{thm:b2:funcseq:weierstrass} met een snelheid.

\textbf{9.} $L$-Lipschitz betekent $\omega(\delta) \leq L\delta$:
snelheid $\frac{3L}{2\sqrt n}$. $\alpha$-HĂ¶lder betekent
$\omega(\delta) \leq C\delta^\alpha$: snelheid $\frac{3C}{2}
n^{-\alpha/2}$.

\textbf{10.} Via $k\binom{2m}k = 2m\binom{2m-1}{k-1}$:
\[
\sum_{k=m+1}^{2m}k\binom{2m}k
= 2m\sum_{j=m}^{2m-1}\binom{2m-1}{j}
= 2m\cdot 2^{2m-2},
\]
omdat $j \mapsto 2m-1-j$ bijecteert $\{m,\dots,2m-1\}$ op
$\{0,\dots,m-1\}$, dus de som is de helft van $2^{2m-1}$. Ook
$\sum_{k=m+1}^{2m}\binom{2m}k = \frac{2^{2m} -
\binom{2m}m}{2}$ (zelfde symmetrie). Dus
\[
\sum_{k=m+1}^{2m}(k-m)\binom{2m}k
= m\,2^{2m-1} - m\,\frac{2^{2m} - \binom{2m}m}{2}
= \frac m2\binom{2m}m .
\]
De substitutie $k \mapsto 2m-k$ beeldt de termen met $k < m$ af
op die met $k > m$ (gelijke binomialen, gelijke $\abs{k-m}$):
de absolute som is tweemaal de eenzijdige som, $m\binom{2m}m$.

\textbf{11.} In $x = \frac12$, $p_k(\tfrac12) =
\binom{2m}k2^{-2m}$ en $f(\tfrac12) = 0$:
\[
B_{2m}f\Bigl(\frac12\Bigr)
= \sum_k\Bigl|\frac{k}{2m} - \frac12\Bigr|
\binom{2m}k 2^{-2m}
= \frac{2^{-2m}}{2m}\,m\binom{2m}m
= \frac{\binom{2m}m}{2\cdot4^m}
\sim \frac{1}{2\sqrt{\pi m}}
\]
door \cref{ex:b2:comparison:centralbinomial}. Aangezien
$\omega_f(\delta) = \delta$ hier (de functie is $1$-Lipschitz
en de grens wordt bereikt), voorspelt vraag~8 hoogstens
$\frac32(2m)^{-1/2}$: de ware fout $\frac{1}{2\sqrt{\pi m}}$
heeft precies de orde $n^{-1/2}$ --- de snelheid is scherp tot
op de constante.

\textbf{12.} $f \leq g$ geeft $g - f \geq 0$, dus $L(g-f) \geq
0$, d.w.z.\ $Lf \leq Lg$. Uit $-\abs f \leq f \leq \abs f$:
$-L\abs f \leq Lf \leq L\abs f$, d.w.z.\ $\abs{Lf} \leq
L\abs f$.

\textbf{13.} Via Heine kies $\delta$ met $\abs{f(s)-f(x)}
\leq \varepsilon$ wanneer $\abs{s-x} \leq \delta$. Als
$\abs{s - x} > \delta$, dan $\frac{(s-x)^2}{\delta^2} > 1$ en
$\abs{f(s)-f(x)} \leq 2M \leq \frac{2M}{\delta^2}(s-x)^2$. In
beide gevallen geldt de gestelde grens.

\textbf{14.} Fixeer $x$; vraag~13 zegt, als functies van $s$:
\[
-\varepsilon e_0 - \frac{2M}{\delta^2}q_x
\;\leq\; f - f(x)e_0
\;\leq\; \varepsilon e_0 + \frac{2M}{\delta^2}q_x,
\qquad q_x = e_2 - 2x\,e_1 + x^2e_0 .
\]
Pas de monotone lineaire $L_n$ toe (vraag~12) en evalueer in
$x$:
\[
\abs{L_nf(x) - f(x)L_ne_0(x)}
\leq \varepsilon L_ne_0(x)
+ \frac{2M}{\delta^2}\bigl(L_ne_2(x) - 2xL_ne_1(x)
+ x^2L_ne_0(x)\bigr) .
\]

\textbf{15.} Schrijf $\alpha_j = L_ne_j - e_j$, dus
$\norm{\alpha_j}_\infty \to 0$. Aangezien $e_2(x) - 2xe_1(x) +
x^2e_0(x) = 0$:
\[
L_ne_2(x) - 2xL_ne_1(x) + x^2L_ne_0(x)
= \alpha_2(x) - 2x\,\alpha_1(x) + x^2\alpha_0(x),
\]
van \omterm{def:b2:nvs:norm}{sup-norm} hoogstens $\norm{\alpha_2} + 2\norm{\alpha_1} +
\norm{\alpha_0} \to 0$. Ook $L_ne_0 \to e_0$ \omterm{def:b2:funcseq:def}{uniform}, dus
$L_ne_0 \leq 2$ voor grote $n$, en $\abs{f(x)}\abs{L_ne_0(x) -
1} \leq M\norm{\alpha_0} \to 0$. Assembleren met vraag~14:
voor grote $n$, \omterm{def:b2:funcseq:def}{uniform} in $x$,
\[
\abs{L_nf(x) - f(x)} \leq 2\varepsilon +
\frac{2M}{\delta^2}\,o(1) + M\,o(1) \leq 3\varepsilon :
\]
$L_nf \to f$ \omterm{def:b2:funcseq:def}{uniform} --- de stelling van Korovkin.

\textbf{16.} $B_ne_0 = e_0$ en $B_ne_1 = e_1$ exact, en
$\norm{B_ne_2 - e_2}_\infty = \max_x\frac{x(1-x)}{n} =
\frac{1}{4n} \to 0$ (vraag~2): Korovkin is van toepassing, en
Weierstrass volgt voor de derde maal.

\textbf{17.} $I_nf$ is lineair in $f$ (knoopwaarden zijn dat), en
op elke cel is de affiene interpolant van niet-negatieve
knoopwaarden niet-negatief: positief. $I_ne_0 = e_0$ en
$I_ne_1 = e_1$ omdat een affiene functie gelijk is aan haar
eigen interpolant. Op een cel $\intcc ab$ ($b - a = \frac1n$)
is de affiene interpolant van $e_2$ gelijk aan $L(t) = (a+b)t
- ab$, en
\[
L(t) - t^2 = (t-a)(b-t) \in
\intcc{0}{\tfrac{(b-a)^2}{4}} ,
\]
met maximum in het midden: $\norm{I_ne_2 - e_2}_\infty =
\frac{1}{4n^2} \to 0$. Korovkin: $I_nf \to f$ \omterm{def:b2:funcseq:def}{uniform} voor elke
continue $f$ --- polygonale benadering, zonder verdere
schatting.

\textbf{18.} Gegeven $f$ en $\varepsilon$: Weierstrass levert
een veelterm $P = \sum_{j=0}^d a_jx^j$ met $\norm{f -
P}_\infty \leq \frac\varepsilon2$; elk $a_j$ vervangen door een
rationaal $b_j$ met $\abs{a_j - b_j} \leq
\frac{\varepsilon}{2(d+1)}$ verplaatst de \omterm{def:b2:nvs:norm}{sup-norm} op
$\intcc{0}{1}$ met hoogstens $\frac\varepsilon2$. De verzameling
van veeltermen met rationale coĂ«fficiĂ«nten is een aftelbare
vereniging (over $d$) van \omterm{def:b2:structures:countable}{aftelbare verzamelingen}, dus
\omterm{def:b2:structures:countable}{aftelbaar}, en dicht: $C(\intcc{0}{1})$ is separabel.

\textbf{19.} Door lineariteit $\int_0^1 fP = 0$ voor elke
veelterm $P$. Kies veeltermen $P_n \to f$ \omterm{def:b2:funcseq:def}{uniform} (Weierstrass):
\[
\Bigl|\int_0^1 f^2\Bigr|
= \Bigl|\int_0^1 f\,(f - P_n)\Bigr|
\leq \norm f_\infty\,\norm{f - P_n}_\infty
\longrightarrow 0 ,
\]
dus $\int_0^1 f^2 = 0$. Als $f(x_0) \neq 0$, geeft continuĂŻteit
$f^2 \geq c > 0$ op een subinterval, in tegenspraak met de
verdwijnende integraal: $f = 0$. Bijgevolg vallen twee continue
functies met dezelfde momenten $\int f t^n$ samen.

\textbf{20.} Met $p_{n,k}(x) = \binom nk x^k(1-x)^{n-k}$ en de
conventies $p_{n-1,-1} = p_{n-1,n} = 0$, geven de productregel
en $k\binom nk = n\binom{n-1}{k-1}$, $(n-k)\binom nk =
n\binom{n-1}{k}$ dat
\[
p_{n,k}'(x) = n\bigl(p_{n-1,k-1}(x) - p_{n-1,k}(x)\bigr) .
\]
Sommeren tegen $f(k/n)$ en verschuiven van de index in de eerste
som (Abelsommatie):
\[
(B_nf)'(x) = n\sum_{j=0}^{n-1}\Bigl(f\Bigl(\frac{j+1}n\Bigr)
- f\Bigl(\frac jn\Bigr)\Bigr)p_{n-1,j}(x) .
\]

\textbf{21.} Via de middelwaardestelling, $f(\frac{j+1}n) -
f(\frac jn) = \frac1n f'(\xi_j)$ met $\xi_j \in
\intoo{j/n}{(j+1)/n}$, dus $(B_nf)'(x) = \sum_j
f'(\xi_j)\,p_{n-1,j}(x)$. De knoop $\frac{j}{n-1}$ ligt ook in
$\intcc{j/n}{(j+1)/n}$ (beide ongelijkheden reduceren tot $j \leq
n-1$), dus $\abs{\xi_j - \frac j{n-1}} \leq \frac1n$ en
\[
\bigl|(B_nf)'(x) - B_{n-1}(f')(x)\bigr|
\leq \sum_j\Bigl|f'(\xi_j) -
f'\Bigl(\frac{j}{n-1}\Bigr)\Bigr| p_{n-1,j}(x)
\leq \omega_{f'}\Bigl(\frac1n\Bigr) \longrightarrow 0
\]
\omterm{def:b2:funcseq:def}{uniform}. Aangezien $B_{n-1}(f') \to f'$ \omterm{def:b2:funcseq:def}{uniform}
(\cref{thm:b2:funcseq:weierstrass} toegepast op de continue
$f'$), geeft de driehoeksongelijkheid $(B_nf)' \to f'$
\omterm{def:b2:funcseq:def}{uniform}. De veeltermen $P_n = B_nf$ convergeren dan naar $f$
in de $C^1$-zin.

\textbf{22.} Taylor--Lagrange in $x$: $f(\frac kn) - f(x) =
f'(x)(\frac kn - x) + \frac{f''(\xi_k)}2(\frac kn - x)^2$.
Sommeren tegen $p_k$, de lineaire term sterft (vraag~2):
\[
\abs{B_nf(x) - f(x)}
\leq \frac{\norm{f''}_\infty}{2}\sum_k\Bigl(\frac kn -
x\Bigr)^2p_k
= \frac{\norm{f''}_\infty}{2}\cdot\frac{x(1-x)}{n}
\leq \frac{\norm{f''}_\infty}{8n} .
\]

\textbf{23.} Nog twee differentiaties van $(x+y)^n$ geven de
factoriĂ«le momenten, met $n_{(j)} = n(n-1)\cdots(n-j+1)$:
\[
\sum_k k_{(j)}\,p_k = n_{(j)}\,x^j \qquad (j = 3, 4),
\]
en $k^3 = k_{(3)} + 3k_{(2)} + k$, $k^4 = k_{(4)} + 6k_{(3)} +
7k_{(2)} + k$ zetten ze om in machtmomenten:
\[
\sum_k k^3p_k = n_{(3)}x^3 + 3n_{(2)}x^2 + nx,
\qquad
\sum_k k^4p_k = n_{(4)}x^4 + 6n_{(3)}x^3 + 7n_{(2)}x^2 + nx .
\]
Ontwikkelen van $(k - nx)^4$ en verzamelen (een geduldige maar
puur mechanische berekening met de vier machtmomenten):
\[
\sum_k(k-nx)^4p_k = nx(1-x)\bigl(1 + 3(n-2)x(1-x)\bigr) .
\]
Met $x(1-x) \leq \frac14$: de rechterkant is hoogstens
$\frac n4\bigl(1 + \frac{3n}4\bigr) = \frac{3n^2}{16} +
\frac n4 \leq n^2$ voor $n \geq 1$.

\textbf{24.} De Peano-vorm van Taylor in $x$ definieert $\eta(t)
= \frac{f(t) - f(x) - f'(x)(t-x) - \frac12f''(x)(t-x)^2}
{(t-x)^2}$ voor $t \neq x$, $\eta(x) = 0$: door Taylor--Lagrange
is $\eta(t) = \frac12\bigl(f''(\xi) - f''(x)\bigr)$ voor enige
$\xi$ tussen $t$ en $x$, dus $\abs\eta \leq \norm{f''}_\infty$
en $\eta(t) \to 0$ als $t \to x$ (continuĂŻteit van $f''$).
Sommeren van de ontwikkeling tegen $p_k$ en gebruiken van
vragen~2--3:
\[
n\bigl(B_nf(x) - f(x)\bigr)
= \frac{x(1-x)}{2}f''(x)
+ n\sum_k\eta\Bigl(\frac kn\Bigr)\Bigl(\frac kn -
x\Bigr)^2p_k .
\]
Gegeven $\varepsilon$, kies $\delta$ met $\abs\eta \leq
\varepsilon$ op $\abs{t - x}\leq\delta$. Dichtbij-deel: hoogstens
$\varepsilon\,n\cdot\frac{x(1-x)}n \leq \varepsilon$. Ver-deel:
met $C = \norm{f''}_\infty$ en vraag~23,
\[
n\,C\sum_{\abs{k/n-x}>\delta}\Bigl(\frac kn -
x\Bigr)^2p_k
\leq \frac{nC}{\delta^2}\sum_k\Bigl(\frac kn -
x\Bigr)^4p_k
= \frac{nC}{\delta^2 n^4}\sum_k(k-nx)^4p_k
\leq \frac{C}{\delta^2 n} \longrightarrow 0 .
\]
Dus $n(B_nf(x) - f(x)) \to \frac{x(1-x)}2f''(x)$ --- de
stelling van Voronovskaya. Voor $f = e_2$ is dit exact bij elke
$n$ (\cref{exo:b2:funcseq:7}): het $\frac1n$-plafond is reĂ«el.

\textbf{25.} (i) Positiviteit zette puntsgewijze ongelijkheden
om in operatorongelijkheden: zij gaf de normgrens, Jensen,
Chebyshev, en heel Korovkin --- lineariteit alleen bewijst hier
niets. (ii) Compactheid kwam binnen via Heine (vragen~6, 13),
via begrensdheid van $f$, en via de \omterm{def:b2:nvs:norm}{norm} $\norm\cdot_\infty$
die eindig is. (iii) Drie testfuncties volstaan omdat
positiviteit alles reduceert tot het controleren van $L_n$ op
de enige familie $(s-x)^2 = e_2 - 2xe_1 + x^2e_0$, waarvan de
span die van $e_0, e_1, e_2$ is. (iv) Bernstein convergeert met
de eerlijke snelheid $\omega(n^{-1/2})$ voor elke continue $f$
(scherp, vraag~11) maar verzadigt bij $\frac1n$ voor gladde $f$
(vraag~24); het Fourierhoofdstuk voert hetzelfde programma uit
voor periodieke functies met de kern van FejĂ©r --- een andere
positieve operator met dezelfde deugden en dezelfde bescheidenheid.
\end{solution}
