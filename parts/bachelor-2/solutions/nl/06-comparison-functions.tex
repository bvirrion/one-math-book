\section*{Hoofdstuk \ref{ch:b2:comparison} --- Vergelijking van functies}

\begin{solution}{exo:b2:comparison:1}
$\sqrt{x^2 + x + 1} = x\sqrt{1 + \tfrac1x + \tfrac{1}{x^2}} = x +
\frac12 + \frac38\cdot\frac1x + o\bigl(\frac1x\bigr)$ (binomiale
ontwikkeling: $\frac12 u - \frac18 u^2$ met $u = \frac1x +
\frac{1}{x^2}$ geeft $\frac{1}{2x} + \frac{1}{2x^2} -
\frac{1}{8x^2} = \frac{1}{2x} + \frac{3}{8x^2}$, dan vermenigvuldigen
met $x$).

$\ln(x^2 + x) - 2\ln x = \ln\bigl(1 + \tfrac1x\bigr) = \frac1x -
\frac{1}{2x^2} + o\bigl(\frac{1}{x^2}\bigr)$.

Derde functie: ontwikkel elke factor,
\[
\frac{x + \sin x}{x - \ln x}
= \Bigl(1 + \frac{\sin x}{x}\Bigr)
\Bigl(1 + \frac{\ln x}{x} + \frac{(\ln x)^2}{x^2} +
O\Bigl(\frac{(\ln x)^3}{x^3}\Bigr)\Bigr).
\]
Orden de bijdragen op de schaal in $+\infty$: $\frac{\ln x}{x}
\gg \frac{1}{x} \geq \bigl|\frac{\sin x}{x}\bigr| \gg \frac{(\ln
x)^2}{x^2}$. De twee termen na de leidende $1$ zijn daarom
$\frac{\ln x}{x}$, dan de begrensd-oscillerende term $\frac{\sin
x}{x}$:
\[
\frac{x + \sin x}{x - \ln x}
= 1 + \frac{\ln x}{x} + \frac{\sin x}{x}
+ O\Bigl(\frac{(\ln x)^2}{x^2}\Bigr).
\]
\end{solution}

\begin{solution}{exo:b2:comparison:2}
$f(t) = t^\alpha$ ($t \geq 1$).

$\alpha > -1$: divergentie, en door
\cref{thm:b2:comparison:seriesintegral}~(2), $\sum_{k\leq n}
k^\alpha = \frac{n^{\alpha+1}}{\alpha+1} + C + o(1)$ als $\alpha <
0$ (waar $f$ daalt); voor $\alpha \geq 0$ ($f$ stijgend) geeft
dezelfde inkleming met omgekeerde ongelijkheden $\sum_{k \leq n}
k^\alpha \sim \frac{n^{\alpha + 1}}{\alpha + 1}$.

$\alpha = -1$: $H_n = \ln n + \gamma + o(1)$
(\cref{ex:b2:comparison:harmonic}).

$\alpha < -1$: convergentie, met rest $\sum_{k > n} k^\alpha
\sim \frac{n^{\alpha+1}}{-(\alpha+1)}$ door de inkleming (1) (beide
integraalgrenzen zijn equivalent met die waarde).
\end{solution}

\begin{solution}{exo:b2:comparison:3}
Zij $v_n = H_n - \ln n - \gamma \to 0$. Dan
\[
v_n - v_{n+1} = \ln\frac{n+1}{n} - \frac{1}{n+1}
= \Bigl(\frac1n - \frac{1}{2n^2}\Bigr) - \Bigl(\frac1n -
\frac{1}{n^2}\Bigr) + O\Bigl(\frac{1}{n^3}\Bigr)
= \frac{1}{2n^2} + O\Bigl(\frac{1}{n^3}\Bigr),
\]
met $\frac{1}{n+1} = \frac1n - \frac{1}{n^2} + O(n^{-3})$. Aangezien
$v_n \to 0$, telescoperen van de staart:
\[
v_n = \sum_{k \geq n} (v_k - v_{k+1})
= \sum_{k\geq n} \Bigl(\frac{1}{2k^2} + O(k^{-3})\Bigr)
= \frac{1}{2n} + O\Bigl(\frac{1}{n^2}\Bigr),
\]
door \cref{thm:b2:comparison:seriesintegral}~(1) toegepast op
$t^{-2}$ (rest $\sim \frac1n$, gehalveerd) en op $t^{-3}$. Dus $H_n =
\ln n + \gamma + \frac{1}{2n} + o(\frac1n)$.
\end{solution}

\begin{solution}{exo:b2:comparison:4}
Stirling driemaal:
\[
\frac{(3n)!}{(n!)^3}
\sim \frac{\sqrt{6\pi n}\,(3n/\eu)^{3n}}
{(2\pi n)^{3/2}\,(n/\eu)^{3n}}
= \frac{\sqrt{6}\; 27^{\,n}}{2\pi n} \cdot
\frac{1}{\sqrt{2\pi n}}\cdot\sqrt{2\pi n}\;
= \frac{\sqrt3\,27^n}{2\pi n} .
\]
(Zorgvuldig: $\frac{\sqrt{6\pi n}}{(2\pi n)^{3/2}} =
\frac{\sqrt6}{(2\pi n)\sqrt{2\pi n}}\sqrt{\pi n} =
\frac{\sqrt3}{2\pi n}$.)

$\dfrac{n!}{n^n} \sim \sqrt{2\pi n}\,\eu^{-n}$.

$\sqrt[n]{n!} = \exp\bigl(\frac{\ln n!}{n}\bigr)$ met $\ln n! = n\ln
n - n + \frac12\ln(2\pi n) + o(1)$:
\[
\sqrt[n]{n!} = \exp\Bigl(\ln n - 1 + \frac{\ln(2\pi n)}{2n} +
o\Bigl(\frac{\ln n}{n}\Bigr)\Bigr)
= \frac{n}{\eu}\Bigl(1 + \frac{\ln(2\pi n)}{2n} +
o\Bigl(\frac{\ln n}{n}\Bigr)\Bigr).
\]
\end{solution}

\begin{solution}{exo:b2:comparison:5}
$g(x) = x^n + x - 1$ stijgt strikt op $\intcc{0}{1}$ van $-1$ naar
$1$: unieke wortel $x_n$. Aangezien $x_n^n = 1 - x_n \in
\intoo{0}{1}$: als $x_n \leq c < 1$ langs een deelrij, dan $x_n^n
\leq c^n \to 0$, dus $1 - x_n \to 0$: tegenspraak met $x_n \leq
c$. Dus $x_n \to 1$.

Schrijf $x_n = 1 - \varepsilon_n$, $\varepsilon_n \to 0^+$. De
vergelijking luidt $(1 - \varepsilon_n)^n = \varepsilon_n$, d.w.z.
\[
n\ln(1 - \varepsilon_n) = \ln \varepsilon_n
\quad\Longrightarrow\quad
-n\varepsilon_n\bigl(1 + o(1)\bigr) = \ln\varepsilon_n .
\]
Dus $n\varepsilon_n = -\ln\varepsilon_n\,(1 + o(1)) \to +\infty$, en
opnieuw logaritmen nemen: $\ln n + \ln\varepsilon_n =
\ln(-\ln\varepsilon_n) + o(1)$. Aangezien $\ln(-\ln \varepsilon_n) =
o(\ln(1/\varepsilon_n))$, geeft dit $\ln\varepsilon_n \sim -\ln
n$, vandaar $\varepsilon_n = \frac{-\ln\varepsilon_n}{n}(1 + o(1))
\sim \frac{\ln n}{n}$:
\[
x_n = 1 - \frac{\ln n}{n} + o\Bigl(\frac{\ln n}{n}\Bigr) .
\]
\end{solution}

\begin{solution}{exo:b2:comparison:6}
Exacte relatie: $\cot y_n = x_n = n\pi + \frac\pi2 - y_n$, met $y_n
\sim \frac{1}{n\pi}$ (\cref{ex:b2:comparison:tan}). Ontwikkel $\cot y
= \frac1y - \frac y3 + O(y^3)$:
\[
\frac{1}{y_n} - \frac{y_n}{3} + O(y_n^3) = n\pi + \frac\pi2 - y_n
\quad\Longrightarrow\quad
\frac{1}{y_n} = n\pi + \frac\pi2 + O\Bigl(\frac1n\Bigr),
\]
(de termen $-y_n$ en $-\frac{y_n}{3}$ zijn $O(\frac1n)$). Inverteer:
\[
y_n = \frac{1}{n\pi}\cdot\frac{1}{1 + \frac{1}{2n} + O(n^{-2})}
= \frac{1}{n\pi}\Bigl(1 - \frac{1}{2n} + O\Bigl(\frac{1}{n^2}\Bigr)\Bigr)
= \frac{1}{n\pi} - \frac{1}{2n^2\pi} + O\Bigl(\frac{1}{n^3}\Bigr).
\]
Dus
\[
x_n = n\pi + \frac{\pi}{2} - y_n
= n\pi + \frac\pi2 - \frac{1}{n\pi} + \frac{1}{2n^2\pi} +
o\Bigl(\frac{1}{n^2}\Bigr).
\]
\end{solution}

\begin{solution}{exo:b2:comparison:7}
Splits de twee grootste termen af:
\[
\sum_{k=0}^{n} k! = n! + (n-1)! + \sum_{k \leq n-2} k! ,
\qquad
\sum_{k\leq n-2} k! \leq (n-1)\,(n-2)! = (n-1)! .
\]
Dus $1 \leq \frac{1}{n!}\sum k! \leq 1 + \frac{2}{n}$: de limiet is
$1$. Verfijnen: $\frac{(n-1)!}{n!} = \frac1n$ en de ruwe grens
$\sum_{k \leq n-2}k! \leq (n-1)!$ kan op dezelfde manier worden
aangescherpt: $\sum_{k\leq n-2} k! = (n-2)!\,(1 + O(\frac1n)) =
O\bigl(\frac{n!}{n^2}\bigr)$. Dus
\[
\sum_{k=0}^{n} k! = n!\Bigl(1 + \frac1n + O\Bigl(\frac{1}{n^2}\Bigr)\Bigr).
\]
\end{solution}

\begin{solution}{exo:b2:comparison:8}
$(u_n)$ stijgt; als begrensd zou zij naar $\ell$ convergeren met
$\ell = \ell + \frac1\ell$: absurd. Dus $u_n \to \infty$.

Kwadraten: $u_{n+1}^2 = u_n^2 + 2 + u_n^{-2}$, dus
\[
u_n^2 = u_0^2 + 2n + \sum_{k=0}^{n-1} \frac{1}{u_k^2} .
\]
De som is $o(n)$ (termen gaan naar $0$, CesĂ ro), dus $u_n^2 \sim 2n$
en $u_n \sim \sqrt{2n}$.

Verfijning: $\frac{1}{u_k^2} \sim \frac{1}{2k}$, dus door
vergelijking (\cref{thm:b2:comparison:seriesintegral}, of
equivalenten van partiĂ«le sommen van positieve reeksen)
$\sum_{k<n} u_k^{-2} \sim \frac12 \ln n$. Dus
\[
u_n^2 = 2n + \frac{\ln n}{2}\,(1 + o(1)) + O(1)
\quad\Longrightarrow\quad
u_n = \sqrt{2n}\sqrt{1 + \frac{\ln n}{4n} + o\Bigl(\frac{\ln
n}{n}\Bigr)}
= \sqrt{2n}\Bigl(1 + \frac{\ln n}{8n} + o\Bigl(\frac{\ln
n}{n}\Bigr)\Bigr).
\]
\end{solution}

\begin{solution}{exo:b2:comparison:9}
Factoriseer $n$ en stel $t = \ln n$:
\[
S_n = \frac1n \sum_{k=1}^{n} \frac{1}{1 + t\,\frac kn} .
\]
Voor vaste $t$ is de som een Riemannsom van $u \mapsto \frac{1}{1 +
tu}$ op $\intcc{0}{1}$; de functie is monotoon in $u$, dus wordt de
Riemannsom ingeklemd door de integraal verschoven over Ă©Ă©n mesh:
\[
\int_0^1 \frac{\dd u}{1 + tu} - \frac1n
\leq S_n \leq \int_0^1 \frac{\dd u}{1 + tu} + \frac1n
\]
(vergelijking van Riemannsommen van een monotone functie met haar
integraal, geldig voor elke $n$ met haar eigen $t = \ln n$). Nu
$\int_0^1 \frac{\dd u}{1 + tu} = \frac{\ln(1 + t)}{t}$, en
$\frac1n = o\bigl(\frac{\ln t}{t}\bigr)$: dus
\[
S_n = \frac{\ln(1 + \ln n)}{\ln n} + O\Bigl(\frac 1n\Bigr)
\;\sim\; \frac{\ln\ln n}{\ln n} .
\]
\end{solution}

\begin{solution}{exo:b2:comparison:10}
Identiteit: $(\ln n)^{\ln n} = \eu^{\ln n\,\ln\ln n} =
\bigl(\eu^{\ln n}\bigr)^{\ln\ln n} = n^{\ln\ln n}$. Rangschikking:
vergelijk logaritmen. $\ln(n^2) = 2\ln n$; $\ln\bigl((\ln
n)^{\ln n}\bigr) = \ln n\ln\ln n$; $\ln(2^n) = n\ln2$;
$\ln(n!) = n\ln n - n + O(\ln n)$ (Stirling, of de ruwere
inkleming $\ln n! \sim n\ln n$); $\ln(n^n) = n\ln n$. Aangezien
$2\ln n = o(\ln n\ln\ln n)$, $\ln n\ln\ln n = o(n)$, $n\ln 2 =
o(n\ln n - n)$, en $n \ln n - n \sim n\ln n$ maar $n! / n^n \to
0$ (het verschil van logs is $-n + O(\ln n) \to -\infty$):
\[
n^2 = o\bigl((\ln n)^{\ln n}\bigr),\quad
(\ln n)^{\ln n} = o(2^n),\quad
2^n = o(n!),\quad
n! = o(n^n).
\]
(Voor elke stap: het verschil van logaritmen gaat naar $+\infty$,
dus de ratio naar $0$.)
\end{solution}

\begin{solution}{exo:b2:comparison:11}
Ontbind $\frac1{k^2} = \frac1{k(k+1)} + \frac1{k^2(k+1)}$ en
sommeer voor $k > n$:
\[
\sum_{k>n}\frac1{k^2} = \frac1{n+1} +
\sum_{k>n}\frac{1}{k^2(k+1)} ,
\]
de eerste som exact telescoperend ($\frac1{k(k+1)} = \frac1k -
\frac1{k+1}$). Voor de tweede: $\frac{1}{k^2(k+1)} = \frac1{k^3}
+ O\bigl(\frac1{k^4}\bigr)$ (aangezien $\frac{1}{k^2(k+1)} -
\frac1{k^3} = \frac{-1}{k^3(k+1)}$), en door de
integraalvergelijking $\sum_{k>n}\frac1{k^3} = \frac1{2n^2} +
O\bigl(\frac1{n^3}\bigr)$, $\sum_{k>n}\frac1{k^4} =
O\bigl(\frac1{n^3}\bigr)$. Dus
\[
\sum_{k>n}\frac1{k^2}
= \frac1{n+1} + \frac{1}{2n^2} + O\Bigl(\frac1{n^3}\Bigr)
= \frac1n - \frac1{n^2} + \frac{1}{2n^2} +
O\Bigl(\frac1{n^3}\Bigr)
= \frac1n - \frac{1}{2n^2} + O\Bigl(\frac1{n^3}\Bigr),
\]
met $\frac1{n+1} = \frac1n - \frac1{n^2} +
O\bigl(\frac1{n^3}\bigr)$.
\end{solution}

\begin{solution}{exo:b2:comparison:12}
$(u_n)$ stijgt; als zij begrensd was zou zij naar een eindige
$\ell$ convergeren met $\ell = \ell + \eu^{-\ell}$: onmogelijk. Dus
$u_n \to \infty$. Zij $v_n = \eu^{u_n} \to \infty$: dan
\[
v_{n+1} = \eu^{u_n + \eu^{-u_n}} = v_n\,\eu^{1/v_n}
= v_n\Bigl(1 + \frac1{v_n} + \frac1{2v_n^2} +
O\bigl(v_n^{-3}\bigr)\Bigr)
= v_n + 1 + \frac{1}{2v_n} + O\bigl(v_n^{-2}\bigr).
\]
Sommeren van $v_{k+1} - v_k = 1 + O(1)$ geeft eerst $v_n = n +
O(n)$, dus $v_n \geq cn$ uiteindelijk; opnieuw sommeren met
$\frac1{2v_k} = O(\frac1k)$ geeft $v_n = n + O(\ln n)$. Nog een
doorgang: $\frac{1}{2v_k} = \frac{1}{2k}\bigl(1 +
O\bigl(\tfrac{\ln k}k\bigr)\bigr)$, dus
\[
v_n = n + \sum_{k<n}\frac1{2k} + O(1) = n + \frac{\ln n}2 +
O(1).
\]
Ten slotte $u_n = \ln v_n = \ln n + \ln\Bigl(1 + \frac{\ln n}{2n} +
O\bigl(\tfrac1n\bigr)\Bigr) = \ln n + \frac{\ln n}{2n} +
O\bigl(\tfrac1n\bigr)$.
\end{solution}

\begin{solution}{pb:b2:comparison:1}
\textbf{1.} Aftrekken van de twee ontwikkelingen: $\sum_i (c_i -
c_i')\varphi_i = o(\varphi_k)$. Als zekere coĂ«fficiĂ«nt verschilt,
zij $i_0$ de eerste: delen door $\varphi_{i_0}$ en gebruiken
$\varphi_j = o(\varphi_{i_0})$ voor $j > i_0$ geeft $c_{i_0} -
c_{i_0}' = o(1)$: nul, tegenspraak. Voor de ontwikkeling: met $u =
\frac{\ln x}x \to 0$,
\[
\frac{1}{x - \ln x} = \frac1x\cdot\frac{1}{1 - u}
= \frac1x\bigl(1 + u + u^2 + O(u^3)\bigr)
= \frac1x + \frac{\ln x}{x^2} + \frac{(\ln x)^2}{x^3} +
o\Bigl(\frac{(\ln x)^2}{x^3}\Bigr).
\]
Geen $\frac c{x^2}$-term verschijnt omdat de ontwikkeling een
meetkundige reeks in $u = \frac{\ln x}{x}$ is: elke term draagt even
veel machten van $\ln x$ als van $\frac1x$ voorbij de eerste; de
schaal trede $\frac1{x^2}$ (coĂ«fficiĂ«nt van $(\ln x)^0$) is simpelweg
afwezig, met coĂ«fficiĂ«nt $0$.

\textbf{2.} $f(x) = \eu^x + x$ is continu, strikt stijgend, met
limieten $-\infty$ en $+\infty$: een bijectie $\R \to \R$, dus $x_n
= f^{-1}(n)$ bestaat en is uniek, en $x_n \to +\infty$ ($f^{-1}$
stijgt naar $+\infty$). Uit $\eu^{x_n} = n - x_n$: $x_n = \ln(n -
x_n) \leq \ln n$, dus $x_n/n \to 0$ en $x_n = \ln n + \ln(1 -
x_n/n) = \ln n + o(1) \sim \ln n$.

\textbf{3.} Schrijf $u_n = x_n/n$. Tweede doorgang: $u_n =
\frac{\ln n + o(1)}{n}$, dus
\[
x_n = \ln n + \ln(1 - u_n) = \ln n - u_n + O(u_n^2)
= \ln n - \frac{\ln n}{n} + o\Bigl(\frac{\ln n}n\Bigr).
\]
Derde doorgang: nu $u_n = \frac{\ln n}{n} - \frac{\ln n}{n^2} +
o\bigl(\frac{\ln n}{n^2}\bigr)$, en $\ln(1 - u_n) = -u_n -
\frac{u_n^2}2 + O(u_n^3)$:
\[
x_n = \ln n - \frac{\ln n}n + \frac{\ln n}{n^2}
- \frac{(\ln n)^2}{2n^2} + o\Bigl(\frac{(\ln
n)^2}{n^2}\Bigr)
= \ln n - \frac{\ln n}{n} - \frac{(\ln n)^2}{2n^2} +
o\Bigl(\frac{(\ln n)^2}{n^2}\Bigr),
\]
de term $\frac{\ln n}{n^2}$ wordt geabsorbeerd in
$o\bigl(\frac{(\ln n)^2}{n^2}\bigr)$.

\textbf{4.} Bij $n = 1000$: $\ln 1000 \approx 6.90776$ (fout
$7\cdot10^{-3}$); twee termen: $6.90085$ (fout $2\cdot10^{-5}$);
drie termen: $6.90082$ (fout onder $10^{-5}$), tegen $x_{1000}
\approx 6.90083$. Elke doorgang koopt ongeveer de voorspelde factor
$\frac{\ln n}{n}$.

\textbf{5.} Twee partiĂ«le integraties, startend van rechts: met
$\frac{\dd}{\dd t}\bigl[\tfrac12t(1-t)\bigr] = \tfrac12 - t$ en
$t(1-t)$ verdwijnend aan beide einden,
\[
\frac12\int_0^1 t(1-t)g''(t)\dd t
= -\int_0^1\Bigl(\frac12 - t\Bigr)g'(t)\dd t
= -\Bigl[\Bigl(\frac12 - t\Bigr)g\Bigr]_0^1 - \int_0^1 g
= \frac{g(0) + g(1)}2 - \int_0^1 g .
\]
Herschikt is dit de gestelde identiteit.

\textbf{6.} Bereken het increment, pas dan vraag 5 toe op $g(t) =
f(n + t)$:
\begin{align*}
E_{n+1} - E_n
&= f(n{+}1) - \int_n^{n+1}\!f - \frac{f(n{+}1) - f(n)}2 \\
&= \frac{f(n) + f(n{+}1)}2 - \int_n^{n+1}\!f
= \frac12\int_0^1 t(1-t)f''(n+t)\dd t .
\end{align*}
Aangezien $0 \leq t(1-t) \leq \frac14$: $\abs{E_{n+1} - E_n} \leq
\frac18\int_n^{n+1}\abs{f''}$, waarvan de som over $n$ convergeert
door de hypothese: $(E_n)$ convergeert (absoluut sommeerbare
incrementen) naar zekere $E$, met
\[
\abs{E - E_n} \leq \sum_{k\geq n}\abs{E_{k+1} - E_k} \leq
\frac18\int_n^\infty\abs{f''} .
\]

\textbf{7.} $f(t) = \frac1t$: $f''(t) = \frac2{t^3}$,
$\int_1^\infty\abs{f''} = 1 < \infty$. Vraag 6:
\[
H_n = \ln n + \frac{1 + \frac1n}{2} + E + (E_n - E)
= \ln n + \Bigl(E + \frac12\Bigr) + \frac1{2n} +
\varepsilon_n,
\]
met $\abs{\varepsilon_n} = \abs{E_n - E} \leq
\frac18\int_n^\infty\frac{2\dd t}{t^3} = \frac1{8n^2}$.
Vergelijken met $H_n = \ln n + \gamma + o(1)$
(\cref{ex:b2:comparison:harmonic}) identificeert $E + \frac12 =
\gamma$.

\textbf{8.} Uit de incrementformule van vraag 6,
\[
\varepsilon_n = E_n - E = -\sum_{k\geq n}\frac12\int_0^1
t(1-t)\,\frac{2\,\dd t}{(k+t)^3}
= -\sum_{k \geq n}\Bigl(\frac1{k^3}\int_0^1t(1-t)\dd t +
O\Bigl(\frac1{k^4}\Bigr)\Bigr),
\]
met $\frac{1}{(k+t)^3} = \frac1{k^3} + O\bigl(\frac1{k^4}\bigr)$
uniform voor $t \in \intcc01$. Met $\int_0^1 t(1-t) = \frac16$ en
$\sum_{k\geq n}\frac1{k^3} \sim \frac{1}{2n^2}$
(\cref{thm:b2:comparison:seriesintegral}):
\[
\varepsilon_n = -\frac16\cdot\frac{1}{2n^2} +
o\Bigl(\frac1{n^2}\Bigr) = -\frac{1}{12n^2} +
o\Bigl(\frac{1}{n^2}\Bigr).
\]

\textbf{9.} $f = \ln$: $f''(t) = -\frac1{t^2}$, absoluut
integreerbaar. Vraag 6 geeft
\[
\ln n! = \int_1^n\ln t\,\dd t + \frac{\ln n}2 + E + O\Bigl(
\frac1{8}\int_n^\infty\frac{\dd t}{t^2}\Bigr)
= \Bigl(n + \frac12\Bigr)\ln n - n + 1 + E +
O\Bigl(\frac1n\Bigr),
\]
dus $d_n = 1 + E + O\bigl(\frac1n\bigr)$: convergentie van $(d_n)$
--- Stap 1 van \cref{thm:b2:comparison:stirling} --- plus de
snelheid $O(1/n)$. (De Stirlingwaarde van de limiet geeft $E =
\ln\sqrt{2\pi} - 1$.)

\textbf{10.} $f(t) = t^{-1/2}$: $f''(t) = \frac34 t^{-5/2}$,
absoluut integreerbaar. Vraag 6:
\[
\sum_{k=1}^n \frac1{\sqrt k}
= 2\sqrt n - 2 + \frac{1 + \frac1{\sqrt n}}2 + E +
O\bigl(n^{-3/2}\bigr)
= 2\sqrt n + c + \frac{1}{2\sqrt n} +
O\bigl(n^{-3/2}\bigr),
\]
met $c = E - \frac32$. Bij $n = 10^4$: $2\sqrt n = 200$, $c
\approx -1.46035$, $\frac1{2\sqrt n} = 0.005$: voorspeld
$198.54465$, en inderdaad $\sum_{k\leq10^4}k^{-1/2} =
198.544645\dots$ --- drie termen, zeven cijfers.

\textbf{11.} $t \mapsto t\ln t$ is continu en strikt stijgend op
$\intco1\infty$ (afgeleide $\ln t + 1 \geq 1$), van $0$ naar
$+\infty$: een unieke $x_n$ bestaat, en $x_n \to \infty$ (anders
zou $x_n\ln x_n$ begrensd blijven). Logaritmen nemen in $x_n\ln
x_n = n$: $\ln x_n + \ln\ln x_n = \ln n$; aangezien $\ln\ln x_n =
o(\ln x_n)$, geeft delen door $\ln x_n$ $\frac{\ln n}{\ln x_n} \to
1$: $\ln x_n \sim \ln n$.

\textbf{12.} Uit $x_n = \frac{n}{\ln x_n}$ en $\ln x_n \sim \ln
n$: $x_n \sim \frac{n}{\ln n}$. Volgende doorgang: $\ln\ln x_n =
\ln\bigl(\ln n\,(1 + o(1))\bigr) = \ln\ln n + o(1)$, dus $\ln
x_n = \ln n - \ln\ln n + o(1)$ en
\[
x_n = \frac{n}{\ln n - \ln\ln n + o(1)}
= \frac{n}{\ln n}\cdot\frac{1}{1 - \frac{\ln\ln n +
o(1)}{\ln n}}
= \frac{n}{\ln n}\Bigl(1 + \frac{\ln\ln n}{\ln n} +
o\Bigl(\frac{\ln\ln n}{\ln n}\Bigr)\Bigr).
\]

\textbf{13.} Bij $n = 10^6$: $\frac{n}{\ln n} \approx 72\,382$
($18\%$ ernaast), twee termen geven $\approx 86\,140$ ($1.9\%$
ernaast), tegen de ware $x \approx 87\,848$. De winst per doorgang
is slechts de factor $\frac{\ln\ln n}{\ln n} \approx
\frac{2.63}{13.8} \approx 0.19$: logaritmische schalen convergeren
met waanzinnige traagheid --- een feit van het leven overal waar
priemen betrokken zijn.

\textbf{14.} Er zijn precies $n$ priemen $\leq p_n$ (namelijk $p_1,
\dots, p_n$): $\pi(p_n) = n$. De priemgetalstelling (toegegeven;
volume van Jaar~3) geeft $n = \pi(p_n) \sim \frac{p_n}{\ln p_n}$,
d.w.z.\ $p_n \sim n\ln p_n$: dit is de vergelijking $x\ln x
\approx n$ achterstevoren gelezen. Logaritmen nemen: $\ln p_n =
\ln n + \ln\ln p_n + o(1)$, en $\ln\ln p_n = o(\ln p_n)$ forceert
$\ln p_n \sim \ln n$ zoals in vraag 11. Terugsubstitueren:
\[
p_n \sim n\ln p_n = n\,\ln n\,\frac{\ln p_n}{\ln n} \sim n\ln
n .
\]

\textbf{15.} (a) Fixeer $\varepsilon > 0$; voor grote $k$, $(1 -
\varepsilon)k\ln k \leq p_k \leq (1 + \varepsilon)k\ln k$. Door
vergelijking met de stijgende $t\ln t$
(\cref{thm:b2:comparison:seriesintegral}-achtige inkleming),
$\sum_{k\leq n}k\ln k = \int_1^n t\ln t\,\dd t + O(n\ln n) =
\frac{n^2\ln n}2 - \frac{n^2}4 + O(n\ln n) \sim
\frac{n^2\ln n}2$. Dus $\sum_{k\leq n}p_k = \frac{n^2\ln
n}{2}(1 + O(\varepsilon) + o(1))$ voor elke $\varepsilon$:
$\sum_{k\leq n}p_k \sim \frac{n^2\ln n}2$. (b) Door de
priemgetalstelling is onder de gehele getallen tot $10^{100}$ een
proportie $\sim \frac{1}{\ln 10^{100}} = \frac1{230.26\dots}$
priem: een uniform willekeurig $100$-cijferig geheel getal is
priem met kans ongeveer $\frac1{230}$.

\textbf{16.} Op $\intoo{n\pi}{n\pi + \frac\pi2}$ is $g(x) = \tan

\textbf{17.} $\tan z_n \sim z_n$ en $\frac1{x_n} \sim

\textbf{18.} $z_n = \arctan\frac1{x_n}$ en $\arctan u = u +

\[

\]

\textbf{19.} Steek $z_n = \frac1{n\pi} + O(n^{-3})$ in de vorige

\textbf{21.} $\sin u < u$ op $\intoo0\pi$ en $\sin$ beeldt

\textbf{24.} (a) Waar: $\ln u_n - \ln v_n = \ln\frac{u_n}{v_n}

\textbf{25.} De lus van \cref{met:b2:comparison:implicit} liep
