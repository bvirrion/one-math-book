\section*{Hoofdstuk \ref{ch:b2:randomvar} --- Discrete stochastische variabelen}

\begin{solution}{exo:b2:randomvar:1}
\emph{Binomiaal:} $X = \sum_{i=1}^n X_i$ met onafhankelijke
Bernoulli $X_i$; $V(X_i) = \E(X_i^2) - \E(X_i)^2 = p - p^2$, en
\omterm{def:b2:randomvar:variance}{varianties} van onafhankelijke variabelen tellen op
(\cref{thm:b2:randomvar:variancerules}):
\[
\E(X) = np, \qquad V(X) = np(1-p) .
\]
\emph{Poisson:} $\E\bigl(X(X-1)\bigr) =
\sum_{k\geq2}k(k-1)e^{-\lambda}\frac{\lambda^k}{k!} = \lambda^2
e^{-\lambda}\sum_{j\geq0}\frac{\lambda^j}{j!} = \lambda^2$, dus
\[
V(X) = \E(X^2) - \E(X)^2
= \lambda^2 + \lambda - \lambda^2 = \lambda .
\]
\emph{Geometrisch} ($q = 1 - p$): differentiëren van
$\sum_{k\geq0}q^k = \frac{1}{1-q}$ tweemaal binnen de schijf
(\cref{ch:b2:powerseries}), $\sum_{k\geq2}k(k-1)q^{k-2} =
\frac{2}{(1-q)^3}$, dus
\[
\E\bigl(X(X-1)\bigr) = pq\sum_{k\geq2}k(k-1)q^{k-2}
= \frac{2q}{p^2},
\qquad
V(X) = \frac{2q}{p^2} + \frac1p - \frac{1}{p^2}
= \frac{q}{p^2} = \frac{1-p}{p^2} .
\]
\end{solution}

\begin{solution}{exo:b2:randomvar:2}
\emph{Som:} voor $n \in \N$, door disjunctheid en
onafhankelijkheid,
\[
\P(X + Y = n)
= \sum_{k=0}^n \P(X = k)\P(Y = n - k)
= e^{-(\lambda + \mu)}\frac{1}{n!}
\sum_{k=0}^n \binom nk \lambda^k\mu^{n-k}
= e^{-(\lambda+\mu)}\frac{(\lambda + \mu)^n}{n!}
\]
door het binomium van Newton: $X + Y \sim \mathcal{P}(\lambda
+ \mu)$.
\emph{Voorwaardelijke \omterm{def:b2:randomvar:law}{wet}:} voor $0 \leq k \leq n$,
\[
\P(X = k \mid X + Y = n)
= \frac{\P(X = k)\P(Y = n - k)}{\P(X + Y = n)}
= \binom nk
\Bigl(\frac{\lambda}{\lambda+\mu}\Bigr)^{k}
\Bigl(\frac{\mu}{\lambda+\mu}\Bigr)^{n-k} ,
\]
de binomiale \omterm{def:b2:randomvar:law}{wet} $\mathcal{B}\bigl(n,
\frac{\lambda}{\lambda+\mu}\bigr)$: gegeven de totale telling
``kiest'' elke \omterm{def:b2:proba:space}{gebeurtenis} \omterm{def:b2:proba:independence}{onafhankelijk} de eerste bron met
waarschijnlijkheid evenredig aan haar tempo.
\end{solution}

\begin{solution}{exo:b2:randomvar:3}
Wanneer er nog $k$ speeltjes ontbreken, brengt elke nieuwe
doos een nieuw speeltje met waarschijnlijkheid $\frac kn$,
\omterm{def:b2:proba:independence}{onafhankelijk} van het verleden: de wachttijd $W_k$ tot het
volgende nieuwe speeltje is geometrisch
$\mathcal{G}\bigl(\frac kn\bigr)$, met $\E(W_k) = \frac nk$,
en $T_n = W_n + W_{n-1} + \dots + W_1$ (eerste doos geeft
altijd een nieuw speeltje: $W_n = 1$, consistent met $\E =
n/n$). Door lineariteit,
\[
\E(T_n) = \sum_{k=1}^n \frac nk = n\sum_{k=1}^n\frac1k
\sim n\ln n ,
\]
met $\sum_{k\leq n}\frac1k = \ln n + \gamma + o(1)$
(\cref{ch:b2:comparison}). De laatste paar speeltjes
verzamelen is wat kost: de helft van de dozen gaat naar de
finale handvol.
\end{solution}

\begin{solution}{exo:b2:randomvar:4}
Puntsgewijs, $X(\omega) = \#\{n \geq 1 : X(\omega) \geq n\} =
\sum_{n\geq1}\mathbf{1}_{X \geq n}(\omega)$. De dubbele
familie $\bigl(\mathbf{1}_{X \geq n}(\omega)\,\P(\{\omega\})
\bigr)_{n, \omega}$ is niet-negatief, dus Fubini voor
families (\cref{ch:b2:series}) past onvoorwaardelijk toe:
eerst in $n$ sommeren geeft $\E(X)$, eerst in $\omega$
sommeren geeft $\sum_n \P(X \geq n)$; de twee zijn
gelijktijdig eindig en gelijk. Voor $X \sim \mathcal{G}(p)$:
$\P(X \geq n) = q^{n-1}$ ($q = 1-p$), dus $\E(X) =
\sum_{n\geq1}q^{n-1} = \frac{1}{1 - q} = \frac1p$.
\end{solution}

\begin{solution}{exo:b2:randomvar:5}
\emph{Symmetrie:} de $i$-de getrokken bal is een uniform
willekeurige bal van de urn (elk van de $N$ ballen is even
waarschijnlijk in positie $i$ van de trekvolgorde te landen),
dus $\P(Y_i = 1) = \frac MN = p$ en $\E(X) = np$ door
lineariteit --- geen onafhankelijkheid nodig.

\emph{\omterm{def:b2:randomvar:variance}{Covariantie}:} voor $i \neq j$, $\E(Y_iY_j) =
\P(\text{trekkingen } i, j \text{ beide wit}) =
\frac{M(M-1)}{N(N-1)}$ (geordende paren van verschillende
posities krijgen een geordend paar verschillende ballen,
uniform). Dus
\[
\operatorname{Cov}(Y_i, Y_j)
= \frac{M(M-1)}{N(N-1)} - \frac{M^2}{N^2}
= \frac{M(N - M)}{N^2}\cdot\frac{-1}{N-1}
= -\frac{p(1-p)}{N-1} < 0 :
\]
een witte bal trekken maakt witten schaarser voor de andere
trekkingen. Door \cref{thm:b2:randomvar:variancerules},
\[
V(X) = np(1-p) + n(n-1)\Bigl(-\frac{p(1-p)}{N-1}\Bigr)
= np(1-p)\,\frac{N - n}{N - 1} \leq np(1-p) :
\]
steekproef zonder teruglegging heeft hetzelfde gemiddelde
maar \emph{kleinere} \omterm{def:b2:randomvar:variance}{variantie} dan met teruglegging
(gelijkheid alleen voor $n = 1$), de negatieve correlaties
werkend als stabilizer. Voor $n = N$ verdwijnt de \omterm{def:b2:randomvar:variance}{variantie}:
de telling is dan deterministisch.
\end{solution}

\begin{solution}{exo:b2:randomvar:6}
Ontwikkelen rond $m = \E(X)$:
\[
\E\bigl((X - c)^2\bigr)
= \E\bigl((X - m)^2\bigr) + 2(m - c)\,\E(X - m) + (m - c)^2
= V(X) + (m - c)^2 ,
\]
minimaal precies in $c = m$ met waarde $V(X)$ --- \omterm{def:b2:randomvar:expectation}{verwachting}
is de beste constante voorspeller in kwadratisch gemiddelde.

Als $\P(X = m) = 1$ dan verdwijnt $(X - m)^2$ met
waarschijnlijkheid $1$, dus $V(X) = 0$ (de definiërende
familie heeft nultermen behalve op een nulverzameling).
Omgekeerd, als $V(X) = 0$, geeft Chebyshev
(\cref{thm:b2:randomvar:markov}) $\P\bigl(\abs{X - m} \geq
\frac1n\bigr) \leq n^2\,V(X) = 0$ voor elke $n$; de
\omterm{def:b2:proba:space}{gebeurtenissen} $\bigl\{\abs{X - m} \geq \frac1n\bigr\}$
stijgen naar $\{X \neq m\}$, dus monotone continuïteit
(\cref{thm:b2:proba:continuity}) levert $\P(X \neq m) = 0$.
\end{solution}

\begin{solution}{exo:b2:randomvar:7}
$\E(S_n) = \frac n2$ en $V(S_n) = \frac n4$.
\emph{Markov:} $\P\bigl(S_n \geq \frac{3n}4\bigr) \leq
\frac{n/2}{3n/4} = \frac23$ --- een constante grens, nutteloos
voor grote $n$.
\emph{Chebyshev:} de \omterm{def:b2:proba:space}{gebeurtenis} impliceert
$\abs{S_n - \frac n2} \geq \frac n4$, dus de
waarschijnlijkheid is hoogstens
$\frac{n/4}{(n/4)^2} = \frac4n$ --- die daalt, maar alleen
polynomiaal.
\emph{Chernoff:} door onafhankelijkheid, $\E(e^{tS_n}) =
\prod_{i=1}^n\E(e^{tX_i}) = \bigl(\frac{1 + e^t}{2}\bigr)^n$,
en Markov toegepast op $e^{tS_n} \geq e^{3nt/4}$ geeft, voor
elke $t > 0$,
\[
\P\Bigl(S_n \geq \frac{3n}4\Bigr)
\leq \Bigl(\frac{1 + e^t}{2}\Bigr)^n e^{-3nt/4}
= \exp\Bigl(n\bigl(\ln\tfrac{1 + e^t}{2} - \tfrac{3t}4\bigr)\Bigr).
\]
Minimaliseer de exponent: $\frac{\dd}{\dd t}\ln\frac{1+e^t}{2}
= \frac{e^t}{1 + e^t} = \frac34$ bij $e^t = 3$, d.w.z.\ $t =
\ln 3$, wat geeft
\[
\P\Bigl(S_n \geq \frac{3n}4\Bigr)
\leq \Bigl(\frac{4}{2}\Bigr)^n 3^{-3n/4}
= \bigl(2 \cdot 3^{-3/4}\bigr)^n \approx (0.877)^n ,
\]
exponentieel klein. De hiërarchie Markov $\to$ Chebyshev $\to$
Chernoff is de standaardladder: elke sport past Markov toe op
een sneller groeiende functie van de variabele.
\end{solution}

\begin{solution}{exo:b2:randomvar:8}
Door de overdrachtsstelling (\cref{thm:b2:randomvar:transfer})
toegepast op $f\bigl(\frac{S_n}{n}\bigr)$ met $S_n \sim
\mathcal{B}(n, x)$:
\[
\E\Bigl[f\Bigl(\frac{S_n}{n}\Bigr)\Bigr]
= \sum_{k=0}^n f\Bigl(\frac kn\Bigr)\binom nk x^k(1-x)^{n-k}
= B_nf(x) .
\]
Fixeer $\delta > 0$ en splits $\abs{f(S_n/n) - f(x)}$ op de
\omterm{def:b2:proba:space}{gebeurtenis} $D = \bigl\{\abs{\frac{S_n}{n} - x} \geq
\delta\bigr\}$: buiten $D$ is het verschil ten hoogste de
continuïteitsmodulus $\omega_f(\delta) =
\sup_{\abs{s - t}\leq\delta}\abs{f(s) - f(t)}$; op $D$, ten
hoogste $2\norm f_\infty$. \omterm{def:b2:randomvar:expectation}{Verwachtingen} nemend en Chebyshev
gebruikend met $V\bigl(\frac{S_n}{n}\bigr) =
\frac{x(1-x)}{n} \leq \frac{1}{4n}$:
\[
\abs{B_nf(x) - f(x)}
\leq \E\,\abs{f(S_n/n) - f(x)}
\leq \omega_f(\delta)
+ 2\norm f_\infty\,\P(D)
\leq \omega_f(\delta) + \frac{2\norm f_\infty}{4n\delta^2} .
\]
Uniforme continuïteit van $f$ op $[0, 1]$ maakt
$\omega_f(\delta) \to 0$: kies $\delta$ dan $n$, en $B_nf
\to f$ uniform --- de Weierstrass-benaderingsstelling van
\cref{ch:b2:funcseq}, waarvan het ``tellemma'' nu herkenbaar
is als de ongelijkheid van Chebyshev voor de binomiale \omterm{def:b2:randomvar:law}{wet}.
\end{solution}

\begin{solution}{exo:b2:randomvar:9}
Ontwikkel $S_n^4 = \sum_{i,j,k,l}X_iX_jX_kX_l$ en neem
\omterm{def:b2:randomvar:expectation}{verwachtingen}. Door onafhankelijkheid en centrering verdwijnt
elke term die een index bevat die precies één keer verschijnt
($\E(X_i) = 0$ factort uit). Overlevende termen: de $n$
diagonaaltermen $\E(X_i^4)$, en de termen die twee paren
gelijke indices paren, $\E(X_i^2X_j^2) = \E(X_1^2)^2$ voor $i
\neq j$, verschijnend $3n(n-1)$ keer: kies het ongeordende
paar waarden ($\binom n2$ manieren), dan de
$\frac{4!}{2!\,2!} = 6$ manieren om ze in de vier slots te
plaatsen --- $6\binom n2 = 3n(n-1)$. Dus, met $\E(X_1^2)^2
\leq \E(X_1^4)$ (Jensen of Cauchy--Schwarz),
\[
\E(S_n^4) = n\,\E(X_1^4) + 3n(n-1)\,\E(X_1^2)^2
\leq C n^2,
\qquad C = 4\,\E(X_1^4) .
\]
Markov van orde 4:
\[
\P\Bigl(\Bigl|\frac{S_n}{n}\Bigr| \geq \varepsilon\Bigr)
= \P\bigl(S_n^4 \geq n^4\varepsilon^4\bigr)
\leq \frac{Cn^2}{n^4\varepsilon^4}
= \frac{C}{n^2\varepsilon^4} ,
\]
een sommeerbare reeks. Door Borel--Cantelli~1
(\cref{thm:b2:proba:borelcantelli}) heeft voor elke $j$ de
\omterm{def:b2:proba:space}{gebeurtenis} $B_j = \limsup_n\bigl\{\abs{S_n/n} \geq
\frac1j\bigr\}$ waarschijnlijkheid $0$, dus
$\P\bigl(\bigcup_j B_j\bigr) = 0$ door aftelbare
subadditiviteit. Op het complement --- van
waarschijnlijkheid $1$ --- bestaat voor elke $j$ een $N$ met
$\abs{S_n/n} < \frac1j$ voor alle $n \geq N$: precies $S_n/n
\to 0$. De sterke wet van de grote aantallen geldt onder een
vierde moment; die hypothese verwijderen (stelling van
Kolmogorov) is Bachelor jaar~3-werk.
\end{solution}

\begin{solution}{exo:b2:randomvar:10}
$\P(M \leq k) = \bigl(\frac k6\bigr)^2$ (beide stenen ten
hoogste $k$, \omterm{def:b2:proba:independence}{onafhankelijk}), dus $\P(M \geq k) = 1 -
\bigl(\frac{k-1}6\bigr)^2$ en
\[
\E(M) = \sum_{k=1}^6\P(M \geq k)
= 6 - \frac{0 + 1 + 4 + 9 + 16 + 25}{36}
= 6 - \frac{55}{36} = \frac{161}{36} \approx 4.47 ,
\]
comfortabel boven het gemiddelde $3.5$ van een enkele steen,
zoals een maximum behoort te zijn.
\end{solution}

\begin{solution}{exo:b2:randomvar:11}
Met $I_i = \mathbf 1_{\sigma(i) = i}$: $\P(\sigma(i) = i) =
\frac{(n-1)!}{n!} = \frac1n$, dus $\E(F_n) = n\cdot\frac1n =
1$. Voor $i \neq j$: $\P(\sigma(i) = i, \sigma(j) = j) =
\frac{(n-2)!}{n!} = \frac1{n(n-1)}$, dus
\[
\operatorname{Cov}(I_i, I_j) = \frac1{n(n-1)} - \frac1{n^2}
= \frac{1}{n^2(n-1)} .
\]
Door het variantiegereedschap
(\cref{thm:b2:randomvar:variancerules}),
\[
V(F_n) = n\cdot\frac1n\Bigl(1 - \frac1n\Bigr)
+ n(n-1)\cdot\frac1{n^2(n-1)}
= 1 - \frac1n + \frac1n = 1 .
\]
Gemiddelde $1$, \omterm{def:b2:randomvar:variance}{variantie} $1$, onafhankelijk van $n$ ---
consistent met de Poisson-limiet van het matchingprobleem
(\cref{exo:b2:proba:5}).
\end{solution}

\begin{solution}{exo:b2:randomvar:12}
$T_n = \sum_{k=1}^nG_k$ waar $G_k \sim \mathcal G(k/n)$ de
tijd is tot een nieuw speeltje wanneer er nog $k$ ontbreken,
de stadia \omterm{def:b2:proba:independence}{onafhankelijk}. Dus
\[
V(T_n) = \sum_{k=1}^n\frac{1 - k/n}{(k/n)^2}
\leq \sum_{k=1}^n\frac{n^2}{k^2}
\leq \frac{\pi^2}6\,n^2 ,
\]
door \cref{ex:b2:fourier:basel}. Met $\E(T_n) = nH_n$, $H_n =
\sum_1^n\frac1k$ (\cref{exo:b2:randomvar:3}), geeft Chebyshev,
voor $\varepsilon > 0$,
\[
\P\bigl(\abs{T_n - nH_n} \geq \varepsilon\,n\ln n\bigr)
\leq \frac{\pi^2n^2/6}{\varepsilon^2n^2\ln^2 n}
= \frac{\pi^2}{6\,\varepsilon^2\ln^2n}
\xrightarrow[n\to\infty]{} 0 .
\]
Omdat $H_n \sim \ln n$, toont delen door $n\ln n$ dat
$T_n/(n\ln n) \to 1$ in waarschijnlijkheid: de fluctuaties
van $T_n$ zijn van orde $n$, verwaarloosbaar tegen het
gemiddelde $n\ln n$.
\end{solution}

\begin{solution}{pb:b2:randomvar:1}
\textbf{1.} $\E(S_n) = \frac n2$ en Markov
(\cref{thm:b2:randomvar:markov}) geven $\P(S_n \geq an) \leq
\frac{n/2}{an} = \frac1{2a}$. Markov kent alleen het
gemiddelde: zij kan een variabele geconcentreerd in $n/2$
niet onderscheiden van één gespreid tussen $0$ en $n$, dus
prijst zij de staart alsof alle massa daar kon zitten.

\textbf{2.} De eerlijke binomiaal is \omterm{def:b2:quadratic:adjoint}{symmetrisch} om $n/2$
($S_n$ en $n - S_n$ hebben dezelfde \omterm{def:b2:randomvar:law}{wet}), dus met $x = n(a -
\frac12) > 0$ zijn de twee \omterm{def:b2:proba:space}{gebeurtenissen} $\{S_n - \frac n2
\geq x\}$ en $\{S_n - \frac n2 \leq -x\}$ disjunct en
equiprobable: $\P(S_n \geq an) = \frac12\P(\abs{S_n -
\frac n2} \geq x)$. Chebyshev met $V(S_n) = \frac n4$:
\[
\P(S_n \geq an)
\leq \frac12\cdot\frac{n/4}{n^2(a - 1/2)^2}
= \frac1{8n(a - 1/2)^2},
\]
wat $\frac2n$ is bij $a = \frac34$.

\textbf{3.} Door onafhankelijkheid en de productstelling,
$\E(\eu^{tS_n}) = \bigl(\E \eu^{tX_1}\bigr)^n = \bigl(\frac{1
+ \eu^t}2\bigr)^n$. Markov toegepast op $\eu^{tS_n}$:
\[
\P(S_n \geq an) \leq \eu^{-tan}\Bigl(\frac{1 +
\eu^t}2\Bigr)^{\!n} = \exp\Bigl(n\bigl(\ln\tfrac{1 +
\eu^t}2 - ta\bigr)\Bigr).
\]
De afgeleide van de exponent in $t$ is $\frac{\eu^t}{1 +
\eu^t} - a$, nul in $\eu^t = \frac a{1-a}$, d.w.z.\ $t^* =
\ln\frac a{1-a} > 0$; daar $\frac{1 + \eu^{t^*}}2 =
\frac1{2(1-a)}$ en de exponent is gelijk aan
\[
n\Bigl(-\ln 2 - \ln(1-a) - a\ln\frac a{1-a}\Bigr)
= -n\bigl(\ln2 + a\ln a + (1-a)\ln(1-a)\bigr) = -n\,I(a),
\]
met $I(\frac12) = 0$ en $I'(a) = \ln\frac a{1-a} > 0$ op
$\intoo{\frac12}1$: $I(a) > 0$. Bij $a = \frac34$:
$\eu^{-I(3/4)} = \frac12(\tfrac34)^{-3/4}(\tfrac14)^{-1/4} =
2\cdot3^{-3/4}$, de grens van \cref{exo:b2:randomvar:7}.

\textbf{4.} De $n + 1$ getallen $\binom nja^j(1-a)^{n-j}$
sommeren tot $1$, en de grootste is die bij $j = k = an$ (de
modus van $\mathcal B(n, a)$ is $\floor{(n+1)a} = k$ hier).
Een maximum van $n + 1$ getallen die tot $1$ sommeren is
minstens $\frac1{n+1}$:
\[
\binom nk a^k(1-a)^{n-k} \geq \frac1{n+1}
\quad\Longrightarrow\quad
\binom nk \geq \frac{a^{-an}(1-a)^{-n(1-a)}}{n+1}
= \frac{\eu^{nH(a)}}{n+1}.
\]
Dus $\P(S_n \geq an) \geq \binom{n}{an}2^{-n} \geq
\eu^{n(H(a) - \ln2)}/(n+1) = \eu^{-nI(a)}/(n+1)$: tot op de
polynomiale factor $n + 1$ is de exponent van Chernoff de
waarheid.

\textbf{5.} $n = 100$, $a = \frac34$: Markov $\frac23$;
Chebyshev $\frac2{100} = 0.02$; Chernoff
$(2\cdot3^{-3/4})^{100} = \eu^{-100\,I(3/4)} \approx
2.1\cdot10^{-6}$, tegenover de exacte $2.8\cdot10^{-7}$.
Moraal: elk moment informatie deelt de grens polynomiaal;
het exponentiële moment verandert haar \emph{aard}.

\textbf{6.} $\cosh t = \sum_{k\geq0}\frac{t^{2k}}{(2k)!}$ en
$\eu^{t^2/2} = \sum_{k\geq0}\frac{t^{2k}}{2^kk!}$; de
bewering volgt term voor term uit $(2k)! \geq 2^kk!$, wat
door inductie geldt: $(2k)! = 2k(2k-1)\cdot(2k-2)! \geq
2k\cdot 2^{k-1}(k-1)! = 2^kk!\cdot(2k-1) \geq 2^kk!$.

\textbf{7.} Door onafhankelijkheid,
$\E\bigl(\eu^{t\sum\varepsilon_i}\bigr) = (\cosh t)^n \leq
\eu^{nt^2/2}$, dus Markov geeft $\P(\sum\varepsilon_i \geq s)
\leq \eu^{nt^2/2 - ts}$; minimaliseren in $t = s/n$ levert
$\eu^{-s^2/(2n)}$.

\textbf{8.} Met $X_i = \frac{1 + \varepsilon_i}2$, $\widehat
p_n - \frac12 = \frac1{2n}\sum\varepsilon_i$, dus
$\{\widehat p_n - \frac12 \geq \delta\} =
\{\sum\varepsilon_i \geq 2n\delta\}$ en vraag 7 geeft de
grens $\eu^{-(2n\delta)^2/(2n)} = \eu^{-2n\delta^2}$. De
symmetrische \omterm{def:b2:proba:space}{gebeurtenis} heeft dezelfde grens, vandaar de
factor $2$ voor $\abs{\widehat p_n - \frac12} \geq \delta$.

\textbf{9.} $\E(\eu^{tX}) = \sum_x\eu^{tx}\P(X = x)$ is een
reeks van gladde functies van $t$ waarvan de
term-voor-term-afgeleiden, op elk \omterm{def:b2:metric:compact}{compact} $t$-interval,
gedomineerd worden door $\eu^{\abs t}\P(X = x)$ (omdat $0
\leq x \leq 1$): door de differentiatiestelling voor
normaal convergente reeksen
(\cref{thm:b2:funcseq:differentiation}) is zij tweemaal
\omterm{def:b2:diffcalc:differential}{differentieerbaar}, en de quotiëntregel geeft $\psi' =
\E_t(X)$ en $\psi'' = \E_t(X^2) - \E_t(X)^2$, waar $\E_t$ de
\omterm{def:b2:randomvar:expectation}{verwachting} is voor de herwogen gewichten
$\eu^{tx}\P(X{=}x)/\E(\eu^{tX})$ --- niet-negatief, sommerend
tot $1$, gedragen door dezelfde waarden $x \in \intcc01$.
Een \omterm{def:b2:randomvar:variance}{variantie} van een $\intcc01$-waardige variabele is ten
hoogste $\frac14$: door \cref{exo:b2:randomvar:6} is zij
$\min_c\E_t((X - c)^2) \leq \E_t\bigl((X -
\tfrac12)^2\bigr) \leq \tfrac14$. Taylor met
integraalrestterm, met $\psi(0) = 0$, $\psi'(0) = p$:
\[
\psi(t) = tp + \int_0^t(t - s)\,\psi''(s)\,\dd s
\leq tp + \frac{t^2}2\cdot\frac14,
\]
d.w.z.\ $\E(\eu^{t(X - p)}) \leq \eu^{t^2/8}$ voor alle reële
$t$.

\textbf{10.} Door onafhankelijkheid, $\E\bigl(\eu^{t(S_n -
np)}\bigr) \leq \eu^{nt^2/8}$; Markov en de optimalisatie $t
= 4\delta$ geven
\[
\P(\widehat p_n - p \geq \delta)
\leq \eu^{nt^2/8 - tn\delta}\Big|_{t = 4\delta}
= \eu^{-2n\delta^2};
\]
dit toepassen op de variabelen $1 - X_i$ (ook in $\intcc01$)
begrenst de andere staart, vandaar de tweezijdige
$2\eu^{-2n\delta^2}$.

\textbf{11.} Chebyshev eist alleen een tweede moment en
geeft $\frac{p(1-p)}{n\delta^2}$; Hoeffding eist
\emph{begrensdheid} en geeft $2\eu^{-2n\delta^2}$. Bij $p =
\frac12$, $\delta = 0.03$: de grenzen zijn $\frac{278}{n}$
(ongeveer) versus $2\eu^{-0.0018n}$; zij kruisen nabij $n
\approx 1200$, waarna de exponentiële grens wint, en wild zo
($n = 5000$: $0.056$ versus $2.5\cdot10^{-4}$).

\textbf{12.} Door Hoeffding (vraag 10), $\P(\abs{\widehat
p_n - p} \geq \delta) \leq 2\eu^{-2n\delta^2} \leq \alpha$
zodra $2n\delta^2 \geq \ln\frac2\alpha$, d.w.z.\ $n \geq
\frac{\ln(2/\alpha)}{2\delta^2}$.

\textbf{13.} $\delta = 0.03$, $\alpha = 0.05$: $n \geq
\frac{\ln 40}{2\cdot0.0009} \approx 2049.4$: $2050$ mensen.
Voor $\delta = 0.01$: $n \geq \frac{\ln40}{0.0002} \approx
18\,445$. De populatiegrootte verschijnt nooit omdat elke
bemonsterde kiezer gemodelleerd is als een verse
Bernoulli$(p)$-trekking: de moeilijkheid van de peiling is
de \omterm{def:b2:randomvar:variance}{variantie} van een munt, niet de grootte van het land.
De marge halveren kost vier keer de steekproef --- de
$1/\delta^2$-wet.

\textbf{14.} Chebyshev: $\P(\abs{\widehat p_n - p} \geq
\delta) \leq \frac{p(1-p)}{n\delta^2} \leq
\frac1{4n\delta^2} \leq \alpha$ voor $n \geq
\frac1{4\alpha\delta^2}$, d.w.z.\ $5556$ bij drie punten ---
ongeveer $2.7$ keer de Hoeffding-eis. Zonder teruglegging
wordt de \omterm{def:b2:randomvar:variance}{variantie} vermenigvuldigd met $\frac{N - n}{N-1} <
1$ (\cref{exo:b2:randomvar:5}), dus dezelfde $n$ kan alleen
beter doen: de met-teruglegging-berekening is de
conservatieve.

\textbf{15.} $\{\widehat p_n \leq \frac12\} \subseteq
\{\widehat p_n - 0.52 \leq -0.02\}$, dus door de eenzijdige
Hoeffding-grens $\P(\widehat p_n \leq \tfrac12) \leq
\eu^{-2n(0.02)^2} \leq 0.01$ zodra $n \geq \frac{\ln
100}{2\cdot0.0004} \approx 5756.5$: $5757$ kiezers. De kost
schaalt als het omgekeerde kwadraat van de \emph{voorsprong},
niet van de gewenste precisie: nipte races zijn duur.

\textbf{16.} Gebruikt: de steekproef is uniform en
\omterm{def:b2:proba:independence}{onafhankelijk} uit het electoraat getrokken; elke
bemonsterde persoon antwoordt, eerlijk, en $p$ beweegt niet
tijdens peilen. Echte peilingen schenden alle drie:
bereikbare en willige respondenten zijn geen uniforme
steekproef (selectie- en non-responsvertekening), en
antwoorden kunnen onwaarachtig of onstabiel zijn. Dit zijn
\emph{vertekeningsfouten}: zij verschuiven $\E(\widehat
p_n)$ weg van $p$ met een bedrag onafhankelijk van $n$, dus
geen steekproefgrootte reduceert ze --- de wiskunde van dit
Deel controleert alleen de fluktuatieterm.

\textbf{17.} Chebyshev voor één groep van grootte $m$:
$\P(\abs{\widehat p^{(i)} - p} \geq \delta) \leq
\frac{1}{4m\delta^2} \leq \frac18$ voor $m \geq
\frac2{\delta^2}$. Als minder dan $k/2$ groepen fout gaan,
liggen meer dan $k/2$ van de waarden $\widehat p^{(i)}$ in
het open interval $\intoo{p - \delta}{p + \delta}$, en dus
ook hun mediaan; dus dwingt $\{\abs{M - p} \geq \delta\}$
minstens $\lceil k/2\rceil$ fouten onder $k$
onafhankelijke groepen. De unie-ongelijkheid over de
$\binom k{\lceil k/2\rceil}$ mogelijke verzamelingen
foutgaande groepen geeft
\[
\P(\abs{M - p} \geq \delta)
\leq \binom{k}{\lceil k/2\rceil}
\Bigl(\frac18\Bigr)^{k/2}
\leq 2^k\,8^{-k/2} = 2^{-k/2} :
\]
exponentieel verval in het aantal groepen, gekocht met
niets dan \omterm{def:b2:randomvar:variance}{varianties} --- nuttig precies wanneer de
sommanden onbegrensd zijn en Hoeffding niet beschikbaar is.

\textbf{18.} Cauchy--Schwarz
(\cref{thm:b2:randomvar:jensen}):
\[
\E(X) = \E(X\,\mathbf 1_{X>0})
\leq \sqrt{\E(X^2)}\sqrt{\E(\mathbf 1_{X>0}^2)}
= \sqrt{\E(X^2)\,\P(X > 0)} ;
\]
kwadrateer en deel.

\textbf{19.} Laat $h(a) = I(a) - 2(a - \tfrac12)^2$. Dan
$h(\tfrac12) = 0$, $h'(a) = \ln\frac a{1-a} - 4(a -
\tfrac12)$ verdwijnt in $\tfrac12$, en
\[
h''(a) = \frac1a + \frac1{1-a} - 4 = \frac{1}{a(1-a)} - 4
\geq 0
\]
omdat $a(1-a) \leq \frac14$. Dus $h'$ stijgt van $0$ op
$\intco{\frac12}1$, vandaar $h' \geq 0$ en $h \geq 0$: $I(a)
\geq 2(a - \tfrac12)^2$.

\textbf{20.} $I(\tfrac12) = I'(\tfrac12) = 0$, $I''(a) =
\frac1{a(1-a)}$ geeft $I''(\tfrac12) = 4$, en $I'''(\tfrac12)
= 0$ (de functie is \omterm{def:b2:quadratic:adjoint}{symmetrisch} om $\tfrac12$), dus
$I(\tfrac12 + \delta) = 2\delta^2 + O(\delta^4)$. Vraag 4
begrenst dan de ware staart \emph{van onderen} door
$\eu^{-n(2\delta^2 + O(\delta^4))}/(n+1)$: voor kleine
$\delta$ is de Hoeffding-exponent $2n\delta^2$ asymptotisch
exact --- alleen polynomiaal-in-$n$ verbeteringen zijn
mogelijk.

\textbf{21.} Markov: één moment, verval $1/a$, nuttig alleen
als motor achter de anderen (vraag 1 toont haar vlak).
Chebyshev: twee momenten, verval $\frac{V}{n\delta^2}$,
scherp zonder verdere hypothesen
(\cref{ex:b2:randomvar:chebsharp}), en het beste
gereedschap bij vraag 23. Vierde moment
(\cref{exo:b2:randomvar:9}): verval $C/n^2$, net genoeg
sommeerbaarheid voor een sterke wet. Hoeffding: begrensde
variabelen, verval $2\eu^{-2n\delta^2}$, het werkpaard van
Deel III. Chernoff met het exacte tempo $I(a)$: volle
exponentiële momenten, onverslaanbare exponent (vragen 4,
20), het referentiepunt voor al het andere.

\textbf{22.} Als de munt eerlijk is: $\P(\widehat p_n >
0.525) \leq \P(\widehat p_n - \tfrac12 \geq 0.025) \leq
\eu^{-2n(0.025)^2}$. Als $p = 0.55$: $\P(\widehat p_n \leq
0.525) \leq \P(\widehat p_n - 0.55 \leq -0.025) \leq
\eu^{-2n(0.025)^2}$. Beide fouten liggen onder $0.01$ wanneer
$2n(0.025)^2 \geq \ln 100$, d.w.z.\ $n \geq 3684.2$: $3685$
worpen. (Hypothesen $2.5$ punten uit elkaar onderscheiden
kost wat schatten tot $\pm2.5$ punten kost.)

\textbf{23.} Hoeffding: $n \geq \frac{\ln 40}{2(0.005)^2}
\approx 73\,778$. Chebyshev met de ware \omterm{def:b2:randomvar:variance}{variantie} $p(1-p)
= 0.0099$: $n \geq \frac{0.0099}{0.05\cdot(0.005)^2} =
7920$ --- negen keer goedkoper. De exponent van Hoeffding
$2n\delta^2$ prijst de \omterm{def:b2:randomvar:variance}{variantie} op haar slechtste geval
$\frac14$, absurd pessimistisch wanneer $p = 0.01$; het
nederige tweede moment weet beter. Het ontbrekende
gereedschap is een variantie-bewuste exponentiële grens
(ongelijkheid van Bernstein, Bachelor jaar~3) --- of, voor
zeldzame \omterm{def:b2:proba:space}{gebeurtenissen}, de Poisson-benadering bewezen in
\cref{ch:b2:genfun}, die op de natuurlijke relatieve schaal
werkt.

\textbf{24.} Fixeer $\delta > 0$: $\sum_n 2\eu^{-2n\delta^2}
< \infty$ (geometrisch-achtige reeks), dus
Borel--Cantelli~1 (\cref{thm:b2:proba:borelcantelli}) geeft
$\P(\abs{\widehat p_n - p} \geq \delta \text{ oneindig
vaak}) = 0$, d.w.z.\ de \omterm{def:b2:proba:space}{gebeurtenis} $E_j =
\bigcup_N\bigcap_{n\geq N}\{\abs{\widehat p_n - p} <
\tfrac1j\}$ heeft waarschijnlijkheid $1$ voor elke $j$. De
aftelbare doorsnede $\bigcap_jE_j$ heeft nog steeds
waarschijnlijkheid $1$ (subadditiviteit op de complementen),
en daarop $\widehat p_n \to p$: de sterke wet van de grote
aantallen voor muntworpen, waarbij begrensdheid de rol speelt
die het vierde moment speelde in
\cref{exo:b2:randomvar:9}.

\textbf{25.} Eén moment koopt een platte grens; twee kopen
$1/(n\delta^2)$, en niet meer (het scherptevoorbeeld); vier
kopen $1/n^2$, genoeg om tot een bijna-zekere \omterm{def:b2:randomvar:law}{wet} te
telescoperen; begrensdheid koopt $\eu^{-2n\delta^2}$; en het
volle exponentiële moment koopt het exacte tempo $I$, dat
geen methode verslaat. Peilen van $2050$ mensen volstaat
voor elk land omdat de fluctuatie van de steekproef
geregeerd wordt door de \omterm{def:b2:randomvar:variance}{variantie} van de munt, niet de
grootte van de populatie --- de $1/\delta^2$- en
$\ln(1/\alpha)$-prijskaartjes zijn universeel. De centrale
limietstelling van het Bachelor jaar~3-volume vervangt deze
ongelijkheden, op de $\sqrt n$-schaal, door een exacte
limietwet met expliciete constanten --- elke grens van dit
probleem omzettend in een asymptotische gelijkheid.
\end{solution}
