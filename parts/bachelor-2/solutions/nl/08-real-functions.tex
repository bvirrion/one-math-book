\section*{Hoofdstuk \ref{ch:b2:realfun} --- Functies van een reële veranderlijke}

\begin{solution}{exo:b2:realfun:1}
$\lfloor x\rfloor$: sprongen van grootte $1$ in elk geheel getal. $x -
\lfloor x\rfloor$: sprongen van grootte $-1$ in gehele getallen
(linkerlimiet $1$, waarde $0$). $\lfloor x\rfloor + \sqrt{x -
\lfloor x\rfloor}$: in een geheel getal $n$, linkerlimiet $(n - 1)
+ 1 = n$ en waarde $n$: \emph{continu} overal (de vierkantswortel
herstelt de sprong), al niet differentieerbaar in gehele getallen.
De rationale-sprongfunctie: bij restrictie van de constructie tot
een enumeratie van de dyadischen springt zij met $2^{-n}$ precies
in de $n$-de dyadische rationaal en is elders continu.
\end{solution}

\begin{solution}{exo:b2:realfun:2}
Stel $f$ stijgend heeft een discontinuïteit in een inwendig $a$:
dan $f(a^-) < f(a^+)$ (\cref{thm:b2:realfun:monotone}) en het
beeld van $I$ mist het niet-lege open interval
$\intoo{f(a^-)}{f(a^+)}$ behalve mogelijk de enkele waarde $f(a)$:
het beeld van enig deelinterval dat $a$ inwendig bevat is geen
interval (het heeft een gat aan minstens één kant van $f(a)$). Dit
spreekt de tussenwaarde-eigenschap tegen. Eindpuntdiscontinuïteiten
worden op dezelfde manier uitgesloten met eenzijdige gaten.
\end{solution}

\begin{solution}{exo:b2:realfun:3}
$x\ln x$: tweede afgeleide $\frac1x > 0$: convex. $\ln(1 +
\eu^x)$: afgeleide $\frac{\eu^x}{1 + \eu^x} = 1 - \frac{1}{1 +
\eu^x}$, stijgend: convex. $\sqrt{1 + x^2}$: tweede afgeleide
$(1 + x^2)^{-3/2} > 0$: convex. $x^3$: niet convex op $\R$ ($f'' =
6x$ wisselt van teken); convex enkel op $\R_+$.
\end{solution}

\begin{solution}{exo:b2:realfun:4}
Stel $f(a) \neq f(b)$, zeg $f(b) > f(a)$ met $a < b$ (het geval
$f(b) < f(a)$ is symmetrisch, kijkend naar links). Voor $x > b$
geeft de hellingongelijkheid (\cref{lem:b2:realfun:slopes}) op $a
< b < x$
\[
\frac{f(x) - f(a)}{x - a} \geq \frac{f(b) - f(a)}{b - a} = m > 0
\quad\Longrightarrow\quad
f(x) \geq f(a) + m(x - a) \xrightarrow[x\to+\infty]{} +\infty,
\]
in tegenspraak met begrensdheid van boven. Dus is $f$ constant.

Asymptoten: als $f(x) - (\alpha x + \beta) \to 0$ in $+\infty$ en
$f(x) - (\alpha' x + \beta') \to 0$ in $-\infty$, is de convexe
functie $g(x) = f(x) - (\alpha x + \beta)$ van boven begrensd
nabij $+\infty$; convexiteit plus een asymptoot in $-\infty$ (die
$\alpha' \leq \alpha$ dwingt, dan $\alpha' = \alpha$ door
hellingen te vergelijken in $\mp\infty$: hellingen van een
convexe functie stijgen) maakt $g$ van boven begrensd op heel
$\R$, dus constant $= 0$ in de limiet: $f$ is affien.
\end{solution}

\begin{solution}{exo:b2:realfun:5}
$f'$ is monotoon, dus door \cref{thm:b2:realfun:monotone} zijn
haar enige mogelijke discontinuïteiten sprongen, met eenzijdige
limieten overal bestaand. Door \cref{cor:b2:realfun:nojumps} heeft
een afgeleide geen sprongdiscontinuïteiten. Dus heeft $f'$ geen
discontinuïteiten: continu.
\end{solution}

\begin{solution}{exo:b2:realfun:6}
Neem logaritmen:
\[
\frac1p \ln\Bigl(\sum_i \lambda_i x_i^p\Bigr)
= \frac1p \ln\Bigl(1 + p\sum_i \lambda_i \ln x_i +
O(p^2)\Bigr)
= \sum_i \lambda_i \ln x_i + O(p)
\xrightarrow[p \to 0^+]{} \sum_i \lambda_i \ln x_i ,
\]
met $\sum\lambda_i = 1$ en $\ln(1 + u) = u + O(u^2)$.
Exponentiëren geeft het meetkundig gemiddelde. Nu voor elke $p
\in \intoo{0}{1}$ geeft de machtsgemiddelde-ongelijkheid
(\cref{ex:b2:realfun:powermeans}, exponenten $p < 1$)
\[
\Bigl(\sum_i \lambda_i x_i^p\Bigr)^{1/p} \leq \sum_i\lambda_i x_i ;
\]
$p \to 0^+$ links levert $\prod_i x_i^{\lambda_i} \leq
\sum_i \lambda_i x_i$: de gewogen AM--GM-ongelijkheid.
\end{solution}

\begin{solution}{exo:b2:realfun:7}
Met $\varphi(t) = t\ln t$ (convex: $\varphi'' = \frac1t > 0$) en
gewichten $q_i$ in de punten $t_i = \frac{p_i}{q_i}$:
\[
\sum_i p_i \ln\frac{p_i}{q_i}
= \sum_i q_i\, \varphi\Bigl(\frac{p_i}{q_i}\Bigr)
\;\geq\; \varphi\Bigl(\sum_i q_i \frac{p_i}{q_i}\Bigr)
= \varphi(1) = 0 ,
\]
door Jensen (\cref{thm:b2:realfun:convexreg}~(3)). Gelijkheid in
Jensen voor een \emph{strikt} convexe functie dwingt alle punten
$t_i$ samen te vallen: $\frac{p_i}{q_i}$ constant, en sommeren,
de constante is $1$: $p = q$. (Deze grootheid --- de
Kullback--Leibler-divergentie --- keert terug in de wereld van
\cref{ch:b2:randomvar}.)
\end{solution}

\begin{solution}{exo:b2:realfun:8}
\emph{Dyadische gewichten.} Door inductie op $m$: het geval $m =
1$ is de hypothese. Voor gewicht $\lambda = \frac{k}{2^{m+1}}$
(oneven $k$), schrijf $\lambda = \frac12(\lambda_1 + \lambda_2)$
met $\lambda_j = \frac{k \mp 1}{2^{m+1}}$, beide van noemer
$2^m$ na vereenvoudiging; dan
\[
f\bigl(\lambda x + (1{-}\lambda)y\bigr)
= f\Bigl(\tfrac{u + v}{2}\Bigr)
\leq \frac{f(u) + f(v)}{2}
\leq \lambda f(x) + (1 - \lambda) f(y),
\]
waar $u = \lambda_1 x + (1 - \lambda_1)y$ en $v = \lambda_2 x +
(1-\lambda_2)y$, met midpuntconvexiteit dan de inductiehypothese
op $u, v$.

\emph{Overgang naar de limiet.} Voor willekeurige $\lambda \in
\intcc{0}{1}$, neem dyadischen $\lambda_n \to \lambda$:
continuïteit van $f$ en van de affiene afbeeldingen draagt de
ongelijkheid $f(\lambda_n x + (1-\lambda_n)y) \leq \lambda_n
f(x) + (1-\lambda_n)f(y)$ over naar de limiet: $f$ is convex.
\end{solution}

\begin{solution}{exo:b2:realfun:9}
Voor $0 < x < y$: de hellingongelijkheid
(\cref{lem:b2:realfun:slopes}) in de punten $0 < x < y$ geeft
\[
\frac{f(x) - f(0)}{x} \leq \frac{f(y) - f(0)}{y},
\qquad\text{d.w.z.}\qquad
\frac{f(x)}{x} \leq \frac{f(y)}{y} +
f(0)\Bigl(\frac1x - \frac1y\Bigr).
\]
Aangezien $f(0) \leq 0$ en $\frac1x - \frac1y > 0$, is de
laatste term $\leq 0$: $\frac{f(x)}{x} \leq \frac{f(y)}{y}$. Dus
stijgt $x \mapsto \frac{f(x)}x$.

Superadditiviteit voor $f(0) = 0$: voor $x, y > 0$ (gevallen met
een nule veranderlijke zijn triviaal),
\[
f(x) = x\,\frac{f(x)}{x} \leq x\,\frac{f(x+y)}{x+y},
\qquad
f(y) \leq y\,\frac{f(x+y)}{x+y},
\]
door de zojuist bewezen monotonie; optellen geeft $f(x) + f(y)
\leq f(x+y)$.
\end{solution}

\begin{solution}{exo:b2:realfun:10}
Voor $x, y \in I$ en $\lambda \in \intcc01$: convexiteit van $f$,
dan monotonie van $g$, dan convexiteit van $g$:
\[
g\bigl(f(\lambda x + (1{-}\lambda)y)\bigr)
\leq g\bigl(\lambda f(x) + (1{-}\lambda)f(y)\bigr)
\leq \lambda\,g(f(x)) + (1{-}\lambda)\,g(f(y)).
\]
Tegenvoorbeeld zonder monotonie: $g(t) = -t$ is convex (affien)
maar dalend, $f(x) = x^2$ is convex, en $g \circ f = -x^2$ is
strikt concaaf.
\end{solution}

\begin{solution}{exo:b2:realfun:11}
\emph{Linkerongelijkheid:} zij $m = \frac{a+b}2$ en neem een
steunlijn in $m$ (\cref{thm:b2:realfun:convexreg}~(2)): $f(t)
\geq f(m) + \mu(t - m)$ voor alle $t \in \intcc ab$. Integreren
over $\intcc{a}{b}$: de lineaire term integreert tot
$\mu\int_a^b(t - m)\dd t = 0$ (symmetrie rond $m$), dus $\int_a^b
f \geq (b - a)f(m)$.

\emph{Rechterongelijkheid:} op $\intcc ab$ begrensd convexiteit
$f$ door haar koorde: $f(t) \leq f(a) + \frac{f(b) - f(a)}{b -
a}(t - a)$. Integreren: $\int_a^b f \leq (b-a)f(a) +
\frac{f(b) - f(a)}{b - a}\cdot\frac{(b-a)^2}2 = (b -
a)\,\frac{f(a) + f(b)}2$. Deel door $b - a$.
\end{solution}

\begin{solution}{exo:b2:realfun:12}
Zij $a - \delta < a \leq x < y \leq b < b + \delta$, alle in $I$.
Twee toepassingen van de hellingongelijkheid
(\cref{lem:b2:realfun:slopes}), eerst op $a - \delta < a \leq x <
y$, dan op $x < y \leq b < b + \delta$:
\[
\frac{f(a) - f(a - \delta)}{\delta}
\leq \frac{f(y) - f(x)}{y - x}
\leq \frac{f(b + \delta) - f(b)}{\delta}
\]
(koordehellingen stijgen wanneer beide eindpunten naar rechts
bewegen). Dus is elke koordehelling binnen $\intcc ab$ gevangen
tussen twee vaste getallen, en
\[
\abs{f(y) - f(x)} \leq K\,\abs{y - x},
\qquad
K = \max\Bigl(\Bigl|\frac{f(a) - f(a-\delta)}{\delta}\Bigr|,
\Bigl|\frac{f(b+\delta) - f(b)}{\delta}\Bigr|\Bigr):
\]
$f$ is \omterm{def:b2:metric:continuity}{Lipschitz} op $\intcc ab$. Elk punt van het open $I$ heeft
zo'n segment-met-marge eromheen: lokaal \omterm{def:b2:metric:continuity}{Lipschitz}, dus (opnieuw)
continu op $I$.
\end{solution}

\begin{solution}{pb:b2:realfun:1}
\textbf{1.} \emph{Convex $\Rightarrow$ $f'$ stijgend:} voor $a <
b$ geeft de hellingongelijkheid, voor kleine $h > 0$,
$\frac{f(a+h) - f(a)}h \leq \frac{f(b) - f(a)}{b-a} \leq
\frac{f(b) - f(b-h)}{h}$; $h \to 0$: $f'(a) \leq \frac{f(b) -
f(a)}{b - a} \leq f'(b)$. \emph{Omgekeerd}, als $f'$ stijgt en
$x < y < z$: de middelwaardestelling geeft $c_1 \in
\intoo{x}{y}$, $c_2 \in \intoo yz$ met
\[
\frac{f(y) - f(x)}{y - x} = f'(c_1) \leq f'(c_2)
= \frac{f(z) - f(y)}{z - y},
\]
en deze driewegs hellingongelijkheid, toegepast met $y = \lambda
x + (1 - \lambda)z$, herschikt tot de convexiteitsongelijkheid.
Voor $C^2$: $f'' \geq 0$ d.e.s.d.a.\ $f'$ stijgt.

\textbf{2.} Als $f'' > 0$, is $f'$ strikt stijgend, en de
middelwaardeberekening hierboven geeft een \emph{strikte}
ongelijkheid tussen de twee koordehellingen: strikte
convexiteit. \emph{Strikte steun:} in een inwendig $a$ met
steunhelling $m$, als $f(x_0) = f(a) + m(x_0 - a)$ voor enige
$x_0 \neq a$, dan vallen op het segment van $a$ tot $x_0$ de
steunlijn en de koorde samen, en strikte convexiteit in het
middenpunt geeft $f\bigl(\frac{a + x_0}2\bigr) < f(a) +
m\,\frac{x_0 - a}2$, in tegenspraak met de
steunongelijkheid. Dus $f(x) > f(a) + m(x - a)$ voor alle $x
\neq a$. \emph{Strikt Jensen:} met $a = \sum\lambda_ix_i$,
gemiddelden van de steunongelijkheden geven
$\sum\lambda_if(x_i) \geq f(a)$, met gelijkheid d.e.s.d.a.\ elke
term een gelijkheid is, d.w.z.\ d.e.s.d.a.\ elke $x_i = a$.

\textbf{3.} $(-\ln)'' = \frac1{t^2} > 0$; $(t^r)'' = r(r -
1)t^{r-2}$, positief voor $r > 1$, negatief voor $0 < r < 1$;
$\exp'' = \exp > 0$. Alle strikt door vraag 2.

\textbf{4.} De gevallen $ab = 0$ zijn triviaal. Voor $a, b > 0$,
concaviteit van $\ln$ in de punten $a^p, b^q$ met gewichten
$\frac1p, \frac1q$:
\[
\ln\Bigl(\frac{a^p}p + \frac{b^q}q\Bigr) \geq
\frac1p\ln(a^p) + \frac1q\ln(b^q) = \ln(ab),
\]
en $\ln$ stijgt: $ab \leq \frac{a^p}p + \frac{b^q}q$.
Gelijkheid d.e.s.d.a.\ de twee punten samenvallen (strikte
concaviteit): $a^p = b^q$.

\textbf{5.} Concaviteit van $\ln$ met gewichten $\lambda_i$:
$\ln\bigl(\sum\lambda_ix_i\bigr) \geq \sum\lambda_i\ln x_i =
\ln\prod x_i^{\lambda_i}$; exponentieer. Gelijkheid d.e.s.d.a.\
alle $x_i$ gelijk (vraag 2). De route van
\cref{exo:b2:realfun:6} verkreeg dezelfde ongelijkheid als
limiet van machtsgemiddelden; hier is het één toepassing van
Jensen --- de gereedschapskist heeft redundantie ingebouwd.

\textbf{6.} Als $a = 0$ of $b = 0$ is de ongelijkheid triviaal.
Normaliseer: vervang $a$ door $a/\norm a_p$ en $b$ door $b/\norm
b_q$, we mogen aannemen $\norm a_p = \norm b_q = 1$ en moeten
tonen $\sum\abs{a_ib_i} \leq 1$. Young termsgewijs:
\[
\sum_i\abs{a_i}\abs{b_i} \leq
\sum_i\Bigl(\frac{\abs{a_i}^p}{p} +
\frac{\abs{b_i}^q}{q}\Bigr) = \frac1p + \frac1q = 1 .
\]
Gelijkheid d.e.s.d.a.\ elke Young-ongelijkheid strak is:
$\abs{a_i}^p = \abs{b_i}^q$ voor alle $i$ --- na ongedaan maken
van de normalisatie, $(\abs{a_i}^p)$ evenredig met
$(\abs{b_i}^q)$.

\textbf{7.} $p = q = 2$ is Cauchy--Schwarz met hetzelfde
gelijkheidsgeval (evenredigheid). Eindpunt: $\sum\abs{a_ib_i}
\leq \bigl(\max_i\abs{b_i}\bigr)\sum_i\abs{a_i} =
\norm a_1\norm b_\infty$, onmiddellijk termsgewijs.

\textbf{8.} Voor $p = 1$ is het de driehoeksongelijkheid
termsgewijs. Voor $p > 1$, met $q$ geconjugeerd:
\[
\norm{a+b}_p^p = \sum_i\abs{a_i + b_i}^p
\leq \sum_i\abs{a_i+b_i}^{p-1}\abs{a_i} +
\sum_i\abs{a_i+b_i}^{p-1}\abs{b_i},
\]
en Hölder op elke som, met $(p - 1)q = p$:
\[
\sum_i\abs{a_i+b_i}^{p-1}\abs{a_i} \leq
\Bigl(\sum_i\abs{a_i+b_i}^{p}\Bigr)^{1/q}\norm a_p
= \norm{a + b}_p^{p/q}\,\norm a_p ,
\]
evenzo met $b$. Dus $\norm{a+b}_p^p \leq \norm{a +
b}_p^{p/q}\bigl(\norm a_p + \norm b_p\bigr)$; als $a + b \neq
0$, deel door $\norm{a+b}_p^{p/q}$ en gebruik $p - \frac pq =
1$. Met homogeniteit en scheiding (duidelijk) is
$\norm\cdot_p$ een \omterm{def:b2:nvs:norm}{norm} op $\R^n$.

\textbf{9.} Voor continue $f, g$ op $\intcc ab$: Hölder
\[
\int_a^b\abs{fg} \leq \Bigl(\int_a^b\abs
f^p\Bigr)^{1/p}\Bigl(\int_a^b\abs g^q\Bigr)^{1/q}
\]
door dezelfde normalisatie plus puntsgewijze Young, geïntegreerd;
en Minkowski $\norm{f + g}_p \leq \norm f_p + \norm g_p$ door
dezelfde splitsing, Hölder op elk stuk. Scheiding van de \omterm{def:b2:nvs:norm}{norm}
gebruikt strikte positiviteit: een continue $\abs f^p$ met
nulintegraal verdwijnt identiek (boekdeel van Jaar~1).

\textbf{10.} \emph{Monotonie:} we mogen aannemen $\norm a_p =
1$; dan elke $\abs{a_i} \leq 1$, dus $\abs{a_i}^q \leq
\abs{a_i}^p$ en $\norm a_q^q \leq 1$: $\norm a_q \leq 1 =
\norm a_p$. Gelijkheid eist $\abs{a_i}^q = \abs{a_i}^p$ voor
elke $i$, d.w.z.\ elke $\abs{a_i} \in \{0, 1\}$; met
$\sum\abs{a_i}^p = 1$ blijft precies één coördinaat van modulus
$1$: gelijkheid d.e.s.d.a.\ $a$ ten hoogste één niet-nul
coördinaat heeft. \emph{Limiet:} $\norm a_\infty \leq
\norm a_p \leq n^{1/p}\norm a_\infty$, en $n^{1/p} \to 1$.
\emph{Omgekeerde vergelijking:} Hölder met exponenten $\frac
qp$ en haar geconjugeerde $\frac{q}{q-p}$, toegepast op
$\abs{a_i}^p\cdot 1$:
\[
\norm a_p^p = \sum_i\abs{a_i}^p\cdot 1 \leq
\Bigl(\sum_i\abs{a_i}^{q}\Bigr)^{p/q}\,n^{1 - p/q}
= \norm a_q^{p}\; n^{1-p/q},
\]
vandaar $\norm a_p \leq n^{\frac1p - \frac1q}\norm a_q$, met
gelijkheid d.e.s.d.a.\ alle $\abs{a_i}$ gelijk zijn (het
Hölder-gelijkheidsgeval tegen de constante vector).

\textbf{11.} Schrijf $\abs{a_i}^r = \abs{a_i}^{\theta
r}\,\abs{a_i}^{(1-\theta)r}$ en pas Hölder toe met de
geconjugeerde exponenten $\frac{p}{\theta r}$ en
$\frac{q}{(1-\theta)r}$ (geconjugeerd precies omdat
$\frac{\theta r}p + \frac{(1-\theta)r}q = 1$):
\[
\norm a_r^r = \sum_i \abs{a_i}^{\theta r}\abs{a_i}^{(1-\theta)r}
\leq \Bigl(\sum_i\abs{a_i}^{p}\Bigr)^{\theta r/p}
\Bigl(\sum_i\abs{a_i}^{q}\Bigr)^{(1-\theta)r/q}
= \norm a_p^{\theta r}\,\norm a_q^{(1-\theta)r} .
\]
Neem $r$-de wortels: de $p$-normen zijn log-convex in $\frac1p$.

\textbf{12.} \emph{Beide negatief:} als $p < q < 0$ dan $0 < -q
< -p$, en $M_{-q}(y) \leq M_{-p}(y)$ voor de positieve
exponenten (cursusgeval, \cref{ex:b2:realfun:powermeans})
toegepast op $y = (1/x_i)$; inverteren van de identiteit
$M_p(x) = M_{-p}(1/x)^{-1}$ keert de ongelijkheid om tot $M_p(x)
\leq M_q(x)$. \emph{Brug:} voor $q > 0$ geeft concaviteit van
$\ln$ $\ln M_q = \frac1q\ln\bigl(\sum\lambda_ix_i^q\bigr)
\geq \frac1q\sum\lambda_i\ln x_i^q = \ln M_0$; voor $p < 0$
geeft dezelfde concaviteit
$\ln\bigl(\sum\lambda_ix_i^p\bigr) \geq p\sum\lambda_i\ln
x_i$, en delen door $p < 0$ keert om: $\ln M_p \leq \ln M_0$.
Dus $M_p \leq M_0 \leq M_q$ wanneer $p < 0 < q$: met de twee
zelfde-teken-gevallen stijgt $M$ op heel $\R^*$ (en door $0$).

\textbf{13.} Zij $x_{\max} = \max x_i$, bereikt in $i^*$. Voor
$p > 0$:
\[
\lambda_{i^*}^{1/p}\,x_{\max} \leq M_p \leq x_{\max},
\]
en $\lambda_{i^*}^{1/p} \to 1$: $M_p \to x_{\max}$. Voor $p \to
-\infty$: $M_p(x) = M_{-p}(1/x)^{-1} \to
\bigl(\max_i\frac1{x_i}\bigr)^{-1} = \min_ix_i$.

\textbf{14.} Met $\lambda_i = \frac1n$ leest de keten $M_{-\infty}
\leq M_{-1} \leq M_0 \leq M_1 \leq M_2 \leq M_{+\infty}$ als
\[
\min \leq \frac{n}{\sum\frac1{a_i}} \leq \Bigl(\prod
a_i\Bigr)^{1/n} \leq \frac{\sum a_i}{n} \leq
\sqrt{\frac{\sum a_i^2}{n}} \leq \max .
\]
AM--HM ($M_{-1} \leq M_1$) herschikt rechtstreeks tot
$\bigl(\sum a_i\bigr)\bigl(\sum\frac1{a_i}\bigr) \geq n^2$.

\textbf{15.} Met gelijke gewichten is $M_p(x) =
\bigl(\frac1n\sum\abs{x_i}^p\bigr)^{1/p} = n^{-1/p}\norm x_p$.
Als $p$ groeit, daalt $\norm x_p$ (vraag 10) maar de
normalisator $n^{-1/p}$ stijgt sneller, en het product stijgt
(vraag 12): gemiddelden middelen, \omterm{def:b2:nvs:norm}{normen} accumuleren, en de
factor $n^{-1/p}$ is precies de wisselkoers tussen de twee
boekhoudconventies.

\textbf{16.} Elke schakel is een instantie van strikt Jensen
(vraag 2) met de strikt convexe/concave functies van vraag 3
($t^{q/p}$, $\ln$), dus dwingt gelijkheid in enige schakel alle
$x_i$ gelijk; en $\min = M_p$ of $M_p = \max$ dwingt evenzo
alle waarden gelijk aan het gemeenschappelijk extremum. De
keten is strikt zodra twee $x_i$ verschillen.

\textbf{17.} Pas Young (vraag 4) toe op het paar
$\varepsilon^{1/p}a$ en $\varepsilon^{-1/p}b$:
\[
ab = (\varepsilon^{1/p}a)(\varepsilon^{-1/p}b)
\leq \varepsilon\,\frac{a^p}p +
\varepsilon^{-q/p}\,\frac{b^q}q .
\]
Voor $p = q = 2$, vervang $\varepsilon$ door $2\varepsilon$:
$ab \leq \varepsilon a^2 + \frac{b^2}{4\varepsilon}$ --- de
absorptieongelijkheid: een product wordt geruild voor een klein
veelvoud van één kwadraat plus een groot veelvoud van het
andere.

\textbf{18.} Telescoperend:
\[
\prod_{k=1}^{n}c_k = \frac{\prod_{k=1}^n(k+1)^k}
{\prod_{k=1}^{n}k^{k-1}}
= \frac{2^1\,3^2\cdots(n+1)^n}{1^0\,2^1\cdots n^{n-1}}
= (n+1)^n,
\]
elke factor $(k+1)^k$ van de teller cancelt tegen de volgende
term van de noemer. AM--GM op de $n$ getallen $c_ka_k$:
\[
(a_1\cdots a_n)^{1/n} =
\frac{\bigl(\prod_k c_ka_k\bigr)^{1/n}}{(n+1)}
\leq \frac{1}{n+1}\cdot\frac1n\sum_{k=1}^{n}c_ka_k .
\]

\textbf{19.} Sommeren over $n$ en wisselen van de twee
sommaties (alle termen positief: \cref{thm:b2:series:fubini}):
\[
\sum_{n\geq1}(a_1\cdots a_n)^{1/n}
\leq \sum_{n\geq1}\frac{1}{n(n+1)}\sum_{k=1}^{n}c_ka_k
= \sum_{k\geq1}c_ka_k\sum_{n\geq k}\frac1{n(n+1)}
= \sum_{k\geq1}\frac{c_ka_k}{k},
\]
met de telescoperende $\sum_{n\geq k}\bigl(\frac1n -
\frac1{n+1}\bigr) = \frac1k$. Ten slotte $\frac{c_k}k =
\frac{(k+1)^k}{k^k} = \bigl(1 + \frac1k\bigr)^k < \eu$
(stijgende rij met limiet $\eu$, boekdeel van Jaar~1):
\[
\sum_{n\geq1}(a_1\cdots a_n)^{1/n} \leq
\eu\sum_{k\geq1}a_k :
\]
de ongelijkheid van Carleman. (De constante $\eu$ is optimaal,
al bewijzen we dat niet.)

\textbf{20.} Cauchy--Schwarz (vraag 9, $p = q = 2$) toegepast
op $\sqrt f$ en $\frac1{\sqrt f}$:
\[
1 = \Bigl(\int_0^1\sqrt f\cdot\frac{1}{\sqrt f}\Bigr)^{2}
\leq \Bigl(\int_0^1 f\Bigr)\Bigl(\int_0^1\frac1f\Bigr).
\]
Gelijkheid d.e.s.d.a.\ $\sqrt f$ en $\frac1{\sqrt f}$ evenredig
zijn, d.w.z.\ $f^2$ constant, d.w.z.\ $f$ constant ($f > 0$
continu).

\textbf{21.} Zij $1 < p < \infty$, $\norm a_p = \norm b_p = 1$,
$a \neq b$, en stel $\bigl\Vert\frac{a+b}2\bigr\Vert_p = 1$,
d.w.z.\ Minkowski is een gelijkheid voor $a, b$. Het bewijs van
vraag 8 volgend dwingt gelijkheid in beide Hölder-toepassingen
en in de termsgewijze driehoeksongelijkheden: $(\abs{a_i}^p)$ en
$(\abs{b_i}^p)$ beide evenredig met $(\abs{a_i + b_i}^p)$, en
$a_i, b_i$ van hetzelfde teken --- dus $b = ta$ voor enige $t
\geq 0$, en $\norm b_p = \norm a_p$ geeft $t = 1$: $b = a$,
tegenspraak. Dus bevat de $p$-sfeer geen middenpunt van
onderscheiden sfeerpunten: geen segment. Voor $p = \infty$ in
$\R^2$: alle $(1, t)$, $\abs t \leq 1$, liggen op de
eenheidssfeer --- een platte rand; voor $p = 1$: het segment
$(t, 1 - t)$, $t \in \intcc01$, doet dat.

\textbf{22.} Voor $a = 0$ verdwijnen beide kanten. Anders begrensd
Hölder elke $\sum a_ib_i$ door $\norm a_p\norm b_q \leq \norm
a_p$. Bereiking: neem
\[
b_i = \frac{\operatorname{sign}(a_i)\,\abs{a_i}^{p-1}}
{\norm a_p^{p/q}} :
\qquad
\norm b_q^q = \frac{\sum_i\abs{a_i}^{(p-1)q}}{\norm a_p^{p}}
= \frac{\norm a_p^p}{\norm a_p^p} = 1,
\quad
\sum_ia_ib_i = \frac{\norm a_p^p}{\norm a_p^{p/q}} =
\norm a_p ,
\]
met $(p-1)q = p$ en $p - \frac pq = 1$. Dus is het supremum een
maximum, gelijk aan $\norm a_p$: elke $p$-norm is de duale \omterm{def:b2:nvs:norm}{norm}
van haar geconjugeerde --- de kiem van $L^p$--$L^q$-dualiteit.

\textbf{23.} $\E[X^r] = \sum_i\lambda_ix_i^r$, dus
$\E[X^r]^{1/r} = M_r(x; \lambda)$, stijgend in $r$ door vraag
12 (en door $r \to 0, \pm\infty$ door vragen 12--13): de
momentongelijkheid van Lyapunov, puur een uitspraak over
gewogen machtsgemiddelden. Zij keert terug voor echte
stochastische veranderlijken in \cref{ch:b2:randomvar}.

\textbf{24.} (i) Machtsgemiddelden $M_1 \leq M_3$ met gelijke
gewichten: $\frac{a+b+c}3 \leq
\bigl(\frac{a^3+b^3+c^3}3\bigr)^{1/3}$; kubus en
vermenigvuldig met $3$: $a^3 + b^3 + c^3 \geq
\frac{(a+b+c)^3}9$. (ii) Cauchy--Schwarz tegen de constante
vector: $\sum_i\sqrt{x_i}\cdot1 \leq
\bigl(\sum_ix_i\bigr)^{1/2}n^{1/2}$; kwadrateer.

\textbf{25.} De koordedefinitie levert het hellingslemma door
één algebraïsche herschikking; hellingen samengeknepen in een
punt produceren eenzijdige afgeleiden en steunlijnen, waarvan
het gewogen gemiddelde Jensen is. Toegepast op $-\ln$ wordt
Jensen Young, die gesommeerd tegen genormaliseerde vectoren
Hölder is, die gesplitst en geabsorbeerd Minkowski is --- en de
$p$-normen van \cref{ch:b2:nvs} zijn geboren, met hun
dualiteit (vraag 22) en hun meetkunde (vraag 21). Jensen
toegepast langs de schaal van machten ketent alle gemiddelden
van $\min$ tot $\max$ (vragen 12--14), die gelezen op
stochastische veranderlijken de momentongelijkheid is (vraag
23). En AM--GM, gewogen door één telescoperende truc, levert de
grens van Carleman met haar onherleidbare constante $\eu$
(vragen 18--19). Toppen: Hölder--Minkowski, en Carleman.
Bestemming: de $L^p$-ruimten van het boekdeel van Jaar~3,
waarvan de stichtende axioma's precies vragen 6 en 8 zijn met
integralen in plaats van sommen.
\end{solution}
