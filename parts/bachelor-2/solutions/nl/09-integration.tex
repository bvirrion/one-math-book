\section*{Hoofdstuk \ref{ch:b2:integration} --- Integratie}

\begin{solution}{exo:b2:integration:1}
$\int_0^1 \frac{\dd t}{\sqrt{t(1-t)}}$: nabij $0$, $\sim t^{-1/2}$
($\alpha = \frac12 < 1$: \omterm{def:b2:integration:improper}{convergeert}); nabij $1$, $\sim
(1-t)^{-1/2}$: \omterm{def:b2:integration:improper}{convergeert}. Convergent (haar waarde is $\pi$, via
de substitutie $t = \sin^2\theta$).

$\int_1^\infty \frac{\ln t}{t^2}$: $\frac{\ln t}{t^2} = o(t^{-3/2})$:
convergent (waarde $1$ via partiĂ«le integratie).

$\int_0^\infty \frac{\dd t}{1 + t^2\sin^2 t}$: divergent. Nabij $t =
n\pi$, schrijf $t = n\pi + u$: $\sin^2 t = \sin^2 u \leq u^2$, dus
op $\abs u \leq \frac{1}{n}$ is $1 + t^2\sin^2 t \leq 1 + (n\pi +
1)^2u^2 \leq C n^2 u^2 + 1$; vandaar
\[
\int_{n\pi - 1/n}^{n\pi + 1/n} \frac{\dd t}{1 + t^2\sin^2 t}
\geq \int_{-1/n}^{1/n} \frac{\dd u}{1 + Cn^2u^2}
= \frac{2\arctan\sqrt C}{\sqrt C}\cdot\frac{1}{n} ,
\]
een term van een divergente harmonische-type reeks: sommeren over
$n$ maakt de primitieve onbegrensd.
\end{solution}

\begin{solution}{exo:b2:integration:2}
Substitueer $u = \lambda t$:
\[
\int_0^\infty t^n \eu^{-\lambda t}\dd t
= \frac{1}{\lambda^{n+1}}\int_0^\infty u^n\eu^{-u}\dd u
= \frac{\Gamma(n+1)}{\lambda^{n+1}} = \frac{n!}{\lambda^{n+1}} .
\]
Met $t = \eu^{-u}$ ($\dd t = -\eu^{-u}\dd u$):
\[
\int_0^1 (\ln t)^n \dd t = \int_0^{\infty} (-u)^n \eu^{-u}\,\dd u
= (-1)^n\, n! .
\]
\end{solution}

\begin{solution}{exo:b2:integration:3}
Convergentie: de integrand is $\leq \frac{1}{1+t^2}$ nabij $\infty$
en begrensd nabij $0$ (beide factoren van onder begrensd weg van
$0$): absoluut convergent, voor elke $x$. Substitutie $t =
\frac1u$ ($\dd t = -\frac{\dd u}{u^2}$):
\[
I(x) = \int_0^\infty \frac{1}{\bigl(1 +
\frac1{u^2}\bigr)\bigl(1 + u^{-x}\bigr)}\cdot\frac{\dd u}{u^2}
= \int_0^\infty \frac{u^x}{(1 + u^2)(1 + u^x)}\,\dd u .
\]
Optellen van de twee uitdrukkingen van $I(x)$:
\[
2I(x) = \int_0^\infty \frac{1 + t^x}{(1+t^2)(1+t^x)}\dd t
= \int_0^\infty \frac{\dd t}{1 + t^2} = \frac{\pi}{2} :
\]
$I(x) = \frac\pi4$, onafhankelijk van $x$.
\end{solution}

\begin{solution}{exo:b2:integration:4}
Nabij $0^+$, met $u = \abs{\ln t} \to \infty$. Als $\alpha < 1$:
convergentie ongeacht $\beta$ (vergelijk met $t^{-\alpha'}$ voor
$\alpha < \alpha' < 1$: de logfactor wordt verslagen). Als
$\alpha > 1$: divergentie ongeacht $\beta$ (vergelijk met
$t^{-\alpha''}$, $1 < \alpha'' < \alpha$). Als $\alpha = 1$:
substitueer $t = \eu^{-u}$:
\[
\int_0^{1/2} \frac{\dd t}{t\,\abs{\ln t}^\beta}
= \int_{\ln 2}^{\infty} \frac{\dd u}{u^\beta},
\]
convergent d.e.s.d.a.\ $\beta > 1$. Samenvatting: convergentie
d.e.s.d.a.\ $\alpha < 1$, of ($\alpha = 1$ en $\beta > 1$) ---
de spiegel van de Bertrand-reeksen.
\end{solution}

\begin{solution}{exo:b2:integration:5}
ContinuĂŻteit op $\intco{0}{\infty}$: dominantie
$\bigl|\frac{\eu^{-xt}}{1+t^2}\bigr| \leq \frac{1}{1+t^2}$,
integreerbaar, uniform in $x \geq 0$:
\cref{thm:b2:integration:continuity}.

$C^2$ op $\intoo{0}{\infty}$: op $x \geq a > 0$ worden de eerste
twee $x$-afgeleiden $\frac{-t\,\eu^{-xt}}{1+t^2}$ en
$\frac{t^2\eu^{-xt}}{1+t^2}$ gedomineerd door $t\,\eu^{-at}$ en
$\eu^{-at}$: twee toepassingen van
\cref{thm:b2:integration:leibnizrule}. Dan
\[
F''(x) + F(x) = \int_0^\infty \frac{t^2 + 1}{1 + t^2}\,\eu^{-xt}\dd
t = \int_0^\infty \eu^{-xt}\dd t = \frac1x .
\]
Limiet: $0 \leq F(x) \leq \int_0^\infty \eu^{-xt}\dd t = \frac1x
\to 0$.
\end{solution}

\begin{solution}{exo:b2:integration:6}
Op $\intcc{\varepsilon}{M}$, substitueer $u = at$ en $u = bt$ in
de twee helften:
\[
\int_\varepsilon^M \frac{f(at) - f(bt)}{t}\dd t
= \int_{a\varepsilon}^{aM}\frac{f(u)}{u}\dd u
- \int_{b\varepsilon}^{bM}\frac{f(u)}{u}\dd u
= \int_{a\varepsilon}^{b\varepsilon} \frac{f(u)}{u}\dd u
- \int_{aM}^{bM} \frac{f(u)}{u}\dd u .
\]
Eerste stuk: $f(u) = f(0) + o(1)$ nabij $0$, en
$\int_{a\varepsilon}^{b\varepsilon} \frac{\dd u}{u} = \ln\frac
ba$: het stuk streeft naar $f(0)\ln\frac ba$. Tweede stuk: $f(u)
\to f(\infty)$, dezelfde berekening: streeft naar
$f(\infty)\ln\frac ba$. Dus \omterm{def:b2:integration:improper}{convergeert} de \omterm{def:b2:integration:improper}{oneigenlijke integraal}
naar $\bigl(f(0) - f(\infty)\bigr)\ln\frac ba$.

Met $f(t) = \eu^{-t}$ ($f(0) = 1$, $f(\infty) = 0$), $a = 1$, $b
= 2$:
\[
\int_0^\infty \frac{\eu^{-t} - \eu^{-2t}}{t}\dd t = \ln 2 .
\]
\end{solution}

\begin{solution}{exo:b2:integration:7}
Zet de integrand voort met $0$ voorbij $t = n$: $g_n(t) = (1 -
\frac tn)^n t^{x-1}\mathbf{1}_{t \leq n}$. Puntsgewijs, $g_n(t)
\to \eu^{-t}t^{x-1}$ (de samengestelde-interestlimiet, boekdeel
van Jaar~1). Dominantie: $\ln(1 - u) \leq -u$ geeft $(1 -
\frac tn)^n \leq \eu^{-t}$ op $\intcc{0}{n}$, dus $\abs{g_n(t)}
\leq \eu^{-t}t^{x-1} = \varphi(t)$, integreerbaar. Gedomineerde
convergentie:
\[
\int_0^n \Bigl(1 - \frac tn\Bigr)^n t^{x-1}\dd t
\xrightarrow[n\to\infty]{} \int_0^\infty \eu^{-t}t^{x-1}\dd t
= \Gamma(x) .
\]
(Berekenen van de linkerkant via herhaalde partiĂ«le integratie
geeft de productvorm van Euler $\Gamma(x) = \lim
\frac{n!\,n^x}{x(x+1)\cdots(x+n)}$.)
\end{solution}

\begin{solution}{exo:b2:integration:8}
$H$ is differentieerbaar in $x$ (integrand $C^1$ in $x$,
afgeleide $-2x(1+t^2)\cdot\frac{\eu^{-x^2(1+t^2)}}{1+t^2} =
-2x\,\eu^{-x^2}\eu^{-x^2t^2}$, continu en begrensd op compacta
van $x$, dominantie over $t \in \intcc{0}{1}$ triviaal):
\[
H'(x) = -2x\,\eu^{-x^2}\int_0^1 \eu^{-x^2t^2}\,\dd t
\overset{u = xt}{=} -2\,\eu^{-x^2}\int_0^x \eu^{-u^2}\,\dd u
= -G'(x),
\]
aangezien $G'(x) = 2\eu^{-x^2}\int_0^x \eu^{-t^2}\dd t$
(kettingregel op het kwadraat, hoofdstelling). Dus is $G + H$
constant, gelijk aan $G(0) + H(0) = 0 + \int_0^1 \frac{\dd
t}{1+t^2} = \frac\pi4$.

Als $x \to \infty$: $0 \leq H(x) \leq \eu^{-x^2}\int_0^1 \dd t
\to 0$, dus $G(x) \to \frac\pi4$:
\[
\int_0^\infty \eu^{-t^2}\dd t = \sqrt{\frac\pi4} =
\frac{\sqrt\pi}{2} .
\]
(Dus $\Gamma\bigl(\frac12\bigr) = 2\int_0^\infty
\eu^{-t^2}\dd t = \sqrt\pi$, via de substitutie $t = \sqrt u$.)
\end{solution}

\begin{solution}{exo:b2:integration:9}
Cauchy--Schwarz (boekdeel van Jaar~1, geldig op
$\intcc{\varepsilon}{M}$ en overgedragen naar de limiet)
toegepast op de factorisatie $t^{\frac{x+y}{2}-1}\eu^{-t} =
\bigl(t^{x-1}\eu^{-t}\bigr)^{1/2}
\bigl(t^{y-1}\eu^{-t}\bigr)^{1/2}$:
\[
\Gamma\Bigl(\frac{x+y}{2}\Bigr) \leq
\Gamma(x)^{1/2}\,\Gamma(y)^{1/2}
\quad\Longrightarrow\quad
\ln\Gamma\Bigl(\frac{x+y}{2}\Bigr) \leq \frac{\ln\Gamma(x) +
\ln\Gamma(y)}{2} :
\]
$\ln\Gamma$ is midpuntconvex; zijnde continu
(\cref{thm:b2:integration:gammaprops}), is zij convex
(\cref{exo:b2:realfun:8}). (Log-convexiteit spijkert $\Gamma$
uniek vast onder interpolaties van de faculteit --- de stelling
van Bohr--Mollerup, een parel van Jaar~3.)
\end{solution}

\begin{solution}{exo:b2:integration:10}
\begin{enumerate}
  \item Op $x \geq a > 0$: de $x$-afgeleide van de integrand is
        $-\sin t\,\eu^{-xt}$, gedomineerd door $\eu^{-at}$:
        \cref{thm:b2:integration:leibnizrule} geeft $F'(x) =
        -\int_0^\infty \eu^{-xt}\sin t\,\dd t$. Twee partiĂ«le
        integraties (of de complexe exponentiĂ«le):
        \[
        \int_0^\infty \eu^{-xt}\sin t\,\dd t
        = \Im \int_0^\infty \eu^{(-x+\iu)t}\dd t
        = \Im\frac{1}{x - \iu} = \frac{1}{1 + x^2} .
        \]
  \item $\abs{F(x)} \leq \int_0^\infty \eu^{-xt}\dd t = \frac1x
        \to 0$. Integreren van $F' = -\frac{1}{1+x^2}$ van $x$
        tot $\infty$: $0 - F(x) = -\bigl(\frac\pi2 - \arctan
        x\bigr)$, dus $F(x) = \frac\pi2 - \arctan x$.
  \item $x \to 0^+$ met de aangenomen continuĂŻteit: $F(0^+) =
        \frac\pi2$, en $F(0) = \int_0^\infty \frac{\sin
        t}{t}\dd t$ (de semi-convergente Dirichlet-integraal,
        \cref{ex:b2:integration:sint}): haar waarde is
        $\frac\pi2$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b2:integration:11}
Convergentie: nabij $0$ zet de integrand zich continu voort met
waarde $1$ ($\sin t \sim t$); in oneindig is zij $\leq t^{-2}$:
absolute convergentie. Op $\intcc{\varepsilon}{M}$, partiĂ«le
integratie met $u = \sin^2 t$, $v' = t^{-2}$:
\[
\int_\varepsilon^M \frac{\sin^2 t}{t^2}\dd t
= \Bigl[-\frac{\sin^2 t}{t}\Bigr]_\varepsilon^M
+ \int_\varepsilon^M \frac{2\sin t\cos t}{t}\dd t
= \Bigl[-\frac{\sin^2 t}{t}\Bigr]_\varepsilon^M
+ \int_{2\varepsilon}^{2M} \frac{\sin u}{u}\dd u
\]
($u = 2t$ in de laatste integraal). De haak streeft naar $0$
aan beide einden ($\sin^2\varepsilon/\varepsilon \leq
\varepsilon$; $\sin^2 M/M \leq 1/M$), en de laatste integraal
streeft naar $\int_0^\infty \frac{\sin u}{u}\dd u = \frac\pi2$
(\cref{exo:b2:integration:10}). Dus
\[
\int_0^\infty \Bigl(\frac{\sin t}{t}\Bigr)^{\!2}\dd t
= \frac\pi2 .
\]
\end{solution}

\begin{solution}{exo:b2:integration:12}
\begin{enumerate}
  \item PartiĂ«le integratie met $u = \frac{-1}{2t}$, $v' =
        -2t\,\eu^{-t^2}$ (dus $v = \eu^{-t^2}$):
        \[
        T(x) = \Bigl[\frac{-\eu^{-t^2}}{2t}\Bigr]_x^\infty
        - \int_x^\infty \frac{\eu^{-t^2}}{2t^2}\dd t
        = \frac{\eu^{-x^2}}{2x}
        - \int_x^\infty \frac{\eu^{-t^2}}{2t^2}\dd t .
        \]
        Zelfde apparaat op de nieuwe integraal ($u =
        \frac{-1}{4t^3}$, $v' = -2t\,\eu^{-t^2}$):
        \[
        \int_x^\infty \frac{\eu^{-t^2}}{2t^2}\dd t
        = \frac{\eu^{-x^2}}{4x^3}
        - \frac34\int_x^\infty \frac{\eu^{-t^2}}{t^4}\dd t ,
        \]
        vandaar de aangekondigde identiteit.
  \item Nog Ă©Ă©n partiĂ«le integratie begrensd de rest:
        \[
        \int_x^\infty \frac{\eu^{-t^2}}{t^4}\dd t
        = \frac{\eu^{-x^2}}{2x^5}
        - \frac52\int_x^\infty\frac{\eu^{-t^2}}{t^6}\dd t
        \leq \frac{\eu^{-x^2}}{2x^5},
        \]
        dus $0 \leq \frac34\int_x^\infty t^{-4}\eu^{-t^2}\dd t
        \leq \frac{3}{8x^5}\eu^{-x^2}$. Weglaten van de
        (positieve) rest in de identiteit van vraag 1 geeft de
        ondergrens; weglaten van de (negatieve) tweede term van
        de eerste partiĂ«le integratie geeft $T(x) \leq
        \frac{\eu^{-x^2}}{2x}$. Delen van de insluiting door
        $\frac{\eu^{-x^2}}{2x}$: de verhouding wordt ingesloten
        tussen $1 - \frac{1}{2x^2}$ en $1$, dus $T(x) \sim
        \frac{\eu^{-x^2}}{2x}$.
  \item Itereren van de partiĂ«le integratie produceert de
        formele reeks
        \[
        T(x) \approx \frac{\eu^{-x^2}}{2x}\Bigl(1 -
        \frac{1}{2x^2} + \frac{1\cdot3}{(2x^2)^2} -
        \frac{1\cdot3\cdot5}{(2x^2)^3} + \cdots\Bigr),
        \]
        waarvan de $k$-de coĂ«fficiĂ«nt $1\cdot3\cdots(2k-1) =
        \frac{(2k)!}{2^k k!}$ sneller groeit dan elke
        meetkundige rij: voor vaste $x$ streven de termen
        $\frac{1\cdot3\cdots(2k-1)}{(2x^2)^k}$ naar oneindig
        (hun quotiĂ«nt is $\frac{2k+1}{2x^2} \to \infty$), dus
        de reeks divergeert voor elke $x$. Het is een
        \emph{asymptotische} ontwikkeling: afgeknot op enige
        vaste orde is de fout van de orde van de eerste
        weggelaten term als $x \to \infty$ --- maar nooit een
        convergente reeks. (Deze staartschatting is de
        standaard Gaussische staartgrens van de
        kansrekeningshoofdstukken.)
\end{enumerate}
\end{solution}

\begin{solution}{pb:b2:integration:1}
\textbf{1.} In $0^+$ is de integrand $\sim t^{x-1}$: de
eindige-eindpuntschaal \omterm{def:b2:integration:improper}{convergeert} d.e.s.d.a.\ $1 - x < 1$,
d.w.z.\ $x > 0$ (en voor $x \leq 0$ is $t^{x-1} \geq t^{-1}$
divergent); in $+\infty$ is $t^{x-1}\eu^{-t} = o(t^{-2})$
convergent voor elke $x$. Dan $\Gamma(x) =
\frac{\Gamma(x+1)}{x}$ en $\Gamma(x+1) \to \Gamma(1) = 1$ als
$x \to 0^+$ (continuĂŻteit, \cref{thm:b2:integration:gammaprops}):
$\Gamma(x) \sim \frac1x$.

\textbf{2.} Met $t = u^2$, $\dd t = 2u\,\dd u$:
\[
\Gamma\Bigl(\frac12\Bigr) = \int_0^\infty
t^{-1/2}\eu^{-t}\dd t
= \int_0^\infty \frac{\eu^{-u^2}}{u}\,2u\,\dd u
= 2\int_0^\infty \eu^{-u^2}\dd u = \sqrt\pi
\]
door \cref{exo:b2:integration:8}. Met $u = v/\sqrt2$:
\[
\int_\R \eu^{-v^2/2}\dd v
= 2\sqrt2\int_0^\infty \eu^{-u^2}\dd u
= \sqrt2\,\sqrt\pi = \sqrt{2\pi} .
\]

\textbf{3.} Waar voor $n = 0$ (beide kanten $\sqrt\pi$). Als
$\Gamma(n + \frac12) = \frac{(2n)!}{4^n n!}\sqrt\pi$, geeft de
functionaalvergelijking
\[
\Gamma\Bigl(n + 1 + \frac12\Bigr)
= \Bigl(n + \frac12\Bigr)\Gamma\Bigl(n + \frac12\Bigr)
= \frac{2n+1}{2}\cdot\frac{(2n)!}{4^n n!}\sqrt\pi
= \frac{(2n+2)!}{4^{n+1}(n+1)!}\sqrt\pi ,
\]
de laatste stap omdat $\frac{(2n+2)!}{(2n)!} = (2n+2)(2n+1)$
en $\frac{2n+1}{2} = \frac{(2n+2)(2n+1)}{4(n+1)}$.

\textbf{4.} \cref{thm:b2:integration:gammaprops} geeft
$\Gamma''(x) = \int_0^\infty t^{x-1}\eu^{-t}(\ln t)^2\dd t$
(twee toepassingen van de Leibniz-regel, dominanties als in het
bewijs van de stelling); de integrand is $\geq 0$ en niet
identiek nul, dus $\Gamma'' > 0$: $\Gamma$ is strikt convex en
$\Gamma'$ is strikt stijgend. Aangezien $\Gamma(1) = \Gamma(2)
= 1$, levert Rolle $x_0 \in \intoo12$ met $\Gamma'(x_0) = 0$;
strikte monotonie van $\Gamma'$ maakt $x_0$ haar unieke nulpunt,
met $\Gamma' < 0$ ervoor en $\Gamma' > 0$ erna: $\Gamma$ daalt
op $\intoo0{x_0}$, stijgt op $\intoo{x_0}\infty$, en $x_0$ is
het unieke minimum.

\textbf{5.} Zij $k \in \N$ en $x \geq 3$; kies het gehele $n$
met $n + 1 \leq x < n + 2$ (dus $n \geq 1$). Door de monotonie
van vraag 4 (geldig vanaf $x_0 < 2$): $\Gamma(x) \geq \Gamma(n
+ 1) = n!$, terwijl $x^k \leq (n+2)^k$. Aangezien
$\frac{n!}{(n+2)^k} \to \infty$ (faculteiten verslaan machten,
boekdeel van Jaar~1), $\frac{\Gamma(x)}{x^k} \geq
\frac{n!}{(n+2)^k} \to \infty$ als $x \to \infty$: $x^k =
o(\Gamma(x))$.

\textbf{6.} Nabij $0$ is de integrand $\sim t^{x-1}$
(convergent d.e.s.d.a.\ $x > 0$), nabij $1$ is zij $\sim
(1-t)^{y-1}$ (d.e.s.d.a.\ $y > 0$); beide vergelijkingen zijn
tussen positieve functies, dus \omterm{def:b2:integration:improper}{convergeert} $B(x,y)$ precies
voor $x, y > 0$. De substitutie $t \mapsto 1 - t$ wisselt de
twee factoren: $B(x,y) = B(y,x)$.

\textbf{7.} $B(x,1) = \int_0^1 t^{x-1}\dd t = \frac1x$.
PartiĂ«le integratie op $\intcc\varepsilon{1-\varepsilon}$ met
$u = (1-t)^y$, $v = \frac{t^x}{x}$:
\[
\int t^{x-1}(1-t)^{y}\dd t
= \Bigl[\frac{t^x(1-t)^y}{x}\Bigr]
+ \frac{y}{x}\int t^{x}(1-t)^{y-1}\dd t ;
\]
de haak verdwijnt aan beide einden als $\varepsilon \to 0$ ($x >
0$ in $0$, $y > 0$ in $1$), en houdt $B(x, y+1) = \frac
yx\,B(x+1, y)$ over.

\textbf{8.} Aangezien $t + (1-t) = 1$:
\[
t^{x-1}(1-t)^{y-1} = t^{x}(1-t)^{y-1} + t^{x-1}(1-t)^{y},
\]
dus $B(x,y) = B(x+1,y) + B(x,y+1)$. Vraag 7 leest $B(x+1,y) =
\frac xy B(x,y+1)$; substitueren,
\[
B(x,y) = \Bigl(\frac xy + 1\Bigr)B(x,y+1)
= \frac{x+y}{y}\,B(x,y+1),
\]
d.w.z.\ $B(x,y+1) = \frac{y}{x+y}B(x,y)$; de tweelingrelatie
volgt door de symmetrie van vraag 6.

\textbf{9.} Inductie op $n$ bij vaste $m$: $B(m,1) = \frac1m =
\frac{(m-1)!\,0!}{m!}$, en als de formule geldt in $n$,
\[
B(m, n+1) = \frac{n}{m+n}\,B(m,n)
= \frac{n}{m+n}\cdot\frac{(m-1)!(n-1)!}{(m+n-1)!}
= \frac{(m-1)!\,n!}{(m+n)!} .
\]
Herschrijven: $B(m,n) = \frac{(m-1)!(n-1)!}{(m+n-1)!} =
\bigl[(m+n-1)\binom{m+n-2}{m-1}\bigr]^{-1}$.

\textbf{10.} Beide kanten van $B(x,n) =
\frac{\Gamma(x)\Gamma(n)}{\Gamma(x+n)}$ zijn gelijk $\frac1x$
in $n = 1$ ($\Gamma(1) = 1$, $\Gamma(x+1) = x\Gamma(x)$). Als
zij overeenkomen in $n$, dan door de afdaalrelatie en de
functionaalvergelijking:
\[
B(x, n+1) = \frac{n}{x+n}\,B(x,n), \qquad
\frac{\Gamma(x)\Gamma(n+1)}{\Gamma(x+n+1)}
= \frac{n}{x+n}\cdot
\frac{\Gamma(x)\Gamma(n)}{\Gamma(x+n)} :
\]
de twee rijen gehoorzamen dezelfde recurrente uit hetzelfde
zaad, dus komen overeen voor alle $n \in \N^*$ en alle $x >
0$.

\textbf{11.} Met $t = \sin^2\theta$ ($\theta \in
\intoo0{\pi/2}$, $\dd t = 2\sin\theta\cos\theta\,\dd\theta$),
$t^{x-1} = \sin^{2x-2}\theta$ en $(1-t)^{y-1} =
\cos^{2y-2}\theta$:
\[
B(x,y) = \int_0^{\pi/2}\sin^{2x-2}\theta\,\cos^{2y-2}\theta
\cdot 2\sin\theta\cos\theta\,\dd\theta
= 2\int_0^{\pi/2}\sin^{2x-1}\theta\,\cos^{2y-1}\theta\,
\dd\theta .
\]

\textbf{12.} Neem $y = \frac12$ (doodt de cosinusfactor) en
$2x - 1 = n$: $B\bigl(\frac{n+1}2, \frac12\bigr) = 2W_n$,
d.w.z.\ $W_n = \frac12 B\bigl(\frac{n+1}2,\frac12\bigr)$. De
afdaalrelatie in de eerste veranderlijke geeft
\[
\frac{W_n}{W_{n-2}}
= \frac{B\bigl(\frac{n-1}2 + 1, \frac12\bigr)}
{B\bigl(\frac{n-1}2, \frac12\bigr)}
= \frac{\frac{n-1}2}{\frac{n-1}2 + \frac12}
= \frac{n-1}{n} :
\]
de Wallis-recurrente, ditmaal zonder partiĂ«le integratie op
sinussen --- Deel II deed het werk eens en voor altijd.

\textbf{13.} $B\bigl(\frac12,\frac12\bigr) = 2W_0 =
2\cdot\frac\pi2 = \pi$, terwijl
$\Gamma\bigl(\frac12\bigr)^2/\Gamma(1) = (\sqrt\pi)^2 = \pi$:
de formule van Euler geldt in $\bigl(\frac12,\frac12\bigr)$.

\textbf{14.} Itereren van $W_{2n} = \frac{2n-1}{2n}W_{2n-2}$
vanuit $W_0 = \frac\pi2$:
\[
W_{2n} = \frac\pi2\prod_{k=1}^{n}\frac{2k-1}{2k}
= \frac\pi2\cdot\frac{(2n)!}{4^n(n!)^2},
\]
want $\prod(2k-1) = \frac{(2n)!}{2^n n!}$ en $\prod 2k = 2^n
n!$. Dus, met vraag 3:
\[
B\Bigl(n+\frac12, \frac12\Bigr) = 2W_{2n}
= \pi\,\frac{(2n)!}{4^n(n!)^2}
= \frac{(2n)!\sqrt\pi}{4^n n!}\cdot\frac{\sqrt\pi}{n!}
= \frac{\Gamma\bigl(n+\frac12\bigr)\Gamma\bigl(\frac12\bigr)}
{\Gamma(n+1)} .
\]
Fixeer nu $x \in \frac12\N^*$. De formule van Euler geldt in
$(x, \frac12)$: voor $x$ geheel is dit vraag 10 (met
symmetrie), voor $x = n + \frac12$ is het de weergave
hierboven. Beide kanten van de formule van Euler gehoorzamen
de afdaalrecurrente $y \mapsto y + 1$ (vraag 8 links, de
functionaalvergelijking rechts, als in vraag 10): inductie
plant de formule voort van $y = \frac12$ en $y = 1$ naar elke
$y \in \frac12\N^*$. De formule van Euler geldt dus wanneer
$2x, 2y \in \N^*$.

\textbf{15.} Met $u = \frac{t}{1-t}$, d.w.z.\ $t =
\frac{u}{1+u}$, $1 - t = \frac{1}{1+u}$, $\dd t =
\frac{\dd u}{(1+u)^2}$:
\[
B(x,y) = \int_0^\infty
\Bigl(\frac{u}{1+u}\Bigr)^{x-1}
\Bigl(\frac{1}{1+u}\Bigr)^{y-1}
\frac{\dd u}{(1+u)^2}
= \int_0^\infty \frac{u^{x-1}}{(1+u)^{x+y}}\,\dd u .
\]
In $x = y = \frac12$, met $u = v^2$:
\[
\int_0^\infty \frac{u^{-1/2}}{1+u}\dd u
= \int_0^\infty \frac{2\,\dd v}{1+v^2} = \pi
= B\Bigl(\frac12,\frac12\Bigr) . \checkmark
\]

\textbf{16.} EĂ©n partiĂ«le integratie, voor $1 \leq k \leq n$
en $s > 0$ ($u = (1 - t/n)^k$, $v = t^s/s$; de randtermen
verdwijnen):
\[
\int_0^n \Bigl(1-\frac tn\Bigr)^{\!k} t^{s-1}\dd t
= \frac{k}{ns}\int_0^n \Bigl(1-\frac tn\Bigr)^{\!k-1}
t^{s}\dd t .
\]
Startend vanuit $k = n$, $s = x$ en $n$ keer itererend:
\[
\int_0^n \Bigl(1-\frac tn\Bigr)^{\!n} t^{x-1}\dd t
= \frac{n(n-1)\cdots1}{n^n\,x(x+1)\cdots(x+n-1)}
\int_0^n t^{x+n-1}\dd t
= \frac{n!}{n^n}\cdot
\frac{n^{x+n}}{x(x+1)\cdots(x+n)} ,
\]
wat $\dfrac{n!\,n^x}{x(x+1)\cdots(x+n)}$ is.

\textbf{17.} Door \cref{exo:b2:integration:7} streeft de
linkerkant naar $\Gamma(x)$ (gedomineerde convergentie met
dominator $t^{x-1}\eu^{-t}$); de rechterkant is het
Gauss-quotiĂ«nt:
\[
\Gamma(x) = \lim_{n\to\infty}
\frac{n!\,n^x}{x(x+1)\cdots(x+n)} .
\]

\textbf{18.} Logaritmen nemend in het quotiĂ«nt $G_n(x)$ van
vraag 16 en splitsend $\ln(x+k) = \ln k + \ln(1 + x/k)$ voor $k
\geq 1$:
\[
\ln G_n(x) = x\ln n - \ln x - \sum_{k=1}^n
\ln\Bigl(1+\frac xk\Bigr)
= -\ln x + x(\ln n - H_n)
+ \sum_{k=1}^n\Bigl(\frac xk -
\ln\Bigl(1+\frac xk\Bigr)\Bigr).
\]
Voor $u \geq 0$ is $u - \frac{u^2}2 \leq \ln(1+u) \leq u$, dus
ligt de algemene term in $\intcc{0}{x^2/(2k^2)}$: de reeks
\omterm{def:b2:integration:improper}{convergeert} (vergelijking met $\sum k^{-2}$). Aangezien $\ln n
- H_n \to -\gamma$ (\cref{ex:b2:comparison:harmonic}) en $\ln
G_n(x) \to \ln\Gamma(x)$ (vraag 17 en continuĂŻteit van $\ln$):
\[
\ln\Gamma(x) = -\ln x - \gamma x
+ \sum_{k=1}^\infty\Bigl(\frac xk -
\ln\Bigl(1+\frac xk\Bigr)\Bigr) .
\]

\textbf{19.} In $x = \frac12$ is de noemer
$\prod_{k=0}^n\bigl(k+\frac12\bigr) =
\frac{(2n+1)!}{2^{2n+1}n!}$ (de helften uitwerken), dus
\[
G_n\Bigl(\frac12\Bigr)
= \frac{n!\,\sqrt n\;2^{2n+1}n!}{(2n+1)!}
= \frac{2\sqrt n\;4^n}{(2n+1)\binom{2n}{n}} .
\]
Met $\binom{2n}{n} \sim \frac{4^n}{\sqrt{\pi n}}$
(\cref{ex:b2:comparison:centralbinomial}):
\[
G_n\Bigl(\frac12\Bigr)
\sim \frac{2\sqrt n\,\sqrt{\pi n}}{2n+1}
\longrightarrow \sqrt\pi = \Gamma\Bigl(\frac12\Bigr) .
\]
De $\sqrt\pi$ van de centrale binomiaalcoĂ«fficiĂ«nt (die van
Wallis kwam, dus van de constante van Stirling) en de
$\sqrt\pi$ van de Gaussische integraal zijn hetzelfde getal.

\textbf{20.} De identiteit is directe algebra: vermenigvuldig
$\frac{n!\,n^x}{x(x+1)\cdots(x+n)}$ met $\frac{nx}{x+n+1}$ en
absorbeer $x$ in het product, $n$ in $n^x$. $n \to \infty$:
de linkerkant streeft naar $\Gamma(x+1)$ (Gauss in $x+1$), de
rechterkant naar $\Gamma(x)\cdot x\cdot 1$ want
$\frac{n}{x+n+1} \to 1$: $\Gamma(x+1) = x\Gamma(x)$ ---
hersteld zonder een enkele partiĂ«le integratie.

\textbf{21.} Met $u = t^a$, $t = u^{1/a}$, $\dd t = \frac1a
u^{1/a - 1}\dd u$:
\[
\int_0^\infty \eu^{-t^a}\dd t
= \frac1a\int_0^\infty u^{\frac1a - 1}\eu^{-u}\dd u
= \frac1a\,\Gamma\Bigl(\frac1a\Bigr)
= \Gamma\Bigl(1 + \frac1a\Bigr) .
\]
Als $a \to +\infty$ (langs enige rij): $\eu^{-t^a} \to 1$ voor
$0 < t < 1$, $\to \eu^{-1}$ in $t = 1$, $\to 0$ voor $t > 1$;
voor $a \geq 2$ domineer door $\mathbf 1_{t \leq 1} +
\eu^{-t^2}\mathbf 1_{t > 1}$ ($t^a \geq t^2$ voor $t \geq 1$),
integreerbaar. Gedomineerde convergentie: de integraal streeft
naar $\int_0^1 1\,\dd t = 1$ --- zoals het moet, want
$\Gamma(1 + \frac1a) \to \Gamma(1) = 1$ door continuĂŻteit.

\textbf{22.} Met $u = t^n$, $\dd t = \frac1n u^{1/n - 1}\dd
u$:
\[
\int_0^1 \frac{\dd t}{\sqrt{1-t^n}}
= \frac1n\int_0^1 u^{\frac1n-1}(1-u)^{-1/2}\dd u
= \frac1n\,B\Bigl(\frac1n, \frac12\Bigr) .
\]
$n = 1$: $B\bigl(1,\frac12\bigr) = B\bigl(\frac12,1\bigr) = 2$,
passend bij $\int_0^1\frac{\dd t}{\sqrt{1-t}} = 2$. $n = 2$:
$\frac12 B\bigl(\frac12,\frac12\bigr) = \frac\pi2 = \arcsin 1$.
Voor $n = 4$ is de waarde $\frac14 B\bigl(\frac14,\frac12\bigr)$
de lemniscaatconstante: geen elementaire gesloten vorm.

\textbf{23.} Itereren van de functionaalvergelijking:
\[
\frac{1}{\Gamma(x)}\int_0^\infty t^{x+k-1}\eu^{-t}\dd t
= \frac{\Gamma(x+k)}{\Gamma(x)}
= (x+k-1)(x+k-2)\cdots x ,
\]
de stijgende faculteit met $k$ factoren. In $x = 1$:
$\Gamma(1+k)/\Gamma(1) = k!$, de momenten van $\eu^{-t}$ uit
\cref{exo:b2:integration:2}.

\textbf{24.} Substitueer $t = \frac{1+s}2$ ($s \in
\intoo{-1}1$, $\dd t = \frac{\dd s}2$, $t(1-t) =
\frac{1-s^2}4$):
\[
B(x,x) = \int_{-1}^{1}\Bigl(\frac{1-s^2}{4}\Bigr)^{x-1}
\frac{\dd s}{2}
= 4^{1-x}\int_0^1 (1-s^2)^{x-1}\dd s
\]
(de integrand is even). Dan $s = \sqrt v$ ($\dd s =
\frac{\dd v}{2\sqrt v}$):
\[
B(x,x) = \frac{4^{1-x}}{2}\int_0^1
v^{-1/2}(1-v)^{x-1}\dd v
= 2^{1-2x}\,B\Bigl(\frac12, x\Bigr) .
\]
Voor $2x \in \N^*$ liggen alle argumenten in zicht in
$\frac12\N^*$, dus past de formule van Euler (vraag 14) op
beide kanten:
\[
\frac{\Gamma(x)^2}{\Gamma(2x)}
= 2^{1-2x}\,
\frac{\Gamma\bigl(\frac12\bigr)\Gamma(x)}
{\Gamma\bigl(x+\frac12\bigr)}
\quad\Longleftrightarrow\quad
\Gamma(x)\,\Gamma\Bigl(x+\frac12\Bigr)
= 2^{1-2x}\sqrt\pi\;\Gamma(2x) .
\]
Directe controle in $x = n$: de linkerkant is $(n-1)!\cdot
\frac{(2n)!\sqrt\pi}{4^n n!} = \frac{(2n)!\sqrt\pi}{4^n n}$,
de rechterkant $2\cdot4^{-n}\sqrt\pi\,(2n-1)! =
\frac{(2n)!\sqrt\pi}{4^n n}$: gelijk.
