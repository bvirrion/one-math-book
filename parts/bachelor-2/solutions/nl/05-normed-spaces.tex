\section*{Hoofdstuk \ref{ch:b2:nvs} --- Genormeerde vectorruimten}

\begin{solution}{exo:b2:nvs:1}
Eenheidsbollen: een ruit ($\norm\cdot_1$), een schijf
($\norm\cdot_2$), een vierkant ($\norm\cdot_\infty$), genest in die
volgorde. Ongelijkheden: $\norm x_\infty \leq \norm x_2$ (één kwadraat
is ten hoogste de som); $\norm x_2 \leq \norm x_1$ (kwadrateren:
$x_1^2 + x_2^2 \leq (\abs{x_1} + \abs{x_2})^2$); $\norm x_1 \leq
2\norm x_\infty$ (twee termen, elk $\leq \max$). Scherpte: $(1, 0)$
maakt de eerste twee tot gelijkheden; $(1, 1)$ maakt $\norm x_1 =
2\norm x_\infty$ en toont ook dat $\norm x_2 = \sqrt2 \norm
x_\infty$ en $\norm x_1 = \sqrt2 \norm x_2$ de extreme ratio's in de
andere richting zijn.
\end{solution}

\begin{solution}{exo:b2:nvs:2}
Normaxioma's: homogeniteit en driehoeksongelijkheid worden
termgewijs geërfd; scheiding: $N(f) = 0$ forceert $f' = 0$ (dus $f$
constant) en $f(0) = 0$: $f = 0$. Het is een \omterm{def:b2:nvs:norm}{norm}.

Vergelijking: $\norm f_\infty \leq N(f)$, aangezien $\abs{f(x)} \leq
\abs{f(0)} + \abs{\int_0^x f'} \leq \abs{f(0)} +
\norm{f'}_\infty$. Het omgekeerde faalt: neem $f_n(x) = \frac1n
\sin(nx)$: dan $\norm{f_n}_\infty \leq \frac1n \to 0$ terwijl
$N(f_n) = 0 + \norm{\cos(nx)}_\infty = 1$. Geen constante $C$ geeft
$N \leq C\norm\cdot_\infty$.
\end{solution}

\begin{solution}{exo:b2:nvs:3}
$\abs{u(f)} \leq \norm f_\infty \int_0^1 \eu^t\,\dd t = (\eu -
1)\norm f_\infty$, met gelijkheid voor $f \equiv 1$: $\vertiii u =
\eu - 1$.

Verschuiving: $\norm{S(x)}_\infty = \max(\abs{x_2}, \dots,
\abs{x_n}) \leq \norm x_\infty$, gelijkheid in $x = e_2$: $\vertiii
S = 1$ (voor $n \geq 2$).
\end{solution}

\begin{solution}{exo:b2:nvs:4}
Bovengrens: voor $\norm x_\infty \leq 1$,
\[
\abs{(Ax)_i} = \Bigl|\sum_j a_{ij}x_j\Bigr| \leq \sum_j
\abs{a_{ij}},
\]
dus $\norm{Ax}_\infty \leq \max_i \sum_j \abs{a_{ij}}$. Bereikt: zij
$i_0$ het max realiseren en neem $x_j = \operatorname{sign}(a_{i_0
j})$ (modulus-$1$-entries): dan $(Ax)_{i_0} = \sum_j \abs{a_{i_0
j}}$. Vandaar de formule. Voor de gegeven matrix: rijsommen $3$ en
$4$: $\vertiii A_\infty = 4$.
\end{solution}

\begin{solution}{exo:b2:nvs:5}
\emph{Meetkundige reeks:} voor $\vertiii B < 1$ convergeert de reeks
$\sum B^k$ absoluut in de Banach $\mathcal{M}_n(K)$
(\cref{thm:b2:nvs:absoluteconvergence}, $\vertiii{B^k} \leq
\vertiii B^k$), en
\[
(I - B)\sum_{k=0}^{K} B^k = I - B^{K+1} \longrightarrow I :
\]
door continuïteit van het product
(\cref{prop:b2:nvs:bilinear}), $(I - B)\sum_{k\geq0} B^k = I$, dus
$I - B$ is inverteerbaar met inverse de som (en
$\vertiii{(I-B)^{-1}} \leq \frac{1}{1 - \vertiii B}$).

\emph{Openheid:} voor $A$ inverteerbaar en $\vertiii H <
\frac{1}{\vertiii{A^{-1}}}$: $A + H = A(I + A^{-1}H)$ met
$\vertiii{A^{-1}H} \leq \vertiii{A^{-1}}\vertiii H < 1$:
inverteerbaar. Dus blijft een bol rond $A$ in $GL_n$.

\emph{Continuïteit van inversie:} met $B = -A^{-1}H$,
\[
(A + H)^{-1} - A^{-1} = \bigl((I - B)^{-1} - I\bigr)A^{-1}
= \Bigl(\sum_{k \geq 1} B^k\Bigr) A^{-1},
\]
van \omterm{def:b2:nvs:norm}{norm} $\leq \frac{\vertiii B}{1 - \vertiii B}\vertiii{A^{-1}}
\to 0$ als $H \to 0$.
\end{solution}

\begin{solution}{exo:b2:nvs:6}
Continu $\Rightarrow$ gesloten kern: origineel van de gesloten
$\{0\}$ (\cref{thm:b2:metric:globalcontinuity}).

Omgekeerd, veronderstel $\ker\varphi$ gesloten en $\varphi \neq 0$.
Kies $a$ met $\varphi(a) = 1$; aangezien $a \notin \ker\varphi$ en
de kern gesloten is, mist zekere bol $B(a, r)$ haar. Zij nu $h \in
E$ met $\varphi(h) \neq 0$: de vector $a - \frac{h}{\varphi(h)}$
ligt in $\ker\varphi$, dus buiten $B(a, r)$:
\[
\Bigl\Vert \frac{h}{\varphi(h)} \Bigr\Vert \geq r
\quad\Longrightarrow\quad
\abs{\varphi(h)} \leq \frac{\norm h}{r},
\]
een ongelijkheid die triviaal ook geldt wanneer $\varphi(h) = 0$:
grens (4) van \cref{thm:b2:nvs:continuouslinear}: continu.
\end{solution}

\begin{solution}{exo:b2:nvs:7}
Zij $f_n$ gelijk aan $0$ op $\intcc{0}{\frac12 - \frac1n}$, affien
omhoog tot waarde $1$ in $\frac12$, dan $1$ op $\intcc{\frac12}{1}$.
Voor $m \geq n$ is $f_m - f_n$ gesteund op een interval van lengte
$\frac1n$ met waarden in $\intcc{-1}{1}$: $\norm{f_m - f_n}_1 \leq
\frac1n$: Cauchy.

Veronderstel $f_n \to f$ in $\norm\cdot_1$ met $f$ continu. Op
$\intcc{0}{\frac12 - \delta}$ (vaste $\delta$): $\int \abs{f} =
\int\abs{f - f_n} \leq \norm{f - f_n}_1 \to 0$ voor $n >
\frac1\delta$, dus $\int_0^{1/2 - \delta}\abs f = 0$, en door
strikte positiviteit $f = 0$ daar --- voor elke $\delta$: $f = 0$
op $\intoo{0}{\frac12}$. Evenzo $f = 1$ op $\intcc{\frac12}{1}$
(de $f_n$ zijn daar allemaal $1$). Door continuïteit in $\frac12$:
$0 = 1$, absurd. Geen limiet bestaat: de ruimte is niet \omterm{def:b2:metric:complete}{volledig}.
\end{solution}

\begin{solution}{exo:b2:nvs:8}
Welgedefinieerd en continu: $\abs{\varphi(f)} \leq \sum 2^{-n}
\norm f_\infty = \norm f_\infty$, dus $\varphi$ is een lineaire
vorm met $\vertiii\varphi \leq 1$ (de reeks convergeert absoluut
voor elke $f$).

\omterm{def:b2:nvs:norm}{Norm} $1$: fixeer $K$; de punten $1, \frac12, \dots, \frac1K$ zijn
paarsgewijs verschillend, dus bestaat een continue $f_K$,
$\norm{f_K}_\infty \leq 1$, met $f_K(\frac1n) = (-1)^n$ voor $n
\leq K$ (stuksgewijs affiene interpolatie, constant nabij $0$). Dan
\[
\varphi(f_K) \geq \sum_{n=1}^{K} 2^{-n} - \sum_{n > K} 2^{-n}
= 1 - 2^{-K+1} \xrightarrow[K \to \infty]{} 1 .
\]

Niet bereikt: als $\norm f_\infty \leq 1$ en $\varphi(f) = 1$, moet
elke term haar maximum bijdragen: $(-1)^n f(\frac1n) = 1$ voor elke
$n$ (anders kan het strikte tekort van één term niet gecompenseerd
worden, alle termen $\leq 2^{-n}$). Dus $f(\frac1n) = (-1)^n$; maar
$\frac1n \to 0$ en $f$ is continu in $0$, wat de tegensprekende
convergentie van $(-1)^n$ forceert. Dus is het supremum geen
maximum --- onmogelijk in eindige dimensie, waar de gesloten
eenheidsbol \omterm{def:b2:metric:compact}{compact} is.
\end{solution}

\begin{solution}{exo:b2:nvs:9}
\emph{Begrensde rijen:} een begrensde rij ligt in zekere gesloten
bol $\overline B(0, R) = R\,\overline B(0,1)$, \omterm{def:b2:metric:compact}{compact} (beeld van
de compacte eenheidsbol onder het homeomorfisme $x \mapsto Rx$):
extraheer daar.

\emph{Lineaire vormen:} als $\dim E < \infty$, is elke lineaire
afbeelding vanuit $E$ continu (\cref{thm:b2:nvs:finitedim}).
Omgekeerd, veronderstel $\dim E = \infty$ (wat door
\cref{thm:b2:nvs:riesz} eigenlijk uitgesloten wordt door de
compactheidshypothese --- het punt van deze vraag is de implicatie
tussen de twee eigenschappen in algemene genormeerde ruimten):
kies een genormaliseerde lineair onafhankelijke rij $(e_n)$, vul
haar aan tot een algebraïsche basis (toegegeven), en definieer
$\varphi(e_n) = n$, $\varphi = 0$ op de andere basisvectoren,
lineair uitgebreid. Dan $\abs{\varphi(e_n)} = n$ met $\norm{e_n} =
1$: onbegrensd op de eenheidsbol, discontinu. Dus forceert ``alle
vormen continu'' eindige dimensie.
\end{solution}

\begin{solution}{exo:b2:nvs:10}
Cauchy--Schwarz met de constante functie $1$: $\norm f_1 =
\int_0^1 \abs f\cdot 1 \leq \bigl(\int_0^1
f^2\bigr)^{1/2}\bigl(\int_0^1 1\bigr)^{1/2} = \norm f_2$. En
$\norm f_2^2 = \int f^2 \leq \norm f_\infty^2$. Voor $f_n(x) =
x^n$:
\[
\norm{f_n}_1 = \frac1{n+1}, \qquad
\norm{f_n}_2 = \frac1{\sqrt{2n+1}}, \qquad
\norm{f_n}_\infty = 1 .
\]
Dan $\norm{f_n}_2/\norm{f_n}_1 = \frac{n+1}{\sqrt{2n+1}} \to
\infty$ en $\norm{f_n}_\infty/\norm{f_n}_2 = \sqrt{2n+1} \to
\infty$: geen omgekeerde ongelijkheden, geen paar is \omterm{def:b2:nvs:equivalent}{equivalent}.
\end{solution}

\begin{solution}{exo:b2:nvs:11}
\emph{Ondergrens voor de afstand:} voor $h \in \ker\varphi$,
$\abs{\varphi(x)} = \abs{\varphi(x - h)} \leq
\vertiii\varphi\,\norm{x - h}$; neem het infimum over $h$:
$d(x, \ker\varphi) \geq \abs{\varphi(x)}/\vertiii\varphi$.

\emph{Bovengrens:} we mogen $\varphi(x) \neq 0$ aannemen. Gegeven
$\varepsilon > 0$, kies een eenheid $u$ met $\abs{\varphi(u)} \geq
\vertiii\varphi - \varepsilon > 0$ en stel $h = x -
\frac{\varphi(x)}{\varphi(u)}\,u$: dan $\varphi(h) = 0$ en
\[
\norm{x - h} = \frac{\abs{\varphi(x)}}{\abs{\varphi(u)}}
\leq \frac{\abs{\varphi(x)}}{\vertiii\varphi - \varepsilon}.
\]
Laat $\varepsilon \to 0$: $d(x, \ker\varphi) \leq
\abs{\varphi(x)}/\vertiii\varphi$; gelijkheid.

\emph{Niet-bereiking:} neem $\varphi$ uit \cref{exo:b2:nvs:8}
($\vertiii\varphi = 1$, niet bereikt) en elke $x$ met $\varphi(x)
\neq 0$. Als zekere $h \in \ker\varphi$ $\norm{x - h} =
\abs{\varphi(x)}$ realiseerde, zou de eenheidsvector $v = (x -
h)/\norm{x - h}$ voldoen aan $\abs{\varphi(v)} =
\abs{\varphi(x)}/\norm{x - h} = 1 = \vertiii\varphi$: de
\omterm{thm:b2:nvs:continuouslinear}{operatornorm} zou bereikt worden --- tegenspraak.
\end{solution}

\begin{solution}{exo:b2:nvs:12}
$N_1 \leq N_2$ is monotoniciteit van de sup in het domein, dus de
identiteit $(E, N_2) \to (E, N_1)$ is $1$-Lipschitz. Voor $P_n(x) =
(x/2)^n$: $N_2(P_n) = 1$ (bereikt in $x = 2$) terwijl $N_1(P_n) =
2^{-n}$: een grens $N_2 \leq CN_1$ zou $1 \leq C2^{-n}$ voor alle
$n$ geven: onmogelijk. Dus zijn de \omterm{def:b2:nvs:norm}{normen} niet \omterm{def:b2:nvs:equivalent}{equivalent} en is de
inverse identiteit een discontinue lineaire bijectie.

\emph{Onvolledigheid:} zij $S_n = \sum_{k=0}^{n}\frac{X^k}{k!}$.
Voor $m > n$, $N_1(S_m - S_n) \leq \sum_{k>n}\frac1{k!} \to 0$:
Cauchy voor $N_1$. Als $S_n \to P$ in $(E, N_1)$, dan puntgewijs
$P(x) = \lim S_n(x) = \eu^x$ op $\intcc{0}{1}$; maar een polynoom
van graad $d$ kan niet gelijk zijn aan $\eu^x$ op een interval
(differentieer $d + 1$ keer: de linkerkant sterft, $\eu^x$ niet).
Geen limiet in $E$: niet \omterm{def:b2:metric:complete}{volledig}.

In eindige dimensie zijn alle drie de fenomenen onmogelijk: alle
\omterm{def:b2:nvs:norm}{normen} zijn \omterm{def:b2:nvs:equivalent}{equivalent}, elke genormeerde ruimte is \omterm{def:b2:metric:complete}{volledig}, en de
inverse van een lineaire bijectie is lineair vanuit een
eindigdimensionale ruimte, dus continu (\cref{thm:b2:nvs:finitedim}).
\end{solution}

\begin{solution}{pb:b2:nvs:1}
\textbf{1.} De kandidaat-waardige vormen $K = \{f \in F : \norm{x
- f} \leq \norm x\}$: niet-leeg ($0 \in K$), gesloten (origineel
van een gesloten interval onder de continue $f \mapsto \norm{x -
f}$, doorsneden met de gesloten $F$,
\cref{cor:b2:nvs:closedsubspace}), begrensd ($\norm f \leq
\norm{f - x} + \norm x \leq 2\norm x$). In de eindigdimensionale
$F$ betekent gesloten en begrensd \omterm{def:b2:metric:compact}{compact}
(\cref{thm:b2:nvs:finitedim}); de continue functie $f \mapsto
\norm{x - f}$ bereikt op $K$ haar infimum, dat gelijk is aan het
infimum over heel $F$ (elke $f \notin K$ geeft $\norm{x - f} >
\norm x \geq \inf$).

\textbf{2.} Parallellogramidentiteit in $(\R^n, \norm\cdot_2)$:
$\norm{u + v}^2 + \norm{u - v}^2 = 2\norm u^2 + 2\norm v^2$
(ontwikkel de kwadraten van coördinaatsommen). Voor eenheden $u
\neq v$:
\[
\Bigl\Vert\frac{u+v}2\Bigr\Vert^2 = 1 - \frac{\norm{u -
v}^2}{4} < 1 .
\]
Niet strikt convex: voor $\norm\cdot_\infty$, neem $u = (1, 1, 0,
\dots)$, $v = (1, -1, 0, \dots)$: eenheidsvectoren met middenpunt
$(1, 0, \dots)$ van \omterm{def:b2:nvs:norm}{norm} $1$; voor $\norm\cdot_1$, neem $u = (1,
0, \dots)$, $v = (0, 1, 0, \dots)$: middenpunt $(\frac12,
\frac12, 0, \dots)$ van \omterm{def:b2:nvs:norm}{norm} $1$.

\textbf{3.} Zij $d = d(x, F)$. Als $d = 0$: $x \in \overline F =
F$ en de enige minimizer is $x$. Als $d > 0$ en $f_1 \neq f_2$
beide minimaliseren: $u = \frac{x - f_1}{d}$ en $v = \frac{x -
f_2}{d}$ zijn verschillende eenheidsvectoren, dus
\[
\Bigl\Vert x - \frac{f_1 + f_2}2\Bigr\Vert
= d\,\Bigl\Vert\frac{u + v}2\Bigr\Vert < d ,
\]
met $\frac{f_1 + f_2}2 \in F$: tegenspraak met de definitie van
$d$. Unieke minimizer.

\textbf{4.} $\norm{(0,1) - t(1,0)}_\infty = \max(\abs t, 1)
\geq 1$, met gelijkheid iff $\abs t \leq 1$: de minimizers vormen
het segment $\{t(1, 0) : t \in \intcc{-1}{1}\}$, allen op afstand
$1$ --- uniciteit faalt precies omdat de vierkante bol platte
zijden heeft (vraag 2).

\textbf{5.} Zij $M = \max f$, $m = \min f$ (bereikt: compactheid).
Voor elke constante $c$: $\sup\abs{f - c} \geq \max(M - c,\, c -
m) \geq \frac{M - m}2$, de laatste ongelijkheid omdat de twee
grootheden middelen tot $\frac{M-m}2$; gelijkheid in beide forceert
$M - c = c - m$, d.w.z.\ $c = c^* = \frac{M + m}2$. Omgekeerd
$\sup\abs{f - c^*} = \max(M - c^*, c^* - m) = \frac{M - m}2$.
Unieke beste constante. Voor $f(x) = x^2$ op $\intcc01$: $c^* =
\frac12$, afstand $\frac12$.

\textbf{6.} Uit $\cos(n{+}1)\theta + \cos(n{-}1)\theta =
2\cos\theta\cos n\theta$: de polynomen gedefinieerd door $T_0 =
1$, $T_1 = X$, $T_{n+1} = 2XT_n - T_{n-1}$ voldoen aan
$T_n(\cos\theta) = \cos n\theta$ door inductie. Uniciteit: twee
polynomen die op $\intcc{-1}{1}$ (oneindig veel punten) overeenkomen
zijn gelijk. Opnieuw inductie: $\deg T_n = n$ met leidende
coëfficiënt $2^{n-1}$ voor $n \geq 1$ ($T_1$: coëfficiënt $1 =
2^0$; de recurrentie verdubbelt haar).

\textbf{7.} Elke $x \in \intcc{-1}{1}$ is $\cos\theta$, en
$\abs{\cos n\theta} \leq 1$. In $\eta_k = \cos\frac{k\pi}n$:
$T_n(\eta_k) = \cos k\pi = (-1)^k$, en $1 = \eta_0 > \eta_1 >
\dots > \eta_n = -1$. Wortels: $\cos n\theta = 0$ iff $\theta =
\frac{(2k-1)\pi}{2n}$: de $n$ verschillende punten
$\cos\frac{(2k-1)\pi}{2n}$, en aangezien $\frac{(k-1)\pi}n <
\frac{(2k-1)\pi}{2n} < \frac{k\pi}n$, ligt elke wortel strikt
tussen twee opeenvolgende extrema.

\textbf{8.} $T_2 = 2X^2 - 1$, $T_3 = 4X^3 - 3X$, $T_4 = 8X^4 -
8X^2 + 1$. Voor $T_3$: $T_3(1) = 1$, $T_3(\tfrac12) = \tfrac12 -
\tfrac32 = -1$, $T_3(-\tfrac12) = 1$, $T_3(-1) = -1$: perfecte
alternatie.

\textbf{9.} Zij $u_\pm = x \pm \sqrt{x^2 - 1}$ voor $x \geq 1$: de
wortels van $z^2 - 2xz + 1$, met $u_+u_- = 1$. De rij $s_n =
\frac{u_+^n + u_-^n}2$ voldoet aan $s_{n+1} = 2x\,s_n - s_{n-1}$
(Newton-achtige recurrentie uit de kwadratische), $s_0 = 1$, $s_1
= x$: dezelfde recurrentie en beginwaarden als $n \mapsto T_n(x)$,
dus $s_n = T_n(x)$ voor alle $n$. Aangezien $0 < u_- \leq 1 \leq
u_+$ met $u_+ > 1$ voor $x > 1$: $T_n(x) \geq \frac{u_+^n}2 \to
\infty$ en $T_n(x) \sim \frac12\bigl(x + \sqrt{x^2-1}\bigr)^n$.
(Voor $x \leq -1$ gebruik de pariteit $T_n(-x) = (-1)^nT_n(x)$,
duidelijk uit de recurrentie.)

\textbf{10.} Voor elke $\theta$: $T_m\bigl(T_n(\cos\theta)\bigr) =
T_m(\cos n\theta) = \cos mn\theta = T_{mn}(\cos\theta)$. De
polynomen $T_m \circ T_n$ en $T_{mn}$ komen overeen op
$\intcc{-1}{1}$, dus zijn gelijk.

\textbf{11.} $D = Q_n - P$ heeft graad $\leq n - 1$ (de monische
leidende termen heffen op). In de extrema: $(-1)^kD(\eta_k) =
2^{1-n} - (-1)^kP(\eta_k) \geq 2^{1-n} - \abs{P(\eta_k)} > 0$ door
de hypothese. Dus neemt $D$ niet-nulle waarden van wisselend teken
aan in de $n + 1$ dalende punten $\eta_0 > \dots > \eta_n$: door de
tussenwaardestelling heeft zij minstens $n$ verschillende
nulpunten, één in elk open interval
$\intoo{\eta_{k}}{\eta_{k-1}}$. Een niet-nul polynoom van graad
$\leq n - 1$ kan geen $n$ nulpunten hebben; en $D = 0$ spreekt de
strikte tekens tegen. Tegenspraak: $\sup\abs P \geq 2^{1-n}$ voor
elke monische $P$ van graad $n$.

\textbf{12.} Nu $(-1)^kD(\eta_k) = 2^{1-n} - (-1)^kP(\eta_k)
\geq 0$ aangezien $\abs{P} \leq 2^{1-n}$. Op elk
$\intcc{\eta_k}{\eta_{k-1}}$ ($k = 1, \dots, n$) hebben de
eindpuntwaarden van $D$ tegengestelde zwakke tekens: de
tussenwaardestelling levert een nulpunt $z_k$ in het gesloten
interval. Als de $z_k$ paarsgewijs verschillend gekozen kunnen
worden, heeft $D \neq 0$ van graad $\leq n-1$ $n$ nulpunten:
tegenspraak. Twee opeenvolgende intervallen kunnen alleen het
nulpunt $z_k = z_{k+1} = \eta_k$ met $0 < k < n$ delen (inwendig).
Daar betekent $D(\eta_k) = 0$ dat $P(\eta_k) = (-1)^k2^{1-n}$, een
extremale waarde van $P$ op $\intcc{-1}{1}$ bereikt in een
\emph{inwendig} punt: $P'(\eta_k) = 0$; en $\eta_k$ is ook een
inwendig extremum van $T_n$: $Q_n'(\eta_k) = 0$. Dus $D'(\eta_k)
= 0$: $\eta_k$ is een nulpunt van multipliciteit $\geq 2$, dat het
gedeelde interval compenseert. In alle gevallen heeft $D$ minstens
$n$ nulpunten geteld met multipliciteit, graad $\leq n - 1$, dus
$D = 0$: $P = Q_n$. De extremale stelling van Chebyshev is bewezen:
de unieke monische minimizer is $2^{1-n}T_n$, van \omterm{def:b2:nvs:norm}{sup-norm}
$2^{1-n}$.

\textbf{13.} Monische polynomen van graad $n$ zijn precies de $X^n
- R$ met $R \in \R_{n-1}[X]$, dus
\[
d_\infty\bigl(X^n, \R_{n-1}[X]\bigr) = \min_{P \text{
monisch}}\ \sup_{\intcc{-1}{1}}\abs P = 2^{1-n},
\]
uniek in $R^* = X^n - Q_n$. Substitutie $x = \frac{1+t}2$: als $P$
monisch van graad $n$ is op $\intcc01$, dan is $t \mapsto
2^nP\bigl(\frac{1+t}2\bigr)$ monisch op $\intcc{-1}{1}$ met sup
gelijk aan $2^n\sup_{\intcc01}\abs P$: dus
$\sup_{\intcc01}\abs P \geq 2^{-n}\cdot2^{1-n} = 2^{1-2n}$, met
gelijkheid precies voor $P^*(x) = 2^{-n}Q_n(2x - 1)$: op
$\intcc{0}{1}$ is de afstand $2^{1-2n}$.

\textbf{14.} $\omega$ is monisch van graad $n$, dus
$\sup_{\intcc{-1}{1}}\abs\omega \geq 2^{1-n}$ met gelijkheid iff
$\omega = Q_n = 2^{1-n}T_n$, d.w.z.\ iff de knopen de $n$ wortels
van $T_n$ zijn: $x_k = \cos\frac{(2k-1)\pi}{2n}$ --- de
\emph{Chebyshevknopen}. Minimale waarde: $2^{1-n}$. Equidistante
knopen zijn strikt slechter; de interpolatiefoutfactor wordt
geminimaliseerd door knopen nabij de eindpunten te clusteren.

\textbf{15.} $n = 2$ met de hand: $x^2$ loopt over $\intcc01$, dus
$\sup_{\intcc{-1}{1}}\abs{x^2 - c} = \max(\abs c, \abs{1 - c})
\geq \frac12$, geminimaliseerd in $c = \frac12$: minimale monische
kwadratische $x^2 - \frac12 = \frac12(2x^2 - 1) = Q_2$, waarde
$\frac12 = 2^{1-2}$. Voor $n = 10$ op $\intcc{0}{1}$: $2^{1-20} =
2^{-19} \approx 1.9\cdot10^{-6}$. Zeker polynoom van graad $9$
blijft binnen twee miljoensten van $x^{10}$ op heel $\intcc01$: op
deze schaal zijn de twee grafieken niet te onderscheiden --- de
vlakheid van $x^{10}$ nabij $0$ laat lagere graden al het werk doen.

\textbf{16.} $\norm{g_k}_\infty = g_k(1) = 1$. Voor $j > k$ stel
$a = 2^k$, $b = 2^j \geq 2a$, en evalueer in $x_0 = 2^{-1/a}$
(zodat $x_0^a = \frac12$):
\[
g_k(x_0) - g_j(x_0) = \frac12 - \Bigl(\frac12\Bigr)^{b/a}
\geq \frac12 - \frac14 = \frac14 .
\]
Dus $\norm{g_k - g_j}_\infty \geq \frac14$ voor alle $j \neq k$:
geen deelrij is Cauchy, geen convergeert. De gesloten eenheidsbol
van $\bigl(C(\intcc01), \norm\cdot_\infty\bigr)$ is niet \omterm{def:b2:metric:compact}{compact}.

\textbf{17.} $F$ is gesloten (\cref{cor:b2:nvs:closedsubspace}) en
echt: kies $y \notin F$, dus $\delta = d(y, F) > 0$. Door vraag 1
wordt de afstand bereikt in zekere $f^* \in F$. Stel $x =
\frac{y - f^*}{\delta}$, een eenheidsvector ($\norm{y - f^*} =
\delta$). Voor elke $g \in F$:
\[
\norm{x - g} = \frac{\norm{y - (f^* + \delta g)}}{\delta}
\geq \frac{\delta}{\delta} = 1 ,
\]
aangezien $f^* + \delta g \in F$. Dus $d(x, F) \geq 1$; en $d(x,
F) \leq \norm{x - 0} = 1$: precies $1$.

\textbf{18.} In een oneindigdimensionale $E$, bouw eenheidsvectoren
inductief: $x_1$ willekeurig; gegeven $x_1, \dots, x_k$, is de
deelruimte $F_k = \operatorname{Vect}(x_1, \dots, x_k)$
eindigdimensionaal, dus echt, en vraag 17 levert een eenheid
$x_{k+1}$ met $d(x_{k+1}, F_k) = 1$: in het bijzonder
$\norm{x_{k+1} - x_i} \geq 1$ voor $i \leq k$. De rij heeft
paarsgewijze afstanden $\geq 1$: de eenheidsbol bevat een rij
zonder convergente deelrij, is dus niet \omterm{def:b2:metric:compact}{compact} --- de stelling van
Riesz, met de scherpe constante $1$.

\textbf{19.} Zij $h_n$ affien op elke helft van
$\bigl[\frac1{n+1}, \frac1n\bigr]$, van $0$ omhoog tot $1$ in het
middenpunt en terug naar $0$, en nul elders: continu, $\norm{h_n}_\infty
= 1$. Voor $n \neq m$ ontmoeten de steunen hoogstens in een
gemeenschappelijk eindpunt, waar beide verdwijnen; in de piek van
$h_n$ is $h_m = 0$: $\norm{h_n - h_m}_\infty = 1$ precies. Voor
vaste $x > 0$: $h_n(x) = 0$ zodra $\frac1n < x$, en $h_n(0) = 0$
altijd: $h_n \to 0$ puntgewijs; maar $\norm{h_n - 0}_\infty = 1$:
niet uniform. Een expliciete constellatie op onderlinge afstand $1$
binnen de eenheidsbol.

\textbf{20.} Een bol van straal $\frac13$ heeft diameter $\leq
\frac23 < 1$, dus bevat ten hoogste één van de $h_n$ (twee liggen
op afstand $1$). Eindig veel zulke bollen bevatten eindig veel van
de oneindig veel $h_n$: zij kunnen de eenheidsbol niet overdekken.
Totale begrensdheid --- die \omterm{def:b2:metric:compact}{compacte metrische ruimten} genieten,
door het bewijs van \cref{thm:b2:metric:borellebesgue} --- faalt zo
erg als maar kan.

\textbf{21.} Als $Ax = \lambda x$ met $x \neq 0$:
$\abs\lambda\,\norm x = \norm{Ax} \leq \vertiii A\,\norm x$, dus
$\abs\lambda \leq \vertiii A$. Voor
$\begin{psmallmatrix}1 & -2\\ 3 & 1\end{psmallmatrix}$:
$\vertiii A_\infty = \max(1 + 2,\ 3 + 1) = 4$
(\cref{exo:b2:nvs:4}), dus heeft elke \omterm{def:b2:reduction:eigen}{eigenwaarde} modulus $\leq
4$; in feite geeft $\chi_A = X^2 - 2X + 7$ $\lambda = 1 \pm
\iu\sqrt6$ van modulus $\sqrt7 \approx 2.65$: de grens is geldig,
niet scherp.

\textbf{22.} $N_P$ is een \omterm{def:b2:nvs:norm}{norm}: $N_P(x) = 0$ forceert $P^{-1}x =
0$, dus $x = 0$; homogeniteit en driehoeksongelijkheid worden
geërfd van $\norm\cdot_\infty$ via de lineaire $P^{-1}$.
\omterm{thm:b2:nvs:continuouslinear}{Operatornorm}: met $y = P^{-1}x$ en $D = \mathrm{diag}(\lambda_i)$,
\[
N_P(Ax) = \norm{P^{-1}AP\,y}_\infty = \norm{Dy}_\infty,
\]
dus is $\vertiii A_{N_P}$ de $\norm\cdot_\infty$-operatornorm van
$D$, die haar grootste absolute rijsom is
(\cref{exo:b2:nvs:4}): $\max_i\abs{\lambda_i}$.

\textbf{23.} Als alle $\abs{\lambda_i} < 1$: $N_P(A^kx) \leq
\rho^kN_P(x)$ met $\rho = \max\abs{\lambda_i} < 1$, dus $A^kx \to
0$ voor elke $x$, en $A^k \to 0$ in elke \omterm{def:b2:nvs:norm}{norm} op $\mathcal{M}_n$
(alle \omterm{def:b2:nvs:equivalent}{equivalent} in eindige dimensie,
\cref{thm:b2:nvs:finitedim}; convergentie van $A^ke_j$ voor elke
$j$ is entrygewijze convergentie). Als zekere $\abs{\lambda} \geq
1$ met \omterm{def:b2:reduction:eigen}{eigenvector} $x$: $\norm{A^kx} = \abs\lambda^k\norm x
\not\to 0$. Voor $A = \frac14\begin{psmallmatrix}1 & 2\\ 2 &
1\end{psmallmatrix}$: \omterm{def:b2:reduction:eigen}{eigenwaarden} $\frac14(1 \pm 2) = \frac34,
-\frac14$, beide van modulus $< 1$: $A^k \to 0$.

\textbf{24.} Beide zijn \omterm{def:b2:nvs:norm}{normen} op de eindigdimensionale $\R_n[X]$,
dus \omterm{def:b2:nvs:equivalent}{equivalent} voor elke $n$. Neem de minimale monische $P^*_n$ van
vraag 13 op $\intcc01$: haar coëfficiënt van $X^n$ is $1$, dus
$N_c(P^*_n) \geq 1$, terwijl $\norm{P^*_n}_{\intcc01} = 2^{1-2n}$.
Dus
\[
C_n \geq \frac{N_c(P^*_n)}{\norm{P^*_n}_{\intcc01}}
\geq 2^{2n-1} .
\]
De equivalentieconstanten exploderen met de dimensie: op de unie
$\R[X]$ dient geen enkele constante, precies de niet-equivalentie
gezien in \cref{exo:b2:nvs:12} --- ``alle \omterm{def:b2:nvs:norm}{normen} zijn \omterm{def:b2:nvs:equivalent}{equivalent}''
is een stelling over één dimensie tegelijk, en oneindige dimensie
is waar zij sterft.

\textbf{25.} Compactheid van gesloten begrensde verzamelingen in
eindige dimensie produceerde bestaan van beste benaderingen (vraag
1) en voedde de nulpuntstelling in de $n + 1$ extremale punten
(vragen 11--12, via bereikte sups). Strikte convexiteit bestuurt
uniciteit van beste benadering --- ronde bollen geven één
minimizer, plat-zijdige bollen geven segmenten ervan (vragen
2--4). De constellatie van eenheidsvectoren op onderlinge afstand
$1$ (vragen 17--19) vernietigt compactheid van de eenheidsbol en
totale begrensdheid mee (vraag 20). Vraag 24 kwantificeert de
ineenstorting: de constanten die twee \omterm{def:b2:nvs:norm}{normen} op $\R_n[X]$
relateren groeien als $4^n$, dus overleeft geen uniforme
vergelijking de overgang naar $\R[X]$. De top is de extremale
stelling van Chebyshev (vragen 11--12): de unieke monische
minimizer $2^{1-n}T_n$. Beste benadering vindt haar moderne thuis
in Hilbertruimten, waar de projectiestelling compactheid vervangt
door volledigheid plus de parallellogramidentiteit --- eerlijk
bewijs in het volume van Jaar~3.
\end{solution}
