\section*{Hoofdstuk \ref{ch:b2:powerseries} --- Machtreeksen}

\begin{solution}{exo:b2:powerseries:1}
$\sum \frac{n^2}{2^n}z^n$: quotiënt $\frac{(n+1)^2}{2^{n+1}}\cdot
\frac{2^n}{n^2} \to \frac12$: $R = 2$.

$\sum \frac{z^n}{\binom{2n}{n}}$: $\binom{2n}{n} \sim
\frac{4^n}{\sqrt{\pi n}}$ (\cref{ex:b2:comparison:centralbinomial}),
dus $\abs{a_n}^{-1} \approx 4^n$ tot op polynomiale factoren: $R = 4$
(quotiëntcriterium: $\frac{\binom{2n}{n}}{\binom{2n+2}{n+1}} =
\frac{(n+1)^2}{(2n+1)(2n+2)} \to \frac14$).

$\sum z^{n!}$: coëfficiënten $a_k = 1$ als $k = n!$, anders $0$. Voor
$\abs z < 1$ convergeert $\sum \abs z^{n!}$ (overheerst door
meetkundig); voor $\abs z \geq 1$ gaan de termen niet naar $0$: $R =
1$.

$\sum (2 + (-1)^n)^n z^n$: coëfficiënten $3^n$ (even $n$) en $1$
(oneven $n$). Begrensdheid van $a_nr^n$ eist $3r \leq 1$; en $r <
\frac13$ werkt: $R = \frac13$.
\end{solution}

\begin{solution}{exo:b2:powerseries:2}
Alle drie hebben straal $1$ (quotiëntcriterium). In $z = 1$: $\sum 1$
divergeert; $\sum\frac1n$ divergeert; $\sum\frac{1}{n^2}$ convergeert. In
$z = -1$: $\sum(-1)^n$ divergeert; $\sum\frac{(-1)^n}{n}$ convergeert
(alternerend); $\sum\frac{(-1)^n}{n^2}$ convergeert (absoluut).
Randgedrag is onzichtbaar voor de straal.
\end{solution}

\begin{solution}{exo:b2:powerseries:3}
Uit $\frac{1}{1-x} = \sum x^n$, differentieer en vermenigvuldig met $x$
(\cref{thm:b2:powerseries:calculus}):
\[
\sum n x^n = \frac{x}{(1-x)^2} .
\]
Nogmaals differentiëren, opnieuw met $x$ vermenigvuldigen:
\[
\sum n^2 x^n = x\,\frac{\dd}{\dd x}\Bigl(\frac{x}{(1-x)^2}\Bigr)
= \frac{x(1 + x)}{(1-x)^3} .
\]
Derde som: het is het oneven deel van $-\ln(1 - x)$:
\[
\sum_{n\geq0} \frac{x^{2n+1}}{2n+1}
= \frac{-\ln(1-x) + \ln(1+x)}{2}
= \frac12 \ln\frac{1+x}{1-x}
= \operatorname{artanh} x .
\]
\end{solution}

\begin{solution}{exo:b2:powerseries:4}
Partiële breuken: $\frac{1}{(1-x)(2-x)} = \frac{1}{1-x} -
\frac{1}{2 - x} = \sum x^n - \frac12\sum \bigl(\frac x2\bigr)^n$:
\[
\frac{1}{(1-x)(2-x)} = \sum_{n\geq0} \Bigl(1 -
\frac{1}{2^{n+1}}\Bigr)x^n,
\qquad R = 1 .
\]

$\ln(1 + x + x^2) = \ln\frac{1 - x^3}{1 - x} = \ln(1 - x^3) -
\ln(1 - x) = \sum_{n\geq1}\frac{x^n}{n} -
\sum_{m\geq1}\frac{x^{3m}}{m}$: de coëfficiënt van $x^n$ is
$\frac1n$ als $3 \nmid n$, en $\frac1n - \frac{3}{n} = -\frac2n$ als
$3 \mid n$. Straal $1$ (de dichtstbijzijnde obstructie: de reeks voor
$\ln(1-x^3)$).
\end{solution}

\begin{solution}{exo:b2:powerseries:5}
Cauchyproduct van $\sum_{m \geq 0} x^m$ (coëfficiënten $1$) en
$\sum_{k\geq1} \frac{x^k}{k}$ (coëfficiënten $\frac1k$, $k \geq
1$), beide absoluut convergent voor $\abs x < 1$: de coëfficiënt
van $x^n$ in het product is $\sum_{k=1}^{n} \frac1k \cdot 1 = H_n$.
Dus
\[
\Bigl(\sum x^m\Bigr)\Bigl(\sum \frac{x^k}{k}\Bigr)
= \frac{1}{1-x}\cdot\bigl(-\ln(1-x)\bigr)
= \sum_{n\geq1} H_n x^n .
\]
\end{solution}

\begin{solution}{exo:b2:powerseries:6}
Vermenigvuldig de recurrentie met $x^{n+1}$ en sommeer ($\abs x <
\frac12$):
\[
U(x) - 1 = 2x\,U(x) + \sum_{n\geq0} n\,x^{n+1}
= 2x\,U(x) + \frac{x^2}{(1-x)^2} ,
\]
via \cref{exo:b2:powerseries:3}. Dus
\[
U(x) = \frac{1}{1 - 2x}\Bigl(1 + \frac{x^2}{(1-x)^2}\Bigr)
= \frac{1 - 2x + 2x^2}{(1-2x)(1-x)^2} .
\]
Partiële breuken (cover-up in $x = \frac12$ geeft coëfficiënt $2$;
in de dubbele pool $x = 1$, coëfficiënt $-1$; de middelste
coëfficiënt verdwijnt door evaluatie in $x = 0$):
\[
U(x) = \frac{2}{1-2x} - \frac{1}{(1 - x)^2} .
\]
Beide ontwikkelen:
\[
u_n = 2\cdot 2^n - (n + 1) = 2^{n+1} - n - 1 .
\]
(Controle: $u_0 = 1$, $u_1 = 2u_0 + 0 = 2 = 4 - 2$.)
\end{solution}

\begin{solution}{exo:b2:powerseries:7}
Inductie: $f'(x) = \frac{2}{x^3}\eu^{-1/x^2}$, en als $f^{(n)}(x)
= P_n(\tfrac1x)\eu^{-1/x^2}$ dan
\[
f^{(n+1)}(x) =
\Bigl(-\frac{1}{x^2}\,P_n'\Bigl(\frac1x\Bigr) +
\frac{2}{x^3}\,P_n\Bigl(\frac1x\Bigr)\Bigr)\eu^{-1/x^2} :
\]
opnieuw van de gestelde vorm. In $0$: differentiequotiënten
$\frac{f^{(n)}(h)}{h} = \frac1h P_n(\frac1h)\eu^{-1/h^2} \to 0$
als $h \to 0$, want $Q(u)\,\eu^{-u^2} \to 0$ als $u \to \pm\infty$
voor elke veelterm $Q$ (exponentieel verslaat machten): door
inductie bestaan alle $f^{(n)}(0)$ en verdwijnen, en elke
$f^{(n)}$ is continu in $0$ door dezelfde limiet. Dus $f \in
C^\infty$ met nule Taylorreeks in $0$; de Taylorreeks sommeert
naar $0 \neq f$: niet \omterm{def:b2:powerseries:analytic}{analytisch} in $0$.
\end{solution}

\begin{solution}{exo:b2:powerseries:8}
Binomiaalreeks: $\sqrt{1-4x} = \sum_{k\geq0}
\binom{1/2}{k}(-4x)^k$. Voor $k = n + 1 \geq 1$:
\begin{align*}
\binom{1/2}{n+1}(-4)^{n+1}
&= \frac{\frac12\bigl(\frac12 - 1\bigr)\cdots\bigl(\frac12 -
n\bigr)}{(n+1)!}\,(-4)^{n+1}\\
&= \frac{(-1)^n\,1\cdot3\cdots(2n-1)}{2^{n+1}(n+1)!}\,(-4)^{n+1}
= -\frac{2}{n+1}\cdot\frac{(2n)!}{n!\,n!} ,
\end{align*}
via $1\cdot3\cdots(2n-1) = \frac{(2n)!}{2^n n!}$. Dus
\[
C(x) = \frac{1 - \sqrt{1-4x}}{2x}
= \frac{1}{2x}\sum_{n\geq0}\frac{2}{n+1}\binom{2n}{n}x^{n+1}
= \sum_{n\geq0} \frac{1}{n+1}\binom{2n}{n}\,x^n :
\]
$C_n = \frac{1}{n+1}\binom{2n}{n}$. Straal: $\frac14$ (de
binomiaalreeks in $4x$). Asymptotiek via
\cref{ex:b2:comparison:centralbinomial}:
\[
C_n \sim \frac{4^n}{\sqrt{\pi}\; n^{3/2}} .
\]
\end{solution}

\begin{solution}{exo:b2:powerseries:9}
Met $A_n = \sum_{k \leq n} a_k \to A$: Abelsommatie geeft, voor
$0 \leq x < 1$,
\[
\sum_{n=0}^{\infty} a_n x^n = (1 - x)\sum_{n=0}^{\infty} A_n x^n
\]
(beide zijden convergeren: $(A_n)$ is begrensd; en de identiteit
volgt uit $a_n = A_n - A_{n-1}$ en herindexering). Aangezien $(1 -
x)\sum x^n = 1$:
\[
\sum_n a_nx^n - A = (1-x)\sum_{n} (A_n - A)x^n .
\]
Gegeven $\varepsilon$, kies $N$ met $\abs{A_n - A} \leq
\varepsilon$ voor $n > N$; dan
\[
\Bigl|\sum a_nx^n - A\Bigr|
\leq (1-x)\sum_{n \leq N}\abs{A_n - A} + \varepsilon(1 -
x)\sum_{n > N}x^n
\leq (1-x)\,C_N + \varepsilon ,
\]
en $x \to 1^-$: limsup $\leq \varepsilon$ voor alle
$\varepsilon$. Dus de radiale limiet is $A$.

Toepassingen: $\sum \frac{(-1)^{n-1}}{n}$ convergeert
(alternerend), en voor $x < 1$ sommeert haar machtreeks naar
$\ln(1 + x)$: door Abel is de som $\ln 2$. Evenzo geeft
$\sum\frac{(-1)^n}{2n+1}x^{2n+1} = \arctan x$ de waarde
$\frac\pi4$ in $x = 1$ --- de integraalbewijzen van Jaar~1, nu
structureel.
\end{solution}

\begin{solution}{exo:b2:powerseries:10}
Zowel $\sum\frac{x^n}{n}$ als $\sum\frac{x^n}{n+1}$ hebben straal
$1$, en $\frac{1}{n(n+1)} = \frac1n - \frac1{n+1}$, dus voor $0
< \abs x < 1$:
\[
\sum_{n\geq1}\frac{x^n}{n(n+1)}
= -\ln(1-x) - \frac1x\sum_{n\geq1}\frac{x^{n+1}}{n+1}
= -\ln(1-x) - \frac{-\ln(1-x) - x}{x}
= 1 + \frac{1-x}{x}\ln(1-x) .
\]
Straal $1$; en $\norm{x^n/(n(n+1))}_{\infty,\intcc{-1}1} =
\frac{1}{n(n+1)}$ is \omterm{def:b2:series:summable}{sommeerbaar}: \omterm{def:b2:funcseq:series}{normale convergentie} op
$\intcc{-1}{1}$, dus de som is daar continu. Als $x \to
1^-$, $(1-x)\ln(1-x) \to 0$ en de gesloten vorm gaat naar $1$ ---
in overeenstemming met de telescoperende waarde
$\sum\frac{1}{n(n+1)} = \lim_N\bigl(1 - \frac{1}{N+1}\bigr) = 1$
in $x = 1$.
\end{solution}

\begin{solution}{exo:b2:powerseries:11}
Sorteren van de $n!$ permutaties naar hun vaste-puntenverzameling:
de $k$ vaste punten kiezen ($\binom nk$ manieren) en de andere
$n - k$ objecten derangeren geeft $n! = \sum_{k=0}^n\binom nk
D_{n-k}$. Delen door $n!$:
\[
1 = \sum_{k=0}^{n}\frac{1}{k!}\cdot\frac{D_{n-k}}{(n-k)!} ,
\]
wat precies zegt dat het Cauchyproduct van $\eu^x =
\sum\frac{x^k}{k!}$ en $D(x) = \sum D_n\frac{x^n}{n!}$ gelijk is
aan $\sum x^n = \frac{1}{1-x}$. Beide factoren convergeren
absoluut voor $\abs x < 1$ ($D_n \leq n!$, dus $D$ wordt
overheerst door de meetkundige reeks): de productidentiteit is
legitiem (\cref{prop:b2:powerseries:operations}), en
\[
D(x) = \frac{\eu^{-x}}{1-x} .
\]
Cauchyproduct van $\eu^{-x} = \sum\frac{(-1)^kx^k}{k!}$ en
$\sum x^m$: de coëfficiënt van $x^n$ is
$\sum_{k=0}^{n}\frac{(-1)^k}{k!}$, en door uniciteit van
machtreekscoëfficiënten (\cref{thm:b2:powerseries:calculus}):
\[
\frac{D_n}{n!} = \sum_{k=0}^{n}\frac{(-1)^k}{k!}
\xrightarrow[n\to\infty]{} \eu^{-1} :
\]
ongeveer $37\%$ van alle permutaties zijn derangements, ongeacht
$n$.
\end{solution}

\begin{solution}{exo:b2:powerseries:12}
Binomiaalreeks met $\alpha = -\frac12$ in $-4x$:
\[
\binom{-1/2}{n}(-4)^n
= \frac{\bigl(-\frac12\bigr)\bigl(-\frac32\bigr)\cdots
\bigl(-\frac{2n-1}2\bigr)}{n!}(-4)^n
= \frac{1\cdot3\cdots(2n-1)}{2^n\,n!}\,4^n
= \frac{(2n)!}{2^n n!}\cdot\frac{2^n}{n!}
= \binom{2n}{n},
\]
via $1\cdot3\cdots(2n-1) = \frac{(2n)!}{2^nn!}$. Dus
$(1-4x)^{-1/2} = \sum\binom{2n}nx^n$ voor $\abs{4x} < 1$.
Kwadrateren (Cauchyproduct, legitiem door absolute convergentie)
en vergelijken met $\frac{1}{1-4x} = \sum 4^nx^n$: de
coëfficiënt van $x^n$ in het kwadraat is
$\sum_{k=0}^n\binom{2k}k\binom{2n-2k}{n-k}$, en uniciteit van
coëfficiënten geeft
\[
\sum_{k=0}^{n}\binom{2k}{k}\binom{2n-2k}{n-k} = 4^n .
\]
\end{solution}

\begin{solution}{pb:b2:powerseries:1}
\textbf{1.} Met $a_n = r_n - r_{n+1}$, partiële sommatie:
\[
\sum_{n=N}^{M} a_nx^n
= r_Nx^N + \sum_{n=N+1}^{M} r_n\bigl(x^n - x^{n-1}\bigr)
- r_{M+1}x^M .
\]
Voor $0 \leq x \leq 1$ zijn de toenames $x^{n-1} - x^n$
niet-negatief en telescoperen tot $x^N - x^M$; met $s =
\sup_{n\geq N}\abs{r_n}$:
\[
\Bigl|\sum_{n=N}^{M}a_nx^n\Bigr|
\leq s\bigl(x^N + (x^N - x^M) + x^M\bigr) = 2s\,x^N \leq 2s .
\]

\textbf{2.} Aangezien $r_n \to 0$, $\sup_{n\geq N}\abs{r_n} \to
0$: vraag~1 is precies het uniforme Cauchy-criterium op
$\intcc{0}{1}$, dus $\sum a_nx^n$ convergeert daar uniform en
haar som is continu (\cref{thm:b2:funcseq:seriestransfer}). De
waarde in $1$ zijnde $\sum a_n$, is continuïteit in $1$ de
radiale limiet van \cref{exo:b2:powerseries:9}.

\textbf{3.} Voor $\abs x < 1$ convergeren alle drie machtreeksen
absoluut en $\bigl(\sum a_nx^n\bigr)\bigl(\sum b_nx^n\bigr)
= \sum c_nx^n$ (\cref{prop:b2:powerseries:operations}). Door
vraag~2 zijn elke factor en de productzijde continu op
$\intcc{0}{1}$ (hun coëfficiëntenreeksen convergeren per
hypothese); $x \to 1^-$ in de identiteit: $AB = C$.

\textbf{4.} Hier
\[
\abs{c_n} = \sum_{k=0}^{n}
\frac{1}{\sqrt{(k+1)(n-k+1)}}
\geq \sum_{k=0}^{n}\frac{2}{n+2}
= \frac{2(n+1)}{n+2} \geq 1,
\]
via AM--GM: $\sqrt{(k+1)(n-k+1)} \leq \frac{(k+1) +
(n-k+1)}{2} = \frac{n+2}{2}$. De algemene term van $\sum c_n$
gaat niet naar $0$: het Cauchyproduct divergeert, hoewel beide
factoren convergeren (alternerende reeks).

\textbf{5.} \cref{exo:b2:powerseries:5} geeft
$\frac{-\ln(1-x)}{1-x} = \sum H_nx^n$ ($\abs x < 1$).
Term-voor-term primitieven
(\cref{thm:b2:powerseries:calculus}~(2)), beide zijden
verdwijnend in $0$:
\[
\frac{\bigl(\ln(1-x)\bigr)^2}{2}
= \sum_{n\geq1}\frac{H_n}{n+1}\,x^{n+1} .
\]
Daling: $(n+2)H_n \geq (n+1)H_{n+1}$ komt neer op $H_n \geq
1$, waar voor $n \geq 1$; en $\frac{H_n}{n+1} \sim \frac{\ln
n}{n} \to 0$: in $x = -1$ convergeert de reeks door de
alternerende test. Substitutie $x \mapsto -x$ en toepassen van
vraag~2 in $x = 1$:
\[
\frac{(\ln 2)^2}{2}
= \sum_{n\geq1}\frac{H_n}{n+1}(-1)^{n+1},
\]
de aangekondigde waarde.

\textbf{6.} $f(x) = \sum(-1)^nx^n = \frac{1}{1+x} \to \frac12$
als $x \to 1^-$: \omterm{thm:b2:series:abel}{Abel-sommeerbaar} tot $\frac12$. Maar de
partiële sommen zijn $1, 0, 1, 0, \dots$: divergent.

\textbf{7.} Gegeven $\varepsilon > 0$ kies $m$ met $\abs{u_n}
\leq \varepsilon$ voor $n > m$; voor $N \geq m$:
\[
\Bigl|\frac{u_1 + \dots + u_N}{N}\Bigr|
\leq \frac{\abs{u_1} + \dots + \abs{u_m}}{N} +
\varepsilon\,\frac{N - m}{N}
\leq \frac{C_m}{N} + \varepsilon,
\]
dus $\limsup \leq \varepsilon$ voor elke $\varepsilon$: de
middelen gaan naar $0$.

\textbf{8.} Voor $0 \leq x \leq 1$: $1 - x^n = (1-x)(1 + x +
\dots + x^{n-1}) \leq n(1-x)$, dus
\[
\Bigl|\sum_{n=0}^N a_n(1 - x_N^n)\Bigr|
\leq (1 - x_N)\sum_{n=1}^N n\abs{a_n}
= \frac1N\sum_{n=1}^{N}n\abs{a_n} .
\]
Voor $n > N$: $\abs{a_n} = \frac{n\abs{a_n}}{n} \leq
\frac{1}{N}\sup_{m>N}m\abs{a_m}$, en $\sum_{n>N}x_N^n \leq
\frac{1}{1 - x_N} = N$:
\[
\Bigl|\sum_{n>N}a_nx_N^n\Bigr|
\leq \frac{\sup_{m>N}m\abs{a_m}}{N}\cdot N
= \sup_{m>N}\,m\abs{a_m} .
\]

\textbf{9.} Ontbind
\[
A_N - L = \sum_{n=0}^{N}a_n\bigl(1 - x_N^n\bigr)
- \sum_{n>N}a_nx_N^n + \bigl(f(x_N) - L\bigr) .
\]
De eerste term gaat naar $0$ door vraag~7 (de middelen van
$n\abs{a_n} \to 0$), de tweede door vraag~8 (het sup gaat naar
$0$), de derde omdat $x_N \to 1^-$ en $f(x) \to L$. Dus $A_N
\to L$: de stelling van Tauber.

\textbf{10.} Voor $x \in \intco{0}{1}$ en elke $N$:
$\sum_{n\leq N}a_nx^n \leq f(x) \leq M$ (niet-negatieve
termen). Laat $x \to 1^-$ in de eindige som: $\sum_{n\leq
N}a_n \leq M$. De partiële sommen zijn stijgend en begrensd:
$\sum a_n$ convergeert, en dan geeft vraag~2
$\lim_{x\to1^-}f(x) = \sum a_n$.

\textbf{11.} $\sigma_N - L$ is het gemiddelde van de $N$ getallen
$A_n - L$ ($0 \leq n < N$), die naar $0$ gaan: vraag~7.

\textbf{12.} $A_n = 1$ voor even $n$, $0$ voor oneven: $A_0 +
\dots + A_{N-1} = \lceil N/2\rceil$, dus $\sigma_N =
\frac{\lceil N/2\rceil}{N} \to \frac12$, de Abel-waarde van
vraag~6.

\textbf{13.} Onder $\sigma_N \to L$ heeft men $S_n = O(n)$, dus
$A_n = S_n - S_{n-1} = O(n)$ en $a_n = O(n)$: alle reeksen
hieronder hebben straal $\geq 1$. Voor $\abs x < 1$, uit $a_n =
A_n - A_{n-1}$ en $A_nx^n \to 0$:
\[
(1-x)\sum_n A_nx^n = \sum_n A_nx^n - \sum_n A_nx^{n+1}
= \sum_n a_nx^n = f(x),
\]
en identiek $(1-x)\sum S_nx^n = \sum A_nx^n$, dus $f(x) =
(1-x)^2\sum_n S_nx^n = (1-x)^2\sum_n(n+1)\sigma_{n+1}x^n$.
Tenslotte $\sum(n+1)x^n = \frac{1}{(1-x)^2}$
(\cref{exo:b2:powerseries:3}), wat de tweede identiteit is.

\textbf{14.} Aftrekken van $L$ maal de tweede identiteit van de
eerste:
\[
f(x) - L = (1-x)^2\sum_{n\geq0}(n+1)
\bigl(\sigma_{n+1} - L\bigr)x^n .
\]
Gegeven $\varepsilon$, kies $N$ met $\abs{\sigma_{n+1} - L}
\leq \varepsilon$ voor $n \geq N$; dan
\[
\abs{f(x) - L} \leq (1-x)^2 C_N +
\varepsilon(1-x)^2\sum_{n}(n+1)x^n
= (1-x)^2C_N + \varepsilon ,
\]
en $x \to 1^-$: $\limsup \leq \varepsilon$. Dus $f(x) \to L$:
de stelling van Frobenius.

\textbf{15.} $f(x) = \sum(-1)^n(n+1)x^n = \frac{1}{(1+x)^2}$
(differentieer de meetkundige reeks in $-x$): Abel-waarde
$\frac14$. Partiële sommen: $A_{2k} = k+1$, $A_{2k+1} =
-(k+1)$ (directe inductie). Dan $S_{2m-1} = 0$ (opeenvolgende
paren cancelleren) en $S_{2m} = m + 1$, dus
\[
\sigma_{2m} = \frac{S_{2m-1}}{2m} = 0,
\qquad
\sigma_{2m+1} = \frac{m+1}{2m+1} \to \frac12 :
\]
$(\sigma_N)$ heeft twee verschillende clusterwaarden: niet
\omterm{thm:b2:series:abel}{Cesàro-sommeerbaar}. Met vragen~6 en~11--14 is de hiërarchie
convergent $\Rightarrow$ Cesàro $\Rightarrow$ Abel strikt bij
beide pijlen.

\textbf{16.} De primitievenreeks $F(x) =
\sum\frac{a_n}{n+1}x^{n+1}$ heeft dezelfde straal en $F' = f$
op $\intco{0}{1}$ (\cref{thm:b2:powerseries:calculus});
aangezien $\sum\frac{a_n}{n+1}$ convergeert, maakt Deel~I
(vraag~2) $F$ continu op $\intcc{0}{1}$. Omdat $\int_0^x f =
F(x)$ (gelijke afgeleiden, gelijke waarde $0$ in $0$),
\[
\int_0^x f \xrightarrow[x\to1^-]{} F(1)
= \sum_{n\geq0}\frac{a_n}{n+1} :
\]
de \omterm{def:b2:integration:improper}{oneigenlijke integraal} bestaat met de gestelde waarde.

\textbf{17.} $\frac{\ln(1+x)}{x} =
\sum_{n\geq1}\frac{(-1)^{n-1}}{n}x^{n-1}$ (straal $1$;
continu in $0$). De reeks van $\frac{a_m}{m+1}$ is
$\sum_{n\geq1}\frac{(-1)^{n-1}}{n^2}$, absoluut convergent:
vraag~16 geeft $\int_0^1\frac{\ln(1+x)}{x}\dd x = \eta$. In de
absoluut convergente $\sum\frac1{n^2}$, hergroepeer even en
oneven:
\[
\eta = \sum_{\text{oneven}}\frac1{n^2} -
\sum_{\text{even}}\frac1{n^2}
= \sum_{n}\frac1{n^2} - 2\sum_{k}\frac1{(2k)^2}
= \Bigl(1 - \frac12\Bigr)\sum_n\frac1{n^2}
= \frac12\sum_{n\geq1}\frac1{n^2} .
\]

\textbf{18.} Leibniz: $\frac{1}{3n+1}\downarrow0$, de reeks
convergeert. Op $\intco{0}{1}$, $\sum(-1)^nx^{3n} =
\frac{1}{1+x^3}$, en $\sum\frac{(-1)^n}{3n+1}$ convergeert:
vraag~16 geeft $\sum\frac{(-1)^n}{3n+1} =
\int_0^1\frac{\dd x}{1+x^3}$. Partiële breuken (controle:
$\frac13(x^2-x+1) + \frac{2-x}{3}(1+x) = 1$):
\[
\int_0^1\frac{\dd x}{1+x^3}
= \frac13\ln2 + \frac13\int_0^1\frac{2-x}{x^2-x+1}\dd x .
\]
Schrijvend $2 - x = -\frac12(2x-1) + \frac32$: het
$\ln(x^2-x+1)$-deel verdwijnt in beide uiteinden, en
\[
\frac32\int_0^1\frac{\dd x}{(x-\frac12)^2 + \frac34}
= \frac32\cdot\frac{2}{\sqrt3}
\Bigl[\arctan\frac{2x-1}{\sqrt3}\Bigr]_0^1
= \sqrt3\cdot\frac{\pi}{3} = \frac{\pi}{\sqrt3} .
\]
Totaal: $\frac13\bigl(\ln2 + \frac{\pi}{\sqrt3}\bigr)$.

\textbf{19.} Substitutie $t = x^2$ in de reeks van
\cref{exo:b2:powerseries:12} en term-voor-term integreren
(de primitieve van $(1-x^2)^{-1/2}$ die in $0$ verdwijnt is
$\arcsin$):
\[
\arcsin x = \sum_{n\geq0}
\frac{\binom{2n}n}{4^n(2n+1)}x^{2n+1}
\qquad(\abs x < 1) .
\]
De coëfficiënten zijn $\sim \frac{1}{2\sqrt\pi\,n^{3/2}}$
(\cref{ex:b2:comparison:centralbinomial}), \omterm{def:b2:series:summable}{sommeerbaar}: de
reeks convergeert \emph{normaal} op $\intcc{-1}{1}$, haar som
is daar continu, en valt samen met de continue $\arcsin$ op
$\intoo{-1}{1}$, dus ook in $x = 1$:
\[
\sum_{n\geq0}\frac{\binom{2n}n}{4^n(2n+1)}
= \arcsin 1 = \frac\pi2 .
\]

\textbf{20.} $C_n4^{-n} \sim \frac{1}{\sqrt\pi\,n^{3/2}}$
(\cref{exo:b2:powerseries:8}): \omterm{def:b2:funcseq:series}{normale convergentie} van $\sum
C_nx^n$ op $\intcc{0}{\frac14}$, dus haar som is daar
continu; op $\intoo{0}{\frac14}$ is zij gelijk aan
$\frac{1-\sqrt{1-4x}}{2x}$
(\cref{ex:b2:powerseries:catalan}), waarvan de limiet in
$\frac14^-$ gelijk is aan $\frac{1-0}{1/2} = 2$. Dus
$\sum_{n\geq0} C_n4^{-n} = 2$.

\textbf{21.} $f(x) = \sum_{n\geq m}a_nx^n = x^m g(x)$ met
$g(x) = \sum_{k\geq0}a_{m+k}x^k$; als $(a_nr^n)$ begrensd is
dan is $(a_{m+k}r^k)$ dat ook (deel door $r^m$): $g$ heeft
straal $\geq R$. $g$ is continu met $g(0) = a_m \neq 0$, dus
$g \neq 0$ op enige $\intcc{-\delta}{\delta}$, en $f(x) =
x^mg(x) \neq 0$ voor $0 < \abs x \leq \delta$.

\textbf{22.} $d = f - h$ is de som van een machtreeks bij $0$
die verdwijnt in de niet-nule punten $x_k \to 0$. Als enige
coëfficiënt van $d$ niet-nul was, zou vraag~21 een
gepuncteerde omgeving van $0$ vrij van nulpunten van $d$ geven
--- in tegenspraak met $d(x_k) = 0$. Dus alle coëfficiënten
van $d$ verdwijnen: $f$ en $h$ hebben gelijke coëfficiënten en
vallen samen bij $0$.

\textbf{23.} De functie $h(x) = \frac{1}{1+x^2} =
\sum(-1)^nx^{2n}$ (straal $1$) voldoet aan $h(\frac1k) =
\frac{1}{1 + 1/k^2} = \frac{k^2}{k^2+1}$. Elke analytische $f$
met dezelfde waarden valt samen met $h$ in de punten $\frac1k
\to 0$: door de identiteitsstelling (vraag~22) is $f =
\frac{1}{1+x^2}$ bij $0$ --- de unieke oplossing.

\textbf{24.} Zij $Z$ de verzameling van punten van $I$ met een
omgeving waarop $f$ identiek verdwijnt: open per definitie,
niet-leeg (het subinterval). Gesloten in $I$: als $y \in I$
limiet is van punten van $Z$, dan is $y$ een ophopingspunt van
nulpunten van $f$; ontwikkelen van $f$ in een machtreeks in $y$
(analyticiteit) en toepassen van vragen~21--22 hercentreerd in
$y$, verdwijnen alle coëfficiënten in $y$, dus $f \equiv 0$ bij
$y$: $y \in Z$. Een interval is \omterm{def:b2:metric:connected}{samenhangend}, dus $Z = I$: $f
\equiv 0$ op $I$. In het bijzonder verdwijnt een \omterm{def:b2:powerseries:analytic}{analytische
functie} op $\R$ die buiten een compacte verzameling verdwijnt
op een interval, dus overal: geen niet-nule analytische bult.
De $C^\infty$-wereld is anders: lijmen van de platte functie
van \cref{exo:b2:powerseries:7} (bijv.\ $x \mapsto
\eu^{-1/x^2}\mathbf 1_{x>0}$ en haar spiegel) produceert gladde
bulten met compacte drager.

\textbf{25.} (i) \omterm{def:b2:funcseq:series}{Normale convergentie} leeft op compacte
subschijven strikt binnen de schijf; de stelling van Abel
breidt continuïteit uit tot een randpunt, onder de enige
hypothese dat de coëfficiëntenreeks daar convergeert. (ii) De
omkering geldt onder de voorwaarde van Tauber $na_n \to 0$
(vraag~9), en Cesàro-sommeerbaarheid zit strikt tussen
convergentie en Abel-sommeerbaarheid (Frobenius,
vragen~14--15). (iii) Voor een vriend:
$\sum\frac{(-1)^n}{3n+1} = \int_0^1\frac{\dd x}{1+x^3}$ door
de meetkundige reeks tot de rand te integreren, dan partiële
breuken. (iv) Cesàromiddelden keren terug in het
Fourierhoofdstuk als de stelling van Fejér, waar middelen van
de partiële sommen het falen van \omterm{def:b2:funcseq:def}{puntsgewijze convergentie}
herstellen --- dezelfde medicijn, nieuwe patiënt.
\end{solution}
