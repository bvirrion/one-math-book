\begin{solution}{exo:b2:genfun:2}
\emph{Binomiaal:} $G(t) = (1 - p + pt)^n$, $G'(t) = np(1 - p
+ pt)^{n-1}$, $G''(t) = n(n-1)p^2(1 - p + pt)^{n-2}$, dus
\[
\E(X) = G'(1) = np,
\qquad
V(X) = G''(1) + G'(1) - G'(1)^2
= n(n-1)p^2 + np - n^2p^2 = np(1-p).
\]
\emph{Geometrisch} ($q = 1 - p$): $G(t) = \frac{pt}{1 -
qt}$, dus $G'(t) = \frac{p}{(1 - qt)^2}$ en $G''(t) =
\frac{2pq}{(1 - qt)^3}$; in $t = 1$ (met $1 - q = p$):
\[
\E(X) = \frac{p}{p^2} = \frac1p,
\qquad
V(X) = \frac{2q}{p^2} + \frac1p - \frac{1}{p^2}
= \frac{2q + p - 1}{p^2}
= \frac{q}{p^2} ,
\]
overeenkomend met \cref{exo:b2:randomvar:1} met minder werk.
\end{solution}

\begin{solution}{exo:b2:genfun:4}
Door \cref{thm:b2:genfun:compound} met $G_N(s) =
e^{\lambda(s-1)}$ en $G_X(t) = 1 - p + pt$:
\[
G_S(t) = e^{\lambda(1 - p + pt - 1)} = e^{\lambda p(t - 1)} :
\]
$S \sim \mathcal{P}(\lambda p)$. De verworpen telling $D = N
- S$ telt dezelfde items gehouden met waarschijnlijkheid $1 -
p$, dus door dezelfde berekening $D \sim
\mathcal{P}(\lambda(1 - p))$. Onafhankelijkheid, direct:
voor $j, k \in \N$,
\begin{align*}
\P(S = j,\ D = k)
&= \P(N = j + k)\,\binom{j+k}{j}p^jq^k
= e^{-\lambda}\frac{\lambda^{j+k}}{(j+k)!}\,
\frac{(j+k)!}{j!\,k!}\,p^jq^k\\
&= \Bigl(e^{-\lambda p}\frac{(\lambda p)^j}{j!}\Bigr)
\Bigl(e^{-\lambda q}\frac{(\lambda q)^k}{k!}\Bigr)
\end{align*}
met $q = 1 - p$: de gezamenlijke wet factort als
$\mathcal{P}(\lambda p) \otimes \mathcal{P}(\lambda q)$. Een
Poisson-stroom willekeurig gesplitst levert
\emph{onafhankelijke} Poisson-stromen --- een klein wonder
constant gebruikt in wachtrijtheorie.
\end{solution}

\begin{solution}{exo:b2:genfun:5}
De wachttijden tussen opeenvolgende koppen zijn
onafhankelijke geometrische $\mathcal{G}(p)$-variabelen
(geheugenloosheid: na elke kop herstart het spel), dus $T_r =
W_1 + \dots + W_r$ en multiplicativiteit
(\cref{thm:b2:genfun:product}) geeft
\[
G_{T_r}(t) = \Bigl(\frac{pt}{1 - qt}\Bigr)^{r},
\qquad
\E(T_r) = r\,\E(W_1) = \frac rp,
\qquad
V(T_r) = r\,V(W_1) = \frac{rq}{p^2}
\]
($q = 1 - p$; \omterm{def:b2:randomvar:variance}{varianties} tellen op door onafhankelijkheid).
Ontwikkeling: door de veralgemeende binomiale reeks
(\cref{ch:b2:powerseries}), $(1 - qt)^{-r} = \sum_{m\geq0}
\binom{m + r - 1}{r - 1}q^mt^m$, dus de coÃ«fficiÃ«nt van $t^n$
in $p^rt^r(1 - qt)^{-r}$ is (met $m = n - r$)
\[
\P(T_r = n) = \binom{n-1}{r-1}p^r(1-p)^{n-r},
\qquad n \geq r ,
\]
de \emph{negatief binomiale} wet --- combinatorisch: de
$r$-de kop valt op worp $n$ iff de vorige $r - 1$ koppen hun
plaatsen kiezen onder de eerste $n - 1$ worpen.
\end{solution}

\begin{solution}{exo:b2:genfun:7}
\emph{\omterm{def:b2:randomvar:expectation}{Verwachting}.} Eerst $\E(Z_n) = m^n$: door
\cref{thm:b2:genfun:compound}, $\E(Z_{n+1}) = \E(Z_n)\,m$,
en $\E(Z_0) = 1$. De familie $\bigl(Z_n(\omega)\P(\{\omega\})
\bigr)_{n, \omega}$ is niet-negatief, dus Fubini voor
families past onvoorwaardelijk toe:
\[
\E(Y) = \sum_{n=0}^{\infty}\E(Z_n)
= \sum_{n=0}^\infty m^n = \frac{1}{1 - m} < \infty
\]
(in het bijzonder is $Y$ bijna zeker eindig: consistent met
zeker uitsterven in het subkritische geval).

\emph{Functionaalvergelijking.} Ontbind de populatie naar de
kinderen van de voorouder: als de voorouder $Z_1 = k$
kinderen heeft, is de totale nakomelingschap $Y = 1 + Y_1 +
\dots + Y_k$, waar $Y_i$ de totale nakomelingschap van de
lijn van het $i$-de kind is --- en de $Y_i$ zijn
onafhankelijke kopieÃ«n van $Y$, onafhankelijk van $Z_1$
(verschillende lijnen gebruiken disjuncte, onafhankelijke
voortplantingsgebeurtenissen). Conditioneren op $Z_1$ zoals
in \cref{thm:b2:genfun:compound}:
\[
H(t) = \E\bigl(t^Y\bigr)
= t\sum_{k=0}^\infty \P(Z_1 = k)\,H(t)^k
= t\,G\bigl(H(t)\bigr),
\]
de factor $t$ die de voorouder zelf meerekent. (Voor de wet
$p_0 = 1 - p$, $p_2 = p$ van binaire vertakking kan deze
kwadratische vergelijking in $H$ expliciet opgelost en
ontwikkeld worden --- de \omterm{ex:b2:powerseries:catalan}{Catalan-getallen} van
\cref{ch:b2:powerseries} tellen de familiestambomen.)
\end{solution}

\begin{solution}{exo:b2:genfun:8}
Laat $R > 1$ de straal zijn en fixeer $t \in \intoo{1}{R}$.
Dan $\E(t^X) = G(t) < \infty$, en de ongelijkheid van Markov
(\cref{thm:b2:randomvar:markov}) toegepast op de
niet-negatieve variabele $t^X$ op niveau $t^n$:
\[
\P(X \geq n) = \P\bigl(t^X \geq t^n\bigr)
\leq \frac{G(t)}{t^n} = C\rho^n,
\qquad C = G(t),\quad \rho = \frac1t \in \intoo{0}{1}.
\]
\emph{Omgekeerd:} als $\P(X \geq n) \leq C\rho^n$, dan $p_n
\leq \P(X \geq n) \leq C\rho^n$, dus voor $\abs t <
\frac1\rho$ wordt de reeks $\sum p_n\abs t^n$ gedomineerd
door de convergente geometrische reeks $C\sum(\rho\abs
t)^n$: de straal is minstens $\frac1\rho > 1$. Straal van de
\omterm{ex:b2:powerseries:fibonacci}{genererende functie} en geometrisch verval van de staart zijn
twee gezichten van dezelfde eigenschap.
\end{solution}

\begin{solution}{exo:b2:genfun:9}
Schrijf $p_k^{(n)} = \P(X_n = k)$, $p_k = \P(X = k)$.

\emph{Geval $k = 0$.} Voor $t \in \intoo{0}{1}$ en elke wet
$(q_j)$ met $\sum_j q_j \leq 1$:
\[
\Bigl|\,q_0 - \sum_j q_jt^j\Bigr|
= \sum_{j \geq 1} q_j t^j
\leq \sum_{j\geq1}t^j = \frac{t}{1 - t} .
\]
Dus
\[
\abs{p_0^{(n)} - p_0}
\leq \frac{2t}{1 - t}
+ \abs{G_{X_n}(t) - G_X(t)} .
\]
Gegeven $\varepsilon > 0$, kies $t$ met $\frac{2t}{1-t} <
\frac\varepsilon2$, dan $n_0$ zodat de laatste term $<
\frac\varepsilon2$ is voor $n \geq n_0$: dus $p_0^{(n)} \to
p_0$.

\emph{Inductiestap.} Neem aan $p_j^{(n)} \to p_j$ voor $j <
k$. Beschouw de \emph{verschoven} functies
\[
g_n(t) = \frac{G_{X_n}(t) - p^{(n)}_0}{t}
= \sum_{j\geq0} p^{(n)}_{j+1}t^j,
\qquad
g(t) = \frac{G_X(t) - p_0}{t} ,
\]
\omterm{ex:b2:powerseries:fibonacci}{genererende functies} van de sub-waarschijnlijkheidsrijen
$(p^{(n)}_{j+1})_j$ (totale massa $\leq 1$, wat het $k =
0$-argument allemaal gebruikte). Voor vaste $t \in
\intoo{0}{1}$, $g_n(t) \to g(t)$ door de hypothese en het
geval $k = 0$. Het $k = 0$-argument toepassen op $g_n$ geeft
$p_1^{(n)} \to p_1$; de verschuiving $k$ keer itereren geeft
$p_k^{(n)} \to p_k$ voor elke $k$. (Dit is het discrete,
elementaire geval van de continuÃ¯teitsstelling van LÃ©vy,
wier algemene vorm --- voor karakteristieke functies --- een
mijlpaal van Bachelor jaar~3 is.)
\end{solution}

\begin{solution}{exo:b2:genfun:10}
Puntsgewijs is $\frac{1 + (-1)^X}{2}$ gelijk aan $1$ wanneer
$X$ even is en $0$ wanneer oneven, dus \omterm{def:b2:randomvar:expectation}{verwachtingen} nemend
(overdracht),
\[
\P(X \text{ even}) = \frac{1 + \E\bigl((-1)^X\bigr)}2 =
\frac{1 + G_X(-1)}2 .
\]
Poisson: $\frac{1 + \eu^{-2\lambda}}2 \to \frac12$ als
$\lambda$ groeit. Binomiaal: $\frac{1 + (1 - 2p)^n}2$. In
beide gevallen zegt $G_X(-1) \to 0$ dat de pariteit van $X$
een eerlijke munt wordt: de wet spreidt over vele gehelen en
vergeet haar pariteit.
\end{solution}

\begin{solution}{exo:b2:genfun:11}
$t + \dots + t^6 = t\,\frac{1 - t^6}{1 - t}$ en $1 - t^6 =
(1 - t)(1 + t)(1 + t + t^2)(1 - t + t^2)$, wat de gestelde
factorisatie geeft. Voor de eerste steen, $(1 + t)(1 + t +
t^2) = 1 + 2t + 2t^2 + t^3$, dus
$\frac{t(1+t)(1+t+t^2)}6 = \frac{t + 2t^2 + 2t^3 + t^4}6$:
vlakken $\{1, 2, 2, 3, 3, 4\}$. Voor de tweede, ontwikkelen
\[
(1 + 2t + 2t^2 + t^3)(1 - t + t^2)^2 = 1 + t^2 + t^3 + t^4 +
t^5 + t^7,
\]
dus $\frac{t(1+t)(1+t+t^2)(1-t+t^2)^2}6 = \frac{t + t^3 +
t^4 + t^5 + t^6 + t^8}6$: vlakken $\{1, 3, 4, 5, 6, 8\}$.
Het product van de twee \omterm{ex:b2:powerseries:fibonacci}{genererende functies} hergroepeert de
zes factoren tot $\bigl(\frac{t(1+t)(1+t+t^2)(1-t+t^2)}6
\bigr)^2$, het kwadraat van de functie van de
standaardsteen: het Sicherman-paar heeft precies de
standaardwet voor het totaal --- \omterm{ex:b2:powerseries:fibonacci}{genererende functies}
classificeren alle zulke hergroeperingen.
\end{solution}

\begin{solution}{exo:b2:genfun:12}
Laat $A$ en $B$ de \omterm{ex:b2:powerseries:fibonacci}{genererende functies} zijn van de
resterende duur startend vanuit ``geen huidige kop'' en

\qquad

\[

\]

$G_T(1) = 1$, het spel eindigt bijna zeker (zoals
\cref{exo:b2:proba:6} door recursie toonde).
Logaritmische differentiatie in $1$: $\E(T) = 2 -

-q - 2pq$:
\[
\E(T) = 2 + \frac{q + 2pq}{p^2} = \frac{2p^2 + q + 2pq}{p^2}
= \frac{1 + p}{p^2},
\]
wat $6$ is voor $p = \frac12$.
\end{solution}

\begin{solution}{pb:b2:genfun:1}
\textbf{1.} Voor $t \in \intoo01$ geeft de kettingregel op
$G_n = G \circ G_{n-1}$ dat $G_n'(t) =
G'\bigl(G_{n-1}(t)\bigr)G_{n-1}'(t)$. Als $t \to 1^-$, $G_{n-1}(t)
\uparrow 1$, en $G'$ is niet-dalend met linkerlimiet $m$ in
$1$, dus de eerste factor neigt naar $m$; door inductie neigt
de tweede naar $m^{n-1}$. Door
\cref{thm:b2:genfun:moments}, $\E(Z_n) = G_n'(1^-) = m^n$.

\textbf{2.} Nog eens differentiÃ«ren,
\[
G_n'' = G''(G_{n-1})\,(G_{n-1}')^2 +
G'(G_{n-1})\,G_{n-1}'',
\]
en $t \to 1^-$: $a_n = G''(1)m^{2(n-1)} + m\, a_{n-1}$ met
$a_n = G_n''(1)$, $a_1 = G''(1)$. Voor $m \neq 1$ controleert
men door inductie dat $a_n = G''(1)\,m^{n-1}\frac{m^n -
1}{m - 1}$ (de recursie voegt $G''(1)m^{2n-2}$ toe aan
$m\cdot G''(1)m^{n-2}\frac{m^{n-1}-1}{m-1}$, en $m^{n-1} +
\frac{m^{n-1}-1}{m-1} = \frac{m^n - 1}{m-1}$); voor $m =
1$, $a_n = a_{n-1} + G''(1) = n\,G''(1)$.

\textbf{3.} $V(Z_n) = a_n + m^n - m^{2n}$ en $G''(1) =
\sigma^2 + m^2 - m$. Voor $m \neq 1$ heft het stuk $(m^2 -
m)m^{n-1}\frac{m^n-1}{m-1} = m^n(m^n - 1)$ precies $m^n -
m^{2n}$ op, wat $V(Z_n) =
\sigma^2m^{n-1}\frac{m^n-1}{m-1}$ overlaat. Voor $m = 1$:
$V(Z_n) = nG''(1) = n\sigma^2$.

\textbf{4.} $Z_n$ is een niet-negatieve geheelwaardige
variabele, dus $\P(Z_n > 0) = \P(Z_n \geq 1) \leq \E(Z_n) =
m^n$ door Markov (\cref{thm:b2:randomvar:markov}). Voor $m <
1$ daalt dit geometrisch --- en \omterm{def:b2:series:summable}{sommeerbaar}, dus geeft
Borel--Cantelli zelfs dat slechts eindig veel generaties
niet-leeg zijn, wat opnieuw uitsterven is.

\textbf{5.} Cauchy--Schwarz: $\E(Z_n)^2 = \E(Z_n\mathbf
1_{Z_n>0})^2 \leq \E(Z_n^2)\,\P(Z_n > 0)$. Met vraag 3 en $m
< 1$:
\[
\E(Z_n^2) = V(Z_n) + m^{2n}
\leq \frac{\sigma^2m^{n-1}}{1-m} + m^{2n},
\]
dus, $m^{2n}$ delend door deze grens en vereenvoudigend met
$m^n$,
\[
\P(Z_n > 0) \geq \frac{m^n}{\frac{\sigma^2}{m(1-m)} + m^n}
\geq \Bigl(\frac{\sigma^2}{m(1-m)} + 1\Bigr)^{-1}m^n ,
\]
met $m^n \leq 1$ in de noemer. Met vraag 4: $\P(Z_n > 0)
\asymp m^n$.

\textbf{6.} $G(t) = q\sum_k(pt)^k = \frac{q}{1 - pt}$, en $m
= G'(1) = \frac{pq}{(1-p)^2} = \frac pq$. Subkritisch voor
$p < \frac12$, kritiek voor $p = \frac12$, superkritisch
voor $p > \frac12$.

\textbf{7.} $G(t) = t$ leest $pt^2 - t + q = 0$, met wortels
$\frac{1 \pm \abs{p - q}}{2p}$, d.w.z.\ $1$ en $\frac qp =
\frac1m$. De uitsterfkans is het kleinste vast punt in
$\intcc01$ (\cref{thm:b2:genfun:extinction}):
$q_{\mathrm{ext}} = 1$ als $m \leq 1$, en $\frac1m$ als $m
> 1$.

\textbf{8.} Voor $m \neq 1$, met $p = \frac m{m+1}$, $q =
\frac1{m+1}$: als $q_n = \frac{m^n - 1}{m^{n+1} - 1}$, dan
\[
1 - p\,q_n = \frac{(m+1)(m^{n+1} - 1) - m(m^n - 1)}
{(m+1)(m^{n+1} - 1)} = \frac{m^{n+2} - 1}{(m+1)(m^{n+1} -
1)},
\]
dus $q_{n+1} = \frac{q}{1 - pq_n} = \frac{m^{n+1} -
1}{m^{n+2} - 1}$; het basgeval $q_0 = 0$ houdt. Voor $m =
1$: $G(t) = \frac1{2 - t}$ en $q_{n+1} = \frac1{2 -
\frac{n}{n+1}} = \frac{n+1}{n+2}$, met $q_0 = 0$.

\textbf{9.} $1 - q_n = \frac{m^n(m - 1)}{m^{n+1} - 1}$. Voor
$m < 1$ neigt de noemer naar $-1$: $1 - q_n \sim (1 -
m)\,m^n$. Voor $m > 1$:
\[
q_{\mathrm{ext}} - q_n = \frac1m - \frac{m^n - 1}{m^{n+1} -
1} = \frac{m - 1}{m\,(m^{n+1} - 1)} \sim \frac{m -
1}{m^{2}}\;m^{-n} .
\]
En $G'(t) = \frac{pq}{(1 - pt)^2}$ geÃ«valueerd in $t =
\frac qp$ (waar $1 - pt = 1 - q = p$) geeft
$G'(q_{\mathrm{ext}}) = \frac qp = \frac1m$: de
waargenomen verhouding $m^{-1}$ is precies de afgeleide in
het aantrekkende vast punt.

\textbf{10.} Voor $p = \frac12$: $G''(t) = \frac{1/4}{(1 -
t/2)^3}$, dus $G''(1) = 2$ en $\sigma^2 = G''(1) + m - m^2 =
2$. De gesloten vorm geeft $1 - q_n = \frac1{n+1}$: de
overlevingskans daalt als $1/n$ --- te traag om \omterm{def:b2:series:summable}{sommeerbaar}
te zijn, in tegenstelling tot elk subkritisch tempo.

\textbf{11.} Inductie: $G_1(t) = \frac1{2-t}$ past bij de
formule voor $n = 1$, en
\[
G(G_n(t)) = \cfrac{1}{2 - \cfrac{n - (n-1)t}{n+1 - nt}}
= \frac{n + 1 - nt}{2(n+1) - 2nt - n + (n-1)t}
= \frac{n+1 - nt}{n + 2 - (n+1)t} .
\]
Dan
\[
\frac{G_n(t) - q_n}{1 - q_n}
= (n+1)\,\Bigl(\frac{n - (n-1)t}{n+1 - nt} -
\frac{n}{n+1}\Bigr)
= \frac{t}{n + 1 - nt}
= \frac{\frac{t}{n+1}}{1 - \frac{n}{n+1}t} ,
\]
de pgf van de geometrische wet $\mathcal G\bigl(\frac1{n+1}
\bigr)$ op $\N^*$ (\cref{ex:b2:genfun:classical}): gegeven
overleving, $\P(Z_n = k \mid Z_n > 0) =
\frac1{n+1}\bigl(\frac n{n+1}\bigr)^{k-1}$, met
voorwaardelijk gemiddelde $n + 1$. Het onvoorwaardelijke
gemiddelde $1 = \E(Z_n)$ is het product van een verdwijnende
overlevingskans en een lineair groeiende voorwaardelijke
grootte.

\textbf{12.} Als de lijn uitsterft in generatie $n$, dan is
$Y = Z_0 + \dots + Z_{n-1}$ eindig; als zij nooit uitsterft,
$Y \geq \sum_n 1 = \infty$. Dus is $\{Y < \infty\}$ de
uitsterfgebeurtenis en $\P(Y < \infty) = q_{\mathrm{ext}}$.

\textbf{13.} Met $G(s) = \frac{1 + s^2}2$ leest de

(\cref{ex:b2:powerseries:catalan}): $H(t) = \frac

= 2k+1) = C_k2^{-2k-1}$. Controles: $\P(Y = 1) = C_0/2 =
\frac12$ (de voorouder heeft geen kind); $\P(Y = 3) =
C_1/8 = \frac18$ (twee kinderen, beide kinderloos:
$\frac12\cdot\frac12\cdot\frac12$).

\textbf{14.} DifferentiÃ«ren van $H = tG(H)$ op $\intoo01$ en

\textbf{15.} $C_k = \frac1{k+1}\binom{2k}k \sim

\[

De staart sommeren (vergelijking met $\int_K^\infty

een zware staart met oneindig gemiddelde, die vraag 14

\textbf{16.} Beide kritieke objecten --- de

\textbf{17.} $G''(t) = \sum_{n\geq2}n(n-1)p_nt^{n-2}$ heeft

\textbf{18.} Reducerend tot een gemeenschappelijke noemer,

\textbf{19.} Uit de definitie van $h$ in $t = q_j$ en
