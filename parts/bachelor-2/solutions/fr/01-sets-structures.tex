\section*{Chapitre \ref{ch:b2:structures} --- Ensembles et structures}

\begin{solution}{exo:b2:structures:1}
\emph{Parties finies de $\N$~:} \omterm{def:b2:structures:countable}{dénombrable} --- l'ensemble des parties
de $\intint{0}{n}$ est fini, et les parties finies forment l'union
\omterm{def:b2:structures:countable}{dénombrable} sur $n$ de celles-ci
(\cref{prop:b2:structures:countablestable}~(3))~; infini car il
contient tous les singletons.

\emph{Toutes les parties de $\N$~:} non \omterm{def:b2:structures:countable}{dénombrable}, par le théorème de
Cantor (\cref{thm:b2:structures:cantor}~(1) avec $E = \N$).

\emph{$\R \setminus \Q$~:} non \omterm{def:b2:structures:countable}{dénombrable} --- sinon $\R = \Q \cup
(\R\setminus\Q)$ serait une union de deux \omterm{def:b2:structures:countable}{ensembles dénombrables},
contredisant le~\cref{thm:b2:structures:cantor}~(2).

\emph{Polynômes sur $\Q$~:} \omterm{def:b2:structures:countable}{dénombrable} --- les polynômes de degré
$\leq n$ s'injectent dans $\Q^{n+1}$ (produits finis d'\omterm{def:b2:structures:countable}{ensembles
dénombrables}), et prendre l'union sur $n$.

\emph{Suites binaires nulles à partir d'un certain rang~:} \omterm{def:b2:structures:countable}{dénombrable}
--- elles sont en bijection avec les parties finies de $\N$ (le
support).
\end{solution}

\begin{solution}{exo:b2:structures:2}
Calculer élément par élément, en appliquant d'abord le facteur de
droite. $\sigma\tau$ envoie $1 \mapsto \sigma(1) = 4$, $\;2 \mapsto
\sigma(3) = 5$, $\;3 \mapsto \sigma(7) = 7$, $\;4 \mapsto \sigma(4) =
2$, $\;5 \mapsto \sigma(5) = 3$, $\;6 \mapsto \sigma(6) = 1$, $\;7
\mapsto \sigma(2) = 6$~:
\[
\sigma\tau = (1\,4\,2\,5\,3\,7\,6),
\]
un $7$-cycle. De même $\tau\sigma$ envoie $1 \mapsto \tau(4) = 4$,
$\;2 \mapsto \tau(6) = 6$, $\;3 \mapsto \tau(5) = 5$, $\;4 \mapsto
\tau(2) = 3$, $\;5 \mapsto \tau(3) = 7$, $\;6 \mapsto \tau(1) = 1$,
$\;7 \mapsto \tau(7) = 2$~:
\[
\tau\sigma = (1\,4\,3\,5\,7\,2\,6),
\]
également un $7$-cycle (comme prévu~: $\sigma\tau$ et $\tau\sigma$ sont
conjugués, donc partagent leur type de \omterm{def:b2:structures:sn}{cycle}).

\omterm{def:b2:structures:generated}{Ordres} et signatures~: $\sigma$ a le type de \omterm{def:b2:structures:sn}{cycle} $(4,2)$~: \omterm{def:b2:structures:generated}{ordre}
$\operatorname{lcm}(4,2) = 4$, signature $(-1)^3(-1)^1 = +1$~; $\tau$
est un $3$-cycle~: \omterm{def:b2:structures:generated}{ordre} $3$, signature $+1$~; les deux produits sont
des $7$-cycles~: \omterm{def:b2:structures:generated}{ordre} $7$, signature $(-1)^6 = +1$.

$\sigma^{2026}$~: $2026 = 4 \times 506 + 2$, donc $\sigma^{2026} =
\sigma^2 = (1\,2)(4\,6)$ (élever le $4$-cycle au carré~; la
\omterm{def:b2:structures:sn}{transposition} disparaît en s'élevant au carré).
\end{solution}

\begin{solution}{exo:b2:structures:3}
$\{0,1\}^{\N} \to \mathcal{P}(\N)$~: une suite s'envoie sur son support
--- une bijection (fonctions indicatrices), aucun théorème requis.

$\{0,1\}^{\N} \to \intcc{0}{1}$~: l'application en base $3$ $(a_n)
\mapsto \sum 2a_n 3^{-n-1}$ est injective (deux suites distinctes
diffèrent d'abord au rang $N$~; les queues ne peuvent compenser un
écart de $2\cdot 3^{-N-1}$, puisque $\sum_{n > N} 2\cdot 3^{-n-1} =
3^{-N-1} < 2\cdot3^{-N-1}$).

$\intcc{0}{1} \to \{0,1\}^{\N}$~: développement binaire, en choisissant
(disons) le développement qui ne se termine pas par une suite infinie
de $1$~: injective.

Par Cantor--Bernstein (\cref{thm:b2:structures:cantorbernstein})
appliqué aux deux dernières injections, $\intcc{0}{1}$ et
$\{0,1\}^{\N}$ sont \omterm{def:b2:structures:countable}{équipotents}, donc les trois ensembles le sont.
\end{solution}

\begin{solution}{exo:b2:structures:4}
Soit $c = ab = ba$ et $d = \operatorname{ord}(c)$. D'abord $c^{mn} =
a^{mn} b^{mn} = e$ (la commutation permet de scinder la puissance),
donc $d \mid mn$. Réciproquement $c^d = e$ donne $a^d = b^{-d}$~; cet
élément appartient à $\langle a\rangle \cap \langle b\rangle$, un
sous-groupe dont l'\omterm{def:b2:structures:generated}{ordre} divise à la fois $m$ et $n$ (Lagrange dans
chaque \omterm{def:b2:structures:generated}{groupe cyclique}), donc est trivial~: $a^d = b^d = e$, donc $m
\mid d$ et $n \mid d$, et par la coprimalité $mn \mid d$. Donc $d = mn$.

Dans $\mathfrak{S}_3$~: prendre $a = (1\,2)$ (\omterm{def:b2:structures:generated}{ordre} $2$) et $b =
(1\,2\,3)$ (\omterm{def:b2:structures:generated}{ordre} $3$), d'\omterm{def:b2:structures:generated}{ordres} premiers entre eux, qui ne commutent
pas~: $ab = (2\,3)$ est d'\omterm{def:b2:structures:generated}{ordre} $2 \neq 6$ --- en effet
$\mathfrak{S}_3$ n'a pas d'élément d'\omterm{def:b2:structures:generated}{ordre} $6$. La commutation est
essentielle.
\end{solution}

\begin{solution}{exo:b2:structures:5}
Apparier tout $x \in G$ avec $x^{-1}$. Les paires $\{x, x^{-1}\}$ avec
$x \neq x^{-1}$ ont deux éléments et partitionnent leur union~; les
éléments restants sont exactement ceux avec $x = x^{-1}$, c.-à-d.\ $x^2
= e$. Comme $\abs G$ est pair et que les paires à deux éléments
couvrent un nombre pair d'éléments, l'ensemble $\{x : x^2 = e\}$ est de
cardinal pair~; il contient $e$, donc il contient au moins un autre
élément $x \neq e$ --- un élément d'\omterm{def:b2:structures:generated}{ordre} $2$.
\end{solution}

\begin{solution}{exo:b2:structures:6}
Tout élément de $A_n$ est un produit d'un nombre pair de
\omterm{def:b2:structures:sn}{transpositions} (\cref{thm:b2:structures:signature}~: décomposer en
\omterm{def:b2:structures:sn}{transpositions}~; le compte est pair puisque la signature est $+1$). Il
suffit d'écrire chaque produit de deux \omterm{def:b2:structures:sn}{transpositions} à l'aide de
$3$-cycles~:
\[
(a\,b)(a\,c) = (a\,c\,b),
\qquad
(a\,b)(c\,d) = (a\,c\,b)(a\,c\,d) \quad (a,b,c,d \text{ distincts}),
\]
(vérifier par évaluation), et $(a\,b)(a\,b) = \mathrm{id}$. Donc les
$3$-cycles engendrent $A_n$.
\end{solution}

\begin{solution}{exo:b2:structures:7}
\emph{$(\Q,+) \to (\Z,+)$~:} seul le morphisme nul. Pour tout $x$ et
tout $n \geq 1$, $f(x) = n f\bigl(\frac xn\bigr)$ est divisible par
$n$ dans $\Z$~; le seul entier divisible par tout $n$ est $0$, donc
$f(x) = 0$ pour tout $x$.

\emph{$(\Z/n\Z, +) \to (\Z/m\Z, +)$~:} un morphisme est déterminé par
$c = f(\overline 1)$, qui doit vérifier $n c \equiv 0 \pmod m$,
c.-à-d.\ $c$ est un multiple de $\frac{m}{\gcd(m,n)}$~; il y a
$\gcd(m,n)$ telles classes, et chaque choix définit bien un morphisme
(factoriser $k \mapsto kc$ par $\Z/n\Z$ grâce à la propriété
universelle).

\emph{$(\Q, +) \to (\Q_+^*, \times)$~:} seul le morphisme trivial. Si
$f(x) = y$, alors pour tout $n$, $y = f(n \cdot \frac xn) =
f(\frac xn)^n$ est une puissance $n$-ième dans $\Q_+^*$. Mais un
rationnel $y \neq 1$ ne peut être une puissance $n$-ième pour tout
$n$~: un certain nombre premier apparaît dans $y$ avec un exposant non
nul $v$, et $n \nmid v$ pour $n > \abs v$ (les exposants des puissances
$n$-ièmes sont des multiples de $n$, par unicité de la factorisation).
Donc $f \equiv 1$.
\end{solution}

\begin{solution}{exo:b2:structures:8}
$360 = 2^3 \cdot 3^2 \cdot 5$~:
$\varphi(360) = 360\bigl(1 - \tfrac12\bigr)\bigl(1 -
\tfrac13\bigr)\bigl(1 - \tfrac15\bigr) = 360 \cdot \tfrac12 \cdot
\tfrac23 \cdot \tfrac45 = 96$.

Système~: modules $8, 9, 5$ deux à deux premiers entre eux, total
$360$. De $x \equiv 3 \pmod 8$ et $x \equiv 5 \pmod 9$~: $x = 3 + 8k$
avec $3 + 8k \equiv 5 \pmod 9$, c.-à-d.\ $-k \equiv 2$, $k \equiv -2
\equiv 7 \pmod 9$~: $x \equiv 3 + 56 = 59 \pmod{72}$. Puis $59 +
72\ell \equiv 2 \pmod 5$~: $4 + 2\ell \equiv 2$, $2\ell \equiv 3 \equiv
8$, $\ell \equiv 4 \pmod 5$~: $x \equiv 59 + 288 = 347 \pmod{360}$.

Deux derniers chiffres de $3^{2026}$~: modulo $4$, $3^{2026} =
9^{1013} \equiv 1$. Modulo $25$~: $\varphi(25) = 20$ et $2026 =
20\cdot101 + 6$, donc $3^{2026} \equiv 3^6 = 729 \equiv 4 \pmod{25}$.
Résoudre $x \equiv 1 \pmod 4$, $x \equiv 4 \pmod{25}$~: $x = 4 + 25k
\equiv 1 \pmod 4$ donne $k \equiv 1 \pmod 4$~: $x \equiv 29 \pmod{100}$.
Les deux derniers chiffres sont $29$.
\end{solution}

\begin{solution}{exo:b2:structures:9}
Soit $A$ un anneau intègre fini et $a \in A$, $a \neq 0$.
L'application $x \mapsto ax$ est injective ($ax = ay \implies a(x - y)
= 0 \implies x = y$, pas de diviseurs de zéro)~; une application
injective d'un ensemble fini dans lui-même est surjective (volume de
première année, l'équivalence des tiroirs). Donc $1 = ab$ pour un
certain $b$~: tout élément non nul est inversible, $A$ est un corps.

$\Z/n\Z$~: si $n$ est premier c'est un anneau intègre ($n \mid ab
\implies n \mid a$ ou $n \mid b$, lemme d'Euclide), fini, donc un
corps~; si $n = rs$ est composé, $\overline r\,\overline s =
\overline 0$ exhibe des diviseurs de zéro.
\end{solution}

\begin{solution}{exo:b2:structures:10}
Soit $m = \max\{\operatorname{ord}(x) : x \in G\}$, atteint en $a$.

\emph{Affirmation~: tout $x \in G$ a un \omterm{def:b2:structures:generated}{ordre} divisant $m$.} Supposons
qu'un certain $x$ ait un \omterm{def:b2:structures:generated}{ordre} $q$ avec $q \nmid m$~: alors une
certaine puissance de premier $p^k$ divise $q$ mais pas $m$. Écrire $m
= p^j m'$ avec $p \nmid m'$ et $j < k$. L'élément $a^{p^j}$ est
d'\omterm{def:b2:structures:generated}{ordre} $m'$~; l'élément $x^{q/p^k}$ est d'\omterm{def:b2:structures:generated}{ordre} $p^k$~; ces \omterm{def:b2:structures:generated}{ordres}
sont premiers entre eux et les deux éléments commutent ($G \subseteq
K^*$ est abélien), donc par l'\cref{exo:b2:structures:4} leur produit
est d'\omterm{def:b2:structures:generated}{ordre} $p^k m' > p^j m' = m$~: contredisant la maximalité.

Donc tout $x \in G$ vérifie $x^m = 1$~: le polynôme $X^m - 1$ a au
moins $\abs G$ racines dans le corps $K$, d'où $\abs G \leq m$
(un polynôme non nul de degré $m$ a au plus $m$ racines, volume de
première année). Mais $m = \operatorname{ord}(a) \leq \abs G$ par
Lagrange. Donc $m = \abs G$ et $\langle a \rangle$, de cardinal $m =
\abs G$, est tout $G$~: \omterm{def:b2:structures:generated}{cyclique}.

Pour $K = \Z/p\Z$~: $(\Z/p\Z)^*$ est un sous-groupe fini de $K^*$,
donc \omterm{def:b2:structures:generated}{cyclique} (d'\omterm{def:b2:structures:generated}{ordre} $p - 1$).
\end{solution}

\begin{solution}{exo:b2:structures:11}
\emph{Non \omterm{def:b2:structures:generated}{cyclique}~:} le sous-groupe $\langle \frac pq\rangle$ est
constitué des multiples entiers de $\frac pq$, qui ont tous un
dénominateur divisant $q$ (sous forme irréductible)~; il rate donc
$\frac{1}{2q}$. Aucun générateur unique ne peut atteindre les
dénominateurs non bornés de $\Q$.

\emph{Non de type fini~:} le sous-groupe engendré par
$\frac{p_1}{q_1}, \dots, \frac{p_k}{q_k}$ est constitué de rationnels
dont les dénominateurs divisent $Q = q_1 \cdots q_k$ (les combinaisons
entières ont un dénominateur divisant $Q$)~: il rate $\frac{1}{2Q}$.

\emph{Les sous-groupes de type fini sont \omterm{def:b2:structures:generated}{cycliques}~:} avec $Q$ comme
ci-dessus, le sous-groupe $H = \langle \frac{p_1}{q_1}, \dots,
\frac{p_k}{q_k}\rangle$ est contenu dans $\frac{1}{Q}\Z$. L'application
$x \mapsto Qx$ est un isomorphisme de $\frac1Q\Z$ sur $\Z$ portant
$H$ sur un sous-groupe de $\Z$, qui est $n\Z$ pour un certain $n$
(volume de première année)~: donc $H = \frac{n}{Q}\Z$ est \omterm{def:b2:structures:generated}{cyclique},
engendré par $\frac nQ$.
\end{solution}

\begin{solution}{exo:b2:structures:12}
\emph{Une partie \omterm{def:b2:structures:countable}{dénombrable}.} Soit $E$ infini. Construire $a_0,
a_1, a_2, \dots$ par récurrence~: $E$ est non vide, choisir $a_0 \in
E$~; si $a_0, \dots, a_n$ sont choisis, $E \setminus \{a_0, \dots,
a_n\}$ est non vide ($E$ n'est pas fini), y choisir $a_{n+1}$. Les
$a_n$ sont deux à deux distincts par construction, donc $A = \{a_n : n
\in \N\}$ est une partie \omterm{def:b2:structures:countable}{dénombrable} de $E$.

\emph{Infini $\implies$ équipotent à une partie propre.} Définir
$f \colon E \to E \setminus \{a_0\}$ par $f(a_n) = a_{n+1}$ et
$f(x) = x$ pour $x \notin A$. Elle est injective (les deux morceaux
sont injectifs et d'images disjointes) et surjective sur $E \setminus
\{a_0\}$~: tout $a_{n+1}$ est atteint, tout $x \notin A$ est atteint.
Donc $E$ est équipotent à la partie propre $E \setminus \{a_0\}$.

\emph{Réciproque.} Si $E$ est fini et $g \colon E \to F$ est une
bijection sur $F \subseteq E$ avec $F \neq E$, alors $g$ est une
injection de $E$ dans lui-même qui n'est pas surjective,
contredisant le principe des tiroirs (volume de première année~: une
auto-application injective d'un ensemble fini est bijective). Donc un
ensemble équipotent à une partie propre est infini.
\end{solution}

\begin{solution}{pb:b2:structures:1}
\textbf{1.} Une configuration attribue à chacune des $16$ cases l'un
des $16$ contenus (tuiles $1$--$15$ ou la case vide $b = 16$),
chacun exactement une fois~: précisément une bijection $\intint1{16}
\to \intint1{16}$, un élément de $\mathfrak{S}_{16}$~; il y en a $16!
= 20\,922\,789\,888\,000$. Un coup légal fait glisser une tuile
adjacente à la case vide, donc le nombre de coups est le nombre de
voisines de la case de la case vide~: $2$ pour les quatre cases de
coin, $3$ pour les huit cases de bord, $4$ pour les quatre cases
intérieures.

\textbf{2.} Après le glissement, la case $p$ porte l'ancien contenu
de $c$ et la case $c$ porte la case vide~; toutes les autres cases
sont inchangées~: $\sigma'(p) = \sigma(c)$, $\sigma'(c) = \sigma(p) =
16$, $\sigma' = \sigma$ ailleurs. C'est exactement $\sigma' = \sigma
\circ (p\ c)$. Comme $\varepsilon$ est un morphisme et
$\varepsilon\bigl((p\ c)\bigr) = -1$~:
$\varepsilon(\sigma') = -\varepsilon(\sigma)$.

\textbf{3.} Des cases adjacentes diffèrent d'un pas dans exactement
l'une des deux coordonnées, donc $i + j$ change de parité~: $\chi$
prend des valeurs opposées sur des cases adjacentes. Un coup
transfère la case vide de $p$ à la case adjacente $c$, changeant
$\chi(\text{case de la case vide})$ de signe. Le long d'un chemin
fermé de la case vide, $\chi$ change de signe une fois par coup et
revient à sa valeur initiale~: le nombre de coups est pair.

\textbf{4.} Par les questions 2 et 3, un coup change de signe les
deux facteurs de $I(\sigma) = \varepsilon(\sigma)\chi(\sigma^{-1}(16))$~;
leur produit est inchangé. Pour la configuration résolue~:
$\varepsilon(\mathrm{id}) = +1$ et la case vide est à la case $16$,
ligne $4$, colonne $4$~: $\chi(16) = (-1)^{8} = +1$, donc
$I(\mathrm{id}) = +1$.

\textbf{5.} $\sigma_L$ est la \omterm{def:b2:structures:sn}{transposition} $(14\ 15)$ des cases~:
$\varepsilon(\sigma_L) = -1$~; sa case vide est au domicile, $\chi(16)
= +1$~: $I(\sigma_L) = -1 \neq +1 = I(\mathrm{id})$. Comme $I$ est
préservé par tout coup, aucune suite de coups ne relie $\sigma_L$
et $\mathrm{id}$. Le prix était structurellement à l'abri.

\textbf{6.} Fixer une case $p$ et deux autres cases $c \neq d$
distinctes de $p$, et poser $\tau_0 = (c\ d)$. Sur l'ensemble des
configurations à case vide en $p$, l'application $\sigma \mapsto
\sigma \circ \tau_0$ est une involution (elle préserve $\sigma(p) =
16$ puisque $\tau_0$ fixe $p$) et change $\varepsilon$ de signe, donc
change $I$ de signe~: elle apparie les configurations avec $I = +1$
bijectivement avec celles avec $I = -1$. Donc chacune des $16$
positions de la case vide contribue $15!/2$ configurations avec $I =
+1$, et
\[
\abs{\{I = +1\}} = 16 \cdot \frac{15!}{2} = \frac{16!}{2}.
\]

\textbf{7.} Le coup faisant glisser la tuile de $c$ dans $p$ est
annulé en faisant glisser cette même tuile (maintenant en $p$) de
retour dans $c$~: composer deux fois avec $(p\ c)$ est l'identité.
D'où~: réflexivité (suite vide), symétrie (renverser la suite, en
annulant chaque coup), transitivité (concaténer)~: une relation
d'équivalence. Tout $\sigma \in R$ a $I(\sigma) = I(\mathrm{id}) = +1$
par la question 4, donc $R \subseteq \{I = +1\}$~; et $\sigma_L \notin
R$ donne une seconde classe.

\textbf{8.} Si $\sigma(16) = 16$, alors $\sigma$ permute les cases
$1, \dots, 15$~; appelons $\rho$ cette restriction. Ajouter un point
fixe ne change ni le type de \omterm{def:b2:structures:sn}{cycle} ni la signature (décomposer $\rho$
en \omterm{def:b2:structures:sn}{transpositions}~; le même produit convient dans
$\mathfrak{S}_{16}$), donc $\varepsilon(\sigma) = \varepsilon(\rho)$,
et $\chi(16) = +1$ donne $I(\sigma) = \varepsilon(\rho)$. Toute
configuration peut être amenée à une configuration à case vide au
domicile~: la grille est connexe, donc on promène la case vide le
long d'un chemin de cases adjacentes jusqu'à la case $16$ (chaque pas
est un coup légal). Supposons maintenant que tout $\rho \in
\mathfrak{S}_{15}$ pair est réalisé par un programme. Étant donné
$\sigma$ avec $I(\sigma) = +1$~: promener la case vide au domicile
pour atteindre $\widetilde\sigma$ (équivalent à $\sigma$), avec
$I(\widetilde\sigma) = +1$, c.-à-d.\ sa restriction $\rho$ est
paire~; le programme réalisant $\rho$ amène $\widetilde\sigma$ à
$\widetilde\sigma \circ \rho^{-1} = \mathrm{id}$ (voir la question
9). Par transitivité $\sigma \in R$, d'où $\{I = +1\} \subseteq R$ et
l'égalité.

\textbf{9.} \emph{Coup isolé~:} le contenu de $c$ finit en $p$ et la
case vide en $c$~: l'effet est $\pi = (p\ c)$, et en effet $\sigma' =
\sigma \circ (p\ c) = \sigma \circ \pi^{-1}$. \emph{Récurrence~:} si
une suite a pour effet $\pi_1$ et amène $\sigma$ à $\sigma \circ
\pi_1^{-1}$, la faire suivre d'un coup d'effet $\pi_2 = (p'\ c')$
donne $(\sigma \circ \pi_1^{-1}) \circ \pi_2^{-1} = \sigma \circ
(\pi_2\pi_1)^{-1}$, et les contenus se déplacent par $\pi_2 \circ
\pi_1$ (d'abord $\pi_1$, puis $\pi_2$). Donc les effets se composent,
et un programme exécuté depuis $\sigma$ aboutit à $\sigma \circ
\pi^{-1}$. \emph{Sous-groupe~:} le programme vide a pour effet
$\mathrm{id}$~; la concaténation donne les produits~; renverser un
programme (question 7) donne les inverses. L'effet d'un programme fixe
la case $16$ (la case vide part et revient au domicile), donc $H \leq
\mathfrak{S}_{15}$. \emph{Parité~:} un programme de $k$ coups a $k$
pair (question 3), et $\varepsilon(\sigma \circ \pi^{-1}) =
(-1)^k\varepsilon(\sigma)$ force $\varepsilon(\pi) = +1$~: $H
\subseteq A_{15}$.

\textbf{10.} Suivre les quatre glissements depuis la case vide en
$16$~: le coup $16 \to 12$ envoie le contenu de $12$ en $16$~; le
coup $12 \to 11$ envoie le contenu de $11$ en $12$~; le coup $11 \to
15$ envoie le contenu de $15$ en $11$~; le coup $15 \to 16$ envoie le
contenu garé en $16$ (venant à l'origine de $12$) en $15$. Bilan~: $11
\mapsto 12$, $12 \mapsto 15$, $15 \mapsto 11$, case vide au domicile~:
l'effet est $(11\ 12\ 15)$. Le tour inverse l'annule~: effet $(11\
12\ 15)^{-1} = (11\ 15\ 12)$. Les deux sont des effets de programmes,
donc dans $H$.

\textbf{11.} Adjacence des cases consécutives~: dans chaque paire
listée les cases diffèrent de $1$ dans la même ligne ($16{-}15$,
$15{-}14$, $14{-}13$~; $1{-}2$, $2{-}3$, $3{-}4$~; $8{-}7$,
$7{-}6$~; $10{-}11$, $11{-}12$) ou de $4$ dans une colonne
($13{-}9$, $9{-}5$, $5{-}1$~; $4{-}8$~; $6{-}10$~; $12{-}16$)~: un
chemin fermé passant par les $16$ cases, de longueur $16$. Effet~:
comme à la question 10, en écrivant les cases visitées $c_0 = 16, c_1
= 15, \dots, c_{15} = 12$~: le contenu de $c_i$ se déplace en
$c_{i-1}$ pour $i = 2, \dots, 15$, et le contenu de $c_1$, garé en
$16$ après le premier coup, est amené en $c_{15}$ par le dernier
coup. Donc l'effet envoie $15 \mapsto 12$, et $14 \mapsto 15$, $13
\mapsto 14$, $9 \mapsto 13$, $5 \mapsto 9$, $1 \mapsto 5$, $2 \mapsto
1$, $3 \mapsto 2$, $4 \mapsto 3$, $8 \mapsto 4$, $7 \mapsto 8$, $6
\mapsto 7$, $10 \mapsto 6$, $11 \mapsto 10$, $12 \mapsto 11$~:
exactement le $15$-cycle $\zeta$. Son \omterm{def:b2:structures:generated}{ordre} \omterm{def:b2:structures:generated}{cyclique} commence par
$x_0 = 15$, $x_1 = 12$, $x_2 = 11$, et $(x_0\ x_1\ x_2) = (15\ 12\
11)$ envoie $15 \mapsto 12 \mapsto 11 \mapsto 15$ --- ce qui est
précisément $(11\ 15\ 12)$, le tour élémentaire inverse.

\textbf{12.} Soit $\gamma = (a\ b\ c)$ et $x \in \intint1n$. Si $x =
g(a)$~: $g\gamma g^{-1}(x) = g(\gamma(a)) = g(b)$~; de même $g(b)
\mapsto g(c)$ et $g(c) \mapsto g(a)$. Si $x \notin \{g(a), g(b),
g(c)\}$, alors $g^{-1}(x) \notin \{a,b,c\}$ est fixé par $\gamma$,
donc $x$ est fixé. Donc $g\gamma g^{-1} = (g(a)\ g(b)\ g(c))$. Et pour
$g, h \in H$, $ghg^{-1} \in H$ par les axiomes de sous-groupe.

\textbf{13.} $\zeta \in H$ (question 11) et $s_0 = (x_0\ x_1\ x_2)
\in H$ (questions 10--11). Comme $\zeta(x_i) = x_{i+1}$ (indices
modulo $15$), la question 12 donne
\[
\zeta^{t}\,s_0\,\zeta^{-t}
= \bigl(\zeta^t(x_0)\ \zeta^t(x_1)\ \zeta^t(x_2)\bigr)
= (x_t\ x_{t+1}\ x_{t+2}) = s_t \in H
\qquad (t = 0, 1, \dots, 14).
\]

\textbf{14.} Quitte à inverser, supposer $s = (a\ b\ c)$ et $t =
(b\ c\ d)$ (un $3$-cycle sur $\{a,b,c\}$ est $(a\ b\ c)$ ou son
inverse~; de même sur $\{b,c,d\}$~; remplacer un générateur par son
inverse laisse $\langle s, t\rangle$ inchangé). Alors, en appliquant
$t$ d'abord,
\[
st \colon a \mapsto b,\quad b \mapsto a,\quad c \mapsto d,\quad
d \mapsto c, \qquad\text{c.-à-d.}\quad st = (a\ b)(c\ d),
\]
une double \omterm{def:b2:structures:sn}{transposition}. Le sous-groupe $G = \langle s, t\rangle$
est constitué de permutations paires des quatre lettres, donc $G \leq
A_4$ et $\abs G \mid 12$~; il contient un élément d'\omterm{def:b2:structures:generated}{ordre} $3$
et un d'\omterm{def:b2:structures:generated}{ordre} $2$, donc $6 \mid \abs G$ (Lagrange,
\cref{thm:b2:structures:lagrange}, appliqué aux deux sous-groupes
\omterm{def:b2:structures:generated}{cycliques}). Si $A_4$ avait un sous-groupe $K$ d'\omterm{def:b2:structures:generated}{ordre} $6$, il serait
d'indice $2$, et alors $g^2 \in K$ pour tout $g \in A_4$~: pour
$g \in K$ c'est clair~; pour $g \notin K$ les seules classes sont
$K$ et $gK$, donc la classe $g^2K$ est $K$ ou $gK$, et $g^2K =
gK$ forcerait $g \in K$. Donc tout carré est dans $K$. Mais tout
$3$-cycle $\gamma$ est un carré, $\gamma = (\gamma^2)^2$, et
$A_4$ contient huit $3$-cycles~: $8 > 6$, contradiction. Donc
$\abs G = 12$~: $G = A_4$.

\textbf{15.} Prolonger $u \mapsto a$, $v \mapsto b$ en une bijection
$g_0$ de $X$ (envoyer les $k - 2$ lettres restantes bijectivement
n'importe où sur le complémentaire de $\{a, b\}$). Si $g_0$ est
impaire, choisir deux lettres distinctes $s_1, t_1 \in X \setminus
\{u, v\}$ (possible~: $k \geq 4$) et remplacer $g_0$ par $g_0 \circ
(s_1\ t_1)$, qui est paire et envoie toujours $u \mapsto a$, $v
\mapsto b$. Prolonger par l'identité hors de $X$~: une permutation
paire $g \in G$ (c'est une permutation paire de $X$). Alors la
question 12~:
\[
g\,(u\ v\ w)\,g^{-1} = (g(u)\ g(v)\ g(w)) = (a\ b\ w) \in G,
\]
en utilisant $g(w) = w$.

\textbf{16.} Tout $3$-cycle de $X \cup \{w\}$ est dans $G$~: ceux à
support dans $X$ sont des permutations paires de $X$~; un à support
$\{a, b, w\}$ est $(a\ b\ w)$ ou $(b\ a\ w)$, tous deux livrés par la
question 15. Par l'\cref{exo:b2:structures:6}, les $3$-cycles de
l'ensemble à $(k+1)$ éléments $X \cup \{w\}$ engendrent son groupe
alterné, donc $G$ contient toute permutation paire de $X \cup \{w\}$.
\emph{Enchaînement~:} soit $G = \langle s_0, \dots, s_{12}\rangle$. Le
lemme A appliqué à $s_0 = (x_0\ x_1\ x_2)$ et $s_1 = (x_1\ x_2\
x_3)$ (supports partageant $\{x_1, x_2\}$) donne toutes les
permutations paires de $X_4 = \{x_0, x_1, x_2, x_3\}$. Si $G$ contient
toutes les permutations paires de $X_m = \{x_0, \dots, x_{m-1}\}$ ($4
\leq m \leq 14$), alors $s_{m-2} = (x_{m-2}\ x_{m-1}\ x_m)$ a $u =
x_{m-2}, v = x_{m-1} \in X_m$ et nouvelle lettre $w = x_m$~: le lemme
B et la première partie donnent toutes les permutations paires de
$X_{m+1}$. Récurrence jusqu'à $m = 14$~: $G \supseteq A_{15}$
(permutations paires des quinze cases), et $G \subseteq A_{15}$
puisque chaque $s_t$ est pair~: $\langle s_0, \dots, s_{12}\rangle =
A_{15}$.

\textbf{17.} Questions 13 et 16~: $A_{15} = \langle s_0, \dots,
s_{12}\rangle \subseteq H$~; question 9~: $H \subseteq A_{15}$. Donc
$H = A_{15}$, d'\omterm{def:b2:structures:generated}{ordre} $15!/2 = 653\,837\,184\,000$~: tout
réarrangement pair des quinze tuiles est l'effet d'un programme.

\textbf{18.} La question 8 a réduit $R = \{I = +1\}$ à la
réalisation de tout $\rho \in \mathfrak{S}_{15}$ pair par un
programme~: fait par la question 17. Avec la question 6, $\abs R =
16!/2 = 10\,461\,394\,944\,000$. \emph{Deux classes~:} soit $t_0 =
(14\ 15)$ agissant sur les \emph{contenus}~: $\varphi(\sigma) = t_0
\circ \sigma$. Un coup légal depuis $\sigma$ est un coup légal depuis
$\varphi(\sigma)$ (la case de la case vide est inchangée~:
$(t_0\sigma)^{-1}(16) = \sigma^{-1}(t_0(16)) = \sigma^{-1}(16)$,
et la case déplacée est la même), et $\varphi(\sigma \circ \tau)
= \varphi(\sigma) \circ \tau$~: $\varphi$ envoie les suites de coups
sur des suites de coups, bijectivement (c'est une involution). Elle
change $I$ de signe~: $\varepsilon(t_0\sigma) = -\varepsilon(\sigma)$,
même case de la case vide. Donc $\varphi$ envoie la classe $R = \{I =
+1\}$ de $\mathrm{id}$ bijectivement sur la classe de
$\varphi(\mathrm{id}) = \sigma_L$, qui est donc tout $\{I = -1\}$~:
exactement deux classes. C'est le théorème de Johnson--Story.

\textbf{19.} Indexer les cases dans l'\omterm{def:b2:structures:generated}{ordre} de lecture et soit $k =
4(i - 1) + j$ la case de la case vide. Compter les inversions de
$\sigma$ (paires de cases $x < y$ avec $\sigma(x) > \sigma(y)$)~:
les paires de deux cases de tuiles contribuent $N$~; les paires
impliquant la case vide~: les cases après la case vide portent toutes
des tuiles $< 16$, chacune inversée ($16 - k$ paires), les cases
avant elle ne sont jamais inversées. Donc
$\varepsilon(\sigma) = (-1)^{N + 16 - k} = (-1)^{N + k}$. Comme
$k = 4(i-1) + j \equiv j \pmod 2$,
\[
I(\sigma) = (-1)^{N + j}\,(-1)^{i + j} = (-1)^{N + i}
= (-1)^{N + r + 1}
\]
en utilisant $i = 5 - r$. Par la question 18, $\sigma$ est résoluble
si et seulement si $I(\sigma) = +1$ si et seulement si $N + r$ est
impair. Vérification~: résolu, $N = 0$, $r = 1$~: impair, résoluble~;
Loyd, $N = 1$, $r = 1$~: pair, non résoluble.

\textbf{20.} \emph{Action~:} $e \cdot \sigma = \sigma \circ
\mathrm{id} = \sigma$ et $g \cdot (h \cdot \sigma) = \sigma
\circ h^{-1} \circ g^{-1} = \sigma \circ (gh)^{-1} = (gh) \cdot
\sigma$~; et $\sigma \circ h^{-1}$ est encore une configuration à
case vide au domicile ($h$ fixe la case $16$). \emph{Libre~:} $\sigma
\circ h^{-1} = \sigma$ donne $h^{-1} = \mathrm{id}$ (composer avec
$\sigma^{-1}$). \emph{Orbites = classes de programmes~:} la question
9 dit que les configurations atteignables depuis $\sigma$ par
programmes sont exactement les $\sigma \circ \pi^{-1}$, $\pi \in H$~:
l'orbite $H \cdot \sigma$. \emph{Décompte~:} la liberté rend $h
\mapsto h \cdot \sigma$ injective, donc toute orbite a $\abs H =
15!/2$ éléments~; les $15!$ configurations à case vide au domicile se
répartissent donc en $15!\,/\,(15!/2) = 2$ orbites --- l'ombre à case
vide au domicile des deux classes de Johnson--Story.

\textbf{21.} La grille $3 \times 3$ est bipartie pour le coloriage en
damier~: chaque pas d'un chemin change de couleur, donc tout chemin
\emph{fermé} est de longueur paire. Un chemin fermé visitant chacune
des $9$ cases exactement une fois serait de longueur $9$, impaire~:
impossible. La construction du grand tour de la partie III est donc
indisponible sur le taquin à huit.

\textbf{22.} \emph{Tour périmétrique} $9 \to 8 \to 7 \to 4 \to 1
\to 2 \to 3 \to 6 \to 9$ (tous les pas adjacents~; longueur $8$,
paire)~: par la comptabilité de la question 11 avec $c_1 = 8, c_2 =
7, c_3 = 4, c_4 = 1, c_5 = 2, c_6 = 3, c_7 = 6$, l'effet est
\[
\zeta' = (8\ 6\ 3\ 2\ 1\ 4\ 7),
\]
un $7$-cycle fixant le centre $5$ (le contenu de $7$ se déplace en
$8$, de $4$ en $7$, de $1$ en $4$, de $2$ en $1$, de $3$ en $2$, de
$6$ en $3$, et de $8$ en $6$). \emph{Tour de coin} $9 \to 6 \to
5 \to 8 \to 9$~: effet $(6\ 8\ 5)$ (le contenu de $5$ se déplace en
$6$, de $8$ en $5$, de $6$ --- garé en $9$ --- en $8$). Poser
$y_t = \zeta'^{\,t}(8)$~: $y_0 = 8, y_1 = 6, y_2 = 3, y_3 = 2,
y_4 = 1, y_5 = 4, y_6 = 7$. Conjugaison (question 12)~:
\[
\zeta'^{\,t}\,(6\ 8\ 5)\,\zeta'^{-t}
= (y_{t+1}\ y_t\ 5) =: T_t \in H_{3\times3},
\]
puisque $\zeta'$ fixe $5$. Les supports de $T_0 = (y_1\ y_0\ 5)$
et $T_1 = (y_2\ y_1\ 5)$ partagent exactement $\{y_1, 5\}$~: le
lemme A donne toutes les permutations paires de $\{y_0, y_1, y_2,
5\}$. Puis $T_2 = (y_3\ y_2\ 5)$ adjoint $y_3$ par le lemme B (ses
lettres $y_2, 5$ sont dans l'ensemble courant, $k = 4$), et $T_3,
T_4, T_5$ adjoignent $y_4, y_5, y_6$ à leur tour~: toutes les
permutations paires des huit cases hors domicile sont dans le groupe
des programmes, qui est aussi constitué de permutations paires
(l'argument de la question 9 est indépendant du plateau). Donc
$H_{3\times3} = A_8$, et le raisonnement des questions 6, 8, 18 ---
lui aussi indépendant du plateau --- montre que les configurations
atteignables sont exactement celles avec $I = +1$~: la moitié de
$9!$, soit $181\,440$.

\textbf{23.} Étiqueter les cases $0, \dots, n-1$ autour du \omterm{def:b2:structures:sn}{cycle}.
Un coup échange la case vide avec l'une de ses deux voisines. Lire
les tuiles dans l'\omterm{def:b2:structures:generated}{ordre} \omterm{def:b2:structures:generated}{cyclique} en partant juste après la case
vide~: un mot $w$ listant les $n - 1$ tuiles. Déplacer la case vide
d'un pas en avant remplace $(p, w)$ par $(p + 1, \rho w)$, où $p$ est
la case de la case vide et $\rho$ fait tourner cycliquement le mot
d'un cran~; le coup en arrière est l'inverse. L'\omterm{def:b2:structures:generated}{ordre} \emph{\omterm{def:b2:structures:generated}{cyclique}}
des tuiles (le mot à rotation près) est donc invariant. La classe
atteignable de $(p, w)$ est l'orbite de l'application $g \colon (p, w)
\mapsto (p+1, \rho w)$, un élément d'\omterm{def:b2:structures:generated}{ordre} $\operatorname{lcm}(n, n-1)
= n(n-1)$ dans le produit des deux \omterm{def:b2:structures:generated}{groupes cycliques} (translations de
$\Z/n\Z$ et rotations des $n-1$ positions du mot), le ppcm valant
$n(n-1)$ car $\gcd(n, n-1) = 1$~: chaque classe a exactement $n(n-1)$
configurations, toutes avec le même collier. Classes~:
$n!\,/\,\bigl(n(n-1)\bigr) = (n-2)!$. Pour $n \geq 5$,
$(n-2)! > 2$~: l'invariant de parité (deux classes au mieux) est
aveugle à presque toute l'obstruction~; la richesse du plateau $4
\times 4$ --- où la parité est la \emph{seule} obstruction --- est
un fait véritablement géométrique, pas formel.

\textbf{24.} Les deux plateaux ont les tuiles en \omterm{def:b2:structures:generated}{ordre} entièrement
inversé, donc $N = \binom{15}{2} = 105$ dans les deux cas (toute
paire de tuiles est inversée). \emph{Case vide au domicile~:} $r = 1$,
$N + r = 106$ pair~: non résoluble. \emph{Case vide dans la case
$1$~:} la case vide est dans la ligne du haut, $r = 4$, $N + r = 109$
impair~: résoluble. Deux plateaux qui ne diffèrent que par
l'emplacement du trou tombent de part et d'autre du mur.

\textbf{25.} \emph{Propriété de morphisme~:} elle convertit «~un coup
= une \omterm{def:b2:structures:sn}{transposition}~» en «~un coup = un changement de signe~»
(questions 2, 4), rendant $I$ calculable coup par coup.
\emph{Lagrange~:} il a forcé $6 \mid \abs{\langle s, t\rangle}$ dans
le lemme A et a dimensionné les classes dans l'exclusion de l'\omterm{def:b2:structures:generated}{ordre}
$6$ (question 14). \emph{Génération par les $3$-cycles~:} elle a
converti «~$H$ contient assez de $3$-cycles~» en «~$H$ contient tout
$A_{15}$~» (question 16). \emph{Conjugaison~:} elle a fabriqué les
quinze $3$-cycles consécutifs à partir d'un seul tour $2 \times 2$
transporté par le grand tour (questions 12--13), et les $3$-cycles
$(a\ b\ w)$ du lemme B. \emph{Méta-principe~:} un invariant démontre
l'impossibilité, une construction explicite démontre la possibilité,
et un problème est entièrement résolu exactement lorsque les deux
bornes se rejoignent --- ici, à une moitié.
\end{solution}
