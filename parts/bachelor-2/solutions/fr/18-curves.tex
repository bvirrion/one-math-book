\section*{Chapitre \ref{ch:b2:curves} --- Courbes}

\begin{solution}{exo:b2:curves:1}
$\gamma'(t) = (1 - \cos t,\ \sin t)$, donc
\[
\norm{\gamma'(t)}^2 = (1 - \cos t)^2 + \sin^2 t
= 2 - 2\cos t = 4\sin^2\tfrac t2 ,
\]
et $\norm{\gamma'(t)} = 2\sin\frac t2$ (positif sur $[0,
2\pi]$). D'où
\[
L = \int_0^{2\pi} 2\sin\tfrac t2\,\dd t
= \Bigl[-4\cos\tfrac t2\Bigr]_0^{2\pi} = 8 :
\]
une arche de la cycloïde a pour \omterm{def:b2:curves:length}{longueur} $8$ (pour une roue de
rayon $1$) --- un résultat célèbre de Wren, sans aucun $\pi$ en
vue.
\end{solution}

\begin{solution}{exo:b2:curves:2}
Avec $x = a\cos t$, $y = b\sin t$~: $x' = -a\sin t$, $y' = b\cos
t$, $x'' = -a\cos t$, $y'' = -b\sin t$, donc par
\cref{prop:b2:curves:kappaformula}
\[
\kappa(t)
= \frac{x'y'' - y'x''}{(x'^2 + y'^2)^{3/2}}
= \frac{ab\sin^2 t + ab\cos^2 t}
       {(a^2\sin^2 t + b^2\cos^2 t)^{3/2}}
= \frac{ab}{(a^2\sin^2 t + b^2\cos^2 t)^{3/2}} .
\]
Le dénominateur est minimal quand $\sin t = 0$ (valeur $b^3$,
points $(\pm a, 0)$) et maximal quand $\cos t = 0$ (valeur $a^3$,
points $(0, \pm b)$), puisque $a > b$. Donc $\kappa$ est maximale
aux extrémités du grand axe, $\kappa_{\max} = a/b^2$, et minimale
aux extrémités du petit axe, $\kappa_{\min} = b/a^2$~: l'ellipse
se courbe le plus fortement aux sommets de son grand axe.
\end{solution}

\begin{solution}{exo:b2:curves:3}
Pour le graphe $\gamma(x) = (x, \cosh x)$~: $\norm{\gamma'(x)} =
\sqrt{1 + \sinh^2 x} = \cosh x$, donc la \omterm{def:b2:curves:length}{longueur d'arc} de $0$ à
$x$
est $\int_0^x \cosh u\,\dd u = \sinh x$. \omterm{thm:b2:curves:frenet2d}{Courbure} d'un graphe
(\cref{prop:b2:curves:kappaformula})~:
\[
\kappa(x) = \frac{f''(x)}{(1 + f'(x)^2)^{3/2}}
= \frac{\cosh x}{\cosh^3 x} = \frac{1}{\cosh^2 x} ,
\]
donc $R(x) = \cosh^2 x$, comme annoncé. Notons la jolie
coïncidence $R(x) = 1 + s(x)^2$ avec $s = \sinh x$ la \omterm{def:b2:curves:length}{longueur
d'arc}~: le \omterm{def:b2:curves:curvature}{rayon de courbure} de la chaînette croît comme le carré
de la \omterm{def:b2:curves:length}{longueur d'arc} depuis le sommet.
\end{solution}

\begin{solution}{exo:b2:curves:4}
$\gamma'(t) = e^t(\cos t - \sin t,\ \sin t + \cos t)$, donc
\[
\langle \gamma(t), \gamma'(t)\rangle = e^{2t}
\bigl(\cos t(\cos t - \sin t) + \sin t(\sin t + \cos t)\bigr)
= e^{2t},
\]
tandis que $\norm{\gamma(t)} = e^t$ et $\norm{\gamma'(t)} =
e^t\sqrt 2$. D'où
\[
\cos\angle\bigl(\gamma, \gamma'\bigr)
= \frac{e^{2t}}{e^t \cdot e^t\sqrt2} = \frac{1}{\sqrt2} :
\]
la \omterm{def:b2:curves:arc}{tangente} fait toujours l'angle $\pi/4$ avec le rayon --- la
propriété \emph{équiangle} de la spirale logarithmique. \omterm{def:b2:curves:length}{Longueur
d'arc} sur $(-\infty, 0]$~:
\[
\int_{-\infty}^0 \norm{\gamma'(t)}\,\dd t
= \sqrt2\int_{-\infty}^0 e^t\,\dd t = \sqrt 2 ,
\]
finie bien que la spirale s'enroule une infinité de fois autour de
l'origine. \omterm{thm:b2:curves:frenet2d}{Courbure}~: avec $x'y'' - y'x''$ calculé à partir de
$\gamma'' =
e^t(-2\sin t,\ 2\cos t)$,
\[
x'y'' - y'x'' = e^{2t}\bigl(2\cos t(\cos t - \sin t)
+ 2\sin t(\sin t + \cos t)\bigr) = 2e^{2t},
\]
donc $\kappa(t) = \dfrac{2e^{2t}}{(e^t\sqrt2)^3} =
\dfrac{1}{e^t\sqrt2}$~: la \omterm{thm:b2:curves:frenet2d}{courbure} est $1/(\sqrt2\,
\norm{\gamma})$, décroissant à mesure que la spirale grandit.
\end{solution}

\begin{solution}{exo:b2:curves:5}
\emph{Premier arc~:} $\gamma(t) = (t^2,\ t^4 + t^5)$. Dérivées en
$0$~: $\gamma'' = (2, 0) \neq 0$, donc $p = 2$. Puis $\gamma^{(3)}(0)
= (0, 0)$, $\gamma^{(4)}(0) = (0, 24)$, non colinéaire à $(2,
0)$~: $q = 4$. Les deux pairs~: \emph{point de rebroussement de
seconde espèce} --- les deux
branches partent dans la direction $+u = (1,0)$ et restent du même
côté de la \omterm{def:b2:curves:arc}{tangente}. (En effet $y = x^2 \pm x^{5/2}$ sur les deux
branches~: même signe pour $x$ petit.)

\emph{Second arc~:} $\gamma(t) = (t^3, t^4)$. $\gamma'(0) =
\gamma''(0) = 0$, $\gamma^{(3)}(0) = (6, 0)$~: $p = 3$, impair. Puis
$\gamma^{(4)}(0) = (0, 24)$~: $q = 4$, pair. Impair--pair~:
\emph{point ordinaire} --- malgré l'annulation de la vitesse, la
trajectoire $y = x^{4/3}$ traverse l'origine de manière lisse,
restant au-dessus de sa \omterm{def:b2:curves:arc}{tangente} $y = 0$.
\end{solution}

\begin{solution}{exo:b2:curves:6}
Dérivons $c(s) = \gamma(s) + \dfrac{1}{\kappa(s)}N(s)$ à l'aide des
formules de Frenet dans le plan
(\cref{thm:b2:curves:frenet2d})~:
\[
c'(s) = T(s) - \frac{\kappa'(s)}{\kappa(s)^2}N(s)
+ \frac{1}{\kappa(s)}\,\bigl(-\kappa(s)T(s)\bigr)
= -\frac{\kappa'(s)}{\kappa(s)^2}\,N(s) ,
\]
les termes tangentiels s'annulant exactement. Ainsi la vitesse de
la développée est portée par $N(s)$, qui dirige la normale de
$\gamma$ en $\gamma(s)$ --- et le point $c(s)$ est sur cette
normale même~: la développée est l'\emph{\omterm{pb:b2:curves:1}{enveloppe} des normales}.
(Là où $\kappa' = 0$ la développée a un point singulier~; c'est
ce qui produit les points de rebroussement de la développée d'une
ellipse.)
\end{solution}

\begin{solution}{exo:b2:curves:7}
Soit $G(s) = F(s)F(s)^{\mathsf T}$. Alors, en utilisant $F' = FA$ et
$(F^{\mathsf T})' = (F')^{\mathsf T} = A^{\mathsf T}F^{\mathsf T}$,
\[
G' = F'F^{\mathsf T} + F(F^{\mathsf T})'
= FAF^{\mathsf T} + FA^{\mathsf T}F^{\mathsf T}
= F(A + A^{\mathsf T})F^{\mathsf T} = 0
\]
par antisymétrie. Donc $G$ est constante sur l'intervalle, égale à
$G(s_0) = F(s_0)F(s_0)^{\mathsf T} = I$~: $F(s)$ est orthogonale pour
tout $s$. (Autrement, sans calculer $G'$ jusqu'à zéro~: $G$ et la
constante $I$ sont toutes deux solutions du système linéaire $Y' = YA +
A^{\mathsf T}Y$ avec la même valeur initiale, et l'unicité de
Cauchy--Lipschitz pour les systèmes linéaires,
\cref{ch:b2:diffeq}, force $G
\equiv I$.)

\emph{Pertinence~:} le système de Frenet $(T, N, B)' = (T, N,
B)\,A(s)$ a pour matrice de coefficients antisymétrique
\[
A = \begin{pmatrix} 0 & -\kappa & 0\\ \kappa & 0 & -\tau\\
0 & \tau & 0\end{pmatrix}
\]
(les colonnes exprimant $T', N', B'$). Le calcul ci-dessus montre
qu'un repère solution qui part orthonormé \emph{reste}
orthonormé --- l'étape clé du théorème fondamental
reconstruisant une courbe à partir de $(\kappa, \tau)$.
\end{solution}

\begin{solution}{exo:b2:curves:8}
Par \cref{thm:b2:curves:fundamental} (partie unicité), il existe
une fonction angle $\varphi$ de classe $\mathcal{C}^1$ avec $T(s) =
(\cos\varphi(s), \sin\varphi(s))$ et $\varphi' = \kappa$. D'où
\[
\int_0^L \kappa(s)\,\dd s = \varphi(L) - \varphi(0) .
\]
Comme l'arc est fermé de période $L$, $T(L) = T(0)$~:
$(\cos\varphi(L), \sin\varphi(L)) = (\cos\varphi(0),
\sin\varphi(0))$, donc $\varphi(L) - \varphi(0) \in 2\pi\Z$. La
\omterm{thm:b2:curves:frenet2d}{courbure} totale d'une courbe fermée est donc toujours un multiple
entier de $2\pi$ --- l'entier étant l'\emph{indice}
de la \omterm{def:b2:curves:arc}{tangente} (le nombre de tours \omterm{def:b2:metric:complete}{complets} que fait $T$). Pour le
cercle de rayon $R$~: $\kappa = 1/R$ et $L = 2\pi R$, \omterm{thm:b2:curves:frenet2d}{courbure}
totale $2\pi$, indice $1$~; le théorème des \omterm{def:b2:curves:arc}{tangentes}
tournantes affirme que cette valeur vaut pour toute courbe fermée
simple.
\end{solution}

\begin{solution}{exo:b2:curves:9}
Comme $\tau \equiv 0$, la courbe est contenue dans un plan
(\cref{prop:b2:curves:torsion})~; travaillons dans ce plan.
Considérons le
candidat centre
\[
c(s) = \gamma(s) + \frac{1}{\kappa}N(s)
\qquad (\kappa \text{ constant}).
\]
En dérivant avec les formules de Frenet ($N' = -\kappa T + \tau B
= -\kappa T$ ici)~:
\[
c'(s) = T + \frac1\kappa(-\kappa T) = 0 ,
\]
donc $c$ est un point constant $\Omega$. Alors $\norm{\gamma(s) -
\Omega} = \norm{-\frac1\kappa N(s)} = \frac1\kappa$ pour tout $s$~:
la courbe est sur le cercle de centre $\Omega$ et de rayon
$1/\kappa$ (dans son plan), et étant un arc non constant de
celui-ci, elle est un arc de ce cercle.
\end{solution}

\begin{solution}{exo:b2:curves:10}
Pour le graphe $\gamma(x) = (x, x^2/2)$, $\norm{\gamma'(x)} =
\sqrt{1 + x^2}$, donc $L = \int_0^a\sqrt{1+x^2}\,\dd x$.
En substituant $x = \sinh u$ ($\dd x = \cosh u\,\dd u$, $u$ de
$0$ à $u_a = \ln(a + \sqrt{1+a^2})$)~:
\[
L = \int_0^{u_a}\cosh^2 u\,\dd u
= \frac12\bigl[u + \sinh u\cosh u\bigr]_0^{u_a}
= \frac12\Bigl(\ln\bigl(a + \sqrt{1+a^2}\bigr) +
a\sqrt{1+a^2}\Bigr),
\]
en utilisant $\cosh^2 u = \frac{1 + \cosh 2u}2$ et $\sinh u_a = a$,
$\cosh u_a = \sqrt{1 + a^2}$.
\end{solution}

\begin{solution}{exo:b2:curves:11}
Paramétrons par \omterm{def:b2:curves:length}{longueur d'arc}
(\cref{thm:b2:curves:arclength}) et posons $\lambda(s) =
\langle P - \gamma(s), T(s)\rangle$, une fonction $\mathcal C^1$~;
puisque $P$ est sur la \omterm{def:b2:curves:arc}{tangente} en $\gamma(s)$,
le vecteur $P - \gamma(s)$ est colinéaire à $T(s)$, donc $P =
\gamma(s) + \lambda(s)T(s)$. En dérivant,
\[
0 = T(s) + \lambda'(s)T(s) + \lambda(s)T'(s)
= \bigl(1 + \lambda'(s)\bigr)T(s) + \lambda(s)T'(s),
\]
et $T'(s) \perp T(s)$ (dériver $\norm T^2 = 1$), donc
les deux composantes s'annulent~: $\lambda' = -1$ et $\lambda T' = 0$.
Alors $\lambda(s) = c - s$ s'annule au plus une fois, donc $T' = 0$
sur un ensemble dense, d'où partout par continuité~: $T$ est un
vecteur \omterm{def:b2:hermitian:adjoint}{unitaire} constant et $\gamma(s) = \gamma(s_0) + (s -
s_0)T$~: une droite (passant par $P$, comme il se doit).
\end{solution}

\begin{solution}{exo:b2:curves:12}
(a) La matrice de Frenet
\[
A(s) = \begin{pmatrix} 0 & -\kappa & 0\\ \kappa & 0 & -\tau\\
0 & \tau & 0\end{pmatrix}
\]
a des coefficients \omterm{def:b2:metric:continuity}{continus}, donc le système linéaire $F' = FA(s)$,
$F(s_0) = F_0$ (une matrice orthonormée directe) a une unique
solution sur tout $J$ (\cref{thm:b2:diffeq:linear}). Par
\cref{exo:b2:curves:7}, $F(s)$ est orthogonale pour tout $s$~;
$\det F$ est \omterm{def:b2:metric:continuity}{continu} à valeurs dans $\{\pm1\}$ et vaut
$1$ en $s_0$, donc $F(s)$ est direct pour tout $s$.

(b) On lit les lignes $T, N, B$ de $F$ (de sorte que $T' = \kappa
N$, $N' = -\kappa T + \tau B$, $B' = -\tau N$) et on pose
$\gamma(s) = \gamma_0 + \int_{s_0}^s T(u)\,\dd u$. Alors
$\gamma' = T$ est un vecteur \omterm{def:b2:hermitian:adjoint}{unitaire}~: vitesse unité~; $T' = \kappa N$
avec $\kappa > 0$ et $N$ \omterm{def:b2:hermitian:adjoint}{unitaire} orthogonal à $T$, donc $\gamma$
est birégulier de \omterm{thm:b2:curves:frenet2d}{courbure} $\norm{T'} = \kappa$ et
normale principale $N$~; la binormale est $T \wedge N = B$
(repère orthonormé direct), et $B' = -\tau N$ identifie la
\omterm{thm:b2:curves:frenet3d}{torsion} à $\tau$.

(c) Soient $\gamma_1, \gamma_2$ deux courbes birégulières
paramétrées par \omterm{def:b2:curves:length}{longueur d'arc} de même $(\kappa, \tau)$. Il existe
une unique isométrie directe $\Phi = \rho + w$ ($\rho \in SO(3)$)
envoyant
$\gamma_1(s_0)$ sur $\gamma_2(s_0)$ et le repère de Frenet de
$\gamma_1$ en $s_0$ sur celui de $\gamma_2$ en $s_0$. La courbe
$\Phi\circ\gamma_1$ est paramétrée par \omterm{def:b2:curves:length}{longueur d'arc}, de mêmes
invariants
(son repère est $\rho$ appliqué à celui de $\gamma_1$, et
$\rho$ préserve les produits vectoriels, étant directe). Or les
repères de $\Phi\circ\gamma_1$ et $\gamma_2$ sont tous deux
solutions de $F' =
FA(s)$ avec la même valeur initiale, donc ils coïncident par
unicité~; en particulier les \omterm{def:b2:curves:arc}{tangentes} coïncident, et en intégrant
depuis le point commun $s_0$~: $\Phi\circ\gamma_1 = \gamma_2$.
\end{solution}

\begin{solution}{pb:b2:curves:1}
\textbf{1.} Le système est linéaire en $(x, y)$ de \omterm{def:b2:linalg:det}{déterminant}
$\Delta(t) = a b' - a'b \neq 0$~: la règle de Cramer donne
l'unique solution
\[
x = \frac{c b' - c' b}{\Delta}, \qquad
y = \frac{a c' - a' c}{\Delta}.
\]

\textbf{2.} Comme $a(t)x(t) + b(t)y(t) = c(t)$ identiquement,
en dérivant on obtient $a'x + b'y + ax' + by' = c'$~; la
seconde équation caractéristique élimine $a'x + b'y - c'$, donc
$a(t)x'(t) + b(t)y'(t) = 0$~: $E'(t)$ est orthogonal à $(a,
b)$, donc parallèle à $(-b, a)$, la direction de $D_t$. Comme
$E(t) \in D_t$ (première équation) et $E'(t) \neq 0$, la droite
$D_t$ est exactement la \omterm{def:b2:curves:arc}{tangente} de la courbe $E$ en $E(t)$.

\textbf{3.} Ici $(a, b, c) = (t, -1, t^2/2)$, donc $\Delta =
t\cdot0 - 1\cdot(-1) = 1$ et
\[
x = \frac{c b' - c' b}{\Delta} = \tfrac{t^2}2\cdot 0 +
t = t, \qquad
y = \frac{a c' - a' c}{\Delta} = t\cdot t - \tfrac{t^2}2 =
\tfrac{t^2}2 :
\]
l'\omterm{pb:b2:curves:1}{enveloppe} des \omterm{def:b2:curves:arc}{tangentes} de la parabole est la
parabole, comme il se doit.

\textbf{4.} La normale est $\langle M, T(s)\rangle =
\langle\gamma(s), T(s)\rangle$~: coefficients $a = T_1$, $b =
T_2$, $c = \langle\gamma, T\rangle$. En dérivant avec
Frenet ($T' = \kappa N$)~: $a' = \kappa N_1$, $b' = \kappa
N_2$, et $c' = \langle T, T\rangle + \langle\gamma, \kappa
N\rangle = 1 + \kappa\langle\gamma, N\rangle$. La seconde
équation caractéristique $\kappa\langle M, N\rangle = 1 +
\kappa\langle\gamma, N\rangle$ s'écrit $\kappa\langle M -
\gamma, N\rangle = 1$. La première dit $M - \gamma \perp T$,
donc $M - \gamma = \mu N$ avec $\mu = 1/\kappa$~: le
point caractéristique est $\gamma + \frac1\kappa N$, le \omterm{def:b2:curves:curvature}{centre
de courbure}, et l'\omterm{pb:b2:curves:1}{enveloppe} des normales est la développée
de \cref{exo:b2:curves:6}. Enfin $\Delta = T_1\kappa N_2 -
\kappa N_1 T_2 = \kappa\det(T, N) = \kappa \neq 0$.

\textbf{5.} Faisceau~: le système $x\cos\theta + y\sin\theta =
0$, $-x\sin\theta + y\cos\theta = 0$ a pour \omterm{def:b2:linalg:det}{déterminant} $1$ et
solution $(0,0)$ pour tout $\theta$~: $E \equiv (0,0)$, $E'
\equiv 0$, et il n'y a pas de courbe --- \omterm{def:b2:funcseq:def}{simplement} le point
commun de toutes les droites. Famille parallèle~: $a' = b' = 0$
donne
$\Delta \equiv 0$ et la seconde équation $0 = c'(t)$, qui
échoue dès que la famille bouge réellement~: pas de point
caractéristique, et en effet une famille de droites parallèles ne
touche aucune courbe le long de tous ses membres.

\textbf{6.} La droite passant par $(\cos t, 0)$ et $(0, \sin t)$
est $\frac x{\cos t} + \frac y{\sin t} = 1$, c'est-à-dire $x\sin t +
y\cos t = \sin t\cos t$. Avec $(a, b, c) = (\sin t, \cos t,
\sin t\cos t)$~: $a' = \cos t$, $b' = -\sin t$, $c' = \cos
2t$, $\Delta = -\sin^2 t - \cos^2 t = -1$. Cramer~:
\begin{align*}
x &= \frac{cb' - c'b}{-1} = \sin^2 t\cos t + \cos 2t\cos t
= \cos t\,(\sin^2 t + \cos^2 t - \sin^2 t) = \cos^3 t,\\
y &= \frac{ac' - a'c}{-1} = \sin t\cos^2 t - \sin t\cos 2t
= \sin t\,(\cos^2 t - \cos^2 t + \sin^2 t) = \sin^3 t .
\end{align*}
Et $(\cos^3t)^{2/3} + (\sin^3t)^{2/3} = 1$~: l'\omterm{pb:b2:curves:1}{astroïde}.

\textbf{7.} $E'(t) = 3(-\cos^2 t\sin t,\ \sin^2 t\cos t)$
s'annule en $t = 0$. Là, $E''(0) = (-3, 0) \neq 0$ donne
$p = 2$~; la composante en $x$ de $E$ est paire en $t$, donc
$E'''(0) = (0, 6)$, non colinéaire~: $q = 3$. Pair--impair~:
point de rebroussement de première espèce en $(1, 0)$
(\cref{prop:b2:curves:local}), de \omterm{def:b2:curves:arc}{tangente} selon l'axe des
$x$. Les symétries $x \mapsto -x$, $y \mapsto -y$, $(x,
y) \mapsto (y, x)$ de l'\omterm{pb:b2:curves:1}{astroïde} transportent le rebroussement en
$(-1, 0)$ et $(0, \pm1)$.

\textbf{8.} $E'(t) = 3\sin t\cos t\,(-\cos t, \sin t)$, donc
$\norm{E'(t)} = 3\abs{\sin t\cos t} = \tfrac32\abs{\sin 2t}$.
Un quadrant~: $\int_0^{\pi/2}\tfrac32\sin 2t\,\dd t =
\tfrac32$, et par symétrie la \omterm{def:b2:curves:length}{longueur} totale est $4 \cdot
\tfrac32 = 6$.

\textbf{9.} $E(t) - P_t = (\cos^3 t - \cos t,\ \sin^3 t) =
\sin^2 t\,(-\cos t,\ \sin t) = \sin^2 t\,(Q_t - P_t)$. Ainsi le
point de contact est le \omterm{def:b2:affine:barycenter}{barycentre} de $(P_t, \cos^2 t)$ et
$(Q_t, \sin^2 t)$~: lorsque $t$ va de $0$ à $\pi/2$ il glisse
de l'extrémité au sol à l'extrémité au mur de l'échelle.

\textbf{10.} Dans le premier quadrant la région sous
l'\omterm{pb:b2:curves:1}{astroïde} a pour aire $\int_0^1 y\,\dd x$ avec $x = \cos^3 t$
décroissant de $1$ à $0$ quand $t$ va de $0$ à $\pi/2$~:
\[
\int_0^1 y\,\dd x
= \int_{\pi/2}^{0}\sin^3 t\,(-3\cos^2 t\sin t)\,\dd t
= 3\int_0^{\pi/2}\sin^4 t\cos^2 t\,\dd t .
\]
Linéarisons~: $\sin^4 t\cos^2 t = (\sin t\cos t)^2\sin^2 t =
\tfrac18\bigl(\sin^2 2t - \sin^2 2t\cos 2t\bigr)$, et
$\int_0^{\pi/2}\sin^2 2t\,\dd t = \tfrac\pi4$ tandis que
$\int_0^{\pi/2}\sin^2 2t\cos 2t\,\dd t =
\bigl[\tfrac{\sin^3 2t}6\bigr]_0^{\pi/2} = 0$. Donc
l'intégrale vaut $\tfrac\pi{32}$, l'aire du quadrant
$\tfrac{3\pi}{32}$, et l'aire délimitée $4 \cdot
\tfrac{3\pi}{32} = \tfrac{3\pi}8$.

\textbf{11.} La \omterm{def:b2:curves:arc}{tangente} en $\gamma(t) = (t, t^2/2)$ est
dirigée par $(1, t)$, donc la normale est $\{(x, y) : (x -
t) + t\,(y - t^2/2) = 0\}$, c'est-à-dire $x + t\,y = t +
\tfrac{t^3}2$.

\textbf{12.} $(a, b, c) = (1, t, t + t^3/2)$~: $a' = 0$, $b' =
1$, $c' = 1 + \tfrac32 t^2$, $\Delta = 1$. Alors $y = ac' -
a'c = 1 + \tfrac32t^2$ et
\[
x = cb' - c'b = t + \tfrac{t^3}2 - t\Bigl(1 +
\tfrac32t^2\Bigr) = -t^3 .
\]
En éliminant $t$~: $t^2 = \tfrac23(y - 1)$ et $x^2 = t^6 =
\tfrac8{27}(y - 1)^3$~: une parabole semi-cubique de sommet
$(0, 1)$.

\textbf{13.} $T = (1, t)/\sqrt{1+t^2}$, $N = (-t,
1)/\sqrt{1+t^2}$, et $\kappa = (1+t^2)^{-3/2}$, donc
\[
\gamma + \frac1\kappa N = (t, \tfrac{t^2}2) +
(1+t^2)\,(-t, 1) = \Bigl(-t^3,\ 1 + \tfrac32t^2\Bigr),
\]
la même courbe~: l'\omterm{pb:b2:curves:1}{enveloppe} des normales est le lieu des
\omterm{def:b2:curves:curvature}{centres de courbure}, comme la question 4 le promettait.

\textbf{14.} $E'(t) = (-3t^2, 3t)$ s'annule en $t = 0$~;
$E''(0) = (0, 3) \neq 0$ donne $p = 2$ et $E'''(0) = (-6,
0)$ donne $q = 3$~: un point de rebroussement de première espèce
en $(0, 1)$. Le
sommet est là où $\kappa = (1+t^2)^{-3/2}$ est maximale, donc
$\kappa'(0) = 0$ et la vitesse de la développée
$-\frac{\kappa'}{\kappa^2}N$ s'annule exactement là~: les
points de rebroussement de la développée sont aux extrema de
\omterm{thm:b2:curves:frenet2d}{courbure}.

\textbf{15.} La normale au paramètre $t$ passe par
$(x_0, y_0)$ ssi $x_0 + t\,y_0 = t + \tfrac{t^3}2$,
c'est-à-dire
\[
\frac{t^3}2 + (1 - y_0)\,t - x_0 = 0 ,
\]
une cubique en $t$~: une ou trois racines réelles (comptées sans
multiplicité, pour des points génériques). Sur l'axe $x_0 = 0$
elle se factorise en $t\bigl(\tfrac{t^2}2 + 1 - y_0\bigr) = 0$~: la
racine $t = 0$ (l'axe est la normale au sommet), plus
$t = \pm\sqrt{2(y_0 - 1)}$ lorsque $y_0 > 1$. Donc~: une normale
pour $y_0 < 1$, trois pour $y_0 > 1$, et en $y_0 = 1$ la
racine triple marque le point de rebroussement de la développée.
En général une
racine double de la cubique signifie que le point satisfait à la
fois l'équation de la droite et sa dérivée en $t$ --- il est
\emph{sur l'\omterm{pb:b2:curves:1}{enveloppe}}~: la développée est précisément la
frontière entre les régions à une et à trois normales.

\textbf{16.} La réflexion dans le miroir inverse la composante
normale de la direction et conserve la tangentielle~: en
écrivant $u = \langle u, n\rangle n + u_{\mathrm{tan}}$, la
direction réfléchie est $u_{\mathrm{tan}} - \langle u,
n\rangle n = u - 2\langle u, n\rangle n$. Ici $\langle u,
n\rangle = \cos\theta$, donc
\[
v = (1, 0) - 2\cos\theta\,(\cos\theta, \sin\theta)
= (1 - 2\cos^2\theta,\ -2\sin\theta\cos\theta)
= -(\cos2\theta,\ \sin2\theta).
\]

\textbf{17.} Le rayon réfléchi passe par $P_\theta =
(\cos\theta, \sin\theta)$ de direction $(\cos2\theta,
\sin2\theta)$~; un vecteur normal est $(-\sin2\theta,
\cos2\theta)$, donc la droite est
\[
-\sin2\theta\,(x - \cos\theta) + \cos2\theta\,(y -
\sin\theta) = 0,
\]
et la constante est $-\sin2\theta\cos\theta +
\cos2\theta\sin\theta = -\sin\theta$~: en multipliant par $-1$,
$x\sin2\theta - y\cos2\theta = \sin\theta$.

\textbf{18.} $(a, b, c) = (\sin2\theta, -\cos2\theta,
\sin\theta)$~: $a' = 2\cos2\theta$, $b' = 2\sin2\theta$, $c' =
\cos\theta$, $\Delta = 2\sin^22\theta + 2\cos^22\theta = 2$.
Cramer, puis les formules de produit en somme~:
\begin{align*}
x &= \frac{2\sin\theta\sin2\theta +
\cos\theta\cos2\theta}{2}
= \frac{(\cos\theta - \cos3\theta) + \frac12(\cos\theta +
\cos3\theta)}{2} = \frac{3\cos\theta - \cos3\theta}4,\\
y &= \frac{\sin2\theta\cos\theta - 2\cos2\theta\sin\theta}2
= \frac{\frac12(\sin3\theta + \sin\theta) - (\sin3\theta -
\sin\theta)}2 = \frac{3\sin\theta - \sin3\theta}4 :
\end{align*}
la \omterm{pb:b2:curves:1}{néphroïde}, une courbe fermée à deux points de rebroussement.

\textbf{19.} En dérivant et en factorisant avec $\sin3\theta
- \sin\theta = 2\cos2\theta\sin\theta$, $\cos\theta -
\cos3\theta = 2\sin2\theta\sin\theta$~:
\[
E'(\theta) = \tfrac34\bigl(\sin3\theta - \sin\theta,\
\cos\theta - \cos3\theta\bigr)
= \tfrac32\sin\theta\,(\cos2\theta, \sin2\theta),
\]
parallèle à la direction réfléchie de la question 16~: chaque
rayon réfléchi est tangent à la caustique, comme l'exige la
propriété d'\omterm{pb:b2:curves:1}{enveloppe}. $E' = 0$ exactement en $\sin\theta = 0$~:
$E(0) = (\tfrac12, 0)$ et $E(\pi) = (-\tfrac12, 0)$, les deux
points de rebroussement. En posant $y = 0$ dans l'équation de la
droite~: $x\sin2\theta =
\sin\theta$, donc $x = \frac1{2\cos\theta} \to \frac12$ quand
$\theta \to 0$~: les rayons paraxiaux se focalisent à distance
$R/2$ du centre --- la distance focale du miroir sphérique.

\textbf{20.} $\norm{E'(\theta)} = \tfrac32\abs{\sin\theta}$,
donc la \omterm{def:b2:curves:length}{longueur} est $\tfrac32\int_0^{2\pi}\abs{\sin\theta}\,
\dd\theta = \tfrac32\cdot4 = 6$. Près de $\theta = 0$, avec
$\cos k\theta = 1 - \tfrac{k^2\theta^2}2 + O(\theta^4)$ et
$\sin k\theta = k\theta - \tfrac{k^3\theta^3}6 +
O(\theta^5)$~:
\[
x - \tfrac12 = \frac{3\theta^2 + O(\theta^4)}{4}
= \tfrac34\theta^2 + O(\theta^4),
\qquad
y = \frac{4\theta^3 + O(\theta^5)}{4}
= \theta^3 + O(\theta^5) :
\]
$p = 2$, $q = 3$, un point de rebroussement de première espèce
pointant le long de l'axe --- le point brillant de la caustique de
la tasse de café.

\textbf{21.} Paramétrons les points éclairés par
$\Phi(\theta, r) = P_\theta + r\,v_\theta$ (position le long
de chaque rayon réfléchi). Le \omterm{def:b2:linalg:det}{déterminant} jacobien
$\det(\partial_\theta\Phi, \partial_r\Phi) =
\det(P_\theta' + rv_\theta', v_\theta)$ est \omterm{def:b2:affine:subspace}{affine} en $r$ et
s'annule pour exactement un $r = r_*(\theta)$ --- et
$\Phi(\theta, r_*(\theta))$ est le point caractéristique, puisque
là la direction du rayon et la variation de la famille deviennent
dépendantes. Hors de l'\omterm{pb:b2:curves:1}{enveloppe} l'application est un
difféomorphisme local, et un point juste à l'intérieur de la
caustique est atteint par deux rayons voisins
(deux solutions $\theta$), un point à l'extérieur par aucun de
cette partie de la famille~; sur la caustique les deux fusionnent.
L'intensité lumineuse est inversement proportionnelle à la valeur
absolue du jacobien, donc elle explose le long de l'\omterm{pb:b2:curves:1}{enveloppe}~: la
caustique est la
courbe brillante, la plus brillante de toutes au point de
rebroussement, où la dégénérescence est la pire.

\textbf{22.} $x' = 1 - \cos t$, $y' = \sin t$, $x'' = \sin
t$, $y'' = \cos t$, donc $x'y'' - y'x'' = \cos t - 1$ et
$\norm{\gamma'}^2 = 2(1 - \cos t) = 4\sin^2\tfrac t2$~;
par la~\cref{prop:b2:curves:kappaformula},
\[
\kappa(t) = \frac{-(1 - \cos t)}{8\sin^3\tfrac t2}
= \frac{-2\sin^2\tfrac t2}{8\sin^3\tfrac t2}
= -\frac1{4\sin\tfrac t2}
\qquad (0 < t < 2\pi).
\]
Avec $T = (\sin\tfrac t2, \cos\tfrac t2)$ (diviser $\gamma'$
par $2\sin\tfrac t2$) et $N = (-\cos\tfrac t2, \sin\tfrac
t2)$~:
\[
\gamma + \frac1\kappa N
= \gamma - 4\sin\tfrac t2\,\Bigl(-\cos\tfrac t2,\
\sin\tfrac t2\Bigr)
= \bigl(t - \sin t + 2\sin t,\ 1 - \cos t - 2(1 - \cos
t)\bigr),
\]
c'est-à-dire $c(t) = (t + \sin t,\ \cos t - 1)$. En substituant $t = u
+ \pi$~:
\[
c = \bigl(u + \pi - \sin u,\ -\cos u - 1\bigr)
= \bigl((u - \sin u) + \pi,\ (1 - \cos u) - 2\bigr) :
\]
la cycloïde $\gamma(u)$ translatée de $(\pi, -2)$. La
développée d'une cycloïde est une cycloïde congruente, suspendue
un niveau plus bas.

\textbf{23.} $R(t) = 1/\abs{\kappa(t)} = 4\sin\tfrac t2$, donc
$R(\pi) = 4$~: la moitié de la \omterm{def:b2:curves:length}{longueur} d'arche $8$ calculée dans
\cref{exo:b2:curves:1}. La vitesse de la développée $c'(t) = (1 +
\cos t, -\sin t)$ s'annule en $t = \pi$~: le point de
rebroussement est $c(\pi)
= (\pi, -2)$, directement sous le sommet $\gamma(\pi) = (\pi,
2)$, à distance $4 = R(\pi)$, exactement la \omterm{def:b2:curves:length}{longueur} du
rayon osculateur en ce point.

\textbf{24.} D'après \cref{exo:b2:curves:6}, $c'(s) =
-\frac{\kappa'(s)}{\kappa(s)^2}N(s) = R'(s)\,N(s)$, donc
$\norm{c'(s)} = \abs{R'(s)}$ et, pour $R$ monotone,
\[
\int_{s_0}^{s_1}\norm{c'(s)}\,\dd s =
\Bigl|\int_{s_0}^{s_1}R'(s)\,\dd s\Bigr| = \abs{R(s_1) -
R(s_0)} .
\]
Disons que $R$ décroît. Un fil posé le long de la développée
au-delà de $c(s_0)$ et prolongé par le segment de $c(s_0)$ à
$\gamma(s_0)$ (qui est tangent à la développée, par la question 4)
a, une fois déroulé jusqu'à $c(s)$ et tendu, une partie droite de
\omterm{def:b2:curves:length}{longueur} $R(s_0) - \bigl(R(s_0) - R(s)\bigr) = R(s)$
pointant de $c(s)$ le long de la normale --- atterrissant
exactement sur $\gamma(s)$~: l'extrémité libre trace la courbe
originale («~développante~»). Huygens a suspendu un pendule entre
deux joues cycloïdales~: le fil s'enroule sur la développée, donc
la masse décrit une cycloïde --- la tautochrone, dont la période
d'oscillation ne dépend pas de l'amplitude.

\textbf{25.} \omterm{def:b2:curves:arc}{Tangentes} de la parabole~: $\Delta = 1$ et $E' =
(1, t) \neq 0$ partout --- \omterm{pb:b2:curves:1}{enveloppe} lisse (la parabole
elle-même). Échelle qui glisse~: $\Delta = -1$, mais $E' =
\tfrac32\sin 2t\,(-\cos t, \sin t)$ s'annule aux extrémités des
quadrants --- les quatre points de rebroussement de l'\omterm{pb:b2:curves:1}{astroïde}.
Normales de la parabole et de la cycloïde~: $\Delta = \kappa \neq
0$, et
$E' = R'N$ s'annule exactement là où la \omterm{thm:b2:curves:frenet2d}{courbure} est extrémale
--- points de rebroussement des développées en $(0,1)$ et $(\pi,
-2)$. Caustique~:
$\Delta = 2$, et $E' = \tfrac32\sin\theta\,(\cos2\theta,
\sin2\theta)$ s'annule en $\theta = 0, \pi$ --- les deux points de
rebroussement de la \omterm{pb:b2:curves:1}{néphroïde}, les foyers du miroir. Les extrema de
\omterm{thm:b2:curves:frenet2d}{courbure} \emph{doivent} produire des points de rebroussement sur
l'\omterm{pb:b2:curves:1}{enveloppe} des normales, puisque la vitesse de la développée est
$R'N$~: c'est pourquoi la développée de l'ellipse a quatre points
de rebroussement (quatre sommets),
et les familles dégénérées (faisceau, parallèles) sont les cas
où la machine produit un point ou rien du tout.
\end{solution}
