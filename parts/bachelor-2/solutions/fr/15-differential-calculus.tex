\section*{Chapitre \ref{ch:b2:diffcalc} --- Calcul différentiel}

\begin{solution}{exo:b2:diffcalc:1}
$J_f = \begin{pmatrix} 2x & -2y\\ 2y & 2x\end{pmatrix}$, $\det J_f
= 4(x^2 + y^2)$~: inversible en dehors de l'origine. (Ce $f$ est $z
\mapsto z^2$ déguisé en complexe.)

$J_\Phi = \begin{pmatrix} \cos\theta & -r\sin\theta & 0\\
\sin\theta & r\cos\theta & 0\\ 0 & 0 & 1\end{pmatrix}$, $\det = r$~:
inversible pour $r \neq 0$ (coordonnées cylindriques).
\end{solution}

\begin{solution}{exo:b2:diffcalc:2}
Pour $v = (a, b) \neq 0$~: $\frac{f(tv) - 0}{t} = \frac{t^3a^3}{t\cdot
t^2(a^2+b^2)} = \frac{a^3}{a^2 + b^2}$~: toute dérivée directionnelle
existe, de valeur $D_v = \frac{a^3}{a^2+b^2}$. Mais $v
\mapsto D_v$ n'est pas linéaire ($D_{(1,0)} = 1$, $D_{(0,1)} = 0$,
$D_{(1,1)} = \frac12 \neq 1$)~: aucune application linéaire ne peut
produire ces valeurs, donc $f$ n'est pas \omterm{def:b2:diffcalc:differential}{différentiable} en $0$ (la
\omterm{def:b2:diffcalc:differential}{différentielle} devrait être $v \mapsto D_v$).
\end{solution}

\begin{solution}{exo:b2:diffcalc:3}
$f = x^3 + y^3 - 3xy$~: points critiques $(0,0)$ et $(1,1)$ (calcul
de Licence~1). Hessiennes~: $H = \begin{pmatrix} 6x & -3\\ -3 &
6y\end{pmatrix}$. En $(0,0)$~: signes propres mixtes ($\det = -9 < 0$)~:
point col. En $(1,1)$~: $\det = 27 > 0$, trace $> 0$~: définie
positive, minimum local strict --- désormais justifié par le
\cref{thm:b2:diffcalc:extrema} plutôt que décrété.

$g = x^4 + y^4 - 2(x-y)^2$~: $\nabla g = (4x^3 - 4(x - y),\; 4y^3 +
4(x-y))$~; points critiques $(0,0)$, $(\sqrt2, -\sqrt2)$,
$(-\sqrt2, \sqrt2)$ (Licence~1). En $(\pm\sqrt2, \mp\sqrt2)$~: $H =
\begin{pmatrix} 12\cdot2 - 4 & 4\\ 4 & 20\end{pmatrix} =
\begin{pmatrix} 20 & 4\\ 4 & 20\end{pmatrix}$~: définie positive
(à diagonale dominante~; \omterm{def:b2:reduction:eigen}{valeurs propres} $24, 16$)~: minima locaux
stricts. En $(0,0)$~: $H = \begin{pmatrix} -4 & 4\\ 4 & -4\end{pmatrix}$,
singulière semi-définie négative~: le test est muet~; l'étude
directionnelle ($g(x,x) = 2x^4 > 0$, $g(x,-x) = 2x^4 - 8x^2 < 0$
pour $x$ petit) montre un point de type col --- pas d'extremum.
\end{solution}

\begin{solution}{exo:b2:diffcalc:4}
Dériver $t \mapsto f(tx)$ en $t = 1$~: par la règle de la chaîne,
$\langle \nabla f(x), x\rangle$~; par homogénéité la même fonction
vaut $t^pf(x)$, de dérivée $pf(x)$ en $t = 1$~: l'identité d'Euler.

Réciproque~: fixer $x \neq 0$ et poser $\varphi(t) = f(tx) - t^p f(x)$
sur $t > 0$. Alors $\varphi'(t) = \langle\nabla f(tx), x\rangle -
pt^{p-1}f(x) = \frac1t\bigl(\langle \nabla f(tx), tx\rangle -
p\,t^pf(x)\bigr)$. L'hypothèse --- l'identité d'Euler au point
$tx$ --- évalue le crochet à $p\,f(tx) - p\,t^pf(x) =
p\,\varphi(t)$. Donc $\varphi' = \frac{p}{t}\varphi$ avec $\varphi(1)
= 0$~: l'unique solution de cette EDO linéaire est $\varphi \equiv 0$
(unicité de Licence~1), c.-à-d.\ $f(tx) = t^pf(x)$.
\end{solution}

\begin{solution}{exo:b2:diffcalc:5}
En développant $f(x + h) - f(x) = \langle Ax - b, h\rangle +
\frac12\langle Ah, h\rangle$ (symétrie de $A$)~: $\nabla f(x) = Ax -
b$ et $H_f = A$ partout. $f$ est convexe si et seulement si $A$ est
semi-définie positive (le test de la hessienne, global ici puisque
$H$ est constante~: la formule de Taylor du second \omterm{def:b2:structures:generated}{ordre} est exacte).
Si $A$ est définie positive~: l'unique point critique est $x^* =
A^{-1}b$, et $f(x^* + h) - f(x^*) = \frac12\langle Ah, h\rangle \geq
\frac{\lambda_{\min}}{2}\norm h^2 > 0$ pour $h \neq 0$~: minimum
global strict.
\end{solution}

\begin{solution}{exo:b2:diffcalc:6}
Puisque $\nabla g(a) \neq 0$, une dérivée partielle, disons
$\frac{\partial g}{\partial y}(a) \neq 0$~: le théorème des fonctions
implicites (le compagnon du \cref{thm:b2:diffcalc:inverse}) paramétrise
$\{g = 0\}$ près de $a = (a_1, a_2)$ par $\gamma(t) = (t, y(t))$ avec
$y$ $C^1$, $y'(t) = -\frac{\partial_x g}{\partial_y g}(\gamma(t))$
(dériver $g(t, y(t)) = 0$). La fonction à une variable $t
\mapsto f(\gamma(t))$ a un extremum local en $t = a_1$~:
\[
0 = \frac{\dd}{\dd t}f(\gamma(t))\Big|_{a_1}
= \partial_x f(a) + \partial_y f(a)\,y'(a_1)
= \partial_x f(a) - \partial_yf(a)\frac{\partial_x
g(a)}{\partial_y g(a)} :
\]
les vecteurs $\nabla f(a)$ et $\nabla g(a)$ ont des coordonnées
proportionnelles~: $\nabla f(a) = \lambda \nabla g(a)$ avec $\lambda =
\frac{\partial_y f(a)}{\partial_y g(a)}$.

Application~: sur le cercle, $\nabla(xy) = (y, x)$ parallèle à
$(2x, 2y)$ force $y^2 = x^2$~; avec la contrainte, les candidats
sont $\pm\bigl(\tfrac{1}{\sqrt2}, \tfrac{1}{\sqrt2}\bigr)$ (valeur
$\frac12$) et $\pm\bigl(\tfrac{1}{\sqrt2},
-\tfrac{1}{\sqrt2}\bigr)$ (valeur $-\frac12$)~: max $\frac12$, min
$-\frac12$ (atteints~: le cercle est \omterm{def:b2:metric:compact}{compact}).
\end{solution}

\begin{solution}{exo:b2:diffcalc:7}
$\nabla f = (2x - y + 1,\; 2y - x - 1) = 0$~: en résolvant, $x =
-\frac13$, $y = \frac13$. Hessienne $\begin{pmatrix} 2 & -1\\ -1 &
2\end{pmatrix}$, définie positive~: minimum global de la quadratique
$f$, valeur $f\bigl(-\frac13, \frac13\bigr) = -\frac13$.

Sur le triangle $T$~: le point critique intérieur $(-\frac13,
\frac13) \notin T$ ($x$ négatif). Arêtes~: sur $y = 0$, $x \in
\intcc{0}{1}$~: $f = x^2 + x$, croissante~: extrêmes $0$ et $2$. Sur
$x = 0$~: $f = y^2 - y$, minimum $-\frac14$ en $y = \frac12$, valeurs
$0$ et $0$ aux extrémités. Sur $x + y = 1$~: substituer $y = 1 - x$,
$f = x^2 + (1-x)^2 - x(1-x) + x - (1-x) = 3x^2 - x$~; sur
$\intcc{0}{1}$~: minimum $-\frac{1}{12}$ en $x = \frac16$, valeurs
$0$ (en $x=0$) et $2$ (en $x=1$). Sommets~: $f(0,0) = 0$, $f(1,0)
= 2$, $f(0,1) = 0$. Global sur $T$~: minimum $-\frac14$ en $(0,
\frac12)$, maximum $2$ en $(1, 0)$.
\end{solution}

\begin{solution}{exo:b2:diffcalc:8}
$J_f = \begin{pmatrix} 1 & 2y\\ 2x & 1\end{pmatrix}$, $\det J_f =
1 - 4xy$. En $0$~: $\det = 1 \neq 0$~: difféomorphisme local
(\cref{thm:b2:diffcalc:inverse}), avec
\[
\dd(f^{-1})_{(0,0)} = (J_f(0))^{-1} = I_2 .
\]
Inversibilité sur une boule~: il faut $4\abs{xy} < 1$ partout~; sur
$\norm{(x,y)}_2 < r$, $\abs{xy} \leq \frac{x^2 + y^2}{2} <
\frac{r^2}{2}$, donc $r = \frac{1}{\sqrt2}$ convient~; et c'est le
plus grand~: en $(x, y) = \bigl(\tfrac12, \tfrac12\bigr)$, de \omterm{def:b2:nvs:norm}{norme}
$\frac{1}{\sqrt2}$, $\det J_f = 0$.
\end{solution}

\begin{solution}{exo:b2:diffcalc:9}
$f(t) = (\cos t, \sin t)$~: $f(0) = f(2\pi) = (1, 0)$, pourtant $f'(t) =
(-\sin t, \cos t)$ est de \omterm{def:b2:nvs:norm}{norme} $1$, jamais nulle~: aucun point où la
dérivée s'annule --- Rolle n'a pas d'analogue vectoriel. L'inégalité
des accroissements finis tient largement~: $\norm{f(2\pi) - f(0)} = 0 \leq
2\pi \cdot \sup\norm{f'} = 2\pi$.
\end{solution}

\begin{solution}{exo:b2:diffcalc:10}
$f(x+h) - f(x) = 2\langle x, h\rangle + \norm h^2$~: $\dd f_x =
2\langle x, \cdot\rangle$, $\nabla f(x) = 2x$, hessienne $2I$
(constante). Pour $g$~:
\[
g(x + h) - g(x) = \langle Ax, h\rangle + \langle Ah, x\rangle
+ \langle Ah, h\rangle
= \bigl\langle (A + A^{\mathsf T})x,\ h\bigr\rangle +
O(\norm h^2),
\]
donc $\nabla g(x) = (A + A^{\mathsf T})x$ et $H_g = A +
A^{\mathsf T}$, constante. D'après l'\cref{exo:b2:diffcalc:11}, $g$ est
convexe si et seulement si $A + A^{\mathsf T}$ est semi-définie
positive --- seule la partie \omterm{def:b2:quadratic:adjoint}{symétrique} de $A$ compte, comme en effet
$g(x) = \langle \frac{A + A^{\mathsf T}}2 x, x\rangle$.
\end{solution}

\begin{solution}{exo:b2:diffcalc:11}
$f$ est convexe si et seulement si sa restriction à tout segment est
convexe, c.-à-d.\ si et seulement si tout $\varphi(t) = f(a + tv)$ est
convexe. Par la règle de la chaîne $\varphi''(t) = \langle H_{a+tv}\,v,\
v\rangle$.

Si toutes les hessiennes sont semi-définies positives~: $\varphi'' \geq 0$,
donc chaque $\varphi$ est convexe (volume de Licence~1) et $f$ est
convexe. Réciproquement si $f$ est convexe, chaque $\varphi$ est
convexe, donc $\varphi''(0) \geq 0$~: $\langle H_a v, v\rangle \geq 0$
pour tout $a$ et toute direction $v$~: toutes les hessiennes sont
semi-définies positives.
\end{solution}

\begin{solution}{exo:b2:diffcalc:12}
\begin{enumerate}
  \item Par l'inégalité des accroissements finis
        (\cref{thm:b2:diffcalc:mvi}) appliquée à $g$~:
        $\norm{g(x) - g(y)} \leq k\norm{x-y}$, donc
        \[
        \norm{f(x) - f(y)} \geq \norm{x - y} - \norm{g(x) -
        g(y)} \geq (1 - k)\norm{x - y} :
        \]
        $f$ est injective et $f^{-1}$ (défini sur l'image) est
        $\frac{1}{1-k}$-lipschitzien.
  \item Fixer $y$~; $T(x) = y - g(x)$ vérifie $\norm{T(x) -
        T(x')} = \norm{g(x') - g(x)} \leq k\norm{x - x'}$~: une
        contraction de l'espace \omterm{def:b2:metric:complete}{complet} $\R^n$. Banach
        (\cref{thm:b2:metric:banach}) donne un point fixe $x^*
        = y - g(x^*)$, c.-à-d.\ $f(x^*) = y$~: $f$ est surjective.
  \item $f$ est ainsi une bijection de $\R^n$ de réciproque
        $\frac{1}{1-k}$-lipschitzienne~: un théorème d'inversion
        globale, où la petitesse de $\dd g$ partout remplace
        l'hypothèse d'inversibilité locale du
        \cref{thm:b2:diffcalc:inverse}.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b2:diffcalc:1}
\textbf{1.} $\norm{ABx} \leq \vertiii A\norm{Bx} \leq
\vertiii A\vertiii B\norm x$~: prendre le sup sur $\norm x = 1$.
Les produits de matrices, $\det$ et $\operatorname{tr}$ sont des
fonctions polynomiales des coefficients, donc \omterm{def:b2:metric:continuity}{continues} ($\mathcal M_n(\R)
\simeq \R^{n^2}$, toutes les \omterm{def:b2:nvs:equivalent}{normes équivalentes}~:
\cref{thm:b2:nvs:finitedim}).

\textbf{2.} $\sum\vertiii{X^k} \leq \sum\vertiii X^k <
\infty$~: la série \omterm{def:b2:integration:improper}{converge} \omterm{def:b2:series:def}{absolument}, donc \omterm{def:b2:integration:improper}{converge}
(\cref{thm:b2:nvs:absoluteconvergence}). De $(I -
X)\sum_{k\leq N}X^k = I - X^{N+1} \to I$~: la somme est $(I -
X)^{-1}$. \omterm{def:b2:nvs:norm}{Norme}~: $\leq \sum\vertiii X^k = \frac{1}{1 -
\vertiii X}$~; et
\[
(I - X)^{-1} - I - X = \sum_{k\geq2}X^k = X^2(I - X)^{-1},
\qquad
\vertiii{X^2(I-X)^{-1}} \leq
\frac{\vertiii X^2}{1 - \vertiii X} = O(\vertiii X^2).
\]

\textbf{3.} $A + H = A(I + A^{-1}H)$ avec $\vertiii{A^{-1}H}
\leq \vertiii{A^{-1}}\vertiii H < 1$~: inversible par la question 2~:
la boule ouverte de rayon $\vertiii{A^{-1}}^{-1}$ autour de $A$ est
incluse dans $GL_n(\R)$. Densité~: $\det(A + \varepsilon I)$ est un
polynôme de degré $n$ en $\varepsilon$ de coefficient dominant $1$~:
il a un nombre fini de racines, donc il existe $\varepsilon_k \to 0$
tels que $A + \varepsilon_kI$ soit inversible, convergeant vers $A$.

\textbf{4.} Pour $\vertiii H < \frac{1}{2\vertiii{A^{-1}}}$~:
\[
(A + H)^{-1} - A^{-1}
= \bigl[(I + A^{-1}H)^{-1} - I\bigr]A^{-1},
\]
de \omterm{def:b2:nvs:norm}{norme} au plus $\frac{\vertiii{A^{-1}H}}{1 -
\vertiii{A^{-1}H}}\,\vertiii{A^{-1}} \leq
2\vertiii{A^{-1}}^2\vertiii H \to 0$~: $\Phi$ est \omterm{def:b2:metric:continuity}{continue} en
tout $A \in GL_n(\R)$.

\textbf{5.} $(A+H)^2 = A^2 + AH + HA + H^2$~: l'application $H
\mapsto AH + HA$ est linéaire et l'erreur $H^2$ est un
$O(\vertiii H^2)$. En développant $(A + H)^k$ et en triant par le
nombre de facteurs $H$~: la partie linéaire est
$\sum_{i=0}^{k-1}A^iHA^{k-1-i}$, et les termes à $\geq 2$
facteurs $H$ sont majorés par $\binom k2$ produits de \omterm{def:b2:nvs:norm}{norme}
$\leq \vertiii A^{k-2}\vertiii H^2$~: $O(\vertiii H^2)$.
On ne peut pas réduire la somme à $kA^{k-1}H$ car $H$ et $A$
ne commutent pas en général --- la somme est la bonne dérivée
non commutative.

\textbf{6.} Pour $H$ petit~:
\[
(A+H)^{-1} = (I + A^{-1}H)^{-1}A^{-1}
= \bigl(I - A^{-1}H + O(\vertiii H^2)\bigr)A^{-1}
= A^{-1} - A^{-1}HA^{-1} + O(\vertiii H^2) :
\]
$\dd\Phi_A(H) = -A^{-1}HA^{-1}$, linéaire en $H$. Pour $n = 1$~:
$\dd(1/a)(h) = -h/a^2$, la dérivée familière.

\textbf{7.} Règle de la chaîne le long de la courbe~:
$\bigl(A(t)^{-1}\bigr)' = \dd\Phi_{A(t)}(A'(t)) =
-A(t)^{-1}A'(t)A(t)^{-1}$. En $A(t) = I + tB$, $t = 0$~: $(I +
tB)^{-1} = I - tB + O(t^2)$.

\textbf{8.} Par la question 5 et l'invariance \omterm{def:b2:structures:generated}{cyclique} de la trace~:
\[
\dd f_A(H) = \operatorname{tr}\Bigl(\sum_{i=0}^{k-1}
A^iHA^{k-1-i}\Bigr) = k\operatorname{tr}\bigl(A^{k-1}H\bigr) .
\]
Contre le produit de Frobenius, $\dd f_A(H) =
\operatorname{tr}\bigl((\nabla f)^{\mathsf T}H\bigr)$ requiert
$(\nabla f)^{\mathsf T} = kA^{k-1}$~: $\nabla f(A) =
k\,(A^{k-1})^{\mathsf T}$.

\textbf{9.} $f(A + H) - f(A) = 2\operatorname{tr}
(A^{\mathsf T}H) + \operatorname{tr}(H^{\mathsf T}H)$~: la
\omterm{def:b2:diffcalc:differential}{différentielle} est $H \mapsto 2\operatorname{tr}(A^{\mathsf T}H) =
2\langle A, H\rangle$, donc $\nabla f(A) = 2A$~; le terme du second
\omterm{def:b2:structures:generated}{ordre} est exactement $\norm H_F^2$~: la hessienne est le double de
la \omterm{def:b2:quadratic:def}{forme quadratique} identité, définie positive et constante, donc
$\norm\cdot_F^2$ est strictement convexe (la formule de Taylor est
exacte ici).

\textbf{10.} Par multilinéarité en les colonnes, $\det(I + H) =
\sum_{S\subseteq\{1,\dots,n\}}\det(M_S)$ où $M_S$ a pour colonne
$h_j$ pour $j \in S$ et $e_j$ sinon. $S = \varnothing$
donne $1$~; $S = \{j\}$ donne le \omterm{def:b2:linalg:det}{déterminant} de $I$ avec la colonne
$j$ remplacée par $h_j$, à savoir son $j$-ème coefficient $h_{jj}$,
de somme $\operatorname{tr}H$~; chaque terme à $\abs S \geq 2$
est un \omterm{def:b2:linalg:det}{déterminant} ayant au moins deux colonnes de taille
$O(\vertiii H)$, donc $O(\vertiii H^2)$ (les applications
multilinéaires sur un espace de dimension finie sont bornées), et
il y en a un nombre fini. Donc $\det(I + H) = 1 + \operatorname{tr}H +
O(\vertiii H^2)$~: $\dd(\det)_I = \operatorname{tr}$.

\textbf{11.} $\det(A + H) = \det A\,\det(I + A^{-1}H) =
\det A\,\bigl(1 + \operatorname{tr}(A^{-1}H) +
O(\vertiii H^2)\bigr)$~: la \omterm{def:b2:diffcalc:differential}{différentielle} est $H \mapsto
\det(A)\operatorname{tr}(A^{-1}H)$.

\textbf{12.} Développement de Laplace selon la ligne $i$~: $\det A =
\sum_j a_{ij}C_{ij}$, et les cofacteurs $C_{ij}$ ne font pas
intervenir la ligne $i$~: $\frac{\partial\det}{\partial a_{ij}} = C_{ij}$.
D'où
\[
\dd(\det)_A(H) = \sum_{i,j}C_{ij}h_{ij}
= \operatorname{tr}\bigl(\operatorname{com}(A)^{\mathsf T}
H\bigr)
= \operatorname{tr}\bigl(\operatorname{adj}(A)H\bigr),
\qquad
\nabla(\det)(A) = \operatorname{com}(A) .
\]
Pour $A$ inversible, $\operatorname{adj}A = \det(A)A^{-1}$
retrouve la question 11. Le long d'une courbe $C^1$, la règle de la
chaîne s'écrit $(\det A(t))' = \operatorname{tr}(\operatorname{adj}
(A(t))\,A'(t))$~: la formule de Jacobi.

\textbf{13.} Avec $A' = MA$ et
$\operatorname{adj}(A)A = \det(A)I$~:
\[
(\det A)' = \operatorname{tr}\bigl(\operatorname{adj}(A)MA
\bigr)
= \operatorname{tr}\bigl(A\operatorname{adj}(A)M\bigr)
= \det A\;\operatorname{tr}M
\]
(cyclicité~; $A\operatorname{adj}A = \det(A) I$ également). L'EDO
scalaire linéaire $y' = \operatorname{tr}(M(t))\,y$ a pour
unique solution $y(t) = y(0)\exp\bigl(\int_0^t
\operatorname{tr}M\bigr)$ (Licence~1)~: la formule de Liouville.

\textbf{14.} En $A \in SL_n(\R)$, prendre $H = \frac1nA$~:
$\dd(\det)_A\bigl(\tfrac1nA\bigr) = \frac1n\det(A)
\operatorname{tr}(A^{-1}A) = \frac1n\cdot1\cdot n = 1 \neq 0$~:
la \omterm{def:b2:diffcalc:differential}{différentielle} est une forme linéaire non nulle, donc surjective
sur $\R$ en tout point de l'ensemble de niveau~: $SL_n(\R)$ est un
ensemble de niveau lisse (de dimension $n^2 - 1$).

\textbf{15.} $\sum_k\vertiii{A^k/k!} \leq
\sum\vertiii A^k/k! = \eu^{\vertiii A}$~: convergence absolue
(l'argument de complétude de la question 2), avec \omterm{def:b2:funcseq:series}{convergence
normale} sur toute boule $\vertiii A \leq R$ (majoration $R^k/k!$
indépendante de $A$). Chaque somme partielle est \omterm{def:b2:metric:continuity}{continue}
(polynomiale)~; la limite uniforme sur les boules est \omterm{def:b2:metric:continuity}{continue}~: $A
\mapsto \eu^A$ est \omterm{def:b2:metric:continuity}{continue}, avec $\vertiii{\eu^A} \leq
\eu^{\vertiii A}$.

\textbf{16.} Les deux séries convergent \omterm{def:b2:series:def}{absolument}, donc le \omterm{thm:b2:series:fubini}{produit
de Cauchy} est légitime (\cref{thm:b2:series:fubini})~:
\[
\eu^A\eu^B = \sum_{n\geq0}\frac{1}{n!}\sum_{k=0}^n\binom
nkA^kB^{n-k}
= \sum_{n\geq0}\frac{(A+B)^n}{n!} = \eu^{A+B},
\]
l'identité binomiale exigeant $AB = BA$. Avec $B = -A$~:
$\eu^A\eu^{-A} = \eu^0 = I$~: tout $\eu^A \in GL_n(\R)$.

\textbf{17.} La série $\sum t^kA^k/k!$ et sa série dérivée
$\sum t^{k-1}A^k/(k-1)! = A\sum t^{k-1}A^{k-1}/(k-1)!$ convergent
\omterm{def:b2:funcseq:series}{normalement} sur tout segment $\abs t \leq T$ (majorations
$T^k\vertiii A^k/k!$)~: le théorème de dérivation des séries
(\cref{thm:b2:funcseq:seriestransfer}, appliqué coefficient par
coefficient) donne $\frac{\dd}{\dd t}\eu^{tA} = A\eu^{tA}$~; en
factorisant $A$ à droite on obtient plutôt $\eu^{tA}A$. En itérant~:
$C^\infty$.

\textbf{18.} $A(t) = \eu^{tA}$ satisfait $A'(t) = A\,A(t)$~:
la formule de Liouville (question 13) avec $M = A$ constant donne
$\det\eu^{tA} = \eu^{t\operatorname{tr}A}$ (valeur $1$ en $t =
0$)~; en $t = 1$, $\det\eu^A = \eu^{\operatorname{tr}A}$.
Vérifications~: $n = 1$ est l'exponentielle elle-même~; une $A$
nilpotente a $\operatorname{tr}A = 0$ et $\eu^A$ unipotente de
\omterm{def:b2:linalg:det}{déterminant} $1$~; $\operatorname{tr}A = 0$ donne $\det\eu^A = 1$~:
les matrices de trace nulle sont envoyées dans $SL_n(\R)$.

\textbf{19.} $\eu^H - I - H = \sum_{k\geq2}H^k/k!$, de \omterm{def:b2:nvs:norm}{norme}
$\leq \vertiii H^2\eu^{\vertiii H} = O(\vertiii H^2)$~:
$\dd(\exp)_0 = \mathrm{id}$, inversible. De plus $\exp$ est
$C^1$~: par la question 5, le candidat différentiel $H \mapsto
\sum_k\frac1{k!}\sum_iA^iHA^{k-1-i}$ est une série \omterm{def:b2:funcseq:series}{normalement}
convergente d'applications linéaires dépendant continûment de $A$
(majorations $\vertiii A^{k-1}/(k-1)!$ sur les boules), donc les
dérivées partielles existent et sont \omterm{def:b2:metric:continuity}{continues}
(\cref{thm:b2:diffcalc:c1} et le théorème de transfert des séries).
Le théorème d'inversion locale
(\cref{thm:b2:diffcalc:inverse}) s'applique en $0$~: $\exp$ est un
difféomorphisme $C^1$ d'un voisinage de $0$ sur un voisinage de $I$
--- les matrices proches de $I$ ont des logarithmes.

\textbf{20.} Pour $S = PDP^{\mathsf T}$ \omterm{def:b2:quadratic:adjoint}{symétrique} (théorème
spectral)~: $\eu^S = P\eu^DP^{\mathsf T}$ est \omterm{def:b2:quadratic:adjoint}{symétrique} de
\omterm{def:b2:reduction:eigen}{valeurs propres} $\eu^{\lambda_i} > 0$~: définie positive.
Surjectivité~: une $Q =
P\operatorname{diag}(\mu_i)P^{\mathsf T}$ définie positive ($\mu_i > 0$)
est $\eu^S$ pour $S = P\operatorname{diag}(\ln\mu_i)P^{\mathsf T}$.
Injectivité~: $\eu^S$ détermine ses sous-espaces propres, qui sont
exactement ceux de $S$ (sur chaque sous-espace propre de $S$ pour
$\lambda$, $\eu^S$ agit comme $\eu^\lambda$~; des $\lambda$ distincts
donnent des $\eu^\lambda$ distincts), et prendre le $\ln$ des \omterm{def:b2:reduction:eigen}{valeurs
propres} retrouve $S$. Donc $\exp$ est une bijection des matrices
\omterm{def:b2:quadratic:adjoint}{symétriques} sur les matrices définies positives.

\textbf{21.} $F(A + H) = A^{\mathsf T}A + A^{\mathsf T}H +
H^{\mathsf T}A + H^{\mathsf T}H$~: $\dd F_A(H) =
A^{\mathsf T}H + H^{\mathsf T}A$ (à valeurs dans $S_n$~; erreur
$O(\vertiii H^2)$). En $A \in O_n$ et pour $S \in S_n$, le
choix $H = \frac12AS$ donne
\[
A^{\mathsf T}\cdot\tfrac12AS + \tfrac12(AS)^{\mathsf T}A
= \tfrac12 S + \tfrac12 S^{\mathsf T} = S :
\]
surjective. $O_n = F^{-1}(I)$ est un ensemble de niveau lisse de
dimension $n^2 - \dim S_n = \frac{n(n-1)}2$.

\textbf{22.} En dérivant $A(t)^{\mathsf T}A(t) = I$ en $t
= 0$ (avec $A(0) = I$)~: $A'(0)^{\mathsf T} + A'(0) = 0$~:
antisymétrique. Réciproquement, pour $K^{\mathsf T} = -K$~:
$(\eu^{tK})^{\mathsf T}\eu^{tK} = \eu^{tK^{\mathsf T}}
\eu^{tK} = \eu^{-tK}\eu^{tK} = I$ (transposer la série
terme à terme~; les exposants commutent)~: la courbe reste dans
$O_n$, de vitesse $K$ en $t = 0$. Espace tangent en $I$ $=$ les
matrices antisymétriques, de la dimension attendue
$\frac{n(n-1)}2$.

\textbf{23.} $\operatorname{tr}K = 0$ pour $K$ antisymétrique,
donc $\det\eu^K = \eu^0 = 1$ (question 18)~: l'exponentielle
atterrit dans $SO_n$. Pour $n = 2$~: $J^2 = -I$, donc $J^{2m} =
(-1)^mI$, $J^{2m+1} = (-1)^mJ$, et
\[
\eu^{\theta J}
= \Bigl(\sum_m\frac{(-1)^m\theta^{2m}}{(2m)!}\Bigr)I
+ \Bigl(\sum_m\frac{(-1)^m\theta^{2m+1}}{(2m+1)!}\Bigr)J
= \cos\theta\,I + \sin\theta\,J ,
\]
la rotation d'angle $\theta$~: les séries du sinus et du cosinus
vivent à l'intérieur de l'\omterm{ex:b2:nvs:matrixexp}{exponentielle de matrice}.

\textbf{24.} Pour $A, B$ inversibles~: $\operatorname{adj}(AB) =
\det(AB)(AB)^{-1} = \det(B)\det(A)B^{-1}A^{-1} =
\operatorname{adj}(B)\operatorname{adj}(A)$. Les deux membres de
l'identité sont des applications polynomiales (donc \omterm{def:b2:metric:continuity}{continues}) des
coefficients de $(A, B)$~; ils coïncident sur la partie dense
$GL_n\times GL_n$ de $\mathcal M_n\times\mathcal M_n$ (question 3~:
approcher chaque facteur), donc ils coïncident partout.

\textbf{25.} (i) La Partie I reposait sur la complétude des espaces
normés de dimension finie (les séries \omterm{def:b2:series:def}{absolument} convergentes
convergent), la Partie IV sur la même chose plus le théorème
spectral pour la question 20, le logarithme local de la Partie V sur
le théorème d'inversion locale. (ii) Le chapitre sur les équations
\omterm{def:b2:diffcalc:differential}{différentielles} s'appuie sur la formule de Liouville (question 13)
pour le wronskien des systèmes linéaires, et sur
$\frac{\dd}{\dd t}\eu^{tA} = A\eu^{tA}$ (question 17), qui est
l'énoncé que $\eu^{tA}$ résout $X' = AX$. (iii)
$\dd(\det)_I = \operatorname{tr}$ dit que la trace est le taux
infinitésimal de variation de volume, et $\det\eu^A =
\eu^{\operatorname{tr}A}$ intègre globalement cet énoncé.
(iv) Le volume de Licence~3 nomme les structures~: $O_n$ et
$SL_n(\R)$ sont des groupes de Lie, leurs espaces tangents en $I$
(matrices antisymétriques et de trace nulle) sont des \omterm{def:b2:structures:algebra}{algèbres} de
Lie, et $\exp$ est le pont entre eux.
