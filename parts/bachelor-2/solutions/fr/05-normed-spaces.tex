\section*{Chapitre \ref{ch:b2:nvs} --- Espaces vectoriels normés}

\begin{solution}{exo:b2:nvs:1}
Boules unités~: un losange ($\norm\cdot_1$), un disque ($\norm\cdot_2$),
un carré ($\norm\cdot_\infty$), emboîtés dans cet \omterm{def:b2:structures:generated}{ordre}. Inégalités~:
$\norm x_\infty \leq \norm x_2$ (un carré est au plus la somme)~;
$\norm x_2 \leq \norm x_1$ (en élevant au carré~: $x_1^2 + x_2^2 \leq (\abs{x_1}
+ \abs{x_2})^2$)~; $\norm x_1 \leq 2\norm x_\infty$ (deux termes, chacun
$\leq \max$). Optimalité~: $(1, 0)$ rend les deux premières des égalités~;
$(1, 1)$ rend $\norm x_1 = 2\norm x_\infty$ et montre aussi que
$\norm x_2 = \sqrt2 \norm x_\infty$ et $\norm x_1 = \sqrt2 \norm
x_2$ sont les rapports extrêmes dans l'autre sens.
\end{solution}

\begin{solution}{exo:b2:nvs:2}
Axiomes de \omterm{def:b2:nvs:norm}{norme}~: l'homogénéité et l'inégalité triangulaire sont héritées
terme à terme~; séparation~: $N(f) = 0$ force $f' = 0$ (donc $f$ constante)
et $f(0) = 0$~: $f = 0$. C'est une \omterm{def:b2:nvs:norm}{norme}.

Comparaison~: $\norm f_\infty \leq N(f)$, car $\abs{f(x)} \leq
\abs{f(0)} + \abs{\int_0^x f'} \leq \abs{f(0)} +
\norm{f'}_\infty$. La réciproque échoue~: prendre $f_n(x) = \frac1n
\sin(nx)$~: alors $\norm{f_n}_\infty \leq \frac1n \to 0$ tandis que
$N(f_n) = 0 + \norm{\cos(nx)}_\infty = 1$. Aucune constante $C$ ne donne
$N \leq C\norm\cdot_\infty$.
\end{solution}

\begin{solution}{exo:b2:nvs:3}
$\abs{u(f)} \leq \norm f_\infty \int_0^1 \eu^t\,\dd t = (\eu -
1)\norm f_\infty$, avec égalité pour $f \equiv 1$~: $\vertiii u =
\eu - 1$.

Décalage~: $\norm{S(x)}_\infty = \max(\abs{x_2}, \dots, \abs{x_n}) \leq
\norm x_\infty$, égalité en $x = e_2$~: $\vertiii S = 1$ (pour $n
\geq 2$).
\end{solution}

\begin{solution}{exo:b2:nvs:4}
Borne supérieure~: pour $\norm x_\infty \leq 1$,
\[
\abs{(Ax)_i} = \Bigl|\sum_j a_{ij}x_j\Bigr| \leq \sum_j
\abs{a_{ij}},
\]
donc $\norm{Ax}_\infty \leq \max_i \sum_j \abs{a_{ij}}$. Atteinte~: soit
$i_0$ réalisant le maximum et prenons $x_j = \operatorname{sign}(a_{i_0
j})$ (coefficients de module $1$)~: alors $(Ax)_{i_0} = \sum_j \abs{a_{i_0
j}}$. D'où la formule. Pour la matrice donnée~: sommes de lignes $3$ et
$4$~: $\vertiii A_\infty = 4$.
\end{solution}

\begin{solution}{exo:b2:nvs:5}
\emph{Série géométrique~:} pour $\vertiii B < 1$, la série $\sum
B^k$ converge absolument dans l'\omterm{def:b2:nvs:banach}{espace de Banach} $\mathcal{M}_n(K)$
(\cref{thm:b2:nvs:absoluteconvergence}, $\vertiii{B^k} \leq
\vertiii B^k$), et
\[
(I - B)\sum_{k=0}^{K} B^k = I - B^{K+1} \longrightarrow I :
\]
par continuité du produit
(\cref{prop:b2:nvs:bilinear}), $(I - B)\sum_{k\geq0} B^k = I$, donc
$I - B$ est inversible d'inverse la somme (et
$\vertiii{(I-B)^{-1}} \leq \frac{1}{1 - \vertiii B}$).

\emph{Ouverture~:} pour $A$ inversible et $\vertiii H <
\frac{1}{\vertiii{A^{-1}}}$~: $A + H = A(I + A^{-1}H)$ avec
$\vertiii{A^{-1}H} \leq \vertiii{A^{-1}}\vertiii H < 1$~: inversible.
Donc une boule autour de $A$ reste dans $GL_n$.

\emph{Continuité de l'inversion~:} avec $B = -A^{-1}H$,
\[
(A + H)^{-1} - A^{-1} = \bigl((I - B)^{-1} - I\bigr)A^{-1}
= \Bigl(\sum_{k \geq 1} B^k\Bigr) A^{-1},
\]
de \omterm{def:b2:nvs:norm}{norme} $\leq \frac{\vertiii B}{1 - \vertiii B}\vertiii{A^{-1}}
\to 0$ lorsque $H \to 0$.
\end{solution}

\begin{solution}{exo:b2:nvs:6}
\omterm{def:b2:metric:continuity}{Continue} $\Rightarrow$ noyau fermé~: image réciproque du fermé
$\{0\}$ (\cref{thm:b2:metric:globalcontinuity}).

Réciproquement, supposons $\ker\varphi$ fermé et $\varphi \neq 0$.
Choisissons $a$ tel que $\varphi(a) = 1$~; puisque $a \notin \ker\varphi$
et que le noyau est fermé, une boule $B(a, r)$ l'évite. Soit maintenant
$h \in E$ avec $\varphi(h) \neq 0$~: le vecteur $a - \frac{h}{\varphi(h)}$
appartient à $\ker\varphi$, donc est hors de $B(a, r)$~:
\[
\Bigl\Vert \frac{h}{\varphi(h)} \Bigr\Vert \geq r
\quad\Longrightarrow\quad
\abs{\varphi(h)} \leq \frac{\norm h}{r},
\]
inégalité trivialement vraie lorsque $\varphi(h) = 0$ aussi~: la borne
(4) du~\cref{thm:b2:nvs:continuouslinear}~: \omterm{def:b2:metric:continuity}{continue}.
\end{solution}

\begin{solution}{exo:b2:nvs:7}
Soit $f_n$ nulle sur $\intcc{0}{\frac12 - \frac1n}$, affine jusqu'à la
valeur $1$ en $\frac12$, puis égale à $1$ sur $\intcc{\frac12}{1}$. Pour $m
\geq n$, $f_m - f_n$ est à support dans un intervalle de longueur
$\frac1n$ à valeurs dans $\intcc{-1}{1}$~: $\norm{f_m - f_n}_1 \leq
\frac1n$~: de Cauchy.

Supposons $f_n \to f$ en $\norm\cdot_1$ avec $f$ \omterm{def:b2:metric:continuity}{continue}. Sur
$\intcc{0}{\frac12 - \delta}$ ($\delta$ fixé)~: $\int \abs{f} =
\int\abs{f - f_n} \leq \norm{f - f_n}_1 \to 0$ pour $n >
\frac1\delta$, donc $\int_0^{1/2 - \delta}\abs f = 0$, et par stricte
positivité $f = 0$ là --- pour tout $\delta$~: $f = 0$ sur
$\intoo{0}{\frac12}$. De même $f = 1$ sur $\intcc{\frac12}{1}$
(les $f_n$ y valent toutes $1$). Par continuité en $\frac12$~: $0 =
1$, absurde. Aucune limite n'existe~: l'espace n'est pas \omterm{def:b2:metric:complete}{complet}.
\end{solution}

\begin{solution}{exo:b2:nvs:8}
Bien définie et \omterm{def:b2:metric:continuity}{continue}~: $\abs{\varphi(f)} \leq \sum 2^{-n}
\norm f_\infty = \norm f_\infty$, donc $\varphi$ est une forme linéaire
avec $\vertiii\varphi \leq 1$ (la série converge absolument pour
chaque $f$).

\omterm{def:b2:nvs:norm}{Norme} $1$~: fixons $K$~; les points $1, \frac12, \dots, \frac1K$ sont
deux à deux distincts, donc il existe une fonction \omterm{def:b2:metric:continuity}{continue} $f_K$, $\norm{f_K}_\infty
\leq 1$, avec $f_K(\frac1n) = (-1)^n$ pour $n \leq K$ (interpolation
affine par morceaux, constante près de $0$). Alors
\[
\varphi(f_K) \geq \sum_{n=1}^{K} 2^{-n} - \sum_{n > K} 2^{-n}
= 1 - 2^{-K+1} \xrightarrow[K \to \infty]{} 1 .
\]

Non atteinte~: si $\norm f_\infty \leq 1$ et $\varphi(f) = 1$, chaque
terme doit apporter son maximum~: $(-1)^n f(\frac1n) = 1$ pour tout
$n$ (sinon le déficit strict d'un terme ne peut être compensé,
tous les termes étant $\leq 2^{-n}$). Donc $f(\frac1n) = (-1)^n$~; mais
$\frac1n \to 0$ et $f$ est \omterm{def:b2:metric:continuity}{continue} en $0$, ce qui force la
convergence contradictoire de $(-1)^n$. Ainsi le supremum n'est pas un
maximum --- impossible en dimension finie, où la boule unité fermée
est compacte.
\end{solution}

\begin{solution}{exo:b2:nvs:9}
\emph{Suites bornées~:} une suite bornée se trouve dans une boule
fermée $\overline B(0, R) = R\,\overline B(0,1)$, compacte (image de
la boule unité compacte par l'homéomorphisme $x \mapsto Rx$)~:
y extraire.

\emph{Formes linéaires~:} si $\dim E < \infty$, toute application linéaire
issue de $E$ est \omterm{def:b2:metric:continuity}{continue} (\cref{thm:b2:nvs:finitedim}). Réciproquement,
supposons $\dim E = \infty$ (ce qui, d'après le~\cref{thm:b2:nvs:riesz},
est en fait exclu par l'hypothèse de compacité --- l'objet de
cette question est l'implication entre les deux propriétés dans les
espaces normés généraux)~: choisir une suite normalisée libre $(e_n)$,
la compléter en une base algébrique (admis), et poser $\varphi(e_n) = n$,
$\varphi = 0$ sur les autres vecteurs de base, prolongée linéairement.
Alors $\abs{\varphi(e_n)} = n$ avec $\norm{e_n} = 1$~: non bornée sur la
boule unité, discontinue. Ainsi «~toutes les formes \omterm{def:b2:metric:continuity}{continues}~» force la
dimension finie.
\end{solution}

\begin{solution}{exo:b2:nvs:10}
Cauchy--Schwarz avec la fonction constante $1$~: $\norm f_1 =
\int_0^1 \abs f\cdot 1 \leq \bigl(\int_0^1
f^2\bigr)^{1/2}\bigl(\int_0^1 1\bigr)^{1/2} = \norm f_2$. Et
$\norm f_2^2 = \int f^2 \leq \norm f_\infty^2$. Pour $f_n(x) =
x^n$~:
\[
\norm{f_n}_1 = \frac1{n+1}, \qquad
\norm{f_n}_2 = \frac1{\sqrt{2n+1}}, \qquad
\norm{f_n}_\infty = 1 .
\]
Alors $\norm{f_n}_2/\norm{f_n}_1 = \frac{n+1}{\sqrt{2n+1}} \to
\infty$ et $\norm{f_n}_\infty/\norm{f_n}_2 = \sqrt{2n+1} \to
\infty$~: aucune inégalité inverse, aucune paire n'est équivalente.
\end{solution}

\begin{solution}{exo:b2:nvs:11}
\emph{Minoration de la distance~:} pour $h \in \ker\varphi$,
$\abs{\varphi(x)} = \abs{\varphi(x - h)} \leq
\vertiii\varphi\,\norm{x - h}$~; prendre l'infimum sur $h$~:
$d(x, \ker\varphi) \geq \abs{\varphi(x)}/\vertiii\varphi$.

\emph{Majoration~:} on peut supposer $\varphi(x) \neq 0$. Étant donné
$\varepsilon > 0$, choisir un vecteur unitaire $u$ avec $\abs{\varphi(u)} \geq
\vertiii\varphi - \varepsilon > 0$ et poser $h = x -
\frac{\varphi(x)}{\varphi(u)}\,u$~: alors $\varphi(h) = 0$ et
\[
\norm{x - h} = \frac{\abs{\varphi(x)}}{\abs{\varphi(u)}}
\leq \frac{\abs{\varphi(x)}}{\vertiii\varphi - \varepsilon}.
\]
Faire $\varepsilon \to 0$~: $d(x, \ker\varphi) \leq
\abs{\varphi(x)}/\vertiii\varphi$~; égalité.

\emph{Non-atteinte~:} prendre $\varphi$ de l'\cref{exo:b2:nvs:8}
($\vertiii\varphi = 1$, non atteinte) et un $x$ quelconque avec
$\varphi(x) \neq 0$. Si un $h \in \ker\varphi$ réalisait
$\norm{x - h} = \abs{\varphi(x)}$, le vecteur unitaire $v = (x -
h)/\norm{x - h}$ vérifierait $\abs{\varphi(v)} =
\abs{\varphi(x)}/\norm{x - h} = 1 = \vertiii\varphi$~: la
\omterm{thm:b2:nvs:continuouslinear}{norme d'opérateur} serait atteinte --- contradiction.
\end{solution}

\begin{solution}{exo:b2:nvs:12}
$N_1 \leq N_2$ est la monotonie du sup en fonction du domaine, donc
l'identité $(E, N_2) \to (E, N_1)$ est $1$-lipschitzienne. Pour $P_n(x) =
(x/2)^n$~: $N_2(P_n) = 1$ (atteinte en $x = 2$) tandis que $N_1(P_n) =
2^{-n}$~: une borne $N_2 \leq CN_1$ donnerait $1 \leq C2^{-n}$ pour
tout $n$~: impossible. Donc les \omterm{def:b2:nvs:norm}{normes} ne sont pas \omterm{def:b2:nvs:equivalent}{équivalentes} et
l'identité inverse est une bijection linéaire discontinue.

\emph{Incomplétude~:} posons $S_n = \sum_{k=0}^{n}\frac{X^k}{k!}$.
Pour $m > n$, $N_1(S_m - S_n) \leq \sum_{k>n}\frac1{k!} \to 0$~:
de Cauchy pour $N_1$. Si $S_n \to P$ dans $(E, N_1)$, alors
ponctuellement $P(x) = \lim S_n(x) = \eu^x$ sur $\intcc{0}{1}$~; mais un
polynôme de degré $d$ ne peut égaler $\eu^x$ sur un intervalle (dériver
$d + 1$ fois~: le membre de gauche meurt, $\eu^x$ non). Aucune limite
dans $E$~: non \omterm{def:b2:metric:complete}{complet}.

En dimension finie, les trois phénomènes sont impossibles~: toutes les
\omterm{def:b2:nvs:norm}{normes} sont \omterm{def:b2:nvs:equivalent}{équivalentes}, tout espace normé est \omterm{def:b2:metric:complete}{complet}, et l'inverse
d'une bijection linéaire est linéaire issue d'un espace de dimension
finie, donc \omterm{def:b2:metric:continuity}{continu} (\cref{thm:b2:nvs:finitedim}).
\end{solution}

\begin{solution}{pb:b2:nvs:1}
\textbf{1.} Les candidats à considérer forment $K = \{f \in F
: \norm{x - f} \leq \norm x\}$~: non vide ($0 \in K$), fermé
(image réciproque d'un intervalle fermé par l'application \omterm{def:b2:metric:continuity}{continue} $f \mapsto
\norm{x - f}$, intersectée avec le fermé $F$,
\cref{cor:b2:nvs:closedsubspace}), borné ($\norm f \leq
\norm{f - x} + \norm x \leq 2\norm x$). Dans $F$, de dimension
finie, fermé et borné signifie \omterm{def:b2:metric:compact}{compact}
(\cref{thm:b2:nvs:finitedim})~; la fonction \omterm{def:b2:metric:continuity}{continue} $f
\mapsto \norm{x - f}$ atteint sur $K$ son infimum, qui égale
l'infimum sur $F$ tout entier (tout $f \notin K$ donne $\norm{x -
f} > \norm x \geq \inf$).

\textbf{2.} Identité du parallélogramme dans $(\R^n,
\norm\cdot_2)$~: $\norm{u + v}^2 + \norm{u - v}^2 = 2\norm u^2 +
2\norm v^2$ (développer les carrés de sommes de coordonnées). Pour
$u \neq v$ unitaires~:
\[
\Bigl\Vert\frac{u+v}2\Bigr\Vert^2 = 1 - \frac{\norm{u -
v}^2}{4} < 1 .
\]
Non strictement convexe~: pour $\norm\cdot_\infty$, prendre $u = (1, 1, 0,
\dots)$, $v = (1, -1, 0, \dots)$~: vecteurs unitaires de milieu
$(1, 0, \dots)$ de \omterm{def:b2:nvs:norm}{norme} $1$~; pour $\norm\cdot_1$, prendre $u = (1,
0, \dots)$, $v = (0, 1, 0, \dots)$~: milieu $(\frac12,
\frac12, 0, \dots)$ de \omterm{def:b2:nvs:norm}{norme} $1$.

\textbf{3.} Posons $d = d(x, F)$. Si $d = 0$~: $x \in \overline F =
F$ et l'unique minimiseur est $x$. Si $d > 0$ et que $f_1 \neq f_2$
minimisent tous deux~: $u = \frac{x - f_1}{d}$ et $v = \frac{x -
f_2}{d}$ sont des vecteurs unitaires distincts, donc
\[
\Bigl\Vert x - \frac{f_1 + f_2}2\Bigr\Vert
= d\,\Bigl\Vert\frac{u + v}2\Bigr\Vert < d ,
\]
avec $\frac{f_1 + f_2}2 \in F$~: contredit la définition de
$d$. Minimiseur unique.

\textbf{4.} $\norm{(0,1) - t(1,0)}_\infty = \max(\abs t, 1)
\geq 1$, avec égalité si et seulement si $\abs t \leq 1$~: les minimiseurs forment
le segment $\{t(1, 0) : t \in \intcc{-1}{1}\}$, tous à distance
$1$ --- l'unicité échoue précisément parce que la boule carrée a des
côtés plats (question 2).

\textbf{5.} Posons $M = \max f$, $m = \min f$ (atteints~:
compacité). Pour toute constante $c$~: $\sup\abs{f - c} \geq
\max(M - c,\, c - m) \geq \frac{M - m}2$, la dernière inégalité
car les deux quantités ont pour moyenne $\frac{M-m}2$~; l'égalité dans les
deux force $M - c = c - m$, c'est-à-dire $c = c^* = \frac{M + m}2$.
Réciproquement $\sup\abs{f - c^*} = \max(M - c^*, c^* - m) =
\frac{M - m}2$. Meilleure constante unique. Pour $f(x) = x^2$ sur
$\intcc01$~: $c^* = \frac12$, distance $\frac12$.

\textbf{6.} De $\cos(n{+}1)\theta + \cos(n{-}1)\theta =
2\cos\theta\cos n\theta$~: les polynômes définis par $T_0 = 1$,
$T_1 = X$, $T_{n+1} = 2XT_n - T_{n-1}$ vérifient $T_n(\cos\theta)
= \cos n\theta$ par récurrence. Unicité~: deux polynômes
coïncidant sur $\intcc{-1}{1}$ (une infinité de points) sont égaux.
De nouveau par récurrence~: $\deg T_n = n$ de coefficient dominant
$2^{n-1}$ pour $n \geq 1$ ($T_1$~: coefficient $1 = 2^0$~; la
récurrence le double).

\textbf{7.} Tout $x \in \intcc{-1}{1}$ s'écrit $\cos\theta$, et
$\abs{\cos n\theta} \leq 1$. En $\eta_k = \cos\frac{k\pi}n$~:
$T_n(\eta_k) = \cos k\pi = (-1)^k$, et $1 = \eta_0 > \eta_1 >
\dots > \eta_n = -1$. Racines~: $\cos n\theta = 0$ si et seulement si $\theta =
\frac{(2k-1)\pi}{2n}$~: les $n$ points distincts
$\cos\frac{(2k-1)\pi}{2n}$, et comme $\frac{(k-1)\pi}n <
\frac{(2k-1)\pi}{2n} < \frac{k\pi}n$, chaque racine est strictement
comprise entre deux extrema consécutifs.

\textbf{8.} $T_2 = 2X^2 - 1$, $T_3 = 4X^3 - 3X$, $T_4 = 8X^4 -
8X^2 + 1$. Pour $T_3$~: $T_3(1) = 1$, $T_3(\tfrac12) = \tfrac12 -
\tfrac32 = -1$, $T_3(-\tfrac12) = 1$, $T_3(-1) = -1$~: alternance
parfaite.

\textbf{9.} Posons $u_\pm = x \pm \sqrt{x^2 - 1}$ pour $x \geq 1$~:
les racines de $z^2 - 2xz + 1$, avec $u_+u_- = 1$. La suite
$s_n = \frac{u_+^n + u_-^n}2$ vérifie $s_{n+1} = 2x\,s_n -
s_{n-1}$ (récurrence de type Newton issue de l'équation du second
degré), $s_0 = 1$,
$s_1 = x$~: la même récurrence et les mêmes valeurs initiales que $n \mapsto
T_n(x)$, donc $s_n = T_n(x)$ pour tout $n$. Comme $0 < u_- \leq 1
\leq u_+$ avec $u_+ > 1$ pour $x > 1$~: $T_n(x) \geq
\frac{u_+^n}2 \to \infty$ et $T_n(x) \sim \frac12\bigl(x +
\sqrt{x^2-1}\bigr)^n$. (Pour $x \leq -1$, utiliser la parité $T_n(-x)
= (-1)^nT_n(x)$, claire d'après la récurrence.)

\textbf{10.} Pour tout $\theta$~: $T_m\bigl(T_n(\cos\theta)\bigr)
= T_m(\cos n\theta) = \cos mn\theta =
T_{mn}(\cos\theta)$. Les polynômes $T_m \circ T_n$ et
$T_{mn}$ coïncident sur $\intcc{-1}{1}$, donc sont égaux.

\textbf{11.} $D = Q_n - P$ est de degré $\leq n - 1$ (les termes
dominants unitaires s'annulent). Aux extrema~: $(-1)^kD(\eta_k) =
2^{1-n} - (-1)^kP(\eta_k) \geq 2^{1-n} - \abs{P(\eta_k)} > 0$ par
hypothèse. Ainsi $D$ prend des valeurs non nulles de signes alternés aux
$n + 1$ points décroissants $\eta_0 > \dots > \eta_n$~: par le
théorème des valeurs intermédiaires, il possède au moins $n$ racines
distinctes, une dans chaque intervalle \omterm{def:b2:metric:topology}{ouvert} $\intoo{\eta_{k}}{\eta_{k-1}}$. Un
polynôme non nul de degré $\leq n - 1$ ne peut avoir $n$ racines~;
et $D = 0$ contredit les signes stricts. Contradiction~:
$\sup\abs P \geq 2^{1-n}$ pour tout $P$ unitaire de degré $n$.

\textbf{12.} Maintenant $(-1)^kD(\eta_k) = 2^{1-n} - (-1)^kP(\eta_k)
\geq 0$ puisque $\abs{P} \leq 2^{1-n}$. Sur chaque
$\intcc{\eta_k}{\eta_{k-1}}$ ($k = 1, \dots, n$), les valeurs de $D$
aux extrémités ont des signes larges opposés~: le théorème des valeurs
intermédiaires fournit un zéro $z_k$ dans l'intervalle fermé. Si les $z_k$
peuvent être choisis deux à deux distincts, $D \neq 0$ de degré $\leq n-1$
a $n$ racines~: contradiction. Deux intervalles consécutifs ne peuvent
partager que le zéro $z_k = z_{k+1} = \eta_k$ avec $0 < k < n$
(intérieur). Là, $D(\eta_k) = 0$ signifie $P(\eta_k) =
(-1)^k2^{1-n}$, une valeur extrémale de $P$ sur $\intcc{-1}{1}$
atteinte en un point \emph{intérieur}~: $P'(\eta_k) = 0$~; et
$\eta_k$ est aussi un extremum intérieur de $T_n$~: $Q_n'(\eta_k) =
0$. Donc $D'(\eta_k) = 0$~: $\eta_k$ est une racine de multiplicité
$\geq 2$, compensant l'intervalle partagé. Dans tous les cas $D$ possède
au moins $n$ racines comptées avec multiplicité, de degré $\leq n -
1$, donc $D = 0$~: $P = Q_n$. Le théorème extrémal de Tchebychev est
démontré~: l'unique minimiseur unitaire est $2^{1-n}T_n$, de \omterm{def:b2:nvs:norm}{norme} sup
$2^{1-n}$.

\textbf{13.} Les polynômes unitaires de degré $n$ sont exactement les
$X^n - R$ avec $R \in \R_{n-1}[X]$, donc
\[
d_\infty\bigl(X^n, \R_{n-1}[X]\bigr) = \min_{P \text{
unitaire}}\ \sup_{\intcc{-1}{1}}\abs P = 2^{1-n},
\]
uniquement en $R^* = X^n - Q_n$. Substitution $x = \frac{1+t}2$~:
si $P$ est unitaire de degré $n$ sur $\intcc01$, alors $t \mapsto
2^nP\bigl(\frac{1+t}2\bigr)$ est unitaire sur $\intcc{-1}{1}$ de
sup égal à $2^n\sup_{\intcc01}\abs P$~: d'où
$\sup_{\intcc01}\abs P \geq 2^{-n}\cdot2^{1-n} = 2^{1-2n}$, avec
égalité exactement pour $P^*(x) = 2^{-n}Q_n(2x - 1)$~: sur
$\intcc{0}{1}$ la distance vaut $2^{1-2n}$.

\textbf{14.} $\omega$ est unitaire de degré $n$, donc
$\sup_{\intcc{-1}{1}}\abs\omega \geq 2^{1-n}$ avec égalité si et seulement si
$\omega = Q_n = 2^{1-n}T_n$, c'est-à-dire si et seulement si les nœuds sont les $n$
racines de $T_n$~: $x_k = \cos\frac{(2k-1)\pi}{2n}$ --- les
\emph{nœuds de Tchebychev}. Valeur minimale~: $2^{1-n}$. Des nœuds
équidistants sont strictement moins bons~; le facteur d'erreur
d'interpolation est minimisé en concentrant les nœuds près des
extrémités.

\textbf{15.} $n = 2$ à la main~: $x^2$ parcourt $\intcc01$, donc
$\sup_{\intcc{-1}{1}}\abs{x^2 - c} = \max(\abs c, \abs{1 - c})
\geq \frac12$, minimisé en $c = \frac12$~: le trinôme unitaire
minimal $x^2 - \frac12 = \frac12(2x^2 - 1) = Q_2$, valeur
$\frac12 = 2^{1-2}$. Pour $n = 10$ sur $\intcc{0}{1}$~: $2^{1-20} =
2^{-19} \approx 1.9\cdot10^{-6}$. Un certain polynôme de degré $9$
reste à deux millionièmes près de $x^{10}$ sur tout
$\intcc01$~: à cette échelle les deux graphes sont indiscernables
--- la platitude de $x^{10}$ près de $0$ laisse les degrés inférieurs
faire tout le travail.

\textbf{16.} $\norm{g_k}_\infty = g_k(1) = 1$. Pour $j > k$, posons
$a = 2^k$, $b = 2^j \geq 2a$, et évaluons en $x_0 = 2^{-1/a}$
(de sorte que $x_0^a = \frac12$)~:
\[
g_k(x_0) - g_j(x_0) = \frac12 - \Bigl(\frac12\Bigr)^{b/a}
\geq \frac12 - \frac14 = \frac14 .
\]
Ainsi $\norm{g_k - g_j}_\infty \geq \frac14$ pour tous $j \neq k$~: aucune
sous-suite n'est de Cauchy, aucune ne converge. La boule unité fermée de
$\bigl(C(\intcc01), \norm\cdot_\infty\bigr)$ n'est pas compacte.

\textbf{17.} $F$ est fermé (\cref{cor:b2:nvs:closedsubspace})
et propre~: choisir $y \notin F$, donc $\delta = d(y, F) > 0$. D'après la
question 1, la distance est atteinte en un certain $f^* \in F$. Posons $x
= \frac{y - f^*}{\delta}$, un vecteur unitaire ($\norm{y - f^*} =
\delta$). Pour tout $g \in F$~:
\[
\norm{x - g} = \frac{\norm{y - (f^* + \delta g)}}{\delta}
\geq \frac{\delta}{\delta} = 1 ,
\]
car $f^* + \delta g \in F$. Donc $d(x, F) \geq 1$~; et $d(x,
F) \leq \norm{x - 0} = 1$~: exactement $1$.

\textbf{18.} Dans un $E$ de dimension infinie, construire des vecteurs
unitaires par récurrence~: $x_1$ arbitraire~; étant donnés $x_1, \dots, x_k$, le
sous-espace $F_k = \operatorname{Vect}(x_1, \dots, x_k)$ est
de dimension finie, donc propre, et la question 17 donne un vecteur unitaire
$x_{k+1}$ avec $d(x_{k+1}, F_k) = 1$~: en particulier
$\norm{x_{k+1} - x_i} \geq 1$ pour $i \leq k$. La suite a des
distances mutuelles $\geq 1$~: la boule unité contient une suite
sans sous-suite convergente, donc n'est pas compacte ---
le théorème de Riesz, avec la constante optimale $1$.

\textbf{19.} Soit $h_n$ affine sur chaque moitié de
$\bigl[\frac1{n+1}, \frac1n\bigr]$, de $0$ jusqu'à $1$ au
milieu puis de retour à $0$, et nulle ailleurs~: \omterm{def:b2:metric:continuity}{continue},
$\norm{h_n}_\infty = 1$. Pour $n \neq m$, les supports se rencontrent au
plus en une extrémité commune, où toutes deux s'annulent~; au sommet de
$h_n$, $h_m = 0$~: $\norm{h_n - h_m}_\infty = 1$ exactement. Pour
$x > 0$ fixé~: $h_n(x) = 0$ dès que $\frac1n < x$, et
$h_n(0) = 0$ toujours~: $h_n \to 0$ simplement~; mais
$\norm{h_n - 0}_\infty = 1$~: pas uniformément. Une constellation
explicite à distance mutuelle $1$ dans la boule unité.

\textbf{20.} Une boule de rayon $\frac13$ a un diamètre $\leq
\frac23 < 1$, donc contient au plus un des $h_n$ (deux sont
à distance $1$). Un nombre fini de telles boules contient un nombre
fini des $h_n$, qui sont en nombre infini~: elles ne peuvent recouvrir
la boule unité. La précompacité --- dont jouissent les \omterm{def:b2:metric:compact}{espaces
métriques compacts}, d'après la preuve du~\cref{thm:b2:metric:borellebesgue}
--- échoue aussi complètement que possible.

\textbf{21.} Si $Ax = \lambda x$ avec $x \neq 0$~:
$\abs\lambda\,\norm x = \norm{Ax} \leq \vertiii A\,\norm x$,
donc $\abs\lambda \leq \vertiii A$. Pour
$\begin{psmallmatrix}1 & -2\\ 3 & 1\end{psmallmatrix}$~:
$\vertiii A_\infty = \max(1 + 2,\ 3 + 1) = 4$
(\cref{exo:b2:nvs:4}), donc toute \omterm{def:b2:reduction:eigen}{valeur propre} a un module $\leq
4$~; en fait $\chi_A = X^2 - 2X + 7$ donne $\lambda = 1 \pm
\iu\sqrt6$ de module $\sqrt7 \approx 2.65$~: la borne est
valide, non optimale.

\textbf{22.} $N_P$ est une \omterm{def:b2:nvs:norm}{norme}~: $N_P(x) = 0$ force $P^{-1}x = 0$,
donc $x = 0$~; l'homogénéité et l'inégalité triangulaire sont héritées
de $\norm\cdot_\infty$ à travers l'application linéaire $P^{-1}$. \omterm{thm:b2:nvs:continuouslinear}{Norme
d'opérateur}~: avec $y = P^{-1}x$ et $D = \mathrm{diag}(\lambda_i)$,
\[
N_P(Ax) = \norm{P^{-1}AP\,y}_\infty = \norm{Dy}_\infty,
\]
donc $\vertiii A_{N_P}$ est la \omterm{thm:b2:nvs:continuouslinear}{norme d'opérateur} en $\norm\cdot_\infty$
de $D$, qui est sa plus grande somme de ligne en valeur absolue
(\cref{exo:b2:nvs:4})~: $\max_i\abs{\lambda_i}$.

\textbf{23.} Si tous les $\abs{\lambda_i} < 1$~: $N_P(A^kx) \leq
\rho^kN_P(x)$ avec $\rho = \max\abs{\lambda_i} < 1$, donc $A^kx
\to 0$ pour tout $x$, et $A^k \to 0$ dans toute \omterm{def:b2:nvs:norm}{norme} sur
$\mathcal{M}_n$ (toutes \omterm{def:b2:nvs:equivalent}{équivalentes} en dimension finie,
\cref{thm:b2:nvs:finitedim}~; la convergence de $A^ke_j$ pour chaque
$j$ est la convergence coefficient par coefficient). Si un $\abs{\lambda} \geq 1$
de vecteur propre $x$~: $\norm{A^kx} = \abs\lambda^k\norm x
\not\to 0$. Pour $A = \frac14\begin{psmallmatrix}1 & 2\\ 2 &
1\end{psmallmatrix}$~: \omterm{def:b2:reduction:eigen}{valeurs propres} $\frac14(1 \pm 2) =
\frac34, -\frac14$, toutes deux de module $< 1$~: $A^k \to 0$.

\textbf{24.} Ce sont deux \omterm{def:b2:nvs:norm}{normes} sur $\R_n[X]$, de dimension finie,
donc \omterm{def:b2:nvs:equivalent}{équivalentes} pour chaque $n$. Prenons le minimiseur unitaire
$P^*_n$ de la question 13 sur $\intcc01$~: son coefficient de
$X^n$ vaut $1$, donc $N_c(P^*_n) \geq 1$, tandis que
$\norm{P^*_n}_{\intcc01} = 2^{1-2n}$. D'où
\[
C_n \geq \frac{N_c(P^*_n)}{\norm{P^*_n}_{\intcc01}}
\geq 2^{2n-1} .
\]
Les constantes d'équivalence explosent avec la dimension~: sur la
réunion $\R[X]$, aucune constante unique ne convient, ce qui est
exactement la non-équivalence observée dans l'\cref{exo:b2:nvs:12} ---
«~toutes les \omterm{def:b2:nvs:norm}{normes} sont \omterm{def:b2:nvs:equivalent}{équivalentes}~» est un théorème sur une
dimension à la fois, et la dimension infinie est là où il meurt.

\textbf{25.} La compacité des parties fermées bornées en dimension
finie a produit l'existence des meilleures approximations (question
1) et a alimenté le comptage des zéros aux $n + 1$ points extrémaux
(questions 11--12, via des sups atteints). La stricte convexité gouverne
l'unicité de la meilleure approximation --- les boules rondes donnent un
minimiseur, les boules à côtés plats en donnent des segments (questions
2--4). La constellation de vecteurs unitaires à distance mutuelle
$1$ (questions 17--19) détruit la compacité de la boule unité
et, avec elle, la précompacité (question 20). La question 24
quantifie l'effondrement~: les constantes reliant deux \omterm{def:b2:nvs:norm}{normes} sur
$\R_n[X]$ croissent comme $4^n$, si bien qu'aucune comparaison uniforme
ne survit au passage à $\R[X]$. Le sommet est le théorème extrémal de
Tchebychev (questions 11--12)~: l'unique minimiseur unitaire
$2^{1-n}T_n$. La meilleure approximation trouve sa maison moderne dans
les espaces de Hilbert, où le théorème de projection remplace
la compacité par la complétude plus l'identité du parallélogramme ---
démontré honnêtement dans le volume de troisième année.
\end{solution}
