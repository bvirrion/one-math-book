\section*{Chapitre \ref{ch:b2:proba} --- Probabilités sur les espaces dénombrables}

\begin{solution}{exo:b2:proba:1}
Par symétrie~: l'\omterm{def:b2:structures:generated}{ordre} de tirage induit un \omterm{def:b2:structures:generated}{ordre} relatif uniforme
sur les boules $1$ et $2$, de sorte que $\P(1 \text{ avant } 2) =
\frac12$. Formellement~: échanger les positions des boules $1$ et $2$
dans une suite de tirages est une bijection des issues
(équiprobables) qui échange l'\omterm{def:b2:proba:space}{événement} avec son complémentaire. Parmi les boules
$1, \dots, k$~: l'\omterm{def:b2:structures:generated}{ordre} relatif de ces $k$ boules est uniforme
parmi les $k!$ ordonnancements, et la boule $1$ est première dans $(k-1)!$
d'entre eux~: probabilité $\frac{(k-1)!}{k!} = \frac1k$.
\end{solution}

\begin{solution}{exo:b2:proba:2}
$\frac{1}{k(k+1)} = \frac1k - \frac1{k+1}$, de sorte que $\sum_{k\geq1} p_k$
télescope vers $1$~: une \omterm{def:b2:proba:space}{mesure de probabilité}. Résultats pairs~:
\[
\P(2\N^*) = \sum_{j=1}^{\infty}\frac{1}{2j(2j+1)}
= \sum_{j=1}^{\infty}\Bigl(\frac{1}{2j} - \frac{1}{2j+1}\Bigr)
= \frac12 - \frac13 + \frac14 - \frac15 + \cdots
\]
C'est la série harmonique \omterm{def:b2:linalg:alternating}{alternée} avec son premier terme
retiré et les signes inversés~: puisque $\ln 2 = 1 - \frac12 + \frac13 -
\frac14 + \cdots$ (\cref{ch:b2:series}),
\[
\P(2\N^*) = -\bigl(\ln 2 - 1\bigr) = 1 - \ln 2 \approx 0.307 .
\]
\end{solution}

\begin{solution}{exo:b2:proba:3}
Soit $S$ = malade, $+$ = test positif. Bayes
(\cref{thm:b2:proba:bayes}) avec la partition $\{S, S^c\}$~:
\[
\P(S \mid +)
= \frac{0.99 \times 10^{-4}}
       {0.99 \times 10^{-4} + 0.01 \times 0.9999}
= \frac{0.000099}{0.000099 + 0.009999}
\approx 0.0098 ,
\]
en dessous de $1\%$. Bien que le test soit «~exact à 99\%~», un résultat
positif vous laisse environ $99\%$ de chances d'être bien portant~: les faux
positifs parmi la vaste majorité bien portante submergent les vrais
positifs de la minuscule minorité malade. Les tests de dépistage de
conditions rares doivent toujours être lus à travers ce calcul du
taux de base.
\end{solution}

\begin{solution}{exo:b2:proba:4}
Ponctuellement sur $\Omega$~: $\omega \in \bigcup A_i$ ssi un certain facteur
$1 - \mathbf{1}_{A_i}(\omega)$ s'annule, de sorte que
\[
\mathbf{1}_{\bigcup A_i}
= 1 - \prod_{i=1}^n\bigl(1 - \mathbf{1}_{A_i}\bigr)
= \sum_{\emptyset \neq J \subseteq \{1,\dots,n\}}
(-1)^{\abs J + 1}\prod_{i \in J}\mathbf{1}_{A_i} ,
\]
en développant le produit et en faisant passer le $1$. Or
$\prod_{i\in J}\mathbf{1}_{A_i} = \mathbf{1}_{\bigcap_{i \in J}
A_i}$, et sommer contre les poids $\P(\{\omega\})$ ---
légitime~: un nombre fini de termes bornés, chaque \omterm{def:b2:series:summable}{famille sommable} ---
transforme chaque indicatrice en la probabilité de son \omterm{def:b2:proba:space}{événement}, donnant
la formule.
\end{solution}

\begin{solution}{exo:b2:proba:5}
Soit $A_i$ = «~la lettre $i$ est dans la bonne \omterm{pb:b2:curves:1}{enveloppe}~». Pour $J$ de
taille $k$, $\P\bigl(\bigcap_{i\in J}A_i\bigr) =
\frac{(n-k)!}{n!}$ (fixer $k$ lettres, permuter le reste). Par
inclusion--exclusion,
\[
\P\Bigl(\bigcup A_i\Bigr)
= \sum_{k=1}^n (-1)^{k+1}\binom nk \frac{(n-k)!}{n!}
= \sum_{k=1}^n \frac{(-1)^{k+1}}{k!} ,
\]
de sorte que
\[
\P(\text{aucune correspondance})
= 1 - \P\Bigl(\bigcup A_i\Bigr)
= \sum_{k=0}^{n}\frac{(-1)^k}{k!}
\xrightarrow[n\to\infty]{} e^{-1} \approx 0.368 .
\]
Exactement une correspondance~: une permutation avec exactement un point fixe est
déterminée par le choix de la lettre fixe ($n$ façons) et un
\emph{dérangement} (arrangement sans correspondance) des $n - 1$ autres~;
en notant $D_{n-1} = (n-1)!\sum_{k=0}^{n-1}\frac{(-1)^k}{k!}$ pour
le nombre de dérangements (la première partie, multipliée par
$(n-1)!$),
\[
\P(\text{exactement une correspondance})
= \frac{n\,D_{n-1}}{n!}
= \frac{D_{n-1}}{(n-1)!}
= \sum_{k=0}^{n-1}\frac{(-1)^k}{k!}
\xrightarrow[n\to\infty]{} e^{-1} :
\]
à la limite, «~aucune correspondance~» et «~exactement une correspondance~» sont également
probables, chacune de probabilité $e^{-1}$.
\end{solution}

\begin{solution}{exo:b2:proba:6}
Conditionnons par le début (règle des probabilités composées /
\cref{thm:b2:proba:bayes})~:
\begin{itemize}
\item premier lancer face (probabilité $1 - p$)~: le jeu recommence
à neuf~; durer plus de $n$ signifie durer plus de $n - 1$ à partir de
là~: contribution $(1-p)\,q_{n-1}$~;
\item premiers lancers PF (probabilité $p(1-p)$)~: redémarrage après deux
lancers~: contribution $p(1-p)\,q_{n-2}$~;
\item premiers lancers PP~: le jeu s'est terminé (en $n$ lancers, $n
\geq 2$)~: contribue $0$.
\end{itemize}
Donc $q_n = (1-p)q_{n-1} + p(1-p)q_{n-2}$. L'équation
caractéristique $r^2 = (1-p)r + p(1-p)$ a pour racines
\[
r_\pm = \frac{(1-p) \pm \sqrt{(1-p)^2 + 4p(1-p)}}{2},
\]
avec $\abs{r_\pm} < 1$~: en effet le polynôme $\chi(r) = r^2 -
(1-p)r - p(1-p)$ satisfait $\chi(1) = 1 - (1-p) - p(1-p) = p^2 >
0$ et $\chi(-1) = 1 + (1-p) - p(1-p) > 0$, tandis que $\chi(0) =
-p(1-p) < 0$~: une racine dans $\intoo{-1}{0}$, une dans $\intoo{0}{1}$.
Donc $q_n = \alpha r_+^n + \beta r_-^n \to 0$. Les \omterm{def:b2:proba:space}{événements} «~le jeu
dure plus de $n$~» décroissent vers «~le jeu ne se termine jamais~»~; la continuité
monotone (\cref{thm:b2:proba:continuity}) donne
$\P(\text{ne se termine jamais}) = \lim q_n = 0$~: le jeu se termine presque
sûrement.
\end{solution}

\begin{solution}{exo:b2:proba:7}
\emph{$\P(R_n) = 1/n$~:} parmi les $n$ premiers tirages, chacune des $n$
positions relatives du dernier tirage est équiprobable (uniformité
de l'\omterm{def:b2:structures:generated}{ordre} relatif), et $R_n$ est l'\omterm{def:b2:proba:space}{événement} où c'est la
plus grande~: probabilité $1/n$.

\emph{Une infinité de records~:} $\sum_n \P(R_n) = \sum 1/n =
\infty$ et les $R_n$ sont \omterm{def:b2:proba:independence}{indépendants} (admis), de sorte que
Borel--Cantelli~2 (\cref{thm:b2:proba:borelcantelli}) donne
$\P(\limsup R_n) = 1$~: les records ne cessent jamais, presque sûrement --- mais
ils se raréfient logarithmiquement.

\emph{Records consécutifs~:} par indépendance,
\[
\sum_n \P(R_n \cap R_{n+1})
= \sum_n \frac{1}{n(n+1)} < \infty ,
\]
de sorte que Borel--Cantelli~1 s'applique~: presque sûrement, seul un nombre fini de
fois un record est immédiatement suivi d'un autre record. Les deux
moitiés du lemme travaillent de concert~: une infinité de records, mais
(p.s.) à partir d'un certain rang jamais deux d'affilée.
\end{solution}

\begin{solution}{exo:b2:proba:8}
\emph{Première partie~:} $\sum \P(A_n) = \sum\frac{1}{n+1} = \infty$
avec indépendance~: Borel--Cantelli~2 donne $\P(\limsup A_n) = 1$.
Chaque $A_n$ individuel est de plus en plus improbable, pourtant presque tout
$\omega$ appartient à une infinité d'entre eux.

\emph{Contre-exemple sans indépendance~:} prenons $\Omega = \N^*$
avec les poids $p_k = \frac{1}{k(k+1)}$ de
l'\cref{exo:b2:proba:2}, et $A_n = \{k \in \N^* : k \geq n\}$.
Alors
\[
\P(A_n) = \sum_{k \geq n}\Bigl(\frac1k - \frac1{k+1}\Bigr)
= \frac1n ,
\qquad
\sum_n \P(A_n) = \infty ,
\]
mais les $A_n$ sont décroissants, de sorte que $\limsup_n A_n = \bigcap_n A_n =
\emptyset$~: $\P(\limsup A_n) = 0$. La divergence de $\sum\P(A_n)$
seule ne garantit rien lorsque les \omterm{def:b2:proba:space}{événements} s'accumulent sur une partie
rétrécissante de l'espace --- l'indépendance est ce qui interdit cette
conspiration.
\end{solution}

\begin{solution}{exo:b2:proba:9}
Avec $q = 1 - p$, le premier pile tombe au rang $2j + 1$ avec
probabilité $q^{2j}p$, de sorte que
\[
\P(\text{rang impair}) = \sum_{j\geq0}q^{2j}p
= \frac{p}{1 - q^2} = \frac{1}{1 + q} .
\]
Pour une pièce équilibrée~: $\frac1{1 + 1/2} = \frac23$. (Vérification de bon sens~:
les rangs impairs devraient être plus probables, puisque le rang $1$ vient en premier ---
et en effet $\frac1{1+q} > \frac12$ toujours.)
\end{solution}

\begin{solution}{exo:b2:proba:10}
Les \omterm{def:b2:proba:space}{événements} $B_N = \bigcap_{n=1}^N A_n^c$ décroissent vers
$\bigcap_nA_n^c$, et par indépendance des complémentaires
$\P(B_N) = \prod_{n=1}^N(1 - p_n)$~; la continuité monotone
(\cref{thm:b2:proba:continuity}) donne la limite affichée.
En prenant les logarithmes, $\prod(1 - p_n) > 0$ ssi $\sum-\ln(1 -
p_n) < \infty$. Si $\sum p_n < \infty$ alors $p_n \to 0$ et
$-\ln(1 - p_n) \sim p_n$~: la série des logarithmes \omterm{def:b2:integration:improper}{converge}. Si
$\sum p_n = \infty$, alors $-\ln(1 - p_n) \geq p_n$ force la
divergence, de sorte que le produit est $0$. Cela concorde avec
Borel--Cantelli~2~: pour $\sum p_n = \infty$, non seulement
$\P(\text{aucun }A_n\text{ ne se produit}) = 0$, mais presque sûrement
une infinité de $A_n$ se produisent.
\end{solution}

\begin{solution}{exo:b2:proba:11}
Disons que la boîte $A$ est celle découverte vide en premier, l'autre
boîte contenant $k$. Cela signifie~: parmi les $2n - k$
premiers gestes, exactement $n$ sont allés vers $A$ et $n - k$ vers $B$ (dans un
certain \omterm{def:b2:structures:generated}{ordre}), et le geste numéro $2n - k + 1$ est reparti vers $A$,
la trouvant vide. Les gestes sont des choix équilibrés \omterm{def:b2:proba:independence}{indépendants},
de sorte que cet \omterm{def:b2:proba:space}{événement} a probabilité
$\binom{2n-k}{n}2^{-(2n-k)}\cdot\frac12$~; en doublant (la boîte vide
peut être l'une ou l'autre) on obtient
\[
\P(\text{l'autre boîte a }k) = \binom{2n-k}{n}\,2^{-(2n-k)} .
\]
Pour $n = 1$~: $k = 1$ donne $\binom11 2^{-1} = \frac12$ et $k
= 0$ donne $\binom21 2^{-2} = \frac12$~: total $1$, comme il se doit.
\end{solution}

\begin{solution}{exo:b2:proba:12}
(a) Si $\P(\{n\}) = c$ pour tout $n$, la $\sigma$-additivité
impose $1 = \sum_nc$~: impossible, que $c = 0$ (somme $0$)
ou $c > 0$ (somme infinie). Il n'existe pas de probabilité uniforme sur
$\N$.

(b) Si $A \cap B = \emptyset$ et $d(A)$, $d(B)$ existent, alors
$\abs{(A \sqcup B)\cap\intint1n} = \abs{A\cap\intint1n} +
\abs{B\cap\intint1n}$, de sorte que $d(A \sqcup B) = d(A) + d(B)$~:
additivité finie sur de telles paires. Chaque singleton a une fonction
de comptage constante à partir d'un certain rang, donc densité $0$, tandis que
$d(\N^*) = 1$. Si $d$ était $\sigma$-additive, $\N^* =
\bigsqcup_k\{k\}$ donnerait $1 = \sum_k 0 = 0$~: la densité est
finiment additive mais non $\sigma$-additive --- l'axiome a un
contenu.

(c) Soit $A = \bigcup_{k\geq0}\intint{4^k}{2\cdot4^k - 1}$
(blocs de $4^k$ à $2\cdot4^k - 1$). En $n = 2\cdot4^K -
1$ le compte est $\sum_{k\leq K}4^k \sim \frac43 4^K$, donnant
un rapport $\to \frac23$~; en $n = 4^{K+1} - 1$ le compte est
inchangé, donnant un rapport $\to \frac13$. Le rapport oscille
entre les limites $\frac13$ et $\frac23$~: pas de densité.
\end{solution}

\begin{solution}{pb:b2:proba:1}
\textbf{1.} Un chemin de \omterm{def:b2:curves:length}{longueur} $n$ est déterminé par l'ensemble de
ses pas montants~; terminer en $k$ signifie $u$ pas montants et $n - u$
pas descendants avec $u - (n - u) = k$, c'est-à-dire $u = \frac{n+k}2$~:
possible ssi $n + k$ est pair et $\abs k \leq n$, de
$\binom{n}{(n+k)/2}$ façons. Chaque chemin spécifique est un point de
la mesure produit équilibrée sur $n$ lancers~: probabilité $2^{-n}$.
Donc $\P(S_n = k) = N_n(k)2^{-n}$.

\textbf{2.} $S_n$ a la parité de $n$, de sorte que $S_{2n+1} \neq
0$~; et $u_n = N_{2n}(0)4^{-n} = \binom{2n}n4^{-n}$. Valeurs~:
$u_1 = \frac12$, $u_2 = \frac6{16} = \frac38$, $u_3 =
\frac{20}{64} = \frac5{16}$.

\textbf{3.} $\dfrac{u_n}{u_{n-1}} =
\dfrac{\binom{2n}n}{4\binom{2n-2}{n-1}} =
\dfrac{(2n)(2n-1)}{4n^2} = \dfrac{2n-1}{2n} < 1$~: décroissante.
Par l'\cref{ex:b2:comparison:centralbinomial}, $\binom{2n}n \sim
\frac{4^n}{\sqrt{\pi n}}$, de sorte que $u_n \sim \frac1{\sqrt{\pi n}}
\to 0$, et $\sum u_n$ diverge par comparaison avec $\sum
n^{-1/2}$.

\textbf{4.} Étant donné un chemin de $1$ à $k$ touchant $0$,
réfléchir son segment initial (jusqu'au \emph{premier} passage à
$0$) par rapport à l'axe horizontal~: le résultat est un chemin de
$-1$ à $k$, et l'opération est une involution --- tout
chemin de $-1$ à $k \geq 1$ doit croiser $0$, et réfléchir
son segment initial en retour retrouve l'original. Donc les
chemins touchants sont au nombre de $N_{n-1}(k + 1)$ (de $-1$ à $k$ le
déplacement est $k + 1$). Un chemin de $0$ à $k$ restant $>
0$ après l'instant $0$ commence par un pas montant puis va de
$1$ à $k$ en $n - 1$ pas sans toucher $0$~: il y en a
$N_{n-1}(k-1) - N_{n-1}(k+1)$.

\textbf{5.} Avec $m = \frac{n+k}2$, en utilisant
$\binom{n-1}{m-1} = \frac mn\binom nm$ et $\binom{n-1}{m} =
\frac{n-m}n\binom nm$~:
\[
\frac{N_{n-1}(k-1) - N_{n-1}(k+1)}{N_n(k)}
= \frac{\binom{n-1}{m-1} - \binom{n-1}{m}}{\binom nm}
= \frac{m - (n - m)}{n} = \frac kn .
\]
Pour $n = 3$, $k = 1$~: $N_3(1) = 3$ chemins ($++-$, $+-+$,
$-++$), dont seul $++-$ reste positif ($+-+$ revient à
$0$ à l'instant $2$)~: un sur trois, et $\frac kn = \frac13$.

\textbf{6.} Par symétrie la probabilité est $2\P(S_i > 0\ \forall
i \leq 2n)$. En sommant sur le point d'arrivée $2k$ et en utilisant
la question~4 (avec $n$ remplacé par $2n$)~:
\[
\P(S_i > 0\ \forall i) = 2^{-2n}\sum_{k\geq1}
\bigl(N_{2n-1}(2k-1) - N_{2n-1}(2k+1)\bigr)
= 2^{-2n}\,N_{2n-1}(1),
\]
une somme télescopique. Or $N_{2n-1}(1) = \binom{2n-1}{n}$ et
$2\binom{2n-1}n = \binom{2n}n$ (Pascal), de sorte que la
probabilité affichée est $2\cdot2^{-2n}\binom{2n-1}n =
\binom{2n}n4^{-n} = u_n$.

\textbf{7.} Les \omterm{def:b2:proba:space}{événements} $D_n = \{S_i \neq 0,\ i \leq 2n\}$
décroissent, d'intersection «~aucun retour jamais~»~; par continuité
monotone et la question~6, $\P(\text{aucun retour}) = \lim u_n =
0$~: la marche revient presque sûrement. De plus $f_n = \P(D_{n-1})
- \P(D_n) = u_{n-1} - u_n$, et par la question~3
\[
u_{n-1} - u_n = u_n\Bigl(\frac{2n}{2n-1} - 1\Bigr) =
\frac{u_n}{2n-1};
\qquad
\sum_{n\geq1}f_n = u_0 - \lim u_n = 1 .
\]

\textbf{8.} $2n\,f_n = \frac{2n}{2n-1}u_n \geq u_n$, et
$\sum u_n = \infty$ (question~3)~: la série $\sum 2nf_n$
diverge. Le premier retour est certain mais n'a pas de temps
d'attente moyen fini --- la marche est \emph{récurrente nulle}, dans le
vocabulaire que le \cref{ch:b2:randomvar} fournira.

\textbf{9.} L'\omterm{def:b2:proba:space}{événement} «~au moins $k$ retours~» est l'union
\omterm{def:b2:structures:countable}{dénombrable} disjointe, sur $0 < n_1 < \dots < n_k$, des
\omterm{def:b2:proba:space}{événements} «~les $k$ premiers retours se produisent exactement aux instants
$2n_1, \dots, 2n_k$~». Un tel \omterm{def:b2:proba:space}{événement} est l'intersection de
$k$ \omterm{def:b2:proba:space}{événements} dépendant des blocs de lancers disjoints
$\intint1{2n_1}$, $\intint{2n_1+1}{2n_2}$, \dots, chaque bloc
exigeant d'une marche fraîche qu'elle fasse son premier retour après
exactement le nombre alloué de pas~; par indépendance des
blocs sa probabilité est $f_{n_1}f_{n_2-n_1}\cdots
f_{n_k-n_{k-1}}$. En sommant par paquets (\cref{ch:b2:series},
tous les termes positifs)~:
\[
\P(\text{au moins }k\text{ retours})
= \Bigl(\sum_{n\geq1}f_n\Bigr)^{\!k} = 1^k = 1 .
\]
Les \omterm{def:b2:proba:space}{événements} décroissent en $k$, de sorte que par continuité monotone
$\P(\text{une infinité de retours}) = 1$~: récurrence.

\textbf{10.} Par la question~9 la marche fait une infinité
d'excursions loin de $0$. Le premier pas de chaque excursion est
une pièce fraîche, indépendante de tout ce qui précède~: la
probabilité que les $m$ premières excursions commencent toutes vers le bas
est $2^{-m}$. Pour atteindre $1$ la marche n'a besoin que d'un début
d'excursion vers le haut (depuis $<0$ elle doit passer par $0$ avant
d'atteindre $1$, les pas étant $\pm1$), de sorte que $\P(\text{n'atteint jamais }1)
\leq 2^{-m}$ pour tout $m$~: la marche atteint $1$ presque sûrement.
En décomposant sur le temps d'atteinte (presque sûrement fini), la
marche redémarrée là est une marche fraîche partie de $1$~: par
récurrence elle atteint chaque $k \geq 1$ presque sûrement, et par
symétrie chaque $k \leq -1$. Enfin, en redémarrant au premier
passage en $k$, la question~9 s'applique à la marche fraîche~: chaque
site est visité une infinité de fois, presque sûrement.

\textbf{11.} Les \omterm{def:b2:proba:space}{événements} $A_n = \{S_{2n} = 0\}$ sont loin d'être
\omterm{def:b2:proba:independence}{indépendants} (être en $0$ à l'instant $2n$ rend le fait d'être en $0$ à
l'instant $2n + 2$ bien plus probable que $u_{n+1}$), de sorte que
Borel--Cantelli~2 est indisponible, et en effet tout le travail
de la Partie~II était de le remplacer. L'autre direction ne requiert aucune
indépendance~: \emph{si} $\sum\P(A_n)$ \omterm{def:b2:integration:improper}{converge},
Borel--Cantelli~1 donne un nombre fini de retours presque sûrement.
Cette implication est le moteur de toute preuve de transience
ci-dessous.

\textbf{12.} Un retour à l'instant $2n$ requiert $n$ pas montants et $n$
pas descendants~: $\P(S_{2n} = 0) = \binom{2n}np^nq^n =
u_n(4pq)^n$, et $4pq = 1 - (p - q)^2 < 1$ pour $p \neq
\frac12$. Puisque $u_n \leq 1$, la série $\sum\P(S_{2n} = 0)$
est dominée par la géométrique $\sum(4pq)^n$~: convergente. Par
Borel--Cantelli~1, $\P(S_{2n} = 0 \text{ une infinité de fois}) =
0$~: un nombre fini de retours, presque sûrement.

\textbf{13.} Pour $n + k$ pair, $\P(S_n = k) =
\binom{n}{\frac{n+k}2}p^{\frac{n+k}2}q^{\frac{n-k}2}$~; le
coefficient binomial est au plus le coefficient central, et
$p^{\frac{n+k}2}q^{\frac{n-k}2} = (pq)^{n/2}(p/q)^{k/2}$,
donnant la borne annoncée $\leq
2^n(pq)^{n/2}(p/q)^{k/2} = (4pq)^{n/2}(p/q)^{k/2}$, \omterm{def:b2:series:summable}{sommable}
en $n$ puisque $\sqrt{4pq} < 1$. Borel--Cantelli~1~: le site $k$ est
visité un nombre fini de fois presque sûrement~; l'union sur $k \in
\Z$ des \omterm{def:b2:proba:space}{événements} nuls exceptionnels est encore nulle (sous-additivité
\omterm{def:b2:structures:countable}{dénombrable}). Presque sûrement chaque site est visité un nombre fini de fois,
de sorte que la suite d'entiers $(S_n)$ quitte définitivement toute fenêtre
bornée~: $\abs{S_n} \to \infty$.

\textbf{14.} $\P(S_n \neq 0,\ 1 \leq n \leq 200) = u_{100} =
\binom{200}{100}4^{-100} \approx \frac1{\sqrt{100\pi}}
\approx 0.056$~: plus d'une chance sur vingt que $200$
lancers équilibrés ne s'égalisent jamais. La décroissance $1/\sqrt{\pi n}$ est
atrocement lente~: la certitude d'une égalité (question~7) est
compatible avec de très longues plages sans égalité --- un premier
avant-goût des phénomènes d'arcsinus de la Partie~IV.

\textbf{15.} Conditionnons par le premier pas. Si $X_1 = +1$ alors
$T_1 = 1$, et $f_1 = \frac12$ concorde. Si $X_1 = -1$, la
marche doit remonter de $-1$ à $1$~; par la décomposition en blocs,
revenir à $0$ pour la première fois à l'instant $2n$ se scinde en~:
un pas vers le bas, puis une marche fraîche partie de $-1$ atteignant
$0$ pour la première fois --- de façon équivalente une marche fraîche atteignant
$+1$ pour la première fois --- en $2n - 1$ pas, ou l'\omterm{def:b2:proba:space}{événement} \omterm{def:b2:quadratic:adjoint}{symétrique} vers le haut.
Les deux signes contribuent également~:
\[
f_n = 2\cdot\tfrac12\,\P(T_1 = 2n - 1) = \P(T_1 = 2n-1) .
\]
Donc $\P(T_1 < \infty) = \sum f_n = 1$, tandis que $\sum_n(2n -
1)f_n = \sum_n u_n = \infty$ par la question~7~: la marche atteint
$1$ presque sûrement, en temps moyen infini.

\textbf{16.} Partitionnons $\{M_n \geq k\}$ par la valeur terminale
$S_n = m$. Pour $m \geq k$ la condition $M_n \geq k$ est
automatique. Pour $m < k$, réfléchir le chemin après son
\emph{premier} passage au niveau $k$~: c'est une bijection entre
$\{M_n \geq k, S_n = m\}$ et $\{S_n = 2k - m\}$ (tout chemin
terminant en $2k - m > k$ visite $k$~; réfléchir en retour est
l'inverse). Donc
\[
\P(M_n \geq k)
= \sum_{m > k}\P(S_n = m) + \P(S_n = k)
+ \sum_{m < k}\P(S_n = 2k - m)
= 2\P(S_n > k) + \P(S_n = k).
\]

\textbf{17.} À l'instant pair $2n$ avec $k = 1$~: $\P(S_{2n} = 1)
= 0$ et $\P(S_{2n} > 1) = \P(S_{2n} \geq 2)$, de sorte que
\[
\P(M_{2n} \geq 1) = 2\P(S_{2n} \geq 2)
= \P(S_{2n} \geq 2) + \P(S_{2n} \leq -2)
= 1 - u_n .
\]
Ainsi $\P(S_i \leq 0\ \forall i \leq 2n) = u_n$~: la marche
ne mène jamais dans les premiers $2n$ pas exactement aussi souvent qu'elle
ne s'égalise jamais (question~6) --- deux \omterm{def:b2:proba:space}{événements} bien différents,
portés par le même $u_n$.

\textbf{18.} $\{L_{2n} = 2k\} = \{S_{2k} = 0\} \cap
\{\text{la marche des lancers } 2k+1, \dots, 2n \text{ n'a
aucun zéro}\}$. Les deux \omterm{def:b2:proba:space}{événements} dépendent de blocs de lancers disjoints,
de sorte qu'ils sont \omterm{def:b2:proba:independence}{indépendants}~; le premier a probabilité $u_k$, le
second $u_{n-k}$ par la question~6 appliquée à la marche fraîche à
$(2n-2k)$ pas. Donc $\P(L_{2n} = 2k) = u_ku_{n-k}$.
Puisque $L_{2n}$ prend exactement les valeurs $0, 2, \dots, 2n$,
ces probabilités somment à $1$~:
$\sum_{k=0}^nu_ku_{n-k} = 1$, une identité binomiale livrée
par une partition probabiliste.

\textbf{19.} La symétrie est immédiate~: $u_ku_{n-k} =
u_{n-k}u_k$. Comme $u_j$ décroît en $j$, le produit
$u_ku_{n-k}$ est le plus petit pour $k$ central et le plus grand aux
extrêmes $k \in \{0, n\}$, où il vaut $u_n$~;
quantitativement $u_ku_{n-k} \approx \frac1{\pi\sqrt{k(n-k)}}$
dans le corps, contre $u_n \approx \frac1{\sqrt{\pi n}}$ aux
bords. Pour $n = 5$~: $\P(L_{10} = 0) = \P(L_{10} = 10) =
u_5 = \frac{63}{256} \approx 0.246$, tandis que $\P(L_{10} = 4) =
u_2u_3 = \frac38\cdot\frac5{16} = \frac{15}{128} \approx
0.117$. Dans un long jeu équilibré la dernière égalisation est le plus
probablement près du tout début ou de la toute fin~: un joueur
mène typiquement pendant d'énormes plages, sans aucun biais dans la
pièce.

\textbf{20.} L'image~: à l'instant $n$ la marche vit à l'échelle
$\sqrt n$ (l'étalement binomial de la question~3 --- $u_n \sim
1/\sqrt{\pi n}$ est la hauteur du pic central)~; elle
revient à $0$ une infinité de fois avec probabilité $1$ (Partie~II),
pourtant le temps d'attente entre retours a une moyenne divergente
(question~8), ce qui explique pourquoi des excursions isolées peuvent occuper une
fraction positive de tout horizon~; corrélativement la dernière
égalité d'un jeu à $2n$ pas est étalée avec les valeurs extrêmes les
plus probables (questions~18--19), et le fait de ne jamais mener a la
même probabilité lentement décroissante $u_n$ que le fait de ne jamais s'égaliser
(question~17). Certitude à la limite, persistance à chaque
horizon fini~: voilà la marche équilibrée.

\textbf{21.} Partitionnons $\{S_{2n} = 0\}$ ($n \geq 1$) par le
premier instant de retour $2k$, $1 \leq k \leq n$~: le premier bloc de
$2k$ lancers réalise un premier retour, les $2n - 2k$
lancers restants réalisent un retour d'une marche fraîche, et les blocs sont
\omterm{def:b2:proba:independence}{indépendants}~: $u_n = \sum_{k=1}^nf_ku_{n-k}$. Les deux séries
$U(x) = \sum u_nx^n$, $F(x) = \sum f_nx^n$ ont un rayon $\geq
1$ (coefficients dans $\intcc01$), et le \omterm{thm:b2:series:fubini}{produit de Cauchy}
(\cref{ch:b2:powerseries}) donne, pour $0 \leq x < 1$,
\[
U(x) - 1 = \sum_{n\geq1}\Bigl(\sum_{k=1}^n
f_ku_{n-k}\Bigr)x^n = F(x)\,U(x),
\qquad\text{c'est-à-dire}\qquad
U(x)\bigl(1 - F(x)\bigr) = 1 .
\]

\textbf{22.} Lorsque $x \uparrow 1$, $U(x)$ et $F(x)$ croissent
(coefficients positifs)~; toute somme partielle $\sum_{n\leq
N}u_n$ est une limite de $\sum_{n\leq N}u_nx^n \leq U(x)$, de sorte que
$U(x) \uparrow \sum u_n \in \intoc0{+\infty}$, et de même
$F(x) \uparrow f = \sum f_n$. Si $\sum u_n = \infty$~: $1 -
F(x) = 1/U(x) \to 0$, de sorte que $f = 1$. Si $\sum u_n = S < \infty$~:
$1 - f = 1/S > 0$, de sorte que $f < 1$. Vérifications~: marche équilibrée, $\sum u_n =
\infty$ et $f = 1$ (questions~3, 7)~; marche biaisée, $\sum
u_n(4pq)^n < \infty$ et corrélativement $f = 1 -
1/\sum_{n\geq0}u_n(4pq)^n < 1$, cohérent avec la
finitude presque sûre du nombre de retours (question~12).

\textbf{23.} Pour les quatre pas $(\pm1, 0), (0, \pm1)$ de la
marche sur $\Z^2$, les accroissements de $U = X + Y$ et $V = X - Y$
sont~: $(+,+)$ pour $(1,0)$, $(+,-)$ pour $(0,1)$, $(-,+)$ pour
$(0,-1)$, $(-,-)$ pour $(-1,0)$ --- chaque paire de signes de
probabilité $\frac14 = \frac12\cdot\frac12$~: les deux
marches de coordonnées $(U_n)$ et $(V_n)$ sont des marches
équilibrées indépendantes sur $\Z$. Puisque $S^{(2)}_{2n} = (0,0)$ ssi $U_{2n} = 0$
et $V_{2n} = 0$,
\[
\P\bigl(S^{(2)}_{2n} = (0,0)\bigr) = u_n^2 \sim \frac1{\pi
n}, \qquad \sum_nu_n^2 = \infty .
\]
L'identité de renouvellement de la question~21 et la dichotomie de
la question~22 n'ont rien utilisé d'unidimensionnel (seulement la
décomposition sur le premier retour et l'indépendance des blocs
disjoints), de sorte que $\sum u_n^{(2)} = \infty$ donne $f^{(2)} =
1$, et l'argument de la question~9 le renforce~: la marche sur
$\Z^2$ revient à l'origine une infinité de fois presque sûrement.

\textbf{24.} Avec la borne admise $\P(S^{(3)}_{2n} = 0)
\leq Cn^{-3/2}$, la série \omterm{def:b2:integration:improper}{converge}, et Borel--Cantelli~1
donne un nombre fini de retours presque sûrement~: la marche sur
$\Z^3$ est transiente (et la même borne d'exposant $-d/2$
traite tout $d \geq 3$). En tout~: le \emph{théorème de Pólya}
--- la \omterm{pb:b2:proba:1}{marche aléatoire simple} est récurrente sur $\Z$ et
$\Z^2$, transiente sur $\Z^d$ pour $d \geq 3$. Un homme ivre trouve
son chemin de retour~; un oiseau ivre peut ne pas le trouver.

\textbf{25.} Le dénombrement de chemins et la réflexion ont produit les lois
\omterm{def:b2:multint:exact}{exactes} ($u_n$, le théorème du scrutin, $f_n$, le maximum, le dernier
zéro)~; la continuité monotone a converti tout énoncé
limite («~revient au moins une fois~», «~une infinité de fois~»)
en une limite de probabilités à horizon fini~;
l'indépendance des blocs disjoints a alimenté les décompositions de
renouvellement (questions~9, 18, 21) --- c'est le squelette
\omterm{def:b2:structures:countable}{dénombrable} de la propriété de Markov~; Borel--Cantelli~1 a été l'arme
de transience (questions~12--13, 24), ne requérant aucune
indépendance~; l'identité de renouvellement a organisé le tout en
la dichotomie $\sum u_n = \infty \iff$ récurrence. L'unique
apport \omterm{def:b2:powerseries:analytic}{analytique} est l'estimation locale $u_n \sim 1/\sqrt{\pi
n}$~: son carré $1/(\pi n)$ diverge encore (dimension $2$,
récurrente), tandis que $n^{-3/2}$ \omterm{def:b2:integration:improper}{converge} (dimension $3$,
transiente) --- le théorème de Pólya est, au fond, un énoncé
sur la divergence de $\sum n^{-d/2}$.
\end{solution}
