\begin{solution}{exo:b2:funcseq:1}
$f_n(x) = \frac{x}{1 + nx}$: \omterm{def:b2:funcseq:def}{puntualmente} límite $0$ en $\intcc{0}{1}$.
\omterm{def:b2:funcseq:def}{uniformemente}: $f_n$ aumenta en $\intcc{0}{1}$ (derivado
$\frac{1}{(1+nx)^2} > 0$), entonces $\norm{f_n}_\infty = f_n(1) =
\frac{1}{1+n} \to 0$: \emph{uniforme} en $\intcc{0}{1}$.

$g_n(x) = nx\,\eu^{-nx^2}$: límite \omterm{def:b2:funcseq:def}{puntualmente} $0$ (exponencial
latidos). Sup: $g_n' = n\eu^{-nx^2}(1 - 2nx^2)$ desaparece en $x_n =
\frac{1}{\sqrt{2n}}$, donde $g_n(x_n) = \sqrt{\frac n2}\,\eu^{-1/2}
\to \infty$: no uniforme en $\intcc{0}{1}$ --- pero sí uniforme en
$\intco{\delta}{1}$, ya que allí $g_n(x) \leq n\,\eu^{-n\delta^2}
\to 0$.

$h_n(x) = x^n(1 - x^n)$: \omterm{def:b2:funcseq:def}{puntualmente} límite $0$ en $\intcc{0}{1}$
(ambos factores; en $x = 1$, $h_n = 0$). Sup: con $u = x^n \in
\intcc{0}{1}$, $u(1-u) \leq \frac14$ alcanzado en $u = \frac12$,
es decir,\ $x = 2^{-1/n} \in \intoo{0}{1}$: $\norm{h_n}_\infty =
\frac14 \not\to 0$: no uniforme en $\intcc{0}{1}$; uniforme en
$\intcc{0}{a}$ ($\sup \leq a^n \to 0$).
\end{solution}

\begin{solution}{exo:b2:funcseq:2}
$\int_0^1 nx\,\eu^{-nx^2}\dd x = \bigl[-\tfrac12
\eu^{-nx^2}\bigr]_0^1 = \frac{1 - \eu^{-n}}{2} \to \frac12$, mientras
$\int_0^1 \lim g_n = 0$. Sin contradicción:
\cref{thm:b2:funcseq:integration} requiere \emph{uniforme}
convergencia en el segmento, que falla aquí (el golpe de altura
$\sim\sqrt n$ se desliza hacia $0$).
\end{solution}

\begin{solution}{exo:b2:funcseq:3}
\omterm{def:b2:funcseq:series}{Convergencia normal} en $\intcc{-1}{1}$: $\norm{x^n/n^2}_\infty =
\frac{1}{n^2}$, \omterm{def:b2:series:summable}{sumable}: $S$ es \omterm{def:b2:metric:continuity}{continuo} allí
(\cref{thm:b2:funcseq:seriestransfer}).

Derivada: la serie derivada $\sum \frac{x^{n-1}}{n}$ converge
\omterm{def:b2:funcseq:series}{normalmente} en cada $\intcc{-a}{a}$, $a < 1$ ($\sup =
\frac{a^{n-1}}{n}$): $S$es $C^1$ en $\intoo{-1}{1}$ con
\[
S'(x) = \sum_{n\geq1} \frac{x^{n-1}}{n} = \frac1x \sum_{n\geq1}
\frac{x^n}{n} = -\frac{\ln(1 - x)}{x}
\qquad (0 < \abs x < 1),
\]
siendo la última identidad la serie logarítmica del año 1 (rederivada
sinceramente en \cref{ch:b2:powerseries}).
\end{solution}

\begin{solution}{exo:b2:funcseq:4}
Para $x > 0$ fijo la serie se alterna con $\frac{1}{n + x}
\downarrow 0$: \omterm{def:b2:funcseq:def}{convergencia puntual}, y el resto cota
$\abs{R_N(x)} \leq \frac{1}{N + 1 + x} \leq \frac{1}{N+1}$ es
\emph{uniforme} en $\intco{\delta}{\infty}$ (de hecho en
$\intoo{0}{\infty}$): \omterm{def:b2:funcseq:def}{convergencia uniforme}. (No es normal:
$\norm{\frac{(-1)^n}{n+x}}_\infty = \frac{1}{n + \delta}$,
divergente.) \omterm{def:b2:metric:continuity}{Continuidad} se sigue de
\cref{thm:b2:funcseq:seriestransfer}.

Ecuación funcional: reindexar $F(x + 1)$ con $m = n + 1$:
\[
F(x+1) = \sum_{n\geq0} \frac{(-1)^n}{n + 1 + x}
= \sum_{m\geq1}\frac{(-1)^{m-1}}{m+x} ,
\]
entonces, aislando el término $m = 0$ de $F(x)$,
\[
F(x) + F(x+1)
= \frac{1}{x} + \sum_{m\geq1}
\frac{(-1)^m + (-1)^{m-1}}{m+x} = \frac1x .
\]
\end{solution}

\begin{solution}{exo:b2:funcseq:5}
Establecer $g_n = f_n - f \geq 0$ (decreciente en $n$, por hipótesis; el
el límite es $0$ \omterm{def:b2:funcseq:def}{puntualmente}); cada $g_n$ es \omterm{def:b2:metric:continuity}{continuo}. Arregle $\varepsilon
> 0$y deje que $U_n = \{x : g_n(x) < \varepsilon\}$: \omterm{def:b2:metric:topology}{abierto} (imagen previa
de un conjunto \omterm{def:b2:metric:topology}{abierto}), aumentando ($g_{n+1} \leq g_n$) y cubriendo $K$
(\omterm{def:b2:funcseq:def}{convergencia puntual}). Por Borel--Lebesgue
(\cref{thm:b2:metric:borellebesgue}), un número finito de $U_{n_1}
\subseteq \dots \subseteq U_{n_k}$ cubren $K$: entonces $K = U_{n_k}$,
es decir,\ $\norm{g_{n_k}}_\infty \leq \varepsilon$, y por monotonicidad
$\norm{g_n}_\infty \leq \varepsilon$ para todos los $n \geq n_k$: \omterm{def:b2:funcseq:def}{uniforme
convergencia}. (La monotonicidad es esencial: las protuberancias deslizantes de
\cref{exo:b2:funcseq:2} converge \omterm{def:b2:funcseq:def}{puntualmente} en un \omterm{def:b2:metric:compact}{compacto} sin
uniformidad.)
\end{solution}

\begin{solution}{exo:b2:funcseq:6}
Sustituir $u = nx$:
\[
\int_0^1 \frac{n f(x)}{1 + n^2x^2}\dd x
= \int_0^n \frac{f(u/n)}{1 + u^2}\,\dd u .
\]
Los integrandos $h_n(u) = \frac{f(u/n)}{1+u^2}\mathbf{1}_{u \leq n}$
convergen \omterm{def:b2:funcseq:def}{puntualmente} a $\frac{f(0)}{1+u^2}$ (\omterm{def:b2:metric:continuity}{continuidad} de $f$ en
$0$) y están dominados por $\frac{\norm f_\infty}{1 + u^2}$,
integrable en $\intco{0}{\infty}$: convergencia dominada
(\cref{thm:b2:integration:dominated}) da el límite
\[
\int_0^\infty \frac{f(0)}{1 + u^2}\dd u = \frac{\pi}{2} f(0) .
\]
(Los granos se concentran en $0$: una identidad aproximada).
\end{solution}

\begin{solution}{exo:b2:funcseq:7}
Para $f(x) = x^2$, utilice la segunda familia de identidad binomial de la
prueba: $\sum_k k^2 p_k = n(n-1)x^2 + nx$. Por lo tanto
\[
B_n(f)(x) = \sum_k \frac{k^2}{n^2}\,p_k
= \frac{n(n-1)x^2 + nx}{n^2}
= x^2 + \frac{x(1 - x)}{n} :
\]
$B_n(f) - f = \frac{x(1-x)}{n}$, de sup \omterm{def:b2:nvs:norm}{norma} $\frac{1}{4n} =
O\bigl(\frac1n\bigr)$, como se predijo.
\end{solution}

\begin{solution}{exo:b2:funcseq:8}
Para $\varepsilon = 1$ existe $N$ con $\norm{P_n - P_m}_{\infty,
\R} \leq 1$ para $m, n \geq N$. Un polinomio acotado en $\R$ es
constante (una no constante tiende a $\pm\infty$): $P_n - P_m =
c_{n,m}$, constantes. Entonces para $n \geq N$: $P_n = P_N + c_n$ con
$c_n = P_n(0) - P_N(0)$ convergente (\omterm{def:b2:funcseq:def}{convergencia puntual} en $0$).
Por tanto $f = \lim P_n = P_N + \lim c_n$: un polinomio.
\end{solution}

\begin{solution}{exo:b2:funcseq:9}
(a) $\norm{(3/4)^n\varphi(4^n\cdot)}_\infty = \frac12
(3/4)^n$: \omterm{def:b2:funcseq:series}{convergencia normal}, so $W$ is \omterm{def:b2:metric:continuity}{continuo}
(\cref{thm:b2:funcseq:seriestransfer}).

(b) Corrección $x$, $m$; elige el signo de $h_m = \pm\frac12 4^{-m}$ para que
que el segmento $\intcc{4^mx}{4^m(x + h_m)}$ (de longitud $\frac12$)
no contiene ningún medio entero, lo que hace que $\varphi$ sea afín a la pendiente $\pm1$
en él (posible: un intervalo de longitud $\frac12$ coincide como máximo con un
punto medio entero; elige el lado que lo evita).

Para $n > m$: $4^n h_m = \pm\frac12 4^{n-m}$ es un número entero y
$\varphi$ es $1$-periódico: el $n$-ésimo término de la diferencia
desaparece.

Para $n = m$: $\abs{\varphi(4^m x + 4^m h_m) - \varphi(4^m x)} =
\abs{4^m h_m} = \frac12$($\varphi $afín a la pendiente $\pm 1$ en el
segmento), por lo que el término contribuye exactamente $(3/4)^m \cdot
\frac{1/2}{\abs{h_m}} = (3/4)^m\,4^m = 3^m$ en el cociente.

Para $n < m$: el $1$-Lipschitz $\varphi$ proporciona
$\bigl|(3/4)^n\bigl(\varphi(4^nx + 4^nh_m) -
\varphi(4^nx)\bigr)\bigr| \leq (3/4)^n 4^n\abs{h_m} =
3^n\abs{h_m}$: cada uno contribuye como máximo $3^n$ al cociente.

Por lo tanto
\[
\Bigl|\frac{W(x + h_m) - W(x)}{h_m}\Bigr|
\geq 3^m - \sum_{n=0}^{m-1} 3^n
= 3^m - \frac{3^m - 1}{2} = \frac{3^m + 1}{2}
\longrightarrow \infty .
\]
Si $W$ fuera diferenciable en $x$, cada cociente de diferencias a lo largo
$h_m \to 0$ convergería a $W'(x)$: contradicción. $W$ es
\omterm{def:b2:metric:continuity}{continuo} en todas partes, diferenciable en ninguna parte.
\end{solution}

\begin{solution}{exo:b2:funcseq:10}
\omterm{def:b2:funcseq:def}{Puntualmente}: para $x \in \intco{0}{1}$ la serie es geométrica de
relación $-x$,
\[
\sum_{n\geq0}(-1)^n x^n(1-x) = \frac{1-x}{1+x},
\]
y en $x = 1$ todos los términos desaparecen: suma $0 = \frac{1-1}{2}$,
consistente --- la suma es \omterm{def:b2:metric:continuity}{continuo} en $\intcc{0}{1}$.
Uniformidad: el resto es una cola geométrica,
\[
\abs{R_N(x)} = \frac{x^{N+1}(1-x)}{1+x} \leq x^{N+1}(1 - x)
\leq \max_{\intcc01} t^{N+1}(1-t)
= \frac{1}{N+2}\Bigl(\frac{N+1}{N+2}\Bigr)^{\!N+1}
\leq \frac{1}{N+2} \to 0 ,
\]
uniformemente en $x$. No es normal: $\norm{u_n}_\infty =
\max x^n(1-x) = \frac{1}{n+1}\bigl(\frac{n}{n+1}\bigr)^n \sim
\frac{1}{\eu\,n}$y $\sum \frac1{\eu n}$ divergen. Contraste:
$\sum x^n(1-x)$ tiene sumas parciales $1 - x^{N+1}$, convergentes
\omterm{def:b2:funcseq:def}{puntualmente} al \emph{discontinuo} $\mathbf 1_{\intco01}$: por
\cref{thm:b2:funcseq:continuity}, esa convergencia no puede ser
uniforme en $\intcc{0}{1}$.
\end{solution}

\begin{solution}{exo:b2:funcseq:11}
El límite $f$ es \omterm{def:b2:metric:continuity}{continuo} (\cref{thm:b2:funcseq:continuity}).
entonces
\[
\abs{f_n(x_n) - f(x)}
\leq \abs{f_n(x_n) - f(x_n)} + \abs{f(x_n) - f(x)}
\leq \norm{f_n - f}_\infty + \abs{f(x_n) - f(x)} ,
\]
y ambos términos tienden a $0$ (\omterm{def:b2:funcseq:def}{convergencia uniforme}; \omterm{def:b2:metric:continuity}{continuidad} de
$f$ y $x$). Contraejemplo bajo el mero \omterm{def:b2:funcseq:def}{convergencia puntual}:
$g_n(x) = nx\,\eu^{-nx^2} \to 0$ \omterm{def:b2:funcseq:def}{puntualmente} en $\intcc{0}{1}$
con $g_n$ y el límite \omterm{def:b2:metric:continuity}{continuo}, aún en $x_n =
\frac{1}{\sqrt{2n}} \to 0$:
\[
g_n(x_n) = \sqrt{\frac n2}\,\eu^{-1/2} \longrightarrow +\infty
\neq 0 = f(0) .
\]
\end{solution}

\begin{solution}{exo:b2:funcseq:12}
\begin{enumerate}
 \item Inducción. $n = 1$ es la definición. Supongamos la fórmula
 para $n$ y configure $g(x) = \int_0^x \frac{(x-t)^n}{n!}
 f(t)\dd t$. Para un integrando \omterm{def:b2:metric:continuity}{continuo} en $(x,t)$ y
 $C^1$ en $x$, el límite variable \omterm{thm:b2:integration:continuity}{integral de parámetros}
 se diferencia como
 \[
 g'(x) = \frac{(x-x)^n}{n!}f(x)
 + \int_0^x \frac{(x-t)^{n-1}}{(n-1)!}f(t)\dd t
 = T^nf(x)
 \]
 (divida $g(x+h) - g(x)$ en la tira
 $\int_x^{x+h}$, que es $O(h\cdot\sup)$ con el
 integrando que desaparece en $t = x$ como $h^n$, y el fijo
 Integral del incremento $x$, manejada por el valor medio.
 desigualdad y \omterm{def:b2:metric:continuity}{continuidad}). También $(T^{n+1}f)' = T^nf$
 (teorema fundamental del cálculo) y $g(0) =
 T^{n+1}f(0) = 0$: dos primitivas de $T^nf$ que desaparecen en
 $0$ coinciden, por lo que $T^{n+1}f = g$. El límite:
 \[
 \abs{T^nf(x)} \leq \norm f_\infty
 \int_0^x \frac{(x-t)^{n-1}}{(n-1)!}\dd t
 = \norm f_\infty\,\frac{x^n}{n!}
 \leq \frac{\norm f_\infty}{n!} .
 \]
 \item $\sum_n \norm{T^nf}_\infty \leq \eu\,\norm f_\infty$:
 convergencia normal, por tanto uniforme; $S$ es \omterm{def:b2:metric:continuity}{continuo}.
 Las sumas parciales satisfacen $S_N = f + T S_{N-1}$ y $T$.
 es $1$-Lipschitz para $\norm\cdot_\infty$
 ($\abs{Tg(x)} \leq x\norm g_\infty$): dejar $N \to
 \infty $en ambos lados da $S = f + TS$.
 \item Establecer $V(x) = f(x) + \eu^x\int_0^x \eu^{-t}f(t)\dd t$.
 Entonces $V - f$ es $C^1$ con $(V-f)'(x) =
 \eu^x\int_0^x\eu^{-t}f + f(x) = V(x)$y $(TV)' = V$.
 con $(V - f)(0) = TV(0) = 0$: entonces $V - f = TV$, es decir\
 $V$ resuelve la ecuación. Unicidad: si $S_1, S_2$ son
 Soluciones \omterm{def:b2:metric:continuity}{continuo}, $D = S_1 - S_2$ satisface $D =
 TD$, por lo tanto $D = T^nD$para todos $n$y $\norm D_\infty
 \leq \frac{\norm D_\infty}{n!} \to 0$: $D = 0$.
 Por lo tanto
 \[
 \sum_{n\geq0} T^nf(x) = f(x) +
 \int_0^x \eu^{x-t}f(t)\,\dd t .
 \]
 (La serie $\sum T^n$ es una serie geométrica de
 operadores: una primera muestra del solvente
 $(\mathrm{Id} - T)^{-1}$, desarrollado en el Año 3
 volumen.)
\end{enumerate}
\end{solution}

\begin{solution}{pb:b2:funcseq:1}
\textbf{1.} La linealidad se desprende claramente de la fórmula. Positividad:
los pesos $p_k(x) \geq 0$, por lo que $f \geq 0$ fuerza a $B_nf \geq
0$; la monotonicidad sigue aplicada a $g - f$. Cota: $\pm f \leq
\norm f_\infty $da $\pm B_nf \leq \norm f_\infty B_ne_0 =
\norm f_\infty$. Puntos finales: $p_k(0) = \mathbf 1_{k=0}$ y
$p_k(1) = \mathbf 1_{k=n}$, entonces $B_nf(0) = f(0)$, $B_nf(1) =
f(1)$.

\textbf{2.} Diferenciar $(x+y)^n = \sum_k\binom nk x^ky^{n-k}$
en $x$, multiplique por $x$ y configure $y = 1 - x$:
\[
nx = \sum_k k\,p_k(x) ;
\]
dos veces, multiplicando por $x^2$: $n(n-1)x^2 = \sum_k k(k-1)p_k(x)$.
Por lo tanto $B_ne_0 = 1$ (teorema del binomio), $B_ne_1(x) =
\frac{nx}{n} = x$ y
\[
B_ne_2(x) = \frac{\sum_k k^2p_k}{n^2}
= \frac{n(n-1)x^2 + nx}{n^2}
= x^2 + \frac{x(1-x)}{n} .
\]

\textbf{3.} Ampliar:
\[
\sum_k\Bigl(\frac kn - x\Bigr)^{\!2} p_k
= B_ne_2(x) - 2x\,B_ne_1(x) + x^2
= \frac{x(1-x)}{n} .
\]
Cauchy--Schwarz con la división $\abs{k/n - x}\sqrt{p_k}
\cdot \sqrt{p_k}$:
\[
\sum_k\Bigl|\frac kn - x\Bigr| p_k
\leq \Bigl(\sum_k\Bigl(\frac kn - x\Bigr)^2
p_k\Bigr)^{\!1/2}
= \sqrt{\frac{x(1-x)}{n}} \leq \frac{1}{2\sqrt n},
\]
utilizando $x(1-x) \leq \frac14$.

\textbf{4.} Los pesos $p_k(x)$ no son negativos con la suma $1$
y baricentro $\sum_k \frac kn p_k(x) = x$ (pregunta 2). el
desigualdad finita de Jensen para el convexo $f$ (inducción de la
definición de dos puntos, volumen del año 1) da
\[
f(x) = f\Bigl(\sum_k \frac kn\,p_k\Bigr)
\leq \sum_k f\Bigl(\frac kn\Bigr)p_k = B_nf(x) .
\]

\textbf{5.} En $\{k : \abs{k/n - x} > \delta\}$ se tiene
$\bigl(\frac{k/n - x}{\delta}\bigr)^2 > 1$, entonces
\[
\sum_{\abs{k/n-x}>\delta} p_k
\leq \frac{1}{\delta^2}\sum_k\Bigl(\frac kn -
x\Bigr)^2p_k
= \frac{x(1-x)}{n\delta^2} \leq \frac{1}{4n\delta^2} .
\]
Lectura: $p_k(x)$ es la ley de una frecuencia de muestreo $S_n/n$ de
$n$ lanzamientos de moneda al aire libre $x$; su media es $x$, su varianza
$\frac{x(1-x)}n \to 0$, y la pantalla es de Chebyshev.
desigualdad: la masa se concentra en $x$, por lo que promedia $f$
en contra reproduce $f(x)$ en el límite
(\cref{ch:b2:genfun} oficializa el vocabulario).

\textbf{6.}$\omega \leq 2M < \infty$; la monotonicidad es clara
(superior a un conjunto más grande). Heine: $f$ \omterm{def:b2:metric:continuity}{continuo} en un \omterm{def:b2:metric:compact}{compacto} es
uniformemente continuo, que dice exactamente $\omega(\delta) \to 0$
como $\delta \to 0^+$. Subaditividad: si $\abs{s - t} \leq
\delta_1 + \delta_2$, el punto $u$en el segmento $\intcc st$
a distancia $\min(\delta_1, \abs{s-t})$ de $s$ satisface
$\abs{s-u} \leq \delta_1$, $\abs{u-t} \leq \delta_2$ y
$\abs{f(s)-f(t)} \leq \abs{f(s)-f(u)} + \abs{f(u)-f(t)}$.
Iterando, $\omega(p\delta) \leq p\,\omega(\delta)$ para $p \in
\N^*$; para $\lambda > 0$, con $p = \lceil\lambda\rceil \leq 1
+ \lambda$: $\omega(\lambda\delta) \leq \omega(p\delta) \leq
p\,\omega(\delta) \leq (1+\lambda)\omega(\delta)$.

\textbf{7.} Desde $\sum p_k = 1$:
\[
\abs{B_nf(x) - f(x)}
= \Bigl|\sum_k\bigl(f(k/n) - f(x)\bigr)p_k\Bigr|
\leq \sum_k\omega\bigl(\abs{k/n - x}\bigr)p_k .
\]
Para cada $k$, pregunta 6 con $\lambda = \abs{k/n - x}/\delta$
da $\omega(\abs{k/n-x}) \leq \bigl(1 +
\frac{\abs{k/n-x}}\delta\bigr)\omega(\delta)$; sumando en contra
el $p_k$ produce la estimación maestra.

\textbf{8.} Insertar el límite de la pregunta 3:
\[
\abs{B_nf(x) - f(x)} \leq \Bigl(1 +
\frac{1}{2\delta\sqrt n}\Bigr)\omega(\delta),
\]
uniformemente en $x$; con $\delta = n^{-1/2}$ el paréntesis es
$\frac32$: $\norm{B_nf - f}_\infty \leq
\frac32\omega(n^{-1/2}) \to 0$ por pregunta 6 (Heine). esto
reprende \cref{thm:b2:funcseq:weierstrass} con una tarifa.

\textbf{9.} $L$-Lipschitz significa $\omega(\delta) \leq L\delta$:
tasa $\frac{3L}{2\sqrt n}$. $\alpha$-Hölder
$\omega(\delta) \leq C\delta^\alpha$: tarifa $\frac{3C}{2}
n^{-\alpha/2}$.

\textbf{10.} Usando $k\binom{2m}k = 2m\binom{2m-1}{k-1}$:
\[
\sum_{k=m+1}^{2m}k\binom{2m}k
= 2m\sum_{j=m}^{2m-1}\binom{2m-1}{j}
= 2m\cdot 2^{2m-2},
\]
porque $j \mapsto 2m-1-j$ biyecta $\{m,\dots,2m-1\}$ en
$\{0,\dots,m-1\}$, por lo que la suma es la mitad de $2^{2m-1}$. también
$\sum_{k=m+1}^{2m}\binom{2m}k = \frac{2^{2m} -
\binom{2m}m}{2}$ (misma simetría). Por lo tanto
\[
\sum_{k=m+1}^{2m}(k-m)\binom{2m}k
= m\,2^{2m-1} - m\,\frac{2^{2m} - \binom{2m}m}{2}
= \frac m2\binom{2m}m .
\]
La sustitución $k \mapsto 2m-k$ asigna los términos con $k < m$
sobre aquellos con $k > m$ (binomios iguales, $\abs{k-m}$ igual):
la suma absoluta es el doble de la suma unilateral, $m\binom{2m}m$.

\textbf{11.} En $x = \frac12$, $p_k(\tfrac12) =
\binom{2m}k2^{-2m}$y $f(\tfrac12) = 0$:
\[
B_{2m}f\Bigl(\frac12\Bigr)
= \sum_k\Bigl|\frac{k}{2m} - \frac12\Bigr|
\binom{2m}k 2^{-2m}
= \frac{2^{-2m}}{2m}\,m\binom{2m}m
= \frac{\binom{2m}m}{2\cdot4^m}
\sim \frac{1}{2\sqrt{\pi m}}
\]
por \cref{ex:b2:comparison:centralbinomial}. Desde $\omega_f
(\delta) = \delta $aquí (la función es $1$-Lipschitz y el
se alcanza el límite), la pregunta 8 predice como máximo
$\frac32(2m)^{-1/2}$: el verdadero error $\frac{1}{2\sqrt{\pi m}}$
tiene exactamente el orden $n^{-1/2}$ --- la tasa es aguda hasta
la constante.

\textbf{12.} $f \leq g$ da $g - f \geq 0$, por lo que $L(g-f) \geq
0$, es decir, $Lf \leq Lg$. De$-\abs f \leq f \leq \abs f$:
$-L\abs f \leq Lf \leq L\abs f$, es decir $\abs{Lf} \leq
L\abs f$.

\textbf{13.} By Heine elige $\delta$ con $\abs{f(s)-f(x)}
\leq \varepsilon $siempre que $\abs{s-x} \leq \delta$. si
$\abs{s - x} > \delta$, luego $\frac{(s-x)^2}{\delta^2} > 1$ y
$\abs{f(s)-f(x)} \leq 2M \leq \frac{2M}{\delta^2}(s-x)^2$. en
En ambos casos se mantiene el límite reclamado.

\textbf{14.} Reparar $x$; la pregunta 13 dice, como funciones de $s$:
\[
-\varepsilon e_0 - \frac{2M}{\delta^2}q_x
\;\leq\; f - f(x)e_0
\;\leq\; \varepsilon e_0 + \frac{2M}{\delta^2}q_x,
\qquad q_x = e_2 - 2x\,e_1 + x^2e_0 .
\]
Aplicar el lineal monótono $L_n$ (pregunta 12) y evaluar en
$x$:
\[
\abs{L_nf(x) - f(x)L_ne_0(x)}
\leq \varepsilon L_ne_0(x)
+ \frac{2M}{\delta^2}\bigl(L_ne_2(x) - 2xL_ne_1(x)
+ x^2L_ne_0(x)\bigr) .
\]

\textbf{15.} Escribe $\alpha_j = L_ne_j - e_j$, entonces
$\norm{\alpha_j}_\infty \to 0$. Desde $e_2(x) - 2xe_1(x) +
x^2e_0(x) = 0$:
\[
L_ne_2(x) - 2xL_ne_1(x) + x^2L_ne_0(x)
= \alpha_2(x) - 2x\,\alpha_1(x) + x^2\alpha_0(x),
\]
de sup \omterm{def:b2:nvs:norm}{norma} como máximo $\norm{\alpha_2} + 2\norm{\alpha_1} +
\norm{\alpha_0} \to 0$. También $L_ne_0 \to e_0$ \omterm{def:b2:funcseq:def}{uniformemente}, entonces
$L_ne_0 \leq 2$ para $n$ y $\abs{f(x)}\abs{L_ne_0(x) -
1} \leq M\norm{\alpha_0} \to 0$ grandes. Asamblea con la pregunta 14:
para $n$ grande, uniformemente en $x$,
\[
\abs{L_nf(x) - f(x)} \leq 2\varepsilon +
\frac{2M}{\delta^2}\,o(1) + M\,o(1) \leq 3\varepsilon :
\]
$L_nf \to f$ \omterm{def:b2:funcseq:def}{uniformemente} --- Teorema de Korovkin.

\textbf{16.} $B_ne_0 = e_0$ y $B_ne_1 = e_1$ exactamente, y
$\norm{B_ne_2 - e_2}_\infty = \max_x\frac{x(1-x)}{n} =
\frac{1}{4n} \to 0$ (pregunta 2): Korovkin aplica, y
Weierstrass lo sigue por tercera vez.

\textbf{17.} $I_nf$ es lineal en $f$ (los valores nodales lo son), y en
cada celda el interpolante afín de valores nodales no negativos es
no negativo: positivo. $I_ne_0 = e_0$ y $I_ne_1 = e_1$
porque una función afín es igual a su propio interpolante. en un
celda $\intcc ab$ ($b - a = \frac1n$), el interpolante afín de
$e_2$ es $L(t) = (a+b)t - ab$, y
\[
L(t) - t^2 = (t-a)(b-t) \in
\intcc{0}{\tfrac{(b-a)^2}{4}} ,
\]
con el máximo en el punto medio: $\norm{I_ne_2 - e_2}_\infty =
\frac{1}{4n^2} \to 0$. Korovkin: $I_nf \to f$ \omterm{def:b2:funcseq:def}{uniformemente} para
cada \omterm{def:b2:metric:continuity}{continuo} $f$ --- aproximación poligonal, sin
Se necesita una estimación adicional.

\textbf{18.} Dados $f$ y $\varepsilon$: Weierstrass proporciona
un polinomio $P = \sum_{j=0}^d a_jx^j$ con $\norm{f -
P}_\infty \leq \frac\varepsilon2$; reemplazando cada $a_j$ por un
racional $b_j$ con $\abs{a_j - b_j} \leq
\frac{\varepsilon}{2(d+1)}$ mueve el sup \omterm{def:b2:nvs:norm}{norma} en
$\intcc{0}{1}$ por como máximo $\frac\varepsilon2$. el conjunto de
polinomios de coeficiente racional es una unión \omterm{def:b2:structures:countable}{contable} (más de
$d$) de \omterm{def:b2:structures:countable}{conjuntos contables}, por lo tanto \omterm{def:b2:structures:countable}{contable}, y denso:
$C(\intcc{0}{1})$ es separable.

\textbf{19.} Por linealidad $\int_0^1 fP = 0$ para cada
polinomio $P$. Elija polinomios $P_n \to f$ \omterm{def:b2:funcseq:def}{uniformemente}
(Weierstraße):
\[
\Bigl|\int_0^1 f^2\Bigr|
= \Bigl|\int_0^1 f\,(f - P_n)\Bigr|
\leq \norm f_\infty\,\norm{f - P_n}_\infty
\longrightarrow 0 ,
\]
entonces $\int_0^1 f^2 = 0$. Si $f(x_0) \neq 0$, \omterm{def:b2:metric:continuity}{continuidad} da
$f^2 \geq c > 0$ en un subintervalo, contradiciendo la desaparición
Integral: $f = 0$. En consecuencia dos funciones \omterm{def:b2:metric:continuity}{continuo} con
coinciden los mismos momentos $\int f t^n$.

\textbf{20.} Con $p_{n,k}(x) = \binom nk x^k(1-x)^{n-k}$ y
las convenciones $p_{n-1,-1} = p_{n-1,n} = 0$, la regla del producto
y $k\binom nk = n\binom{n-1}{k-1}$, $(n-k)\binom nk =
n\binom{n-1}{k}$ dan
\[
p_{n,k}'(x) = n\bigl(p_{n-1,k-1}(x) - p_{n-1,k}(x)\bigr) .
\]
Sumando contra $f(k/n)$ y desplazando el índice en el primero
suma (\omterm{thm:b2:series:abel}{resumen de abel}):
\[
(B_nf)'(x) = n\sum_{j=0}^{n-1}\Bigl(f\Bigl(\frac{j+1}n\Bigr)
- f\Bigl(\frac jn\Bigr)\Bigr)p_{n-1,j}(x) .
\]

\textbf{21.} Según el teorema del valor medio, $f(\frac{j+1}n) -
f(\frac jn) = \frac1n f'(\xi_j)$con $\xi_j \in
\intoo{j/n}{(j+1)/n}$, entonces $(B_nf)'(x) = \sum_j
f'(\xi_j)\,p_{n-1,j}(x)$. El nodo $\frac{j}{n-1}$ también se encuentra en
$\intcc{j/n}{(j+1)/n}$ (ambas desigualdades se reducen a $j \leq
n-1$), por lo tanto $\abs{\xi_j - \frac j{n-1}} \leq \frac1n$ y
\[
\bigl|(B_nf)'(x) - B_{n-1}(f')(x)\bigr|
\leq \sum_j\Bigl|f'(\xi_j) -
f'\Bigl(\frac{j}{n-1}\Bigr)\Bigr| p_{n-1,j}(x)
\leq \omega_{f'}\Bigl(\frac1n\Bigr) \longrightarrow 0
\]
\omterm{def:b2:funcseq:def}{uniformemente}. Desde $B_{n-1}(f') \to f'$ \omterm{def:b2:funcseq:def}{uniformemente}
(\cref{thm:b2:funcseq:weierstrass} aplicado al \omterm{def:b2:metric:continuity}{continuo}
$f'$), la desigualdad del triángulo da $(B_nf)' \to f'$
\omterm{def:b2:funcseq:def}{uniformemente}. Los polinomios $P_n = B_nf$ luego convergen a $f$
en el sentido $C^1$.

\textbf{22.} Taylor--Lagrange en $x$: $f(\frac kn) - f(x) =
f'(x)(\frac kn - x) + \frac{f''(\xi_k)}2(\frac kn - x)^2$.
Sumando contra $p_k$, el término lineal muere (pregunta 2):
\[
\abs{B_nf(x) - f(x)}
\leq \frac{\norm{f''}_\infty}{2}\sum_k\Bigl(\frac kn -
x\Bigr)^2p_k
= \frac{\norm{f''}_\infty}{2}\cdot\frac{x(1-x)}{n}
\leq \frac{\norm{f''}_\infty}{8n} .
\]

\textbf{23.} Dos diferenciaciones más de $(x+y)^n$ dan la
momentos factoriales, con $n_{(j)} = n(n-1)\cdots(n-j+1)$:
\[
\sum_k k_{(j)}\,p_k = n_{(j)}\,x^j \qquad (j = 3, 4),
\]
y $k^3 = k_{(3)} + 3k_{(2)} + k$, $k^4 = k_{(4)} + 6k_{(3)} +
7k_{(2)} + k$ los convierten a momentos de potencia:
\[
\sum_k k^3p_k = n_{(3)}x^3 + 3n_{(2)}x^2 + nx,
\qquad
\sum_k k^4p_k = n_{(4)}x^4 + 6n_{(3)}x^3 + 7n_{(2)}x^2 + nx .
\]
Expandiendo $(k - nx)^4$ y recogiendo (un paciente pero puramente
Cálculo mecánico con los cuatro momentos de potencia):
\[
\sum_k(k-nx)^4p_k = nx(1-x)\bigl(1 + 3(n-2)x(1-x)\bigr) .
\]
Con $x(1-x) \leq \frac14$: el lado derecho es como máximo
$\frac n4\bigl(1 + \frac{3n}4\bigr) = \frac{3n^2}{16} +
\frac n4 \leq n^2$ para $n \geq 1$.

\textbf{24.} La forma Peano de Taylor en $x$ define $\eta(t) =
\frac{f(t) - f(x) - f'(x)(t-x) - \frac12f''(x)(t-x)^2}
{(t-x)^2}$ para $t \neq x$, $\eta(x) = 0$: por Taylor--Lagrange
$\eta(t) = \frac12\bigl(f''(\xi) - f''(x)\bigr)$ para algunos $\xi$
entre $t$ y $x$, por lo que $\abs\eta \leq \norm{f''}_\infty$ y
$\eta(t) \to 0$ como $t \to x$ (\omterm{def:b2:metric:continuity}{continuidad} de $f''$). sumando
la expansión contra $p_k$ y usando las preguntas 2--3:
\[
n\bigl(B_nf(x) - f(x)\bigr)
= \frac{x(1-x)}{2}f''(x)
+ n\sum_k\eta\Bigl(\frac kn\Bigr)\Bigl(\frac kn -
x\Bigr)^2p_k .
\]
Dado $\varepsilon$, elija $\delta$ con $\abs\eta \leq
\varepsilon $en $\abs{t - x}\leq\delta$. Parte cercana: como máximo
$\varepsilon\,n\cdot\frac{x(1-x)}n \leq \varepsilon$. Parte lejana:
con $C = \norm{f''}_\infty$ y la pregunta 23,
\[
n\,C\sum_{\abs{k/n-x}>\delta}\Bigl(\frac kn -
x\Bigr)^2p_k
\leq \frac{nC}{\delta^2}\sum_k\Bigl(\frac kn -
x\Bigr)^4p_k
= \frac{nC}{\delta^2 n^4}\sum_k(k-nx)^4p_k
\leq \frac{C}{\delta^2 n} \longrightarrow 0 .
\]
Por lo tanto $n(B_nf(x) - f(x)) \to \frac{x(1-x)}2f''(x)$ ---
Teorema de Voronovskaya. Para $f = e_2$ esto es exacto en cada
$n$ (\cref{exo:b2:funcseq:7}): el techo $\frac1n$ es real.

\textbf{25.} (i) La positividad convirtió las desigualdades \omterm{def:b2:funcseq:def}{puntualmente} en
desigualdades del operador: dio el límite \omterm{def:b2:nvs:norm}{norma}, Jensen,
Chebyshev y todo Korovkin: sólo la linealidad prueba
nada aquí. (ii) \omterm{def:b2:metric:compact}{Compacidad} ingresado a través de Heine (preguntas
6, 13), a través de la acotación de $f$, y a través del propio \omterm{def:b2:nvs:norm}{norma}
$\norm\cdot_\infty$ siendo finito. (iii) Tres funciones de prueba
basta porque la positividad lo reduce todo a controlar
$L_n$ en la unifamiliar $(s-x)^2 = e_2 - 2xe_1 + x^2e_0$,
cuyo lapso es el de $e_0, e_1, e_2$. (iv) Bernstein converge
al precio honesto $\omega(n^{-1/2})$ por cada \omterm{def:b2:metric:continuity}{continuo} $f$
(agudo, pregunta 11) pero se satura en $\frac1n$ para un $f$ fluido
(pregunta 24); el capítulo de Fourier ejecuta el mismo programa para
funciones periódicas con el núcleo de Fejér --- otro positivo
operador con las mismas virtudes y la misma modestia.
