\section*{Capítulo \ref{ch:b2:randomvar} --- Discrete Random Variables}

\begin{solution}{exo:b2:randomvar:1}
\emph{Binomio:} $X = \sum_{i=1}^n X_i$ con \omterm{def:b2:proba:independence}{independiente} Bernoulli
$X_i$; $V(X_i) = \E(X_i^2) - \E(X_i)^2 = p - p^2$ y \omterm{def:b2:randomvar:variance}{variaciones}
de variables \omterm{def:b2:proba:independence}{independiente} agregar
(\cref{thm:b2:randomvar:variancerules}):
\[
\E(X) = np, \qquad V(X) = np(1-p) .
\]
\emph{veneno:} $\E\bigl(X(X-1)\bigr) =
\sum_{k\geq2}k(k-1)e^{-\lambda}\frac{\lambda^k}{k!} = \lambda^2
e^{-\lambda}\sum_{j\geq0}\frac{\lambda^j}{j!} = \lambda^2$, entonces
\[
V(X) = \E(X^2) - \E(X)^2
= \lambda^2 + \lambda - \lambda^2 = \lambda .
\]
\emph{Geométrico} ($q = 1 - p$): diferenciando
$\sum_{k\geq0}q^k = \frac{1}{1-q}$ dos veces dentro del disco
(\cref{ch:b2:powerseries}), $\sum_{k\geq2}k(k-1)q^{k-2} =
\frac{2}{(1-q)^3}$, entonces
\[
\E\bigl(X(X-1)\bigr) = pq\sum_{k\geq2}k(k-1)q^{k-2}
= \frac{2q}{p^2},
\qquad
V(X) = \frac{2q}{p^2} + \frac1p - \frac{1}{p^2}
= \frac{q}{p^2} = \frac{1-p}{p^2} .
\]
\end{solution}

\begin{solution}{exo:b2:randomvar:2}
\emph{Suma:} por $n \in \N$, por desunión y \omterm{def:b2:proba:independence}{independencia},
\[
\P(X + Y = n)
= \sum_{k=0}^n \P(X = k)\P(Y = n - k)
= e^{-(\lambda + \mu)}\frac{1}{n!}
\sum_{k=0}^n \binom nk \lambda^k\mu^{n-k}
= e^{-(\lambda+\mu)}\frac{(\lambda + \mu)^n}{n!}
\]
por el teorema del binomio: $X + Y \sim \mathcal{P}(\lambda + \mu)$.
\emph{Conditional \omterm{def:b2:randomvar:law}{ley}:} para $0 \leq k \leq n$,
\[
\P(X = k \mid X + Y = n)
= \frac{\P(X = k)\P(Y = n - k)}{\P(X + Y = n)}
= \binom nk
\Bigl(\frac{\lambda}{\lambda+\mu}\Bigr)^{k}
\Bigl(\frac{\mu}{\lambda+\mu}\Bigr)^{n-k} ,
\]
el binomio \omterm{def:b2:randomvar:law}{ley} $\mathcal{B}\bigl(n,
\frac{\lambda}{\lambda+\mu}\bigr)$: dado el conteo total, cada
\omterm{def:b2:proba:space}{evento} "elige" independientemente la primera fuente con probabilidad
proporcional a su tasa.
\end{solution}

\begin{solution}{exo:b2:randomvar:3}
Cuando aún faltan juguetes $k$, cada caja nueva trae un juguete nuevo
con probabilidad $\frac kn$, independientemente del pasado: el
El tiempo de espera $W_k$ para el próximo juguete nuevo es geométrico.
$\mathcal{G}\bigl(\frac kn\bigr)$, con $\E(W_k) = \frac nk$, y
$T_n = W_n + W_{n-1} + \dots + W_1$ (el primer cuadro siempre proporciona una nueva
juguete: $W_n = 1$, compatible con $\E = n/n$). Por linealidad,
\[
\E(T_n) = \sum_{k=1}^n \frac nk = n\sum_{k=1}^n\frac1k
\sim n\ln n ,
\]
usando $\sum_{k\leq n}\frac1k = \ln n + \gamma + o(1)$
(\cref{ch:b2:comparison}). Recoger los últimos juguetes es lo que
costos: la mitad de las cajas van al último puñado.
\end{solution}

\begin{solution}{exo:b2:randomvar:4}
\omterm{def:b2:funcseq:def}{Puntualmente}, $X(\omega) = \#\{n \geq 1 : X(\omega) \geq n\} =
\sum_{n\geq1}\mathbf{1}_{X \geq n}(\omega)$. la doble familia
$\bigl(\mathbf{1}_{X \geq n}(\omega)\,\P(\{\omega\})\bigr)_{n,
\omega}$ no es negativo, por lo que Fubini para familias
(\cref{ch:b2:series}) se aplica incondicionalmente: sumando primero en
$n$ da $\E(X)$, sumando primero en $\omega$ da $\sum_n \P(X
\geq n)$; los dos son simultáneamente finitos e iguales. Para $X \sim
\mathcal{G}(p)$: $\P(X \geq n) = q^{n-1}$($q = 1-p$), entonces $\E(X) =
\sum_{n\geq1}q^{n-1} = \frac{1}{1 - q} = \frac1p$.
\end{solution}

\begin{solution}{exo:b2:randomvar:5}
\emph{Simetría:} la $i$-ésima bola extraída es una bola uniformemente aleatoria
de la urna (cualquiera de las bolas $N$ tiene la misma probabilidad de aterrizar en
posición $i$ del orden de dibujo), por lo que $\P(Y_i = 1) = \frac MN =
p $y$\E(X) = np$ por linealidad --- no se necesita \omterm{def:b2:proba:independence}{independencia}.

\emph{\omterm{def:b2:randomvar:variance}{Covarianza}:} para $i \neq j$, $\E(Y_iY_j) = \P(\text{draws }
i, j \text{ both white}) = \frac{M(M-1)}{N(N-1)}$ (pares ordenados
de posiciones distintas obtienen un par ordenado de bolas distintas,
uniformemente). Por lo tanto
\[
\operatorname{Cov}(Y_i, Y_j)
= \frac{M(M-1)}{N(N-1)} - \frac{M^2}{N^2}
= \frac{M(N - M)}{N^2}\cdot\frac{-1}{N-1}
= -\frac{p(1-p)}{N-1} < 0 :
\]
Sacar una bola blanca hace que las blancas sean más escasas para los otros sorteos.
Por \cref{thm:b2:randomvar:variancerules},
\[
V(X) = np(1-p) + n(n-1)\Bigl(-\frac{p(1-p)}{N-1}\Bigr)
= np(1-p)\,\frac{N - n}{N - 1} \leq np(1-p) :
\]
el muestreo sin reemplazo tiene la misma media pero \emph{menor}
\omterm{def:b2:randomvar:variance}{diferencia} que con reemplazo (igualdad solo para $n = 1$), el
correlaciones negativas que actúan como estabilizador. Para $n = N$ el
\omterm{def:b2:randomvar:variance}{diferencia} desaparece: el recuento es entonces determinista.
\end{solution}

\begin{solution}{exo:b2:randomvar:6}
Ampliando alrededor de $m = \E(X)$:
\[
\E\bigl((X - c)^2\bigr)
= \E\bigl((X - m)^2\bigr) + 2(m - c)\,\E(X - m) + (m - c)^2
= V(X) + (m - c)^2 ,
\]
mínimo exactamente en $c = m$ con valor $V(X)$ --- \omterm{def:b2:randomvar:expectation}{expectativa} es
el mejor predictor constante en cuadrado medio.

Si $\P(X = m) = 1$ entonces $(X - m)^2$ desaparece con probabilidad
$1$, por lo que $V(X) = 0$ (la familia definitoria tiene términos nulos excepto en un
conjunto nulo). Por el contrario, si $V(X) = 0$, Chebyshev
(\cref{thm:b2:randomvar:markov}) da $\P\bigl(\abs{X - m} \geq
\frac1n\bigr) \leq n^2\,V(X) = 0$ por cada $n$; el \omterm{def:b2:proba:space}{eventos}
$\bigl\{\abs{X - m} \geq \frac1n\bigr\}$ aumenta a $\{X \neq
m\}$, por lo que es monótono \omterm{def:b2:metric:continuity}{continuidad} (\cref{thm:b2:proba:continuity})
produce $\P(X \neq m) = 0$.
\end{solution}

\begin{solution}{exo:b2:randomvar:7}
$\E(S_n) = \frac n2$ y $V(S_n) = \frac n4$.
\emph{Markov:} $\P\bigl(S_n \geq \frac{3n}4\bigr) \leq
\frac{n/2}{3n/4} = \frac23$ --- un límite constante, inútil para
grande $n$.
\emph{Chebyshev:} el \omterm{def:b2:proba:space}{evento} implica $\abs{S_n - \frac n2} \geq
\frac n4$, por lo que la probabilidad es $\leq \frac{n/4}{(n/4)^2} =
\frac4n$ --- decae, pero solo polinomialmente.
\emph{Chernoff:} por \omterm{def:b2:proba:independence}{independencia}, $\E(e^{tS_n}) =
\prod_{i=1}^n\E(e^{tX_i}) = \bigl(\frac{1 + e^t}{2}\bigr)^n$ y
Markov aplicado a $e^{tS_n} \geq e^{3nt/4}$ da, por cada $t >
0$,
\[
\P\Bigl(S_n \geq \frac{3n}4\Bigr)
\leq \Bigl(\frac{1 + e^t}{2}\Bigr)^n e^{-3nt/4}
= \exp\Bigl(n\bigl(\ln\tfrac{1 + e^t}{2} - \tfrac{3t}4\bigr)\Bigr).
\]
Minimiza el exponente: $\frac{\dd}{\dd t}\ln\frac{1+e^t}{2} =
\frac{e^t}{1 + e^t} = \frac34$ en $e^t = 3$, es decir $t = \ln 3$,
dando
\[
\P\Bigl(S_n \geq \frac{3n}4\Bigr)
\leq \Bigl(\frac{4}{2}\Bigr)^n 3^{-3n/4}
= \bigl(2 \cdot 3^{-3/4}\bigr)^n \approx (0.877)^n ,
\]
exponencialmente pequeño. La jerarquía Markov $\to$ Chebyshev $\to$
Chernoff es la escalera estándar: cada peldaño aplica a Markov a un
función de la variable que crece más rápidamente.
\end{solution}

\begin{solution}{exo:b2:randomvar:8}
Por el teorema de la transferencia (\cref{thm:b2:randomvar:transfer})
aplicado a $f\bigl(\frac{S_n}{n}\bigr)$ con $S_n \sim
\mathcal{B}(n, x)$:
\[
\E\Bigl[f\Bigl(\frac{S_n}{n}\Bigr)\Bigr]
= \sum_{k=0}^n f\Bigl(\frac kn\Bigr)\binom nk x^k(1-x)^{n-k}
= B_nf(x) .
\]
Repare $\delta > 0$ y divida $\abs{f(S_n/n) - f(x)}$ en \omterm{def:b2:proba:space}{evento} $D
= \bigl\{\abs{\frac{S_n}{n} - x} \geq \delta\bigr\}$:
apagado $D$, la diferencia es como máximo el módulo de \omterm{def:b2:metric:continuity}{continuidad}
$\omega_f(\delta) = \sup_{\abs{s - t}\leq\delta}\abs{f(s) -
f(t)}$; en $D$, como máximo $2\norm f_\infty$. Tomando \omterm{def:b2:randomvar:expectation}{esperanzas de heredar} y
usando Chebyshev con $V\bigl(\frac{S_n}{n}\bigr) =
\frac{x(1-x)}{n} \leq \frac{1}{4n}$:
\[
\abs{B_nf(x) - f(x)}
\leq \E\,\abs{f(S_n/n) - f(x)}
\leq \omega_f(\delta)
+ 2\norm f_\infty\,\P(D)
\leq \omega_f(\delta) + \frac{2\norm f_\infty}{4n\delta^2} .
\]
El uniforme \omterm{def:b2:metric:continuity}{continuidad} de $f$ en $[0, 1]$ hace $\omega_f(\delta) \to
0$: elija $\delta$, luego $n$y $B_nf \to f$ \omterm{def:b2:funcseq:def}{uniformemente} --- el
Teorema de aproximación de Weierstrass de \cref{ch:b2:funcseq}, cuyo
El "lema de conteo" ahora es reconocible como la desigualdad de Chebyshev
para el binomio \omterm{def:b2:randomvar:law}{ley}.
\end{solution}

\begin{solution}{exo:b2:randomvar:9}
Expanda $S_n^4 = \sum_{i,j,k,l}X_iX_jX_kX_l$ y tome
\omterm{def:b2:randomvar:expectation}{esperanzas de heredar}. Por \omterm{def:b2:proba:independence}{independencia} y centrado, cualquier término que contenga
un índice que aparece exactamente una vez desaparece (factores $\E(X_i) = 0$
fuera). Términos supervivientes: los términos diagonales $n$ $\E(X_i^4)$ y los
términos que emparejan dos pares de índices iguales, $\E(X_i^2X_j^2) =
\E(X_1^2)^2$ para $i \neq j$, que aparecen $3n(n-1)$ veces: elija
el par desordenado de valores (formas $\binom n2$), luego el
$\frac{4!}{2!\,2!} = 6$ formas de colocarlos en las cuatro ranuras ---
$6\binom n2 = 3n(n-1)$. Por lo tanto, con $\E(X_1^2)^2
\leq \E(X_1^4)$ (Jensen o Cauchy--Schwarz),
\[
\E(S_n^4) = n\,\E(X_1^4) + 3n(n-1)\,\E(X_1^2)^2
\leq C n^2,
\qquad C = 4\,\E(X_1^4) .
\]
Markov en el orden 4:
\[
\P\Bigl(\Bigl|\frac{S_n}{n}\Bigr| \geq \varepsilon\Bigr)
= \P\bigl(S_n^4 \geq n^4\varepsilon^4\bigr)
\leq \frac{Cn^2}{n^4\varepsilon^4}
= \frac{C}{n^2\varepsilon^4} ,
\]
una serie \omterm{def:b2:series:summable}{sumable}. Por Borel--Cantelli~1
(\cref{thm:b2:proba:borelcantelli}), por cada $j$ el \omterm{def:b2:proba:space}{evento}
$B_j = \limsup_n\bigl\{\abs{S_n/n} \geq \frac1j\bigr\}$ tiene
probabilidad $0$, entonces $\P\bigl(\bigcup_j B_j\bigr) = 0$ por
\omterm{def:b2:structures:countable}{contable} subaditividad. Sobre el complemento --- de probabilidad $1$
--- por cada $j$ hay $N$ con $\abs{S_n/n} < \frac1j$ para
todos $n \geq N$: precisamente $S_n/n \to 0$. La fuerte ley de las grandes
los números se mantienen bajo un cuarto momento; eliminando esa hipótesis
(Teorema de Kolmogorov) es el trabajo del año 3.
\end{solution}

\begin{solution}{exo:b2:randomvar:10}
$\P(M \leq k) = \bigl(\frac k6\bigr)^2$ (ambos dados como máximo
$k$, de forma independiente), por lo que $\P(M \geq k) = 1 -
\bigl(\frac{k-1}6\bigr)^2$ y
\[
\E(M) = \sum_{k=1}^6\P(M \geq k)
= 6 - \frac{0 + 1 + 4 + 9 + 16 + 25}{36}
= 6 - \frac{55}{36} = \frac{161}{36} \approx 4.47 ,
\]
cómodamente por encima de la media $3.5$ de un solo dado, como
máximo debería ser.
\end{solution}

\begin{solution}{exo:b2:randomvar:11}
Con $I_i = \mathbf 1_{\sigma(i) = i}$: $\P(\sigma(i) = i) =
\frac{(n-1)!}{n!} = \frac1n$, entonces $\E(F_n) = n\cdot\frac1n =
1$. Para $i \neq j$: $\P(\sigma(i) = i, \sigma(j) = j) =
\frac{(n-2)!}{n!} = \frac1{n(n-1)}$, por lo tanto
\[
\operatorname{Cov}(I_i, I_j) = \frac1{n(n-1)} - \frac1{n^2}
= \frac{1}{n^2(n-1)} .
\]
Por el kit de herramientas \omterm{def:b2:randomvar:variance}{diferencia}
(\cref{thm:b2:randomvar:variancerules}),
\[
V(F_n) = n\cdot\frac1n\Bigl(1 - \frac1n\Bigr)
+ n(n-1)\cdot\frac1{n^2(n-1)}
= 1 - \frac1n + \frac1n = 1 .
\]
Media $1$, \omterm{def:b2:randomvar:variance}{diferencia} $1$, independiente de $n$ --- consistente
con el límite de Poisson del problema de emparejamiento
(\cref{exo:b2:proba:5}).
\end{solution}

\begin{solution}{exo:b2:randomvar:12}
$T_n = \sum_{k=1}^nG_k$ donde $G_k \sim \mathcal G(k/n)$ es
el momento de ver un juguete nuevo cuando falta $k$, las etapas
siendo \omterm{def:b2:proba:independence}{independiente}. Por lo tanto
\[
V(T_n) = \sum_{k=1}^n\frac{1 - k/n}{(k/n)^2}
\leq \sum_{k=1}^n\frac{n^2}{k^2}
\leq \frac{\pi^2}6\,n^2 ,
\]
por \cref{ex:b2:fourier:basel}. Con $\E(T_n) = nH_n$, $H_n =
\sum_1^n\frac1k$ (\cref{exo:b2:randomvar:3}), Chebyshev da,
para $\varepsilon > 0$,
\[
\P\bigl(\abs{T_n - nH_n} \geq \varepsilon\,n\ln n\bigr)
\leq \frac{\pi^2n^2/6}{\varepsilon^2n^2\ln^2 n}
= \frac{\pi^2}{6\,\varepsilon^2\ln^2n}
\xrightarrow[n\to\infty]{} 0 .
\]
Desde $H_n \sim \ln n$, al dividir por $n\ln n$ se muestra
$T_n/(n\ln n) \to 1$ en probabilidad: las fluctuaciones de
$T_n$ son del orden $n$, insignificantes frente a la media $n\ln n$.
\end{solution}

\begin{solution}{pb:b2:randomvar:1}
\textbf{1.} $\E(S_n) = \frac n2$ y Markov
(\cref{thm:b2:randomvar:markov}) dé $\P(S_n \geq an) \leq
\frac{n/2}{an} = \frac1{2a}$. Markov sólo conoce el medio:
no se puede distinguir una variable concentrada en $n/2$ de una
repartido entre $0$ y $n$, por lo que fija el precio de la cola como si todos
la masa podría sentarse allí.

\textbf{2.} El binomio justo es \omterm{def:b2:quadratic:adjoint}{simétrico} sobre $n/2$
($S_n$ y $n - S_n$ tienen el mismo \omterm{def:b2:randomvar:law}{ley}), por lo que con $x = n(a -
\frac12) > 0$ los dos \omterm{def:b2:proba:space}{eventos}$\{S_n - \frac n2 \geq x\}$ y
$\{S_n - \frac n2 \leq -x\}$ son disjuntos y equiprobables:
$\P(S_n \geq an) = \frac12\P(\abs{S_n - \frac n2} \geq x)$.
Chebyshev con $V(S_n) = \frac n4$:
\[
\P(S_n \geq an)
\leq \frac12\cdot\frac{n/4}{n^2(a - 1/2)^2}
= \frac1{8n(a - 1/2)^2},
\]
que es $\frac2n$ en $a = \frac34$.

\textbf{3.} Por \omterm{def:b2:proba:independence}{independencia} y el teorema del producto,
$\E(\eu^{tS_n}) = \bigl(\E \eu^{tX_1}\bigr)^n = \bigl(\frac{1
+ \eu^t}2\bigr)^n$. Markov aplicó a $\eu^{tS_n}$:
\[
\P(S_n \geq an) \leq \eu^{-tan}\Bigl(\frac{1 +
\eu^t}2\Bigr)^{\!n} = \exp\Bigl(n\bigl(\ln\tfrac{1 +
\eu^t}2 - ta\bigr)\Bigr).
\]
La derivada del exponente en $t$ es $\frac{\eu^t}{1 + \eu^t}
- a $y desaparece en $\eu^t = \frac a{1-a}$, es decir, $t^* =
\ln\frac a{1-a} > 0$; ahí $\frac{1 + \eu^{t^*}}2 =
\frac1{2(1-a)}$ y el exponente es igual
\[
n\Bigl(-\ln 2 - \ln(1-a) - a\ln\frac a{1-a}\Bigr)
= -n\bigl(\ln2 + a\ln a + (1-a)\ln(1-a)\bigr) = -n\,I(a),
\]
con $I(\frac12) = 0$ y $I'(a) = \ln\frac a{1-a} > 0$ encendidos
$\intoo{\frac12}1$: $I(a) > 0$. En $a = \frac34$:
$\eu^{-I(3/4)} = \frac12(\tfrac34)^{-3/4}(\tfrac14)^{-1/4} =
2\cdot3^{-3/4}$, el límite de \cref{exo:b2:randomvar:7}.

\textbf{4.} La suma de los números $n + 1$ $\binom nja^j(1-a)^{n-j}$
a $1$, y el más grande es el de $j = k = an$ (el modo
de $\mathcal B(n, a)$ es $\floor{(n+1)a} = k$ aquí). un
máximo de números $n + 1$ que suman $1$ es al menos
$\frac1{n+1}$:
\[
\binom nk a^k(1-a)^{n-k} \geq \frac1{n+1}
\quad\Longrightarrow\quad
\binom nk \geq \frac{a^{-an}(1-a)^{-n(1-a)}}{n+1}
= \frac{\eu^{nH(a)}}{n+1}.
\]
Por lo tanto $\P(S_n \geq an) \geq \binom{n}{an}2^{-n} \geq
\eu^{n(H(a) - \ln2)}/(n+1) = \eu^{-nI(a)}/(n+1)$: hasta el
factor polinómico $n + 1$, el exponente de Chernoff es la verdad.

\textbf{5.} $n = 100$, $a = \frac34$: Markov $\frac23$;
Chebyshev $\frac2{100} = 0.02$; Chernoff
$(2\cdot3^{-3/4})^{100} = \eu^{-100\,I(3/4)} \approx
2.1\cdot10^{-6}$, contra el $2.8\cdot10^{-7}$ exacto. Moraleja:
cada momento de información divide polinomialmente el límite;
el momento exponencial cambia su \emph{naturaleza}.

\textbf{6.} $\cosh t = \sum_{k\geq0}\frac{t^{2k}}{(2k)!}$ y
$\eu^{t^2/2} = \sum_{k\geq0}\frac{t^{2k}}{2^kk!}$; el reclamo
sigue término por término de $(2k)! \geq 2^kk!$, que se mantiene por
inducción: $(2k)! = 2k(2k-1)\cdot(2k-2)! \geq 2k\cdot
2^{k-1}(k-1)! = 2^kk!\cdot(2k-1) \geq 2^kk!$.

\textbf{7.} Por \omterm{def:b2:proba:independence}{independencia}, $\E\bigl(\eu^{t\sum\varepsilon_i}
\bigr) = (\cosh t)^n \leq \eu^{nt^2/2}$, entonces Markov da
$\P(\sum\varepsilon_i \geq s) \leq \eu^{nt^2/2 - ts}$;
minimizar en $t = s/n$ produce $\eu^{-s^2/(2n)}$.

\textbf{8.} Con $X_i = \frac{1 + \varepsilon_i}2$, $\widehat
p_n - \frac12 = \frac1{2n}\sum\varepsilon_i$, entonces
$\{\widehat p_n - \frac12 \geq \delta\} =
\{\sum\varepsilon_i \geq 2n\delta\}$ y la pregunta 7 da la
cota $\eu^{-(2n\delta)^2/(2n)} = \eu^{-2n\delta^2}$. el
evento simétrico tiene el mismo límite, de donde el factor $2$ para
$\abs{\widehat p_n - \frac12} \geq \delta$.

\textbf{9.} $\E(\eu^{tX}) = \sum_x\eu^{tx}\P(X = x)$ es un
serie de funciones suaves de $t$ cuyo término por término
Los derivados están dominados, en cada intervalo \omterm{def:b2:metric:compact}{compacto} $t$, por
$\eu^{\abs t}\P(X = x)$ (como $0 \leq x \leq 1$): por el
teorema de diferenciación para series convergentes \omterm{def:b2:funcseq:series}{normalmente}
(\cref{thm:b2:funcseq:differentiation}) es el doble
\omterm{def:b2:diffcalc:differential}{diferenciable}, y la regla del cociente da $\psi' = \E_t(X)$
y $\psi'' = \E_t(X^2) - \E_t(X)^2$, donde $\E_t$ es el
\omterm{def:b2:randomvar:expectation}{expectativa} para los pesos reponderados
$\eu^{tx}\P(X{=}x)/\E(\eu^{tX})$ --- no negativo, sumando a
$1$, llevado por los mismos valores $x \in \intcc01$. A\omterm{def:b2:randomvar:variance}{diferencia}
de una variable con valor $\intcc01$ es como máximo $\frac14$: por
\cref{exo:b2:randomvar:6}, es $\min_c\E_t((X - c)^2) \leq
\E_t\bigl((X - \tfrac12)^2\bigr) \leq \tfrac14$. taylor con
resto integral, usando $\psi(0) = 0$, $\psi'(0) = p$:
\[
\psi(t) = tp + \int_0^t(t - s)\,\psi''(s)\,\dd s
\leq tp + \frac{t^2}2\cdot\frac14,
\]
es decir,\ $\E(\eu^{t(X - p)}) \leq \eu^{t^2/8}$ para todos los $t$ reales.
