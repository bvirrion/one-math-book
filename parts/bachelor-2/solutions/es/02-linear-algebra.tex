\begin{solution}{exo:b2:linalg:6}

conjugaciĂłn, $M = PM'P^{-1}$ es invertible y
$\operatorname{tr}(AM) = \operatorname{tr}(BM') = 0$: $M \in H_A$.
\end{solution}

\begin{solution}{exo:b2:linalg:9}
El mapa $\Theta \colon A \mapsto \operatorname{tr}(A\,\cdot)$ es
lineal desde $\mathcal{M}_n(K)$ hasta su \omterm{def:b2:linalg:dual}{dual}, entre espacios de igual

\begin{solution}{exo:b2:linalg:12}
\emph{$D$ reversible.} Multiplicar a la derecha por la matriz del bloque
$T = \begin{psmallmatrix} I & 0\\ -D^{-1}C & I\end{psmallmatrix}$,

= \begin{pmatrix} A - BD^{-1}C & B\\ C - DD^{-1}C & D\end{pmatrix}

\begin{solution}{pb:b2:linalg:1}
\textbf{1.} $\Phi$ es lineal con $\ker\Phi = \bigcap_i
\ker\varphi_i $(una tupla $p$ desaparece si cada entrada lo hace). Para
las coordenadas forman $\varepsilon_i$ de $K^p$: $\Phi^{\mathsf
T}(\varepsilon_i) = \varepsilon_i \circ \Phi = \varphi_i$, entonces
$\operatorname{im}\Phi^{\mathsf T} \supseteq
\operatorname{Vect}(\varphi_i)$; por el contrario
$\operatorname{im}\Phi^{\mathsf T}$ estĂˇ abarcado por el
$\Phi^{\mathsf T}(\varepsilon_i)$ (el intervalo $\varepsilon_i$
$(K^p)^*$). Entonces $\operatorname{rk}\Phi = \operatorname{rk}
\Phi^{\mathsf T} = \dim\operatorname{Vect}(\varphi_1, \dots,
\varphi_p) =: r$ (\cref{prop:b2:linalg:transposerank}), y
rango: la nulidad da $\dim\bigcap_i\ker\varphi_i = n - r$.

\textbf{2.} ($\Leftarrow$) Mantener una subfamilia libre mĂˇxima, digamos
$\varphi_1, \dots, \varphi_r$, que abarca el mismo espacio (por lo que el
La hipĂłtesis todavĂ­a dice $\bigcap_{i \leq r}\ker\varphi_i \subseteq
\ker\varphi$: la intersecciĂłn de todo $i$ es igual a la de encima
$i \leq r$, siendo cada forma descartada una combinaciĂłn). el mapa
$\Psi = (\varphi_1, \dots, \varphi_r) \colon E \to K^r$ es
sobreyectiva (pregunta 1: su rango es $r$). Si $\Psi(x) = \Psi(y)$
luego $x - y \in \ker\Psi \subseteq \ker\varphi$, entonces $\varphi(x) =
\varphi(y)$: $\varphi $se factoriza como $\varphi = \lambda \circ \Psi$
con $\lambda \colon K^r \to K$ bien definido; $\lambda$ es lineal
porque $\Psi$ es lineal y sobreyectivo (para $t = \Psi(x)$, $t' =
\Psi(x')$: $\lambda(t + \alpha t') = \varphi(x + \alpha x') =
\lambda(t) + \alpha\lambda(t')$). Escritura $\lambda = \sum c_i
\varepsilon_i$: $\varphi = \sum_{i \leq r} c_i\varphi_i$.
($\Rightarrow$) Si $\varphi = \sum c_i \varphi_i$, cualquier $x$
matar a todos los $\varphi_i$ mata a $\varphi$.

\textbf{3.} Por pregunta 1, $\dim\bigcap\ker\varphi_i = n - r$
con $r = \dim\operatorname{Vect}(\varphi_i) \leq p$ y $r = p$
si la familia es libre. Un subespacio $F$ de codimensiĂłn $p$: es
\omterm{def:b2:linalg:annihilator}{aniquilador} tiene dimensiĂłn $p$
(\cref{thm:b2:linalg:annihilator}); una base $(\varphi_1, \dots,
\varphi_p)$de $F^\circ $da $F = \bigcap_i\ker\varphi_i$ (el
fĂłrmula de recuperaciĂłn). Menos: una intersecciĂłn de hiperplanos $q$ tiene
dimensiĂłn $\geq n - q > n - p$ por la pregunta 1.

\textbf{4.} Calcular $\ker\varphi_1 \cap \ker\varphi_2$: a partir de $x
+ y - z = 0$y $y + z - t = 0$, parametrizar por $(y, z)$: $x = z
- y$, $t = y + z$, dando el plano de los vectores $(z - y,\; y,\;
z,\; y + z)$. En Ă©l, $\psi = x + 2y - t = (z - y) + 2y - (y + z)
= 0$: por el lema de factorizaciĂłn $\psi \in
\operatorname{Vect}(\varphi_1, \varphi_2)$--- de hecho $\psi =
\varphi_1 + \varphi_2$. Pero $\psi' = x + y + t = (z - y) + y +
(y + z) = y + 2z $no es exactamente cero allĂ­ ($y = 1, z = 0$
da $1$): $\psi' \notin \operatorname{Vect}(\varphi_1,
\varphi_2)$.

\textbf{5.} Tres formas en un espacio $3$-dimensional: libertad
basta. Si es $a\psi_0 + b\psi_1 + c\psi_2 = 0$, pruebe en $1, X,
X^2$: $a + b + c = 0$, $b + \frac c2 = 0$, $b + \frac c3 = 0$;
restando los dos Ăşltimos da $c = 0$, luego $b = 0$, $a = 0$.
Base antedual: escritura $P = \alpha + \beta X + \gamma X^2$ y
resolviendo $\psi_i(P_j) = \delta_{ij}$ ($P(0) = \alpha$, $P(1) =
\alpha + \beta + \gamma$, $\int_0^1 P = \alpha + \frac\beta2 +
\frac\gamma3$):
\[
P_0 = 1 - 4X + 3X^2, \qquad
P_1 = -2X + 3X^2, \qquad
P_2 = 6X - 6X^2 .
\]
(Marque, por ejemplo: $\int_0^1 P_2 = 3 - 2 = 1$, $P_2(0) = P_2(1) = 0$.)
El problema de interpolaciĂłn se resuelve mediante coordenadas en el
base anterior:
\[
P = 1\cdot P_0 + 2\cdot P_1 + \tfrac32\, P_2 = 1 + X
\]
($X$-coeficiente $-4 - 4 + 9 = 1$, $X^2$-coeficiente $3 + 6 - 9 =
0$); de hecho $P(0) = 1$, $P(1) = 2$, $\int_0^1 P = \frac32$.

\textbf{6.} Linealidad: para cada $\varphi$, $J(x + \alpha
y)(\varphi) = \varphi(x + \alpha y) = J(x)(\varphi) + \alpha
J(y)(\varphi)$, es decir $J(x + \alpha y) = J(x) + \alpha J(y)$. Inyectividad: si $x \neq 0$,
complete $x = e_1$ en una base; la forma de coordenadas $e_1^*$ tiene
$J(x)(e_1^*) = 1 \neq 0$. Desde $\dim E^{**} = \dim E^* = \dim
E$, inyectivo implica biyectivo.

\textbf{7.} InclusiĂłn: para $x \in F$ y $\varphi \in F^\circ$,
$J(x)(\varphi) = \varphi(x) = 0$, entonces $J(F) \subseteq
F^{\circ\circ}$. Dimensiones
(\cref{thm:b2:linalg:annihilator} dos veces):
\[
\dim F^{\circ\circ} = \dim E^* - \dim F^\circ
= n - (n - \dim F) = \dim F = \dim J(F),
\]
$J$ es inyectivo. Por lo tanto $J(F) = F^{\circ\circ}$.

\textbf{8.} Primera identidad: $\varphi$ mata a $F + G$ si mata
tanto $F$ como $G$ (mata las sumas si mata las piezas):
$(F+G)^\circ = F^\circ \cap G^\circ$. Segundo: la inclusiĂłn
$F^\circ + G^\circ \subseteq (F \cap G)^\circ$ es claro (cada
orden mata a $F \cap G$). Dimensiones, usando la primera identidad.
y Grassman:
\[
\dim(F^\circ + G^\circ) = \dim F^\circ + \dim G^\circ -
\dim(F^\circ \cap G^\circ)
= (n - \dim F) + (n - \dim G) - \bigl(n - \dim(F +
G)\bigr),
\]
que por Grassmann en $E$ es igual a $n - \dim(F \cap G) = \dim(F
\cap G)^\circ$: igualdad.

\textbf{9.} A travĂ©s del lema: $\ker\psi \subseteq \ker\varphi$ con
$p = 1$ da $\varphi \in \operatorname{Vect}(\psi)$, y
$\varphi \neq 0$ hace que el escalar sea distinto de cero. Directamente: elija $x_0$
con $\psi(x_0) \neq 0$; cada $x$ escribe $x = \bigl(x -
