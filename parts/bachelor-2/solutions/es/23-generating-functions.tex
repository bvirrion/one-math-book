\begin{solution}{exo:b2:genfun:1}
Troquel justo: $G(t) = \frac16(t + t^2 + \dots + t^6) = \frac t6(1 + t
+ \dots + t^5)$. Si la suma de dos dados justos fuera uniforme en
$\{2, \dots, 12\}$, entonces
\[
G(t)^2 = \frac{t^2}{36}\,h(t)^2
= \frac{t^2}{11}\sum_{k=0}^{10}t^k ,
\qquad h(t) = 1 + t + \dots + t^5 .
\]
Ahora $h$ es un polinomio real de grado impar $5$, por lo que tiene un polinomio real
raíz (teorema del valor intermedio; concretamente $h(-1) = 0$), por lo tanto
$h^2$ tiene una raíz real. Pero $\sum_{k=0}^{10}t^k$ no tiene ninguno: es
positivo para $t \geq 0$, y para $t < 0$ es igual
$\frac{t^{11} - 1}{t - 1}$, cociente de dos números negativos.
Contradicción --- la suma de dos dados justos nunca es uniforme (como
el conocido triangular \omterm{def:b2:randomvar:law}{distribución} de sumas de dados lo confirma).
\end{solution}

\begin{solution}{exo:b2:genfun:2}
\emph{Binomio:} $G(t) = (1 - p + pt)^n$, $G'(t) = np(1 - p +
pt)^{n-1}$, $G''(t) = n(n-1)p^2(1 - p + pt)^{n-2}$, entonces
\[
\E(X) = G'(1) = np,
\qquad
V(X) = G''(1) + G'(1) - G'(1)^2
= n(n-1)p^2 + np - n^2p^2 = np(1-p).
\]
\emph{Geométrico} ($q = 1 - p$): $G(t) = \frac{pt}{1 - qt}$, entonces
$G'(t) = \frac{p}{(1 - qt)^2}$ y $G''(t) = \frac{2pq}{(1 -
qt)^3}$; en $t = 1$(usando $1 - q = p$):
\[
\E(X) = \frac{p}{p^2} = \frac1p,
\qquad
V(X) = \frac{2q}{p^2} + \frac1p - \frac{1}{p^2}
= \frac{2q + p - 1}{p^2}
= \frac{q}{p^2} ,
\]
haciendo coincidir \cref{exo:b2:randomvar:1} con menos trabajo.
\end{solution}

\begin{solution}{exo:b2:genfun:4}
Por \cref{thm:b2:genfun:compound} con $G_N(s) = e^{\lambda(s-1)}$
y $G_X(t) = 1 - p + pt$:
\[
G_S(t) = e^{\lambda(1 - p + pt - 1)} = e^{\lambda p(t - 1)} :
\]
$S \sim \mathcal{P}(\lambda p)$. El conteo descartado $D = N - S$
cuenta los mismos elementos mantenidos con probabilidad $1 - p$, por lo que por
mismo cálculo $D \sim \mathcal{P}(\lambda(1 - p))$.
\omterm{def:b2:proba:independence}{Independencia}, directamente: para $j, k \in \N$,
\begin{align*}
\P(S = j,\ D = k)
&= \P(N = j + k)\,\binom{j+k}{j}p^jq^k
= e^{-\lambda}\frac{\lambda^{j+k}}{(j+k)!}\,
\frac{(j+k)!}{j!\,k!}\,p^jq^k\\
&= \Bigl(e^{-\lambda p}\frac{(\lambda p)^j}{j!}\Bigr)
\Bigl(e^{-\lambda q}\frac{(\lambda q)^k}{k!}\Bigr)
\end{align*}
con $q = 1 - p$: los factores conjuntos \omterm{def:b2:randomvar:law}{ley} como
$\mathcal{P}(\lambda p) \otimes \mathcal{P}(\lambda q)$. Un veneno
la división del flujo al azar produce \emph{\omterm{def:b2:proba:independence}{independiente}} flujos de Poisson ---
un pequeño milagro que se utiliza constantemente en la teoría de colas.
\end{solution}

\begin{solution}{exo:b2:genfun:5}
Los tiempos de espera entre cabezas consecutivas son \omterm{def:b2:proba:independence}{independiente}
variables geométricas $\mathcal{G}(p)$ (falta de memoria: después de cada
cabeza el juego se reinicia), por lo $T_r = W_1 + \dots + W_r$ y
multiplicatividad (\cref{thm:b2:genfun:product}) da
\[
G_{T_r}(t) = \Bigl(\frac{pt}{1 - qt}\Bigr)^{r},
\qquad
\E(T_r) = r\,\E(W_1) = \frac rp,
\qquad
V(T_r) = r\,V(W_1) = \frac{rq}{p^2}
\]
($q = 1 - p$; \omterm{def:b2:randomvar:variance}{variaciones} agregado por \omterm{def:b2:proba:independence}{independencia}). Ampliación: por el
serie binomial generalizada
(\cref{ch:b2:powerseries}), $(1 - qt)^{-r} = \sum_{m\geq0}
\binom{m + r - 1}{r - 1}q^mt^m$, por lo que el coeficiente de $t^n$ en
$p^rt^r(1 - qt)^{-r}$ es (con $m = n - r$)
\[
\P(T_r = n) = \binom{n-1}{r-1}p^r(1-p)^{n-r},
\qquad n \geq r ,
\]
el \emph{binomio negativo} \omterm{def:b2:randomvar:law}{ley} --- combinatoriamente: el $r$-ésimo
la cabeza cae en el lanzamiento $n$ si las cabezas $r - 1$ anteriores eligen su
lugares entre los primeros $n - 1$ lanzamientos.
\end{solution}

\begin{solution}{exo:b2:genfun:7}
\emph{\omterm{def:b2:randomvar:expectation}{Expectativa}.} Primer $\E(Z_n) = m^n$: por
\cref{thm:b2:genfun:compound}, $\E(Z_{n+1}) = \E(Z_n)\,m$ y
$\E(Z_0) = 1$. La familia $\bigl(Z_n(\omega)\P(\{\omega\})
\bigr)_{n, \omega}$ no es negativa, por lo que aplica Fubini para familias.
incondicionalmente:
\[
\E(Y) = \sum_{n=0}^{\infty}\E(Z_n)
= \sum_{n=0}^\infty m^n = \frac{1}{1 - m} < \infty
\]
(en particular, $Y$ es casi seguramente finito: consistente con
extinción segura en el caso subcrítico).

\emph{Ecuación funcional.} Descomponer la población por el
hijos del antepasado: si el antepasado tiene hijos $Z_1 = k$, el
la progenie total es $Y = 1 + Y_1 + \dots + Y_k$, donde $Y_i$ es el
La progenie total de la línea infantil $i$-ésima --- y la $Y_i$ son
\omterm{def:b2:proba:independence}{independiente} copias de $Y$, independientes de $Z_1$ (líneas distintas
utilizar disjunto, reproducción \omterm{def:b2:proba:independence}{independiente} \omterm{def:b2:proba:space}{eventos}). Acondicionamiento en
$Z_1$ como en \cref{thm:b2:genfun:compound}:
\[
H(t) = \E\bigl(t^Y\bigr)
= t\sum_{k=0}^\infty \P(Z_1 = k)\,H(t)^k
= t\,G\bigl(H(t)\bigr),
\]
el factor $t$ representa al antepasado mismo. (Para el
\omterm{def:b2:randomvar:law}{ley} $p_0 = 1 - p$, $p_2 = p$ de ramificación binaria, esta cuadrática
La ecuación en $H$ se puede resolver explícitamente y expandirse --- la
\omterm{ex:b2:powerseries:catalan}{números catalanes} de \cref{ch:b2:powerseries} cuenta la familia
árboles.)
\end{solution}

\begin{solution}{exo:b2:genfun:9}
Escriba $p_k^{(n)} = \P(X_n = k)$, $p_k = \P(X = k)$.

\emph{Caso $k = 0$.} Para $t \in \intoo{0}{1}$ y cualquier \omterm{def:b2:randomvar:law}{ley}
$(q_j)$ con $\sum_j q_j \leq 1$:
\[
\Bigl|\,q_0 - \sum_j q_jt^j\Bigr|
= \sum_{j \geq 1} q_j t^j
\leq \sum_{j\geq1}t^j = \frac{t}{1 - t} .
\]
Por lo tanto
\[
\abs{p_0^{(n)} - p_0}
\leq \frac{2t}{1 - t}
+ \abs{G_{X_n}(t) - G_X(t)} .
\]
Dado $\varepsilon > 0$, elija $t$ con $\frac{2t}{1-t} <
\frac\varepsilon2$, luego $n_0$ de modo que el último término sea$<
\frac\varepsilon2$ para $n \geq n_0$: entonces $p_0^{(n)} \to p_0$.

\emph{Paso de inducción.} Supongamos $p_j^{(n)} \to p_j$ para $j < k$.
Considere las funciones \emph{desplazado}
\[
g_n(t) = \frac{G_{X_n}(t) - p^{(n)}_0}{t}
= \sum_{j\geq0} p^{(n)}_{j+1}t^j,
\qquad
g(t) = \frac{G_X(t) - p_0}{t} ,
\]
\omterm{ex:b2:powerseries:fibonacci}{funciones generadoras} de las secuencias de subprobabilidad
$(p^{(n)}_{j+1})_j$ (masa total $\leq 1$, que es toda la
$k = 0$ argumento utilizado). Para el fijo $t \in \intoo{0}{1}$, $g_n(t)
\to g(t)$por hipótesis y el caso $k = 0$. Al aplicar el argumento $k =
0$a $g_n $se obtiene $p_1^{(n)} \to p_1$; iterando el
cambiar $k$ veces da $p_k^{(n)} \to p_k$ por cada $k$. (Esto es
el ejemplo discreto y elemental del teorema \omterm{def:b2:metric:continuity}{continuidad} de Lévy,
cuya forma general --- para funciones características --- es un Año
3 punto de referencia.)
\end{solution}

\begin{solution}{exo:b2:genfun:10}
\omterm{def:b2:funcseq:def}{Puntualmente}, $\frac{1 + (-1)^X}{2}$ es igual a $1$ cuando $X$ es par
y $0$ cuando es impar, por lo que tomando \omterm{def:b2:randomvar:expectation}{esperanzas de heredar} (transferencia),
\[
\P(X \text{ even}) = \frac{1 + \E\bigl((-1)^X\bigr)}2 =
\frac{1 + G_X(-1)}2 .
\]
Poisson: $\frac{1 + \eu^{-2\lambda}}2 \to \frac12$ como
$\lambda$ crece. Binomio: $\frac{1 + (1 - 2p)^n}2$. en ambos
casos $G_X(-1) \to 0$ dice que la paridad de $X$ se vuelve justa
moneda: el \omterm{def:b2:randomvar:law}{ley} se extiende sobre muchos números enteros y olvida su
paridad.
\end{solution}

\begin{solution}{exo:b2:genfun:12}
Sea $A$ y $B$ el \omterm{ex:b2:powerseries:fibonacci}{funciones generadoras} del resto
duración comenzó desde ``sin cabezal actual'' y ``un cabezal actual
cabeza''. Se gasta un lanzamiento, entonces: del estado $0$, cruz
regresa al estado $0$, la cabeza pasa al estado $1$; del estado
$1$, cara termina el juego, cruz vuelve al estado $0$:
\[
A(t) = t\bigl(q\,A(t) + p\,B(t)\bigr),
\qquad
B(t) = t\bigl(p + q\,A(t)\bigr).
\]
Sustituyendo: $A(1 - qt) = pt\,B = pt(pt + qtA)$, entonces
\[
G_T(t) = A(t) = \frac{p^2t^2}{1 - qt - pq\,t^2} .
\]
En $t = 1$ el denominador es $1 - q - pq = p(1 - q) = p^2$:
$G_T(1) = 1$, el juego termina casi con seguridad (como
\cref{exo:b2:proba:6} mostrado por recursividad). logarítmico
diferenciación en $1$: $\E(T) = 2 - \frac{D'(1)}{D(1)}$ con
$D(t) = 1 - qt - pqt^2$, $D'(1) = -q - 2pq$:
\[
\E(T) = 2 + \frac{q + 2pq}{p^2} = \frac{2p^2 + q + 2pq}{p^2}
= \frac{1 + p}{p^2},
\]
que es $6$ para $p = \frac12$.
\end{solution}

\begin{solution}{pb:b2:genfun:1}
\textbf{1.} Para $t \in \intoo01$, la regla de la cadena en $G_n = G
\circ G_{n-1}$ da $G_n'(t) =
G'\bigl(G_{n-1}(t)\bigr)G_{n-1}'(t)$. Como $t \to 1^-$,
$G_{n-1}(t) \uparrow 1$ y $G'$ no son decrecientes con
límite izquierdo $m$ en $1$, por lo que el primer factor tiende a $m$; por
inducción el segundo tiende a $m^{n-1}$. Por
\cref{thm:b2:genfun:moments}, $\E(Z_n) = G_n'(1^-) = m^n$.

\textbf{2.} Diferenciándose una vez más,
\[
G_n'' = G''(G_{n-1})\,(G_{n-1}')^2 +
G'(G_{n-1})\,G_{n-1}'',
\]
y dejando $t \to 1^-$: $a_n = G''(1)m^{2(n-1)} + m\,
a_{n-1}$ con $a_n = G_n''(1)$, $a_1 = G''(1)$. Para $m \neq
1$ se comprueba por inducción que $a_n = G''(1)\,m^{n-1}
\frac{m^n - 1}{m - 1}$(la recursividad añade $G''(1)m^{2n-2}$
a $m\cdot G''(1)m^{n-2}\frac{m^{n-1}-1}{m-1}$, y
$m^{n-1} + \frac{m^{n-1}-1}{m-1} = \frac{m^n - 1}{m-1}$);
para $m = 1$, $a_n = a_{n-1} + G''(1) = n\,G''(1)$.

\textbf{3.} $V(Z_n) = a_n + m^n - m^{2n}$ y $G''(1) =
\sigma^2 + m^2 - m$. Para $m \neq 1$, la pieza $(m^2 -
m)m^{n-1}\frac{m^n-1}{m-1} = m^n(m^n - 1)$cancela exactamente $m^n -
m^{2n}$, quedando $V(Z_n) =
\sigma^2m^{n-1}\frac{m^n-1}{m-1}$. Para $m = 1$: $V(Z_n) =
nG''(1) = n\sigma^2$.

\textbf{4.} $Z_n$ es una variable entera no negativa, por lo que
$\P(Z_n > 0) = \P(Z_n \geq 1) \leq \E(Z_n) = m^n$ por Markov
(\cref{thm:b2:randomvar:markov}). Para $m < 1$ esto decae
geométricamente --- y \omterm{def:b2:series:summable}{sumablemente}, por lo que Borel--Cantelli incluso da
que sólo un número finito de generaciones no están vacías, lo cual es
extinción nuevamente.

\textbf{5.} Cauchy--Negro: $\E(Z_n)^2 = \E(Z_n\mathbf
1_{Z_n>0})^2 \leq \E(Z_n^2)\,\P(Z_n > 0)$. con la pregunta 3
y $m < 1$:
\[
\E(Z_n^2) = V(Z_n) + m^{2n}
\leq \frac{\sigma^2m^{n-1}}{1-m} + m^{2n},
\]
entonces, dividiendo $m^{2n}$ por este límite y simplificando por
$m^n$,
\[
\P(Z_n > 0) \geq \frac{m^n}{\frac{\sigma^2}{m(1-m)} + m^n}
\geq \Bigl(\frac{\sigma^2}{m(1-m)} + 1\Bigr)^{-1}m^n ,
\]
usando $m^n \leq 1$ en el denominador. Con la pregunta 4:
$\P(Z_n > 0) \asymp m^n$.

\textbf{6.} $G(t) = q\sum_k(pt)^k = \frac{q}{1 - pt}$ y $m
= G'(1) = \frac{pq}{(1-p)^2} = \frac pq$. Subcrítico para $p
< \frac12$, crítico para $p = \frac12$, supercrítico para $p
> \frac12$.

\textbf{7.} $G(t) = t$ lee $pt^2 - t + q = 0$, con raíces
$\frac{1 \pm \abs{p - q}}{2p}$, es decir,\ $1$ y $\frac qp =
\frac1m$. La probabilidad de extinción es la más pequeña fija.
punto en $\intcc01$ (\cref{thm:b2:genfun:extinction}):
$q_{\mathrm{ext}} = 1$ si $m \leq 1$ y $\frac1m$ si $m >
1$.

\textbf{8.} Para $m \neq 1$, con $p = \frac m{m+1}$, $q =
\frac1{m+1}$: si $q_n = \frac{m^n - 1}{m^{n+1} - 1}$, entonces
\[
1 - p\,q_n = \frac{(m+1)(m^{n+1} - 1) - m(m^n - 1)}
{(m+1)(m^{n+1} - 1)} = \frac{m^{n+2} - 1}{(m+1)(m^{n+1} -
1)},
\]
entonces $q_{n+1} = \frac{q}{1 - pq_n} = \frac{m^{n+1} -
1}{m^{n+2} - 1}$; el caso base $q_0 = 0$ se mantiene. Para $m = 1$:
$G(t) = \frac1{2 - t}$ y $q_{n+1} = \frac1{2 -
\frac{n}{n+1}} = \frac{n+1}{n+2}$, con $q_0 = 0$.

\textbf{9.}$1 - q_n = \frac{m^n(m - 1)}{m^{n+1} - 1}$. Para
$m < 1$ el denominador tiende a $-1$: $1 - q_n \sim (1 -
m)\,m^n$. Para $m > 1$:
\[
q_{\mathrm{ext}} - q_n = \frac1m - \frac{m^n - 1}{m^{n+1} -
1} = \frac{m - 1}{m\,(m^{n+1} - 1)} \sim \frac{m -
1}{m^{2}}\;m^{-n} .
\]
Y $G'(t) = \frac{pq}{(1 - pt)^2}$ evaluado en $t = \frac
qp $(donde $1 - pt = 1 - q = p$) da $G'(q_{\mathrm{ext}})
= \frac qp = \frac1m$: la relación observada $m^{-1}$ es exactamente
la derivada en el punto fijo de atracción.

\textbf{10.} Para $p = \frac12$: $G''(t) = \frac{1/4}{(1 -
t/2)^3}$, entonces $G''(1) = 2$y $\sigma^2 = G''(1) + m - m^2 =
2$. La forma cerrada da $1 - q_n = \frac1{n+1}$: el
La probabilidad de supervivencia decae como $1/n$ --- demasiado lentamente para ser
\omterm{def:b2:series:summable}{sumable}, a diferencia de cualquier tasa subcrítica.

\textbf{11.} Inducción: $G_1(t) = \frac1{2-t}$ coincide con el
fórmula para $n = 1$, y
\[
G(G_n(t)) = \cfrac{1}{2 - \cfrac{n - (n-1)t}{n+1 - nt}}
= \frac{n + 1 - nt}{2(n+1) - 2nt - n + (n-1)t}
= \frac{n+1 - nt}{n + 2 - (n+1)t} .
\]
entonces
\[
\frac{G_n(t) - q_n}{1 - q_n}
= (n+1)\,\Bigl(\frac{n - (n-1)t}{n+1 - nt} -
\frac{n}{n+1}\Bigr)
= \frac{t}{n + 1 - nt}
= \frac{\frac{t}{n+1}}{1 - \frac{n}{n+1}t} ,
\]
el pgf del geométrico \omterm{def:b2:randomvar:law}{ley} $\mathcal G\bigl(\frac1{n+1}
\bigr)$en $\N^*$ (\cref{ex:b2:genfun:classical}): dado
supervivencia, $\P(Z_n = k \mid Z_n > 0) =
\frac1{n+1}\bigl(\frac n{n+1}\bigr)^{k-1}$, con condicional
significa $n + 1$. La media incondicional $1 = \E(Z_n)$ es la
producto de una probabilidad de supervivencia que desaparece y una probabilidad lineal
tamaño condicional creciente.

\textbf{12.} Si la línea se extingue en la generación $n$, entonces
$Y = Z_0 + \dots + Z_{n-1}$ es finito; si nunca va

\textbf{14.} Diferenciando $H = tG(H)$ de $\intoo01$ y

\textbf{15.} $C_k = \frac1{k+1}\binom{2k}k \sim

\textbf{24.} \emph{$G'(q) < 1$:} $G - \mathrm{id}$ es \omterm{def:b2:affine:convex}{convexo}

\emph{Proceso complementario:} $\widehat G(t) = G(qt)/q =
