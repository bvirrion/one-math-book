\begin{solution}{exo:b2:multint:3}
El dominio es $\{0 \leq x \leq 1,\ x^2 \leq y \leq x\}$.
$y$-primero:
\[
\int_0^1\!\int_{x^2}^{x}(x + y)\,\dd y\,\dd x
= \int_0^1\Bigl(x(x - x^2)
+ \frac{x^2 - x^4}{2}\Bigr)\dd x
= \int_0^1\Bigl(\frac{3x^2}{2} - x^3 - \frac{x^4}{2}\Bigr)\dd x
= \frac12 - \frac14 - \frac1{10} = \frac{3}{20} .
\]
$x$-primero: el corte a la altura $y \in [0, 1]$ es $y \leq x \leq
\sqrt y$, por lo que
\[
\int_0^1\!\int_{y}^{\sqrt y}(x + y)\,\dd x\,\dd y
= \int_0^1\Bigl(\frac{y - y^2}{2}
+ y(\sqrt y - y)\Bigr)\dd y
= \frac14 - \frac16 + \frac25 - \frac13 = \frac{3}{20} .
\]
\end{solution}

\begin{solution}{exo:b2:multint:4}
\emph{Primera integral.} En el disco $D_R$, en coordenadas polares:
\[
\iint_{D_R}\frac{\dd x\,\dd y}{(1 + x^2 + y^2)^2}
= \int_0^{2\pi}\!\!\int_0^R
\frac{\rho\,\dd\rho\,\dd\alpha}{(1 + \rho^2)^2}
= 2\pi\Bigl[-\frac{1}{2(1 + \rho^2)}\Bigr]_0^R
= \pi\Bigl(1 - \frac{1}{1 + R^2}\Bigr)
\xrightarrow[R \to \infty]{} \pi .
\]
\emph{Segunda integral.} El cuarto de disco es $0 \leq \alpha \leq
\frac\pi2$, $0 \leq \rho \leq 1$y $xy =
\rho^2\cos\alpha\sin\alpha$:
\[
\iint_{D'}xy\,\dd x\,\dd y
= \int_0^{\pi/2}\!\!\cos\alpha\sin\alpha\,\dd\alpha
\int_0^1 \rho^3\,\dd\rho
= \frac12\cdot\frac14 = \frac18 .
\]
\end{solution}

\begin{solution}{exo:b2:multint:5}

(El \omterm{pb:b2:curves:1}{astroide} cabe en el disco unitario de \omterm{def:b2:multint:domain}{ĂĄrea} $\pi$; tres octavos de

\end{solution}

\begin{solution}{exo:b2:multint:8}
El integrando $g(t; x, y) = xP(tx, ty) + yQ(tx, ty)$ es
$\mathcal{C}^1$ en $(x, y)$, \omterm{def:b2:metric:continuity}{continuo} en $t$, con parcial
derivados \omterm{def:b2:metric:continuity}{continuo} sobre $[0,1] \times U$; diferenciaciĂłn bajo

\[
f_x(x, y)
= \int_0^1 \bigl(P(tx, ty) + tx\,P_x(tx, ty)
+ ty\,Q_x(tx, ty)\bigr)\dd t .
\]
Usando el cierre $Q_x = P_y$:
\[
tx\,P_x(tx, ty) + ty\,P_y(tx, ty)
= t\,\frac{\dd}{\dd t}\bigl[P(tx, ty)\bigr] ,
\]
entonces el integrando es $P(tx, ty) + t\frac{\dd}{\dd t}P(tx, ty) =
\frac{\dd}{\dd t}\bigl[t\,P(tx, ty)\bigr]$ y
\[
f_x(x, y) = \Bigl[t\,P(tx, ty)\Bigr]_0^1 = P(x, y) .
\]
SimĂŠtricamente $f_y = Q$ (se utiliza el mismo cĂĄlculo con $P_y = Q_x$)
al revĂŠs). Note donde entra la hipĂłtesis: $f$ estĂĄ definido
integrando a lo largo del segmento $[0, M]$, que se encuentra en $U$
Precisamente porque $U$ tiene forma de estrella.
\end{solution}

\begin{solution}{exo:b2:multint:9}
En la tira $[0, A] \times [0, \infty)$ la funciĂłn $(x, y)
\mapsto e^{-xy}\sin x $no es \omterm{def:b2:series:def}{absolutamente} integrable hasta $y =
\infty$ uniformemente en un sentido ingenuo, pero cada integral iterada
converge y su igualdad se sigue de Fubini en $[0, A]
\times [0, B]$mĂĄs un lĂ­mite $B \to \infty$ (la cola
$\int_0^A\int_B^\infty e^{-xy}\abs{\sin x}\,\dd y\,\dd x \leq
\int_0^A \frac{e^{-Bx}\abs{\sin x}}{x}\dd x \leq \int_0^A e^{-Bx}
\dd x\to 0$, utilizando $\abs{\sin x} \leq x$).

\emph{$y$ primero:} $\int_0^\infty e^{-xy}\,\dd y = \frac1x$ para $x
> 0$, por lo que la primera integral es $\int_0^A \frac{\sin x}{x}\,\dd x$.

\emph{$x$ primero:} dos integraciones por partes (o tomando la
parte imaginaria de $\int_0^A e^{(i - y)x}\dd x$) dar
\[
\int_0^A e^{-xy}\sin x\,\dd x
= \frac{1 - e^{-Ay}(y\sin A + \cos A)}{1 + y^2} .
\]
Integrando $y$ sobre $[0, \infty)$, el tĂŠrmino $\int_0^\infty
\frac{\dd y}{1 + y^2} = \frac\pi2$ se escinde:
\[
\int_0^A \frac{\sin x}{x}\,\dd x
= \frac{\pi}{2}
- \int_0^\infty e^{-Ay}\,\frac{y\sin A + \cos A}{1 + y^2}\,\dd y .
\]
El resto estĂĄ limitado por $\int_0^\infty e^{-Ay}\frac{y +
1}{1 + y^2}\dd y \leq \int_0^\infty e^{-Ay}\cdot\frac{1+y}{1+y^2}
\,\dd y \to 0$ como $A \to \infty$ (convergencia dominada, o la
crudo atado $\frac{1 + y}{1 + y^2} \leq \frac32$ dando
$\frac{3}{2A}$). Por lo tanto $\int_0^\infty\frac{\sin x}{x}\dd x =
\frac\pi2$ --- el mismo valor obtenido en
\cref{ch:b2:integration} diferenciando un \omterm{thm:b2:integration:continuity}{integral de parĂĄmetros};
aquĂ­ Fubini hace el trabajo.
\end{solution}

\begin{solution}{exo:b2:multint:10}
Parametrizar por \omterm{def:b2:curves:length}{longitud del arco} $s \in [0, 2\pi]$, entonces $x'^2 + y'^2 = 1$,
y traducir para que $\int_0^{2\pi} x(s)\,\dd s = 0$. Por
\cref{cor:b2:multint:area},
\[
2A = \oint x\,\dd y - y\,\dd x
= \int_0^{2\pi}\bigl(xy' - yx'\bigr)\dd s .
\]
Integrando $\oint y\,\dd x$ por partes a lo largo del periodo (lĂ­mite
tĂŠrminos se cancelan por periodicidad), $-\int yx' = \int y'x$, por lo que de hecho
$2A = 2\int_0^{2\pi}xy'\,\dd s$. Entonces $2xy' \leq x^2 + y'^2$
da
\[
2A \leq \int_0^{2\pi}\bigl(x^2 + y'^2\bigr)\dd s
= \int_0^{2\pi} x^2 + \int_0^{2\pi}\bigl(1 - x'^2\bigr)
= 2\pi - \int_0^{2\pi}\bigl(x'^2 - x^2\bigr)\dd s .
\]

$\int x^2 \leq \int x'^2$) hace que la Ăşltima integral no sea negativa:
$A \leq \pi$. La igualdad requiere igualdad en Wirtinger ($x(s) =
a\cos s + b\sin s$) y en $2xy' \leq x^2 + y'^2$($y' = x$
\omterm{def:b2:funcseq:def}{puntualmente}), lo que fuerza a $y = a\sin s - b\cos s + c$: la curva es
el cĂ­rculo unitario (adecuadamente centrado). Desde una curva de \omterm{def:b2:curves:length}{longitud} $L$
reescala a \omterm{def:b2:curves:length}{longitud} $2\pi$, la declaraciĂłn general es $A \leq
\frac{L^2}{4\pi}$: entre todas las curvas cerradas de un perĂ­metro dado, la
El cĂ­rculo encierra el \omterm{def:b2:multint:domain}{ĂĄrea} mĂĄs grande.
\end{solution}

\begin{solution}{pb:b2:multint:1}
\textbf{1.} La pelota $B_n(R)$ se describe mediante lĂ­mites iterados.
$-R \leq x_n \leq R$, luego $\abs{x_{n-1}} \leq \sqrt{R^2 -
x_n^2}$, y asĂ­ sucesivamente; sustituyendo $x_i = Ru_i$ en cada uno de los
$n$ integrales de una variable multiplica cada una por $R$ y asigna
los lĂ­mites a los de $B_n(1)$: $v_n(R) = R^n\,v_n(1) =
V_nR^n$.

\textbf{2.} Cortando a lo largo de $x_n = t$: la rebanada de $B_n(1)$ es
la pelota $B_{n-1}\bigl(\sqrt{1 - t^2}\bigr)$, entonces, por pregunta
1,
\[
V_n = \int_{-1}^1 v_{n-1}\bigl(\sqrt{1 - t^2}\bigr)\,\dd t
= V_{n-1}\int_{-1}^{1}(1 - t^2)^{\frac{n-1}2}\,\dd t .
\]

\textbf{3.} Con $t = \sin\theta$, $\dd t =
\cos\theta\,\dd\theta $y$(1 - t^2)^{\frac{n-1}2} =
\cos^{n-1}\theta $en $\intcc{-\pi/2}{\pi/2}$:
\[
\int_{-1}^1(1 - t^2)^{\frac{n-1}2}\dd t
= \int_{-\pi/2}^{\pi/2}\cos^n\theta\,\dd\theta
= 2\int_0^{\pi/2}\cos^n\theta\,\dd\theta = 2W_n,
\]
y $\theta \mapsto \frac\pi2 - \theta$ intercambia el seno
y formas cosenos de $W_n$.

\textbf{4.} Escribe $\sin^n = \sin^{n-2}(1 - \cos^2)$ y
integrar $\int\sin^{n-2}\cos\cdot\cos$ por partes ($v =
\frac{\sin^{n-1}}{n-1}$):
\[
W_n = W_{n-2} - \frac{W_n}{n-1}
\quad\Longrightarrow\quad
W_n = \frac{n-1}{n}W_{n-2}.
\]
De ahĂ­ $nW_nW_{n-1} = (n-1)W_{n-1}W_{n-2}$: la secuencia
$(nW_nW_{n-1})$ es constante, igual a $1\cdot W_1W_0 =
1\cdot\frac\pi2$, por lo que $W_nW_{n-1} = \frac{\pi}{2n}$.

\textbf{5.} Las preguntas 2--3 dan $V_n = 2W_nV_{n-1}$, dos veces:
\[
V_n = 2W_n\cdot 2W_{n-1}\,V_{n-2}
= 4\,\frac{\pi}{2n}\,V_{n-2} = \frac{2\pi}n\,V_{n-2}.
\]

\textbf{6.} De $V_0 = 1$: $V_{2k} = \frac{2\pi}{2k}V_{2k-2}
= \frac\pi kV_{2k-2}$, entonces $V_{2k} = \frac{\pi^k}{k!}$ por
inducciĂłn. De $V_1 = 2$: $V_{2k+1} =
\frac{2\pi}{2k+1}V_{2k-1}$, entonces
\[
V_{2k+1} = 2\prod_{j=1}^k\frac{2\pi}{2j+1}
= \frac{2^{k+1}\pi^k}{1\cdot3\cdot5\cdots(2k+1)} .
\]

\textbf{7.} $V_1 = 2$, $V_2 = \pi \approx 3.142$, $V_3 =
\frac{4\pi}3 \approx 4.189$, $V_4 = \frac{\pi^2}2 \approx
4.935$, $V_5 = \frac{8\pi^2}{15} \approx 5.264$, $V_6 =
\frac{\pi^3}6 \approx 5.168$, $V_7 = \frac{16\pi^3}{105}
\approx 4.725$. La relaciĂłn de un paso es $V_n/V_{n-1} = 2W_n$,
y $(W_n)$ estĂĄ disminuyendo ($\sin^n \leq \sin^{n-1}$
\omterm{def:b2:funcseq:def}{puntualmente}). Ahora $2W_5 = 2\cdot\frac45\cdot\frac23 =
\frac{16}{15} > 1$ mientras que $2W_6 =
2\cdot\frac56\cdot\frac34\cdot\frac12\cdot\frac\pi2 =
\frac{5\pi}{16} < 1$: los ratios superan $1$ hasta $n = 5$ y
estĂĄn por debajo de $1$ desde $n = 6$ en --- $(V_n)$ aumenta a su
mĂĄximo $V_5$ y luego disminuye.

\textbf{8.} Para $n \geq 13 > 4\pi$: $V_n/V_{n-2} = 2\pi/n <
\tfrac12$, entonces $V_{n} \leq C\cdot 2^{-n/2}$ con un fijo
constante; mejor, para cualquier $q > 0$, $2\pi/n < q^2$ para grandes
$n$, entonces $V_n/q^n \to 0$: la decadencia supera a toda geometrĂ­a
secuencia. La convergencia de $\sum V_n$ se deriva de la relaciĂłn
$V_n/V_{n-2} \to 0$ (compĂĄrese con una serie geomĂŠtrica de
algĂşn rango en).

\textbf{9.} $\sum_{k\geq0}V_{2k}x^{2k} =
\sum_{k\geq0}\frac{(\pi x^2)^k}{k!} = \eu^{\pi x^2}$, el
serie exponencial (\cref{ch:b2:powerseries}), convergente para
cada $x$. En $x = 1$: $\sum_kV_{2k} = \eu^\pi \approx
23.14$.

\textbf{10.} En el cubo $\intcc{-R}R^n$ el integrando es el
producto $\prod_i\eu^{-x_i^2}$, por lo que la integral iterada
factoriza: $\bigl(\int_{-R}^R\eu^{-t^2}\dd t\bigr)^n$.
Dejar $R \to \infty$ y usar
$\int_\R\eu^{-t^2}\dd t = \sqrt\pi$
(\cref{ex:b2:multint:polar}): $I_n = \pi^{n/2}$.

\textbf{11.} $t = u^2$ da $\Gamma(\tfrac12) =
\int_0^\infty t^{-1/2}\eu^{-t}\dd t =
2\int_0^\infty\eu^{-u^2}\dd u = \sqrt\pi$. Iterando
$\Gamma(s+1) = s\Gamma(s)$: $\Gamma(k+1) = k!\,\Gamma(1) =
k!$, y
\[
\Gamma\Bigl(k + \frac32\Bigr)
= \Bigl(k + \frac12\Bigr)\Bigl(k - \frac12\Bigr)\cdots
\frac12\cdot\Gamma\Bigl(\frac12\Bigr)
= \frac{(2k+1)(2k-1)\cdots1}{2^{k+1}}\,\sqrt\pi .
\]

\textbf{12.} Establecer $F_n = \pi^{n/2}/\Gamma(\frac n2 + 1)$.
Desde $\Gamma(\frac n2 + 1) = \frac n2\,\Gamma(\frac n2) =
\frac n2\,\Gamma(\frac{n-2}2 + 1)$, obtenemos $F_n =
\frac{2\pi}nF_{n-2}$: la misma recursividad que $V_n$ (pregunta
5). Bases: $F_1 = \sqrt\pi/\Gamma(\frac32) =
\sqrt\pi/(\frac{\sqrt\pi}2) = 2 = V_1$y $F_2 =
\pi/\Gamma(2) = \pi = V_2$. Por inducciĂłn $V_n = F_n$ para todos
$n$; La pregunta 11 convierte esto nuevamente en los dos \omterm{def:b2:multint:exact}{formas cerradas} de
pregunta 6.

\textbf{13.} Con $r = \sqrt t$: $\int_0^\infty
\eu^{-r^2}r^{n-1}\dd r = \frac12\int_0^\infty
t^{\frac n2 - 1}\eu^{-t}\dd t = \frac12\Gamma(\frac n2)$.
Por lo tanto
\[
n\,V_n\int_0^\infty\eu^{-r^2}r^{n-1}\dd r
= V_n\cdot\frac n2\,\Gamma\Bigl(\frac n2\Bigr)
= V_n\,\Gamma\Bigl(\frac n2 + 1\Bigr) = \pi^{n/2} = I_n .
\]
Una vez probadas ambas partes, la identidad puede ser \emph{leer} como
la descomposiciĂłn de capas de la integral gaussiana: la esfera
de radio $r$ lleva \omterm{def:b2:multint:domain}{ĂĄrea} $nV_nr^{n-1}$, y el gaussiano
El peso $\eu^{-r^2}$ estĂĄ integrado sobre las carcasas.

\textbf{14.} $s_0 = V_1 = 2$ (la esfera $0$ tiene dos puntos),
$s_1 = 2V_2 = 2\pi$, $s_2 = 3V_3 = 4\pi$, $s_3 = 4V_4 =
2\pi^2$; y $\int_0^R s_{n-1}r^{n-1}\dd r = V_nR^n =
v_n(R)$: \omterm{def:b2:multint:domain}{ĂĄrea} es la derivada radial del volumen.

\textbf{15.} Para $n = 2k$, Stirling
(\cref{thm:b2:comparison:stirling}) da $k! \sim
\sqrt{2\pi k}\,(k/\eu)^k$, entonces
\[

\sim \frac{(\pi\eu/k)^k}{\sqrt{2\pi k}}
= \frac1{\sqrt{\pi n}}\Bigl(\frac{2\pi\eu}n\Bigr)^{n/2}
