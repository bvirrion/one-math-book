\begin{solution}{exo:b2:hermitian:1}
$\langle x, y\rangle = \conj{1}\cdot\iu + \conj{\iu}\cdot 1 = \iu -
\iu = 0$: ortogonal.$\norm x = \norm y = \sqrt{1 + 1} = \sqrt2$.
Base ortonormal: $\bigl(\frac{1}{\sqrt2}(1, \iu),\;
\frac{1}{\sqrt2}(\iu, 1)\bigr)$ --- el par normalizado en sí.
\end{solution}

\begin{solution}{exo:b2:hermitian:2}
Primero: es igual a su conjugado \omterm{def:b2:linalg:transpose}{transponer} (diagonal real, $\conj{\iu} =
-\iu $intercambiada): \omterm{def:b2:hermitian:adjoint}{hermitiano} (por lo tanto normal); no \omterm{def:b2:hermitian:adjoint}{unitario} ($A^\dagger A
\neq I$: columnas no unidad).

Segundo: $A^\dagger A = \frac12\begin{pmatrix} 1 & -\iu\\ -\iu &
1\end{pmatrix}\begin{pmatrix} 1 & \iu\\ \iu & 1\end{pmatrix} =
\frac12\begin{pmatrix} 2 & 0\\ 0 & 2\end{pmatrix} = I$: \omterm{def:b2:hermitian:adjoint}{unitario}
(por lo tanto normal); no \omterm{def:b2:hermitian:adjoint}{hermitiano}.

Tercero: $A^\dagger A = E_{22} \neq E_{11} = AA^\dagger$: no normal
(por lo que ni \omterm{def:b2:hermitian:adjoint}{hermitiano} ni \omterm{def:b2:hermitian:adjoint}{unitario}) --- el nilpotente estándar
contraejemplo.
\end{solution}

\begin{solution}{exo:b2:hermitian:3}
Unicidad: $A = H + \iu K$ con $H^\dagger = H$, $K^\dagger = K$
fuerza a $A^\dagger = H - \iu K$, por lo que $H = \frac{A + A^\dagger}{2}$,
$K = \frac{A - A^\dagger}{2\iu}$; estas fórmulas son \omterm{def:b2:hermitian:adjoint}{hermitiano}
(verificar: $\bigl(\frac{A - A^\dagger}{2\iu}\bigr)^\dagger = \frac{
A^\dagger - A}{-2\iu} = K$) y reconstruir $A$: existencia.

Normalidad: $A^\dagger A - AA^\dagger = (H - \iu K)(H + \iu K) - (H
+ \iu K)(H - \iu K) = 2\iu(HK - KH)$: desaparece si $HK = KH$.
\end{solution}

\begin{solution}{exo:b2:hermitian:4}
$\chi_A = (X-2)^2 - 1$: \omterm{def:b2:reduction:eigen}{valores propios} $3$ y $1$. \omterm{def:b2:reduction:eigen}{Vectores propios}, por
cálculo directo:
\[
A\begin{pmatrix}1\\ -\iu\end{pmatrix}
= \begin{pmatrix} 2 + \iu(-\iu)\\ -\iu + 2(-\iu)\end{pmatrix}
= \begin{pmatrix} 3\\ -3\iu \end{pmatrix}
= 3\begin{pmatrix}1\\ -\iu\end{pmatrix},
\qquad
A\begin{pmatrix}1\\ \iu\end{pmatrix}
= \begin{pmatrix} 2 + \iu\cdot\iu\\ -\iu + 2\iu\end{pmatrix}
= \begin{pmatrix}1\\ \iu\end{pmatrix}.
\]
Base propia ortonormal: $u_1 = \frac{1}{\sqrt2}(1, -\iu)$
(\omterm{def:b2:reduction:eigen}{valor propio} $3$), $u_2 = \frac{1}{\sqrt2}(1, \iu)$ (\omterm{def:b2:reduction:eigen}{valor propio}
$1$); ortogonalidad como en \cref{exo:b2:hermitian:1}. poderes, a través de
las proyecciones espectrales $A^k = 3^k u_1u_1^\dagger + 1^k\,
u_2u_2^\dagger$:
\[
A^k = U\begin{pmatrix} 3^k & 0\\ 0 & 1\end{pmatrix}U^\dagger
= \frac{3^k}{2}\begin{pmatrix} 1 & \iu\\ -\iu & 1\end{pmatrix}
+ \frac{1}{2}\begin{pmatrix} 1 & -\iu\\ \iu & 1\end{pmatrix}
= \frac12\begin{pmatrix}
3^k + 1 & (3^k - 1)\iu\\
-(3^k-1)\iu & 3^k + 1
\end{pmatrix}.
\]
(Marque $k = 1$: recupera $A$).
\end{solution}

\begin{solution}{exo:b2:hermitian:5}

\begin{solution}{exo:b2:hermitian:6}
Según el teorema espectral, trabaje en una base propia ortonormal de $H$:
todo se reduce a escalares $\lambda \in \R$ (el \omterm{def:b2:reduction:eigen}{valores propios}).
$H + \iu I$ tiene \omterm{def:b2:reduction:eigen}{valores propios} $\lambda + \iu \neq 0$: invertible. $U$
tiene \omterm{def:b2:reduction:eigen}{valores propios} $\mu = \frac{\lambda - \iu}{\lambda + \iu}$, de
módulo $1$ ($\abs{\lambda - \iu} = \abs{\lambda + \iu}$ de verdad
$\lambda$): $U^\dagger U = I$ se mantiene ya que $U$ es unitariamente
\omterm{def:b2:reduction:diag}{diagonalizable} con \omterm{def:b2:reduction:eigen}{valores propios} unimodular (es diagonal en el
base ortonormal elegida). Y $\mu = 1$ forzaría a $-\iu = \iu$:
imposible, entonces $1 \notin \operatorname{Sp} U$. (El Cayley
transformar mapas \omterm{def:b2:hermitian:adjoint}{hermitiano} a unitario menos un punto --- la matriz
versión del mapa de $\R$ al círculo.)
\end{solution}

\begin{solution}{exo:b2:hermitian:7}

\begin{solution}{exo:b2:hermitian:8}
\begin{enumerate}
 \item $\norm{u(x)}^2 = \langle u(x), u(x)\rangle = \langle x,
 u^*u(x)\rangle = \langle x, uu^*(x)\rangle =
 \norm{u^*(x)}^2$. Aplicando esto al $u -
 \lambda\,\mathrm{id}$ normal (su \omterm{def:b2:hermitian:adjoint}{adjunto} es $u^* -
 \conj\lambda$, y la normalidad se hereda): $\norm{(u -
 \lambda)x} = \norm{(u^* - \conj\lambda)x}$, entonces los núcleos
 de acuerdo.
 \item Para \omterm{def:b2:reduction:eigen}{vectores propios} $u(x) = \lambda x$, $u(y) = \mu y$
 ($\lambda \neq \mu$): usando (1), $u^*(x) = \conj\lambda x$;
 entonces
 \[
 \lambda\langle y, x\rangle = \langle y, u(x)\rangle
 = \langle u^*(y), x\rangle = \langle \conj\mu\, y, x\rangle
 = \mu \langle y, x\rangle ,
 \]
 entonces $\langle y, x\rangle = 0$. Estabilidad de $E_\lambda^\perp$:
 para $x \perp E_\lambda$ y $z \in E_\lambda$, $\langle z,
 u(x)\rangle = \langle u^*(z), x\rangle =
 \langle\conj\lambda z, x\rangle = 0$.
 \item Inducción en dimensión: sobre $\C$, $u$ tiene un \omterm{def:b2:reduction:eigen}{vector propio}
 $e_1$ (normalizar); su complemento ortogonal es estable
 bajo $u$ (por (2)) \emph{y} bajo $u^*$ (mismo argumento
 con roles intercambiados), por lo que la restricción es normal: inducir
 y concatenar bases propias ortonormales. Por el contrario, un
 unitariamente \omterm{def:b2:reduction:diag}{diagonalizable} $u = UDU^\dagger$ satisface $u^*u
 = U\conj D D U^\dagger = U D\conj D U^\dagger = uu^*$:
 normales.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b2:hermitian:9}
\omterm{def:b2:reduction:eigen}{Valores propios}: para $u(x) = \lambda x$, $x \neq 0$:
\[
\lambda\norm x^2 = \langle x, u(x)\rangle
= \langle u^*(x), x\rangle = -\langle u(x), x\rangle
= -\conj{\langle x, u(x)\rangle} = -\conj\lambda\,\norm x^2 ,
\]
entonces $\lambda = -\conj\lambda$: puramente imaginario. Desde $(\iu
u)^* = -\iu\,u^*$ (el \omterm{def:b2:hermitian:adjoint}{adjunto} es lineal conjugado en escalares),
$u^* = u$ da $(\iu u)^* = -\iu u$: el mapa $u \mapsto \iu u$
envía \omterm{def:b2:hermitian:adjoint}{hermitiano} a \omterm{def:b2:hermitian:adjoint}{sesgado-hermitiano}, con $w \mapsto -\iu
w $inverso: una biyección. A \omterm{def:b2:hermitian:adjoint}{sesgado-hermitiano}$u $satisface $u^*u = -u^2 =
uu^*$: normal, por lo tanto unitariamente \omterm{def:b2:reduction:diag}{diagonalizable} por
\cref{exo:b2:hermitian:8}.
\end{solution}

\begin{solution}{exo:b2:hermitian:11}
($\Leftarrow$) Vamos $P^2 = P = P^*$. Cada $v$ se divide como $v = Pv
+ (v - Pv)$con $Pv \in \operatorname{im} P$y $P(v - Pv) =
0$. Las dos piezas son ortogonales: para cualquier $x, y$,
\[
\langle Px, (I - P)y\rangle = \langle x, P(I-P)y\rangle
= \langle x, (P - P^2)y\rangle = 0 :
\]
$\ker P \perp \operatorname{im} P$, entonces $P$ es el ortogonal
proyección sobre su imagen. ($\Rightarrow$) Si $P$ es el
proyección ortogonal sobre $F = \operatorname{im}P$: para todos
$x, y$, $\langle Px, y\rangle = \langle Px, Py\rangle$ (el
componente $y - Py \perp F$ cae) y simétricamente $\langle
x, Py\rangle = \langle Px, Py\rangle$: $\langle Px, y\rangle =
\langle x, Py\rangle$, es decir $P^* = P$. Matrices: sobre $\C v$
($\norm v = 1$): $Px = v\,\langle v, x\rangle$, es decir\ $P =
vv^\dagger$; en $\operatorname{Vect}(v_1, \dots, v_k)$
ortonormal: $P = \sum_i v_iv_i^\dagger$.
\end{solution}

\begin{solution}{exo:b2:hermitian:12}
Diagonalizar $A = U D U^\dagger$ (teorema espectral), $D$ diagonal
con entradas entre el $\lambda_i$. Entonces $P_i = L_i(A) = U
L_i(D)U^\dagger $y$L_i(D)$ son diagonales con entradas
$L_i(\lambda_j) = \delta_{ij}$: los que están exactamente en las ranuras de
$E_i$. Entonces $P_i$ es \omterm{def:b2:hermitian:adjoint}{hermitiano} ($L_i$ real, $D$ real), idempotente,
con imagen $E_i$ y kernel $\bigoplus_{j\neq i}E_j = E_i^\perp$

\begin{solution}{pb:b2:hermitian:1}
\textbf{1.} $\conj{\langle x, Ax\rangle} = \langle Ax, x\rangle
= \langle x, A^*x\rangle = \langle x, Ax\rangle$: real. escribiendo
$x = \sum c_ie_i$:
\[
R_A(x) = \frac{\sum_i\lambda_i\abs{c_i}^2}
{\sum_i\abs{c_i}^2} ,
\]
un promedio ponderado del \omterm{def:b2:reduction:eigen}{valores propios}: se encuentra en
$\intcc{\lambda_n}{\lambda_1}$, con los límites alcanzados en
$e_1$ y $e_n$.

\textbf{2.} Para $t$ real y cualquier $v$, expanda $R_A(x + tv) =
\frac{N(t)}{D(t)}$ con
\[
N(t) = \langle x, Ax\rangle + 2t\Re\langle v, Ax\rangle +
t^2\langle v, Av\rangle,
\quad
D(t) = \norm x^2 + 2t\Re\langle v, x\rangle + t^2\norm v^2 .
\]
La derivada en $t = 0$ es
\[
\frac{2}{\norm x^2}\,
\Re\bigl\langle v,\ Ax - R_A(x)\,x\bigr\rangle .
\]
Desaparece para todos $v$ si $\Re\langle v, w\rangle = 0$ para
todos $v$, donde $w = Ax - R_A(x)x$; reemplazando $v$ por $\iu v$
mata también la parte imaginaria: $w = 0$, es decir\ $Ax = R_A(x)x$.
Los puntos críticos de $R_A$ son exactamente \omterm{def:b2:reduction:eigen}{vectores propios}, con
valor crítico el \omterm{def:b2:reduction:eigen}{valor propio}.

\textbf{3.} Para $x = \sum_{i\leq k}c_ie_i \in V_k$: $R_A(x)$ es
un promedio ponderado de $\lambda_1, \dots, \lambda_k$, por lo tanto
$\geq \lambda_k$, con igualdad en $e_k$: $\min_{V_k} R_A =
\lambda_k$. Simétricamente en $W_k$ el promedio involucra
$\lambda_k, \dots, \lambda_n$: $\max_{W_k}R_A = \lambda_k$.

\textbf{4.} Vamos $\dim V = k$. Entonces $\dim V + \dim W_k = n + 1 >
n$, por lo que hay una unidad $x \in V \cap W_k $y$R_A(x) \leq
\lambda_k $(pregunta 3): $\min_{V}R_A \leq \lambda_k$ para cada
tal $V$. Dado que $V_k$ alcanza $\lambda_k$, el máximo-mínimo es igual
$\lambda_k$. La fórmula min-max es el mismo argumento con la
roles invertidos ($\dim W = n - k + 1$ fuerza a $W \cap V_k \neq
\{0\}$, por lo que $\max_W R_A \geq \lambda_k $se alcanza en $W_k$).

\textbf{5.} $R_B(x) = R_A(x) + \frac{\langle x,
(B-A)x\rangle}{\norm x^2} \geq R_A(x)$ \omterm{def:b2:funcseq:def}{puntualmente}. tomando
$\min$ sobre cualquier $k$-dimensional $V$ y luego $\max$ sobre $V$:
$\lambda_k(B) \geq \lambda_k(A)$ por la pregunta 4.

\textbf{6.} Grassmann: $\dim(V\cap W) = \dim V + \dim W -
\dim(V + W) \geq \dim V + \dim W - n > 0$. Aplicando esto dos veces:
\[
\dim(V_1\cap V_2\cap V_3)
\geq \dim(V_1\cap V_2) + \dim V_3 - n
\geq \dim V_1 + \dim V_2 + \dim V_3 - 2n .
\]

\textbf{7.} Los subespacios $W_i(A)$, $W_j(B)$ (pregunta 3, para
$A$ y $B$) y $V_{i+j-1}(A+B)$ tienen dimensiones $(n-i+1) +
(n-j+1) + (i+j-1) = 2n + 1 > 2n$: por la pregunta 6 hay una unidad
