\begin{solution}{exo:b2:integration:1}
$\int_0^1 \frac{\dd t}{\sqrt{t(1-t)}}$: cerca de $0$, $\sim t^{-1/2}$
($\alpha = \frac12 < 1$: \omterm{def:b2:integration:improper}{converge}); cerca de $1$, $\sim
(1-t)^{-1/2}$: \omterm{def:b2:integration:improper}{converge}. Convergente (su valor es $\pi$, por el
sustituciﺃ٣n $t = \sin^2\theta$).

$\int_1^\infty \frac{\ln t}{t^2}$: $\frac{\ln t}{t^2} = o(t^{-3/2})$:
convergente (valor $1$ por partes).

$\int_0^\infty \frac{\dd t}{1 + t^2\sin^2 t}$: divergente. Cerca de $t =
n\pi$, escriba $t = n\pi + u$: $\sin^2 t = \sin^2 u \leq u^2$, asﺃﺝ sucesivamente
$\abs u \leq \frac{1}{n}$, $1 + t^2\sin^2 t \leq 1 + (n\pi +
1)^2u^2 \leq C n^2 u^2 + 1$; por lo tanto
\[
\int_{n\pi - 1/n}^{n\pi + 1/n} \frac{\dd t}{1 + t^2\sin^2 t}
\geq \int_{-1/n}^{1/n} \frac{\dd u}{1 + Cn^2u^2}
= \frac{2\arctan\sqrt C}{\sqrt C}\cdot\frac{1}{n} ,
\]
un tﺃﺏrmino de una serie de tipo armﺃ٣nico divergente: sumando $n$, el
primitivo es ilimitado.
\end{solution}

\begin{solution}{exo:b2:integration:2}
Sustituir $u = \lambda t$:
\[
\int_0^\infty t^n \eu^{-\lambda t}\dd t
= \frac{1}{\lambda^{n+1}}\int_0^\infty u^n\eu^{-u}\dd u
= \frac{\Gamma(n+1)}{\lambda^{n+1}} = \frac{n!}{\lambda^{n+1}} .
\]
Con $t = \eu^{-u}$ ($\dd t = -\eu^{-u}\dd u$):
\[
\int_0^1 (\ln t)^n \dd t = \int_0^{\infty} (-u)^n \eu^{-u}\,\dd u
= (-1)^n\, n! .
\]
\end{solution}

\begin{solution}{exo:b2:integration:3}
Convergencia: el integrando es $\leq \frac{1}{1+t^2}$ cerca de $\infty$
y acotado cerca de $0$ (ambos factores acotados a continuaciﺃ٣n lejos de $0$):
\omterm{def:b2:series:def}{absolutamente} convergente, por cada $x$. Sustituyendo $t = \frac1u$
($\dd t = -\frac{\dd u}{u^2}$):
\[
I(x) = \int_0^\infty \frac{1}{\bigl(1 +
\frac1{u^2}\bigr)\bigl(1 + u^{-x}\bigr)}\cdot\frac{\dd u}{u^2}
= \int_0^\infty \frac{u^x}{(1 + u^2)(1 + u^x)}\,\dd u .
\]
Sumando las dos expresiones de $I(x)$:
\[
2I(x) = \int_0^\infty \frac{1 + t^x}{(1+t^2)(1+t^x)}\dd t
= \int_0^\infty \frac{\dd t}{1 + t^2} = \frac{\pi}{2} :
\]
$I(x) = \frac\pi4$, independiente de $x$.
\end{solution}

\begin{solution}{exo:b2:integration:4}
Cerca de $0^+$, con $u = \abs{\ln t} \to \infty$. Si $\alpha < 1$:
convergencia independientemente de $\beta$ (compﺃ­rese con $t^{-\alpha'}$ para
$\alpha < \alpha' < 1$: se supera el factor logarﺃﺝtmico). Si $\alpha > 1$:
divergencia independientemente de $\beta$ (compﺃ­rese con $t^{-\alpha''}$,
$1 < \alpha'' < \alpha$). Si $\alpha = 1$: sustituir $t =
\eu^{-u}$:
\[
\int_0^{1/2} \frac{\dd t}{t\,\abs{\ln t}^\beta}
= \int_{\ln 2}^{\infty} \frac{\dd u}{u^\beta},
\]
convergente si $\beta > 1$. Resumen: convergencia si $\alpha < 1$,
o ($\alpha = 1$ y $\beta > 1$) --- el espejo del Bertrand
serie.
\end{solution}

\begin{solution}{exo:b2:integration:5}
\omterm{def:b2:metric:continuity}{Continuidad} en $\intco{0}{\infty}$: dominaciﺃ٣n
$\bigl|\frac{\eu^{-xt}}{1+t^2}\bigr| \leq \frac{1}{1+t^2}$,
integrable, uniforme en $x \geq 0$:
\cref{thm:b2:integration:continuity}.

$C^2$ en $\intoo{0}{\infty}$: en $x \geq a > 0$, los dos primeros
$x$-derivados $\frac{-t\,\eu^{-xt}}{1+t^2}$ y
$\frac{t^2\eu^{-xt}}{1+t^2}$ estﺃ­n dominados por $t\,\eu^{-at}$ y
$\eu^{-at}$: dos aplicaciones de
\cref{thm:b2:integration:leibnizrule}. entonces
\[
F''(x) + F(x) = \int_0^\infty \frac{t^2 + 1}{1 + t^2}\,\eu^{-xt}\dd
t = \int_0^\infty \eu^{-xt}\dd t = \frac1x .
\]
Lﺃﺝmite: $0 \leq F(x) \leq \int_0^\infty \eu^{-xt}\dd t = \frac1x \to
0$.
\end{solution}

\begin{solution}{exo:b2:integration:6}
En $\intcc{\varepsilon}{M}$, sustituya $u = at$ y $u = bt$ en
las dos mitades:
\[
\int_\varepsilon^M \frac{f(at) - f(bt)}{t}\dd t
= \int_{a\varepsilon}^{aM}\frac{f(u)}{u}\dd u
- \int_{b\varepsilon}^{bM}\frac{f(u)}{u}\dd u
= \int_{a\varepsilon}^{b\varepsilon} \frac{f(u)}{u}\dd u
- \int_{aM}^{bM} \frac{f(u)}{u}\dd u .
\]
Primera pieza: $f(u) = f(0) + o(1)$ cerca de $0$, y $\int_{a\varepsilon}
^{b\varepsilon} \frac{\dd u}{u} = \ln\frac ba$: la pieza tiende a
$f(0)\ln\frac ba$. Segunda pieza: $f(u) \to f(\infty)$, igual
Cﺃ­lculo: tiende a $f(\infty)\ln\frac ba$. De ahﺃﺝ el \omterm{def:b2:integration:improper}{inadecuado
integral} \omterm{def:b2:integration:improper}{converge} a $\bigl(f(0) - f(\infty)\bigr)\ln\frac ba$.

Con $f(t) = \eu^{-t}$ ($f(0) = 1$, $f(\infty) = 0$), $a = 1$, $b
= 2$:
\[
\int_0^\infty \frac{\eu^{-t} - \eu^{-2t}}{t}\dd t = \ln 2 .
\]
\end{solution}

\begin{solution}{exo:b2:integration:7}
Extienda el integrando en $0$ mﺃ­s allﺃ­ de $t = n$: $g_n(t) = (1 -
\frac tn)^n t^{x-1}\mathbf{1}_{t \leq n}$. Puntualmente, $g_n(t) \to
\eu^{-t}t^{x-1}$ (el lﺃﺝmite de interﺃﺏs compuesto, volumen del aﺃ١o 1).
Dominaciﺃ٣n: $\ln(1 - u) \leq -u$ da $(1 - \frac tn)^n \leq
\eu^{-t}$ sobre $\intcc{0}{n}$, por lo que $\abs{g_n(t)} \leq
\eu^{-t}t^{x-1} = \varphi(t)$, integrable. Convergencia dominada:
\[
\int_0^n \Bigl(1 - \frac tn\Bigr)^n t^{x-1}\dd t
\xrightarrow[n\to\infty]{} \int_0^\infty \eu^{-t}t^{x-1}\dd t
= \Gamma(x) .
\]
(Al calcular el lado izquierdo por partes repetidas se obtiene el producto de Euler
formulario $\Gamma(x) = \lim \frac{n!\,n^x}{x(x+1)\cdots(x+n)}$.)
\end{solution}

\begin{solution}{exo:b2:integration:8}
$H$ es diferenciable en $x$ (integrando $C^1$ en $x$, derivada
$-2x(1+t^2)\cdot\frac{\eu^{-x^2(1+t^2)}}{1+t^2} =
-2x\,\eu^{-x^2}\eu^{-x^2t^2}$, \omterm{def:b2:metric:continuity}{continuo} y acotado en \omterm{def:b2:metric:compact}{compactos}
de $x$, dominio sobre $t \in \intcc{0}{1}$ trivial):
\[
H'(x) = -2x\,\eu^{-x^2}\int_0^1 \eu^{-x^2t^2}\,\dd t
\overset{u = xt}{=} -2\,\eu^{-x^2}\int_0^x \eu^{-u^2}\,\dd u
= -G'(x),
\]
desde $G'(x) = 2\eu^{-x^2}\int_0^x \eu^{-t^2}\dd t$ (regla de cadena en
el cuadrado, teorema fundamental del cﺃ­lculo). Entonces $G + H$ es
constante, igual a $G(0) + H(0) = 0 + \int_0^1 \frac{\dd t}{1+t^2}
= \frac\pi4$.

Como $x \to \infty$: $0 \leq H(x) \leq \eu^{-x^2}\int_0^1 \dd t \to
0$, entonces $G(x) \to \frac\pi4$:
\[
\int_0^\infty \eu^{-t^2}\dd t = \sqrt{\frac\pi4} =
\frac{\sqrt\pi}{2} .
\]
(En consecuencia $\Gamma\bigl(\frac12\bigr) = 2\int_0^\infty
\eu^{-t^2}\dd t = \sqrt\pi$, por la sustituciﺃ٣n $t = \sqrt u$.)
\end{solution}

\begin{solution}{exo:b2:integration:9}
Cauchy--Schwarz (volumen del aﺃ١o 1, vﺃ­lido en $\intcc{\varepsilon}{M}$
y pasado al lﺃﺝmite) aplicado a la factorizaciﺃ٣n
$t^{\frac{x+y}{2}-1}\eu^{-t} = \bigl(t^{x-1}\eu^{-t}\bigr)^{1/2}
\bigl(t^{y-1}\eu^{-t}\bigr)^{1/2}$:
\[
\Gamma\Bigl(\frac{x+y}{2}\Bigr) \leq
\Gamma(x)^{1/2}\,\Gamma(y)^{1/2}
\quad\Longrightarrow\quad
\ln\Gamma\Bigl(\frac{x+y}{2}\Bigr) \leq \frac{\ln\Gamma(x) +
\ln\Gamma(y)}{2} :
\]
$\ln\Gamma$ es el punto medio convexo; siendo \omterm{def:b2:metric:continuity}{continuo}
(\cref{thm:b2:integration:gammaprops}), es convexo
(\cref{exo:b2:realfun:8}). (La convexidad logarﺃﺝtmica fija $\Gamma$ hacia abajo
ﺃﻑnicamente entre las interpolaciones del factorial --- el
Teorema de Bohr-Mollerup, una perla del aﺃ١o 3.)
\end{solution}

\begin{solution}{exo:b2:integration:10}
\begin{enumerate}
 \item En $x \geq a > 0$: la derivada $x$ del integrando es
 $-\sin t\,\eu^{-xt}$, dominado por $\eu^{-at}$:
 \cref{thm:b2:integration:leibnizrule} da $F'(x) =
 -\int_0^\infty \eu^{-xt}\sin t\,\dd t$. Dos integraciones por
 partes (o el exponencial complejo):
 \[
 \int_0^\infty \eu^{-xt}\sin t\,\dd t
 = \Im \int_0^\infty \eu^{(-x+\iu)t}\dd t
 = \Im\frac{1}{x - \iu} = \frac{1}{1 + x^2} .
 \]
 \item $\abs{F(x)} \leq \int_0^\infty \eu^{-xt}\dd t = \frac1x \to
 0$. Integrando $F' = -\frac{1}{1+x^2}$ de $x$ a
 $\infty$: $0 - F(x) = -\bigl(\frac\pi2 - \arctan x\bigr)$,
 entonces $F(x) = \frac\pi2 - \arctan x$.
 \item Dejemos $x \to 0^+$ con el \omterm{def:b2:metric:continuity}{continuidad} admitido: $F(0^+) =
 \frac\pi2$y $F(0) = \int_0^\infty \frac{\sin t}{t}\dd
 t$ (la integral de Dirichlet semiconvergente,
 \cref{ex:b2:integration:sint}): su valor es
 $\frac\pi2$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b2:integration:11}
Convergencia: cerca de $0$ el integrando extiende a \omterm{def:b2:metric:continuity}{continuamente} por el
valor $1$ ($\sin t \sim t$); en el infinito es $\leq t^{-2}$:
convergencia absoluta. En $\intcc{\varepsilon}{M}$, integrar por
piezas con $u = \sin^2 t$, $v' = t^{-2}$:
\[
\int_\varepsilon^M \frac{\sin^2 t}{t^2}\dd t
= \Bigl[-\frac{\sin^2 t}{t}\Bigr]_\varepsilon^M
+ \int_\varepsilon^M \frac{2\sin t\cos t}{t}\dd t
= \Bigl[-\frac{\sin^2 t}{t}\Bigr]_\varepsilon^M
+ \int_{2\varepsilon}^{2M} \frac{\sin u}{u}\dd u
\]
($u = 2t$ en la ﺃﻑltima integral). El soporte tiende a $0$ en ambos
termina ($\sin^2\varepsilon/\varepsilon \leq \varepsilon$;
$\sin^2 M/M \leq 1/M$), y la ﺃﻑltima integral tiende a
$\int_0^\infty \frac{\sin u}{u}\dd u = \frac\pi2$
(\cref{exo:b2:integration:10}). Por lo tanto
\[
\int_0^\infty \Bigl(\frac{\sin t}{t}\Bigr)^{\!2}\dd t
= \frac\pi2 .
\]
\end{solution}

\begin{solution}{pb:b2:integration:1}
\textbf{1.} En $0^+$ el integrando es $\sim t^{x-1}$: el
escala de punto final finito \omterm{def:b2:integration:improper}{converge} iff $1 - x < 1$, es decir,\ $x > 0$
(y para $x \leq 0$, $t^{x-1} \geq t^{-1}$ diverge); en
$+\infty$, $t^{x-1}\eu^{-t} = o(t^{-2})$ \omterm{def:b2:integration:improper}{converge} para cada
$x$. Luego $\Gamma(x) = \frac{\Gamma(x+1)}{x}$ y $\Gamma(x+1)
\to \Gamma(1) = 1$ como $x \to 0^+$ (\omterm{def:b2:metric:continuity}{continuidad},
\cref{thm:b2:integration:gammaprops}): $\Gamma(x) \sim
\frac1x$.

\textbf{2.} Con $t = u^2$, $\dd t = 2u\,\dd u$:
\[
\Gamma\Bigl(\frac12\Bigr) = \int_0^\infty
t^{-1/2}\eu^{-t}\dd t
= \int_0^\infty \frac{\eu^{-u^2}}{u}\,2u\,\dd u
= 2\int_0^\infty \eu^{-u^2}\dd u = \sqrt\pi
\]
por \cref{exo:b2:integration:8}. Con $u = v/\sqrt2$:
\[
\int_\R \eu^{-v^2/2}\dd v
= 2\sqrt2\int_0^\infty \eu^{-u^2}\dd u
= \sqrt2\,\sqrt\pi = \sqrt{2\pi} .
\]

\textbf{3.} Verdadero para $n = 0$ (ambos lados $\sqrt\pi$). si
$\Gamma(n + \frac12) = \frac{(2n)!}{4^n n!}\sqrt\pi$, el
ecuaciﺃ٣n funcional da
\[
\Gamma\Bigl(n + 1 + \frac12\Bigr)
= \Bigl(n + \frac12\Bigr)\Gamma\Bigl(n + \frac12\Bigr)
= \frac{2n+1}{2}\cdot\frac{(2n)!}{4^n n!}\sqrt\pi
= \frac{(2n+2)!}{4^{n+1}(n+1)!}\sqrt\pi ,
\]
el ﺃﻑltimo paso porque $\frac{(2n+2)!}{(2n)!} = (2n+2)(2n+1)$
y $\frac{2n+1}{2} = \frac{(2n+2)(2n+1)}{4(n+1)}$.

\textbf{4.} \cref{thm:b2:integration:gammaprops} da
$\Gamma''(x) = \int_0^\infty t^{x-1}\eu^{-t}(\ln t)^2\dd t$
(dos aplicaciones de la regla de Leibniz, dominaciones como en el
demostraciﺃ٣n del teorema); el integrando es $\geq 0$ y no es idﺃﺏntico
cero, entonces $\Gamma'' > 0$: $\Gamma$ es estrictamente convexo y
$\Gamma'$ es estrictamente creciente. Desde $\Gamma(1) = \Gamma(2)
= 1$, Rolle proporciona $x_0 \in \intoo12$ con $\Gamma'(x_0) =
0$; La estricta monotonicidad de $\Gamma'$hace que $x_0$ sea ﺃﻑnico.
cero, con $\Gamma' < 0$ antes y $\Gamma' > 0$ despuﺃﺏs:
$\Gamma$ disminuye en $\intoo0{x_0}$, aumenta en
$\intoo{x_0}\infty$ y $x_0$ es el mﺃﺝnimo ﺃﻑnico.

\textbf{5.} Dejemos que $k \in \N$ y $x \geq 3$; elige el numero entero
$n$ con $n + 1 \leq x < n + 2$ (es decir, $n \geq 1$). por el
monotonﺃﺝa de la pregunta 4 (vﺃ­lida desde $x_0 < 2$ en adelante): $\Gamma(x)
\geq \Gamma(n + 1) = n!$, mientras que $x^k \leq (n+2)^k$. desde
$\frac{n!}{(n+2)^k} \to \infty$ (factoriales vencen a las potencias, Aﺃ١o 1
volumen), $\frac{\Gamma(x)}{x^k} \geq \frac{n!}{(n+2)^k} \to
\infty $como $x \to \infty$: $x^k = o(\Gamma(x))$.

\textbf{6.} Cerca de $0$ el integrando es $\sim t^{x-1}$
(iff convergente $x > 0$), cerca de $1$ es $\sim (1-t)^{y-1}$
(iff $y > 0$); ambas comparaciones son entre funciones positivas,
entonces $B(x,y)$ \omterm{def:b2:integration:improper}{converge} exactamente para $x, y > 0$. la sustitucion
$t \mapsto 1 - t$ intercambia los dos factores: $B(x,y) = B(y,x)$.

\textbf{7.} $B(x,1) = \int_0^1 t^{x-1}\dd t = \frac1x$. Piezas
en $\intcc\varepsilon{1-\varepsilon}$ con $u = (1-t)^y$, $v =
\frac{t^x}{x}$:
\[
\int t^{x-1}(1-t)^{y}\dd t
= \Bigl[\frac{t^x(1-t)^y}{x}\Bigr]
+ \frac{y}{x}\int t^{x}(1-t)^{y-1}\dd t ;
\]
el soporte desaparece en ambos extremos como $\varepsilon \to 0$ ($x >
0$ en $0$, $y > 0$ en $1$), dejando $B(x, y+1) = \frac
yx\,B(x+1, y)$.

\textbf{8.} Desde $t + (1-t) = 1$:
\[
t^{x-1}(1-t)^{y-1} = t^{x}(1-t)^{y-1} + t^{x-1}(1-t)^{y},
\]
entonces $B(x,y) = B(x+1,y) + B(x,y+1)$. La pregunta 7 dice $B(x+1,y) =
\frac xy B(x,y+1)$; sustituyendo,
\[
B(x,y) = \Bigl(\frac xy + 1\Bigr)B(x,y+1)
= \frac{x+y}{y}\,B(x,y+1),
\]
es decir\ $B(x,y+1) = \frac{y}{x+y}B(x,y)$; la relaciﺃ٣n gemela
sigue por la simetrﺃﺝa de la pregunta 6.

\textbf{9.} Inducciﺃ٣n en $n$ en $m$ fijo: $B(m,1) = \frac1m =
\frac{(m-1)!\,0!}{m!}$, y si la fﺃ٣rmula se mantiene en $n$,
\[
B(m, n+1) = \frac{n}{m+n}\,B(m,n)
= \frac{n}{m+n}\cdot\frac{(m-1)!(n-1)!}{(m+n-1)!}
= \frac{(m-1)!\,n!}{(m+n)!} .
\]
Reescritura: $B(m,n) = \frac{(m-1)!(n-1)!}{(m+n-1)!} =
\bigl[(m+n-1)\binom{m+n-2}{m-1}\bigr]^{-1}$.

\textbf{10.} Ambos lados de $B(x,n) =
\frac{\Gamma(x)\Gamma(n)}{\Gamma(x+n)}$ son iguales a $\frac1x $y$n =
1$($\Gamma(1) = 1$, $\Gamma(x+1) = x\Gamma(x)$). si estan de acuerdo
en $n$, luego por la relaciﺃ٣n de descenso y el funcional
ecuaciﺃ٣n:
\[
B(x, n+1) = \frac{n}{x+n}\,B(x,n), \qquad
\frac{\Gamma(x)\Gamma(n+1)}{\Gamma(x+n+1)}
= \frac{n}{x+n}\cdot
\frac{\Gamma(x)\Gamma(n)}{\Gamma(x+n)} :
\]
las dos secuencias obedecen a la misma recursividad de la misma semilla,
por lo tanto, acepta para todos los $n \in \N^*$ y todos los $x > 0$.

\textbf{11.} Con $t = \sin^2\theta$ ($\theta \in
\intoo0{\pi/2}$, $\dd t = 2\sin\theta\cos\theta\,\dd\theta$),
$t^{x-1} = \sin^{2x-2}\theta$ y $(1-t)^{y-1} =
\cos^{2y-2}\theta$:
\[
B(x,y) = \int_0^{\pi/2}\sin^{2x-2}\theta\,\cos^{2y-2}\theta
\cdot 2\sin\theta\cos\theta\,\dd\theta
= 2\int_0^{\pi/2}\sin^{2x-1}\theta\,\cos^{2y-1}\theta\,
\dd\theta .
\]

\textbf{12.} Tome $y = \frac12$ (eliminando el factor coseno) y
$2x - 1 = n$: $B\bigl(\frac{n+1}2, \frac12\bigr) = 2W_n$, es decir\
$W_n = \frac12 B\bigl(\frac{n+1}2,\frac12\bigr)$. el descenso
relaciﺃ٣n en la primera variable da
\[
\frac{W_n}{W_{n-2}}
= \frac{B\bigl(\frac{n-1}2 + 1, \frac12\bigr)}
{B\bigl(\frac{n-1}2, \frac12\bigr)}
= \frac{\frac{n-1}2}{\frac{n-1}2 + \frac12}
= \frac{n-1}{n} :
