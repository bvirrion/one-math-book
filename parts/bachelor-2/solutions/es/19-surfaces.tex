\begin{solution}{exo:b2:surfaces:1}
Con $\sigma(\theta, r) = (r\cos\theta,\ r\sin\theta,\ r)$:
\[
\sigma_\theta = (-r\sin\theta,\ r\cos\theta,\ 0),
\qquad
\sigma_r = (\cos\theta,\ \sin\theta,\ 1),
\]
independiente para $r > 0$. El \omterm{def:b2:surfaces:tangent}{plano tangente} en $M_0 =
\sigma(\theta_0, r_0)$pasa por $M_0$ con indicaciones
$\sigma_\theta, \sigma_r$. Ahora $M_0 - O = r_0(\cos\theta_0,
\sin\theta_0, 1) = r_0\,\sigma_r(\theta_0, r_0)$ es en sí mismo un
Dirección tangente: el vértice $O$ se encuentra en el \omterm{def:b2:surfaces:tangent}{plano tangente}. (Este
es el comportamiento general de los conos: están regidos por líneas que pasan por
el ápice, y un \omterm{def:b2:surfaces:tangent}{plano tangente} contiene la línea gobernante a través del
punto de tangencia.)
\end{solution}

\begin{solution}{exo:b2:surfaces:2}
Aplicar \cref{prop:b2:surfaces:gradient} a $f(x,y,z) =
\frac{x^2}{a^2} + \frac{y^2}{b^2} + \frac{z^2}{c^2}$: $\nabla
f(x_0,y_0,z_0) = 2\bigl(\frac{x_0}{a^2}, \frac{y_0}{b^2},
\frac{z_0}{c^2}\bigr) \neq 0$ en la superficie. El \omterm{def:b2:surfaces:tangent}{plano tangente} es
\[
\frac{x_0}{a^2}(x - x_0) + \frac{y_0}{b^2}(y - y_0)
+ \frac{z_0}{c^2}(z - z_0) = 0,
\qquad\text{i.e.}\qquad
\frac{x_0\,x}{a^2} + \frac{y_0\,y}{b^2} + \frac{z_0\,z}{c^2} = 1,
\]
usando ese $(x_0, y_0, z_0)$ satisface la ecuación del elipsoide ---
la regla de "dividir los cuadrados" que generaliza el $\langle
M_0, M \rangle = R^2$ de la esfera.
\end{solution}

\begin{solution}{exo:b2:surfaces:3}
$\sigma_u = (-v\sin u,\ v\cos u,\ a)$ y $\sigma_v = (\cos u,\
\sin u,\ 0)$, entonces
\[
E = v^2 + a^2, \qquad F = 0, \qquad G = 1,
\qquad
\sqrt{EG - F^2} = \sqrt{v^2 + a^2} > 0
\]
(el helicoidal es regular en todas partes, incluso en su eje $v =
0$). \omterm{def:b2:surfaces:area}{Área} de la pieza:
\[
\mathcal{A} = \int_0^{2\pi}\!\!\int_0^1
\sqrt{v^2 + a^2}\;\dd v\,\dd u
= 2\pi\cdot\frac12\Bigl[v\sqrt{v^2 + a^2}
+ a^2\ln\bigl(v + \sqrt{v^2 + a^2}\bigr)\Bigr]_0^1 ,
\]
es decir,\ $\mathcal{A} = \pi\Bigl(\sqrt{1 + a^2} +
a^2\ln\frac{1 + \sqrt{1 + a^2}}{a}\Bigr)$.
\end{solution}

\begin{solution}{exo:b2:surfaces:4}
Derivados:
\[
\sigma_\theta = \bigl(-(R + r\cos\psi)\sin\theta,\
(R + r\cos\psi)\cos\theta,\ 0\bigr),
\qquad
\sigma_\psi = \bigl(-r\sin\psi\cos\theta,\
-r\sin\psi\sin\theta,\ r\cos\psi\bigr).
\]
entonces
\[
E = (R + r\cos\psi)^2,
\qquad
F = r\sin\psi\cos\psi\,(R + r\cos\psi)
\bigl(\sin\theta\cos\theta - \sin\theta\cos\theta\bigr) = 0,
\qquad
G = r^2 ,
\]
entonces $\sqrt{EG - F^2} = r(R + r\cos\psi) \geq r(R - r) > 0$:
Regular en todas partes. \omterm{def:b2:surfaces:area}{Área}:
\[
\mathcal{A}
= \int_0^{2\pi}\!\!\int_0^{2\pi} r(R + r\cos\psi)
\,\dd\theta\,\dd\psi
= 2\pi r \Bigl(2\pi R + r\int_0^{2\pi}\cos\psi\,\dd\psi\Bigr)
= 4\pi^2 R r ,
\]
el término $\cos\psi$ integrándose a cero --- teorema de Pappus: \omterm{def:b2:surfaces:area}{área}
$=$ (\omterm{def:b2:curves:length}{longitud} del círculo girado) $\times$ (distancia recorrida por
su centro).
\end{solution}

\begin{solution}{exo:b2:surfaces:5}
Para $\sigma(x, y) = (x, y, f(x,y))$: $\sigma_x = (1, 0, f_x)$,
$\sigma_y = (0, 1, f_y)$, entonces $E = 1 + f_x^2$, $F = f_xf_y$, $G = 1
+ f_y^2$ y
\[
EG - F^2 = (1 + f_x^2)(1 + f_y^2) - f_x^2f_y^2
= 1 + f_x^2 + f_y^2 ,
\]
dando la fórmula \omterm{def:b2:surfaces:area}{área} indicada. Para $f = \frac12(x^2 + y^2)$ en
el disco unitario: $1 + f_x^2 + f_y^2 = 1 + x^2 + y^2$, y en polar
coordenadas ($x = \rho\cos\alpha$, $y = \rho\sin\alpha$, jacobiano
$\rho$, \cref{ch:b2:multint}):
\[
\mathcal{A}
= \int_0^{2\pi}\!\!\int_0^1 \sqrt{1 + \rho^2}\;\rho
\,\dd\rho\,\dd\alpha
= 2\pi\Bigl[\tfrac13(1 + \rho^2)^{3/2}\Bigr]_0^1
= \frac{2\pi}{3}\bigl(2\sqrt2 - 1\bigr).
\]
\end{solution}

\begin{solution}{exo:b2:surfaces:6}
Del cálculo de \cref{ex:b2:surfaces:spherearea}, $E =
R^2\cos^2\varphi$, $F = 0$, $G = R^2$, por lo que por
\cref{prop:b2:surfaces:curvelength}
\[
L = \int_a^b R\sqrt{\cos^2\varphi(t)\,\theta'(t)^2
+ \varphi'(t)^2}\;\dd t .
\]
Deje que los puntos finales sean $(\theta_0, \varphi_1)$ y $(\theta_0,
\varphi_2)$, $\varphi_1 < \varphi_2$. Para cualquier curva de unión,
\[
L \geq \int_a^b R\,\abs{\varphi'(t)}\,\dd t
\geq R\,\Bigl|\int_a^b \varphi'(t)\,\dd t\Bigr|
= R\,(\varphi_2 - \varphi_1),
\]
eliminando el término no negativo $\cos^2\varphi\,\theta'^2$ y usando
la desigualdad triangular para integrales. El arco meridiano
$\theta \equiv \theta_0$, $\varphi$ aumentando de $\varphi_1$ a
$\varphi_2$, tiene \omterm{def:b2:curves:length}{longitud} exactamente $R(\varphi_2 - \varphi_1)$: es
más corto. (Los meridianos son grandes círculos; este es el primero,
caso elemental del hecho de que las geodésicas de la esfera son grandes
círculos.)
\end{solution}

\begin{solution}{exo:b2:surfaces:7}
Fije una curva $\gamma$ dibujada en $S$ y deje $g(t) = \norm{\gamma(t)
- \Omega}^2$. Entonces $g'(t) = 2\langle \gamma'(t),\ \gamma(t) -
\Omega\rangle$. La línea normal en $M = \gamma(t)$ pasa por
$\Omega$ por hipótesis, entonces $\gamma(t) - \Omega$ es un \emph{normal}
vector, ortogonal al \omterm{def:b2:surfaces:tangent}{plano tangente}, en particular al
La velocidad $\gamma'(t)$ (\cref{prop:b2:surfaces:velocity}): $g' =
0$y $g$ es constante a lo largo de cada curva dibujada.

Ahora el conjunto $S_c = \{M \in S : \norm{M - \Omega}^2 = c\}$ es
cerrado en $S$; también es \omterm{def:b2:metric:topology}{abierto} en $S$: cerca de cualquiera de sus puntos, $S$
es un gráfico regular, por lo que cualquier punto cercano de $S$ está unido a él por
una curva dibujada (un segmento elevado), a lo largo de la cual $g$ es constante. como
$S$ es \omterm{def:b2:metric:connected}{conectado} y $S_c$ no está vacío para la derecha $c$, $S = S_c
\subseteq $la esfera de centro $\Omega $y radio $\sqrt c$
(\cref{ch:b2:metric}: argumento \omterm{def:b2:metric:connected}{conectividad}).
\end{solution}

\begin{solution}{exo:b2:surfaces:8}
Escribe $\Phi(u', v') = (u, v)$. Por la regla de la cadena,
\[
\tilde\sigma_{u'} = \frac{\partial u}{\partial u'}\sigma_u
+ \frac{\partial v}{\partial u'}\sigma_v,
\qquad
\tilde\sigma_{v'} = \frac{\partial u}{\partial v'}\sigma_u
+ \frac{\partial v}{\partial v'}\sigma_v ,
\]
parciales de $\sigma$ evaluados en $\Phi(u',v')$. Ampliando el
producto cruzado bilinealmente y usando $\sigma_u \wedge \sigma_u =
\sigma_v \wedge \sigma_v = 0$, $\sigma_v \wedge \sigma_u =
-\sigma_u \wedge \sigma_v$:
\[
\tilde\sigma_{u'} \wedge \tilde\sigma_{v'}
= \Bigl(\frac{\partial u}{\partial u'}
\frac{\partial v}{\partial v'}
- \frac{\partial v}{\partial u'}
\frac{\partial u}{\partial v'}\Bigr)\,
\sigma_u \wedge \sigma_v
= \det J_\Phi \cdot (\sigma_u \wedge \sigma_v)\circ\Phi .
\]
Tomar \omterm{def:b2:nvs:norm}{normas} da la identidad. Luego, por el cambio de variables
fórmula (\cref{ch:b2:multint}) aplicada al mapa $\Phi$ en $K' =
\Phi^{-1}(K)$:
\[
\iint_{K'} \norm{\tilde\sigma_{u'} \wedge \tilde\sigma_{v'}}
\,\dd u'\dd v'
= \iint_{K'} \norm{\sigma_u \wedge \sigma_v}\circ\Phi\;
\abs{\det J_\Phi}\,\dd u'\dd v'
= \iint_K \norm{\sigma_u \wedge \sigma_v}\,\dd u\,\dd v :
\]
las dos parametrizaciones asignan el mismo \omterm{def:b2:surfaces:area}{área} a la misma pieza de
superficie.
\end{solution}

\begin{solution}{exo:b2:surfaces:9}
En la carta esférica, $z = R\sin\varphi$, y la zona
corresponde a $\varphi_1 \leq \varphi \leq \varphi_2$ con

\[
\mathcal A = \int_{\varphi_1}^{\varphi_2}\!\!\int_0^{2\pi}
R^2\cos\varphi\,\dd\theta\,\dd\varphi
= 2\pi R^2(\sin\varphi_2 - \sin\varphi_1)
= 2\pi R\,(z_2 - z_1) .
\]
El resultado depende sólo de la altura $z_2 - z_1$: rodajas de
Igual espesor lleva igual \omterm{def:b2:surfaces:area}{áreas}, ya sea cortado en el ecuador
o en el polo --- Teorema de la caja del sombrero de Arquímedes, y la razón

\begin{solution}{exo:b2:surfaces:10}
$\sigma_\theta = (-r\sin\theta,\ r\cos\theta,\ 0)$ y

\sigma_\theta \wedge \sigma_z
= (r\cos\theta,\ r\sin\theta,\ -r\,r') ,

t \mapsto \bigl(r(1 + t)\cos\theta,\ r(1 + t)\sin\theta,\

\begin{solution}{exo:b2:surfaces:11}
$\sigma_u = (-\sin u, \cos u, 0)$, $\sigma_v = (0, 0, 1)$: $E
= 1$, $F = 0$, $G = 1$. Por
\cref{prop:b2:surfaces:curvelength} el \omterm{def:b2:curves:length}{longitud} de un sorteo
curva es $\int\sqrt{u'^2 + v'^2}\,\dd t$ --- el \omterm{def:b2:curves:length}{longitud} de su
sombra del parámetro $(u(t), v(t))$ en el plano: el gráfico es un
isometría local (el desenrollamiento del cilindro). la hélice
$t \mapsto (\cos t, \sin t, ct)$, $t \in \intcc0{2\pi}$, tiene
sombree el segmento de $(0,0)$ a $(2\pi, 2\pi c)$, de \omterm{def:b2:curves:length}{longitud}
$2\pi\sqrt{1 + c^2}$. Cualquier curva dibujada desde $(1,0,0)$ a $(1,
0, 2\pi c)$ que realice una vuelta completa tiene una sombra \omterm{def:b2:metric:continuity}{continuo}.
uniendo $(0, 0)$ a $(2\pi, 2\pi c)$, del plano \omterm{def:b2:curves:length}{longitud} $\geq$
el segmento recto; Dado que \omterm{def:b2:curves:length}{longitudes} está de acuerdo, la hélice es
más corto.
\end{solution}

\begin{solution}{exo:b2:surfaces:12}
La función $g(M) = \norm{M}^2$ es \omterm{def:b2:metric:continuity}{continuo} en el \omterm{def:b2:metric:compact}{compacto}
$S$, por lo que alcanza su máximo en algún momento $M_0$. Para cada curva
$\gamma$ dibujada en $S$ con $\gamma(0) = M_0$, la función $t
\mapsto \norm{\gamma(t)}^2$ tiene un máximo en $t = 0$, por lo que
derivado $2\langle\gamma'(0), M_0\rangle$ desaparece: $M_0$ es
ortogonal a cada vector tangente, es decir, normal a $S$ en
$M_0$. Como $\nabla f(M_0) \neq 0$ también dirige la línea normal.
(\cref{prop:b2:surfaces:gradient}), $\nabla f(M_0)$ y
$\vect{OM_0}$ son colineales. Para el elipsoide, radialidad
significa
\[
\Bigl(\frac{x_0}{a^2}, \frac{y_0}{b^2},
\frac{z_0}{c^2}\Bigr) = \mu\,(x_0, y_0, z_0) :
\]
cada coordenada satisface $x_0(\frac1{a^2} - \mu) = 0$, etc.;
ya que $a^{-2}, b^{-2}, c^{-2}$ son distintos, como máximo uno
La coordenada es distinta de cero y las soluciones en la superficie son
los puntos finales de seis ejes $(\pm a, 0, 0)$, $(0, \pm b, 0)$, $(0,
0, \pm c)$. Los puntos más lejanos son $(\pm a, 0, 0)$, en
distancia $a = \max(a,b,c)$.
\end{solution}

\begin{solution}{pb:b2:surfaces:1}
\textbf{1.} Expandiendo y usando la simetría de $A$
($t^{\mathsf T}\!APY = (At)^{\mathsf T}PY$):
\[
q(PY + t) = Y^{\mathsf T}P^{\mathsf T}\!APY +
2\,(At + b)^{\mathsf T}PY + \bigl(t^{\mathsf T}\!At +
2b^{\mathsf T}t + c\bigr),
\]
que es el triple mostrado. $A' = P^{\mathsf T}\!AP =
P^{-1}AP$es similar a $A$: mismo \omterm{def:b2:reduction:charpoly}{polinomio característico},
\omterm{def:b2:reduction:eigen}{espectro}, rango, firma. Finalmente $\{q = 0\} = \{sq = 0\}$
para $s \neq 0$, por lo que sólo la ecuación \emph{hasta un escalar} es
adjunto al conjunto; escalar por $s$ multiplica todo \omterm{def:b2:reduction:eigen}{valores propios}
por $s$.

\textbf{2.} El teorema espectral proporciona a $P \in O(3)$
$P^{\mathsf T}\!AP = \operatorname{diag}(\lambda_1, \lambda_2,
\lambda_3)$; La pregunta 1 con $t = 0$ convierte la ecuación en
$\sum\lambda_iy_i^2 + 2\sum\beta_iy_i + c = 0$, donde $\beta =
P^{\mathsf T}b$.

\textbf{3.} Para $\lambda_i \neq 0$: $\lambda_iy_i^2 +
2\beta_iy_i = \lambda_i\bigl(y_i +
\frac{\beta_i}{\lambda_i}\bigr)^2 -
\frac{\beta_i^2}{\lambda_i}$; la traducción $z_i = y_i +
\beta_i/\lambda_i $(y $z_i = y_i $para $i > r$) da
\[
\sum_{i \leq r}\lambda_iz_i^2 + 2\sum_{i > r}\beta_iz_i +
c'' = 0, \qquad
c'' = c - \sum_{i\leq r}\frac{\beta_i^2}{\lambda_i}.
\]

\textbf{4.}$q(2\Omega - X) = (2\Omega - X)^{\mathsf T}
A(2\Omega - X) + 2b^{\mathsf T}(2\Omega - X) + c$; expandiéndose
y restando $q(X)$, los términos cuadráticos se cancelan y
\[
q(2\Omega - X) - q(X) = -4\,\langle A\Omega + b,\ X\rangle +
4\,\langle A\Omega + b,\ \Omega\rangle .
\]
Si $A\Omega + b = 0$ esto desaparece de manera idéntica: el punto
La reflexión conserva $q$, de ahí el \omterm{pb:b2:surfaces:1}{cuádrico}. Por el contrario,
``$q(2\Omega - X) = q(X)$ para todos los $X$'' dice que el \omterm{def:b2:affine:subspace}{afín}
La función anterior desaparece en todo $\R^3$, lo que fuerza su
la parte lineal $A\Omega + b$ sea cero. Así centra $=$ soluciones de
$A\Omega = -b$: un conjunto no vacío si $b \in \operatorname{im}A$
(un subespacio \omterm{def:b2:affine:subspace}{afín} dirigido por $\ker A$) y un único punto
si $A$ es invertible.

\textbf{5.} Firma $(3,0)$ (todos $\lambda_i > 0$): $\delta >
0$ da $\frac{x^2}{a^2} + \frac{y^2}{b^2} + \frac{z^2}{c^2}
= 1$ con $a = \sqrt{\delta/\lambda_1}$, etc.\ --- an
elipsoide; $\delta = 0$: el punto único $O$; $\delta < 0$:
vacío. Firma $(2,1)$ ($\lambda_1, \lambda_2 > 0 >
\lambda_3$): $\delta > 0$: $\frac{x^2}{a^2} + \frac{y^2}{b^2}
- \frac{z^2}{c^2} = 1$, el hiperboloide de una hoja; $\delta =
0$: el cono $\frac{x^2}{a^2} + \frac{y^2}{b^2} =
\frac{z^2}{c^2}$; $\delta < 0$: $\frac{z^2}{c^2} -
\frac{x^2}{a^2} - \frac{y^2}{b^2} = 1$, el de dos hojas
hiperboloide ($\abs z \geq c$: dos componentes).

\textbf{6.} La parte cuadrática tiene matriz $A$ con diagonal
$1$ y fuera de diagonal $2$: $A = 2J - I$. Desde $J$ tiene \omterm{def:b2:reduction:eigen}{espectro}
$\{3, 0, 0\}$ (\omterm{def:b2:reduction:eigen}{vector propio} $(1,1,1)$ para $3$), $A$ tiene
\omterm{def:b2:reduction:eigen}{espectro} $\{5, -1, -1\}$, el \omterm{def:b2:reduction:eigen}{valor propio} $5$ llevado por
$\R(1,1,1)$. En las coordenadas rotadas: $5u^2 - v^2 - w^2 =
1$, es decir $\frac{u^2}{1/5} - v^2 - w^2 = 1$: una hoja de dos
hiperboloide, de revolución (igual \omterm{def:b2:reduction:eigen}{valores propios} $-1$) sobre el
eje $\R(1,1,1)$.

\textbf{7.} La superficie es $\bigl(\frac xa - \frac
zc\bigr)\bigl(\frac xa + \frac zc\bigr) = \bigl(1 - \frac
yb\bigr)\bigl(1 + \frac yb\bigr)$. Para $(\lambda : \mu) \neq
(0:0)$ defina la línea
\[
D_{\lambda:\mu} :\quad
\lambda\Bigl(\frac xa - \frac zc\Bigr) = \mu\Bigl(1 - \frac
yb\Bigr), \qquad
\mu\Bigl(\frac xa + \frac zc\Bigr) = \lambda\Bigl(1 + \frac
yb\Bigr)
\]
(dos ecuaciones \omterm{def:b2:affine:subspace}{afín} independientes: una recta). multiplicando el
dos ecuaciones muestran que cada punto de $D_{\lambda:\mu}$ se encuentra en
la superficie cuando $\lambda\mu \neq 0$; los casos $\lambda = 0$
o $\mu = 0$ se marcan directamente (por ejemplo,\ $\lambda = 0$: $y =
b$, $\frac xa = -\frac zc$, que satisface la ecuación). el
segunda familia $D'_{\lambda:\mu}$ intercambia los dos derechos
factores.

\textbf{8.} Fijar $M$ en la superficie. Las condiciones para $M \in
D_{\lambda:\mu}$ forman un sistema lineal homogéneo $2\times2$.
en $(\lambda, \mu)$ cuyo \omterm{def:b2:linalg:det}{determinante} es
\[
\Bigl(\frac{x^2}{a^2} - \frac{z^2}{c^2}\Bigr) - \Bigl(1 -
\frac{y^2}{b^2}\Bigr) = 0
\]
precisamente porque $M$ se encuentra en el \omterm{pb:b2:surfaces:1}{cuádrico}: un no trivial
La solución $(\lambda : \mu)$ existe. La matriz de coeficientes es
nunca cero (eso forzaría $1 - \frac yb = 1 + \frac yb =
0$), por lo que su rango es $1$ y la solución es única hasta
escala: exactamente una línea de la familia pasa por $M$.
Lo mismo se aplica a la segunda familia, y las dos líneas son
distintos (en $(a, 0, 0)$ son $\{x = a,\ \frac zc = \frac
yb\}$y $\{x = a,\ \frac zc = -\frac yb\}$): el de una hoja
El hiperboloide está doblemente reglado.

\textbf{9.} Sea $M = (x, y, z)$ en el hiperboloide, $\rho^2 =
\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1 + \frac{z^2}{c^2} \geq
1$. El punto $N = (x,\ y,\ \varepsilon c\rho)$ con
$\varepsilon$ el signo de $z$ satisface $\frac{x^2}{a^2} +
\frac{y^2}{b^2} - \frac{(c\rho)^2}{c^2} = 0$: $N \in C$, y
\[
d(M, C) \leq \abs{z - \varepsilon c\rho}
= c\bigl(\rho - \sqrt{\rho^2 - 1}\bigr)
= \frac{c}{\rho + \sqrt{\rho^2 - 1}} .
\]
Si $\norm M \to \infty$ entonces $\rho \to \infty$ (los tres
las coordenadas están limitadas por múltiplos de $\rho$), por lo que $d(M, C)
\to 0$. La sección $x = a$: $\frac{y^2}{b^2} -
\frac{z^2}{c^2} = 0$, el par de líneas cruzadas de pregunta
8 --- y también en $x = -a$.

\textbf{10.} Con $\lambda_3 = 0$, sale la pregunta 3
$\lambda_1z_1^2 + \lambda_2z_2^2 + 2\beta_3z_3 + c'' = 0$. en
la base propia, $\operatorname{im}A = \operatorname{Vect}(e_1,
e_2)$, por lo que según la pregunta existen 4 centros si $\beta_3 = 0$. si
$\beta_3 \neq 0$: la traducción $z_3 \mapsto z_3 -
c''/(2\beta_3)$elimina la constante, dejando $\lambda_1z_1^2
+ \lambda_2z_2^2 + 2\beta_3z_3 = 0$, es decir $z_3 = px^2 +
qy^2$ después de cambiar el nombre: un \emph{paraboloide elíptico} si
$\lambda_1\lambda_2 > 0$, un \emph{paraboloide hiperbólico} si
$\lambda_1\lambda_2 < 0$ --- y de hecho no hay centro. si
$\beta_3 = 0$: la ecuación $\lambda_1z_1^2 + \lambda_2z_2^2 +
c'' = 0$ no involucra a $z_3$: el \omterm{pb:b2:surfaces:1}{cuádrico} es un cilindro
sobre el plano correspondiente cónico --- cilindro elíptico,
línea o conjunto vacío cuando $\lambda_1\lambda_2 > 0$; hiperbólico
cilindro o par de planos que se cruzan cuando
$\lambda_1\lambda_2 < 0$ --- con toda una línea de centros
$\{(z_1^*, z_2^*)\} \times \R$.
