\begin{solution}{exo:b2:powerseries:3}

\begin{solution}{exo:b2:powerseries:8}

\begin{solution}{exo:b2:powerseries:9}
Con $A_n = \sum_{k \leq n} a_k \to A$: \omterm{thm:b2:series:abel}{resumen de abel} da, por
$0 \leq x < 1$,
\[
\sum_{n=0}^{\infty} a_n x^n = (1 - x)\sum_{n=0}^{\infty} A_n x^n
\]
(ambos lados convergen: $(A_n)$ estĂˇ acotado; y la identidad sigue
desde $a_n = A_n - A_{n-1}$ y reindexaciĂłn). Desde $(1 -
x)\sum x^n = 1$:
\[
\sum_n a_nx^n - A = (1-x)\sum_{n} (A_n - A)x^n .
\]
Dado $\varepsilon$, elija $N$ con $\abs{A_n - A} \leq
\varepsilon $para $n > N$; entonces
\[
\Bigl|\sum a_nx^n - A\Bigr|
\leq (1-x)\sum_{n \leq N}\abs{A_n - A} + \varepsilon(1 -
x)\sum_{n > N}x^n
\leq (1-x)\,C_N + \varepsilon ,
\]
y dejando $x \to 1^-$: limsup $\leq \varepsilon$ para todos
$\varepsilon$. Por tanto, el lĂ­mite radial es $A$.

Aplicaciones: $\sum \frac{(-1)^{n-1}}{n}$ converge (alternando),
y para $x < 1$ su serie de potencias suma $\ln(1 + x)$: por Abel,
la suma es $\ln 2$. Del mismo modo, $\sum\frac{(-1)^n}{2n+1}x^{2n+1} =
\arctan x $da $\frac\pi4$ en $x = 1$ --- la integral del aĂ±o 1
pruebas, ahora estructurales.
\end{solution}

\begin{solution}{pb:b2:powerseries:1}
\textbf{1.} Con $a_n = r_n - r_{n+1}$, sumatoria por partes:
\[
\sum_{n=N}^{M} a_nx^n
= r_Nx^N + \sum_{n=N+1}^{M} r_n\bigl(x^n - x^{n-1}\bigr)
- r_{M+1}x^M .
\]
Para $0 \leq x \leq 1$ los incrementos $x^{n-1} - x^n$ son
no negativo y telescopio a $x^N - x^M$; con $s = \sup_{n\geq
N}\abs{r_n}$:
\[
\Bigl|\sum_{n=N}^{M}a_nx^n\Bigr|
\leq s\bigl(x^N + (x^N - x^M) + x^M\bigr) = 2s\,x^N \leq 2s .
\]

\textbf{2.} Desde $r_n \to 0$, $\sup_{n\geq N}\abs{r_n} \to 0$:
La pregunta 1 es exactamente el criterio uniforme de Cauchy sobre
$\intcc{0}{1}$, por lo que $\sum a_nx^n$ converge \omterm{def:b2:funcseq:def}{uniformemente} allĂ­ y
su suma es \omterm{def:b2:metric:continuity}{continuo} (\cref{thm:b2:funcseq:seriestransfer}).
El valor en $1$ es $\sum a_n$, \omterm{def:b2:metric:continuity}{continuidad} en $1$ es el
lĂ­mite radial de \cref{exo:b2:powerseries:9}.

\textbf{3.} Para $\abs x < 1$ convergen las tres series de potencias
\omterm{def:b2:series:def}{absolutamente} y $\bigl(\sum a_nx^n\bigr)\bigl(\sum b_nx^n\bigr)
= \sum c_nx^n$ (\cref{prop:b2:powerseries:operations}). Por
pregunta 2 cada factor y el lado del producto son \omterm{def:b2:metric:continuity}{continuo} en
$\intcc{0}{1}$ (sus series de coeficientes convergen por
hipĂłtesis); dejando a $x \to 1^-$ la identidad: $AB = C$.

\textbf{4.} AquĂ­
\[
\abs{c_n} = \sum_{k=0}^{n}
\frac{1}{\sqrt{(k+1)(n-k+1)}}
\geq \sum_{k=0}^{n}\frac{2}{n+2}
= \frac{2(n+1)}{n+2} \geq 1,
\]
por AM--GM: $\sqrt{(k+1)(n-k+1)} \leq \frac{(k+1) +
(n-k+1)}{2} = \frac{n+2}{2}$. El tĂ©rmino general de $\sum c_n$
no tiende a $0$: el \omterm{thm:b2:series:fubini}{producto cauchy} diverge, aunque
ambos factores convergen (series alternas).

\textbf{5.} \cref{exo:b2:powerseries:5} da
$\frac{-\ln(1-x)}{1-x} = \sum H_nx^n$ ($\abs x < 1$).
Primitivas tĂ©rmino por tĂ©rmino
(\cref{thm:b2:powerseries:calculus}~(2)), ambos lados desapareciendo
en $0$:
\[
\frac{\bigl(\ln(1-x)\bigr)^2}{2}
= \sum_{n\geq1}\frac{H_n}{n+1}\,x^{n+1} .
\]
DisminuciĂłn: $(n+2)H_n \geq (n+1)H_{n+1}$ equivale a $H_n \geq
1$, verdadero para $n \geq 1$; y $\frac{H_n}{n+1} \sim \frac{\ln
n}{n} \to 0$: en $x = -1$ la serie converge por el
prueba alternante. Sustituyendo $x \mapsto -x$ y aplicando
pregunta 2 en $x = 1$:
\[
\frac{(\ln 2)^2}{2}
= \sum_{n\geq1}\frac{H_n}{n+1}(-1)^{n+1},
\]
el valor anunciado.

\textbf{6.}$f(x) = \sum(-1)^nx^n = \frac{1}{1+x} \to \frac12$
como $x \to 1^-$: Abel-sumable a $\frac12$. Pero el parcial
las sumas son $1, 0, 1, 0, \dots$: divergentes.

\textbf{7.} Dado $\varepsilon > 0$, elija $m$ con $\abs{u_n}
\leq \varepsilon $para $n > m$; para $N \geq m$:
\[
\Bigl|\frac{u_1 + \dots + u_N}{N}\Bigr|
\leq \frac{\abs{u_1} + \dots + \abs{u_m}}{N} +
\varepsilon\,\frac{N - m}{N}
\leq \frac{C_m}{N} + \varepsilon,
\]
entonces $\limsup \leq \varepsilon$ por cada $\varepsilon$: el
significa tender a $0$.

\textbf{8.} Para $0 \leq x \leq 1$: $1 - x^n = (1-x)(1 + x +
\dots + x^{n-1}) \leq n(1-x)$, entonces
\[
\Bigl|\sum_{n=0}^N a_n(1 - x_N^n)\Bigr|
\leq (1 - x_N)\sum_{n=1}^N n\abs{a_n}
= \frac1N\sum_{n=1}^{N}n\abs{a_n} .
\]
Para $n > N$: $\abs{a_n} = \frac{n\abs{a_n}}{n} \leq
\frac{1}{N}\sup_{m>N}m\abs{a_m}$y $\sum_{n>N}x_N^n \leq
\frac{1}{1 - x_N} = N$:
\[
\Bigl|\sum_{n>N}a_nx_N^n\Bigr|
\leq \frac{\sup_{m>N}m\abs{a_m}}{N}\cdot N
= \sup_{m>N}\,m\abs{a_m} .
\]

\textbf{9.} Descomponer
\[
A_N - L = \sum_{n=0}^{N}a_n\bigl(1 - x_N^n\bigr)
- \sum_{n>N}a_nx_N^n + \bigl(f(x_N) - L\bigr) .
\]
El primer tĂ©rmino tiende a $0$ por la pregunta 7 (la media de
$n\abs{a_n} \to 0$), el segundo por la pregunta 8 (el sup tiende a
$0$), el tercero porque $x_N \to 1^-$ y $f(x) \to L$. Por lo tanto
$A_N \to L$: Teorema de Tauber.

\textbf{10.} Para $x \in \intco{0}{1}$ y cualquier $N$:
$\sum_{n\leq N}a_nx^n \leq f(x) \leq M$ (tĂ©rminos no negativos).
Sea $x \to 1^-$ en la suma finita: $\sum_{n\leq N}a_n \leq M$.
Las sumas parciales son crecientes y acotadas: $\sum a_n$
converge, y luego la pregunta 2 da $\lim_{x\to1^-}f(x) =
\sum a_n$.

\textbf{11.} $\sigma_N - L$ es la media de los nĂşmeros $N$
$A_n - L$ ($0 \leq n < N$), que tienden a $0$: pregunta 7.

\textbf{12.} $A_n = 1$ para $n$ par, $0$ para impar: $A_0 + \dots
+ A_{N-1} = \lceil N/2\rceil$, por lo que $\sigma_N =

\textbf{13.} Bajo $\sigma_N \to L$ uno tiene $S_n = O(n)$, por lo tanto

a continuaciĂłn tienen radio $\geq 1$. Para $\abs x < 1$, de $a_n = A_n -
A_{n-1}$y $A_nx^n \to 0$:
\[

= \sum_n a_nx^n = f(x),
\]
e idĂ©nticamente $(1-x)\sum S_nx^n = \sum A_nx^n$, entonces $f(x) =
(1-x)^2\sum_n S_nx^n = (1-x)^2\sum_n(n+1)\sigma_{n+1}x^n$.
Finalmente $\sum(n+1)x^n = \frac{1}{(1-x)^2}$
(\cref{exo:b2:powerseries:3}), que es la segunda identidad.

\textbf{14.} Restando $L$ veces la segunda identidad del
primero:
\[

\]
Dado $\varepsilon$, elija $N$ con $\abs{\sigma_{n+1} - L}
\leq \varepsilon $para $n \geq N$; entonces
\[
\abs{f(x) - L} \leq (1-x)^2 C_N +

\]

$f(x) \to L$: Teorema de Frobenius.
