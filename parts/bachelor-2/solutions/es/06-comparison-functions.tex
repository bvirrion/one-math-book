\begin{solution}{exo:b2:comparison:1}
$\sqrt{x^2 + x + 1} = x\sqrt{1 + \tfrac1x + \tfrac{1}{x^2}} = x +
\frac12 + \frac38\cdot\frac1x + o\bigl(\frac1x\bigr)$ (binomial
expansiĂłn: $\frac12 u - \frac18 u^2$ con $u = \frac1x +
\frac{1}{x^2}$ da $\frac{1}{2x} + \frac{1}{2x^2} -
\frac{1}{8x^2} = \frac{1}{2x} + \frac{3}{8x^2}$, luego multiplica por

$\ln(x^2 + x) - 2\ln x = \ln\bigl(1 + \tfrac1x\bigr) = \frac1x -
\frac{1}{2x^2} + o\bigl(\frac{1}{x^2}\bigr)$.

Tercera funciĂłn: expandir cada factor,
\[
\frac{x + \sin x}{x - \ln x}
= \Bigl(1 + \frac{\sin x}{x}\Bigr)
\Bigl(1 + \frac{\ln x}{x} + \frac{(\ln x)^2}{x^2} +
O\Bigl(\frac{(\ln x)^3}{x^3}\Bigr)\Bigr).
\]
Ordene las aportaciones en la escala $+\infty$: $\frac{\ln x}{x}
\gg \frac{1}{x} \geq \bigl|\frac{\sin x}{x}\bigr| \gg \frac{(\ln
x)^2}{x^2}$. Por lo tanto, los dos tĂ©rminos despuĂ©s del $1$ inicial son
$\frac{\ln x}{x}$, luego el tĂ©rmino de oscilaciĂłn acotada $\frac{\sin
x}{x}$:
\[
\frac{x + \sin x}{x - \ln x}
= 1 + \frac{\ln x}{x} + \frac{\sin x}{x}
+ O\Bigl(\frac{(\ln x)^2}{x^2}\Bigr).
\]
\end{solution}

\begin{solution}{exo:b2:comparison:3}
Deje $v_n = H_n - \ln n - \gamma \to 0$. entonces

\begin{solution}{exo:b2:comparison:5}
$g(x) = x^n + x - 1$ aumenta estrictamente en $\intcc{0}{1}$ desde $-1$
a $1$: raĂ­z Ăşnica $x_n$. Desde $x_n^n = 1 - x_n \in
\intoo{0}{1}$: si $x_n \leq c < 1$a lo largo de una subsecuencia, entonces $x_n^n
\leq c^n \to 0$, entonces $1 - x_n \to 0$: contradicciĂłn con $x_n \leq
c$. Por lo tanto $x_n \to 1$.

Escriba $x_n = 1 - \varepsilon_n$, $\varepsilon_n \to 0^+$. el
la ecuaciĂłn dice $(1 - \varepsilon_n)^n = \varepsilon_n$, es decir
\[
n\ln(1 - \varepsilon_n) = \ln \varepsilon_n
\quad\Longrightarrow\quad
-n\varepsilon_n\bigl(1 + o(1)\bigr) = \ln\varepsilon_n .
\]
Entonces $n\varepsilon_n = -\ln\varepsilon_n\,(1 + o(1)) \to +\infty$, y
tomando logaritmos nuevamente: $\ln n + \ln\varepsilon_n =
\ln(-\ln\varepsilon_n) + o(1)$. Desde $\ln(-\ln \varepsilon_n) =
o(\ln(1/\varepsilon_n))$, esto da $\ln\varepsilon_n \sim -\ln
n$, de donde $\varepsilon_n = \frac{-\ln\varepsilon_n}{n}(1 + o(1))
\sim \frac{\ln n}{n}$:
\[
x_n = 1 - \frac{\ln n}{n} + o\Bigl(\frac{\ln n}{n}\Bigr) .
\]
\end{solution}

\begin{solution}{exo:b2:comparison:6}

\begin{solution}{exo:b2:comparison:7}
Separe los dos tĂ©rminos mĂˇs grandes:
\[

\qquad
\sum_{k\leq n-2} k! \leq (n-1)\,(n-2)! = (n-1)! .

Por lo tanto

\end{solution}

\begin{solution}{exo:b2:comparison:9}
Factorice $n$ y configure $t = \ln n$:
\[
S_n = \frac1n \sum_{k=1}^{n} \frac{1}{1 + t\,\frac kn} .
\]
Para $t$ fijo, la suma es una suma de Riemann de $u \mapsto \frac{1}{1 +
tu}$ sobre $\intcc{0}{1}$; la funciĂłn es monĂłtona en $u$, por lo que la
La suma de Riemann estĂˇ entre parĂ©ntesis por la integral desplazada una malla:
\[

\leq S_n \leq \int_0^1 \frac{\dd u}{1 + tu} + \frac1n

\begin{solution}{exo:b2:comparison:10}

\begin{solution}{exo:b2:comparison:11}
