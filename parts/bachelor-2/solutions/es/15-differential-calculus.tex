\begin{solution}{exo:b2:diffcalc:4}
Diferenciar $t \mapsto f(tx)$ en $t = 1$: por la regla de la cadena,
$\langle \nabla f(x), x\rangle$; por homogeneidad la misma función
es $t^pf(x)$, de la derivada $pf(x)$ en $t = 1$: identidad de Euler.

Converse: arregla $x \neq 0$ y deja que $\varphi(t) = f(tx) - t^p f(x)$
en $t > 0$. Entonces $\varphi'(t) = \langle\nabla f(tx), x\rangle -
pt^{p-1}f(x) = \frac1t\bigl(\langle \nabla f(tx), tx\rangle -
p\,t^pf(x)\bigr)$. La hipótesis --- la identidad de Euler en el punto
$tx$ --- evalúa el soporte como $p\,f(tx) - p\,t^pf(x) =
p\,\varphi(t)$. Entonces $\varphi' = \frac{p}{t}\varphi $con $\varphi(1)
= 0$: la solución única de la EDO lineal es $\varphi \equiv 0$ (Año
1 unicidad), es decir,\ $f(tx) = t^pf(x)$.
\end{solution}

\begin{solution}{exo:b2:diffcalc:6}
Desde $\nabla g(a) \neq 0$, un parcial, digamos $\frac{\partial
g}{\partial y}(a) \neq 0$: el teorema de la función implícita
(Compañero de \cref{thm:b2:diffcalc:inverse}) parametriza $\{g =
0\}$ cerca de $a = (a_1, a_2)$como $\gamma(t) = (t, y(t))$con $y$
$C^1$, $y'(t) = -\frac{\partial_x g}{\partial_y g}(\gamma(t))$
(diferenciar $g(t, y(t)) = 0$). La función de una variable $t
\mapsto f(\gamma(t))$tiene un extremo local en $t = a_1$:
\[
0 = \frac{\dd}{\dd t}f(\gamma(t))\Big|_{a_1}
= \partial_x f(a) + \partial_y f(a)\,y'(a_1)
= \partial_x f(a) - \partial_yf(a)\frac{\partial_x
g(a)}{\partial_y g(a)} :
\]
los vectores $\nabla f(a)$ y $\nabla g(a)$ tienen proporcional
coordenadas: $\nabla f(a) = \lambda \nabla g(a)$ con $\lambda =
\frac{\partial_y f(a)}{\partial_y g(a)}$.

Aplicación: en el círculo, $\nabla(xy) = (y, x)$ paralelo a
$(2x, 2y)$ fuerza a $y^2 = x^2$; con la restricción, los candidatos
son $\pm\bigl(\tfrac{1}{\sqrt2}, \tfrac{1}{\sqrt2}\bigr)$ (valor
$\frac12$) y $\pm\bigl(\tfrac{1}{\sqrt2},
-\tfrac{1}{\sqrt2}\bigr)$(valor$-\frac12$): máx.$\frac12$, mín.
$-\frac12$ (alcanzado: el círculo es \omterm{def:b2:metric:compact}{compacto}).
\end{solution}

\begin{solution}{exo:b2:diffcalc:10}
$f(x+h) - f(x) = 2\langle x, h\rangle + \norm h^2$: $\dd f_x =
2\langle x, \cdot\rangle$, $\nabla f(x) = 2x$, Arpillera $2I$
(constante). Para $g$:
\[
g(x + h) - g(x) = \langle Ax, h\rangle + \langle Ah, x\rangle
+ \langle Ah, h\rangle
= \bigl\langle (A + A^{\mathsf T})x,\ h\bigr\rangle +

\]
entonces $\nabla g(x) = (A + A^{\mathsf T})x$ y $H_g = A +

sólo importa la parte \omterm{def:b2:quadratic:adjoint}{simétrico} de $A$, como de hecho $g(x) =

\begin{solution}{exo:b2:diffcalc:11}
$f$ es convexo si su restricción a cada segmento es convexa,
es decir, si cada $\varphi(t) = f(a + tv)$ es convexo. por la cadena
regla $\varphi''(t) = \langle H_{a+tv}\,v,\ v\rangle$.

Si todos los hessianos son semidefinidos positivos: $\varphi'' \geq 0$,
por lo que cada $\varphi$ es convexo (volumen del año 1) y $f$ es convexo.
Por el contrario, si $f$ es convexo, cada $\varphi$ es convexo, por lo que
$\varphi''(0) \geq 0$: $\langle H_a v, v\rangle \geq 0$ para
cada $a$ y cada dirección $v$: todos los hessianos son positivos
semidefinido.
\end{solution}

 \item Por el valor medio de la desigualdad

\begin{solution}{pb:b2:diffcalc:1}
\textbf{1.} $\norm{ABx} \leq \vertiii A\norm{Bx} \leq
\vertiii A\vertiii B\norm x$: toma el control de $\norm x = 1$.
Los productos matriciales, $\det$ y $\operatorname{tr}$ son polinomiales.
funciones de las entradas, por lo tanto \omterm{def:b2:metric:continuity}{continuo} ($\mathcal M_n(\R)
\simeq \R^{n^2}$, todos los \omterm{def:b2:nvs:norm}{normas} equivalentes:
\cref{thm:b2:nvs:finitedim}).

\textbf{2.} $\sum\vertiii{X^k} \leq \sum\vertiii X^k <
\infty$: la serie converge \omterm{def:b2:series:def}{absolutamente}, por lo tanto converge
(\cref{thm:b2:nvs:absoluteconvergence}). De $(I -
X)\sum_{k\leq N}X^k = I - X^{N+1} \to I$: la suma es $(I -
X)^{-1}$. \omterm{def:b2:nvs:norm}{Norma}: $\leq \sum\vertiii X^k = \frac{1}{1 -
\vertiii X}$; y
\[
(I - X)^{-1} - I - X = \sum_{k\geq2}X^k = X^2(I - X)^{-1},
\qquad
\vertiii{X^2(I-X)^{-1}} \leq
\frac{\vertiii X^2}{1 - \vertiii X} = O(\vertiii X^2).
\]

\textbf{3.} $A + H = A(I + A^{-1}H)$ con $\vertiii{A^{-1}H}
\leq \vertiii{A^{-1}}\vertiii H < 1$: invertible por pregunta 2:
la bola \omterm{def:b2:metric:topology}{abierto} de radio $\vertiii{A^{-1}}^{-1}$ alrededor de $A$ se encuentra
en $GL_n(\R)$. Densidad: $\det(A + \varepsilon I)$ es un grado-$n$
polinomio en $\varepsilon$ con coeficiente principal $1$:
tiene un número finito de raíces, por lo que hay $\varepsilon_k \to 0$
con $A + \varepsilon_kI$ invertible, convergiendo a $A$.

\textbf{4.} Para $\vertiii H < \frac{1}{2\vertiii{A^{-1}}}$:
\[
(A + H)^{-1} - A^{-1}
= \bigl[(I + A^{-1}H)^{-1} - I\bigr]A^{-1},
\]
de \omterm{def:b2:nvs:norm}{norma} como máximo $\frac{\vertiii{A^{-1}H}}{1 -
\vertiii{A^{-1}H}}\,\vertiii{A^{-1}} \leq
2\vertiii{A^{-1}}^2\vertiii H \to 0$: $\Phi$ es \omterm{def:b2:metric:continuity}{continuo} en
cada $A \in GL_n(\R)$.

\textbf{5.} $(A+H)^2 = A^2 + AH + HA + H^2$: el mapa $H
\mapsto AH + HA$es lineal y el error $H^2$ es
$O(\vertiii H^2)$. Expandiendo $(A + H)^k$ y ordenando por
número de factores $H$: la parte lineal es
$\sum_{i=0}^{k-1}A^iHA^{k-1-i}$ y los términos con $\geq 2$
los factores $H$ están limitados por $\binom k2$; muchos productos de \omterm{def:b2:nvs:norm}{norma}
$\leq \vertiii A^{k-2}\vertiii H^2$-escala: $O(\vertiii H^2)$.
No se puede contraer la suma a $kA^{k-1}H$ porque $H$ y $A$
no necesita conmutar --- la suma es el no conmutativo correcto
derivado.

\textbf{6.} Para pequeño $H$:
\[
(A+H)^{-1} = (I + A^{-1}H)^{-1}A^{-1}
= \bigl(I - A^{-1}H + O(\vertiii H^2)\bigr)A^{-1}
= A^{-1} - A^{-1}HA^{-1} + O(\vertiii H^2) :
\]
$\dd\Phi_A(H) = -A^{-1}HA^{-1}$, lineal en $H$. Para $n = 1$:
$\dd(1/a)(h) = -h/a^2$, el derivado familiar.

\textbf{7.} Regla de cadena a lo largo de la curva:
$\bigl(A(t)^{-1}\bigr)' = \dd\Phi_{A(t)}(A'(t)) =
-A(t)^{-1}A'(t)A(t)^{-1}$. En $A(t) = I + tB$, $t = 0$: $(I +
tB)^{-1} = I - tB + O(t^2)$.

\textbf{8.} Por pregunta 5 e invariancia cíclica de la traza:
\[
\dd f_A(H) = \operatorname{tr}\Bigl(\sum_{i=0}^{k-1}
A^iHA^{k-1-i}\Bigr) = k\operatorname{tr}\bigl(A^{k-1}H\bigr) .
\]
Contra el producto Frobenius, $\dd f_A(H) =
\operatorname{tr}\bigl((\nabla f)^{\mathsf T}H\bigr)$ requiere
$(\nabla f)^{\mathsf T} = kA^{k-1}$: $\nabla f(A) =
k\,(A^{k-1})^{\mathsf T}$.

\textbf{9.} $f(A + H) - f(A) = 2\operatorname{tr}
(A^{\mathsf T}H) + \operatorname{tr}(H^{\mathsf T}H)$: el
\omterm{def:b2:diffcalc:differential}{diferencial} es $H \mapsto 2\operatorname{tr}(A^{\mathsf T}H) =
2\langle A, H\rangle$, por lo que $\nabla f(A) = 2A$; el segundo orden
término es exactamente $\norm H_F^2$: el hessiano es el doble del
identidad \omterm{def:b2:quadratic:def}{forma cuadrática}, positiva definida y constante, por lo que
$\norm\cdot_F^2$ es estrictamente convexo (la fórmula de Taylor es
exacto aquí).

\textbf{10.} Por multilinealidad en las columnas, $\det(I + H) =
\sum_{S\subseteq\{1,\dots,n\}}\det(M_S)$donde $M_S$ tiene columna
$h_j$ para $j \in S$ y $e_j$ en caso contrario. $S = \varnothing$
da $1$; $S = \{j\}$ da el \omterm{def:b2:linalg:det}{determinante} de $I$ con columna
$j$ reemplazado por $h_j$, es decir, su $j$-ésima entrada $h_{jj}$,
sumando a $\operatorname{tr}H$; cada término con $\abs S \geq 2$
es un \omterm{def:b2:linalg:det}{determinante} con al menos dos columnas de tamaño
$O(\vertiii H)$, por lo tanto $O(\vertiii H^2)$ (mapas multilineales en un
espacio de dimensión finita están acotados), y hay finitamente
muchos. Entonces $\det(I + H) = 1 + \operatorname{tr}H +
O(\vertiii H^2)$: $\dd(\det)_I = \operatorname{tr}$.

\textbf{11.} $\det(A + H) = \det A\,\det(I + A^{-1}H) =
\det A\,\bigl(1 + \operatorname{tr}(A^{-1}H) +
O(\vertiii H^2)\bigr)$: el \omterm{def:b2:diffcalc:differential}{diferencial} es $H \mapsto
\det(A)\operatorname{tr}(A^{-1}H)$.

\textbf{12.} Expansión de Laplace a lo largo de la fila $i$: $\det A =
\sum_j a_{ij}C_{ij}$, y los cofactores $C_{ij}$ no involucran
fila $i$: $\frac{\partial\det}{\partial a_{ij}} = C_{ij}$. Por lo tanto
\[
\dd(\det)_A(H) = \sum_{i,j}C_{ij}h_{ij}
= \operatorname{tr}\bigl(\operatorname{com}(A)^{\mathsf T}
H\bigr)
= \operatorname{tr}\bigl(\operatorname{adj}(A)H\bigr),
\qquad
\nabla(\det)(A) = \operatorname{com}(A) .
\]
Para reversibles $A$, $\operatorname{adj}A = \det(A)A^{-1}$
recupera la pregunta 11. A lo largo de una curva $C^1$, la regla de la cadena dice
$(\det A(t))' = \operatorname{tr}(\operatorname{adj}
(A(t))\,A'(t))$: Fórmula de Jacobi.

\textbf{13.} Con $A' = MA$ y
$\operatorname{adj}(A)A = \det(A)I$:
\[
(\det A)' = \operatorname{tr}\bigl(\operatorname{adj}(A)MA
\bigr)
= \operatorname{tr}\bigl(A\operatorname{adj}(A)M\bigr)
= \det A\;\operatorname{tr}M
\]
(ciclicidad; $A\operatorname{adj}A = \det(A) I$ también). el
La ODE escalar lineal $y' = \operatorname{tr}(M(t))\,y$ tiene la
solución única $y(t) = y(0)\exp\bigl(\int_0^t
\operatorname{tr}M\bigr)$ (Año 1): Fórmula de Liouville.

\textbf{14.} En $A \in SL_n(\R)$, tome $H = \frac1nA$:
$\dd(\det)_A\bigl(\tfrac1nA\bigr) = \frac1n\det(A)
\operatorname{tr}(A^{-1}A) = \frac1n\cdot1\cdot n = 1 \neq 0$:
el \omterm{def:b2:diffcalc:differential}{diferencial} es una forma lineal distinta de cero, por lo tanto sobreyectiva
en $\R$ en cada punto del nivel establecido: $SL_n(\R)$ es un
conjunto de nivel liso (de dimensión $n^2 - 1$).

\textbf{15.} $\sum_k\vertiii{A^k/k!} \leq
\sum\vertiii A^k/k! = \eu^{\vertiii A}$: convergencia absoluta
(argumento \omterm{def:b2:metric:complete}{lo completo} de la pregunta 2), con \omterm{def:b2:funcseq:series}{convergencia normal}
en cada bola $\vertiii A \leq R$ (unida $R^k/k!$ independiente
de $A$). Cada suma parcial es \omterm{def:b2:metric:continuity}{continuo} (polinomio); el
El límite uniforme de bolas es \omterm{def:b2:metric:continuity}{continuo}: $A \mapsto \eu^A$ es
\omterm{def:b2:metric:continuity}{continuo}, con $\vertiii{\eu^A} \leq \eu^{\vertiii A}$.

\textbf{16.} Ambas series convergen \omterm{def:b2:series:def}{absolutamente}, por lo que \omterm{thm:b2:series:fubini}{cauchy
producto} es legítima (\cref{thm:b2:series:fubini}):
\[
\eu^A\eu^B = \sum_{n\geq0}\frac{1}{n!}\sum_{k=0}^n\binom
nkA^kB^{n-k}
= \sum_{n\geq0}\frac{(A+B)^n}{n!} = \eu^{A+B},
\]
