\end{solution}

\begin{solution}{exo:b2:diffeq:2}
$y' = \sin(ty)$: $\abs{\sin(ty) - \sin(tz)} \leq \abs t\,\abs{y -
z}$--- \omterm{def:b2:metric:continuity}{Lipschitz} en $y$ \omterm{def:b2:funcseq:def}{uniformemente} en cada segmento de tiempos:
soluciones globales únicas en $\R$ (aplicar el teorema en cada
segmento).

$y' = y^2$: localmente \omterm{def:b2:metric:continuity}{Lipschitz} solamente; ningún teorema global, y de hecho
$y(0) = 1$ explota en $t = 1$.

$y' = \abs y$: $\abs\cdot$ es $1$-Lipschitz: existencia global y
singularidad. Con $y(0) = 0$: $y \equiv 0$ (¡singularidad!). Con $y(0)
= 1$: $y$permanece positivo (no puede cruzar la solución cero), por lo que $y' =
y$: $y = \eu^t$.
\end{solution}

\begin{solution}{exo:b2:diffeq:5}
\emph{Serie:} $A = \alpha I + \beta J$ con $J =
\begin{pmatrix}0 & -1\\ 1 & 0\end{pmatrix}$, $J^2 = -I$; los dos
los pedidos conmutan, por lo que $\eu^{tA} = \eu^{\alpha t}\,\eu^{\beta tJ}$,
y la serie de $\eu^{\beta t J}$ se divide en potencias pares/impares
en $\cos(\beta t)I + \sin(\beta t)J$: la escala de rotación indicada
matriz.

\emph{Complejo:} identifica $(x, y) \in \R^2$ con $z = x + \iu y$;
el sistema $X' = AX$ lee $z' = (\alpha + \iu\beta)z$, cuyo
La solución $z(t) = \eu^{\alpha t}\eu^{\iu\beta t}z_0$ es exactamente la
espiral: módulo $\eu^{\alpha t}$, argumento que avanza a gran velocidad
$\beta$.
\end{solution}

\begin{solution}{exo:b2:diffeq:6}
Deje $v(t) = \eu^{-k(t - t_0)}\int_{t_0}^t u$. entonces
\[
v'(t) = \eu^{-k(t-t_0)}\Bigl(u(t) - k\int_{t_0}^t u\Bigr)
\leq C\,\eu^{-k(t-t_0)},
\]
por la hipótesis. Integrando de $t_0$ a $t$ ($v(t_0) = 0$):
$v(t) \leq \frac{C}{k}\bigl(1 - \eu^{-k(t - t_0)}\bigr)$, es decir\
$k\int_{t_0}^t u \leq C\bigl(\eu^{k(t-t_0)} - 1\bigr)$; alimentación
esto de nuevo en la hipótesis: $u(t) \leq C\eu^{k(t-t_0)}$.

Unicidad/dependencia: dos soluciones $y, z$ de la integral
ecuación satisface
\[
\norm{y(t) - z(t)} \leq \norm{y_0 - z_0} +
k\int_{t_0}^{t}\norm{y - z},
\]
y Gronwall con $C = \norm{y_0 - z_0}$ da el exponencial
atado; $C = 0$ da singularidad.
\end{solution}

\begin{solution}{exo:b2:diffeq:8}
Dunford: $A = D + N$ desplazamientos, $D$ \omterm{def:b2:reduction:diag}{diagonalizable} con el mismo
\omterm{def:b2:reduction:eigen}{valores propios}, $N$ nilpotente, entonces
\[
\eu^{tA} = \eu^{tD}\,\eu^{tN},
\qquad
\eu^{tN} = \sum_{k < n} \frac{t^kN^k}{k!}
\ \text{(a matrix polynomial in } t).
\]
Vamos $-2\alpha = \max_i \Re\lambda_i < 0$. En una base diagonalizando
$D$, $\vertiii{\eu^{tD}} \leq \eu^{-2\alpha t}$ (entradas
$\eu^{t\lambda_i}$ del módulo $\eu^{t\Re\lambda_i}$); \omterm{def:b2:nvs:norm}{normas} en
diferentes bases se diferencian por constantes. Por lo tanto
\[
\norm{X(t)} \leq \vertiii{\eu^{tA}}\,\norm{X_0}
\leq C'\,\eu^{-2\alpha t}\,(1 + t)^{n-1}\,\norm{X_0}
\leq C\,\eu^{-\alpha t}\norm{X_0} ,
\]
absorbiendo el polinomio en un factor exponencial
($\eu^{-\alpha t}(1+t)^{n-1} \to 0$, por lo tanto acotado).
\end{solution}

\begin{solution}{exo:b2:diffeq:9}
Sea $y$ una solución máxima en $\intco{0}{T}$, $T \leq \infty$,
y supongamos $T < \infty$. La forma integral da, para $t < T$,
\[
\norm{y(t)} \leq \norm{y_0} + \int_0^t \bigl(a\norm{y(s)} +
b\bigr)\dd s
\leq \bigl(\norm{y_0} + bT\bigr) + a\int_0^t\norm{y(s)}\,\dd s ,
\]
y Gronwall limita $\norm{y(t)} \leq (\norm{y_0} +
bT)\,\eu^{aT} =: M$con $\intco{0}{T}$: la solución se queda en un
\omterm{def:b2:metric:compact}{compacto} bola. Entonces $y' = f(t, y)$ está delimitado cerca de $T$, por lo que $y$ es
\omterm{def:b2:metric:continuity}{Lipschitz} cerca de $T$ y extiende \omterm{def:b2:metric:continuity}{continuamente} a $T$ (Cauchy
criterio); resolver el problema de Cauchy en $(T, y(T))$ prolonga $y$
más allá de $T$, contradiciendo la maximalidad. Por lo tanto $T = \infty$: no
escape de tiempo finito bajo crecimiento lineal.
\end{solution}

\begin{solution}{pb:b2:diffeq:1}

\textbf{2.} $X(t) = a\,\eu^{\mu t}v_- + b\,\eu^{\lambda t}v_+$.

\textbf{18.} Entre ceros consecutivos $a < b$, $y_1$ mantiene
un signo, diga $y_1 > 0$ en $\intoo ab$: luego $y_1'(a) > 0$ y
$y_1'(b) < 0$ (ceros simples). Evaluando la constante $W =
y_1y_2' - y_1'y_2$ en $a$y $b$:
\[
W = -y_1'(a)\,y_2(a) = -y_1'(b)\,y_2(b) ,
\]
entonces $y_2(a)$ y $y_2(b)$ tienen signos opuestos ($W \neq 0$
prohíbe desaparecer): $y_2$ desaparece en $\intoo ab$
(valores intermedios). Allí no puede desaparecer dos veces: dos ceros.
de $y_2$ flanquearía un cero de $y_1$ por el mismo argumento con
los roles intercambiados, contradiciendo la consecutivaidad: exactamente
uno cero --- entrelazado.

\textbf{19.} Supongamos que $z$ no tiene cero en $\intoo ab$; reemplazando
$y, z$ por sus negativos, asume $y > 0$ y $z > 0$ en
$\intoo ab$. Configure $\varphi = yz' - y'z$: $\varphi' = yz'' -
y''z = (q_1 - q_2)\,yz \leq 0$ en $\intoo ab$: $\varphi$ es
no creciente. Pero $\varphi(a) = -y'(a)z(a) \leq 0$ (como
$y'(a) > 0$, $z(a) \geq 0$) y $\varphi(b) = -y'(b)z(b) \geq
0$(como $y'(b) < 0$, $z(b) \geq 0$): una función no creciente
que se ejecuta desde $\leq 0$ a $\geq 0$ desaparece de manera idéntica, por lo que
$(q_1 - q_2)yz \equiv 0$ en $\intoo ab$. Si $q_1 < q_2$
en algún lugar de $\intoo ab$, esto es absurdo ($y, z > 0$ allí):
$z$ debe desaparecer estrictamente en el interior. En general ($q_1 \leq q_2$),
$z$ desaparece en $\intoo ab$ o $\varphi \equiv 0$
fuerza a $z$ proporcional a $y$, que desaparece en $a$ y $b$:
en todos los casos $z$ tiene un cero en $\intcc ab$.

\textbf{20.} Límite superior: compare $y$ (coeficiente $q \geq
m^2$) con $u(t) = \sin(m(t - a))$(coeficiente $m^2 \leq q$,
entonces $y$ desempeña el papel de $z$ en la pregunta 19): si $y$ no tuviera
cero en $\intoc{a}{a + \pi/m}$, los ceros $a$ y $a +
\frac\pi m $de $u$serían consecutivos con $y \neq 0$
entre ellos, contradiciendo la pregunta 19: ceros consecutivos de
$y$ están a la distancia $\leq \frac\pi m$. Límite inferior: si dos
ceros consecutivos $a < b$ de $y$ tenían $b - a < \frac\pi M$,
entonces $z(t) = \sin(M(t-a))$ (coeficiente $M^2 \geq q$) sería
tiene que desaparecer en $\intcc ab \subset \intoo{a}{a + \pi/M}
\cup\{a\}$, donde su único cero es el propio $a$ --- pero
La pregunta 19 se aplicó en $\intoo{a}{b}$ con rigor en el
los puntos finales dan un cero en $\intcc ab$ y $z > 0$ en
$\intoc ab$: contradicción. Por lo tanto $\frac\pi M \leq b - a \leq
\frac\pi m$; para $q \equiv \omega^2$ ambos límites colapsan a
el espaciado exacto $\frac\pi\omega$ del armónico
oscilador.

\textbf{21.} Con $u = ty$: $u'' = ty'' + 2y'$, entonces $ty'' + 2y'
+ ty = u'' + u = 0$: $u = A\sin t + B\cos t = R\sin(t +
\varphi)$, y $y = \frac{u}{t}$ recupera $\frac{\sin t}t$ y
$\frac{\cos t}t$ (\cref{exo:b2:diffeq:7}) sin descenso de
orden. Los ceros de cualquier solución distinta de cero son los de
$R\sin(t + \varphi)$: espaciado exactamente $\pi$ --- Sturm's
Filosofía en acción: el coeficiente $q \equiv 1$ dicta
los ceros, con fórmula o sin fórmula.

\textbf{22.} Establecer $x(t) =
\frac1\omega\int_0^t\sin(\omega(t-s))F(s)\dd s$. Entonces $x(0) =
0$;
\[
x'(t) = \frac1\omega\sin(0)F(t) +
\int_0^t\cos(\omega(t-s))F(s)\dd s
= \int_0^t\cos(\omega(t-s))F(s)\dd s ,
\]
entonces $x'(0) = 0$; y $x''(t) = F(t) -
\omega\int_0^t\sin(\omega(t-s))F(s)\dd s = F(t) - \omega^2x(t)$
(diferenciación de un \omterm{thm:b2:integration:continuity}{integral de parámetros} con límite variable,
como en el capítulo de integración). Con $F(s) = F\cos\omega s$,
producto a suma da
\[
\int_0^t\sin(\omega(t-s))\cos(\omega s)\dd s
= \frac12\int_0^t\bigl(\sin\omega t + \sin(\omega t -
2\omega s)\bigr)\dd s
= \frac{t}{2}\sin\omega t ,
\]
(la segunda pieza se integra a cero), entonces $x =
\frac{F}{2\omega}t\sin\omega t$: pregunta 15 nuevamente, de
Duhamel.

\textbf{23.} En formato de sistema $X' = AX + (0, F(t))^{\mathsf T}$
con $\operatorname{Sp}A$ de partes reales negativas ($\zeta >
0$): variación de constantes y \cref{exo:b2:diffeq:8}
($\vertiii{\eu^{tA}} \leq C\eu^{-\alpha t}$) dar
\[
\norm{X(t)} \leq C\eu^{-\alpha t}\norm{X_0}
+ \int_0^t C\eu^{-\alpha(t-s)}\norm{F}_\infty\dd s
\leq C\norm{X_0} + \frac{C\norm F_\infty}{\alpha} :
\]
entrada limitada, salida limitada --- uniformemente en el inicial
datos después del transitorio.

El forzado falla exactamente en una frecuencia --- \omterm{pb:b2:diffeq:1}{resonancia} ---
