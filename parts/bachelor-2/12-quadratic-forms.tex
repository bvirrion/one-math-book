\chapter{Quadratic Forms}\label{ch:b2:quadratic}

A quadratic form is the algebraic shadow of a geometry: signature
zero parts flatten, positive parts curve one way, negative the
other. This chapter reduces every real quadratic form to a sum of
$\pm$ squares (Gauss), proves that the counts of signs are intrinsic
(Sylvester), and then crowns Euclidean geometry with the
\emph{spectral theorem}: symmetric endomorphisms diagonalize in
orthonormal bases --- the single most-used theorem of applied
linear algebra.

\section{Bilinear and quadratic forms}

\begin{definition}\label{def:b2:quadratic:def}
A \emph{symmetric bilinear form} on a real vector space $E$ is a
bilinear $\varphi \colon E \times E \to \R$ with $\varphi(x, y) =
\varphi(y, x)$; the associated \emph{quadratic
form}\index{quadratic form} is $q(x) = \varphi(x, x)$. The form
$\varphi$ is recovered from $q$ by
\emph{polarization}\index{polarization identity}:
\[
\varphi(x, y) = \tfrac12\bigl(q(x + y) - q(x) - q(y)\bigr).
\]
In a basis $(e_i)$, the \emph{matrix} of $\varphi$ is the symmetric
$B = (\varphi(e_i, e_j))$, with $q(x) = X^{\mathsf T} B X$; a change
of basis with matrix $P$ replaces $B$ by $P^{\mathsf T} B P$
(\emph{congruence}\index{congruent matrices} --- not similarity!).
The \emph{rank} of $q$ is $\operatorname{rk} B$ (invariant:
congruence multiplies by invertible matrices).
\end{definition}

\begin{example}
On $\R^2$: $q(x, y) = x^2 + 4xy + y^2$ has matrix $\begin{pmatrix}
1 & 2\\ 2 & 1\end{pmatrix}$. An inner product is exactly a
symmetric bilinear form whose quadratic form is positive definite;
this chapter studies the general, sign-indefinite case.
\end{example}

\begin{example}[Congruence in action]\label{ex:b2:quadratic:congruence}
Take $q(x, y) = x^2 + 4xy + y^2$ (matrix $B = \begin{pmatrix} 1
& 2\\ 2 & 1\end{pmatrix}$) and the new basis $e_1' = (1, 1)$,
$e_2' = (1, -1)$, i.e.\ $P = \begin{pmatrix} 1 & 1\\ 1 &
-1\end{pmatrix}$. Then
\[
P^{\mathsf T}BP
= \begin{pmatrix} 1 & 1\\ 1 & -1\end{pmatrix}
\begin{pmatrix} 1 & 2\\ 2 & 1\end{pmatrix}
\begin{pmatrix} 1 & 1\\ 1 & -1\end{pmatrix}
= \begin{pmatrix} 6 & 0\\ 0 & -2\end{pmatrix} :
\]
in the coordinates $(u, v)$ along the new basis, $q = 6u^2 -
2v^2$ --- check: $x = u + v$, $y = u - v$ gives $x^2 + 4xy +
y^2 = 6u^2 - 2v^2$ directly. Note the new diagonal entries $6,
-2$ are \emph{not} the eigenvalues $3, -1$ of $B$: congruence
rescales, only similarity preserves spectra --- but the signs
agree, as Sylvester's theorem demands. (The basis here is
orthogonal but not orthonormal; normalizing it by
$\frac{1}{\sqrt2}$ would divide the diagonal by $2$ and
recover the eigenvalues.)
\end{example}

\begin{example}[Gram determinants measure area]\label{ex:b2:quadratic:gramarea}
For $v_1, v_2$ in a Euclidean space, the Gram matrix $G =
\bigl(\langle v_i, v_j\rangle\bigr)$ packages lengths and
angle; its determinant packages \emph{area}:
\[
\det G = \norm{v_1}^2\norm{v_2}^2 - \langle v_1, v_2\rangle^2
= \norm{v_1}^2\norm{v_2}^2\bigl(1 - \cos^2\theta\bigr)
= \bigl(\norm{v_1}\,\norm{v_2}\sin\theta\bigr)^2 ,
\]
the squared area of the parallelogram on $v_1, v_2$ ---
and Cauchy--Schwarz is exactly the statement $\det G \geq 0$.
Worked instance: $v_1 = (1, 2, 2)$, $v_2 = (2, 1, -2)$ in
$\R^3$:
\[
G = \begin{pmatrix} 9 & 0\\ 0 & 9 \end{pmatrix},
\qquad
\det G = 81 :
\]
the vectors are orthogonal of length $3$, spanning a
parallelogram (here, a square) of area $\sqrt{81} = 9$.
Closing insight: no cross product and no dimension-$3$ magic
was used --- $\sqrt{\det G}$ measures $k$-dimensional volume
in \emph{any} dimension, the starting point of the weekend
problem's Part I and of the surface-area integrals later in
this volume.
\end{example}

\section{Gauss reduction and Sylvester's inertia}

\begin{theorem}[Gauss reduction]\label{thm:b2:quadratic:gauss}
Every quadratic form $q$ on a finite-dimensional real space can be
written\index{Gauss reduction}
\[
q = \sum_{i=1}^{s} \ell_i^2 - \sum_{j=1}^{t} m_j^2 ,
\]
where $\ell_1, \dots, \ell_s, m_1, \dots, m_t$ are linearly
independent linear forms; equivalently, some basis makes the
matrix of $q$ diagonal with entries $+1$ ($s$ times), $-1$ ($t$
times), $0$.
\end{theorem}

\begin{proof}
Induction on the number of variables, in coordinates: $q(x_1,
\dots, x_n)$.

\emph{Case 1: some square appears}, say the coefficient $a$ of
$x_1^2$ is nonzero. Group all the $x_1$-terms and complete the
square:
\[
q = a\Bigl(x_1 + \frac{1}{a}\,\lambda(x_2, \dots,
x_n)\Bigr)^{\!2} + q_1(x_2, \dots, x_n),
\]
where $\lambda$ is linear and $q_1$ quadratic in the remaining
variables: one independent form split off (it involves $x_1$, the
others do not), induction applies to $q_1$, and $\pm$ signs come
from the sign of $a$ after rescaling by $\sqrt{\abs a}$.

\emph{Case 2: no square, but some cross term}, say $b\,x_1x_2$ with
$b \neq 0$. Use the identity
\[
x_1x_2 = \tfrac14\bigl((x_1 + x_2)^2 - (x_1 - x_2)^2\bigr)
\]
after grouping: writing $q = b\,x_1x_2 + x_1\alpha + x_2\beta +
q_2$ (with $\alpha, \beta, q_2$ in the other variables), one checks
\[
q = \frac{b}{4}\Bigl[\Bigl(x_1 + x_2 + \frac{\alpha +
\beta}{b}\Bigr)^{2} - \Bigl(x_1 - x_2 + \frac{\beta -
\alpha}{b}\Bigr)^{2}\Bigr] + \widetilde q ,
\]
with $\widetilde q$ free of $x_1, x_2$: two independent forms split
off, and induction finishes.

Independence of the collected forms: order the batches as
produced. The forms of the first batch contain $x_1$ (Case 1)
or $x_1, x_2$ (Case 2); all later forms are free of those
variables. Suppose a linear combination of all collected forms
vanishes. Reading the coefficient of $x_1$ (and $x_2$): only
the first batch contributes, and within that batch the one or
two forms are visibly independent ($\ell$ alone; or $\ell \pm
m$ with independent $\ell, m$): the first-batch coefficients
vanish. Strip the batch and repeat: by induction along the
batches, all coefficients vanish --- the whole family is free,
triangularity made explicit.
\end{proof}

\begin{theorem}[Sylvester's law of inertia]\label{thm:b2:quadratic:sylvester}
The pair $(s, t)$ in \cref{thm:b2:quadratic:gauss} depends only on
$q$, not on the reduction: it is the
\emph{signature}\index{signature of a quadratic form} of $q$.
Moreover
\[
s = \max\{\dim F : q|_F \text{ positive definite}\},
\]
and symmetrically for $t$.
\end{theorem}

\begin{proof}
Let $q = \sum_{i \leq s}\ell_i^2 - \sum_{j\leq t} m_j^2$ and let
$F_+$ be the span of the (pre-)dual vectors on which $(\ell_i)$
restrict to coordinates --- concretely: complete the
independent family $(\ell_1, \dots, \ell_s, m_1, \dots, m_t)$
into a basis of the dual $E^*$, and let $(u_1, \dots, u_n)$ be
the basis of $E$ whose coordinate forms these are (the
pre-dual basis: $\ell_i(u_k) = \delta_{ik}$ for $k \leq s$,
and the later forms vanish on the earlier vectors). Set $F_+ =
\operatorname{Vect}(u_1, \dots, u_s)$: for $x = \sum_{i\leq
s}x_iu_i \in F_+$,
\[
\ell_i(x) = x_i,
\qquad m_j(x) = 0,
\qquad\text{so}\qquad
q(x) = \sum_{i\leq s}x_i^2 > 0 \quad (x \neq 0) :
\]
$q|_{F_+}$ is positive definite and the max in the display
is $\geq s$. Conversely, let $F$ be any subspace with $q|_F$
positive definite, and $G = \{x : \ell_1(x) = \dots = \ell_s(x) =
0\}$, of codimension $\leq s$; on $G$, $q(x) = -\sum m_j^2 \leq 0$.
Then $F \cap G = \{0\}$ (a nonzero vector there would have $q > 0$
and $q \leq 0$), so $\dim F \leq \dim E - \dim G \leq s$. Hence the
max equals $s$ for \emph{every} reduction: $s$ is intrinsic, and $t
= \operatorname{rk} q - s$ likewise.
\end{proof}

\begin{example}\label{ex:b2:quadratic:gaussexample}
$q(x, y, z) = xy + yz + zx$ (no squares). With $x_1 = x$, $x_2 =
y$: $q = xy + z(x + y)$, and the two-square identity gives
\[
q = \tfrac14(x + y + 2z)^2 - \tfrac14(x - y)^2 - z^2 ,
\]
(expand to check). Three independent forms: signature $(1, 2)$,
rank $3$. One positive direction, two negative: the ``light cone''
geometry of this form.
\end{example}

\begin{example}[A degenerate form, reduced in full]\label{ex:b2:quadratic:degenerate}
$q(x, y, z) = xy + yz$ on $\R^3$: no squares, so Case 2 with the
grouping $q = y(x + z)$. The two-square identity on the product
of the independent forms $y$ and $x + z$:
\[
q = \frac14\bigl(y + x + z\bigr)^2 -
\frac14\bigl(y - x - z\bigr)^2 .
\]
The two linear forms $y + x + z$ and $y - x - z$ are
independent (their difference is $2(x+z)$, their sum $2y$), so
Sylvester reads off: signature $(1, 1)$, rank $2$ ---
\emph{degenerate}. The kernel of the polar form is found by
solving $\varphi(v, \cdot) = 0$: with matrix
$\frac12\begin{pmatrix} 0&1&0\\ 1&0&1\\ 0&1&0\end{pmatrix}$,
the kernel is $\{y = 0,\ x + z = 0\} = \R\,(1, 0, -1)$, the
direction along which $q$ sees nothing. Closing insight: rank
deficiency shows up in Gauss as ``running out of variables''
--- the reduction produced only two squares out of three
dimensions, and the missing dimension is exactly the kernel.
\end{example}

\begin{example}[One form, two roads to the signature]\label{ex:b2:quadratic:tworoads}
$q(x, y, z) = 2x^2 + 2y^2 + 2z^2 + 2xy + 2yz$, matrix
$\begin{pmatrix} 2 & 1 & 0\\ 1 & 2 & 1\\ 0 & 1 & 2
\end{pmatrix}$. \emph{Road 1, Gauss:} complete squares in
order,
\[
q = 2\Bigl(x + \frac y2\Bigr)^{\!2} + \frac32 y^2 + 2yz + 2z^2
= 2\Bigl(x + \frac y2\Bigr)^{\!2}
+ \frac32\Bigl(y + \frac{2z}{3}\Bigr)^{\!2} + \frac43 z^2 :
\]
three positive squares on independent forms, signature $(3,
0)$: positive definite. \emph{Road 2, eigenvalues:} the matrix
is the tridiagonal $2I + N$ with $N$ the neighbor matrix; its
eigenvalues are $2 + \sqrt2$, $2$, $2 - \sqrt2$ (check the
eigenvectors $(1, \pm\sqrt2, 1)$ and $(1, 0, -1)$), all
positive: same verdict, by
\cref{cor:b2:quadratic:principalaxes}. Gauss is faster;
eigenvalues say more (they give the principal axes and the
extreme values of $q$ on the sphere). Closing insight: the
positive pivots $2, \frac32, \frac43$ of Gauss are exactly the
ratios $\frac{\Delta_k}{\Delta_{k-1}}$ of leading principal
minors ($\Delta_1 = 2$, $\Delta_2 = 3$, $\Delta_3 = 4$) --- the
weekend problem proves this in general.
\end{example}

\begin{method}[Computing a signature: three routes]\label{met:b2:quadratic:signature}
\begin{enumerate}
  \item \emph{Gauss} (always works, fastest by hand): complete
        squares in order, Case 2 when no square is available;
        count the signs. Check that the collected linear forms
        are independent --- fewer forms than variables means a
        kernel (\cref{ex:b2:quadratic:degenerate}).
  \item \emph{Leading minors} (for definiteness tests): all
        $\Delta_k > 0$ iff positive definite
        (\cref{exo:b2:quadratic:8}); the pivots
        $\Delta_k/\Delta_{k-1}$ even give the Gauss
        coefficients (weekend problem). Fails silently if some
        $\Delta_k = 0$: fall back on route 1.
  \item \emph{Eigenvalues} (most informative, most expensive):
        signs of the spectrum
        (\cref{cor:b2:quadratic:principalaxes}); also yields
        the principal axes and the extreme values of $q$ on
        the unit sphere. Prefer it when the eigenstructure is
        needed anyway.
\end{enumerate}
\end{method}

\section{The spectral theorem}

Now let $E$ be Euclidean (inner product $\langle\cdot,\cdot\rangle$,
Year 1 volume).

\begin{definition}[Adjoint; symmetric endomorphisms]\label{def:b2:quadratic:adjoint}
For $u \in \mathcal{L}(E)$, the \emph{adjoint}\index{adjoint}
$u^*$ is the unique endomorphism with
\[
\langle u(x), y\rangle = \langle x, u^*(y)\rangle
\qquad (x, y \in E);
\]
in an orthonormal basis, $\operatorname{Mat}(u^*) =
\operatorname{Mat}(u)^{\mathsf T}$. $u$ is
\emph{symmetric}\index{symmetric endomorphism} (self-adjoint) when
$u^* = u$ --- equivalently its matrix in an orthonormal basis is
symmetric.
\end{definition}

\begin{proof}[Existence and uniqueness of the adjoint]
For fixed $y$, the form $x \mapsto \langle u(x), y\rangle$ is
linear, hence (finite dimension) of the shape $\langle x,
z_y\rangle$ for a unique $z_y$ --- the map $y \mapsto z_y =:
u^*(y)$ being linear by uniqueness. The matrix identification:
$\langle u(e_i), e_j\rangle$ read both ways.
\end{proof}

\begin{example}[The adjoint depends on the inner product]\label{ex:b2:quadratic:weightedadjoint}
On $\R^2$ take the \emph{weighted} inner product $\langle x,
y\rangle_D = x_1y_1 + 2x_2y_2$ (matrix $D =
\operatorname{diag}(1,2)$) and $u$ of matrix $A =
\begin{pmatrix} 0 & 1\\ 0 & 0\end{pmatrix}$ in the canonical
basis. From $\langle u(x), y\rangle_D = (Ax)^{\mathsf T}Dy =
x^{\mathsf T}(A^{\mathsf T}D)y$ and $\langle x, u^*(y)\rangle_D
= x^{\mathsf T}(DA^*)y$, the adjoint's matrix is
\[
A^* = D^{-1}A^{\mathsf T}D
= \begin{pmatrix} 1 & 0\\ 0 & \tfrac12\end{pmatrix}
\begin{pmatrix} 0 & 0\\ 1 & 0\end{pmatrix}
\begin{pmatrix} 1 & 0\\ 0 & 2\end{pmatrix}
= \begin{pmatrix} 0 & 0\\ \tfrac12 & 0\end{pmatrix}
\neq A^{\mathsf T} .
\]
Sanity check on $x = (1,0)$, $y = (0,1)$:
\[
\langle u(x), y\rangle_D = \langle (0,0), y\rangle_D = 0,
\quad
\langle x, u^*(y)\rangle_D
= \langle(1,0), (0,\tfrac12)\rangle_D = 0 ;
\]
on $x = (0,1)$, $y = (1,0)$:
\[
\langle u(x), y\rangle_D = \langle(1,0),(1,0)\rangle_D = 1,
\quad
\langle x, u^*(y)\rangle_D =
\langle(0,1),(0,\tfrac12)\rangle_D = 1 .
\] Closing insight: ``$\operatorname{Mat}(u^*)
= \operatorname{Mat}(u)^{\mathsf T}$'' is a statement about
\emph{orthonormal} bases only; in general the metric $D$
intervenes, exactly as in the simultaneous reduction of the
weekend problem.
\end{example}

\begin{theorem}[Spectral theorem]\label{thm:b2:quadratic:spectral}
Let $u$ be a symmetric endomorphism of a Euclidean space $E$. Then
$E$ has an \emph{orthonormal basis of eigenvectors} of $u$; all
eigenvalues are real, and eigenspaces for distinct eigenvalues are
orthogonal. Matrix form: every real symmetric matrix $A$ writes
\[
A = P\,D\,P^{\mathsf T},
\qquad P \text{ orthogonal } (P^{\mathsf T} P = I),\ D
\text{ diagonal}.\index{spectral theorem}
\]
\end{theorem}

\begin{proof}
\emph{An eigenvector exists.} The function $x \mapsto \langle u(x),
x\rangle$ is continuous on the unit sphere $S$ of $E$, which is
compact (finite dimension, \cref{thm:b2:nvs:finitedim}): it attains
its maximum $\lambda$ at some $a \in S$. We claim $u(a) = \lambda
a$. For any $y \perp a$ with $\norm y = 1$ and $t \in \R$, the
vector $x_t = \frac{a + ty}{\sqrt{1 + t^2}}$ lies on $S$
($\norm{a + ty}^2 = 1 + t^2$ by Pythagoras); expanding
the maximized function,
\[
g(t) = \langle u(x_t), x_t\rangle
= \frac{\langle u(a), a\rangle + 2t\langle u(a), y\rangle +
t^2\langle u(y), y\rangle}{1 + t^2}
\]
(symmetry of $u$ merged the two cross terms: $\langle u(a),
y\rangle = \langle a, u(y)\rangle = \langle u(y), a\rangle$).
$g$ is a differentiable function of $t$ with a maximum at $t =
0$; the quotient rule at $0$ gives
\[
g'(0) = \frac{2\langle u(a), y\rangle\cdot 1 - \langle u(a),
a\rangle\cdot 0}{1} = 2\langle u(a), y\rangle = 0 .
\]
So $u(a)$ is orthogonal to the whole hyperplane $a^\perp$: $u(a)
\in (a^{\perp})^{\perp} = \R a$, i.e.\ $u(a) = \mu a$; and $\mu =
\langle u(a), a\rangle = \lambda$.

\emph{Induction.} The orthogonal complement $F = a^\perp$ is
$u$-stable: for $x \perp a$, $\langle u(x), a\rangle = \langle x,
u(a)\rangle = \lambda\langle x, a\rangle = 0$. The restriction
$u|_F$ is symmetric for the induced inner product; by induction on
the dimension, $F$ has an orthonormal eigenbasis; prepend $a$.

\emph{Complements.} Eigenvalues are the real numbers $\langle u(e),
e\rangle$ on the eigenbasis. Orthogonality of eigenspaces: $u(x) =
\lambda x$, $u(y) = \mu y$ give $\lambda\langle x, y\rangle =
\langle u(x), y\rangle = \langle x, u(y)\rangle = \mu\langle x,
y\rangle$, so $\langle x, y\rangle = 0$ when $\lambda \neq \mu$.
Matrix form: columns of $P$ = the orthonormal eigenbasis.
\end{proof}

\begin{example}[A full spectral run]\label{ex:b2:quadratic:fullrun}
Diagonalize orthogonally $A = \begin{pmatrix} 1 & 2\\ 2 &
1\end{pmatrix}$. Characteristic polynomial $(1 - \lambda)^2 -
4$: eigenvalues $3$ and $-1$. Eigenvectors: $(A - 3I)v = 0$
gives $v_1 = \frac{1}{\sqrt2}(1,1)$; $(A + I)v = 0$ gives $v_2
= \frac{1}{\sqrt2}(1,-1)$ --- orthogonal, as
\cref{thm:b2:quadratic:spectral} guarantees without
computation. With $P = (v_1\ v_2)$ (a rotation by
$\frac\pi4$):
\[
P^{\mathsf T}AP = \begin{pmatrix} 3 & 0\\ 0 & -1
\end{pmatrix},
\qquad
x^2 + 4xy + y^2 = 3u^2 - v^2
\quad\text{in the rotated frame} .
\]
So the form of \cref{exo:b2:quadratic:1} is a hyperbola-type
form: signature $(1,1)$, consistent with its Gauss reduction
$(x + 2y)^2 - 3y^2$ --- different squares, same signature, as
Sylvester demands. Closing insight: Gauss gave the answer
faster, but the spectral route also reports that on the unit
circle $q$ ranges exactly over $\intcc{-1}{3}$, attained along
$v_2$ and $v_1$: the extra work buys geometry.
\end{example}

\begin{corollary}[Principal axes; positivity tests]\label{cor:b2:quadratic:principalaxes}
\begin{enumerate}
  \item Every quadratic form $q$ on a Euclidean space diagonalizes
        in some \emph{orthonormal} basis: $q(x) = \sum_i \lambda_i
        x_i^2$ with $\lambda_i$ the eigenvalues of the symmetric
        matrix of $q$; the signature counts positive and negative
        eigenvalues.
  \item A symmetric matrix is positive semidefinite (resp.\
        definite) iff all its eigenvalues are $\geq 0$ (resp.\ $>
        0$); and then the extreme values of the Rayleigh quotient
        are
        \[
        \min_{\norm x = 1} \langle Ax, x\rangle = \lambda_{\min},
        \qquad
        \max_{\norm x = 1} \langle Ax, x\rangle = \lambda_{\max} .
        \]
\end{enumerate}
\end{corollary}

\begin{proof}
(1) Write $q(x) = \langle A x, x\rangle$ with $A$ symmetric (the
matrix of $q$ in an orthonormal basis); diagonalize $A$ by the
spectral theorem: for $x = \sum x_ie_i$ in the orthonormal
eigenbasis,
\[
q(x) = \Bigl\langle \sum_i \lambda_ix_ie_i,\ \sum_j
x_je_j\Bigr\rangle = \sum_i \lambda_i x_i^2
\]
(orthonormality kills the cross terms). The $\pm$ signs of the
$\lambda_i$ count the signature by Sylvester: rescaling each
coordinate by $\sqrt{\abs{\lambda_i}}$ exhibits a Gauss
reduction with independent forms.

(2) In the eigenbasis, $\langle Ax, x\rangle = \sum \lambda_i
x_i^2$, trapped between $\lambda_{\min}\norm x^2$ and
$\lambda_{\max}\norm x^2$, with equality at the corresponding
eigenvectors; positivity of all eigenvalues is thus equivalent
to positivity of the form.
\end{proof}

\begin{example}\label{ex:b2:quadratic:spectralexample}
$A = \begin{pmatrix} 2 & 1\\ 1 & 2 \end{pmatrix}$: eigenvalues $3$
(eigenvector $\frac{1}{\sqrt2}(1,1)$) and $1$
($\frac{1}{\sqrt2}(1,-1)$). The quadratic form $2x^2 + 2xy + 2y^2$
becomes $3X^2 + Y^2$ in the rotated orthonormal frame: an ellipse's
principal axes, computed. The Gauss reduction reaches a diagonal
form too, but only the spectral theorem reaches it \emph{without
distorting lengths}.
\end{example}

\begin{example}[An ellipse fully identified]\label{ex:b2:quadratic:ellipserun}
Which curve is $5x^2 + 4xy + 2y^2 = 6$? The matrix
$\begin{pmatrix} 5 & 2\\ 2 & 2\end{pmatrix}$ has characteristic
polynomial $\lambda^2 - 7\lambda + 6 = (\lambda - 1)(\lambda -
6)$: eigenvalues $1$ and $6$, both positive --- an ellipse.
Orthonormal eigenvectors: for $\lambda = 1$, solve
$\begin{pmatrix} 4 & 2\\ 2 & 1\end{pmatrix}v = 0$: $v_1 =
\frac{1}{\sqrt5}(1, -2)$; for $\lambda = 6$: $v_2 =
\frac{1}{\sqrt5}(2, 1)$. In the rotated coordinates $(X, Y)$
along $(v_2, v_1)$ the equation becomes
\[
6X^2 + Y^2 = 6,
\qquad\text{i.e.}\qquad
X^2 + \frac{Y^2}{6} = 1 :
\]
semi-axes $1$ (along $v_2$) and $\sqrt6$ (along $v_1$).
Closing insight: the rough shape was free of charge --- $\det
= 6 > 0$ and positive trace announce an ellipse before any
eigenvector is computed --- but only the spectral theorem
delivers the axes' directions and lengths, i.e.\ the actual
geometry.
\end{example}

\begin{example}[Extremes on the sphere, read off the spectrum]\label{ex:b2:quadratic:sphereextremes}
What are the extreme values of $q(x,y,z) = 2xy + 2yz + 2zx$ on
the unit sphere? Its matrix (the all-ones off-diagonal of
\cref{exo:b2:quadratic:2}) has eigenvalues $2$ and $-1$
(double), so by \cref{cor:b2:quadratic:principalaxes}~(2):
\[
\max_{\norm v = 1} q(v) = 2
\ \text{ at } v = \tfrac{1}{\sqrt3}(1,1,1),
\qquad
\min_{\norm v = 1} q(v) = -1
\ \text{ on the circle } x + y + z = 0 .
\]
No calculus, no Lagrange multipliers: the spectral theorem
solves this constrained optimization outright --- and exhibits
the maximizer. Closing insight: compare with the multiplier
method of the differential-calculus chapter, which finds the
same critical points with more work; for \emph{quadratic}
objectives on spheres, spectra are the royal road (the weekend
problem of the Hermitian chapter builds the whole
Courant--Fischer theory on this observation).
\end{example}

\begin{remark}[Common pitfalls]
\emph{(i) Congruence is not similarity:} a change of basis for
a form acts by $P^{\mathsf T}BP$, not $P^{-1}BP$; eigenvalues
are \emph{not} invariants of a quadratic form ($I$ and $4I$
are congruent via $P = 2I$) --- only their signs are
(Sylvester). Speak of the eigenvalues of a form only once an
inner product is fixed. \emph{(ii) Positive entries prove
nothing:} $\begin{pmatrix} 1 & 2\\ 2 & 1\end{pmatrix}$ has all
entries positive yet signature $(1,1)$ ($\det = -3$);
conversely a positive definite matrix may have negative
off-diagonal entries (\cref{ex:b2:quadratic:tworoads} shifted:
$2I - N$ works equally). Use
\cref{met:b2:quadratic:signature}. \emph{(iii) Dependent
squares:} writing $q = \ell_1^2 - \ell_2^2$ says nothing if
$\ell_1, \ell_2$ are proportional --- $x^2 + 2xy + y^2 =
(x+y)^2$ has rank $1$, not $2$; always check independence
before reading off the signature. \emph{(iv) Sphere extremes
without compactness:} the Rayleigh bounds of
\cref{cor:b2:quadratic:principalaxes} are attained because the
sphere is compact; on the open ball or the whole space, an
indefinite form has neither maximum nor minimum.
\end{remark}

\begin{remark}[Where this is used]
The spectral theorem is the single most exported result of this
book: statistics diagonalizes covariance matrices with it
(principal component analysis), mechanics extracts normal modes
of oscillation from it (the simultaneous reduction of the
weekend problem), numerical analysis builds Cholesky and
singular value decompositions on it (same problem), and the next
chapter transports it to complex Hermitian spaces. The Year 3
volume proves its infinite-dimensional avatar for compact
self-adjoint operators, where the finite-dimensional proof's
compactness argument becomes the whole story.
\end{remark}

\begin{remark}[Perspectives within this volume]
Quadratic forms thread through the rest of Book 4 under three
disguises. As \emph{Hessians}: the differential-calculus
chapter classifies critical points by the signature of the
second-order form, so Sylvester's invariance is what makes
``saddle'' a well-defined word. As \emph{energies}: the
differential-equations chapter's oscillators carry the
quadratic energy $\frac12x'^2 + \frac12\omega^2x^2$, and the
simultaneous reduction of this chapter's weekend problem is
exactly the extraction of normal modes. As \emph{geometry}:
the conics of this chapter grow into the quadric surfaces of
the geometry chapters, where the second fundamental form of a
surface --- a quadratic form on each tangent plane --- has
signature deciding whether the surface curves like a bowl or
a saddle. The Hermitian chapter, next, replays the whole score
over $\C$.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b2:quadratic:1}
Gauss-reduce and give rank and signature:
\[
q_1(x,y) = x^2 + 4xy + y^2,
\qquad
q_2(x,y,z) = x^2 + 2y^2 + 3z^2 + 2xy + 2yz .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:quadratic:2}
Diagonalize orthogonally $A = \begin{pmatrix} 0 & 1 & 1\\ 1 & 0 &
1\\ 1 & 1 & 0\end{pmatrix}$ (eigenvalues from
\cref{ch:b2:reduction}'s computation; now make the basis
orthonormal) and reduce the form $q(x,y,z) = 2xy + 2yz + 2zx$ to
principal axes.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:quadratic:3}
Prove that $u^{**} = u$, $(u \circ v)^* = v^* \circ u^*$, and that
$\ker u^* = (\operatorname{im} u)^{\perp}$. Deduce
$\operatorname{rk} u^* = \operatorname{rk} u$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:quadratic:4}
Let $A$ be real symmetric with $A^3 = A$. Prove that $A^2$ is the
matrix of an orthogonal projection. More generally, relate the
spectral decompositions of $A$ and $P(A)$ for a polynomial $P$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:quadratic:5}
Prove that $O(n) = \{P : P^{\mathsf T}P = I\}$ is a compact subset
of $\mathcal{M}_n(\R)$ \emph{(closed: preimage of $\{I\}$ under a
continuous map; bounded: columns are unit vectors)}. Is it
connected?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:quadratic:6}
(Square root) Let $A$ be symmetric positive semidefinite. Construct
a symmetric positive semidefinite $B$ with $B^2 = A$, and prove
that it is \emph{unique} \emph{(existence: take square roots of the
eigenvalues in a spectral basis; uniqueness: a candidate $B$
commutes with $A = B^2$, so preserves its eigenspaces --- reduce to
the scalar case on each)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:quadratic:7}
For real symmetric $A$, prove that $\vertiii{A}_2 :=
\sup_{\norm x_2 = 1}\norm{Ax}_2 = \max_i \abs{\lambda_i}$
(spectral radius), and compute $\vertiii{A}_2$ for $A =
\begin{pmatrix} 1 & 2\\ 2 & 1\end{pmatrix}$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:quadratic:8}
(Sylvester's criterion) Let $A$ be real symmetric with leading
principal minors $\Delta_1, \dots, \Delta_n$ (determinants of the
top-left blocks). Prove that $A$ is positive definite if and only
if all $\Delta_k > 0$. \emph{(For $\Rightarrow$: restrictions of a
definite form are definite, and the determinant of a positive
definite matrix --- the product of its eigenvalues --- is positive.
For $\Leftarrow$: induct on $n$; the top-left $(n-1)$-block is
positive definite, diagonalize the form on that subspace and
complete the square in the last variable; the sign of the last
diagonal entry is governed by $\det A = \Delta_n > 0$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:quadratic:9}
(Courant--Fischer, second eigenvalue) Let $u$ be symmetric with
eigenvalues $\lambda_1 \geq \lambda_2 \geq \dots \geq \lambda_n$.
Prove
\[
\lambda_2 = \min_{\substack{H \text{ hyperplane}}}\;
\max_{\substack{x \in H,\ \norm x = 1}} \langle u(x), x\rangle .
\]
\emph{(For $\leq$: any hyperplane meets the $2$-plane spanned by
the top two eigenvectors. For $\geq$: choose $H =
(e_1)^{\perp}$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:quadratic:10}
Determine rank and signature of $q(x_1, \dots, x_n) = \sum_{i <
j} x_ix_j$ on $\R^n$ ($n \geq 2$), in two ways: by the algebraic
identity $2q = \bigl(\sum x_i\bigr)^2 - \sum x_i^2$ together
with the restriction of $q$ to the hyperplane $\sum x_i = 0$;
and by computing the eigenvalues of its matrix $\frac12(J - I)$,
where $J$ is the all-ones matrix.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:quadratic:11}
Let $A$, $B$ be real symmetric with $B$ positive semidefinite.
Prove
\[
\lambda_{\min}(A)\operatorname{tr} B
\;\leq\;
\operatorname{tr}(AB)
\;\leq\;
\lambda_{\max}(A)\operatorname{tr} B .
\]
\emph{(Write $B = C^{\mathsf T}C$ and $\operatorname{tr}(AB) =
\sum_i \langle A c_i, c_i\rangle$ over the columns $c_i$ of
$C^{\mathsf T}$.)} In particular $\operatorname{tr}(AB) \geq 0$
when both are positive semidefinite.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:quadratic:12}
On $E = \mathcal{M}_n(\R)$, consider $q(M) =
\operatorname{tr}(M^2)$.
\begin{enumerate}
  \item Show that $q$ is a quadratic form with polar form
        $\varphi(M, N) = \operatorname{tr}(MN)$.
  \item Show that the symmetric and antisymmetric matrices form
        $\varphi$-orthogonal subspaces on which $q$ is
        respectively positive definite and negative definite
        \emph{(compute $\operatorname{tr}(M^2)$ entrywise in
        each case)}.
  \item Conclude: $q$ has signature $\bigl(\frac{n(n+1)}{2},
        \frac{n(n-1)}{2}\bigr)$ and rank $n^2$.
\end{enumerate}
\end{exercise}

\section{Problem: Cholesky, Hadamard, and the polar
decomposition}

\begin{problem}\label{pb:b2:quadratic:1}
The spectral theorem is a microscope; this problem uses it as a
factory. From Gram matrices we manufacture the \emph{Cholesky
factorization}\index{Cholesky factorization} (and identify the
Gauss pivots as ratios of minors), then prove \emph{Hadamard's
inequality}\index{Hadamard's inequality} on determinants, build
the \emph{polar decomposition}\index{polar decomposition} $A =
QS$ and the \emph{singular value decomposition}, classify plane
conics, and end with the simultaneous reduction of two forms ---
the theorem behind normal modes of oscillation. Throughout, $E =
\R^n$ with its standard inner product.

\medskip
\noindent\textbf{Part I --- Gram matrices and Cholesky.} For
vectors $v_1, \dots, v_n \in E$, their \emph{Gram matrix} is
$G = \bigl(\langle v_i, v_j\rangle\bigr)_{i,j}$.
\begin{enumerate}
  \item Show that $G$ is symmetric positive semidefinite, and
        positive definite if and only if $(v_1, \dots, v_n)$ is
        linearly independent \emph{(compute $X^{\mathsf T}GX$)}.
  \item Conversely, show that every symmetric positive
        semidefinite $A$ is a Gram matrix: $A = C^{\mathsf T}C$
        for some $C$ (use the square root of
        \cref{exo:b2:quadratic:6}), with $C$ invertible iff $A$
        is definite.
  \item Deduce that a positive semidefinite $A$ satisfies
        $\abs{a_{ij}} \leq \sqrt{a_{ii}\,a_{jj}}$ for all $i,
        j$ \emph{(restrict to two coordinates)} --- the
        Cauchy--Schwarz inequality, reread matricially.
  \item (Cholesky) Let $A$ be positive definite. Prove that
        there is a \emph{unique} upper triangular $T$ with
        positive diagonal entries such that
        \[
        A = T^{\mathsf T}\,T
        \]
        \emph{(existence: apply Gram--Schmidt to vectors
        realizing $A$ as a Gram matrix; uniqueness: if
        $T_1^{\mathsf T}T_1 = T_2^{\mathsf T}T_2$, show
        $T_1T_2^{-1}$ is orthogonal and triangular with
        positive diagonal, hence $I$)}.
  \item Show that the leading principal minors satisfy
        $\Delta_k = (t_{11}\cdots t_{kk})^2$, and deduce that
        the pivots produced by the Gauss reduction of a
        positive definite form, taken in the natural variable
        order, are
        \[
        d_k = \frac{\Delta_k}{\Delta_{k-1}}
        \qquad (\Delta_0 = 1) :
        \]
        the minors of Sylvester's criterion
        (\cref{exo:b2:quadratic:8}) and the pivots of Gauss are
        the same data. Check on
        \cref{ex:b2:quadratic:tworoads}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Hadamard's inequality.}
\begin{enumerate}[resume]
  \item Let $A$ be positive definite. Prove
        \[
        \det A \leq a_{11}\,a_{22}\cdots a_{nn}
        \]
        \emph{(normalize: $B = DAD$ with $D =
        \operatorname{diag}(a_{ii}^{-1/2})$ has unit diagonal;
        bound $\det B = \prod \mu_i$ by AM--GM against
        $\operatorname{tr} B = n$)}.
  \item Show that equality holds iff $A$ is diagonal.
  \item Deduce \emph{Hadamard's inequality}: for every real
        square matrix $M$ with columns $c_1, \dots, c_n$,
        \[
        \abs{\det M} \leq \prod_{i=1}^{n}\norm{c_i}_2 ,
        \]
        with equality (for invertible $M$) iff the columns are
        pairwise orthogonal \emph{(apply questions 6--7 to
        $M^{\mathsf T}M$)}.
  \item Geometric and combinatorial dividends: interpret
        question 8 as ``the volume of a parallelepiped is at
        most the product of its edge lengths''; and show that a
        matrix with all entries in $\intcc{-1}{1}$ has
        $\abs{\det M} \leq n^{n/2}$. (Matrices attaining this
        bound --- Hadamard matrices --- exist for $n = 1, 2$
        and many multiples of $4$; whether for \emph{all}
        multiples of $4$ is a famous open problem.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Polar decomposition and singular
values.}
\begin{enumerate}[resume]
  \item Let $A$ be invertible. Show that $A^{\mathsf T}A$ is
        positive definite, and that
        \[
        S = \sqrt{A^{\mathsf T}A}
        \quad\text{(the square root of
        \cref{exo:b2:quadratic:6})},
        \qquad Q = AS^{-1}
        \]
        give a factorization $A = QS$ with $Q$ orthogonal and
        $S$ positive definite.
  \item Prove that this factorization of an invertible $A$ is
        unique.
  \item Extend existence to arbitrary $A$: choose
        $\varepsilon_k \to 0$ with $A + \varepsilon_k I$
        invertible, write $A + \varepsilon_kI = Q_kS_k$, and
        use the compactness of $O(n)$
        (\cref{exo:b2:quadratic:5}) to extract $Q_k \to Q$;
        show $S_k = Q_k^{\mathsf T}(A + \varepsilon_kI)$
        converges to some positive semidefinite $S$ with $A =
        QS$ and $S = \sqrt{A^{\mathsf T}A}$. Where does
        uniqueness fail for singular $A$?
  \item (Singular value decomposition) Deduce that every real
        square $A$ writes
        \[
        A = U\,\Sigma\,V^{\mathsf T},
        \qquad U, V \in O(n),\quad
        \Sigma = \operatorname{diag}(\sigma_1, \dots,
        \sigma_n),\ \sigma_i \geq 0 ,
        \]
        where the $\sigma_i$ (the \emph{singular values}) are
        the eigenvalues of $\sqrt{A^{\mathsf T}A}$.
  \item Three consequences: $\vertiii{A}_2 = \sigma_{\max}$ for
        \emph{every} real $A$ (generalizing
        \cref{exo:b2:quadratic:7}); $\abs{\det A} =
        \sigma_1\cdots\sigma_n$; and the image of the unit
        sphere under an invertible $A$ is an ellipsoid with
        semi-axes $\sigma_1, \dots, \sigma_n$ along the columns
        of $U$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Conics, by the spectral theorem.}
A plane conic is the zero set of $f(x) = q(x) + \langle b,
x\rangle + c$, with $q \neq 0$ a quadratic form of matrix $A$,
$b \in \R^2$, $c \in \R$.
\begin{enumerate}[resume]
  \item Reduce $f$ by a rotation (principal axes,
        \cref{cor:b2:quadratic:principalaxes}) followed by a
        translation, and classify the possible nonempty,
        nondegenerate shapes by the signature of $q$: ellipse
        ($\det A > 0$), hyperbola ($\det A < 0$), parabola
        ($\det A = 0$, rank $1$, with the linear term not
        absorbed).
  \item Run the reduction in full for
        \[
        x^2 + 4xy + y^2 + 2x - 2y = 4 :
        \]
        rotated coordinates, reduced equation, nature and
        center of the conic.
  \item (Central conics) Suppose $\det A \neq 0$. Show the
        \emph{center} is $x_0 = -\frac12 A^{-1}b$, and that the
        congruence by $\begin{pmatrix} I & x_0\\ 0 &
        1\end{pmatrix}$ of the $3\times3$ matrix $\widetilde Q
        = \begin{pmatrix} A & b/2 \\ b^{\mathsf T}/2 &
        c\end{pmatrix}$ yields
        \[
        \det\widetilde Q = f(x_0)\,\det A :
        \]
        the central conic is degenerate (a point or two lines)
        exactly when $\det\widetilde Q = 0$.
  \item Verify question 17 on the example of question 16:
        compute $x_0$, $f(x_0)$ and $\det\widetilde Q$, and
        conclude again that the conic is a nondegenerate
        hyperbola.
  \item (A quadric pencil) For $\lambda \in \R$, classify the
        surface
        \[
        x^2 + y^2 + z^2 + 2\lambda(xy + yz + zx) = 1
        \]
        by the eigenvalues of its matrix \emph{(all-ones
        structure: eigenvalues $1 + 2\lambda$ and $1 -
        \lambda$ double)}: sphere/ellipsoid, cylinder, pair of
        planes, hyperboloids of one and two sheets, according
        to $\lambda$.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Two forms at once: simultaneous
reduction.}
\begin{enumerate}[resume]
  \item Let $q$ be positive definite and $q'$ an arbitrary
        quadratic form on $E$. Prove that there is a basis of
        $E$ which is orthonormal for $q$ and orthogonal for
        $q'$: in it, $q = \sum x_i^2$ and $q' = \sum \mu_i
        x_i^2$ \emph{(use $q$ as the inner product and apply
        the spectral theorem to the endomorphism representing
        $q'$)}.
  \item Matrix form: for $A$ positive definite and $B$
        symmetric, there is an invertible $P$ with
        $P^{\mathsf T}AP = I$ and $P^{\mathsf T}BP =
        \operatorname{diag}(\mu_1, \dots, \mu_n)$, where the
        $\mu_i$ are the roots of $\det(B - \mu A) = 0$.
  \item Run it in full for
        \[
        A = \begin{pmatrix} 2 & 1\\ 1 & 1 \end{pmatrix},
        \qquad
        B = \begin{pmatrix} 0 & 1\\ 1 & 0 \end{pmatrix} :
        \]
        the generalized eigenvalues $\mu_\pm$, and vectors
        diagonalizing both forms.
  \item Show that positive definiteness cannot be dropped: for
        \[
        A = \begin{pmatrix} 1 & 0\\ 0 & -1 \end{pmatrix},
        \qquad
        B = \begin{pmatrix} 0 & 1\\ 1 & 0 \end{pmatrix},
        \]
        no basis diagonalizes both forms \emph{(if $P$
        diagonalized both, $\det(B - \mu A)$ would split with
        real roots; compute it)}.
  \item Show that the $\mu_i$ of question 21 are the
        eigenvalues of $A^{-1}B$, and that $A^{-1}B$, while not
        symmetric in general, is always diagonalizable with
        real eigenvalues \emph{(conjugate by $\sqrt A$)}.
  \item Synthesis. In one sentence each: (i) the one theorem
        every Part leaned on; (ii) which results of Parts
        I--III survive for positive \emph{semi}definite
        matrices, and which need definiteness; (iii) the
        physical system whose small oscillations questions
        20--22 diagonalize (kinetic and potential energy as
        the two forms), and what the $\mu_i$ mean there.
\end{enumerate}
\end{problem}
