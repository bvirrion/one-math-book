\chapter{Affine Spaces}\label{ch:b2:affine}

Vector spaces have a privileged point --- the origin --- that
geometry does not want. An \emph{affine space} is a vector space
that has forgotten its origin: points and vectors become different
species, related by translation. This short chapter builds the
dictionary (points, barycenters, affine subspaces and maps), the
affine view of convexity, and the classification tools used in the
geometry chapters ahead.

\section{Points and vectors}

\begin{definition}\label{def:b2:affine:def}
An \emph{affine space}\index{affine space} directed by a real
vector space $E$ is a nonempty set $\mathcal{E}$ with a map
$(A, B) \mapsto \vect{AB} \in E$ satisfying
\[
\vect{AB} + \vect{BC} = \vect{AC}
\quad \text{(Chasles)},
\qquad
\text{for each } A,\ B \mapsto \vect{AB} \text{ is a bijection }
\mathcal{E} \to E .
\]
One writes $B = A + u$ for the unique point with $\vect{AB} = u$.
The \emph{dimension} of $\mathcal{E}$ is $\dim E$. Every vector
space is an affine space over itself ($\vect{AB} = B - A$); every
choice of origin $O \in \mathcal{E}$ identifies $\mathcal{E}$ with
$E$ via $M \mapsto \vect{OM}$.
\end{definition}

\begin{example}[An affine space with no natural
origin]\label{ex:b2:affine:planeexample}
The solution plane $\mathcal E = \{(x, y, z) \in \R^3 : x + y
+ z = 1\}$ is not a vector subspace ($0 \notin \mathcal E$),
but it is an affine space directed by $E = \{x + y + z =
0\}$: for $A, B \in \mathcal E$ the difference $\vect{AB} = B
- A$ lands in $E$ (the sums cancel), Chasles is inherited
from $\R^3$, and $B \mapsto \vect{AB}$ is bijective onto $E$.
No point of $\mathcal E$ is distinguished --- any choice of
``origin'' $O \in \mathcal E$ works equally well, and all the
identifications $M \mapsto \vect{OM}$ differ by translations.
This is the typical situation: solution sets of inhomogeneous
linear problems (linear systems, linear differential
equations in \cref{ch:b2:diffeq}) are affine, never linear,
and the slogan ``particular solution plus kernel'' is exactly
the statement $\mathcal F = A + F$ of the next definition.
\end{example}

\begin{definition}[Barycenter]\label{def:b2:affine:barycenter}
Let $(A_i, \lambda_i)_{i \leq k}$ be weighted points with
$\sum\lambda_i \neq 0$. The \emph{barycenter}\index{barycenter} $G
= \operatorname{bar}\bigl((A_i, \lambda_i)\bigr)$ is the unique
point with
\[
\sum_i \lambda_i\, \vect{GA_i} = 0
\qquad\text{equivalently}\qquad
\vect{OG} = \frac{1}{\sum\lambda_i}\sum_i \lambda_i\,\vect{OA_i}
\quad (\text{any } O).
\]
Barycenters are \emph{associative} (subgroups of points may be
replaced by their partial barycenter with the summed weight) and
invariant under rescaling of all weights.
\end{definition}

\begin{proof}[Proof of existence and the formulas]
Fix $O$ and write $s = \sum_i\lambda_i \neq 0$. By Chasles,
\[
\sum_i\lambda_i\,\vect{GA_i} = 0
\iff \sum_i\lambda_i\bigl(\vect{GO} + \vect{OA_i}\bigr) = 0
\iff s\,\vect{OG} = \sum_i\lambda_i\,\vect{OA_i},
\]
which determines $G = O + \frac1s\sum_i\lambda_i\vect{OA_i}$
uniquely. \emph{Independence of $O$:} for another origin
$O'$,
\[
\frac1s\sum_i\lambda_i\,\vect{O'A_i}
= \frac1s\sum_i\lambda_i\bigl(\vect{O'O} + \vect{OA_i}\bigr)
= \vect{O'O} + \frac1s\sum_i\lambda_i\,\vect{OA_i}
= \vect{O'G} :
\]
the same point $G$. \emph{Associativity:} split the index set
as $I \sqcup J$ with $s_I = \sum_{i\in I}\lambda_i \neq 0$,
and let $G_I$ be the barycenter of $(A_i, \lambda_i)_{i\in
I}$, so that $\sum_{i\in I}\lambda_i\vect{OA_i} =
s_I\,\vect{OG_I}$. Then
\[
s\,\vect{OG}
= \sum_{i\in I}\lambda_i\vect{OA_i} +
\sum_{j\in J}\lambda_j\vect{OA_j}
= s_I\,\vect{OG_I} + \sum_{j\in J}\lambda_j\vect{OA_j} :
\]
$G$ is the barycenter of $(G_I, s_I)$ together with $(A_j,
\lambda_j)_{j\in J}$, as claimed. \emph{Rescaling:} replacing
each $\lambda_i$ by $t\lambda_i$ ($t \neq 0$) multiplies $s$
and the weighted sum by $t$, leaving $\vect{OG}$ unchanged.
\end{proof}

\begin{remark}[Common pitfalls]
Two traps surround the definition. First, if the weights sum
to \emph{zero}, there is no barycenter: the map $O \mapsto
\sum\lambda_i\vect{OA_i}$ is then independent of $O$ and
defines a \emph{vector}, not a point --- for instance
$(A, -1;\ B, 1)$ encodes $\vect{AB}$. Keeping track of which
of the two objects a computation produces is half of
barycentric hygiene. Second, weights are only meaningful up
to a common nonzero factor; formulas like ``the coordinates
of $G$ are $\lambda_1, \dots, \lambda_k$'' presuppose a
normalization (usually $\sum\lambda_i = 1$), and forgetting
to normalize is the standard source of wrong ratios on a
figure.
\end{remark}

\begin{definition}[Affine subspaces; affine maps]\label{def:b2:affine:subspace}
An \emph{affine subspace} is a set $\mathcal{F} = A + F = \{A + u :
u \in F\}$ with $F$ a vector subspace (its \emph{direction});
equivalently, a nonempty set stable under barycenters. Affine
subspaces of $\R^n$ are exactly the solution sets of linear systems
$MX = B$ (Year 1: particular solution plus kernel). A map $f \colon
\mathcal{E} \to \mathcal{E}'$ is \emph{affine}\index{affine map}
when it preserves barycenters --- equivalently when
\[
f(A + u) = f(A) + \varphi(u)
\]
for a (unique) linear map $\varphi = \vec f$, the \emph{linear
part}. Affine maps of $\R^n$: $X \mapsto MX + C$. Compositions are
affine with composed linear parts; $f$ is bijective iff $\vec f$
is.
\end{definition}

\begin{proof}[Proof of the equivalence for maps]
If $f(A + u) = f(A) + \varphi(u)$: for a barycenter $G$ of $(A_i,
\lambda_i)$, expanding every point from $A$, $f(G) = f(A) +
\varphi(\vect{AG})$ and $\varphi(\vect{AG}) =
\frac{\sum\lambda_i\varphi(\vect{AA_i})}{\sum\lambda_i}$: $f(G)$ is
the barycenter of the images. Conversely, fix $A$ and define
$\varphi(u) = \vect{f(A)\,f(A + u)}$. \emph{Homogeneity:} $A + tu =
\operatorname{bar}\bigl(A, 1-t;\ A + u, t\bigr)$ for every real
$t$, so preservation of barycenters (with arbitrary real weights,
as hypothesized) gives $\varphi(tu) = t\,\varphi(u)$ directly.
\emph{Additivity:} $A + u + v = \operatorname{bar}\bigl(A + 2u,
\tfrac12;\ A + 2v, \tfrac12\bigr)$, so $\varphi(u + v) =
\tfrac12\varphi(2u) + \tfrac12\varphi(2v) = \varphi(u) +
\varphi(v)$, using homogeneity. Hence $\varphi$ is linear.
\end{proof}

\begin{remark}
The proof used barycenters with \emph{arbitrary real}
weights: the homogeneity step takes $t$ outside $\intcc01$.
If a map is only assumed to preserve barycenters with
nonnegative weights --- equivalently, midpoints and segments
--- linearity of the vector map no longer comes for free:
one only gets $\Q$-linearity, and a continuity hypothesis is
needed to conclude, exactly as in
\cref{exo:b2:affine:5}. Distinguishing ``preserves all
barycenters'' from ``preserves convex combinations'' is a
small but real subtlety of the affine vocabulary.
\end{remark}

\begin{example}[Classical barycenter geometry]\label{ex:b2:affine:median}
The centroid of a triangle $ABC$ is the barycenter $G =
\operatorname{bar}(A,1; B,1; C,1)$. Associativity with the
midpoint $A' = \operatorname{bar}(B, 1; C, 1)$ shows
\[
G = \operatorname{bar}(A, 1;\ A', 2) :
\]
$G$ lies on the median $AA'$ at two-thirds of it --- and likewise
for the other two medians: the three medians are concurrent, in
one line of barycentric calculus.
\end{example}

\begin{example}[The bimedians of a
quadrilateral]\label{ex:b2:affine:bimedians}
Let $ABCD$ be any quadrilateral (planar or not!) and consider
its \emph{bimedians}: the segments joining the midpoints of
opposite sides, $M_{AB}M_{CD}$ and $M_{BC}M_{DA}$. Introduce
the barycenter $G$ of $(A,1; B,1; C,1; D,1)$ and group the
weights two ways:
\[
G = \operatorname{bar}\bigl(M_{AB}, 2;\ M_{CD}, 2\bigr)
= \operatorname{bar}\bigl(M_{BC}, 2;\ M_{DA}, 2\bigr) :
\]
$G$ is the midpoint of \emph{both} bimedians --- so the two
bimedians always bisect each other, and the quadrilateral of
the four midpoints is a parallelogram (its diagonals are the
bimedians). No case analysis, no coordinates, and the
argument survives unchanged for a skew quadrilateral in
$\R^3$, where a picture-based proof would already be
delicate: associativity does not care about dimension.
\end{example}

\begin{example}[Classifying an affine map, start to
finish]\label{ex:b2:affine:dilation}
Let $f(x, y) = (2x - 1,\ 3y - 4)$ on $\R^2$. Its linear part
is $\varphi = \operatorname{diag}(2, 3)$, whose spectrum
$\{2, 3\}$ avoids $1$: by the fixed-point criterion proved
below (\cref{prop:b2:affine:fixedpoint}), $f$ has exactly one
fixed point, found by solving
\[
x = 2x - 1, \qquad y = 3y - 4
\qquad\Longrightarrow\qquad \Omega = (1, 2).
\]
Recentering at $\Omega$ (set $x = 1 + u$, $y = 2 + v$):
\[
f(1 + u,\ 2 + v) = (1 + 2u,\ 2 + 3v) :
\]
in the frame at $\Omega$, $f$ \emph{is} its linear part, an
anisotropic dilation stretching by $2$ horizontally and $3$
vertically from the center $(1, 2)$. The general lesson: an
affine map is ``linear map plus location data'', and the
location data collapses to one well-chosen origin whenever
$1$ is not an eigenvalue. Conversely, translating the origin
badly \emph{creates} the constant terms: affine geometry is
the art of choosing where to put $0$.
\end{example}

\begin{remark}[Method: concurrency and alignment by
barycenters]
\cref{ex:b2:affine:median} is an instance of a general
recipe. To prove that three cevians of a triangle are
concurrent, exhibit a \emph{single} weighted system $(A,
\alpha; B, \beta; C, \gamma)$ and use associativity three
ways: grouping $(B, C)$ shows the barycenter lies on the
cevian from $A$, grouping $(C, A)$ on the cevian from $B$,
grouping $(A, B)$ on the third. For the medians, the system
$(A, 1; B, 1; C, 1)$ does all the work; for cevians cutting
the sides in prescribed ratios, the weights are read off the
ratios. To prove three points \emph{aligned}, write one as a
barycenter of the other two (\cref{exo:b2:affine:2}), or use
the determinant criterion of \cref{exo:b2:affine:11}. Both
recipes replace geometric ingenuity by weight bookkeeping ---
this is precisely what barycentric calculus is for.
\end{remark}

\section{Convexity, affinely}

\begin{definition}\label{def:b2:affine:convex}
A subset $C$ of an affine space is \emph{convex} when it contains
every barycenter with \emph{nonnegative} weights of its points ---
equivalently, every segment $\intcc{A}{B} = \{\operatorname{bar}(A,
1-t; B, t) : t \in \intcc{0}{1}\}$ between its points. The
\emph{convex hull} $\operatorname{conv}(S)$ is the set of all
nonnegative-weight barycenters of points of $S$ --- the smallest
convex set containing $S$.
\end{definition}

\begin{example}[Epigraphs are convex
sets]\label{ex:b2:affine:epigraph}
The region $C = \{(x, y) : y \geq x^2\}$ above the parabola
is convex: for $(x_1, y_1), (x_2, y_2) \in C$ and $t \in
\intcc01$, the convexity inequality of the square function
gives
\[
\bigl((1-t)x_1 + tx_2\bigr)^2 \leq (1-t)x_1^2 + tx_2^2
\leq (1-t)y_1 + ty_2 ,
\]
so the barycenter stays above the parabola. The computation
is general: $\{y \geq f(x)\}$ is convex exactly when $f$ is a
convex function --- convex \emph{sets} and convex
\emph{functions} (\cref{ch:b2:realfun}) are two faces of one
notion, epigraphs being the dictionary. This is the geometric
reason support lines exist for convex functions, the fact
that will prove Jensen's inequality in
\cref{ch:b2:randomvar}.
\end{example}

\begin{example}[Redundant generators of a convex
hull]\label{ex:b2:affine:squarehull}
Let $S = \{(0,0), (2,0), (2,2), (0,2), (1,1)\}$. The fifth
point is the barycenter
\[
(1,1) = \operatorname{bar}\bigl((0,0), \tfrac12;\ (2,2),
\tfrac12\bigr),
\]
so it already lies in the hull of the other four:
$\operatorname{conv}(S)$ is the square with the four corners
as vertices. In general, a point of $S$ that is a
nonnegative-weight barycenter of the \emph{other} points of
$S$ can be deleted without changing the hull; the points that
can never be deleted (here the four corners) are the
\emph{extreme points} of the hull. Determining them is a pure
barycenter computation: $(2,0)$, say, cannot be written as
$\operatorname{bar}$ of the remaining points with nonnegative
weights, because the first coordinate would force all weight
onto points with $x = 2$, and the second coordinate then
fails. Convexity questions reduce, again and again, to
solving small weighted systems.
\end{example}

\begin{theorem}[Carathéodory]\label{thm:b2:affine:caratheodory}
In an affine space of dimension $n$, every point of
$\operatorname{conv}(S)$ is a barycenter of at most $n + 1$ points
of $S$.\index{Carathéodory's theorem}
\end{theorem}

\begin{proof}
Let $G = \operatorname{bar}(A_0, \lambda_0; \dots; A_k, \lambda_k)$
with $\lambda_i > 0$, $\sum\lambda_i = 1$, and $k + 1 > n + 1$
points. The $k$ vectors $\vect{A_0A_i}$ ($i \geq 1$) are linked
($k > n$): $\sum_{i\geq1}\mu_i \vect{A_0A_i} = 0$ nontrivially;
setting $\mu_0 = -\sum_{i\geq1}\mu_i$, we get weights $(\mu_i)$
with $\sum\mu_i = 0$, $\sum \mu_i\,\vect{OA_i} = 0$ (any $O$), not
all zero. Then for every real $t$ the weights $\lambda_i - t\mu_i$
still sum to $1$ and, since $\sum_i\mu_i\vect{OA_i} = 0$,
\[
\sum_i(\lambda_i - t\mu_i)\,\vect{OA_i}
= \sum_i\lambda_i\,\vect{OA_i} :
\]
they produce the \emph{same} point $G$. Now slide $t$ from
$0$: some $\mu_i$ is positive (they sum to zero and are not
all zero), so
\[
t^* = \min\Bigl\{\frac{\lambda_i}{\mu_i} : \mu_i > 0\Bigr\}
\]
is well defined and positive. At $t = t^*$: for indices with
$\mu_i > 0$, $\lambda_i - t^*\mu_i \geq 0$ by minimality,
with equality at a minimizing index; for indices with $\mu_i
\leq 0$, $\lambda_i - t^*\mu_i \geq \lambda_i > 0$. All
weights remain nonnegative and at least one has died: $G$ is
rewritten as a barycenter of fewer points. Iterate while more
than $n + 1$ points remain.
\end{proof}

\begin{example}
In the plane ($n = 2$): every point of the convex hull of a finite
set lies in a triangle with vertices in the set --- the geometric
content of Carathéodory, used in optimization and probability
(mixtures) alike.
\end{example}

\begin{example}[Running Carath\'eodory's
algorithm]\label{ex:b2:affine:caratheodoryrun}
Write the center of the square of
\cref{ex:b2:affine:squarehull} with its four corners $A_1 =
(0,0)$, $A_2 = (2,0)$, $A_3 = (2,2)$, $A_4 = (0,2)$:
\[
(1,1) = \operatorname{bar}\bigl(A_1, \tfrac14;\ A_2,
\tfrac14;\ A_3, \tfrac14;\ A_4, \tfrac14\bigr),
\]
four points in dimension $2$ --- one too many. The proof's
recipe asks for weights $(\mu_i)$ with $\sum\mu_i = 0$ and
$\sum\mu_i\vect{OA_i} = 0$: here $\mu = (1, -1, 1, -1)$ works
(the two diagonals share their midpoint). Sliding $\lambda_i
\mapsto \lambda_i - t\mu_i$ keeps the barycenter fixed for
every $t$; the extremal admissible value $t = \frac14$ makes
the weights $(0, \tfrac12, 0, \tfrac12)$, killing $A_1$ and
$A_3$ simultaneously:
\[
(1,1) = \operatorname{bar}\bigl(A_2, \tfrac12;\ A_4,
\tfrac12\bigr),
\]
a representation by two points --- even better than the three
that the theorem guarantees, because the center happens to
lie on a segment between generators. The algorithm is
entirely mechanical: find a dependence, slide until a weight
dies, repeat.
\end{example}

\section{Affine classification tools}

\begin{proposition}[Fixed points of affine maps]\label{prop:b2:affine:fixedpoint}
Let $f$ be an affine endomorphism of a finite-dimensional affine
space with linear part $\varphi$. If $1 \notin
\operatorname{Sp}(\varphi)$, then $f$ has exactly one fixed point
$\Omega$, and in the vectorialization at $\Omega$, $f$ \emph{is}
its linear part. (Translations, with $\varphi = \mathrm{id}$ and no
fixed point, are the basic obstruction.)
\end{proposition}

\begin{proof}
Fix $O$ and write $f(O + x) = f(O) + \varphi(x)$. The point
$O + x$ is fixed iff $O + x = f(O) + \varphi(x)$, i.e.
\[
(\mathrm{id} - \varphi)(x) = \vect{O f(O)} .
\]
In finite dimension, $\mathrm{id} - \varphi$ is invertible
iff $0$ is not an eigenvalue of $\mathrm{id} - \varphi$, iff
$1 \notin \operatorname{Sp}\varphi$ --- and in that case the
displayed equation has exactly one solution $x^*$, giving the
unique fixed point $\Omega = O + x^*$. Recentering: for any
vector $u$,
\[
f(\Omega + u) = f(\Omega) + \varphi(u) = \Omega + \varphi(u),
\]
so in the frame with origin $\Omega$ the map reads $u \mapsto
\varphi(u)$: purely linear. When $1 \in
\operatorname{Sp}\varphi$, either no fixed point exists (the
displayed equation may be unsolvable, as for a translation)
or a whole affine subspace of them does (add any eigenvector
of eigenvalue $1$ to a solution): uniqueness is exactly the
spectral condition.
\end{proof}

\begin{example}[Plane isometries, completed]\label{ex:b2:affine:isometries}
An affine isometry of the Euclidean plane has linear part in
$O(2)$: a rotation $R_\theta$ or a reflection (Year 1 volume). If
$\theta \neq 0$: $1 \notin \operatorname{Sp} R_\theta$, so the map
is a \emph{rotation about a unique center}
(\cref{prop:b2:affine:fixedpoint}). If the linear part is a
reflection: either a reflection in an axis (fixed points exist) or
a \emph{glide reflection} (reflection composed with a translation
along the axis, no fixed point). With translations, this is the
complete classification of plane isometries.
\end{example}

\begin{remark}[The plane isometries, at a glance]
Collecting the cases: identity; translations ($\vec f =
\mathrm{id}$, no fixed point unless trivial); rotations
(linear part $R_\theta$, $\theta \neq 0$: one center);
reflections (linear part a reflection, a line of fixed
points); glide reflections (same linear part, no fixed
point). Four families plus the identity, each recognized by
two data only: the linear part and the fixed-point set ---
the pattern of \cref{prop:b2:affine:fixedpoint} made
exhaustive.
\end{remark}

\begin{example}[A glide reflection, caught in the
act]\label{ex:b2:affine:glide}
Let $f(x, y) = (y + 1,\ x + 1)$. The linear part $(x, y)
\mapsto (y, x)$ is the reflection in the diagonal $y = x$, so
$1 \in \operatorname{Sp}\vec f$ and
\cref{prop:b2:affine:fixedpoint} is silent. Fixed points
would need $x = y + 1$ and $y = x + 1$ simultaneously:
impossible --- none exist, so $f$ is not a reflection. Squaring
settles the classification:
\[
f\bigl(f(x, y)\bigr) = f(y + 1,\ x + 1) = (x + 2,\ y + 2),
\]
the translation by $(2, 2)$: $f$ is the \emph{glide
reflection} with axis the line $y = x$ (shifted
appropriately: the midpoint of $M$ and $f(M)$ always lies on
$y = x + {}$constant, here $y = x$, as one checks on $M = (0,
0) \mapsto (1,1)$) and glide vector $(1, 1)$, half of $f
\circ f$. Compare with \cref{exo:b2:affine:6}, where the same
linear part but a different constant produced an honest
reflection: with eigenvalue $1$ present, the constant term
decides everything.
\end{example}

\begin{example}[Affine recursions are affine
dynamics]\label{ex:b2:affine:recursion}
The classical recursion $u_{n+1} = au_n + b$ ($a \neq 1$)
iterates the affine map $f(x) = ax + b$ of the line, whose
linear part $a$ avoids the eigenvalue $1$: there is a unique
fixed point $\omega = \frac{b}{1-a}$, and recentering there
(the one-dimensional case of the proposition above) turns
$f$ into multiplication by $a$:
\[
u_{n+1} - \omega = a\,(u_n - \omega)
\qquad\Longrightarrow\qquad
u_n = \omega + a^n(u_0 - \omega) .
\]
For $u_{n+1} = \frac{u_n}2 + 3$: $\omega = 6$ and $u_n = 6 +
(u_0 - 6)2^{-n} \to 6$. The recipe taught for such
recursions in \cref{ch:b2:series} --- ``subtract the fixed
point'' --- is exactly the vectorialization of an affine map
at its fixed point; convergence for $\abs a < 1$ is the
contraction phenomenon that \cref{ch:b2:metric} turned into
the Banach fixed-point theorem. One idea, three chapters.
\end{example}

\begin{example}[Finding the center of a rotation]\label{ex:b2:affine:rotationcenter}
Let $f(x, y) = (-y + 2,\ x)$. The linear part is
$\varphi(x, y) = (-y, x)$: the rotation of angle $\frac\pi2$,
whose spectrum $\{\iu, -\iu\}$ avoids $1$. By
\cref{prop:b2:affine:fixedpoint} there is exactly one fixed
point: $x = -y + 2$ and $y = x$ give $x = 1$, $y = 1$, so
$\Omega = (1, 1)$, and $f$ \emph{is} the rotation of center
$(1, 1)$ and angle $\frac\pi2$. The general lesson: when $1
\notin \operatorname{Sp}\vec f$, classifying $f$ costs one
linear system --- the geometry is entirely in the linear part,
the arithmetic entirely in locating the center.
\end{example}

\begin{remark}[Where affine language is used next]
Barycenters and affine maps are the grammar of the geometry
chapters ahead: tangent lines and planes are affine objects
(\cref{ch:b2:curves,ch:b2:surfaces}), an affine change of
variables multiplies areas and volumes by
$\abs{\det \vec f}$ (\cref{ch:b2:multint}), and expectation
is a barycenter with weights given by a probability law, which
is why convexity governs Jensen's inequality
(\cref{ch:b2:randomvar}). In the Year 3 volume the same
convexity vocabulary carries the study of $L^p$ norms and of
integral inequalities.
\end{remark}

\begin{remark}[Perspectives within this volume]
Two threads leave this chapter. The \emph{affine} thread:
tangent lines (\cref{ch:b2:curves}) and tangent planes
(\cref{ch:b2:surfaces}) are affine subspaces attached to
nonlinear objects, and the classification of quadrics in the
surfaces chapter runs on this chapter's center equation
$A\Omega = -b$. The \emph{convex} thread is longer: convexity
of half-planes and disks powers the Helly theory of the
weekend problem; convexity of functions gives Jensen's
inequality (\cref{ch:b2:randomvar}); and the final theorem of
the book --- the extinction criterion for branching processes
(\cref{ch:b2:genfun}) --- is decided by the position of a
convex curve relative to the diagonal, a picture that belongs
to this chapter as much as to probability. Barycenters
return there too: an expectation is a barycenter with
probability weights.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b2:affine:1}
In $\R^3$, are the following affine subspaces? Give directions and
dimensions. $\{x + y + z = 1\}$; $\;\{x + y + z = 1,\ x - z =
3\}$; $\;\{x^2 + y^2 = 1\}$; the solution set of $MX = B$ for a
given compatible system.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:affine:2}
Prove that three distinct points $A, B, C$ of an affine space are
aligned iff $C$ is a barycenter of $A$ and $B$, iff the vectors
$\vect{AB}, \vect{AC}$ are linked. Deduce Menelaus-style weight
bookkeeping: if $C = \operatorname{bar}(A, 1 - t; B, t)$, locate
$C$ for $t = \frac12$, $t = 2$, $t = -1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:affine:3}
Let $f$ be the affine map of $\R^2$ given by $f(X) = MX + C$ with
$M = \frac12\begin{pmatrix} 1 & 1\\ 1 & 1\end{pmatrix}$ and $C =
(1, 0)^{\mathsf T}$. Determine the image of $f$, its fixed points
(if any), and $f \circ f$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:affine:4}
(Associativity in action) In a triangle $ABC$, let $I, J, K$ divide
$BC$, $CA$, $AB$ in ratios $\vect{BI} = \frac13\vect{BC}$,
$\vect{CJ} = \frac13\vect{CA}$, $\vect{AK} = \frac13\vect{AB}$.
Express $I, J, K$ as barycenters and compute the barycenter of
$(I,1;J,1;K,1)$: what do you find, and why was it predictable?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:affine:5}
Prove that a map $f \colon \R^n \to \R^n$ preserving \emph{midpoints}
($f\bigl(\frac{A+B}{2}\bigr) = \frac{f(A) + f(B)}{2}$) and
\emph{continuous} is affine. \emph{(Show the vector map $u \mapsto
f(O + u) - f(O)$ is additive via midpoints, then
$\Q$-homogeneous, then $\R$-homogeneous by continuity ---
the same density strategy used for Cauchy's functional equation in
the Year 1 volume; re-derive the needed steps here.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:affine:6}
Classify the affine map $f(x, y) = (y + 1,\; x - 1)$ of the
Euclidean plane: linear part, fixed points, geometric nature
(reflection? glide?). Compute $f \circ f$ and conclude.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:affine:7}
(Radon) Let $A_1, \dots, A_{n+2}$ be points of an affine space of
dimension $n$. Prove that they can be split into two disjoint
groups whose convex hulls intersect. \emph{(As in Carathéodory's
proof, find weights $\mu_i$, not all zero, with $\sum\mu_i = 0$ and
$\sum\mu_i\vect{OA_i} = 0$; separate positive and negative weights
and normalize both sides.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:affine:8}
Let $f$ be an affine endomorphism of $\R^n$ with $f \circ f = f$.
Prove that $f$ is the affine projection onto the affine subspace
$\operatorname{Fix}(f) = \operatorname{im} f$ along the direction
$\ker\vec f$, and that conversely all such projections are
idempotent. \emph{(Show first that $\operatorname{im} f$ consists
of fixed points.)}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:affine:9}
Let $G = \operatorname{bar}(A, 1;\ B, 2;\ C, 3)$ in a triangle
$ABC$. Using associativity, show that the line $AG$ meets $BC$
at $M = \operatorname{bar}(B, 2;\ C, 3)$, and locate $G$ on the
segment $\intcc AM$; locate likewise the intersection of $BG$
with $CA$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:affine:10}
For $\lambda \neq 0$, the \emph{homothety} $h_{\Omega,
\lambda}$ is the affine map fixing $\Omega$ with linear part
$\lambda\,\mathrm{id}$. Prove that the composition
$h_{\Omega', \mu} \circ h_{\Omega, \lambda}$ is a homothety of
ratio $\lambda\mu$ when $\lambda\mu \neq 1$, and a translation
when $\lambda\mu = 1$; in the case $\lambda = \mu = -1$ (two
point reflections), compute the translation vector.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:affine:11}
(Menelaus) In a triangle $ABC$, let $A' \in (BC)$, $B' \in
(CA)$, $C' \in (AB)$, all distinct from the vertices, and
define $\alpha, \beta, \gamma$ by $\vect{A'B} =
\alpha\,\vect{A'C}$, $\vect{B'C} = \beta\,\vect{B'A}$,
$\vect{C'A} = \gamma\,\vect{C'B}$. Prove that $A', B', C'$ are
aligned if and only if $\alpha\beta\gamma = 1$. \emph{(Write
each point as a barycenter of two vertices; show that three
points are aligned iff their barycentric coordinate rows with
respect to $(A, B, C)$ form a singular $3 \times 3$ matrix.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:affine:12}
Prove that the convex hull of a compact subset $K$ of $\R^n$
is compact. \emph{(By \cref{thm:b2:affine:caratheodory},
$\operatorname{conv}(K)$ is the image of a compact set under a
continuous map.)} Show by an example in $\R^2$ that the convex
hull of a \emph{closed} set need not be closed.
\end{exercise}

theorem}

\begin{problem}[{Weekend problem --- Helly's theorem and two of
its dividends}]\label{pb:b2:affine:1}
Radon's lemma (\cref{exo:b2:affine:7}) says that $n + 2$
points of an $n$-dimensional affine space always split into
two groups with intersecting convex hulls. This problem turns
that one linear-algebra fact into a chain of theorems of
combinatorial geometry: Helly's intersection theorem, the
centerpoint theorem (a two-dimensional median), and Jung's
covering theorem. Throughout, the plane is $\R^2$ with its
usual Euclidean structure, and $\det$ is the determinant in
the canonical basis.\index{Helly's theorem}\index{Radon's
lemma}\index{Jung's theorem}\index{centerpoint}

\medskip
\noindent\textbf{Part I --- Barycentric coordinates.} Points
$A_0, \dots, A_k$ are \emph{affinely independent} when the
vectors $\vect{A_0A_1}, \dots, \vect{A_0A_k}$ are linearly
independent.
\begin{enumerate}
  \item Show that affine independence does not depend on the
        choice of the base point $A_0$, and that it is
        equivalent to: whenever two families of weights, each
        summing to $1$, define the same barycenter of $(A_0,
        \dots, A_k)$, the weights coincide.
  \item Let $(A, B, C)$ be affinely independent in the plane.
        Show that every point $M$ admits a unique triple
        $(\alpha, \beta, \gamma)$ with $\alpha + \beta +
        \gamma = 1$ and $M = \operatorname{bar}(A, \alpha; B,
        \beta; C, \gamma)$ --- its \emph{barycentric
        coordinates}.
  \item Prove the determinant formulas
        \[
        \alpha = \frac{\det(\vect{MB},
        \vect{MC})}{\det(\vect{AB}, \vect{AC})},
        \qquad
        \beta = \frac{\det(\vect{MC},
        \vect{MA})}{\det(\vect{AB}, \vect{AC})},
        \qquad
        \gamma = \frac{\det(\vect{MA},
        \vect{MB})}{\det(\vect{AB}, \vect{AC})} :
        \]
        barycentric coordinates are ratios of signed areas.
  \item The lines $BC$, $CA$, $AB$ are the coordinate lines
        $\{\alpha = 0\}$, $\{\beta = 0\}$, $\{\gamma = 0\}$.
        Show that $M$ lies in the closed triangle
        $\operatorname{conv}\{A, B, C\}$ iff $\alpha, \beta,
        \gamma \geq 0$, and that the three lines cut the plane
        into exactly seven regions, classified by the signs of
        $(\alpha, \beta, \gamma)$ (the sign pattern $(-, -,
        -)$ being impossible).
  \item Let $u \colon \R^2 \to \R$ be an affine map (an
        \emph{affine form}). Show $u(M) = \alpha\,u(A) +
        \beta\,u(B) + \gamma\,u(C)$, that the level sets of a
        nonconstant affine form are lines, that every line
        arises this way, and that the closed half-planes $\{u
        \geq c\}$ are convex.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Radon partitions, refined.} A
family of $n + 2$ points of $\R^n$ is in \emph{general
position} when every $n + 1$ of them are affinely independent.
An \emph{affine dependence} of $(A_1, \dots, A_{n+2})$ is a
family $(\mu_i)$ with $\sum_i \mu_i = 0$ and $\sum_i
\mu_i\,\vect{OA_i} = 0$ for one (hence every) origin $O$.
\begin{enumerate}[resume]
  \item Compute a nonzero affine dependence of the four
        points $A_1 = (0,0)$, $A_2 = (3,0)$, $A_3 = (0,3)$,
        $A_4 = (1,1)$; give the Radon partition and the Radon
        point.
  \item Show that for points in general position the vector
        space of affine dependences has dimension exactly
        $1$, and that a nonzero dependence has \emph{no}
        vanishing coefficient.
  \item Deduce that the Radon partition of $n + 2$ points in
        general position is unique (up to swapping the two
        blocks), each block being the set of indices where
        $\mu_i$ has one fixed sign.
  \item For four points of the plane in general position,
        show the dichotomy: either the partition has type
        $(1, 3)$ --- one point interior to the triangle of
        the other three --- or type $(2, 2)$: the four points
        are in convex position and the segments joining the
        two pairs (the diagonals) intersect, at the Radon
        point.
  \item Carry out question 6 for the unit square $(0,0)$,
        $(1,0)$, $(1,1)$, $(0,1)$: dependence, partition,
        Radon point.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Helly's theorem in the plane.}
\begin{enumerate}[resume]
  \item Let $C_1, C_2, C_3, C_4$ be convex subsets of $\R^2$,
        any three of which have a common point. Pick $x_i \in
        \bigcap_{j \neq i} C_j$ and apply Radon's lemma to
        $x_1, \dots, x_4$: show that the Radon point belongs
        to all four sets. \emph{(For each $k$, the block not
        containing $x_k$ consists of points of $C_k$.)}
  \item (Helly) Let $C_1, \dots, C_m$ ($m \geq 3$) be convex
        subsets of $\R^2$, any three of which intersect.
        Prove $\bigcap_{i=1}^m C_i \neq \emptyset$, by
        induction on $m$: replace $C_{m-1}$ and $C_m$ by
        $C_{m-1} \cap C_m$ and check the hypothesis for the
        new family using question 11.
  \item Three counterexamples, one per hypothesis: (a) the
        three closed edges of a triangle pairwise intersect
        but have no common point ($3$ cannot be lowered to
        $2$); (b) the four sets $S_i = \{x_1, \dots, x_4\}
        \setminus \{x_i\}$, for four points in general
        position, satisfy the triple-intersection hypothesis
        but not the conclusion (convexity matters); (c) the
        closed half-planes $H_k = \intco{k}{+\infty} \times
        \R$, $k \in \N$, pairwise and triple-wise intersect
        but $\bigcap_k H_k = \emptyset$ (infinite families
        need compactness).
  \item (Compact Helly) Let $(K_i)_{i \in I}$ be an arbitrary
        family of compact convex subsets of $\R^2$, any three
        of which intersect. Using question 12 and the
        Borel--Lebesgue property
        (\cref{thm:b2:metric:borellebesgue}), show
        $\bigcap_{i \in I} K_i \neq \emptyset$.
  \item (First dividend) Let $S$ be a finite set of points of
        the plane and $r > 0$. Show: if every three points of
        $S$ lie in some closed disk of radius $r$, then $S$
        lies in one closed disk of radius $r$. \emph{(Apply
        Helly to the disks $\overline D(p, r)$, $p \in S$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The centerpoint theorem.} A
\emph{centerpoint} of a finite set $S$ of $n$ points of the
plane is a point $c$ (not necessarily in $S$) such that every
closed half-plane containing $c$ contains at least $n/3$
points of $S$.
\begin{enumerate}[resume]
  \item (Dimension $1$) For reals $x_1 \leq \dots \leq x_n$,
        show that the median $c = x_{\lceil n/2 \rceil}$
        satisfies: every closed half-line containing $c$
        contains at least $n/2$ of the $x_i$.
  \item (Counting lemma) If $A, B, C$ are subsets of $S$ with
        $\abs A, \abs B, \abs C > \tfrac{2n}3$, show $A \cap
        B \cap C \neq \emptyset$.
  \item Let $m = \floor{2n/3} + 1$ and let $\mathcal F$ be
        the (finite) family of the convex hulls
        $\operatorname{conv}(T)$, $T \subseteq S$, $\abs T =
        m$. Show that any three members of $\mathcal F$ have
        a common point, and deduce from Helly a point $c$
        common to all of them.
  \item Prove that this $c$ is a centerpoint of $S$: the
        \emph{centerpoint theorem}. \emph{(If a closed
        half-plane through $c$ contained fewer than $n/3$
        points, its open complement would contain a set $T$
        of $m$ points, and $\operatorname{conv}(T)$ would
        avoid $c$.)}
  \item Sharpness: let $n = 3k$ and place $k$ points in each
        of three disks of small radius $\varepsilon$ centered
        at the vertices of a large triangle. Show that for
        \emph{every} point $c$ of the plane some closed
        half-plane containing $c$ contains at most $n/3$
        points of $S$, so the constant $1/3$ cannot be
        improved. \emph{(Among the three directions from $c$
        to the disk centers, two make an angle at most
        $2\pi/3$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Jung's theorem and synthesis.}
\begin{enumerate}[resume]
  \item (Triangle lemma) Let $P, Q, R$ be three points with
        pairwise distances $\leq 1$. Show they lie in a
        closed disk of radius $1/\sqrt3$. \emph{(If some
        angle is $\geq \pi/2$, take the disk on the longest
        side as diameter, using the median formula
        $\norm{RM}^2 = \tfrac12\norm{RP}^2 +
        \tfrac12\norm{RQ}^2 - \tfrac14\norm{PQ}^2$; if the
        triangle is acute, bound the circumradius $a/(2\sin
        \widehat A)$ using its largest angle, which lies in
        $\intco{\pi/3}{\pi/2}$.)}
  \item (Jung) Deduce: every compact subset of the plane of
        diameter $\leq 1$ is contained in a closed disk of
        radius $1/\sqrt3$.
  \item Sharpness: for the equilateral triangle $A_1A_2A_3$
        of side $1$ with centroid $G$, prove the Leibniz
        identity $\sum_i \norm{\vect{OA_i}}^2 =
        3\norm{\vect{OG}}^2 + \sum_i \norm{\vect{GA_i}}^2$
        for every point $O$, and conclude that any disk
        containing the three vertices has radius $\geq
        1/\sqrt3$, with equality only for the circumdisk.
  \item (Helly in $\R^n$) State and prove Helly's theorem in
        $\R^n$: if finitely many convex sets are such that
        any $n + 1$ of them intersect, then all of them
        intersect. \emph{(Radon's lemma
        \cref{exo:b2:affine:7} handles $n + 2$ sets; then
        induct as in question 12.)}
  \item Synthesis. Assemble the chain
        \[
        \text{affine dependence} \Rightarrow \text{Radon}
        \Rightarrow \text{Helly} \Rightarrow
        \text{centerpoint and Jung},
        \]
        indicating in one sentence each: where linear algebra
        enters, where the signs of the weights enter, where
        convexity enters, and which single step used the
        dimension of the plane. What do the constants $3$ (in
        Helly), $1/3$ (centerpoint) and $1/\sqrt3$ (Jung)
        become in $\R^n$? (State without proof.)
\end{enumerate}
\end{problem}
