Este capítulo extiende la integración de intervalos a curvas y a
dominios del plano y del espacio. \omterm{def:b2:multint:lineint}{Integrales de línea} integrar un
\emph{\omterm{def:b2:multint:lineint}{forma diferencial}} $P\,\dd x + Q\,\dd y$ a lo largo de una orientación
arco; integrales dobles y triples integran funciones sobre

\section{Integrales de línea}

\begin{definition}[Forma diferencial; integral de linea]\label{def:b2:multint:lineint}
Sea $U \subseteq \R^2$ \omterm{def:b2:metric:topology}{abierto}. A \emph{diferencial
forma}\index{forma diferencial} de
grado $1$ y clase $\mathcal{C}^0$ en $U$ es una expresión
$\omega = P\,\dd x + Q\,\dd y$ con $P, Q \colon U \to \R$
\omterm{def:b2:metric:continuity}{continuo} --- formalmente, un mapa \omterm{def:b2:metric:continuity}{continuo} de $U$ al \omterm{def:b2:linalg:dual}{dual}
de $\R^2$, $\omega(M) = P(M)\,e_1^* + Q(M)\,e_2^*$. por un
$\mathcal{C}^1$ arco $\gamma \colon [a, b] \to U$, $\gamma(t) =
(x(t), y(t))$, el \emph{integral de línea}\index{integral de línea} de
$\omega$ junto $\gamma$
es
\[
\int_\gamma \omega
= \int_a^b \Bigl(P(\gamma(t))\,x'(t)
+ Q(\gamma(t))\,y'(t)\Bigr)\,\dd t .
\]
Las definiciones se extienden textualmente a $\R^3$ (formas $P\,\dd x +
Q\,\dd y + R\,\dd z$) y a los arcos $\mathcal{C}^1$ por partes (suma
sobre las piezas).
\end{definition}

\begin{proposition}[Invariancia y orientación]\label{prop:b2:multint:lineinv}
El \omterm{def:b2:multint:lineint}{integral de línea} no cambia bajo un $\mathcal{C}^1$ creciente.
\omterm{def:b2:curves:reparam}{cambio de parametro}, y cambia de signo a uno decreciente. eso
por lo tanto solo depende del \emph{orientado} \omterm{def:b2:curves:reparam}{arco geométrico}.
\end{proposition}

\begin{proof}
Si $\theta \colon [c, d] \to [a, b]$ es un \omterm{def:b2:curves:reparam}{cambio de parametro} y
$\tilde\gamma = \gamma \circ \theta$, luego por la regla de la cadena y
el cambio de una variable de variables $t = \theta(u)$,
\[
\int_{\tilde\gamma}\omega
= \int_c^d \bigl(P(\gamma(\theta(u)))\,x'(\theta(u))
+ Q(\gamma(\theta(u)))\,y'(\theta(u))\bigr)\,\theta'(u)\,\dd u
= \pm\int_a^b \bigl(Px' + Qy'\bigr)(t)\,\dd t ,
\]
con signo $+$ si $\theta$ es creciente ($\theta(c) = a$) y $-$
si disminuye (los límites se intercambian).
\end{proof}

\begin{example}[Trabajo de una fuerza; circulación]\label{ex:b2:multint:work}
Si $F = (P, Q)$ es un campo de fuerza, $\int_\gamma P\dd x + Q\dd y =
\int_a^b \langle F(\gamma(t)), \gamma'(t)\rangle\,\dd t$ es el
\emph{trabajar} de $F$ junto con $\gamma$. Para $\omega = -y\,\dd x +
x\,\dd y $a lo largo del círculo unitario $\gamma(t) =
(\cos t, \sin t)$ en sentido antihorario:
\[
\int_\gamma \omega
= \int_0^{2\pi}\bigl((-\sin t)(-\sin t)
+ \cos t\cos t\bigr)\,\dd t = 2\pi ,
\]
dos veces el \omterm{def:b2:multint:domain}{área} adjunto --- un primer indicio de Green--Riemann.
\end{example}

\begin{example}[Una integral, dos parametrizaciones, una
trampa de señales]\label{ex:b2:multint:semicircle}
Calcule $\int_\gamma x\,\dd y$ a lo largo de la unidad superior
semicírculo de $(1, 0)$ a $(-1, 0)$. Con $\gamma(t) =
(\cos t, \sin t)$, $t \in \intcc0\pi$:
\[
\int_0^\pi\cos t\cdot\cos t\,\dd t = \frac\pi2 .
\]
Con la parametrización del gráfico $x \mapsto (x, \sqrt{1 -
x^2})$, $x$de $1$a$-1$ (¡tenga en cuenta la dirección!):
\[
\int_1^{-1}x\cdot\frac{-x}{\sqrt{1 - x^2}}\,\dd x
= \int_{-1}^{1}\frac{x^2}{\sqrt{1 - x^2}}\,\dd x
= \frac\pi2
\]
($x = \sin u$ lo reduce a una integral de Wallis). Mismo valor,
como garantiza \cref{prop:b2:multint:lineinv} --- pero sólo
porque ambas corridas van de $(1,0)$ a $(-1,0)$; revertir el
viajar invierte el cartel. Cerrando el camino a lo largo del eje $x$
(donde $\dd y = 0$) no agrega nada, y el total $\frac\pi2$
es el \omterm{def:b2:multint:domain}{área} del medio disco: la primera instancia del
\omterm{def:b2:multint:domain}{área límite} fórmulas de Green--Riemann a continuación.
\end{example}

\begin{definition}[Formularios exactos y cerrados]\label{def:b2:multint:exact}
El formulario $\omega = P\,\dd x + Q\,\dd y$ de la clase $\mathcal{C}^0$
es \emph{exacto}\index{forma exacta} en $U$ si hay $f \in \mathcal{C}^1(U)$
(a \emph{potencial}\index{potencial}) con $\omega = \dd f$, es decir\ $P = f_x$ y $Q
= f_y$. Un formulario $\mathcal{C}^1$ es \emph{cerrado}\index{forma cerrada} si $P_y = Q_x$ está en
$U$.
\end{definition}

\begin{theorem}[Teorema fundamental de la recta
integrales]\label{thm:b2:multint:ftc}
Si $\omega = \dd f$ es \omterm{def:b2:multint:exact}{exacto} y $\gamma$ es por partes
$\mathcal{C}^1$ arco en $U$ de $A$ a $B$, luego
\[
\int_\gamma \omega = f(B) - f(A) .
\]
En particular, la integral de un \omterm{def:b2:multint:exact}{forma exacta} a lo largo de cualquier arco cerrado
es cero y cada formulario \omterm{def:b2:multint:exact}{exacto} $\mathcal{C}^1$ está cerrado.
\end{theorem}

\begin{proof}
$\frac{\dd}{\dd t}f(\gamma(t)) = f_x(\gamma(t))x'(t) +
f_y(\gamma(t))y'(t)$ por la regla de la cadena
(\cref{ch:b2:diffcalc}), por lo que el integrando en
\cref{def:b2:multint:lineint} es la derivada de $t \mapsto
f(\gamma(t))$, y el teorema fundamental del cálculo da la
resultado en cada pieza; El telescopio de valores intermedios. cerrazón
de las formas \omterm{def:b2:multint:exact}{exacto} $\mathcal{C}^1$ es el teorema de Schwarz: $P_y =
f_{xy} = f_{yx} = Q_x$.
\end{proof}

\begin{example}[Reconstruyendo un potencial]\label{ex:b2:multint:potential}
Deja $\omega = y\,\eu^{xy}\,\dd x + (x\,\eu^{xy} + 2y)\,\dd y$
en $\R^2$. Es cerrado: ambas derivadas cruzadas son iguales.
$\eu^{xy}(1 + xy)$. Para encontrar un \omterm{def:b2:multint:exact}{potencial}, integre $P$ en
$x$ en $y$ fijo:
\[
f(x, y) = \int y\,\eu^{xy}\,\dd x = \eu^{xy} + c(y),
\]
luego ajuste $c$ haciendo coincidir $f_y$: $x\,\eu^{xy} + c'(y) =
x\,\eu^{xy} + 2y $da $c(y) = y^2$. Entonces $f(x,y) = \eu^{xy} +
y^2$, y para cualquier arco $\mathcal C^1$ por partes de $(0,0)$
a $(1,1)$,
\[
\int_\gamma\omega = f(1,1) - f(0,0) = (\eu + 1) - 1 = \eu ,
\]
independientemente de la ruta --- la receta de dos pasos (integrar
en $x$, correcto en $y$) es lo contrario práctico de
\cref{thm:b2:multint:ftc} en dominios donde está \omterm{def:b2:multint:exact}{formas cerradas}
\omterm{def:b2:multint:exact}{exacto}.
\end{example}

\begin{example}[Cerrado no implica exacto]\label{ex:b2:multint:angleform}
En $U = \R^2 \setminus \{0\}$, el \emph{forma de ángulo}
\[
\omega = \frac{-y\,\dd x + x\,\dd y}{x^2 + y^2}
\]
está cerrado (cálculo directo: tanto $P_y$ como $Q_x$ son iguales
$\frac{y^2 - x^2}{(x^2+y^2)^2}$), pero es integral a lo largo de la unidad.
el círculo es $2\pi \neq 0$ (el mismo cálculo que
\cref{ex:b2:multint:work}, dividido por $1$): $\omega$ no es \omterm{def:b2:multint:exact}{exacto}
en $U$. Localmente, $\omega = \dd\theta$ para una determinación.
$\theta$ del ángulo polar; el fracaso es global --- el ángulo
No se puede definir \omterm{def:b2:metric:continuity}{continuamente} alrededor del pinchazo. En dominios
sin agujeros la patología desaparece: en un \emph{en forma de estrella}
Conjunto \omterm{def:b2:metric:topology}{abierto}, cada formulario $\mathcal{C}^1$ cerrado es \omterm{def:b2:multint:exact}{exacto}
(Lema de Poincaré, \cref{exo:b2:multint:8}).
\end{example}

\section{Integrales dobles}

Damos por sentada la teoría de la integral de Riemann en una
variable (volumen del año 1 y \cref{ch:b2:integration}) y boceto
su versión de dos variables. Una función $f$ \omterm{def:b2:metric:continuity}{continuo} en un rectángulo
$R = [a, b] \times [c, d]$ tiene una integral doble $\iint_R f$,
definido por sumas de Riemann sobre cuadrículas exactamente como en una variable, y
calculado por iteración:

\begin{theorem}[Fubini sobre un rectángulo]\label{thm:b2:multint:fubini}
Para $f$ \omterm{def:b2:metric:continuity}{continuo} en $R = [a,b] \times [c,d]$,
\[
\iint_R f
= \int_a^b \Bigl(\int_c^d f(x, y)\,\dd y\Bigr)\dd x
= \int_c^d \Bigl(\int_a^b f(x, y)\,\dd x\Bigr)\dd y .
\]
\end{theorem}

\begin{proof}
Establezca $F(x) = \int_c^d f(x, y)\,\dd y$. Uniforme \omterm{def:b2:metric:continuity}{continuidad} de $f$ en
el \omterm{def:b2:metric:compact}{compacto} $R$ hace $F$ \omterm{def:b2:metric:continuity}{continuo} (estimación dominada:
$\abs{F(x) - F(x')} \leq (d - c)\sup_y\abs{f(x,y) - f(x',y)}$).
Ahora subdivida $[a,b]$ y $[c,d]$ en $n$ partes iguales, dando un
cuadrícula de celdas $R_{ij}$ de \omterm{def:b2:multint:domain}{área} $\Delta x\,\Delta y$. En cada celda,
$\inf_{R_{ij}} f \cdot \Delta x \Delta y \leq
\int_{x_{i-1}}^{x_i}\int_{y_{j-1}}^{y_j} f(x,y)\,\dd y\,\dd x \leq
\sup_{R_{ij}} f \cdot \Delta x \Delta y$ por monotonicidad del
Integral de una variable (aplicada dos veces). Sumando las celdas, el
La integral iterada $\int_a^b F$ se aprieta entre las partes inferior y
sumas de Riemann superiores de la cuadrícula; por uniforme \omterm{def:b2:metric:continuity}{continuidad} ambas sumas
convergen al valor común que define $\iint_R f$ como $n \to
\infty$. El mismo argumento se aplica con los roles de $x$y $y$.
intercambiadas, por lo que ambas integrales iteradas son iguales a $\iint_R f$.
\end{proof}

\begin{remark}
\omterm{def:b2:metric:continuity}{Continuidad} en un rectángulo \omterm{def:b2:metric:compact}{compacto} está haciendo un trabajo real en
La prueba de Fubini: proporciona el uniforme \omterm{def:b2:metric:continuity}{continuidad} que
exprime las sumas de Riemann. Para integrandos más salvajes, el
declaración realmente falla --- hay funciones cuyos dos
las integrales iteradas existen y difieren. El general honesto
El teorema, con la integrabilidad como única hipótesis, es
El teorema de Fubini para la integral de Lebesgue, demostrado en el
Volumen del año 3; Todo en este capítulo se mantiene dentro del
\omterm{def:b2:metric:continuity}{continuo} configuración donde se encuentra la prueba elemental anterior
\omterm{def:b2:metric:complete}{completo}.
\end{remark}

\begin{definition}[Dominios elementales]\label{def:b2:multint:domain}
Un dominio $D \subseteq \R^2$ es \emph{$y$-elemental} si
\[
D = \{(x, y) : a \leq x \leq b,\
\varphi_1(x) \leq y \leq \varphi_2(x)\}
\]
con $\varphi_1 \leq \varphi_2$ \omterm{def:b2:metric:continuity}{continuo} en $[a,b]$
($x$-elemental: simétricamente). Para $f$ \omterm{def:b2:metric:continuity}{continuo} en un
$y$-elemental $D$,
\[
\iint_D f
= \int_a^b\Bigl(
\int_{\varphi_1(x)}^{\varphi_2(x)} f(x,y)\,\dd y\Bigr)\dd x ,
\]
y uno verifica (extendiendo $f$ mediante un argumento de aproximación, o
subdividiendo) que cuando $D$ es elemental en ambas direcciones el
dos integrales iteradas concuerdan. Dominios cortados en un número finito
Las piezas elementales se manejan por aditividad, y las
\emph{área}\index{área!de un dominio plano} de $D$ es $\operatorname
{Area}(D) = \iint_D 1$.
\end{definition}

\begin{example}\label{ex:b2:multint:triangle}
En el triángulo $D = \{0 \leq x \leq 1,\ 0 \leq y \leq x\}$:
\[
\iint_D xy \,\dd x\,\dd y
= \int_0^1 x\Bigl(\int_0^x y\,\dd y\Bigr)\dd x
= \int_0^1 x\cdot\frac{x^2}{2}\,\dd x = \frac18 .
\]
Intercambiando el orden ($x$ de $y$ a $1$): $\int_0^1
y\bigl(\int_y^1 x\,\dd x\bigr)\dd y = \int_0^1
y\,\frac{1 - y^2}{2}\,\dd y = \frac18$ --- mismo valor, diferente
cálculo: elegir bien el orden de integración es la mitad del
artesanía.
\end{example}

\begin{example}[Cuando solo funciona un pedido]\label{ex:b2:multint:swaporder}
Calcule $I = \displaystyle\int_0^1\!\!\int_x^1
\eu^{y^2}\,\dd y\,\dd x$. Como está escrito, la integral interna
$\int\eu^{y^2}\dd y$ no tiene antiderivada elemental: la
el cálculo está atascado. Pero el dominio es el triángulo $0 \leq
x \leq y \leq 1$, que es elemental en ambas direcciones;
cambiando el orden,
\[
I = \int_0^1\!\!\int_0^y \eu^{y^2}\,\dd x\,\dd y
= \int_0^1 y\,\eu^{y^2}\,\dd y
= \Bigl[\tfrac12\eu^{y^2}\Bigr]_0^1 = \frac{\eu - 1}{2} .
\]
La variable interna $x$ no apareció en ninguna parte del integrando, por lo que
integrarlo primero produjo exactamente el factor $y$ que
hace que la integral exterior sea inmediata. Moraleja: Fubini no es justo
una licencia para iterar --- es una licencia para \emph{elegir},
y el orden correcto puede convertir una integral imposible en una
una sola línea. Siempre dibuje el dominio y lea ambos.
descripciones antes de comenzar.
\end{example}

\begin{theorem}[Cambio de variables]\label{thm:b2:multint:changevar}
Sea $\Phi \colon U' \to U$ un difeomorfismo $\mathcal{C}^1$
entre conjuntos \omterm{def:b2:metric:topology}{abierto} de $\R^2$, sea $K \subseteq U$ un \omterm{def:b2:metric:compact}{compacto}
dominio cortado en pedazos elementales con $K' = \Phi^{-1}(K)$, y
sea $f$ \omterm{def:b2:metric:continuity}{continuo} en $K$. entonces
\[
\iint_K f(x, y)\,\dd x\,\dd y
= \iint_{K'} f\bigl(\Phi(u, v)\bigr)\,
\abs{\det J_\Phi(u, v)}\,\dd u\,\dd v .
\]
\end{theorem}
\admitted

\begin{remark}
La prueba \omterm{def:b2:metric:complete}{completo} --- se aproxima a $\Phi$ por su \omterm{def:b2:diffcalc:differential}{diferencial} en
una cuadrícula fina y controlar las celdas límite --- aunque es largo
no profundo; se realiza íntegramente en la teoría de medidas del Año 3, como
consecuencia de la teoría de Lebesgue. La heurística es la imagen.
ya usado para la superficie \omterm{def:b2:multint:domain}{área}: un pequeño cuadrado de lado $\dd u$ en
$(u, v)$ se asigna, en primer orden, al paralelogramo abarcado
por $\Phi_u\,\dd u$ y $\Phi_v\,\dd v$, cuyo \omterm{def:b2:multint:domain}{área} es
$\abs{\det J_\Phi}\,\dd u\,\dd v$
(\cref{lem:b2:surfaces:lagrange}).
\end{remark}

\begin{remark}[Método: elegir el cambio de variables]
Tres reflejos cubren la mayoría de los casos. \emph{Simetría de la
integrando}: $x^2 + y^2$ pide polar, una estructura de producto
Requiere mantener los ejes cartesianos. \emph{Forma de la
límite}: límites $u(x,y) = c_1$, $v(x,y) = c_2$ ruegan por
las coordenadas $(u, v)$, como en el hiperbólico
ejemplo de región a continuación: el dominio se convierte en un rectángulo,
que es toda la victoria. \emph{estructura lineal}:
Las expresiones en $x + y$ y $x - y$ invitan al grado $45$.
rotación o cizallamiento (\cref{ex:b2:multint:affine}). en total
casos, tres casillas a marcar antes de integrar: el mapa es un
biyección del nuevo dominio sobre el antiguo; es jacobiano
se calcula \emph{en la dirección realmente utilizada} (invertir en
el final si es más fácil); y entra el jacobiano con su
valor absoluto.
\end{remark}

\begin{example}[Coordenadas polares]\label{ex:b2:multint:polar}
$\Phi(\rho, \alpha) = (\rho\cos\alpha,\ \rho\sin\alpha)$ tiene
\[
J_\Phi = \begin{pmatrix}
\cos\alpha & -\rho\sin\alpha\\
\sin\alpha & \rho\cos\alpha
\end{pmatrix},
\qquad \det J_\Phi = \rho ,
\]
entonces $\dd x\,\dd y = \rho\,\dd\rho\,\dd\alpha$. Para el disco $D_R$
de radio $R$:
\[
\iint_{D_R} e^{-(x^2 + y^2)}\,\dd x\,\dd y
= \int_0^{2\pi}\!\!\int_0^R e^{-\rho^2}\rho\,\dd\rho\,\dd\alpha
= \pi\bigl(1 - e^{-R^2}\bigr)
\xrightarrow[R\to\infty]{} \pi .
\]
Comparando con el cuadrado $[-R, R]^2$ (que está apretado entre
los discos $D_R$ y $D_{R\sqrt2}$, todos los integrandos positivos) da
$\bigl(\int_{-\infty}^\infty e^{-x^2}\dd x\bigr)^2 = \pi$:
\[
\boxed{\ \int_{-\infty}^{+\infty} e^{-x^2}\,\dd x = \sqrt{\pi}\ }
\]
--- la integral de Gauss nuevamente, ahora por su prueba más famosa
(compare la derivación de una variable en
\cref{ch:b2:integration}).
\end{example}

\begin{example}[Cambios afines de variables]\label{ex:b2:multint:affine}
Para un \omterm{def:b2:affine:subspace}{aplicación afín} $\Phi(u, v) = M(u, v)^{\mathsf T} + C$ con
$M$ invertible, el jacobiano es la matriz constante $M$:
\omterm{def:b2:multint:domain}{áreas} se multiplican por el factor constante $\abs{\det M}$
--- la promesa hecha en \cref{ch:b2:affine} es ahora un teorema.
Dos usos inmediatos. La elipse $\frac{x^2}{a^2} +
\frac{y^2}{b^2} \leq 1$ es la imagen del disco unitario bajo
$(u, v) \mapsto (au, bv)$, por lo que \omterm{def:b2:multint:domain}{área} es $ab \cdot \pi$ ---
sin cálculo. Y para la integral de $f(x + y)$ sobre el
cuadrado $K = \intcc01^2$, la cizalla $\Phi(u, v) = (u - v, v)$
(\omterm{def:b2:linalg:det}{determinante} $1$) lo convierte en una integral de $f(u)$ sobre un
paralelogramo, que Fubini corta a $u$ constante: con $f =
\exp$,
\[
\iint_K \eu^{x+y}\,\dd x\,\dd y
= \Bigl(\int_0^1 \eu^x\,\dd x\Bigr)^2 = (\eu - 1)^2,
\]
como lo confirma la estructura del producto. Elegir coordenadas
adaptado al integrando --- no al dominio --- es el
la otra mitad de la embarcación.
\end{example}


\begin{example}[Coordenadas adaptadas a un curvilíneo
dominio]\label{ex:b2:multint:hyperbolicregion}
Sea $D$ la región del primer cuadrante acotada por el
hipérbolas $xy = 1$ y $xy = 3$ y las líneas $y = x$ y
$y = 3x$. En las coordenadas $u = xy$, $v = y/x$ el dominio
se convierte en el cuadrado $\intcc13 \times \intcc13$; invertir,
\[
x = \sqrt{u/v}, \qquad y = \sqrt{uv},
\qquad
\det J = x_uy_v - x_vy_u = \frac{1}{2v}
\]
(un cálculo de dos líneas con $x = u^{1/2}v^{-1/2}$, $y =
u^{1/2}v^{1/2}$). Por lo tanto
\[
\operatorname{Area}(D)
= \int_1^3\!\!\int_1^3\frac{\dd u\,\dd v}{2v}
= 2\cdot\frac{\ln 3}{2} = \ln 3 \approx 1.10 .
\]
Intentar cortar $D$ en coordenadas cartesianas significa cortarlo
en tres piezas con límites hiperbólicos y lineales ---
factible, triste y propenso a errores. La moraleja se repite
\cref{ex:b2:multint:affine} con toda su fuerza: lea el
ecuaciones de contorno, y deje que \emph{a ellos} elija la
coordenadas; el jacobiano luego convierte el curvilíneo
celda de malla \omterm{def:b2:multint:domain}{área}, exactamente como lo hizo $\rho$ para polar
coordenadas.
\end{example}

\begin{example}[Valores medios]\label{ex:b2:multint:average}
El \emph{valor medio} de $f$ sobre un dominio $D$ es
$\frac1{\operatorname{Area}(D)}\iint_Df$. Muestra: el promedio
distancia desde el centro para un punto uniformemente elegido de la
disco de radio $R$ es
\[
\frac{1}{\pi R^2}\int_0^{2\pi}\!\!\int_0^R
\rho\cdot\rho\,\dd\rho\,\dd\alpha
= \frac{2\pi R^3/3}{\pi R^2} = \frac{2R}3 ,
\]
no $R/2$: \omterm{def:b2:multint:domain}{área} uniforme pone más masa en radios grandes (el
el anillo en el radio $\rho$ tiene un peso proporcional a $\rho$),
por lo que el promedio se sitúa más allá de la mitad del camino. Conseguir esto
El factor derecho es exactamente el jacobiano polar en acción, y el
misma ponderación explica el centroide $\bar z = 3R/8$ del
media bola se calcula más adelante en el capítulo en lugar de $R/2$.
\end{example}

\section{El teorema de Green-Riemann}

\begin{theorem}[Verde--Riemann]\label{thm:b2:multint:green}
Sea $K \subseteq \R^2$ un dominio \omterm{def:b2:metric:compact}{compacto} que es elemental en
ambas direcciones (o una unión finita de tales segmentos pegados),
con límite $\partial K$ a trozos $\mathcal{C}^1$ cerrado
orientado a curvas \emph{sinistrórsum} (el dominio permanece en la
izquierda). Para $P, Q$ de clase $\mathcal{C}^1$ en una vecindad de
$K$:
\[
\oint_{\partial K} P\,\dd x + Q\,\dd y
= \iint_K \Bigl(\frac{\partial Q}{\partial x}
- \frac{\partial P}{\partial y}\Bigr)\,\dd x\,\dd y .
\]
\end{theorem}

\begin{proof}
Primero, ambos lados son aditivos bajo el corte $K$ a lo largo de un
segmentar en dos piezas $K_1, K_2$: las integrales dobles suman
por aditividad de $\iint$; en cuanto a las integrales de frontera, la
límites en sentido antihorario de $K_1$ y $K_2$ cada recorrido
el interior cortado una vez, en direcciones \emph{opuesto}, por lo que en
la suma
\[
\oint_{\partial K_1} + \oint_{\partial K_2}
= \oint_{\partial K} + (\text{cut, both ways})
= \oint_{\partial K},
\]
las dos pasadas a lo largo del corte se cancelan
(\cref{prop:b2:multint:lineinv}) y solo el límite exterior
sobrevive. Iterando un número finito de cortes, es suficiente tratar
un dominio elemental. Probamos $\oint P\,\dd x = -\iint_K P_y$ en un
$y$-dominio elemental $D = \{a \leq x \leq b,\ \varphi_1(x) \leq y
\leq \varphi_2(x)\}$; la identidad $\oint Q\,\dd y = \iint_K Q_x$
es \omterm{def:b2:quadratic:adjoint}{simétrico} ($x$-elemental), y el teorema es su suma.

Calcular la integral doble de Fubini y la de una variable
teorema fundamental:
\[
\iint_D \frac{\partial P}{\partial y}\,\dd x\,\dd y
= \int_a^b \bigl(P(x, \varphi_2(x)) - P(x, \varphi_1(x))\bigr)
\,\dd x .
\]
Ahora el límite de $D$, en sentido antihorario, consta de: la parte inferior
gráfico $y = \varphi_1(x)$ recorrido de izquierda a derecha, la derecha
segmento vertical $x = b$ (hacia arriba), el gráfico superior $y =
\varphi_2(x)$ atravesó \emph{derecha a izquierda}, el vertical izquierdo
segmento $x = a$ (hacia abajo). A lo largo de los segmentos verticales $x$ está
constante, por lo que contribuyen $0$ a $\oint P\,\dd x$; los gráficos,
parametrizado por $x$, dé
\[
\oint_{\partial D} P\,\dd x
= \int_a^b P(x, \varphi_1(x))\,\dd x
- \int_a^b P(x, \varphi_2(x))\,\dd x
= -\iint_D \frac{\partial P}{\partial y}\,\dd x\,\dd y .
\qedhere
\]
\end{proof}

\begin{corollary}[Área por el límite]\label{cor:b2:multint:area}
Bajo las hipótesis de \cref{thm:b2:multint:green},
\[
\operatorname{Area}(K)
= \oint_{\partial K} x\,\dd y
= -\oint_{\partial K} y\,\dd x
= \frac12\oint_{\partial K} x\,\dd y - y\,\dd x .
\]
\end{corollary}

\begin{proof}
Aplique Verde--Riemann a $(P, Q) = (0, x)$, $(-y, 0)$ y
$\frac12(-y, x)$: cada vez $Q_x - P_y = 1$.
\end{proof}

\begin{remark}[Elegir entre las tres fórmulas de área]
Las tres fórmulas de límites son iguales, no intercambiables en
práctica. Utilice $\oint x\,\dd y$ cuando la parametrización haga
$\dd y$ simple (gráficos sobre el eje $y$), $-\oint y\,\dd x$
simétricamente, y la media suma \omterm{def:b2:quadratic:adjoint}{simétrico} cuando el
la parametrización trata a $x$ y $y$ de manera uniforme --- para la elipse
produjo un integrando constante, sin trigonometría
linealización en absoluto. En límites poligonales la media suma
se convierte en la fórmula de cordones de zapatos de \cref{exo:b2:multint:12},
El algoritmo de los topógrafos. Y cuando se traspasa el límite
en el sentido de las agujas del reloj según la parametrización dada, las tres fórmulas
devuelve \emph{menos} o \omterm{def:b2:multint:domain}{área}: un resultado negativo no es un
error de cálculo pero un informe de orientación --- voltee el
signo o la parametrización.
\end{remark}

\begin{example}[Área de la elipse]\label{ex:b2:multint:ellipsearea}
Para $x = a\cos t$, $y = b\sin t$, $t \in [0, 2\pi]$:
\[
\operatorname{Area}
= \frac12\int_0^{2\pi}\bigl(a\cos t \cdot b\cos t
- b\sin t\cdot(-a\sin t)\bigr)\,\dd t
= \frac{ab}{2}\int_0^{2\pi}\dd t = \pi ab .
\]
\end{example}

\begin{example}[Verde--Riemann como
verificación cruzada]\label{ex:b2:multint:greencheck}
Tome $P = -y^3$, $Q = x^3$ en el disco unitario cerrado $D$.
Lado límite, con $\gamma(t) = (\cos t, \sin t)$:
\[
\oint_{\partial D}P\,\dd x + Q\,\dd y
= \int_0^{2\pi}\bigl(\sin^4 t + \cos^4 t\bigr)\dd t
= 2\pi\cdot\Bigl(\frac38 + \frac38\Bigr) = \frac{3\pi}2 ,
\]
por linealización ($\sin^4 + \cos^4 = \tfrac34 +
\tfrac14\cos4t$). Lado interior:
\[
\iint_D(Q_x - P_y)\,\dd x\,\dd y
= \iint_D 3(x^2 + y^2)\,\dd x\,\dd y
= 3\int_0^{2\pi}\!\!\int_0^1\rho^3\,\dd\rho\,\dd\alpha
= \frac{3\pi}2 .
\]
Mismo número, dos cálculos muy diferentes --- y eso es
el uso práctico: cualquier lado de la identidad de Green es
más fácil resulta el cálculo, el otro una verificación. Para
circulaciones de campos polinomiales alrededor de curvas cerradas, la
La integral doble es casi siempre el lado fácil.
\end{example}

\begin{remark}
Green--Riemann explica \cref{ex:b2:multint:angleform}: para un
\omterm{def:b2:multint:exact}{forma cerrada} ($Q_x = P_y$), la integral alrededor del límite de cualquier
El dominio contenido en $U$ desaparece. La forma del ángulo no es \omterm{def:b2:multint:exact}{exacto}
sólo porque la punción en el origen impide que el disco se limite

\omterm{def:b2:multint:exact}{formas cerradas} detecta los agujeros del dominio. (Empujado más,
esta observación se convierte en cohomología de De Rham.)
\end{remark}

\section{Integrales triples}

La teoría se extiende a tres variables sin idea nueva: Fubini
reduce $\iiint$ a tres integrales de una variable (ya sea por
\emph{rebanar}: $\iiint_K f = \int\bigl(\iint_{K_z}
f\bigr)\dd z $sobre los cortes horizontales $K_z$, o por
\emph{apilado}: integrando en $z$ primero a lo largo de palos verticales),
y la formula de cambio de variables se cumple con el $3 \times 3$
Jacobiano.

\begin{example}[Coordenadas cilíndricas y esféricas]\label{ex:b2:multint:spherical}
\emph{Cilíndrico} $(x, y, z) = (\rho\cos\alpha, \rho\sin\alpha,
z)$: $\dd x\,\dd y\,\dd z = \rho\,\dd\rho\,\dd\alpha\,\dd z$.
\emph{Esférico} $(x, y, z) = (r\cos\theta\cos\varphi,\
r\sin\theta\cos\varphi,\ r\sin\varphi)$($\theta$ longitud,
Latitud $\varphi \in [-\frac\pi2, \frac\pi2]$): expandiendo $3
\times 3$ \omterm{def:b2:linalg:det}{determinante} a lo largo de la última fila,
\[
\det J = r^2\cos\varphi ,
\qquad
\dd x\,\dd y\,\dd z
= r^2\cos\varphi\;\dd r\,\dd\theta\,\dd\varphi .
\]
Volumen de la bola de radio $R$:
\[
V = \int_0^R\!\!\int_0^{2\pi}\!\!\int_{-\pi/2}^{\pi/2}
r^2\cos\varphi\;\dd\varphi\,\dd\theta\,\dd r
= \frac{R^3}{3}\cdot 2\pi \cdot 2
= \boxed{\frac43\pi R^3} ,
\]
descargando por fin la fórmula admitida en los capítulos del volumen de
los libros anteriores.
\end{example}

\begin{example}[El tetraedro, dos veces]
\label{ex:b2:multint:tetra}
El volumen de $T = \{x, y, z \geq 0,\ x + y + z \leq 1\}$, por
apilamiento: para $(x, y)$ fijo en el triángulo $x + y \leq 1$,
$z$ pasa por encima de $\intcc0{1 - x - y}$, por lo que
\[
V = \int_0^1\!\!\int_0^{1-x}(1 - x - y)\,\dd y\,\dd x
= \int_0^1\frac{(1 - x)^2}{2}\,\dd x = \frac16 .
\]
Por corte: la sección a la altura $z$ es el triángulo $\{x, y
\geq 0,\ x + y \leq 1 - z\}$, de \omterm{def:b2:multint:domain}{área}$\frac{(1-z)^2}2$, y
$V = \int_0^1\frac{(1-z)^2}2\,\dd z = \frac16$ nuevamente --- el
dos cálculos son las mismas integrales en una forma diferente
orden, que es todo lo que afirma Fubini. El valor $\frac16 =
\frac13\cdot\frac12\cdot1$ es la fórmula del cono.
(\cref{ex:b2:multint:cone}) con base triangular, y el
$n$ versión tridimensional $1/n!$ se demuestra exactamente con esto
Rebanar el problema del fin de semana.
\end{example}

\begin{example}[Centroide de media bola]\label{ex:b2:multint:centroid}
Para la media bola superior $H$ de radio $R$ ($z \geq 0$), la
La altura del centroide es $\bar z = \frac1{V}\iiint_H z$, con $V =
\frac23\pi R^3$. En coordenadas esféricas ($z =
r\sin\varphi$, $\varphi \in \intcc0{\pi/2}$):
\[
\iiint_H z
= \int_0^R r^3\,\dd r\int_0^{2\pi}\dd\theta
\int_0^{\pi/2}\sin\varphi\cos\varphi\,\dd\varphi
= \frac{R^4}4\cdot2\pi\cdot\frac12 = \frac{\pi R^4}4 ,
\]
entonces
\[
\bar z = \frac{\pi R^4/4}{2\pi R^3/3} = \frac{3R}8 :
\]
El punto de equilibrio de un hemisferio sólido se encuentra a tres octavos de
el radio por encima de la cara plana --- por debajo de la media altura
$R/2$, como debe ser, ya que el sólido es más gordo cerca de la base.
Cada cálculo de centroide tiene esta forma: un momento
Integral, un volumen, una relación y una verificación de plausibilidad.
contra la geometría.
\end{example}

\begin{example}[Volumen por corte: el cono]\label{ex:b2:multint:cone}
Un cono de base \omterm{def:b2:multint:domain}{área} $A$ y altura $h$ (ápice hacia arriba, base en $z = 0$):
el corte en altura $z$ es la base escalada por el factor $(1 -
z/h)$, de \omterm{def:b2:multint:domain}{área}$A(1 - z/h)^2$. Por lo tanto
\[
V = \int_0^h A\Bigl(1 - \frac zh\Bigr)^2\dd z = \frac{Ah}{3} :
\]
el tercio de las fórmulas escolares, válidas para la base \emph{cualquier}
forma --- cortar la convierte en la integral de un cuadrado.
\end{example}

\begin{example}[Umbrales de integrabilidad en el
avión]\label{ex:b2:multint:threshold}
¿Por qué $\alpha > 0$ converge $\iint_{D}\rho^{-\alpha}\,\dd
x\,\dd y $en el disco unitario perforado $D$ (límite
sobre los anillos $\varepsilon \leq \rho \leq 1$)? en polares
coordenadas,
\[
\int_0^{2\pi}\!\!\int_\varepsilon^1\rho^{-\alpha}\,
\rho\,\dd\rho\,\dd\alpha
= 2\pi\int_\varepsilon^1\rho^{1-\alpha}\,\dd\rho ,
\]
que converge como $\varepsilon \to 0$ si $1 - \alpha > -1$,
es decir\ $\alpha < 2$: en la dimensión $2$ la singularidad crítica
exponente es la dimensión misma, el $\rho$ extra del
Jacobiano suavizando la singularidad por un poder. (Asimismo
$\alpha < 3$ para una singularidad puntual en el espacio, vía $r^2$.)
Una contabilidad radial de este tipo es la forma de integración
decidido de un vistazo en el marco de Lebesgue del Año 3 ---
y es la razón por la que $\iiint 1/r$ convergió sin esfuerzo en
\cref{exo:b2:multint:7}.
\end{example}

\begin{remark}[Errores comunes]
(i) \emph{Orientación}: un \omterm{def:b2:multint:lineint}{integral de línea} cambia de signo con el
dirección de viaje, y Green-Riemann requiere que el límite
en sentido antihorario (dominio de la izquierda); para un dominio con un
agujero, se atraviesa el límite interior \emph{dextrorso}. (ii)
\emph{El jacobiano entra con valor absoluto}: cambio de
variables nunca produce \omterm{def:b2:multint:domain}{área} negativo, y olvidar
$\abs{\det}$ normalmente voltea las señales exactamente cuando el mapa
invierte la orientación. (iii) \emph{The polar factor $\rho$}:
$\dd x\,\dd y = \rho\,\dd\rho\,\dd\alpha$, no
$\dd\rho\,\dd\alpha$ --- el error más común en todo
capítulo; El análisis dimensional lo capta, ya que
$\dd\rho\,\dd\alpha$ tiene la dimensión de un \omterm{def:b2:curves:length}{longitud}, no de un
\omterm{def:b2:multint:domain}{área}. (iv) \emph{Integrales dobles impropias}: límites superiores
Los discos crecientes y los cuadrados crecientes concuerdan aquí porque el
los integrandos son positivos (apretando); para cambiar de signo
integrandos el límite puede depender del agotamiento, y no
La afirmación se hace sin una convergencia absoluta. (v)
\emph{Dominios versus integrandos}: un integrando de producto en un
dominio que no es producto, \emph{no} factoriza la integral ---
la factorización necesita ambos, como en \cref{ex:b2:multint:polar}
cuadrado.
\end{remark}

\begin{remark}[Perspectivas dentro de este volumen]
La integral de Gauss calculada aquí está silenciosamente en todas partes
los capítulos de probabilidad: la constante $\sqrt\pi$ dentro
La fórmula de Stirling (\cref{thm:b2:comparison:stirling}) es
la integral de este capítulo y, a través de Stirling, fija la
$1/\sqrt{\pi n}$ asintóticas del retorno del paseo aleatorio
probabilidades en \cref{ch:b2:proba}. El \omterm{pb:b2:multint:1}{Integrales de Wallis} de
El problema del fin de semana vuelve a aparecer ahí también, conduciendo igual.
estimaciones binomiales centrales. En la otra dirección, el \omterm{def:b2:multint:domain}{área}
y los elementos de volumen de este capítulo completan la geometría de
\cref{ch:b2:surfaces}, y la fórmula de Green vuelve a calcular el
\omterm{pb:b2:curves:1}{sobre} \omterm{def:b2:multint:domain}{áreas} de \cref{ch:b2:curves} (el \omterm{pb:b2:curves:1}{astroide}, en
\cref{exo:b2:multint:5}). Un capítulo, tres servicios:
medida para la geometría, constantes para la probabilidad y la
disciplina de cambio de variables utilizada por ambos.
\end{remark}

\section{Problema: el volumen de la pelota en la dimensión $n$}

\begin{problem}[{Problema del fin de semana --- $V_n =
\pi^{n/2}/\Gamma(\frac n2 + 1)$, y la extrañeza de la alta
dimensiones}]\label{pb:b2:multint:1}
El disco tiene \omterm{def:b2:multint:domain}{área} $\pi$, \omterm{pb:b2:multint:1}{volumen de la bola} $\frac43\pi$ --- y
entonces? Este problema calcula el volumen de la bola unitaria de
$\R^n$ por cada $n$, dos veces (mediante una recursividad de corte impulsada por
el \omterm{pb:b2:multint:1}{Integrales de Wallis}, luego a través de la función $\Gamma$ y el
Integral de Gauss de \cref{ex:b2:multint:polar}), y luego lee
fuera de la geometría: los volúmenes alcanzan su punto máximo en la dimensión cinco y se precipitan
a cero, y casi toda una bola de alta dimensión se esconde en un
capa delgada cerca de su límite.\index{volumen de la bola}\index{wallis
integrales} Para una función \omterm{def:b2:metric:continuity}{continuo} en una bola de $\R^n$, la
integral se entiende como la integral iterada $n$ veces
(cortando una coordenada a la vez, como en el capítulo $n
\leq 3$); escribimos $B_n(R)$para la bola cerrada de radio $R$
centrado en $0$, $v_n(R)$ para su volumen, y $V_n =
v_n(1)$, con $V_0 = 1$ por convención.

\medskip
\noindent\textbf{Parte I --- The slicing recursion.}
\begin{enumerate}
 \item Sustituyendo $x_i = Ru_i$ en cada uno de los $n$ iterados
 integrales, muestre $v_n(R) = V_nR^n$.
 \item Cortando $B_n(1)$ a lo largo de su última coordenada, muestra
 \[
 V_n = V_{n-1}\int_{-1}^{1}(1 -
 t^2)^{\frac{n-1}2}\,\dd t .
 \]
 \item Con $t = \sin\theta$, identifique la integral como
 Integral de Wallis: $\int_{-1}^1(1 -
 t^2)^{\frac{n-1}2}\dd t = 2W_n$, donde $W_n =
 \int_0^{\pi/2}\cos^n\theta\,\dd\theta =
 \int_0^{\pi/2}\sin^n\theta\,\dd\theta$.
 \item Demostrar las dos identidades de Wallis (integrar por partes;
 luego telescopio $nW_nW_{n-1}$):
 \[
 W_n = \frac{n-1}nW_{n-2} \quad (n \geq 2),
 \qquad
 W_nW_{n-1} = \frac{\pi}{2n} \quad (n \geq 1).
 \]
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The recursion solved.}
\begin{enumerate}[resume]
 \item Combine las preguntas 2 a 4 en la recursividad de dos pasos
 \[
 V_n = \frac{2\pi}{n}\,V_{n-2}
 \qquad (n \geq 2).
 \]
 \item Deduce el \omterm{def:b2:multint:exact}{formas cerradas}, para $k \geq 0$:
 \[
 V_{2k} = \frac{\pi^k}{k!},
 \qquad
 V_{2k+1} = \frac{2^{k+1}\pi^k}{1\cdot3\cdot5\cdots
 (2k+1)} .
 \]
 \item Tabule $V_1, \dots, V_7$ numéricamente. Usando el
 relación $V_n/V_{n-2} = 2\pi/n$ y los valores de $2W_5$
 y $2W_6$, demuestre que la secuencia $(V_n)$ aumenta
 hasta su máximo $V_5 = \frac{8\pi^2}{15} \approx
 5.26$ y disminuye a partir de ahí.
 \item Demuestre que $V_n \to 0$ es más rápido que cualquier patrón geométrico.
 secuencia, y que $\sum_{n\geq1} V_n$ converge: todos
 las bolas unitarias juntas tienen un volumen total finito.
 \item Probar la identidad generadora
 \[
 \sum_{k\geq0} V_{2k}\,x^{2k} = \eu^{\pi x^2}
 \qquad (x \in \R),
 \]
 y deducir $\sum_{k\geq0}V_{2k} = \eu^\pi \approx
 23.14$.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Second route: $\Gamma$ and the
Gauss integral.}
\begin{enumerate}[resume]
 \item Muestra por Fubini (el integrando es un producto) que
 \[
 I_n = \int_{\R^n}\eu^{-\norm
 x^2}\dd x
 = \Bigl(\int_{-\infty}^{+\infty}
 \eu^{-t^2}\dd t\Bigr)^{\!n} = \pi^{n/2},
 \]
 la integral de Gauss $n$-dimensional, entendida como una
 límite sobre cubos $\intcc{-R}{R}^n$.
 \item Recuperar $\Gamma(s) = \int_0^\infty t^{s-1}\eu^{-t}\dd
 t$ (\cref{def:b2:integration:gamma}). De
 $\Gamma(s+1) = s\,\Gamma(s)$
 (\cref{thm:b2:integration:gammaprops}) y
 $\Gamma(\tfrac12) = \sqrt\pi$ (sustituto $t = u^2$
 e invocar la integral de Gauss), calcular
 \[
 \Gamma(k + 1) = k!,
 \qquad
 \Gamma\Bigl(k + \frac32\Bigr) =
 \frac{1\cdot3\cdots(2k+1)}{2^{k+1}}\,\sqrt\pi .
 \]
 \item Demostrar, por inducción mediante la recursividad de la pregunta
 5, la fórmula única
 \[
 V_n = \frac{\pi^{n/2}}{\Gamma\bigl(\frac n2 +
 1\bigr)} \qquad (n \geq 1),
 \]
 y verifique que reproduzca tanto \omterm{def:b2:multint:exact}{formas cerradas} de la pregunta
 6.
 \item Muestra $\int_0^\infty \eu^{-r^2}r^{n-1}\dd r =
 \tfrac12\Gamma\bigl(\tfrac n2\bigr)$ y deduce el
 identidad
 \[
 I_n = n\,V_n\int_0^\infty \eu^{-r^2}\,r^{n-1}\,\dd r.
 \]
 Interpretarlo: la masa gaussiana de $\R^n$ es
 recogidos a lo largo de conchas esféricas cuyo
 ``$(n-1)$-dimensional \omterm{def:b2:multint:domain}{área}'' en el radio $r$ es
 $nV_nr^{n-1}$ --- ambas partes ahora están probadas
 de forma independiente, por lo que la interpretación no cuesta nada.
 \item Establezca $s_{n-1} = nV_n$ (el \omterm{def:b2:multint:domain}{área} de la unidad de esfera
 $S^{n-1}$, consistente con $v_n(R) = \int_0^R
 s_{n-1}r^{n-1}\dd r$). Tabular $s_0, \dots, s_3$ y
 marque $s_1 = 2\pi$, $s_2 = 4\pi$, $s_3 = 2\pi^2$.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- High dimensions are strange.}
\begin{enumerate}[resume]
 \item De la fórmula de Stirling
 (\cref{thm:b2:comparison:stirling}) aplicado a $k!$,
 mostrar incluso $n = 2k$:
 \[
 V_n \sim \frac{1}{\sqrt{\pi n}}
 \Bigl(\frac{2\pi\eu}{n}\Bigr)^{n/2}
 \qquad (n \to \infty, \ n \text{ even}),
 \]
 y explicar por qué la misma desintegración supergeométrica se limita
 se extiende al impar $n$ mediante la recursividad.
 \item La bola unitaria se encuentra en el cubo $\intcc{-1}1^n$ de
 volumen $2^n$. Calcule la relación de llenado $V_n/2^n$ para
 $n = 2, 3, 10$, y muestra que tiende a $0$: en alto
 dimensión, esencialmente todo el cubo se encuentra en su
 esquinas.
 \item Demuestre que la fracción de $v_n(1)$ que se encuentra dentro
 la distancia $\varepsilon$ de la esfera límite es $1 -
 (1 - \varepsilon)^n \to 1$; numéricamente, ¿qué
 fracción de una bola de dimensiones $100$ se encuentra en el
 capa exterior de espesor $1\%$?
 \item Demuestre el $W_n \sim
 \sqrt{\dfrac{\pi}{2n}}$ \emph{(monotonicity of
 $(W_n)$, the ratio $W_n/W_{n-2} \to 1$, and $W_nW_{n-1}
 = \frac\pi{2n}$)} asintótico de Wallis y el límite inferior $W_n \geq
 \sqrt{\dfrac{\pi}{2(n+1)}}$ para todos los $n$.
 \item (Concentración en una losa) La fracción de la unidad
 bola con la primera coordenada más allá de $\delta$ es
 $\int_\delta^1(1 - x^2)^{\frac{n-1}2}\dd x \,\big/\,
 (2W_n)$. Usando $1 - u \leq \eu^{-u}$ y la cola
 atado $\int_\delta^\infty \eu^{-a x^2}\dd x \leq
 \frac{\eu^{-a\delta^2}}{2a\delta}$, muestra esto
 la fracción es como máximo
 \[
 \frac{\eu^{-(n-1)\delta^2/2}}{(n-1)\,\delta}
 \Big/ \sqrt{\frac{2\pi}{n+1}}
 \]
 y concluimos: para $\delta = s/\sqrt{n-1}$, todos menos un
 la fracción $O(\eu^{-s^2/2}/s)$ de la pelota se encuentra en el
 losa $\abs{x_1} \leq s/\sqrt{n-1}$. una bola de alto
 dimensión es, estadísticamente, un panqueque delgado en cada
 dirección a la vez.
 \item Reúna las preguntas 16 a 19 en un párrafo: dónde
 el volumen de $B_n(1)$ se encuentra (cerca del límite
 esfera, pero dentro de $O(1/\sqrt n)$ losas de cada
 hiperplano que pasa por el centro), y por qué estos dos
 las declaraciones no se contradicen entre sí.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Other bodies, and synthesis.}
\begin{enumerate}[resume]
 \item (Simplex) Deje $\Delta_n = \{x \in \R^n : x_i \geq 0,\
 \sum x_i \leq 1\}$. Demuestre por corte e inducción que
 $\operatorname{vol}(\Delta_n) = \frac1{n!}$.
 \item (Politopo cruzado) Deduzca que $C_n = \{x :
 \sum\abs{x_i} \leq 1\}$ tiene volumen $\frac{2^n}{n!}$,
 y verificar el sándwich $C_n \subseteq B_n(1)
 \subseteq \intcc{-1}1^n$ a nivel de volúmenes:
 $\frac{2^n}{n!} \leq V_n \leq 2^n$.
 \item Calcule $V_4$ de una tercera manera: corte $\R^4 = \R^2 \times
 \R^2$, integre el \omterm{def:b2:multint:domain}{área} del disco $(z, w)$
 el disco $(x, y)$ en coordenadas polares y recuperar
 $V_4 = \frac{\pi^2}2$.
 \item (Monte Carlo en problemas) Se sortea un punto uniformemente
 en el cubo $\intcc{-1}1^{20}$. Demuestre que el
 la probabilidad de que caiga en la bola inscrita es
 $V_{20}/2^{20} \approx 2.5\cdot10^{-8}$, más o menos
 Se necesitan cuarenta millones de sorteos antes del primer golpe.
 se espera: estimación de $V_n$ mediante muestreo de rechazo
 colapsa en alta dimensión (la maldición de
 dimensionalidad).
 \item Síntesis. Dos derivaciones independientes se encontraron en $V_n =
 \pi^{n/2}/\Gamma(\frac n2 + 1)$: enumera qué teorema
 de este capítulo cada uno utilizado (Fubini, cambio de
 variables, la integral polar de Gauss), y que
 entradas de una variable (Wallis, $\Gamma$, Stirling).
 ¿Dónde rehace este cálculo el volumen del año 3?
 con la teoría de Lebesgue, y ¿qué aporta?
\end{enumerate}
\end{problem}
