\section{Espacios de probabilidad}

\begin{definition}[Espacio de probabilidad contable]\label{def:b2:proba:space}
Sea $\Omega$ un finito no vacío o \omterm{def:b2:structures:countable}{conjunto contable} (el
\emph{espacio muestral}\index{espacio muestral}). Un \emph{probabilidad
medida}\index{medida de probabilidad} en $\Omega$ es
un mapa $\P$ del conjunto $\mathcal{P}(\Omega)$ de todos los subconjuntos de
$\Omega$ (\emph{eventos}\index{evento}) a $[0, 1]$ tal que:
\begin{enumerate}
\item $\P(\Omega) = 1$;
\item ($\sigma$-aditividad) para cada secuencia $(A_n)_{n\in\N}$ de
eventos disjuntos por pares,
\[
\P\Bigl(\,\bigcup_{n \in \N} A_n\Bigr)
= \sum_{n=0}^{\infty} \P(A_n) .
\]
\end{enumerate}
El par $(\Omega, \P)$ es un (\omterm{def:b2:structures:countable}{contable}) \emph{probabilidad
espacio}\index{espacio de probabilidad}.
\end{definition}

\begin{proposition}[reglas elementales]\label{prop:b2:proba:rules}
Para \omterm{def:b2:proba:space}{eventos} $A, B$ y \omterm{def:b2:proba:space}{medida de probabilidad} $\P$:
$\P(\emptyset) = 0$; $\P$ es finitamente aditivo; $\P(A^c) = 1 -
\P(A)$; si $A \subseteq B$entonces $\P(A) \leq \P(B)$; y
\[
\P(A \cup B) = \P(A) + \P(B) - \P(A \cap B) .
\]
\end{proposition}

\begin{proof}
Aplicando la aditividad $\sigma$ a $A_0 = \Omega$, $A_n = \emptyset$
($n \geq 1$) da $1 = 1 + \sum_{n\geq1}\P(\emptyset)$, entonces
$\P(\emptyset) = 0$; rellenar una unión finita disjunta con vacío
conjuntos entonces da aditividad finita. El resto sigue como en el
caso finito (volumen de escuela secundaria): $1 = \P(A) + \P(A^c)$ de
$\Omega = A \sqcup A^c$; $\P(B) = \P(A) + \P(B \setminus A)
\geq \P(A)$cuando $A \subseteq B$; y descomponerse en tres
piezas separadas,
\begin{align*}
\P(A \cup B) &= \P(A \setminus B) + \P(B \setminus A) +
\P(A \cap B)\\
&= \bigl(\P(A) - \P(A\cap B)\bigr) + \bigl(\P(B) - \P(A\cap
B)\bigr) + \P(A \cap B),
\end{align*}
que es inclusión-exclusión; la versión general del conjunto $n$ es
\cref{exo:b2:proba:4}.
\end{proof}

\begin{proposition}[Distribuciones en un contable
espacio]\label{prop:b2:proba:distribution}
Dar un \omterm{def:b2:proba:space}{medida de probabilidad} a un \omterm{def:b2:structures:countable}{contable} $\Omega =
\{\omega_0, \omega_1, \dots\}$ equivale exactamente a dar pesos
$p_i = \P(\{\omega_i\}) \geq 0$ con $\sum_i p_i = 1$; entonces para
cada $A \subseteq \Omega$,
\[
\P(A) = \sum_{\omega \in A} \P(\{\omega\}) ,
\]
una subsuma (\omterm{def:b2:series:def}{absolutamente} convergente) de la familia $(p_i)$.
\end{proposition}

\begin{proof}
Dado $\P$, los singletons $\{\omega\}$, $\omega \in A$, forman un
\omterm{def:b2:structures:countable}{contable} cubierta separada de $A$, por lo que $\sigma$-fuerzas de aditividad
\[
\P(A) = \sum_{\omega\in A}\P(\{\omega\}),
\]
una subsuma incondicional del \omterm{def:b2:series:summable}{familia sumable} no negativo
$(p_i)$ --- reordenar es inofensivo precisamente porque el
los términos no son negativos (\cref{ch:b2:series}); en particular
$\sum_ip_i = \P(\Omega) = 1$.
Por el contrario, dados los pesos no negativos de la suma total $1$, defina
$\P(A) = \sum_{\omega \in A}p_\omega$: la familia es \omterm{def:b2:series:summable}{sumable}, y
$\sigma$-la aditividad es exactamente el teorema de la suma por paquetes
de \cref{ch:b2:series} aplicado a la partición de $\bigcup A_n$
en el $A_n$.
\end{proof}

\begin{example}[Modelo geométrico: esperando el primero

Lanza una moneda con probabilidad de cara $p \in \intoo{0}{1}$ repetidamente,

\]

\begin{theorem}[Continuidad monótona]\label{thm:b2:proba:continuity}
Sea $(A_n)$ una secuencia de \omterm{def:b2:proba:space}{eventos}.
\begin{enumerate}
\item Si $A_n \subseteq A_{n+1}$ para todos $n$
(\emph{creciente}), entonces
$\P\bigl(\bigcup_n A_n\bigr) = \lim_{n\to\infty} \P(A_n)$.
\item Si $A_n \supseteq A_{n+1}$ para todos $n$
(\emph{decreciente}), entonces
$\P\bigl(\bigcap_n A_n\bigr) = \lim_{n\to\infty} \P(A_n)$.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Desunir: sean $B_0 = A_0$ y $B_n = A_n \setminus
A_{n-1}$. Los $B_n $están separados por pares de $\bigcup_{k \leq n}
B_k = A_n $y$\bigcup_n B_n = \bigcup_n A_n$. Por
$\sigma$-aditividad y aditividad finita,
\[
\P\Bigl(\bigcup_n A_n\Bigr)
= \sum_{n=0}^\infty \P(B_n)
= \lim_{N\to\infty}\sum_{n=0}^N \P(B_n)
= \lim_{N\to\infty}\P(A_N) .
\]
\emph{2.} Pase a complementos: $(A_n^c)$ es creciente con unión
$\bigl(\bigcap A_n\bigr)^c$ y aplique la parte 1:
$1 - \P(\bigcap A_n) = \lim (1 - \P(A_n))$.
\end{proof}

\begin{corollary}[Subaditividad contable]\label{cor:b2:proba:subadd}
Para cualquier secuencia de \omterm{def:b2:proba:space}{eventos}, $\P\bigl(\bigcup_n A_n\bigr) \leq
\sum_{n=0}^\infty \P(A_n)$.
\end{corollary}

\begin{proof}
La subaditividad finita $\P(A_0 \cup \dots \cup A_N) \leq
\sum_0^N \P(A_n)$ se deriva de la inclusión-exclusión por inducción
(o de aditividad sobre el $B_n \subseteq A_n$ desarticulado).
Sea $N \to \infty$: el lado izquierdo converge a
$\P(\bigcup_n A_n)$ por el monótono \omterm{def:b2:metric:continuity}{continuidad} aplicado al
secuencia creciente $C_N = A_0 \cup \dots \cup A_N$.
\end{proof}

La subaditividad con un número finito de \omterm{def:b2:proba:space}{eventos} --- el \emph{unión

\begin{example}[Al final llega un seis]\label{ex:b2:proba:sixeventually}
Tira un dado justo para siempre y deja que $B_n = {}$``al menos un seis

\begin{theorem}[Probabilidades compuestas, probabilidad total,
Bayes]\label{thm:b2:proba:bayes}
\leavevmode
\begin{enumerate}
\item (Regla de cadena) Si $\P(A_1 \cap \dots \cap A_{n-1}) > 0$,
\[
\P(A_1 \cap \dots \cap A_n)
= \P(A_1)\,\P(A_2 \mid A_1)\cdots
\P(A_n \mid A_1 \cap \dots \cap A_{n-1}) .
\]
\item (Probabilidad total) Si $(B_i)_{i \in I}$ es finito o
\omterm{def:b2:structures:countable}{contable} partición de $\Omega$ con $\P(B_i) > 0$, luego para cada
\omterm{def:b2:proba:space}{evento}$A$:
\[
\P(A) = \sum_{i \in I} \P(A \mid B_i)\,\P(B_i) .
\]
\item (Bayes) Bajo las mismas hipótesis, si además $\P(A) > 0$:
\[
\P(B_j \mid A)
= \frac{\P(A \mid B_j)\,\P(B_j)}
 {\sum_{i \in I} \P(A \mid B_i)\,\P(B_i)} .
\]
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Escribe cada \omterm{def:b2:proba:conditional}{probabilidad condicional} como cociente:
el lado derecho es
\[
\P(A_1)\cdot\frac{\P(A_1 \cap A_2)}{\P(A_1)}\cdot
\frac{\P(A_1 \cap A_2 \cap A_3)}{\P(A_1 \cap A_2)}\cdots
\frac{\P(A_1 \cap \dots \cap A_n)}{\P(A_1 \cap \dots \cap
A_{n-1})},
\]
un producto telescópico: cada denominador cancela el
numerador anterior, dejando $\P(A_1 \cap \dots \cap A_n)$.
Todos los denominadores son $\geq \P(A_1 \cap \dots \cap A_{n-1}) >
0$ por monotonía, por lo que nada desaparece. (La hipótesis
protege exactamente esto: condicionado a un \omterm{def:b2:proba:space}{evento} de probabilidad
cero no está definido.)
\emph{2.} Los conjuntos $A \cap B_i$ son disjuntos por pares con unión
$A$; aplicar ($\sigma$-)la aditividad y la definición de
acondicionamiento.
\emph{3.} Ambos lados de $\P(B_j \mid A)\P(A) = \P(A \mid
B_j)\P(B_j)$son iguales a $\P(A \cap B_j)$; dividir por $\P(A)$ y expandir
$\P(A)$ por probabilidad total.
\end{proof}

Tres confusiones recurrentes, todas visibles en los ejemplos.
arriba. (i) \emph{inversión}: $\pcond BA$ y $\pcond AB$
difieren por el factor $\P(A)/\P(B)$ --- una prueba que es
$99\%$ preciso sobre los enfermos aún puede dejar un positivo
Es casi seguro que el paciente está sano cuando la enfermedad es rara.
(\cref{exo:b2:proba:3}); citar $\pcond{\text{sick}}{
\text{positive}}$ donde se refiere a $\pcond{\text{positive}}{
\text{sick}}$ es la falacia de la tasa base. (ii)
\emph{Conditioning on the wrong \omterm{def:b2:proba:space}{evento}}: en Monty Hall, el
El acondicionamiento correcto \omterm{def:b2:proba:space}{evento} es "el host abrió la puerta $3$",
no ``el premio no está detrás de la puerta $3$''; los dos llevan
información diferente, y todo el $\frac23$ cuelga en el
diferencia. (iii) \emph{Disjoint versus \omterm{def:b2:proba:independence}{independiente}}:
\omterm{def:b2:proba:space}{eventos} disjuntos de probabilidad positiva nunca son
\omterm{def:b2:proba:independence}{independiente} ($\P(A\cap B) = 0 \neq \P(A)\P(B)$) ---
\omterm{def:b2:proba:independence}{independencia} es compatibilidad de información, no ausencia de
superposición.
\end{remark}

\begin{definition}[Independencia]\label{def:b2:proba:independence}
\omterm{def:b2:proba:space}{Eventos} $A$ y $B$ son \emph{independiente}\index{eventos independientes} si $\P(A \cap B) =
\P(A)\P(B)$. Una familia $(A_i)_{i \in I}$ de \omterm{def:b2:proba:space}{eventos} es
\emph{(mutuamente) independientes} si para cada subconjunto finito $J
\subseteq I$,
\[
\P\Bigl(\bigcap_{i \in J} A_i\Bigr)
= \prod_{i \in J} \P(A_i) .
\]
\end{definition}

\begin{remark}
Mutual \omterm{def:b2:proba:independence}{independencia} es estrictamente más fuerte que por pares
\omterm{def:b2:proba:independence}{independencia}: con dos lanzamientos de moneda justos, el \omterm{def:b2:proba:space}{eventos} ``primero es un
cabeza'', ``la segunda es una cabeza'', ``ambos de acuerdo'' son pares
\omterm{def:b2:proba:independence}{independiente} (cada par tiene probabilidad de intersección $\frac14 =
\frac12\cdot\frac12$), pero la intersección triple tiene probabilidad
$\frac14 \neq \frac18$. Tenga en cuenta también que si $A, B$ es \omterm{def:b2:proba:independence}{independiente},
también lo son $A, B^c$ (calcular: $\P(A \cap B^c) = \P(A) - \P(A\cap B) =
\P(A)(1 - \P(B))$), por lo tanto también $A^c, B^c$.
\end{remark}

\section{El lema de Borel-Cantelli}

\begin{definition}[Límite superior de eventos]\label{def:b2:proba:limsup}
Para una secuencia $(A_n)$ de \omterm{def:b2:proba:space}{eventos}, el \omterm{def:b2:proba:space}{evento}
\[
\limsup_n A_n
= \bigcap_{N=0}^{\infty}\ \bigcup_{n \geq N} A_n
= \{\omega \in \Omega : \omega \in A_n
\text{ for infinitely many } n\}
\]
es el \omterm{def:b2:proba:space}{evento} ``$A_n$ ocurre infinitamente con frecuencia''.
\end{definition}

\begin{example}[Traduciendo ``infinitamente frecuente'' y
``eventualmente'']\label{ex:b2:proba:limsuplang}
El complemento de $\limsup_nA_n$ es, por de Morgan,
\[
\Bigl(\bigcap_N\bigcup_{n\geq N}A_n\Bigr)^{\!c}
= \bigcup_N\bigcap_{n\geq N}A_n^c
= \{\omega : \omega \notin A_n \text{ for all large }n\},
\]
el \omterm{def:b2:proba:space}{evento} ``\emph{eventualmente}, $A_n$ falla'' (escrito
$\liminf_nA_n^c$). Entonces ``$A_n$ infinitamente frecuente'' y
``$A_n^c$ eventualmente'' son complementarios --- manteniendo esto
El diccionario directo previene la mayoría de los accidentes con cuantificadores.
Ejemplos de traducciones para lanzar una moneda al aire: ``infinitamente muchos
cabezas'' es $\limsup\{X_n = H\}$; ``sólo un número finito de ejecuciones
de $100$ cabezas'' es el complemento de un limsup; ``el
La frecuencia de funcionamiento converge a $\frac12$'' es
$\bigcap_j\bigcup_N\bigcap_{n\geq N}\{\abs{\widehat p_n -
\tfrac12} < \tfrac1j\}$ --- \omterm{def:b2:structures:countable}{contable} operaciones en todo momento,
entonces todos estos son honestos \omterm{def:b2:proba:space}{eventos}.
\end{example}

\begin{theorem}[Borel--Cantelli]\label{thm:b2:proba:borelcantelli}
\leavevmode
\begin{enumerate}
\item Si $\sum_{n} \P(A_n) < \infty$, entonces
$\P\bigl(\limsup_n A_n\bigr) = 0$.
\item Si \omterm{def:b2:proba:space}{eventos} $A_n$ son \omterm{def:b2:proba:independence}{independiente} y $\sum_n \P(A_n) =
\infty$, entonces $\P\bigl(\limsup_n A_n\bigr) = 1$.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Dejar $C_N = \bigcup_{n \geq N}A_n$; la secuencia $(C_N)$
está disminuyendo con la intersección $\limsup A_n$ y por \omterm{def:b2:structures:countable}{contable}
subaditividad (\cref{cor:b2:proba:subadd})
\[
\P(C_N) \leq \sum_{n \geq N}\P(A_n)
\xrightarrow[N\to\infty]{} 0
\]
(cola de una serie convergente). Monótono \omterm{def:b2:metric:continuity}{continuidad}
(\cref{thm:b2:proba:continuity}) concluye: $\P(\limsup A_n) =
\lim_N \P(C_N) = 0$.

\emph{2.} Basta mostrar $\P\bigl(\bigcup_{n\geq N}A_n\bigr)
= 1$ por cada $N$: efectivamente, si \omterm{def:b2:proba:space}{eventos}$B_N$ todos tienen
probabilidad $1$, entonces
\[
\P\Bigl(\Bigl(\bigcap_NB_N\Bigr)^{\!c}\Bigr)
= \P\Bigl(\bigcup_NB_N^c\Bigr)
\leq \sum_N\P(B_N^c) = 0
\]
por \omterm{def:b2:structures:countable}{contable} subaditividad
(\cref{cor:b2:proba:subadd}), por lo que la intersección \omterm{def:b2:structures:countable}{contable}
$\limsup A_n = \bigcap_N\bigcup_{n\geq N}A_n$ todavía tiene
probabilidad $1$. Repare $N$ y considere para $M > N$ el
complemento:
\[
\P\Bigl(\bigcap_{n=N}^{M} A_n^c\Bigr)
= \prod_{n=N}^{M}\bigl(1 - \P(A_n)\bigr)
\leq \prod_{n=N}^{M} e^{-\P(A_n)}
= \exp\Bigl(-\sum_{n=N}^M \P(A_n)\Bigr) ,
\]
usando \omterm{def:b2:proba:independence}{independencia} de los complementos y la convexidad ligada a $1 -
x \leq e^{-x}$. Como $M \to \infty $el exponente tiende a$-\infty$
por divergencia de la serie, por lo monótono \omterm{def:b2:metric:continuity}{continuidad}
(secuencia decreciente) $\P\bigl(\bigcap_{n \geq N}A_n^c\bigr) = 0$,
es decir,\ $\P\bigl(\bigcup_{n \geq N}A_n\bigr) = 1$.
\end{proof}

 deducir que el número de caminos de $0$ a $k$ que
 permanezca $> 0$ después de que el tiempo $0$ sea $N_{n-1}(k-1) -
 N_{n-1}(k+1)$.
 \item (Teorema de la papeleta) Deduce que
 \[
 \P\bigl(S_1 > 0, \dots, S_{n-1} > 0 \bigm| S_n =
 k\bigr) = \frac kn \qquad (k \geq 1) :
 \]
 en un conteo donde el ganador lidera por $k$ de $n$
 boletas, la probabilidad de que el ganador liderara en todas partes
 el recuento es $k/n$. Verifique manualmente para $n = 3$, $k =
 1$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Return to the origin.}
\begin{enumerate}[resume]
 \item Demostrar la identidad de la clave
 \[
 \P(S_1 \neq 0,\ S_2 \neq 0,\ \dots,\ S_{2n} \neq 0) =
 u_n
 \]
 \emph{(condition on the first step, sum the counts of
 question 4 over the endpoint, and telescope; finish
 with $2\binom{2n-1}{n} = \binom{2n}{n}$)}.
 \item Deducir de monótono \omterm{def:b2:metric:continuity}{continuidad}
 (\cref{thm:b2:proba:continuity}) que el paseo
 regresa a $0$ al menos una vez con probabilidad $1$,
 y que $f_n := \P(\text{first return at time }2n)$
 satisface
 \[
 f_n = u_{n-1} - u_n = \frac{u_n}{2n-1},
 \qquad \sum_{n\geq1}f_n = 1 .
 \]
 \item Demuestre que $\sum_n 2n\,f_n = \infty$: la devolución es
 seguro, pero la serie que calcularía la media
 el tiempo de espera diverge (en el vocabulario de
 \cref{ch:b2:randomvar}, el tiempo de retorno tiene infinito
 expectativa).
 \item Demuestre que por cada $k \geq 1$, $\P(\text{at least }
 k\text{ returns to }0) = 1$ \emph{(decompose over the
 times of the first $k$ returns: the corresponding
 toss blocks are disjoint, so the probabilities
 multiply and sum to $(\sum_nf_n)^k$)}; concluir con
 monótono \omterm{def:b2:metric:continuity}{continuidad}:
 \[
 \P(S_n = 0 \text{ for infinitely many } n) = 1 :
 \]
 el \omterm{pb:b2:proba:1}{paseo aleatorio simple} en $\Z$ es \emph{recurrente}.
 \item Demuestra que la caminata visita todos los sitios $k \in \Z$
 casi seguramente, por lo tanto (por recurrencia, reiniciado en el
 primera visita) infinitamente a menudo. \emph{(The signs of
 the successive excursions from $0$ are \omterm{def:b2:proba:independence}{independiente}
 fair coins; a positive excursion visits $1$.)}
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Borel--Cantelli and the biased
walk.}
\begin{enumerate}[resume]
 \item El \omterm{def:b2:proba:space}{eventos} $A_n = \{S_{2n} = 0\}$ satisface $\sum\P(A_n)
 = \infty$; explica por qué Borel--Cantelli~2 no
 \emph{no} se aplican a ellos, y lo que Borel--Cantelli~1
 daría si la serie convergiera. (Este es el
 estrategia de toda la Parte.)
 \item Ahora deja que la moneda tenga sesgo $p \neq \frac12$, $q = 1 -
 p$. Mostrar $\P(S_{2n} = 0) = \binom{2n}n(pq)^n =
 u_n\,(4pq)^n $con $4pq < 1$, deducir
 $\sum_n\P(S_{2n} = 0) < \infty$, y concluir con
 Borel--Cantelli~1 que el paseo sesgado regresa a $0$
 sólo un número finito de veces, casi con seguridad.
 \item Fotograma de $p \neq \frac12$: mostrar $\P(S_n = k) \leq
 \binom{n}{\floor{n/2}}\,(pq)^{n/2}\,(p/q)^{k/2}$ para
 cada $k$ fijo, deduzca que cada sitio es visitado
 finitamente a menudo casi con seguridad, y concluimos
 $\abs{S_n} \to \infty$ casi seguro: el paseo sesgado
 es \emph{transitorio}.
 \item Volviendo a la moneda justa: utilizando la pregunta 6, calcule el
 probabilidad de que los lanzamientos $200$ produzcan un empate \emph{no}
 ($S_n \neq 0$ para $1 \leq n \leq 200$), numéricamente
 $u_{100} \approx 0.056$. Comenta sobre lo lento
 $1/\sqrt{\pi n}$ decadencia: los lazos son seguros a largo plazo
 correr pero más raro de lo que sugiere la intuición.
 \item (Primer pasaje) Que $T_1$ sea la primera vez que caminamos.
 llega a $1$. Utilizando el principio de reflexión para
 máximo $M_n = \max_{i\leq n}S_i$ (probado en cuestión
 16, que no depende de éste), o directamente
 de la pregunta 7 condicionando el primer paso,
 mostrar $\P(T_1 = 2n - 1) = f_n$; deducir $\P(T_1 <
 \infty) = 1$ mientras la serie de tiempo medio
 $\sum(2n-1)f_n$ diverge.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Maxima, last zero, long leads.}
\begin{enumerate}[resume]
 \item (Reflexión para el máximo) Para $k \geq 1$, demostrar
 \[
 \P(M_n \geq k) = 2\,\P(S_n > k) + \P(S_n = k)
 \]
 reflejando el camino tras su primera visita a
 nivel $k$.
 \item Deducir $\P(M_{2n} \geq 1) = 1 - u_n$, es decir\
 $\P(S_i \leq 0 \text{ for all } i \leq 2n) = u_n$:
 la probabilidad de nunca estar adelante es igual a la
 probabilidad de nunca estar en cero (pregunta 6) ---
 dos \omterm{def:b2:proba:space}{eventos} diferentes, una probabilidad.
 \item (Último cero) Deje $L_{2n} = \max\{k \leq 2n : S_k =
 0\}$ (par). Combinando la pregunta 6 con \omterm{def:b2:proba:independence}{independencia}
 de bloques de lanzamiento desunidos, mostrar
 \[
 \P(L_{2n} = 2k) = u_k\,u_{n-k}
 \qquad (0 \leq k \leq n),
 \]
 y deducir, sin más cálculos, el binomio
 identidad $\sum_{k=0}^n u_ku_{n-k} = 1$.
 \item Demuestre que la ley de $L_{2n}$ es \omterm{def:b2:quadratic:adjoint}{simétrico} ($\P(L =
 2k) = \P(L = 2n - 2k)$) y, utilizando $u_j \sim
 1/\sqrt{\pi j}$, que sus extremos son los más
 valores probables. Tabule para $n = 5$: $\P(L_{10} = 0)
 = u_5 \approx 0.246$ versus $\P(L_{10} = 4) = u_2u_3
 \approx 0.117$. Interpretar: en un juego limpio y largo, el
 El último empate tiende a ser muy temprano o muy tarde ---
 Las pistas largas son la regla, no la excepción.
 \item Reúna las preguntas 16 a 19 en un párrafo sobre
 imagen de fluctuación del paseo justo: la difusión
 escala sugerida por la pregunta 3, la certeza de
 retorno contra el tiempo de espera medio divergente, y
 la persistencia de los cables con sabor a arcoseno.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The renewal identity and
Pólya's theorem.}
\begin{enumerate}[resume]
 \item Demuestre, dividiendo $\{S_{2n} = 0\}$ a lo largo del tiempo
 de la primera devolución, el \emph{renovación de identidad}
 \[
 u_n = \sum_{k=1}^n f_k\,u_{n-k} \quad (n \geq 1),
 \qquad\text{hence}\qquad
 U(x)\bigl(1 - F(x)\bigr) = 1 \quad (0 \leq x < 1),
 \]
 donde $U(x) = \sum_{n\geq0}u_nx^n$ y $F(x) =
 \sum_{n\geq1}f_nx^n$ (justificar los radios y el
 producto de serie con \cref{ch:b2:powerseries}).
 \item Deducir el \emph{dicotomía de recurrencia}: dejando $x
 \to 1^-$ (límites monótonos de series con no negativos
 coeficientes),
 \[
 \sum_n u_n = \infty \iff \sum_n f_n = 1 ,
 \]
 y compárelo con las preguntas 3, 7 (camino justo) y
 12 (paseo sesgado).
 \item (Dimensión $2$) El simple paseo por $\Z^2$ requiere pasos
 $(\pm1, 0)$, $(0, \pm1)$ uniformemente. Demuestre que el
 Las coordenadas rotadas $U_n = X_n + Y_n$ y $V_n = X_n
 - Y_n$ realizan paseos justos \emph{\omterm{def:b2:proba:independence}{independiente}} en
 $\Z$, deducir
 \[
 \P\bigl(S^{(2)}_{2n} = (0,0)\bigr) = u_n^2 \sim
 \frac1{\pi n},
 \qquad \sum_n u_n^2 = \infty ,
 \]
 y concluir con las preguntas 21--22 (cuyas pruebas
 transferir textualmente) que el paseo por $\Z^2$ es
 recurrente.
 \item (Dimensión $3$) Para el simple paseo por $\Z^3$, admita
 la estimación local $\P(S^{(3)}_{2n} = 0) \leq
 C\,n^{-3/2}$ (probada con el teorema del límite local en
 el volumen del año 3). Deducir de Borel--Cantelli~1
 que el paseo por $\Z^3$ es transitorio e indique la
 resultado completo: \emph{Pólya's theorem} --- el \omterm{pb:b2:proba:1}{sencillo
 paseo aleatorio} es recurrente en las dimensiones $1$ y $2$,
 transitorio en dimensión $3$ y superior.
 \item Síntesis. Enumere el papel exacto desempeñado por: ruta
 contar y reflexionar; monótono \omterm{def:b2:metric:continuity}{continuidad};
 \omterm{def:b2:proba:independence}{independencia} de bloques de lanzamiento desunidos;
 Borel--Cantelli~1; la renovación de la identidad. cual soltero
 Hecho \omterm{def:b2:powerseries:analytic}{analítico} ($u_n \sim 1/\sqrt{\pi n}$, por lo tanto
 $\sum u_n = \infty$ pero $\sum u_n^2 = \infty$ y
 $\sum n^{-3/2} < \infty$) decide entre recurrencia
 y fugacidad en cada dimensión?
\end{enumerate}
\end{problem}
