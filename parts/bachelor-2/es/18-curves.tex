\chapter{Curvas}\label{ch:b2:curves}

La geometría ahora se vuelve diferencial. Una curva es un punto que se mueve a través de
espacio; el cálculo nos da su velocidad y aceleración, y la geometría
pregunta qué es intrínseco, independiente de qué tan rápido atravesamos el
trayectoria. Las respuestas son \emph{\omterm{def:b2:curves:length}{longitud del arco}}, que mide la
trayectoria en sí, y \emph{\omterm{thm:b2:curves:frenet2d}{curvatura}}, que mide cómo
se dobla. En la dimensión $3$ un segundo invariante, \emph{\omterm{thm:b2:curves:frenet3d}{torsión}},
Mide cómo la curva se sale de su plano. la contabilidad
El dispositivo para todo esto es el \emph{Marco de frenet} en movimiento.

\section{Arcos parametrizados}

\begin{definition}[Arco parametrizado]\label{def:b2:curves:arc}
A \emph{arco parametrizado}\index{arco parametrizado} de clase
$\mathcal{C}^k$ ($k \geq 1$) es
un mapa $\gamma \colon I \to \R^n$ de clase $\mathcal{C}^k$ en un
intervalo $I$. Un punto $\gamma(t)$ es \emph{regular} si
$\gamma'(t) \neq 0$, y el arco es \emph{regular} si todos sus
los puntos son. La línea que pasa por $\gamma(t)$ dirigida por $\gamma'(t)$
es el \emph{recta tangente}\index{recta tangente} en un punto regular.
\end{definition}

\begin{definition}[Cambio de parámetro]\label{def:b2:curves:reparam}
A \emph{cambio de parametro}\index{cambio de parametro} de clase
$\mathcal{C}^k$ es un
$\mathcal{C}^k$ difeomorfismo $\theta \colon J \to I$ entre
intervalos ($\theta' \neq 0$ en todas partes). Los arcos $\gamma$ y
$\gamma \circ \theta$ se llaman \emph{equivalente}; un
\emph{arco geométrico}\index{arco geométrico} (o curva) es una
clase de equivalencia. Nociones
invariante bajo cambio de parámetro --- la trayectoria, \omterm{def:b2:curves:arc}{tangente
linea}, \omterm{def:b2:curves:length}{longitud del arco}, \omterm{thm:b2:curves:frenet2d}{curvatura} --- se llaman \emph{geométrico}.
\end{definition}

\begin{remark}
La trayectoria por sí sola no determina el \omterm{def:b2:curves:reparam}{arco geométrico}: el
parametrizaciones $t \mapsto (\cos t, \sin t)$ en $[0, 2\pi]$ y en
$[0, 4\pi]$ tiene la misma imagen pero recorre el círculo una y otra vez.
dos veces. Un \omterm{def:b2:curves:reparam}{arco geométrico} recuerda la multiplicidad y orientación.
del recorrido, no de su velocidad.
\end{remark}

\begin{example}[La velocidad no cambia nada
geométrico]\label{ex:b2:curves:speedinvariance}
Parametrizar el círculo unitario mediante
\[
\gamma(t) = (\cos t^2,\ \sin t^2),
\qquad t \in \intcc0{\sqrt{2\pi}} .
\]
La velocidad $\norm{\gamma'(t)} = 2t$ crece linealmente, aún
\[
L = \int_0^{\sqrt{2\pi}}2t\,\dd t = 2\pi ,
\]
el mismo \omterm{def:b2:curves:length}{longitud} que a velocidad constante --- como
Promesas \cref{thm:b2:curves:lengthinv}, a través de \omterm{def:b2:curves:reparam}{cambio de
parámetro} $t \mapsto t^2$. El \omterm{def:b2:curves:arc}{recta tangente}, el \omterm{thm:b2:curves:frenet2d}{curvatura}
calculado a partir de \cref{prop:b2:curves:kappaformula}, y cada
otras cantidades geométricas también concuerdan; solo $t = 0$
Merece un vistazo, donde $\gamma'(0) = 0$ hace que \emph{esto
parametrización} sea irregular aunque la trayectoria es una
círculo perfecto. Las declaraciones geométricas toleran el mal.
parametrizaciones mal: reparar primero, concluir
segundo.
\end{example}

\begin{example}\label{ex:b2:curves:cusp}
El arco $\gamma(t) = (t^2, t^3)$ es $\mathcal{C}^\infty$ pero no
normal: $\gamma'(0) = (0, 0)$. Su trayectoria, la semicúbica.
parábola $y^2 = x^3$, tiene un \emph{cúspide} en el origen: suavidad
de la parametrización no impide una singularidad geométrica
donde la velocidad desaparece. Por eso la hipótesis de la regularidad
$\gamma' \neq 0$ no es cosmético.
\end{example}

\section{Longitud del arco}

\begin{definition}[longitud del arco]\label{def:b2:curves:length}
Sea $\gamma \colon [a, b] \to \R^n$ un arco $\mathcal{C}^1$. su
\emph{longitud}\index{longitud del arco} es
\[
L(\gamma) = \int_a^b \norm{\gamma'(t)}\, \dd t ,
\]
donde $\norm{\cdot}$ es el euclidiano \omterm{def:b2:nvs:norm}{norma}. El \emph{longitud de arco
función}\index{función de longitud de arco} basado en $t_0$ es $s(t) = \int_{t_0}^t
\norm{\gamma'(u)}\,\dd u$.
\end{definition}

\begin{theorem}[La longitud es geométrica; poligonal
caracterización]\label{thm:b2:curves:lengthinv}
\leavevmode
\begin{enumerate}
\item Si $\theta \colon [c, d] \to [a, b]$ es un $\mathcal{C}^1$
\omterm{def:b2:curves:reparam}{cambio de parametro}, luego $L(\gamma \circ \theta) = L(\gamma)$.
\item $L(\gamma)$ es el supremo del \omterm{def:b2:curves:length}{longitudes} de inscrito
polígonos:
\[
L(\gamma)
= \sup\Bigl\{\, \sum_{i=1}^{m} \norm{\gamma(t_i) -
\gamma(t_{i-1})} \;:\; a = t_0 < t_1 < \dots < t_m = b \,\Bigr\}.
\]
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Por el cambio de variables $t = \theta(u)$ (Año 1
volumen, válido ya que $\theta$ es $\mathcal{C}^1$ monótono),
\[
\int_c^d \norm{(\gamma\circ\theta)'(u)}\,\dd u
= \int_c^d \norm{\gamma'(\theta(u))}\,\abs{\theta'(u)}\,\dd u
= \int_a^b \norm{\gamma'(t)}\,\dd t ,
\]
donde usamos $(\gamma\circ\theta)' = \theta'\cdot
(\gamma'\circ\theta)$y, si $\theta$ es decreciente, el signo de
$\theta'$ es absorbido por la inversión de los límites.

\emph{2.} Para cualquier subdivisión, $\gamma(t_i) - \gamma(t_{i-1}) =
\int_{t_{i-1}}^{t_i}\gamma'(t)\,\dd t$, por lo que por el triángulo
desigualdad para integrales $\norm{\gamma(t_i) - \gamma(t_{i-1})}
\leq \int_{t_{i-1}}^{t_i}\norm{\gamma'}$: cada polígono es más corto
que $L(\gamma)$, por lo tanto $\sup \leq L(\gamma)$.

Para la desigualdad inversa, sea $\varepsilon > 0$. Desde $\gamma'$
es \omterm{def:b2:metric:continuity}{continuo} en \omterm{def:b2:metric:compact}{compacto} $[a,b]$, es uniformemente continuo:
hay $\delta > 0$ con $\norm{\gamma'(t) - \gamma'(u)} \leq
\varepsilon $siempre que $\abs{t - u} \leq \delta$. tomar una subdivisión
del paso $\leq \delta$. En cada pieza, para $t \in [t_{i-1}, t_i]$,
\[
\gamma(t_i) - \gamma(t_{i-1})
= \int_{t_{i-1}}^{t_i}\gamma'(t)\,\dd t
= (t_i - t_{i-1})\,\gamma'(t_{i-1}) + R_i,
\qquad
\norm{R_i} \leq \varepsilon\,(t_i - t_{i-1}),
\]
desde $R_i = \int_{t_{i-1}}^{t_i}(\gamma'(t) -
\gamma'(t_{i-1}))\,\dd t$. Por lo tanto
\[
\norm{\gamma(t_i) - \gamma(t_{i-1})}
\geq (t_i - t_{i-1})\norm{\gamma'(t_{i-1})}
- \varepsilon (t_i - t_{i-1}) .
\]
Sumando y comparando $\sum (t_i -
t_{i-1})\norm{\gamma'(t_{i-1})}$ con $\int_a^b\norm{\gamma'}$ (un
Suma de Riemann de la función \omterm{def:b2:metric:continuity}{continuo} $\norm{\gamma'}$, dentro de
$\varepsilon(b - a)$ de la integral para $\delta$ lo suficientemente pequeña por
uniforme \omterm{def:b2:metric:continuity}{continuidad} nuevamente), obtenemos un polígono de \omterm{def:b2:curves:length}{longitud} $\geq
L(\gamma) - 2\varepsilon(b - a)$. Dejando $\varepsilon \to 0$
prueba la afirmación.
\end{proof}

\begin{example}[Arquímedes y los inscritos
polígonos]\label{ex:b2:curves:archimedes}
Para el círculo unitario, el gon regular $n$ inscrito tiene \omterm{def:b2:curves:length}{longitud}
$L_n = 2n\sin\frac\pi n$, y la expansión $\sin x = x -
\frac{x^3}6 + O(x^5)$ da
\[
L_n = 2\pi - \frac{\pi^3}{3n^2} + O\Bigl(\frac1{n^4}\Bigr) :
\]
el poligonal \omterm{def:b2:curves:length}{longitudes} de \cref{thm:b2:curves:lengthinv}
converger \emph{cuadráticamente}. Numéricamente: $L_6 = 6$
(el hexágono, dando el crudo $\pi > 3$), mientras que $L_{96} =
192\sin\frac{\pi}{96} \approx 6.28206$ contra $2\pi \approx
6.28319$--- el error $0.00113$ concuerda con lo previsto
$\pi^3/(3\cdot96^2) \approx 0.00112$. Por eso Arquímedes,
duplicar el hexágono cinco veces a los lados $96$, podría entre paréntesis
$\pi$ a tres dígitos a mano: cada duplicación divide el
error por cuatro. El supremo en la poligonal.
la caracterización no se alcanza sólo en el límite; es
alcanzado \emph{rápido}, porque una curva suave se separa de
sus acordes sólo en segundo orden.
\end{example}

\begin{theorem}[Parametrización de longitud de arco]\label{thm:b2:curves:arclength}
Sea $\gamma \colon I \to \R^n$ un arco $\mathcal{C}^k$ normal
($k \geq 1$). El \omterm{def:b2:curves:length}{función de longitud de arco} $s$ es un $\mathcal{C}^k$
difeomorfismo de $I$ en un intervalo $J$, y $\tilde\gamma =
\gamma \circ s^{-1}$ satisface $\norm{\tilde\gamma'} = 1$
en todas partes. Hasta la traducción del parámetro y la orientación,
esta parametrización \emph{\omterm{def:b2:curves:length}{longitud de arco}} (o \emph{velocidad unitaria}) es
único.
\end{theorem}

\begin{proof}
$s'(t) = \norm{\gamma'(t)} > 0$ y $s'$ es $\mathcal{C}^{k-1}$
(composición del mapa $\mathcal{C}^{k-1}$ $\gamma'$ con el \omterm{def:b2:nvs:norm}{norma},
suave lejos de $0$), por lo que $s$ es $\mathcal{C}^k$, estrictamente
creciente, una biyección sobre $J = s(I)$, y su inversa es
$\mathcal{C}^k$ por el teorema de la función inversa en la dimensión $1$
(Volumen del año 1). entonces
\[
\tilde\gamma'(\sigma)
= \frac{\gamma'(t)}{s'(t)}
= \frac{\gamma'(t)}{\norm{\gamma'(t)}},
\qquad t = s^{-1}(\sigma),
\]
un vector unitario. Si $\hat\gamma = \gamma\circ\theta$ es otro
parametrización de velocidad unitaria, luego $\abs{\theta'} = 1$, por lo que $\theta'
= \pm 1$ es constante (\omterm{def:b2:metric:continuity}{continuidad}), es decir, $\theta(u) = \pm u + c$.
\end{proof}

\begin{remark}
\omterm{def:b2:curves:length}{Longitud del arco} es el parámetro que separa la geometría de
dinámica. Una trayectoria se puede recorrer con cualquier velocidad.
perfil --- la física del movimiento --- pero cada
pregunta invariante de parametrización (forma, flexión,
osculación) tiene un reloj canónico, la distancia recorrida.
Es por eso que todas las fórmulas \omterm{thm:b2:curves:frenet2d}{curvatura} a continuación son \emph{definido}
a velocidad unitaria y luego \emph{traducido} a arbitrario
parametrizaciones por la regla de la cadena: los factores de traducción
son poderes de $v = s'$, y rastrearlos correctamente es el
contenido completo de \cref{prop:b2:curves:kappaformula}.
\end{remark}

\begin{example}[Círculo y hélice]\label{ex:b2:curves:helix}
Para el círculo $\gamma(t) = (R\cos t, R\sin t)$, $\norm{\gamma'} =
R$, entonces $s = Rt $y el \omterm{def:b2:curves:length}{longitud} de una vuelta completa es $2\pi R$. Para el
hélice $\gamma(t) = (a\cos t,\ a\sin t,\ bt)$ con $a > 0$,
$\norm{\gamma'(t)} = \sqrt{a^2 + b^2}$ es constante: la hélice es
recorrido a velocidad constante, y $s = t\sqrt{a^2 + b^2}$.
\end{example}

\begin{example}[Longitud del arco en polar
coordenadas]\label{ex:b2:curves:polarlength}
Una curva polar $r = r(\theta)$ es el arco $\gamma(\theta) =
(r\cos\theta,\ r\sin\theta)$, con
\[
\gamma'(\theta) = (r'\cos\theta - r\sin\theta,\
r'\sin\theta + r\cos\theta),
\qquad
\norm{\gamma'(\theta)}^2 = r'^2 + r^2
\]
(los términos cruzados se cancelan): el elemento polar \omterm{def:b2:curves:length}{longitud} es
$\sqrt{r^2 + r'^2}\,\dd\theta$. Para el cardioide $r = 1 +
\cos\theta$:
\[
r^2 + r'^2 = (1 + \cos\theta)^2 + \sin^2\theta = 2 +
2\cos\theta = 4\cos^2\tfrac\theta2,
\]
y en $\intcc{-\pi}{\pi}$ el coseno del medio ángulo es
no negativo, entonces
\[
L = \int_{-\pi}^{\pi}2\cos\tfrac\theta2\,\dd\theta
= \Bigl[4\sin\tfrac\theta2\Bigr]_{-\pi}^{\pi}
= 4 - (-4) = 8 :
\]
como el arco cicloide de \cref{exo:b2:curves:1}, una curva
construido a partir de círculos tiene un \omterm{def:b2:curves:length}{longitud} racional, sin $\pi$
en cualquier lugar. La factorización de medio ángulo es el truco estándar.
para \omterm{def:b2:curves:length}{longitudes} de curvas generadas por círculos; cuando falla (el
elipse), el \omterm{def:b2:curves:length}{longitud} es una función genuinamente nueva --- una
Integral elíptica, más allá de las formas cerradas elementales.
\end{example}

\section{Curvatura en el plano.}

A lo largo de esta sección, los arcos son $\mathcal{C}^2$ y regulares en
el plano euclidiano orientado. Parametrizamos por \omterm{def:b2:curves:length}{longitud del arco} y
escriba $T(s) = \tilde\gamma'(s)$ para la unidad tangente y $N(s)$
para el vector unitario directamente ortogonal a $T(s)$ (rotación de
$T$ por $+\pi/2$).

\begin{theorem}[Fórmulas del plano Frenet]\label{thm:b2:curves:frenet2d}
Sea $\tilde\gamma$ un arco $\mathcal{C}^2$ de velocidad unitaria en el
plano orientado. Hay una función \omterm{def:b2:metric:continuity}{continuo} $\kappa$, la
(algebraico) \emph{curvatura}\index{curvatura}, tal que
\[
T'(s) = \kappa(s)\, N(s),
\qquad
N'(s) = -\kappa(s)\, T(s).
\]
\end{theorem}

\begin{proof}
Desde $\norm{T(s)}^2 = 1$ para todo $s$, diferenciando el escalar
el producto da $2\langle T'(s), T(s)\rangle = 0$: $T'(s)$ es ortogonal a
$T(s)$, por lo tanto colineal con $N(s)$ (dimensión $2$); escriba $T'(s)
= \kappa(s) N(s)$con $\kappa(s) = \langle T'(s), N(s)\rangle$,
\omterm{def:b2:metric:continuity}{continuo}. Del mismo modo $N' \perp N$, entonces $N' = \lambda T$; y
diferenciando $\langle T, N\rangle = 0$ da $\langle T', N\rangle +
\langle T, N'\rangle = \kappa + \lambda = 0$.
\end{proof}

\begin{definition}\label{def:b2:curves:curvature}
Cuando $\kappa(s) \neq 0$, \emph{radio de
curvatura}\index{radio de curvatura} es $R(s) =
1/\abs{\kappa(s)}$ y \emph{centro de curvatura}\index{centro de
curvatura} es
$\tilde\gamma(s) + \frac{1}{\kappa(s)} N(s)$; el circulo con eso
centro y radio $R(s)$ es el \emph{círculo osculador}\index{círculo osculador}, el
mejor aproximación circular de la curva en $\tilde\gamma(s)$.
\end{definition}

\begin{example}[El círculo oscilador del
exponencial]\label{ex:b2:curves:oscexp}
Para $y = \eu^x$ en el punto $(0, 1)$: $f'(0) = f''(0) = 1$,
entonces, según la fórmula gráfica a continuación,
\[
\kappa(0) = \frac{1}{(1 + 1)^{3/2}} = \frac1{2\sqrt2},
\qquad R = 2\sqrt2 .
\]
La unidad tangente es $T = \frac{(1, 1)}{\sqrt2}$, la directa
$N = \frac{(-1, 1)}{\sqrt2}$ normal y \omterm{def:b2:curves:curvature}{centro de
curvatura} es
\[
(0, 1) + 2\sqrt2\cdot\frac{(-1, 1)}{\sqrt2} = (-2,\ 3) :
\]
el \omterm{def:b2:curves:curvature}{círculo osculador} tiene la ecuación $(x + 2)^2 + (y - 3)^2 =
8$. Como comprobación de la afirmación de "mejor aproximación circular":
resolviendo la ecuación del círculo para $y$ cerca de $(0,1)$ y
expandiéndose da $y = 1 + x + \frac{x^2}2 + O(x^3)$ ---
exactamente la expansión de Taylor de segundo orden de $\eu^x$. el
\omterm{def:b2:curves:curvature}{círculo osculador} coincide con el valor, pendiente \emph{y} segundo
derivado; un círculo tangente ordinario coincidiría sólo con el
los dos primeros.
\end{example}

\begin{example}[La evolución de un círculo es su
centro]\label{ex:b2:curves:circleevolute}
Para el círculo de radio $R$ recorrido en sentido antihorario,
$\kappa = 1/R$ y $N$ apuntan hacia el centro, por lo que el
\omterm{def:b2:curves:curvature}{centro de curvatura} $\tilde\gamma + \frac1\kappa N$ es el
centro del círculo, por cada $s$: el \omterm{def:b2:curves:curvature}{círculo osculador}
de un círculo es el círculo mismo, y el lugar geométrico de los centros
de \omterm{thm:b2:curves:frenet2d}{curvatura} colapsa hasta un punto. Este caso degenerado
calibra \cref{exo:b2:curves:6}: ahí está la evoluta
la velocidad es $-\frac{\kappa'}{\kappa^2}N$, que desaparece
de forma idéntica precisamente cuando $\kappa$ es constante.
\end{example}

\begin{proposition}[Curvatura en forma arbitraria
parametrización]\label{prop:b2:curves:kappaformula}
Para un arco plano normal $\mathcal{C}^2$ $\gamma(t) = (x(t), y(t))$,
\[
\kappa(t)
= \frac{x'(t)\,y''(t) - y'(t)\,x''(t)}
 {\bigl(x'(t)^2 + y'(t)^2\bigr)^{3/2}} ,
\]
en particular $\kappa = \dfrac{y''}{(1 + y'^2)^{3/2}}$ para un gráfico
$y = f(x)$.
\end{proposition}

\begin{proof}
Escribe $v(t) = \norm{\gamma'(t)} = s'(t)$, entonces $\gamma' = vT$
(componiendo los datos de velocidad unitaria con $s$). diferenciando,
\[
\gamma'' = v'T + v\,T'\cdot s' = v'T + v^2\kappa N .
\]
Ahora tome el \omterm{def:b2:linalg:det}{determinante} (en la base orientada canónicamente) de
$(\gamma', \gamma'')$: desde $\det(T, T) = 0$ y $\det(T, N) = 1$,
\[
\det(\gamma', \gamma'') = \det(vT,\ v'T + v^2\kappa N)
= v^3\kappa .
\]
El lado izquierdo es $x'y'' - y'x''$ y $v^3 = (x'^2 +
y'^2)^{3/2}$. El caso del gráfico es la parametrización $t \mapsto (t,
f(t))$.
\end{proof}

\begin{example}[Círculo, recta, parábola]\label{ex:b2:curves:kappaexamples}
Una línea tiene $\kappa = 0$ (y a la inversa: $T' = 0$ significa $T$
constante, entonces $\tilde\gamma(s) = \tilde\gamma(0) + sT$, una línea). el
El círculo de radio $R$ recorrido en sentido antihorario tiene $\kappa = 1/R$:
con $\gamma(t) = (R\cos t, R\sin t)$, la fórmula da $\kappa =
R^2/R^3$. Para la parábola $y = x^2/2$: $\kappa(x) = 1/(1 +
x^2)^{3/2}$, máxima en el vértice --- la parábola es más pronunciada
doblado donde gira.
\end{example}

\begin{remark}[Errores comunes en torno a la curvatura]
(i) El \omterm{thm:b2:curves:frenet2d}{curvatura} algebraico de un arco plano cambia de signo cuando
la orientación del arco o del plano se invierte: sólo
$\abs\kappa$ y $R = 1/\abs\kappa$ son puramente geométricos. un
El círculo recorrido en el sentido de las agujas del reloj tiene $\kappa = -1/R$. (ii) El
fórmula gráfica $\kappa = f''/(1 + f'^2)^{3/2}$ en silencio
elige la parametrización por $x$; aplicándolo a una curva
que no es una gráfica cerca del punto (tangente vertical) es la
error clásico. (iii) En un punto donde $\gamma' = 0$
no hay nada definido --- ni $T$ ni $\kappa$ --- y el
la trayectoria realmente puede romperse
(\cref{ex:b2:curves:cusp}); comprobar siempre la regularidad antes
derivando la tangente unitaria. (iv) En el espacio, $\kappa =
\norm{T'} \geq 0$ por convención: no hay señal para llegar
incorrecto, pero tampoco hay señal para explotar --- tipo de inflexión
la información pasa al \omterm{thm:b2:curves:frenet3d}{torsión}. (v) Finalmente, $\kappa$ es
un derivado \emph{with respect to \omterm{def:b2:curves:length}{longitud del arco}}: para un
parametrización no unitaria de velocidad, olvidándose del factor $v^3$
en \cref{prop:b2:curves:kappaformula} es el más frecuente
error en la práctica.
\end{remark}

\begin{theorem}[La curvatura determina la curva.]\label{thm:b2:curves:fundamental}
Sea $\kappa \colon J \to \R$ \omterm{def:b2:metric:continuity}{continuo}. existe un
arco de velocidad unitaria $\mathcal{C}^2$ en el plano con \omterm{thm:b2:curves:frenet2d}{curvatura}
$\kappa$, y es único hasta una isometría directa (rotación
seguido de traducción).
\end{theorem}

\begin{proof}
\emph{Existencia.} Repare $s_0 \in J$ y configure $\varphi(s) =
\int_{s_0}^s \kappa(u)\,\dd u$, luego
\[
\tilde\gamma(s) = \Bigl(\int_{s_0}^s \cos\varphi(u)\,\dd u,\
\int_{s_0}^s \sin\varphi(u)\,\dd u\Bigr).
\]
Entonces $T(s) = (\cos\varphi(s), \sin\varphi(s))$ es un vector unitario,
$N(s) = (-\sin\varphi, \cos\varphi)$, y
\[
T'(s) = \varphi'(s)\,(-\sin\varphi, \cos\varphi)
= \kappa(s)\,N(s) :
\]
el arco es de velocidad unitaria con \omterm{thm:b2:curves:frenet2d}{curvatura} $\kappa$.

\emph{Unicidad.} Sean $\gamma_1, \gamma_2$ arcos de velocidad unitaria con
el mismo \omterm{thm:b2:curves:frenet2d}{curvatura}. Cada unidad de tangente se eleva formando un ángulo.
función, mediante una construcción explícita: vea $T_j$ como el
número complejo $z_j = a_j + \iu b_j$ del módulo $1$, elija
$\varphi_j(0)$ con $z_j(0) = \eu^{\iu\varphi_j(0)}$ y establezca
\[
\varphi_j(s) = \varphi_j(0) + \int_0^s\det\bigl(T_j,
T_j'\bigr)(u)\,\dd u .
\]
De $\abs{z_j} = 1$: $\operatorname{Re}(\conj{z_j}\,z_j') =
0$, entonces $\conj{z_j}\,z_j' = \iu\det(T_j, T_j') =
\iu\,\varphi_j'$, es decir $z_j' = \iu\varphi_j'z_j$; entonces
\[
\bigl(z_j\,\eu^{-\iu\varphi_j}\bigr)'
= \eu^{-\iu\varphi_j}\bigl(z_j' - \iu\varphi_j'z_j\bigr) = 0,
\]
entonces $z_j = \eu^{\iu\varphi_j}$ en todas partes: $T_j =
(\cos\varphi_j, \sin\varphi_j)$con $\varphi_j$ de clase
$\mathcal C^1$. Además $\det(T_j, T_j') = \det(T_j,
\kappa N_j) = \kappa$, entonces $\varphi_j' = \kappa$. Por lo tanto $\varphi_2 = \varphi_1 + c$ para un
constante $c$: $T_2$ es $T_1$ girado por el ángulo fijo $c$, por lo que
integrando, $\gamma_2 = \rho(\gamma_1) + w$ donde $\rho$ es el
rotación del ángulo $c$ y $w$ un vector constante.
\end{proof}

\section{Marco Frenet en el espacio.}

Ahora sea $\tilde\gamma \colon J \to \R^3$ una unidad de velocidad.
$\mathcal{C}^3$ arco que es \emph{biregular}: $T'(s) \neq 0$ para
todos $s$. Entonces $\kappa(s) = \norm{T'(s)} > 0$ define el
\emph{\omterm{thm:b2:curves:frenet2d}{curvatura}} (sin espacio para iniciar sesión: no hay ningún espacio normal preferido)
orientación), y establecemos:
\[
N(s) = \frac{T'(s)}{\kappa(s)}
\quad\text{(principal normal)},
\qquad
B(s) = T(s) \wedge N(s)
\quad\text{(binormal)} ,
\]
de modo que $(T, N, B)$ es un marco ortonormal directo, el
\emph{Marco de frenet}. El avión que pasó por $\tilde\gamma(s)$ atravesó
$T, N$ es el \emph{plano osculador}.

\begin{theorem}[Fórmulas de Frenet en el espacio]\label{thm:b2:curves:frenet3d}
Existe una función \omterm{def:b2:metric:continuity}{continuo} $\tau$, la \emph{torsión}\index{torsión}, con
\[
T' = \kappa N, \qquad
N' = -\kappa T + \tau B, \qquad
B' = -\tau N .
\]
\end{theorem}

\begin{proof}
La primera fórmula es la definición de $N$. cada uno de los vectores
$T, N, B$ tiene constante \omterm{def:b2:nvs:norm}{norma} $1$ y son ortogonales por pares;
diferenciando las seis relaciones $\langle X, Y\rangle = \delta_{XY}$
muestra la matriz de $(T', N', B')$ en la base $(T, N, B)$ es
antisimétrico: de hecho $\langle X', Y\rangle + \langle X, Y'\rangle = 0$ y
$\langle X', X\rangle = 0$. Su entrada $(N, T)$ es $\langle N', T\rangle =
-\langle N, T'\rangle = -\kappa $y su entrada de columna $(T, B)$
$\langle T', B\rangle = \kappa\langle N, B\rangle = 0$. Nombrando el resto
la entrada gratuita $\tau = \langle N', B\rangle$ proporciona exactamente los tres mostrados
fórmulas: la antisimetría completa $\langle B', N\rangle = -\tau$ y
$\langle B', T\rangle = 0$. \omterm{def:b2:metric:continuity}{Continuidad} de $\tau = \langle N', B\rangle$ está claro
ya que $N'$ y $B$ son \omterm{def:b2:metric:continuity}{continuo} ($\tilde\gamma$ es
$\mathcal{C}^3$, por lo que $N = T'/\kappa$ es $\mathcal{C}^1$).
\end{proof}

\begin{example}[El vector de Darboux]\label{ex:b2:curves:darboux}
Las tres fórmulas de Frenet se comprimen en una. Configure $\omega(s) =
\tau\,T + \kappa\,B$(el \emph{vector de Darboux}). Usando $B
\wedge T = N$, $T \wedge N = B$, $N \wedge B = T$:
\[
\omega \wedge T = \kappa\,N = T', \qquad
\omega \wedge N = \tau\,B - \kappa\,T = N', \qquad
\omega \wedge B = -\tau\,N = B' :
\]
cada vector de cuadro evoluciona en $X' = \omega \wedge X$, el
firma cinemática de un \emph{rotación instantánea} con
vector de velocidad angular $\omega$. El marco gira a un ritmo
$\norm\omega = \sqrt{\kappa^2 + \tau^2}$ sobre la mudanza
eje $\omega$; \omterm{thm:b2:curves:frenet2d}{curvatura} es el componente del giro sobre
el binormal, \omterm{thm:b2:curves:frenet3d}{torsión} el componente respecto a la tangente. Para
la hélice, $\omega$ es un vector constante a lo largo del cilindro
eje --- que es exactamente por qué el marco de la hélice precede
uniformemente. La antisimetría de la matriz de Frenet, explotada en
\cref{exo:b2:curves:7}, es la forma matricial de este sencillo
hecho geométrico.
\end{example}

\begin{proposition}[Medidas de torsión planaridad]\label{prop:b2:curves:torsion}
Un arco biregular está contenido en un plano si y sólo si $\tau
\equiv 0$; en ese caso el plano es el osculador (constante)
avión.
\end{proposition}

\begin{proof}
Si $\tau \equiv 0$, entonces $B' = 0$, entonces $B$ es una unidad constante
vector $B_0$, y
\[
\frac{\dd}{\dd s}\langle \tilde\gamma(s), B_0\rangle
= \langle T(s), B_0\rangle = 0 :
\]
$\langle \tilde\gamma, B_0\rangle$ es constante, por lo que el arco se encuentra en un plano.
ortogonal a $B_0$. Por el contrario, si el arco se encuentra en un plano $P$,
entonces $T$ y $T'$ (por lo tanto $N$) son paralelos a la dirección de
$P$ para todos los $s$; entonces $B = T \wedge N$ es una de las dos unidades
normales de $P$, y al ser \omterm{def:b2:metric:continuity}{continuo} es constante; luego $0 = B'
= -\tau N$con $N \neq 0$ fuerza a $\tau \equiv 0$.
\end{proof}

\begin{example}[Un círculo inclinado tiene torsión cero.]
El arco $\gamma(t) = \bigl(\cos t,\ \tfrac{\sin t}{\sqrt2},\
\tfrac{\sin t}{\sqrt2}\bigr)$se encuentra en el plano $y = z$, y
es el círculo unitario de ese plano (verifique: $\norm{\gamma(t)} =
1$y la base ortonormal del avión $(1,0,0)$,
$(0, \tfrac1{\sqrt2}, \tfrac1{\sqrt2})$ exhibe el estándar
parametrización). Sin ningún cálculo de Frenet,
\cref{prop:b2:curves:torsion} predice $\tau \equiv 0$, y
el binormal fijo debe ser la unidad normal del avión $\pm(0,
\tfrac1{\sqrt2}, -\tfrac1{\sqrt2})$. \omterm{thm:b2:curves:frenet3d}{Torsión} no mide
estar "inclinado en el espacio"; mide \emph{partida} a
avión. Sólo el $b$ distinto de cero de la hélice a continuación produce
\omterm{thm:b2:curves:frenet3d}{torsión}.
\end{example}

\begin{example}[La helice]\label{ex:b2:curves:helixfrenet}
Para la hélice $\gamma(t) = (a\cos t, a\sin t, bt)$, $a > 0$,
calculado $s = ct$ con $c = \sqrt{a^2 + b^2}$. entonces
\[
T = \frac1c(-a\sin t,\ a\cos t,\ b),
\qquad
T' \cdot \frac{\dd t}{\dd s}
= \frac{1}{c^2}(-a\cos t, -a\sin t, 0),
\]

0)$: la normal principal apunta horizontalmente hacia el eje.
Siguiente $B = T \wedge N = \frac1c(b\sin t, -b\cos t, a)$ y $B'
\frac{\dd t}{\dd s} = \frac{b}{c^2}(\cos t, \sin t, 0) = -\tau N$
da
\[
\boxed{\ \kappa = \frac{a}{a^2 + b^2}, \qquad
\tau = \frac{b}{a^2 + b^2}. \ }
\]
Ambas invariantes son constantes --- y se puede demostrar, a la inversa, que
las únicas curvas biregulares con constante $\kappa > 0$ y constante
$\tau$ son hélices (círculos cuando $\tau = 0$). Tenga en cuenta los signos: $b >
0$ da una hélice derecha con \omterm{thm:b2:curves:frenet3d}{torsión} positivo.
\end{example}

\section{Estudio local: posición respecto de la tangente}

\begin{proposition}[Forma local en forma regular.
punto]\label{prop:b2:curves:local}
Sea $\gamma$ un arco plano de clase $\mathcal{C}^k$ en $t_0$, con
$p$ el índice más pequeño con $\gamma^{(p)}(t_0) \neq 0$ y $q$ el
índice más pequeño $> p$ con $\gamma^{(q)}(t_0)$ no colineal con
$\gamma^{(p)}(t_0)$ (suponiendo que ambos existan, $q \leq k$). en la base
$(u, v) = (\gamma^{(p)}(t_0), \gamma^{(q)}(t_0))$ centrado en
$\gamma(t_0)$, Taylor--Young da coordenadas
\[
X(t) \sim \frac{(t - t_0)^p}{p!},
\qquad
Y(t) \sim \frac{(t - t_0)^q}{q!} .
\]
La imagen local depende únicamente de las paridades de $p$ y $q$:
\begin{center}
\begin{tabular}{c c p{0.42\textwidth}}
$p$ impar, $q$ par &
  \emph{punto ordinario} &
  no cruza la tangente; queda de un lado\\
$p$ impar, $q$ impar &
  \emph{punto de inflexión} &
  cruza su tangente\\
$p$ par, $q$ impar &
  \emph{cúspide de primer tipo} &
  ramas a lados opuestos de la tangente\\
$p$ par, $q$ par &
  \emph{cúspide de segundo tipo} &
  ambas ramas del mismo lado\\
\end{tabular}
\end{center}
\end{proposition}

\begin{proof}
Taylor--Young en el orden $q$ (la función $\gamma$ es
$\mathcal{C}^q$ cerca de $t_0$):
\[
\gamma(t) - \gamma(t_0)
= \sum_{j=p}^{q} \frac{(t-t_0)^j}{j!}\,\gamma^{(j)}(t_0)
+ o\bigl((t-t_0)^q\bigr).
\]
Mediante la elección de $p$ y $q$, cada $\gamma^{(j)}(t_0)$ con $p
\leq j < q $es colineal con $u$; Recolectando componentes en el
base $(u, v)$: $X(t) = \frac{(t-t_0)^p}{p!}(1 + o(1))$ y
$Y(t) = \frac{(t-t_0)^q}{q!}(1 + o(1))$. La tabla de signos de $X$ y
$Y$ para $t \gtrless t_0$ --- regido exactamente por las paridades ---
da las cuatro imágenes: por ejemplo, si $p$ es par, $X > 0$ en
ambos lados (ambas ramas salen en dirección $+u$: una cúspide),
y el costado del \omterm{def:b2:curves:arc}{recta tangente} ($\operatorname{sign} Y$) se voltea
con $q$ impar.
\end{proof}

\begin{example}
Para $\gamma(t) = (t^2, t^3)$ en $t_0 = 0$
(\cref{ex:b2:curves:cusp}): $\gamma'' (0)= (2, 0)$, $\gamma'''(0) =
(0, 6)$, entonces $p = 2$, $q = 3$: una cúspide del primer tipo, la
imagen familiar de la parábola semicúbica. Para $\gamma(t) = (t,
t^3)$en $0$: $p = 1$, $q = 3$: inflexión --- las cruces cúbicas
su tangente.
\end{example}

\begin{remark}[Método: ejecutar el estudio local]
En la práctica, la clasificación es una rutina de cuatro pasos.
\emph{Uno}, diferenciar en $t_0$ hasta el primer distinto de cero
aparece la derivada: su índice es $p$, su valor el vector
$u$. \emph{Dos}, seguir diferenciando hasta obtener una derivada
Aparece no colineal con $u$: índice $q$, vector $v$.

Y se permiten derivadas intermedias colineales con $u$

\end{remark}

Determine $p$, $q$ y la forma local (ordinaria, inflexión,
cúspide) de $\gamma(t) = (t^2,\ t^4 + t^5)$ en $t = 0$, y de
$\gamma(t) = (t^3,\ t^4)$ en $t = 0$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:curves:6}
Sea $\gamma$ un arco plano de velocidad unitaria con $\kappa(s) > 0$ para
todo $s$, y sea $c(s) = \gamma(s) + \frac{1}{\kappa(s)}N(s)$
el \omterm{def:b2:curves:curvature}{centro de curvatura} (la curva $c$ es la \emph{evolucionar}).
Suponiendo que $\kappa$ es $\mathcal{C}^1$, demuestre que $c'(s) =
-\frac{\kappa'(s)}{\kappa(s)^2}N(s)$: la evoluta es tangente a
las líneas normales de $\gamma$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:curves:7}
Sea $A(s)$ una familia \omterm{def:b2:metric:continuity}{continuo} de $3 \times 3$ antisimétrico
matrices y $F' = F A$ una solución matricial con $F(s_0)$
ortogonal. Demuestre que $F(s)$ es ortogonal para todo $s$.
\emph{(Differentiate $G = F F^{\mathsf T}$ and use uniqueness in
Cauchy--Lipschitz.)} Explique la relevancia para el sistema Frenet.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:curves:8}
(Total \omterm{thm:b2:curves:frenet2d}{curvatura} de una curva convexa cerrada) Sea $\tilde\gamma$ un
velocidad unitaria $\mathcal{C}^2$ arco plano cerrado de \omterm{def:b2:curves:length}{longitud} $L$
(entonces $\tilde\gamma(s + L) = \tilde\gamma(s)$), atravesado una vez
en sentido antihorario. Usando la función de ángulo $\varphi$ con $T =
(\cos\varphi, \sin\varphi)$ de
\cref{thm:b2:curves:fundamental}, explique por qué $\varphi(L) -
\varphi(0)$es un múltiplo de $2\pi$ y demuestre que
$\int_0^L \kappa(s)\,\dd s = \varphi(L) - \varphi(0)$. (Por un
círculo de radio $R$: $\int \kappa = \frac1R \cdot 2\pi R = 2\pi$.
El teorema de las tangentes de giro afirma el valor $2\pi$ para
cada curva cerrada simple; no se le pide que lo demuestre).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:curves:9}
Demuestre que una curva espacial biregular con constante $\kappa > 0$ y
$\tau = 0$ es (un arco de) un círculo de radio $1/\kappa$.
\emph{(Use \cref{prop:b2:curves:torsion}, then show the center
$\gamma + \frac1\kappa N$ is constant.)}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:curves:10}
Calcule el \omterm{def:b2:curves:length}{longitud del arco} de la parábola $y = x^2/2$ sobre
$\intcc0a$ y demostrar que es igual
\[
\tfrac12\Bigl(a\sqrt{1 + a^2} + \ln\bigl(a + \sqrt{1 +
a^2}\bigr)\Bigr).
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:curves:11}
Sea $\gamma$ un arco $\mathcal C^2$ normal en $\R^n$ todos
cuyo \omterm{def:b2:curves:arc}{rectas tangentes} pasa por un punto fijo $P$. demostrar que
la trayectoria de $\gamma$ está contenida en una línea recta.

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:curves:12}
(Teorema fundamental para curvas espaciales) Sean $\kappa > 0$ y

\begin{problem}[{Problema del fin de semana --- la máquina de sobres y
cuatro curvas clásicas}]\label{pb:b2:curves:1}
Una familia de líneas de un solo parámetro generalmente no cubre la
plano uniformemente: las líneas se acumulan a lo largo de una curva tangente a todas
de ellos, su \emph{sobre}\index{sobre}. rayos de luz
hacer que \omterm{pb:b2:curves:1}{sobres} sea visible como \emph{cáusticos} --- el brillante
curva cúspide en una taza de café. Este problema construye el
máquina general \omterm{pb:b2:curves:1}{sobre}, luego la ejecuta cuatro veces: el deslizamiento
escalera (\omterm{pb:b2:curves:1}{astroide}), las normales de la parábola y de la
cicloide (evoluta, con el péndulo de Huygens al final), y la
Cáustico para taza de café (\omterm{pb:b2:curves:1}{nefroide}). En todo momento, $D_t$ denota el
línea de ecuación $a(t)\,x + b(t)\,y = c(t)$, donde $a, b, c$
son funciones $\mathcal C^2$ con $(a(t), b(t)) \neq (0,0)$,
y $\Delta(t) = a(t)b'(t) - a'(t)b(t)$.

\medskip
\noindent\textbf{Parte I --- The \omterm{pb:b2:curves:1}{sobre} machine.}
\begin{enumerate}
 \item Supongamos que $\Delta(t) \neq 0$. Demuestre que el
 \emph{sistema caracteristico}
 \[
 \begin{cases} a(t)\,x + b(t)\,y = c(t)\\
 a'(t)\,x + b'(t)\,y = c'(t)\end{cases}
 \]
 tiene una solución única $E(t) = (x(t), y(t))$, dada por
 $x = \dfrac{cb' - c'b}{\Delta}$, $y = \dfrac{ac' -
 a'c}{\Delta}$.
 \item Supongamos además que $E$ es $\mathcal C^1$ cerca de $t$
 con $E'(t) \neq 0$. Diferenciando el primero
 ecuación del sistema, muestre $a(t)\,x'(t) +
 b(t)\,y'(t) = 0$y concluya que la curva $E$
 pasa por un punto de $D_t$ \emph{con el
 dirección de} $D_t$: la familia es tangente a $E$,
 que se llama \emph{\omterm{pb:b2:curves:1}{sobre}}.
 \item Control de cordura: el \omterm{def:b2:curves:arc}{rectas tangentes} de la parábola $y =
 x^2/2$ en los puntos $(t, t^2/2)$son $tx - y =
 t^2/2$. Verifique que la máquina \omterm{pb:b2:curves:1}{sobre} devuelva el
 parábola misma.
 \item (\omterm{pb:b2:curves:1}{Sobre} de las normales) Sea $\gamma$ la unidad de velocidad
 con $\kappa(s) \neq 0$. La línea normal en
 $\gamma(s)$ es $\{M : \langle M - \gamma(s), T(s)
 \rangle = 0\}$. Demuestre que su sistema característico
 fuerzas $\langle M - \gamma(s), N(s)\rangle =
 1/\kappa(s)$, por lo que el punto característico es
 el \omterm{def:b2:curves:curvature}{centro de curvatura}: \emph{the \omterm{pb:b2:curves:1}{sobre} of the
 normals is the evolute}, recuperando
 \cref{exo:b2:curves:6}. Consulte $\Delta(s) =
 \kappa(s)$.
 \item Dos degeneraciones. Para el lápiz $D_\theta : x\cos
 \theta + y\sin\theta = 0$, demuestre que el
 El punto característico es el origen de cada $\theta$.
 (el ``\omterm{pb:b2:curves:1}{sobre}'' colapsa hasta un punto, y $E' = 0$:
 la pregunta 2 no se aplica). Para una familia de paralelo
 líneas (constante $a, b$), muestra $\Delta \equiv 0$ y
 que el sistema característico es en general
 inconsistente: no \omterm{pb:b2:curves:1}{sobre}.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The sliding ladder and the
\omterm{pb:b2:curves:1}{astroide}.} Un segmento de \omterm{def:b2:curves:length}{longitud} $1$ se desliza con un extremo $P_t =
(\cos t, 0)$en el suelo y el otro $Q_t = (0, \sin t)$ en
la pared, $t \in \intoo0{\pi/2}$.
\begin{enumerate}[resume]
 \item Demuestre que la línea $(P_tQ_t)$ tiene la ecuación $x\sin t +
 y\cos t = \sin t\cos t$ y que la máquina \omterm{pb:b2:curves:1}{sobre}
 da el punto característico
 \[
 E(t) = (\cos^3 t,\ \sin^3 t) :
 \]
 el \emph{astroide}\index{astroide}, de implícito
 ecuación $x^{2/3} + y^{2/3} = 1$ (extendida a la
 otros cuadrantes por simetría).
 \item Muestre que $E'(0) = 0$ y, utilizando el local
 clasificación (\cref{prop:b2:curves:local}), que
 el \omterm{pb:b2:curves:1}{astroide} tiene una cúspide del primer tipo en $(1, 0)$
 --- y lo mismo en sus cuatro puntos del eje.
 \item Calcular $\norm{E'(t)} = \tfrac32\abs{\sin 2t}$ y
 deducir que el total \omterm{def:b2:curves:length}{longitud} del \omterm{pb:b2:curves:1}{astroide} es $6$.
 \item ¿Dónde toca la escalera el \omterm{pb:b2:curves:1}{astroide}? Mostrar $E(t) =
 P_t + \sin^2 t\,(Q_t - P_t)$: el punto de contacto
 divide la escalera en la proporción $\sin^2 t : \cos^2
 t$, barriéndola de un extremo al otro a medida que
 toboganes de escalera.
 \item Calcule el área encerrada por \omterm{pb:b2:curves:1}{astroide}: demuestre que
 el área del primer cuadrante es $3\int_0^{\pi/2}\sin^4
 t\cos^2 t\,\dd t$, evalúe la integral mediante
 linealización ($\sin^2 2t = \tfrac{1 - \cos 4t}2$),
 y concluir que el área total es $3\pi/8$.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The evolute of the parabola.}
Deje $\gamma(t) = (t, t^2/2)$.
\begin{enumerate}[resume]
 \item Demuestre que la recta normal en $\gamma(t)$ tiene la ecuación
 $x + t\,y = t + t^3/2$.
 \item Ejecute la máquina \omterm{pb:b2:curves:1}{sobre}: muestre que el \omterm{pb:b2:curves:1}{sobre} de
 lo normal es
 \[
 E(t) = \Bigl(-t^3,\ 1 + \tfrac32 t^2\Bigr),
 \]
 con ecuación implícita $x^2 = \tfrac8{27}(y - 1)^3$:
 una parábola semicúbica.
 \item Verifique con la pregunta 4: calcule el \omterm{def:b2:curves:curvature}{centro de
 curvatura} $\gamma(t) + \frac1{\kappa(t)}N(t)$ a partir de
 $\kappa(t) = (1 + t^2)^{-3/2}$
 (\cref{ex:b2:curves:kappaexamples}) y recuperar el
 mismo punto.
 \item Demuestre que la evoluta tiene una cúspide de primer tipo en
 $(0, 1)$, el \omterm{def:b2:curves:curvature}{centro de curvatura} en el vértice ---
 el punto donde $\kappa$ es extremo, como lo predice
 la fórmula $c' = -\frac{\kappa'}{\kappa^2}N$ de
 \cref{exo:b2:curves:6}.
 \item ¿Cuántas normales de la parábola pasan por un determinado
 punto $(x_0, y_0)$? Mostrar que la respuesta se rige por
 el cúbico $\tfrac{t^3}2 + (1 - y_0)\,t - x_0 = 0$;
 tratar el caso del eje $x_0 = 0$ por completo (uno normal
 para $y_0 < 1$, tres para $y_0 > 1$), e interpreta
 la evolución como curva de transición.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The coffee-cup caustic.} Paralelo
Los rayos de dirección $(1, 0)$ inciden en el interior del espejo.
círculo $x^2 + y^2 = 1$; el rayo que incide en $P_\theta =
(\cos\theta, \sin\theta)$ se refleja según la ley de
reflexión.
\begin{enumerate}[resume]
 \item A partir de la simetría especular en la normal (el radio),
 justifique que la dirección reflejada es $v = u -
 2\langle u, n\rangle n $con $u = (1,0)$, $n =
 (\cos\theta, \sin\theta)$y calcule $v =
 -(\cos2\theta, \sin2\theta)$.
 \item Demuestre que el rayo reflejado se encuentra en la recta
 \[
 x\sin 2\theta - y\cos 2\theta = \sin\theta .
 \]
 \item Ejecute la máquina \omterm{pb:b2:curves:1}{sobre} ($\Delta = 2$): muestre que
 la cáustica es
 \[
 E(\theta) = \Bigl(\tfrac{3\cos\theta -
 \cos3\theta}4,\ \tfrac{3\sin\theta -
 \sin3\theta}4\Bigr),
 \]
 el \emph{nefroide}\index{nefroide}.
 \item Calcular $E'(\theta) = \tfrac32\sin\theta\,
 (\cos2\theta, \sin2\theta)$; comprobar que la tangente
 La dirección es la dirección del rayo reflejado (pregunta).
 16), localice las dos cúspides $(\pm\tfrac12, 0)$, y
 mostrar que el rayo reflejado cruza el eje $y = 0$
 en $x = \frac1{2\cos\theta}$ --- rayos casi axiales
 enfocar en $x = \tfrac12$: el \omterm{def:b2:curves:length}{longitud} $R/2$ focal de un
 espejo de radio $R$.
 \item Demuestre que el \omterm{pb:b2:curves:1}{nefroide} tiene un total de \omterm{def:b2:curves:length}{longitud} $6$ y que
 cerca de $\theta = 0$,
 \[
 E(\theta) - \bigl(\tfrac12, 0\bigr) =
 \bigl(\tfrac34\theta^2 + o(\theta^2),\ \theta^3 +
 o(\theta^3)\bigr) :
 \]
 una cúspide del primer tipo, apuntando a lo largo del eje.
 \item Explique en un párrafo por qué la cáustica es brillante:
 a través de cada punto justo fuera del paso cáustico dos
 rayos reflejados, a través de cada punto en él los rayos
 están "infinitamente concentrados" (el mapa $(\theta,
 \text{distance along ray}) \mapsto \R^2$ tiene un
 punto crítico exactamente en el \omterm{pb:b2:curves:1}{sobre}).
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Huygens: the cycloid is its own
evolute.} Sean $\gamma(t) = (t - \sin t,\ 1 - \cos t)$, $t \in
\intoo0{2\pi}$, un arco de la cicloide.
\begin{enumerate}[resume]
 \item Calcular $\kappa(t) = -\dfrac1{4\sin(t/2)}$ y el
 \omterm{def:b2:curves:curvature}{centro de curvatura}; Demuestre que la evolución es
 \[
 c(t) = (t + \sin t,\ \cos t - 1),
 \]
 y que la sustitución $t = u + \pi$ lo exhibe
 como la cicloide original traducida por $(\pi, -2)$:
 \emph{la evolución de una cicloide es congruente
 cicloide} (Huygens).
 \item Verifique que \omterm{def:b2:curves:curvature}{radio de curvatura} en el ápice $t =
 \pi $sea igual a $4$, que es la mitad de \omterm{def:b2:curves:length}{longitud}$8$ de uno
 arco (\cref{exo:b2:curves:1}); localizar la cúspide de la
 evolucionan directamente debajo del ápice, a la distancia $4$.
 \item (La propiedad de cadena) Sea $\gamma$ la unidad de velocidad con
 $\kappa > 0$, $\kappa$ de clase $\mathcal C^1$ y
 $R = 1/\kappa$ estrictamente monótono. Usando $c' = R'N$,
 mostrar que el \omterm{def:b2:curves:length}{longitud del arco} de la evolución entre
 $c(s_0)$ y $c(s_1)$ es $\abs{R(s_1) - R(s_0)}$.
 Interpretar: una cuerda tensa desenrollada de la evoluta,
 de \omterm{def:b2:curves:length}{longitud} $R(s_0)$ al inicio, tiene su extremo libre
 trazar la curva original --- entonces un péndulo oscilando
 entre dos mejillas cicloidales del reloj de Huygens
 describe una cicloide.
 \item Síntesis. La máquina de la Parte I produjo el
 \omterm{pb:b2:curves:1}{astroide}, una parábola semicúbica, una \omterm{pb:b2:curves:1}{nefroide} y una
 cicloide. Para cada una de las cuatro familias, indique en una
 oración donde se cumplen las hipótesis $\Delta \neq 0$ y
 $E' \neq 0$ se celebró o falló, y qué evento geométrico
 (cúspide, foco, degeneración) cada falla de $E' \neq 0$
 señalizado. ¿Dónde deben aparecer los extremos de \omterm{thm:b2:curves:frenet2d}{curvatura}?
 el \omterm{pb:b2:curves:1}{sobre} de los normales, ¿y por qué?
\end{enumerate}
\end{problem}
