\chapter{Espacios vectoriales normados}\label{ch:b2:nvs}

Cuando el \omterm{def:b2:metric:def}{espacio métrico} es un espacio vectorial y la distancia proviene de
a \omterm{def:b2:nvs:norm}{norma}, la topología y el álgebra lineal comienzan a interactuar: mapas lineales
son \omterm{def:b2:metric:continuity}{continuo} exactamente cuando están acotados en la bola unitaria,
dimensión finita obliga a todos los \omterm{def:b2:nvs:norm}{normas} a estar de acuerdo, y \omterm{def:b2:metric:complete}{lo completo} se vuelve
series absolutamente convergentes en series convergentes. el
división de dimensión finita/infinita --- cristalizada en el teorema de Riesz
--- es la lección más profunda del capítulo.

En todo momento, $E, F$ son espacios vectoriales sobre $K = \R$ o $\C$.

\section{Normas}

\begin{definition}\label{def:b2:nvs:norm}
A \emph{norma}\index{norma} en $E$ es un mapa $\norm{\,\cdot\,} \colon E
\to \R_+$con, para todos $x, y \in E$, $\lambda \in K$:
\[
\norm x = 0 \iff x = 0,
\qquad
\norm{\lambda x} = \abs\lambda\,\norm x,
\qquad
\norm{x + y} \leq \norm x + \norm y .
\]
Entonces $d(x, y) = \norm{x - y}$ es una distancia, y todos
Se aplica \cref{ch:b2:metric}. La desigualdad del triángulo inverso
$\bigl|\norm x - \norm y\bigr| \leq \norm{x - y}$ hace la norma
sí mismo $1$-Lipschitz; la suma y la multiplicación escalar son
\omterm{def:b2:metric:continuity}{continuo} (estimaciones $\norm{(x + y) - (x' + y')} \leq \norm{x -
x'} + \norm{y - y'}$, etc.).
\end{definition}

\begin{example}\label{ex:b2:nvs:examples}
En $K^n$:
\[
\norm{x}_1 = \sum_i \abs{x_i},
\qquad
\norm{x}_2 = \Bigl(\sum_i \abs{x_i}^2\Bigr)^{1/2},
\qquad
\norm{x}_\infty = \max_i \abs{x_i}
\]
($\norm\cdot_2$ es un \omterm{def:b2:nvs:norm}{norma} de Cauchy--Schwarz, volumen del año 1). encendido
$C(\intcc{a}{b})$:
\[
\norm f_\infty = \sup \abs f,
\qquad
\norm f_1 = \int_a^b \abs f,
\qquad
\norm f_2 = \Bigl(\int_a^b \abs f^2\Bigr)^{1/2},
\]
siendo los dos últimos \omterm{def:b2:nvs:norm}{normas} gracias a la estricta positividad del
Cauchy integral e integral--Schwarz (volumen del año 1). En matrices:
cualquier \omterm{def:b2:nvs:norm}{norma} en $\mathcal M_n(K) \simeq K^{n^2}$; el \omterm{thm:b2:nvs:continuouslinear}{normas del operador}
a continuación se muestran los estructuralmente importantes.
\end{example}

\begin{definition}[Normas equivalentes]\label{def:b2:nvs:equivalent}
Dos \omterm{def:b2:nvs:norm}{normas} $N_1, N_2$ en $E$ son \emph{equivalente}\index{equivalente
normas} cuando hay constantes $c, C > 0$ con
\[
c\,N_1 \leq N_2 \leq C\, N_1 .
\]
\omterm{def:b2:nvs:norm}{normas} equivalente tiene los mismos conjuntos \omterm{def:b2:metric:topology}{abierto}, los mismos convergentes y
Secuencias de Cauchy, los mismos subconjuntos \omterm{def:b2:metric:compact}{compacto} y \omterm{def:b2:metric:complete}{completo}: los mismos
análisis.
\end{definition}

\begin{example}[No equivalencia en dimensión infinita]\label{ex:b2:nvs:nonequivalent}
En $C(\intcc{0}{1})$: $\norm f_1 \leq \norm f_\infty$ siempre, pero
sin retenciones de límite inverso: $f_n(x) = x^n$ tiene $\norm{f_n}_\infty = 1$
y $\norm{f_n}_1 = \frac{1}{n+1} \to 0$. Entonces $f_n \to 0$ para
$\norm\cdot_1$ pero no para $\norm\cdot_\infty$: los dos \omterm{def:b2:nvs:norm}{normas}
no están de acuerdo sobre la convergencia misma.
\end{example}

\begin{example}[Constantess en la dimensión $n$]\label{ex:b2:nvs:constants}
En $K^n$ los tres clásicos \omterm{def:b2:nvs:norm}{normas} son \omterm{def:b2:nvs:equivalent}{equivalente} con agudos
constantes:
\[
\norm x_\infty \leq \norm x_2 \leq \norm x_1
\leq \sqrt n\,\norm x_2 \leq n\,\norm x_\infty ,
\]
el límite medio $\norm x_1 \leq \sqrt n\norm x_2$ proveniente de
Cauchy-Schwarz contra el vector de todos unos. Vectores extremos:
$e_1$ convierte las dos primeras desigualdades en igualdades, $(1, 1,
\dots, 1)$las dos últimas. La dimensión $n$ se encuentra visiblemente en el
constantes --- la semilla cuantitativa del fracaso en infinito
dimensión: como $n \to \infty$ no sobrevive ninguna constante uniforme,
que es exactamente lo que muestra \cref{ex:b2:nvs:nonequivalent} en
espacios funcionales.
\end{example}

\section{Mapas lineales continuos}

\begin{theorem}[Caracterización]\label{thm:b2:nvs:continuouslinear}
Para un mapa lineal $u \colon E \to F$ entre espacios normados, el
Los siguientes son \omterm{def:b2:nvs:equivalent}{equivalente}:
\begin{enumerate}
 \item $u$ es \omterm{def:b2:metric:continuity}{continuo};
 \item $u$ es \omterm{def:b2:metric:continuity}{continuo} en $0$;
 \item $u$ está acotado en la bola unitaria cerrada: $\sup_{\norm x \leq
 1} \norm{u(x)} < \infty$;
 \item hay $C \geq 0$ con $\norm{u(x)} \leq C \norm x$ para
 todos $x$;
 \item $u$ es \omterm{def:b2:metric:continuity}{Lipschitz}.
\end{enumerate}
El $C$ menos parecido es el \emph{norma del operador}\index{norma del operador}.
$\vertiii{u} = \sup_{\norm x \leq 1}\norm{u(x)} = \sup_{x \neq 0}
\frac{\norm{u(x)}}{\norm x}$; hace que el espacio $\mathcal{L}_c(E,
F)$ de los mapas lineales \omterm{def:b2:metric:continuity}{continuo} sea un espacio normado, con
\[
\vertiii{v \circ u} \leq \vertiii v\, \vertiii u .
\]
\end{theorem}

\begin{proof}
(1 $\Rightarrow$ 2) trivial. (2 $\Rightarrow$ 3): \omterm{def:b2:metric:continuity}{continuidad} en $0$
con $\varepsilon = 1$ da $\delta$ con $\norm x \leq \delta
\Rightarrow \norm{u(x)} \leq 1$; la homogeneidad luego escala cualquier $x$
con $\norm x \leq 1$ hacia abajo en esa bola y de regreso:
\[
\norm{u(x)} = \frac1\delta\,\norm{u(\delta x)} \leq
\frac1\delta ,
\]
desde $\norm{\delta x} \leq \delta$. (3 $\Rightarrow$ 4):
para $x \neq 0$, aplique el límite a $\frac{x}{\norm x}$. (4
$\Rightarrow$ 5): $\norm{u(x) - u(y)} = \norm{u(x - y)} \leq
C\norm{x - y}$. (5$\Rightarrow$ 1) conocido.

\omterm{def:b2:nvs:norm}{Norma} axiomas para $\vertiii\cdot$: homogeneidad y separación son
claro ($\vertiii u = 0$ fuerza a $u = 0$ en la pelota, por lo tanto
en todas partes); desigualdad triangular de $\norm{(u + v)(x)} \leq
\norm{u(x)} + \norm{v(x)}$. Submultiplicatividad: $\norm{v(u(x))}
\leq \vertiii v\,\norm{u(x)} \leq \vertiii v \vertiii u \norm x$.
\end{proof}

\begin{example}\label{ex:b2:nvs:operatornorms}
En $\bigl(C(\intcc{0}{1}), \norm\cdot_\infty\bigr)$: evaluación
$f \mapsto f(0)$ tiene \omterm{thm:b2:nvs:continuouslinear}{norma del operador} $1$; la integración $f \mapsto
\int_0^1 f $tiene \omterm{def:b2:nvs:norm}{norma}$1$; el mapa $f \mapsto \int_0^1 t f(t)\dd t$
tiene \omterm{def:b2:nvs:norm}{norma} $\int_0^1 t\,\dd t = \frac12$ (límite superior por el triángulo
desigualdad para integrales; obtenido en $f \equiv 1$). pero
la diferenciación, de $(C^1, \norm\cdot_\infty)$ a $(C^0,
\norm\cdot_\infty)$, es \emph{no} \omterm{def:b2:metric:continuity}{continuo}: $\norm{\sin(nx)}_\infty
= 1$ mientras que la derivada tiene sup \omterm{def:b2:nvs:norm}{norma}$n$. Lineal no implica
\omterm{def:b2:metric:continuity}{continuo} en dimensión infinita.
\end{example}

\begin{example}[Dos normas, dos veredictos en una secuencia]\label{ex:b2:nvs:sqrtnxn}
En $C(\intcc01)$, deje $g_n(x) = \sqrt{n}\,x^n$. entonces
\[
\norm{g_n}_1 = \frac{\sqrt n}{n + 1} \longrightarrow 0,
\qquad
\norm{g_n}_2^2 = \frac{n}{2n + 1} \longrightarrow \frac12,
\qquad
\norm{g_n}_\infty = \sqrt n \longrightarrow \infty :
\]
una secuencia, tres \omterm{def:b2:nvs:norm}{normas}, tres comportamientos --- convergencia a
cero, sin convergencia (\omterm{def:b2:nvs:norm}{normas} se estabiliza en $\frac1{\sqrt2}$ pero
el límite puntual es $0$) y explosión. concentración de masa
cerca de $x = 1$ es invisible para $\norm\cdot_1$, medio visible para
$\norm\cdot_2$, dominante para $\norm\cdot_\infty$. en infinito
dimensión, "¿converge?" no es una pregunta sobre una
secuencia: se trata de una secuencia \emph{y} a \omterm{def:b2:nvs:norm}{norma}.
\end{example}

\begin{method}[Cálculo de una norma de operador]\label{met:b2:nvs:opnorm}
Siempre en dos movimientos. \emph{Límite superior:} estimación
$\norm{u(x)}$ por $C\norm x$ usando desigualdades triangulares,
Cauchy--Schwarz, o límites integrales--- esto prueba $\vertiii u
\leq C$. \emph{Testigo:} exhibe un $x_0 \neq 0$ específico
con $\norm{u(x_0)} = C\norm{x_0}$ (se alcanza el límite), o un
secuencia de vectores unitarios $x_n$ con $\norm{u(x_n)} \to C$ (el
se acerca el límite). Ambos movimientos son obligatorios: un límite superior
solo da solo $\vertiii u \leq C$, solo un testigo solo
$\vertiii u \geq C$. En dimensión infinita el testigo puede tener
ser una secuencia --- no es necesario alcanzar el supremo
(\cref{exo:b2:nvs:8}).
\end{method}

\begin{example}[Los operadores diagonales ven todas las normas por igual.]\label{ex:b2:nvs:diagonalnorm}
Para $D = \operatorname{diag}(d_1, \dots, d_n)$ en $K^n$ con
\emph{cualquier} del \omterm{def:b2:nvs:norm}{normas} $\norm\cdot_1, \norm\cdot_2,
\norm\cdot_\infty$: desde $\abs{d_ix_i} \leq
\bigl(\max_j\abs{d_j}\bigr)\abs{x_i}$ por coordenadas,
$\norm{Dx} \leq \max_j\abs{d_j}\,\norm x$; y $x = e_{j_0}$
(un índice maximizador) lo alcanza. Entonces $\vertiii D =
\max_j\abs{d_j}$ en los tres casos: para mapas diagonales, todos
razonable \omterm{def:b2:nvs:norm}{normas} cuenta la misma historia, el tramo más grande
factor. Todo lo difícil de \omterm{thm:b2:nvs:continuouslinear}{normas del operador} se trata de
\emph{no}-comportamiento diagonal --- por eso el adaptado
\omterm{def:b2:nvs:norm}{normas} del problema de fin de semana de este capítulo (pregunta 22) funciona por
obligando a una matriz a volverse diagonal primero.
\end{example}

\begin{example}[Sumas de columnas: el gemelo normativo $1$ de \cref{exo:b2:nvs:4}]\label{ex:b2:nvs:columnsums}
En $(\R^n, \norm\cdot_1)$, el \omterm{thm:b2:nvs:continuouslinear}{norma del operador} de una matriz $A$ es
la mayor suma absoluta \emph{columna}. Ejecute el método: para
$\norm x_1 \leq 1$,
\[
\norm{Ax}_1 = \sum_i\Bigl|\sum_j a_{ij}x_j\Bigr|
\leq \sum_j \abs{x_j}\sum_i\abs{a_{ij}}
\leq \Bigl(\max_j\sum_i\abs{a_{ij}}\Bigr)\norm x_1 ,
\]
y el límite se alcanza en $x = e_{j_0}$ para una maximización
columna $j_0$ --- el testigo más ordenado imaginable. Entonces para $A =
\begin{psmallmatrix}1 & -2\\ 3 & 1\end{psmallmatrix}$:
$\vertiii A_1 = \max(1 + 3,\ 2 + 1) = 4$, mientras
$\vertiii A_\infty = 4$ también (filas) --- una coincidencia aquí, no
una ley: \omterm{def:b2:linalg:transpose}{transponer} las entradas de la matriz de forma asimétrica y las dos
\omterm{def:b2:nvs:norm}{normas} parte de la empresa. Filas para $\norm\cdot_\infty$, columnas para
$\norm\cdot_1$: el mnemónico es que los vectores unitarios de cada \omterm{def:b2:nvs:norm}{norma}
(patrones de signos, resp. \ vectores básicos) seleccione la coincidencia
sumas.
\end{example}

\begin{proposition}[mapas bilineales]\label{prop:b2:nvs:bilinear}
Un mapa bilineal $b \colon E \times F \to G$ es \omterm{def:b2:metric:continuity}{continuo} si y así
$\norm{b(x,y)} \leq C\norm x\,\norm y$ para algunos $C$; entonces es
\omterm{def:b2:metric:continuity}{Lipschitz} en conjuntos acotados. (Mismo patrón de prueba; el producto
$(u, v) \mapsto v \circ u$ y la multiplicación de matrices son la clave
ejemplos.)
\end{proposition}

\begin{proof}
Si el límite se cumple:
\[
b(x,y) - b(x_0,y_0) = b(x - x_0,\, y) + b(x_0,\, y - y_0),
\]
entonces $\norm{b(x,y) - b(x_0,y_0)} \leq C\norm{x - x_0}\norm y +
C\norm{x_0}\norm{y - y_0}$: \omterm{def:b2:metric:continuity}{continuidad} en $(x_0, y_0)$, y un
\omterm{def:b2:metric:continuity}{Lipschitz} con destino a $\norm x, \norm y \leq R$. Por el contrario,
\omterm{def:b2:metric:continuity}{continuidad} en $(0,0)$ da $\delta$ con $\norm{b(x,y)} \leq 1$ encendido
$\norm x, \norm y \leq \delta$; escalar ambas variables.
\end{proof}

\section{Dimensión finita}

\begin{theorem}[Equivalencia de normas en dimensión finita]\label{thm:b2:nvs:finitedim}
En un espacio de dimensión finita, \emph{all \omterm{def:b2:nvs:norm}{normas} are
\omterm{def:b2:nvs:equivalent}{equivalente}}.\index{equivalencia de normas} En consecuencia, en finito
dimensión: convergencia, apertura, \omterm{def:b2:metric:compact}{compacidad}, \omterm{def:b2:metric:complete}{lo completo} son
nociones independientes de normas; \omterm{def:b2:metric:compact}{compacto} $=$ cerrado y acotado;
el espacio es \omterm{def:b2:metric:complete}{completo}; y cada mapa lineal (o multilineal)
\emph{de} un espacio de dimensión finita es \omterm{def:b2:metric:continuity}{continuo}.
\end{theorem}

\begin{proof}
Fijar una base e identificar $E \simeq K^n$; basta con comparar cualquier
\omterm{def:b2:nvs:norm}{norma} $N$ con $\norm\cdot_\infty$.

\emph{Una dirección es el álgebra:} $N(x) = N(\sum x_i e_i) \leq
\sum \abs{x_i} N(e_i) \leq C \norm x_\infty $con $C = \sum N(e_i)$.
Esto también muestra que $N$ es \omterm{def:b2:metric:continuity}{continuo} en $(K^n, \norm\cdot_\infty)$.
(es $C$-Lipschitz: $\abs{N(x) - N(y)} \leq N(x - y)$).

\emph{La otra es la topología:} la esfera unitaria $S = \{x :
\norm{x}_\infty = 1\}$ está cerrada y acotada en $(K^n,
\norm\cdot_\infty)$, por lo tanto \omterm{def:b2:metric:compact}{compacto}
(\cref{thm:b2:metric:compactprops}~(2), válido para $\C^n \simeq
\R^{2n}$). La función \omterm{def:b2:metric:continuity}{continuo}$N$alcanza su mínimo $c$ en
$S$; $c > 0$ ya que $N$ desaparece solo en $0 \notin S$. Homogeneidad
extiende el límite: $N(x) \geq c \norm{x}_\infty$ para todos los $x$.

Consecuencias: todas las declaraciones se reducen a $(K^n,
\norm\cdot_\infty)$, donde se conocen.
(\cref{thm:b2:metric:rncomplete},
\cref{thm:b2:metric:compactprops}); a linear $u$ from
de dimensión finita $E$ satisface $\norm{u(x)} \leq \sum\abs{x_i}
\norm{u(e_i)} \leq C'\norm{x}_\infty$: ligado (4) de
\cref{thm:b2:nvs:continuouslinear}.
\end{proof}

\begin{corollary}\label{cor:b2:nvs:closedsubspace}
Un subespacio de dimensión finita de cualquier espacio normado es cerrado.
\end{corollary}

\begin{proof}
Es \omterm{def:b2:metric:complete}{completo} para el \omterm{def:b2:nvs:norm}{norma} inducido.
(\cref{thm:b2:nvs:finitedim}) y \omterm{def:b2:metric:complete}{completo} están cerrados
(\cref{def:b2:metric:complete}).
\end{proof}

\begin{example}[Una mejor aproximación calculada por simetría.]\label{ex:b2:nvs:bestaffine}
En $\bigl(C(\intcc{-1}{1}), \norm\cdot_\infty\bigr)$, ¿a qué distancia está
$f(x) = \abs x$ del subespacio (cerrado, bidimensional) de
funciones afines $a + bx$? Por simetría, sustituyendo $a + bx$ por
$a - bx$ deja $\norm{f - (a \pm bx)}_\infty$ sin cambios y
el punto medio $a$ lo hace al menos tan bien (desigualdad del triángulo en
la media): basta con considerar constantes. por una constante
$a$:
\[
\norm{\abs x - a}_\infty = \max\,(1 - a,\ a)
\geq \frac12 ,
\]
minimizado en $a = \frac12$: la distancia es $\frac12$, alcanzada
por la constante $\frac12$. Tenga en cuenta la curva de error $\abs x -
\frac12$: llega a $\pm\frac12$ alternativamente en $x = -1, 0,
1$ --- tres extremos de signo alterno para una mejor
aproximación de una familia de dos parámetros. eso
El patrón \emph{equioscilación} no es casualidad; es el
firma de optimización que el problema del fin de semana de este capítulo
se convierte en el teorema de Chebyshev.
\end{example}

\begin{example}[Subespacios cerrados versus densos]\label{ex:b2:nvs:closeddense}
En $E = \bigl(C(\intcc{0}{1}), \norm\cdot_\infty\bigr)$: cada uno
$\R_n[X]$ (polinomios de grado $\leq n$, restringidos a
$\intcc01$) es un subespacio de dimensión finita, por lo tanto \emph{cerrado}.
--- un límite uniforme de polinomios de grado $\leq n$ es uno. pero
la unión $\R[X]$ de todos ellos es \emph{denso} en $E$ (el
Teorema de aproximación de Weierstrass, demostrado en
\cref{ch:b2:funcseq}), y los subespacios propios densos son como
no cerrado como puede ser. La moraleja: el carácter cerrado de los subespacios es una
privilegio de dimensión finita; apilar pisos cerrados puede construir un
denso rascacielos.
\end{example}

\begin{theorem}[Riesz]\label{thm:b2:nvs:riesz}
La bola unitaria cerrada de un espacio normado $E$ es \omterm{def:b2:metric:compact}{compacto} \emph{si y
solo si} $\dim E < \infty$.\index{teorema de riesz}
\end{theorem}

\begin{proof}
Dimensión finita: basta cerrada y acotada
(\cref{thm:b2:nvs:finitedim}).

Por el contrario, supongamos $\dim E = \infty$. \emph{Lema de Riesz:} para
Para cada subespacio cerrado adecuado $F \subsetneq E$ y $\varepsilon \in
\intoo{0}{1}$, existe un vector unitario $x$con $d(x, F) \geq 1 -
\varepsilon$. Prueba: elija $y \notin F$, deje $\delta = d(y, F) > 0$
($F$ cerrado), elija $f \in F$ con $\norm{y - f} \leq
\frac{\delta}{1 - \varepsilon}$y configure $x = \frac{y - f}{\norm{y -
f}}$: para cualquier $g \in F$,
\[
\norm{x - g} = \frac{\norm{y - (f + \norm{y-f}\,g)}}{\norm{y - f}}
\geq \frac{\delta}{\norm{y-f}} \geq 1 - \varepsilon ,
\]
siendo el numerador una distancia desde $y$ hasta un punto de $F$.

Ahora construya vectores unitarios $x_1, x_2, \dots$ de forma inductiva: $F_k =
\operatorname{Vect}(x_1, \dots, x_k)$ es de dimensión finita, por lo tanto
cerrado (\cref{cor:b2:nvs:closedsubspace}) y adecuado; Lema de Riesz
con $\varepsilon = \frac12$ proporciona una unidad $x_{k+1}$ con $d(x_{k+1},
F_k) \geq \frac12$. La secuencia satisface $\norm{x_p - x_q} \geq
\frac12$ para $p \neq q$: no hay subsecuencia convergente --- la bola unitaria
no es \omterm{def:b2:metric:compact}{compacto}.
\end{proof}

\begin{example}[Riesz como detector de dimensiones]\label{ex:b2:nvs:rieszdetector}
¿$C(\intcc01)$ es de dimensión finita? Riesz responde sin
exhibiendo cualquier familia libre infinita explícita: la secuencia
$f_n(x) = x^n$ se encuentra en la bola unitaria cerrada y satisface, por

en puntos adecuados --- cuantificados claramente en la descripción de este capítulo.
problema de fin de semana (pregunta 16), donde una subsecuencia permanece en

la pelota no es \omterm{def:b2:metric:compact}{compacto}, entonces $\dim C(\intcc01) = \infty$ por
\cref{thm:b2:nvs:riesz}. \omterm{def:b2:metric:compact}{Compacidad} de la bola unitaria es un
dicotomía perfecta: se mantiene en una dimensión finita, falla en
Dimensión infinita, sin término medio: solo geometría.
lee el tipo de dimensión.
\end{example}

\section{espacios de banach}

\begin{definition}\label{def:b2:nvs:banach}
Un \emph{espacio banach}\index{espacio banach} es un \omterm{def:b2:metric:complete}{completo} normalizado.
espacio. Ejemplos: todo espacio normado de dimensión finita
(\cref{thm:b2:nvs:finitedim}); $\bigl(C(\intcc{a}{b}),
\norm\cdot_\infty\bigr)$ (\cref{thm:b2:metric:rncomplete});
$\mathcal{L}_c(E, F)$ para $F$ Banach (mismo patrón de prueba que para
\omterm{def:b2:metric:continuity}{continuo} funciones). No ejemplo: $\bigl(C(\intcc{0}{1}),
\norm\cdot_1\bigr)$ (\cref{exo:b2:nvs:7}).
\end{definition}

\begin{example}[La norma de integración del operador.]\label{ex:b2:nvs:integrationnorm}
En $\bigl(C(\intcc01), \norm\cdot_\infty\bigr)$, sea $T(f)(x) =
\int_0^x f(t)\,\dd t $(un endomorfismo: $T(f)$ es \omterm{def:b2:metric:continuity}{continuo}).
Ejecute \cref{met:b2:nvs:opnorm}. Límite superior:
\[
\abs{T(f)(x)} \leq \int_0^x\abs f \leq x\,\norm f_\infty \leq
\norm f_\infty ,
\]
entonces $\vertiii T \leq 1$. Testigo: $f \equiv 1$ da $T(f)(x) =
x $y$\norm{T(f)}_\infty = 1 = \norm f_\infty$: alcanzado,
$\vertiii T = 1$. Pero tenga en cuenta $\vertiii{T^2} = \frac12 \neq
\vertiii T^2$: efectivamente $T^2(f)(x) = \int_0^x(x - t)f(t)\dd t$ tiene
$\abs{T^2(f)(x)} \leq \frac{x^2}2\norm f_\infty$, alcanzado nuevamente
en $f \equiv 1$; y en general $\vertiii{T^n} = \frac1{n!}$ ---
el límite submultiplicativo $\vertiii T^n = 1$ está desviado por un
factoriales. Este es exactamente el fenómeno del truco iterativo de
El problema del fin de semana de \cref{ch:b2:metric} se convierte en global
Solubilidad de ecuaciones diferenciales lineales.
\end{example}

\begin{theorem}[Convergencia absoluta en espacios de Banach]\label{thm:b2:nvs:absoluteconvergence}
En un \omterm{def:b2:nvs:banach}{espacio banach}, si $\sum \norm{u_n} < \infty$ entonces $\sum u_n$
converge, y $\norm{\sum u_n} \leq \sum\norm{u_n}$. (El completo
La teoría de series en espacios normados es \cref{ch:b2:series}.)
\end{theorem}

\begin{proof}
Sumas parciales $S_N$: para $q > p$, $\norm{S_q - S_p} \leq
\sum_{n=p+1}^{q}\norm{u_n}$, que tiende a $0$ (criterio de Cauchy
para la serie real de \omterm{def:b2:nvs:norm}{normas}): $(S_N)$ es Cauchy, por lo tanto convergente.
La desigualdad pasa al límite del triángulo finito.
desigualdad.
\end{proof}

\begin{example}[Matriz exponencial, primer contacto]\label{ex:b2:nvs:matrixexp}
El \emph{matriz exponencial}\index{matriz exponencial}:
$\mathcal{M}_n(K)$ con cualquier submultiplicativo \omterm{def:b2:nvs:norm}{norma} ($\vertiii{AB}
\leq \vertiii A \vertiii B$) es Banach (dimensión finita). Entonces, por
cada $A$,
\[
\eu^A = \sum_{k=0}^{\infty} \frac{A^k}{k!}
\]
converge absolutamente ($\vertiii{A^k/k!} \leq \vertiii A^k /k!$,
sumable): bien definido. \cref{ch:b2:diffeq} lo explota
sistemáticamente.
\end{example}

\begin{example}[El exponencial de un generador de rotacion.]\label{ex:b2:nvs:rotationexp}
Tome $A = \begin{psmallmatrix}0 & -\theta\\ \theta &

+ \Bigl(\sum_{j}\frac{(-1)^j\theta^{2j+1}}{(2j+1)!}\Bigr)

\cos\theta & -\sin\theta\\
\sin\theta & \cos\theta
\end{pmatrix},

\end{example}

\begin{remark}[Errores comunes]\label{rem:b2:nvs:pitfalls}
(i) Un \omterm{thm:b2:nvs:continuouslinear}{norma del operador} depende del \emph{ambos} elegido \omterm{def:b2:nvs:norm}{normas}: el
la misma matriz tiene $\vertiii\cdot_\infty$ dada por sumas de filas
(\cref{exo:b2:nvs:4}) y un $\vertiii\cdot_1$ diferente (columna
sumas); citando ``el'' \omterm{def:b2:nvs:norm}{norma} de una matriz sin nombrar el
\omterm{def:b2:nvs:norm}{normas} subyacente no tiene sentido. (ii) $\vertiii{AB} \leq
\vertiii A\,\vertiii B$ es una desigualdad, generalmente estricta ---
Los poderes pueden reducirse mucho más rápido que el límite $\vertiii A^k$.
sugiere, que es el objetivo del \omterm{def:b2:nvs:norm}{normas} adaptado (este
problema de fin de semana del capítulo, pregunta 22). (iii) ``Lineal implica
\omterm{def:b2:metric:continuity}{continuo}'' es un privilegio de dimensión finita: diferenciación
en polinomios es lineal y ilimitado
(\cref{ex:b2:nvs:operatornorms}). (iv) Convergencia absoluta de
$\sum u_n$ solo ayuda cuando el espacio es \omterm{def:b2:metric:complete}{completo}
(\cref{exo:b2:series:9} construye el contraejemplo). (v) En
dimensión infinita un supremo sobre la bola unitaria es un genuino
supremo: no asuma que se logra
(\cref{exo:b2:nvs:8}).
\end{remark}

\section{Problema: mejor aproximación y teorema de Chebyshev}

¿Qué tan bien se puede aproximar una función mediante polinomios de un determinado
grado, y ¿qué polinomio funciona mejor? Del lado de la existencia,
la respuesta pertenece a este capítulo: \omterm{def:b2:metric:compact}{compacidad} en finito
dimensión hace que existan las mejores aproximaciones. Del lado explícito,
un caso no trivial se puede resolver completamente con las manos desnudas ---
entre todos los polinomios \emph{mónico} de grado $n$, el de
el sup más pequeño \omterm{def:b2:nvs:norm}{norma} en $\intcc{-1}{1}$ es el Chebyshev (normalizado)
polinomio, de \omterm{def:b2:nvs:norm}{norma} $2^{1-n}$: \emph{El extremo de Chebyshev
teorema}. El problema prueba ambas partes y luego mide cuán gravemente
\omterm{def:b2:metric:compact}{compacidad} falla en dimensión infinita: la bola unitaria de
$C(\intcc{0}{1})$ contiene infinitas constelaciones de puntos en
distancia mutua $1$.

\begin{problem}[{Problema del fin de semana --- Teorema extremo de Chebyshev
y la geometria de la bola unitaria}]
\label{pb:b2:nvs:1}
\omterm{def:b2:nvs:norm}{Normas} sin subíndice son sup \omterm{def:b2:nvs:norm}{normas} en el segmento indicado.

\medskip
\noindent\textbf{Parte I --- Best approximation in normed
spaces.}
\begin{enumerate}
 \item Sea $F$ un subespacio de dimensión finita de una norma
 espacio $E$ y $x \in E$. Demuestre que la distancia $d(x,
 F) = \inf_{f \in F}\norm{x - f}$ es \emph{alcanzado}
 \emph{(reduce to a closed bounded subset of $F$ and use
 \cref{thm:b2:nvs:finitedim})}.
 \item A \omterm{def:b2:nvs:norm}{norma} es \emph{estrictamente convexo} cuando $\norm u = \norm v
 = 1$y $u \neq v $implican $\bigl\Vert\frac{u +
 v}2\bigr\Vert < 1$. Mostrar que $\norm\cdot_2$ en $\R^n$
 es estrictamente convexo \emph{(identidad del paralelogramo)}, y
 que $\norm\cdot_1$ y $\norm\cdot_\infty$ no son para
 $n \geq 2$.
 \item Demuestre que para un \omterm{def:b2:nvs:norm}{norma} estrictamente convexo, el mejor
 La aproximación de la pregunta 1 es \emph{único}.
 \item En $(\R^2, \norm\cdot_\infty)$, calcule todo lo mejor
 aproximaciones de $x = (0, 1)$ por la línea $F =
 \operatorname{Vect}\bigl((1,0)\bigr)$: un intervalo de
 minimizadores.
 \item En $\bigl(C(\intcc{a}{b}), \norm\cdot_\infty\bigr)$,
 demuestre que la mejor aproximación de $f$ por
 \emph{constantes} es único, igual a $c^* = \frac{\max f
 + \min f}{2}$, con distancia $\frac{\max f - \min
 f}{2}$; calcular ambos para $f(x) = x^2$ en
 $\intcc{0}{1}$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Chebyshev polynomials.}
\begin{enumerate}[resume]
 \item Muestra que hay exactamente un polinomio $T_n$ con
 $T_n(\cos\theta) = \cos n\theta$ para todos $\theta$
 \emph{(recurrence $T_{n+1} = 2XT_n - T_{n-1}$ from the
 cosine addition formula)}, ese $\deg T_n = n$, y ese
 su coeficiente principal es $2^{n-1}$ para $n \geq 1$.
 \item Mostrar $\abs{T_n} \leq 1$ en $\intcc{-1}{1}$, con
 $T_n(\eta_k) = (-1)^k$ en los puntos $n + 1$ $\eta_k =
 \cos\frac{k\pi}{n}$($k = 0, \dots, n$), y que el
 las raíces de $T_n$ son los puntos $n$
 $\cos\frac{(2k-1)\pi}{2n}$, entrelazando el $\eta_k$.
 \item Calcule $T_2, T_3, T_4$ y verifique la alternancia de
 $T_3$ en $\eta_0, \dots, \eta_3 = 1, \frac12, -\frac12,
 -1$ por evaluación directa.
 \item Para $\abs x \geq 1$, demostrar
 \[
 T_n(x) = \frac{\bigl(x + \sqrt{x^2 - 1}\bigr)^n +
 \bigl(x - \sqrt{x^2 - 1}\bigr)^n}{2},
 \]
 y deduzca $T_n(x) \sim \frac12\bigl(x + \sqrt{x^2 -
 1}\bigr)^n \to \infty $geométricamente para $x > 1$ fijo.
 \item Probar la ley de composición $T_m \circ T_n = T_{mn}$
 \emph{(check on $\intcc{-1}{1}$ and invoke the rigidity
 of polynomials)}.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Chebyshev's extremal theorem.}
Escribe $Q_n = 2^{1-n}T_n$ (mónico, por pregunta 6).
\begin{enumerate}[resume]
 \item Sea $P$ mónico de grado $n \geq 1$ con
 $\sup_{\intcc{-1}{1}}\abs P < 2^{1-n}$. Evaluando $D
 = Q_n - P$en los puntos $\eta_k$ y contando el signo
 cambios, derivan una contradicción. Concluir:
 \[
 \sup_{\intcc{-1}{1}}\abs P \;\geq\; 2^{1-n}
 \qquad\text{for every monic } P \text{ of degree } n.
 \]
 \item (Caso de igualdad) Supongamos $\sup_{\intcc{-1}{1}}\abs P =
 2^{1-n}$ con $P$monic de grado $n$, y sea $D = Q_n
 - P \neq 0$. Mostrar $(-1)^kD(\eta_k) \geq 0$ para todos los $k$;
 mostrar que cada uno de los intervalos $n$
 $\intcc{\eta_{k}}{\eta_{k-1}}$ contiene un cero de $D$,
 y que un cero compartido por dos intervalos consecutivos es
 un punto \emph{interior} $\eta_k$ donde $D' = 0$ como
 bueno. Concluya que $D$ tiene ceros $n$ contados
 multiplicidad, por lo tanto $D = 0$: el minimizador es exactamente
 $Q_n$ --- \emph{Teorema extremo de Chebyshev}.
 \item Reformule el teorema como una distancia: en $\intcc{-1}{1}$,
 \[
 d_\infty\bigl(X^n,\ \R_{n-1}[X]\bigr) = 2^{1-n},

\noindent\textbf{Parte IV --- The unit ball of
$C(\intcc{0}{1})$.}
\begin{enumerate}[resume]
 \item Vamos $g_k(x) = x^{2^k}$. Mostrar $\norm{g_k}_\infty = 1$ y
 $\norm{g_k - g_j}_\infty \geq \frac14$ para $j > k$
 \emph{(evaluate at the point where $x^{2^k} =
 \frac12$)}: una secuencia acotada explícita sin
 subsecuencia convergente --- la bola unitaria cerrada no es
 \omterm{def:b2:metric:compact}{compacto}, con las manos desnudas.
 \item (lema de Riesz, agudizado) Sea $F$ un
 \emph{de dimensión finita} subespacio propio de una norma
 espacio $E$. Usando la pregunta 1, genere un vector unitario $x$
 con $d(x, F) = 1$ exactamente --- no solo $\geq 1 -
 \varepsilon$ como en el lema de \cref{thm:b2:nvs:riesz}.
 \item Deducir: en todo espacio normado de dimensión infinita hay
 es una secuencia de vectores unitarios con distancias por pares
 $\geq 1$, y volver a derivar el teorema de Riesz a partir de él.
 \item En $C(\intcc{0}{1})$, exhibe dicha constelación.
 explícitamente: las funciones de la tienda $h_n$ son compatibles con
 $\bigl[\frac1{n+1}, \frac1n\bigr]$ con valor pico $1$.
 Verifique $\norm{h_n} = 1$, $\norm{h_n - h_m} = 1$ para $n
 \neq m $y tenga en cuenta que $h_n \to 0$ puntualmente pero no
 uniformemente.
 \item (La acotación total falla) Demuestre que la unidad de bola cerrada
 de $C(\intcc{0}{1})$ no puede ser cubierto por un número finito
 bolas de radio $\frac13$ \emph{(cada una de estas bolas contiene
 como máximo un $h_n$)} --- contrastan con las
 paso de acotación total en la prueba de
 \cref{thm:b2:metric:borellebesgue}.
\end{enumerate}

\noindent\textbf{Part V --- \omterm{def:b2:nvs:norm}{Normas} at work on matrices, and
synthesis.}
\begin{enumerate}[resume]
 \item Demuestre que cada \omterm{def:b2:reduction:eigen}{valor propio} $\lambda$ de $A \in
 \mathcal{M}_n(\C)$satisface $\abs\lambda \leq
 \vertiii A$ para cada \omterm{thm:b2:nvs:continuouslinear}{norma del operador}; aplicar
 \cref{exo:b2:nvs:4} para vincular el \omterm{def:b2:reduction:eigen}{valores propios} de
 $\begin{psmallmatrix}1 & -2\\ 3 & 1\end{psmallmatrix}$
 y comparar con su módulo verdadero.
 \item (Adaptado \omterm{def:b2:nvs:norm}{normas}) Sea $A$ \omterm{def:b2:reduction:diag}{diagonalizable}, $A =
 P\,\mathrm{diag}(\lambda_1, \dots, \lambda_n)\,P^{-1}$.
 Mostrar $N_P(x) = \norm{P^{-1}x}_\infty$ es un \omterm{def:b2:nvs:norm}{norma} cuyo
 \omterm{thm:b2:nvs:continuouslinear}{norma del operador} satisface $\vertiii A_{N_P} =
 \max_i\abs{\lambda_i}$.
 \item Deducir: para \omterm{def:b2:reduction:diag}{diagonalizable} $A$, $A^k \to 0$ si y sólo
 si todos \omterm{def:b2:reduction:eigen}{valores propios} satisfacen $\abs{\lambda_i} < 1$ ---
 \omterm{thm:b2:nvs:finitedim}{equivalencia de normas} llega a la conclusión
 independiente de las normas. Consulte $A =
 \frac14\begin{psmallmatrix}1 & 2\\ 2 &
 1\end{psmallmatrix}$.
 \item (Las constantes de equivalencia aumentan) En $\R_n[X]$, compare
 $N_c(P) = \max_k \abs{a_k}$ (coeficientes) y
 $\norm{P}_{\intcc{0}{1}}$: ambos son \omterm{def:b2:nvs:norm}{normas}, por lo que son
 \omterm{def:b2:nvs:equivalent}{equivalente} para cada $n$ fijo; pero muestra, usando el mónico
 minimizador de la pregunta 13 sobre $\intcc{0}{1}$, que el
 la mejor constante $C_n$ en $N_c \leq
 C_n\norm\cdot_{\intcc{0}{1}}$ satisface $C_n \geq
 2^{2n-1}$. Concluya en una oración por qué "todos los \omterm{def:b2:nvs:norm}{normas} son
 \omterm{def:b2:nvs:equivalent}{equivalente}'' muere en dimensión infinita.
 \item (Síntesis) Una frase cada uno: donde \omterm{def:b2:metric:compact}{compacidad} de
 bolas de dimensiones finitas trabajadas (preguntas 1, 12); que
 gobierna la convexidad estricta; cual es la constelación de
 las preguntas 18 y 19 destruyen; y como pregunta 24
 cuantifica el fracaso. Nombra la cumbre (Chebyshev's
 teorema extremo) y estado donde la mejor aproximación
 encuentra su hogar moderno (el teorema de proyección de Hilbert
 espacios, volumen del año 3, donde \omterm{def:b2:metric:complete}{lo completo} reemplaza
 \omterm{def:b2:metric:compact}{compacidad}).
\end{enumerate}
