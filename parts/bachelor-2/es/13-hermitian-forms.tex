\chapter{Formas hermitianas}\label{ch:b2:hermitian}

Los espacios vectoriales complejos tienen su propia geometría de producto interno, con
un giro: linealidad en una variable, \emph{conjugado}-linealidad en
el otro. La recompensa por aceptar el giro es una teoría espectral.
incluso más limpio que el real --- \omterm{def:b2:hermitian:adjoint}{Endomorfismos hermitianos} tiene real
\omterm{def:b2:reduction:eigen}{valores propios}, \omterm{def:b2:hermitian:adjoint}{unitario} tienen \omterm{def:b2:reduction:eigen}{valores propios} unimodular, y ambos
diagonalizar en bases ortonormales. Este breve capítulo recorre el
Programa euclidiano de \cref{ch:b2:quadratic} sobre $\C$.

\section{Productos internos hermitianos}

\begin{definition}\label{def:b2:hermitian:def}
A \emph{Producto interior hermitiano}\index{Producto interior hermitiano} en
un espacio vectorial complejo $E$ es un mapa $\langle\cdot,\cdot\rangle
\colon E \times E \to \C$ que es lineal en la segunda variable,
conjugado-simétrico ($\langle y, x\rangle = \conj{\langle x,
y\rangle}$ --- por lo tanto conjugado-lineal en la primera variable), y
positivo definido ($\langle x, x\rangle > 0$ para $x \neq 0$). el
ejemplo estándar en $\C^n$:
\[
\langle x, y \rangle = \sum_{i=1}^{n} \conj{x_i}\, y_i ;
\]
en funciones \omterm{def:b2:metric:continuity}{continuo}, $\langle f, g\rangle = \int_a^b \conj
f\,g$. \omterm{def:b2:nvs:norm}{Norma}: $\norm x = \sqrt{\langle x,x\rangle}$; un
El espacio complejo de dimensión finita así equipado es un \emph{hermitiano
espacio}\index{espacio hermitiano}.
\end{definition}

\begin{example}[primeros cálculos]\label{ex:b2:hermitian:firstcomp}
En $\C^2$, tome $x = (1+\iu,\ 2-\iu)$ y $y = (\iu,\ 1)$.
entonces
\[
\norm x^2 = \abs{1+\iu}^2 + \abs{2-\iu}^2 = 2 + 5 = 7,
\qquad
\norm y^2 = 1 + 1 = 2,
\]
y, conjugando el primer argumento,
\[
\langle x, y\rangle
= \conj{(1+\iu)}\,\iu + \conj{(2-\iu)}\cdot1
= (1-\iu)\iu + (2+\iu)
= (\iu + 1) + (2 + \iu) = 3 + 2\iu .
\]
Cauchy--Schwarz revisa: $\abs{\langle x, y\rangle}^2 = 9 +
4 = 13 \leq 14 = \norm x^2\norm y^2$ --- cerca de la igualdad,
porque $x$ está cerca de un múltiplo de $y$. Nota también
$\langle y, x\rangle = \conj{3 + 2\iu} = 3 - 2\iu$:
simetría conjugada en acción, y la razón $\langle x,
x\rangle$ es siempre real.
\end{example}

\begin{theorem}[Cauchy--Negro, complejo]\label{thm:b2:hermitian:cs}
$\abs{\langle x, y\rangle} \leq \norm x \norm y$, con igualdad y sii
$x, y$ son linealmente dependientes; $\norm\cdot$ es un \omterm{def:b2:nvs:norm}{norma}. Además
existen bases ortonormales (Gram-Schmidt corre textualmente), con
\[
x = \sum_i \langle e_i, x\rangle\, e_i,
\qquad
\norm x^2 = \sum_i \abs{\langle e_i, x\rangle}^2 .
\]
\end{theorem}

\begin{proof}
Para $y \neq 0$ y $t \in \C$: $0 \leq \norm{x - ty}^2 = \norm x^2
- 2\Re\bigl(\conj t\langle y, x\rangle\bigr) + \abs t^2\norm y^2$.
Elija $t = \frac{\langle y, x\rangle}{\norm y^2}$:
\[
0 \leq \norm x^2 - \frac{\abs{\langle y, x\rangle}^2}{\norm y^2},
\]
cuál es la desigualdad; fuerzas de igualdad $x = ty$. Triangulo
desigualdad, en su totalidad:
\[
\norm{x + y}^2 = \norm x^2 + 2\,\Re\langle x, y\rangle
+ \norm y^2
\leq \norm x^2 + 2\,\abs{\langle x, y\rangle} + \norm y^2
\leq \bigl(\norm x + \norm y\bigr)^2 ,
\]
usando $\Re z \leq \abs z$ y luego Cauchy--Schwarz; homogeneidad
y la separación son inmediatas, por lo que $\norm\cdot$ es un \omterm{def:b2:nvs:norm}{norma}.
Gram-Schmidt: como en el caso real, con conjugados
colocado por la definición (tenga en cuenta la convención: nuestros productos son
lineal conjugado en la ranura \emph{primero}, por lo que las coordenadas son
$\langle e_i, x\rangle$; el siguiente ejemplo ejecuta el algoritmo
una vez en su totalidad).
\end{proof}

\begin{example}[Gramo complejo--Schmidt, ejecutado completo]\label{ex:b2:hermitian:gramschmidt}
Ortonormalizar la base $v_1 = (1, \iu)$, $v_2 = (0, 1)$ de
$\C^2$. Primer vector: $\norm{v_1}^2 = \abs1^2 + \abs\iu^2 = 2$,
Entonces $e_1 = \frac{1}{\sqrt2}(1, \iu)$. Proyecto $v_2$ --- con el
conjugar en la primera ranura:
\[
\langle e_1, v_2\rangle
= \frac{1}{\sqrt2}\bigl(\conj{1}\cdot0 +
\conj{\iu}\cdot1\bigr)
= \frac{-\iu}{\sqrt2} ,
\qquad
v_2 - \langle e_1, v_2\rangle e_1
= (0,1) + \frac{\iu}{2}\,(1, \iu)
= \Bigl(\frac\iu2,\ \frac12\Bigr) .
\]
Su \omterm{def:b2:nvs:norm}{norma} es $\sqrt{\frac14 + \frac14} = \frac{1}{\sqrt2}$:
$e_2 = \frac{1}{\sqrt2}(\iu, 1)$. Verificar: $\langle e_1,
e_2\rangle = \frac12(\conj1\cdot\iu + \conj\iu\cdot1) =
\frac12(\iu - \iu) = 0$. Coordenadas de $v_2$ en la nueva base:
$v_2 = \langle e_1, v_2\rangle e_1 + \langle e_2, v_2\rangle
e_2$ con $\langle e_2, v_2\rangle = \frac{1}{\sqrt2}$ ---
Tenga en cuenta el orden: $\langle v_2, e_1\rangle$ le daría el
Coeficiente \emph{conjugado}. Información final: el algoritmo es
el euclidiano palabra por palabra; La única trampa es donde
la conjugación cae, y el cálculo $\norm{v_2 -
\text{proj}}^2 > 0$ usa silenciosamente la positividad --- el axioma
eso hace que toda la geometría funcione.
\end{example}

\section{Endomorfismos adjuntos, hermitianos y unitarios.}

\begin{definition}\label{def:b2:hermitian:adjoint}
El \emph{adjunto} $u^*$ de $u \in \mathcal{L}(E)$ está definido por
$\langle u^*(x), y\rangle = \langle x, u(y)\rangle$; en un
base ortonormal, $\operatorname{Mat}(u^*) = \conj{A}^{\mathsf T} =:
A^{\dagger}$(conjugado \omterm{def:b2:linalg:transpose}{transponer}) --- de hecho, si $B =
(b_{ij})$es la matriz de $u^*$ en base ortonormal
$(e_i)$, luego $b_{ij} = \langle e_i, u^*(e_j)\rangle$ y el
definir la identidad da
\[
\conj{b_{ij}}
= \langle u^*(e_j), e_i\rangle
= \langle e_j, u(e_i)\rangle = a_{ji} ,
\qquad\text{so}\qquad
B = \conj{A}^{\mathsf T} .
\] $u$ es
\emph{hermitiano}\index{endomorfismo hermitiano} cuando $u^* = u$
($A^\dagger = A$), \emph{unitario}\index{grupo unitario} cuando $u^*u =
\mathrm{id}$($A^\dagger A = I$: el grupo $U(n)$), \emph{normal}
cuando $u^*u = uu^*$.
\end{definition}

\begin{proposition}\label{prop:b2:hermitian:eigenvalues}
\omterm{def:b2:reduction:eigen}{Valores propios} de un \omterm{def:b2:hermitian:adjoint}{endomorfismo hermitiano} son \emph{real};
\omterm{def:b2:reduction:eigen}{valores propios} de un endomorfismo \omterm{def:b2:hermitian:adjoint}{unitario} tiene \emph{módulo $1$}; en
ambos casos \omterm{def:b2:reduction:eigen}{espacios propios} para distintos \omterm{def:b2:reduction:eigen}{valores propios} son ortogonales.
\end{proposition}

\begin{proof}
\omterm{def:b2:hermitian:adjoint}{hermitiano}, $u(x) = \lambda x$, $x \neq 0$:
\[
\lambda \norm x^2 = \langle x, u(x)\rangle = \langle u(x), x\rangle
= \conj{\langle x, u(x)\rangle} = \conj\lambda\,\norm x^2 ,
\]
entonces $\lambda \in \R$. \omterm{def:b2:hermitian:adjoint}{Unitario}: $\norm{u(x)} = \norm x$ (de $u^*u =
\mathrm{id}$), entonces $\abs\lambda\norm x = \norm x$. Ortogonalidad
(caso \omterm{def:b2:hermitian:adjoint}{hermitiano}): $\lambda\langle x, y\rangle = \langle u(x),
y\rangle = \langle x, u(y)\rangle = \mu\langle x, y\rangle$ para
\omterm{def:b2:reduction:eigen}{vectores propios} con $\lambda \neq \mu$ real. Caso \omterm{def:b2:hermitian:adjoint}{Unitario}, en
completo: para $u(x) = \lambda x$, $u(y) = \mu y$ con $\lambda \neq
\mu$ (ambos unimodulares),
\[
\langle x, y\rangle = \langle u(x), u(y)\rangle
= \conj\lambda\mu\,\langle x, y\rangle ,
\]
y $\conj\lambda\mu = \frac{\mu}{\lambda} \neq 1$: el factor
no es $1$, entonces $\langle x, y\rangle = 0$.
\end{proof}

\begin{example}[Una matriz sesgada-hermitiana, diagonalizada]\label{ex:b2:hermitian:skewexample}
$A = \begin{pmatrix} 0 & -2\\ 2 & 0\end{pmatrix}$ satisface
$A^\dagger = A^{\mathsf T} = -A$: \omterm{def:b2:hermitian:adjoint}{sesgado-hermitiano} (también real
antisimétrico --- sobre $\R$ no tiene \omterm{def:b2:reduction:eigen}{valores propios}).
\omterm{def:b2:reduction:charpoly}{Polinomio característico} $X^2 + 4$: \omterm{def:b2:reduction:eigen}{valores propios} $\pm2\iu$,
puramente imaginario, como predice \cref{exo:b2:hermitian:9} en
generales. \omterm{def:b2:reduction:eigen}{Vectores propios}: $(A - 2\iu I)v = 0$ da $v_1 =
\frac{1}{\sqrt2}(1, \iu)$y $v_2 = \frac{1}{\sqrt2}(1,
-\iu)$para$-2\iu$; son ortogonales:
\[
\langle v_1, v_2\rangle
= \tfrac12\bigl(\conj{1}\cdot1 +

\]
Entonces $A = U\operatorname{diag}(2\iu, -2\iu)\,U^\dagger$ con $U =
(v_1\ v_2)$\omterm{def:b2:hermitian:adjoint}{unitario}. Información final: $H = -\iu A =
\begin{pmatrix} 0 & 2\iu\\ -2\iu & 0\end{pmatrix}$ es \omterm{def:b2:hermitian:adjoint}{hermitiano}

\omterm{def:b2:reduction:eigen}{vectores propios} --- la biyección $u \mapsto \iu u$ entre

\begin{theorem}[Teorema espectral de Hermitiano]\label{thm:b2:hermitian:spectral}
Cada \omterm{def:b2:hermitian:adjoint}{endomorfismo hermitiano} de un \omterm{def:b2:hermitian:def}{espacio hermitiano} tiene una dirección ortonormal.
base de \omterm{def:b2:reduction:eigen}{vectores propios} (con \omterm{def:b2:reduction:eigen}{valores propios} real): $A^\dagger = A$
implica $A = U D U^{\dagger}$ con $U \in U(n)$ y $D$ reales
diagonal.\index{teorema espectral!hermitiano}
\end{theorem}

\begin{proof}
Sobre $\C$, el \omterm{def:b2:reduction:charpoly}{polinomio característico} se divide: un \omterm{def:b2:reduction:eigen}{vector propio}
$e_1$ existe (\cref{ch:b2:reduction}) --- no hay argumento \omterm{def:b2:metric:compact}{compacidad}
necesario, una ventaja de $\C$. Normalízalo. Su ortogonal
el complemento $F = e_1^\perp$ es estable: para $x \perp e_1$,
\[
\langle e_1, u(x)\rangle = \langle u(e_1), x\rangle
= \lambda_1\langle e_1, x\rangle = 0
\]
($\lambda_1$ reales). La restricción es \omterm{def:b2:hermitian:adjoint}{hermitiano}; inducción en el
dimensionar y concatenar. En detalle: la restricción $u|_F$ es
un endomorfismo del \omterm{def:b2:hermitian:def}{espacio hermitiano} $F$ (dimensión $n - 1$)
con $\langle u|_F(x), y\rangle = \langle x, u|_F(y)\rangle$
heredado de $u$; La hipótesis de la inducción proporciona una
base ortonormal $(e_2, \dots, e_n)$ de $F$ de \omterm{def:b2:reduction:eigen}{vectores propios},
y $(e_1, e_2, \dots, e_n)$ es ortonormal en $E$ ($e_1 \perp
F$) y está hecho de \omterm{def:b2:reduction:eigen}{vectores propios} de $u$. Traducción de matrices: el
Las columnas de $U$ son las $e_i$, $U^\dagger U = I$ expresa su
ortonormalidad, y $AU = UD$ recopila el \omterm{def:b2:reduction:eigen}{valor propio}
ecuaciones, de donde $A = UDU^\dagger$ con $D$ diagonal real
(\cref{prop:b2:hermitian:eigenvalues}).
\end{proof}

\begin{remark}[Errores comunes]
\emph{(i) Dónde cae la barra:} este libro conjuga la
Ranura \emph{primero}, por lo que las coordenadas son $\langle e_i, x\rangle$
y $\langle\lambda x, y\rangle = \conj\lambda\langle x,
y\rangle$; muchos textos conjugan la segunda ranura ---
traducir antes de comparar fórmulas o signos de $\iu$ ir
mal en silencio. \emph{(ii) Complex \omterm{def:b2:quadratic:def}{polarización} is stronger:}
sobre $\C$, si $\langle x, u(x)\rangle = 0$ para \emph{todo} $x$
luego $u = 0$ (expandir $x + y$ y $x + \iu y$: ambos el real
y partes imaginarias de $\langle x, u(y)\rangle$ desaparecen); encima
$\R$ esto falla --- la rotación por $\frac\pi2$ satisface
$\langle x, u(x)\rangle = 0$ en todas partes. En consecuencia, más
Sólo $\C$, ``$\langle x, u(x)\rangle \in \R$ para todos los $x$''
ya fuerza $u$ \omterm{def:b2:hermitian:adjoint}{hermitiano}. \emph{(iii) Real normal is not
\omterm{def:b2:reduction:diag}{diagonalizable}:} la matriz de
\cref{ex:b2:hermitian:skewexample} es normal pero no tiene real
\omterm{def:b2:reduction:eigen}{valor propio}; \omterm{def:b2:hermitian:adjoint}{unitario} la diagonalización es un teorema \emph{encima}
$\C$ y a partir de $\R$ solo se obtienen reducciones en bloque.
\emph{(iv) Comprobación de la unitaridad:} $U^\dagger U = I$ significa el
\emph{columnas} son ortonormales para el producto \omterm{def:b2:hermitian:adjoint}{hermitiano} ---
probando $UU^{\mathsf T}$, u olvidando la conjugación en el
productos de columna, son las dos formas clásicas de certificar un error
matriz.
\end{remark}

\begin{example}[Las isometrías son exactamente las unitarias.]\label{ex:b2:hermitian:isometry}
La preservación \omterm{def:b2:nvs:norm}{Norma} parece más débil que la unitaridad, pero más allá de $\C$
no es: si $\norm{u(x)} = \norm x$ para todos los $x$, entonces $u^*u =
\mathrm{id}$. De hecho $v = u^*u - \mathrm{id}$ es \omterm{def:b2:hermitian:adjoint}{hermitiano} y
satisface $\langle x, v(x)\rangle = \norm{u(x)}^2 - \norm x^2 =
0$ para cada $x$; por el complejo \omterm{def:b2:quadratic:def}{polarización} (obstáculo (ii)
arriba), un mapa con una ``diagonal'' que desaparece idénticamente es cero:
$v = 0$. En concreto, el \omterm{def:b2:quadratic:def}{polarización} funciona
\[
0 = \langle x + y, v(x+y)\rangle
= \langle x, v(y)\rangle + \langle y, v(x)\rangle,
\qquad
0 = \langle x + \iu y, v(x + \iu y)\rangle
= \iu\langle x, v(y)\rangle - \iu\langle y, v(x)\rangle ,
\]
y las dos líneas juntas fuerzan $\langle x, v(y)\rangle = 0$
para todos los $x, y$. Información final: esta es la razón por la que ``\omterm{def:b2:hermitian:adjoint}{unitario}'' puede
comprobarse midiendo longitudes únicamente --- rigidez que el
Capítulo de Fourier explotará, donde preservar la energía.
$\norm f_2$ (Parseval) es lo mismo que preservar todo el interior
productos de coeficientes.
\end{example}

La maquinaria \omterm{def:b2:hermitian:adjoint}{hermitiano} construida aquí se consume casi
inmediatamente. El capítulo de Fourier \emph{is} \omterm{def:b2:hermitian:adjoint}{hermitiano} geometría
en dimensión infinita: los exponenciales $(e_n)$ son un
familia ortonormal para $\langle f, g\rangle =
\frac{1}{2\pi}\int\conj fg$, la desigualdad de Bessel es la
estimación de proyección de este capítulo
\cref{thm:b2:hermitian:cs}, y Parseval es su limitante
igualdad. La transformada finita de Fourier
(\cref{exo:b2:hermitian:10}) reaparece cada vez que hay convolución
debe estar diagonalizado. Y el problema del fin de semana de este capítulo ---
Courant--Fischer, Weyl, entrelazado--- proporciona la
\omterm{def:b2:reduction:eigen}{valor propio} estabilidad que el capítulo de ecuaciones diferenciales
invoca cuando afirma que pequeñas perturbaciones de un sistema
mueve sus frecuencias sólo ligeramente. Al revés, todo aquí.
es el espejo complejo del capítulo de formas cuadráticas: mantenga la
dos diccionarios uno al lado del otro ($A^{\mathsf T} \leftrightarrow
A^\dagger$, ortogonal $\leftrightarrow$ \omterm{def:b2:hermitian:adjoint}{unitario}, Rayleigh
real en ambos).
\end{remark}

\begin{remark}[endomorfismos normales]
Sobre $\C$ la declaración definitiva es: $u$ es \emph{unitarily
\omterm{def:b2:reduction:diag}{diagonalizable} if and only if it is normal} ($u^*u = uu^*$) ---
cubriendo los mapas \omterm{def:b2:hermitian:adjoint}{hermitiano}, \omterm{def:b2:hermitian:adjoint}{unitario} y \omterm{def:b2:hermitian:adjoint}{sesgado-hermitiano} a la vez. el
La prueba es un agradable fortalecimiento del argumento anterior.
(\cref{exo:b2:hermitian:8}). Más de $\R$, por el contrario, sólo normalidad.
compra diagonalización de bloques (bloques de rotación): la geometría compleja es
genuinamente más simple.
\end{remark}

\begin{remark}[donde se utiliza]
\omterm{def:b2:hermitian:adjoint}{hermitiano} la teoría espectral es la matemática cuántica
mecánica: los observables son modelados por operadores \omterm{def:b2:hermitian:adjoint}{hermitiano}
(espectros reales = valores medibles), evolución temporal por \omterm{def:b2:hermitian:adjoint}{unitario}
unos (\omterm{def:b2:nvs:norm}{norma} preservación = conservación de probabilidad). dentro
En este libro, el capítulo de Fourier se basa en la ortonormalidad de
los exponenciales --- una declaración de producto interno \omterm{def:b2:hermitian:adjoint}{hermitiano} --- y
la diagonalización de matrices circulantes
(\cref{exo:b2:hermitian:10}) es la transformada finita de Fourier.
El problema del fin de semana desarrolla el cálculo variacional de
\omterm{def:b2:reduction:eigen}{valores propios} (Courant--Fischer, Weyl, entrelazado), el diario
pan de análisis numérico y física matemática; el año
3 volumen lo extiende a los operadores \omterm{def:b2:metric:compact}{compacto} \omterm{def:b2:hermitian:adjoint}{autoadjunto} en
Espacios de Hilbert.
\end{remark}

\begin{exercise}[$\star$]\label{exo:b2:hermitian:1}
En $\C^2$: calcule $\langle x, y\rangle$, $\norm x$, $\norm y$ para
$x = (1, \iu)$, $y = (\iu, 1)$; son ortogonales? dar un
base ortonormal que contiene $\frac{x}{\norm x}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:hermitian:2}
¿Cuáles son \omterm{def:b2:hermitian:adjoint}{hermitiano}? \omterm{def:b2:hermitian:adjoint}{unitario}? ¿normal?
\[
\begin{pmatrix} 1 & \iu\\ -\iu & 2 \end{pmatrix},
\qquad
\frac{1}{\sqrt2}\begin{pmatrix} 1 & \iu\\ \iu & 1\end{pmatrix},
\qquad
\begin{pmatrix} 0 & 1\\ 0 & 0 \end{pmatrix}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:hermitian:3}
Demuestre que una matriz $A \in \mathcal{M}_n(\C)$ escribe de forma única $A =
H + \iu K$con $H, K$ \omterm{def:b2:hermitian:adjoint}{hermitiano} \emph{(the ``real and imaginary
parts'' $H = \frac{A + A^\dagger}{2}$, $K = \frac{A -
A^\dagger}{2\iu}$)}, y que $A$es normal si $H$y $K$
conmutar.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:hermitian:4}
Diagonalice en forma ortonormal: $A = \begin{pmatrix} 2 & \iu\\
-\iu & 2\end{pmatrix}$y calcule $A^k $para $k \in \N$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:hermitian:5}
Demuestre que $U(n)$ es \omterm{def:b2:metric:compact}{compacto} y que el mapa \omterm{def:b2:reduction:eigen}{valor propio} es
en: cada $\lambda$ unimodular surge para alguna matriz \omterm{def:b2:hermitian:adjoint}{unitario}.
Demuestre que $\det U \in \mathbb{U}$ (el círculo unitario) para $U \in
U(n)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:hermitian:6}
(Transformada de Cayley) Sea $H$ \omterm{def:b2:hermitian:adjoint}{hermitiano}. Demuestre que $H + \iu I$ es
invertible y que $U = (H - \iu I)(H + \iu I)^{-1}$ es \omterm{def:b2:hermitian:adjoint}{unitario},
con $1 \notin \operatorname{Sp}(U)$. \emph{(Trabajar espectralmente: en un
base propia de $H$, todo es escalar.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:hermitian:7}
Para $A$ \omterm{def:b2:hermitian:adjoint}{hermitiano} positivo definido ($\langle x, Ax\rangle > 0$ para
$x \neq 0$), probar que $\operatorname{Sp}(A) \subseteq
\intoo{0}{\infty}$, que $A = B^2$ para un \omterm{def:b2:hermitian:adjoint}{hermitiano} positivo
definido $B$, y que $\det A > 0$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:hermitian:8}
(Teorema espectral para endomorfismos normales) Sea $u$ normal en un
\omterm{def:b2:hermitian:def}{espacio hermitiano}.
\begin{enumerate}
 \item Demuestre $\norm{u(x)} = \norm{u^*(x)}$ para todos los $x$ y deduzca
 $\ker(u - \lambda) = \ker(u^* - \conj\lambda)$.
 \item Demuestre que \omterm{def:b2:reduction:eigen}{espacios propios} de $u$ para distintos \omterm{def:b2:reduction:eigen}{valores propios} son
 ortogonal, y que el complemento ortogonal de un
 \omterm{def:b2:reduction:eigen}{espacio propio} es $u$ estable.
 \item Concluir por inducción que $u$ es unitariamente \omterm{def:b2:reduction:diag}{diagonalizable};
 y a la inversa.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:hermitian:9}
Un endomorfismo es \emph{\omterm{def:b2:hermitian:adjoint}{sesgado-hermitiano}} cuando $u^* = -u$. probar
que sus \omterm{def:b2:reduction:eigen}{valores propios} son puramente imaginarios, que $u \mapsto \iu
u$ es una biyección de \omterm{def:b2:hermitian:adjoint}{hermitiano} a \omterm{def:b2:hermitian:adjoint}{sesgado-hermitiano}
endomorfismos, y que los endomorfismos \omterm{def:b2:hermitian:adjoint}{sesgado-hermitiano} son
unitariamente \omterm{def:b2:reduction:diag}{diagonalizable} (\cref{exo:b2:hermitian:8}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:hermitian:10}
(La transformada finita de Fourier) Sea $S$ el desplazamiento \omterm{def:b2:structures:generated}{cíclico} de
$\C^n$: $S(x_0, x_1, \dots, x_{n-1}) = (x_{n-1}, x_0, \dots,
x_{n-2})$y $\omega = \eu^{2\iu\pi/n}$.
\begin{enumerate}
 \item Demuestre que $S$ es \omterm{def:b2:hermitian:adjoint}{unitario}, y que los vectores $f_k =
 \frac{1}{\sqrt n}\bigl(1, \omega^k, \omega^{2k}, \dots,
 \omega^{(n-1)k}\bigr)$, $0 \leq k < n$, forman un
 base ortonormal de \omterm{def:b2:reduction:eigen}{vectores propios}: $Sf_k = \omega^{-k}
 f_k$.
 \item Deduzca que toda matriz \emph{circulante} $C =
 \sum_{j=0}^{n-1} c_jS^j$ es normal, diagonalizada por
 misma base, con \omterm{def:b2:reduction:eigen}{valores propios} $\widehat c(k) = \sum_j
 c_j\,\omega^{-jk}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:hermitian:11}
Sea $P$ un endomorfismo idempotente ($P^2 = P$) de un \omterm{def:b2:hermitian:def}{hermitiano
espacio}. Demuestre que $P$ es la proyección \emph{ortogonal} sobre
$\operatorname{im} P$ si y sólo si $P^* = P$. dar la matriz
de la proyección ortogonal sobre $\C v$ ($\norm v = 1$), y
en un subespacio con base ortonormal $(v_1, \dots, v_k)$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:hermitian:12}
(Proyectores espectrales por interpolación) Sea $A$ \omterm{def:b2:hermitian:adjoint}{hermitiano} con
\emph{distinto} \omterm{def:b2:reduction:eigen}{valores propios} $\lambda_1, \dots, \lambda_p$ y
\omterm{def:b2:reduction:eigen}{espacio propio} descomposición $E = \bigoplus_i E_i$. Definir el
Polinomios de Lagrange $L_i(X) = \prod_{j\neq i}\frac{X -
\lambda_j}{\lambda_i - \lambda_j}$. Demuestre que $P_i = L_i(A)$
es la proyección ortogonal sobre $E_i$, que $P_iP_j = 0$ para
$i \neq j$, $\sum_i P_i = I$ y $A = \sum_i \lambda_iP_i$
(el \emph{descomposición espectral}); expresa $f(A)$ para cualquier
polinomio $f$ en términos del $P_i$.
\end{exercise}

\section{Problema: Courant--Fischer, Weyl y el cálculo de
valores propios}

\begin{problem}\label{pb:b2:hermitian:1}
Las \omterm{def:b2:reduction:eigen}{valores propios} de una matriz \omterm{def:b2:hermitian:adjoint}{hermitiano} no son solo raíces de una
polinomio: son soluciones de \emph{problemas de optimización}.
Ese punto de vista variacional --- \omterm{pb:b2:hermitian:1}{cocientes de Rayleigh} y el
\emph{Teorema mín-máx de Courant-Fischer}\index{Courant--Fischer
teorema} --- hace que \omterm{def:b2:reduction:eigen}{valores propios} sea comparable, estable y
computable, y este problema cosecha sus cosechas clásicas:
\emph{Desigualdades de perturbación de Weyl}\index{Weyl's
desigualdad}, \emph{entrelazado de cauchy}, el fan de Schur y Ky
trazar desigualdades, la monotonía de la raíz cuadrada de la matriz,
y el \omterm{def:b2:reduction:eigen}{espectro} del laplaciano discreto. En todo momento, $A, B,
E$son \omterm{def:b2:hermitian:adjoint}{hermitiano} en $E = \C^n$ y \omterm{def:b2:reduction:eigen}{valores propios} figuran en
orden decreciente $\lambda_1(A) \geq \dots \geq \lambda_n(A)$,
y $R_A(x) = \frac{\langle x, Ax\rangle}{\langle x, x\rangle}$
para $x \neq 0$ es el \emph{cociente de Rayleigh}\index{Rayleigh
cociente}.

\medskip
\noindent\textbf{Parte I --- \omterm{pb:b2:hermitian:1}{cocientes de Rayleigh} and min-max.}
Arreglar una base propia ortonormal $(e_1, \dots, e_n)$, $Ae_i =
\lambda_ie_i$.
\begin{enumerate}
 \item Demuestre que $R_A(x)$ es real y que
 \[
 \lambda_n \leq R_A(x) \leq \lambda_1
 \qquad (x \neq 0),
 \]
 ambos límites alcanzados: $\lambda_1 = \max R_A$,
 $\lambda_n = \min R_A$.
 \item Demuestre que los puntos críticos de $R_A$ son exactamente los
 \omterm{def:b2:reduction:eigen}{vectores propios} de $A$ \emph{(expand $t \mapsto R_A(x +
 tv)$at $t = 0$ for $v$arbitrary, then replace $v$ by
 $\iu v$)}.
 \item Deja que $V_k = \operatorname{Vect}(e_1, \dots, e_k)$ y
 $W_k = \operatorname{Vect}(e_k, \dots, e_n)$. Mostrar
 \[
 \min_{x \in V_k\setminus\{0\}} R_A(x) = \lambda_k
 = \max_{x \in W_k\setminus\{0\}} R_A(x) .
 \]
 \item Pruebe el \emph{Teorema de Courant-Fischer}: para $1 \leq
 k \leq n$,
 \[
 \lambda_k = \max_{\dim V = k}\;\min_{x \in
 V\setminus\{0\}} R_A(x)
 = \min_{\dim W = n-k+1}\;\max_{x \in
 W\setminus\{0\}} R_A(x)
 \]
 \emph{(for any $V$ of dimension $k$: $V \cap W_k \neq
 \{0\}$ by Grassmann, so $\min_V R_A \leq \lambda_k$;
 question 3 shows the bound is attained)}.
 \item (Monotonicidad) Escribe $A \leq B$ cuando $B - A$ sea
 semidefinido positivo. Deducir de la pregunta 4: $A \leq
 B$implica $\lambda_k(A) \leq \lambda_k(B)$ por cada
 $k$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Weyl's inequalities.}
\begin{enumerate}[resume]
 \item Demuestre que los subespacios $V, W \subseteq \C^n$ con $\dim V
 + \dim W > n$ se cruzan de manera no trivial y generalice:
 $\dim(V_1 \cap V_2 \cap V_3) \geq \dim V_1 + \dim V_2 +
 \dim V_3 - 2n$.
 \item Demuestre \emph{La desigualdad de Weyl}: para $i + j - 1 \leq n$,
 \[
 \lambda_{i+j-1}(A + B) \leq \lambda_i(A) +
 \lambda_j(B)
 \]
 \emph{(intersect the subspaces $W_i(A)$, $W_j(B)$ and
 $V_{i+j-1}(A+B)$ of question 3 and count
 dimensions)}.
 \item Definir $\vertiii{E}_2 = \max_{\norm x = 1}\norm{Ex}$
 y mostrar $\vertiii E_2 = \max_k\abs{\lambda_k(E)}$ para
 \omterm{def:b2:hermitian:adjoint}{hermitiano}$E$. Deducir \emph{La perturbación de Weyl
 teorema}:
 \[
 \bigl|\lambda_k(A + E) - \lambda_k(A)\bigr| \leq
 \vertiii{E}_2
 \qquad (1 \leq k \leq n) :
 \]
 cada \omterm{def:b2:reduction:eigen}{valor propio} es una función $1$-Lipschitz del
 matriz.
 \item (Perturbaciones de rango uno) Sea $P$ \omterm{def:b2:hermitian:adjoint}{hermitiano} positivo
 semidefinido de rango $1$. Mostrar
 \[
 \lambda_k(A) \leq \lambda_k(A + P) \leq
 \lambda_{k-1}(A) \qquad (2 \leq k \leq n),
 \]
 junto con $\lambda_1(A) \leq \lambda_1(A+P)$: el
 nuevos \omterm{def:b2:reduction:eigen}{valores propios} \emph{entrelazar} los antiguos.
 \item Marque la pregunta 8 numéricamente: $A = \begin{pmatrix} 2 &
 \iu\\ -\iu & 2\end{pmatrix}$
 (\cref{exo:b2:hermitian:4}: \omterm{def:b2:reduction:eigen}{espectro} $\{3, 1\}$) y
 $E = \begin{pmatrix} 0 & 1\\ 1 & 0\end{pmatrix}$
 (\omterm{def:b2:reduction:eigen}{espectro} $\{1, -1\}$): calcula el \omterm{def:b2:reduction:eigen}{espectro} de $A +
 E$ y ambos lados de la desigualdad.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Interlacing and trace
inequalities.}
\begin{enumerate}[resume]
 \item (entrelazado de Cauchy) Sea $B$ la submatriz principal $(n-1)
 \times(n-1)$de $A$. probar
 \[
 \lambda_{k+1}(A) \leq \lambda_k(B) \leq \lambda_k(A)
 \qquad (1 \leq k \leq n-1)
 \]
 \emph{(view $\C^{n-1} \subseteq \C^n$; on it, $R_B$ is
 the restriction of $R_A$; apply Courant--Fischer on
 both levels)}.
 \item Iterar: para una submatriz principal $B$ de tamaño $n - m$,
 $\lambda_{k+m}(A) \leq \lambda_k(B) \leq
 \lambda_k(A)$.
 \item (Schur) Sea $d_1 \geq d_2 \geq \dots \geq d_n$ el
 Entradas diagonales de $A$, ordenadas. Demuestre, para cada $k$:
 \[
 \sum_{i=1}^{k} d_i \leq \sum_{i=1}^{k}\lambda_i(A),
 \]
 con igualdad en $k = n$ (la traza)
 \emph{(the $k$ chosen diagonal entries form a
 principal $k\times k$ submatrix; bound its trace by
 question 12)}.
 \item (Ky Fan) Demuestra
 \[
 \sum_{i=1}^{k}\lambda_i(A)
 = \max\Bigl\{\sum_{i=1}^{k}\langle x_i, Ax_i\rangle
 : (x_1, \dots, x_k) \text{ orthonormal}\Bigr\} .
 \]
 \item Verifique las preguntas 11 y 13 del
 \[
 A = \begin{pmatrix} 2 & 1 & 0\\ 1 & 2 & 1\\
 0 & 1 & 2\end{pmatrix}
 \qquad
 \bigl(\operatorname{Sp} = \{2 - \sqrt2,\ 2,\
 2+\sqrt2\}\bigr)
 \]
 contra su bloque líder $2\times2$ (\omterm{def:b2:reduction:eigen}{espectro} $\{1,
 3\}$) y su diagonal.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The Loewner order.}$A \leq B$
todavía significa $B - A$ semidefinido positivo; todas las matrices en este
parte son \omterm{def:b2:hermitian:adjoint}{hermitiano}.
\begin{enumerate}[resume]
 \item Mostrar: $A \leq B$ implica $a_{ii} \leq b_{ii}$ para todos
 $i$, $\operatorname{tr} A \leq \operatorname{tr} B$,
 y $C^\dagger AC \leq C^\dagger BC$ para \emph{cada}
 matriz compleja $C$.
 \item Demuestre que elevar al cuadrado es \emph{no} monótono: para
 \[
 A = \begin{pmatrix} 1 & 0\\ 0 & 0\end{pmatrix},
 \qquad
 B = \begin{pmatrix} 2 & 1\\ 1 & 1\end{pmatrix},
 \]
 marque $0 \leq A \leq B$ pero $A^2 \not\leq B^2$.
 \item Demuestre que la raíz cuadrada \emph{is} monótona: $0 \leq
 A \leq B$implica $\sqrt A \leq \sqrt B$ \emph{(let
 $\mu$ be an \omterm{def:b2:reduction:eigen}{valor propio} of $\sqrt B - \sqrt A$ with unit
 \omterm{def:b2:reduction:eigen}{vector propio} $v$; compute $\langle v, (B - A)v\rangle =
 \mu\bigl(\langle v, \sqrt B\,v\rangle + \langle v,
 \sqrt A\,v\rangle\bigr)$ and discuss)}.
 \item Demuestre que la inversión es antítono sobre definida positiva
 matrices: $0 < A \leq B$ implica $B^{-1} \leq A^{-1}$
 \emph{(congruate by $A^{-1/2}$ to reduce to $I \leq M
 \Rightarrow M^{-1} \leq I$, which is scalar in a
 spectral basis)}.
 \item Sea $A, B$ positivo definido. Demuestre que el
 \omterm{def:b2:reduction:eigen}{valores propios} de $AB$ (¡no \omterm{def:b2:hermitian:adjoint}{hermitiano} en general!) son
 real y positivo, y que
 \[
 \lambda_{\max}(AB) \leq
 \lambda_{\max}(A)\,\lambda_{\max}(B)
 \]
 \emph{(conjugate by $\sqrt A$: $AB \sim \sqrt
 A\,B\sqrt A$)}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The discrete Laplacian, worked.}
Sea $T_n$ la matriz tridiagonal $n \times n$ con $2$ en
la diagonal y $-1$ en las dos diagonales adyacentes.
\begin{enumerate}[resume]
 \item Con $\theta_k = \frac{k\pi}{n+1}$, verifique que el
 vectores $v_k = \bigl(\sin(j\theta_k)\bigr)_{1\leq j\leq
 n}$ satisfacen $T_nv_k = (2 - 2\cos\theta_k)\,v_k$
 \emph{(identidad producto-suma; verifique las filas límite
 $j = 1, n$)}. Concluir:
 \[
 \operatorname{Sp}(T_n) = \Bigl\{4\sin^2
 \frac{k\pi}{2(n+1)} : 1 \leq k \leq n\Bigr\},
 \]
 Todo simple, todo positivo: $T_n$ es positivo definido.
 \item (potencial A) Para una diagonal real $D =
 \operatorname{diag}(d_1, \dots, d_n)$, intercale el
 \omterm{def:b2:reduction:eigen}{espectro}: por cada $k$,
 \[
 \lambda_k(T_n) + \min_i d_i \;\leq\;
 \lambda_k(T_n + D) \;\leq\; \lambda_k(T_n) + \max_i
 d_i .
 \]
 \item Verifique el entrelazado de Cauchy entre $T_3$ y $T_2$
 explícitamente (espectros $\{2 \pm \sqrt2, 2\}$ y $\{1,
 3\}$), e interpreta: $T_2$ es $T_3$ con un punto final
 del camino eliminado.
 \item Muestra la satisfacción extrema \omterm{def:b2:reduction:eigen}{valores propios}, como $n \to
 \infty$:
 \[
 \lambda_{\min}(T_n) = 4\sin^2\frac{\pi}{2(n+1)}
 \sim \frac{\pi^2}{(n+1)^2},
 \qquad
 \lambda_{\max}(T_n) \to 4 ,
 \]
 entonces el número de condición $\kappa_n =
 \lambda_{\max}/\lambda_{\min}$ crece como
 $\frac{4(n+1)^2}{\pi^2}$: discretizando un segundo
 derivada en una cuadrícula cada vez más fina es
 intrínsecamente mal condicionado.
 \item Síntesis. En una oración cada uno: (i) por qué el
 La caracterización variacional, no el \omterm{def:b2:reduction:charpoly}{característica
 polinomio}, es lo que hace que \omterm{def:b2:reduction:eigen}{valores propios} sea estable.
 (preguntas 8--9); (ii) qué preguntas se utilizaron únicamente
 $\lambda_1 = \max R_A$ y cuál necesitaba el completo
 mín-máx; (iii) lo que la orden Loewner añade a la
 historia; (iv) dónde reaparecen estas herramientas (números
 análisis de las matrices de rigidez de la pregunta 24; cuántico
 teoría de la perturbación; y, en el volumen del año 3, el
 Principio mínimo-máximo para \omterm{def:b2:metric:compact}{compacto} \omterm{def:b2:hermitian:adjoint}{autoadjunto}
 operadores).
\end{enumerate}
\end{problem}
