\chapter{Secuencias y Series}\label{ch:b2:series}

La teoría de las series numéricas (volumen del año 1) madura aquí en tres
direcciones: serie con valores en \omterm{def:b2:nvs:banach}{espacios de banach}, donde absoluto
la convergencia hace el trabajo; las pruebas más finas para series reales (\omterm{thm:b2:series:abel}{abel
sumatoria}); y \emph{\omterm{def:b2:series:summable}{sumable} families} --- suma liberada de
el orden de los términos --- con el teorema de Fubini para sumas dobles
y el \omterm{thm:b2:series:fubini}{producto cauchy}. Estas herramientas llevan toda la serie de funciones.
capítulos por delante.

\section{Series en espacios normados}

\begin{definition}\label{def:b2:series:def}
Para una secuencia $(u_n)$ en un espacio normado $E$, la serie $\sum u_n$
converge cuando lo hacen sus sumas parciales; converge
\emph{absolutamente}\index{convergencia absoluta!en un espacio de Banach} cuando
$\sum \norm{u_n} < \infty$. En un espacio \emph{Banach}, absoluto
convergencia implica convergencia (\cref{thm:b2:nvs:absoluteconvergence});
en un espacio \omterm{def:b2:metric:complete}{incompleto} esto puede fallar
(\cref{exo:b2:series:9}).
\end{definition}

\begin{example}\label{ex:b2:series:neumann}
En $\mathcal{M}_n(K)$ (o $\mathcal{L}_c(E)$, $E$ Banach): para
$\vertiii A < 1$, el \emph{serie neumann} $\sum A^k$ converge
\omterm{def:b2:series:def}{absolutamente} a $(I - A)^{-1}$ (probado en \cref{exo:b2:nvs:5});
$\sum \frac{A^k}{k!}$ converge \omterm{def:b2:series:def}{absolutamente} a $\eu^A$ por cada $A$
(\cref{ex:b2:nvs:matrixexp}). Geométrico valorado por el operador y
las series exponenciales se comportan como sus modelos escalares --- el conjunto
punto del marco de Banach.
\end{example}

\section{resumen de abel}

\begin{theorem}[Resumen y prueba de Abel]\label{thm:b2:series:abel}
(Suma por partes) Para escalares $a_n$ y vectores $b_n$, con $B_n
= \sum_{k=0}^{n} b_k$:\index{resumen de abel}
\[
\sum_{n=0}^{N} a_n b_n
= a_N B_N - \sum_{n=0}^{N-1} (a_{n+1} - a_n) B_n .
\]
(Prueba de Abel) Si $(a_n)$ es una secuencia real, disminuyendo a $0$, y
las sumas parciales $B_n$ son \emph{encerrado} (en un \omterm{def:b2:nvs:banach}{espacio banach}), entonces
$\sum a_n b_n$ converge.
\end{theorem}

\begin{proof}
La identidad, paso a paso: con $B_{-1} = 0$, escribe $b_n = B_n -
B_{n-1}$ y divide,
\[
\sum_{n=0}^{N} a_nb_n
= \sum_{n=0}^{N} a_nB_n - \sum_{n=0}^{N} a_nB_{n-1}
= \sum_{n=0}^{N} a_nB_n - \sum_{n=0}^{N-1} a_{n+1}B_{n} ,
\]
reindexando la segunda suma por $n \mapsto n + 1$ (el $B_{-1}$
término desaparece); recogiendo el rango común $0 \leq n \leq N-1$
deja $a_NB_N$ más $\sum_{n\leq N-1}(a_n - a_{n+1})B_n$: el
fórmula indicada. Es la integración discreta por partes, con
$(B_n)$ como antiderivada de $(b_n)$ y la diferencia $a_{n+1}
- a_n $como derivada de $(a_n)$. Para la prueba, con $\norm{B_n}
\leq M$: el
término límite $a_N B_N \to 0$; la serie $\sum (a_n -
a_{n+1})B_n$ converge \omterm{def:b2:series:def}{absolutamente}, ya que
\[
\sum_n \norm{(a_{n+1} - a_n)B_n} \leq M \sum_n (a_n - a_{n+1})
= M a_0 < \infty
\]
(telescópico, $a_n \downarrow 0$). Ambas piezas de la identidad.
convergen, por lo tanto también lo hace $\sum a_n b_n$.
\end{proof}

\begin{example}\label{ex:b2:series:sinn}
$\sum \frac{\sin n}{n}$ converge: $a_n = \frac1n \downarrow 0$ y
$B_n = \sum_{k=1}^{n} \sin k$ está acotado --- de hecho $B_n =
\Im\sum_{k \leq n} \eu^{\iu k} = \Im\,\frac{\eu^{\iu}(\eu^{\iu n} -
1)}{\eu^{\iu} - 1}$, de módulo $\leq \frac{2}{\abs{\eu^{\iu} - 1}}$.
\emph{no} converge, \omterm{def:b2:series:def}{absolutamente} ($\abs{\sin n} \geq \sin^2 n
= \frac{1 - \cos 2n}{2}$y $\sum \frac{1 - \cos 2n}{2n}$ divergen).
ya que $\sum \frac{\cos 2n}{n}$ converge por la misma prueba de Abel mientras que
$\sum \frac{1}{2n}$ diverge). La prueba de series alternas es la
caso especial $b_n = (-1)^n$.
\end{example}

\begin{example}[Abel en el círculo de convergencia]\label{ex:b2:series:abelboundary}
¿Para qué complejo $z$ con $\abs z = 1$ converge $\sum_{n \geq 1}
\frac{z^n}{n}$? En $z = 1$ es la serie armónica:
divergente. Para $z \neq 1$ en el círculo, se aplica la prueba de Abel.
con $a_n = \frac1n \downarrow 0$ y $b_n = z^n$, cuyos parciales
las sumas están limitadas independientemente de $N$:
\[
\Bigl|\sum_{n=1}^{N} z^n\Bigr|
= \Bigl|\frac{z(z^N - 1)}{z - 1}\Bigr|
\leq \frac{2}{\abs{z - 1}} .
\]
Convergente --- aunque nunca \omterm{def:b2:series:def}{absolutamente} ($\sum\frac1n$). uno
serie, un círculo de comportamientos: divergencia en un solo punto,
semiconvergencia en todos los demás lugares. Este es el límite estándar
comportamiento de la serie de potencias (\cref{ch:b2:powerseries}), conocido aquí
con las manos desnudas; en $z = -1$ recupera la alternancia
serie armónica, y en $z = \eu^{\iu\theta}$ es real y
Las piezas imaginarias son las series $\sum\frac{\cos
n\theta}{n}$y $\sum\frac{\sin n\theta}{n}$ de
\cref{exo:b2:series:4}.
\end{example}

\begin{example}[Una serie alternativa de trampas explosivas.]\label{ex:b2:series:trap}
¿Converge $\sum_{n\geq2} \dfrac{(-1)^n}{\sqrt n + (-1)^n}$?
Los signos se alternan y los términos tienden a $0$ --- sin embargo el
prueba alterna se aplica \emph{no}: los módulos
$\frac{1}{\sqrt n + (-1)^n}$ no están disminuyendo (salta en
cada impar $n$). Expandir en su lugar:
\[
\frac{(-1)^n}{\sqrt n + (-1)^n}
= \frac{(-1)^n}{\sqrt n}\cdot
\frac{1}{1 + \frac{(-1)^n}{\sqrt n}}
= \frac{(-1)^n}{\sqrt n} - \frac{1}{n}
+ O\Bigl(\frac{1}{n^{3/2}}\Bigr).
\]
La primera pieza converge (prueba alternada, honestamente aplicada a
$\frac1{\sqrt n}\downarrow0$), el tercero converge \omterm{def:b2:series:def}{absolutamente}
--- pero la pieza intermedia es la serie armónica divergente: la
suma \emph{diverge} a $-\infty$. La idea final: cuando
la monotonicidad falla, expanda hasta que cada pieza sea
\omterm{def:b2:series:def}{absolutamente} caso de prueba convergente o limpio; lo escondido
$-\frac1n$ es invisible para el recuento de signos.
\end{example}

\begin{remark}[Errores comunes]\label{rem:b2:series:pitfalls}
(i) ``Los términos tienden a $0$'' no prueba nada: la serie armónica
diverge. (ii) La prueba alterna requiere \emph{decreciente}
módulos --- \cref{ex:b2:series:trap} es el canónico
contraejemplo, y la tercera serie de \cref{exo:b2:series:1} la
taladro. (iii) Las series condicionalmente convergentes pueden no ser
reorganizados (\cref{ex:b2:series:rearrange}), y su \omterm{thm:b2:series:fubini}{cauchy
productos} puede divergir: para $\sum\frac{(-1)^n}{\sqrt{n+1}}$
al cuadrado, los términos de la diagonal satisfacen
\[
\abs{c_k} = \sum_{m=0}^{k}
\frac{1}{\sqrt{(m+1)(k-m+1)}}
\geq (k+1)\cdot\frac{2}{k+2} \longrightarrow 2 \neq 0
\]
(cada factor es como máximo $\frac{k+2}2$ por AM--GM), por lo que $\sum c_k$
diverge --- convergencia absoluta de al menos un factor
(\cref{exo:b2:series:8}) no es un lujo. (iv) \omterm{def:b2:series:summable}{Sumabilidad} es
sobre los límites \emph{absoluto} por definición: no existe tal
cosa como condicionalmente \omterm{def:b2:series:summable}{familia sumable}.
\end{remark}

\section{familias sumables}

\begin{definition}\label{def:b2:series:summable}
Sea $I$ un conjunto de índices \omterm{def:b2:structures:countable}{contable}. Una familia $(u_i)_{i \in I}$ de
\emph{reales no negativos} es \emph{sumable}\index{familia sumable}
cuando las sumas parciales finitas están acotadas; su suma es
\[
\sum_{i \in I} u_i = \sup_{F \subseteq I \text{ finite}} \sum_{i
\in F} u_i \in \intcc{0}{+\infty} .
\]
Una familia de reales o complejos (o vectores de Banach) es sumable cuando
$(\norm{u_i})$ es; su suma se define luego dividiéndola en
partes positivas/negativas (o reales/imaginarias) --- de manera equivalente, como
valor común de $\sum_{n} u_{\sigma(n)}$ en todas las enumeraciones
$\sigma$ de $I$ (ver más abajo).
\end{definition}

\begin{method}[Elegir una prueba]\label{met:b2:series:choosing}
Frente a $\sum u_n$, en orden: (1) si $u_n \not\to 0$, divergencia,
detenerse. (2) Si los términos tienen signo constante, compare: encuentre un
equivalente (\cref{ch:b2:comparison}) y colóquelo en el
Mapa de Riemann-Bertrand. (3) Si los signos se alternan con
Módulos \emph{decreciente}, la prueba alterna; si los módulos
no son monótonos, expanda el término hasta que cada pieza sea \omterm{def:b2:series:def}{absolutamente}
caso de prueba convergente o limpio (\cref{ex:b2:series:trap}). (4)
Si el patrón de signos es oscilatorio pero estructurado ($\sin
n\theta$, $\eu^{\iu n\theta}$, potencias matriciales), la prueba de Abel con
sumas parciales acotadas. (5) La convergencia absoluta siempre vale la pena
comprobando primero: es más fuerte, a prueba de pedidos y desbloquea \omterm{thm:b2:series:fubini}{cauchy
productos} y Fubini.
\end{method}

\begin{example}[Sumabilidad por conteo diagonal]\label{ex:b2:series:diagonalcount}
Para cual $s > 0$ es la familia
$\bigl((m + n)^{-s}\bigr)_{m, n \geq 1}$\omterm{def:b2:series:summable}{sumable}? Agrupa el
sumas parciales finitas por diagonales $m + n = k$: la diagonal $k$
lleva pares $k - 1$, cada uno de los cuales contribuye con $k^{-s}$, por lo que el
las sumas finitas están exactamente limitadas por (y agotadas)
\[
\sum_{k \geq 2} \frac{k - 1}{k^{s}} ,
\]
una serie con términos positivos equivalente a $k^{1-s}$: \omterm{def:b2:series:summable}{sumable}
si es $s - 1 > 1$, es decir\ $s > 2$. El índice bidimensional come.
una potencia completa: un plano de términos es "una dimensión más
divergente'' que una línea --- la geometría de conteo del índice
establece, no el tamaño de los términos individuales, decide \omterm{def:b2:series:summable}{sumabilidad}.
(El mismo censo muestra que $\bigl((m^2 + n^2)^{-1}\bigr)$ no es
\omterm{def:b2:series:summable}{sumable}: en la diagonal $m + n = k$, cada término es al menos
$k^{-2}$ y $(k-1)\cdot k^{-2}$ suman como el armónico
serie.)
\end{example}

\begin{theorem}[sumabilidad y orden]\label{thm:b2:series:rearrangement}
\begin{enumerate}
 \item Para familias no negativas, la suma es invariante bajo cualquier
 enumeración: $\sum_{i} u_i = \sum_{n=0}^{\infty}
 u_{\sigma(n)}$ para cada biyección $\sigma \colon \N \to I$.
 \item Una serie real o compleja $\sum u_n$ es \emph{conmutativamente
 convergente} (todo reordenamiento converge, con el mismo
 suma) si y sólo si es \omterm{def:b2:series:def}{absolutamente} convergente.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Toda suma parcial $\sum_{n \leq N} u_{\sigma(n)}$ es finita
suma parcial de la familia (entonces $\leq$ el sup); todo finito $F$ es
contenido en algunos $\{\sigma(0), \dots, \sigma(N)\}$ (por lo que el sup
$\leq$ el límite de la serie). Los dos límites coinciden.

(2) Si $\sum\abs{u_n} < \infty$: para cualquier reordenamiento $\sigma$ y
$\varepsilon > 0$, elija $N$ con $\sum_{n > N}\abs{u_n} \leq
\varepsilon$; más allá del rango donde $\sigma$ ha agotado
$\intint{0}{N}$, las sumas parciales reordenadas difieren de las
límite original en como máximo $\varepsilon$: misma suma. Si $\sum
\abs{u_n} = \infty $pero $\sum u_n$ convergen (caso real; complejo
sigue en forma de coordenadas): las partes positiva y negativa, ambas
divergen, y uno puede reorganizarse para alcanzar cualquier límite prescrito
--- Teorema de Riemann, realizado en \cref{exo:b2:series:5} ---
entonces la convergencia conmutativa falla.
\end{proof}

\begin{example}[Un reordenamiento pillado con las manos en la masa]\label{ex:b2:series:rearrange}
La serie armónica alterna suma
$\sum_{n\geq1}\frac{(-1)^{n-1}}{n} = \ln 2$ (Año 1
volumen). Reorganícelo como "uno positivo, dos negativos":
\[
1 - \frac12 - \frac14 + \frac13 - \frac16 - \frac18 + \frac15 -
\cdots
\]
Agrupando cada bloque de tres,
\[
\frac{1}{2k-1} - \frac{1}{4k-2} - \frac1{4k}
= \frac{1}{4k-2} - \frac{1}{4k}
= \frac12\Bigl(\frac{1}{2k-1} - \frac1{2k}\Bigr),
\]
entonces la serie reordenada converge a $\frac12\ln 2$ --- la mitad de la
suma original, con exactamente los mismos términos. \omterm{def:b2:series:def}{No absolutamente}
las series convergentes recuerdan el orden de sus términos; \omterm{def:b2:series:summable}{sumable}
las familias son precisamente las que no lo hacen.
\end{example}

\begin{example}[Agrupar es seguro, desagrupar no]\label{ex:b2:series:grouping}
Agrupar términos consecutivos de una serie \emph{convergente} nunca
cambia la suma: las sumas parciales agrupadas forman una subsecuencia de
los originales. Está prohibida la operación inversa:
\[
(1 - 1) + (1 - 1) + (1 - 1) + \cdots = 0 + 0 + \cdots = 0,
\]
sin embargo, el $1 - 1 + 1 - 1 + \cdots$ desagrupado diverge (parcialmente
las sumas oscilan entre $1$ y $0$). La desagrupación es legítima
sólo con una hipótesis compensatoria --- por ejemplo, términos
tendiendo a $0$ con longitudes de grupo acotadas: luego entre dos
sumas parciales agrupadas las originales derivan como máximo en una suma de
$o(1)$ términos de número acotado y transferencias de convergencia hacia atrás.
Ésa es exactamente la cláusula bajo la cual el cálculo en bloque de
\cref{ex:b2:series:rearrange} es una prueba y no una prestidigitación
mano.
\end{example}

\begin{theorem}[Fubini para familias; Productos Cauchy]\label{thm:b2:series:fubini}
Sea $(u_{m,n})_{(m,n) \in \N^2}$ una familia doble \omterm{def:b2:series:summable}{sumable} (es decir,\
$\sup_F \sum_F \abs{u_{m,n}} < \infty$). Entonces\index{teorema de fubini
para serie}
\[
\sum_{(m,n)} u_{m,n}
= \sum_{m=0}^{\infty}\Bigl(\sum_{n=0}^{\infty} u_{m,n}\Bigr)
= \sum_{n=0}^{\infty}\Bigl(\sum_{m=0}^{\infty} u_{m,n}\Bigr)
= \sum_{k=0}^{\infty}\Bigl(\sum_{m+n=k} u_{m,n}\Bigr),
\]
todas las series internas convergentes (\omterm{def:b2:series:def}{absolutamente}). En particular, si $\sum
a_m $y$\sum b_n$ convergen \omterm{def:b2:series:def}{absolutamente}, su \emph{cauchy
producto}\index{producto cauchy} converge \omterm{def:b2:series:def}{absolutamente} con
\[
\Bigl(\sum_m a_m\Bigr)\Bigl(\sum_n b_n\Bigr)
= \sum_{k=0}^{\infty} c_k,
\qquad
c_k = \sum_{m=0}^{k} a_m b_{k-m} .
\]
\end{theorem}

\begin{proof}
\emph{Caso no negativo.} Cada agrupación (por filas, columnas o
diagonales) calcula el mismo supremo: cualquier conjunto finito de pares es
contenido en un bloque finito de filas (limitando cada suma agrupada debajo
por sumas parciales finitas y arriba por el total), y monótono
la convergencia de sumas parciales hace el resto --- concretamente, para filas:
$\sum_{m \leq M}\sum_{n \leq N} u_{m,n} \leq S$ da, dejando que $N
\to \infty $luego $M \to \infty$, $\sum_m \sum_n u_{m,n} \leq S$;
a la inversa, cada $F$ finito se encuentra en dicho rectángulo, por lo que $S \leq
\sum_m\sum_n u_{m,n}$. Diagonales: los mismos dos límites con triángulos
en lugar de rectángulos.

\emph{Caso general.} Dividir en positivo y negativo (real y
imaginarias), cada una de las cuales es una familia no negativa \omterm{def:b2:series:summable}{sumable}; los cuatro
los agrupamientos coinciden en cada parte y, por tanto, en la diferencia; absoluto
la convergencia de la serie interna proviene del caso no negativo
aplicado a $\abs{u_{m,n}}$.

\emph{\omterm{thm:b2:series:fubini}{producto cauchy}.} La familia $u_{m,n} = a_m b_n$ es \omterm{def:b2:series:summable}{sumable}:
sumas parciales finitas de $\abs{a_mb_n}$ están limitadas por
$\bigl(\sum\abs{a_m}\bigr)\bigl(\sum\abs{b_n}\bigr)$. Las filas dan
$\bigl(\sum a_m\bigr)\bigl(\sum b_n\bigr)$; diagonales dan $\sum_k
c_k$.
\end{proof}

\begin{example}[La identidad exponencial, sinceramente.]\label{ex:b2:series:exp}
Para $a, b \in \C$ (o matrices de conmutación):
\[
\Bigl(\sum_m \frac{a^m}{m!}\Bigr)\Bigl(\sum_n
\frac{b^n}{n!}\Bigr)
= \sum_k \sum_{m+n=k} \frac{a^m b^n}{m!\,n!}
= \sum_k \frac{(a + b)^k}{k!},
\]
por el teorema del binomio en cada diagonal: $\eu^a \eu^b = \eu^{a+b}$
--- la ecuación funcional de $\exp$ derivada solo de la serie.
(La conmutación se utiliza en el paso binomial; para no conmutación
matrices la identidad realmente falla,
\cref{ch:b2:diffeq}.)
\end{example}

\begin{example}[Productos Cauchy como dispositivo informático]\label{ex:b2:series:squaresum}
De la serie geométrica y $\sum_{n
\geq 1} nz^n = \frac{z}{(1-z)^2}$($\abs z < 1$) de \cref{exo:b2:series:2}, una más
\omterm{thm:b2:series:fubini}{producto cauchy} finaliza el segundo momento. Multiplica $\sum_m
mz^m $por $\sum_n z^n$: el coeficiente diagonal es $\sum_{m=0}^k
m = \frac{k(k+1)}2$, entonces
\[
\frac{z}{(1-z)^3} = \sum_{k\geq0}\frac{k(k+1)}{2}\,z^k ,
\]
y la identidad $n^2 = 2\cdot\frac{n(n+1)}2 - n$ ensambla
\[
\sum_{n\geq1} n^2z^n = \frac{2z}{(1-z)^3} - \frac{z}{(1-z)^2}
= \frac{z(1+z)}{(1-z)^3} .
\]
En $z = \frac12$: $\sum_{n\geq1}\frac{n^2}{2^n} =
\frac{\frac12\cdot\frac32}{\frac18} = 6$ --- un valor cerrado con
no hay diferenciación en ninguna parte, solo series convergentes \omterm{def:b2:series:def}{absolutamente}
multiplicados como polinomios. El mismo telescopio de identidades
calcula cada $\sum n^dz^n$, y los probabilistas reconocerán
el segundo momento factorial de la ley geométrica
(\cref{ch:b2:genfun}).
\end{example}

\begin{example}[Una evaluación de doble suma]\label{ex:b2:series:doublesum}
Para $s > 1$ real, sea $\zeta(s) = \sum_{n\geq1} n^{-s}$. Contando
divisors by double summation --- the family $(m^{-s}n^{-s})$ over
$(m,n) \in (\N^*)^2$ es \omterm{def:b2:series:summable}{sumable} (producto de convergencia positiva
series) --- y agrupando por producto $q = mn$:
\[
\zeta(s)^2 = \sum_{m,n} \frac{1}{(mn)^s}
= \sum_{q=1}^{\infty} \frac{d(q)}{q^s},
\]
donde $d(q)$ es el número de divisores de $q$. \omterm{def:b2:series:summable}{Sumable} familias
convertir la combinatoria en análisis.
\end{example}

\begin{example}[Una evaluación de Fubini: $\sum_n (\zeta(n) - 1) = 1$]\label{ex:b2:series:zetasum}
Para números enteros $n \geq 2$, $\zeta(n) - 1 = \sum_{k \geq 2} k^{-n}$.
La doble familia $(k^{-n})_{k, n \geq 2}$ es \omterm{def:b2:series:summable}{sumable}: sumando
las columnas geométricas primero,
\[
\sum_{k\geq2}\sum_{n\geq2} \frac{1}{k^n}
= \sum_{k\geq2} \frac{1/k^2}{1 - 1/k}
= \sum_{k\geq2} \frac{1}{k(k-1)} = 1
\]
(telescópico), y todos los términos son positivos, por lo que
\cref{thm:b2:series:fubini} autoriza la suma por filas en su lugar:
\[
\sum_{n\geq2}\bigl(\zeta(n) - 1\bigr) = 1 .
\]
Los infinitos valores $\zeta$, cada uno de ellos de aspecto trascendental,
tienen colas que suman exactamente $1$. Perspectiva final: cuando un
La suma doble tiene términos positivos, calcúlala en cualquier orden.
se derrumba --- aquí las columnas son geométricas, las filas son misteriosas,
y Fubini traslada el colapso.
\end{example}

\begin{example}[La serie geométrica resuelve una ecuación.]\label{ex:b2:series:neumannsolve}
En \omterm{def:b2:nvs:banach}{espacio banach} $\bigl(C(\intcc01),
\norm\cdot_\infty\bigr)$, resuelva $x - K(x) = y $donde $K(f)$ es
la función constante $\frac12\int_0^1 f$. El \omterm{thm:b2:nvs:continuouslinear}{norma del operador} es
$\vertiii K \leq \frac12 < 1$, por lo que se aplica la serie Neumann
(\cref{ex:b2:series:neumann}): $x = \sum_{n\geq0} K^n(y)$.
Calcule las iteraciones: $K(y) = \frac12\int_0^1 y$ (una constante),
y aplicando $K$ a una constante $c$ da $\frac c2$, entonces
$K^n(y) = \frac{1}{2^{n-1}}\cdot\frac12\int_0^1 y$ para $n \geq
1$. Sumando las constantes geométricas:
\[
x = y + \Bigl(\int_0^1 y\Bigr)
\sum_{n\geq1}\frac{1}{2^n} = y + \int_0^1 y .
\]
Verificar: $x - K(x) = y + \int y - \frac12\bigl(\int y + \int
y\bigr) = y$. Una serie infinita, una respuesta finita y una
verificación de una línea --- la serie geométrica es una inversión
algoritmo, no sólo una declaración de convergencia.
\end{example}

\begin{example}[Telescópico por fracciones parciales]\label{ex:b2:series:telescope}
La suma exacta es rara; telescópico es su principal proveedor.
descomponer
\[
\frac{1}{n(n+1)(n+2)}
= \frac{1}{2}\Bigl(\frac{1}{n(n+1)} -
\frac{1}{(n+1)(n+2)}\Bigr),
\]
(verifique por reducción al denominador común), por lo que el parcial
las sumas colapsan:
\[
\sum_{n=1}^{N}\frac{1}{n(n+1)(n+2)}
= \frac12\Bigl(\frac{1}{1\cdot2} -
\frac{1}{(N+1)(N+2)}\Bigr)
\longrightarrow \frac14 .
\]
El mismo patrón --- escriba el término como $c(u_n - u_{n+1})$ para
un $(u_n)$ explícito --- resuelto \cref{exo:b2:series:10}
(arctagentes) y calcula cada $\sum\frac{1}{n(n+1)\cdots(n +
k)} = \frac{1}{k\cdot k!}$. Cuando existe una suma exacta en este
nivel, normalmente se esconde un telescopio en el término.
\end{example}

\begin{remark}[Perspectivas dentro de este volumen]\label{rem:b2:series:perspectives}
Los tres capítulos siguientes son clientes directos. Para
\cref{ch:b2:funcseq}: la convergencia normal de $\sum f_n$ es
convergencia absoluta de $\sum\norm{f_n}_\infty$ en \omterm{def:b2:nvs:banach}{Banach
espacio} $\bigl(C, \norm\cdot_\infty\bigr)$ --- este capítulo
\cref{thm:b2:nvs:absoluteconvergence} disfrazado. Para
\cref{ch:b2:powerseries}: dentro del disco de convergencia
todo es absoluto y \omterm{def:b2:series:summable}{sumable}, entonces \omterm{thm:b2:series:fubini}{productos cauchy} y
Los reordenamientos funcionan libremente (es por eso que las series de potencias se multiplican como
polinomios); en el límite, la prueba de Abel se hace cargo
(\cref{ex:b2:series:abelboundary}). Para \cref{ch:b2:genfun}:
Las funciones generadoras de probabilidad son series de potencias
manipulaciones --- productos para sumas de variables independientes,
sumas dobles para leyes compuestas --- todos están autorizados por
\cref{thm:b2:series:fubini}. \omterm{def:b2:series:summable}{Sumable} las familias son las legales
departamento de análisis por venir.
\end{remark}

\begin{remark}[Dónde se utiliza este capítulo]
Todo lo que tiene suma infinita pasa por aquí: series de potencias
(\cref{ch:b2:powerseries}) son familias \omterm{def:b2:series:summable}{sumable} disfrazadas,
Los coeficientes de Fourier se multiplican por \omterm{thm:b2:series:fubini}{productos cauchy} y
reorganizado por Parseval (\cref{ch:b2:fourier}) y probabilidad
funciones generadoras (\cref{ch:b2:genfun}) son el teorema de Fubini
aplicado a las expectativas. El volumen del año 3 absorbe \omterm{def:b2:series:summable}{sumable}
familias en la integración de Lebesgue sobre la medida de conteo ---
donde \cref{thm:b2:series:fubini} se convierte en un caso especial del
Teorema de Fubini-Tonelli.
\end{remark}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:series:3}

Demuestre, mediante \omterm{thm:b2:series:abel}{resumen de abel}, que $\dfrac{1}{n}\sum_{k=1}^{n} k\,b_k
\to 0$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:series:4}
Estudiar la convergencia de $\sum \dfrac{\sin(n\theta)}{n^\alpha}$
($\theta \in \R$, $\alpha > 0$) --- para el cual $(\theta, \alpha)$ es
¿\omterm{def:b2:series:def}{absolutamente} es convergente, semiconvergente, divergente?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:series:5}
(Reordenamiento de Riemann) Sea $\sum u_n$ un convergente pero no
\omterm{def:b2:series:def}{absolutamente} series reales convergentes y $\ell \in \R$. demostrar que
alguna reordenación de $\sum u_n$ converge a $\ell$.
\emph{(Show both subseries of positive and negative terms diverge;
then greedily alternate: take positive terms until exceeding
$\ell$, then negative until dropping below, and so on; the terms
tend to $0$, forcing convergence to $\ell$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:series:6}
demostrar que la familia
$\Bigl(\dfrac{x^{m+n}}{m!\,n!}\Bigr)_{(m,n)\in\N^2}$ es \omterm{def:b2:series:summable}{sumable}
para cada $x \in \R$, y vuelva a derivar la identidad $(\eu^x)^2 =
\eu^{2x}$ agrupando la suma doble a lo largo de las diagonales $m + n =
k$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:series:7}
Demostrar que la familia $\bigl(\frac{1}{m^2 n^2}\bigr)_{m,n \geq 1}$
es \omterm{def:b2:series:summable}{sumable}, y esa agrupación por $\gcd$: con $q = \gcd(m,n)$,
\[
\zeta(2)^2 = \sum_{q\geq1} \frac{1}{q^4}
\sum_{\substack{a,b \geq 1\\ \gcd(a,b)=1}} \frac{1}{a^2b^2}
= \zeta(4) \cdot S,
\]
donde $S = \sum_{\gcd(a,b)=1} \frac{1}{a^2b^2}$: deducir $S =
\zeta(2)^2/\zeta(4)$. \emph{(Every pair $(m,n)$ writes uniquely
$(qa, qb)$ with $\gcd(a,b) = 1$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:series:8}
(Teorema de Abel sobre productos, versión ligera) Supongamos $\sum a_n$
converge \omterm{def:b2:series:def}{absolutamente} y $\sum b_n$ converge. Demostrar que su
\omterm{thm:b2:series:fubini}{producto cauchy} $\sum c_n$ converge con $\sum c_n = (\sum
a_n)(\sum b_n)$. \emph{(Write $C_N = \sum_{k\leq N} c_k = \sum_n
a_n B_{N-n}$ with $B$the partial sums of $b$; split according to
$n \leq N/2$ or not, using boundedness of $(B_m)$ and the absolute
tail of $(a_n)$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:series:9}
En el espacio (\omterm{def:b2:metric:complete}{incompleto}) $E$ de secuencias reales eventualmente cero
con el sup \omterm{def:b2:nvs:norm}{norma}, exhibe una serie convergente \omterm{def:b2:series:def}{absolutamente} que no
no converge en $E$. \emph{(Try $u_n = 2^{-n} e_n$ with $(e_n)$ the
canonical sequences.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:series:10}
Verifique la identidad $\arctan(n+1) - \arctan(n) =
\arctan\dfrac{1}{n^2 + n + 1}$ y deduzca el valor exacto de

\sum_{n=1}^{\infty} \arctan\frac{1}{n^2 + n + 1} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:series:11}
Determinar la naturaleza (con equivalentes) de

\section{Problema: $\zeta(2) = \pi^2/6$ de Euler, por Cauchy
Suma cotangente}

La identidad más famosa de Euler, $1 + \frac14 + \frac19 + \cdots =
\frac{\pi^2}6$, admite una prueba completamente elemental, debido a
Cauchy: la fórmula de de Moivre produce un polinomio cuyas raíces son
los números $\cot^2\frac{k\pi}{2n+1}$, Vieta suma esas raíces
exactamente, y el apretón $\cot^2\theta < \frac{1}{\theta^2} <
1 + \cot^2\theta $aplasta las sumas parciales de $\sum\frac1{k^2}$
entre dos límites racionales explícitos. Ejecutamos la prueba en su totalidad,
extraiga $\zeta(4) = \frac{\pi^4}{90}$ por el mismo método, luego
mapear toda la frontera entre convergencia y divergencia con
la serie Bertrand --- y demostrar que la frontera no lleva
serie convergente más lenta.

\begin{problem}[{Problema del fin de semana --- $\zeta(2) = \pi^2/6$ y el
Panorama de Bertrand}]
\label{pb:b2:series:1}
En todo momento, $n \geq 1$ y $\theta_k = \dfrac{k\pi}{2n+1}$ para $k
= 1, \dots, n$; nota $0 < \theta_k < \frac\pi2$.

\medskip
\noindent\textbf{Parte I --- The cotangent identity.}
\begin{enumerate}
 \item Pruebe la fórmula de Moivre $(\cos\theta +
 \iu\sin\theta)^m = \cos m\theta + \iu\sin m\theta $($m
 \in \N$), y deduzca, para $m = 2n + 1$,
 \[
 \sin\bigl((2n{+}1)\theta\bigr) = \sum_{j=0}^{n}
 (-1)^j\binom{2n+1}{2j+1}\cos^{2(n-j)}\theta\,
 \sin^{2j+1}\theta .
 \]
 \item Deduce que para $\theta \in \intoo{0}{\frac\pi2}$,
 \[
 \sin\bigl((2n{+}1)\theta\bigr) =
 \sin^{2n+1}\theta\; P_n(\cot^2\theta),
 \qquad
 P_n(x) = \sum_{j=0}^{n}
 (-1)^j\binom{2n+1}{2j+1}x^{\,n-j},
 \]
 un polinomio de grado $n$ con coeficiente principal $2n
 + 1$.
 \item Demuestre que $x_k = \cot^2\theta_k$, $k = 1, \dots, n$, son
 $n$ \emph{distinto} raíces de $P_n$ --- de ahí todos
 ellos.
 \item Por Vieta, prueba la identidad exacta
 \[
 \sum_{k=1}^{n} \cot^2\frac{k\pi}{2n+1}
 = \frac{n(2n-1)}{3}.
 \]
 \item Deducir también $\displaystyle\sum_{k=1}^{n}
 \frac{1}{\sin^2\theta_k} = \frac{2n(n+1)}{3}$.
 \item Pruebe el apretón: $\cot^2\theta < \dfrac1{\theta^2} <
 \dfrac{1}{\sin^2\theta}$ para $\theta \in
 \intoo{0}{\frac\pi2}$\emph{(from $\sin\theta < \theta <
 \tan\theta$)}.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The squeeze closes: Euler's
theorem.}
\begin{enumerate}[resume]
 \item Sumando la pregunta 6 sobre $k = 1, \dots, n$ con $\theta =
 \theta_k$, establezca
 \[
 \frac{n(2n-1)}{3} \;<\;
 \frac{(2n+1)^2}{\pi^2}\sum_{k=1}^{n}\frac1{k^2}
 \;<\; \frac{2n(n+1)}{3}.
 \]
 \item Concluir (\emph{Teorema de Euler, por la prueba de Cauchy}):
 \[
 \zeta(2) = \sum_{k=1}^{\infty}\frac{1}{k^2} =
 \frac{\pi^2}{6}.
 \]
 \item Extraer una tarifa del sándwich: mostrar
 \[
 \Bigl|\sum_{k=1}^{n}\frac1{k^2} - \frac{\pi^2}6\Bigr|
 = O\Bigl(\frac1n\Bigr),
 \]
 consistente con la cola exacta $\sum_{k>n}k^{-2} =
 \frac1n - \frac{1}{2n^2} + O(n^{-3})$ de
 \cref{exo:b2:comparison:11}.
 \item Haga funcionar la máquina un piso más arriba: usando el segundo Vieta
 función de $P_n$, mostrar
 \[
 \sum_{k=1}^{n}\cot^4\theta_k =
 \Bigl(\frac{n(2n-1)}3\Bigr)^{\!2} -
 \frac{2n(2n-1)(2n-2)(2n-3)}{60}
 \;\sim\; \frac{8n^4}{45},
 \]
 y apriete con $\cot^4 < \theta^{-4} < (1 +
 \cot^2)^2$ para obtener $\zeta(4) = \dfrac{\pi^4}{90}$.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Dividends.}
\begin{enumerate}[resume]
 \item Deducir de $\zeta(2) = \frac{\pi^2}6$:
 \[
 \sum_{k\geq0}\frac{1}{(2k+1)^2} = \frac{\pi^2}{8},
 \qquad
 \sum_{k\geq1}\frac{(-1)^{k-1}}{k^2} = \frac{\pi^2}{12}.
 \]
 \item Combinar con \cref{exo:b2:series:7}: calcular $S =
 \sum_{\gcd(a,b)=1}\frac{1}{a^2b^2} =
 \frac{\zeta(2)^2}{\zeta(4)} = \frac52$ e interpretar
 $\frac{1}{\zeta(2)} = \frac{6}{\pi^2} \approx 0.608$ como
 la densidad de pares coprimos (explique la heurística
 Sinceramente: el recuento riguroso es un volumen del año 3.
 materia).
 \item (Aceleración certificada) La fórmula de la cola de la pregunta 9
 da $\sum_{k\leq n}k^{-2} + \frac1n - \frac1{2n^2} =
 \frac{\pi^2}6 + O(n^{-3})$. Compara el trabajo necesario para
 seis dígitos de $\zeta(2)$: suma directa versus el
 suma corregida en $n = 100$ (donde el error es
 $1.7\cdot10^{-7}$).
 \item Verifique la pregunta 4 manualmente en $n = 1$ y $n = 2$ (el
 valores $\cot^2\frac\pi3 = \frac13$ y
 $\cot^2\frac\pi5 + \cot^2\frac{2\pi}5 = 2$), utilizando
 $\cos\frac\pi5 = \frac{1+\sqrt5}4$ o un número
 evaluación.
 \item Comprobar la identidad del acompañante
 \[
 \sum_{k=1}^{n}\tan^2\frac{k\pi}{2n+1} = n(2n+1)
 \]
 \emph{(the numbers $\tan^2\theta_k$ are the roots of the
 reversed polynomial $x^nP_n(1/x)$)} y verifíquelo en $n
 = 1$.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The Bertrand panorama.} Para $\alpha,
\beta \in \R$, considere el \emph{Serie Bertrand}
\[
\sum_{n \geq 3} \frac{1}{n^{\alpha}(\ln n)^{\beta}} .
\]
\begin{enumerate}[resume]
 \item Demuestre que para $\alpha > 1$ la serie converge,
 lo que sea $\beta$ \emph{(compare with
 $n^{-(1+\alpha)/2}$)}.
 \item Demuestre que para $\alpha < 1$ diverge, sea cual sea
 $\beta$.
 \item Para $\alpha = 1$: usando la comparación serie-integral
 (\cref{thm:b2:comparison:seriesintegral}) con $f(t) =
 \frac{1}{t(\ln t)^\beta}$, demuestre la convergencia si es $\beta
 > 1$.
 \item Iterar la frontera: mostrar $\sum\frac{1}{n\ln n\,\ln\ln
 n}$ diverge mientras $\sum\frac{1}{n\ln n\,(\ln\ln n)^2}$
 converge.
 \item Dos trampas: determinar la naturaleza de
 \[
 \sum_n \frac{1}{n^{1 + 1/\ln n}}
 \qquad\text{and}\qquad
 \sum_n \frac{1}{n^{1 + 1/\ln\ln n}}
 \]
 \emph{(compute $n^{1/\ln n}$ exactly; compare
 $n^{1/\ln\ln n}$ with every power of $\ln n$)}.
 \item (No hay series convergentes más lentas) Sea $\sum a_n$ cualquier
 serie convergente con $a_n > 0$ y $R_n = \sum_{k
 \geq n}a_k $sus colas. Demuestre que $\sum
 \frac{a_n}{\sqrt{R_n}}$ sigue siendo \emph{converge}
 \emph{(compare with the telescoping $2(\sqrt{R_n} -
 \sqrt{R_{n+1}})$)}, aunque $\frac{a_n/\sqrt{R_n}}{a_n}
 \to \infty$: toda serie convergente es estrictamente
 dominada por otra serie convergente. la frontera de
 La convergencia no es una curva sino una niebla.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Cross-checks and synthesis.}
\begin{enumerate}[resume]
 \item (Condensación de Cauchy) Prueba: para $(a_n)$ positivo
 decreciente, $\sum a_n$ converge si $\sum 2^k a_{2^k}$
 converge. Vuelva a derivar la frontera de la pregunta 18 a partir de ella.
 \item (El costo de la lentitud) Para $\sum\frac1{n(\ln n)^2}$,
 unir la cola por una integral y demostrar que resumiendo
 a $N = 10^6$ todavía deja un error mayor que
 $0.07$: la convergencia certificada por la teoría puede ser inútil
 para números --- contraste con la pregunta 13.
 \item Clasificar (con justificaciones de una línea):
 $\sum\frac1{n\ln n}$, $\sum\frac1{n^{1.01}}$,
 $\sum\frac{(\ln n)^{100}}{n^{1.001}}$,
 $\sum\frac1{n(\ln n)(\ln\ln n)^{3}}$.
 \item (Síntesis) Una frase cada uno: cómo de Moivre se convirtió en
 identidad trigonométrica en un polinomio con computable
 sumas de raíces; donde se necesitaba el apretón \emph{exacto}
 identidades de puntos finales en lugar de equivalentes; cual herramienta
 de \cref{ch:b2:comparison} motorizado Parte IV; y que
 La pregunta 21 dice sobre el sueño de un "universal".
 prueba de comparación''. Nombra las dos cumbres: Euler
 $\zeta(2) = \frac{\pi^2}6$ (y su piso de arriba,
 $\zeta(4) = \frac{\pi^4}{90}$), y el Bertrand
 clasificación. Tenga en cuenta dónde se probará $\zeta(2)$
 otra vez: por Parseval en \cref{ch:b2:fourier} --- uno
 Teorema, dos civilizaciones.
\end{enumerate}
\end{problem}
