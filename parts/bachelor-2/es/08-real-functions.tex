\chapter{Funciones de una variable real}\label{ch:b2:realfun}

Antes de que el análisis pase a funciones de funciones (\cref{ch:b2:funcseq})
vale la pena conocer el paisaje de una variable con mayor detalle que
Se requiere el año 1: qué tan discontinua puede ser una función monótona, qué tan
regular debe ser una función convexa y qué propiedades especiales
disfrute de derivados (Darboux). Estos resultados estructurales son breves,
agudo y querido por los examinadores.

\section{Funciones monótonas}

\begin{theorem}[Regularidad de funciones monótonas]\label{thm:b2:realfun:monotone}
Sea $f \colon I \to \R$ creciente en un intervalo.
\begin{enumerate}
 \item En cada punto interior $a$, existen límites unilaterales:
 \[
 f(a^-) = \sup_{x < a} f(x) \;\leq\; f(a) \;\leq\;
 f(a^+) = \inf_{x > a} f(x) ;
 \]
 cada discontinuidad es un \emph{saltar}.
 \item El conjunto de discontinuidades de $f$ es como máximo
 \omterm{def:b2:structures:countable}{contable}.\index{función monótona!discontinuidades}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) El conjunto $\{f(x) : x < a\}$ no está vacío, limitado arriba por $f(a)$:
su supremo $s$ satisface $f(x) \to s$ como $x \to a^-$ (dado
$\varepsilon$, algo de $f(x_0) > s - \varepsilon$ y monotonicidad
atrapa $f(x) \in \intoc{s - \varepsilon}{s}$ para $x \in
\intoo{x_0}{a}$). Simétricamente a la derecha.

(2) A cada discontinuidad $a$ adjunte el intervalo \omterm{def:b2:metric:topology}{abierto} no vacío
$J_a = \intoo{f(a^-)}{f(a^+)}$ (un auténtico salto). Para $a < b$
discontinuidades, $J_a$ y $J_b$ son disjuntas: $f(a^+) \leq f(c)
\leq f(b^-)$para cualquier $c$entre ellas. Cada $J_a$ contiene un racional;
discontinuidades distintas obtienen racionales distintos: una inyección de la
discontinuidad establecida en $\Q$, que es \omterm{def:b2:structures:countable}{contable}
(\cref{prop:b2:structures:countablestable}).
\end{proof}

\begin{example}
El límite es claro: corrija una enumeración $(r_n)$ de $\Q \cap
\intoo{0}{1}$y establezca $f(x) = \sum_{n : r_n \leq x} 2^{-n}$
(una definición de familia resumible,
\cref{def:b2:series:summable}). Entonces $f$ está aumentando en
$\intcc{0}{1}$ y discontinuo exactamente en cada racional de
$\intoo{0}{1}$ (salte $2^{-n}$ en $r_n$): una función monótona
\emph{poder} será discontinuo sobre un \omterm{def:b2:structures:countable}{conjunto contable} denso.
\end{example}

\begin{example}[Los saltos no pueden pesar más que la subida.]\label{ex:b2:realfun:jumpbudget}
Para $f$ aumentando en $\intcc{a}{b}$, los saltos tienen un presupuesto:
si $a < c_1 < \dots < c_m < b$ son discontinuidades con saltos
$s_i = f(c_i^+) - f(c_i^-) > 0$, luego eligiendo entrelazado
puntos $a < c_1 < t_1 < c_2 < \dots$ y usando monotonicidad en
cada pieza,
\[
\sum_{i=1}^{m} s_i \;\leq\; f(b) - f(a) :
\]
el ascenso total limita el salto total. Consecuencia: para
cada $k$, como máximo $k\,\bigl(f(b) - f(a)\bigr)$ discontinuidades
tiene salto $\geq \frac1k$ --- un refinamiento cuantitativo de
\cref{thm:b2:realfun:monotone}~(2), desde el conjunto de discontinuidades
es la unión \omterm{def:b2:structures:countable}{contable} sobre $k$ de estos conjuntos finitos. en el
función de salto racional anterior, el presupuesto se gasta exactamente:
los saltos $2^{-n}$ suman a $1 = f(1^+) - f(0^-)$ en lo obvio
sentido extendido. Las funciones monótonas pueden saltar densamente, pero sólo
con una estricta asignación.
\end{example}

\section{Funciones convexas}

\begin{lemma}[Desigualdad de pendiente]\label{lem:b2:realfun:slopes}
Sea $f$ convexo en $I$ y $x < y < z$ en $I$. entonces
\[
\frac{f(y) - f(x)}{y - x}
\;\leq\; \frac{f(z) - f(x)}{z - x}
\;\leq\; \frac{f(z) - f(y)}{z - y} :
\]
las pendientes de las cuerdas aumentan en ambos puntos finales.\index{convexo
función!desigualdad de pendiente}
\end{lemma}

\begin{proof}
Escribe $y = \frac{z - y}{z - x}\,x + \frac{y - x}{z - x}\,z$: a
combinación convexa, ya que los dos coeficientes son positivos y
suma a $1$. La convexidad da
\[
f(y) \;\leq\; \frac{z-y}{z-x}\,f(x) + \frac{y-x}{z-x}\,f(z).
\]
Para la desigualdad izquierda, reste $f(x)$ de ambos lados, usando
$\frac{z-y}{z-x} - 1 = -\frac{y-x}{z-x}$:
\[
f(y) - f(x) \leq \frac{y - x}{z - x}\bigl(f(z) - f(x)\bigr),
\]
y dividir por $y - x > 0$. Para la desigualdad correcta, reste
en lugar de $f(z)$:
\[
f(z) - f(y) \geq f(z) - \frac{z-y}{z-x}f(x) -
\frac{y-x}{z-x}f(z)
= \frac{z - y}{z - x}\bigl(f(z) - f(x)\bigr),
\]
y dividir por $z - y > 0$. Ambos pasos mostrados son iguales.
identidad baricéntrica leída frente a un punto final diferente.
\end{proof}

\begin{theorem}[Regularidad de funciones convexas]\label{thm:b2:realfun:convexreg}
Sea $f$ convexo en un intervalo $I$.
\begin{enumerate}
 \item En cada punto interior, $f$ tiene finitos unilaterales.
 derivados $f'_g \leq f'_d$; ambas son funciones crecientes
 del punto; en particular $f$ es \omterm{def:b2:metric:continuity}{continuo} en el
 interior de $I$ (pero posiblemente no en los puntos finales).
 \item $f$ se encuentra encima de cada \emph{línea de apoyo}: para interior $a$
 y cualquier $m \in \intcc{f'_g(a)}{f'_d(a)}$,
 \[
 f(x) \geq f(a) + m(x - a) \qquad (x \in I).
 \]
 \item (Jensen, ponderado) Para $x_i \in I$ y pesos $\lambda_i
 \geq 0$, $\sum\lambda_i = 1$:
 \[
 f\Bigl(\sum_i \lambda_i x_i\Bigr) \leq \sum_i \lambda_i
 f(x_i) .\index{La desigualdad de Jensen}
 \]
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Repare el interior de $a$. Por \cref{lem:b2:realfun:slopes}, la pendiente
$\tau(h) = \frac{f(a + h) - f(a)}{h}$ es una función creciente de
$h$ (en ambos lados, y $\tau(h_-) \leq \tau(h_+)$ para $h_- < 0 <
h_+$). Por lo tanto, $\tau $tiene un límite finito como $h \to 0^-$ (creciente,
delimitado arriba por cualquier pendiente recta) --- esto es $f'_g(a)$ --- y como
$h \to 0^+$ ($f'_d(a)$), con $f'_g(a) \leq f'_d(a)$. finito
Las derivadas unilaterales fuerzan \omterm{def:b2:metric:continuity}{continuidad} en $a$. Monotonicidad en el
punto: para $a < b$ interior, $f'_d(a) \leq \frac{f(b) - f(a)}{b -
a} \leq f'_g(b)$, nuevamente por la desigualdad de pendiente.

(2) Para $x > a$: $\frac{f(x) - f(a)}{x - a} \geq f'_d(a) \geq m$;
para $x < a$: $\frac{f(a) - f(x)}{a - x} \leq f'_g(a) \leq m$. ambos
reorganizar al reclamo.

(3) Inducción en el número de puntos exactamente como en el Año 1
volumen (el caso de dos puntos es la definición) --- o de un solo trazo:
aplicar (2) en $a = \sum\lambda_i x_i$ y promediar la línea de soporte
desigualdades en los puntos $x_i$ con pesos $\lambda_i$: $\sum_i
\lambda_i f(x_i) \geq f(a) + m\sum_i\lambda_i(x_i - a) = f(a)$.
\end{proof}

\begin{example}[Entropía máxima]\label{ex:b2:realfun:entropy}
Para un vector de probabilidad $(p_1, \dots, p_n)$ (positivo,
sumando $1$), la entropía $H(p) = -\sum_i p_i\ln p_i$
satisface
\[
H(p) \leq \ln n ,
\qquad\text{with equality iff } p_i = \frac1n \text{ for all }
i .
\]
Prueba de Jensen (\cref{thm:b2:realfun:convexreg}~(3)) aplicada a
el \emph{cóncavo} $\ln$ con pesos $p_i$ en los puntos
$\frac{1}{p_i}$:
\[
H(p) = \sum_i p_i\ln\frac{1}{p_i}
\leq \ln\Bigl(\sum_i p_i\,\frac1{p_i}\Bigr) = \ln n ,
\]
igualdad obligando a todos los puntos $\frac1{p_i}$ iguales (estricta
concavidad), es decir,\ $p$ uniforme. De manera equivalente, esto es
\cref{exo:b2:realfun:7} con uniforme $q$. La incertidumbre es
maximizado por la ignorancia uniformemente extendida --- la variación
Principio detrás de la codificación, la mecánica estadística y la entropía.
apariciones de \cref{ch:b2:randomvar}.
\end{example}

\begin{method}[Encontrar la función convexa detrás de una desigualdad]\label{met:b2:realfun:spotting}
La mayoría de las desigualdades clásicas son Jensen disfrazado; desnudarse
uno: (1) normalizar para que aparezca \emph{promedio ponderado}
(pesos positivos, sumando $1$ --- dividir por una masa total si
necesario); (2) observe qué función se aplica dentro versus
fuera del promedio: el reclamo ``$f(\text{average}) \leq$
el promedio de $f$'' nombra el convexo $f$; (3) certificar la convexidad mediante
la segunda derivada, y manejar la igualdad a través del rigor; (4)
si no se ve ningún promedio, tome primero los logaritmos --- productos
y las potencias se convierten en promedios, y la concavidad de $\ln$ lleva
AM--GM, Young y sus familiares (el fin de semana de este capítulo
problema ejecuta los pasos 1 a 4 en cada uno de ellos). Si incluso los logaritmos lo hacen
no revela un promedio, intente leer la desigualdad como
monotonicidad de pendientes (\cref{lem:b2:realfun:slopes}) ---
declaraciones de superaditividad como \cref{exo:b2:realfun:9} en vivo
allí.
\end{method}

\begin{remark}[Errores comunes]\label{rem:b2:realfun:pitfalls}
(i) Los productos $x$ y $(x - 1)^2$ no conservan la convexidad.
son convexos en $\intcc{0}{2}$, pero su producto $x(x-1)^2$ tiene
segunda derivada $6x - 4$, negativa en
$\intco{0}{\frac23}$ --- no convexo; ni se conserva la convexidad
por composición sin monotonicidad
(\cref{exo:b2:realfun:10}). (ii) Jensen gira hacia el cóncavo
funciones: la mitad de las desigualdades clásicas son las cóncavas
$\ln$-versión; aplicar la forma convexa a $\ln$ es la
forma más rápida de demostrar AM--GM \emph{hacia atrás}. (iii) Punto medio
la convexidad por sí sola no implica convexidad --- \omterm{def:b2:metric:continuity}{continuidad} (o
se necesita mera delimitación) (\cref{exo:b2:realfun:8}); el
Los contraejemplos patológicos viven más allá de los axiomas de este libro. (v)
Una función convexa en un intervalo \emph{\omterm{def:b2:metric:topology}{abierto}} es \omterm{def:b2:metric:continuity}{continuo},
incluso localmente \omterm{def:b2:metric:continuity}{Lipschitz} (\cref{exo:b2:realfun:12}); en
puntos finales, nada es gratis. (v) Los derivados obedecen a Darboux pero
No es necesario que sea \omterm{def:b2:metric:continuity}{continuo} (\cref{ex:b2:realfun:oscillation}):
``$f'$ no tiene saltos'' nunca significa ``$f'$ es \omterm{def:b2:metric:continuity}{continuo}''.
\end{remark}

\section{La propiedad Darboux}

\begin{theorem}[Darboux]\label{thm:b2:realfun:darboux}
Sea $f$ diferenciable en un intervalo $I$. Entonces $f'$ toma cada
valor entre dos de sus valores --- aunque $f'$ no necesita ser
\omterm{def:b2:metric:continuity}{continuo}.\index{teorema de darboux}
\end{theorem}

\begin{proof}
Deje $a < b$ en $I$ y $v$ estrictamente entre $f'(a)$ y $f'(b)$,
diga $f'(a) < v < f'(b)$. La función $g(x) = f(x) - vx$ es
diferenciable con $g'(a) < 0 < g'(b)$: su mínimo en
$\intcc{a}{b}$ (obtenido: \omterm{def:b2:metric:continuity}{continuidad} en un \omterm{def:b2:metric:compact}{compacto}) no está en $a$
(justo después de $a$, $g$ disminuye por debajo de $g(a)$) ni en $b$ (justo
antes de $b$, $g$ está debajo de $g(b)$): es interior, y allí $g'(c)
= 0$, es decir $f'(c) = v$. (Este fue un ejercicio destacado del Año 1; es
El lugar en la teoría es aquí.)
\end{proof}

\begin{example}[¿Qué funciones son derivadas?]\label{ex:b2:realfun:whichderivatives}
El teorema de Darboux es una máquina de la inexistencia. La función del suelo
$\lfloor x\rfloor$ no es la derivada de ninguna función en
$\R$: toma los valores $0$ y $1$ pero omite $\frac12$ en
$\intcc{0}{1}$, que \cref{thm:b2:realfun:darboux} prohíbe
derivados. El mismo veredicto afecta a todas las funciones con un salto
--- signo, Heaviside, todas las funciones escalonadas --- por inocentes que sean
ellos miran; sus ``antiderivadas'' ($\abs x$ para signo, etc.)
existe sólo lejos del salto y anuda allí con una esquina.
Contraste: el tremendamente discontinuo $f'$ de
\cref{ex:b2:realfun:oscillation} \emph{is} un derivado --- es
la discontinuidad es una oscilación que Darboux tolera. el
El límite entre los dos comportamientos es exactamente el mismo.
Corolario sin saltos a continuación.
\end{example}

\begin{corollary}\label{cor:b2:realfun:nojumps}
Un derivado no tiene discontinuidades de salto: si $f'(a^-)$ y
$f'(a^+)$ existen, son iguales a $f'(a)$. Las discontinuidades de un
derivadas son siempre de tipo oscilación ($x^2\sin\frac1x$'s
derivado en $0$, volumen del año 1).
\end{corollary}

\begin{proof}
Si $f'(a^+) = \lim_{x\to a^+} f'(x)$ existe y difiere de
$f'(a)$, los valores estrictamente entre ellos serán omitidos por $f'$ en
un barrio correcto --- contradiciendo a Darboux a intervalos
$\intcc{a}{a + h}$. (Alternativamente: el teorema del valor medio obliga
$f'(a) = \lim_{h\to0^+} \frac{f(a+h)-f(a)}{h} = f'(a^+)$, el
el cociente de diferencias es un valor $f'$ en un punto intermedio).
Lo mismo a la izquierda.
\end{proof}

\begin{example}[La derivada oscilante canónica]\label{ex:b2:realfun:oscillation}
Deje $f(x) = x^2\sin\frac1x$ para $x \neq 0$ y $f(0) = 0$. en
$0$: $\bigl|\frac{f(h) - f(0)}{h}\bigr| = \abs{h\sin\frac1h}
\leq \abs h \to 0$, por lo que $f'(0) = 0$ existe. Lejos de $0$,
\[
f'(x) = 2x\sin\frac1x - \cos\frac1x ,
\]
cuyo primer término tiende a $0$ mientras que $\cos\frac1x$ oscila
hasta $\intcc{-1}{1}$ en cada intervalo $\intoo{0}{\delta}$: el
El límite $f'(0^+)$ no existe. Entonces $f'$ está definido en todas partes menos
discontinuo en $0$ --- y, exactamente como
\cref{cor:b2:realfun:nojumps} predice, la discontinuidad es una
oscilación, no un salto: en cada $\intoo{0}{\delta}$, $f'$ todavía
barre un intervalo completo alrededor de $0$. Los derivados pueden ser salvajes, pero
sólo de forma compatible con Darboux.
\end{example}

\begin{remark}[Dónde se utiliza este capítulo]
La convexidad es el motor de la industria de la desigualdad: esto
El problema del fin de semana del capítulo fabrica Young, Hölder, Minkowski
y la cadena de potencia-significa de ella, que \cref{ch:b2:nvs}'s \omterm{def:b2:nvs:norm}{norma}
la teoría y las estimaciones integrales de \cref{ch:b2:integration} consumen;
Jensen reaparece en probabilidad como el momento de las desigualdades de
\cref{ch:b2:randomvar}. La regularidad monótona regresa en
\cref{ch:b2:integration} (las funciones monótonas son integrables) y,
en el volumen del Año 3, como la diferenciabilidad casi en todas partes
de funciones monótonas --- donde ``\omterm{def:b2:structures:countable}{contablemente} muchos saltos'' se convierte en
el primer paso de la teoría de Lebesgue.
\end{remark}

Sea $f$ diferenciable en $I$ con $f'$ monótono. demostrar que
$f'$ es \omterm{def:b2:metric:continuity}{continuo} \emph{(combine
\cref{thm:b2:realfun:monotone} and
\cref{cor:b2:realfun:nojumps})}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:realfun:6}
(Media geométrica como límite) Para $x_i$ positivo y pesos
$\lambda_i$ sumando $1$, probar
\[
\lim_{p \to 0^+} \Bigl(\sum_i \lambda_i x_i^p\Bigr)^{1/p}
= \prod_i x_i^{\lambda_i} ,
\]
vía $x_i^p = \eu^{p\ln x_i} = 1 + p\ln x_i + O(p^2)$, y deducir
la desigualdad AM-GM ponderada de
\cref{ex:b2:realfun:powermeans}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:realfun:7}
(Desigualdad de entropía) Usando la convexidad estricta de $t \mapsto t\ln t$,
demuestre que para $p_i, q_i$ positivo con $\sum p_i = \sum q_i = 1$:
\[
\sum_i p_i \ln\frac{p_i}{q_i} \geq 0 ,
\]
con igualdad si $p = q$. \emph{(Write the left side as $\sum q_i\,
\varphi\bigl(\frac{p_i}{q_i}\bigr)$with $\varphi(t) = t\ln t$ and
apply Jensen with weights $q_i$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:realfun:8}
(Convexidad del punto medio) $f \colon I \to \R$ es \emph{punto medio convexo}
cuando $f\bigl(\frac{x+y}{2}\bigr) \leq \frac{f(x) + f(y)}{2}$
siempre. Demuestre que una función convexa de punto medio \emph{\omterm{def:b2:metric:continuity}{continuo}} es
convexo. \emph{(Establish the convexity inequality for dyadic
weights $\frac{k}{2^m}$ by induction on $m$, then pass to the limit
using density and \omterm{def:b2:metric:continuity}{continuidad}.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:realfun:9}
Sea $f$ convexo en $\intco{0}{+\infty}$ con $f(0) \leq 0$. probar
que $x \mapsto \frac{f(x)}{x}$ está aumentando en
$\intoo{0}{+\infty}$, y deduce que para $f$ convexo con $f(0) =
0$: $f(x + y) \geq f(x) + f(y)$para $x, y \geq 0$
(superaditividad).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:realfun:10}
Sea $f$ convexo en $I$ y $g$ convexo \emph{creciente} en un
intervalo que contiene $f(I)$. Demuestre que $g \circ f$ es convexo y
demuestre con un contraejemplo que la monotonicidad de $g$ no puede ser
cayó.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:realfun:11}
(Hermite--Hadamard) Sea $f$ convexo y \omterm{def:b2:metric:continuity}{continuo} en
$\intcc{a}{b}$. probar
\[
f\Bigl(\frac{a+b}{2}\Bigr) \;\leq\; \frac{1}{b -
a}\int_a^b f(t)\,\dd t \;\leq\; \frac{f(a) + f(b)}{2} .
\]
\emph{(Izquierda: integrar una línea de soporte en el punto medio. Derecha:
atado $f$ por la cuerda.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:realfun:12}
Demuestre que una función convexa en un intervalo \emph{\omterm{def:b2:metric:topology}{abierto}} $I$ es
localmente \omterm{def:b2:metric:continuity}{Lipschitz}: para cada segmento $\intcc{a}{b} \subseteq I$
y margen $\delta > 0$ con $\intcc{a - \delta}{b + \delta}
\subseteq I$, la restricción de $f$a $\intcc{a}{b}$ es
\omterm{def:b2:metric:continuity}{Lipschitz}, con constante
$\max\Bigl(\bigl|\frac{f(a) - f(a - \delta)}{\delta}\bigr|,
\bigl|\frac{f(b + \delta) - f(b)}{\delta}\bigr|\Bigr)$
\emph{(atrapa cada pendiente de cuerda entre estos dos por la pendiente
desigualdad)}.
\end{exercise}

\section{Problema: la caja de herramientas de convexidad}

Una definición --- la cuerda encima del gráfico --- genera la
toda la caja de herramientas de las desigualdades clásicas. Este problema de fin de semana
lo construye en orden lógico: criterios de convexidad y estricto Jensen,
luego Young, Hölder y Minkowski (las partidas de nacimiento de los
$p$-normas), la cadena potencia-media completa desde el mínimo hasta el
máximo, y dos dividendos de corona --- la desigualdad de Carleman, y
Hölder leído como una dualidad. Todo está probado; nada es
importado.

\begin{problem}[{Problema de fin de semana --- Joven, Hölder, Minkowski,
y la cadena potencia-media}]
\label{pb:b2:realfun:1}
En todo momento, $p, q > 1$ son \emph{exponentes conjugados}: $\frac1p +
\frac1q = 1$; los vectores son $a = (a_1, \dots, a_n) \in \R^n$;
los pesos $\lambda_i > 0$ satisfacen $\sum_i\lambda_i = 1$.

\medskip
\noindent\textbf{Parte I --- Criteria and strict Jensen.}
\begin{enumerate}
 \item Sea $f$ diferenciable en un intervalo $I$. demostrar que
 $f$ es convexo si y sólo si $f'$ es creciente
 \emph{(one direction by passing to the limit in the
 slope inequality \cref{lem:b2:realfun:slopes}; the
 other by the mean value theorem)}. Deducir el $C^2$
 criterio $f'' \geq 0$.
 \item Supongamos $f''> 0$ en $I$. Demuestre que $f$ es
 \emph{estrictamente} convexo (desigualdad estricta para $x \neq y$
 y $\lambda \in \intoo01$), y que un estrictamente convexo
 la función satisface la desigualdad de Jensen
 (\cref{thm:b2:realfun:convexreg}~(3)) con igualdad
 \emph{solo} cuando coincidan todos los $x_i$.
 \item Certificar las materias primas de la caja de herramientas: $-\ln$ es estrictamente
 convexo en $\intoo{0}{+\infty}$; $t \mapsto t^r$ es
 estrictamente convexo allí para $r > 1$ y estrictamente cóncavo
 para $0 < r < 1$; $\exp$ es estrictamente convexo en $\R$.
 \item (Desigualdad de Young) Para $a, b \geq 0$, demuestre
 \[
 ab \;\leq\; \frac{a^p}{p} + \frac{b^q}{q},
 \]
 con igualdad si y solo si $a^p = b^q$ \emph{(apply
 the concavity of $\ln$ to the two points $a^p, b^q$
 with weights $\frac1p, \frac1q$)}.
 \item Vuelva a derivar AM-GM ponderado en una línea del
 concavidad de $\ln$:
 \[
 \prod_i x_i^{\lambda_i} \leq \sum_i\lambda_ix_i \qquad
 (x_i > 0),
 \]
 con el caso de igualdad; comparar con la ruta límite de
 \cref{exo:b2:realfun:6}.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Hölder and Minkowski.} Escribir
$\norm{a}_p = \bigl(\sum_i \abs{a_i}^p\bigr)^{1/p}$ y
$\norm{a}_\infty = \max_i\abs{a_i}$.
\begin{enumerate}[resume]
 \item (Hölder) Demostrar
 \[
 \sum_{i=1}^{n}\abs{a_ib_i} \;\leq\;
 \norm a_p\,\norm b_q ,
 \]
 con igualdad si y solo los vectores $(\abs{a_i}^p)$ y
 $(\abs{b_i}^q)$ son proporcionales \emph{(normalize
 $\norm a_p = \norm b_q = 1$ and apply Young termwise)}.
 \item Identificar los casos especiales: $p = q = 2$
 (Cauchy--Schwarz) y el par de puntos finales $(p, q) = (1,
 \infty)$: estado y prueba $\sum\abs{a_ib_i} \leq
 \norm a_1\norm b_\infty$.
 \item (Minkowski) Para $p \geq 1$, demuestre
 \[
 \norm{a + b}_p \leq \norm a_p + \norm b_p
 \]
 \emph{(write $\abs{a_i + b_i}^p \leq \abs{a_i +
 b_i}^{p-1}(\abs{a_i} + \abs{b_i})$ and apply Hölder
 to each product)}. Concluir: $\norm\cdot_p$ es un \omterm{def:b2:nvs:norm}{norma}
 en $\R^n$ por cada $p \in \intco{1}{+\infty}$,
 completando la imagen de \cref{ch:b2:nvs}.
 \item Versiones integrales: para $f, g$ \omterm{def:b2:metric:continuity}{continuo} en
 $\intcc{a}{b}$, declarar y probar Hölder y Minkowski
 para $\norm f_p = \bigl(\int_a^b\abs
 f^p\bigr)^{1/p}$ \emph{(mismas pruebas, con la estricta
 positividad de la integral para la igualdad
 discusión)}.
 \item Demuestre la monotonicidad $\norm a_q \leq \norm a_p$ para $1
 \leq p \leq q$, el límite $\norm a_p \to
 \norm a_\infty $como $p \to \infty$ y a la inversa.
 comparación con la constante aguda:
 \[
 \norm a_p \leq n^{\frac1p - \frac1q}\,\norm a_q
 \]
 \emph{(Hölder against the constant vector)}. Identificar
 los vectores que logran cada igualdad.
 \item (Interpolación) Para $1 \leq p < r < q$ y $\theta \in
 \intoo01$ con $\frac1r = \frac\theta p +
 \frac{1-\theta}q$, demuestre
 \[
 \norm a_r \leq \norm a_p^{\theta}\,
 \norm a_q^{1-\theta}
 \]
 \emph{(apply Hölder with exponents $\frac{p}{\theta
 r}$ and $\frac{q}{(1-\theta)r}$ to $\abs{a_i}^{\theta
 r}\abs{a_i}^{(1-\theta)r}$)}.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The power-mean chain, complete.}
Para $p \neq 0$, configure $M_p = \bigl(\sum_i\lambda_i
x_i^p\bigr)^{1/p}$($x_i > 0$) y $M_0 = \prod_i
x_i^{\lambda_i}$.
\begin{enumerate}[resume]
 \item Demuestre que $p \mapsto M_p$ está aumentando en todos
 $\R^*$: tratar $p < q < 0$ por la identidad recíproca
 $M_{-p}(x) = M_p(1/x)^{-1}$ y puentear a través de $0$ mediante
 mostrando $M_p \leq M_0 \leq M_q$ para $p < 0 < q$
 \emph{(apply the concavity of $\ln$ to $x_i^q$, and the
 reversed inequality for negative exponents)}.
 \item Demuestre los límites $M_p \to \max_i x_i$ como $p \to
 +\infty $y$M_p \to \min_i x_i $como $p \to -\infty$.
 \item Escribe la cadena $\min \leq \mathrm{HM} \leq
 \mathrm{GM} \leq \mathrm{AM} \leq \mathrm{QM} \leq
 \max$ para pesos iguales y demuestra el clásico.
 consecuencia: para $a_1, \dots, a_n$ positivo,
 \[
 \Bigl(\sum_i a_i\Bigr)\Bigl(\sum_i\frac1{a_i}\Bigr)
 \geq n^2 .
 \]
 \item Relacionar medias con \omterm{def:b2:nvs:norm}{normas}: para pesos iguales $\lambda_i =
 \frac1n$, $M_p(x) = n^{-1/p}\norm x_p$. reconciliar el
 dos monotonicidades --- significa \emph{aumentar} con $p$
 mientras que \omterm{def:b2:nvs:norm}{normas} \emph{disminuir} (pregunta 10) --- en uno
 oración sobre el factor $n^{-1/p}$.
 \item Determinar los casos de igualdad a lo largo de toda la cadena de
 Pregunta 14 (ponderaciones positivas): igualdad en cualquier lugar
 obliga a todos los $x_i$ a ser iguales --- la convexidad estricta vale la pena.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Dividends.}
\begin{enumerate}[resume]
 \item (Joven con pomo) Para $a, b \geq 0$ y $\varepsilon >
 0$, probar
 \[
 ab \leq \varepsilon\,\frac{a^p}{p} +
 \varepsilon^{-q/p}\,\frac{b^q}{q},
 \]
 y el caso del caballo de batalla $ab \leq \varepsilon a^2 +
 \frac{b^2}{4\varepsilon}$: el truco de absorción utilizado
 a lo largo del análisis.
 \item (Hacia Carleman) Vamos $c_k = \frac{(k+1)^k}{k^{k-1}}$.
 Pruebe la identidad telescópica $\prod_{k=1}^{n}c_k =
 (n+1)^n$ y deduzca, por AM--GM aplicado a los números
 $c_ka_k$,
 \[
 (a_1a_2\cdots a_n)^{1/n} \leq
 \frac{1}{n(n+1)}\sum_{k=1}^{n} c_k a_k
 \qquad (a_k > 0).
 \]
 \item (desigualdad de Carleman) Suma sobre $n$, intercambia el
 orden de suma (familias \omterm{def:b2:series:summable}{sumable} positivas,
 \cref{thm:b2:series:fubini}), y utilice $\sum_{n \geq
 k}\frac{1}{n(n+1)} = \frac1k $y$c_k/k = \bigl(1 +
 \frac1k\bigr)^k < \eu$ para concluir: por cada
 $\sum a_k$ convergente con términos positivos,
 \[
 \sum_{n=1}^{\infty}(a_1a_2\cdots a_n)^{1/n}
 \;\leq\; \eu\sum_{k=1}^{\infty}a_k .
 \]
 \item Para $f$ \omterm{def:b2:metric:continuity}{continuo} y positivo en $\intcc{0}{1}$,
 probar
 \[
 \Bigl(\int_0^1 f\Bigr)\Bigl(\int_0^1\frac1f\Bigr)
 \geq 1,
 \]
 con igualdad si $f$ es constante
 \emph{(Cauchy--Schwarz on $\sqrt f\cdot\frac1{\sqrt
 f}$)}.
 \item (Geometría de las bolas) Usando el caso de igualdad de
 Minkowski, demuestre que para $1 < p < \infty$ la unidad
 esfera de $\norm\cdot_p$ no contiene ningún segmento (el \omterm{def:b2:nvs:norm}{norma}
 es estrictamente convexo en el sentido de
 \cref{pb:b2:nvs:1}), mientras que para $p = 1$ y $p =
 \infty$ sí: exhibe las piezas planas.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Duality and synthesis.}
\begin{enumerate}[resume]
 \item (Hölder como dualidad) Demuestre que por cada $a \in
 \R^n$,
 \[
 \norm a_p = \max_{\norm b_q \leq 1}\ \sum_i a_ib_i ,
 \]
 exhibiendo una maximización $b$ explícitamente. (La norma $p$
 es el \omterm{def:b2:linalg:dual}{dual} de la norma $q$ --- el de dimensión finita
 germen de la dualidad $L^p$.)
 \item (Momentos) Sea $X$ una variable aleatoria que tome números finitos
 muchos valores positivos $x_i$ con probabilidades
 $\lambda_i$. Reformule la pregunta 12 como: $r \mapsto
 \E[X^r]^{1/r}$ está aumentando --- el momento
 (Lyapunov) desigualdad, para ser reutilizada en
 \cref{ch:b2:randomvar}.
 \item Resuelva con las herramientas nombradas, en dos líneas cada una: (i) para
 positivo $a, b, c$: $a^3 + b^3 + c^3 \geq
 \frac{(a+b+c)^3}{9}$; (ii) para $x_1, \dots,
 x_n $positivo: $\bigl(\sum_i\sqrt{x_i}\bigr)^2 \leq
 n\sum_i x_i$.
 \item (Síntesis) Dibuja la genealogía en cinco frases: acorde
 definición del lema de pendiente; pendientes para soportar líneas de
 Jensen; Concavidad de $\ln$ a Joven a Hölder a
 Minkowski según las normas $p$; Jensen al poder-medio
 cadena a momentos; AM--GM a Carleman. Nombra las cumbres
 (Hölder--Minkowski; Carleman), e indique dónde se encuentra
 La caja de herramientas se titula: los espacios $L^p$ del Año 3.
 volumen, cuyos axiomas son exactamente las preguntas 6 y 8.
\end{enumerate}
\end{problem}
