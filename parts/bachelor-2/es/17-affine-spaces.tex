\chapter{Espacios afines}\label{ch:b2:affine}

Los espacios vectoriales tienen un punto privilegiado --- el origen --- que
La geometría no quiere. Un \emph{\omterm{def:b2:affine:def}{espacio afín}} es un espacio vectorial
que ha olvidado su origen: puntos y vectores se vuelven diferentes
especies, relacionadas por traducción. Este breve capítulo construye el
diccionario (puntos, \omterm{def:b2:affine:barycenter}{baricentros}, \omterm{def:b2:affine:subspace}{afín} subespacios y mapas), el
\omterm{def:b2:affine:subspace}{afín} vista de la convexidad y las herramientas de clasificación utilizadas en el
capítulos de geometría a continuación.

\section{Puntos y vectores}

\begin{definition}\label{def:b2:affine:def}
Un \emph{espacio afín}\index{espacio afín} dirigido por un real
El espacio vectorial $E$ es un conjunto no vacío $\mathcal{E}$ con un mapa.
$(A, B) \mapsto \vect{AB} \in E$ satisfactorio
\[
\vect{AB} + \vect{BC} = \vect{AC}
\quad \text{(Chasles)},
\qquad
\text{for each } A,\ B \mapsto \vect{AB} \text{ is a bijection }
\mathcal{E} \to E .
\]
Se escribe $B = A + u$ para el punto único con $\vect{AB} = u$.
El \emph{dimensión} de $\mathcal{E}$ es $\dim E$. Cada vector
el espacio es un espacio \omterm{def:b2:affine:subspace}{afín} sobre sí mismo ($\vect{AB} = B - A$); cada
elección de origen $O \in \mathcal{E}$ identifica $\mathcal{E}$ con
$E$ vía $M \mapsto \vect{OM}$.
\end{definition}

\begin{example}[Un espacio afín sin natural
origen]\label{ex:b2:affine:planeexample}
El plano de solución $\mathcal E = \{(x, y, z) \in \R^3 : x + y
+ z = 1\}$ no es un subespacio vectorial ($0 \notin \mathcal E$),
pero es un \omterm{def:b2:affine:def}{espacio afín} dirigido por $E = \{x + y + z =
0\}$: para $A, B \in \mathcal E$la diferencia $\vect{AB} = B
- A$cae en $E$ (las sumas se cancelan), Chasles se hereda
de $\R^3$ y $B \mapsto \vect{AB}$ es biyectivo en $E$.
No se distingue ningún punto de $\mathcal E$ --- cualquier elección de
``origen'' $O \in \mathcal E$ funciona igualmente bien, y todos los
las identificaciones $M \mapsto \vect{OM}$ difieren según las traducciones.
Ésta es la situación típica: conjuntos de soluciones de elementos no homogéneos.
problemas lineales (sistemas lineales, diferencial lineal
ecuaciones en \cref{ch:b2:diffeq}) son \omterm{def:b2:affine:subspace}{afín}, nunca lineales,
y el eslogan ``solución particular más núcleo'' es exactamente
la declaración $\mathcal F = A + F$ de la siguiente definición.
\end{example}

\begin{definition}[Barycentro]\label{def:b2:affine:barycenter}
Sea $(A_i, \lambda_i)_{i \leq k}$ puntos ponderados con
$\sum\lambda_i \neq 0$. El \emph{baricentro}\index{baricentro} $G
= \operatorname{bar}\bigl((A_i, \lambda_i)\bigr)$ es el único
punto con
\[
\sum_i \lambda_i\, \vect{GA_i} = 0
\qquad\text{equivalently}\qquad
\vect{OG} = \frac{1}{\sum\lambda_i}\sum_i \lambda_i\,\vect{OA_i}
\quad (\text{any } O).
\]
Los baricentros son \emph{de asociación} (se pueden crear subgrupos de puntos).
reemplazados por su baricentro parcial con el peso sumado) y
invariante bajo reescalado de todos los pesos.
\end{definition}

\begin{proof}[Prueba de existencia y las fórmulas]
Corrija $O$ y escriba $s = \sum_i\lambda_i \neq 0$. Por Chasles,
\[
\sum_i\lambda_i\,\vect{GA_i} = 0
\iff \sum_i\lambda_i\bigl(\vect{GO} + \vect{OA_i}\bigr) = 0
\iff s\,\vect{OG} = \sum_i\lambda_i\,\vect{OA_i},
\]
que determina $G = O + \frac1s\sum_i\lambda_i\vect{OA_i}$
únicamente. \emph{Independencia de $O$:} para otro origen
$O'$,
\[
\frac1s\sum_i\lambda_i\,\vect{O'A_i}
= \frac1s\sum_i\lambda_i\bigl(\vect{O'O} + \vect{OA_i}\bigr)
= \vect{O'O} + \frac1s\sum_i\lambda_i\,\vect{OA_i}
= \vect{O'G} :
\]
el mismo punto $G$. \emph{Asociatividad:} dividir el conjunto de índices
como $I \sqcup J$ con $s_I = \sum_{i\in I}\lambda_i \neq 0$,
y sea $G_I$ el \omterm{def:b2:affine:barycenter}{baricentro} de $(A_i, \lambda_i)_{i\in
I}$, de modo que $\sum_{i\in I}\lambda_i\vect{OA_i} =
s_I\,\vect{OG_I}$. entonces
\[
s\,\vect{OG}
= \sum_{i\in I}\lambda_i\vect{OA_i} +
\sum_{j\in J}\lambda_j\vect{OA_j}
= s_I\,\vect{OG_I} + \sum_{j\in J}\lambda_j\vect{OA_j} :
\]
$G$ es el \omterm{def:b2:affine:barycenter}{baricentro} de $(G_I, s_I)$ junto con $(A_j,
\lambda_j)_{j\in J}$, como se afirma. \emph{Reescalado:} reemplazando
cada $\lambda_i$ por $t\lambda_i$ ($t \neq 0$) multiplica $s$
y la suma ponderada por $t$, dejando $\vect{OG}$ sin cambios.
\end{proof}

\begin{remark}[Errores comunes]
Dos trampas rodean la definición. Primero, si los pesos suman
a \emph{cero}, no hay \omterm{def:b2:affine:barycenter}{baricentro}: el mapa $O \mapsto
\sum\lambda_i\vect{OA_i}$ es entonces independiente de $O$ y
define un \emph{vector}, no un punto --- por ejemplo
$(A, -1;\ B, 1)$ codifica $\vect{AB}$. Haciendo un seguimiento de cuáles
de los dos objetos que produce un cálculo es la mitad de
Higiene baricéntrica. En segundo lugar, los pesos sólo son significativos hacia arriba.
a un factor común distinto de cero; fórmulas como ``las coordenadas
de $G$ son $\lambda_1, \dots, \lambda_k$'' presuponen una
normalización (generalmente $\sum\lambda_i = 1$) y olvido
normalizar es la fuente estándar de ratios incorrectos en un
figura.
\end{remark}

\begin{definition}[Subespacios afines; mapas afines]\label{def:b2:affine:subspace}
Un \emph{subespacio afín} es un conjunto $\mathcal{F} = A + F = \{A + u :
u \in F\}$ con $F$ un subespacio vectorial (su \emph{dirección});
de manera equivalente, un conjunto no vacío estable bajo \omterm{def:b2:affine:barycenter}{baricentros}. afín
los subespacios de $\R^n$ son exactamente los conjuntos de soluciones de sistemas lineales
$MX = B$ (Año 1: solución particular más kernel). Un mapa $f \colon
\mathcal{E} \to \mathcal{E}'$ es \emph{afín}\index{aplicación afín}
cuando conserva \omterm{def:b2:affine:barycenter}{baricentros} --- de manera equivalente cuando
\[
f(A + u) = f(A) + \varphi(u)
\]
para un mapa lineal (único) $\varphi = \vec f$, el \emph{lineal
parte}. Mapas afines de $\R^n$: $X \mapsto MX + C$. Las composiciones son
afín con partes lineales compuestas; $f$ es biyectivo si $\vec f$
es.
\end{definition}

\begin{proof}[Comprobante de equivalencia para mapas]
Si $f(A + u) = f(A) + \varphi(u)$: para un \omterm{def:b2:affine:barycenter}{baricentro} $G$ de $(A_i,
\lambda_i)$, expandir cada punto desde $A$, $f(G) = f(A) +
\varphi(\vect{AG})$y $\varphi(\vect{AG}) =
\frac{\sum\lambda_i\varphi(\vect{AA_i})}{\sum\lambda_i}$: $f(G)$ es
el \omterm{def:b2:affine:barycenter}{baricentro} de las imágenes. Por el contrario, solucione $A$ y defina
$\varphi(u) = \vect{f(A)\,f(A + u)}$. \emph{Homogeneidad:} $A + tu =
\operatorname{bar}\bigl(A, 1-t;\ A + u, t\bigr)$ por cada real
$t$, por lo que se conserva \omterm{def:b2:affine:barycenter}{baricentros} (con pesos reales arbitrarios,
según la hipótesis) da $\varphi(tu) = t\,\varphi(u)$ directamente.
\emph{Aditividad:} $A + u + v = \operatorname{bar}\bigl(A + 2u,
\tfrac12;\ A + 2v, \tfrac12\bigr)$, entonces $\varphi(u + v) =
\tfrac12\varphi(2u) + \tfrac12\varphi(2v) = \varphi(u) +
\varphi(v)$, usando homogeneidad. Por tanto, $\varphi$ es lineal.
\end{proof}

\begin{remark}
La prueba utilizó \omterm{def:b2:affine:barycenter}{baricentros} con \emph{real arbitrario}.
pesos: el paso de homogeneidad lleva a $t$ fuera de $\intcc01$.
Si se supone que un mapa solo conserva \omterm{def:b2:affine:barycenter}{baricentros} con
pesos no negativos --- equivalentemente, puntos medios y segmentos
--- la linealidad del mapa vectorial ya no es gratuita:
uno solo obtiene la linealidad $\Q$, y una hipótesis \omterm{def:b2:metric:continuity}{continuidad} es
necesario para concluir, exactamente como en
\cref{exo:b2:affine:5}. Distinguiendo ``preserva todo
\omterm{def:b2:affine:barycenter}{baricentros}'' de ``conserva combinaciones \omterm{def:b2:affine:convex}{convexo}'' es un
Pequeña pero real sutileza del vocabulario \omterm{def:b2:affine:subspace}{afín}.
\end{remark}

\begin{example}[Geometría baricéntrica clásica]\label{ex:b2:affine:median}
El centroide de un triángulo $ABC$ es el \omterm{def:b2:affine:barycenter}{baricentro} $G =
\operatorname{bar}(A,1; B,1; C,1)$. asociatividad con el
punto medio $A' = \operatorname{bar}(B, 1; C, 1)$ muestra
\[
G = \operatorname{bar}(A, 1;\ A', 2) :
\]
$G$ se encuentra en la mediana $AA'$ en dos tercios de ella --- y de la misma manera
para las otras dos medianas: las tres medianas son concurrentes, en
una línea de cálculo baricéntrico.
\end{example}

\begin{example}[Las bimedianas de un
cuadrilátero]\label{ex:b2:affine:bimedians}
Sea $ABCD$ cualquier cuadrilátero (¡plano o no!) y considere
es \emph{bimedianas}: los segmentos que unen los puntos medios de
lados opuestos, $M_{AB}M_{CD}$ y $M_{BC}M_{DA}$. Introducir
el \omterm{def:b2:affine:barycenter}{baricentro} $G$ de $(A,1; B,1; C,1; D,1)$ y agrupar el
pesa de dos maneras:
\[
G = \operatorname{bar}\bigl(M_{AB}, 2;\ M_{CD}, 2\bigr)
= \operatorname{bar}\bigl(M_{BC}, 2;\ M_{DA}, 2\bigr) :
\]
$G$ es el punto medio de las bimedianas \emph{ambos} --- entonces las dos
bimedianas siempre se bisecan entre sí, y el cuadrilátero de
los cuatro puntos medios es un paralelogramo (sus diagonales son las
bimedianas). Sin análisis de casos, sin coordenadas, y el
El argumento sobrevive sin cambios para un cuadrilátero sesgado en
$\R^3$, donde ya estaría disponible una prueba basada en imágenes
delicado: a la asociatividad no le importa la dimensión.
\end{example}

\begin{example}[Clasificando un aplicación afín, comienza a
terminar]\label{ex:b2:affine:dilation}
Deje $f(x, y) = (2x - 1,\ 3y - 4)$ en $\R^2$. Su parte lineal
es $\varphi = \operatorname{diag}(2, 3)$, cuyo \omterm{def:b2:reduction:eigen}{espectro}
$\{2, 3\}$ evita $1$: según el criterio del punto fijo demostrado
a continuación (\cref{prop:b2:affine:fixedpoint}), $f$ tiene exactamente uno
punto fijo, encontrado resolviendo
\[
x = 2x - 1, \qquad y = 3y - 4
\qquad\Longrightarrow\qquad \Omega = (1, 2).
\]
Centrándose en $\Omega$ (establezca $x = 1 + u$, $y = 2 + v$):
\[
f(1 + u,\ 2 + v) = (1 + 2u,\ 2 + 3v) :
\]
en el marco en $\Omega$, $f$ \emph{is} su parte lineal, una
dilatación anisotrópica que se extiende por $2$ horizontalmente y $3$
verticalmente desde el centro $(1, 2)$. La lección general: una
\omterm{def:b2:affine:subspace}{aplicación afín} es "mapa lineal más datos de ubicación", y el
Los datos de ubicación colapsan en un origen bien elegido siempre que
$1$ no es un \omterm{def:b2:reduction:eigen}{valor propio}. Por el contrario, traducir el origen
mal \emph{crea} los términos constantes: \omterm{def:b2:affine:subspace}{afín} la geometría es
el arte de elegir dónde poner $0$.
\end{example}

\begin{remark}[Método: concurrencia y alineación por
baricentro]
\cref{ex:b2:affine:median} es una instancia de un general
receta. Para demostrar que tres cevianas de un triángulo son

$(A, 1; B, 1; C, 1)$ hace todo el trabajo; para corte de cevianas
los lados en proporciones prescritas, los pesos se leen en el
proporciones. Para probar tres puntos \emph{alineado}, escribe uno como

el criterio \omterm{def:b2:linalg:det}{determinante} de \cref{exo:b2:affine:11}. ambos
Las recetas reemplazan el ingenio geométrico por la contabilidad del peso.
para eso precisamente está el cálculo baricéntrico.
\end{remark}

\section{Convexidad, afinidad}

\begin{definition}\label{def:b2:affine:convex}
Un subconjunto $C$ de un \omterm{def:b2:affine:def}{espacio afín} es \emph{convexo}\index{conjunto convexo} cuando contiene
cada \omterm{def:b2:affine:barycenter}{baricentro} con \emph{no negativo} pesos de sus puntos ---
de manera equivalente, cada segmento $\intcc{A}{B} = \{\operatorname{bar}(A,
1-t; B, t) : t \in \intcc{0}{1}\}$ entre sus puntos. el
\emph{casco convexo}\index{casco convexo} $\operatorname{conv}(S)$ es el conjunto de todos
peso no negativo \omterm{def:b2:affine:barycenter}{baricentros} de puntos de $S$ --- el más pequeño
conjunto convexo que contiene $S$.
\end{definition}

\begin{example}[Los epígrafes son convexos
conjuntos]\label{ex:b2:affine:epigraph}
La región $C = \{(x, y) : y \geq x^2\}$ encima de la parábola
es \omterm{def:b2:affine:convex}{convexo}: para $(x_1, y_1), (x_2, y_2) \in C$ y $t \in
\intcc01$, la desigualdad de convexidad de la función cuadrada
da
\[
\bigl((1-t)x_1 + tx_2\bigr)^2 \leq (1-t)x_1^2 + tx_2^2
\leq (1-t)y_1 + ty_2 ,
\]
entonces el \omterm{def:b2:affine:barycenter}{baricentro} permanece por encima de la parábola. el computo
es general: $\{y \geq f(x)\}$ es \omterm{def:b2:affine:convex}{convexo} exactamente cuando $f$ es un
función convexa --- \omterm{def:b2:affine:convex}{convexo} \emph{conjuntos} y \omterm{def:b2:affine:convex}{convexo}
\emph{funciones} (\cref{ch:b2:realfun}) son dos caras de una
noción, siendo los epígrafes el diccionario. esta es la geométrica
razón por la que existen líneas de soporte para funciones convexas, el hecho
eso probará la desigualdad de Jensen en
\cref{ch:b2:randomvar}.
\end{example}

\begin{example}[Generadores redundantes de un casco
convexo]\label{ex:b2:affine:squarehull}
Sea
\[
S = \{(0,0), (2,0), (2,2), (0,2), (1,1)\}.
\]
El quinto punto es el \omterm{def:b2:affine:barycenter}{baricentro}
\[
(1,1) = \operatorname{bar}\bigl((0,0), \tfrac12;\ (2,2),
\tfrac12\bigr),
\]
entonces ya se encuentra en el casco de los otros cuatro:
$\operatorname{conv}(S)$ es el cuadrado con las cuatro esquinas
como vértices. En general, un punto de $S$ que es un
peso no negativo \omterm{def:b2:affine:barycenter}{baricentro} de los puntos \emph{otro} de
$S$ se puede eliminar sin cambiar el casco; los puntos que
nunca se puede borrar (aquí las cuatro esquinas) son las
\emph{puntos extremos} del casco. Determinarlos es un puro
Cálculo \omterm{def:b2:affine:barycenter}{baricentro}: $(2,0)$, digamos, no se puede escribir como
$\operatorname{bar}$ de los puntos restantes con no negativos
pesos, porque la primera coordenada obligaría a todos los pesos
en puntos con $x = 2$, y la segunda coordenada luego
falla. Las preguntas sobre la convexidad se reducen, una y otra vez, a
Resolución de pequeños sistemas ponderados.
\end{example}

\begin{theorem}[Carathéodory]\label{thm:b2:affine:caratheodory}
En un \omterm{def:b2:affine:def}{espacio afín} de dimensión $n$, cada punto de
$\operatorname{conv}(S)$ es un \omterm{def:b2:affine:barycenter}{baricentro} de como máximo puntos $n + 1$
de $S$.\index{Teorema de Carathéodory}
\end{theorem}

\begin{proof}
Deja $G = \operatorname{bar}(A_0, \lambda_0; \dots; A_k, \lambda_k)$
con $\lambda_i > 0$, $\sum\lambda_i = 1$ y $k + 1 > n + 1$
puntos. Los vectores $k$ $\vect{A_0A_i}$ ($i \geq 1$) están vinculados
($k > n$): $\sum_{i\geq1}\mu_i \vect{A_0A_i} = 0$ no trivial;
configurando $\mu_0 = -\sum_{i\geq1}\mu_i$, obtenemos pesos $(\mu_i)$
con $\sum\mu_i = 0$, $\sum \mu_i\,\vect{OA_i} = 0$ (cualquier $O$), no
todo cero. Luego por cada $t$ real los pesos $\lambda_i - t\mu_i$
todavía suma $1$ y, desde $\sum_i\mu_i\vect{OA_i} = 0$,
\[
\sum_i(\lambda_i - t\mu_i)\,\vect{OA_i}
= \sum_i\lambda_i\,\vect{OA_i} :
\]
Producen el punto \emph{mismo} $G$. Ahora deslice $t$ desde
$0$: algunos $\mu_i$ son positivos (suman cero y no son
todo cero), entonces
\[
t^* = \min\Bigl\{\frac{\lambda_i}{\mu_i} : \mu_i > 0\Bigr\}
\]
está bien definido y es positivo. En $t = t^*$: para índices con
$\mu_i > 0$, $\lambda_i - t^*\mu_i \geq 0$ por minimalidad,
con igualdad en un índice minimizador; para índices con $\mu_i
\leq 0$, $\lambda_i - t^*\mu_i \geq \lambda_i > 0$. Todos
los pesos siguen siendo no negativos y al menos uno ha muerto: $G$ es
reescrito como \omterm{def:b2:affine:barycenter}{baricentro} de menos puntos. Iterar mientras más
Quedan más de $n + 1$ puntos.
\end{proof}

En el plano ($n = 2$): cada punto del \omterm{def:b2:affine:convex}{casco convexo} de un finito
conjunto se encuentra en un triángulo con vértices en el conjunto --- el elemento geométrico
Contenido de Carathéodory, utilizado en optimización y probabilidad.
(mezclas) por igual.
\end{example}

\section{Herramientas de clasificación afines}

\begin{proposition}[Puntos fijos de mapas afines]\label{prop:b2:affine:fixedpoint}
Sea $f$ un endomorfismo \omterm{def:b2:affine:subspace}{afín} de un \omterm{def:b2:affine:def}{afín
espacio} de dimensión finita con parte lineal $\varphi$. Si $1 \notin
\operatorname{Sp}(\varphi)$, entonces $f$ tiene exactamente un punto fijo
$\Omega$, y en la vectorialización en $\Omega$, $f$ \emph{is}
su parte lineal. (Traducciones, con $\varphi = \mathrm{id}$ y no
punto fijo, son la obstrucción básica.)
\end{proposition}

\begin{proof}
Corrige $O$ y escribe $f(O + x) = f(O) + \varphi(x)$. el punto
$O + x$ se fija si $O + x = f(O) + \varphi(x)$, es decir
\[
(\mathrm{id} - \varphi)(x) = \vect{O f(O)} .
\]
En dimensión finita, $\mathrm{id} - \varphi$ es invertible
si $0$ no es un \omterm{def:b2:reduction:eigen}{valor propio} de $\mathrm{id} - \varphi$, si
$1 \notin \operatorname{Sp}\varphi$ --- y en ese caso el
La ecuación mostrada tiene exactamente una solución $x^*$, lo que da la
punto fijo único $\Omega = O + x^*$. Centrado: para cualquier
vector $u$,
\[
f(\Omega + u) = f(\Omega) + \varphi(u) = \Omega + \varphi(u),
\]
entonces en el cuadro con origen $\Omega$ el mapa dice $u \mapsto
\varphi(u)$: puramente lineal. Cuando $1 \in
\operatorname{Sp}\varphi$, no existe ningún punto fijo (el
la ecuación mostrada puede no tener solución, como en el caso de una traducción)
o un subespacio \omterm{def:b2:affine:subspace}{afín} completo de ellos (agregue cualquier \omterm{def:b2:reduction:eigen}{vector propio}
de \omterm{def:b2:reduction:eigen}{valor propio} $1$ a una solución): la unicidad es exactamente la
condición espectral.
\end{proof}

\begin{example}[Isometrías planas, completadas]\label{ex:b2:affine:isometries}
Una isometría \omterm{def:b2:affine:subspace}{afín} del plano euclidiano tiene parte lineal en
$O(2)$: una rotación $R_\theta$ o una reflexión (volumen Año 1). si
$\theta \neq 0$: $1 \notin \operatorname{Sp} R_\theta$, entonces el mapa
es un \emph{rotación sobre un centro único}
(\cref{prop:b2:affine:fixedpoint}). Si la parte lineal es una
reflexión: ya sea una reflexión en un eje (existen puntos fijos) o
a \emph{reflejo de deslizamiento} (reflexión compuesta con una traducción
a lo largo del eje, sin punto fijo). Con las traducciones, este es el
\omterm{def:b2:metric:complete}{completo} clasificación de isometrías planas.
\end{example}

\begin{remark}[Las isometrías del plano, de un vistazo]
Recopilación de los casos: identidad; traducciones ($\vec f =
\mathrm{id}$, sin punto fijo a menos que sea trivial); rotaciones
(parte lineal $R_\theta$, $\theta \neq 0$: un centro);
reflexiones (parte lineal una reflexión, una línea de fijo
puntos); reflejos de deslizamiento (misma parte lineal, sin fijas)
punto). Cuatro familias más la identidad, cada una reconocida por
sólo dos datos: la parte lineal y el conjunto de punto fijo ---
el patrón de \cref{prop:b2:affine:fixedpoint} hecho
exhaustivo.
\end{remark}

\begin{example}[Un reflejo deslizante, atrapado en el
acto]\label{ex:b2:affine:glide}
Deje $f(x, y) = (y + 1,\ x + 1)$. La parte lineal $(x, y)
\mapsto (y, x)$es el reflejo en la diagonal $y = x$, entonces
$1 \in \operatorname{Sp}\vec f$ y
\cref{prop:b2:affine:fixedpoint} guarda silencio. Puntos fijos
necesitaría $x = y + 1$ y $y = x + 1$ simultáneamente:
imposible --- no existe ninguno, por lo que $f$ no es un reflejo. cuadratura
resuelve la clasificación:
\[
f\bigl(f(x, y)\bigr) = f(y + 1,\ x + 1) = (x + 2,\ y + 2),
\]
la traducción por $(2, 2)$: $f$ es el \emph{deslizarse
reflexión} con eje la línea $y = x$ (desplazada
apropiadamente: el punto medio de $M$ y $f(M)$ siempre se encuentra en
$y = x + {}$ constante, aquí $y = x$, como se verifica en $M = (0,
0) \mapsto (1,1)$) y vector de deslizamiento $(1, 1)$, la mitad de $f
\circ f$. Comparar con \cref{exo:b2:affine:6}, donde lo mismo
parte lineal pero una constante diferente produjo una honesta
reflexión: con \omterm{def:b2:reduction:eigen}{valor propio} $1$ presente, el término constante
decide todo.
\end{example}

\begin{example}[Las recursiones afines son afines
dinámico]\label{ex:b2:affine:recursion}
La recursividad clásica $u_{n+1} = au_n + b$ ($a \neq 1$)
itera el \omterm{def:b2:affine:subspace}{aplicación afín} $f(x) = ax + b$ de la línea, cuyo
La pieza lineal $a$ evita el \omterm{def:b2:reduction:eigen}{valor propio} $1$: hay un único
punto fijo $\omega = \frac{b}{1-a}$, y volver a centrar allí
(el caso unidimensional de la proposición anterior) se convierte
$f$ en multiplicación por $a$:
\[
u_{n+1} - \omega = a\,(u_n - \omega)
\qquad\Longrightarrow\qquad
u_n = \omega + a^n(u_0 - \omega) .
\]
Para $u_{n+1} = \frac{u_n}2 + 3$: $\omega = 6$ y $u_n = 6 +
(u_0 - 6)2^{-n} \to 6$. La receta enseñada para tal
recursiones en \cref{ch:b2:series} --- ``resta el valor fijo
point'' --- es exactamente la vectorialización de un \omterm{def:b2:affine:subspace}{aplicación afín}
en su punto fijo; La convergencia para $\abs a < 1$ es la
fenómeno de contracción en el que se convirtió \cref{ch:b2:metric}
el teorema del punto fijo de Banach. Una idea, tres capítulos.
\end{example}

\begin{example}[Encontrar el centro de una rotación]\label{ex:b2:affine:rotationcenter}
Deje $f(x, y) = (-y + 2,\ x)$. La parte lineal es
$\varphi(x, y) = (-y, x)$: la rotación del ángulo $\frac\pi2$,
cuyo \omterm{def:b2:reduction:eigen}{espectro} $\{\iu, -\iu\}$ evita $1$. Por
\cref{prop:b2:affine:fixedpoint} hay exactamente uno arreglado
punto: $x = -y + 2$ y $y = x$ dan $x = 1$, $y = 1$, entonces
$\Omega = (1, 1)$ y $f$ \emph{is} la rotación del centro
$(1, 1)$ y ángulo $\frac\pi2$. La lección general: cuando $1
\notin \operatorname{Sp}\vec f$, clasificar $f$ cuesta uno
sistema lineal --- la geometría está completamente en la parte lineal,
la aritmética por completo en localizar el centro.
\end{example}

plano es un punto $c$ (no necesariamente en $S$) tal que cada
el semiplano cerrado que contiene $c$ contiene al menos $n/3$
puntos de $S$.
\begin{enumerate}[resume]
 \item (Dimensión $1$) Para reales $x_1 \leq \dots \leq x_n$,
 demuestre que la mediana $c = x_{\lceil n/2 \rceil}$
 satisface: cada media línea cerrada que contiene $c$
 contiene al menos $n/2$ del $x_i$.
 \item (Lema de conteo) Si $A, B, C$ son subconjuntos de $S$ con
 $\abs A, \abs B, \abs C > \tfrac{2n}3$, muestra $A \cap
 B \cap C \neq \emptyset$.
 \item Dejemos que $m = \floor{2n/3} + 1$ y $\mathcal F$ sean
 la familia (finita) del \omterm{def:b2:affine:convex}{cascos convexos}
 $\operatorname{conv}(T)$, $T \subseteq S$, $\abs T =
 m$. Demuestre que tres miembros cualesquiera de $\mathcal F$ tienen
 un punto en común, y deducir de Helly un punto $c$
 común a todos ellos.
 \item Demuestre que este $c$ es un punto central de $S$: el
 \emph{teorema del punto central}. \emph{(If a closed
 half-plane through $c$ contained fewer than $n/3$
 points, its \omterm{def:b2:metric:topology}{abierto} complement would contain a set $T$
 of $m$ points, and $\operatorname{conv}(T)$ would
 avoid $c$.)}
 \item Nitidez: deja $n = 3k$ y coloca puntos $k$ en cada
 de tres discos de pequeño radio $\varepsilon$ centrados
 en los vértices de un gran triángulo. Muéstralo por
 \emph{cada} punto $c$ del avión algunos cerrados
 el semiplano que contiene $c$ contiene como máximo $n/3$
 puntos de $S$, por lo que la constante $1/3$ no puede
 mejorado. \emph{(Among the three directions from $c$
 to the disk centers, two make an angle at most
 $2\pi/3$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Jung's theorem and synthesis.}
\begin{enumerate}[resume]
 \item (Lema del triángulo) Sean $P, Q, R$ tres puntos con
 distancias por pares $\leq 1$. Muestre que se encuentran en un
 disco cerrado de radio $1/\sqrt3$. \emph{(If some
 angle is $\geq \pi/2$, take the disk on the longest
 side as diameter, using the median formula
 $\norm{RM}^2 = \tfrac12\norm{RP}^2 +
 \tfrac12\norm{RQ}^2 - \tfrac14\norm{PQ}^2$; if the
 triangle is acute, bound the circumradius $a/(2\sin
 \widehat A)$ using its largest angle, which lies in
 $\intco{\pi/3}{\pi/2}$.)}
 \item (Jung) Deducir: cada \omterm{def:b2:metric:compact}{compacto} subconjunto del plano de
 diámetro $\leq 1$ está contenido en un disco cerrado de
 radio $1/\sqrt3$.
 \item Nitidez: para el triángulo equilátero $A_1A_2A_3$
 del lado $1$ con centroide $G$, prueba Leibniz
 identidad $\sum_i \norm{\vect{OA_i}}^2 =
 3\norm{\vect{OG}}^2 + \sum_i \norm{\vect{GA_i}}^2$
 para cada punto $O$ y concluir que cualquier disco
 que contiene los tres vértices tiene radio $\geq
 1/\sqrt3$, siendo igual solo para el circundisco.
 \item (Helly en $\R^n$) Enuncie y demuestre el teorema de Helly en
 $\R^n$: si un número finito de \omterm{def:b2:affine:convex}{conjuntos convexos} son tales que
 cualquier $n + 1$ de ellos se cruza, entonces todos
 cruzarse. \emph{(Radon's lemma
 \cref{exo:b2:affine:7} handles $n + 2$ sets; then
 induct as in question 12.)}
 \item Síntesis. Montar la cadena
 \[
 \text{affine dependence} \Rightarrow \text{Radon}
 \Rightarrow \text{Helly} \Rightarrow
 \text{centerpoint and Jung},
 \]
 indicando en una oración cada uno: donde álgebra lineal
 entra, por donde entran los signos de los pesos, por donde
 entra la convexidad, y en qué único paso se utilizó el
 dimensión del avión. ¿Qué significan las constantes $3$ (en
 Helly), $1/3$ (punto central) y $1/\sqrt3$ (Jung)
 convertirse en $\R^n$? (Declaración sin pruebas).
\end{enumerate}
\end{problem}
