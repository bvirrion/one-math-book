\section{Superficies parametrizadas}

\begin{definition}[Superficie regular parametrizada]\label{def:b2:surfaces:param}
Sea $U \subseteq \R^2$ \omterm{def:b2:metric:topology}{abierto}. A \emph{parametrizado
superficie}\index{superficie parametrizada} de
la clase $\mathcal{C}^k$ ($k \geq 1$) es un mapa $\sigma \colon U \to
\R^3$, $(u, v) \mapsto \sigma(u, v)$, de clase $\mathcal{C}^k$. un
el punto es \emph{regular} si los vectores de derivada parcial
\[
\sigma_u(u, v) = \frac{\partial\sigma}{\partial u}(u,v),
\qquad
\sigma_v(u, v) = \frac{\partial\sigma}{\partial v}(u,v)
\]
son linealmente independientes, es decir\ $\sigma_u \wedge \sigma_v \neq 0$;
la superficie es \emph{regular} si todos los puntos lo son.
\end{definition}

\begin{example}[Las tres descripciones estándar]\label{ex:b2:surfaces:standard}
\leavevmode
\begin{enumerate}
\item \emph{Gráfico:} $\sigma(u, v) = (u,\ v,\ f(u, v))$ para $f \in
\mathcal{C}^1(U)$. Siempre regular: $\sigma_u = (1, 0, f_u)$ y
$\sigma_v = (0, 1, f_v)$ son independientes.
\item \emph{Esfera (coordenadas esféricas):} para el ámbito de
radio $R$,
\[
\sigma(\theta, \varphi)
= (R\cos\theta\cos\varphi,\ R\sin\theta\cos\varphi,\
R\sin\varphi),
\qquad
(\theta, \varphi) \in \R \times
\bigl(-\tfrac\pi2, \tfrac\pi2\bigr),
\]
siendo $\theta$ la longitud y $\varphi$ la latitud. uno comprueba
$\norm{\sigma_\theta \wedge \sigma_\varphi} = R^2\cos\varphi > 0$:
regular lejos de los polos (que este gráfico omite).
\item \emph{Conjunto de niveles:} $S = \{(x,y,z) : f(x, y, z) = c\}$ donde
$f$ es $\mathcal{C}^1$ y $\nabla f \neq 0$ en $S$. Cerca de cada uno
punto, una coordenada se puede expresar en función de la otra
dos por el teorema de la función implícita (\cref{ch:b2:diffcalc}), por lo que
$S$ es localmente un gráfico.
\end{enumerate}
\end{example}

\section{Plano tangente y normal}

\begin{definition}[Plano tangente]\label{def:b2:surfaces:tangent}
Dejemos que $\sigma$ sea regular en $(u_0, v_0)$, $M_0 = \sigma(u_0, v_0)$.
El \emph{plano tangente}\index{plano tangente} $T_{M_0}S$ es el avión que pasa por $M_0$
dirigido por $\operatorname{Vect}(\sigma_u, \sigma_v)$ (parciales en
$(u_0, v_0)$). El \emph{unidad normal}\index{unidad normal} es
\[
n(u_0, v_0)
= \frac{\sigma_u \wedge \sigma_v}
 {\norm{\sigma_u \wedge \sigma_v}} .
\]
\end{definition}

\begin{proposition}[Los vectores tangentes son velocidad.
vectores]\label{prop:b2:surfaces:velocity}
La dirección de $T_{M_0}S$ es exactamente el conjunto de vectores.
$\gamma'(0)$, donde $\gamma = \sigma \circ c$ abarca el
$\mathcal{C}^1$ curvas dibujadas en la superficie hasta $M_0$ (es decir,\
$c \colon (-\varepsilon, \varepsilon) \to U$ es $\mathcal{C}^1$
con $c(0) = (u_0, v_0)$).
\end{proposition}

\begin{proof}
Si $c(t) = (u(t), v(t))$, la regla de la cadena
(\cref{ch:b2:diffcalc}) da
\[
\gamma'(0) = u'(0)\,\sigma_u + v'(0)\,\sigma_v
\in \operatorname{Vect}(\sigma_u, \sigma_v).
\]
Por el contrario, el vector $a\sigma_u + b\sigma_v$ se obtiene mediante el
curva $c(t) = (u_0 + at,\ v_0 + bt)$, que permanece en el conjunto \omterm{def:b2:metric:topology}{abierto}
$U$ para $\abs t$ pequeño.
\end{proof}

\begin{proposition}[Normal de una superficie nivelada]\label{prop:b2:surfaces:gradient}
Sea $S = \{f = c\}$ con $f$ de clase $\mathcal{C}^1$ y $\nabla
f(M_0) \neq 0$. Entonces el \omterm{def:b2:surfaces:tangent}{plano tangente} de $S$en $M_0$ es el
plano que pasa por $M_0$ ortogonal a $\nabla f(M_0)$:
\[
T_{M_0}S :\quad
\langle \nabla f(M_0),\ M - M_0\rangle = 0 .
\]
\end{proposition}

\begin{proof}
Para cualquier curva $\gamma$ dibujada de $S$ a $M_0$, $f(\gamma(t)) =
c $es idéntica, por lo que la regla de la cadena da $\langle
\nabla f(M_0), \gamma'(0)\rangle = 0$: todos los vectores de velocidad son
ortogonal al gradiente, por lo que la dirección tangente está contenida
en el avión $\nabla f(M_0)^\perp$. Ambos son $2$-dimensionales
subespacios --- la dirección tangente porque $S$ es localmente un
gráfico regular (\cref{ex:b2:surfaces:standard}), el ortogonal
complemento porque $\nabla f(M_0) \neq 0$ --- por lo tanto son
igual.
\end{proof}

\begin{example}\label{ex:b2:surfaces:spheretangent}

\end{example}

\begin{example}[Planos tangentes de cuádricas: el
regla de polarización]\label{ex:b2:surfaces:polarization}
Sean $S : xy + yz + zx = 1$ y $M_0 = (1, 1, 0) \in S$. aquí
$\nabla f = (y + z,\ x + z,\ x + y)$, entonces $\nabla f(M_0) = (1,
1, 2)$ y \omterm{def:b2:surfaces:tangent}{plano tangente} son
\[
(x - 1) + (y - 1) + 2z = 0,
\qquad\text{i.e.}\qquad x + y + 2z = 2 .
\]
La misma respuesta proviene del \emph{\omterm{def:b2:quadratic:def}{polarización} rule} que
generaliza \cref{ex:b2:surfaces:spheretangent} y
\cref{exo:b2:surfaces:2}: en la ecuación de \omterm{pb:b2:surfaces:1}{cuádrico}, reemplace
$x^2$ por $x_0x$, y cada producto $xy$ por $\frac{x_0y +
y_0x}2$ (y cíclicamente):
\[
\frac{x_0y + y_0x}2 + \frac{y_0z + z_0y}2 + \frac{z_0x +
x_0z}2 = 1
\;\xrightarrow{\ M_0 = (1,1,0)\ }\;

\]
que es $x + y + 2z = 2$ nuevamente. La regla funciona porque
$\nabla$ de \omterm{def:b2:quadratic:def}{forma cuadrática} es la forma bilineal asociada
evaluado contra el punto base --- la tangencia a \omterm{pb:b2:surfaces:1}{cuádrico} es

\end{example}

\section{La primera forma fundamental.}

\begin{definition}[Primera forma fundamental]\label{def:b2:surfaces:fff}
Dejemos que $\sigma \colon U \to \R^3$ sea un $\mathcal{C}^1$ normal.
superficie. Su \emph{primera forma fundamental}\index{primer fundamental
forma} en $(u,v)$ es el
positivo definitivo \omterm{def:b2:quadratic:def}{forma cuadrática} en $\R^2$
\[
I(h, k) = \norm{h\,\sigma_u + k\,\sigma_v}^2
= E\,h^2 + 2F\,hk + G\,k^2,
\]
donde
\[
E = \norm{\sigma_u}^2,
\qquad
F = \langle \sigma_u, \sigma_v\rangle,
\qquad
G = \norm{\sigma_v}^2 .
\]
\end{definition}

\begin{remark}
$I$ es la restricción del producto escalar euclidiano ambiental a
el \omterm{def:b2:surfaces:tangent}{plano tangente}, leído en la base $(\sigma_u, \sigma_v)$: es
positivo definido precisamente porque $\sigma_u, \sigma_v$ son
independiente (\cref{ch:b2:quadratic}). Cada cantidad métrica en el
superficie --- \omterm{def:b2:curves:length}{longitudes} de curvas dibujadas, ángulos entre ellas, \omterm{def:b2:surfaces:area}{áreas} ---
se calcula solo a partir de $E, F, G$. Dos superficies con el mismo $E,
F, G$ en parámetros adecuados son \emph{isométrico} incluso si se asientan
diferente en el espacio: este es el punto de partida de la intrínseca
geometría.
\end{remark}

\begin{example}[Ángulos entre coordenadas
curvas]\label{ex:b2:surfaces:coordangle}
El \omterm{def:b2:surfaces:fff}{primera forma fundamental} también mide ángulos: el
Las curvas de coordenadas $u \mapsto \sigma(u, v_0)$ y $v \mapsto
\sigma(u_0, v)$se encuentran en el ángulo $\theta$ con
\[
\cos\theta = \frac{\langle\sigma_u,
\sigma_v\rangle}{\norm{\sigma_u}\,\norm{\sigma_v}}
= \frac{F}{\sqrt{EG}} :
\]
el coeficiente único $F$ decide la ortogonalidad del
red de parámetros. Para el gráfico de esferas y el helicoidal, $F =
0$: los meridianos cortan paralelos y las hélices cortan la horizontal
fallos, en ángulos rectos --- por eso su \omterm{def:b2:surfaces:area}{área}
Los integrandos colapsaron en $\sqrt{EG}$. Para un gráfico,
$F = f_xf_y$ desaparece sólo cuando una derivada parcial:
la red de coordenadas de un gráfico inclinado es \emph{no}
ortogonal, aunque la red $(x, y)$ de abajo lo es. cuando
Los cálculos en una superficie parecen pesados, el primer paso es
busque un gráfico con $F = 0$.
\end{example}

\begin{example}[El gráfico de silla de montar.]\label{ex:b2:surfaces:saddlechart}
Para el sillín $z = xy$ con tabla $\sigma(u, v) = (u, v,
uv)$:
\[
\sigma_u = (1, 0, v), \qquad \sigma_v = (0, 1, u),
\qquad
E = 1 + v^2, \quad F = uv, \quad G = 1 + u^2 ,
\]
y $EG - F^2 = 1 + u^2 + v^2 > 0$: regular en todas partes. el
dos curvas de coordenadas que pasan por un punto son \emph{recto
lineas} de $\R^3$ (fijar $u$ o arreglar $v$: las reglas de la
silla con doble regla), pero $F \neq 0$ fuera de los ejes: reglas
que pasan por un punto genérico no son ortogonales. Ambas sentencias mienten
en el \omterm{def:b2:surfaces:tangent}{plano tangente}, que abarcan --- por lo que el \omterm{def:b2:surfaces:tangent}{tangente
avión} corta la superficie a lo largo de dos líneas enteras, el extremo
opuesto de la esfera, cuyo \omterm{def:b2:surfaces:tangent}{planos tangentes} se toca a la vez
solo punto. El signo del "contacto de segundo orden" entre
una superficie y su \omterm{def:b2:surfaces:tangent}{planos tangentes} es una historia \omterm{thm:b2:curves:frenet2d}{curvatura},

\end{example}

\begin{proposition}[Longitud de una curva dibujada en un
superficie]\label{prop:b2:surfaces:curvelength}
Si $\gamma(t) = \sigma(u(t), v(t))$, $t \in [a, b]$, es
$\mathcal{C}^1$, entonces
\[
L(\gamma) = \int_a^b
\sqrt{E\,u'^2 + 2F\,u'v' + G\,v'^2}\;\dd t ,
\]
con $E, F, G$ evaluado en $(u(t), v(t))$.
\end{proposition}

\begin{proof}
$\gamma' = u'\sigma_u + v'\sigma_v$ por la regla de la cadena, por lo que
$\norm{\gamma'}^2 = I(u', v') = Eu'^2 + 2Fu'v' + Gv'^2$; integrar
$\norm{\gamma'}$ (\cref{def:b2:curves:length}).
\end{proof}

\begin{lemma}[Identidad Lagrange]\label{lem:b2:surfaces:lagrange}
Para todos los $a, b \in \R^3$:
$\norm{a \wedge b}^2 = \norm a^2 \norm b^2 - \langle a, b\rangle^2$.
en particular
\[
\norm{\sigma_u \wedge \sigma_v} = \sqrt{EG - F^2}.
\]
\end{lemma}

\begin{proof}
Ambos lados no cambian si reemplazamos $b$ por su componente $b_\perp
= b - \frac{\langle a, b\rangle}{\norm a^2}a $ortogonal a $a$
(para $a \neq 0$; el caso $a = 0$ es trivial): el lado izquierdo
porque $a \wedge a = 0$, el lado derecho expandiendo $\norm
{b_\perp}^2 = \norm b^2 - \frac{\langle a,b\rangle^2}{\norm a^2}$
y $\langle a, b_\perp\rangle = 0$. Por lo que basta con demostrar la
identidad para $a, b$ ortogonal, donde se lee $\norm{a \wedge
b} = \norm a \norm b$: verdadero, ya que para vectores ortogonales el
El producto cruzado tiene \omterm{def:b2:nvs:norm}{norma} $\norm a\norm b\,\abs{\sin\frac\pi2}$. el
La fórmula mostrada es el caso $a = \sigma_u$, $b = \sigma_v$.
\end{proof}

\begin{remark}
La identidad de Lagrange dice que $EG - F^2 = \det\begin{psmallmatrix}
E & F\\ F & G\end{psmallmatrix}$ es el \emph{Gram
\omterm{def:b2:linalg:det}{determinante}} de $(\sigma_u, \sigma_v)$: el \omterm{def:b2:surfaces:area}{área} al cuadrado de
el paralelogramo que abarcan. Regularidad, positiva
precisión del \omterm{def:b2:surfaces:fff}{primera forma fundamental} y positividad del
el Gram \omterm{def:b2:linalg:det}{determinante} son tres frases de una condición ---
es por eso que el integrando \omterm{def:b2:surfaces:area}{área} a continuación nunca desaparece en un
gráfico regular.
\end{remark}

\begin{definition}[Área]\label{def:b2:surfaces:area}
Sea $\sigma \colon U \to \R^3$ una inyectiva regular.
superficie $\mathcal{C}^1$ y $K \subseteq U$ un dominio \omterm{def:b2:metric:compact}{compacto} en
qué integrales dobles tienen sentido (\cref{ch:b2:multint}). el
\emph{área}\index{área!de una superficie} de la pieza $\sigma(K)$ es
\[
\mathcal{A}
= \iint_K \norm{\sigma_u \wedge \sigma_v}\,\dd u\,\dd v
= \iint_K \sqrt{EG - F^2}\;\dd u\,\dd v .
\]
\end{definition}

\sigma_v}\,\dd u\,\dd v$: la definición integra el local
factor de distorsión de área, exactamente como \omterm{def:b2:curves:length}{longitud del arco} integra el local
velocidad. La coherencia con \omterm{def:b2:curves:reparam}{cambio de parametros} es
\cref{exo:b2:surfaces:8}; coherencia con el cambio de variables
La fórmula para integrales dobles se analiza en

\end{remark}

-\sin\theta\sin\varphi,\ \cos\varphi),

entonces $E = R^2\cos^2\varphi$, $F = 0$, $G = R^2$ y $\sqrt{EG - F^2}
= R^2\cos\varphi$. Por lo tanto
\[
\mathcal{A}
= \int_{-\pi/2}^{\pi/2}\!\!\int_0^{2\pi}
R^2\cos\varphi\;\dd\theta\,\dd\varphi
= 2\pi R^2\,\bigl[\sin\varphi\bigr]_{-\pi/2}^{\pi/2}
= \boxed{4\pi R^2} .
\]
\end{example}

\begin{remark}[Controles de sanidad para áreas]
Tres comprobaciones instantáneas detectan la mayoría de los errores en un \omterm{def:b2:surfaces:area}{área}
cálculo. \emph{Escalada}: dilatando una superficie por $\lambda$
multiplica $E, F, G$ por $\lambda^2$ y \omterm{def:b2:surfaces:area}{área} por
$\lambda^2$ --- una respuesta cuya dependencia de $R$ no es
cuadrática (como $4\pi R^2$) es incorrecta. \emph{Positividad de la
elemento}: $\sqrt{EG - F^2}$ debe ser estrictamente positivo en el
interior del gráfico; un valor que desaparece indica una degeneración del gráfico,
para ser extirpado como en los polos de la esfera. \emph{Simetría}: un
El cálculo sobre una pieza \omterm{def:b2:quadratic:adjoint}{simétrico} debe ser consistente con
sumando sus partes congruentes --- sería mejor que el hemisferio
proporcione $2\pi R^2$.
\end{remark}

\begin{example}[Superficie de revolución]\label{ex:b2:surfaces:revolution}
Girar la curva $z \mapsto (r(z), 0, z)$, $r > 0$ de clase
$\mathcal{C}^1$, alrededor del eje $z$:
\[
\sigma(\theta, z) = (r(z)\cos\theta,\ r(z)\sin\theta,\ z).
\]
Entonces $E = r(z)^2$, $F = 0$, $G = 1 + r'(z)^2$, entonces
\[
\mathcal{A} = \int_a^b\!\!\int_0^{2\pi}
r(z)\sqrt{1 + r'(z)^2}\;\dd\theta\,\dd z
= 2\pi\int_a^b r(z)\sqrt{1 + r'(z)^2}\,\dd z ,
\]
la fórmula clásica (circunferencia $2\pi r$ veces la inclinación
\omterm{def:b2:curves:length}{longitud} elemento). Para el cono $r(z) = kz$, $z \in [0, h]$: $\mathcal
A = 2\pi k\sqrt{1 + k^2}\,\frac{h^2}{2} = \pi \rho \ell$ con
$\rho = kh$ el radio base y $\ell = h\sqrt{1 + k^2}$ la inclinación
altura: la fórmula escolar, ahora derivada en lugar de admitida.
\end{example}

\begin{example}[El catenoide]\label{ex:b2:surfaces:catenoid}
Girar la catenaria $r(z) = \cosh z$, $z \in \intcc{-1}{1}$,
alrededor de su eje: el \emph{catenoide} resultante tiene, por
fórmula de revolución y $1 + \sinh^2 z = \cosh^2 z$,
\[
\mathcal A = 2\pi\int_{-1}^1\cosh z\,\sqrt{1 + \sinh^2z}\,
\dd z = 2\pi\int_{-1}^1\cosh^2z\,\dd z
= \pi\bigl[z + \sinh z\cosh z\bigr]_{-1}^{1},
\]
eso es
\[
\mathcal A = 2\pi + \pi\sinh 2 \approx 17.68 .
\]
El factor de inclinación $\sqrt{1 + r'^2}$ se fusionó con el radio
en un cuadrado perfecto --- la misma identidad que hizo el
elemental \omterm{def:b2:curves:length}{longitud del arco} de catenaria en el capítulo de curvas. esto
No es casualidad del álgebra: entre todas las superficies de revolución.
Al abarcar los dos círculos límite, la catenoide minimiza
\omterm{def:b2:surfaces:area}{área} (es la forma de una película de jabón entre dos anillos), y
Esta propiedad variacional es precisamente lo que distingue
$\cosh$; el volumen de Year 3 lo demuestra con el cálculo de
variaciones.
\end{example}

\begin{remark}[Errores comunes]
(yo) \emph{Las singularidades del gráfico no son singularidades de superficie.}:
La carta esférica degenera en los polos.
($\cos\varphi = 0$), pero la esfera allí es perfectamente lisa.

0)$no es regular a lo largo de $u = 0$ aunque su imagen es una
avión. (iii) La fórmula \omterm{def:b2:surfaces:area}{área} requiere $\sigma$

hasta firmar por la superficie pero es \emph{preferido} por el gráfico
(orden de $u, v$); Las declaraciones que involucran orientación deben corregirse.
esa elección. (v) Finalmente, $EG - F^2 > 0$ no es un extra
Hipótesis: es exactamente regularidad, según Lagrange.
identidad --- si desaparece en alguna parte, el problema es la
gráfico, y allí no se aplica ninguna fórmula \omterm{def:b2:surfaces:area}{área} o del plano tangente.
\end{remark}

\section{Problema: la clasificación de las cuádricas de
$\R^3$}

\begin{problem}[{Problema de fin de semana --- cada superficie cuádrica,
ordenados por el teorema espectral}]\label{pb:b2:surfaces:1}
A \emph{cuádrico}\index{cuádrico} es el cero establecido en $\R^3$ de un
polinomio de grado dos
\[
q(X) = X^{\mathsf T}\!AX + 2\,\langle b, X\rangle + c,
\qquad A \in \mathcal S_3(\R),\ A \neq 0,\ b \in \R^3,\
c \in \R .
\]
Las superficies de las figuras de este capítulo --- esferas,
elipsoides, sillas de montar, conos, cilindros --- son todos \omterm{pb:b2:surfaces:1}{cuádricas}.
Este problema los clasifica \emph{todo}: el teorema espectral
(\cref{thm:b2:quadratic:spectral}) endereza la cuadrática
parte, las traducciones \omterm{def:b2:affine:subspace}{afín} (\cref{ch:b2:affine}) absorben
parte lineal, y lo que queda es una lista breve y completa de
formas normales.

\medskip
\noindent\textbf{Parte I --- The reduction machine.}
\begin{enumerate}
 \item Deje $X = PY + t$ con $P \in O(3)$ y $t \in \R^3$ (un
 cambio rígido de coordenadas). Demuestre que $q(PY + t) =
 Y^{\mathsf T}\!A'Y + 2\langle b', Y\rangle + c'$ con
 \[
 A' = P^{\mathsf T}\!AP, \qquad
 b' = P^{\mathsf T}(At + b), \qquad
 c' = q(t) .
 \]
 Deducir que el \omterm{def:b2:reduction:eigen}{espectro} de $A$ (de ahí su rango y
 firma) es un invariante rígido de la ecuación, y
 explique por qué la ecuación de un \omterm{pb:b2:surfaces:1}{cuádrico} dado es solo
 determinado hasta un factor escalar distinto de cero.
 \item Utilizando el teorema espectral, demuestre que después de una rotación
 la ecuación pasa a ser $\sum_i \lambda_i y_i^2 +
 2\sum_i\beta_iy_i + c = 0$ con $\lambda_1, \lambda_2,
 \lambda_3$ el \omterm{def:b2:reduction:eigen}{valores propios} de $A$.
 \item Por cada $i$ con $\lambda_i \neq 0$, absorber
 $\beta_iy_i$ mediante una traducción ($y_i \mapsto y_i -
 \beta_i/\lambda_i$). Escribe la ecuación reducida cuando
 $\operatorname{rank} A = r$: $\sum_{i\leq r}\lambda_i
 z_i^2 + 2\sum_{i > r}\beta_iz_i + c'' = 0$.
 \item Un \emph{centro} del \omterm{pb:b2:surfaces:1}{cuádrico} de la ecuación $q = 0$ es
 un punto $\Omega$ con $q(2\Omega - X) = q(X)$ para todos
 $X$: la reflexión del punto en $\Omega$ preserva la
 ecuación, de ahí la superficie. Demuestre que $q(2\Omega - X)
 - q(X) = -4\langle A\Omega + b, X\rangle + 4\langle
 A\Omega + b, \Omega\rangle$ y deduzca: los centros
 son exactamente las soluciones de $A\Omega = -b$; ellos
 existe si $b \in \operatorname{im}A$, y el centro
 es único si $A$ es invertible.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Central \omterm{pb:b2:surfaces:1}{cuádricas}
($\operatorname{rank}A = 3$).} Aquí la ecuación reducida es
$\lambda_1z_1^2 + \lambda_2z_2^2 + \lambda_3z_3^2 = \delta$.
\begin{enumerate}[resume]
 \item Multiplicando por $-1$ si es necesario, asume al menos dos
 $\lambda_i > 0$. Enumere las posibilidades:
 firma $(3, 0)$ con $\delta > 0$, $= 0$, $< 0$,
 y firma $(2, 1)$ con $\delta > 0$, $= 0$, $<
 0$; nombrar los seis conjuntos resultantes (elipsoide, punto,
 conjunto vacío, hiperboloide de una hoja, cono, dos hojas
 hiperboloide) y poner cada uno en su normal euclidiana
 formulario ($\frac{x^2}{a^2} + \frac{y^2}{b^2} +
 \frac{z^2}{c^2} = 1$, etc.).
 \item Clasificar $x^2 + y^2 + z^2 + 4xy + 4yz + 4zx = 1$:
 muestre $A = 2J - I$ con $J$ la matriz de todos unos,
 calcular el \omterm{def:b2:reduction:eigen}{espectro} $\{5, -1, -1\}$ e identificar un
