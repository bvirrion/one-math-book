\section{El diferencial}

\begin{definition}\label{def:b2:diffcalc:differential}
$f$ es \emph{diferenciable}\index{diferenciable} en $a$ cuando hay
es un mapa lineal (\omterm{def:b2:metric:continuity}{continuo}) $\dd f_a \colon \R^n \to \R^m$ con
\[
f(a + h) = f(a) + \dd f_a(h) + o(\norm h)
\qquad (h \to 0).
\]
El mapa $\dd f_a$, el \emph{diferencial}\index{diferencial} de
$f$ en $a$, es único; su matriz en las bases canónicas es
el \emph{matriz jacobiana}\index{matriz jacobiana} $J_f(a) =
\bigl(\frac{\partial f_i}{\partial x_j}(a)\bigr)$.
La diferenciabilidad implica \omterm{def:b2:metric:continuity}{continuidad} y la existencia de todos
derivados direccionales $\dd f_a(v) = \lim_{t\to0}\frac{f(a + tv)
- f(a)}{t}$; lo contrario falla
(\cref{exo:b2:diffcalc:2}). Para $m = 1$, $\dd f_a(h) = \langle
\nabla f(a), h\rangle$: el gradiente del Año 1, ahora entendido como
el vector que representa el diferencial.
\end{definition}

\begin{theorem}[criterio $C^1$]\label{thm:b2:diffcalc:c1}
Si todas las derivadas parciales de $f$ existen en $U$ y son \omterm{def:b2:metric:continuity}{continuo}
en $a$, entonces $f$ es \omterm{def:b2:diffcalc:differential}{diferenciable} en $a$. ``$C^1$ en $U$'' ---
\omterm{def:b2:metric:continuity}{continuo} parciales --- por lo tanto implica \omterm{def:b2:diffcalc:differential}{diferenciabilidad}
en todas partes, con \omterm{def:b2:metric:continuity}{continuo} \omterm{def:b2:diffcalc:differential}{diferencial}.
\end{theorem}

\begin{proof}
Componente por componente (basta con $m = 1$). La prueba del Año 1 para dos
variables --- mover una coordenada a la vez, aplicar la variable de una
teorema del valor medio en cada cateto y use \omterm{def:b2:metric:continuity}{continuidad} de los parciales
en $a$ --- generaliza palabra por palabra a las piernas $n$:
\[
f(a + h) - f(a) = \sum_{j=1}^{n} \bigl(f(a + h^{(j)}) - f(a +
h^{(j-1)})\bigr)
= \sum_j h_j\,\frac{\partial f}{\partial x_j}(\xi_j),
\]
donde $h^{(j)}$ congela las primeras coordenadas $j$ de $h$ y
$\xi_j$ se encuentra en el tramo $j$; \omterm{def:b2:metric:continuity}{continuidad} vueltas cada una
$\frac{\partial f}{\partial x_j}(\xi_j)$ en $\frac{\partial
f}{\partial x_j}(a) + o(1)$y el error es $o(\norm h)$.
\end{proof}

\begin{theorem}[regla de cadena]\label{thm:b2:diffcalc:chainrule}
Si $f$ es \omterm{def:b2:diffcalc:differential}{diferenciable} en $a$ y $g$ en $f(a)$, entonces $g \circ f$
es \omterm{def:b2:diffcalc:differential}{diferenciable} en $a$ con
\[
\dd(g \circ f)_a = \dd g_{f(a)} \circ \dd f_a ,
\qquad
J_{g\circ f}(a) = J_g\bigl(f(a)\bigr)\,J_f(a) :
\]
Los jacobianos se multiplican.\index{regla de la cadena}
\end{theorem}

\begin{proof}
Escribe $f(a + h) = f(a) + \dd f_a(h) + \norm h\,\varepsilon_1(h)$
y $g(b + k) = g(b) + \dd g_b(k) + \norm k\,\varepsilon_2(k)$ con
$b = f(a)$, $k = k(h) = \dd f_a(h) + \norm h \varepsilon_1(h)$.
Sustituyendo,
\[
g(f(a+h)) = g(b) + \dd g_b\bigl(\dd f_a(h)\bigr)
+ \norm h\,\dd g_b(\varepsilon_1(h)) + \norm{k}\,\varepsilon_2(k),
\]
y ambos términos de error son $o(\norm h)$: el primero desde $\dd g_b$ es
\omterm{def:b2:metric:continuity}{continuo} y $\varepsilon_1 \to 0$; el segundo desde $\norm k
\leq C\norm h$ (mapa lineal acotado más término pequeño) y
$\varepsilon_2(k) \to 0$ como $h \to 0$.
\end{proof}

\begin{theorem}[Desigualdad del valor medio]\label{thm:b2:diffcalc:mvi}
Sea $f$ \omterm{def:b2:diffcalc:differential}{diferenciable} en $U$ y deje que el segmento $\intcc{a}{b}
= \{a + t(b-a)\}$ esté en $U$. entonces
\[
\norm{f(b) - f(a)} \leq \norm{b - a}\,
\sup_{x \in \intcc{a}{b}} \vertiii{\dd f_x} .
\]
En particular, un mapa \omterm{def:b2:diffcalc:differential}{diferenciable} con cero \omterm{def:b2:diffcalc:differential}{diferencial} en un
\emph{\omterm{def:b2:metric:connected}{conectado}} \omterm{def:b2:metric:topology}{abierto} establecido es constante.
\end{theorem}

\begin{proof}
La función $\varphi(t) = f(a + t(b-a))$ es \omterm{def:b2:diffcalc:differential}{diferenciable} en
$\intcc{0}{1}$ con $\varphi'(t) = \dd f_{a + t(b-a)}(b - a)$
(regla de la cadena), de \omterm{def:b2:nvs:norm}{norma} $\leq M\norm{b-a}$ con $M$ el mostrado
sup. Para $f$ con valor $\R$, la desigualdad del valor medio de una variable
concluye; para valores vectoriales, aplíquelo a $t \mapsto \langle u,
\varphi(t)\rangle $con $u$el vector unitario a lo largo de $f(b) - f(a)$.
Constancia: localmente constante (segmentos en bolas) más \omterm{def:b2:metric:connected}{conectividad}
(el conjunto donde $f$ es igual a un valor dado es \omterm{def:b2:metric:topology}{abierto} y cerrado:
\cref{ch:b2:metric}).
\end{proof}

\section{Segundas derivadas}

\begin{theorem}[negro]\label{thm:b2:diffcalc:schwarz}
Si $f$ es $C^2$ en $U$ (todos los segundos parciales existen y son
\omterm{def:b2:metric:continuity}{continuo}), luego para todos $i, j$:\index{teorema de schwarz}
\[
\frac{\partial^2 f}{\partial x_i\,\partial x_j}
= \frac{\partial^2 f}{\partial x_j\,\partial x_i} .
\]
\end{theorem}

\begin{proof}
Bastan dos variables ($x = x_i$, $y = x_j$, otras congeladas).
Considere la segunda diferencia
\[
\Delta(h) = f(a + h, b + h) - f(a + h, b) - f(a, b + h) + f(a,b) .
\]
Repare $h$ y configure $\varphi(x) = f(x, b+h) - f(x, b)$: luego $\Delta(h)
= \varphi(a + h) - \varphi(a)$ y dos aplicaciones de la media.
teorema del valor dar
\[
\Delta(h) = h\,\varphi'(\xi)
= h\Bigl(\frac{\partial f}{\partial x}(\xi, b+h) -
\frac{\partial f}{\partial x}(\xi, b)\Bigr)
= h^2\,\frac{\partial^2 f}{\partial y\,\partial x}(\xi, \eta),
\]
con $\xi \in \intoo{a}{a+h}$, $\eta \in \intoo{b}{b+h}$. Por
\omterm{def:b2:metric:continuity}{continuidad}, $\frac{\Delta(h)}{h^2} \to \frac{\partial^2 f}{\partial
y\partial x}(a, b)$como $h \to 0$. El mismo cálculo con el
roles de las variables intercambiadas (congelar primero la segunda variable)
da $\frac{\Delta(h)}{h^2} \to \frac{\partial^2 f}{\partial x
\partial y}(a,b)$: los dos límites de la misma cantidad coinciden.
\end{proof}

\begin{example}[Por qué se necesita $C^2$: contraejemplo de Peano]\label{ex:b2:diffcalc:peano}
Sea $f(x, y) = \dfrac{xy(x^2 - y^2)}{x^2 + y^2}$, $f(0,0) = 0$.
Lejos del origen $f$ está $C^\infty$; en el origen todo primero
y segundos parciales existen, pero los mixtos no están de acuerdo. calcular
a lo largo de los ejes: $f(x, 0) = f(0, y) = 0$, y para $y \neq 0$,
\[
\frac{\partial f}{\partial x}(0, y)
= \lim_{x\to0}\frac{f(x,y)}{x}
= \frac{y(0 - y^2)}{y^2} = -y ,
\qquad\text{symmetrically}\qquad
\frac{\partial f}{\partial y}(x, 0) = x .
\]
Por lo tanto
\[
\frac{\partial^2 f}{\partial y\,\partial x}(0,0)
= \frac{\dd}{\dd y}\Bigl[\frac{\partial f}{\partial
x}(0,y)\Bigr]_{y=0} = -1,
\qquad
\frac{\partial^2 f}{\partial x\,\partial y}(0,0) = +1 :
\]
los dos parciales mixtos existen y difieren. Ninguna contradicción con
\cref{thm:b2:diffcalc:schwarz}: los segundos parciales de $f$ son
no \omterm{def:b2:metric:continuity}{continuo} en $0$ (prueba junto con $y = tx$). Información final:
El teorema de Schwarz es un teorema genuino sobre \emph{\omterm{def:b2:metric:continuity}{continuidad}},
no es una identidad formal --- y la hipótesis ``$C^2$'' en
Taylor--Young a continuación está haciendo un trabajo real.
\end{example}

\begin{theorem}[Taylor--Joven en la orden 2]\label{thm:b2:diffcalc:taylor}
Sea $f \colon U \to \R$ $C^2$ y $a \in U$. Entonces, como $h \to 0$,
\[
f(a + h) = f(a) + \langle\nabla f(a), h\rangle
+ \frac12\, \langle H_a h,\, h\rangle + o\bigl(\norm h^2\bigr),
\]
donde $H_a = \bigl(\frac{\partial^2 f}{\partial x_i\partial
x_j}(a)\bigr)$ es el (\omterm{def:b2:quadratic:adjoint}{simétrico}, de Schwarz) \emph{arpillera
matriz}\index{matriz de arpillera}.
\end{theorem}

\begin{proof}
Aplique el teorema de Taylor-Young de una variable (volumen del año 1) a
$\varphi(t) = f(a + th)$ en $\intcc{0}{1}$: según la regla de la cadena,
$\varphi'(t) = \langle \nabla f(a + th), h\rangle$ y $\varphi''(t)
= \langle H_{a+th}h, h\rangle$, ambos \omterm{def:b2:metric:continuity}{continuo} en $t$. entonces
$\varphi(1) = \varphi(0) + \varphi'(0) + \frac12\varphi''(\theta)$
(Taylor--Lagrange) con $\theta \in \intoo{0}{1}$ y \omterm{def:b2:metric:continuity}{continuidad}
del segundo parcial convierte $\varphi''(\theta) =
\varphi''(0) + o(1)\cdot\norm h^2$ de manera uniforme: el mostrado
expansión.
\end{proof}

\begin{example}[Una expansión de dos vías]\label{ex:b2:diffcalc:taylortwoways}
Expanda $f(x, y) = \eu^x\cos y$ en el origen para ordenar $2$.

reproduce $1 + x + \frac12(x^2 - y^2)$. Los dos cálculos
de acuerdo, y la ruta de composición fue más rápida --- no hay segundo
parciales en absoluto. Información final: el origen es \emph{no} a
punto crítico ($\nabla f \neq 0$), por lo que a pesar del tiempo indefinido

y la prueba de segundo orden sólo habla en momentos críticos.
puntos.
\end{example}

\begin{theorem}[Prueba extrema de segundo orden, probada]\label{thm:b2:diffcalc:extrema}
Sea $f$ $C^2$ cerca de un punto crítico $a$ ($\nabla f(a) = 0$),
con arpillera $H = H_a$.
\begin{enumerate}
 \item Si $H$ es positivo definido, $a$ es un mínimo local estricto
 (definida negativa: máximo).
 \item Si $H$ tiene \omterm{def:b2:reduction:eigen}{valores propios} de ambos signos, $a$ es una silla de montar: no
 extremo.
 \item Si $H$ es singular (y semidefinido), no hay conclusión.
\end{enumerate}
La prueba del Año 1 ``$rt - s^2$'' es el caso $n = 2$: $\det H = rt -
s^2$, $\operatorname{tr}$ lectura de signos $r$.
\end{theorem}

\begin{proof}
Según el teorema espectral (\cref{thm:b2:quadratic:spectral}), $H$
\omterm{def:b2:quadratic:def}{forma cuadrática} está apretado entre su extremo \omterm{def:b2:reduction:eigen}{valores propios}:
$\lambda_{\min}\norm h^2 \leq \langle Hh, h\rangle \leq
\lambda_{\max}\norm h^2$.

(1) Si $\lambda_{\min} > 0$: Taylor--Young da
\[
f(a + h) - f(a) \geq \frac{\lambda_{\min}}{2}\norm h^2 -
o(\norm h^2) > 0
\]
para pequeños $h \neq 0$: mínimo local estricto.

(2) A lo largo de un \omterm{def:b2:reduction:eigen}{vector propio} $v_+$ con $\lambda_+ > 0$: $f(a + tv_+) -
f(a) = \frac{\lambda_+}{2}t^2 + o(t^2) > 0$ para $t$ pequeño; a lo largo
$v_-$ con $\lambda_- < 0$ la diferencia es negativa: ambos signos
ocurren en cada barrio.

(3) $f(x,y) = x^2 + y^4$, $x^2 - y^4$, $x^2 + y^3$ comparten lo mismo
Arpillera semidefinida singular en $0$ con tres diferentes
comportamientos.
\end{proof}

\begin{example}[Una clasificación completa, global incluida]\label{ex:b2:diffcalc:quarticrun}
Clasifique todos los extremos de $f(x, y) = x^4 + y^4 - 4xy$ en $\R^2$.
\emph{Puntos críticos:} $\nabla f = (4x^3 - 4y,\ 4y^3 - 4x) =
0$ proporciona $y = x^3$y $x = y^3 = x^9$, por lo que $x(x^8 - 1) = 0$:
las soluciones reales son $(0,0)$, $(1,1)$, $(-1,-1)$.
\emph{Arpilleras:}$H = \begin{pmatrix} 12x^2 & -4\\ -4 &
12y^2\end{pmatrix}$. En $(\pm1, \pm1)$: $\begin{pmatrix} 12 &
-4\\ -4 & 12\end{pmatrix}$, \omterm{def:b2:reduction:eigen}{valores propios}$8$y $16$: positivo
mínimos locales definidos y estrictos con $f = -2$. En $(0,0)$:
$\begin{pmatrix} 0 & -4\\ -4 & 0\end{pmatrix}$, \omterm{def:b2:reduction:eigen}{valores propios}
$\pm4$: una silla de montar. \emph{Globalidad:} de $2\abs{xy} \leq x^2 +
y^2$,
\[
f(x, y) \geq x^4 + y^4 - 2(x^2 + y^2)
= (x^2 - 1)^2 + (y^2 - 1)^2 + x^2 + y^2 - 2
\xrightarrow[\norm{(x,y)}\to\infty]{} +\infty :
\]
$f$ es coercitivo, por lo que alcanza un mínimo global
(\omterm{def:b2:metric:compact}{compacidad} de conjuntos de subniveles), necesariamente en un punto crítico
punto: el valor $-2$, tanto en $(1,1)$ como en $(-1,-1)$, es el
mínimo global; no hay un máximo ($f$ ilimitado arriba).
Perspectiva final: la prueba local clasifica a los candidatos, pero
Sólo un argumento de crecimiento convierte lo "local" en "global".
patrón de dos pasos de cada prueba de optimización en este libro.
\end{example}

\begin{method}[Clasificando los extremos de $f \colon \R^n \to \R$]\label{met:b2:diffcalc:extrema}
\begin{enumerate}
 \item Resolver $\nabla f = 0$ (todos los puntos críticos; en un dominio
 con límite, trate el límite por separado como en
 \cref{exo:b2:diffcalc:7}).
 \item En cada punto crítico, calcule el hessiano y su
 \omterm{def:b2:reduction:eigen}{valor propio} signos --- en dimensión $2$, solo $\det H$
 y $\operatorname{tr} H$: silla $\det < 0$; $\det >
 0$ extremo, del tipo dado por el signo de la
 rastro; $\det = 0$: la prueba es silenciosa, estudia $f$
 a lo largo de curvas.
 \item Para declaraciones globales, agregue \omterm{def:b2:metric:compact}{compacidad} o coercitividad
 argumento ($f \to +\infty$ en el infinito, o un \omterm{def:b2:metric:compact}{compacto}
 conjunto de restricciones), luego compare los valores críticos.
\end{enumerate}
\end{method}

\section{El teorema de la función inversa}

\begin{theorem}[Teorema de la función inversa]\label{thm:b2:diffcalc:inverse}
Sea $f \colon U \to \R^n$ $C^1$ y $a \in U$ con $\dd f_a$
\emph{reversible}. Luego hay barrios \omterm{def:b2:metric:topology}{abierto} $V \ni a$, $W
\ni f(a)$tales que $f \colon V \to W$es una biyección con $C^1$
inversa, y\index{teorema de la función inversa}
\[
\dd (f^{-1})_{f(x)} = (\dd f_x)^{-1} \qquad (x \in V).
\]
\end{theorem}
\admitted

\begin{remark}[Por qué es cierto: la estrategia del punto fijo]
Resolver $f(x) = y$ cerca de $a$ se reescribe como la ecuación de punto fijo
$x = x + \dd f_a^{-1}\bigl(y - f(x)\bigr) =: \Phi_y(x)$; el mapa
$\Phi_y$ tiene \omterm{def:b2:diffcalc:differential}{diferencial} $\mathrm{id} - \dd f_a^{-1}\dd f_x$,
pequeño cerca de $a$ por \omterm{def:b2:metric:continuity}{continuidad} de $\dd f$, por lo que $\Phi_y$ es un
contracción sobre una pequeña bola cerrada y el punto fijo de Banach
teorema (\cref{thm:b2:metric:banach}) proporciona el único local
solución $x = f^{-1}(y)$. \omterm{def:b2:metric:continuity}{Continuidad} y \omterm{def:b2:diffcalc:differential}{diferenciabilidad} de la
La inversa se sigue entonces de las estimaciones en la contracción. el completo
la contabilidad se lleva a cabo en el año 3; la estrategia --- y la
declaración --- se utilizan libremente a partir de ahora. el compañero
\emph{teorema de la función implícita} (resolviendo $F(x, y) = 0$ para $y(x)$
cuando $\frac{\partial F}{\partial y}$ es invertible) sigue por
aplicando el teorema a $(x, y) \mapsto (x, F(x,y))$.
\end{remark}

\begin{example}[Coordenadas polares]\label{ex:b2:diffcalc:polar}
$\Phi(r, \theta) = (r\cos\theta, r\sin\theta)$ tiene jacobiano
\[
J_\Phi = \begin{pmatrix} \cos\theta & -r\sin\theta\\ \sin\theta &
r\cos\theta \end{pmatrix},
\qquad \det J_\Phi = r :
\]
invertible para $r \neq 0$, por lo que $\Phi$ es un $C^1$ local
difeomorfismo lejos del origen --- la licencia para "cambiar a
coordenadas polares'', renovadas para las integrales múltiples de
\cref{ch:b2:multint}.
\end{example}

\begin{example}[Local en todas partes, global en ninguna]\label{ex:b2:diffcalc:localnotglobal}
Deja que $f(x, y) = \bigl(\eu^x\cos y,\ \eu^x\sin y\bigr)$ encienda
$\R^2$. Es jacobiano,
\[
J_f = \begin{pmatrix}
\eu^x\cos y & -\eu^x\sin y\\
\eu^x\sin y & \eu^x\cos y
\end{pmatrix},
\qquad
\det J_f = \eu^{2x} > 0 ,
\]
nunca desaparece: por \cref{thm:b2:diffcalc:inverse}, $f$ es un
Difeomorfismo local $C^1$ en el punto \emph{cada} del plano.
Sin embargo, $f$ está lejos de ser inyectivo: $f(x, y + 2\pi) = f(x, y)$, por lo que
cada valor se toma infinitamente a menudo; y no es sobreyectivo
tampoco, ya que $\norm{f(x, y)} = \eu^x > 0$ pierde el origen.
Idea final: el teorema de la función inversa es irreducible
\emph{local} --- la invertibilidad de cada $\dd f_a$ produce un
mosaico de inversos locales que no necesitan unirse en uno solo.
(Los lectores que conocen números complejos reconocerán $z \mapsto
\eu^z$; el mosaico es la familia de ramas de logaritmos).
Compárese \cref{exo:b2:diffcalc:12}, donde se muestra un análisis global cuantitativo.
La hipótesis fuerza una inversa global.
\end{example}

\begin{remark}[Errores comunes]
\emph{(i) Directional derivatives are cheap, \omterm{def:b2:diffcalc:differential}{diferenciales} are
not:} todas las derivadas direccionales pueden existir --- e incluso fallar
depender linealmente de la dirección --- sin
\omterm{def:b2:diffcalc:differential}{diferenciabilidad} (\cref{exo:b2:diffcalc:2}); solo el $C^1$
criterio
(\cref{thm:b2:diffcalc:c1}) actualiza parciales a un
\omterm{def:b2:diffcalc:differential}{diferencial}. \emph{(ii) Crítico no significa extremo:}
sillines (\cref{ex:b2:diffcalc:quarticrun}) y el silencioso
caso singular (\cref{thm:b2:diffcalc:extrema}~(3)) ambos ocultan
detrás de $\nabla f = 0$. \emph{(iii) Sin valor medio valorado por vector
igualdad:} solo el \emph{desigualdad} de
\cref{thm:b2:diffcalc:mvi} sobrevive
(\cref{exo:b2:diffcalc:9}); nunca escriba $f(b) - f(a) = \dd
f_c(b-a)$para $f$con valores en $\R^m$, $m \geq 2$.
\emph{(iv) La invertibilidad local no es inyectividad:}
\cref{ex:b2:diffcalc:localnotglobal}. \emph{(v) El gradiente
pertenece al producto interno:} $\nabla f$ es el vector
representando $\dd f_a$ en un producto interior elegido; cambiar el
producto (como en el ejemplo ponderado de las formas cuadráticas
capítulo) y el gradiente gira, mientras que el \omterm{def:b2:diffcalc:differential}{diferencial} ---
el objeto intrínseco --- no se mueve.
\end{remark}

\begin{remark}[donde se utiliza]
Todo lo que sigue en este capítulo es cálculo diferencial.
aplicado: el capítulo de ecuaciones diferenciales linealiza los flujos y
utiliza la fórmula \omterm{def:b2:linalg:det}{determinante} de Liouville (probada en el capítulo
problema de fin de semana); los capítulos sobre curvas y superficies estudian
conjuntos de niveles y parametrizaciones a través de la función implícita
teorema; integrales múltiples cambian variables a través de jacobianos.
El problema del fin de semana desarrolla el cálculo \emph{en el espacio de
matrices en sí} --- \omterm{def:b2:diffcalc:differential}{diferencial} del \omterm{def:b2:linalg:det}{determinante}, del
inversa, la \omterm{ex:b2:nvs:matrixexp}{matriz exponencial}, y el grupo ortogonal como
conjunto de niveles suaves --- la sombra del Año 2 de lo que fue el Año 3
El volumen se formaliza como variedades y grupos de Lie.
\end{remark}

\begin{exercise}[$\star$]\label{exo:b2:diffcalc:1}
Calcule las matrices jacobianas de $f(x,y) = (x^2 - y^2,\, 2xy)$ y
de $\Phi(r,\theta,z) = (r\cos\theta, r\sin\theta, z)$; donde estan
¿El \omterm{def:b2:diffcalc:differential}{diferenciales} invertible?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:diffcalc:2}
Deje $f(x,y) = \frac{x^3}{x^2 + y^2}$ ($f(0,0) = 0$). demostrar que todo
Existen derivadas direccionales de $f$ en $0$, pero $f$ no es
\omterm{def:b2:diffcalc:differential}{diferenciable} en $0$ \emph{(the map $v \mapsto$ directional
derivative is not linear)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:diffcalc:3}
Encuentre y clasifique los puntos críticos de $f(x, y) = x^3 + y^3 -
3xy $usando \cref{thm:b2:diffcalc:extrema} y de $g(x,y) = x^4 +
y^4 - 2(x - y)^2$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:diffcalc:4}
Sea $f \colon \R^n \to \R$ $C^1$ y \emph{homogéneo de grado
$p$}: $f(tx) = t^pf(x)$ para $t > 0$. Demostrar la identidad de Euler
\[
\langle \nabla f(x), x\rangle = p\,f(x) ,
\]
y su recíproco para funciones $C^1$ en $\R^n\setminus\{0\}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:diffcalc:5}
Sea $A$ \omterm{def:b2:quadratic:adjoint}{simétrico} y $f(x) = \frac12\langle Ax, x\rangle -
\langle b, x\rangle$. Calcule $\nabla f $y$H_f$; cuando es $f$
convexo? Suponiendo $A$ positivo definido, demuestre que $f$ tiene un único
mínimo global en la solución de $Ax = b$ --- descenso de gradiente
razón de être.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:diffcalc:6}
(Multiplicador de Lagrange, una restricción, probado manualmente) Sea $f, g$
$C^1$ en $\R^2$, y supongamos que $f$ alcanza, en $a$, un extremo local
en el nivel establecido $\{g = 0\}$, con $\nabla g(a) \neq 0$. demostrar que
$\nabla f(a) = \lambda\nabla g(a)$ para algunos $\lambda$.
\emph{(Parametrize the level set near $a$ by the implicit function
theorem and differentiate $t \mapsto f(\gamma(t))$.)} Aplicación:
extremos de $f(x,y) = xy$ en el círculo $x^2 + y^2 = 1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:diffcalc:7}
Determine los extremos de $f(x, y) = x^2 + y^2 - xy + x - y$ en
$\R^2$, luego su máximo y mínimo en el triángulo cerrado con
vértices $(0,0)$, $(1,0)$, $(0,1)$ \emph{(puntos críticos interiores,
luego las tres aristas, luego los vértices)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:diffcalc:8}
Sea $f \colon \R^2 \to \R^2$, $f(x, y) = (x + y^2,\; y + x^2)$.
Demuestre que $f$ es un difeomorfismo local cerca de $0$, calcule
$\dd(f^{-1})_{(0,0)}$ y encuentre el $r$ más grande tal que $\dd f$
es invertible en la pelota $\norm{(x,y)}_2 < r$ \emph{(compute
$\det J_f$)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:diffcalc:9}
(Rolle falla, el valor medio sobrevive) Dé $f \colon \R \to \R^2$,
$C^1$, con $f(0) = f(2\pi)$ pero $f'(t) \neq 0$ para todos $t$
(sin Rolle con valor vectorial). Luego verifique en su ejemplo el
valor medio \emph{desigualdad} de \cref{thm:b2:diffcalc:mvi}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b2:diffcalc:10}
Calcule el \omterm{def:b2:diffcalc:differential}{diferencial} y el gradiente de $f(x) = \norm
x_2^2$y de $g(x) = \langle Ax, x\rangle $en $\R^n $($A$ a
matriz cuadrada, no asumida \omterm{def:b2:quadratic:adjoint}{simétrico}), y el hessiano de cada uno.
¿Para qué $A$ es $g$ convexo?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b2:diffcalc:11}
Sea $f \colon \R^n \to \R$ $C^2$. Demuestre que $f$ es convexo si
y sólo si su hessiano $H_x$ es semidefinido positivo en cada
$x $\emph{(reduce to one variable: $t \mapsto f(a + t(b-a))$;
use Taylor--Lagrange in one direction, and for the converse
evaluate $\varphi''$)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b2:diffcalc:12}
(Un teorema inverso global) Sea $g \colon \R^n \to \R^n$ $C^1$
con $\vertiii{\dd g_x} \leq k < 1$ para todos los $x$ y $f =
\mathrm{id} + g$.
\begin{enumerate}
 \item Mostrar $\norm{f(x) - f(y)} \geq (1 - k)\norm{x - y}$: $f$
 es inyectivo, con \omterm{def:b2:metric:continuity}{continuo} inverso en su imagen.
 \item Demuestre que para cada $y \in \R^n$ el mapa $x \mapsto y -
 g(x)$es una contracción del espacio \omterm{def:b2:metric:complete}{completo}$\R^n$,
 y concluir por el teorema del punto fijo de Banach
 (\cref{thm:b2:metric:banach}) que $f$ es sobreyectivo.
 \item Concluimos que $f$ es una biyección de $\R^n$ con
 $(1-k)^{-1}$-Lipschitz inversa --- a \emph{global}
 contraparte de \cref{thm:b2:diffcalc:inverse} (que,
 por el contrario, es puramente local).
\end{enumerate}
\end{exercise}

\section{Problema: El cálculo de matrices --- Jacobi,
exponencial y el grupo ortogonal}

\begin{problem}\label{pb:b2:diffcalc:1}
El campo de juego más limpio para el cálculo diferencial es el espacio
$\mathcal M_n(\R) \simeq \R^{n^2}$ en sí: sus mapas más naturales
--- producto, inverso, \omterm{def:b2:linalg:det}{determinante}, exponencial --- tener
\omterm{def:b2:diffcalc:differential}{diferenciales} de sorprendente elegancia. Este problema los calcula.
todos: el \emph{serie neumann}, el \omterm{def:b2:diffcalc:differential}{diferencial} del inverso,
\emph{la fórmula de jacobi}\index{la fórmula de jacobi} para el
\omterm{def:b2:linalg:det}{determinante} con \emph{La fórmula de Liouville.} como dividendo, el
\emph{matriz exponencial}\index{matriz exponencial} con
$\det\eu^A = \eu^{\operatorname{tr}A}$, y finalmente el
grupo ortogonal $O_n$ como un nivel suave establecido con el
matrices antisimétricas como espacio tangente --- diferencial
Geometría en embrión. En todo momento, $\vertiii\cdot$ es el \omterm{thm:b2:nvs:continuouslinear}{operador
norma} subordinado a $\norm\cdot_2$ y $\langle X, Y\rangle =
\operatorname{tr}(X^{\mathsf T}Y)$ el producto interno de Frobenius.

\medskip
\noindent\textbf{Parte I --- The Neumann series.}
\begin{enumerate}
 \item Demuestre la submultiplicatividad, $\vertiii{AB} \leq
 \vertiii A\,\vertiii B$, y deduzca ese polinomio
 Los mapas de $A$ (productos de matriz, \omterm{def:b2:linalg:det}{determinante}, traza) son
 \omterm{def:b2:metric:continuity}{continuo} en $\mathcal M_n(\R)$.
 \item Para $\vertiii X < 1$, muestre que $\sum_{k\geq0}X^k$
 converge \omterm{def:b2:series:def}{absolutamente} en $\mathcal M_n(\R)$
 (\cref{thm:b2:nvs:absoluteconvergence}), que su suma
 es $(I - X)^{-1}$, y que
 \[
 \vertiii{(I - X)^{-1}} \leq \frac{1}{1 - \vertiii X},
 \qquad
 (I - X)^{-1} = I + X + O\bigl(\vertiii X^2\bigr) .
 \]
 \item Deduzca que $GL_n(\R)$ es \emph{\omterm{def:b2:metric:topology}{abierto}}: si $A$ es
 invertible y $\vertiii H <
 \frac{1}{\vertiii{A^{-1}}}$, entonces $A + H$ es
 reversible. Deduzca también que $GL_n(\R)$ es \emph{denso}
 en $\mathcal M_n(\R)$ \emph{(perturb $A$ by
 $\varepsilon I$: $\det(A + \varepsilon I)$ is a nonzero
 polynomial in $\varepsilon$)}.
 \item Demuestre que el mapa de inversión $\Phi(A) = A^{-1}$ es
 \omterm{def:b2:metric:continuity}{continuo} en $GL_n(\R)$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- First \omterm{def:b2:diffcalc:differential}{diferenciales}.}
\begin{enumerate}[resume]
 \item Demuestre que el mapa de cuadratura $A \mapsto A^2$ es
 \omterm{def:b2:diffcalc:differential}{diferenciable} con \omterm{def:b2:diffcalc:differential}{diferencial} $H \mapsto AH + HA$,
 y más generalmente que $A \mapsto A^k$ tiene
 \omterm{def:b2:diffcalc:differential}{diferencial}$H \mapsto \sum_{i=0}^{k-1}
 A^iHA^{k-1-i}$. ¿Por qué uno \emph{no} puede escribir $kA^{k-1}H$?
 en general?
 \item Demuestre que $\Phi(A) = A^{-1}$ es \omterm{def:b2:diffcalc:differential}{diferenciable} en
 $GL_n(\R)$ con
 \[
 \dd\Phi_A(H) = -A^{-1}HA^{-1}
 \]
 \emph{(write $(A + H)^{-1} = (I + A^{-1}H)^{-1}A^{-1}$
 and expand by question 2)}. Compara la fórmula con
 el caso escalar $n = 1$.
 \item Para una curva $C^1$ $t \mapsto A(t) \in GL_n(\R)$, deduzca
 $\bigl(A(t)^{-1}\bigr)' = -A^{-1}A'A^{-1}$ y expandir
 $t \mapsto (I + tB)^{-1}$ al primer pedido en $t = 0$.
 \item Calcule el \omterm{def:b2:diffcalc:differential}{diferencial} de $f(A) =
 \operatorname{tr}(A^k)$ e identifique su gradiente para
 el producto interno de Frobenius:
 \[
 \dd f_A(H) = k\operatorname{tr}\bigl(A^{k-1}H\bigr),
 \qquad
 \nabla f(A) = k\,\bigl(A^{k-1}\bigr)^{\mathsf T} .
 \]
 \item Las mismas preguntas para $f(A) = \operatorname{tr}
 (A^{\mathsf T}A) = \norm A_F^2$: \omterm{def:b2:diffcalc:differential}{diferencial},
 gradiente y el hessiano (constante); concluir que
 $\norm\cdot_F^2$ es estrictamente convexo.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Jacobi's formula.}
\begin{enumerate}[resume]
 \item Demostrar
 \[
 \det(I + H) = 1 + \operatorname{tr}H +
 O\bigl(\vertiii H^2\bigr)
 \]
 \emph{(expand $\det(e_1 + h_1, \dots, e_n + h_n)$ by
 multilinearity in the columns: terms with at least two
 $h$-columns are $O(\vertiii H^2)$)}: $\dd(\det)_I =
 \operatorname{tr}$.
 \item Para $A$ invertible, deduzca
 \[
 \dd(\det)_A(H) = \det(A)\,
 \operatorname{tr}\bigl(A^{-1}H\bigr) .
 \]
 \item Muestre que para \emph{cada} $A$ (invertible o no),
 $\frac{\partial\det}{\partial a_{ij}}(A) = C_{ij}$, el
 $(i,j)$ cofactor \emph{(Expansión de Laplace a lo largo de la fila
 $i$)}, de modo que con el adjunto $\operatorname{adj}A =
 \operatorname{com}(A)^{\mathsf T}$:
 \[
 \dd(\det)_A(H) =
 \operatorname{tr}\bigl(\operatorname{adj}(A)\,H\bigr),
 \qquad
 \nabla(\det)(A) = \operatorname{com}(A) ,
 \]
 recuperando la pregunta 11 cuando $A$ es invertible
 ($\operatorname{adj}A = \det(A)A^{-1}$). esto es
 \emph{la fórmula de jacobi}: $\bigl(\det
 A(t)\bigr)' = \operatorname{tr}\bigl(
 \operatorname{adj}(A(t))\,A'(t)\bigr)$.
 \item (fórmula de Liouville) Sea $A(t)$ una curva $C^1$ de
 matrices que satisfacen la ecuación diferencial lineal
 $A'(t) = M(t)A(t)$. probar
 \[
 \bigl(\det A(t)\bigr)' =
 \operatorname{tr}\bigl(M(t)\bigr)\,\det A(t),
 \qquad\text{hence}\qquad
 \det A(t) = \det A(0)\,
 \exp\Bigl(\int_0^t\operatorname{tr}M\Bigr)
 \]
 \emph{(use $\operatorname{adj}(A)\,A = \det(A)I$ and
 the cyclic invariance of the trace)} --- el wronskiano
 identidad que el capítulo de ecuaciones diferenciales
 utilizar constantemente.
 \item Demuestre que $SL_n(\R) = \{\det = 1\}$ es un nivel suave
 establecer: en cada $A \in SL_n(\R)$ el \omterm{def:b2:diffcalc:differential}{diferencial}
 $\dd(\det)_A$ es un mapa lineal \emph{sobreyectivo} en
 $\R$\emph{(evaluate it at $H = \frac1nA$)}.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The \omterm{ex:b2:nvs:matrixexp}{matriz exponencial}.}
\begin{enumerate}[resume]
 \item Muestra que $\eu^A = \sum_{k\geq0}\frac{A^k}{k!}$
 converge \omterm{def:b2:series:def}{absolutamente} para cada $A$, \omterm{def:b2:funcseq:series}{normalmente} en cada
 bola, con $\vertiii{\eu^A} \leq
 \eu^{\vertiii A}$; y que $\eu^A$ depende
 \omterm{def:b2:metric:continuity}{continuamente} en $A$.
 \item Demuestre que $AB = BA$ implica $\eu^{A+B} =
 \eu^A\eu^B$ \emph{(\omterm{thm:b2:series:fubini}{producto cauchy}, legitimate by
 absolute convergence)}; deducir que $\eu^A$ es siempre
 invertible, con mapas inversos $\eu^{-A}$: $\exp$
 $\mathcal M_n(\R)$ en $GL_n(\R)$.
 \item Demuestre que $t \mapsto \eu^{tA}$ es $C^1$ (de hecho
 $C^\infty$) con
 \[
 \frac{\dd}{\dd t}\,\eu^{tA} = A\,\eu^{tA} =
 \eu^{tA}A
 \]
 \emph{(differentiate the series term by term on
 segments: the derived series converges \omterm{def:b2:funcseq:series}{normalmente})}.
 \item Probar la identidad
 \[
 \det\bigl(\eu^{A}\bigr) = \eu^{\operatorname{tr}A}
 \]
 \emph{(apply Liouville's formula, question 13, to
 $A(t) = \eu^{tA}$)}. Comprobaciones de cordura: $n = 1$; nilpotente
 $A$; y las matrices de traza cero aterrizan en
 $SL_n(\R)$.
 \item Mostrar $\eu^H = I + H + O(\vertiii H^2)$, por lo que $\exp$ es
 \omterm{def:b2:diffcalc:differential}{diferenciable} en $0$ con $\dd(\exp)_0 =
 \mathrm{id}$; concluir con la función inversa
 teorema (\cref{thm:b2:diffcalc:inverse}) que $\exp$
 es un difeomorfismo $C^1$ de una vecindad de $0$
 en una vecindad de $I$: cada matriz cercana a la
 La identidad tiene un logaritmo.
 \item Muestre que $\exp$ asigna matrices \omterm{def:b2:quadratic:adjoint}{simétrico} a \omterm{def:b2:quadratic:adjoint}{simétrico}
 \emph{positivo definido} matrices, biyectivamente
 \emph{(diagonalizar; la inversa es la espectral
 logaritmo)}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The orthogonal group as a level
set.}
\begin{enumerate}[resume]
 \item Deje $F(A) = A^{\mathsf T}A$, de $\mathcal M_n(\R)$ a
 las matrices \omterm{def:b2:quadratic:adjoint}{simétrico} $S_n$. Calcule $\dd F_A(H) =
 A^{\mathsf T}H + H^{\mathsf T}A$ y demuestre que en cada
 $A \in O_n = F^{-1}(I)$ este \omterm{def:b2:diffcalc:differential}{diferencial} es
 \emph{sobreyectivo} en $S_n$ \emph{(given $S \in S_n$,
 try $H = \frac12 AS$)}: $O_n$ es un conjunto de niveles suaves,
 de dimensión $n^2 - \frac{n(n+1)}2 = \frac{n(n-1)}2$.
 \item Demuestre que cada curva $C^1$ $A(t) \in O_n$ con $A(0) =
 I$tiene una velocidad \emph{antisimétrico}$A'(0)$, y
 por el contrario, para $K$ antisimétrico la curva
 $\eu^{tK}$ se queda en $O_n$: el espacio tangente de $O_n$
 en $I$ son exactamente las matrices antisimétricas.
 \item Mostrar $\det\eu^{K} = 1$ para $K$ antisimétrico (pregunta
 18): la curva exponencial vive en el grupo de rotación
 $SO_n$. Calcúlelo completamente para $n = 2$: con $J =
 \begin{pmatrix}0 & -1\\ 1 & 0\end{pmatrix}$, demuestre
 \[
 \eu^{\theta J} = \begin{pmatrix} \cos\theta &
 -\sin\theta\\ \sin\theta & \cos\theta\end{pmatrix} :
 \]
 el \omterm{ex:b2:nvs:matrixexp}{matriz exponencial} \emph{is} la rotación por
 $\theta$, y las definiciones de series de coseno y seno
 reaparecer dentro de una matriz.
 \item (El truco de la densidad) Usando la densidad de $GL_n(\R)$
 (pregunta 3) y \omterm{def:b2:metric:continuity}{continuidad}, se extienden de invertible a
 todas las matrices la identidad
 \[
 \operatorname{adj}(AB) =
 \operatorname{adj}(B)\operatorname{adj}(A)
 \]
 \emph{(for invertible $A, B$ both sides equal
 $\det(AB)(AB)^{-1}$; both sides are polynomial in the
 entries)}.
 \item Síntesis. En una oración cada uno: (i) cuál anteriormente
 Los capítulos proporcionan el motor de cada parte.
 (\omterm{def:b2:metric:complete}{lo completo} y álgebras normadas; el espectro
 teorema; el teorema de la función inversa); (ii) cuál
 fórmula de este problema las ecuaciones diferenciales
 el capítulo se apoyará y dónde; (iii) qué
 $\dd(\det)_I = \operatorname{tr}$ y $\det\eu^A =
 \eu^{\operatorname{tr}A}$ dicen sobre rastreo y
 \omterm{def:b2:linalg:det}{determinante} como ``volumen infinitesimal y global'';
 (iv) qué hace el volumen del Año 3 con las preguntas
 21--23 (Grupos de Lie y sus álgebras de Lie).
\end{enumerate}
\end{problem}
