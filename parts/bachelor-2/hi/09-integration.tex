\chapter{एकीकरण}\label{ch:b2:integration}

वर्ष 1 खंड ने एक खंड पर अभिन्न अंग का निर्माण किया। यह अध्याय
इसे मनमाने अंतरालों तक विस्तारित करता है (अनुचित अभिन्न, पूर्ण के साथ
तुलना टूलकिट), फिर इंटीग्रल का अध्ययन करता है \emph{ए पर निर्भर करता है

संकेत --- प्रभुत्व \omterm{def:b2:integration:improper}{अभिसरण} प्रमेय द्वारा संचालित, एक
विश्वास पर लिये गये इस अध्याय का परिणाम. $\Gamma $ कार्य करता है
चल रहे उदाहरण के रूप में, और विशेष के आधे भाग के प्रवेश द्वार के रूप में
गणित के कार्य.

\section{एक मनमाना अंतराल पर इंटीग्रल}

\begin{definition}\label{def:b2:integration:improper}
होने देना $ f $ टुकड़े-टुकड़े हो जाओ \omterm{def:b2:metric:continuity}{निरंतर} पर $\intco{a}{b}$ ($ b \in \R $ या
$+\infty $). अभिन्न \emph{अभिसरण}\index{अभिसरण}
कब $\lim_{x \to b^-} \int_a^x f $ मौजूद है; फिर कोई लिखता है
$\int_a^b f $ सीमा के लिए. (इसी प्रकार चालू $\intoc{a}{b}$, और पर
$\intoo{a}{b}$ एक आंतरिक बिंदु पर विभाजन करके --- चुनाव होता है
कोई फर्क नहीं पड़ता, चैसल्स द्वारा।) अभिन्न \emph{अभिसरण \omterm{def:b2:series:def}{बिल्कुल}}
कब $\int_a^b \abs f $ अभिसरण; पूर्ण अभिसरण का तात्पर्य है
कॉची मानदंड द्वारा अभिसरण:
\[
\Bigl| \int_x^{y} f \Bigr| \leq \int_x^{y} \abs f
\]
और पूर्णता का $\R $ (आदिम के पास कॉची संपत्ति है)।
विस्तार से: चलो $ F(x) = \int_a^x f $ और $ G(x) = \int_a^x \abs
f $. अगर $\int^b\abs f $ जुटता है, $ G $ पर एक सीमा है $ b^-$, तो
प्रत्येक के लिए $\varepsilon > 0$ वहाँ है $ c < b $ साथ $ G(y) - G(x)
\leq \varepsilon $ जब कभी भी $ c \leq x \leq y < b $; प्रदर्शन
इस कॉची संपत्ति को स्थानांतरित करता है $ F $. किसी भी क्रम के लिए $ x_n
\to b^-$ मूल्य $ F(x_n)$ फिर एक कॉची अनुक्रम बनाएं
वास्तविक, द्वारा अभिसरण पूर्णता, और ऐसे दो को आपस में जोड़ना
अनुक्रम दर्शाते हैं कि सीमा उन सभी के लिए समान है: $ F $ है
पर एक सीमा $ b^-$.
\end{definition}

\begin{theorem}[सकारात्मक तुलना टूलकिट]\label{thm:b2:integration:comparison}
के लिए $ f, g \geq 0$ टुकड़े-टुकड़े में \omterm{def:b2:metric:continuity}{निरंतर} पर $\intco{a}{b}$:
\begin{enumerate}
  \item $\int_a^b f $ \omterm{def:b2:integration:improper}{अभिसरण} यदि आदिम $ x \mapsto \int_a^x
        f $ घिरा है;
  \item $ f \leq g $: का \omterm{def:b2:integration:improper}{अभिसरण} $\int g $ उसे मजबूर करता है $\int f $;
        विचलन दूसरे तरीके से स्थानांतरित होता है;
  \item $ f \sim g $ पर $ b $: दो अभिन्नों की प्रकृति समान है;
  \item संदर्भ तराजू: पर $+\infty $, $\int^{\infty}
        \frac{\dd t}{t^\alpha}$ \omterm{def:b2:integration:improper}{अभिसरण} आईएफएफ $\alpha > 1$, और
        $\int^\infty \frac{\dd t}{t(\ln t)^\beta}$ आईएफएफ $\beta > 1$;
        एक सीमित समापन बिंदु पर $ b $, $\int^b \frac{\dd t}{(b -
        t)^\alpha}$ \omterm{def:b2:integration:improper}{अभिसरण} आईएफएफ $\alpha < 1$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1)आदिम $ F(x) = \int_a^x f $ घट नहीं रहा है ($ f \geq
0$). यदि यह सीमाबद्ध है, $\ell = \sup_{x < b}F $ परिमित है और
$ F(x) \to \ell $: दिया गया $\varepsilon > 0$, कुछ $ F(x_0) > \ell -
\varepsilon $, और एकरसता जाल $ F(x) \in
\intoc{\ell - \varepsilon}{\ell}$ के लिए $ x_0 \leq x < b $. यदि यह है
असीमित, $ F \to +\infty $: विचलन.

(2) से $ f \leq g $: $\int_a^x f \leq \int_a^x g $ सभी के लिए $ x $;
अगर $\int^b g $ \omterm{def:b2:integration:improper}{अभिसरण}, दाहिना भाग घिरा हुआ है, इसलिए ऐसा है
बाएँ, और (1) समाप्त होता है। अंतर्विरोध विचलन को स्थानांतरित करता है
दूसरे तरीके से.

(3) $ f \sim g $ पर $ b $ प्रदान $ c < b $ साथ
\[
\tfrac12\,g(t) \;\leq\; f(t) \;\leq\; 2\,g(t)
\qquad (c \leq t < b) :
\]
(2) द्वारा दोनों तरीकों से लागू किया गया $\intco{c}{b}$, दो अभिन्न अंग
समान स्वभाव रखें; प्रारंभिक टुकड़ा $\intcc{a}{c}$ एक है
उचित अभिन्न और कुछ भी नहीं बदलता है।

(4) स्पष्ट आदिम: के लिए $\alpha \neq 1$ और $\beta \neq 1$,
\[
\int_c^x \frac{\dd t}{t^\alpha}
= \frac{x^{1-\alpha} - c^{1-\alpha}}{1 - \alpha},
\qquad
\int_c^x \frac{\dd t}{t(\ln t)^\beta}
= \frac{(\ln x)^{1-\beta} - (\ln c)^{1-\beta}}{1 - \beta},
\]
बहिष्कृत मामलों में लघुगणक के साथ: के रूप में परिबद्ध $ x \to
+\infty $ \omterm{def:b2:series:def}{बिल्कुल} कब $\alpha > 1$, सम्मान।\ $\beta > 1$. ए पर
परिमित समापन बिंदु, प्रतिस्थापन $ u = b - t $ को कम कर देता है
स्केल $\int_0 u^{-\alpha}\,\dd u $, परिबद्ध iff $\alpha < 1$.
हर बार (1) लागू करें।
\end{proof}

\begin{example}[दो वार्म-अप, अंत तक काम आए]\label{ex:b2:integration:warmups}
\emph{(ए)} $\displaystyle\int_0^1 \ln t\,\dd t $: इंटीग्रैंड
पर फूट पड़ता है $0^+$, लेकिन $\abs{\ln t} = o\bigl(t^{-1/2}\bigr)$
वहां (लघुगणक घातों से हार जाते हैं), और $\int_0 t^{-1/2}$
\omterm{def:b2:integration:improper}{अभिसरण}: पूर्ण \omterm{def:b2:integration:improper}{अभिसरण}. मूल्य, भागों द्वारा
$\intcc{\varepsilon}{1}$:
\[
\int_\varepsilon^1 \ln t\,\dd t
= \bigl[t\ln t - t\bigr]_\varepsilon^1
= -1 - \varepsilon\ln\varepsilon + \varepsilon
\xrightarrow[\varepsilon\to0^+]{} -1 .
\]
\emph{(बी)} $\displaystyle\int_0^\infty \frac{\ln t}{1 +
t^2}\,\dd t $: दोनों तरफ से परेशानी, इसलिए बंटवारा $1$. पास में $0$:
$\abs{\ln t}$ (ए) के रूप में अभिन्न; पास में $\infty $: $\frac{\ln
t}{1+t^2} = o(t^{-3/2})$: \omterm{def:b2:series:def}{बिल्कुल} अभिसारी। द
प्रतिस्थापन $ t = \frac1u $ एमएपीएस $\intoo{0}{1}$ पर
$\intoo{1}{\infty}$ और
\[
\int_0^1 \frac{\ln t}{1+t^2}\,\dd t
= \int_1^{\infty} \frac{-\ln u}{1 + u^{-2}}\cdot
\frac{\dd u}{u^2}
= -\int_1^\infty \frac{\ln u}{1+u^2}\,\dd u :
\]
दो हिस्से रद्द हो जाते हैं, और अभिन्न है $0$. समापन
अंतर्दृष्टि: समरूपता के तहत $ t \mapsto \frac1t $ के एक पेज के लायक है
गणना --- वही चाल पहले से ही संचालित है
\cref{exo:b2:integration:3}.
\end{example}

\begin{example}[एक मान, तीन अभिन्न]\label{ex:b2:integration:onecos}
अध्ययन $ I = \displaystyle\int_0^{\infty} \frac{1 - \cos
t}{t^2}\,\dd t $. पर $0$: $1 - \cos t \sim \frac{t^2}2$, तो
इंटीग्रैंड का विस्तार होता है लगातार मूल्य से $\frac12$ --- नहीं
\omterm{def:b2:series:def}{बिल्कुल} विलक्षणता. पर $\infty $: $0 \leq \frac{1 - \cos t}{t^2}
\leq \frac{2}{t^2}$: पूर्ण \omterm{def:b2:integration:improper}{अभिसरण}
(\cref{thm:b2:integration:comparison}). मूल्य: द्वारा एकीकृत करें
भागों पर $\intcc{\varepsilon}{M}$ साथ $ u = 1 - \cos t $, $ v' =
t^{-2}$:
\[
\int_\varepsilon^M \frac{1 - \cos t}{t^2}\,\dd t
= \Bigl[-\frac{1 - \cos t}{t}\Bigr]_\varepsilon^M
+ \int_\varepsilon^M \frac{\sin t}{t}\,\dd t .
\]
ब्रैकेट दोनों सिरों पर गायब हो जाता है ($\frac{1 -
\cos\varepsilon}{\varepsilon} \sim \frac\varepsilon2$; घिरा हुआ
अंश पर $ M $), और इंटीग्रल डिरिचलेट मान की ओर प्रवृत्त होता है
$\frac\pi2$ (\cref{exo:b2:integration:10}): $ I = \frac\pi2$.
समापन अंतर्दृष्टि: साथ $1 - \cos t = 2\sin^2\frac t2$ और $ u =
\frac t2$,
\[
I = \int_0^\infty \frac{2\sin^2 u}{(2u)^2}\;2\,\dd u
= \int_0^\infty \Bigl(\frac{\sin u}{u}\Bigr)^{\!2}\dd u :
\]
तीन क्लासिक्स $\int_0^\infty\frac{\sin t}{t}\dd t $,
$\int_0^\infty\bigl(\frac{\sin t}{t}\bigr)^2\dd t $
(\cref{exo:b2:integration:11}) और $ I $ सभी मूल्य साझा करते हैं
$\frac\pi2$, भागों और प्रतिस्थापन द्वारा पारित --- और
केवल पहला अर्\omterm{def:b2:integration:improper}{ध-अभिसरण} है: व्यापारित भागों द्वारा एकीकरण
एक सरल एकीकरण के लिए पूर्ण \omterm{def:b2:integration:improper}{अभिसरण} को दूर करें।
\end{example}

\begin{example}[एक अर्ध-अभिसरण अभिन्न]\label{ex:b2:integration:sint}
$\displaystyle\int_1^{\infty} \frac{\sin t}{t}\,\dd t $ \omterm{def:b2:integration:improper}{अभिसरण}:
भागों द्वारा एकीकृत करें,
\[
\int_1^x \frac{\sin t}{t}\dd t
= \Bigl[\frac{-\cos t}{t}\Bigr]_1^x - \int_1^x \frac{\cos
t}{t^2}\dd t ,
\]
जहां ब्रैकेट की एक सीमा और अंतिम अभिन्न अंग है \omterm{def:b2:integration:improper}{अभिसरण}
\omterm{def:b2:series:def}{बिल्कुल} ($\abs{\cos t}/t^2 \leq t^{-2}$). लेकिन नहीं \omterm{def:b2:series:def}{बिल्कुल}:
से $\abs{\sin t} \geq \sin^2 t $,
\[
\int_1^x \frac{\abs{\sin t}}{t}\,\dd t
\;\geq\; \int_1^x \frac{\sin^2t}{t}\,\dd t
= \underbrace{\int_1^x \frac{\dd t}{2t}}_{=\ \frac12\ln x
\ \to\ \infty}
\;-\; \underbrace{\int_1^x \frac{\cos 2t}{2t}\,\dd
t}_{\text{convergent}} ,
\]
जहां अंतिम अभिन्न \omterm{def:b2:integration:improper}{अभिसरण} से \emph{वही}
ऊपर बताए अनुसार भागों द्वारा एकीकरण (के साथ)। $\sin 2t $ कोष्ठक में):
एक भिन्न टुकड़ा घटा एक अभिसारी टुकड़ा अलग हो जाता है। इसलिए
$\int_1^\infty\frac{\sin t}{t}\dd t $ \omterm{def:b2:integration:improper}{अभिसरण} बिना
एकत्रित होना \omterm{def:b2:series:def}{बिल्कुल} --- का अभिन्न एनालॉग
भूमिका निभाने वाले भागों द्वारा एकीकरण के साथ, वैकल्पिक श्रृंखला
वैकल्पिक परीक्षण का.
\end{example}

\section{अभिसरण प्रमेय}

\begin{theorem}[प्रभुत्व अभिसरण]\label{thm:b2:integration:dominated}
होने देना $(f_n)$ टुकड़े-टुकड़े हो जाओ \omterm{def:b2:metric:continuity}{निरंतर} एक अंतराल पर $ I $, \omterm{def:b2:integration:improper}{अभिसरण}
बिंदुवार से टुकड़ेवार तक \omterm{def:b2:metric:continuity}{निरंतर} $ f $, और मान लीजिए कि वहाँ एक है
निश्चित अभिन्न $\varphi \geq 0$ ($\int_I \varphi < \infty $)
साथ\index{प्रभुत्व अभिसरण प्रमेय}
\[
\abs{f_n(t)} \leq \varphi(t) \qquad (t \in I,\ n \in \N).
\]
फिर सब $\int_I f_n $ और $\int_I f $ एकाग्र \omterm{def:b2:series:def}{बिल्कुल}, और
\[
\int_I f_n \xrightarrow[n \to \infty]{} \int_I f .
\]
\end{theorem}
\admitted

\begin{remark}
ईमानदार प्रमाण वर्ष के लेबेस्ग एकीकरण सिद्धांत से संबंधित है
3; हालाँकि, यह कथन अब से लगातार उपयोग किया जा रहा है।
\emph{प्रभुत्व} परिकल्पना संपूर्ण बिंदु है: बिंदुवार
अकेले \omterm{def:b2:integration:improper}{अभिसरण} पर्याप्त नहीं है ($ f_n = n\,\mathbf{1}_{\intoo{0}{1/n}}$,
फिसलने वाले उभार: $\int f_n = 1 \not\to 0 = \int f $). प्रमेय
ए के लिए भी धारण करता है \omterm{def:b2:metric:continuity}{निरंतर} पैरामीटर ($ f_\lambda $, $\lambda \to
\lambda_0$), सीमाओं के अनुक्रमिक लक्षण वर्णन द्वारा।
\end{remark}

\begin{example}[प्रभुत्व द्वारा एक गाऊसी सीमा]\label{ex:b2:integration:wallislimit}
गणना करना $\displaystyle\lim_{n\to\infty} I_n $ कहाँ $ I_n =
\int_0^\infty \Bigl(1 + \frac{t^2}{n}\Bigr)^{\!-n}\dd t $.
बिंदुवार, $(1 + t^2/n)^n \to \eu^{t^2}$ (चक्रवृद्धि-ब्याज
सीमा), इसलिए इंटीग्रैंड्स की प्रवृत्ति होती है $\eu^{-t^2}$. प्रभुत्व:
क्रम $ n \mapsto (1 + u/n)^n $ के लिए घटता नहीं है $ u \geq 0$
(AM--GM पर $ n + 1$ कारकों $1, 1 + \frac un, \dots, 1 +
\frac un $ देता है $(1 + \frac u{n+1})^{n+1} \geq (1 + \frac
un)^n $), अभीतक के लिए तो $ n \geq 2$:
\[
\Bigl(1 + \frac{t^2}{n}\Bigr)^{\!-n}
\leq \Bigl(1 + \frac{t^2}{2}\Bigr)^{\!-2},
\]
एक अभिन्न प्रभुत्वकर्ता ($\sim 4t^{-4}$ अनंत पर)। प्रभुत्व
\omterm{def:b2:integration:improper}{अभिसरण}:
\[
I_n \xrightarrow[n\to\infty]{} \int_0^\infty \eu^{-t^2}\dd t
= \frac{\sqrt\pi}{2}
\]
(गाऊसी अभिन्न अंग \cref{exo:b2:integration:8}). समापन
जाँच करें: प्रतिस्थापन $ t = \sqrt n\tan\theta $ गणना करता है $ I_n $
\omterm{def:b2:series:def}{बिल्कुल}, $ I_n = \sqrt n\int_0^{\pi/2}\cos^{2n-2}\theta\,\dd\theta
= \sqrt n\,W_{2n-2}$, और वालिस एसिम्प्टोटिक्स $ W_m \sim
\sqrt{\pi/(2m)}$ (\cref{lem:b2:comparison:wallis}) देना $\sqrt
n\,W_{2n-2} \to \frac{\sqrt\pi}2$ फिर से: इसके दो स्तंभ
अध्याय और अंतिम सहमत हैं।
\end{example}

\begin{example}[प्रभुत्व अभिसरण, सतत पैरामीटर]
\label{ex:b2:integration:arctanlimit}
गणना करना
\[
\lim_{x\to+\infty}\int_0^\infty
\frac{\arctan(xt)}{1+t^2}\,\dd t .
\]
प्रत्येक के लिए $ t > 0$,
$\arctan(xt) \to \frac\pi2$ जैसा $ x \to \infty $; और
प्रभुत्व
\[
\Bigl|\frac{\arctan(xt)}{1+t^2}\Bigr|
\leq \frac{\pi/2}{1+t^2},
\qquad\text{integrable, independent of } x,
\]
सभी के लिए धारण करता है $ x $. के सतत-पैरामीटर रूप द्वारा
\cref{thm:b2:integration:dominated} (क्रमिक
लक्षण वर्णन: प्रत्येक के साथ परीक्षण करें $ x_n \to \infty $),
\[
\int_0^\infty\frac{\arctan(xt)}{1+t^2}\,\dd t
\xrightarrow[x\to+\infty]{}
\frac\pi2\int_0^\infty\frac{\dd t}{1+t^2}
= \frac{\pi^2}{4} .
\]
समापन अंतर्दृष्टि: एकल बिंदु $ t = 0$, जहां बिंदुवार
सीमा है $0$ इसके बजाय $\frac\pi2$, कुछ भी नहीं बदलता---
सीमा फ़ंक्शन केवल इसके अभिन्न अंग के माध्यम से प्रवेश करता है, इनमें से एक
प्रमेय की शांत दया.
\end{example}

\section{एक पैरामीटर के साथ इंटीग्रल}

\begin{theorem}[अभिन्न चिह्न के अंतर्गत निरंतरता]\label{thm:b2:integration:continuity}
होने देना $ f \colon A \times I \to \R $ ($ A $ ए \omterm{def:b2:metric:def}{मीट्रिक स्थान}, $ I $ एक
अंतराल) के साथ: $ t \mapsto f(x, t)$ टुकड़े-टुकड़े में \omterm{def:b2:metric:continuity}{निरंतर} प्रत्येक के लिए
$ x $; $ x \mapsto f(x, t)$ \omterm{def:b2:metric:continuity}{निरंतर} प्रत्येक के लिए $ t $; और ए
\emph{प्रभुत्व} $\abs{f(x,t)} \leq \varphi(t)$ ($\varphi $
पर एकीकृत $ I $, से स्वतंत्र $ x $). तब
\[
F(x) = \int_I f(x, t)\,\dd t
\]
परिभाषित किया गया है और \omterm{def:b2:metric:continuity}{निरंतर} पर $ A $.\index{पैरामीटर अभिन्न}
\end{theorem}

\begin{proof}
परिभाषा: प्रभुत्व पूर्ण \omterm{def:b2:integration:improper}{अभिसरण} देता है। निरंतरता पर
$ x_0$: किसी भी क्रम के लिए $ x_n \to x_0$, कार्य $ g_n(t) =
f(x_n, t)$ बिंदुवार \omterm{def:b2:integration:improper}{अभिसरण} करें $ f(x_0, t)$ (निरंतरता में $ x $)
निश्चित प्रभुत्व के तहत $\varphi $: प्रभुत्व \omterm{def:b2:integration:improper}{अभिसरण} देता है
$ F(x_n) \to F(x_0)$; के अनुक्रमिक लक्षण वर्णन द्वारा निष्कर्ष निकालें
निरंतरता (\cref{def:b2:metric:continuity}).
\end{proof}

\begin{theorem}[अभिन्न चिह्न के अंतर्गत भेद]\label{thm:b2:integration:leibnizrule}
होने देना $ f \colon J \times I \to \R $ ($ J $ मापदंडों का एक अंतराल)
के साथ: $ t \mapsto f(x,t)$ पर एकीकृत $ I $ प्रत्येक के लिए $ x $; $ x \mapsto
f(x,t)$ कक्षा का $ C^1$ प्रत्येक के लिए $ t $, आंशिक व्युत्पन्न
$\frac{\partial f}{\partial x}$ टुकड़े-टुकड़े होना \omterm{def:b2:metric:continuity}{निरंतर} में $ t $
और प्रभुत्व: $\bigl|\frac{\partial f}{\partial x}(x,t)\bigr| \leq
\psi(t)$ साथ $\psi $ अभिन्न. तब $ F(x) = \int_I f(x,t)\dd t $
है $ C^1$ पर $ J $ और
\[
F'(x) = \int_I \frac{\partial f}{\partial x}(x, t)\,\dd t .
\]
\end{theorem}

\begin{proof}
हल करना $ x $ और $ h_n \to 0$. अंतर भागफल
\[
\frac{F(x + h_n) - F(x)}{h_n}
= \int_I \frac{f(x + h_n, t) - f(x, t)}{h_n}\,\dd t
\]
इंटीग्रैंड्स बिंदुवार \omterm{def:b2:integration:improper}{अभिसरण} कर रहे हैं $\frac{\partial f}{\partial
x}(x, t)$, और का प्रभुत्व है $\psi(t)$: माध्य मान से
असमानता लागू होती है $ x $ निश्चित पर $ t $,
\[
\Bigl|\frac{f(x + h_n, t) - f(x,t)}{h_n}\Bigr|
\leq \sup_{\xi} \Bigl|\frac{\partial f}{\partial x}(\xi, t)\Bigr|
\leq \psi(t) .
\]
प्रभुत्व \omterm{def:b2:integration:improper}{अभिसरण} सीमा देता है $\int_I \frac{\partial
f}{\partial x}(x,t)\dd t $ भागफल का: $ F $ विभेदनीय है
घोषित व्युत्पन्न के साथ, जो है \omterm{def:b2:metric:continuity}{निरंतर} द्वारा
\cref{thm:b2:integration:continuity} के लिए आवेदन किया $\frac{\partial
f}{\partial x}$.
\end{proof}

\begin{example}[एक पैरामीटर इंटीग्रल को एक सूत्र के विरुद्ध जांचा गया]\label{ex:b2:integration:arctanparam}
होने देना $ F(x) = \displaystyle\int_0^\infty \frac{\dd t}{t^2 + x}$
के लिए $ x > 0$. सभी पर $\intcc{a}{b} \subset \intoo{0}{\infty}$,
इंटीग्रैंड का प्रभुत्व है $\frac{1}{t^2 + a}$, अभिन्न
और स्वतंत्र $ x $: $ F $ है \omterm{def:b2:metric:continuity}{निरंतर}
(\cref{thm:b2:integration:continuity}). यहां प्रमेय हो सकता है
किसी स्पष्ट मान के विरुद्ध जाँच की गई:
\[
F(x) = \Bigl[\frac{1}{\sqrt x}\arctan\frac{t}{\sqrt
x}\Bigr]_0^\infty = \frac{\pi}{2\sqrt x} ,
\]
दिख \omterm{def:b2:metric:continuity}{निरंतर}. अब अभिन्न के अंतर्गत अंतर करें: द
$ x $-व्युत्पन्न $-\frac{1}{(t^2+x)^2}$ पर हावी है
$\intcc ab $ द्वारा $\frac{1}{(t^2+a)^2}$, अभिन्न:
\cref{thm:b2:integration:leibnizrule} देता है
\[
F'(x) = -\int_0^\infty \frac{\dd t}{(t^2 + x)^2}
\qquad\text{while}\qquad
F'(x) = -\frac{\pi}{4}\,x^{-3/2} ,
\]
तो हमारे पास है \emph{गणना} निःशुल्क एक नया अभिन्न अंग:
$\int_0^\infty\frac{\dd t}{(t^2+x)^2} = \frac{\pi}{4x^{3/2}}$.
समापन अंतर्दृष्टि: किसी ज्ञात को अलग करना \omterm{thm:b2:integration:continuity}{पैरामीटर अभिन्न} एक है
नए फ़ार्मुलों का कारखाना---पुनरावृत्ति देता है
$\int_0^\infty\frac{\dd t}{(t^2+1)^n}$ हरएक के लिए $ n $, नहीं के साथ
त्रिकोणमितीय प्रतिस्थापन.
\end{example}

\begin{method}[एक अनुचित अभिन्न अंग का अध्ययन]\label{met:b2:integration:study}
दिया गया $\int_a^b f $:
\begin{enumerate}
  \item समस्या का पता लगाएं: अंतिम बिंदुओं (या आंतरिक) को सूचीबद्ध करें
        अंक) कहाँ $ f $ असीमित है या अंतराल है
        अनंत, और विभाजित ताकि प्रत्येक टुकड़े में \omterm{def:b2:series:def}{बिल्कुल} एक हो
        परेशानी भरा अंत.
  \item अगर $ f $ उस सिरे के पास स्थिर चिह्न है, एक खोजें
        समतुल्य और के संदर्भ पैमानों के साथ तुलना करें
        \cref{thm:b2:integration:comparison}.
  \item अगर $ f $ दोलन करता है, परीक्षण करता है $\abs f $ प्रथम (पूर्ण)
        \omterm{def:b2:integration:improper}{अभिसरण})। अगर $\int\abs f $ विचलन, एकीकरण का प्रयास करें
        क्षय के लिए दोलन का व्यापार करने के लिए भागों द्वारा, जैसे कि
        \cref{ex:b2:integration:sint}; लघुकरण जैसे
        $\abs{\sin t} \geq \sin^2t $ असली का पता लगाएं
        अर्\omterm{def:b2:integration:improper}{ध-अभिसरण}.
  \item एक मूल्य के लिए, न कि केवल प्रकृति के लिए: भाग, प्रतिस्थापन,
        या एक पैरामीटर (एक सरल इंटीग्रल को अलग करें, जैसे कि
        \cref{ex:b2:integration:arctanparam} और
        \cref{ex:b2:integration:laplace}).
  \item विवेक किसी भी गणना किए गए मूल्य पर जांच करता है: संकेत और मोटा
        क्रूड बाउंड के विरुद्ध आकार ($\int_0^\infty
        \eu^{-t^2}\dd t \in \intoo{0}{1 + \int_1^\infty
        \eu^{-t}}$, इसलिए $\frac{\sqrt\pi}{2} \approx 0.886$ है
        प्रशंसनीय); और स्केलिंग के तहत आयामी स्थिरता
        ($ t \mapsto \lambda t $ दोनों पक्षों को फिर से स्केल करना होगा
        उसी तरह --- किसी खोए हुए कारक का सबसे तेज़ डिटेक्टर)।
\end{enumerate}
\end{method}

\begin{remark}[सामान्य ख़तरे]
तीन आवर्ती त्रुटियाँ. \emph{(i) पैरामीटर-निर्भर
प्रभुत्व रखने वाले:} प्रभुत्व $\abs{f(x,t)} \leq \varphi(t)$ अवश्य
में एक समान हो $ x $ सेट पर विचार किया गया; यह आमतौर पर कायम रहता है
खंड $\intcc ab $ लेकिन विश्व स्तर पर नहीं --- के लिए
$\int_0^\infty\eu^{-xt}\dd t $ कोई अभिन्न प्रभुत्वकर्ता नहीं है
सभी के लिए मान्य $ x > 0$, फिर भी हावी है $ x \geq a > 0$ है
समग्र रूप से काम करने के लिए पर्याप्त है \omterm{def:b2:metric:topology}{खुला} अर्ध-पंक्ति, तब से निरंतरता
और व्युत्पन्न स्थानीय धारणाएँ हैं। \emph{(ii) \omterm{thm:b2:structures:signature}{हस्ताक्षर}ित तुलना करना
एकीकृत:} तुलना टूलकिट के लिए है \emph{गैर नकारात्मक}
कार्य; से $\abs f \leq g $ साथ $\int g $ भिन्न हो सकता है
कुछ भी निष्कर्ष नहीं निकालें --- $\int_1^\infty\frac{\sin t}t\,\dd t $
\omterm{def:b2:integration:improper}{अभिसरण} हालाँकि हर तुलना के साथ $\frac1t $ विफल रहता है.
\emph{(iii) आधी परेशानी भूल जाना:} पर
$\intoo{0}{\infty}$ हमेशा दोनों सिरों का अलग-अलग अध्ययन करें;
$\int_0^\infty\frac{\dd t}{t}$ पर विचलन करता है \emph{दोनों}, और ए
अभिसरण-दिखने वाला विभाजन चुपचाप दो अनन्तताओं को रद्द कर सकता है।
सेफ रिफ्लेक्स की चेकलिस्ट है
\cref{met:b2:integration:study}.
\end{remark}

\begin{example}[एक बर्ट्रेंड सीमा मामला, अंक तक]\label{ex:b2:integration:bertrandboundary}
पैमाना $\int^\infty\frac{\dd t}{t(\ln t)^\beta}$ का
\cref{thm:b2:integration:comparison} \omterm{def:b2:series:def}{बिल्कुल} किनारे पर बैठता है
शक्ति तराजू का; इसके सीमा मामले एक पूर्ण के पात्र हैं
गणना. के लिए $\beta = 2$:
\[
\int_\eu^{\infty}\frac{\dd t}{t(\ln t)^2}
= \Bigl[-\frac{1}{\ln t}\Bigr]_\eu^{\infty}
= 0 - (-1) = 1 ,
\]
सुखद सटीक मूल्य के साथ एक \omterm{def:b2:integration:improper}{अभिसरण} अभिन्न अंग; जबकि इसके लिए
$\beta = 1$,
\[
\int_\eu^{x}\frac{\dd t}{t\ln t}
= \bigl[\ln\ln t\bigr]_\eu^{x} = \ln\ln x
\longrightarrow \infty ,
\]
भिन्न---लेकिन इतनी धीमी गति से कि पहुँचना $\ln\ln x = 10$
आवश्यक है $ x = \eu^{\eu^{10}} \approx 10^{9566}$. समापन
अंतर्दृष्टि: ``प्रत्येक शक्ति के बीच $ t^{-1-\varepsilon}$
\omterm{def:b2:integration:improper}{अभिसरण}'' और ``$ t^{-1}$ विचलन'' एक अनंत सीढ़ी जीता है
लघुगणकीय पैमानों में से प्रत्येक अंतिम को परिष्कृत करता है; प्रतिस्थापन
$ u = \ln t $ प्रत्येक पायदान को पिछले पायदान पर ढहा देता है, जो
यही कारण है कि बर्ट्रेंड मानदंड रीमैन मानदंडों को एक स्तर पर प्रतिध्वनित करते हैं
ऊपर.
\end{example}

\begin{remark}[इस खंड के भीतर परिप्रेक्ष्य]
इस अध्याय के उपकरण हर जगह उपलब्ध होने वाले हैं। प्रभुत्व
\omterm{def:b2:integration:improper}{अभिसरण} की अनुमानित पहचान के पीछे का इंजन है
अगला अध्याय (स्लाइडिंग कर्नेल, बर्नस्टीन और फ़ेज़\'एर
एक जैसे); निरंतरता और अभिन्न चिह्न के अंतर्गत विभेदन
फ़ोरियर में फ़ोरियर गुणांकों का कलन तैयार करें
अध्याय, जहां हर $ c_n(f)$ एक है \omterm{thm:b2:integration:continuity}{पैरामीटर अभिन्न} में
भेष बदलना। $\Gamma $ फ़ंक्शन दो बार लौटता है: अध्याय में
एकाधिक इंटीग्रल पर, जहां एक दोहरा इंटीग्रल अंततः सिद्ध होता है
यूलर का बीटा--गामा सूत्र पूर्ण रूप से, और संभाव्यता में
अध्याय, कहाँ $\Gamma $-टाइप इंटीग्रल्स मानक को सामान्य बनाते हैं
घनत्व और उनके क्षणों की गणना करें। और अर्ध-अभिसारी
$\int\frac{\sin t}{t}$ के गिब्स स्थिरांक के रूप में पुनः प्रकट होता है
फूरियर अध्याय --- वही अभिन्न, ओवरशूट को मापना
एक छलांग में आंशिक रकम की.
\end{remark}

\begin{definition}[ $\Gamma $ समारोह]\label{def:b2:integration:gamma}
के लिए $ x > 0$:\index{गामा फ़ंक्शन}
\[
\Gamma(x) = \int_0^{\infty} t^{x-1}\,\eu^{-t}\,\dd t ,
\]
दोनों सिरों पर \omterm{def:b2:integration:improper}{अभिसरण} ($ t^{x-1}$ पर अभिन्न $0^+$ के लिए $ x > 0$;
पर घातीय क्षय $\infty $).
\end{definition}

\begin{theorem}\label{thm:b2:integration:gammaprops}
$\Gamma $ है \omterm{def:b2:metric:continuity}{निरंतर} पर $\intoo{0}{+\infty}$, को संतुष्ट करता है
कार्यात्मक समीकरण
\[
\Gamma(x + 1) = x\,\Gamma(x),
\qquad \Gamma(1) = 1,
\qquad\text{hence}\qquad \Gamma(n + 1) = n! ,
\]
और क्लास का है $ C^1$ (वास्तव में $ C^\infty $) साथ $\Gamma'(x) =
\int_0^\infty t^{x-1}\eu^{-t}\ln t\,\dd t $.
\end{theorem}

\begin{proof}
कार्यात्मक समीकरण: भागों द्वारा एकीकृत करें $\intcc{\varepsilon}{M}$
और सिरों को जाने दो: $\int t^{x}\eu^{-t} = [-t^x\eu^{-t}] + x\int
t^{x-1}\eu^{-t}$, सीमा शब्द लुप्त हो रहे हैं --- वास्तव में
$\varepsilon^x\eu^{-\varepsilon} \to 0$ जैसा $\varepsilon \to
0^+$ क्योंकि $ x > 0$, और $ M^x\eu^{-M} \to 0$ जैसा $ M \to \infty $
क्योंकि घातांक हर शक्ति को मात देता है; दोनों को काट दिया गया
अभिन्न अंग \omterm{def:b2:integration:improper}{अभिसरण} द्वारा अपने अनुचित मूल्यों में परिवर्तित हो जाते हैं
में स्थापित \cref{def:b2:integration:gamma}. $\Gamma(1) = \int
\eu^{-t} = 1$; प्रेरण फैक्टोरियल देता है।

निरंतरता पर $\intcc{a}{b} \subset \intoo{0}{\infty}$: हावी होना
$ t^{x-1}\eu^{-t}$ द्वारा $\varphi(t) = (t^{a-1} + t^{b-1})\eu^{-t}$,
अभिन्न और स्वतंत्र $ x \in \intcc{a}{b}$:
\cref{thm:b2:integration:continuity} ऐसे हर खंड पर लागू होता है,
इसलिए पूरी आधी पंक्ति पर। भिन्नता: $ x $-व्युत्पन्न
$ t^{x-1}\eu^{-t}\ln t $ पर हावी है $\intcc{a}{b}$ द्वारा $(t^{a-1}
+ t^{b-1})\eu^{-t}\,\abs{\ln t}$, अभी भी अभिन्न:
\cref{thm:b2:integration:leibnizrule}; पुनरावृत्त करने से सब कुछ मिलता है
व्युत्पन्न (प्रत्येक एक शक्ति जोड़ता है $\ln t $, हानिरहित)।
\end{proof}

\begin{example}[अर्ध-पूर्णांक भाज्य]\label{ex:b2:integration:halffactorial}
कार्यात्मक समीकरण और $\Gamma\bigl(\frac12\bigr) =
\sqrt\pi $ (से दूर एक प्रतिस्थापन
\cref{exo:b2:integration:8}: तय करना $ t = u^2$ परिभाषित करने में
इंटीग्रल) सभी अर्ध-पूर्णांक मान उत्पन्न करता है:
\[
\Gamma\Bigl(\frac32\Bigr) = \frac12\,\Gamma\Bigl(\frac12\Bigr)
= \frac{\sqrt\pi}{2},
\qquad
\Gamma\Bigl(\frac52\Bigr) = \frac32\cdot\frac{\sqrt\pi}{2}
= \frac{3\sqrt\pi}{4},
\qquad
\Gamma\Bigl(\frac72\Bigr) = \frac{15\sqrt\pi}{8} .
\]
तब से $\Gamma(n+1) = n!$, यह कहना उचित है ``$\frac12! =
\frac{\sqrt\pi}{2} \approx 0.886$'': फैक्टोरियल रहा है
अंतर्वेशित होता है, और अंतर्वेशित वक्र नीचे की ओर झुकता है $1$
बीच में $0! = 1$ और $1! = 1$ (यह न्यूनतम है $\approx 0.8856$ पर
$ x \approx 1.4616$ की उत्तलता चित्र से मेल खाता है
सप्ताहांत समस्या का भाग I)। समापन अंतर्दृष्टि: इसमें कुछ भी नहीं
अभिन्न $\int_0^\infty t^{x-1}\eu^{-t}\dd t $ विशेषाधिकार
पूर्णांक --- भाज्य की विसंगति एक दुर्घटना थी
गिनती, और $\sqrt\pi $ वह है जो बीच में रहता है $1$ और $1$.
\end{example}

\begin{remark}[कहाँ $\Gamma $ यहाँ से जाता है]
इस अध्याय की सप्ताहांत समस्या संपूर्ण यूलर का निर्माण करती है
चारों ओर गणना $\Gamma $: द \omterm{pb:b2:integration:1}{बीटा फ़ंक्शन}, इसका एकीकरण-दर-भाग
रिकर्सन, वालिस बीटा मान और गॉस के रूप में एकीकृत होता है
सीमा सूत्र. एकाधिक अभिन्नों पर अध्याय यूलर की बात को सिद्ध करता है
बीटा--डबल इंटीग्रल द्वारा सभी तर्कों के लिए गामा सूत्र; द
संभाव्यता अध्याय मिलते हैं $\Gamma $ पुनः सामान्यीकरण में
सबसे आम घनत्व और प्रतीक्षा समय के क्षणों में।
वर्ष 3 खंड का पुनर्निर्माण $\Gamma $ लेब्सेग फाउंडेशन पर,
बोह्र-मोलेरुप विशिष्टता प्रमेय को सिद्ध करता है, और विस्तारित करता है
पूर्णांकों से वास्तविक अर्ध-पंक्ति तक स्टर्लिंग का सूत्र
प्रभुत्व \omterm{def:b2:integration:improper}{अभिसरण}.
\end{remark}

\begin{example}[विभेदीकरण द्वारा एक शास्त्रीय गणना]\label{ex:b2:integration:laplace}
के लिए $ x \in \R $, होने देना $ F(x) = \int_0^{\infty} \eu^{-t^2}\cos(xt)\,\dd
t $ (\omterm{def:b2:series:def}{बिल्कुल} अभिसारी, का प्रभुत्व $\eu^{-t^2}$). द्वारा
\cref{thm:b2:integration:leibnizrule} (का वर्चस्व
$ x $-व्युत्पन्न द्वारा $ t\,\eu^{-t^2}$, अभिन्न):
\[
F'(x) = -\int_0^\infty t\,\eu^{-t^2}\sin(xt)\,\dd t
= \Bigl[\frac{\eu^{-t^2}}{2}\sin(xt)\Bigr]_0^\infty
- \frac x2\int_0^\infty \eu^{-t^2}\cos(xt)\,\dd t
= -\frac x2\,F(x),
\]
(भागों के साथ $ u' = t\eu^{-t^2}$). विभेदक समीकरण $ F' =
-\frac x2 F $ को एकीकृत करता है $ F(x) = F(0)\,\eu^{-x^2/4}$: द
गाऊसी-प्रकार का अभिन्न अंग स्वयं को पुन: उत्पन्न करता है। स्थिरांक $ F(0) =
\int_0^\infty \eu^{-t^2}\dd t = \frac{\sqrt\pi}{2}$ में गणना की जाती है
\cref{exo:b2:integration:8} --- और फिर, दोहरे एकीकरण द्वारा,
में \cref{ch:b2:multint}.
\end{example}

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:integration:5}
होने देना $ F(x) = \displaystyle\int_0^{\infty} \frac{\eu^{-xt}}{1 +
t^2}\,\dd t $ के लिए $ x \geq 0$. साबित करें कि $ F $ है \omterm{def:b2:metric:continuity}{निरंतर} पर
$\intco{0}{\infty}$, $ C^2$ पर $\intoo{0}{\infty}$, संतुष्ट करता है $ F'' +
F = \frac1x $ वहाँ, और वह $ F(x) \to 0$ जैसा $ x \to +\infty $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:integration:6}
(फ्रूल्लानी) चलो $ f $ होना \omterm{def:b2:metric:continuity}{निरंतर} पर $\intco{0}{+\infty}$ एक के साथ
परिमित सीमा $ f(\infty)$ पर $+\infty $. इसके लिए सिद्ध करें $ a, b > 0$:
\[
\int_0^{\infty} \frac{f(at) - f(bt)}{t}\,\dd t
= \bigl(f(0) - f(\infty)\bigr)\,\ln\frac ba .
\]
\emph{(पर $\intcc{\varepsilon}{M}$, प्रत्येक टुकड़े में स्थानापन्न और
में पुनः समूहित करें $\int_{a\varepsilon}^{b\varepsilon} -
\int_{aM}^{bM}$ का $\frac{f(u)}u\,\dd u $; का उपयोग करके निचोड़ें
निरंतरता पर $0$ और सीमा पर $\infty $.)} गणना करना
$\int_0^\infty \frac{\eu^{-t} - \eu^{-2t}}{t}\dd t $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:integration:7}
औचित्य सिद्ध करें और गणना करें $\lim_{n\to\infty} \displaystyle\int_0^n
\Bigl(1 - \frac tn\Bigr)^{\!n} t^{x-1}\,\dd t $ के लिए $ x > 0$
\emph{(के साथ \omterm{def:b2:integration:improper}{अभिसरण} हावी है $\varphi(t) = \eu^{-t}t^{x-1}$,
का उपयोग कर रहे हैं $(1 - t/n)^n \leq \eu^{-t}$; सीमा है $\Gamma(x)$)}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:integration:8}
(एक पैरामीटर ट्रिक द्वारा गॉसियन इंटीग्रल) के लिए $ x \geq 0$ तय करना
\[
G(x) = \Bigl(\int_0^x \eu^{-t^2}\dd t\Bigr)^{\!2},
\qquad
H(x) = \int_0^1 \frac{\eu^{-x^2(1+t^2)}}{1 + t^2}\,\dd t .
\]
साबित करें कि $ G' + H' = 0$ (अंतर करें $ H $ अभिन्न के तहत और
स्थानापन्न $ u = xt $ परिणामी अभिन्न में), निष्कर्ष निकालें $ G(x) + H(x)
= \frac\pi4$ सभी के लिए $ x $, और निष्कर्ष निकालें
\[
\int_0^{\infty} \eu^{-t^2}\,\dd t = \frac{\sqrt\pi}{2} .
\]
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:integration:9}
साबित करें कि $\Gamma $ है \emph{लॉग-उत्तल}: $\ln\Gamma $ पर उत्तल है
$\intoo{0}{\infty}$. \emph{(कॉची--श्वार्ज़ इंटीग्रल के लिए लागू
को $ t^{(x+y)/2 - 1}\eu^{-t} = \bigl(t^{x-1}\eu^{-t}\bigr)^{1/2}
\bigl(t^{y-1}\eu^{-t}\bigr)^{1/2}$ देता है
$\Gamma\bigl(\frac{x+y}{2}\bigr)^2 \leq \Gamma(x)\Gamma(y)$;
के साथ संयुक्त निरंतरता और \cref{exo:b2:realfun:8}.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:integration:10}
(डिरिचलेट इंटीग्रल) सेट $ F(x) = \displaystyle\int_0^{\infty}
\frac{\sin t}{t}\,\eu^{-xt}\,\dd t $ के लिए $ x > 0$.
\begin{enumerate}
  \item औचित्य $ F'(x) = -\frac{1}{1 + x^2}$ (अंतर्गत अंतर करें
        अभिन्न; गणना करें $\int_0^\infty \eu^{-xt}\sin t\,\dd
        t $ भागों द्वारा दो एकीकरण द्वारा)।
  \item सिद्ध करना $ F(x) \to 0$ जैसा $ x \to +\infty $ और निष्कर्ष निकालें $ F(x) =
        \frac\pi2 - \arctan x $.
  \item स्वीकार कर रहा हूँ निरंतरता का $ F $ पर $0^+$ (एक हाबिल-प्रकार
        प्रमेय), अर्\omterm{def:b2:integration:improper}{ध-अभिसरण} का मान समाप्त करें
        अभिन्न:
        \[
        \int_0^{\infty} \frac{\sin t}{t}\,\dd t = \frac{\pi}{2}.
        \]
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:integration:11}
के \omterm{def:b2:integration:improper}{अभिसरण} का औचित्य सिद्ध कीजिए $\displaystyle\int_0^\infty
\Bigl(\frac{\sin t}{t}\Bigr)^{\!2}\dd t $, फिर इसे एक से गिनें
भागों द्वारा एकीकरण और \cref{exo:b2:integration:10}:
\[
\int_0^\infty \Bigl(\frac{\sin t}{t}\Bigr)^{\!2}\dd t
= \frac{\pi}{2} .
\]
(समान मान $\int_0^\infty \frac{\sin t}{t}\dd t $ --- लेकिन
इस बार \omterm{def:b2:integration:improper}{अभिसरण} पूर्ण है।)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:integration:12}
(गाऊसी पूँछ) के लिए $ x > 0$ तय करना $ T(x) = \displaystyle
\int_x^\infty \eu^{-t^2}\dd t $.
\begin{enumerate}
  \item लिखना $\eu^{-t^2} = \frac{1}{-2t}\cdot(-2t\,\eu^{-t^2})$,
        प्राप्त करने के लिए दो बार भागों द्वारा एकीकृत करें
        \[
        T(x) = \eu^{-x^2}\Bigl(\frac{1}{2x} -
        \frac{1}{4x^3}\Bigr) +
        \frac34\int_x^\infty \frac{\eu^{-t^2}}{t^4}\,\dd t .
        \]
  \item शेष को बांधें: $0 \leq \frac34\int_x^\infty
        t^{-4}\eu^{-t^2}\dd t \leq \frac{3}{8x^5}\,\eu^{-x^2}$,
        और ब्रैकेटिंग निकालें
        \[
        \eu^{-x^2}\Bigl(\frac{1}{2x} - \frac{1}{4x^3}\Bigr)
        \leq T(x) \leq \frac{\eu^{-x^2}}{2x},
        \qquad\text{hence}\qquad
        T(x) \sim \frac{\eu^{-x^2}}{2x} \quad (x \to +\infty).
        \]
  \item पुनरावृत्त करके पूर्ण प्रत्यावर्ती श्रृंखला क्यों प्राप्त की जा सकती है?
        हिस्से कभी भी स्थिर के लिए एकत्रित नहीं होते $ x $? \emph{(तुलना करें
        गुणांकों की वृद्धि $1\cdot3\cdots(2k-1)$ के साथ
        शक्तियां $(2x^2)^k $.)}
\end{enumerate}
\end{exercise}

\section{समस्या: यूलर इंटीग्रल्स --- बीटा, गामा और गॉस
सीमा सूत्र}

\begin{problem}\label{pb:b2:integration:1}
 $\Gamma $ का कार्य \cref{def:b2:integration:gamma} एक है
यूलर के अभिन्नों की गणना का आधा; दूसरा भाग है
\emph{बीटा फ़ंक्शन}\index{बीटा फ़ंक्शन}
\[
B(x, y) = \int_0^1 t^{x-1}(1 - t)^{y-1}\,\dd t .
\]
यह समस्या जोड़ी को विकसित करती है $(\Gamma, B)$ के उपकरणों के साथ
यह अध्याय केवल --- भागों द्वारा एकीकरण, प्रतिस्थापन,
प्रभुत्व \omterm{def:b2:integration:improper}{अभिसरण} --- और समापन होता है \emph{यूलर का
बीटा--गामा सूत्र} $ B(x,y) = \frac{\Gamma(x)\Gamma(y)}
{\Gamma(x+y)}$ अर्ध-पूर्णांकों पर और में \emph{गॉस की सीमा
सूत्र}\index{गॉस की सीमा
सूत्र} के लिए $\Gamma $. रास्ते में
वालिस का अभिन्न अंग \cref{lem:b2:comparison:wallis} पुनः प्रकट होना
बीटा मान के रूप में, और लिजेंड्रे का दोहराव फॉर्मूला समाप्त हो जाता है।

\medskip
\noindent\textbf{Part I --- Fine structure of $\Gamma $.}
\begin{enumerate}
  \item स्मरण करो क्यों $\Gamma(x) = \int_0^\infty t^{x-1}\eu^{-t}\dd
        t $ \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:series:def}{बिल्कुल} के लिए $ x > 0$, और दिखाओ
        \[
        \Gamma(x) \sim \frac1x \qquad (x \to 0^+)
        \]
        \emph{(कार्यात्मक समीकरण प्लस निरंतरता का $\Gamma $ पर
        $1$)}.
  \item सिद्ध करना $\Gamma\bigl(\tfrac12\bigr) = \sqrt\pi $
        \emph{(विकल्प $ t = u^2$ और आह्वान करें
        \cref{exo:b2:integration:8})}, और निष्कर्ष निकालें
        $\int_\R \eu^{-u^2/2}\dd u = \sqrt{2\pi}$.
  \item प्रेरण द्वारा दिखाएँ, के लिए $ n \in \N $:
        \[
        \Gamma\Bigl(n + \frac12\Bigr)
        = \frac{(2n)!}{4^n\,n!}\,\sqrt\pi .
        \]
  \item औचित्य $\Gamma''(x) = \int_0^\infty
        t^{x-1}\eu^{-t}(\ln t)^2\dd t > 0$, और उसका निष्कर्ष निकालो
        $\Gamma $ सख्ती से उत्तल है, एक अद्वितीय न्यूनतम प्राप्त करता है
        कुछ $ x_0 \in \intoo{1}{2}$ \emph{($\Gamma(1) = \Gamma(2)
        = 1$ और रोले)}, पर घटता है $\intoo{0}{x_0}$ और
        पर बढ़ता है $\intoo{x_0}{\infty}$.
  \item बताते हैं कि $\Gamma $ हर शक्ति को धड़कता है: प्रत्येक के लिए $ k \in
        \N $, $ x^k = o\bigl(\Gamma(x)\bigr)$ जैसा $ x \to +\infty $
        \emph{(निचोड़ $ x $ पूर्णांकों और उपयोग के बीच $\Gamma(n+1)
        = n!$ प्रश्न 4 की एकरसता के साथ)}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The \omterm{pb:b2:integration:1}{बीटा फ़ंक्शन}, by parts.}
\begin{enumerate}[resume]
  \item बताते हैं कि $ B(x,y)$ \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:series:def}{बिल्कुल} के लिए $ x > 0$ और $ y >
        0$, ओर वो $ B(x,y) = B(y,x)$.
  \item गणना करना $ B(x, 1) = \frac1x $, और एकीकरण द्वारा सिद्ध करें
        भागों, के लिए $ x, y > 0$:
        \[
        B(x, y+1) = \frac{y}{x}\,B(x+1, y) .
        \]
  \item बंटवारे से $ t^{x-1}(1-t)^{y-1} =
        t^{x}(1-t)^{y-1} + t^{x-1}(1-t)^{y}$ परिणाम निकालना $ B(x,y) =
        B(x+1,y) + B(x,y+1)$, और प्रश्न 7 के साथ संयोजित करें
        द \emph{वंश संबंध}
        \[
        B(x, y+1) = \frac{y}{x+y}\,B(x,y),
        \qquad
        B(x+1, y) = \frac{x}{x+y}\,B(x,y) .
        \]
  \item निष्कर्ष निकालें, के लिए $ m, n \geq 1$ पूर्णांक:
        \[
        B(m, n) = \frac{(m-1)!\,(n-1)!}{(m+n-1)!}
        = \frac{1}{(m+n-1)\binom{m+n-2}{m-1}} .
        \]
  \item सिद्ध करना \emph{एक पूर्णांक तर्क के लिए यूलर का सूत्र}:
        हरएक के लिए $ x > 0$ और $ n \in \N^*$,
        \[
        B(x, n) = \frac{\Gamma(x)\,\Gamma(n)}{\Gamma(x + n)}
        \]
        \emph{(प्रेरण पर $ n $: दोनों पक्ष बराबर $\frac1x $ पर $ n
        = 1$ और उसी वंश संबंध का पालन करें)}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Wallis integrals as Beta values.}
\begin{enumerate}[resume]
  \item विकल्प $ t = \sin^2\theta $ त्रिकोणमिति प्राप्त करने के लिए
        प्रपत्र
        \[
        B(x, y) = 2\int_0^{\pi/2}
        \sin^{2x-1}\theta\,\cos^{2y-1}\theta\,\dd\theta .
        \]
  \item परिणाम निकालना $ W_n = \frac12\,B\bigl(\frac{n+1}2, \frac12\bigr)$
        वालिस इंटीग्रल के लिए $ W_n = \int_0^{\pi/2}\sin^n
        \theta\,\dd\theta $, और पुनरावृत्ति पुनर्प्राप्त करें $ W_n =
        \frac{n-1}{n}W_{n-2}$ का \cref{lem:b2:comparison:wallis}
        अकेले प्रश्न 8 के वंश संबंधों से।
  \item गणना करना $ B\bigl(\frac12, \frac12\bigr) = 2W_0 = \pi $ और
        इसके विरुद्ध जाँच करें $\Gamma\bigl(\frac12\bigr)^2/\Gamma(1)$:
        यूलर का सूत्र कायम है $\bigl(\frac12,
        \frac12\bigr)$.
  \item बंद प्रपत्र प्राप्त करें $ W_{2n} = \frac\pi2\,
        \frac{(2n)!}{4^n(n!)^2}$ पुनरावृत्ति से, और सत्यापित करें
        \[
        B\Bigl(n + \frac12, \frac12\Bigr)
        = \frac{\Gamma\bigl(n + \frac12\bigr)\Gamma\bigl(
        \frac12\bigr)}{\Gamma(n+1)} .
        \]
        अवरोही संबंधों के साथ प्रेरण द्वारा निष्कर्ष निकालें
        यूलर का सूत्र $ B(x,y) =
        \frac{\Gamma(x)\Gamma(y)}{\Gamma(x+y)}$ जब भी धारण करता है
        $2x $ और $2y $ धनात्मक पूर्णांक हैं.
  \item विकल्प $ u = \frac{t}{1-t}$ तीसरा पाने के लिए
        शास्त्रीय रूप
        \[
        B(x,y) = \int_0^\infty
        \frac{u^{x-1}}{(1+u)^{x+y}}\,\dd u ,
        \]
        और मामले की जांच करें $ x = y = \frac12$ सीधे ($ u = v^2$
        इसे कम कर देता है $\int_0^\infty\frac{2\,\dd v}{1+v^2}$).
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Gauss's limit formula.}
\begin{enumerate}[resume]
  \item के लिए $ x > 0$ और $ n \in \N^*$, द्वारा सिद्ध करें $ n $ क्रमिक
        भागों द्वारा एकीकरण:
        \[
        \int_0^n \Bigl(1 - \frac tn\Bigr)^{\!n} t^{x-1}\,\dd t
        = \frac{n!\;n^x}{x(x+1)\cdots(x+n)} .
        \]
  \item के साथ समापन करें \cref{exo:b2:integration:7} (प्रभुत्व
        \omterm{def:b2:integration:improper}{अभिसरण}) \emph{गॉस का सीमा सूत्र}:
        \[
        \Gamma(x) = \lim_{n\to\infty}
        \frac{n!\;n^x}{x(x+1)\cdots(x+n)}
        \qquad (x > 0).
        \]
  \item लघुगणक लेते हुए, उसे दिखाएँ $ x > 0$:
        \[
        \ln\Gamma(x) = -\ln x - \gamma x +
        \sum_{k=1}^{\infty}\Bigl(\frac xk -
        \ln\Bigl(1 + \frac xk\Bigr)\Bigr),
        \]
        कहाँ $\gamma $ यूलर स्थिरांक है
        (\cref{ex:b2:comparison:harmonic}); का औचित्य सिद्ध करें
        श्रृंखला का \omterm{def:b2:integration:improper}{अभिसरण} \emph{(सामान्य शब्द है
        $\sim \frac{x^2}{2k^2}$)}.
  \item गॉस के सूत्र का प्रयोग करें $ x = \frac12$ और केंद्रीय
        द्विपद स्पर्शोन्मुखता $\binom{2n}{n} \sim
        \frac{4^n}{\sqrt{\pi n}}$
        (\cref{ex:b2:comparison:centralbinomial}) पुनर्गणना करना
        $\Gamma\bigl(\frac12\bigr) = \sqrt\pi $: स्टर्लिंग का
        स्थिरांक और गाऊसी समाकलन एक ही संख्या हैं
        दो वेश में.
  \item जांचें कि गॉस का सूत्र कार्यात्मकता को दोहराता है
        समीकरण: सटीक पहचान से
        \[
        \frac{n!\,n^{x+1}}{(x+1)\cdots(x+n+1)}
        = \frac{n!\,n^{x}}{x(x+1)\cdots(x+n)}\cdot
        \frac{n\,x}{x+n+1},
        \]
        निष्कर्ष $\Gamma(x+1) = x\,\Gamma(x)$ दोबारा। (गॉस का
        सूत्र निर्धारित करता है $\Gamma $ एकमुश्त; वर्ष 3 खंड
        अधिक तीव्र बोह्र-मोलेरुप प्रमेय को सिद्ध करता है:
        कार्यात्मक समीकरण प्लस लॉग-उत्तलता पहले से ही पिन
        $\Gamma $ नीचे।)
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends.}
\begin{enumerate}[resume]
  \item के लिए $ a > 0$ दिखाओ $\int_0^\infty \eu^{-t^a}\dd t =
        \Gamma\bigl(1 + \frac1a\bigr)$, और सीमा की गणना इस प्रकार करें
        $ a \to +\infty $ प्रभुत्व \omterm{def:b2:integration:improper}{अभिसरण} द्वारा \emph{(बिंदुवार
        सीमा $\mathbf 1_{t < 1}$; का प्रभुत्व $1$ पर
        $\intoc{0}{1}$ और तक $\eu^{-t^2}$ परे, के लिए $ a \geq
        2$)}. के सामने उत्तर की जाँच करें निरंतरता का
        $\Gamma $.
  \item के लिए $ n \geq 1$ दिखाओ
        \[
        \int_0^1 \frac{\dd t}{\sqrt{1 - t^n}}
        = \frac1n\,B\Bigl(\frac1n, \frac12\Bigr),
        \]
        और मूल्यों को पुनर्प्राप्त करें $2$ ($ n = 1$) और $\frac\pi2$ ($ n
        = 2$). (के लिए $ n = 4$ ये है \emph{लेम्निस्केट
        स्थिर}, जिसका कोई प्रारंभिक बंद रूप नहीं है; इसका
        कहानी अण्डाकार समाकलन के सिद्धांत से संबंधित है।)
  \item (क्षण) के लिए $ x > 0$ और $ k \in \N $, दिखाओ
        \[
        \frac{1}{\Gamma(x)}\int_0^\infty
        t^{k}\,t^{x-1}\eu^{-t}\,\dd t
        = \frac{\Gamma(x+k)}{\Gamma(x)}
        = x(x+1)\cdots(x+k-1),
        \]
        बढ़ती फैक्टोरियल; जाँच करें कि $ x = 1$ देता है $ k!$. (में
        संभाव्यता अध्याय यह है $ k $-ए का वां क्षण
        मानक प्रतीक्षा-समय घनत्व।)
  \item बीटा पहचान साबित करें, के लिए मान्य \emph{सभी} $ x > 0$:
        \[
        B(x, x) = 2^{1-2x}\,B\Bigl(x, \frac12\Bigr)
        \]
        \emph{(विकल्प $ t = \frac{1+s}2$, शोषण करें
        में समरूपता $ s $, फिर सेट करें $ s = \sqrt v $)}. निष्कर्ष निकालें, के लिए
        $2x \in \N^*$, \emph{लीजेंड्रे का दोहराव सूत्र}
        \[
        \Gamma(x)\,\Gamma\Bigl(x + \frac12\Bigr)
        = 2^{1-2x}\,\sqrt\pi\;\Gamma(2x),
        \]
        और इसे सीधे सत्यापित करें $ x = n $ प्रश्न 3 के माध्यम से (के लिए)
        सामान्य $ x $ यह एक बार एक ही पहचान से चलता है
        यूलर का सूत्र सभी तर्कों के लिए जाना जाता है ---
        एकाधिक पर अध्याय में दोहरा-अभिन्न प्रमाण
        अभिन्न।)
  \item संश्लेषण। प्रत्येक एक वाक्य में: (i) जहां एकीकरण
        भागों द्वारा संपूर्ण भाग II को आगे बढ़ाया गया; (ii) कहां
        प्रभुत्व वाले \omterm{def:b2:integration:improper}{अभिसरण} ने भाग IV और V में प्रवेश किया; (iii) जो
        तुलना से स्पर्शोन्मुख इनपुट आयात किए गए थे
        अध्याय; (iv) अब यूलर के सूत्र से क्या सिद्ध हो गया है
        $ B(x,y) = \Gamma(x)\Gamma(y)/\Gamma(x+y)$, और क्या
        दोहरे अभिन्न अंग का निपटान बाकी है।
\end{enumerate}
\end{problem}
