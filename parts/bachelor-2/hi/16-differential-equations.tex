\chapter{विभेदक समीकरण}\label{ch:b2:diffeq}

वर्ष 1 ने उन रैखिक समीकरणों को हल किया जो सूत्रों को स्वीकार करते हैं। यह अध्याय
वह आपूर्ति करता है जो सूत्र नहीं कर सकते: द \emph{कॉची--लिप्सचिट्ज़ प्रमेय}
--- अस्तित्व और विशिष्टता के लिए $y' = f(t, y)$ --- के साथ साबित हुआ
बानाच निश्चित बिंदु प्रमेय, \omterm{def:b2:series:def}{बिल्कुल} वैसा ही जैसा वादा किया गया था
\cref{ch:b2:metric}; फिर का \omterm{def:b2:metric:complete}{पूरा} सिद्धांत \emph{रैखिक
सिस्टम} $ X' = A(t)X + B(t)$, साथ \omterm{ex:b2:nvs:matrixexp}{मैट्रिक्स घातांक} और यह
\omterm{def:b2:diffeq:wronskian}{रोन्स्कियन} कम्प्यूटेशनल इंजन के रूप में।

\section{कॉची-लिप्सचिट्ज़ प्रमेय}

\begin{theorem}[कॉची--लिप्सचिट्ज़, वैश्विक लिप्सचिट्ज़ संस्करण]\label{thm:b2:diffeq:cauchylipschitz}
होने देना $ I $ एक खंड हो, $ f \colon I \times \R^n \to \R^n $ \omterm{def:b2:metric:continuity}{निरंतर}
और \emph{\omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} दूसरे वेरिएबल में}, \omterm{def:b2:funcseq:def}{समान रूप से} में
प्रथम: $\norm{f(t, y) - f(t, z)} \leq k\,\norm{y - z}$ सभी के लिए $ t
\in I $. फिर हर एक के लिए $(t_0, y_0) \in I \times \R^n $, कॉची
समस्या\index{कॉची--लिप्सचिट्ज़ प्रमेय}
\[
y' = f(t, y), \qquad y(t_0) = y_0
\]
\omterm{def:b2:series:def}{बिल्कुल} एक ही समाधान है $ y \colon I \to \R^n $ कक्षा का $ C^1$.
\end{theorem}

\begin{proof}
\emph{सुधार.} ए \omterm{def:b2:metric:continuity}{निरंतर} $ y $ यदि यह समस्या हल करता है
अभिन्न समीकरण को संतुष्ट करता है
\[
y(t) = y_0 + \int_{t_0}^{t} f\bigl(s, y(s)\bigr)\,\dd s
=: T(y)(t)
\]
(दोनों तरीकों से कलन का मौलिक प्रमेय; a \omterm{def:b2:metric:continuity}{निरंतर} समाधान
अभिन्न समीकरण का स्वचालित रूप से है $ C^1$).

\emph{पुन: सामान्यीकरण के बाद एक संकुचन।} पर \omterm{def:b2:nvs:banach}{बनच स्थान} $ E =
C(I, \R^n)$ साथ \emph{भारित} आदर्श
\[
N(y) = \sup_{t \in I}\; \eu^{-2k\abs{t - t_0}}\,\norm{y(t)} ,
\]
(सुपर के बराबर) आदर्श: वजन ऊपर और नीचे बंधा हुआ है
खंड पर $ I $, इसलिए $ E $ रहता है \omterm{def:b2:metric:complete}{पूरा}), के लिए अनुमान $ y, z \in
E $ और कहते हैं, $ t \geq t_0$:
\[
\norm{T(y)(t) - T(z)(t)}
\leq \int_{t_0}^{t} k\,\norm{y(s) - z(s)}\,\dd s
\leq k\,N(y - z)\int_{t_0}^{t} \eu^{2k(s - t_0)}\dd s
\leq \frac{N(y-z)}{2}\,\eu^{2k(t - t_0)} .
\]
से गुणा करना $\eu^{-2k(t - t_0)}$ और समर्थन लेना (मामला)। $ t
< t_0$ है \omterm{def:b2:quadratic:adjoint}{सममित}): $ N\bigl(T(y) - T(z)\bigr) \leq \frac12 N(y -
z)$: $ T $ एक है $\frac12$- का संकुचन \omterm{def:b2:metric:complete}{पूरा} $(E, N)$. द
बानाच निश्चित बिंदु प्रमेय (\cref{thm:b2:metric:banach}) एक उपज देता है
अद्वितीय निश्चित बिंदु: अद्वितीय समाधान।
\end{proof}

\begin{remark}
के लिए $ f $ केवल $ C^1$ (स्थानीय तौर पर \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़}), प्रमेय कायम है
\emph{स्थानीय रूप से}, अधिकतम पर अधिकतम समाधान के साथ \omterm{def:b2:metric:topology}{खुला} \omterm{def:b2:diffcalc:differential}{अंतर}ाल;
समाधान सीमित समय में विस्फोटित हो सकते हैं ($ y' = y^2$, $ y(0) = 1$: $ y(t)
= \frac{1}{1-t}$, पर चला गया $ t = 1$). वैश्विक \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} परिकल्पना
वही है जो पूरे खंड को खरीदता है। उत्कीर्णन के लायक दो परिणाम:
एक ODE के समाधान वक्र के साथ \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} मैदान \emph{कभी पार न करें};
और शून्य फलन कहीं भी गायब होने वाला एकमात्र समाधान है
रैखिक सजातीय समीकरण.
\end{remark}

\begin{example}[विशिष्टता एक प्रमेय है: एक ऐसा क्षेत्र जो लीक हो जाता है]\label{ex:b2:diffeq:leaky}
विचार करना $ y' = 2\sqrt{\abs y}$ साथ $ y(0) = 0$. शून्य
फ़ंक्शन इसे हल करता है; ऐसा करता है
\[
y(t) = \begin{cases} 0 & t \leq 0,\\ t^2 & t \geq 0,
\end{cases}
\]
जो है $ C^1$ (दोनों टुकड़े व्युत्पन्न हैं $0$ गोंद पर
बिंदु) और संतुष्ट करता है $ y'(t) = 2t = 2\sqrt{t^2}$ के लिए $ t > 0$ ---
वास्तव में, टेक-ऑफ में देरी करने से इसका समाधान मिल जाता है \emph{प्रत्येक}
जारी करने का समय $ c \geq 0$: के माध्यम से अनंत रूप से कई समाधान
वही प्रारंभिक डेटा. के साथ कोई विरोधाभास नहीं
\cref{thm:b2:diffeq:cauchylipschitz}: पास में $ y = 0$,
\[
\frac{\abs{2\sqrt y - 2\sqrt z}}{\abs{y - z}}
= \frac{2}{\sqrt y + \sqrt z} \longrightarrow +\infty ,
\]
फ़ील्ड नहीं है \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} में $ y $, और प्रमेय मौन है।
समापन \omterm{def:b2:diffcalc:differential}{अंतर}्दृष्टि: भौतिक पढ़ना एक बाल्टी नाली है
गुरुत्वाकर्षण के तहत पीछे की ओर दौड़ें --- खाली अवस्था से, एक
कब भरना शुरू हुआ, कह नहीं सकते; ODE का नियतिवाद है
बिलकुल \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} परिस्थिति, प्रकृति का नियम नहीं।
\end{example}

\section{रैखिक प्रणाली}

\begin{theorem}[रैखिक प्रणालियों की संरचना]\label{thm:b2:diffeq:linear}
होने देना $ A \colon I \to \mathcal{M}_n(\R)$ और $ B \colon I \to \R^n $ होना
\omterm{def:b2:metric:continuity}{निरंतर} एक \omterm{def:b2:diffcalc:differential}{अंतर}ाल पर $ I $. हरएक के लिए $(t_0, X_0)$ समस्या
\[
X' = A(t)X + B(t), \qquad X(t_0) = X_0
\]
\omterm{def:b2:series:def}{बिल्कुल} एक समाधान चालू है \emph{सभी} का $ I $. के समाधान
सजातीय प्रणाली ($ B = 0$) एक सदिश समष्टि बनाएं $\mathcal{S}_H $ का
आयाम \omterm{def:b2:series:def}{बिल्कुल} $ n $, और मूल्यांकन $ X \mapsto X(t_0)$ एक है
समरूपता $\mathcal{S}_H \to \R^n $; सामान्य समाधान $=$
विशिष्ट $+$ सजातीय.
\end{theorem}

\begin{proof}
हर खंड पर $ J \subseteq I $ युक्त $ t_0$: $ f(t, X) =
A(t)X + B(t)$ है \omterm{def:b2:metric:continuity}{निरंतर}, और \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} में $ X $ स्थिरांक के साथ $ k
= \sup_J \vertiii{A(t)}$ (परिमित: \omterm{def:b2:metric:continuity}{निरंतर} एक खंड पर):
\cref{thm:b2:diffeq:cauchylipschitz} पर लागू होता है $ J $; दे $ J $
निकास $ I $, विशिष्टता समाधानों को एक में जोड़ती है $ I $.
समाधान सेट और मूल्यांकन मानचित्र की रैखिकता स्पष्ट है;
मूल्यांकन अस्तित्व (सजेक्टिव) और विशिष्टता द्वारा व्यक्तिपरक है
(इंजेक्शन): $\dim \mathcal{S}_H = n $. एफ़िन संरचना वर्ष है
1 का तर्क शब्दशः।
\end{proof}

\begin{example}[मूल्यांकन समरूपता, ठोस रूप से]\label{ex:b2:diffeq:evaliso}
के लिए $ y'' + y = 0$, सिस्टम के रूप में देखा जाता है $ X' =
\begin{pmatrix} 0 & 1\\ -1 & 0\end{pmatrix}X $ साथ $ X = (y,
y')$: प्रमेय कहता है कि समाधान स्थान एक समतल है, और वह
$ X \mapsto X(0) = (y(0), y'(0))$ पर एक समरूपता है
$\R^2$. समाधान $\cos $ और $\sin $ का मूल्यांकन करें $(1, 0)$
और $(0, 1)$ --- का विहित आधार $\R^2$ --- तो वे बनते हैं
समाधान स्थान का एक आधार, और \emph{प्रत्येक} समाधान है
\[
y(t) = y(0)\cos t + y'(0)\sin t ,
\]
गुणांक के साथ प्रारंभिक डेटा को सीधे पढ़ें, नहीं
हल करने के लिए रैखिक प्रणाली। समापन \omterm{def:b2:diffcalc:differential}{अंतर}्दृष्टि: चुनना
\omterm{def:b2:diffeq:wronskian}{मौलिक प्रणाली} जिनके प्रारंभिक मान विहित हैं
आधार (यहाँ) $\cos, \sin $) वास्तव में कॉलम का चयन कर रहा है
$\eu^{tA}$; मूल्यांकन समरूपता प्रारंभिक क्यों है
स्थितियाँ पैरामीट्रिज़ प्रक्षेपवक्र --- ज्यामितीय सामग्री
रैखिक समीकरणों के लिए ``नियतात्मक गतिशीलता'' की।
\end{example}

\begin{definition}[रोन्स्कियन]\label{def:b2:diffeq:wronskian}
समाधान के लिए $ X_1, \dots, X_n $ सजातीय प्रणाली का,
\emph{रोन्स्कियन}\index{रोन्स्कियन} है $ W(t) = \det\bigl(X_1(t),
\dots, X_n(t)\bigr)$. उपरोक्त समरूपता द्वारा, या तो $ W $ गायब हो जाता है
\omterm{def:b2:funcseq:def}{समान रूप से} (परिवार जुड़ा हुआ है) या कभी नहीं (ए)। \emph{मौलिक
प्रणाली}\index{मौलिक
प्रणाली}); मात्रात्मक रूप से, $ W' = \operatorname{tr}\bigl(A(t)\bigr)
W $, इसलिए
\[
W(t) = W(t_0)\,\exp\Bigl(\int_{t_0}^{t}
\operatorname{tr} A(s)\,\dd s\Bigr)
\quad \text{(Liouville's formula)}.
\]
\end{definition}

\begin{example}[लिउविले ने यूलर समीकरण पर जाँच की]\label{ex:b2:diffeq:eulerwronskian}
पर $\intoo{0}{\infty}$, समीकरण $ t^2y'' + ty' - y = 0$ है
समाधान $ y_1(t) = t $ और $ y_2(t) = \frac1t $ (विकल्प)।
उनका \omterm{def:b2:diffeq:wronskian}{रोन्स्कियन}:
\[
W(t) = \det\begin{pmatrix} t & \tfrac1t\\[2pt]
1 & -\tfrac{1}{t^2}\end{pmatrix}
= -\frac1t - \frac1t = -\frac2t ,
\]
कभी शून्य नहीं: ए \omterm{def:b2:diffeq:wronskian}{मौलिक प्रणाली}. अब लिउविले की जाँच करें: में
सामान्यीकृत रूप $ y'' + \frac1t\,y' - \frac{1}{t^2}\,y = 0$, द
साथी मैट्रिक्स $ A(t) = \begin{pmatrix} 0 & 1\\ \frac{1}{t^2}
& -\frac1t\end{pmatrix}$ निशान है $-\frac1t $, इसलिए
\[
W(t) = W(1)\exp\Bigl(-\int_1^t\frac{\dd s}{s}\Bigr)
= -2\,\eu^{-\ln t} = -\frac2t . \checkmark
\]
समापन \omterm{def:b2:diffcalc:differential}{अंतर}्दृष्टि: लिउविल भविष्यवाणी करता है \emph{आकार} की
\omterm{def:b2:diffeq:wronskian}{रोन्स्कियन} इससे पहले कि कोई समाधान ज्ञात हो --- यहाँ, वह $ W $ अवश्य
हो $\frac{c}{t}$; यही शक्ति है
\omterm{def:b2:structures:generated}{आदेश} में कमी की विधि
(\cref{prop:b2:diffeq:secondorder}), जहां जानना $ y_1$ और यह रोन्स्कियनका स्वरूप निर्धारित करता है $ y_2$ एक चतुर्भुज द्वारा.
\end{example}

\begin{proof}[लिउविल के सूत्र का प्रमाण]
$ W(t) = \det M(t)$ साथ $ M' = AM $. भेद करना सिद्ध
स्तंभों के एक बहुरेखीय कार्य के रूप में,
\[
W'(t) = \sum_j \det(X_1, \dots, X_j', \dots, X_n)
= \sum_j \det(X_1, \dots, AX_j, \dots, X_n) .
\]
अब नक्शा $(C_1, \dots, C_n) \mapsto \sum_j \det(C_1, \dots,
AC_j, \dots, C_n)$ है $ n $-रैखिक और प्रत्यावर्ती (दो बराबर के साथ)।
कॉलम $ C_i = C_k $, शर्तें $ j \notin \{i, k\}$ एकदम गायब हो जाना,
और शर्तें $ j = i $ और $ j = k $ एक कॉलम के बाद \omterm{def:b2:quadratic:adjoint}{जोड़}े में रद्द करें
स्वैप): विशिष्टता प्रमेय द्वारा
(\cref{thm:b2:linalg:detspace}) यह है $ c \cdot \det $, साथ $ c $ पढ़ें
विहित स्तंभों पर: $ c = \sum_j \det(e_1, \dots, Ae_j, \dots,
e_n) = \sum_j a_{jj} = \operatorname{tr} A $. इस तरह $ W' =
\operatorname{tr}\bigl(A(t)\bigr)W $: एक अदिश रैखिक ODE, द्वारा हल किया गया
वर्ष 1 सूत्र.
\end{proof}

\section{लगातार गुणांक: मैट्रिक्स घातांक}

\begin{theorem}\label{thm:b2:diffeq:matrixexp}
के लिए $ A \in \mathcal{M}_n(\R)$ (या $\C $), घातीय $\eu^{tA}
= \sum_k \frac{(tA)^k}{k!}$ (\cref{ex:b2:nvs:matrixexp}) संतुष्ट करता है:
$ t \mapsto \eu^{tA}$ है $ C^1$ (वास्तव में $ C^\infty $)
साथ\index{मैट्रिक्स घातांक}
\[
\frac{\dd}{\dd t}\,\eu^{tA} = A\,\eu^{tA} = \eu^{tA}A ,
\qquad
\eu^{(s+t)A} = \eu^{sA}\,\eu^{tA},
\qquad
(\eu^{A})^{-1} = \eu^{-A} ;
\]
और $\eu^{A + B} = \eu^A\eu^B $ \emph{कब} $ AB = BA $. द कॉची
समस्या $ X' = AX $, $ X(0) = X_0$ अद्वितीय समाधान है $ X(t) =
\eu^{tA}X_0$; एक स्रोत के साथ, \emph{स्थिरांक की भिन्नता}
सूत्र धारण करता है:
\[
X(t) = \eu^{(t - t_0)A}X_0 + \int_{t_0}^{t} \eu^{(t-s)A}B(s)\,\dd
s .
\]
\end{theorem}

\begin{proof}
\emph{भिन्नता:} शृंखला $\sum \frac{t^kA^k}{k!}$ और
इसकी शब्द-व्युत्पन्न श्रृंखला $\sum \frac{t^{k-1}A^k}{(k-1)!} = A\sum
\frac{(tA)^{k-1}}{(k-1)!}$ एकाग्र \omterm{def:b2:funcseq:series}{सामान्यतः} हर खंड पर
(मानदंड $\leq \frac{(\abs t\,\vertiii A)^k}{k!}$): \omterm{def:b2:diffcalc:differential}{अंतर} करना
पद दर पद (\cref{thm:b2:funcseq:seriestransfer}, वेक्टर-मूल्यवान)।
दो \omterm{def:b2:structures:generated}{आदेश} $ A\eu^{tA}$ और $\eu^{tA}A $ प्रत्येक से सहमत हूँ
आंशिक राशि साथ चलती है $ A $.

\emph{समूह कानून:} आवागमन के लिए $ A, B $, द \omterm{thm:b2:series:fubini}{कॉची उत्पाद} का
दो घातीय श्रृंखलाएं द्विपद प्रमेय द्वारा सटीक रूप से पुनर्गठित होती हैं
जैसा कि \cref{ex:b2:series:exp} (बानाच में पूर्ण \omterm{def:b2:integration:improper}{अभिसरण}
बीजगणित इसे उचित ठहराता है): $\eu^{A+B} = \eu^A\eu^B $; साथ $ B = sA $ यह
एक-पैरामीटर समूह कानून देता है, और $ B = -A $ उलटा.

\emph{कॉची समस्या:} $ X(t) = \eu^{tA}X_0$ इसे हल करता है (\omterm{def:b2:diffcalc:differential}{अंतर});
द्वारा विशिष्टता \cref{thm:b2:diffeq:linear}. स्थिरांक का परिवर्तन:
सेट $ Y(t) = \eu^{-tA}X(t)$; विभेद करना, $ Y' = \eu^{-tA}(X' -
AX) = \eu^{-tA}B(t)$; से एकीकृत करें $ t_0$ को $ t $ और गुणा करें
पीछे से $\eu^{tA}$.
\end{proof}

\begin{method}[कम्प्यूटिंग $\eu^{tA}$]\label{met:b2:diffeq:computeexp}
कम करना $ A $ (\cref{ch:b2:reduction}): अगर $ A = PDP^{-1}$ विकर्ण,
$\eu^{tA} = P\,\eu^{tD}P^{-1}$ साथ $\eu^{tD}$ का विकर्ण
$\eu^{t\lambda_i}$; सामान्य तौर पर डनफोर्ड का उपयोग करें $ A = D + N $
(आवागमन): $\eu^{tA} = \eu^{tD}\,\eu^{tN}$ साथ $\eu^{tN}$ ए
\emph{बहुपद} में $ t $ (निलपोटेंस श्रृंखला को छोटा कर देता है)। जटिल
eigenvalues घूर्णन-समय-घातीय ब्लॉकों में युग्मित करें
(\cref{exo:b2:diffeq:5}).
\end{method}

\begin{remark}[सामान्य ख़तरे]
\emph{(मैं) $\eu^{A+B} \neq \eu^A\eu^B $ बिना रूपांतरण के:}
लेना $ A = \begin{pmatrix} 0 & 1\\ 0 & 0\end{pmatrix}$, $ B =
\begin{pmatrix} 0 & 0\\ 1 & 0\end{pmatrix}$. तब $\eu^A = I +
A $, $\eu^B = I + B $ (निलपोटेंस), इसलिए
\[
\eu^A\eu^B = \begin{pmatrix} 2 & 1\\ 1 & 1\end{pmatrix},
\qquad\text{while}\qquad
\eu^{A+B} = \cosh(1)\,I + \sinh(1)\,(A + B)
= \begin{pmatrix} \cosh 1 & \sinh 1\\ \sinh 1 & \cosh 1
\end{pmatrix},
\]
का उपयोग करते हुए $(A+B)^2 = I $; और $\cosh 1 \approx 1.54 \neq 2$. द
का समूह कानून \cref{thm:b2:diffeq:matrixexp} एक असली ले जाता है
परिकल्पना. \emph{(ii) रैखिक मैदान पर अरेखीय \omterm{def:b2:diffcalc:differential}{अंतर}्ज्ञान:}
ए के समाधान \emph{रेखीय} सिस्टम के साथ \omterm{def:b2:metric:continuity}{निरंतर}
गुणांक पूरे \omterm{def:b2:diffcalc:differential}{अंतर}ाल पर रहते हैं
(\cref{thm:b2:diffeq:linear}) --- यदि कोई उम्मीदवार समाधान करता है
अंदर फूट पड़ता है $ I $, समीकरण रैखिक नहीं था या
गणना ग़लत है; इसके विपरीत, अरेखीय समीकरणों के लिए
बिना तर्क के कभी भी वैश्विकता का वादा न करें ($ y' = y^2$).
\emph{(iii) अज्ञात द्वारा विभाजन:} में चरों को अलग करना
$ y' = y(1-y)$ \omterm{def:b2:metric:continuity}{निरंतर} समाधानों को चुपचाप त्याग देता है $0$ और
$1$ --- \omterm{def:b2:series:def}{बिल्कुल} वही जो चरण रेखा को व्यवस्थित करते हैं
(\cref{exo:b2:diffeq:3}); पहले \omterm{def:b2:metric:continuity}{निरंतर} समाधानों की सूची बनाएं।
\emph{(iv) प्रारंभिक डेटा सदिशों को ठीक करता है, अदिश को नहीं:} एक $ n $-वें
\omterm{def:b2:structures:generated}{आदेश} अदिश समीकरण की जरूरत है $ n $ स्थितियाँ ($ y, y', \dots $ पर
$ t_0$); केवल मिलान $ y(t_0)$ एक छोड़ देता है
$(n-1)$-पैरामीटर परिवार, ``खोया हुआ'' का एक क्लासिक स्रोत
स्थिरांक
\end{remark}

\begin{example}[ए $3\times3$ डनफोर्ड द्वारा घातीय]\label{ex:b2:diffeq:dunford3}
हल करना $ X' = AX $ के लिए $ A = \begin{pmatrix} 2 & 1 & 0\\ 0 & 2 &
0\\ 0 & 0 & 3\end{pmatrix}$. ब्लॉक द्वारा डनफोर्ड: $ A = D + N $
साथ $ D = \operatorname{diag}(2, 2, 3)$ और $ N = E_{12}$, जो
आवागमन ($ N $ \omterm{def:b2:reduction:eigen}{eigenvalue} के अंदर रहता है-$2$ ब्लॉक), और $ N^2 =
0$:
\[
\eu^{tA} = \eu^{tD}\,\eu^{tN}
= \begin{pmatrix}
\eu^{2t} & t\,\eu^{2t} & 0\\
0 & \eu^{2t} & 0\\
0 & 0 & \eu^{3t}
\end{pmatrix} .
\]
सामान्य समाधान कॉलम दर कॉलम पढ़ता है: $ X(t) =
\bigl(\eu^{2t}(x_0 + ty_0),\ \eu^{2t}y_0,\
\eu^{3t}z_0\bigr)$. स्वच्छता जांच: पर $ t = 0$ मैट्रिक्स है
$ I $; इसका सिद्ध है $\eu^{7t} =
\eu^{t\operatorname{tr}A}$, जैसा कि लिउविल मांग करता है; और
कारक $ t $ \omterm{def:b2:series:def}{बिल्कुल} वहीं दिखाई देता है जहां \omterm{def:b2:reduction:eigen}{eigenvalue} $2$ टूटा हुआ है। समापन \omterm{def:b2:diffcalc:differential}{अंतर}्दृष्टि: बहुपद गुणा घातांक हैं
याद रखने योग्य कोई अनुमान नहीं --- वे काट-छाँट की गई शृंखला हैं
$\eu^{tN}$, और उनकी डिग्री शून्यक्षमता से बंधी होती है
सूचकांक, कभी अधिक नहीं.
\end{example}

\begin{method}[हल $ X' = AX + B(t)$, समाप्त करने के लिए प्रारंभ करें]\label{met:b2:diffeq:solvelinear}
\begin{enumerate}
  \item \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} का $ A $; तब $\eu^{tA}$ द्वारा
        \cref{met:b2:diffeq:computeexp} (विकर्णीकृत करें; या
        डनफोर्ड के रूप में \cref{ex:b2:diffeq:dunford3}; या ए
        बहुपद युक्ति जैसे $ A^2 = -I $).
  \item एक विशेष समाधान: स्थिरांक की भिन्नता
        $\int_{t_0}^t\eu^{(t-s)A}B(s)\dd s $ हमेशा काम करता है;
        घातीय-बहुपद के लिए $ B $, उसी का एक अंश
        आकार (डिग्री पर उठाया गया \omterm{pb:b2:diffeq:1}{गूंज},
        \cref{exo:b2:diffeq:10}) तेज है.
  \item सामान्य समाधान $= \eu^{(t-t_0)A}X_0 +$ विशिष्ट;
        प्रारंभिक डेटा को फिट करें \emph{अंतिम}, पूर्ण पर
        सूत्र.
  \item स्वच्छता जांच: $ X(t_0)$ सही; सजातीय भाग
        विकास मेल खाता है \omterm{def:b2:reduction:eigen}{eigenvalue} वास्तविक भाग
        (\cref{exo:b2:diffeq:8}); और $\det $ एक मौलिक का
        मैट्रिक्स लिउविले का पालन करता है।
\end{enumerate}
\end{method}

\begin{example}[एक चरण चित्र]\label{ex:b2:diffeq:phaseportrait}
$ X' = AX $ साथ $ A = \begin{pmatrix} 0 & 1\\ -1 & 0\end{pmatrix}$:
$ A^2 = -I $, इसलिए श्रृंखला विभाजित हो जाती है
\[
\eu^{tA} = (\cos t)\,I + (\sin t)\,A
= \begin{pmatrix} \cos t & \sin t\\ -\sin t & \cos t
\end{pmatrix} :
\]

द eigenvalues का $ A $ की वृद्धि या क्षय तय करें $\norm{X(t)}$

\end{example}

\section{चर गुणांकों के साथ दूसरा क्रम}

\begin{proposition}\label{prop:b2:diffeq:secondorder}
समीकरण $ y'' + a(t)y' + b(t)y = c(t)$ ($ a, b, c $ \omterm{def:b2:metric:continuity}{निरंतर} पर
$ I $) प्रणाली है $ X' = A(t)X + B(t)$ के लिए $ X = (y, y')$: समाधान
मौजूद हैं और सभी में अद्वितीय हैं $ I $ किसी भी प्रारंभिक डेटा के लिए $(y(t_0),
y'(t_0))$; समांगी विलयन एक तल बनाते हैं। यदि कोई लुप्त न हो
सजातीय समाधान $ y_1$ ज्ञात है, दूसरा स्वतंत्र है
द्वारा पाया गया \emph{ऑर्डर कम करना}: सेटिंग $ y = y_1 z $ बदल देता है
सजातीय समीकरण को प्रथम-क्रम समीकरण में बदलें $ z'$, हल हो गया
चतुर्भुज द्वारा.
\end{proposition}

\begin{proof}
सिस्टम फॉर्म और \cref{thm:b2:diffeq:linear} सब कुछ दे दो
संरचनात्मक. कम करना: प्रतिस्थापित करना $ y = y_1z $,
\[
y_1 z'' + (2y_1' + a y_1)z' + \underbrace{(y_1'' + ay_1' +
by_1)}_{=\,0}\,z = 0 :
\]
प्रथम कोटि का रैखिक समीकरण $ u = z'$, वर्ष 1 तक हल करने योग्य
सूत्र; एकीकृत $ u $ देता है $ z $, इस तरह $ y_2 = y_1 z $,
से स्वतंत्र $ y_1$ जब कभी भी $ z $ अचर है.
\end{proof}

\begin{example}\label{ex:b2:diffeq:lowering}
$ t^2y'' - 2y = 0$ पर $\intoo{0}{\infty}$: $ y_1 = t^2$ एक है
समाधान. विकल्प $ y = t^2z $: से $ y' = t^2z' + 2tz $ और $ y''
= t^2z'' + 4tz' + 2z $,
\[
t^2y'' - 2y = t^4 z'' + 4t^3z' = 0,
\qquad\text{i.e.}\qquad \frac{z''}{z'} = -\frac4t :
\]
$ z' = t^{-4}$ (एक स्थिरांक तक), $ z = -\frac{1}{3t^3}$, और $ y_2
= t^2z = -\frac{1}{3t}$. सामान्य समाधान: $ y = \alpha t^2 +
\frac{\beta}{t}$.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b2:diffeq:1}
हल करना $ X' = AX $, $ X(0) = (1, 0)^{\mathsf T}$, के लिए $ A =
\begin{pmatrix} 1 & 1\\ 0 & 2\end{pmatrix}$ (विकर्ण) और $ A =
\begin{pmatrix} 2 & 1\\ 0 & 2 \end{pmatrix}$ (डनफोर्ड)।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:diffeq:2}
किन कॉची समस्याओं के अद्वितीय वैश्विक समाधान हैं? $\R $ द्वारा
\cref{thm:b2:diffeq:cauchylipschitz}? $ y' = \sin(ty)$; $\;y' =
y^2$; $\;y' = \abs y $. आखिरी के लिए, स्पष्ट रूप से हल करें $ y(0) =
0$ और $ y(0) = 1$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:diffeq:3}
साबित करें कि दो अलग-अलग अधिकतम समाधान हैं $ y' = f(t,y)$ ($ f $
\omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} में $ y $) कभी भी एक ही समय में समान मान न लें, और
का समाधान निकालें $ y' = y(1 - y)$ में शुरू
$\intoo{0}{1}$ में रहो $\intoo{0}{1}$ हमेशा के लिए।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffeq:4}
गणना करना $\eu^{tA}$ के लिए $ A = \begin{pmatrix} 3 & 1\\ -1 &
1\end{pmatrix}$ \emph{(डनफोर्ड: $(A - 2I)^2 = 0$)}, और हल करें $ X' =
AX + \begin{pmatrix} \eu^{2t}\\ 0\end{pmatrix}$, $ X(0) = 0$, द्वारा
स्थिरांक की भिन्नता.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffeq:5}
के लिए $ A = \begin{pmatrix} \alpha & -\beta\\ \beta &
\alpha\end{pmatrix}$, सिद्ध करना $\eu^{tA} =
\eu^{\alpha t}\begin{pmatrix} \cos\beta t & -\sin\beta t\\
\sin\beta t & \cos\beta t\end{pmatrix}$ --- सर्पिल प्रक्षेपवक्र ---
दो तरीके: श्रृंखला के माध्यम से (लिखें $ A = \alpha I + \beta J $, $ J^2 =
-I $), और जटिल पहचान के माध्यम से $ z' = (\alpha +
\iu\beta)z $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffeq:6}
(ग्रोनवाल की लेम्मा) चलो $ u $ होना \omterm{def:b2:metric:continuity}{निरंतर} के साथ गैरनकारात्मक $ u(t)
\leq C + k\int_{t_0}^{t} u(s)\,\dd s $ पर $\intco{t_0}{T}$. सिद्ध करना
$ u(t) \leq C\,\eu^{k(t - t_0)}$ \emph{(\omterm{def:b2:diffcalc:differential}{अंतर} करें $ v(t) =
\eu^{-kt}\int_{t_0}^t u $)}. फिर से विशिष्टता का पता लगाएं
कॉची--लिप्सचिट्ज़ और \omterm{def:b2:metric:continuity}{निरंतर} निर्भरता $\norm{y(t) -
z(t)} \leq \norm{y_0 - z_0}\,\eu^{k\abs{t - t_0}}$ दो के लिए
विभिन्न प्रारंभिक डेटा के साथ समाधान।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffeq:7}
जानते हुए भी $ y_1(t) = \frac{\sin t}{t}$ हल करती है $ ty'' + 2y' + ty =
0$ पर $\intoo{0}{\pi}$, द्वारा दूसरा स्वतंत्र समाधान खोजें
\omterm{def:b2:structures:generated}{आदेश} कम करें, और सामान्य समाधान दें।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:diffeq:8}
होने देना $ A \in \mathcal{M}_n(\C)$ सभी के साथ eigenvalues का (कड़ाई से)
नकारात्मक वास्तविक भाग. सिद्ध करें कि प्रत्येक समाधान $ X' = AX $ आदत है
$0$ जैसा $ t \to +\infty $, एक घातीय दर के साथ: $\norm{X(t)}
\leq C\,\eu^{-\alpha t}$ कुछ के लिए $\alpha > 0$.
\emph{(त्रिकोणीकरण करें; अंतिम पंक्ति से \omterm{def:b2:reduction:diag}{त्रिकोणीय} प्रणाली का इलाज करें
ऊपर, या डनफोर्ड का उपयोग करें: $\eu^{tA} = \eu^{tD}\eu^{tN}$ साथ
$\norm{\eu^{tD}} \leq \eu^{-\alpha' t}$ और $\eu^{tN}$ बहुपद
में $ t $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:diffeq:9}
(रैखिक विकास के लिए सीमित समय में कोई बच नहीं सकता) मान लीजिए $ f $ है
\omterm{def:b2:metric:continuity}{निरंतर} साथ $\norm{f(t, y)} \leq a\norm y + b $ पर
$\intco{0}{\infty} \times \R^n $, स्थानीय स्तर पर \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} में $ y $. उपयोग करना
ग्रोनवाल (\cref{exo:b2:diffeq:6}) अभिन्न रूप पर, सिद्ध करें
अधिकतम समाधान वैश्विक हैं (सभी पर परिभाषित)।
$\intco{0}{\infty}$).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:diffeq:10}
हल करना $ y'' - 3y' + 2y = \eu^{t}$: सजातीय समाधान, फिर ए
प्रपत्र का विशेष समाधान $\alpha t\,\eu^{t}$ \emph{(क्यों?
क्या भोला अनुमान लगाता है $\alpha\eu^t $ असफल?)}; सामान्य समाधान और
समाधान के साथ $ y(0) = y'(0) = 0$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffeq:11}
गणना करना $\eu^{tA}$ जॉर्डन ब्लॉक के लिए
\[
A = \begin{pmatrix} \lambda & 1 & 0\\ 0 & \lambda & 1\\
0 & 0 & \lambda\end{pmatrix},
\]
और सभी समाधानों का वर्णन करें $ X' = AX $: घातीय बार
बहुपद सदिश, तक की घातों के साथ $2$. कहाँ होता है
बहुपद डिग्री कहाँ से आती है?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:diffeq:12}
(आवधिक दबाव, आवधिक प्रतिक्रिया) चलो $ A \in
\mathcal{M}_n(\R)$ और $ B \colon \R \to \R^n $ \omterm{def:b2:metric:continuity}{निरंतर} और
$ T $-आवधिक.
\begin{enumerate}
  \item इसका कोई समाधान दिखाएँ $ X' = AX + B(t)$ है
        $ T $-आवधिक यदि और केवल यदि $ X(T) = X(0)$
        \emph{(तुलना करना $ X(\cdot + T)$ और $ X $)}.
  \item बताते हैं कि eigenvalues का $\eu^{TA}$ हैं
        $\eu^{T\lambda}$, $\lambda \in \operatorname{Sp}A $
        \emph{(तिकोणीकरण करें $\C $)}. निष्कर्ष निकालें: यदि नहीं
        \omterm{def:b2:reduction:eigen}{eigenvalue} का $ A $ में निहित है $\frac{2\iu\pi}{T}\Z $, तब
        $ I - \eu^{TA}$ उलटा है.
  \item उस परिकल्पना के तहत, साबित करें कि सिस्टम में है
        \omterm{def:b2:series:def}{बिल्कुल} एक $ T $-आवधिक समाधान, साथ
        \[
        X(0) = \bigl(I - \eu^{TA}\bigr)^{-1}
        \int_0^{T}\eu^{(T-s)A}B(s)\,\dd s .
        \]
        बहिष्कृत मामला किससे मेल खाता है, के लिए
        हार्मोनिक थरथरानवाला? (सप्ताहांत समस्या का उत्तर:
        \omterm{pb:b2:diffeq:1}{गूंज}.)
\end{enumerate}
\end{exercise}

\section{समस्या: दोलन, प्रतिध्वनि, और स्टर्म
तुलना प्रमेय}

\begin{problem}\label{pb:b2:diffeq:1}
एक समीकरण यांत्रिक और विद्युत दुनिया को नियंत्रित करता है:
\[
x'' + 2\zeta\omega\,x' + \omega^2 x = F(t),
\qquad \omega > 0,\ \zeta \geq 0 .
\]
इस समस्या का पूर्ण रूप से अध्ययन किया जाता है --- के माध्यम से
ट्रेस--तलीय रैखिक प्रणालियों का निर्धारक वर्गीकरण,
तीन अवमंदन व्यवस्थाएँ, स्थिर-अवस्था प्रतिक्रिया
इसके साथ आवधिक जबरदस्ती \emph{गूंज}\index{गूंज}
शिखर और \omterm{pb:b2:diffeq:1}{गूंज} प्रलय---फिर स्थिर छोड़ देता है
के लिए गुणांक \emph{स्टर्म का पृथक्करण और तुलना
प्रमेय}\index{स्टर्म का पृथक्करण और तुलना
प्रमेय}, जो नियंत्रित करता है
के समाधान के शून्य $ y'' + q(t)y = 0$ बिना किसी फ़ॉर्मूले के
सब.

\medskip
\noindent\textbf{Part I --- The trace--determinant plane.} होने देना
$ A \in \mathcal M_2(\R)$, $\tau = \operatorname{tr}A $, $\delta
= \det A $, $\Delta = \tau^2 - 4\delta $.
\begin{enumerate}
  \item दिखाओ कि eigenvalues का $ A $ हैं
        $\frac{\tau\pm\sqrt\Delta}{2}$ और वर्गीकृत करें: दो वास्तविक
        eigenvalues विपरीत चिह्नों का iff $\delta < 0$; असली
        eigenvalues उसी चिह्न का iff $\delta > 0$, $\Delta
        \geq 0$ (\omterm{thm:b2:structures:signature}{हस्ताक्षर} करना $\tau $); अवास्तविक संयुग्मी युग्म यदि
        $\Delta < 0$ (वास्तविक भाग $\frac\tau2$).
  \item (काठी, $\delta < 0$) साथ eigenvalues $\mu < 0 <
        \lambda $ और eigenvectors $ v_\pm $, सामान्य लिखें
        समाधान और प्रक्षेप पथ का वर्णन करें: दो स्थिर और
        दो अस्थिर किरणें, अन्य सभी कक्षाएँ स्पर्शोन्मुख हैं
        दोनों. लेकिन कोई समाधान क्यों नहीं हो सकता $0$ सब पर बंधे रहो
        का $\R $?
  \item (नोड्स, $\delta > 0$, $\Delta > 0$) के लिए $\mu < \lambda
        < 0$: दिखाएँ कि प्रत्येक गैर-शून्य समाधान की प्रवृत्ति होती है $0$ और
        कि तीव्र अक्ष पर स्थित कक्षाओं को छोड़कर सभी कक्षाएँ आ जाती हैं
        \emph{धीमी ईजेनडायरेक्शन के स्पर्शरेखा}
        \emph{(तुलना करना $\eu^{\mu t}$ और $\eu^{\lambda t}$)}.
  \item (सर्पिल और केंद्र, $\Delta < 0$) लिखना
        eigenvalues $\alpha \pm \iu\beta $, उपयोग
        \cref{exo:b2:diffeq:5} (आधार में वास्तविक परिवर्तन के बाद,
        उस व्यापकता में स्वीकार किया गया या प्रणालियों के लिए सिद्ध किया गया
        भाग II का, जिसका वर्णन करने के लिए नीचे उपयोग किया गया है)।
        कक्षाएँ: सर्पिल \omterm{def:b2:integration:improper}{अभिसरण} के लिए $\alpha =
        \frac\tau2 < 0$, के लिए विचलन $\tau > 0$, बंद
        वक्र (केंद्र) के लिए $\tau = 0$.
  \item (सीमा मामले) एक डबल के लिए \omterm{def:b2:reduction:eigen}{eigenvalue} ($\Delta =
        0$): दिखाओ $\eu^{tA} = \eu^{\lambda t}(I + tN)$ साथ $ N
        = A - \lambda I $ निलपोटेंट, और तारे को अलग करें
        ($ N = 0$) अनुचित नोड से ($ N \neq 0$).
        भाग I को ट्रेस-निर्धारक चित्र में सारांशित करें
        इस अध्याय का चित्र.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The damped oscillator.} अब $ F =
0$: $ x'' + 2\zeta\omega x' + \omega^2x = 0$, यानी\ $ X' = AX $
साथ $ A = \begin{pmatrix} 0 & 1\\ -\omega^2 &
-2\zeta\omega\end{pmatrix}$.
\begin{enumerate}[resume]
  \item गणना करना $\tau, \delta, \Delta $ और तीनों को रखें
        ट्रेस-निर्धारक विमान में शासन:
        \emph{कम नमीयुक्त} $0 < \zeta < 1$ (स्थिर सर्पिल),
        \emph{गंभीर रूप से नम} $\zeta = 1$ (दोहरा
        \omterm{def:b2:reduction:eigen}{eigenvalue}), \emph{अत्यधिक नमीयुक्त} $\zeta > 1$ (स्थिर
        नोड); $\zeta = 0$ केंद्र है.
  \item तीन व्यवस्थाओं को स्पष्ट रूप से हल करें:
        \[
        \zeta < 1:\ \eu^{-\zeta\omega t}\bigl(a\cos\omega_d t
        + b\sin\omega_dt\bigr),\ \omega_d =
        \omega\sqrt{1-\zeta^2};
        \qquad
        \zeta = 1:\ (a + bt)\,\eu^{-\omega t};
        \]
        $\zeta > 1$: दो वास्तविक घातांक। परिभाषित करें
        छद्म काल $\frac{2\pi}{\omega_d}$ और दिखाओ
        की क्रमिक अधिकतमा का अनुपात $\abs x $ है
        स्थिर $\eu^{-2\pi\zeta/\sqrt{1-\zeta^2}}$ (द
        लघुगणकीय कमी)।
  \item (दरवाजा-बंद करने का सिद्धांत) के लिए $\zeta \geq 1$ द
        क्षय दर सबसे धीमी गति से नियंत्रित होती है \omterm{def:b2:reduction:eigen}{eigenvalue}
        $\lambda_{\mathrm{slow}} = -\omega\bigl(\zeta -
        \sqrt{\zeta^2-1}\bigr)$. बताते हैं कि
        $\abs{\lambda_{\mathrm{slow}}} =
        \frac{\omega}{\zeta + \sqrt{\zeta^2 - 1}}$ एक है
        \emph{घटते} का कार्य $\zeta \geq 1$:
        गंभीर अवमंदन $\zeta = 1$ सबसे तेज देता है
        आराम की ओर गैर-दोलनात्मक वापसी।
  \item (ऊर्जा) चलो $ E(t) = \frac12x'^2 +
        \frac12\omega^2x^2$. सिद्ध करना $ E' = -2\zeta\omega\,x'^2
        \leq 0$, और उसके लिए निष्कर्ष निकालें $\zeta > 0$ समीकरण
        इसका कोई शून्येतर आवर्त समाधान नहीं है \emph{(एक अवधि होगी
        बल $ E $ स्थिर, इसलिए $ x' \equiv 0$)}.
  \item दो वाक्यों में स्पष्ट कीजिए कि केन्द्र क्यों है? $\zeta = 0$
        है \emph{संरचनात्मक रूप से नाजुक}: कोई $\zeta > 0$,
        चाहे कितना ही छोटा क्यों न हो, आवधिकता को नष्ट कर देता है --- और कहाँ
        जो ट्रेस-निर्धारक तल (द) में दिखता है
        केंद्र रेखा का आंतरिक भाग खाली है)।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Forced oscillations and
\omterm{pb:b2:diffeq:1}{गूंज}.} अब $ F(t) = F\cos(\gamma t)$ साथ $ F, \gamma > 0$.
\begin{enumerate}[resume]
  \item ($\zeta > 0$: स्थिर अवस्था) खोजें $ x_p =
        \Re\bigl(z\,\eu^{\iu\gamma t}\bigr)$: दिखाओ
        \[
        z = \frac{F}{\omega^2 - \gamma^2 +
        2\iu\zeta\omega\gamma},
        \qquad
        A(\gamma) := \abs z = \frac{F}{\sqrt{(\omega^2 -
        \gamma^2)^2 + 4\zeta^2\omega^2\gamma^2}} ,
        \]
        और लिखा $ x_p = A(\gamma)\cos(\gamma t - \varphi)$
        साथ $\tan\varphi =
        \frac{2\zeta\omega\gamma}{\omega^2-\gamma^2}$.
  \item बताते हैं कि \emph{प्रत्येक} समाधान है $ x_p $ प्लस ए
        भाग II से क्षणिक, जिसकी प्रवृत्ति होती है $0$: जो भी हो
        प्रारंभिक डेटा, सिस्टम स्थिर पर लॉक हो जाता है
        अवस्था --- आयाम $ A(\gamma)$, चरण \omterm{def:b2:diffcalc:differential}{अंतर}ाल
        $\varphi $.
  \item (द \omterm{pb:b2:diffeq:1}{गूंज} वक्र) अधिकतम करें $ A $: बताते हैं कि
        $ A(\gamma)$ आंतरिक अधिकतम iff है $\zeta <
        \frac{1}{\sqrt2}$, पर
        \[
        \gamma_* = \omega\sqrt{1 - 2\zeta^2},
        \qquad
        A(\gamma_*) =
        \frac{F}{2\zeta\omega^2\sqrt{1-\zeta^2}} ,
        \]
        और वह छोटे के लिए $\zeta $ शिखर को बढ़ाता है
        स्थैतिक प्रतिक्रिया $ A(0) = \frac F{\omega^2}$ द्वारा
        कारक $\approx \frac{1}{2\zeta}$.
  \item ($\zeta = 0$, बंद \omterm{pb:b2:diffeq:1}{गूंज}) के लिए $\gamma \neq \omega $,
        दिखाएँ कि समाधान के साथ $ x(0) = x'(0) = 0$ है
        \[
        x(t) = \frac{F}{\omega^2 -
        \gamma^2}\bigl(\cos\gamma t - \cos\omega t\bigr)
        = \frac{2F}{\omega^2-\gamma^2}
        \sin\frac{(\omega-\gamma)t}{2}
        \sin\frac{(\omega+\gamma)t}{2} :
        \]
        घिरा हुआ, के साथ \emph{धड़कता है} --- एक तेज़ दोलन
        एक धीमे लिफ़ाफ़े के नीचे --- कब $\gamma $ के करीब है
        $\omega $.
  \item ($\zeta = 0$, \omterm{pb:b2:diffeq:1}{गूंज}) के लिए $\gamma = \omega $, बताते हैं कि $ x_p(t) = \frac{F}{2\omega}\,t\sin(\omega t)$ है
        एक समाधान, और इसे प्रश्न की सीमा के रूप में पुनर्प्राप्त करें
        14 के रूप में $\gamma \to \omega $: आयाम बढ़ता है
        रैखिक रूप से हमेशा के लिए--- \omterm{pb:b2:diffeq:1}{गूंज} तबाही.
  \item (फूरियर लिंक) एक सामान्य आवधिक बल विघटित होता है
        हार्मोनिक्स में (फूरियर अध्याय); रैखिकता द्वारा
        स्थिर अवस्था हार्मोनिक प्रतिक्रियाओं का योग है। के लिए
        आवृत्ति का एक अवमंदित थरथरानवाला $\omega = 3$
        की वर्ग तरंग द्वारा मजबूर
        \cref{exo:b2:fourier:1}-प्रकार (हार्मोनिक्स \omterm{def:b2:series:def}{बिल्कुल} विषम
        पूर्णांक), कौन सा हार्मोनिक प्रतिध्वनित होता है? एक वाक्य पर
        इंजीनियर वर्ग तरंगों से क्यों डरते हैं?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Sturm's theorems.} विचार करना $ y''
+ q(t)\,y = 0$ एक \omterm{def:b2:diffcalc:differential}{अंतर}ाल पर $ I $, $ q $ \omterm{def:b2:metric:continuity}{निरंतर}. (कोई भी
समीकरण $ y'' + ay' + by = 0$ द्वारा इस सामान्य रूप में कम कर देता है
प्रतिस्थापन $ y = u\exp\bigl(-\frac12\int a\bigr)$;
प्रश्न 21 क्रियान्वित चाल का एक प्रकार दिखाता है।)
\begin{enumerate}[resume]
  \item दो समाधान के लिए $ y_1, y_2$, दिखाओ कि \omterm{def:b2:diffeq:wronskian}{रोन्स्कियन}
        $ W = y_1y_2' - y_1'y_2$ है \emph{स्थिर}, शून्य आईएफएफ
        समाधान आनुपातिक हैं; और वह एक शून्येतर
        समाधान ही है \emph{सरल, पृथक} शून्य.
  \item (स्टर्म अलगाव) चलो $ y_1, y_2$ स्वतंत्र रहो
        समाधान और $ a < b $ के लगातार दो शून्य $ y_1$.
        साबित करें कि $ y_2$ ठीक एक बार गायब हो जाता है
        $\intoo{a}{b}$ \emph{(स्थिरांक का मूल्यांकन करें $ W $ पर $ a $
        और $ b $: $ W = y_1'y_2$ वहाँ और $ y_1'(a)$,
        $ y_1'(b)$ विपरीत संकेत हैं)}: स्वतंत्र के शून्य
        समाधान जिल्द.
  \item (स्टर्म तुलना) चलो $ q_1 \leq q_2$ पर $ I $, $ y \neq
        0$ साथ $ y'' + q_1y = 0$, $ z \neq 0$ साथ $ z'' + q_2z
        = 0$, और $ a < b $ के लगातार शून्य $ y $. बताते हैं कि
        $ z $ में गायब हो जाता है $\intcc{a}{b}$ --- सख्ती से अंदर अगर
        $ q_1 < q_2$ कहीं पर $\intoo ab $ \emph{(अगर $ z \neq
        0$ पर $\intoo ab $, अध्ययन $(yz' - y'z)' = (q_1 -
        q_2)yz $ के लिए निश्चित चिह्नों के साथ $ y, z $ और तुलना करें
        सीमा मान)}.
  \item निष्कर्ष निकालें \emph{\omterm{def:b2:diffcalc:differential}{अंतर} सीमा}: अगर $0 < m^2 \leq
        q(t) \leq M^2$ पर $ I $, फिर कोई दो लगातार शून्य
        $ a < b $ के एक शून्येतर समाधान का $ y'' + qy = 0$
        संतुष्ट
        \[
        \frac{\pi}{M} \;\leq\; b - a \;\leq\; \frac{\pi}{m}
        \]
        \emph{(साथ तुलना करें $ u'' + M^2u = 0$ और $ u'' + m^2u
        = 0$, जिनके शून्य का \omterm{def:b2:diffcalc:differential}{अंतर} है $\frac\pi M $ और
        $\frac\pi m $)}. हार्मोनिक ऑसिलेटर की जाँच करें।
  \item परिवर्तन $ ty'' + 2y' + ty = 0$
        (\cref{exo:b2:diffeq:7}) द्वारा $ u = ty $ में $ u'' + u =
        0$, इसके समाधान पुनर्प्राप्त करें $\frac{\sin t}t $,
        $\frac{\cos t}{t}$ तुरंत, और निष्कर्ष निकालें कि
        प्रत्येक गैर-शून्य समाधान के शून्य \omterm{def:b2:series:def}{बिल्कुल} ठीक दूरी पर हैं
        $\pi $: स्टर्म का विश्व दृष्टिकोण---शून्य किसके द्वारा नियंत्रित होते हैं
        गुणांक $ q $, सूत्र या नहीं।
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Duhamel and the boundedness
frontier.}
\begin{enumerate}[resume]
  \item (ऑसिलेटर के लिए डुहामेल) इसके लिए दिखाएँ
        \omterm{def:b2:metric:continuity}{निरंतर} $ F $, का समाधान $ x'' + \omega^2x =
        F(t)$ साथ $ x(0) = x'(0) = 0$ है
        \[
        x(t) = \frac1\omega\int_0^t\sin\bigl(\omega(t -
        s)\bigr)F(s)\,\dd s ,
        \]
        और प्रश्न 15 का गुंजयमान समाधान पुनः प्राप्त करें
        इसके साथ से $ F(s) = F\cos(\omega s)$
        \emph{(उत्पाद-से-योग)}.
  \item ($\zeta > 0$: परिबद्ध इनपुट, परिबद्ध आउटपुट) दिखाएँ
        उस के लिए $\zeta > 0$ और \emph{कोई} घिरे \omterm{def:b2:metric:continuity}{निरंतर}
        $ F $, अवमंदित समीकरण का प्रत्येक समाधान परिबद्ध है
        पर $\intco{0}{\infty}$ \emph{(स्थिरांक का परिवर्तन
        साथ ही घातांकीय क्षय $\vertiii{\eu^{tA}} \leq
        C\eu^{-\alpha t}$ का \cref{exo:b2:diffeq:8})}.
  \item ($\zeta = 0$) दिखाएँ कि बिना किसी अवमंदन के, परिबद्ध
        आवधिक बल सभी समाधानों को सीमित रखता है
        \emph{के अलावा} बिलकुल पर \omterm{pb:b2:diffeq:1}{गूंज} ($\gamma =
        \omega $, प्रश्न 15 बनाम प्रश्न 14): अवमंदन है
        जो सीमाबद्ध सीमा को एकरूपता में बदल देता है
        स्थिरता.
  \item संश्लेषण। प्रत्येक एक वाक्य में: (i) कैसे
        ट्रेस--निर्धारक विमान भाग I--II और को व्यवस्थित करता है
        जहां जबरदस्ती (भाग III) इसे छोड़ देती है; (ii) द
        का भौतिक अर्थ $\gamma_*$, $ A(\gamma_*)$ और
        कारक $\frac1{2\zeta}$; (iii) स्टर्म क्या है
        प्रमेय कहते हैं कि स्पष्ट सूत्र स्पष्ट नहीं कर सकते; (iv)
        इस समस्या के बाकी कौन से दो परिणाम हैं
        पुस्तक चुपचाप पुन: उपयोग होगी (लिउविल-स्थिरांक)।
        ronskians; बाउंडेड-इनपुट स्थिरता)।
\end{enumerate}
