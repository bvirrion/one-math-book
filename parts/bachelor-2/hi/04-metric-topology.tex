\chapter{मीट्रिक रिक्त स्थान की टोपोलॉजी}\label{ch:b2:metric}

वास्तविक रेखा की टोपोलॉजी (वर्ष 1 खंड) लगभग सामान्यीकृत होती है
एक भी शब्द बदले बिना, दूरी से \omterm{def:b2:structures:countable}{सुसज्जित} किसी भी सेट पर। द
लाभ बहुत बड़ा है: कार्यों का क्रम, आव्यूह, वक्र --- सभी
के बिंदु बन जाते हैं मीट्रिक रिक्त स्थान, और यहां तीन स्तंभ सिद्ध हुए
--- पूर्णता बानाच निश्चित बिंदु प्रमेय के साथ, सघनता,
संयुक्तता --- उन पर समान रूप से लागू करें. यह अध्याय है
संपूर्ण विश्लेषण की रीढ़ पुस्तक का आधा भाग।

\section{मीट्रिक स्थान}

\begin{definition}\label{def:b2:metric:def}
ए \emph{मीट्रिक स्थान}\index{मीट्रिक स्थान} एक सेट है $X $ एक मानचित्र के साथ $ d
\colon X \times X \to \R_+$ ऐसा कि, सभी के लिए $ x, y, z $:
\[
d(x,y) = 0 \iff x = y,
\qquad
d(x,y) = d(y,x),
\qquad
d(x,z) \leq d(x,y) + d(y,z).
\]
बॉल्स: $ B(a, r) = \{x : d(a,x) < r\}$ (\omterm{def:b2:metric:topology}{खुला}), $\overline B(a,r) =
\{x : d(a,x) \leq r\}$ (बंद किया हुआ)। एक उपसमुच्चय $ A \subseteq X $ एक बन जाता है
प्रेरित दूरी के साथ मीट्रिक स्थान।
\end{definition}

\begin{example}\label{ex:b2:metric:examples}
$\R $ साथ $\abs{x - y}$; $\R^n $ किसी के साथ
\[
d_1(x,y) = \sum_i \abs{x_i - y_i},
\quad
d_2(x,y) = \Bigl(\sum_i (x_i - y_i)^2\Bigr)^{1/2},
\quad
d_\infty(x,y) = \max_i \abs{x_i - y_i};
\]
सेट $ C(\intcc{a}{b})$ का \omterm{def:b2:metric:continuity}{निरंतर} के साथ कार्य करता है
\emph{सुपर दूरी} $ d_\infty(f, g) = \sup_{\intcc{a}{b}} \abs{f -
g}$ (परिमित: $ f - g $ घिरा है); के साथ कोई भी सेट \emph{अलग}
दूरी ($ d(x,y) = 1$ के लिए $ x \neq y $). मानदंडों से आ रही दूरियां
का विषय हैं \cref{ch:b2:nvs}.
\end{example}

\begin{definition}[मीट्रिक स्थान की टोपोलॉजी]\label{def:b2:metric:topology}
$ U \subseteq X $ है \emph{खुला}\index{खुला}
जब प्रत्येक बिंदु $ U $ में निहित एक गेंद का केंद्र है $ U $;
$ F $ है \emph{बंद किया हुआ} जब इसका कॉम्प्लीमेंट खुला हो. पड़ोस,
आंतरिक, समापन, घनत्व, सीमा को बिल्कुल उसी तरह परिभाषित किया गया है जैसे कि
वास्तविक रेखा (वर्ष 1 खंड), अंतरालों की जगह लेने वाली गेंदों के साथ, और
कथन वहाँ सिद्ध हुए --- खुले समुच्चयों के मिलन/प्रतिच्छेदन,
आंतरिक और समापन की विशेषताएं, सबसे छोटा समापन
बंद सुपरसेट---समान प्रमाणों के साथ जारी रखें। खुली गेंदें हैं
खुली, बंद गेंदें बंद हैं (त्रिकोण असमानता)।
\end{definition}

\begin{example}[एक सेट पर आंतरिक, समापन, सीमा]\label{ex:b2:metric:intclosure}
में $\R $, होने देना $ A = \intoc{0}{1} \cup \{2\}$. आंतरिक भाग:
$\intoo{0}{1}$ --- किसी के आसपास $ x \in \intoo01$ एक छोटी सी गेंद
में रहता है $ A $; आस-पास $1$, हर गेंद $\intoo{1-r}{1+r}$ लीक
से बाहर $ A $ दाईं ओर, इसलिए $1$ आंतरिक नहीं है; और
पृथक $2$ आंतरिक भी नहीं है. समापन:
$\intcc{0}{1} \cup \{2\}$ (मुद्दा $0$ की एक सीमा है $ A $,
और कुछ नहीं जोड़ा गया है)। सीमा (आंतरिक भाग को छोड़कर बंद करना):
$\{0, 1, 2\}$. याद रखने लायक विषमताओं पर ध्यान दें: a
समापनबिंदु आंतरिक न होते हुए भी किसी समुच्चय से संबंधित हो सकता है ($1$), कर सकते हैं
बिना किसी संबंध के अनुयायी बनें ($0$), और एक पृथक बिंदु है
इसकी अपनी सीमा ($2$). वही बहीखाता शब्दशः चलता है
कोई भी \omterm{def:b2:metric:def}{मीट्रिक स्थान}, अंतराल के स्थान पर गेंदों के साथ।
\end{example}

\begin{definition}[सीमा, निरंतरता]\label{def:b2:metric:continuity}
$ x_n \to x $ में $ X $ कब $ d(x_n, x) \to 0$. नक्षा $ f \colon X \to Y $
बीच में मीट्रिक रिक्त स्थान है \emph{निरंतर}\index{निरंतर} पर $ a $ कब
\[
\forall \varepsilon > 0,\ \exists\delta > 0,\quad
d_X(x, a) \leq \delta \implies d_Y\bigl(f(x), f(a)\bigr) \leq
\varepsilon ;
\]
समकक्ष रूप से (समान प्रमाण)। $\R $), $ f(x_n) \to f(a)$ प्रत्येक के लिए
क्रम $ x_n \to a $. $ f $ है \emph{लिप्सचिट्ज़}\index{लिप्सचिट्ज़} स्थिरांक के साथ $ k $ कब
$ d_Y(f(x), f(y)) \leq k\, d_X(x, y)$ सदैव --- फिर समान रूप से
निरंतर, इसलिए निरंतर.
\end{definition}

\begin{theorem}[निरंतरता का वैश्विक लक्षण वर्णन]\label{thm:b2:metric:globalcontinuity}
$ f \colon X \to Y $ है \omterm{def:b2:metric:continuity}{निरंतर} (प्रत्येक बिंदु पर) यदि और केवल यदि
प्रत्येक की पूर्वछवि \omterm{def:b2:metric:topology}{खुला} सेट है \omterm{def:b2:metric:topology}{खुला} --- यदि प्रत्येक की पूर्वछवि
बंद सेट बंद है.
\end{theorem}

\begin{proof}
($\Rightarrow $) होने देना $ V \subseteq Y $ होना \omterm{def:b2:metric:topology}{खुला} और $ a \in f^{-1}(V)$:
कुछ गेंद $ B(f(a), \varepsilon) \subseteq V $; निरंतरता पर $ a $
प्रदान $\delta $ साथ $ f(B(a, \delta)) \subseteq B(f(a),
\varepsilon)$, इसलिए $ B(a, \delta) \subseteq f^{-1}(V)$.

($\Leftarrow $) दिया गया $ a $ और $\varepsilon $: $ f^{-1}\bigl(B(f(a),
\varepsilon)\bigr)$ है \omterm{def:b2:metric:topology}{खुला} और शामिल है $ a $, इसलिए इसमें एक गेंद होती है
$ B(a, \delta)$: यही परिभाषा है निरंतरता पर $ a $. बंद
सेट: पूरक (\cref{prop:b2:structures:images}).
\end{proof}

\section{पूर्ण रिक्त स्थान}

\begin{definition}\label{def:b2:metric:complete}
एक क्रम $(x_n)$ है \emph{कॉची} कब $\sup_{p, q \geq N} d(x_p,
x_q) \to 0$ जैसा $ N \to \infty $. ए \omterm{def:b2:metric:def}{मीट्रिक स्थान} है
\emph{पूरा}\index{पूरा} जब हर कॉची
अनुक्रम एकत्रित होता है। संमिलित $\Rightarrow $ कॉची हमेशा; बंद
पूर्ण स्थानों के उपसमुच्चय पूर्ण होते हैं, और किसी के पूर्ण उपसमुच्चय
स्थान बंद हैं (समान प्रमाण चालू हैं)। $\R $: वर्ष 1 खंड).
\end{definition}

\begin{example}[बिना किसी सीमा के कॉची]\label{ex:b2:metric:cauchynolimit}
में $ X = \Q $ सामान्य दूरी के साथ, दशमलव काट-छाँट
$\sqrt2$,
\[
x_0 = 1,\quad x_1 = 1.4,\quad x_2 = 1.41,\quad x_3 = 1.414,
\quad\dots
\]
संतुष्ट $\abs{x_p - x_q} \leq 10^{-\min(p,q)}$: कॉची इन $\Q $.
में एक सीमा $\Q $ में भी सीमा होगी $\R $, अर्थात्
$\sqrt2 \notin \Q $: में कोई सीमा मौजूद नहीं है $ X $. अपूर्णता है
ऐसी ``प्रेत सीमाओं'' की उपस्थिति; पूर्णता का $\R $
प्रत्येक को सटीक रूप से देने के लिए वर्ष 1 खंड में इंजीनियर किया गया था
कॉची अनुक्रम एक घर.
\end{example}

\begin{theorem}\label{thm:b2:metric:rncomplete}
$\R^n $ (तीनों दूरियों में से कोई भी
\cref{ex:b2:metric:examples}) और $\bigl(C(\intcc{a}{b}),
d_\infty\bigr)$ हैं \omterm{def:b2:metric:complete}{पूरा}.
\end{theorem}

\begin{proof}
$\R^n $: प्रत्येक निर्देशांक (प्रत्येक) में एक कॉची अनुक्रम कॉची है
$\abs{x_i - y_i} \leq d(x,y)$ सभी तीन दूरियों के लिए), इसलिए प्रत्येक
समन्वय अभिसरण (पूर्णता का $\R $, वर्ष 1 खंड), और
समन्वयवार अभिसरण का तात्पर्य अभिसरण से है $ d_\infty $
(अंततः कई निर्देशांक), इसलिए तीनों के लिए (तीन)।
स्थिर कारकों के भीतर दूरियाँ एक दूसरे पर हावी होती हैं: $ d_\infty
\leq d_2 \leq d_1 \leq n\,d_\infty $).

$ C(\intcc{a}{b})$: होने देना $(f_n)$ होना $ d_\infty $-कौची. प्रत्येक के लिए $ x $,
$(f_n(x))$ कॉची में है $\R $ ($\abs{f_p(x) - f_q(x)} \leq
d_\infty(f_p, f_q)$): कुछ में परिवर्तित हो जाता है $ f(x)$. के लिए गुजर रहा है
में सीमा $\abs{f_p(x) - f_q(x)} \leq \varepsilon $ (के लिए मान्य $ p, q
\geq N_\varepsilon $, सभी $ x $) जैसा $ q \to \infty $: $\abs{f_p(x) -
f(x)} \leq \varepsilon $ सभी के लिए $ x $, यानी\ $ d_\infty(f_p, f) \leq
\varepsilon $: एकसमान अभिसरण. हद है \omterm{def:b2:metric:continuity}{निरंतर}: दिया गया
$\varepsilon $, चुनना $ p $ साथ $\sup\abs{f_p - f} \leq \varepsilon $,
फिर उपयोग करें निरंतरता का $ f_p $ पर $ a $ और तीन-अवधि का विभाजन
\[
\abs{f(x) - f(a)} \leq \abs{f(x) - f_p(x)} + \abs{f_p(x) - f_p(a)}
+ \abs{f_p(a) - f(a)} \leq 3\varepsilon
\]
के लिए $ x $ के करीब $ a $. (यह ``$3\varepsilon $ तर्क'' के रूप में लौटता है
की एकसमान-सीमा प्रमेय \cref{ch:b2:funcseq}.)
\end{proof}

\begin{example}[खुले और बंद सेटों को निरंतरता द्वारा पहचाना जाता है]\label{ex:b2:metric:recognize}
वैश्विक लक्षण वर्णन
(\cref{thm:b2:metric:globalcontinuity}) के लिए रोजमर्रा का उपकरण है
टोपोलॉजिकल बहीखाता। में $\R^2$: सेट $\{(x, y) : x^2 + y^2
< 1,\ y > x^3\}$ है \omterm{def:b2:metric:topology}{खुला} --- यह है $ g^{-1}(\intoo{-\infty}{1})
\cap h^{-1}(\intoo{0}{+\infty})$ के लिए \omterm{def:b2:metric:continuity}{निरंतर} $ g(x,y) =
x^2 + y^2$ और $ h(x, y) = y - x^3$, दो का प्रतिच्छेदन \omterm{def:b2:metric:topology}{खुला}
पूर्वछवियाँ। में $\bigl(C(\intcc01), d_\infty\bigr)$: का सेट
के साथ कार्य करता है $ f(0) = f(1)$ और $\int_0^1 f = 0$ बंद है ---
की पूर्वछवि $\{(0,0)\}$ नीचे \omterm{def:b2:metric:continuity}{निरंतर} नक्शा $ f \mapsto
\bigl(f(0) - f(1),\ \int_0^1 f\bigr)$ में $\R^2$ (प्रत्येक
समन्वय है $1$-लिप्सचिट्ज़, जैसा कि \cref{exo:b2:metric:3}). द
विधि कभी चित्र नहीं बनाती: प्रदर्शन a \omterm{def:b2:metric:continuity}{निरंतर} मानचित्र, पढ़ें
एक पूर्वछवि के रूप में सेट करें, प्रमेय उद्धृत करें।
\end{example}

\begin{example}[अपरिमित अनेक स्थितियों द्वारा परिभाषित एक बंद समुच्चय]\label{ex:b2:metric:lipclosed}
में $\bigl(C(\intcc01), d_\infty\bigr)$, सेट
\[
L = \{f : \abs{f(x) - f(y)} \leq \abs{x - y}
\ \text{for all } x, y\}
\]
का $1$-लिप्सचिट्ज़ फ़ंक्शंस बंद है, हालाँकि इसे काट दिया गया है
बेशुमार कई शर्तें: प्रत्येक निश्चित जोड़ी के लिए $(x, y)$, वो नक्शा $ f \mapsto \abs{f(x) - f(y)} - \abs{x - y}$ है \omterm{def:b2:metric:continuity}{निरंतर}
(मूल्यांकन हैं $1$-लिप्सचिट्ज़), इसलिए प्रत्येक एक शर्त
एक बंद सेट को परिभाषित करता है, और $ L $ है \emph{चौराहा} का
यह परिवार --- बंद सेटों का एक मनमाना प्रतिच्छेदन है
बंद. वही टेम्प्लेट मोनोटोन के लिए क्लोजनेस को प्रमाणित करता है
फ़ंक्शंस, उत्तल फ़ंक्शंस, एक निश्चित से बंधे फ़ंक्शंस $ g $:
एक परिवार के रूप में व्यक्त की जाने वाली प्रत्येक संपत्ति को समान सीमाएँ विरासत में मिलती हैं
बंद बिंदुवार बाधाओं का। एक समान सीमाएँ क्या करती हैं
\emph{नहीं} स्वचालित रूप से विरासत में मिलता है --- भिन्नता, के लिए
एक --- बिल्कुल वही है \cref{ch:b2:funcseq} के लिए श्रम करना होगा.
\end{example}

\begin{theorem}[बानाच निश्चित बिंदु प्रमेय]\label{thm:b2:metric:banach}
होने देना $ X $ एक शून्य होना पूर्ण \omterm{def:b2:metric:def}{मीट्रिक स्थान} और $ f \colon X \to X $ ए
\emph{सिकुड़न}: \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} स्थिरांक के साथ $ k < 1$. तब $ f $ एक है
अद्वितीय निश्चित बिंदु $\ell $, और हर कक्षा $ x_{n+1} = f(x_n)$
में एकत्रित हो जाता है $\ell $, साथ\index{बानाच निश्चित बिंदु प्रमेय}
\[
d(x_n, \ell) \leq \frac{k^n}{1 - k}\, d(x_1, x_0) .
\]
\end{theorem}

\begin{proof}
विशिष्टता: दो निश्चित बिंदु दूरी पर हैं $\leq k $ कई बार ही.
अस्तित्व: $ d(x_{n+1}, x_n) \leq k^n d(x_1, x_0)$ प्रेरण द्वारा, इसलिए
के लिए $ q > p $,
\[
d(x_q, x_p) \leq \sum_{j=p}^{q-1} d(x_{j+1}, x_j)
\leq d(x_1, x_0) \sum_{j \geq p} k^j
= \frac{k^p}{1-k}\, d(x_1, x_0) \xrightarrow[p\to\infty]{} 0 :
\]
कॉची, इसलिए कुछ के लिए अभिसरण $\ell $; निरंतरता का $ f $ गुजरता है
$ x_{n+1} = f(x_n)$ सीमा तक: $\ell = f(\ell)$. त्रुटि बाध्य है
के साथ प्रदर्शित अनुमान है $ q \to \infty $.
\end{proof}

\begin{example}[एक अभिन्न समीकरण]\label{ex:b2:metric:integralequation}
पर $ X = C(\intcc{0}{1})$ (\omterm{def:b2:metric:complete}{पूरा},
\cref{thm:b2:metric:rncomplete}), विचार करना $ T(f)(x) = 1 + \frac12
\int_0^x f(t)\,\dd t $. के लिए $ f, g \in X $:
\[
\abs{T(f)(x) - T(g)(x)} \leq \frac12 \int_0^x \abs{f - g} \leq
\frac12\, d_\infty(f, g),
\]
इसलिए $ T $ एक है $\frac12$-संकुचन: इसमें एक अद्वितीयता है \omterm{def:b2:metric:continuity}{निरंतर} निश्चित
बिंदु---का समाधान $ f' = \frac f2$, $ f(0) = 1$, अर्थात्
$\eu^{x/2}$. यह योजना औद्योगीकृत हो जाती है
कॉची--लिप्सचिट्ज़ प्रमेय \cref{ch:b2:diffeq}.
\end{example}

\begin{example}[एक संख्यात्मक निश्चित बिंदु: $ x = \cos x $]\label{ex:b2:metric:cosfix}
पर \omterm{def:b2:metric:complete}{पूरा} $ X = \intcc{0}{1}$, वो नक्शा $ f = \cos $ भेजता है $ X $
में $\intcc{\cos 1}{1} \subseteq X $ और एक संकुचन है: द्वारा
माध्य मूल्य असमानता,
\[
\abs{\cos x - \cos y} \leq \bigl(\sup_{\intcc01}\abs{\sin}\bigr)
\abs{x - y} = (\sin 1)\abs{x - y},
\qquad \sin 1 \approx 0.841 < 1 .
\]
बनच: का एक अनोखा समाधान $ x = \cos x $ में $\intcc01$ (इसलिए में
$\R $: कोई भी वास्तविक निश्चित बिंदु निहित है $\intcc{-1}{1}$, में फिर
$\intcc{\cos 1}{1}$ एक आवेदन के बाद), और पुनरावृत्ति
$ x_{n+1} = \cos x_n $ किसी भी शुरुआत से इसमें जुट जाता है: $ x_\infty
\approx 0.739085$, हथौड़े से ठोकने पर प्राप्त प्रसिद्ध संख्या
कैलकुलेटर की कोसाइन कुंजी। त्रुटि बाध्य भविष्यवाणी करती है
$(\sin1)^n/(1 - \sin1)$ क्षय---लगभग एक अंक प्रति $13$
दबाता है; इस अध्याय के सप्ताहांत का पिछला भाग
समस्या (प्रश्न 14) प्रत्येक चरण को तुरंत प्रमाणित करती है।
\end{example}

\section{सघनता}

\begin{definition}\label{def:b2:metric:compact}
ए \omterm{def:b2:metric:def}{मीट्रिक स्थान} $ X $ है \emph{कॉम्पैक्ट}\index{कॉम्पैक्ट}
जब हर क्रम में $ X $ एक परिणामी अभिसरण है \emph{में}
$ X $ (बोल्ज़ानो--वीयरस्ट्रैस संपत्ति)। एक उपसमुच्चय संहत होता है जब यह
प्रेरित दूरी के साथ भी ऐसा ही है।
\end{definition}

\begin{theorem}[प्रथम गुण]\label{thm:b2:metric:compactprops}
\begin{enumerate}
  \item ए \omterm{def:b2:metric:compact}{कॉम्पैक्ट} उपसमुच्चय बंद और परिबद्ध है; ए का एक बंद उपसमुच्चय
        \omterm{def:b2:metric:compact}{कॉम्पैक्ट} अंतरिक्ष है \omterm{def:b2:metric:compact}{कॉम्पैक्ट}.
  \item में $\R^n $, व्युत्क्रम धारण करता है: \omterm{def:b2:metric:compact}{कॉम्पैक्ट} $\iff $ बंद और
        घिरा हुआ.
  \item ए \omterm{def:b2:metric:continuity}{निरंतर} एक की छवि \omterm{def:b2:metric:compact}{कॉम्पैक्ट} अंतरिक्ष है \omterm{def:b2:metric:compact}{कॉम्पैक्ट}; ए
        \omterm{def:b2:metric:continuity}{निरंतर} एक गैर-रिक्त पर वास्तविक कार्य \omterm{def:b2:metric:compact}{कॉम्पैक्ट} अंतरिक्ष है
        बंधा हुआ और अपनी सीमा को प्राप्त करता है।
  \item (हेन) ए \omterm{def:b2:metric:continuity}{निरंतर} ए पर नक्शा \omterm{def:b2:metric:compact}{कॉम्पैक्ट} स्थान समान रूप से है
        \omterm{def:b2:metric:continuity}{निरंतर}.
  \item उत्पाद: यदि $ X, Y $ हैं \omterm{def:b2:metric:compact}{कॉम्पैक्ट}, ऐसा ही है $ X \times Y $ (साथ
        $ d\bigl((x,y),(x',y')\bigr) = d(x,x') + d(y,y')$).
\end{enumerate}
\end{theorem}

\begin{proof}
(1) लाइन पर समान तर्क (वर्ष 1 खंड): एक एस्केपिंग-टू-%
अनंत या अभिसरण-बाहर अनुक्रम का कोई अनुवर्ती नहीं है
अंदर एकत्रित होना; दूसरे दावे के लिए, परिवेश में निकालें
\omterm{def:b2:metric:compact}{कॉम्पैक्ट} और बंदपन का उपयोग करें.

(2) बाउंडेड सीक्वेंस $\R^n $ घटकवार अभिसरण है
अनुवर्ती: पहले निर्देशांक पर निकालें
(बोलजानो--वीयरस्ट्रैस ऑन $\R $), फिर, उसके बाद से, पर
दूसरा, और इसी तरह ($ n $ क्रमिक निष्कर्षण); बंदपन रहता है
अंदर की सीमा.

(3) दिया गया $(f(x_n))$, निकालना $ x_{\varphi(n)} \to x \in X $;
निरंतरता देता है $ f(x_{\varphi(n)}) \to f(x) \in f(X)$. वास्तविक मामला:
सघनता का $ f(X) \subseteq \R $ इसे बंद और घिरा बनाता है, और
$\sup f(X) \in f(X)$ (एक सेट का समर्थन इसके अनुरूप है, और
$ f(X)$ बन्द है)।

(4) वर्ष 1 प्रमाण शब्दशः स्थानांतरित होता है; यह यहाँ है, मीट्रिक में
पोशाक. कल्पना करना $ f \colon X \to Y $ \omterm{def:b2:metric:continuity}{निरंतर} पर \omterm{def:b2:metric:compact}{कॉम्पैक्ट} $ X $
लेकिन समान रूप से \omterm{def:b2:metric:continuity}{निरंतर} नहीं: कुछ $\varepsilon > 0$ स्वीकार करता है, के लिए
प्रत्येक $ n $, के साथ अंक
\[
d_X(x_n, y_n) \leq \frac{1}{n+1}
\qquad\text{and}\qquad
d_Y\bigl(f(x_n), f(y_n)\bigr) > \varepsilon .
\]
निकालना $ x_{\varphi(n)} \to a \in X $; तब $ y_{\varphi(n)} \to
a $ भी (आपसी दूरियाँ बढ़ती हैं।) $0$). निरंतरता पर $ a $
दोनों छवि अनुक्रम भेजता है $ f(a)$, इसलिए
$ d_Y\bigl(f(x_{\varphi(n)}), f(y_{\varphi(n)})\bigr) \to 0$ ---
एकसमान अंतर का खंडन $> \varepsilon $. सघनता
बिल्कुल एक चीज़ की आपूर्ति की गई: क्लस्टर बिंदु $ a $ जिस पर
सादा लगाएं निरंतरता.

(5) पर निकालें $ X $-निर्देशांक, फिर से पर
$ Y $-निर्देशांक.
\end{proof}

\begin{example}[हेन का प्रमेय, सघनता के साथ और उसके बिना]\label{ex:b2:metric:heinework}
पर $\intcc{0}{1}$, समारोह $ x \mapsto x^2$ समान रूप से है
\omterm{def:b2:metric:continuity}{निरंतर} --- हेन बिना किसी गणना के, बल्कि प्रत्यक्ष रूप से ऐसा कहता है
अनुमान शिक्षाप्रद है:
\[
\abs{x^2 - y^2} = \abs{x + y}\,\abs{x - y} \leq 2\abs{x - y},
\]
इसलिए $\delta = \varepsilon/2$ काम करता है \emph{प्रत्येक बिंदु के लिए
एक बार}. पर $\R $ समान फ़ंक्शन समान रूप से \omterm{def:b2:metric:continuity}{निरंतर} नहीं है:
के साथ $ x_n = n $ और $ y_n = n + \frac1n $, अन्तर $\abs{x_n - y_n}
= \frac1n \to 0$ जबकि $\abs{x_n^2 - y_n^2} = 2 + \frac1{n^2}
\geq 2$: एक भी नही $\delta $ सेवा करता है $\varepsilon = 1$. द
तंत्र दिखाई देता है: स्थानीय \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} स्थिर $\abs{x + y}$
ए पर बँधा हुआ है \omterm{def:b2:metric:compact}{कॉम्पैक्ट} और असीमित पर $\R $ --- हेन का
प्रमेय बिल्कुल वही कथन है सघनता कैप्स ऐसे
स्थानीय स्थिरांक समान रूप से।
\end{example}

\begin{method}[यह साबित करना कि एक सेट कॉम्पैक्ट है]\label{met:b2:metric:compactness}
आवृत्ति के क्रम में तीन मार्ग। (1) \emph{परिवेश
मान्यता:} में $\R^n $ (या कोई परिमित-आयामी मानक
अंतरिक्ष, \cref{ch:b2:nvs}), सत्यापित बंद --- आम तौर पर एक के रूप में
पूर्वछवि, \cref{ex:b2:metric:recognize} --- और बाध्य. (2)
\emph{विरासत:} किसी ज्ञात का एक बंद उपसमुच्चय \omterm{def:b2:metric:compact}{कॉम्पैक्ट} अंतरिक्ष है
\omterm{def:b2:metric:compact}{कॉम्पैक्ट}; एक परिमित संघ या का एक उत्पाद \omterm{def:b2:metric:compact}{कॉम्पैक्ट} कैमरों है \omterm{def:b2:metric:compact}{कॉम्पैक्ट}; ए
\omterm{def:b2:metric:continuity}{निरंतर} एक की छवि \omterm{def:b2:metric:compact}{कॉम्पैक्ट} है \omterm{def:b2:metric:compact}{कॉम्पैक्ट}. (3) \emph{नंगे हाथों:} एक मनमाना से एक अभिसरण परिणाम निकालें
अनुक्रम --- आमतौर पर क्रमिक निष्कर्षण द्वारा समन्वयित होता है
समन्वय करें. साबित करना \emph{गैर}-संक्षिप्तता, एक गवाह
पर्याप्त: एक अनुक्रम जिसमें कोई अभिसरण अनुवर्ती नहीं होता है, अधिकतर
आपसी दूरी पर बिंदु $\geq \varepsilon $.
\end{method}

\begin{example}[सेटों के बीच की दूरी: सघनता अपनी पकड़ बनाए रखती है]\label{ex:b2:metric:setdistance}
होने देना $ K $ होना \omterm{def:b2:metric:compact}{कॉम्पैक्ट}, $ F $ बंद किया हुआ, $ K \cap F = \emptyset $ में एक
\omterm{def:b2:metric:def}{मीट्रिक स्थान}. तब
\[
d(K, F) = \inf\,\{d(x, y) : x \in K,\ y \in F\} > 0 :
\]
समारोह $ x \mapsto d(x, F)$ है \omterm{def:b2:metric:continuity}{निरंतर}
(\cref{exo:b2:metric:11}) और सकारात्मक पर $ K $ ($ d(x, F) = 0$
डालूँगा $ x \in \overline F = F $), इसलिए यह एक सकारात्मकता प्राप्त करता है
पर न्यूनतम \omterm{def:b2:metric:compact}{कॉम्पैक्ट} $ K $
(\cref{thm:b2:metric:compactprops}~(3)). सघनता नहीं है
सजावटी: दो के लिए \emph{बंद किया हुआ} सेट न्यूनतम गायब हो सकता है
बिना प्राप्त किये---में $\R^2$, अतिपरवलय $ F_1 = \{xy
= 1\}$ और धुरी $ F_2 = \{y = 0\}$ असंयुक्त बंद समुच्चय हैं
के साथ $ d(F_1, F_2) = 0$ (बिन्दु $(n, \frac1n)$ के पास जाओ
अक्ष). अनन्त तक पलायन बिल्कुल यही है सघनता मना करता है.
\end{example}

\begin{theorem}[बोरेल--लेब्सग्यू]\label{thm:b2:metric:borellebesgue}
ए \omterm{def:b2:metric:def}{मीट्रिक स्थान} $ X $ है \omterm{def:b2:metric:compact}{कॉम्पैक्ट} यदि और केवल यदि प्रत्येक आवरण $ X $ द्वारा
\omterm{def:b2:metric:topology}{खुला} सेट में एक है \emph{परिमित} छिपाना।\index{बोरेल--लेब्सग्यू
प्रमेय}
\end{theorem}

\begin{proof}
($\Leftarrow $) कल्पना करना $(x_n)$ इसका कोई अभिसारी परिणाम नहीं है। हम
प्रत्येक का दावा करें $ x \in X $ एक गेंद है $ B(x, r_x)$ युक्त $ x_n $
केवल सीमित अनेक सूचकांकों के लिए $ n $: अन्यथा, हर गेंद
$ B(x, \frac1{k+1})$ इसमें अपरिमित रूप से अनेक पद होंगे, और
सूचकांक चुनना
\[
\varphi(0) < \varphi(1) < \varphi(2) < \cdots
\quad\text{with}\quad
x_{\varphi(k)} \in B\Bigl(x, \frac{1}{k+1}\Bigr)
\]
(प्रत्येक चरण पर सटीक रूप से संभव है क्योंकि अनंत रूप से अनेक
उम्मीदवार बने रहेंगे) एक अनुवर्ती अभिसरण का निर्माण करेंगे $ x $.
गेंदें $ B(x, r_x)$ ढकना $ X $; यदि उनमें से बहुत सारे हैं
ढका हुआ $ X $, सूचकांक सेट $\N $ का एक परिमित मिलन होगा
परिमित समुच्चय: बेतुका।

($\Rightarrow $) दो चरण. \emph{लेब्सेग संख्या:} एक के लिए \omterm{def:b2:metric:topology}{खुला} ढकना
$(U_i)$ एक का \omterm{def:b2:metric:compact}{कॉम्पैक्ट} $ X $, वहाँ है $\rho > 0$ ऐसे कि हर गेंद
त्रिज्या का $\rho $ कुछ में झूठ है $ U_i $. अन्यथा, प्रत्येक के लिए $ n $ चुनना
$ x_n $ साथ $ B(x_n, \frac{1}{n+1})$ नहीं में $ U_i $; निकालना
$ x_{\varphi(n)} \to x \in U_{i_0} \supseteq B(x, r)$; बड़े के लिए $ n $,
$ B(x_{\varphi(n)}, \frac{1}{\varphi(n)+1}) \subseteq B(x, r)
\subseteq U_{i_0}$: विरोधाभास। \emph{कुल सीमा:} हरएक के लिए $\rho > 0$, त्रिज्या की परिमित रूप से अनेक गेंदें $\rho $ ढकना $ X $.
अन्यथा आगमनात्मक रूप से चुनें $ x_{n+1}$ बाहर $ B(x_0, \rho) \cup
\dots \cup B(x_n, \rho)$: अनुक्रम में जोड़ीवार दूरियाँ हैं $\geq
\rho $, इसलिए कोई कॉची नहीं --- इसलिए कोई अभिसरण नहीं --- अनुवर्ती:
विरोधाभास. संयोजन: आवरण $ X $ त्रिज्या की बहुत सारी गेंदों द्वारा
$\rho $ (लेब्सग्यू नंबर), प्रत्येक कुछ के अंदर $ U_i $: एक परिमित
छिपाना।
\end{proof}

\begin{example}[एक $\varepsilon $-नेट, गिना गया]\label{ex:b2:metric:epsnet}
कुल सीमा (प्रमाण से)
\cref{thm:b2:metric:borellebesgue}) पर बहुत ठोस है
$\intcc{0}{1}$: के लिए $\varepsilon > 0$, द $\lceil

\begin{example}[कवर पर सघनता पढ़ना]\label{ex:b2:metric:covers}
 आधा \omterm{def:b2:metric:topology}{खुला} अंतराल $\intoc{0}{1}$ द्वारा कवर किया गया है \omterm{def:b2:metric:topology}{खुला} सेट
$ U_n = \intoo{\frac1n}{2}$, $ n \geq 1$; किसी भी परिमित उपपरिवार में एक होता है
सबसे बड़ा सूचकांक $ N $ और चूक जाता है $\intoc{0}{\frac1N}$: कोई परिमित नहीं
सबकवर, तो $\intoc{0}{1}$ क्या नहीं है \omterm{def:b2:metric:compact}{कॉम्पैक्ट} --- जो
अनुक्रमिक परिभाषा के माध्यम से देखता है $ x_n = \frac1n $, जिसकी सीमा
$0$ भाग जाता है. दूसरी ओर, एकल बिंदु जोड़ना $0$
दोनों निदानों को एक साथ ठीक करता है: चालू $\intcc{0}{1}$ ऐसे हर कवर
युक्त एक सेट होना चाहिए $0$, जो संपूर्ण प्रारंभिक को निगल जाता है
खंड, और सीमित रूप से कई सेट शेष को \omterm{def:b2:metric:complete}{पूरा} करते हैं। दो भाषाएँ
का \cref{thm:b2:metric:borellebesgue} हमेशा असफल या सफल
एक साथ --- कवर ठीक उसी स्थान पर भागने का पता लगाते हैं जहां अनुक्रम करते हैं।
\end{example}

\begin{remark}[इस खंड के भीतर परिप्रेक्ष्य]\label{rem:b2:metric:perspectives}
यह अध्याय खंड की भार वहन करने वाली दीवार है; देखो कहाँ
प्रत्येक स्तंभ भार वहन करता है। \emph{संपूर्णता}: कॉची
मानदंड बनच में श्रृंखला के लिए अभिसरण परीक्षण बन जाता है
रिक्त स्थान (\cref{ch:b2:series}), में एक समान अभिसरण
\cref{ch:b2:funcseq} में बिल्कुल अभिसरण है \omterm{def:b2:metric:complete}{पूरा}
$\bigl(C, d_\infty\bigr)$, और कॉची--लिप्सचिट्ज़
(\cref{ch:b2:diffeq}) बानाच निश्चित बिंदु प्रमेय पहन रहा है
एक अभिन्न समीकरण. \emph{सघनता}: यह साबित होता है
मानदंडों की समानता (\cref{ch:b2:nvs}), की प्राप्ति
के अनुकूलन के लिए एक्स्ट्रेमा \cref{ch:b2:diffcalc}, और
सर्वोत्तम सन्निकटन का अस्तित्व (\cref{ch:b2:nvs}का सप्ताहांत
समस्या) \emph{संयुक्तता}: यह स्थानीय बयानों का वैश्वीकरण करता है
---विभेदक समीकरणों के समाधान की विशिष्टता,
वक्रों पर मध्यवर्ती मान प्रमेय (\cref{ch:b2:curves}), और
के दो घटक $ GL_n(\R)$ वह अभिविन्यास सिद्धांत
(\cref{ch:b2:multint}) अलग रखेंगे.
\end{remark}

\begin{remark}[सामान्य ख़तरे]\label{rem:b2:metric:pitfalls}
(i) ``बंद और परिबद्ध का तात्पर्य है \omterm{def:b2:metric:compact}{कॉम्पैक्ट}'' के बारे में एक प्रमेय है
$\R^n $, के बारे में नहीं मीट्रिक रिक्त स्थान: के साथ एक अनंत सेट
असतत मीट्रिक अपने आप में बंद और परिबद्ध है फिर भी नहीं \omterm{def:b2:metric:compact}{कॉम्पैक्ट}
(\cref{exo:b2:metric:4}), और बंद इकाई गेंद
$ C(\intcc01)$ भी विफल रहता है (\cref{ch:b2:nvs}). (ii) संपूर्णता है
की एक संपत्ति \emph{दूरी}, टोपोलॉजी का नहीं: $\R $
साथ $ d(x,y) = \abs{\arctan x - \arctan y}$ सामान्य है
अभिसरण अनुक्रम लेकिन अधूरा है
(\cref{exo:b2:metric:1}). (iii) ए \omterm{def:b2:metric:continuity}{निरंतर} आपत्ति की आवश्यकता नहीं है
एक होमोमोर्फिज्म बनें --- सर्कल पैरामीट्रिजेशन
\cref{exo:b2:metric:7}; सघनता स्रोत की मरम्मत करता है।
(iv) बानाच की प्रमेय आवश्यकताएँ $ k < 1$ \emph{समान रूप से}: स्थिति $ d(f(x), f(y)) < d(x,y)$ अकेले किसी चीज़ की गारंटी नहीं देता
गैर \omterm{def:b2:metric:compact}{कॉम्पैक्ट} अंतरिक्ष (\cref{exo:b2:metric:5}). (वी) \omterm{def:b2:metric:connected}{जुड़े हुए} करता है
संकेत नहीं पथ से जुड़ा हुआ सामान्य तौर पर --- लेकिन के लिए \omterm{def:b2:metric:topology}{खुला}
इस पुस्तक में मानक स्थानों के उपसमुच्चय मिले हैं, दोनों सहमत हैं
(\cref{ch:b2:nvs}).
\end{remark}

\section{संयुक्तता}

\begin{definition}\label{def:b2:metric:connected}
$ X $ है \emph{जुड़े हुए}\index{जुड़े हुए} जब यह स्वीकार करता है कि नहीं
दो अरिक्त में विभाजन \omterm{def:b2:metric:topology}{खुला} उपसमुच्चय --- समतुल्य, जब यह
केवल दोनों उपसमुच्चय \omterm{def:b2:metric:topology}{खुला} और बंद हैं $\emptyset $ और $ X $. $ X $ है
\emph{पथ से जुड़ा हुआ} जब किन्हीं दो बिंदुओं को a से जोड़ा जाता है \omterm{def:b2:metric:continuity}{निरंतर}
नक्शा $\gamma \colon \intcc{0}{1} \to X $.
\end{definition}

\begin{theorem}\label{thm:b2:metric:connectedness}
\begin{enumerate}
  \item  \omterm{def:b2:metric:connected}{जुड़े हुए} के उपसमुच्चय $\R $ बिल्कुल अंतराल हैं.
  \item ए \omterm{def:b2:metric:continuity}{निरंतर} एक की छवि जुड़ा हुआ स्थान है \omterm{def:b2:metric:connected}{जुड़े हुए} ---
        सामान्य मध्यवर्ती मूल्य प्रमेय कहां से आया: a \omterm{def:b2:metric:continuity}{निरंतर}
        ए पर वास्तविक कार्य जुड़ा हुआ स्थान के बीच प्रत्येक मान लेता है
        इसके कोई दो मान.
  \item पथ से जुड़ा हुआ $\Rightarrow $ \omterm{def:b2:metric:connected}{जुड़े हुए}. (वार्तालाप विफल रहता है
        सामान्य तौर पर; यह कायम है \omterm{def:b2:metric:topology}{खुला} मानक स्थानों के उपसमुच्चय,
        \cref{ch:b2:nvs}.)
\end{enumerate}
\end{theorem}

\begin{proof}
(1) एक गैर-अंतराल $ A $ कुछ याद आता है $ z $ इसके दो बिंदुओं के बीच:
$ A = (A \cap \intoo{-\infty}{z}) \cup (A \cap \intoo{z}{+\infty})$
इसे दो अरिक्त भागों में विभाजित करता है \omterm{def:b2:metric:topology}{खुला} (में $ A $) टुकड़े। इसके विपरीत, चलो
$ I $ एक अंतराल हो और $ I = U \cup V $ गैररिक्त में एक विभाजन
अपेक्षाकृत \omterm{def:b2:metric:topology}{खुला} सेट; चुनना $ a \in U $, $ b \in V $, कहना $ a < b $, और सेट करें $ s = \sup\,(U \cap \intcc{a}{b})$, का एक बिंदु $\intcc{a}{b}
\subseteq I $. अगर $ s \in U $: तब $ s \neq b $, और सापेक्ष खुलापन
का $ U $ चारों ओर एक संपूर्ण अंतराल डालता है $ s $ (के साथ प्रतिच्छेद किया गया $ I $)
अंदर $ U $ --- तो के अंक $ U \cap \intcc{a}{b}$ से अधिक $ s $,
सर्वोच्च का खंडन. अगर $ s \in V $: सापेक्ष खुलापन $ V $
एक अंतराल डालता है $\intoo{s - r}{s + r} \cap I $ अंदर $ V $; लेकिन
सर्वोच्च का अनुयायी है $ U \cap \intcc{a}{b}$, जो उससे मिलना चाहिए
अंतराल---विरोधाभास के साथ $ U \cap V = \emptyset $. (यह है
वर्ष 1 के लिए क्लोपेन तर्क $\R $, अंदर भागो $ I $.)

(2) यदि $ f(X) = U' \cup V'$ अपेक्षाकृत गैररिक्त में विभाजित हो जाता है \omterm{def:b2:metric:topology}{खुला}
सेट, फिर $ X = f^{-1}(U') \cup f^{-1}(V')$ फूट $ X $
(\cref{thm:b2:metric:globalcontinuity}). आईवीटी: $ f(X) \subseteq \R $
है \omterm{def:b2:metric:connected}{जुड़े हुए}, इसलिए (1) से एक अंतराल।

(3) मान लीजिये $ X = U \cup V $, दोनों गैर-रिक्त \omterm{def:b2:metric:topology}{खुला}, और शामिल हों $ a \in U $
को $ b \in V $ एक पथ से $\gamma $: तब $\gamma^{-1}(U),
\gamma^{-1}(V)$ विभाजित करना $\intcc{0}{1}$, विरोधाभासी (1)।
\end{proof}

\begin{example}\label{ex:b2:metric:glnr}
$ GL_n(\R)$ क्या नहीं है \omterm{def:b2:metric:connected}{जुड़े हुए}: $\det $ है \omterm{def:b2:metric:continuity}{निरंतर} (एक बहुपद में
प्रविष्टियाँ) पर $\R^*$, जो नहीं है \omterm{def:b2:metric:connected}{जुड़े हुए}; की पूर्वछवियाँ
$\R_+^*$ और $\R_-^*$ विभाजित करना $ GL_n(\R)$. (प्रत्येक टुकड़ा वास्तव में है
पथ से जुड़ा हुआ --- हमारी आवश्यकताओं से परे एक सुखद अभ्यास।) द्वारा
विरोधाभास $ GL_n(\C)$ \emph{है} पथ से जुड़ा हुआ:
\cref{exo:b2:metric:10}.
\end{example}

\begin{example}[अकेले संयोजकता से एक निश्चित बिंदु]\label{ex:b2:metric:ivtfixed}
प्रत्येक \omterm{def:b2:metric:continuity}{निरंतर} $ f \colon \intcc01 \to \intcc01$ एक निश्चित है
बिंदु --- कोई संकुचन परिकल्पना नहीं, कोई पुनरावृत्ति नहीं। विचार करना
$ g(x) = f(x) - x $, \omterm{def:b2:metric:continuity}{निरंतर} पर \omterm{def:b2:metric:connected}{जुड़े हुए} $\intcc01$:
\[
g(0) = f(0) \geq 0,
\qquad
g(1) = f(1) - 1 \leq 0 ,
\]
और मध्यवर्ती मूल्य प्रमेय
(\cref{thm:b2:metric:connectedness}~(2)) का शून्य प्रदान करता है
$ g $, यानी\ का एक निश्चित बिंदु $ f $. बानाच से तुलना करें
(\cref{thm:b2:metric:banach}): यहां अस्तित्व टोपोलॉजिकल है
और मुफ़्त, लेकिन विशिष्टता और एल्गोरिदम खो गए हैं --- $ f =
\mathrm{id}$ प्रत्येक बिंदु निश्चित है, और a की पुनरावृत्ति है
गैर-ठेका $ f $ हमेशा के लिए \omterm{def:b2:structures:sn}{चक्र} हो सकता है। दो निश्चित बिंदु
इस अध्याय के प्रमेय विभिन्न प्रश्नों के उत्तर देते हैं
विभिन्न मुद्राएँ।
\end{example}

\begin{example}[$\R $ और $\R^2$ होम्योमॉर्फिक नहीं हैं]\label{ex:b2:metric:rnotr2}
संयुक्तता एक टोपोलॉजिकल फ़िंगरप्रिंट है. कल्पना करना $ h \colon
\R^2 \to \R $ एक होमोमोर्फिज्म थे (ए \omterm{def:b2:metric:continuity}{निरंतर} आपत्ति के साथ
\omterm{def:b2:metric:continuity}{निरंतर} श्लोक में)। एक बिंदु हटा दें $ a \in \R^2$: द
प्रतिबंध $ h \colon \R^2\setminus\{a\} \to
\R\setminus\{h(a)\}$ अभी भी एक होमियोमोर्फिज्म है। लेकिन $\R^2$ माइनस
एक बिंदु है पथ से जुड़ा हुआ --- किन्हीं दो बिंदुओं को एक खंड द्वारा जोड़ें,
एक सहायक बिंदु के माध्यम से दूसरे खंड के साथ चक्कर लगाना यदि
$ a $ प्रत्यक्ष को अवरुद्ध करता है --- इसलिए \omterm{def:b2:metric:connected}{जुड़े हुए}
(\cref{thm:b2:metric:connectedness}~(3)); जबकि $\R $ माइनस ए
बिंदु दो अरिक्त में विभाजित हो जाता है \omterm{def:b2:metric:topology}{खुला} आधी पंक्तियाँ: नहीं \omterm{def:b2:metric:connected}{जुड़े हुए}.
संयुक्तता द्वारा संरक्षित है \omterm{def:b2:metric:continuity}{निरंतर} मानचित्र: विरोधाभास.
समतल और रेखा वास्तव में भिन्न हैं \emph{जैसे
टोपोलॉजिकल रिक्त स्थान} --- एक तथ्य यह है कि अकेले कार्डिनैलिटी
(\cref{exo:b2:structures:3}-स्टाइल आक्षेप मौजूद हैं!) भी है
देखने में मोटा.
\end{example}

\begin{exercise}[$\star $]\label{exo:b2:metric:3}
में $\bigl(C(\intcc{0}{1}), d_\infty\bigr)$, दूरी की गणना करें

$\overline B(0, 1)$; और सिद्ध करें कि समुच्चय $\{f : f(0) = 0\}$ है
जबकि बंद है $\{f : f(0) > 0\}$ है \omterm{def:b2:metric:topology}{खुला}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:metric:4}
सिद्ध कीजिए कि असतत \omterm{def:b2:metric:def}{मीट्रिक स्थान} $ X $ (कोई भी सेट) है \omterm{def:b2:metric:complete}{पूरा}, और
वह यह है \omterm{def:b2:metric:compact}{कॉम्पैक्ट} अगर और केवल अगर $ X $ परिमित है. कौन से उपसमुच्चय हैं
\omterm{def:b2:metric:connected}{जुड़े हुए}?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:metric:5}
होने देना $ X $ होना \omterm{def:b2:metric:compact}{कॉम्पैक्ट} और $ f \colon X \to X $ साथ
\[
d\bigl(f(x), f(y)\bigr) < d(x, y) \quad \text{for all } x \neq y .
\]
साबित करें कि $ f $ एक अद्वितीय निश्चित बिंदु है \emph{(छोटा करना $ x \mapsto
d(x, f(x))$)}, और एक उदाहरण दीजिए $ X = \intco{1}{+\infty}$
(नहीं \omterm{def:b2:metric:compact}{कॉम्पैक्ट}) बिना किसी निश्चित बिंदु के।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:metric:6}
(नेस्टेड \omterm{def:b2:metric:compact}{कॉम्पैक्ट} कैमरों) होने देना $(K_n)$ गैर-रिक्त का घटता क्रम बनें
\omterm{def:b2:metric:compact}{कॉम्पैक्ट} ए के उपसमुच्चय \omterm{def:b2:metric:def}{मीट्रिक स्थान}. साबित करें कि $\bigcap_n K_n \neq
\emptyset $ \emph{(चुनना $ x_n \in K_n $ और निकालें)}. उदाहरण देकर दिखाओ
वह गैर-रिक्त घोंसला \emph{बंद किया हुआ} में सेट $\R $ खाली हो सकता है
चौराहा.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:metric:7}
होने देना $ K $ होना \omterm{def:b2:metric:compact}{कॉम्पैक्ट} और $ f \colon K \to Y $ \omterm{def:b2:metric:continuity}{निरंतर} और विशेषण.
साबित करें कि $ f^{-1}$ है \omterm{def:b2:metric:continuity}{निरंतर} \emph{(बंद सेट का उपयोग करें:
\cref{thm:b2:metric:globalcontinuity} और
\cref{thm:b2:metric:compactprops})}. बिना प्रति-उदाहरण दीजिए
सघनता ($\gamma(t) = (\cos t, \sin t)$ पर
$\intco{0}{2\pi}$).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:metric:8}
 \emph{कैंटर सेट} $ C $ से प्राप्त किया जाता है $\intcc{0}{1}$ द्वारा
बार-बार हटाना \omterm{def:b2:metric:topology}{खुला} मध्य तिहाई. साबित करें कि $ C $ है \omterm{def:b2:metric:compact}{कॉम्पैक्ट},
इसका आंतरिक भाग ख़ाली है, और वास्तव में अनंत है \omterm{def:b2:structures:countable}{सुसज्जित} को
$\{0,1\}^{\N}$ \emph{(अंकों के साथ टर्नरी विस्तार $0,2$;
\cref{exo:b2:structures:3})}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:metric:9}
होने देना $ X $ एक हो \emph{\omterm{def:b2:metric:compact}{कॉम्पैक्ट}} \omterm{def:b2:metric:def}{मीट्रिक स्थान} और $ f \colon X \to X $ एक
आइसोमेट्री: $ d(f(x), f(y)) = d(x, y)$. साबित करें कि $ f $ विशेषण है.
\emph{संकेत: यदि $ a \notin f(X)$, तब $\varepsilon = d(a, f(X)) > 0$
(क्यों?); कक्षा का अध्ययन करें $ a, f(a), f^2(a), \dots $ और इसे दिखाओ
अंक जोड़ीवार हैं $\geq \varepsilon $ अलग --- के साथ विरोधाभास
सघनता.}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:metric:10}
साबित करें कि $ GL_n(\C)$ है पथ से जुड़ा हुआ. \emph{संकेत: दिया गया $ A, B $
उलटा, विचार करें $ p(z) = \det\bigl((1 - z)A + zB\bigr)$ के लिए $ z
\in \C $: में एक बहुपद $ z $, समान रूप से शून्य नहीं है, इसलिए यह है
परिमित रूप से अनेक जड़ें; जोड़ना $0$ को $1$ में $\C $ परहेज़ के रास्ते से
उन्हें.}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:metric:11}
के लिए $\emptyset \neq A \subseteq X $, तय करना $ d(x, A) = \inf_{a \in A}
d(x, a)$. साबित करें कि $ x \mapsto d(x, A)$ है $1$-लिप्सचिट्ज़, वह
$ d(x, A) = 0$ आईएफएफ $ x \in \overline A $, और वह असंयुक्त के लिए
गैररिक्त \emph{बंद किया हुआ} सेट $ A, B $ समारोह
\[
\varphi(x) = \frac{d(x, A)}{d(x, A) + d(x, B)}
\]
अच्छी तरह से परिभाषित है, \omterm{def:b2:metric:continuity}{निरंतर}, के बराबर $0$ बिलकुल चालू $ A $ और को
$1$ बिलकुल चालू $ B $ --- ए \omterm{def:b2:metric:continuity}{निरंतर} किन्हीं दो को अलग करने वाला ``स्विच''
असंयुक्त बंद सेट.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:metric:12}
(बेयर) चलो $ X $ एक हो पूर्ण \omterm{def:b2:metric:def}{मीट्रिक स्थान} और $(U_n)_{n\geq1}$ ए
सघनता का क्रम \omterm{def:b2:metric:topology}{खुला} उपसमुच्चय साबित करें कि $\bigcap_n U_n $ है
में घना $ X $ \emph{(किसी भी गेंद के अंदर, नेस्टेड बंद गेंदें बनाएं
$\overline B(x_n, r_n) \subseteq U_n $ साथ $ r_n \to 0$ और उपयोग करें
पूर्णता)}. उसका निष्कर्ष निकालो $\R $ नहीं है कोई \omterm{def:b2:structures:countable}{गणनीय} का संघ
खाली इंटीरियर के साथ बंद सेट, और पुनर्प्राप्त करें --- फिर से --- वह
$\R $ बेशुमार है.
\end{exercise}

\section{समस्या: पिकार्ड पुनरावृत्ति}

संपूर्णता प्लस संकुचन एक समाधान मशीन है: इसे फ़ीड करें
समीकरण को एक निश्चित बिंदु समस्या के रूप में लिखा जाता है, और यह वापस आ जाता है
अस्तित्व, विशिष्टता, एक एल्गोरिथ्म, और त्रुटि पट्टियाँ। इस सप्ताहांत
समस्या समीकरण पर मशीन को पूरी शक्ति से चलाती है $ y' = f(t,
y)$: हम स्थानीय साबित करते हैं \emph{पिकार्ड--लिंडेल\"प्रमेय का} (द
का अरेखीय हृदय \cref{ch:b2:diffeq}कॉची--लिप्सचिट्ज़
सिद्धांत), हर परिकल्पना को \omterm{def:b2:metric:complete}{पूरा} होते हुए देखें
प्रतिउदाहरण, और विशुद्ध रूप से मीट्रिक लाभांश एकत्र करें ---
\omterm{def:b2:metric:continuity}{निरंतर} डेटा पर निर्भरता, केपलर का समीकरण, और
कैंटर सेट की स्व-समानता।

\begin{problem}[{सप्ताहांत समस्या --- पिकार्ड--लिंडेल\"का
प्रमेय}]
\label{pb:b2:metric:1}
लगातार, $ t_0 \in \R $, $ y_0 \in \R $, $ a, b > 0$, और $ f $ एक है
\omterm{def:b2:metric:continuity}{निरंतर} आयत पर कार्य करें $ R = \intcc{t_0 - a}{t_0 +
a} \times \intcc{y_0 - b}{y_0 + b}$, से घिरा $ M =
\sup_R\,\abs f $, और \emph{$ L $-लिप्सचिट्ज़ अपने दूसरे चर में}:
$\abs{f(t, y) - f(t, z)} \leq L\abs{y - z}$ जब भी दोनों बिंदु
लेट जाओ $ R $. तय करना
\[
h = \min\Bigl(a, \frac bM\Bigr) \quad (\text{with } h = a
\text{ if } M = 0), \qquad I = \intcc{t_0 - h}{t_0 + h}.
\]

\medskip
\noindent\textbf{Part I --- The \omterm{def:b2:metric:complete}{पूरा} stage.}
\begin{enumerate}
  \item उद्धृत दोनों कथनों को सिद्ध कीजिए
        \cref{def:b2:metric:complete}: a का एक बंद उपसमुच्चय
        पूर्ण \omterm{def:b2:metric:def}{मीट्रिक स्थान} है \omterm{def:b2:metric:complete}{पूरा}, और ए \omterm{def:b2:metric:complete}{पूरा} उपसमुच्चय
        किसी का \omterm{def:b2:metric:def}{मीट्रिक स्थान} बन्द है। यह निष्कर्ष निकालें कि प्रत्येक बंद है
        का उपसमुच्चय $\bigl(C(I), d_\infty\bigr)$ एक है पूर्ण
        \omterm{def:b2:metric:def}{मीट्रिक स्थान}.
  \item वह नक्शा दिखाओ $ C(I) \to C(I)$, $ y \mapsto \bigl(t
        \mapsto \int_{t_0}^{t}y(s)\,\dd s\bigr)$, है
        $ h $-लिप्सचिट्ज़ के लिए $ d_\infty $.
  \item (निश्चित बिंदु मानचित्रों की तुलना में कम चलते हैं) चलो $ g \colon X
        \to X $ एक हो $ k $-ए का संकुचन \omterm{def:b2:metric:def}{मीट्रिक स्थान} निश्चित के साथ
        बिंदु $\ell_g $, और $\widetilde g \colon X \to X $
        \emph{कोई} एक निश्चित बिंदु वाला मानचित्र $\ell_{\widetilde g}$.
        सिद्ध करना
        \[
        d(\ell_g, \ell_{\widetilde g}) \leq
        \frac{d\bigl(g(\ell_{\widetilde g}),
        \widetilde g(\ell_{\widetilde g})\bigr)}{1 - k}
        \leq \frac{\sup_{x \in X} d\bigl(g(x), \widetilde
        g(x)\bigr)}{1 - k}.
        \]
  \item (पुनरावृत्त युक्ति) चलो $ X $ होना \omterm{def:b2:metric:complete}{पूरा} गैररिक्त और $ g
        \colon X \to X $ एक नक्शा --- अनुमान नहीं लगाया गया \omterm{def:b2:metric:continuity}{निरंतर} ---
        इस प्रकार कि कुछ पुनरावृत्त होते हैं $ g^m $ एक है $ k $-सिकुड़न।
        साबित करें कि $ g $ एक अद्वितीय निश्चित बिंदु है $\ell $ ओर वो
        \emph{प्रत्येक} कक्षा $ x_{n+1} = g(x_n)$ में एकत्रित हो जाता है
        $\ell $. \emph{(के निश्चित बिंदु $ g $ के निश्चित बिंदु हैं
        $ g^m $; इसके विपरीत $ g(\ell)$ का एक निश्चित बिंदु है $ g^m $;
        अवशेष मॉड के साथ कक्षा को विभाजित करें $ m $.)}
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The Picard--Lindelöf theorem.}
\begin{enumerate}[resume]
  \item वह एक फ़ंक्शन दिखाएँ $ y \colon I \to \intcc{y_0 - b}{y_0
        + b}$ है $ C^1$ साथ $ y(t_0) = y_0$ और $ y' = f(t, y)$ पर
        $ I $ यदि और केवल यदि यह है \omterm{def:b2:metric:continuity}{निरंतर} और संतुष्ट करता है
        अभिन्न समीकरण
        \[
        y(t) = y_0 + \int_{t_0}^{t} f\bigl(s, y(s)\bigr)\dd s
        \qquad (t \in I).
        \]
  \item होने देना $ X_h = \{y \in C(I) : \abs{y(t) - y_0} \leq b
        \text{ on } I\}$ और चलो $ T $ द्वारा परिभाषित किया जाए $ T(y)(t) =
        y_0 + \int_{t_0}^{t}f(s, y(s))\dd s $. बताते हैं कि $ X_h $ है
        का एक गैररिक्त बंद उपसमुच्चय $ C(I)$, इस तरह \omterm{def:b2:metric:complete}{पूरा}, ओर वो $ T $ एमएपीएस $ X_h $ में $ X_h $ --- यहीं पर $ h \leq
        b/M $ काम करता है.
  \item बताते हैं कि $ d_\infty\bigl(T(y), T(z)\bigr) \leq
        Lh\,d_\infty(y, z)$ पर $ X_h $: अगर $ Lh < 1$, बनच का
        प्रमेय पहले ही समाप्त हो चुका है। हम इस छोटेपन को दूर करते हैं
        अगली शर्त.
  \item प्रेरण द्वारा सिद्ध करें $ n $:
        \[
        \abs{T^n(y)(t) - T^n(z)(t)} \leq
        \frac{\bigl(L\abs{t - t_0}\bigr)^n}{n!}\,
        d_\infty(y, z) \qquad (y, z \in X_h,\ t \in I),
        \]
        तो कुछ पुनरावृत्त $ T $ एक संकुचन है. के साथ समापन करें
        प्रश्न 4 (\emph{पिकार्ड--लिंडेल\"प्रमेय का}): द
        कॉची समस्या $ y' = f(t,y)$, $ y(t_0) = y_0$ बिलकुल है
        एक समाधान चालू $ I = \intcc{t_0 - h}{t_0 + h}$ के साथ
        में मान $\intcc{y_0 - b}{y_0 + b}$.
  \item दिखाएँ कि प्रतिबंध ``मानों के साथ $\intcc{y_0 -
        b}{y_0 + b}$'' स्वचालित है: कॉची का कोई भी समाधान
        समस्या पर परिभाषित किया गया $ I $ जिसका ग्राफ शुरू होता है $ R $ रहता है
        में $\intcc{y_0 - b}{y_0 + b}$ \emph{(पहले पर विचार करें
        बाहर निकलने का समय और सीमा $\abs{y(t) - y_0}$ द्वारा $ M\abs{t -
        t_0}$)}. इसलिए सभी समाधानों में विशिष्टता कायम है
        $ I $.
  \item मशीन चालू करें $ y' = y $, $ y(0) = 1$, से शुरू
        स्थिरांक $ y^{(0)} \equiv 1$: पिकार्ड की गणना करें
        पुनरावृत्त करता है $ y^{(n)}$, उन्हें पहचानें, और उनका वर्णन करें
        अभिसरण.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Every hypothesis earns its keep.}
\begin{enumerate}[resume]
  \item (\omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} विफल रहता है, विशिष्टता विफल रहती है) के लिए $ y' =
        2\sqrt{\abs y}$, $ y(0) = 0$: जाँच करें कि $ y \equiv 0$
        और, प्रत्येक के लिए $ c \geq 0$, समारोह $ y_c(t) = 0$ के लिए
        $ t \leq c $, $ y_c(t) = (t - c)^2$ के लिए $ t > c $, सभी हैं
        $ C^1$ समाधान चालू $\R $. वास्तव में कहाँ होता है $ y \mapsto
        2\sqrt{\abs y}$ होने में असफल \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़}?
  \item (इलाका वास्तविक है) के लिए $ y' = y^2$, $ y(0) = 1$: हल करें
        स्पष्ट रूप से, अस्तित्व का अधिकतम अंतराल दें, और
        सर्वोत्तम की गणना करें $ h $ प्रमेय सब पर प्रमाणित कर सकता है
        आयत के विकल्प ($ a $ बड़ा, $ b $ मुफ़्त): दिखाएँ
        $ h_{\max} = \sup_{b>0} \frac{b}{(1+b)^2} = \frac14$,
        जबकि सच्चा समाधान जीवित रहता है $\intoo{-\infty}{1}$.
  \item (संपूर्णता सजावट नहीं है) चालू $ X = \Q \cap
        \intcc{1}{2}$ सामान्य दूरी के साथ, चलो $ g(x) =
        \frac x2 + \frac1x $. दिखाओ $ g(X) \subseteq X $, वह $ g $
        एक है $\frac12$-सिकुड़न \emph{(माध्य मान
        असमानता)}, ओर वो $ g $ में कोई निश्चित बिंदु नहीं है $ X $.
        बानाच के प्रमेय की कौन सी परिकल्पना विफल होती है, और कौन सी है
        समापन में निश्चित बिंदु?
  \item (त्रुटि पट्टियाँ) एक के लिए $ k $-सिकुड़न $ g $ एक पर \omterm{def:b2:metric:complete}{पूरा}
        स्थान, सिद्ध करो \emph{वापस} अनुमान लगाना $ d(x_n,
        \ell) \leq \frac{k}{1-k}\,d(x_n, x_{n-1})$. बगुला के लिए
        नक्शा $ g(x) = \frac x2 + \frac 1x $ पर $\intcc{1}{2}$
        (नियत बिन्दु $\sqrt2$), से शुरू $ x_0 = \frac32$:
        कितने कदम है \emph{संभवतः} अवश्यंभावी
        $\frac{k^n}{1-k}d(x_1, x_0)$ सटीकता की मांग
        $10^{-6}$, और वास्तविकता में कितने कदम पर्याप्त हैं?
        (गणना करें $ x_1, x_2, x_3$ और उनकी त्रुटियाँ; द
        संकुचन की बाध्यता ईमानदार लेकिन निराशावादी है ---
        बगुला चतुष्कोणीय रूप से अभिसरण करता है।)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- \omterm{def:b2:metric:continuity}{निरंतर} dependence.} इस हिस्से में $ Lh < 1$, इसलिए $ T $ अपने आप में एक संकुचन है $ X_h $ (प्रश्न
7); लिखना $ y[\,y_0\,]$ प्रारंभिक मूल्य वाले समाधान के लिए
$ y_0$.
\begin{enumerate}[resume]
  \item (प्रारंभिक मूल्य पर निर्भरता) चलो $ z_0$ दूसरा बनो
        प्रारंभिक मूल्य के साथ $\abs{z_0 - y_0}$ इतना छोटा कि
        दोनों समस्याएँ आयत में फिट होती हैं। प्रश्न 3 का उपयोग करते हुए,
        साबित करो
        \[
        d_\infty\bigl(y[y_0], y[z_0]\bigr) \leq
        \frac{\abs{y_0 - z_0}}{1 - Lh}.
        \]
  \item (श्रृंखला द्वारा लंबे अंतराल) मान लीजिए कि समाधान मौजूद हैं
        एक लंबे खंड पर काटा गया $ m $ लगातार टुकड़े चालू
        जिनमें से प्रत्येक के साथ पिछली सीमा लागू होती है $ Lh \leq
        \frac12$. दिखाएँ कि विचलन अधिकतम एक कारक से बढ़ता है
        $2$ प्रति टुकड़ा, इसलिए $ d_\infty \leq 2^m\abs{y_0 - z_0}$
        कुल मिलाकर --- एक घातीय-लंबाई से बंधा हुआ, असतत
        की छाया $\eu^{L\abs{t - t_0}}$ ग्रोनवाल की लेम्मा का
        (\cref{ch:b2:diffeq}).
  \item (मैदान पर निर्भरता) चलो $ g $ पर एक और फ़ील्ड बनें
        $ R $, भी $ L $-लिप्सचिट्ज़ में $ y $, साथ $\sup_R \abs{f - g}
        \leq \varepsilon $. सिद्ध कीजिए कि संगत
        समाधान संतुष्ट करते हैं $ d_\infty \leq
        \frac{\varepsilon h}{1 - Lh}$: मॉडलिंग त्रुटि
        रैखिक रूप से प्रचारित होता है।
  \item (पैरामीटर) यदि एक परिवार $ f_\lambda $ फ़ील्ड का है
        $ L $-लिप्सचिट्ज़ में $ y $ समान रूप से और $\sup_R\abs{f_\lambda
        - f_\mu} \leq C\abs{\lambda - \mu}$, वह निष्कर्ष निकालो
        $\lambda \mapsto y_\lambda $ है \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} से
        पैरामीटर स्थान में $\bigl(C(I), d_\infty\bigr)$.
  \item (सिस्टम की लागत कुछ भी नहीं) स्पष्ट करें कि भाग I, II और IV क्यों हैं
        के लिए शब्दशः पकड़ें $ y $ मूल्यों के साथ $\R^n $ (सुपर
        की किसी भी दूरी पर बनी दूरियाँ
        \cref{ex:b2:metric:examples}), फिर सभी पिकार्ड की गणना करें
        सिस्टम के लिए पुनरावृत्त होता है $ y' = Ay $, $ y(0) = (c_1,
        c_2)$, $ A = \begin{psmallmatrix} 0 & 1\\ 0 &
        0\end{psmallmatrix}$: दिखाएँ पुनरावृत्ति बन जाती है
        एक चरण के बाद सटीक समाधान पर स्थिर।
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Metric dividends and synthesis.}
\begin{enumerate}[resume]
  \item (पहचान में गड़बड़ी) चलो $ X $ एक हो \omterm{def:b2:metric:complete}{पूरा}
        मानक-अंतरिक्ष-जैसी अवस्था: ले लो $ X = C(I)$ या $\R^n $. अगर
        $\eta \colon X \to X $ है $ k $-लिप्सचिट्ज़ के साथ $ k < 1$,
        साबित करें कि $ x \mapsto x + \eta(x)$ का आक्षेप है $ X $
        जिसका व्युत्क्रम है $\frac1{1-k}$-लिप्सचिट्ज़ \emph{(प्रत्येक के लिए
        $ y $, बैनच को लागू करें $ x \mapsto y - \eta(x)$)}. ये है
        व्युत्क्रम फलन प्रमेय का मीट्रिक हृदय
        (\cref{ch:b2:diffcalc}).
  \item (केपलर समीकरण) के लिए $0 \leq e < 1$ और $ m \in \R $,
        साबित करें कि $ x = m + e\sin x $ बिल्कुल एक ही समाधान है,
        वह पुनरावृत्ति $ x_{n+1} = m + e\sin x_n $ जुटता है
        इसे किसी भी शुरुआत से, और अनुमान: के लिए $ e = \frac12$,
        $ m = 1$, कितने पुनरावृत्तियाँ त्रुटि की गारंटी देती हैं $\leq
        10^{-3}$ एक प्राथमिकता से बाध्य? (समाधान है $ x
        \approx 1.4987$.)
  \item (कैंटर सेट एक निश्चित बिंदु है) चलो $ S_1(x) = \frac
        x3$ और $ S_2(x) = \frac x3 + \frac23$ पर $\R $, और चलो
        $ C $ का कैंटर सेट हो \cref{exo:b2:metric:8}. सिद्ध करना
        $ C = S_1(C) \cup S_2(C)$, और एक वाक्य में समझाएं
        क्यों नहीं? \emph{अन्य} अरिक्त \omterm{def:b2:metric:compact}{कॉम्पैक्ट} सेट इसे संतुष्ट करता है
        समीकरण (मानचित्र $ A \mapsto S_1(A) \cup S_2(A)$ एक है
        के बीच की दूरी के लिए संकुचन \omterm{def:b2:metric:compact}{कॉम्पैक्ट} सेट ---
        हॉसडॉर्फ़ दूरी, वर्ष 3 खंड में ईमानदार बनाई गई)।
  \item (संयुक्तता विशिष्टता को वैश्वीकृत करता है) चलो $ f $ स्थानीय रूप से हो
        \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} में $ y $ एक पर \omterm{def:b2:metric:topology}{खुला} सेट करें, और जाने दें $ y, z $ दो हो
        के समाधान $ y' = f(t, y)$ एक सामान्य अंतराल पर $ J $
        साथ $ y(t_0) = z(t_0)$. सिद्ध करना $ y = z $ पर $ J $: बताते हैं कि
        $\{t \in J : y(t) = z(t)\}$ शून्य नहीं है, अंदर बंद है $ J $,
        और \omterm{def:b2:metric:topology}{खुला} में $ J $ (स्थानीय विशिष्टता द्वारा), और उपयोग करें
        संयुक्तता अंतराल का
        (\cref{thm:b2:metric:connectedness}).
  \item (रैखिक समीकरणों के लिए कोई लघुता नहीं) के लिए $ y' =
        \alpha(t)y + \beta(t)$ साथ $\alpha, \beta $ \omterm{def:b2:metric:continuity}{निरंतर}
        एक खंड पर $\intcc{A}{B}$, दिखाने के लिए प्रश्न 8 को अनुकूलित करें
        फैक्टोरियल बाउंड पर कायम है \emph{साबुत} खंड,
        इसलिए अस्तित्व और विशिष्टता वहां वैश्विक हैं---
        का अदिश मामला \cref{ch:b2:diffeq}कॉची--लिप्सचिट्ज़
        प्रमेय, लंबाई पर कोई प्रतिबंध नहीं $ B - A $.
  \item (संश्लेषण) एक-एक वाक्य: क्या पूर्णता
        योगदान दिया; संकुचन ने क्या योगदान दिया; क्या
        सादे बानाच की तुलना में खरीदी गई पुनरावृत्त चाल; कहाँ
        संयुक्तता प्रविष्टि की; और भाग का कौन सा प्रति उदाहरण
        III किस परिकल्पना की रक्षा करता है। शिखर प्रमेय का नाम बताएं,
        और बताएं कि कब संकुचन की जगह क्या लेता है $ f$ मात्र है
        \omterm{def:b2:metric:continuity}{निरंतर} (पीनो का प्रमेय, के माध्यम से सघनता समारोह में
        रिक्त स्थान --- वर्ष 3 खंड का अर्ज़ेल\`ए--एस्कोली)।
\end{enumerate}
\end{problem}
