\chapter{द्विघात रूप}\label{ch:b2:quadratic}

ए द्विघात रूप ज्यामिति की बीजगणितीय छाया है: \omterm{thm:b2:structures:signature}{हस्ताक्षर}
शून्य हिस्से चपटे होते हैं, सकारात्मक हिस्से एक तरफ मुड़ते हैं, नकारात्मक हिस्से एक तरफ मुड़ते हैं
अन्य. यह अध्याय प्रत्येक यथार्थ को कम करता है द्विघात रूप के योग तक
$\pm $ वर्ग (गॉस), साबित करता है कि संकेतों की गिनती आंतरिक है
(सिल्वेस्टर), और फिर यूक्लिडियन ज्यामिति को ताज पहनाया
\emph{वर्णक्रमीय प्रमेय}: \omterm{def:b2:quadratic:adjoint}{सममित} एंडोमोर्फिज्म में विकर्ण बनाना
ऑर्थोनॉर्मल आधार --- लागू होने वाला एकल सबसे अधिक उपयोग किया जाने वाला प्रमेय
रैखिक बीजगणित.

\section{द्विरेखीय और द्विघात रूप}

\begin{definition}\label{def:b2:quadratic:def}
ए \emph{\omterm{def:b2:quadratic:adjoint}{सममित} द्विरेखीय रूप} एक वास्तविक सदिश स्थान पर $ E $ एक है
द्विरेखीय $\varphi \colon E \times E \to \R $ साथ $\varphi(x, y) =
\varphi(y, x)$; जुड़े \emph{द्विघात
प्रपत्र}\index{द्विघात
प्रपत्र} है $ q(x) = \varphi(x, x)$. फार्म
$\varphi $ से बरामद किया गया है $ q $ द्वारा
\emph{ध्रुवीकरण}\index{ध्रुवीकरण}:
\[
\varphi(x, y) = \tfrac12\bigl(q(x + y) - q(x) - q(y)\bigr).
\]
एक आधार में $(e_i)$, द \emph{मैट्रिक्स} का $\varphi $ है \omterm{def:b2:quadratic:adjoint}{सममित}
$ B = (\varphi(e_i, e_j))$, साथ $ q(x) = X^{\mathsf T} B X $; एक बदलाव
मैट्रिक्स के साथ आधार का $ P $ के स्थान पर $ B $ द्वारा $ P^{\mathsf T} B P $
(\emph{अनुरूपता}\index{अनुरूपता} ---समानता नहीं!)
 \emph{रैंक} का $ q $ है $\operatorname{rk} B $ (अपरिवर्तनीय:
सर्वांगसमता व्युत्क्रमणीय आव्यूहों से गुणा होती है)।
\end{definition}

\begin{example}
पर $\R^2$: $ q(x, y) = x^2 + 4xy + y^2$ मैट्रिक्स है $\begin{pmatrix}
1 & 2\\ 2 & 1\end{pmatrix}$. एक आंतरिक उत्पाद वास्तव में एक है
\omterm{def:b2:quadratic:adjoint}{सममित} द्विरेखीय रूप किसका द्विघात रूप सकारात्मक निश्चित है;
यह अध्याय सामान्य, संकेत-अनिश्चित मामले का अध्ययन करता है।
\end{example}

\begin{example}[कार्रवाई में अनुरूपता]\label{ex:b2:quadratic:congruence}
लेना $ q(x, y) = x^2 + 4xy + y^2$ (मैट्रिक्स $ B = \begin{pmatrix} 1
& 2\\ 2 & 1\end{pmatrix}$) और नया आधार $ e_1' = (1, 1)$,
$ e_2' = (1, -1)$, यानी\ $ P = \begin{pmatrix} 1 & 1\\ 1 &
-1\end{pmatrix}$. तब
\[
P^{\mathsf T}BP
= \begin{pmatrix} 1 & 1\\ 1 & -1\end{pmatrix}
\begin{pmatrix} 1 & 2\\ 2 & 1\end{pmatrix}
\begin{pmatrix} 1 & 1\\ 1 & -1\end{pmatrix}
= \begin{pmatrix} 6 & 0\\ 0 & -2\end{pmatrix} :
\]
निर्देशांक में $(u, v)$ नये आधार पर, $ q = 6u^2 -
2v^2$ --- जाँच करना: $ x = u + v $, $ y = u - v $ देता है $ x^2 + 4xy +
y^2 = 6u^2 - 2v^2$ सीधे. नई विकर्ण प्रविष्टियों पर ध्यान दें $6,
-2$ हैं \emph{नहीं}  eigenvalues $3, -1$ का $ B $: \omterm{def:b2:quadratic:def}{अनुरूपता}
पुनर्मूल्यांकन, केवल समानता स्पेक्ट्रा को संरक्षित करती है --- लेकिन संकेत
सहमत हूँ, जैसा कि सिल्वेस्टर का प्रमेय मांग करता है। (यहाँ आधार है
ऑर्थोगोनल लेकिन ऑर्थोनॉर्मल नहीं; द्वारा इसे सामान्य किया जा रहा है
$\frac{1}{\sqrt2}$ विकर्ण को इससे विभाजित करेंगे $2$ और
पुनर्प्राप्त करें eigenvalues.)
\end{example}

\begin{example}[ग्राम निर्धारक क्षेत्रफल मापते हैं]\label{ex:b2:quadratic:gramarea}
के लिए $ v_1, v_2$ यूक्लिडियन स्थान में, ग्राम मैट्रिक्स $ G =
\bigl(\langle v_i, v_j\rangle\bigr)$ पैकेज की लंबाई और
कोण; इसका सिद्ध संकुल \emph{क्षेत्र}:
\[
\det G = \norm{v_1}^2\norm{v_2}^2 - \langle v_1, v_2\rangle^2
= \norm{v_1}^2\norm{v_2}^2\bigl(1 - \cos^2\theta\bigr)
= \bigl(\norm{v_1}\,\norm{v_2}\sin\theta\bigr)^2 ,
\]
समांतर चतुर्भुज का वर्ग क्षेत्रफल $ v_1, v_2$ ---
और कॉची-श्वार्ज़ \omterm{def:b2:series:def}{बिल्कुल} यही कथन है $\det G \geq 0$.
काम किया गया उदाहरण: $ v_1 = (1, 2, 2)$, $ v_2 = (2, 1, -2)$ में
$\R^3$:
\[
G = \begin{pmatrix} 9 & 0\\ 0 & 9 \end{pmatrix},
\qquad
\det G = 81 :
\]
वेक्टर लंबाई के ऑर्थोगोनल हैं $3$, फैला हुआ ए
क्षेत्रफल का समांतर चतुर्भुज (यहाँ, एक वर्ग)। $\sqrt{81} = 9$.
समापन अंतर्दृष्टि: कोई क्रॉस उत्पाद नहीं और कोई आयाम नहीं-$3$ जादू
प्रयोग किया गया --- $\sqrt{\det G}$ पैमाने $ k $-आयामी मात्रा
में \emph{कोई} आयाम, सप्ताहांत का शुरुआती बिंदु
समस्या का भाग I और सतह-क्षेत्र इंटीग्रल बाद में
यह वॉल्यूम.
\end{example}

\section{गॉस में कमी और सिल्वेस्टर की जड़ता}

\begin{theorem}[गॉस में कमी]\label{thm:b2:quadratic:gauss}
प्रत्येक द्विघात रूप $ q $ एक परिमित-आयामी वास्तविक स्थान पर हो सकता है
लिखा हुआ\index{गॉस में कमी}
\[
q = \sum_{i=1}^{s} \ell_i^2 - \sum_{j=1}^{t} m_j^2 ,
\]
कहाँ $\ell_1, \dots, \ell_s, m_1, \dots, m_t $ रैखिक हैं
स्वतंत्र रैखिक रूप; \omterm{def:b2:funcseq:def}{समान रूप से}, कुछ आधार बनाता है
का मैट्रिक्स $ q $ प्रविष्टियों के साथ विकर्ण $+1$ ($ s $ समय), $-1$ ($ t $
समय), $0$.
\end{theorem}

\begin{proof}
निर्देशांक में चर की संख्या पर प्रेरण: $ q(x_1,
\dots, x_n)$.

\emph{केस 1: कुछ वर्ग दिखाई देता है}, गुणांक कहो $ a $ का
$ x_1^2$ शून्येतर है. सभी को समूहित करें $ x_1$-शर्तें और \omterm{def:b2:metric:complete}{पूरा} करें
वर्ग:
\[
q = a\Bigl(x_1 + \frac{1}{a}\,\lambda(x_2, \dots,
x_n)\Bigr)^{\!2} + q_1(x_2, \dots, x_n),
\]
कहाँ $\lambda $ रैखिक है और $ q_1$ शेष में द्विघात
चर: एक स्वतंत्र रूप अलग हो गया (इसमें शामिल है $ x_1$, द
अन्य नहीं), इंडक्शन पर लागू होता है $ q_1$, और $\pm $ संकेत आते हैं
के चिन्ह से $ a $ द्वारा पुनः स्केल करने के बाद $\sqrt{\abs a}$.

\emph{केस 2: कोई वर्ग नहीं, बल्कि कुछ क्रॉस टर्म}, कहना $ b\,x_1x_2$ साथ
$ b \neq 0$. पहचान का प्रयोग करें
\[
x_1x_2 = \tfrac14\bigl((x_1 + x_2)^2 - (x_1 - x_2)^2\bigr)
\]
समूहीकरण के बाद: लिखना $ q = b\,x_1x_2 + x_1\alpha + x_2\beta +
q_2$ (साथ $\alpha, \beta, q_2$ अन्य चरों में), एक जाँच करता है
\[
q = \frac{b}{4}\Bigl[\Bigl(x_1 + x_2 + \frac{\alpha +
\beta}{b}\Bigr)^{2} - \Bigl(x_1 - x_2 + \frac{\beta -
\alpha}{b}\Bigr)^{2}\Bigr] + \widetilde q ,
\]
साथ $\widetilde q $ मुक्त मे $ x_1, x_2$: दो स्वतंत्र रूप विभाजित हो गए
बंद, और प्रेरण समाप्त।

एकत्रित प्रपत्रों की स्वतंत्रता: बैचों को इस प्रकार क्रमबद्ध करें
उत्पादित. पहले बैच के प्रपत्र शामिल हैं $ x_1$ (केस 1)
या $ x_1, x_2$ (केस 2); बाद के सभी फॉर्म उनसे मुक्त हैं
चर. सभी एकत्रित प्रपत्रों का एक रैखिक संयोजन मान लीजिए
गायब हो जाता है. का गुणांक पढ़ना $ x_1$ (और $ x_2$): केवल
पहला बैच योगदान देता है, और उस बैच के भीतर एक या
दो रूप स्पष्ट रूप से स्वतंत्र हैं ($\ell $ अकेला; या $\ell \pm
m $ स्वतंत्र के साथ $\ell, m $): प्रथम-बैच गुणांक
गायब हो जाना. बैच को पट्टी करें और दोहराएं: साथ में प्रेरण द्वारा
बैच, सभी गुणांक गायब हो जाते हैं --- \omterm{def:b2:metric:complete}{पूरा} परिवार स्वतंत्र है,
त्रिकोणीयता स्पष्ट हो गई।
\end{proof}

\begin{theorem}[सिल्वेस्टर का जड़त्व का नियम]\label{thm:b2:quadratic:sylvester}
\omterm{def:b2:quadratic:adjoint}{जोड़}ी $(s, t)$ में \cref{thm:b2:quadratic:gauss} पर ही निर्भर करता है
$ q $, कटौती पर नहीं: यह है
\emph{हस्ताक्षर}\index{हस्ताक्षर} का $ q $.
इसके अतिरिक्त
\[
s = \max\{\dim F : q|_F \text{ positive definite}\},
\]
और \omterm{def:b2:quadratic:adjoint}{सममित} रूप से के लिए $ t $.
\end{theorem}

\begin{proof}
होने देना $ q = \sum_{i \leq s}\ell_i^2 - \sum_{j\leq t} m_j^2$ और चलो
$ F_+$ (पूर्व) की अवधि हो\omterm{def:b2:linalg:dual}{दोहरी} जिस पर वेक्टर $(\ell_i)$
निर्देशांक तक सीमित --- ठोस रूप से: \omterm{def:b2:metric:complete}{पूरा} करें
स्वतंत्र परिवार $(\ell_1, \dots, \ell_s, m_1, \dots, m_t)$
के एक आधार में \omterm{def:b2:linalg:dual}{दोहरी} $ E^*$, और चलो $(u_1, \dots, u_n)$ हो
का आधार $ E $ ये किसके निर्देशांक रूप हैं (द
पूर्व \omterm{def:b2:linalg:dual}{दोहरी} आधार: $\ell_i(u_k) = \delta_{ik}$ के लिए $ k \leq s $,
और बाद के रूप पहले वाले वैक्टर पर लुप्त हो जाते हैं)। तय करना $ F_+ =
\operatorname{Vect}(u_1, \dots, u_s)$: के लिए $ x = \sum_{i\leq
s}x_iu_i \in F_+$,
\[
\ell_i(x) = x_i,
\qquad m_j(x) = 0,
\qquad\text{so}\qquad
q(x) = \sum_{i\leq s}x_i^2 > 0 \quad (x \neq 0) :
\]
$ q|_{F_+}$ सकारात्मक निश्चित है और प्रदर्शन में अधिकतम है
है $\geq s $. इसके विपरीत, चलो $ F $ किसी भी उपस्थान के साथ हो $ q|_F $
सकारात्मक निश्चित, और $ G = \{x : \ell_1(x) = \dots = \ell_s(x) =
0\}$, संहिताकरण का $\leq s $; पर $ G $, $ q(x) = -\sum m_j^2 \leq 0$.
तब $ F \cap G = \{0\}$ (वहां एक शून्येतर सदिश होगा $ q > 0$
और $ q \leq 0$), इसलिए $\dim F \leq \dim E - \dim G \leq s $. इसलिए
अधिकतम बराबर $ s $ के लिए \emph{प्रत्येक} कमी: $ s $ आंतरिक है, और $ t
= \operatorname{rk} q - s $ वैसे ही।
\end{proof}

\begin{example}\label{ex:b2:quadratic:gaussexample}
$ q(x, y, z) = xy + yz + zx $ (कोई वर्ग नहीं). साथ $ x_1 = x $, $ x_2 =
y $: $ q = xy + z(x + y)$, और दो-वर्ग पहचान देता है
\[
q = \tfrac14(x + y + 2z)^2 - \tfrac14(x - y)^2 - z^2 ,
\]
(जांचने के लिए विस्तार करें)। तीन स्वतंत्र प्रपत्र: \omterm{thm:b2:structures:signature}{हस्ताक्षर} $(1, 2)$,
रैंक $3$. एक सकारात्मक दिशा, दो नकारात्मक: ``प्रकाश शंकु''
इस रूप की ज्यामिति.
\end{example}

\begin{example}[एक पतित रूप, पूर्ण रूप से छोटा]\label{ex:b2:quadratic:degenerate}
$ q(x, y, z) = xy + yz $ पर $\R^3$: कोई वर्ग नहीं, इसलिए केस 2 के साथ
समूहीकरण $ q = y(x + z)$. उत्पाद पर दो-वर्ग की पहचान
स्वतंत्र रूपों का $ y $ और $ x + z $:
\[
q = \frac14\bigl(y + x + z\bigr)^2 -
\frac14\bigl(y - x - z\bigr)^2 .
\]
दो रैखिक रूप $ y + x + z $ और $ y - x - z $ हैं
स्वतंत्र (उनका अंतर है $2(x+z)$, उनका योग $2y $), तो
सिल्वेस्टर पढ़ता है: \omterm{thm:b2:structures:signature}{हस्ताक्षर} $(1, 1)$, रैंक $2$ ---
\emph{पतित}. ध्रुवीय रूप की गुठली पाई जाती है
सुलझाना $\varphi(v, \cdot) = 0$: मैट्रिक्स के साथ
$\frac12\begin{pmatrix} 0&1&0\\ 1&0&1\\ 0&1&0\end{pmatrix}$,
कर्नेल है $\{y = 0,\ x + z = 0\} = \R\,(1, 0, -1)$, द
जिस दिशा में $ q $ कुछ नहीं देखता. समापन अंतर्दृष्टि: रैंक
कमी गॉस में ``चरों के समाप्त होने'' के रूप में दिखाई देती है
--- कटौती से तीन में से केवल दो वर्ग उत्पन्न हुए
आयाम, और लुप्त आयाम \omterm{def:b2:series:def}{बिल्कुल} कर्नेल है।
\end{example}

\begin{example}[एक फॉर्म, हस्ताक्षर तक पहुंचने के दो रास्ते]\label{ex:b2:quadratic:tworoads}
$ q(x, y, z) = 2x^2 + 2y^2 + 2z^2 + 2xy + 2yz $, मैट्रिक्स
$\begin{pmatrix} 2 & 1 & 0\\ 1 & 2 & 1\\ 0 & 1 & 2
\end{pmatrix}$. \emph{रोड 1, गॉस:} में पूर्ण वर्ग
\omterm{def:b2:structures:generated}{आदेश},
\[
q = 2\Bigl(x + \frac y2\Bigr)^{\!2} + \frac32 y^2 + 2yz + 2z^2
= 2\Bigl(x + \frac y2\Bigr)^{\!2}
+ \frac32\Bigl(y + \frac{2z}{3}\Bigr)^{\!2} + \frac43 z^2 :
\]
स्वतंत्र प्रपत्रों, \omterm{thm:b2:structures:signature}{हस्ताक्षर} पर तीन सकारात्मक वर्ग $(3,
0)$: सकारात्मक निश्चित. \emph{सड़क 2, eigenvalues:} मैट्रिक्स
त्रिविकर्ण है $2I + N $ साथ $ N $ पड़ोसी मैट्रिक्स; इसका
eigenvalues हैं $2 + \sqrt2$, $2$, $2 - \sqrt2$ (जाँचें
eigenvectors $(1, \pm\sqrt2, 1)$ और $(1, 0, -1)$), सभी
सकारात्मक: वही फैसला, द्वारा
\cref{cor:b2:quadratic:principalaxes}. गॉस तेज़ है;
eigenvalues और कहें (वे मुख्य अक्ष और देते हैं
के चरम मूल्य $ q $ गोले पर)। समापन अंतर्दृष्टि:
सकारात्मक धुरी $2, \frac32, \frac43$ गॉस के \omterm{def:b2:series:def}{बिल्कुल} वही हैं
अनुपात $\frac{\Delta_k}{\Delta_{k-1}}$ अग्रणी प्राचार्य का
नाबालिग ($\Delta_1 = 2$, $\Delta_2 = 3$, $\Delta_3 = 4$) --- द
सप्ताहांत की समस्या आम तौर पर यह साबित करती है।
\end{example}

\begin{method}[हस्ताक्षर की गणना: तीन मार्ग]\label{met:b2:quadratic:signature}
\begin{enumerate}
  \item \emph{गॉस} (हमेशा काम करता है, हाथ से सबसे तेज़): \omterm{def:b2:metric:complete}{पूरा}
        वर्ग क्रम में, केस 2 जब कोई वर्ग उपलब्ध नहीं है;
        चिन्हों को गिनें. जांचें कि एकत्रित रैखिक रूप है
        स्वतंत्र हैं --- चर की तुलना में कम रूप का अर्थ है ए
        कर्नेल (\cref{ex:b2:quadratic:degenerate}).
  \item \emph{नाबालिगों का नेतृत्व करना} (निश्चितता परीक्षण के लिए): सभी
        $\Delta_k > 0$ यदि सकारात्मक निश्चित है
        (\cref{exo:b2:quadratic:8}); धुरी
        $\Delta_k/\Delta_{k-1}$ यहां तक कि गॉस भी दें
        गुणांक (सप्ताहांत समस्या)। यदि कुछ चुपचाप विफल हो जाता है
        $\Delta_k = 0$: मार्ग 1 पर वापस आएं।
  \item \emph{eigenvalues} (सबसे जानकारीपूर्ण, सबसे महंगा):
        के लक्षण \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम}
        (\cref{cor:b2:quadratic:principalaxes}); पैदावार भी होती है
        के प्रमुख अक्ष और चरम मान $ q $ पर
        इकाई क्षेत्र. जब ईजेनस्ट्रक्चर हो तो इसे प्राथमिकता दें
        वैसे भी जरूरत है.
\end{enumerate}
\end{method}

\section{वर्णक्रमीय प्रमेय}

अब चलो $ E $ यूक्लिडियन बनें (आंतरिक उत्पाद $\langle\cdot,\cdot\rangle $,
वर्ष 1 खंड).

\begin{definition}[जोड़; सममित एंडोमोर्फिज्म]\label{def:b2:quadratic:adjoint}
के लिए $ u \in \mathcal{L}(E)$, द \emph{जोड़}\index{जोड़}
$ u^*$ के साथ अद्वितीय एंडोमोर्फिज्म है
\[
\langle u(x), y\rangle = \langle x, u^*(y)\rangle
\qquad (x, y \in E);
\]
दैहिक आधार पर, $\operatorname{Mat}(u^*) =
\operatorname{Mat}(u)^{\mathsf T}$. $ u $ है
\emph{सममित}\index{सममित} (स्वयं-सहायक) जब
$ u^* = u $ --- \omterm{def:b2:funcseq:def}{समान रूप से} इसका मैट्रिक्स ऑर्थोनॉर्मल आधार पर है
सममित.
\end{definition}

\begin{proof}[सहायक का अस्तित्व और विशिष्टता]
तय के लिए $ y $, फार्म $ x \mapsto \langle u(x), y\rangle $ है
रैखिक, इसलिए आकार का (सीमित आयाम)। $\langle x,
z_y\rangle $ एक अद्वितीय के लिए $ z_y $ --- वो नक्शा $ y \mapsto z_y =:
u^*(y)$ विशिष्टता द्वारा रैखिक होना। मैट्रिक्स पहचान:
$\langle u(e_i), e_j\rangle $ दोनों तरह से पढ़ें.
\end{proof}

\begin{example}[जोड़ आंतरिक उत्पाद पर निर्भर करता है]\label{ex:b2:quadratic:weightedadjoint}
पर $\R^2$ ले लो \emph{भारित} अंदरूनी प्रोडक्ट $\langle x,
y\rangle_D = x_1y_1 + 2x_2y_2$ (मैट्रिक्स $ D =
\operatorname{diag}(1,2)$) और $ u $ मैट्रिक्स का $ A =
\begin{pmatrix} 0 & 1\\ 0 & 0\end{pmatrix}$ विहित में
आधार. से $\langle u(x), y\rangle_D = (Ax)^{\mathsf T}Dy =
x^{\mathsf T}(A^{\mathsf T}D)y $ और $\langle x, u^*(y)\rangle_D
= x^{\mathsf T}(DA^*)y $, द जोड़का मैट्रिक्स है
\[
A^* = D^{-1}A^{\mathsf T}D
= \begin{pmatrix} 1 & 0\\ 0 & \tfrac12\end{pmatrix}
\begin{pmatrix} 0 & 0\\ 1 & 0\end{pmatrix}
\begin{pmatrix} 1 & 0\\ 0 & 2\end{pmatrix}
= \begin{pmatrix} 0 & 0\\ \tfrac12 & 0\end{pmatrix}
\neq A^{\mathsf T} .
\]
विवेक की जाँच जारी है $ x = (1,0)$, $ y = (0,1)$:
\[
\langle u(x), y\rangle_D = \langle (0,0), y\rangle_D = 0,
\quad
\langle x, u^*(y)\rangle_D
= \langle(1,0), (0,\tfrac12)\rangle_D = 0 ;
\]
पर $ x = (0,1)$, $ y = (1,0)$:
\[
\langle u(x), y\rangle_D = \langle(1,0),(1,0)\rangle_D = 1,
\quad
\langle x, u^*(y)\rangle_D =
\langle(0,1),(0,\tfrac12)\rangle_D = 1 .
\] समापन अंतर्दृष्टि: ``$\operatorname{Mat}(u^*)
= \operatorname{Mat}(u)^{\mathsf T}$'' के बारे में एक कथन है
\emph{ऑर्थोनॉर्मल} केवल आधार; सामान्य तौर पर मीट्रिक $ D $
हस्तक्षेप करता है, ठीक वैसे ही जैसे कि एक साथ कमी लाने में
सप्ताहांत की समस्या.
\end{example}

\begin{theorem}[वर्णक्रमीय प्रमेय]\label{thm:b2:quadratic:spectral}
होने देना $ u $ एक हो \omterm{def:b2:quadratic:adjoint}{सममित} एंडोमोर्फिज्म यूक्लिडियन स्थान का $ E $. तब
$ E $ एक है \emph{का ऑर्थोनॉर्मल आधार eigenvectors} का $ u $; सभी
eigenvalues वास्तविक हैं, और eigenspaces विशिष्ट के लिए eigenvalues हैं
ओर्थोगोनल. मैट्रिक्स फॉर्म: हर वास्तविक \omterm{def:b2:quadratic:adjoint}{सममित} मैट्रिक्स $ A $ लिखते हैं
\[
A = P\,D\,P^{\mathsf T},
\qquad P \text{ orthogonal } (P^{\mathsf T} P = I),\ D
\text{ diagonal}.\index{वर्णक्रमीय प्रमेय}
\]
\end{theorem}

\begin{proof}
\emph{एक आइजन्वेक्टर मौजूद है.} समारोह $ x \mapsto \langle u(x),
x\rangle $ है \omterm{def:b2:metric:continuity}{निरंतर} इकाई क्षेत्र पर $ S $ का $ E $, जो है
\omterm{def:b2:metric:compact}{कॉम्पैक्ट} (परिमित आयाम, \cref{thm:b2:nvs:finitedim}): यह प्राप्त होता है
यह अधिकतम है $\lambda $ कुछ $ a \in S $. हम दावा करते हैं $ u(a) = \lambda
a $. किसी के लिए $ y \perp a $ साथ $\norm y = 1$ और $ t \in \R $, द
वेक्टर $ x_t = \frac{a + ty}{\sqrt{1 + t^2}}$ पर स्थित है $ S $
($\norm{a + ty}^2 = 1 + t^2$ पाइथागोरस द्वारा); विस्तार
अधिकतम फ़ंक्शन,
\[
g(t) = \langle u(x_t), x_t\rangle
= \frac{\langle u(a), a\rangle + 2t\langle u(a), y\rangle +
t^2\langle u(y), y\rangle}{1 + t^2}
\]
(की समरूपता $ u $ दो क्रॉस टर्म्स को मिला दिया: $\langle u(a),
y\rangle = \langle a, u(y)\rangle = \langle u(y), a\rangle $).
$ g $ का एक भिन्न कार्य है $ t $ अधिकतम के साथ $ t =
0$; पर भागफल नियम $0$ देता है
\[
g'(0) = \frac{2\langle u(a), y\rangle\cdot 1 - \langle u(a),
a\rangle\cdot 0}{1} = 2\langle u(a), y\rangle = 0 .
\]
इसलिए $ u(a)$ संपूर्ण हाइपरप्लेन के लिए ऑर्थोगोनल है $ a^\perp $: $ u(a)
\in (a^{\perp})^{\perp} = \R a $, यानी\ $ u(a) = \mu a $; और $\mu =
\langle u(a), a\rangle = \lambda $.

\emph{प्रेरण।} ओर्थोगोनल पूरक $ F = a^\perp $ है
$ u $-स्थिर: के लिए $ x \perp a $, $\langle u(x), a\rangle = \langle x,
u(a)\rangle = \lambda\langle x, a\rangle = 0$. प्रतिबंध
$ u|_F $ है \omterm{def:b2:quadratic:adjoint}{सममित} प्रेरित आंतरिक उत्पाद के लिए; पर प्रेरण द्वारा
आयाम, $ F $ एक ऑर्थोनॉर्मल ईजेनबैसिस है; आगे जोड़ते $ a $.

\emph{पूरक.} eigenvalues वास्तविक संख्याएँ हैं $\langle u(e),
e\rangle $ ईजेनबेसिस पर. की रूढ़िवादिता eigenspaces: $ u(x) =
\lambda x $, $ u(y) = \mu y $ देना $\lambda\langle x, y\rangle =
\langle u(x), y\rangle = \langle x, u(y)\rangle = \mu\langle x,
y\rangle $, इसलिए $\langle x, y\rangle = 0$ कब $\lambda \neq \mu $.
मैट्रिक्स फॉर्म: के कॉलम $ P $ = ऑर्थोनॉर्मल ईजेनबैसिस।
\end{proof}

\begin{example}[एक पूर्ण वर्णक्रमीय दौड़]\label{ex:b2:quadratic:fullrun}
ओर्थोगोनल रूप से विकर्णित करें $ A = \begin{pmatrix} 1 & 2\\ 2 &
1\end{pmatrix}$. अभिलक्षणिक बहुपद $(1 - \lambda)^2 -
4$: eigenvalues $3$ और $-1$. आइगेनवेक्टर: $(A - 3I)v = 0$
देता है $ v_1 = \frac{1}{\sqrt2}(1,1)$; $(A + I)v = 0$ देता है $ v_2
= \frac{1}{\sqrt2}(1,-1)$ --- ऑर्थोगोनल, जैसे
\cref{thm:b2:quadratic:spectral} बिना गारंटी
गणना. साथ $ P = (v_1\ v_2)$ (द्वारा एक रोटेशन
$\frac\pi4$):
\[
P^{\mathsf T}AP = \begin{pmatrix} 3 & 0\\ 0 & -1
\end{pmatrix},
\qquad
x^2 + 4xy + y^2 = 3u^2 - v^2
\quad\text{in the rotated frame} .
\]
तो का स्वरूप \cref{exo:b2:quadratic:1} अतिपरवलय-प्रकार है
प्रपत्र: \omterm{thm:b2:structures:signature}{हस्ताक्षर} $(1,1)$, इसके अनुरूप \omterm{thm:b2:quadratic:gauss}{गॉस में कमी}
$(x + 2y)^2 - 3y^2$ --- अलग-अलग वर्ग, एक ही \omterm{thm:b2:structures:signature}{हस्ताक्षर}, जैसे
सिलवेस्टर की मांग. समापन अंतर्दृष्टि: गॉस ने उत्तर दिया
तेज़, लेकिन वर्णक्रमीय मार्ग भी इकाई पर रिपोर्ट करता है
वृत्त $ q $ पर्वतमाला \omterm{def:b2:series:def}{बिल्कुल} खत्म हो गई है $\intcc{-1}{3}$, साथ ही प्राप्त किया
$ v_2$ और $ v_1$: अतिरिक्त कार्य ज्यामिति खरीदता है।
\end{example}

\begin{corollary}[प्रधान अक्ष; सकारात्मकता परीक्षण]\label{cor:b2:quadratic:principalaxes}
\begin{enumerate}
  \item प्रत्येक द्विघात रूप $ q $ यूक्लिडियन स्थान पर विकर्ण होता है
        कुछ में \emph{ऑर्थोनॉर्मल} आधार: $ q(x) = \sum_i \lambda_i
        x_i^2$ साथ $\lambda_i $  eigenvalues की \omterm{def:b2:quadratic:adjoint}{सममित}
        का मैट्रिक्स $ q $; \omterm{thm:b2:structures:signature}{हस्ताक्षर} सकारात्मक और नकारात्मक मायने रखता है
        eigenvalues.
  \item ए \omterm{def:b2:quadratic:adjoint}{सममित} मैट्रिक्स सकारात्मक अर्धनिश्चित है (resp.\
        निश्चित) यदि यह सब है eigenvalues हैं $\geq 0$ (सम्मान.\ $>
        0$); और फिर रेले भागफल के चरम मूल्य
        हैं
        \[
        \min_{\norm x = 1} \langle Ax, x\rangle = \lambda_{\min},
        \qquad
        \max_{\norm x = 1} \langle Ax, x\rangle = \lambda_{\max} .
        \]
\end{enumerate}
\end{corollary}

\begin{proof}
(1)लिखो $ q(x) = \langle A x, x\rangle $ साथ $ A $ \omterm{def:b2:quadratic:adjoint}{सममित} (द
का मैट्रिक्स $ q $ ऑर्थोनॉर्मल आधार पर); विकर्ण बनाना $ A $ द्वारा
वर्णक्रमीय प्रमेय: के लिए $ x = \sum x_ie_i $ ऑर्थोनॉर्मल में
ईजेनबेसिस,
\[
q(x) = \Bigl\langle \sum_i \lambda_ix_ie_i,\ \sum_j
x_je_j\Bigr\rangle = \sum_i \lambda_i x_i^2
\]
(ऑर्थोनॉर्मलिटी क्रॉस टर्म्स को खत्म कर देती है)। $\pm $ के लक्षण
$\lambda_i $ सिल्वेस्टर द्वारा \omterm{thm:b2:structures:signature}{हस्ताक्षर} गिनें: प्रत्येक को पुनः स्केल करना
द्वारा समन्वयित करें $\sqrt{\abs{\lambda_i}}$ एक प्रदर्शित करता है गॉस
कमी स्वतंत्र रूपों के साथ.

(2) ईजेनबासिस में, $\langle Ax, x\rangle = \sum \lambda_i
x_i^2$, बीच में फंसा हुआ $\lambda_{\min}\norm x^2$ और
$\lambda_{\max}\norm x^2$, संगत पर समानता के साथ
eigenvectors; सभी की सकारात्मकता eigenvalues इस प्रकार समतुल्य है
रूप की सकारात्मकता के लिए.
\end{proof}

\begin{example}\label{ex:b2:quadratic:spectralexample}
$ A = \begin{pmatrix} 2 & 1\\ 1 & 2 \end{pmatrix}$: eigenvalues $3$
(आइजन्वेक्टर $\frac{1}{\sqrt2}(1,1)$) और $1$
($\frac{1}{\sqrt2}(1,-1)$). द्विघात रूप $2x^2 + 2xy + 2y^2$
बन जाता है $3X^2 + Y^2$ घुमाए गए ऑर्थोनॉर्मल फ्रेम में: एक दीर्घवृत्त
प्रमुख अक्षों की गणना की गई। \omterm{thm:b2:quadratic:gauss}{गॉस में कमी} एक विकर्ण तक पहुँचता है
रूप भी, लेकिन केवल वर्णक्रमीय प्रमेय ही उस तक पहुंचता है \emph{बिना
विकृत लंबाई}.
\end{example}

\begin{example}[एक दीर्घवृत्त पूरी तरह से पहचाना गया]\label{ex:b2:quadratic:ellipserun}
कौन सा वक्र है $5x^2 + 4xy + 2y^2 = 6$? गणित का सवाल
$\begin{pmatrix} 5 & 2\\ 2 & 2\end{pmatrix}$ है \omterm{def:b2:reduction:charpoly}{विशेषता
बहुपद} $\lambda^2 - 7\lambda + 6 = (\lambda - 1)(\lambda -
6)$: eigenvalues $1$ और $6$, दोनों सकारात्मक --- एक दीर्घवृत्त।
ऑर्थोनॉर्मल eigenvectors: के लिए $\lambda = 1$, हल करना
$\begin{pmatrix} 4 & 2\\ 2 & 1\end{pmatrix}v = 0$: $ v_1 =
\frac{1}{\sqrt5}(1, -2)$; के लिए $\lambda = 6$: $ v_2 =
\frac{1}{\sqrt5}(2, 1)$. घुमाए गए निर्देशांक में $(X, Y)$
साथ में $(v_2, v_1)$ समीकरण बन जाता है
\[
6X^2 + Y^2 = 6,
\qquad\text{i.e.}\qquad
X^2 + \frac{Y^2}{6} = 1 :
\]
अर्ध-अक्ष $1$ (साथ में $ v_2$) और $\sqrt6$ (साथ में $ v_1$).
समापन अंतर्दृष्टि: खुरदरा आकार नि:शुल्क था --- $\det
= 6 > 0$ और सकारात्मक ट्रेस किसी से भी पहले एक दीर्घवृत्त की घोषणा करता है
आइजन्वेक्टर गणना की जाती है --- लेकिन केवल वर्णक्रमीय प्रमेय
अक्षों की दिशा और लंबाई प्रदान करता है, अर्थात\ वास्तविक
ज्यामिति.
\end{example}

\begin{example}[क्षेत्र पर चरम, स्पेक्ट्रम से पढ़ें]\label{ex:b2:quadratic:sphereextremes}
के चरम मूल्य क्या हैं $ q(x,y,z) = 2xy + 2yz + 2zx $ पर
इकाई क्षेत्र? इसका मैट्रिक्स (सभी का विकर्ण बंद)।
\cref{exo:b2:quadratic:2}) है eigenvalues $2$ और $-1$
(डबल), तो द्वारा \cref{cor:b2:quadratic:principalaxes}~(2):
\[
\max_{\norm v = 1} q(v) = 2
\ \text{ at } v = \tfrac{1}{\sqrt3}(1,1,1),
\qquad
\min_{\norm v = 1} q(v) = -1
\ \text{ on the circle } x + y + z = 0 .
\]
कोई कैलकुलस नहीं, कोई लैग्रेंज गुणक नहीं: वर्णक्रमीय प्रमेय
इस बाधित अनुकूलन को पूरी तरह से हल करता है --- और प्रदर्शित करता है
अधिकतमकर्ता. समापन अंतर्दृष्टि: गुणक के साथ तुलना करें
डिफरेंशियल-कैलकुलस अध्याय की विधि, जो पाता है
अधिक कार्य के साथ समान महत्वपूर्ण बिंदु; के लिए \emph{द्विघात}
क्षेत्रों पर उद्देश्य, स्पेक्ट्रा शाही सड़क (सप्ताहांत) हैं
हर्मिटियन अध्याय की समस्या संपूर्ण का निर्माण करती है
कूरेंट--इस अवलोकन पर फिशर सिद्धांत)।
\end{example}

\begin{remark}[सामान्य ख़तरे]
\emph{(मैं) \omterm{def:b2:quadratic:def}{अनुरूपता} समानता नहीं है:} के लिए आधार परिवर्तन
एक प्रपत्र द्वारा कार्य करता है $ P^{\mathsf T}BP $, नहीं $ P^{-1}BP $; eigenvalues
हैं \emph{नहीं} ए के अपरिवर्तनीय द्विघात रूप ($ I $ और $4I $
के माध्यम से सर्वांगसम हैं $ P = 2I $) ---सिर्फ उनकी निशानियाँ हैं
(सिल्वेस्टर)। की बात करें eigenvalues एक फॉर्म का केवल एक बार
आंतरिक उत्पाद स्थिर है. \emph{(ii) सकारात्मक प्रविष्टियाँ सिद्ध होती हैं
कुछ नहीं:} $\begin{pmatrix} 1 & 2\\ 2 & 1\end{pmatrix}$ सब कुछ है
प्रविष्टियाँ सकारात्मक फिर भी \omterm{thm:b2:structures:signature}{हस्ताक्षर} $(1,1)$ ($\det = -3$);
इसके विपरीत एक सकारात्मक निश्चित मैट्रिक्स नकारात्मक हो सकता है
ऑफ-विकर्ण प्रविष्टियाँ (\cref{ex:b2:quadratic:tworoads} स्थानांतरित:
$2I - N $ \omterm{def:b2:funcseq:def}{समान रूप से} काम करता है)। उपयोग
\cref{met:b2:quadratic:signature}. \emph{(iii) आश्रित
वर्ग:} लिखना $ q = \ell_1^2 - \ell_2^2$ कुछ नहीं कहता अगर
$\ell_1, \ell_2$ आनुपातिक हैं --- $ x^2 + 2xy + y^2 =
(x+y)^2$ रैंक है $1$, नहीं $2$; हमेशा स्वतंत्रता की जांच करें
\omterm{thm:b2:structures:signature}{हस्ताक्षर} पढ़ने से पहले. \emph{(iv) क्षेत्र की चरम सीमाएँ
बिना सघनता:} रेले की सीमा
\cref{cor:b2:quadratic:principalaxes} प्राप्त होते हैं क्योंकि
गोला है \omterm{def:b2:metric:compact}{कॉम्पैक्ट}; पर \omterm{def:b2:metric:topology}{खुला} गेंद या संपूर्ण स्थान, ए
अनिश्चित रूप में न तो अधिकतम होता है और न ही न्यूनतम।
\end{remark}

\begin{remark}[इसका उपयोग कहां किया जाता है]
वर्णक्रमीय प्रमेय इसका सबसे अधिक निर्यात किया जाने वाला परिणाम है
पुस्तक: आँकड़े इसके साथ सहप्रसरण मैट्रिक्स को विकर्णित करते हैं
(प्रमुख घटक विश्लेषण), यांत्रिकी सामान्य मोड निकालता है
इससे दोलन की (एक साथ कमी)
सप्ताहांत समस्या), संख्यात्मक विश्लेषण चॉल्स्की और बनाता है
इस पर एकवचन मूल्य अपघटन (समान समस्या), और अगला
अध्याय इसे जटिल हर्मिटियन स्थानों तक पहुँचाता है। वर्ष 3
आयतन इसके अनंत-आयामी अवतार को सिद्ध करता है \omterm{def:b2:metric:compact}{कॉम्पैक्ट}
स्वयं adjoint ऑपरेटर्स, जहां परिमित-आयामी प्रमाण है
सघनता तर्क पूरी कहानी बन जाता है.
\end{remark}

\begin{remark}[इस खंड के भीतर परिप्रेक्ष्य]
द्विघात रूप पुस्तक 4 के शेष भाग को तीन के अंतर्गत पिरोएं
भेष बदलना। जैसा \emph{हेस्सियन}: अंतर-कैलकुलस
अध्याय के \omterm{thm:b2:structures:signature}{हस्ताक्षर} द्वारा महत्वपूर्ण बिंदुओं को वर्गीकृत किया गया है
दूसरे क्रम का रूप, इसलिए सिल्वेस्टर की अपरिवर्तनशीलता ही इसे बनाती है
``काठी'' एक सुपरिभाषित शब्द है। जैसा \emph{ऊर्जा}: द
अंतर-समीकरण अध्याय के ऑसिलेटर ले जाते हैं
द्विघात ऊर्जा $\frac12x'^2 + \frac12\omega^2x^2$, और
इस अध्याय की सप्ताहांत समस्या का एक साथ निवारण है
\omterm{def:b2:series:def}{बिल्कुल} सामान्य मोड का निष्कर्षण। जैसा \emph{ज्यामिति}:
इस अध्याय के शांकव चतुर्भुज सतहों में विकसित होते हैं
ज्यामिति अध्याय, जहां ए का दूसरा मौलिक रूप
सतह --- ए द्विघात रूप प्रत्येक स्पर्शरेखा तल पर --- है
\omterm{thm:b2:structures:signature}{हस्ताक्षर} यह तय करते हैं कि सतह कटोरे की तरह मुड़ती है या नहीं
एक काठी. अगला, हर्मिटियन अध्याय, पूरे स्कोर को दोहराता है
खत्म $\C $.
\end{remark}

एक ऑर्थोगोनल प्रक्षेपण का मैट्रिक्स। अधिक \omterm{def:b2:funcseq:series}{सामान्यतः}, संबंधित करें

\begin{exercise}[$\star\star $]\label{exo:b2:quadratic:5}

का $\mathcal{M}_n(\R)$ \emph{(बंद: की पूर्वछवि $\{I\}$ एक के तहत

\begin{exercise}[$\star\star $]\label{exo:b2:quadratic:6}
(वर्गमूल) चलो $ A $ होना \omterm{def:b2:quadratic:adjoint}{सममित} सकारात्मक अर्धनिश्चित. निर्माण
ए \omterm{def:b2:quadratic:adjoint}{सममित} सकारात्मक अर्धनिश्चित $ B $ साथ $ B^2 = A $, और साबित करो
वह यह है \emph{अद्वितीय} \emph{(अस्तित्व: का वर्गमूल लें

के साथ आवागमन करता है $ A = B^2$, इसलिए इसे सुरक्षित रखता है eigenspaces --- कम करें
प्रत्येक पर अदिश मामला)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:quadratic:7}

(वर्णक्रमीय त्रिज्या), और गणना करें $\vertiii{A}_2$ के लिए $ A =

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:quadratic:8}
(सिल्वेस्टर की कसौटी) चलो $ A $ स्वाभाविक रहें \omterm{def:b2:quadratic:adjoint}{सममित} अग्रणी के साथ
प्रमुख अवयस्क $\Delta_1, \dots, \Delta_n $ (निर्धारकों का
शीर्ष-बाएँ ब्लॉक)। साबित करें कि $ A $ सकारात्मक निश्चित है यदि और केवल
यदि सभी $\Delta_k > 0$. \emph{(के लिए $\Rightarrow $: ए के प्रतिबंध
निश्चित रूप निश्चित हैं, और सिद्ध एक सकारात्मक का
निश्चित मैट्रिक्स --- इसका उत्पाद eigenvalues --- सकारात्मक है।
के लिए $\Leftarrow $: पर शामिल करें $ n $; शीर्ष-बाएँ $(n-1)$-ब्लॉक है
सकारात्मक निश्चित, उस उपस्थान पर रूप को विकर्णित करें और
अंतिम चर में वर्ग \omterm{def:b2:metric:complete}{पूरा} करें; आखिरी का संकेत
विकर्ण प्रविष्टि द्वारा नियंत्रित किया जाता है $\det A = \Delta_n > 0$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:quadratic:9}
(क्यूरेंट--फिशर, दूसरा \omterm{def:b2:reduction:eigen}{eigenvalue}) होने देना $ u $ होना \omterm{def:b2:quadratic:adjoint}{सममित} साथ
eigenvalues $\lambda_1 \geq \lambda_2 \geq \dots \geq \lambda_n $.
सिद्ध करना
\[
\lambda_2 = \min_{\substack{H \text{ hyperplane}}}\;
\max_{\substack{x \in H,\ \norm x = 1}} \langle u(x), x\rangle .
\]
\emph{(के लिए $\leq $: कोई भी हाइपरप्लेन मिलता है $2$-विमान द्वारा फैलाया गया
शीर्ष दो eigenvectors. के लिए $\geq $: चुनना $ H =
(e_1)^{\perp}$.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:quadratic:10}
की रैंक और \omterm{thm:b2:structures:signature}{हस्ताक्षर} निर्धारित करें $ q(x_1, \dots, x_n) = \sum_{i <
j} x_ix_j $ पर $\R^n $ ($ n \geq 2$), दो तरीकों से: बीजगणित द्वारा
पहचान $2q = \bigl(\sum x_i\bigr)^2 - \sum x_i^2$ एक साथ
के प्रतिबंध के साथ $ q $ हाइपरप्लेन के लिए $\sum x_i = 0$;
और गणना करके eigenvalues इसके मैट्रिक्स का $\frac12(J - I)$,
कहाँ $ J $ ऑल-वन्स मैट्रिक्स है।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:quadratic:11}
होने देना $ A $, $ B $ स्वाभाविक रहें \omterm{def:b2:quadratic:adjoint}{सममित} साथ $ B $ सकारात्मक अर्धनिश्चित.
सिद्ध करना
\[
\lambda_{\min}(A)\operatorname{tr} B
\;\leq\;
\operatorname{tr}(AB)
\;\leq\;
\lambda_{\max}(A)\operatorname{tr} B .
\]
\emph{(लिखना $ B = C^{\mathsf T}C $ और $\operatorname{tr}(AB) =
\sum_i \langle A c_i, c_i\rangle $ स्तंभों के ऊपर $ c_i $ का
$ C^{\mathsf T}$.)} विशेष रूप से $\operatorname{tr}(AB) \geq 0$
जब दोनों सकारात्मक अर्धनिश्चित हों।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:quadratic:12}
पर $ E = \mathcal{M}_n(\R)$, विचार करना $ q(M) =
\operatorname{tr}(M^2)$.
\begin{enumerate}
  \item बताते हैं कि $ q $ एक है द्विघात रूप ध्रुवीय रूप के साथ
        $\varphi(M, N) = \operatorname{tr}(MN)$.
  \item दिखाओ कि \omterm{def:b2:quadratic:adjoint}{सममित} और एंटीसिमेट्रिक मैट्रिसेस बनते हैं
        $\varphi $-ऑर्थोगोनल उपस्थान जिस पर $ q $ है
        क्रमशः सकारात्मक निश्चित और नकारात्मक निश्चित
        \emph{(गणना करें $\operatorname{tr}(M^2)$ प्रवेशवार में
        प्रत्येक मामला)}.
  \item निष्कर्ष: $ q $ \omterm{thm:b2:structures:signature}{हस्ताक्षर} है $\bigl(\frac{n(n+1)}{2},
        \frac{n(n-1)}{2}\bigr)$ और रैंक $ n^2$.
\end{enumerate}
\end{exercise}

\section{समस्या: चोलेस्की, हैडामर्ड और ध्रुवीय
विघटन}

\begin{problem}\label{pb:b2:quadratic:1}
वर्णक्रमीय प्रमेय एक सूक्ष्मदर्शी है; यह समस्या इसे एक के रूप में उपयोग करती है
फ़ैक्टरी. ग्राम मैट्रिस से हम निर्माण करते हैं \emph{चोलेस्की
गुणनखंडीकरण}\index{चोलेस्की
गुणनखंडीकरण} (और पहचानें
गॉस अवयस्कों के अनुपात के रूप में धुरी बनाता है), फिर सिद्ध करें \emph{हैडामर्ड का
असमानता}\index{हैडामर्ड का
असमानता} पर निर्धारकों, निर्माण
द \emph{ध्रुवीय अपघटन}\index{ध्रुवीय अपघटन} $ A =
QS $ और यह \emph{एकल मूल्य अपघटन}, विमान को वर्गीकृत करें
शांकव, और दो रूपों की एक साथ कमी के साथ समाप्त होता है ---
दोलन के सामान्य तरीकों के पीछे का प्रमेय। लगातार, $ E =
\R^n $ अपने मानक आंतरिक उत्पाद के साथ।

\medskip
\noindent\textbf{Part I --- Gram matrices and Cholesky.} के लिए
वेक्टर $ v_1, \dots, v_n \in E $, उनका \emph{ग्राम मैट्रिक्स} है
$ G = \bigl(\langle v_i, v_j\rangle\bigr)_{i,j}$.
\begin{enumerate}
  \item बताते हैं कि $ G $ है \omterm{def:b2:quadratic:adjoint}{सममित} सकारात्मक अर्धनिश्चित, और
        सकारात्मक निश्चित यदि और केवल यदि $(v_1, \dots, v_n)$ है
        रैखिक रूप से स्वतंत्र \emph{(गणना करें $ X^{\mathsf T}GX $)}.
  \item इसके विपरीत, यह दिखाएँ कि प्रत्येक \omterm{def:b2:quadratic:adjoint}{सममित} सकारात्मक
        अर्धनिश्चित $ A $ एक ग्राम मैट्रिक्स है: $ A = C^{\mathsf T}C $
        कुछ के लिए $ C $ (के वर्गमूल का प्रयोग करें
        \cref{exo:b2:quadratic:6}), साथ $ C $ उलटा यदि $ A $
        निश्चित है.
  \item यह निष्कर्ष निकालें कि यह एक सकारात्मक अर्धनिश्चित है $ A $ संतुष्ट
        $\abs{a_{ij}} \leq \sqrt{a_{ii}\,a_{jj}}$ सभी के लिए $ i,
        j $ \emph{(दो निर्देशांक तक सीमित)} --- द
        कॉची-श्वार्ज़ असमानता, मैट्रिक रूप से पुनः पढ़ें।
  \item (चोल्स्की) चलो $ A $ सकारात्मक निश्चित रहें. यह साबित करो
        वहाँ एक है \emph{अद्वितीय} ऊपरी \omterm{def:b2:reduction:diag}{त्रिकोणीय} $ T $ के साथ
        सकारात्मक विकर्ण प्रविष्टियाँ जैसे कि
        \[
        A = T^{\mathsf T}\,T
        \]
        \emph{(अस्तित्व: वैक्टर पर ग्राम-श्मिट लागू करें
        साकार $ A $ ग्राम मैट्रिक्स के रूप में; विशिष्टता: यदि
        $ T_1^{\mathsf T}T_1 = T_2^{\mathsf T}T_2$, दिखाओ
        $ T_1T_2^{-1}$ ऑर्थोगोनल और \omterm{def:b2:reduction:diag}{त्रिकोणीय} है
        इसलिए, सकारात्मक विकर्ण $ I $)}.
  \item दिखाएँ कि प्रमुख प्रमुख नाबालिग संतुष्ट हैं
        $\Delta_k = (t_{11}\cdots t_{kk})^2$, और उसका निष्कर्ष निकालो
        द्वारा उत्पादित धुरी \omterm{thm:b2:quadratic:gauss}{गॉस में कमी} एक का
        सकारात्मक निश्चित रूप, प्राकृतिक चर में लिया गया
        \omterm{def:b2:structures:generated}{आदेश}, हैं
        \[
        d_k = \frac{\Delta_k}{\Delta_{k-1}}
        \qquad (\Delta_0 = 1) :
        \]
        सिल्वेस्टर की कसौटी के नाबालिग
        (\cref{exo:b2:quadratic:8}) और गॉस की धुरी हैं
        वही डेटा. पर जाँचा
        \cref{ex:b2:quadratic:tworoads}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Hadamard's inequality.}
\begin{enumerate}[resume]
  \item होने देना $ A $ सकारात्मक निश्चित रहें. सिद्ध करना
        \[
        \det A \leq a_{11}\,a_{22}\cdots a_{nn}
        \]
        \emph{(सामान्यीकृत करें: $ B = DAD $ साथ $ D =
        \operatorname{diag}(a_{ii}^{-1/2})$ इकाई विकर्ण है;
        अवश्यंभावी $\det B = \prod \mu_i $ एएम--जीएम द्वारा विरुद्ध
        $\operatorname{tr} B = n $)}.
  \item दिखाएँ कि समानता iff रखती है $ A $ विकर्ण है.
  \item परिणाम निकालना \emph{हैडामर्ड की असमानता}: हर वास्तविक के लिए
        वर्ग मैट्रिक्स $ M $ स्तंभों के साथ $ c_1, \dots, c_n $,
        \[
        \abs{\det M} \leq \prod_{i=1}^{n}\norm{c_i}_2 ,
        \]
        समानता के साथ (उलटा के लिए) $ M $) यदि कॉलम हैं
        \omterm{def:b2:quadratic:adjoint}{जोड़}ीवार ओर्थोगोनल \emph{(प्रश्न 6-7 को लागू करें
        $ M^{\mathsf T}M $)}.
  \item ज्यामितीय और संयोजक लाभांश: व्याख्या करें
        प्रश्न 8 के रूप में ``एक समांतर चतुर्भुज का आयतन पर है
        ''इसके किनारे की लंबाई का अधिकांश गुणनफल''; और दिखाओ कि ए
        सभी प्रविष्टियों के साथ मैट्रिक्स $\intcc{-1}{1}$ है
        $\abs{\det M} \leq n^{n/2}$. (मैट्रिसेस इसे प्राप्त कर रहा है
        बाउंड --- हैडामर्ड मैट्रिसेस --- के लिए मौजूद हैं $ n = 1, 2$
        और के कई गुणज $4$; चाहे के लिए \emph{सभी}
        के गुणज $4$ एक प्रसिद्ध खुली समस्या है।)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- ध्रुवीय विघटन and singular
values.}
\begin{enumerate}[resume]
  \item होने देना $ A $ उलटा हो. बताते हैं कि $ A^{\mathsf T}A $ है
        सकारात्मक निश्चित, और वह
        \[
        S = \sqrt{A^{\mathsf T}A}
        \quad\text{(the square root of
        \cref{exo:b2:quadratic:6})},
        \qquad Q = AS^{-1}
        \]
        एक गुणनखंडन दीजिए $ A = QS $ साथ $ Q $ ओर्थोगोनल और
        $ S $ सकारात्मक निश्चित.
  \item सिद्ध कीजिए कि यह व्युत्क्रमणीय गुणनखण्ड है $ A $ निराला है।
  \item अस्तित्व को मनमाने ढंग से बढ़ाएँ $ A $: चुनना
        $\varepsilon_k \to 0$ साथ $ A + \varepsilon_k I $
        उलटा, लिखो $ A + \varepsilon_kI = Q_kS_k $, और
        का उपयोग करें सघनता का $ O(n)$
        (\cref{exo:b2:quadratic:5}) निस्सारण करना $ Q_k \to Q $;
        दिखाओ $ S_k = Q_k^{\mathsf T}(A + \varepsilon_kI)$
        कुछ सकारात्मक अर्धनिश्चित में परिवर्तित हो जाता है $ S $ साथ $ A =
        QS $ और $ S = \sqrt{A^{\mathsf T}A}$. कहाँ होता है
        एकवचन के लिए विशिष्टता विफल $ A $?
  \item (एकवचन मूल्य अपघटन) यह निष्कर्ष निकालें कि प्रत्येक वास्तविक
        वर्ग $ A $ लिखते हैं
        \[
        A = U\,\Sigma\,V^{\mathsf T},
        \qquad U, V \in O(n),\quad
        \Sigma = \operatorname{diag}(\sigma_1, \dots,
        \sigma_n),\ \sigma_i \geq 0 ,
        \]
        जहां $\sigma_i $ (द \emph{एकवचन मान}) हैं
        द eigenvalues का $\sqrt{A^{\mathsf T}A}$.
  \item तीन परिणाम: $\vertiii{A}_2 = \sigma_{\max}$ के लिए
        \emph{प्रत्येक} असली $ A $ (सामान्यीकरण
        \cref{exo:b2:quadratic:7}); $\abs{\det A} =
        \sigma_1\cdots\sigma_n $; और इकाई की छवि
        एक उलटे के नीचे का गोला $ A $ के साथ एक दीर्घवृत्ताभ है
        अर्ध-अक्ष $\sigma_1, \dots, \sigma_n $ स्तंभों के साथ
        का $ U $.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Conics, by the spectral theorem.}
एक समतल शंकुव का शून्य समुच्चय है $ f(x) = q(x) + \langle b,
x\rangle + c $, साथ $ q \neq 0$ ए द्विघात रूप मैट्रिक्स का $ A $,
$ b \in \R^2$, $ c \in \R $.
\begin{enumerate}[resume]
  \item कम करना $ f $ एक घूर्णन द्वारा (मुख्य अक्ष,
        \cref{cor:b2:quadratic:principalaxes}) के बाद ए
        अनुवाद, और संभावित गैर-रिक्त को वर्गीकृत करें,
        के \omterm{thm:b2:structures:signature}{हस्ताक्षर} द्वारा अविकृत आकृतियाँ $ q $: दीर्घवृत्त
        ($\det A > 0$), अतिपरवलय ($\det A < 0$), परवलय
        ($\det A = 0$, रैंक $1$, रैखिक पद के साथ नहीं
        अवशोषित)।
  \item कमी को पूर्ण रूप से चलाएँ
        \[
        x^2 + 4xy + y^2 + 2x - 2y = 4 :
        \]
        घुमाए गए निर्देशांक, कम किए गए समीकरण, प्रकृति और
        शंकु का केंद्र.
  \item (केंद्रीय शांकव) मान लीजिए $\det A \neq 0$. दिखाएं
        \emph{केंद्र} है $ x_0 = -\frac12 A^{-1}b $, और वह
        \omterm{def:b2:quadratic:def}{अनुरूपता} द्वारा $\begin{pmatrix} I & x_0\\ 0 &
        1\end{pmatrix}$ की $3\times3$ मैट्रिक्स $\widetilde Q
        = \begin{pmatrix} A & b/2 \\ b^{\mathsf T}/2 &
        c\end{pmatrix}$ पैदावार
        \[
        \det\widetilde Q = f(x_0)\,\det A :
        \]
        केंद्रीय शंकु विकृत है (एक बिंदु या दो रेखाएँ)
        \omterm{def:b2:series:def}{बिल्कुल} कब $\det\widetilde Q = 0$.
  \item प्रश्न 16 के उदाहरण पर प्रश्न 17 को सत्यापित करें:
        गणना करें $ x_0$, $ f(x_0)$ और $\det\widetilde Q $, और
        फिर से निष्कर्ष निकालें कि शंकु एक अपक्षयी है
        अतिपरवलय.
  \item (एक क्वाड्रिक पेंसिल) के लिए $\lambda \in \R $, वर्गीकृत करें
        सतह
        \[
        x^2 + y^2 + z^2 + 2\lambda(xy + yz + zx) = 1
        \]
        से eigenvalues इसके मैट्रिक्स का \emph{(सभी
        संरचना: eigenvalues $1 + 2\lambda $ और $1 -
        \lambda $ दोहरा)}: गोला/दीर्घवृत्त, बेलन, का युग्म
        समतल, एक और दो शीट के हाइपरबोलॉइड, तदनुसार
        को $\lambda $.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Two forms at once: simultaneous
reduction.}
\begin{enumerate}[resume]
  \item होने देना $ q $ सकारात्मक निश्चित रहें और $ q'$ एक मनमाना
        द्विघात रूप पर $ E $. सिद्ध कीजिए कि इसका एक आधार है
        $ E $ जो कि असामान्य है $ q $ और ओर्थोगोनल के लिए
        $ q'$: इस में, $ q = \sum x_i^2$ और $ q' = \sum \mu_i
        x_i^2$ \emph{(उपयोग $ q $ आंतरिक उत्पाद के रूप में और लागू करें
        एन्डोमोर्फिज्म का प्रतिनिधित्व करने के लिए वर्णक्रमीय प्रमेय
        $ q'$)}.
  \item मैट्रिक्स फॉर्म: के लिए $ A $ सकारात्मक निश्चित और $ B $
        \omterm{def:b2:quadratic:adjoint}{सममित}, एक उलटा है $ P $ साथ
        $ P^{\mathsf T}AP = I $ और $ P^{\mathsf T}BP =
        \operatorname{diag}(\mu_1, \dots, \mu_n)$, जहां
        $\mu_i $ की जड़ें हैं $\det(B - \mu A) = 0$.
  \item के लिए इसे पूर्ण रूप से चलाएं
        \[
        A = \begin{pmatrix} 2 & 1\\ 1 & 1 \end{pmatrix},
        \qquad
        B = \begin{pmatrix} 0 & 1\\ 1 & 0 \end{pmatrix} :
        \]
        सामान्यीकृत eigenvalues $\mu_\pm $, और वैक्टर
        दोनों रूपों को विकर्णित करना।
  \item दिखाएँ कि सकारात्मक निश्चितता को छोड़ा नहीं जा सकता: के लिए
        \[
        A = \begin{pmatrix} 1 & 0\\ 0 & -1 \end{pmatrix},
        \qquad
        B = \begin{pmatrix} 0 & 1\\ 1 & 0 \end{pmatrix},
        \]
        कोई भी आधार दोनों रूपों को विकर्ण नहीं करता \emph{(अगर $ P $
        दोनों विकर्णित, $\det(B - \mu A)$ के साथ अलग हो जायेंगे
        असली जड़ें; इसकी गणना करें)}.
  \item दिखाओ कि $\mu_i $ प्रश्न 21 के हैं
        eigenvalues का $ A^{-1}B $, ओर वो $ A^{-1}B $, जबकि नहीं
        \omterm{def:b2:quadratic:adjoint}{सममित} सामान्य तौर पर, हमेशा होता है \omterm{def:b2:reduction:diag}{विकर्णीय} के साथ
        असली eigenvalues \emph{(द्वारा संयुग्मित करें $\sqrt A $)}.
  \item संश्लेषण। प्रत्येक एक वाक्य में: (i) एक प्रमेय
        प्रत्येक भाग पर झुक गया; (ii) भागों के कौन से परिणाम
        I-III सकारात्मक रूप से जीवित रहता है \emph{अर्द्ध}निश्चित
        मैट्रिक्स, और जिन्हें निश्चितता की आवश्यकता है; (iii) द
        भौतिक तंत्र जिसके छोटे-छोटे दोलन प्रश्न उठाते हैं
        20--22 विकर्णीकरण (गतिज और स्थितिज ऊर्जा के रूप में)।
        दो रूप), और क्या $\mu_i$ मतलब वहाँ.
\end{enumerate}
