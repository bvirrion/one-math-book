\chapter{अनुक्रम और शृंखला}\label{ch:b2:series}

संख्यात्मक श्रृंखला का सिद्धांत (वर्ष 1 खंड) यहां तीन में परिपक्व होता है
दिशानिर्देश: मानों के साथ श्रृंखला बानाच स्थान, जहां निरपेक्ष
अभिसरण कार्य करता है; वास्तविक श्रृंखला के लिए बेहतर परीक्षण (\omterm{thm:b2:series:abel}{हाबिल
सारांश}); और \emph{\omterm{def:b2:series:summable}{संक्षेपणीय} परिवार} --- योग से मुक्त
शब्दों का क्रम --- दोहरे योग के लिए फ़ुबिनी प्रमेय के साथ
और \omterm{thm:b2:series:fubini}{कॉची उत्पाद}. ये उपकरण सभी फ़ंक्शन-श्रृंखला को वहन करते हैं
अध्याय आगे.

\section{मानक स्थानों में श्रृंखला}

\begin{definition}\label{def:b2:series:def}
एक क्रम के लिए $(u_n)$ एक मानक स्थान में $ E $, श्रृंखला $\sum u_n $
जब इसका आंशिक योग अभिसरण होता है; यह एकत्रित हो जाता है
\emph{बिल्कुल}\index{बिल्कुल} कब
$\sum \norm{u_n} < \infty $. में एक \emph{बनच} अंतरिक्ष, निरपेक्ष
अभिसरण का तात्पर्य अभिसरण (\cref{thm:b2:nvs:absoluteconvergence});
में एक गैर \omterm{def:b2:metric:complete}{पूरा} अंतरिक्ष यह विफल हो सकता है
(\cref{exo:b2:series:9}).
\end{definition}

\begin{example}\label{ex:b2:series:neumann}
में $\mathcal{M}_n(K)$ (या $\mathcal{L}_c(E)$, $ E $ बनच): के लिए
$\vertiii A < 1$, द \emph{न्यूमैन श्रृंखला} $\sum A^k $ अभिसरण
\omterm{def:b2:series:def}{बिल्कुल} को $(I - A)^{-1}$ (साबित हुआ) \cref{exo:b2:nvs:5});
$\sum \frac{A^k}{k!}$ अभिसरण \omterm{def:b2:series:def}{बिल्कुल} को $\eu^A $ हरएक के लिए $ A $
(\cref{ex:b2:nvs:matrixexp}). ऑपरेटर-मूल्यवान ज्यामितीय और
घातीय श्रृंखला उनके अदिश मॉडल की तरह व्यवहार करती है --- संपूर्ण
बानाच ढांचे का बिंदु।
\end{example}

\section{हाबिल सारांश}

\begin{theorem}[हाबिल का सारांश और परीक्षण]\label{thm:b2:series:abel}
(भागों द्वारा योग) अदिश राशि वालों के लिए $ a_n $ और वैक्टर $ b_n $, साथ $ B_n
= \sum_{k=0}^{n} b_k $:\index{हाबिल सारांश}
\[
\sum_{n=0}^{N} a_n b_n
= a_N B_N - \sum_{n=0}^{N-1} (a_{n+1} - a_n) B_n .
\]
(हाबिल का परीक्षण) यदि $(a_n)$ एक वास्तविक अनुक्रम है, जो घटता जा रहा है $0$, और
आंशिक रकम $ B_n $ हैं \emph{घिरे} (में एक \omterm{def:b2:nvs:banach}{बनच स्थान}), तब
$\sum a_n b_n $ जुटता है.
\end{theorem}

\begin{proof}
पहचान, कदम दर कदम: साथ $ B_{-1} = 0$, लिखना $ b_n = B_n -
B_{n-1}$ और विभाजित,
\[
\sum_{n=0}^{N} a_nb_n
= \sum_{n=0}^{N} a_nB_n - \sum_{n=0}^{N} a_nB_{n-1}
= \sum_{n=0}^{N} a_nB_n - \sum_{n=0}^{N-1} a_{n+1}B_{n} ,
\]
द्वारा दूसरे योग को पुनः अनुक्रमित करना $ n \mapsto n + 1$ (द $ B_{-1}$
शब्द लुप्त हो जाता है); सामान्य श्रेणी एकत्रित करना $0 \leq n \leq N-1$
पत्तियों $ a_NB_N $ प्लस $\sum_{n\leq N-1}(a_n - a_{n+1})B_n $: द
बताया गया सूत्र. यह भागों द्वारा असतत एकीकरण है
$(B_n)$ के प्रतिव्युत्पन्न के रूप में $(b_n)$ और अंतर $ a_{n+1}
- a_n $ के व्युत्पन्न के रूप में $(a_n)$. परीक्षण के लिए, साथ $\norm{B_n}
\leq M $: द
सीमा अवधि $ a_N B_N \to 0$; शृंखला $\sum (a_n -
a_{n+1})B_n $ अभिसरण \omterm{def:b2:series:def}{बिल्कुल}, तब से
\[
\sum_n \norm{(a_{n+1} - a_n)B_n} \leq M \sum_n (a_n - a_{n+1})
= M a_0 < \infty
\]
(दूरबीन, $ a_n \downarrow 0$). पहचान के दोनों टुकड़े
अभिसरण, इसलिए ऐसा होता है $\sum a_n b_n $.
\end{proof}

\begin{example}\label{ex:b2:series:sinn}
$\sum \frac{\sin n}{n}$ अभिसरण: $ a_n = \frac1n \downarrow 0$ और
$ B_n = \sum_{k=1}^{n} \sin k $ बाध्य है --- वास्तव में $ B_n =
\Im\sum_{k \leq n} \eu^{\iu k} = \Im\,\frac{\eu^{\iu}(\eu^{\iu n} -
1)}{\eu^{\iu} - 1}$, मापांक का $\leq \frac{2}{\abs{\eu^{\iu} - 1}}$.
ऐसा होता है \emph{नहीं} एकाग्र \omterm{def:b2:series:def}{बिल्कुल} ($\abs{\sin n} \geq \sin^2 n
= \frac{1 - \cos 2n}{2}$, और $\sum \frac{1 - \cos 2n}{2n}$ विचलन
तब से $\sum \frac{\cos 2n}{n}$ जबकि उसी एबेल परीक्षण द्वारा अभिसरण होता है
$\sum \frac{1}{2n}$ विचलन)। प्रत्यावर्ती श्रृंखला परीक्षण है
विशेष मामला $ b_n = (-1)^n $.
\end{example}

\begin{example}[अभिसरण के चक्र पर हाबिल]\label{ex:b2:series:abelboundary}
किस कॉम्प्लेक्स के लिए $ z $ साथ $\abs z = 1$ करता है $\sum_{n \geq 1}
\frac{z^n}{n}$ जुटना? पर $ z = 1$ यह हार्मोनिक श्रृंखला है:
भिन्न के लिए $ z \neq 1$ वृत्त पर, हाबिल का परीक्षण लागू होता है
के साथ $ a_n = \frac1n \downarrow 0$ और $ b_n = z^n $, जिसका आंशिक
रकमें स्वतंत्र रूप से सीमाबद्ध होती हैं $ N $:
\[
\Bigl|\sum_{n=1}^{N} z^n\Bigr|
= \Bigl|\frac{z(z^N - 1)}{z - 1}\Bigr|
\leq \frac{2}{\abs{z - 1}} .
\]
अभिसारी --- यद्यपि कभी नहीं \omterm{def:b2:series:def}{बिल्कुल} ($\sum\frac1n $). एक
श्रृंखला, व्यवहारों का एक \omterm{def:b2:structures:sn}{चक्र}: एक बिंदु पर विचलन,
हर जगह अर्ध-अभिसरण। यह मानक सीमा है
शक्ति श्रृंखला का व्यवहार (\cref{ch:b2:powerseries}), यहाँ मिले
नंगे हाथों से; पर $ z = -1$ यह प्रत्यावर्तन को पुनः प्राप्त करता है
हार्मोनिक श्रृंखला, और पर $ z = \eu^{\iu\theta}$ यह वास्तविक है और
काल्पनिक भाग श्रृंखला हैं $\sum\frac{\cos
n\theta}{n}$ और $\sum\frac{\sin n\theta}{n}$ का
\cref{exo:b2:series:4}.
\end{example}

\begin{example}[एक मूर्ख-फँसी हुई वैकल्पिक श्रृंखला]\label{ex:b2:series:trap}
करता है $\sum_{n\geq2} \dfrac{(-1)^n}{\sqrt n + (-1)^n}$ जुटना?
संकेत वैकल्पिक होते हैं और पद बदलते रहते हैं $0$ --- फिर भी
वैकल्पिक परीक्षण करता है \emph{नहीं} लागू करें: मॉड्यूलि
$\frac{1}{\sqrt n + (-1)^n}$ कम नहीं हो रहे हैं (वे उछल पड़ते हैं
प्रत्येक विषम $ n $). इसके बजाय विस्तार करें:
\[
\frac{(-1)^n}{\sqrt n + (-1)^n}
= \frac{(-1)^n}{\sqrt n}\cdot
\frac{1}{1 + \frac{(-1)^n}{\sqrt n}}
= \frac{(-1)^n}{\sqrt n} - \frac{1}{n}
+ O\Bigl(\frac{1}{n^{3/2}}\Bigr).
\]
पहला टुकड़ा अभिसरण करता है (वैकल्पिक परीक्षण, ईमानदारी से लागू किया गया)।
$\frac1{\sqrt n}\downarrow0$), तीसरा अभिसरण करता है \omterm{def:b2:series:def}{बिल्कुल}
--- लेकिन मध्य भाग अपसारी हार्मोनिक श्रृंखला है: द
योग \emph{विचलन} को $-\infty $. समापन अंतर्दृष्टि: कब
एकरसता विफल हो जाती है, तब तक विस्तार करें जब तक कि हर टुकड़ा एक जैसा न हो जाए
\omterm{def:b2:series:def}{बिल्कुल} अभिसारी या साफ़ परीक्षण केस; छिपा हुआ
$-\frac1n $ साइन-काउंटिंग के लिए अदृश्य है।
\end{example}

\begin{remark}[सामान्य ख़तरे]\label{rem:b2:series:pitfalls}
(i) ``शर्तें प्रवृत्त होती हैं $0$'' कुछ भी साबित नहीं होता: हार्मोनिक श्रृंखला
विचलन (ii) वैकल्पिक परीक्षण की आवश्यकता है \emph{घटते}
मोडुली --- \cref{ex:b2:series:trap} विहित है
प्रतिउदाहरण, और \cref{exo:b2:series:1}की तीसरी सीरीज है
ड्रिल. (iii) सशर्त रूप से अभिसरण श्रृंखला नहीं हो सकती है
पुनर्व्यवस्थित (\cref{ex:b2:series:rearrange}), और उनके \omterm{thm:b2:series:fubini}{कॉची
उत्पाद} विचलन हो सकता है: के लिए $\sum\frac{(-1)^n}{\sqrt{n+1}}$
वर्ग, विकर्ण पद संतुष्ट करते हैं
\[
\abs{c_k} = \sum_{m=0}^{k}
\frac{1}{\sqrt{(m+1)(k-m+1)}}
\geq (k+1)\cdot\frac{2}{k+2} \longrightarrow 2 \neq 0
\]
(प्रत्येक कारक अधिकतम है $\frac{k+2}2$ AM--GM द्वारा), तो $\sum c_k $
विचलन --- कम से कम एक कारक का पूर्ण अभिसरण
(\cref{exo:b2:series:8}) कोई विलासिता नहीं है. (iv) संक्षेपण है
के बारे में \emph{निरपेक्ष} परिभाषा के अनुसार सीमा: ऐसा कोई नहीं है
एक सशर्त के रूप में बात सारांशित परिवार.
\end{remark}

\section{सारांशित परिवार}

\begin{definition}\label{def:b2:series:summable}
होने देना $ I $ एक हो \omterm{def:b2:structures:countable}{गणनीय} सूचकांक सेट. एक परिवार $(u_i)_{i \in I}$ का
\emph{गैर-नकारात्मक वास्तविकताएँ} है \emph{संक्षेपणीय}\index{संक्षेपणीय}
जब परिमित आंशिक राशियाँ परिबद्ध होती हैं; इसका योग है
\[
\sum_{i \in I} u_i = \sup_{F \subseteq I \text{ finite}} \sum_{i
\in F} u_i \in \intcc{0}{+\infty} .
\]
वास्तविक या कॉम्प्लेक्स (या बानाच वैक्टर) का एक परिवार तब सारांशित किया जा सकता है
$(\norm{u_i})$ है; इसके बाद इसके योग को विभाजित करके परिभाषित किया जाता है
सकारात्मक/नकारात्मक (या वास्तविक/काल्पनिक) भाग --- \omterm{def:b2:nvs:equivalent}{समकक्ष}, जैसे
का सामान्य मूल्य $\sum_{n} u_{\sigma(n)}$ सभी गणनाओं पर
$\sigma $ का $ I $ (नीचे देखें)।
\end{definition}

\begin{method}[एक परीक्षण चुनना]\label{met:b2:series:choosing}
का सामना करना पड़ $\sum u_n $, क्रम में: (1) यदि $ u_n \not\to 0$, विचलन,
रुकें. (2) यदि पदों में स्थिर चिह्न है, तो तुलना करें: एक खोजें
समतुल्य (\cref{ch:b2:comparison}) और इसे पर रखें
रीमैन--बर्ट्रेंड मानचित्र। (3) यदि संकेत वैकल्पिक हों
\emph{घटते} मॉड्यूलि, वैकल्पिक परीक्षण; यदि मॉड्यूलि
एकरस नहीं हैं, प्रत्येक टुकड़े तक शब्द का विस्तार करें \omterm{def:b2:series:def}{बिल्कुल}
अभिसारी या साफ़ परीक्षण केस (\cref{ex:b2:series:trap}). (4)
यदि साइन पैटर्न दोलनशील है लेकिन संरचित है ($\sin
n\theta $, $\eu^{\iu n\theta}$, मैट्रिक्स शक्तियां), हाबिल का परीक्षण
परिबद्ध आंशिक रकम. (5) पूर्ण अभिसरण सदैव सार्थक होता है
पहले जांचें: यह मजबूत है, ऑर्डर-प्रूफ़ है और अनलॉक करता है \omterm{thm:b2:series:fubini}{कॉची
उत्पाद} और फ़ुबिनी.
\end{method}

\begin{example}[विकर्ण गणना द्वारा योग्यता]\label{ex:b2:series:diagonalcount}
जिसके लिए $ s > 0$ परिवार है
$\bigl((m + n)^{-s}\bigr)_{m, n \geq 1}$ \omterm{def:b2:series:summable}{संक्षेपणीय}? समूह बनाएं
विकर्णों द्वारा परिमित आंशिक योग $ m + n = k $: विकर्ण $ k $
किया जाता है $ k - 1$ जोड़े, प्रत्येक योगदान दे रहा है $ k^{-s}$, तो
परिमित रकम \omterm{def:b2:series:def}{बिल्कुल} (और निकास) से बंधी होती है
\[
\sum_{k \geq 2} \frac{k - 1}{k^{s}} ,
\]
के समतुल्य सकारात्मक पदों वाली एक श्रृंखला $ k^{1-s}$: \omterm{def:b2:series:summable}{संक्षेपणीय}
आईएफएफ $ s - 1 > 1$, यानी\ $ s > 2$. द्वि-आयामी सूचकांक खाता है
एक पूर्ण शक्ति: पदों का एक तल ``एक आयाम अधिक है
एक रेखा से भिन्न'' --- सूचकांक की गिनती ज्यामिति
सेट, व्यक्तिगत शब्दों का आकार नहीं, निर्णय लेता है संक्षेपण.
(वही जनगणना से पता चलता है $\bigl((m^2 + n^2)^{-1}\bigr)$ क्या नहीं है
\omterm{def:b2:series:summable}{संक्षेपणीय}: विकर्ण पर $ m + n = k $, प्रत्येक पद कम से कम है
$ k^{-2}$, और $(k-1)\cdot k^{-2}$ हार्मोनिक की तरह योग
शृंखला.)
\end{example}

\begin{theorem}[संक्षेपण एवं व्यवस्था]\label{thm:b2:series:rearrangement}
\begin{enumerate}
  \item गैर-नकारात्मक परिवारों के लिए, राशि किसी के अंतर्गत अपरिवर्तनीय है
        गणना: $\sum_{i} u_i = \sum_{n=0}^{\infty}
        u_{\sigma(n)}$ हर आपत्ति के लिए $\sigma \colon \N \to I $.
  \item एक वास्तविक या जटिल श्रृंखला $\sum u_n $ है \emph{क्रमविनिमेय रूप से
        अभिसारी} (प्रत्येक पुनर्व्यवस्था उसी के साथ अभिसरण करती है
        योग) यदि और केवल यदि यह है \omterm{def:b2:series:def}{बिल्कुल} अभिसारी।
\end{enumerate}
\end{theorem}

\begin{proof}
(1) प्रत्येक आंशिक योग $\sum_{n \leq N} u_{\sigma(n)}$ एक परिमित है
परिवार का आंशिक योग (तो) $\leq $ सुपर); हर परिमित $ F $ है
कुछ में निहित है $\{\sigma(0), \dots, \sigma(N)\}$ (तो अति
$\leq $ श्रृंखला की सीमा)। दोनों सीमाएँ मेल खाती हैं।

(2) यदि $\sum\abs{u_n} < \infty $: किसी भी पुनर्व्यवस्था के लिए $\sigma $ और
$\varepsilon > 0$, चुनना $ N $ साथ $\sum_{n > N}\abs{u_n} \leq
\varepsilon $; पद से परे जहां $\sigma $ ख़त्म हो गया है
$\intint{0}{N}$, पुनर्व्यवस्थित आंशिक योग इससे भिन्न होते हैं
मूल सीमा अधिक से अधिक $\varepsilon $: समान राशि. अगर $\sum
\abs{u_n} = \infty $ लेकिन $\sum u_n $ अभिसरण (वास्तविक मामला; जटिल
निर्देशांक के अनुसार इस प्रकार है): सकारात्मक और नकारात्मक दोनों भाग
विचलन, और कोई भी किसी भी निर्धारित सीमा तक पहुंचने के लिए पुनर्व्यवस्थित कर सकता है
--- रीमैन का प्रमेय, में किया गया \cref{exo:b2:series:5} ---
इसलिए क्रमविनिमेय अभिसरण विफल हो जाता है।
\end{proof}

\begin{example}[रंगे हाथ पकड़ा गया एक मनचला]\label{ex:b2:series:rearrange}
प्रत्यावर्ती हार्मोनिक श्रृंखला का योग है
$\sum_{n\geq1}\frac{(-1)^{n-1}}{n} = \ln 2$ (वर्ष 1
वॉल्यूम). इसे ``एक सकारात्मक, दो नकारात्मक'' के रूप में पुनर्व्यवस्थित करें:
\[
1 - \frac12 - \frac14 + \frac13 - \frac16 - \frac18 + \frac15 -
\cdots
\]
प्रत्येक ब्लॉक को तीन-तीन का समूह बनाना,
\[
\frac{1}{2k-1} - \frac{1}{4k-2} - \frac1{4k}
= \frac{1}{4k-2} - \frac{1}{4k}
= \frac12\Bigl(\frac{1}{2k-1} - \frac1{2k}\Bigr),
\]
इसलिए पुनर्व्यवस्थित श्रृंखला अभिसरित हो जाती है $\frac12\ln 2$ --- आधा
मूल योग, \omterm{def:b2:series:def}{बिल्कुल} समान शर्तों के साथ। \omterm{def:b2:series:def}{बिल्कुल} नहीं
अभिसरण श्रृंखला उनके पदों के क्रम को याद रखती है; \omterm{def:b2:series:summable}{संक्षेपणीय}
परिवार वास्तव में वही हैं जो ऐसा नहीं करते हैं।
\end{example}

\begin{example}[समूहीकरण सुरक्षित है, असमूहीकरण सुरक्षित नहीं है]\label{ex:b2:series:grouping}
a के क्रमागत पदों को समूहीकृत करना \emph{संमिलित} श्रृंखला कभी नहीं
योग बदलता है: समूहीकृत आंशिक योग एक अनुवर्ती बनाते हैं
मूल वाले. विपरीत संक्रिया निषिद्ध है:
\[
(1 - 1) + (1 - 1) + (1 - 1) + \cdots = 0 + 0 + \cdots = 0,
\]
फिर भी असमूहीकृत $1 - 1 + 1 - 1 + \cdots $ विचलन (आंशिक
रकमें बीच में घूमती रहती हैं $1$ और $0$). अनग्रुपिंग वैध है
केवल एक क्षतिपूर्ति परिकल्पना के साथ --- उदाहरण के लिए, शर्तें
की ओर रुझान $0$ बंधी हुई समूह लंबाई के साथ: फिर दो के बीच
समूहीकृत आंशिक योग, मूल योग अधिक से अधिक योग में बह जाते हैं
$ o(1)$ बंधी हुई संख्या की शर्तें, और अभिसरण वापस स्थानांतरित हो जाता है।
यह \omterm{def:b2:series:def}{बिल्कुल} वही खंड है जिसके तहत ब्लॉक की गणना की जाती है
\cref{ex:b2:series:rearrange} यह एक प्रमाण है न कि दिखावा
हाथ.
\end{example}

\begin{theorem}[परिवारों के लिए फ़ुबिनी; कॉची उत्पाद]\label{thm:b2:series:fubini}
होने देना $(u_{m,n})_{(m,n) \in \N^2}$ एक हो \omterm{def:b2:series:summable}{संक्षेपणीय} दोहरा परिवार (अर्थात्\
$\sup_F \sum_F \abs{u_{m,n}} < \infty $). तब\index{फ़ुबिनी प्रमेय
श्रृंखला के लिए}
\[
\sum_{(m,n)} u_{m,n}
= \sum_{m=0}^{\infty}\Bigl(\sum_{n=0}^{\infty} u_{m,n}\Bigr)
= \sum_{n=0}^{\infty}\Bigl(\sum_{m=0}^{\infty} u_{m,n}\Bigr)
= \sum_{k=0}^{\infty}\Bigl(\sum_{m+n=k} u_{m,n}\Bigr),
\]
सभी आंतरिक श्रृंखला अभिसरण (\omterm{def:b2:series:def}{बिल्कुल}). विशेषकर, यदि $\sum
a_m $ और $\sum b_n $ एकाग्र \omterm{def:b2:series:def}{बिल्कुल}, उनका \emph{कॉची
उत्पाद}\index{कॉची
उत्पाद} अभिसरण \omterm{def:b2:series:def}{बिल्कुल} साथ
\[
\Bigl(\sum_m a_m\Bigr)\Bigl(\sum_n b_n\Bigr)
= \sum_{k=0}^{\infty} c_k,
\qquad
c_k = \sum_{m=0}^{k} a_m b_{k-m} .
\]
\end{theorem}

\begin{proof}
\emph{गैर-नकारात्मक मामला.} प्रत्येक समूहीकरण (पंक्तियों, स्तंभों या द्वारा)।
विकर्ण) समान सर्वोच्च की गणना करता है: जोड़े का कोई भी सीमित सेट है
पंक्तियों के एक सीमित ब्लॉक में समाहित (नीचे प्रत्येक समूहीकृत योग को सीमित करते हुए)।
परिमित आंशिक योग से और ऊपर कुल से), और एकस्वर
आंशिक रकम का अभिसरण बाकी काम करता है --- ठोस रूप से, पंक्तियों के लिए:
$\sum_{m \leq M}\sum_{n \leq N} u_{m,n} \leq S $ देता है, देता है $ N
\to \infty $ तब $ M \to \infty $, $\sum_m \sum_n u_{m,n} \leq S $;
इसके विपरीत प्रत्येक परिमित $ F $ ऐसे आयत में बैठता है, इसलिए $ S \leq
\sum_m\sum_n u_{m,n}$. विकर्ण: त्रिभुजों के साथ समान दो सीमाएँ
आयतों के बजाय.

\emph{सामान्य मामला.} सकारात्मक और नकारात्मक (वास्तविक और) में विभाजित करें
काल्पनिक) भाग, प्रत्येक ए \omterm{def:b2:series:summable}{संक्षेपणीय} गैर-नकारात्मक परिवार; चार
समूह प्रत्येक भाग पर सहमत हैं, इसलिए अंतर पर; निरपेक्ष
आंतरिक श्रृंखला का अभिसरण गैर-नकारात्मक मामले से आता है
पर लागू किया गया $\abs{u_{m,n}}$.

\emph{\omterm{thm:b2:series:fubini}{कॉची उत्पाद}.} परिवार $ u_{m,n} = a_m b_n $ है \omterm{def:b2:series:summable}{संक्षेपणीय}:
का परिमित आंशिक योग $\abs{a_mb_n}$ से बंधे हैं
$\bigl(\sum\abs{a_m}\bigr)\bigl(\sum\abs{b_n}\bigr)$. पंक्तियाँ देते हैं
$\bigl(\sum a_m\bigr)\bigl(\sum b_n\bigr)$; विकर्ण देते हैं $\sum_k
c_k $.
\end{proof}

\begin{example}[घातीय पहचान, ईमानदारी से]\label{ex:b2:series:exp}
के लिए $ a, b \in \C $ (या आवागमन मैट्रिक्स):
\[
\Bigl(\sum_m \frac{a^m}{m!}\Bigr)\Bigl(\sum_n
\frac{b^n}{n!}\Bigr)
= \sum_k \sum_{m+n=k} \frac{a^m b^n}{m!\,n!}
= \sum_k \frac{(a + b)^k}{k!},
\]
प्रत्येक विकर्ण पर द्विपद प्रमेय द्वारा: $\eu^a \eu^b = \eu^{a+b}$
--- का कार्यात्मक समीकरण $\exp $ अकेले श्रृंखला से व्युत्पन्न.
(कम्यूटेशन का उपयोग द्विपद चरण में किया जाता है; गैर-कम्यूटिंग के लिए
मैट्रिक्स की पहचान वास्तव में विफल हो जाती है,
\cref{ch:b2:diffeq}.)
\end{example}

\begin{example}[कंप्यूटिंग डिवाइस के रूप में कॉची उत्पाद]\label{ex:b2:series:squaresum}
ज्यामितीय श्रृंखला से और \cref{exo:b2:series:2}'एस $\sum_{n
\geq 1} nz^n = \frac{z}{(1-z)^2}$ ($\abs z < 1$), और एक
\omterm{thm:b2:series:fubini}{कॉची उत्पाद} दूसरे क्षण समाप्त होता है। गुणा $\sum_m
mz^m $ द्वारा $\sum_n z^n $: विकर्ण गुणांक है $\sum_{m=0}^k
m = \frac{k(k+1)}2$, इसलिए
\[
\frac{z}{(1-z)^3} = \sum_{k\geq0}\frac{k(k+1)}{2}\,z^k ,
\]
और पहचान $ n^2 = 2\cdot\frac{n(n+1)}2 - n $ असेम्बल
\[
\sum_{n\geq1} n^2z^n = \frac{2z}{(1-z)^3} - \frac{z}{(1-z)^2}
= \frac{z(1+z)}{(1-z)^3} .
\]
पर $ z = \frac12$: $\sum_{n\geq1}\frac{n^2}{2^n} =
\frac{\frac12\cdot\frac32}{\frac18} = 6$ --- के साथ एक बंद मूल्य
कहीं कोई भेदभाव नहीं, बस \omterm{def:b2:series:def}{बिल्कुल} अभिसरण श्रृंखला
बहुपद की तरह गुणा किया गया। पहचानों की वही दूरबीन
प्रत्येक की गणना करता है $\sum n^dz^n $, और संभाव्यतावादी पहचान लेंगे
ज्यामितीय कानून का दूसरा तथ्यात्मक क्षण
(\cref{ch:b2:genfun}).
\end{example}

\begin{example}[एक दोहरा-योग मूल्यांकन]\label{ex:b2:series:doublesum}
के लिए $ s > 1$ असली, चलो $\zeta(s) = \sum_{n\geq1} n^{-s}$. गिनती
दोहरे योग द्वारा भाजक --- परिवार $(m^{-s}n^{-s})$ ऊपर
$(m,n) \in (\N^*)^2$ है \omterm{def:b2:series:summable}{संक्षेपणीय} (अभिसरण सकारात्मक का उत्पाद
श्रृंखला) --- और उत्पाद द्वारा समूहीकरण $ q = mn $:
\[
\zeta(s)^2 = \sum_{m,n} \frac{1}{(mn)^s}
= \sum_{q=1}^{\infty} \frac{d(q)}{q^s},
\]
कहाँ $ d(q)$ के भाजक की संख्या है $ q $. \omterm{def:b2:series:summable}{संक्षेपणीय} परिवार
कॉम्बिनेटरिक्स को विश्लेषण में बदलें।
\end{example}

\begin{example}[एक फ़ुबिनी मूल्यांकन: $\sum_n (\zeta(n) - 1) = 1$]\label{ex:b2:series:zetasum}
पूर्णांक के लिए $ n \geq 2$, $\zeta(n) - 1 = \sum_{k \geq 2} k^{-n}$.
दोहरा परिवार $(k^{-n})_{k, n \geq 2}$ है \omterm{def:b2:series:summable}{संक्षेपणीय}: संक्षेप
सबसे पहले ज्यामितीय स्तंभ,
\[
\sum_{k\geq2}\sum_{n\geq2} \frac{1}{k^n}
= \sum_{k\geq2} \frac{1/k^2}{1 - 1/k}
= \sum_{k\geq2} \frac{1}{k(k-1)} = 1
\]
(टेलीस्कोपिंग), और सभी शब्द सकारात्मक हैं, इसलिए
\cref{thm:b2:series:fubini} इसके बजाय पंक्तियों द्वारा योग को अधिकृत करता है:
\[
\sum_{n\geq2}\bigl(\zeta(n) - 1\bigr) = 1 .
\]
अनंत अनेक $\zeta $-मूल्य, प्रत्येक पारलौकिक-दिखने वाला,
ऐसी पूँछें होती हैं जो \omterm{def:b2:series:def}{बिल्कुल} जुड़ती हैं $1$. समापन अंतर्दृष्टि: जब ए
दोहरे योग में धनात्मक पद होते हैं, इसकी गणना किसी भी क्रम में करें
पतन --- यहाँ स्तंभ ज्यामितीय हैं, पंक्तियाँ रहस्यमय हैं,
और फ़ुबिनी पतन को स्थानांतरित करता है।
\end{example}

\begin{example}[ज्यामितीय श्रृंखला एक समीकरण को हल करती है]\label{ex:b2:series:neumannsolve}
में \omterm{def:b2:nvs:banach}{बनच स्थान} $\bigl(C(\intcc01),
\norm\cdot_\infty\bigr)$, हल करना $ x - K(x) = y $ कहाँ $ K(f)$ है
स्थिर कार्य $\frac12\int_0^1 f $. \omterm{thm:b2:nvs:continuouslinear}{ऑपरेटर मानदंड} है
$\vertiii K \leq \frac12 < 1$, इसलिए न्यूमैन श्रृंखला लागू होती है
(\cref{ex:b2:series:neumann}): $ x = \sum_{n\geq0} K^n(y)$.
पुनरावृत्तियों की गणना करें: $ K(y) = \frac12\int_0^1 y $ (एक स्थिरांक),
और आवेदन कर रहे हैं $ K $ एक स्थिरांक के लिए $ c $ देता है $\frac c2$, इसलिए
$ K^n(y) = \frac{1}{2^{n-1}}\cdot\frac12\int_0^1 y $ के लिए $ n \geq
1$. ज्यामितीय स्थिरांकों का योग:
\[
x = y + \Bigl(\int_0^1 y\Bigr)
\sum_{n\geq1}\frac{1}{2^n} = y + \int_0^1 y .
\]
जाँच करना: $ x - K(x) = y + \int y - \frac12\bigl(\int y + \int
y\bigr) = y $. एक अनंत शृंखला, एक सीमित उत्तर, और एक
एक-पंक्ति सत्यापन --- ज्यामितीय श्रृंखला एक व्युत्क्रम है
एल्गोरिदम, केवल एक अभिसरण कथन नहीं।
\end{example}

\begin{example}[आंशिक अंशों द्वारा टेलीस्कोपिंग]\label{ex:b2:series:telescope}
सटीक योग दुर्लभ है; टेलिस्कोपिंग इसका मुख्य आपूर्तिकर्ता है।
विघटित
\[
\frac{1}{n(n+1)(n+2)}
= \frac{1}{2}\Bigl(\frac{1}{n(n+1)} -
\frac{1}{(n+1)(n+2)}\Bigr),
\]
(सामान्य विभाजक को घटाकर जांचें), तो आंशिक
रकम पतन:
\[
\sum_{n=1}^{N}\frac{1}{n(n+1)(n+2)}
= \frac12\Bigl(\frac{1}{1\cdot2} -
\frac{1}{(N+1)(N+2)}\Bigr)
\longrightarrow \frac14 .
\]
वही पैटर्न---शब्द को इस प्रकार लिखें $ c(u_n - u_{n+1})$ के लिए
एक स्पष्ट $(u_n)$ --- सुलझ गया \cref{exo:b2:series:10}
(आर्कटेंजेंट) और प्रत्येक की गणना करता है $\sum\frac{1}{n(n+1)\cdots(n +
k)} = \frac{1}{k\cdot k!}$. जब इस पर एक सटीक योग मौजूद है
स्तर, एक दूरबीन आमतौर पर शब्द में छिपी होती है।
\end{example}

\begin{remark}[इस खंड के भीतर परिप्रेक्ष्य]\label{rem:b2:series:perspectives}
आगे के तीन अध्याय प्रत्यक्ष ग्राहक हैं। के लिए
\cref{ch:b2:funcseq}: का सामान्य अभिसरण $\sum f_n $ है
का पूर्ण अभिसरण $\sum\norm{f_n}_\infty $ में बनच
अंतरिक्ष $\bigl(C, \norm\cdot_\infty\bigr)$ ---इस अध्याय का
\cref{thm:b2:nvs:absoluteconvergence} पोशाक में. के लिए
\cref{ch:b2:powerseries}: अभिसरण की डिस्क के अंदर
सब कुछ पूर्ण है और \omterm{def:b2:series:summable}{संक्षेपणीय}, इसलिए \omterm{thm:b2:series:fubini}{कॉची उत्पाद} और
पुनर्व्यवस्थाएँ निःशुल्क चलती हैं (यही कारण है कि शक्ति श्रृंखला कई गुना बढ़ जाती है
बहुपद); सीमा पर, हाबिल का परीक्षण हावी हो जाता है
(\cref{ex:b2:series:abelboundary}). के लिए \cref{ch:b2:genfun}:
संभाव्यता उत्पन्न करने वाले फलन शक्ति श्रृंखला हैं जिनके
हेरफेर --- स्वतंत्र चर के योग के लिए उत्पाद,
मिश्रित कानूनों के लिए दोगुनी रकम --- सभी द्वारा लाइसेंस प्राप्त हैं
\cref{thm:b2:series:fubini}. \omterm{def:b2:series:summable}{संक्षेपणीय} परिवार कानूनी हैं
विश्लेषण विभाग आने वाला है।
\end{remark}

\begin{remark}[जहाँ इस अध्याय का प्रयोग किया जाता है]
अनंत योग वाली हर चीज यहां से गुजरती है: शक्ति श्रृंखला
(\cref{ch:b2:powerseries}) हैं \omterm{def:b2:series:summable}{संक्षेपणीय} छद्म वेश में परिवार,
फूरियर गुणांक गुणा हो जाते हैं \omterm{thm:b2:series:fubini}{कॉची उत्पाद} और
पार्सेवल द्वारा पुनर्व्यवस्थित (\cref{ch:b2:fourier}), और संभाव्यता
कार्य उत्पन्न करना (\cref{ch:b2:genfun}) फ़ुबिनी के प्रमेय हैं
अपेक्षाओं पर लागू किया गया। वर्ष 3 की मात्रा अवशोषित करती है \omterm{def:b2:series:summable}{संक्षेपणीय}
गिनती के उपाय पर लेबेसेग एकीकरण में परिवार ---
कहाँ \cref{thm:b2:series:fubini} का एक विशेष मामला बन जाता है
फ़ुबिनी--टोनेली प्रमेय।
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b2:series:1}
की प्रकृति: $\sum \dfrac{\cos n}{n}$; $\;\sum \dfrac{(-1)^n}{\ln
n}$; $\;\sum \dfrac{(-1)^n}{n^{3/4} + \cos n}$ \emph{(जैसा विस्तार करें
वर्ष 1 ट्रैप: वैकल्पिक परीक्षण के लिए एकरसता की आवश्यकता होती है)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:series:2}
इसके लिए सिद्ध करें $\abs z < 1$: $\sum_{n\geq1} n z^{n} =
\dfrac{z}{(1-z)^2}$, के माध्यम से \omterm{thm:b2:series:fubini}{कॉची उत्पाद} का $\sum z^n $ के साथ
स्वयं.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:series:3}
(क्रोनकर-प्रकार लेम्मा) चलो $\sum b_n $ एक अभिसरण वास्तविक श्रृंखला बनें।
साबित करो, द्वारा \omterm{thm:b2:series:abel}{हाबिल सारांश}, वह $\dfrac{1}{n}\sum_{k=1}^{n} k\,b_k
\to 0$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:series:4}
के अभिसरण का अध्ययन करें $\sum \dfrac{\sin(n\theta)}{n^\alpha}$
($\theta \in \R $, $\alpha > 0$) --- जिसके लिए $(\theta, \alpha)$ यह है \omterm{def:b2:series:def}{बिल्कुल} अभिसरण, अर्ध-अभिसरण, अपसारी?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:series:5}
(रीमैन पुनर्व्यवस्था) चलो $\sum u_n $ अभिसरण हो लेकिन नहीं
\omterm{def:b2:series:def}{बिल्कुल} अभिसरण वास्तविक श्रृंखला, और $\ell \in \R $. यह साबित करो
की कुछ पुनर्व्यवस्था $\sum u_n $ में एकत्रित हो जाता है $\ell $.
\emph{(सकारात्मक और नकारात्मक शब्दों की दोनों उपश्रेणियाँ अलग-अलग दिखाएँ;
फिर लालच से वैकल्पिक करें: अधिक होने तक सकारात्मक शब्द लें
$\ell $, फिर नीचे गिरने तक नकारात्मक, इत्यादि; शर्तें
करते हैं $0$, अभिसरण के लिए मजबूर करना $\ell $.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:series:6}
साबित करो कि परिवार
$\Bigl(\dfrac{x^{m+n}}{m!\,n!}\Bigr)_{(m,n)\in\N^2}$ है \omterm{def:b2:series:summable}{संक्षेपणीय}
हरएक के लिए $ x \in \R $, और पहचान पुनः प्राप्त करें $(\eu^x)^2 =
\eu^{2x}$ दोहरे योग को विकर्णों के अनुदिश समूहित करके $ m + n =
k $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:series:7}
साबित करो कि परिवार $\bigl(\frac{1}{m^2 n^2}\bigr)_{m,n \geq 1}$
है \omterm{def:b2:series:summable}{संक्षेपणीय}, और वह समूहीकरण $\gcd $: साथ $ q = \gcd(m,n)$,
\[
\zeta(2)^2 = \sum_{q\geq1} \frac{1}{q^4}
\sum_{\substack{a,b \geq 1\\ \gcd(a,b)=1}} \frac{1}{a^2b^2}
= \zeta(4) \cdot S,
\]
कहाँ $ S = \sum_{\gcd(a,b)=1} \frac{1}{a^2b^2}$: निष्कर्ष निकालें $ S =
\zeta(2)^2/\zeta(4)$. \emph{(प्रत्येक जोड़ी $(m,n)$ अनोखा लिखता है
$(qa, qb)$ साथ $\gcd(a,b) = 1$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:series:8}
(उत्पादों पर हाबिल का प्रमेय, हल्का संस्करण) मान लीजिए $\sum a_n $
अभिसरण \omterm{def:b2:series:def}{बिल्कुल} और $\sum b_n $ जुटता है. साबित करें कि उनका
\omterm{thm:b2:series:fubini}{कॉची उत्पाद} $\sum c_n $ अभिसरण, के साथ $\sum c_n = (\sum
a_n)(\sum b_n)$. \emph{(लिखना $ C_N = \sum_{k\leq N} c_k = \sum_n
a_n B_{N-n}$ साथ $ B $ का आंशिक योग $ b $; के अनुसार विभाजित करें
$ n \leq N/2$ या नहीं, की सीमा का उपयोग कर $(B_m)$ और पूर्ण
की पूँछ $(a_n)$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:series:9}
में (गैर \omterm{def:b2:metric:complete}{पूरा}) अंतरिक्ष $ E $ अंततः-शून्य वास्तविक अनुक्रमों का
सुपर के साथ आदर्श, प्रदर्शन एक \omterm{def:b2:series:def}{बिल्कुल} अभिसरण श्रृंखला जो करती है
में जुटना नहीं $ E $. \emph{(कोशिश $ u_n = 2^{-n} e_n $ साथ $(e_n)$ द
विहित अनुक्रम।)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:series:10}
पहचान सत्यापित करें $\arctan(n+1) - \arctan(n) =

\end{exercise}

\section{समस्या: यूलर की $\zeta(2) = \pi^2/6$, कॉची द्वारा
कोटैंजेंट योग}

यूलर की सबसे प्रसिद्ध पहचान, $1 + \frac14 + \frac19 + \cdots =
\frac{\pi^2}6$, एक पूरी तरह से प्राथमिक प्रमाण स्वीकार करता है, के कारण
कॉची: डी मोइवर का सूत्र एक बहुपद बनाता है जिसकी जड़ें हैं
संख्याएँ $\cot^2\frac{k\pi}{2n+1}$, विएटा ने उन जड़ों का सार प्रस्तुत किया
\omterm{def:b2:series:def}{बिल्कुल}, और निचोड़ $\cot^2\theta < \frac{1}{\theta^2} <
1 + \cot^2\theta $ के आंशिक योग को कुचल देता है $\sum\frac1{k^2}$
दो स्पष्ट तर्कसंगत सीमाओं के बीच। हम प्रमाण को पूर्ण रूप से चलाते हैं,
निकालना $\zeta(4) = \frac{\pi^4}{90}$ तो फिर उसी विधि से
अभिसरण और विचलन के बीच की संपूर्ण सीमा का मानचित्रण करें
बर्ट्रेंड श्रृंखला --- और साबित करें कि सीमा का कोई महत्व नहीं है
सबसे धीमी अभिसरण श्रृंखला।

\begin{problem}[{सप्ताहांत समस्या --- $\zeta(2) = \pi^2/6$ और
बर्ट्रेंड पैनोरमा}]
\label{pb:b2:series:1}
लगातार, $ n \geq 1$ और $\theta_k = \dfrac{k\pi}{2n+1}$ के लिए $ k
= 1, \dots, n $; टिप्पणी $0 < \theta_k < \frac\pi2$.

\medskip
\noindent\textbf{Part I --- The cotangent identity.}
\begin{enumerate}
  \item डी मोइवरे का सूत्र सिद्ध करें $(\cos\theta +
        \iu\sin\theta)^m = \cos m\theta + \iu\sin m\theta $ ($ m
        \in \N $), और निष्कर्ष निकालें, के लिए $ m = 2n + 1$,
        \[
        \sin\bigl((2n{+}1)\theta\bigr) = \sum_{j=0}^{n}
        (-1)^j\binom{2n+1}{2j+1}\cos^{2(n-j)}\theta\,
        \sin^{2j+1}\theta .
        \]
  \item उसके लिए निष्कर्ष निकालें $\theta \in \intoo{0}{\frac\pi2}$,
        \[
        \sin\bigl((2n{+}1)\theta\bigr) =
        \sin^{2n+1}\theta\; P_n(\cot^2\theta),
        \qquad
        P_n(x) = \sum_{j=0}^{n}
        (-1)^j\binom{2n+1}{2j+1}x^{\,n-j},
        \]
        डिग्री का एक बहुपद $ n $ अग्रणी गुणांक के साथ $2n
        + 1$.
  \item बताते हैं कि $ x_k = \cot^2\theta_k $, $ k = 1, \dots, n $, हैं
        $ n $ \emph{विशिष्ट} की जड़ें $ P_n $ --- इसलिए सभी
        उन्हें.
  \item विएटा द्वारा, सटीक पहचान साबित करें
        \[
        \sum_{k=1}^{n} \cot^2\frac{k\pi}{2n+1}
        = \frac{n(2n-1)}{3}.
        \]
  \item निष्कर्ष भी निकालें $\displaystyle\sum_{k=1}^{n}
        \frac{1}{\sin^2\theta_k} = \frac{2n(n+1)}{3}$.
  \item निचोड़ साबित करें: $\cot^2\theta < \dfrac1{\theta^2} <
        \dfrac{1}{\sin^2\theta}$ के लिए $\theta \in
        \intoo{0}{\frac\pi2}$ \emph{(से $\sin\theta < \theta <
        \tan\theta $)}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The squeeze closes: Euler's
theorem.}
\begin{enumerate}[resume]
  \item सारांश प्रश्न 6 समाप्त $ k = 1, \dots, n $ साथ $\theta =
        \theta_k $, स्थापित करना
        \[
        \frac{n(2n-1)}{3} \;<\;
        \frac{(2n+1)^2}{\pi^2}\sum_{k=1}^{n}\frac1{k^2}
        \;<\; \frac{2n(n+1)}{3}.
        \]
  \item निष्कर्ष (\emph{कॉची के प्रमाण द्वारा यूलर का प्रमेय}):
        \[
        \zeta(2) = \sum_{k=1}^{\infty}\frac{1}{k^2} =
        \frac{\pi^2}{6}.
        \]
  \item सैंडविच से एक रेट निकालें: दिखाएँ
        \[
        \Bigl|\sum_{k=1}^{n}\frac1{k^2} - \frac{\pi^2}6\Bigr|
        = O\Bigl(\frac1n\Bigr),
        \]
        सटीक पूँछ के अनुरूप $\sum_{k>n}k^{-2} =
        \frac1n - \frac{1}{2n^2} + O(n^{-3})$ का
        \cref{exo:b2:comparison:11}.
  \item मशीन को एक मंजिल ऊपर चलाएँ: दूसरी विएटा का उपयोग करके
        का कार्य $ P_n $, दिखाओ
        \[
        \sum_{k=1}^{n}\cot^4\theta_k =
        \Bigl(\frac{n(2n-1)}3\Bigr)^{\!2} -
        \frac{2n(2n-1)(2n-2)(2n-3)}{60}
        \;\sim\; \frac{8n^4}{45},
        \]
        और साथ निचोड़ें $\cot^4 < \theta^{-4} < (1 +
        \cot^2)^2$ प्राप्त करने के लिए $\zeta(4) = \dfrac{\pi^4}{90}$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Dividends.}
\begin{enumerate}[resume]
  \item से निष्कर्ष निकालें $\zeta(2) = \frac{\pi^2}6$:
        \[
        \sum_{k\geq0}\frac{1}{(2k+1)^2} = \frac{\pi^2}{8},
        \qquad
        \sum_{k\geq1}\frac{(-1)^{k-1}}{k^2} = \frac{\pi^2}{12}.
        \]
  \item के साथ संयुक्त \cref{exo:b2:series:7}: गणना करें $ S =
        \sum_{\gcd(a,b)=1}\frac{1}{a^2b^2} =
        \frac{\zeta(2)^2}{\zeta(4)} = \frac52$, और व्याख्या करें
        $\frac{1}{\zeta(2)} = \frac{6}{\pi^2} \approx 0.608$ जैसे
        सहअभाज्य युग्मों का घनत्व (अनुमानित बताएं)।
        ईमानदारी से: कठोर गणना वर्ष 3 खंड है
        मामला)।
  \item (प्रमाणित त्वरण) प्रश्न 9 का पूँछ सूत्र
        देता है $\sum_{k\leq n}k^{-2} + \frac1n - \frac1{2n^2} =
        \frac{\pi^2}6 + O(n^{-3})$. आवश्यक कार्य की तुलना करें
        के छह अंक $\zeta(2)$: प्रत्यक्ष योग बनाम
        पर सही किया गया योग $ n = 100$ (त्रुटि कहां है
        $1.7\cdot10^{-7}$).
  \item प्रश्न 4 को हाथ से जांचें $ n = 1$ और $ n = 2$ (द
        मूल्य $\cot^2\frac\pi3 = \frac13$ और
        $\cot^2\frac\pi5 + \cot^2\frac{2\pi}5 = 2$), का उपयोग करना
        $\cos\frac\pi5 = \frac{1+\sqrt5}4$ या एक संख्यात्मक
        मूल्यांकन.
  \item साथी की पहचान साबित करें
        \[
        \sum_{k=1}^{n}\tan^2\frac{k\pi}{2n+1} = n(2n+1)
        \]
        \emph{(संख्या $\tan^2\theta_k $ की जड़ें हैं
        उलटा बहुपद $ x^nP_n(1/x)$)}, और इसे सत्यापित करें $ n
        = 1$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The Bertrand panorama.} के लिए $\alpha,
\beta \in \R $, विचार करें \emph{बर्ट्रेंड श्रृंखला}
\[
\sum_{n \geq 3} \frac{1}{n^{\alpha}(\ln n)^{\beta}} .
\]
\begin{enumerate}[resume]
  \item उसके लिए दिखाओ $\alpha > 1$ श्रृंखला मिलती है,
        जो भी हो $\beta $ \emph{(साथ तुलना करें
        $ n^{-(1+\alpha)/2}$)}.
  \item उसके लिए दिखाओ $\alpha < 1$ यह अलग हो जाता है, जो भी हो
        $\beta $.
  \item के लिए $\alpha = 1$: श्रृंखला का उपयोग करना--अभिन्न तुलना
        (\cref{thm:b2:comparison:seriesintegral}) साथ $ f(t) =
        \frac{1}{t(\ln t)^\beta}$, अभिसरण सिद्ध करें यदि $\beta
        > 1$.
  \item सीमा को पुनरावृत्त करें: दिखाएँ $\sum\frac{1}{n\ln n\,\ln\ln
        n}$ जबकि विचलन करता है $\sum\frac{1}{n\ln n\,(\ln\ln n)^2}$
        जुटता है.
  \item दो जाल: की प्रकृति का निर्धारण
        \[
        \sum_n \frac{1}{n^{1 + 1/\ln n}}
        \qquad\text{and}\qquad
        \sum_n \frac{1}{n^{1 + 1/\ln\ln n}}
        \]
        \emph{(गणना करें $ n^{1/\ln n}$ \omterm{def:b2:series:def}{बिल्कुल}; तुलना करना
        $ n^{1/\ln\ln n}$ की हर शक्ति के साथ $\ln n $)}.
  \item (कोई धीमी अभिसरण श्रृंखला नहीं) चलो $\sum a_n $ कोई भी हो
        के साथ अभिसरण श्रृंखला $ a_n > 0$, और $ R_n = \sum_{k
        \geq n}a_k $ इसकी पूँछ. साबित करें कि $\sum
        \frac{a_n}{\sqrt{R_n}}$ फिर भी \emph{अभिसरण}
        \emph{(दूरबीन से तुलना करें $2(\sqrt{R_n} -
        \sqrt{R_{n+1}})$)}, हालांकि $\frac{a_n/\sqrt{R_n}}{a_n}
        \to \infty $: प्रत्येक अभिसरण श्रृंखला सख्ती से है
        एक और अभिसरण श्रृंखला का प्रभुत्व। की सीमा
        अभिसरण एक वक्र नहीं बल्कि एक कोहरा है।
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Cross-checks and synthesis.}
\begin{enumerate}[resume]
  \item (कॉची संक्षेपण) सिद्ध करें: के लिए $(a_n)$ सकारात्मक
        घट रहा है, $\sum a_n $ यदि अभिसरण करता है $\sum 2^k a_{2^k}$
        जुटता है. इससे प्रश्न 18 का सीमांत पुनः प्राप्त करें।
  \item (धीमेपन की कीमत) के लिए $\sum\frac1{n(\ln n)^2}$,
        पूंछ को एक अभिन्न अंग से बांधें और उसका सारांश दिखाएं
        को $ N = 10^6$ फिर भी इससे बड़ी त्रुटि रह जाती है
        $0.07$: सिद्धांत द्वारा प्रमाणित अभिसरण बेकार हो सकता है
        संख्यात्मकता के लिए --- प्रश्न 13 से विरोधाभास।
