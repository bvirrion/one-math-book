\chapter{सामान्य वेक्टर रिक्त स्थान}\label{ch:b2:nvs}

जब \omterm{def:b2:metric:def}{मीट्रिक स्थान} एक सदिश समष्टि है और दूरी से आती है
ए \omterm{def:b2:nvs:norm}{आदर्श}, टोपोलॉजी और रैखिक बीजगणित परस्पर क्रिया करना शुरू करते हैं: रैखिक मानचित्र
हैं \omterm{def:b2:metric:continuity}{निरंतर} ठीक उसी समय जब वे यूनिट बॉल पर बंधे होते हैं,
परिमित आयाम सभी को बाध्य करता है मानदंड सहमत होना, और पूर्णता बदल जाता है
अभिसारी श्रृंखलाओं में पूर्णतया अभिसरण श्रृंखला। द
परिमित/अनंत आयाम विभाजन --- रीज़ के प्रमेय में क्रिस्टलीकृत
--- अध्याय का सबसे गहरा पाठ है.

लगातार, $E, F $ वेक्टर रिक्त स्थान हैं $ K = \R $ या $\C $.

\section{मानदंड}

\begin{definition}\label{def:b2:nvs:norm}
ए \emph{आदर्श}\index{आदर्श} पर $ E $ एक नक्शा है $\norm{\,\cdot\,} \colon E
\to \R_+$ सबके साथ, सबके लिए $ x, y \in E $, $\lambda \in K $:
\[
\norm x = 0 \iff x = 0,
\qquad
\norm{\lambda x} = \abs\lambda\,\norm x,
\qquad
\norm{x + y} \leq \norm x + \norm y .
\]
तब $ d(x, y) = \norm{x - y}$ एक दूरी है, और सब कुछ
\cref{ch:b2:metric} लागू होता है. उलटा त्रिकोण असमानता
$\bigl|\norm x - \norm y\bigr| \leq \norm{x - y}$ आदर्श बनाता है
स्वयं $1$-लिप्सचिट्ज़; जोड़ और अदिश गुणन हैं
\omterm{def:b2:metric:continuity}{निरंतर} (अनुमान $\norm{(x + y) - (x' + y')} \leq \norm{x -
x'} + \norm{y - y'}$, वगैरह।)।
\end{definition}

\begin{example}\label{ex:b2:nvs:examples}
पर $ K^n $:
\[
\norm{x}_1 = \sum_i \abs{x_i},
\qquad
\norm{x}_2 = \Bigl(\sum_i \abs{x_i}^2\Bigr)^{1/2},
\qquad
\norm{x}_\infty = \max_i \abs{x_i}
\]
($\norm\cdot_2$ एक है \omterm{def:b2:nvs:norm}{आदर्श} कॉची द्वारा--श्वार्ज़, वर्ष 1 खंड)। पर
$ C(\intcc{a}{b})$:
\[
\norm f_\infty = \sup \abs f,
\qquad
\norm f_1 = \int_a^b \abs f,
\qquad
\norm f_2 = \Bigl(\int_a^b \abs f^2\Bigr)^{1/2},
\]
अंतिम दो हैं मानदंड की सख्त सकारात्मकता के लिए धन्यवाद
इंटीग्रल और इंटीग्रल कॉची--श्वार्ज़ (वर्ष 1 खंड)। मैट्रिक्स पर:
कोई भी \omterm{def:b2:nvs:norm}{आदर्श} पर $\mathcal M_n(K) \simeq K^{n^2}$; \omterm{thm:b2:nvs:continuouslinear}{ऑपरेटर मानदंड}
नीचे संरचनात्मक रूप से महत्वपूर्ण हैं।
\end{example}

\begin{definition}[समतुल्य मानदंड]\label{def:b2:nvs:equivalent}
दो मानदंड $ N_1, N_2$ पर $ E $ हैं \emph{समकक्ष}\index{समकक्ष} जब स्थिरांक हों $ c, C > 0$ साथ
\[
c\,N_1 \leq N_2 \leq C\, N_1 .
\]
समकक्ष मानदंड एक ही लें \omterm{def:b2:metric:topology}{खुला} सेट, समान अभिसरण और
कॉची अनुक्रम, वही \omterm{def:b2:metric:compact}{कॉम्पैक्ट} और \omterm{def:b2:metric:complete}{पूरा} उपसमुच्चय: वही
विश्लेषण.
\end{definition}

\begin{example}[अनंत आयाम में गैर-बराबरी]\label{ex:b2:nvs:nonequivalent}
पर $ C(\intcc{0}{1})$: $\norm f_1 \leq \norm f_\infty $ हमेशा, लेकिन
कोई रिवर्स बाउंड होल्ड नहीं: $ f_n(x) = x^n $ है $\norm{f_n}_\infty = 1$
और $\norm{f_n}_1 = \frac{1}{n+1} \to 0$. इसलिए $ f_n \to 0$ के लिए
$\norm\cdot_1$ लेकिन के लिए नहीं $\norm\cdot_\infty $: दो मानदंड
अभिसरण के बारे में ही असहमत हैं।
\end{example}

\begin{example}[आयाम में स्पष्ट स्थिरांक $ n $]\label{ex:b2:nvs:constants}
पर $ K^n $ तीन शास्त्रीय मानदंड हैं \omterm{def:b2:nvs:equivalent}{समकक्ष} तेज़ के साथ
स्थिरांक:
\[
\norm x_\infty \leq \norm x_2 \leq \norm x_1
\leq \sqrt n\,\norm x_2 \leq n\,\norm x_\infty ,
\]
मध्य बाउंड $\norm x_1 \leq \sqrt n\norm x_2$ से आ रहा है
कॉची--श्वार्ज़ ऑल-वन्स वेक्टर के विरुद्ध। चरम सदिश:
$ e_1$ प्रथम दो असमानताओं को समानता बनाता है, $(1, 1,
\dots, 1)$ अंतिम दो। आयाम $ n $ में स्पष्ट रूप से बैठता है
स्थिरांक --- अनंत में विफलता का मात्रात्मक बीज
आयाम: जैसे $ n \to \infty $ कोई एकसमान स्थिरांक जीवित नहीं रहता,
जो वास्तव में क्या है \cref{ex:b2:nvs:nonequivalent} पर प्रदर्शित करता है
कार्य स्थान.
\end{example}

\section{सतत रेखीय मानचित्र}

\begin{theorem}[निस्र्पण]\label{thm:b2:nvs:continuouslinear}
एक रेखीय मानचित्र के लिए $ u \colon E \to F $ मानक स्थानों के बीच,
निम्नलिखित हैं \omterm{def:b2:nvs:equivalent}{समकक्ष}:
\begin{enumerate}
  \item $ u $ है \omterm{def:b2:metric:continuity}{निरंतर};
  \item $ u $ है \omterm{def:b2:metric:continuity}{निरंतर} पर $0$;
  \item $ u $ बंद इकाई गेंद पर घिरा है: $\sup_{\norm x \leq
        1} \norm{u(x)} < \infty $;
  \item वहाँ है $ C \geq 0$ साथ $\norm{u(x)} \leq C \norm x $ सभी के लिए $ x $;
  \item $ u $ है \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़}.
\end{enumerate}
ऐसा सबसे कम $ C $ है \emph{ऑपरेटर मानदंड}\index{ऑपरेटर मानदंड}
$\vertiii{u} = \sup_{\norm x \leq 1}\norm{u(x)} = \sup_{x \neq 0}
\frac{\norm{u(x)}}{\norm x}$; यह जगह बनाता है $\mathcal{L}_c(E,
F)$ का \omterm{def:b2:metric:continuity}{निरंतर} रैखिक मानचित्र एक मानक स्थान, के साथ
\[
\vertiii{v \circ u} \leq \vertiii v\, \vertiii u .
\]
\end{theorem}

\begin{proof}
(1 $\Rightarrow $ 2) तुच्छ. (2 $\Rightarrow $ 3): निरंतरता पर $0$
साथ $\varepsilon = 1$ देता है $\delta $ साथ $\norm x \leq \delta
\Rightarrow \norm{u(x)} \leq 1$; एकरूपता फिर किसी भी पैमाने पर होती है $ x $
साथ $\norm x \leq 1$ उस गेंद में नीचे और पीछे:
\[
\norm{u(x)} = \frac1\delta\,\norm{u(\delta x)} \leq
\frac1\delta ,
\]
तब से $\norm{\delta x} \leq \delta $. (3 $\Rightarrow $ 4):
के लिए $ x \neq 0$, बाउंड को लागू करें $\frac{x}{\norm x}$. (4
$\Rightarrow $ 5): $\norm{u(x) - u(y)} = \norm{u(x - y)} \leq
C\norm{x - y}$. (5 $\Rightarrow $ 1) ज्ञात.

\omterm{def:b2:nvs:norm}{आदर्श} के लिए अभिगृहीत $\vertiii\cdot $: एकरूपता और अलगाव हैं
स्पष्ट ($\vertiii u = 0$ बल $ u = 0$ गेंद पर, इसलिए
हर जगह); त्रिकोण असमानता से $\norm{(u + v)(x)} \leq
\norm{u(x)} + \norm{v(x)}$. उपगुणात्मकता: $\norm{v(u(x))}
\leq \vertiii v\,\norm{u(x)} \leq \vertiii v \vertiii u \norm x $.
\end{proof}

\begin{example}\label{ex:b2:nvs:operatornorms}
पर $\bigl(C(\intcc{0}{1}), \norm\cdot_\infty\bigr)$: मूल्यांकन
$ f \mapsto f(0)$ है \omterm{thm:b2:nvs:continuouslinear}{ऑपरेटर मानदंड} $1$; एकीकरण $ f \mapsto
\int_0^1 f $ है \omterm{def:b2:nvs:norm}{आदर्श} $1$; वो नक्शा $ f \mapsto \int_0^1 t f(t)\dd t $
है \omterm{def:b2:nvs:norm}{आदर्श} $\int_0^1 t\,\dd t = \frac12$ (ऊपरी भाग त्रिभुज से घिरा है
अभिन्नों के लिए असमानता; पर प्राप्त किया $ f \equiv 1$). लेकिन
भेदभाव, से $(C^1, \norm\cdot_\infty)$ को $(C^0,
\norm\cdot_\infty)$, है \emph{नहीं} \omterm{def:b2:metric:continuity}{निरंतर}: $\norm{\sin(nx)}_\infty
= 1$ जबकि व्युत्पन्न में समर्थन है \omterm{def:b2:nvs:norm}{आदर्श} $ n $. रैखिक का तात्पर्य नहीं है
\omterm{def:b2:metric:continuity}{निरंतर} अनंत आयाम में.
\end{example}

\begin{example}[एक क्रम पर दो मानदंड, दो फैसले]\label{ex:b2:nvs:sqrtnxn}
पर $ C(\intcc01)$, होने देना $ g_n(x) = \sqrt{n}\,x^n $. तब
\[
\norm{g_n}_1 = \frac{\sqrt n}{n + 1} \longrightarrow 0,
\qquad
\norm{g_n}_2^2 = \frac{n}{2n + 1} \longrightarrow \frac12,
\qquad
\norm{g_n}_\infty = \sqrt n \longrightarrow \infty :
\]
एक क्रम, तीन मानदंड, तीन व्यवहार --- अभिसरण
शून्य, कोई अभिसरण नहीं (मानदंड पर स्थिर हो जाओ $\frac1{\sqrt2}$ लेकिन
बिंदुवार सीमा है $0$), और विस्फोट। द्रव्यमान का ध्यान केन्द्रित करना
निकट $ x = 1$ के लिए अदृश्य है $\norm\cdot_1$, आधा-दृश्यमान
$\norm\cdot_2$, के लिए प्रमुख $\norm\cdot_\infty $. अनंत में
आयाम, ''क्या यह अभिसरण होता है?'' के बारे में कोई प्रश्न नहीं है
अनुक्रम: यह एक अनुक्रम के बारे में प्रश्न है \emph{और} ए \omterm{def:b2:nvs:norm}{आदर्श}.
\end{example}

\begin{method}[एक ऑपरेटर मानदंड की गणना]\label{met:b2:nvs:opnorm}
हमेशा दो चालों में. \emph{ऊपरी सीमा:} अनुमान लगाना
$\norm{u(x)}$ द्वारा $ C\norm x $ त्रिभुज असमानताओं का उपयोग करते हुए,
कॉची--श्वार्ज़, या अभिन्न सीमाएँ---यह सिद्ध होता है $\vertiii u
\leq C $. \emph{गवाह:} या तो एक विशिष्ट प्रदर्शन करें $ x_0 \neq 0$
साथ $\norm{u(x_0)} = C\norm{x_0}$ (सीमा प्राप्त हो गई है), या ए
इकाई सदिशों का क्रम $ x_n $ साथ $\norm{u(x_n)} \to C $ (द
बाउंड से संपर्क किया गया है)। दोनों चालें अनिवार्य हैं: एक ऊपरी सीमा
अकेला ही देता है $\vertiii u \leq C $, केवल एक गवाह
$\vertiii u \geq C $. अनंत आयाम में साक्षी हो सकता है
एक क्रम होने के लिए--सर्वोच्च को प्राप्त करने की आवश्यकता नहीं है
(\cref{exo:b2:nvs:8}).
\end{method}

\begin{example}[विकर्ण संचालक प्रत्येक मानक को एक समान देखते हैं]\label{ex:b2:nvs:diagonalnorm}
के लिए $ D = \operatorname{diag}(d_1, \dots, d_n)$ पर $ K^n $ साथ
\emph{कोई} की मानदंड $\norm\cdot_1, \norm\cdot_2,
\norm\cdot_\infty $: से $\abs{d_ix_i} \leq
\bigl(\max_j\abs{d_j}\bigr)\abs{x_i}$ समन्वय के अनुसार,
$\norm{Dx} \leq \max_j\abs{d_j}\,\norm x $; और $ x = e_{j_0}$
(एक अधिकतम सूचकांक) इसे प्राप्त करता है। इसलिए $\vertiii D =
\max_j\abs{d_j}$ तीनों मामलों में: विकर्ण मानचित्रों के लिए, सभी
उचित मानदंड वही कहानी बताओ, सबसे बड़ा विस्तार
कारक. हर चीज़ के बारे में मुश्किल \omterm{thm:b2:nvs:continuouslinear}{ऑपरेटर मानदंड} के बारे में है
\emph{गैर}-विकर्ण व्यवहार --- यही कारण है कि अनुकूलित
मानदंड इस अध्याय की सप्ताहांत समस्या (प्रश्न 22) द्वारा कार्य
मैट्रिक्स को पहले विकर्ण बनने के लिए बाध्य करना।
\end{example}

\begin{example}[स्तम्भ योग: $1$-आदर्श जुड़वां \cref{exo:b2:nvs:4}]\label{ex:b2:nvs:columnsums}
पर $(\R^n, \norm\cdot_1)$, द \omterm{thm:b2:nvs:continuouslinear}{ऑपरेटर मानदंड} एक मैट्रिक्स का $ A $ है
सबसे बड़ा निरपेक्ष \emph{स्तंभ} जोड़। विधि चलाएँ: के लिए
$\norm x_1 \leq 1$,
\[
\norm{Ax}_1 = \sum_i\Bigl|\sum_j a_{ij}x_j\Bigr|
\leq \sum_j \abs{x_j}\sum_i\abs{a_{ij}}
\leq \Bigl(\max_j\sum_i\abs{a_{ij}}\Bigr)\norm x_1 ,
\]
और सीमा प्राप्त हो जाती है $ x = e_{j_0}$ अधिकतमीकरण के लिए
स्तंभ $ j_0$ --- कल्पना करने योग्य सबसे साफ-सुथरा गवाह। अभीतक के लिए तो $ A =
\begin{psmallmatrix}1 & -2\\ 3 & 1\end{psmallmatrix}$:
$\vertiii A_1 = \max(1 + 3,\ 2 + 1) = 4$, जबकि
$\vertiii A_\infty = 4$ भी (पंक्तियाँ) --- यहाँ एक संयोग है, नहीं
एक कानून: \omterm{def:b2:linalg:transpose}{खिसकाना} मैट्रिक्स असममित रूप से और दो प्रविष्टियाँ करता है
मानदंड अलग हो जाना। के लिए पंक्तियाँ $\norm\cdot_\infty $, के लिए कॉलम
$\norm\cdot_1$: स्मरक यह है कि प्रत्येक आदर्शकी इकाई सदिश
(साइन पैटर्न, सम्मान।\ आधार वेक्टर) मिलान चुनें
रकम.
\end{example}

\begin{proposition}[द्विरेखीय मानचित्र]\label{prop:b2:nvs:bilinear}
एक द्विरेखीय मानचित्र $ b \colon E \times F \to G $ है \omterm{def:b2:metric:continuity}{निरंतर} आईएफएफ
$\norm{b(x,y)} \leq C\norm x\,\norm y $ कुछ के लिए $ C $; तो यह है
\omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} बंधे हुए सेट पर. (समान प्रमाण पैटर्न; उत्पाद
$(u, v) \mapsto v \circ u $ और मैट्रिक्स गुणन कुंजी हैं
उदाहरण.)
\end{proposition}

\begin{proof}
यदि सीमा कायम है:
\[
b(x,y) - b(x_0,y_0) = b(x - x_0,\, y) + b(x_0,\, y - y_0),
\]
इसलिए $\norm{b(x,y) - b(x_0,y_0)} \leq C\norm{x - x_0}\norm y +
C\norm{x_0}\norm{y - y_0}$: निरंतरता पर $(x_0, y_0)$, और ए
\omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} कहाँ बंधा है $\norm x, \norm y \leq R $. इसके विपरीत,
निरंतरता पर $(0,0)$ देता है $\delta $ साथ $\norm{b(x,y)} \leq 1$ पर
$\norm x, \norm y \leq \delta $; दोनों वेरिएबल्स को स्केल करें।
\end{proof}

\section{परिमित आयाम}

\begin{theorem}[परिमित आयाम में मानदंडों की समानता]\label{thm:b2:nvs:finitedim}
एक परिमित-आयामी स्थान पर, \emph{सभी मानदंड हैं
\omterm{def:b2:nvs:equivalent}{समकक्ष}}.\index{मानदंडों की समानता} फलस्वरूप, परिमित में
आयाम: अभिसरण, खुलापन, सघनता, पूर्णता हैं
आदर्श-स्वतंत्र धारणाएँ; \omterm{def:b2:metric:compact}{कॉम्पैक्ट} $=$ बंद और घिरा हुआ;
जगह है \omterm{def:b2:metric:complete}{पूरा}; और प्रत्येक रैखिक (या बहुरेखीय) मानचित्र
\emph{से} एक परिमित-आयामी स्थान है \omterm{def:b2:metric:continuity}{निरंतर}.
\end{theorem}

\begin{proof}
एक आधार तय करें और पहचानें $ E \simeq K^n $; किसी से भी तुलना करना पर्याप्त है
\omterm{def:b2:nvs:norm}{आदर्श} $ N $ साथ $\norm\cdot_\infty $.

\emph{एक दिशा बीजगणित है:} $ N(x) = N(\sum x_i e_i) \leq
\sum \abs{x_i} N(e_i) \leq C \norm x_\infty $ साथ $ C = \sum N(e_i)$.
ये भी पता चलता है $ N $ है \omterm{def:b2:metric:continuity}{निरंतर} पर $(K^n, \norm\cdot_\infty)$
(यह है $ C $-लिप्सचिट्ज़: $\abs{N(x) - N(y)} \leq N(x - y)$).

\emph{दूसरा टोपोलॉजी है:} इकाई क्षेत्र $ S = \{x :
\norm{x}_\infty = 1\}$ बंद और घिरा हुआ है $(K^n,
\norm\cdot_\infty)$, इस तरह \omterm{def:b2:metric:compact}{कॉम्पैक्ट}
(\cref{thm:b2:metric:compactprops}~(2), के लिए मान्य $\C^n \simeq
\R^{2n}$). \omterm{def:b2:metric:continuity}{निरंतर} समारोह $ N $ अपने न्यूनतम को प्राप्त करता है $ c $ पर
$ S $; $ c > 0$ तब से $ N $ पर ही लुप्त हो जाता है $0 \notin S $. एकरूपता
सीमा फैलाता है: $ N(x) \geq c \norm{x}_\infty $ सभी के लिए $ x $.

परिणाम: सभी कथन कम हो जाते हैं $(K^n,
\norm\cdot_\infty)$, जहां वे जाने जाते हैं
(\cref{thm:b2:metric:rncomplete},
\cref{thm:b2:metric:compactprops}); एक रैखिक $ u $ से
परिमित-आयामी $ E $ संतुष्ट $\norm{u(x)} \leq \sum\abs{x_i}
\norm{u(e_i)} \leq C'\norm{x}_\infty $: बाउंड (4) का
\cref{thm:b2:nvs:continuouslinear}.
\end{proof}

\begin{corollary}\label{cor:b2:nvs:closedsubspace}
किसी भी मानक स्थान का एक परिमित-आयामी उपस्थान बंद है।
\end{corollary}

\begin{proof}
यह है \omterm{def:b2:metric:complete}{पूरा} प्रेरित के लिए \omterm{def:b2:nvs:norm}{आदर्श}
(\cref{thm:b2:nvs:finitedim}), और \omterm{def:b2:metric:complete}{पूरा} उपसमुच्चय बंद हैं
(\cref{def:b2:metric:complete}).
\end{proof}

\begin{example}[समरूपता द्वारा गणना किया गया सर्वोत्तम सन्निकटन]\label{ex:b2:nvs:bestaffine}
में $\bigl(C(\intcc{-1}{1}), \norm\cdot_\infty\bigr)$, कितनी दूर है
$ f(x) = \abs x $ के (बंद, द्वि-आयामी) उपस्थान से
एफ़िन फ़ंक्शंस $ a + bx $? समरूपता से, प्रतिस्थापित करना $ a + bx $ द्वारा
$ a - bx $ पत्तियों $\norm{f - (a \pm bx)}_\infty $ अपरिवर्तित, और
मध्यबिंदु $ a $ कम से कम उतना ही अच्छा करता है (त्रिकोण असमानता पर)।
औसत): यह स्थिरांक पर विचार करने के लिए पर्याप्त है। एक स्थिरांक के लिए
$ a $:
\[
\norm{\abs x - a}_\infty = \max\,(1 - a,\ a)
\geq \frac12 ,
\]
पर न्यूनतम किया गया $ a = \frac12$: दूरी है $\frac12$, प्राप्त किया
स्थिरांक द्वारा $\frac12$. त्रुटि वक्र पर ध्यान दें $\abs x -
\frac12$: पहुँच जाता है $\pm\frac12$ बारी-बारी से $ x = -1, 0,
1$ --- सर्वश्रेष्ठ के लिए वैकल्पिक संकेत के तीन चरम
दो-पैरामीटर परिवार से सन्निकटन। वह
\emph{संतुलन} पैटर्न कोई दुर्घटना नहीं है; यह है
इष्टतमता का \omterm{thm:b2:structures:signature}{हस्ताक्षर} जो इस अध्याय की सप्ताहांत समस्या है
चेबीशेव के प्रमेय में बदल जाता है।
\end{example}

\begin{example}[बंद बनाम सघन उपस्थान]\label{ex:b2:nvs:closeddense}
में $ E = \bigl(C(\intcc{0}{1}), \norm\cdot_\infty\bigr)$: प्रत्येक
$\R_n[X]$ (डिग्री के बहुपद $\leq n $, तक सीमित
$\intcc01$) एक परिमित-आयामी है, इसलिए \emph{बंद किया हुआ}, उपस्थान
--- घात के बहुपदों की एक समान सीमा $\leq n $ एक है. लेकिन
संघ $\R[X]$ उन सभी का है \emph{घना} में $ E $ (द
वीयरस्ट्रैस सन्निकटन प्रमेय, में सिद्ध हुआ
\cref{ch:b2:funcseq}), और घने उचित उप-स्थान इस प्रकार हैं
जितना हो सके गैर-बंद। नैतिक: उप-स्थानों का बंद होना एक है
परिमित-आयामी विशेषाधिकार; बंद फर्शों को ढेर करने से निर्माण हो सकता है
घनी गगनचुंबी इमारत.
\end{example}

\begin{theorem}[रिज़्ज़]\label{thm:b2:nvs:riesz}
एक मानक स्थान की बंद इकाई गेंद $ E $ है \omterm{def:b2:metric:compact}{कॉम्पैक्ट} \emph{अगर और केवल अगर} $\dim E < \infty $.\index{रिज़्ज़ का प्रमेय}
\end{theorem}

\begin{proof}
परिमित आयाम: बंद और परिबद्ध पर्याप्त
(\cref{thm:b2:nvs:finitedim}).

इसके विपरीत, मान लीजिए $\dim E = \infty $. \emph{रिज़्ज़ की लेम्मा:} के लिए
प्रत्येक उचित बंद उपस्थान $ F \subsetneq E $ और $\varepsilon \in
\intoo{0}{1}$, एक इकाई सदिश है $ x $ साथ $ d(x, F) \geq 1 -
\varepsilon $. प्रमाण: चुनें $ y \notin F $, होने देना $\delta = d(y, F) > 0$
($ F $ बंद), चुनें $ f \in F $ साथ $\norm{y - f} \leq
\frac{\delta}{1 - \varepsilon}$, और सेट करें $ x = \frac{y - f}{\norm{y -
f}}$: किसी के लिए $ g \in F $,
\[
\norm{x - g} = \frac{\norm{y - (f + \norm{y-f}\,g)}}{\norm{y - f}}
\geq \frac{\delta}{\norm{y-f}} \geq 1 - \varepsilon ,
\]
अंश से दूरी है $ y $ एक बिंदु तक $ F $.

अब यूनिट वैक्टर बनाएं $ x_1, x_2, \dots $ आगमनात्मक रूप से: $ F_k =
\operatorname{Vect}(x_1, \dots, x_k)$ इसलिए, परिमित-आयामी है
बंद (\cref{cor:b2:nvs:closedsubspace}) और उचित; रिज़्ज़ की लेम्मा
के साथ $\varepsilon = \frac12$ एक इकाई प्रदान करता है $ x_{k+1}$ साथ $ d(x_{k+1},
F_k) \geq \frac12$. क्रम संतुष्ट करता है $\norm{x_p - x_q} \geq
\frac12$ के लिए $ p \neq q $: कोई अभिसरण अनुवर्ती नहीं --- यूनिट बॉल
नहीं है \omterm{def:b2:metric:compact}{कॉम्पैक्ट}.
\end{proof}

\begin{example}[एक आयाम डिटेक्टर के रूप में रिज़्ज़]\label{ex:b2:nvs:rieszdetector}
है $ C(\intcc01)$ परिमित-आयामी? रिज़्ज़ बिना उत्तर दिए
किसी भी स्पष्ट अनंत मुक्त परिवार का प्रदर्शन: अनुक्रम
$ f_n(x) = x^n $ बंद इकाई गेंद में निहित है और संतुष्ट करता है, के लिए
$ m > n $, $\norm{f_n - f_m}_\infty \geq f_n(x_0) - f_m(x_0) > 0$
उपयुक्त बिंदुओं पर --- इस अध्याय में स्पष्ट रूप से परिमाणित किया गया है
सप्ताहांत समस्या (प्रश्न 16), जहां एक अनुवर्ती रहता है
आपसी दूरी $\geq \frac14$. कोई अभिसरण अनुवर्ती नहीं, इसलिए
गेंद नहीं है \omterm{def:b2:metric:compact}{कॉम्पैक्ट}, इसलिए $\dim C(\intcc01) = \infty $ द्वारा
\cref{thm:b2:nvs:riesz}. सघनता यूनिट बॉल का एक है
पूर्ण द्वंद्व: यह सीमित आयाम में रहता है, विफल रहता है
अनंत आयाम, बिना किसी मध्य मार्ग के --- केवल ज्यामिति
आयाम प्रकार को पढ़ता है।
\end{example}

\section{बानाच स्थान}

\begin{definition}\label{def:b2:nvs:banach}
ए \emph{बनच स्थान}\index{बनच स्थान} एक है \omterm{def:b2:metric:complete}{पूरा} मानकीकृत
अंतरिक्ष. उदाहरण: प्रत्येक परिमित-आयामी मानक स्थान
(\cref{thm:b2:nvs:finitedim}); $\bigl(C(\intcc{a}{b}),
\norm\cdot_\infty\bigr)$ (\cref{thm:b2:metric:rncomplete});
$\mathcal{L}_c(E, F)$ के लिए $ F $ बानाच (के लिए समान प्रमाण पैटर्न)।
\omterm{def:b2:metric:continuity}{निरंतर} फ़ंक्शंस)। गैर-उदाहरण: $\bigl(C(\intcc{0}{1}),
\norm\cdot_1\bigr)$ (\cref{exo:b2:nvs:7}).
\end{definition}

\begin{example}[एकीकरण का ऑपरेटर मानदंड]\label{ex:b2:nvs:integrationnorm}
पर $\bigl(C(\intcc01), \norm\cdot_\infty\bigr)$, होने देना $ T(f)(x) =
\int_0^x f(t)\,\dd t $ (एक एंडोमोर्फिज्म: $ T(f)$ है \omterm{def:b2:metric:continuity}{निरंतर}).
दौड़ना \cref{met:b2:nvs:opnorm}. ऊपरी सीमा:
\[
\abs{T(f)(x)} \leq \int_0^x\abs f \leq x\,\norm f_\infty \leq
\norm f_\infty ,
\]
इसलिए $\vertiii T \leq 1$. गवाह: $ f \equiv 1$ देता है $ T(f)(x) =
x $ और $\norm{T(f)}_\infty = 1 = \norm f_\infty $: प्राप्त,
$\vertiii T = 1$. लेकिन ध्यान दें $\vertiii{T^2} = \frac12 \neq
\vertiii T^2$: वास्तव में $ T^2(f)(x) = \int_0^x(x - t)f(t)\dd t $ है
$\abs{T^2(f)(x)} \leq \frac{x^2}2\norm f_\infty $, पुनः प्राप्त किया
पर $ f \equiv 1$; और आम तौर पर $\vertiii{T^n} = \frac1{n!}$ ---
उपगुणात्मक बाध्यता $\vertiii T^n = 1$ ए द्वारा बंद है
भाज्य. यह बिल्कुल वही घटना है जिसकी पुनरावृत्ति चाल है
\cref{ch:b2:metric}सप्ताहांत की समस्या वैश्विक में परिवर्तित हो जाती है
रैखिक अंतर समीकरणों की सॉल्वैबिलिटी।
\end{example}

\begin{theorem}[बानाच स्थानों में पूर्ण अभिसरण]\label{thm:b2:nvs:absoluteconvergence}
में एक \omterm{def:b2:nvs:banach}{बनच स्थान}, अगर $\sum \norm{u_n} < \infty $ तब $\sum u_n $
अभिसरण, और $\norm{\sum u_n} \leq \sum\norm{u_n}$. (पूर्ण
मानक स्थानों में श्रृंखला का सिद्धांत है \cref{ch:b2:series}.)
\end{theorem}

\begin{proof}
आंशिक रकम $ S_N $: के लिए $ q > p $, $\norm{S_q - S_p} \leq
\sum_{n=p+1}^{q}\norm{u_n}$, जिसकी प्रवृत्ति होती है $0$ (कौची मानदंड
की वास्तविक श्रृंखला के लिए मानदंड): $(S_N)$ कॉची है, इसलिए अभिसरण है।
असमानता परिमित त्रिभुज से सीमा तक गुजरती है
असमानता.
\end{proof}

\begin{example}[मैट्रिक्स घातांक, पहला संपर्क]\label{ex:b2:nvs:matrixexp}
 \emph{मैट्रिक्स घातांक}\index{मैट्रिक्स घातांक}:
$\mathcal{M}_n(K)$ किसी भी उपगुणक के साथ \omterm{def:b2:nvs:norm}{आदर्श} ($\vertiii{AB}
\leq \vertiii A \vertiii B $) बनच (परिमित आयाम) है। फिर, के लिए
प्रत्येक $ A $,
\[
\eu^A = \sum_{k=0}^{\infty} \frac{A^k}{k!}
\]
बिल्कुल एकाग्र होता है ($\vertiii{A^k/k!} \leq \vertiii A^k /k!$,
संक्षेपणीय): अच्छी तरह से परिभाषित। \cref{ch:b2:diffeq} इसका शोषण करता है
व्यवस्थित रूप से.
\end{example}

\begin{example}[एक न्यूमैन श्रृंखला जो समाप्त होती है]\label{ex:b2:nvs:neumannfinite}
के लिए $ A = \begin{psmallmatrix}0 & \frac12\\ 0 &
0\end{psmallmatrix}$: $\vertiii A < 1$ मेँ कोई \omterm{thm:b2:nvs:continuouslinear}{ऑपरेटर मानदंड}
बनाया गया \cref{ex:b2:nvs:examples}'एस मानदंड, और $ A^2 = 0$, तो
ज्यामितीय श्रृंखला ढह जाती है:
\[
(I - A)^{-1} = \sum_{k \geq 0} A^k = I + A =
\begin{pmatrix}1 & \tfrac12\\ 0 & 1\end{pmatrix},
\]
द्वारा सत्यापित $(I - A)(I + A) = I - A^2 = I $. निलपोटेंसी
श्रृंखला को ठीक वैसे ही छोटा करता है जैसे इसमें घातांक को छोटा किया गया था
\cref{ch:b2:reduction}; और उदाहरण अंशांकित होता है
अपेक्षाएँ: न्यूमैन व्युत्क्रम एक अनंत श्रृंखला है
सामान्यतः, एक बहुपद ठीक तब होता है जब क्षोभ होता है
निलपोटेंट, और उसके बाद त्रुटि $ N $ शर्तें हमेशा से बंधी होती हैं
ज्यामितीय पूंछ $\vertiii A^{N+1}/(1 - \vertiii A)$.
\end{example}

\begin{example}[एक घूर्णन जनरेटर का घातांक]\label{ex:b2:nvs:rotationexp}
लेना $ A = \begin{psmallmatrix}0 & -\theta\\ \theta &
0\end{psmallmatrix}$. तब $ A^2 = -\theta^2 I $, तो शक्तियां
\omterm{def:b2:structures:sn}{चक्र} अवधि चार के साथ, और श्रृंखला सम और विषम में विभाजित हो जाती है
भाग:
\[
\eu^{A} = \sum_{k}\frac{A^k}{k!}
= \Bigl(\sum_{j}\frac{(-1)^j\theta^{2j}}{(2j)!}\Bigr) I
+ \Bigl(\sum_{j}\frac{(-1)^j\theta^{2j+1}}{(2j+1)!}\Bigr)
\frac{A}{\theta}
= \begin{pmatrix}
\cos\theta & -\sin\theta\\
\sin\theta & \cos\theta
\end{pmatrix},
\]
सभी पुनर्व्यवस्थाओं को पूर्ण अभिसरण द्वारा लाइसेंस दिया जा रहा है। द
एक एंटीसिमेट्रिक जनरेटर का घातांक एक घूर्णन है ---
यहां पूरी तरह से श्रृंखला से तीन अध्याय पहले की गणना की गई है
विभेदक समीकरण $ x' = Ax $ (\cref{ch:b2:diffeq}) समझाता है
\emph{क्यों}: $\eu^{tA}$ एकसमान वृत्ताकार गति है। समापन
अंतर्दृष्टि: मैट्रिक्स श्रृंखला के बीच की पहचान बिल्कुल समान साबित होती है
अदिश वाले, एक बार उपगुणक \omterm{def:b2:nvs:norm}{आदर्श} पूर्ण प्रमाणित करता है
अभिसरण.
\end{example}

\begin{example}[सुपर नॉर्म का अर्थ है वर्दी: शब्दकोश]\label{ex:b2:nvs:uniformdictionary}
कथन $\norm{f_n - f}_\infty \to 0$ \emph{है} वर्दी
अभिसरण: एक संख्या, $\sup_x\abs{f_n(x) - f(x)}$, सीमा
हर बिंदु पर एक साथ त्रुटि. शब्दकोश में
पर कार्रवाई $ f_n(x) = x^n $ ऊपर $\intcc{0}{1}$:बिंदुवार, $ f_n
\to 0$ पर $\intco{0}{1}$ और $ f_n(1) = 1$; में \omterm{def:b2:nvs:norm}{आदर्श},
$\norm{f_n - 0}_\infty = 1 \not\to 0$, और वास्तव में बिंदुवार
सीमा असंतुलित है, इसलिए a की पहुंच से बाहर है
$\norm\cdot_\infty $-सीमा में $ C(\intcc01)$ (जो बंद है
एक समान सीमा के अंतर्गत, \cref{thm:b2:metric:rncomplete}). पर
$\intcc{0}{a}$, $ a < 1$: $\norm{f_n}_\infty = a^n \to 0$ ---
डोमेन को सिकोड़कर एक समान अभिसरण बहाल किया गया। हर
का अभिसरण कथन \cref{ch:b2:funcseq} एक कथन है
इस बारे में \omterm{def:b2:nvs:norm}{आदर्श}; शब्दकोश को आधा ध्यान में रखते हुए
वह अध्याय.
\end{example}

\begin{remark}[सामान्य ख़तरे]\label{rem:b2:nvs:pitfalls}
(i) एक \omterm{thm:b2:nvs:continuouslinear}{ऑपरेटर मानदंड} पर निर्भर करता है \emph{दोनों} चुना गया मानदंड: द
वही मैट्रिक्स है $\vertiii\cdot_\infty $ पंक्ति योग द्वारा दिया गया
(\cref{exo:b2:nvs:4}) और एक अलग $\vertiii\cdot_1$ (कॉलम
रकम); ``द'' को उद्धृत करते हुए \omterm{def:b2:nvs:norm}{आदर्श} नाम दिए बिना एक मैट्रिक्स का
अंतर्निहित मानदंड अर्थहीन है. (ii) $\vertiii{AB} \leq
\vertiii A\,\vertiii B $ एक असमानता है, आमतौर पर सख्त ---
शक्तियाँ सीमा से कहीं अधिक तेजी से सिकुड़ सकती हैं $\vertiii A^k $
सुझाव देता है, जो अनुकूलित का संपूर्ण बिंदु है मानदंड (यह
अध्याय की सप्ताहांत समस्या, प्रश्न 22)। (iii) ``रैखिक का तात्पर्य है
\omterm{def:b2:metric:continuity}{निरंतर}'' एक परिमित-आयामी विशेषाधिकार है: विभेदीकरण
बहुपदों पर रैखिक और असीमित है
(\cref{ex:b2:nvs:operatornorms}). (iv) का पूर्ण अभिसरण
$\sum u_n $ केवल तभी मदद करता है जब जगह हो \omterm{def:b2:metric:complete}{पूरा}
(\cref{exo:b2:series:9} प्रति उदाहरण बनाता है)। (v)इन
यूनिट बॉल पर अनंत आयाम का सर्वोच्च एक वास्तविक है
सर्वोच्च: यह मत समझो कि यह प्राप्त हो गया है
(\cref{exo:b2:nvs:8}).
\end{remark}

\begin{remark}[इस खंड के भीतर परिप्रेक्ष्य]\label{rem:b2:nvs:perspectives}
अब तीन नियुक्तियां तय हो गई हैं। साथ \cref{ch:b2:series}: में एक \omterm{def:b2:nvs:banach}{बनच स्थान}, बिल्कुल अभिसरण श्रृंखला अभिसरण, तो
ऑपरेटरों की ज्यामितीय और घातांकीय श्रृंखला प्रतिदिन बनती जाती है
उपकरण---उलटा करना $ I - A $, परिभाषित करना $\eu^{A}$
(\cref{ex:b2:series:neumann}). साथ
\cref{ch:b2:funcseq} और \cref{ch:b2:powerseries}: अभिसरण
फ़ंक्शन अनुक्रम और पावर श्रृंखला का अभिसरण होता है
$\bigl(C, \norm\cdot_\infty\bigr)$
(\cref{ex:b2:nvs:uniformdictionary}), और की त्रिज्या
अभिसरण किस ज्यामितीय श्रृंखला के बारे में एक कथन है
प्रभुत्व. साथ \cref{ch:b2:fourier}: द मानदंड
$\norm\cdot_2$ और $\norm\cdot_\infty $ वास्तव में असहमत हैं
$ C(\intcc{0}{1})$ (\cref{ex:b2:nvs:sqrtnxn}), जो बिल्कुल सही है
फूरियर श्रृंखला और वर्दी का माध्य-वर्ग अभिसरण क्यों
अभिसरण दो अलग-अलग प्रमेय हैं जिनमें दो भिन्न प्रमेय हैं
कीमतें.
\end{remark}

\begin{remark}[जहाँ इस अध्याय का प्रयोग किया जाता है]
\omterm{thm:b2:nvs:continuouslinear}{ऑपरेटर मानदंड} और ज्यामितीय श्रृंखला गड़बड़ी को संचालित करती है
के तर्क \cref{ch:b2:diffcalc} (व्युत्क्रम फलन प्रमेय) और
द \omterm{ex:b2:nvs:matrixexp}{मैट्रिक्स घातांक} का \cref{ch:b2:diffeq}; की समतुल्यता
मानदंड चुपचाप प्रत्येक ``अपना पसंदीदा चुनें'' को अधिकृत करता है \omterm{def:b2:nvs:norm}{आदर्श}''
में तर्क \cref{ch:b2:funcseq} और इसके बाद में; और
रिज़्ज़ के प्रमेय का परिमित/अनंत विभाजन --- मात्रात्मक बनाया गया
इस अध्याय की सप्ताहांत समस्या --- वर्ष 3 का कारण है
वॉल्यूम को नए टूल की आवश्यकता है (कमजोर अभिसरण, आर्ज़ेल\`ए--अस्कोली,
हिल्बर्ट अंतरिक्ष प्रक्षेपण) जहां यह मात्रा अभी भी निकाली जा सकती है
अभिसरण अनुवर्ती.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b2:nvs:1}
पर $\R^2$, की इकाई गेंदें बनाएं $\norm\cdot_1$, $\norm\cdot_2$,
$\norm\cdot_\infty $, और असमानताओं को सिद्ध करें $\norm x_\infty
\leq \norm x_2 \leq \norm x_1 \leq 2\norm x_\infty $ सर्वोत्तम के साथ
आयाम में स्थिरांक $2$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:nvs:2}
है $ N(f) = \abs{f(0)} + \norm{f'}_\infty $ ए \omterm{def:b2:nvs:norm}{आदर्श} पर
$ C^1(\intcc{0}{1})$? इसकी तुलना करें $\norm{f}_\infty $: एक
असमानता कायम रहती है, अन्य विफल रहता है (प्रदर्शन)।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:nvs:3}
की गणना करें \omterm{thm:b2:nvs:continuouslinear}{ऑपरेटर मानदंड} का $ u(f) = \int_0^1 f(t)\,\eu^t\,\dd t $
पर $\bigl(C(\intcc{0}{1}), \norm\cdot_\infty\bigr) \to \R $, और का
बदलाव $ S(x_1, x_2, \dots, x_n) = (x_2, \dots, x_n, 0)$ पर
$(K^n, \norm\cdot_\infty)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:nvs:4}
पर $(\R^n, \norm\cdot_\infty)$, सिद्ध करें कि \omterm{thm:b2:nvs:continuouslinear}{ऑपरेटर मानदंड} एक का
मैट्रिक्स $ A $ है $\vertiii A_\infty = \max_i \sum_j \abs{a_{ij}}$ (द
सबसे बड़ी निरपेक्ष पंक्ति योग)। इसके लिए गणना करें $\begin{pmatrix} 1 & -2\\
3 & 1\end{pmatrix}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:nvs:5}
साबित करें कि $ GL_n(K)$ है \omterm{def:b2:metric:topology}{खुला} में $\mathcal{M}_n(K)$ ओर वो $ A
\mapsto A^{-1}$ है \omterm{def:b2:metric:continuity}{निरंतर} उस पर. \emph{संकेत: खुलेपन के लिए, यदि
$\vertiii H < \frac{1}{\vertiii{A^{-1}}}$ तब $ A + H = A(I +
A^{-1}H)$ साथ $\vertiii{A^{-1}H} < 1$, और $ I + B $ उलटा है
के लिए $\vertiii B < 1$ ज्यामितीय श्रृंखला द्वारा
(\cref{thm:b2:nvs:absoluteconvergence}); के लिए निरंतरता, अवश्यंभावी
$(A+H)^{-1} - A^{-1}$ उसी श्रृंखला का उपयोग करना।}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:nvs:6}
होने देना $\varphi $ एक मानक स्थान पर एक रैखिक रूप बनें $ E $. साबित करें कि
$\varphi $ है \omterm{def:b2:metric:continuity}{निरंतर} अगर और केवल अगर $\ker\varphi $ बन्द है।
\emph{(अगर $\ker\varphi $ बंद है और $\varphi \neq 0$, चुनना $ a $
साथ $\varphi(a) = 1$ और $ r > 0$ साथ $ B(a, r) \cap \ker\varphi =
\emptyset $; परिणाम निकालना $\abs{\varphi(h)} \leq \frac{1}{r}\norm h $ ए द्वारा
स्केलिंग तर्क चालू $ a - \frac{h}{\varphi(h)}$.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:nvs:7}
साबित करें कि $\bigl(C(\intcc{0}{1}), \norm\cdot_1\bigr)$ क्या नहीं है
\omterm{def:b2:metric:complete}{पूरा}:दिखाएँ कि कार्य $ f_n $, रैंप से जुड़ें $0$ को
$1$ ऊपर $\bigl[\frac12 - \frac1n, \frac12\bigr]$ (कीमत $0$
पहले, $1$ बाद में), संख्या के साथ एक कॉची अनुक्रम बनाएं \omterm{def:b2:metric:continuity}{निरंतर}
$\norm\cdot_1$-सीमा.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:nvs:8}
पर $ E = C(\intcc{0}{1})$ साथ $\norm\cdot_\infty $, विचार करना
\[
\varphi(f) = \sum_{n \geq 1} (-1)^n\, 2^{-n} f\bigl(\tfrac1n\bigr).
\]
साबित करें कि $\varphi $ एक अच्छी तरह से परिभाषित है \omterm{def:b2:metric:continuity}{निरंतर} रैखिक रूप के साथ
$\vertiii\varphi = 1$, लेकिन वह सर्वोच्च परिभाषा है
$\vertiii\varphi $ है \emph{प्राप्त नहीं हुआ} बंद यूनिट बॉल पर.
\emph{(ऊपरी सीमा: त्रिकोण असमानता। \omterm{def:b2:nvs:norm}{आदर्श} $= 1$: निर्माण
\omterm{def:b2:metric:continuity}{निरंतर} $ f_K $ साथ $\norm{f_K}_\infty \leq 1$ और $ f_K(\frac1n)
= (-1)^n $ के लिए $ n \leq K $ --- बिन्दु $\frac1n $ से पृथक हैं
एक दूसरे. अप्राप्ति: समता बल देगी $ f(\frac1n) =
(-1)^n $ हरएक के लिए $ n $, के साथ असंगत निरंतरता का $ f $ पर $0$
तब से $\frac1n \to 0$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:nvs:9}
होने देना $ E $ एक मानक स्थान बनें जिसमें बंद इकाई गेंद हो \omterm{def:b2:metric:compact}{कॉम्पैक्ट}.
बिना उद्धृत किये पुनः प्राप्त करें \cref{thm:b2:nvs:riesz}, वह हर
परिबद्ध अनुक्रम में एक अभिसरण अनुवर्ती होता है, और सिद्ध करें कि प्रत्येक
रैखिक रूप पर $ E $ है \omterm{def:b2:metric:continuity}{निरंतर} अगर और केवल अगर $\dim E < \infty $.
\emph{(अनंत आयाम के लिए, एक असंतत रूप का निर्माण करें
इसे एक रैखिक रूप से स्वतंत्र सामान्यीकृत अनुक्रम पर स्वतंत्र रूप से परिभाषित करना
और विस्तार --- बीजगणित के अस्तित्व को स्वीकार करना
पूरक.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:nvs:10}
पर $ C(\intcc{0}{1})$, सिद्ध करना $\norm f_1 \leq \norm f_2 \leq \norm
f_\infty $ \emph{(कौची--श्वार्ज़ प्रथम)}, और साथ दिखाओ
परिवार $ f_n(x) = x^n $ कि किसी भी असमानता को उलटा नहीं किया जा सकता है
एक स्थिरांक तक: तीन मानदंड जोड़ीवार हैं गैर बराबर.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:nvs:11}
(हाइपरप्लेन से दूरी) चलो $\varphi $ शून्येतर हो \omterm{def:b2:metric:continuity}{निरंतर}
एक मानक स्थान पर रैखिक रूप $ E $. साबित करें कि
\[
d\bigl(x, \ker\varphi\bigr) =
\frac{\abs{\varphi(x)}}{\vertiii\varphi}
\qquad (x \in E),
\]
और जाँच करें \cref{exo:b2:nvs:8} कि न्यूनतम की आवश्यकता नहीं है
हाइपरप्लेन के किसी भी बिंदु द्वारा प्राप्त किया गया।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:nvs:12}
पर $ E = \R[X]$ (सभी बहुपद), चलो $ N_1(P) =
\sup_{\intcc{0}{1}}\abs P $ और $ N_2(P) =
\sup_{\intcc{0}{2}}\abs P $. बताते हैं कि $ N_1 \leq N_2$ लेकिन उस
$ N_1$ और $ N_2$ हैं \emph{नहीं} \omterm{def:b2:nvs:equivalent}{समकक्ष}; यह निष्कर्ष निकालें कि
पहचान $(E, N_2) \to (E, N_1)$ एक है \omterm{def:b2:metric:continuity}{निरंतर} रैखिक आक्षेप
जिसका व्युत्क्रम असंतत है। अंततः वह दिखाओ $(E, N_1)$ क्या नहीं है \omterm{def:b2:metric:complete}{पूरा} \emph{(टेलर का आंशिक योग $\eu^x $)}. तीनों
परिमित आयाम में घटनाएँ असंभव हैं---कहें क्यों।
\end{exercise}

\section{समस्या: सर्वोत्तम सन्निकटन और चेबीशेव का प्रमेय}

किसी दिए गए बहुपद द्वारा किसी फलन का कितनी अच्छी तरह अनुमान लगाया जा सकता है
डिग्री, और कौन सा बहुपद इसे सर्वोत्तम बनाता है? अस्तित्व की ओर,
उत्तर इस अध्याय से संबंधित है: सघनता परिमित में
आयाम सर्वोत्तम सन्निकटन अस्तित्व में लाता है। स्पष्ट पक्ष पर,
एक गैर-तुच्छ मामले को नंगे हाथों से पूरी तरह से हल किया जा सकता है ---
सबके बीच \emph{राक्षसी} डिग्री के बहुपद $ n $, में से एक
सबसे छोटा समर्थन \omterm{def:b2:nvs:norm}{आदर्श} पर $\intcc{-1}{1}$ (सामान्यीकृत) चेबीशेव है
बहुपद, का \omterm{def:b2:nvs:norm}{आदर्श} $2^{1-n}$: \emph{चेबीशेव का चरम
प्रमेय}. समस्या दोनों पक्षों को साबित करती है, फिर कितनी बुरी तरह मापती है
सघनता अनंत आयाम में विफल रहता है: की इकाई गेंद
$ C(\intcc{0}{1})$ इसमें बिंदुओं के अनंत तारामंडल शामिल हैं
आपसी दूरी $1$.

\begin{problem}[{सप्ताहांत समस्या---चेबीशेव का चरम प्रमेय
और यूनिट बॉल की ज्यामिति}]
\label{pb:b2:nvs:1}
मानदंड बिना सबस्क्रिप्ट के सुपर हैं मानदंड संकेतित खंड पर.

\medskip
\noindent\textbf{Part I --- Best approximation in normed
spaces.}
\begin{enumerate}
  \item होने देना $ F $ एक मानक का एक परिमित-आयामी उपस्थान बनें
        अंतरिक्ष $ E $ और $ x \in E $. दूरी सिद्ध करें $ d(x,
        F) = \inf_{f \in F}\norm{x - f}$ है \emph{प्राप्त}
        \emph{(एक बंद परिबद्ध उपसमुच्चय को कम करें $ F $ और उपयोग करें
        \cref{thm:b2:nvs:finitedim})}.
  \item ए \omterm{def:b2:nvs:norm}{आदर्श} है \emph{सख्ती से उत्तल} कब $\norm u = \norm v
        = 1$ और $ u \neq v $ संकेत करना $\bigl\Vert\frac{u +
        v}2\bigr\Vert < 1$. बताते हैं कि $\norm\cdot_2$ पर $\R^n $
        सख्ती से उत्तल है \emph{(समानांतर चतुर्भुज पहचान)}, ओर वो $\norm\cdot_1$ और $\norm\cdot_\infty $ के लिए नहीं हैं
        $ n \geq 2$.
  \item इसे सख्ती से उत्तल के लिए साबित करें \omterm{def:b2:nvs:norm}{आदर्श}, सर्वोत्तम
        प्रश्न 1 का सन्निकटन है \emph{अद्वितीय}.
  \item में $(\R^2, \norm\cdot_\infty)$, सर्वोत्तम गणना करें
        का अनुमान $ x = (0, 1)$ लाइन से $ F =
        \operatorname{Vect}\bigl((1,0)\bigr)$: का एक अंतराल
        न्यूनतम करनेवाला।
  \item में $\bigl(C(\intcc{a}{b}), \norm\cdot_\infty\bigr)$,
        दिखाएँ कि इसका सबसे अच्छा सन्निकटन है $ f $ द्वारा
        \emph{स्थिरांक} अद्वितीय है, बराबर है $ c^* = \frac{\max f
        + \min f}{2}$, दूरी के साथ $\frac{\max f - \min
        f}{2}$; दोनों के लिए गणना करें $ f(x) = x^2$ पर
        $\intcc{0}{1}$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Chebyshev polynomials.}
\begin{enumerate}[resume]
  \item दिखाएँ कि वास्तव में एक बहुपद है $ T_n $ साथ
        $ T_n(\cos\theta) = \cos n\theta $ सभी के लिए $\theta $
        \emph{(पुनरावृत्ति $ T_{n+1} = 2XT_n - T_{n-1}$ से
        कोसाइन जोड़ सूत्र)}, वह $\deg T_n = n $, और वह
        इसका अग्रणी गुणांक है $2^{n-1}$ के लिए $ n \geq 1$.
  \item दिखाओ $\abs{T_n} \leq 1$ पर $\intcc{-1}{1}$, साथ
        $ T_n(\eta_k) = (-1)^k $ पर $ n + 1$ अंक $\eta_k =
        \cos\frac{k\pi}{n}$ ($ k = 0, \dots, n $), और वह
        की जड़ें $ T_n $ हैं $ n $ अंक
        $\cos\frac{(2k-1)\pi}{2n}$, इंटरलेसिंग $\eta_k $.
  \item गणना करना $ T_2, T_3, T_4$, और के विकल्प को सत्यापित करें
        $ T_3$ पर $\eta_0, \dots, \eta_3 = 1, \frac12, -\frac12,
        -1$ प्रत्यक्ष मूल्यांकन द्वारा.
  \item के लिए $\abs x \geq 1$, सिद्ध करना
        \[
        T_n(x) = \frac{\bigl(x + \sqrt{x^2 - 1}\bigr)^n +
        \bigl(x - \sqrt{x^2 - 1}\bigr)^n}{2},
        \]
        और निष्कर्ष निकालें $ T_n(x) \sim \frac12\bigl(x + \sqrt{x^2 -
        1}\bigr)^n \to \infty $ निश्चित के लिए ज्यामितीय रूप से $ x > 1$.
  \item संघटन नियम सिद्ध करें $ T_m \circ T_n = T_{mn}$
        \emph{(पर जाँचा $\intcc{-1}{1}$ और कठोरता का आह्वान करें
        बहुपदों का)}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Chebyshev's extremal theorem.}
लिखना $ Q_n = 2^{1-n}T_n $ (मोनिक, प्रश्न 6 द्वारा)।
\begin{enumerate}[resume]
  \item होने देना $ P $ डिग्री के प्रति उदासीन रहें $ n \geq 1$ साथ
        $\sup_{\intcc{-1}{1}}\abs P < 2^{1-n}$. मूल्यांकन करके $ D
        = Q_n - P $ बिंदुओं पर $\eta_k $ और गिनती का चिह्न
        परिवर्तन, एक विरोधाभास उत्पन्न करता है। निष्कर्ष:
        \[
        \sup_{\intcc{-1}{1}}\abs P \;\geq\; 2^{1-n}
        \qquad\text{for every monic } P \text{ of degree } n.
        \]
  \item (समानता का मामला) मान लीजिए $\sup_{\intcc{-1}{1}}\abs P =
        2^{1-n}$ साथ $ P $ डिग्री का मोनिक $ n $, और चलो $ D = Q_n
        - P \neq 0$. दिखाओ $(-1)^kD(\eta_k) \geq 0$ सभी के लिए $ k $;
        दिखाएँ कि प्रत्येक $ n $ अंतराल
        $\intcc{\eta_{k}}{\eta_{k-1}}$ का एक शून्य शामिल है $ D $,
        और यह कि दो लगातार अंतरालों द्वारा साझा किया गया एक शून्य है
        एक \emph{आंतरिक भाग} बिंदु $\eta_k $ कहाँ $ D' = 0$ भी। समाप्त करें कि $ D $ है $ n $ शून्य के साथ गिना गया
        बहुलता, इसलिए $ D = 0$: मिनिमाइज़र बिल्कुल सही है
        $ Q_n $ --- \emph{चेबीशेव का चरम प्रमेय}.
  \item प्रमेय को दूरी के रूप में पुनः दोहराएँ: पर $\intcc{-1}{1}$,
        \[
        d_\infty\bigl(X^n,\ \R_{n-1}[X]\bigr) = 2^{1-n},
        \]
        अद्वितीय सर्वोत्तम सन्निकटन के साथ $ X^n - Q_n $; और दिखाओ
        एफ़िन प्रतिस्थापन $ x = \frac{1+t}2$ उस पर
        $\intcc{0}{1}$ दूरी बन जाती है $2^{1-2n}$.
  \item (इष्टतम इंटरपोलेशन नोड्स) के लिए $ n $ नोड्स $ x_1, \dots,
        x_n \in \intcc{-1}{1}$, नोड बहुपद
        $\omega(x) = \prod_i(x - x_i)$ डिग्री का राक्षसी है $ n $.
        प्रश्न 12 से निष्कर्ष निकालें कि नोड्स का कौन सा विकल्प न्यूनतम करता है
        $\sup_{\intcc{-1}{1}}\abs\omega $, नोड-निर्भर
        शास्त्रीय प्रक्षेप त्रुटि का कारक बाध्य, और
        न्यूनतम मूल्य दें.
  \item मामले का सत्यापन करें $ n = 2$ हाथ से प्रमेय का (खोजें)
        $\inf_c \sup_{\intcc{-1}{1}}\abs{x^2 - c}$ सीधे),
        और प्रश्न 13 की दूरी की संख्यात्मक गणना करें
        $\intcc{0}{1}$ के लिए $ n = 10$. इसका साइज क्या कहता है
        के ग्राफ के बारे में $ x^{10}$?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The unit ball of
$ C(\intcc{0}{1})$.}
\begin{enumerate}[resume]
  \item होने देना $ g_k(x) = x^{2^k}$. दिखाओ $\norm{g_k}_\infty = 1$ और
        $\norm{g_k - g_j}_\infty \geq \frac14$ के लिए $ j > k $
        \emph{(उस बिंदु पर मूल्यांकन करें जहां $ x^{2^k} =
        \frac12$)}: संख्या के साथ एक स्पष्ट परिबद्ध अनुक्रम
        अभिसरण अनुवर्ती --- बंद इकाई गेंद नहीं है
        \omterm{def:b2:metric:compact}{कॉम्पैक्ट}, नंगे हाथों से.
  \item (रिस्ज़ की लेम्मा, तेज) चलो $ F $ एक हो
        \emph{परिमित-आयामी} एक मानक का उचित उपस्थान
        अंतरिक्ष $ E $. प्रश्न 1 का उपयोग करके, एक इकाई वेक्टर तैयार करें $ x $
        साथ $ d(x, F) = 1$ बिल्कुल --- बस नहीं $\geq 1 -
        \varepsilon $ के रूप में \cref{thm:b2:nvs:riesz}की लेम्मा.
  \item निष्कर्ष निकालें: प्रत्येक अनंत-आयामी मानक स्थान में
        जोड़ीदार दूरियों के साथ इकाई सदिशों का एक क्रम है
        $\geq 1$, और इससे रिज़्ज़ के प्रमेय को पुनः प्राप्त करें।
  \item में $ C(\intcc{0}{1})$, ऐसे नक्षत्र का प्रदर्शन करें
        स्पष्ट रूप से: तम्बू कार्य करता है $ h_n $ पर समर्थित
        $\bigl[\frac1{n+1}, \frac1n\bigr]$ चरम मूल्य के साथ $1$.
        सत्यापित करें $\norm{h_n} = 1$, $\norm{h_n - h_m} = 1$ के लिए $ n
        \neq m $, और उस पर ध्यान दें $ h_n \to 0$ बिंदुवार लेकिन नहीं
        समान रूप से.
  \item (कुल सीमा विफल हो जाती है) दिखाएँ कि बंद इकाई गेंद
        का $ C(\intcc{0}{1})$ बहुत से लोगों द्वारा कवर नहीं किया जा सकता
        त्रिज्या की गेंदें $\frac13$ \emph{(ऐसी प्रत्येक गेंद में शामिल है
        अधिक से अधिक एक $ h_n $)} --- के साथ विरोधाभास
        के प्रमाण में कुल-बद्धता चरण
        \cref{thm:b2:metric:borellebesgue}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- मानदंड at work on matrices, and
synthesis.}
\begin{enumerate}[resume]
  \item प्रत्येक को सिद्ध करें \omterm{def:b2:reduction:eigen}{eigenvalue} $\lambda $ का $ A \in
        \mathcal{M}_n(\C)$ संतुष्ट $\abs\lambda \leq
        \vertiii A $ हरएक के लिए \omterm{thm:b2:nvs:continuouslinear}{ऑपरेटर मानदंड}; आवेदन करना
        \cref{exo:b2:nvs:4} बाध्य करने के लिए eigenvalues का
        $\begin{psmallmatrix}1 & -2\\ 3 & 1\end{psmallmatrix}$
        और उनके वास्तविक मापांक से तुलना करें।
  \item (अनुकूलित मानदंड) होने देना $ A $ होना \omterm{def:b2:reduction:diag}{विकर्णीय}, $ A =
        P\,\mathrm{diag}(\lambda_1, \dots, \lambda_n)\,P^{-1}$.
        दिखाओ $ N_P(x) = \norm{P^{-1}x}_\infty $ एक है \omterm{def:b2:nvs:norm}{आदर्श} किसका
        \omterm{thm:b2:nvs:continuouslinear}{ऑपरेटर मानदंड} संतुष्ट $\vertiii A_{N_P} =
        \max_i\abs{\lambda_i}$.
  \item निष्कर्ष: के लिए \omterm{def:b2:reduction:diag}{विकर्णीय} $ A $, $ A^k \to 0$ यदि और केवल
        यदि सभी eigenvalues संतुष्ट $\abs{\lambda_i} < 1$ ---
        मानदंडों की समानता निष्कर्ष निकालता है
        मानक-स्वतंत्र। पर जाँचा $ A =
        \frac14\begin{psmallmatrix}1 & 2\\ 2 &
        1\end{psmallmatrix}$.
  \item (समतुल्यता स्थिरांक उड़ जाते हैं) चालू $\R_n[X]$, तुलना करना
        $ N_c(P) = \max_k \abs{a_k}$ (गुणांक) और
        $\norm{P}_{\intcc{0}{1}}$: दोनों हैं मानदंड, तो वे हैं
        \omterm{def:b2:nvs:equivalent}{समकक्ष} प्रत्येक के लिए निश्चित $ n $; लेकिन मोनिक का उपयोग करके दिखाओ
        प्रश्न 13 का मिनिमाइज़र $\intcc{0}{1}$, कि
        सर्वोत्तम स्थिरांक $ C_n $ में $ N_c \leq
        C_n\norm\cdot_{\intcc{0}{1}}$ संतुष्ट $ C_n \geq
        2^{2n-1}$. एक वाक्य में निष्कर्ष निकालें कि ``सभी'' क्यों मानदंड हैं
        \omterm{def:b2:nvs:equivalent}{समकक्ष}'' अनंत आयाम में मर जाता है।
  \item (संश्लेषण) एक-एक वाक्य: कहाँ सघनता का
        परिमित-आयामी गेंदों ने काम किया (प्रश्न 1, 12); क्या
        सख्त उत्तलता नियंत्रित करती है; का नक्षत्र क्या है
        प्रश्न 18--19 नष्ट करता है; और कैसे प्रश्न 24
        विफलता को मापता है। शिखर का नाम बताइए (चेबीशेव का)।
        चरम प्रमेय) और बताएं कि सबसे अच्छा सन्निकटन कहाँ है
        अपना आधुनिक घर ढूंढता है (हिल्बर्ट पर प्रक्षेपण प्रमेय)।
        रिक्त स्थान, वर्ष 3 खंड, कहाँ पूर्णता के स्थान पर
        सघनता).
\end{enumerate}
