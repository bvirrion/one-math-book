\chapter{सेट और संरचनाएँ}\label{ch:b2:structures}

यह आरंभिक अध्याय वर्ष 1 में रखी गई नींव को धारदार बनाता है
व्यापार के कामकाजी उपकरणों में मात्रा: सेट की गणना और
भागफल, अनंत सेटों की तुलना (गणनीयता,
कैंटर--बर्नस्टीन), और समूहों और छल्लों का संरचनात्मक सिद्धांत ---
लैग्रेंज प्रमेय, अपने \omterm{thm:b2:structures:signature}{हस्ताक्षर} के साथ सममित समूह, \omterm{def:b2:structures:ideal}{आदर्श}ों और
चीनी शेषफल प्रमेय. यहां हर चीज़ का निरंतर उपयोग किया जाता है
पुस्तक का शेष भाग: \omterm{thm:b2:structures:signature}{हस्ताक्षर} निर्धारक का निर्माण करता है
(\cref{ch:b2:linalg}), भागफल के छल्ले ड्राइव अंकगणित, और
गणनीयता टोपोलॉजी और संभाव्यता दोनों को रेखांकित करता है।

\section{समुच्चय, मानचित्र, भागफल}

हम सेट, मानचित्र और समतुल्यता आदि की भाषा का स्वतंत्र रूप से उपयोग करते हैं \omterm{def:b2:structures:generated}{आदेश}
वर्ष 1 खंड में स्थापित संबंध। दो उन्नयन उचित के पात्र हैं
बयान.

\begin{proposition}[परिवारों की छवियाँ और पूर्वछवियाँ]\label{prop:b2:structures:images}
होने देना $f \colon E \to F $ और चलो $(A_i)_{i \in I}$, $(B_j)_{j \in J}$
के उपसमूहों के परिवार बनें $ E $, सम्मान।\ $ F $. तब
\[
f^{-1}\Bigl(\bigcup_j B_j\Bigr) = \bigcup_j f^{-1}(B_j),
\qquad
f^{-1}\Bigl(\bigcap_j B_j\Bigr) = \bigcap_j f^{-1}(B_j),
\qquad
f^{-1}(F \setminus B) = E \setminus f^{-1}(B),
\]
\[
f\Bigl(\bigcup_i A_i\Bigr) = \bigcup_i f(A_i),
\qquad
f\Bigl(\bigcap_i A_i\Bigr) \subseteq \bigcap_i f(A_i)
\quad (\text{equality for injective } f).
\]
\end{proposition}

\begin{proof}
प्रत्येक पहचान परिभाषाओं का एक प्रकार है; उदाहरण के लिए $ x \in
f^{-1}(\bigcap B_j) \iff f(x) \in B_j $ सभी के लिए $ j $ $\iff x \in
f^{-1}(B_j)$ सभी के लिए $ j $. छवि की पहचान और विफलता
प्रतिच्छेदन मामले में समानता (इंजेक्टिविटी फिक्स के साथ) थी
दो सेटों के लिए वर्ष 1 खंड में सिद्ध; तर्क समान हैं
परिवारों के लिए.
\end{proof}

\begin{example}[जहां छवि समावेशन सख्त है]\label{ex:b2:structures:imagestrict}
लेना $ f \colon \R \to \R $, $ f(x) = x^2$, साथ $ A_1 =
\intcc{-1}{0}$ और $ A_2 = \intcc{0}{1}$. तब
\[
f(A_1 \cap A_2) = f(\{0\}) = \{0\},
\qquad
f(A_1) \cap f(A_2) = \intcc{0}{1} \cap \intcc{0}{1} =
\intcc{0}{1} :
\]
का समावेश \cref{prop:b2:structures:images} उतना ही सख्त है
जैसा कि हो सकता है --- दो पूर्वछवि बिंदु $\pm x $ एक सामान्य मूल्य का
अलग-अलग रहते हैं $ A_i $. इंजेक्शन बिल्कुल वही है जो मना करता है
यह विभाजन, यही कारण है कि पूर्वछवियाँ (जो कभी विलीन नहीं होतीं
अंक) सभी चार पहचानों को बिना शर्त संतुष्ट करते हैं
छवियां चौराहों के बारे में खो देती हैं। के लिए अंगूठे का नियम
पूरी किताब: धक्का \emph{पूर्वछवियाँ} सेट ऑपरेशन के माध्यम से
स्वतंत्र रूप से; छवियों को सावधानी से संभालें.
\end{example}

\begin{definition}[भागफल सेट]\label{def:b2:structures:quotient}
होने देना $\mathcal{R}$ पर एक तुल्यता संबंध हो $ E $.
\emph{भागफल सेट}\index{भागफल सेट} $ E/\mathcal{R}$ का सेट है
तुल्यता वर्ग; अनुमान $\pi \colon E \to E/\mathcal{R}$,
$ x \mapsto \mathrm{cl}(x)$, है \emph{विहित
प्रक्षेपण}\index{विहित
प्रक्षेपण}.

\emph{सार्वभौमिक संपत्ति (कारकीकरण):} अगर $ f \colon E \to F $ है
\emph{अनुकूल} साथ $\mathcal{R}$ (अर्थात्\ $ x \mathbin{\mathcal{R}}
y \implies f(x) = f(y)$), बिल्कुल एक नक्शा है $\overline f
\colon E/\mathcal{R} \to F $ साथ $ f = \overline f \circ \pi $.
\end{definition}

\begin{proof}[सार्वभौमिक संपत्ति का प्रमाण]
विशिष्टता: आवश्यकता $ f = \overline f \circ \pi $ पढ़ता
\[
\overline f\bigl(\mathrm{cl}(x)\bigr) = f(x)
\qquad (x \in E),
\]
और तबसे $\pi $ विशेषण है, प्रत्येक तत्व
$ E/\mathcal{R}$ कुछ है $\mathrm{cl}(x)$: के मान
$\overline f $ सब मजबूर हैं. अस्तित्व: प्रदर्शन को इस रूप में लें
द \emph{परिभाषा} का $\overline f $; यह स्पष्ट है
ठीक अनुकूलता से --- यदि $\mathrm{cl}(x) =
\mathrm{cl}(y)$, तब $ x \mathbin{\mathcal{R}} y $, इसलिए $ f(x) =
f(y)$ और दो उम्मीदवार मान सहमत हैं --- और यह कारक बनता है
$ f $ निर्माण द्वारा. श्रम के विभाजन पर ध्यान दें: प्रत्यक्षता
का $\pi $ विशिष्टता देती है, अनुकूलता अस्तित्व देती है।
\end{proof}

\begin{example}
$\Z/n\Z $ का भागफल है $\Z $ सर्वांगसमता मॉड्यूलो द्वारा $ n $; द
वर्ष 1 खंड की सुस्पष्टता जाँच इसके उदाहरण थे
सार्वभौमिक संपत्ति. भागफल ``संगत निर्माणों को चालू करते हैं
प्रतिनिधियों'' को ईमानदार मानचित्रों में --- हम इसका उपयोग लगातार नीचे करते हैं।
\end{example}

\section{गणनीयता और प्रमुखता}

\begin{definition}[साम्यशक्ति, गणनीयता]\label{def:b2:structures:countable}
दो सेट हैं \emph{सुसज्जित}\index{सुसज्जित} जब एक आपत्ति
उनसे जुड़ जाता है. एक सेट है \emph{गणनीय}\index{गणनीय} जब यह है
से सुसज्जित $\N $ (कुछ लेखकों में परिमित सेट शामिल हैं; हम कहते हैं
\emph{अधिक से अधिक गणनीय} ``परिमित या गणनीय'' के लिए)।
\end{definition}

\begin{proposition}[स्थिरता गुण]\label{prop:b2:structures:countablestable}
\begin{enumerate}
  \item का प्रत्येक अनंत उपसमुच्चय $\N $ है \omterm{def:b2:structures:countable}{गणनीय}; एक सेट अधिकतम है
        \omterm{def:b2:structures:countable}{गणनीय} अगर यह इंजेक्ट करता है $\N $ यदि यह खाली है या a
        की विशेषण छवि $\N $.
  \item $\N \times \N $ है \omterm{def:b2:structures:countable}{गणनीय}; अधिकतम दो का एक उत्पाद
        \omterm{def:b2:structures:countable}{गणनीय} समुच्चय अधिकतम है \omterm{def:b2:structures:countable}{गणनीय}.
  \item अधिक से अधिक \omterm{def:b2:structures:countable}{गणनीय} अधिक से अधिक का मिलन \omterm{def:b2:structures:countable}{गणनीय} समुच्चय पर है
        सबसे \omterm{def:b2:structures:countable}{गणनीय}.
  \item $\Z $ और $\Q $ हैं \omterm{def:b2:structures:countable}{गणनीय}.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) अनंत की सूची बनाएं $ A \subseteq \N $ बार-बार मिनिमा द्वारा: $ a_0 =
\min A $, $ a_{k+1} = \min\,(A \setminus \{a_0, \dots, a_k\})$
(तब से खाली नहीं है $ A $ अनंत है); वो नक्शा $ k \mapsto a_k $ सख्ती से है
बढ़ाना, विशेषण, और विशेषण पर $ A $ (प्रत्येक $ a \in A $
के केवल सीमित रूप से कई तत्वों से अधिक है $ A $, तो यह पहुंच गया है)। अगर $ E $
में इंजेक्ट करता है $\N $ के जरिए $\varphi $, तब $ E $ है \omterm{def:b2:structures:countable}{सुसज्जित} को
$\varphi(E) \subseteq \N $: परिमित या \omterm{def:b2:structures:countable}{गणनीय}. अगर $ s \colon \N \to
E $ तो, विशेषण है $ x \mapsto \min s^{-1}(\{x\})$ इंजेक्शन $ E $
में $\N $.

(2) नक्शा $(p, q) \mapsto 2^p(2q + 1) - 1$ एक आक्षेप है $\N^2
\to \N $ (प्रत्येक सकारात्मक पूर्णांक में एक अद्वितीय विषम-सम विभाजन होता है
$2^p m $ साथ $ m $ विषम, अद्वितीय गुणनखंडन द्वारा)। उत्पाद: लिखें
इंजेक्शन.

(3) दिए गए सेट $ E_n $ अनुमान के साथ $ s_n \colon \N \to E_n $
(हानिकारक जब कुछ $ E_n $ परिमित है: मान दोहराएँ), मानचित्र $(n, k)
\mapsto s_n(k)$ का एक अनुमान है \omterm{def:b2:structures:countable}{गणनीय} $\N^2$ पर
$\bigcup E_n $.

(4) $\Z = \N \cup (-\N^*)$: \omterm{def:b2:structures:countable}{गणनीय} संघ. $\Q $ एक विशेषण है
की छवि $\Z \times \N^*$ (अंश मानचित्र), इसलिए अधिक से अधिक
\omterm{def:b2:structures:countable}{गणनीय}, और अनंत.
\end{proof}

\begin{example}[एक युग्मन फलन, परिकलित]\label{ex:b2:structures:pairing}
आपत्ति $(p, q) \mapsto 2^p(2q + 1) - 1$ प्रमाण का
काम पर देखे जाने योग्य है। इसका पहला मान:
\[
\begin{array}{c|ccccc}
 & q = 0 & q = 1 & q = 2 & q = 3 & q = 4\\
\hline
p = 0 & 0 & 2 & 4 & 6 & 8\\
p = 1 & 1 & 5 & 9 & 13 & 17\\
p = 2 & 3 & 11 & 19 & 27 & 35\\
p = 3 & 7 & 23 & 39 & 55 & 71
\end{array}
\]
पंक्ति $ p $ पूर्णांक एकत्रित करता है $ n $ जिसके लिए $ n + 1$ बिलकुल है
से विभाज्य $2^p $: प्रत्येक प्राकृत संख्या ठीक एक बार आती है।
डिकोडिंग एन्कोडिंग जितनी ही स्पष्ट है: के लिए $ n = 43$, कारक $ n + 1
= 44 = 2^2\cdot 11 = 2^2(2\cdot5 + 1)$, इसलिए $(p, q) = (2, 5)$.
समापन अंतर्दृष्टि: गणनीयता प्रमाण अक्सर होते हैं
\emph{एल्गोरिदम} भेस में --- यहाँ, ``दो को अलग करें''।
\end{example}

\begin{example}[बीजगणितीय संख्याएँ गणनीय हैं]\label{ex:b2:structures:algebraic}
एक सम्मिश्र संख्या है \emph{बीजगणितीय} जब यह कुछ को नष्ट कर देता है
परिमेय गुणांकों के साथ शून्येतर बहुपद। सेट
$\overline\Q $ बीजगणितीय संख्याओं का है \omterm{def:b2:structures:countable}{गणनीय}: के बहुपद
डिग्री $\leq d $ ऊपर $\Q $ में इंजेक्ट करें $\Q^{d+1}$, एक परिमित
का उत्पाद \omterm{def:b2:structures:countable}{गणनीय} समुच्चय
(\cref{prop:b2:structures:countablestable}~(2)); संघ ख़त्म
$ d $ शून्येतर परिमेय बहुपदों की गणना इस प्रकार करता है $ P_0, P_1,
P_2, \dots $; प्रत्येक $ P_k $ इसकी बहुत सारी जड़ें हैं; और
\[
\overline\Q = \bigcup_{k \in \N}\ \{\text{roots of } P_k\}
\]
एक है \omterm{def:b2:structures:countable}{गणनीय} परिमित समुच्चयों का मिलन
(\cref{prop:b2:structures:countablestable}~(3)), अनंत से
इसमें शामिल है $\Q $. की बेशुमारता के साथ संयुक्त $\R $
(\cref{thm:b2:structures:cantor} नीचे), यह साबित होता है --- बिना
एक ही चीज़ का प्रदर्शन --- कि पारलौकिक संख्याएँ मौजूद हैं
और एक बेशुमार बहुमत बनाएं: कैंटर का गिनती का तर्क
1874, केवल कार्डिनैलिटी द्वारा अस्तित्व।
\end{example}

\begin{theorem}[कैंटर; की बेशुमारता $\R $]\label{thm:b2:structures:cantor}
\begin{enumerate}
  \item हर सेट के लिए $ E $, कोई आपत्ति नहीं है $ E \to
        \mathcal{P}(E)$.
  \item $\R $ है \emph{नहीं} \omterm{def:b2:structures:countable}{गणनीय}.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) वर्ष 1 खंड (विकर्ण सेट) में सिद्ध किया गया था $ D = \{x : x
\notin f(x)\}$).

(2) मान लीजिये $(x_n)_{n \in \N}$ विश्लेषण करता है $\R $. नेस्टेड खंड बनाएं
$ I_0 \supseteq I_1 \supseteq \dots $ साथ $\abs{I_n} = 3^{-n}$ और
$ x_n \notin I_n $: वर्तमान खंड को तीन बंद तिहाई में विभाजित करें;
कम से कम एक तिहाई बचता है $ x_n $ (एक बिंदु अधिकतम दो से मिलता है
तीन). नेस्टेड-सेगमेंट प्रमेय (आसन्न समापन बिंदु) प्रदान करता है
$\ell \in \bigcap_n I_n $; लेकिन $\ell = x_N $ कुछ के लिए $ N $, और $ x_N
\notin I_N $: विरोधाभास।
\end{proof}

\begin{theorem}[कैंटर--बर्नस्टीन]\label{thm:b2:structures:cantorbernstein}
अगर $ E $ में इंजेक्ट करता है $ F $ और $ F $ में इंजेक्ट करता है $ E $, तब $ E $ और $ F $
हैं \omterm{def:b2:structures:countable}{सुसज्जित}.\index{कैंटर--बर्नस्टीन प्रमेय}
\end{theorem}

\begin{proof}
होने देना $ f \colon E \to F $ और $ g \colon F \to E $ इंजेक्शन हो. के लिए
प्रत्येक बिंदु (के $ E $ या $ F $), इसका पता लगाएं \emph{पूर्वज श्रृंखला} का
क्रमिक पूर्वछवियाँ, $ x \mapsto g^{-1}(x) \mapsto
f^{-1}(g^{-1}(x)) \mapsto \dots $ --- प्रत्येक चरण को तब तक परिभाषित किया जाता है
वर्तमान बिंदु प्रासंगिक इंजेक्शन की छवि में निहित है, और
फिर इंजेक्शन द्वारा अद्वितीय है। तीन परस्पर अनन्य भाग्य:
श्रृंखला एक बिंदु पर रुकती है $ E \setminus g(F)$ (\emph{में उत्पत्ति
$ E $}), के एक बिंदु पर रुकता है $ F \setminus f(E)$ (\emph{में उत्पत्ति
$ F $}), या कभी नहीं रुकता. यह विभाजन $ E = E_E \cup E_F \cup
E_\infty $ और $ F = F_E \cup F_F \cup F_\infty $ के अनुसार
उत्पत्ति.

अब निरीक्षण करें: $ f $ एमएपीएस $ E_E $ \emph{पर} $ F_E $ --- की शृंखला
$ f(x)$ की शृंखला है $ x $ एक चरण के साथ उपसर्ग, ताकि मूल मेल खाता हो;
और हर $ y \in F_E $ कम से कम एक कदम (इसकी उत्पत्ति) के साथ एक श्रृंखला है
में निहित है $ E $), इसलिए $ y = f(x)$ साथ $ x \in E_E $. वही तर्क
आक्षेप देता है $ f \colon E_\infty \to F_\infty $ और $ g \colon F_F
\to E_F $. चिपकाना,
\[
h(x) =
\begin{cases}
f(x) & \text{if } x \in E_E \cup E_\infty,\\
g^{-1}(x) & \text{if } x \in E_F,
\end{cases}
\]
से एक आक्षेप है $ E $ पर $ F = F_E \cup F_\infty \cup F_F $: यह है
विशेषण टुकड़ेवार, और तीन लक्ष्य टुकड़े असंयुक्त हैं।
\end{proof}

\begin{example}\label{ex:b2:structures:cbexample}
$\intoo{0}{1}$ और $\intcc{0}{1}$ हैं \omterm{def:b2:structures:countable}{सुसज्जित}: पहचान
एक तरह से इंजेक्ट करता है, $ x \mapsto \frac{x + 1}{3}$ दूसरा; प्रमेय
(आवश्यक रूप से असंतत) आक्षेप का निर्माण करता है। वैसे ही
$\R $, $\intoo{0}{1}$ (के जरिए $\tanh $-प्रकार आक्षेप) और
$\mathcal{P}(\N)$ (बाइनरी विस्तार, \cref{exo:b2:structures:3})
सभी हैं \omterm{def:b2:structures:countable}{सुसज्जित}: ``सातत्य की प्रमुखता''।
\end{example}

\begin{example}[खंड और वर्ग]\label{ex:b2:structures:squareseg}
$\intcc{0}{1}$ और $\intcc{0}{1}^2$ हैं \omterm{def:b2:structures:countable}{सुसज्जित} --- आयाम
कार्डिनैलिटी के लिए अदृश्य है। एक इंजेक्शन तुच्छ है: $ x
\mapsto (x, 0)$. दूसरे के लिए, भेजें $(x, y)$ असली को जिसका
दशमलव अंक उन अंकों को आपस में जोड़ते हैं $ x $ और $ y $,
\[
(0.x_1x_2x_3\dots,\ 0.y_1y_2y_3\dots)
\;\longmapsto\; 0.x_1y_1x_2y_2x_3y_3\dots,
\]
प्रत्येक समन्वय के लिए उस विस्तार को चुनना जो समाप्त नहीं होता है
सब $9$'s: उस परिपाटी के साथ छवि के अंक
का निर्धारण करें $ x $ और $ y $, इसलिए मानचित्र इंजेक्टिव है (इसकी आवश्यकता है
विशेषणात्मक न हों --- छवियाँ कभी भी, मान लीजिए, विषम स्थिति में नहीं होतीं
अंततः अंक $9$ --- और यह ठीक है)।
कैंटर--बर्नस्टीन (\cref{thm:b2:structures:cantorbernstein})
एक वास्तविक आक्षेप को एकत्रित करता है। निःसंदेह, निरंतरता है
निराशाजनक: उनके बीच निरंतर आपत्ति असंभव है ---
मीट्रिक अध्याय बताते हैं कि क्यों (कनेक्टिविटी अलग होती है
हवाई जहाज़ से लाइन, \cref{ch:b2:metric}).
\end{example}

\section{समूह}

\begin{definition}[उत्पन्न उपसमूह; आदेश]\label{def:b2:structures:generated}
होने देना $ G $ एक समूह बनें और $ A \subseteq G $. उपसमूह \emph{आप जेनरेट हुई}
द्वारा $ A $, लिखा हुआ $\langle A \rangle $, सबसे छोटा उपसमूह है
युक्त $ A $ --- ठोस रूप से, तत्वों के सभी परिमित उत्पाद $ A $
और उनके व्युत्क्रम. एक समूह है \emph{चक्रीय}\index{चक्रीय} कब
एक तत्व द्वारा उत्पन्न: $\langle a\rangle = \{a^k : k \in \Z\}$.
 \emph{आदेश}\index{आदेश} का $ a \in G $ है
$\operatorname{ord}(a) = \abs{\langle a \rangle}$ (संभवतः
अनंत); जब परिमित होता है, तो यह सबसे छोटा होता है $ n \geq 1$ साथ $ a^n = e $,
और $ a^k = e \iff \operatorname{ord}(a) \mid k $.
\end{definition}

\begin{proof}[आदेश लक्षण वर्णन का प्रमाण]
यदि कुछ $ a^m = e $ साथ $ m \geq 1$, होने देना $ n \geq 1$ कम से कम साथ रहो $ a^n
= e $. तत्व $ e, a, \dots, a^{n-1}$ जोड़ीवार भिन्न हैं
($ a^{i} = a^{j}$ साथ $0 \leq i < j < n $ देता है $ a^{j-i} = e $,
न्यूनतमता का खंडन), और प्रत्येक $ a^k $ उनमें से एक को कम कर देता है
यूक्लिडियन प्रभाग $ k = nq + r $: $\langle a\rangle $ बिलकुल है $ n $
तत्व, और $ a^k = a^r = e \iff r = 0 \iff n \mid k $. यदि कोई शक्ति नहीं है
सब तुच्छ है $ a^k $ ($ k \in \Z $) अलग हैं (समान प्रभाग
तर्क) और \omterm{def:b2:structures:generated}{आदेश} अनंत है.
\end{proof}

\begin{theorem}[लैग्रेंज]\label{thm:b2:structures:lagrange}
होने देना $ G $ एक परिमित समूह बनें और $ H $ एक उपसमूह. तब $\abs H $ विभाजित
$\abs G $.\index{लैग्रेंज का प्रमेय} विशेषकर \omterm{def:b2:structures:generated}{आदेश} प्रत्येक का
तत्व विभाजित होता है $\abs G $, और $ a^{\abs G} = e $ सभी के लिए $ a \in G $.
\end{theorem}

\begin{proof}
रिश्ता $ x \sim y \iff x^{-1}y \in H $ एक तुल्यता है
(प्रतिवर्ती: $ e \in H $; सममित: व्युत्क्रम; सकर्मक: उत्पाद)।
की कक्षा $ x $ है \emph{बायां कोसेट} $ xH = \{xh : h \in H\}$,
और $ h \mapsto xh $ एक आक्षेप है $ H \to xH $ (श्लोक में $ y \mapsto
x^{-1}y $): सभी वर्गों के पास है $\abs H $ तत्व. वर्ग विभाजन $ G $
(वर्ष 1 खंड का सामान्य विभाजन प्रमेय), इसलिए $\abs G =
\abs H \times (\text{number of cosets})$. किसी तत्व के लिए: इसे लागू करें
को $ H = \langle a\rangle $; तब $ a^{\abs G} = (a^{\operatorname{ord}
a})^{\abs G / \operatorname{ord} a} = e $.
\end{proof}

\begin{example}[कार्रवाई में कोसेट: $ A_3$ अंदर $\mathfrak{S}_3$]\label{ex:b2:structures:cosets}
लेना $ G = \mathfrak{S}_3$ (\omterm{def:b2:structures:generated}{आदेश} $6$) और $ H = A_3 =
\{\mathrm{id},\ (1\,2\,3),\ (1\,3\,2)\}$. बाएँ सहसमुच्चय हैं
\[
H = \{\mathrm{id},\ (1\,2\,3),\ (1\,3\,2)\},
\qquad
(1\,2)H = \{(1\,2),\ (2\,3),\ (1\,3)\} :
\]
तीन तत्वों के विभाजन के दो वर्ग $ G $, बिलकुल वैसे ही जैसे
गिनती $\abs G = \abs H \times (\text{number of cosets})$
मांग --- और स्पष्ट रूप से सम और विषम में विभाजन
क्रमपरिवर्तन. टिप्पणी $(1\,3)H = (1\,2)H $ हालांकि $(1\,3) \neq
(1\,2)$:कोसेट हैं \emph{कक्षाओं}, उनके द्वारा लेबल नहीं किया गया
प्रतिनिधि, और $ x^{-1}y \in H $ एकमात्र वैध है
तुलना. यह दो-वर्ग का चित्र सामान्य है
\omterm{thm:b2:structures:signature}{हस्ताक्षर}: $ A_n $ और इसका अकेला साथी कोसेट अलग हो गया
$\mathfrak{S}_n $ आधे में, जो सप्ताहांत की समस्या है
पहुंच योग्य पहेली स्थितियों की गणना करता है।
\end{example}

\begin{proposition}[चक्रीय समूह]\label{prop:b2:structures:cyclic}
होने देना $ G = \langle a \rangle $ होना \omterm{def:b2:structures:generated}{चक्रीय} का \omterm{def:b2:structures:generated}{आदेश} $ n $.
\begin{enumerate}
  \item $ G $ के लिए समरूपी है $(\Z/n\Z, +)$, के जरिए $\overline k \mapsto
        a^k $.
  \item का प्रत्येक उपसमूह $ G $ है \omterm{def:b2:structures:generated}{चक्रीय}; प्रत्येक विभाजक के लिए $ d \mid n $
        का बिल्कुल एक उपसमूह है \omterm{def:b2:structures:generated}{आदेश} $ d $, अर्थात् $\langle
        a^{n/d}\rangle $.
  \item $ a^k $ उत्पन्न करता है $ G $ अगर और केवल अगर $\gcd(k, n) = 1$: $ G $ है
        $\varphi(n)$ जेनरेटर (यूलर का कार्य\index{यूलर का
        टोटिएंट फ़ंक्शन}).
\end{enumerate}
\end{proposition}

\begin{proof}
(1) नक्शा $ k \mapsto a^k $ से $\Z $ पर $ G $ के साथ संगत है
सर्वांगसमता मॉड $ n $ ($ a^{k} = a^{k'} \iff n \mid k - k'$, से \omterm{def:b2:structures:generated}{आदेश}
लक्षण वर्णन); सार्वभौमिक संपत्ति
(\cref{def:b2:structures:quotient}) एक अच्छी तरह से परिभाषित विशेषण उत्पन्न करता है
रूपवाद से $\Z/n\Z $.

(2) चलो $ H \leq G $ गैरतुच्छ हो और $ m $ कम से कम $\geq 1$ साथ $ a^m \in
H $. यूक्लिडियन विभाजन दिखाता है $ H = \langle a^m\rangle $ (के लिए $ a^k \in
H $: $ k = mq + r $ बल $ a^r \in H $, इसलिए $ r = 0$), और $ m \mid n $
(विभाजित करना $ n $ द्वारा $ m $: $ a^{n \bmod m} \in H $). तब $\abs H = n/m $;
ले रहा हूँ $ m = n/d $ प्रत्येक विभाजक का एहसास करता है $ d $. विशिष्टता: कोई भी उपसमूह
का \omterm{def:b2:structures:generated}{आदेश} $ d $ उपरोक्त के अनुसार, स्वरूप का है $\langle a^m \rangle $
साथ $ n/m = d $ --- इसलिए $ m = n/d $ मजबूर है और उपसमूह है
निर्धारित.

(3) हम दावा करते हैं $\operatorname{ord}(a^k) = \frac{n}{\gcd(k, n)}$.
लिखना $ d = \gcd(k, n)$. किसी के लिए $ m \geq 1$, द \omterm{def:b2:structures:generated}{आदेश}
का लक्षण वर्णन \cref{def:b2:structures:generated} देता है
समतुल्यता की शृंखला
\[
(a^k)^m = e
\iff n \mid km
\iff \frac{n}{d} \,\Big|\, \frac{k}{d}\,m
\iff \frac{n}{d} \,\Big|\, m ,
\]
गॉस की लेम्मा द्वारा अंतिम चरण, तब से $\frac nd $ और $\frac kd $
सहअभाज्य हैं. ऐसा सबसे कम $ m $ है $\frac nd $:
$\operatorname{ord}(a^k) = \frac n{\gcd(k,n)}$, जो बराबर है $ n $
आईएफएफ $\gcd(k, n) = 1$. वहाँ हैं $\varphi(n)$ ऐसी कक्षाएं $ k $
सापेक्ष $ n $.
\end{proof}

\section{सममित समूह}

\begin{definition}\label{def:b2:structures:sn}
$\mathfrak{S}_n $ के क्रमपरिवर्तन का समूह है $\intint{1}{n}$
(\omterm{def:b2:structures:generated}{आदेश} $ n!$). ए \emph{चक्र}\index{चक्र} $(a_1\,a_2\,\cdots\,a_k)$
एमएपीएस $ a_1 \mapsto a_2 \mapsto \dots \mapsto a_k \mapsto a_1$ और
बाकी सब ठीक करता है; $ k $ इसका है \emph{लंबाई}, ए $2$-साइकिल एक है
\emph{स्थानांतरण}\index{स्थानांतरण}. दो चक्र हैं
\emph{विभिन्न करना} जब उनका
समर्थन (गैर-निश्चित बिंदु) हैं।
\end{definition}

\begin{theorem}[चक्र अपघटन]\label{thm:b2:structures:cycles}
प्रत्येक क्रमपरिवर्तन $\sigma \neq \mathrm{id}$ जोड़ीवार का एक उत्पाद है
विच्छेद \omterm{def:b2:structures:sn}{चक्र}, कारकों के क्रम तक विशिष्ट रूप से। विभिन्न करना
\omterm{def:b2:structures:sn}{चक्र} आवागमन, और $\operatorname{ord}(\sigma)$ का एलसीएम है
लंबाई.
\end{theorem}

\begin{proof}
के समर्थन पर ``कक्षा'' संबंध पर विचार करें $\sigma $: $ x \sim y $
आईएफएफ $ y = \sigma^k(x)$ कुछ के लिए $ k \in \Z $ --- एक तुल्यता
संबंध. प्रत्येक वर्ग $\{x, \sigma(x), \dots, \sigma^{k-1}(x)\}$
(परिमित, इसलिए पुनरावृत्त होता है \omterm{def:b2:structures:sn}{चक्र} वापस---पहली पुनरावृत्ति होनी चाहिए
पर लौटें $ x $ इंजेक्शन द्वारा) वहन करता है \omterm{def:b2:structures:sn}{चक्र} $(x\ \sigma(x)\
\cdots\ \sigma^{k-1}(x))$, और $\sigma $ इन्हीं का उत्पाद है
\omterm{def:b2:structures:sn}{चक्र}: प्रत्येक कक्षा पर, केवल संगत \omterm{def:b2:structures:sn}{चक्र} कार्य करता है.
विशिष्टता: कोई भी असंयुक्\omterm{def:b2:structures:sn}{त-चक्र} गुणनखंडन बिल्कुल पुनरुत्पादित करता है
कक्षाएँ (द \omterm{def:b2:structures:sn}{चक्र} के माध्यम से $ x $ होना ही चाहिए $(x\ \sigma(x)\ \cdots)$).
विभिन्न करना \omterm{def:b2:structures:sn}{चक्र} यात्रा करें क्योंकि वे असंयुक्त बिंदुओं पर चलते हैं; \omterm{def:b2:structures:generated}{आदेश}
कथन अनुसरण करता है क्योंकि $\sigma^m = \mathrm{id}$ यदि प्रत्येक \omterm{def:b2:structures:sn}{चक्र}'एस
$ m $-th शक्ति (असंयुक्तता) है, यदि प्रत्येक लंबाई विभाजित होती है $ m $.
\end{proof}

\begin{example}[जनगणना के रूप में साइकिल का प्रकार]\label{ex:b2:structures:cycletype}
के कितने क्रमपरिवर्तन $\mathfrak{S}_9$ लीजिए \omterm{def:b2:structures:sn}{चक्र} प्रकार
$(4, 3, 2)$ --- एक $4$-साइकिल, एक $3$-साइकिल, एक \omterm{def:b2:structures:sn}{स्थानांतरण}?
समर्थन और \omterm{def:b2:structures:generated}{चक्रीय} \omterm{def:b2:structures:generated}{आदेश} चुनें:
\[
\frac{9!}{4\cdot 3\cdot 2}
= \frac{362\,880}{24} = 15\,120 :
\]
नौ प्रतीकों को एक पंक्ति में सूचीबद्ध करें ($9!$ तरीके), पहले ब्रैकेट करें
चार, अगले तीन, अंतिम दो \omterm{def:b2:structures:sn}{चक्र}, और से विभाजित करें
प्रत्येक ब्रैकेट के अंदर घुमाव ($4$, $3$ और $2$ उनमें से) जो
वही क्रमपरिवर्तन दें. (विशिष्ट \omterm{def:b2:structures:sn}{चक्र} \emph{लंबाई} यहाँ,
इसलिए आगे कोई विभाजन नहीं; समान लंबाई की भी आवश्यकता होगी
समान कोष्ठकों के क्रमपरिवर्तन द्वारा विभाजित करना।) प्रत्येक ऐसा
क्रमपरिवर्तन है \omterm{def:b2:structures:generated}{आदेश} $\operatorname{lcm}(4,3,2) = 12$ और
\omterm{thm:b2:structures:signature}{हस्ताक्षर} $(-1)^3(-1)^2(-1)^1 = +1$
(\cref{thm:b2:structures:cycles} और \omterm{thm:b2:structures:signature}{हस्ताक्षर} प्रमेय
नीचे)। का एक विभाजन $9$, एक संयुग्मन वर्ग, एक जनगणना
--- का कॉम्बिनेटरिक्स $\mathfrak{S}_n $ का अंकगणित है
विभाजन.
\end{example}

\begin{theorem}[हस्ताक्षर]\label{thm:b2:structures:signature}
वास्तव में एक समूह रूपवाद है $\varepsilon \colon
\mathfrak{S}_n \to \{\pm 1\}$ (के लिए $ n \geq 2$) मान लेना $-1$
पर प्रतिस्थापन: द \emph{हस्ताक्षर}\index{हस्ताक्षर}. इसके अतिरिक्त
$\varepsilon(\sigma) = (-1)^{I(\sigma)}$ कहाँ $ I(\sigma)$ है
की संख्या \emph{इन्वर्ज़न} (जोड़े $ i < j $ साथ $\sigma(i) >
\sigma(j)$), ए $ k $-साइकिल पर \omterm{thm:b2:structures:signature}{हस्ताक्षर} हैं $(-1)^{k-1}$, और यह
\emph{वैकल्पिक समूह} $ A_n = \ker\varepsilon $ है \omterm{def:b2:structures:generated}{आदेश}
$\frac{n!}{2}$.
\end{theorem}

\begin{proof}
\emph{अस्तित्व।} के लिए $\sigma \in \mathfrak{S}_n $ तय करना
\[
\varepsilon(\sigma)
= \prod_{1 \leq i < j \leq n}
\frac{\sigma(j) - \sigma(i)}{j - i} .
\]
कारकों का निरपेक्ष मान गुणा हो जाता है $1$ (अनियंत्रित जोड़े
$\{\sigma(i), \sigma(j)\}$ सभी जोड़ियों पर दौड़ें), इसलिए
$\varepsilon(\sigma) = (-1)^{I(\sigma)} \in \{\pm1\}$. रूपवाद: के लिए
$\sigma, \tau $,
\[
\varepsilon(\sigma\tau)
= \prod_{i<j} \frac{\sigma(\tau(j)) - \sigma(\tau(i))}{j - i}
= \prod_{i<j} \frac{\sigma(\tau(j)) - \sigma(\tau(i))}{\tau(j) -
\tau(i)} \cdot \prod_{i<j} \frac{\tau(j) - \tau(i)}{j - i}
= \varepsilon(\sigma)\,\varepsilon(\tau),
\]
मध्य उत्पाद है $\varepsilon(\sigma)$ द्वारा पुन: अनुक्रमणित करने के बाद
जोड़े $\{\tau(i), \tau(j)\}$ (प्रत्येक अव्यवस्थित जोड़ा एक बार प्रकट होता है,
और अंश और हर एक साथ फ़्लिप चिह्न)। ए \omterm{def:b2:structures:sn}{स्थानांतरण}
$\tau = (a\,b)$ साथ $ a < b $ व्युत्क्रमणों की विषम संख्या है;
बिल्कुल सही गिना गया: उल्टे जोड़े $(i, j)$, $ i < j $, साथ
$\tau(i) > \tau(j)$ हैं
\[
(a, j) \ \text{for } a < j < b, \qquad
(i, b) \ \text{for } a < i < b, \qquad
(a, b) \ \text{itself},
\]
वह है $(b - a - 1) + (b - a - 1) + 1 = 2(b - a) - 1$ उनमें से,
अजीब. (वैकल्पिक रूप से: जांचें $(1\,2)$ सीधे, एक के साथ
व्युत्क्रम, और संयुग्म --- संयुग्मों के \omterm{thm:b2:structures:signature}{हस्ताक्षर} समान होते हैं
तब से $\varepsilon $ एबेलियन समूह के लिए एक रूपवाद है।) इसलिए
$\varepsilon((a\,b)) = (-1)^{2(b-a)-1} = -1$.

\emph{विशिष्टता.} प्रतिस्थापन उत्पन्न $\mathfrak{S}_n $ (कोई
\omterm{def:b2:structures:sn}{चक्र} $(a_1\cdots a_k) = (a_1\,a_k)(a_1\,a_{k-1})\cdots(a_1\,a_2)$,
और \cref{thm:b2:structures:cycles} ख़त्म); के लिए एक रूपवाद
$\{\pm1\}$ जनरेटर पर इसके मूल्यों द्वारा निर्धारित किया जाता है।

\emph{नतीजे।}  \omterm{def:b2:structures:sn}{चक्र} ऊपर पहचान लिखती है a $ k $-चक्र के रूप में
$ k - 1$ प्रतिस्थापन: \omterm{thm:b2:structures:signature}{हस्ताक्षर} $(-1)^{k-1}$. $ A_n $: रूपवाद
$\varepsilon $ विशेषण है (प्रतिस्थापन के लिए मौजूद है $ n \geq 2$),
और दो ``कोसेट'' $ A_n $ और $(1\,2)A_n $ हैं \omterm{def:b2:structures:countable}{सुसज्जित} और
विभाजन $\mathfrak{S}_n $ (लैग्रेंज का तर्क): $\abs{A_n} =
\frac{n!}{2}$.
\end{proof}

\begin{example}\label{ex:b2:structures:snexample}
$\sigma = \begin{pmatrix} 1&2&3&4&5&6\\ 3&6&5&4&1&2 \end{pmatrix}
= (1\,3\,5)(2\,6)$: \omterm{def:b2:structures:generated}{आदेश} $\operatorname{lcm}(3,2) = 6$, \omterm{thm:b2:structures:signature}{हस्ताक्षर}
$(-1)^{2}\cdot(-1)^{1} = -1$. \omterm{thm:b2:structures:signature}{हस्ताक्षर} सबसे तेज़ है
फेरबदल पर समता जांच --- और निर्धारक का इंजन
\cref{ch:b2:linalg}.
\end{example}

\begin{example}[एक चिन्ह तक तीन सड़कें]\label{ex:b2:structures:threeroads}
होने देना $\sigma \in \mathfrak{S}_5$ भेजना $1, 2, 3, 4, 5$ को $3, 5, 4,
1, 2$. \emph{के जरिए \omterm{def:b2:structures:sn}{चक्र}:} $1 \mapsto 3 \mapsto 4 \mapsto 1$ और $2
\mapsto 5 \mapsto 2$, इसलिए $\sigma = (1\,3\,4)(2\,5)$ और
$\varepsilon(\sigma) = (-1)^{2}(-1)^{1} = -1$. \emph{वाया
व्युत्क्रम:} मूल्य सूची में $3, 5, 4, 1, 2$ ऑर्डर से बाहर
जोड़े हैं $(3,1)$, $(3,2)$, $(5,4)$, $(5,1)$, $(5,2)$, $(4,1)$,
$(4,2)$: उनमें से सात, और $(-1)^7 = -1$. \emph{के जरिए
प्रतिस्थापन:} $\sigma = (1\,4)(1\,3)(2\,5)$, तीन कारक,
$(-1)^3 = -1$. तीन गणनाएँ, एक समता: में विशिष्टता
\cref{thm:b2:structures:signature} गारंटी देता है कि कोई बहीखाता नहीं है
योजना उन्हें कभी भी असहमत बना सकती है --- जो वास्तव में बनाता है
$\varepsilon $ एक अपरिवर्तनीय के रूप में प्रयोग करने योग्य (सप्ताहांत समस्या देखें)।
\end{example}

\begin{remark}[यहां से हस्ताक्षर कहां जाता है]
\omterm{thm:b2:structures:signature}{हस्ताक्षर} तीन बाद की फ़सलों का बीज है: यह निर्माण करता है
निर्धारक और उसके उत्पाद नियम में \cref{ch:b2:linalg}; यह शक्ति देता है
संयोजक पहेलियों के लिए समता अपरिवर्तनीय (इस अध्याय का सप्ताहांत)।
समस्या इसके साथ पंद्रह पहेली को हल करती है); और प्रत्यावर्तन
समूह $ A_n $ यह वर्ष 3 खंड में केंद्रीय बनने को परिभाषित करता है, जहां
उनकी सादगी के लिए $ n \geq 5$ बताते हैं क्यों डिग्री-$5$ समीकरण
कट्टरपंथियों में कोई समाधान नहीं है.
\end{remark}

\section{वलय, आदर्श, भागफल}

\begin{definition}[आदर्श]\label{def:b2:structures:ideal}
होने देना $ A $ एक क्रमविनिमेय वलय बनें। एक \emph{आदर्श}\index{आदर्श} $ I
\subseteq A $ ऐसा एक योगात्मक उपसमूह है $ a x \in I $ सभी के लिए
$ a \in A $, $ x \in I $. वलय आकारिकी के गुठली आदर्श हैं; $ I = A $
आईएफएफ $1 \in I $ आईएफएफ $ I $ एक इकाई शामिल है. आदर्श \emph{आप जेनरेट हुई} द्वारा
$ x $ है $ xA = \{xa\}$ (ए \emph{प्रधानाचार्य} आदर्श)।
\end{definition}

\begin{theorem}[{$\Z$ और $K[X]$ के आदर्श}]\label{thm:b2:structures:principal}
प्रत्येक \omterm{def:b2:structures:ideal}{आदर्श} का $\Z $ है $ n\Z $ एक अद्वितीय के लिए $ n \in \N $; प्रत्येक \omterm{def:b2:structures:ideal}{आदर्श} का
$ K[X]$ ($ K $ एक फ़ील्ड) है $ P\,K[X]$ एक अद्वितीय मोनिक (या शून्य) के लिए $ P $.
परिणामस्वरूप जीसीडी बेज़आउट संबंधों के साथ दोनों रिंगों में मौजूद है: $ x\Z +
y\Z = \gcd(x,y)\Z $, और इसी प्रकार बहुपदों के लिए भी।
\end{theorem}

\begin{proof}
के लिए $\Z $ यह वर्ष 1 खंड (ए) का उपसमूह प्रमेय था \omterm{def:b2:structures:ideal}{आदर्श}
विशेष रूप से एक उपसमूह है, और $ n\Z $ एक \omterm{def:b2:structures:ideal}{आदर्श}). के लिए $ K[X]$: होने देना
$ I \neq \{0\}$ एक हो \omterm{def:b2:structures:ideal}{आदर्श} और $ P \in I $ न्यूनतम डिग्री का शून्येतर,
सामान्यीकृत मोनिक। के लिए $ F \in I $, यूक्लिडियन डिवीजन $ F = PQ + R $
देता है $ R = F - PQ \in I $ साथ $\deg R < \deg P $: न्यूनतम बल $ R
= 0$, इसलिए $ I = P\,K[X]$. विशिष्टता: दो मोनिक जेनरेटर प्रत्येक को विभाजित करते हैं
अन्य. बेज़ाउट कथन की समानता है \omterm{def:b2:structures:ideal}{आदर्श} $ x\Z +
y\Z $ (सम्मान.\ इसका बहुपद एनालॉग) प्रिंसिपल के साथ \omterm{def:b2:structures:ideal}{आदर्श} का
जीसीडी --- वर्ष 1 में प्रयुक्त जीसीडी की परिभाषा, जिसे अब मान्यता दी गई है
के बारे में एक बयान \omterm{def:b2:structures:ideal}{आदर्श}ों.
\end{proof}

\begin{example}[एक बहुपद जीसीडी, दो तरीके]\label{ex:b2:structures:polygcd}
गणना करना $\gcd(X^3 - 1,\ X^2 - 1)$ में $\Q[X]$. \emph{यूक्लिड द्वारा:}
\[
X^3 - 1 = X\,(X^2 - 1) + (X - 1),
\qquad
X^2 - 1 = (X + 1)(X - 1) + 0 ,
\]
तो जीसीडी है $ X - 1$, और बैक-प्रतिस्थापन बेज़आउट देता है
संबंध
\[
X - 1 = 1\cdot(X^3 - 1) - X\cdot(X^2 - 1).
\]
\emph{द्वारा \omterm{def:b2:structures:ideal}{आदर्श}ों:}  \omterm{def:b2:structures:ideal}{आदर्श} $(X^3 - 1)\Q[X] + (X^2 - 1)\Q[X]$ है
प्रिंसिपल (\cref{thm:b2:structures:principal}); इसमें है $ X -
1$ (प्रदर्शन) और इसमें समाहित है $(X - 1)\Q[X]$ (दोनों
जनरेटर गायब हो जाते हैं $1$, इसलिए के गुणज हैं $ X - 1$): द
मोनिक जनरेटर है $ X - 1$. समापन अंतर्दृष्टि: \omterm{def:b2:structures:ideal}{आदर्श}
दृष्टिकोण gcd की पहचान करता है \emph{बिना विभाजित किये} --- सामान्य
जड़ें पता लगाती हैं \omterm{def:b2:structures:ideal}{आदर्श}, और यूक्लिड केवल इसे प्रमाणित करता है।
\end{example}

\begin{definition}[भागफल अंगूठी $\Z/n\Z $, पुनरीक्षित]\label{def:b2:structures:quotientring}
एक के लिए \omterm{def:b2:structures:ideal}{आदर्श} $ I $ का $ A $, संबंध $ x \sim y \iff x - y \in I $ है
के साथ संगत एक तुल्यता $+$ और $\times $; \omterm{def:b2:structures:quotient}{भागफल सेट}
$ A/I $ एक वलय संरचना विरासत में मिली है --- \emph{भागफल
अंगूठी}\index{भागफल
अंगूठी} ---बनाना $\pi \colon A \to A/I $ ए
कर्नेल के साथ रूपवाद $ I $. के लिए $ A = \Z $, $ I = n\Z $ ये है
$\Z/n\Z $ वर्ष का 1 खंड, अब अपनी सार्वभौमिक संपत्ति के साथ: कोई भी
रूपवाद हत्या $ I $ के माध्यम से कारक $ A/I $.
\end{definition}

\begin{theorem}[चीनी शेषफल प्रमेय, वलय रूप]\label{thm:b2:structures:crt}
अगर $\gcd(m, n) = 1$, वो नक्शा
\[
\Z/mn\Z \longrightarrow \Z/m\Z \times \Z/n\Z,
\qquad
\overline{x} \longmapsto (x \bmod m,\; x \bmod n)
\]
एक वलय समरूपता है।\index{चीनी शेषफल प्रमेय} फलस्वरूप
$\varphi(mn) = \varphi(m)\varphi(n)$ सहप्राइम के लिए $ m, n $, और
\[
\varphi(n) = n \prod_{p \mid n} \Bigl(1 - \frac 1p\Bigr)
\quad (p \text{ prime}).
\]
\end{theorem}

\begin{proof}
मानचित्र एक अच्छी तरह से परिभाषित रिंग आकारिकी है (संगतताएं हैं
तत्काल)। इंजेक्शन: $ x \equiv 0$ आधुनिक $ m $ और मॉड $ n $ साथ
$\gcd(m,n) = 1$ बल $ mn \mid x $ (गॉस)। विशेषणात्मकता: दोनों पक्ष
है $ mn $ तत्व, इसलिए इंजेक्शन पर्याप्त है (परिमित बराबर)।
कार्डिनैलिटीज़) --- या स्पष्ट रूप से: बी से\'ईज़आउट संबंध $ um +
vn = 1$, का वर्ग
\[
x = b\,um + a\,vn
\]
के लिए मानचित्र $(a \bmod m,\ b \bmod n)$, तब से $ vn = 1 - um \equiv 1
\pmod m $ बनाता है $ x \equiv a \pmod m $, और सममित रूप से मॉड $ n $
--- नुस्खा संख्यात्मक रूप से प्रयोग किया जाता है
\cref{ex:b2:structures:crtinverse}. इकाइयाँ इकाइयों के जोड़े (एक उत्पाद रिंग) के अनुरूप होती हैं
इकाइयाँ इकाइयों के जोड़े हैं), इसलिए $\varphi(mn) =
\varphi(m)\varphi(n)$. एक प्रमुख शक्ति के लिए, $\varphi(p^k) = p^k -
p^{k-1}$ (गैर-इकाइयाँ मॉड $ p^k $ के गुणज हैं $ p $);
गुणात्मकता उत्पाद सूत्र को एकत्रित करती है।
\end{proof}

\begin{example}[चीनी समरूपता को उलटना]\label{ex:b2:structures:crtinverse}
लेना $ m = 8$, $ n = 9$. समरूपता का व्युत्क्रम बनता है
दोनों द्वारा स्पष्ट \emph{बेरोजगार}: तलाश $ u \equiv 1 \pmod 8$,
$ u \equiv 0 \pmod 9$ और $ v \equiv 0 \pmod 8$, $ v \equiv 1 \pmod
9$. से $ u = 9k \equiv 1 \pmod 8$: $ k \equiv 1$, इसलिए $ u = 9$; से
$ v = 8k \equiv 1 \pmod 9$: $-k \equiv 1$, $ k \equiv 8$, इसलिए $ v =
64$. फिर की कक्षा $ x = 9a + 64b $ सापेक्ष $72$ अद्वितीय है
का समाधान $ x \equiv a \pmod 8$, $ x \equiv b \pmod 9$: के लिए $ a =
3$, $ b = 5$ एक मिलता है $27 + 320 = 347 \equiv 59 \pmod{72}$ ---
में प्रतिस्थापन द्वारा पाया गया बिल्कुल मध्यवर्ती मान
\cref{exo:b2:structures:8}. समापन अंतर्दृष्टि: $ u $ और $ v $
संतुष्ट $ u + v \equiv 1$, $ uv \equiv 0$, $ u^2 \equiv u $, $ v^2
\equiv v $ सापेक्ष $72$; वे की छवियाँ हैं $(1, 0)$ और $(0,
1)$, और प्रत्येक चीनी अपघटन नीचे एक अपघटन है
का $1$ ऑर्थोगोनल इडेम्पोटेंट्स में।
\end{example}

\begin{theorem}[यूलर; फ़र्मेट ने दोबारा गौर किया]\label{thm:b2:structures:euler}
की इकाइयाँ $\Z/n\Z $ का एक समूह बनाएं \omterm{def:b2:structures:generated}{आदेश} $\varphi(n)$; इसलिए के लिए
$\gcd(a, n) = 1$:
\[
a^{\varphi(n)} \equiv 1 \pmod n
\qquad (\text{Euler's theorem\index{यूलर का प्रमेय}}),
\]
और फ़र्मेट का छोटा प्रमेय मामला है $ n = p $ प्राइम, अब एक पंक्ति
लैग्रेंज से.
\end{theorem}

\begin{proof}
व्युत्क्रमणीय वर्ग बिल्कुल उन पूर्णांकों के होते हैं जिनसे सहअभाज्य होता है $ n $
(वर्ष 1 खंड): $\varphi(n)$ उनमें से, एक समूह का गठन
गुणन. लैग्रेंज (\cref{thm:b2:structures:lagrange}):
समूह की शक्ति के लिए प्रत्येक तत्व \omterm{def:b2:structures:generated}{आदेश} पहचान है.
\end{proof}

\begin{example}[जनरेटर के बिना एक इकाई समूह]\label{ex:b2:structures:units15}
समूह $(\Z/15\Z)^*$ है $\varphi(15) = \varphi(3)\varphi(5)
= 8$ तत्व. यह है \omterm{def:b2:structures:generated}{चक्रीय}? गणना करना \omterm{def:b2:structures:generated}{आदेश} चीनी का उपयोग करना
समरूपता $(\Z/15\Z)^* \simeq (\Z/3\Z)^* \times (\Z/5\Z)^*$
(एक यूनिट मॉड $15$ इकाइयों की एक जोड़ी है): कारकों में है \omterm{def:b2:structures:generated}{आदेश}
$2$ और $4$, तो हर तत्व का \omterm{def:b2:structures:generated}{आदेश} विभाजित
$\operatorname{lcm}(2, 4) = 4 < 8$ --- कोई तत्व उत्पन्न नहीं होता.
ठोस रूप से:
\[
2^4 = 16 \equiv 1, \qquad
4^2 = 16 \equiv 1, \qquad
7^4 \equiv 1, \qquad
11^2 = 121 \equiv 1, \qquad
14^2 \equiv 1 \pmod{15} :
\]
\omterm{def:b2:structures:generated}{आदेश} $4, 2, 4, 2, 2$ और कभी नहीं $8$. के साथ तुलना करें
\cref{exo:b2:structures:10}: $(\Z/p\Z)^*$ \emph{है} \omterm{def:b2:structures:generated}{चक्रीय} के लिए
$ p $ अभाज्य, क्योंकि वहां इकाई समूह एक क्षेत्र के अंदर बैठता है।
यूलर का प्रमेय अभी भी घातांक के साथ लागू होता है $\varphi(15) = 8$,
लेकिन सच्चा सार्वभौमिक प्रतिपादक यहीं है $4$ --- यूलर एक है
ऊपरी सीमा, हमेशा तेज़ नहीं।
\end{example}

\begin{definition}[बीजगणित]\label{def:b2:structures:algebra}
ए \emph{$ K $-बीजगणित}\index{$ K $-बीजगणित} एक है $ K $-सदिश स्थल $ A $ एक के साथ
वलय संरचना जिसका गुणन है $ K $-बिलिनियर. उदाहरण:
$ K[X]$, $\mathcal{M}_n(K)$, $\mathcal{L}(E)$, फ़ंक्शन स्थान
$\mathcal{F}(X, K)$, $\C $ एक के रूप में $\R $-बीजगणित. बीजगणित की आकृतियाँ
रैखिक वलय आकारिकी हैं; \emph{मूल्यांकन} $ P \mapsto P(u)$
से $ K[X]$ को $\mathcal{L}(E)$ (या $\mathcal{M}_n(K)$) है
केंद्रीय उदाहरण, ड्राइविंग \cref{ch:b2:reduction}.
\end{definition}

\begin{example}[एक मूल्यांकन रूपवाद और उसका मूल]\label{ex:b2:structures:evaluation}
लेना $ A = \begin{pmatrix}0 & 1\\ 0 & 0\end{pmatrix}$ और
मूल्यांकन $\varepsilon_A \colon \R[X] \to \mathcal{M}_2(\R)$,
$ P \mapsto P(A)$. तब से $ A^2 = 0$,
\[
P(A) = P(0)\,I + P'(0)\,A =
\begin{pmatrix} P(0) & P'(0)\\ 0 & P(0)\end{pmatrix},
\]
(केवल अचर और रैखिक पद $ P $ जीवित बचना)। इस तरह
$\ker\varepsilon_A = \{P : P(0) = P'(0) = 0\} = X^2\,\R[X]$: ए
प्रधान \omterm{def:b2:structures:ideal}{आदर्श}, बिल्कुल वैसा ही \cref{thm:b2:structures:principal}
भविष्यवाणी करता है, आप जेनरेट हुई मोनिक द्वारा $ X^2$ में न्यूनतम डिग्री की
कर्नेल --- \emph{न्यूनतम बहुपद} का $ A $, का सितारा
\cref{ch:b2:reduction}. छवि द्वि-आयामी है
क्रमविनिमेय बीजगणित $\{aI + bA\}$: मूल्यांकन आकारिकी सिकुड़ती है
अनंत-आयामी $\R[X]$ छोटे, गणना योग्य पर अल्जेब्रास.
\end{example}

\begin{remark}[परिप्रेक्ष्य: सुनने के लिए तीन धुनें]\label{rem:b2:structures:perspectives}
इस अध्याय से तीन संरचनात्मक विचार पूरे अध्याय में दोहराए जाते हैं
वॉल्यूम, हर बार भारी ऑर्केस्ट्रेशन में। \emph{गुणनखंडीकरण
एक भागफल के माध्यम से} (\cref{def:b2:structures:quotient}): यह
बनाता है $\Z/n\Z $ यहां, रैखिक के समाधान सेट पर मानचित्रों को परिभाषित करता है
सिस्टम में \cref{ch:b2:linalg}, और चुपचाप हर एक को रेखांकित करता है
``वर्गों पर अच्छी तरह से परिभाषित'' तर्क। \emph{अपरिवर्तनशीलताओं}: द
\omterm{thm:b2:structures:signature}{हस्ताक्षर} एक रूपवाद है $\{\pm1\}$ कोई कानूनी कदम नहीं उठा सकता
चकमा --- वही तर्क निर्धारक का उत्पाद नियम देता है
(\cref{ch:b2:linalg}), ट्रेस की समानता अपरिवर्तनीयता, और
की संरक्षित मात्रा \cref{ch:b2:diffeq}.
\emph{एक संरचना के विरुद्ध गिनती}: लैग्रेंज की गिनती होती है
कोसेट, आयाम आधारों के माध्यम से गिना जाता है
(\cref{ch:b2:linalg}), बहुलता बहुपद के माध्यम से गिना जाता है
डिग्री (\cref{ch:b2:reduction}); जब भी कोई बंधा दिखता है
चमत्कारी, कुछ विभाजन या ग्रेडिंग गिनती कर रही है।
\end{remark}

\begin{remark}[सामान्य ख़तरे]\label{rem:b2:structures:pitfalls}
चार क्लासिक्स. (i) भागफल पर मानचित्र की जाँच अवश्य की जानी चाहिए
\emph{अच्छी तरह से परिभाषित}: ``$\overline x \mapsto $ (सूत्र पर $ x $)''
केवल तभी वैध है जब सूत्र वर्गों पर स्थिर हो ---
की अनुकूलता \cref{def:b2:structures:quotient}, नहीं ए
औपचारिकता. (ii) $\operatorname{ord}(ab) =
\operatorname{lcm}(\operatorname{ord}a, \operatorname{ord}b)$
है \emph{असत्य} सामान्य तौर पर, आने-जाने वाले तत्वों के लिए भी ($ a $
और $ a^{-1}$); \cref{exo:b2:structures:4} सही देता है
सहप्राइम-एंड-कम्यूटिंग कथन, और असंयुक्त \omterm{def:b2:structures:sn}{चक्र} द
सही क्रमपरिवर्तन संस्करण. (iii) गणनीयता जीवित
\omterm{def:b2:structures:countable}{गणनीय} \emph{यूनियन} और परिमित \emph{उत्पादों}, लेकिन नहीं
\omterm{def:b2:structures:countable}{गणनीय} उत्पाद: $\{0,1\}^{\N}$ बेशुमार है
(\cref{exo:b2:structures:3}) हालाँकि प्रत्येक कारक में दो होते हैं
तत्व. (iv) कैंटर--बर्नस्टीन को केवल दोनों इंजेक्शनों की आवश्यकता है
तरीके, लेकिन इससे जो आपत्ति पैदा होती है वह आम तौर पर असंतत होती है और
गैर-स्पष्ट --- किसी सूत्र की अपेक्षा न करें
(\cref{ex:b2:structures:cbexample}).
\end{remark}

\begin{remark}[जहाँ इस अध्याय का प्रयोग किया जाता है]
लगभग हर जगह. \omterm{thm:b2:structures:signature}{हस्ताक्षर} निर्धारकों का निर्माण करता है
(\cref{ch:b2:linalg}); मूल्यांकन रूपवाद $ P \mapsto P(u)$
और प्रिंसिपल \omterm{def:b2:structures:ideal}{आदर्श}ों का $ K[X]$ न्यूनतम बहुपद उत्पन्न करें और
की गिरी का विघटन \cref{ch:b2:reduction}; गणनीयता
वह मंच है जिस पर \cref{ch:b2:proba} प्रदर्शन करता है (संभावना पर
\omterm{def:b2:structures:countable}{गणनीय} रिक्त स्थान) और कारण टोपोलॉजी उत्पन्न होती रहती है \omterm{def:b2:structures:countable}{गणनीय}
घने सेट (\cref{ch:b2:metric}). भागफल निर्माण $ A/I $
फ़ील्ड बनाने के लिए वर्ष 3 खंड में पुनः तैनात किया गया है $ K[X]/(P)$ और,
उनसे, गैलोज़ सिद्धांत: सार्वभौम संपत्ति यहाँ साबित हुई है
वहाँ शब्द दर शब्द प्रयोग किया गया।
\end{remark}

\begin{exercise}[$\star $]\label{exo:b2:structures:1}
निम्नलिखित में से कौन से सेट हैं \omterm{def:b2:structures:countable}{गणनीय}? परिमित उपसमुच्चय का समुच्चय
का $\N $; के समुच्चय \emph{सभी} के उपसमुच्चय $\N $; $\R \setminus \Q $;
परिमेय गुणांक वाले बहुपदों का समुच्चय; का सेट
के अनुक्रम $0$'रेत $1$ जो अंततः शून्य हैं।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:structures:2}
में $\mathfrak{S}_7$, होने देना $\sigma = (1\,4\,2\,6)(3\,5)$ और $\tau =
(2\,3\,7)$. गणना करना $\sigma\tau $ और $\tau\sigma $ में असंयुक्\omterm{def:b2:structures:sn}{त-चक्र}
इससे \omterm{def:b2:structures:generated}{आदेश} और सभी चार क्रमपरिवर्तन के \omterm{thm:b2:structures:signature}{हस्ताक्षर}, और
$\sigma^{2026}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:structures:3}
इसे दर्शाते हुए स्पष्ट इंजेक्शनों का निर्माण करें $\mathcal{P}(\N)$,
$\intcc{0}{1}$ और सेट $\{0,1\}^{\N}$ बाइनरी अनुक्रम हैं
जोड़ीदार \omterm{def:b2:structures:countable}{सुसज्जित} \emph{(द्विआधारी विस्तार दोनों तरह से;
कैंटर--बर्नस्टीन दोहरे-प्रतिनिधित्व उपद्रव को अवशोषित करता है)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:structures:4}
होने देना $ G $ एक समूह बनें और $ a, b \in G $ परिमित के आने वाले तत्व
सहप्राइम \omterm{def:b2:structures:generated}{आदेश} $ m $ और $ n $. साबित करें कि $\operatorname{ord}(ab) =
mn $. में एक उदाहरण देकर दिखाएँ $\mathfrak{S}_3$ वह रूपान्तरण है
आवश्यक.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:structures:5}
होने देना $ G $ सम का एक परिमित समूह बनें \omterm{def:b2:structures:generated}{आदेश}. साबित करें कि $ G $ एक शामिल है
का तत्व \omterm{def:b2:structures:generated}{आदेश} $2$. \emph{(प्रत्येक तत्व को उसके व्युत्क्रम के साथ जोड़ें;
स्व-युग्मित लोगों की गिनती करें।)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:structures:6}
साबित करें कि $ A_n $ ($ n \geq 3$) है आप जेनरेट हुई से $3$-चक्र।
\emph{(दो का एक उत्पाद प्रतिस्थापन एक है $3$-साइकिल या एक उत्पाद
दो का $3$-चक्र।)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:structures:7}
सभी समूह आकारिकी निर्धारित करें: से $(\Q, +)$ को $(\Z, +)$; से

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:structures:8}
चीनी शेषफल प्रमेय का उपयोग करके गणना करें $\varphi(360)$, खोजें
सब $ x $ साथ $ x \equiv 3 \pmod 8$, $ x \equiv 5 \pmod 9$ और $ x
\equiv 2 \pmod 5$, और अंतिम दो अंकों की गणना करें $3^{2026}$
\emph{(यूलर मॉड $100$; सावधान: कार्य मॉड $4$ और मॉड $25$)}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:structures:9}
साबित करें कि एक परिमित अभिन्न डोमेन एक क्षेत्र है। उसका निष्कर्ष निकालो
$\Z/n\Z $ एक फ़ील्ड है iff $ n $ प्रधान (फिर से) है.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:structures:10}
(एक क्लासिक) चलो $ K $ एक क्षेत्र हो और $ G $ ए \emph{परिमित} का उपसमूह
$(K^*, \times)$. साबित करें कि $ G $ है \omterm{def:b2:structures:generated}{चक्रीय}.
\emph{संकेत: चलो $ m $ अधिकतम हो \omterm{def:b2:structures:generated}{आदेश} के तत्वों के बीच $ G $;
प्रत्येक तत्व दिखाएँ \omterm{def:b2:structures:generated}{आदेश} विभाजित $ m $ (का उपयोग करते हुए

संतुष्ट $ x^m = 1$; की जड़ें गिनें $ X^m - 1$.}
विशेष रूप से $(\Z/p\Z)^*$ है \omterm{def:b2:structures:generated}{चक्रीय}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:structures:11}
सिद्ध करें कि समूह $(\Q, +)$ क्या नहीं है \omterm{def:b2:structures:generated}{चक्रीय}, और इससे भी बदतर: यह नहीं है

\end{exercise}

\section{समस्या: पंद्रह पहेली}

पन्द्रह पहेली एक है $4 \times 4$ पन्द्रह स्लाइडिंग रखने वाली ट्रे
टाइल्स क्रमांकित $1$ को $15$ और एक खाली सेल; एक चाल इनमें से एक को स्लाइड करती है
इसमें खाली सेल से सटी हुई टाइलें। 1890 के दशक में सैम लोयड
की पेशकश करके पहेली को लोकप्रिय बनाया \$ जो कोई भी कर सकता है उसे 1000 रु
टाइल्स बदलें $14$ और $15$ और प्रत्येक अन्य टाइल को उसके स्थान पर लौटा दें
जगह. किसी ने कभी भी संग्रह नहीं किया, और इस सप्ताहांत की समस्या दोनों को साबित करती है
कारण के आधे: के \omterm{thm:b2:structures:signature}{हस्ताक्षर}
\cref{thm:b2:structures:signature} लोयड के आदान-प्रदान को प्रतिबंधित करता है, और

\begin{problem}[{सप्ताहांत समस्या---जॉनसन--कहानी
सॉल्वेबिलिटी प्रमेय}]
\label{pb:b2:structures:1}
कोशिकाओं को क्रमांकित करें $1$ को $16$ पढ़ने के क्रम में (बाएँ से दाएँ, ऊपर)।
नीचे तक), तो वह सेल $ k $ पंक्ति में बैठता है $ i $ और स्तंभ $ j $ साथ $ k
= 4(i - 1) + j $. कक्ष $16$ (नीचे दाएं) है \emph{घर} का
खाली सेल; हम खाली सेल को सोलहवीं टाइल के रूप में मानते हैं,
लिखा हुआ $ b $ और नंबर से पहचान की गई $16$. ए
\emph{विन्यास} एक आक्षेप है $\sigma \colon \intint1{16}
\to \intint1{16}$, कक्ष $\mapsto $ सामग्री; \emph{हल हो गया}
कॉन्फ़िगरेशन है $\sigma = \mathrm{id}$. लगातार,
$\varepsilon $ का \omterm{thm:b2:structures:signature}{हस्ताक्षर} है
\cref{thm:b2:structures:signature} और दो कोशिकाएँ हैं
\emph{नज़दीक} जब वे ट्रे का एक किनारा साझा करते हैं।

\medskip
\noindent\textbf{Part I --- Configurations, moves, signatures.}
\begin{enumerate}
  \item पुष्टि करें कि कॉन्फ़िगरेशन बिल्कुल तत्व हैं
        का $\mathfrak{S}_{16}$, तो वहाँ हैं $16! =
        20\,922\,789\,888\,000$ उनमें से, और वह संख्या
        किसी दिए गए कॉन्फ़िगरेशन से कानूनी चालें हैं $2$, $3$ या $4$,
        इसके अनुसार खाली सेल एक कोने में पड़ा है या नहीं
        एक किनारा, या आंतरिक भाग में।
  \item होने देना $\sigma $ एक विन्यास हो, $ p = \sigma^{-1}(16)$ द
        रिक्त स्थान का कक्ष, और $ c $ बगल में एक सेल $ p $. दिखाओ
        वह टाइल खिसका रहा है $ c $ में $ p $ का उत्पादन करता है
        विन्यास $\sigma' = \sigma \circ \tau $ साथ $\tau =
        (p\ c)$, और यह निष्कर्ष निकालें कि हर कदम पर \omterm{thm:b2:structures:signature}{हस्ताक्षर} पलट जाता है:
        $\varepsilon(\sigma') = -\varepsilon(\sigma)$.
  \item ट्रे को चेकरबोर्ड करें: $\chi(k) = (-1)^{i+j}$ सेल के लिए
        $ k $ पंक्ति में $ i $, स्तंभ $ j $. दिखाएँ कि हर चाल उलट जाती है
        $\chi(\text{cell of the blank})$, और यह निष्कर्ष निकालें कि a
        रिक्त स्थान को उसके आरंभिक सेल में लौटाने की चालों का क्रम
        सम लंबाई है.
  \item बताते हैं कि
        \[
        I(\sigma) = \varepsilon(\sigma)\,
        \chi\bigl(\sigma^{-1}(16)\bigr)
        \]
        प्रत्येक कानूनी कदम के तहत अपरिवर्तनीय है, और गणना करें
        $ I(\mathrm{id})$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Loyd's bounty: the invariant at
work.}
\begin{enumerate}[resume]
  \item लॉयड का विन्यास $\sigma_L $ समाधान से सहमत हैं
        सिवाय उस कोशिकाओं के $14$ और $15$ टाइल्स पकड़ो $15$ और $14$.
        गणना करना $ I(\sigma_L)$ और निष्कर्ष निकाला कि इसका कोई क्रम नहीं है
        लिंक ले जाता है $\sigma_L $ हल किए गए कॉन्फ़िगरेशन के लिए:
        लॉयड का \$1000 कभी ख़तरे में नहीं था.
  \item दिखाएँ कि सभी कॉन्फ़िगरेशन का बिल्कुल आधा हिस्सा संतुष्ट करता है $ I =
        +1$: $\abs{\{\sigma : I(\sigma) = +1\}} = 16!/2$.
        \emph{(एक निश्चित रिक्त सेल के लिए, जोड़ी कॉन्फ़िगरेशन
        एक निश्चित के साथ रचना \omterm{def:b2:structures:sn}{स्थानांतरण} दो अन्य का
        कोशिकाएँ।)}
  \item दिखाएँ कि हर कदम कानूनी कदम से पूर्ववत हो जाता है
        ``$\sigma'$ से पहुंचा जा सकता है $\sigma $ कानूनी चालों से'' है
        एक तुल्यता संबंध, और वह वर्ग $ R $ का
        हल किया गया कॉन्फ़िगरेशन संतुष्ट करता है $ R \subseteq \{I = +1\}$.
        निष्कर्ष निकालें कि कम से कम दो वर्ग हैं।
  \item मान लीजिए रिक्त स्थान घर है: $\sigma(16) = 16$. बताते हैं कि
        $ I(\sigma) = \varepsilon(\rho)$ कहाँ $\rho \in
        \mathfrak{S}_{15}$ का प्रतिबंध है $\sigma $ को
        कोशिकाएं $1, \dots, 15$, और वह कोई भी कॉन्फ़िगरेशन हो सकता है
        कानूनी चालों द्वारा खाली घर में ले जाया गया।
        निष्कर्ष निकालना: सिद्ध करना $ R = \{I = +1\}$ यह एहसास करने के लिए पर्याप्त है
        प्रत्येक \emph{यहां तक की} पन्द्रह गैर-घर का क्रमपरिवर्तन
        कोशिकाओं को चालों के क्रम से प्रारंभ और समाप्त करना
        खाली घर.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Blank tours and the program
group.} ए \emph{कार्यक्रम} कानूनी चालों का एक सीमित क्रम है,
खाली घर के साथ एक विन्यास से शुरू हुआ, जिसका अंतिम
कॉन्फ़िगरेशन में फिर से रिक्त घर है। इसका \emph{प्रभाव} है
क्रमपरिवर्तन $\pi $ द्वारा परिभाषित कोशिकाओं की संख्या: कोशिका की सामग्री $ x $
सेल में समाप्त होता है $\pi(x)$.
\begin{enumerate}[resume]
  \item दिखाएँ कि एक प्रोग्राम कहाँ से चलता है $\sigma $ पर समाप्त होता है $\sigma
        \circ \pi^{-1}$; जो लगातार दो प्रोग्राम चला रहा है
        उनके प्रभाव रचता है; और वह सेट $ H $ सबमें से
        प्रभाव का एक उपसमूह है $\mathfrak{S}_{15}$ (क्रमपरिवर्तन
        कोशिकाओं का $1, \dots, 15$) प्रत्यावर्तन में निहित है
        समूह $ A_{15}$.
  \item (प्रारंभिक दौरा) घर में रिक्त स्थान से, स्लाइड करें
        नीचे दाईं ओर खाली $2 \times 2$ ब्लॉक: कोशिकाएं
        $16 \to 12 \to 11 \to 15 \to 16$. दिखाओ प्रभाव है
        $3$-चक्र $(11\ 12\ 15)$, और वह रिवर्स टूर देता है
        $(11\ 15\ 12)$. दोनों पड़े हैं $ H $.
  \item (भव्य दौरा) उसे सत्यापित करें
        \[
        16 \to 15 \to 14 \to 13 \to 9 \to 5 \to 1 \to 2 \to 3
        \to 4 \to 8 \to 7 \to 6 \to 10 \to 11 \to 12 \to 16
        \]
        सभी सोलह कक्षों (आसन्न चरणों) के माध्यम से एक बंद रास्ता है
        केवल), और वह इसका प्रभाव है $15$-चक्र
        \[
        \zeta = (15\ 12\ 11\ 10\ 6\ 7\ 8\ 4\ 3\ 2\ 1\ 5\ 9\ 13\
        14) .
        \]
        लिखना $ x_0 = 15$, $ x_1 = 12$, $ x_2 = 11$, \dots, $ x_{14}
        = 14$ इसके लिए \omterm{def:b2:structures:sn}{चक्र} \omterm{def:b2:structures:generated}{आदेश}, जांचें कि इसका उलटा है
        प्रश्न 10 का प्रारंभिक दौरा बिल्कुल सही है $(x_0\ x_1\
        x_2)$.
  \item किसी में संयुग्मन सूत्र सिद्ध करें $\mathfrak{S}_n $: के लिए
        एक क्रमपरिवर्तन $ g $ और ए $3$-चक्र,
        \[
        g\,(a\ b\ c)\,g^{-1} = \bigl(g(a)\ g(b)\ g(c)\bigr),
        \]
        और उस पर ध्यान दें $ H $, एक समूह होने के नाते, के अंतर्गत बंद है
        अपने स्वयं के तत्वों द्वारा संयुग्मन।
  \item उसका निष्कर्ष निकालो $ H $ सभी पन्द्रह सम्मिलित हैं \emph{लगातार}
        $3$-भव्य यात्रा के \omterm{def:b2:structures:sn}{चक्र}:
        \[
        s_t = (x_t\ x_{t+1}\ x_{t+2}) \qquad (t \in \Z/15\Z,
        \text{ indices mod } 15).
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Generating the alternating group.}
\begin{enumerate}[resume]
  \item (लेम्मा ए) चलो $ s $ और $ t $ होना $3$-साइकिल जिनका समर्थन करता है
        ठीक दो बिंदु साझा करें, समर्थन कहें $\{a, b, c\}$ और
        $\{b, c, d\}$. बदलने के बाद दिखाओ $ s $ या $ t $ द्वारा
        यदि आवश्यक हो तो इसका उलटा (जो कुछ भी नहीं बदलता है)।
        आप जेनरेट हुई उपसमूह), उत्पाद $ st $ एक दोहरा है
        \omterm{def:b2:structures:sn}{स्थानांतरण}; बताते हैं कि $ A_4$ का कोई उपसमूह शामिल नहीं है
        \omterm{def:b2:structures:generated}{आदेश} $6$ \emph{(सूचकांक का एक उपसमूह $2$ प्रत्येक शामिल है
        वर्गाकार; गिनें $3$-वर्गों के बीच साइकिल)}; और निष्कर्ष निकालो
        वह $\langle s, t\rangle $ संपूर्ण वैकल्पिक समूह है
        चार अक्षरों में से $\{a, b, c, d\}$.
  \item (लेम्मा बी) चलो $ X $ का एक सेट हो $ k \geq 4$ पत्र, $ w
        \notin X $, और चलो $ G $ कुछ का उपसमूह बनें
        $\mathfrak{S}_n $ के प्रत्येक सम क्रमपरिवर्तन से युक्त $ X $
        और एक $3$-चक्र $(u\ v\ w)$ साथ $ u, v \in X $. वो दिखाओ
        सभी के लिए विशिष्ट $ a, b \in X $ वहाँ है एक \emph{यहां तक की}
        परिवर्तन $ g $ का $ X $ साथ $ g(u) = a $, $ g(v) = b $, और
        निष्कर्ष निकालना $(a\ b\ w) \in G $.
  \item यह निष्कर्ष निकालें कि समूह $ G $ लेम्मा बी में प्रत्येक सम शामिल है
        का क्रमपरिवर्तन $ X \cup \{w\}$ \emph{(उपयोग
        \cref{exo:b2:structures:6}: द $3$-चक्र उत्पन्न)}.
        फिर, लेम्मास ए और बी को लगातार श्रृंखलाबद्ध करते हुए
        $3$-चक्र $ s_0, s_1, \dots, s_{12}$ प्रश्न 13 का,
        इसे साबित करो $\langle s_0, \dots, s_{12}\rangle = A_{15}$.
  \item समाप्त करें कि $ H = A_{15}$: \emph{प्रत्येक सम पुनर्व्यवस्था
        पंद्रह टाइल्स में से एक प्रोग्राम द्वारा प्राप्त किया जा सकता है}, और $ H $
        है $15!/2 = 653\,837\,184\,000$ तत्व.
  \item (द जॉनसन--स्टोरी थ्योरम, 1879) प्रश्न 6 इकट्ठा करें,
        7, 8 और 17: हल किए गए कॉन्फ़िगरेशन से पहुंच योग्य
        एक हैं \emph{बिल्कुल}  $16!/2 =
        10\,461\,394\,944\,000$ के साथ कॉन्फ़िगरेशन $ I = +1$;
        और पहुंच योग्यता बिल्कुल है \emph{दो} कक्षाएं, कक्षा
        of the solved configuration and the class of Loyd's
        $\sigma_L $. \emph{(दूसरे बिंदु के लिए, टाइल्स को पुनः लेबल करें
        $14$ और $15$: दिखाओ $\sigma \mapsto (14\ 15) \circ
        \sigma $ मानचित्र अनुक्रमों को स्थानांतरित करने के लिए अनुक्रमों को स्थानांतरित करते हैं और
        आदान-प्रदान $\{I = +1\}$ साथ $\{I = -1\}$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Criteria, variants, and the view
from above.}
\begin{enumerate}[resume]
  \item (व्यावहारिक मानदंड) पंद्रह टाइलें पढ़ें
        उनके कक्षों का पढ़ने का क्रम, रिक्त स्थान छोड़ें, और जाने दें
        $ N $ इस सूची के व्युत्क्रमों की संख्या हो; होने देना $ r $ हो
        रिक्त स्थान की पंक्ति से गिना गया \emph{तल}. बताते हैं कि $ I(\sigma) = (-1)^{N + r + 1}$, ताकि $\sigma $ है
        हल करने योग्य यदि और केवल यदि $ N + r $ अजीब है.
  \item (समूह क्रियाएँ) An \emph{कार्रवाई} एक समूह का $ G $ एक सेट पर
        $ X $ एक नक्शा है $ G \times X \to X $, $(g, x) \mapsto g \cdot
        x $, साथ $ e \cdot x = x $ और $ g \cdot (h \cdot x) =
        (gh) \cdot x $; \emph{कक्षा} का $ x $ है $ G \cdot x $, और
        कार्रवाई है \emph{मुक्त} कब $ g \cdot x = x $ बल $ g =
        e $. बताते हैं कि $ h \cdot \sigma = \sigma \circ h^{-1}$
        की एक स्वतंत्र कार्रवाई को परिभाषित करता है $ H $ ब्लैंक-होम के सेट पर
        विन्यास, कि इसकी कक्षाएँ बिल्कुल उसी वर्ग की हैं
        कार्यक्रमों द्वारा आपसी पहुंच, और से उबरना
        कक्षा गणना कि ये विन्यास बिल्कुल विभाजित हो गए
        $15!\,/\,\abs H = 2$ कक्षाएं.
  \item (द $3 \times 3$ रुकावट) दिखाएँ कि $3 \times 3$
        बोर्ड मानता है \emph{नहीं} हर कोशिका का दौरा करते हुए बंद चलना
        ठीक एक बार: भाग III की भव्य यात्रा रणनीति विफल हो जाती है
        आठ पहेली के लिए. \emph{(नौ को चेकरबोर्ड करें
        कोशिकाएँ।)}
  \item (मरम्मत) पर $3 \times 3$ कोशिकाओं के साथ बोर्ड $1$ को
        $9$ पढ़ने के क्रम में और घर पर $9$: के प्रभावों की गणना करें
        परिधि यात्रा $9 \to 8 \to 7 \to 4 \to 1 \to 2 \to 3
        \to 6 \to 9$ (ए $7$-चक्र $\zeta'$ केंद्र को ठीक करना $5$)
        और कोने का दौरा $9 \to 6 \to 5 \to 8 \to 9$ (ए
        $3$-केंद्र के माध्यम से \omterm{def:b2:structures:sn}{चक्र})। उत्तरार्द्ध को संयुग्मित करना
        की शक्तियां $\zeta'$ और लेम्मास ए और बी को चेनिंग करते हुए साबित करें
        कि आठ पहेली का कार्यक्रम समूह सभी का है $ A_8$,
        इसलिए बिल्कुल वैसा ही $9!/2 = 181\,440$ की $9! =
        362\,880$ कॉन्फ़िगरेशन हल करने योग्य हैं.
  \item (एक घटिया बोर्ड) अब बोर्ड को सिंगल रहने दीजिए \omterm{def:b2:structures:sn}{चक्र} का $ n
        \geq 4$ कोशिकाएं ले जा रही हैं $ n - 1$ टाइल्स. दिखाएँ कि \omterm{def:b2:structures:generated}{चक्रीय}
        टाइल्स का क्रम अपरिवर्तनीय है, प्रत्येक पहुंच योग्यता
        क्लास बिल्कुल सही है $ n(n - 1)$ विन्यास \emph{(द
        कक्षाएँ a की कक्षाएँ हैं \omterm{def:b2:structures:generated}{चक्रीय} समूह का \omterm{def:b2:structures:generated}{आदेश}
        $\operatorname{lcm}(n, n-1) = n(n-1)$)}, और वह वहां
        हैं $(n - 2)!$ कक्षाएं --- के लिए $ n \geq 5$ से कहीं अधिक
        $2$: एक पतले बोर्ड पर समता अपरिवर्तनीय लगभग कब्जा कर लेता है
        कुछ भी नहीं, और ज्यामिति नियम।
  \item प्रश्न 19 की कसौटी पर दो निर्णय: पूर्णतः
        उलटी ट्रे (टाइल्स) $15, 14, \dots, 1$ कोशिकाओं में $1$
        को $15$, खाली घर) और सेल में खाली जगह वाली ट्रे
        $1$ टाइल्स के बाद $15, 14, \dots, 1$ कोशिकाओं में $2$

  \item (संश्लेषण) प्रमाण के दो स्वतंत्र स्तंभ हैं: एक
        \emph{अचल} ($ I $, \omterm{thm:b2:structures:signature}{हस्ताक्षर} रूपवाद से निर्मित)
