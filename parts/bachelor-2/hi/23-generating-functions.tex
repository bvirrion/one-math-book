\chapter{संभाव्यता उत्पन्न करने वाले कार्य}\label{ch:b2:genfun}

की शक्ति श्रृंखला \cref{ch:b2:powerseries} ए के साथ वापसी
संभाव्य मिशन: एक के लिए $\N $-मूल्यांकित \omterm{def:b2:randomvar:law}{अनियमित परिवर्तनशील वस्तु} हम संलग्न करते हैं
गुणांकों के साथ शक्ति श्रृंखला $\P(X = n)$. यह
\emph{\omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन}} का योग परिवर्तित करता है \omterm{def:b2:proba:independence}{स्वतंत्र} चर
उत्पादों में, क्षणों में डेरिवेटिव में $1$, और कठिन
संयोजनात्मक सर्वसमिकाओं को एक-पंक्ति गुणन में। द
अध्याय पुस्तक को दो शोपीस के साथ बंद करता है: पॉइसन
दुर्लभ का सन्निकटन \omterm{def:b2:proba:space}{घटनाएँ}, और विलुप्त होने की कसौटी
शाखाकरण प्रक्रियाएँ --- वास्तव में अनंत संभाव्यता
गणना पूरी तरह से \omterm{def:b2:affine:convex}{उत्तल} वक्र की ज्यामिति द्वारा हल की जाती है।

\section{परिभाषा और बुनियादी गुण}

\begin{definition}[संभाव्यता उत्पन्न करने वाला कार्य]\label{def:b2:genfun:pgf}
होने देना $ X $ एक हो $\N $-मूल्यांकित \omterm{def:b2:randomvar:law}{अनियमित परिवर्तनशील वस्तु}, $ p_n = \P(X = n)$.
\emph{संभाव्यता उत्पन्न करने वाला कार्य}\index{संभाव्यता उत्पन्न करने वाला कार्य} का $ X $ का योग है
शक्ति श्रृंखला
\[
G_X(t) = \E\bigl(t^X\bigr) = \sum_{n=0}^{\infty} p_n\,t^n .
\]
\end{definition}

\begin{example}[पहली सजगता]\label{ex:b2:genfun:firstreflexes}
एक स्थिर चर $ X = c $ है $ G_X(t) = t^c $; एक बदलाव का पालन करता है
$ G_{X+c}(t) = t^c\,G_X(t)$; और विशेष बिंदुओं पर मूल्यांकन कर रहे हैं
बिना किसी विस्तार के जानकारी को पढ़ता है: $ G_X(0) = \P(X
= 0)$, $ G_X(1) = 1$, और $ G_X(-1) = \P(X\text{ even}) -
\P(X\text{ odd})$, समता संतुलन का शोषण किया गया
\cref{exo:b2:genfun:10}. इन वन-लाइनर्स का उपयोग चुपचाप किया जाता है
नीचे हर जगह --- और मूल्यांकन $ G_X(0)$ बिलकुल है
विलुप्त होने की संभावनाओं को पुनरावृत्त से कैसे निकाला जाएगा
कार्य उत्पन्न करना अध्याय के अंत में.
\end{example}

\begin{proposition}[त्रिज्या और प्रथम गुण]\label{prop:b2:genfun:radius}
परिभाषित करने वाली श्रृंखला $ G_X $ है \omterm{def:b2:powerseries:radius}{अभिसरण की त्रिज्या} $\geq 1$;
$ G_X $ परिभाषित किया गया है और \omterm{def:b2:metric:continuity}{निरंतर} पर $\intcc{-1}{1}$,
$\mathcal{C}^\infty $ पर $\intoo{-1}{1}$, साथ $ G_X(1) = 1$ और
$\abs{G_X(t)} \leq 1$ वहाँ। इसके अतिरिक्त $ G_X $ निर्धारित करता है \omterm{def:b2:randomvar:law}{कानून} का
$ X $:
\[
p_n = \frac{G_X^{(n)}(0)}{n!} .
\]
\end{proposition}

\begin{proof}
तब से $\sum p_n = 1$ \omterm{def:b2:integration:improper}{अभिसरण}, शर्तें $ p_n\,1^n $ बंधे हुए हैं,
तो त्रिज्या है $\geq 1$ (हाबिल की लेम्मा,
\cref{ch:b2:powerseries}); पर $ t = \pm1$ श्रृंखला एकत्रित होती है
\omterm{def:b2:series:def}{बिल्कुल} ($\sum p_n = 1$ हावी है); बेहतर, कुल मिलाकर
\omterm{def:b2:diffcalc:differential}{अंतर}ाल $\intcc{-1}1$,
\[
\sup_{\abs t\leq1}\,\abs{p_nt^n} = p_n
\quad\text{with}\quad \sum_np_n < \infty :
\]
श्रृंखला एकत्रित होती है \emph{\omterm{def:b2:funcseq:series}{सामान्यतः}} पर $\intcc{-1}1$, तो
इसका योग है \omterm{def:b2:metric:continuity}{निरंतर} वहाँ
(\cref{thm:b2:funcseq:weierstrass,thm:b2:funcseq:continuity}). अंदर चिकनाई और गुणांक
सूत्र शक्ति श्रृंखला के सामान्य सिद्धांत हैं; गुणांक
पुनर्प्राप्ति योग्य होने के कारण, दो चर समान हैं \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन}
एक ही लें \omterm{def:b2:randomvar:law}{कानून}.
\end{proof}

\begin{example}[शास्त्रीय कानून]\label{ex:b2:genfun:classical}
\leavevmode
\begin{itemize}
\item Bernoulli $\mathcal{B}(p)$: $ G(t) = 1 - p + pt $.
\item द्विपद $\mathcal{B}(n, p)$: $ G(t) = \sum_k \binom nk
(pt)^k(1-p)^{n-k} = (1 - p + pt)^n $ (द्विपद प्रमेय)।
\item ज्यामितिक $\mathcal{G}(p)$: $ G(t) =
\sum_{k\geq1}(1-p)^{k-1}p\,t^k = \dfrac{pt}{1 - (1-p)t}$ (त्रिज्या
$\frac{1}{1-p} > 1$).
\item प्वासों $\mathcal{P}(\lambda)$: $ G(t) =
\sum_k e^{-\lambda}\frac{(\lambda t)^k}{k!} = e^{\lambda(t - 1)}$
(त्रिज्या $\infty $).
\end{itemize}
\end{example}

\begin{example}[जनरेटिंग को एकीकृत करना
समारोह]\label{ex:b2:genfun:integraltrick}
के व्युत्पन्न $ G_X $ पर $1$ सकारात्मक क्षण दें;
\emph{अभिन्न} एक नकारात्मक देता है. से $\int_0^1t^k\dd t
= \frac1{k+1}$ और अवधि-दर-अवधि एकीकरण (सामान्य
\omterm{def:b2:integration:improper}{अभिसरण} पर $\intcc01$):
\[
\int_0^1G_X(t)\,\dd t = \sum_{k\geq0}\frac{\P(X =
k)}{k+1} = \E\Bigl(\frac1{1+X}\Bigr).
\]
के लिए $ X \sim \mathcal P(\lambda)$:
\[
\E\Bigl(\frac1{1+X}\Bigr) =
\int_0^1\eu^{\lambda(t-1)}\,\dd t = \frac{1 -
\eu^{-\lambda}}{\lambda},
\]
की श्रृंखला गणना को एक पंक्ति में पुनर्प्राप्त करना
\cref{ex:b2:randomvar:transferex}. \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन}
एक दोतरफा उपकरण है: पर \omterm{def:b2:diffcalc:differential}{अंतर} करें $1$ के लिए
क्षण $\E(X)$, $\E(X(X-1))$, एकीकृत करें $\intcc01$
के लिए $\E\bigl(\frac1{1+X}\bigr)$ --- एक \omterm{def:b2:powerseries:analytic}{विश्लेषणात्मक} वस्तु,
समस्या की जिस भी दिशा में आवश्यकता हो, उस दिशा में पूछताछ की जाए।
\end{example}

\begin{example}[बिल्कुल त्रिज्या वाला एक कानून
एक]\label{ex:b2:genfun:heavytail}
होने देना $\P(X = k) = \dfrac{6}{\pi^2k^2}$ के लिए $ k \geq 1$ --- ए
संभाव्यता \omterm{def:b2:randomvar:law}{कानून} बेसल की पहचान से
(\cref{ex:b2:fourier:basel}). इसका \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन}
$ G(t) = \frac6{\pi^2}\sum_{k\geq1}\frac{t^k}{k^2}$ है त्रिज्या
\omterm{def:b2:integration:improper}{अभिसरण} का \omterm{def:b2:series:def}{बिल्कुल} $1$: सामान्य सीमा ``त्रिज्या $\geq
1$'' का \cref{prop:b2:genfun:radius} सुधार नहीं किया जा सकता. और
मतलब है
\[
\sum_{k\geq1}k\,\P(X = k) =
\frac6{\pi^2}\sum_{k\geq1}\frac1k = \infty :
\]
$ G $ है \omterm{def:b2:metric:continuity}{निरंतर} पर $\intcc{-1}1$, अंदर से चिकना, लेकिन यह
व्युत्पन्न पर विस्फोट होता है $1^-$ --- ग्राफ़ पर आता है
बिंदु $(1, 1)$ एक ऊर्ध्वाधर स्पर्शरेखा के साथ. भारी पूंछ हैं
दृश्यमान \emph{ज्यामितीय} पर \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन}, पर
एकल बिंदु $ t = 1$; नीचे दिए गए क्षण प्रमेय बनाता है
यह पत्राचार सटीक.
\end{example}

\begin{theorem}[सृजन से क्षण
समारोह]\label{thm:b2:genfun:moments}
$ X $ एक है \omterm{def:b2:randomvar:expectation}{अपेक्षा} अगर और केवल अगर $ G_X $ है \omterm{def:b2:diffcalc:differential}{विभेदक} पर
$1^-$ (बाएं व्युत्पन्न, परिमित), और फिर $\E(X) = G_X'(1)$.
उसी प्रकार $ X $ एक दूसरा क्षण है iff $ G_X $ दो बार है
\omterm{def:b2:diffcalc:differential}{विभेदक} पर $1^-$, और तब
\[
\E\bigl(X(X - 1)\bigr) = G_X''(1),
\qquad
V(X) = G_X''(1) + G_X'(1) - G_X'(1)^2 .
\]
\end{theorem}

\begin{proof}
के लिए $ t \in \intoo{0}{1}$, शब्द-दर-अवधि विभेदन के अंदर
डिस्क देता है $ G_X'(t) = \sum_{n\geq1} np_n t^{n-1}$, के साथ एक श्रृंखला
अऋणात्मक गुणांक: $ t \mapsto G_X'(t)$ पर घट नहीं रहा है
$\intoo{0}{1}$, और आंशिक रकम के मोनोटोन \omterm{def:b2:integration:improper}{अभिसरण} द्वारा (या
अऋणात्मक गुणांकों के लिए हाबिल का प्रमेय,
\cref{ch:b2:powerseries}),
\[
\lim_{t \to 1^-} G_X'(t)
= \sum_{n\geq1} n\,p_n \in \intcc{0}{+\infty} ,
\]
प्रत्येक पक्ष \omterm{def:b2:series:def}{बिल्कुल} परिमित होता है जब दूसरा होता है। जब परिमित, मतलब
मूल्य प्रमेय \omterm{def:b2:diffcalc:differential}{अंतर} भागफल को निचोड़ता है $\frac{G_X(1) -
G_X(t)}{1 - t}$ के मूल्यों के बीच $ G_X'$, इसलिए $ G_X $ है
\omterm{def:b2:diffcalc:differential}{विभेदक} पर $1^-$ साथ $ G_X'(1) = \sum np_n = \E(X)$ (द्वारा
\omterm{def:b2:structures:sn}{स्थानांतरण})। दूसरे क्रम का कथन तर्क को एक स्तर पर दोहराता है
ऊपर: $ G''_X(t) = \sum_{n\geq2}n(n-1)p_nt^{n-2}$ है
न घटनेवाला $\intoo01$ मोनोटोन सीमा के साथ
$\sum_nn(n-1)p_n = \E(X(X-1))$, परिमित \omterm{def:b2:series:def}{बिल्कुल} कब $ X $ है
एक दूसरा क्षण. \omterm{def:b2:randomvar:variance}{झगड़ा} सूत्र तब से अनुसरण करता है
के\"ओनिग--ह्यजेंस:
\[
V(X) = \E(X^2) - \E(X)^2 = \E\bigl(X(X-1)\bigr) + \E(X) -
\E(X)^2 = G''_X(1) + G'_X(1) - G'_X(1)^2 .
\]
\end{proof}

\begin{example}
पॉइसन: $ G'(t) = \lambda e^{\lambda(t-1)}$, इसलिए $\E(X) = \lambda $;
$ G''(1) = \lambda^2$, इसलिए $ V(X) = \lambda^2 + \lambda - \lambda^2 =
\lambda $ --- की गणना \cref{ch:b2:randomvar} एक में
प्रत्येक पंक्ति.
\end{example}

\begin{example}[पॉइसन कानून की विधि]
\label{ex:b2:genfun:poissonmode}
कहाँ है $\P(X = k)$ के लिए सबसे बड़ा $ X \sim \mathcal
P(\lambda)$? अनुपात के माध्यम से लगातार वजन की तुलना की जाती है
\[
\frac{\P(X = k+1)}{\P(X = k)} = \frac{\lambda}{k + 1} ,
\]
जो अधिक है $1$ जबकि $ k < \lambda - 1$ और नीचे गिर जाता है
$1$ एक बार $ k > \lambda - 1$: वजन बढ़ता है और फिर गिरता है
मोड $\floor\lambda $ (और बीच में टाई $\lambda - 1$ और
$\lambda $ कब $\lambda $ एक पूर्णांक है: के लिए $\lambda = 3$,
$\P(X = 2) = \P(X = 3) = \frac92\eu^{-3} \approx 0.224$).
गुणांकों पर अनुपात परीक्षण अक्सर सबसे तेज़ मार्ग होते हैं
किसी असतत के बारे में गुणात्मक तथ्य \omterm{def:b2:randomvar:law}{कानून} --- नहीं उत्पन्न करना
समारोह आवश्यक है, लेकिन गुणांक \emph{हैं} 
\omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन}, शब्द दर शब्द पढ़ें।
\end{example}

\section{स्वतंत्र चरों का योग}

\begin{theorem}[बहुलता]\label{thm:b2:genfun:product}
अगर $ X $ और $ Y $ हैं \omterm{def:b2:proba:independence}{स्वतंत्र} $\N $-मूल्यांकित यादृच्छिक चर, तब
\[
G_{X + Y}(t) = G_X(t)\,G_Y(t)
\qquad (\abs t \leq 1),
\]
और प्रेरण द्वारा $ G_{X_1 + \dots + X_n} = \prod_i G_{X_i}$ के लिए
\omterm{def:b2:proba:independence}{स्वतंत्र} $ X_1, \dots, X_n $.
\end{theorem}

\begin{proof}
दो प्रमाण, दोनों शिक्षाप्रद। \emph{के जरिए अपेक्षाएं:} $ t^X $ और
$ t^Y $ हैं \omterm{def:b2:proba:independence}{स्वतंत्र} बंधे हुए चर, इसलिए
(\cref{thm:b2:randomvar:product})
\[
G_{X+Y}(t) = \E\bigl(t^{X+Y}\bigr)
= \E\bigl(t^X t^Y\bigr)
= \E\bigl(t^X\bigr)\E\bigl(t^Y\bigr) .
\]
\emph{के जरिए \omterm{thm:b2:series:fubini}{कॉची उत्पाद}:}  \omterm{def:b2:randomvar:law}{कानून} का $ X + Y $ कनवल्शन है
$\P(X + Y = n) = \sum_{k=0}^n \P(X = k)\P(Y = n - k)$, और यह
\omterm{thm:b2:series:fubini}{कॉची उत्पाद} के लिए प्रमेय \omterm{def:b2:series:def}{बिल्कुल} \omterm{def:b2:integration:improper}{अभिसरण} श्रृंखला
(\cref{ch:b2:series}) दो शक्ति श्रृंखलाओं को सटीक रूप से गुणा करता है
इस कनवल्शन के साथ.
\end{proof}

\begin{example}[शास्त्रीय कानूनों की स्थिरता]\label{ex:b2:genfun:stability}
\omterm{def:b2:proba:independence}{स्वतंत्र} उसी के साथ द्विपद $ p $ जोड़ना: $(1 - p + pt)^m(1 -
p + pt)^n = (1 - p + pt)^{m+n}$, इसलिए $\mathcal{B}(m, p) +
\mathcal{B}(n, p) = \mathcal{B}(m + n, p)$ --- विशेष रूप से एक राशि
का $ n $ \omterm{def:b2:proba:independence}{स्वतंत्र} बर्नौली चर द्विपद है, इसे पुनः सिद्ध करना
\omterm{def:b2:randomvar:law}{कानून} सफलताओं की संख्या का. \omterm{def:b2:proba:independence}{स्वतंत्र} पॉइसन्स जोड़ते हैं:
$ e^{\lambda(t-1)}e^{\mu(t-1)} = e^{(\lambda + \mu)(t-1)}$, इसलिए
$\mathcal{P}(\lambda) + \mathcal{P}(\mu) = \mathcal{P}(\lambda +
\mu)$ --- कनवल्शन गणना
\cref{exo:b2:randomvar:2}, अब बिना गणना के।
\end{example}

\begin{example}[दो पासे, एक बहुपद
चुकता]\label{ex:b2:genfun:twodice}
एक निष्पक्ष पासे के लिए, $ G(t) = \frac{t + t^2 + \dots + t^6}{6}$;
दो के योग के लिए,
\[
G(t)^2 = \frac{1}{36}\bigl(t^2 + 2t^3 + 3t^4 + 4t^5 + 5t^6 +
6t^7 + 5t^8 + 4t^9 + 3t^{10} + 2t^{11} + t^{12}\bigr) :
\]
\omterm{def:b2:reduction:diag}{त्रिकोणीय} \omterm{def:b2:randomvar:law}{कानून} पासों का योग ($7$ मोड है, के साथ
संभाव्यता $\frac6{36} = \frac16$), एक बहुपद को पढ़ें
वर्ग जिसे जीवन में एक बार गुणा किया जाता है। द
कनवल्शन फ़ॉर्मूले के लिए ग्यारह अलग-अलग की आवश्यकता होगी
तर्क गिनना; \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन} वो सब करता है
एक साथ, क्योंकि बहुपदों को गुणा करना \emph{है}
संघटित गुणांक. यह यांत्रिक अनुवाद ---
\omterm{def:b2:randomvar:law}{कानून} गुणांकों का, उत्पादों का योग --- संपूर्ण है
अध्याय का व्यवसाय मॉडल, और
\cref{exo:b2:genfun:11} इसे आश्चर्य की ओर धकेलता है
सिचरमन पासा.
\end{example}

\begin{example}[तीन पासे और एक गुणांक
निष्कर्षण]\label{ex:b2:genfun:threedice}
राशि के लिए $ S $ तीन निष्पक्ष पासों में से, $\P(S = 10)$ है
का गुणांक $ t^{10}$ में $\bigl(\frac{t + \dots +
t^6}6\bigr)^3$. द्विपद और के साथ गुणनखंड और विस्तार करें
ज्यामितीय श्रृंखला:
\[
\Bigl(\frac{t(1 - t^6)}{6(1 - t)}\Bigr)^{\!3}
= \frac{t^3}{216}\,\bigl(1 - 3t^6 + 3t^{12} -
t^{18}\bigr)\sum_{j\geq0}\binom{j+2}2t^j .
\]
का गुणांक $ t^{10}$ आवश्यक है $ t^7$ उत्पाद से:
$ j = 7$ शब्द के साथ $1$, और $ j = 1$ शब्द के साथ
$-3t^6$:
\[
\P(S = 10) = \frac{1}{216}\Bigl(\binom92 -
3\binom32\Bigr) = \frac{36 - 9}{216} = \frac{27}{216} =
\frac18 .
\]
की सीधी गणना $27$ त्रिगुण त्रुटि-प्रवण है; द
बीजगणित यांत्रिक है और किसी भी संख्या के पासों को मापता है ---
समावेश--बहिष्करण दिखाई देता है $(1 - t^6)^3$ कर रहा है
मामले का विश्लेषण स्वचालित रूप से।
\end{example}

\begin{example}[किसी कानून को उसके निर्माण से पढ़ना
समारोह]\label{ex:b2:genfun:identify}
कौन \omterm{def:b2:randomvar:law}{कानून} है $ G(t) = \dfrac1{2 - t}$? एक शक्ति में विस्तार करें
शृंखला:
\[
\frac{1}{2 - t} = \frac12\cdot\frac1{1 - t/2}
= \sum_{k\geq0}\frac{t^k}{2^{k+1}} :
\]
अऋणात्मक गुणांकों का योग $ G(1) = 1$, तो यह एक है
वास्तविक \omterm{def:b2:randomvar:law}{कानून}, $\P(X = k) = 2^{-(k+1)}$ पर $\N $ --- ए
ज्यामितीय \omterm{def:b2:randomvar:law}{कानून} से शुरू हो रहा है $0$. विशिष्टता से
(\cref{prop:b2:genfun:radius}), कोई अन्य नहीं \omterm{def:b2:randomvar:law}{कानून} इसे साझा करता है
$ G $. मान्यता देना \omterm{def:b2:randomvar:law}{कानून} उनके से कार्य उत्पन्न करना एक है
ड्रिलिंग के लायक कौशल: यह महत्वपूर्ण शाखा कैसे है
पुनरावृत्त करना $ G_n(t) = \frac{n - (n-1)t}{n+1 - nt}$ का
सप्ताहांत की समस्या एक ज्यामितीय के रूप में उजागर हुई है \omterm{def:b2:randomvar:law}{कानून} वातानुकूलित
अस्तित्व पर.
\end{example}

\begin{remark}
स्थिरता केवल एक ही रास्ते पर चलती है: का योग \omterm{def:b2:proba:independence}{स्वतंत्र} पॉइज़न हैं
पॉइसन, लेकिन मतभेद नहीं हैं --- $ X - Y $ नकारात्मक लेता है
मूल्य, इसलिए इसका कोई नहीं है \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन} \omterm{def:b2:series:def}{बिल्कुल}, और यह \omterm{def:b2:randomvar:law}{कानून}
(स्केलम \omterm{def:b2:randomvar:law}{वितरण}) इस अध्याय के बाहर है
टूलकिट. वैसे ही $\mathcal B(m, p) + \mathcal B(n, p')$ साथ
$ p \neq p'$ है \emph{नहीं} द्विपद: उत्पाद $(1 - p +
pt)^m(1 - p' + p't)^n $ जबकि, इसके दो अलग-अलग मूल स्थान हैं
प्रत्येक द्विपद पीजीएफ में एक ही दोहराया गया मूल होता है। पढ़ना
रूट पैटर्न से स्थिरता कितनी है इसका एक छोटा सा पूर्वावलोकन है
बहुपद एन्कोड की संरचना करें।
\end{remark}

\begin{remark}[एकता की जड़ें फिल्टर]
पर मूल्यांकन कर रहे हैं $-1$ सम को विषम से अलग करता है; \omterm{def:b2:series:def}{बिल्कुल} मूल्यांकन करना
$ m $- एकता की जड़ें प्रत्येक अवशेष वर्ग को अलग करती हैं: साथ
$\omega = \eu^{2\iu\pi/m}$,
\[
\P(X \equiv r \bmod m)
= \frac1m\sum_{j=0}^{m-1}\omega^{-jr}\,G_X(\omega^j),
\]
औसत के बाद से $\omega^{j(k-r)}$ ऊपर $ j $ पैदावार $1$ अगर $ k
\equiv r $ और $0$ अन्यथा। नमूना लाभांश: राशि के लिए $ S $
दो निष्पक्ष पासों में से प्रत्येक $ G(\omega^j) = \frac16\sum_{k=1}^6
\omega^{jk} = -\frac16$ के लिए $ j \neq 0$ (सात सातवां
एकता की जड़ों का योग शून्य होता है), इसलिए
\[
\P(7 \mid S) = \frac17\Bigl(1 +
6\cdot\frac1{36}\Bigr) = \frac16 ,
\]
से गिनती की पुष्टि करना \cref{ex:b2:genfun:twodice} ---
और विधि उन प्रश्नों को मापती है जहां सीधी गिनती होती है
नहीं करता.
\end{remark}

\begin{theorem}[यादृच्छिक रकम: उत्पन्न करने के लिए वाल्ड की पहचान
कार्य]\label{thm:b2:genfun:compound}
होने देना $(X_k)_{k\geq1}$ होना \omterm{def:b2:proba:independence}{स्वतंत्र} $\N $-मूल्यवान चर के साथ
वही \omterm{def:b2:randomvar:law}{कानून} और \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन} $ G_X $, और चलो $ N $ एक हो
$\N $-मूल्यवान चर \omterm{def:b2:proba:independence}{स्वतंत्र} की $ X_k $, साथ उत्पन्न करना
समारोह $ G_N $. फिर यादृच्छिक योग $ S = X_1 + \dots + X_N $ (साथ
$ S = 0$ कब $ N = 0$) है \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन}
\[
G_S = G_N \circ G_X .
\]
विशेषकर, यदि $ N $ और $ X_1$ पास होना अपेक्षाएं, $\E(S) =
\E(N)\,\E(X_1)$.
\end{theorem}

\begin{proof}
शर्त चालू $ N $ (कुल \omterm{def:b2:multint:exact}{संभावना},
\cref{thm:b2:proba:bayes}): के लिए $\abs t \leq 1$,
\[
G_S(t) = \sum_{n=0}^\infty \P(N = n)\,
\E\bigl(t^{X_1 + \dots + X_n}\bigr)
= \sum_{n=0}^\infty \P(N = n)\,G_X(t)^n
= G_N\bigl(G_X(t)\bigr),
\]
प्रत्येक निश्चित के लिए गुणनशीलता का उपयोग करना $ n $ और यह संक्षेपण का
\omterm{def:b2:metric:complete}{पूरा} दोहरा परिवार ($\abs{G_X(t)} \leq 1$). का आदान-प्रदान
सारांश फ़ुबिनी के लिए है \omterm{def:b2:series:summable}{संक्षेपणीय} परिवार
(\cref{ch:b2:series}). पर विभेद करना $1^-$ शृंखला नियम द्वारा
और \cref{thm:b2:genfun:moments}: $\E(S) = G_N'(G_X(1))\,G_X'(1)
= G_N'(1)G_X'(1) = \E(N)\E(X_1)$.
\end{proof}

\begin{example}[कंपाउंड पॉइसन: वार्षिक बीमा
घाटा]\label{ex:b2:genfun:compoundpoisson}
एक बीमाकर्ता प्राप्त करता है $ N \sim \mathcal P(\lambda)$ ए में दावा
वर्ष, प्रत्येक दावे की लागत $ X_k $ (पूर्णांक इकाइयाँ, i.i.d., pgf
$ G_X $, अर्थ $\mu $, से \omterm{def:b2:proba:independence}{स्वतंत्र} $ N $). द्वारा
\cref{thm:b2:genfun:compound}, कुल हानि $ S $ है
\[
G_S(t) = \eu^{\lambda(G_X(t) - 1)},
\qquad
\E(S) = \lambda\mu ,
\]
और दो बार \omterm{def:b2:diffcalc:differential}{अंतर} करना $1^-$:
\[
V(S) = \lambda\,G_X''(1) + \lambda^2\mu^2 + \lambda\mu -
(\lambda\mu)^2 = \lambda\,\E(X^2) .
\]
 \omterm{def:b2:randomvar:variance}{झगड़ा} शामिल है \emph{दूसरा} एकल का क्षण
दावा, इसका नहीं \omterm{def:b2:randomvar:variance}{झगड़ा}: एक यौगिक पॉइसन योग लगता है
कभी-कभार बड़ा दावा दो बार --- एक बार से कितने तक, एक बार
कितने बड़े के माध्यम से. के लिए $\lambda = 10$ ज्यामितीय के दावे \omterm{def:b2:randomvar:law}{कानून}
मतलब के साथ $2$ ($\E X^2 = 6$): $\E S = 20$, $ V(S) = 60$, और
चेबीशेव (\cref{ch:b2:randomvar}) पहले से ही उपयोग योग्य उपज देता है
सॉल्वेंसी मार्जिन। यह ``यादृच्छिक रूप से रोका गया योग'' पैटर्न है
वही जो ब्रांचिंग रिकर्सन को चलाएगा
\cref{prop:b2:genfun:branching}: की रचना उत्पन्न करना
कार्य यादृच्छिक आबादी का बीजगणित है।
\end{example}

\begin{remark}
 स्वतंत्रता का $ N $ सारांश से सजावटी नहीं है.
लेना $ X_k \in \{0, 2\}$ समान संभावनाओं के साथ और चलो $ N =
X_1$ (स्पष्ट रूप से निर्भर): फिर $ S = X_1 + \dots + X_N $ है
$0$ कब $ X_1 = 0$, और $2 + X_2$ कब $ X_1 = 2$, इसलिए $\E(S) =
\frac12(2 + 1) = \frac32$, जबकि $\E(N)\E(X_1) = 1\cdot1 =
1$: वाल्ड की पहचान विफल हो गई। जब पदों की संख्या है
करने की अनुमति दी \emph{प्रतिक्रिया} स्वयं की शर्तों के लिए, स्वच्छ
उत्पाद संरचना ध्वस्त हो जाती है---इस तरह का \omterm{def:b2:metric:complete}{पूरा} सिद्धांत
``रोकना'' नियम वर्ष 3 का मार्टिंगेल अध्याय है
आयतन.
\end{remark}

\section{पॉइसन सन्निकटन}

\begin{theorem}[दुर्लभ घटनाओं का नियम]\label{thm:b2:genfun:rareevents}
होने देना $ X_n \sim \mathcal{B}(n, p_n)$ साथ $ n\,p_n \to \lambda > 0$.
फिर हर एक के लिए $ k \in \N $:
\[
\P(X_n = k)
\xrightarrow[n\to\infty]{}
e^{-\lambda}\frac{\lambda^k}{k!} :
\]
द्विपद \omterm{def:b2:randomvar:law}{कानून} बहुत से दुर्लभ \omterm{def:b2:proba:independence}{स्वतंत्र} \omterm{def:b2:proba:space}{घटनाएँ} में परिवर्तित हो जाता है
पॉइसन \omterm{def:b2:randomvar:law}{कानून} पैरामीटर का $\lambda $.
\end{theorem}

\begin{proof}
के साथ सीधी गणना $ p_n = \frac{\lambda_n}{n}$, $\lambda_n
\to \lambda $:
\[
\P(X_n = k)
= \binom nk p_n^k(1 - p_n)^{n-k}
= \frac{n(n-1)\cdots(n-k+1)}{n^k}\cdot
\frac{\lambda_n^k}{k!}\,
\bigl(1 - \tfrac{\lambda_n}{n}\bigr)^{n-k} .
\]
जैसा $ n \to \infty $ साथ $ k $ निश्चित: पहला कारक प्रवृत्त होता है $1$
(का उत्पाद $ k $ कारकों $\to 1$); $\lambda_n^k \to \lambda^k $;
और $\bigl(1 - \frac{\lambda_n}{n}\bigr)^{n-k} =
\exp\bigl((n-k)\ln(1 - \frac{\lambda_n}{n})\bigr) \to
e^{-\lambda}$ तब से $(n - k)\ln\bigl(1 - \frac{\lambda_n}{n}\bigr)
\sim -\lambda_n \to -\lambda $
(\cref{ch:b2:comparison}). वैकल्पिक रूप से, के स्तर पर
कार्य उत्पन्न करना: $ G_{X_n}(t) = \bigl(1 +
\frac{\lambda_n(t-1)}{n}\bigr)^n \to e^{\lambda(t - 1)} =
G_{\mathcal{P}(\lambda)}(t)$ प्रत्येक के लिए निश्चित $ t \in [0, 1]$ ---
का \omterm{def:b2:integration:improper}{अभिसरण} कार्य उत्पन्न करना, जो (के लिए $\N $-मूल्यांकित
चर) प्रत्येक के \omterm{def:b2:integration:improper}{अभिसरण} के बराबर है $\P(X_n = k)$;
देखना \cref{exo:b2:genfun:9}.
\end{proof}

\begin{remark}
यही कारण है कि पॉइसन \omterm{def:b2:randomvar:law}{कानून} दुर्लभ की मॉडल गणना \omterm{def:b2:proba:space}{घटनाएँ} --- टाइपो प्रति
पृष्ठ, प्रति सेकंड रेडियोधर्मी क्षय, प्रति दिन दुर्\omterm{def:b2:proba:space}{घटनाएँ}
पार करना: प्रत्येक अवसर लगभग नगण्य है, अवसर हैं
अनेक, और केवल औसत दर $\lambda $ सीमा में जीवित रहता है.
\end{remark}

\begin{example}[पॉइसन सीमा पर नजर रखना
एकत्रित होना]\label{ex:b2:genfun:poissonnumerics}
हल करना $\lambda = 2$ और चलो $ X_n \sim \mathcal B(n, 2/n)$.
कोई घटना नहीं \omterm{def:b2:multint:exact}{संभावना} \omterm{def:b2:series:def}{बिल्कुल} सही है $\P(X_n = 0) = (1 - 2/n)^n $:
\[
n = 10:\ 0.107, \qquad
n = 20:\ 0.122, \qquad
n = 50:\ 0.130, \qquad
n = 100:\ 0.133,
\]
सीमा के विरुद्ध $\eu^{-2} \approx 0.135$. \omterm{def:b2:integration:improper}{अभिसरण}
एकस्वर और गति का है $ O(1/n)$ --- विस्तार, $(1 -
2/n)^n = \eu^{-2}\bigl(1 - \tfrac2n + O(n^{-2})\bigr)$ ---
अभीतक के लिए तो $ n $ सैकड़ों की संख्या में पॉइसन मॉडल पहले से ही मौजूद है
तीसरे अंक तक सटीक. यह व्यावहारिक सामग्री है
दुर्लभ घटनाओं के नियम के बारे में: मॉडेलर कभी नहीं जानता $ n $ और
$ p $ अलग से (एक टाइपो के लिए कितने सूक्ष्म अवसर होते हैं
एक पेज होल्ड?), लेकिन केवल उनका उत्पाद $\lambda $, और
सीमा \omterm{def:b2:randomvar:law}{कानून} दया किसी और चीज़ पर निर्भर नहीं करती।
\end{example}

\section{शाखाकरण प्रक्रियाएँ}

एक पूर्वज से शुरू होने वाली जनसंख्या पर विचार करें; प्रत्येक व्यक्ति,
\omterm{def:b2:proba:independence}{स्वतंत्र} रूप से, उसके पास यादृच्छिक संख्या में बच्चे हैं \omterm{def:b2:randomvar:law}{कानून} $(p_k)_{k
\in \N}$ और \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन} $ G $ (द \emph{वंश
\omterm{def:b2:randomvar:law}{वितरण}}). होने देना $ Z_n $ पीढ़ी का आकार हो $ n $ ($ Z_0 =
1$), और चलो $ m = G'(1) = \E(Z_1)$ औसत संतान संख्या हो.

\begin{proposition}\label{prop:b2:genfun:branching}
 \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन} का $ Z_n $ है $ n $-वें पुनरावृत्त
$ G_{Z_n} = G \circ G \circ \dots \circ G $ ($ n $ टाइम्स), और
विलुप्त होने की संभावनाएँ $ q_n = \P(Z_n = 0)$ संतुष्ट
\[
q_0 = 0, \qquad q_{n+1} = G(q_n),
\]
और \omterm{def:b2:multint:exact}{संभावना} बढ़ जाती है $ q $ अंततः विलुप्त होने का, जो
का एक निश्चित बिंदु है $ G $.
\end{proposition}

\begin{proof}
पीढ़ी $ n + 1$ की संतानों का यादृच्छिक योग है $ Z_n $
पीढ़ी के सदस्य $ n $, गिनती के साथ \omterm{def:b2:proba:independence}{स्वतंत्र} एक दूसरे के
और का $ Z_n $: \cref{thm:b2:genfun:compound} देता है $ G_{Z_{n+1}} =
G_{Z_n} \circ G $, और प्रेरण से $ G_{Z_0}(t) = t $ की पैदावार होती है
$ n $-गुना पुनरावृत्त --- जो, रचना की साहचर्यता द्वारा, कर सकता है
\omterm{def:b2:funcseq:def}{समान रूप से} पढ़ा जाए $ G_{Z_{n+1}} = G \circ G_{Z_n}$. मूल्यांकन करना
यह दूसरा रूप है $0$: $ q_{n+1} = G_{Z_{n+1}}(0) =
G\bigl(G_{Z_n}(0)\bigr) = G(q_n)$. \omterm{def:b2:proba:space}{घटनाएँ} $\{Z_n = 0\}$ वृद्धि (विलुप्त आबादी बनी रहती है
विलुप्त), तो $ q_n \uparrow q = \P\bigl(\bigcup_n\{Z_n = 0\}\bigr)$
मोनोटोन द्वारा निरंतरता (\cref{thm:b2:proba:continuity}), और
निरंतरता का $ G $ पर $[0, 1]$ मोड़ों $ q_{n+1} = G(q_n)$ में $ q =
G(q)$ सीमा पर.
\end{proof}

\begin{example}[विलुप्ति को एकत्रित होते देखना]
\label{ex:b2:genfun:cobwebnumerics}
संतान के लिए \omterm{def:b2:randomvar:law}{कानून} $(p_0, p_1, p_2) = (\tfrac14,
\tfrac14, \tfrac12)$ का \cref{ex:b2:genfun:branchingexample},
$ G(t) = \tfrac14 + \tfrac14t + \tfrac12t^2$ और पुनरावृत्ति
$ q_{n+1} = G(q_n)$ देता है
\[
q_1 = 0.25, \quad q_2 = 0.34375, \quad q_3 \approx 0.39502,
\quad q_4 \approx 0.42678, \quad q_5 \approx 0.44776,
\]
विलुप्त होने की \omterm{def:b2:multint:exact}{संभावना} की ओर चढ़ना $ q = \tfrac12$.
\omterm{def:b2:diffcalc:differential}{अंतर}ाल $ q - q_n $ हैं $0.25$, $0.156$, $0.105$, $0.073$,
$0.052$: प्रत्येक मोटे तौर पर है $\tfrac34$ पिछले का, और
वास्तव में माध्य मान प्रमेय देता है $ q - q_{n+1} = G'(c_n)(q
- q_n)$ साथ $ G'(q) = \tfrac14 + q = \tfrac34$. दो नैतिकताएँ: ए
पीढ़ी दर पीढ़ी वंशावली अभी भी जीवित है $ n $ में बनाया गया है
वही गणना, संभाव्यता $ q - q_n $ बर्बाद होने का
बाद में; और सीढ़ी की \omterm{def:b2:integration:improper}{अभिसरण} दर
नीचे दिया गया चित्र निश्चित बिंदु --- पर व्युत्पन्न है
सप्ताहांत समस्या दोनों अवलोकनों को प्रमेयों में बदल देती है।
\end{example}

\begin{theorem}[विलुप्ति की कसौटी]\label{thm:b2:genfun:extinction}
मान लीजिए $ p_1 \neq 1$. विलुप्त होने की \omterm{def:b2:multint:exact}{संभावना} $ q $ है
\emph{सबसे छोटा} का निश्चित बिंदु $ G $ में $\intcc{0}{1}$, और:
\begin{itemize}
\item अगर $ m \leq 1$ (सबक्रिटिकल या क्रिटिकल), $ q = 1$: विलुप्ति
निश्चित है;
\item अगर $ m > 1$ (सुपरक्रिटिकल), $ q < 1$: जनसंख्या जीवित रहती है
हमेशा के लिए सकारात्मक \omterm{def:b2:multint:exact}{संभावना} के साथ $1 - q $.
\end{itemize}
\end{theorem}

\begin{proof}
$ G $ है \omterm{def:b2:affine:convex}{उत्तल} पर $\intcc{0}{1}$ (गैर-नकारात्मक के साथ शक्ति श्रृंखला
गुणांक: $ G'' \geq 0$), न घटने वाला, साथ $ G(1) = 1$.

\emph{सबसे छोटा निश्चित बिंदु:} होने देना $ r \in \intcc{0}{1}$ कोई भी निश्चित हो
बिंदु. तब $ q_0 = 0 \leq r $, और आगमनात्मक रूप से $ q_{n+1} = G(q_n)
\leq G(r) = r $ (एकरसता): तो $ q = \lim q_n \leq r $.

\emph{मामला $ m \leq 1$:} कल्पना करना $ r < 1$ एक निश्चित बिंदु है. द्वारा
माध्य मान प्रमेय पर $[r, 1]$, वहाँ है $ c \in \intoo{r}{1}$ साथ
$ G'(c) = \frac{G(1) - G(r)}{1 - r} = \frac{1 - r}{1 - r} = 1$. लेकिन
$ G'$ के साथ गैर-घटती (उत्तलता) नहीं है $\lim_{t\to1^-}G'(t) = m
\leq 1$, इसलिए $ G' \leq 1$ पर $\intoo{0}{1}$; समानता $ G'(c) =
1$ फिर मजबूर करता है $ G'$ के बराबर स्थिर रहना $1$ पर $\intco{c}{1}$,
इस तरह $ G'' = \sum n(n-1)p_nt^{n-2} \equiv 0$ वहाँ। एक शक्ति श्रृंखला
एक \omterm{def:b2:diffcalc:differential}{अंतर}ाल पर गायब होने वाले गैर-नकारात्मक गुणांक के साथ सब कुछ है
ये गुणांक शून्य: $ p_n = 0$ के लिए $ n \geq 2$, इसलिए $ G(t) = p_0
+ p_1t $ और $1 = G'(c) = p_1$ --- परिकल्पना का खंडन
$ p_1 \neq 1$. इसलिए $1$ एकमात्र निश्चित बिंदु है: $ q = 1$.

\emph{मामला $ m > 1$:} पास में $1$, $ G(t) - t $ व्युत्पन्न है $ G'(t) -
1 \to m - 1 > 0$ जैसा $ t \to 1^-$, इसलिए $ G(t) - t < G(1) - 1 = 0$ पर
कुछ \omterm{def:b2:diffcalc:differential}{अंतर}ाल $\intoo{1 - \delta}{1}$: द \omterm{def:b2:metric:continuity}{निरंतर} समारोह
$ G(t) - t $ है $\geq 0$ पर $ t = 0$ ($ G(0) = p_0 \geq 0$) और $< 0$
बस नीचे $1$, तो कुछ पर यह गायब हो जाता है $ r < 1$ (मध्यवर्ती मूल्य
प्रमेय). तब सबसे छोटा स्थिर बिंदु है $ q \leq r < 1$.
\end{proof}

\begin{remark}[मकड़ी का जाला कैसे पढ़ें]
चित्र में, एक ऊर्ध्वाधर चाल लागू होती है $ G $ (से $(q_n, q_n)$
तक $(q_n, G(q_n))$), विकर्ण की ओर एक क्षैतिज चाल
आउटपुट को इनपुट में परिवर्तित करता है: सीढ़ी \emph{है} द
प्रत्यावर्तन $ q_{n+1} = G(q_n)$. की उत्तलता $ G $ और $ G(1) = 1$
केवल दो ज्यामितियाँ छोड़ें। या तो वक्र ऊपर रहता है
विकर्ण पर $\intco01$ (अर्थ $ m \leq 1$): सीढ़ी है
पहले रुकने की कोई जगह नहीं $1$. या वक्र कुछ पर पार हो जाता है $ q <
1$ ($ m > 1$): सीढ़ी क्रॉसिंग के नीचे फंसी हुई है
और ज्यामितीय दर पर उसमें परिवर्तित हो जाता है $ G'(q) < 1$
में परिमाणित किया गया \cref{ex:b2:genfun:cobwebnumerics}. सभी
विलुप्ति प्रमेय का विश्लेषण इसमें दिखाई देता है
चित्र --- यही कारण है कि यह पहले खींचने लायक है
कंप्यूटिंग.
\end{remark}

\begin{example}\label{ex:b2:genfun:branchingexample}
वंशज \omterm{def:b2:randomvar:law}{कानून}: कोई बच्चा नहीं, एक बच्चा, दो बच्चे साथ
सम्भावनाएँ $\frac14, \frac14, \frac12$. तब $ m = \frac14 + 1 =
\frac54 > 1$ और $ G(t) = \frac14 + \frac14 t + \frac12 t^2$. ठीक किया गया
अंक: $\frac12 t^2 - \frac34 t + \frac14 = 0$, यानी\ $2t^2 - 3t
+ 1 = (2t - 1)(t - 1) = 0$: $ q = \frac12$. वंश-परम्परा मर जाती है
\omterm{def:b2:multint:exact}{संभावना} के साथ बाहर $\frac12$ --- और संभाव्यता के साथ $\frac12$
यह सदैव जीवित रहता है।
\end{example}

\begin{remark}[इस खंड के भीतर परिप्रेक्ष्य]
अध्याय पुस्तक का चौराहा और प्रत्येक घटक है
एक नामित स्थान से आया: श्रृंखला बीजगणित से
\cref{ch:b2:series} और \cref{ch:b2:powerseries}, द
\omterm{def:b2:multint:exact}{संभावना} से \cref{ch:b2:proba} (नीरस निरंतरता
को सिद्ध करता $ q_n \uparrow q $) और \cref{ch:b2:randomvar} ($ G_X =
\E(t^X)$ एक \omterm{def:b2:randomvar:expectation}{अपेक्षा}, बहुलता गुणनफल है
प्रमेय), से उत्तलता \cref{ch:b2:realfun} के माध्यम से
\cref{ch:b2:affine}. यहां तक कि हेवी-टेल पैथोलॉजी भी जुड़ती है:
पिछले अध्याय का सेंट पीटर्सबर्ग वैरिएबल है
$ G(t) = \sum_k2^{-k}t^{2^k}$, एक पूर्णतः \omterm{def:b2:integration:improper}{अभिसरण} श्रृंखला
पर $\intcc01$ जिसका व्युत्पन्न पर $1^-$ विचलन ---
अनंत माध्य, एक नज़र में दिखाई देने वाला। एक वस्तु, हर उपकरण
वर्ष का: एक उपयुक्त अंतिम अध्याय।
\end{remark}

\begin{remark}[सामान्य ख़तरे]
(मैं) कार्य उत्पन्न करना पर लागू $\N $-केवल मूल्यवान चर:
ऑब्जेक्ट पर \omterm{thm:b2:structures:signature}{हस्ताक्षर}ित या गैर-पूर्णांक चर के लिए $\E(t^X)$
अपनी शक्ति-श्रृंखला संरचना खो देता है (वर्ष 3 इसे प्रतिस्थापित करता है
परिवर्तनों को अनुकूलित किया गया $\R $). (ii) की पहली विवेक जांच
किसी भी गणना $ G $ है $ G(1) = 1$; दूसरा यह है कि
गुणांक अऋणात्मक होते हैं --- एक ऋणात्मक गुणांक
इसका मतलब बीजगणित की पर्ची है, नई नहीं \omterm{def:b2:randomvar:law}{कानून}. (iii) यादृच्छिक योगों में,
रचना क्रम मायने रखता है: $ G_S = G_N \circ G_X $, द
\emph{आउटर} शर्तें गिनने का कार्य; की रचना
अन्य तरीका अर्थहीन है ($ G_X \circ G_N $ वस्तुओं की गिनती करेगा
वस्तुओं का) (iv) गुणात्मकता की आवश्यकता स्वतंत्रता और
यादृच्छिकता के विशिष्ट स्रोत: $ G_{2X}(t) = G_X(t^2)$, नहीं
$ G_X(t)^2$. (v) पर विभेद करना $1$ एक सीमा है
ऑपरेशन: जब त्रिज्या \omterm{def:b2:series:def}{बिल्कुल} है $1$, के रूप में
\cref{ex:b2:genfun:heavytail}, $ G'(1^-)$ अनंत हो सकता है,
और क्षण प्रमेय का मोनोटोन-सीमा सूत्रीकरण नहीं है
एक पांडित्यपूर्ण बारीकियों लेकिन ईमानदार बयान।
\end{remark}

दो भरे हुए पासे: क्या दो पासों को लोड करना संभव है (\omterm{def:b2:proba:independence}{स्वतंत्र} रूप से,
\omterm{def:b2:funcseq:def}{समान रूप से} या नहीं) ताकि उनका योग एक समान हो $\{2, \dots,
12\}$? \emph{(समान गुणनखंडन बाधा के रूप में
\cref{exo:b2:genfun:1}: अलग-अलग होने पर भी जवाब नहीं है
लोड हो रहा है, क्योंकि प्रत्येक कारक $ G_X(t)/t $ अजीब डिग्री है $5$,
इसलिए वास्तविक जड़ है, जबकि लक्ष्य में कोई नहीं है।)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:genfun:4}
होने देना $ X_1, X_2, \dots $ होना \omterm{def:b2:proba:independence}{स्वतंत्र} Bernoulli $\mathcal{B}(p)$
और $ N \sim \mathcal{P}(\lambda)$ \omterm{def:b2:proba:independence}{स्वतंत्र} उनमें से. दिखाओ,
के माध्यम से \cref{thm:b2:genfun:compound}, वह $ S = X_1 + \dots + X_N
\sim \mathcal{P}(\lambda p)$: प्रत्येक रखी गई वस्तुओं की एक पॉइसन संख्या
संभाव्यता के साथ $ p $, एक पॉइसन नंबर छोड़ता है --- \emph{पतले}.
की भी गणना करें \omterm{def:b2:randomvar:law}{कानून} छोड़े गए गिनती और प्रशंसा की: यह है
$\mathcal{P}(\lambda(1-p))$, और कोई यह दिखा सकता है कि यह \omterm{def:b2:proba:independence}{स्वतंत्र} है
का $ S $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:genfun:5}
(ऋणात्मक द्विपद) चलो $ T_r $ प्राप्त करने के लिए टॉस की संख्या हो
$ r $ शीर्ष (सिर \omterm{def:b2:multint:exact}{संभावना}) $ p $). लिखना $ T_r $ के योग के रूप में $ r $
\omterm{def:b2:proba:independence}{स्वतंत्र} ज्यामितीय चर, निष्कर्ष निकालें
\[
G_{T_r}(t) = \Bigl(\frac{pt}{1 - (1-p)t}\Bigr)^{r},
\qquad
\E(T_r) = \frac rp,
\qquad
V(T_r) = \frac{r(1-p)}{p^2},
\]
और विस्तार करें $ G_{T_r}$ ढूँढ़ने के लिए $\P(T_r = n) = \binom{n-1}{r-1}
p^r(1-p)^{n-r}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:genfun:6}
संतान के लिए \omterm{def:b2:randomvar:law}{कानून} $ p_0 = \frac18$, $ p_1 = \frac38$, $ p_2 =
\frac38$, $ p_3 = \frac18$: गणना करें $ m $, निर्णय लें
सुपरक्रिटिकलिटी, और विलुप्त होने की \omterm{def:b2:multint:exact}{संभावना} की गणना करें $ q $
\omterm{def:b2:series:def}{बिल्कुल}। \emph{(मूल का गुणनखंड निकालें $ t = 1$ का $ G(t) - t $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:genfun:7}
(कुल संतान) एक उपक्रान्ति में \omterm{pb:b2:genfun:1}{शाखाकरण प्रक्रिया} ($ m < 1$), होने देना
$ Y = \sum_{n\geq0} Z_n $ अब तक व्यक्तियों की कुल संख्या हो
पैदा हुआ. दिखाओ $\E(Y) = \sum_n m^n = \frac{1}{1 - m}$ (उचित ठहराएं
योगों का आदान-प्रदान), और साबित करें कि \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन}
$ H = G_Y $ कार्यात्मक समीकरण को संतुष्ट करता है $ H(t) = t\,G(H(t))$.
\emph{(पूर्वज, साथ ही उसके प्रत्येक की कुल संतानें
बच्चे, जो हैं \omterm{def:b2:proba:independence}{स्वतंत्र} की प्रतियाँ $ Y $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:genfun:8}
होने देना $ X $ पास होना \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन} $ G $ साथ \omterm{def:b2:powerseries:radius}{अभिसरण की त्रिज्या} $>
1$. सिद्ध करो \emph{घातांकीय पूँछ बंधी हुई}: वहाँ हैं $ C > 0$ और
$\rho \in \intoo{0}{1}$ साथ $\P(X \geq n) \leq C\rho^n $.
\emph{(मार्कोव ने आवेदन किया $ t^X $ एक निश्चित के लिए $ t > 1$ के अंदर
डिस्क.)} इसके विपरीत, दिखाएँ कि यदि $\P(X \geq n) \leq C\rho^n $ साथ
$\rho < 1$, की त्रिज्या $ G $ है $\geq 1/\rho > 1$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:genfun:9}
(निरंतरता प्रमेय, प्रारंभिक मामला) चलो $ X, X_1, X_2, \dots $ होना
$\N $-के साथ मूल्यवान $ G_{X_n}(t) \to G_X(t)$ हरएक के लिए $ t \in
\intco{0}{1}$. बताते हैं कि $\P(X_n = k) \to \P(X = k)$ हरएक के लिए
$ k $. \emph{(प्रेरण चालू $ k $: के लिए $ k = 0$ लेना $ t \to 0$ ---
ध्यान से: ठीक करें $ t $ छोटा, उपयोग $\abs{\P(X_n = 0) - G_{X_n}(t)}
\leq \frac{t}{1-t}$, पूँछ से मान्य $\sum_{j \geq 1}p_jt^j
\leq \frac{t}{1 - t}$; फिर विकर्ण करें। प्रेरण चरण के लिए,
विचार करें $\frac{G(t) - \P(X = 0)}{t}$, द \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन} का
एक स्थानांतरित \omterm{def:b2:randomvar:law}{कानून}.)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:genfun:10}
(समता युक्ति) इसे एक के लिए दिखाएँ $\N $-मूल्यवान चर $ X $,
\[
\P(X \text{ even}) = \frac{1 + G_X(-1)}{2} ,
\]
और इस \omterm{def:b2:multint:exact}{संभावना} की गणना करें $ X \sim \mathcal P(\lambda)$
और $ X \sim \mathcal B(n, p)$. क्या करता है $ G_X(-1) \to 0$ मतलब
संभाव्यतापूर्वक?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:genfun:11}
(सिचेरमैन पासा) फेयर-डाई के गुणनखंडन को सत्यापित करें
\omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन}
\[
\frac{t + t^2 + \dots + t^6}{6}
= \frac{t\,(1 + t)(1 + t + t^2)(1 - t + t^2)}{6},
\]
और दिखाओ कि दो पासे फलक वाले हैं $\{1, 2, 2, 3, 3, 4\}$
और $\{1, 3, 4, 5, 6, 8\}$ पास होना कार्य उत्पन्न करना
$\frac{t(1+t)(1+t+t^2)}6$ और
$\frac{t(1+t)(1+t+t^2)(1-t+t^2)^2}6$, जिसका उत्पाद है
दो मानक पासे: ये विदेशी पासे कुल मिलाकर उत्पादन करते हैं $2,
\dots, 12$ \omterm{def:b2:series:def}{बिल्कुल} मानक संभावनाओं के साथ।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:genfun:12}
(एक पंक्ति में दो चितों की प्रतीक्षा में) चित \omterm{def:b2:multint:exact}{संभावना} वाला एक सिक्का
$ p $ लगातार दो सिर आने तक उछाला जाता है; होने देना $ T $ हो
टॉस की संख्या (खेल) \cref{exo:b2:proba:6}).
पहले टॉस पर कंडीशनिंग, के लिए एक रैखिक प्रणाली प्राप्त करें
द कार्य उत्पन्न करना राज्यों से ``कोई वर्तमान प्रमुख नहीं''
और ``एक वर्तमान प्रमुख'', और निष्कर्ष निकालें
\[
G_T(t) = \frac{p^2t^2}{1 - qt - pqt^2}
\qquad (q = 1 - p);
\]
जाँच करना $ G_T(1) = 1$ और $\E(T) = \dfrac{1 + p}{p^2}$ ($= 6$
एक उचित सिक्के के लिए)।
\end{exercise}

\section{समस्या: गैल्टन-वॉटसन प्रक्रिया, हल हो गई}

\begin{problem}[{सप्ताहांत समस्या --- विकास दर, सटीक
समाधान, कुल संतान, और कोलमोगोरोव की आलोचना
अनुमान}]\label{pb:b2:genfun:1}
विलुप्त होने की कसौटी (\cref{thm:b2:genfun:extinction})
शाखाकरण प्रक्रियाओं को विभाजित करता है\index{शाखाकरण प्रक्रिया} में
सबक्रिटिकल, क्रिटिकल और सुपरक्रिटिकल --- लेकिन यह कहता है
के बारे में कुछ नहीं \emph{दरें}: कितनी तेजी से एक बर्बाद रेखा मर जाती है, कैसे
जीवित रहने वाला बड़ा होता है। यह समस्या उनकी गणना करती है। हम रखते हैं
अध्याय का संकेत: संतान \omterm{def:b2:randomvar:law}{कानून} $(p_k)$ पीजीएफ के साथ $ G $,
अर्थ $ m = G'(1)$, पीढ़ी का आकार $ Z_n $ ($ Z_0 = 1$), पुनरावृत्त करता है
$ G_n = G_{Z_n}$, विलुप्त होने की संभावनाएँ $ q_n = \P(Z_n = 0)
\uparrow q $; हम हमेशा मान लेते हैं $ p_1 \neq 1$ और, दूसरा कहां
क्षण प्रकट होते हैं, $ G''(1) < \infty $, और हम लिखते हैं $\sigma^2 =
V(Z_1)$.

\medskip
\noindent\textbf{Part I --- Moments of the generations.}
\begin{enumerate}
  \item दिखाओ $\E(Z_n) = m^n $ \emph{(श्रृंखला नियम चालू $ G_n = G
        \circ G_{n-1}$ पर $1^-$, का उपयोग करना $ G_{n-1}(1) = 1$ और
        \cref{thm:b2:genfun:moments})}.
  \item प्रत्यावर्तन स्थापित करें $ G_n''(1) =
        G''(1)\,m^{2(n-1)} + m\,G_{n-1}''(1)$ और इसे हल करें:
        $ G_n''(1) = G''(1)\,m^{n-1}\dfrac{m^n - 1}{m - 1}$
        के लिए $ m \neq 1$, और $ G_n''(1) = n\,G''(1)$ के लिए $ m =
        1$.
  \item परिणाम निकालना
        \[
        V(Z_n) = \sigma^2m^{n-1}\,\frac{m^n - 1}{m - 1}
        \quad (m \neq 1),
        \qquad
        V(Z_n) = n\,\sigma^2 \quad (m = 1).
        \]
  \item (सबक्रिटिकल दर, ऊपरी सीमा) के लिए $ m < 1$, दिखाओ
        $\P(Z_n > 0) \leq m^n $ \emph{(मार्कोव पर
        पूर्णांक-मूल्यवान $ Z_n $)}: ए के साथ विलुप्ति निश्चित है
        ज्यामितीय दर --- का एक मात्रात्मक शोधन
        अध्याय की कसौटी.
  \item (सबक्रिटिकल दर, निचली सीमा) का उपयोग करना
        कॉची--श्वार्ज़ ऑन $ Z_n\mathbf 1_{Z_n > 0}$, दिखाओ
        \[
        \P(Z_n > 0) \geq \frac{\E(Z_n)^2}{\E(Z_n^2)}
        \geq c\,m^{n}
        \quad\text{with}\quad
        c = \Bigl(\frac{\sigma^2}{m(1-m)} + 1\Bigr)^{-1} :
        \]
        ज्यामितीय दर $ m^n $ स्थिरांक तक सटीक है।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The geometric family, solved
exactly.} संतान होने दो \omterm{def:b2:randomvar:law}{कानून} पर ज्यामितीय हो $\N $: $ p_k =
qp^k $ ($ k \geq 0$), साथ $0 < p < 1$, $ q = 1 - p $.
\begin{enumerate}[resume]
  \item गणना करना $ G(t) = \dfrac{q}{1 - pt}$ और $ m = \dfrac
        pq $; के संदर्भ में तीन व्यवस्थाओं का पता लगाएं $ p $.
  \item हल करना $ G(t) = t $:दिखाएँ कि निश्चित बिंदु हैं $1$ और
        $ q/p = 1/m $, और विलुप्त होने की \omterm{def:b2:multint:exact}{संभावना} को पुनः प्राप्त करें
        $ q_{\mathrm{ext}} = \min(1, 1/m)$.
  \item बंद प्रपत्रों को प्रेरण द्वारा सिद्ध करें
        \[
        q_n = \frac{m^n - 1}{m^{n+1} - 1} \quad (m \neq 1),
        \qquad
        q_n = \frac{n}{n+1} \quad (m = 1).
        \]
  \item सटीक दरें निकालें: $1 - q_n \sim (1 - m)\,m^n $
        सबक्रिटिकल मामले में, और $ q_{\mathrm{ext}} - q_n
        \sim \dfrac{m - 1}{m^{2}}\cdot m^{-n}$ में
        सुपरक्रिटिकल मामला; जांचें कि सुपरक्रिटिकल
        संकुचन अनुपात है $ G'(q_{\mathrm{ext}}) = 1/m $.
  \item गंभीर मामला ($ p = \tfrac12$): गणना करें $\sigma^2 =
        2$ और ध्यान दें $1 - q_n = \frac1{n+1}$: अस्तित्व समाप्त हो जाता है
        जैसे $\frac1n $ --- न तो ज्यामितीय और न ही \omterm{def:b2:series:summable}{संक्षेपणीय}.
  \item अभी भी महत्वपूर्ण: पूर्ण पुनरावृत्ति को प्रेरण द्वारा सिद्ध करें
        \[
        G_n(t) = \frac{n - (n-1)t}{n + 1 - nt},
        \]
        और जीवित रहने पर आधारित निष्कर्ष निकालें, $ Z_n $ है
        ज्यामितीय चालू $\N^*$ पैरामीटर के साथ $\frac1{n+1}$:
        \[
        \P(Z_n = k \mid Z_n > 0) = \frac1{n+1}
        \Bigl(\frac{n}{n+1}\Bigr)^{k-1},
        \qquad
        \E(Z_n \mid Z_n > 0) = n + 1 .
        \]
        औसत रेखा मर जाती है, लेकिन जीवित रेखाएँ मर जाती हैं
        ऑर्डर का आकार $ n $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Total progeny.} होने देना $ Y =
\sum_{n\geq0}Z_n \in \N^* \cup \{\infty\}$ कुल संख्या हो
अब तक जन्मे व्यक्तियों का, और $ H(t) =
\sum_{k\geq1}\P(Y = k)t^k $.
\begin{enumerate}[resume]
  \item औचित्य $\P(Y < \infty) = q_{\mathrm{ext}}$, और
        से याद करें \cref{exo:b2:genfun:7} कार्यात्मक
        समीकरण $ H(t) = t\,G(H(t))$ (जिसकी व्युत्पत्ति हुई
        उपयोग नहीं $ m < 1$).
  \item (बाइनरी ब्रांचिंग) के लिए $ p_0 = p_2 = \frac12$
        (महत्वपूर्ण), कार्यात्मक समीकरण हल करें:
        \[
        H(t) = \frac{1 - \sqrt{1 - t^2}}{t},
        \]
        और साथ विस्तार करें \cref{ex:b2:powerseries:catalan} पाने के
        \[
        \P(Y = 2k + 1) = \frac{C_k}{2^{2k+1}},
        \qquad C_k = \frac1{k+1}\binom{2k}k ;
        \]
        मूल्यों की जाँच करें $\P(Y = 1) = \frac12$ और $\P(Y = 3)
        = \frac18$ सीधी गिनती से.
  \item कार्यात्मक समीकरण को विभेदित करना $1^-$,
        बताते हैं कि $\E(Y) = \frac{1}{1-m}$ के लिए $ m < 1$, जबकि
        आलोचनात्मकता बल $\E(Y) = \infty $: आलोचनात्मक
        कुल संतान लगभग निश्चित रूप से अनंत के साथ सीमित है
        मतलब.
  \item केंद्रीय द्विपद स्पर्शोन्मुखता के साथ
        (\cref{ex:b2:comparison:centralbinomial}), दिखाओ
        \[
        \P(Y = 2k+1) \sim \frac{1}{2\sqrt\pi\,k^{3/2}},
        \]
        एक भारी $ k^{-3/2}$ पूँछ, और निष्कर्ष निकालें $\P(Y > n)
        \asymp n^{-1/2}$ (इसकी ऊपरी और निचली सीमाएं
        ऑर्डर पर्याप्त है)।
  \item मेले से तुलना करें यादृच्छिक चाल (सप्ताहान्त
        की समस्या \cref{ch:b2:proba}): निश्चित लेकिन
        अनंत-माध्य वापसी समय वहाँ, निश्चित लेकिन
        अनंत-माध्य कुल संतान यहाँ, दोनों के साथ
        $ n^{-3/2}$ स्थानीय \omterm{def:b2:randomvar:law}{कानून}. क्यों पर एक अनुच्छेद
        आलोचनात्मकता इस \omterm{thm:b2:structures:signature}{हस्ताक्षर} का निर्माण करती है।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Kolmogorov's estimate at
criticality.} मान लीजिए $ m = 1$, $0 < \sigma^2 = G''(1) <
\infty $.
\begin{enumerate}[resume]
  \item बताते हैं कि $ G''$ का विस्तार लगातार को $\intcc01$
        \emph{(परिमित सीमा के साथ गैर-नकारात्मक वृद्धि)} और
        टेलर विस्तार का अनुमान लगाएं $1$:
        \[
        G(t) = t + b\,(1-t)^2 + o\bigl((1-t)^2\bigr),
        \qquad b = \frac{G''(1)}2 = \frac{\sigma^2}2 .
        \]
  \item के लिए $ t \in \intco01$, तय करना $ h(t) = \dfrac1{1 - G(t)} -
        \dfrac1{1 - t}$. दिखाओ
        \[
        h(t) = \frac{G(t) - t}{(1 - G(t))(1 - t)}
        \xrightarrow[t\to1^-]{} b .
        \]
  \item पुनरावृत्ति के साथ टेलीस्कोप $ q_{j+1} = G(q_j)$:
        \[
        \frac1{1 - q_n} = 1 + \sum_{j=0}^{n-1}h(q_j),
        \]
        और सेसरो के तर्क के साथ समाप्त करें
        \[
        \P(Z_n > 0) = 1 - q_n \sim \frac{2}{\sigma^2\,n}
        \]
        --- \emph{कोलमोगोरोव का अनुमान}: हर आलोचनात्मक
        \omterm{pb:b2:genfun:1}{शाखाकरण प्रक्रिया} सार्वभौमिक दर पर मर जाता है $1/n $,
        केवल संतानों को \omterm{def:b2:metric:continuity}{निरंतर} याद रखने के साथ \omterm{def:b2:randomvar:law}{कानून}.
  \item क्रिटिकल ज्योमेट्रिक के विरुद्ध अनुमान की जाँच करें
        प्रश्न 10 का मामला.
  \item परिणाम निकालना $\E(Z_n \mid Z_n > 0) = \dfrac{1}{1 - q_n}
        \sim \dfrac{\sigma^2 n}{2}$ \emph{(टिप्पणी $\E(Z_n
        \mathbf 1_{Z_n>0}) = \E(Z_n) = 1$)}, और इसकी जांच करें
        प्रश्न 11 के विरुद्ध: जीवित रहने पर आधारित,
        जनसंख्या बढ़ती है \emph{रैखिक} --- आलोचनात्मक
        मौत और विस्फोट के बीच रस्सी।
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Applications and synthesis.}
\begin{enumerate}[resume]
  \item (महामारी, श्रृंखला प्रतिक्रियाएं) पॉइसन संतान के लिए
        \omterm{def:b2:randomvar:law}{कानून} $\mathcal P(\lambda)$ --- प्रत्येक मामला संक्रमित करता है
        $\mathcal P(\lambda)$ नए मामले---लिखें
        विलुप्ति समीकरण $ q = \eu^{\lambda(q-1)}$ और
        इसे संख्यात्मक रूप से हल करें $\lambda = 1.5$ ($ q \approx
        0.417$) और $\lambda = 2$ ($ q \approx 0.203$):
        एक मामले से शुरू होकर, एक बड़ा प्रकोप है
        \emph{नहीं} निश्चित तब भी जब $\lambda > 1$. समझाओ
        पुनरावृत्ति क्यों? $ q_{n+1} = \eu^{\lambda(q_n - 1)}$
        से $ q_0 = 0$ दाहिनी जड़ में एकत्रित हो जाता है।
  \item से शुरू $ k $ एक के बदले पूर्वज, वह दिखाओ
        विलुप्त होने की \omterm{def:b2:multint:exact}{संभावना} है $ q^k $. आवेदन:
        के साथ $\lambda = 1.5$, कितने प्रारंभिक मामले बनते हैं
        कम से कम प्रकोप $99\%$ संभावित?
  \item (विलुप्त होने पर एक सुपरक्रिटिकल प्रक्रिया की पुष्टि)
        के लिए $ m > 1$ विलुप्त होने की \omterm{def:b2:multint:exact}{संभावना} के साथ $ q \in
        \intoo01$: पहले उत्तलता से सिद्ध करें कि $ G'(q) < 1$
        सबसे छोटे निश्चित बिंदु पर, और निष्कर्ष निकालें
        $ q_{\mathrm{ext}} - q_n = O\bigl(G'(q)^n\bigr)$
        (ज्यामितीय \omterm{def:b2:integration:improper}{अभिसरण}, जैसा कि प्रश्न 9 में उदाहरण दिया गया है)।
        फिर वो दिखाओ $\widehat G(t) = G(qt)/q $ पीजीएफ है
        एक प्रामाणिक संतान का \omterm{def:b2:randomvar:law}{कानून}, मतलब के साथ $\widehat m =
        G'(q) < 1$: एक उप-महत्वपूर्ण साथी प्रक्रिया। सत्यापित करें
        ज्यामितीय परिवार पर: कंडीशनिंग
        सुपरक्रिटिकल $(p, q)$-विलुप्त होने की अदला-बदली पर प्रक्रिया
        $ p $ और $ q $. (\omterm{def:b2:metric:complete}{पूरा} बयान --- वातानुकूलित
        प्रक्रिया \emph{है} सहचर प्रक्रिया --- सिद्ध होती है
        वर्ष 3 खंड में; यहां आपने इसका सत्यापन कर लिया है
        जनरेटिंग-फ़ंक्शन छाया।)
  \item संश्लेषण: ट्राइकोटॉमी तालिका बनाएं --- के लिए $ m <
        1$, $ m = 1$, $ m > 1$: का मूल्य $ q $; का मूल्य $\P(Z_n >
        0)$ या का $ q - q_n $; $\E(Y)$; जीवित रहने का आकार
        पीढ़ी. प्रति उपकरण एक वाक्य में बताएं कि कैसे
        पीजीएफएस की संरचना, उत्तलता, टेलर एट $1^-$, और
        सिजेरो ने औसत रूप से पूरी समस्या को उठाया, और क्या
        वर्ष 3 खंड जोड़ता है (मार्टिंगेल)। $ Z_n/m^n$ और
        याग्लोम की घातीय सीमा \omterm{def:b2:randomvar:law}{कानून}).
\end{enumerate}
\end{problem}
