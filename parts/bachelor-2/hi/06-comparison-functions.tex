\chapter{कार्यों की तुलना}\label{ch:b2:comparison}

एसिम्प्टोटिक विश्लेषण --- एक जटिल मात्रा को बदलने की कला
एक साधारण प्लस एक नियंत्रित त्रुटि द्वारा --- वर्ष 1 में शुरू किया गया था
टेलर विस्तार के साथ वॉल्यूम। यह अध्याय इसे एक अनुशासन बनाता है
इसका अपना: सामान्य पैमानों पर विस्तार, श्रृंखला--अभिन्न
इसकी पूर्ण स्पर्शोन्मुख शक्ति के साथ तुलना, स्टर्लिंग का सूत्र (साबित)
पूरी तरह से), और अंतर्निहित रूप से परिभाषित का व्यवस्थित अध्ययन
क्रम. ये तकनीकें एसिम्प्टोटिक की रोजमर्रा की रोटी हैं
विश्लेषण, और प्रत्येक बाद का अध्याय जो कुछ भी अनुमान लगाता है ---
श्रृंखला, अभिन्न, संभावनाएँ --- इस तालिका से खाती है।

\section{तुलना संबंध और पैमाने}

\begin{definition}\label{def:b2:comparison:landau}
एक बिंदु के निकट $a $ ($ a \in \R $ या $\pm\infty $), फ़ंक्शंस के लिए (या
अनुक्रम, के साथ $ n \to \infty $): $ f = o(g)$, $ f = O(g)$, $ f \sim g $
जैसा कि वर्ष 1 खंड में है। ए \emph{तुलना
स्केल}\index{तुलना
स्केल} पर $ a $ सकारात्मक का परिवार है
फ़ंक्शन, जोड़ीवार तुलनीय, पूरी तरह से क्रमबद्ध $ o(\cdot)$ ---
मानक पैमाने पर $+\infty $ प्राणी
\[
x^{\alpha} (\ln x)^{\beta}
\qquad (\alpha, \beta \in \R),
\]
में शब्दकोषीय रूप से \omterm{def:b2:structures:generated}{आदेश} दिया गया $(\alpha, \beta)$, आवश्यकता पड़ने पर परिष्कृत किया जाता है
घातांक द्वारा $\eu^{\gamma x}$.
\end{definition}

\begin{definition}[स्पर्शोन्मुख विस्तार]\label{def:b2:comparison:expansion}
$ f $ मानता है \emph{स्पर्शोन्मुख विस्तार}\index{स्पर्शोन्मुख विस्तार}
\[
f = c_1 \varphi_1 + c_2\varphi_2 + \dots + c_k \varphi_k +
o(\varphi_k)
\qquad (\varphi_{i+1} = o(\varphi_i) \text{ in the scale})
\]
जब क्रमिक शेष प्रदर्शित अनुमानों को संतुष्ट करते हैं। द
गुणांक तब अद्वितीय होते हैं: $ c_1 = \lim f/\varphi_1$, और
आगमनात्मक रूप से $ c_{i+1} = \lim\,(f - \sum_{j \leq i}
c_j\varphi_j)/\varphi_{i+1}$.
\end{definition}

\begin{example}
टेलर विस्तार हैं \omterm{def:b2:comparison:expansion}{स्पर्शोन्मुख विस्तार} पैमाने के साथ $(x -
a)^k $ पर $ a $. लेकिन यह धारणा पूरी तरह से व्यापक है: पर $+\infty $,
\[
\frac{1}{x - \ln x}
= \frac1x \cdot \frac{1}{1 - \frac{\ln x}{x}}
= \frac1x + \frac{\ln x}{x^2} + o\Bigl(\frac{\ln x}{x^2}\Bigr),
\]
मिश्रित पैमाने पर विस्तार --- कोई टेलर प्रमेय लागू नहीं होता,
केवल ज्यामितीय विस्तार और गणना $ o $'एस।
\end{example}

\begin{example}[मानक पैमाने का वास्तव में आदेश दिया गया है]\label{ex:b2:comparison:scaleorder}
का शब्दकोषीय दावा \cref{def:b2:comparison:landau}
प्रति मामले में प्रमाण की एक पंक्ति की आवश्यकता होती है। तुलना करना
$ x^{\alpha}(\ln x)^{\beta}$ और $ x^{\alpha'}(\ln x)^{\beta'}$
पर $+\infty $. अगर $\alpha < \alpha'$: अनुपात है
$ x^{\alpha - \alpha'}(\ln x)^{\beta - \beta'} \to 0$, क्योंकि ए
की नकारात्मक शक्ति $ x $ की किसी भी शक्ति को कुचल देता है $\ln x $ (तय करना $ x =
\eu^t $: $\eu^{(\alpha - \alpha')t}\,t^{\beta - \beta'} \to 0$
वर्ष 1 की घातीय-बीट्स-बहुपद सीमा द्वारा
वॉल्यूम). अगर $\alpha = \alpha'$ और $\beta < \beta'$: अनुपात
है $(\ln x)^{\beta - \beta'} \to 0$ सीधे. तो जोड़े
$(\alpha, \beta)$, शब्दकोषीय ढंग से \omterm{def:b2:structures:generated}{आदेश} दिया, पैमाने का \omterm{def:b2:structures:generated}{आदेश} दिया
द्वारा $ o(\cdot)$ --- और प्रतिस्थापन $ x = \eu^t $ है
मिश्रित पावर-लॉग तुलनाओं के लिए एक-आकार-सभी के लिए उपयुक्त युक्ति।
\end{example}

\begin{example}[एक चिड़ियाघर की रैंकिंग]\label{ex:b2:comparison:ranking}
तराजू होना चाहिए \emph{\omterm{def:b2:structures:generated}{आदेश} दिया}; यहाँ मानक ड्रिल है. पर
$+\infty $, तुलना करना $ n^{10}$, $\eu^{\sqrt{\ln n}\,\cdot\,\sqrt n}$,
$2^n $ और $ n^{\ln n}$ लघुगणक लेकर:
\[
10\ln n
\;\ll\; (\ln n)^2
\;\ll\; \sqrt{n\ln n}
\;\ll\; n\ln 2 ,
\]
कहाँ $ a_n \ll b_n $ मतलब $ a_n = o(b_n)$; दूसरी प्रविष्टि है
$\ln(n^{\ln n})$. घातांक इन सख्त अंतरालों को संरक्षित करते हैं (यदि
$\ln u_n - \ln v_n \to -\infty $ तब $ u_n/v_n \to 0$), इसलिए
\[
n^{10} = o\bigl(n^{\ln n}\bigr),
\qquad
n^{\ln n} = o\bigl(\eu^{\sqrt{n\ln n}}\bigr),
\qquad
\eu^{\sqrt{n\ln n}} = o(2^n) .
\]
नैतिक, दो बार खत्म: हमेशा लघुगणक के माध्यम से तुलना करें
(लॉग के अंतर, लॉग के अनुपात नहीं), और कभी निष्कर्ष नहीं निकालते
$ u_n \sim v_n $ से $\ln u_n \sim \ln v_n $ --- जोड़ी $ n^{10}$
और $ n^{\ln n}$ है $\ln $-अनुपात की ओर रुझान $\infty $, लेकिन $2^n $
और $4^n $ पास होना $\ln $-अनुपात बिल्कुल $2$ और बेतहाशा हैं
असमान.
\end{example}

\section{शृंखला--अभिन्न तुलना, स्पर्शोन्मुख रूप से}

\begin{theorem}\label{thm:b2:comparison:seriesintegral}
होने देना $ f $ होना \omterm{def:b2:metric:continuity}{निरंतर}, सकारात्मक, \emph{घटते} पर
$\intco{1}{+\infty}$.
\begin{enumerate}
  \item अगर $\int_1^{\infty} f $ अभिसरण होता है, शेष संतुष्ट होते हैं
        \[
        \int_{n+1}^{\infty} f \;\leq\; \sum_{k > n} f(k) \;\leq\;
        \int_{n}^{\infty} f .
        \]
  \item अगर $\int_1^\infty f $ विचलन, आंशिक रकम संतुष्ट करती है
        $\sum_{k=1}^{n} f(k) = \int_1^n f + C + o(1)$ कुछ के लिए
        स्थिर $ C $: के अंतर $\sum_{k \leq n} f(k) -
        \int_1^n f $ \emph{अभिसरण}.
\end{enumerate}
\end{theorem}

\begin{proof}
ब्रैकेटिंग $ f(k+1) \leq \int_k^{k+1} f \leq f(k)$ (कमी) थी
वर्ष 1 डिवाइस; संक्षेप में $ k \geq n+1$, सम्मान।\ $ k \geq n $,
देता है (1). (2) के लिए, सेट करें $ u_k = f(k) - \int_k^{k+1} f $: द्वारा
ब्रैकेटिंग, $0 \leq u_k \leq f(k) - f(k+1)$, तो का आंशिक योग
$\sum u_k $ दूरबीन से बंधे हैं $ f(1) - f(n+1) \leq
f(1)$: शृंखला अभिसरित होती है। इसके अलावा क्रम $\bigl(\int_n^{n+1}
f\bigr)_n $ नहीं बढ़ रहा है ($ f $ घटती है) और गैर-नकारात्मक, इसलिए
अभिसारी। लिखना
\[
\sum_{k=1}^{n} f(k) - \int_1^n f
= \sum_{k=1}^{n} u_k + \int_n^{n+1} f ,
\]
दाहिनी ओर के रूप में अभिसरण होता है $ n \to \infty $: अंतर जुट जाता है
एक स्थिरांक के लिए $ C $, जो कथन (2) है।
\end{proof}

\begin{example}[हार्मोनिक विस्तार]\label{ex:b2:comparison:harmonic}
के लिए $ f(t) = \frac1t $: $ H_n = \ln n + \gamma + o(1)$, ठीक हो रहा है
क्लीनर प्रमाण के साथ यूलर स्थिरांक (वर्ष 1 आयतन)। एक को धक्का देना
आगे ऑर्डर करें (\cref{exo:b2:comparison:3}):
\[
H_n = \ln n + \gamma + \frac{1}{2n} + o\Bigl(\frac1n\Bigr).
\]
संख्याएँ लाभ को दृश्यमान बनाती हैं $ n = 10$: $ H_{10} =
2.928968\dots $ और $\ln 10 = 2.302585\dots $, तो कच्चा अनुमान
का $\gamma $ है $ H_{10} - \ln 10 = 0.626383$, द्वारा बंद $0.049$;
सुधार घटाना $\frac1{20}$ देता है $0.576383$,
से दूर $\gamma = 0.577216$ केवल द्वारा $8.3\cdot10^{-4}$ ---
जो स्वयं अगला पद है $\frac{1}{12\cdot100}$ का
विस्तार, जैसा कि सप्ताहांत की समस्या साबित होती है (प्रश्न 8)।
\end{example}

\begin{example}[एक कच्चा $\ln(n!)$ स्टर्लिंग के बिना]\label{ex:b2:comparison:lnfactorial}
ब्रैकेटिंग डिवाइस अकेले पहले से ही पता लगा लेता है $\ln(n!)$. तब से
$\ln $ बढ़ता है,
\[
\int_{k-1}^{k}\ln t\,\dd t \;\leq\; \ln k \;\leq\;
\int_{k}^{k+1}\ln t\,\dd t ,
\]
और संक्षेप में बताएं $ k = 2, \dots, n $ (साथ $\int_1^n\ln = n\ln n
- n + 1$):
\[
n\ln n - n + 1 \;\leq\; \ln(n!) \;\leq\; (n+1)\ln(n+1) - n .
\]
दोनों बाड़ हैं $ n\ln n - n + O(\ln n)$: इस तरह $\ln(n!) = n\ln
n - n + O(\ln n)$, खास तरीके से $\ln(n!) \sim n\ln n $.
स्टर्लिंग जो जोड़ता है वह अगले दो पायदान हैं --- $\frac12\ln n $
और स्थिरांक $\ln\sqrt{2\pi}$ ---जिसकी कीमत अधिक थी
की दूरबीन \cref{thm:b2:comparison:stirling}. जानना
प्रत्येक उपकरण जो परिशुद्धता खरीदता है वह असममित शिल्प का आधा है।
\end{example}

\begin{example}[रद्दीकरण विस्तार की मांग करता है]\label{ex:b2:comparison:cancellation}
की सीमा की गणना करें $\sqrt{n^2 + n} - n $. दोनों शर्तें हैं
$\sim n $, और ``$\sim n - n $'' अर्थहीन है: \omterm{def:b2:nvs:equivalent}{समकक्ष}
घटाया नहीं जा सकता. इसके बजाय विस्तार करें:
\[
\sqrt{n^2 + n} - n
= n\Bigl(\sqrt{1 + \tfrac1n} - 1\Bigr)
= n\Bigl(\frac{1}{2n} - \frac{1}{8n^2} +
O\Bigl(\frac1{n^3}\Bigr)\Bigr)
= \frac12 - \frac{1}{8n} + O\Bigl(\frac1{n^2}\Bigr) :
\]
आप LIMIT $\frac12$, दृष्टिकोण गति के साथ $\frac1{8n}$ एक के रूप में
बोनस. तंत्र एक नाम का हकदार है: दो बड़े का अंतर
समतुल्य मात्राएँ पूरी तरह से उन्हीं में रहती हैं \emph{अगला}
शर्तें, इसलिए किसी को पहले क्रम तक विस्तार करना चाहिए जिस पर दोनों
पक्ष अलग-अलग हैं --- और इसे प्रमाणित करने के लिए शेष को साथ रखें
उस क्रम में और कुछ भी नहीं बचता।
\end{example}

\begin{example}[एक भिन्न तुलना ने काम किया]\label{ex:b2:comparison:loglog}
के लिए $ f(t) = \frac{1}{t\ln t}$ पर $\intco{2}{+\infty}$
(\omterm{def:b2:metric:continuity}{निरंतर}, सकारात्मक, घटता हुआ): $\int_2^x f = \ln\ln x -
\ln\ln 2 \to \infty $, तो द्वारा
\cref{thm:b2:comparison:seriesintegral}~(2),
\[
\sum_{k=2}^{n}\frac{1}{k\ln k} = \ln\ln n + C + o(1)
\]
कुछ स्थिरांक के लिए $ C $. दो पाठ. सबसे पहले, विचलन है
वास्तविक लेकिन हिमनद: आंशिक योग पहले से अधिक है $4$ आस-पास $ n
\approx \eu^{\eu^{4 - C}}$, खगोलीय रूप से बड़ा। दूसरा,
\emph{आकार} $\ln\ln n $ एक प्रतिव्युत्पन्न द्वारा वितरित किया गया था, नहीं
अनुमान लगाया गया: मोनोटोन शब्दों के लिए, अभिन्न अंग विहित है
योग उपकरण, और स्थिरांक $ C $ --- यूलर की तरह $\gamma $
--- प्रारंभिक शब्दों की स्मृति है।
\end{example}

\section{स्टर्लिंग का सूत्र}

\begin{lemma}[वालिस इंटीग्रल्स, दोबारा गौर किया गया]\label{lem:b2:comparison:wallis}
होने देना $ W_n = \int_0^{\pi/2} \sin^n t\,\dd t $. तब $ nW_nW_{n-1} =
\frac\pi2$ के लिए $ n \geq 1$, $(W_n)$ घट जाती है, और $ W_n \sim
\sqrt{\dfrac{\pi}{2n}}$.
\end{lemma}

\begin{proof}
भागों द्वारा एकीकरण देता है $ nW_n = (n-1)W_{n-2}$ ($ n \geq 2$), इसलिए
$ nW_nW_{n-1}$ में स्थिर है $ n $, के बराबर $1 \cdot W_1 W_0 =
\frac\pi2$. घटाना: $\sin^{n+1} \leq \sin^n $ पर
$\intcc{0}{\frac\pi2}$. निचोड़, विस्तार से: एकरसता
देता है $ W_{n+1} \leq W_n \leq W_{n-1}$, और से विभाजित करना $ W_{n-1}
> 0$,
\[
\frac{n}{n+1} = \frac{W_{n+1}}{W_{n-1}} \leq
\frac{W_n}{W_{n-1}} \leq 1 ,
\]
सूचकांक पर पुनरावृत्ति से बाईं पहचान $ n + 1$. दोनों
सीमाएँ होती हैं $1$: $ W_n \sim W_{n-1}$, कहाँ से
\[
nW_n^2 \sim nW_nW_{n-1} = \frac\pi2
\qquad\Longrightarrow\qquad
W_n \sim \sqrt{\frac{\pi}{2n}} .
\]
\end{proof}

\begin{example}[पहला वालिस इंटीग्रल्स]\label{ex:b2:comparison:wallisvalues}
से $ W_0 = \frac\pi2$, $ W_1 = 1$ और पुनरावृत्ति $ nW_n =
(n-1)W_{n-2}$:
\[
W_2 = \frac\pi4, \qquad
W_3 = \frac23, \qquad
W_4 = \frac{3\pi}{16}, \qquad
W_5 = \frac{8}{15}, \qquad
W_6 = \frac{5\pi}{32}.
\]
यहां तक कि सूचकांक भी ले जाते हैं $\pi $, विषम तर्कसंगत हैं --- दो
बंद रूपों के इंटरलीव्ड उत्पाद। संख्यानुसार $ W_6
\approx 0.4909$ स्पर्शोन्मुख के विरुद्ध
$\sqrt{\pi/12} \approx 0.5116$: पर $ n = 6$ समतुल्य है
पहले से ही भीतर $5\%$, और उत्पाद की पहचान हर जगह सटीक है
$ n $: $6\,W_6W_5 = 6\cdot\frac{5\pi}{32}\cdot\frac8{15} =
\frac\pi2$. इस तरह की छोटी टेबलें सबसे सस्ता तरीका हैं
इससे पहले कि यह एक असममित तर्क को संक्रमित कर दे, बीजगणित की पर्ची पकड़ लें।
\end{example}

\begin{theorem}[स्टर्लिंग]\label{thm:b2:comparison:stirling}
\[
n! \;\sim\; \sqrt{2\pi n}\, \Bigl(\frac{n}{\eu}\Bigr)^{\!n}
.\index{स्टर्लिंग का सूत्र}
\]
\end{theorem}

\begin{proof}
\emph{स्टेप 1: $ n! \sim C \sqrt n\, (n/\eu)^n $ कुछ स्थिरांक के लिए $ C >
0$.} तय करना
\[
d_n = \ln(n!) - \Bigl(n + \frac12\Bigr)\ln n + n .
\]
तब
\[
d_n - d_{n+1}
= \Bigl(n + \frac12\Bigr) \ln\frac{n+1}{n} - 1
= \Bigl(n + \frac12\Bigr)\Bigl(\frac1n - \frac{1}{2n^2} +
\frac{1}{3n^3} + o\bigl(n^{-3}\bigr)\Bigr) - 1
= \frac{1}{12n^2} + o\Bigl(\frac{1}{n^2}\Bigr),
\]
के टेलर विस्तार द्वारा $\ln(1 + \frac1n)$. शृंखला $\sum (d_n
- d_{n+1})$ इस प्रकार पूर्ण रूप से अभिसरण होता है (तुलना के साथ)। $\sum
n^{-2}$), इसलिए $(d_n)$ जुटता है, कहता है $ d $; प्रतिपादक, $ n! \sim
C\sqrt n\,(n/\eu)^n $ साथ $ C = \eu^{d}$.

\emph{चरण दो: $ C = \sqrt{2\pi}$ वालिस के माध्यम से.} बंद प्रपत्र $ W_{2p}
= \frac{(2p)!}{4^p (p!)^2}\cdot\frac\pi2$ (पुनरावृत्ति से,
वर्ष 1 में गणना पुनः की गई \cref{lem:b2:comparison:wallis}का
सेटिंग) चरण 1 के साथ संयोजित होती है:
\[
W_{2p} \sim \frac{C\sqrt{2p}\,(2p/\eu)^{2p}}
{4^p\,\bigl(C\sqrt p\,(p/\eu)^p\bigr)^2}\cdot\frac{\pi}{2}
= \frac{\sqrt{2p}}{C\,p}\cdot\frac{\pi}{2}
= \frac{\pi}{C}\cdot\frac{1}{\sqrt{2p}} .
\]
के साथ तुलना करना $ W_{2p} \sim \sqrt{\frac{\pi}{4p}}$
(\cref{lem:b2:comparison:wallis}): $\frac{\pi}{C\sqrt{2p}} =
\sqrt{\frac{\pi}{4p}}\,(1 + o(1))$ बल $ C = \pi
\sqrt{\frac{4p}{2p\,\pi}} = \sqrt{2\pi}$.
\end{proof}

\begin{example}[केंद्रीय द्विपद गुणांक]\label{ex:b2:comparison:centralbinomial}
\[
\binom{2n}{n} = \frac{(2n)!}{(n!)^2}
\sim \frac{\sqrt{4\pi n}\,(2n/\eu)^{2n}}{2\pi n\,(n/\eu)^{2n}}
= \frac{4^n}{\sqrt{\pi n}} :
\]
संभावना है कि एक सममित यादृच्छिक चलना वापस आ जाता है $0$ समय पर
$2n $ है $\sim \frac{1}{\sqrt{\pi n}}$ ---की एक घोषणा
\cref{ch:b2:randomvar}.
\end{example}

\begin{remark}[इस खंड के भीतर परिप्रेक्ष्य]\label{rem:b2:comparison:perspectives}
आगे का प्रत्येक मात्रात्मक अध्याय इस अध्याय की बात कहता है
भाषा. \cref{ch:b2:series} तुलना करके श्रृंखला का वर्गीकरण करता है
पैमाने की शर्तें $ n^{-\alpha}(\ln n)^{-\beta}$ --- इसका
सप्ताहांत समस्या मानचित्र उस सीमा को पूरी तरह से दर्शाता है।
\cref{ch:b2:integration} अनुचित अभिन्नों के लिए भी यही करता है,
में समान पैमाने के साथ \omterm{def:b2:metric:continuity}{निरंतर} चर।
\cref{ch:b2:powerseries} से अभिसरण की त्रिज्या की गणना करता है
$\limsup\abs{a_n}^{1/n}$, एक समकक्ष-$ n $-वें-जड़ें
व्यायाम करें जहां स्टर्लिंग मानक कुंजी है ($\sqrt[n]{n!}
\sim \frac n\eu $, \cref{exo:b2:comparison:4}). और
संभाव्यता अध्याय नकद स्टर्लिंग सीधे: स्थानीय
का अनुमान \cref{ch:b2:randomvar} द्विपद गुणांक के लिए
हैं \cref{ex:b2:comparison:centralbinomial} और
\cref{ex:b2:comparison:lopsided} शब्दशः. एसिम्प्टोटिक्स नहीं है
यहाँ एक अध्याय; यह वॉल्यूम का उच्चारण है।
\end{remark}

\begin{method}[बूटस्ट्रैप चेकलिस्ट]\label{met:b2:comparison:checklist}
बूटस्ट्रैप्ड विस्तार पर भरोसा करने से पहले, चार बिंदुओं का ऑडिट करें।
(1) \emph{अस्तित्व पहले:} रूट या अनुक्रम को पिन किया जाना चाहिए
किसी भी विस्तार से पहले नीचे (एकरसता, मध्यवर्ती मूल्य)।
--- बिना सन्दर्भ वाले प्रतीक सुन्दरतापूर्वक विस्तार करते हैं और अर्थपूर्ण होते हैं
कुछ नहीं. (2) \emph{प्रति पास एक \omterm{def:b2:structures:generated}{आदेश}:} प्रत्येक प्रतिस्थापन हो सकता है
केवल दिए गए अनुमान के क्रम पर ही भरोसा किया जाए; निकालना
एक पास से दो नए शब्द गलत का क्लासिक स्रोत हैं
गुणांक. (3) \emph{शेष साथ चलते हैं:} ले जाओ
$ o(\cdot)$ प्रत्येक बीजगणितीय चरण के माध्यम से और अवशोषण करें
(छोटे शब्दों को बड़े अवशेषों द्वारा निगल लिया जाता है) पर होता है
अंत, स्पष्ट रूप से. (4) \emph{संख्यात्मक लेखापरीक्षा:} एक पर मूल्यांकन करें
का ईमानदार मूल्य $ n $; एक गुणांक त्रुटि बीजगणितीय बनी रहती है
पुनर्वत्पत्ति आश्चर्यजनक रूप से अक्सर होती है, और लगभग कभी भी जीवित नहीं रहती है
अंकगणित
\end{method}

\begin{remark}[सामान्य ख़तरे]\label{rem:b2:comparison:pitfalls}
(i)समतुल्य \emph{जोड़ना} बुरी तरह से: से $ u_n \sim n + \ln n $ और
$ v_n \sim -n $ कोई भी हो सकता है \emph{नहीं} निष्कर्ष $ u_n + v_n \sim \ln
n $; रद्दीकरण स्पष्ट अवशेषों के साथ विस्तार की मांग करता है,
\omterm{def:b2:nvs:equivalent}{समकक्ष}ों को कभी उजागर न करें। (ii) कभी भी तुल्यता का घातांक न लगाएं:
$ n + 1 \sim n $ लेकिन $\eu^{n+1} \not\sim \eu^n $; सुरक्षित
दिशा समतुल्य के लघुगणक ले रही है
$+\infty $ (इस अध्याय की सप्ताहांत समस्या, प्रश्न 24)। (iii)
एक \omterm{def:b2:comparison:expansion}{स्पर्शोन्मुख विस्तार} ए से जुड़ा हुआ है \emph{पैमाना}: लिखना
$ f = \frac1x + o\bigl(\frac1{x^2}\bigr)$ से अधिक का दावा करता है $ f =
\frac1x + o\bigl(\frac1x\bigr)$, और दोनों को मिलाने से अमान्य हो जाता है
बाद का बीजगणित. (iv) बूटस्ट्रैप में, स्थानापन्न करें
\emph{साबुत} वर्तमान विस्तार, शेष सम्मिलित --- गिरना
एक $ o(\cdot)$ मध्य-पास प्रशंसनीय लेकिन गलत पैदा करता है
गुणांक. (v) शृंखला--अभिन्न तुलना की आवश्यकता है
एकरसता: दोलनशील शब्दों के लिए यह पूरी तरह से विफल हो जाता है (तुलना करें)।
$\sum\frac{\sin k}k $, \cref{ch:b2:series}).
\end{remark}

\begin{example}[संख्या में स्टर्लिंग]\label{ex:b2:comparison:stirlingnum}
पर $ n = 10$: सूत्र देता है $\sqrt{20\pi}\,(10/\eu)^{10}
\approx 3\,598\,696$ ख़िलाफ़ $10! = 3\,628\,800$: सापेक्ष त्रुटि
$8.3\cdot10^{-3}$, एक ``एसिम्प्टोटिक'' कथन के लिए उल्लेखनीय
$ n = 10$. त्रुटि की एक संरचना होती है --- सटीक परिशोधन $ n!
= \sqrt{2\pi n}\,(n/\eu)^n\bigl(1 + \frac1{12n} +
O(n^{-2})\bigr)$ --- पहला सुधार किसका $\frac1{120} \approx
8.3\cdot10^{-3}$ देखे गए अंतर को लगभग सटीक रूप से समझाता है। द
सप्ताहांत की समस्या यूलर--मैकलॉरिन मशीनरी बिल्कुल सही है
ऐसे सुधार शब्दों का व्यवस्थित स्रोत।
\end{example}

\begin{remark}[जहाँ इस अध्याय का प्रयोग किया जाता है]
स्पर्शोन्मुख तुलना हर मात्रात्मक चीज़ का व्याकरण है
डाउनस्ट्रीम: अभिसरण परीक्षण और बर्ट्रेंड पैनोरमा
\cref{ch:b2:series}, की अभिन्नता मानदंड
\cref{ch:b2:integration}, अभिसरण की त्रिज्या गणना
का \cref{ch:b2:powerseries}, और की सीमा प्रमेय
\cref{ch:b2:randomvar} (जहां स्टर्लिंग डी चलाता है
मोइवर--लाप्लास अनुमान)। वर्ष 3 का खंड औद्योगिकीकरण करता है
एक विचार जिसे हम यहां हाथ से साबित करते हैं --- मुख्य शब्द निकालें, बाध्य
बाकी --- लाप्लास विधि और प्रभुत्व अभिसरण में।
\end{remark}

\begin{example}[स्वयं की तुलना में एक अभिन्न: $\int_2^x \frac{\dd t}{\ln t}$]\label{ex:b2:comparison:liworked}
तुलना टूलबॉक्स भी इंटीग्रल पर चलता है। होने देना $ F(x) =
\int_2^x\frac{\dd t}{\ln t}$ (इंटीग्रैंड है \omterm{def:b2:metric:continuity}{निरंतर} पर
$\intco2\infty $). भागों द्वारा एकीकृत करें:
\[
F(x) = \Bigl[\frac{t}{\ln t}\Bigr]_2^x +
\int_2^x\frac{\dd t}{(\ln t)^2}
= \frac{x}{\ln x} + O\Bigl(\int_2^x\frac{\dd t}{(\ln
t)^2}\Bigr) + O(1),
\]
और शेष अभिन्न है $ o\bigl(\frac{x}{\ln x}\bigr)$:
इसे विभाजित करें $\sqrt x $, से घिरा हुआ
\[
\int_2^{\sqrt x}\frac{\dd t}{(\ln t)^2} \leq \sqrt x
\qquad\text{and}\qquad
\int_{\sqrt x}^{x}\frac{\dd t}{(\ln t)^2} \leq
\frac{x}{(\ln\sqrt x)^2} = \frac{4x}{(\ln x)^2} .
\]
इस तरह $ F(x) \sim \frac{x}{\ln x}$. जो पाठक प्राइम से मिले
इस अध्याय की सप्ताहांत समस्या में संख्या प्रमेय को पहचानेंगे
$ F $: यह लघुगणकीय अभिन्न अंग है, इसका बेहतर अनुमानक है
$\pi(x)$, और गणना से पता चलता है कि यह इससे सहमत है
$\frac{x}{\ln x}$ पहले ऑर्डर करने के लिए.
\end{example}

\begin{example}[एक असंतुलित द्विपद पर स्टर्लिंग]\label{ex:b2:comparison:lopsided}
के लिए वही तीन-कारकीय दिनचर्या
\cref{ex:b2:comparison:centralbinomial} देता है, के लिए
$\binom{3n}{n} = \frac{(3n)!}{n!\,(2n)!}$:
\[
\binom{3n}{n} \sim
\frac{\sqrt{6\pi n}\,(3n/\eu)^{3n}}
{\sqrt{2\pi n}\,(n/\eu)^{n}\cdot\sqrt{4\pi n}\,(2n/\eu)^{2n}}
= \sqrt{\frac{3}{4\pi n}}\,
\Bigl(\frac{27}{4}\Bigr)^{\!n} .
\]
घातीय दर $\frac{27}4 = \frac{3^3}{2^2}$ है
$\eu^{3n\,H(1/3)}$ सूचना के एन्ट्रापी अंकन में
सिद्धांत: असंतुलित द्विपद केंद्रीय की तुलना में सख्ती से धीमी गति से बढ़ते हैं
$4^n $ प्रति दो चरण --- यहाँ $(27/4)^{1/3} \approx 1.89 < 2$ प्रति
कदम. कॉम्बिनेटरिक्स और संभाव्यता में प्रत्येक द्विपद स्पर्शोन्मुख
(\cref{ch:b2:randomvar}) क्या यह अलग के साथ एक गणना है
वजन.
\end{example}

\section{स्पष्ट रूप से परिभाषित अनुक्रम}

\begin{method}\label{met:b2:comparison:implicit}
समाधानों के स्पर्शोन्मुखता का पता लगाना $ x_n $ एक समीकरण का $ F(x, n) =
0$:\index{निहित अनुक्रम}
\begin{enumerate}
  \item \emph{स्थानीय बनाना}: के अस्तित्व और विशिष्टता को साबित करें $ x_n $ में
        एक निश्चित अंतराल (एकरसता, मध्यवर्ती मूल्य
        प्रमेय), और इसके अपरिष्कृत व्यवहार (सीमा, क्रम) का पता लगाएं
        विकास)
  \item \emph{बूटस्ट्रैप}: अपरिष्कृत रूप को प्रतिस्थापित करें $ x_n = (\text{main
        term})(1 + \varepsilon_n)$ समीकरण में डालें और हल करें
        का अगला \omterm{def:b2:structures:generated}{आदेश} $\varepsilon_n $; दोहराएँ, प्रत्येक पास
        एक \omterm{def:b2:structures:generated}{आदेश} को परिष्कृत करना।
\end{enumerate}
\end{method}

\begin{example}\label{ex:b2:comparison:tan}
के लिए $ n \geq 1$, समीकरण $\tan x = x $ बिल्कुल एक ही समाधान है
$ x_n $ में $\intoo{n\pi - \frac\pi2}{n\pi + \frac\pi2}$ (फ़ंक्शन
$\tan x - x $ से बढ़ता है $-\infty $ को $+\infty $ वहाँ, यह
व्युत्पन्न अस्तित्व $\tan^2 x \geq 0$). \emph{अपरिष्कृत:} $ x_n = n\pi +
\frac\pi2 - y_n $ साथ $ y_n \in \intoo{0}{\pi}$; तब से $ x_n \to
\infty $ और $\tan x_n = x_n \to +\infty $, $ x_n $ के पास पहुंचता है
बाईं ओर से स्पर्शोन्मुख: $ y_n \to 0$. \emph{बूटस्ट्रैप:} $\tan x_n =
\cot y_n = \frac{1}{\tan y_n} \sim \frac{1}{y_n}$, और समीकरण
$\cot y_n = x_n \sim n\pi $ देता है $ y_n \sim \frac{1}{n\pi}$. इस तरह
\[
x_n = n\pi + \frac\pi2 - \frac{1}{n\pi} +
o\Bigl(\frac1n\Bigr),
\]
और यह प्रक्रिया किसी भी क्रम में जारी रहती है
(\cref{exo:b2:comparison:6}).
\end{example}

\begin{example}[विधि का दूसरा भाग]\label{ex:b2:comparison:xplusln}
हल करना $ x + \ln x = n $ स्पर्शोन्मुख रूप से। \emph{स्थानीयकरण:} $ x
\mapsto x + \ln x $ से बढ़ता है $-\infty $ को $+\infty $ पर
$\intoo{0}{+\infty}$: एक अनोखी जड़ $ x_n $, और $ x_n \to \infty $.
\emph{अपरिष्कृत:} $\ln x_n = o(x_n)$ देता है $ x_n \sim n $.
\emph{बूटस्ट्रैप:} से $ x_n = n - \ln x_n $ और $\ln x_n = \ln n
+ o(1)$ (\omterm{def:b2:nvs:equivalent}{समकक्ष}ों के लघुगणक, दोनों पक्ष $\to \infty $):
\[
x_n = n - \ln n + o(1) ;
\]
एक और पास, साथ में $\ln x_n = \ln\bigl(n - \ln n + o(1)\bigr) =
\ln n - \frac{\ln n}{n} + o\bigl(\frac{\ln n}n\bigr)$:
\[
x_n = n - \ln n + \frac{\ln n}{n} +
o\Bigl(\frac{\ln n}{n}\Bigr).
\]
(पर जांचें $ n = 100$: जड़ है $ x \approx 95.4415$; द
तीन अवधि का फार्मूला देता है $100 - 4.6052 + 0.0461 = 95.4409$, द
दो-अवधि वाला $95.3948$ --- प्रत्येक पास पूर्वानुमानित लाभ प्राप्त करता है
क्रम।) वही लूप, तीसरा परिदृश्य: की विधि
\cref{met:b2:comparison:implicit} परवाह नहीं क्या
समीकरण ऐसा दिखता है, केवल यह कि प्रत्येक पास प्रमुख को अलग करता है
अज्ञात.
\end{example}

दो शब्दों के लिए सटीक बनाया गया)।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:comparison:5}
के लिए $ n \geq 2$, साबित करें कि $ x^n + x = 1$ एक अनोखा समाधान है $ x_n
\in \intoo{0}{1}$, वह $ x_n \to 1$, और स्थापित करें
\[
x_n = 1 - \frac{\ln n}{n} + o\Bigl(\frac{\ln n}{n}\Bigr).
\]
\emph{(से $ x_n^n = 1 - x_n $: लघुगणक और बूटस्ट्रैप साथ लें
$ x_n = 1 - \varepsilon_n $.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:comparison:6}
धकेलना \cref{ex:b2:comparison:tan} एक \omterm{def:b2:structures:generated}{आदेश} आगे:
\[
x_n = n\pi + \frac\pi2 - \frac{1}{n\pi} + \frac{1}{2n^2\pi} +
o\Bigl(\frac{1}{n^2}\Bigr).
\]
\emph{(लिखना $\cot y_n = x_n $ बिल्कुल, विस्तार करें $\cot y = \frac1y -
\frac y3 + o(y)$ और $ x_n = n\pi(1 + \frac{1}{2n} - \dots)$, और
पहचानें।)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:comparison:7}
ठानना $\lim_{n\to\infty} \dfrac{1}{n!}\sum_{k=0}^{n} k!$
\emph{(अंतिम दो को छोड़कर सभी पदों के योग को सीमित करें)}, और निष्कर्ष निकालें
\omterm{def:b2:comparison:expansion}{स्पर्शोन्मुख विस्तार} $\sum_{k \leq n} k! = n!\bigl(1 + \frac1n +
O(n^{-2})\bigr)$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:comparison:8}
होने देना $ u_0 > 0$ और $ u_{n+1} = u_n + \dfrac{1}{u_n}$. साबित करें कि $ u_n
\to \infty $, फिर उस $ u_n \sim \sqrt{2n}$ \emph{(अध्ययन $ u_n^2$:
इसकी वृद्धि है $2 + u_n^{-2}$; जोड़)}, और परिष्कृत करें:
\[
u_n = \sqrt{2n}\Bigl(1 + \frac{\ln n}{8n} +
o\Bigl(\frac{\ln n}{n}\Bigr)\Bigr).
\]
\emph{(से $ u_n^2 = 2n + \sum_{k<n} u_k^{-2} + u_0^2$ और $ u_k^2
\sim 2k $: योग है $\sim \frac12\ln n $ द्वारा
\cref{thm:b2:comparison:seriesintegral}.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:comparison:9}
(एक मोड़ के साथ एक रीमैन योग) के स्पर्शोन्मुख व्यवहार का निर्धारण करें
\[
S_n = \sum_{k=1}^{n} \frac{1}{n + k\ln n} .
\]
\emph{(कारक $ n $: $ S_n = \frac1n\sum_k \bigl(1 +
\frac{k\ln n}{n}\bigr)^{-1}$; रीमैन-प्रकार के योग को a से पहचानें
धीरे-धीरे बदलता पैरामीटर $ t = \ln n $, गणना करें $\int_0^1
\frac{\dd u}{1 + tu} = \frac{\ln(1+t)}{t}$, और निष्कर्ष निकालें $ S_n \sim
\frac{\ln\ln n}{\ln n}$.)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:comparison:10}
पहचान साबित करें $(\ln n)^{\ln n} = n^{\ln\ln n}$, फिर रैंक करें
बढ़ते हुए अनुसरण कर रहे हैं $ o(\cdot)$ अनंत पर \omterm{def:b2:structures:generated}{आदेश}, साथ
प्रमाण: $ n^2$, $(\ln n)^{\ln n}$, $2^n $, $ n!$, $ n^n $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:comparison:11}
(की पूँछ $\sum 1/k^2$, दो शब्द) सटीक टेलीस्कोपिंग का उपयोग करना
$\sum_{k > n} \frac{1}{k(k+1)} = \frac{1}{n+1}$ और
विघटन $\frac1{k^2} = \frac{1}{k(k+1)} +
\frac{1}{k^2(k+1)}$, सिद्ध करना
\[
\sum_{k > n} \frac{1}{k^2}
= \frac1n - \frac{1}{2n^2} + O\Bigl(\frac{1}{n^3}\Bigr).
\]
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:comparison:12}
होने देना $ u_0 = \frac12$ और $ u_{n+1} = u_n + \eu^{-u_n}$. साबित करें कि
$ u_n \to \infty $, फिर --- सेटिंग $ v_n = \eu^{u_n}$ और दिखा रहा हूँ
$ v_{n+1} = v_n + 1 + \frac{1}{2v_n} + O\bigl(v_n^{-2}\bigr)$ ---
स्थापित करना
\[
u_n = \ln n + \frac{\ln n}{2n} + O\Bigl(\frac1n\Bigr).
\]
\end{exercise}

\section{समस्या: बूटस्ट्रैपिंग, यूलर--मैकलॉरिन से
अभाज्य}

एक अंतर्निहित या संचित मात्रा शायद ही कभी इसे सौंपती है
एक बार में स्पर्शोन्मुखता; कोई उन्हें पासों में निकालता है, प्रत्येक पास में
पिछले अनुमान को परिभाषित संबंध में वापस फीड करना।
इस सप्ताहांत की समस्या उस लूप को ताज़ा समीकरणों पर प्रशिक्षित करती है, यह साबित होता है
प्रथम क्रम \emph{यूलर - मैकलॉरिन फार्मूला} (ट्रेपेज़ॉइड
श्रृंखला का उन्नयन--अभिन्न तुलना, कठोर त्रुटि के साथ
बार), उलटा $ x\ln x = n $, और कैश विधि सबसे प्रसिद्ध है
जाँच: स्वीकृत अभाज्य संख्या प्रमेय से, स्पर्शोन्मुख
कानून $ p_n \sim n\ln n $ की $ n $-वें प्रधान.

\begin{problem}[{सप्ताहांत समस्या---यूलर--मैकलॉरिन
सुधार और स्पर्शोन्मुखता $ n $-वाँ अभाज्य}]
\label{pb:b2:comparison:1}

\medskip
\noindent\textbf{Part I --- The bootstrap loop on a fresh
equation.}
\begin{enumerate}
  \item की विशिष्टता का दावा सिद्ध करें
        \cref{def:b2:comparison:expansion}: अगर $ f = \sum_{i\leq
        k} c_i\varphi_i + o(\varphi_k) = \sum_{i \leq k}
        c_i'\varphi_i + o(\varphi_k)$ फिर, उसी पैमाने पर
        $ c_i = c_i'$ सभी के लिए $ i $. फिर पाठ्यक्रम को मिश्रित करें
        उदाहरण एक पायदान आगे:
        \[
        \frac{1}{x - \ln x} = \frac1x + \frac{\ln x}{x^2} +
        \frac{(\ln x)^2}{x^3} + o\Bigl(\frac{(\ln
        x)^2}{x^3}\Bigr) \qquad (x \to +\infty),
        \]
        और बताएं कि कोई शब्द क्यों नहीं $\frac{c}{x^2}$ प्रकट होता है।
  \item इसे हर एक के लिए दिखाएँ $ n \geq 1$ समीकरण $\eu^x + x =
        n $ बिल्कुल एक वास्तविक समाधान है $ x_n $, ओर वो $ x_n
        \to +\infty $ साथ $ x_n \sim \ln n $.
  \item बूटस्ट्रैप दो बार:
        \[
        x_n = \ln n - \frac{\ln n}{n} - \frac{(\ln n)^2}{2n^2} +
        o\Bigl(\frac{(\ln n)^2}{n^2}\Bigr).
        \]
  \item संख्यात्मक रूप से जाँच करें $ n = 1000$: तुलना करना $ x_{1000}
        \approx 6.90083$ एक-, दो- और तीन-अवधि के साथ
        प्रश्न 3 का मान, पाँच दशमलव तक।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Euler--Maclaurin, order one.}
\begin{enumerate}[resume]
  \item ट्रैपेज़ॉइड कर्नेल पहचान साबित करें: के लिए $ g $ कक्षा का
        $ C^2$ पर $\intcc{0}{1}$,
        \[
        \int_0^1 g(t)\,\dd t = \frac{g(0) + g(1)}{2}
        - \frac12\int_0^1 t(1 - t)\,g''(t)\,\dd t
        \]
        \emph{(एकीकृत करें $\frac12 t(1-t)g''$ भागों द्वारा दो बार)}.
  \item होने देना $ f $ होना $ C^2$ पर $\intco{1}{+\infty}$ साथ
        $\int_1^\infty \abs{f''} < \infty $. बताते हैं कि
        \[
        E_n = \sum_{k=1}^{n} f(k) - \int_1^n f -
        \frac{f(1) + f(n)}{2}
        \]
        एक स्थिरांक में परिवर्तित हो जाता है $ E $, पूँछ बंधी हुई
        $\abs{E - E_n} \leq \frac18\int_n^\infty\abs{f''}$: द
        \emph{यूलर - मैकलॉरिन फार्मूला} पहले ऑर्डर करने के लिए.
  \item इसे लागू करें $ f(t) = \frac1t $: सिद्ध करना
        \[
        H_n = \ln n + \gamma + \frac{1}{2n} + \varepsilon_n,
        \qquad \abs{\varepsilon_n} \leq \frac{1}{8n^2},
        \]
        को सुदृढ़ \cref{exo:b2:comparison:3} (पहचानें
        के साथ \omterm{def:b2:metric:continuity}{निरंतर} $\gamma $ तुलना करके
        \cref{ex:b2:comparison:harmonic}).
  \item अगला गुणांक निकालें: दिखाएँ $\varepsilon_n =
        -\frac{1}{12n^2} + o\bigl(\frac1{n^2}\bigr)$
        \emph{(की वृद्धि) $ E_n $ हैं
        $\frac12\int_0^1t(1-t)f''(n+t)\dd t =
        \frac1{12}f''(n) + o(f''(n))$; पूँछ का योग करें
        \cref{thm:b2:comparison:seriesintegral})}.
  \item प्रश्न 6 को लागू करें $ f = \ln $: तीन पंक्तियों में पुनः प्राप्त करें
        का अभिसरण $ d_n = \ln n! - (n +
        \frac12)\ln n + n $ (चरण 1)
        \cref{thm:b2:comparison:stirling}), बोनस त्रुटि के साथ
        दर $ d_n = d + O\bigl(\frac1n\bigr)$.
  \item प्रश्न 6 को लागू करें $ f(t) = \frac{1}{\sqrt t}$: दिखाओ
        \[
        \sum_{k=1}^{n}\frac1{\sqrt k} = 2\sqrt n + c +
        \frac{1}{2\sqrt n} + O\Bigl(\frac{1}{n^{3/2}}\Bigr)
        \]
        कुछ स्थिरांक के लिए $ c $, और सभी शर्तों का मूल्यांकन करें $ n =
        10^4$ (स्थिरांक है $ c \approx -1.4604$).
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Inversion: the equation $ x\ln x =
n $.}
\begin{enumerate}[resume]
  \item बताते हैं कि $ x\ln x = n $ बिल्कुल एक ही समाधान है $ x_n \in
        \intco{1}{+\infty}$ के लिए $ n \geq 1$, वह $ x_n \to
        \infty $, ओर वो $\ln x_n \sim \ln n $.
  \item एक-अवधि का व्युत्क्रम निकालें $ x_n \sim
        \dfrac{n}{\ln n}$, फिर एक बार फिर बूटस्ट्रैप करें:
        \[
        \ln x_n = \ln n - \ln\ln n + o(1),
        \qquad
        x_n = \frac{n}{\ln n}\Bigl(1 + \frac{\ln\ln n}{\ln n}
        + o\Bigl(\frac{\ln\ln n}{\ln n}\Bigr)\Bigr).
        \]
  \item पर परीक्षण करें $ n = 10^6$:सच्ची जड़ है $ x \approx
        87\,848$; एक-अवधि के साथ तुलना करें ($\approx 72\,382$)
        और दो-अवधि ($\approx 86\,140$) मान, और व्याख्या करें
        धीमा लाभ (विस्तार पैरामीटर है $\frac{\ln\ln
        n}{\ln n}$, केवल $\approx 0.19$ पर $ n = 10^6$).
  \item हम अब \emph{स्वीकार करते हैं} अभाज्य संख्या प्रमेय: संख्या
        $\pi(x)$ अभाज्य संख्याओं का $\leq x $ संतुष्ट $\pi(x) \sim
        \frac{x}{\ln x}$ जैसा $ x \to \infty $ (ईमानदारी से साबित हुआ
        वर्ष 3 खंड)। लिखना $ p_n $ के लिए $ n $-वें प्रधान,
        उचित ठहराना $\pi(p_n) = n $, और का उलटा चलाएँ
        प्रश्न 11-12 सिद्ध करने के लिए
        \[
        p_n \sim n \ln n .
        \]
  \item लाभांश: (ए) दिखाएँ $\sum_{k \leq n} p_k \sim
        \frac{n^2\ln n}{2}$ \emph{(तुलना करना $\sum k\ln k $ साथ
        $\int t\ln t\,\dd t $)}; (बी) अनुमानित गणना करें
        संभावना है कि एक समान रूप से यादृच्छिक पूर्णांक $100$
        अंक अभाज्य है ($\ln 10^{100} \approx 230.26$: के बारे में
        एक में $230$).
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The method exported: $ x\tan x =
1$.}
\begin{enumerate}[resume]
  \item इसे प्रत्येक के लिए दिखाएँ $ n \geq 1$ समीकरण $\tan x =
        \frac1x $ बिल्कुल एक ही समाधान है $ x_n $ में
        $\intoo{n\pi}{\,n\pi + \frac\pi2}$, ओर वो $ z_n = x_n
        - n\pi \to 0^+$.
  \item एक कार्यकाल: $ z_n \sim \dfrac{1}{n\pi}$.
  \item दिखाएँ कि का विस्तार $ z_n $ है \emph{नहीं}
        $\frac{c}{n^2}$ अवधि: $ z_n = \frac1{n\pi} +
        O\bigl(\frac{1}{n^3}\bigr)$.
  \item तीन पद: उपयोग करना $\arctan u = u - \frac{u^3}3 +
        O(u^5)$ और $\frac1{x_n} = \frac{1}{n\pi} -
        \frac{z_n}{(n\pi)^2} + O(n^{-3}\cdot z_n^2)$, सिद्ध करना
        \[
        x_n = n\pi + \frac{1}{n\pi} -
        \frac{4}{3\pi^3 n^3} + o\Bigl(\frac{1}{n^3}\Bigr).
        \]
  \item पर जांचें $ n = 3$: सच्ची जड़ $ x_3 \approx 9.5293344$;
        एक- और तीन-अवधि के मानों की तुलना करें, और इसमें अंतर करें
        पाठ्यक्रम के साथ एक वाक्य $\tan x = x $
        (\cref{ex:b2:comparison:tan}): जहां प्रत्येक अनुक्रम बैठता है
        इसकी खिड़की में, और क्यों।
\end{enumerate}

\medskip
\noindent\textbf{Part V --- A dynamical bootstrap, rules of the
game, synthesis.}
\begin{enumerate}[resume]
  \item होने देना $ u_0 \in \intoo{0}{\pi}$ और $ u_{n+1} = \sin u_n $.
        दिखाओ $ u_n \to 0$ घटते हुए, और की सीमा की गणना करें
        $\dfrac{1}{u_{n+1}^2} - \dfrac{1}{u_n^2}$ \emph{(बढ़ाना
        $\sin^{-2}$ के जरिए $\sin u = u - \frac{u^3}6 + o(u^3)$)}.
  \item सेस के माध्यम से निष्कर्ष निकालें\`एआरओ का अर्थ है (वर्ष 1 खंड), क्लासिक
        \[
        u_n \sim \sqrt{\frac{3}{n}} .
        \]
  \item (प्रमाणित संख्याएँ) प्रश्न 7 की कठोर सीमा का उपयोग करते हुए,
        वह मूल्यांकन करके दिखाओ $\ln n + \gamma + \frac1{2n}$ पर $ n
        = 10^6$ पैदावार $ H_{10^6}$ अधिकतम त्रुटि के साथ
        $1.25\cdot10^{-13}$ --- एक मिलियन-टर्म राशि की गणना की गई
        तीन पदों द्वारा तेरह अंक।
  \item (खेल के नियम) प्रमाण सहित सिद्ध या खंडन करें
        प्रतिउदाहरण: (ए) यदि $ u_n \sim v_n \to +\infty $ तब
        $\ln u_n \sim \ln v_n $; (बी) यदि $ u_n \sim v_n $ तब
        $\eu^{u_n} \sim \eu^{v_n}$; (सी) यदि $ f \sim g $ पर
        $+\infty $ ($ f, g $ भिन्न) फिर $ f' \sim g'$.
  \item (संश्लेषण) प्रत्येक एक वाक्य में: बूटस्ट्रैप लूप
        \cref{met:b2:comparison:implicit} जैसा कि भाग I में उपयोग किया गया है,
        तृतीय, चतुर्थ; ट्रैपेज़ॉइड सुधार क्या जोड़ता है
        \cref{thm:b2:comparison:seriesintegral}; उलटा क्यों
        का $ x\ln x $ बिल्कुल पुल से है $\pi(x)$ को
        $ p_n $; और प्रश्न 24 के कौन से नियम किसकी रक्षा करते हैं
        कदम. दो शिखरों के नाम बताएं: यूलर-मैकलॉरिन फॉर्मूला
        (प्रथम क्रम), और का स्पर्शोन्मुख नियम $ n$-वें
        प्रधान.
\end{enumerate}
\end{problem}
