\chapter{बिजली की श्रृंखला}\label{ch:b2:powerseries}

पावर श्रृंखला कार्यों की सर्वोत्तम व्यवहार वाली श्रृंखला है
गणित: \omterm{def:b2:integration:improper}{अभिसरण} की अपनी डिस्क के अंदर वे \omterm{def:b2:integration:improper}{अभिसरण} करते हैं
\omterm{def:b2:funcseq:series}{सामान्यतः} पर \omterm{def:b2:metric:compact}{कॉम्पैक्ट} कैमरों, द्वारा विभेदित और एकीकृत किया जा सकता है
बिना दोबारा सोचे शब्द, और उनका योग --- \emph{\omterm{def:b2:powerseries:analytic}{विश्लेषणात्मक}
कार्य} --- उनके गुणांकों द्वारा निर्धारित किया जाता है। यह अध्याय
उस पूरे पैकेज को साबित करता है और ईमानदारी से, प्रत्येक टेलर को पुनः प्राप्त करता है
वर्ष की श्रृंखला 1 खंड; कार्य उत्पन्न करना इसके साथ बंद करें
बीजगणितीय लाभांश.

\section{अभिसरण की त्रिज्या}

\begin{lemma}[हाबिल]\label{lem:b2:powerseries:abel}
यदि क्रम $(a_n z_0^n)$ कुछ के लिए बाध्य है $ z_0 \neq 0$,
तब $\sum a_n z^n $ \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:series:def}{बिल्कुल} हरएक के लिए $\abs z < \abs
{z_0}$, और \omterm{def:b2:funcseq:series}{सामान्यतः} प्रत्येक डिस्क पर $\abs z \leq r < \abs{z_0}$.
\end{lemma}

\begin{proof}
साथ $\abs{a_n z_0^n} \leq M $ और $\abs z \leq r $:
\[
\abs{a_n z^n} = \abs{a_n z_0^n}\,\Bigl|\frac{z}{z_0}\Bigr|^n
\leq M\Bigl(\frac{r}{\abs{z_0}}\Bigr)^{\!n},
\]
एक अभिसारी ज्यामितीय सीमा, डिस्क पर एकसमान।
\end{proof}

\begin{definition}[अभिसरण की त्रिज्या]\label{def:b2:powerseries:radius}
 \emph{अभिसरण की त्रिज्या}\index{अभिसरण की त्रिज्या} का
$\sum a_n z^n $ है
\[
R = \sup\{r \geq 0 : (a_n r^n) \text{ bounded}\} \in
\intcc{0}{+\infty} .
\]
द्वारा \cref{lem:b2:powerseries:abel}: के लिए पूर्ण \omterm{def:b2:integration:improper}{अभिसरण} $\abs
z < R $ (सामान्य पर \omterm{def:b2:metric:compact}{कॉम्पैक्ट} सबडिस्क), विचलन --- वास्तव में असीमित
शर्तें --- के लिए $\abs z > R $. सीमा \omterm{def:b2:structures:sn}{चक्र} पर कुछ भी हो सकता है
घटित (\cref{exo:b2:powerseries:2}). व्यवहार में $ R $ गणना की जाती है
डी'एलेम्बर्ट के अनुपात परीक्षण द्वारा $\abs{a_n}\abs z^n $ या तुलना करके.
\end{definition}

\begin{example}[बिना किसी अनुपात परीक्षण वाला त्रिज्या]\label{ex:b2:powerseries:sinn}
की त्रिज्या क्या है $\sum \sin(n)\,z^n $? अनुपात
$\abs{\sin(n+1)/\sin n}$ इसकी कोई सीमा नहीं है, लेकिन परिभाषा काम करती है
सीधे. \emph{$ R \geq 1$:} $\abs{\sin n} \leq 1$, इसलिए $(\sin
n\cdot r^n)$ प्रत्येक के लिए बाध्य है $ r < 1$ --- वास्तव में के लिए $ r =
1$. \emph{$ R \leq 1$:} इतना ही काफी है $\sin n \not\to 0$.
कल्पना करना $\sin n \to 0$; जोड़ सूत्र
\[
\sin(n+1) = \sin n\cos 1 + \cos n\sin 1
\]
जबरदस्ती करेगा $\cos n \to 0$ (के लिए हल करें) $\cos n $: $\sin 1 \neq
0$), विरोधाभासी $\sin^2 n + \cos^2 n = 1$. तो शर्तें
$\sin(n)\,1^n $ प्रवृत्ति मत करो $0$: श्रृंखला अलग हो जाती है $ z =
1$, और $ R \leq 1$. निष्कर्ष: $ R = 1$. समापन अंतर्दृष्टि:
त्रिज्या के बारे में एक कथन है $\abs{a_n}r^n $ बाध्य किया जा रहा है ---
अनुपातों की किसी सीमा की कभी आवश्यकता नहीं होती, और सीमाबद्धता तर्क
उन मामलों का निपटारा करें जिन्हें अनुपात परीक्षण छू नहीं सकता (तुलना करें)।
के दोलन गुणांक \cref{exo:b2:powerseries:1}).
\end{example}

\begin{proposition}[संचालन]\label{prop:b2:powerseries:operations}
होने देना $\sum a_nz^n $, $\sum b_nz^n $ त्रिज्या है $ R_a, R_b $. फिर, के लिए
$\abs z < \min(R_a, R_b)$:
\[
\sum (a_n + b_n)z^n = \sum a_nz^n + \sum b_nz^n,
\qquad
\Bigl(\sum a_nz^n\Bigr)\Bigl(\sum b_nz^n\Bigr) = \sum c_n z^n,
\quad c_n = \sum_{k=0}^{n} a_kb_{n-k},
\]
दोनों श्रृंखलाओं में त्रिज्या है $\geq \min(R_a, R_b)$. (उत्पाद है
द \omterm{thm:b2:series:fubini}{कॉची उत्पाद}, पूर्ण \omterm{def:b2:integration:improper}{अभिसरण} द्वारा वैध और
\cref{thm:b2:series:fubini}.)
\end{proposition}

\begin{proof}
योग सूत्र \omterm{def:b2:integration:improper}{अभिसरण} श्रृंखला की रैखिकता है, और
$(a_n + b_n)r^n $ जब भी दोनों बंधे होते हैं $ a_nr^n $ और $ b_nr^n $
हैं: त्रिज्या $\geq \min(R_a, R_b)$. उत्पाद के लिए, ठीक करें $\abs z
< \min(R_a, R_b)$: दोनों शृंखलाएँ अभिसरित हैं \emph{\omterm{def:b2:series:def}{बिल्कुल}}
वहाँ (\cref{lem:b2:powerseries:abel}), इसलिए दोगुना अनुक्रमित
परिवार $(a_kz^k\,b_lz^l)_{k,l}$ है \omterm{def:b2:series:summable}{संक्षेपणीय}, और
\cref{thm:b2:series:fubini} किसी भी समूहीकरण की अनुमति देता है। द्वारा समूहीकरण $ k
+ l = n $:
\[
\Bigl(\sum_k a_kz^k\Bigr)\Bigl(\sum_l b_lz^l\Bigr)
= \sum_{n\geq0}\Bigl(\sum_{k+l=n}a_kb_l\Bigr)z^n
= \sum_{n\geq0}c_nz^n ,
\]
\omterm{def:b2:series:def}{बिल्कुल} ऐसे प्रत्येक के लिए \omterm{def:b2:integration:improper}{अभिसरण} $ z $: उत्पाद श्रृंखला
त्रिज्या है $\geq \min(R_a, R_b)$ भी।
\end{proof}

\begin{example}[एक कॉची स्क्वायर, क्रॉस-चेक किया गया]\label{ex:b2:powerseries:cauchysquare}
ज्यामितीय श्रृंखला को वर्गित करें: के लिए $\abs x < 1$, गुणांक
का $ x^n $ में $\bigl(\sum x^k\bigr)^2$ है $ c_n = \sum_{k+l=n}
1\cdot1 = n + 1$, इसलिए
\[
\frac{1}{(1-x)^2} = \sum_{n\geq0}(n+1)\,x^n .
\]
शब्द-दर-अवधि विभेदन द्वारा क्रॉस-चेक करें
(\cref{thm:b2:powerseries:calculus} नीचे): विभेद करना
$\frac{1}{1-x} = \sum x^n $ देता है $\frac{1}{(1-x)^2} = \sum
nx^{n-1} = \sum(n+1)x^n $ --- दो असंबद्ध द्वारा एक ही श्रृंखला
तंत्र. समापन अंतर्दृष्टि: जब एक गुणांक पहचान दिखती है
रहस्यमय, इन दो इंजनों में से एक (कन्वोल्यूशन या
विभेदन) आमतौर पर इसे एक पंक्ति में उत्पन्न करता है; सप्ताहांत
समस्या का प्रश्न चालू है $\sum\binom{2k}k\binom{2n-2k}{n-k} = 4^n $
कनवल्शन इंजन को पूरी शक्ति से चलाता है।
\end{example}

\begin{example}[से गुणा करना $\frac{1}{1-x}$ गुणांकों का योग करता है]\label{ex:b2:powerseries:partialsums}
ए \omterm{thm:b2:series:fubini}{कॉची उत्पाद} ज्यामितीय श्रृंखला के विरुद्ध एक यादगार है
अर्थ: किसी के लिए $\sum a_nx^n $ त्रिज्या का $ R > 0$ और $\abs x <
\min(R, 1)$,
\[
\frac{1}{1-x}\sum_{n\geq0}a_nx^n
= \sum_{n\geq0}\Bigl(\sum_{k=0}^{n}a_k\Bigr)x^n :
\]
से गुणा $\frac{1}{1-x}$ द्वारा गुणांक प्रतिस्थापित करता है
उनके आंशिक योग (सभी-व्यक्ति अनुक्रम के साथ \omterm{def:b2:metric:connected}{जुड़े हुए})।
उदाहरण: $\dfrac{\eu^x}{1-x} = \sum_n s_n x^n $ साथ $ s_n =
\sum_{k\leq n}\frac{1}{k!}$, का आंशिक योग $\eu $ ---
तुलना करना \cref{exo:b2:powerseries:11}, जहां वही उत्पाद
के साथ $\eu^{-x}$ विक्षोभ की गणना को एन्कोड करता है। समापन
अंतर्दृष्टि: पावर श्रृंखला पर संचालन चालू हैं
भेष में गुणांक अनुक्रम (से गुणा करें)। $\frac1{1-x}$:
जोड़; गुणा करके $ x $: बदलाव; अंतर करना: से गुणा करना $ n $
और शिफ्ट) --- एक शब्दकोश जनरेटिंग-फ़ंक्शन अध्याय
धाराप्रवाह पढ़ेंगे.
\end{example}

\section{योग की नियमितता}

\begin{theorem}[टर्म-दर-टर्म कैलकुलस]\label{thm:b2:powerseries:calculus}
होने देना $ f(x) = \sum_{n\geq0} a_n x^n $ त्रिज्या है $ R > 0$ (वास्तविक
परिवर्तनशील $ x \in \intoo{-R}{R}$).
\begin{enumerate}
  \item व्युत्पन्न श्रृंखला $\sum n\,a_n x^{n-1}$ के पास है \emph{वही}
        RADIUS $ R $, और $ f $ है $ C^1$ साथ $ f'(x) = \sum_{n \geq 1}
        n a_n x^{n-1}$. पुनरावृत्त, $ f $ है $ C^\infty $ और
        \[
        a_n = \frac{f^{(n)}(0)}{n!} :
        \]
        एक शक्ति श्रृंखला के गुणांक अद्वितीय हैं (दो श्रृंखला
        उसी राशि के पास $0$ समान गुणांक हैं)।
  \item शब्द-दर-अवधि आदिम: $\sum \frac{a_n}{n+1}x^{n+1}$ है
        त्रिज्या $ R $ और व्युत्पन्न $ f $.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{समान त्रिज्या:} अगर $(a_nr^n)$ घिरा हुआ है और $ r' < r $, तब
$ n\abs{a_n} r'^{\,n-1} = \frac{n}{r'}\abs{a_nr^n}
\bigl(\frac{r'}{r}\bigr)^n $ घिरा हुआ है (वास्तव में) $\to 0$: ज्यामितीय
धड़कता है $ n $), इसलिए $ R' \geq R $; इसके विपरीत $\abs{a_n x^n} \leq \abs x
\cdot n\abs{a_n}\abs x^{n-1}$ देता है $ R \geq R'$.

\emph{भेदभाव:} पर $\intcc{-r}{r}$, $ r < R $, व्युत्पन्न
शृंखला एकत्रित होती है \omterm{def:b2:funcseq:series}{सामान्यतः} ($ n\abs{a_n}r^{n-1}$ \omterm{def:b2:series:summable}{संक्षेपणीय} द्वारा
त्रिज्या गणना); मूल पर \omterm{def:b2:integration:improper}{अभिसरण} होता है $ x = 0$: द
श्रृंखला के लिए विभेदन प्रमेय
(\cref{thm:b2:funcseq:seriestransfer}) ऐसे प्रत्येक पर लागू होता है
खंड, इसलिए आगे $\intoo{-R}{R}$. बार-बार दोहराना $ k $ समय और
पर मूल्यांकन कर रहे हैं $0$: स्पष्ट रूप से, $ k $-वीं व्युत्पन्न श्रृंखला है
\[
f^{(k)}(x) = \sum_{n\geq k} n(n-1)\cdots(n-k+1)\,a_n\,x^{n-k},
\]
और कम से $ x = 0$ प्रत्येक पद के साथ $ n > k $ मिट जाता है, केवल छोड़कर
स्थिर पद $ k(k-1)\cdots1\cdot a_k $: $ f^{(k)}(0) =
k!\,a_k $. गुणांकों की विशिष्टता इस प्रकार है: दो शक्ति श्रृंखला
उसी राशि के पास $0$ पर समान व्युत्पन्न हैं $0$,
इसलिए वही है $ a_k $. आदिम: समान त्रिज्या समान
गणना, और पद दर पद अंतर करना।
\end{proof}

\begin{example}[एक बिंदु पर एक श्रृंखला का मूल्यांकन करना]\label{ex:b2:powerseries:evaluate}
क्या है $\sum_{n\geq1}\dfrac{n^2}{2^n}$? यह योग है
$\sum n^2x^n $ का \cref{exo:b2:powerseries:3} का मूल्यांकन
\emph{अंदर} डिस्क, पर $ x = \frac12 < 1 = R $, जहां हर
बंद फॉर्म प्राप्त करने के लिए इस्तेमाल किया गया हेरफेर वैध था:
\[
\sum_{n\geq1} n^2x^n = \frac{x(1+x)}{(1-x)^3}
\quad\Longrightarrow\quad
\sum_{n\geq1}\frac{n^2}{2^n}
= \frac{\frac12\cdot\frac32}{(\frac12)^3}
= \frac{3/4}{1/8} = 6 .
\]
वही इंजन, अन्य डायल: $ x = \frac13$ देता है
$\sum\frac{n^2}{3^n} = \frac{\frac13\cdot\frac43}{(2/3)^3} =
\frac32$. समापन अंतर्दृष्टि: एक शक्ति-श्रृंखला पहचान एक है
मशीन, एक भी फार्मूला नहीं --- एक-एक व्युत्पत्ति कीमतें
संख्यात्मक श्रृंखला $\sum n^2q^n $ एक बार में, सभी के लिए $\abs q < 1$;
जनरेटिंग-फ़ंक्शन अध्याय इस प्रकार गणना करेगा
अपेक्षाएँ और भिन्नताएँ थोक में।
\end{example}

\begin{example}[क्लासिक्स, इस बार ईमानदारी से]\label{ex:b2:powerseries:classics}
$\displaystyle\frac{1}{1 - x} = \sum x^n $ ($ R = 1$); एकीकरण
पद दर पद (\cref{thm:b2:powerseries:calculus}~(2)):
\[
-\ln(1 - x) = \sum_{n\geq1} \frac{x^n}{n},
\qquad
\arctan x = \sum_{n \geq 0} \frac{(-1)^n x^{2n+1}}{2n+1}
\quad (\abs x < 1),
\]
दो चरणों में दूसरा: स्थानापन्न $-x^2$ ज्यामितीय में
प्राप्त करने के लिए श्रृंखला $\frac{1}{1+x^2} = \sum(-1)^nx^{2n}$ (त्रिज्या
$1$, तब से $\abs{x^2} < 1 \iff \abs x < 1$), तो ले लो
शब्द-दर-अवधि आदिम लुप्त हो रहा है $0$; दोनों पक्ष हैं
समान मान के साथ समान फ़ंक्शन के आदिम $0$,
इसलिए बराबर है $\intoo{-1}{1}$. और $\exp $:
शृंखला $ E(x) = \sum \frac{x^n}{n!}$ ($ R = \infty $) संतुष्ट करता है
$ E' = E $, $ E(0) = 1$ पदवार विभेदन द्वारा, इसलिए $ E = \exp $ द्वारा
वर्ष 1 विशिष्टता. वर्ष 1 का प्रत्येक ``मानक विस्तार''
वॉल्यूम अब इसकी पूर्ण शक्ति श्रृंखला के बारे में एक प्रमेय है।
\end{example}

\begin{example}[डिस्क के अंदर से गणना किया गया एक लघुगणक]\label{ex:b2:powerseries:lntwo}
का मूल्यांकन $-\ln(1-x) = \sum\frac{x^n}{n}$ आंतरिक भाग पर
बिंदु $ x = \frac12$:
\[
\sum_{n\geq1}\frac{1}{n\,2^n} = \ln 2 ,
\]
का तेजी से परिवर्तित होने वाला प्रतिनिधित्व $\ln 2$ (दस शर्तें पहले से ही
देना $0.69306\ldots $ ख़िलाफ़ $\ln 2 = 0.69314\ldots $), दूर
वैकल्पिक श्रृंखला से बेहतर $1 - \frac12 + \frac13 -
\dots $ केवल सीमा पर उपलब्ध है। समापन अंतर्दृष्टि:
जब भी कोई स्थिरांक रिम और सख्ती दोनों पर पहुंच योग्य होता है
डिस्क के अंदर, इंटीरियर संख्यात्मक रूप से जीतता है --- ज्यामितीय
हार्मोनिक क्षय के विरुद्ध क्षय।
\end{example}

\begin{example}[विभेदन त्रिज्या को सुरक्षित रखता है, सीमा को नहीं]\label{ex:b2:powerseries:boundaryloss}
शृंखला $\sum_{n\geq1}\frac{x^n}{n^2}$ त्रिज्या है $1$ और
पर एकत्रित होता है \emph{दोनों} समापनबिंदु ($\sum\frac1{n^2}$ और उसके
वैकल्पिक जुड़वां)। इसकी व्युत्पन्न श्रृंखला,
\[
\sum_{n\geq1}\frac{x^{n-1}}{n} ,
\]
समान त्रिज्या है $1$ --- जैसा
\cref{thm:b2:powerseries:calculus} गारंटी --- लेकिन अभी
पर विचलन करता है $ x = 1$ (हार्मोनिक श्रृंखला) अभी भी \omterm{def:b2:integration:improper}{अभिसरण} करते हुए
पर $ x = -1$ (बदलते हुए)। एक और भेदभाव पैदा होता है
$\sum_{n\geq2}\frac{n-1}{n}x^{n-2}$, दोनों सिरों पर भिन्न
(शर्तें प्रवृत्त नहीं होतीं $0$). समापन अंतर्दृष्टि: प्रत्येक
विभेदन गुणांकों को कई गुना बढ़ा देता है $ n $, जो
त्रिज्या को कभी नहीं हिलाता (ज्यामितीय बहुपद को हराता है) लेकिन खाता है
सीमा क्षय का एक क्रम; टर्म-दर-टर्म कैलकुलस एक है
आंतरिक खेल, और रिम पर जो कुछ भी होता है वह अवश्य होना चाहिए
पुनः परीक्षण किया गया --- सप्ताहांत समस्या का एबेल--टाउबर सिद्धांत है
बिलकुल वही पुनः परीक्षा.
\end{example}

\begin{example}[किसी शृंखला को अवशेषों से विभाजित करना --- अंत तक काम किया]\label{ex:b2:powerseries:quarter}
गणना करना $ f(x) = \sum_{n\geq0} \dfrac{x^{4n}}{(4n)!}$ बंद में
प्रपत्र. दोनों $\cosh x = \sum \frac{x^{2m}}{(2m)!}$ और $\cos x =
\sum \frac{(-1)^m x^{2m}}{(2m)!}$ त्रिज्या है $\infty $, तो उनके
औसत की गणना पद दर पद की जा सकती है:
\[
\frac{\cosh x + \cos x}{2}
= \sum_{m\geq0}\frac{1 + (-1)^m}{2}\,\frac{x^{2m}}{(2m)!}
= \sum_{m \text{ even}}\frac{x^{2m}}{(2m)!}
= \sum_{n\geq0}\frac{x^{4n}}{(4n)!} = f(x) .
\]
फ़िल्टर $\frac{1+(-1)^m}{2}$ \omterm{def:b2:series:def}{बिल्कुल} सम रखता है $ m $: यह
रूट्स-ऑफ़-यूनिटी फ़िल्टर (कॉम्प्लेक्स) का वास्तविक अवतार है
संस्करण, के साथ $\iu^n $, अवशेष मॉड्यूलो निकालता है $4$ एक में
स्ट्रोक)। समापन जांच: $ f $ हल करती है $ f'''' = f $ साथ $ f(0) = 1$,
$ f'(0) = f''(0) = f'''(0) = 0$ ---श्रृंखला चार में अंतर करें
समय (\cref{thm:b2:powerseries:calculus}) और इसे देखो
स्वयं को पुनरुत्पादित करें; $\frac{\cosh + \cos}{2}$ उसी को संतुष्ट करता है
डेटा.
\end{example}

\begin{definition}[विश्लेषणात्मक कार्य]\label{def:b2:powerseries:analytic}
$ f $ है \emph{विश्लेषणात्मक}\index{विश्लेषणात्मक} पर $ x_0$ जब यह है
में एक शक्ति श्रृंखला का योग $(x - x_0)$ पड़ोस पर; एक पर
अंतराल, जब हर बिंदु पर. शक्ति श्रृंखला के योग विश्लेषणात्मक हैं
उनकी डिस्क के अंदर (विस्तार की पुनर्व्यवस्था --- पर स्वीकार किया गया
पुनः केन्द्रित करने के लिए यह स्तर, मामला $ x_0 = 0$ प्राणी
\cref{thm:b2:powerseries:calculus}). विश्लेषणात्मक तात्पर्य है $ C^\infty $;
संभाषण \emph{विफल रहता है}: फ्लैट फ़ंक्शन $\eu^{-1/x^2}$
(\cref{exo:b2:powerseries:7}).
\end{definition}

\begin{example}[पुन: केन्द्रित, और दूरी के रूप में त्रिज्या]\label{ex:b2:powerseries:recenter}
बढ़ाना $ f(x) = \frac{1}{1-x}$ आस-पास $ x_0 = \frac12$: लिखना
$ x = \frac12 + h $,
\[
\frac{1}{1 - x} = \frac{1}{\frac12 - h}
= \frac{2}{1 - 2h}
= \sum_{n\geq0} 2^{n+1}\,h^n
= \sum_{n\geq0} 2^{n+1}\Bigl(x - \frac12\Bigr)^{\!n},
\]
के लिए मान्य $\abs{2h} < 1$, यानी\ $\abs{x - \frac12} < \frac12$.
नई त्रिज्या \omterm{def:b2:series:def}{बिल्कुल} नए केंद्र से दूरी है
विलक्षणता $ x = 1$: पुन: केन्द्रित होने से सिकुड़ता है (या बढ़ता है)।
निकटतम बाधा को फिट करने के लिए डिस्क। समापन अंतर्दृष्टि: यह है
विश्लेषणात्मकता की परिभाषा के पीछे की तस्वीर --- एक
फ़ंक्शन, कई स्थानीय बिजली श्रृंखलाएं, प्रत्येक सबसे बड़े पर रहती हैं
परेशानी से बचने वाली डिस्क; वर्ष 3 खंड बदल जाता है
अनुमानी ``त्रिज्या $=$ निकटतम परिसर से दूरी
विलक्षणता'' एक प्रमेय में।
\end{example}

\begin{remark}[सामान्य ख़तरे]
\emph{(i) अनुपात परीक्षण पर्याप्त है, आवश्यक नहीं:} कब
$\abs{a_{n+1}/a_n}$ कोई सीमा नहीं है
(\cref{ex:b2:powerseries:sinn}, \cref{exo:b2:powerseries:1}),
परिभाषा पर वापस जाएँ: $ R = \sup\{r : (a_nr^n)$
घिरे $\}$. \emph{(ii) कोई भी चीज मुफ्त में सीमा पार नहीं करती:}
पद-दर-अवधि विभेदीकरण और एकीकरण प्रमेय हैं
\emph{अंदर}  \omterm{def:b2:metric:topology}{खुला} डिस्क; पर $\abs x = R $ प्रत्येक श्रृंखला अवश्य होनी चाहिए
दोबारा जांच की जाए (यह सप्ताहांत का \omterm{def:b2:metric:complete}{पूरा} विषय है
समस्या) \emph{(iii) किसी राशि की त्रिज्या:} $\min(R_a, R_b)$ है
केवल निचली सीमा --- रद्दीकरण ही इसे बढ़ा सकता है ($ a_n = 1,
b_n = -1$: \omterm{def:b2:funcseq:def}{समान रूप से} योग $0$, त्रिज्या $\infty $). \emph{(iv)
$ C^\infty $ क्या नहीं है \omterm{def:b2:powerseries:analytic}{विश्लेषणात्मक}:} एक \omterm{def:b2:integration:improper}{अभिसरण} टेलर श्रृंखला हो सकती है
में एकत्रित हों \emph{गलत} समारोह
(\cref{exo:b2:powerseries:7}); लिखने से पहले $ f(x) = \sum
\frac{f^{(n)}(0)}{n!}x^n $, इसे साबित करें --- एक ODE के माध्यम से
(\cref{met:b2:powerseries:ode}), एक शेष अनुमान, या एक
अभिन्न सूत्र.
\end{remark}

\begin{remark}[इसका उपयोग कहां किया जाता है]
पावर सीरीज़ तीन बाद के अध्यायों का वर्कहॉर्स है: द
अंतर-समीकरण अध्याय इंजेक्शन द्वारा रैखिक ODE को हल करता है
$\sum a_nx^n $ (नीचे विधि बॉक्स, औद्योगीकृत); द
जनरेटिंग-फंक्शन अध्याय के बारे में पहचान परिवर्तित करता है
शक्ति श्रृंखला और पीठ के बारे में पहचान में संभावनाएं; और
वर्ष 3 खंड जटिल चर को आधिकारिक बनाता है, जहां
विश्लेषणात्मकता जटिल भिन्नता के समतुल्य हो जाती है और
उपरोक्त ``स्वीकृत पुनः केन्द्रीकरण'' से इसका ईमानदार प्रमाण मिलता है। द
सप्ताहांत समस्या इस अध्याय के प्रमेयों का एक स्थान पर अन्वेषण करती है
चुप रहो: सीमा $\abs x = R $ स्वयं.
\end{remark}

\begin{method}[विभेदक समीकरण के माध्यम से विस्तार]\label{met:b2:powerseries:ode}
किसी फ़ंक्शन का विस्तार करना $ f $ एक शक्ति श्रृंखला में: एक रैखिक ODE खोजें
बहुपद गुणांक से संतुष्ट $ f $; इंजेक्षन $\sum a_nx^n $;
पुनरावृत्ति प्राप्त करने के लिए गुणांकों की पहचान करें $(a_n)$; हल करें, और
त्रिज्या और प्रारंभिक स्थितियों की जाँच करें। उदाहरण --- द्विपद
शृंखला: $ f(x) = (1+x)^\alpha $ संतुष्ट $(1+x)f' = \alpha f $, $ f(0)
= 1$; इंजेक्शन देता है $(n+1)a_{n+1} = (\alpha - n)a_n $, इसलिए $ a_n =
\binom{\alpha}{n}$, त्रिज्या $1$ (अनुपात परीक्षण), और योग, संतोषजनक
समान प्रारंभिक मान के साथ समान ODE, बराबर होता है $(1 + x)^\alpha $
रैखिक ODE के लिए विशिष्टता प्रमेय द्वारा (वर्ष 1 खंड)।
\end{method}

\begin{example}[एक मजबूर समीकरण पर विधि]\label{ex:b2:powerseries:odeforced}
हल करना $ y' = y + x $, $ y(0) = 0$, शक्ति श्रृंखला द्वारा। इंजेक्शन $ y =
\sum a_nx^n $:
\[
\sum_{n\geq0}(n+1)a_{n+1}x^n
= \sum_{n\geq0}a_nx^n + x ,
\]
और गुणांक की पहचान करना: $ a_1 = a_0 = 0$, $2a_2 = a_1 + 1
= 1$, और $(n+1)a_{n+1} = a_n $ के लिए $ n \geq 2$. इसलिए $ a_2 =
\frac{1}{2!}$ और, प्रेरण द्वारा, $ a_n = \frac{1}{n!}$ हरएक के लिए
$ n \geq 2$: त्रिज्या $\infty $, और
\[
y(x) = \sum_{n\geq2}\frac{x^n}{n!} = \eu^x - 1 - x .
\]
जाँच करना: $ y' = \eu^x - 1 = y + x $ और $ y(0) = 0$. समापन
अंतर्दृष्टि: पुनरावृत्ति \emph{है} समीकरण, गुणांक द्वारा
गुणांक; मजबूर करने वाला शब्द केवल बहुत से लोगों को परेशान करता है
प्रारंभिक गुणांक, जिसके बाद सजातीय पैटर्न लेता है
ओवर --- ``विशेष समाधान प्लस'' की एक अलग छाया
सजातीय समाधान ''।
\end{example}

\section{कार्य उत्पन्न करना}

\begin{example}[फाइबोनैचि]\label{ex:b2:powerseries:fibonacci}
होने देना $ F(x) = \sum_{n\geq0} F_n x^n $ (फाइबोनैचि संख्या, $ F_0 = 0$,
$ F_1 = 1$). पुनरावृत्ति $ F_{n+2} = F_{n+1} + F_n $ अनुवाद करता है,
से गुणा करना $ x^{n+2}$ और संक्षेप में, में
\[
F(x) - x = x\,F(x) + x^2 F(x)
\quad\Longrightarrow\quad
F(x) = \frac{x}{1 - x - x^2} ,
\]
मान्य जहां श्रृंखला अभिसरित होती है। त्रिज्या है
$\frac{1}{\varphi}$: से $ F_n \sim
\frac{\varphi^n}{\sqrt5}$ (बिनेट, अगला उदाहरण --- या क्रूड
प्रेरण $ F_n \leq 2^n $ प्लस पुनरावृत्ति), अनुपात परीक्षण
देता है
\[
\frac{F_{n+1}\abs x^{n+1}}{F_n\abs x^n}
\longrightarrow \varphi\abs x ,
\qquad\text{convergence iff } \abs x < \frac1\varphi
\approx 0.618 .
\] आंशिक भिन्न चालू
$\frac{x}{1 - x - x^2}$ और ज्यामितीय श्रृंखला बिनेट को पुनः प्राप्त करती है
सूत्र --- कार्य उत्पन्न करना रैखिक औद्योगीकरण करें
पुनरावृत्ति.\index{जनरेटिंग फ़ंक्शन}
\end{example}

\begin{example}[बिनेट का फार्मूला लागू किया गया]\label{ex:b2:powerseries:binet}
होने देना $\varphi = \frac{1+\sqrt5}{2}$ और $\psi =
\frac{1-\sqrt5}{2}$, की जड़ें $ X^2 = X + 1$; तब से $\varphi
+ \psi = 1$ और $\varphi\psi = -1$,
\[
1 - x - x^2 = (1 - \varphi x)(1 - \psi x) .
\]
आंशिक अंश: तलाश $\frac{x}{(1-\varphi x)(1-\psi x)} =
\frac{A}{1 - \varphi x} + \frac{B}{1 - \psi x}$, स्थिरांक
शब्द देता है $ A + B = 0$ और यह $ x $-गुणांक $-A\psi - B\varphi
= 1$, इसलिए $ A(\varphi - \psi) = 1$: $ A = \frac{1}{\sqrt5} = -B $.
दो ज्यामितीय शृंखलाएँ बाद में,
\[
F(x) = \frac{1}{\sqrt5}\sum_{n\geq0}
\bigl(\varphi^n - \psi^n\bigr)x^n
\quad\Longrightarrow\quad
F_n = \frac{\varphi^n - \psi^n}{\sqrt5}
\]
गुणांकों की विशिष्टता से
(\cref{thm:b2:powerseries:calculus}). तब से $\abs\psi < 1$, शब्द $\frac{\psi^n}{\sqrt5}$ पूर्ण मूल्य है $< \frac12$:
$ F_n $ है \emph{निकटतम पूर्णांक} को
$\frac{\varphi^n}{\sqrt5}$. समापन अंतर्दृष्टि: त्रिज्या
$\frac1\varphi $ का $ F $ प्रमुख जड़ का व्युत्क्रम है
--- गुणांक की वृद्धि और \omterm{def:b2:powerseries:radius}{अभिसरण की त्रिज्या} हैं
एक ही जानकारी विपरीत दिशाओं में पढ़ी जाती है।
\end{example}

\begin{example}[कैटलन संख्या]\label{ex:b2:powerseries:catalan}
 \omterm{ex:b2:powerseries:catalan}{कैटलन संख्या} $ C_n $ (त्रिकोणों की संख्या, कोष्ठक की संख्या,
डाइक पथों का, \dots) संतुष्ट $ C_0 = 1$ और $ C_{n+1} = \sum_{k=0}^n
C_kC_{n-k}$. \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन} $ C(x) = \sum C_nx^n $ फिर
संतुष्ट करता है (\omterm{thm:b2:series:fubini}{कॉची उत्पाद}!)
\[
C(x) = 1 + x\,C(x)^2
\quad\Longrightarrow\quad
C(x) = \frac{1 - \sqrt{1 - 4x}}{2x} ,
\]
के साथ जड़ का चयन करना $ C(0) = 1$: द्विघात को हल करना
$ xC^2 - C + 1 = 0$ दो उम्मीदवार देता है $\frac{1 \pm
\sqrt{1-4x}}{2x}$, और के रूप में $ x \to 0$ ``$+$'' जड़ फूट जाती है
जैसे $\frac1x $ जबकि ``$-$'' जड़ की प्रवृत्ति होती है $1$ (बढ़ाना
$\sqrt{1-4x} = 1 - 2x + O(x^2)$) --- केवल ऋण चिह्न ही कर सकता है
एक पावर सीरीज़ साथ रखें $ C_0 = 1$. विस्तार $\sqrt{1 - 4x}$ द्वारा
द्विपद श्रृंखला बंद रूप देती है
\[
C_n = \frac{1}{n+1}\binom{2n}{n} ,
\]
में किया गया \cref{exo:b2:powerseries:8}.\index{कैटलन संख्या}
\end{example}

\begin{remark}[औपचारिक श्रृंखला बनाम अभिसरण श्रृंखला]
उपरोक्त प्रत्येक जनरेटिंग-फ़ंक्शन गणना आह्वान द्वारा समाप्त होती है
गुणांकों की विशिष्टता, और वह प्रमेय जीवित है
\emph{अंदर} सकारात्मक त्रिज्या की एक डिस्क: ``रीडिंग ऑफ'' से पहले
$ F_n $ या $ C_n $, किसी को पता होना चाहिए $ R > 0$. एक कच्चा एक प्राथमिक बाध्य
पर्याप्त --- $ F_n \leq 2^n $ (तत्काल प्रेरण) देता है $ R \geq
\frac12$ फाइबोनैचि के लिए; $ C_n \leq 4^n $ (प्रत्येक \omterm{ex:b2:powerseries:catalan}{कैटलन संख्या}
पथों के उपसमुच्चय गिनता है) देता है $ R \geq \frac14$. सावधान रहें
पैमाने का पतित अंत: $\sum n!\,x^n $ त्रिज्या है $0$, और
एक फ़ंक्शन के रूप में इसमें हेरफेर करना अर्थहीन है --- पहचान
ऐसी श्रृंखला से संबंधित हैं \emph{औपचारिक} की गणना
गुणांक, अपने स्वयं के (अलग) के साथ एक विशुद्ध रूप से बीजगणितीय खेल
नियम. इस स्तर पर: हमेशा पहले एक सकारात्मक दायरा सुरक्षित करें,
फिर इसके अंदर स्वतंत्र रूप से गणना करें।
\end{remark}

\begin{remark}[इस खंड के भीतर परिप्रेक्ष्य]
पावर सीरीज़ पुस्तक के दो महान विस्तारों में से एक है
मशीनें; दूसरा हार्मोनिक की फूरियर श्रृंखला है
अध्याय, और उनकी तुलना करना शिक्षाप्रद है। एक शक्ति श्रृंखला है
कठोर: इसके गुणांक मजबूर हैं ($ a_n = f^{(n)}(0)/n!$),
इसका \omterm{def:b2:integration:improper}{अभिसरण} निर्दयी है (अंदर सामान्य, बाहर निराशाजनक),
और इसका योग है \omterm{def:b2:powerseries:analytic}{विश्लेषणात्मक} --- असीम रूप से कठोर
(\cref{def:b2:powerseries:analytic}). एक फूरियर श्रृंखला है
कोमल: यह केवल टुकड़े-टुकड़े-सुचारू संकेतों का प्रतिनिधित्व करता है
की सीमा पर नाजुक \omterm{def:b2:integration:improper}{अभिसरण} प्रश्नों की कीमत
चिकनाई. इस अध्याय के सप्ताहांत में दोनों सिद्धांत मिलते हैं
समस्या: सेस\`एआरओ और \omterm{thm:b2:series:abel}{हाबिल सारांश}, के लिए यहाँ विकसित किया गया
सीमा वृत्त, फूरियर अध्याय में फ़ेज़ के रूप में लौटें\'एर का
और पॉइसन की गुठली। इस बीच अंतर-समीकरण
अध्याय सीधे बिजली श्रृंखला का उपभोग करता है ($\eu^{tA}$, श्रृंखला
समाधान), और जनरेटिंग-फ़ंक्शन अध्याय बदल जाता है
\cref{ex:b2:powerseries:fibonacci}एक व्यवस्थित में चाल है
संभावनाओं के लिए गणना.
\end{remark}

\begin{exercise}[$\star $]\label{exo:b2:powerseries:1}
\omterm{def:b2:powerseries:radius}{अभिसरण की त्रिज्या}:
$\sum \dfrac{n^2}{2^n}z^n $; $\;\sum \dfrac{z^n}{\binom{2n}{n}}$;
$\;\sum z^{n!}$; $\;\sum \bigl(2 + (-1)^n\bigr)^n z^n $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:powerseries:2}
बताते हैं कि $\sum z^n $, $\sum \frac{z^n}{n}$, $\sum \frac{z^n}{n^2}$
सभी की त्रिज्या है $1$ लेकिन अलग तरह से व्यवहार करें $ z = 1$ और $ z = -1$:
विचलन/भिन्नता, विचलन/\omterm{def:b2:integration:improper}{अभिसरण},
\omterm{def:b2:integration:improper}{अभिसरण}/\omterm{def:b2:integration:improper}{अभिसरण}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:powerseries:3}
रकम की गणना करें, के लिए $\abs x < 1$:
\[
\sum_{n\geq0} n x^n,
\qquad
\sum_{n\geq0} n^2 x^n,
\qquad
\sum_{n\geq0} \frac{x^{2n+1}}{2n+1} .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:powerseries:4}
पर शक्ति श्रृंखला में विस्तार करें $0$, त्रिज्या के साथ: $\dfrac{1}{(1-x)(2-x)}$
(आंशिक अंश); $\;\ln(1 + x + x^2)$ \emph{(लिखना $1 + x + x^2
= \frac{1 - x^3}{1 - x}$)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:powerseries:5}
साबित करें कि $ f(x) = \sum_{n\geq1} H_n x^n = -\dfrac{\ln(1 -
x)}{1 - x}$ के लिए $\abs x < 1$, कहाँ $ H_n $ हार्मोनिक संख्या है
\emph{(\omterm{thm:b2:series:fubini}{कॉची उत्पाद} का $\sum x^n $ और $\sum \frac{x^n}{n}$)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:powerseries:6}
द्वारा हल करें \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन} पुनरावृत्ति $ u_0 = 1$, $ u_{n+1} =
2u_n + n $: गणना करें $ U(x) = \sum u_nx^n $ बंद रूप में, विघटित,
और पढ़ो $ u_n = 2^{n+1} - n - 1$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:powerseries:7}
होने देना $ f(x) = \eu^{-1/x^2}$ के लिए $ x \neq 0$, $ f(0) = 0$. साबित करें कि
$ f $ है $ C^\infty $ पर $\R $ साथ $ f^{(n)}(0) = 0$ सभी के लिए $ n $
\emph{(प्रेरण द्वारा दिखाएँ कि $ f^{(n)}(x) = P_n\bigl(\frac1x\bigr)
\eu^{-1/x^2}$ बहुपदों के लिए $ P_n $, और विकास का उपयोग करें
तुलना)}. समाप्त करें कि $ f $ क्या नहीं है \omterm{def:b2:powerseries:analytic}{विश्लेषणात्मक} पर $0$: यह टेलर है
श्रृंखला पर $0$ गलत कार्य में परिवर्तित हो जाता है।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:powerseries:8}
\omterm{def:b2:metric:complete}{पूरा} \cref{ex:b2:powerseries:catalan}: बढ़ाना $\sqrt{1 - 4x}$
द्विपद श्रृंखला के साथ, दिखा रहा है
\[
\binom{1/2}{n+1}(-4)^{n+1} = -\frac{2}{n+1}\binom{2n}{n},
\]
और निष्कर्ष निकालें $ C_n = \frac{1}{n+1}\binom{2n}{n}$; निर्धारित करें त्रिज्या
\omterm{def:b2:integration:improper}{अभिसरण} का का $ C(x)$ और के स्पर्शोन्मुखता $ C_n $ के माध्यम से
स्टर्लिंग.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:powerseries:9}
(हाबिल की रेडियल सीमा प्रमेय, विशेष मामला) मान लीजिए $\sum a_n $
जुटता है. साबित करें कि $\lim_{x \to 1^-} \sum_{n} a_n x^n = \sum_n
a_n $. \emph{(\omterm{thm:b2:series:abel}{हाबिल सारांश}: साथ $ A_n $ आंशिक रकम और $ A =
\lim A_n $, लिखना $\sum a_nx^n = (1 - x)\sum A_n x^n $; तब $\sum
a_nx^n - A = (1-x)\sum (A_n - A)x^n $, राशि को बड़े पैमाने पर विभाजित करें
$ N $.)} आवेदन पत्र: $\sum \frac{(-1)^{n-1}}{n} = \ln 2$ और $\sum
\frac{(-1)^n}{2n+1} = \frac\pi4$, शक्ति श्रृंखला से पुनः सिद्ध।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:powerseries:10}
बताते हैं कि $\displaystyle\sum_{n\geq1}\frac{x^n}{n(n+1)} = 1 +
\frac{1-x}{x}\,\ln(1-x)$ के लिए $0 < \abs x < 1$, निर्धारित करें
त्रिज्या, और जांचें कि \omterm{def:b2:integration:improper}{अभिसरण} सामान्य है
$\intcc{-1}{1}$; सत्यापित करें कि मान पर है $ x = 1$ द्वारा भविष्यवाणी की गई
निरंतरता दूरबीन राशि से सहमत हैं $\sum
\frac{1}{n(n+1)} = 1$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:powerseries:11}
(विक्षोभ) चलो $ D_n $ के क्रमपरिवर्तन की संख्या हो $ n $
निश्चित बिंदु के बिना वस्तुएं ($ D_0 = 1$). क्रमबद्ध करना
का क्रमपरिवर्तन $\{1, \dots, n\}$ उनके निश्चित-बिंदु सेट द्वारा देता है
$ n! = \sum_{k=0}^{n}\binom nk D_{n-k}$. गुणा करके
$\frac{x^n}{n!}$, योग, और पहचानें ए \omterm{thm:b2:series:fubini}{कॉची उत्पाद} प्राप्त करना
घातीय \omterm{ex:b2:powerseries:fibonacci}{जनरेटिंग फ़ंक्शन}
\[
\sum_{n\geq0} D_n\,\frac{x^n}{n!} = \frac{\eu^{-x}}{1-x}
\qquad (\abs x < 1),
\]
फिर बंद फॉर्म को पढ़ें $\dfrac{D_n}{n!} =
\sum_{k=0}^{n}\dfrac{(-1)^k}{k!}$ और सीमा $\dfrac{D_n}{n!}
\to \eu^{-1}$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:powerseries:12}
की द्विपद श्रृंखला से सिद्ध कीजिए
\cref{met:b2:powerseries:ode}, वह
\[
\frac{1}{\sqrt{1 - 4x}} = \sum_{n\geq0}\binom{2n}{n}x^n
\qquad \Bigl(\abs x < \frac14\Bigr),
\]
और वर्ग करके (\omterm{thm:b2:series:fubini}{कॉची उत्पाद} ख़िलाफ़
$\frac{1}{1-4x} = \sum 4^nx^n $), कनवल्शन पहचान
\[
\sum_{k=0}^{n}\binom{2k}{k}\binom{2n-2k}{n-k} = 4^n .
\]
\end{exercise}

\section{समस्या: हाबिल, टाउबर और अभिसरण की सीमा}

\begin{problem}\label{pb:b2:powerseries:1}
\omterm{def:b2:integration:improper}{अभिसरण} की डिस्क के अंदर सब कुछ आसान है; सभी
पावर सीरीज का ड्रामा होता है \emph{पर} सीमा. यह
समस्या वास्तविक चर में सीमा सिद्धांत का निर्माण करती है: एबेल की
प्रमेय अपने एकसमान रूप में, टाउबर के अंतर्गत इसका व्युत्क्रम
शर्त, सेस\`एआरओ--संकलन विधियों का एबेल पदानुक्रम
(फ्रोबेनियस प्रमेय के साथ), शब्द-दर-अवधि एकीकरण तक
लाभांश के रूप में शास्त्रीय स्थिरांक के साथ सीमा, और अंत में
की कठोरता \omterm{def:b2:powerseries:analytic}{विश्लेषणात्मक} कार्य ---पहचान
प्रमेय.\index{हाबिल का प्रमेय}\index{टाउबर का प्रमेय}
लगातार, $(a_n)$ एक वास्तविक अनुक्रम है, $ f(x) = \sum_{n\geq0}
a_nx^n $, और $ A_n = a_0 + \dots + a_n $.

\medskip
\noindent\textbf{Part I --- Abel's theorem, \omterm{def:b2:funcseq:def}{समान रूप से}.} मान लीजिये
इस भाग में वह $\sum a_n $ \omterm{def:b2:integration:improper}{अभिसरण}, और सेट $ r_n =
\sum_{k\geq n} a_k $ (इसलिए $ r_n \to 0$ और $ a_n = r_n -
r_{n+1}$).
\begin{enumerate}
  \item भागों द्वारा योग करके सिद्ध करें कि यह सभी के लिए है $0 \leq x
        \leq 1$ और $ N \leq M $:
        \[
        \Bigl|\sum_{n=N}^{M} a_n x^n\Bigr|
        \leq 2\sup_{n \geq N}\,\abs{r_n} .
        \]
  \item उसका निष्कर्ष निकालो $\sum a_nx^n $ \omterm{def:b2:integration:improper}{अभिसरण} \emph{\omterm{def:b2:funcseq:def}{समान रूप से}} पर
        $\intcc{0}{1}$, कि इसका योग है \omterm{def:b2:metric:continuity}{निरंतर} वहाँ, और
        की रेडियल सीमा पुनः प्राप्त करें
        \cref{exo:b2:powerseries:9}: $ f(x) \to \sum a_n $ जैसा $ x
        \to 1^-$.
  \item (हाबिल का प्रमेय \omterm{thm:b2:series:fubini}{कॉची उत्पाद}) होने देना $\sum a_n = A $,
        $\sum b_n = B $ और मान लीजिये \omterm{thm:b2:series:fubini}{कॉची उत्पाद} $\sum
        c_n $, $ c_n = \sum_{k} a_kb_{n-k}$, \emph{\omterm{def:b2:integration:improper}{अभिसरण}},
        योग के साथ $ C $. सिद्ध करना $ C = AB $ \emph{(डिस्क के अंदर
        उत्पाद की पहचान कायम रहती है
        \cref{prop:b2:powerseries:operations}; होने देना $ x \to
        1^-$)}.
  \item दिखाएँ परिकल्पना मायने रखती है: के लिए $ a_n = b_n =
        \frac{(-1)^n}{\sqrt{n+1}}$, दोनों शृंखलाएँ अभी भी अभिसरित हैं
        $\abs{c_n} \geq \frac{2(n+1)}{n+2} \geq 1$ \emph{(बाध्य
        प्रत्येक कारक $\sqrt{(k+1)(n-k+1)}$ एएम--जीएम द्वारा)}: द
        \omterm{thm:b2:series:fubini}{कॉची उत्पाद} दो अभिसारी श्रृंखलाएँ भिन्न हो सकती हैं।
  \item (का एक लाभांश \cref{exo:b2:powerseries:5}) बताते हैं कि
        $\bigl(\ln(1-x)\bigr)^2 = 2\sum_{n\geq1}
        \frac{H_n}{n+1}x^{n+1}$ पर $\intoo{-1}{1}$, जाँच करें कि
        $\bigl(\frac{H_n}{n+1}\bigr)_{n\geq1}$ तक कम हो जाता है
        $0$, और हाबिल के साथ समाप्त करें:
        \[
        \sum_{n\geq1} (-1)^{n+1}\,\frac{H_n}{n+1}
        = \frac{(\ln 2)^2}{2} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Tauber's converse.} पुकारना $\sum
a_n $ \emph{हाबिल-सारांश} को $ L $ कब $ f(x) \to L $ जैसा $ x \to
1^-$.
\begin{enumerate}[resume]
  \item बताते हैं कि $\sum (-1)^n $ हाबिल-योग्य है $\frac12$
        फिर भी भिन्न: हाबिल के प्रमेय में कोई शर्त नहीं है
        बातचीत
  \item (सेस\`एरो लेम्मा) यदि $ u_n \to 0$ तब $\frac{u_1 + \dots
        + u_N}{N} \to 0$ \emph{(राशि को एक निश्चित पर विभाजित करें $ m $)}.
  \item अब मान लीजिये $ n\,a_n \to 0$ और $ f(x) \to L $. साथ $ x_N =
        1 - \frac1N $, दो अनुमान सिद्ध करें
        \[
        \Bigl|\sum_{n=0}^{N} a_n\bigl(1 - x_N^n\bigr)\Bigr|
        \leq \frac{1}{N}\sum_{n=1}^{N} n\,\abs{a_n},
        \qquad
        \Bigl|\sum_{n>N} a_n x_N^n\Bigr|
        \leq \sup_{n>N}\bigl(n\abs{a_n}\bigr)
        \]
        \emph{(पहले के लिए, $1 - x^n \leq n(1-x)$; के लिए
        दूसरा, $\abs{a_n} \leq \frac{1}{N}\sup_{m>N}
        m\abs{a_m}$ और $\sum x_N^n \leq N $)}.
  \item निष्कर्ष निकालना \emph{टाउबर का प्रमेय}: अगर $ n\,a_n \to 0$ और
        $\sum a_n $ हाबिल-योग्य है $ L $, तब $\sum a_n $
        में एकत्रित हो जाता है $ L $.
  \item (सकारात्मक गुणांकों के लिए आसान टूबेरियन) यदि $ a_n
        \geq 0$ और $ f $ पर सीमाबद्ध है $\intco{0}{1}$, बताते हैं कि
        $\sum a_n $ \omterm{def:b2:integration:improper}{अभिसरण} और $\sum a_n = \lim_{x\to1^-}
        f(x)$ \emph{(अवश्यंभावी $\sum_{n\leq N}a_nx^n \leq f(x)$ और
        चलो $ x \to 1^-$, फिर एबेल का उपयोग करें)}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Cesàro means and Frobenius'
theorem.} पुकारना $\sum a_n $ \emph{सीईएस\`एआरओ-योग्य} को $ L $ कब
$\sigma_N = \frac{A_0 + \dots + A_{N-1}}{N} \to L $.
\begin{enumerate}[resume]
  \item दिखाएँ कि एक \omterm{def:b2:integration:improper}{अभिसरण} श्रृंखला Ces है\`एआरओ-सारांश करने योग्य
        इसका योग \emph{(प्रश्न 7 पर लागू होता है $ A_n - L $)}.
  \item सेस की गणना करें\`का एआरओ मान $\sum(-1)^n $ और इसकी जांच करें
        हाबिल मूल्य से सहमत हैं $\frac12$ प्रश्न 6 का.
  \item साथ $ S_n = A_0 + \dots + A_n = (n+1)\,\sigma_{n+1}$,
        दोनों सर्वसमिकाओं को सिद्ध करें $0 \leq x < 1$:
        \[
        f(x) = (1-x)^2\sum_{n\geq0}(n+1)\,\sigma_{n+1}x^n,
        \qquad
        (1-x)^2\sum_{n\geq0}(n+1)x^n = 1 .
        \]
  \item (फ्रोबेनियस) निष्कर्ष निकालें: यदि $\sigma_N \to L $ तब $ f(x) \to
        L $ जैसा $ x \to 1^-$ --- सेस\`एआरओ-सारांश का तात्पर्य है
        हाबिल-सारांश, समान मूल्य के लिए \emph{(दोनों को घटाएं
        पहचान और राशि को बड़े पैमाने पर विभाजित करें $ N $, के रूप में
        \cref{exo:b2:powerseries:9})}.
  \item दिखाएँ कि पदानुक्रम
        \[
        \text{convergent} \;\Longrightarrow\;
        \text{Cesàro-summable} \;\Longrightarrow\;
        \text{Abel-summable}
        \]
        दोनों तीरों पर सख्त है: पहले के लिए प्रश्न 6;
        दूसरे के लिए, वह दिखाओ $\sum(-1)^n(n+1)$ है
        हाबिल-सारांश करने योग्य $\frac14$ (गणना करें $ f $) लेकिन नहीं
        सेस\`एआरओ-सारांश (गणना)। $\sigma_N $ के लिए अलग से
        सम और विषम $ N $).
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Integrating to the boundary.}
\begin{enumerate}[resume]
  \item कल्पना करना $\sum a_nx^n $ पर एकत्रित हो जाता है $\intco{0}{1}$ और
        $\sum \frac{a_n}{n+1}$ जुटता है. सिद्ध करें कि
        अभिन्न अनुचित $\int_0^1 f $ मौजूद है और
        \[
        \int_0^1 \Bigl(\sum_{n\geq0} a_nx^n\Bigr)\dd x
        = \sum_{n\geq0}\frac{a_n}{n+1}
        \]
        \emph{(आदिम $ F(x) = \sum\frac{a_n}{n+1}x^{n+1}$
        है \omterm{def:b2:metric:continuity}{निरंतर} पर $1$ भाग I द्वारा)}.
  \item होने देना $\eta = \sum_{n\geq1}\frac{(-1)^{n-1}}{n^2}$. दिखाओ
        $\int_0^1 \frac{\ln(1+x)}{x}\dd x = \eta $ और, द्वारा
        सम और विषम सूचकांकों को विभाजित करना \omterm{def:b2:series:def}{बिल्कुल}
        संमिलित $\sum \frac1{n^2}$, वह $\eta =
        \frac12\sum_{n\geq1}\frac{1}{n^2}$. (द फूरियर
        अध्याय की सप्ताहांत समस्या का मूल्यांकन करता है
        $\sum\frac1{n^2} = \frac{\pi^2}{6}$.)
  \item सिद्ध करना
        \[
        \sum_{n\geq0}\frac{(-1)^n}{3n+1}
        = \int_0^1\frac{\dd x}{1+x^3}
        = \frac13\Bigl(\ln 2 + \frac{\pi}{\sqrt3}\Bigr)
        \]
        \emph{(श्रृंखला लीबनिज़ द्वारा \omterm{def:b2:integration:improper}{अभिसरण} करती है; एकीकृत करें
        ज्यामितीय श्रृंखला $\sum(-1)^nx^{3n}$ प्रश्न 16 के साथ;
        फिर आंशिक भिन्न: $\frac{1}{1+x^3} =
        \frac{1/3}{1+x} + \frac{(2-x)/3}{x^2-x+1}$)}.
  \item के लिए द्विपद श्रृंखला से $(1-t)^{-1/2}$
        (\cref{exo:b2:powerseries:12}) व्युत्पन्न
        \[
        \arcsin x = \sum_{n\geq0}
        \frac{\binom{2n}{n}}{4^n(2n+1)}\,x^{2n+1}
        \quad(\abs x < 1),
        \qquad\text{then}\qquad
        \sum_{n\geq0}\frac{\binom{2n}{n}}{4^n(2n+1)}
        = \frac\pi2 ,
        \]
        द्वारा सीमा मान को उचित ठहराना \emph{सामान्य}
        पर \omterm{def:b2:integration:improper}{अभिसरण} $\intcc{-1}{1}$ (उपयोग
        $\binom{2n}n4^{-n} \sim \frac{1}{\sqrt{\pi n}}$,
        \cref{ex:b2:comparison:centralbinomial}) --- यहाँ नहीं
        हाबिल की भी जरूरत है.
  \item (कैटलन सीमा पर) वह दिखाओ $\sum C_n 4^{-n} =
        2$: कैटलन श्रृंखला
        \cref{ex:b2:powerseries:catalan} \omterm{def:b2:integration:improper}{अभिसरण} \emph{पर}
        इसकी त्रिज्या $\frac14$ (असिम्प्टोटिक्स ऑफ
        \cref{exo:b2:powerseries:8}), इसका योग है \omterm{def:b2:metric:continuity}{निरंतर} पर
        $\intcc{0}{\frac14}$, और बंद फॉर्म की सीमा है $2$
        वहाँ।
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Rigidity: the identity theorem.}
\begin{enumerate}[resume]
  \item (पृथक शून्य) चलो $ f = \sum a_nx^n $ त्रिज्या है $ R >
        0$ और सभी नहीं $ a_n = 0$; होने देना $ m $ सबसे छोटा सूचकांक हो
        के साथ $ a_m \neq 0$. दिखाओ $ f(x) = x^m g(x)$ साथ $ g $ ए
        त्रिज्या की शक्ति श्रृंखला $ R $, $ g(0) = a_m \neq 0$, और
        उसका निष्कर्ष निकालो $ f $ कुछ पंक्चर में शून्य नहीं है
        का पड़ोस $0$.
  \item (पहचान प्रमेय) चलो $ f, h $ घात श्रृंखला का योग बनें
        निकट $0$ और $(x_k)$ का एक क्रम \emph{अशून्य}
        के साथ अंक $ x_k \to 0$ और $ f(x_k) = h(x_k)$. साबित करें कि $ f $ और $ h $ इसलिए, उनके गुणांक समान हैं
        निकट संयोग $0$.
  \item खोजो \emph{सभी} कार्य $ f $ \omterm{def:b2:powerseries:analytic}{विश्लेषणात्मक} पास में $0$ साथ
        \[
        f\Bigl(\frac1k\Bigr) = \frac{k^2}{k^2+1}
        \qquad\text{for all large integers } k .
        \]
  \item वह एक फ़ंक्शन दिखाएँ \omterm{def:b2:powerseries:analytic}{विश्लेषणात्मक} एक पर \omterm{def:b2:metric:topology}{खुला} अंतराल $ I $
        जो एक उपअंतराल पर गायब हो जाता है वह \omterm{def:b2:funcseq:def}{समान रूप से} गायब हो जाता है
        $ I $ \emph{(जिसके चारों ओर बिंदुओं का समूह $ f $ गायब हो जाता है
        \omterm{def:b2:funcseq:def}{समान रूप से} है \omterm{def:b2:metric:topology}{खुला} और, पहचान प्रमेय द्वारा
        संचय बिंदुओं पर लागू किया गया, बंद कर दिया गया $ I $)}.
        निष्कर्ष निकालिए कि कोई शून्येतर नहीं \omterm{def:b2:powerseries:analytic}{विश्लेषणात्मक} कार्य पर $\R $ है
        \omterm{def:b2:metric:compact}{कॉम्पैक्ट} समर्थन --- जबकि $ C^\infty$ टक्कर कार्य
        मौजूद (\cref{exo:b2:powerseries:7} की आपूर्ति करता है
        बिल्डिंग ब्लॉक): विश्लेषणात्मकता कठोर है, सहजता है
        पिलपिला।
  \item संश्लेषण। प्रत्येक एक वाक्य में: (i) हाबिल क्या है
        प्रमेय सामान्\omterm{def:b2:integration:improper}{य-अभिसरण} पैकेज में जोड़ता है
        \cref{lem:b2:powerseries:abel}; (ii) सटीक
        परिकल्पनाएँ जिसके अंतर्गत व्युत्क्रम (टाउबर) और धारण करता है
        मध्यवर्ती पायदान (फ्रोबेनियस); (iii) एक सीमा
        अब आप भाग IV से स्थिरांक प्राप्त कर सकते हैं
        दो पंक्तियों में मित्र; (iv) जहां सेस\`एआरओ का अर्थ है इच्छा
        बहुत अलग श्रृंखला के लिए, इस पुस्तक में पुनः प्रकट हों
        दयालु.
\end{enumerate}
\end{problem}
