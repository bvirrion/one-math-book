\chapter{कार्यों का क्रम और श्रृंखला}\label{ch:b2:funcseq}

जब फ़ंक्शन किसी फ़ंक्शन में परिवर्तित हो जाते हैं, तो कौन से गुण जीवित रहते हैं
सीमा तक मार्ग? \omterm{def:b2:funcseq:def}{बिंदुवार} \omterm{def:b2:integration:improper}{अभिसरण} लगभग सुरक्षित रखता है
कुछ नहीं; \emph{वर्दी} अभिसरण---सुपर में \omterm{def:b2:integration:improper}{अभिसरण} आदर्श
--- संरक्षित करता है निरंतरता, खंडों पर अभिन्न अंग और, एक मोड़ के साथ,
व्युत्पन्न। यह अध्याय तीन \omterm{def:b2:structures:sn}{स्थानांतरण} प्रमेयों को सिद्ध करता है,
उनके श्रृंखला संस्करण, और उन्हें वीयरस्ट्रैस का ताज पहनाया गया
सन्निकटन प्रमेय, बर्नस्टीन की सुंदर संभाव्यता से सिद्ध हुआ
बहुपद.

\section{बिंदुवार और एकसमान अभिसरण}

\begin{definition}\label{def:b2:funcseq:def}
होने देना $f_n, f \colon X \to \R $ (या $\C $, या एक मानक स्थान), $ X $ कोई भी
सेट. $(f_n)$ में एकत्रित हो जाता है $ f $
\emph{बिंदुवार}\index{बिंदुवार} कब $ f_n(x) \to
f(x)$ हरएक के लिए $ x $; \emph{समान रूप से}\index{समान रूप से}
कब
\[
\norm{f_n - f}_\infty = \sup_{x \in X}\, \abs{f_n(x) - f(x)}
\xrightarrow[n \to \infty]{} 0 .
\]
वर्दी का तात्पर्य बिंदुवार है; पर $ C(\intcc{a}{b})$, एक समान \omterm{def:b2:integration:improper}{अभिसरण}
में \omterm{def:b2:series:def}{बिल्कुल} \omterm{def:b2:integration:improper}{अभिसरण} है \omterm{def:b2:nvs:banach}{बनच स्थान}
$\bigl(C(\intcc{a}{b}), \norm\cdot_\infty\bigr)$ का
\cref{ch:b2:nvs}.
\end{definition}

\begin{example}\label{ex:b2:funcseq:xn}
पर $\intcc{0}{1}$, $ f_n(x) = x^n $ \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:funcseq:def}{बिंदुवार} तक
\emph{टूटनेवाला} आप LIMIT $ f = \mathbf{1}_{\{1\}}$; \omterm{def:b2:integration:improper}{अभिसरण}
एक समान नहीं है: $\norm{f_n - f}_\infty \geq f_n\bigl(1 -
\tfrac1n\bigr) = (1 - \tfrac1n)^n \to \eu^{-1} \neq 0$. पर
$\intcc{0}{a}$ साथ $ a < 1$ यह \emph{है} वर्दी ($\sup = a^n \to
0$): एकरूपता डोमेन की ही तरह एक संपत्ति है
क्रम.
\end{example}

\begin{example}[दो सीमाएँ जो आवागमन से इंकार करती हैं]\label{ex:b2:funcseq:doublelimit}
\omterm{def:b2:metric:complete}{पूरा} अध्याय अदला-बदली सीमाओं के बारे में है, इसलिए यहाँ है
सबसे छोटी संभव विफलता. होने देना $ a_{n,m} = \dfrac{n}{n+m}$ के लिए
$ n, m \geq 1$. तब
\[
\lim_{m\to\infty}\Bigl(\lim_{n\to\infty}a_{n,m}\Bigr)
= \lim_{m\to\infty} 1 = 1,
\qquad
\lim_{n\to\infty}\Bigl(\lim_{m\to\infty}a_{n,m}\Bigr)
= \lim_{n\to\infty} 0 = 0 :
\]
दोनों पुनरावृत्त सीमाएँ मौजूद हैं और वे भिन्न हैं। हर \omterm{def:b2:structures:sn}{स्थानांतरण}
इस अध्याय का प्रमेय दो सीमाओं को कम करने का लाइसेंस है ---
$\lim_n $ साथ $\lim_{x\to a}$ (निरंतरता), साथ $\int $
(एकीकरण), साथ $\frac{\dd}{\dd x}$ (भेदभाव) ---
और एकसमान \omterm{def:b2:integration:improper}{अभिसरण} वास्तव में वह शुल्क है जो बनाता है
रूपान्तरण कानूनी. समापन अंतर्दृष्टि: जब भी कोई ``प्रमाण''
चुपचाप दो सीमा संचालनों को स्वैप करता है, यह दो-पंक्ति सरणी है
इसके विरुद्ध रखने के लिए प्रति-उदाहरण; के फिसलने वाले उभार
\cref{exo:b2:funcseq:2} एक पहने हुए एक ही घटना हैं
अभिन्न चिह्न.
\end{example}

\section{तीन स्थानांतरण प्रमेय}

\begin{theorem}[निरंतरता]\label{thm:b2:funcseq:continuity}
यदि प्रत्येक $ f_n $ है \omterm{def:b2:metric:continuity}{निरंतर} पर $ a $ और $ f_n \to f $ \omterm{def:b2:funcseq:def}{समान रूप से} एक पर
का पड़ोस $ a $, तब $ f $ है \omterm{def:b2:metric:continuity}{निरंतर} पर $ a $. एक समान सीमा
का \omterm{def:b2:metric:continuity}{निरंतर} कार्य है \omterm{def:b2:metric:continuity}{निरंतर}.
\end{theorem}

\begin{proof}
 $3\varepsilon $ तर्क पहले से ही प्रयोग किया गया है
\cref{thm:b2:metric:rncomplete}: चुनना $ n $ साथ $\norm{f_n -
f}_\infty \leq \varepsilon $, तब $\delta $ से निरंतरता का
$ f_n $ पर $ a $; के लिए $\abs{x - a} \leq \delta $,
\[
\abs{f(x) - f(a)} \leq \abs{f(x) - f_n(x)} + \abs{f_n(x) - f_n(a)}
+ \abs{f_n(a) - f(a)} \leq 3\varepsilon . \qedhere
\]
\end{proof}

\begin{example}[एकरूपता ठीक वहीं विफल हो जाती है जहां सीमा टूटती है]\label{ex:b2:funcseq:threshold}
पर $\intcc{0}{2}$, होने देना $ f_n(x) = \dfrac{x^n}{1 + x^n}$.
\omterm{def:b2:funcseq:def}{बिंदुवार} सीमा एक तीन-टुकड़ा फ़ंक्शन है:
\[
f(x) = \begin{cases} 0 & 0 \leq x < 1,\\[2pt]
\tfrac12 & x = 1,\\[2pt]
1 & 1 < x \leq 2, \end{cases}
\]
पर असंतत $1$, तो द्वारा \cref{thm:b2:funcseq:continuity}
\omterm{def:b2:integration:improper}{अभिसरण} एक समान नहीं हो सकता $\intcc{0}{2}$. पर
बंद टुकड़े उस सीमा से बचते हैं जो यह है: के लिए $0 \leq x \leq
a < 1$,
\[
\sup_{\intcc{0}{a}}\abs{f_n - 0}
= \frac{a^n}{1 + a^n} \leq a^n \to 0 ,
\]
और के लिए $1 < b \leq x \leq 2$,
\[
\sup_{\intcc{b}{2}}\abs{f_n - 1}
= \frac{1}{1 + b^n} \leq b^{-n} \to 0 ,
\]
दोनों सुप्रीमा की एकरसता द्वारा गणना की गई $ u \mapsto
\frac{u}{1+u}$ और का $ x \mapsto x^n $. समापन अंतर्दृष्टि:
एकरूपता की विफलता को असंततता पर स्थानीयकृत किया जाता है
सीमा---समान ज्यामिति \cref{ex:b2:funcseq:xn}, और
''अंदर के हर खंड पर एक समान'' अनुशासन का कारण
पूरे अध्याय की पुनरावृत्ति होती है।
\end{example}

\begin{theorem}[एक खंड पर एकीकरण]\label{thm:b2:funcseq:integration}
अगर $ f_n \to f $ \omterm{def:b2:funcseq:def}{समान रूप से} पर $\intcc{a}{b}$, साथ $ f_n $ टुकड़े-टुकड़े में
\omterm{def:b2:metric:continuity}{निरंतर} ($ f $ इसी तरह), फिर
\[
\int_a^b f_n \longrightarrow \int_a^b f .
\]
\end{theorem}

\begin{proof}
अभिन्नों के लिए रैखिकता और त्रिभुज असमानता देते हैं
\[
\Bigl|\int_a^b f_n - \int_a^b f\Bigr|
= \Bigl|\int_a^b (f_n - f)\Bigr|
\leq \int_a^b\abs{f_n - f}
\leq (b - a)\,\norm{f_n - f}_\infty \longrightarrow 0 .
\]
लंबाई कारक $(b - a)$ कहां है सघनता खंड का
प्रवेश करता है: पर गैर \omterm{def:b2:metric:compact}{कॉम्पैक्ट} अंतराल वही अनुमान उत्पन्न करता है
बेकार बाध्य $\infty\cdot0$, और निष्कर्ष वास्तव में
प्रभुत्व के बिना विफल रहता है --- सपाट उभार $ f_n =
\frac1n\mathbf 1_{\intcc{0}{n}}$ एकाग्र \omterm{def:b2:funcseq:def}{समान रूप से} को $0$ पर
$\intco{0}{\infty}$ फिर भी रखो $\int f_n = 1$ (नुकसान टिप्पणी
नीचे), और फिसलने वाले उभार \cref{ch:b2:integration}का
टिप्पणी के साथ भी ऐसा ही करें \omterm{def:b2:funcseq:def}{बिंदुवार} \omterm{def:b2:integration:improper}{अभिसरण}; वर्दी
\omterm{def:b2:integration:improper}{अभिसरण} ऊँचाई को नियंत्रित करता है, चौड़ाई को कभी नहीं।
\end{proof}

\begin{theorem}[भेदभाव]\label{thm:b2:funcseq:differentiation}
होने देना $ f_n $ होना $ C^1$ एक अंतराल पर $ I $, साथ: $(f_n')$ अभिसारी
\emph{\omterm{def:b2:funcseq:def}{समान रूप से}} पर $ I $ (या के हर खंड पर $ I $) कुछ करने के लिए $ g $,
और $(f_n(x_0))$ एक बिंदु पर एकत्रित होना $ x_0$. तब $(f_n)$
\omterm{def:b2:integration:improper}{अभिसरण} (\omterm{def:b2:funcseq:def}{समान रूप से} खंडों पर) से ए $ C^1$ समारोह $ f $, और $ f'
= g $: कोई सीमा को अलग कर सकता है।
\end{theorem}

\begin{proof}
परिभाषित करना $ f(x) = \lim f_n(x_0) + \int_{x_0}^x g $: वैध, $ g $
प्राणी \omterm{def:b2:metric:continuity}{निरंतर} --- वास्तव में $ g $ है \emph{वर्दी} पर सीमा
के खंड \omterm{def:b2:metric:continuity}{निरंतर} $ f_n'$, इसलिए
\cref{thm:b2:funcseq:continuity} लागू होता है, और का अभिन्न अंग
ए \omterm{def:b2:metric:continuity}{निरंतर} फ़ंक्शन को मूलभूत रूप से अच्छी तरह से परिभाषित किया गया है
कैलकुलस का प्रमेय,
\[
f'(x) = g(x) \qquad (x \in I) :
\]
उम्मीदवार की सीमा है $ C^1$ सही व्युत्पन्न के साथ
\emph{निर्माण द्वारा}, इससे पहले कि कोई \omterm{def:b2:integration:improper}{अभिसरण} सिद्ध हो। द्वारा
मौलिक प्रमेय फिर से, $ f_n(x) = f_n(x_0) + \int_{x_0}^x
f_n'$; घटाना,
\[
\abs{f_n(x) - f(x)} \leq \abs{f_n(x_0) - \lim f_n(x_0)}
+ \abs{x - x_0}\,\norm{f_n' - g}_{\infty} ,
\]
जिसकी प्रवृत्ति होती है $0$ \omterm{def:b2:funcseq:def}{समान रूप से} हर खंड पर. और $ f $ है $ C^1$
साथ $ f' = g $ निर्माण द्वारा.
\end{proof}

\begin{example}[परिकल्पना डेरिवेटिव पर क्यों बैठती है?]\label{ex:b2:funcseq:sqrtabs}
होने देना $ F_n(x) = \sqrt{x^2 + \frac1n}$ पर $\R $. प्रत्येक $ F_n $ है
$ C^1$ (वास्तव में $ C^\infty $), और \omterm{def:b2:integration:improper}{अभिसरण} $\abs x $ है
सब पर एक समान $\R $:
\[
0 \leq F_n(x) - \abs x
= \frac{(x^2 + \frac1n) - x^2}{\sqrt{x^2+\frac1n} + \abs x}
= \frac{1/n}{\sqrt{x^2 + \frac1n} + \abs x}
\leq \frac{1/n}{1/\sqrt n} = \frac{1}{\sqrt n} .
\]
फिर भी सीमा $\abs x $ पर भिन्न नहीं है $0$: वर्दी
\omterm{def:b2:integration:improper}{अभिसरण} की \emph{कार्य}, चाहे कितनी भी तेज हो, स्थानान्तरण नहीं
भिन्नता विफलता डेरिवेटिव पर दिखाई देती है:
\[
F_n'(x) = \frac{x}{\sqrt{x^2 + \frac1n}}
\longrightarrow \begin{cases} 1 & x > 0,\\ 0 & x = 0,\\
-1 & x < 0, \end{cases}
\]
एक असंतत \omterm{def:b2:funcseq:def}{बिंदुवार} सीमा, तो $(F_n')$ जुट नहीं सकता
\omterm{def:b2:funcseq:def}{समान रूप से} पास में $0$ (\cref{thm:b2:funcseq:continuity} दोबारा)।
समापन अंतर्दृष्टि:
\cref{thm:b2:funcseq:differentiation} जानबूझकर मानता है
एकसमान \omterm{def:b2:integration:improper}{अभिसरण} की $ f_n'$, का नहीं $ f_n $ ---
यह उदाहरण कारण है.
\end{example}

\section{कार्यों की श्रृंखला}

\begin{definition}\label{def:b2:funcseq:series}
कार्यों की एक श्रृंखला $\sum u_n $ \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:funcseq:def}{बिंदुवार}/\omterm{def:b2:funcseq:def}{समान रूप से} कब
इसका आंशिक योग होता है। यह एकत्रित हो जाता है
\emph{सामान्यतः}\index{सामान्यतः} (पर $ X $) कब $\sum
\norm{u_n}_\infty < \infty $. सामान्य \omterm{def:b2:integration:improper}{अभिसरण} का तात्पर्य है वर्दी
\omterm{def:b2:integration:improper}{अभिसरण} (में \omterm{def:b2:nvs:banach}{बनच स्थान} बंधे हुए कार्यों का:
\cref{thm:b2:nvs:absoluteconvergence}), जिसका तात्पर्य है \omterm{def:b2:funcseq:def}{बिंदुवार};
दोनों निहितार्थ सख्त हैं.
\end{definition}

\begin{example}[एक शृंखला, तीन फैसले]\label{ex:b2:funcseq:threemodes}
लेना $ u_n(x) = \frac{x^n}{n}$ पर $\intco{0}{1}$. \emph{\omterm{def:b2:funcseq:def}{बिंदुवार}:}
प्रत्येक के लिए \omterm{def:b2:integration:improper}{अभिसरण} होता है $ x \in \intco01$ (के साथ तुलना
ज्यामितीय श्रृंखला)। \emph{सामान्य चालू $\intcc{0}{a}$, $ a < 1$:}
$\norm{u_n}_{\infty,\intcc0a} = \frac{a^n}{n}$, \omterm{def:b2:series:summable}{संक्षेपणीय}.
\emph{पर सामान्य नहीं है $\intco{0}{1}$:}
$\norm{u_n}_{\infty,\intco01} = \frac1n $, और $\sum\frac1n $
विचलन \emph{एक समान भी नहीं $\intco{0}{1}$:} द
शेष निकट प्रतिरोध करता है $1$,
\[
R_N(x) = \sum_{n>N}\frac{x^n}{n}
\geq \sum_{n=N+1}^{2N}\frac{x^n}{n}
\geq \frac{N\,x^{2N}}{2N} = \frac{x^{2N}}{2}
\xrightarrow[x\to1^-]{} \frac12 ,
\]
इसलिए $\sup_{\intco01}\abs{R_N} \geq \frac12$ हरएक के लिए $ N $.
समापन अंतर्दृष्टि: सभी चार फैसले शांतिपूर्वक सह-अस्तित्व में हैं ---
योग $-\ln(1-x)$ है \omterm{def:b2:metric:continuity}{निरंतर} पर $\intco{0}{1}$ क्योंकि
निरंतरता केवल एकरूपता की आवश्यकता है \emph{प्रत्येक बिंदु के पास}, अर्थात\
खंडों पर $\intcc0a $; किनारे पर उड़ना है
योग सही है.
\end{example}

\begin{theorem}[श्रृंखला के लिए स्थानांतरण]\label{thm:b2:funcseq:seriestransfer}
अगर $\sum u_n $ \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:funcseq:def}{समान रूप से} (उदा.\ \omterm{def:b2:funcseq:series}{सामान्यतः}) प्रासंगिक पर
सेट: निरंतरता के सभी $ u_n $ पर $ a $ योग तक जाता है; एकीकरण
एक खंड पर शब्द दर शब्द किया जा सकता है; और यदि $\sum u_n(x_0)$
जबकि \omterm{def:b2:integration:improper}{अभिसरण} होता है $\sum u_n'$ \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:funcseq:def}{समान रूप से} खंडों पर,
योग है $ C^1$ व्युत्पन्न के साथ $\sum u_n'$.
\end{theorem}

\begin{proof}
सब कुछ आंशिक पर लागू संगत प्रमेय है
रकम $ S_N = \sum_{n\leq N}u_n $, जो कि परिमित योग हैं
प्रासंगिक नियमितता के साथ कार्य करता है। निरंतरता: प्रत्येक $ S_N $
है \omterm{def:b2:metric:continuity}{निरंतर} पर $ a $ और $ S_N \to \sum u_n $ \omterm{def:b2:funcseq:def}{समान रूप से}:
\cref{thm:b2:funcseq:continuity}. एकीकरण: खंड पर,
\[
\int_a^b \sum_{n\geq0} u_n
= \lim_N \int_a^b S_N
= \lim_N \sum_{n=0}^{N}\int_a^b u_n
= \sum_{n\geq0}\int_a^b u_n
\]
द्वारा \cref{thm:b2:funcseq:integration} (पहली समानता) और
अभिन्न की रैखिकता (दूसरा)। विभेदन:
$ S_N $ हैं $ C^1$, $ S_N(x_0)$ \omterm{def:b2:integration:improper}{अभिसरण}, और $ S_N' =
\sum_{n\leq N}u_n'$ \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:funcseq:def}{समान रूप से} खंडों पर:
\cref{thm:b2:funcseq:differentiation} देता है कि योग है
$ C^1$ व्युत्पन्न के साथ $\lim S_N' = \sum u_n'$.
\end{proof}

\begin{remark}[सामान्य ख़तरे]
चार जाल, सभी परीक्षा प्रतियों में देखे गए। \emph{(मैं)
आधा-अधूरा जांचा गया सुप्रीमा:} का मूल्यांकन $ f_n $ एक अच्छी तरह से चुने हुए के साथ
क्रम $ x_n $ केवल सीमा $\norm{f_n - f}_\infty $ से
\emph{नीचे} --- एकरूपता को अस्वीकार करने के लिए पर्याप्त है (जैसा कि)
\cref{ex:b2:funcseq:xn}), इसे कभी साबित न करें; इसे साबित करने के लिए बाध्य है
एक गणना द्वारा समर्थन के लिए मान्य है \emph{सभी} $ x $. \emph{(ii)
गलत सेट पर एकरूपता:} सामान्य या एकसमान \omterm{def:b2:integration:improper}{अभिसरण}
अक्सर हर पर पकड़ रखता है $\intcc{-a}{a}$ या $\intco\delta\infty $
लेकिन पर विफल रहता है \omterm{def:b2:metric:topology}{खुला} संघ; वह कोई बाधा नहीं है --- निरंतरता
और भिन्नता स्थानीय है, इसलिए खंड-दर-खंड
का अनुशासन \cref{ex:b2:funcseq:zeta} उन्हें समग्र रूप से देता है
\omterm{def:b2:metric:topology}{खुला} तय करना। \emph{(iii) गैर-खंडों पर एकीकरण:}
\cref{thm:b2:funcseq:integration} के बारे में एक बयान है
खंड; पर $\intco0\infty $, एकसमान \omterm{def:b2:integration:improper}{अभिसरण} नहीं करता
द्रव्यमान को अनंत तक भागने से रोकें ($ f_n =
\frac1n\mathbf 1_{\intcc{0}{n}}$ \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:funcseq:def}{समान रूप से} को $0$,
साथ $\int f_n = 1$) --- वहाँ प्रभुत्व \omterm{def:b2:integration:improper}{अभिसरण} का उपयोग करें।
\emph{(iv) सीमा में अंतर करना:}
\cref{ex:b2:funcseq:sqrtabs}; व्युत्पन्न परिकल्पना चालू है
$(f_n')$, और \omterm{def:b2:integration:improper}{अभिसरण} की कोई दर नहीं $(f_n)$ प्रतिस्थापित कर सकते हैं
यह.
\end{remark}

\begin{example}[रीमैन $\zeta $ समारोह]\label{ex:b2:funcseq:zeta}
$\zeta(s) = \sum_{n\geq1} n^{-s}$ \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:funcseq:series}{सामान्यतः} हर पर
अर्ध-पंक्ति $\intco{a}{+\infty}$, $ a > 1$ ($\norm{n^{-s}}_\infty =
n^{-a}$, \omterm{def:b2:series:summable}{संक्षेपणीय}): $\zeta $ है \omterm{def:b2:metric:continuity}{निरंतर} पर $\intoo{1}{+\infty}$;
शब्दवार विभेदन (व्युत्पन्न श्रृंखला)। $\sum -\ln n\; n^{-s}$
\omterm{def:b2:integration:improper}{अभिसरण} भी करता है \omterm{def:b2:funcseq:series}{सामान्यतः} पर $\intco{a}{\infty}$), $\zeta $ है $ C^1$
--- और, पुनरावृत्त करते हुए, $ C^\infty $ --- साथ $\zeta'(s) = -\sum \frac{\ln
n}{n^s}$. अनुशासन पर ध्यान दें: सामान्य \omterm{def:b2:integration:improper}{अभिसरण} पर जाँच की जाती है
\emph{उप}-आधी-पंक्तियाँ, पर कभी नहीं \omterm{def:b2:metric:topology}{खुला} $\intoo{1}{\infty}$
स्वयं, जहां यह विफल रहता है।
\end{example}

\begin{example}[एक लघुगणक श्रृंखला, अंत तक काम करती रही]\label{ex:b2:funcseq:logseries}
होने देना $ F(x) = \sum_{n\geq1} \frac{\eu^{-nx}}{n}$ पर
$\intoo{0}{\infty}$. प्रत्येक पद पर परिबद्ध है $\intco{\delta}
\infty $ द्वारा $\frac{\eu^{-n\delta}}{n} \leq \eu^{-n\delta}$, ए
\omterm{def:b2:integration:improper}{अभिसरण} ज्यामितीय श्रृंखला: सामान्य \omterm{def:b2:integration:improper}{अभिसरण} सभी पर
$\intco\delta\infty $, इसलिए $ F $ है \omterm{def:b2:metric:continuity}{निरंतर} पर
$\intoo{0}{\infty}$. व्युत्पन्न श्रृंखला $\sum -\eu^{-nx}$ है
इसी तरह \omterm{def:b2:funcseq:series}{सामान्यतः} पर \omterm{def:b2:integration:improper}{अभिसरण} $\intco\delta\infty $
($\norm{\eu^{-nx}}_{\infty,\intco\delta\infty} =
\eu^{-n\delta}$), इसलिए $ F $ है $ C^1$ एक ज्यामितीय व्युत्पन्न के साथ:
\[
F'(x) = -\sum_{n\geq1}\eu^{-nx}
= \frac{-\eu^{-x}}{1 - \eu^{-x}}
= \frac{-1}{\eu^{x} - 1} .
\]
पुनरावृत्त, $ F $ है $ C^\infty $. घालमेल $ F'$ (दोनों $ F $ और $ x
\mapsto -\ln(1 - \eu^{-x})$ पर गायब हो जाओ $+\infty $ और है
पर समान व्युत्पन्न $\intoo0\infty $):
\[
F(x) = -\ln\bigl(1 - \eu^{-x}\bigr),
\]
लघुगणक श्रृंखला पर $ t = \eu^{-x}$. समापन अंतर्दृष्टि: जैसा $ x
\to 0^+$, $ F(x) = -\ln(x + O(x^2)) = \ln\frac1x + O(x)$ --- द
श्रृंखला सीमा पर लघुगणकीय रूप से विचलन करती है, \omterm{def:b2:series:def}{बिल्कुल} वैसे ही
यह जिस हार्मोनिक श्रृंखला पर बन जाता है $ x = 0$; सामान्य \omterm{def:b2:integration:improper}{अभिसरण} पर
$\intco\delta\infty $ लेकिन चालू नहीं $\intoo0\infty $ है
लक्षण.
\end{example}

\begin{method}[एकसमान अभिसरण को सिद्ध या असिद्ध करना]\label{met:b2:funcseq:uniform}
के लिए $ f_n \to f $ \omterm{def:b2:funcseq:def}{बिंदुवार} पर $ X $:
\begin{enumerate}
  \item गणना या बाध्य $\norm{f_n - f}_\infty $: अध्ययन करें
        समारोह $ x \mapsto \abs{f_n(x) - f(x)}$ (व्युत्पन्न,
        एकरसता) इसकी अधिकतम स्थिति का पता लगाने के लिए; के लिए मान्य एक बाध्यता
        सब $ x $ ऐसा होता है $0$ एकरूपता सिद्ध करता है.
  \item को \emph{असत्य सिद्ध करना}: प्रदर्शन बिंदु $ x_n $ साथ
        $\abs{f_n(x_n) - f(x_n)} \not\to 0$ (अक्सर $ x_n $ ट्रैक
        चलती हुई टक्कर, जैसे कि \cref{exo:b2:funcseq:1}); या
        गर्भनिरोधक में \omterm{def:b2:structures:sn}{स्थानांतरण} प्रमेय लागू करें --- ए
        की असंतत सीमा \omterm{def:b2:metric:continuity}{निरंतर} कार्य
        (\cref{ex:b2:funcseq:threshold}), या $\int f_n
        \not\to \int f $ एक खंड पर.
  \item श्रृंखला के लिए, प्रयास करें सामान्य \omterm{def:b2:integration:improper}{अभिसरण} पहला
        ($\sum\sup\abs{u_n} < \infty $); यदि यह विश्व स्तर पर विफल रहता है,
        उन उप-खंडों पर इसका परीक्षण करें जो मायने रखते हैं
        (\cref{ex:b2:funcseq:threemodes}); यदि यह विफल रहता है
        हर जगह, एकसमान \omterm{def:b2:integration:improper}{अभिसरण} के माध्यम से अभी भी धारण कर सकता है
        वैकल्पिक शेष बाध्य
        (\cref{exo:b2:funcseq:4}) या भागों द्वारा योग।
\end{enumerate}
\end{method}

\section{वीयरस्ट्रैस सन्निकटन प्रमेय}

\begin{theorem}[वीयरस्ट्रैस, बर्नस्टीन के माध्यम से]\label{thm:b2:funcseq:weierstrass}
प्रत्येक \omterm{def:b2:metric:continuity}{निरंतर} $ f \colon \intcc{0}{1} \to \R $ एक समान सीमा है
बहुपदों का --- स्पष्ट रूप से, इसका \emph{बर्नस्टीन
बहुपद}\index{बर्नस्टीन
बहुपद}\index{वीयरस्ट्रैस
सन्निकटन प्रमेय}
\[
B_n(f)(x) = \sum_{k=0}^{n} f\Bigl(\frac kn\Bigr)\binom nk x^k
(1-x)^{n-k} .
\]
\end{theorem}

\begin{proof}
हल करना $ x \in \intcc{0}{1}$ और सेट करें $ p_k(x) = \binom nk x^k(1 -
x)^{n-k}$. मूल्यांकन द्वारा प्राप्त तीन द्विपद सर्वसमिकाएँ
$(x + y)^n $ और इसके दो $ x $-डेरिवेटिव पर $ y = 1 - x $:
\[
\sum_k p_k = 1,
\qquad
\sum_k k\,p_k = nx,
\qquad
\sum_k k(k-1) p_k = n(n-1)x^2 .
\]
विस्तार से: $(x+y)^n = \sum_k\binom nk x^ky^{n-k}$ पर $ y = 1-x $
पहला है; में अंतर करना $ x $,
\[
n(x+y)^{n-1} = \sum_k k\binom nk x^{k-1}y^{n-k} ,
\]
फिर से गुणा करना $ x $ और सेटिंग $ y = 1 - x $ देता है
दूसरा; दो बार अंतर करना और गुणा करना $ x^2$ देता है
तीसरा. विस्तार $(k - nx)^2 = k(k-1) + k(1 - 2nx) +
n^2x^2$ और तीनों का संयोजन:
\[
\sum_k (k - nx)^2 p_k
= n(n-1)x^2 + nx(1 - 2nx) + n^2x^2
= nx(1 - x) \leq \frac n4 ,
\]
 \emph{विचरण पहचान}.

अब अनुमान लगाएं, प्रयोग करें $\sum p_k = 1$:
\[
\abs{B_n(f)(x) - f(x)}
\leq \sum_{k} \Bigl| f\Bigl(\frac kn\Bigr) - f(x)\Bigr|\, p_k(x)
= \Sigma_{\text{near}} + \Sigma_{\text{far}} ,
\]
के अनुसार विभाजन $\abs{\frac kn - x} \leq \delta $ या नहीं।
दिया गया $\varepsilon > 0$, वर्दी निरंतरता का $ f $ (हेन) प्रदान करता है
$\delta $ साथ $\Sigma_{\text{near}} \leq \varepsilon $. दूर के लिए
योग, साथ $ M = \norm f_\infty $: विचरण पहचान द्वारा और
चेबीशेव की गिनती की चाल,
\[
\Sigma_{\text{far}} \leq 2M \sum_{\abs{k - nx} > n\delta} p_k
\leq 2M\,\frac{\sum_k (k - nx)^2 p_k}{n^2\delta^2}
\leq \frac{2M}{4 n \delta^2} \cdot 1
= \frac{M}{2n\delta^2}
\xrightarrow[n\to\infty]{} 0 ,
\]
\omterm{def:b2:funcseq:def}{समान रूप से} में $ x $. इसलिए $\norm{B_n(f) - f}_\infty \leq \varepsilon +
\frac{M}{2n\delta^2} \leq 2\varepsilon $ बड़े के लिए $ n $.
\end{proof}

\begin{remark}
एफ़िन प्रतिस्थापन द्वारा प्रमेय किसी भी खंड पर लागू होता है
$\intcc{a}{b}$. यह विफल रहता है $\R $ (बहुपदों की एक समान सीमा

उपबीजगणित (पत्थर--वीयरस्ट्रैस) और को \omterm{def:b2:metric:compact}{कॉम्पैक्ट} रिक्त स्थान
\end{remark}

\begin{example}[बहुभुज सन्निकटन, एक दर के साथ]\label{ex:b2:funcseq:polygonal}
के लिए $ L $-लिप्सचिट्ज़ $ f $ पर $\intcc{0}{1}$, होने देना $ I_nf $ हो
नोड्स पर टुकड़े-टुकड़े-एफ़िन इंटरपोलेंट $\frac kn $. ए पर
सेल $\intcc{\frac kn}{\frac{k+1}n}$, दोनों $ f(x)$ और
$ I_nf(x)$ चरम मूल्यों के बीच स्थित है कि एक $ L $-लिप्सचिट्ज़
फ़ंक्शन दो नोडल मान ले सकता है, इसलिए $ x $ में
सेल, लेखन $ x_k = \frac kn $:
\[
\abs{I_nf(x) - f(x)}
\leq \abs{I_nf(x) - f(x_k)} + \abs{f(x_k) - f(x)}
\leq L\,\abs{x - x_k} + L\,\abs{x - x_k}
\leq \frac{2L}{n}
\]
(इंटरपोलेंट स्वयं है $ L $-सेल पर \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़}: इसका
ढलान का एक अंतर भागफल है $ f $). इस तरह $\norm{I_nf -
f}_\infty \leq \frac{2L}{n}$: का बहुभुज सन्निकटन
\omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} कार्य गति से एकत्रित होते हैं $\frac1n $ ---
\emph{और तेज} बर्नस्टीन की तुलना में $\frac{1}{\sqrt n}$ के लिए
एक ही कक्षा (सप्ताहांत समस्या, भाग II)। समापन अंतर्दृष्टि:
बहुभुज प्रक्षेप करता है लेकिन चिकना नहीं है, बर्नस्टीन चिकना है
लेकिन धीमा; की नियमितता के बीच कोई निःशुल्क दोपहर का भोजन नहीं है
सन्निकटन और गति --- एक व्यापार-बंद द्वारा सटीक बनाया गया
सप्ताहांत समस्या के संतृप्ति परिणाम।
\end{example}

\begin{remark}[इस खंड के भीतर परिप्रेक्ष्य]
एकसमान \omterm{def:b2:integration:improper}{अभिसरण} यहाँ से यह इस पुस्तक का कार्यस्रोत है। द
पावर-सीरीज़ चैप्टर पूरी तरह से चलता है सामान्य \omterm{def:b2:integration:improper}{अभिसरण} पर
\omterm{def:b2:metric:compact}{कॉम्पैक्ट} सबडिस्क --- प्रत्येक पद-दर-पद प्रमेय है
इस अध्याय के \omterm{def:b2:structures:sn}{स्थानांतरण} प्रमेयों का विशेष मामला। फूरियर
अध्याय एक मंजिल ऊपर रहता है: इसका आंशिक योग $ S_N $ विफल
\omterm{def:b2:series:def}{बिल्कुल} वहीं जहां यह अध्याय चेतावनी देता है कि वे (\omterm{def:b2:funcseq:def}{बिंदुवार} लेकिन नहीं
छलांग पर एकसमान), और इसका फ़ेज़\'एर का मतलब उसी से सफल होना है
$3\varepsilon $ यांत्रिकी जो साबित हुई
\cref{thm:b2:funcseq:continuity}. अंतर-समीकरण
अध्याय परिभाषित करता है $\eu^{tA}$ ए द्वारा \omterm{def:b2:funcseq:series}{सामान्यतः} \omterm{def:b2:integration:improper}{अभिसरण} श्रृंखला और
शब्दानुसार इसे अलग करता है --- शाब्दिक रूप से
\cref{thm:b2:funcseq:seriestransfer} मैट्रिक्स पर लागू किया गया
प्रविष्टियाँ। जब पुस्तक में बाद में ``हम क्यों कर सकते हैं'' के बारे में संदेह हुआ
ऐसा करो'', उत्तर आमतौर पर इस अध्याय का एक प्रमेय है।
\end{remark}

\begin{exercise}[$\star\star\star $]\label{exo:b2:funcseq:9}
(ए \omterm{def:b2:metric:continuity}{निरंतर}, कहीं भी भिन्न कार्य नहीं --- निर्देशित) चलो
$\varphi $ निकटतम पूर्णांक की दूरी हो ($1$-आवधिक,
$\norm{\varphi}_\infty = \frac12$, $1$-लिप्सचिट्ज़) और
\[
W(x) = \sum_{n=0}^{\infty} \Bigl(\frac{3}{4}\Bigr)^{\!n}
\varphi(4^n x) .
\]
साबित करें: (ए) $ W $ है \omterm{def:b2:metric:continuity}{निरंतर} पर $\R $ (सामान्य \omterm{def:b2:integration:improper}{अभिसरण}); (बी) के लिए
प्रत्येक $ x $ और हर $ m $, चुनना $ h_m = \pm\frac12\cdot 4^{-m}$
साइन बनाने के साथ $\varphi $ से खंड पर affine $4^m x $
को $4^m(x + h_m)$, अंतर भागफल संतुष्ट करता है
\[
\Bigl|\frac{W(x + h_m) - W(x)}{h_m}\Bigr| \geq 3^m -
\sum_{n<m} 3^n \geq \frac{3^m + 1}{2} \xrightarrow[m\to\infty]{}
\infty
\]
(शर्तें $ n > m $ आवधिकता से गायब हो जाना; अवधि $ n = m $ योगदान देता है
बिलकुल $3^m $; शर्तें $ n < m $ से बंधे हैं \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} संपत्ति)।
समाप्त करें कि $ W $ कहीं भी भिन्न नहीं है.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:funcseq:12}
(श्रृंखला द्वारा एक वोल्टेरा इंटीग्रल समीकरण) के लिए $ f \in
C(\intcc{0}{1})$ परिभाषित करना $ Tf(x) = \int_0^x f(t)\,\dd t $.
\begin{enumerate}
  \item प्रेरण द्वारा दिखाएँ कि के लिए $ n \geq 1$:
        \[
        T^n f(x) = \int_0^x \frac{(x -
        t)^{n-1}}{(n-1)!}\,f(t)\,\dd t,
        \qquad
        \norm{T^n f}_\infty \leq \frac{\norm f_\infty}{n!} .
        \]
  \item उसका निष्कर्ष निकालो $ S = \sum_{n\geq0} T^n f $ \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:funcseq:series}{सामान्यतः}
        पर $\intcc{0}{1}$ और अभिन्न समीकरण को हल करता है $ S = f
        + TS $.
  \item सत्यापित करें कि $ S(x) = f(x) + \int_0^x \eu^{x-t}f(t)\,\dd
        t $ समान समीकरण को हल करता है, और विशिष्टता सिद्ध करता है
        \omterm{def:b2:metric:continuity}{निरंतर} समाधान \emph{(अगर $ S = TS $ तब
        $\norm{S}_\infty \leq \norm{T^nS}_\infty \to 0$)}:
        योग का बंद रूप समाप्त करें।
\end{enumerate}
\end{exercise}

\section{समस्या: सन्निकटन की दरें और कोरोवकिन का प्रमेय}

\begin{problem}\label{pb:b2:funcseq:1}
बर्नस्टीन का प्रमाण \cref{thm:b2:funcseq:weierstrass} छुपाता है
दो खजाने. सबसे पहले, यह है \emph{मात्रात्मक}: कितनी तेजी से $ B_nf
\to f $ के मापांक द्वारा शासित होता है निरंतरता का $ f $, के साथ
द्वारा तीव्र दर प्राप्त की गई $\abs{x - \frac12}$. दूसरा, यह है
\emph{संरचनात्मक}: बस यही मायने रखता था $ B_n $ एक है
सकारात्मक रैखिक ऑपरेटर अच्छा व्यवहार कर रहा है $1$, $ x $, $ x^2$ ---
वह अवलोकन, पृथक, है \emph{कोरोव्किन का
प्रमेय}\index{कोरोव्किन का
प्रमेय}. यह समस्या दोनों को सिद्ध करती है,
और वोरोनोव्स्काया के सटीक स्पर्शोन्मुख के साथ समाप्त होता है। लगातार,
$ f \in C(\intcc{0}{1})$, $ M = \norm f_\infty $, $ p_k(x) = \binom
nk x^k(1-x)^{n-k}$, और $ e_j $ अर्थ है $ x \mapsto x^j $.

\medskip
\noindent\textbf{Part I --- The Bernstein operator.}
\begin{enumerate}
  \item बताते हैं कि $ B_n $ रैखिक है, \emph{सकारात्मक} ($ f \geq 0
        \Rightarrow B_nf \geq 0$), इसलिए मोनोटोन ($ f \leq g
        \Rightarrow B_nf \leq B_ng $), साथ
        $\norm{B_nf}_\infty \leq \norm f_\infty $, ओर वो
        $ B_nf $ प्रक्षेप करता है $ f $ दोनों अंतिम बिंदुओं पर.
  \item पहचान पुनः प्राप्त करें $ B_n e_0 = e_0$, $ B_n e_1 =
        e_1$ और $ B_n e_2 = e_2 + \frac{e_1 - e_2}{n}$
        \emph{(अंतर करें $(x + y)^n $ दो बार और सेट करें $ y = 1 -
        x $)}.
  \item विचरण पहचान निकालें $\sum_k \bigl(\frac kn -
        x\bigr)^2 p_k(x) = \frac{x(1-x)}{n}$ और, द्वारा
        कॉची--श्वार्ज़, प्रथम-क्षण बाध्य
        \[
        \sum_{k=0}^{n}\Bigl|\frac kn - x\Bigr|\,p_k(x)
        \leq \sqrt{\frac{x(1-x)}{n}} \leq \frac{1}{2\sqrt n} .
        \]
  \item दिखाओ अगर $ f $ तो उत्तल है $ B_nf \geq f $ पर
        $\intcc{0}{1}$ \emph{(परिमित जेन्सेन असमानता के लिए
        वजन $ p_k(x)$)}.
  \item (चेबीशेव की गिनती बंधी हुई, पुनर्कथन) के लिए $\delta > 0$
        दिखाओ
        \[
        \sum_{\abs{k/n - x} > \delta} p_k(x)
        \leq \frac{x(1-x)}{n\delta^2}
        \leq \frac{1}{4n\delta^2} ,
        \]
        और संभाव्य रीडिंग दें: $ B_nf(x)$ औसत
        $ f $ एक द्विपद नमूने पर ध्यान केंद्रित करने का मतलब है $ x $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Rates: the modulus of निरंतरता.}
के लिए $\delta > 0$ तय करना $\omega(\delta) = \sup\{\abs{f(s) - f(t)}
: s, t \in \intcc{0}{1},\ \abs{s - t} \leq \delta\}$.
\begin{enumerate}[resume]
  \item दिखाओ: $\omega $ परिमित है, घटता नहीं,
        $\omega(\delta) \to 0$ जैसा $\delta \to 0^+$ (हेन),
        उपयोज्य ($\omega(\delta_1 + \delta_2) \leq
        \omega(\delta_1) + \omega(\delta_2)$), और
        $\omega(\lambda\delta) \leq (1 +
        \lambda)\,\omega(\delta)$ सभी के लिए $\lambda > 0$.
  \item प्रत्येक के लिए मुख्य अनुमान सिद्ध करें $\delta > 0$:
        \[
        \abs{B_nf(x) - f(x)}
        \leq \sum_k \omega\Bigl(\Bigl|\frac kn -
        x\Bigr|\Bigr)p_k(x)
        \leq \Bigl(1 + \frac1\delta\sum_k\Bigl|\frac kn -
        x\Bigr|p_k(x)\Bigr)\,\omega(\delta) .
        \]
  \item चुनना $\delta = n^{-1/2}$ और निष्कर्ष निकालो
        \emph{मात्रात्मक वीयरस्ट्रैस प्रमेय}:
        \[
        \norm{B_nf - f}_\infty \leq
        \frac32\,\omega\Bigl(\frac{1}{\sqrt n}\Bigr)
        \xrightarrow[n\to\infty]{} 0 .
        \]
  \item दरें निकालें: $\norm{B_nf - f}_\infty \leq
        \frac{3L}{2\sqrt n}$ के लिए $ L $-लिप्सचिट्ज़ $ f $, और $\leq
        \frac32 C n^{-\alpha/2}$ के लिए $\alpha $-एच\"पुराने $ f $
        ($\abs{f(s) - f(t)} \leq C\abs{s-t}^\alpha $).
  \item (स्पष्ट उदाहरण --- एक द्विपद पहचान) के लिए $ m
        \geq 1$ सिद्ध करना
        \[
        \sum_{k=m+1}^{2m} (k - m)\binom{2m}{k}
        = \frac{m}{2}\binom{2m}{m},
        \qquad\text{hence}\qquad
        \sum_{k=0}^{2m}\abs{k - m}\binom{2m}{k}
        = m\binom{2m}{m}
        \]
        \emph{(उपयोग $ k\binom{2m}k = 2m\binom{2m-1}{k-1}$ और
        द्विपद रेखा की समरूपता, जो देती है
        $\sum_{j=m}^{2m-1}\binom{2m-1}{j} = 2^{2m-2}$)}.
  \item के लिए $ f(t) = \abs{t - \frac12}$ सटीक मान निकालें
        और इसके स्पर्शोन्मुख (केंद्रीय द्विपद,
        \cref{ex:b2:comparison:centralbinomial}):
        \[
        B_{2m}f\Bigl(\frac12\Bigr) - f\Bigl(\frac12\Bigr)
        = \frac{\binom{2m}{m}}{2\cdot4^{m}}
        \;\sim\; \frac{1}{2\sqrt{\pi m}} :
        \]
        दर $\omega(n^{-1/2})$ प्रश्न 8 का प्राप्त होता है
        (एक स्थिरांक तक) --- मात्र के लिए \omterm{def:b2:metric:continuity}{निरंतर} $ f $,
        बर्नस्टीन का $ n^{-1/2}$ ईमानदार है.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Korovkin's theorem.} होने देना $(L_n)$
का एक क्रम हो \emph{सकारात्मक रैखिक} ऑपरेटरों से
$ C(\intcc{0}{1})$ अपने आप से ऐसा कि $ L_ne_j \to e_j $
\omterm{def:b2:funcseq:def}{समान रूप से} के लिए $ j = 0, 1, 2$.
\begin{enumerate}[resume]
  \item दिखाएँ कि एक सकारात्मक रैखिक $ L $ एकस्वर है और
        संतुष्ट करता है $\abs{Lf} \leq L\abs f $ \omterm{def:b2:funcseq:def}{बिंदुवार}.
  \item दिखाएँ: प्रत्येक के लिए $\varepsilon > 0$ वहाँ है $\delta > 0$
        ऐसे कि के लिए \emph{सभी} $ s, x \in \intcc{0}{1}$:
        \[
        \abs{f(s) - f(x)} \leq \varepsilon +
        \frac{2M}{\delta^2}(s - x)^2
        \]
        \emph{(इलाज $\abs{s - x} \leq \delta $ हेन द्वारा और
        $\abs{s-x} > \delta $ क्रूड बाउंड द्वारा $2M $)}.
  \item हल करना $ x $, आवेदन करना $ L_n $ प्रश्न 13 की असमानता के लिए
        चर में $ s $, और प्राप्त करें
        \[
        \abs{L_nf(x) - f(x)\,L_ne_0(x)}
        \leq \varepsilon\,L_ne_0(x) + \frac{2M}{\delta^2}
        \bigl(L_ne_2(x) - 2x\,L_ne_1(x) + x^2 L_ne_0(x)\bigr).
        \]
  \item बताते हैं कि $\sup_x \bigl(L_ne_2(x) - 2x\,L_ne_1(x) + x^2
        L_ne_0(x)\bigr) \to 0$, फिर इकट्ठा करें
        \emph{कोरोवकिन का प्रमेय}: $ L_nf \to f $ \omterm{def:b2:funcseq:def}{समान रूप से} के लिए
        \emph{प्रत्येक} $ f \in C(\intcc{0}{1})$.
  \item जाँच करें कि $(B_n)$ कोरोव्किन की परिकल्पनाओं को संतुष्ट करता है:
        वीयरस्ट्रैस तीसरी बार, तीन मोनोमियल से।
  \item होने देना $ I_n $ टुकड़े-टुकड़े-एफ़िन इंटरपोलेशन बनें
        नोड्स पर ऑपरेटर $\frac kn $. दिखाओ $ I_n $ है
        सकारात्मक रैखिक, $ I_ne_0 = e_0$, $ I_ne_1 = e_1$, और
        $\norm{I_ne_2 - e_2}_\infty = \frac{1}{4n^2}$
        \emph{(प्रत्येक कोशिका पर एफ़िन इंटरपोलेशन की त्रुटि
        का $ t^2$ है $(t - a)(b - t)$)}. कोरोवकिन द्वारा निष्कर्ष:
        बहुभुज इंटरपोलेंट \omterm{def:b2:integration:improper}{अभिसरण} करते हैं \omterm{def:b2:funcseq:def}{समान रूप से} हरएक के लिए
        \omterm{def:b2:metric:continuity}{निरंतर} $ f $.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Dividends: density, moments,
derivatives.}
\begin{enumerate}[resume]
  \item वह बहुपद दिखाएँ \emph{तर्कसंगत} गुणांक
        में सघन हैं $\bigl(C(\intcc{0}{1}),
        \norm\cdot_\infty\bigr)$: यह \omterm{def:b2:nvs:banach}{बनच स्थान} है
        अलग करने योग्य.
  \item (क्षण कार्य निर्धारित करते हैं) चलो $ f \in
        C(\intcc{0}{1})$ साथ $\int_0^1 f(t)\,t^n \dd t = 0$
        हरएक के लिए $ n \in \N $. दिखाओ $\int_0^1 f P = 0$ प्रत्येक के लिए
        बहुपद, फिर $\int_0^1 f^2 = 0$, तब $ f = 0$.
  \item व्युत्पन्न पहचान सिद्ध करें
        \[
        (B_nf)'(x) = n\sum_{k=0}^{n-1}\Bigl(
        f\Bigl(\frac{k+1}{n}\Bigr) -
        f\Bigl(\frac kn\Bigr)\Bigr)\,
        \binom{n-1}{k}x^k(1-x)^{n-1-k}
        \]
        \emph{(अंतर करें $ p_k $ और पुनः अनुक्रमणिका --- एक \omterm{thm:b2:series:abel}{हाबिल
        सारांश})}.
  \item कल्पना करना $ f $ है $ C^1$. माध्य मान प्रमेय का उपयोग करना
        प्रत्येक वृद्धि और तुलना के साथ $ B_{n-1}(f')$, दिखाओ
        $(B_nf)' \to f'$ \omterm{def:b2:funcseq:def}{समान रूप से} पर $\intcc{0}{1}$. निष्कर्ष:
        के लिए $ f \in C^1$ वहाँ बहुपदों का \omterm{def:b2:integration:improper}{अभिसरण} होता है $ f $
        \emph{के साथ साथ} उनके व्युत्पन्न.
  \item कल्पना करना $ f $ है $ C^2$. टेलर द्वारा - लैग्रेंज एट $ x $ दिखाओ
        \[
        \abs{B_nf(x) - f(x)} \leq
        \frac{\norm{f''}_\infty}{2}\cdot\frac{x(1-x)}{n}
        \leq \frac{\norm{f''}_\infty}{8n} :
        \]
        चिकनाई दर को उन्नत करती है $ n^{-1/2}$ को
        $ n^{-1}$.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Saturation: Voronovskaya's
theorem.}
\begin{enumerate}[resume]
  \item चौथे क्षण की पहचान सिद्ध करें
        \[
        \sum_k (k - nx)^4 p_k(x)
        = nx(1-x)\bigl(1 + 3(n-2)x(1-x)\bigr) \leq n^2
        \quad (n \geq 1)
        \]
        \emph{(बढ़ाना $ k^4$ गिरते फैक्टोरियल में $ k(k-1)\cdots $
        और प्रश्न 2 की विभेदन युक्ति का दो बार उपयोग करें
        अधिक)}.
  \item (वोरोनोव्स्काया) चलो $ f $ होना $ C^2$ और $ x \in
        \intcc{0}{1}$. लिखना $ f(t) = f(x) + f'(x)(t-x) +
        \frac{f''(x)}2(t-x)^2 + \eta(t)(t-x)^2$ साथ $\eta $
        घिरा हुआ और $\eta(t) \to 0$ जैसा $ t \to x $, सिद्ध करना
        \[
        n\bigl(B_nf(x) - f(x)\bigr)
        \xrightarrow[n\to\infty]{}
        \frac{x(1-x)}{2}\,f''(x)
        \]
        \emph{(विभाजित करें $\eta $ योग पर $\abs{t - x} \leq
        \delta $; प्रश्न 23 से सुदूर भाग को नियंत्रित करें)}. तो
        प्रश्न 22 की त्रुटि क्रम में सटीक है \emph{और}
        स्थिर: $ B_n $ \emph{संतृप्त} पर $\frac1n $ हालाँकि
        चिकना $f $ हो सकता है --- तुलना करें
        \cref{exo:b2:funcseq:7}.
  \item संश्लेषण। प्रत्येक एक वाक्य में: (i) कैसी सकारात्मकता
        अकेले खरीदा (भाग I और III); (ii) कहां सघनता
        का $\intcc{0}{1}$ हर हिस्से में प्रवेश किया; (iii) तीन क्यों?
        कोरोवकिन के प्रमेय में परीक्षण कार्य पर्याप्त हैं; (iv) द
        ट्रेड-ऑफ बर्नस्टीन (मजबूत) बनाता है $ n^{-1/2}$ किसी न किसी के लिए
        $ f $, लेकिन ए $\frac1n $ चिकनी के लिए छत $ f$), और कौन सा
        इस पुस्तक के अध्याय के साथ वही खेल खेलेंगे
        त्रिकोणमितीय बहुपद.
\end{enumerate}
\end{problem}
