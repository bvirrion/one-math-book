\chapter{विभेदक कैलकुलस}\label{ch:b2:diffcalc}

वर्ष 1 खंड का दो-चर कलन परिपक्व होता है
मानक स्थानों के बीच मानचित्रों की \omterm{def:b2:diffcalc:differential}{विभेदक} गणना:
\emph{\omterm{def:b2:diffcalc:differential}{अंतर}} सर्वोत्तम रैखिक सन्निकटन के रूप में, श्रृंखला नियम
पूर्ण व्यापकता में, श्वार्ज़ की समरूपता प्रमेय \emph{साबित}, टेलर
सूत्र, और एक्स्ट्रेमा का संपूर्ण दूसरे क्रम का विश्लेषण। द
व्युत्क्रम फलन प्रमेय, सिद्धांत का मुकुट, इसके साथ बताया गया है
प्रमाण रणनीति --- एक बानाच निश्चित बिंदु --- स्पष्ट किया गया।

लगातार, $U $ एक \omterm{def:b2:metric:topology}{खुला} का उपसमुच्चय $\R^n $ (या एक मानक स्थान;
परिमित-आयामी मामला सभी विचारों को वहन करता है), $ f \colon U \to
\R^m $.

\section{अंतर}

\begin{definition}\label{def:b2:diffcalc:differential}
$ f $ है \emph{विभेदक}\index{विभेदक} पर $ a $ जब वहाँ
एक है (\omterm{def:b2:metric:continuity}{निरंतर}) रेखीय मानचित्र $\dd f_a \colon \R^n \to \R^m $ साथ
\[
f(a + h) = f(a) + \dd f_a(h) + o(\norm h)
\qquad (h \to 0).
\]
वो नक्शा $\dd f_a $, द \emph{अंतर}\index{अंतर} का
$ f $ पर $ a $, निराला है; विहित आधारों में इसका मैट्रिक्स है
द \emph{जैकोबियन मैट्रिक्स}\index{जैकोबियन मैट्रिक्स} $ J_f(a) =
\bigl(\frac{\partial f_i}{\partial x_j}(a)\bigr)$.
भिन्नता का तात्पर्य है निरंतरता और सबका अस्तित्व
दिशात्मक व्युत्पन्न $\dd f_a(v) = \lim_{t\to0}\frac{f(a + tv)
- f(a)}{t}$; बातचीत विफल हो जाती है
(\cref{exo:b2:diffcalc:2}). के लिए $ m = 1$, $\dd f_a(h) = \langle
\nabla f(a), h\rangle $: वर्ष 1 का ग्रेडिएंट, अब इस रूप में समझा जाता है
अंतर का प्रतिनिधित्व करने वाला वेक्टर।
\end{definition}

\begin{theorem}[$ C^1$ मानदंड]\label{thm:b2:diffcalc:c1}
यदि सभी आंशिक व्युत्पन्न $ f $ पर मौजूद है $ U $ और हैं \omterm{def:b2:metric:continuity}{निरंतर}
पर $ a $, तब $ f $ है \omterm{def:b2:diffcalc:differential}{विभेदक} पर $ a $. ``$ C^1$ पर $ U $'' ---
\omterm{def:b2:metric:continuity}{निरंतर} आंशिक --- इसलिए तात्पर्य है भिन्नता
हर जगह, साथ \omterm{def:b2:metric:continuity}{निरंतर} \omterm{def:b2:diffcalc:differential}{अंतर}.
\end{theorem}

\begin{proof}
घटक द्वारा घटक ($ m = 1$ पर्याप्त)। दो के लिए वर्ष 1 का प्रमाण
वेरिएबल --- एक समय में एक निर्देशांक को स्थानांतरित करें, एक-वेरिएबल लागू करें
प्रत्येक पैर पर माध्य मान प्रमेय, और उपयोग करें निरंतरता आंशिक का
पर $ a $ --- शब्दशः सामान्यीकरण करता है $ n $ पैर:
\[
f(a + h) - f(a) = \sum_{j=1}^{n} \bigl(f(a + h^{(j)}) - f(a +
h^{(j-1)})\bigr)
= \sum_j h_j\,\frac{\partial f}{\partial x_j}(\xi_j),
\]
कहाँ $ h^{(j)}$ सबसे पहले जमा देता है $ j $ के निर्देशांक $ h $ और
$\xi_j $ पर स्थित है $ j $-वां पैर; निरंतरता प्रत्येक को घुमाता है
$\frac{\partial f}{\partial x_j}(\xi_j)$ में $\frac{\partial
f}{\partial x_j}(a) + o(1)$, और त्रुटि है $ o(\norm h)$.
\end{proof}

\begin{theorem}[शृंखला नियम]\label{thm:b2:diffcalc:chainrule}
अगर $ f $ है \omterm{def:b2:diffcalc:differential}{विभेदक} पर $ a $ और $ g $ पर $ f(a)$, तब $ g \circ f $
है \omterm{def:b2:diffcalc:differential}{विभेदक} पर $ a $ साथ
\[
\dd(g \circ f)_a = \dd g_{f(a)} \circ \dd f_a ,
\qquad
J_{g\circ f}(a) = J_g\bigl(f(a)\bigr)\,J_f(a) :
\]
जैकोबियन गुणा करते हैं।\index{शृंखला नियम}
\end{theorem}

\begin{proof}
लिखना $ f(a + h) = f(a) + \dd f_a(h) + \norm h\,\varepsilon_1(h)$
और $ g(b + k) = g(b) + \dd g_b(k) + \norm k\,\varepsilon_2(k)$ साथ
$ b = f(a)$, $ k = k(h) = \dd f_a(h) + \norm h \varepsilon_1(h)$.
प्रतिस्थापित करना,
\[
g(f(a+h)) = g(b) + \dd g_b\bigl(\dd f_a(h)\bigr)
+ \norm h\,\dd g_b(\varepsilon_1(h)) + \norm{k}\,\varepsilon_2(k),
\]
और दोनों त्रुटि शर्तें हैं $ o(\norm h)$: के बाद से पहली बार $\dd g_b $ है
\omterm{def:b2:metric:continuity}{निरंतर} और $\varepsilon_1 \to 0$; उसके बाद से दूसरा $\norm k
\leq C\norm h $ (सीमाबद्ध रेखीय मानचित्र और लघु अवधि) और
$\varepsilon_2(k) \to 0$ जैसा $ h \to 0$.
\end{proof}

\begin{example}[रेडियल कार्य, एक बार और सभी के लिए]\label{ex:b2:diffcalc:radial}
होने देना $ r(x) = \norm x_2$ पर $\R^n\setminus\{0\}$ और $ f = g \circ
r $ साथ $ g $ ए $ C^1$ एक चर का कार्य. पहला, $ r $ है
\omterm{def:b2:diffcalc:differential}{विभेदक} से दूर $0$: से $ r^2 = \sum x_i^2$,
\[
\frac{\partial r}{\partial x_i} = \frac{x_i}{r},
\qquad\text{i.e.}\qquad
\nabla r(x) = \frac{x}{\norm x} ,
\]
यूनिट रेडियल वेक्टर (\omterm{def:b2:diffcalc:differential}{अंतर}) $ r^2$ और विभाजित करें --- या
श्रृंखला नियम लागू करें $\sqrt{\cdot}$). फिर शृंखला नियम
देता है, प्रत्येक रेडियल फ़ंक्शन के लिए,
\[
\nabla f(x) = g'\bigl(\norm x\bigr)\,\frac{x}{\norm x} .
\]
काम किया गया उदाहरण: $ g(r) = \frac1r $ पैदावार $\nabla\frac{1}{\norm
x} = -\frac{x}{\norm x^3}$, का व्युत्क्रम-वर्ग क्षेत्र
गुरुत्वाकर्षण और इलेक्ट्रोस्टैटिक्स --- दिशा रेडियल, परिमाण
$\frac{1}{\norm x^2}$. समापन \omterm{def:b2:diffcalc:differential}{अंतर}्दृष्टि: रेडियल के ग्रेडिएंट्स
फ़ंक्शन रेडियल हैं क्योंकि स्तर सेट गोले हैं और
ग्रेडिएंट लेवल सेट के लिए ऑर्थोगोनल है; पर $ x = 0$, इसके विपरीत, $ r $ है \emph{नहीं} \omterm{def:b2:diffcalc:differential}{विभेदक} (कोई उम्मीदवार नहीं
रेखीय मानचित्र मेल खाता है $\norm h $ सभी दिशाओं से) --- चिकनी
रेडियल प्रोफाइल की जरूरत है $ g'(0) = 0$ मूल को पार करने के लिए
शालीनता से।
\end{example}

\begin{theorem}[माध्य मूल्य असमानता]\label{thm:b2:diffcalc:mvi}
होने देना $ f $ होना \omterm{def:b2:diffcalc:differential}{विभेदक} पर $ U $ और खंड चलो $\intcc{a}{b}
= \{a + t(b-a)\}$ रिहायश $ U $. तब
\[
\norm{f(b) - f(a)} \leq \norm{b - a}\,
\sup_{x \in \intcc{a}{b}} \vertiii{\dd f_x} .
\]
विशेष रूप से, ए \omterm{def:b2:diffcalc:differential}{विभेदक} शून्य के साथ मानचित्र \omterm{def:b2:diffcalc:differential}{अंतर} एक पर
\emph{\omterm{def:b2:metric:connected}{जुड़े हुए}} \omterm{def:b2:metric:topology}{खुला} सेट स्थिर है.
\end{theorem}

\begin{proof}
समारोह $\varphi(t) = f(a + t(b-a))$ है \omterm{def:b2:diffcalc:differential}{विभेदक} पर
$\intcc{0}{1}$ साथ $\varphi'(t) = \dd f_{a + t(b-a)}(b - a)$
(श्रृंखला नियम), का आदर्श $\leq M\norm{b-a}$ साथ $ M $ प्रदर्शित
सुपर. के लिए $\R $-मूल्यांकित $ f $ एक-चर माध्य मान असमानता
निष्कर्ष; वेक्टर मानों के लिए इसे लागू करें $ t \mapsto \langle u,
\varphi(t)\rangle $ साथ $ u $ यूनिट वेक्टर साथ में $ f(b) - f(a)$.
स्थिरता: स्थानीय रूप से स्थिर (गेंदों में खंड) प्लस संयुक्तता
(वह सेट जहां $ f $ दिए गए मान के बराबर है \omterm{def:b2:metric:topology}{खुला} और बंद:
\cref{ch:b2:metric}).
\end{proof}

\begin{example}[एमवीआई से एक लिप्सचिट्ज़ स्थिरांक]\label{ex:b2:diffcalc:lipschitz}
है $ f(x, y) = \sin x\,\sin y $ \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} पर $\R^2$, और साथ में
कौन सा स्थिरांक? इसका ढाल है $\nabla f = (\cos x\sin y,\
\sin x\cos y)$, चुकता का आदर्श
\[
\cos^2x\sin^2y + \sin^2x\cos^2y
\leq \sin^2 y + \cos^2y\cdot 1 = 1
\]
(अवश्यंभावी $\cos^2x $ और $\sin^2x $ द्वारा $1$ अलग से), तो
$\vertiii{\dd f_{(x,y)}} = \norm{\nabla f} \leq 1$ हर जगह,
और \cref{thm:b2:diffcalc:mvi} किन्हीं दो के बीच के खंड पर
अंक देता है
\[
\abs{f(b) - f(a)} \leq \norm{b - a}_2 :
\]
$ f $ है $1$-लिप्सचिट्ज़, और स्थिरांक तेज है (के पास
उत्पत्ति, $ f(x, \tfrac\pi2) = \sin x $ ढलान है $1$). समापन
\omterm{def:b2:diffcalc:differential}{अंतर}्दृष्टि: औसत मूल्य असमानता परिवर्तित होती है \omterm{def:b2:funcseq:def}{बिंदुवार} बाध्य
पर \omterm{def:b2:diffcalc:differential}{अंतर} के वैश्विक मापांक में निरंतरता ---
के लिए मानक मार्ग \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} हर आयाम में अनुमान,
और अंदर इंजन \cref{exo:b2:diffcalc:12}.
\end{example}

\section{दूसरा व्युत्पन्न}

\begin{theorem}[श्वार्ज]\label{thm:b2:diffcalc:schwarz}
अगर $ f $ है $ C^2$ पर $ U $ (सभी दूसरे भाग मौजूद हैं और हैं
\omterm{def:b2:metric:continuity}{निरंतर}), फिर सभी के लिए $ i, j $:\index{श्वार्ज का प्रमेय}
\[
\frac{\partial^2 f}{\partial x_i\,\partial x_j}
= \frac{\partial^2 f}{\partial x_j\,\partial x_i} .
\]
\end{theorem}

\begin{proof}
दो चर पर्याप्त हैं ($ x = x_i $, $ y = x_j $, अन्य जमे हुए)।
दूसरे \omterm{def:b2:diffcalc:differential}{अंतर} पर विचार करें
\[
\Delta(h) = f(a + h, b + h) - f(a + h, b) - f(a, b + h) + f(a,b) .
\]
हल करना $ h $ और सेट करें $\varphi(x) = f(x, b+h) - f(x, b)$: तब $\Delta(h)
= \varphi(a + h) - \varphi(a)$, और माध्य के दो अनुप्रयोग
मूल्य प्रमेय दे
\[
\Delta(h) = h\,\varphi'(\xi)
= h\Bigl(\frac{\partial f}{\partial x}(\xi, b+h) -
\frac{\partial f}{\partial x}(\xi, b)\Bigr)
= h^2\,\frac{\partial^2 f}{\partial y\,\partial x}(\xi, \eta),
\]
साथ $\xi \in \intoo{a}{a+h}$, $\eta \in \intoo{b}{b+h}$. द्वारा
निरंतरता, $\frac{\Delta(h)}{h^2} \to \frac{\partial^2 f}{\partial
y\partial x}(a, b)$ जैसा $ h \to 0$. के साथ भी यही गणना
आदान-प्रदान किए गए चर की भूमिकाएँ (पहले दूसरे चर को फ़्रीज़ करें)
देता है $\frac{\Delta(h)}{h^2} \to \frac{\partial^2 f}{\partial x
\partial y}(a,b)$: एक ही मात्रा की दो सीमाएँ मेल खाती हैं।
\end{proof}

\begin{example}[क्यों $ C^2$ आवश्यक है: पीनो का प्रति उदाहरण]\label{ex:b2:diffcalc:peano}
होने देना $ f(x, y) = \dfrac{xy(x^2 - y^2)}{x^2 + y^2}$, $ f(0,0) = 0$.
मूल से दूर $ f $ है $ C^\infty $; सबसे पहले मूल में
और दूसरा भाग मौजूद है, लेकिन मिश्रित वाले असहमत हैं। गणना करें
अक्षों के अनुदिश: $ f(x, 0) = f(0, y) = 0$, और के लिए $ y \neq 0$,
\[
\frac{\partial f}{\partial x}(0, y)
= \lim_{x\to0}\frac{f(x,y)}{x}
= \frac{y(0 - y^2)}{y^2} = -y ,
\qquad\text{symmetrically}\qquad
\frac{\partial f}{\partial y}(x, 0) = x .
\]
इस तरह
\[
\frac{\partial^2 f}{\partial y\,\partial x}(0,0)
= \frac{\dd}{\dd y}\Bigl[\frac{\partial f}{\partial
x}(0,y)\Bigr]_{y=0} = -1,
\qquad
\frac{\partial^2 f}{\partial x\,\partial y}(0,0) = +1 :
\]
दो मिश्रित भाग मौजूद हैं और भिन्न हैं। के साथ कोई विरोधाभास नहीं
\cref{thm:b2:diffcalc:schwarz}: का दूसरा भाग $ f $ नहीं हैं \omterm{def:b2:metric:continuity}{निरंतर} पर $0$ (साथ में परीक्षण करें $ y = tx $). समापन \omterm{def:b2:diffcalc:differential}{अंतर}्दृष्टि:
श्वार्ज़ प्रमेय एक वास्तविक प्रमेय है \emph{निरंतरता},
औपचारिक पहचान नहीं --- और परिकल्पना ``$ C^2$'' में
टेलर--यंग नीचे असली काम कर रहा है।
\end{example}

\begin{theorem}[टेलर - क्रम 2 पर युवा]\label{thm:b2:diffcalc:taylor}
होने देना $ f \colon U \to \R $ होना $ C^2$ और $ a \in U $. फिर ऐसे $ h \to 0$,
\[
f(a + h) = f(a) + \langle\nabla f(a), h\rangle
+ \frac12\, \langle H_a h,\, h\rangle + o\bigl(\norm h^2\bigr),
\]
कहाँ $ H_a = \bigl(\frac{\partial^2 f}{\partial x_i\partial
x_j}(a)\bigr)$ है (\omterm{def:b2:quadratic:adjoint}{सममित}, श्वार्ज द्वारा) \emph{हेसियन
मैट्रिक्स}\index{हेसियन
मैट्रिक्स}.
\end{theorem}

\begin{proof}
एक-चर टेलर--यंग प्रमेय (वर्ष 1 खंड) को लागू करें
$\varphi(t) = f(a + th)$ पर $\intcc{0}{1}$: शृंखला नियम द्वारा,
$\varphi'(t) = \langle \nabla f(a + th), h\rangle $ और $\varphi''(t)
= \langle H_{a+th}h, h\rangle $, दोनों \omterm{def:b2:metric:continuity}{निरंतर} में $ t $. तब
$\varphi(1) = \varphi(0) + \varphi'(0) + \frac12\varphi''(\theta)$
(टेलर-लैग्रेंज) के साथ $\theta \in \intoo{0}{1}$, और निरंतरता
दूसरे भाग का रूपांतरण होता है $\varphi''(\theta) =
\varphi''(0) + o(1)\cdot\norm h^2$-समान रूप से: प्रदर्शित
विस्तार.
\end{proof}

\begin{example}[एक विस्तार दो तरह से]\label{ex:b2:diffcalc:taylortwoways}
बढ़ाना $ f(x, y) = \eu^x\cos y $ ऑर्डर करने के लिए मूल स्थान पर $2$.
\emph{एक-चर विस्तार की संरचना द्वारा:}
\[
\eu^x\cos y = \Bigl(1 + x + \frac{x^2}{2} +
o(x^2)\Bigr)\Bigl(1 - \frac{y^2}{2} + o(y^2)\Bigr)
= 1 + x + \frac{x^2 - y^2}{2} + o\bigl(\norm{(x,y)}^2\bigr) .
\]
\emph{आंशिक व्युत्पन्न द्वारा:} $ f_x = \eu^x\cos y $, $ f_y =
-\eu^x\sin y $, इसलिए $\nabla f(0) = (1, 0)$; और $ f_{xx} = f $,
$ f_{yy} = -f $, $ f_{xy} = -\eu^x\sin y $ देना $ H_0 =
\operatorname{diag}(1, -1)$: \cref{thm:b2:diffcalc:taylor}
पुनरुत्पादन करता है $1 + x + \frac12(x^2 - y^2)$. दो गणनाएँ
सहमत हूँ, और रचना मार्ग तेज़ था --- कोई दूसरा नहीं
\omterm{def:b2:series:def}{बिल्कुल} आंशिक. समापन \omterm{def:b2:diffcalc:differential}{अंतर}्दृष्टि: मूल है \emph{नहीं} ए
महत्वपूर्ण बिंदु ($\nabla f \neq 0$), इसलिए अनिश्चित काल के बावजूद
हेसियन घोषित करने के लिए कोई काठी नहीं है: रैखिक शब्द नियम,
और दूसरे क्रम का परीक्षण केवल आलोचनात्मक स्तर पर ही बोलता है
अंक.
\end{example}

\begin{theorem}[दूसरे क्रम का चरम परीक्षण, सिद्ध हुआ]\label{thm:b2:diffcalc:extrema}
होने देना $ f $ होना $ C^2$ एक महत्वपूर्ण बिंदु के निकट $ a $ ($\nabla f(a) = 0$),
हेस्सियन के साथ $ H = H_a $.
\begin{enumerate}
  \item अगर $ H $ सकारात्मक निश्चित है, $ a $ एक सख्त स्थानीय न्यूनतम है
        (नकारात्मक निश्चित: अधिकतम)।
  \item अगर $ H $ है eigenvalues दोनों चिन्हों का, $ a $ एक काठी है: नहीं
        चरम.
  \item अगर $ H $ एकवचन (और अर्धनिश्चित) है, कोई निष्कर्ष नहीं।
\end{enumerate}
वर्ष 1 ``$ rt - s^2$''परीक्षण का मामला है $ n = 2$: $\det H = rt -
s^2$, $\operatorname{tr}$-हस्ताक्षर पढ़ें $ r $.
\end{theorem}

\begin{proof}
वर्णक्रमीय प्रमेय द्वारा (\cref{thm:b2:quadratic:spectral}), $ H $'एस
द्विघात रूप अपने चरम के बीच निचोड़ा हुआ है eigenvalues:
$\lambda_{\min}\norm h^2 \leq \langle Hh, h\rangle \leq
\lambda_{\max}\norm h^2$.

(1) यदि $\lambda_{\min} > 0$: टेलर--यंग देता है
\[
f(a + h) - f(a) \geq \frac{\lambda_{\min}}{2}\norm h^2 -
o(\norm h^2) > 0
\]
छोटे के लिए $ h \neq 0$: सख्त स्थानीय न्यूनतम.

(2) साथ में ए आइजन्वेक्टर $ v_+$ साथ $\lambda_+ > 0$: $ f(a + tv_+) -
f(a) = \frac{\lambda_+}{2}t^2 + o(t^2) > 0$ छोटे के लिए $ t $; साथ में
$ v_-$ साथ $\lambda_- < 0$ \omterm{def:b2:diffcalc:differential}{अंतर} नकारात्मक है: दोनों संकेत
हर मोहल्ले में होता है.

(3) $ f(x,y) = x^2 + y^4$, $ x^2 - y^4$, $ x^2 + y^3$ वही साझा करें
एकवचन अर्धनिश्चित हेस्सियन पर $0$ तीन अलग के साथ
व्यवहार.
\end{proof}

\begin{example}[एक संपूर्ण वर्गीकरण, वैश्विक शामिल]\label{ex:b2:diffcalc:quarticrun}
के सभी चरम को वर्गीकृत करें $ f(x, y) = x^4 + y^4 - 4xy $ पर $\R^2$.
\emph{महत्वपूर्ण बिंदु:} $\nabla f = (4x^3 - 4y,\ 4y^3 - 4x) =
0$ देता है $ y = x^3$ और $ x = y^3 = x^9$, इसलिए $ x(x^8 - 1) = 0$:
असली समाधान हैं $(0,0)$, $(1,1)$, $(-1,-1)$.
\emph{हेस्सियन:} $ H = \begin{pmatrix} 12x^2 & -4\\ -4 &
12y^2\end{pmatrix}$. पर $(\pm1, \pm1)$: $\begin{pmatrix} 12 &
-4\\ -4 & 12\end{pmatrix}$, eigenvalues $8$ और $16$: सकारात्मक
निश्चित, सख्त स्थानीय न्यूनतम के साथ $ f = -2$. पर $(0,0)$:
$\begin{pmatrix} 0 & -4\\ -4 & 0\end{pmatrix}$, eigenvalues
$\pm4$: एक काठी. \emph{वैश्विकता:} से $2\abs{xy} \leq x^2 +
y^2$,
\[
f(x, y) \geq x^4 + y^4 - 2(x^2 + y^2)
= (x^2 - 1)^2 + (y^2 - 1)^2 + x^2 + y^2 - 2
\xrightarrow[\norm{(x,y)}\to\infty]{} +\infty :
\]
$ f $ जबरदस्ती है, इसलिए यह वैश्विक न्यूनतम प्राप्त करता है
(सघनता सबलेवल सेट का), आवश्यक रूप से एक महत्वपूर्ण स्तर पर
बिंदु: मूल्य $-2$, दोनों पर $(1,1)$ और $(-1,-1)$, है
वैश्विक न्यूनतम; कोई अधिकतम नहीं है ($ f $ ऊपर असीमित)।
समापन \omterm{def:b2:diffcalc:differential}{अंतर}्दृष्टि: स्थानीय परीक्षण उम्मीदवारों को वर्गीकृत करता है, लेकिन
केवल विकास का तर्क ``स्थानीय'' को ``वैश्विक'' में बदल देता है ---
इस पुस्तक में प्रत्येक अनुकूलन प्रमाण का दो-चरणीय पैटर्न।
\end{example}

\begin{method}[की चरम सीमा का वर्गीकरण $ f \colon \R^n \to \R $]\label{met:b2:diffcalc:extrema}
\begin{enumerate}
  \item हल करना $\nabla f = 0$ (सभी महत्वपूर्ण बिंदु; एक डोमेन पर
        सीमा के साथ, सीमा को अलग से मानें जैसे कि
        \cref{exo:b2:diffcalc:7}).
  \item प्रत्येक महत्वपूर्ण बिंदु पर, हेस्सियन और इसकी गणना करें
        \omterm{def:b2:reduction:eigen}{eigenvalue} संकेत --- आयाम में $2$, अभी $\det H $
        और $\operatorname{tr} H $: $\det < 0$ काठी; $\det >
        0$ चरम, के संकेत द्वारा दिए गए प्रकार का
        ट्रेस; $\det = 0$: परीक्षण मौन है, अध्ययन करें $ f $
        वक्रों के साथ.
  \item वैश्विक कथनों के लिए, a \omterm{def:b2:quadratic:adjoint}{जोड़}ें सघनता या जबरदस्ती
        तर्क ($ f \to +\infty $ अनंत पर, या ए \omterm{def:b2:metric:compact}{कॉम्पैक्ट}
        बाधा सेट), फिर महत्वपूर्ण मानों की तुलना करें।
\end{enumerate}
\end{method}

\section{व्युत्क्रम फलन प्रमेय}

\begin{theorem}[व्युत्क्रम फलन प्रमेय]\label{thm:b2:diffcalc:inverse}
होने देना $ f \colon U \to \R^n $ होना $ C^1$ और $ a \in U $ साथ $\dd f_a $
\emph{उलटी}. फिर हैं \omterm{def:b2:metric:topology}{खुला} पड़ोस $ V \ni a $, $ W
\ni f(a)$ ऐसा है कि $ f \colon V \to W $ के साथ एक आक्षेप है $ C^1$
उलटा, और\index{व्युत्क्रम फलन प्रमेय}
\[
\dd (f^{-1})_{f(x)} = (\dd f_x)^{-1} \qquad (x \in V).
\]
\end{theorem}
\admitted

\begin{remark}[यह सत्य क्यों है: निश्चित-बिंदु रणनीति]
हल $ f(x) = y $ पास में $ a $ निश्चित-बिंदु समीकरण के रूप में पुनः लिखता है
$ x = x + \dd f_a^{-1}\bigl(y - f(x)\bigr) =: \Phi_y(x)$; वो नक्शा
$\Phi_y $ है \omterm{def:b2:diffcalc:differential}{अंतर} $\mathrm{id} - \dd f_a^{-1}\dd f_x $,
पास में छोटा $ a $ द्वारा निरंतरता का $\dd f $, इसलिए $\Phi_y $ एक है
एक छोटी सी बंद गेंद और बानाच निश्चित बिंदु पर संकुचन
प्रमेय (\cref{thm:b2:metric:banach}) अद्वितीय स्थानीयता प्रदान करता है
समाधान $ x = f^{-1}(y)$. निरंतरता और भिन्नता का
इसके बाद संकुचन में अनुमान से उलटा अनुसरण करें। पूर्ण
बहीखाता पद्धति वर्ष 3 में की जाती है; रणनीति --- और
कथन --- अब से स्वतंत्र रूप से उपयोग किया जाता है। साथी
\emph{निहित कार्य प्रमेय} (समाधान $ F(x, y) = 0$ के लिए $ y(x)$
कब $\frac{\partial F}{\partial y}$ उलटा है) का अनुसरण करता है
प्रमेय को लागू करना $(x, y) \mapsto (x, F(x,y))$.
\end{remark}

\begin{example}[ध्रुवीय निर्देशांक]\label{ex:b2:diffcalc:polar}
$\Phi(r, \theta) = (r\cos\theta, r\sin\theta)$ जैकोबियन है
\[
J_\Phi = \begin{pmatrix} \cos\theta & -r\sin\theta\\ \sin\theta &
r\cos\theta \end{pmatrix},
\qquad \det J_\Phi = r :
\]
के लिए उलटा $ r \neq 0$, इसलिए $\Phi $ स्थानीय है $ C^1$
मूल से दूर भिन्नता--- ``परिवर्तन'' करने का लाइसेंस
ध्रुवीय निर्देशांक'', के एकाधिक अभिन्नों के लिए नवीनीकृत किया गया
\cref{ch:b2:multint}.
\end{example}

\begin{example}[हर जगह स्थानीय, कहीं भी वैश्विक नहीं]\label{ex:b2:diffcalc:localnotglobal}
होने देना $ f(x, y) = \bigl(\eu^x\cos y,\ \eu^x\sin y\bigr)$ पर
$\R^2$. यह जैकोबियन है,
\[
J_f = \begin{pmatrix}
\eu^x\cos y & -\eu^x\sin y\\
\eu^x\sin y & \eu^x\cos y
\end{pmatrix},
\qquad
\det J_f = \eu^{2x} > 0 ,
\]
कभी लुप्त नहीं होता: द्वारा \cref{thm:b2:diffcalc:inverse}, $ f $ एक है
स्थानीय $ C^1$ भिन्नतावाद पर \emph{प्रत्येक} विमान का बिंदु.
अभी तक $ f $ इंजेक्शन से बहुत दूर है: $ f(x, y + 2\pi) = f(x, y)$, तो
प्रत्येक मान को अनंत बार लिया जाता है; और यह विशेषण नहीं है
या तो, तब से $\norm{f(x, y)} = \eu^x > 0$ मूल से चूक जाता है.
समापन \omterm{def:b2:diffcalc:differential}{अंतर}्दृष्टि: व्युत्क्रम फलन प्रमेय अघुलनशील है
\emph{स्थानीय} --- प्रत्येक की व्युत्क्रमणीयता $\dd f_a $ उपज ए
स्थानीय व्युत्क्रमों का पैचवर्क जिसे एक में एकत्रित करने की आवश्यकता नहीं है।
(संमिश्र संख्याओं को जानने वाले पाठक पहचान लेंगे $ z \mapsto
\eu^z $; पैचवर्क लघुगणक शाखाओं का परिवार है।)
तुलना करें \cref{exo:b2:diffcalc:12}, जहां एक मात्रात्मक वैश्विक
परिकल्पना एक वैश्विक व्युत्क्रम को बाध्य करती है।
\end{example}

\begin{remark}[सामान्य ख़तरे]
\emph{(i) दिशात्मक व्युत्पन्न सस्ते हैं, भिन्नता नहीं हैं:} सभी दिशात्मक व्युत्पन्न मौजूद हो सकते हैं --- और विफल भी हो सकते हैं
दिशा पर रैखिक रूप से निर्भर होना --- बिना
भिन्नता (\cref{exo:b2:diffcalc:2}); केवल $ C^1$
कसौटी
(\cref{thm:b2:diffcalc:c1}) आंशिक को ए में अपग्रेड करता है
\omterm{def:b2:diffcalc:differential}{अंतर}. \emph{(ii) क्रिटिकल का मतलब अति नहीं है:}
काठी (\cref{ex:b2:diffcalc:quarticrun}) और चुप
एकवचन मामला (\cref{thm:b2:diffcalc:extrema}~(3)) दोनों छिपते हैं
पीछे $\nabla f = 0$. \emph{(iii) कोई सदिश-मान माध्य मान नहीं
समानता:} केवल \emph{असमानता} का
\cref{thm:b2:diffcalc:mvi} जीवित रहता है
(\cref{exo:b2:diffcalc:9}); कभी मत लिखो $ f(b) - f(a) = \dd
f_c(b-a)$ के लिए $ f $ मूल्यों के साथ $\R^m $, $ m \geq 2$.
\emph{(iv) स्थानीय व्युत्क्रमणता अंतःक्षेपण नहीं है:}
\cref{ex:b2:diffcalc:localnotglobal}. \emph{(v) ढाल
आंतरिक उत्पाद से संबंधित है:} $\nabla f $ वेक्टर है
प्रतिनिधित्व करना $\dd f_a $ चुने हुए आंतरिक उत्पाद में; बदलो
उत्पाद (जैसा कि द्विघात-रूपों के भारित उदाहरण में है
अध्याय) और ग्रेडिएंट घूमता है, जबकि \omterm{def:b2:diffcalc:differential}{अंतर} ---
आंतरिक वस्तु --- हिलती नहीं है।
\end{remark}

\begin{remark}[इसका उपयोग कहां किया जाता है]
इस अध्याय का डाउनस्ट्रीम सब कुछ डिफरेंशियल कैलकुलस है
लागू: अंतर-समीकरण अध्याय प्रवाह को रैखिक बनाता है और
लिउविले का उपयोग करता है सिद्ध सूत्र (इस अध्याय में सिद्ध किया गया है
सप्ताहांत समस्या); वक्रों और सतहों पर अध्याय अध्ययन करते हैं
\omterm{def:b2:diffcalc:differential}{अंतर}्निहित फ़ंक्शन के माध्यम से स्तर सेट और पैरामीट्रिज़ेशन
प्रमेय; एकाधिक अभिन्न अंग जैकोबियन के माध्यम से चर बदलते हैं।
सप्ताहांत की समस्या कैलकुलस विकसित करती है \emph{की जगह पर
मैट्रिसेस स्वयं} --- \omterm{def:b2:diffcalc:differential}{अंतर} की सिद्ध, का
उलटा, \omterm{ex:b2:nvs:matrixexp}{मैट्रिक्स घातांक}, और ओर्थोगोनल समूह एक के रूप में
स्मूथ लेवल सेट --- वर्ष 2 की छाया वर्ष 3 की क्या है
वॉल्यूम मैनिफोल्ड्स और लाई समूहों के रूप में औपचारिक होता है।
\end{remark}

\begin{exercise}[$\star $]\label{exo:b2:diffcalc:1}
के \omterm{def:b2:diffcalc:differential}{जैकोबियन मैट्रिक्स} की गणना करें $ f(x,y) = (x^2 - y^2,\, 2xy)$ और
का $\Phi(r,\theta,z) = (r\cos\theta, r\sin\theta, z)$; कहां हैं
द भिन्नता उलटा?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:diffcalc:2}
होने देना $ f(x,y) = \frac{x^3}{x^2 + y^2}$ ($ f(0,0) = 0$). यह सब साबित करो
के दिशात्मक व्युत्पन्न $ f $ पर $0$ मौजूद है, लेकिन वह $ f $ क्या नहीं है
\omterm{def:b2:diffcalc:differential}{विभेदक} पर $0$ \emph{(वो नक्शा $ v \mapsto $ दिशात्मक
व्युत्पन्न रैखिक नहीं है)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:diffcalc:3}
के महत्वपूर्ण बिंदुओं को खोजें और वर्गीकृत करें $ f(x, y) = x^3 + y^3 -
3xy $ का उपयोग करते हुए \cref{thm:b2:diffcalc:extrema}, और का $ g(x,y) = x^4 +
y^4 - 2(x - y)^2$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffcalc:4}
होने देना $ f \colon \R^n \to \R $ होना $ C^1$ और \emph{डिग्री का सजातीय
$ p $}: $ f(tx) = t^pf(x)$ के लिए $ t > 0$. यूलर की पहचान सिद्ध करें
\[
\langle \nabla f(x), x\rangle = p\,f(x) ,
\]
और इसके विपरीत $ C^1$ पर कार्य करता है $\R^n\setminus\{0\}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffcalc:5}
होने देना $ A $ होना \omterm{def:b2:quadratic:adjoint}{सममित} और $ f(x) = \frac12\langle Ax, x\rangle -
\langle b, x\rangle $. गणना करना $\nabla f $ और $ H_f $; कब है $ f $
उत्तल? यह मानते हुए $ A $ सकारात्मक निश्चित, वह दिखाओ $ f $ एक अनोखा है
के समाधान पर वैश्विक न्यूनतम $ Ax = b $ --- क्रमिक अवतरण
रायसन डी'\^आदि.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffcalc:6}
(लैग्रेंज मल्टीप्लायर, एक बाधा, हाथ से सिद्ध) चलो $ f, g $ होना
$ C^1$ पर $\R^2$, और मान लीजिए $ f $ प्राप्त करता है, पर $ a $, एक स्थानीय चरम
निर्धारित स्तर पर $\{g = 0\}$, साथ $\nabla g(a) \neq 0$. साबित करें कि
$\nabla f(a) = \lambda\nabla g(a)$ कुछ के लिए $\lambda $.
\emph{(पास में निर्धारित स्तर को पैरामीट्रिज करें $ a $ \omterm{def:b2:diffcalc:differential}{अंतर}्निहित कार्य द्वारा
प्रमेय और \omterm{def:b2:diffcalc:differential}{अंतर} बताइए $ t \mapsto f(\gamma(t))$.)} आवेदन:
की चरम सीमा $ f(x,y) = xy $ वृत्त पर $ x^2 + y^2 = 1$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffcalc:7}
की चरम सीमा निर्धारित करें $ f(x, y) = x^2 + y^2 - xy + x - y $ पर
$\R^2$, तो बंद त्रिकोण पर इसका अधिकतम और न्यूनतम
शिखर $(0,0)$, $(1,0)$, $(0,1)$ \emph{(आंतरिक महत्वपूर्ण बिंदु,
फिर तीन किनारे, फिर शीर्ष)}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:diffcalc:8}
होने देना $ f \colon \R^2 \to \R^2$, $ f(x, y) = (x + y^2,\; y + x^2)$.
बताते हैं कि $ f $ निकट एक स्थानीय भिन्नता है $0$, गणना करें
$\dd(f^{-1})_{(0,0)}$, और सबसे बड़ा खोजें $ r $ ऐसा है कि $\dd f $
गेंद पर उलटा है $\norm{(x,y)}_2 < r $ \emph{(गणना करें
$\det J_f $)}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:diffcalc:9}
(रोल विफल रहता है, माध्य मान जीवित रहता है) दीजिए $ f \colon \R \to \R^2$,
$ C^1$, साथ $ f(0) = f(2\pi)$ लेकिन $ f'(t) \neq 0$ सभी के लिए $ t $
(कोई वेक्टर-मूल्यवान रोले नहीं)। फिर अपने उदाहरण पर सत्यापित करें
माध्य मान \emph{असमानता} का \cref{thm:b2:diffcalc:mvi}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:diffcalc:10}
की गणना करें \omterm{def:b2:diffcalc:differential}{अंतर} और की ढाल $ f(x) = \norm
x_2^2$ और का $ g(x) = \langle Ax, x\rangle $ पर $\R^n $ ($ A $ ए
वर्ग मैट्रिक्स, ग्रहण नहीं किया गया \omterm{def:b2:quadratic:adjoint}{सममित}), और प्रत्येक का हेसियन।
जिसके लिए $ A $ है $ g $ उत्तल?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:diffcalc:11}
होने देना $ f \colon \R^n \to \R $ होना $ C^2$. साबित करें कि $ f $ उत्तल है यदि
और केवल यदि यह हेस्सियन है $ H_x $ प्रत्येक पर सकारात्मक अर्धनिश्चित है
$ x $ \emph{(एक चर को कम करें: $ t \mapsto f(a + t(b-a))$;
टेलर-लैग्रेंज का एक दिशा में और इसके विपरीत उपयोग करें
मूल्यांकन करें $\varphi''$)}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:diffcalc:12}
(एक वैश्विक व्युत्क्रम प्रमेय) चलो $ g \colon \R^n \to \R^n $ होना $ C^1$
साथ $\vertiii{\dd g_x} \leq k < 1$ सभी के लिए $ x $, और $ f =
\mathrm{id} + g $.
\begin{enumerate}
  \item दिखाओ $\norm{f(x) - f(y)} \geq (1 - k)\norm{x - y}$: $ f $
        इंजेक्शन है, साथ में \omterm{def:b2:metric:continuity}{निरंतर} इसकी छवि पर उलटा.
  \item इसे प्रत्येक के लिए दिखाएँ $ y \in \R^n $ वो नक्शा $ x \mapsto y -
        g(x)$ का संकुचन है \omterm{def:b2:metric:complete}{पूरा} \omterm{def:b2:diffcalc:differential}{अंतर}िक्ष $\R^n $,
        और बानाच निश्चित बिंदु प्रमेय द्वारा निष्कर्ष निकालें
        (\cref{thm:b2:metric:banach}) वह $ f $ विशेषण है.
  \item समाप्त करें कि $ f $ का आक्षेप है $\R^n $ साथ
        $(1-k)^{-1}$-लिप्सचिट्ज़ उलटा --- ए \emph{वैश्विक}
        के \omterm{def:b2:nvs:equivalent}{समकक्ष} \cref{thm:b2:diffcalc:inverse} (जो,
        इसके विपरीत, पूरी तरह से स्थानीय है)।
\end{enumerate}
\end{exercise}

\section{समस्या: आव्यूहों की गणना --- जैकोबी,
घातांकीय, और ऑर्थोगोनल समूह}

\begin{problem}\label{pb:b2:diffcalc:1}
डिफरेंशियल कैलकुलस के लिए सबसे साफ खेल का मैदान जगह है
$\mathcal M_n(\R) \simeq \R^{n^2}$ स्वयं: इसके सबसे प्राकृतिक मानचित्र
--- उत्पाद, उलटा, सिद्ध, घातीय --- है
भिन्नता आकर्षक लालित्य का. यह समस्या उनकी गणना करती है
सब: द \emph{न्यूमैन श्रृंखला}, द \omterm{def:b2:diffcalc:differential}{अंतर} उलटे का,
\emph{जैकोबी का सूत्र}\index{जैकोबी का सूत्र} के लिए
सिद्ध साथ \emph{लिउविले का सूत्र} लाभांश के रूप में,
\emph{मैट्रिक्स घातांक}\index{मैट्रिक्स घातांक} साथ
$\det\eu^A = \eu^{\operatorname{tr}A}$, और अंत में
ऑर्थोगोनल समूह $ O_n $ के साथ एक सहज स्तर सेट के रूप में
स्पर्शरेखा स्थान के रूप में एंटीसिमेट्रिक मैट्रिक्स --- \omterm{def:b2:diffcalc:differential}{अंतर}
भ्रूण में ज्यामिति. लगातार, $\vertiii\cdot $ है ऑपरेटर
आदर्श के अधीन $\norm\cdot_2$, और $\langle X, Y\rangle =
\operatorname{tr}(X^{\mathsf T}Y)$ फ्रोबेनियस आंतरिक उत्पाद।

\medskip
\noindent\textbf{Part I --- The Neumann series.}
\begin{enumerate}
  \item उपगुणात्मकता सिद्ध करें, $\vertiii{AB} \leq
        \vertiii A\,\vertiii B $, और उस बहुपद को व्युत्पन्न करें
        के मानचित्र $ A $ (मैट्रिक्स उत्पाद, सिद्ध, ट्रेस) हैं
        \omterm{def:b2:metric:continuity}{निरंतर} पर $\mathcal M_n(\R)$.
  \item के लिए $\vertiii X < 1$, बताते हैं कि $\sum_{k\geq0}X^k $
        \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:series:def}{बिल्कुल} में $\mathcal M_n(\R)$
        (\cref{thm:b2:nvs:absoluteconvergence}), कि यह योग है
        है $(I - X)^{-1}$, ओर वो
        \[
        \vertiii{(I - X)^{-1}} \leq \frac{1}{1 - \vertiii X},
        \qquad
        (I - X)^{-1} = I + X + O\bigl(\vertiii X^2\bigr) .
        \]
  \item उसका निष्कर्ष निकालो $ GL_n(\R)$ है \emph{\omterm{def:b2:metric:topology}{खुला}}: अगर $ A $ है
        उलटा और $\vertiii H <
        \frac{1}{\vertiii{A^{-1}}}$, तब $ A + H $ है
        उलटा। उसका भी निष्कर्ष निकालो $ GL_n(\R)$ है \emph{घना}
        में $\mathcal M_n(\R)$ \emph{(परेशान $ A $ द्वारा
        $\varepsilon I $: $\det(A + \varepsilon I)$ एक अशून्य है
        बहुपद में $\varepsilon $)}.
  \item व्युत्क्रम मानचित्र दिखाएँ $\Phi(A) = A^{-1}$ है
        \omterm{def:b2:metric:continuity}{निरंतर} पर $ GL_n(\R)$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- First भिन्नता.}
\begin{enumerate}[resume]
  \item वह वर्ग मानचित्र दिखाएँ $ A \mapsto A^2$ है
        \omterm{def:b2:diffcalc:differential}{विभेदक} साथ \omterm{def:b2:diffcalc:differential}{अंतर} $ H \mapsto AH + HA $,
        और अधिक \omterm{def:b2:funcseq:series}{सामान्यतः} वह $ A \mapsto A^k $ है
        \omterm{def:b2:diffcalc:differential}{अंतर} $ H \mapsto \sum_{i=0}^{k-1}
        A^iHA^{k-1-i}$. कोई क्यों कर सकता है \emph{नहीं} लिखना $ kA^{k-1}H $
        सामान्य तौर पर?
  \item साबित करें कि $\Phi(A) = A^{-1}$ है \omterm{def:b2:diffcalc:differential}{विभेदक} पर
        $ GL_n(\R)$ साथ
        \[
        \dd\Phi_A(H) = -A^{-1}HA^{-1}
        \]
        \emph{(लिखना $(A + H)^{-1} = (I + A^{-1}H)^{-1}A^{-1}$
        और प्रश्न 2 द्वारा विस्तार करें)}. विरुद्ध सूत्र की जाँच करें
        अदिश मामला $ n = 1$.
  \item एक के लिए $ C^1$ वक्र $ t \mapsto A(t) \in GL_n(\R)$, निष्कर्ष निकालना
        $\bigl(A(t)^{-1}\bigr)' = -A^{-1}A'A^{-1}$, और विस्तार करें
        $ t \mapsto (I + tB)^{-1}$ पहले ऑर्डर करने के लिए $ t = 0$.
  \item की गणना करें \omterm{def:b2:diffcalc:differential}{अंतर} का $ f(A) =
        \operatorname{tr}(A^k)$ और इसके ढाल की पहचान करें
        फ्रोबेनियस आंतरिक उत्पाद:
        \[
        \dd f_A(H) = k\operatorname{tr}\bigl(A^{k-1}H\bigr),
        \qquad
        \nabla f(A) = k\,\bigl(A^{k-1}\bigr)^{\mathsf T} .
        \]
  \item के लिए वही प्रश्न $ f(A) = \operatorname{tr}
        (A^{\mathsf T}A) = \norm A_F^2$: \omterm{def:b2:diffcalc:differential}{अंतर},
        ग्रेडिएंट, और (\omterm{def:b2:metric:continuity}{निरंतर}) हेस्सियन; समाप्त करें कि
        $\norm\cdot_F^2$ सख्ती से उत्तल है.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Jacobi's formula.}
\begin{enumerate}[resume]
  \item सिद्ध करना
        \[
        \det(I + H) = 1 + \operatorname{tr}H +
        O\bigl(\vertiii H^2\bigr)
        \]
        \emph{(बढ़ाना $\det(e_1 + h_1, \dots, e_n + h_n)$ द्वारा
        स्तंभों में बहुरेखीयता: कम से कम दो पद
        $ h $-कॉलम हैं $ O(\vertiii H^2)$)}: $\dd(\det)_I =
        \operatorname{tr}$.
  \item के लिए $ A $ उलटा, निष्कर्ष निकालना
        \[
        \dd(\det)_A(H) = \det(A)\,
        \operatorname{tr}\bigl(A^{-1}H\bigr) .
        \]
  \item उसके लिए दिखाओ \emph{प्रत्येक} $ A $ (उलटा या नहीं),
        $\frac{\partial\det}{\partial a_{ij}}(A) = C_{ij}$, द
        $(i,j)$ सहायक कारक \emph{(पंक्ति के साथ लाप्लास विस्तार
        $ i $)}, ताकि सहायक के साथ $\operatorname{adj}A =
        \operatorname{com}(A)^{\mathsf T}$:
        \[
        \dd(\det)_A(H) =
        \operatorname{tr}\bigl(\operatorname{adj}(A)\,H\bigr),
        \qquad
        \nabla(\det)(A) = \operatorname{com}(A) ,
        \]
        पुनर्प्राप्ति प्रश्न 11 कब $ A $ उलटा है
        ($\operatorname{adj}A = \det(A)A^{-1}$). यह है
        \emph{\omterm{pb:b2:diffcalc:1}{जैकोबी का सूत्र}}: $\bigl(\det
        A(t)\bigr)' = \operatorname{tr}\bigl(
        \operatorname{adj}(A(t))\,A'(t)\bigr)$.
  \item (लिउविले का सूत्र) चलो $ A(t)$ एक हो $ C^1$ का वक्र
        रैखिक अवकल समीकरण को संतुष्ट करने वाले आव्यूह
        $ A'(t) = M(t)A(t)$. सिद्ध करना
        \[
        \bigl(\det A(t)\bigr)' =
        \operatorname{tr}\bigl(M(t)\bigr)\,\det A(t),
        \qquad\text{hence}\qquad
        \det A(t) = \det A(0)\,
        \exp\Bigl(\int_0^t\operatorname{tr}M\Bigr)
        \]
        \emph{(उपयोग $\operatorname{adj}(A)\,A = \det(A)I $ और
        ट्रेस का \omterm{def:b2:structures:generated}{चक्रीय} आविष्कार)} --- व्रोन्स्कियन
        पहचान कि अंतर-समीकरण अध्याय होगा
        लगातार उपयोग करें.
  \item बताते हैं कि $ SL_n(\R) = \{\det = 1\}$ एक सहज स्तर है
        सेट: प्रत्येक पर $ A \in SL_n(\R)$  \omterm{def:b2:diffcalc:differential}{अंतर}
        $\dd(\det)_A $ एक है \emph{विशेषण} रेखीय मानचित्र पर
        $\R $ \emph{(इसका मूल्यांकन करें $ H = \frac1nA $)}.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The \omterm{ex:b2:nvs:matrixexp}{मैट्रिक्स घातांक}.}
\begin{enumerate}[resume]
  \item बताते हैं कि $\eu^A = \sum_{k\geq0}\frac{A^k}{k!}$
        \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:series:def}{बिल्कुल} हरएक के लिए $ A $, \omterm{def:b2:funcseq:series}{सामान्यतः} हर पर
        गेंद, साथ $\vertiii{\eu^A} \leq
        \eu^{\vertiii A}$; ओर वो $\eu^A $ निर्भर करता है
        लगातार पर $ A $.
  \item साबित करें कि $ AB = BA $ तात्पर्य $\eu^{A+B} =
        \eu^A\eu^B $ \emph{(\omterm{thm:b2:series:fubini}{कॉची उत्पाद}, द्वारा वैध
        पूर्ण \omterm{def:b2:integration:improper}{अभिसरण})}; उसका निष्कर्ष निकालो $\eu^A $ हमेशा है
        व्युत्क्रमणीय, व्युत्क्रम के साथ $\eu^{-A}$: $\exp $ एमएपीएस
        $\mathcal M_n(\R)$ में $ GL_n(\R)$.
  \item बताते हैं कि $ t \mapsto \eu^{tA}$ है $ C^1$ (वास्तव में
        $ C^\infty $) साथ
        \[
        \frac{\dd}{\dd t}\,\eu^{tA} = A\,\eu^{tA} =
        \eu^{tA}A
        \]
        \emph{(श्रृंखला के पदों को पदों के आधार पर अलग करें
        खंड: व्युत्पन्न श्रृंखला \omterm{def:b2:integration:improper}{अभिसरण} करती है \omterm{def:b2:funcseq:series}{सामान्यतः})}.
  \item पहचान साबित करें
        \[
        \det\bigl(\eu^{A}\bigr) = \eu^{\operatorname{tr}A}
        \]
        \emph{(लिउविले का सूत्र, प्रश्न 13, लागू करें
        $ A(t) = \eu^{tA}$)}. स्वच्छता जांच: $ n = 1$; निलपोटेंट
        $ A $; और ट्रेस-जीरो मैट्रिसेस अंदर आते हैं
        $ SL_n(\R)$.
  \item दिखाओ $\eu^H = I + H + O(\vertiii H^2)$, इसलिए $\exp $ है
        \omterm{def:b2:diffcalc:differential}{विभेदक} पर $0$ साथ $\dd(\exp)_0 =
        \mathrm{id}$; व्युत्क्रम फलन के साथ समाप्त करें
        प्रमेय (\cref{thm:b2:diffcalc:inverse}) वह $\exp $
        एक है $ C^1$ के पड़ोस से भिन्नता $0$
        के एक पड़ोस पर $ I $: प्रत्येक मैट्रिक्स के करीब
        पहचान में एक लघुगणक होता है।
  \item बताते हैं कि $\exp $ एमएपीएस \omterm{def:b2:quadratic:adjoint}{सममित} मैट्रिक्स को \omterm{def:b2:quadratic:adjoint}{सममित}
        \emph{सकारात्मक निश्चित} आव्यूह, विशेषणात्मक रूप से
        \emph{(विकर्ण करें; व्युत्क्रम वर्णक्रमीय है
        लघुगणक)}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The orthogonal group as a level
set.}
\begin{enumerate}[resume]
  \item होने देना $ F(A) = A^{\mathsf T}A $, से $\mathcal M_n(\R)$ तक \omterm{def:b2:quadratic:adjoint}{सममित} मैट्रिक्स $ S_n $. गणना करना $\dd F_A(H) =
        A^{\mathsf T}H + H^{\mathsf T}A $ और उसे हर हाल में दिखाओ
        $ A \in O_n = F^{-1}(I)$ यह \omterm{def:b2:diffcalc:differential}{अंतर} है
        \emph{विशेषण} पर $ S_n $ \emph{(दिया गया $ S \in S_n $,
        कोशिश $ H = \frac12 AS $)}: $ O_n $ एक सहज स्तर सेट है,
        आयाम का $ n^2 - \frac{n(n+1)}2 = \frac{n(n-1)}2$.
  \item वह हर दिखाओ $ C^1$ वक्र $ A(t) \in O_n $ साथ $ A(0) =
        I $ एक है \emph{antisymmetric} वेग $ A'(0)$, और
        इसके विपरीत एंटीसिमेट्रिक के लिए $ K $ वक्र
        $\eu^{tK}$ में रहता है $ O_n $: का स्पर्शरेखा स्थान $ O_n $
        पर $ I $ \omterm{def:b2:series:def}{बिल्कुल} एंटीसिमेट्रिक मैट्रिक्स है।
  \item दिखाओ $\det\eu^{K} = 1$ एंटीसिमेट्रिक के लिए $ K $ (प्रश्न
        18): घातीय वक्र घूर्णन समूह में रहता है
        $ SO_n $. इसकी पूरी गणना करें $ n = 2$: साथ $ J =
        \begin{pmatrix}0 & -1\\ 1 & 0\end{pmatrix}$, सिद्ध करना
        \[
        \eu^{\theta J} = \begin{pmatrix} \cos\theta &
        -\sin\theta\\ \sin\theta & \cos\theta\end{pmatrix} :
        \]
         \omterm{ex:b2:nvs:matrixexp}{मैट्रिक्स घातांक} \emph{है} द्वारा घूर्णन
        $\theta $, और कोसाइन और साइन की श्रृंखला परिभाषाएँ
        मैट्रिक्स के अंदर पुनः प्रकट होना।
  \item (घनत्व चाल) के घनत्व का उपयोग करना $ GL_n(\R)$
        (प्रश्न 3) और निरंतरता, व्युत्क्रमणीय से विस्तार करें
        सभी आव्यूहों की पहचान
        \[
        \operatorname{adj}(AB) =
        \operatorname{adj}(B)\operatorname{adj}(A)
        \]
        \emph{(उलटा के लिए $ A, B $ दोनों पक्ष बराबर
        $\det(AB)(AB)^{-1}$; में दोनों पक्ष बहुपद हैं
        प्रविष्टियाँ)}.
  \item संश्लेषण। प्रत्येक एक वाक्य में: (i) जो पहले
        अध्यायों ने प्रत्येक भाग का इंजन प्रदान किया
        (पूर्णता और मानकीकृत बीजगणित; वर्णक्रमीय
        प्रमेय; व्युत्क्रम फलन प्रमेय); (ii) जो
        इस समस्या का सूत्र अंतर-समीकरण है
        अध्याय पर निर्भर करेगा, और कहाँ; (iii) क्या
        $\dd(\det)_I = \operatorname{tr}$ और $\det\eu^A =
        \eu^{\operatorname{tr}A}$ ट्रेस और के बारे में कहें
        सिद्ध ``असीमित और वैश्विक मात्रा'' के रूप में;
        (iv) वर्ष 3 खंड में प्रश्नों का क्या अर्थ है
        21--23 (झूठ समूह और उनके झूठ बीजगणित)।
