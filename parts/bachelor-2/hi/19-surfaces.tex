\chapter{सतह}\label{ch:b2:surfaces}

वक्रों के बाद, सतहें: दो-पैरामीटर वाली वस्तुएं $\R^3$. द
की \omterm{def:b2:diffcalc:differential}{विभेदक} गणना \cref{ch:b2:diffcalc} सब कुछ प्रदान करता है
हमें चाहिए --- आंशिक व्युत्पन्न स्पर्शरेखा सदिश, क्रॉस दें
उत्पाद सामान्य देता है, निर्धारकों देना \omterm{def:b2:surfaces:area}{क्षेत्र}ों. हम परिभाषित करते हैं
नियमित \omterm{def:b2:surfaces:param}{पैरामीट्रिज्ड सतह}ें, उनका \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल}, और यह
\emph{\omterm{def:b2:surfaces:fff}{प्रथम मौलिक रूप}}, जो सभी को एन्कोड करता है \omterm{def:b2:curves:length}{लंबाई} और \omterm{def:b2:surfaces:area}{क्षेत्र}
सतह पर माप. लेवल सेट के रूप में सतहें भी उत्पन्न होती हैं
$ f(x, y, z) = c $; फिर ढाल सामान्य को निर्देशित करती है।

\section{पैरामीट्रिज्ड सतहें}

\begin{definition}[नियमित पैरामीट्रिज्ड सतह]\label{def:b2:surfaces:param}
होने देना $ U \subseteq \R^2$ होना \omterm{def:b2:metric:topology}{खुला}. ए \emph{पैरामीट्रिज्ड
सतह}\index{पैरामीट्रिज्ड
सतह} का
कक्षा $\mathcal{C}^k $ ($ k \geq 1$) एक नक्शा है $\sigma \colon U \to
\R^3$, $(u, v) \mapsto \sigma(u, v)$, वर्ग का $\mathcal{C}^k $. ए
बिंदु है \emph{नियमित} यदि आंशिक व्युत्पन्न वैक्टर
\[
\sigma_u(u, v) = \frac{\partial\sigma}{\partial u}(u,v),
\qquad
\sigma_v(u, v) = \frac{\partial\sigma}{\partial v}(u,v)
\]
रैखिक रूप से स्वतंत्र हैं, अर्थात\ $\sigma_u \wedge \sigma_v \neq 0$;
सतह है \emph{नियमित} यदि हर बिंदु है.
\end{definition}

\begin{example}[तीन मानक विवरण]\label{ex:b2:surfaces:standard}
\leavevmode
\begin{enumerate}
\item \emph{ग्राफ़:} $\sigma(u, v) = (u,\ v,\ f(u, v))$ के लिए $ f \in
\mathcal{C}^1(U)$. हमेशा नियमित: $\sigma_u = (1, 0, f_u)$ और
$\sigma_v = (0, 1, f_v)$ स्वतंत्र हैं.
\item \emph{\omterm{def:b2:surfaces:area}{क्षेत्र} (गोलाकार निर्देशांक):} के \omterm{def:b2:surfaces:area}{क्षेत्र} के लिए
त्रिज्या $ R $,
\[
\sigma(\theta, \varphi)
= (R\cos\theta\cos\varphi,\ R\sin\theta\cos\varphi,\
R\sin\varphi),
\qquad
(\theta, \varphi) \in \R \times
\bigl(-\tfrac\pi2, \tfrac\pi2\bigr),
\]
साथ $\theta $ देशांतर और $\varphi $ अक्षांश. एक जाँच करता है
$\norm{\sigma_\theta \wedge \sigma_\varphi} = R^2\cos\varphi > 0$:
ध्रुवों से नियमित रूप से दूर (जिसे यह चार्ट छोड़ देता है)।
\item \emph{लेवल सेट:} $ S = \{(x,y,z) : f(x, y, z) = c\}$ कहाँ
$ f $ है $\mathcal{C}^1$ और $\nabla f \neq 0$ पर $ S $. प्रत्येक के पास
बिंदु, एक निर्देशांक को दूसरे के कार्य के रूप में व्यक्त किया जा सकता है
\omterm{def:b2:diffcalc:differential}{अंतर}्निहित फ़ंक्शन प्रमेय द्वारा दो (\cref{ch:b2:diffcalc}), इसलिए
$ S $ स्थानीय रूप से एक ग्राफ़ है.
\end{enumerate}
\end{example}

\begin{example}[लेवल सेट से लेकर ग्राफ़ तक]
\label{ex:b2:surfaces:levelgraph}
आइटम 3 में निहित फ़ंक्शन प्रमेय एक स्पष्टता के योग्य है
भागो. गोला ले लो $ x^2 + y^2 + z^2 = R^2$ इसके उत्तर के निकट
खंभा $(0, 0, R)$: वहाँ $\frac{\partial f}{\partial z} = 2z =
2R \neq 0$, और के लिए समाधान $ z $ ग्राफ़ चार्ट देता है
\[
z = \sqrt{R^2 - x^2 - y^2},
\qquad x^2 + y^2 < R^2 ,
\]
इस पर हर जगह नियमित (\omterm{def:b2:metric:topology}{खुला}) डोमेन --- सहित
ध्रुव कि गोलाकार चार्ट छूट गया। भूमध्य रेखा बिंदु के पास
जैसे $(R, 0, 0)$ वही प्रमेय हल करता है $ x $ बजाय
($\frac{\partial f}{\partial x} = 2R \neq 0$). का नियम
अंगूठा: एक समतल सतह निर्देशांक तल पर एक ग्राफ है
\emph{ग्रेडिएंट के सबसे बड़े घटक के लिए ऑर्थोगोनल},
और गोले को छह ऐसे ग्राफ़ चार्ट से ढककर एक
बिना किसी त्रिकोणमिति के हर जगह इसकी चिकनाई की जाँच करता है
सब.
\end{example}

\section{स्पर्शरेखा तल और सामान्य}

\begin{definition}[स्पर्शरेखा तल]\label{def:b2:surfaces:tangent}
होने देना $\sigma $ पर नियमित रहें $(u_0, v_0)$, $ M_0 = \sigma(u_0, v_0)$.
 \emph{स्पर्शरेखा तल}\index{स्पर्शरेखा तल} $ T_{M_0}S $ के माध्यम से विमान है $ M_0$
निर्देशक $\operatorname{Vect}(\sigma_u, \sigma_v)$ (आंशिक पर
$(u_0, v_0)$). \emph{इकाई सामान्य}\index{इकाई सामान्य} है
\[
n(u_0, v_0)
= \frac{\sigma_u \wedge \sigma_v}
       {\norm{\sigma_u \wedge \sigma_v}} .
\]
\end{definition}

\begin{proposition}[स्पर्शरेखा सदिश वेग हैं
वेक्टर]\label{prop:b2:surfaces:velocity}
इसकी दिशा $ T_{M_0}S $ \omterm{def:b2:series:def}{बिल्कुल} सदिशों का समुच्चय है
$\gamma'(0)$, कहाँ $\gamma = \sigma \circ c $ से अधिक है
$\mathcal{C}^1$ के माध्यम से सतह पर खींचे गए वक्र $ M_0$ (अर्थात्\
$ c \colon (-\varepsilon, \varepsilon) \to U $ है $\mathcal{C}^1$
साथ $ c(0) = (u_0, v_0)$).
\end{proposition}

\begin{proof}
अगर $ c(t) = (u(t), v(t))$, श्रृंखला नियम
(\cref{ch:b2:diffcalc}) देता है
\[
\gamma'(0) = u'(0)\,\sigma_u + v'(0)\,\sigma_v
\in \operatorname{Vect}(\sigma_u, \sigma_v).
\]
इसके विपरीत, वेक्टर $ a\sigma_u + b\sigma_v $ द्वारा प्राप्त किया जाता है
वक्र $ c(t) = (u_0 + at,\ v_0 + bt)$, जो में रहता है \omterm{def:b2:metric:topology}{खुला} तय करना
$ U $ के लिए $\abs t $ छोटा।
\end{proof}

\begin{example}[हेलिकॉइड का स्पर्शरेखा तल]
\label{ex:b2:surfaces:helicoidtangent}
हेलिकॉइड के लिए $\sigma(u, v) = (v\cos u,\ v\sin u,\ au)$,
बिंदु पर $\sigma(0, 1) = (1, 0, 0)$:
\[
\sigma_u = (0,\ 1,\ a), \qquad \sigma_v = (1,\ 0,\ 0),
\qquad
\sigma_u \wedge \sigma_v = (0,\ a,\ -1),
\]
इतना \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल} है $ a\,y = z $. इसमें संपूर्ण समाविष्ट है
क्षैतिज निर्णय $\{(t, 0, 0)\}$ (दिशा $\sigma_v $): जैसे
के शंकु के लिए \cref{exo:b2:surfaces:1}, एक सतह द्वारा शासित
सीधी रेखाओं में प्रत्येक नियम अंदर पड़ा होता है स्पर्शरेखा
विमान इसके साथ. अन्य स्पर्शरेखा दिशा $\sigma_u $ है
हेलिक्स का वेग $ u \mapsto \sigma(u, 1)$: एक चार्ट,
दो खींचे गए वक्र, और संपूर्ण \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल} फैला हुआ है ---
\cref{prop:b2:surfaces:velocity} कार्रवाई में.
\end{example}

\begin{proposition}[समतल सतह का सामान्य]\label{prop:b2:surfaces:gradient}
होने देना $ S = \{f = c\}$ साथ $ f $ कक्षा का $\mathcal{C}^1$ और $\nabla
f(M_0) \neq 0$. फिर \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल} का $ S $ पर $ M_0$ है
के माध्यम से विमान $ M_0$ ओर्थोगोनल से $\nabla f(M_0)$:
\[
T_{M_0}S :\quad
\langle \nabla f(M_0),\ M - M_0\rangle = 0 .
\]
\end{proposition}

\begin{proof}
किसी भी वक्र के लिए $\gamma $ के नाम आहरित $ S $ के माध्यम से $ M_0$, $ f(\gamma(t)) =
c $ \omterm{def:b2:funcseq:def}{समान रूप से}, इसलिए श्रृंखला नियम देता है $\langle
\nabla f(M_0), \gamma'(0)\rangle = 0$: सभी वेग सदिश हैं
ग्रेडिएंट के लिए ऑर्थोगोनल, इसलिए स्पर्शरेखा दिशा समाहित है
विमान में $\nabla f(M_0)^\perp $. दोनों हैं $2$-आयामी
उपस्थान --- स्पर्शरेखा दिशा क्योंकि $ S $ स्थानीय रूप से एक है
नियमित ग्राफ (\cref{ex:b2:surfaces:standard}), ओर्थोगोनल
पूरक क्योंकि $\nabla f(M_0) \neq 0$ --- इसलिए वे हैं
बराबर.
\end{proof}

\begin{example}\label{ex:b2:surfaces:spheretangent}
गोले के लिए $ x^2 + y^2 + z^2 = R^2$: $\nabla f = 2(x, y, z)$,
इतना \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल} पर $ M_0$ त्रिज्या के लिए ओर्थोगोनल है
$\vect{OM_0}$ ---शास्त्रीय तथ्य यह है कि त्रिज्या और \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल}
समीकरण के साथ लंबवत हैं $\langle M_0, M\rangle = R^2$.
\end{example}

\begin{example}[ग्राफ़ स्पर्शरेखा तल]\label{ex:b2:surfaces:graphtangent}
के लिए $ z = f(x, y)$ पर $(x_0, y_0)$: आवेदन करना
\cref{prop:b2:surfaces:gradient} को $ F(x,y,z) = f(x,y) - z $,
\[
z = f(x_0, y_0)
+ f_x(x_0, y_0)(x - x_0)
+ f_y(x_0, y_0)(y - y_0) ,
\]
 \omterm{def:b2:affine:subspace}{एफ़िन} पहले क्रम के टेलर विस्तार का हिस्सा ---
\omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल} का ग्राफ है \omterm{def:b2:diffcalc:differential}{अंतर}, जैसा कि यह होना चाहिए।
\end{example}

\begin{example}[ए का निकटतम बिंदु
सतह]\label{ex:b2:surfaces:closest}
परवलयज का कौन सा बिंदु $ z = x^2 + y^2$ के सबसे नजदीक है
$ P = (0, 0, 1)$? के साथ वर्ग दूरी कम से कम करें
सतह: साथ $\rho^2 = x^2 + y^2$,
\[
g(\rho^2) = \rho^2 + (\rho^2 - 1)^2,
\qquad
g'(\rho^2) = 1 + 2(\rho^2 - 1) = 0
\iff \rho^2 = \tfrac12 ,
\]
ऊँचाई पर बिन्दुओं का वृत्त देना $ z = \frac12$ और
दूरी $\sqrt{\tfrac12 + \tfrac14} = \frac{\sqrt3}2$. द
न्यूनतमता का ज्यामितीय \omterm{thm:b2:structures:signature}{हस्ताक्षर}: ऐसे बिंदु पर $ M $, द
वेक्टर $\vect{MP}$ होना ही चाहिए \emph{सामान्य} सतह पर ---
अन्यथा वेग वाले खींचे गए वक्र के साथ फिसलते हुए
की ओर घटक $ P $ दूरी कम हो जाएगी. जाँच करना:
$\nabla(z - x^2 - y^2) = (-2x, -2y, 1)$ पर $ M = (x, y,
\tfrac12)$, जबकि $\vect{MP} = (-x, -y, \tfrac12) =
\tfrac12(-2x, -2y, 1)$: समानांतर, जैसा कि अनुमान लगाया गया था। द
प्रथम-क्रम की स्थिति ``लंब का पाद'' है
वही जो लेवल सेट पर एक्स्ट्रेमा को चलाएगा
\cref{exo:b2:surfaces:12}.
\end{example}

\begin{example}[चतुर्भुज के स्पर्शरेखा तल: द
ध्रुवीकरण नियम]\label{ex:b2:surfaces:polarization}
होने देना $ S : xy + yz + zx = 1$ और $ M_0 = (1, 1, 0) \in S $. यहाँ
$\nabla f = (y + z,\ x + z,\ x + y)$, इसलिए $\nabla f(M_0) = (1,
1, 2)$ और यह \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल} है
\[
(x - 1) + (y - 1) + 2z = 0,
\qquad\text{i.e.}\qquad x + y + 2z = 2 .
\]
की तरफ से भी यही जवाब आता है \emph{\omterm{def:b2:quadratic:def}{ध्रुवीकरण} नियम} वह
सामान्यीकरण करता है \cref{ex:b2:surfaces:spheretangent} और
\cref{exo:b2:surfaces:2}: में द्विघातका समीकरण, प्रतिस्थापित करें
$ x^2$ द्वारा $ x_0x $, और प्रत्येक उत्पाद $ xy $ द्वारा $\frac{x_0y +
y_0x}2$ (और \omterm{def:b2:structures:generated}{चक्रीय} रूप से):
\[
\frac{x_0y + y_0x}2 + \frac{y_0z + z_0y}2 + \frac{z_0x +
x_0z}2 = 1
\;\xrightarrow{\ M_0 = (1,1,0)\ }\;
\frac{x + y}2 + \frac z2 + \frac z2 = 1 ,
\]
जो है $ x + y + 2z = 2$ दोबारा। नियम इसलिए काम करता है
$\nabla $ एक का \omterm{pb:b2:surfaces:1}{द्विघात} रूप संबद्ध द्विरेखीय रूप है
आधार बिंदु के विरुद्ध मूल्यांकन किया गया --- ए की स्पर्शरेखा \omterm{pb:b2:surfaces:1}{द्विघात} है
\omterm{def:b2:quadratic:def}{ध्रुवीकरण}, का एक और चेहरा \cref{ch:b2:quadratic}.
\end{example}

\section{पहला मौलिक स्वरूप}

\begin{definition}[प्रथम मौलिक स्वरूप]\label{def:b2:surfaces:fff}
होने देना $\sigma \colon U \to \R^3$ नियमित रहो $\mathcal{C}^1$
सतह। इसका \emph{प्रथम मौलिक रूप}\index{प्रथम मौलिक रूप} पर $(u,v)$ है
सकारात्मक निश्चित \omterm{pb:b2:surfaces:1}{द्विघात} रूप पर $\R^2$
\[
I(h, k) = \norm{h\,\sigma_u + k\,\sigma_v}^2
= E\,h^2 + 2F\,hk + G\,k^2,
\]
कहाँ
\[
E = \norm{\sigma_u}^2,
\qquad
F = \langle \sigma_u, \sigma_v\rangle,
\qquad
G = \norm{\sigma_v}^2 .
\]
\end{definition}

\begin{remark}
$ I $ परिवेशी यूक्लिडियन स्केलर उत्पाद का प्रतिबंध है
द \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल}, आधार में पढ़ें $(\sigma_u, \sigma_v)$: यह है
सकारात्मक निश्चित रूप से ठीक है क्योंकि $\sigma_u, \sigma_v $ हैं
स्वतंत्र (\cref{ch:b2:quadratic}). पर प्रत्येक मीट्रिक मात्रा
सतह --- \omterm{def:b2:curves:length}{लंबाई} खींचे गए वक्रों का, उनके बीच के कोणों का, \omterm{def:b2:surfaces:area}{क्षेत्र}ों ---
से गणना की जाती है $ E, F, G $ अकेला। समान के साथ दो सतहें $ E,
F, G $ उपयुक्त मापदंडों में हैं \emph{\omterm{def:b2:quadratic:adjoint}{सममित}ीय} भले ही वे बैठें
\omterm{def:b2:diffcalc:differential}{अंतर}िक्ष में अलग-अलग: यह आंतरिक का प्रारंभिक बिंदु है
ज्यामिति.
\end{remark}

\begin{example}[समन्वय के बीच के कोण
वक्र]\label{ex:b2:surfaces:coordangle}
 \omterm{def:b2:surfaces:fff}{प्रथम मौलिक रूप} कोणों को भी मापता है:
समन्वय वक्र $ u \mapsto \sigma(u, v_0)$ और $ v \mapsto
\sigma(u_0, v)$ कोण पर मिलें $\theta $ साथ
\[
\cos\theta = \frac{\langle\sigma_u,
\sigma_v\rangle}{\norm{\sigma_u}\,\norm{\sigma_v}}
= \frac{F}{\sqrt{EG}} :
\]
एकल गुणांक $ F $ की रूढ़िवादिता तय करता है
पैरामीटर नेट. गोलाकार चार्ट और हेलिकॉइड के लिए, $ F =
0$: मेरिडियन समानताएं काटते हैं, और हेलिकॉप्टर क्षैतिज काटते हैं
निर्णय, समकोण पर --- यही कारण है कि उनका \omterm{def:b2:surfaces:area}{क्षेत्र}
इंटीग्रैंड्स का पतन हो गया $\sqrt{EG}$. ग्राफ़ चार्ट के लिए,
$ F = f_xf_y $ केवल वहीं गायब हो जाता है जहां आंशिक व्युत्पन्न होता है:
झुके हुए ग्राफ का निर्देशांक जाल है \emph{नहीं}
ऑर्थोगोनल, भले ही $(x, y)$-नीचे नेट है. कब
सतह पर गणनाएँ भारी लगती हैं, पहला कदम यह है
के साथ एक चार्ट खोजें $ F = 0$.
\end{example}

\begin{example}[काठी चार्ट]\label{ex:b2:surfaces:saddlechart}
काठी के लिए $ z = xy $ चार्ट के साथ $\sigma(u, v) = (u, v,
uv)$:
\[
\sigma_u = (1, 0, v), \qquad \sigma_v = (0, 1, u),
\qquad
E = 1 + v^2, \quad F = uv, \quad G = 1 + u^2 ,
\]
और $ EG - F^2 = 1 + u^2 + v^2 > 0$: हर जगह नियमित. द
एक बिंदु से गुजरने वाले दो निर्देशांक वक्र हैं \emph{सीधे पंक्तियां} का $\R^3$ (हल करना $ u $ या ठीक करें $ v $: के फैसले
दोगुना शासित काठी), फिर भी $ F \neq 0$ कुल्हाड़ियों से दूर: निर्णय
एक सामान्य बिंदु के माध्यम से ऑर्थोगोनल नहीं हैं। दोनों फैसले झूठ बोलते हैं
में \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल}, जो वे फैलाते हैं --- तो स्पर्शरेखा
विमान सतह को दो पूर्ण रेखाओं के साथ काटता है, चरम
गोले के विपरीत, जिसका \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल} एक पर स्पर्श करें
केवल बिंदु. के बीच ``दूसरे क्रम के संपर्क'' का चिह्न
एक सतह और उसकी \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल} एक है \omterm{thm:b2:curves:frenet2d}{वक्रता} कहानी,
वर्ष 3 खंड में लिया गया।
\end{example}

\begin{proposition}[a पर खींचे गए वक्र की लंबाई
सतह]\label{prop:b2:surfaces:curvelength}
अगर $\gamma(t) = \sigma(u(t), v(t))$, $ t \in [a, b]$, है
$\mathcal{C}^1$, तब
\[
L(\gamma) = \int_a^b
\sqrt{E\,u'^2 + 2F\,u'v' + G\,v'^2}\;\dd t ,
\]
साथ $ E, F, G $ पर मूल्यांकन किया गया $(u(t), v(t))$.
\end{proposition}

\begin{proof}
$\gamma' = u'\sigma_u + v'\sigma_v $ श्रृंखला नियम से, इसलिए
$\norm{\gamma'}^2 = I(u', v') = Eu'^2 + 2Fu'v' + Gv'^2$; एकीकृत करें
$\norm{\gamma'}$ (\cref{def:b2:curves:length}).
\end{proof}

\begin{example}[हवाई जहाज़ पोल के ऊपर से क्यों उड़ते हैं?]\label{ex:b2:surfaces:polarroute}
दो हवाई अड्डे अक्षांश पर स्थित हैं $\varphi_0$ और विपरीत
देशांतर: $ A = \sigma(0, \varphi_0)$ और $ B = \sigma(\pi,
\varphi_0)$ त्रिज्या के गोले पर $ R $. समानांतर के साथ
($\varphi \equiv \varphi_0$), द \omterm{def:b2:curves:length}{लंबाई} है
$\int_0^\pi R\cos\varphi_0\,\dd\theta = \pi R\cos\varphi_0$.
ध्रुव के ऊपर मार्ग के साथ (मध्याह्न रेखा के ऊपर)। $\theta = 0$,
मध्याह्न रेखा के नीचे $\theta = \pi $), यह है $2R(\frac\pi2 -
\varphi_0)$. अक्षांश पर $\varphi_0 = \frac\pi3$ (साठ
डिग्री): समानांतर मार्ग $\pi R/2 \approx 1.571\,R $, ध्रुवीय
मार्ग $\pi R/3 \approx 1.047\,R $ --- एक तिहाई छोटा. वास्तव में
$\pi\cos\varphi_0 \geq \pi - 2\varphi_0$ पर
$\intcc0{\pi/2}$ (फ़ंक्शन $\pi\cos\varphi - \pi +
2\varphi $ दोनों सिरों और इसके व्युत्पन्न पर गायब हो जाता है $2 -
\pi\sin\varphi $ एक बार संकेत बदलता है, इसलिए यह पहले बढ़ रहा है
फिर घट रहा है, इसलिए गैर-नकारात्मक): ध्रुवीय मार्ग कभी नहीं
हार जाता है. \omterm{def:b2:surfaces:fff}{प्रथम मौलिक रूप} एक नेविगेशन प्रश्न बन गया
दो एक-पंक्ति अभिन्नों में; \cref{exo:b2:surfaces:6} धक्का देता है
मेरिडियन के लिए वास्तविक न्यूनतम प्रमाण का विचार।
\end{example}

\begin{lemma}[लैग्रेंज पहचान]\label{lem:b2:surfaces:lagrange}
सभी के लिए $ a, b \in \R^3$:
$\norm{a \wedge b}^2 = \norm a^2 \norm b^2 - \langle a, b\rangle^2$.
विशेष रूप से
\[
\norm{\sigma_u \wedge \sigma_v} = \sqrt{EG - F^2}.
\]
\end{lemma}

\begin{proof}
यदि हम प्रतिस्थापित करें तो दोनों पक्ष अपरिवर्तित हैं $ b $ इसके घटक द्वारा $ b_\perp
= b - \frac{\langle a, b\rangle}{\norm a^2}a $ ओर्थोगोनल से $ a $
(के लिए $ a \neq 0$; मामला $ a = 0$ तुच्छ है): बाईं ओर
क्योंकि $ a \wedge a = 0$, दाहिनी ओर विस्तार करके $\norm
{b_\perp}^2 = \norm b^2 - \frac{\langle a,b\rangle^2}{\norm a^2}$
और $\langle a, b_\perp\rangle = 0$. तो यह साबित करने के लिए पर्याप्त है
ऑर्थोगोनल के लिए पहचान $ a, b $, जहां यह पढ़ता है $\norm{a \wedge
b} = \norm a \norm b $: सच है, क्योंकि ऑर्थोगोनल वैक्टर के लिए
क्रॉस उत्पाद है आदर्श $\norm a\norm b\,\abs{\sin\frac\pi2}$. द
प्रदर्शित सूत्र मामला है $ a = \sigma_u $, $ b = \sigma_v $.
\end{proof}

\begin{remark}
लैग्रेंज पहचान कहती है $ EG - F^2 = \det\begin{psmallmatrix}
E & F\\ F & G\end{psmallmatrix}$ है \emph{ग्राम
सिद्ध} का $(\sigma_u, \sigma_v)$: चुकता \omterm{def:b2:surfaces:area}{क्षेत्र} का
वे जिस समांतर चतुर्भुज को फैलाते हैं। नियमितता, सकारात्मकता
की निश्चितता \omterm{def:b2:surfaces:fff}{प्रथम मौलिक रूप}, और की सकारात्मकता
ग्राम सिद्ध एक स्थिति के तीन वाक्यांश हैं ---
यही कारण है कि \omterm{def:b2:surfaces:area}{क्षेत्र} नीचे दिया गया इंटीग्रैंड कभी भी गायब नहीं होता है
नियमित चार्ट.
\end{remark}

\begin{definition}[क्षेत्र]\label{def:b2:surfaces:area}
होने देना $\sigma \colon U \to \R^3$ एक नियमित इंजेक्शन बनें
$\mathcal{C}^1$ सतह और $ K \subseteq U $ ए \omterm{def:b2:metric:compact}{कॉम्पैक्ट} डोमेन चालू
कौन से दोहरे अभिन्न अंग समझ में आते हैं (\cref{ch:b2:multint}).
\emph{क्षेत्र}\index{क्षेत्र} टुकड़े का $\sigma(K)$ है
\[
\mathcal{A}
= \iint_K \norm{\sigma_u \wedge \sigma_v}\,\dd u\,\dd v
= \iint_K \sqrt{EG - F^2}\;\dd u\,\dd v .
\]
\end{definition}

\begin{remark}[ये फार्मूला क्यों]
आयत $[u, u + \dd u] \times [v, v + \dd v]$ को मैप किया गया है
प्रथम क्रम, द्वारा फैलाए गए समांतर चतुर्भुज पर $\sigma_u\,\dd u $
और $\sigma_v\,\dd v $, किसका \omterm{def:b2:surfaces:area}{क्षेत्र} है $\norm{\sigma_u \wedge
\sigma_v}\,\dd u\,\dd v $: परिभाषा स्थानीय को एकीकृत करती है
क्षेत्र-विरूपण कारक, \omterm{def:b2:series:def}{बिल्कुल} वैसा ही वक्राकार \omterm{def:b2:curves:length}{लंबाई} स्थानीय को एकीकृत करता है
गति. के साथ संगति मापदंडों का परिवर्तन है
\cref{exo:b2:surfaces:8}; चर-परिवर्तन के साथ संगति
दोहरे समाकलन के सूत्र पर चर्चा की गई है
\cref{ch:b2:multint}.
\end{remark}

\begin{example}[दो अलग-अलग ग्राफ़, एक
क्षेत्र]\label{ex:b2:surfaces:samearea}
यूनिट डिस्क के ऊपर, कटोरे की तुलना करें $ z = \frac12(x^2 + y^2)$
और काठी $ z = xy $. उनका \omterm{def:b2:surfaces:area}{क्षेत्र} integrands
(\cref{exo:b2:surfaces:5}) हैं
\[
\sqrt{1 + x^2 + y^2}
\qquad\text{and}\qquad
\sqrt{1 + y^2 + x^2} :
\]
समान। दो सतहें --- एक समान तरीके से अंदर की ओर मुड़ती हैं
सभी दिशाएँ, अन्य काठी के आकार की --- \omterm{def:b2:series:def}{बिल्कुल} हैं
बराबर \omterm{def:b2:surfaces:area}{क्षेत्र}ों प्रत्येक डोमेन पर, $\frac{2\pi}3(2\sqrt2 - 1)$
यूनिट डिस्क पर. \omterm{def:b2:surfaces:area}{क्षेत्र} तत्व केवल देखता है
\emph{\omterm{def:b2:curves:length}{लंबाई}} ढाल की, व्यवस्था की नहीं
झुकना; काठी से कटोरा बताने की आवश्यकता है
दूसरे क्रम का डेटा (संकेत संरचना में प्रदर्शित)
\cref{fig:b2:surfaces:saddle}), जिसकी कोई राशि नहीं \omterm{def:b2:surfaces:area}{क्षेत्र}
माप से पता चलता है. प्रथम मौलिक स्वरूप: मीट्रिक, अंधा
आकार देना; आकार देखने वाला दूसरा रूप वर्ष 3 का है।
\end{example}

\begin{example}[गोले का क्षेत्रफल]\label{ex:b2:surfaces:spherearea}
के गोलाकार चार्ट के लिए \cref{ex:b2:surfaces:standard}:
\[
\sigma_\theta = R(-\sin\theta\cos\varphi,\
\cos\theta\cos\varphi,\ 0),
\qquad
\sigma_\varphi = R(-\cos\theta\sin\varphi,\
-\sin\theta\sin\varphi,\ \cos\varphi),
\]
इसलिए $ E = R^2\cos^2\varphi $, $ F = 0$, $ G = R^2$ और $\sqrt{EG - F^2}
= R^2\cos\varphi $. इस तरह
\[
\mathcal{A}
= \int_{-\pi/2}^{\pi/2}\!\!\int_0^{2\pi}
R^2\cos\varphi\;\dd\theta\,\dd\varphi
= 2\pi R^2\,\bigl[\sin\varphi\bigr]_{-\pi/2}^{\pi/2}
= \boxed{4\pi R^2} .
\]
\end{example}

\begin{example}[शंकु, के विरुद्ध जाँच की गई
स्कूल फार्मूला]\label{ex:b2:surfaces:conearea}
शंकु के लिए $ z = \sqrt{x^2 + y^2}$ वलय के ऊपर $ a \leq
\rho \leq b $, का ग्राफ़ सूत्र
\cref{exo:b2:surfaces:5} देता है $1 + f_x^2 + f_y^2 = 2$
(गणना करें $ f_x = x/\rho $, $ f_y = y/\rho $), इसलिए
\[
\mathcal A = \sqrt2\,\pi\,(b^2 - a^2) .
\]
तिरछी-ऊंचाई सूत्र के साथ संगति की जांच करें $\pi\rho\ell $
का \cref{ex:b2:surfaces:revolution}: आधार के पूर्ण शंकु
त्रिज्या $ b $ और $ a $ पार्श्व है \omterm{def:b2:surfaces:area}{क्षेत्र}ों $\pi b\cdot b\sqrt2$ और
$\pi a\cdot a\sqrt2$, जिसका \omterm{def:b2:diffcalc:differential}{अंतर} \omterm{def:b2:series:def}{बिल्कुल} सही है
$\sqrt2\pi(b^2 - a^2)$. दो चार्ट, दो सूत्र, एक \omterm{def:b2:surfaces:area}{क्षेत्र}
--- अपरिवर्तनशीलता सिद्ध हुई \cref{exo:b2:surfaces:8}, देखा
जंगली में.
\end{example}

\begin{remark}[क्षेत्रों के लिए स्वच्छता जांच]
तीन त्वरित जाँचें अधिकांश त्रुटियाँ पकड़ती हैं \omterm{def:b2:surfaces:area}{क्षेत्र}
गणना. \emph{स्केलिंग}: किसी सतह को चौड़ा करना $\lambda $
गुणा $ E, F, G $ द्वारा $\lambda^2$ और यह \omterm{def:b2:surfaces:area}{क्षेत्र} द्वारा
$\lambda^2$ --- एक उत्तर जिसकी निर्भरता पर $ R $ नहीं है
\omterm{pb:b2:surfaces:1}{द्विघात} (जैसे $4\pi R^2$) ग़लत है. \emph{की सकारात्मकता
तत्व}: $\sqrt{EG - F^2}$ पर सख्ती से सकारात्मक होना चाहिए
चार्ट का आंतरिक भाग; लुप्त होता मूल्य चार्ट की गिरावट को दर्शाता है,
गोले के ध्रुवों की तरह एक्साइज किया जाना है। \emph{समरूपता}: ए
ए पर गणना \omterm{def:b2:quadratic:adjoint}{सममित} टुकड़ा सुसंगत होना चाहिए
इसके सर्वांगसम भागों का योग --- गोलार्ध बेहतर था
देना $2\pi R^2$.
\end{remark}

\begin{example}[क्रांति की सतह]\label{ex:b2:surfaces:revolution}
वक्र घुमाएँ $ z \mapsto (r(z), 0, z)$, $ r > 0$ कक्षा का
$\mathcal{C}^1$, चारों ओर $ z $-अक्ष:
\[
\sigma(\theta, z) = (r(z)\cos\theta,\ r(z)\sin\theta,\ z).
\]
तब $ E = r(z)^2$, $ F = 0$, $ G = 1 + r'(z)^2$, इसलिए
\[
\mathcal{A} = \int_a^b\!\!\int_0^{2\pi}
r(z)\sqrt{1 + r'(z)^2}\;\dd\theta\,\dd z
= 2\pi\int_a^b r(z)\sqrt{1 + r'(z)^2}\,\dd z ,
\]
शास्त्रीय सूत्र (परिधि $2\pi r $ तिरछा समय
\omterm{def:b2:curves:length}{लंबाई} तत्व)। शंकु के लिए $ r(z) = kz $, $ z \in [0, h]$: $\mathcal
A = 2\pi k\sqrt{1 + k^2}\,\frac{h^2}{2} = \pi \rho \ell $ साथ
$\rho = kh $ आधार त्रिज्या और $\ell = h\sqrt{1 + k^2}$ तिरछा
ऊंचाई --- स्कूल का फॉर्मूला, अब प्रवेश के बजाय प्राप्त किया गया है।
\end{example}

\begin{example}[कैटेनॉयड]\label{ex:b2:surfaces:catenoid}
कैटेनरी घुमाएँ $ r(z) = \cosh z $, $ z \in \intcc{-1}{1}$,
अपनी धुरी के चारों ओर: परिणामी \emph{कैटेनॉइड} है, द्वारा
क्रांति सूत्र और $1 + \sinh^2 z = \cosh^2 z $,
\[
\mathcal A = 2\pi\int_{-1}^1\cosh z\,\sqrt{1 + \sinh^2z}\,
\dd z = 2\pi\int_{-1}^1\cosh^2z\,\dd z
= \pi\bigl[z + \sinh z\cosh z\bigr]_{-1}^{1},
\]
वह है
\[
\mathcal A = 2\pi + \pi\sinh 2 \approx 17.68 .
\]
तिरछा कारक $\sqrt{1 + r'^2}$ त्रिज्या के साथ विलीन हो गया
एक पूर्ण वर्ग में --- वही पहचान जिसने बनाई थी
कैटेनरी का वक्राकार \omterm{def:b2:curves:length}{लंबाई} वक्र अध्याय में प्राथमिक। यह
बीजगणित कोई दुर्घटना नहीं है: क्रांति की सभी सतहों के बीच
दो सीमा वृत्तों को फैलाते हुए, कैटेनॉइड न्यूनतम हो जाता है
\omterm{def:b2:surfaces:area}{क्षेत्र} (यह दो छल्लों के बीच साबुन की फिल्म का आकार है), और
यह परिवर्तनशील गुण वास्तव में वही है जो अलग करता है
$\cosh $; वर्ष 3 का खंड इसकी गणना से इसे सिद्ध करता है
विविधताएँ।
\end{example}

\begin{remark}[सामान्य ख़तरे]
(मैं) \emph{चार्ट विलक्षणताएँ सतही विलक्षणताएँ नहीं हैं}:
गोलाकार चार्ट ध्रुवों पर ख़राब हो जाता है
($\cos\varphi = 0$), लेकिन वहां गोला \omterm{def:b2:series:def}{बिल्कुल} चिकना है
--- एक अन्य चार्ट (अक्षों की भूमिकाओं का आदान-प्रदान) नियमित है
ध्रुवों पर. किसी बिंदु को एकवचन घोषित करने से पहले, दूसरा प्रयास करें
पैरामीट्रिजेशन. (ii) \emph{की नियमितता $\sigma $ चिंताएँ
पैरामीट्रिज़ेशन, छवि नहीं}: $\sigma(u, v) = (u^3, v,
0)$ साथ में गैर-नियमित है $ u = 0$ हालाँकि इसकी छवि एक है
विमान. (iii) द \omterm{def:b2:surfaces:area}{क्षेत्र} सूत्र की आवश्यकता है $\sigma $
\emph{इंजेक्शन} पर $ K $: एक चार्ट जो एक टुकड़े को दो बार कवर करता है
इसे दो बार गिनता है ($\theta $ भागना $\intcc0{4\pi}$
गोले को दोगुना कर देता है \omterm{def:b2:surfaces:area}{क्षेत्र}). (iv) द \omterm{def:b2:surfaces:tangent}{इकाई सामान्य} परिभाषित किया गया है
सतह से \omterm{thm:b2:structures:signature}{हस्ताक्षर} करने तक लेकिन है \emph{चुना गया} चार्ट द्वारा
(का \omterm{def:b2:structures:generated}{आदेश} $ u, v $); अभिविन्यास से जुड़े बयानों को ठीक करना होगा
वह विकल्प. (v) अंततः, $ EG - F^2 > 0$ अतिरिक्त नहीं है
परिकल्पना: लैग्रेंज के अनुसार यह \omterm{def:b2:series:def}{बिल्कुल} नियमितता है
पहचान--- कहीं लुप्त हो जाए तो समस्या है
चार्ट, और नहीं \omterm{def:b2:surfaces:area}{क्षेत्र} या स्पर्शरेखा-समतल सूत्र वहां लागू होता है।
\end{remark}

\begin{remark}[इस खंड के भीतर परिप्रेक्ष्य]
यहां से लिंक अग्रेषित करें। \omterm{def:b2:surfaces:area}{क्षेत्र} तत्व $\sqrt{EG -
F^2}\,\dd u\,\dd v $ में एक द्वि-आयामी जैकोबियन है
भेस, और \cref{ch:b2:multint} सादृश्य को सटीक बनाता है
परिवर्तन-परिवर्तन प्रमेय के साथ --- सतह
इस अध्याय में अभिन्न अंग हैं \omterm{def:b2:surfaces:area}{क्षेत्र}ों एक इंटीग्रैंड के साथ
बोर्ड. \omterm{def:b2:surfaces:fff}{प्रथम मौलिक रूप} सकारात्मकता का \omterm{def:b2:surfaces:area}{क्षेत्र} है
\omterm{pb:b2:surfaces:1}{द्विघात} रूप, संभाला \omterm{def:b2:funcseq:def}{बिंदुवार} के उपकरणों द्वारा
\cref{ch:b2:quadratic}, जिसका वर्णक्रमीय प्रमेय भी शक्तियाँ देता है
इस अध्याय का सप्ताहांत वर्गीकरण चतुर्भुज. और
सामान्य लाइन बाधा सेट पर चरम समस्याओं को जन्म देती है
(\cref{ex:b2:surfaces:closest}), का ज्यामितीय रोगाणु
लैग्रेंज मल्टीप्लायर विधि का रेखाचित्र बनाया गया
\cref{thm:b2:diffcalc:extrema}.
\end{remark}

की सतह-अभिन्न माप है \cref{ch:b2:multint}, कहाँ
यह ग्रीन के फार्मूले से मिलता है; \omterm{def:b2:surfaces:fff}{प्रथम मौलिक रूप} है
के एक \omterm{def:b2:surfaces:area}{क्षेत्र} का प्रोटोटाइप \omterm{pb:b2:surfaces:1}{द्विघात} रूप, अध्ययन किया \omterm{def:b2:funcseq:def}{बिंदुवार}
के उपकरणों के साथ \cref{ch:b2:quadratic}; और इस अध्याय का
सप्ताहांत समस्या सभी को वर्गीकृत करती है \omterm{pb:b2:surfaces:1}{द्विघात} के साथ सतह
वर्णक्रमीय प्रमेय. वर्ष 3 का खंड सतह पर लौट आया है
\omterm{def:b2:diffcalc:differential}{अंतर} रूप और विचलन प्रमेय, और आंतरिक
\omterm{thm:b2:curves:frenet2d}{वक्रता} --- क्या $ E, F, G $ झुकने के बारे में जानिए---है
\omterm{def:b2:diffcalc:differential}{विभेदक} ज्यामिति का प्रवेश द्वार उचित।
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b2:surfaces:1}
दिखाओ कि \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल} शंकु का $ z = \sqrt{x^2 + y^2}$
(इसके शीर्ष को घटाकर) सभी शीर्ष से होकर गुजरते हैं। \emph{(पैरामीटराइज़ द्वारा
$\sigma(\theta, r) = (r\cos\theta, r\sin\theta, r)$, $ r > 0$.)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:surfaces:2}
खोजें \omterm{def:b2:surfaces:tangent}{स्पर्शरेखा तल} दीर्घवृत्ताभ का $\frac{x^2}{a^2} +
\frac{y^2}{b^2} + \frac{z^2}{c^2} = 1$ एक बिंदु पर $(x_0, y_0,
z_0)$ सतह का.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:surfaces:3}
गणना करना $ E, F, G $ हेलिकॉइड के लिए $\sigma(u, v) = (v\cos u,\
v\sin u,\ au)$, $ a > 0$, और यह \omterm{def:b2:surfaces:area}{क्षेत्र} टुकड़े का $0 \leq u \leq
2\pi $, $0 \leq v \leq 1$, एक अभिन्न के रूप में (इसका उपयोग करके मूल्यांकन करें
$\int\sqrt{v^2 + a^2}\,\dd v = \frac12\bigl(v\sqrt{v^2+a^2} +
a^2\ln(v + \sqrt{v^2 + a^2})\bigr) + C $).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:surfaces:4}
(टोरस) के वृत्त को घुमाकर प्राप्त टोरस को पैरामीट्रिज करें
केंद्र $(R, 0, 0)$ और त्रिज्या $ r < R $ में $ xz $- चारों ओर विमान
$ z $-अक्ष:
\[
\sigma(\theta, \psi) = \bigl((R + r\cos\psi)\cos\theta,\
(R + r\cos\psi)\sin\theta,\ r\sin\psi\bigr).
\]
गणना करना $ E, F, G $, नियमितता की जांच करें, और दिखाएं कि \omterm{def:b2:surfaces:area}{क्षेत्र} है
$4\pi^2 R r $ (पप्पस: माध्य परिधि $2\pi R $ समय \omterm{def:b2:structures:sn}{चक्र}
\omterm{def:b2:curves:length}{लंबाई} $2\pi r $).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:surfaces:5}
दिखाओ कि \omterm{def:b2:surfaces:area}{क्षेत्र} के ग्राफ का $ f \in \mathcal{C}^1(K)$ है
$\iint_K \sqrt{1 + f_x^2 + f_y^2}\,\dd x\,\dd y $, और इसकी गणना करें
परवलयिक टुकड़े के लिए $ z = \frac12(x^2 + y^2)$ डिस्क के ऊपर
$ x^2 + y^2 \leq 1$ (ध्रुवीय निर्देशांक,
\cref{ch:b2:multint}).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:surfaces:6}
एक खींचा हुआ वक्र $\gamma(t) = \sigma(u(t), v(t))$ के \omterm{def:b2:surfaces:area}{क्षेत्र} पर
त्रिज्या $ R $ (गोलाकार चार्ट) है $ u = \theta(t)$, $ v =
\varphi(t)$. इसे लिखें \omterm{def:b2:curves:length}{लंबाई} में एक अभिन्न अंग के रूप में $\theta, \varphi $
और सिद्ध करें कि दो समान बिंदुओं को जोड़ने वाले वक्रों के बीच
मध्याह्न रेखा $\theta = \theta_0$, मेरिडियन चाप सबसे छोटा है।
\emph{(नीचे दिए गए इंटीग्रैंड को इससे बांधें $ R\abs{\varphi'}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:surfaces:7}
(किसी गोले की सामान्य रेखाएँ) चलो $ S $ एक नियमित स्तर की सतह बनें
$\{f = c\}$, \omterm{def:b2:metric:connected}{जुड़े हुए}, जिनकी सभी सामान्य रेखाएँ a से होकर गुजरती हैं
निश्चित बिंदु $\Omega $. बताते हैं कि $ S $ एक गोले में समाहित है
पर केन्द्रित $\Omega $. \emph{(बताते हैं कि $\norm{M - \Omega}^2$ है
खींचे गए प्रत्येक वक्र के साथ शून्य व्युत्पन्न $ S $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:surfaces:8}
(\omterm{def:b2:surfaces:area}{क्षेत्र} ज्यामितीय है) चलो $\Phi \colon U' \to U $ एक हो $\mathcal{C}^1$
के बीच भिन्नता \omterm{def:b2:metric:topology}{खुला} के सेट $\R^2$ और $\tilde\sigma =
\sigma \circ \Phi $. बताते हैं कि
\[
\norm{\tilde\sigma_{u'} \wedge \tilde\sigma_{v'}}
= \abs{\det J_\Phi}\,
\norm{(\sigma_u \wedge \sigma_v)\circ\Phi} ,
\]
और चर-परिवर्तन के सूत्र का उपयोग करके निष्कर्ष निकालें
\cref{ch:b2:multint}, कि \omterm{def:b2:surfaces:area}{क्षेत्र} का
\cref{def:b2:surfaces:area} चुने गए नियमित पर निर्भर नहीं है
पैरामीट्रिजेशन.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:surfaces:9}
(आर्किमिडीज़ का हैट-बॉक्स प्रमेय) त्रिज्या के गोले पर $ R $, द
\emph{\omterm{def:b2:surfaces:area}{क्षेत्र}} अक्षांशों के बीच $ z_1 \leq z \leq z_2$
($-R \leq z_1 < z_2 \leq R $) है \omterm{def:b2:surfaces:area}{क्षेत्र} $2\pi R\,(z_2 - z_1)$:
इसे गोलाकार चार्ट से सिद्ध करें और निष्कर्ष निकालें कि यह एक \omterm{def:b2:surfaces:area}{क्षेत्र} है
\omterm{def:b2:surfaces:area}{क्षेत्र} यह केवल उसकी ऊंचाई पर निर्भर करता है --- संतरे को टुकड़ों में काटना
समान मोटाई के स्लाइस से समान मात्रा में छिलका मिलता है।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:surfaces:10}
दिखाएँ कि सतह की प्रत्येक सामान्य रेखा परिक्रमण करती है
$\sigma(\theta, z) = (r(z)\cos\theta,\ r(z)\sin\theta,\ z)$
($ r > 0$ कक्षा का $\mathcal C^1$) के अक्ष से मिलता है
क्रांति, और प्रतिच्छेदन बिंदु का पता लगाएं।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:surfaces:11}
(सिलेंडर को खोलते हुए) चार्ट $\sigma(u, v) = (\cos u,\
\sin u,\ v)$ यूनिट सिलेंडर की है $ E = G = 1$, $ F = 0$:
इसे सत्यापित करें, और बताएं कि प्रत्येक खींचा गया वक्र क्यों है $ t \mapsto
\sigma(u(t), v(t))$ वही है \omterm{def:b2:curves:length}{लंबाई} समतल वक्र के रूप में $ t
\mapsto (u(t), v(t))$. यह निष्कर्ष निकालें कि हेलिक्स से $(1, 0, 0)$
को $(1, 0, 2\pi c)$ एक मोड़ बनाना है \omterm{def:b2:curves:length}{लंबाई}
$2\pi\sqrt{1 + c^2}$, और उसके साथ कोई खींचा हुआ वक्र नहीं है
समापन बिंदु और एक पूर्ण मोड़ छोटा होता है।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:surfaces:12}
होने देना $ S = \{f = c\}$ एक हो \omterm{def:b2:metric:compact}{कॉम्पैक्ट} नियमित स्तर की सतह और
$ M_0 \in S $ मूल बिंदु से अधिकतम दूरी पर एक बिंदु। बताते हैं कि $\nabla f(M_0)$ के साथ संरेख है $\vect{OM_0}$ --- द
सबसे दूर बिंदु पर सामान्य रेडियल है। दीर्घवृत्त पर लागू करें
$\frac{x^2}{a^2} + \frac{y^2}{b^2} + \frac{z^2}{c^2} = 1$ ($ a >
b > c > 0$): उन सभी बिंदुओं को ढूंढें जहां सामान्य रेडियल है, और
सबसे दूर वालों की पहचान करें.
\end{exercise}

\section{समस्या: चतुर्भुज का वर्गीकरण
$\R^3$}

\begin{problem}[{सप्ताहांत समस्या --- प्रत्येक चतुर्भुज सतह,
वर्णक्रमीय प्रमेय द्वारा क्रमबद्ध}]\label{pb:b2:surfaces:1}
ए \emph{द्विघात}\index{द्विघात} शून्य निर्धारित है $\R^3$ एक का
डिग्री-दो बहुपद
\[
q(X) = X^{\mathsf T}\!AX + 2\,\langle b, X\rangle + c,
\qquad A \in \mathcal S_3(\R),\ A \neq 0,\ b \in \R^3,\
c \in \R .
\]
इस अध्याय की आकृतियों की सतहें---गोले,
दीर्घवृत्ताभ, काठी, शंकु, बेलन --- सभी हैं चतुर्भुज.
यह समस्या उन्हें वर्गीकृत करती है \emph{सभी}: वर्णक्रमीय प्रमेय
(\cref{thm:b2:quadratic:spectral}) \omterm{pb:b2:surfaces:1}{द्विघात} को सीधा करता है
भाग, \omterm{def:b2:affine:subspace}{एफ़िन} अनुवाद (\cref{ch:b2:affine}) को अवशोषित करें
रैखिक भाग, और जो बचता है वह एक छोटी, पूरी सूची है
सामान्य रूप.

\medskip
\noindent\textbf{Part I --- The reduction machine.}
\begin{enumerate}
  \item होने देना $ X = PY + t $ साथ $ P \in O(3)$ और $ t \in \R^3$ (ए
        निर्देशांक का कठोर परिवर्तन)। बताते हैं कि $ q(PY + t) =
        Y^{\mathsf T}\!A'Y + 2\langle b', Y\rangle + c'$ साथ
        \[
        A' = P^{\mathsf T}\!AP, \qquad
        b' = P^{\mathsf T}(At + b), \qquad
        c' = q(t) .
        \]
        यह निष्कर्ष निकालें कि \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} का $ A $ (इसलिए इसकी रैंक और
        \omterm{thm:b2:structures:signature}{हस्ताक्षर}) समीकरण का एक कठोर अपरिवर्तनीय है, और
        समझाएं कि किसी दिए गए का समीकरण क्यों है \omterm{pb:b2:surfaces:1}{द्विघात} केवल है
        एक शून्येतर अदिश गुणनखंड तक निर्धारित।
  \item वर्णक्रमीय प्रमेय का उपयोग करके दिखाएँ कि एक घूर्णन के बाद
        समीकरण बन जाता है $\sum_i \lambda_i y_i^2 +
        2\sum_i\beta_iy_i + c = 0$ साथ $\lambda_1, \lambda_2,
        \lambda_3$  eigenvalues का $ A $.
  \item हरएक के लिए $ i $ साथ $\lambda_i \neq 0$, अवशोषित करना
        $\beta_iy_i $ अनुवाद द्वारा ($ y_i \mapsto y_i -
        \beta_i/\lambda_i $). घटा हुआ समीकरण कब लिखें
        $\operatorname{rank} A = r $: $\sum_{i\leq r}\lambda_i
        z_i^2 + 2\sum_{i > r}\beta_iz_i + c'' = 0$.
  \item ए \emph{केंद्र} की \omterm{pb:b2:surfaces:1}{द्विघात} समीकरण का $ q = 0$ है
        एक बिंदु $\Omega $ साथ $ q(2\Omega - X) = q(X)$ सभी के लिए
        $ X $: बिंदु प्रतिबिंब में $\Omega $ को सुरक्षित रखता है
        समीकरण, इसलिए सतह। बताते हैं कि $ q(2\Omega - X)
        - q(X) = -4\langle A\Omega + b, X\rangle + 4\langle
        A\Omega + b, \Omega\rangle $, और निष्कर्ष निकालें: केंद्र
        \omterm{def:b2:series:def}{बिल्कुल} समाधान हैं $ A\Omega = -b $; वे
        मौजूद है अगर $ b \in \operatorname{im}A $, और केंद्र
        अद्वितीय है iff $ A $ उलटा है.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Central चतुर्भुज
($\operatorname{rank}A = 3$).} यहां घटा हुआ समीकरण है
$\lambda_1z_1^2 + \lambda_2z_2^2 + \lambda_3z_3^2 = \delta $.
\begin{enumerate}[resume]
  \item से गुणा करना $-1$ यदि आवश्यक हो, तो कम से कम दो मान लें
        $\lambda_i > 0$. संभावनाएं गिनाएं:
        \omterm{thm:b2:structures:signature}{हस्ताक्षर} $(3, 0)$ साथ $\delta > 0$, $= 0$, $< 0$,
        और \omterm{thm:b2:structures:signature}{हस्ताक्षर} $(2, 1)$ साथ $\delta > 0$, $= 0$, $<
        0$; छह परिणामी सेटों को नाम दें (दीर्घवृत्ताकार, बिंदु,
        खाली सेट, एक-शीट हाइपरबोलॉइड, शंकु, दो-शीट
        हाइपरबोलॉइड) और प्रत्येक को उसके यूक्लिडियन सामान्य में रखें
        फॉर्म ($\frac{x^2}{a^2} + \frac{y^2}{b^2} +
        \frac{z^2}{c^2} = 1$, वगैरह।)।
  \item वर्गीकृत करें $ x^2 + y^2 + z^2 + 4xy + 4yz + 4zx = 1$:
        दिखाओ $ A = 2J - I $ साथ $ J $ सभी मैट्रिक्स,
        की गणना करें \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} $\{5, -1, -1\}$, और पहचानें a
        अक्ष के परितः परिक्रमण की दो-शीट हाइपरबोलॉइड
        $\R(1,1,1)$.
  \item (नियम) एक-शीट हाइपरबोलॉइड के लिए
        $\frac{x^2}{a^2} + \frac{y^2}{b^2} - \frac{z^2}{c^2}
        = 1$, कारक
        \[
        \Bigl(\frac xa - \frac zc\Bigr)
        \Bigl(\frac xa + \frac zc\Bigr)
        = \Bigl(1 - \frac yb\Bigr)\Bigl(1 + \frac yb\Bigr)
        \]
        और सीधे दो एक-पैरामीटर परिवारों का उत्पादन करें
        सतह पर पड़ी रेखाएँ।
  \item इसे एक-पत्रक के प्रत्येक बिंदु के माध्यम से दिखाएँ
        हाइपरबोलॉइड प्रत्येक परिवार की ठीक एक पंक्ति से गुजरता है:
        सतह है \emph{दोगुना शासन किया}.
  \item  \emph{स्पर्शोन्मुख शंकु} एक शीट का
        हाइपरबोलाइड है $ C : \frac{x^2}{a^2} + \frac{y^2}{b^2}
        - \frac{z^2}{c^2} = 0$. साथ $\rho^2 = \frac{x^2}{a^2}
        + \frac{y^2}{b^2}$, दिखाओ कि कोई भी बिंदु
        हाइपरबोलॉइड अधिकतम दूरी पर है $ c\bigl(\rho -
        \sqrt{\rho^2 - 1}\bigr) = \frac{c}{\rho +
        \sqrt{\rho^2 - 1}}$ से $ C $, इसलिए सतह उसे गले लगा लेती है
        अनंत पर शंकु. के अनुभाग कौन-कौन से हैं
        समतलों द्वारा अतिपरवलयज $ x = \pm a $?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Rank $2$ and rank $1$:
paraboloids, cylinders, planes.}
\begin{enumerate}[resume]
  \item कल्पना करना $\operatorname{rank}A = 2$, कहना $\lambda_1,
        \lambda_2 \neq 0 = \lambda_3$. सवाल से शुरू
        3, के अनुसार दो मामलों में विभाजित $\beta_3 \neq 0$
        (कोई केंद्र नहीं, प्रश्न 4 से) या $\beta_3 = 0$ (एक पंक्ति
        केंद्रों की), और कम करें
        \[
        \lambda_1z_1^2 + \lambda_2z_2^2 + 2\beta_3z_3 = 0
        \qquad\text{or}\qquad
        \lambda_1z_1^2 + \lambda_2z_2^2 + c'' = 0 :
        \]
        पहले मामले में अण्डाकार/अतिपरवलयिक पैराबोलॉइड,
        केंद्रीय शांकवों (या के \omterm{def:b2:quadratic:adjoint}{जोड़}े) पर सिलेंडर
        प्रतिच्छेदी तल, एक रेखा, खाली सेट) में
        दूसरा.
  \item वह काठी दिखाओ $ z = xy $ अतिशयोक्तिपूर्ण है
        पैराबोलॉइड: द्वारा घुमाएँ $\pi/4$ में $ xy $-हवाई जहाज़ के लिए
        पहुंचें $ z = \tfrac12(u^2 - v^2)$, की सतह
        \cref{fig:b2:surfaces:saddle} पैमाने तक.
  \item वह काठी दिखाओ $ z = xy $ दो लाइन ले जाता है
        परिवार $\{x = x_0,\ z = x_0y\}$ और $\{y = y_0,\ z
        = xy_0\}$, प्रत्येक की \omterm{def:b2:series:def}{बिल्कुल} एक पंक्ति के साथ
        हर बिंदु: दूसरे ने दोगुना शासन किया \omterm{pb:b2:surfaces:1}{द्विघात}.
  \item वर्गीकृत करें $ x^2 + y^2 - 2x + 4y + 3 = 0$ में $\R^3$
        (वर्गों को \omterm{def:b2:metric:complete}{पूरा} करें; एक लंब वृत्तीय की पहचान करें
        सिलेंडर, और इसकी धुरी और त्रिज्या दें)।
  \item अब चलो $\operatorname{rank}A = 1$, कहना $\lambda_1
        \neq 0 = \lambda_2 = \lambda_3$. के भीतर घूम रहा है
        कर्नेल विमान और अनुवाद, कम करें
        \[
        \lambda_1z_1^2 + 2\beta z_2 = 0 \quad (\beta \neq 0)
        \qquad\text{or}\qquad
        \lambda_1z_1^2 + c'' = 0 :
        \]
        एक परवलयिक सिलेंडर, या समानांतर विमानों की एक \omterm{def:b2:quadratic:adjoint}{जोड़}ी, ए
        डबल प्लेन, या खाली सेट। वर्गीकृत करें $(x + y)^2 =
        z $ पूरी तरह से (सामान्य रूप, अनुवाद की धुरी
        अपरिवर्तनीय)।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The classification theorem.}
\begin{enumerate}[resume]
  \item भाग I--III को एक प्रमेय में एकत्रित करें: \emph{प्रत्येक
        \omterm{pb:b2:surfaces:1}{द्विघात} का $\R^3$ पर एक कठोर गति द्वारा मैप किया जाता है
        \omterm{def:b2:series:def}{बिल्कुल} एक सामान्य रूप}. सत्रह की सूची बनाओ \omterm{def:b2:affine:subspace}{एफ़िन}
        प्रकार (खाली वेरिएंट और पतित सेट की गिनती करें),
        और नौ में से एकल \omterm{pb:b2:surfaces:1}{द्विघात} \emph{सतह}:
        दीर्घवृत्ताकार, एक- और दो-शीट हाइपरबोलॉइड, शंकु,
        अण्डाकार और अतिशयोक्तिपूर्ण परवलय, अण्डाकार,
        अतिशयोक्तिपूर्ण और परवलयिक सिलेंडर.
  \item वर्गीकरण लिखिए \emph{एल्गोरिदम}: दिया गया $(A,
        b, c)$, आप किस मात्रा की गणना करते हैं, किसमें
        ऑर्डर, और कौन सी शाखा किस प्रकार का निर्णय लेती है? औचित्य सिद्ध करें
        कि प्रत्येक चरण प्रभावी है (eigenvalues एक का
        \omterm{def:b2:quadratic:adjoint}{सममित} $3\times3$ मैट्रिक्स, रैंक, सॉल्वैबिलिटी
        $ A\Omega = -b $).
  \item एल्गोरिथम चालू करें $ x^2 + y^2 + z^2 - 2xy - 2yz -
        2zx = 1$: दिखाओ $ A = 2I - J $ है \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} $\{2, 2,
        -1\}$ और निष्कर्ष निकालें: का एक-पत्रक हाइपरबोलाइड
        क्रांति के बारे में $\R(1,1,1)$.
  \item इसे चालू करें $ x^2 + y^2 - z^2 - 2x + 4y + 2z + 4 = 0$:
        केंद्र ढूंढें और पहचानें \omterm{pb:b2:surfaces:1}{द्विघात}.
  \item इसे चालू करें $ x^2 + 4xy + y^2 = 2z $: को विकर्णित करें
        $ xy $-अवरोध पैदा करना ($ u = \frac{x+y}{\sqrt2}$, $ v =
        \frac{x-y}{\sqrt2}$) और पहचानें \omterm{pb:b2:surfaces:1}{द्विघात}.
  \item यूक्लिडियन बनाम \omterm{def:b2:affine:subspace}{एफ़िन}. दिखाएँ कि दो केन्द्रीय
        चतुर्भुज सामान्य रूप में हैं \emph{सख्ती से} \omterm{def:b2:nvs:equivalent}{समकक्ष}
        यदि उनके पास समान गुणांक सूचियाँ हैं (तक)।
        क्रमपरिवर्तन और पर एक सामान्य सकारात्मक अदिश
        समीकरण), जबकि \emph{स्नेहपूर्वक}, केवल \omterm{thm:b2:structures:signature}{हस्ताक्षर}
        डेटा जीवित रहता है: प्रत्येक दीर्घवृत्त एक है \omterm{def:b2:affine:subspace}{एफ़िन} की छवि
        गोल गोला. कौन सा प्रमेय गारंटी देता है कि
        रास्ते में \omterm{thm:b2:structures:signature}{हस्ताक्षर} नहीं बदल सकते
        (\cref{thm:b2:quadratic:sylvester})?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends.}
\begin{enumerate}[resume]
  \item दिखाएँ कि प्रत्येक अनुभाग a \omterm{pb:b2:surfaces:1}{द्विघात} एक द्वारा \omterm{def:b2:affine:subspace}{एफ़िन}
        समतल उस समतल का एक शंकु (संभवतः पतित) है।
        काठी के अनुभागों को पहचानें $ z = xy $ द्वारा
        विमान $ z = c $ ($ c \neq 0$ और $ c = 0$).
  \item कौन \omterm{pb:b2:surfaces:1}{द्विघात} सतहों पर सीधी रेखाएँ होती हैं? दिखाओ
        वह दीर्घवृत्ताकार, दो-शीट हाइपरबोलॉइड और
        अण्डाकार पैराबोलॉइड में कोई नहीं होता है \emph{(प्रतिबंध लगाना $ q $
        एक पंक्ति में जाएँ और इसके लिए कॉची-श्वार्ज़ असमानता का उपयोग करें
        दो-पत्रक मामला)}; वह शंकु और बेलन हैं
        एक परिवार द्वारा शासित; और निष्कर्ष निकालें कि दोगुना
        शासन किया \omterm{pb:b2:surfaces:1}{द्विघात} सतहें \omterm{def:b2:series:def}{बिल्कुल} एक-शीट वाली हैं
        हाइपरबोलॉइड और हाइपरबोलिक पैराबोलॉइड।
  \item जब केवल \emph{\omterm{def:b2:affine:subspace}{एफ़िन}} प्रकार चाहिए, गॉस का
        कमी (\cref{thm:b2:quadratic:gauss}) सस्ता है
        विकर्णीकरण से। गॉस के साथ प्रश्न 17 को दोबारा करें
        एल्गोरिदम और \omterm{thm:b2:structures:signature}{हस्ताक्षर} की जाँच करें $(2, 1)$; क्या
        क्या गॉस यूक्लिडियन जानकारी खो देता है?
  \item सभी eigenvalues एक का \omterm{def:b2:quadratic:adjoint}{सममित} मैट्रिक्स वास्तविक हैं; दिखाओ
        जिसके परिणामस्वरूप \emph{लक्षण} की eigenvalues
        का $ A $ से पढ़ा जा सकता है \omterm{def:b2:reduction:charpoly}{विशेषता बहुपद}
        डेसकार्टेस के संकेतों के नियम के अनुसार, और इसे सत्यापित करें
        प्रश्न 6: $\chi_A(\lambda) = \lambda^3 - 3\lambda^2
        - 9\lambda - 5$ इसलिए, \omterm{def:b2:series:def}{बिल्कुल} एक चिह्न परिवर्तन हुआ है
        \omterm{thm:b2:structures:signature}{हस्ताक्षर} $(1, 2)$.
  \item संश्लेषण। एल्गोरिदम को कुछ पंक्तियों में सारांशित करें;
        (i) वर्णक्रम द्वारा निभाई गई सटीक भूमिका बताएं
        प्रमेय, (ii) केंद्र समीकरण $ A\Omega = -b $,
        (iii) सिल्वेस्टर का जड़त्व प्रमेय, (iv) गॉस का
        कमी. एक ही मशीन से क्या मिलता है?
        $\R^2$, और इसमें क्या परिवर्तन होता है $\R^n$?
\end{enumerate}
\end{problem}
