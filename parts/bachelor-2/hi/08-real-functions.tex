तीक्ष्ण और परीक्षकों का प्रिय।

\section{मोनोटोन कार्य}

\begin{theorem}[मोनोटोन कार्यों की नियमितता]\label{thm:b2:realfun:monotone}
होने देना $f \colon I \to \R $ एक अंतराल पर बढ़ता जा रहा है।
\begin{enumerate}
  \item हर आंतरिक बिंदु पर $ a $, एकतरफ़ा सीमाएँ मौजूद हैं:
        \[
        f(a^-) = \sup_{x < a} f(x) \;\leq\; f(a) \;\leq\;
        f(a^+) = \inf_{x > a} f(x) ;
        \]
        हर असंततता एक है \emph{कूदना}.
  \item के असंततता का समुच्चय $ f $ अधिकतम है
        \omterm{def:b2:structures:countable}{गणनीय}.\index{मोनोटोन फ़ंक्शन!असंतोष}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) सेट $\{f(x) : x < a\}$ शून्य नहीं है, ऊपर से घिरा हुआ है $ f(a)$:
यह सर्वोच्च है $ s $ संतुष्ट $ f(x) \to s $ जैसा $ x \to a^-$ (दिया गया
$\varepsilon $, कुछ $ f(x_0) > s - \varepsilon $, और एकरसता
जाल $ f(x) \in \intoc{s - \varepsilon}{s}$ के लिए $ x \in
\intoo{x_0}{a}$). सममित रूप से दाहिनी ओर।

(2) प्रत्येक असंततता को $ a $ गैर-रिक्त को संलग्न करें \omterm{def:b2:metric:topology}{खुला} अंतराल
$ J_a = \intoo{f(a^-)}{f(a^+)}$ (एक वास्तविक छलांग)। के लिए $ a < b $
असंततता, $ J_a $ और $ J_b $ असंयुक्त हैं: $ f(a^+) \leq f(c)
\leq f(b^-)$ किसी के लिए $ c $ बीच में। प्रत्येक $ J_a $ एक तर्कसंगत शामिल है;
अलग-अलग असंततताओं को अलग-अलग तर्कसंगतताएं मिलती हैं: का एक इंजेक्शन
असंततता स्थापित हो गई $\Q $, जो है \omterm{def:b2:structures:countable}{गणनीय}
(\cref{prop:b2:structures:countablestable}).
\end{proof}

\begin{example}
सीमा तीव्र है: एक गणना ठीक करें $(r_n)$ का $\Q \cap
\intoo{0}{1}$ और सेट करें $ f(x) = \sum_{n : r_n \leq x} 2^{-n}$
(एक सारांश-परिवार परिभाषा,
\cref{def:b2:series:summable}). तब $ f $ बढ़ता जा रहा है
$\intcc{0}{1}$ और प्रत्येक तर्कसंगत पर \omterm{def:b2:series:def}{बिल्कुल} असंतत
$\intoo{0}{1}$ (कूदना $2^{-n}$ पर $ r_n $): एक मोनोटोन फ़ंक्शन
\emph{कर सकना} एक घने पर असंतुलित हो \omterm{def:b2:structures:countable}{गणनीय} समुच्चय.
\end{example}

\begin{example}[छलांग वृद्धि से अधिक नहीं हो सकती]\label{ex:b2:realfun:jumpbudget}
के लिए $ f $ बढ़ता जा रहा है $\intcc{a}{b}$, जंप का एक बजट होता है:
यदि $ a < c_1 < \dots < c_m < b $ छलांग के साथ असंततताएं हैं
$ s_i = f(c_i^+) - f(c_i^-) > 0$, फिर इंटरलेसिंग चुनना
अंक $ a < c_1 < t_1 < c_2 < \dots $ और एकरसता का उपयोग कर रहे हैं
प्रत्येक टुकड़ा,
\[
\sum_{i=1}^{m} s_i \;\leq\; f(b) - f(a) :
\]
कुल चढ़ाई कुल छलांग को सीमित करती है। परिणाम: के लिए
प्रत्येक $ k $, अधिक से अधिक $ k\,\bigl(f(b) - f(a)\bigr)$ असंततता
छलांग लगाओ $\geq \frac1k $ --- का एक मात्रात्मक शोधन
\cref{thm:b2:realfun:monotone}~(2), चूंकि असंततता सेट है
है \omterm{def:b2:structures:countable}{गणनीय} संघ ख़त्म $ k $ इन परिमित समुच्चयों में से. पर
उपरोक्त तर्कसंगत-कूद फ़ंक्शन, बजट \omterm{def:b2:series:def}{बिल्कुल} खर्च किया जाता है:
छलांग $2^{-n}$ का योग $1 = f(1^+) - f(0^-)$ स्पष्ट में
विस्तारित अर्थ. मोनोटोन फ़ंक्शंस सघन रूप से कूद सकते हैं, लेकिन केवल
सख्त भत्ते पर.
\end{example}

\section{उत्तल कार्य}

\begin{lemma}[ढलान असमानता]\label{lem:b2:realfun:slopes}
होने देना $ f $ पर उत्तल हो $ I $ और $ x < y < z $ में $ I $. तब
\[
\frac{f(y) - f(x)}{y - x}
\;\leq\; \frac{f(z) - f(x)}{z - x}
\;\leq\; \frac{f(z) - f(y)}{z - y} :
\]
दोनों समापन बिंदुओं पर जीवाओं का ढलान बढ़ जाता है।\index{उत्तल
कार्य!ढलान असमानता}
\end{lemma}

\begin{proof}
लिखना $ y = \frac{z - y}{z - x}\,x + \frac{y - x}{z - x}\,z $: ए
उत्तल संयोजन, चूंकि दो गुणांक सकारात्मक हैं और
का योग $1$. उत्तलता देता है
\[
f(y) \;\leq\; \frac{z-y}{z-x}\,f(x) + \frac{y-x}{z-x}\,f(z).
\]
बाएँ असमानता के लिए, घटाएँ $ f(x)$ दोनों तरफ से, उपयोग करके
$\frac{z-y}{z-x} - 1 = -\frac{y-x}{z-x}$:
\[
f(y) - f(x) \leq \frac{y - x}{z - x}\bigl(f(z) - f(x)\bigr),
\]
और से विभाजित करें $ y - x > 0$. सही असमानता के लिए, घटाएँ
इसके बजाय से $ f(z)$:
\[
f(z) - f(y) \geq f(z) - \frac{z-y}{z-x}f(x) -
\frac{y-x}{z-x}f(z)
= \frac{z - y}{z - x}\bigl(f(z) - f(x)\bigr),
\]
और से विभाजित करें $ z - y > 0$. प्रदर्शित दोनों चरण समान हैं
बैरीसेंट्रिक पहचान एक अलग समापन बिंदु के विरुद्ध पढ़ी जाती है।
\end{proof}

\begin{theorem}[उत्तल कार्यों की नियमितता]\label{thm:b2:realfun:convexreg}
होने देना $ f $ एक अंतराल पर उत्तल हो $ I $.
\begin{enumerate}
  \item प्रत्येक आंतरिक बिंदु पर, $ f $ परिमित एकतरफ़ा है
        व्युत्पन्न $ f'_g \leq f'_d $; दोनों बढ़ते हुए कार्य हैं
        मुद्दे की बात; विशेष रूप से $ f $ है \omterm{def:b2:metric:continuity}{निरंतर} पर
        का आंतरिक भाग $ I $ (लेकिन संभवतः अंतिम बिंदु पर नहीं)।
  \item $ f $ प्रत्येक के ऊपर स्थित है \emph{समर्थन लाइन}: के लिए $ a $ आंतरिक
        और कोई भी $ m \in \intcc{f'_g(a)}{f'_d(a)}$,
        \[
        f(x) \geq f(a) + m(x - a) \qquad (x \in I).
        \]
  \item (जेन्सेन, भारित) के लिए $ x_i \in I $ और वजन $\lambda_i
        \geq 0$, $\sum\lambda_i = 1$:
        \[
        f\Bigl(\sum_i \lambda_i x_i\Bigr) \leq \sum_i \lambda_i
        f(x_i) .\index{जेन्सेन की असमानता}
        \]
\end{enumerate}
\end{theorem}

\begin{proof}
(1) ठीक करें $ a $ आंतरिक भाग। द्वारा \cref{lem:b2:realfun:slopes}, ढाल
$\tau(h) = \frac{f(a + h) - f(a)}{h}$ का बढ़ता हुआ कार्य है
$ h $ (दोनों तरफ, और $\tau(h_-) \leq \tau(h_+)$ के लिए $ h_- < 0 <
h_+$). इस तरह $\tau $ की एक सीमित सीमा होती है $ h \to 0^-$ (बढ़ते हुए,
ऊपर किसी दाएँ ढलान से घिरा हुआ) --- यह है $ f'_g(a)$ --- और के रूप में
$ h \to 0^+$ ($ f'_d(a)$), साथ $ f'_g(a) \leq f'_d(a)$. परिमित
एकतरफ़ा व्युत्पन्न बल निरंतरता पर $ a $. में एकरसता
बिंदु: के लिए $ a < b $ आंतरिक भाग, $ f'_d(a) \leq \frac{f(b) - f(a)}{b -
a} \leq f'_g(b)$, फिर से ढलान असमानता से।

(2) के लिए $ x > a $: $\frac{f(x) - f(a)}{x - a} \geq f'_d(a) \geq m $;
के लिए $ x < a $: $\frac{f(a) - f(x)}{a - x} \leq f'_g(a) \leq m $. दोनों
दावे को पुनर्व्यवस्थित करें।

(3) वर्ष 1 की तरह ही अंकों की संख्या पर प्रेरण
आयतन (दो-बिंदु मामला परिभाषा है) --- या एक झटके में:
(2) पर लागू करें $ a = \sum\lambda_i x_i $ और समर्थन-रेखा को औसत करें
बिन्दुओं पर असमानताएँ $ x_i $ वज़न के साथ $\lambda_i $: $\sum_i
\lambda_i f(x_i) \geq f(a) + m\sum_i\lambda_i(x_i - a) = f(a)$.
\end{proof}

\begin{example}[समापन बिंदु असंततता]
पर $\intcc{0}{1}$, समारोह $ f(0) = 1$, $ f(x) = 0$ के लिए $ x > 0$
उत्तल है लेकिन अंतिम बिंदु पर असंतत है $0$: कथन (1) है
तेज़.
\end{example}

\begin{example}[समर्थन रेखाओं के कोने और ढेर]\label{ex:b2:realfun:corner}
के लिए $ f(x) = \abs x $ पर $ a = 0$: एकतरफ़ा व्युत्पन्न हैं
$ f'_g(0) = -1$ और $ f'_d(0) = +1$, और
\cref{thm:b2:realfun:convexreg}~(2) के लिए एक समर्थन लाइन सौंपता है
\emph{प्रत्येक} ढलान $ m \in \intcc{-1}{1}$:
\[
\abs x \geq m\,x \qquad (x \in \R,\ -1 \leq m \leq 1),
\]
प्रत्येक एक समानता \omterm{def:b2:series:def}{बिल्कुल} आधी रेखा पर या पर $0$. एक उत्तल
फ़ंक्शन अलग-अलग है $ a $ ठीक जब पूला
एक पंक्ति में सिमट जाता है ($ f'_g(a) = f'_d(a)$); कोने ले जाते हैं
स्पर्शरेखाओं का अंतराल. यह पुलिंदा परिमित-आयामी है
का रोगाणु \emph{उपविभेदक} उत्तल अनुकूलन का ---
और कारण उत्तल कार्य इतने मजबूत हैं: यहां तक कि जहां भी
व्युत्पन्न विफल हो जाता है, सहायक ज्यामिति बची रहती है, बस इतना ही
जेन्सेन के प्रमाण का उपयोग किया गया।
\end{example}

\begin{example}[शक्ति का मतलब असमानता]\label{ex:b2:realfun:powermeans}
के लिए $0 < p < q $ और सकारात्मक $ x_i $ वज़न के साथ $\lambda_i $
संक्षेप में $1$, जेन्सेन को उत्तल पर लगाना $ t \mapsto t^{q/p}$
बिंदुओं पर $ x_i^p $:
\[
\Bigl(\sum \lambda_i x_i^{p}\Bigr)^{1/p}
\leq \Bigl(\sum \lambda_i x_i^{q}\Bigr)^{1/q} :
\]
शक्ति का अर्थ है घातांक --- युक्त AM--QM के साथ वृद्धि, और,
सीमा में $ p \to 0$ (\cref{exo:b2:realfun:6}), एएम - जीएम
एक बार फिर असमानता.
\end{example}

\begin{example}[अधिकतम एन्ट्रापी]\label{ex:b2:realfun:entropy}
संभाव्यता वेक्टर के लिए $(p_1, \dots, p_n)$ (सकारात्मक,
संक्षेप में $1$), एन्ट्रापी $ H(p) = -\sum_i p_i\ln p_i $
संतुष्ट
\[
H(p) \leq \ln n ,
\qquad\text{with equality iff } p_i = \frac1n \text{ for all }
i .
\]
जेन्सेन द्वारा प्रमाण (\cref{thm:b2:realfun:convexreg}~(3)) पर लागू किया गया
द \emph{नतोदर} $\ln $ वज़न के साथ $ p_i $ बिंदुओं पर
$\frac{1}{p_i}$:
\[
H(p) = \sum_i p_i\ln\frac{1}{p_i}
\leq \ln\Bigl(\sum_i p_i\,\frac1{p_i}\Bigr) = \ln n ,
\]
समानता सभी बिंदुओं पर बल देती है $\frac1{p_i}$ बराबर (सख्त)
अवतलता), यानी\ $ p $ वर्दी। समान रूप से, यह है
\cref{exo:b2:realfun:7} साथ $ q $ वर्दी। अनिश्चितता है
अज्ञानता द्वारा अधिकतम समान रूप से फैलाया गया --- परिवर्तनशील
कोडिंग, सांख्यिकीय यांत्रिकी और एन्ट्रापी के पीछे सिद्धांत
की उपस्थिति \cref{ch:b2:randomvar}.
\end{example}

\begin{method}[असमानता के पीछे उत्तल फलन ढूँढना]\label{met:b2:realfun:spotting}
अधिकांश शास्त्रीय असमानताएं जेन्सेन की पोशाक में हैं; कपड़े उतारना
एक: (1) सामान्य करें ताकि ए \emph{भारित औसत} प्रकट होता है
(वजन सकारात्मक है, संक्षेप में $1$ --- यदि कुल द्रव्यमान से विभाजित करें
आवश्यक); (2) देखें कि बनाम के अंदर कौन सा फ़ंक्शन लागू होता है
औसत से बाहर: दावा ``$ f(\text{average}) \leq $
का औसत $ f $'' उत्तल को नाम देता है $ f $; (3) द्वारा उत्तलता प्रमाणित करें
दूसरा व्युत्पन्न, और कठोरता के माध्यम से समानता को संभालें; (4)
यदि कोई औसत दिखाई नहीं दे रहा है, तो पहले लघुगणक लें --- उत्पाद
और शक्तियाँ औसत हो जाती हैं, और की अवतलता $\ln $ ले जाता है
एएम--जीएम, यंग और उनके रिश्तेदार (इस अध्याय का सप्ताहांत
समस्या उनमें से प्रत्येक पर चरण 1--4 चलाती है)। यदि लघुगणक भी करते हैं
औसत प्रकट न करें, असमानता को पढ़ने का प्रयास करें
ढलानों की एकरसता (\cref{lem:b2:realfun:slopes}) ---
सुपरएडिटिविटी कथन जैसे \cref{exo:b2:realfun:9} जीना
वहाँ.
\end{method}

\begin{remark}[सामान्य ख़तरे]\label{rem:b2:realfun:pitfalls}
(i) उत्तलता उत्पादों द्वारा संरक्षित नहीं है: $ x $ और $(x - 1)^2$
पर उत्तल हैं $\intcc{0}{2}$, लेकिन उनका उत्पाद $ x(x-1)^2$ है
दूसरा व्युत्पन्न $6x - 4$, नकारात्मक पर
$\intco{0}{\frac23}$ --- उत्तल नहीं; न ही उत्तलता संरक्षित है
एकरसता के बिना रचना द्वारा
(\cref{exo:b2:realfun:10}). (ii) जेन्सेन अवतल के लिए फ़्लिप करता है
कार्य: आधी शास्त्रीय असमानताएँ अवतल हैं
$\ln $-संस्करण; उत्तल रूप को लागू करना $\ln $ है
AM--GM सिद्ध करने का सबसे तेज़ तरीका \emph{पीछे की ओर}. (iii) मध्यबिंदु
केवल उत्तलता का अर्थ उत्तलता नहीं है --- निरंतरता (या
मात्र सीमाबद्धता) की आवश्यकता है (\cref{exo:b2:realfun:8}); द
पैथोलॉजिकल प्रति उदाहरण इस पुस्तक के सिद्धांतों से परे रहते हैं। (iv)
एक पर एक उत्तल कार्य \emph{\omterm{def:b2:metric:topology}{खुला}} अंतराल है \omterm{def:b2:metric:continuity}{निरंतर},
यहां तक कि स्थानीय स्तर पर भी \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़} (\cref{exo:b2:realfun:12}); पर
अंतिम बिंदु, कुछ भी मुफ़्त नहीं है। (v) डेरिवेटिव्स डार्बोक्स का पालन करते हैं लेकिन
होने की आवश्यकता नहीं है \omterm{def:b2:metric:continuity}{निरंतर} (\cref{ex:b2:realfun:oscillation}):
``$ f'$ इसमें कोई छलांग नहीं है'' का अर्थ कभी भी ``'' नहीं होता $ f'$ है \omterm{def:b2:metric:continuity}{निरंतर}''.
\end{remark}

\section{डारबौक्स संपत्ति}

\begin{theorem}[डार्बौक्स]\label{thm:b2:realfun:darboux}
होने देना $ f $ एक अंतराल पर भिन्न होना $ I $. तब $ f'$ हर लेता है
इसके किन्हीं दो मानों के बीच का मान --- भले ही $ f'$ होने की आवश्यकता नहीं है
\omterm{def:b2:metric:continuity}{निरंतर}.\index{डार्बौक्स का प्रमेय}
\end{theorem}

\begin{proof}
होने देना $ a < b $ में $ I $ और $ v $ सख्ती से बीच में $ f'(a)$ और $ f'(b)$,
कहना $ f'(a) < v < f'(b)$. समारोह $ g(x) = f(x) - vx $ है
के साथ भिन्न $ g'(a) < 0 < g'(b)$: यह न्यूनतम है
$\intcc{a}{b}$ (प्राप्त: निरंतरता एक पर \omterm{def:b2:metric:compact}{कॉम्पैक्ट}) पर नहीं है $ a $
(तुरन्त बाद $ a $, $ g $ नीचे घट जाती है $ g(a)$) न ही पर $ b $ (से ठीक पहले $ b $, $ g $ नीचे है $ g(b)$): यह आंतरिक है, और वहाँ है $ g'(c)
= 0$, यानी\ $ f'(c) = v $. (यह प्रथम वर्ष का तारांकित अभ्यास था; यह
सिद्धांत में स्थान यहाँ है।)
\end{proof}

\begin{example}[कौन से फ़ंक्शन व्युत्पन्न हैं?]\label{ex:b2:realfun:whichderivatives}
डार्बौक्स का प्रमेय एक गैर-अस्तित्व वाली मशीन है। फर्श का कार्य
$\lfloor x\rfloor $ किसी भी फ़ंक्शन का व्युत्पन्न नहीं है
$\R $: यह मान लेता है $0$ और $1$ लेकिन छूट जाता है $\frac12$ पर
$\intcc{0}{1}$, कौन \cref{thm:b2:realfun:darboux} के लिए मना करता है
व्युत्पन्न। एक ही फैसला हर कार्य पर उछाल मारता है
--- साइन, हेविसाइड, सभी चरण कार्य --- हालांकि निर्दोष
वे देखते हैं; उनके ``एंटीडेरिवेटिव्स'' ($\abs x $ संकेत आदि के लिए)
केवल एक कोने के साथ छलांग और गाँठ से दूर मौजूद हैं।
कंट्रास्ट: बेतहाशा असंतत $ f'$ का
\cref{ex:b2:realfun:oscillation} \emph{है} एक व्युत्पन्न --- इसका
असंततता एक दोलन है, जिसे डार्बौक्स सहन करता है। द
दोनों व्यवहारों के बीच की सीमा \omterm{def:b2:series:def}{बिल्कुल} वही है
नीचे नो-जंप का परिणाम है।
\end{example}

\begin{corollary}\label{cor:b2:realfun:nojumps}
एक व्युत्पन्न में कोई जम्प असंततता नहीं है: यदि $ f'(a^-)$ और
$ f'(a^+)$ मौजूद हैं, वे बराबर हैं $ f'(a)$. ए की असंततता
व्युत्पन्न हमेशा दोलन प्रकार के होते हैं ($ x^2\sin\frac1x $का
पर व्युत्पन्न $0$, वर्ष 1 खंड).
\end{corollary}

\begin{proof}
अगर $ f'(a^+) = \lim_{x\to a^+} f'(x)$ मौजूद है और इससे भिन्न है
$ f'(a)$, उनके बीच सख्ती से मूल्यों को छोड़ दिया जाएगा $ f'$ पर
एक सही पड़ोस --- अंतराल पर डार्बोक्स का खंडन करता है
$\intcc{a}{a + h}$. (वैकल्पिक रूप से: माध्य मान प्रमेय बल
$ f'(a) = \lim_{h\to0^+} \frac{f(a+h)-f(a)}{h} = f'(a^+)$, द
अंतर भागफल एक है $ f'$-एक मध्यवर्ती बिंदु पर मूल्य।)
बायीं ओर भी वैसा ही.
\end{proof}

\begin{example}[विहित दोलन व्युत्पन्न]\label{ex:b2:realfun:oscillation}
होने देना $ f(x) = x^2\sin\frac1x $ के लिए $ x \neq 0$ और $ f(0) = 0$. पर
$0$: $\bigl|\frac{f(h) - f(0)}{h}\bigr| = \abs{h\sin\frac1h}
\leq \abs h \to 0$, इसलिए $ f'(0) = 0$ मौजूद है. से दूर $0$,
\[
f'(x) = 2x\sin\frac1x - \cos\frac1x ,
\]
जिसका प्रथम पद प्रवृत्त होता है $0$ जबकि $\cos\frac1x $ दोलन करता है
के माध्यम से $\intcc{-1}{1}$ हर अंतराल पर $\intoo{0}{\delta}$: द
सीमा $ f'(0^+)$ मौजूद नहीं होना। इसलिए $ f'$ लेकिन हर जगह परिभाषित किया गया है
पर असंतत $0$ --- और, \omterm{def:b2:series:def}{बिल्कुल} वैसा ही
\cref{cor:b2:realfun:nojumps} भविष्यवाणी करता है, असंततता एक है
दोलन, छलांग नहीं: प्रत्येक पर $\intoo{0}{\delta}$, $ f'$ अभी भी
चारों ओर एक पूर्ण अंतराल स्वीप करता है $0$. डेरिवेटिव जंगली हो सकते हैं, लेकिन
केवल डार्बौक्स-संगत तरीके से।
\end{example}

\begin{remark}[जहाँ इस अध्याय का प्रयोग किया जाता है]
उत्तलता असमानता उद्योग का इंजन है: यह
अध्याय की सप्ताहांत समस्या यंग, एच का निर्माण करती है\"वृद्ध, मिन्कोव्स्की
और इससे शक्ति-माध्य श्रृंखला, जो \cref{ch:b2:nvs}'एस आदर्श
सिद्धांत और \cref{ch:b2:integration}का अभिन्न अनुमान उपभोग करता है;
जेन्सेन संभाव्यता में पल की असमानताओं के रूप में फिर से प्रकट होता है
\cref{ch:b2:randomvar}. एकस्वर नियमितता लौट आती है
\cref{ch:b2:integration} (मोनोटोन फ़ंक्शन पूर्णांक हैं) और,
वर्ष 3 खंड में, लगभग हर जगह भिन्नता के रूप में
मोनोटोन फ़ंक्शंस --- जहां ``\omterm{def:b2:structures:countable}{गणनीय} कई छलाँगें'' बन जाती हैं
लेब्सग्यू के सिद्धांत का पहला चरण।
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b2:realfun:1}
असंततता सेट और जंप आकार निर्धारित करें: $\lfloor x
\rfloor $; $\;x - \lfloor x\rfloor $; $\;\lfloor x \rfloor + \sqrt{x
- \lfloor x\rfloor}$; निम्नलिखित उदाहरण का कार्य
\cref{thm:b2:realfun:monotone} डायडिक परिमेय तक ही सीमित है
$ r_n $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:realfun:2}
सिद्ध कीजिए कि यह एक बढ़ता हुआ फलन है $ f \colon I \to \R $ के साथ
मध्यवर्ती मूल्य संपत्ति (किसी भी उपअंतराल की इसकी छवि एक है
अंतराल) है \omterm{def:b2:metric:continuity}{निरंतर}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:realfun:3}
निम्नलिखित में से कौन अपने डोमेन पर उत्तल हैं? $ x \mapsto
x\ln x $ ($ x > 0$); $\;x \mapsto \ln(1 + \eu^x)$; $\;x \mapsto
\sqrt{1 + x^2}$; $\;x \mapsto x^3$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:realfun:4}
होने देना $ f $ पर उत्तल हो $\R $ और ऊपर से घिरा हुआ है. साबित करें कि $ f $ है
स्थिर. \emph{(अगर $ f(a) \neq f(b)$, ढलान असमानता
गैर-शून्य कॉर्ड ढलान का प्रचार करता है: बड़े बिंदु से परे
मूल्य, $ f $ कम से कम रैखिक रूप से बढ़ता है --- सीमा के विपरीत।
ढलान के दोनों लक्षणों का इलाज करें।)} यह निष्कर्ष निकालें कि उत्तल फलन चालू है
$\R $ दोनों सिरों पर एक अनंतस्पर्शी के साथ एफ़िन है।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:realfun:5}
होने देना $ f $ पर भिन्न होना $ I $ साथ $ f'$ नीरस. साबित करें कि
$ f'$ है \omterm{def:b2:metric:continuity}{निरंतर} \emph{(मिलाना
\cref{thm:b2:realfun:monotone} और
\cref{cor:b2:realfun:nojumps})}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:realfun:6}
(एक सीमा के रूप में ज्यामितीय माध्य) सकारात्मक के लिए $ x_i $ और वजन
$\lambda_i $ संक्षेप में $1$, सिद्ध करना
\[
\lim_{p \to 0^+} \Bigl(\sum_i \lambda_i x_i^p\Bigr)^{1/p}
= \prod_i x_i^{\lambda_i} ,
\]
के जरिए $ x_i^p = \eu^{p\ln x_i} = 1 + p\ln x_i + O(p^2)$, और निष्कर्ष निकालें
भारित AM--GM असमानता
\cref{ex:b2:realfun:powermeans}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:realfun:7}
(एंट्रॉपी असमानता) की सख्त उत्तलता का उपयोग करना $ t \mapsto t\ln t $,
इसे सकारात्मक रूप से सिद्ध करें $ p_i, q_i $ साथ $\sum p_i = \sum q_i = 1$:
\[
\sum_i p_i \ln\frac{p_i}{q_i} \geq 0 ,
\]
समानता के साथ यदि $ p = q $. \emph{(बायीं ओर को इस प्रकार लिखें $\sum q_i\,
\varphi\bigl(\frac{p_i}{q_i}\bigr)$ साथ $\varphi(t) = t\ln t $ और
जेन्सेन को वज़न के साथ लगाएं $ q_i $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:realfun:8}
(मध्यबिंदु उत्तलता) $ f \colon I \to \R $ है \emph{मध्यबिंदु उत्तल}
कब $ f\bigl(\frac{x+y}{2}\bigr) \leq \frac{f(x) + f(y)}{2}$
हमेशा। साबित करें कि ए \emph{\omterm{def:b2:metric:continuity}{निरंतर}} मध्यबिंदु उत्तल फलन है
उत्तल. \emph{(डायडिक के लिए उत्तलता असमानता स्थापित करें
वजन $\frac{k}{2^m}$ पर प्रेरण द्वारा $ m $, फिर सीमा तक जाएं
घनत्व का उपयोग करना और निरंतरता.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:realfun:9}
होने देना $ f $ पर उत्तल हो $\intco{0}{+\infty}$ साथ $ f(0) \leq 0$. साबित करें कि $ x \mapsto \frac{f(x)}{x}$ बढ़ता जा रहा है
$\intoo{0}{+\infty}$, और उत्तल के लिए इसे घटाएं $ f $ साथ $ f(0) =
0$: $ f(x + y) \geq f(x) + f(y)$ के लिए $ x, y \geq 0$
(सुपरएडिटिविटी)।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:realfun:10}
होने देना $ f $ पर उत्तल हो $ I $ और $ g $ उत्तल \emph{की बढ़ती} एक पर
अंतराल युक्त $ f(I)$. साबित करें कि $ g \circ f $ उत्तल है, और
एक प्रति उदाहरण द्वारा दिखाएँ कि एकरसता $ g $ नहीं हो सकता
गिरा दिया.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:realfun:11}
(हर्माइट--हैडमार्ड) चलो $ f $ उत्तल हो और \omterm{def:b2:metric:continuity}{निरंतर} पर
$\intcc{a}{b}$. सिद्ध करना
\[
f\Bigl(\frac{a+b}{2}\Bigr) \;\leq\; \frac{1}{b -
a}\int_a^b f(t)\,\dd t \;\leq\; \frac{f(a) + f(b)}{2} .
\]
\emph{(बाएं: मध्यबिंदु पर एक समर्थन लाइन एकीकृत करें। दाएं:
बाध्य $ f $ तार से.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:realfun:12}
सिद्ध कीजिए कि a पर उत्तल फलन है \emph{\omterm{def:b2:metric:topology}{खुला}} अंतराल $ I $ है
स्थानीय स्तर पर \omterm{def:b2:metric:continuity}{लिप्सचिट्ज़}: प्रत्येक खंड के लिए $\intcc{a}{b} \subseteq I $
और मार्जिन $\delta > 0$ साथ $\intcc{a - \delta}{b + \delta}
\subseteq I $, का प्रतिबंध $ f $ को $\intcc{a}{b}$ है
\omterm{def:b2:metric:continuity}{लिप्सचिट्ज़}, स्थिरांक के साथ
$\max\Bigl(\bigl|\frac{f(a) - f(a - \delta)}{\delta}\bigr|,
\bigl|\frac{f(b + \delta) - f(b)}{\delta}\bigr|\Bigr)$
\emph{(इन दोनों के बीच प्रत्येक कॉर्ड ढलान को ढलान से फंसाएं
असमानता)}.
\end{exercise}

\section{समस्या: उत्तलता टूलबॉक्स}

एक परिभाषा --- ग्राफ़ के ऊपर का तार --- उत्पन्न करता है
शास्त्रीय असमानताओं का संपूर्ण टूलबॉक्स। इस सप्ताहांत समस्या
इसे तार्किक क्रम में बनाता है: उत्तलता मानदंड और सख्त जेन्सेन,
फिर यंग, एच\"वृद्ध और मिन्कोवस्की (का जन्म प्रमाण पत्र)
$ p $-मानदंड), न्यूनतम से पूर्ण शक्ति-माध्य श्रृंखला
अधिकतम, और दो मुकुट लाभांश --- कार्लेमैन की असमानता, और
एच\"पुराने को द्वंद्व के रूप में पढ़ा जाता है। सब कुछ सिद्ध है; कुछ भी नहीं है
आयातित.

\begin{problem}[{सप्ताहांत समस्या --- यंग, एच\"वृद्ध, मिन्कोव्स्की,
और शक्ति-माध्य श्रृंखला}]
\label{pb:b2:realfun:1}
लगातार, $ p, q > 1$ हैं \emph{संयुग्म घातांक}: $\frac1p +
\frac1q = 1$; वेक्टर हैं $ a = (a_1, \dots, a_n) \in \R^n $;
तौल $\lambda_i > 0$ संतुष्ट $\sum_i\lambda_i = 1$.

\medskip
\noindent\textbf{Part I --- Criteria and strict Jensen.}
\begin{enumerate}
  \item होने देना $ f $ एक अंतराल पर भिन्न होना $ I $. साबित करें कि
        $ f $ उत्तल है यदि और केवल यदि $ f'$ यह बढ़ रहा है
        \emph{(सीमा में पार करके एक दिशा
        ढलान असमानता \cref{lem:b2:realfun:slopes}; द
        माध्य मान प्रमेय द्वारा अन्य)}. निष्कर्ष निकालें $ C^2$
        मानदंड $ f'' \geq 0$.
  \item कल्पना करना $ f''> 0$ पर $ I $. साबित करें कि $ f $ है
        \emph{कठोरता से} उत्तल (के लिए सख्त असमानता $ x \neq y $
        और $\lambda \in \intoo01$), और वह सख्ती से उत्तल है
        फ़ंक्शन जेन्सेन की असमानता को संतुष्ट करता है
        (\cref{thm:b2:realfun:convexreg}~(3)) समानता के साथ
        \emph{केवल} जब सभी $ x_i $ संयोग.
  \item टूलबॉक्स के कच्चे माल को प्रमाणित करें: $-\ln $ सख्ती से है
        उत्तल पर $\intoo{0}{+\infty}$; $ t \mapsto t^r $ है
        वहाँ के लिए सख्ती से उत्तल $ r > 1$ और सख्ती से अवतल
        के लिए $0 < r < 1$; $\exp $ सख्ती से उत्तल है $\R $.
  \item (यंग की असमानता) के लिए $ a, b \geq 0$, सिद्ध करना
        \[
        ab \;\leq\; \frac{a^p}{p} + \frac{b^q}{q},
        \]
        समानता के साथ यदि और केवल यदि $ a^p = b^q $ \emph{(लागू करें
        की अवतलता $\ln $ दो बिंदुओं पर $ a^p, b^q $
        वज़न के साथ $\frac1p, \frac1q $)}.
  \item भारित AM--GM को एक पंक्ति में पुनः प्राप्त करें
        की अवतलता $\ln $:
        \[
        \prod_i x_i^{\lambda_i} \leq \sum_i\lambda_ix_i \qquad
        (x_i > 0),
        \]
        समानता के मामले के साथ; के सीमा मार्ग से तुलना करें
        \cref{exo:b2:realfun:6}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Hölder and Minkowski.} लिखना
$\norm{a}_p = \bigl(\sum_i \abs{a_i}^p\bigr)^{1/p}$ और
$\norm{a}_\infty = \max_i\abs{a_i}$.
\begin{enumerate}[resume]
  \item (एच\"पुराना) सिद्ध करें
        \[
        \sum_{i=1}^{n}\abs{a_ib_i} \;\leq\;
        \norm a_p\,\norm b_q ,
        \]
        सदिशों की समानता के साथ $(\abs{a_i}^p)$ और
        $(\abs{b_i}^q)$ आनुपातिक हैं \emph{(सामान्यीकृत करें
        $\norm a_p = \norm b_q = 1$ और यंग टर्मवाइज लागू करें)}.
  \item विशेष मामलों की पहचान करें: $ p = q = 2$
        (कॉची-श्वार्ज़), और समापन बिंदु जोड़ी $(p, q) = (1,
        \infty)$: बताएं और साबित करें $\sum\abs{a_ib_i} \leq
        \norm a_1\norm b_\infty $.
  \item (मिन्कोव्स्की) के लिए $ p \geq 1$, सिद्ध करना
        \[
        \norm{a + b}_p \leq \norm a_p + \norm b_p
        \]
        \emph{(लिखना $\abs{a_i + b_i}^p \leq \abs{a_i +
        b_i}^{p-1}(\abs{a_i} + \abs{b_i})$ और एच लागू करें\"पुराना
        प्रत्येक उत्पाद के लिए)}. निष्कर्ष: $\norm\cdot_p $ एक है आदर्श
        पर $\R^n $ हरएक के लिए $ p \in \intco{1}{+\infty}$,
        की तस्वीर को \omterm{def:b2:metric:complete}{पूरा} करना \cref{ch:b2:nvs}.
  \item अभिन्न संस्करण: के लिए $ f, g $ \omterm{def:b2:metric:continuity}{निरंतर} पर
        $\intcc{a}{b}$, बताएं और H सिद्ध करें\"वृद्ध और मिन्कोव्स्की
        के लिए $\norm f_p = \bigl(\int_a^b\abs
        f^p\bigr)^{1/p}$ \emph{(समान प्रमाण, सख्त के साथ
        समानता के लिए अभिन्न की सकारात्मकता
        चर्चा)}.
  \item एकरसता सिद्ध करें $\norm a_q \leq \norm a_p $ के लिए $1
        \leq p \leq q $, सीमा $\norm a_p \to
        \norm a_\infty $ जैसा $ p \to \infty $, और उलटा
        तीव्र स्थिरांक के साथ तुलना:
        \[
        \norm a_p \leq n^{\frac1p - \frac1q}\,\norm a_q
        \]
        \emph{(एच\"स्थिर वेक्टर के विरुद्ध पुराना)}. पहचानें
        प्रत्येक समानता प्राप्त करने वाले सदिश।
  \item (इंटरपोलेशन) के लिए $1 \leq p < r < q $ और $\theta \in
        \intoo01$ साथ $\frac1r = \frac\theta p +
        \frac{1-\theta}q $, सिद्ध करना
        \[
        \norm a_r \leq \norm a_p^{\theta}\,
        \norm a_q^{1-\theta}
        \]
        \emph{(एच लागू करें\"घातांक के साथ पुराने $\frac{p}{\theta
        r}$ और $\frac{q}{(1-\theta)r}$ को $\abs{a_i}^{\theta
        r}\abs{a_i}^{(1-\theta)r}$)}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The power-mean chain, complete.}
के लिए $ p \neq 0$ तय करना $ M_p = \bigl(\sum_i\lambda_i
x_i^p\bigr)^{1/p}$ ($ x_i > 0$), और $ M_0 = \prod_i
x_i^{\lambda_i}$.
\begin{enumerate}[resume]
  \item साबित करें कि $ p \mapsto M_p $ सब पर बढ़ रहा है
        $\R^*$: इलाज $ p < q < 0$ पारस्परिक पहचान से
        $ M_{-p}(x) = M_p(1/x)^{-1}$, और पुल के माध्यम से $0$ द्वारा
        दिखा रहा हूँ $ M_p \leq M_0 \leq M_q $ के लिए $ p < 0 < q $
        \emph{(की अवतलता लागू करें $\ln $ को $ x_i^q $, और
        नकारात्मक घातांक के लिए उलटी असमानता)}.
  \item सीमाएं सिद्ध करें $ M_p \to \max_i x_i $ जैसा $ p \to
        +\infty $ और $ M_p \to \min_i x_i $ जैसा $ p \to -\infty $.
  \item शृंखला लिखिए $\min \leq \mathrm{HM} \leq
        \mathrm{GM} \leq \mathrm{AM} \leq \mathrm{QM} \leq
        \max $ समान वजन के लिए, और क्लासिक साबित करें
        परिणाम: सकारात्मक के लिए $ a_1, \dots, a_n $,
        \[
        \Bigl(\sum_i a_i\Bigr)\Bigl(\sum_i\frac1{a_i}\Bigr)
        \geq n^2 .
        \]
  \item सम्बंधित का मतलब होता है मानदंड: समान वजन के लिए $\lambda_i =
        \frac1n $, $ M_p(x) = n^{-1/p}\norm x_p $. सामंजस्य स्थापित करें
        दो एकरसताएँ --- साधन \emph{बढ़ोतरी} साथ $ p $
        जबकि मानदंड \emph{घटाना} (प्रश्न 10) ---एक में
        कारक के बारे में वाक्य $ n^{-1/p}$.
  \item की पूरी श्रृंखला में समानता के मामले निर्धारित करें
        प्रश्न 14 (सकारात्मक भार): कहीं भी समानता
        सभी को मजबूर करता है $ x_i $ बराबर --- सख्त उत्तलता से लाभ होता है।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Dividends.}
\begin{enumerate}[resume]
  \item (घुंडी वाला युवा) के लिए $ a, b \geq 0$ और $\varepsilon >
        0$, सिद्ध करना
        \[
        ab \leq \varepsilon\,\frac{a^p}{p} +
        \varepsilon^{-q/p}\,\frac{b^q}{q},
        \]
        और वर्कहॉर्स मामला $ ab \leq \varepsilon a^2 +
        \frac{b^2}{4\varepsilon}$: उपयोग की गई अवशोषण युक्ति
        पूरे विश्लेषण में.
  \item (कार्लमैन की ओर) आइए $ c_k = \frac{(k+1)^k}{k^{k-1}}$.
        दूरबीन पहचान सिद्ध करें $\prod_{k=1}^{n}c_k =
        (n+1)^n $, और संख्याओं पर लागू AM--GM द्वारा निष्कर्ष निकालें
        $ c_ka_k $,
        \[
        (a_1a_2\cdots a_n)^{1/n} \leq
        \frac{1}{n(n+1)}\sum_{k=1}^{n} c_k a_k
        \qquad (a_k > 0).
        \]
  \item (कार्लमैन की असमानता) योग समाप्त $ n $, विनिमय करें
        योग का क्रम (सकारात्मक) \omterm{def:b2:series:summable}{संक्षेपणीय} परिवार,
        \cref{thm:b2:series:fubini}), और उपयोग करें $\sum_{n \geq
        k}\frac{1}{n(n+1)} = \frac1k $ और $ c_k/k = \bigl(1 +
        \frac1k\bigr)^k < \eu $ निष्कर्ष निकालना: प्रत्येक के लिए
        अभिसारी $\sum a_k $ सकारात्मक शर्तों के साथ,
        \[
        \sum_{n=1}^{\infty}(a_1a_2\cdots a_n)^{1/n}
        \;\leq\; \eu\sum_{k=1}^{\infty}a_k .
        \]
  \item के लिए $ f $ \omterm{def:b2:metric:continuity}{निरंतर} और सकारात्मक पर $\intcc{0}{1}$,
        सिद्ध करना
        \[
        \Bigl(\int_0^1 f\Bigr)\Bigl(\int_0^1\frac1f\Bigr)
        \geq 1,
        \]
        समानता के साथ यदि $ f $ स्थिर है
        \emph{(कॉची--श्वार्ज़ ऑन $\sqrt f\cdot\frac1{\sqrt
        f}$)}.
  \item (गेंदों की ज्यामिति) के समानता मामले का उपयोग करना
        मिन्कोव्स्की, इसके लिए दिखाओ $1 < p < \infty $ इकाई
        का क्षेत्र $\norm\cdot_p $ इसमें कोई खंड नहीं है (द आदर्श
        के अर्थ में सख्ती से उत्तल है
        \cref{pb:b2:nvs:1}), जबकि के लिए $ p = 1$ और $ p =
        \infty $ यह करता है: सपाट टुकड़े प्रदर्शित करें।
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Duality and synthesis.}
\begin{enumerate}[resume]
  \item (एच\"द्वैत जितना पुराना) इसे प्रत्येक के लिए सिद्ध करें $ a \in
        \R^n $,
        \[
        \norm a_p = \max_{\norm b_q \leq 1}\ \sum_i a_ib_i ,
        \]
        अधिकतमीकरण प्रदर्शित करना $ b $ स्पष्ट रूप से. (द $ p $-मानदंड
        है \omterm{def:b2:linalg:dual}{दोहरी} की $ q $-मानदंड --- परिमित-आयामी
        का रोगाणु $ L^p $ द्वंद्व।)
  \item (क्षण) चलो $ X $ परिमित रूप से लेने वाला एक यादृच्छिक चर बनें
        अनेक सकारात्मक मूल्य $ x_i $ संभावनाओं के साथ
        $\lambda_i $. प्रश्न 12 को इस प्रकार पुनः दोहराएँ: $ r \mapsto
        \E[X^r]^{1/r}$ बढ़ रहा है---पल
        (लायपुनोव) असमानता, का पुन: उपयोग किया जाना है
        \cref{ch:b2:randomvar}.
  \item नामित टूल से प्रत्येक दो पंक्तियों में हल करें: (i) के लिए
        सकारात्मक $ a, b, c $: $ a^3 + b^3 + c^3 \geq
        \frac{(a+b+c)^3}{9}$; (ii) सकारात्मक के लिए $ x_1, \dots,
        x_n $: $\bigl(\sum_i\sqrt{x_i}\bigr)^2 \leq
        n\sum_i x_i $.
  \item (संश्लेषण) वंशावली को पाँच वाक्यों में बनाएँ: राग
        ढलान लेम्मा की परिभाषा; लाइनों का समर्थन करने के लिए ढलान
        जेन्सेन; $\ln $ यंग से एच तक की अवतलता\"से अधिक उम्र का
        मिन्कोवस्की को $p $-मानदंड; पावर-मीन के लिए जेन्सेन
        क्षणों की शृंखला; एएम--जीएम से कार्लमैन। शिखरों के नाम बताएं
        (एच\"वृद्ध--मिन्कोव्स्की; कार्लेमैन), और बताएं कि कहां
        टूलबॉक्स का नेतृत्व किया गया है: द $ L^p$ वर्ष के स्थान 3
        आयतन, जिसके अभिगृहीत \omterm{def:b2:series:def}{बिल्कुल} प्रश्न 6 और 8 हैं।
\end{enumerate}
\end{problem}
