\chapter{हर्मिटियन फॉर्म}\label{ch:b2:hermitian}

जटिल सदिश स्थानों की अपनी आंतरिक-उत्पाद ज्यामिति होती है
एक मोड़: एक चर में रैखिकता, \emph{संयुग्मित}-रैखिकता में
दूसरा. मोड़ को स्वीकार करने का भुगतान एक वर्णक्रमीय सिद्धांत है
असली से भी ज्यादा साफ--- \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} एंडोमोर्फिज्म असली है
eigenvalues, \omterm{def:b2:hermitian:adjoint}{अमली} लोगों के पास यूनिमॉड्यूलर है eigenvalues, और दोनों
लम्बवत आधारों में विकर्णीकरण करें। यह छोटा अध्याय चलता है
का यूक्लिडियन कार्यक्रम \cref{ch:b2:quadratic} ऊपर $\C $.

\section{हर्मिटियन आंतरिक उत्पाद}

\begin{definition}\label{def:b2:hermitian:def}
ए \emph{हर्मिटियन आंतरिक उत्पाद}\index{हर्मिटियन आंतरिक उत्पाद} पर
एक जटिल सदिश स्थान $ E $ एक नक्शा है $\langle\cdot,\cdot\rangle
\colon E \times E \to \C $ वह दूसरे चर में रैखिक है,
संयुग्\omterm{def:b2:quadratic:adjoint}{म-सममित} ($\langle y, x\rangle = \conj{\langle x,
y\rangle}$ --- इसलिए पहले चर में संयुग्म-रैखिक), और
सकारात्मक निश्चित ($\langle x, x\rangle > 0$ के लिए $ x \neq 0$). द
मानक उदाहरण पर $\C^n $:
\[
\langle x, y \rangle = \sum_{i=1}^{n} \conj{x_i}\, y_i ;
\]
पर \omterm{def:b2:metric:continuity}{निरंतर} कार्य, $\langle f, g\rangle = \int_a^b \conj
f\,g $. आदर्श: $\norm x = \sqrt{\langle x,x\rangle}$; ए
परिमित-आयामी जटिल स्थान इतना \omterm{def:b2:structures:countable}{सुसज्जित} है \emph{हर्मिटियन
अंतरिक्ष}\index{हर्मिटियन
अंतरिक्ष}.
\end{definition}

\begin{example}[पहली गणना]\label{ex:b2:hermitian:firstcomp}
में $\C^2$, लेना $ x = (1+\iu,\ 2-\iu)$ और $ y = (\iu,\ 1)$.
तब
\[
\norm x^2 = \abs{1+\iu}^2 + \abs{2-\iu}^2 = 2 + 5 = 7,
\qquad
\norm y^2 = 1 + 1 = 2,
\]
और, पहले तर्क को जोड़ते हुए,
\[
\langle x, y\rangle
= \conj{(1+\iu)}\,\iu + \conj{(2-\iu)}\cdot1
= (1-\iu)\iu + (2+\iu)
= (\iu + 1) + (2 + \iu) = 3 + 2\iu .
\]
कॉची-श्वार्ज़ जाँच करता है: $\abs{\langle x, y\rangle}^2 = 9 +
4 = 13 \leq 14 = \norm x^2\norm y^2$ --- समानता के करीब,
क्योंकि $ x $ के गुणज के करीब है $ y $. यह भी नोट करें
$\langle y, x\rangle = \conj{3 + 2\iu} = 3 - 2\iu $:
क्रिया में संयुग्म-समरूपता, और कारण $\langle x,
x\rangle $ हमेशा वास्तविक होता है.
\end{example}

\begin{theorem}[कॉची-श्वार्ज़, जटिल]\label{thm:b2:hermitian:cs}
$\abs{\langle x, y\rangle} \leq \norm x \norm y $, समानता के साथ यदि
$ x, y $ रैखिक रूप से निर्भर हैं; $\norm\cdot $ एक है आदर्श. इसके अलावा
ऑर्थोनॉर्मल आधार मौजूद हैं (ग्राम--श्मिट शब्दशः चलता है), साथ
\[
x = \sum_i \langle e_i, x\rangle\, e_i,
\qquad
\norm x^2 = \sum_i \abs{\langle e_i, x\rangle}^2 .
\]
\end{theorem}

\begin{proof}
के लिए $ y \neq 0$ और $ t \in \C $: $0 \leq \norm{x - ty}^2 = \norm x^2
- 2\Re\bigl(\conj t\langle y, x\rangle\bigr) + \abs t^2\norm y^2$.
चुनना $ t = \frac{\langle y, x\rangle}{\norm y^2}$:
\[
0 \leq \norm x^2 - \frac{\abs{\langle y, x\rangle}^2}{\norm y^2},
\]
जो असमानता है; समानता बल $ x = ty $. त्रिकोण
असमानता, पूर्णतः:
\[
\norm{x + y}^2 = \norm x^2 + 2\,\Re\langle x, y\rangle
+ \norm y^2
\leq \norm x^2 + 2\,\abs{\langle x, y\rangle} + \norm y^2
\leq \bigl(\norm x + \norm y\bigr)^2 ,
\]
का उपयोग करते हुए $\Re z \leq \abs z $ और फिर कॉची-श्वार्ज़; एकरूपता
और अलगाव तत्काल है, इसलिए $\norm\cdot $ एक है आदर्श.
ग्राम--श्मिट: जैसा कि वास्तविक मामले में, संयुग्मों के साथ होता है
परिभाषा के अनुसार रखा गया है (सम्मेलन पर ध्यान दें: हमारे उत्पाद हैं
संयुग्म-रैखिक में \emph{पहला} स्लॉट, इसलिए निर्देशांक हैं
$\langle e_i, x\rangle $; अगला उदाहरण एल्गोरिथम चलाता है
एक बार \omterm{def:b2:metric:complete}{पूरा})।
\end{proof}

\begin{example}[कॉम्प्लेक्स ग्राम--श्मिट, पूर्ण रूप से चलाएँ]\label{ex:b2:hermitian:gramschmidt}
आधार को सामान्यीकृत करें $ v_1 = (1, \iu)$, $ v_2 = (0, 1)$ का
$\C^2$. पहला वेक्टर: $\norm{v_1}^2 = \abs1^2 + \abs\iu^2 = 2$,
इसलिए $ e_1 = \frac{1}{\sqrt2}(1, \iu)$. परियोजना $ v_2$ --- के साथ
पहले स्लॉट में संयुग्मित करें:
\[
\langle e_1, v_2\rangle
= \frac{1}{\sqrt2}\bigl(\conj{1}\cdot0 +
\conj{\iu}\cdot1\bigr)
= \frac{-\iu}{\sqrt2} ,
\qquad
v_2 - \langle e_1, v_2\rangle e_1
= (0,1) + \frac{\iu}{2}\,(1, \iu)
= \Bigl(\frac\iu2,\ \frac12\Bigr) .
\]
इसका आदर्श है $\sqrt{\frac14 + \frac14} = \frac{1}{\sqrt2}$:
$ e_2 = \frac{1}{\sqrt2}(\iu, 1)$. जाँच करना: $\langle e_1,
e_2\rangle = \frac12(\conj1\cdot\iu + \conj\iu\cdot1) =
\frac12(\iu - \iu) = 0$. के निर्देशांक $ v_2$ नये आधार पर:
$ v_2 = \langle e_1, v_2\rangle e_1 + \langle e_2, v_2\rangle
e_2$ साथ $\langle e_2, v_2\rangle = \frac{1}{\sqrt2}$ ---
\omterm{def:b2:structures:generated}{आदेश} पर ध्यान दें: $\langle v_2, e_1\rangle $ दे देंगे
\emph{संयुग्मित} गुणांक. समापन अंतर्दृष्टि: एल्गोरिथ्म है
यूक्लिडियन एक शब्दशः; एकमात्र जाल वहीं है जहां
संयुग्मन गिरता है, और कंप्यूटिंग $\norm{v_2 -
\text{proj}}^2 > 0$ चुपचाप सकारात्मकता--स्वयंसिद्धि का उपयोग करता है
जो संपूर्ण ज्यामिति को कार्यान्वित करता है।
\end{example}

\section{एडजॉइंट, हर्मिटियन और एकात्मक एंडोमोर्फिज्म}

\begin{definition}\label{def:b2:hermitian:adjoint}
 \emph{\omterm{def:b2:quadratic:adjoint}{जोड़}} $ u^*$ का $ u \in \mathcal{L}(E)$ द्वारा परिभाषित किया गया है
$\langle u^*(x), y\rangle = \langle x, u(y)\rangle $; एक में
ऑर्थोनॉर्मल आधार, $\operatorname{Mat}(u^*) = \conj{A}^{\mathsf T} =:
A^{\dagger}$ (संयुग्मित \omterm{def:b2:linalg:transpose}{खिसकाना}) --- वास्तव में, यदि $ B =
(b_{ij})$ का मैट्रिक्स है $ u^*$ ऑर्थोनॉर्मल आधार पर
$(e_i)$, तब $ b_{ij} = \langle e_i, u^*(e_j)\rangle $, और
परिभाषित पहचान देता है
\[
\conj{b_{ij}}
= \langle u^*(e_j), e_i\rangle
= \langle e_j, u(e_i)\rangle = a_{ji} ,
\qquad\text{so}\qquad
B = \conj{A}^{\mathsf T} .
\] $ u $ है
\emph{हर्मिटियन}\index{हर्मिटियन} कब $ u^* = u $
($ A^\dagger = A $), \emph{अमली}\index{अमली} कब $ u^*u =
\mathrm{id}$ ($ A^\dagger A = I $: समूह $ U(n)$), \emph{सामान्य}
कब $ u^*u = uu^*$.
\end{definition}

\begin{proposition}\label{prop:b2:hermitian:eigenvalues}
eigenvalues एक का \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} एंडोमोर्फिज्म हैं \emph{असली};
eigenvalues एक का \omterm{def:b2:hermitian:adjoint}{अमली} एंडोमोर्फिज्म है \emph{मापांक $1$}; दोनों ही मामलों में eigenspaces विशिष्ट के लिए eigenvalues ऑर्थोगोनल हैं.
\end{proposition}

\begin{proof}
\omterm{def:b2:hermitian:adjoint}{हर्मिटियन}, $ u(x) = \lambda x $, $ x \neq 0$:
\[
\lambda \norm x^2 = \langle x, u(x)\rangle = \langle u(x), x\rangle
= \conj{\langle x, u(x)\rangle} = \conj\lambda\,\norm x^2 ,
\]
इसलिए $\lambda \in \R $. \omterm{def:b2:hermitian:adjoint}{अमली}: $\norm{u(x)} = \norm x $ (से $ u^*u =
\mathrm{id}$), इसलिए $\abs\lambda\norm x = \norm x $. रूढ़िवादिता
(\omterm{def:b2:hermitian:adjoint}{हर्मिटियन} मामला): $\lambda\langle x, y\rangle = \langle u(x),
y\rangle = \langle x, u(y)\rangle = \mu\langle x, y\rangle $ के लिए
eigenvectors असली के साथ $\lambda \neq \mu $. \omterm{def:b2:hermitian:adjoint}{अमली} केस, में
पूर्ण: के लिए $ u(x) = \lambda x $, $ u(y) = \mu y $ साथ $\lambda \neq
\mu $ (दोनों यूनिमॉड्यूलर),
\[
\langle x, y\rangle = \langle u(x), u(y)\rangle
= \conj\lambda\mu\,\langle x, y\rangle ,
\]
और $\conj\lambda\mu = \frac{\mu}{\lambda} \neq 1$: कारक
नहीं है $1$, इसलिए $\langle x, y\rangle = 0$.
\end{proof}

\begin{example}[एक तिरछा-हर्मिटियन मैट्रिक्स, विकर्णित]\label{ex:b2:hermitian:skewexample}
$ A = \begin{pmatrix} 0 & -2\\ 2 & 0\end{pmatrix}$ संतुष्ट
$ A^\dagger = A^{\mathsf T} = -A $: तिरछा-Hermitian (वास्तविक भी
एंटीसिमेट्रिक --- खत्म $\R $ इसमें नहीं है eigenvalues \omterm{def:b2:series:def}{बिल्कुल} भी)।
अभिलक्षणिक बहुपद $ X^2 + 4$: eigenvalues $\pm2\iu $,
विशुद्ध रूप से काल्पनिक, जैसे \cref{exo:b2:hermitian:9} में भविष्यवाणी करता है
सामान्य. आइगेनवेक्टर: $(A - 2\iu I)v = 0$ देता है $ v_1 =
\frac{1}{\sqrt2}(1, \iu)$, और $ v_2 = \frac{1}{\sqrt2}(1,
-\iu)$ के लिए $-2\iu $; वे ऑर्थोगोनल हैं:
\[
\langle v_1, v_2\rangle
= \tfrac12\bigl(\conj{1}\cdot1 +
\conj{\iu}\cdot(-\iu)\bigr)
= \tfrac12(1 - 1) = 0 .
\]
इसलिए $ A = U\operatorname{diag}(2\iu, -2\iu)\,U^\dagger $ साथ $ U =
(v_1\ v_2)$ \omterm{def:b2:hermitian:adjoint}{अमली}. समापन अंतर्दृष्टि: $ H = -\iu A =
\begin{pmatrix} 0 & 2\iu\\ -2\iu & 0\end{pmatrix}$ है \omterm{def:b2:hermitian:adjoint}{हर्मिटियन}
असली के साथ \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} $\{\pm2\}$ और यह \emph{वही}
eigenvectors ---आक्षेप $ u \mapsto \iu u $ बीच में
\omterm{def:b2:hermitian:adjoint}{हर्मिटियन} और तिरछा-Hermitian एंडोमोर्फिज्म
(\cref{exo:b2:hermitian:9}), मैट्रिक्स द्वारा मैट्रिक्स देखा; ऊपर $\R $
जो उसी $ A $ संख्या के साथ एक रोटेशन-स्केलिंग है eigenvectors पर
सभी, और केवल मार्ग $\C $ अपना सामान्य स्वरूप प्रकट करता है।
\end{example}

\begin{theorem}[हर्मिटियन वर्णक्रमीय प्रमेय]\label{thm:b2:hermitian:spectral}
प्रत्येक \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} एंडोमोर्फिज्म एक का \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} स्थान एक ऑर्थोनॉर्मल है
का आधार eigenvectors (असली के साथ eigenvalues): $ A^\dagger = A $
तात्पर्य $ A = U D U^{\dagger}$ साथ $ U \in U(n)$ और $ D $ असली
विकर्ण.\index{वर्णक्रमीय प्रमेय!हर्मिटियन}
\end{theorem}

\begin{proof}
ऊपर $\C $, द \omterm{def:b2:reduction:charpoly}{विशेषता बहुपद} विभाजन: एक आइजन्वेक्टर
$ e_1$ मौजूद है (\cref{ch:b2:reduction}) --- नहीं सघनता तर्क
की जरूरत है, का एक फायदा $\C $. इसे सामान्य करें. यह ऑर्थोगोनल है
पूरक $ F = e_1^\perp $ स्थिर है: के लिए $ x \perp e_1$,
\[
\langle e_1, u(x)\rangle = \langle u(e_1), x\rangle
= \lambda_1\langle e_1, x\rangle = 0
\]
($\lambda_1$ असली)। प्रतिबंध है \omterm{def:b2:hermitian:adjoint}{हर्मिटियन}; पर शामिल करें
आयाम और संयोजन. विस्तार से: प्रतिबंध $ u|_F $ है
की एक एंडोमोर्फिज्म \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} स्थान $ F $ (आयाम $ n - 1$)
साथ $\langle u|_F(x), y\rangle = \langle x, u|_F(y)\rangle $
विरासत में मिला $ u $; प्रेरण परिकल्पना एक प्रदान करती है
ऑर्थोनॉर्मल आधार $(e_2, \dots, e_n)$ का $ F $ का eigenvectors,
और $(e_1, e_2, \dots, e_n)$ में ऑर्थोनॉर्मल है $ E $ ($ e_1 \perp
F $) और से बना है eigenvectors का $ u $. मैट्रिक्स अनुवाद:
के कॉलम $ U $ हैं $ e_i $, $ U^\dagger U = I $ उनका व्यक्त करता है
ऑर्थोनोर्मैलिटी, और $ AU = UD $ एकत्रित करता है \omterm{def:b2:reduction:eigen}{eigenvalue}
समीकरण, कहाँ से $ A = UDU^\dagger $ साथ $ D $ वास्तविक विकर्ण
(\cref{prop:b2:hermitian:eigenvalues}).
\end{proof}

\begin{example}\label{ex:b2:hermitian:pauli}
$ A = \begin{pmatrix} 0 & -\iu\\ \iu & 0\end{pmatrix}$ है \omterm{def:b2:hermitian:adjoint}{हर्मिटियन}
($ A^\dagger = A $): eigenvalues से $\chi_A = X^2 - 1$: $\pm 1$
(वास्तविक, जैसा कि वादा किया गया था), ऑर्थोनॉर्मल के साथ eigenvectors
$\frac{1}{\sqrt2}(1, \iu)^{\mathsf T}$ और $\frac{1}{\sqrt2}(1,
-\iu)^{\mathsf T}$. (भौतिक विज्ञानी जानते हैं $ A $ पाउली मैट्रिक्स के रूप में; द
की वास्तविकता \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} स्पेक्ट्रा के कारण ही क्वांटम वेधशालाओं का मॉडल तैयार किया जाता है
द्वारा \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} ऑपरेटर्स.)
\end{example}

\begin{example}[एक सकारात्मक निश्चित हर्मिटियन मैट्रिक्स ने काम किया]\label{ex:b2:hermitian:pdrun}
$ A = \begin{pmatrix} 2 & 1-\iu\\ 1+\iu & 3\end{pmatrix}$:
\omterm{def:b2:hermitian:adjoint}{हर्मिटियन}, चूँकि विकर्ण वास्तविक है और विविकर्ण वास्तविक है
प्रविष्टियाँ संयुग्मी हैं। अभिलक्षणिक बहुपद:
\[
(2-\lambda)(3-\lambda) - \abs{1-\iu}^2
= \lambda^2 - 5\lambda + 4
= (\lambda - 1)(\lambda - 4) :
\]
\omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} $\{1, 4\}$, वास्तविक और सकारात्मक --- $ A $ सकारात्मक है
निश्चित. आइगेनवेक्टर: के लिए $\lambda = 4$, सिस्टम $(A -
4I)v = 0$ देता है $ v_4 = (1 - \iu,\ 2)$ (दूसरी पंक्ति जांचें:
$(1+\iu)(1-\iu) - 2 = 0$); के लिए $\lambda = 1$, $ v_1 = (1 - \iu,\
-1)$. पहले स्लॉट में संयुग्म के साथ ऑर्थोगोनैलिटी:
\[
\langle v_4, v_1\rangle
= \conj{(1-\iu)}\,(1-\iu) + \conj{2}\,(-1)
= 2 - 2 = 0 . \checkmark
\]
सामान्यीकरण ($\norm{v_4}^2 = 2 + 4 = 6$, $\norm{v_1}^2 = 2 + 1
= 3$) देता है \omterm{def:b2:hermitian:adjoint}{अमली} $ U = \bigl(\frac{v_4}{\sqrt6}\
\frac{v_1}{\sqrt3}\bigr)$ साथ $ A =
U\operatorname{diag}(4,1)U^\dagger $. समापन अंतर्दृष्टि:
रेले की रीडिंग तत्काल है --- के इकाई क्षेत्र पर
$\C^2$, $\langle x, Ax\rangle $ तक फैली हुई है $\intcc{1}{4}$,
दो पर प्राप्त किया eigenvectors; ये है $ n = 2$ का रोगाणु
कूरेंट--फिशर सिद्धांत सप्ताहांत समस्या में निर्मित।
स्पॉट-चेक करें कि फॉर्म असली है या नहीं eigenvectors भी: पर
$ x = (1, \iu)$,
\[
Ax = \bigl(2 + (1-\iu)\iu,\ (1+\iu) + 3\iu\bigr)
= (3 + \iu,\ 1 + 4\iu),
\]
\[
\langle x, Ax\rangle = \conj{1}\,(3+\iu) +
\conj{\iu}\,(1+4\iu)
= (3 + \iu) + (-\iu)(1 + 4\iu) = 3 + \iu - \iu + 4 = 7 \in
\R ,
\]
जैसा \cref{prop:b2:hermitian:eigenvalues}का प्रमाण तंत्र
(संयुग्म-समरूपता विरुद्ध $ A^\dagger = A $) के लिए गारंटी
प्रत्येक $ x $.
\end{example}

\begin{example}[एकात्मक मैट्रिक्स विकर्णित]\label{ex:b2:hermitian:unitary}
$ U = \frac{1}{\sqrt2}\begin{pmatrix} 1 & \iu\\ \iu & 1
\end{pmatrix}$ (\omterm{def:b2:hermitian:adjoint}{अमली} द्वारा \cref{exo:b2:hermitian:2}). इसका
\omterm{def:b2:reduction:charpoly}{विशेषता बहुपद} है $\bigl(X - \frac{1}{\sqrt2}\bigr)^2
+ \frac12$, जड़ों के साथ
\[
\lambda_\pm = \frac{1 \pm \iu}{\sqrt2} = \eu^{\pm\iu\pi/4},
\]
यूनिमॉड्यूलर के रूप में \cref{prop:b2:hermitian:eigenvalues} वादा किया, और
ऑर्थोनॉर्मल eigenvectors $\frac{1}{\sqrt2}(1, \pm1)$. इसलिए $ U =
V\operatorname{diag}(\eu^{\iu\pi/4},
\eu^{-\iu\pi/4})V^\dagger $: सही आधार पर, $ U $ की एक \omterm{def:b2:quadratic:adjoint}{जोड़}ी है
द्वारा विमान का घूर्णन $\pm\frac\pi4$ --- एक वास्तविक रोटेशन मैट्रिक्स है
कोई वास्तविक नहीं eigenvectors, लेकिन खत्म $\C $ यह दो भागों में विभाजित हो जाता है
यूनिमॉड्यूलर स्केलर। समापन अंतर्दृष्टि: \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} स्पेक्ट्रा चालू है
असली लाइन, \omterm{def:b2:hermitian:adjoint}{अमली} यूनिट सर्कल पर स्पेक्ट्रा; दोनों हैं
समान सामान्यता की छाया, और केली रूपांतरण
\cref{exo:b2:hermitian:6} एक चित्र को दूसरे चित्र पर मैप करता है।
\end{example}

\begin{example}[केली रूपांतरण की गणना की गई]\label{ex:b2:hermitian:cayleyrun}
दौड़ना \cref{exo:b2:hermitian:6} पर $ H = \begin{pmatrix} 0 & 1\\
1 & 0\end{pmatrix}$ (\omterm{def:b2:hermitian:adjoint}{हर्मिटियन}, \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} $\{1, -1\}$,
ऑर्थोनॉर्मल eigenvectors $\frac{1}{\sqrt2}(1, \pm1)$). में
आइजेनबैसिस सब कुछ अदिश है: परिवर्तन $\lambda
\mapsto \frac{\lambda - \iu}{\lambda + \iu}$ भेजता है
\[
1 \longmapsto \frac{1 - \iu}{1 + \iu} = -\iu,
\qquad
-1 \longmapsto \frac{-1 - \iu}{-1 + \iu} = \iu ,
\]
(हर के संयुग्मक से गुणा करें), इसलिए $ U = (H -
\iu I)(H + \iu I)^{-1}$ है \omterm{def:b2:hermitian:adjoint}{अमली} साथ eigenvalues
$\mp\iu $ उन्हीं पर eigenvectors:
\[
U = \frac{1}{2}\begin{pmatrix} 1 & 1\\ 1 & -1\end{pmatrix}
\begin{pmatrix} -\iu & 0\\ 0 & \iu\end{pmatrix}
\begin{pmatrix} 1 & 1\\ 1 & -1\end{pmatrix}
= \begin{pmatrix} 0 & -\iu\\ -\iu & 0\end{pmatrix} .
\]
जाँच करना: $ U^\dagger U = I $, और $1 \notin \operatorname{Sp}U =
\{\pm\iu\}$, जैसा कि सिद्धांत वादा करता है। समापन अंतर्दृष्टि: वास्तविक
बिंदु को घटाकर इकाई वृत्त पर रेखा मानचित्र $1$ ---
\omterm{def:b2:reduction:eigen}{eigenvalue} द्वारा \omterm{def:b2:reduction:eigen}{eigenvalue}, केली परिवर्तन है
एम\"ओबियस मानचित्र $\frac{\lambda-\iu}{\lambda+\iu}$, और मैट्रिक्स
बस उनके स्पेक्ट्रा का पालन करें।
\end{example}

\begin{remark}[सामान्य ख़तरे]
\emph{(i) जहां बार गिरता है:} यह पुस्तक संयुग्मित करती है
\emph{पहला} स्लॉट, इसलिए निर्देशांक हैं $\langle e_i, x\rangle $
और $\langle\lambda x, y\rangle = \conj\lambda\langle x,
y\rangle $; कई पाठ इसके बजाय दूसरे स्लॉट को संयुग्मित करते हैं ---
सूत्रों, या संकेतों की तुलना करने से पहले अनुवाद करें $\iu $ जाओ
चुपचाप गलत. \emph{(ii) जटिल \omterm{def:b2:quadratic:def}{ध्रुवीकरण} अधिक बलवान है:}
ऊपर $\C $, अगर $\langle x, u(x)\rangle = 0$ के लिए \emph{सभी} $ x $
तब $ u = 0$ (बढ़ाना $ x + y $ और $ x + \iu y $: दोनों असली
और के काल्पनिक भाग $\langle x, u(y)\rangle $ गायब होना); ऊपर
$\R $ यह विफल रहता है --- द्वारा घूर्णन $\frac\pi2$ संतुष्ट
$\langle x, u(x)\rangle = 0$ हर जगह. नतीजतन, खत्म
$\C $ केवल, ``$\langle x, u(x)\rangle \in \R $ सभी के लिए $ x $''
पहले से ही बल $ u $ \omterm{def:b2:hermitian:adjoint}{हर्मिटियन}. \emph{(iii) वास्तविक सामान्य नहीं है
\omterm{def:b2:reduction:diag}{विकर्णीय}:} का मैट्रिक्स
\cref{ex:b2:hermitian:skewexample} सामान्य है लेकिन कोई वास्तविक नहीं है
\omterm{def:b2:reduction:eigen}{eigenvalue}; \omterm{def:b2:hermitian:adjoint}{अमली} विकर्णीकरण एक प्रमेय है \emph{ऊपर}
$\C $, और खत्म होता है $\R $ किसी को केवल ब्लॉक कटौती मिलती है।
\emph{(iv) एकात्मकता की जाँच करना:} $ U^\dagger U = I $ का मतलब है
\emph{कॉलम} के लिए अलौकिक हैं \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} उत्पाद का परीक्षण करना $ UU^{\mathsf T}$, या में संयुग्मन को भूल जाना
कॉलम उत्पाद, किसी ग़लती को प्रमाणित करने के दो क्लासिक तरीके हैं
मैट्रिक्स.
\end{remark}

\begin{example}[आइसोमेट्री बिल्कुल एकात्मक हैं]\label{ex:b2:hermitian:isometry}
आदर्श संरक्षण एकात्मकता की तुलना में कमजोर दिखता है, लेकिन खत्म हो गया है $\C $ यह
नहीं है: यदि $\norm{u(x)} = \norm x $ सभी के लिए $ x $, तब $ u^*u =
\mathrm{id}$. वास्तव में $ v = u^*u - \mathrm{id}$ है \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} और
संतुष्ट करता है $\langle x, v(x)\rangle = \norm{u(x)}^2 - \norm x^2 =
0$ हरएक के लिए $ x $; जटिल द्वारा \omterm{def:b2:quadratic:def}{ध्रुवीकरण} (नुकसान (ii)
ऊपर), \omterm{def:b2:funcseq:def}{समान रूप से} लुप्त हो रहे ``विकर्ण'' वाला नक्शा शून्य है:
$ v = 0$. सीधे तौर पर, \omterm{def:b2:quadratic:def}{ध्रुवीकरण} चलता है
\[
0 = \langle x + y, v(x+y)\rangle
= \langle x, v(y)\rangle + \langle y, v(x)\rangle,
\qquad
0 = \langle x + \iu y, v(x + \iu y)\rangle
= \iu\langle x, v(y)\rangle - \iu\langle y, v(x)\rangle ,
\]
और दो रेखाएं मिलकर बल लगाती हैं $\langle x, v(y)\rangle = 0$
सभी के लिए $ x, y $. समापन अंतर्दृष्टि: यही कारण है कि ``\omterm{def:b2:hermitian:adjoint}{अमली}'' कर सकते हैं
केवल लंबाई मापकर जाँच की जानी चाहिए --- कठोरता कि
फूरियर अध्याय शोषण करेगा, जहां ऊर्जा का संरक्षण होगा
$\norm f_2$ (पारसेवल) सभी आंतरिक को संरक्षित करने के समान है
गुणांक के उत्पाद.
\end{example}

\begin{remark}[इस खंड के भीतर परिप्रेक्ष्य]
 \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} यहां बनी मशीनरी लगभग खप जाती है
तुरंत. फूरियर अध्याय \emph{है} \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} ज्यामिति
अनंत आयाम में: घातांक $(e_n)$ एक हैं
ऑर्थोनॉर्मल परिवार के लिए $\langle f, g\rangle =
\frac{1}{2\pi}\int\conj fg $, बेसेल की असमानता है
इस अध्याय का प्रक्षेपण अनुमान
\cref{thm:b2:hermitian:cs}, और पार्सेवल इसकी सीमा है
समानता. परिमित फूरियर रूपांतरण
(\cref{exo:b2:hermitian:10}) जब भी कनवल्शन होता है तो पुनः प्रकट होता है
विकर्ण होना चाहिए. और इस अध्याय की सप्ताहांत समस्या ---
कूरेंट--फिशर, वेइल, इंटरलेसिंग--- आपूर्ति करता है
\omterm{def:b2:reduction:eigen}{eigenvalue} स्थिरता कि अंतर-समीकरण अध्याय
यह तब लागू होता है जब यह किसी सिस्टम की छोटी-छोटी गड़बड़ियों पर जोर देता है
इसकी आवृत्तियों को केवल थोड़ा सा स्थानांतरित करें। पिछड़ा, सब कुछ यहीं
द्विघात-रूप अध्याय का जटिल दर्पण है: रखें
दो शब्दकोश अगल-बगल ($ A^{\mathsf T} \leftrightarrow
A^\dagger $, ओर्थोगोनल $\leftrightarrow $ \omterm{def:b2:hermitian:adjoint}{अमली}, रेले
दोनों में वास्तविक)।
\end{remark}

\begin{remark}[सामान्य एंडोमोर्फिज्म]
ऊपर $\C $ निश्चित कथन है: $ u $ है \emph{एकात्मक रूप से
\omterm{def:b2:reduction:diag}{विकर्णीय} यदि और केवल यदि यह सामान्य है} ($ u^*u = uu^*$) ---
ढकना \omterm{def:b2:hermitian:adjoint}{हर्मिटियन}, \omterm{def:b2:hermitian:adjoint}{अमली} और तिरछा-Hermitian एक ही बार में मानचित्र. द
सबूत उपरोक्त तर्क का एक सुखद सुदृढ़ीकरण है
(\cref{exo:b2:hermitian:8}). ऊपर $\R $, इसके विपरीत, केवल सामान्यता
ब्लॉक-विकर्णीकरण (रोटेशन ब्लॉक) खरीदता है: जटिल ज्यामिति है
वास्तव में सरल.
\end{remark}

\begin{remark}[इसका उपयोग कहां किया जाता है]
\omterm{def:b2:hermitian:adjoint}{हर्मिटियन} वर्णक्रमीय सिद्धांत क्वांटम का गणित है
यांत्रिकी: अवलोकन योग्य वस्तुओं का प्रतिरूपण किया जाता है \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} संचालक
(वास्तविक स्पेक्ट्रा = मापने योग्य मान), समय विकास द्वारा \omterm{def:b2:hermitian:adjoint}{अमली}
वाले (आदर्श संरक्षण = संभाव्यता का संरक्षण)। भीतर
यह पुस्तक, फूरियर अध्याय की रूढ़िवादिता पर आधारित है
घातांक --- ए \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} आंतरिक-उत्पाद कथन --- और
सर्कुलर मैट्रिक्स का विकर्णीकरण
(\cref{exo:b2:hermitian:10}) परिमित फूरियर रूपांतरण है।
सप्ताहांत समस्या का वैरिएबल कैलकुलस विकसित होता है
eigenvalues (कूरेंट--फिशर, वेइल, इंटरलेसिंग), दैनिक
संख्यात्मक विश्लेषण और गणितीय भौतिकी की रोटी; वर्ष
3 खंड इसे विस्तारित करता है \omterm{def:b2:metric:compact}{कॉम्पैक्ट} स्वयं adjoint ऑपरेटरों पर
हिल्बर्ट स्थान.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b2:hermitian:1}
पर $\C^2$: गणना करें $\langle x, y\rangle $, $\norm x $, $\norm y $ के लिए
$ x = (1, \iu)$, $ y = (\iu, 1)$; क्या वे ऑर्थोगोनल हैं? एक दीजिए
ऑर्थोनॉर्मल आधार युक्त $\frac{x}{\norm x}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:hermitian:2}
जो हैं \omterm{def:b2:hermitian:adjoint}{हर्मिटियन}? \omterm{def:b2:hermitian:adjoint}{अमली}? सामान्य?
\[
\begin{pmatrix} 1 & \iu\\ -\iu & 2 \end{pmatrix},
\qquad
\frac{1}{\sqrt2}\begin{pmatrix} 1 & \iu\\ \iu & 1\end{pmatrix},
\qquad
\begin{pmatrix} 0 & 1\\ 0 & 0 \end{pmatrix}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:hermitian:3}
इसे एक मैट्रिक्स सिद्ध करें $ A \in \mathcal{M}_n(\C)$ अनोखा लिखता है $ A =
H + \iu K $ साथ $ H, K $ \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} \emph{(``वास्तविक और काल्पनिक
भाग'' $ H = \frac{A + A^\dagger}{2}$, $ K = \frac{A -
A^\dagger}{2\iu}$)}, ओर वो $ A $ सामान्य है यदि $ H $ और $ K $
आना-जाना।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:hermitian:4}
ऑर्थोनॉर्मल आधार पर विकर्णीकरण करें: $ A = \begin{pmatrix} 2 & \iu\\
-\iu & 2\end{pmatrix}$, और गणना करें $ A^k $ के लिए $ k \in \N $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:hermitian:5}
साबित करें कि $ U(n)$ है \omterm{def:b2:metric:compact}{कॉम्पैक्ट}, और वह \omterm{def:b2:reduction:eigen}{eigenvalue} नक्शा है
पर: प्रत्येक यूनिमॉड्यूलर $\lambda $ कुछ के लिए उठता है \omterm{def:b2:hermitian:adjoint}{अमली} मैट्रिक्स.
साबित करें कि $\det U \in \mathbb{U}$ (यूनिट सर्कल) के लिए $ U \in
U(n)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:hermitian:6}
(केली ट्रांसफॉर्म) चलो $ H $ होना \omterm{def:b2:hermitian:adjoint}{हर्मिटियन}. साबित करें कि $ H + \iu I $ है
उलटा और वह $ U = (H - \iu I)(H + \iu I)^{-1}$ है \omterm{def:b2:hermitian:adjoint}{अमली},
साथ $1 \notin \operatorname{Sp}(U)$. \emph{(वर्णक्रम से काम करें: एक पर
का eigenbasis $ H $, सब कुछ अदिश है।)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:hermitian:7}
के लिए $ A $ \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} सकारात्मक निश्चित ($\langle x, Ax\rangle > 0$ के लिए
$ x \neq 0$), साबित करें कि $\operatorname{Sp}(A) \subseteq
\intoo{0}{\infty}$, वह $ A = B^2$ एक के लिए \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} सकारात्मक
निश्चित $ B $, ओर वो $\det A > 0$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:hermitian:8}
(सामान्य एंडोमोर्फिज्म के लिए वर्णक्रमीय प्रमेय) आइए $ u $ ए पर सामान्य रहें
\omterm{def:b2:hermitian:adjoint}{हर्मिटियन} स्थान.
\begin{enumerate}
  \item सिद्ध करना $\norm{u(x)} = \norm{u^*(x)}$ सभी के लिए $ x $, और निष्कर्ष निकालें
        $\ker(u - \lambda) = \ker(u^* - \conj\lambda)$.
  \item साबित करें कि eigenspaces का $ u $ विशिष्ट के लिए eigenvalues हैं
        ऑर्थोगोनल, और वह ऑर्थोगोनल पूरक है
        \omterm{def:b2:reduction:eigen}{eigenspace} है $ u $-स्थिर।
  \item प्रेरण द्वारा निष्कर्ष निकालें $ u $ एकात्मक रूप से है \omterm{def:b2:reduction:diag}{विकर्णीय};
        और इसके विपरीत.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:hermitian:9}
एक एंडोमोर्फिज्म है \emph{तिरछा-Hermitian} कब $ u^* = -u $. साबित करो
वह यह है eigenvalues ये पूर्णतया काल्पनिक हैं $ u \mapsto \iu
u $ से एक आक्षेप है \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} को तिरछा-Hermitian
एंडोमोर्फिज्म, और वह तिरछा-Hermitian एंडोमोर्फिज्म हैं
एकात्मक रूप से \omterm{def:b2:reduction:diag}{विकर्णीय} (\cref{exo:b2:hermitian:8}).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:hermitian:10}
(परिमित फूरियर रूपांतरण) चलो $ S $ हो \omterm{def:b2:structures:generated}{चक्रीय} की शिफ्ट
$\C^n $: $ S(x_0, x_1, \dots, x_{n-1}) = (x_{n-1}, x_0, \dots,
x_{n-2})$, और $\omega = \eu^{2\iu\pi/n}$.
\begin{enumerate}
  \item बताते हैं कि $ S $ है \omterm{def:b2:hermitian:adjoint}{अमली}, और वह वेक्टर $ f_k =
        \frac{1}{\sqrt n}\bigl(1, \omega^k, \omega^{2k}, \dots,
        \omega^{(n-1)k}\bigr)$, $0 \leq k < n $, फॉर्म ए
        का ऑर्थोनॉर्मल आधार eigenvectors: $ Sf_k = \omega^{-k}
        f_k $.
  \item यह निष्कर्ष निकालें कि प्रत्येक \emph{परिचालित} मैट्रिक्स $ C =
        \sum_{j=0}^{n-1} c_jS^j $ सामान्य है, द्वारा विकर्णित
        एक ही आधार, साथ eigenvalues $\widehat c(k) = \sum_j
        c_j\,\omega^{-jk}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:hermitian:11}
होने देना $ P $ एक निष्क्रिय बनो ($ P^2 = P $) ए की एंडोमोर्फिज्म \omterm{def:b2:hermitian:def}{हर्मिटियन
अंतरिक्ष}. साबित करें कि $ P $ है \emph{ओर्थोगोनल} पर प्रक्षेपण
$\operatorname{im} P $ अगर और केवल अगर $ P^* = P $. मैट्रिक्स दीजिए
ओर्थोगोनल प्रक्षेपण का $\C v $ ($\norm v = 1$), और
लम्बवत् आधार पर एक उपस्थान पर $(v_1, \dots, v_k)$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:hermitian:12}
(इंटरपोलेशन द्वारा स्पेक्ट्रल प्रोजेक्टर) चलो $ A $ होना \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} साथ
\emph{विशिष्ट} eigenvalues $\lambda_1, \dots, \lambda_p $ और
\omterm{def:b2:reduction:eigen}{eigenspace} सड़न $ E = \bigoplus_i E_i $. परिभाषित करें
लैग्रेंज बहुपद $ L_i(X) = \prod_{j\neq i}\frac{X -
\lambda_j}{\lambda_i - \lambda_j}$. साबित करें कि $ P_i = L_i(A)$
पर ओर्थोगोनल प्रक्षेपण है $ E_i $, वह $ P_iP_j = 0$ के लिए
$ i \neq j $, $\sum_i P_i = I $, और $ A = \sum_i \lambda_iP_i $
(द \emph{वर्णक्रमीय अपघटन}); अभिव्यक्त करना $ f(A)$ किसी के लिए
बहुपद $ f $ के रूप में $ P_i $.
\end{exercise}

\section{समस्या: कूरेंट--फिशर, वेइल, और कैलकुलस
eigenvalues}

\begin{problem}\label{pb:b2:hermitian:1}
 eigenvalues एक का \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} मैट्रिक्स केवल a की जड़ें नहीं हैं
बहुपद: वे के समाधान हैं \emph{अनुकूलन समस्याएं}.
वह परिवर्तनशील दृष्टिकोण --- \omterm{pb:b2:hermitian:1}{रेले भागफल} और यह
\emph{कूरेंट--फिशर न्यूनतम-अधिकतम प्रमेय}\index{कूरेंट--फिशर न्यूनतम-अधिकतम प्रमेय} --- बनाता है eigenvalues तुलनीय, स्थिर, और
गणना योग्य, और यह समस्या अपनी शास्त्रीय फ़सलें उत्पन्न करती है:
\emph{वेइल की गड़बड़ी असमानताएँ}\index{वेइल की गड़बड़ी असमानताएँ}, \emph{कॉची इंटरलेसिंग}, शूर और क्यू फैन
असमानताओं का पता लगाएं, मैट्रिक्स वर्गमूल की एकरसता,
और \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} असतत लाप्लासियन का। लगातार, $ A, B,
E $ हैं \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} पर $ E = \C^n $ साथ eigenvalues में सूचीबद्ध
घटता क्रम $\lambda_1(A) \geq \dots \geq \lambda_n(A)$,
और $ R_A(x) = \frac{\langle x, Ax\rangle}{\langle x, x\rangle}$
के लिए $ x \neq 0$ है \emph{रेले भागफल}\index{रेले भागफल}.

\medskip
\noindent\textbf{Part I --- \omterm{pb:b2:hermitian:1}{रेले भागफल} and min-max.}
एक ऑर्थोनॉर्मल ईजेनबासिस को ठीक करें $(e_1, \dots, e_n)$, $ Ae_i =
\lambda_ie_i $.
\begin{enumerate}
  \item बताते हैं कि $ R_A(x)$ वास्तविक है, और वह
        \[
        \lambda_n \leq R_A(x) \leq \lambda_1
        \qquad (x \neq 0),
        \]
        दोनों सीमाएँ प्राप्त हुईं: $\lambda_1 = \max R_A $,
        $\lambda_n = \min R_A $.
  \item दिखाएँ कि के महत्वपूर्ण बिंदु $ R_A $ \omterm{def:b2:series:def}{बिल्कुल} वही हैं
        eigenvectors का $ A $ \emph{(बढ़ाना $ t \mapsto R_A(x +
        tv)$ पर $ t = 0$ के लिए $ v $ मनमाना, फिर बदलें $ v $ द्वारा
        $\iu v $)}.
  \item होने देना $ V_k = \operatorname{Vect}(e_1, \dots, e_k)$ और
        $ W_k = \operatorname{Vect}(e_k, \dots, e_n)$. दिखाओ
        \[
        \min_{x \in V_k\setminus\{0\}} R_A(x) = \lambda_k
        = \max_{x \in W_k\setminus\{0\}} R_A(x) .
        \]
  \item सिद्ध करो \emph{कूरेंट--फिशर प्रमेय}: के लिए $1 \leq
        k \leq n $,
        \[
        \lambda_k = \max_{\dim V = k}\;\min_{x \in
        V\setminus\{0\}} R_A(x)
        = \min_{\dim W = n-k+1}\;\max_{x \in
        W\setminus\{0\}} R_A(x)
        \]
        \emph{(किसी के लिए $ V $ आयाम का $ k $: $ V \cap W_k \neq
        \{0\}$ ग्रासमैन द्वारा, इसलिए $\min_V R_A \leq \lambda_k $;
        प्रश्न 3 दर्शाता है कि सीमा प्राप्त हो गई है)}.
  \item (एकरसता) लिखें $ A \leq B $ कब $ B - A $ है
        सकारात्मक अर्धनिश्चित. प्रश्न 4 से निष्कर्ष निकालें: $ A \leq
        B $ तात्पर्य $\lambda_k(A) \leq \lambda_k(B)$ हरएक के लिए
        $ k $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Weyl's inequalities.}
\begin{enumerate}[resume]
  \item वह उप-स्थान दिखाएँ $ V, W \subseteq \C^n $ साथ $\dim V
        + \dim W > n $ गैर-तुच्छ रूप से प्रतिच्छेद करें, और सामान्यीकरण करें:
        $\dim(V_1 \cap V_2 \cap V_3) \geq \dim V_1 + \dim V_2 +
        \dim V_3 - 2n $.
  \item सिद्ध करना \emph{वेइल की असमानता}: के लिए $ i + j - 1 \leq n $,
        \[
        \lambda_{i+j-1}(A + B) \leq \lambda_i(A) +
        \lambda_j(B)
        \]
        \emph{(उपस्थानों को प्रतिच्छेद करें $ W_i(A)$, $ W_j(B)$ और
        $ V_{i+j-1}(A+B)$ प्रश्न 3 और गिनती का
        आयाम)}.
  \item परिभाषित करना $\vertiii{E}_2 = \max_{\norm x = 1}\norm{Ex}$
        और दिखाओ $\vertiii E_2 = \max_k\abs{\lambda_k(E)}$ के लिए
        \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} $ E $. परिणाम निकालना \emph{वेइल की परेशानी
        प्रमेय}:
        \[
        \bigl|\lambda_k(A + E) - \lambda_k(A)\bigr| \leq
        \vertiii{E}_2
        \qquad (1 \leq k \leq n) :
        \]
        प्रत्येक \omterm{def:b2:reduction:eigen}{eigenvalue} एक है $1$-लिप्सचिट्ज़ फ़ंक्शन का
        मैट्रिक्स.
  \item (रैंक-एक गड़बड़ी) चलो $ P $ होना \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} सकारात्मक
        रैंक का अर्धनिश्चित $1$. दिखाओ
        \[
        \lambda_k(A) \leq \lambda_k(A + P) \leq
        \lambda_{k-1}(A) \qquad (2 \leq k \leq n),
        \]
        के साथ साथ $\lambda_1(A) \leq \lambda_1(A+P)$: नई eigenvalues \emph{जिल्द} पुराने वाले.
  \item प्रश्न 8 को संख्यात्मक रूप से जांचें: $ A = \begin{pmatrix} 2 &
        \iu\\ -\iu & 2\end{pmatrix}$
        (\cref{exo:b2:hermitian:4}: \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} $\{3, 1\}$) और
        $ E = \begin{pmatrix} 0 & 1\\ 1 & 0\end{pmatrix}$
        (\omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} $\{1, -1\}$): गणना करें \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} का $ A +
        E $ और असमानता के दोनों पक्ष।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Interlacing and trace
inequalities.}
\begin{enumerate}[resume]
  \item (कॉची इंटरलेसिंग) चलो $ B $ अग्रणी बनें $(n-1)
        \times(n-1)$ का प्रमुख सबमैट्रिक्स $ A $. सिद्ध करना
        \[
        \lambda_{k+1}(A) \leq \lambda_k(B) \leq \lambda_k(A)
        \qquad (1 \leq k \leq n-1)
        \]
        \emph{(देखना $\C^{n-1} \subseteq \C^n $; उस पर, $ R_B $ है
        का प्रतिबंध $ R_A $; कूरेंट--फिशर को लागू करें
        दोनों स्तर)}.
  \item पुनरावृति: एक प्रमुख सबमैट्रिक्स के लिए $ B $ आकार का $ n - m $,
        $\lambda_{k+m}(A) \leq \lambda_k(B) \leq
        \lambda_k(A)$.
  \item (शूर) चलो $ d_1 \geq d_2 \geq \dots \geq d_n $ हो
        की विकर्ण प्रविष्टियाँ $ A $, क्रमबद्ध। प्रत्येक के लिए सिद्ध करें $ k $:
        \[
        \sum_{i=1}^{k} d_i \leq \sum_{i=1}^{k}\lambda_i(A),
        \]
        समानता के साथ $ k = n $ (निशान)
        \emph{(द $ k $ चुनी गई विकर्ण प्रविष्टियाँ फॉर्म ए
        प्रधान $ k\times k $ सबमैट्रिक्स; इसके निशान को बांध दिया
        प्रश्न 12)}.
  \item (के फैन) साबित करो
        \[
        \sum_{i=1}^{k}\lambda_i(A)
        = \max\Bigl\{\sum_{i=1}^{k}\langle x_i, Ax_i\rangle
        : (x_1, \dots, x_k) \text{ orthonormal}\Bigr\} .
        \]
  \item प्रश्न 11 और 13 को सत्यापित करें
        \[
        A = \begin{pmatrix} 2 & 1 & 0\\ 1 & 2 & 1\\
        0 & 1 & 2\end{pmatrix}
        \qquad
        \bigl(\operatorname{Sp} = \{2 - \sqrt2,\ 2,\
        2+\sqrt2\}\bigr)
        \]
        इसके नेतृत्व के खिलाफ $2\times2$ अवरोध पैदा करना (\omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} $\{1,
        3\}$) और इसका विकर्ण।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The Loewner order.} $ A \leq B $
अभी भी मतलब है $ B - A $ सकारात्मक अर्धनिश्चित; इसमें सभी मैट्रिक्स
भाग हैं \omterm{def:b2:hermitian:adjoint}{हर्मिटियन}.
\begin{enumerate}[resume]
  \item दिखाओ: $ A \leq B $ तात्पर्य $ a_{ii} \leq b_{ii}$ सभी के लिए
        $ i $, $\operatorname{tr} A \leq \operatorname{tr} B $,
        और $ C^\dagger AC \leq C^\dagger BC $ के लिए \emph{प्रत्येक}
        जटिल मैट्रिक्स $ C $.
  \item दिखाएँ कि वर्ग करना है \emph{नहीं} मोनोटोन: के लिए
        \[
        A = \begin{pmatrix} 1 & 0\\ 0 & 0\end{pmatrix},
        \qquad
        B = \begin{pmatrix} 2 & 1\\ 1 & 1\end{pmatrix},
        \]
        जाँच करना $0 \leq A \leq B $ लेकिन $ A^2 \not\leq B^2$.
  \item सिद्ध कीजिए कि वर्गमूल \emph{है} एकस्वर: $0 \leq
        A \leq B $ तात्पर्य $\sqrt A \leq \sqrt B $ \emph{(होने देना
        $\mu $ एक हो \omterm{def:b2:reduction:eigen}{eigenvalue} का $\sqrt B - \sqrt A $ इकाई के साथ
        आइजन्वेक्टर $ v $; गणना $\langle v, (B - A)v\rangle =
        \mu\bigl(\langle v, \sqrt B\,v\rangle + \langle v,
        \sqrt A\,v\rangle\bigr)$ और चर्चा करें)}.
  \item सिद्ध कीजिए कि व्युत्क्रम सकारात्मक निश्चित पर प्रतिस्वर है
        मैट्रिक्स: $0 < A \leq B $ तात्पर्य $ B^{-1} \leq A^{-1}$
        \emph{(बधाई दें $ A^{-1/2}$ कम करने के लिए $ I \leq M
        \Rightarrow M^{-1} \leq I $, जो कि अदिश राशि है
        वर्णक्रमीय आधार)}.
  \item होने देना $ A, B $ सकारात्मक निश्चित रहें. दिखाओ कि
        eigenvalues का $ AB $ (नहीं \omterm{def:b2:hermitian:adjoint}{हर्मिटियन} सामान्य तौर पर!) हैं
        वास्तविक और सकारात्मक, और वह
        \[
        \lambda_{\max}(AB) \leq
        \lambda_{\max}(A)\,\lambda_{\max}(B)
        \]
        \emph{(द्वारा संयुग्मित करें $\sqrt A $: $ AB \sim \sqrt
        A\,B\sqrt A $)}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The discrete Laplacian, worked.}
होने देना $ T_n $ हो $ n \times n $ त्रिविकर्ण मैट्रिक्स के साथ $2$ पर
विकर्ण और $-1$ दो आसन्न विकर्णों पर.
\begin{enumerate}[resume]
  \item साथ $\theta_k = \frac{k\pi}{n+1}$, सत्यापित करें कि
        वेक्टर $ v_k = \bigl(\sin(j\theta_k)\bigr)_{1\leq j\leq
        n}$ संतुष्ट $ T_nv_k = (2 - 2\cos\theta_k)\,v_k $
        \emph{(उत्पाद-से-योग पहचान; सीमा पंक्तियों की जाँच करें
        $ j = 1, n $)}. निष्कर्ष:
        \[
        \operatorname{Sp}(T_n) = \Bigl\{4\sin^2
        \frac{k\pi}{2(n+1)} : 1 \leq k \leq n\Bigr\},
        \]
        सब कुछ सरल, सब कुछ सकारात्मक: $ T_n $ सकारात्मक निश्चित है.
  \item (एक क्षमता) एक वास्तविक विकर्ण के लिए $ D =
        \operatorname{diag}(d_1, \dots, d_n)$, सैंडविच
        \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम}: हरएक के लिए $ k $,
        \[
        \lambda_k(T_n) + \min_i d_i \;\leq\;
        \lambda_k(T_n + D) \;\leq\; \lambda_k(T_n) + \max_i
        d_i .
        \]
  \item बीच में कॉची इंटरलेसिंग की जाँच करें $ T_3$ और $ T_2$
        स्पष्ट रूप से (स्पेक्ट्रा $\{2 \pm \sqrt2, 2\}$ और $\{1,
        3\}$), और व्याख्या करें: $ T_2$ है $ T_3$ एक समापन बिंदु के साथ
        पथ का हटाया गया.
  \item चरम दिखाओ eigenvalues संतुष्ट करें, जैसे $ n \to
        \infty $:
        \[
        \lambda_{\min}(T_n) = 4\sin^2\frac{\pi}{2(n+1)}
        \sim \frac{\pi^2}{(n+1)^2},
        \qquad
        \lambda_{\max}(T_n) \to 4 ,
        \]
        तो शर्त संख्या $\kappa_n =
        \lambda_{\max}/\lambda_{\min}$ जैसे बढ़ता है
        $\frac{4(n+1)^2}{\pi^2}$: एक सेकंड का विवेक करना
        एक बेहतर और बेहतर ग्रिड पर व्युत्पन्न है
        आंतरिक रूप से अस्वस्थ.
  \item संश्लेषण। प्रत्येक एक वाक्य में: (i) क्यों
        परिवर्तनशील लक्षण वर्णन, नहीं \omterm{def:b2:reduction:charpoly}{विशेषता
        बहुपद}, वही बनाता है eigenvalues स्थिर
        (प्रश्न 8--9); (ii) केवल कौन से प्रश्न प्रयुक्त हुए
        $\lambda_1 = \max R_A$ और जिसकी पूरी जरूरत थी
        न्यूनतम-अधिकतम; (iii) लोवेनर का \omterm{def:b2:structures:generated}{आदेश} इसमें क्या जोड़ता है
        कहानी; (iv) जहां ये उपकरण पुनः प्रकट होते हैं (संख्यात्मक)।
        प्रश्न 24 की कठोरता मैट्रिक्स का विश्लेषण; क्वांटम
        गड़बड़ी सिद्धांत; और, वर्ष 3 खंड में,
