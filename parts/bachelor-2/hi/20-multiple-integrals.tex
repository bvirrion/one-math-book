\chapter{लाइन इंटीग्रल और मल्टीपल इंटीग्रल}\label{ch:b2:multint}

यह अध्याय \omterm{def:b2:diffcalc:differential}{अंतर}ालों से लेकर वक्रों तक एकीकरण का विस्तार करता है
समतल और \omterm{def:b2:diffcalc:differential}{अंतर}िक्ष के डोमेन। रेखा अभिन्न एकीकृत करें ए
\emph{\omterm{def:b2:diffcalc:differential}{विभेदक} रूप} $P\,\dd x + Q\,\dd y $ एक उन्मुख के साथ
चाप; डबल और ट्रिपल इंटीग्रल्स कार्यों को एकीकृत करते हैं
द्वि- और त्रि-आयामी डोमेन। दोनों सिद्धांत मिलते हैं
\emph{ग्रीन--रीमैन प्रमेय}, द्वि-आयामी मौलिक
कैलकुलस का प्रमेय, और संपूर्ण मुख्य कम्प्यूटेशनल उपकरण है
द \emph{चर सूत्र का परिवर्तन}, जिसका विकृति कारक है
पूर्ण जैकोबियन सिद्ध.

\section{रेखा अभिन्न}

\begin{definition}[विभेदक रूप; लाइन इंटीग्रल]\label{def:b2:multint:lineint}
होने देना $ U \subseteq \R^2$ होना \omterm{def:b2:metric:topology}{खुला}. ए \emph{अंतर
प्रपत्र}\index{अंतर
प्रपत्र} का
डिग्री $1$ और वर्ग $\mathcal{C}^0$ पर $ U $ अभिव्यक्ति है
$\omega = P\,\dd x + Q\,\dd y $ साथ $ P, Q \colon U \to \R $
\omterm{def:b2:metric:continuity}{निरंतर} --- औपचारिक रूप से, ए \omterm{def:b2:metric:continuity}{निरंतर} से मानचित्र $ U $ में \omterm{def:b2:linalg:dual}{दोहरी}
का $\R^2$, $\omega(M) = P(M)\,e_1^* + Q(M)\,e_2^*$. एक के लिए
$\mathcal{C}^1$ आर्क $\gamma \colon [a, b] \to U $, $\gamma(t) =
(x(t), y(t))$, द \emph{लाइन इंटीग्रल}\index{लाइन इंटीग्रल} का
$\omega $ साथ में $\gamma $
है
\[
\int_\gamma \omega
= \int_a^b \Bigl(P(\gamma(t))\,x'(t)
+ Q(\gamma(t))\,y'(t)\Bigr)\,\dd t .
\]
परिभाषाएँ शब्दशः विस्तारित होती हैं $\R^3$ (रूप $ P\,\dd x +
Q\,\dd y + R\,\dd z $) और टुकड़े-टुकड़े करने के लिए $\mathcal{C}^1$ चाप (योग)
टुकड़ों के ऊपर)।
\end{definition}

\begin{proposition}[अपरिवर्तनशीलता और अभिविन्यास]\label{prop:b2:multint:lineinv}
 \omterm{def:b2:multint:lineint}{लाइन इंटीग्रल} बढ़ते हुए के तहत अपरिवर्तित है $\mathcal{C}^1$
\omterm{def:b2:curves:reparam}{पैरामीटर का परिवर्तन}, और घटते हुए चिह्न के नीचे चिह्न बदलता है। यह
इसलिए केवल पर निर्भर करता है \emph{उन्मुखी} \omterm{def:b2:curves:reparam}{ज्यामितीय चाप}.
\end{proposition}

\begin{proof}
अगर $\theta \colon [c, d] \to [a, b]$ एक है \omterm{def:b2:curves:reparam}{पैरामीटर का परिवर्तन} और
$\tilde\gamma = \gamma \circ \theta $, फिर श्रृंखला नियम द्वारा और
चरों का एक-परिवर्तनीय परिवर्तन $ t = \theta(u)$,
\[
\int_{\tilde\gamma}\omega
= \int_c^d \bigl(P(\gamma(\theta(u)))\,x'(\theta(u))
+ Q(\gamma(\theta(u)))\,y'(\theta(u))\bigr)\,\theta'(u)\,\dd u
= \pm\int_a^b \bigl(Px' + Qy'\bigr)(t)\,\dd t ,
\]
संकेत के साथ $+$ अगर $\theta $ यह बढ़ रहा है ($\theta(c) = a $) और $-$
यदि घट रही है (सीमाएं बदल जाती हैं)।
\end{proof}

\begin{example}[एक बल का कार्य; प्रसार]\label{ex:b2:multint:work}
अगर $ F = (P, Q)$ एक बल \omterm{def:b2:multint:domain}{क्षेत्र} है, $\int_\gamma P\dd x + Q\dd y =
\int_a^b \langle F(\gamma(t)), \gamma'(t)\rangle\,\dd t $ है
\emph{काम} का $ F $ साथ में $\gamma $. के लिए $\omega = -y\,\dd x +
x\,\dd y $ वामावर्त इकाई वृत्त के साथ $\gamma(t) =
(\cos t, \sin t)$:
\[
\int_\gamma \omega
= \int_0^{2\pi}\bigl((-\sin t)(-\sin t)
+ \cos t\cos t\bigr)\,\dd t = 2\pi ,
\]
संलग्न से दोगुना \omterm{def:b2:multint:domain}{क्षेत्र} --- ग्रीन--रीमैन का पहला संकेत।
\end{example}

\begin{example}[एक अभिन्न, दो पैरामीट्रिजेशन, एक
संकेत जाल]\label{ex:b2:multint:semicircle}
गणना करना $\int_\gamma x\,\dd y $ ऊपरी इकाई के साथ
से अर्धवृत्त $(1, 0)$ को $(-1, 0)$. साथ $\gamma(t) =
(\cos t, \sin t)$, $ t \in \intcc0\pi $:
\[
\int_0^\pi\cos t\cdot\cos t\,\dd t = \frac\pi2 .
\]
ग्राफ पैरामीट्रिजेशन के साथ $ x \mapsto (x, \sqrt{1 -
x^2})$, $ x $ से $1$ को $-1$ (दिशा ध्यान दें!):
\[
\int_1^{-1}x\cdot\frac{-x}{\sqrt{1 - x^2}}\,\dd x
= \int_{-1}^{1}\frac{x^2}{\sqrt{1 - x^2}}\,\dd x
= \frac\pi2
\]
($ x = \sin u $ इसे वालिस इंटीग्रल में कम कर देता है)। वही मूल्य,
जैसे \cref{prop:b2:multint:lineinv} गारंटी --- लेकिन केवल
क्योंकि दोनों रन चलते हैं $(1,0)$ को $(-1,0)$; को उलट देना
यात्रा संकेत फ़्लिप करती है। के साथ रास्ता बंद करना $ x $-अक्ष
(कहां $\dd y = 0$) कुछ नहीं जोड़ता, और कुल $\frac\pi2$
है \omterm{def:b2:multint:domain}{क्षेत्र} अर्ध-डिस्क का: का पहला उदाहरण
सीमा-क्षेत्र ग्रीन--रीमैन के सूत्र नीचे।
\end{example}

\begin{definition}[सटीक और बंद प्रपत्र]\label{def:b2:multint:exact}
फार्म $\omega = P\,\dd x + Q\,\dd y $ कक्षा का $\mathcal{C}^0$
है \emph{एकदम सही}\index{एकदम सही} पर $ U $ अगर वहाँ होता $ f \in \mathcal{C}^1(U)$
(ए \emph{संभावना}\index{संभावना}) साथ $\omega = \dd f $, यानी\ $ P = f_x $ और $ Q
= f_y $. ए $\mathcal{C}^1$ रूप है \emph{बंद किया हुआ}\index{बंद किया हुआ} अगर $ P_y = Q_x $ पर
$ U $.
\end{definition}

\begin{theorem}[पंक्ति के लिए मौलिक प्रमेय
अभिन्न]\label{thm:b2:multint:ftc}
अगर $\omega = \dd f $ है \omterm{def:b2:multint:exact}{एकदम सही} और $\gamma $ एक टुकड़ावार है
$\mathcal{C}^1$ अंदर आर्क $ U $ से $ A $ को $ B $, तब
\[
\int_\gamma \omega = f(B) - f(A) .
\]
विशेष रूप से एक का अभिन्न अंग सटीक रूप किसी भी बंद चाप के साथ
शून्य है, और प्रत्येक \omterm{def:b2:multint:exact}{एकदम सही} $\mathcal{C}^1$ फॉर्म बंद है.
\end{theorem}

\begin{proof}
$\frac{\dd}{\dd t}f(\gamma(t)) = f_x(\gamma(t))x'(t) +
f_y(\gamma(t))y'(t)$ शृंखला नियम द्वारा
(\cref{ch:b2:diffcalc}), इसलिए इंटीग्रैंड इन
\cref{def:b2:multint:lineint} का व्युत्पन्न है $ t \mapsto
f(\gamma(t))$, और कैलकुलस का मौलिक प्रमेय देता है
प्रत्येक टुकड़े पर परिणाम; मध्यवर्ती मान दूरबीन। बंदपन
का \omterm{def:b2:multint:exact}{एकदम सही} $\mathcal{C}^1$ प्रपत्र श्वार्ज का प्रमेय है: $ P_y =
f_{xy} = f_{yx} = Q_x $.
\end{proof}

\begin{example}[एक क्षमता का पुनर्निर्माण]\label{ex:b2:multint:potential}
होने देना $\omega = y\,\eu^{xy}\,\dd x + (x\,\eu^{xy} + 2y)\,\dd y $
पर $\R^2$. यह बंद है: दोनों क्रॉस डेरिवेटिव बराबर हैं
$\eu^{xy}(1 + xy)$. एक खोजने के लिए \omterm{def:b2:multint:exact}{संभावना}, एकीकृत करें $ P $ में
$ x $ निश्चित पर $ y $:
\[
f(x, y) = \int y\,\eu^{xy}\,\dd x = \eu^{xy} + c(y),
\]
फिर समायोजित करें $ c $ मिलान करके $ f_y $: $ x\,\eu^{xy} + c'(y) =
x\,\eu^{xy} + 2y $ देता है $ c(y) = y^2$. इसलिए $ f(x,y) = \eu^{xy} +
y^2$, और किसी भी टुकड़े के लिए $\mathcal C^1$ चाप से $(0,0)$
को $(1,1)$,
\[
\int_\gamma\omega = f(1,1) - f(0,0) = (\eu + 1) - 1 = \eu ,
\]
पथ से स्वतंत्र --- दो-चरणीय नुस्खा (एकीकृत करें)।
में $ x $, में सही $ y $) का व्यावहारिक विपरीतार्थक है
\cref{thm:b2:multint:ftc} डोमेन पर जहां बंद प्रपत्र हैं
\omterm{def:b2:multint:exact}{एकदम सही}.
\end{example}

\begin{example}[बंद का मतलब सटीक नहीं है]\label{ex:b2:multint:angleform}
पर $ U = \R^2 \setminus \{0\}$, द \emph{कोण रूप}
\[
\omega = \frac{-y\,\dd x + x\,\dd y}{x^2 + y^2}
\]
बंद है (प्रत्यक्ष गणना: दोनों $ P_y $ और $ Q_x $ बराबर
$\frac{y^2 - x^2}{(x^2+y^2)^2}$), लेकिन यह इकाई के साथ अभिन्न है
वृत्त है $2\pi \neq 0$ (समान गणना
\cref{ex:b2:multint:work}, द्वारा विभाजित $1$): $\omega $ क्या नहीं है \omterm{def:b2:multint:exact}{एकदम सही}
पर $ U $. स्थानीय स्तर पर, $\omega = \dd\theta $ एक संकल्प के लिए
$\theta $ ध्रुवीय कोण का; विफलता वैश्विक है---कोण
परिभाषित नहीं किया जा सकता लगातार पंचर के आसपास. डोमेन पर
छेद के बिना विकृति गायब हो जाती है: पर ए \emph{तारे के आकार का}
\omterm{def:b2:metric:topology}{खुला} सेट, हर बंद $\mathcal{C}^1$ रूप है \omterm{def:b2:multint:exact}{एकदम सही}
(पोंकारे की लेम्मा, \cref{exo:b2:multint:8}).
\end{example}

\section{दोहरा अभिन्न अंग}

हम रीमैन इंटीग्रल के सिद्धांत को एक रूप में स्वीकार करते हैं
परिवर्तनशील (वर्ष 1 खंड, और \cref{ch:b2:integration}) और स्केच
इसका दो-चर संस्करण। एक समारोह $ f $ \omterm{def:b2:metric:continuity}{निरंतर} एक आयत पर
$ R = [a, b] \times [c, d]$ दोहरा अभिन्न अंग है $\iint_R f $,
रीमैन द्वारा ग्रिड पर योगों को \omterm{def:b2:series:def}{बिल्कुल} एक चर के रूप में परिभाषित किया गया है, और
पुनरावृत्ति द्वारा गणना:

\begin{theorem}[एक आयत पर फ़ुबिनी]\label{thm:b2:multint:fubini}
के लिए $ f $ \omterm{def:b2:metric:continuity}{निरंतर} पर $ R = [a,b] \times [c,d]$,
\[
\iint_R f
= \int_a^b \Bigl(\int_c^d f(x, y)\,\dd y\Bigr)\dd x
= \int_c^d \Bigl(\int_a^b f(x, y)\,\dd x\Bigr)\dd y .
\]
\end{theorem}

\begin{proof}
तय करना $ F(x) = \int_c^d f(x, y)\,\dd y $. वर्दी निरंतरता का $ f $ पर \omterm{def:b2:metric:compact}{कॉम्पैक्ट} $ R $ बनाता है $ F $ \omterm{def:b2:metric:continuity}{निरंतर} (प्रभुत्वपूर्ण अनुमान:
$\abs{F(x) - F(x')} \leq (d - c)\sup_y\abs{f(x,y) - f(x',y)}$).
अब उपविभाजन करें $[a,b]$ और $[c,d]$ में $ n $ समान भाग, एक देना
कोशिकाओं का ग्रिड $ R_{ij}$ का \omterm{def:b2:multint:domain}{क्षेत्र} $\Delta x\,\Delta y $. प्रत्येक कोशिका पर,
$\inf_{R_{ij}} f \cdot \Delta x \Delta y \leq
\int_{x_{i-1}}^{x_i}\int_{y_{j-1}}^{y_j} f(x,y)\,\dd y\,\dd x \leq
\sup_{R_{ij}} f \cdot \Delta x \Delta y $ की एकरसता से
एक-चर अभिन्न (दो बार लागू)। कोशिकाओं पर संक्षेपण,
पुनरावृत्त अभिन्न $\int_a^b F $ निचले और के बीच निचोड़ा हुआ है
ग्रिड के ऊपरी रीमैन योग; वर्दी द्वारा निरंतरता दोनों योग
सामान्य मूल्य परिभाषित करने के लिए \omterm{def:b2:integration:improper}{अभिसरण} करें $\iint_R f $ जैसा $ n \to
\infty $. यही तर्क भूमिकाओं के साथ भी लागू होता है $ x $ और $ y $
आदान-प्रदान किया गया, इसलिए दोनों पुनरावृत्त अभिन्न अंग बराबर हैं $\iint_R f $.
\end{proof}

\begin{remark}
निरंतरता एक पर \omterm{def:b2:metric:compact}{कॉम्पैक्ट} आयत वास्तविक कार्य कर रहा है
फ़ुबिनी का प्रमाण: यह वर्दी प्रदान करता है निरंतरता वह
रीमैन रकम को निचोड़ता है। वाइल्डर इंटीग्रैंड्स के लिए
कथन वास्तव में विफल रहता है --- ऐसे कार्य हैं जिनके दो
पुनरावृत्त अभिन्न अंग मौजूद हैं और भिन्न हैं। ईमानदार जनरल
प्रमेय, पूर्णीकरणीयता के साथ एकमात्र परिकल्पना है
लेबेस्ग इंटीग्रल के लिए फ़ुबिनी का प्रमेय, में सिद्ध हुआ
वर्ष 3 खंड; इस अध्याय में सब कुछ भीतर रहता है
\omterm{def:b2:metric:continuity}{निरंतर} यह निर्धारित करना कि उपरोक्त प्राथमिक प्रमाण कहाँ है
\omterm{def:b2:metric:complete}{पूरा}.
\end{remark}

\begin{definition}[प्राथमिक डोमेन]\label{def:b2:multint:domain}
एक डोमेन $ D \subseteq \R^2$ है \emph{$ y $-प्राथमिक} अगर
\[
D = \{(x, y) : a \leq x \leq b,\
\varphi_1(x) \leq y \leq \varphi_2(x)\}
\]
साथ $\varphi_1 \leq \varphi_2$ \omterm{def:b2:metric:continuity}{निरंतर} पर $[a,b]$
($ x $-प्राथमिक: \omterm{def:b2:quadratic:adjoint}{सममित} रूप से)। के लिए $ f $ \omterm{def:b2:metric:continuity}{निरंतर} एक पर
$ y $-प्राथमिक $ D $,
\[
\iint_D f
= \int_a^b\Bigl(
\int_{\varphi_1(x)}^{\varphi_2(x)} f(x,y)\,\dd y\Bigr)\dd x ,
\]
और एक जाँच करता है (विस्तार करके $ f $ सन्निकटन तर्क द्वारा, या
उपविभाजन करके) कि कब $ D $ दोनों दिशाओं में प्राथमिक है
दो पुनरावृत्त अभिन्न अंग सहमत हैं। डोमेन को बहुत से टुकड़ों में काटा जाता है
प्राथमिक टुकड़ों को एडिटिविटी द्वारा नियंत्रित किया जाता है, और
\emph{क्षेत्र}\index{क्षेत्र} का $ D $ है $\operatorname
{Area}(D) = \iint_D 1$.
\end{definition}

\begin{example}\label{ex:b2:multint:triangle}
त्रिकोण पर $ D = \{0 \leq x \leq 1,\ 0 \leq y \leq x\}$:
\[
\iint_D xy \,\dd x\,\dd y
= \int_0^1 x\Bigl(\int_0^x y\,\dd y\Bigr)\dd x
= \int_0^1 x\cdot\frac{x^2}{2}\,\dd x = \frac18 .
\]
ऑर्डर की अदला-बदली ($ x $ से $ y $ को $1$): $\int_0^1
y\bigl(\int_y^1 x\,\dd x\bigr)\dd y = \int_0^1
y\,\frac{1 - y^2}{2}\,\dd y = \frac18$ --- समान मूल्य, भिन्न
गणना: एकीकरण के क्रम को चुनना अच्छी तरह से आधा है
शिल्प.
\end{example}

\begin{example}[जब केवल एक ही आदेश काम करता है]\label{ex:b2:multint:swaporder}
गणना करना $ I = \displaystyle\int_0^1\!\!\int_x^1
\eu^{y^2}\,\dd y\,\dd x $. जैसा लिखा है, आंतरिक अभिन्न
$\int\eu^{y^2}\dd y $ इसका कोई प्राथमिक प्रतिअवकलन नहीं है: the
गणना अटकी हुई है. लेकिन डोमेन त्रिकोण है $0 \leq
x \leq y \leq 1$, जो दोनों दिशाओं में प्राथमिक है;
ऑर्डर की अदला-बदली,
\[
I = \int_0^1\!\!\int_0^y \eu^{y^2}\,\dd x\,\dd y
= \int_0^1 y\,\eu^{y^2}\,\dd y
= \Bigl[\tfrac12\eu^{y^2}\Bigr]_0^1 = \frac{\eu - 1}{2} .
\]
आंतरिक चर $ x $ इंटीग्रैंड में कहीं भी दिखाई नहीं दिया, इसलिए
इसे एकीकृत करने से सबसे पहले सटीक कारक उत्पन्न हुआ $ y $ वह
बाहरी अभिन्न को तत्काल बनाता है। नैतिक: फ़ुबिनी बस नहीं है
पुनरावृत्त करने का लाइसेंस --- यह करने का लाइसेंस है \emph{चुनना},
और सही क्रम एक असंभव अभिन्न अंग को एक में बदल सकता है
एक पंक्ति. हमेशा डोमेन का स्केच बनाएं और दोनों को पढ़ें
शुरू करने से पहले विवरण.
\end{example}

\begin{theorem}[चरों का परिवर्तन]\label{thm:b2:multint:changevar}
होने देना $\Phi \colon U' \to U $ एक हो $\mathcal{C}^1$ भिन्नता
के बीच \omterm{def:b2:metric:topology}{खुला} के सेट $\R^2$, होने देना $ K \subseteq U $ एक हो \omterm{def:b2:metric:compact}{कॉम्पैक्ट}
डोमेन को प्राथमिक टुकड़ों में काटा गया $ K' = \Phi^{-1}(K)$, और
चलो $ f $ होना \omterm{def:b2:metric:continuity}{निरंतर} पर $ K $. तब
\[
\iint_K f(x, y)\,\dd x\,\dd y
= \iint_{K'} f\bigl(\Phi(u, v)\bigr)\,
\abs{\det J_\Phi(u, v)}\,\dd u\,\dd v .
\]
\end{theorem}
\admitted

\begin{remark}
 \omterm{def:b2:metric:complete}{पूरा} प्रमाण---अनुमानित करना $\Phi $ इसके द्वारा \omterm{def:b2:diffcalc:differential}{अंतर} पर
एक बढ़िया ग्रिड और सीमा कोशिकाओं को नियंत्रित करना --- हालांकि लंबा है
गहरा नहीं; यह वर्ष 3 के माप सिद्धांत में पूर्ण रूप से किया गया है
लेबेस्ग्यू सिद्धांत का परिणाम. अनुमानी चित्र है
सतह के लिए पहले से ही उपयोग किया जा चुका है \omterm{def:b2:multint:domain}{क्षेत्र}: भुजा का एक छोटा वर्ग $\dd u $ पर
$(u, v)$ पहले क्रम में, फैले हुए समांतर चतुर्भुज पर मैप किया जाता है
द्वारा $\Phi_u\,\dd u $ और $\Phi_v\,\dd v $, किसका \omterm{def:b2:multint:domain}{क्षेत्र} है
$\abs{\det J_\Phi}\,\dd u\,\dd v $
(\cref{lem:b2:surfaces:lagrange}).
\end{remark}

\begin{remark}[विधि: चरों में परिवर्तन का चयन करना]
तीन प्रतिवर्त अधिकांश मामलों को कवर करते हैं। \emph{की समरूपता
एकीकृत}: $ x^2 + y^2$ ध्रुवीय, एक उत्पाद संरचना की मांग करता है
कार्टेशियन अक्षों को रखने का आह्वान करता है। \emph{का आकार
सीमा}: सीमाएँ $ u(x,y) = c_1$, $ v(x,y) = c_2$ भीख मांगो
निर्देशांक $(u, v)$ स्वयं, अतिशयोक्तिपूर्ण के रूप में
\omterm{def:b2:multint:domain}{क्षेत्र} का उदाहरण नीचे दिया गया है --- डोमेन एक आयत बन जाता है,
जो पूरी जीत है. \emph{रैखिक संरचना}:
में अभिव्यक्तियाँ $ x + y $ और $ x - y $ को आमंत्रित करें $45$-डिग्री
घूर्णन या कतरनी (\cref{ex:b2:multint:affine}). कुल मिलाकर
मामले, एकीकृत करने से पहले टिक करने के लिए तीन बक्से: नक्शा एक है
पुराने डोमेन पर नए डोमेन का आक्षेप; यह जैकोबियन है
गणना की जाती है \emph{वास्तव में उपयोग की जाने वाली दिशा में} (पर पलटें
यदि आसान हो तो अंत); और जैकोबियन इसके साथ प्रवेश करता है
निरपेक्ष मूल्य.
\end{remark}

\begin{example}[ध्रुवीय निर्देशांक]\label{ex:b2:multint:polar}
$\Phi(\rho, \alpha) = (\rho\cos\alpha,\ \rho\sin\alpha)$ है
\[
J_\Phi = \begin{pmatrix}
\cos\alpha & -\rho\sin\alpha\\
\sin\alpha & \rho\cos\alpha
\end{pmatrix},
\qquad \det J_\Phi = \rho ,
\]
इसलिए $\dd x\,\dd y = \rho\,\dd\rho\,\dd\alpha $. डिस्क के लिए $ D_R $
त्रिज्या का $ R $:
\[
\iint_{D_R} e^{-(x^2 + y^2)}\,\dd x\,\dd y
= \int_0^{2\pi}\!\!\int_0^R e^{-\rho^2}\rho\,\dd\rho\,\dd\alpha
= \pi\bigl(1 - e^{-R^2}\bigr)
\xrightarrow[R\to\infty]{} \pi .
\]
वर्ग से तुलना $[-R, R]^2$ (जो बीच में निचोड़ा हुआ है
डिस्क $ D_R $ और $ D_{R\sqrt2}$, सभी इंटीग्रैंड्स पॉजिटिव) देता है
$\bigl(\int_{-\infty}^\infty e^{-x^2}\dd x\bigr)^2 = \pi $:
\[
\boxed{\ \int_{-\infty}^{+\infty} e^{-x^2}\,\dd x = \sqrt{\pi}\ }
\]
--- गॉस इंटीग्रल फिर से, अब इसके सबसे प्रसिद्ध प्रमाण से
(एक-चर व्युत्पत्ति की तुलना करें
\cref{ch:b2:integration}).
\end{example}

\begin{example}[चरों के परिवर्तन को प्रभावित करें]\label{ex:b2:multint:affine}
एक के लिए \omterm{def:b2:affine:subspace}{एफ़िन} मानचित्र $\Phi(u, v) = M(u, v)^{\mathsf T} + C $ साथ
$ M $ उलटा, जैकोबियन स्थिर मैट्रिक्स है $ M $:
\omterm{def:b2:multint:domain}{क्षेत्र}ों स्थिर कारक से गुणा किया जाता है $\abs{\det M}$
---में किया गया वादा \cref{ch:b2:affine} अब एक प्रमेय है.
दो तात्कालिक उपयोग. दीर्घवृत्त $\frac{x^2}{a^2} +
\frac{y^2}{b^2} \leq 1$ नीचे यूनिट डिस्क की छवि है
$(u, v) \mapsto (au, bv)$, तो यह है \omterm{def:b2:multint:domain}{क्षेत्र} है $ ab \cdot \pi $ ---
कोई गणना नहीं. और के अभिन्न अंग के लिए $ f(x + y)$ के ऊपर
वर्ग $ K = \intcc01^2$, कतरनी $\Phi(u, v) = (u - v, v)$
(सिद्ध $1$) इसे एक अभिन्न अंग में बदल देता है $ f(u)$ एक से अधिक
समांतर चतुर्भुज, जिसे फ़ुबिनी लगातार काटती है $ u $: साथ $ f =
\exp $,
\[
\iint_K \eu^{x+y}\,\dd x\,\dd y
= \Bigl(\int_0^1 \eu^x\,\dd x\Bigr)^2 = (\eu - 1)^2,
\]
जैसा कि उत्पाद संरचना पुष्टि करती है। निर्देशांक चुनना
इंटीग्रैंड के लिए अनुकूलित --- डोमेन के लिए नहीं --- है
शिल्प का दूसरा भाग.
\end{example}


\begin{example}[निर्देशांक एक वक्ररेखीय के लिए अनुकूलित
डोमेन]\label{ex:b2:multint:hyperbolicregion}
होने देना $ D $ से घिरा प्रथम चतुर्थांश का \omterm{def:b2:multint:domain}{क्षेत्र} हो

x = \sqrt{u/v}, \qquad y = \sqrt{uv},
\qquad
\det J = x_uy_v - x_vy_u = \frac{1}{2v}
\]
(एक दो-पंक्ति गणना के साथ $ x = u^{1/2}v^{-1/2}$, $ y =

\[
\operatorname{Area}(D)
= \int_1^3\!\!\int_1^3\frac{\dd u\,\dd v}{2v}
= 2\cdot\frac{\ln 3}{2} = \ln 3 \approx 1.10 .

टुकड़े करने की कोशिश कर रहा हूँ $ D $ कार्तीय निर्देशांक में इसका अर्थ है इसे काटना

व्यवहार्य, आनंदहीन और त्रुटि-प्रवण। नैतिक दोहराता है
\cref{ex:b2:multint:affine} पूरी ताकत से: पढ़ें
सीमा समीकरण, और चलो \emph{उन्हें} चुनें
निर्देशांक; जैकोबियन फिर वक्ररेखीय को परिवर्तित करता है
जाल की कोशिका \omterm{def:b2:multint:domain}{क्षेत्र}, \omterm{def:b2:series:def}{बिल्कुल} वैसा ही $\rho $ ध्रुवीय के लिए किया
निर्देशांक
\end{example}

\begin{example}[औसत मान]\label{ex:b2:multint:average}
 \emph{औसत मूल्य} का $ f $ एक डोमेन पर $ D $ है

\section{ग्रीन--रीमैन प्रमेय}

\begin{theorem}[हरा--रीमैन]\label{thm:b2:multint:green}
होने देना $ K \subseteq \R^2$ एक हो \omterm{def:b2:metric:compact}{कॉम्पैक्ट} डोमेन जो प्राथमिक है
दोनों दिशाएँ (या ऐसे चिपके हुए खंडों का एक सीमित संघ),
सीमा के साथ $\partial K $ एक टुकड़ावार $\mathcal{C}^1$ बंद
वक्र उन्मुख \emph{वामावर्त} (डोमेन पर रहता है
बाएं). के लिए $ P, Q $ कक्षा का $\mathcal{C}^1$ के एक पड़ोस पर
$ K $:
\[
\oint_{\partial K} P\,\dd x + Q\,\dd y
= \iint_K \Bigl(\frac{\partial Q}{\partial x}
- \frac{\partial P}{\partial y}\Bigr)\,\dd x\,\dd y .
\]
\end{theorem}

\begin{proof}
सबसे पहले, दोनों पक्ष काटने के तहत योजक हैं $ K $ साथ में ए
दो टुकड़ों में विभाजित करें $ K_1, K_2$: दोहरे अभिन्न अंग जोड़ते हैं
की additiveity द्वारा $\iint $; जहाँ तक सीमा अभिन्नों का प्रश्न है, द
की वामावर्त सीमाएँ $ K_1$ और $ K_2$ प्रत्येक यात्रा
आंतरिक भाग को एक बार अंदर काटा गया \emph{विलोम} दिशा-निर्देश, इत्यादि
योग
\[
\oint_{\partial K_1} + \oint_{\partial K_2}
= \oint_{\partial K} + (\text{cut, both ways})
= \oint_{\partial K},
\]
कट के साथ दो पास रद्द हो जाते हैं
(\cref{prop:b2:multint:lineinv}) और केवल बाहरी सीमा
जीवित रहता है. बहुत से कटों को बार-बार दोहराते हुए, यह इलाज के लिए पर्याप्त है
एक प्राथमिक डोमेन. हम साबित करते हैं $\oint P\,\dd x = -\iint_K P_y $ एक पर
$ y $-प्राथमिक डोमेन $ D = \{a \leq x \leq b,\ \varphi_1(x) \leq y
\leq \varphi_2(x)\}$; पहचान $\oint Q\,\dd y = \iint_K Q_x $
है \omterm{def:b2:quadratic:adjoint}{सममित} ($ x $-प्राथमिक), और प्रमेय उनका योग है।

फ़ुबिनी और एक-चर द्वारा दोहरे अभिन्न अंग की गणना करें
मौलिक प्रमेय:
\[
\iint_D \frac{\partial P}{\partial y}\,\dd x\,\dd y
= \int_a^b \bigl(P(x, \varphi_2(x)) - P(x, \varphi_1(x))\bigr)
\,\dd x .
\]
अब की सीमा $ D $, वामावर्त, इसमें शामिल हैं: निचला
ग्राफ $ y = \varphi_1(x)$ बाएँ से दाएँ, दाएँ घुमाया गया
ऊर्ध्वाधर खंड $ x = b $ (ऊपर की ओर), ऊपरी ग्राफ़ $ y =
\varphi_2(x)$ चल \emph{दाएं से बाएं}, बायां लंबवत
खंड $ x = a $ (नीचे की ओर)। ऊर्ध्वाधर खंडों के साथ $ x $ है
स्थिर, इसलिए वे योगदान करते हैं $0$ को $\oint P\,\dd x $; रेखांकन,
द्वारा पैरामीट्रिज्ड $ x $, देना
\[
\oint_{\partial D} P\,\dd x
= \int_a^b P(x, \varphi_1(x))\,\dd x
- \int_a^b P(x, \varphi_2(x))\,\dd x
= -\iint_D \frac{\partial P}{\partial y}\,\dd x\,\dd y .
\qedhere
\]
\end{proof}

\begin{corollary}[सीमा द्वारा क्षेत्र]\label{cor:b2:multint:area}
की परिकल्पना के तहत \cref{thm:b2:multint:green},
\[
\operatorname{Area}(K)
= \oint_{\partial K} x\,\dd y
= -\oint_{\partial K} y\,\dd x
= \frac12\oint_{\partial K} x\,\dd y - y\,\dd x .
\]
\end{corollary}

\begin{proof}
ग्रीन--रिमैन को लागू करें $(P, Q) = (0, x)$, $(-y, 0)$ और
$\frac12(-y, x)$: हर बार $ Q_x - P_y = 1$.
\end{proof}

\begin{remark}[तीन क्षेत्र सूत्रों में से चयन करना]
तीन सीमा सूत्र समान हैं, विनिमेय नहीं हैं
अभ्यास. उपयोग $\oint x\,\dd y $ जब पैरामीट्रिजेशन बनता है
$\dd y $ सरल (ग्राफ़ के ऊपर $ y $-अक्ष), $-\oint y\,\dd x $
\omterm{def:b2:quadratic:adjoint}{सममित} रूप से, और \omterm{def:b2:quadratic:adjoint}{सममित} आधी राशि जब
पैरामीट्रिजेशन व्यवहार करता है $ x $ और $ y $ \omterm{def:b2:funcseq:def}{समान रूप से} --- दीर्घवृत्त के लिए
इसने बिना किसी त्रिकोणमिति के, एक स्थिर समाकलन उत्पन्न किया
\omterm{def:b2:series:def}{बिल्कुल} रैखिकरण। बहुभुज सीमाओं पर आधा योग
जूते के फीते का फार्मूला बन जाता है \cref{exo:b2:multint:12},
सर्वेक्षकों का एल्गोरिदम. और जब सीमा पार हो जाती है
दिए गए पैरामीट्रिजेशन द्वारा, सभी तीन सूत्रों को दक्षिणावर्त
वापसी \emph{ऋण}  \omterm{def:b2:multint:domain}{क्षेत्र}: एक नकारात्मक परिणाम नहीं है
कम्प्यूटेशनल त्रुटि लेकिन एक ओरिएंटेशन रिपोर्ट --- पलटें
संकेत, या पैरामीट्रिज़ेशन।
\end{remark}

\begin{example}[दीर्घवृत्त का क्षेत्रफल]\label{ex:b2:multint:ellipsearea}
के लिए $ x = a\cos t $, $ y = b\sin t $, $ t \in [0, 2\pi]$:
\[
\operatorname{Area}
= \frac12\int_0^{2\pi}\bigl(a\cos t \cdot b\cos t
- b\sin t\cdot(-a\sin t)\bigr)\,\dd t
= \frac{ab}{2}\int_0^{2\pi}\dd t = \pi ab .
\]
\end{example}

\begin{example}[ग्रीन - रीमैन एक के रूप में
क्रॉस-चेक करें]\label{ex:b2:multint:greencheck}
लेना $ P = -y^3$, $ Q = x^3$ बंद इकाई डिस्क पर $ D $.
सीमा की ओर, साथ $\gamma(t) = (\cos t, \sin t)$:
\[
\oint_{\partial D}P\,\dd x + Q\,\dd y
= \int_0^{2\pi}\bigl(\sin^4 t + \cos^4 t\bigr)\dd t
= 2\pi\cdot\Bigl(\frac38 + \frac38\Bigr) = \frac{3\pi}2 ,
\]
रैखिककरण द्वारा ($\sin^4 + \cos^4 = \tfrac34 +
\tfrac14\cos4t $). आंतरिक पक्ष:
\[
\iint_D(Q_x - P_y)\,\dd x\,\dd y
= \iint_D 3(x^2 + y^2)\,\dd x\,\dd y
= 3\int_0^{2\pi}\!\!\int_0^1\rho^3\,\dd\rho\,\dd\alpha
= \frac{3\pi}2 .
\]
एक ही संख्या, दो \omterm{def:b2:series:def}{बिल्कुल} भिन्न गणनाएँ --- और वह यह है
व्यावहारिक उपयोग: ग्रीन की पहचान का जो भी पक्ष हो
गणना आसान हो जाती है, दूसरा सत्यापन। के लिए
बंद वक्रों के चारों ओर बहुपद \omterm{def:b2:multint:domain}{क्षेत्र}ों का परिचलन
डबल इंटीग्रल लगभग हमेशा आसान पक्ष होता है।
\end{example}

\begin{remark}
ग्रीन - रीमैन बताते हैं \cref{ex:b2:multint:angleform}: एक के लिए
बंद प्रपत्र ($ Q_x = P_y $), किसी की सीमा के चारों ओर अभिन्न
डोमेन में निहित है $ U $ गायब हो जाता है. कोण प्रपत्र होने में विफल रहता है \omterm{def:b2:multint:exact}{एकदम सही}
केवल इसलिए क्योंकि मूल में पंचर डिस्क को सीमित होने से रोकता है
यूनिट सर्कल द्वारा अंदर लेटने से $ U $ --- \omterm{def:b2:multint:lineint}{लाइन इंटीग्रल} का
बंद प्रपत्र डोमेन के छिद्रों का पता लगाएं। (आगे बढ़ाया,
यह अवलोकन डी राम कोहोमोलॉजी बन जाता है।)
\end{remark}

\section{त्रिगुण अभिन्न}

सिद्धांत बिना किसी नए विचार के तीन चरों तक फैला हुआ है: फ़ुबिनी
कम कर देता है $\iiint $ तीन एक-चर इंटीग्रल्स के लिए (या तो द्वारा
\emph{टुकड़ा करने की क्रिया}: $\iiint_K f = \int\bigl(\iint_{K_z}
f\bigr)\dd z $ क्षैतिज स्लाइस के ऊपर $ K_z $, या द्वारा
\emph{स्टैकिंग}: में एकीकृत करना $ z $ पहले ऊर्ध्वाधर छड़ियों के साथ),
और चर परिवर्तन सूत्र के साथ रहता है $3 \times 3$
जैकोबियन.

\begin{example}[बेलनाकार और गोलाकार निर्देशांक]\label{ex:b2:multint:spherical}
\emph{बेलनाकार} $(x, y, z) = (\rho\cos\alpha, \rho\sin\alpha,
z)$: $\dd x\,\dd y\,\dd z = \rho\,\dd\rho\,\dd\alpha\,\dd z $.
\emph{गोलाकार} $(x, y, z) = (r\cos\theta\cos\varphi,\
r\sin\theta\cos\varphi,\ r\sin\varphi)$ ($\theta $ देशांतर,
$\varphi \in [-\frac\pi2, \frac\pi2]$ अक्षांश): का विस्तार $3
\times 3$ सिद्ध आखिरी पंक्ति के साथ,
\[
\det J = r^2\cos\varphi ,
\qquad
\dd x\,\dd y\,\dd z
= r^2\cos\varphi\;\dd r\,\dd\theta\,\dd\varphi .
\]
त्रिज्या की गेंद का आयतन $ R $:
\[
V = \int_0^R\!\!\int_0^{2\pi}\!\!\int_{-\pi/2}^{\pi/2}
r^2\cos\varphi\;\dd\varphi\,\dd\theta\,\dd r
= \frac{R^3}{3}\cdot 2\pi \cdot 2
= \boxed{\frac43\pi R^3} ,
\]
के खंड अध्यायों में स्वीकृत सूत्र का अंत में निर्वहन
पहले की किताबें.
\end{example}

\begin{example}[चतुष्फलक, दो बार]
\label{ex:b2:multint:tetra}
की मात्रा $ T = \{x, y, z \geq 0,\ x + y + z \leq 1\}$, द्वारा
स्टैकिंग: निश्चित के लिए $(x, y)$ त्रिकोण में $ x + y \leq 1$,
$ z $ पर चलाता है $\intcc0{1 - x - y}$, इसलिए
\[
V = \int_0^1\!\!\int_0^{1-x}(1 - x - y)\,\dd y\,\dd x
= \int_0^1\frac{(1 - x)^2}{2}\,\dd x = \frac16 .
\]
टुकड़ा करके: ऊंचाई पर अनुभाग $ z $ त्रिकोण है $\{x, y
\geq 0,\ x + y \leq 1 - z\}$, का \omterm{def:b2:multint:domain}{क्षेत्र} $\frac{(1-z)^2}2$, और
$ V = \int_0^1\frac{(1-z)^2}2\,\dd z = \frac16$ फिर से ---
दो संगणनाएँ भिन्न में समान अभिन्न अंग हैं
\omterm{def:b2:structures:generated}{आदेश}, जो फ़ुबिनी के सभी दावे हैं। मूल्य $\frac16 =
\frac13\cdot\frac12\cdot1$ शंकु सूत्र है
(\cref{ex:b2:multint:cone}) \omterm{def:b2:reduction:diag}{त्रिकोणीय} आधार के साथ, और
$ n $-आयामी संस्करण $1/n!$ \omterm{def:b2:series:def}{बिल्कुल} इसी से सिद्ध होता है
सप्ताहांत की समस्या में कटौती।
\end{example}

\begin{example}[आधी गेंद का केन्द्रक]\label{ex:b2:multint:centroid}
ऊपरी आधी गेंद के लिए $ H $ त्रिज्या का $ R $ ($ z \geq 0$), द
\omterm{def:b2:affine:barycenter}{केन्द्रक} ऊंचाई है $\bar z = \frac1{V}\iiint_H z $, साथ $ V =
\frac23\pi R^3$. गोलाकार निर्देशांक में ($ z =
r\sin\varphi $, $\varphi \in \intcc0{\pi/2}$):
\[
\iiint_H z
= \int_0^R r^3\,\dd r\int_0^{2\pi}\dd\theta
\int_0^{\pi/2}\sin\varphi\cos\varphi\,\dd\varphi
= \frac{R^4}4\cdot2\pi\cdot\frac12 = \frac{\pi R^4}4 ,
\]
इसलिए
\[
\bar z = \frac{\pi R^4/4}{2\pi R^3/3} = \frac{3R}8 :
\]
एक ठोस गोलार्ध का संतुलन बिंदु तीन-आठवां बैठता है
सपाट सतह के ऊपर की त्रिज्या --- आधी ऊंचाई के नीचे
$ R/2$, जैसा कि होना ही चाहिए, क्योंकि आधार के पास ठोस मोटा होता है।
प्रत्येक \omterm{def:b2:affine:barycenter}{केन्द्रक} गणना का यह आकार होता है: एक क्षण
अभिन्न, एक मात्रा, एक अनुपात, और एक संभाव्यता जांच
ज्यामिति के विरुद्ध.
\end{example}

\begin{example}[टुकड़ा करके आयतन: शंकु]\label{ex:b2:multint:cone}
आधार का एक शंकु \omterm{def:b2:multint:domain}{क्षेत्र} $ A $ और ऊंचाई $ h $ (शीर्ष ऊपर, आधार पर $ z = 0$):
ऊंचाई पर टुकड़ा $ z $ कारक द्वारा मापा गया आधार है $(1 -
z/h)$, का \omterm{def:b2:multint:domain}{क्षेत्र} $ A(1 - z/h)^2$. इस तरह
\[
V = \int_0^h A\Bigl(1 - \frac zh\Bigr)^2\dd z = \frac{Ah}{3} :
\]
स्कूल के एक-तिहाई फ़ॉर्मूले, के लिए मान्य हैं \emph{कोई} आधार
आकार --- टुकड़ा करने से यह एक वर्ग के अभिन्न अंग में बदल जाता है।
\end{example}

\begin{example}[इंटीग्रेबिलिटी थ्रेशोल्ड में
विमान]\label{ex:b2:multint:threshold}
जिसके लिए $\alpha > 0$ करता है $\iint_{D}\rho^{-\alpha}\,\dd
x\,\dd y $ पंचर यूनिट डिस्क पर एकत्रित हों $ D $ (सीमा
वार्षिकी के ऊपर $\varepsilon \leq \rho \leq 1$)? ध्रुवीय में
निर्देशांक,
\[
\int_0^{2\pi}\!\!\int_\varepsilon^1\rho^{-\alpha}\,
\rho\,\dd\rho\,\dd\alpha
= 2\pi\int_\varepsilon^1\rho^{1-\alpha}\,\dd\rho ,
\]
जो के रूप में \omterm{def:b2:integration:improper}{अभिसरण} होता है $\varepsilon \to 0$ आईएफएफ $1 - \alpha > -1$,
यानी\ $\alpha < 2$: आयाम में $2$ आलोचनात्मक विलक्षणता
प्रतिपादक आयाम ही है, अतिरिक्त $\rho $ से
जैकोबियन एक शक्ति द्वारा विलक्षणता को नरम कर रहा है। (वैसे ही
$\alpha < 3$ \omterm{def:b2:diffcalc:differential}{अंतर}िक्ष में एक बिंदु विलक्षणता के लिए, के माध्यम से $ r^2$.)
इस प्रकार की रेडियल बहीखाता पद्धति कितनी अभिन्नता है
वर्ष 3 के लेबेस्ग ढांचे में एक नज़र में निर्णय लिया गया ---
और यही कारण है $\iiint 1/r $ सहजता से एकाग्र हो गया
\cref{exo:b2:multint:7}.
\end{example}

\begin{remark}[सामान्य ख़तरे]
(मैं) \emph{अभिविन्यास}: ए \omterm{def:b2:multint:lineint}{लाइन इंटीग्रल} के साथ चिन्ह बदलता है
यात्रा की दिशा, और हरा--रीमैन को सीमा की आवश्यकता है
वामावर्त (बाईं ओर डोमेन); a वाले डोमेन के लिए
छेद, भीतरी सीमा पार हो गई है \emph{दक्षिणावर्त}. (ii)
\emph{जैकोबियन पूर्ण मूल्य के साथ प्रवेश करता है}: का परिवर्तन
चर कभी भी नकारात्मक उत्पन्न नहीं करते \omterm{def:b2:multint:domain}{क्षेत्र}, और भूल जाना
$\abs{\det}$ आम तौर पर जब नक्शा ठीक होता है तो संकेत फ़्लिप हो जाते हैं
अभिविन्यास उलट देता है। (iii) \emph{ध्रुवीय कारक $\rho $}:
$\dd x\,\dd y = \rho\,\dd\rho\,\dd\alpha $, नहीं
$\dd\rho\,\dd\alpha $ --- पूरी तरह से सबसे आम त्रुटि
अध्याय; चूँकि, आयामी विश्लेषण इसे पकड़ता है
$\dd\rho\,\dd\alpha $ का आयाम है \omterm{def:b2:curves:length}{लंबाई}, एक नहीं
\omterm{def:b2:multint:domain}{क्षेत्र}. (iv) \emph{अनुचित दोहरा इंटीग्रल}: सीमा खत्म
बढ़ती हुई डिस्क और बढ़ते हुए वर्ग यहाँ सहमत हैं क्योंकि
इंटीग्रैंड्स सकारात्मक हैं (निचोड़ रहे हैं); साइन-चेंजिंग के लिए
एकीकृत सीमा थकावट पर निर्भर हो सकती है, और नहीं
दावा पूर्ण \omterm{def:b2:integration:improper}{अभिसरण} के बिना किया गया है। (वी)
\emph{डोमेन बनाम इंटीग्रैंड्स}: एक उत्पाद एक पर एकीकृत
गैर-उत्पाद डोमेन करता है \emph{नहीं} अभिन्न कारक ---
गुणनखंडन के लिए दोनों की आवश्यकता होती है, जैसे कि \cref{ex:b2:multint:polar}का
वर्ग.
\end{remark}

\begin{remark}[इस खंड के भीतर परिप्रेक्ष्य]
यहां गणना किया गया गॉस इंटीग्रल हर जगह चुपचाप मौजूद है
संभाव्यता अध्याय: स्थिरांक $\sqrt\pi $ अंदर
स्टर्लिंग का सूत्र (\cref{thm:b2:comparison:stirling}) है
यह अध्याय अभिन्न है, और स्टर्लिंग के माध्यम से इसे ठीक करता है
$1/\sqrt{\pi n}$ रैंडम वॉक की वापसी के स्पर्शोन्मुखता
संभावनाओं में \cref{ch:b2:proba}. वालिस इंटीग्रल्स का
सप्ताहांत की समस्या वहाँ भी फिर से प्रकट होती है, वही ड्राइविंग
केंद्रीय-द्विपद अनुमान. दूसरी दिशा में, \omterm{def:b2:multint:domain}{क्षेत्र}
और इस अध्याय के आयतन तत्व की ज्यामिति को \omterm{def:b2:metric:complete}{पूरा} करते हैं
\cref{ch:b2:surfaces}, और ग्रीन का सूत्र पुनःगणना करता है
\omterm{pb:b2:curves:1}{लिफ़ाफ़ा} \omterm{def:b2:multint:domain}{क्षेत्र}ों का \cref{ch:b2:curves} (द \omterm{pb:b2:curves:1}{astroid}, में
\cref{exo:b2:multint:5}). एक अध्याय, तीन सेवाएँ:
ज्यामिति के लिए माप, संभाव्यता के लिए स्थिरांक, और
परिवर्तन-परिवर्तन अनुशासन दोनों द्वारा उपयोग किया जाता है।
\end{remark}

गणना करना $\int_\gamma y^2\,\dd x + x\,\dd y $ साथ में: (ए) खंड
से $(0,0)$ को $(1,1)$; (बी) परवलय चाप $ y = x^2$ से

\begin{exercise}[$\star $]\label{exo:b2:multint:2}
बताते हैं कि $\omega = (2xy + y^3)\,\dd x + (x^2 + 3xy^2 + 1)\,\dd y $
पर बंद है $\R^2$, लगता है \omterm{def:b2:multint:exact}{संभावना}, और गणना करें
$\int_\gamma\omega $ से किसी भी चाप के साथ $(0, 0)$ को $(1, 2)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:multint:3}
गणना करना $\iint_D (x + y)\,\dd x\,\dd y $ कहाँ $ D $ डोमेन है
से घिरा हुआ $ y = x^2$ और $ y = x $ ($0 \leq x \leq 1$), दोनों में
एकीकरण के \omterm{def:b2:structures:generated}{आदेश}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:multint:4}
ध्रुवीय निर्देशांक का उपयोग करके गणना करें $\iint_D \frac{\dd x\,\dd y}{(1 +
x^2 + y^2)^2}$ पूरे तल पर (डिस्क पर एक सीमा के रूप में), और
$\iint_{D'} xy\,\dd x\,\dd y $ क्वार्टर डिस्क पर $ D' = \{x, y
\geq 0,\ x^2 + y^2 \leq 1\}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:multint:5}
की गणना करें \omterm{def:b2:multint:domain}{क्षेत्र} द्वारा संलग्न \omterm{pb:b2:curves:1}{astroid} $ x = \cos^3 t $, $ y =
\sin^3 t $, $ t \in [0, 2\pi]$, का उपयोग करना
\cref{cor:b2:multint:area}. \emph{(रैखिकीकरण करें $\sin^2 t\cos^2 t $.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:multint:6}
नीचे पैराबोलॉइड से घिरे ठोस के आयतन की गणना करें $ z
= x^2 + y^2$ और ऊपर हवाई जहाज़ से $ z = 1$, दोनों तरीकों से:
स्टैकिंग (एकीकृत) $1 - x^2 - y^2$ यूनिट डिस्क पर, ध्रुवीय
निर्देशांक) और स्लाइसिंग (क्षैतिज स्लाइस त्रिज्या की डिस्क हैं
$\sqrt z $).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:multint:7}
(गेंद का गुरुत्वाकर्षण आकर्षण --- न्यूटन का प्रमेय, विशेष
केस) दिखाएँ कि गोलाकार खोल का आयतन $ a \leq r \leq
b $ है $\frac43\pi(b^3 - a^3)$ और गणना करें $\iiint_{B}
\frac{\dd x\,\dd y\,\dd z}{r}$ गेंद के ऊपर $ B $ त्रिज्या का $ R $
($ r $ उद्गम स्थल से दूरी) \emph{(गोलाकार
निर्देशांक.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:multint:8}
(पोंकारे की लेम्मा, तारे के आकार का मामला) चलो $ U $ तारे के आकार का हो
के प्रति सम्मान $0$ (अर्थात्\ $ M \in U \Rightarrow [0, M] \subseteq U $)
और $\omega = P\dd x + Q\dd y $ एक बंद $\mathcal{C}^1$ पर प्रपत्र
$ U $. परिभाषित करना
\[
f(x, y) = \int_0^1 \bigl(x\,P(tx, ty) + y\,Q(tx, ty)\bigr)\dd t .
\]
अभिन्न चिह्न के \omterm{def:b2:diffcalc:differential}{अंतर}्गत विभेदन का उपयोग करना
(\cref{ch:b2:integration}) और $ P_y = Q_x $, बताते हैं कि $ f_x = P $
और $ f_y = Q $: प्रत्येक बंद प्रपत्र एक तारे के आकार पर \omterm{def:b2:metric:topology}{खुला} सेट है
\omterm{def:b2:multint:exact}{एकदम सही}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:multint:9}
(डबल इंटीग्रेशन द्वारा डिरिचलेट इंटीग्रल) औचित्य सिद्ध करें और शोषण करें
\[
\int_0^\infty\!\!\int_0^\infty e^{-xy}\sin x\;\dd y\,\dd x
\quad\text{vs}\quad
\int_0^\infty\!\!\int_0^\infty e^{-xy}\sin x\;\dd x\,\dd y
\]
पर $[0, A] \times [0, \infty)$: दिखाओ $\int_0^A \frac{\sin
x}{x}\dd x = \frac\pi2 - \int_0^\infty
e^{-Ay}\frac{y\sin A + \cos A}{1 + y^2}\dd y $ और ठीक हो जाओ
$\int_0^\infty \frac{\sin x}{x}\,\dd x = \frac\pi2$, तुलना करना
के पैरामीटर-अभिन्न प्रमाण के साथ
\cref{ch:b2:integration}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:multint:10}
(विर्टिंगर के माध्यम से आइसोपेरिमेट्रिक असमानता) आइए $\gamma $ सरल बनो
बंद $\mathcal{C}^1$ का वक्र \omterm{def:b2:curves:length}{लंबाई} $2\pi $, द्वारा पैरामीट्रिज्ड वक्राकार \omterm{def:b2:curves:length}{लंबाई} पर $[0, 2\pi]$, घेरना \omterm{def:b2:multint:domain}{क्षेत्र} $ A $. का उपयोग करते हुए
\cref{cor:b2:multint:area}, पार्सेवल और विर्टिंगर असमानता
(के व्यायाम \cref{ch:b2:fourier}), सिद्ध करना $ A \leq \pi $, के साथ
वृत्त के लिए समानता. \emph{(सामान्यीकृत करें $\int_0^{2\pi} x(s)\dd s
= 0$; लिखना $2A = \oint x\,\dd y - y\,\dd x $ और बाध्य $2A \leq
\int (x^2 + y'^2)$ के माध्यम से ध्यान से $2A = \int_0^{2\pi}(xy' - yx')\dd
s $ और $ x^2 + y'^2 \geq 2xy'$.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:multint:11}
(गेंद के क्षण) गेंद के लिए $ B $ त्रिज्या का $ R $ में
$\R^3$, गणना करें $\iiint_B z^2\,\dd x\,\dd y\,\dd z $ में
गोलाकार निर्देशांक, और निष्कर्ष निकालें $\iiint_B (x^2 + y^2 +
z^2)\,\dd x\,\dd y\,\dd z $ समरूपता द्वारा. क्रॉस-चेक करें
शेल गणना के विरुद्ध उत्तरार्द्ध $\int_0^R r^2\cdot4\pi
r^2\,\dd r $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:multint:12}
(जूते का फीता सूत्र) चलो $ K $ शीर्षों वाला बहुभुज बनें $(x_1,
y_1), \dots, (x_m, y_m)$ वामावर्त क्रम में (सूचकांक)
मॉड $ m $). से निष्कर्ष निकालें \cref{cor:b2:multint:area} वह
\[
\operatorname{Area}(K)
= \frac12\sum_{i=1}^m
\bigl(x_iy_{i+1} - x_{i+1}y_i\bigr) ,
\]
और त्रिभुज पर सूत्र की जाँच करें $(0,0)$, $(1,0)$,
$(0,1)$.
\end{exercise}

\section{समस्या: आयाम में गेंद का आयतन $ n $}

\begin{problem}[{सप्ताहांत समस्या --- $ V_n =
\pi^{n/2}/\Gamma(\frac n2 + 1)$, और उच्च की विचित्रता
आयाम}]\label{pb:b2:multint:1}
डिस्क है \omterm{def:b2:multint:domain}{क्षेत्र} $\pi $, द \omterm{pb:b2:multint:1}{गेंद की मात्रा} $\frac43\pi $ --- और तब? यह समस्या यूनिट बॉल की मात्रा की गणना करती है
$\R^n $ हरएक के लिए $ n $, दो बार (एक स्लाइसिंग रिकर्सन द्वारा संचालित
द वालिस इंटीग्रल्स, फिर के माध्यम से $\Gamma $ फ़ंक्शन और
गॉस का अभिन्न अंग \cref{ex:b2:multint:polar}), और फिर पढ़ता है
ज्यामिति से बाहर: वॉल्यूम पांच आयाम में चरम पर है और तेजी से बढ़ता है
शून्य तक, और लगभग सभी उच्च-आयामी गेंद एक में छिप जाती है
इसकी सीमा के पास पतला खोल।\index{गेंद की मात्रा}\index{वालिस
अभिन्न} एक के लिए \omterm{def:b2:metric:continuity}{निरंतर} की एक गेंद पर कार्य करें $\R^n $, द
अभिन्न को समझा जाता है $ n $-गुना पुनरावृत्त अभिन्न
(जैसा कि अध्याय में बताया गया है, एक समय में एक निर्देशांक को काटना $ n
\leq 3$); हम लिखते हैं $ B_n(R)$ त्रिज्या की बंद गेंद के लिए $ R $
पर केन्द्रित $0$, $ v_n(R)$ इसकी मात्रा के लिए, और $ V_n =
v_n(1)$, साथ $ V_0 = 1$ रिवाज के सन्दर्भ मे।

\medskip
\noindent\textbf{Part I --- The slicing recursion.}
\begin{enumerate}
  \item स्थानापन्न $ x_i = Ru_i $ प्रत्येक में $ n $ पुनरावृत्त
        अभिन्न, दिखाओ $ v_n(R) = V_nR^n $.
  \item टुकड़ा करने की क्रिया $ B_n(1)$ इसके अंतिम निर्देशांक के साथ दिखाएँ
        \[
        V_n = V_{n-1}\int_{-1}^{1}(1 -
        t^2)^{\frac{n-1}2}\,\dd t .
        \]
  \item साथ $ t = \sin\theta $, अभिन्न को एक के रूप में पहचानें
        वालिस इंटीग्रल: $\int_{-1}^1(1 -
        t^2)^{\frac{n-1}2}\dd t = 2W_n $, कहाँ $ W_n =
        \int_0^{\pi/2}\cos^n\theta\,\dd\theta =
        \int_0^{\pi/2}\sin^n\theta\,\dd\theta $.
  \item दो वालिस सर्वसमिकाओं को सिद्ध करें (भागों द्वारा एकीकृत करें;
        फिर दूरबीन $ nW_nW_{n-1}$):
        \[
        W_n = \frac{n-1}nW_{n-2} \quad (n \geq 2),
        \qquad
        W_nW_{n-1} = \frac{\pi}{2n} \quad (n \geq 1).
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The recursion solved.}
\begin{enumerate}[resume]
  \item प्रश्न 2-4 को दो-चरणीय पुनरावृत्ति में संयोजित करें
        \[
        V_n = \frac{2\pi}{n}\,V_{n-2}
        \qquad (n \geq 2).
        \]
  \item निष्कर्ष निकालें बंद प्रपत्र, के लिए $ k \geq 0$:
        \[
        V_{2k} = \frac{\pi^k}{k!},
        \qquad
        V_{2k+1} = \frac{2^{k+1}\pi^k}{1\cdot3\cdot5\cdots
        (2k+1)} .
        \]
  \item समतल $ V_1, \dots, V_7$ संख्यात्मक रूप से. का उपयोग करना
        अनुपात $ V_n/V_{n-2} = 2\pi/n $ और के मूल्य $2W_5$
        और $2W_6$, सिद्ध कीजिए कि अनुक्रम $(V_n)$ बढ़ता है
        इसकी अधिकतम सीमा तक $ V_5 = \frac{8\pi^2}{15} \approx
        5.26$ और तब से घटता जाता है।
  \item बताते हैं कि $ V_n \to 0$ किसी भी ज्यामितीय से तेज़
        अनुक्रम, और वह $\sum_{n\geq1} V_n $ \omterm{def:b2:integration:improper}{अभिसरण}: सब
        इकाई गेंदों का कुल आयतन सीमित होता है।
  \item उत्पन्न करने वाली पहचान सिद्ध करें
        \[
        \sum_{k\geq0} V_{2k}\,x^{2k} = \eu^{\pi x^2}
        \qquad (x \in \R),
        \]
        और निष्कर्ष निकालें $\sum_{k\geq0}V_{2k} = \eu^\pi \approx
        23.14$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Second route: $\Gamma $ and the
Gauss integral.}
\begin{enumerate}[resume]
  \item फ़ुबिनी द्वारा दिखाएँ (इंटीग्रैंड एक उत्पाद है) वह
        \[
        I_n = \int_{\R^n}\eu^{-\norm
        x^2}\dd x
        = \Bigl(\int_{-\infty}^{+\infty}
        \eu^{-t^2}\dd t\Bigr)^{\!n} = \pi^{n/2},
        \]
         $ n $-आयामी गॉस इंटीग्रल, के रूप में समझा जाता है
        घनों पर सीमा $\intcc{-R}{R}^n $.
  \item याद करना $\Gamma(s) = \int_0^\infty t^{s-1}\eu^{-t}\dd
        t $ (\cref{def:b2:integration:gamma}). से
        $\Gamma(s+1) = s\,\Gamma(s)$
        (\cref{thm:b2:integration:gammaprops}) और
        $\Gamma(\tfrac12) = \sqrt\pi $ (विकल्प $ t = u^2$
        और गॉस इंटीग्रल का आह्वान करें), गणना करें
        \[
        \Gamma(k + 1) = k!,
        \qquad
        \Gamma\Bigl(k + \frac32\Bigr) =
        \frac{1\cdot3\cdots(2k+1)}{2^{k+1}}\,\sqrt\pi .
        \]
  \item प्रश्न की पुनरावृत्ति के माध्यम से प्रेरण द्वारा सिद्ध करें
        5, एकल सूत्र
        \[
        V_n = \frac{\pi^{n/2}}{\Gamma\bigl(\frac n2 +
        1\bigr)} \qquad (n \geq 1),
        \]
        और जांचें कि यह दोनों को पुन: उत्पन्न करता है बंद प्रपत्र प्रश्न का
        6.
  \item दिखाओ $\int_0^\infty \eu^{-r^2}r^{n-1}\dd r =
        \tfrac12\Gamma\bigl(\tfrac n2\bigr)$ और निष्कर्ष निकालें
        पहचान
        \[
        I_n = n\,V_n\int_0^\infty \eu^{-r^2}\,r^{n-1}\,\dd r.
        \]
        इसकी व्याख्या करें: का गाऊसी द्रव्यमान $\R^n $ है
        गोलाकार कोशों के साथ एकत्र किया गया जिसका
        ``$(n-1)$ आयामी \omterm{def:b2:multint:domain}{क्षेत्र}'' त्रिज्या पर $r $ है
        $ nV_nr^{n-1}$ ---दोनों पक्ष अब साबित हो गए हैं
        स्वतंत्र रूप से, इसलिए व्याख्या की कोई लागत नहीं है।
  \item तय करना $ s_{n-1} = nV_n $ (द \omterm{def:b2:multint:domain}{क्षेत्र} इकाई \omterm{def:b2:multint:domain}{क्षेत्र} का
        $ S^{n-1}$, लगातार साथ $ v_n(R) = \int_0^R
        s_{n-1}r^{n-1}\dd r $). समतल $ s_0, \dots, s_3$ और
        जांचें $ s_1 = 2\pi $, $ s_2 = 4\pi $, $ s_3 = 2\pi^2$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- High dimensions are strange.}
\begin{enumerate}[resume]
  \item स्टर्लिंग के सूत्र से
        (\cref{thm:b2:comparison:stirling}) के लिए आवेदन किया $ k!$,
        सम के लिए दिखाओ $ n = 2k $:
        \[
        V_n \sim \frac{1}{\sqrt{\pi n}}
        \Bigl(\frac{2\pi\eu}{n}\Bigr)^{n/2}
        \qquad (n \to \infty, \ n \text{ even}),
        \]
        और समझाएं कि समान सुपर-ज्यामितीय क्षय क्यों बाध्य है
        विषम तक फैला हुआ है $ n $ प्रत्यावर्तन के माध्यम से.
  \item यूनिट बॉल क्यूब में बैठती है $\intcc{-1}1^n $ का
        आयतन $2^n $. भरण अनुपात की गणना करें $ V_n/2^n $ के लिए
        $ n = 2, 3, 10$, और दिखाओ कि ऐसा होता है $0$: उच्च में
        आयाम, मूलतः सारा घन इसी में निहित है
        कोने.
  \item दिखाएँ कि का अंश $ v_n(1)$ भीतर पड़ा हुआ
        दूरी $\varepsilon $ सीमा \omterm{def:b2:multint:domain}{क्षेत्र} का है $1 -
        (1 - \varepsilon)^n \to 1$; संख्यात्मक रूप से, क्या
        ए का अंश $100$-आयामी गेंद में निहित है
        मोटाई का बाहरी आवरण $1\%$?
  \item वालिस को स्पर्शोन्मुख सिद्ध करें $ W_n \sim
        \sqrt{\dfrac{\pi}{2n}}$ \emph{(की नीरसता
        $(W_n)$, अनुपात $ W_n/W_{n-2} \to 1$, और $ W_nW_{n-1}
        = \frac\pi{2n}$)}, और निचली सीमा $ W_n \geq
        \sqrt{\dfrac{\pi}{2(n+1)}}$ सभी के लिए $ n $.
  \item (एक स्लैब पर एकाग्रता) इकाई का अंश
        पहले निर्देशांक से परे गेंद $\delta $ है
        $\int_\delta^1(1 - x^2)^{\frac{n-1}2}\dd x \,\big/\,
        (2W_n)$. का उपयोग करते हुए $1 - u \leq \eu^{-u}$ और पूँछ
        बाध्य $\int_\delta^\infty \eu^{-a x^2}\dd x \leq
        \frac{\eu^{-a\delta^2}}{2a\delta}$, ये दिखाओ
        अंश अधिकतम है
        \[
        \frac{\eu^{-(n-1)\delta^2/2}}{(n-1)\,\delta}
        \Big/ \sqrt{\frac{2\pi}{n+1}}
        \]
        और निष्कर्ष निकालें: के लिए $\delta = s/\sqrt{n-1}$, ए को छोड़कर सभी
        अंश $ O(\eu^{-s^2/2}/s)$ गेंद के अंदर निहित है
        स्लैब $\abs{x_1} \leq s/\sqrt{n-1}$. ऊंचाई की एक गेंद
        आयाम, सांख्यिकीय रूप से, प्रत्येक में एक पतला पैनकेक है
        एक ही बार में दिशा.
  \item प्रश्न 16--19 को एक पैराग्राफ में एकत्रित करें: कहाँ
        की मात्रा $ B_n(1)$ बैठता है (सीमा के पास)।
        \omterm{def:b2:multint:domain}{क्षेत्र}, फिर भी भीतर $ O(1/\sqrt n)$ प्रत्येक के स्लैब
        केंद्र के माध्यम से हाइपरप्लेन), और ये दोनों क्यों
        कथन एक दूसरे का खंडन नहीं करते.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Other bodies, and synthesis.}
\begin{enumerate}[resume]
  \item (सिंप्लेक्स) चलो $\Delta_n = \{x \in \R^n : x_i \geq 0,\
        \sum x_i \leq 1\}$. स्लाइसिंग और इंडक्शन करके दिखाओ
        $\operatorname{vol}(\Delta_n) = \frac1{n!}$.
  \item (क्रॉस-पॉलीटोप) उसका निष्कर्ष निकालें $ C_n = \{x :
        \sum\abs{x_i} \leq 1\}$ मात्रा है $\frac{2^n}{n!}$,
        और सैंडविच को सत्यापित करें $ C_n \subseteq B_n(1)
        \subseteq \intcc{-1}1^n $ मात्रा के स्तर पर:
        $\frac{2^n}{n!} \leq V_n \leq 2^n $.
  \item गणना करना $ V_4$ तीसरा तरीका: टुकड़ा $\R^4 = \R^2 \times
        \R^2$, एकीकृत करें \omterm{def:b2:multint:domain}{क्षेत्र} की $(z, w)$-डिस्क खत्म
        द $(x, y)$-ध्रुवीय निर्देशांक में डिस्क, और पुनर्प्राप्त करें
        $ V_4 = \frac{\pi^2}2$.
  \item (मोंटे कार्लो मुसीबत में) एक बिंदु \omterm{def:b2:funcseq:def}{समान रूप से} खींचा गया है
        घन में $\intcc{-1}1^{20}$. दिखाओ कि
        प्रायिकता यह है कि यह अंकित गेंद में गिरेगा
        $ V_{20}/2^{20} \approx 2.5\cdot10^{-8}$, तो के बारे में
        पहली हिट से पहले चालीस मिलियन ड्रॉ की आवश्यकता होती है
        अपेक्षित है: अनुमान लगाना $ V_n $ अस्वीकृति नमूनाकरण द्वारा
        उच्च आयाम में ढह जाता है (का अभिशाप)
        आयामीता)।
  \item संश्लेषण। दो स्वतंत्र व्युत्पत्तियाँ मिलीं $ V_n =
        \pi^{n/2}/\Gamma(\frac n2 + 1)$: कौन सा प्रमेय सूचीबद्ध करें
        इस अध्याय में से प्रत्येक ने उपयोग किया (फ़ुबिनी, का परिवर्तन)।
        चर, ध्रुवीय गॉस अभिन्न), और कौन सा
        एक-चर इनपुट (वालिस, $\Gamma$, स्टर्लिंग)।
        वर्ष 3 खंड इस गणना को फिर से कहां करता है
        लेब्सग्यू के सिद्धांत के साथ, और यह क्या जोड़ता है?
\end{enumerate}
\end{problem}
