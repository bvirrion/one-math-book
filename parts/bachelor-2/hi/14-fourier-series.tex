\chapter{फूरियर श्रृंखला}\label{ch:b2:fourier}

क्या प्रत्येक आवधिक संकेत को शुद्ध साइन और कोसाइन से फिर से बनाया जा सकता है?
फूरियर की साहसी ``हाँ'' ने विश्लेषण की एक सदी बना दी। यह
अध्याय एमपी* पहुंच के भीतर दो स्तंभों को साबित करता है: \emph{डिरिचलेट का
प्रमेय} (\omterm{def:b2:funcseq:def}{बिंदुवार} टुकड़े-टुकड़े के लिए पुनर्निर्माण $C^1$ कार्य,
के माध्यम से \omterm{lem:b2:fourier:kernel}{डिरिचलेट कर्नेल}) और \emph{\omterm{thm:b2:fourier:parseval}{पारसेवल की पहचान}} (ऊर्जा
किसी सिग्नल का उसके हार्मोनिक्स की ऊर्जाओं का योग होता है), और
शास्त्रीय संख्यात्मक श्रृंखला --- बेसल का उपयोग करता है $\sum 1/n^2 =
\pi^2/6$ उनमें से पहला.

कुल मिलाकर, कार्य हैं $2\pi $-आवधिक, टुकड़े-टुकड़े \omterm{def:b2:metric:continuity}{निरंतर},
जटिल-मूल्यवान; $\mathcal{C}$ को दर्शाता है \omterm{def:b2:metric:continuity}{निरंतर} वाले.

\section{फूरियर गुणांक}

\begin{definition}\label{def:b2:fourier:coefficients}
 \emph{फूरियर गुणांक}\index{फूरियर गुणांक} का $ f $
हैं
\[
c_n(f) = \frac{1}{2\pi}\int_{-\pi}^{\pi} f(t)\,\eu^{-\iu n t}\,\dd
t \qquad (n \in \Z),
\]
और वास्तविक-रूप गुणांक $ a_n = c_n + c_{-n}$, $ b_n = \iu(c_n
- c_{-n})$, वैसा ही किया \emph{फूरियर आंशिक रकम}\index{फूरियर आंशिक रकम} हैं
\[
S_N(f)(t) = \sum_{n=-N}^{N} c_n(f)\,\eu^{\iu nt}
= \frac{a_0}{2} + \sum_{n=1}^{N}\bigl(a_n\cos nt + b_n \sin
nt\bigr).
\]
पर $\mathcal{C}$, को परिभाषित करो \omterm{def:b2:hermitian:def}{हर्मिटियन आंतरिक उत्पाद} $\langle f,
g\rangle = \frac{1}{2\pi}\int_{-\pi}^{\pi}\conj f\,g $: द
घातांक $ e_n(t) = \eu^{\iu nt}$ हैं \emph{ऑर्थोनॉर्मल}
($\langle e_m, e_n\rangle = \delta_{mn}$, प्रत्यक्ष गणना), और
$ c_n(f) = \langle e_n, f\rangle $: फूरियर विश्लेषण है \omterm{def:b2:hermitian:adjoint}{हर्मिटियन}
ज्यामिति (\cref{ch:b2:hermitian}) अनंत आयाम में।
\end{definition}

\begin{proposition}[बेसेल की असमानता]\label{prop:b2:fourier:bessel}
$ S_N(f)$ का ओर्थोगोनल प्रक्षेपण है $ f $ अंतरिक्ष पर
$\mathcal{T}_N $ डिग्री के त्रिकोणमितीय बहुपदों की $\leq N $,
और\index{बेसेल की असमानता}
\[
\sum_{n=-N}^{N} \abs{c_n(f)}^2 \leq \norm f_2^2 =
\frac{1}{2\pi}\int_{-\pi}^{\pi} \abs f^2 :
\]
शृंखला $\sum \abs{c_n}^2$ \omterm{def:b2:integration:improper}{अभिसरण}, और $ c_n(f) \to 0$ जैसा
$\abs n \to \infty $ (रीमैन--गुणांक के लिए लेब्सग्यू)।
\end{proposition}

\begin{proof}
$ f - S_N(f)$ प्रत्येक के लिए ओर्थोगोनल है $ e_k $, $\abs k \leq N $ ($\langle
e_k, f - S_N f\rangle = c_k - c_k = 0$): $ S_Nf $ ऑर्थोगोनल है
पर प्रक्षेपण $\mathcal{T}_N = \operatorname{Vect}(e_{-N}, \dots,
e_N)$ (वर्ष 1 खंड का प्रक्षेपण प्रमेय, शब्दशः
\omterm{def:b2:hermitian:adjoint}{हर्मिटियन} सेटिंग)। पाइथागोरस: $\norm f_2^2 = \norm{S_Nf}_2^2 +
\norm{f - S_Nf}_2^2 \geq \norm{S_Nf}_2^2 = \sum_{\abs n \leq N}
\abs{c_n}^2$; होने देना $ N \to \infty $.
\end{proof}

\begin{example}[सर्वोत्तम सन्निकटन, मापा गया]\label{ex:b2:fourier:bestapprox}
निम्न-डिग्री त्रिकोणमितीय बहुपद कितनी अच्छी तरह अनुमानित होते हैं
लकड़ी का चूरा $ f(t) = t $ (पर $\intoo{-\pi}{\pi}$) में
द्विघात माध्य? द्वारा \cref{prop:b2:fourier:bessel} सबसे अच्छा
डिग्री-$ N $ सन्निकटन \emph{है} $ S_N(f)$, चुकता त्रुटि के साथ
\[
\norm{f - S_Nf}_2^2 = \norm f_2^2 -
\sum_{\abs n\leq N}\abs{c_n}^2 .
\]
यहाँ $\norm f_2^2 = \frac{1}{2\pi}\int_{-\pi}^\pi t^2\dd t =
\frac{\pi^2}{3}$, और से $ b_n = \frac{2(-1)^{n+1}}{n}$
(\cref{ex:b2:fourier:basel}): $\abs{c_n}^2 + \abs{c_{-n}}^2 =
\frac{b_n^2}{2} = \frac{2}{n^2}$. इसलिए
\[
\norm{f - S_Nf}_2^2
= \frac{\pi^2}{3} - \sum_{n=1}^{N}\frac{2}{n^2}
\qquad\text{: numerically } 1.29,\ 0.79,\ 0.57,\ 0.44
\]
के लिए $ N = 1, 2, 3, 4$ --- घट रही है, लेकिन धीरे-धीरे: पूँछ
$\sum_{n>N}\frac2{n^2} \sim \frac2N $ धीमी गति से शासित होता है
$\frac1n $ गुणांकों का क्षय, स्वयं का \omterm{thm:b2:structures:signature}{हस्ताक्षर}
छलांग (\cref{exo:b2:fourier:6} पीछे की ओर पढ़ें)। समापन
अंतर्दृष्टि: पार्सेवल सन्निकटन गुणवत्ता को पूंछ में बदल देता है
संख्यात्मक श्रृंखला --- और किसी भी चित्र से पहले, भविष्यवाणी करती है
छलांगें फूरियर श्रृंखला को अनिच्छा से एकाग्र करती हैं।
\end{example}

\begin{method}[फूरियर गुणांक की कुशलतापूर्वक गणना करना]\label{met:b2:fourier:coefficients}
कुछ भी एकीकृत करने से पहले:
\begin{enumerate}
  \item \emph{समता:} यहां तक की $ f $ है $ b_n = 0$, विषम $ f $ है $ a_n
        = 0$, और बचे हुए अभिन्न अंग कम हो जाते हैं $\frac2\pi
        \int_0^\pi $ --- आधा काम, दोगुनी विश्वसनीयता।
  \item \emph{त्रिकोणमितीय बहुपद पहले ही हो चुके हैं:}
        उत्पादों को रैखिक बनाना ($\cos^3$, $\sin^2\cos $, \dots) और
        गुणांकों को पढ़ें
        (\cref{exo:b2:fourier:9}); रूढ़िवादिता किसी को भी बनाती है
        आगे एकीकरण बेमानी.
  \item \emph{घातांक के लिए जटिल घातांक:} के लिए
        कारक $\eu^{at}$ या नम दोलन, गणना करें
        $ c_n $ सीधे --- का एक अभिन्न अंग $\eu^{(a - \iu
        n)t}$ भागों द्वारा दो एकीकरणों को हराता है
        (\cref{exo:b2:fourier:10}).
  \item \emph{ज्ञात विस्तार को अलग करें:} अगर $ f' $ है
        ज्ञात गुणांक और $ f $ है \omterm{def:b2:metric:continuity}{निरंतर}, $ c_n(f) =
        \frac{c_n(f')}{\iu n}$ ($ n \neq 0$) लेकिन सभी को पुनः प्राप्त करता है
        $ c_0$, जो माध्य है --- अक्सर सबसे तेज़ मार्ग,
        और \omterm{def:b2:series:def}{बिल्कुल} की परिकल्पना के तहत वैध है
        \cref{thm:b2:fourier:parseval}~(1).
\end{enumerate}
\end{method}

\section{डिरिचलेट का प्रमेय}

\begin{lemma}[डिरिचलेट कर्नेल]\label{lem:b2:fourier:kernel}
$ S_N(f)(x) = \frac{1}{2\pi}\int_{-\pi}^{\pi} f(x + u)\,D_N(u)\,\dd
u $, कहाँ\index{डिरिचलेट कर्नेल}
\[
D_N(u) = \sum_{n=-N}^{N} \eu^{\iu nu}
= \frac{\sin\bigl((N + \frac12)u\bigr)}{\sin\frac u2}
\quad (u \notin 2\pi\Z),
\qquad
\frac{1}{2\pi}\int_{-\pi}^{\pi} D_N = 1 .
\]
\end{lemma}

\begin{proof}
की परिभाषा डालें $ c_n $ में $ S_N $ और स्वैप योग और अभिन्न
(वैध: योग सीमित है):
\[
S_N(f)(x)
= \sum_{n=-N}^{N}\Bigl(\frac{1}{2\pi}\int_{-\pi}^{\pi}
f(t)\,\eu^{-\iu nt}\dd t\Bigr)\eu^{\iu nx}
= \frac{1}{2\pi}\int_{-\pi}^{\pi} f(t)
\sum_{n=-N}^{N}\eu^{\iu n(x-t)}\,\dd t ;
\]
विकल्प $ u = t - x $ और एकीकरण खंड को पीछे की ओर स्लाइड करें
को $\intcc{-\pi}{\pi}$ द्वारा $2\pi $-एकीकरण की आवधिकता;
 \omterm{def:b2:quadratic:adjoint}{सममित} सूचकांक सीमा बनाता है $\sum_n\eu^{-\iu nu} = D_N(u)$.
बंद प्रपत्र: अनुपात के साथ ज्यामितीय योग $\eu^{\iu u}$,
\[
D_N(u) = \eu^{-\iu Nu}\,\frac{\eu^{\iu(2N+1)u} - 1}{\eu^{\iu u} -
1}
= \frac{\eu^{\iu(N + \frac12)u} - \eu^{-\iu(N+\frac12)u}}
{\eu^{\iu u/2} - \eu^{-\iu u/2}} ,
\]
जो साइन भागफल है. इसका माध्य है $1$: केवल $ n = 0$
योगदान देता है.
\end{proof}

\begin{theorem}[रीमैन--लेब्सग्यू लेम्मा]\label{thm:b2:fourier:riemannlebesgue}
के लिए $ g $ टुकड़े-टुकड़े में \omterm{def:b2:metric:continuity}{निरंतर} एक खंड पर,
$\int_a^b g(t)\sin(\lambda t + \varphi)\,\dd t \to 0$ जैसा $\lambda
\to +\infty $.\index{रीमैन--लेब्सग्यू लेम्मा}
\end{theorem}

\begin{proof}
अनुमानित $ g $ \omterm{def:b2:funcseq:def}{समान रूप से} चरणबद्ध कार्यों द्वारा
(\cref{thm:b2:funcseq:weierstrass} आवश्यक नहीं है---प्राथमिक
टुकड़े-टुकड़े का चरण-कार्य सन्निकटन \omterm{def:b2:metric:continuity}{निरंतर} कार्य
पर्याप्त) और प्रत्येक चरण को स्पष्ट रूप से एकीकृत करें: प्रत्येक टुकड़ा
योगदान देता है $ O\bigl(\frac1\lambda\bigr)$, और सन्निकटन
त्रुटि योगदान करती है $\varepsilon(b - a)$. यह तर्क रखा गया
वर्ष 1 खंड के अंतिम अभ्यास के रूप में पूर्णतः प्रकाशित
एकीकरण अध्याय; के लिए $ C^1$ इसके बजाय टुकड़ों को एकीकृत किया जा सकता है
भागों और से बंधे $\frac C\lambda $.
\end{proof}

\begin{example}[गुणांक कितनी तेजी से मरते हैं?]\label{ex:b2:fourier:decaytable}
रीमैन--लेब्सग्यू का कहना है कि गुणांक की प्रवृत्ति होती है $0$; उनका
\emph{दर} एक चिकनाई मीटर है. इसके तीन नमूने
अध्याय और उसके अभ्यास:
\[
\text{square wave: } b_n = \frac{4}{\pi n}\ (n\ \text{odd}),
\qquad
\abs t : a_n = \frac{-4}{\pi n^2}\ (n\ \text{odd}),
\qquad
\abs{\sin t} : a_{2k} = \frac{-4}{\pi(4k^2-1)} .
\]
एक छलांग $ f $ (स्क्वायर वेव, सॉटूथ) के गुणांक छोड़ देता है
\omterm{def:b2:structures:generated}{आदेश} $\frac1n $: नहीं सामान्य \omterm{def:b2:integration:improper}{अभिसरण}, गिब्स ओवरशूट करते हैं
छलांग. निरंतरता एक कोने के साथ --- एक छलांग $ f'$ केवल ---
करने के लिए \omterm{def:b2:structures:generated}{आदेश} में सुधार करता है $\frac{1}{n^2}$: सामान्य \omterm{def:b2:integration:improper}{अभिसरण},
एकसमान पुनर्निर्माण. सामान्य तौर पर $ k $ डेरिवेटिव खरीदें $ c_n =
O(n^{-k})$ (\cref{exo:b2:fourier:6}), और इसके विपरीत ए \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम}
हर शक्ति की तुलना में तेजी से क्षय हो रहा है $ f $ होना $ C^\infty $
(शब्द दर शब्द अंतर करें, अब वैध रूप से)। समापन
अंतर्दृष्टि: सिग्नल की नियमितता और क्षय \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम}
ये वही जानकारी हैं --- एक इंजीनियर इनमें से एक को पढ़ता है
फ़ंक्शन को प्लॉट किए बिना दूसरे का ढलान।
\end{example}

\begin{theorem}[Dirichlet]\label{thm:b2:fourier:dirichlet}
होने देना $ f $ होना $2\pi $-आवधिक और टुकड़ेवार $ C^1$. फिर हर एक के लिए $ x $,
\[
S_N(f)(x) \xrightarrow[N \to \infty]{}
\frac{f(x^+) + f(x^-)}{2}
\]
(एकतरफा सीमाओं का माध्य) --- विशेष रूप से $ S_N(f)(x) \to
f(x)$ के हर बिंदु पर निरंतरता.\index{डिरिचलेट का प्रमेय}
\end{theorem}


\begin{proof}
कर्नेल लेम्मा और इसकी इकाई से तात्पर्य, अभिन्न को विभाजित करना है
$ u > 0$ और $ u < 0$ आधे (प्रत्येक का मतलब) $\frac12$):
\begin{align*}
S_N(f)(x) - \frac{f(x^+) + f(x^-)}{2}
&= \frac{1}{2\pi}\int_{0}^{\pi} \bigl(f(x+u) -
f(x^+)\bigr)D_N(u)\,\dd u\\
&\quad+ \frac{1}{2\pi}\int_{-\pi}^{0}\bigl(f(x+u) -
f(x^-)\bigr)D_N(u)\,\dd u .
\end{align*}
पहले का इलाज करें (दूसरा है)। \omterm{def:b2:quadratic:adjoint}{सममित}). लिखना
\[
\bigl(f(x + u) - f(x^+)\bigr)\,D_N(u)
= \underbrace{\frac{f(x+u) - f(x^+)}{\sin\frac u2}}_{g(u)}\,
\sin\Bigl(\Bigl(N + \frac12\Bigr)u\Bigr) .
\]
समारोह $ g $ टुकड़े-टुकड़े में है \omterm{def:b2:metric:continuity}{निरंतर} पर $\intoc{0}{\pi}$ और
एक है \emph{पर परिमित सीमा $0^+$}: लिखना
\[
g(u) = \frac{f(x+u) - f(x^+)}{u}\cdot\frac{u}{\sin\frac u2} ,
\]
पहला कारक प्रवृत्त होता है $ f'(x^+)$ (एकतरफ़ा
भिन्नता, टुकड़े-टुकड़े से $ C^1$) और दूसरा को $2$
(मानक सीमा $\frac{\sin v}{v} \to 1$ पर $ v = \frac
u2$): $ g(0^+) = 2f'(x^+)$ मौजूद है. इसलिए $ g $ टुकड़े-टुकड़े में विस्तारित होता है
लगातार को $\intcc{0}{\pi}$, और रीमैन--लेब्सग्यू
(\cref{thm:b2:fourier:riemannlebesgue}) को अभिन्न भेजता है
$0$. यह परिकल्पना का संपूर्ण बिंदु है: बिना
एकतरफ़ा व्युत्पन्न, कारक $\frac{1}{\sin(u/2)}$ मारता है
ऊपर $0$ रीमैन की तुलना में तेज़--लेबेस्ग क्षतिपूर्ति कर सकता है, और
\omterm{def:b2:funcseq:def}{बिंदुवार} \omterm{def:b2:integration:improper}{अभिसरण} वास्तव में केवल के लिए विफल हो सकता है \omterm{def:b2:metric:continuity}{निरंतर}
$ f $ --- फ़ेज़ का अंतर\'एर का प्रमेय (सप्ताहांत समस्या)
औसत से बंद होता है.
\end{proof}

\begin{example}[एक छलांग में डिरिचलेट]\label{ex:b2:fourier:jumpvalue}
लकड़ी के चूरे के लिए $ f(t) = t $ पर $\intoo{-\pi}{\pi}$
(\cref{ex:b2:fourier:basel} नीचे), आवधिक विस्तार
पर कूदता है $ t = \pi $ से $ f(\pi^-) = \pi $ को $ f(\pi^+) = -\pi $.
डिरिचलेट मूल्य का वादा करता है $\frac{\pi + (-\pi)}{2} = 0$
वहाँ, और वास्तव में प्रत्येक पद $\sum
\frac{2(-1)^{n+1}}{n}\sin nt $ पर गायब हो जाता है $ t = \pi $: द
शृंखला विनम्रतापूर्वक दोनों को नज़रअंदाज करते हुए मध्यबिंदु पर आ जाती है
एकतरफ़ा मूल्य. मूल्यांकन बिंदु को आगे ले जाना $ t =
\frac\pi2$ इसके बजाय (ए निरंतरता बिंदु) समान श्रृंखला में बदल जाता है
लीबनिज में $\frac\pi4$. एक श्रृंखला, दो व्यवहार ---
प्रमेय के \omterm{def:b2:series:def}{बिल्कुल} दो खंड।
\end{example}

\begin{theorem}[के लिए सामान्य अभिसरण $ C^1$; पारसेवल]\label{thm:b2:fourier:parseval}
\begin{enumerate}
  \item अगर $ f $ है \omterm{def:b2:metric:continuity}{निरंतर}, $2\pi $-आवधिक और टुकड़ेवार $ C^1$,
        तब $ c_n(f') = \iu n\,c_n(f)$, फूरियर श्रृंखला $ f $
        \omterm{def:b2:integration:improper}{अभिसरण} \emph{\omterm{def:b2:funcseq:series}{सामान्यतः}} पर $\R $, और इसका योग है $ f $.
  \item (पारसेवल) प्रत्येक टुकड़े के लिए \omterm{def:b2:metric:continuity}{निरंतर} $2\pi $-आवधिक
        $ f $:\index{पारसेवल की पहचान}
        \[
        \frac{1}{2\pi}\int_{-\pi}^{\pi}\abs{f}^2
        = \sum_{n=-\infty}^{\infty} \abs{c_n(f)}^2
        = \frac{\abs{a_0}^2}{4} + \frac12\sum_{n\geq1}
        \bigl(\abs{a_n}^2 + \abs{b_n}^2\bigr).
        \]
        \emph{(यहाँ के लिए साबित हुआ $ f $ \omterm{def:b2:metric:continuity}{निरंतर} टुकड़े-टुकड़े में $ C^1$;
        सामान्य रूप से स्वीकार किया गया।)}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) प्रत्येक पर भागों द्वारा एकीकरण $ C^1$ टुकड़ा (सीमा शर्तें रद्द करें)
द्वारा निरंतरता और आवधिकता): $ c_n(f') = \iu n c_n(f)$. फिर, द्वारा
कॉची--श्वार्ज़ दो पर वर्ग-योगयोग्य परिवार
(\cref{prop:b2:fourier:bessel} के लिए $ f'$):
\[
\sum_{n \neq 0} \abs{c_n(f)} = \sum_{n\neq0}
\frac{\abs{c_n(f')}}{\abs n}
\leq \Bigl(\sum \abs{c_n(f')}^2\Bigr)^{1/2}
\Bigl(\sum_{n\neq0}\frac{1}{n^2}\Bigr)^{1/2} < \infty :
\]
सामान्य \omterm{def:b2:integration:improper}{अभिसरण} फूरियर श्रृंखला के. इसका योग है \omterm{def:b2:metric:continuity}{निरंतर} और
के साथ मेल खाता है $ f $ डिरिचलेट द्वारा हर बिंदु पर
(\cref{thm:b2:fourier:dirichlet}: $ f $ है \omterm{def:b2:metric:continuity}{निरंतर}: श्रृंखला
में एकत्रित हो जाता है $ f $, \omterm{def:b2:funcseq:def}{समान रूप से}.

(2) ऐसे के लिए $ f $: $ S_N f \to f $ \omterm{def:b2:funcseq:def}{समान रूप से}, इसलिए $\norm{f - S_Nf}_2
\leq \norm{f - S_Nf}_\infty \to 0$, और पाइथागोरस
($\norm f_2^2 = \sum_{\abs n \leq N}\abs{c_n}^2 + \norm{f -
S_Nf}_2^2$) सीमा तक गुजरता है। वास्तविक स्वरूप बहीखाता पद्धति का है
$ a_n, b_n $.
\end{proof}

\begin{example}[ $ C^1$ पूँछ बंधी, मात्रात्मक बनाई गई]\label{ex:b2:fourier:tailbound}
का प्रमाण \cref{thm:b2:fourier:parseval}~(1) एक प्रयोग योग्य को छुपाता है
अनुमान. के लिए \omterm{def:b2:metric:continuity}{निरंतर} टुकड़े-टुकड़े-$ C^1$ $ f $, वही
कॉची--श्वार्ज़ को पूँछ पर लगाने से ही लाभ मिलता है
\[
\sum_{\abs n > N}\abs{c_n(f)}
= \sum_{\abs n > N}\frac{\abs{c_n(f')}}{\abs n}
\leq \Bigl(\sum_{\abs n > N}\abs{c_n(f')}^2\Bigr)^{\!1/2}
\Bigl(\sum_{\abs n>N}\frac{1}{n^2}\Bigr)^{\!1/2}
\leq \norm{f'}_2\,\sqrt{\frac{2}{N}} ,
\]
के लिए बेसेल का उपयोग करना $ f'$ और $\sum_{n>N}n^{-2} \leq \frac1N $. तो
आंशिक योगों की एकसमान त्रुटि का पालन होता है
\[
\norm{f - S_Nf}_\infty
\leq \sum_{\abs n>N}\abs{c_n(f)}
\leq \norm{f'}_2\,\sqrt{\frac2N} .
\]
के लिए $ f(t) = \abs t $: $\norm{f'}_2 = 1$ (व्युत्पन्न है
$\pm1$), इसलिए दस पद पहले से ही पुनर्निर्माण कर चुके हैं $\abs t $ \omterm{def:b2:funcseq:def}{समान रूप से}
अंदर $\sqrt{0.2} \approx 0.45$, और $ N = 10^4$ अंदर
$0.015$. समापन अंतर्दृष्टि: एक डेरिवेटिव खरीदता है
$\frac{1}{\sqrt N}$ एकसमान दर; के साथ तुलना करना
$\frac1n $-वर्ग तरंग का गुणांक विश्व (संख्या) वर्दी
\omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:series:def}{बिल्कुल}), का शब्दकोश
\cref{ex:b2:fourier:decaytable} संख्याएँ प्राप्त करता है।
\end{example}

\begin{example}[बेसल और दोस्त]\label{ex:b2:fourier:basel}
होने देना $ f(t) = t $ पर $\intoo{-\pi}{\pi}$, विस्तारित $2\pi $-समय-समय पर
(एक चूरा, टुकड़े-टुकड़े में $ C^1$). कंप्यूटिंग, $ a_n = 0$ (विषमता) और
\[
b_n = \frac{1}{\pi}\int_{-\pi}^{\pi} t\sin nt\,\dd t
= \frac{2(-1)^{n+1}}{n} .
\]
डिरिचलेट पर $ t = \frac\pi2$ लीबनिज को पुनः प्राप्त करता है $\frac\pi4 = 1 -
\frac13 + \frac15 - \dots $; पारसेवल देता है
\[
\frac{1}{2\pi}\int_{-\pi}^{\pi} t^2\,\dd t = \frac{\pi^2}{3}
= \frac12\sum_{n\geq1}\frac{4}{n^2}
\quad\Longrightarrow\quad
\boxed{\;\sum_{n\geq1}\frac{1}{n^2} = \frac{\pi^2}{6}\;}
\]
--- यूलर का बेसल योग, दो पंक्तियों में। समारोह $ f(t) = t^2$
इसी तरह पैदावार होती है $\sum \frac1{n^4} = \frac{\pi^4}{90}$
(\cref{exo:b2:fourier:3}).
\end{example}

\begin{example}[अंतर्निर्मित जांच के साथ पूर्ण विस्तार: $\abs{\sin t}$]\label{ex:b2:fourier:abssin}
समारोह $ f(t) = \abs{\sin t}$ है \omterm{def:b2:metric:continuity}{निरंतर}, यहां तक की,
$\pi $-आवधिक (इसलिए $2\pi $-आवधिक), टुकड़े-टुकड़े $ C^1$.
समता मार देती है $ b_n $; $ a_0 = \frac1\pi\int_{-\pi}^{\pi}
\abs{\sin t}\dd t = \frac4\pi $; और के लिए $ n \geq 1$,
उत्पाद-से-योग देता है
\[
a_n = \frac2\pi\int_0^\pi \sin t\cos nt\,\dd t
= \frac{1}{\pi}\int_0^\pi\bigl(\sin(1+n)t +
\sin(1-n)t\bigr)\dd t
= \frac2\pi\cdot\frac{1 + \cos n\pi}{1 - n^2}
\]
के लिए $ n \neq 1$ (और $ a_1 = 0$ सीधे): विषम के लिए शून्य $ n $, और
$ a_{2k} = \frac{-4}{\pi(4k^2-1)}$. द्वारा
\cref{thm:b2:fourier:parseval}~(1) \omterm{def:b2:integration:improper}{अभिसरण} सामान्य है,
और
\[
\abs{\sin t} = \frac{2}{\pi} - \frac{4}{\pi}
\sum_{k\geq1}\frac{\cos(2kt)}{4k^2 - 1}
\qquad (t \in \R) .
\]
बिल्ट-इन चेक एट $ t = 0$: पहचान मांगती है
$\sum_{k\geq1}\frac{1}{4k^2-1} = \frac12$, जो दूरबीन
पुष्टि करता है:
\[
\sum_{k\geq1}\frac{1}{4k^2-1}
= \frac12\sum_{k\geq1}\Bigl(\frac{1}{2k-1} -
\frac{1}{2k+1}\Bigr) = \frac12 . \checkmark
\]
समापन अंतर्दृष्टि: \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} का $\abs{\sin}$ पर रहता है
\emph{यहां तक की} केवल आवृत्तियाँ --- एक साइन को सुधारने से वह दोगुनी हो जाती है
आवृत्ति सामग्री, यही कारण है कि फुल-वेव रेक्टिफायर्स गुनगुनाते हैं
$100$ या $120$ हर्ट्ज़, मुख्य आवृत्ति से दोगुना।
\end{example}

\begin{example}[एक कंप्यूटिंग डिवाइस के रूप में पार्सेवल]\label{ex:b2:fourier:zetafourodd}
पार्सेवल विस्तार को संख्यात्मक श्रृंखला थोक में बदल देता है।
इसे लागू करें $ f(t) = \abs t $ (\cref{exo:b2:fourier:2}: $ a_0 =
\pi $, $ a_n = \frac{-4}{\pi n^2}$ विषम के लिए $ n $, शेष शून्य):
\[
\frac{1}{2\pi}\int_{-\pi}^{\pi}t^2\,\dd t = \frac{\pi^2}{3}
= \frac{a_0^2}{4} + \frac12\sum_{n \text{ odd}} a_n^2
= \frac{\pi^2}{4} +
\frac{8}{\pi^2}\sum_{n\text{ odd}}\frac{1}{n^4} ,
\]
जहां से
\[
\sum_{n\text{ odd}}\frac{1}{n^4}
= \frac{\pi^2}{8}\Bigl(\frac{\pi^2}{3} -
\frac{\pi^2}{4}\Bigr) = \frac{\pi^4}{96} .
\]
के विरुद्ध क्रॉस-चेक करें \cref{exo:b2:fourier:3}: विभाजन
$\sum\frac1{n^4}$ विषम और सम भागों में देता है $\zeta $-शैली
बहीखाता $ S = S_{\mathrm{odd}} + \frac{S}{16}$, इसलिए $ S =
\frac{16}{15}\cdot\frac{\pi^4}{96} = \frac{\pi^4}{90}$ ---
\omterm{def:b2:series:def}{बिल्कुल} वही मान जो किसी भिन्न फ़ंक्शन द्वारा वहां पाया गया। दो
विस्तार, एक संख्या: संगति पार्सेवल की आइसोमेट्री है
काम पर. समापन अंतर्दृष्टि: प्रत्येक नया फूरियर विस्तार एक है
श्रृंखला पहचान के लिए जनरेटिंग मशीन; सप्ताहांत
समस्या का भाग II बताता है कि मशीन कभी ऐसा क्यों नहीं कर सकती
स्वयं का खंडन करें.
\end{example}

\begin{example}[अनुवाद और मॉड्यूलेशन]\label{ex:b2:fourier:shiftrules}
दो एक-पंक्ति नियम एक से कई विस्तार उत्पन्न करते हैं। के लिए $ a
\in \R $, प्रतिस्थापित करना $ s = t - a $:
\[
c_n\bigl(f(\cdot - a)\bigr)
= \frac{1}{2\pi}\int_{-\pi}^{\pi}f(t - a)\eu^{-\iu nt}\dd t
= \eu^{-\iu na}\,c_n(f)
\qquad\text{(translation modulates the spectrum)},
\]
और सीधे परिभाषा से,
\[
c_n\bigl(\eu^{\iu kt}f\bigr) = c_{n-k}(f)
\qquad\text{(modulation translates the spectrum)}.
\]
काम किया गया उदाहरण: चूरा द्वारा स्थानांतरित किया गया $\pi $, $ g(t) = f(t -
\pi)$ साथ $ f(t) = t $, है $ b_n $-गुणांक
$(-1)^n\cdot\frac{2(-1)^{n+1}}{n} = -\frac2n $: विस्तार
$ g \sim -2\sum\frac{\sin nt}{n}$ उस चूरे का जो कूदता है
$0$ पर के बजाय $\pi $ --- कोई अभिन्न पुनर्गणना नहीं। समापन
अंतर्दृष्टि: समय परिवर्तन केवल चरण बदलता है, आयाम कभी नहीं
($\abs{c_n}$ शिफ्ट-अपरिवर्तनीय है), यही कारण है कि ऊर्जा
(पारसेवल) और \omterm{def:b2:integration:improper}{अभिसरण} वर्ग सिग्नल के गुण हैं
\emph{आकार}, यह नहीं कि घड़ी कहाँ से शुरू होती है।
\end{example}

\begin{remark}[सामान्य ख़तरे]
\emph{(i) तीन \omterm{def:b2:integration:improper}{अभिसरण}, तीन मुद्राएँ:} \omterm{def:b2:funcseq:def}{बिंदुवार}
(डिरिचलेट: टुकड़े-टुकड़े की जरूरत है $ C^1$, भुगतान करता है \emph{मध्य} पर
प्रत्येक छलांग --- कभी भी एकतरफ़ा मूल्य नहीं), एकसमान (एक की आवश्यकता होती है)।
\omterm{def:b2:metric:continuity}{निरंतर} सीमा; एक छलांग पार करना असंभव है, गिब्स है
दृश्यमान लक्षण), और द्विघात माध्य (पारसेवल: सबसे अधिक)।
मजबूत, व्यक्तिगत बिंदुओं पर ध्यान न देना)। हमेशा आप कौन सा नाम बताएं
दावा कर रहे हैं. \emph{(ii) कोई पदवार विभेदन नहीं
डिफ़ॉल्ट:} सॉटूथ श्रृंखला को अलग करना
\cref{ex:b2:fourier:basel} अवधि दर अवधि पैदावार $\sum
2(-1)^{n+1}\cos nt $, जिनकी शर्तें भी लागू नहीं होतीं $0$ ---
फ़ंक्शन-अनुक्रम अध्याय के \omterm{def:b2:structures:sn}{स्थानांतरण} प्रमेय की आवश्यकता है
एकसमान \omterm{def:b2:integration:improper}{अभिसरण} की \emph{व्युत्पन्न} श्रृंखला, जो
छलांग नष्ट कर देती है. चिकनाई पहले, भेदभाव दूसरा
(\cref{exo:b2:fourier:6} शब्दकोश है) \emph{(iii)
\omterm{def:b2:quadratic:adjoint}{सममित} आंशिक रकम:} डिरिचलेट का प्रमेय चिंता का विषय है $ S_N =
\sum_{-N}^{N}$; पहले एक पक्ष को पुनर्व्यवस्थित या संक्षेपित कर सकते हैं
विचलन को \omterm{def:b2:integration:improper}{अभिसरण} और विपरीत में बदलें। \emph{(iv)
सामान्यीकरण बहाव:} पुस्तकों में परंपराएँ भिन्न-भिन्न होती हैं
($\frac{1}{2\pi}$ या $\frac1\pi $ सामने, अवधि $2\pi $ या
$1$); विश्वसनीय अपरिवर्तनीयता ऑर्थोनॉर्मलिटी संबंध हैं
--- पुनर्गणना करें $\langle e_m, e_n\rangle $ सम्मेलन में
किसी भी फॉर्मूले पर भरोसा करने से पहले हाथ धो लें.
\end{remark}

\begin{remark}[इसका उपयोग कहां किया जाता है]
पारसेवल का रोगाणु है $ L^2$ फूरियर श्रृंखला का सिद्धांत:
वर्ष 3 खंड चित्र को \omterm{def:b2:metric:complete}{पूरा} करता है (घातांक एक हैं
हिल्बर्ट का आधार $ L^2$, और नक्शा $ f \mapsto (c_n)$ एक है
बायजेक्टिव आइसोमेट्री)। इस खंड के अंतर्गत, सप्ताहांत समस्या
साबित करता है \emph{फेज\'एर का प्रमेय} --- सेस\`एआरओ का मतलब है
फूरियर श्रृंखला \omterm{def:b2:integration:improper}{अभिसरण} \omterm{def:b2:funcseq:def}{समान रूप से} हरएक के लिए \omterm{def:b2:metric:continuity}{निरंतर} आवधिक
$ f $ --- जो स्वीकृत सामान्य पार्सेवल को अपग्रेड करता है
प्रमेय, त्रिकोणमितीय वीयरस्ट्रैस प्रमेय उत्पन्न करता है, और भुगतान करता है
दो शानदार लाभांश: वेइल का समवितरण प्रमेय और
आइसोपेरिमेट्रिक असमानता। अनुप्रयुक्त गणित इसे पढ़ता है
अध्याय दैनिक: संकेतों का स्पेक्ट्रा, कंपन के हार्मोनिक्स
सिस्टम, और तेज फूरियर ट्रांसफॉर्म (परिमित अवतार था)।
\cref{exo:b2:hermitian:10}).
\end{remark}

\begin{remark}[इस खंड के भीतर परिप्रेक्ष्य]
तीन अध्याय इसी से बातचीत करते हैं। पिछड़ा: द \omterm{def:b2:hermitian:adjoint}{हर्मिटियन}
अध्याय ने ज्यामिति की आपूर्ति की (ऑर्थोनॉर्मल परिवार,
अनुमान, बेसेल), और फ़ंक्शन-अनुक्रम अध्याय
विश्लेषण (एकसमान \omterm{def:b2:integration:improper}{अभिसरण}, \omterm{def:b2:structures:sn}{स्थानांतरण} प्रमेय, अनुमानित
पहचान --- Fej\'एर का कर्नेल फूरियर श्रृंखला से क्या है
बर्नस्टीन के बहुपद वीयरस्ट्रैस के लिए थे)। बग़ल में:
शक्ति-श्रृंखला अध्याय का सीमा सिद्धांत वापस आता है \omterm{thm:b2:series:abel}{हाबिल
सारांश}, यहाँ पॉइसन कर्नेल द्वारा एहसास हुआ
(\cref{exo:b2:fourier:12}) --- डिस्क की त्रिज्या $ r $ खेल रहे हैं
योग पैरामीटर की भूमिका. आगे: द
अंतर-समीकरण अध्याय आवधिक बल को विघटित करता है
हार्मोनिक्स में और प्रत्येक को ऑसिलेटर की आवृत्ति पर फ़ीड करता है
प्रतिक्रिया; अनुनाद तब होता है जब इनपुट का फूरियर मोड होता है
प्राकृतिक आवृत्ति से मेल खाता है, यही कारण है कि इसकी सप्ताहांत समस्या है
और यह अध्याय एक ही कहानी के दो भाग हैं।
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b2:fourier:1}
की गणना करें \omterm{def:b2:fourier:coefficients}{फूरियर गुणांक} वर्ग तरंग का ($ f = -1$ पर
$\intoo{-\pi}{0}$, $+1$ पर $\intoo{0}{\pi}$), डिरिचलेट का राज्य
पर निष्कर्ष $ t = \frac\pi2$ और छलांग पर $ t = 0$, और ठीक हो जाओ
लीबनिज़ की श्रृंखला.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:fourier:2}
बढ़ाना $ f(t) = \abs t $ ($\abs t \leq \pi $, $2\pi $-आवधिक) में
फूरियर श्रृंखला; औचित्य \emph{सामान्य} \omterm{def:b2:integration:improper}{अभिसरण}; पर मूल्यांकन करें $ t =
0$ प्राप्त करने के लिए $\sum_{k\geq0}\frac{1}{(2k+1)^2} = \frac{\pi^2}{8}$,
और इससे बेसल को पुनः प्राप्त करें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:fourier:3}
बढ़ाना $ f(t) = t^2$ ($\abs t \leq \pi $) और निष्कर्ष निकालें
\[
\sum_{n\geq1}\frac{(-1)^{n+1}}{n^2} = \frac{\pi^2}{12},
\qquad
\sum_{n\geq1}\frac{1}{n^4} = \frac{\pi^4}{90}
\quad\text{(Parseval)}.
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:fourier:4}
होने देना $ f $ होना \omterm{def:b2:metric:continuity}{निरंतर} $2\pi $-आवधिक साथ $ c_n(f) = 0$ सभी के लिए
$ n $. सिद्ध करना $ f = 0$ \emph{(पारसेवल --- यह किस वर्ग के लिए सिद्ध है
यहाँ? उसे उचित ठहराओ निरंतरता साथ ही टुकड़े-टुकड़े में $ C^1$ गिराया जा सकता है
सामान्य पार्सेवल को स्वीकार करके, या घनत्व तर्क दें
रूपरेखा)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:fourier:5}
के लिए $\alpha \notin \Z $, बढ़ाना $ f(t) = \cos(\alpha t)$ ($\abs t
\leq \pi $) और के आंशिक-अंश विस्तार को कम करें
कोटैंजेंट:
\[
\pi\cot(\pi\alpha) = \frac{1}{\alpha} + \sum_{n\geq1}
\frac{2\alpha}{\alpha^2 - n^2} .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:fourier:6}
यदि ऐसा हो तो सिद्ध करें $ f $ है $2\pi $-आवधिक और $ C^k $ साथ $ f^{(k)}$
टुकड़े-टुकड़े में \omterm{def:b2:metric:continuity}{निरंतर}, तब $ c_n(f) = O\bigl(\abs n^{-k}\bigr)$:
सिग्नल की चिकनाई $=$ इसका क्षय \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:fourier:7}
(विर्टिंगर की असमानता) आइए $ f $ होना $ C^1$, $2\pi $-आवधिक, साथ
$\int_{-\pi}^{\pi} f = 0$. सिद्ध करना
\[
\int_{-\pi}^{\pi} \abs{f}^2 \leq \int_{-\pi}^{\pi} \abs{f'}^2 ,
\]
समानता के साथ यदि $ f(t) = a\cos t + b\sin t $. \emph{(परसेवल पर
दोनों तरफ; तुलना करें $\abs{c_n}^2$ और $ n^2\abs{c_n}^2$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:fourier:8}
(गिब्स स्थिरांक) की वर्ग तरंग के लिए
\cref{exo:b2:fourier:1}, आंशिक योग का मूल्यांकन करें $ x_N =
\frac{\pi}{2N}$: लिखना $ u_k = \frac{(2k+1)\pi}{2N}$ और $\Delta u
= \frac{\pi}{N}$, बताते हैं कि
\[
S_{2N-1}\Bigl(\frac{\pi}{2N}\Bigr)
= \frac{4}{\pi}\sum_{k=0}^{N-1} \frac{\sin u_k}{2k+1}
= \frac{2}{\pi}\sum_{k=0}^{N-1} \frac{\sin u_k}{u_k}\,\Delta u
\xrightarrow[N\to\infty]{}
\frac{2}{\pi}\int_0^{\pi}\frac{\sin u}{u}\,\dd u \approx 1.179 :
\]
रीमैन का योग $\frac{2}{\pi}\cdot\frac{\sin u}{u}$ पर
$\intcc{0}{\pi}$ मध्यबिंदु पर. निष्कर्ष निकालिए कि ओवरशूट परे है
छलांग मूल्य $1$ के रूप में गायब नहीं होता $ N \to \infty $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:fourier:9}
बढ़ाना $\cos^3 t $ और $\sin^2 t\,\cos t $ फूरियर श्रृंखला में
\emph{(रैखिकीकरण; एक त्रिकोणमितीय बहुपद का अपना फूरियर होता है
श्रृंखला, गुणांक की विशिष्टता द्वारा)}. क्या हैं $ c_n $,
$ a_n $, $ b_n $ प्रत्येक के लिए?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:fourier:10}
होने देना $ a > 0$ और $ f(t) = \eu^{at}$ पर $\intoc{-\pi}{\pi}$,
विस्तारित $2\pi $-समय-समय पर. गणना करना
\[
c_n(f) = \frac{(-1)^n\sinh(a\pi)}{\pi(a - \iu n)} ,
\]
कूदने पर डिरिचलेट प्रमेय लागू करें $ t = \pi $, और निष्कर्ष निकालें
हाइपरबोलिक कोटैंजेंट का आंशिक-अंश विस्तार:
\[
\coth(\pi a) = \frac{1}{\pi a} + \sum_{n\geq1}
\frac{2a}{\pi(a^2 + n^2)} .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:fourier:11}
(कन्वोल्यूशन) के लिए $ f, g $ \omterm{def:b2:metric:continuity}{निरंतर} और $2\pi $-आवधिक परिभाषा
\[
(f * g)(x) = \frac{1}{2\pi}\int_{-\pi}^{\pi} f(x - t)\,g(t)\,
\dd t .
\]
बताते हैं कि $ f * g = g * f $, वह $ c_n(f*g) = c_n(f)\,c_n(g)$
\emph{(ए के दो अभिन्नों को स्वैप करने के लिए \omterm{def:b2:metric:continuity}{निरंतर} एकीकरण,
ऊपरी सीमा के दो कार्यों की तुलना करें: दोनों गायब हो जाते हैं
बाएँ समापनबिंदु और समान व्युत्पन्न है, द्वारा निरंतरता और
अभिन्न चिन्ह के तहत विभेदन)}, ओर वो $ S_N(f) =
f * D_N $ के लिए \omterm{lem:b2:fourier:kernel}{डिरिचलेट कर्नेल}. (सप्ताहांत की समस्या
फेज\'एर साधन इसी प्रकार संवलन हैं, $\sigma_N(f) = f *
F_N $.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:fourier:12}
(पॉइसन कर्नेल: एबेल का अर्थ फूरियर श्रृंखला है) के लिए $0 \leq r <
1$ तय करना $ P_r(t) = \sum_{n\in\Z} r^{\abs n}\eu^{\iu nt}$.
\begin{enumerate}
  \item दो ज्यामितीय श्रृंखलाओं का योग करें और दिखाएँ
        \[
        P_r(t) = \frac{1 - r^2}{1 - 2r\cos t + r^2} > 0,
        \qquad
        \frac{1}{2\pi}\int_{-\pi}^{\pi}P_r = 1 .
        \]
  \item उसके लिए दिखाओ $\delta \leq \abs t \leq \pi $: $ P_r(t)
        \leq \frac{1 - r^2}{1 - 2r\cos\delta + r^2} \to 0$ जैसा
        $ r \to 1^-$, \omterm{def:b2:funcseq:def}{समान रूप से}.
  \item प्रत्येक के लिए यह निष्कर्ष निकालें \omterm{def:b2:metric:continuity}{निरंतर} $2\pi $-आवधिक $ f $,
        द \emph{हाबिल का मतलब है} $(f * P_r)(x) = \sum_n
        r^{\abs n}c_n(f)\,\eu^{\iu nx}$ में जुटना $ f $
        \omterm{def:b2:funcseq:def}{समान रूप से} जैसा $ r \to 1^-$ --- सतत-पैरामीटर
        फ़ेज़ का भाई\'एर का प्रमेय, और फूरियर
        का अवतार \omterm{thm:b2:series:abel}{हाबिल सारांश} पावर-श्रृंखला से
        अध्याय.
\end{enumerate}
\end{exercise}

\section{समस्या: फेज\'एर का प्रमेय और उसका लाभांश}

\begin{problem}\label{pb:b2:fourier:1}
डिरिचलेट की प्रमेय की आवश्यकता है $ f $ टुकड़े-टुकड़े में $ C^1$; मात्र के लिए
\omterm{def:b2:metric:continuity}{निरंतर} $ f $ आंशिक रकम $ S_N(f)$ दुर्व्यवहार कर सकते हैं. फेज\'एर का
खोज: उनका \emph{सीईएस\`एआरओ का मतलब है} कभी मत करो. इंजन है
फेज की सकारात्मकता\'एर गिरी, और फसल है
अपार: एकसमान त्रिकोणमितीय सन्निकटन (वीयरस्ट्रैस),
की विशिष्टता \omterm{def:b2:fourier:coefficients}{फूरियर गुणांक}, प्रत्येक फ़ंक्शन के लिए पारसेवल
इस अध्याय में (``स्वीकृत'' को हटाते हुए)
\cref{thm:b2:fourier:parseval}), वेइल का \omterm{pb:b2:fourier:1}{समान वितरण}
प्रमेय, और --- ज्यामिति की एक सदी का ताज ---
आइसोपेरिमेट्रिक असमानता।\index{Fejer's theorem@Fejér's
theorem}\index{समान वितरण}\index{आइसोपेरिमेट्रिक असमानता}
लगातार, $ f $ है $2\pi $-आवधिक और टुकड़ेवार \omterm{def:b2:metric:continuity}{निरंतर},
और
\[
\sigma_N(f) = \frac{S_0(f) + S_1(f) + \dots +
S_{N-1}(f)}{N} .
\]

\medskip
\noindent\textbf{Part I --- The Fejér kernel.}
\begin{enumerate}
  \item बताते हैं कि $\sigma_N(f)(x) =
        \frac{1}{2\pi}\int_{-\pi}^{\pi}f(x+u)\,F_N(u)\,\dd u $
        साथ $ F_N = \frac{D_0 + \dots + D_{N-1}}{N}$, ओर वो $\frac{1}{2\pi}\int_{-\pi}^{\pi}F_N = 1$.
  \item के लिए, बंद प्रपत्र सिद्ध करें $ u \notin 2\pi\Z $:
        \[
        F_N(u) = \frac{1}{N}\,
        \frac{\sin^2\bigl(\frac{Nu}{2}\bigr)}
        {\sin^2\bigl(\frac u2\bigr)} \;\geq\; 0
        \]
        \emph{(\omterm{def:b2:quadratic:adjoint}{जोड़} $\sin\bigl((n+\frac12)u\bigr)$ के रूप में
        एक ज्यामितीय श्रृंखला का काल्पनिक भाग)}.
  \item एकाग्रता अनुमान दिखाएँ: के लिए $0 < \delta \leq
        \abs u \leq \pi $,
        \[
        F_N(u) \leq \frac{1}{N\sin^2\frac\delta2}
        \xrightarrow[N\to\infty]{} 0
        \quad\text{uniformly} :
        \]
        $(F_N)$ एक सकारात्मक अनुमानित पहचान है.
  \item (Fej\'एर का प्रमेय) सिद्ध करें: यदि $ f $ है \omterm{def:b2:metric:continuity}{निरंतर} और
        $2\pi $-आवधिक, फिर $\sigma_N(f) \to f $
        \emph{\omterm{def:b2:funcseq:def}{समान रूप से}} पर $\R $ \emph{(के अभिन्न को विभाजित करें
        $\bigl(f(x+u) - f(x)\bigr)F_N(u)$ पर $\abs u =
        \delta $; हेन और प्रश्न 1--3 का उपयोग करें)}.
  \item टुकड़े-टुकड़े के लिए \omterm{def:b2:metric:continuity}{निरंतर} $ f $, दिखाएं \omterm{def:b2:funcseq:def}{बिंदुवार}
        संस्करण $\sigma_N(f)(x) \to \frac{f(x^+) +
        f(x^-)}{2}$ प्रत्येक पर $ x $, और वर्दी बाध्य
        $\norm{\sigma_N(f)}_\infty \leq \norm f_\infty $
        \emph{(सकारात्मकता!)}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Weierstrass, uniqueness,
Parseval.}
\begin{enumerate}[resume]
  \item (त्रिकोणमितीय वीयरस्ट्रैस) निष्कर्ष निकालें: प्रत्येक \omterm{def:b2:metric:continuity}{निरंतर}
        $2\pi $-आवधिक फलन की एक समान सीमा है
        त्रिकोणमितीय बहुपद.
  \item (विशिष्टता) निष्कर्ष: a \omterm{def:b2:metric:continuity}{निरंतर} $ f $ साथ $ c_n(f) = 0$
        सभी के लिए $ n $ \omterm{def:b2:funcseq:def}{समान रूप से} शून्य है --- दो \omterm{def:b2:metric:continuity}{निरंतर}
        उसी के साथ आवधिक कार्य \omterm{def:b2:fourier:coefficients}{फूरियर गुणांक}
        संयोग (\cref{exo:b2:fourier:4}, अब नहीं के साथ
        प्रवेश)।
  \item (पारसेवल, \omterm{def:b2:metric:continuity}{निरंतर} केस) प्रक्षेपण का उपयोग करना
        की संपत्ति $ S_N $ (\cref{prop:b2:fourier:bessel}) और
        $\sigma_Nf \in \mathcal T_{N-1} \subseteq \mathcal
        T_N $, सिद्ध करना
        \[
        \norm{f - S_Nf}_2 \leq \norm{f - \sigma_Nf}_2
        \leq \norm{f - \sigma_Nf}_\infty
        \xrightarrow[N\to\infty]{} 0 ,
        \]
        और प्रत्येक के लिए \omterm{thm:b2:fourier:parseval}{पारसेवल की पहचान} का निष्कर्ष निकालें
        \emph{\omterm{def:b2:metric:continuity}{निरंतर}} $2\pi $-आवधिक $ f $.
  \item (पारसेवल, टुकड़े-टुकड़े में \omterm{def:b2:metric:continuity}{निरंतर} केस) दिया गया $ f $
        टुकड़े-टुकड़े में \omterm{def:b2:metric:continuity}{निरंतर} और $\varepsilon > 0$, निर्माण ए
        \omterm{def:b2:metric:continuity}{निरंतर} आवधिक $ g $ साथ $\norm{f - g}_2 \leq
        \varepsilon $ \emph{(प्रतिस्थापित करें $ f $ एक एफ़िन द्वारा
        छलांग के चारों ओर छोटे अंतराल पर प्रक्षेप)},
        और निष्कर्ष निकालें $\norm{f - S_Nf}_2 \to 0$
        \emph{(बेसेल का उपयोग करें: $\norm{S_Nh}_2 \leq \norm h_2$)}:
        पार्सेवल में बताई गई पूर्ण व्यापकता में है
        \cref{thm:b2:fourier:parseval} --- ``भर्ती''
        चला गया.
  \item (फेज के लिए कोई गिब्स नहीं\'एर) के साथ तुलना करें
        \cref{exo:b2:fourier:8}: वर्गाकार तरंग के लिए दिखाएँ
        $ f $, $\abs{\sigma_N(f)} \leq 1$ हर जगह, हर किसी के लिए
        $ N $ --- सेस\`एआरओ औसत उस ओवरशूट को मिटा देता है
        सताता है $ S_N $. एक वाक्य में बताएं कि कौन सा गुण है
        का $ F_N $ जिम्मेदार है.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Rates.}
\begin{enumerate}[resume]
  \item इसके लिए दो कर्नेल सीमाएँ सिद्ध करें $0 < \abs u \leq
        \pi $:
        \[
        F_N(u) \leq N,
        \qquad
        F_N(u) \leq \frac{\pi^2}{N u^2}
        \]
        \emph{(पहले के लिए, $\abs{\sin N\theta} \leq
        N\abs{\sin\theta}$ प्रेरण द्वारा; दूसरे के लिए,
        $\sin\frac u2 \geq \frac{u}{\pi}$ पर
        $\intcc{0}{\pi}$)}.
  \item पहले क्षण का अनुमान निकालें
        \[
        \frac{1}{2\pi}\int_{-\pi}^{\pi}\abs u\,F_N(u)\,\dd u
        \;\leq\; \frac{C\,\ln N}{N}
        \qquad (N \geq 2)
        \]
        एक स्पष्ट स्थिरांक के लिए \emph{(पर विभाजित $\abs u =
        \frac1N $)}.
  \item निष्कर्ष: यदि $ f $ है $ L $-लिप्सचिट्ज़ और $2\pi $-आवधिक,
        फिर
        \[
        \norm{\sigma_N f - f}_\infty \leq \frac{C\,L\ln N}{N} .
        \]
  \item (संतृप्ति) गणना $\sigma_N(e_1)$ के लिए $ e_1(t) =
        \eu^{\iu t}$ और दिखाओ $\norm{\sigma_Ne_1 - e_1}_\infty
        = \frac1N $: यहां तक कि सबसे आसान कार्यों के लिए भी, Fej\'एर
        से अधिक तेजी से \omterm{def:b2:integration:improper}{अभिसरण} नहीं होता है $\frac1N $ --- सटीक
        में बर्नस्टीन संतृप्ति का एनालॉग
        फ़ंक्शन-अनुक्रम अध्याय की सप्ताहांत समस्या।
  \item (स्थानीयकरण) दिखाएँ: यदि $ f $ (टुकड़े-टुकड़े में \omterm{def:b2:metric:continuity}{निरंतर})
        पर गायब हो जाता है $\intoo{x - \delta}{x + \delta}$, तब
        $\sigma_N(f)(x) \to 0$, चाहे कितना भी जंगली हो $ f $ अन्यत्र है
        --- साधनों का \omterm{def:b2:integration:improper}{अभिसरण} $ x $ केवल देखता है $ f $ पास में
        $ x $.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Weyl's equidistribution theorem.}
एक क्रम $(x_n)_{n\geq1}$ में $\intco{0}{1}$ है
\emph{\omterm{def:b2:funcseq:def}{समान रूप से} वितरित} कब, प्रत्येक अंतराल के लिए
$\intcc{a}{b} \subseteq \intcc{0}{1}$,
\[
\frac{\#\{n \leq N : x_n \in \intcc ab\}}{N}
\xrightarrow[N\to\infty]{} b - a .
\]
\begin{enumerate}[resume]
  \item बताते हैं कि $(x_n)$ जैसे ही \omterm{def:b2:funcseq:def}{समान रूप से} वितरित किया जाता है
        $\frac1N\sum_{n\leq N}f(x_n) \to \int_0^1 f $ हरएक के लिए
        \emph{\omterm{def:b2:metric:continuity}{निरंतर}} $1$-आवधिक $ f $ \emph{(निचोड़ें
        का सूचक $\intcc ab $ दो के बीच में \omterm{def:b2:metric:continuity}{निरंतर}
        टुकड़े-टुकड़े-एफ़िन फ़ंक्शंस जिनके इंटीग्रल अलग-अलग होते हैं
        $\varepsilon $)}.
  \item (वेइल की कसौटी, पर्याप्तता) मान लीजिए
        \[
        \frac{1}{N}\sum_{n=1}^{N}\eu^{2\iu\pi kx_n}
        \xrightarrow[N\to\infty]{} 0
        \qquad\text{for every } k \in \Z\setminus\{0\} .
        \]
        दिखाओ $\frac1N\sum f(x_n) \to \int_0^1f $ पहले के लिए
        त्रिकोणमितीय बहुपद, फिर सभी के लिए \omterm{def:b2:metric:continuity}{निरंतर}
        $1$-आवधिक $ f $ प्रश्न 6 द्वारा (अवधि में ले जाया गया
        $1$): प्रश्न 16 के साथ, $(x_n)$ \omterm{def:b2:funcseq:def}{समान रूप से} वितरित है.
  \item होने देना $\alpha $ तर्कहीन हो और $ x_n = \{n\alpha\}$
        (आंशिक हिस्सा)। ज्यामितीय योग से बंधे
        \[
        \Bigl|\sum_{n=1}^{N}\eu^{2\iu\pi kn\alpha}\Bigr|
        \leq \frac{2}{\abs{1 - \eu^{2\iu\pi k\alpha}}}
        \qquad (k \neq 0),
        \]
        और निष्कर्ष निकालो \emph{वेइल का प्रमेय}: $(\{n\alpha\})$ है
        में \omterm{def:b2:funcseq:def}{समान रूप से} वितरित $\intco{0}{1}$.
  \item उसका निष्कर्ष निकालो $(\{n\alpha\})$ में सघन है
        $\intcc{0}{1}$ तर्कहीन के लिए $\alpha $, और इसमें समझाएं
        एक वाक्य कि \omterm{pb:b2:fourier:1}{समान वितरण} सख्ती से मजबूत क्यों है
        घनत्व से.
  \item (अग्रणी अंक) सिद्ध करें कि पूर्णांकों का अनुपात
        $ n \leq N $ ऐसा है कि $2^n $ अग्रणी (दशमलव) अंक है
        $1$ आदत है $\log_{10}2 \approx 0.301$ \emph{(अग्रणी
        अंक $1$ मतलब $\{n\log_{10}2\} \in
        \intco{0}{\log_{10}2}$; दिखाओ $\log_{10}2$ है
        तर्कहीन)}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The isoperimetric inequality.} होने देना
$\Gamma $ एक बंद सरल हो $ C^1$ लंबाई का वक्र $ L $ घेरना
एक \omterm{thm:b2:structures:signature}{हस्ताक्षर}ित क्षेत्र $ A $, स्केल्ड चाप लंबाई द्वारा पैरामीट्रिज्ड: $ z(t) =
x(t) + \iu y(t)$, $2\pi $-आवधिक, साथ $\abs{z'(t)} =
\frac{L}{2\pi}$ स्थिर; संलग्न क्षेत्र है
\[
A = \frac12\int_0^{2\pi}\bigl(x\,y' - y\,x'\bigr)\dd t
= \frac{1}{2}\,\Im\int_0^{2\pi}\conj{z}\,z'\,\dd t
\]
(यहां \omterm{thm:b2:structures:signature}{हस्ताक्षर}ित क्षेत्र की परिभाषा के रूप में लिया गया है; अध्याय
एकाधिक अभिन्नों पर यह साबित होता है कि यह सहज ज्ञान से सहमत है,
ग्रीन के सूत्र के माध्यम से)।
\begin{enumerate}[resume]
  \item बढ़ाना $ z(t) = \sum_{n\in\Z}c_n\eu^{\iu nt}$ (द
        ए की श्रृंखला $ C^1$ समारोह, \omterm{def:b2:funcseq:series}{सामान्यतः} \omterm{def:b2:integration:improper}{अभिसरण}) और
        साबित करें, पार्सेवल द्वारा लागू किया गया $ z'$:
        \[
        \frac{L^2}{2\pi} = \int_0^{2\pi}\abs{z'}^2\dd t
        = 2\pi\sum_{n\in\Z}n^2\abs{c_n}^2 .
        \]
  \item वैसे ही सिद्ध करो $ A = \pi\sum_{n\in\Z}
        n\,\abs{c_n}^2$ \emph{(पारसेवल अपने ध्रुवीकृत रूप में:
        $\frac{1}{2\pi}\int\conj f g = \sum
        \conj{c_n(f)}c_n(g)$, के लिए आवेदन किया $ f = z $, $ g = z'$)}.
  \item (हर्विट्ज़) निष्कर्ष:
        \[
        L^2 - 4\pi A = 4\pi^2\sum_{n\in\Z}
        (n^2 - n)\abs{c_n}^2 \;\geq\; 0 ,
        \]
        समानता के साथ यदि और केवल यदि $ z(t) = c_0 +
        c_1\eu^{\iu t}$ --- एक \omterm{def:b2:structures:sn}{चक्र}। आइसोपेरिमेट्रिक
        असमानता: लंबाई के बंद वक्रों के बीच $ L $, केवल
        वृत्त क्षेत्र को घेरता है $\frac{L^2}{4\pi}$.
  \item विवेक जांच: सर्कल के लिए समानता सत्यापित करें
        त्रिज्या $ R $ और के वर्ग के लिए सख्त असमानता
        ओर $ a $; समझाइए क्यों $ n^2 - n \geq 0$ प्रत्येक के लिए
        पूर्णांक $ n $, जिसमें नकारात्मक भी शामिल हैं, और कहां
        पैरामीट्रिजेशन की स्थिर गति का उपयोग किया गया था।
  \item संश्लेषण। प्रत्येक एक वाक्य में: (i) एकल
        की संपत्ति $ F_N $ जिसमें से भाग I--III प्रवाहित होता है, और
        जो $ D_N $ अभाव; (ii) कैसे सेस\`एआरओ सारांश यहाँ
        पावर-सीरीज़ चैप्टर की सप्ताहांत समस्या से संबंधित है
        (फ्रोबेनियस); (iii) केवल किस लाभांश का उपयोग किया गया
        वीयरस्ट्रैस (प्रश्न 6) और जिसे पूर्ण की आवश्यकता थी
        पारसेवल; (iv) वर्ष 3 खंड पर एक वाक्य
        जोड़ता है ($ L^2$ पूर्णता: हिल्बर्ट के रूप में फूरियर श्रृंखला
        आधार)।
\end{enumerate}
\end{problem}
