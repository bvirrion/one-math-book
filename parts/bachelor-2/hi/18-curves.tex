\chapter{घटता}\label{ch:b2:curves}

ज्यामिति अब भिन्न हो गई है। वक्र एक बिंदु है जिससे होकर घूम रहा है
स्थान; कैलकुलस हमें इसका वेग, त्वरण और ज्यामिति देता है
पूछता है कि आंतरिक क्या है --- इससे स्वतंत्र कि हम कितनी तेजी से पार करते हैं
प्रक्षेपवक्र. उत्तर हैं \emph{वक्राकार \omterm{def:b2:curves:length}{लंबाई}}, जो मापता है
प्रक्षेपवक्र ही, और \emph{\omterm{thm:b2:curves:frenet2d}{वक्रता}}, जो मापता है कि यह कैसे है
झुकता है. आयाम में $3$ एक दूसरा अपरिवर्तनीय, \emph{\omterm{thm:b2:curves:frenet3d}{टोशन}},
मापता है कि वक्र अपने तल से कैसे मुड़ता है। बहीखाता
इन सबके लिए उपकरण गतिशील है \emph{फ़्रेनेट फ़्रेम}.

\section{पैरामीट्रिज्ड चाप}

\begin{definition}[पैरामीट्रिज्ड चाप]\label{def:b2:curves:arc}
ए \emph{पैरामीट्रिज्ड चाप}\index{पैरामीट्रिज्ड चाप} कक्षा का
$\mathcal{C}^k $ ($ k \geq 1$) है
एक नक्शा $\gamma \colon I \to \R^n $ कक्षा का $\mathcal{C}^k $ एक पर
\omterm{def:b2:diffcalc:differential}{अंतर}ाल $ I $. एक बिंदु $\gamma(t)$ है \emph{नियमित} अगर
$\gamma'(t) \neq 0$, और चाप है \emph{नियमित} यदि यह सब
अंक हैं. के माध्यम से लाइन $\gamma(t)$ निर्देशक $\gamma'(t)$
है \emph{स्पर्श रेखा}\index{स्पर्श रेखा} एक नियमित बिंदु पर.
\end{definition}

\begin{definition}[पैरामीटर का परिवर्तन]\label{def:b2:curves:reparam}
ए \emph{पैरामीटर का परिवर्तन}\index{पैरामीटर का परिवर्तन} कक्षा का
$\mathcal{C}^k $ एक है
$\mathcal{C}^k $ भिन्नता $\theta \colon J \to I $ के बीच
\omterm{def:b2:diffcalc:differential}{अंतर}ाल ($\theta' \neq 0$ हर जगह)। चाप $\gamma $ और
$\gamma \circ \theta $ कहा जाता है \emph{\omterm{def:b2:nvs:equivalent}{समकक्ष}}; ए
\emph{ज्यामितीय चाप}\index{ज्यामितीय चाप} (या वक्र) एक है
तुल्यता वर्ग. धारणाएँ
पैरामीटर के परिवर्तन के तहत अपरिवर्तनीय --- प्रक्षेपवक्र, \omterm{def:b2:curves:arc}{स्पर्श रेखा}, वक्राकार \omterm{def:b2:curves:length}{लंबाई}, \omterm{thm:b2:curves:frenet2d}{वक्रता} --- कहा जाता है \emph{ज्यामितिक}.
\end{definition}

\begin{remark}
केवल प्रक्षेप पथ ही इसका निर्धारण नहीं करता है \omterm{def:b2:curves:reparam}{ज्यामितीय चाप}: द
पैरामीट्रिज़ेशन $ t \mapsto (\cos t, \sin t)$ पर $[0, 2\pi]$ और पर
$[0, 4\pi]$ एक ही छवि है लेकिन एक बार वृत्त की यात्रा करें
दो बार. ए \omterm{def:b2:curves:reparam}{ज्यामितीय चाप} बहुलता और अभिविन्यास को याद रखता है
ट्रैवर्सल का, उसकी गति का नहीं।
\end{remark}

\begin{example}[गति से कुछ नहीं बदलता
ज्यामितीय]\label{ex:b2:curves:speedinvariance}
यूनिट सर्कल को पैरामीट्रिज करें
\[
\gamma(t) = (\cos t^2,\ \sin t^2),
\qquad t \in \intcc0{\sqrt{2\pi}} .
\]
गति $\norm{\gamma'(t)} = 2t $ फिर भी, रैखिक रूप से बढ़ता है
\[
L = \int_0^{\sqrt{2\pi}}2t\,\dd t = 2\pi ,
\]
जो उसी \omterm{def:b2:curves:length}{लंबाई} स्थिर गति के रूप में --- जैसे
\cref{thm:b2:curves:lengthinv} वादे, के माध्यम से का परिवर्तन
पैरामीटर $ t \mapsto t^2$. \omterm{def:b2:curves:arc}{स्पर्श रेखा}, द \omterm{thm:b2:curves:frenet2d}{वक्रता}
से गणना की गई \cref{prop:b2:curves:kappaformula}, और हर
अन्य ज्यामितीय मात्राएँ भी सहमत हैं; केवल $ t = 0$
एक नज़र डालने लायक है, कहाँ $\gamma'(0) = 0$ बनाता है \emph{यह
पैरामीट्रिजेशन} यद्यपि प्रक्षेप पथ अनियमित है
पूर्ण \omterm{def:b2:structures:sn}{चक्र}. ज्यामितीय कथन बुरे को सहन करते हैं
पैरामीट्रिजेशन बुरी तरह से: पहले पुनः पैरामीट्रिजेशन करें, निष्कर्ष निकालें
दूसरा.
\end{example}

\begin{example}\label{ex:b2:curves:cusp}
चाप $\gamma(t) = (t^2, t^3)$ है $\mathcal{C}^\infty $ लेकिन नहीं
नियमित: $\gamma'(0) = (0, 0)$. इसका प्रक्षेप पथ अर्धघनाकार है
परवलय $ y^2 = x^3$, एक \emph{उभार} मूल में: चिकनाई
पैरामीट्रिज़ेशन एक ज्यामितीय विलक्षणता को नहीं रोकता है
जहां वेग लुप्त हो जाता है. यही कारण है कि नियमितता परिकल्पना
$\gamma' \neq 0$ कॉस्मेटिक नहीं है.
\end{example}

\section{वक्राकार लंबाई}

\begin{definition}[वक्राकार लंबाई]\label{def:b2:curves:length}
होने देना $\gamma \colon [a, b] \to \R^n $ एक हो $\mathcal{C}^1$ चाप. इसका
\emph{लंबाई}\index{लंबाई} है
\[
L(\gamma) = \int_a^b \norm{\gamma'(t)}\, \dd t ,
\]
कहाँ $\norm{\cdot}$ यूक्लिडियन है आदर्श. \emph{चाप-लंबाई
समारोह}\index{चाप-लंबाई
समारोह} पर आधारित $ t_0$ है $ s(t) = \int_{t_0}^t
\norm{\gamma'(u)}\,\dd u $.
\end{definition}

\begin{theorem}[लंबाई ज्यामितीय है; बहुभुज
लक्षण वर्णन]\label{thm:b2:curves:lengthinv}
\leavevmode
\begin{enumerate}
\item अगर $\theta \colon [c, d] \to [a, b]$ एक है $\mathcal{C}^1$
\omterm{def:b2:curves:reparam}{पैरामीटर का परिवर्तन}, तब $ L(\gamma \circ \theta) = L(\gamma)$.
\item $ L(\gamma)$ का सर्वोच्च है \omterm{def:b2:curves:length}{लंबाई} अंकित का
बहुभुज:
\[
L(\gamma)
= \sup\Bigl\{\, \sum_{i=1}^{m} \norm{\gamma(t_i) -
\gamma(t_{i-1})} \;:\; a = t_0 < t_1 < \dots < t_m = b \,\Bigr\}.
\]
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} चरों के परिवर्तन से $ t = \theta(u)$ (वर्ष 1
मात्रा, तब से वैध है $\theta $ है $\mathcal{C}^1$ मोनोटोन),
\[
\int_c^d \norm{(\gamma\circ\theta)'(u)}\,\dd u
= \int_c^d \norm{\gamma'(\theta(u))}\,\abs{\theta'(u)}\,\dd u
= \int_a^b \norm{\gamma'(t)}\,\dd t ,
\]
हमने कहां उपयोग किया $(\gamma\circ\theta)' = \theta'\cdot
(\gamma'\circ\theta)$ और, यदि $\theta $ घट रहा है, का संकेत
$\theta'$ सीमा के उलटने से अवशोषित हो जाता है।

\emph{2.} किसी भी उपविभाग के लिए, $\gamma(t_i) - \gamma(t_{i-1}) =
\int_{t_{i-1}}^{t_i}\gamma'(t)\,\dd t $, तो त्रिकोण द्वारा
अभिन्नों के लिए असमानता $\norm{\gamma(t_i) - \gamma(t_{i-1})}
\leq \int_{t_{i-1}}^{t_i}\norm{\gamma'}$: प्रत्येक बहुभुज छोटा है
से $ L(\gamma)$, इस तरह $\sup \leq L(\gamma)$.

विपरीत असमानता के लिए, आइए $\varepsilon > 0$. तब से $\gamma'$
है \omterm{def:b2:metric:continuity}{निरंतर} पर \omterm{def:b2:metric:compact}{कॉम्पैक्ट} $[a,b]$, यह \omterm{def:b2:funcseq:def}{समान रूप से} \omterm{def:b2:metric:continuity}{निरंतर} है:
वहाँ है $\delta > 0$ साथ $\norm{\gamma'(t) - \gamma'(u)} \leq
\varepsilon $ जब कभी भी $\abs{t - u} \leq \delta $. एक उपविभाग लीजिए
कदम का $\leq \delta $. प्रत्येक टुकड़े पर, के लिए $ t \in [t_{i-1}, t_i]$,
\[
\gamma(t_i) - \gamma(t_{i-1})
= \int_{t_{i-1}}^{t_i}\gamma'(t)\,\dd t
= (t_i - t_{i-1})\,\gamma'(t_{i-1}) + R_i,
\qquad
\norm{R_i} \leq \varepsilon\,(t_i - t_{i-1}),
\]
तब से $ R_i = \int_{t_{i-1}}^{t_i}(\gamma'(t) -
\gamma'(t_{i-1}))\,\dd t $. इस तरह
\[
\norm{\gamma(t_i) - \gamma(t_{i-1})}
\geq (t_i - t_{i-1})\norm{\gamma'(t_{i-1})}
- \varepsilon (t_i - t_{i-1}) .
\]
संक्षेप करना, और तुलना करना $\sum (t_i -
t_{i-1})\norm{\gamma'(t_{i-1})}$ साथ $\int_a^b\norm{\gamma'}$ (ए
रीमैन का योग \omterm{def:b2:metric:continuity}{निरंतर} समारोह $\norm{\gamma'}$, अंदर
$\varepsilon(b - a)$ के लिए अभिन्न का $\delta $ द्वारा काफी छोटा
वर्दी निरंतरता पुनः), हमें एक बहुभुज मिलता है \omterm{def:b2:curves:length}{लंबाई} $\geq
L(\gamma) - 2\varepsilon(b - a)$. दे $\varepsilon \to 0$
दावा साबित करता है.
\end{proof}

\begin{example}[आर्किमिडीज़ और उत्कीर्ण
बहुभुज]\label{ex:b2:curves:archimedes}
यूनिट सर्कल के लिए, नियमित रूप से अंकित $ n $-गॉन के पास है \omterm{def:b2:curves:length}{लंबाई}
$ L_n = 2n\sin\frac\pi n $, और विस्तार $\sin x = x -
\frac{x^3}6 + O(x^5)$ देता है
\[
L_n = 2\pi - \frac{\pi^3}{3n^2} + O\Bigl(\frac1{n^4}\Bigr) :
\]
बहुभुज \omterm{def:b2:curves:length}{लंबाई} का \cref{thm:b2:curves:lengthinv}
एकाग्र \emph{चतुर्भुज रूप से}. संख्यात्मक रूप से: $ L_6 = 6$
(षट्कोण, क्रूड दे रहा है $\pi > 3$), जबकि $ L_{96} =
192\sin\frac{\pi}{96} \approx 6.28206$ ख़िलाफ़ $2\pi \approx
6.28319$ ---त्रुटि $0.00113$ भविष्यवाणी से सहमत हैं
$\pi^3/(3\cdot96^2) \approx 0.00112$. यही कारण है कि आर्किमिडीज़,
षट्भुज को पांच बार दोगुना करना $96$ पक्ष, ब्रैकेट कर सकते हैं
$\pi $ हाथ से तीन अंकों तक: प्रत्येक दोहरीकरण विभाजित करता है
चार की त्रुटि. बहुभुज में सर्वोच्च
लक्षण वर्णन केवल सीमा में ही प्राप्त नहीं होता; यह है
हासिल किया \emph{तेज़}, क्योंकि एक चिकना वक्र अलग हो जाता है
इसके तार केवल दूसरे क्रम पर हैं।
\end{example}

\begin{theorem}[चाप-लंबाई पैरामीट्रिजेशन]\label{thm:b2:curves:arclength}
होने देना $\gamma \colon I \to \R^n $ नियमित रहो $\mathcal{C}^k $ चाप
($ k \geq 1$). चा\omterm{def:b2:curves:length}{प-लंबाई} फ़ंक्शन $ s $ एक है $\mathcal{C}^k $
भिन्नतारूपता से $ I $ एक \omterm{def:b2:diffcalc:differential}{अंतर}ाल पर $ J $, और $\tilde\gamma =
\gamma \circ s^{-1}$ संतुष्ट $\norm{\tilde\gamma'} = 1$
हर जगह. पैरामीटर और ओरिएंटेशन के अनुवाद तक,
यह \emph{वक्राकार \omterm{def:b2:curves:length}{लंबाई}} (या \emph{इकाई गति}) पैरामीट्रिजेशन है
अद्वितीय.
\end{theorem}

\begin{proof}
$ s'(t) = \norm{\gamma'(t)} > 0$ और $ s'$ है $\mathcal{C}^{k-1}$
(की रचना $\mathcal{C}^{k-1}$ नक्शा $\gamma'$ साथ आदर्श,
से दूर चिकना $0$), इसलिए $ s $ है $\mathcal{C}^k $, सख्ती से
बढ़ रहा है, पर एक आक्षेप $ J = s(I)$, और इसका उलटा है
$\mathcal{C}^k $ आयाम में व्युत्क्रम फलन प्रमेय द्वारा $1$
(वर्ष 1 खंड). तब
\[
\tilde\gamma'(\sigma)
= \frac{\gamma'(t)}{s'(t)}
= \frac{\gamma'(t)}{\norm{\gamma'(t)}},
\qquad t = s^{-1}(\sigma),
\]
एक इकाई वेक्टर. अगर $\hat\gamma = \gamma\circ\theta $ दूसरा है
यूनिट-स्पीड पैरामीट्रिजेशन, फिर $\abs{\theta'} = 1$, इसलिए $\theta'
= \pm 1$ स्थिर (निरंतरता), यानी\ $\theta(u) = \pm u + c $.
\end{proof}

\begin{remark}
वक्राकार \omterm{def:b2:curves:length}{लंबाई} वह पैरामीटर है जो ज्यामिति को अलग करता है
गतिकी। प्रक्षेप पथ को किसी भी गति से पार किया जा सकता है
प्रोफ़ाइल --- गति की भौतिकी --- लेकिन हर
पैरामीट्रिजेशन-अपरिवर्तनीय प्रश्न (आकार, झुकना,
ऑस्क्यूलेशन) में एक विहित घड़ी होती है, जो दूरी तय की जाती है।
यही कारण है कि सभी \omterm{thm:b2:curves:frenet2d}{वक्रता} नीचे दिए गए सूत्र हैं \emph{परिभाषित}
इकाई गति से और फिर \emph{अनुवाद} मनमाना करने के लिए
श्रृंखला नियम द्वारा पैरामीट्रिज़ेशन: अनुवाद कारक
की शक्तियां हैं $ v = s'$, और उन्हें सही ढंग से ट्रैक करना है
की संपूर्ण सामग्री \cref{prop:b2:curves:kappaformula}.
\end{remark}

\begin{example}[वृत्त और हेलिक्स]\label{ex:b2:curves:helix}
सर्कल के लिए $\gamma(t) = (R\cos t, R\sin t)$, $\norm{\gamma'} =
R $, इसलिए $ s = Rt $ और यह \omterm{def:b2:curves:length}{लंबाई} का एक पूर्ण मोड़ है $2\pi R $. के लिए
हेलिक्स $\gamma(t) = (a\cos t,\ a\sin t,\ bt)$ साथ $ a > 0$,
$\norm{\gamma'(t)} = \sqrt{a^2 + b^2}$ स्थिर है: हेलिक्स है
\omterm{def:b2:metric:continuity}{निरंतर} गति से पार किया गया, और $ s = t\sqrt{a^2 + b^2}$.
\end{example}

\begin{example}[ध्रुवीय में चाप की लंबाई
निर्देशांक]\label{ex:b2:curves:polarlength}
एक ध्रुवीय वक्र $ r = r(\theta)$ चाप है $\gamma(\theta) =
(r\cos\theta,\ r\sin\theta)$, साथ
\[
\gamma'(\theta) = (r'\cos\theta - r\sin\theta,\
r'\sin\theta + r\cos\theta),
\qquad
\norm{\gamma'(\theta)}^2 = r'^2 + r^2
\]
(क्रॉस शर्तें रद्द करें): ध्रुवीय \omterm{def:b2:curves:length}{लंबाई} तत्व है
$\sqrt{r^2 + r'^2}\,\dd\theta $. कार्डियोइड के लिए $ r = 1 +
\cos\theta $:
\[
r^2 + r'^2 = (1 + \cos\theta)^2 + \sin^2\theta = 2 +
2\cos\theta = 4\cos^2\tfrac\theta2,
\]
और पर $\intcc{-\pi}{\pi}$ अर्धकोण कोज्या है
गैर-नकारात्मक, इसलिए
\[
L = \int_{-\pi}^{\pi}2\cos\tfrac\theta2\,\dd\theta
= \Bigl[4\sin\tfrac\theta2\Bigr]_{-\pi}^{\pi}
= 4 - (-4) = 8 :
\]
के साइक्लोइड आर्क की तरह \cref{exo:b2:curves:1}, एक वक्र
वृत्तों से निर्मित एक तर्कसंगत है \omterm{def:b2:curves:length}{लंबाई}, नहीं के साथ $\pi $
कहीं भी. अर्ध-कोण गुणनखंडन मानक चाल है
के लिए \omterm{def:b2:curves:length}{लंबाई} वृत्त-जनित वक्रों का; जब यह विफल हो जाता है (
दीर्घवृत्त), द \omterm{def:b2:curves:length}{लंबाई} वास्तव में एक नया कार्य है --- an
प्राथमिक बंद रूपों से परे, अण्डाकार अभिन्न।
\end{example}

\section{समतल में वक्रता}

इस पूरे खंड में, चाप हैं $\mathcal{C}^2$ और नियमित रूप से
उन्मुख यूक्लिडियन विमान. हम द्वारा पैरामीटराइज़ करते हैं वक्राकार \omterm{def:b2:curves:length}{लंबाई} और लिखा $ T(s) = \tilde\gamma'(s)$ इकाई स्पर्शरेखा के लिए, और $ N(s)$
यूनिट वेक्टर के लिए सीधे ओर्थोगोनल $ T(s)$ (का घूर्णन
$ T $ द्वारा $+\pi/2$).

\begin{theorem}[समतल फ्रेनेट सूत्र]\label{thm:b2:curves:frenet2d}
होने देना $\tilde\gamma $ एक इकाई-गति हो $\mathcal{C}^2$ में चाप
उन्मुख विमान. वहां एक है \omterm{def:b2:metric:continuity}{निरंतर} समारोह $\kappa $, द
(बीजगणितीय) \emph{वक्रता}\index{वक्रता}, ऐसा है कि
\[
T'(s) = \kappa(s)\, N(s),
\qquad
N'(s) = -\kappa(s)\, T(s).
\]
\end{theorem}

\begin{proof}
तब से $\norm{T(s)}^2 = 1$ सभी के लिए $ s $, अदिश को विभेदित करना
उत्पाद देता है $2\langle T'(s), T(s)\rangle = 0$: $ T'(s)$ के लिए ओर्थोगोनल है
$ T(s)$, इसलिए के साथ संरेख $ N(s)$ (आयाम $2$); लिखना $ T'(s)
= \kappa(s) N(s)$ साथ $\kappa(s) = \langle T'(s), N(s)\rangle $,
\omterm{def:b2:metric:continuity}{निरंतर}. वैसे ही $ N' \perp N $, इसलिए $ N' = \lambda T $; और
विभेद करना $\langle T, N\rangle = 0$ देता है $\langle T', N\rangle +
\langle T, N'\rangle = \kappa + \lambda = 0$.
\end{proof}

\begin{definition}\label{def:b2:curves:curvature}
कब $\kappa(s) \neq 0$, द \emph{वक्रता त्रिज्या}\index{वक्रता त्रिज्या} है $ R(s) =
1/\abs{\kappa(s)}$ और यह \emph{वक्रता का केंद्र}\index{वक्रता का केंद्र} है
$\tilde\gamma(s) + \frac{1}{\kappa(s)} N(s)$; उसके साथ वृत्त
केंद्र और त्रिज्या $ R(s)$ है \emph{ऑस्कुलेटिंग सर्कल}\index{ऑस्कुलेटिंग सर्कल}, द
वक्र का सर्वोत्तम गोलाकार सन्निकटन $\tilde\gamma(s)$.
\end{definition}

\begin{example}[का दोलन चक्र
घातीय]\label{ex:b2:curves:oscexp}
के लिए $ y = \eu^x $ बिंदु पर $(0, 1)$: $ f'(0) = f''(0) = 1$,
तो नीचे दिए गए ग्राफ़ सूत्र द्वारा,
\[
\kappa(0) = \frac{1}{(1 + 1)^{3/2}} = \frac1{2\sqrt2},
\qquad R = 2\sqrt2 .
\]
इकाई स्पर्शरेखा है $ T = \frac{(1, 1)}{\sqrt2}$, प्रत्यक्ष
सामान्य $ N = \frac{(-1, 1)}{\sqrt2}$, और यह का केंद्र
\omterm{thm:b2:curves:frenet2d}{वक्रता} है
\[
(0, 1) + 2\sqrt2\cdot\frac{(-1, 1)}{\sqrt2} = (-2,\ 3) :
\]
 \omterm{def:b2:curves:curvature}{ऑस्कुलेटिंग सर्कल} समीकरण है $(x + 2)^2 + (y - 3)^2 =
8$. ``सर्वोत्तम परिपत्र सन्निकटन'' दावे की जाँच के रूप में:
वृत्त के समीकरण को हल करना $ y $ पास में $(0,1)$ और
विस्तार देता है $ y = 1 + x + \frac{x^2}2 + O(x^3)$ ---
\omterm{def:b2:series:def}{बिल्कुल} दूसरे क्रम का टेलर विस्तार $\eu^x $.
\omterm{def:b2:curves:curvature}{ऑस्कुलेटिंग सर्कल} मान, ढलान से मेल खाता है \emph{और} दूसरा
व्युत्पन्न; एक साधारण स्पर्शरेखा वृत्त केवल से मेल खाएगा
पहले दो.
\end{example}

\begin{example}[एक वृत्त का विकासक्रम उसका है
केंद्र]\label{ex:b2:curves:circleevolute}
त्रिज्या के वृत्त के लिए $ R $ वामावर्त घुमाया,
$\kappa = 1/R $ और $ N $ केंद्र की ओर इंगित करता है, इसलिए
\omterm{def:b2:curves:curvature}{वक्रता का केंद्र} $\tilde\gamma + \frac1\kappa N $ है
वृत्त का केंद्र, प्रत्येक के लिए $ s $: द \omterm{def:b2:curves:curvature}{ऑस्कुलेटिंग सर्कल}
एक वृत्त का वृत्त ही होता है, और केंद्रों का स्थान होता है
का \omterm{thm:b2:curves:frenet2d}{वक्रता} एक बिंदु तक ढह जाता है. ये घटिया मामला
अंशांकन करता है \cref{exo:b2:curves:6}: वहाँ evolute है
वेग है $-\frac{\kappa'}{\kappa^2}N $, जो गायब हो जाता है
\omterm{def:b2:series:def}{बिल्कुल} ठीक कब $\kappa $ स्थिर है.
\end{example}

\begin{proposition}[मनमाने ढंग से वक्रता
पैरामीट्रिजेशन]\label{prop:b2:curves:kappaformula}
एक नियमित के लिए $\mathcal{C}^2$ समतल चाप $\gamma(t) = (x(t), y(t))$,
\[
\kappa(t)
= \frac{x'(t)\,y''(t) - y'(t)\,x''(t)}
       {\bigl(x'(t)^2 + y'(t)^2\bigr)^{3/2}} ,
\]
विशेष रूप से $\kappa = \dfrac{y''}{(1 + y'^2)^{3/2}}$ एक ग्राफ के लिए
$ y = f(x)$.
\end{proposition}

\begin{proof}
लिखना $ v(t) = \norm{\gamma'(t)} = s'(t)$, इसलिए $\gamma' = vT $
(यूनिट-स्पीड डेटा के साथ रचना करना $ s $). विभेद करना,
\[
\gamma'' = v'T + v\,T'\cdot s' = v'T + v^2\kappa N .
\]
अब लीजिए सिद्ध (विहित उन्मुख आधार में) का
$(\gamma', \gamma'')$: तब से $\det(T, T) = 0$ और $\det(T, N) = 1$,
\[
\det(\gamma', \gamma'') = \det(vT,\ v'T + v^2\kappa N)
= v^3\kappa .
\]
बाईं ओर है $ x'y'' - y'x''$, और $ v^3 = (x'^2 +
y'^2)^{3/2}$. ग्राफ़ केस पैरामीट्रिज़ेशन है $ t \mapsto (t,
f(t))$.
\end{proof}

\begin{example}[वृत्त, रेखा, परवलय]\label{ex:b2:curves:kappaexamples}
एक लाइन है $\kappa = 0$ (और इसके विपरीत: $ T' = 0$ मतलब $ T $
स्थिर, इसलिए $\tilde\gamma(s) = \tilde\gamma(0) + sT $, एक पंक्ति). द
त्रिज्या का वृत्त $ R $ वामावर्त घुमाया गया है $\kappa = 1/R $:
साथ $\gamma(t) = (R\cos t, R\sin t)$, सूत्र देता है $\kappa =
R^2/R^3$. परवलय के लिए $ y = x^2/2$: $\kappa(x) = 1/(1 +
x^2)^{3/2}$, शीर्ष पर अधिकतम --- परवलय सबसे तीव्र होता है
जहां मुड़ता है वहीं झुक जाता है।
\end{example}

\begin{remark}[वक्रता के आसपास आम नुकसान]
(i) बीजगणितीय \omterm{thm:b2:curves:frenet2d}{वक्रता} एक समतल चाप का चिन्ह कब बदलता है?
चाप या समतल का अभिविन्यास उलटा है: केवल
$\abs\kappa $ और $ R = 1/\abs\kappa $ विशुद्ध रूप से ज्यामितीय हैं. ए
वृत्त को दक्षिणावर्त दिशा में घुमाया गया है $\kappa = -1/R $. (ii) द
ग्राफ़ सूत्र $\kappa = f''/(1 + f'^2)^{3/2}$ चुपचाप
द्वारा पैरामीट्रिजेशन चुनता है $ x $; इसे एक वक्र पर लागू करना
वह बिंदु (ऊर्ध्वाधर स्पर्शरेखा) के निकट ग्राफ़ नहीं है
शास्त्रीय भूल. (iii) एक बिंदु पर जहां $\gamma' = 0$
कुछ भी परिभाषित नहीं है --- न ही $ T $ और न $\kappa $ --- और
प्रक्षेप पथ वास्तव में टूट सकता है
(\cref{ex:b2:curves:cusp}); हमेशा पहले नियमितता की जाँच करें
इकाई स्पर्शरेखा को विभेदित करना। (iv) \omterm{def:b2:diffcalc:differential}{अंतर}िक्ष में, $\kappa =
\norm{T'} \geq 0$ परंपरा के अनुसार: प्राप्त करने का कोई संकेत नहीं है
ग़लत, लेकिन शोषण का कोई संकेत भी नहीं --- विभक्ति-प्रकार
जानकारी अंदर चली जाती है \omterm{thm:b2:curves:frenet3d}{टोशन}. (v) अंततः, $\kappa $ है
एक व्युत्पन्न \emph{इसके संबंध में वक्राकार \omterm{def:b2:curves:length}{लंबाई}}: एक के लिए
गैर-इकाई-गति पैरामीट्रिजेशन, कारक को भूल जाना $ v^3$
में \cref{prop:b2:curves:kappaformula} सबसे अधिक बार होता है
व्यवहार में त्रुटि.
\end{remark}

\begin{theorem}[वक्रता वक्र का निर्धारण करती है]\label{thm:b2:curves:fundamental}
होने देना $\kappa \colon J \to \R $ होना \omterm{def:b2:metric:continuity}{निरंतर}. वहाँ एक मौजूद है
इकाई गति $\mathcal{C}^2$ विमान में चाप के साथ \omterm{thm:b2:curves:frenet2d}{वक्रता}
$\kappa $, और यह प्रत्यक्ष आइसोमेट्री (रोटेशन) तक अद्वितीय है
अनुवाद के बाद)।
\end{theorem}

\begin{proof}
\emph{अस्तित्व।} हल करना $ s_0 \in J $ और सेट करें $\varphi(s) =
\int_{s_0}^s \kappa(u)\,\dd u $, तब
\[
\tilde\gamma(s) = \Bigl(\int_{s_0}^s \cos\varphi(u)\,\dd u,\
\int_{s_0}^s \sin\varphi(u)\,\dd u\Bigr).
\]
तब $ T(s) = (\cos\varphi(s), \sin\varphi(s))$ एक इकाई वेक्टर है,
$ N(s) = (-\sin\varphi, \cos\varphi)$, और
\[
T'(s) = \varphi'(s)\,(-\sin\varphi, \cos\varphi)
= \kappa(s)\,N(s) :
\]
चाप इकाई-गति है \omterm{thm:b2:curves:frenet2d}{वक्रता} $\kappa $.

\emph{विशिष्टता.} होने देना $\gamma_1, \gamma_2$ यूनिट-स्पीड आर्क्स के साथ बनें
वही \omterm{thm:b2:curves:frenet2d}{वक्रता}. प्रत्येक इकाई स्पर्शरेखा एक कोण तक उठाती है
फ़ंक्शन, एक स्पष्ट निर्माण द्वारा: दृश्य $ T_j $ के रूप में
सम्मिश्र संख्या $ z_j = a_j + \iu b_j $ मापांक का $1$, चुनना
$\varphi_j(0)$ साथ $ z_j(0) = \eu^{\iu\varphi_j(0)}$, और सेट करें
\[
\varphi_j(s) = \varphi_j(0) + \int_0^s\det\bigl(T_j,
T_j'\bigr)(u)\,\dd u .
\]
से $\abs{z_j} = 1$: $\operatorname{Re}(\conj{z_j}\,z_j') =
0$, इसलिए $\conj{z_j}\,z_j' = \iu\det(T_j, T_j') =
\iu\,\varphi_j'$, यानी\ $ z_j' = \iu\varphi_j'z_j $; तब
\[
\bigl(z_j\,\eu^{-\iu\varphi_j}\bigr)'
= \eu^{-\iu\varphi_j}\bigl(z_j' - \iu\varphi_j'z_j\bigr) = 0,
\]
इसलिए $ z_j = \eu^{\iu\varphi_j}$ लगातार: $ T_j =
(\cos\varphi_j, \sin\varphi_j)$ साथ $\varphi_j $ कक्षा का
$\mathcal C^1$. इसके अतिरिक्त $\det(T_j, T_j') = \det(T_j,
\kappa N_j) = \kappa $, इसलिए $\varphi_j' = \kappa $. इस तरह $\varphi_2 = \varphi_1 + c $ एक के लिए
स्थिर $ c $: $ T_2$ है $ T_1$ निश्चित कोण से घुमाया जाता है $ c $, तो
एकीकृत करना, $\gamma_2 = \rho(\gamma_1) + w $ कहाँ $\rho $ है
कोण का घूमना $ c $ और $ w $ एक स्थिर सदिश.
\end{proof}

\begin{remark}
यह एक आयामी प्रोटोटाइप है \emph{मौलिक
ज्यामिति का प्रमेय}: ए \omterm{def:b2:metric:complete}{पूरा} स्थानीय अपरिवर्तनीयों का सेट (यहाँ, एक
फ़ंक्शन) वस्तु को कठोर गति तक वर्गीकृत करता है। द
नीचे दिए गए त्रि-आयामी संस्करण के लिए दो अपरिवर्तनीयों की आवश्यकता है।
\end{remark}

\begin{example}[निरंतर वक्रता का अर्थ है
वृत्त]\label{ex:b2:curves:constantkappa}
लेना $\kappa \equiv \kappa_0 > 0$ अस्तित्व सूत्र में:
$\varphi(s) = \kappa_0 s $ और
\[
\tilde\gamma(s) = \Bigl(\frac{\sin\kappa_0s}{\kappa_0},\
\frac{1 - \cos\kappa_0s}{\kappa_0}\Bigr) :
\]
त्रिज्या का \omterm{def:b2:structures:sn}{चक्र} $1/\kappa_0$ पर केन्द्रित $(0,
1/\kappa_0)$, इकाई गति से पार किया गया। विशिष्टता से
प्रमेय का आधा भाग, \emph{प्रत्येक} स्थिरांक की इकाई-गति चाप
\omterm{thm:b2:curves:frenet2d}{वक्रता} $\kappa_0$ त्रिज्या के एक वृत्त का एक टुकड़ा है
$1/\kappa_0$ (या एक पंक्ति यदि $\kappa_0 = 0$) ---विपरीत
में गणना का \cref{ex:b2:curves:kappaexamples}, और
का विमान मामला \cref{exo:b2:curves:9}.
\end{example}

\begin{example}[इससे एक वक्र का पुनर्निर्माण करना
वक्रता]\label{ex:b2:curves:reconstruction}
कौन सी इकाई-गति वक्र है \omterm{def:b2:curves:curvature}{वक्रता त्रिज्या} $ R(s) = 1 +
s^2$? अस्तित्व प्रमाण के साथ पालन करें $\kappa(s) =
\frac1{1+s^2}$ और $ s_0 = 0$: $\varphi(s) = \arctan s $, इसलिए
\[
T(s) = (\cos\arctan s,\ \sin\arctan s)
= \Bigl(\frac{1}{\sqrt{1+s^2}},\
\frac{s}{\sqrt{1+s^2}}\Bigr),
\]
और एकीकृत करना,
\[
\tilde\gamma(s) = \Bigl(\ln\bigl(s + \sqrt{1 + s^2}\bigr),\
\sqrt{1 + s^2} - 1\Bigr).
\]
सेटिंग $ x = \ln(s + \sqrt{1+s^2})$, यानी\ $ s = \sinh x $, द
दूसरा समन्वय है $\cosh x - 1$: वक्र है
\emph{ज़ंजीर का} $ y = \cosh x - 1$. इससे लूप बंद हो जाता है
\cref{exo:b2:curves:3}, जहां हमने गणना की $ R = \cosh^2 x = 1 +
\sinh^2 x = 1 + s^2$ सीधे: मौलिक प्रमेय
गारंटी देता है कि कैटेनरी है \emph{केवल} इसके साथ वक्र
\omterm{thm:b2:curves:frenet2d}{वक्रता} प्रोफ़ाइल, प्रत्यक्ष आइसोमेट्री तक।
\end{example}

\begin{remark}[जहां आगे वक्रता का प्रयोग किया जाता है]
विघटन $\gamma'' = v'T + v^2\kappa N $ में प्राप्त किया गया
का प्रमाण \cref{prop:b2:curves:kappaformula} है
प्रत्येक घुमावदार गति की गतिकी: स्पर्शरेखीय बनाम
केन्द्राभिमुख त्वरण. \omterm{thm:b2:curves:frenet2d}{वक्रता} सतहों के लिए रिटर्न
(\cref{ch:b2:surfaces}) के माध्यम से \omterm{thm:b2:curves:frenet2d}{वक्रता} खींचे गए वक्रों का
उन पर, और \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} इस अध्याय के सप्ताहांत की गणना
समस्या --- विकसित, कास्टिक --- की ज्यामितीय प्रकाशिकी है
वेवफ्रंट्स वर्ष 3 खंड का आंतरिक बिंदु लेता है
के उपमानों के लिए फिर से देखें $\R^n $.
\end{remark}

\section{अंतरिक्ष में फ़्रेनेट फ़्रेम}

अब चलो $\tilde\gamma \colon J \to \R^3$ एक इकाई-गति हो
$\mathcal{C}^3$ चाप वह है \emph{biregular}: $ T'(s) \neq 0$ सभी के लिए $ s $. तब $\kappa(s) = \norm{T'(s)} > 0$ को परिभाषित करता है
\emph{\omterm{thm:b2:curves:frenet2d}{वक्रता}} (\omterm{def:b2:diffcalc:differential}{अंतर}िक्ष में कोई चिह्न नहीं: कोई पसंदीदा सामान्य नहीं है
ओरिएंटेशन), और हम सेट करते हैं:
\[
N(s) = \frac{T'(s)}{\kappa(s)}
\quad\text{(principal normal)},
\qquad
B(s) = T(s) \wedge N(s)
\quad\text{(binormal)} ,
\]
ताकि $(T, N, B)$ एक सीधा ऑर्थोनॉर्मल फ्रेम है,
\emph{फ़्रेनेट फ़्रेम}. विमान के माध्यम से $\tilde\gamma(s)$ द्वारा फैलाया गया
$ T, N $ है \emph{ऑस्कुलेटिंग प्लेन}.

\begin{theorem}[अंतरिक्ष में फ़्रेनेट सूत्र]\label{thm:b2:curves:frenet3d}
वहां एक है \omterm{def:b2:metric:continuity}{निरंतर} समारोह $\tau $, द \emph{टोशन}\index{टोशन}, साथ
\[
T' = \kappa N, \qquad
N' = -\kappa T + \tau B, \qquad
B' = -\tau N .
\]
\end{theorem}

\begin{proof}
पहला सूत्र है इसकी परिभाषा $ N $. प्रत्येक सदिश
$ T, N, B $ स्थिरांक है आदर्श $1$ और वे \omterm{def:b2:quadratic:adjoint}{जोड़}ीवार ओर्थोगोनल हैं;
छह संबंधों को अलग करना $\langle X, Y\rangle = \delta_{XY}$
का मैट्रिक्स दिखाता है $(T', N', B')$ आधार में $(T, N, B)$ है
एंटीसिमेट्रिक: वास्तव में $\langle X', Y\rangle + \langle X, Y'\rangle = 0$ और
$\langle X', X\rangle = 0$. इसका $(N, T)$ प्रवेश है $\langle N', T\rangle =
-\langle N, T'\rangle = -\kappa $, और इसके $(T, B)$-कॉलम प्रविष्टि
$\langle T', B\rangle = \kappa\langle N, B\rangle = 0$. शेष का नामकरण
निःशुल्क प्रवेश $\tau = \langle N', B\rangle $ \omterm{def:b2:series:def}{बिल्कुल} प्रदर्शित तीन देता है
सूत्र: एंटीसिमेट्री भरता है $\langle B', N\rangle = -\tau $ और
$\langle B', T\rangle = 0$. निरंतरता का $\tau = \langle N', B\rangle $ स्पष्ट है
तब से $ N'$ और $ B $ हैं \omterm{def:b2:metric:continuity}{निरंतर} ($\tilde\gamma $ है
$\mathcal{C}^3$, इसलिए $ N = T'/\kappa $ है $\mathcal{C}^1$).
\end{proof}

\begin{example}[डार्बौक्स वेक्टर]\label{ex:b2:curves:darboux}
तीन फ़्रेनेट सूत्र एक में संकुचित हो जाते हैं। तय करना $\omega(s) =
\tau\,T + \kappa\,B $ (द \emph{डार्बौक्स वेक्टर}). का उपयोग करते हुए $ B
\wedge T = N $, $ T \wedge N = B $, $ N \wedge B = T $:
\[
\omega \wedge T = \kappa\,N = T', \qquad
\omega \wedge N = \tau\,B - \kappa\,T = N', \qquad
\omega \wedge B = -\tau\,N = B' :
\]
प्रत्येक फ्रेम वेक्टर विकसित होता है $ X' = \omega \wedge X $, द
एक का गतिज \omterm{thm:b2:structures:signature}{हस्ताक्षर} \emph{तात्कालिक घूर्णन} के साथ
कोणीय वेग वेक्टर $\omega $. फ़्रेम दर से घूमता है
$\norm\omega = \sqrt{\kappa^2 + \tau^2}$ चलने के बारे में
अक्ष $\omega $; \omterm{thm:b2:curves:frenet2d}{वक्रता} चारों ओर घूमने का घटक है
द्विसामान्य, \omterm{thm:b2:curves:frenet3d}{टोशन} स्पर्शरेखा के बारे में घटक. के लिए
हेलिक्स, $\omega $ सिलेंडर के अनुदिश एक स्थिर सदिश है
अक्ष --- यही कारण है कि हेलिक्स का फ्रेम आगे बढ़ता है
\omterm{def:b2:funcseq:def}{समान रूप से}. फ़्रेनेट मैट्रिक्स की एंटीसिममेट्री का उपयोग किया गया
\cref{exo:b2:curves:7}, इस एकल का मैट्रिक्स रूप है
ज्यामितीय तथ्य.
\end{example}

\begin{proposition}[मरोड़ ग्रहीयता को मापता है]\label{prop:b2:curves:torsion}
एक द्विनियमित चाप एक समतल में समाहित होता है यदि और केवल यदि $\tau
\equiv 0$; उस स्थिति में विमान (स्थिर) दोलनशील है
विमान.
\end{proposition}

\begin{proof}
अगर $\tau \equiv 0$, तब $ B' = 0$, इसलिए $ B $ एक स्थिर इकाई है
वेक्टर $ B_0$, और
\[
\frac{\dd}{\dd s}\langle \tilde\gamma(s), B_0\rangle
= \langle T(s), B_0\rangle = 0 :
\]
$\langle \tilde\gamma, B_0\rangle $ स्थिर है, इसलिए चाप एक समतल में स्थित है
ओर्थोगोनल से $ B_0$. इसके विपरीत, यदि चाप एक समतल में स्थित है $ P $,
तब $ T $ और $ T'$ (इस तरह $ N $) की दिशा के समानांतर हैं
$ P $ सभी के लिए $ s $; इसलिए $ B = T \wedge N $ दो इकाइयों में से एक है
के सामान्य $ P $, और होना \omterm{def:b2:metric:continuity}{निरंतर} यह स्थिर है; तब $0 = B'
= -\tau N $ साथ $ N \neq 0$ बल $\tau \equiv 0$.
\end{proof}

\begin{example}[एक झुके हुए वृत्त में शून्य मरोड़ होता है]
चाप $\gamma(t) = \bigl(\cos t,\ \tfrac{\sin t}{\sqrt2},\
\tfrac{\sin t}{\sqrt2}\bigr)$ विमान में पड़ा है $ y = z $, और
उस समतल का इकाई वृत्त है (जांचें: $\norm{\gamma(t)} =
1$ और विमान का लम्बवत् आधार $(1,0,0)$,
$(0, \tfrac1{\sqrt2}, \tfrac1{\sqrt2})$ मानक प्रदर्शित करता है
पैरामीट्रिजेशन)। बिना किसी फ़्रेनेट गणना के,
\cref{prop:b2:curves:torsion} भविष्यवाणी $\tau \equiv 0$, और
निश्चित द्विसामान्य विमान की इकाई सामान्य होनी चाहिए $\pm(0,
\tfrac1{\sqrt2}, -\tfrac1{\sqrt2})$. \omterm{thm:b2:curves:frenet3d}{टोशन} मापता नहीं
``\omterm{def:b2:diffcalc:differential}{अंतर}िक्ष में झुका हुआ''; यह उपाय \emph{छोड़कर} एक हवाई जहाज। केवल हेलिक्स अशून्य है $ b $ नीचे वास्तविक उत्पादन होता है
\omterm{thm:b2:curves:frenet3d}{टोशन}.
\end{example}

\begin{example}[हेलिक्स]\label{ex:b2:curves:helixfrenet}
हेलिक्स के लिए $\gamma(t) = (a\cos t, a\sin t, bt)$, $ a > 0$, हम
गणना की गई $ s = ct $ साथ $ c = \sqrt{a^2 + b^2}$. तब
\[
T = \frac1c(-a\sin t,\ a\cos t,\ b),
\qquad
T' \cdot \frac{\dd t}{\dd s}
= \frac{1}{c^2}(-a\cos t, -a\sin t, 0),
\]
इसलिए $\kappa = a/c^2 = a/(a^2 + b^2)$ और $ N = (-\cos t, -\sin t,
0)$: मुख्य सामान्य अक्ष की ओर क्षैतिज रूप से इंगित करता है।
अगला $ B = T \wedge N = \frac1c(b\sin t, -b\cos t, a)$, और $ B'
\frac{\dd t}{\dd s} = \frac{b}{c^2}(\cos t, \sin t, 0) = -\tau N $
देता है
\[
\boxed{\ \kappa = \frac{a}{a^2 + b^2}, \qquad
\tau = \frac{b}{a^2 + b^2}. \ }
\]
दोनों अपरिवर्तनीय स्थिर हैं --- और कोई भी, इसके विपरीत, दिखा सकता है
स्थिरांक के साथ एकमात्र द्विनियमित वक्र $\kappa > 0$ और स्थिर
$\tau $ हेलिकॉप्टर हैं (सर्कल जब $\tau = 0$). संकेतों पर ध्यान दें: $ b >
0$ सकारात्मकता के साथ दाएँ हाथ का हेलिक्स देता है \omterm{thm:b2:curves:frenet3d}{टोशन}.
\end{example}

\begin{example}[चाप की लंबाई के बिना वक्रता और मरोड़; द
मुड़ा हुआ घन]\label{ex:b2:curves:twistedcubic}
द्वारा पुनःपैरामेट्रिज़िंग वक्राकार \omterm{def:b2:curves:length}{लंबाई} बंद में आमतौर पर असंभव है
रूप, इसलिए अपरिवर्तनीयों को कच्चे से निकाला जाना चाहिए
व्युत्पन्न। लिखना $ v = \norm{\gamma'} = s'$; तब $\gamma' =
vT $ और, जैसा कि प्रमाण में है
\cref{prop:b2:curves:kappaformula},
\[
\gamma'' = v'T + v^2\kappa N,
\qquad
\gamma' \wedge \gamma'' = v^3\kappa\,(T \wedge N) =
v^3\kappa\,B .
\]
लेना मानदंड ($\kappa \geq 0$ \omterm{def:b2:diffcalc:differential}{अंतर}िक्ष में):
\[
\kappa = \frac{\norm{\gamma' \wedge \gamma''}}{v^3} .
\]
फर्क $\gamma''$ एक बार फिर और परिवर्तित कर रहा हूँ $ N' =
v(-\kappa T + \tau B)$ (श्रृंखला नियम के माध्यम से $ s $), केवल
$ B $-घटक अंतिम पद से आता है:
\[
\gamma''' = \bigl(v'' - v^3\kappa^2\bigr)T +
\bigl(v'v\kappa + (v^2\kappa)'\bigr)N + v^3\kappa\tau\,B,
\]
ताकि, के साथ \omterm{def:b2:quadratic:adjoint}{जोड़}ी बनाई जा सके $\gamma' \wedge \gamma'' = v^3\kappa B $,
\[
\det(\gamma', \gamma'', \gamma''') = \langle\gamma' \wedge
\gamma'',\ \gamma'''\rangle = v^6\kappa^2\tau,
\qquad\text{i.e.}\qquad
\tau = \frac{\det(\gamma', \gamma'',
\gamma''')}{\norm{\gamma' \wedge \gamma''}^2} .
\]
के लिए आवेदन \emph{मुड़ा हुआ घन} $\gamma(t) = (t,\
t^2,\ t^3)$ पर $ t = 0$: $\gamma' = (1, 0, 0)$, $\gamma'' =
(0, 2, 0)$, $\gamma''' = (0, 0, 6)$, इसलिए $ v = 1$,
\[
\gamma' \wedge \gamma'' = (0, 0, 2), \qquad \kappa(0) = 2,
\qquad
\det(\gamma', \gamma'', \gamma''') = 12, \qquad \tau(0) =
\frac{12}{4} = 3 .
\]
समापन \omterm{def:b2:diffcalc:differential}{अंतर}्दृष्टि: दोनों सूत्र अनुपात हैं जिनमें गति
$ v $ आवश्यक सटीक डिग्री तक रद्द करता है --- $\kappa $ तराजू
प्रति इकाई दूसरे व्युत्पन्न की तरह \omterm{def:b2:curves:length}{लंबाई}, $\tau $ की तरह
प्रति वर्ग क्षेत्र तीन डेरिवेटिव की मिश्रित मात्रा --- जो
है \emph{क्यों} जबकि वे ज्यामितीय हैं $\gamma''$ स्वयं है
नहीं.
\end{example}

\begin{remark}[अंतरिक्ष वक्रों के लिए मौलिक प्रमेय]
जैसे विमान में, \omterm{def:b2:quadratic:adjoint}{जोड़}ी $(\kappa, \tau)$ साथ $\kappa > 0$
की सीधी आइसोमेट्री तक एक द्विनियमित चाप निर्धारित करता है $\R^3$: द
फ़्रेनेट सूत्र फ़्रेम के लिए एक रैखिक \omterm{def:b2:diffcalc:differential}{अंतर} प्रणाली बनाते हैं
$(T, N, B)$, जिसका कॉची-लिप्सचिट्ज़ सिद्धांत है
\cref{ch:b2:diffeq} लागू होता है; समाधान फ़्रेम की ऑर्थोनॉर्मलिटी
संरक्षित है क्योंकि गुणांक मैट्रिक्स एंटीसिमेट्रिक (समान) है
ग्राम-मैट्रिक्स तर्क जैसा कि \cref{exo:b2:curves:7}), और वक्र
एकीकृत करके पुनर्प्राप्त किया जाता है $ T $. हम विवरण इस पर छोड़ते हैं
पाठक को एक महत्वपूर्ण लेकिन शिक्षाप्रद अभ्यास के रूप में।
\end{remark}

\section{स्थानीय अध्ययन: स्पर्शरेखा के संबंध में स्थिति}

\begin{proposition}[नियमित रूप से स्थानीय आकार
बिंदु]\label{prop:b2:curves:local}
होने देना $\gamma $ वर्ग का एक समतल चाप बनें $\mathcal{C}^k $ पर $ t_0$, साथ
$ p $ के साथ सबसे छोटा सूचकांक $\gamma^{(p)}(t_0) \neq 0$ और $ q $ द
सबसे छोटा सूचकांक $> p $ साथ $\gamma^{(q)}(t_0)$ के साथ संरेख नहीं
$\gamma^{(p)}(t_0)$ (यह मानते हुए कि दोनों मौजूद हैं, $ q \leq k $). आधार में
$(u, v) = (\gamma^{(p)}(t_0), \gamma^{(q)}(t_0))$ पर केन्द्रित
$\gamma(t_0)$, टेलर--यंग निर्देशांक देता है
\[
X(t) \sim \frac{(t - t_0)^p}{p!},
\qquad
Y(t) \sim \frac{(t - t_0)^q}{q!} .
\]
स्थानीय तस्वीर केवल की समानता पर निर्भर करती है $ p $ और $ q $:
\begin{center}
\begin{tabular}{c c l}
$ p $ odd, $ q $ even & \emph{साधारण बिंदु} & curve crosses no line,
stays on one side of tangent\\
$ p $ odd, $ q $ odd & \emph{विभक्ति बिंदु} & curve crosses its
tangent\\
$ p $ even, $ q $ odd & \emph{पहली तरह का शिखर} & both branches
on opposite sides of tangent\\
$ p $ even, $ q $ even & \emph{दूसरे प्रकार का पुच्छल} & both
branches on the same side\\
\end{tabular}
\end{center}
\end{proposition}

\begin{proof}
टेलर - युवा क्रम में $ q $ (फ़ंक्शन $\gamma $ है
$\mathcal{C}^q $ पास में $ t_0$):
\[
\gamma(t) - \gamma(t_0)
= \sum_{j=p}^{q} \frac{(t-t_0)^j}{j!}\,\gamma^{(j)}(t_0)
+ o\bigl((t-t_0)^q\bigr).
\]
की पसंद से $ p $ और $ q $, प्रत्येक $\gamma^{(j)}(t_0)$ साथ $ p
\leq j < q $ के साथ संरेख है $ u $; में घटकों को एकत्रित करना
आधार $(u, v)$: $ X(t) = \frac{(t-t_0)^p}{p!}(1 + o(1))$ और
$ Y(t) = \frac{(t-t_0)^q}{q!}(1 + o(1))$. की साइन टेबल $ X $ और
$ Y $ के लिए $ t \gtrless t_0$ --- \omterm{def:b2:series:def}{बिल्कुल} समानताओं द्वारा शासित ---
चार चित्र देता है: उदाहरण के लिए यदि $ p $ सम है, $ X > 0$ पर
दोनों तरफ (दोनों शाखाएँ दिशा में निकलती हैं $+u $: एक पुच्छ),
और की ओर \omterm{def:b2:curves:arc}{स्पर्श रेखा} ($\operatorname{sign} Y $) फ़्लिप
के साथ $ q $ विषम।
\end{proof}

\begin{example}
के लिए $\gamma(t) = (t^2, t^3)$ पर $ t_0 = 0$
(\cref{ex:b2:curves:cusp}): $\gamma'' (0)= (2, 0)$, $\gamma'''(0) =
(0, 6)$, इसलिए $ p = 2$, $ q = 3$: पहली तरह का एक पुच्छल,
अर्धघनाकार परवलय का परिचित चित्र। के लिए $\gamma(t) = (t,
t^3)$ पर $0$: $ p = 1$, $ q = 3$: विभक्ति ---घन पार
यह स्पर्शरेखा है.
\end{example}

\begin{remark}[इस खंड के भीतर परिप्रेक्ष्य]
कर्व्स अगले अध्यायों को तीन तरीकों से फीड करते हैं। ए पर खींचा गया
सतह, वे इसके स्पर्शरेखा तलों और इसके पहले को परिभाषित करते हैं
मौलिक रूप (\cref{ch:b2:surfaces}), और उनके \omterm{def:b2:curves:length}{लंबाई}
परिवेश मीट्रिक --- को प्रतिबंधित करके गणना की जाती है
आगे का अध्याय काफी हद तक इस अध्याय से संबंधित है।
\omterm{pb:b2:curves:1}{लिफ़ाफ़ा} सप्ताहांत समस्या का कैलकुलस दोगुना मिलता है
में अभिन्न \cref{ch:b2:multint}, जहां astroidका क्षेत्र
ग्रीन के सूत्र द्वारा पुनर्गणना की गई है
(\cref{exo:b2:multint:5}) --- एक वक्र, दो सिद्धांत,
मिलते-जुलते उत्तर. और फ़्रेनेट सिस्टम पहले से ही उपयोग किया गया है
के रैखिक \omterm{def:b2:diffcalc:differential}{अंतर} समीकरण \cref{ch:b2:diffeq}
(अस्तित्व, विशिष्टता, और रूढ़िवादिता-संरक्षण
का तर्क \cref{exo:b2:curves:7}): मौलिक प्रमेय
वक्रों का एक \omterm{def:b2:diffcalc:differential}{विभेदक} समीकरण प्रमेय है
ज्यामितीय कपड़े.
\end{remark}

\begin{remark}[विधि: स्थानीय अध्ययन चलाना]
व्यवहार में वर्गीकरण चार चरणों वाली दिनचर्या है।
\emph{एक}, पर \omterm{def:b2:diffcalc:differential}{अंतर} करें $ t_0$ प्रथम अशून्य तक
व्युत्पन्न प्रकट होता है: इसका सूचकांक है $ p $, इसका मान वेक्टर है
$ u $. \emph{दो}, व्युत्पन्न होने तक विभेद करते रहें
के साथ संरेख नहीं $ u $ प्रकट होता है: सूचकांक $ q $, वेक्टर $ v $.
\emph{तीन}, समताएँ पढ़ें $(p, q)$ मेज पर।
\emph{चार}, ड्रा: वक्र साथ छोड़ देता है $+u $ अगर $ p $ अजीब है
(साथ में $ u $ फिर वापस साथ $ u $ अगर $ p $ सम है), किनारे पर
का $ v $ के संकेत द्वारा निर्देशित $ Y $. दो सावधानियाँ. फ़्रेम
$(u, v)$ आम तौर पर है \emph{नहीं} ऑर्थोनॉर्मल--- टेबल
के सापेक्ष स्थितियों का वर्णन करता है \omterm{def:b2:curves:arc}{स्पर्श रेखा}, कोण नहीं
या दूरियाँ, इसलिए मत पढ़ो \omterm{thm:b2:curves:frenet2d}{वक्रता} चित्र से बाहर.
और मध्यवर्ती डेरिवेटिव के साथ संरेख $ u $ अनुमति है
रैंकों के बीच $ p $ और $ q $ (वे केवल विस्तार को स्थानांतरित करते हैं
$ X $); क्या चाहिए \emph{नहीं} घटित होना पहले पर रुकना है
शून्येतर व्युत्पन्न और अनुमान लगाना $ q = p + 1$: के लिए $\gamma(t)
= (t^2, t^4 + t^5)$ भोला अनुमान $ q = 3$ गलत है, जैसे
$\gamma^{(3)}(0)$ के साथ अभी भी संरेख है $\gamma''(0)$ ---
यह \omterm{def:b2:series:def}{बिल्कुल} सही है \cref{exo:b2:curves:5}.
\end{remark}

\begin{exercise}[$\star $]\label{exo:b2:curves:3}
दिखाओ कि वक्राकार \omterm{def:b2:curves:length}{लंबाई} के ग्राफ का $ f(x) = \cosh x $ ऊपर

वक्र (द \emph{ज़ंजीर का}). सत्यापित करें कि $ R(x) = 1/\kappa(x) =

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:curves:4}
(लघुगणकीय सर्पिल) चलो $\gamma(t) = e^{t}(\cos t,\ \sin t)$, $ t
\in \R $.
दिखाओ कि बीच का कोण है $\gamma(t)$ और $\gamma'(t)$ है
स्थिरांक, गणना करें वक्राकार \omterm{def:b2:curves:length}{लंबाई} का $\gamma $ पर $(-\infty, 0]$
(परिमित!), और \omterm{thm:b2:curves:frenet2d}{वक्रता}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:curves:5}
ठानना $ p $, $ q $ और स्थानीय आकार (साधारण, विभक्ति,
पुच्छल) का $\gamma(t) = (t^2,\ t^4 + t^5)$ पर $ t = 0$, और का
$\gamma(t) = (t^3,\ t^4)$ पर $ t = 0$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:curves:6}
होने देना $\gamma $ एक यूनिट-स्पीड प्लेन आर्क के साथ बनें $\kappa(s) > 0$ सभी के लिए $ s $, और चलो $ c(s) = \gamma(s) + \frac{1}{\kappa(s)}N(s)$ हो \omterm{def:b2:curves:curvature}{वक्रता का केंद्र} (वक्र $ c $ है \emph{केन्द्रज}).
यह मानते हुए $\kappa $ है $\mathcal{C}^1$, बताते हैं कि $ c'(s) =
-\frac{\kappa'(s)}{\kappa(s)^2}N(s)$: evolute स्पर्शरेखा है
की सामान्य रेखाएँ $\gamma $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:curves:7}
होने देना $ A(s)$ एक हो \omterm{def:b2:metric:continuity}{निरंतर} एंटीसिमेट्रिक का परिवार $3 \times 3$
मैट्रिक्स और $ F' = F A $ के साथ एक मैट्रिक्स समाधान $ F(s_0)$
ओर्थोगोनल. बताते हैं कि $ F(s)$ सभी के लिए ओर्थोगोनल है $ s $.
\emph{(\omterm{def:b2:diffcalc:differential}{अंतर} करें $ G = F F^{\mathsf T}$ और विशिष्टता का उपयोग करें
कॉची--लिप्सचिट्ज़।)} फ्रेनेट प्रणाली की प्रासंगिकता स्पष्ट करें।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:curves:8}
(कुल \omterm{thm:b2:curves:frenet2d}{वक्रता} एक बंद \omterm{def:b2:affine:convex}{उत्तल} वक्र का) चलो $\tilde\gamma $ एक हो
इकाई गति $\mathcal{C}^2$ का बंद समतल चाप \omterm{def:b2:curves:length}{लंबाई} $ L $
(इसलिए $\tilde\gamma(s + L) = \tilde\gamma(s)$), एक बार ट्रैवर्स किया गया
वामावर्त. कोण फ़ंक्शन का उपयोग करना $\varphi $ साथ $ T =
(\cos\varphi, \sin\varphi)$ से
\cref{thm:b2:curves:fundamental}, समझाइए क्यों $\varphi(L) -
\varphi(0)$ का गुणज है $2\pi $, और दिखाओ
$\int_0^L \kappa(s)\,\dd s = \varphi(L) - \varphi(0)$. (ए के लिए
त्रिज्या का वृत्त $ R $: $\int \kappa = \frac1R \cdot 2\pi R = 2\pi $.
स्पर्शरेखाओं को मोड़ने का प्रमेय मूल्य पर जोर देता है $2\pi $ के लिए
हर साधारण बंद वक्र; आपसे यह साबित करने के लिए नहीं कहा गया है।)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:curves:9}
दिखाएँ कि स्थिरांक के साथ एक द्विनियमित \omterm{def:b2:diffcalc:differential}{अंतर}िक्ष वक्र $\kappa > 0$ और
$\tau = 0$ त्रिज्या का एक वृत्त (एक चाप) है $1/\kappa $.
\emph{(उपयोग \cref{prop:b2:curves:torsion}, फिर केंद्र दिखाएँ
$\gamma + \frac1\kappa N $ स्थिर है.)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:curves:10}
की गणना करें वक्राकार \omterm{def:b2:curves:length}{लंबाई} परवलय का $ y = x^2/2$ ऊपर

\[

\]

का प्रक्षेप पथ $\gamma $ एक सीधी रेखा में समाहित है.
\emph{(पैरामीटराइज़ द्वारा वक्राकार \omterm{def:b2:curves:length}{लंबाई}, लिखना $\gamma(s) +

\begin{exercise}[$\star\star\star $]\label{exo:b2:curves:12}
(\omterm{def:b2:diffcalc:differential}{अंतर}िक्ष वक्रों के लिए मौलिक प्रमेय) चलो $\kappa > 0$ और
$\tau $ होना \omterm{def:b2:metric:continuity}{निरंतर} एक \omterm{def:b2:diffcalc:differential}{अंतर}ाल पर कार्य करता है $ J $. निभाना
निम्नलिखित टिप्पणी का कार्यक्रम
\cref{ex:b2:curves:helixfrenet}: (ए) दिखाएं कि रैखिक
प्रणाली $ F' = FA(s)$, साथ $ A(s)$ एंटीसिमेट्रिक फ्रेनेट
मैट्रिक्स से निर्मित $\kappa, \tau $ और $ F(s_0)$ एक प्रत्यक्ष
ऑर्थोनॉर्मल फ़्रेम का एक अद्वितीय वैश्विक समाधान है, जो बना हुआ है

(सी) प्रत्यक्ष आइसोमेट्री तक विशिष्टता साबित करें $\R^3$.
\end{exercise}

\begin{problem}[{सप्ताहांत समस्या---लिफाफा मशीन और
चार शास्त्रीय वक्र}]\label{pb:b2:curves:1}
लाइनों का एक-पैरामीटर परिवार आमतौर पर कवर करने में विफल रहता है
\omterm{def:b2:funcseq:def}{समान रूप से} समतल: रेखाएँ सभी के स्पर्शरेखा वक्र के साथ ढेर हो जाती हैं
उनमें से, उनके \emph{लिफ़ाफ़ा}\index{लिफ़ाफ़ा}. प्रकाश किरणें
बनाओ लिफाफे के रूप में दिखाई देता है \emph{कास्टिक} --- उज्ज्वल
कॉफ़ी के एक मग में घुमावदार वक्र। यह समस्या उत्पन्न करती है
सामान्य \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} मशीन, फिर इसे चार बार चलाती है: स्लाइडिंग
सीढ़ी (\omterm{pb:b2:curves:1}{astroid}), परवलय और के सामान्य
साइक्लोइड (अंत में ह्यूजेंस के पेंडुलम के साथ विकसित होता है), और
कॉफ़ी-कप कास्टिक (\omterm{pb:b2:curves:1}{नेफ्रोइड}). लगातार, $ D_t $ को दर्शाता है
समीकरण की रेखा $ a(t)\,x + b(t)\,y = c(t)$, कहाँ $ a, b, c $
हैं $\mathcal C^2$ के साथ कार्य करता है $(a(t), b(t)) \neq (0,0)$,
और $\Delta(t) = a(t)b'(t) - a'(t)b(t)$.

\medskip
\noindent\textbf{Part I --- The \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} machine.}
\begin{enumerate}
  \item कल्पना करना $\Delta(t) \neq 0$. दिखाओ कि
        \emph{विशेषता प्रणाली}
        \[
        \begin{cases} a(t)\,x + b(t)\,y = c(t)\\
        a'(t)\,x + b'(t)\,y = c'(t)\end{cases}
        \]
        एक अनोखा समाधान है $ E(t) = (x(t), y(t))$, द्वारा दिए गए
        $ x = \dfrac{cb' - c'b}{\Delta}$, $ y = \dfrac{ac' -
        a'c}{\Delta}$.
  \item इसके अलावा यह भी मान लीजिए $ E $ है $\mathcal C^1$ पास में $ t $
        साथ $ E'(t) \neq 0$. पहले भेद करना
        सिस्टम का समीकरण, दिखाएँ $ a(t)\,x'(t) +
        b(t)\,y'(t) = 0$, और निष्कर्ष निकालें कि वक्र $ E $
        के एक बिंदु से होकर गुजरता है $ D_t $ \emph{के साथ
        की दिशा} $ D_t $: परिवार स्पर्शरेखा है $ E $,
        जिसे इसका कहा जाता है \emph{\omterm{pb:b2:curves:1}{लिफ़ाफ़ा}}.
  \item विवेक जांच: स्पर्शरेखा रेखाएँ परवलय का $ y =
        x^2/2$ बिंदुओं पर $(t, t^2/2)$ हैं $ tx - y =
        t^2/2$. सत्यापित करें कि \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} मशीन लौटा देती है
        परवलय ही.
  \item (\omterm{pb:b2:curves:1}{लिफ़ाफ़ा} सामान्यों में से) चलो $\gamma $ इकाई-गति हो
        के साथ $\kappa(s) \neq 0$. सामान्य रेखा पर
        $\gamma(s)$ है $\{M : \langle M - \gamma(s), T(s)
        \rangle = 0\}$. दिखाएँ कि इसकी विशेषता प्रणाली है
        बल $\langle M - \gamma(s), N(s)\rangle =
        1/\kappa(s)$, इसलिए यह विशेषता बिंदु है
        द \omterm{def:b2:curves:curvature}{वक्रता का केंद्र}: \emph{ \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} का
        नॉर्मल्स एवोलुट है}, ठीक हो रहा है
        \cref{exo:b2:curves:6}. जाँच करना $\Delta(s) =
        \kappa(s)$.
  \item दो पतन. पेंसिल के लिए $ D_\theta : x\cos
        \theta + y\sin\theta = 0$, दिखाओ कि
        विशेषता बिंदु प्रत्येक का मूल है $\theta $
        (''\omterm{pb:b2:curves:1}{लिफ़ाफ़ा}'' एक बिंदु तक ढह जाता है, और $ E' = 0$:
        प्रश्न 2 लागू नहीं होता) समानांतर के एक परिवार के लिए
        पंक्तियाँ ($ a, b $ स्थिरांक), दिखाएँ $\Delta \equiv 0$ और
        वह विशेषता प्रणाली सामान्य रूप से है
        असंगत: नहीं \omterm{pb:b2:curves:1}{लिफ़ाफ़ा}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The sliding ladder and the
\omterm{pb:b2:curves:1}{astroid}.} का एक खंड \omterm{def:b2:curves:length}{लंबाई} $1$ एक सिरे से स्लाइड $ P_t =
(\cos t, 0)$ फर्श पर और दूसरा $ Q_t = (0, \sin t)$ पर
दीवार, $ t \in \intoo0{\pi/2}$.
\begin{enumerate}[resume]
  \item वह रेखा दिखाओ $(P_tQ_t)$ समीकरण है $ x\sin t +
        y\cos t = \sin t\cos t $, और वह \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} मशीन
        विशेषता बिंदु देता है
        \[
        E(t) = (\cos^3 t,\ \sin^3 t) :
        \]
         \emph{astroid}\index{astroid}, अन्तर्निहित का
        समीकरण $ x^{2/3} + y^{2/3} = 1$ (तक विस्तारित
        समरूपता द्वारा अन्य चतुर्थांश)।
  \item बताते हैं कि $ E'(0) = 0$ और, स्थानीय का उपयोग करना
        वर्गीकरण (\cref{prop:b2:curves:local}), वह
        द \omterm{pb:b2:curves:1}{astroid} पर पहली तरह का एक पुच्छल भाग है $(1, 0)$
        --- और इसी प्रकार इसके चार अक्ष बिंदुओं पर भी।
  \item गणना करना $\norm{E'(t)} = \tfrac32\abs{\sin 2t}$ और
        वह कुल निकालें \omterm{def:b2:curves:length}{लंबाई} की \omterm{pb:b2:curves:1}{astroid} है $6$.
  \item सीढ़ी कहां छूती है \omterm{pb:b2:curves:1}{astroid}? दिखाओ $ E(t) =
        P_t + \sin^2 t\,(Q_t - P_t)$: संपर्क बिंदु
        सीढ़ी को अनुपात में विभाजित करता है $\sin^2 t : \cos^2
        t $, इसे एक सिरे से दूसरे सिरे तक घुमाते हुए
        सीढ़ी स्लाइड.
  \item से घिरे क्षेत्र की गणना करें \omterm{pb:b2:curves:1}{astroid}: वह दिखाओ
        प्रथम-चतुर्थांश क्षेत्र है $3\int_0^{\pi/2}\sin^4
        t\cos^2 t\,\dd t $, द्वारा अभिन्न का मूल्यांकन करें
        रैखिककरण ($\sin^2 2t = \tfrac{1 - \cos 4t}2$),
        और निष्कर्ष निकालें कि कुल क्षेत्रफल है $3\pi/8$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The evolute of the parabola.}
होने देना $\gamma(t) = (t, t^2/2)$.
\begin{enumerate}[resume]
  \item दिखाएँ कि सामान्य रेखा पर है $\gamma(t)$ समीकरण है
        $ x + t\,y = t + t^3/2$.
  \item चलाएँ \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} मशीन: दिखाओ कि \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} का
        सामान्य है
        \[
        E(t) = \Bigl(-t^3,\ 1 + \tfrac32 t^2\Bigr),
        \]
        निहित समीकरण के साथ $ x^2 = \tfrac8{27}(y - 1)^3$:
        एक अर्धघनाकार परवलय.
  \item प्रश्न 4 के साथ क्रॉस-चेक करें: गणना करें का केंद्र
        \omterm{thm:b2:curves:frenet2d}{वक्रता} $\gamma(t) + \frac1{\kappa(t)}N(t)$ से
        $\kappa(t) = (1 + t^2)^{-3/2}$
        (\cref{ex:b2:curves:kappaexamples}) और पुनर्प्राप्त करें
        एक ही बात.
  \item दिखाएँ कि evolute पर पहली तरह का एक पुच्छल है
        $(0, 1)$, द \omterm{def:b2:curves:curvature}{वक्रता का केंद्र} शीर्ष पर ---
        वह बिंदु जहां $\kappa $ चरम है, जैसा कि भविष्यवाणी की गई थी
        सूत्र $ c' = -\frac{\kappa'}{\kappa^2}N $ का
        \cref{exo:b2:curves:6}.
  \item परवलय के कितने अभिलंब किसी दिए गए बिंदु से होकर गुजरते हैं
        बिंदु $(x_0, y_0)$? दिखाएँ कि उत्तर किसके द्वारा शासित है
        घन $\tfrac{t^3}2 + (1 - y_0)\,t - x_0 = 0$;
        अक्ष मामले का इलाज करें $ x_0 = 0$ पूरी तरह से (एक सामान्य
        के लिए $ y_0 < 1$, तीन के लिए $ y_0 > 1$), और व्याख्या करें
        संक्रमण वक्र के रूप में विकसित।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The coffee-cup caustic.} समानांतर
दिशा की किरणें $(1, 0)$ दर्पण के अंदर प्रहार करो
वृत्त $ x^2 + y^2 = 1$; किरण मार रही है $ P_\theta =
(\cos\theta, \sin\theta)$ के कानून के अनुसार प्रतिबिंबित करता है
प्रतिबिंब.
\begin{enumerate}[resume]
  \item सामान्य (त्रिज्या) में दर्पण समरूपता से,
        औचित्य सिद्ध करें कि प्रतिबिंबित दिशा है $ v = u -
        2\langle u, n\rangle n $ साथ $ u = (1,0)$, $ n =
        (\cos\theta, \sin\theta)$, और गणना करें $ v =
        -(\cos2\theta, \sin2\theta)$.
  \item दिखाएँ कि परावर्तित किरण रेखा पर स्थित है
        \[
        x\sin 2\theta - y\cos 2\theta = \sin\theta .
        \]
  \item चलाएँ \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} मशीन ($\Delta = 2$): वो दिखाओ
        कास्टिक है
        \[
        E(\theta) = \Bigl(\tfrac{3\cos\theta -
        \cos3\theta}4,\ \tfrac{3\sin\theta -
        \sin3\theta}4\Bigr),
        \]
         \emph{नेफ्रोइड}\index{नेफ्रोइड}.
  \item गणना करना $ E'(\theta) = \tfrac32\sin\theta\,
        (\cos2\theta, \sin2\theta)$; जांचें कि स्पर्शरेखा
        दिशा परावर्तित-किरण दिशा है (प्रश्न)।
        16), दो पुच्छों का पता लगाएं $(\pm\tfrac12, 0)$, और
        दिखाएँ कि परावर्तित किरण अक्ष को पार करती है $ y = 0$
        पर $ x = \frac1{2\cos\theta}$ --- तो लगभग अक्षीय किरणें
        पर ध्यान केंद्रित करें $ x = \tfrac12$: फोकल \omterm{def:b2:curves:length}{लंबाई} $ R/2$ एक का
        त्रिज्या का दर्पण $ R $.
  \item दिखाओ कि \omterm{pb:b2:curves:1}{नेफ्रोइड} कुल है \omterm{def:b2:curves:length}{लंबाई} $6$ और वह
        निकट $\theta = 0$,
        \[
        E(\theta) - \bigl(\tfrac12, 0\bigr) =
        \bigl(\tfrac34\theta^2 + o(\theta^2),\ \theta^3 +
        o(\theta^3)\bigr) :
        \]
        पहली तरह का एक पुच्छ, अक्ष की ओर इशारा करता हुआ।
  \item एक पैराग्राफ में बताएं कि कास्टिक चमकीला क्यों है:
        कास्टिक के ठीक बाहर प्रत्येक बिंदु से होकर दो पास करें
        परावर्तित किरणें, उस पर प्रत्येक बिंदु से होकर गुजरती हैं किरणें
        ``असीमित रूप से केंद्रित'' हैं (नक्शा $(\theta,
        \text{distance along ray}) \mapsto \R^2$ एक है
        \omterm{def:b2:series:def}{बिल्कुल} महत्वपूर्ण बिंदु पर \omterm{pb:b2:curves:1}{लिफ़ाफ़ा}).
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Huygens: the cycloid is its own
evolute.} होने देना $\gamma(t) = (t - \sin t,\ 1 - \cos t)$, $ t \in
\intoo0{2\pi}$, चक्रवात का एक आर्क।
\begin{enumerate}[resume]
  \item गणना करना $\kappa(t) = -\dfrac1{4\sin(t/2)}$ और यह
        \omterm{def:b2:curves:curvature}{वक्रता का केंद्र}; दिखाएँ कि evolute है
        \[
        c(t) = (t + \sin t,\ \cos t - 1),
        \]
        और वह प्रतिस्थापन $ t = u + \pi $ इसे प्रदर्शित करता है
        द्वारा अनुवादित मूल चक्रवात के रूप में $(\pi, -2)$:
        \emph{एक चक्रवात का विकास सर्वांगसम होता है
        चक्रज} (ह्यूजेन्स)।
  \item सत्यापित करें कि \omterm{def:b2:curves:curvature}{वक्रता त्रिज्या} शीर्ष पर $ t =
        \pi $ के बराबर होती है $4$, जो कि आधा है \omterm{def:b2:curves:length}{लंबाई} $8$ एक का
        मेहराब (\cref{exo:b2:curves:1}); के शिखर का पता लगाएं
        दूरी पर सीधे शीर्ष के नीचे विकसित होता है $4$.
  \item (स्ट्रिंग गुण) चलो $\gamma $ इकाई-गति के साथ हो
        $\kappa > 0$, $\kappa $ कक्षा का $\mathcal C^1$ और
        $ R = 1/\kappa $ सख्ती से नीरस. का उपयोग करते हुए $ c' = R'N $,
        दिखाओ कि वक्राकार \omterm{def:b2:curves:length}{लंबाई} के बीच विकसित
        $ c(s_0)$ और $ c(s_1)$ है $\abs{R(s_1) - R(s_0)}$.
        व्याख्या करें: उत्क्रांति से खुली हुई एक तनी हुई डोरी,
        का \omterm{def:b2:curves:length}{लंबाई} $ R(s_0)$ शुरुआत में, इसका स्वतंत्र अंत होता है
        मूल वक्र का पता लगाएं --- तो एक पेंडुलम झूल रहा है
        ह्यूजेंस की घड़ी के दो साइक्लोइडल गालों के बीच
        एक चक्रवात का वर्णन करता है।
  \item संश्लेषण। भाग I की मशीन ने उत्पादन किया
        \omterm{pb:b2:curves:1}{astroid}, एक अर्धघनाकार परवलय, ए \omterm{pb:b2:curves:1}{नेफ्रोइड} और ए
        चक्रज चार परिवारों में से प्रत्येक के लिए, एक में बताएं
        वाक्य जहां परिकल्पनाएं $\Delta \neq 0$ और
        $ E' \neq 0$ आयोजित या विफल, और कौन सी ज्यामितीय घटना
        (पुच्छल, फोकस, पतन) की प्रत्येक विफलता $ E' \neq 0$
        संकेत दिया. कहाँ की चरम सीमा होनी चाहिए \omterm{thm:b2:curves:frenet2d}{वक्रता} पर दिखाई देते हैं
        द \omterm{pb:b2:curves:1}{लिफ़ाफ़ा} सामान्य का, और क्यों?
\end{enumerate}
\end{problem}
