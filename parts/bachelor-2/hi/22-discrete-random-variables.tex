\chapter{असतत यादृच्छिक चर}\label{ch:b2:randomvar}

यादृच्छिक चर चारों ओर संभाव्यता गणना व्यवस्थित करें
के बजाय कार्य करता है \omterm{def:b2:proba:space}{घटनाएँ}. पर \omterm{def:b2:structures:countable}{गणनीय} रिक्त स्थान सिद्धांत है
द्वारा संचालित \omterm{def:b2:series:summable}{संक्षेपणीय} के परिवार \cref{ch:b2:series}:
\omterm{def:b2:randomvar:expectation}{अपेक्षा} द्वारा अनुक्रमित परिवार का योग है \omterm{def:b2:proba:space}{नमूना अंतरिक्ष},
और इसके सभी गुण --- रैखिकता, \omterm{def:b2:structures:sn}{स्थानांतरण}, उत्पाद सूत्र
के लिए \omterm{def:b2:proba:independence}{स्वतंत्र} चर --- के बारे में प्रमेय हैं \omterm{def:b2:series:summable}{संक्षेपणीय}
परिवार. अध्याय मार्कोव की प्रमुख असमानताओं को सिद्ध करता है,
चेबीशेव, कॉची--श्वार्ज़ और जेन्सेन, और शास्त्रीय के साथ समाप्त होता है
\omterm{def:b2:randomvar:law}{कानून} और बड़ी संख्याओं का कमज़ोर नियम, जिसका प्रमाण दो पंक्तियाँ हैं
एक बार चेबीशेव उपलब्ध हो जाए।

\section{यादृच्छिक चर और उनके नियम}

\begin{definition}[असतत यादृच्छिक चर; कानून]\label{def:b2:randomvar:law}
होने देना $(\Omega, \P)$ एक हो \omterm{def:b2:structures:countable}{गणनीय} संभाव्यता स्थान. ए
\emph{अनियमित परिवर्तनशील वस्तु}\index{अनियमित परिवर्तनशील वस्तु} एक नक्शा है $ X \colon \Omega \to E $ ($ E $ कोई भी
सेट; \emph{असली} यादृच्छिक चर जब $ E = \R $). इसका \emph{कानून}\index{कानून}
(या \emph{वितरण}\index{वितरण}) है संभाव्यता माप $\P_X $ पर
\omterm{def:b2:structures:countable}{गणनीय} समुच्चय $ X(\Omega)$ द्वारा परिभाषित
\[
\P_X(\{x\}) = \P(X = x)
= \P\bigl(\{\omega : X(\omega) = x\}\bigr) .
\]
\end{definition}

\begin{example}[शास्त्रीय कानून]\label{ex:b2:randomvar:laws}
\leavevmode
\begin{itemize}
\item \emph{Bernoulli} $\mathcal{B}(p)$: $ X \in \{0, 1\}$, $\P(X =
1) = p $. एक का सूचक आयोजन.
\item \emph{द्विपद} $\mathcal{B}(n, p)$: $\P(X = k) =
\binom nk p^k(1-p)^{n-k}$, $0 \leq k \leq n $: सफलताओं की संख्या
में $ n $ \omterm{def:b2:proba:independence}{स्वतंत्र} बर्नौली परीक्षण (हाई स्कूल वॉल्यूम; पुनः सिद्ध)।
के योग के माध्यम से नीचे \omterm{def:b2:proba:independence}{स्वतंत्र} चर)।
\item \emph{ज्यामितिक} $\mathcal{G}(p)$: $\P(X = k) =
(1-p)^{k-1}p $, $ k \in \N^*$: पहली सफलता की रैंक
(\cref{ex:b2:proba:geometric}).
\item \emph{प्वासों} $\mathcal{P}(\lambda)$: $\P(X = k) =
e^{-\lambda}\frac{\lambda^k}{k!}$, $ k \in \N $ --- ए \omterm{def:b2:proba:space}{संभाव्यता
उपाय} घातीय श्रृंखला द्वारा. दुर्लभ घटनाओं का नियम
(\cref{ch:b2:genfun}).
\end{itemize}
\end{example}

\begin{remark}[कौन सा कानून क्या मॉडल बनाता है]
चार \omterm{def:b2:randomvar:law}{कानून} चार आदिम प्रश्नों के उत्तर दें: बर्नौली,
``क्या ऐसा हुआ?''; द्विपद, ``कितनी बार में $ n $
कोशिश करता है?''; ज्यामितीय, ``पहली बार कब तक?'';
पॉइसन, ``कितने \omterm{def:b2:proba:space}{घटनाएँ} एक निश्चित दर पर, जब कोशिशें होती हैं
अनेक और व्यक्तिगत रूप से असंभावित?'' प्रश्न को पहचानना
मॉडलिंग का नौवां हिस्सा है: संकेतकों का योग इंगित करता है
द्विपद, ज्यामितीय के लिए प्रतीक्षा समय, दुर्लभ-घटना
पोइसन में गिना जाता है --- द्विपद से परिच्छेद के साथ
पॉइसन ने दुर्लभ घटनाओं के नियम को सटीक बनाया
\cref{ch:b2:genfun}.
\end{remark}

\begin{proposition}[ज्यामितीय की स्मृतिहीनता
कानून]\label{prop:b2:randomvar:memoryless}
अगर $ X \sim \mathcal{G}(p)$, तो सभी के लिए $ m, n \in \N $:
\[
\P(X > m + n \mid X > m) = \P(X > n) ,
\]
और ज्यामितीय \omterm{def:b2:randomvar:law}{कानून} एकमात्र हैं \omterm{def:b2:randomvar:law}{कानून} पर $\N^*$ इसके साथ
संपत्ति.
\end{proposition}

\begin{proof}
ज्यामितीय भारों का योग, $\P(X > n) = (1-p)^n $. इस तरह
\[
\P(X > m + n \mid X > m)
= \frac{\P(X > m + n)}{\P(X > m)}
= \frac{(1-p)^{m+n}}{(1-p)^m} = (1-p)^n = \P(X > n).
\]
इसके विपरीत, यदि $ G(n) = \P(X > n)$ संतुष्ट $ G(m + n) =
G(m)G(n)$ साथ $ G(0) = 1$, तब $ G(n) = G(1)^n $ प्रेरण द्वारा; $ q =
G(1) \in \intco{0}{1}$, और $ q = 0$ या \omterm{def:b2:randomvar:law}{कानून} है
$\mathcal{G}(1 - q)$: $\P(X = k) = G(k-1) - G(k) =
q^{k-1}(1 - q)$.
\end{proof}

\begin{example}[कोई भी नंबर कभी भी ``देय'' नहीं होता]
पासा घुमाओ, छः का इंतज़ार करो: इंतज़ार का समय है $ X \sim
\mathcal G(1/6)$. स्मृतिहीनता कहती है कि बाद में $10$
फलहीन रोल, द \emph{शेष} इंतज़ार $ X - 10$, दिया गया
$ X > 10$, फिर से है $\mathcal G(1/6)$: सशर्त
अपेक्षित प्रतीक्षा अभी भी है $6$ रोल, \omterm{def:b2:series:def}{बिल्कुल} शुरुआत की तरह।
मरने वाले को याद नहीं रहता, और कोई भी छक्का कभी ``देय'' नहीं होता ---
जुआरी का भ्रम यह विश्वास है कि सशर्त \omterm{def:b2:randomvar:law}{कानून}
स्थानांतरित कर देना चाहिए था. इसके विपरीत, प्रस्ताव की विशिष्टता
आधा ये उदासीनता कहता है \emph{की विशेषता} ज्यामितीय
प्रतीक्षा समय: कोई भी प्रतीक्षा समय जिसका पूर्वानुमान कभी अद्यतन नहीं होता
ज्यामितीय है. वास्तविक कतारें और जीवनकाल आमतौर पर अद्यतन होते हैं,
ठीक इसी तरह कोई पता लगाता है कि वे नहीं हैं
ज्यामितीय.
\end{example}

\section{अपेक्षा}

\begin{definition}[अपेक्षा]\label{def:b2:randomvar:expectation}
एक असली \omterm{def:b2:randomvar:law}{अनियमित परिवर्तनशील वस्तु} $ X $ पर $(\Omega, \P)$ \emph{एक है
उम्मीद} यदि परिवार
$\bigl(X(\omega)\,\P(\{\omega\})\bigr)_{\omega \in \Omega}$ है
\omterm{def:b2:series:summable}{संक्षेपणीय} (\cref{ch:b2:series}); इसका \emph{अपेक्षा}\index{अपेक्षा} तब है
\[
\E(X) = \sum_{\omega \in \Omega} X(\omega)\,\P(\{\omega\}) .
\]
\end{definition}

\begin{theorem}[स्थानांतरण प्रमेय]\label{thm:b2:randomvar:transfer}
$ X $ एक है \omterm{def:b2:randomvar:expectation}{अपेक्षा} यदि और केवल यदि परिवार $\bigl(x\,\P(X =
x)\bigr)_{x \in X(\Omega)}$ है \omterm{def:b2:series:summable}{संक्षेपणीय}, और तब
\[
\E(X) = \sum_{x \in X(\Omega)} x\,\P(X = x) .
\]
अधिक \omterm{def:b2:funcseq:series}{सामान्यतः}, के लिए $ f \colon X(\Omega) \to \R $, परिवर्तनशील
$ f(X)$ एक है \omterm{def:b2:randomvar:expectation}{अपेक्षा} आईएफएफ $\sum_x \abs{f(x)}\,\P(X = x) <
\infty $, और तब $\E(f(X)) = \sum_x f(x)\,\P(X = x)$.
\end{theorem}

\begin{proof}
PARTITION $\Omega $ स्तर सेट में $\Omega_x = \{X = x\}$, $ x
\in X(\Omega)$. सारांश-दर-पैकेट प्रमेय द्वारा \omterm{def:b2:series:summable}{संक्षेपणीय}
परिवार (\cref{ch:b2:series}), परिवार
$(X(\omega)\P(\{\omega\}))_\omega $ है \omterm{def:b2:series:summable}{संक्षेपणीय} यदि प्रत्येक पैकेट है
(स्वचालित: $\sum_{\omega \in \Omega_x}\abs{x}\P(\{\omega\}) =
\abs x\,\P(X = x)$) \emph{और} पैकेट रकम का परिवार
$\bigl(x\,\P(X = x)\bigr)_x $ है \omterm{def:b2:series:summable}{संक्षेपणीय} --- और फिर कुल
रकम सहमत है. के लिए $ f(X)$: सिद्ध कथन को वेरिएबल पर लागू करें $ Y =
f \circ X $, जिसका लेवल सेट है $\{Y = y\} =
\bigsqcup_{x : f(x) = y}\{X = x\}$; द्वारा दूसरा सारांश
पैकेट परिवर्तित हो जाते हैं $\sum_y y\,\P(Y = y)$ में $\sum_x
f(x)\,\P(X = x)$, पैकेट अब मानों को समूहीकृत कर रहे हैं $ x $
उनकी छवि से $ f(x)$, निरपेक्ष के साथ संक्षेपण एक का
दूसरे के बराबर परिवार।
\end{proof}

\begin{theorem}[अपेक्षा के गुण]\label{thm:b2:randomvar:linearity}
के सेट पर यादृच्छिक चर साथ \omterm{def:b2:randomvar:expectation}{अपेक्षा}:
\begin{enumerate}
\item (रैखिकता) $\E(aX + bY) = a\,\E(X) + b\,\E(Y)$.
\item (सकारात्मकता और एकरसता) $ X \geq 0 \Rightarrow \E(X)
\geq 0$; $ X \leq Y \Rightarrow \E(X) \leq \E(Y)$; और
$\abs{\E(X)} \leq \E(\abs X)$.
\item (वर्चस्व) यदि $\abs X \leq Z $ और $ Z $ एक है \omterm{def:b2:randomvar:expectation}{अपेक्षा},
ऐसा करता है $ X $.
\end{enumerate}
\end{theorem}

\begin{proof}
सभी योग के गुण हैं \omterm{def:b2:series:summable}{संक्षेपणीय} परिवार
(\cref{ch:b2:series}): योग की रैखिकता, सकारात्मकता पद द्वारा
अवधि, और वर्चस्व की कसौटी संक्षेपण. (ध्यान दें
रैखिकता तत्काल है \emph{परिभाषा} ऊपर $\Omega $,
जबकि \omterm{def:b2:structures:sn}{स्थानांतरण} फार्मूले पर यह अजीब होगा --- एक लाभ
परिभाषित करने का $\E $ अपस्ट्रीम।)
\end{proof}

\begin{example}\label{ex:b2:randomvar:expectations}
$ X \sim \mathcal{B}(n, p)$: लिखना $ X = X_1 + \dots + X_n $ एक के रूप में
बर्नौली संकेतकों का योग और रैखिकता का उपयोग करते हुए, $\E(X) = np $ ---
किसी द्विपद गुणांक की आवश्यकता नहीं. $ X \sim \mathcal{G}(p)$: $\E(X) =
\sum_{k\geq1}k(1-p)^{k-1}p = p\cdot\frac{1}{(1 - (1-p))^2} =
\frac1p $, इसकी डिस्क के अंदर ज्यामितीय श्रृंखला को विभेदित करके
(\cref{ch:b2:powerseries}). $ X \sim \mathcal{P}(\lambda)$: $\E(X)
= \sum_{k\geq1}k e^{-\lambda}\frac{\lambda^k}{k!} = \lambda
e^{-\lambda}\sum_{j\geq0}\frac{\lambda^j}{j!} = \lambda $.
\end{example}

\begin{example}[कार्रवाई में स्थानांतरण]\label{ex:b2:randomvar:transferex}
के लिए $ X \sim \mathcal P(\lambda)$, गणना करें
$\E\bigl(\frac1{1+X}\bigr)$ --- द \omterm{def:b2:randomvar:law}{कानून} का $\frac1{1+X}$
अपने आप में अजीब है, लेकिन \omterm{def:b2:structures:sn}{स्थानांतरण} कभी इसकी मांग नहीं करता:
\[
\E\Bigl(\frac1{1+X}\Bigr)
= \sum_{k\geq0}\frac{1}{k+1}\,\eu^{-\lambda}
\frac{\lambda^k}{k!}
= \frac{\eu^{-\lambda}}{\lambda}\sum_{k\geq0}
\frac{\lambda^{k+1}}{(k+1)!}
= \frac{\eu^{-\lambda}}{\lambda}\bigl(\eu^\lambda - 1\bigr)
= \frac{1 - \eu^{-\lambda}}{\lambda} .
\]
दो पाठ. कम्प्यूटेशनल रूप से: एक स्थानांतरित को पहचानना
घातीय श्रृंखला संपूर्ण कार्य है --- \omterm{def:b2:structures:sn}{स्थानांतरण} कम हो जाता है
अपेक्षाएं का $ f(X)$ श्रृंखला में हेरफेर करने के लिए. संरचनात्मक रूप से:
अनुभवहीन प्लग-इन मान होगा $\frac1{1 + \E X} =
\frac1{1 + \lambda}$, और सच्चा उत्तर बड़ा है,
\[
\frac{1 - \eu^{-\lambda}}{\lambda} \geq
\frac{1}{1 + \lambda},
\]
\omterm{def:b2:series:def}{बिल्कुल} \omterm{def:b2:affine:convex}{उत्तल} फलन के लिए जेन्सेन की असमानता के समान $ t
\mapsto \frac1{1+t}$ मांग. अपेक्षाएं का
\omterm{def:b2:affine:convex}{उत्तल} छवियाँ अनुभवहीन प्लग-इन मान के ऊपर बैठती हैं, और स्थानांतरित होती हैं
साथ ही एक श्रृंखला जांच अमूर्त असमानता को ठोस बनाती है।
\end{example}

\begin{theorem}[स्वतंत्रता और उत्पाद]\label{thm:b2:randomvar:product}
यादृच्छिक चर $ X, Y $ हैं \emph{\omterm{def:b2:proba:independence}{स्वतंत्र}} अगर $\P(X = x, Y =
y) = \P(X = x)\P(Y = y)$ सभी के लिए $ x, y $ --- \omterm{def:b2:nvs:equivalent}{समकक्ष},
\omterm{def:b2:proba:space}{घटनाएँ} $\{X \in A\}$ और $\{Y \in B\}$ हैं \omterm{def:b2:proba:independence}{स्वतंत्र} सभी के लिए $ A,
B $. अगर $ X $ और $ Y $ हैं \omterm{def:b2:proba:independence}{स्वतंत्र} वास्तविक चर के साथ
अपेक्षाएं, तब $ XY $ एक है \omterm{def:b2:randomvar:expectation}{अपेक्षा} और
\[
\E(XY) = \E(X)\,\E(Y) .
\]
\end{theorem}

\begin{proof}
दोनों योगों की तुल्यता योग द्वारा अनुसरण करती है
\omterm{def:b2:funcseq:def}{बिंदुवार} पहचान ख़त्म $(x, y) \in A \times B $
($\sigma $-एडिटिविटी दो बार)। उत्पाद के लिए: दोहरा परिवार
$\bigl(xy\,\P(X = x)\P(Y = y)\bigr)_{(x,y)}$ है \omterm{def:b2:series:summable}{संक्षेपणीय}, के बाद से
परिवारों के लिए फ़ुबिनी (\cref{ch:b2:series})
\[
\sum_{x, y}\abs x \abs y\,\P(X{=}x)\P(Y{=}y)
= \Bigl(\sum_x \abs x \P(X{=}x)\Bigr)
\Bigl(\sum_y \abs y \P(Y{=}y)\Bigr) < \infty ;
\]
द्वारा स्वतंत्रता यह परिवार बिलकुल सही है $\bigl(xy\,\P(X = x, Y =
y)\bigr)$, जिसका योग है $\E(XY)$ \omterm{def:b2:structures:sn}{स्थानांतरण} द्वारा लागू किया गया
परिवर्तनशील $(X, Y) \mapsto xy $; फ़ुबिनी फिर से अहस्ताक्षरित का मूल्यांकन करता है
उत्पाद के रूप में योग $\E(X)\E(Y)$.
\end{proof}

\begin{example}[उत्पाद, साथ में और बिना
स्वतंत्रता]\label{ex:b2:randomvar:productcontrast}
दो अच्छे पासे घुमाओ। अगर $ Y $ दूसरा पासा है (\omterm{def:b2:proba:independence}{स्वतंत्र} का
पहला), $\E(XY) = \E(X)\E(Y) = 3.5^2 = 12.25$. यदि इसके बजाय
$ Y = X $ (स्वयं पासे का ``उत्पाद''),
\[
\E(X^2) = \frac{1 + 4 + 9 + 16 + 25 + 36}{6} = \frac{91}{6}
\approx 15.17 \neq 12.25 :
\]
वही सीमांत \omterm{def:b2:randomvar:law}{कानून} दोनों परिदृश्यों में, अलग-अलग \omterm{def:b2:quadratic:adjoint}{जोड़} \omterm{def:b2:randomvar:law}{कानून},
अलग उत्पाद अपेक्षाएं. नैतिक, उत्कीर्णन के लायक:
$\E(XY)$ की एक कार्यात्मकता है \emph{\omterm{def:b2:quadratic:adjoint}{जोड़}ा}, दोनों में से नहीं
सीमांत --- और \omterm{def:b2:diffcalc:differential}{अंतर} $\E(X^2) - \E(X)^2 \approx 2.92$
है, के द्वारा\"ओनिग - ह्यूजेन्स, \omterm{def:b2:series:def}{बिल्कुल} \omterm{def:b2:randomvar:variance}{झगड़ा}
$\frac{35}{12}$ मरने का.
\end{example}

\section{प्रसरण, सहप्रसरण, और शास्त्रीय असमानताएँ}

\begin{definition}[क्षण, विचरण]\label{def:b2:randomvar:variance}
$ X $ एक \emph{\omterm{def:b2:structures:generated}{आदेश} का क्षण 2} अगर $ X^2$ एक है \omterm{def:b2:randomvar:expectation}{अपेक्षा}
(तो ऐसा ही होता है $ X $, प्रभुत्व द्वारा: $\abs X \leq \frac{1 +
X^2}{2}$). इसका \emph{झगड़ा}\index{झगड़ा} और \emph{मानक
विचलन}\index{मानक
विचलन} फिर वे
\[
V(X) = \E\bigl((X - \E(X))^2\bigr)
= \E(X^2) - \E(X)^2 ,
\qquad
\sigma(X) = \sqrt{V(X)} ,
\]
(दूसरा रूप --- द \emph{के\"ओनिग - ह्यूजेन्स} सूत्र --- द्वारा
वर्ग का विस्तार करना और रैखिकता का उपयोग करना:
\[
\E\bigl((X - \E X)^2\bigr)
= \E\bigl(X^2 - 2X\,\E X + \E(X)^2\bigr)
= \E(X^2) - 2\,\E(X)^2 + \E(X)^2 ,
\]
उसका उपयोग करते हुए मध्य पद $\E X $ एक स्थिरांक है)। के लिए $ X, Y $ दूसरे क्षणों के साथ,
द \emph{सहप्रसरण}\index{सहप्रसरण} है
\[
\operatorname{Cov}(X, Y)
= \E\bigl((X - \E X)(Y - \E Y)\bigr)
= \E(XY) - \E(X)\E(Y) .
\]
\end{definition}

\begin{theorem}[वेरिएंस टूलकिट]\label{thm:b2:randomvar:variancerules}
दूसरे क्षण वाले चर के लिए:
\begin{enumerate}
\item $ V(aX + b) = a^2\,V(X)$;
\item $ V(X + Y) = V(X) + V(Y) + 2\operatorname{Cov}(X, Y)$, और
अधिक \omterm{def:b2:funcseq:series}{सामान्यतः}
\[
V\Bigl(\sum_{i=1}^n X_i\Bigr)
= \sum_{i=1}^n V(X_i)
+ 2\sum_{i < j}\operatorname{Cov}(X_i, X_j) ;
\]
\item अगर $ X, Y $ हैं \omterm{def:b2:proba:independence}{स्वतंत्र}, $\operatorname{Cov}(X, Y) = 0$
(विपरीत गलत है), इसलिए प्रसरण का \omterm{def:b2:proba:independence}{स्वतंत्र} चर
\omterm{def:b2:quadratic:adjoint}{जोड़}ें.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1} और \emph{2} वर्गों का विस्तार प्लस रैखिकता है;
उत्पाद $ X_iX_j $ पास होना अपेक्षाएं नीचे कॉची-श्वार्ज़ द्वारा
(या द्वारा $\abs{X_iX_j} \leq \frac{X_i^2 + X_j^2}{2}$). \emph{3} है
\cref{thm:b2:randomvar:product} केंद्र पर लागू किया गया
चर. इसके विपरीत एक मानक प्रति उदाहरण: $ X $ वर्दी
पर $\{-1, 0, 1\}$ और $ Y = X^2$ असंबंधित हैं
($\E(XY) = \E(X^3) = 0 = \E X \cdot \E Y $) लेकिन स्पष्ट रूप से निर्भर।
\end{proof}

\begin{theorem}[मार्कोव और चेबीशेव
असमानताएँ]\label{thm:b2:randomvar:markov}
\leavevmode
\begin{enumerate}
\item (मार्कोव) यदि $ X \geq 0$ एक है \omterm{def:b2:randomvar:expectation}{अपेक्षा}, फिर प्रत्येक के लिए $ a
> 0$:
\[
\P(X \geq a) \leq \frac{\E(X)}{a} .
\]
\item (चेबीशेव) यदि $ X $ एक दूसरा क्षण है, फिर प्रत्येक के लिए
$\varepsilon > 0$:
\[
\P\bigl(\abs{X - \E(X)} \geq \varepsilon\bigr)
\leq \frac{V(X)}{\varepsilon^2} .
\]
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} \omterm{def:b2:funcseq:def}{बिंदुवार}, $ a\,\mathbf{1}_{X \geq a} \leq X $ (पर
आयोजन बाईं ओर है $ a \leq X $; इससे दूर, $0 \leq X $). लेना
अपेक्षाएं: $ a\,\P(X \geq a) \leq \E(X)$ एकरसता से और
$\E(\mathbf{1}_A) = \P(A)$.
\emph{2.} मार्कोव को गैर-नकारात्मक चर पर लागू करें $(X - \E
X)^2$ स्तर पर $ a = \varepsilon^2$: द आयोजन $\{(X - \E X)^2
\geq \varepsilon^2\}$ बिलकुल है $\{\abs{X - \E X} \geq
\varepsilon\}$.
\end{proof}

\begin{example}[असंबंधित परंतु चिपका हुआ
एक साथ]\label{ex:b2:randomvar:sumdiff}
दो निष्पक्ष पासे घुमाओ, $ X $ और $ Y $ \omterm{def:b2:proba:independence}{स्वतंत्र}, और सेट करें $ S = X +
Y $, $ D = X - Y $. की द्विरेखीयता द्वारा \omterm{def:b2:randomvar:variance}{सहप्रसरण},
\[
\operatorname{Cov}(S, D) = V(X) - V(Y) +
\operatorname{Cov}(Y, X) - \operatorname{Cov}(X, Y) = V(X) -
V(Y) = 0 :
\]
योग और \omterm{def:b2:diffcalc:differential}{अंतर} असंबद्ध हैं। \omterm{def:b2:proba:independence}{स्वतंत्र}? हरगिज नहीं: $ S = 12$ बल $ D = 0$, जबकि $\P(D = 0) =
\frac16$ बिना शर्त. सहसंबंध केवल परीक्षण करता है
\emph{रेखीय} निर्भरता का हिस्सा; यहाँ निर्भरता है
उस बाधा द्वारा किया गया $ S $ और $ D $ समान है
समता, अदृश्य \omterm{def:b2:randomvar:variance}{सहप्रसरण}. (इस \omterm{def:b2:quadratic:adjoint}{जोड़}ी के लिए, शून्य
\omterm{def:b2:randomvar:variance}{सहप्रसरण} आवश्यकता है $ V(X) = V(Y)$: \omterm{pb:b2:fourier:1}{समान वितरण},
नहीं स्वतंत्रता, काम किया।)
\end{example}

\begin{example}[जब मार्कोव सटीक है]\label{ex:b2:randomvar:markovsharp}
मार्कोव की असमानता ठीक उसी स्थिति में समानता है जब कुछ भी नहीं है
सीमा में बर्बाद हो गया $ a\,\mathbf 1_{X\geq a} \leq X $: द
वेरिएबल को केवल मान लेना चाहिए $0$ और $ a $. ठोस रूप से,
यदि $\P(X = a) = \pi $ और $\P(X = 0) = 1 - \pi $, तब $\E(X)
= a\pi $ और
\[
\P(X \geq a) = \pi = \frac{\E(X)}{a} .
\]
एक यथार्थवादी अध्ययन: ऐसी आबादी में जहां औसत संपत्ति है
$100$ और धन या तो है $0$ या $10^6$, का अनुपात
\omterm{def:b2:series:def}{बिल्कुल} करोड़पति है $10^{-4}$ --- मार्कोव बाउंड, हिट
\omterm{def:b2:series:def}{बिल्कुल} अधिकतम असमानता से। जब कभी भी $ X $ फैल जाता है
मध्यवर्ती मूल्यों की सीमा सख्त है, अक्सर बेतहाशा;
लेकिन जैसा कि चरम मामले से पता चलता है, इससे बेहतर कोई असमानता नहीं हो सकती
केवल माध्य से निकाला गया।
\end{example}

\begin{example}[चेबीशेव तेज है --- बिना आगे के
परिकल्पनाएँ]\label{ex:b2:randomvar:chebsharp}
हल करना $\varepsilon > 0$, $ q \in \intoc01$, और चलो $ X $ ले लो
मूल्य $\pm\varepsilon $ संभाव्यता के साथ $\frac q2$ प्रत्येक और
$0$ संभाव्यता के साथ $1 - q $. तब $\E(X) = 0$, $ V(X) =
q\varepsilon^2$, और
\[
\P\bigl(\abs{X - \E X} \geq \varepsilon\bigr) = q
= \frac{V(X)}{\varepsilon^2} :
\]
चेबीशेव में समानता। अतः असमानता में सुधार नहीं किया जा सकता
केवल का उपयोग कर रहा हूँ \omterm{def:b2:randomvar:variance}{झगड़ा} --- द $1/\varepsilon^2$ क्षय है
दूसरे क्षण की जानकारी की सटीक कीमत। तेजी से क्षय
मजबूत परिकल्पनाओं की आवश्यकता है: चर की सीमा
खरीदता है \emph{घातीय} एकाग्रता, जैसे
\cref{exo:b2:randomvar:7} पूर्वावलोकन और इस अध्याय का सप्ताहांत
समस्या व्यवस्थित रूप से विकसित होती है।
\end{example}

\begin{theorem}[कॉची-श्वार्ज़ और जेन्सेन]\label{thm:b2:randomvar:jensen}
\leavevmode
\begin{enumerate}
\item (कॉची-श्वार्ज़) यदि $ X, Y $ दूसरे पल हैं, $ XY $ एक है
\omterm{def:b2:randomvar:expectation}{अपेक्षा} और $\E(XY)^2 \leq \E(X^2)\,\E(Y^2)$; फलस्वरूप
$\operatorname{Cov}(X,Y)^2 \leq V(X)V(Y)$.
\item (जेन्सेन) यदि $\varphi \colon I \to \R $ है \omterm{def:b2:affine:convex}{उत्तल} एक पर
\omterm{def:b2:diffcalc:differential}{अंतर}ाल युक्त $ X(\Omega)$, और $ X $, $\varphi(X)$ पास होना
अपेक्षाएं, तब
\[
\varphi\bigl(\E(X)\bigr) \leq \E\bigl(\varphi(X)\bigr) .
\]
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} संक्षेपण का $ XY $: $\abs{XY} \leq \frac{X^2 + Y^2}2$.
वो नक्शा $(X, Y) \mapsto \E(XY)$ एक सकारात्मक है \omterm{def:b2:quadratic:adjoint}{सममित} द्विरेखीय
प्रपत्र दूसरे क्षणों के साथ चर के स्थान पर, इसलिए सार
कॉची-श्वार्ज़ असमानता \cref{ch:b2:quadratic} लागू होता है
(सकारात्मक \emph{अर्द्ध}-असमानता के लिए निश्चित पर्याप्तता)।
इसे केन्द्रित चरों पर लागू करने से प्राप्त होता है \omterm{def:b2:randomvar:variance}{सहप्रसरण} अवश्यंभावी।

\emph{2.} पहला, $ m = \E(X)$ में निहित है $ I $: $ I $ एक \omterm{def:b2:diffcalc:differential}{अंतर}ाल है
के सभी मानों से युक्त $ X $, और \omterm{def:b2:randomvar:expectation}{अपेक्षा} एकरस है, इसलिए $ m $
के बीच है $\inf X(\Omega)$ और $\sup X(\Omega)$. द्वारा
\omterm{def:b2:affine:convex}{उत्तल} कार्यों के लिए समर्थन-रेखा प्रमेय
(\cref{ch:b2:realfun}), वहाँ हैं $\alpha, \beta $ साथ
$\varphi(t) \geq \alpha t + \beta $ सभी के लिए $ t \in I $ और
$\varphi(m) = \alpha m + \beta $. तब, \omterm{def:b2:funcseq:def}{बिंदुवार} पर $\Omega $,
$\varphi(X) \geq \alpha X + \beta $; ले रहा हूँ अपेक्षाएं,
\[
\E\bigl(\varphi(X)\bigr) \geq \alpha\,\E(X) + \beta
= \varphi\bigl(\E(X)\bigr). \qedhere
\]
\end{proof}

\begin{example}
जेन्सेन के साथ $\varphi(t) = t^2$ देता है $\E(X)^2 \leq \E(X^2)$ ---
की सकारात्मकता \omterm{def:b2:randomvar:variance}{झगड़ा}; साथ $\varphi(t) = 1/t $ पर
$\intoo{0}{\infty}$: $\frac{1}{\E X} \leq \E\bigl(\frac1X\bigr)$
--- हार्मोनिक माध्य अंकगणितीय माध्य से नीचे है, अब यादृच्छिक रूप में
प्रपत्र.
\end{example}

\begin{remark}[सामान्य ख़तरे]
(मैं) $\E(XY) = \E(X)\E(Y)$ \emph{आवश्यक है} स्वतंत्रता (या पर
न्यूनतम शून्य \omterm{def:b2:randomvar:variance}{सहप्रसरण}): लेना $ Y = X $ देता है $\E(X^2) \neq
\E(X)^2$ जब कभी भी $ V(X) > 0$. (ii) इसी प्रकार $ V(X + X) =
4V(X)$, नहीं $2V(X)$: प्रसरण केवल पार \omterm{def:b2:quadratic:adjoint}{जोड़}ें \omterm{def:b2:proba:independence}{स्वतंत्र}
(या असंबंधित) सारांश। (iii) $\E(f(X))$ क्या नहीं है
$ f(\E(X))$; के लिए \omterm{def:b2:affine:convex}{उत्तल} $ f $ जेन्सेन आपको यह भी बताता है
त्रुटि की दिशा, जैसे कि
\cref{ex:b2:randomvar:transferex}. (iv) अस्तित्व सत्य है
परिकल्पना: सेंट पीटर्सबर्ग चर के लिए $ X = 2^K $ साथ
$\P(K = k) = 2^{-k}$ ($ k \geq 1$),
\[
\sum_{k\geq1}2^k\cdot2^{-k} = \sum_{k\geq1}1 = \infty :
\]
$ X $ लगभग निश्चित रूप से परिमित है फिर भी नहीं है \omterm{def:b2:randomvar:expectation}{अपेक्षा}, और नहीं
खेल के लिए उचित प्रवेश मूल्य मौजूद है। संक्षेपण में
की परिभाषा $\E $ बहीखाता पांडित्य नहीं है --- यह है
जहां भारी पूंछ का पता चलता है। (v) अंत में, \omterm{def:b2:structures:sn}{स्थानांतरण}
प्रमेय की जरूरत है \emph{निरपेक्ष} संक्षेपण किसी से पहले
मूल्यों पर योग की पुनर्व्यवस्था वैध है
(\cref{ch:b2:series}).
\end{remark}


\begin{example}[चेबीशेव एक सौ पर
उछालता है]\label{ex:b2:randomvar:hundredtosses}
के लिए $ X \sim \mathcal B(100, \frac12)$: $\E X = 50$, $ V(X) =
25$. चेबीशेव के साथ $\varepsilon = 6$:
\[
\P(45 \leq X \leq 55) = \P(\abs{X - 50} < 6)
\geq 1 - \frac{25}{36} \approx 0.31 ,
\]
जबकि सटीक द्विपद योग देता है $\approx 0.73$. द
गारंटी $31\%$ सच्चाई से कोसों दूर है, लेकिन इसकी आवश्यकता है
\emph{केवल} मतलब और \omterm{def:b2:randomvar:variance}{झगड़ा} --- वही प्रमाणपत्र
किसी भी वेरिएबल पर शब्दशः लागू होता है $\E = 50$, $ V = 25$,
हालाँकि विदेशी, और \cref{ex:b2:randomvar:chebsharp} दिखाता है
कुछ ऐसे परिवर्तन इसे संतृप्त करते हैं। सार्वभौमिकता की एक कीमत होती है;
जब \omterm{def:b2:randomvar:law}{वितरण} वास्तव में द्विपद, घातीय है
सप्ताहांत समस्या के उपकरण अधिकांश \omterm{def:b2:diffcalc:differential}{अंतर} को पाट देते हैं।
\end{example}

\begin{example}[किसी भाग का उसके साथ सहसंबंध
संपूर्ण]\label{ex:b2:randomvar:partwhole}
के लिए \omterm{def:b2:proba:independence}{स्वतंत्र} \omterm{def:b2:funcseq:def}{समान रूप से} वितरित $ X, Y $ साथ \omterm{def:b2:randomvar:variance}{झगड़ा}
$\sigma^2 > 0$, एक सारांश योग के साथ कितना सहसंबद्ध है
$ S = X + Y $? गणना करना
\[
\operatorname{Cov}(X, S) = \operatorname{Cov}(X, X) +
\operatorname{Cov}(X, Y) = \sigma^2 + 0 = \sigma^2,
\qquad V(S) = 2\sigma^2,
\]
तो सहसंबंध गुणांक है
\[
\rho(X, S) = \frac{\operatorname{Cov}(X,
S)}{\sigma(X)\,\sigma(S)}
= \frac{\sigma^2}{\sigma\cdot\sigma\sqrt2}
= \frac{1}{\sqrt2} \approx 0.707 ,
\]
जो भी आम है \omterm{def:b2:randomvar:law}{कानून} --- पासा, सिक्के, पॉइसन गिनती। साथ
$ n $ सारांश वही गणना देता है $\rho(X_1, S_n) =
1/\sqrt n $: कुल पर प्रत्येक व्यक्तिगत पद का प्रभाव
एक वर्गमूल की तरह पतला हो जाता है, जो कि सहसंबद्ध छाया है
का $\sqrt n $ उतार-चढ़ाव का पैमाना. कॉची-श्वार्ज़
गारंटी देता है $\abs\rho \leq 1$ हमेशा; यहाँ सीमा पूरी हो गई है
\omterm{def:b2:series:def}{बिल्कुल} पतित मामले में $ n = 1$ और पूर्वानुमानित रूप से क्षय होता है
के बाद.
\end{example}

\begin{example}[भारित AM--GM से
जेन्सेन]\label{ex:b2:randomvar:amgm}
होने देना $ Y $ सकारात्मक मूल्य लें $ a_1, \dots, a_k $ के साथ
सम्भावनाएँ $\lambda_1, \dots, \lambda_k $. समारोह
$-\ln $ है \omterm{def:b2:affine:convex}{उत्तल} पर $\intoo0\infty $, तो जेन्सेन देता है
$-\ln\E(Y) \leq \E(-\ln Y)$, यानी
\[
a_1^{\lambda_1}a_2^{\lambda_2}\cdots a_k^{\lambda_k}
\;\leq\; \lambda_1a_1 + \lambda_2a_2 + \dots + \lambda_ka_k :
\]
भारित अंकगणित--ज्यामितीय असमानता, समानता के साथ
यदि $ Y $ स्थिर है. बराबर वजन $\lambda_i = \frac1k $
शास्त्रीय AM--GM को पुनः प्राप्त करें। संभाव्यता चुपचाप सिद्ध हो गई है
एक विशुद्ध बीजगणितीय प्रमेय: एक संभाव्यता चुनना \omterm{def:b2:randomvar:law}{कानून} है
के लिए बस एक बहीखाता उपकरण \omterm{def:b2:affine:convex}{उत्तल} संयोजन ---
का बैरीसेंट्रिक दृष्टिकोण \cref{ch:b2:affine} एक बार फिर, अभी
इंजन के रूप में जेन्सेन के साथ।
\end{example}

\section{बड़ी संख्या का कमजोर नियम}

\begin{theorem}[बड़ी संख्या का कमजोर नियम]\label{thm:b2:randomvar:lln}
होने देना $(X_k)_{k \geq 1}$ \omterm{def:b2:quadratic:adjoint}{जोड़}ीदार हो \omterm{def:b2:proba:independence}{स्वतंत्र} यादृच्छिक चर
उसी के साथ \omterm{def:b2:randomvar:law}{कानून}, दूसरे क्षण को स्वीकार करते हुए; लिखना $ m = \E(X_1)$
और $ S_n = X_1 + \dots + X_n $. फिर हर एक के लिए $\varepsilon > 0$:
\[
\P\Bigl(\,\Bigl|\frac{S_n}{n} - m\Bigr| \geq \varepsilon\Bigr)
\;\leq\; \frac{V(X_1)}{n\,\varepsilon^2}
\xrightarrow[n \to \infty]{} 0 .
\]
\end{theorem}

\begin{proof}
रैखिकता से $\E(S_n/n) = m $; द्वारा
\cref{thm:b2:randomvar:variancerules} (\omterm{def:b2:quadratic:adjoint}{जोड़}ी में स्वतंत्रता
को मारता है \omterm{def:b2:randomvar:variance}{सहप्रसरण}) $ V(S_n) = n\,V(X_1)$, इसलिए $ V(S_n/n) =
V(X_1)/n $. चेबीशेव की असमानता लागू होती है $ S_n/n $ देता है
बाध्य.
\end{proof}

\begin{remark}
यह वह प्रमेय है जो संभाव्यता को आवृत्ति से जोड़ता है: के लिए
$ X_k $ एक का सूचक आयोजन $ A $ में \omterm{def:b2:proba:independence}{स्वतंत्र} दोहराव,
$ S_n/n $ की प्रेक्षित आवृत्ति है $ A $, और बड़े का \omterm{def:b2:randomvar:law}{कानून}
संख्याएँ कहती हैं कि यह चारों ओर ध्यान केंद्रित करता है $\P(A)$ दर पर
$\frac{p(1-p)}{n\varepsilon^2}$. \emph{मज़बूत} \omterm{def:b2:randomvar:law}{कानून} ($ S_n/n \to
m $ लगभग निश्चित रूप से) एक वर्ष 3 प्रमेय है --- यह चौथे के लिए प्रमाण है
हालाँकि, क्षण पहुंच के भीतर है: देखें \cref{exo:b2:randomvar:9},
जो चेबीशेव-प्रकार की सीमा पर बोरेल-कैंटेली चलाता है। वैसा ही
चेबीशेव अनुमान ने बर्नस्टीन-बहुपद प्रमाण को संचालित किया
वीयरस्ट्रैस सन्निकटन प्रमेय \cref{ch:b2:funcseq} --- द
वहाँ लेम्मा गिनती \emph{था} बड़ी संख्या का कमजोर नियम
भेष बदलना।
\end{remark}

\begin{example}[पचास कूपन एकत्रित कर रहा हूँ]
\label{ex:b2:randomvar:couponnumeric}
कूपन संग्राहक \cref{exo:b2:randomvar:3} साथ $ n =
50$ विशिष्ट खिलौने: अपेक्षित कुल है
\[
\E(T_{50}) = 50\,H_{50} = 50\sum_{k=1}^{50}\frac1k
\approx 50 \times 4.499 \approx 225
\]
बक्से --- भोले अनुमान से साढ़े चार गुना $50$. द
हार्मोनिक विकास पूरी कहानी है: पहला $25$ खिलौने
के बारे में पहुंचें $50\ln2 \approx 35$ बक्से, जबकि
\emph{अंतिम} अकेले खिलौने की लागत $50$ औसतन बक्से (ए
पैरामीटर की ज्यामितीय प्रतीक्षा $\frac1{50}$). समापन
समस्याएँ अपने अंतिम खेल पर हावी होती हैं, यही कारण है
\cref{exo:b2:randomvar:12} क्रम में उतार-चढ़ाव पाता है $ n $
--- उस अंतिम ज्यामितीय प्रतीक्षा का आकार --- के आसपास
मतलब $ n\ln n $.
\end{example}

\begin{example}[कितना बड़ा होना चाहिए $ n $ होना?]\label{ex:b2:randomvar:llnnumeric}
प्रेक्षित आवृत्ति को भीतर पिन करने के लिए $\varepsilon = 0.01$ का
$\P(A)$ आत्मविश्वास के साथ $95\%$, चेबीशेव की बाध्य मांगें
\[
\frac{p(1-p)}{n\varepsilon^2} \leq \frac{1}{4n\varepsilon^2}
\leq 0.05,
\qquad\text{i.e.}\qquad
n \geq \frac{1}{4\cdot0.05\cdot(0.01)^2} = 50\,000 .
\]
निर्भरता क्रूर है $\varepsilon $ (\omterm{pb:b2:surfaces:1}{द्विघात}) और
आत्मविश्वास में हल्का (रैखिक में) $1/\alpha $). दोनों विशेषताएं
के गुण हैं \emph{अवश्यंभावी}, सत्य का नहीं: द
सप्ताहांत समस्या की घातीय असमानताएँ कीमत कम करती हैं
से विश्वास $1/\alpha $ को $\ln(1/\alpha)$ --- वही
विशिष्टता के बारे में लागत आएगी $18\,500$ वहाँ नमूने ---
जबकि $1/\varepsilon^2$ पैमाना असली है और
असुधार्य यह जानना कि सीमा का कौन सा भाग ढीला है
स्वयं बाध्य के रूप में उपयोगी।
\end{example}

आगे, यहाँ सब कुछ खिलाता है \cref{ch:b2:genfun}: द
\omterm{def:b2:randomvar:expectation}{अपेक्षा} $\E(t^X)$ के एक चालाक कार्य का $ X $ पैक
संपूर्ण \omterm{def:b2:randomvar:law}{कानून} एक शक्ति श्रृंखला में, क्षण बन जाते हैं
व्युत्पन्न पर $1$, और यादृच्छिक रकम के लिए वाल्ड-प्रकार की पहचान

\omterm{def:b2:proba:independence}{स्वतंत्र} चर की बहुलता बन जाती है उत्पन्न करना
कार्य. पिछड़ा, \omterm{def:b2:randomvar:expectation}{अपेक्षा} एक है \omterm{def:b2:affine:barycenter}{केन्द्रक} के साथ
संभाव्यता भार (\cref{ch:b2:affine}), जेन्सेन का
असमानता \omterm{def:b2:affine:convex}{उत्तल} कार्यों की समर्थन-रेखा ज्यामिति है
(\cref{ch:b2:realfun}), और घातीय-क्षण विधि
इस अध्याय की सप्ताहांत समस्या मार्कोव पर लागू होती है
$\eu^{tX}$ --- एक असमानता, एक अच्छे बदलाव से उन्नत
चर का, तीन अध्यायों में फैला हुआ।
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b2:randomvar:1}
गणना करना $\E(X)$ और $ V(X)$ के लिए $ X \sim \mathcal{B}(n, p)$ (के माध्यम से)
संकेतक), $ X \sim \mathcal{P}(\lambda)$ (दिखाओ $ V(X) =
\lambda $), और $ X \sim \mathcal{G}(p)$ (दिखाओ $ V(X) =
\frac{1-p}{p^2}$; उपयोग $\E(X(X-1))$ और का दूसरा व्युत्पन्न
ज्यामितीय श्रृंखला)।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:randomvar:2}
होने देना $ X \sim \mathcal{P}(\lambda)$ और $ Y \sim \mathcal{P}(\mu)$ होना
\omterm{def:b2:proba:independence}{स्वतंत्र}. बताते हैं कि $ X + Y \sim \mathcal{P}(\lambda + \mu)$
(वजन का कनवल्शन; द्विपद प्रमेय), और वह
सशर्त \omterm{def:b2:randomvar:law}{कानून} का $ X $ दिया गया $ X + Y = n $ द्विपद है
$\mathcal{B}\bigl(n, \frac{\lambda}{\lambda + \mu}\bigr)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:randomvar:3}
(कूपन संग्राहक, \omterm{def:b2:randomvar:expectation}{अपेक्षा}) एक अनाज ब्रांड इनमें से एक को छुपाता है $ n $
प्रत्येक डिब्बे में \omterm{def:b2:funcseq:def}{समान रूप से} अलग-अलग खिलौने। होने देना $ T_n $ की संख्या हो
सभी को एकत्रित करने के लिए बक्सों की आवश्यकता है $ n $ खिलौने. लिखना $ T_n $ के योग के रूप में
\omterm{def:b2:proba:independence}{स्वतंत्र} ज्यामितीय चर (देखने का समय)। \emph{नया} खिलौना कब
$ k $ अभी भी गायब हैं), दिखाएँ
\[
\E(T_n) = n\sum_{k=1}^{n}\frac{1}{k} \sim n\ln n
\]
(श्रृंखला के बराबर--की अभिन्न तुलना
\cref{ch:b2:comparison}).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:randomvar:4}
होने देना $ X \geq 0$ पूर्णांक-मूल्यवान बनें। सिद्ध करो \emph{पूंछ सूत्र}
\[
\E(X) = \sum_{n=1}^{\infty} \P(X \geq n)
\]
(जब कोई भी पक्ष परिमित हो), लिखकर $ X =
\sum_{n\geq1}\mathbf{1}_{X \geq n}$ और सारांश का आदान-प्रदान
(नॉननेगेटिव परिवारों के लिए फ़ुबिनी)। वापस पाना $\E(X) = \frac1p $ के लिए
ज्यामितीय \omterm{def:b2:randomvar:law}{कानून}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:randomvar:5}
(प्रतिस्थापन के बिना नमूना अधिक केंद्रित है) एक कलश है $ N $
गेंदें, $ M $ उनमें से सफेद. खींचना $ n \leq N $ प्रतिस्थापन के बिना और
चलो $ X $ गोरों को गिनें (\emph{अतिज्यामितीय} \omterm{def:b2:randomvar:law}{कानून}). उपयोग करना
संकेतक $ X = \sum_{i=1}^n Y_i $ साथ $ Y_i $  $ i $-वां ड्रा:
प्रत्येक को दिखाओ $ Y_i $ पैरामीटर का बर्नौली है $ p = M/N $ (समरूपता!),
निष्कर्ष $\E(X) = np $ \omterm{def:b2:series:def}{बिल्कुल} प्रतिस्थापन के समान, और दिखाएँ
$\operatorname{Cov}(Y_i, Y_j) = -\frac{p(1-p)}{N-1} < 0$ के लिए $ i
\neq j $, इस तरह $ V(X) = np(1-p)\frac{N - n}{N - 1} \leq np(1-p)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:randomvar:6}
होने देना $ X $ एक दूसरा क्षण लो. बताते हैं कि $ c \mapsto \E\bigl((X -
c)^2\bigr)$ \omterm{def:b2:series:def}{बिल्कुल} न्यूनतम है $ c = \E(X)$, न्यूनतम के साथ
$ V(X)$. फिर वो दिखाओ $\P(X = \E(X)) = 1$ अगर और केवल अगर $ V(X) =
0$. \emph{(दूसरे बिंदु के लिए: यदि $ V(X) = 0$, चेबीशेव का उपयोग करें
के साथ $\varepsilon = 1/n $ और नीरस निरंतरता,
\cref{thm:b2:proba:continuity}.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:randomvar:7}
(एकाग्रता मार्कोव को मात देती है) चलो $ S_n \sim \mathcal{B}(n,
\frac12)$ (प्रमुखों की संख्या $ n $ उचित उछाल)। तुलना करें
मार्कोव द्वारा दी गई सीमाएं ($\P(S_n \geq \frac{3n}{4})$), चेबीशेव द्वारा,
और घातीय (चेर्नॉफ़) विधि द्वारा:
\[
\P\Bigl(S_n \geq \frac{3n}4\Bigr)
\leq \E\bigl(e^{tS_n}\bigr)e^{-3nt/4}
= \Bigl(\frac{1 + e^t}{2}\Bigr)^n e^{-3nt/4}
\quad (t > 0),
\]
और अनुकूलन करें $ t $ एक घातीय रूप से छोटी सीमा प्राप्त करने के लिए। \emph{(पर
$ t = \ln 3$: अवश्यंभावी $\bigl(2\cdot 3^{-3/4}\bigr)^n \approx
(0.877)^n $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:randomvar:8}
(वीयरस्ट्रैस फिर से, संभाव्य रूप से) चलो $ f \colon [0,1] \to \R $
होना \omterm{def:b2:metric:continuity}{निरंतर} और $ S_n \sim \mathcal{B}(n, x)$. दिखाओ कि
\omterm{thm:b2:funcseq:weierstrass}{बर्नस्टीन बहुपद} $ B_nf(x) = \sum_{k=0}^n f\bigl(\frac
kn\bigr)\binom nk x^k(1-x)^{n-k}$ के बराबर होती है
$\E\bigl[f\bigl(\frac{S_n}{n}\bigr)\bigr]$, और पुनः प्राप्त करें
अनुमान $\abs{B_nf(x) - f(x)} \leq \omega_f(\delta) +
\frac{2\norm f_\infty}{4n\delta^2}$ का \cref{ch:b2:funcseq} में
यह संभाव्य भाषा (विभाजित) $\bigl|\frac{S_n}{n} -
x\bigr| \geq \delta $ और चेबीशेव का उपयोग करें)।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:randomvar:9}
(मज़बूत \omterm{def:b2:randomvar:law}{कानून} चौथे क्षण के \omterm{def:b2:diffcalc:differential}{अंतर}्गत) चलो $(X_k)$ होना \omterm{def:b2:proba:independence}{स्वतंत्र},
\omterm{def:b2:funcseq:def}{समान रूप से} वितरित, केन्द्रित ($\E X_1 = 0$), साथ
$\E(X_1^4) < \infty $. विस्तार $\E(S_n^4)$ और गिनती
जीवित शर्तें (केवल $\E(X_i^4)$ और $\E(X_i^2X_j^2)$ शर्तें, $ i
\neq j $), दिखाओ $\E(S_n^4) \leq C n^2$ एक स्थिरांक के लिए $ C $. परिणाम निकालना
$\sum_n \P\bigl(\abs{S_n/n} \geq \varepsilon\bigr) < \infty $
प्रत्येक के लिए $\varepsilon > 0$ (मार्कोव क्रम 4 पर) और इसके साथ समाप्त करें
बोरेल--कैंटेली (\cref{thm:b2:proba:borelcantelli}) वह
$ S_n/n \to 0$ लगभग निश्चित रूप से एक उपयुक्त सूत्रीकरण के साथ: द
आयोजन $\bigcap_{j}\bigcup_N\bigcap_{n \geq N}\{\abs{S_n/n} <
\frac1j\}$ \omterm{def:b2:multint:exact}{संभावना} है $1$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:randomvar:10}
दो निष्पक्ष पासे फेंके जाते हैं; होने देना $ M $ दोनों में से बड़ा हो
परिणाम. के पूँछ सूत्र का उपयोग करना \cref{exo:b2:randomvar:4}
(परिमित संस्करण), दिखाएँ
\[
\E(M) = \sum_{k=1}^{6}\P(M \geq k)
= 6 - \sum_{j=0}^5\Bigl(\frac j6\Bigr)^2 = \frac{161}{36}
\approx 4.47 .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:randomvar:11}
होने देना $ F_n $ एक \omterm{def:b2:funcseq:def}{समान रूप से} यादृच्छिक के निश्चित बिंदुओं की संख्या हो
का क्रमपरिवर्तन $\{1, \dots, n\}$ ($ n \geq 2$). लिखना $ F_n =
\sum_i\mathbf 1_{\sigma(i) = i}$, गणना करें $\E(F_n) = 1$,
$\operatorname{Cov}(\mathbf 1_{\sigma(i)=i}, \mathbf
1_{\sigma(j)=j}) = \frac1{n^2(n-1)}$ के लिए $ i \neq j $, और
निष्कर्ष $ V(F_n) = 1$: औसतन एक अक्षर निश्चित होता है, साथ में
\omterm{def:b2:randomvar:variance}{झगड़ा} \omterm{def:b2:series:def}{बिल्कुल} $1$, जो कुछ भी $ n $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:randomvar:12}
(कूपन संग्राहक, एकाग्रता) की सेटिंग में
\cref{exo:b2:randomvar:3}, दिखाओ
\[
V(T_n) = \sum_{k=1}^n\frac{1 - k/n}{(k/n)^2}
\leq n^2\sum_{k=1}^n\frac{1}{k^2} \leq \frac{\pi^2}{6}n^2,
\]
का उपयोग करते हुए स्वतंत्रता ज्यामितीय चरणों का और $ V(\mathcal
G(p)) = \frac{1-p}{p^2}$ (\cref{exo:b2:randomvar:1}; मूल्य
$\pi^2/6$ है \cref{ex:b2:fourier:basel}). के साथ निष्कर्ष निकालें
चेबीशेव वह $\dfrac{T_n}{n\ln n} \to 1$ \emph{में
संभाव्यता}:कलेक्टर का कुल समय है $ n\ln n $ तक
क्रम में उतार-चढ़ाव $ n $.
\end{exercise}

\section{समस्या: एकाग्रता टूलबॉक्स, मार्कोव से
होफ़्डिंग}

\begin{problem}[{सप्ताहांत समस्या---घातीय एकाग्रता
हाथ से, और सर्वेक्षण में कितने लोगों से पूछा जाना चाहिए}]
\label{pb:b2:randomvar:1}
मार्कोव की असमानता एक पल की लागत है और एक खरीदता है $1/a $ क्षय;
चेबीशेव दो क्षण खर्च करता है और खरीदता है $1/\varepsilon^2$ ---
और \cref{ex:b2:randomvar:chebsharp} दिखाता है कि बस इतना ही है
क्षण खरीद सकते हैं. यह समस्या शेष सीढ़ी पर चढ़ती है:
इसके साथ घातीय (चेर्नॉफ़) विधि \emph{\omterm{def:b2:multint:exact}{एकदम सही}} दर
सिक्का उछालने के लिए, होफ़डिंग की असमानता सभी के लिए सीमित है
चर, और भुगतान --- स्पष्ट, ईमानदार नमूना आकार
मतदान, चुनाव कॉल और सिक्का परीक्षण के लिए।
\index{एकाग्रता असमानताएँ}\index{होफ़डिंग का
असमानता}\index{चेर्नॉफ़ बाध्य} लगातार, $ S_n \sim
\mathcal B(n, p)$ का योग है $ n $ \omterm{def:b2:proba:independence}{स्वतंत्र} बरनौली
चर और $\widehat p_n = S_n/n $ अनुभवजन्य आवृत्ति.

\medskip
\noindent\textbf{Part I --- Calibration on the fair coin.}
यहाँ $ p = \frac12$ और $ a \in \intoo{\frac12}{1}$.
\begin{enumerate}
  \item मार्कोव स्तर पर $ an $: दिखाओ $\P(S_n \geq an) \leq
        \frac1{2a}$, एक ऐसी सीमा जिसकी ओर झुकाव भी नहीं होता $0$.
        मार्कोव इतना कहाँ खो देता है?
  \item चेबीशेव: निष्पक्ष द्विपद की समरूपता का उपयोग करना
        के बारे में $ n/2$, दिखाओ
        \[
        \P(S_n \geq an) = \tfrac12\,
        \P\bigl(\abs{S_n - \tfrac n2} \geq n(a -
        \tfrac12)\bigr)
        \leq \frac{1}{8n(a - 1/2)^2},
        \]
        यानी\ $\frac2n $ पर $ a = \frac34$: बहुपद क्षय पर
        अंतिम.
  \item (चेर्नॉफ़, सामान्य स्तर) गणना करें
        $\E(\eu^{tS_n}) = \bigl(\frac{1 + \eu^t}2\bigr)^n $ और
        अनुकूलित करें $\P(S_n \geq an) \leq
        \E(\eu^{tS_n})\eu^{-tan}$ ऊपर $ t > 0$: दिखाओ
        इष्टतम $ t $ है $\ln\frac{a}{1-a}$ और
        \[
        \P(S_n \geq an) \leq \eu^{-n\,I(a)},
        \qquad
        I(a) = \ln 2 + a\ln a + (1-a)\ln(1-a) > 0 .
        \]
        जाँच करें कि $ a = \frac34$ बाध्य को पुनः प्राप्त करता है
        $\bigl(2\cdot3^{-3/4}\bigr)^n $ का
        \cref{exo:b2:randomvar:7}.
  \item (प्रतिपादक सटीक है) चलो $ k = an $ एक पूर्णांक हो.
        इस तथ्य से कि $\binom nk a^k(1-a)^{n-k}$ है
        का सबसे बड़ा $ n + 1$ संभाव्यता की शर्तें
        \omterm{def:b2:randomvar:law}{वितरण}, सिद्ध करना $\binom nk \geq
        \frac{\eu^{nH(a)}}{n+1}$ साथ $ H(a) = -a\ln a -
        (1-a)\ln(1-a)$, और मिलान निचली सीमा का अनुमान लगाएं
        \[
        \P(S_n \geq an) \geq \binom{n}{an}2^{-n}
        \geq \frac{\eu^{-n\,I(a)}}{n + 1} .
        \]
  \item तीन सीमाओं को सारणीबद्ध करें $ n = 100$, $ a =
        \frac34$: मार्कोव $\frac23$, चेबीशेव $0.02$,
        चेर्नॉफ़ $\approx 2.1\cdot10^{-6}$ (सच्चा मूल्य है
        $\approx 2.8\cdot10^{-7}$). नैतिक, एक वाक्य में?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Hoeffding's inequality.}
\begin{enumerate}[resume]
  \item (रेडमेकर केस) के लिए $\varepsilon = \pm1$ के साथ
        संभाव्यता $\frac12$ प्रत्येक, साबित करो
        \[
        \E(\eu^{t\varepsilon}) = \cosh t \leq \eu^{t^2/2}
        \qquad (t \in \R)
        \]
        दो श्रृंखलाओं की पद दर पद तुलना करके ($(2k)! \geq
        2^kk!$).
  \item निष्कर्ष निकालें, के लिए \omterm{def:b2:proba:independence}{स्वतंत्र} रेडमेकर चर
        $\varepsilon_1, \dots, \varepsilon_n $ और हर $ s >
        0$:
        \[
        \P\Bigl(\sum_{i=1}^n\varepsilon_i \geq s\Bigr)
        \leq \eu^{-s^2/(2n)} .
        \]
  \item उचित सिक्कों में अनुवाद करें ($ X_i =
        \frac{1+\varepsilon_i}2$): $\P\bigl(\widehat p_n -
        \tfrac12 \geq \delta\bigr) \leq \eu^{-2n\delta^2}$,
        और एक कारक के साथ दो तरफा संस्करण $2$.
  \item (हॉफडिंग की लेम्मा) चलो $ X \in \intcc01$ साथ $\E X =
        p $, और $\psi(t) = \ln\E(\eu^{tX})$. उसे उचित ठहराओ
        $\psi $ दो बार है \omterm{def:b2:diffcalc:differential}{विभेदक} साथ
        \[
        \psi''(t) = \E_t(X^2) - \E_t(X)^2, \qquad
        \E_t(Y) := \frac{\E(Y\eu^{tX})}{\E(\eu^{tX})},
        \]
        ए \emph{\omterm{def:b2:randomvar:variance}{झगड़ा}} एक पुनः भारित चर का अभी भी
        मूल्यों को अंदर लेना $\intcc01$; इसके द्वारा बाध्य $\frac14$
        (\cref{exo:b2:randomvar:6}न्यूनतमता तर्क) और
        टेलर द्वारा निष्कर्ष:
        \[
        \E\bigl(\eu^{t(X - p)}\bigr) \leq \eu^{t^2/8} .
        \]
  \item (हॉफडिंग की असमानता) के लिए \omterm{def:b2:proba:independence}{स्वतंत्र} $ X_i \in
        \intcc01$ सामान्य माध्य के साथ $ p $, निष्कर्ष निकालना
        \[
        \P\bigl(\abs{\widehat p_n - p} \geq \delta\bigr)
        \leq 2\,\eu^{-2n\delta^2}
        \qquad (\delta > 0).
        \]
  \item चेबीशेव दर की तुलना करें
        $\frac{p(1-p)}{n\delta^2}$ हॉफडिंग के साथ
        $2\eu^{-2n\delta^2}$: प्रत्येक परिकल्पना कौन सी करती है
        आवश्यकता है, और किससे $ n $ (मोटे तौर पर) करता है
        घातीय बाध्य जीत पर $\delta = 0.03$, $ p =
        \frac12$?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- How many people must a poll
ask?} एक सर्वेक्षण पूछता है $ n $ \omterm{def:b2:proba:independence}{स्वतंत्र}, \omterm{def:b2:funcseq:def}{समान रूप से} चुने गए मतदाता;
प्रत्येक ईमानदारी से उत्तर देता है; $ p $ सच्चा स्कोर है, $\widehat p_n $
सर्वेक्षण का आंकड़ा.
\begin{enumerate}[resume]
  \item दिखाएँ कि सर्वेक्षण सटीक है $\pm\delta $ के साथ
        आत्मविश्वास $1 - \alpha $ (अर्थात्\
        $\P(\abs{\widehat p_n - p} \geq \delta) \leq \alpha $)
        जैसे ही
        \[
        n \;\geq\; \frac{\ln(2/\alpha)}{2\,\delta^2} .
        \]
  \item आवश्यक की गणना करें $ n $ मानक ``तीन के लिए
        अंक, पचानवे प्रतिशत'' विशिष्टता ($\delta
        = 0.03$, $\alpha = 0.05$): $ n \geq 2050$; और एक के लिए
        बिंदु: $ n \geq 18\,445$. निरीक्षण करें --- और समझाएं ---
        आश्चर्यजनक तथ्य यह है कि उत्तर में यह शामिल नहीं है
        जनसंख्या का आकार.
  \item चेबीशेव के साथ प्रश्न 13 दोबारा करें ($ V(X_1) = p(1-p)
        \leq \frac14$): $ n \geq \frac1{4\alpha\delta^2} =
        5556$ तीन बिंदुओं पर. ध्यान दें कि नमूनाकरण
        \emph{बिना} प्रतिस्थापन केवल मदद करता है
        (\cref{exo:b2:randomvar:5}: द \omterm{def:b2:randomvar:variance}{झगड़ा} से सिकुड़ जाता है
        $\frac{N-n}{N-1}$).
  \item (चुनाव की घोषणा) एक उम्मीदवार का असली स्कोर है $ p
        = 0.52$. इसके लिए कितने मतदाताओं का मतदान होना चाहिए
        $\P(\widehat p_n \leq \tfrac12) \leq 0.01$? दिखाओ $ n
        \geq \frac{\ln 100}{2\cdot(0.02)^2} \approx 5757$ ---
        एक करीबी दौड़ बुलाने की लागत अनुमान लगाने से कहीं अधिक है
        स्कोर.
  \item गणित क्या करता है \emph{नहीं} कवर: सूची बनाएं
        प्रयुक्त मॉडलिंग धारणाएँ (\omterm{def:b2:proba:independence}{स्वतंत्र} वर्दी
        नमूनाकरण, ईमानदार उत्तर, निश्चित $ p $), और इसमें समझाएं
        एक संक्षिप्त अनुच्छेद कि वास्तविक मतदान त्रुटियाँ क्यों होती हैं
        का प्रभुत्व है \emph{पक्षपात} (गैर-समान नमूनाकरण,
        गैर-प्रतिक्रिया), जिसमें कोई वृद्धि नहीं हुई $ n $ कम कर देता है.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Sharper and cheaper.}
\begin{enumerate}[resume]
  \item (साधन का माध्य: दो क्षणों से चरघातांकीय क्षय)
        का बजट विभाजित करें $ km $ में नमूने $ k $ \omterm{def:b2:proba:independence}{स्वतंत्र}
        के समूह $ m $; होने देना $\widehat p^{(1)}, \dots, \widehat
        p^{(k)}$ समूह का मतलब हो और $ M $ उनका माध्यिका.
        चुनना $ m $ ताकि प्रत्येक समूह संतुष्ट हो
        $\P(\abs{\widehat p^{(i)} - p} \geq \delta) \leq
        \frac18$ (चेबीशेव: $ m \geq \frac2{\delta^2}$
        पर्याप्त)। दिखाओ अगर $\abs{M - p} \geq \delta $
        तो फिर कम से कम $ k/2$ समूह गलती करते हैं, और निष्कर्ष निकालते हैं
        \[
        \P(\abs{M - p} \geq \delta)
        \leq \binom{k}{\lceil k/2\rceil}\Bigl(\frac18
        \Bigr)^{k/2}
        \leq 2^k\cdot 8^{-k/2} = 2^{-k/2} :
        \]
        इससे परे किसी चीज़ का उपयोग करके घातांकीय एकाग्रता
        प्रसरण.
  \item (पेली--ज़िगमंड) के लिए $ X \geq 0$ दूसरे क्षण के साथ,
        साबित करो $\P(X > 0) \geq \dfrac{\E(X)^2}{\E(X^2)}$
        \emph{(कॉची--श्वार्ज़ ऑन $ X\mathbf 1_{X>0}$)}: द
        रिवर्स-दिशा उपकरण --- क्षण भी मजबूर कर सकते हैं
        \omterm{def:b2:proba:space}{घटनाएँ} घटित होना।
  \item (पिंसकर-लाइट) दिखाएँ $ I(a) \geq 2\bigl(a -
        \tfrac12\bigr)^2$ पर $\intoo{\frac12}1$ \emph{(द
        दूसरे क्रम पर \omterm{def:b2:diffcalc:differential}{अंतर} गायब हो जाता है $\frac12$ और
        इसका दूसरा व्युत्पन्न है $\frac1{a(1-a)} - 4 \geq
        0$)}: चेर्नॉफ़ का सटीक प्रतिपादक हमेशा धड़कता है
        होफ़डिंग का \omterm{pb:b2:surfaces:1}{द्विघात}.
  \item बढ़ाना $ I\bigl(\tfrac12 + \delta\bigr) = 2\delta^2 +
        O(\delta^4)$ और प्रश्न 4 के साथ संयोजित करें: छोटे के लिए
        हॉफडिंग प्रतिपादक का विचलन $2n\delta^2$ है
        स्पर्शोन्मुख रूप से \emph{\omterm{def:b2:multint:exact}{एकदम सही}} --- कोई भी विधि इसे हरा नहीं सकती
        बहुपद कारकों से अधिक द्वारा।
  \item टूलबॉक्स तालिका बनाएं: मार्कोव, चेबीशेव, के लिए
        चौथे क्षण का बंधन \cref{exo:b2:randomvar:9},
        होफ़डिंग, और चेर्नॉफ़ प्रतिपादक के साथ $ I $, राज्य में
        प्रत्येक एक पंक्ति: परिकल्पना आवश्यक, क्षय प्राप्त,
        और इस समस्या में सवाल यह है कि यह कहाँ था
        सबसे तेज़.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends.}
\begin{enumerate}[resume]
  \item (सिक्के का परीक्षण) सिक्का या तो निष्पक्ष होता है या पक्षपातपूर्ण
        $ p = 0.55$. आप इसे पलटें $ n $ बार और घोषित करें
        ``पक्षपातपूर्ण'' कब $\widehat p_n > 0.525$. वो दिखाओ
        दोनों त्रुटि संभावनाएँ अधिकतम हैं
        $\eu^{-2n(0.025)^2}$, ओर वो $ n \geq 3685$ फ़्लिप
        नीचे दोनों की गारंटी दें $1\%$.
  \item (दुर्लभ \omterm{def:b2:proba:space}{घटनाएँ} एक विचरण-जागरूक बाउंड की आवश्यकता है) चलो $ p =
        0.01$ और सापेक्ष विशिष्टता लें $\delta =
        p/2 = 0.005$, $\alpha = 0.05$. नमूने की तुलना करें
        होफ़डिंग द्वारा मांगे गए आकार ($ n \approx 74\,000$) और
        सच के साथ चेबीशेव द्वारा \omterm{def:b2:randomvar:variance}{झगड़ा} $ p(1-p)$ ($ n
        \approx 7920$): विचरण-अंधा घातीय सीमा
        दूसरे ही क्षण विनम्र से हार जाता है। नैतिक बताओ,
        और जहां गायब उपकरण (एक भिन्नता-जागरूक)
        घातीय सीमा; का पॉइसन सन्निकटन
        \cref{ch:b2:genfun}) से आएगा.
  \item (मज़बूत \omterm{def:b2:randomvar:law}{कानून} सिक्कों के लिए) से $\sum_n
        2\eu^{-2n\delta^2} < \infty $ और बोरेल--कैंटेली
        (\cref{thm:b2:proba:borelcantelli}), साबित करें कि
        $\widehat p_n \to p $ लगभग निश्चित रूप से के लिए \omterm{def:b2:proba:independence}{स्वतंत्र}
        सिक्का उछालना: लगभग निश्चित सूत्र तैयार करना आयोजन जैसा
        $\bigcap_j\bigcup_N\bigcap_{n\geq N}
        \{\abs{\widehat p_n - p} < \tfrac1j\}$ के रूप में
        \cref{exo:b2:randomvar:9}, और निष्कर्ष निकालें। (सीमा
        वहां प्रयुक्त चौथे क्षण को प्रतिस्थापित करता है।)
  \item संश्लेषण। पाँच वाक्यों में: प्रत्येक पायदान पर क्या है
        सीढ़ी (क्षण एक, दो, चार; परिबद्ध घातीय;
        सटीक प्रतिपादक) लागत और खरीद; मतदान क्यों? $2050$
        किसी भी आकार के देश के लिए लोग पर्याप्त हैं; और कौन सा
        इन सीमाओं में से वर्ष 3 का खंड तीव्र हो जाएगा
        केंद्रीय सीमा प्रमेय के सटीक स्थिरांक।
\end{enumerate}
\end{problem}
