\chapter{एंडोमोर्फिज्म में कमी}\label{ch:b2:reduction}

एंडोमोर्फिज्म को समझने के लिए, केवल इसकी दिशाएँ खोजें
फैला हुआ यह अध्याय मशीनरी का निर्माण करता है --- eigenvalues,
विशेषता और \omterm{def:b2:reduction:polyu}{न्यूनतम बहुपद}, कर्नेल-विघटन
लेम्मा --- और इसके पुरस्कार: विकर्णीकरण और त्रिकोणीकरण
मानदंड, केली--हैमिल्टन, द \omterm{thm:b2:reduction:dunford}{डनफोर्ड अपघटन}, और
शक्तियों और घातांक की गणना
\cref{ch:b2:diffeq} खिलाऊंगा. लगातार, $E $ एक है
परिमित-आयामी $ K $-सदिश स्थल ($ K = \R $ या $\C $) और $ u \in
\mathcal{L}(E)$, $ n = \dim E $.

\section{आइगेनवैल्यूज़ एवं आइगेनवेक्टर्स}

\begin{definition}\label{def:b2:reduction:eigen}
$\lambda \in K $ एक \emph{eigenvalue}\index{eigenvalue} का $ u $
कब $ u(x) = \lambda x $ कुछ के लिए $ x \neq 0$ (एक
\emph{आइजन्वेक्टर}); \emph{eigenspace} है $ E_\lambda(u) =
\ker(u - \lambda\,\mathrm{id})$. eigenvalues का सेट है
\emph{स्पेक्ट्रम}\index{स्पेक्ट्रम} $\operatorname{Sp}(u)$. एक उपस्थान $ F $ है
\emph{स्थिर} कब $ u(F) \subseteq F $; eigenspaces स्थिर हैं, और
स्थिर उपस्थान प्रेरित एंडोमोर्फिज्म की अनुमति देते हैं $ u|_F $.
\end{definition}

\begin{theorem}[आइजेनस्पेस की स्वतंत्रता]\label{thm:b2:reduction:independence}
आइगेनवेक्टर जोड़ीवार अलग से जुड़ा हुआ eigenvalues फॉर्म ए
मुक्त परिवार; समकक्ष, का योग eigenspaces
$ E_{\lambda_1} + \dots + E_{\lambda_r}$ (विशिष्ट $\lambda_i $) है
प्रत्यक्ष. विशेष रूप से $ u $ अधिकतम है $ n $ eigenvalues.
\end{theorem}

\begin{proof}
पर प्रेरण द्वारा $ r $. कल्पना करना $ x_1 + \dots + x_r = 0$ साथ $ x_i \in
E_{\lambda_i}$, वह कथन जिसके लिए जाना जाता है $ r - 1$. आवेदन करना $ u $ और
घटाना $\lambda_r $ संबंध का समय:
\[
\sum_{i=1}^{r-1} (\lambda_i - \lambda_r)\, x_i = 0 ,
\]
तो प्रत्येक प्रेरण द्वारा $(\lambda_i - \lambda_r)x_i = 0$, यानी\ $ x_i =
0$ के लिए $ i < r $, तब $ x_r = 0$. ए में शून्येतर रिक्त स्थानों का प्रत्यक्ष योग
आयाम का स्थान $ n $ अधिक से अधिक है $ n $ सारांश
\end{proof}

\begin{definition}[अभिलक्षणिक बहुपद]\label{def:b2:reduction:charpoly}
$\chi_u(X) = \det(X\,\mathrm{id} - u)$\index{विशेषता
बहुपद} --- किसी भी आधार पर गणना की जाती है $\det(XI_n - A)$, एक राक्षसी
डिग्री का बहुपद $ n $, समानता के अंतर्गत अपरिवर्तनीय
(\cref{thm:b2:linalg:detrules}). इसकी जड़ें $ K $ बिल्कुल वही हैं
eigenvalues ($\lambda $ \omterm{def:b2:reduction:eigen}{eigenvalue} $\iff u - \lambda\,\mathrm{id}$
इंजेक्शन नहीं $\iff \chi_u(\lambda) = 0$), और
\[
\chi_u(X) = X^n - (\operatorname{tr} u)\, X^{n-1} + \dots +
(-1)^n \det u .
\]
 \emph{बीजगणितीय बहुलता}\index{बीजगणितीय बहुलता}
$ m_\lambda $ की एक \omterm{def:b2:reduction:eigen}{eigenvalue} है
इसकी बहुलता एक जड़ के रूप में है $\chi_u $; \emph{ज्यामितीय
बहुलता}\index{ज्यामितीय
बहुलता} है $\dim E_\lambda $, और $1 \leq \dim E_\lambda \leq
m_\lambda $.
\end{definition}

\begin{proof}[कथित तथ्यों का प्रमाण]
गुणांक का दावा है: विस्तार करें $\det(XI - A)$ क्रमपरिवर्तन द्वारा
सूत्र; पहचान क्रमपरिवर्तन योगदान देता है $\prod_i (X - a_{ii})
= X^n - (\sum a_{ii})X^{n-1} + \dots $, और हर अन्य क्रमपरिवर्तन
अधिकतम ठीक करता है $ n - 2$ विकर्ण स्थिति, योगदान की डिग्री $\leq
n - 2$: शीर्ष दो गुणांक बताए गए अनुसार हैं; $ X = 0$ देता है
स्थिर अवधि $\det(-A) = (-1)^n\det A $.

ज्यामितिक $\leq $ बीजगणितीय: चलो $ d = \dim E_\lambda $ और पूरा करें a
का आधार $ E_\lambda $ के आधार में $ E $; का मैट्रिक्स $ u $ ब्लॉक है
शीर्ष-बाएँ ब्लॉक के साथ ऊपरी-त्रिकोणीय $\lambda I_d $, इसलिए $\chi_u(X) =
(X - \lambda)^d\, \chi_{\text{(bottom block)}}(X)$: बहुलता
का $\lambda $ कम से कम है $ d $.
\end{proof}

\begin{example}[वही $\chi $, अलग ज्यामिति]\label{ex:b2:reduction:samechi}
मैट्रिक्स
\[
\begin{pmatrix}2 & 0\\ 0 & 2\end{pmatrix}
\qquad\text{and}\qquad
\begin{pmatrix}2 & 1\\ 0 & 2\end{pmatrix}
\]
बांटो \omterm{def:b2:reduction:charpoly}{विशेषता बहुपद} $(X - 2)^2$, निशान,
सिद्ध, द \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} --- अभी तक समान नहीं हैं: पहले के पास है
$ E_2$ आयाम का $2$ (\omterm{def:b2:reduction:charpoly}{ज्यामितीय बहुलता} $2$), दूसरा
आयाम का $1$. \omterm{def:b2:reduction:charpoly}{विशेषता बहुपद} केवल देखता है
बीजगणितीय बहुलताएँ; \omterm{def:b2:reduction:eigen}{eigenspace} आयाम हैं
बेहतर अपरिवर्तनीय, और \omterm{def:b2:reduction:polyu}{न्यूनतम बहुपद} मध्यस्थता ($ X - 2$
बनाम $(X - 2)^2$). सभी के लिए नैतिक विकर्णता
चर्चाएँ: $\chi $ उम्मीदवारों को शॉर्टलिस्ट करता है, लेकिन कर्नेल कास्ट करता है
वोट.
\end{example}

\begin{definition}[विकर्णीकरणीय, त्रिकोणीयकरणीय]\label{def:b2:reduction:diag}
$ u $ है \emph{विकर्णीय}\index{विकर्णीय} कब $ E $ एक है
का आधार eigenvectors (मैट्रिक्स: एक विकर्ण मैट्रिक्स के समान);
\emph{त्रिकोणीय}\index{त्रिकोणीय} जब इसका मैट्रिक्स कुछ में होता है
आधार ऊपरी त्रिकोणीय है.
\end{definition}

\begin{theorem}[विकर्णीकरण मानदंड]\label{thm:b2:reduction:diagcrit}
निम्नलिखित समतुल्य हैं:
\begin{enumerate}
  \item $ u $ है \omterm{def:b2:reduction:diag}{विकर्णीय};
  \item $ E = \bigoplus_{\lambda \in \operatorname{Sp} u} E_\lambda $;
  \item $\chi_u $ विभाजित हो जाता है $ K $ और $\dim E_\lambda = m_\lambda $
        हरएक के लिए \omterm{def:b2:reduction:eigen}{eigenvalue};
  \item (पर्याप्त, आवश्यक नहीं) $\chi_u $ है $ n $ विशिष्ट जड़ें
        में $ K $.
\end{enumerate}
\end{theorem}

\begin{proof}
(1 $\iff $ 2): का एक आधार eigenvectors के आधारों में क्रमबद्ध करता है
$ E_\lambda $का, और इसके विपरीत प्रत्यक्ष के आधारों को जोड़ना
सारांश एक आधार देता है $ E $
(\cref{thm:b2:reduction:independence} योग को सीधा बनाता है; समानता
आयामों का यह इसे सब कुछ बनाता है)।

(2 $\iff $ 3): विकर्ण आधार पर, $\chi_u = \prod (X -
\lambda)^{\dim E_\lambda}$ मेल खाती बहुलताओं के साथ विभाजन।
इसके विपरीत, मान लीजिए $\chi_u $ के साथ विभाजित हो जाता है $\dim E_\lambda =
m_\lambda $ लगातार; तो का सीधा योग eigenspaces
(प्रत्यक्ष द्वारा \cref{thm:b2:reduction:independence}) का आयाम है
\[
\sum_{\lambda}\dim E_\lambda = \sum_{\lambda} m_\lambda =
\deg\chi_u = n ,
\]
मध्य समानता क्योंकि एक विभाजित बहुपद की डिग्री है
इसकी मूल बहुलताओं का योग: योग सभी का है $ E $. नोट
जहां प्रत्येक परिकल्पना ने काम किया: विभाजन ने डिग्री भर दी,
बहुलताओं की समानता ने आयाम भर दिए।

(4 $\Rightarrow $ 1): $ n $ विशिष्ट eigenvalues देना $ n $ स्वतंत्र
eigenvectors (\cref{thm:b2:reduction:independence}): एक आधार.
\end{proof}

\begin{method}[विकर्णता का निर्णय करना]\label{met:b2:reduction:decide}
व्यवहार में, इस क्रम में परीक्षण करें --- प्रत्येक चरण समाप्त हो सकता है
नौकरी. (1) सरल विभाजन के साथ एक विनाशकारी बहुपद बनाता है
जड़ें स्वयं मौजूद हैं ($ u^2 = \mathrm{id}$, $ u^2 = u $, $ u^k =
\mathrm{id}$)? यदि हां: \omterm{def:b2:reduction:diag}{विकर्णीय}, कोई गणना नहीं
(\cref{cor:b2:reduction:minpolycrit} नीचे)। (2) गणना करें
$\chi_u $; अगर यह है $ n $ में विशिष्ट जड़ें $ K $: \omterm{def:b2:reduction:diag}{विकर्णीय}
(\cref{thm:b2:reduction:diagcrit}~(4)). (3) अन्यथा, प्रत्येक के लिए
एकाधिक जड़ $\lambda $ केवल, तुलना करें $\dim\ker(u -
\lambda\,\mathrm{id})$ बहुलता के साथ $ m_\lambda $: कोई भी
कमी मार देती है विकर्णता; सर्वत्र समानता यह सिद्ध करती है।
कभी गणना न करें eigenspaces सरल जड़ों का (उनका आयाम है
होने को मजबूर $1$), और केवल निर्णय लेने के लिए कभी भी त्रिकोणीकरण न करें।
\end{method}

\begin{example}[विकर्णीकरण को काम पर लगाया गया]\label{ex:b2:reduction:diagwork}
$ A = I + J = \begin{psmallmatrix}2 & 1 & 1\\ 1 & 2 & 1\\ 1 & 1 &
2\end{psmallmatrix}$, साथ $ J $ ऑल-वन्स मैट्रिक्स: से
$\operatorname{Sp}(J) = \{3, 0\}$
(\cref{ex:b2:linalg:onesmatrix}), $\operatorname{Sp}(A) = \{4,
1\}$, साथ eigenspaces $\R(1,1,1)$ और विमान $\{x + y + z =
0\}$: आयाम $1 + 2 = 3$, \omterm{def:b2:reduction:diag}{विकर्णीय}
(\cref{thm:b2:reduction:diagcrit}~(2)). बिना किसी शक्ति के
परिवर्तन-के-आधार मैट्रिक्स: के साथ $\Pi = J/3$ प्रोजेक्टर चालू
$\R(1,1,1)$,
\[
A = 4\,\Pi + 1\cdot(I - \Pi)
\quad\Longrightarrow\quad
A^k = 4^k\,\Pi + (I - \Pi)
= \frac{4^k - 1}{3}\,J + I .
\]
(जाँच करना $ k = 1$: $\frac{4-1}3 J + I = A $.) समापन अंतर्दृष्टि:
जब eigenspaces दृश्यमान, वर्णक्रमीय हैं \emph{प्रोजेक्टर}
से अधिक तेजी से शक्तियों की गणना करें $ PDP^{-1}$ कभी भी --- और
सूत्र गतिशीलता प्रदर्शित करता है: $ A^k $ जैसे बढ़ता है $4^k $ साथ में
$(1,1,1)$ और ओर्थोगोनल तल पर रखा रहता है।
\end{example}

\begin{theorem}[त्रिकोणीकरण]\label{thm:b2:reduction:trigonalization}
$ u $ है \omterm{def:b2:reduction:diag}{त्रिकोणीय} ऊपर $ K $ अगर और केवल अगर $\chi_u $ विभाजित हो जाता है
$ K $. विशेष रूप से ए की प्रत्येक एंडोमोर्फिज्म $\C $-वेक्टर स्पेस है
\omterm{def:b2:reduction:diag}{त्रिकोणीय}.\index{त्रिकोणीकरण}
\end{theorem}

\begin{proof}
($\Rightarrow $) द \omterm{def:b2:reduction:charpoly}{विशेषता बहुपद} एक \omterm{def:b2:reduction:diag}{त्रिकोणीय} मैट्रिक्स का
है $\prod(X - t_{ii})$: विभाजित करना।

($\Leftarrow $) प्रेरण चालू $ n $. तब से $\chi_u $ विभाजन, इसमें एक है
जड़ $\lambda $: एक चुनें आइजन्वेक्टर $ e_1$. से शुरू होने वाले आधार में
$ e_1$, मैट्रिक्स है $\begin{pmatrix} \lambda & \ast\\ 0 &
B\end{pmatrix}$, और $\chi_u = (X - \lambda)\chi_B $: $\chi_B $ विभाजन
भी. प्रेरण परिकल्पना द्वारा लागू किया गया $(n-1) \times
(n-1)$ मैट्रिक्स $ B $, एक उलटा है $ Q $ साथ $ Q^{-1}BQ $ ऊपरी
\omterm{def:b2:reduction:diag}{त्रिकोणीय}; संपूर्ण मैट्रिक्स को संयुग्मित करना $\begin{pmatrix}1 & 0\\
0 & Q\end{pmatrix}$ इसे \omterm{def:b2:reduction:diag}{त्रिकोणीय} बनाता है.
\end{proof}

\begin{example}[हाथ से त्रिकोणीकरण]\label{ex:b2:reduction:trigwork}
$ B = \begin{pmatrix}3 & -1\\ 1 & 1\end{pmatrix}$: $\chi_B = X^2 -
4X + 4 = (X - 2)^2$, और $\ker(B - 2I) =
\ker\begin{psmallmatrix}1 & -1\\ 1 & -1\end{psmallmatrix}$ है
द्वारा फैलाई गई रेखा $ e_1' = (1, 1)$: एक \omterm{def:b2:reduction:eigen}{eigenvalue}, ए
एक आयामी \omterm{def:b2:reduction:eigen}{eigenspace} --- नहीं \omterm{def:b2:reduction:diag}{विकर्णीय}, लेकिन
\omterm{def:b2:reduction:diag}{त्रिकोणीय} (\cref{thm:b2:reduction:trigonalization}).
के साथ आधार पूरा करें $ e_2' = (1, 0)$ और गणना करें:
\[
u(e_1') = (2, 2) = 2e_1',
\qquad
u(e_2') = (3, 1) = 1\cdot e_1' + 2\, e_2' ,
\]
तो आधार में $(e_1', e_2')$ मैट्रिक्स है $ T =
\begin{psmallmatrix}2 & 1\\ 0 & 2\end{psmallmatrix}$. द
समापन अंतर्दृष्टि: का विकर्ण $ T $ मजबूर किया गया था (दोनों प्रविष्टियाँ
दोगुना होना चाहिए \omterm{def:b2:reduction:eigen}{eigenvalue} $2$); केवल कोने में प्रवेश
की पसंद पर निर्भर था $ e_2'$, और पुनर्स्केलिंग $ e_2'$ बना सकते हैं
यह कोई भी गैर-शून्य मान है --- प्रतिरोधी ``$1$'' की छाया है
निलपोटेंट भाग जिसे डनफोर्ड अलग करेगा।
\end{example}

\section{एक एंडोमोर्फिज्म के बहुपद}

\begin{definition}\label{def:b2:reduction:polyu}
के लिए $ P = \sum a_k X^k \in K[X]$, तय करना $ P(u) = \sum a_k u^k \in
\mathcal{L}(E)$. वो नक्शा $ P \mapsto P(u)$ बीजगणित का एक रूपवाद है
$ K[X] \to \mathcal{L}(E)$ (\cref{def:b2:structures:algebra});
इसकी गिरी $\{P : P(u) = 0\}$ एक आदर्श का $ K[X]$, शून्येतर (द
परिवार $(\mathrm{id}, u, \dots, u^{n^2})$ में जुड़ा हुआ है
$ n^2$आयामी $\mathcal{L}(E)$), इस तरह आप जेनरेट हुई एक अद्वितीय द्वारा
राक्षसी बहुपद $\mu_u $: द \emph{न्यूनतम
बहुपद}\index{न्यूनतम
बहुपद}
(\cref{thm:b2:structures:principal}).
\end{definition}

\begin{proposition}\label{prop:b2:reduction:minpoly}
\begin{enumerate}
  \item $ P(u) = 0 \iff \mu_u \mid P $; eigenvalues का $ u $ जड़ें हैं
        प्रत्येक विनाशकारी बहुपद की, और जड़ों की $\mu_u $
        हैं \emph{बिल्कुल}  eigenvalues.
  \item अगर $ F $ स्थिर है, $\mu_{u|_F} \mid \mu_u $.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) विभाज्यता एक जनरेटर की परिभाषा है। अगर $ u(x) =
\lambda x $, $ x \neq 0$, तब $0 = P(u)(x) = P(\lambda)x $, इसलिए
$ P(\lambda) = 0$: eigenvalues की जड़ें हैं विनाशक, में
का विशेष $\mu_u $. इसके विपरीत, यदि $\lambda $ एक जड़ है, $\mu_u =
(X - \lambda)Q $ साथ $ Q(u) \neq 0$ (की डिग्री $\mu_u $ न्यूनतम): चुनें
$ y $ साथ $ Q(u)(y) \neq 0$; तब $(u - \lambda)(Q(u)(y)) = \mu_u(u)(y)
= 0$ प्रदर्शित करता है आइजन्वेक्टर $ Q(u)(y)$.

(2) $\mu_u(u|_F) = \mu_u(u)|_F = 0$, और (1) लागू करें $ u|_F $.
\end{proof}

\begin{example}[हाथ से पाए गए न्यूनतम बहुपद]\label{ex:b2:reduction:minpolyhand}
 \omterm{def:b2:reduction:polyu}{न्यूनतम बहुपद} क्रमिक परीक्षण द्वारा गणना की जाती है
डिग्री. सभी मैट्रिक्स के लिए $ J \in \mathcal{M}_3(\R)$:
$ J \neq \lambda I $ (डिग्री $1$ बाहर है), और $ J^2 = 3J $, इसलिए
\[
\mu_J = X^2 - 3X = X(X - 3) :
\]
डिग्री $2$, विभाजित, सरल जड़ें --- $ J $ है \omterm{def:b2:reduction:diag}{विकर्णीय} साथ
\omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} $\{0, 3\}$ (\cref{cor:b2:reduction:minpolycrit} नीचे),
पुष्टि कर रहा है \cref{ex:b2:linalg:onesmatrix} एक भी बिना
सिद्ध. स्वैप मैट्रिक्स के लिए $ A $ का
\cref{ex:b2:reduction:projectorswork}: $ A \neq \pm I $ और $ A^2
= I $ देना $\mu_A = X^2 - 1$. दोनों ही मामलों में पैटर्न है
वही: संरचना से निम्न-डिग्री पहचान का अनुमान लगाएं
(रैंक एक बल $ J^2 = (\operatorname{tr}J)\,J $; एक समावेश
बल $ A^2 = I $), फिर जांचें कि कोई उचित भाजक नहीं है
सत्यानाश कर देता है. \omterm{def:b2:reduction:polyu}{न्यूनतम बहुपद} आमतौर पर हैं \emph{मिला}, नहीं
से गणना की गई $\chi $.
\end{example}

\begin{theorem}[कर्नेल अपघटन लेम्मा]\label{thm:b2:reduction:kernels}
अगर $ P = P_1 P_2 \cdots P_r $ साथ $ P_i $ जोड़ीवार सहअभाज्य,
फिर\index{कर्नेल अपघटन लेम्मा}
\[
\ker P(u) = \ker P_1(u) \oplus \dots \oplus \ker P_r(u),
\]
और सारांश पर प्रक्षेपण बहुपद हैं $ u $.
\end{theorem}

\begin{proof}
यह इलाज के लिए पर्याप्त है $ r = 2$ और शामिल करें. बेज़ाउट इन $ K[X]$
(\cref{thm:b2:structures:principal}): $ U P_1 + V P_2 = 1$, तो के लिए
प्रत्येक $ x $,
\[
x = \underbrace{U(u)P_1(u)(x)}_{=:\,x_2}
+ \underbrace{V(u)P_2(u)(x)}_{=:\,x_1}.
\]
अगर $ x \in \ker P(u)$: $ P_2(u)(x_2) = U(u)\,P(u)(x) = 0$ (बहुपद
में $ u $ आवागमन), तो $ x_2 \in \ker P_2(u)$, और सममित रूप से $ x_1
\in \ker P_1(u)$: योग भरता है $\ker P(u)$; दोनों सारांश बैठते हैं
अंदर $\ker P(u)$ ($ P_i \mid P $). प्रत्यक्षता: $ x \in \ker P_1(u)
\cap \ker P_2(u)$ देता है $ x = U(u)P_1(u)x + V(u)P_2(u)x = 0$. द
के लिए सूत्र $ x_1, x_2$ अनुमानों को इस रूप में प्रदर्शित करें $ V(u)P_2(u)$ और
$ U(u)P_1(u)$.
\end{proof}

\begin{example}[स्पष्ट प्रोजेक्टर के साथ कर्नेल लेम्मा]\label{ex:b2:reduction:projectorswork}
होने देना $ A = \begin{psmallmatrix}0 & 1 & 0\\ 1 & 0 & 0\\ 0 & 0 &
1\end{psmallmatrix}$ (पहले दो निर्देशांक बदलें)। तब $ A^2
= I $: बहुपद $ X^2 - 1 = (X - 1)(X + 1)$ सत्यानाश कर देता है $ A $,
इसके कारक सहअभाज्य हैं, और बीézout स्पष्ट है:
\[
\frac{1}{2}(X + 1) - \frac12(X - 1) = 1 .
\]
के प्रमाण के बाद \cref{thm:b2:reduction:kernels}, द
पर प्रक्षेपण $\ker(A - I)$ और $\ker(A + I)$ हैं
बहुपद में $ A $
\[
\pi_+ = \frac{A + I}{2} = \frac12\begin{pmatrix}
1 & 1 & 0\\ 1 & 1 & 0\\ 0 & 0 & 2\end{pmatrix},
\qquad
\pi_- = \frac{I - A}{2} = \frac12\begin{pmatrix}
1 & -1 & 0\\ -1 & 1 & 0\\ 0 & 0 & 0\end{pmatrix}.
\]
जाँच करना: $\pi_+ + \pi_- = I $, $\pi_+\pi_- = 0$, $\pi_\pm^2 =
\pi_\pm $, और छवियाँ विमान हैं $\{x = y\}$ (सममित
वेक्टर, \omterm{def:b2:reduction:eigen}{eigenvalue} $1$) और रेखा $\R(1, -1, 0)$
(एंटीसिमेट्रिक, \omterm{def:b2:reduction:eigen}{eigenvalue} $-1$). कर्नेल लेम्मा एक नहीं है
अस्तित्व कथन: बी\'ईज़आउट गुणांक \emph{हैं} द
प्रोजेक्टर सूत्र.
\end{example}

\begin{example}[प्रोजेक्टर घातांक की भी गणना करते हैं]\label{ex:b2:reduction:expprojectors}
वही स्वैप मैट्रिक्स, एक लाभांश आगे। तब से $ A = \pi_+ -
\pi_-$ ऑर्थोगोनल-इन-द-बीजगणितीय-अर्थ प्रोजेक्टर के साथ
($\pi_+\pi_- = 0$), हर शक्ति आज्ञा का पालन करती है $ A^k = \pi_+ +
(-1)^k\pi_-$, और घातीय श्रृंखला प्रोजेक्टर द्वारा पुन: समूहित होती है:
\[
\eu^{tA} = \sum_k \frac{t^k}{k!}\bigl(\pi_+ +
(-1)^k\pi_-\bigr)
= \eu^{t}\,\pi_+ + \eu^{-t}\,\pi_- =
\begin{pmatrix}
\cosh t & \sinh t & 0\\
\sinh t & \cosh t & 0\\
0 & 0 & \eu^{t}
\end{pmatrix}.
\]
(जाँच करना $ t = 0$: पहचान; पर व्युत्पन्न $0$: $ A $.) द
eigendecomposition एक मैट्रिक्स श्रृंखला को दो अदिश में परिवर्तित करता है
श्रृंखला --- सटीक तंत्र जो \cref{ch:b2:diffeq} होगा
हर पर चलाएँ \omterm{def:b2:reduction:diag}{विकर्णीय} प्रणाली, और कारण अतिशयोक्तिपूर्ण
कार्य सममित युग्मन को नियंत्रित करते हैं।
\end{example}

\begin{corollary}[न्यूनतम बहुपद के माध्यम से विकर्णता]\label{cor:b2:reduction:minpolycrit}
$ u $ है \omterm{def:b2:reduction:diag}{विकर्णीय} $\iff $ $\mu_u $ विभाजित हो जाता है $ K $ साथ
\emph{सरल} जड़ों $\iff $ के कुछ विनाशकारी बहुपद $ u $
साधारण जड़ों से विभाजित होता है।
\end{corollary}

\begin{proof}
अगर $ P(u) = 0$ साथ $ P = \prod_{i}(X - \lambda_i)$ (विशिष्ट
$\lambda_i $), लेम्मा देता है $ E = \ker P(u) = \bigoplus_i \ker(u -
\lambda_i)$: का सीधा योग eigenspaces, इसलिए $ u $ है \omterm{def:b2:reduction:diag}{विकर्णीय}
(\cref{thm:b2:reduction:diagcrit}). इसके विपरीत, ए \omterm{def:b2:reduction:diag}{विकर्णीय}
$ u $ द्वारा मारा जाता है $\prod_{\lambda \in \operatorname{Sp}u}(X -
\lambda)$ (यह प्रत्येक को मारता है \omterm{def:b2:reduction:eigen}{eigenspace}), जो सरल से विभाजित होता है
जड़ें; और $\mu_u $ समान जड़ें रखते हुए इसे विभाजित करता है
(\cref{prop:b2:reduction:minpoly}): $\mu_u $ बिल्कुल वही उत्पाद है.
\end{proof}

\begin{example}\label{ex:b2:reduction:projections}
अनुमान संतुष्ट करते हैं $ p^2 = p $: द्वारा नष्ट कर दिया गया $ X(X-1)$, सरल विभाजन
जड़ें --- \omterm{def:b2:reduction:diag}{विकर्णीय} साथ \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} $\subseteq \{0, 1\}$, और $ E =
\ker p \oplus \ker(p - \mathrm{id})$: वर्ष 1 का ज्यामितीय विश्लेषण,
एक पंक्ति में पुनः सिद्ध किया गया। समरूपता ($ s^2 = \mathrm{id}$,
\omterm{def:b2:linalg:annihilator}{समुच्छेदक} $ X^2 - 1$): \omterm{def:b2:reduction:diag}{विकर्णीय} कब $\operatorname{char} K
\neq 2$, \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} $\subseteq \{\pm 1\}$. के साथ एक एंडोमोर्फिज्म $ u^3 =
u^2$ और $ u^2 \neq u $: द्वारा नष्ट कर दिया गया $ X^2(X - 1)$, \emph{नहीं}
अनिवार्य रूप से \omterm{def:b2:reduction:diag}{विकर्णीय} --- मानदंड इसका पता लगाता है (डबल रूट)।
$0$ परीक्षण किया जाना चाहिए: \omterm{def:b2:reduction:diag}{विकर्णीय} इसके अलावा यदि $\ker u^2 = \ker
u $).
\end{example}

\begin{example}[क्षेत्र निर्णय लेता है: अंदर एक घूर्णन $\R^3$]\label{ex:b2:reduction:rotation3d}
होने देना $ R $ चारों ओर क्वार्टर-टर्न बनें $ z $-अक्ष:
\[
R = \begin{pmatrix}
0 & -1 & 0\\
1 & 0 & 0\\
0 & 0 & 1
\end{pmatrix},
\qquad
\chi_R = (X - 1)(X^2 + 1).
\]
ऊपर $\R $: केवल \omterm{def:b2:reduction:eigen}{eigenvalue} है $1$, साथ \omterm{def:b2:reduction:eigen}{eigenspace} धुरी
$\R e_3$ --- निश्चित सदिशों की एक पंक्ति, और आगे नहीं
कमी: $ R $ नहीं है \omterm{def:b2:reduction:diag}{विकर्णीय} और न \omterm{def:b2:reduction:diag}{त्रिकोणीय} में
$\mathcal{M}_3(\R)$ ($\chi_R $ विभाजित नहीं होता) ऊपर $\C $: तीन
विशिष्ट eigenvalues $1, \iu, -\iu $, इसलिए $ R $ है \omterm{def:b2:reduction:diag}{विकर्णीय},
साथ eigenvectors $ e_3$ और $ e_1 \mp \iu e_2$. ज्यामिति थी
बीजगणित में श्रव्य: समतल में घूर्णन का कोई वास्तविक अर्थ नहीं होता
अपरिवर्तनीय दिशाएँ, और जटिल eigenvalues $\pm\iu $ का
मापांक $1$ कोण संग्रहीत करें ($\pm\frac\pi2$) वह असली है
मैट्रिक्स केवल निर्देशांकों को मिलाकर ही व्यक्त किया जा सकता है।
\end{example}

\begin{example}[न्यूनतम बनाम विशेषता]\label{ex:b2:reduction:minvschar}
के लिए $ D = \operatorname{diag}(2, 2, 3)$: $\chi_D = (X - 2)^2(X -
3)$ लेकिन $\mu_D = (X - 2)(X - 3)$, तब से $(D - 2I)(D - 3I) = 0$
(विहित आधार पर जाँच करें) जबकि कोई भी कारक अकेले नहीं मारता
$ D $. शिफ्ट ब्लॉक के लिए $ N = \begin{psmallmatrix}0 & 1\\ 0 &
0\end{psmallmatrix} \oplus (3)$, यानी\ $ N' =
\begin{psmallmatrix}0 & 1 & 0\\ 0 & 0 & 0\\ 0 & 0 &
3\end{psmallmatrix}$: $\chi_{N'} = X^2(X - 3)$ \emph{और}
$\mu_{N'} = X^2(X - 3)$ --- डबल रूट की वास्तव में आवश्यकता है
क्योंकि $ N'$ क्या नहीं है \omterm{def:b2:reduction:diag}{विकर्णीय} पर $\ker $-ओर ($ N'e_2 =
e_1 \neq 0$). अंगूठे का नियम: $\mu $ और $\chi $ उनकी जड़ें साझा करें
(\cref{prop:b2:reduction:minpoly}); में बहुलता $\mu $
सबसे बड़े निलपोटेंट ब्लॉक के आकार को मापता है, जो कि एक है
$\chi $ विशेषता उपस्थान का कुल आयाम।
\end{example}

\begin{theorem}[केली--हैमिल्टन]\label{thm:b2:reduction:cayleyhamilton}
$\chi_u(u) = 0$; फलस्वरूप $\mu_u \mid \chi_u $, और $\deg \mu_u
\leq n $.\index{केली - हैमिल्टन प्रमेय}
\end{theorem}

\begin{proof}
हल करना $ x \neq 0$ और चलो $ d $ के साथ अधिकतम हो $(x, u(x), \dots,
u^{d-1}(x))$ मुक्त; लिखना
\[
u^d(x) = -a_0 x - a_1 u(x) - \dots - a_{d-1}u^{d-1}(x),
\]
और सेट करें $ P_x = X^d + a_{d-1}X^{d-1} + \dots + a_0$, इसलिए $ P_x(u)(x) =
0$. मुक्त परिवार को आधार बनाकर पूरा करें $ E $: उस पर, $ u $ है
ब्लॉक फॉर्म $\begin{pmatrix} C & \ast\\ 0 & D\end{pmatrix}$ कहाँ $ C $
का साथी मैट्रिक्स है $ P_x $, किसका \omterm{def:b2:reduction:charpoly}{विशेषता बहुपद} है
$ P_x $ (बढ़ाना $\det(XI - C)$ पहले कॉलम के साथ, इंडक्शन द्वारा
$ d $). इस तरह $\chi_u = P_x \cdot \chi_D $, और
\[
\chi_u(u)(x) = \chi_D(u)\bigl(P_x(u)(x)\bigr) = 0 .
\]
यह तर्क हर किसी के लिए सही है $ x $: $\chi_u(u) = 0$.
\end{proof}

\begin{example}[केली--हैमिल्टन काम पर]\label{ex:b2:reduction:chwork}
$ A = \begin{pmatrix} 1 & 2\\ 3 & 4\end{pmatrix}$: $\chi_A = X^2 -
5X - 2$, इसलिए $ A^2 = 5A + 2I $. की हर शक्ति $ A $ ए में ढह जाता है
का संयोजन $ I $ और $ A $:
\[
A^4 = (5A + 2I)^2 = 25A^2 + 20A + 4I = 145A + 54I =
\begin{pmatrix} 199 & 290\\ 435 & 634\end{pmatrix},
\]
और उलटा मुफ़्त आता है: $ A(A - 5I) = 2I $ देता है
\[
A^{-1} = \tfrac12(A - 5I)
= \begin{pmatrix} -2 & 1\\ 3/2 & -1/2\end{pmatrix}.
\]
समापन अंतर्दृष्टि: केली--हैमिल्टन पूरे बीजगणित को संपीड़ित करता है
$ K[A]$ में $\operatorname{Vect}(I, A, \dots, A^{n-1})$ ---
$\dim K[A] = \deg\mu_A \leq n $, चाहे आपको जितनी भी बड़ी शक्तियों की आवश्यकता हो।
\end{example}

\begin{remark}[सामान्य ख़तरे]\label{rem:b2:reduction:pitfalls}
(मैं) eigenvalues जोडो मत: $\operatorname{Sp}(A + B)$ क्या नहीं है
$\operatorname{Sp}A + \operatorname{Sp}B $, और का योग
\omterm{def:b2:reduction:diag}{विकर्णीय} मैट्रिक्स की आवश्यकता नहीं है \omterm{def:b2:reduction:diag}{विकर्णीय} ---
$\begin{psmallmatrix}1 & 1\\ 0 & 0\end{psmallmatrix} +
\begin{psmallmatrix}0 & 0\\ 0 & 1\end{psmallmatrix} =
\begin{psmallmatrix}1 & 1\\ 0 & 1\end{psmallmatrix}$ का योग है
दो \omterm{def:b2:reduction:diag}{विकर्णीय} मैट्रिसेस (प्रत्येक का अलग-अलग होता है eigenvalues) और
नहीं है \omterm{def:b2:reduction:diag}{विकर्णीय}; केवल \emph{आने} परिवार व्यवहार करते हैं
(\cref{exo:b2:reduction:9}). (ii) ``$\chi_u $ स्प्लिट्स'' एक है
के बारे में परिकल्पना \emph{मैदान}: एक समतल घूर्णन होता है $\chi =
X^2 - 2\cos\theta\,X + 1$, से अधिक विभक्त $\C $, समाप्त नहीं हुआ है $\R $ ---
\omterm{def:b2:reduction:diag}{विकर्णीय} में $\mathcal{M}_2(\C)$, नहीं \omterm{def:b2:reduction:diag}{त्रिकोणीय} में
$\mathcal{M}_2(\R)$. (iii) असमानता ज्यामितीय चलती है $\leq $
बीजीय, कभी भी उलटा नहीं; केवल परीक्षण $\dim E_\lambda \geq
1$ के बारे में कुछ भी साबित नहीं करता विकर्णता. (iv) $\mu_u $ क्या नहीं है
$\chi_u $: समानता बिल्कुल तब लागू होती है जब प्रत्येक \omterm{def:b2:reduction:eigen}{eigenvalue} एक है
एकल ब्लॉक श्रृंखला (उदा.\ साथी मैट्रिक्स, इस अध्याय का
सप्ताहांत समस्या); का उपयोग करते हुए $\chi $ कहाँ $\mu $ फुलाने की जरूरत है
हर शक्ति गणना. (v) डनफोर्ड का $ d $ और $\nu $ हैं
बहुपद में $ u $ --- एक विघटन $ u = d' + \nu'$ के साथ
सही गुण लेकिन $ d'\nu' \neq \nu'd'$ है \emph{नहीं} डनफोर्ड
और कभी अद्वितीय नहीं होता.
\end{remark}

\begin{remark}[जहाँ इस अध्याय का प्रयोग किया जाता है]
न्यूनीकरण पुस्तक के शेष भाग का मुख्य कार्य है: शक्तियाँ और
मैट्रिक्स के घातांक रैखिक अंतर प्रणालियों को संचालित करते हैं
\cref{ch:b2:diffeq}; का वर्णक्रमीय प्रमेय \cref{ch:b2:quadratic}
क्या विकर्णीकरण को ओर्थोगोनल बनाया गया है; कार्य उत्पन्न करना
(\cref{ch:b2:genfun}) इसकी पुनरावृत्ति स्पर्शोन्मुखता को पुनः प्राप्त करें
अध्याय की सप्ताहांत समस्या विश्लेषणात्मक रूप से। वर्ष 3 खंड में
एक ही प्रोग्राम अनंत आयाम में चलता है: का वर्णक्रमीय सिद्धांत
कॉम्पैक्ट स्व-सहायक ऑपरेटर, जहां \omterm{def:b2:reduction:eigen}{eigenvalue} अनुक्रम प्रतिस्थापित करते हैं
परिमित स्पेक्ट्रा, और पेरोन-फ्रोबेनियस सकारात्मक सिद्धांत
मैट्रिक्स, जो समझाता है \emph{क्यों} प्रमुख eigenvalues का
गिनती की समस्याएँ सकारात्मक और सरल हैं।
\end{remark}

\section{निलपोटेंट्स और डनफोर्ड अपघटन}

\begin{proposition}[निलपोटेंट एंडोमोर्फिज्म]\label{prop:b2:reduction:nilpotent}
के लिए $ u $ साथ $\chi_u $ विभाजन, निम्नलिखित समतुल्य हैं: $ u^n =
0$; $ u^k = 0$ कुछ के लिए $ k $; $\operatorname{Sp}(u) = \{0\}$; $\chi_u
= X^n $; $ u $ है \omterm{def:b2:reduction:diag}{त्रिकोणीय} शून्य विकर्ण के साथ. ए \omterm{prop:b2:reduction:nilpotent}{निलपोटेंट
एंडोमोर्फिज्म} है $\mu_u = X^{\text{(nilpotence index)}}$, और सूचकांक
$\leq n $.\index{निलपोटेंट एंडोमोर्फिज्म}
\end{proposition}

\begin{proof}
$ u^k = 0$ हर बनाता है \omterm{def:b2:reduction:eigen}{eigenvalue} की एक जड़ $ X^k $: \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} $\{0\}$
(गैर-रिक्त जब $\chi $ विभाजन --- खत्म $\C $ हमेशा)। तब $\chi_u =
X^n $ (सभी मूल शून्य) और केली--हैमिल्टन देते हैं $ u^n = 0$;
त्रिकोणीकरण (\cref{thm:b2:reduction:trigonalization}) डालता है
विकर्ण पर शून्य (विकर्ण वहन करता है eigenvalues).
इसके विपरीत, चलो $ A $ सख्ती से ऊपरी \omterm{def:b2:reduction:diag}{त्रिकोणीय} हो: $ a_{ij} = 0$
के लिए $ j \leq i $. हम इसे प्रेरण द्वारा दिखाते हैं
\[
(A^k)_{ij} = 0 \qquad \text{whenever } j \leq i + k - 1,
\]
यानी\ प्रत्येक शक्ति शून्य क्षेत्र को एक विकर्ण ऊपर धकेलती है। के लिए
$ k = 1$ यह परिकल्पना है. कदम के लिए,
\[
(A^{k+1})_{ij} = \sum_{\ell} (A^k)_{i\ell}\,a_{\ell j} ,
\]
और प्रत्येक शब्द गायब हो जाता है: या तो $\ell \leq i + k - 1$ (पहला
कारक है $0$ प्रेरण द्वारा) या $\ell \geq i + k $, किस स्थिति में
$ j \leq i + k \leq \ell $ दूसरे कारक को मारता है। पर $ k = n $ स्थिति $ j \leq i + n - 1$ सभी के लिए धारण करता है $ i, j \leq n $: $ A^n =
0$. \omterm{def:b2:reduction:polyu}{न्यूनतम बहुपद} विभाजित $ X^n $ और विनाश
सूचकांक को परिभाषित करता है.
\end{proof}

\begin{theorem}[डनफोर्ड अपघटन]\label{thm:b2:reduction:dunford}
कल्पना करना $\chi_u $ विभाजित हो जाता है $ K $ (स्वचालित के लिए $ K = \C $). फिर
वहाँ एक है \emph{अद्वितीय} जोड़ा $(d, \nu)$ साथ
\[
u = d + \nu, \qquad d \text{ diagonalizable}, \quad \nu
\text{ nilpotent}, \quad d\nu = \nu d ,
\]
और इसके अलावा $ d $ और $\nu $ में बहुपद हैं
$ u $.\index{डनफोर्ड अपघटन}
\end{theorem}

\begin{proof}
\emph{अस्तित्व।} लिखना $\chi_u = \prod_{i=1}^{r} (X -
\lambda_i)^{m_i}$ (विशिष्ट $\lambda_i $) और सेट करें $ N_i = \ker(u -
\lambda_i)^{m_i}$, द \emph{विशेषता उपस्थान}. द्वारा
केली - हैमिल्टन और कर्नेल लेम्मा
(\cref{thm:b2:reduction:kernels}),
\[
E = N_1 \oplus \dots \oplus N_r ,
\]
अनुमानों के साथ $\pi_i $ बहुपद में $ u $; प्रत्येक $ N_i $ स्थिर है
(बहुपद में $ u $ के साथ आवागमन करें $ u $). परिभाषित करना $ d = \sum_i \lambda_i
\pi_i $: में एक बहुपद $ u $, \omterm{def:b2:reduction:diag}{विकर्णीय} (यह इस प्रकार कार्य करता है $\lambda_i $
पर $ N_i $, इसलिए $ E $ अपने में विघटित हो जाता है eigenspaces). तब $\nu = u -
d $ में एक बहुपद है $ u $ (इसलिए साथ आवागमन करता है $ d $), और प्रत्येक पर
$ N_i $ यह के रूप में कार्य करता है $ u - \lambda_i $, साथ $(u - \lambda_i)^{m_i} = 0$
वहाँ: $\nu^{\max m_i} = 0$ प्रत्येक सारांश पर, इसलिए $\nu $ शून्यशक्तिशाली है.

\emph{विशिष्टता.} होने देना $ u = d' + \nu'$ ऐसी एक और जोड़ी बनें. तब से
$ d'$ और $\nu'$ एक दूसरे के साथ आवागमन करते हैं, वे साथ आवागमन करते हैं $ u = d' +
\nu'$, इसलिए प्रत्येक बहुपद के साथ $ u $ ---विशेष रूप से के साथ $ d $
और $\nu $. तब $ d - d'$ है \omterm{def:b2:reduction:diag}{विकर्णीय} (दो यात्रा कर रहे हैं
\omterm{def:b2:reduction:diag}{विकर्णीय} नक्शे एक साथ हैं \omterm{def:b2:reduction:diag}{विकर्णीय}:
\cref{exo:b2:reduction:9}) और बराबर है $\nu' - \nu $, जो है
निलपोटेंट: यदि $\nu^k = 0$ और $\nu'^{k'} = 0$, रूपान्तरण
द्विपद विस्तार को लाइसेंस देता है
\[
(\nu' - \nu)^{k + k' - 1}
= \sum_{j=0}^{k+k'-1}\binom{k + k' - 1}{j}
\,\nu'^{\,j}\,(-\nu)^{k + k' - 1 - j} ,
\]
जिसमें प्रत्येक पद मर जाता है: या तो $ j \geq k'$ (पहला कारक
शून्य) या $ k + k' - 1 - j \geq k $ (दूसरा कारक शून्य), और एक
दोनों में से एक हमेशा कायम रहता है। ए \omterm{def:b2:reduction:diag}{विकर्णीय} निलपोटेंट शून्य है (इसका)
\omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} है $\{0\}$ और यह कुछ आधार पर विकर्ण है): $ d = d'$,
$\nu = \nu'$.
\end{proof}

\begin{example}[शक्तियां और घातांक]\label{ex:b2:reduction:powers}
$ A = \begin{pmatrix} 3 & 1\\ -1 & 1\end{pmatrix}$: $\chi_A = X^2 -
4X + 4 = (X-2)^2$, अकेला \omterm{def:b2:reduction:eigen}{eigenvalue} $2$, \omterm{def:b2:reduction:eigen}{eigenspace} आयाम का
$1$: नहीं \omterm{def:b2:reduction:diag}{विकर्णीय}. डनफोर्ड: $ D = 2I $, $ N = A - 2I =
\begin{pmatrix} 1 & 1\\ -1 & -1\end{pmatrix}$, $ N^2 = 0$. तब
\[
A^k = (2I + N)^k = 2^k I + k\,2^{k-1} N ,
\qquad
\eu^{tA} = \eu^{2t}(I + tN),
\]
कम्यूटिंग द्विपद प्रमेय द्वारा, सम्मान।\ घातीय श्रृंखला
(\cref{ch:b2:diffeq}) आने-जाने के \omterm{def:b2:structures:generated}{आदेश}ों पर विभाजन। कमी मोड़
मैट्रिक्स गतिकी को अदिश गतिकी में।
\end{example}

\begin{remark}[इस खंड के भीतर परिप्रेक्ष्य]\label{rem:b2:reduction:perspectives}
न्यूनीकरण एक केन्द्र है; यहाँ देखने लायक चार तीलियाँ हैं। में
\cref{ch:b2:nvs}, अनुकूलित मानदंड ``सभी को बदल देते हैं eigenvalues का
मापांक $< 1$'' कुछ ऑपरेटर मानदंड में $< 1$'', बनाना
स्पेक्ट्रा शक्तियों और श्रृंखला के अभिसरण को नियंत्रित करता है। में
\cref{ch:b2:diffeq}, की रेसिपी
\cref{ex:b2:reduction:powers} का सामान्य समाधान बन जाता है
$ X' = AX $: डनफोर्ड विभाजित हो गया $\eu^{tA}$ में
बहुपद-समय-घातीय ब्लॉक, और स्थिरता समाप्त हो जाती है
के वास्तविक भाग eigenvalues. में \cref{ch:b2:quadratic}, ए
अदिश उत्पाद वह बल देता है जो केवल रैखिक बीजगणित नहीं कर सकता: सममित
मैट्रिक्स बन जाते हैं \emph{ओर्थोगोनली} \omterm{def:b2:reduction:diag}{विकर्णीय}, नहीं के साथ
बिल्कुल शून्यशक्तिशाली भाग। और में \cref{ch:b2:genfun}, द
प्रमुख-आइजेनवैल्यू इस अध्याय के सप्ताहांत की स्पर्शोन्मुखता
समस्या विश्लेषणात्मक रूप से पुनः प्रकट होती है, a की सबसे छोटी विलक्षणता के रूप में
जनरेटिंग फ़ंक्शन --- एक विकास दर के लिए दो भाषाएँ।
\end{remark}

\begin{exercise}[$\star $]\label{exo:b2:reduction:2}
बताते हैं कि $ C = \begin{pmatrix} 1 & 1\\ 0 & 1\end{pmatrix}$ क्या नहीं है
\omterm{def:b2:reduction:diag}{विकर्णीय}, दो बार: के माध्यम से eigenspaces, और के माध्यम से \omterm{def:b2:reduction:polyu}{न्यूनतम

\end{exercise}

\begin{exercise}[$\star $]\label{exo:b2:reduction:3}
होने देना $ u $ संतुष्ट $ u^2 - 5u + 6\,\mathrm{id} = 0$. साबित करें कि $ u $ है
\omterm{def:b2:reduction:diag}{विकर्णीय}, संभावित स्पेक्ट्रा निर्धारित करें, और गणना करें $ u^k $
के संयोजन के रूप में $\mathrm{id}$ और $ u $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:reduction:4}
होने देना $ u $ होना \omterm{def:b2:reduction:diag}{विकर्णीय} और $ F $ एक स्थिर उपस्थान. साबित करें कि
$ u|_F $ है \omterm{def:b2:reduction:diag}{विकर्णीय} \emph{(एक विनाशकारी बहुपद को प्रतिबंधित करें
सरल विभाजित जड़ों के साथ)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:reduction:5}
(फाइबोनैचि) चलो $ A = \begin{pmatrix} 1 & 1\\ 1 & 0\end{pmatrix}$.
विकर्ण बनाना $ A $ ऊपर $\R $, और इसके लिए बिनेट का सूत्र निकालें
फाइबोनैचि अनुक्रम ($ F_0 = 0$, $ F_1 = 1$, $ F_{n+1} = F_n +
F_{n-1}$):
\[
F_n = \frac{\varphi^n - \psi^n}{\sqrt 5},
\qquad \varphi = \frac{1 + \sqrt5}{2},\ \psi = \frac{1 -
\sqrt5}{2}.
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:reduction:6}
होने देना $ u \in \mathcal{L}(E)$ साथ $ u^2$ \omterm{def:b2:reduction:diag}{विकर्णीय} और $ u $
उलटा ($ K = \C $). साबित करें कि $ u $ है \omterm{def:b2:reduction:diag}{विकर्णीय}. ए दीजिए
प्रति उदाहरण जब $ u $ उलटा नहीं है.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:reduction:7}
की गणना करें \omterm{thm:b2:reduction:dunford}{डनफोर्ड अपघटन}, $ A^k $, और $\eu^{tA}$ के लिए
\[
A = \begin{pmatrix} 2 & 1 & 0\\ 0 & 2 & 1\\ 0 & 0 & 2
\end{pmatrix}.
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:reduction:8}
होने देना $ A \in \mathcal{M}_n(\C)$ साथ $ A^k = I $ कुछ के लिए $ k \geq 1$.
साबित करें कि $ A $ है \omterm{def:b2:reduction:diag}{विकर्णीय} और इसके eigenvalues हैं $ k $-वें
एकता की जड़ें. एक परिमित-क्रम व्युत्क्रमणीय जटिल मैट्रिक्स का निष्कर्ष निकालें
इकाई विकर्ण के साथ एक \omterm{def:b2:reduction:diag}{त्रिकोणीय} मैट्रिक्स के समान पहचान है।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:reduction:9}
(एक साथ विकर्णीकरण) चलो $ u, v $ होना \omterm{def:b2:reduction:diag}{विकर्णीय} और
आवागमन. साबित करो कि वे हैं \emph{इसके साथ ही}
\omterm{def:b2:reduction:diag}{विकर्णीय}: कुछ आधार दोनों को विकर्णित करते हैं। \emph{(प्रत्येक \omterm{def:b2:reduction:eigen}{eigenspace}
का $ u $ है $ v $-स्थिर; के प्रतिबंधों को विकर्णित करें $ v $ वहाँ,
का उपयोग कर रहे हैं \cref{exo:b2:reduction:4}.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:reduction:10}
होने देना $ u \in \mathcal{L}(\C^n)$. साबित करें कि $ u $ है \omterm{def:b2:reduction:diag}{विकर्णीय} यदि
और केवल यदि प्रत्येक $ u $-स्थिर उपस्थान में एक है $ u $-स्थिर
पूरक उपस्थान. \emph{(के लिए $\Leftarrow $: संपत्ति लागू करें
को $ F = \sum_\lambda E_\lambda(u)$, सबका योग eigenspaces; यदि ए
स्थिर पूरक $ G $ शून्येतर, त्रिकोणीकरण कर रहे थे $ u|_G $ होगा
एक उत्पादन करें आइजन्वेक्टर का $ u $ अंदर $ G $ --- विरोधाभासी $ G \cap
F = \{0\}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b2:reduction:11}
(गेलफैंड-लाइट द्वारा वर्णक्रमीय त्रिज्या, $2\times2$ विश्लेषण का स्वाद
आओ) चलो $ A \in \mathcal{M}_2(\C)$ दोनों के साथ eigenvalues का
मापांक $< 1$. साबित करें कि $ A^k \to 0$ प्रवेशवार के रूप में $ k \to \infty $.
\emph{(त्रिकोणीकरण करें: $ A = P(T)P^{-1}$ साथ $ T $ ऊपरी \omterm{def:b2:reduction:diag}{त्रिकोणीय};
गणना $ T^k $ स्पष्ट रूप से --- समान और विशिष्ट भेद करें
eigenvalues --- और बाध्य।)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b2:reduction:12}
होने देना $ u \in \mathcal{L}(\C^n)$ साथ $\operatorname{rk} u = 1$ ($ n
\geq 2$). बताते हैं कि $\chi_u = X^{n-1}(X - \operatorname{tr} u)$,
ओर वो $ u $ है \omterm{def:b2:reduction:diag}{विकर्णीय} अगर और केवल अगर $\operatorname{tr} u
\neq 0$. \emph{(से याद करें \cref{exo:b2:linalg:5} वह $ u^2 =
(\operatorname{tr} u)\,u $.)}
\end{exercise}

\section{समस्या: रैखिक पुनरावृत्तियाँ और सहयोगी मैट्रिक्स}

एक रेखीय पुनरावृत्ति $ u_{n+k} = a_{k-1}u_{n+k-1} + \dots + a_0 u_n $
भेष में एक मैट्रिक्स शक्ति है, और कमी इसे परिवर्तित करती है
बंद सूत्र, विकास दर और त्रुटि अनुमान। इस सप्ताहांत
समस्या शब्दकोश विकसित करती है --- एक पर साथी मैट्रिक्स
एक ओर, दूसरी ओर अनुक्रमों के स्थान पर शिफ्ट ऑपरेटर
--- सिद्ध करता है \emph{रैखिक पुनरावृत्तियों का मौलिक प्रमेय}
(सामान्य समाधान है $\sum_i Q_i(n)\lambda_i^n $ जड़ों के ऊपर
का \omterm{def:b2:reduction:charpoly}{विशेषता बहुपद}), और लाभांश खर्च करता है
डायोफैंटाइन सन्निकटन $\sqrt2$, चलने की गिनती पर
और शब्द, और केवल युग्मित अनुक्रमों की एक अंगूठी पर
एक साथ विकर्णीकरण उलझाव को सुलझा सकता है।

\begin{problem}[{सप्ताहांत समस्या --- का मौलिक प्रमेय
रैखिक पुनरावृत्ति}]
\label{pb:b2:reduction:1}
हल करना $ k \geq 1$, अदिश $ a_0, \dots, a_{k-1} \in \C $ साथ $ a_0
\neq 0$, राक्षसी बहुपद $ P = X^k - a_{k-1}X^{k-1} - \dots -
a_1 X - a_0$, और पुनरावृत्ति
\[
(\mathcal R)\colon\quad u_{n+k} = a_{k-1}u_{n+k-1} + \dots +
a_1 u_{n+1} + a_0 u_n \qquad (n \geq 0).
\]
 \emph{साथी मैट्रिक्स} का $ P $ है
\[
C =
\begin{pmatrix}
0 & 1 & & \\
 & \ddots & \ddots & \\
 & & 0 & 1\\
a_0 & a_1 & \cdots & a_{k-1}
\end{pmatrix}
\in \mathcal{M}_k(\C).
\]

\medskip
\noindent\textbf{Part I --- The companion dictionary.}
\begin{enumerate}
  \item वह क्रम दिखाओ $(u_n)$ संतुष्ट $(\mathcal R)$ यदि
        और केवल यदि सदिश $ v_n = (u_n, u_{n+1}, \dots,
        u_{n+k-1})^{\mathsf T}$ संतुष्ट $ v_{n+1} = Cv_n $, इस तरह
        $ v_n = C^n v_0$.
  \item साबित करें कि $\chi_C = P $ (बढ़ाना $\det(XI - C)$ साथ में
        पहला कॉलम और शामिल करें $ k $), फिर उस $\mu_C = P $
        भी \emph{(को पास किया $ C^{\mathsf T}$, जिसके लिए $ e_1$
        है \omterm{def:b2:structures:generated}{चक्रीय}, और ध्यान दें कि एक मैट्रिक्स और उसका \omterm{def:b2:linalg:transpose}{खिसकाना} एक ही लें \omterm{def:b2:reduction:polyu}{न्यूनतम बहुपद})}.
  \item प्रत्येक रूट के लिए इसे दिखाएँ $\lambda $ का $ P $, वेक्टर
        $(1, \lambda, \dots, \lambda^{k-1})^{\mathsf T}$ फैला हुआ
        द \omterm{def:b2:reduction:eigen}{eigenspace} का $ C $ के लिए $\lambda $; उसका निष्कर्ष निकालो
        \emph{प्रत्येक} \omterm{def:b2:reduction:eigen}{eigenspace} का $ C $ आयाम है $1$, ओर वो $ C $ है \omterm{def:b2:reduction:diag}{विकर्णीय} अगर और केवल अगर $ P $ है $ k $
        विशिष्ट जड़ें.
  \item कल्पना करना $ P $ विशिष्ट जड़ें हैं $\lambda_1, \dots,
        \lambda_k $. दिखाएँ कि ज्यामितीय अनुक्रम
        $(\lambda_i^n)_n $ समाधानों के स्थान का आधार बनाएं
        का $(\mathcal R)$, तो हर समाधान है $ u_n = \sum_i
        c_i\lambda_i^n $ अद्वितीय स्थिरांक के लिए $ c_i $.
  \item पूर्णतः हल करें: $ u_{n+2} = u_{n+1} + 6u_n $, $ u_0 = 1$,
        $ u_1 = 8$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The shift operator and the
fundamental theorem.} होने देना $\mathcal{S}$ हो $\C $-वेक्टर स्थान
सभी जटिल अनुक्रमों का और $ S \in
\mathcal{L}(\mathcal{S})$ स्थानान्तरित करना, $ S\bigl((u_n)_n\bigr) =
(u_{n+1})_n $.
\begin{enumerate}[resume]
  \item दिखाएँ कि समाधान सेट $(\mathcal R)$ है $\ker
        P(S)$, और इसका सटीक आयाम है $ k $ \emph{(मानचित्र ए
        इसके प्रारंभिक मूल्यों का समाधान)}.
  \item बताएं कि \omterm{thm:b2:reduction:kernels}{कर्नेल अपघटन लेम्मा} क्यों है
        (\cref{thm:b2:reduction:kernels}) पर लागू होता है $ S $ पर
        अनंत-आयामी $\mathcal{S}$ बिना किसी बदलाव के,
        और परिणामी अपघटन लिखिए $\ker P(S)$ के लिए
        $ P = \prod_{i=1}^{r}(X - \lambda_i)^{m_i}$ (विशिष्ट
        $\lambda_i $, तब से सभी गैर-शून्य $ a_0 \neq 0$).
  \item के लिए $\lambda \neq 0$ और $ m \geq 1$, दिखाओ
        \[
        \ker\,(S - \lambda\,\mathrm{id})^m
        = \bigl\{\,\bigl(Q(n)\,\lambda^n\bigr)_n : Q \in
        \C_{m-1}[X]\,\bigr\},
        \]
        आयाम का $ m $. \emph{(गणना करें $(S -
        \lambda)\bigl(Q(n)\lambda^n\bigr) =
        \lambda^{n+1}(\Delta Q)(n)$ साथ $\Delta Q = Q(X + 1) -
        Q(X)$, और उसका उपयोग करें $\Delta $ डिग्री गिरा देता है; के लिए
        आयाम, इससे बंधे $ m $ प्रारंभिक मूल्यों के माध्यम से।)}
  \item (रैखिक पुनरावृत्ति का मौलिक प्रमेय) निष्कर्ष निकालें:
        यदि $ P = \prod_{i=1}^{r}(X - \lambda_i)^{m_i}$ साथ
        $\lambda_i $ विशिष्ट और शून्येतर, के समाधान
        $(\mathcal R)$ बिल्कुल अनुक्रम हैं
        \[
        u_n = \sum_{i=1}^{r} Q_i(n)\,\lambda_i^n,
        \qquad Q_i \in \C_{m_i - 1}[X],
        \]
        विशिष्ट रूप से निर्धारित बहुपदों के साथ $ Q_i $.
  \item पूर्णतः हल करें: $ u_{n+2} = 4u_{n+1} - 4u_n $, $ u_0 =
        1$, $ u_1 = 0$, और उत्तर की जाँच करें $ u_2$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Dominant roots and Diophantine
dividends.}
\begin{enumerate}[resume]
  \item मान लीजिए कि जड़ें सरल हैं $\abs{\lambda_1} >
        \abs{\lambda_i}$ के लिए $ i \geq 2$, और $ u_n = \sum_i c_i
        \lambda_i^n $ साथ $ c_1 \neq 0$. दिखाओ $ u_n \sim
        c_1\lambda_1^n $ और $ u_{n+1}/u_n \to \lambda_1$.
  \item (पेल) परिभाषित करें $ a_{n+1} = a_n + 2b_n $, $ b_{n+1} = a_n +
        b_n $, $ a_0 = b_0 = 1$. बताते हैं कि $ q(a, b) = a^2 - 2b^2$
        संतुष्ट $ q(a_{n+1}, b_{n+1}) = -q(a_n, b_n)$, इस तरह
        $ a_n^2 - 2b_n^2 = (-1)^{n+1}$; इसे इससे संबंधित करें
        सिद्ध का $ M = \begin{psmallmatrix}1 & 2\\ 1 &
        1\end{psmallmatrix}$.
  \item त्रुटि अनुमान निकालें
        \[
        \Bigl|\frac{a_n}{b_n} - \sqrt2\Bigr|
        = \frac{1}{b_n\,(a_n + \sqrt2\,b_n)}
        \leq \frac{1}{2b_n^2},
        \]
        और दिखाएँ कि यह अनुपात के साथ ज्यामितीय रूप से क्षय होता है $3 -
        2\sqrt2$ \emph{(खोजें eigenvalues का $ M $ और
        की वृद्धि $ b_n $)}.
  \item (सामान्य वृद्धि) प्रश्न 9 से सिद्ध करें: (ए) यदि प्रत्येक
        जड़ संतुष्ट करती है $\abs{\lambda_i} \leq \rho $, तब
        $\abs{u_n} \leq C\,n^{m-1}\rho^n $ साथ $ m = \max_i m_i $;
        (बी) यदि कोई अद्वितीय जड़ है $\lambda_1$ अधिकतम का
        मापांक और $ Q_1 \neq 0$, तब $ u_{n+1}/u_n \to
        \lambda_1$ --- इसे प्रश्न 10 के हल पर जाँचें।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Counting walks and words.} एक के लिए
शीर्ष सेट के साथ परिमित ग्राफ़ $\{1, \dots, N\}$, द
\emph{सहखंडज मैट्रिक्स} $ A $ है $ A_{ij} = 1$ अगर $ ij $ एक किनारा है,
अन्यथा $0$.
\begin{enumerate}[resume]
  \item साबित करें कि $(A^n)_{ij}$ लंबाई में चलने की संख्या है
        $ n $ से $ i $ को $ j $ (के अनुक्रम $ n $ किनारे, प्रत्येक चरण
        एक किनारे के साथ)।
  \item (त्रिभुज) संपूर्ण ग्राफ़ के लिए $3$ शिखर,
        $ A = J - I $: का उपयोग \omterm{def:b2:reduction:eigen}{स्पेक्ट्रम} का $ J $
        (\cref{ex:b2:linalg:onesmatrix}), दिखाओ
        \[
        (A^n)_{ii} = \frac{2^n + 2(-1)^n}{3},
        \qquad
        (A^n)_{ij} = \frac{2^n - (-1)^n}{3} \quad (i \neq j),
        \]
        और दोनों की जाँच करें $ n = 2$ चलने की सूची बनाकर।
  \item (बिना शब्द $11$) होने देना $ w_n $ बाइनरी की संख्या हो
        लंबाई के शब्द $ n $ लगातार दो के साथ नहीं $1$'एस।
        \omterm{def:b2:structures:sn}{स्थानांतरण} प्राप्त करने के लिए शब्दों को उनके अंतिम अक्षर से कूटबद्ध करें
        मैट्रिक्स, दिखाओ $ w_{n+2} = w_{n+1} + w_n $, निष्कर्ष निकालना $ w_n =
        F_{n+2}$ (फाइबोनैचि, \cref{exo:b2:reduction:5}), और
        विकास दर दीजिए $\lim w_{n+1}/w_n $.
  \item (पथ) पथ ग्राफ़ के लिए $1 - 2 - 3$, दिखाएं
        eigenvalues का $ A $ हैं $\sqrt2, 0, -\sqrt2$ साथ
        eigenvectors $(1, \pm\sqrt2, 1)$ और $(1, 0, -1)$, और
        लंबाई में चलने की संख्या ज्ञात कीजिए $ n $ अंत से
        ख़त्म करना है $\bigl((\sqrt2)^n + (-\sqrt2)^n\bigr)/4$:
        विषम के लिए शून्य $ n $, और $2^{\,n/2 - 1}$ एक जैसे के लिए $ n $.
        पर जांचें $ n = 4$.
  \item (ट्रेस सूत्र) दिखाएँ कि बंद की कुल संख्या
        लंबाई की सैर $ n $ (सभी शुरुआती बिंदु) है
        $\operatorname{tr}(A^n) = \sum_i \lambda_i^n $, और
        इसे त्रिभुज पर सत्यापित करें।
\end{enumerate}

\medskip
\noindent\textbf{Part V --- A ring of sequences: simultaneous
diagonalization.} हल करना $ k \geq 3$, होने देना $\omega = \eu^{2\iu\pi/k}$,
और चलो $ W \in \mathcal{M}_k(\C)$ हो \omterm{def:b2:structures:generated}{चक्रीय} बदलाव: $ W e_i =
e_{i+1}$ (सूचकांक मॉड $ k $, कॉलम अनुक्रमित $0, \dots, k-1$).
\begin{enumerate}[resume]
  \item बताते हैं कि $ W^{\mathsf T}$ का साथी मैट्रिक्स है
        $ X^k - 1$, निष्कर्ष निकालना $\chi_W = \mu_W = X^k - 1$, ओर वो $ W $ है
        \omterm{def:b2:reduction:diag}{विकर्णीय} साथ $ k $ सरल eigenvalues
        $\omega^j $ और eigenvectors $ f_j = (1, \omega^{-j},
        \omega^{-2j}, \dots, \omega^{-(k-1)j})^{\mathsf T}$.
  \item ए \emph{परिचालित} मैट्रिक्स है $ C = c_0 I + c_1 W + \dots
        + c_{k-1}W^{k-1}$. दिखाएँ कि सभी परिचालक आवागमन करते हैं,
        वह आधार $(f_0, \dots, f_{k-1})$ विकर्णीकरण करता है
        \emph{सभी} उनमें से एक साथ, और वह
        eigenvalues का $ C $ हैं $\widehat c(\omega^j) = \sum_m
        c_m \omega^{jm}$, $ j = 0, \dots, k-1$.
  \item परिणाम निकालना $\det C = \prod_{j=0}^{k-1} \widehat
        c(\omega^j)$, और उसकी जांच करें $ k = 3$ को पुनः प्राप्त करता है
        का गुणनखंडन \cref{exo:b2:linalg:8}.
  \item (हार औसत) चलो $ x^{(n+1)} = Mx^{(n)}$ साथ
        $ M = \frac12(W + W^{-1})$: की प्रत्येक $ k $ संख्याओं की व्यवस्था की गई
        एक रिंग में उसके दो के औसत को बदल दिया जाता है
        पड़ोसी. दिखाएं eigenvalues का $ M $ हैं
        $\cos(2\pi j/k)$, और वह का गुणांक $ x^{(0)}$
        पर $ f_0$ मतलब है $\frac1k\sum_m x^{(0)}_m $
        \emph{(के निर्देशांकों का योग करें $ f_j $)}.
  \item निष्कर्ष: विषम के लिए $ k $, $ x^{(n)}$ में परिवर्तित हो जाता है
        स्थिर सदिश जिसका मान प्रारंभिक का माध्य है
        मूल्य; के लिए $ k = 4$, प्रदर्शित करें \omterm{def:b2:reduction:eigen}{eigenvalue} जिम्मेदार
        गैर-अभिसरण और सटीक रुकावट के लिए
        (एक प्रत्यावर्ती-माध्य गुणांक जो लुप्त हो जाना चाहिए)।
  \item (संश्लेषण) एक-एक वाक्य में: कैसे साथी
        मैट्रिक्स का विश्लेषण परिवर्तित करता है $(\mathcal R)$ में
        कमी; जहां \omterm{thm:b2:reduction:kernels}{कर्नेल अपघटन लेम्मा} की आवश्यकता है
        कोई सीमित आयाम नहीं; प्रभुत्वशाली क्यों eigenvalues शासन
        विकास दर और डायोफैंटाइन त्रुटि; की शक्तियां क्यों
        आसन्न मैट्रिक्स गिनती चलता है; और क्या आना-जाना
        मैट्रिसेस खरीदें। दो शिखरों के नाम बताएं: मौलिक
        रैखिक पुनरावृत्तियों का प्रमेय, और --- सकारात्मक के लिए
        भाग IV के आव्यूह, वर्ष 3 खंड में ---
        पेरोन--फ्रोबेनियस प्रमेय।
\end{enumerate}
\end{problem}
