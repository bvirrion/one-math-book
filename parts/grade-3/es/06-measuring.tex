\chapter{Medición}\label{ch:g3:measure}

¿Cuánto tiempo, cuánto pesa, cuánto aguanta, qué tan tarde es? Medición
convierte el mundo en números, siempre que todos usen el mismo
unidades. Este capítulo presenta las unidades de longitud, \omterm{def:g2:measure:units}{masa} y \omterm{def:g2:measure:units}{capacidad},
y enseña a leer un reloj.

\section{Longitudes}

\begin{definition}[Unidades de longitud]\label{def:g3:measure:length}
La unidad básica de longitud es el \emph{metro} (m). Para longitudes pequeñas:
el centímetro (cm) y el milímetro (mm); para largas distancias: el
kilómetro (km).
\[
1 \text{ m} = 100 \text{ cm}, \qquad
1 \text{ cm} = 10 \text{ mm}, \qquad
1 \text{ km} = 1\,000 \text{ m}.
\]
\end{definition}

\begin{example}[Elegir la unidad adecuada]\label{ex:g3:measure:choose}
Una hormiga: aproximadamente $5$ mm. Un lápiz: unos $18$ cm. Una puerta: aproximadamente $2$ m.
Un paseo hasta el siguiente pueblo: unos $3$ km. Dando "la longitud de un
el lápiz es $18$'' no significa nada --- la unidad es parte de la respuesta.
\end{example}

\begin{method}[Conversión de unidades]\label{met:g3:measure:convert}
Para convertir, multiplica o divide por $10$, $100$ o $1\,000$:
\begin{enumerate}
  \item de una unidad grande a una pequeña, \emph{multiplicar}:
        $3$ m $= 3 \times 100 = 300$ cm;
  \item de una unidad pequeña a una grande, \emph{dividir}:
        $250$ cm $= 250 \div 100 = 2$ m y $50$ cm;
  \item para agregar longitudes, conviértalas primero a la unidad \emph{mismo}:
        $1$ m $+ 35$ cm $= 100 + 35 = 135$ cm.
\end{enumerate}
\end{method}

\section{Masas y capacidades}

\begin{definition}[Unidades de masa y capacidad]\label{def:g3:measure:mass}
Las masas se miden en \emph{gramos} (g) y \emph{kilogramos} (kg),
capacidades (cuánto cabe en un contenedor) en \emph{litros} (L) y
\emph{centilitros} (CL):
\[
1 \text{ kg} = 1\,000 \text{ g},
\qquad
1 \text{ L} = 100 \text{ cL}.
\]
\end{definition}

\begin{example}\label{ex:g3:measure:mass}
Una carta: aproximadamente $20$ g. Un saco de harina: $1$ kg. Una bañera: aproximadamente
$150$ L. Un vaso: aproximadamente $20$ cL. Para equilibrar una balanza que sostiene una
$1$ kg de saco contra $650$ g de manzanas, añadir $1\,000 - 650 = 350$ g sobre
el lado de las manzanas.
\end{example}

\section{leyendo el reloj}

\begin{method}[leyendo la hora]\label{met:g3:measure:clock}
En un reloj con manecillas:
\begin{enumerate}
  \item la manecilla \emph{corto} da la hora: lee el número que tiene
        pasó último;
  \item la manecilla \emph{largo} da los minutos: cada marca grande es $5$
        minutos, por lo que una manecilla larga en $3$ significa $15$ minutos;
  \item después de media hora también se puede leer ``hasta la próxima hora'':
        $10{:}50$ es "once menos diez".
\end{enumerate}
\end{method}

\begin{example}[Duración]\label{ex:g3:measure:duration}
La película comienza en $14{:}40$ y termina en $16{:}10$. ¿Cuánto dura?
Cuente en dos saltos:
\begin{enumerate}
  \item de $14{:}40$ a $15{:}00$: $20$ minutos;
  \item de $15{:}00$ a $16{:}10$: $1$ hora $10$ minutos.
\end{enumerate}
Total: $1$ h $30$ mín. Cuidado: una hora tiene $60$ minutos, así que
¡$16{:}10$ menos $14{:}40$ no se pueden calcular como $1610 - 1440$!
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g3:measure:1}
¿Qué unidad usarías (mm, cm, mo km) para: el ancho de un clavo;
la altura de un compañero de clase; la longitud del patio de la escuela; el
¿Distancia entre dos pueblos?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:2}
Convertir: $5$ m a cm; \quad $70$ mm en cm; \quad $4$ km en m;
\quad $600$ cm en m.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:3}
Mide con tu regla (al milímetro más cercano) los lados de un
triángulo dibujado por un compañero de clase y calcular la longitud total de su
frontera.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:4}
Calcule, convirtiendo primero: $2$ m $+ 45$ cm (en cm); \quad
$1$ km $+ 250$ m (en m); \quad $3$ cm $+ 8$ mm (en mm).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:5}
Convertir: $3$ kg a g; \quad $2\,500$ g en kg y g; \quad
$4$ L en cL; \quad $350$ cL en L y cL.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:6}
Una receta necesita $750$ g de harina. Mona tiene un saco de $1$ kg. ella tiene
suficiente? ¿Cuánto quedará?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:7}
Lea los tiempos: la mano corta está entre $2$ y $3$, la mano larga
en el $6$; la manecilla corta del $9$, la manecilla larga del $12$;
la manecilla corta casi en el $5$, la manecilla larga en el $10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:8}
La escuela comienza a las $8{:}30$ y la mañana termina a las $11{:}45$. cuanto tiempo
es la mañana? (Cuente en saltos, como en
\cref{ex:g3:measure:duration}.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:9}
El tren sale a las $9{:}50$ y el viaje dura $45$ minutos. ¿En qué?
hora llega? (Salte primero a $10{:}00$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:measure:10}
Una botella contiene $1$ L. Paul llena vasos de $20$ cL. cuantos llenos
¿Puede servir vasos? (Convierta primero).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:measure:11}
Lena camina $600$ m hasta la escuela y de regreso todos los días. cuantos metros
¿Camina en una semana escolar de $5$ días? Da la respuesta en metros,
luego en kilómetros.
\end{exercise}
