\chapter{Compartir y dividir}\label{ch:g3:division}

Veinte canicas repartidas equitativamente entre cuatro niños: ¿cuántas cada uno?
La división responde a todas las preguntas sobre compartir --- y la agrupación
preguntas también --- y es el gemelo de la multiplicación: uno deshace el
otro.

\section{Dos caras de la división}

\begin{definition}[División]\label{def:g3:division:def}
La división $20 \div 4$ responde a dos tipos de preguntas:
\begin{itemize}
  \item \emph{intercambio}: $20$ canicas compartidas entre $4$ niños ---
        ¿Cuántas canicas \emph{cada}? ($5$ cada uno);
  \item \emph{agrupamiento}: Canicas $20$ puestas en bolsas de $4$ --- cuantas
        \emph{bolsas}? ($5$ bolsas).
\end{itemize}
Ambas preguntas tienen la misma respuesta, porque ambas deshacen lo mismo
multiplicación: $4 \times 5 = 20$.
\end{definition}

\begin{method}[Dividiendo con las tablas]\label{met:g3:division:tables}
Para calcular $42 \div 6$, pregunte: ``¿$6$ multiplicado por \emph{qué} da $42$?''
La tabla de $6$ responde: $6 \times 7 = 42$, entonces $42 \div 6 = 7$.
Cada pregunta de división es una pregunta de multiplicación leída.
hacia atrás.
\end{method}

\section{Cuando no sale ni siquiera}

\begin{definition}[Resto]\label{def:g3:division:remainder}
Compartir canicas $23$ entre niños $4$ da canicas $5$ a cada uno ---
y $3$ canicas \emph{sobrante}, no suficientes para otra ronda. Nosotros
escribir
\[
23 = 4 \times 5 + 3 ,
\]
y diga: el \emph{cociente}\index{cociente} es $5$, el
\emph{resto}\index{resto} es $3$. El resto es siempre
menor que el número de niños --- de lo contrario, otra ronda de
compartir sería posible.
\end{definition}

\begin{example}\label{ex:g3:division:remainder}
Divida $58$ por $7$. Escanee la tabla de $7$:
$7 \times 8 = 56$ es el \omterm{def:g2:mult:def}{producto} más grande que cabe debajo de $58$
($7 \times 9 = 63$ es demasiado). entonces
\[
58 = 7 \times 8 + 2 :
\]
\omterm{def:g3:division:remainder}{cociente} $8$, \omterm{def:g3:division:remainder}{resto} $2$. Verifique: $56 + 2 = 58$ y $2 < 7$.
\end{example}

\begin{method}[Interpretando el resto]\label{met:g3:division:interpret}
Después de dividir, regrese a la historia:
\begin{enumerate}
  \item ``¿cuántos cada uno/cuántos grupos completos?'' --- el \omterm{def:g3:division:remainder}{cociente};
  \item ``¿cuántos sobran?'' --- el \omterm{def:g3:division:remainder}{resto};
  \item ``¿cuántas cajas caben todo?'' --- el \omterm{def:g3:division:remainder}{cociente}
        \emph{más uno} si el \omterm{def:g3:division:remainder}{resto} no es \omterm{def:g1:counting:zero}{cero}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:division:interpret}
$58$ alumnos van en canoa, $7$ por canoa. $58 = 7 \times 8 + 2$:
ocho canoas están llenas y quedan estudiantes $2$ --- necesitan una canoa
también. Por lo tanto, se deben reservar las canoas $9$.
\end{example}

\section{Mitades, tercios, cuartos}

\begin{definition}[Mitad, tercio, cuarto]\label{def:g3:division:half}
El \emph{medio}\index{medio} de un número es el resultado de dividirlo entre $2$; el
\emph{tercero}, por $3$; el \emph{cuarto}\index{cuarto}, por $4$. Por ejemplo el
la mitad de $18$ es $9$, la tercera parte de $18$ es $6$, la cuarta parte de $36$
es $9$.
\end{definition}

\begin{example}[Dobles y mitades van juntos]\label{ex:g3:division:double}
El \omterm{def:g3:division:half}{medio} de $46$: desde $46 = 40 + 6$, \omterm{def:g3:division:half}{medio} de cada parte da
$20 + 3 = 23$. Consulta con el doble: $23 + 23 = 46$. sabiendo
duplica de memoria ($25 \to 50$, $45 \to 90$, $150 \to 300$) hace
mitades instantáneas.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g3:division:1}
Calcular usando las tablas al revés: $36 \div 4$; \quad$63 \div 9$;
\quad$40 \div 5$; \quad $56 \div 8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:2}
Para cada división, escribe la igualdad $a = b \times q + r$ y da
el \omterm{def:g3:division:remainder}{cociente} y el \omterm{def:g3:division:remainder}{resto}: $17 \div 5$; \quad$30 \div 4$; \quad
$50 \div 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:3}
Las canicas $28$ se comparten entre los niños $6$. ¿Cuántas canicas cada una?
y cuantos sobran? Dibuja el compartir si te ayuda.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:4}
Calcular: el \omterm{def:g3:division:half}{medio} de $68$; el tercero de $27$; el \omterm{def:g3:division:half}{cuarto} de $48$;
el \omterm{def:g3:division:half}{medio} de $150$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:5}
¿Verdadero o falso? Compruébalo con una multiplicación.
``$45 \div 5 = 9$''; \quad ``$52 \div 6 = 8$ \omterm{def:g3:division:remainder}{resto} $4$'';
\quad ``$70 \div 8 = 9$ \omterm{def:g3:division:remainder}{resto} $2$''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:6}
Las fotos $32$ están pegadas en un álbum, $4$ por página. cuantas paginas son
usado? ¿Qué \omterm{def:g2:shapes:solids}{rostro} de división es esta (compartir o agrupar)?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:7}
Una cuerda de $75$ cm se corta en trozos de $9$ cm. ¿Cuántas piezas completas?
¿Y cuánto mide el trozo sobrante?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:8}
$50$ los niños van de viaje en coches de $4$ plazas. cuantos autos son
necesario? (Cuidado: ¡todos deben viajar!)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:9}
Encuentra los números que faltan: $? \div 7 = 6$; \quad
$54 \div ? = 6$; \quad el \omterm{def:g3:division:half}{medio} de $?$ es $35$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:division:10}
Amina comparte sus pegatinas $47$ entre sus amigos $5$, de la manera más justa.
posible y se queda con el resto para ella. ¿Cuántas pegatinas tiene?
cada amigo recibe, y ¿cuántos se queda Amina?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:division:11}
Cuando se divide un número por $6$, ¿cuáles son todas las posibles?
\omterm{def:g3:division:remainder}{restos}? ¿Cuál es el mayor número cuya división por $6$ tiene
\omterm{def:g3:division:remainder}{cociente}$7$?
\end{exercise}
