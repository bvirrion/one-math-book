\chapter{Números hasta 10\,000}\label{ch:g3:numbers}

Mil, dos mil, tres mil\,\dots\ los números grandes son
construido a partir de pequeños, utilizando una idea maravillosa: el \emph{lugar} de un
dígito indica su valor. Este capítulo enseña cómo leer, escribir,
compare y solicite números hasta $10\,000$.

\section{Leer y escribir números.}

\begin{definition}[valor posicional]\label{def:g3:numbers:place}
Se escribe un número con los dígitos $0, 1, 2, \dots, 9$. Leyendo de
a la derecha, los lugares son: unidades, decenas, centenas, miles. en
$3\,254$:
\[
3\,254 = 3 \text{ thousands} + 2 \text{ hundreds} + 5 \text{ tens}
+ 4 \text{ units}.
\]
\end{definition}

\begin{example}[El cero que ocupa un lugar]\label{ex:g3:numbers:zero}
``Cuatro mil siete'' está escrito $4\,007$: un $4$ en el
miles, un $7$ en las unidades y un \emph{\omterm{def:g1:counting:zero}{ceros}} en el medio para mantener
cada dígito en su lugar correcto. Sin ellos, $47$ sería un muy
número diferente!
\end{example}

\begin{example}[Intercambio]\label{ex:g3:numbers:exchange}
Diez unidades suman una decena; diez decenas suman cien; diez centenas hacen
mil. Entonces $10$ centenas $= 1\,000$ y $32$ decenas
$= 320$: el número $320$ contiene $32$ decenas (no solo el dígito
$2$!).
\end{example}

\section{Comparar y ordenar}

\begin{method}[Comparando dos números]\label{met:g3:numbers:compare}
\begin{enumerate}
  \item Cuente los dígitos: más dígitos significa un número mayor
        ($1\,002 > 998$).
  \item Mismo número de dígitos: comparar dígito por dígito comenzando desde
        el \emph{izquierda}. La primera diferencia decide:
        $5\,614 > 5\,589$ porque $6 > 5$ en las centenas.
  \item Escribe la respuesta con los signos $<$ (menor que) o $>$
        (mayor que): el lado abierto siempre \omterm{def:g2:shapes:solids}{caras} el número mayor.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:numbers:order}
Ordene de menor a mayor: $780$, $8\,700$, $807$, $87$.
Primero por el número de dígitos: $87$ (dos), luego $780$ y $807$
(tres), luego $8\,700$ (cuatro). Entre $780$ y $807$: comparar el
centenas de dígitos, $7 < 8$, por lo que $780 < 807$. Orden final:
\[
87 < 780 < 807 < 8\,700 .
\]
\end{example}

\section{par e impar}

\begin{definition}[par, impar]\label{def:g3:numbers:evenodd}
Un número es \emph{incluso}\index{número par} cuando se puede dividir en
dos partes enteras iguales --- su dígito de unidades es $0$, $2$, $4$, $6$ o
$8$. De lo contrario, es \emph{extraño}\index{número impar} (dígitos de unidades $1$,
$3$, $5$, $7$ o $9$).
\end{definition}

\begin{example}
$46$ es \omterm{def:g3:numbers:evenodd}{incluso} ($46 = 23 + 23$); $47$ es \omterm{def:g3:numbers:evenodd}{extraño} (dos partes iguales serían
Necesito medio objeto). Sólo importa el dígito \emph{último}: $3\,578$ es
\omterm{def:g3:numbers:evenodd}{incluso}, $2\,341$ es \omterm{def:g3:numbers:evenodd}{extraño}.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g3:numbers:1}
Escriba en dígitos: ``dos mil trescientos cuarenta y cinco''; ``seis
mil cincuenta''; ``nueve mil uno''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:2}
Escribe en palabras: $1\,208$; \quad $4\,070$; \quad $9\,999$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:3}
En $7\,382$: ¿qué dígito está en el lugar de las decenas? en miles
lugar? ¿Cuántos \emph{cientos} contiene el número en total?
(cuidado: no sólo el dígito de las centenas, piense en
\cref{ex:g3:numbers:exchange})?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:4}
Descomponer como en \cref{def:g3:numbers:place}:
$5\,631$; \quad$2\,046$; \quad $8\,500$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:5}
Copiar y completar con $<$ o $>$:
\[
456 \;?\; 465, \qquad
1\,203 \;?\; 989, \qquad
6\,090 \;?\; 6\,009, \qquad
9\,999 \;?\; 10\,000 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:6}
Ordenar de menor a mayor: $530$; $3\,500$; $353$; $5\,033$;
$503$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:7}
En una recta numérica graduada en centenas de $0$ a $1\,000$, coloque
(aproximadamente) los números $150$, $480$, $820$ y $990$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:8}
\omterm{def:g3:numbers:evenodd}{Incluso} or \omterm{def:g3:numbers:evenodd}{extraño}? \ $34$; \ $57$; \ $70$; \ $2\,481$; \ $6\,548$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:9}
Cuente por pasos: desde $2\,970$, cuente cinco pasos de $10$; de $860$,
cuente cuatro pasos de $100$; desde $9\,996$, cuente cuatro pasos de $1$.
¿Qué notas al final del último?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:numbers:10}
Usando cada uno de los dígitos $3$, $7$, $0$, $5$ exactamente una vez, escriba el
mayor número posible de cuatro dígitos, luego el más pequeño (un número
no puede comenzar con $0$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:numbers:11}
Soy un número de tres dígitos. Mis dígitos son $2$, $5$ y $8$, soy \omterm{def:g3:numbers:evenodd}{incluso},
y soy más grande que $800$. ¿Quién soy yo?
\end{exercise}
