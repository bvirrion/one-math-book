\chapter{Multiplicación}\label{ch:g3:mult}

Tres bolsas de siete canicas: podrías agregar $7 + 7 + 7$, pero hay
una forma más rápida: multiplicación, la operación de suma repetida.
Este capítulo construye las tablas de multiplicar a partir de imágenes, luego
enseña el método de la columna.

\section{que significa multiplicar}

\begin{definition}[multiplicacion]\label{def:g3:mult:def}
\omterm{def:g2:mult:def}{producto} $3 \times 7$ significa "tres por siete":
\[
3 \times 7 = 7 + 7 + 7 = 21 .
\]
En una imagen, $3 \times 7$ es un rectángulo de puntos: $3$ filas de $7$
puntos.
\end{definition}

{\small Los mismos puntos $21$, contados por filas o por columnas: el orden

\end{omfigure}

\begin{proposition}[Reglas que ayudan]\label{prop:g3:mult:rules}
\begin{enumerate}
  \item El orden no importa: $a \times b = b \times a$.
  \item Multiplicar por $1$ no cambia nada; multiplicar por $0$ da
        $0$.
  \item Al multiplicar por $10$ se agrega un \omterm{def:g1:counting:zero}{cero}: $34 \times 10 = 340$;
        por $100$, dos \omterm{def:g1:counting:zero}{ceros}.
  \item Se puede reconstruir una tabla a partir de sus vecinos:
        $6 \times 8 = 5 \times 8 + 8 = 40 + 8 = 48$.
\end{enumerate}
\end{proposition}
\admitted

\section{Multiplicación de columnas}

\begin{method}[Multiplicar por un número de un dígito]\label{met:g3:mult:column}
Para calcular $234 \times 6$:
\begin{enumerate}
  \item multiplica las unidades: $4 \times 6 = 24$: escribe $4$, lleva $2$;
  \item multiplica las decenas y suma el acarreo:
        $3 \times 6 + 2 = 20$: escribe $0$, lleva $2$;
  \item multiplica las centenas y suma el acarreo:
        $2 \times 6 + 2 = 14$: escribe $14$.
\end{enumerate}
\[
\begin{array}{r}
2\,3\,4 \\
\times \quad 6 \\
\hline
1\,4\,0\,4
\end{array}
\]
Estimación para verificar: $234$ está cerca de $200$, y
$200 \times 6 = 1\,200$: la respuesta $1\,404$ es plausible.
\end{method}

\begin{example}[Dividir para multiplicar mentalmente]\label{ex:g3:mult:split}
El método de la columna divide secretamente el número por lugares. mentalmente, hazlo
lo mismo:
\[
23 \times 4 = (20 + 3) \times 4 = 20 \times 4 + 3 \times 4
= 80 + 12 = 92 .
\]
Y con un vecino amigable: $99 \times 5 = 100 \times 5 - 5 =
495$.
\end{example}

\section{Multiplicación en problemas}

\begin{example}\label{ex:g3:mult:problem}
Un minibús transporta pasajeros $9$. ¿Cuántos pasajeros pueden $27$?
llevan los minibuses?
\begin{enumerate}
  \item Es una multiplicación: $27$ grupos de $9$.
  \item $27 \times 9 = 243$ (columnas, o
        $27 \times 9 = 27 \times 10 - 27 = 270 - 27$).
  \item Frase: los minibuses pueden transportar pasajeros $243$.
\end{enumerate}
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g3:mult:1}
Escribe como una multiplicación, luego calcula:
$8 + 8 + 8 + 8$; \quad$5 + 5 + 5$; \quad
$20 + 20 + 20 + 20 + 20$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:2}
Recitar y completar: $6 \times 7$; $8 \times 4$; $9 \times 9$;
$7 \times 8$; $6 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:3}
Dibuja un rectángulo de puntos para $4 \times 6$, luego uno para $6 \times 4$.
¿Qué notas sobre los dos cargos?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:4}
Calcule usando las reglas de \cref{prop:g3:mult:rules}:
$56 \times 10$; \quad$8 \times 100$; \quad $73 \times 0$; \quad
$45 \times 1$; \quad $30 \times 10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:5}
Calcular en columnas: $132 \times 3$; \quad$217 \times 4$; \quad
$408 \times 7$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:6}
Calcule mentalmente dividiendo (como en \cref{ex:g3:mult:split}):
$31 \times 5$; \quad$42 \times 3$; \quad $25 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:7}
Calcular con un vecino amigable: $99 \times 7$; \quad
$101 \times 6$; \quad $98 \times 5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:8}
Primero estime, luego calcule: $389 \times 5$ (estima con $400$);
$612 \times 8$ (estimación con $600$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:9}
Una caja de huevos contiene huevos $6$. ¿Cuántos huevos hay en las cajas $38$? escribir
la operación, calcular, responder con una frase.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:mult:10}
Un salón de clases tiene $8$ filas de escritorios $4$; cada escritorio tiene capacidad para estudiantes $2$.
¿Cuántos estudiantes tiene capacidad para el salón? (Dos multiplicaciones).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:mult:11}
Encuentre todas las formas de organizar las sillas $24$ en filas iguales (filas de
$1$, de $2$, \dots). ¿Qué arreglos son posibles? (Usa la tabla
cuadrado: busque $24$ dentro de él).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:mult:12}
Zoe calcula $47 \times 6$ así: $40 \times 6 = 240$,
$7 \times 6 = 42$ y luego $240 + 42 = 282$. Lea calcula
$47 \times 6 = 50 \times 6 - 3 \times 6 = 300 - 18 = 282$. explicar
cada método. ¿Cuál prefieres para $38 \times 4$? Calcular ambos
maneras.
\end{exercise}
