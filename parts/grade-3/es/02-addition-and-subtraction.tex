\chapter{Suma y resta}\label{ch:g3:addsub}

Sumar significa juntar; restar significa quitar, o
encontrar lo que falta. Este capítulo entrena los métodos de dos columnas.
--- con sus famosos acarreos y préstamos --- y los trucos mentales
que a menudo hacen que las columnas sean innecesarias.

\section{Adición de columnas}

\begin{method}[Añadiendo en columnas]\label{met:g3:addsub:add}
\begin{enumerate}
  \item Escribe los números uno debajo del otro, unidades debajo de unidades,
        decenas bajo decenas (alinearse en el \emph{bien});
  \item agregar las unidades; si el \omterm{def:g1:addition:def}{suma} pasa $9$, escriba el dígito de sus unidades
        y \emph{llevar} la decena a la siguiente columna (una pequeña $1$);
  \item continuamos con las decenas, las centenas, \dots, sumando el carry
        cada vez.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:addsub:add}
Calcular $457 + 368$:
\[
\begin{array}{r}
{\scriptstyle 1}\ \ {\scriptstyle 1}\ \ \phantom{0} \\[-3pt]
4\,5\,7 \\
+\ 3\,6\,8 \\
\hline
8\,2\,5
\end{array}
\]
Unidades: $7 + 8 = 15$: escriba $5$, lleve $1$. Decenas:
$5 + 6 + 1 = 12$: escribe $2$, lleva $1$. Cientos: $4 + 3 + 1 = 8$.
Resultado: $825$.
\end{example}

\begin{example}[adicion mental]\label{ex:g3:addsub:mental}
Busque números amigables antes de buscar columnas:
\begin{itemize}
  \item \emph{hacer un diez}: $47 + 8 = 47 + 3 + 5 = 50 + 5 = 55$;
  \item \emph{saltar por partes}: $256 + 130 = 256 + 100 + 30 = 386$;
  \item \emph{reordenar}: $17 + 59 + 3 = (17 + 3) + 59 = 79$.
\end{itemize}
\end{example}

\section{Resta de columnas}

\begin{method}[Resta en columnas]\label{met:g3:addsub:sub}
\begin{enumerate}
  \item Escribe el número mayor arriba, alineado a la derecha;
  \item restar columna por columna, partiendo de las unidades;
  \item si el dígito superior es demasiado pequeño, \emph{pedir prestado}: toma uno de
        el siguiente lugar en la fila superior (vale diez en la fila actual)
        columna);
  \item \emph{controlar}: el resultado más el número restado debe dar
        retroceda el número superior.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:addsub:sub}
Calcular $603 - 147$:
\[
\begin{array}{r}
6\,0\,3 \\
-\ 1\,4\,7 \\
\hline
4\,5\,6
\end{array}
\]
Unidades: $3 - 7$ imposible; pedir prestado a través de: $603$ se convierte
$5$ centenas, $9$ decenas, $13$ unidades. Entonces $13 - 7 = 6$;
decenas $9 - 4 = 5$; cientos $5 - 1 = 4$. Resultado: $456$.
Verificar: $456 + 147 = 603$.
\end{example}

\begin{example}[Resta como parte faltante]\label{ex:g3:addsub:missing}
``Lila tiene $35$ canicas, Tom tiene $52$: ¿cuántas más tiene Tom?'' es
la resta $52 - 35$. Mentalmente, cuente \emph{up} desde $35$:
$35 \to 40$ es $5$, luego $40 \to 52$ es $12$; en total
$5 + 12 = 17$. Contar es a menudo más fácil que pedir prestado.
\end{example}

\section{Órdenes de magnitud}

\begin{method}[Estimación antes de calcular]\label{met:g3:addsub:estimate}
Antes de cualquier cálculo de columnas, redondee los números a números amigables y
calcular un \emph{estimar}. Después de calcular, compare: un resultado lejano
de la estimación indica un error.
\end{method}

\begin{example}
Para $487 + 312$: la estimación es $500 + 300 = 800$. la columna
El resultado, $799$, es cercano: plausible. Si hubieras encontrado $1\,099$
(una alineación perdida), la estimación le habría advertido.
\end{example}

\begin{exercise}[$\star$]\label{exo:g3:addsub:7}

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:8}

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:9}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:addsub:10}
