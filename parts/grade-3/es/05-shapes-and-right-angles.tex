\chapter{Formas y ángulos rectos}\label{ch:g3:shapes}

Cuadrados, rectángulos, triángulos, círculos: las formas son viejos amigos.
Este capítulo los analiza con ojos de geómetra: ¿qué es exactamente?
hace que un cuadrado sea un cuadrado? --- e introduce la herramienta que decide:
la escuadra y su \omterm{def:g3:shapes:rightangle}{ángulo recto}.

\section{El ángulo correcto}

\begin{definition}[ángulo recto]\label{def:g3:shapes:rightangle}
A \emph{ángulo recto}\index{ángulo recto} es el ángulo de esquina de un
hoja de papel, o de la escuadra. Dos líneas que forman un ángulo recto.
son \emph{perpendicular}. En las figuras, un ángulo recto está marcado con un
pequeña plaza en la esquina.
\end{definition}

\begin{method}[Comprobando un ángulo recto]\label{met:g3:shapes:check}
\begin{enumerate}
  \item Coloque la esquina de la escuadra exactamente en la esquina para
        prueba;
  \item deslice un borde de la escuadra a lo largo de un lado del ángulo;
  \item si el otro lado del ángulo corre exactamente a lo largo del otro
        borde, el ángulo es correcto --- si se abre más o menos,
        no lo es.
\end{enumerate}
\end{method}

\section{Las formas y sus reglas.}

\begin{definition}[Cuadrado, rectángulo, triángulo]\label{def:g3:shapes:shapes}
\begin{itemize}
  \item A \emph{rectángulo}\index{rectángulo} tiene lados $4$ y $4$
        \omterm{def:g3:shapes:rightangle}{ángulos rectos}; sus lados opuestos tienen la misma longitud.
  \item A \emph{cuadrado}\index{cuadrado} es un rectángulo cuyos lados $4$
        todos tienen la misma longitud.
  \item A \emph{triángulo}\index{triángulo} tiene lados $3$; un
        \emph{triangulo rectángulo} tiene un \omterm{def:g3:shapes:rightangle}{ángulo recto}.
  \item Se dibuja con el compás un \emph{círculo}\index{círculo}: todos
        sus puntos están a la misma distancia (el \emph{radio}\index{radio}) de
        el \emph{centro}.
\end{itemize}
\end{definition}

\begin{example}[Por qué son importantes las marcas]\label{ex:g3:shapes:marks}
Una forma dibujada ligeramente inclinada puede \emph{mirar} como un cuadrado sin
siendo uno. Las marcas dicen la verdad: $4$ cuadrados en ángulo recto $+$ $4$
marcas de verificación $=$ un cuadrado real. Cuando dibujes, pon las marcas; cuando tu
lea una figura, confíe sólo en las marcas y las medidas dadas.
\end{example}

\section{Dibujo sobre papel cuadriculado}

\begin{method}[Construcciones en retícula]\label{met:g3:shapes:grid}
El papel cuadriculado ofrece \omterm{def:g3:shapes:rightangle}{ángulos rectos} gratis: sus líneas son perpendiculares.
\begin{enumerate}
  \item Para dibujar un rectángulo $5$ por $3$: elija un punto de cruce,
        cuente $5$ cuadrados hacia la derecha, $3$ cuadrados hacia arriba y cierre la forma
        a lo largo de las líneas de la cuadrícula;
  \item para comprobar longitudes iguales, contar cuadrados;
  \item los pasos diagonales forman líneas inclinadas: yendo ``$1$ hacia la derecha, $1$ hacia arriba''
        Dibuja repetidamente una línea $45^\circ$.
\end{enumerate}
\end{method}

\begin{example}[Un programa de construcción]\label{ex:g3:shapes:program}
Una figura se puede describir mediante instrucciones:
``1. Dibuja un cuadrado $ABCD$ de lado $4$ cm. 2. Dibuja el círculo con
centro $A$ pasando por $B$.'' Siguiendo un programa línea por línea,
cualquiera puede volver a dibujar la misma figura, y escribir una te obliga a
nombrar cada punto. Pruébalo con un compañero: uno escribe, el otro
dibuja.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g3:shapes:1}
En el aula, encuentre (sin medir) tres \omterm{def:g3:shapes:rightangle}{ángulos rectos} y
un ángulo que no es correcto. ¿Cómo puedes \emph{controlar} cada uno?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:2}
Dibuja un rectángulo de $6$ cm por $4$ cm con regla y escuadra. marca
su \omterm{def:g3:shapes:rightangle}{ángulos rectos} y sus lados iguales.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:3}
Dibuja un cuadrado de lado $5$ cm. ¿Cuántos \omterm{def:g3:shapes:rightangle}{ángulos rectos} tiene?
¿Cuantos lados iguales? Marca todo.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:4}
En papel cuadriculado, dibuja: un cuadrado $4 \times 4$; a $7 \times 2$
rectángulo; un triángulo rectángulo con catetos de cuadrados $3$ y $5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:5}
¿Verdadero o falso? Explicar.
``Cada cuadrado es un rectángulo.'' \quad
``Todo rectángulo es un cuadrado.'' \quad
``Un triángulo puede tener dos \omterm{def:g3:shapes:rightangle}{ángulos rectos}.'' (¡Intenta dibujar uno!)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:6}
Dibuja un círculo de \omterm{def:g3:shapes:shapes}{radio} $4$ cm. Marque su centro $O$, un punto $A$ en
el círculo y dibuja el \omterm{def:g3:shapes:shapes}{radio} $[OA]$. ¿Cuánto dura $OA$? Y si $B$
es otro punto del círculo, ¿cuánto mide $OB$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:7}
Siga este programa: ``1. Dibuja un rectángulo $ABCD$ con $AB = 6$ cm
y $BC = 3$ cm. 2. Marque el punto $M$ en medio de $[AB]$.
3. Une $M$ a $C$ y a $D$.'' ¿Qué formas ha tenido el rectángulo?
¿cortar?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:8}
Ordena las formas: ¿cuál de ellas tiene al menos un \omterm{def:g3:shapes:rightangle}{ángulo recto}?
Un cuadrado; un rectángulo (no cuadrado); un triángulo rectángulo; un círculo; el
Letra L dibujada con trazos gruesos.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:shapes:9}
En papel cuadriculado, dibuje un cuadrado inclinado cuyos lados vayan ``$2$ a la derecha, $1$
arriba'' (y los giros correspondientes). Cuenta cuadrados para convencerte
los cuatro lados tienen la misma longitud.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:shapes:10}
Escribir un programa de construcción (instrucciones numeradas, puntos nombrados)
para: un cuadrado de lado $3$ cm colocado encima de un rectángulo
$3$ cm $\times$ $2$ cm, como una casa en su planta baja. prueba tu
programa en un compañero de clase.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:shapes:11}
¿Cuántos cuadrados puedes encontrar en una cuadrícula $3 \times 3$ de pequeños
cuadrados? (Cuente el $1 \times 1$, el $2 \times 2$ y el
$3 \times 3$ unos.)
\end{exercise}
