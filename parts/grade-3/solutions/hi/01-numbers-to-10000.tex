\section*{अध्याय \ref{ch:g3:numbers} --- 10 तक की संख्या\,000}

\begin{solution}{exo:g3:numbers:1}
$2\,345$; \quad $6\,050$; \quad $9\,001$.
\end{solution}

\begin{solution}{exo:g3:numbers:2}
$1\,208$: एक हजार दो सौ आठ. \quad
$4\,070$: चार हजार सत्तर. \quad
$9\,999$: नौ हजार नौ सौ निन्यानवे।
\end{solution}

\begin{solution}{exo:g3:numbers:3}
दहाई का अंक: $8$. हज़ार अंक: $7$. कुल मिलाकर सैकड़ों शामिल हैं:
$73$ (क्योंकि $7\,382 = 73$ सैकड़ों $+ 82$, अर्थात\
$73 \times 100 + 82$).
\end{solution}

\begin{solution}{exo:g3:numbers:4}
$5\,631 = 5$ हजारों $+ 6$ सैकड़ों $+ 3$ दसियों $+ 1$ इकाई।

$2\,046 = 2$ हजारों $+ 0$ सैकड़ों $+ 4$ दसियों $+ 6$ इकाइयाँ।

$8\,500 = 8$ हजारों $+ 5$ सैकड़ों.
\end{solution}

\begin{solution}{exo:g3:numbers:5}
$456 < 465$ (दस: $5 < 6$); \quad
$1\,203 > 989$ (चार अंक तीन को हरा दें); \quad
$6\,090 > 6\,009$ (दस: $9 > 0$); \quad
$9\,999 < 10\,000$.
\end{solution}

\begin{solution}{exo:g3:numbers:6}
$353 < 503 < 530 < 3\,500 < 5\,033$.
\end{solution}

\begin{solution}{exo:g3:numbers:7}
$150$ बीच में आधा बैठता है $100$ और $200$; $480$ से ठीक पहले
$500$; $820$ तुरन्त बाद $800$; $990$ लगभग पर $1\,000$.
\end{solution}

\begin{solution}{exo:g3:numbers:8}
\omterm{def:g3:numbers:evenodd}{यहां तक की}: $34$, $70$, $6\,548$ (अंतिम अंक $4$, $0$, $8$). \omterm{def:g3:numbers:evenodd}{विषम}: $57$,
$2\,481$ (अंतिम अंक $7$, $1$).
\end{solution}

\begin{solution}{exo:g3:numbers:9}
से $2\,970$: $2\,980$, $2\,990$, $3\,000$, $3\,010$, $3\,020$.

से $860$: $960$, $1\,060$, $1\,160$, $1\,260$.

से $9\,996$: $9\,997$, $9\,998$, $9\,999$, $10\,000$ --- आखिरी
चरण पाँच अंकों की संख्या में पार हो जाता है: हमारे पूरे अध्याय की सीमा!
\end{solution}

\begin{solution}{exo:g3:numbers:10}
सबसे बड़ा: $7\,530$ (अंक घटते क्रम में). सबसे छोटा: $3\,057$
(पहले सबसे छोटा अशून्य अंक, फिर शेष बढ़ता हुआ --- $0$
के ठीक बाद $3$).
\end{solution}

\begin{solution}{exo:g3:numbers:11}
से भी बड़ा $800$: सैकड़े का अंक है $8$. \omterm{def:g3:numbers:evenodd}{यहां तक की}: इकाई अंक
होना ही चाहिए $2$ (प्रत्येक का उपयोग करके $2$, $5$, $8$ एक बार)। तो संख्या है
$852$.
\end{solution}
