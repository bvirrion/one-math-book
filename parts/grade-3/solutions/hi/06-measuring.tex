\section*{अध्याय \ref{ch:g3:measure} --- मापने}

\begin{solution}{exo:g3:measure:1}
नाखून की चौड़ाई: मिमी. सहपाठी की ऊंचाई: सेमी (या मी)। विद्यालय प्रांगण: एम.
शहरों के बीच की दूरी: किमी.
\end{solution}

\begin{solution}{exo:g3:measure:2}
$5$ एम $= 500$ सेमी; \quad $70$ मिमी $= 7$ सेमी; \quad
$4$ किमी $= 4\,000$ एम; \quad $600$ सेमी $= 6$ एम।
\end{solution}

\begin{solution}{exo:g3:measure:3}
उत्तर त्रिभुज पर निर्भर करते हैं; सीमा की कुल लंबाई है
\omterm{def:g1:addition:def}{जोड़} मापी गई तीन भुजाओं में से (इस कुल को कहा जाता है
\emph{परिधि}, में अध्ययन किया \cref{ch:g4:measure}).
\end{solution}

\begin{solution}{exo:g3:measure:4}
$2$ एम $+ 45$ सेमी $= 200 + 45 = 245$ सेमी।

$1$ किमी $+ 250$ एम $= 1\,000 + 250 = 1\,250$ एम।

$3$ सेमी $+ 8$ मिमी $= 30 + 8 = 38$ मिमी.
\end{solution}

\begin{solution}{exo:g3:measure:5}
$3$ किग्रा $= 3\,000$ जी; \quad $2\,500$ जी $= 2$ किलो और $500$ जी; \quad
$4$ एल $= 400$ सीएल; \quad $350$ क्लोरीन $= 3$ भूमि $50$ सीएल.
\end{solution}

\begin{solution}{exo:g3:measure:6}
$1$ किग्रा $= 1\,000$ जी $> 750$ जी: हाँ, उसके पास पर्याप्त है, और
$1\,000 - 750 = 250$ जी बचेगा.
\end{solution}

\begin{solution}{exo:g3:measure:7}
बीच में छोटा हाथ $2$ और $3$, पर लंबा हाथ $6$: $2{:}30$.
छोटा हाथ $9$, लंबा हाथ $12$: $9{:}00$. छोटा हाथ लगभग चालू
$5$, लंबा हाथ $10$: $4{:}50$ (``दस से पांच'' --- घंटे की सुई
तक नहीं पहुंचा है $5$ अभी तक)।
\end{solution}

\begin{solution}{exo:g3:measure:8}
से $8{:}30$ को $9{:}00$: $30$ मिनट; से $9{:}00$ को $11{:}45$:
$2$ एच $45$ मि. कुल: $3$ एच $15$ मि.
\end{solution}

\begin{solution}{exo:g3:measure:9}
से $9{:}50$ को $10{:}00$: $10$ मिनट; वहाँ रहते हैं
$45 - 10 = 35$ मिनट के बाद $10{:}00$. आगमन: $10{:}35$.
\end{solution}

\begin{solution}{exo:g3:measure:10}
$1$ एल $= 100$ सीएल, और $100 = 20 \times 5$: पॉल डाल सकते हैं $5$ पूर्ण
चश्मा.
\end{solution}

\begin{solution}{exo:g3:measure:11}
प्रत्येक दिन: $600 + 600 = 1\,200$ एम। पाँच दिनों में:
$1\,200 \times 5 = 6\,000$ एम $= 6$ किमी.
\end{solution}
