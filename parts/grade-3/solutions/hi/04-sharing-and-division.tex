\section*{अध्याय \ref{ch:g3:division} --- साझा करना और विभाजन}

\begin{solution}{exo:g3:division:1}
$36 \div 4 = 9$ (तब से $4 \times 9 = 36$); \quad
$63 \div 9 = 7$; \quad $40 \div 5 = 8$; \quad $56 \div 8 = 7$.
\end{solution}

\begin{solution}{exo:g3:division:2}
$17 = 5 \times 3 + 2$: \omterm{def:g3:division:remainder}{भागफल} $3$, \omterm{def:g3:division:remainder}{शेष} $2$.

$30 = 4 \times 7 + 2$: \omterm{def:g3:division:remainder}{भागफल} $7$, \omterm{def:g3:division:remainder}{शेष} $2$.

$50 = 6 \times 8 + 2$: \omterm{def:g3:division:remainder}{भागफल} $8$, \omterm{def:g3:division:remainder}{शेष} $2$.
\end{solution}

\begin{solution}{exo:g3:division:3}
$28 = 6 \times 4 + 4$: प्रत्येक बच्चे को मिलता है $4$ पत्थर, और $4$ पत्थर
बचे हुए हैं.
\end{solution}

\begin{solution}{exo:g3:division:4}
\omterm{def:g3:division:half}{आधा} का $68$: $34$. का तीसरा $27$: $9$. \omterm{def:g3:division:half}{तिमाही} का $48$: $12$. \omterm{def:g3:division:half}{आधा}
का $150$: $75$.
\end{solution}

\begin{solution}{exo:g3:division:5}
``$45 \div 5 = 9$'': सत्य, $5 \times 9 = 45$.

``$52 \div 6 = 8$ \omterm{def:g3:division:remainder}{शेष} $4$'': सत्य, $6 \times 8 + 4 = 52$ और
$4 < 6$.

``$70 \div 8 = 9$ \omterm{def:g3:division:remainder}{शेष} $2$'': असत्य --- $8 \times 9 + 2 = 74$,
नहीं $70$. सही: $70 = 8 \times 8 + 6$.
\end{solution}

\begin{solution}{exo:g3:division:6}
$32 \div 4 = 8$ पन्ने. यह है \emph{समूहन}: फोटो लगाए गए हैं
के समूहों में $4$, और हम समूहों की गिनती करते हैं।
\end{solution}

\begin{solution}{exo:g3:division:7}
$75 = 9 \times 8 + 3$: आठ पूर्ण टुकड़े, और एक बचा हुआ टुकड़ा
$3$ सेमी।
\end{solution}

\begin{solution}{exo:g3:division:8}
$50 = 4 \times 12 + 2$: बारह गाड़ियाँ चलती हैं $48$ बच्चे, और $2$
बच्चों को अभी भी सवारी की जरूरत है. $13$ कारों की जरूरत है.
\end{solution}

\begin{solution}{exo:g3:division:9}
$? = 7 \times 6 = 42$. \quad
$54 \div 9 = 6$, इसलिए लुप्त भाजक है $9$. \quad
वह संख्या जिसका \omterm{def:g3:division:half}{आधा} है $35$ है $70$.
\end{solution}

\begin{solution}{exo:g3:division:10}
$47 = 5 \times 9 + 2$: प्रत्येक मित्र को मिलता है $9$ स्टिकर, और अमीना
रखता है \omterm{def:g3:division:remainder}{शेष}, $2$ स्टिकर.
\end{solution}

\begin{solution}{exo:g3:division:11}
द्वारा विभाजित करना $6$, द \omterm{def:g3:division:remainder}{शेष} हमेशा से छोटा होता है $6$: यह हो सकता है
$0$, $1$, $2$, $3$, $4$ या $5$. के साथ सबसे बड़ी संख्या \omterm{def:g3:division:remainder}{भागफल} $7$
सबसे बड़ा उपयोग करता है \omterm{def:g3:division:remainder}{शेष}: $6 \times 7 + 5 = 47$.
\end{solution}
