\begin{solution}{exo:g3:mult:1}
$8 + 8 + 8 + 8 = 4 \times 8 = 32$; \quad
$5 + 5 + 5 = 3 \times 5 = 15$; \quad
$20 \times 5 = 100$.
\end{solution}

\begin{solution}{exo:g3:mult:6}

\begin{solution}{exo:g3:mult:7}
$99 \times 7 = 700 - 7 = 693$; \quad
$101 \times 6 = 600 + 6 = 606$; \quad
$98 \times 5 = 500 - 10 = 490$.
\end{solution}

\begin{solution}{exo:g3:mult:10}

\begin{solution}{exo:g3:mult:11}
$24$ तालिका में वर्गाकार रूप में दिखाई देता है $1 \times 24$, $2 \times 12$,
$3 \times 8$, $4 \times 6$ --- और वही \omterm{def:g1:addition:def}{जोड़}े उलट गए। संभव
व्यवस्थाएँ: $1$ की पंक्ति $24$, $2$ की पंक्तियाँ $12$, $3$ की पंक्तियाँ $8$,
$4$ की पंक्तियाँ $6$, $6$ की पंक्तियाँ $4$, $8$ की पंक्तियाँ $3$, $12$ की पंक्तियाँ $2$,
$24$ की पंक्तियाँ $1$.
\end{solution}

\begin{solution}{exo:g3:mult:12}
ज़ो विभाजित हो जाता है $47$ में $40 + 7$ (उसके सिर में स्तंभ विधि); लीया
प्रतिस्थापित करता है $47$ मित्रवत द्वारा $50$ और दूर करके सही करता है
$3 \times 6$ बहुत ज्यादा गिना गया. के लिए $38 \times 4$: ज़ो का रास्ता,
$120 + 32 = 152$; लीया का रास्ता, $40 \times 4 - 2 \times 4 = 160 - 8 =
152$. दोनों काम करते हैं --- यहां लीया जल्दी है क्योंकि $38$ के करीब है
$40$.
\end{solution}
