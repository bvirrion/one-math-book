\section*{Chapter \ref{ch:g3:measure} --- Measuring}

\begin{solution}{exo:g3:measure:1}
Nail width: mm. Classmate's height: cm (or m). Schoolyard: m.
Distance between towns: km.
\end{solution}

\begin{solution}{exo:g3:measure:2}
$5$ m $= 500$ cm; \quad $70$ mm $= 7$ cm; \quad
$4$ km $= 4\,000$ m; \quad $600$ cm $= 6$ m.
\end{solution}

\begin{solution}{exo:g3:measure:3}
Answers depend on the triangle; the total length of the border is the
\omterm{def:g1:addition:def}{sum} of the three measured sides (this total is called the
\emph{perimeter}, studied in \cref{ch:g4:measure}).
\end{solution}

\begin{solution}{exo:g3:measure:4}
$2$ m $+ 45$ cm $= 200 + 45 = 245$ cm.

$1$ km $+ 250$ m $= 1\,000 + 250 = 1\,250$ m.

$3$ cm $+ 8$ mm $= 30 + 8 = 38$ mm.
\end{solution}

\begin{solution}{exo:g3:measure:5}
$3$ kg $= 3\,000$ g; \quad $2\,500$ g $= 2$ kg and $500$ g; \quad
$4$ L $= 400$ cL; \quad $350$ cL $= 3$ L and $50$ cL.
\end{solution}

\begin{solution}{exo:g3:measure:6}
$1$ kg $= 1\,000$ g $> 750$ g: yes, she has enough, and
$1\,000 - 750 = 250$ g will be left.
\end{solution}

\begin{solution}{exo:g3:measure:7}
Short hand between $2$ and $3$, long hand on the $6$: $2{:}30$.
Short hand on $9$, long hand on $12$: $9{:}00$. Short hand almost on
$5$, long hand on $10$: $4{:}50$ (``ten to five'' --- the hour hand
has not reached the $5$ yet).
\end{solution}

\begin{solution}{exo:g3:measure:8}
From $8{:}30$ to $9{:}00$: $30$ min; from $9{:}00$ to $11{:}45$:
$2$ h $45$ min. Total: $3$ h $15$ min.
\end{solution}

\begin{solution}{exo:g3:measure:9}
From $9{:}50$ to $10{:}00$: $10$ min; there remain
$45 - 10 = 35$ min after $10{:}00$. Arrival: $10{:}35$.
\end{solution}

\begin{solution}{exo:g3:measure:10}
$1$ L $= 100$ cL, and $100 = 20 \times 5$: Paul can pour $5$ full
glasses.
\end{solution}

\begin{solution}{exo:g3:measure:11}
Each day: $600 + 600 = 1\,200$ m. In five days:
$1\,200 \times 5 = 6\,000$ m $= 6$ km.
\end{solution}
