\section*{Chapter \ref{ch:g3:mult} --- Multiplication}

\begin{solution}{exo:g3:mult:1}
$8 + 8 + 8 + 8 = 4 \times 8 = 32$; \quad
$5 + 5 + 5 = 3 \times 5 = 15$; \quad
$20 \times 5 = 100$.
\end{solution}

\begin{solution}{exo:g3:mult:2}
$42$; \quad $32$; \quad $81$; \quad $56$; \quad $36$.
\end{solution}

\begin{solution}{exo:g3:mult:3}
Both rectangles contain the same $24$ dots --- one is the other
turned on its side: $4 \times 6 = 6 \times 4$.
\end{solution}

\begin{solution}{exo:g3:mult:4}
$560$; \quad $800$; \quad $0$; \quad $45$; \quad $300$.
\end{solution}

\begin{solution}{exo:g3:mult:5}
$132 \times 3 = 396$; \quad
$217 \times 4 = 868$; \quad
$408 \times 7 = 2\,856$.
\end{solution}

\begin{solution}{exo:g3:mult:6}
$31 \times 5 = 30 \times 5 + 5 = 155$; \quad
$42 \times 3 = 120 + 6 = 126$; \quad
$25 \times 6 = 20 \times 6 + 5 \times 6 = 120 + 30 = 150$.
\end{solution}

\begin{solution}{exo:g3:mult:7}
$99 \times 7 = 700 - 7 = 693$; \quad
$101 \times 6 = 600 + 6 = 606$; \quad
$98 \times 5 = 500 - 10 = 490$.
\end{solution}

\begin{solution}{exo:g3:mult:8}
$389 \times 5$: estimate $400 \times 5 = 2\,000$; exactly
$1\,945$.

$612 \times 8$: estimate $600 \times 8 = 4\,800$; exactly $4\,896$.
\end{solution}

\begin{solution}{exo:g3:mult:9}
$38 \times 6 = 228$. There are $228$ eggs in the boxes.
\end{solution}

\begin{solution}{exo:g3:mult:10}
Desks: $8 \times 4 = 32$. Students: $32 \times 2 = 64$. The room can
seat $64$ students.
\end{solution}

\begin{solution}{exo:g3:mult:11}
$24$ appears in the table square as $1 \times 24$, $2 \times 12$,
$3 \times 8$, $4 \times 6$ --- and the same pairs reversed. Possible
arrangements: $1$ row of $24$, $2$ rows of $12$, $3$ rows of $8$,
$4$ rows of $6$, $6$ rows of $4$, $8$ rows of $3$, $12$ rows of $2$,
$24$ rows of $1$.
\end{solution}

\begin{solution}{exo:g3:mult:12}
Zoe splits $47$ into $40 + 7$ (the column method in her head); Léa
replaces $47$ by the friendly $50$ and corrects by taking away the
$3 \times 6$ counted too much. For $38 \times 4$: Zoe's way,
$120 + 32 = 152$; Léa's way, $40 \times 4 - 2 \times 4 = 160 - 8 =
152$. Both work --- Léa's is quicker here because $38$ is close to
$40$.
\end{solution}
