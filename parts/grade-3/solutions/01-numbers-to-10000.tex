\section*{Chapter \ref{ch:g3:numbers} --- Numbers up to 10\,000}

\begin{solution}{exo:g3:numbers:1}
$2\,345$; \quad $6\,050$; \quad $9\,001$.
\end{solution}

\begin{solution}{exo:g3:numbers:2}
$1\,208$: one thousand two hundred eight. \quad
$4\,070$: four thousand seventy. \quad
$9\,999$: nine thousand nine hundred ninety-nine.
\end{solution}

\begin{solution}{exo:g3:numbers:3}
Tens digit: $8$. Thousands digit: $7$. Hundreds contained in total:
$73$ (because $7\,382 = 73$ hundreds $+ 82$, i.e.\
$73 \times 100 + 82$).
\end{solution}

\begin{solution}{exo:g3:numbers:4}
$5\,631 = 5$ thousands $+ 6$ hundreds $+ 3$ tens $+ 1$ unit.

$2\,046 = 2$ thousands $+ 0$ hundreds $+ 4$ tens $+ 6$ units.

$8\,500 = 8$ thousands $+ 5$ hundreds.
\end{solution}

\begin{solution}{exo:g3:numbers:5}
$456 < 465$ (tens: $5 < 6$); \quad
$1\,203 > 989$ (four digits beat three); \quad
$6\,090 > 6\,009$ (tens: $9 > 0$); \quad
$9\,999 < 10\,000$.
\end{solution}

\begin{solution}{exo:g3:numbers:6}
$353 < 503 < 530 < 3\,500 < 5\,033$.
\end{solution}

\begin{solution}{exo:g3:numbers:7}
$150$ sits halfway between $100$ and $200$; $480$ just before
$500$; $820$ just after $800$; $990$ almost at $1\,000$.
\end{solution}

\begin{solution}{exo:g3:numbers:8}
Even: $34$, $70$, $6\,548$ (last digit $4$, $0$, $8$). Odd: $57$,
$2\,481$ (last digit $7$, $1$).
\end{solution}

\begin{solution}{exo:g3:numbers:9}
From $2\,970$: $2\,980$, $2\,990$, $3\,000$, $3\,010$, $3\,020$.

From $860$: $960$, $1\,060$, $1\,160$, $1\,260$.

From $9\,996$: $9\,997$, $9\,998$, $9\,999$, $10\,000$ --- the last
step crosses into a five-digit number: our whole chapter's limit!
\end{solution}

\begin{solution}{exo:g3:numbers:10}
Biggest: $7\,530$ (digits in decreasing order). Smallest: $3\,057$
(smallest nonzero digit first, then the rest increasing --- the $0$
right after the $3$).
\end{solution}

\begin{solution}{exo:g3:numbers:11}
Bigger than $800$: the hundreds digit is $8$. Even: the units digit
must be $2$ (using each of $2$, $5$, $8$ once). So the number is
$852$.
\end{solution}
