\section*{Chapter \ref{ch:g3:problems} --- Solving Problems}

\begin{solution}{exo:g3:problems:1}
Flocks joining: addition. Sharing cherries: division. Boxes of
pencils: multiplication. Cards given away: subtraction.
\end{solution}

\begin{solution}{exo:g3:problems:2}
Asked: the number of cars now. Known: $246$, then $158$ more.
Operation: addition. Compute: $246 + 158 = 404$. Sentence: there are
now $404$ cars in the parking lot.
\end{solution}

\begin{solution}{exo:g3:problems:3}
Grouping division: $90 \div 9 = 10$. The rope gives $10$ pieces.
\end{solution}

\begin{solution}{exo:g3:problems:4}
Multiplication: $26 \times 8 = 208$. The tickets cost $208$.
\end{solution}

\begin{solution}{exo:g3:problems:5}
Comparison bar: Emma's bar $64$, Hugo's bar shorter by $29$.
Subtraction: $64 - 29 = 35$. Hugo has $35$ stickers.
\end{solution}

\begin{solution}{exo:g3:problems:6}
Received: $6 \times 45 = 270$ apples. Remaining:
$270 - 178 = 92$ apples.
\end{solution}

\begin{solution}{exo:g3:problems:7}
Science books: $6 + 8 = 14$. Novels: $12 - 9 = 3$ more in week 2.
\end{solution}

\begin{solution}{exo:g3:problems:8}
Monday $12$, Tuesday $20$, Wednesday $12$: total
$12 + 20 + 12 = 44$ cakes. The best day was Tuesday.
\end{solution}

\begin{solution}{exo:g3:problems:9}
The useless number is the price ($12$). Division:
$210 \div 35 = 6$ (since $35 \times 6 = 210$). Leo needs $6$ days.
\end{solution}

\begin{solution}{exo:g3:problems:10}
Adults: $2 \times 9 = 18$. Children: $3 \times 4 = 12$. Total price:
$18 + 12 = 30$. Change: $50 - 30 = 20$. They get $20$ back.
\end{solution}

\begin{solution}{exo:g3:problems:11}
Many correct answers --- for instance: ``A florist makes $8$ bunches
of $15$ flowers, then removes $20$ wilted flowers. How many flowers
remain?'' Solution: $8 \times 15 = 120$, then $120 - 20 = 100$.
\end{solution}
