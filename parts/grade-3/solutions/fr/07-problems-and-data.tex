\section*{Chapitre \ref{ch:g3:problems} --- R\'esoudre des probl\`emes}

\begin{solution}{exo:g3:problems:1}
Troupeaux qui se rejoignent~: addition. Partage de cerises~:
division. Bo\^ites de crayons~: multiplication. Cartes donn\'ees~:
soustraction.
\end{solution}

\begin{solution}{exo:g3:problems:2}
Demand\'e~: le nombre de voitures maintenant. Connu~: $246$, puis
$158$ de plus. Op\'eration~: addition. Calcul~:
$246 + 158 = 404$. Phrase~: il y a maintenant $404$ voitures dans le
parking.
\end{solution}

\begin{solution}{exo:g3:problems:3}
Division de groupement~: $90 \div 9 = 10$. La corde donne $10$
morceaux.
\end{solution}

\begin{solution}{exo:g3:problems:4}
Multiplication~: $26 \times 8 = 208$. Les billets co\^utent $208$.
\end{solution}

\begin{solution}{exo:g3:problems:5}
Barre de comparaison~: barre d'Emma $64$, barre d'Hugo plus courte de
$29$. Soustraction~: $64 - 29 = 35$. Hugo a $35$ autocollants.
\end{solution}

\begin{solution}{exo:g3:problems:6}
Re\c{c}us~: $6 \times 45 = 270$ pommes. Restantes~:
$270 - 178 = 92$ pommes.
\end{solution}

\begin{solution}{exo:g3:problems:7}
Livres de sciences~: $6 + 8 = 14$. Romans~: $12 - 9 = 3$ de plus en
semaine 2.
\end{solution}

\begin{solution}{exo:g3:problems:8}
Lundi $12$, mardi $20$, mercredi $12$~: total
$12 + 20 + 12 = 44$ g\^ateaux. Le meilleur jour \'etait mardi.
\end{solution}

\begin{solution}{exo:g3:problems:9}
Le nombre inutile est le prix ($12$). Division~:
$210 \div 35 = 6$ (car $35 \times 6 = 210$). L\'eo a besoin de $6$
jours.
\end{solution}

\begin{solution}{exo:g3:problems:10}
Adultes~: $2 \times 9 = 18$. Enfants~: $3 \times 4 = 12$. Prix
total~: $18 + 12 = 30$. Monnaie~: $50 - 30 = 20$. Ils re\c{c}oivent
$20$.
\end{solution}

\begin{solution}{exo:g3:problems:11}
Beaucoup de r\'eponses correctes --- par exemple~:
\guillemotleft{} Un fleuriste fait $8$ bouquets de $15$ fleurs, puis
retire $20$ fleurs fan\'ees. Combien de fleurs restent-elles~?
\guillemotright{} Solution~: $8 \times 15 = 120$, puis
$120 - 20 = 100$.
\end{solution}
