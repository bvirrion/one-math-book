\begin{solution}{exo:g3:addsub:2}

\begin{solution}{exo:g3:addsub:5}
De $47$ \`a $50$~: $3$~; de $50$ \`a $63$~: $13$~; total
$3 + 13 = 16$. Les colonnes confirment~: $63 - 47 = 16$.
\end{solution}

\begin{solution}{exo:g3:addsub:9}
Unit\'es~: $7 + \text{?} = 2$ est impossible, donc $7 + 5 = 12$~: le
trou des unit\'es est $5$, retenue $1$. Dizaines~:
$\text{?} + 2 + 1 = 6$~: le trou des dizaines est $3$ (pas de
retenue). Centaines~: $3 + \text{?} = 8$~: le trou des centaines est
$5$. Donc $337 + 525 = 862$, et chaque trou n'avait qu'un seul
chiffre possible --- il n'y a qu'une seule fa\c{c}on.
\end{solution}

\begin{solution}{exo:g3:addsub:10}
Vendus~: $145 + 89 = 234$ croissants. Cuits~: vendus plus invendus,
$234 + 17 = 251$ croissants.
\end{solution}

\begin{solution}{exo:g3:addsub:11}
Il a raison~: $500 - 200 = 300$, plus $1$ donne $301$~; en colonnes,
$500 - 199 = 301$ aussi. Enlever $200$ enl\`eve \emph{un de trop},
donc il faut en rendre un --- soustraire un voisin ami et corriger
est souvent plus rapide qu'emprunter.
\end{solution}
