\section*{Chapitre \ref{ch:g3:mult} --- Multiplication}

\begin{solution}{exo:g3:mult:1}
$8 + 8 + 8 + 8 = 4 \times 8 = 32$~; \quad
$5 + 5 + 5 = 3 \times 5 = 15$~; \quad
$20 \times 5 = 100$.
\end{solution}

\begin{solution}{exo:g3:mult:2}
$42$~; \quad $32$~; \quad $81$~; \quad $56$~; \quad $36$.
\end{solution}

\begin{solution}{exo:g3:mult:3}
Les deux rectangles contiennent les m\^emes $24$ points --- l'un est
l'autre tourn\'e sur le c\^ot\'e~: $4 \times 6 = 6 \times 4$.
\end{solution}

\begin{solution}{exo:g3:mult:4}
$560$~; \quad $800$~; \quad $0$~; \quad $45$~; \quad $300$.
\end{solution}

\begin{solution}{exo:g3:mult:5}
$132 \times 3 = 396$~; \quad
$217 \times 4 = 868$~; \quad
$408 \times 7 = 2\,856$.
\end{solution}

\begin{solution}{exo:g3:mult:6}
$31 \times 5 = 30 \times 5 + 5 = 155$~; \quad
$42 \times 3 = 120 + 6 = 126$~; \quad
$25 \times 6 = 20 \times 6 + 5 \times 6 = 120 + 30 = 150$.
\end{solution}

\begin{solution}{exo:g3:mult:7}
$99 \times 7 = 700 - 7 = 693$~; \quad
$101 \times 6 = 600 + 6 = 606$~; \quad
$98 \times 5 = 500 - 10 = 490$.
\end{solution}

\begin{solution}{exo:g3:mult:8}
$389 \times 5$~: estimation $400 \times 5 = 2\,000$~; exactement
$1\,945$.

$612 \times 8$~: estimation $600 \times 8 = 4\,800$~; exactement
$4\,896$.
\end{solution}

\begin{solution}{exo:g3:mult:9}
$38 \times 6 = 228$. Il y a $228$ {\oe}ufs dans les bo\^ites.
\end{solution}

\begin{solution}{exo:g3:mult:10}
Bureaux~: $8 \times 4 = 32$. \'El\`eves~: $32 \times 2 = 64$. La
salle peut accueillir $64$ \'el\`eves.
\end{solution}

\begin{solution}{exo:g3:mult:11}
$24$ appara\^it dans le carr\'e des tables comme $1 \times 24$,
$2 \times 12$, $3 \times 8$, $4 \times 6$ --- et les m\^emes paires
invers\'ees. Arrangements possibles~: $1$ ligne de $24$, $2$ lignes
de $12$, $3$ lignes de $8$, $4$ lignes de $6$, $6$ lignes de $4$,
$8$ lignes de $3$, $12$ lignes de $2$, $24$ lignes de $1$.
\end{solution}

\begin{solution}{exo:g3:mult:12}
Zo\'e d\'ecoupe $47$ en $40 + 7$ (la technique en colonnes dans sa
t\^ete)~; L\'ea remplace $47$ par le $50$ ami et corrige en enlevant
les $3 \times 6$ compt\'es en trop. Pour $38 \times 4$~: fa\c{c}on
Zo\'e, $120 + 32 = 152$~; fa\c{c}on L\'ea,
$40 \times 4 - 2 \times 4 = 160 - 8 = 152$. Les deux fonctionnent ---
celle de L\'ea est plus rapide ici car $38$ est proche de $40$.
\end{solution}
