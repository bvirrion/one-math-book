\section*{Chapitre \ref{ch:g3:numbers} --- Les nombres jusqu'\`a 10\,000}

\begin{solution}{exo:g3:numbers:1}
$2\,345$~; \quad $6\,050$~; \quad $9\,001$.
\end{solution}

\begin{solution}{exo:g3:numbers:2}
$1\,208$~: mille deux cent huit. \quad
$4\,070$~: quatre mille soixante-dix. \quad
$9\,999$~: neuf mille neuf cent quatre-vingt-dix-neuf.
\end{solution}

\begin{solution}{exo:g3:numbers:3}
Chiffre des dizaines~: $8$. Chiffre des milliers~: $7$. Centaines
contenues en tout~: $73$ (parce que $7\,382 = 73$ centaines $+ 82$,
c'est-\`a-dire $73 \times 100 + 82$).
\end{solution}

\begin{solution}{exo:g3:numbers:4}
$5\,631 = 5$ milliers $+ 6$ centaines $+ 3$ dizaines $+ 1$ unit\'e.

$2\,046 = 2$ milliers $+ 0$ centaine $+ 4$ dizaines $+ 6$ unit\'es.

$8\,500 = 8$ milliers $+ 5$ centaines.
\end{solution}

\begin{solution}{exo:g3:numbers:5}
$456 < 465$ (dizaines~: $5 < 6$)~; \quad
$1\,203 > 989$ (quatre chiffres battent trois)~; \quad
$6\,090 > 6\,009$ (dizaines~: $9 > 0$)~; \quad
$9\,999 < 10\,000$.
\end{solution}

\begin{solution}{exo:g3:numbers:6}
$353 < 503 < 530 < 3\,500 < 5\,033$.
\end{solution}

\begin{solution}{exo:g3:numbers:7}
$150$ se situe \`a mi-chemin entre $100$ et $200$~; $480$ juste avant
$500$~; $820$ juste apr\`es $800$~; $990$ presque \`a $1\,000$.
\end{solution}

\begin{solution}{exo:g3:numbers:8}
\omterm{def:g3:numbers:evenodd}{Pairs}~: $34$, $70$, $6\,548$ (dernier chiffre $4$, $0$, $8$).
\omterm{def:g3:numbers:evenodd}{Impairs}~: $57$, $2\,481$ (dernier chiffre $7$, $1$).
\end{solution}

\begin{solution}{exo:g3:numbers:9}
\`A partir de $2\,970$~: $2\,980$, $2\,990$, $3\,000$, $3\,010$,
$3\,020$.

\`A partir de $860$~: $960$, $1\,060$, $1\,160$, $1\,260$.

\`A partir de $9\,996$~: $9\,997$, $9\,998$, $9\,999$, $10\,000$ ---
le dernier pas franchit un nombre \`a cinq chiffres~: la limite de
tout le chapitre~!
\end{solution}

\begin{solution}{exo:g3:numbers:10}
Le plus grand~: $7\,530$ (chiffres en ordre d\'ecroissant). Le plus
petit~: $3\,057$ (plus petit chiffre non nul en premier, puis le
reste en croissant --- le $0$ juste apr\`es le $3$).
\end{solution}

\begin{solution}{exo:g3:numbers:11}
Plus grand que $800$~: le chiffre des centaines est $8$. \omterm{def:g3:numbers:evenodd}{Pair}~: le
chiffre des unit\'es doit \^etre $2$ (en utilisant chacun de $2$,
$5$, $8$ une fois). Donc le nombre est $852$.
\end{solution}
