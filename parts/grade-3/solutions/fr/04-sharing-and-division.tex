\section*{Chapitre \ref{ch:g3:division} --- Partage et division}

\begin{solution}{exo:g3:division:1}
$36 \div 4 = 9$ (car $4 \times 9 = 36$)~; \quad
$63 \div 9 = 7$~; \quad $40 \div 5 = 8$~; \quad $56 \div 8 = 7$.
\end{solution}

\begin{solution}{exo:g3:division:2}
$17 = 5 \times 3 + 2$~: \omterm{def:g3:division:remainder}{quotient} $3$, \omterm{def:g3:division:remainder}{reste} $2$.

$30 = 4 \times 7 + 2$~: \omterm{def:g3:division:remainder}{quotient} $7$, \omterm{def:g3:division:remainder}{reste} $2$.

$50 = 6 \times 8 + 2$~: \omterm{def:g3:division:remainder}{quotient} $8$, \omterm{def:g3:division:remainder}{reste} $2$.
\end{solution}

\begin{solution}{exo:g3:division:3}
$28 = 6 \times 4 + 4$~: chaque enfant re\c{c}oit $4$ billes, et $4$
billes restent en trop.
\end{solution}

\begin{solution}{exo:g3:division:4}
\omterm{def:g3:division:half}{Moiti\'e} de $68$~: $34$. Tiers de $27$~: $9$. \omterm{def:g3:division:half}{Quart} de $48$~: $12$.
\omterm{def:g3:division:half}{Moiti\'e} de $150$~: $75$.
\end{solution}

\begin{solution}{exo:g3:division:5}
\guillemotleft{} $45 \div 5 = 9$ \guillemotright{}~: vrai,
$5 \times 9 = 45$.

\guillemotleft{} $52 \div 6 = 8$ \omterm{def:g3:division:remainder}{reste} $4$ \guillemotright{}~: vrai,
$6 \times 8 + 4 = 52$ et $4 < 6$.

\guillemotleft{} $70 \div 8 = 9$ \omterm{def:g3:division:remainder}{reste} $2$ \guillemotright{}~: faux
--- $8 \times 9 + 2 = 74$, pas $70$. Correct~:
$70 = 8 \times 8 + 6$.
\end{solution}

\begin{solution}{exo:g3:division:6}
$32 \div 4 = 8$ pages. C'est un \emph{groupement}~: les photos sont
mises en groupes de $4$, et on compte les groupes.
\end{solution}

\begin{solution}{exo:g3:division:7}
$75 = 9 \times 8 + 3$~: huit morceaux complets, et un \omterm{def:g3:division:remainder}{reste} de
$3$~cm.
\end{solution}

\begin{solution}{exo:g3:division:8}
$50 = 4 \times 12 + 2$~: douze voitures transportent $48$ enfants, et
$2$ enfants ont encore besoin d'une place. Il faut $13$ voitures.
\end{solution}

\begin{solution}{exo:g3:division:9}
$? = 7 \times 6 = 42$. \quad
$54 \div 9 = 6$, donc le diviseur manquant est $9$. \quad
Le nombre dont la \omterm{def:g3:division:half}{moiti\'e} est $35$ est $70$.
\end{solution}

\begin{solution}{exo:g3:division:10}
$47 = 5 \times 9 + 2$~: chaque ami re\c{c}oit $9$ autocollants, et
Amina garde le \omterm{def:g3:division:remainder}{reste}, $2$ autocollants.
\end{solution}

\begin{solution}{exo:g3:division:11}
En divisant par $6$, le \omterm{def:g3:division:remainder}{reste} est toujours plus petit que $6$~: il
peut \^etre $0$, $1$, $2$, $3$, $4$ ou $5$. Le plus grand nombre de
\omterm{def:g3:division:remainder}{quotient} $7$ utilise le plus grand \omterm{def:g3:division:remainder}{reste}~: $6 \times 7 + 5 = 47$.
\end{solution}
