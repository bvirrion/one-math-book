\section*{Chapitre \ref{ch:g3:measure} --- Mesures}

\begin{solution}{exo:g3:measure:1}
Largeur d'un clou~: mm. Taille d'un camarade~: cm (ou m). Cour~: m.
Distance entre villes~: km.
\end{solution}

\begin{solution}{exo:g3:measure:2}
$5$~m $= 500$~cm~; \quad $70$~mm $= 7$~cm~; \quad
$4$~km $= 4\,000$~m~; \quad $600$~cm $= 6$~m.
\end{solution}

\begin{solution}{exo:g3:measure:3}
Les r\'eponses d\'ependent du triangle~; la longueur totale du
contour est la \omterm{def:g1:addition:def}{somme} des trois c\^ot\'es mesur\'es (ce total
s'appelle le \emph{p\'erim\`etre}, \'etudi\'e dans
\cref{ch:g4:measure}).
\end{solution}

\begin{solution}{exo:g3:measure:4}
$2$~m $+ 45$~cm $= 200 + 45 = 245$~cm.

$1$~km $+ 250$~m $= 1\,000 + 250 = 1\,250$~m.

$3$~cm $+ 8$~mm $= 30 + 8 = 38$~mm.
\end{solution}

\begin{solution}{exo:g3:measure:5}
$3$~kg $= 3\,000$~g~; \quad $2\,500$~g $= 2$~kg et $500$~g~; \quad
$4$~L $= 400$~cL~; \quad $350$~cL $= 3$~L et $50$~cL.
\end{solution}

\begin{solution}{exo:g3:measure:6}
$1$~kg $= 1\,000$~g $> 750$~g~: oui, elle en a assez, et
$1\,000 - 750 = 250$~g resteront.
\end{solution}

\begin{solution}{exo:g3:measure:7}
Petite aiguille entre $2$ et $3$, grande sur le $6$~: $2{:}30$.
Petite aiguille sur $9$, grande sur $12$~: $9{:}00$. Petite aiguille
presque sur $5$, grande sur $10$~: $4{:}50$
(\guillemotleft{} cinq heures moins dix \guillemotright{} ---
l'aiguille des heures n'a pas encore atteint le $5$).
\end{solution}

\begin{solution}{exo:g3:measure:8}
De $8{:}30$ \`a $9{:}00$~: $30$~min~; de $9{:}00$ \`a $11{:}45$~:
$2$~h $45$~min. Total~: $3$~h $15$~min.
\end{solution}

\begin{solution}{exo:g3:measure:9}
De $9{:}50$ \`a $10{:}00$~: $10$~min~; il reste
$45 - 10 = 35$~min apr\`es $10{:}00$. Arriv\'ee~: $10{:}35$.
\end{solution}

\begin{solution}{exo:g3:measure:10}
$1$~L $= 100$~cL, et $100 = 20 \times 5$~: Paul peut verser $5$
verres pleins.
\end{solution}

\begin{solution}{exo:g3:measure:11}
Chaque jour~: $600 + 600 = 1\,200$~m. En cinq jours~:
$1\,200 \times 5 = 6\,000$~m $= 6$~km.
\end{solution}
