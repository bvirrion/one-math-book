\section*{Chapitre \ref{ch:g3:shapes} --- Figures et angles droits}

\begin{solution}{exo:g3:shapes:1}
Angles droits typiques~: les coins du tableau, d'une feuille de
papier, de la porte. Un angle non droit~: l'ouverture d'une paire de
ciseaux, les aiguilles de l'horloge \`a $1$~heure. Chacun se
v\'erifie \`a l'\'equerre (\cref{met:g3:shapes:check}) --- jamais
\`a l'{\oe}il seul.
\end{solution}

\begin{solution}{exo:g3:shapes:2}
Construction~: un c\^ot\'e de $6$~cm, angles droits aux deux
extr\'emit\'es avec l'\'equerre, c\^ot\'es de $4$~cm, fermer la
figure. Marques~: $4$ petits carr\'es dans les coins, un trait sur
chaque c\^ot\'e de $6$~cm, un double trait sur chaque c\^ot\'e de
$4$~cm.
\end{solution}

\begin{solution}{exo:g3:shapes:3}
Un carr\'e a $4$ angles droits et $4$ c\^ot\'es \'egaux --- tous
marqu\'es.
\end{solution}

\begin{solution}{exo:g3:shapes:4}
Constructions sur quadrillage~; le quadrillage garantit les angles
droits, et le comptage des cases garantit les longueurs.
\end{solution}

\begin{solution}{exo:g3:shapes:5}
\guillemotleft{} Tout carr\'e est un rectangle \guillemotright{}~:
\emph{vrai} --- il a $4$ angles droits, c'est tout ce qu'un
rectangle demande.

\guillemotleft{} Tout rectangle est un carr\'e \guillemotright{}~:
\emph{faux} --- un rectangle $6 \times 4$ a des c\^ot\'es in\'egaux.

\guillemotleft{} Un triangle peut avoir deux angles droits
\guillemotright{}~: \emph{faux} --- les deux c\^ot\'es tir\'es des
deux angles droits seraient perpendiculaires \`a la m\^eme base,
donc ne se rencontreraient jamais pour fermer le triangle.
\end{solution}

\begin{solution}{exo:g3:shapes:6}
$OA = 4$~cm (le rayon). $OB = 4$~cm aussi~: \emph{tout} point du
cercle est \`a la m\^eme distance du centre.
\end{solution}

\begin{solution}{exo:g3:shapes:7}
Le rectangle est d\'ecoup\'e en trois triangles~: deux triangles
rectangles ($AMD$ et $MBC$, rectangles en $A$ et $B$) et le triangle
du milieu $MCD$.
\end{solution}

\begin{solution}{exo:g3:shapes:8}
Avec au moins un angle droit~: le carr\'e, le rectangle, le triangle
rectangle, et la lettre L (ses coins int\'erieurs et ext\'erieurs).
Le cercle n'a aucun coin.
\end{solution}

\begin{solution}{exo:g3:shapes:9}
Chaque c\^ot\'e croise \guillemotleft{} $2$ \`a droite, $1$ en haut
\guillemotright{} (ou les versions tourn\'ees \guillemotleft{} $1$
\`a droite, $2$ en bas \guillemotright{}, etc.)~: chaque c\^ot\'e est
la diagonale d'un bloc $2 \times 1$ de cases, donc les quatre ont la
m\^eme longueur.
\end{solution}

\begin{solution}{exo:g3:shapes:10}
Un programme correct~: \guillemotleft{} 1. Dessiner un rectangle
$ABCD$ avec $AB = 3$~cm (bas) et $BC = 2$~cm. 2. Sur le c\^ot\'e
$[DC]$, dessiner le carr\'e $DCEF$ de c\^ot\'e $3$~cm, \`a
l'ext\'erieur du rectangle. \guillemotright{} Tout programme clair,
num\'erot\'e et complet est correct --- le tester en le suivant mot
\`a mot.
\end{solution}

\begin{solution}{exo:g3:shapes:11}
Carr\'es $1 \times 1$~: $9$. Carr\'es $2 \times 2$~: $4$.
$3 \times 3$~: $1$. Total~: $9 + 4 + 1 = 14$ carr\'es.
\end{solution}
