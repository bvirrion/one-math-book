\section*{Capítulo \ref{ch:g3:mult} --- Multiplication}

\begin{solution}{exo:g3:mult:1}
$8 + 8 + 8 + 8 = 4 \times 8 = 32$; \quad
$5 + 5 + 5 = 3 \times 5 = 15$; \quad
$20 \times 5 = 100$.
\end{solution}

\begin{solution}{exo:g3:mult:2}
$42$; \quad $32$; \quad $81$; \quad $56$; \quad $36$.
\end{solution}

\begin{solution}{exo:g3:mult:3}
Ambos rectángulos contienen los mismos puntos $24$ --- uno es el otro
volteado de lado: $4 \times 6 = 6 \times 4$.
\end{solution}

\begin{solution}{exo:g3:mult:4}
$560$; \quad $800$; \quad $0$; \quad $45$; \quad $300$.
\end{solution}

\begin{solution}{exo:g3:mult:5}
$132 \times 3 = 396$; \quad
$217 \times 4 = 868$; \quad
$408 \times 7 = 2\,856$.
\end{solution}

\begin{solution}{exo:g3:mult:6}
$31 \times 5 = 30 \times 5 + 5 = 155$; \quad
$42 \times 3 = 120 + 6 = 126$; \quad
$25 \times 6 = 20 \times 6 + 5 \times 6 = 120 + 30 = 150$.
\end{solution}

\begin{solution}{exo:g3:mult:7}
$99 \times 7 = 700 - 7 = 693$; \quad
$101 \times 6 = 600 + 6 = 606$; \quad
$98 \times 5 = 500 - 10 = 490$.
\end{solution}

\begin{solution}{exo:g3:mult:8}
$389 \times 5$: estimación $400 \times 5 = 2\,000$; exactamente
$1\,945$.

$612 \times 8$: estimación $600 \times 8 = 4\,800$; exactamente $4\,896$.
\end{solution}

\begin{solution}{exo:g3:mult:9}
$38 \times 6 = 228$. Hay huevos $228$ en las cajas.
\end{solution}

\begin{solution}{exo:g3:mult:10}
Escritorios: $8 \times 4 = 32$. Estudiantes: $32 \times 2 = 64$. la habitación puede
asiento $64$ estudiantes.
\end{solution}

\begin{solution}{exo:g3:mult:11}
$24$ aparece en el cuadro de la tabla como $1 \times 24$, $2 \times 12$,
$3 \times 8$, $4 \times 6$ --- y los mismos pares invertidos. Posible
arreglos: $1$ fila de $24$, $2$ filas de $12$, $3$ filas de $8$,
$4$ filas de $6$, $6$ filas de $4$, $8$ filas de $3$, $12$ filas de $2$,
$24$ filas de $1$.
\end{solution}

\begin{solution}{exo:g3:mult:12}
Zoe divide $47$ en $40 + 7$ (el método de columna en su cabeza); lea
reemplaza $47$ por el simpático $50$ y corrige quitando el
$3 \times 6$ contó demasiado. Para $38 \times 4$: el camino de Zoe,
$120 + 32 = 152$; El camino de Léa, $40 \times 4 - 2 \times 4 = 160 - 8 =
152$. Ambos funcionan --- Léa es más rápido aquí porque $38$ está cerca de
$40$.
\end{solution}
