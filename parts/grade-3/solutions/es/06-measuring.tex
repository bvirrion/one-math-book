\section*{Capítulo \ref{ch:g3:measure} --- Measuring}

\begin{solution}{exo:g3:measure:1}
Ancho de clavo: mm. Altura del compañero: cm (o m). Patio de la escuela: m.
Distancia entre pueblos: km.
\end{solution}

\begin{solution}{exo:g3:measure:2}
$5$ m $= 500$ cm; \quad $70$ mm $= 7$ cm; \quad
$4$ km $= 4\,000$ m; \quad $600$ cm $= 6$ m.
\end{solution}

\begin{solution}{exo:g3:measure:3}
Las respuestas dependen del triángulo; la longitud total de la frontera es la
\omterm{def:g1:addition:def}{suma} de los tres lados medidos (este total se llama
\emph{perímetro}, estudiado en \cref{ch:g4:measure}).
\end{solution}

\begin{solution}{exo:g3:measure:4}
$2$ m $+ 45$ cm $= 200 + 45 = 245$ cm.

$1$ km $+ 250$ m $= 1\,000 + 250 = 1\,250$ m.

$3$ cm $+ 8$ mm $= 30 + 8 = 38$ mm.
\end{solution}

\begin{solution}{exo:g3:measure:5}
$3$ kg $= 3\,000$ g; \quad $2\,500$ g $= 2$ kg y $500$ g; \quad
$4$ L $= 400$ cL; \quad $350$ cL $= 3$ L y $50$ cL.
\end{solution}

\begin{solution}{exo:g3:measure:6}
$1$ kg $= 1\,000$ g $> 750$ g: si, tiene suficiente, y
Quedarán $1\,000 - 750 = 250$ g.
\end{solution}

\begin{solution}{exo:g3:measure:7}
Manecilla corta entre $2$ y $3$, manecilla larga en $6$: $2{:}30$.
Mano corta en $9$, mano larga en $12$: $9{:}00$. Mano corta casi en
$5$, manecilla larga en $10$: $4{:}50$ (``diez menos cinco'' --- la manecilla de las horas
aún no ha llegado al $5$).
\end{solution}

\begin{solution}{exo:g3:measure:8}
De $8{:}30$ a $9{:}00$: $30$ mín; de $9{:}00$ a $11{:}45$:
$2$ h $45$ mín. Total: $3$ h $15$ mín.
\end{solution}

\begin{solution}{exo:g3:measure:9}
De $9{:}50$ a $10{:}00$: $10$ mín; quedan
$45 - 10 = 35$ minutos después de $10{:}00$. Llegada: $10{:}35$.
\end{solution}

\begin{solution}{exo:g3:measure:10}
$1$ L $= 100$ cL y $100 = 20 \times 5$: Paul puede servir $5$ completo
gafas.
\end{solution}

\begin{solution}{exo:g3:measure:11}
Cada día: $600 + 600 = 1\,200$ m. En cinco días:
$1\,200 \times 5 = 6\,000$ m $= 6$ km.
\end{solution}
