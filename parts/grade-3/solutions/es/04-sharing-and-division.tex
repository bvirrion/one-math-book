\section*{Capítulo \ref{ch:g3:division} --- Sharing and Division}

\begin{solution}{exo:g3:division:1}
$36 \div 4 = 9$ (desde $4 \times 9 = 36$); \quad
$63 \div 9 = 7$; \quad$40 \div 5 = 8$; \quad $56 \div 8 = 7$.
\end{solution}

\begin{solution}{exo:g3:division:2}
$17 = 5 \times 3 + 2$: \omterm{def:g3:division:remainder}{cociente} $3$, \omterm{def:g3:division:remainder}{resto} $2$.

$30 = 4 \times 7 + 2$: \omterm{def:g3:division:remainder}{cociente} $7$, \omterm{def:g3:division:remainder}{resto} $2$.

$50 = 6 \times 8 + 2$: \omterm{def:g3:division:remainder}{cociente} $8$, \omterm{def:g3:division:remainder}{resto} $2$.
\end{solution}

\begin{solution}{exo:g3:division:3}
$28 = 6 \times 4 + 4$: cada niño recibe canicas $4$ y canicas $4$
sobran.
\end{solution}

\begin{solution}{exo:g3:division:4}
\omterm{def:g3:division:half}{Medio} de $68$: $34$. Tercero de $27$: $9$. \omterm{def:g3:division:half}{Cuarto} de $48$: $12$. \omterm{def:g3:division:half}{Medio}
de $150$: $75$.
\end{solution}

\begin{solution}{exo:g3:division:5}
``$45 \div 5 = 9$'': verdadero, $5 \times 9 = 45$.

``$52 \div 6 = 8$ \omterm{def:g3:division:remainder}{resto} $4$'': verdadero, $6 \times 8 + 4 = 52$ y
$4 < 6$.

``$70 \div 8 = 9$ \omterm{def:g3:division:remainder}{resto} $2$'': falso --- $8 \times 9 + 2 = 74$,
no $70$. Correcto: $70 = 8 \times 8 + 6$.
\end{solution}

\begin{solution}{exo:g3:division:6}
$32 \div 4 = 8$ páginas. Este es \emph{agrupamiento}: las fotos están puestas
en grupos de $4$, y contamos los grupos.
\end{solution}

\begin{solution}{exo:g3:division:7}
$75 = 9 \times 8 + 3$: ocho piezas completas y un poco de sobra
$3$cm.
\end{solution}

\begin{solution}{exo:g3:division:8}
$50 = 4 \times 12 + 2$: doce coches transportan niños $48$, y $2$
los niños todavía necesitan que los lleven. Se necesitan coches $13$.
\end{solution}

\begin{solution}{exo:g3:division:9}
$? = 7 \times 6 = 42$. \quad
$54 \div 9 = 6$, por lo que el divisor que falta es $9$. \quad
El número cuyo \omterm{def:g3:division:half}{medio} es $35$ es $70$.
\end{solution}

\begin{solution}{exo:g3:division:10}
$47 = 5 \times 9 + 2$: cada amigo recibe pegatinas $9$ y Amina
conserva las pegatinas \omterm{def:g3:division:remainder}{resto}, $2$.
\end{solution}

\begin{solution}{exo:g3:division:11}
Dividiendo por $6$, el \omterm{def:g3:division:remainder}{resto} siempre es menor que $6$: puede ser
$0$, $1$, $2$, $3$, $4$ o $5$. El número más grande con \omterm{def:g3:division:remainder}{cociente} $7$
utiliza el \omterm{def:g3:division:remainder}{resto} más grande: $6 \times 7 + 5 = 47$.
\end{solution}
