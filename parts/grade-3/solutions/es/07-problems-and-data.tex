\section*{Capítulo \ref{ch:g3:problems} --- Solving Problems}

\begin{solution}{exo:g3:problems:1}
Unión de bandadas: suma. Compartir cerezas: división. Cajas de
lápices: multiplicación. Tarjetas regaladas: resta.
\end{solution}

\begin{solution}{exo:g3:problems:2}
Preguntado: el número de coches ahora. Conocido: $246$, luego $158$ más.
Operación: suma. Calcular: $246 + 158 = 404$. Oración: hay
ahora $404$ autos en el estacionamiento.
\end{solution}

\begin{solution}{exo:g3:problems:3}
División de agrupación: $90 \div 9 = 10$. La cuerda da $10$ piezas.
\end{solution}

\begin{solution}{exo:g3:problems:4}
Multiplicación: $26 \times 8 = 208$. Los billetes cuestan $208$.
\end{solution}

\begin{solution}{exo:g3:problems:5}
Barra de comparación: barra de Emma $64$, barra de Hugo más corta en $29$.
Resta: $64 - 29 = 35$. Hugo tiene pegatinas $35$.
\end{solution}

\begin{solution}{exo:g3:problems:6}
Recibido: $6 \times 45 = 270$ manzanas. Restante:
$270 - 178 = 92$ manzanas.
\end{solution}

\begin{solution}{exo:g3:problems:7}
Libros de ciencias: $6 + 8 = 14$. Novelas: $12 - 9 = 3$ más en la semana 2.
\end{solution}

\begin{solution}{exo:g3:problems:8}
Lunes $12$, martes $20$, miércoles $12$: total
$12 + 20 + 12 = 44$ pasteles. El mejor día fue el martes.
\end{solution}

\begin{solution}{exo:g3:problems:9}
El número inútil es el precio ($12$). División:
$210 \div 35 = 6$ (desde $35 \times 6 = 210$). Leo necesita $6$ días.
\end{solution}

\begin{solution}{exo:g3:problems:10}
Adultos: $2 \times 9 = 18$. Niños: $3 \times 4 = 12$. Precio total:
$18 + 12 = 30$. Cambio: $50 - 30 = 20$. Recuperan $20$.
\end{solution}

\begin{solution}{exo:g3:problems:11}
Muchas respuestas correctas --- por ejemplo: ``Un florista hace $8$ racimos
de flores $15$ y luego elimina las flores marchitas $20$. cuantas flores
¿Permanecen?'' Solución: $8 \times 15 = 120$, luego $120 - 20 = 100$.
\end{solution}
