\section*{Hoofdstuk \ref{ch:g3:addsub} --- Optellen en aftrekken}

\begin{solution}{exo:g3:addsub:1}
$346 + 253 = 599$; \quad
$478 + 265 = 743$; \quad
$1\,536 + 2\,876 = 4\,412$.
\end{solution}

\begin{solution}{exo:g3:addsub:2}
$587 - 234 = 353$ (controle: $353 + 234 = 587$).

$805 - 348 = 457$ (controle: $457 + 348 = 805$).

$1\,000 - 463 = 537$ (controle: $537 + 463 = 1\,000$).
\end{solution}

\begin{solution}{exo:g3:addsub:3}
$38 + 7 = 38 + 2 + 5 = 45$; \quad
$56 + 9 = 56 + 4 + 5 = 65$; \quad
$45 + 28 = 45 + 30 - 2 = 73$.
\end{solution}

\begin{solution}{exo:g3:addsub:4}
$340 + 220 = 560$; \quad $675 + 210 = 885$; \quad
$512 - 300 = 212$; \quad $840 - 220 = 620$.
\end{solution}

\begin{solution}{exo:g3:addsub:5}
Van $47$ naar $50$: $3$; van $50$ naar $63$: $13$; totaal
$3 + 13 = 16$. Kolommen bevestigen: $63 - 47 = 16$.
\end{solution}

\begin{solution}{exo:g3:addsub:6}
$492 + 218$: schatting $500 + 200 = 700$; precies $710$.

$913 - 388$: schatting $900 - 400 = 500$; precies $525$.
\end{solution}

\begin{solution}{exo:g3:addsub:7}
$348 + 296 = 644$. De school heeft $644$ leerlingen.
\end{solution}

\begin{solution}{exo:g3:addsub:8}
$224 - 178 = 46$. Nora heeft nog $46$ bladzijden te lezen.
\end{solution}

\begin{solution}{exo:g3:addsub:9}
Eenheden: $7 + \text{?} = 2$ is onmogelijk, dus $7 + 5 = 12$: het
eenhedengat is $5$, onthoud $1$. Tientallen: $\text{?} + 2 + 1 = 6$:
het tientallengat is $3$ (geen onthouden). Honderdtallen:
$3 + \text{?} = 8$: het honderdtallengat is $5$. Dus
$337 + 525 = 862$, en elk gat had precies \'{e}\'{e}n mogelijk cijfer
--- er is maar \'{e}\'{e}n manier.
\end{solution}

\begin{solution}{exo:g3:addsub:10}
Verkocht: $145 + 89 = 234$ croissants. Gebakken: verkocht plus
onverkocht, $234 + 17 = 251$ croissants.
\end{solution}

\begin{solution}{exo:g3:addsub:11}
Hij heeft gelijk: $500 - 200 = 300$, plus $1$ geeft $301$; in
kolommen is $500 - 199 = 301$ ook. $200$ wegnemen neemt
\emph{\'{e}\'{e}n te veel} weg, dus moet er \'{e}\'{e}n teruggegeven worden
--- aftrekken met een vriendelijke buur en corrigeren is vaak sneller
dan lenen.
\end{solution}
