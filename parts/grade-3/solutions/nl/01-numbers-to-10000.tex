\section*{Hoofdstuk \ref{ch:g3:numbers} --- Getallen tot 10\,000}

\begin{solution}{exo:g3:numbers:1}
$2\,345$; \quad $6\,050$; \quad $9\,001$.
\end{solution}

\begin{solution}{exo:g3:numbers:2}
$1\,208$: duizend tweehonderdacht. \quad
$4\,070$: vierduizend zeventig. \quad
$9\,999$: negenduizend negenhonderdnegenennegentig.
\end{solution}

\begin{solution}{exo:g3:numbers:3}
Tientallencijfer: $8$. Duizendtallencijfer: $7$. Honderdtallen in
totaal: $73$ (want $7\,382 = 73$ honderdtallen $+ 82$, d.w.z.\
$73 \times 100 + 82$).
\end{solution}

\begin{solution}{exo:g3:numbers:4}
$5\,631 = 5$ duizendtallen $+ 6$ honderdtallen $+ 3$ tientallen $+ 1$
eenheid.

$2\,046 = 2$ duizendtallen $+ 0$ honderdtallen $+ 4$ tientallen $+ 6$
eenheden.

$8\,500 = 8$ duizendtallen $+ 5$ honderdtallen.
\end{solution}

\begin{solution}{exo:g3:numbers:5}
$456 < 465$ (tientallen: $5 < 6$); \quad
$1\,203 > 989$ (vier cijfers winnen van drie); \quad
$6\,090 > 6\,009$ (tientallen: $9 > 0$); \quad
$9\,999 < 10\,000$.
\end{solution}

\begin{solution}{exo:g3:numbers:6}
$353 < 503 < 530 < 3\,500 < 5\,033$.
\end{solution}

\begin{solution}{exo:g3:numbers:7}
$150$ ligt halverwege tussen $100$ en $200$; $480$ net v\'{o}\'{o}r
$500$; $820$ net na $800$; $990$ bijna bij $1\,000$.
\end{solution}

\begin{solution}{exo:g3:numbers:8}
Even: $34$, $70$, $6\,548$ (laatste cijfer $4$, $0$, $8$). Oneven:
$57$, $2\,481$ (laatste cijfer $7$, $1$).
\end{solution}

\begin{solution}{exo:g3:numbers:9}
Vanaf $2\,970$: $2\,980$, $2\,990$, $3\,000$, $3\,010$, $3\,020$.

Vanaf $860$: $960$, $1\,060$, $1\,160$, $1\,260$.

Vanaf $9\,996$: $9\,997$, $9\,998$, $9\,999$, $10\,000$ --- de
laatste stap gaat over naar een vijfcijferig getal: de limiet van
heel dit hoofdstuk!
\end{solution}

\begin{solution}{exo:g3:numbers:10}
Grootste: $7\,530$ (cijfers in aflopende volgorde). Kleinste:
$3\,057$ (kleinste niet-nul cijfer eerst, dan de rest stijgend ---
de $0$ direct na de $3$).
\end{solution}

\begin{solution}{exo:g3:numbers:11}
Groter dan $800$: het honderdtallencijfer is $8$. Even: het
eenhedencijfer moet $2$ zijn (elk van $2$, $5$, $8$ \'{e}\'{e}n keer).
Dus het getal is $852$.
\end{solution}
