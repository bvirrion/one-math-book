\section*{Hoofdstuk \ref{ch:g3:division} --- Verdelen en delen}

\begin{solution}{exo:g3:division:1}
$36 \div 4 = 9$ (want $4 \times 9 = 36$); \quad
$63 \div 9 = 7$; \quad $40 \div 5 = 8$; \quad $56 \div 8 = 7$.
\end{solution}

\begin{solution}{exo:g3:division:2}
$17 = 5 \times 3 + 2$: \omterm{def:g3:division:remainder}{quoti\"ent} $3$, \omterm{def:g3:division:remainder}{rest} $2$.

$30 = 4 \times 7 + 2$: \omterm{def:g3:division:remainder}{quoti\"ent} $7$, \omterm{def:g3:division:remainder}{rest} $2$.

$50 = 6 \times 8 + 2$: \omterm{def:g3:division:remainder}{quoti\"ent} $8$, \omterm{def:g3:division:remainder}{rest} $2$.
\end{solution}

\begin{solution}{exo:g3:division:3}
$28 = 6 \times 4 + 4$: elk kind krijgt $4$ knikkers, en $4$
knikkers blijven over.
\end{solution}

\begin{solution}{exo:g3:division:4}
\omterm{def:g3:division:half}{Helft} van $68$: $34$. Derde van $27$: $9$. \omterm{def:g3:division:half}{Kwart} van $48$: $12$.
\omterm{def:g3:division:half}{Helft} van $150$: $75$.
\end{solution}

\begin{solution}{exo:g3:division:5}
``$45 \div 5 = 9$'': waar, $5 \times 9 = 45$.

``$52 \div 6 = 8$ \omterm{def:g3:division:remainder}{rest} $4$'': waar, $6 \times 8 + 4 = 52$ en
$4 < 6$.

``$70 \div 8 = 9$ \omterm{def:g3:division:remainder}{rest} $2$'': niet waar --- $8 \times 9 + 2 = 74$,
niet $70$. Juist: $70 = 8 \times 8 + 6$.
\end{solution}

\begin{solution}{exo:g3:division:6}
$32 \div 4 = 8$ bladen. Dit is \emph{groeperen}: de foto's worden
in groepen van $4$ gezet, en we tellen de groepen.
\end{solution}

\begin{solution}{exo:g3:division:7}
$75 = 9 \times 8 + 3$: acht volle stukken, en een reststuk van
$3$ cm.
\end{solution}

\begin{solution}{exo:g3:division:8}
$50 = 4 \times 12 + 2$: twaalf auto's vervoeren $48$ kinderen, en
$2$ kinderen hebben nog een rit nodig. Er zijn $13$ auto's nodig.
\end{solution}

\begin{solution}{exo:g3:division:9}
$? = 7 \times 6 = 42$. \quad
$54 \div 9 = 6$, dus de ontbrekende deler is $9$. \quad
Het getal waarvan de \omterm{def:g3:division:half}{helft} $35$ is, is $70$.
\end{solution}

\begin{solution}{exo:g3:division:10}
$47 = 5 \times 9 + 2$: elke vriend krijgt $9$ stickers, en Amina
houdt de \omterm{def:g3:division:remainder}{rest}, $2$ stickers.
\end{solution}

\begin{solution}{exo:g3:division:11}
Bij delen door $6$ is de \omterm{def:g3:division:remainder}{rest} altijd kleiner dan $6$: die kan
$0$, $1$, $2$, $3$, $4$ of $5$ zijn. Het grootste getal met
\omterm{def:g3:division:remainder}{quoti\"ent} $7$ gebruikt de grootste \omterm{def:g3:division:remainder}{rest}: $6 \times 7 + 5 = 47$.
\end{solution}
