\section*{Hoofdstuk \ref{ch:g3:measure} --- Meten}

\begin{solution}{exo:g3:measure:1}
Breedte van een spijker: mm. Lengte van een klasgenoot: cm (of m).
Speelplaats: m. Afstand tussen steden: km.
\end{solution}

\begin{solution}{exo:g3:measure:2}
$5$ m $= 500$ cm; \quad $70$ mm $= 7$ cm; \quad
$4$ km $= 4\,000$ m; \quad $600$ cm $= 6$ m.
\end{solution}

\begin{solution}{exo:g3:measure:3}
Antwoorden hangen af van de driehoek; de totale lengte van de omtrek
is de \omterm{def:g1:addition:def}{som} van de drie gemeten zijden (dit totaal heet de
\emph{omtrek}, bestudeerd in \cref{ch:g4:measure}).
\end{solution}

\begin{solution}{exo:g3:measure:4}
$2$ m $+ 45$ cm $= 200 + 45 = 245$ cm.

$1$ km $+ 250$ m $= 1\,000 + 250 = 1\,250$ m.

$3$ cm $+ 8$ mm $= 30 + 8 = 38$ mm.
\end{solution}

\begin{solution}{exo:g3:measure:5}
$3$ kg $= 3\,000$ g; \quad $2\,500$ g $= 2$ kg en $500$ g; \quad
$4$ L $= 400$ cL; \quad $350$ cL $= 3$ L en $50$ cL.
\end{solution}

\begin{solution}{exo:g3:measure:6}
$1$ kg $= 1\,000$ g $> 750$ g: ja, ze heeft genoeg, en
$1\,000 - 750 = 250$ g blijft over.
\end{solution}

\begin{solution}{exo:g3:measure:7}
Korte wijzer tussen $2$ en $3$, lange wijzer op de $6$: $2{:}30$.
Korte wijzer op $9$, lange wijzer op $12$: $9{:}00$. Korte wijzer
bijna op $5$, lange wijzer op $10$: $4{:}50$ (``tien voor vijf'' ---
de uurwijzer heeft de $5$ nog niet bereikt).
\end{solution}

\begin{solution}{exo:g3:measure:8}
Van $8{:}30$ naar $9{:}00$: $30$ min; van $9{:}00$ naar $11{:}45$:
$2$ u $45$ min. Totaal: $3$ u $15$ min.
\end{solution}

\begin{solution}{exo:g3:measure:9}
Van $9{:}50$ naar $10{:}00$: $10$ min; er blijven
$45 - 10 = 35$ min na $10{:}00$. Aankomst: $10{:}35$.
\end{solution}

\begin{solution}{exo:g3:measure:10}
$1$ L $= 100$ cL, en $100 = 20 \times 5$: Paul kan $5$ volle glazen
inschenken.
\end{solution}

\begin{solution}{exo:g3:measure:11}
Elke dag: $600 + 600 = 1\,200$ m. In vijf dagen:
$1\,200 \times 5 = 6\,000$ m $= 6$ km.
\end{solution}
