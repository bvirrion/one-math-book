\section*{Hoofdstuk \ref{ch:g3:problems} --- Problemen oplossen}

\begin{solution}{exo:g3:problems:1}
Kuddes bij elkaar: optelling. Kersen verdelen: deling. Dozen
potloden: vermenigvuldiging. Kaarten weggegeven: aftrekking.
\end{solution}

\begin{solution}{exo:g3:problems:2}
Gevraagd: het aantal auto's nu. Bekend: $246$, daarna $158$ erbij.
Bewerking: optelling. Reken: $246 + 158 = 404$. Zin: er zijn nu
$404$ auto's op de parkeerplaats.
\end{solution}

\begin{solution}{exo:g3:problems:3}
Groepeerdeling: $90 \div 9 = 10$. Het touw geeft $10$ stukken.
\end{solution}

\begin{solution}{exo:g3:problems:4}
Vermenigvuldiging: $26 \times 8 = 208$. De kaartjes kosten $208$.
\end{solution}

\begin{solution}{exo:g3:problems:5}
Vergelijkingsstaaf: Emma's staaf $64$, Hugo's staaf korter met $29$.
Aftrekking: $64 - 29 = 35$. Hugo heeft $35$ stickers.
\end{solution}

\begin{solution}{exo:g3:problems:6}
Ontvangen: $6 \times 45 = 270$ appels. Overblijvend:
$270 - 178 = 92$ appels.
\end{solution}

\begin{solution}{exo:g3:problems:7}
Wetenschapsboeken: $6 + 8 = 14$. Romans: $12 - 9 = 3$ meer in week
2.
\end{solution}

\begin{solution}{exo:g3:problems:8}
Maandag $12$, dinsdag $20$, woensdag $12$: totaal
$12 + 20 + 12 = 44$ cakes. De beste dag was dinsdag.
\end{solution}

\begin{solution}{exo:g3:problems:9}
Het nutteloze getal is de prijs ($12$). Deling:
$210 \div 35 = 6$ (want $35 \times 6 = 210$). Leo heeft $6$ dagen
nodig.
\end{solution}

\begin{solution}{exo:g3:problems:10}
Volwassenen: $2 \times 9 = 18$. Kinderen: $3 \times 4 = 12$. Totale
prijs: $18 + 12 = 30$. Wisselgeld: $50 - 30 = 20$. Ze krijgen $20$
terug.
\end{solution}

\begin{solution}{exo:g3:problems:11}
Veel juiste antwoorden --- bijvoorbeeld: ``Een bloemist maakt $8$
bossen van $15$ bloemen, en haalt er dan $20$ verwelkte bloemen uit.
Hoeveel bloemen blijven er over?'' Oplossing: $8 \times 15 = 120$,
dan $120 - 20 = 100$.
\end{solution}
