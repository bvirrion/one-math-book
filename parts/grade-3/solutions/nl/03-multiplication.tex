\section*{Hoofdstuk \ref{ch:g3:mult} --- Vermenigvuldigen}

\begin{solution}{exo:g3:mult:1}
$8 + 8 + 8 + 8 = 4 \times 8 = 32$; \quad
$5 + 5 + 5 = 3 \times 5 = 15$; \quad
$20 \times 5 = 100$.
\end{solution}

\begin{solution}{exo:g3:mult:2}
$42$; \quad $32$; \quad $81$; \quad $56$; \quad $36$.
\end{solution}

\begin{solution}{exo:g3:mult:3}
Beide rechthoeken bevatten dezelfde $24$ stippen --- de ene is de
andere op de zijkant gekeerd: $4 \times 6 = 6 \times 4$.
\end{solution}

\begin{solution}{exo:g3:mult:4}
$560$; \quad $800$; \quad $0$; \quad $45$; \quad $300$.
\end{solution}

\begin{solution}{exo:g3:mult:5}
$132 \times 3 = 396$; \quad
$217 \times 4 = 868$; \quad
$408 \times 7 = 2\,856$.
\end{solution}

\begin{solution}{exo:g3:mult:6}
$31 \times 5 = 30 \times 5 + 5 = 155$; \quad
$42 \times 3 = 120 + 6 = 126$; \quad
$25 \times 6 = 20 \times 6 + 5 \times 6 = 120 + 30 = 150$.
\end{solution}

\begin{solution}{exo:g3:mult:7}
$99 \times 7 = 700 - 7 = 693$; \quad
$101 \times 6 = 600 + 6 = 606$; \quad
$98 \times 5 = 500 - 10 = 490$.
\end{solution}

\begin{solution}{exo:g3:mult:8}
$389 \times 5$: schatting $400 \times 5 = 2\,000$; precies
$1\,945$.

$612 \times 8$: schatting $600 \times 8 = 4\,800$; precies $4\,896$.
\end{solution}

\begin{solution}{exo:g3:mult:9}
$38 \times 6 = 228$. Er zitten $228$ eieren in de dozen.
\end{solution}

\begin{solution}{exo:g3:mult:10}
Bureaus: $8 \times 4 = 32$. Leerlingen: $32 \times 2 = 64$. Het
lokaal kan $64$ leerlingen plaatsen.
\end{solution}

\begin{solution}{exo:g3:mult:11}
$24$ komt in het tafelsquare voor als $1 \times 24$, $2 \times 12$,
$3 \times 8$, $4 \times 6$ --- en dezelfde paren omgekeerd. Mogelijke
opstellingen: $1$ rij van $24$, $2$ rijen van $12$, $3$ rijen van
$8$, $4$ rijen van $6$, $6$ rijen van $4$, $8$ rijen van $3$, $12$
rijen van $2$, $24$ rijen van $1$.
\end{solution}

\begin{solution}{exo:g3:mult:12}
Zoe splitst $47$ in $40 + 7$ (de kolommethode in haar hoofd); L\'{e}a
vervangt $47$ door het vriendelijke $50$ en corrigeert door de
$3 \times 6$ die te veel geteld is weg te nemen. Voor $38 \times 4$:
Zoe's manier, $120 + 32 = 152$; L\'{e}a's manier,
$40 \times 4 - 2 \times 4 = 160 - 8 = 152$. Beide werken --- L\'{e}a's
is hier sneller omdat $38$ dicht bij $40$ ligt.
\end{solution}
