\section*{Capítulo \ref{ch:g3:measure} --- Medindo}

\begin{solution}{exo:g3:measure:1}
Largura da unha: mm. Altura do colega: cm (ou m). Pátio da escola: m.
Distância entre cidades: km.
\end{solution}

\begin{solution}{exo:g3:measure:2}
$5$ eu $= 500$ cm; \quad $70$ milímetros $= 7$ cm; \quad
$4$ quilômetros $= 4\,000$ m; \quad $600$ cm $= 6$ m.
\end{solution}

\begin{solution}{exo:g3:measure:3}
As respostas dependem do triângulo; o comprimento total da fronteira é o
\omterm{def:g1:addition:def}{soma} dos três lados medidos (este total é chamado de
\emph{perímetro}, estudou em \cref{ch:g4:measure}).
\end{solution}

\begin{solution}{exo:g3:measure:4}
$2$ eu $+ 45$ cm $= 200 + 45 = 245$ cm.

$1$ quilômetros $+ 250$ eu $= 1\,000 + 250 = 1\,250$ m.

$3$ cm $+ 8$ milímetros $= 30 + 8 = 38$ mm.
\end{solution}

\begin{solution}{exo:g3:measure:5}
$3$ kg $= 3\,000$ g; \quad $2\,500$ g $= 2$ kg e $500$ g; \quad
$4$ eu $= 400$ cl; \quad $350$ cL $= 3$ Terra $50$ cl.
\end{solution}

\begin{solution}{exo:g3:measure:6}
$1$ kg $= 1\,000$ g $> 750$ g: sim, ela tem o suficiente, e
$1\,000 - 750 = 250$ restará g.
\end{solution}

\begin{solution}{exo:g3:measure:7}
Ponteiro curto entre $2$ e $3$, mão longa no $6$: $2{:}30$.
Mão curta $9$, mão longa $12$: $9{:}00$. Ponteiro curto quase ligado
$5$, mão longa $10$: $4{:}50$ (``dez para cinco'' --- o ponteiro das horas
não atingiu o $5$ ainda).
\end{solution}

\begin{solution}{exo:g3:measure:8}
De $8{:}30$ para $9{:}00$: $30$ min; de $9{:}00$ para $11{:}45$:
$2$ h $45$ min. Total: $3$ h $15$ min.
\end{solution}

\begin{solution}{exo:g3:measure:9}
De $9{:}50$ para $10{:}00$: $10$ min; resta
$45 - 10 = 35$ minutos depois $10{:}00$. Chegada: $10{:}35$.
\end{solution}

\begin{solution}{exo:g3:measure:10}
$1$ eu $= 100$ cL, e $100 = 20 \times 5$: Paul pode derramar $5$ cheio
óculos.
\end{solution}

\begin{solution}{exo:g3:measure:11}
Todos os dias: $600 + 600 = 1\,200$ m. Em cinco dias:
$1\,200 \times 5 = 6\,000$ eu $= 6$ km.
\end{solution}
