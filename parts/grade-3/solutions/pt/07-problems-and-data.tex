\section*{Capítulo \ref{ch:g3:problems} --- Resolvendo Problemas}

\begin{solution}{exo:g3:problems:1}
Junção de bandos: adição. Compartilhando cerejas: divisão. Caixas de
lápis: multiplicação. Cartas distribuídas: subtração.
\end{solution}

\begin{solution}{exo:g3:problems:2}
Perguntado: o número de carros agora. Conhecido: $246$, então $158$ mais.
Operação: adição. Calcular: $246 + 158 = 404$. Frase: há
agora $404$ carros no estacionamento.
\end{solution}

\begin{solution}{exo:g3:problems:3}
Divisão de agrupamento: $90 \div 9 = 10$. A corda dá $10$ peças.
\end{solution}

\begin{solution}{exo:g3:problems:4}
Multiplicação: $26 \times 8 = 208$. Os ingressos custam $208$.
\end{solution}

\begin{solution}{exo:g3:problems:5}
Barra de comparação: barra de Emma $64$, a barra de Hugo é mais curta em $29$.
Subtração: $64 - 29 = 35$. Hugo tem $35$ adesivos.
\end{solution}

\begin{solution}{exo:g3:problems:6}
Recebido: $6 \times 45 = 270$ maçãs. \omterm{def:g3:division:remainder}{Restante}:
$270 - 178 = 92$ maçãs.
\end{solution}

\begin{solution}{exo:g3:problems:7}
Livros de ciências: $6 + 8 = 14$. Romances: $12 - 9 = 3$ mais na semana 2.
\end{solution}

\begin{solution}{exo:g3:problems:8}
Segunda-feira $12$, Terça-feira $20$, Quarta-feira $12$: total
$12 + 20 + 12 = 44$ bolos. O melhor dia foi terça-feira.
\end{solution}

\begin{solution}{exo:g3:problems:9}
O número inútil é o preço ($12$). Divisão:
$210 \div 35 = 6$ (desde $35 \times 6 = 210$). Leão precisa $6$ dias.
\end{solution}

\begin{solution}{exo:g3:problems:10}
Adultos: $2 \times 9 = 18$. Crianças: $3 \times 4 = 12$. Preço total:
$18 + 12 = 30$. Mudar: $50 - 30 = 20$. Eles conseguem $20$ voltar.
\end{solution}

\begin{solution}{exo:g3:problems:11}
Muitas respostas corretas --- por exemplo: ``Uma florista faz $8$ cachos
de $15$ flores e depois remove $20$ flores murchas. Quantas flores
permanecer?'' Solução: $8 \times 15 = 120$, então $120 - 20 = 100$.
\end{solution}
