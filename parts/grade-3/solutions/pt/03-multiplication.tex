\section*{Capítulo \ref{ch:g3:mult} --- Multiplicação}

\begin{solution}{exo:g3:mult:1}
$8 + 8 + 8 + 8 = 4 \times 8 = 32$; \quad
$5 + 5 + 5 = 3 \times 5 = 15$; \quad
$20 \times 5 = 100$.
\end{solution}

\begin{solution}{exo:g3:mult:2}
$42$; \quad $32$; \quad $81$; \quad $56$; \quad $36$.
\end{solution}

\begin{solution}{exo:g3:mult:3}
Ambos os retângulos contêm o mesmo $24$ pontos --- um é o outro
virado de lado: $4 \times 6 = 6 \times 4$.
\end{solution}

\begin{solution}{exo:g3:mult:4}
$560$; \quad $800$; \quad $0$; \quad $45$; \quad $300$.
\end{solution}

\begin{solution}{exo:g3:mult:5}
$132 \times 3 = 396$; \quad
$217 \times 4 = 868$; \quad
$408 \times 7 = 2\,856$.
\end{solution}

\begin{solution}{exo:g3:mult:6}
$31 \times 5 = 30 \times 5 + 5 = 155$; \quad
$42 \times 3 = 120 + 6 = 126$; \quad
$25 \times 6 = 20 \times 6 + 5 \times 6 = 120 + 30 = 150$.
\end{solution}

\begin{solution}{exo:g3:mult:7}
$99 \times 7 = 700 - 7 = 693$; \quad
$101 \times 6 = 600 + 6 = 606$; \quad
$98 \times 5 = 500 - 10 = 490$.
\end{solution}

\begin{solution}{exo:g3:mult:8}
$389 \times 5$: estimativa $400 \times 5 = 2\,000$; exatamente
$1\,945$.

$612 \times 8$: estimativa $600 \times 8 = 4\,800$; exatamente $4\,896$.
\end{solution}

\begin{solution}{exo:g3:mult:9}
$38 \times 6 = 228$. Há $228$ ovos nas caixas.
\end{solution}

\begin{solution}{exo:g3:mult:10}
Mesas: $8 \times 4 = 32$. Alunos: $32 \times 2 = 64$. A sala pode
assento $64$ estudantes.
\end{solution}

\begin{solution}{exo:g3:mult:11}
$24$ aparece no quadrado da tabela como $1 \times 24$, $2 \times 12$,
$3 \times 8$, $4 \times 6$ --- e os mesmos \omterm{def:g3:numbers:evenodd}{pares} invertidos. Possível
arranjos: $1$ linha de $24$, $2$ linhas de $12$, $3$ linhas de $8$,
$4$ linhas de $6$, $6$ linhas de $4$, $8$ linhas de $3$, $12$ linhas de $2$,
$24$ linhas de $1$.
\end{solution}

\begin{solution}{exo:g3:mult:12}
Zoe se separa $47$ em $40 + 7$ (o método da coluna na cabeça dela); Léa
substitui $47$ pelo amigável $50$ e corrige tirando o
$3 \times 6$ contava demais. Para $38 \times 4$: Do jeito de Zoe,
$120 + 32 = 152$; Do jeito de Léa, $40 \times 4 - 2 \times 4 = 160 - 8 =
152$. Ambos funcionam --- o da Léa é mais rápido aqui porque $38$ está perto de
$40$.
\end{solution}
