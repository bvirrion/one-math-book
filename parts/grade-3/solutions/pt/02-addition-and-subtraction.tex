\begin{solution}{exo:g3:addsub:1}

$478 + 265 = 743$; \quad
$1\,536 + 2\,876 = 4\,412$.
\end{solution}

\begin{solution}{exo:g3:addsub:2}
$587 - 234 = 353$ (verificar: $353 + 234 = 587$).

$805 - 348 = 457$ (verificar: $457 + 348 = 805$).

$1\,000 - 463 = 537$ (verificar: $537 + 463 = 1\,000$).
\end{solution}

\begin{solution}{exo:g3:addsub:5}
De $47$ para $50$: $3$; de $50$ para $63$: $13$; total
$3 + 13 = 16$. Colunas confirmam: $63 - 47 = 16$.
\end{solution}

\begin{solution}{exo:g3:addsub:9}
Unidades: $7 + \text{?} = 2$ é impossível, então $7 + 5 = 12$: as unidades
buraco é $5$, carregar $1$. Dezenas: $\text{?} + 2 + 1 = 6$: o buraco das dezenas
é $3$ (sem transporte). Centenas: $3 + \text{?} = 8$: o buraco das centenas é
$5$. Então $337 + 525 = 862$, e cada buraco tinha exatamente um possível
dígito --- só existe uma maneira.
\end{solution}

\begin{solution}{exo:g3:addsub:10}
Vendido: $145 + 89 = 234$ croissants. Assado: vendido mais não vendido,
$234 + 17 = 251$ croissants.
\end{solution}

\begin{solution}{exo:g3:addsub:11}
Ele está certo: $500 - 200 = 300$, mais $1$ dá $301$; em colunas,
$500 - 199 = 301$ também. Tirando $200$ remove \emph{um demais},
então é preciso devolver --- subtraindo um vizinho amigável e
corrigir é muitas vezes mais rápido do que pedir emprestado.
\end{solution}
