\section*{Capítulo \ref{ch:g3:numbers} --- Números até 10\,000}

\begin{solution}{exo:g3:numbers:1}
$2\,345$; \quad $6\,050$; \quad $9\,001$.
\end{solution}

\begin{solution}{exo:g3:numbers:2}
$1\,208$: mil duzentos e oito. \quad
$4\,070$: quatro mil e setenta. \quad
$9\,999$: nove mil novecentos e noventa e nove.
\end{solution}

\begin{solution}{exo:g3:numbers:3}
Dígito das dezenas: $8$. Dígito dos milhares: $7$. Centenas contidas no total:
$73$ (porque $7\,382 = 73$ centenas $+ 82$, ou seja\
$73 \times 100 + 82$).
\end{solution}

\begin{solution}{exo:g3:numbers:4}
$5\,631 = 5$ milhares $+ 6$ centenas $+ 3$ dezenas $+ 1$ unidade.

$2\,046 = 2$ milhares $+ 0$ centenas $+ 4$ dezenas $+ 6$ unidades.

$8\,500 = 8$ milhares $+ 5$ centenas.
\end{solution}

\begin{solution}{exo:g3:numbers:5}
$456 < 465$ (dezenas: $5 < 6$); \quad
$1\,203 > 989$ (quatro dígitos vencem três); \quad
$6\,090 > 6\,009$ (dezenas: $9 > 0$); \quad
$9\,999 < 10\,000$.
\end{solution}

\begin{solution}{exo:g3:numbers:6}
$353 < 503 < 530 < 3\,500 < 5\,033$.
\end{solution}

\begin{solution}{exo:g3:numbers:7}
$150$ fica no meio do caminho $100$ e $200$; $480$ pouco antes
$500$; $820$ logo depois $800$; $990$ quase em $1\,000$.
\end{solution}

\begin{solution}{exo:g3:numbers:8}
\omterm{def:g3:numbers:evenodd}{Par}: $34$, $70$, $6\,548$ (último dígito $4$, $0$, $8$). \omterm{def:g3:numbers:evenodd}{Ímpar}: $57$,
$2\,481$ (último dígito $7$, $1$).
\end{solution}

\begin{solution}{exo:g3:numbers:9}
De $2\,970$: $2\,980$, $2\,990$, $3\,000$, $3\,010$, $3\,020$.

De $860$: $960$, $1\,060$, $1\,160$, $1\,260$.

De $9\,996$: $9\,997$, $9\,998$, $9\,999$, $10\,000$ --- o último
passo se transforma em um número de cinco dígitos: o limite de todo o nosso capítulo!
\end{solution}

\begin{solution}{exo:g3:numbers:10}
O maior: $7\,530$ (dígitos em ordem decrescente). Menor: $3\,057$
(primeiro o menor dígito diferente de \omterm{def:g1:counting:zero}{zero}, depois o resto aumentando --- o $0$
logo após o $3$).
\end{solution}

\begin{solution}{exo:g3:numbers:11}
Maior que $800$: o algarismo das centenas é $8$. \omterm{def:g3:numbers:evenodd}{Par}: o dígito das unidades
deve ser $2$ (usando cada um $2$, $5$, $8$ uma vez). Então o número é
$852$.
\end{solution}
