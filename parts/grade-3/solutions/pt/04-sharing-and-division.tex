\section*{Capítulo \ref{ch:g3:division} --- Compartilhamento e Divisão}

\begin{solution}{exo:g3:division:1}
$36 \div 4 = 9$ (desde $4 \times 9 = 36$); \quad
$63 \div 9 = 7$; \quad $40 \div 5 = 8$; \quad $56 \div 8 = 7$.
\end{solution}

\begin{solution}{exo:g3:division:2}
$17 = 5 \times 3 + 2$: \omterm{def:g3:division:remainder}{quociente} $3$, \omterm{def:g3:division:remainder}{restante} $2$.

$30 = 4 \times 7 + 2$: \omterm{def:g3:division:remainder}{quociente} $7$, \omterm{def:g3:division:remainder}{restante} $2$.

$50 = 6 \times 8 + 2$: \omterm{def:g3:division:remainder}{quociente} $8$, \omterm{def:g3:division:remainder}{restante} $2$.
\end{solution}

\begin{solution}{exo:g3:division:3}
$28 = 6 \times 4 + 4$: cada criança recebe $4$ bolinhas de gude e $4$ bolinhas de gude
sobraram.
\end{solution}

\begin{solution}{exo:g3:division:4}
\omterm{def:g3:division:half}{Metade} de $68$: $34$. Terceiro de $27$: $9$. \omterm{def:g3:division:half}{Trimestre} de $48$: $12$. \omterm{def:g3:division:half}{Metade}
de $150$: $75$.
\end{solution}

\begin{solution}{exo:g3:division:5}
``$45 \div 5 = 9$'': verdadeiro, $5 \times 9 = 45$.

``$52 \div 6 = 8$ \omterm{def:g3:division:remainder}{restante} $4$'': verdadeiro, $6 \times 8 + 4 = 52$ e
$4 < 6$.

``$70 \div 8 = 9$ \omterm{def:g3:division:remainder}{restante} $2$'': falso --- $8 \times 9 + 2 = 74$,
não $70$. Correto: $70 = 8 \times 8 + 6$.
\end{solution}

\begin{solution}{exo:g3:division:6}
$32 \div 4 = 8$ páginas. Isso é \emph{agrupamento}: as fotos são colocadas
em grupos de $4$, e contamos os grupos.
\end{solution}

\begin{solution}{exo:g3:division:7}
$75 = 9 \times 8 + 3$: oito peças inteiras e um pouco de sobra
$3$ cm.
\end{solution}

\begin{solution}{exo:g3:division:8}
$50 = 4 \times 12 + 2$: doze carros transportam $48$ crianças, e $2$
as crianças ainda precisam de carona. $13$ carros são necessários.
\end{solution}

\begin{solution}{exo:g3:division:9}
$? = 7 \times 6 = 42$. \quad
$54 \div 9 = 6$, então o divisor que falta é $9$. \quad
O número cujo \omterm{def:g3:division:half}{metade} é $35$ é $70$.
\end{solution}

\begin{solution}{exo:g3:division:10}
$47 = 5 \times 9 + 2$: cada amigo recebe $9$ adesivos e Amina
mantém o \omterm{def:g3:division:remainder}{restante}, $2$ adesivos.
\end{solution}

\begin{solution}{exo:g3:division:11}
Dividindo por $6$, o \omterm{def:g3:division:remainder}{restante} é sempre menor que $6$: pode ser
$0$, $1$, $2$, $3$, $4$ ou $5$. O maior número com \omterm{def:g3:division:remainder}{quociente} $7$
usa o maior \omterm{def:g3:division:remainder}{restante}: $6 \times 7 + 5 = 47$.
\end{solution}
