\section*{Chapter \ref{ch:g3:division} --- Sharing and Division}

\begin{solution}{exo:g3:division:1}
$36 \div 4 = 9$ (since $4 \times 9 = 36$); \quad
$63 \div 9 = 7$; \quad $40 \div 5 = 8$; \quad $56 \div 8 = 7$.
\end{solution}

\begin{solution}{exo:g3:division:2}
$17 = 5 \times 3 + 2$: \omterm{def:g3:division:remainder}{quotient} $3$, \omterm{def:g3:division:remainder}{remainder} $2$.

$30 = 4 \times 7 + 2$: \omterm{def:g3:division:remainder}{quotient} $7$, \omterm{def:g3:division:remainder}{remainder} $2$.

$50 = 6 \times 8 + 2$: \omterm{def:g3:division:remainder}{quotient} $8$, \omterm{def:g3:division:remainder}{remainder} $2$.
\end{solution}

\begin{solution}{exo:g3:division:3}
$28 = 6 \times 4 + 4$: each child gets $4$ marbles, and $4$ marbles
are left over.
\end{solution}

\begin{solution}{exo:g3:division:4}
\omterm{def:g3:division:half}{Half} of $68$: $34$. Third of $27$: $9$. \omterm{def:g3:division:half}{Quarter} of $48$: $12$. \omterm{def:g3:division:half}{Half}
of $150$: $75$.
\end{solution}

\begin{solution}{exo:g3:division:5}
``$45 \div 5 = 9$'': true, $5 \times 9 = 45$.

``$52 \div 6 = 8$ \omterm{def:g3:division:remainder}{remainder} $4$'': true, $6 \times 8 + 4 = 52$ and
$4 < 6$.

``$70 \div 8 = 9$ \omterm{def:g3:division:remainder}{remainder} $2$'': false --- $8 \times 9 + 2 = 74$,
not $70$. Correct: $70 = 8 \times 8 + 6$.
\end{solution}

\begin{solution}{exo:g3:division:6}
$32 \div 4 = 8$ pages. This is \emph{grouping}: the photos are put
in groups of $4$, and we count the groups.
\end{solution}

\begin{solution}{exo:g3:division:7}
$75 = 9 \times 8 + 3$: eight full pieces, and a leftover bit of
$3$ cm.
\end{solution}

\begin{solution}{exo:g3:division:8}
$50 = 4 \times 12 + 2$: twelve cars carry $48$ children, and $2$
children still need a ride. $13$ cars are needed.
\end{solution}

\begin{solution}{exo:g3:division:9}
$? = 7 \times 6 = 42$. \quad
$54 \div 9 = 6$, so the missing divisor is $9$. \quad
The number whose \omterm{def:g3:division:half}{half} is $35$ is $70$.
\end{solution}

\begin{solution}{exo:g3:division:10}
$47 = 5 \times 9 + 2$: each friend gets $9$ stickers, and Amina
keeps the \omterm{def:g3:division:remainder}{remainder}, $2$ stickers.
\end{solution}

\begin{solution}{exo:g3:division:11}
Dividing by $6$, the \omterm{def:g3:division:remainder}{remainder} is always smaller than $6$: it can be
$0$, $1$, $2$, $3$, $4$ or $5$. The biggest number with \omterm{def:g3:division:remainder}{quotient} $7$
uses the biggest \omterm{def:g3:division:remainder}{remainder}: $6 \times 7 + 5 = 47$.
\end{solution}
