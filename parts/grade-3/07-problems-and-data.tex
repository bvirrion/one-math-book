\chapter{Solving Problems}\label{ch:g3:problems}

Knowing the four operations is one thing; knowing \emph{which one} a
story needs is another. This chapter gives a method for word
problems --- the same four steps every time --- and teaches how to
read information from tables and pictograms.

\section{The four steps}

\begin{method}[Solving a word problem]\label{met:g3:problems:steps}
\begin{enumerate}
  \item \emph{Read} the problem twice; say in your own words what is
        asked, and what is known.
  \item \emph{Choose} the operation: draw a quick sketch or a bar
        picture if the choice is not obvious.
  \item \emph{Compute}, in columns or mentally, with an estimate as a
        safety net.
  \item \emph{Answer} with a full sentence, including the unit ---
        and check the answer makes sense (a child does not weigh
        $300$ kg).
\end{enumerate}
\end{method}

\begin{example}[Which operation?]\label{ex:g3:problems:which}
The words hint at the operation, the sketch decides:
\begin{itemize}
  \item \emph{putting together / in all}: addition;
  \item \emph{taking away / how many more / what is missing}:
        subtraction;
  \item \emph{so many groups of so many / so many times}:
        multiplication;
  \item \emph{sharing fairly / how many groups}: division.
\end{itemize}
\end{example}

\begin{example}[A two-step problem]\label{ex:g3:problems:twostep}
``Maya buys $3$ packs of $6$ bottles and drinks $2$ bottles. How many
bottles are left?''
\begin{enumerate}
  \item What is asked: the number of bottles remaining. Known: $3$
        packs of $6$; $2$ drunk.
  \item Sketch: $3$ groups of $6$, then take $2$ away. Two
        operations!
  \item Compute: $3 \times 6 = 18$, then $18 - 2 = 16$.
  \item Sentence: $16$ bottles are left.
\end{enumerate}
Never stop after the first operation: reread the question to see
whether the story is finished.
\end{example}

\begin{example}[A useless number]\label{ex:g3:problems:trap}
``A baker, who is $52$ years old, bakes $120$ rolls and sells $85$.
How many rolls are left?'' The age is there to distract you: the
answer only needs $120 - 85 = 35$ rolls. Before computing, always
sort the numbers: which ones does the question really use?
\end{example}

\section{Reading tables and pictograms}

\begin{example}[A table]\label{ex:g3:problems:table}
Books borrowed at the library:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cccc}
 & comics & novels & science & total \\
\hline
week 1 & $14$ & $9$ & $6$ & $29$ \\
week 2 & $11$ & $12$ & $8$ & $31$ \\
\end{tabular}
\end{center}
Reading: week 2, novels: $12$. Computing from the table: in the two
weeks together, $14 + 11 = 25$ comics were borrowed; the busiest
week was week 2 ($31 > 29$).
\end{example}

\begin{example}[A pictogram]\label{ex:g3:problems:pictogram}
In a pictogram, one symbol stands for a fixed amount --- read the
key first!

\begin{omfigure}
\begin{tikzpicture}[scale=0.62]
  \node[left, font=\small] at (0,2.4) {Monday};
  \foreach \i in {0,1,2} \fill[omDef] (0.5+\i*0.8,2.4) circle (0.26);
  \node[left, font=\small] at (0,1.2) {Tuesday};
  \foreach \i in {0,1,2,3,4} \fill[omDef] (0.5+\i*0.8,1.2) circle (0.26);
  \node[left, font=\small] at (0,0) {Wednesday};
  \foreach \i in {0,1} \fill[omDef] (0.5+\i*0.8,0) circle (0.26);
  \fill[omDef] (0.5+1.6,0) circle (0.26);
  \node[right, font=\small] at (5.4,1.2)
    {key: one dot $= 4$ cakes sold};
\end{tikzpicture}

{\small A pictogram: Monday shows $3$ dots, so $3 \times 4 = 12$
cakes; Tuesday $5$ dots, $20$ cakes; Wednesday $3$ dots, $12$
cakes.}
\end{omfigure}
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g3:problems:1}
For each story, just \emph{name} the operation (do not compute):
a flock of $48$ sheep joins a flock of $37$; \ $56$ cherries shared
among $7$ children; \ $9$ boxes of $8$ pencils; \ Ali had $71$ cards
and gave away $18$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:2}
Solve with the four steps: ``A parking lot has $246$ cars in the
morning; $158$ more arrive at noon. How many cars are there now?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:3}
Solve: ``A rope of $90$ m is cut into pieces of $9$ m. How many
pieces?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:4}
Solve: ``Tickets cost $8$ each. What do $26$ tickets cost?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:5}
Draw a bar picture, then solve: ``Emma has $64$ stickers, Hugo has
$29$ fewer. How many has Hugo?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:6}
Two steps: ``A crate holds $45$ apples. A grocer receives $6$ crates
and sells $178$ apples. How many apples remain?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:7}
Using the table of \cref{ex:g3:problems:table}: how many science
books were borrowed in the two weeks together? How many more novels
in week 2 than in week 1?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:8}
Using the pictogram of \cref{ex:g3:problems:pictogram}: how many
cakes were sold in the three days together? On which day were the
most cakes sold?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:9}
This problem has a useless number --- find it, then solve: ``A
$210$-page book costs $12$. Leo reads $35$ pages a day. How many
days does he need to finish the book?''
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:problems:10}
``A family of $2$ adults and $3$ children visits the museum. Adult
tickets cost $9$, children tickets $4$. They pay with a $50$ bill.
How much change do they get?'' (Three steps: adults, children, then
the change.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:problems:11}
Make up your own two-step problem whose answer is $100$, write its
solution with the four steps, and test it on a classmate.
\end{exercise}
