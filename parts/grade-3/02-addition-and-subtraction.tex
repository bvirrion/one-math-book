\chapter{Addition and Subtraction}\label{ch:g3:addsub}

Adding means putting together; subtracting means taking away, or
finding what is missing. This chapter trains the two column methods
--- with their famous carries and borrows --- and the mental tricks
that often make columns unnecessary.

\section{Column addition}

\begin{method}[Adding in columns]\label{met:g3:addsub:add}
\begin{enumerate}
  \item Write the numbers one under the other, units under units,
        tens under tens (line up on the \emph{right});
  \item add the units; if the sum passes $9$, write its units digit
        and \emph{carry} the ten to the next column (a small $1$);
  \item continue with the tens, the hundreds, \dots, adding the carry
        each time.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:addsub:add}
Compute $457 + 368$:
\[
\begin{array}{r}
{\scriptstyle 1}\ \ {\scriptstyle 1}\ \ \phantom{0} \\[-3pt]
4\,5\,7 \\
+\ 3\,6\,8 \\
\hline
8\,2\,5
\end{array}
\]
Units: $7 + 8 = 15$: write $5$, carry $1$. Tens:
$5 + 6 + 1 = 12$: write $2$, carry $1$. Hundreds: $4 + 3 + 1 = 8$.
Result: $825$.
\end{example}

\begin{example}[Mental addition]\label{ex:g3:addsub:mental}
Look for friendly numbers before reaching for columns:
\begin{itemize}
  \item \emph{make a ten}: $47 + 8 = 47 + 3 + 5 = 50 + 5 = 55$;
  \item \emph{jump by parts}: $256 + 130 = 256 + 100 + 30 = 386$;
  \item \emph{reorder}: $17 + 59 + 3 = (17 + 3) + 59 = 79$.
\end{itemize}
\end{example}

\section{Column subtraction}

\begin{method}[Subtracting in columns]\label{met:g3:addsub:sub}
\begin{enumerate}
  \item Write the bigger number on top, aligned on the right;
  \item subtract column by column, starting from the units;
  \item if the top digit is too small, \emph{borrow}: take one from
        the next place on the top row (it is worth ten in the current
        column);
  \item \emph{check}: the result plus the subtracted number must give
        back the top number.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:addsub:sub}
Compute $603 - 147$:
\[
\begin{array}{r}
6\,0\,3 \\
-\ 1\,4\,7 \\
\hline
4\,5\,6
\end{array}
\]
Units: $3 - 7$ impossible; borrow through: $603$ becomes
$5$ hundreds, $9$ tens, $13$ units. Then $13 - 7 = 6$;
tens $9 - 4 = 5$; hundreds $5 - 1 = 4$. Result: $456$.
Check: $456 + 147 = 603$.
\end{example}

\begin{example}[Subtraction as a missing part]\label{ex:g3:addsub:missing}
``Lila has $35$ marbles, Tom has $52$: how many more has Tom?'' is
the subtraction $52 - 35$. Mentally, count \emph{up} from $35$:
$35 \to 40$ is $5$, then $40 \to 52$ is $12$; in total
$5 + 12 = 17$. Counting up is often easier than borrowing.
\end{example}

\section{Orders of magnitude}

\begin{method}[Estimating before computing]\label{met:g3:addsub:estimate}
Before any column computation, round the numbers to friendly ones and
compute an \emph{estimate}. After computing, compare: a result far
from the estimate signals a mistake.
\end{method}

\begin{example}
For $487 + 312$: the estimate is $500 + 300 = 800$. The column
result, $799$, is close: plausible. If you had found $1\,099$
(a missed alignment), the estimate would have warned you.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g3:addsub:1}
Compute in columns: $346 + 253$; \quad $478 + 265$; \quad
$1\,536 + 2\,876$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:2}
Compute in columns, then check by an addition: $587 - 234$; \quad
$805 - 348$; \quad $1\,000 - 463$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:3}
Compute mentally by making tens: $38 + 7$; \quad $56 + 9$; \quad
$45 + 28$ (add $30$, take away $2$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:4}
Compute mentally by jumps: $340 + 220$; \quad $675 + 210$; \quad
$512 - 300$; \quad $840 - 220$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:5}
Compute $63 - 47$ by counting up (as in
\cref{ex:g3:addsub:missing}), then check in columns.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:6}
Estimate first (round to hundreds), then compute exactly:
$492 + 218$; \quad $913 - 388$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:7}
A school has $348$ girls and $296$ boys. How many students in all?
Write the operation, compute, and answer with a sentence.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:8}
A book has $224$ pages; Nora has read $178$. How many pages are left
to read?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:9}
Complete the holes (work column by column):
\[
\begin{array}{r}
3\,?\,7 \\
+\ ?\,2\,? \\
\hline
8\,6\,2
\end{array}
\]
Is there only one way to fill the three holes? Explain.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:addsub:10}
A bakery sold $145$ croissants in the morning and $89$ in the
afternoon; $17$ croissants were left unsold at closing time. How many
croissants were baked that day? (Two steps.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:addsub:11}
Theo says: ``to compute $500 - 199$, I compute $500 - 200$ and then
add $1$.'' Is he right? Compute both ways, and explain his trick.
\end{exercise}
