\chapter{Shapes and Right Angles}\label{ch:g3:shapes}

Squares, rectangles, triangles, circles: the shapes are old friends.
This chapter looks at them with a geometer's eye --- what exactly
makes a square a square? --- and introduces the tool that decides:
the set square and its \omterm{def:g3:shapes:rightangle}{right angle}.

\section{The right angle}

\begin{definition}[Right angle]\label{def:g3:shapes:rightangle}
A \emph{right angle}\index{right angle} is the corner angle of a
sheet of paper, or of the set square. Two lines making a right angle
are \emph{perpendicular}. In figures, a right angle is marked with a
small square in the corner.
\end{definition}

\begin{method}[Checking a right angle]\label{met:g3:shapes:check}
\begin{enumerate}
  \item Place the corner of the set square exactly on the corner to
        test;
  \item slide one edge of the set square along one side of the angle;
  \item if the other side of the angle runs exactly along the other
        edge, the angle is right --- if it opens wider or narrower,
        it is not.
\end{enumerate}
\end{method}

\section{The shapes and their rules}

\begin{definition}[Square, rectangle, triangle]\label{def:g3:shapes:shapes}
\begin{itemize}
  \item A \emph{rectangle}\index{rectangle} has $4$ sides and $4$
        \omterm{def:g3:shapes:rightangle}{right angles}; its opposite sides have the same length.
  \item A \emph{square}\index{square} is a rectangle whose $4$ sides
        all have the same length.
  \item A \emph{triangle}\index{triangle} has $3$ sides; a
        \emph{right triangle} has one \omterm{def:g3:shapes:rightangle}{right angle}.
  \item A \emph{circle}\index{circle} is drawn with the compass: all
        its points are at the same distance (the \emph{radius}\index{radius}) from
        the \emph{center}.
\end{itemize}
\end{definition}

\begin{example}[Why marks matter]\label{ex:g3:shapes:marks}
A shape drawn slightly tilted can \emph{look} like a square without
being one. The marks tell the truth: $4$ right-angle squares $+$ $4$
tick marks $=$ a real square. When you draw, put the marks; when you
read a figure, trust only the marks and the given measures.
\end{example}

\section{Drawing on grid paper}

\begin{method}[Constructions on a grid]\label{met:g3:shapes:grid}
Grid paper offers free \omterm{def:g3:shapes:rightangle}{right angles}: its lines are perpendicular.
\begin{enumerate}
  \item To draw a rectangle $5$ by $3$: choose a crossing point,
        count $5$ squares right, $3$ squares up, and close the shape
        along the grid lines;
  \item to check equal lengths, count squares;
  \item diagonal steps make tilted lines: going ``$1$ right, $1$ up''
        repeatedly draws a $45^\circ$ line.
\end{enumerate}
\end{method}

\begin{example}[A construction program]\label{ex:g3:shapes:program}
A figure can be described by instructions:
``1. Draw a square $ABCD$ of side $4$ cm. 2. Draw the circle with
center $A$ passing through $B$.'' Following a program line by line,
anyone can redraw the same figure --- and writing one forces you to
name every point. Try it with a partner: one writes, the other
draws.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g3:shapes:1}
In the classroom, find (without measuring) three \omterm{def:g3:shapes:rightangle}{right angles} and
one angle that is not right. How can you \emph{check} each one?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:2}
Draw a rectangle of $6$ cm by $4$ cm with ruler and set square. Mark
its \omterm{def:g3:shapes:rightangle}{right angles} and its equal sides.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:3}
Draw a square of side $5$ cm. How many \omterm{def:g3:shapes:rightangle}{right angles} does it have?
How many equal sides? Mark everything.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:4}
On grid paper, draw: a $4 \times 4$ square; a $7 \times 2$
rectangle; a right triangle with legs of $3$ and $5$ squares.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:5}
True or false? Explain.
``Every square is a rectangle.'' \quad
``Every rectangle is a square.'' \quad
``A triangle can have two \omterm{def:g3:shapes:rightangle}{right angles}.'' (Try to draw one!)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:6}
Draw a circle of \omterm{def:g3:shapes:shapes}{radius} $4$ cm. Mark its center $O$, a point $A$ on
the circle, and draw the \omterm{def:g3:shapes:shapes}{radius} $[OA]$. How long is $OA$? And if $B$
is another point of the circle, how long is $OB$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:7}
Follow this program: ``1. Draw a rectangle $ABCD$ with $AB = 6$ cm
and $BC = 3$ cm. 2. Mark the point $M$ in the middle of $[AB]$.
3. Join $M$ to $C$ and to $D$.'' What shapes has the rectangle been
cut into?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:8}
Sort the shapes: which of these have at least one \omterm{def:g3:shapes:rightangle}{right angle}?
A square; a (non-square) rectangle; a right triangle; a circle; the
letter L drawn with thick strokes.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:shapes:9}
On grid paper, draw a tilted square whose sides go ``$2$ right, $1$
up'' (and the matching turns). Count squares to convince yourself
the four sides have the same length.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:shapes:10}
Write a construction program (numbered instructions, named points)
for: a square of side $3$ cm sitting on top of a rectangle
$3$ cm $\times$ $2$ cm, like a house on its ground floor. Test your
program on a classmate.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:shapes:11}
How many squares can you find in a $3 \times 3$ grid of small
squares? (Count the $1 \times 1$, the $2 \times 2$ and the
$3 \times 3$ ones.)
\end{exercise}
