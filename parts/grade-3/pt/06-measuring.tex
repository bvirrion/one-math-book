\chapter{Medindo}\label{ch:g3:measure}

Quanto tempo, quão pesado, quanto aguenta, quão tarde é? Medindo
transforma o mundo em números --- desde que todos usem o mesmo
unidades. Este capítulo apresenta as unidades de comprimento, \omterm{def:g2:measure:units}{massa} e \omterm{def:g2:measure:units}{capacidade},
e ensina como ler um relógio.

\section{Comprimentos}

\begin{definition}[Unidades de comprimento]\label{def:g3:measure:length}
A unidade básica de comprimento é o \emph{medidor} (m). Para comprimentos pequenos:
o centímetro (cm) e o milímetro (mm); para longas distâncias: o
quilômetro (km).
\[
1 \text{ m} = 100 \text{ cm}, \qquad
1 \text{ cm} = 10 \text{ mm}, \qquad
1 \text{ km} = 1\,000 \text{ m}.
\]
\end{definition}

\begin{example}[Escolhendo a unidade certa]\label{ex:g3:measure:choose}
Uma formiga: sobre $5$ mm. Um lápis: cerca de $18$ cm. Uma porta: cerca de $2$ m.
Uma caminhada até a próxima aldeia: cerca de $3$ km. Dando ``o comprimento de um
lápis é $18$'' não significa nada --- a unidade faz parte da resposta.
\end{example}

\begin{method}[Convertendo unidades]\label{met:g3:measure:convert}
Para converter, multiplicar ou dividir por $10$, $100$ ou $1\,000$:
\begin{enumerate}
  \item de uma unidade grande para uma pequena, \emph{multiplicar}:
        $3$ eu $= 3 \times 100 = 300$ cm;
  \item de uma unidade pequena para uma grande, \emph{dividir}:
        $250$ cm $= 250 \div 100 = 2$ eu e $50$ cm;
  \item para adicionar comprimentos, converta-os para o \emph{mesmo} unidade primeiro:
        $1$ eu $+ 35$ cm $= 100 + 35 = 135$ cm.
\end{enumerate}
\end{method}

\section{Massas e capacidades}

\begin{definition}[Unidades de massa e capacidade]\label{def:g3:measure:mass}
As \omterm{def:g2:measure:units}{massas} são medidas em \emph{gramas} (g) e \emph{quilogramas} (kg),
\omterm{def:g2:measure:units}{capacidades} (quanto um contêiner comporta) em \emph{litros} (Terra
\emph{centilitros} (cL):
\[
1 \text{ kg} = 1\,000 \text{ g},
\qquad
1 \text{ L} = 100 \text{ cL}.
\]
\end{definition}

\begin{example}\label{ex:g3:measure:mass}
Uma carta: sobre $20$ g. Um saco de farinha: $1$ kg. Uma banheira: cerca de
$150$ L. Um copo: cerca de $20$ cl. Para equilibrar uma balança segurando um
$1$ saco de kg contra $650$ g de maçãs, adicione $1\,000 - 650 = 350$ vá em frente
lado das maçãs.
\end{example}

\section{Lendo o relógio}

\begin{method}[Lendo o tempo]\label{met:g3:measure:clock}
Em um relógio com ponteiros:
\begin{enumerate}
  \item o \emph{curto} ponteiro dá a hora: leia o número que tem
        passou por último;
  \item o \emph{longo} ponteiro indica os minutos: cada marca grande é $5$
        minutos, então um ponteiro longo no $3$ significa $15$ minutos;
  \item depois de meia hora, também se pode ler ``até a próxima hora'':
        $10{:}50$ é ``dez para as onze''.
\end{enumerate}
\end{method}

\begin{example}[Durações]\label{ex:g3:measure:duration}
O filme começa às $14{:}40$ e termina em $16{:}10$. Quanto tempo dura?
Conte em dois saltos:
\begin{enumerate}
  \item de $14{:}40$ para $15{:}00$: $20$ minutos;
  \item de $15{:}00$ para $16{:}10$: $1$ hora $10$ minutos.
\end{enumerate}
Total: $1$ h $30$ min. Cuidado: uma hora se passou $60$ minutos, então
$16{:}10$ menos $14{:}40$ não pode ser computado como $1610 - 1440$!
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g3:measure:1}
Qual unidade você usaria (mm, cm, m ou km) para: largura de um prego;
a altura de um colega de classe; a extensão do pátio da escola; o
distância entre duas cidades?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:2}
Converter: $5$ m em cm; \quad $70$ mm em cm; \quad $4$ km em m;
\quad $600$ cm em m.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:3}
Meça com sua régua (com aproximação ao milímetro) os lados de um
triângulo desenhado por um colega de classe e calcule o comprimento total de seu
fronteira.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:4}
Calcule, convertendo primeiro: $2$ eu $+ 45$ cm (em cm); \quad
$1$ quilômetros $+ 250$ m (em m); \quad $3$ cm $+ 8$ mm (em mm).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:5}
Converter: $3$ kg em g; \quad $2\,500$ g em kg e g; \quad
$4$ L em cL; \quad $350$ cL em L e cL.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:6}
Uma receita precisa $750$ g de farinha. Mona tem um $1$ saco de kg. Ela tem
o suficiente? Quanto vai sobrar?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:7}
Leia os tempos: o ponteiro abreviado está entre $2$ e $3$, a mão longa
no $6$; o ponteiro curto no $9$, o ponteiro longo no $12$;
o ponteiro curto quase no $5$, o ponteiro longo no $10$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:8}
A escola começa às $8{:}30$ e a manhã termina às $11{:}45$. Quanto tempo
é de manhã? (Conte em lúpulo, como em
\cref{ex:g3:measure:duration}.)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:9}
O trem sai às $9{:}50$ e o passeio dura $45$ minutos. Em que
hora chega? (Pule para $10{:}00$ primeiro.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:measure:10}
Uma garrafa contém $1$ L. Paul enche copos de $20$ cl. Quantos completos
copos ele pode servir? (Converta primeiro.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:measure:11}
Lena caminha $600$ Vou para a escola e volto todos os dias. Quantos metros
ela anda em um $5$ semana escolar? Dê a resposta em metros,
depois em quilômetros.
\end{exercise}
