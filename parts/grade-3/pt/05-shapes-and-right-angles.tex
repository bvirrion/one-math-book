\chapter{Formas e ângulos retos}\label{ch:g3:shapes}

Quadrados, retângulos, triângulos, círculos: as formas são velhas amigas.
Este capítulo olha para eles com olhos de geômetra --- o que exatamente
faz de um quadrado um quadrado? --- e apresenta a ferramenta que decide:
o esquadro e seu \omterm{def:g3:shapes:rightangle}{ângulo reto}.

\section{O ângulo reto}

\begin{definition}[Ângulo reto]\label{def:g3:shapes:rightangle}
UM \emph{ângulo reto}\index{ângulo reto} é o ângulo do canto de um
folha de papel ou do esquadro. Duas linhas formando um ângulo reto
são \emph{perpendicular}. Nas figuras, um ângulo reto é marcado com um
pequeno quadrado no canto.
\end{definition}

\begin{method}[Verificando um ângulo reto]\label{met:g3:shapes:check}
\begin{enumerate}
  \item Coloque o canto do esquadro exatamente no canto para
        teste;
  \item deslize uma borda do esquadro ao longo de um lado do ângulo;
  \item se o outro lado do ângulo corre exatamente ao longo do outro
        borda, o ângulo está certo --- se abrir mais ou mais estreito,
        não é.
\end{enumerate}
\end{method}

\section{As formas e suas regras}

\begin{definition}[Quadrado, retângulo, triângulo]\label{def:g3:shapes:shapes}
\begin{itemize}
  \item UM \emph{retângulo}\index{retângulo} tem $4$ lados e $4$
        ângulos retos; seus lados opostos têm o mesmo comprimento.
  \item UM \emph{quadrado}\index{quadrado} é um retângulo cujo $4$ lados
        todos têm o mesmo comprimento.
  \item UM \emph{triângulo}\index{triângulo} tem $3$ lados; um
        \emph{triângulo retângulo} tem um \omterm{def:g3:shapes:rightangle}{ângulo reto}.
  \item UM \emph{círculo}\index{círculo} é desenhado com o compasso: tudo
        seus pontos estão à mesma distância (o \emph{raio}\index{raio}) de
        o \emph{centro}.
\end{itemize}
\end{definition}

\begin{example}[Por que as marcas são importantes]\label{ex:g3:shapes:marks}
Uma forma desenhada ligeiramente inclinada pode \emph{olhar} como um quadrado sem
sendo um. As marcas dizem a verdade: $4$ quadrados de \omterm{def:g3:shapes:rightangle}{ângulo reto} $+$ $4$
marcas de escala $=$ um verdadeiro quadrado. Ao desenhar, coloque as marcas; quando você
leia uma figura, confie apenas nas marcas e nas medidas fornecidas.
\end{example}

\section{Desenhando em papel quadriculado}

\begin{method}[Construções em grade]\label{met:g3:shapes:grid}
O papel quadriculado oferece gratuitamente ângulos retos: suas linhas são perpendiculares.
\begin{enumerate}
  \item Para desenhar um retângulo $5$ por $3$: escolha um ponto de passagem,
        contar $5$ quadrados à direita, $3$ quadrados e feche a forma
        ao longo das linhas da grade;
  \item para verificar comprimentos iguais, conte quadrados;
  \item passos diagonais formam linhas inclinadas: indo ``$1$ certo, $1$ para cima''
        desenha repetidamente um $45^\circ $ linha.
\end{enumerate}
\end{method}

\begin{example}[Um programa de construção]\label{ex:g3:shapes:program}
Uma figura pode ser descrita por instruções:
``1. Desenhe um quadrado $ ABCD $ de lado $4$ cm. 2. Desenhe o círculo com
centro $ A $ passando $ B $.'' Seguindo um programa linha por linha,
qualquer um pode redesenhar a mesma figura --- e escrever uma força você a
nomeie cada ponto. Experimente com um parceiro: um escreve, o outro
empates.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g3:shapes:1}
Na sala de aula, encontre (sem medir) três ângulos retos e
um ângulo que não está certo. Como você pode \emph{verificar} cada um?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:shapes:2}
Desenhe um retângulo de $6$ cm por $4$ cm com régua e esquadro. Marca
é ângulos retos e seus lados iguais.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:shapes:3}
Desenhe um quadrado de lado $5$ cm. Quantos ângulos retos isso tem?
Quantos lados iguais? Marque tudo.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:shapes:4}
Em papel quadriculado, desenhe: a $4 \times 4$ quadrado; um $7 \times 2$
retângulo; um triângulo retângulo com pernas de $3$ e $5$ quadrados.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:shapes:5}
Verdadeiro ou falso? Explicar.
``Todo quadrado é um retângulo.'' \quad
``Todo retângulo é um quadrado.'' \quad
``Um triângulo pode ter dois ângulos retos.'' (Tente desenhar um!)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:shapes:6}
Desenhe um círculo de \omterm{def:g3:shapes:shapes}{raio} $4$ cm. Marque seu centro $ O $, um ponto $ A $ ligado
o círculo e desenhe o \omterm{def:g3:shapes:shapes}{raio} $[OA]$. Quanto tempo é $ OA $? E se $ B $
é outro ponto do círculo, qual é o comprimento $ OB $?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:shapes:7}
Siga este programa: ``1. Desenhe um retângulo $ ABCD $ com $ AB = 6$ centímetros
e $ BC = 3$ cm. 2. Marque o ponto $ M $ no meio de $[AB]$.
3. Junte-se $ M $ para $ C $ e para $ D $.'' Quais são as formas do retângulo
cortado em?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:shapes:8}
Classifique as formas: quais delas têm pelo menos um \omterm{def:g3:shapes:rightangle}{ângulo reto}?
Um quadrado; um retângulo (não quadrado); um triângulo retângulo; um círculo; o
letra L desenhada com traços grossos.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:shapes:9}
Em papel quadriculado, desenhe um quadrado inclinado cujos lados vão ``$2$ certo, $1$
up'' (e os turnos correspondentes). Conte quadrados para se convencer
os quatro lados têm o mesmo comprimento.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:shapes:10}
Escreva um programa de construção (instruções numeradas, pontos nomeados)
para: um quadrado de lado $3$ cm sentado em cima de um retângulo
$3$ cm $\times $ $2$ cm, como uma casa no térreo. Teste o seu
programa em um colega de classe.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:shapes:11}
Quantos quadrados você consegue encontrar em um $3 \times 3$ grade de pequenos
quadrados? (Conte o $1 \times 1$, o $2 \times 2$ e o
$3 \times 3$ uns.)
\end{exercise}
