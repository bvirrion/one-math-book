\chapter{Números até 10\,000}\label{ch:g3:numbers}

Mil, dois mil, três mil\,\dots\ números grandes são
construído a partir de pequenos, usando uma ideia maravilhosa: o \emph{lugar} de um
dígito indica seu valor. Este capítulo ensina como ler, escrever,
compare e ordene números até $10\,000$.

\section{Ler e escrever números}

\begin{definition}[Valor posicional]\label{def:g3:numbers:place}
Um número é escrito com os algarismos $0, 1, 2, \dots, 9$. Lendo de
à direita, os lugares são: unidades, dezenas, centenas, milhares. Em
$3\,254$:
\[
3\,254 = 3 \text{ thousands} + 2 \text{ hundreds} + 5 \text{ tens}
+ 4 \text{ units}.
\]
\end{definition}

\begin{example}[O zero que ocupa um lugar]\label{ex:g3:numbers:zero}
``Quatro mil e sete'' está escrito $4\,007$: um $4$ no
milhares, um $7$ nas unidades e \emph{\omterm{def:g1:counting:zero}{zeros}} no meio para manter
cada dígito em seu lugar certo. Sem eles, $47$ seria muito
número diferente!
\end{example}

\begin{example}[Troca]\label{ex:g3:numbers:exchange}
Dez unidades equivalem a um dez; dez dezenas equivalem a cem; dez centenas fazem
mil. Então $10$ centenas $= 1\,000$, e $32$ dezenas
$= 320$: o número $320$ contém $32$ dezenas (não apenas o dígito
$2$!).
\end{example}

\section{Comparando e ordenando}

\begin{method}[Comparando dois números]\label{met:g3:numbers:compare}
\begin{enumerate}
  \item Conte os dígitos: mais dígitos significa um número maior
        ($1\,002 > 998$).
  \item Mesmo número de dígitos: compare dígito por dígito começando em
        o \emph{esquerda}. A primeira \omterm{ex:g1:subtraction:difference}{diferença} decide:
        $5\,614 > 5\,589$ porque $6 > 5$ às centenas.
  \item Escreva a resposta com os sinais $<$ (menor que) ou $>$
        (maior que): o lado aberto sempre \omterm{def:g2:shapes:solids}{rostos} o número maior.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:numbers:order}
Ordene do menor para o maior: $780$, $8\,700$, $807$, $87$.
Primeiro pelo número de dígitos: $87$ (dois), então $780$ e $807$
(três), então $8\,700$ (quatro). Entre $780$ e $807$: compare o
centenas de dígitos, $7 < 8$, então $780 < 807$. Pedido final:
\[
87 < 780 < 807 < 8\,700 .
\]
\end{example}

\section{Par e ímpar}

\begin{definition}[Par, ímpar]\label{def:g3:numbers:evenodd}
Um número é \emph{par}\index{par} quando pode ser dividido em
duas partes inteiras iguais --- seu dígito das unidades é $0$, $2$, $4$, $6$ ou
$8$. Caso contrário, é \emph{ímpar}\index{ímpar} (seu dígito das unidades é $1$,
$3$, $5$, $7$ ou $9$).
\end{definition}

\begin{example}
$46$ é \omterm{def:g3:numbers:evenodd}{par} ($46 = 23 + 23$); $47$ é \omterm{def:g3:numbers:evenodd}{ímpar} (duas partes iguais seriam
preciso de meio objeto). Apenas o \emph{último} dígito importa: $3\,578$ é
\omterm{def:g3:numbers:evenodd}{par}, $2\,341$ é \omterm{def:g3:numbers:evenodd}{ímpar}.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g3:numbers:1}
Escreva em algarismos: ``dois mil trezentos e quarenta e cinco''; ``seis
mil e cinquenta''; ``nove mil e um''.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:2}
Escreva em palavras: $1\,208$; \quad $4\,070$; \quad $9\,999$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:3}
Em $7\,382$: qual dígito está na casa das dezenas? Na casa dos milhares
lugar? Quantos \emph{centenas} o número contém no total
(cuidado: não apenas o dígito das centenas --- pense em
\cref{ex:g3:numbers:exchange})?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:4}
Decompor como em \cref{def:g3:numbers:place}:
$5\,631$; \quad $2\,046$; \quad $8\,500$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:5}
Copie e complete com $<$ ou $>$:
\[
456 \;?\; 465, \qquad
1\,203 \;?\; 989, \qquad
6\,090 \;?\; 6\,009, \qquad
9\,999 \;?\; 10\,000 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:6}
Ordene do menor para o maior: $530$; \ $3\,500$; \ $353$; \ $5\,033$;
\ $503$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:7}
Em uma reta numérica graduada em centenas de $0$ para $1\,000$, lugar
(aproximadamente) os números $150$, $480$, $820$ e $990$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:8}
\omterm{def:g3:numbers:evenodd}{Par} ou \omterm{def:g3:numbers:evenodd}{ímpar}? \ $34$; \ $57$; \ $70$; \ $2\,481$; \ $6\,548$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:9}
Conte por etapas: a partir de $2\,970$, conte cinco passos de $10$; de $860$,
conte quatro passos de $100$; de $9\,996$, conte quatro passos de $1$.
O que você percebe no final do último?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:numbers:10}
Usando cada um dos dígitos $3$, $7$, $0$, $5$ exatamente uma vez, escreva o
maior número possível de quatro dígitos, depois o menor (um número
não é possível começar com $0$).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:numbers:11}
Eu sou um número de três dígitos. Meus dígitos são $2$, $5$ e $8$, Eu sou \omterm{def:g3:numbers:evenodd}{par},
e eu sou maior que $800$. Quem sou eu?
\end{exercise}
