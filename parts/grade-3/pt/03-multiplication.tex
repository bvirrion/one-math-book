\chapter{Multiplicação}\label{ch:g3:mult}

Três sacos com sete bolinhas de gude: você pode adicionar $7 + 7 + 7$, mas há
uma maneira mais rápida --- multiplicação, a operação de adição repetida.
Este capítulo constrói a tabuada a partir de imagens e, em seguida,
ensina o método da coluna.

\section{O que significa multiplicar}

\begin{definition}[Multiplicação]\label{def:g3:mult:def}
O \omterm{def:g2:mult:def}{produto} $3 \times 7$ significa ``três vezes sete'':
\[
3 \times 7 = 7 + 7 + 7 = 21 .
\]
Em uma foto, $3 \times 7$ é um retângulo de pontos: $3$ linhas de $7$
pontos.
\end{definition}

{\small O mesmo $21$ pontos, contados por linhas ou colunas: a ordem
de um \omterm{def:g2:mult:def}{produto} Não importa. Sabendo $3 \times 7$ dá
$7 \times 3$ de graça --- metade das mesas para aprender!}
\end{omfigure}

\begin{proposition}[Regras que ajudam]\label{prop:g3:mult:rules}
\begin{enumerate}
  \item A ordem não importa: $ a \times b = b \times a $.
  \item Multiplicando por $1$ não muda nada; multiplicando por $0$ dá
        $0$.
  \item Multiplicando por $10$ anexa um \omterm{def:g1:counting:zero}{zero}: $34 \times 10 = 340$;
        por $100$, dois \omterm{def:g1:counting:zero}{zeros}.
  \item Uma tabela pode ser reconstruída a partir de suas vizinhas:
        $6 \times 8 = 5 \times 8 + 8 = 40 + 8 = 48$.
\end{enumerate}
\end{proposition}
\admitted

\section{Multiplicação de coluna}

\begin{method}[Multiplicando por um número de um dígito]\label{met:g3:mult:column}
Para calcular $234 \times 6$:
\begin{enumerate}
  \item multiplique as unidades: $4 \times 6 = 24$: escrever $4$, carregar $2$;
  \item multiplique as dezenas e adicione o carry:
        $3 \times 6 + 2 = 20$: escrever $0$, carregar $2$;
  \item multiplique as centenas e adicione o carry:
        $2 \times 6 + 2 = 14$: escrever $14$.
\end{enumerate}
\[
\begin{array}{r}
2\,3\,4 \\
\times \quad 6 \\
\hline
1\,4\,0\,4
\end{array}
\]
Estimativa para verificar: $234$ está perto de $200$, e
$200 \times 6 = 1\,200$: a resposta $1\,404$ é plausível.
\end{method}

\begin{example}[Dividir para multiplicar mentalmente]\label{ex:g3:mult:split}
O método da coluna divide secretamente o número por lugares. Mentalmente, faça
o mesmo:
\[
23 \times 4 = (20 + 3) \times 4 = 20 \times 4 + 3 \times 4
= 80 + 12 = 92 .
\]
E com um vizinho amigável: $99 \times 5 = 100 \times 5 - 5 =
495$.
\end{example}

\section{Multiplicação em problemas}

\begin{example}\label{ex:g3:mult:problem}
Um microônibus transporta $9$ passageiros. Quantos passageiros podem $27$
microônibus carregam?
\begin{enumerate}
  \item É uma multiplicação: $27$ grupos de $9$.
  \item $27 \times 9 = 243$ (colunas ou
        $27 \times 9 = 27 \times 10 - 27 = 270 - 27$).
  \item Frase: os microônibus podem transportar $243$ passageiros.
\end{enumerate}
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g3:mult:1}
Escreva como uma multiplicação e calcule:
$8 + 8 + 8 + 8$; \quad $5 + 5 + 5$; \quad
$20 + 20 + 20 + 20 + 20$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:2}
Recite e complete: $6 \times 7$; \ $8 \times 4$; \ $9 \times 9$;
\ $7 \times 8$; \ $6 \times 6$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:3}
Desenhe um retângulo de pontos para $4 \times 6$, então um para $6 \times 4$.
O que você nota sobre as duas contagens?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:4}
Calcule usando as regras de \cref{prop:g3:mult:rules}:
$56 \times 10$; \quad $8 \times 100$; \quad $73 \times 0$; \quad
$45 \times 1$; \quad $30 \times 10$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:5}
Calcule em colunas: $132 \times 3$; \quad $217 \times 4$; \quad
$408 \times 7$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:6}
Calcule mentalmente dividindo (como em \cref{ex:g3:mult:split}):
$31 \times 5$; \quad $42 \times 3$; \quad $25 \times 6$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:7}
Calcule com um vizinho amigável: $99 \times 7$; \quad
$101 \times 6$; \quad $98 \times 5$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:8}
Estime primeiro e depois calcule: $389 \times 5$ (estimativa com $400$);
$612 \times 8$ (estimativa com $600$).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:9}
Uma caixa de ovos contém $6$ ovos. Quantos ovos tem $38$ caixas? Escreva
a operação, calcule, responda com uma frase.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:mult:10}
Uma sala de aula tem $8$ linhas de $4$ mesas; cada assento de mesa $2$ estudantes.
Quantos alunos a sala pode acomodar? (Duas multiplicações.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:mult:11}
Encontre todas as formas de organizar $24$ cadeiras em fileiras iguais (fileiras de
$1$, de $2$, \dots). Quais arranjos são possíveis? (Use a tabela
quadrado: procure $24$ dentro dele.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:mult:12}
Zoe calcula $47 \times 6$ assim: $40 \times 6 = 240$,
$7 \times 6 = 42$, e então $240 + 42 = 282$. Léa calcula
$47 \times 6 = 50 \times 6 - 3 \times 6 = 300 - 18 = 282$. Explique
cada método. Para qual você prefere $38 \times 4$? Calcule ambos
maneiras.
\end{exercise}
