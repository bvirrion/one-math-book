\chapter{Compartilhamento e Divisão}\label{ch:g3:division}

Vinte bolinhas de gude divididas de forma justa entre quatro crianças: quantas cada?
A divisão responde a todas as questões de compartilhamento --- e o agrupamento
perguntas também --- e é o gêmeo da multiplicação: desfaz-se o
outro.

\section{Duas faces da divisão}

\begin{definition}[Divisão]\label{def:g3:division:def}
A divisão $20 \div 4$ responde a dois tipos de perguntas:
\begin{itemize}
  \item \emph{compartilhamento}: $20$ bolinhas de gude compartilhadas entre $4$ crianças ---
        quantas bolinhas de gude \emph{cada}? ($5$ cada);
  \item \emph{agrupamento}: $20$ bolas de gude colocadas em sacos de $4$ --- quantos
        \emph{bolsas}? ($5$ bolsas).
\end{itemize}
Ambas as perguntas têm a mesma resposta, porque ambas desfazem o mesmo
multiplicação: $4 \times 5 = 20$.
\end{definition}

{\small Compartilhamento $20$ em $4$ grupos iguais. Agrupando por $4$ em vez disso

\end{omfigure}

\begin{method}[Dividindo com as mesas]\label{met:g3:division:tables}
Para calcular $42 \div 6$, pergunte: ``$6$ vezes \emph{o que} faz $42$?''
A mesa de $6$ respostas: $6 \times 7 = 42$, então $42 \div 6 = 7$.
Cada questão de divisão é uma questão de multiplicação lida
para trás.
\end{method}

\section{Quando não sai mesmo}

\begin{definition}[Restante]\label{def:g3:division:remainder}
Compartilhamento $23$ bolinhas de gude entre $4$ crianças dá $5$ bolinhas de gude cada ---
e $3$ bolinhas de gude \emph{sobra}, não o suficiente para outra rodada. Nós
escrever
\[
23 = 4 \times 5 + 3 ,
\]
e diga: o \emph{quociente}\index{quociente} é $5$, o
\emph{restante}\index{restante} é $3$. O resto é sempre
menor que o número de crianças --- caso contrário, outra rodada de
o compartilhamento seria possível.
\end{definition}

\end{omfigure}

\begin{example}\label{ex:g3:division:remainder}
Dividir $58$ por $7$. Digitalize a tabela de $7$:
$7 \times 8 = 56$ é o maior \omterm{def:g2:mult:def}{produto} que cabe sob $58$
($7 \times 9 = 63$ é demais). Então
\[
58 = 7 \times 8 + 2 :
\]
\omterm{def:g3:division:remainder}{quociente} $8$, \omterm{def:g3:division:remainder}{restante} $2$. Verificar: $56 + 2 = 58$, e $2 < 7$.
\end{example}

\begin{method}[Interpretando o restante]\label{met:g3:division:interpret}
Após a divisão, volte à história:
\begin{enumerate}
  \item ``quantos cada / quantos grupos completos?'' --- o \omterm{def:g3:division:remainder}{quociente};
  \item ``quantos sobraram?'' --- o \omterm{def:g3:division:remainder}{restante};
  \item ``quantas caixas para guardar tudo?'' --- o \omterm{def:g3:division:remainder}{quociente}
        \emph{mais um} se o \omterm{def:g3:division:remainder}{restante} não é \omterm{def:g1:counting:zero}{zero}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:division:interpret}
$58$ os alunos praticam canoagem, $7$ por canoa. $58 = 7 \times 8 + 2$:
oito canoas estão cheias, e $2$ os alunos permanecem --- eles precisam de uma canoa
também. Então $9$ canoas devem ser reservadas.
\end{example}

\section{Metades, terços, quartos}

\begin{definition}[Metade, terceiro, quarto]\label{def:g3:division:half}
O \emph{metade}\index{metade} de um número é o resultado da divisão dele por $2$; o
\emph{terceiro}, por $3$; o \emph{trimestre}\index{trimestre}, por $4$. Por exemplo o
metade de $18$ é $9$, o terceiro de $18$ é $6$, o quarto de $36$
é $9$.
\end{definition}

\begin{example}[Duplas e metades andam juntas]\label{ex:g3:division:double}
O \omterm{def:g3:division:half}{metade} de $46$: desde $46 = 40 + 6$, \omterm{def:g3:division:half}{metade} de cada parte dá
$20 + 3 = 23$. Verifique com o duplo: $23 + 23 = 46$. Sabendo
dobra de cor ($25 \to 50$, $45 \to 90$, $150 \to 300$) faz
\omterm{def:g3:division:half}{metades} instantâneas.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g3:division:1}
Calcule usando as tabelas ao contrário: $36 \div 4$; \quad $63 \div 9$;
\quad $40 \div 5$; \quad $56 \div 8$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:2}
Para cada divisão, escreva a igualdade $ a = b \times q + r $ e dar
o \omterm{def:g3:division:remainder}{quociente} e o \omterm{def:g3:division:remainder}{restante}: $17 \div 5$; \quad $30 \div 4$; \quad
$50 \div 6$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:3}
$28$ bolinhas de gude são compartilhadas entre $6$ crianças. Quantas bolinhas de gude cada,
e quantos sobraram? Desenhe o compartilhamento se ajudar.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:4}
Calcular: o \omterm{def:g3:division:half}{metade} de $68$; o terceiro de $27$; o \omterm{def:g3:division:half}{trimestre} de $48$;
o \omterm{def:g3:division:half}{metade} de $150$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:5}
Verdadeiro ou falso? Verifique com uma multiplicação.
``$45 \div 5 = 9$''; \quad ``$52 \div 6 = 8$ \omterm{def:g3:division:remainder}{restante} $4$'';
\quad ``$70 \div 8 = 9$ \omterm{def:g3:division:remainder}{restante} $2$''.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:6}
$32$ as fotos são coladas em um álbum, $4$ por página. Quantas páginas são
usado? Qual face de divisão é essa (compartilhar ou agrupar)?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:7}
Uma corda de $75$ cm é cortado em pedaços de $9$ cm. Quantas peças inteiras,
e quanto tempo dura a sobra?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:8}
$50$ crianças viajam em carros de $4$ assentos. Quantos carros são
necessário? (Cuidado: todos devem andar!)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:9}
Encontre os números que faltam: $? \div 7 = 6$; \quad
$54 \div ? = 6$; \quad o \omterm{def:g3:division:half}{metade} de $?$ é $35$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:division:10}
Amina compartilha ela $47$ adesivos entre ela $5$ amigos, tão justamente quanto
possível, e guarda o que sobrou para si. Quantos adesivos tem
cada amigo ganha e quantos Amina fica?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:division:11}
Quando um número é dividido por $6$, quais são todos os possíveis
restos? Qual é o maior número cuja divisão por $6$ tem
\omterm{def:g3:division:remainder}{quociente} $7$?
\end{exercise}
