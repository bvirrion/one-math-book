\chapter{Adição e Subtração}\label{ch:g3:addsub}

Adicionar significa juntar; subtrair significa tirar, ou
encontrar o que está faltando. Este capítulo treina os dois métodos de coluna
--- com seus famosos carregamentos e empréstimos --- e os truques mentais
que muitas vezes tornam as colunas desnecessárias.

\section{Adição de coluna}

\begin{method}[Adicionando colunas]\label{met:g3:addsub:add}
\begin{enumerate}
  \item Escreva os números um abaixo do outro, unidades sob unidades,
        dezenas sob dezenas (alinhe-se no \emph{certo});
  \item adicione as unidades; se o \omterm{def:g1:addition:def}{soma} passes $9$, escreva seu algarismo de unidades
        e \emph{carregar} os dez para a próxima coluna (um pequeno $1$);
  \item continue com as dezenas, as centenas, \dots, adicionando o carry
        cada vez.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:addsub:add}
Calcular $457 + 368$:
\[
\begin{array}{r}
{\scriptstyle 1}\ \ {\scriptstyle 1}\ \ \phantom{0} \\[-3pt]
4\,5\,7 \\
+\ 3\,6\,8 \\
\hline
8\,2\,5
\end{array}
\]
Unidades: $7 + 8 = 15$: escrever $5$, carregar $1$. Dezenas:
$5 + 6 + 1 = 12$: escrever $2$, carregar $1$. Centenas: $4 + 3 + 1 = 8$.
Resultado: $825$.
\end{example}

\begin{example}[Adição mental]\label{ex:g3:addsub:mental}
Procure números amigáveis antes de chegar às colunas:
\begin{itemize}
  \item \emph{faça um dez}: $47 + 8 = 47 + 3 + 5 = 50 + 5 = 55$;
  \item \emph{pular por partes}: $256 + 130 = 256 + 100 + 30 = 386$;
  \item \emph{reordenar}: $17 + 59 + 3 = (17 + 3) + 59 = 79$.
\end{itemize}
\end{example}

\section{Subtração de coluna}

\begin{method}[Subtraindo em colunas]\label{met:g3:addsub:sub}
\begin{enumerate}
  \item Escreva o número maior em cima, alinhado à direita;
  \item subtrair coluna por coluna, começando pelas unidades;
  \item se o dígito superior for muito pequeno, \emph{emprestar}: pegue um de
        o próximo lugar na linha superior (vale dez no atual
        coluna);
  \item \emph{verificar}: o resultado mais o número subtraído deve dar
        de volta o número superior.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:addsub:sub}
Calcular $603 - 147$:
\[
\begin{array}{r}
6\,0\,3 \\
-\ 1\,4\,7 \\
\hline
4\,5\,6
\end{array}
\]
Unidades: $3 - 7$ impossível; empréstimo através de: $603$ torna-se
$5$ centenas, $9$ dezenas, $13$ unidades. Então $13 - 7 = 6$;
dezenas $9 - 4 = 5$; centenas $5 - 1 = 4$. Resultado: $456$.
Verificar: $456 + 147 = 603$.
\end{example}

\begin{example}[Subtração como parte faltante]\label{ex:g3:addsub:missing}
“Lila tem $35$ bolinhas de gude, Tom tem $52$: quantos mais o Tom tem?'' é
a subtração $52 - 35$. Mentalmente, conte \emph{acima} de $35$:
$35 \to 40$ é $5$, então $40 \to 52$ é $12$; no total
$5 + 12 = 17$. Contar sempre é mais fácil do que pedir emprestado.
\end{example}

\end{omfigure}

\section{Ordens de magnitude}

\begin{method}[Estimando antes de calcular]\label{met:g3:addsub:estimate}
Antes de qualquer cálculo de coluna, arredonde os números para números amigáveis e
calcular um \emph{estimativa}. Depois de calcular, compare: um resultado muito
da estimativa sinaliza um erro.
\end{method}

\begin{example}
Para $487 + 312$: a estimativa é $500 + 300 = 800$. A coluna
resultado, $799$, está próximo: plausível. Se você tivesse encontrado $1\,099$
(um alinhamento errado), a estimativa teria avisado você.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g3:addsub:1}
Calcule em colunas: $346 + 253$; \quad $478 + 265$; \quad
$1\,536 + 2\,876$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:addsub:2}
Calcule em colunas e verifique por adição: $587 - 234$; \quad
$805 - 348$; \quad $1\,000 - 463$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:addsub:3}
Calcule mentalmente formando dezenas: $38 + 7$; \quad $56 + 9$; \quad
$45 + 28$ (adicionar $30$, remover $2$).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:addsub:4}
Calcule mentalmente por saltos: $340 + 220$; \quad $675 + 210$; \quad
$512 - 300$; \quad $840 - 220$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:addsub:5}
Calcular $63 - 47$ contando (como em
\cref{ex:g3:addsub:missing}) e faça check-in das colunas.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:addsub:6}
Estime primeiro (arredondar para centenas) e depois calcule exatamente:
$492 + 218$; \quad $913 - 388$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:addsub:7}
Uma escola tem $348$ meninas e $296$ meninos. Quantos alunos ao todo?
Escreva a operação, calcule e responda com uma frase.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:addsub:8}
Um livro tem $224$ páginas; Nora leu $178$. Quantas páginas faltam
ler?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:addsub:9}
Complete os furos (trabalhe coluna por coluna):
\[
\begin{array}{r}
3\,?\,7 \\
+\ ?\,2\,? \\
\hline
8\,6\,2
\end{array}
\]
Existe apenas uma maneira de preencher os três buracos? Explicar.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:addsub:10}
Uma padaria vendida $145$ croissants pela manhã e $89$ no
tarde; $17$ croissants não foram vendidos na hora de fechar. Quantos
croissants foram assados naquele dia? (Dois passos.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:addsub:11}
Theo diz: ``para calcular $500 - 199$, eu calculo $500 - 200$ e então
adicionar $1$.'' Ele está certo? Calcule nos dois sentidos e explique seu truque.
\end{exercise}
