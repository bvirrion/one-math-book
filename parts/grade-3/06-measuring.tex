\chapter{Measuring}\label{ch:g3:measure}

How long, how heavy, how much does it hold, how late is it? Measuring
turns the world into numbers --- provided everyone uses the same
units. This chapter presents the units of length, \omterm{def:g2:measure:units}{mass} and \omterm{def:g2:measure:units}{capacity},
and teaches how to read a clock.

\section{Lengths}

\begin{definition}[Units of length]\label{def:g3:measure:length}
The basic unit of length is the \emph{meter} (m). For small lengths:
the centimeter (cm) and the millimeter (mm); for long distances: the
kilometer (km).
\[
1 \text{ m} = 100 \text{ cm}, \qquad
1 \text{ cm} = 10 \text{ mm}, \qquad
1 \text{ km} = 1\,000 \text{ m}.
\]
\end{definition}

\begin{example}[Choosing the right unit]\label{ex:g3:measure:choose}
An ant: about $5$ mm. A pencil: about $18$ cm. A door: about $2$ m.
A walk to the next village: about $3$ km. Giving ``the length of a
pencil is $18$'' means nothing --- the unit is part of the answer.
\end{example}

\begin{method}[Converting units]\label{met:g3:measure:convert}
To convert, multiply or divide by $10$, $100$ or $1\,000$:
\begin{enumerate}
  \item from a big unit to a small one, \emph{multiply}:
        $3$ m $= 3 \times 100 = 300$ cm;
  \item from a small unit to a big one, \emph{divide}:
        $250$ cm $= 250 \div 100 = 2$ m and $50$ cm;
  \item to add lengths, convert them to the \emph{same} unit first:
        $1$ m $+ 35$ cm $= 100 + 35 = 135$ cm.
\end{enumerate}
\end{method}

\section{Masses and capacities}

\begin{definition}[Units of mass and capacity]\label{def:g3:measure:mass}
Masses are measured in \emph{grams} (g) and \emph{kilograms} (kg),
capacities (how much a container holds) in \emph{liters} (L) and
\emph{centiliters} (cL):
\[
1 \text{ kg} = 1\,000 \text{ g},
\qquad
1 \text{ L} = 100 \text{ cL}.
\]
\end{definition}

\begin{example}\label{ex:g3:measure:mass}
A letter: about $20$ g. A bag of flour: $1$ kg. A bathtub: about
$150$ L. A glass: about $20$ cL. To balance a scale holding a
$1$ kg bag against $650$ g of apples, add $1\,000 - 650 = 350$ g on
the apples' side.
\end{example}

\section{Reading the clock}

\begin{method}[Reading the time]\label{met:g3:measure:clock}
On a clock with hands:
\begin{enumerate}
  \item the \emph{short} hand gives the hour: read the number it has
        passed last;
  \item the \emph{long} hand gives the minutes: each big mark is $5$
        minutes, so a long hand on the $3$ means $15$ minutes;
  \item after half past, one can also read ``to the next hour'':
        $10{:}50$ is ``ten to eleven''.
\end{enumerate}
\end{method}

\begin{example}[Durations]\label{ex:g3:measure:duration}
The film starts at $14{:}40$ and ends at $16{:}10$. How long is it?
Count in two hops:
\begin{enumerate}
  \item from $14{:}40$ to $15{:}00$: $20$ minutes;
  \item from $15{:}00$ to $16{:}10$: $1$ hour $10$ minutes.
\end{enumerate}
Total: $1$ h $30$ min. Careful: an hour has $60$ minutes, so
$16{:}10$ minus $14{:}40$ cannot be computed like $1610 - 1440$!
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g3:measure:1}
Which unit would you use (mm, cm, m or km) for: the width of a nail;
the height of a classmate; the length of the schoolyard; the
distance between two towns?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:2}
Convert: $5$ m into cm; \quad $70$ mm into cm; \quad $4$ km into m;
\quad $600$ cm into m.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:3}
Measure with your ruler (to the nearest millimeter) the sides of a
triangle drawn by a classmate, and compute the total length of its
border.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:4}
Compute, converting first: $2$ m $+ 45$ cm (in cm); \quad
$1$ km $+ 250$ m (in m); \quad $3$ cm $+ 8$ mm (in mm).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:5}
Convert: $3$ kg into g; \quad $2\,500$ g into kg and g; \quad
$4$ L into cL; \quad $350$ cL into L and cL.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:6}
A recipe needs $750$ g of flour. Mona has a $1$ kg bag. Does she have
enough? How much will be left?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:7}
Read the times: the short hand is between $2$ and $3$, the long hand
on the $6$; the short hand on the $9$, the long hand on the $12$;
the short hand almost on the $5$, the long hand on the $10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:8}
School starts at $8{:}30$ and the morning ends at $11{:}45$. How long
is the morning? (Count in hops, as in
\cref{ex:g3:measure:duration}.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:9}
The train leaves at $9{:}50$ and the ride lasts $45$ minutes. At what
time does it arrive? (Hop to $10{:}00$ first.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:measure:10}
A bottle holds $1$ L. Paul fills glasses of $20$ cL. How many full
glasses can he pour? (Convert first.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:measure:11}
Lena walks $600$ m to school, and back, every day. How many meters
does she walk in a $5$-day school week? Give the answer in meters,
then in kilometers.
\end{exercise}
