\chapter{Les nombres jusqu'\`a 10\,000}\label{ch:g3:numbers}

Mille, deux mille, trois mille\,\dots\ les grands nombres se
construisent \`a partir des petits, gr\^ace \`a une id\'ee
merveilleuse~: la \emph{place} d'un chiffre dit sa valeur. Ce chapitre
apprend \`a lire, \'ecrire, comparer et ranger les nombres jusqu'\`a
$10\,000$.

\section{Lire et \'ecrire les nombres}

\begin{definition}[Valeur de position]\label{def:g3:numbers:place}
Un nombre s'\'ecrit avec les chiffres $0, 1, 2, \dots, 9$. En lisant
de droite \`a gauche, les places sont~: unit\'es, dizaines, centaines,
milliers. Dans $3\,254$~:
\[
3\,254 = 3 \text{ milliers} + 2 \text{ centaines} + 5 \text{ dizaines}
+ 4 \text{ unit\'es}.
\]
\end{definition}

\begin{example}[Le z\'ero qui garde la place]\label{ex:g3:numbers:zero}
\guillemotleft{} Quatre mille sept \guillemotright{} s'\'ecrit
$4\,007$~: un $4$ aux milliers, un $7$ aux unit\'es, et des
\emph{\omterm{def:g1:counting:zero}{z\'eros}} entre les deux pour garder chaque chiffre \`a sa
place. Sans eux, $47$ serait un nombre bien diff\'erent~!
\end{example}

\begin{example}[\'Echanges]\label{ex:g3:numbers:exchange}
Dix unit\'es font une dizaine~; dix dizaines font une centaine~; dix
centaines font un millier. Donc $10$ centaines $= 1\,000$, et $32$
dizaines $= 320$~: le nombre $320$ contient $32$ dizaines (pas
seulement le chiffre $2$~!).
\end{example}

\section{Comparer et ranger}

\begin{method}[Comparer deux nombres]\label{met:g3:numbers:compare}
\begin{enumerate}
  \item Compter les chiffres~: plus de chiffres signifie un plus grand
        nombre ($1\,002 > 998$).
  \item M\^eme nombre de chiffres~: comparer chiffre par chiffre en
        partant de la \emph{gauche}. La premi\`ere \omterm{ex:g1:subtraction:difference}{diff\'erence}
        d\'ecide~: $5\,614 > 5\,589$ parce que $6 > 5$ aux centaines.
  \item \'Ecrire la r\'eponse avec les signes $<$ (plus petit que) ou
        $>$ (plus grand que)~: le c\^ot\'e ouvert fait toujours face
        au plus grand nombre.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:numbers:order}
Ranger du plus petit au plus grand~: $780$, $8\,700$, $807$, $87$.
D'abord par le nombre de chiffres~: $87$ (deux), puis $780$ et $807$
(trois), puis $8\,700$ (quatre). Entre $780$ et $807$~: comparer les
chiffres des centaines, $7 < 8$, donc $780 < 807$. Ordre final~:
\[
87 < 780 < 807 < 8\,700 .
\]
\end{example}

\section{Pairs et impairs}

\begin{definition}[Pair, impair]\label{def:g3:numbers:evenodd}
Un nombre est \emph{pair}\index{nombre pair} quand on peut le
partager en deux parties enti\`eres \'egales --- son chiffre des
unit\'es est $0$, $2$, $4$, $6$ ou $8$. Sinon il est
\emph{impair}\index{nombre impair} (chiffre des unit\'es $1$, $3$,
$5$, $7$ ou $9$).
\end{definition}

\begin{example}
$46$ est \omterm{def:g3:numbers:evenodd}{pair} ($46 = 23 + 23$)~; $47$ est \omterm{def:g3:numbers:evenodd}{impair} (deux parties
\'egales demanderaient un demi-objet). Seul le \emph{dernier} chiffre
compte~: $3\,578$ est \omterm{def:g3:numbers:evenodd}{pair}, $2\,341$ est \omterm{def:g3:numbers:evenodd}{impair}.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g3:numbers:1}
\'Ecris en chiffres~: \guillemotleft{} deux mille trois cent
quarante-cinq \guillemotright{}~; \guillemotleft{} six mille
cinquante \guillemotright{}~; \guillemotleft{} neuf mille un
\guillemotright{}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:2}
\'Ecris en lettres~: $1\,208$~; \quad $4\,070$~; \quad $9\,999$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:3}
Dans $7\,382$~: quel chiffre est \`a la place des dizaines~? \`A la
place des milliers~? Combien de \emph{centaines} le nombre contient-il
en tout (attention~: pas seulement le chiffre des centaines ---
pense \`a \cref{ex:g3:numbers:exchange})~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:4}
D\'ecompose comme dans \cref{def:g3:numbers:place}~:
$5\,631$~; \quad $2\,046$~; \quad $8\,500$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:5}
Recopie et compl\`ete avec $<$ ou $>$~:
\[
456 \;?\; 465, \qquad
1\,203 \;?\; 989, \qquad
6\,090 \;?\; 6\,009, \qquad
9\,999 \;?\; 10\,000 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:6}
Range du plus petit au plus grand~: $530$~; \ $3\,500$~; \ $353$~; \
$5\,033$~; \ $503$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:7}
Sur une droite num\'erique gradu\'ee en centaines de $0$ \`a
$1\,000$, place (approximativement) les nombres $150$, $480$, $820$
et $990$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:8}
\omterm{def:g3:numbers:evenodd}{Pair} ou \omterm{def:g3:numbers:evenodd}{impair}~? \ $34$~; \ $57$~; \ $70$~; \ $2\,481$~; \ $6\,548$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:9}
Compte par pas~: \`a partir de $2\,970$, compte cinq pas de $10$~; \`a
partir de $860$, compte quatre pas de $100$~; \`a partir de $9\,996$,
compte quatre pas de $1$. Que remarques-tu \`a la fin du dernier~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:numbers:10}
En utilisant chacun des chiffres $3$, $7$, $0$, $5$ exactement une
fois, \'ecris le plus grand nombre \`a quatre chiffres possible, puis
le plus petit (un nombre ne peut pas commencer par $0$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:numbers:11}
Je suis un nombre \`a trois chiffres. Mes chiffres sont $2$, $5$ et
$8$, je suis \omterm{def:g3:numbers:evenodd}{pair}, et je suis plus grand que $800$. Qui suis-je~?
\end{exercise}
