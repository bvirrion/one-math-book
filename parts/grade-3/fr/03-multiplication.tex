\chapter{Multiplication}\label{ch:g3:mult}

Trois sacs de sept billes~: on pourrait additionner $7 + 7 + 7$, mais
il y a un moyen plus rapide --- la multiplication, l'op\'eration de
l'addition r\'ep\'et\'ee. Ce chapitre construit les tables de
multiplication \`a partir d'images, puis enseigne la technique en
colonnes.

\section{Ce que signifie multiplier}

\begin{definition}[Multiplication]\label{def:g3:mult:def}
Le \omterm{def:g2:mult:def}{produit} $3 \times 7$ signifie \guillemotleft{} trois fois sept
\guillemotright{}~:
\[
3 \times 7 = 7 + 7 + 7 = 21 .
\]
Sur une image, $3 \times 7$ est un rectangle de points~: $3$ lignes
de $7$ points.
\end{definition}

{\small Les m\^emes $21$ points, compt\'es par lignes ou par
colonnes~: l'ordre d'un \omterm{def:g2:mult:def}{produit} n'a pas d'importance. Conna\^itre
$3 \times 7$ donne $7 \times 3$ gratuitement --- la moiti\'e des
tables \`a apprendre~!}
\end{omfigure}

\begin{proposition}[R\`egles utiles]\label{prop:g3:mult:rules}
\begin{enumerate}
  \item L'ordre n'a pas d'importance~: $a \times b = b \times a$.
  \item Multiplier par $1$ ne change rien~; multiplier par $0$ donne
        $0$.
  \item Multiplier par $10$ ajoute un \omterm{def:g1:counting:zero}{z\'ero}~: $34 \times 10 = 340$~;
        par $100$, deux \omterm{def:g1:counting:zero}{z\'eros}.
  \item Une table se reconstruit \`a partir de ses voisins~:
        $6 \times 8 = 5 \times 8 + 8 = 40 + 8 = 48$.
\end{enumerate}
\end{proposition}
\admitted

\section{Multiplication en colonnes}

\begin{method}[Multiplier par un nombre \`a un chiffre]\label{met:g3:mult:column}
Pour calculer $234 \times 6$~:
\begin{enumerate}
  \item multiplier les unit\'es~: $4 \times 6 = 24$~: \'ecrire $4$,
        retenir $2$~;
  \item multiplier les dizaines et ajouter la retenue~:
        $3 \times 6 + 2 = 20$~: \'ecrire $0$, retenir $2$~;
  \item multiplier les centaines et ajouter la retenue~:
        $2 \times 6 + 2 = 14$~: \'ecrire $14$.
\end{enumerate}
\[
\begin{array}{r}
2\,3\,4 \\
\times \quad 6 \\
\hline
1\,4\,0\,4
\end{array}
\]
Estimer pour v\'erifier~: $234$ est proche de $200$, et
$200 \times 6 = 1\,200$~: la r\'eponse $1\,404$ est plausible.
\end{method}

\begin{example}[D\'ecouper pour multiplier de t\^ete]\label{ex:g3:mult:split}
La technique en colonnes d\'ecoupe secr\`etement le nombre par places.
De t\^ete, faire pareil~:
\[
23 \times 4 = (20 + 3) \times 4 = 20 \times 4 + 3 \times 4
= 80 + 12 = 92 .
\]
Et avec un voisin ami~: $99 \times 5 = 100 \times 5 - 5 = 495$.
\end{example}

\end{omfigure}

\section{La multiplication dans les probl\`emes}

\begin{example}\label{ex:g3:mult:problem}
Un minibus transporte $9$ passagers. Combien de passagers peuvent
transporter $27$ minibus~?
\begin{enumerate}
  \item C'est une multiplication~: $27$ groupes de $9$.
  \item $27 \times 9 = 243$ (en colonnes, ou
        $27 \times 9 = 27 \times 10 - 27 = 270 - 27$).
  \item Phrase~: les minibus peuvent transporter $243$ passagers.
\end{enumerate}
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g3:mult:1}
\'Ecris sous forme de multiplication, puis calcule~:
$8 + 8 + 8 + 8$~; \quad $5 + 5 + 5$~; \quad
$20 + 20 + 20 + 20 + 20$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:2}
R\'ecite et compl\`ete~: $6 \times 7$~; \ $8 \times 4$~; \
$9 \times 9$~; \ $7 \times 8$~; \ $6 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:3}
Dessine un rectangle de points pour $4 \times 6$, puis un pour
$6 \times 4$. Que remarques-tu sur les deux comptages~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:4}
Calcule en utilisant les r\`egles de \cref{prop:g3:mult:rules}~:
$56 \times 10$~; \quad $8 \times 100$~; \quad $73 \times 0$~; \quad
$45 \times 1$~; \quad $30 \times 10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:5}
Calcule en colonnes~: $132 \times 3$~; \quad $217 \times 4$~; \quad
$408 \times 7$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:6}
Calcule de t\^ete en d\'ecoupant (comme dans
\cref{ex:g3:mult:split})~: $31 \times 5$~; \quad $42 \times 3$~;
\quad $25 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:7}
Calcule avec un voisin ami~: $99 \times 7$~; \quad
$101 \times 6$~; \quad $98 \times 5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:8}
Estime d'abord, puis calcule~: $389 \times 5$ (estime avec $400$)~;
$612 \times 8$ (estime avec $600$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:9}
Une bo\^ite d'{\oe}ufs contient $6$ {\oe}ufs. Combien d'{\oe}ufs y
a-t-il dans $38$ bo\^ites~? \'Ecris l'op\'eration, calcule, r\'eponds
par une phrase.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:mult:10}
Une classe a $8$ rang\'ees de $4$ bureaux~; chaque bureau accueille
$2$ \'el\`eves. Combien d'\'el\`eves la salle peut-elle accueillir~?
(Deux multiplications.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:mult:11}
Trouve toutes les fa\c{c}ons d'arranger $24$ chaises en lignes
\'egales (lignes de $1$, de $2$, \dots). Quels arrangements sont
possibles~? (Utilise le carr\'e des tables~: cherche $24$ \`a
l'int\'erieur.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:mult:12}
Zo\'e calcule $47 \times 6$ ainsi~: $40 \times 6 = 240$,
$7 \times 6 = 42$, puis $240 + 42 = 282$. L\'ea calcule
$47 \times 6 = 50 \times 6 - 3 \times 6 = 300 - 18 = 282$. Explique
chaque m\'ethode. Laquelle pr\'ef\`eres-tu pour $38 \times 4$~?
Calcule des deux fa\c{c}ons.
\end{exercise}
