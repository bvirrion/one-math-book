\chapter{Figures et angles droits}\label{ch:g3:shapes}

Carr\'es, rectangles, triangles, cercles~: les figures sont de
vieilles amies. Ce chapitre les regarde avec l'{\oe}il du
g\'eom\`etre --- qu'est-ce qui fait exactement d'un carr\'e un
carr\'e~? --- et pr\'esente l'outil qui d\'ecide~: l'\'equerre et
son \omterm{def:g3:shapes:rightangle}{angle droit}.

\section{L'angle droit}

\begin{definition}[Angle droit]\label{def:g3:shapes:rightangle}
Un \emph{angle droit}\index{angle droit} est l'angle d'un coin de
feuille de papier, ou de l'\'equerre. Deux droites qui forment un
angle droit sont \emph{perpendiculaires}. Dans les figures, un angle
droit se marque par un petit carr\'e dans le coin.
\end{definition}

\begin{method}[V\'erifier un angle droit]\label{met:g3:shapes:check}
\begin{enumerate}
  \item Placer le coin de l'\'equerre exactement sur le coin \`a
        tester~;
  \item glisser un bord de l'\'equerre le long d'un c\^ot\'e de
        l'angle~;
  \item si l'autre c\^ot\'e de l'angle suit exactement l'autre bord,
        l'angle est droit --- s'il s'ouvre plus large ou plus
        \'etroit, ce n'est pas un \omterm{def:g3:shapes:rightangle}{angle droit}.
\end{enumerate}
\end{method}

\section{Les figures et leurs r\`egles}

\begin{definition}[Carr\'e, rectangle, triangle]\label{def:g3:shapes:shapes}
\begin{itemize}
  \item Un \emph{rectangle}\index{rectangle} a $4$ c\^ot\'es et $4$
        \omterm{def:g3:shapes:rightangle}{angles droits}~; ses c\^ot\'es oppos\'es ont la m\^eme
        longueur.
  \item Un \emph{carr\'e}\index{carr\'e} est un rectangle dont les
        $4$ c\^ot\'es ont tous la m\^eme longueur.
  \item Un \emph{triangle}\index{triangle} a $3$ c\^ot\'es~; un
        \emph{triangle rectangle} a un \omterm{def:g3:shapes:rightangle}{angle droit}.
  \item Un \emph{cercle}\index{cercle} se trace au compas~: tous ses
        points sont \`a la m\^eme distance (le \emph{rayon}\index{rayon}) du
        \emph{centre}.
\end{itemize}
\end{definition}

\begin{example}[Pourquoi les marques comptent]\label{ex:g3:shapes:marks}
Une figure un peu pench\'ee peut \emph{ressembler} \`a un carr\'e
sans l'\^etre. Les marques disent la v\'erit\'e~: $4$ petits carr\'es
d'\omterm{def:g3:shapes:rightangle}{angle droit} $+$ $4$ traits d'\'egalit\'e $=$ un vrai carr\'e.
Quand on dessine, mettre les marques~; quand on lit une figure, ne
se fier qu'aux marques et aux mesures donn\'ees.
\end{example}

\section{Dessiner sur du papier quadrill\'e}

\begin{method}[Constructions sur un quadrillage]\label{met:g3:shapes:grid}
Le papier quadrill\'e offre des \omterm{def:g3:shapes:rightangle}{angles droits} gratuits~: ses lignes
sont perpendiculaires.
\begin{enumerate}
  \item Pour dessiner un rectangle $5$ sur $3$~: choisir un croisement,
        compter $5$ cases \`a droite, $3$ cases vers le haut, et
        fermer la figure le long des lignes du quadrillage~;
  \item pour v\'erifier des longueurs \'egales, compter les cases~;
  \item les pas en diagonale font des lignes pench\'ees~: aller
        \guillemotleft{} $1$ \`a droite, $1$ en haut \guillemotright{}
        r\'ep\'et\'ement trace une ligne \`a $45^\circ$.
\end{enumerate}
\end{method}

\begin{example}[Un programme de construction]\label{ex:g3:shapes:program}
Une figure peut se d\'ecrire par des instructions~:
\guillemotleft{} 1. Dessiner un carr\'e $ABCD$ de c\^ot\'e $4$~cm.
2. Dessiner le cercle de centre $A$ passant par $B$.
\guillemotright{} En suivant un programme ligne par ligne, n'importe
qui peut redessiner la m\^eme figure --- et en \'ecrire un force \`a
nommer chaque point. Essaie avec un camarade~: l'un \'ecrit, l'autre
dessine.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g3:shapes:1}
Dans la classe, trouve (sans mesurer) trois \omterm{def:g3:shapes:rightangle}{angles droits} et un
angle qui n'est pas droit. Comment peux-tu \emph{v\'erifier} chacun~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:2}
Dessine un rectangle de $6$~cm sur $4$~cm \`a la r\`egle et \`a
l'\'equerre. Marque ses \omterm{def:g3:shapes:rightangle}{angles droits} et ses c\^ot\'es \'egaux.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:3}
Dessine un carr\'e de c\^ot\'e $5$~cm. Combien d'\omterm{def:g3:shapes:rightangle}{angles droits}
a-t-il~? Combien de c\^ot\'es \'egaux~? Marque tout.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:4}
Sur du papier quadrill\'e, dessine~: un carr\'e $4 \times 4$~; un
rectangle $7 \times 2$~; un triangle rectangle de c\^ot\'es $3$ et
$5$ cases.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:5}
Vrai ou faux~? Explique.
\guillemotleft{} Tout carr\'e est un rectangle. \guillemotright{}
\quad \guillemotleft{} Tout rectangle est un carr\'e.
\guillemotright{} \quad \guillemotleft{} Un triangle peut avoir deux
\omterm{def:g3:shapes:rightangle}{angles droits}. \guillemotright{} (Essaie d'en dessiner un~!)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:6}
Dessine un cercle de \omterm{def:g3:shapes:shapes}{rayon} $4$~cm. Marque son centre $O$, un point
$A$ sur le cercle, et trace le \omterm{def:g3:shapes:shapes}{rayon} $[OA]$. Quelle est la longueur
de $OA$~? Et si $B$ est un autre point du cercle, quelle est la
longueur de $OB$~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:7}
Suis ce programme~: \guillemotleft{} 1. Dessine un rectangle $ABCD$
avec $AB = 6$~cm et $BC = 3$~cm. 2. Marque le point $M$ au milieu de
$[AB]$. 3. Relie $M$ \`a $C$ et \`a $D$. \guillemotright{} En
quelles figures le rectangle a-t-il \'et\'e d\'ecoup\'e~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:8}
Classe les figures~: lesquelles ont au moins un \omterm{def:g3:shapes:rightangle}{angle droit}~? Un
carr\'e~; un rectangle (non carr\'e)~; un triangle rectangle~; un
cercle~; la lettre L dessin\'ee avec des traits \'epais.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:shapes:9}
Sur du papier quadrill\'e, dessine un carr\'e pench\'e dont les
c\^ot\'es vont \guillemotleft{} $2$ \`a droite, $1$ en haut
\guillemotright{} (et les tours correspondants). Compte les cases
pour te convaincre que les quatre c\^ot\'es ont la m\^eme longueur.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:shapes:10}
\'Ecris un programme de construction (instructions num\'erot\'ees,
points nomm\'es) pour~: un carr\'e de c\^ot\'e $3$~cm pos\'e sur un
rectangle $3$~cm $\times$ $2$~cm, comme une maison sur son rez-de-
chauss\'ee. Teste ton programme sur un camarade.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:shapes:11}
Combien de carr\'es peux-tu trouver dans une grille $3 \times 3$ de
petits carr\'es~? (Compte les $1 \times 1$, les $2 \times 2$ et les
$3 \times 3$.)
\end{exercise}
