\chapter{Partage et division}\label{ch:g3:division}

Vingt billes partag\'ees \'equitablement entre quatre enfants~:
combien chacun~? La division r\'epond \`a toutes les questions de
partage --- et aux questions de groupement aussi --- et c'est la
jumelle de la multiplication~: l'une d\'efait l'autre.

\section{Deux visages de la division}

\begin{definition}[Division]\label{def:g3:division:def}
La division $20 \div 4$ r\'epond \`a deux sortes de questions~:
\begin{itemize}
  \item \emph{partage}~: $20$ billes partag\'ees entre $4$ enfants ---
        combien de billes \emph{chacun}~? ($5$ chacun)~;
  \item \emph{groupement}~: $20$ billes mises en sacs de $4$ ---
        combien de \emph{sacs}~? ($5$ sacs).
\end{itemize}
Les deux questions ont la m\^eme r\'eponse, car toutes deux d\'efont
la m\^eme multiplication~: $4 \times 5 = 20$.
\end{definition}

\begin{method}[Diviser avec les tables]\label{met:g3:division:tables}
Pour calculer $42 \div 6$, se demander~: \guillemotleft{} $6$ fois
\emph{quoi} fait $42$~? \guillemotright{} La table de $6$ r\'epond~:
$6 \times 7 = 42$, donc $42 \div 6 = 7$. Chaque question de division
est une question de multiplication lue \`a l'envers.
\end{method}

\section{Quand \c{c}a ne tombe pas juste}

\begin{definition}[Reste]\label{def:g3:division:remainder}
Partager $23$ billes entre $4$ enfants donne $5$ billes chacun ---
et $3$ billes \emph{en trop}, pas assez pour un autre tour. On
\'ecrit
\[
23 = 4 \times 5 + 3 ,
\]
et on dit~: le \emph{quotient} est $5$, le
\emph{reste}\index{reste} est $3$. Le reste est toujours plus petit
que le nombre d'enfants --- sinon un autre tour de partage serait
possible.
\end{definition}

\begin{example}\label{ex:g3:division:remainder}
Diviser $58$ par $7$. Parcourir la table de $7$~:
$7 \times 8 = 56$ est le plus grand produit qui tient sous $58$
($7 \times 9 = 63$ est trop grand). Donc
\[
58 = 7 \times 8 + 2 :
\]
quotient $8$, reste $2$. V\'erification~: $56 + 2 = 58$, et
$2 < 7$.
\end{example}

\begin{method}[Interpr\'eter le reste]\label{met:g3:division:interpret}
Apr\`es la division, revenir \`a l'histoire~:
\begin{enumerate}
  \item \guillemotleft{} combien chacun / combien de groupes
        complets~? \guillemotright{} --- le quotient~;
  \item \guillemotleft{} combien en trop~? \guillemotright{} --- le
        reste~;
  \item \guillemotleft{} combien de bo\^ites pour tout contenir~?
        \guillemotright{} --- le quotient \emph{plus un} si le reste
        n'est pas z\'ero.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:division:interpret}
$58$ \'el\`eves vont en cano\"e, $7$ par cano\"e.
$58 = 7 \times 8 + 2$~: huit cano\"es sont pleins, et $2$ \'el\`eves
restent --- ils ont besoin d'un cano\"e aussi. Donc $9$ cano\"es
doivent \^etre r\'eserv\'es.
\end{example}

\section{Moiti\'es, tiers, quarts}

\begin{definition}[Moiti\'e, tiers, quart]\label{def:g3:division:half}
La \emph{moiti\'e} d'un nombre est le r\'esultat de sa division par
$2$~; le \emph{tiers}, par $3$~; le \emph{quart}, par $4$. Par
exemple la moiti\'e de $18$ est $9$, le tiers de $18$ est $6$, le
quart de $36$ est $9$.
\end{definition}

\begin{example}[Doubles et moiti\'es vont ensemble]\label{ex:g3:division:double}
La moiti\'e de $46$~: comme $46 = 40 + 6$, la moiti\'e de chaque
partie donne $20 + 3 = 23$. V\'erifier avec le double~:
$23 + 23 = 46$. Conna\^itre les doubles par c{\oe}ur ($25 \to 50$,
$45 \to 90$, $150 \to 300$) rend les moiti\'es instantan\'ees.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g3:division:1}
Calcule en utilisant les tables \`a l'envers~: $36 \div 4$~; \quad
$63 \div 9$~; \quad $40 \div 5$~; \quad $56 \div 8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:2}
Pour chaque division, \'ecris l'\'egalit\'e $a = b \times q + r$ et
donne le quotient et le reste~: $17 \div 5$~; \quad $30 \div 4$~;
\quad $50 \div 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:3}
$28$ billes sont partag\'ees entre $6$ enfants. Combien de billes
chacun, et combien en trop~? Dessine le partage si cela aide.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:4}
Calcule~: la moiti\'e de $68$~; le tiers de $27$~; le quart de
$48$~; la moiti\'e de $150$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:5}
Vrai ou faux~? V\'erifie avec une multiplication.
\guillemotleft{} $45 \div 5 = 9$ \guillemotright{}~; \quad
\guillemotleft{} $52 \div 6 = 8$ reste $4$ \guillemotright{}~;
\quad \guillemotleft{} $70 \div 8 = 9$ reste $2$
\guillemotright{}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:6}
$32$ photos sont coll\'ees dans un album, $4$ par page. Combien de
pages sont utilis\'ees~? Quel visage de la division est-ce (partage
ou groupement)~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:7}
Une corde de $75$~cm est coup\'ee en morceaux de $9$~cm. Combien de
morceaux complets, et quelle longueur reste-t-il~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:8}
$50$ enfants partent en voyage dans des voitures de $4$ places.
Combien de voitures faut-il~? (Attention~: tout le monde doit
monter~!)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:9}
Trouve les nombres manquants~: $? \div 7 = 6$~; \quad
$54 \div ? = 6$~; \quad la moiti\'e de $?$ est $35$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:division:10}
Amina partage ses $47$ autocollants entre ses $5$ amis, le plus
\'equitablement possible, et garde le reste pour elle. Combien
d'autocollants chaque ami re\c{c}oit-il, et combien Amina en
garde-t-elle~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:division:11}
Quand on divise un nombre par $6$, quels sont tous les restes
possibles~? Quel est le plus grand nombre dont la division par $6$ a
pour quotient $7$~?
\end{exercise}
