\chapter{Mesures}\label{ch:g3:measure}

Combien de long, combien de lourd, combien \c{c}a contient, quelle
heure est-il~? Mesurer transforme le monde en nombres --- \`a
condition que tout le monde utilise les m\^emes unit\'es. Ce chapitre
pr\'esente les unit\'es de longueur, de masse et de contenance, et
apprend \`a lire une horloge.

\section{Longueurs}

\begin{definition}[Unit\'es de longueur]\label{def:g3:measure:length}
L'unit\'e de base de longueur est le \emph{m\`etre} (m). Pour les
petites longueurs~: le centim\`etre (cm) et le millim\`etre (mm)~;
pour les longues distances~: le kilom\`etre (km).
\[
1 \text{ m} = 100 \text{ cm}, \qquad
1 \text{ cm} = 10 \text{ mm}, \qquad
1 \text{ km} = 1\,000 \text{ m}.
\]
\end{definition}

\begin{example}[Choisir la bonne unit\'e]\label{ex:g3:measure:choose}
Une fourmi~: environ $5$~mm. Un crayon~: environ $18$~cm. Une
porte~: environ $2$~m. Une marche jusqu'au village voisin~: environ
$3$~km. Dire \guillemotleft{} la longueur d'un crayon est $18$
\guillemotright{} ne veut rien dire --- l'unit\'e fait partie de la
r\'eponse.
\end{example}

\begin{method}[Convertir des unit\'es]\label{met:g3:measure:convert}
Pour convertir, multiplier ou diviser par $10$, $100$ ou $1\,000$~:
\begin{enumerate}
  \item d'une grande unit\'e vers une petite, \emph{multiplier}~:
        $3$~m $= 3 \times 100 = 300$~cm~;
  \item d'une petite unit\'e vers une grande, \emph{diviser}~:
        $250$~cm $= 250 \div 100 = 2$~m et $50$~cm~;
  \item pour additionner des longueurs, les convertir d'abord \`a la
        \emph{m\^eme} unit\'e~: $1$~m $+ 35$~cm $= 100 + 35 =
        135$~cm.
\end{enumerate}
\end{method}

\section{Masses et contenances}

\begin{definition}[Unit\'es de masse et de contenance]\label{def:g3:measure:mass}
Les masses se mesurent en \emph{grammes} (g) et \emph{kilogrammes}
(kg), les contenances (ce qu'un r\'ecipient peut contenir) en
\emph{litres} (L) et \emph{centilitres} (cL)~:
\[
1 \text{ kg} = 1\,000 \text{ g},
\qquad
1 \text{ L} = 100 \text{ cL}.
\]
\end{definition}

\begin{example}\label{ex:g3:measure:mass}
Une lettre~: environ $20$~g. Un sac de farine~: $1$~kg. Une baignoire~:
environ $150$~L. Un verre~: environ $20$~cL. Pour \'equilibrer une
balance portant un sac de $1$~kg contre $650$~g de pommes, ajouter
$1\,000 - 650 = 350$~g du c\^ot\'e des pommes.
\end{example}

\section{Lire l'horloge}

\begin{method}[Lire l'heure]\label{met:g3:measure:clock}
Sur une horloge \`a aiguilles~:
\begin{enumerate}
  \item la \emph{petite} aiguille donne l'heure~: lire le nombre
        qu'elle a d\'epass\'e en dernier~;
  \item la \emph{grande} aiguille donne les minutes~: chaque grande
        marque vaut $5$ minutes, donc une grande aiguille sur le $3$
        signifie $15$ minutes~;
  \item apr\`es la demie, on peut aussi lire \guillemotleft{} avant
        l'heure suivante \guillemotright{}~: $10{:}50$ c'est
        \guillemotleft{} onze heures moins dix \guillemotright{}.
\end{enumerate}
\end{method}

\begin{example}[Dur\'ees]\label{ex:g3:measure:duration}
Le film commence \`a $14{:}40$ et se termine \`a $16{:}10$. Combien
dure-t-il~? Compter en deux sauts~:
\begin{enumerate}
  \item de $14{:}40$ \`a $15{:}00$~: $20$ minutes~;
  \item de $15{:}00$ \`a $16{:}10$~: $1$ heure $10$ minutes.
\end{enumerate}
Total~: $1$~h $30$~min. Attention~: une heure a $60$ minutes, donc
$16{:}10$ moins $14{:}40$ ne se calcule pas comme $1610 - 1440$~!
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g3:measure:1}
Quelle unit\'e utiliserais-tu (mm, cm, m ou km) pour~: la largeur
d'un clou~; la taille d'un camarade~; la longueur de la cour~; la
distance entre deux villes~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:2}
Convertis~: $5$~m en cm~; \quad $70$~mm en cm~; \quad $4$~km en m~;
\quad $600$~cm en m.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:3}
Mesure avec ta r\`egle (au millim\`etre pr\`es) les c\^ot\'es d'un
triangle dessin\'e par un camarade, et calcule la longueur totale de
son contour.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:4}
Calcule, en convertissant d'abord~: $2$~m $+ 45$~cm (en cm)~; \quad
$1$~km $+ 250$~m (en m)~; \quad $3$~cm $+ 8$~mm (en mm).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:5}
Convertis~: $3$~kg en g~; \quad $2\,500$~g en kg et g~; \quad
$4$~L en cL~; \quad $350$~cL en L et cL.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:6}
Une recette demande $750$~g de farine. Mona a un sac de $1$~kg. En
a-t-elle assez~? Combien en restera-t-il~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:7}
Lis les heures~: la petite aiguille est entre $2$ et $3$, la grande
sur le $6$~; la petite aiguille sur le $9$, la grande sur le $12$~;
la petite aiguille presque sur le $5$, la grande sur le $10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:8}
L'\'ecole commence \`a $8{:}30$ et la matin\'ee se termine \`a
$11{:}45$. Combien dure la matin\'ee~? (Compte par sauts, comme dans
\cref{ex:g3:measure:duration}.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:9}
Le train part \`a $9{:}50$ et le trajet dure $45$ minutes. \`A
quelle heure arrive-t-il~? (Saute d'abord jusqu'\`a $10{:}00$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:measure:10}
Une bouteille contient $1$~L. Paul remplit des verres de $20$~cL.
Combien de verres pleins peut-il verser~? (Convertis d'abord.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:measure:11}
L\'ena marche $600$~m pour aller \`a l'\'ecole, et autant pour
revenir, chaque jour. Combien de m\`etres marche-t-elle en une
semaine scolaire de $5$~jours~? Donne la r\'eponse en m\`etres, puis
en kilom\`etres.
\end{exercise}
