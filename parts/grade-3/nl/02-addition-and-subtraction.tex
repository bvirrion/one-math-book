\chapter{Optellen en aftrekken}\label{ch:g3:addsub}

Optellen betekent bij elkaar doen; aftrekken betekent wegnemen, of
vinden wat ontbreekt. Dit hoofdstuk oefent de twee kolommethodes ---
met hun beroemde onthouden en lenen --- en de mentale trucs die
kolommen vaak overbodig maken.

\section{Kolomsgewijs optellen}

\begin{method}[Optellen in kolommen]\label{met:g3:addsub:add}
\begin{enumerate}
  \item Schrijf de getallen onder elkaar, eenheden onder eenheden,
        tientallen onder tientallen (lijn uit naar \emph{rechts});
  \item tel de eenheden op; als de som over $9$ gaat, schrijf het
        eenhedencijfer en \emph{onthoud} het tiental naar de
        volgende kolom (een kleine $1$);
  \item ga door met de tientallen, de honderdtallen, \dots, en tel
        telkens het onthouden mee.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:addsub:add}
Reken uit $457 + 368$:
\[
\begin{array}{r}
{\scriptstyle 1}\ \ {\scriptstyle 1}\ \ \phantom{0} \\[-3pt]
4\,5\,7 \\
+\ 3\,6\,8 \\
\hline
8\,2\,5
\end{array}
\]
Eenheden: $7 + 8 = 15$: schrijf $5$, onthoud $1$. Tientallen:
$5 + 6 + 1 = 12$: schrijf $2$, onthoud $1$. Honderdtallen:
$4 + 3 + 1 = 8$. Resultaat: $825$.
\end{example}

\begin{example}[Uit het hoofd optellen]\label{ex:g3:addsub:mental}
Zoek vriendelijke getallen v\'{o}\'{o}r je naar kolommen grijpt:
\begin{itemize}
  \item \emph{maak een tiental}: $47 + 8 = 47 + 3 + 5 = 50 + 5 = 55$;
  \item \emph{spring in delen}: $256 + 130 = 256 + 100 + 30 = 386$;
  \item \emph{herschik}: $17 + 59 + 3 = (17 + 3) + 59 = 79$.
\end{itemize}
\end{example}

\section{Kolomsgewijs aftrekken}

\begin{method}[Aftrekken in kolommen]\label{met:g3:addsub:sub}
\begin{enumerate}
  \item Schrijf het grotere getal boven, rechts uitgelijnd;
  \item trek kolom voor kolom af, beginnend bij de eenheden;
  \item als het bovenste cijfer te klein is, \emph{leen}: neem \'{e}\'{e}n
        van de volgende plaats op de bovenste rij (het is tien waard
        in de huidige kolom);
  \item \emph{controle}: het resultaat plus het afgetrokken getal
        moet het bovenste getal teruggeven.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:addsub:sub}
Reken uit $603 - 147$:
\[
\begin{array}{r}
6\,0\,3 \\
-\ 1\,4\,7 \\
\hline
4\,5\,6
\end{array}
\]
Eenheden: $3 - 7$ onmogelijk; leen door: $603$ wordt $5$
honderdtallen, $9$ tientallen, $13$ eenheden. Dan $13 - 7 = 6$;
tientallen $9 - 4 = 5$; honderdtallen $5 - 1 = 4$. Resultaat: $456$.
Controle: $456 + 147 = 603$.
\end{example}

\begin{example}[Aftrekken als ontbrekend deel]\label{ex:g3:addsub:missing}
``Lila heeft $35$ knikkers, Tom heeft er $52$: hoeveel meer heeft
Tom?'' is de aftrekking $52 - 35$. Uit het hoofd, tel \emph{vooruit}
vanaf $35$: $35 \to 40$ is $5$, dan $40 \to 52$ is $12$; in totaal
$5 + 12 = 17$. Vooruit tellen is vaak makkelijker dan lenen.
\end{example}

\end{omfigure}

\section{Ordes van grootte}

\begin{method}[Schatten v\'{o}\'{o}r het rekenen]\label{met:g3:addsub:estimate}
V\'{o}\'{o}r elke kolomberekening, rond de getallen af op vriendelijke
getallen en reken een \emph{schatting}. Na het rekenen, vergelijk:
een resultaat ver van de schatting wijst op een fout.
\end{method}

\begin{example}
Voor $487 + 312$: de schatting is $500 + 300 = 800$. Het
kolomresultaat, $799$, is dichtbij: aannemelijk. Als je $1\,099$ had
gevonden (een gemiste uitlijning), zou de schatting je gewaarschuwd
hebben.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g3:addsub:1}
Reken kolomsgewijs uit: $346 + 253$; \quad $478 + 265$; \quad
$1\,536 + 2\,876$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:2}
Reken kolomsgewijs uit, controleer dan met een optelling:
$587 - 234$; \quad $805 - 348$; \quad $1\,000 - 463$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:3}
Reken uit het hoofd door tientallen te maken: $38 + 7$; \quad
$56 + 9$; \quad $45 + 28$ (tel $30$ op, neem $2$ weg).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:4}
Reken uit het hoofd met sprongen: $340 + 220$; \quad $675 + 210$;
\quad $512 - 300$; \quad $840 - 220$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:5}
Reken $63 - 47$ uit door vooruit te tellen (zoals in
\cref{ex:g3:addsub:missing}), controleer dan in kolommen.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:6}
Schat eerst (rond af op honderdtallen), reken dan precies:
$492 + 218$; \quad $913 - 388$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:7}
Een school heeft $348$ meisjes en $296$ jongens. Hoeveel leerlingen
in totaal? Schrijf de bewerking, reken uit, en antwoord met een zin.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:8}
Een boek heeft $224$ bladzijden; Nora heeft er $178$ gelezen.
Hoeveel bladzijden zijn er nog te lezen?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:addsub:9}
Vul de gaten (werk kolom voor kolom):
\[
\begin{array}{r}
3\,?\,7 \\
+\ ?\,2\,? \\
\hline
8\,6\,2
\end{array}
\]
Is er maar \'{e}\'{e}n manier om de drie gaten te vullen? Leg uit.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:addsub:10}
Een bakkerij verkocht $145$ croissants 's ochtends en $89$ 's
middags; $17$ croissants bleven onverkocht bij sluitingstijd. Hoeveel
croissants werden die dag gebakken? (Twee stappen.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:addsub:11}
Theo zegt: ``om $500 - 199$ uit te rekenen, reken ik $500 - 200$ en
tel er dan $1$ bij.'' Heeft hij gelijk? Reken op beide manieren, en
leg zijn truc uit.
\end{exercise}
