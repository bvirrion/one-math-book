\chapter{Verdelen en delen}\label{ch:g3:division}

Twintig knikkers eerlijk verdelen onder vier kinderen: hoeveel elk?
Delen beantwoordt alle verdeelvragen --- en de groepeer-vragen ook
--- en het is de tweeling van vermenigvuldigen: de ene maakt de
andere ongedaan.

\section{Twee gezichten van delen}

\begin{definition}[Deling]\label{def:g3:division:def}
De deling $20 \div 4$ beantwoordt twee soorten vragen:
\begin{itemize}
  \item \emph{verdelen}: $20$ knikkers verdeeld onder $4$ kinderen
        --- hoeveel knikkers \emph{elk}? ($5$ elk);
  \item \emph{groeperen}: $20$ knikkers in zakjes van $4$ ---
        hoeveel \emph{zakjes}? ($5$ zakjes).
\end{itemize}
Beide vragen hebben hetzelfde antwoord, omdat beide dezelfde
vermenigvuldiging ongedaan maken: $4 \times 5 = 20$.
\end{definition}

\begin{method}[Delen met de tafels]\label{met:g3:division:tables}
Om $42 \div 6$ uit te rekenen, vraag: ``$6$ keer \emph{wat} maakt
$42$?'' De tafel van $6$ antwoordt: $6 \times 7 = 42$, dus
$42 \div 6 = 7$. Elke deelvraag is een vermenigvuldigingsvraag
achterstevoren gelezen.
\end{method}

\section{Als het niet precies uitkomt}

\begin{definition}[Rest]\label{def:g3:division:remainder}
$23$ knikkers verdelen onder $4$ kinderen geeft $5$ knikkers elk ---
en $3$ knikkers \emph{over}, te weinig voor nog een ronde. We
schrijven
\[
23 = 4 \times 5 + 3 ,
\]
en zeggen: het \emph{quoti\"ent}\index{quoti\"ent} is $5$, de
\emph{rest}\index{rest} is $3$. De rest is altijd kleiner dan het
aantal kinderen --- anders zou nog een ronde verdelen mogelijk zijn.
\end{definition}

\begin{example}\label{ex:g3:division:remainder}
Deel $58$ door $7$. Scan de tafel van $7$:
$7 \times 8 = 56$ is het grootste \omterm{def:g2:mult:def}{product} dat onder $58$ past
($7 \times 9 = 63$ is te veel). Dus
\[
58 = 7 \times 8 + 2 :
\]
\omterm{def:g3:division:remainder}{quoti\"ent} $8$, \omterm{def:g3:division:remainder}{rest} $2$. Controle: $56 + 2 = 58$, en $2 < 7$.
\end{example}

\begin{method}[De rest interpreteren]\label{met:g3:division:interpret}
Na het delen, ga terug naar het verhaal:
\begin{enumerate}
  \item ``hoeveel elk / hoeveel volle groepen?'' --- het
        \omterm{def:g3:division:remainder}{quoti\"ent};
  \item ``hoeveel over?'' --- de \omterm{def:g3:division:remainder}{rest};
  \item ``hoeveel dozen om alles te houden?'' --- het \omterm{def:g3:division:remainder}{quoti\"ent}
        \emph{plus \'{e}\'{e}n} als de \omterm{def:g3:division:remainder}{rest} niet \omterm{def:g1:counting:zero}{nul} is.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:division:interpret}
$58$ leerlingen gaan kano\"{e}n, $7$ per kano. $58 = 7 \times 8 + 2$:
acht kano's zijn vol, en $2$ leerlingen blijven over --- zij hebben
ook een kano nodig. Dus moeten er $9$ kano's geboekt worden.
\end{example}

\section{Helften, derden, kwarten}

\begin{definition}[Helft, derde, kwart]\label{def:g3:division:half}
De \emph{helft}\index{helft} van een getal is het resultaat van delen door $2$;
het \emph{derde}, door $3$; het \emph{kwart}\index{kwart}, door $4$. Bijvoorbeeld
is de helft van $18$ gelijk aan $9$, het derde van $18$ is $6$, het
kwart van $36$ is $9$.
\end{definition}

\begin{example}[Verdubbelen en halveren horen bij elkaar]\label{ex:g3:division:double}
De \omterm{def:g3:division:half}{helft} van $46$: omdat $46 = 40 + 6$, geeft de \omterm{def:g3:division:half}{helft} van elk deel
$20 + 3 = 23$. Controleer met de verdubbeling: $23 + 23 = 46$.
Verdubbelingen uit het hoofd ($25 \to 50$, $45 \to 90$,
$150 \to 300$) maken \omterm{def:g3:division:half}{helften} direct.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g3:division:1}
Reken uit met de tafels achterstevoren: $36 \div 4$; \quad
$63 \div 9$; \quad $40 \div 5$; \quad $56 \div 8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:2}
Schrijf voor elke deling de gelijkheid $a = b \times q + r$ en geef
het \omterm{def:g3:division:remainder}{quoti\"ent} en de \omterm{def:g3:division:remainder}{rest}: $17 \div 5$; \quad $30 \div 4$; \quad
$50 \div 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:3}
$28$ knikkers worden verdeeld onder $6$ kinderen. Hoeveel knikkers
elk, en hoeveel over? Teken het verdelen als dat helpt.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:4}
Reken uit: de \omterm{def:g3:division:half}{helft} van $68$; het derde van $27$; het \omterm{def:g3:division:half}{kwart} van
$48$; de \omterm{def:g3:division:half}{helft} van $150$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:5}
Waar of niet waar? Controleer met een vermenigvuldiging.
``$45 \div 5 = 9$''; \quad ``$52 \div 6 = 8$ \omterm{def:g3:division:remainder}{rest} $4$'';
\quad ``$70 \div 8 = 9$ \omterm{def:g3:division:remainder}{rest} $2$''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:6}
$32$ foto's worden in een album geplakt, $4$ per blad. Hoeveel
bladen worden gebruikt? Welk gezicht van delen is dit (verdelen of
groeperen)?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:7}
Een touw van $75$ cm wordt in stukken van $9$ cm geknipt. Hoeveel
volle stukken, en hoe lang is het reststuk?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:8}
$50$ kinderen gaan op reis in auto's van $4$ zitplaatsen. Hoeveel
auto's zijn nodig? (Let op: iedereen moet meerijden!)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:division:9}
Vind de ontbrekende getallen: $? \div 7 = 6$; \quad
$54 \div ? = 6$; \quad de \omterm{def:g3:division:half}{helft} van $?$ is $35$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:division:10}
Amina verdeelt haar $47$ stickers onder haar $5$ vrienden, zo eerlijk
mogelijk, en houdt de \omterm{def:g3:division:remainder}{rest} zelf. Hoeveel stickers krijgt elke
vriend, en hoeveel houdt Amina?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:division:11}
Als een getal door $6$ wordt gedeeld, wat zijn alle mogelijke
\omterm{def:g3:division:remainder}{resten}? Wat is het grootste getal waarvan de deling door $6$
\omterm{def:g3:division:remainder}{quoti\"ent} $7$ heeft?
\end{exercise}
