\chapter{Getallen tot 10\,000}\label{ch:g3:numbers}

Duizend, tweeduizend, drieduizend\,\dots\ grote getallen worden
gebouwd uit kleine, met een prachtig idee: de \emph{plaats} van een
cijfer vertelt zijn waarde. Dit hoofdstuk leert lezen, schrijven,
vergelijken en ordenen van getallen tot $10\,000$.

\section{Getallen lezen en schrijven}

\begin{definition}[Plaatswaarde]\label{def:g3:numbers:place}
Een getal wordt geschreven met de cijfers $0, 1, 2, \dots, 9$. Van
rechts gelezen zijn de plaatsen: eenheden, tientallen, honderdtallen,
duizendtallen. In $3\,254$:
\[
3\,254 = 3 \text{ duizendtallen} + 2 \text{ honderdtallen} + 5 \text{ tientallen}
+ 4 \text{ eenheden}.
\]
\end{definition}

\begin{example}[De nul die de plaats houdt]\label{ex:g3:numbers:zero}
``Vierduizend zeven'' wordt geschreven $4\,007$: een $4$ bij de
duizendtallen, een $7$ bij de eenheden, en \emph{nullen} ertussen om
elk cijfer op de juiste plaats te houden. Zonder hen zou $47$ een
heel ander getal zijn!
\end{example}

\begin{example}[Omwisselen]\label{ex:g3:numbers:exchange}
Tien eenheden maken \'{e}\'{e}n tiental; tien tientallen maken \'{e}\'{e}n
honderdtal; tien honderdtallen maken \'{e}\'{e}n duizendtal. Dus $10$
honderdtallen $= 1\,000$, en $32$ tientallen $= 320$: het getal $320$
bevat $32$ tientallen (niet alleen het cijfer $2$!).
\end{example}

\section{Vergelijken en ordenen}

\begin{method}[Twee getallen vergelijken]\label{met:g3:numbers:compare}
\begin{enumerate}
  \item Tel de cijfers: meer cijfers betekent een groter getal
        ($1\,002 > 998$).
  \item Zelfde aantal cijfers: vergelijk cijfer voor cijfer vanaf
        links. Het eerste \omterm{ex:g1:subtraction:difference}{verschil} beslist:
        $5\,614 > 5\,589$ omdat $6 > 5$ bij de honderdtallen.
  \item Schrijf het antwoord met de tekens $<$ (kleiner dan) of $>$
        (groter dan): de open kant kijkt altijd naar het grotere
        getal.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:numbers:order}
Zet op volgorde van klein naar groot: $780$, $8\,700$, $807$, $87$.
Eerst op aantal cijfers: $87$ (twee), dan $780$ en $807$ (drie), dan
$8\,700$ (vier). Tussen $780$ en $807$: vergelijk de
honderdtallen, $7 < 8$, dus $780 < 807$. Eindvolgorde:
\[
87 < 780 < 807 < 8\,700 .
\]
\end{example}

\section{Even en oneven}

\begin{definition}[Even, oneven]\label{def:g3:numbers:evenodd}
Een getal is \emph{even}\index{even getal} als het in twee gelijke
hele delen gesplitst kan worden --- zijn eenhedencijfer is $0$, $2$,
$4$, $6$ of $8$. Anders is het \emph{oneven}\index{oneven getal}
(eenhedencijfer $1$, $3$, $5$, $7$ of $9$).
\end{definition}

\begin{example}
$46$ is \omterm{def:g3:numbers:evenodd}{even} ($46 = 23 + 23$); $47$ is \omterm{def:g3:numbers:evenodd}{oneven} (twee gelijke delen
zouden een half voorwerp nodig hebben). Alleen het \emph{laatste}
cijfer telt: $3\,578$ is \omterm{def:g3:numbers:evenodd}{even}, $2\,341$ is \omterm{def:g3:numbers:evenodd}{oneven}.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g3:numbers:1}
Schrijf in cijfers: ``tweeduizend driehonderdvijfenveertig'';
``zesduizend vijftig''; ``negenduizend \'{e}\'{e}n''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:2}
Schrijf in woorden: $1\,208$; \quad $4\,070$; \quad $9\,999$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:3}
In $7\,382$: welk cijfer staat bij de tientallen? Bij de
duizendtallen? Hoeveel \emph{honderdtallen} bevat het getal in
totaal (let op: niet alleen het honderdtallencijfer --- denk aan
\cref{ex:g3:numbers:exchange})?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:4}
Ontbind zoals in \cref{def:g3:numbers:place}:
$5\,631$; \quad $2\,046$; \quad $8\,500$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:5}
Schrijf over en vul in met $<$ of $>$:
\[
456 \;?\; 465, \qquad
1\,203 \;?\; 989, \qquad
6\,090 \;?\; 6\,009, \qquad
9\,999 \;?\; 10\,000 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:6}
Zet op volgorde van klein naar groot: $530$; \ $3\,500$; \ $353$; \
$5\,033$; \ $503$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:7}
Op een getallenlijn in honderdtallen van $0$ tot $1\,000$, plaats
(bij benadering) de getallen $150$, $480$, $820$ en $990$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:8}
\omterm{def:g3:numbers:evenodd}{Even} of \omterm{def:g3:numbers:evenodd}{oneven}? \ $34$; \ $57$; \ $70$; \ $2\,481$; \ $6\,548$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:numbers:9}
Tel met stappen: vanaf $2\,970$, vijf stappen van $10$; vanaf $860$,
vier stappen van $100$; vanaf $9\,996$, vier stappen van $1$. Wat
val je op aan het einde van de laatste?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:numbers:10}
Met elk van de cijfers $3$, $7$, $0$, $5$ precies \'{e}\'{e}n keer,
schrijf het grootst mogelijke viercijferige getal, dan het kleinste
(een getal mag niet met $0$ beginnen).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:numbers:11}
Ik ben een driedijferig getal. Mijn cijfers zijn $2$, $5$ en $8$, ik
ben \omterm{def:g3:numbers:evenodd}{even}, en ik ben groter dan $800$. Wie ben ik?
\end{exercise}
