\chapter{Vermenigvuldigen}\label{ch:g3:mult}

Drie zakjes van zeven knikkers: je zou $7 + 7 + 7$ kunnen optellen,
maar er is een snellere manier --- vermenigvuldigen, de bewerking van
herhaald optellen. Dit hoofdstuk bouwt de tafels uit plaatjes op, en
leert daarna de kolommethode.

\section{Wat vermenigvuldigen betekent}

\begin{definition}[Vermenigvuldiging]\label{def:g3:mult:def}
Het product $3 \times 7$ betekent ``drie keer zeven'':
\[
3 \times 7 = 7 + 7 + 7 = 21 .
\]
Op een plaatje is $3 \times 7$ een rechthoek van stippen: $3$ rijen
van $7$ stippen.
\end{definition}

  \end{scope}
\end{tikzpicture}

{\small Dezelfde $21$ stippen, geteld per rij of per kolom: de
volgorde van een product doet er niet toe. $3 \times 7$ kennen geeft
$7 \times 3$ gratis --- de helft van de tafels te leren!}
\end{omfigure}

\begin{proposition}[Regels die helpen]\label{prop:g3:mult:rules}
\begin{enumerate}
  \item De volgorde doet er niet toe: $a \times b = b \times a$.
  \item Vermenigvuldigen met $1$ verandert niets; vermenigvuldigen
        met $0$ geeft $0$.
  \item Vermenigvuldigen met $10$ voegt een nul toe:
        $34 \times 10 = 340$; met $100$, twee nullen.
  \item Een tafel kan herbouwd worden uit zijn buren:
        $6 \times 8 = 5 \times 8 + 8 = 40 + 8 = 48$.
\end{enumerate}
\end{proposition}
\admitted

\section{Kolomsgewijs vermenigvuldigen}

\begin{method}[Vermenigvuldigen met een eencijferig getal]\label{met:g3:mult:column}
Om $234 \times 6$ uit te rekenen:
\begin{enumerate}
  \item vermenigvuldig de eenheden: $4 \times 6 = 24$: schrijf $4$,
        onthoud $2$;
  \item vermenigvuldig de tientallen en tel het onthouden op:
        $3 \times 6 + 2 = 20$: schrijf $0$, onthoud $2$;
  \item vermenigvuldig de honderdtallen en tel het onthouden op:
        $2 \times 6 + 2 = 14$: schrijf $14$.
\end{enumerate}
\[
\begin{array}{r}
2\,3\,4 \\
\times \quad 6 \\
\hline
1\,4\,0\,4
\end{array}
\]
Schat om te controleren: $234$ ligt dicht bij $200$, en
$200 \times 6 = 1\,200$: het antwoord $1\,404$ is aannemelijk.
\end{method}

\begin{example}[Splitsen om uit het hoofd te vermenigvuldigen]\label{ex:g3:mult:split}
De kolommethode splitst stiekem het getal naar plaatsen. Uit het
hoofd doe je hetzelfde:
\[
23 \times 4 = (20 + 3) \times 4 = 20 \times 4 + 3 \times 4
= 80 + 12 = 92 .
\]
En met een vriendelijke buur: $99 \times 5 = 100 \times 5 - 5 =
495$.
\end{example}

\end{omfigure}

\section{Vermenigvuldigen in sommen}

\begin{example}\label{ex:g3:mult:problem}
Een minibus vervoert $9$ passagiers. Hoeveel passagiers kunnen $27$
minibussen vervoeren?
\begin{enumerate}
  \item Het is een vermenigvuldiging: $27$ groepen van $9$.
  \item $27 \times 9 = 243$ (kolommen, of
        $27 \times 9 = 27 \times 10 - 27 = 270 - 27$).
  \item Zin: de minibussen kunnen $243$ passagiers vervoeren.
\end{enumerate}
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g3:mult:1}
Schrijf als vermenigvuldiging, reken dan uit:
$8 + 8 + 8 + 8$; \quad $5 + 5 + 5$; \quad
$20 + 20 + 20 + 20 + 20$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:2}
Zeg op en vul in: $6 \times 7$; \ $8 \times 4$; \ $9 \times 9$;
\ $7 \times 8$; \ $6 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:3}
Teken een stipperechthoek voor $4 \times 6$, dan \'{e}\'{e}n voor
$6 \times 4$. Wat val je op aan de twee tellingen?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:4}
Reken uit met de regels van \cref{prop:g3:mult:rules}:
$56 \times 10$; \quad $8 \times 100$; \quad $73 \times 0$; \quad
$45 \times 1$; \quad $30 \times 10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:5}
Reken kolomsgewijs uit: $132 \times 3$; \quad $217 \times 4$; \quad
$408 \times 7$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:6}
Reken uit het hoofd door te splitsen (zoals in
\cref{ex:g3:mult:split}): $31 \times 5$; \quad $42 \times 3$;
\quad $25 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:7}
Reken uit met een vriendelijke buur: $99 \times 7$; \quad
$101 \times 6$; \quad $98 \times 5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:8}
Schat eerst, reken dan uit: $389 \times 5$ (schat met $400$);
$612 \times 8$ (schat met $600$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:9}
Een eierdoos bevat $6$ eieren. Hoeveel eieren zitten er in $38$
dozen? Schrijf de bewerking, reken uit, antwoord met een zin.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:mult:10}
Een klaslokaal heeft $8$ rijen van $4$ bureaus; elk bureau biedt
plaats aan $2$ leerlingen. Hoeveel leerlingen kan het lokaal
plaatsen? (Twee vermenigvuldigingen.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:mult:11}
Zoek alle manieren om $24$ stoelen in gelijke rijen te zetten (rijen
van $1$, van $2$, \dots). Welke opstellingen zijn mogelijk? (Gebruik
het tafelsquare: zoek $24$ erin.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:mult:12}
Zoe rekent $47 \times 6$ zo: $40 \times 6 = 240$,
$7 \times 6 = 42$, en dan $240 + 42 = 282$. L\'{e}a rekent
$47 \times 6 = 50 \times 6 - 3 \times 6 = 300 - 18 = 282$. Leg elke
methode uit. Welke vind jij beter voor $38 \times 4$? Reken het op
beide manieren uit.
\end{exercise}
