\chapter{Problemen oplossen}\label{ch:g3:problems}

De vier bewerkingen kennen is \'{e}\'{e}n ding; weten \emph{welke} een
verhaal nodig heeft, is iets anders. Dit hoofdstuk geeft een methode
voor woordproblemen --- steeds dezelfde vier stappen --- en leert
informatie lezen uit tabellen en pictogrammen.

\section{De vier stappen}

\begin{method}[Een woordprobleem oplossen]\label{met:g3:problems:steps}
\begin{enumerate}
  \item \emph{Lees} het probleem twee keer; zeg in je eigen woorden
        wat gevraagd wordt, en wat bekend is.
  \item \emph{Kies} de bewerking: teken een snelle schets of een
        staafjesplaatje als de keuze niet duidelijk is.
  \item \emph{Reken}, in kolommen of uit het hoofd, met een
        schatting als vangnet.
  \item \emph{Antwoord} met een volledige zin, inclusief de eenheid
        --- en controleer of het antwoord zinvol is (een kind weegt
        geen $300$ kg).
\end{enumerate}
\end{method}

\begin{example}[Welke bewerking?]\label{ex:g3:problems:which}
De woorden wijzen naar de bewerking, de schets beslist:
\begin{itemize}
  \item \emph{bij elkaar / in totaal}: optelling;
  \item \emph{wegnemen / hoeveel meer / wat ontbreekt}:
        aftrekking;
  \item \emph{zoveel groepen van zoveel / zoveel keer}:
        vermenigvuldiging;
  \item \emph{eerlijk verdelen / hoeveel groepen}: deling.
\end{itemize}
\end{example}

\begin{example}[Een probleem in twee stappen]\label{ex:g3:problems:twostep}
``Maya koopt $3$ pakken van $6$ flessen en drinkt $2$ flessen.
Hoeveel flessen blijven er over?''
\begin{enumerate}
  \item Wat gevraagd: het aantal overgebleven flessen. Bekend: $3$
        pakken van $6$; $2$ gedronken.
  \item Schets: $3$ groepen van $6$, dan $2$ weg. Twee bewerkingen!
  \item Reken: $3 \times 6 = 18$, dan $18 - 2 = 16$.
  \item Zin: er zijn $16$ flessen over.
\end{enumerate}
Stop nooit na de eerste bewerking: lees de vraag opnieuw om te zien
of het verhaal klaar is.
\end{example}

\begin{example}[Een nutteloos getal]\label{ex:g3:problems:trap}
``Een bakker, die $52$ jaar oud is, bakt $120$ broodjes en verkoopt
er $85$. Hoeveel broodjes blijven er over?'' De leeftijd is er om je
af te leiden: het antwoord heeft alleen $120 - 85 = 35$ broodjes
nodig. Sorteer v\'{o}\'{o}r het rekenen altijd de getallen: welke gebruikt
de vraag echt?
\end{example}

\section{Tabellen en pictogrammen lezen}

\begin{example}[Een tabel]\label{ex:g3:problems:table}
Uitgeleende boeken in de bibliotheek:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cccc}
 & strips & romans & wetenschap & totaal \\
\hline
week 1 & $14$ & $9$ & $6$ & $29$ \\
week 2 & $11$ & $12$ & $8$ & $31$ \\
\end{tabular}
\end{center}
Lezen: week 2, romans: $12$. Rekenen uit de tabel: in de twee weken
samen werden $14 + 11 = 25$ strips uitgeleend; de drukste week was
week 2 ($31 > 29$).
\end{example}

\begin{example}[Een pictogram]\label{ex:g3:problems:pictogram}
In een pictogram staat \'{e}\'{e}n symbool voor een vast bedrag --- lees
eerst de legenda!

{\small Een pictogram: maandag toont $3$ stippen, dus
$3 \times 4 = 12$ cakes; dinsdag $5$ stippen, $20$ cakes; woensdag
$3$ stippen, $12$ cakes.}
\end{omfigure}
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g3:problems:1}
Noem voor elk verhaal alleen de \emph{bewerking} (reken niet uit):
een kudde van $48$ schapen voegt zich bij een kudde van $37$; \
$56$ kersen verdeeld onder $7$ kinderen; \ $9$ dozen van $8$
potloden; \ Ali had $71$ kaarten en gaf er $18$ weg.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:2}
Los op met de vier stappen: ``Een parkeerplaats heeft $246$ auto's
's ochtends; er komen er $158$ bij om twaalf uur. Hoeveel auto's
zijn er nu?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:3}
Los op: ``Een touw van $90$ m wordt in stukken van $9$ m geknipt.
Hoeveel stukken?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:4}
Los op: ``Kaartjes kosten elk $8$. Wat kosten $26$ kaartjes?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:5}
Teken een staafjesplaatje, los dan op: ``Emma heeft $64$ stickers,
Hugo heeft er $29$ minder. Hoeveel heeft Hugo?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:6}
Twee stappen: ``Een krat bevat $45$ appels. Een groenteboer ontvangt
$6$ kratten en verkoopt $178$ appels. Hoeveel appels blijven er
over?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:7}
Met de tabel van \cref{ex:g3:problems:table}: hoeveel
wetenschapsboeken werden in de twee weken samen uitgeleend? Hoeveel
meer romans in week 2 dan in week 1?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:8}
Met het pictogram van \cref{ex:g3:problems:pictogram}: hoeveel
cakes werden in de drie dagen samen verkocht? Op welke dag werden de
meeste cakes verkocht?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:problems:9}
Dit probleem heeft een nutteloos getal --- zoek het, los dan op:
``Een boek van $210$ bladzijden kost $12$. Leo leest $35$
bladzijden per dag. Hoeveel dagen heeft hij nodig om het boek uit te
lezen?''
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:problems:10}
``Een gezin van $2$ volwassenen en $3$ kinderen bezoekt het museum.
Volwassenkaartjes kosten $9$, kindkaartjes $4$. Ze betalen met een
biljet van $50$. Hoeveel wisselgeld krijgen ze?'' (Drie stappen:
volwassenen, kinderen, dan het wisselgeld.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:problems:11}
Bedenk je eigen probleem in twee stappen waarvan het antwoord $100$
is, schrijf de oplossing met de vier stappen, en test het bij een
klasgenoot.
\end{exercise}
