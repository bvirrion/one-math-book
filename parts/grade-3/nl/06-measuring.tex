\chapter{Meten}\label{ch:g3:measure}

Hoe lang, hoe zwaar, hoeveel past erin, hoe laat is het? Meten
verandert de wereld in getallen --- mits iedereen dezelfde eenheden
gebruikt. Dit hoofdstuk presenteert de eenheden van lengte, \omterm{def:g2:measure:units}{massa} en
\omterm{def:g2:measure:units}{inhoud}, en leert een klok te lezen.

\section{Lengtes}

\begin{definition}[Lengte-eenheden]\label{def:g3:measure:length}
De basiseenheid van lengte is de \emph{meter} (m). Voor kleine
lengtes: de centimeter (cm) en de millimeter (mm); voor lange
afstanden: de kilometer (km).
\[
1 \text{ m} = 100 \text{ cm}, \qquad
1 \text{ cm} = 10 \text{ mm}, \qquad
1 \text{ km} = 1\,000 \text{ m}.
\]
\end{definition}

\begin{example}[De juiste eenheid kiezen]\label{ex:g3:measure:choose}
Een mier: ongeveer $5$ mm. Een potlood: ongeveer $18$ cm. Een deur:
ongeveer $2$ m. Een wandeling naar het volgende dorp: ongeveer
$3$ km. Zeggen ``de lengte van een potlood is $18$'' betekent niets
--- de eenheid is deel van het antwoord.
\end{example}

\begin{method}[Eenheden omzetten]\label{met:g3:measure:convert}
Om om te zetten, vermenigvuldig of deel door $10$, $100$ of $1\,000$:
\begin{enumerate}
  \item van een grote eenheid naar een kleine, \emph{vermenigvuldig}:
        $3$ m $= 3 \times 100 = 300$ cm;
  \item van een kleine eenheid naar een grote, \emph{deel}:
        $250$ cm $= 250 \div 100 = 2$ m en $50$ cm;
  \item om lengtes op te tellen, zet ze eerst om naar dezelfde
        eenheid: $1$ m $+ 35$ cm $= 100 + 35 = 135$ cm.
\end{enumerate}
\end{method}

\section{Massa's en inhouden}

\begin{definition}[Eenheden van massa en inhoud]\label{def:g3:measure:mass}
\omterm{def:g2:measure:units}{Massa}'s worden gemeten in \emph{gram} (g) en \emph{kilogram} (kg),
\omterm{def:g2:measure:units}{inhouden} (hoeveel een vat houdt) in \emph{liter} (L) en
\emph{centiliter} (cL):
\[
1 \text{ kg} = 1\,000 \text{ g},
\qquad
1 \text{ L} = 100 \text{ cL}.
\]
\end{definition}

\begin{example}\label{ex:g3:measure:mass}
Een brief: ongeveer $20$ g. Een zak meel: $1$ kg. Een badkuip:
ongeveer $150$ L. Een glas: ongeveer $20$ cL. Om een weegschaal in
evenwicht te brengen met een zak van $1$ kg tegen $650$ g appels,
voeg $1\,000 - 650 = 350$ g toe aan de kant van de appels.
\end{example}

\section{De klok lezen}

\begin{method}[De tijd lezen]\label{met:g3:measure:clock}
Op een klok met wijzers:
\begin{enumerate}
  \item de \emph{korte} wijzer geeft het uur: lees het getal dat hij
        als laatste is gepasseerd;
  \item de \emph{lange} wijzer geeft de minuten: elke grote streep
        is $5$ minuten, dus een lange wijzer op de $3$ betekent
        $15$ minuten;
  \item na half kun je ook lezen ``v\'{o}\'{o}r het volgende uur'':
        $10{:}50$ is ``tien voor elf''.
\end{enumerate}
\end{method}

\begin{example}[Duur]\label{ex:g3:measure:duration}
De film begint om $14{:}40$ en eindigt om $16{:}10$. Hoe lang duurt
hij? Tel in twee sprongen:
\begin{enumerate}
  \item van $14{:}40$ naar $15{:}00$: $20$ minuten;
  \item van $15{:}00$ naar $16{:}10$: $1$ uur $10$ minuten.
\end{enumerate}
Totaal: $1$ u $30$ min. Let op: een uur heeft $60$ minuten, dus
$16{:}10$ min $14{:}40$ kun je niet rekenen als $1610 - 1440$!
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g3:measure:1}
Welke eenheid zou je gebruiken (mm, cm, m of km) voor: de breedte
van een spijker; de lengte van een klasgenoot; de lengte van de
speelplaats; de afstand tussen twee steden?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:2}
Zet om: $5$ m in cm; \quad $70$ mm in cm; \quad $4$ km in m;
\quad $600$ cm in m.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:3}
Meet met je liniaal (tot op de dichtstbijzijnde millimeter) de
zijden van een driehoek getekend door een klasgenoot, en reken de
totale lengte van de omtrek uit.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:4}
Reken uit, zet eerst om: $2$ m $+ 45$ cm (in cm); \quad
$1$ km $+ 250$ m (in m); \quad $3$ cm $+ 8$ mm (in mm).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:5}
Zet om: $3$ kg in g; \quad $2\,500$ g in kg en g; \quad
$4$ L in cL; \quad $350$ cL in L en cL.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:6}
Een recept heeft $750$ g meel nodig. Mona heeft een zak van $1$ kg.
Heeft ze genoeg? Hoeveel blijft er over?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:7}
Lees de tijden: de korte wijzer zit tussen $2$ en $3$, de lange
wijzer op de $6$; de korte wijzer op de $9$, de lange wijzer op de
$12$; de korte wijzer bijna op de $5$, de lange wijzer op de $10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:8}
School begint om $8{:}30$ en de ochtend eindigt om $11{:}45$. Hoe
lang duurt de ochtend? (Tel in sprongen, zoals in
\cref{ex:g3:measure:duration}.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:measure:9}
De trein vertrekt om $9{:}50$ en de rit duurt $45$ minuten. Hoe laat
komt hij aan? (Spring eerst naar $10{:}00$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:measure:10}
Een fles houdt $1$ L. Paul vult glazen van $20$ cL. Hoeveel volle
glazen kan hij inschenken? (Zet eerst om.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:measure:11}
Lena loopt $600$ m naar school, en terug, elke dag. Hoeveel meter
loopt ze in een schoolweek van $5$ dagen? Geef het antwoord in
meters, dan in kilometers.
\end{exercise}
