\chapter{Vormen en rechte hoeken}\label{ch:g3:shapes}

Vierkanten, rechthoeken, driehoeken, cirkels: de vormen zijn oude
vrienden. Dit hoofdstuk bekijkt ze met een meetkundig oog --- wat
maakt een vierkant precies tot een vierkant? --- en introduceert het
hulpmiddel dat beslist: de geodriehoek en zijn \omterm{def:g3:shapes:rightangle}{rechte hoek}.

\section{De rechte hoek}

\begin{definition}[Rechte hoek]\label{def:g3:shapes:rightangle}
Een \emph{rechte hoek}\index{rechte hoek} is de hoek van een
vel papier, of van de geodriehoek. Twee lijnen die een rechte hoek
maken, zijn \emph{loodrecht}. In figuren wordt een rechte hoek
gemarkeerd met een klein vierkantje in de hoek.
\end{definition}

\begin{method}[Een rechte hoek controleren]\label{met:g3:shapes:check}
\begin{enumerate}
  \item Leg de hoek van de geodriehoek precies op de te testen hoek;
  \item schuif \'{e}\'{e}n zijde van de geodriehoek langs \'{e}\'{e}n zijde
        van de hoek;
  \item als de andere zijde van de hoek precies langs de andere rand
        loopt, is de hoek recht --- als hij breder of smaller
        opent, is hij het niet.
\end{enumerate}
\end{method}

\section{De vormen en hun regels}

\begin{definition}[Vierkant, rechthoek, driehoek]\label{def:g3:shapes:shapes}
\begin{itemize}
  \item Een \emph{rechthoek}\index{rechthoek} heeft $4$ zijden en $4$
        \omterm{def:g3:shapes:rightangle}{rechte hoeken}; zijn overstaande zijden hebben dezelfde
        lengte.
  \item Een \emph{vierkant}\index{vierkant} is een rechthoek waarvan
        de $4$ zijden allemaal dezelfde lengte hebben.
  \item Een \emph{driehoek}\index{driehoek} heeft $3$ zijden; een
        \emph{rechthoekige driehoek} heeft \'{e}\'{e}n \omterm{def:g3:shapes:rightangle}{rechte hoek}.
  \item Een \emph{cirkel}\index{cirkel} wordt getekend met de passer:
        al zijn punten liggen op dezelfde afstand (de
        \emph{straal}\index{straal}) van het \emph{middelpunt}.
\end{itemize}
\end{definition}

\begin{example}[Waarom markeringen ertoe doen]\label{ex:g3:shapes:marks}
Een iets scheef getekende vorm kan eruitzien als een vierkant zonder
het te zijn. De markeringen vertellen de waarheid: $4$
recht-hoek-vierkantjes $+$ $4$ streepjes $=$ een echt vierkant. Als
je tekent, zet de markeringen; als je een figuur leest, vertrouw
alleen de markeringen en de gegeven maten.
\end{example}

\section{Tekenen op ruitjespapier}

\begin{method}[Constructies op een raster]\label{met:g3:shapes:grid}
Ruitjespapier biedt gratis \omterm{def:g3:shapes:rightangle}{rechte hoeken}: zijn lijnen zijn loodrecht.
\begin{enumerate}
  \item Om een rechthoek $5$ bij $3$ te tekenen: kies een
        snijpunt, tel $5$ vakjes naar rechts, $3$ omhoog, en sluit
        de vorm langs de rasterlijnen;
  \item om gelijke lengtes te controleren, tel vakjes;
  \item diagonale stappen maken scheve lijnen: herhaald ``$1$
        rechts, $1$ omhoog'' tekent een lijn van $45^\circ$.
\end{enumerate}
\end{method}

\begin{example}[Een constructieprogramma]\label{ex:g3:shapes:program}
Een figuur kan met instructies beschreven worden:
``1. Teken een vierkant $ABCD$ van zijde $4$ cm. 2. Teken de cirkel
met middelpunt $A$ door $B$.'' Door een programma regel voor regel
te volgen, kan iedereen dezelfde figuur natekenen --- en \'{e}\'{e}n
schrijven dwingt je elk punt te benoemen. Probeer het met een
partner: de een schrijft, de ander tekent.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g3:shapes:1}
Zoek in de klas (zonder te meten) drie \omterm{def:g3:shapes:rightangle}{rechte hoeken} en \'{e}\'{e}n
hoek die niet recht is. Hoe kun je elk \emph{controleren}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:2}
Teken een rechthoek van $6$ cm bij $4$ cm met liniaal en
geodriehoek. Markeer de \omterm{def:g3:shapes:rightangle}{rechte hoeken} en de gelijke zijden.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:3}
Teken een vierkant van zijde $5$ cm. Hoeveel \omterm{def:g3:shapes:rightangle}{rechte hoeken} heeft
het? Hoeveel gelijke zijden? Markeer alles.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:4}
Teken op ruitjespapier: een $4 \times 4$-vierkant; een
$7 \times 2$-rechthoek; een rechthoekige driehoek met benen van $3$
en $5$ vakjes.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:5}
Waar of niet waar? Leg uit.
``Elk vierkant is een rechthoek.'' \quad
``Elke rechthoek is een vierkant.'' \quad
``Een driehoek kan twee \omterm{def:g3:shapes:rightangle}{rechte hoeken} hebben.'' (Probeer er \'{e}\'{e}n
te tekenen!)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:6}
Teken een cirkel van \omterm{def:g3:shapes:shapes}{straal} $4$ cm. Markeer het middelpunt $O$, een
punt $A$ op de cirkel, en teken de \omterm{def:g3:shapes:shapes}{straal} $[OA]$. Hoe lang is $OA$?
En als $B$ een ander punt van de cirkel is, hoe lang is $OB$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:7}
Volg dit programma: ``1. Teken een rechthoek $ABCD$ met $AB = 6$ cm
en $BC = 3$ cm. 2. Markeer het punt $M$ in het midden van $[AB]$.
3. Verbind $M$ met $C$ en met $D$.'' In welke vormen is de
rechthoek gesneden?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:shapes:8}
Sorteer de vormen: welke hebben minstens \'{e}\'{e}n \omterm{def:g3:shapes:rightangle}{rechte hoek}?
Een vierkant; een (niet-vierkante) rechthoek; een rechthoekige
driehoek; een cirkel; de letter L getekend met dikke streken.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:shapes:9}
Teken op ruitjespapier een scheef vierkant waarvan de zijden ``$2$
rechts, $1$ omhoog'' gaan (en de bijpassende bochten). Tel vakjes om
jezelf te overtuigen dat de vier zijden dezelfde lengte hebben.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:shapes:10}
Schrijf een constructieprogramma (genummerde instructies, benoemde
punten) voor: een vierkant van zijde $3$ cm bovenop een rechthoek
$3$ cm $\times$ $2$ cm, als een huis op zijn begane grond. Test je
programma bij een klasgenoot.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:shapes:11}
Hoeveel vierkanten kun je vinden in een $3 \times 3$-raster van
kleine vierkantjes? (Tel de $1 \times 1$, de $2 \times 2$ en de
$3 \times 3$.)
\end{exercise}
