\chapter{Multiplication}\label{ch:g3:mult}

Three bags of seven marbles: you could add $7 + 7 + 7$, but there is
a faster way --- multiplication, the operation of repeated addition.
This chapter builds the multiplication tables from pictures, then
teaches the column method.

\section{What multiplying means}

\begin{definition}[Multiplication]\label{def:g3:mult:def}
The product $3 \times 7$ means ``three times seven'':
\[
3 \times 7 = 7 + 7 + 7 = 21 .
\]
On a picture, $3 \times 7$ is a rectangle of dots: $3$ rows of $7$
dots.
\end{definition}

{\small The same $21$ dots, counted by rows or by columns: the order
of a product does not matter. Knowing $3 \times 7$ gives

\end{omfigure}

\begin{proposition}[Rules that help]\label{prop:g3:mult:rules}
\begin{enumerate}
  \item The order does not matter: $a \times b = b \times a$.
  \item Multiplying by $1$ changes nothing; multiplying by $0$ gives
        $0$.
  \item Multiplying by $10$ appends a zero: $34 \times 10 = 340$;
        by $100$, two zeros.
  \item A table can be rebuilt from its neighbors:
        $6 \times 8 = 5 \times 8 + 8 = 40 + 8 = 48$.
\end{enumerate}
\end{proposition}
\admitted

\begin{example}[The table square]\label{ex:g3:mult:table}

column $7$.

\section{Column multiplication}

\begin{method}[Multiplying by a one-digit number]\label{met:g3:mult:column}
To compute $234 \times 6$:
\begin{enumerate}
  \item multiply the units: $4 \times 6 = 24$: write $4$, carry $2$;
  \item multiply the tens and add the carry:
        $3 \times 6 + 2 = 20$: write $0$, carry $2$;
  \item multiply the hundreds and add the carry:
        $2 \times 6 + 2 = 14$: write $14$.
\end{enumerate}
\[
\begin{array}{r}
2\,3\,4 \\
\times \quad 6 \\
\hline
1\,4\,0\,4
\end{array}
\]
Estimate to check: $234$ is close to $200$, and
$200 \times 6 = 1\,200$: the answer $1\,404$ is plausible.
\end{method}

\begin{example}[Splitting to multiply mentally]\label{ex:g3:mult:split}
The column method secretly splits the number by places. Mentally, do
the same:
\[
23 \times 4 = (20 + 3) \times 4 = 20 \times 4 + 3 \times 4
= 80 + 12 = 92 .
\]
And with a friendly neighbor: $99 \times 5 = 100 \times 5 - 5 =
495$.
\end{example}

\section{Multiplication in problems}

\begin{example}\label{ex:g3:mult:problem}
A minibus carries $9$ passengers. How many passengers can $27$
minibuses carry?
\begin{enumerate}
  \item It is a multiplication: $27$ groups of $9$.
  \item $27 \times 9 = 243$ (columns, or
        $27 \times 9 = 27 \times 10 - 27 = 270 - 27$).
  \item Sentence: the minibuses can carry $243$ passengers.
\end{enumerate}
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g3:mult:1}
Write as a multiplication, then compute:
$8 + 8 + 8 + 8$; \quad $5 + 5 + 5$; \quad
$20 + 20 + 20 + 20 + 20$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:2}
Recite and complete: $6 \times 7$; \ $8 \times 4$; \ $9 \times 9$;
\ $7 \times 8$; \ $6 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:3}
Draw a dot rectangle for $4 \times 6$, then one for $6 \times 4$.
What do you notice about the two counts?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:4}
Compute using the rules of \cref{prop:g3:mult:rules}:
$56 \times 10$; \quad $8 \times 100$; \quad $73 \times 0$; \quad
$45 \times 1$; \quad $30 \times 10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:5}
Compute in columns: $132 \times 3$; \quad $217 \times 4$; \quad
$408 \times 7$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:6}
Compute mentally by splitting (as in \cref{ex:g3:mult:split}):
$31 \times 5$; \quad $42 \times 3$; \quad $25 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:7}
Compute with a friendly neighbor: $99 \times 7$; \quad
$101 \times 6$; \quad $98 \times 5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:8}
Estimate first, then compute: $389 \times 5$ (estimate with $400$);
$612 \times 8$ (estimate with $600$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g3:mult:9}
An egg box holds $6$ eggs. How many eggs are in $38$ boxes? Write
the operation, compute, answer with a sentence.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:mult:10}
A classroom has $8$ rows of $4$ desks; each desk seats $2$ students.
How many students can the room seat? (Two multiplications.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:mult:11}
Find all the ways of arranging $24$ chairs into equal rows (rows of
$1$, of $2$, \dots). Which arrangements are possible? (Use the table
square: look for $24$ inside it.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g3:mult:12}
Zoe computes $47 \times 6$ like this: $40 \times 6 = 240$,
$7 \times 6 = 42$, and then $240 + 42 = 282$. Léa computes
$47 \times 6 = 50 \times 6 - 3 \times 6 = 300 - 18 = 282$. Explain
each method. Which do you prefer for $38 \times 4$? Compute it both
ways.
\end{exercise}
