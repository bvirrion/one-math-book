\chapter{गुणा}\label{ch:g3:mult}

सात कंचों के तीन बैग: आप \omterm{def:g1:addition:def}{जोड़} सकते हैं $7 + 7 + 7$, लेकिन वहाँ है
एक तेज़ तरीका --- गुणा, बार-बार जोड़ने की क्रिया।
यह अध्याय चित्रों से गुणन सारणी बनाता है
कॉलम विधि सिखाता है.

\section{गुणा करने का क्या मतलब है}

\begin{definition}[गुणा]\label{def:g3:mult:def}
 \omterm{def:g2:mult:def}{उत्पाद} $3 \times 7$ इसका मतलब है ``तीन गुना सात'':
\[
3 \times 7 = 7 + 7 + 7 = 21 .
\]
एक तस्वीर पर, $3 \times 7$ बिंदुओं का एक आयत है: $3$ की पंक्तियाँ $7$
बिंदु.
\end{definition}

$7 \times 3$ निःशुल्क --- सीखने के लिए आधी टेबलें!}
\end{omfigure}

\begin{proposition}[नियम जो मदद करते हैं]\label{prop:g3:mult:rules}
\begin{enumerate}
  \item आदेश मायने नहीं रखता: $ a \times b = b \times a $.
  \item से गुणा करना $1$ कुछ भी नहीं बदलता; से गुणा करना $0$ देता है
        $0$.
  \item से गुणा करना $10$ एक जोड़ता है \omterm{def:g1:counting:zero}{शून्य}: $34 \times 10 = 340$;
        द्वारा $100$, दो \omterm{def:g1:counting:zero}{शून्य}.
  \item एक टेबल को उसके पड़ोसियों से फिर से बनाया जा सकता है:
        $6 \times 8 = 5 \times 8 + 8 = 40 + 8 = 48$.
\end{enumerate}
\end{proposition}
\admitted

\section{स्तम्भ गुणन}

\begin{method}[एक अंकीय संख्या से गुणा करना]\label{met:g3:mult:column}
गणना करना $234 \times 6$:
\begin{enumerate}
  \item इकाइयों को गुणा करें: $4 \times 6 = 24$: लिखना $4$, ढोना $2$;
  \item दहाई को गुणा करें और कैरी \omterm{def:g1:addition:def}{जोड़}ें:
        $3 \times 6 + 2 = 20$: लिखना $0$, ढोना $2$;
  \item सैकड़ों को गुणा करें और कैरी \omterm{def:g1:addition:def}{जोड़}ें:
        $2 \times 6 + 2 = 14$: लिखना $14$.
\end{enumerate}
\[
\begin{array}{r}
2\,3\,4 \\
\times \quad 6 \\
\hline
1\,4\,0\,4
\end{array}
\]
जांचने का अनुमान: $234$ के करीब है $200$, और
$200 \times 6 = 1\,200$: उत्तर $1\,404$ प्रशंसनीय है.
\end{method}

\begin{example}[मानसिक रूप से गुणा करने के लिए विभाजित करना]\label{ex:g3:mult:split}
स्तंभ विधि गुप्त रूप से संख्या को स्थानों से विभाजित करती है। मानसिक रूप से, करो
वही:
\[
23 \times 4 = (20 + 3) \times 4 = 20 \times 4 + 3 \times 4
= 80 + 12 = 92 .
\]
और एक मिलनसार पड़ोसी के साथ: $99 \times 5 = 100 \times 5 - 5 =
495$.
\end{example}

\section{समस्याओं में गुणा}

\begin{example}\label{ex:g3:mult:problem}
एक मिनीबस ले जाती है $9$ यात्रियों. कितने यात्री कर सकते हैं $27$
मिनी बसें ले जाती हैं?
\begin{enumerate}
  \item यह एक गुणन है: $27$ के समूह $9$.
  \item $27 \times 9 = 243$ (कॉलम, या
        $27 \times 9 = 27 \times 10 - 27 = 270 - 27$).
  \item वाक्य: मिनी बसें ले जा सकती हैं $243$ यात्रियों.
\end{enumerate}
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g3:mult:1}
गुणन के रूप में लिखें, फिर गणना करें:
$8 + 8 + 8 + 8$; \quad $5 + 5 + 5$; \quad
$20 + 20 + 20 + 20 + 20$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:2}
पढ़ें और पूरा करें: $6 \times 7$; \ $8 \times 4$; \ $9 \times 9$;
\ $7 \times 8$; \ $6 \times 6$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:3}
के लिए एक बिंदु आयत बनाएं $4 \times 6$, फिर एक के लिए $6 \times 4$.
आप दोनों मामलों के बारे में क्या नोटिस करते हैं?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:4}
के नियमों का उपयोग करके गणना करें \cref{prop:g3:mult:rules}:
$56 \times 10$; \quad $8 \times 100$; \quad $73 \times 0$; \quad
$45 \times 1$; \quad $30 \times 10$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:5}
कॉलम में गणना करें: $132 \times 3$; \quad $217 \times 4$; \quad
$408 \times 7$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:6}
विभाजित करके मानसिक रूप से गणना करें (जैसे कि) \cref{ex:g3:mult:split}):
$31 \times 5$; \quad $42 \times 3$; \quad $25 \times 6$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:7}
मित्रवत पड़ोसी के साथ हिसाब-किताब करें: $99 \times 7$; \quad
$101 \times 6$; \quad $98 \times 5$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:8}
पहले अनुमान लगाएं, फिर गणना करें: $389 \times 5$ (साथ अनुमान लगाएं $400$);
$612 \times 8$ (साथ अनुमान लगाएं $600$).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:mult:9}
एक अंडे का डिब्बा रखा हुआ है $6$ अंडे. कितने अंडे हैं अंदर $38$ बक्से? लिखो
ऑपरेशन, गणना, एक वाक्य के साथ उत्तर दें।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:mult:10}
एक कक्षा है $8$ की पंक्तियाँ $4$ डेस्क; प्रत्येक डेस्क सीटें $2$ छात्र.
कमरे में कितने छात्र बैठ सकते हैं? (दो गुणन)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:mult:11}
व्यवस्था करने के सभी तरीके खोजें $24$ कुर्सियाँ बराबर पंक्तियों में (पंक्तियाँ)
$1$, का $2$, \dots). कौन सी व्यवस्था संभव है? (तालिका का प्रयोग करें
वर्ग: ढूंढो $24$ इसके अंदर.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:mult:12}
ज़ो गणना करता है $47 \times 6$ इस कदर: $40 \times 6 = 240$,
$7 \times 6 = 42$, और तब $240 + 42 = 282$. लीया गणना करती है
$47 \times 6 = 50 \times 6 - 3 \times 6 = 300 - 18 = 282$. समझाओ
प्रत्येक विधि. आप किसके लिए पसंद करते हैं $38 \times 4$? दोनों की गणना करें
तरीके.
\end{exercise}
