\chapter{10 तक की संख्या\,000}\label{ch:g3:numbers}

एक हजार, दो हजार, तीन हजार\,\dots\ बड़ी संख्याएं हैं
एक अद्भुत विचार का उपयोग करके, छोटे-छोटे टुकड़ों से निर्मित: \emph{जगह} एक का
अंक उसका मान बताता है. यह अध्याय पढ़ना, लिखना सिखाता है
तक की संख्याओं की तुलना करें और क्रमबद्ध करें $10\,000$.

\section{संख्याएँ पढ़ना और लिखना}

\begin{definition}[जगह की मूल्य]\label{def:g3:numbers:place}
अंकों के साथ एक संख्या लिखी जाती है $0, 1, 2, \dots, 9$. से पढ़ रहा हूँ
दाईं ओर, स्थान हैं: इकाइयाँ, दहाई, सैकड़ों, हज़ार। में
$3\,254$:
\[
3\,254 = 3 \text{ thousands} + 2 \text{ hundreds} + 5 \text{ tens}
+ 4 \text{ units}.
\]
\end{definition}

  \node[font=\small] at (1.15,1.275) {thousands};

सौ चौवन''.}
\end{omfigure}

\begin{example}[शून्य जो एक स्थान रखता है]\label{ex:g3:numbers:zero}
''चार हजार सात'' लिखा है $4\,007$: ए $4$ में
हजारों, ए $7$ इकाइयों में, और \emph{\omterm{def:g1:counting:zero}{शून्य}} बीच में रखने के लिए
प्रत्येक अंक अपने सही स्थान पर। उनके बिना, $47$ बहुत होगा
भिन्न संख्या!
\end{example}

\begin{example}[का आदान प्रदान]\label{ex:g3:numbers:exchange}
दस इकाइयाँ एक दस बनाती हैं; दस दहाई एक सौ बनाते हैं; दस सैकड़े बनते हैं
एक हजार. इसलिए $10$ सैकड़ों $= 1\,000$, और $32$ दसियों
$= 320$: जो नंबर $320$ रोकना $32$ दहाई (सिर्फ अंक नहीं)
$2$!).
\end{example}

\section{तुलना करना और ऑर्डर करना}

\begin{method}[दो संख्याओं की तुलना करना]\label{met:g3:numbers:compare}
\begin{enumerate}
  \item अंक गिनें: अधिक अंक का अर्थ है बड़ी संख्या
        ($1\,002 > 998$).
  \item अंकों की समान संख्या: प्रारंभ करते हुए अंक दर अंक की तुलना करें
        द \emph{बाएं}. पहला \omterm{ex:g1:subtraction:difference}{अंतर} तय करता है:
        $5\,614 > 5\,589$ क्योंकि $6 > 5$ सैकड़ों पर.
  \item उत्तर चिन्ह सहित लिखिए $<$ (से छोटा) या $>$
        (इससे बड़ा): हमेशा खुला पक्ष \omterm{def:g2:shapes:solids}{चेहरे} बड़ी संख्या.
\end{enumerate}
\end{method}

\begin{tikzpicture}[scale=1.1]
  \draw[-Stealth] (-0.3,0) -- (10.5,0);
  \foreach \i in {0,1,...,10} \draw (\i,-0.08) -- (\i,0.08);
  \node[below, font=\small] at (0,-0.1) {$0$};
  \node[below, font=\small] at (5,-0.1) {$500$};
  \node[below, font=\small] at (10,-0.1) {$1000$};
  \fill[omThm] (2.4,0) circle (2pt) node[above, font=\small] {$240$};
  \fill[omDef] (6.9,0) circle (2pt) node[above, font=\small] {$690$};
  \fill[omMeth] (9.05,0) circle (2pt) node[above, font=\small] {$905$};
\end{tikzpicture}

{\small एक पंक्ति पर संख्याएँ सैकड़ों में विभाजित होती हैं: आगे सही का मतलब है
बड़ा. $240 < 690 < 905$.}
\end{omfigure}

\begin{example}\label{ex:g3:numbers:order}
सबसे छोटे से सबसे बड़े तक ऑर्डर करें: $780$, $8\,700$, $807$, $87$.
सबसे पहले अंकों की संख्या से: $87$ (दो), फिर $780$ और $807$
(तीन), फिर $8\,700$ (चार). बीच में $780$ और $807$: तुलना करें
सैकड़ों अंक, $7 < 8$, इसलिए $780 < 807$. अंतिम आदेश:
\[
87 < 780 < 807 < 8\,700 .
\]
\end{example}

\section{सम और विषम}

\begin{definition}[और भी अजीब]\label{def:g3:numbers:evenodd}
एक संख्या है \emph{यहां तक की}\index{यहां तक की} जब इसे विभाजित किया जा सकता है
दो बराबर पूर्ण भाग---इसका इकाई अंक है $0$, $2$, $4$, $6$ या
$8$. अन्यथा यह है \emph{विषम}\index{विषम} (इकाई अंक $1$,
$3$, $5$, $7$ या $9$).
\end{definition}

\begin{example}
$46$ है \omterm{def:g3:numbers:evenodd}{यहां तक की} ($46 = 23 + 23$); $47$ है \omterm{def:g3:numbers:evenodd}{विषम} (दो बराबर भाग होंगे
आधी वस्तु चाहिए)। केवल \emph{अंतिम} अंक मायने रखता है: $3\,578$ है
\omterm{def:g3:numbers:evenodd}{यहां तक की}, $2\,341$ है \omterm{def:g3:numbers:evenodd}{विषम}.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g3:numbers:1}
अंकों में लिखें: ``दो हजार तीन सौ पैंतालीस''; ``छह
हजार पचास''; ``नौ हजार एक''।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:2}
शब्दों में लिखो: $1\,208$; \quad $4\,070$; \quad $9\,999$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:3}
में $7\,382$: दहाई के स्थान पर कौन सा अंक है? हजारों की संख्या में
जगह? कितने \emph{सैकड़ों} क्या संख्या में कुल योग शामिल है?
(सावधान: केवल सैकड़ों अंक के बारे में नहीं --- सोचें
\cref{ex:g3:numbers:exchange})?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:4}
के रूप में विघटित करें \cref{def:g3:numbers:place}:
$5\,631$; \quad $2\,046$; \quad $8\,500$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:5}
कॉपी करें और पूरा करें $<$ या $>$:
\[
456 \;?\; 465, \qquad
1\,203 \;?\; 989, \qquad
6\,090 \;?\; 6\,009, \qquad
9\,999 \;?\; 10\,000 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:6}
सबसे छोटे से सबसे बड़े तक ऑर्डर करें: $530$; \ $3\,500$; \ $353$; \ $5\,033$;
\ $503$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:7}
एक संख्या रेखा पर सैकड़ों की संख्या में स्नातक किया गया $0$ को $1\,000$, जगह
(लगभग) संख्याएँ $150$, $480$, $820$ और $990$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:8}
\omterm{def:g3:numbers:evenodd}{यहां तक की} या \omterm{def:g3:numbers:evenodd}{विषम}? \ $34$; \ $57$; \ $70$; \ $2\,481$; \ $6\,548$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:numbers:9}
चरणों से गिनें: से $2\,970$, के पाँच चरण गिनें $10$; से $860$,
के चार चरण गिनें $100$; से $9\,996$, के चार चरण गिनें $1$.
पिछले भाग के अंत में आपने क्या देखा?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:numbers:10}
प्रत्येक अंक का उपयोग करना $3$, $7$, $0$, $5$ बिल्कुल एक बार, लिखें
सबसे बड़ी संभावित चार अंकों की संख्या, फिर सबसे छोटी (एक संख्या)।
से प्रारंभ नहीं किया जा सकता $0$).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:numbers:11}
मैं तीन अंकों की संख्या हूं. मेरे अंक हैं $2$, $5$ और $8$, मैं हूँ \omterm{def:g3:numbers:evenodd}{यहां तक की},
और मैं उससे भी बड़ा हूँ $800$. मैं कौन हूँ?
\end{exercise}
