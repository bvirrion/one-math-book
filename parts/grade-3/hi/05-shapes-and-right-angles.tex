\chapter{आकृतियाँ और समकोण}\label{ch:g3:shapes}

वर्ग, आयत, त्रिभुज, वृत्त: आकृतियाँ पुरानी मित्र हैं।
यह अध्याय उन्हें जियोमीटर की नजर से देखता है --- वास्तव में क्या
एक वर्ग को एक वर्ग बनाता है? --- और उस उपकरण का परिचय देता है जो निर्णय लेता है:
सेट स्क्वायर और उसका \omterm{def:g3:shapes:rightangle}{समकोण}.

\section{समकोण}

\begin{definition}[समकोण]\label{def:g3:shapes:rightangle}
ए \emph{समकोण}\index{समकोण} का कोना कोण है
कागज की शीट, या सेट स्क्वायर की शीट। समकोण बनाती दो रेखाएँ
हैं \emph{सीधा}. आकृतियों में, समकोण को a से चिह्नित किया गया है
कोने में छोटा वर्ग.
\end{definition}

\begin{method}[समकोण की जाँच करना]\label{met:g3:shapes:check}
\begin{enumerate}
  \item सेट स्क्वायर के कोने को बिल्कुल कोने पर रखें
        परीक्षण;
  \item सेट स्क्वायर के एक किनारे को कोण के एक तरफ स्लाइड करें;
  \item यदि कोण का दूसरा किनारा बिल्कुल दूसरे के साथ चलता है
        किनारा, कोण \omterm{def:g3:shapes:rightangle}{समकोण} है --- यदि यह चौड़ा या संकरा खुलता है,
        यह नहीं है.
\end{enumerate}
\end{method}

\section{आकृतियाँ और उनके नियम}

\begin{definition}[वर्ग, आयत, त्रिकोण]\label{def:g3:shapes:shapes}
\begin{itemize}
  \item ए \emph{आयत}\index{आयत} है $4$ पक्ष और $4$
        सही कोण; इसकी विपरीत भुजाओं की लंबाई समान होती है।
  \item ए \emph{वर्ग}\index{वर्ग} एक आयत है जिसका $4$ पक्ष
        सभी की लंबाई समान है.
  \item ए \emph{त्रिकोण}\index{त्रिकोण} है $3$ भुजाएँ; ए
        \emph{सही त्रिकोण} एक है \omterm{def:g3:shapes:rightangle}{समकोण}.
  \item ए \emph{घेरा}\index{घेरा} कम्पास के साथ खींचा गया है: सभी
        इसके बिंदु समान दूरी पर हैं \emph{RADIUS}\index{RADIUS}) से \emph{केंद्र}.
\end{itemize}
\end{definition}

\begin{example}[अंक क्यों मायने रखते हैं?]\label{ex:g3:shapes:marks}
थोड़ा झुका हुआ एक आकृति बनाई जा सकती है \emph{देखना} बिना एक वर्ग की तरह
एक होना. ये निशान सच बताते हैं: $4$ \omterm{def:g3:shapes:rightangle}{समकोण} वर्ग $+$ $4$
टिक चिह्न $=$ एक वास्तविक वर्ग. जब तुम चित्र बनाओ तो चिह्न लगाओ; जब आप
एक आंकड़ा पढ़ें, केवल अंकों और दिए गए मापों पर भरोसा करें।
\end{example}

\section{ग्रिड पेपर पर आरेखण}

\begin{method}[ग्रिड पर निर्माण]\label{met:g3:shapes:grid}
ग्रिड पेपर मुफ़्त ऑफ़र करता है सही कोण: इसकी रेखाएं लंबवत होती हैं।
\begin{enumerate}
  \item एक आयत बनाने के लिए $5$ द्वारा $3$: एक क्रॉसिंग पॉइंट चुनें,
        गिनती $5$ दाएँ वर्ग, $3$ चौकोर बनाएं, और आकृति को बंद करें
        ग्रिड लाइनों के साथ;
  \item समान लंबाई जाँचने के लिए, वर्ग गिनें;
  \item विकर्ण चरण झुकी हुई रेखाएँ बनाते हैं: जा रहे हैं ``$1$ सही, $1$ ऊपर''
        बार-बार खींचता है $45^\circ $ रेखा।
\end{enumerate}
\end{method}

\end{omfigure}

\begin{example}[एक निर्माण कार्यक्रम]\label{ex:g3:shapes:program}
किसी चित्र को निर्देशों द्वारा वर्णित किया जा सकता है:
``1. एक वर्ग बनाएं $ ABCD $ इस पक्ष की $4$ सेमी। 2. वृत्त खींचिए
केंद्र $ A $ के माध्यम से गुजरते हुए $ B $.'' एक कार्यक्रम की पंक्ति दर पंक्ति अनुसरण करते हुए,
कोई भी एक ही आकृति को दोबारा बना सकता है --- और एक लिखना आपको मजबूर करता है
प्रत्येक बिंदु को नाम दें. इसे किसी साथी के साथ आज़माएँ: एक लिखता है, दूसरा
खींचता है.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g3:shapes:1}
कक्षा में, (बिना मापे) तीन खोजें सही कोण और
एक कोण जो सही नहीं है. तुम कैसे \emph{जाँच करना} हर एक को?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:shapes:2}
का एक आयत बनाएं $6$ सेमी द्वारा $4$ रूलर और सेट स्क्वायर के साथ सेमी. निशान
इसका सही कोण और इसकी बराबर भुजाएँ।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:shapes:3}
भुजा का एक वर्ग बनाएं $5$ सेमी। कितने सही कोण क्या इसमें है?
कितनी बराबर भुजाएँ? हर चीज़ को चिह्नित करें.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:shapes:4}
ग्रिड पेपर पर, ड्रा करें: a $4 \times 4$ वर्ग; ए $7 \times 2$
आयत; के पैरों वाला एक \omterm{def:g3:shapes:rightangle}{समकोण} त्रिभुज $3$ और $5$ वर्ग।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:shapes:5}
सच या झूठ? व्याख्या करना।
``प्रत्येक वर्ग एक आयत है।'' \quad
``प्रत्येक आयत एक वर्ग है।'' \quad
``एक त्रिभुज में दो हो सकते हैं सही कोण.'' (एक चित्र बनाने का प्रयास करें!)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:shapes:6}
का एक वृत्त बनाएं \omterm{def:g3:shapes:shapes}{RADIUS} $4$ सेमी। इसके केंद्र को चिह्नित करें $ O $, एक बिंदु $ A $ पर
वृत्त, और आरेखित करें \omterm{def:g3:shapes:shapes}{RADIUS} $[OA]$. कितना लंबा $ OA $? और यदि $ B $
वृत्त का दूसरा बिंदु है, कितना लंबा है $ OB $?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:shapes:7}
इस प्रोग्राम का अनुसरण करें: ``1. एक आयत बनाएं $ ABCD $ साथ $ AB = 6$ सेमी
और $ BC = 3$ सेमी। 2. बिंदु चिन्हित करें $ M $ के बीच में $[AB]$.
3. शामिल हों $ M $ को $ C $ और को $ D $.''आयत किस आकार का है?
में काटें?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:shapes:8}
आकृतियों को क्रमबद्ध करें: इनमें से किसमें कम से कम एक हो \omterm{def:g3:shapes:rightangle}{समकोण}?
एक वर्ग; एक (गैर-वर्ग) आयत; एक \omterm{def:g3:shapes:rightangle}{समकोण} त्रिभुज; एक चक्र; द
अक्षर L मोटे स्ट्रोक से खींचा गया है।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:shapes:9}
ग्रिड पेपर पर, एक झुका हुआ वर्ग बनाएं जिसकी भुजाएँ `` हों$2$ सही, $1$
ऊपर'' (और मिलान बदल जाता है)। स्वयं को आश्वस्त करने के लिए वर्ग गिनें
चारों भुजाओं की लंबाई समान है।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:shapes:10}
एक निर्माण कार्यक्रम लिखें (क्रमांकित निर्देश, नामित बिंदु)
के लिए: भुजा का एक वर्ग $3$ सेमी एक आयत के शीर्ष पर बैठा है
$3$ सेमी $\times $ $2$ सेमी, भूतल पर एक घर की तरह। अपना परीक्षण करें
एक सहपाठी पर कार्यक्रम.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:shapes:11}
आप एक में कितने वर्ग पा सकते हैं? $3 \times 3$ छोटे का ग्रिड
वर्ग? (गिनें $1 \times 1$, द $2 \times 2$ और यह
$3 \times 3$ वाले.)
\end{exercise}
