\chapter{साझा करना और विभाजन}\label{ch:g3:division}

बीस कंचे चार बच्चों के बीच उचित रूप से बँट गए: प्रत्येक को कितने?
प्रभाग सभी साझाकरण प्रश्नों --- और समूहीकरण का उत्तर देता है
प्रश्न भी --- और यह गुणन का जुड़वां है: एक पूर्ववत करता है
अन्य.

\section{विभाजन के दो चेहरे}

\begin{definition}[विभाजन]\label{def:g3:division:def}
प्रखंड $20 \div 4$ दो प्रकार के प्रश्नों के उत्तर देता है:
\begin{itemize}
  \item \emph{बंटवारे}: $20$ मार्बल्स को आपस में बाँटा गया $4$ बच्चे---
        कितने कंचे \emph{प्रत्येक}? ($5$ प्रत्येक);
  \item \emph{समूहन}: $20$ मार्बल्स की थैलियों में डाल दिया $4$ --- कितने
        \emph{थैलियों}? ($5$ बैग)।
\end{itemize}
दोनों प्रश्नों का उत्तर एक ही है, क्योंकि दोनों एक ही को पूर्ववत करते हैं
गुणा: $4 \times 5 = 20$.
\end{definition}

\begin{tikzpicture}[scale=0.55]
  \foreach \g in {0,1,2,3} {
    \draw[omExo, rounded corners] (\g*4.9-0.65,-0.7) rectangle
      (\g*4.9+3.85,0.7);
    \foreach \c in {0,1,2,3,4} \fill[omDef] (\g*4.9+\c*0.8,0) circle (0.27);
  }
  \node[below, font=\small] at (8.95,-1.15)
    {$20$पत्थर,$4$समान समूह:$5$प्रत्येक समूह में};
\end{tikzpicture}

{\small शेयरिंग $20$ में $4$ समान समूह. द्वारा समूहीकरण $4$ इसके बजाय
देगा $5$ समूह---समान विभाजन।}
\end{omfigure}

\begin{method}[तालिकाओं से बाँटना]\label{met:g3:division:tables}
गणना करना $42 \div 6$, पूछें: ``$6$ टाइम्स \emph{क्या} बनाता है $42$?''
की मेज $6$ उत्तर: $6 \times 7 = 42$, इसलिए $42 \div 6 = 7$.
प्रत्येक विभाजन प्रश्न एक गुणन प्रश्न है जिसे पढ़ा जाता है
पीछे की ओर.
\end{method}

\section{जब वह निकलता ही नहीं}

\begin{definition}[शेष]\label{def:g3:division:remainder}
शेयरिंग $23$ बीच में पत्थर $4$ बच्चे देते हैं $5$ प्रत्येक पत्थर ---
और $3$ पत्थर \emph{बचा खुचा}, दूसरे दौर के लिए पर्याप्त नहीं है। हम
लिखो
\[
23 = 4 \times 5 + 3 ,
\]
और कहो: \emph{भागफल}\index{भागफल} है $5$, द
\emph{शेष}\index{शेष} है $3$. शेष सदैव है
बच्चों की संख्या से कम --- अन्यथा एक और दौर
साझा करना संभव होगा.
\end{definition}

\begin{tikzpicture}[scale=0.55]
  \foreach \g in {0,1,2,3} {
    \draw[omExo, rounded corners] (\g*4.9-0.65,-0.7) rectangle
      (\g*4.9+3.85,0.7);
    \foreach \c in {0,1,2,3,4} \fill[omDef] (\g*4.9+\c*0.8,0) circle (0.27);
  }
  \foreach \c in {0,1,2} \fill[omThm] (19.7+\c*0.8,0) circle (0.27);
  \node[above, font=\footnotesize, omThm] at (20.5,0.5) {बचा खुचा};
  \node[below, font=\small] at (10.2,-1.15)
    {$23 = 4 \times 5 + 3$: भागफल$5$, शेष$3$(लाल)};
\end{tikzpicture}

{\small शेयरिंग $23$ के बीच $4$: बाद $5$ दौर, $3$ पत्थर बचे हैं
---ए के लिए बहुत कम $6$ वां दौर.}
\end{omfigure}

\begin{example}\label{ex:g3:division:remainder}
विभाजित करना $58$ द्वारा $7$. की तालिका स्कैन करें $7$:
$7 \times 8 = 56$ सबसे बड़ा है \omterm{def:g2:mult:def}{उत्पाद} वह नीचे फिट बैठता है $58$
($7 \times 9 = 63$ ज्यादा है)। इसलिए
\[
58 = 7 \times 8 + 2 :
\]
\omterm{def:g3:division:remainder}{भागफल} $8$, \omterm{def:g3:division:remainder}{शेष} $2$. जाँच करना: $56 + 2 = 58$, और $2 < 7$.
\end{example}

\begin{method}[शेष की व्याख्या करना]\label{met:g3:division:interpret}
विभाजन के बाद कहानी पर वापस लौटें:
\begin{enumerate}
  \item ``प्रत्येक कितने/कितने पूर्ण समूह?'' --- \omterm{def:g3:division:remainder}{भागफल};
  \item ``कितने बचे हैं?'' --- \omterm{def:g3:division:remainder}{शेष};
  \item ``सबकुछ रखने के लिए कितने बक्से?'' --- \omterm{def:g3:division:remainder}{भागफल}
        \emph{मैं भी सहमत हूं} यदि \omterm{def:g3:division:remainder}{शेष} क्या नहीं है \omterm{def:g1:counting:zero}{शून्य}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g3:division:interpret}
$58$ छात्र कैनोइंग करने जाते हैं, $7$ प्रति डोंगी. $58 = 7 \times 8 + 2$:
आठ डोंगियाँ भरी हुई हैं, और $2$ छात्र रहते हैं --- उन्हें डोंगी की आवश्यकता होती है
भी. इसलिए $9$ डोंगी बुक की जानी चाहिए।
\end{example}

\section{आधा, तिहाई, चौथाई}

\begin{definition}[आधा, तीसरा, चौथाई]\label{def:g3:division:half}
 \emph{आधा}\index{आधा} किसी संख्या को विभाजित करने का परिणाम है $2$;
\emph{तीसरा}, द्वारा $3$; \emph{तिमाही}\index{तिमाही}, द्वारा $4$. उदाहरण के लिए
का आधा $18$ है $9$, का तीसरा $18$ है $6$, का चौथाई $36$
है $9$.
\end{definition}

\begin{example}[डबल्स और हाफ्स एक साथ चलते हैं]\label{ex:g3:division:double}
 \omterm{def:g3:division:half}{आधा} का $46$: तब से $46 = 40 + 6$, \omterm{def:g3:division:half}{आधा} प्रत्येक भाग का देता है
$20 + 3 = 23$. डबल से जांचें: $23 + 23 = 46$. जानना
दिल से दोगुना ($25 \to 50$, $45 \to 90$, $150 \to 300$) बनाता है
\omterm{def:g3:division:half}{आधा} तात्कालिक.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g3:division:1}
तालिकाओं का उपयोग करके पीछे की ओर गणना करें: $36 \div 4$; \quad $63 \div 9$;
\quad $40 \div 5$; \quad $56 \div 8$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:2}
प्रत्येक विभाजन के लिए समानता लिखें $ a = b \times q + r $ और दे
द \omterm{def:g3:division:remainder}{भागफल} और यह \omterm{def:g3:division:remainder}{शेष}: $17 \div 5$; \quad $30 \div 4$; \quad
$50 \div 6$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:3}
$28$ मार्बल आपस में बांटे जाते हैं $6$ बच्चे। प्रत्येक में कितने कंचे,
और कितने बचे हैं? यदि इससे मदद मिलती है तो साझाकरण बनाएं।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:4}
गणना: \omterm{def:g3:division:half}{आधा} का $68$; का तीसरा $27$; \omterm{def:g3:division:half}{तिमाही} का $48$;
 \omterm{def:g3:division:half}{आधा} का $150$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:5}
सच या झूठ? गुणा करके जांचें.
``$45 \div 5 = 9$''; \quad ``$52 \div 6 = 8$ \omterm{def:g3:division:remainder}{शेष} $4$'';
\quad ``$70 \div 8 = 9$ \omterm{def:g3:division:remainder}{शेष} $2$''.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:6}
$32$ तस्वीरें एक एलबम में चिपकी हुई हैं, $4$ प्रति पृष्ठ. कितने पेज हैं
प्रयोग किया गया? कौन चेहरा विभाजन का यह (साझाकरण या समूहीकरण) है?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:7}
की एक रस्सी $75$ सेमी के टुकड़ों में काटा जाता है $9$ सेमी। कितने पूर्ण टुकड़े,
और बचा हुआ टुकड़ा कब तक है?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:8}
$50$ की कारों में बच्चे यात्रा पर जाते हैं $4$ सीटें. कितनी गाड़ियाँ हैं
जरूरत है? (सावधान: हर किसी को सवारी करनी चाहिए!)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:division:9}
लुप्त संख्याएँ ज्ञात करें: $? \div 7 = 6$; \quad
$54 \div ? = 6$; \quad  \omterm{def:g3:division:half}{आधा} का $?$ है $35$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:division:10}
अमीना ने उसे साझा किया $47$ उसके बीच स्टिकर $5$ मित्रो, उतना ही निष्पक्ष
संभव है, और बचे हुए को अपने पास रखती है। कितने स्टीकर करता है
प्रत्येक मित्र को मिलता है, और अमीना कितने रखती है?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:division:11}
जब किसी संख्या को विभाजित किया जाता है $6$, क्या-क्या संभव है
\omterm{def:g3:division:remainder}{शेष}? सबसे बड़ी संख्या कौन सी है जिसका भाग किससे है? $6$ है
\omterm{def:g3:division:remainder}{भागफल} $7$?
\end{exercise}
