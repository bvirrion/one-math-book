\chapter{मापने}\label{ch:g3:measure}

कितनी देर, कितना भारी, कितना टिकता है, कितनी देर है? मापना
दुनिया को संख्याओं में बदल देता है --- बशर्ते हर कोई इसका उपयोग करे
इकाइयाँ। यह अध्याय लंबाई की इकाइयाँ प्रस्तुत करता है, \omterm{def:g2:measure:units}{द्रव्यमान} और \omterm{def:g2:measure:units}{क्षमता},
और घड़ी पढ़ना सिखाता है।

\section{लंबाई}

\begin{definition}[लंबाई की इकाइयाँ]\label{def:g3:measure:length}
लंबाई की मूल इकाई है \emph{मीटर} (एम)। छोटी लंबाई के लिए:
सेंटीमीटर (सेमी) और मिलीमीटर (मिमी); लंबी दूरी के लिए:
किलोमीटर (किमी).
\[
1 \text{ m} = 100 \text{ cm}, \qquad
1 \text{ cm} = 10 \text{ mm}, \qquad
1 \text{ km} = 1\,000 \text{ m}.
\]
\end{definition}

{\small रूलर पढ़ना: सेंटीमीटर के लिए बड़े निशान, सेंटीमीटर के लिए छोटे निशान
मिलीमीटर. वस्तु प्रारंभ होती है $0$ निशान---किनारे पर नहीं
शासक! --- और पर समाप्त होता है $6$ के बाद वां छोटा निशान $3$.}
\end{omfigure}

\begin{example}[सही इकाई का चयन]\label{ex:g3:measure:choose}
एक चींटी: के बारे में $5$ मिमी. एक पेंसिल: के बारे में $18$ सेमी। एक दरवाजा: के बारे में $2$ एम।
अगले गाँव तक पैदल चलें: के बारे में $3$ किमी. ``ए की लंबाई'' देते हुए
पेंसिल है $18$'' का कोई मतलब नहीं --- इकाई उत्तर का हिस्सा है।
\end{example}

\begin{method}[इकाइयों को परिवर्तित करना]\label{met:g3:measure:convert}
से परिवर्तित करना, गुणा करना या भाग देना $10$, $100$ या $1\,000$:
\begin{enumerate}
  \item एक बड़ी इकाई से छोटी इकाई तक, \emph{गुणा}:
        $3$ एम $= 3 \times 100 = 300$ सेमी;
  \item छोटी इकाई से बड़ी इकाई तक, \emph{विभाजित करना}:
        $250$ सेमी $= 250 \div 100 = 2$ मी और $50$ सेमी;
  \item लंबाई जोड़ने के लिए, उन्हें में बदलें \emph{वही} इकाई प्रथम:
        $1$ एम $+ 35$ सेमी $= 100 + 35 = 135$ सेमी।
\end{enumerate}
\end{method}

\section{द्रव्यमान और क्षमताएँ}

\begin{definition}[द्रव्यमान और क्षमता की इकाइयाँ]\label{def:g3:measure:mass}
\omterm{def:g2:measure:units}{द्रव्यमान} को मापा जाता है \emph{ग्राम} (जी) और \emph{किलोग्राम} (किग्रा),
\omterm{def:g2:measure:units}{क्षमता} (एक कंटेनर में कितना सामान है)। \emph{लीटर} (भूमि
\emph{सेंटीलीटर} (सीएल):
\[
1 \text{ kg} = 1\,000 \text{ g},
\qquad
1 \text{ L} = 100 \text{ cL}.
\]
\end{definition}

\begin{example}\label{ex:g3:measure:mass}
एक पत्र: के बारे में $20$ जी। आटे का एक थैला: $1$ किग्रा. एक बाथटब: के बारे में
$150$ एल. एक गिलास: के बारे में $20$ सीएल. एक तराजू को पकड़कर संतुलित करने के लिए
$1$ किलो बैग के खिलाफ $650$ सेब का ग्राम, \omterm{def:g1:addition:def}{जोड़}ें $1\,000 - 650 = 350$ जी चालू
सेब का पक्ष.
\end{example}

\section{घड़ी पढ़ना}

\begin{method}[समय पढ़ना]\label{met:g3:measure:clock}
सुइयों वाली घड़ी पर:
\begin{enumerate}
  \item  \emph{छोटा} हाथ घंटा देता है: उसके पास जो संख्या है उसे पढ़ें
        अंतिम पारित;
  \item  \emph{लंबा} हाथ मिनट देता है: प्रत्येक बड़ा निशान है $5$
        मिनट, इतना लंबा हाथ $3$ मतलब $15$ मिनट;
  \item आधे समय के बाद, कोई ``अगले घंटे तक'' भी पढ़ सकता है:
        $10{:}50$ ``दस से ग्यारह'' है।
\end{enumerate}
\end{method}

\begin{example}[अवधियां]\label{ex:g3:measure:duration}
फिल्म शुरू होती है $14{:}40$ और पर समाप्त होता है $16{:}10$. यह कितना लंबा है?
दो हॉप्स में गिनें:
\begin{enumerate}
  \item से $14{:}40$ को $15{:}00$: $20$ मिनट;
  \item से $15{:}00$ को $16{:}10$: $1$ घंटा $10$ मिनट।
\end{enumerate}
कुल: $1$ एच $30$ मि. सावधान: एक घंटा हो गया है $60$ मिनट, तो
$16{:}10$ ऋण $14{:}40$ की तरह गणना नहीं की जा सकती $1610 - 1440$!
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g3:measure:1}
आप किस इकाई का उपयोग करेंगे (मिमी, सेमी, मी या किमी): एक कील की चौड़ाई;
सहपाठी की ऊंचाई; विद्यालय प्रांगण की लंबाई; द
दो शहरों के बीच की दूरी?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:2}
बदलना: $5$ मी में सेमी; \quad $70$ मिमी से सेमी; \quad $4$ किमी में मी;
\quad $600$ सेमी में मी.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:3}
अपने रूलर से a की भुजाओं को (निकटतम मिलीमीटर तक) मापें
एक सहपाठी द्वारा बनाया गया त्रिभुज, और उसकी कुल लंबाई की गणना करें
सीमा.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:4}
गणना करें, पहले परिवर्तित करें: $2$ एम $+ 45$ सेमी (सेमी में); \quad
$1$ किमी $+ 250$ मी (एम में); \quad $3$ सेमी $+ 8$ मिमी (मिमी में).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:5}
बदलना: $3$ किग्रा में जी; \quad $2\,500$ g को kg और g में; \quad
$4$ एल से सीएल; \quad $350$ सीएल को एल और सीएल में बदलें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:6}
एक नुस्खे की जरूरत है $750$ आटा का ग्राम. मोना के पास है $1$ किलो बैग. क्या उसके पास है
पर्याप्त? कितना बचेगा?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:7}
समय पढ़ें: शॉर्ट हैंड बीच में है $2$ और $3$, लंबा हाथ
पर $6$; पर छोटा हाथ $9$, पर लंबा हाथ $12$;
छोटा हाथ लगभग पर $5$, पर लंबा हाथ $10$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:8}
स्कूल शुरू होता है $8{:}30$ और सुबह समाप्त होती है $11{:}45$. कब तक
क्या सुबह हो गयी है? (हॉप्स में गिनें, जैसे कि
\cref{ex:g3:measure:duration}.)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g3:measure:9}
ट्रेन रवाना हो जाती है $9{:}50$ और सवारी चलती है $45$ मिनट। किस पर
क्या समय आ गया? (हॉप टू $10{:}00$ पहला।)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:measure:10}
एक बोतल रखती है $1$ एल. पॉल का गिलास भरता है $20$ सीएल. कितने भरे हुए हैं
क्या वह चश्मा डाल सकता है? (पहले कनवर्ट करें।)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g3:measure:11}
लीना चलती है $600$ मैं हर दिन स्कूल जाता हूँ और वापस आता हूँ। कितने मीटर
क्या वह अंदर चलती है? $5$-दिन का स्कूल सप्ताह? उत्तर मीटर में दीजिए,
फिर किलोमीटर में.
\end{exercise}
