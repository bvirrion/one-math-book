\chapter{Numbers to 100}\label{ch:g1:tens}

Counting a big pile one by one takes forever. The great trick:
bundle the objects in packs of \emph{ten}, and count the packs. This
is how the numbers up to $100$ are built --- and why they are
written with two digits.

\section{Bundles of ten}

\begin{definition}[Tens and units]\label{def:g1:tens:def}
A \emph{ten} is a bundle of ten units. The number $34$ means $3$
tens and $4$ units:
\[
34 = 10 + 10 + 10 + 4 .
\]
The left digit counts the bundles, the right digit the loose units.
\end{definition}

      ({0.14*\i},0) -- ({0.14*\i},1.6);
    \draw[omExo, thick] (-0.25,0.55) ellipse (1.0 and 0.32);
    \end{scope}
  }
  \foreach \i in {0,...,3} \draw[omThm, very thick]
    (7.4+0.35*\i,0) -- (7.4+0.35*\i,1.6);

    {$3$ bundles and $4$ sticks: $34$};
\end{tikzpicture}

{\small Thirty-four sticks: three bundles of ten (tied with a band)
and four loose sticks. No need to count to $34$ one by one!}
\end{omfigure}

\begin{example}[Reading the numbers]\label{ex:g1:tens:reading}
$20$ is two tens (twenty), $30$ three tens (thirty)\,\dots\ up to
$90$ (ninety) and $100$ (one hundred: ten tens!). In between:
$47$ reads ``forty-seven'' --- forty for the $4$ tens, seven for the
units.
\end{example}

\begin{example}[Same digits, different numbers]\label{ex:g1:tens:place}
$27$ and $72$ use the same digits but are very different: $27$ is
$2$ tens and $7$ units; $72$ is $7$ tens and $2$ units --- much
bigger! The \emph{place} of a digit decides what it counts.
\end{example}

\section{The hundred grid}

\begin{tikzpicture}[scale=0.52]
  \foreach \r in {0,...,9} {
    \foreach \c in {0,...,9} {
      \pgfmathtruncatemacro{\n}{\r*10+\c+1}
      \draw[gray!50] (\c,-\r) +(-0.5,-0.5) rectangle +(0.5,0.5);
      \node[font=\footnotesize] at (\c,-\r) {\n};
    }
  }
  \draw[omThm, very thick] (3,-4) +(-0.5,-0.5) rectangle +(0.5,0.5);
  \draw[omDef, very thick] (3,-5) +(-0.5,-0.5) rectangle +(0.5,0.5);
\end{tikzpicture}

{\small The hundred grid: each row is one ten. One step right is
$+1$; one step \emph{down} is $+10$ (from the red $44$ straight down
to the blue $54$).}
\end{omfigure}

\begin{method}[Moving on the grid]\label{met:g1:tens:grid}
\begin{enumerate}
  \item $+1$ / $-1$: one step right / left;
  \item $+10$ / $-10$: one step down / up --- only the tens digit
        changes: $37 + 10 = 47$;
  \item counting by tens: $10, 20, 30, \dots$ --- follow the last
        column of the grid straight down.
\end{enumerate}
\end{method}

\begin{example}[Counting by 2s and 5s]\label{ex:g1:tens:skip}
By $2$s: $2, 4, 6, 8, 10, 12, \dots$ (the even numbers). By $5$s:
$5, 10, 15, 20, 25, \dots$ --- they end in $5$ or $0$ and make a
nice pattern on the grid. Skip counting prepares multiplication
(\cref{ch:g2:mult}).
\end{example}

\section{Comparing bigger numbers}

\begin{method}[Comparing with tens]\label{met:g1:tens:compare}
More tens wins: $61 > 58$ because $6$ tens beat $5$ tens ---
even though $8$ is more than $1$! Only when the tens are equal do
the units decide: $43 < 47$.
\end{method}

\begin{example}\label{ex:g1:tens:compare}
Order $35$, $53$, $29$: tens first ($2$, $3$, $5$):
\[
29 < 35 < 53 .
\]
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g1:tens:1}
How many tens and units in: $26$; \ $40$; \ $73$; \ $99$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:2}
Write the number: $5$ tens and $2$ units; \ $8$ tens; \ $1$ ten and
$9$ units; \ $10$ tens.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:3}
Write in digits: sixty-three; \ forty; \ eighty-five; \ one
hundred.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:4}
Compute by moving on the hundred grid: $23 + 10$; \quad $67 + 10$;
\quad $45 - 10$; \quad $80 - 10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:5}
Continue: $10, 20, 30, ?, ?, ?$. \quad
$5, 10, 15, ?, ?, ?$. \quad $12, 14, 16, ?, ?, ?$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:6}
Copy and complete with $<$ or $>$:
\[
34 \;?\; 43, \qquad
58 \;?\; 61, \qquad
70 \;?\; 69, \qquad
99 \;?\; 100 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:7}
Order from smallest to biggest: $46$; \ $19$; \ $64$; \ $91$; \
$41$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:8}
A gardener bundles $52$ flowers in bunches of ten. How many full
bunches, and how many loose flowers?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g1:tens:9}
On the hundred grid, start at $36$; go two steps down and one step
left. Where do you land? Which computation did you just do?
\end{exercise}
