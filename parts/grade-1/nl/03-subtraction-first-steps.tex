\chapter{Aftrekken: eerste stappen}\label{ch:g1:subtraction}

Zeven ballonnen, er vliegen er twee weg: hoeveel blijven er over?
Wegnemen is \emph{aftrekken}, geschreven met het teken $-$. En
aftrekken heeft een tweede gezicht: het beantwoordt ook ``hoeveel
meer?''.

\section{Wegnemen}

\begin{definition}[Aftrekking]\label{def:g1:subtraction:def}
\emph{Aftrekken} betekent wegnemen en tellen wat er overblijft. We
schrijven
\[
7 - 2 = 5
\]
en lezen: ``zeven min twee is vijf''.
\end{definition}

\begin{example}[Achteruit tellen]\label{ex:g1:subtraction:countback}
Bij een klein wegnemen, tel achteruit op de lijn: $11 - 3$: zeg
``elf'', en dan drie stappen terug --- ``tien, negen, acht''. Dus
$11 - 3 = 8$.
\end{example}

\section{Hoeveel meer?}

\begin{example}[Het verschil]\label{ex:g1:subtraction:difference}
``Nina heeft $9$ knikkers en Tom heeft er $6$. Hoeveel meer heeft
Nina?'' Er wordt niets weggenomen, en toch is het een aftrekking:
maak paarjes van de knikkers, en tel Nina's knikkers zonder partner:
\[
9 - 6 = 3 .
\]
Nina heeft er $3$ meer. Het antwoord van een aftrekking heet het
\emph{verschil}\index{verschil}.
\end{example}

\begin{method}[Vooruit tellen]\label{met:g1:subtraction:countup}
Om een \omterm{ex:g1:subtraction:difference}{verschil} te vinden, is \emph{vooruit} tellen vaak het
makkelijkst: voor $12 - 9$, begin bij $9$ en klim naar $12$:
``tien, elf, twaalf'' --- drie stappen. Dus $12 - 9 = 3$. (Klim via
$10$ bij grotere gaten: $14 - 8$: van $8$ naar $10$ is $2$, van $10$
naar $14$ is $4$: \omterm{ex:g1:subtraction:difference}{verschil} $6$.)
\end{method}

\section{Optellen en aftrekken zijn tweelingen}

\begin{example}[Rekensommen-families]\label{ex:g1:subtraction:family}
De getallen $3$, $4$ en $7$ vormen een klein gezin:
\[
3 + 4 = 7, \qquad
4 + 3 = 7, \qquad
7 - 4 = 3, \qquad
7 - 3 = 4 .
\]
E\'{e}n \omterm{def:g1:addition:def}{som} van de familie kennen, geeft de drie andere gratis.
Aftrekken maakt optellen ongedaan: na $+4$ brengt $-4$ je terug.
\end{example}

\begin{example}[Controleren]\label{ex:g1:subtraction:check}
$13 - 5 = 8$ uitgerekend? Controleer met de tweelingoptelling:
$8 + 5 = 13$. \checkmark\ Een aftrekking is precies goed als
terugoptellen klopt.
\end{example}

\begin{example}[Welke bewerking?]\label{ex:g1:subtraction:choose}
``$6$ vogels op de draad, er vliegen er $4$ weg'': wegnemen,
$6 - 4 = 2$. ``$6$ vogels, er komen er $4$ bij'': bij elkaar doen,
$6 + 4 = 10$. ``$6$ mussen en $4$ roodborstjes: hoeveel meer
mussen?'': \omterm{ex:g1:subtraction:difference}{verschil}, $6 - 4 = 2$. Teken het verhaal als je twijfelt!
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:1}
Teken, streep door, en reken uit: $6 - 2$; \quad $8 - 5$; \quad
$10 - 4$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:2}
Tel achteruit om uit te rekenen: $9 - 3$; \quad $12 - 2$; \quad
$15 - 4$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:3}
Tel vooruit om het \omterm{ex:g1:subtraction:difference}{verschil} te vinden: $11 - 8$; \quad $13 - 9$;
\quad $16 - 12$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:4}
Schrijf de vier sommen van de familie van $2$, $6$ en $8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:5}
Reken uit, en controleer met de tweelingoptelling: $14 - 6$; \quad
$17 - 9$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:6}
Vul de gaten:
\[
10 - \;?\; = 7, \qquad
\;?\; - 5 = 5, \qquad
12 - \;?\; = 12 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:7}
``Er liggen $15$ kersen in de schaal. De kinderen eten er $8$.
Hoeveel blijven er over?'' Schrijf de aftrekking en het
antwoordzinnetje.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:8}
Kies voor elk verhaal $+$ of $-$, en reken dan uit: ``$9$ eenden op
de vijver, er zwemmen er $5$ weg''; ``$9$ eenden, er komen $5$
ganzen bij''; ``Ali is $9$, zijn zus is $5$: hoeveel jaar ouder is
Ali?''.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g1:subtraction:9}
Moedereend had $12$ eieren. Sommige zijn uitgekomen; nu blijven er
$12 - ?$ eieren over en ze telt $7$ niet-uitgekomen eieren. Hoeveel
eieren zijn uitgekomen? (Welke sommenfamilie is dit?)
\end{exercise}
