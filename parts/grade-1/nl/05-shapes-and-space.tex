\chapter{Vormen en ruimte}\label{ch:g1:shapes}

Ramen zijn rechthoeken, wielen zijn cirkels, daken verbergen
driehoeken: vormen zijn overal. In dit hoofdstuk leer je ze te
herkennen door zijden en hoeken te tellen, en precies te zeggen
\emph{waar} dingen zijn: links, rechts, boven, onder, tussen.

\section{De vier basisvormen}

\begin{definition}[Vierkant, rechthoek, driehoek, cirkel]\label{def:g1:shapes:def}
\begin{itemize}
  \item de \emph{driehoek}\index{driehoek} heeft $3$ zijden en $3$ hoeken;
  \item het \emph{vierkant}\index{vierkant} heeft $4$ gelijke zijden en $4$ hoeken;
  \item de \emph{rechthoek}\index{rechthoek} heeft $4$ zijden en $4$ hoeken, met
        overstaande zijden gelijk (een uitgerekt vierkant);
  \item de \emph{cirkel}\index{cirkel} is perfect rond: geen zijden, geen hoeken.
\end{itemize}
\end{definition}

\begin{example}[Scheve vormen blijven vormen]\label{ex:g1:shapes:tilted}
Een vierkant dat op zijn hoek staat, is nog steeds een vierkant ---
vier gelijke zijden, vier hoeken --- ook al lijkt het op een vlieger.
Een vorm draaien verandert niet wat hij is.
\end{example}

\begin{example}[Vormenjacht]\label{ex:g1:shapes:hunt}
In de klas: het bord is een rechthoek, de klok een cirkel, een
gevouwen servet kan een driehoek zijn, sommige ramen zijn
vierkanten. Echte voorwerpen zijn geen perfecte vormen, maar hun
vlakken lijken erop (lichamen en hun vlakken komen terug in
\cref{ch:g2:shapes}).
\end{example}

\section{Waar is het?}

\begin{definition}[Plaatswoorden]\label{def:g1:shapes:position}
Om te zeggen waar iets is: \emph{links} / \emph{rechts},
\emph{boven} / \emph{onder}, \emph{op} / \emph{onder},
\emph{voor} / \emph{achter}, \emph{tussen},
\emph{binnen} / \emph{buiten}.
\end{definition}

\section{Paden op een raster}

\begin{method}[Een pad beschrijven]\label{met:g1:shapes:path}
Op ruitjespapier is een pad een lijst van stappen:
$\to$ (\'{e}\'{e}n vakje naar rechts), $\leftarrow$ (links),
$\uparrow$ (omhoog), $\downarrow$ (omlaag).
\begin{enumerate}
  \item Om een pad te volgen: begin op het gemarkeerde vakje en volg
        de pijlen \'{e}\'{e}n voor \'{e}\'{e}n;
  \item om een pad te beschrijven: loop het in je hoofd en schrijf
        de pijlen op volgorde;
  \item twee verschillende paden kunnen dezelfde twee vakjes
        verbinden --- sommige korter, sommige langer.
\end{enumerate}
\end{method}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g1:shapes:1}
Tel voor elke vorm de zijden en hoeken en noem hem: een vorm met
$3$ hoeken; een ronde vorm zonder hoek; een vorm met $4$ gelijke
zijden.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:2}
Zoek thuis: twee rechthoeken, \'{e}\'{e}n cirkel, \'{e}\'{e}n driehoek,
\'{e}\'{e}n vierkant. (Deuren? borden? verkeersborden?)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:3}
Waar of niet waar? ``Een vierkant dat op zijn hoek staat, is geen
vierkant meer.'' Leg uit met \cref{ex:g1:shapes:tilted}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:4}
Teken op ruitjespapier: een vierkant van $3$ bij $3$ vakjes; een
rechthoek van $5$ bij $2$; een driehoek met een diagonaal van het
raster.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:5}
Zet drie speelgoedjes op een rij. Beschrijf de rij met
plaatswoorden: welke staat tussen de andere? Welke staat links?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:6}
Teken een cirkel. Teken een kruis \emph{binnen} de cirkel, een
ster \emph{buiten} de cirkel, en een stip \emph{op} de cirkel zelf.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:7}
Markeer op ruitjespapier een startvakje. Volg:
$\to\ \uparrow\ \to\ \uparrow\ \to$. Markeer de aankomst. Schrijf
dan het pad dat recht terug naar de start gaat.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:8}
Hoeveel hoeken in totaal: twee driehoeken en \'{e}\'{e}n vierkant?
E\'{e}n cirkel en twee rechthoeken?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g1:shapes:9}
Neem $12$ lucifers. Kun je een vierkant omlijnen met \emph{alle}
lucifers, met hele lucifers en evenveel aan elke zijde? Hoeveel
lucifers per zijde? Lukt het met $10$ lucifers?
\end{exercise}
