\chapter{Getallen tot 100}\label{ch:g1:tens}

Een grote stapel \'{e}\'{e}n voor \'{e}\'{e}n tellen duurt eeuwig. De
grote truc: bundel de voorwerpen in pakjes van \emph{tien}, en tel de
pakjes. Zo zijn de getallen tot $100$ opgebouwd --- en daarom worden
ze met twee cijfers geschreven.

\section{Bundels van tien}

\begin{definition}[Tientallen en eenheden]\label{def:g1:tens:def}
Een \emph{tiental} is een bundel van tien eenheden. Het getal $34$
betekent $3$ tientallen en $4$ eenheden:
\[
34 = 10 + 10 + 10 + 4 .
\]
Het linker cijfer telt de bundels, het rechter cijfer de losse
eenheden.
\end{definition}

    \begin{scope}[xshift=\b*2.4cm]

      ({0.16*\i},0) -- ({0.16*\i},1.7);

    \end{scope}

  \foreach \i in {0,...,3} \draw[omThm, very thick]

  \node[right, font=\small] at (9.7,0.85)
    {$3$ bundels en $4$ stokjes: $34$};
\end{tikzpicture}

{\small Vierendertig stokjes: drie bundels van tien (elk met een
bandje) en vier losse stokjes. Geen nood om tot $34$ \'{e}\'{e}n voor
\'{e}\'{e}n te tellen!}
\end{omfigure}

\begin{example}[De getallen lezen]\label{ex:g1:tens:reading}
$20$ is twee tientallen (twintig), $30$ drie tientallen
(dertig)\,\dots\ tot $90$ (negentig) en $100$ (honderd: tien
tientallen!). Daartussen: $47$ leest ``zevenenveertig'' --- veertig
voor de $4$ tientallen, zeven voor de eenheden.
\end{example}

  \node[font=\small] at (1.1,1.2) {tientallen};
  \node[font=\small] at (3.3,1.2) {eenheden};

\end{tikzpicture}

{\small De plaatswaardetabel van $34$: de $3$ staat bij de

een nieuwe kolom erbij!}
\end{omfigure}

\begin{example}[Zelfde cijfers, andere getallen]\label{ex:g1:tens:place}
$27$ en $72$ gebruiken dezelfde cijfers maar zijn heel anders: $27$
is $2$ tientallen en $7$ eenheden; $72$ is $7$ tientallen en $2$
eenheden --- veel groter! De \emph{plaats} van een cijfer bepaalt wat
het telt.
\end{example}

\section{Het honderdveld}

\begin{omfigure}
\begin{tikzpicture}[scale=0.52]
  \foreach \r in {0,...,9} {
    \foreach \c in {0,...,9} {
      \pgfmathtruncatemacro{\n}{\r*10+\c+1}
      \draw[gray!50] (\c,-\r) +(-0.5,-0.5) rectangle +(0.5,0.5);
      \node[font=\footnotesize] at (\c,-\r) {\n};
    }
  }
  \draw[omThm, very thick] (3,-4) +(-0.5,-0.5) rectangle +(0.5,0.5);
  \draw[omDef, very thick] (3,-5) +(-0.5,-0.5) rectangle +(0.5,0.5);
\end{tikzpicture}

{\small Het honderdveld: elke rij is \'{e}\'{e}n tiental. E\'{e}n stap
naar rechts is $+1$; \'{e}\'{e}n stap \emph{omlaag} is $+10$ (van de
rode $44$ recht omlaag naar de blauwe $54$).}
\end{omfigure}

\begin{method}[Bewegen op het veld]\label{met:g1:tens:grid}
\begin{enumerate}
  \item $+1$ / $-1$: \'{e}\'{e}n stap naar rechts / links;
  \item $+10$ / $-10$: \'{e}\'{e}n stap omlaag / omhoog --- alleen het
        tientalcijfer verandert: $37 + 10 = 47$;
  \item tellen met tientallen: $10, 20, 30, \dots$ --- volg de
        laatste kolom van het veld recht omlaag.
\end{enumerate}
\end{method}

\begin{example}[Tellen met 2 en met 5]\label{ex:g1:tens:skip}
Met $2$: $2, 4, 6, 8, 10, 12, \dots$ (de even getallen). Met $5$:
$5, 10, 15, 20, 25, \dots$ --- ze eindigen op $5$ of $0$ en maken
een mooi patroon op het veld. Zo tellen bereidt vermenigvuldigen
voor (\cref{ch:g2:mult}).
\end{example}

\section{Grotere getallen vergelijken}

\begin{method}[Vergelijken met tientallen]\label{met:g1:tens:compare}
Meer tientallen wint: $61 > 58$ omdat $6$ tientallen $5$ tientallen
verslaan --- ook al is $8$ meer dan $1$! Alleen als de tientallen
gelijk zijn, beslissen de eenheden: $43 < 47$.
\end{method}

\begin{example}\label{ex:g1:tens:compare}
Zet $35$, $53$, $29$ op volgorde: eerst de tientallen ($2$, $3$,
$5$):
\[
29 < 35 < 53 .
\]
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g1:tens:1}
Hoeveel tientallen en eenheden in: $26$; \ $40$; \ $73$; \ $99$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:2}
Schrijf het getal: $5$ tientallen en $2$ eenheden; \ $8$
tientallen; \ $1$ tiental en $9$ eenheden; \ $10$ tientallen.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:3}
Schrijf in cijfers: drie\"{e}nzeventig; \ veertig; \
vijfentachtig; \ honderd.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:4}
Reken uit door te bewegen op het honderdveld: $23 + 10$; \quad
$67 + 10$; \quad $45 - 10$; \quad $80 - 10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:5}
Ga door: $10, 20, 30, ?, ?, ?$. \quad
$5, 10, 15, ?, ?, ?$. \quad $12, 14, 16, ?, ?, ?$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:6}
Schrijf over en vul in met $<$ of $>$:
\[
34 \;?\; 43, \qquad
58 \;?\; 61, \qquad
70 \;?\; 69, \qquad
99 \;?\; 100 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:7}
Zet op volgorde van klein naar groot: $46$; \ $19$; \ $64$; \ $91$; \
$41$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:8}
Een tuinman bundelt $52$ bloemen in bossen van tien. Hoeveel volle
bossen, en hoeveel losse bloemen?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g1:tens:9}
Op het honderdveld, begin bij $36$; ga twee stappen omlaag en \'{e}\'{e}n
stap naar links. Waar kom je uit? Welke berekening heb je zojuist
gedaan?
\end{exercise}
