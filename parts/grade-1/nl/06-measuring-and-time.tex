\chapter{Meten en tijd}\label{ch:g1:measure}

Welk potlood is langer? Welke dag is het? Hoeveel kost een snoepje?
In dit hoofdstuk vergelijken we lengtes en meten we ze in
centimeters, zetten we de dagen van de week op volgorde, lezen we
hele uren op de klok, en tellen we kleine munten.

\section{Langer en korter}

\begin{method}[Lengtes vergelijken]\label{met:g1:measure:compare}
Om twee lengtes te vergelijken, leg ze naast elkaar met
\emph{\'{e}\'{e}n uiteinde gelijk}:
\begin{enumerate}
  \item leg beide voorwerpen naast elkaar, beginnend vanaf dezelfde
        lijn;
  \item degene die verder reikt is \emph{langer}, de andere is
        \emph{korter};
  \item als beide even ver reiken, hebben ze \emph{dezelfde
        lengte}.
\end{enumerate}
Het begin gelijk leggen is het hele geheim --- vergelijken vanaf
scheve starts bedriegt het oog.
\end{method}

\begin{definition}[De centimeter]\label{def:g1:measure:cm}
Om te zeggen \emph{hoe} lang, meten we met een liniaal in
\emph{centimeters} (cm). Zet de $0$ van de liniaal bij het begin van
het voorwerp, en lees het getal aan het einde.
\end{definition}

\section{De dagen van de week}

\begin{example}[De zeven dagen]\label{ex:g1:measure:days}
De week heeft $7$ dagen, altijd in dezelfde volgorde:
\[
\text{maandag, dinsdag, woensdag, donderdag, vrijdag, zaterdag,
zondag.}
\]
Na zondag begint maandag opnieuw. \emph{Gisteren} is de dag
ervoor, \emph{morgen} de dag erna: als vandaag woensdag is, was
gisteren dinsdag en is morgen donderdag.
\end{example}

\section{Hele uren op de klok}

\begin{method}[Een heel uur lezen]\label{met:g1:measure:clock}
Als de lange wijzer recht omhoog naar de $12$ wijst, is het een heel
uur: lees het getal waar de \emph{korte} wijzer naartoe wijst. Korte
wijzer op de $3$, lange wijzer op de $12$: het is $3$ uur.
\end{method}

\section{Kleine munten}

\begin{example}[Een bedrag maken]\label{ex:g1:measure:coins}
Met munten van $1$, $2$ en $5$ kun je veel bedragen op verschillende
manieren maken. Om $7$ te maken:
\[
5 + 2, \qquad
5 + 1 + 1, \qquad
2 + 2 + 2 + 1 .
\]
De grootste munten eerst is meestal de snelste manier, met de minste
munten.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g1:measure:1}
Leg drie potloden naast elkaar met \'{e}\'{e}n uiteinde gelijk. Welk is
het langst? Welk is het kortst?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:2}
Meet met je liniaal: de lange zijde van dit boek; je pink; een
lepel. Antwoord in hele centimeters.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:3}
Noem de dag: na maandag; v\'{o}\'{o}r zondag; twee dagen na vrijdag.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:4}
Vandaag is donderdag. Welke dag was gisteren? Welke dag is morgen?
Welke dag is over drie dagen?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:5}
Teken een klok die $6$ uur aangeeft. Waar staat de korte wijzer?
Waar staat de lange wijzer?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:6}
Lees het hele uur: korte wijzer op $5$, lange wijzer op $12$; korte
wijzer op $11$, lange wijzer op $12$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:7}
Maak $8$ met munten van $1$, $2$ en $5$, op twee verschillende
manieren.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:8}
Leo heeft een munt van $5$ en drie munten van $2$. Hoeveel geld
heeft hij in totaal?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g1:measure:9}
Een snoepje kost $3$. Zoe betaalt met een munt van $5$. Hoeveel
wisselgeld krijgt ze terug? (Tel vooruit van $3$ naar $5$.)
\end{exercise}
