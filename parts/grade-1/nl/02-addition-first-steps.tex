\chapter{Optellen: eerste stappen}\label{ch:g1:addition}

Drie rode knikkers en twee blauwe knikkers: samen, hoeveel? Bij elkaar
doen en het geheel tellen is \emph{optellen} --- de eerste bewerking,
met haar twee bekende tekens $+$ en $=$.

\section{Bij elkaar doen}

\begin{definition}[Optelling]\label{def:g1:addition:def}
\emph{Optellen} betekent bij elkaar doen en alles tellen. We
schrijven
\[
3 + 2 = 5
\]
en lezen: ``drie plus twee is vijf''. Het antwoord van een optelling
heet de \emph{som}\index{som}.
\end{definition}

  \node[font=\large] at (3.0,0) {$+$};

\end{tikzpicture}

{\small Een optelling in plaatjes: de twee groepen samen maken \'{e}\'{e}n
groep van vijf.}
\end{omfigure}

\begin{example}[De volgorde doet er niet toe]\label{ex:g1:addition:order}
$2 + 3$ en $3 + 2$ doen dezelfde knikkers bij elkaar: beide maken
$5$. Om $2 + 9$ uit te rekenen is het sneller om bij het grotere
getal te beginnen: zeg ``negen'', en tel er twee bij --- ``tien,
elf''. Dus $2 + 9 = 11$.
\end{example}

\section{De tientalvrienden}

\begin{example}[Aanvullingen tot 10]\label{ex:g1:addition:friends}
Twee getallen die samen $10$ maken, zijn ``tientalvrienden'':
\[
1 + 9, \quad 2 + 8, \quad 3 + 7, \quad 4 + 6, \quad 5 + 5 .
\]
Tien vingers laten ze allemaal zien: steek $3$ vingers op, en de
$7$ gevouwen vingers zijn de vriend. De tientalvrienden uit het
hoofd kennen maakt veel optellingen snel.
\end{example}

\end{tikzpicture}

{\small Het tientalraam toont elk paar tientalvrienden in \'{e}\'{e}n
oogopslag: wat gevuld is en wat ontbreekt, maken altijd samen $10$.}
\end{omfigure}

\begin{method}[Over de tien]\label{met:g1:addition:crossten}
Om $8 + 5$ uit te rekenen:
\begin{enumerate}
  \item neem van de $5$ wat de $8$ nodig heeft om $10$ te bereiken:
        dat is $2$ (zijn tientalvriend);
  \item $8 + 2 = 10$, en er blijft $3$ over van de $5$;
  \item $10 + 3 = 13$. Dus $8 + 5 = 13$.
\end{enumerate}
Via $10$ wordt een moeilijke \omterm{def:g1:addition:def}{som} twee makkelijke.
\end{method}

\begin{example}[Verdubbelingen]\label{ex:g1:addition:doubles}
De verdubbelingen zijn de moeite waard om uit het hoofd te leren:
\[
1+1=2, \quad 2+2=4, \quad 3+3=6, \quad 4+4=8, \quad 5+5=10,
\]
\[
6+6=12, \quad 7+7=14, \quad 8+8=16, \quad 9+9=18, \quad 10+10=20 .
\]
Bijna-verdubbelingen krijg je gratis: $6 + 7$ is \'{e}\'{e}n meer dan
$6 + 6$, dus $13$.
\end{example}

\section{Kleine sommen}

\begin{example}\label{ex:g1:addition:problem}
``Lea heeft $7$ stickers. Ze krijgt er $6$ bij. Hoeveel nu?''
Meer krijgen betekent optellen: $7 + 6 = 13$ (via tien:
$7 + 3 = 10$, daarna $+3$). Lea heeft $13$ stickers. Eindig altijd
met een kort zinnetje!
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g1:addition:1}
Teken de knikkers en reken uit: $4 + 3$; \quad $5 + 2$; \quad
$6 + 4$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:addition:2}
Reken uit door bij het grotere getal te beginnen: $3 + 8$; \quad
$2 + 12$; \quad $4 + 9$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:addition:3}
Geef de tientalvriend van: $7$; \ $4$; \ $9$; \ $5$; \ $1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:addition:4}
Reken uit via de tien (\cref{met:g1:addition:crossten}):
$9 + 4$; \quad $8 + 7$; \quad $6 + 5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:addition:5}
Zeg de verdubbelingen op, en reken er dan mee uit: $7 + 8$; \quad
$5 + 6$; \quad $9 + 8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:addition:6}
Vul de gaten:
\[
6 + \;?\; = 10, \qquad
\;?\; + 5 = 12, \qquad
10 + \;?\; = 17 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:addition:7}
``In de bus zitten $8$ kinderen; $5$ volwassenen stappen in. Hoeveel
mensen zitten er in de bus?'' Schrijf de optelling en het
antwoordzinnetje.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:addition:8}
Tom heeft $6$ rode auto's en $7$ blauwe auto's. Mia heeft $10$
auto's. Wie heeft meer auto's?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g1:addition:9}
Drie dobbelstenen laten $4$, $6$ en $5$ zien. Wat is het totaal?
(Tel eerst twee stenen op --- kies de twee die een tientalvriend
maken!)
\end{exercise}
