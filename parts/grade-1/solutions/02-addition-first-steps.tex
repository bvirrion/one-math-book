\section*{Chapter \ref{ch:g1:addition} --- Addition: First Steps}

\begin{solution}{exo:g1:addition:1}
$4 + 3 = 7$; \quad $5 + 2 = 7$; \quad $6 + 4 = 10$.
\end{solution}

\begin{solution}{exo:g1:addition:2}
$3 + 8 = 11$ (start at $8$, count on $3$); \quad
$2 + 12 = 14$; \quad $4 + 9 = 13$.
\end{solution}

\begin{solution}{exo:g1:addition:3}
Ten friends: $7 \to 3$; \ $4 \to 6$; \ $9 \to 1$; \ $5 \to 5$;
\ $1 \to 9$.
\end{solution}

\begin{solution}{exo:g1:addition:4}
$9 + 4 = 13$ ($9 + 1 = 10$, then $+3$); \quad
$8 + 7 = 15$ ($8 + 2 = 10$, then $+5$); \quad
$6 + 5 = 11$ ($6 + 4 = 10$, then $+1$).
\end{solution}

\begin{solution}{exo:g1:addition:5}
$7 + 8 = 15$ (one more than $7 + 7 = 14$); \quad
$5 + 6 = 11$ (one more than $5 + 5$); \quad
$9 + 8 = 17$ (one less than $9 + 9 = 18$).
\end{solution}

\begin{solution}{exo:g1:addition:6}
$6 + 4 = 10$; \quad $7 + 5 = 12$; \quad $10 + 7 = 17$.
\end{solution}

\begin{solution}{exo:g1:addition:7}
$8 + 5 = 13$. There are $13$ people in the bus.
\end{solution}

\begin{solution}{exo:g1:addition:8}
Tom has $6 + 7 = 13$ cars, Mia has $10$: Tom has more.
\end{solution}

\begin{solution}{exo:g1:addition:9}
Add the ten friends $4$ and $6$ first: $4 + 6 = 10$, then
$10 + 5 = 15$. The total is $15$.
\end{solution}
