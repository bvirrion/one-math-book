\section*{अध्याय \ref{ch:g1:addition} --- जोड़: पहला कदम}

\begin{solution}{exo:g1:addition:1}
$4 + 3 = 7$; \quad $5 + 2 = 7$; \quad $6 + 4 = 10$.
\end{solution}

\begin{solution}{exo:g1:addition:2}
$3 + 8 = 11$ (पर शुरू करें $8$, भरोसा रखें $3$); \quad
$2 + 12 = 14$; \quad $4 + 9 = 13$.
\end{solution}

\begin{solution}{exo:g1:addition:3}
दस मित्र: $7 \to 3$; \ $4 \to 6$; \ $9 \to 1$; \ $5 \to 5$;
\ $1 \to 9$.
\end{solution}

\begin{solution}{exo:g1:addition:4}
$9 + 4 = 13$ ($9 + 1 = 10$, तब $+3$); \quad
$8 + 7 = 15$ ($8 + 2 = 10$, तब $+5$); \quad
$6 + 5 = 11$ ($6 + 4 = 10$, तब $+1$).
\end{solution}

\begin{solution}{exo:g1:addition:5}
$7 + 8 = 15$ (एक से अधिक $7 + 7 = 14$); \quad
$5 + 6 = 11$ (एक से अधिक $5 + 5$); \quad
$9 + 8 = 17$ (एक से कम $9 + 9 = 18$).
\end{solution}

\begin{solution}{exo:g1:addition:6}
$6 + 4 = 10$; \quad $7 + 5 = 12$; \quad $10 + 7 = 17$.
\end{solution}

\begin{solution}{exo:g1:addition:7}
$8 + 5 = 13$. वहाँ हैं $13$ बस में लोग.
\end{solution}

\begin{solution}{exo:g1:addition:8}
टॉम के पास है $6 + 7 = 13$ कारें, मिया के पास हैं $10$: टॉम के पास और भी बहुत कुछ है।
\end{solution}

\begin{solution}{exo:g1:addition:9}
दस मित्रों को \omterm{def:g1:addition:def}{जोड़}ें $4$ और $6$ पहला: $4 + 6 = 10$, तब
$10 + 5 = 15$. कुल है $15$.
\end{solution}
