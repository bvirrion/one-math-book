\section*{अध्याय \ref{ch:g1:counting} --- 20 तक गिनती}

\begin{solution}{exo:g1:counting:1}
उत्तर नाम पर निर्भर करते हैं; दो हाथ हैं $10$ उँगलियाँ; दो बिल्लियाँ
है $8$ पैर.
\end{solution}

\begin{solution}{exo:g1:counting:2}
$7$; \ $12$; \ $20$; \ $0$. शब्दों में: चार; पंद्रह; अठारह।
\end{solution}

\begin{solution}{exo:g1:counting:3}
$8, 9, 10, 11, 12, 13$. पिछड़ा: $15, 14, 13, 12, 11, 10$.
\end{solution}

\begin{solution}{exo:g1:counting:4}
से ठीक पहले $10$: $9$. तुरन्त बाद $16$: $17$. बीच में $12$ और
$14$: $13$.
\end{solution}

\begin{solution}{exo:g1:counting:5}
$6 < 9$; \ $14 > 8$; \ $11 < 13$; \ $20 > 19$.
\end{solution}

\begin{solution}{exo:g1:counting:6}
बाँधना $6$ मंडलियों के साथ $8$ पत्तों को पार करता है $2$ ए के बिना पार करता है
साथी: क्रॉस अधिक हैं, द्वारा $2$.
\end{solution}

\begin{solution}{exo:g1:counting:7}
$2 < 7 < 10 < 12 < 17$.
\end{solution}

\begin{solution}{exo:g1:counting:8}

\end{solution}

\begin{solution}{exo:g1:counting:9}
