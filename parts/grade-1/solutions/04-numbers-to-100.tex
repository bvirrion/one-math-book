\section*{Chapter \ref{ch:g1:tens} --- Numbers to 100}

\begin{solution}{exo:g1:tens:1}
$26$: $2$ tens, $6$ units. $40$: $4$ tens, $0$ units. $73$: $7$
tens, $3$ units. $99$: $9$ tens, $9$ units.
\end{solution}

\begin{solution}{exo:g1:tens:2}

\end{solution}

\begin{solution}{exo:g1:tens:3}

\end{solution}

\begin{solution}{exo:g1:tens:4}
$23 + 10 = 33$; \quad $67 + 10 = 77$; \quad $45 - 10 = 35$; \quad
$80 - 10 = 70$.
\end{solution}

\begin{solution}{exo:g1:tens:6}

\end{solution}

\begin{solution}{exo:g1:tens:7}

\end{solution}

\begin{solution}{exo:g1:tens:8}
$52$ is $5$ tens and $2$ units: $5$ full bunches and $2$ loose
flowers.
\end{solution}

\begin{solution}{exo:g1:tens:9}
From $36$, two steps down is $+20$ ($\to 56$), one step left is
$-1$ ($\to 55$). You land on $55$. The computation is
$36 + 20 - 1$, that is $36 + 19$.
\end{solution}
