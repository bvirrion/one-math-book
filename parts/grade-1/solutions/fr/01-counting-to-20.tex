\section*{Chapitre \ref{ch:g1:counting} --- Compter jusqu'\`a 20}

\begin{solution}{exo:g1:counting:1}
Les r\'eponses d\'ependent du pr\'enom~; deux mains ont $10$ doigts~;
deux chats ont $8$ pattes.
\end{solution}

\begin{solution}{exo:g1:counting:2}
$7$~; \ $12$~; \ $20$~; \ $0$. En lettres~: quatre~; quinze~;
dix-huit.
\end{solution}

\begin{solution}{exo:g1:counting:3}
$8, 9, 10, 11, 12, 13$. \`A rebours~: $15, 14, 13, 12, 11, 10$.
\end{solution}

\begin{solution}{exo:g1:counting:4}
Juste avant $10$~: $9$. Juste apr\`es $16$~: $17$. Entre $12$ et
$14$~: $13$.
\end{solution}

\begin{solution}{exo:g1:counting:5}
$6 < 9$~; \ $14 > 8$~; \ $11 < 13$~; \ $20 > 19$.
\end{solution}

\begin{solution}{exo:g1:counting:6}
Apparier $6$ ronds avec $8$ croix laisse $2$ croix sans partenaire~:
les croix sont plus nombreuses, de $2$.
\end{solution}

\begin{solution}{exo:g1:counting:7}
$2 < 7 < 10 < 12 < 17$.
\end{solution}

\begin{solution}{exo:g1:counting:8}
Deuxi\`eme~: Bob. Quatri\`eme~: Dan. Dan est \`a la place $4$.
\end{solution}

\begin{solution}{exo:g1:counting:9}
Trois pas apr\`es $13$~: $16$. Trois pas avant $13$~: $10$.
\end{solution}
