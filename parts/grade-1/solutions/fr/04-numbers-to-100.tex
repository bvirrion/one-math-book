\section*{Chapitre \ref{ch:g1:tens} --- Les nombres jusqu'\`a 100}

\begin{solution}{exo:g1:tens:1}
$26$~: $2$ dizaines, $6$ unit\'es. $40$~: $4$ dizaines, $0$ unit\'e.
$73$~: $7$ dizaines, $3$ unit\'es. $99$~: $9$ dizaines, $9$ unit\'es.
\end{solution}

\begin{solution}{exo:g1:tens:2}
$52$~; \ $80$~; \ $19$~; \ $100$.
\end{solution}

\begin{solution}{exo:g1:tens:3}
$63$~; \ $40$~; \ $85$~; \ $100$.
\end{solution}

\begin{solution}{exo:g1:tens:4}
$23 + 10 = 33$~; \quad $67 + 10 = 77$~; \quad $45 - 10 = 35$~; \quad
$80 - 10 = 70$.
\end{solution}

\begin{solution}{exo:g1:tens:5}
$10, 20, 30, 40, 50, 60$. \quad $5, 10, 15, 20, 25, 30$. \quad
$12, 14, 16, 18, 20, 22$.
\end{solution}

\begin{solution}{exo:g1:tens:6}
$34 < 43$~; \ $58 < 61$~; \ $70 > 69$~; \ $99 < 100$.
\end{solution}

\begin{solution}{exo:g1:tens:7}
$19 < 41 < 46 < 64 < 91$.
\end{solution}

\begin{solution}{exo:g1:tens:8}
$52$ c'est $5$ dizaines et $2$ unit\'es~: $5$ bouquets complets et
$2$ fleurs isol\'ees.
\end{solution}

\begin{solution}{exo:g1:tens:9}
\`A partir de $36$, deux pas vers le bas c'est $+20$ ($\to 56$), un
pas \`a gauche c'est $-1$ ($\to 55$). On arrive sur $55$. Le calcul
est $36 + 20 - 1$, c'est-\`a-dire $36 + 19$.
\end{solution}
