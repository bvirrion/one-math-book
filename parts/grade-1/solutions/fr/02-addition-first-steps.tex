\section*{Chapitre \ref{ch:g1:addition} --- Addition~: premiers pas}

\begin{solution}{exo:g1:addition:1}
$4 + 3 = 7$~; \quad $5 + 2 = 7$~; \quad $6 + 4 = 10$.
\end{solution}

\begin{solution}{exo:g1:addition:2}
$3 + 8 = 11$ (partir de $8$, compter $3$ de plus)~; \quad
$2 + 12 = 14$~; \quad $4 + 9 = 13$.
\end{solution}

\begin{solution}{exo:g1:addition:3}
Amis de $10$~: $7 \to 3$~; \ $4 \to 6$~; \ $9 \to 1$~; \ $5 \to 5$~;
\ $1 \to 9$.
\end{solution}

\begin{solution}{exo:g1:addition:4}
$9 + 4 = 13$ ($9 + 1 = 10$, puis $+3$)~; \quad
$8 + 7 = 15$ ($8 + 2 = 10$, puis $+5$)~; \quad
$6 + 5 = 11$ ($6 + 4 = 10$, puis $+1$).
\end{solution}

\begin{solution}{exo:g1:addition:5}
$7 + 8 = 15$ (un de plus que $7 + 7 = 14$)~; \quad
$5 + 6 = 11$ (un de plus que $5 + 5$)~; \quad
$9 + 8 = 17$ (un de moins que $9 + 9 = 18$).
\end{solution}

\begin{solution}{exo:g1:addition:6}
$6 + 4 = 10$~; \quad $7 + 5 = 12$~; \quad $10 + 7 = 17$.
\end{solution}

\begin{solution}{exo:g1:addition:7}
$8 + 5 = 13$. Il y a $13$ personnes dans le bus.
\end{solution}

\begin{solution}{exo:g1:addition:8}
Tom a $6 + 7 = 13$ voitures, Mia en a $10$~: Tom en a plus.
\end{solution}

\begin{solution}{exo:g1:addition:9}
Ajouter d'abord les amis de $10$ $4$ et $6$~: $4 + 6 = 10$, puis
$10 + 5 = 15$. Le total est $15$.
\end{solution}
