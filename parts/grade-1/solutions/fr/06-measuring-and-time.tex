\section*{Chapitre \ref{ch:g1:measure} --- Mesures et temps}

\begin{solution}{exo:g1:measure:1}
Les r\'eponses d\'ependent des crayons~; le plus long d\'epasse le
plus loin quand ils sont align\'es \`a une extr\'emit\'e, le plus
court d\'epasse le moins.
\end{solution}

\begin{solution}{exo:g1:measure:2}
Les r\'eponses varient~; chacune se lit en pla\c{c}ant le $0$ de la
r\`egle au d\'ebut de l'objet et en lisant le centim\`etre entier \`a
son extr\'emit\'e.
\end{solution}

\begin{solution}{exo:g1:measure:3}
Apr\`es lundi~: mardi. Avant dimanche~: samedi. Deux jours apr\`es
vendredi~: dimanche.
\end{solution}

\begin{solution}{exo:g1:measure:4}
Hier~: mercredi. Demain~: vendredi. Dans trois jours~: dimanche.
\end{solution}

\begin{solution}{exo:g1:measure:5}
\`A $6$~heures, la petite aiguille pointe vers le $6$ et la grande
aiguille pointe vers le haut, sur le $12$.
\end{solution}

\begin{solution}{exo:g1:measure:6}
Petite aiguille sur $5$~: $5$~heures. Petite aiguille sur $11$~:
$11$~heures.
\end{solution}

\begin{solution}{exo:g1:measure:7}
Pour $8$, par exemple $5 + 2 + 1$ ou $2 + 2 + 2 + 2$ (d'autres
r\'eponses correctes existent).
\end{solution}

\begin{solution}{exo:g1:measure:8}
$5 + 2 + 2 + 2 = 11$. L\'eo a $11$.
\end{solution}

\begin{solution}{exo:g1:measure:9}
De $3$ \`a $5$ en montant, c'est $2$~: Zo\'e re\c{c}oit $2$.
\end{solution}
