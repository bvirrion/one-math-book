\section*{Capítulo \ref{ch:g1:measure} --- Measuring and Time}

\begin{solution}{exo:g1:measure:1}
Las respuestas dependen de los lápices; el más largo llega más lejos cuando
están emparejados en un extremo, el más corto llega menos lejos.
\end{solution}

\begin{solution}{exo:g1:measure:2}
Las respuestas varían; cada uno se lee poniendo el $0$ de la regla al principio
del objeto y leyendo todo el centímetro en su extremo.
\end{solution}

\begin{solution}{exo:g1:measure:3}
Después del lunes: martes. Antes del domingo: sábado. Dos días después
Viernes: domingo.
\end{solution}

\begin{solution}{exo:g1:measure:4}
Ayer: miércoles. Mañana: viernes. En tres días: domingo.
\end{solution}

\begin{solution}{exo:g1:measure:5}
A las horas $6$ la manecilla corta apunta al $6$ y la manecilla larga
apunta hasta el $12$.
\end{solution}

\begin{solution}{exo:g1:measure:6}
Manecilla corta en $5$: $5$ en punto. Manecilla corta en $11$: $11$ en punto.
\end{solution}

\begin{solution}{exo:g1:measure:7}
Para $8$, por ejemplo $5 + 2 + 1$ o $2 + 2 + 2 + 2$ (otro
existen respuestas).
\end{solution}

\begin{solution}{exo:g1:measure:8}
$5 + 2 + 2 + 2 = 11$. Leo tiene $11$.
\end{solution}

\begin{solution}{exo:g1:measure:9}
Desde $3$ hasta $5$ es $2$: Zoe recupera $2$.
\end{solution}
