\section*{Capítulo \ref{ch:g1:counting} --- Counting to 20}

\begin{solution}{exo:g1:counting:1}
Las respuestas dependen del nombre; dos manos tienen dedos $10$; dos gatos
tiene patas $8$.
\end{solution}

\begin{solution}{exo:g1:counting:2}
$7$; $12$; $20$; $0$. En palabras: cuatro; quince; dieciocho.
\end{solution}

\begin{solution}{exo:g1:counting:3}
$8, 9, 10, 11, 12, 13$. Atrás: $15, 14, 13, 12, 11, 10$.
\end{solution}

\begin{solution}{exo:g1:counting:4}
Justo antes de $10$: $9$. Justo después de $16$: $17$. Entre $12$ y
$14$: $13$.
\end{solution}

\begin{solution}{exo:g1:counting:5}
$6 < 9$; \ $14 > 8$; \ $11 < 13$; \ $20 > 19$.
\end{solution}

\begin{solution}{exo:g1:counting:6}
Emparejar círculos $6$ con cruces $8$ deja cruces $2$ sin
compañero: las cruces son más, por $2$.
\end{solution}

\begin{solution}{exo:g1:counting:7}
$2 < 7 < 10 < 12 < 17$.
\end{solution}

\begin{solution}{exo:g1:counting:8}
Segundo: Bob. Cuarto: Dan. Dan está en el lugar $4$.
\end{solution}

\begin{solution}{exo:g1:counting:9}
Tres pasos después de $13$: $16$. Tres pasos antes de $13$: $10$.
\end{solution}
