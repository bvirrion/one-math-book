\section*{Capítulo \ref{ch:g1:subtraction} --- Subtraction: First Steps}

\begin{solution}{exo:g1:subtraction:1}
$6 - 2 = 4$; \quad $8 - 5 = 3$; \quad $10 - 4 = 6$.
\end{solution}

\begin{solution}{exo:g1:subtraction:2}
$9 - 3 = 6$; \quad $12 - 2 = 10$; \quad $15 - 4 = 11$.
\end{solution}

\begin{solution}{exo:g1:subtraction:3}
$11 - 8 = 3$ (desde $8$ hasta $11$); \quad
$13 - 9 = 4$; \quad $16 - 12 = 4$.
\end{solution}

\begin{solution}{exo:g1:subtraction:4}
$2 + 6 = 8$, \quad $6 + 2 = 8$, \quad $8 - 6 = 2$, \quad
$8 - 2 = 6$.
\end{solution}

\begin{solution}{exo:g1:subtraction:5}
$14 - 6 = 8$ (comprobar: $8 + 6 = 14$); \quad
$17 - 9 = 8$ (comprobar: $8 + 9 = 17$).
\end{solution}

\begin{solution}{exo:g1:subtraction:6}
$10 - 3 = 7$; \quad $10 - 5 = 5$; \quad $12 - 0 = 12$.
\end{solution}

\begin{solution}{exo:g1:subtraction:7}
$15 - 8 = 7$. Quedan $7$ cerezas.
\end{solution}

\begin{solution}{exo:g1:subtraction:8}
``$5$ nadando'': $9 - 5 = 4$ patos. ``$5$ llegan los gansos'':
$9 + 5 = 14$ pájaros. ``Cuántos años tiene'': $9 - 5 = 4$ años.
\end{solution}

\begin{solution}{exo:g1:subtraction:9}
Familia de $12$, $7$ y $5$: $12 - 7 = 5$ huevos eclosionados (y
$5 + 7 = 12$ lo comprueba).
\end{solution}
