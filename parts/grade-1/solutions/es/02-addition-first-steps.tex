\section*{Capítulo \ref{ch:g1:addition} --- Addition: First Steps}

\begin{solution}{exo:g1:addition:1}
$4 + 3 = 7$; \quad $5 + 2 = 7$; \quad $6 + 4 = 10$.
\end{solution}

\begin{solution}{exo:g1:addition:2}
$3 + 8 = 11$ (comienza en $8$, cuenta con $3$); \quad
$2 + 12 = 14$; \quad$4 + 9 = 13$.
\end{solution}

\begin{solution}{exo:g1:addition:3}
Diez amigos: $7 \to 3$; $4 \to 6$; $9 \to 1$; $5 \to 5$;
$1 \to 9$.
\end{solution}

\begin{solution}{exo:g1:addition:4}
$9 + 4 = 13$ ($9 + 1 = 10$, luego $+3$); \quad
$8 + 7 = 15$ ($8 + 2 = 10$, luego $+5$); \quad
$6 + 5 = 11$ ($6 + 4 = 10$, luego $+1$).
\end{solution}

\begin{solution}{exo:g1:addition:5}
$7 + 8 = 15$ (uno más que $7 + 7 = 14$); \quad
$5 + 6 = 11$ (uno más que $5 + 5$); \quad
$9 + 8 = 17$ (uno menos que $9 + 9 = 18$).
\end{solution}

\begin{solution}{exo:g1:addition:6}
$6 + 4 = 10$; \quad $7 + 5 = 12$; \quad $10 + 7 = 17$.
\end{solution}

\begin{solution}{exo:g1:addition:7}
$8 + 5 = 13$. Hay $13$ personas en el autobús.
\end{solution}

\begin{solution}{exo:g1:addition:8}
Tom tiene $6 + 7 = 13$ autos, Mia tiene $10$: Tom tiene más.
\end{solution}

\begin{solution}{exo:g1:addition:9}
Agrega los diez amigos $4$ y $6$ primero: $4 + 6 = 10$, luego
$10 + 5 = 15$. El total es $15$.
\end{solution}
