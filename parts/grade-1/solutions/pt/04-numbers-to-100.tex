\section*{Capítulo \ref{ch:g1:tens} --- Números até 100}

\begin{solution}{exo:g1:tens:1}
$26$: $2$ dezenas, $6$ unidades. $40$: $4$ dezenas, $0$ unidades. $73$: $7$
dezenas, $3$ unidades. $99$: $9$ dezenas, $9$ unidades.
\end{solution}

\begin{solution}{exo:g1:tens:2}
$52$; \ $80$; \ $19$; \ $100$.
\end{solution}

\begin{solution}{exo:g1:tens:3}
$63$; \ $40$; \ $85$; \ $100$.
\end{solution}

\begin{solution}{exo:g1:tens:4}
$23 + 10 = 33$; \quad $67 + 10 = 77$; \quad $45 - 10 = 35$; \quad
$80 - 10 = 70$.
\end{solution}

\begin{solution}{exo:g1:tens:5}
$10, 20, 30, 40, 50, 60$. \quad $5, 10, 15, 20, 25, 30$. \quad
$12, 14, 16, 18, 20, 22$.
\end{solution}

\begin{solution}{exo:g1:tens:6}
$34 < 43$; \ $58 < 61$; \ $70 > 69$; \ $99 < 100$.
\end{solution}

\begin{solution}{exo:g1:tens:7}
$19 < 41 < 46 < 64 < 91$.
\end{solution}

\begin{solution}{exo:g1:tens:8}
$52$ é $5$ dezenas e $2$ unidades: $5$ cachos completos e $2$ solto
flores.
\end{solution}

\begin{solution}{exo:g1:tens:9}
De $36$, dois passos abaixo é $+20$ ($\to 56$), falta um passo para
$-1$ ($\to 55$). Você pousa $55$. O cálculo é
$36 + 20 - 1$, aquilo é $36 + 19$.
\end{solution}
