\section*{Capítulo \ref{ch:g1:measure} --- Medição e Tempo}

\begin{solution}{exo:g1:measure:1}
As respostas dependem dos lápis; o mais longo chega mais longe quando
eles são combinados em uma extremidade, o mais curto chega ao menos longe.
\end{solution}

\begin{solution}{exo:g1:measure:2}
As respostas variam; cada um é lido colocando o nome da régua $0$ no início
do objeto e lendo todo o centímetro em seu final.
\end{solution}

\begin{solution}{exo:g1:measure:3}
Depois de segunda-feira: terça-feira. Antes de domingo: sábado. Dois dias depois
Sexta-feira: domingo.
\end{solution}

\begin{solution}{exo:g1:measure:4}
Ontem: quarta-feira. Amanhã: sexta-feira. Em três dias: domingo.
\end{solution}

\begin{solution}{exo:g1:measure:5}
No $6$ horas o ponteiro abreviado aponta para o $6$ e a mão longa
aponta para o $12$.
\end{solution}

\begin{solution}{exo:g1:measure:6}
Mão curta $5$: $5$ horas. Mão curta $11$: $11$ horas.
\end{solution}

\begin{solution}{exo:g1:measure:7}
Para $8$, por exemplo $5 + 2 + 1$ ou $2 + 2 + 2 + 2$ (outro correto
existem respostas).
\end{solution}

\begin{solution}{exo:g1:measure:8}
$5 + 2 + 2 + 2 = 11$. Leão tem $11$.
\end{solution}

\begin{solution}{exo:g1:measure:9}
De $3$ até $5$ é $2$: Zoe fica $2$ voltar.
\end{solution}
