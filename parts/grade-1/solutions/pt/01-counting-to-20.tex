\section*{Capítulo \ref{ch:g1:counting} --- Contando até 20}

\begin{solution}{exo:g1:counting:1}
As respostas dependem do nome; duas mãos têm $10$ dedos; dois gatos
tem $8$ pernas.
\end{solution}

\begin{solution}{exo:g1:counting:2}
$7$; \ $12$; \ $20$; \ $0$. Em palavras: quatro; quinze; dezoito.
\end{solution}

\begin{solution}{exo:g1:counting:3}
$8, 9, 10, 11, 12, 13$. Para trás: $15, 14, 13, 12, 11, 10$.
\end{solution}

\begin{solution}{exo:g1:counting:4}
Pouco antes $10$: $9$. Logo depois $16$: $17$. Entre $12$ e
$14$: $13$.
\end{solution}

\begin{solution}{exo:g1:counting:5}
$6 < 9$; \ $14 > 8$; \ $11 < 13$; \ $20 > 19$.
\end{solution}

\begin{solution}{exo:g1:counting:6}
Emparelhamento $6$ círculos com $8$ cruza folhas $2$ cruza sem
parceiro: as cruzes são mais, por $2$.
\end{solution}

\begin{solution}{exo:g1:counting:7}
$2 < 7 < 10 < 12 < 17$.
\end{solution}

\begin{solution}{exo:g1:counting:8}
Segundo: Bob. Quarto: Dan. Dan está no lugar $4$.
\end{solution}

\begin{solution}{exo:g1:counting:9}
Três passos depois $13$: $16$. Três passos antes $13$: $10$.
\end{solution}
