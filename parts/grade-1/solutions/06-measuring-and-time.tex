\section*{Chapter \ref{ch:g1:measure} --- Measuring and Time}

\begin{solution}{exo:g1:measure:1}
Answers depend on the pencils; the longest reaches furthest when
they are matched at one end, the shortest reaches least far.
\end{solution}

\begin{solution}{exo:g1:measure:2}
Answers vary; each is read by putting the ruler's $0$ at the start
of the object and reading the whole centimeter at its end.
\end{solution}

\begin{solution}{exo:g1:measure:3}
After Monday: Tuesday. Before Sunday: Saturday. Two days after
Friday: Sunday.
\end{solution}

\begin{solution}{exo:g1:measure:4}
Yesterday: Wednesday. Tomorrow: Friday. In three days: Sunday.
\end{solution}

\begin{solution}{exo:g1:measure:5}
At $6$ o'clock the short hand points to the $6$ and the long hand
points up to the $12$.
\end{solution}

\begin{solution}{exo:g1:measure:6}
Short hand on $5$: $5$ o'clock. Short hand on $11$: $11$ o'clock.
\end{solution}

\begin{solution}{exo:g1:measure:7}
For $8$, for instance $5 + 2 + 1$ or $2 + 2 + 2 + 2$ (other correct
answers exist).
\end{solution}

\begin{solution}{exo:g1:measure:8}
$5 + 2 + 2 + 2 = 11$. Leo has $11$.
\end{solution}

\begin{solution}{exo:g1:measure:9}
From $3$ up to $5$ is $2$: Zoe gets $2$ back.
\end{solution}
