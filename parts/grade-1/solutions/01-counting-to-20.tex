\section*{Chapter \ref{ch:g1:counting} --- Counting to 20}

\begin{solution}{exo:g1:counting:1}
Answers depend on the name; two hands have $10$ fingers; two cats
have $8$ legs.
\end{solution}

\begin{solution}{exo:g1:counting:2}
$7$; \ $12$; \ $20$; \ $0$. In words: four; fifteen; eighteen.
\end{solution}

\begin{solution}{exo:g1:counting:3}
$8, 9, 10, 11, 12, 13$. Backward: $15, 14, 13, 12, 11, 10$.
\end{solution}

\begin{solution}{exo:g1:counting:4}
Just before $10$: $9$. Just after $16$: $17$. Between $12$ and
$14$: $13$.
\end{solution}

\begin{solution}{exo:g1:counting:5}
$6 < 9$; \ $14 > 8$; \ $11 < 13$; \ $20 > 19$.
\end{solution}

\begin{solution}{exo:g1:counting:6}
Pairing $6$ circles with $8$ crosses leaves $2$ crosses without a
partner: the crosses are more, by $2$.
\end{solution}

\begin{solution}{exo:g1:counting:7}
$2 < 7 < 10 < 12 < 17$.
\end{solution}

\begin{solution}{exo:g1:counting:8}
Second: Bob. Fourth: Dan. Dan is at place $4$.
\end{solution}

\begin{solution}{exo:g1:counting:9}
Three steps after $13$: $16$. Three steps before $13$: $10$.
\end{solution}
