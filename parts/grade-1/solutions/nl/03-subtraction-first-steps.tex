\section*{Hoofdstuk \ref{ch:g1:subtraction} --- Aftrekken: eerste stappen}

\begin{solution}{exo:g1:subtraction:1}
$6 - 2 = 4$; \quad $8 - 5 = 3$; \quad $10 - 4 = 6$.
\end{solution}

\begin{solution}{exo:g1:subtraction:2}
$9 - 3 = 6$; \quad $12 - 2 = 10$; \quad $15 - 4 = 11$.
\end{solution}

\begin{solution}{exo:g1:subtraction:3}
$11 - 8 = 3$ (van $8$ omhoog naar $11$); \quad
$13 - 9 = 4$; \quad $16 - 12 = 4$.
\end{solution}

\begin{solution}{exo:g1:subtraction:4}
$2 + 6 = 8$, \quad $6 + 2 = 8$, \quad $8 - 6 = 2$, \quad
$8 - 2 = 6$.
\end{solution}

\begin{solution}{exo:g1:subtraction:5}
$14 - 6 = 8$ (controle: $8 + 6 = 14$); \quad
$17 - 9 = 8$ (controle: $8 + 9 = 17$).
\end{solution}

\begin{solution}{exo:g1:subtraction:6}
$10 - 3 = 7$; \quad $10 - 5 = 5$; \quad $12 - 0 = 12$.
\end{solution}

\begin{solution}{exo:g1:subtraction:7}
$15 - 8 = 7$. Er zijn nog $7$ kersen over.
\end{solution}

\begin{solution}{exo:g1:subtraction:8}
``Er zwemmen er $5$ weg'': $9 - 5 = 4$ eenden. ``Er komen $5$ ganzen
bij'': $9 + 5 = 14$ vogels. ``Hoeveel jaar ouder'': $9 - 5 = 4$
jaar.
\end{solution}

\begin{solution}{exo:g1:subtraction:9}
Familie van $12$, $7$ en $5$: $12 - 7 = 5$ eieren zijn uitgekomen
(en $5 + 7 = 12$ controleert het).
\end{solution}
