\section*{Hoofdstuk \ref{ch:g1:counting} --- Tellen tot 20}

\begin{solution}{exo:g1:counting:1}
De antwoorden hangen af van de voornaam; twee handen hebben $10$
vingers; twee katten hebben $8$ poten.
\end{solution}

\begin{solution}{exo:g1:counting:2}
$7$; \ $12$; \ $20$; \ $0$. In woorden: vier; vijftien; achttien.
\end{solution}

\begin{solution}{exo:g1:counting:3}
$8, 9, 10, 11, 12, 13$. Achteruit: $15, 14, 13, 12, 11, 10$.
\end{solution}

\begin{solution}{exo:g1:counting:4}
Net v\'{o}\'{o}r $10$: $9$. Net na $16$: $17$. Tussen $12$ en $14$:
$13$.
\end{solution}

\begin{solution}{exo:g1:counting:5}
$6 < 9$; \ $14 > 8$; \ $11 < 13$; \ $20 > 19$.
\end{solution}

\begin{solution}{exo:g1:counting:6}
Paren van $6$ cirkels met $8$ kruisjes laat $2$ kruisjes zonder
partner over: de kruisjes zijn meer, met $2$.
\end{solution}

\begin{solution}{exo:g1:counting:7}
$2 < 7 < 10 < 12 < 17$.
\end{solution}

\begin{solution}{exo:g1:counting:8}
Tweede: Bob. Vierde: Dan. Dan staat op plaats $4$.
\end{solution}

\begin{solution}{exo:g1:counting:9}
Drie stappen na $13$: $16$. Drie stappen v\'{o}\'{o}r $13$: $10$.
\end{solution}
