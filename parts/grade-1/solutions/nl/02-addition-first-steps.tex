\section*{Hoofdstuk \ref{ch:g1:addition} --- Optellen: eerste stappen}

\begin{solution}{exo:g1:addition:1}
$4 + 3 = 7$; \quad $5 + 2 = 7$; \quad $6 + 4 = 10$.
\end{solution}

\begin{solution}{exo:g1:addition:2}
$3 + 8 = 11$ (begin bij $8$, tel er $3$ bij); \quad
$2 + 12 = 14$; \quad $4 + 9 = 13$.
\end{solution}

\begin{solution}{exo:g1:addition:3}
Tientalvrienden: $7 \to 3$; \ $4 \to 6$; \ $9 \to 1$; \ $5 \to 5$;
\ $1 \to 9$.
\end{solution}

\begin{solution}{exo:g1:addition:4}
$9 + 4 = 13$ ($9 + 1 = 10$, daarna $+3$); \quad
$8 + 7 = 15$ ($8 + 2 = 10$, daarna $+5$); \quad
$6 + 5 = 11$ ($6 + 4 = 10$, daarna $+1$).
\end{solution}

\begin{solution}{exo:g1:addition:5}
$7 + 8 = 15$ (\'{e}\'{e}n meer dan $7 + 7 = 14$); \quad
$5 + 6 = 11$ (\'{e}\'{e}n meer dan $5 + 5$); \quad
$9 + 8 = 17$ (\'{e}\'{e}n minder dan $9 + 9 = 18$).
\end{solution}

\begin{solution}{exo:g1:addition:6}
$6 + 4 = 10$; \quad $7 + 5 = 12$; \quad $10 + 7 = 17$.
\end{solution}

\begin{solution}{exo:g1:addition:7}
$8 + 5 = 13$. Er zitten $13$ mensen in de bus.
\end{solution}

\begin{solution}{exo:g1:addition:8}
Tom heeft $6 + 7 = 13$ auto's, Mia heeft $10$: Tom heeft er meer.
\end{solution}

\begin{solution}{exo:g1:addition:9}
Tel eerst de tientalvrienden $4$ en $6$ op: $4 + 6 = 10$, daarna
$10 + 5 = 15$. Het totaal is $15$.
\end{solution}
