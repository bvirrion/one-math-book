\section*{Hoofdstuk \ref{ch:g1:tens} --- Getallen tot 100}

\begin{solution}{exo:g1:tens:1}
$26$: $2$ tientallen, $6$ eenheden. $40$: $4$ tientallen, $0$
eenheden. $73$: $7$ tientallen, $3$ eenheden. $99$: $9$ tientallen,
$9$ eenheden.
\end{solution}

\begin{solution}{exo:g1:tens:2}
$52$; \ $80$; \ $19$; \ $100$.
\end{solution}

\begin{solution}{exo:g1:tens:3}
$63$; \ $40$; \ $85$; \ $100$.
\end{solution}

\begin{solution}{exo:g1:tens:4}
$23 + 10 = 33$; \quad $67 + 10 = 77$; \quad $45 - 10 = 35$; \quad
$80 - 10 = 70$.
\end{solution}

\begin{solution}{exo:g1:tens:5}
$10, 20, 30, 40, 50, 60$. \quad $5, 10, 15, 20, 25, 30$. \quad
$12, 14, 16, 18, 20, 22$.
\end{solution}

\begin{solution}{exo:g1:tens:6}
$34 < 43$; \ $58 < 61$; \ $70 > 69$; \ $99 < 100$.
\end{solution}

\begin{solution}{exo:g1:tens:7}
$19 < 41 < 46 < 64 < 91$.
\end{solution}

\begin{solution}{exo:g1:tens:8}
$52$ is $5$ tientallen en $2$ eenheden: $5$ volle bossen en $2$
losse bloemen.
\end{solution}

\begin{solution}{exo:g1:tens:9}
Vanaf $36$ is twee stappen omlaag $+20$ ($\to 56$), \'{e}\'{e}n stap
links is $-1$ ($\to 55$). Je landt op $55$. De berekening is
$36 + 20 - 1$, dus $36 + 19$.
\end{solution}
