\chapter{Adição: primeiros passos}\label{ch:g1:addition}

Três bolinhas vermelhas e duas bolinhas azuis: juntas, quantas? Colocando
juntos e contando o todo é \emph{adicionando} --- o primeiro
operação, com seus dois famosos sinais $+$ e $=$.

\section{Juntando}

\begin{definition}[Adição]\label{def:g1:addition:def}
\emph{Adicionando} significa juntar e contar tudo. Nós
escrever
\[
3 + 2 = 5
\]
e leia: ``três mais dois é igual a cinco''. A resposta de uma adição
é chamado de \emph{soma}\index{soma}.
\end{definition}

\begin{example}[A ordem não importa]\label{ex:g1:addition:order}
$2 + 3$ e $3 + 2$ junte as mesmas bolinhas de gude: ambas fazem $5$.
Para calcular $2 + 9$, é mais rápido começar com o número maior:
diga ``nove'', então conte com dois --- ``dez, onze''. Então
$2 + 9 = 11$.
\end{example}

\section{Os dez amigos}

\begin{example}[Complementos para 10]\label{ex:g1:addition:friends}
Dois números que fazem $10$ juntos são ``dez amigos'':
\[
1 + 9, \quad 2 + 8, \quad 3 + 7, \quad 4 + 6, \quad 5 + 5 .
\]
Dez dedos mostram todos eles: levante $3$ dedos, e o $7$ dobrado
aqueles são os amigos. Conhecer os dez amigos de cor faz com que muitos
adições rapidamente.
\end{example}

  \node[right, font=\small, align=left] at (5.6,1.0)

     $3$ filled, $7$ empty: $3 + 7 = 10$};
\end{tikzpicture}

{\small O quadro de dez mostra cada par de dez amigos de relance:
o que está preenchido e o que falta sempre faça $10$ junto.}
\end{omfigure}

\begin{method}[Cruzando os dez]\label{met:g1:addition:crossten}
Para calcular $8 + 5$:
\begin{enumerate}
  \item tirar do $5$ o que $8$ precisa alcançar $10$: aquilo é
        $2$ (seus dez amigos);
  \item $8 + 2 = 10$, e $3$ permanece do $5$;
  \item $10 + 3 = 13$. Então $8 + 5 = 13$.
\end{enumerate}
Passando $10$ fica difícil \omterm{def:g1:addition:def}{soma} em dois fáceis.
\end{method}

\begin{example}[Duplas]\label{ex:g1:addition:doubles}
Vale a pena decorar as duplas:
\[
1+1=2, \quad 2+2=4, \quad 3+3=6, \quad 4+4=8, \quad 5+5=10,
\]
\[
6+6=12, \quad 7+7=14, \quad 8+8=16, \quad 9+9=18, \quad 10+10=20 .
\]
Quase duplos são gratuitos: $6 + 7$ é um a mais que $6 + 6$, então $13$.
\end{example}

\section{Pequenos problemas}

\begin{example}\label{ex:g1:addition:problem}
``Lea tem $7$ adesivos. Ela consegue $6$ mais. Quantos agora?
Obter mais significa adicionar: $7 + 6 = 13$ (até dez:
$7 + 3 = 10$, então $+3$). Léia tem $13$ adesivos. Termine sempre com
uma pequena frase!
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g1:addition:1}
Desenhe as bolinhas e calcule: $4 + 3$; \quad $5 + 2$; \quad
$6 + 4$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:addition:2}
Calcule começando pelo número maior: $3 + 8$; \quad
$2 + 12$; \quad $4 + 9$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:addition:3}
Dê os dez amigos de: $7$; \ $4$; \ $9$; \ $5$; \ $1$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:addition:4}
Calcule os dez (\cref{met:g1:addition:crossten}):
$9 + 4$; \quad $8 + 7$; \quad $6 + 5$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:addition:5}
Recite as duplas e calcule usando-as: $7 + 8$; \quad
$5 + 6$; \quad $9 + 8$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:addition:6}
Complete os buracos:
\[
6 + \;?\; = 10, \qquad
\;?\; + 5 = 12, \qquad
10 + \;?\; = 17 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:addition:7}
``No ônibus há $8$ crianças; $5$ os adultos entram. Como
muitas pessoas estão no ônibus?'' Escreva o acréscimo e a resposta
sentença.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:addition:8}
Tom tem $6$ carros vermelhos e $7$ carros azuis. Mia tem $10$ carros. Quem tem
mais carros?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g1:addition:9}
Mostra de três dados $4$, $6$ e $5$. Qual é o total? (Adicione dois dados
primeiro --- escolha os dois que formam dez amigos!)
\end{exercise}
