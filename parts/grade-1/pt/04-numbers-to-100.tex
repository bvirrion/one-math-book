\chapter{Números até 100}\label{ch:g1:tens}

Contar uma grande pilha, uma por uma, leva uma eternidade. O grande truque:
agrupar os objetos em pacotes de \emph{dez}e conte os pacotes. Isto
é como os números até $100$ são construídos --- e por que eles são
escrito com dois dígitos.

\section{Pacotes de dez}

\begin{definition}[Dezenas e unidades]\label{def:g1:tens:def}
UM \emph{dez} é um pacote de dez unidades. O número $34$ significa $3$
dezenas e $4$ unidades:
\[
34 = 10 + 10 + 10 + 4 .
\]
O dígito da esquerda conta os pacotes, o dígito da direita as unidades soltas.
\end{definition}

\end{tikzpicture}

{\small Trinta e quatro varas: três feixes de dez (cada um amarrado com um
banda) e quatro varetas soltas. Não há necessidade de contar até $34$ um por um!}
\end{omfigure}

\begin{example}[Lendo os números]\label{ex:g1:tens:reading}
$20$ é duas dezenas (vinte), $30$ três dezenas (trinta)\,\dots\ até
$90$ (noventa) e $100$ (cem: dez dezenas!). Entre:
$47$ lê ``quarenta e sete'' --- quarenta para o $4$ dezenas, sete para o
unidades.
\end{example}

\end{omfigure}

\begin{example}[Mesmos dígitos, números diferentes]\label{ex:g1:tens:place}
$27$ e $72$ usam os mesmos dígitos, mas são muito diferentes: $27$ é
$2$ dezenas e $7$ unidades; $72$ é $7$ dezenas e $2$ unidades --- muito
maior! O \emph{lugar} de um dígito decide o que conta.
\end{example}

\section{A grade cem}

\begin{omfigure}
\begin{tikzpicture}[scale=0.52]
  \foreach \r in {0,...,9} {
    \foreach \c in {0,...,9} {
      \pgfmathtruncatemacro{\n}{\r*10+\c+1}
      \draw[gray!50] (\c,-\r) +(-0.5,-0.5) rectangle +(0.5,0.5);
      \node[font=\footnotesize] at (\c,-\r) {\n};
    }
  }
  \draw[omThm, very thick] (3,-4) +(-0.5,-0.5) rectangle +(0.5,0.5);
  \draw[omDef, very thick] (3,-5) +(-0.5,-0.5) rectangle +(0.5,0.5);
\end{tikzpicture}

{\small A grade das centenas: cada linha é uma dezena. Um passo certo é
$+1$; um passo \emph{abaixo} é $+10$ (do vermelho $44$ direto para baixo
para o azul $54$).}
\end{omfigure}

\begin{method}[Movendo-se na grade]\label{met:g1:tens:grid}
\begin{enumerate}
  \item $+1$ / $-1$: um passo para a direita/esquerda;
  \item $+10$ / $-10$: um passo para baixo/para cima --- apenas o dígito das dezenas
        mudanças: $37 + 10 = 47$;
  \item contando por dezenas: $10, 20, 30, \dots $ --- segue o último
        coluna da grade para baixo.
\end{enumerate}
\end{method}

\begin{example}[Contando por 2s e 5s]\label{ex:g1:tens:skip}
Por $2$ é: $2, 4, 6, 8, 10, 12, \dots $ (os números pares). Por $5$ é:
$5, 10, 15, 20, 25, \dots $ --- eles terminam em $5$ ou $0$ e faça um
belo padrão na grade. A contagem pulada prepara a multiplicação
(\cref{ch:g2:mult}).
\end{example}

\section{Comparando números maiores}

\begin{method}[Comparando com dezenas]\label{met:g1:tens:compare}
Mais dezenas de vitórias: $61 > 58$ porque $6$ batida de dezenas $5$ dezenas ---
mesmo que $8$ é mais do que $1$! Somente quando as dezenas são iguais é que
as unidades decidem: $43 < 47$.
\end{method}

\begin{example}\label{ex:g1:tens:compare}
Ordem $35$, $53$, $29$: dezenas primeiro ($2$, $3$, $5$):
\[
29 < 35 < 53 .
\]
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g1:tens:1}
Quantas dezenas e unidades tem: $26$; \ $40$; \ $73$; \ $99$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:tens:2}
Escreva o número: $5$ dezenas e $2$ unidades; \ $8$ dezenas; \ $1$ dez e
$9$ unidades; \ $10$ dezenas.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:tens:3}
Escreva em algarismos: sessenta e três; \ quarenta; \ oitenta e cinco; \ um
cem.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:tens:4}
Calcule movendo-se na grade das centenas: $23 + 10$; \quad $67 + 10$;
\quad $45 - 10$; \quad $80 - 10$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:tens:5}
Continuar: $10, 20, 30, ?, ?, ?$. \quad
$5, 10, 15, ?, ?, ?$. \quad $12, 14, 16, ?, ?, ?$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:tens:6}
Copie e complete com $<$ ou $>$:
\[
34 \;?\; 43, \qquad
58 \;?\; 61, \qquad
70 \;?\; 69, \qquad
99 \;?\; 100 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:tens:7}
Ordene do menor para o maior: $46$; \ $19$; \ $64$; \ $91$; \
$41$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:tens:8}
Um jardineiro faz pacotes $52$ flores em cachos de dez. Quantos completos
cachos e quantas flores soltas?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g1:tens:9}
Na grade cem, comece em $36$; desça dois passos e um passo
esquerda. Onde você pousa? Qual cálculo você acabou de fazer?
\end{exercise}
