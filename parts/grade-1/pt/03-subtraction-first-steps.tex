\chapter{Subtração: primeiros passos}\label{ch:g1:subtraction}

Sete balões, dois voam para longe: quantos sobraram? Tirar é
\emph{subtraindo}, escrito com o sinal $-$. E a subtração tem um
segunda face: também responde ``quantos mais?''.

\section{Tirando}

\begin{definition}[Subtração]\label{def:g1:subtraction:def}
\emph{Subtraindo} significa tirar e contar o que resta. Nós
escrever
\[
7 - 2 = 5
\]
e leia: ``sete menos dois é igual a cinco''.
\end{definition}

\begin{example}[Contando regressivamente]\label{ex:g1:subtraction:countback}
Para um pequeno take-away, conte regressivamente na linha: $11 - 3$: diga
``onze'', depois três passos para trás --- ``dez, nove, oito''. Então
$11 - 3 = 8$.
\end{example}

\section{Quantos mais?}

\begin{example}[A diferença]\label{ex:g1:subtraction:difference}
``Nina tem $9$ bolinhas de gude e Tom tem $6$. Quantos mais Nina tem?
Nada é subtraído, mas é uma subtração: junte as bolinhas,
e conte as bolas de gude de Nina sem parceiro:
\[
9 - 6 = 3 .
\]
Nina tem $3$ mais. A resposta de uma subtração é chamada de
\emph{diferença}\index{diferença}.
\end{example}

\begin{method}[Contando]\label{met:g1:subtraction:countup}
Para encontrar um \omterm{ex:g1:subtraction:difference}{diferença}, contando \emph{acima} geralmente é mais fácil:
para $12 - 9$, comece em $9$ e suba para $12$: ``dez, onze,
doze '' --- três etapas. Então $12 - 9 = 3$. (Suba através $10$ para
lacunas maiores: $14 - 8$: de $8$ para $10$ é $2$, de $10$ para $14$
é $4$: \omterm{ex:g1:subtraction:difference}{diferença} $6$.)
\end{method}

\section{Adição e subtração são gêmeas}

\begin{example}[Famílias de fatos]\label{ex:g1:subtraction:family}
Os números $3$, $4$ e $7$ formar uma pequena família:
\[
3 + 4 = 7, \qquad
4 + 3 = 7, \qquad
7 - 4 = 3, \qquad
7 - 3 = 4 .
\]
Conhecer um fato da família dá de graça os outros três.
A subtração desfaz a adição: depois $+4$, fazendo $-4$ traz você de volta.
\end{example}

\begin{example}[Verificando]\label{ex:g1:subtraction:check}
Calculado $13 - 5 = 8$? Verifique com a adição dupla:
$8 + 5 = 13$. \checkmark\ Uma subtração está correta exatamente ao adicionar
volta funciona.
\end{example}

\begin{example}[Qual operação?]\label{ex:g1:subtraction:choose}
``$6$ pássaros no arame, $4$ voar para longe'': tirar, $6 - 4 = 2$.
``$6$ pássaros, então $4$ mais chegam'': juntos, $6 + 4 = 10$.
``$6$ pardais e $4$ robins: quantos pardais a mais?'':
\omterm{ex:g1:subtraction:difference}{diferença}, $6 - 4 = 2$. Desenhe a história quando não tiver certeza!
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g1:subtraction:1}
Desenhe, risque e calcule: $6 - 2$; \quad $8 - 5$; \quad
$10 - 4$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:subtraction:2}
Conte regressivamente para calcular: $9 - 3$; \quad $12 - 2$; \quad
$15 - 4$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:subtraction:3}
Conte para encontrar o \omterm{ex:g1:subtraction:difference}{diferença}: $11 - 8$; \quad $13 - 9$; \quad
$16 - 12$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:subtraction:4}
Escreva os quatro fatos da família de $2$, $6$ e $8$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:subtraction:5}
Calcule e verifique com a adição dupla: $14 - 6$; \quad
$17 - 9$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:subtraction:6}
Complete os buracos:
\[
10 - \;?\; = 7, \qquad
\;?\; - 5 = 5, \qquad
12 - \;?\; = 12 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:subtraction:7}
``Existem $15$ cerejas na tigela. As crianças comem $8$. Como
sobraram muitos?'' Escreva a subtração e a sentença de resposta.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:subtraction:8}
Para cada história, escolha $+$ ou $-$, então calcule: ``$9$ patos no
lagoa, $5$ nadar para longe''; ``$9$ patos, $5$ chegam gansos'';
``Ali é $9$, sua irmã é $5$: quantos anos mais velha é Ali?''.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g1:subtraction:9}
Mamãe pato tinha $12$ ovos. Alguns eclodiram; agora $12 - ?$ os ovos permanecem
e ela conta $7$ ovos não eclodidos. Quantos ovos eclodiram? (Qual
família de fato é essa?)
\end{exercise}
