\chapter{Formas e Espaço}\label{ch:g1:shapes}

As janelas são retângulos, as rodas são círculos, os telhados escondem triângulos:
as formas estão por toda parte. Este capítulo aprende a reconhecê-los
contando lados e cantos, e dizer com precisão \emph{onde} coisas
são: esquerda, direita, acima, abaixo, entre.

\section{As quatro formas básicas}

\begin{definition}[Quadrado, retângulo, triângulo, círculo]\label{def:g1:shapes:def}
\begin{itemize}
  \item o \emph{triângulo}\index{triângulo} tem $3$ lados e $3$ cantos;
  \item o \emph{quadrado}\index{quadrado} tem $4$ lados iguais e $4$ cantos;
  \item o \emph{retângulo}\index{retângulo} tem $4$ lados e $4$ cantos, com
        lados opostos iguais (um quadrado esticado);
  \item o \emph{círculo}\index{círculo} é perfeitamente redondo: sem lados, sem cantos.
\end{itemize}
\end{definition}

\begin{example}[Formas inclinadas ainda são formas]\label{ex:g1:shapes:tilted}
Um quadrado em seu canto ainda é um quadrado --- quatro iguais
lados, quatro cantos --- mesmo que pareça uma pipa. Transformando um
a forma não muda o que é.
\end{example}

\begin{example}[Caça às formas]\label{ex:g1:shapes:hunt}
Na sala de aula: o quadro é um retângulo, o relógio um círculo, um
guardanapo dobrado pode ser um triângulo, algumas janelas são quadradas. Verdadeiro
os objetos não têm formas perfeitas, mas seus rostos estão próximos deles
(sólidos e suas faces retornam em \cref{ch:g2:shapes}).
\end{example}

\section{Cadê?}

\begin{definition}[Palavras de posição]\label{def:g1:shapes:position}
Para dizer onde algo está: \emph{esquerda} / \emph{certo},
\emph{acima} / \emph{abaixo}, \emph{sobre} / \emph{sob},
\emph{em frente de} / \emph{atrás}, \emph{entre},
\emph{dentro} / \emph{fora}.
\end{definition}

\section{Caminhos em uma grade}

\begin{method}[Descrevendo um caminho]\label{met:g1:shapes:path}
Em papel quadriculado, um caminho é uma lista de etapas:
$\to $ (um quadrado à direita), $\leftarrow $ (esquerda), $\uparrow $ (acima),
$\downarrow $ (abaixo).
\begin{enumerate}
  \item Para seguir um caminho: comece no quadrado marcado e obedeça às
        setas uma por uma;
  \item para descrever um caminho: percorra-o em sua cabeça e escreva o
        setas em ordem;
  \item dois caminhos diferentes podem unir os mesmos dois quadrados --- alguns
        mais curto, alguns mais longo.
\end{enumerate}
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g1:shapes:1}
Para cada forma, conte os lados e cantos e nomeie-a: uma forma
com $3$ cantos; formato redondo sem canto; uma forma com $4$
lados iguais.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:shapes:2}
Encontre em casa: dois retângulos, um círculo, um triângulo, um quadrado.
(Portas? Placas? Sinais de trânsito?)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:shapes:3}
Verdadeiro ou falso? ``Um quadrado virado na esquina deixa de ser um
quadrado. '' Explique com \cref{ex:g1:shapes:tilted}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:shapes:4}
Desenhe em papel quadriculado: um quadrado de $3$ quadrados por $3$; um retângulo de
$5$ por $2$; um triângulo usando uma diagonal da grade.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:shapes:5}
Coloque três brinquedos seguidos. Descreva a linha com palavras de posição:
qual fica entre os outros? Qual está à esquerda?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:shapes:6}
Desenhe um círculo. Desenhe uma cruz \emph{dentro} isso, uma estrela
\emph{fora} isso, e um ponto \emph{sobre} o próprio círculo.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:shapes:7}
Em papel quadriculado, marque um quadrado inicial. Seguir:
$\to\ \uparrow\ \to\ \uparrow\ \to $. Marque a chegada. Então escreva
o caminho que vai direto ao início.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:shapes:8}
Quantos cantos no total: dois triângulos e um quadrado? Um círculo
e dois retângulos?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g1:shapes:9}
Pegar $12$ partidas. Você pode delinear um quadrado usando \emph{todos} de
eles, com fósforos inteiros e o mesmo número de cada lado? Quantos
partidas de cada lado? Você poderia fazer isso com $10$ partidas?
\end{exercise}
