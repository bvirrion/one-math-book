\chapter{Medição e Tempo}\label{ch:g1:measure}

Qual lápis é mais longo? Que dia é hoje? Quanto custa um doce?
Este capítulo compara comprimentos e os mede em centímetros,
ordena os dias da semana, lê horas inteiras no relógio,
e conta moedas pequenas.

\section{Mais longo e mais curto}

\begin{method}[Comparando comprimentos]\label{met:g1:measure:compare}
Para comparar dois comprimentos, alinhe-os com \emph{uma extremidade combinada}:
\begin{enumerate}
  \item coloque os dois objetos lado a lado, começando na mesma linha;
  \item aquele que vai mais longe é \emph{mais longo}, o outro é
        \emph{mais curto};
  \item se ambos chegarem ao mesmo lugar, eles são os \emph{mesmo
        comprimento}.
\end{enumerate}
Combinar o início é todo o segredo --- comparar com o torto
começa a enganar os olhos.
\end{method}

\begin{definition}[O centímetro]\label{def:g1:measure:cm}
Para dizer \emph{como} longo, medimos com uma régua em
\emph{centímetros} (cm). Coloque o $0$ do governante no início de
o objeto e leia o número no final.
\end{definition}

\section{Os dias da semana}

\begin{example}[Os sete dias]\label{ex:g1:measure:days}
A semana tem $7$ dias, sempre na mesma ordem:
\[
\text{Monday, Tuesday, Wednesday, Thursday, Friday, Saturday,
Sunday.}
\]
Depois de domingo, segunda-feira começa novamente. \emph{Ontem} é o dia
antes, \emph{amanhã} no dia seguinte: se hoje for quarta-feira,
ontem foi terça e amanhã é quinta.
\end{example}

\section{Horas inteiras no relógio}

\begin{method}[Lendo uma hora inteira]\label{met:g1:measure:clock}
Quando o ponteiro longo aponta diretamente para o $12$, é um todo
hora: leia o número o \emph{curto} mão aponta para. Mão curta
o $3$, mão longa no $12$: isso é $3$ horas.
\end{method}

\section{Moedas pequenas}

\begin{example}[Fazendo uma quantia]\label{ex:g1:measure:coins}
Com moedas de $1$, $2$ e $5$, muitas quantias podem ser feitas em vários
maneiras. Para fazer $7$:
\[
5 + 2, \qquad
5 + 1 + 1, \qquad
2 + 2 + 2 + 1 .
\]
As moedas maiores primeiro geralmente são a maneira mais rápida, usando o
menor número de moedas.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g1:measure:1}
Alinhe três lápis combinados em uma das extremidades. Qual é o mais longo?
Qual é o mais curto?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:measure:2}
Meça com sua régua: o lado longo deste livro; seu pequeno
dedo; uma colher. Responda em centímetros inteiros.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:measure:3}
Nomeie o dia: depois de segunda-feira; antes de domingo; dois dias depois de sexta-feira.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:measure:4}
Hoje é quinta-feira. Que dia foi ontem? Que dia é amanhã?
Que dia é daqui a três dias?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:measure:5}
Desenhe um relógio mostrando $6$ horas. Onde está a abreviação? Onde está
a mão longa?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:measure:6}
Leia a hora inteira: ponteiro abreviado $5$, mão longa $12$; curto
mãos à obra $11$, mão longa $12$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:measure:7}
Fazer $8$ com moedas de $1$, $2$ e $5$, de duas maneiras diferentes.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:measure:8}
Leão tem um $5$ moeda e três $2$ moedas. Quanto dinheiro ele tem
completamente?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g1:measure:9}
Um doce custa $3$. Zoe paga com um $5$ moeda. Quanta mudança faz
ela voltou? (Contar a partir de $3$ para $5$.)
\end{exercise}
