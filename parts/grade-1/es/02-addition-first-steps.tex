\chapter{Adición: primeros pasos}\label{ch:g1:addition}

Tres canicas rojas y dos canicas azules: juntas, ¿cuántas? poniendo
juntos y contando el total es \emph{añadiendo} --- el primero
operación, con sus dos famosos signos $+$ y $=$.

\section{Montaje}

\begin{definition}[Adición]\label{def:g1:addition:def}
\emph{Añadiendo} significa juntar y contar todo. Nosotros
escribir
\[
3 + 2 = 5
\]
y lea: ``tres más dos son cinco''. La respuesta de una suma.
se llama \emph{suma}\index{suma}.
\end{definition}

\begin{example}[El orden no importa]\label{ex:g1:addition:order}
$2 + 3$ y $3 + 2$ juntan las mismas canicas: ambas forman $5$.
Para calcular $2 + 9$, es más rápido comenzar desde el número mayor:
diga "nueve", luego cuente con dos --- "diez, once". entonces
$2 + 9 = 11$.
\end{example}

\section{los diez amigos}

\begin{example}[Complementos de 10]\label{ex:g1:addition:friends}
Dos números que juntos forman $10$ son "diez amigos":
\[
1 + 9, \quad 2 + 8, \quad 3 + 7, \quad 4 + 6, \quad 5 + 5 .
\]
Diez dedos los muestran todos: levanta los dedos $3$ y el $7$ doblado
los unos son el amigo. Saberse de memoria a los diez amigos hace que muchos
adiciones rápidamente.
\end{example}

\begin{method}[Cruzando la decena]\label{met:g1:addition:crossten}
Para calcular $8 + 5$:
\begin{enumerate}
  \item toma del $5$ lo que el $8$ necesita para llegar al $10$: es decir
        $2$ (su diez amigo);
  \item $8 + 2 = 10$ y $3$ permanecen del $5$;
  \item $10 + 3 = 13$. Entonces $8 + 5 = 13$.
\end{enumerate}
Pasar por $10$ convierte un \omterm{def:g1:addition:def}{suma} difícil en dos fáciles.
\end{method}

\begin{example}[dobles]\label{ex:g1:addition:doubles}
Vale la pena aprenderse los dobles de memoria:
\[
1+1=2, \quad 2+2=4, \quad 3+3=6, \quad 4+4=8, \quad 5+5=10,
\]
\[
6+6=12, \quad 7+7=14, \quad 8+8=16, \quad 9+9=18, \quad 10+10=20 .
\]
Los casi dobles vienen gratis: $6 + 7$ es uno más que $6 + 6$, por lo que $13$.
\end{example}

\section{Pequeños problemas}

\begin{example}\label{ex:g1:addition:problem}
``Lea tiene pegatinas $7$. Ella obtiene $6$ más. ¿Cuántos ahora?
Obtener más significa agregar: $7 + 6 = 13$ (hasta diez:
$7 + 3 = 10$, luego $+3$). Lea tiene pegatinas $13$. Termina siempre con
una pequeña frase!
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g1:addition:1}
Dibuja las canicas y calcula: $4 + 3$; \quad$5 + 2$; \quad
$6 + 4$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:addition:2}
Calcule comenzando desde el número mayor: $3 + 8$; \quad
$2 + 12$; \quad$4 + 9$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:addition:3}
Dale los diez amigos de: $7$; $4$; $9$; $5$; $1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:addition:4}
Calcular hasta la decena (\cref{met:g1:addition:crossten}):
$9 + 4$; \quad$8 + 7$; \quad $6 + 5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:addition:5}
Recita los dobles, luego calcula usándolos: $7 + 8$; \quad
$5 + 6$; \quad $9 + 8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:addition:6}
Completa los agujeros:
\[
6 + \;?\; = 10, \qquad
\;?\; + 5 = 12, \qquad
10 + \;?\; = 17 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:addition:7}
``En el autobús hay $8$ niños; $5$ los adultos suben. ¿Cómo
¿Hay mucha gente en el autobús?'' Escribe la suma y la respuesta.
frase.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:addition:8}
Tom tiene autos rojos $6$ y autos azules $7$. Mia tiene autos $10$. quien tiene
¿más autos?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g1:addition:9}
Tres dados muestran $4$, $6$ y $5$. ¿Cuál es el total? (Agrega dos dados
primero --- ¡elige los dos que hacen diez amigos!)
\end{exercise}
