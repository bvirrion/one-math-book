\chapter{Formas y espacio}\label{ch:g1:shapes}

Las ventanas son rectángulos, las ruedas son círculos, los techos esconden triángulos:
Las formas están por todas partes. Este capítulo aprende a reconocerlos
contar lados y esquinas, y decir con precisión \emph{dónde} cosas
son: izquierda, derecha, arriba, abajo, entre.

\section{Las cuatro formas básicas}

\begin{definition}[Cuadrado, rectángulo, triángulo, círculo]\label{def:g1:shapes:def}
\begin{itemize}
  \item el \emph{triángulo}\index{triángulo} tiene lados $3$ y esquinas $3$;
  \item el \emph{cuadrado}\index{cuadrado} tiene $4$ lados iguales y $4$ esquinas;
  \item el \emph{rectángulo}\index{rectángulo} tiene lados $4$ y esquinas $4$, con
        lados opuestos iguales (un cuadrado estirado);
  \item el \emph{círculo}\index{círculo} es perfectamente redondo: sin lados ni esquinas.
\end{itemize}
\end{definition}

\begin{example}[Las formas inclinadas siguen siendo formas.]\label{ex:g1:shapes:tilted}
Un cuadrado que está en su esquina sigue siendo un cuadrado: cuatro iguales.
lados, cuatro esquinas --- incluso si parece una cometa. Girando un
la forma no cambia lo que es.
\end{example}

\begin{example}[Búsqueda de formas]\label{ex:g1:shapes:hunt}
En el aula: el tablero es un rectángulo, el reloj un círculo, un
la servilleta doblada puede ser un triángulo, algunas ventanas son cuadrados. reales
Los objetos no tienen formas perfectas, pero sus caras están cerca de ellos.
(los sólidos y sus caras regresan en \cref{ch:g2:shapes}).
\end{example}

\section{¿Dónde está?}

\begin{definition}[Palabras de posición]\label{def:g1:shapes:position}
Para decir dónde está algo: \emph{izquierda} / \emph{bien},
\emph{arriba} / \emph{abajo}, \emph{on} / \emph{bajo},
\emph{delante de} / \emph{detrás}, \emph{entre},
\emph{adentro} / \emph{afuera}.
\end{definition}

\section{Caminos en una cuadrícula}

\begin{method}[Describiendo un camino]\label{met:g1:shapes:path}
En papel cuadriculado, un camino es una lista de pasos:
$\to$ (un cuadrado a la derecha), $\leftarrow$ (izquierda), $\uparrow$ (arriba),
$\downarrow$ (abajo).
\begin{enumerate}
  \item Para seguir un camino: comience en el cuadrado marcado y obedezca las
        flechas una por una;
  \item para describir un camino: recorrelo mentalmente y escribe el
        flechas en orden;
  \item dos caminos diferentes pueden unir los mismos dos cuadrados --- algunos
        más cortos, algunos más largos.
\end{enumerate}
\end{method}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g1:shapes:1}
Para cada forma, cuenta los lados y las esquinas y nómbrala: una forma
con esquinas $3$; una forma redonda sin esquinas; una forma con $4$
lados iguales.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:2}
Encuentra en casa: dos rectángulos, un círculo, un triángulo, un cuadrado.
(¿Puertas? ¿Placas? ¿Señales de tráfico?)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:3}
¿Verdadero o falso? ``Un cuadrado que gira en su esquina deja de ser un
cuadrado." Explique con \cref{ex:g1:shapes:tilted}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:4}
Dibuja en papel cuadriculado: un cuadrado de $3$ cuadrados por $3$; un rectángulo de
$5$ por $2$; un triángulo usando una diagonal de la cuadrícula.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:5}
Coloque tres juguetes en fila. Describe la fila con palabras de posición:
cual esta entre los demas? ¿Cuál está a la izquierda?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:6}
Dibuja un círculo. Dibuja una cruz \emph{adentro}, una estrella.
\emph{afuera} y un punto \emph{on} el círculo mismo.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:7}
En papel cuadriculado, marque un cuadrado inicial. Seguir:
$\to\ \uparrow\ \to\ \uparrow\ \to$. Marca la llegada. Entonces escribe
el camino que regresa directamente al inicio.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:8}
¿Cuántas esquinas en total: dos triángulos y un cuadrado? un circulo
y dos rectángulos?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g1:shapes:9}
Tome coincidencias $12$. ¿Puedes delinear un cuadrado usando \emph{todo} de
ellos, con cerillas enteras y el mismo número en cada lado? How many
matches per side? ¿Podrías hacerlo con coincidencias $10$?
\end{exercise}
