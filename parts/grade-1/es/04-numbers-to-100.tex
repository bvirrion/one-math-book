\chapter{números hasta 100}\label{ch:g1:tens}

Contar una gran pila, uno por uno, lleva una eternidad. El gran truco:
agrupa los objetos en paquetes de \emph{diez} y cuenta los paquetes. esto
Así es como se construyen los números hasta $100$ --- y por qué se
escrito con dos dígitos.

\section{paquetes de diez}

\begin{definition}[Decenas y unidades]\label{def:g1:tens:def}
Un \emph{diez} es un paquete de diez unidades. El número $34$ significa $3$
decenas y unidades $4$:
\[
34 = 10 + 10 + 10 + 4 .
\]
El dígito de la izquierda cuenta los paquetes, el dígito de la derecha las unidades sueltas.
\end{definition}

\begin{example}[leyendo los numeros]\label{ex:g1:tens:reading}
$20$ es dos decenas (veinte), $30$ tres decenas (treinta)\,\dots\ hasta
$90$ (noventa) y $100$ (cien: ¡diez decenas!). En el medio:
$47$ lee ``cuarenta y siete'' --- cuarenta para las decenas $4$, siete para las
unidades.
\end{example}

\begin{example}[Mismos dígitos, diferentes números.]\label{ex:g1:tens:place}
$27$ y $72$ usan los mismos dígitos pero son muy diferentes: $27$ es
$2$ decenas y $7$ unidades; $72$ es $7$ decenas y $2$ unidades --- mucho
¡más grande! El \emph{lugar} de un dígito decide lo que cuenta.
\end{example}

\section{La cuadrícula de cien}

  \draw[omDef, very thick] (3,-5) +(-0.5,-0.5) rectangle +(0.5,0.5);
\end{tikzpicture}

{\small La cuadrícula de cien: cada fila es una decena. Un paso a la derecha es
$+1$; un paso \emph{abajo} es $+10$ (desde el $44$ rojo hacia abajo
al azul $54$).}
\end{omfigure}

\begin{method}[Moviéndose en la cuadrícula]\label{met:g1:tens:grid}
\begin{enumerate}
  \item $+1$ / $-1$: un paso derecha/izquierda;
  \item $+10$ / $-10$: un paso hacia abajo/arriba --- solo el dígito de las decenas
        cambios: $37 + 10 = 47$;
  \item contando de decenas: $10, 20, 30, \dots$ --- sigue el último
        columna de la cuadrícula hacia abajo.
\end{enumerate}
\end{method}

\begin{example}[Contando de 2 y 5]\label{ex:g1:tens:skip}
Por $2$s: $2, 4, 6, 8, 10, 12, \dots$ (los números pares). Por $5$s:
$5, 10, 15, 20, 25, \dots$ --- terminan en $5$ o $0$ y forman un
Bonito patrón en la cuadrícula. El conteo saltado prepara la multiplicación
(\cref{ch:g2:mult}).
\end{example}

\section{Comparando números más grandes}

\begin{method}[Comparando con decenas]\label{met:g1:tens:compare}
Más victorias en decenas: $61 > 58$ porque $6$ decenas vencieron a $5$ decenas ---
¡aunque $8$ es más que $1$! Sólo cuando las decenas son iguales
las unidades deciden: $43 < 47$.
\end{method}

\begin{example}\label{ex:g1:tens:compare}
Orden $35$, $53$, $29$: decenas primero ($2$, $3$, $5$):
\[
29 < 35 < 53 .
\]
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g1:tens:1}
Cuantas decenas y unidades hay en: $26$; $40$; $73$; \ $99$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:2}
Escribe el número: $5$ decenas y $2$ unidades; $8$ decenas; $1$ diez y
$9$ unidades; $10$ decenas.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:3}
Escribe en dígitos: sesenta y tres; \ cuarenta; \ ochenta y cinco; \ uno
cien.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:4}
Calcular moviéndose en la cuadrícula de cien: $23 + 10$; \quad$67 + 10$;
\quad $45 - 10$; \quad$80 - 10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:5}
Continúa: $10, 20, 30, ?, ?, ?$. \quad
$5, 10, 15, ?, ?, ?$. \quad $12, 14, 16, ?, ?, ?$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:6}
Copiar y completar con $<$ o $>$:
\[
34 \;?\; 43, \qquad
58 \;?\; 61, \qquad
70 \;?\; 69, \qquad
99 \;?\; 100 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:7}
Ordenar de menor a mayor: $46$; $19$; $64$; $91$; \
$41$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:8}
Un jardinero agrupa flores $52$ en racimos de diez. cuantos llenos
racimos y cuantas flores sueltas?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g1:tens:9}
En la cuadrícula de cien, comience en $36$; baja dos escalones y un escalón
izquierda. ¿Dónde aterrizas? ¿Qué cálculo acabas de hacer?
\end{exercise}
