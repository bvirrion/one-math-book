\chapter{Medición y tiempo}\label{ch:g1:measure}

¿Qué lápiz es más largo? ¿Qué día es? ¿Cuánto cuesta un dulce?
Este capítulo compara longitudes y las mide en centímetros,
ordena los días de la semana, lee horas enteras en el reloj,
y cuenta monedas pequeñas.

\section{Más y más corto}

\begin{method}[Comparando longitudes]\label{met:g1:measure:compare}
Para comparar dos longitudes, alinéelas con \emph{un extremo coincide}:
\begin{enumerate}
  \item coloque ambos objetos uno al lado del otro, comenzando desde la misma línea;
  \item el que llega más lejos es \emph{más extenso}, el otro es
        \emph{más corto};
  \item si ambos llegan al mismo lugar, son el \emph{lo mismo
        longitud}.
\end{enumerate}
Hacer coincidir el inicio es todo el secreto: comparar desde torcido
comienza a engañar a la vista.
\end{method}

\begin{definition}[El centimetro]\label{def:g1:measure:cm}
Para decir \emph{cómo} de largo, medimos con una regla en
\emph{centímetros} (cm). Coloque el $0$ de la regla al inicio de
el objeto y lea el número al final.
\end{definition}

\section{Los dias de la semana}

\begin{example}[Los siete días]\label{ex:g1:measure:days}
La semana tiene $7$ días, siempre en el mismo orden:
\[
\text{Monday, Tuesday, Wednesday, Thursday, Friday, Saturday,
Sunday.}
\]
Después del domingo, el lunes empieza de nuevo. \emph{Ayer} es el día
antes, \emph{mañana} el día después: si hoy es miércoles,
ayer fue martes y mañana es jueves.
\end{example}

\section{Horas enteras en el reloj}

\begin{method}[Leyendo una hora entera]\label{met:g1:measure:clock}
Cuando la manecilla larga apunta directamente al $12$, es un todo
hora: lea el número que señala la manecilla \emph{corto}. mano corta
el $3$, manecilla larga del $12$: son las horas $3$.
\end{method}

\section{monedas pequeñas}

\begin{example}[Haciendo una cantidad]\label{ex:g1:measure:coins}
Con las monedas de $1$, $2$ y $5$ se pueden realizar muchas cantidades en varios
maneras. Para hacer $7$:
\[
5 + 2, \qquad
5 + 1 + 1, \qquad
2 + 2 + 2 + 1 .
\]
Las monedas más grandes primero suelen ser la forma más rápida, usando el
menor cantidad de monedas.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g1:measure:1}
Alinea tres lápices iguales en un extremo. ¿Cuál es el más largo?
¿Cuál es el más corto?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:2}
Mide con tu regla: el lado largo de este libro; tu pequeño
dedo; una cuchara. Responde en centímetros enteros.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:3}
Nombra el día: después del lunes; antes del domingo; dos días después del viernes.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:4}
Hoy es jueves. ¿Qué día fue ayer? ¿Qué día es mañana?
¿Qué día es dentro de tres días?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:5}
Dibuja un reloj que muestre $6$ en punto. ¿Dónde está la mano corta? donde esta
la mano larga?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:6}
Leer la hora completa: manecilla corta en $5$, manecilla larga en $12$; corto
mano en $11$, mano larga en $12$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:7}
Haz $8$ con monedas de $1$, $2$ y $5$, de dos formas diferentes.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:8}
Leo tiene una moneda $5$ y tres monedas $2$. cuanto dinero tiene
en total?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g1:measure:9}
Un dulce cuesta $3$. Zoe paga con una moneda $5$. ¿Cuánto cambio
ella regreso? (Cuente desde $3$ hasta $5$).
\end{exercise}
