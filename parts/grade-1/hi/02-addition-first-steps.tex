\chapter{जोड़: पहला कदम}\label{ch:g1:addition}

तीन लाल कंचे और दो नीले कंचे: एक साथ, कितने? डालना
एक साथ और पूरी गिनती है \emph{जोड़ना} --- पहला
ऑपरेशन, इसके दो प्रसिद्ध संकेतों के साथ $+$ और $=$.

\section{एक साथ रखना}

\begin{definition}[जोड़ना]\label{def:g1:addition:def}
\emph{जोड़ा जा रहा है} इसका अर्थ है हर चीज़ को एक साथ रखना और गिनना। हम
लिखो
\[
3 + 2 = 5
\]
and read: ``three plus two equals five''. The answer of an addition
कहा जाता है \emph{जोड़}\index{जोड़}.
\end{definition}

  \node[font=\large] at (3.0,0) {$+$};

{\small चित्रों में एक अतिरिक्त: दो समूह मिलकर एक बनाते हैं

\end{omfigure}

\begin{example}[आदेश कोई मायने नहीं रखता]\label{ex:g1:addition:order}
$2 + 3$ और $3 + 2$ समान कंचों को एक साथ रखें: दोनों बनाते हैं $5$.
गणना करना $2 + 9$, बड़ी संख्या से शुरू करना तेज़ है:
``नौ'' कहें, फिर दो --- ``दस, ग्यारह'' गिनें। इसलिए
$2 + 9 = 11$.
\end{example}

\section{दस दोस्त}

\begin{example}[10 का पूरक]\label{ex:g1:addition:friends}
दो संख्याएँ जो बनाती हैं $10$ एक साथ हैं ``दस दोस्त'':
\[
1 + 9, \quad 2 + 8, \quad 3 + 7, \quad 4 + 6, \quad 5 + 5 .
\]
दस उंगलियाँ उन सभी को दिखाएँ: उठाएँ $3$ उंगलियां, और $7$ मुड़ा हुआ
वे दोस्त हैं. दस दोस्तों को दिल से जानने से कई दोस्त बन जाते हैं
तेजी से परिवर्धन।
\end{example}

{\small दस-फ़्रेम दस दोस्तों की प्रत्येक \omterm{def:g1:addition:def}{जोड़}ी को एक नज़र में दिखाता है:
क्या भरा है और क्या कमी है हमेशा बनाते रहो $10$ एक साथ।}
\end{omfigure}

\begin{method}[दस पार]\label{met:g1:addition:crossten}
गणना करना $8 + 5$:
\begin{enumerate}
  \item से ले लो $5$ क्या $8$ तक पहुँचने की जरूरत है $10$: वह है
        $2$ (इसके दस मित्र);
  \item $8 + 2 = 10$, और $3$ से रहता है $5$;
  \item $10 + 3 = 13$. इसलिए $8 + 5 = 13$.
\end{enumerate}
के माध्यम से गुजरते हुए $10$ कठोर हो जाता है \omterm{def:g1:addition:def}{जोड़} दो आसान में.
\end{method}

\begin{example}[दोगुना हो जाता है]\label{ex:g1:addition:doubles}
युगल दिल से सीखने लायक हैं:
\[
1+1=2, \quad 2+2=4, \quad 3+3=6, \quad 4+4=8, \quad 5+5=10,
\]
\[
6+6=12, \quad 7+7=14, \quad 8+8=16, \quad 9+9=18, \quad 10+10=20 .
\]
नियर-डबल्स मुफ़्त आते हैं: $6 + 7$ से एक अधिक है $6 + 6$, इसलिए $13$.
\end{example}

\section{छोटी समस्याएं}

\begin{example}\label{ex:g1:addition:problem}
``ली के पास है $7$ स्टिकर. उसे मिलेगा $6$ अधिक। ''अभी कितने?''
अधिक प्राप्त करने का अर्थ है जोड़ना: $7 + 6 = 13$ (दस के माध्यम से:
$7 + 3 = 10$, तब $+3$). ली के पास है $13$ स्टिकर. हमेशा के साथ समाप्त करें
एक छोटा सा वाक्य!
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g1:addition:1}
कंचे बनाएं और गणना करें: $4 + 3$; \quad $5 + 2$; \quad
$6 + 4$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:addition:2}
बड़ी संख्या से प्रारंभ करके गणना करें: $3 + 8$; \quad
$2 + 12$; \quad $4 + 9$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:addition:3}
दस मित्र दें: $7$; \ $4$; \ $9$; \ $5$; \ $1$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:addition:4}
दस के माध्यम से गणना करें (\cref{met:g1:addition:crossten}):
$9 + 4$; \quad $8 + 7$; \quad $6 + 5$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:addition:5}
युगलों का पाठ करें, फिर उनका उपयोग करके गणना करें: $7 + 8$; \quad
$5 + 6$; \quad $9 + 8$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:addition:6}
छिद्रों को पूरा करें:
\[
6 + \;?\; = 10, \qquad
\;?\; + 5 = 12, \qquad
10 + \;?\; = 17 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:addition:7}
``बस में हैं $8$ बच्चे; $5$ बड़े लोग चढ़ते हैं। कैसे
बस में बहुत लोग हैं?'' \omterm{def:g1:addition:def}{जोड़} और उत्तर लिखें
वाक्य.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:addition:8}
टॉम के पास है $6$ लाल कारें और $7$ नीली कारें. मिया के पास है $10$ गाड़ियाँ. किसके पास है
अधिक कारें?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g1:addition:9}
तीन पासे दिखाएँ $4$, $6$ और $5$. कुल कितना है? (दो पासे \omterm{def:g1:addition:def}{जोड़}ें
पहला --- उन दो को चुनें जो दस मित्र बनाते हैं!)
\end{exercise}
