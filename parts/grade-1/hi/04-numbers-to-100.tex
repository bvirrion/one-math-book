\chapter{संख्या 100 तक}\label{ch:g1:tens}

एक-एक करके बड़े ढेर को गिनने में बहुत समय लग जाता है। महान युक्ति:
वस्तुओं को पैक में बाँधें \emph{दस}, और पैक्स गिनें। यह
तक की संख्या इस प्रकार है $100$ निर्मित हैं --- और वे क्यों हैं
दो अंको से लिखा गया है.

\section{दस के बंडल}

\begin{definition}[दहाई और इकाई]\label{def:g1:tens:def}
ए \emph{दस} दस इकाइयों का एक बंडल है. जो नंबर $34$ मतलब $3$
दसियों और $4$ इकाइयाँ:
\[
34 = 10 + 10 + 10 + 4 .
\]
बायां अंक बंडलों को गिनता है, दायां अंक ढीली इकाइयों को गिनता है।
\end{definition}

बैंड) और चार ढीली छड़ियाँ। गिनने की जरूरत नहीं $34$ एक क!}
\end{omfigure}

\begin{example}[संख्याएँ पढ़ना]\label{ex:g1:tens:reading}
$20$ दो दहाई (बीस) है, $30$ तीन दहाई (तीस)\,\dots\ तक
$90$ (नब्बे) और $100$ (एक सौ: दस दहाई!)। बीच में:
$47$ ``सैंतालीस'' --- चालीस के लिए पढ़ता है $4$ दहाई, सात के लिए
इकाइयाँ।
\end{example}

वर्ष!}
\end{omfigure}

\begin{example}[समान अंक, भिन्न-भिन्न संख्याएँ]\label{ex:g1:tens:place}
$27$ और $72$ समान अंकों का उपयोग करें लेकिन बहुत भिन्न हैं: $27$ है
$2$ दसियों और $7$ इकाइयाँ; $72$ है $7$ दसियों और $2$ इकाइयाँ---बहुत
बड़ा! \emph{जगह} एक अंक यह तय करता है कि वह क्या मायने रखता है।
\end{example}

{\small सौ ग्रिड: प्रत्येक पंक्ति एक दस है। एक कदम सही है
$+1$; एक कदम \emph{नीचे} है $+10$ (लाल रंग से $44$ सीधे नीचे
नीले रंग के लिए $54$).}
\end{omfigure}

\begin{method}[ग्रिड पर चल रहा है]\label{met:g1:tens:grid}
\begin{enumerate}
  \item $+1$ / $-1$: एक कदम दाएं/बाएं;
  \item $+10$ / $-10$: एक कदम नीचे/ऊपर --- केवल दहाई का अंक
        परिवर्तन: $37 + 10 = 47$;
  \item दहाई से गिनती: $10, 20, 30, \dots $ --- अंतिम का अनुसरण करें
        ग्रिड का स्तंभ सीधा नीचे।
\end{enumerate}
\end{method}

\begin{example}[2 और 5 से गिनती]\label{ex:g1:tens:skip}
द्वारा $2$ एस: $2, 4, 6, 8, 10, 12, \dots $ (सम संख्याएँ)। द्वारा $5$ एस:
$5, 10, 15, 20, 25, \dots $ --- वे समाप्त हो जाते हैं $5$ या $0$ और एक बनाओ
ग्रिड पर अच्छा पैटर्न. गिनती छोड़ें गुणन तैयार करता है
(\cref{ch:g2:mult}).
\end{example}

\section{बड़ी संख्याओं की तुलना करना}

\begin{method}[दहाई से तुलना]\label{met:g1:tens:compare}
अधिक दसियों जीत: $61 > 58$ क्योंकि $6$ दसियों मारो $5$ दहाई ---
फिर भी $8$ से अधिक होता है $1$! ऐसा तभी करें जब दहाई बराबर हों
इकाइयाँ निर्णय लेती हैं: $43 < 47$.
\end{method}

\begin{example}\label{ex:g1:tens:compare}
आदेश $35$, $53$, $29$: दहाई पहले ($2$, $3$, $5$):
\[
29 < 35 < 53 .
\]
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g1:tens:1}
इसमें कितने दहाई और इकाई हैं: $26$; \ $40$; \ $73$; \ $99$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:tens:2}
संख्या लिखें: $5$ दसियों और $2$ इकाइयाँ; \ $8$ दसियों; \ $1$ दस और
$9$ इकाइयाँ; \ $10$ दसियों.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:tens:3}
अंकों में लिखें: तिरसठ; \ चालीस; \ पच्चासी; \ एक
सौ.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:tens:4}
सौ ग्रिड पर जाकर गणना करें: $23 + 10$; \quad $67 + 10$;
\quad $45 - 10$; \quad $80 - 10$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:tens:5}
जारी रखना: $10, 20, 30, ?, ?, ?$. \quad
$5, 10, 15, ?, ?, ?$. \quad $12, 14, 16, ?, ?, ?$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:tens:6}
कॉपी करें और पूरा करें $<$ या $>$:
\[
34 \;?\; 43, \qquad
58 \;?\; 61, \qquad
70 \;?\; 69, \qquad
99 \;?\; 100 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:tens:7}
सबसे छोटे से सबसे बड़े तक ऑर्डर करें: $46$; \ $19$; \ $64$; \ $91$; \
$41$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g1:tens:8}
एक माली बंडल बनाता है $52$ दस के गुच्छों में फूल. कितने भरे हुए हैं
गुच्छे, और कितने ढीले फूल?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g1:tens:9}
सौ ग्रिड पर, प्रारंभ करें $36$; दो कदम नीचे जाओ और एक कदम नीचे जाओ
बायां. तुम कहाँ उतरोगे? आपने अभी कौन सी गणना की?
\end{exercise}
