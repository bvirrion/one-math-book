\chapter{Subtraction: First Steps}\label{ch:g1:subtraction}

Seven balloons, two fly away: how many are left? Taking away is
\emph{subtracting}, written with the sign $-$. And subtraction has a
second face: it also answers ``how many more?''.

\section{Taking away}

\begin{definition}[Subtraction]\label{def:g1:subtraction:def}
\emph{Subtracting} means taking away and counting what is left. We
write
\[
7 - 2 = 5
\]
and read: ``seven minus two equals five''.
\end{definition}

\begin{example}[Counting back]\label{ex:g1:subtraction:countback}
For a small take-away, count backward on the line: $11 - 3$: say
``eleven'', then three steps back --- ``ten, nine, eight''. So
$11 - 3 = 8$.
\end{example}

\section{How many more?}

\begin{example}[The difference]\label{ex:g1:subtraction:difference}
``Nina has $9$ marbles and Tom has $6$. How many more has Nina?''
Nothing is taken away, yet it is a subtraction: pair up the marbles,
and count Nina's marbles without a partner:
\[
9 - 6 = 3 .
\]
Nina has $3$ more. The answer of a subtraction is called the
\emph{difference}.
\end{example}

\begin{method}[Counting up]\label{met:g1:subtraction:countup}
To find a difference, counting \emph{up} is often easiest:
for $12 - 9$, start at $9$ and climb to $12$: ``ten, eleven,
twelve'' --- three steps. So $12 - 9 = 3$. (Climb through $10$ for
bigger gaps: $14 - 8$: from $8$ to $10$ is $2$, from $10$ to $14$
is $4$: difference $6$.)
\end{method}

\section{Addition and subtraction are twins}

\begin{example}[Fact families]\label{ex:g1:subtraction:family}
The numbers $3$, $4$ and $7$ form a little family:
\[
3 + 4 = 7, \qquad
4 + 3 = 7, \qquad
7 - 4 = 3, \qquad
7 - 3 = 4 .
\]
Knowing one fact of the family gives the three others for free.
Subtraction undoes addition: after $+4$, doing $-4$ brings you back.
\end{example}

\begin{example}[Checking]\label{ex:g1:subtraction:check}
Computed $13 - 5 = 8$? Check with the twin addition:
$8 + 5 = 13$. \checkmark\ A subtraction is right exactly when adding
back works.
\end{example}

\begin{example}[Which operation?]\label{ex:g1:subtraction:choose}
``$6$ birds on the wire, $4$ fly away'': take away, $6 - 4 = 2$.
``$6$ birds, then $4$ more arrive'': put together, $6 + 4 = 10$.
``$6$ sparrows and $4$ robins: how many more sparrows?'':
difference, $6 - 4 = 2$. Draw the story when unsure!
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:1}
Draw, cross out, and compute: $6 - 2$; \quad $8 - 5$; \quad
$10 - 4$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:2}
Count backward to compute: $9 - 3$; \quad $12 - 2$; \quad
$15 - 4$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:3}
Count up to find the difference: $11 - 8$; \quad $13 - 9$; \quad
$16 - 12$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:4}
Write the four facts of the family of $2$, $6$ and $8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:5}
Compute, then check with the twin addition: $14 - 6$; \quad
$17 - 9$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:6}
Complete the holes:
\[
10 - \;?\; = 7, \qquad
\;?\; - 5 = 5, \qquad
12 - \;?\; = 12 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:7}
``There are $15$ cherries in the bowl. The children eat $8$. How
many are left?'' Write the subtraction and the answer sentence.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:subtraction:8}
For each story, choose $+$ or $-$, then compute: ``$9$ ducks on the
pond, $5$ swim away''; ``$9$ ducks, $5$ geese arrive'';
``Ali is $9$, his sister is $5$: how many years older is Ali?''.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g1:subtraction:9}
Mother duck had $12$ eggs. Some hatched; now $12 - ?$ eggs remain
and she counts $7$ unhatched eggs. How many eggs hatched? (Which
fact family is this?)
\end{exercise}
