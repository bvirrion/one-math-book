\chapter{Mesures et temps}\label{ch:g1:measure}

Quel crayon est le plus long~? Quel jour sommes-nous~? Combien
co\^ute un bonbon~? Ce chapitre compare des longueurs et les mesure
en centim\`etres, met les jours de la semaine en ordre, lit les
heures piles sur l'horloge, et compte de petites pi\`eces.

\section{Plus long et plus court}

\begin{method}[Comparer des longueurs]\label{met:g1:measure:compare}
Pour comparer deux longueurs, les aligner avec \emph{une extr\'emit\'e
commune}~:
\begin{enumerate}
  \item placer les deux objets c\^ote \`a c\^ote, en partant de la
        m\^eme ligne~;
  \item celui qui d\'epasse le plus est le plus \emph{long}, l'autre
        est le plus \emph{court}~;
  \item s'ils arrivent au m\^eme endroit, ils ont la \emph{m\^eme
        longueur}.
\end{enumerate}
Aligner le d\'epart est tout le secret --- comparer en d\'ecal\'e
trompe l'{\oe}il.
\end{method}

\begin{definition}[Le centim\`etre]\label{def:g1:measure:cm}
Pour dire \emph{combien} c'est long, on mesure avec une r\`egle en
\emph{centim\`etres} (cm). Placer le $0$ de la r\`egle au d\'ebut de
l'objet, et lire le nombre \`a son extr\'emit\'e.
\end{definition}

\section{Les jours de la semaine}

\begin{example}[Les sept jours]\label{ex:g1:measure:days}
La semaine a $7$ jours, toujours dans le m\^eme ordre~:
\[
\text{lundi, mardi, mercredi, jeudi, vendredi, samedi,
dimanche.}
\]
Apr\`es dimanche, lundi recommence. \emph{Hier} est le jour
d'avant, \emph{demain} le jour d'apr\`es~: si aujourd'hui c'est
mercredi, hier c'\'etait mardi et demain c'est jeudi.
\end{example}

\section{Les heures piles sur l'horloge}

\begin{method}[Lire une heure pile]\label{met:g1:measure:clock}
Quand la grande aiguille pointe tout droit vers le $12$, c'est une
heure pile~: lire le nombre que indique la \emph{petite} aiguille.
Petite aiguille sur le $3$, grande aiguille sur le $12$~: il est
$3$~heures.
\end{method}

\section{Petites pi\`eces}

\begin{example}[Composer une somme]\label{ex:g1:measure:coins}
Avec des pi\`eces de $1$, $2$ et $5$, beaucoup de \omterm{def:g1:addition:def}{sommes} se font de
plusieurs fa\c{c}ons. Pour faire $7$~:
\[
5 + 2, \qquad
5 + 1 + 1, \qquad
2 + 2 + 2 + 1 .
\]
Commencer par les plus grosses pi\`eces est en g\'en\'eral le plus
rapide, avec le moins de pi\`eces.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g1:measure:1}
Aligne trois crayons avec une extr\'emit\'e commune. Lequel est le
plus long~? Lequel est le plus court~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:2}
Mesure avec ta r\`egle~: le long c\^ot\'e de ce livre~; ton petit
doigt~; une cuill\`ere. R\'eponds en centim\`etres entiers.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:3}
Nomme le jour~: apr\`es lundi~; avant dimanche~; deux jours apr\`es
vendredi.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:4}
Aujourd'hui c'est jeudi. Quel jour \'etait hier~? Quel jour est
demain~? Quel jour dans trois jours~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:5}
Dessine une horloge qui indique $6$~heures. O\`u est la petite
aiguille~? O\`u est la grande aiguille~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:6}
Lis l'heure pile~: petite aiguille sur $5$, grande aiguille sur
$12$~; petite aiguille sur $11$, grande aiguille sur $12$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:7}
Fais $8$ avec des pi\`eces de $1$, $2$ et $5$, de deux fa\c{c}ons
diff\'erentes.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:measure:8}
L\'eo a une pi\`ece de $5$ et trois pi\`eces de $2$. Combien
d'argent a-t-il en tout~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g1:measure:9}
Un bonbon co\^ute $3$. Zo\'e paie avec une pi\`ece de $5$. Combien
re\c{c}oit-elle de monnaie~? (Compte de $3$ \`a $5$ en montant.)
\end{exercise}
