\chapter{Figures et espace}\label{ch:g1:shapes}

Les fen\^etres sont des rectangles, les roues des cercles, les toits
cachent des triangles~: les figures sont partout. Ce chapitre apprend
\`a les reconna\^itre en comptant c\^ot\'es et sommets, et \`a dire
pr\'ecis\'ement \emph{o\`u} se trouvent les choses~: \`a gauche, \`a
droite, au-dessus, en dessous, entre.

\section{Les quatre figures de base}

\begin{definition}[Carr\'e, rectangle, triangle, cercle]\label{def:g1:shapes:def}
\begin{itemize}
  \item le \emph{triangle}\index{triangle} a $3$ c\^ot\'es et $3$ sommets~;
  \item le \emph{carr\'e}\index{carr\'e} a $4$ c\^ot\'es \'egaux et $4$ sommets~;
  \item le \emph{rectangle}\index{rectangle} a $4$ c\^ot\'es et $4$ sommets, avec les
        c\^ot\'es oppos\'es \'egaux (un carr\'e \'etir\'e)~;
  \item le \emph{cercle}\index{cercle} est parfaitement rond~: pas de c\^ot\'e, pas
        de sommet.
\end{itemize}
\end{definition}

\begin{example}[Les figures pench\'ees restent des figures]\label{ex:g1:shapes:tilted}
Un carr\'e pos\'e sur un sommet reste un carr\'e --- quatre c\^ot\'es
\'egaux, quatre sommets --- m\^eme s'il ressemble \`a un losange.
Tourner une figure ne change pas ce qu'elle est.
\end{example}

\begin{example}[Chasse aux figures]\label{ex:g1:shapes:hunt}
Dans la classe~: le tableau est un rectangle, l'horloge un cercle,
une serviette pli\'ee peut \^etre un triangle, certaines fen\^etres
sont des carr\'es. Les objets r\'eels ne sont pas des figures
parfaites, mais leurs faces s'en rapprochent (les solides et leurs
faces reviennent dans \cref{ch:g2:shapes}).
\end{example}

\section{O\`u est-ce~?}

\begin{definition}[Mots de position]\label{def:g1:shapes:position}
Pour dire o\`u se trouve quelque chose~: \emph{\`a gauche} /
\emph{\`a droite}, \emph{au-dessus} / \emph{en dessous},
\emph{sur} / \emph{sous}, \emph{devant} / \emph{derri\`ere},
\emph{entre}, \emph{\`a l'int\'erieur} / \emph{\`a l'ext\'erieur}.
\end{definition}

\section{Chemins sur un quadrillage}

\begin{method}[D\'ecrire un chemin]\label{met:g1:shapes:path}
Sur du papier quadrill\'e, un chemin est une liste de pas~:
$\to$ (une case \`a droite), $\leftarrow$ (\`a gauche), $\uparrow$
(vers le haut), $\downarrow$ (vers le bas).
\begin{enumerate}
  \item Pour suivre un chemin~: partir de la case marqu\'ee et ob\'eir
        aux fl\`eches une par une~;
  \item pour d\'ecrire un chemin~: le parcourir dans la t\^ete et
        \'ecrire les fl\`eches dans l'ordre~;
  \item deux chemins diff\'erents peuvent relier les m\^emes deux
        cases --- certains plus courts, d'autres plus longs.
\end{enumerate}
\end{method}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g1:shapes:1}
Pour chaque figure, compte les c\^ot\'es et les sommets et nomme-la~:
une figure avec $3$ sommets~; une figure ronde sans sommet~; une
figure avec $4$ c\^ot\'es \'egaux.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:2}
Trouve \`a la maison~: deux rectangles, un cercle, un triangle, un
carr\'e. (Portes~? assiettes~? panneaux de signalisation~?)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:3}
Vrai ou faux~? \guillemotleft{} Un carr\'e pos\'e sur un sommet
n'est plus un carr\'e. \guillemotright{} Explique avec
\cref{ex:g1:shapes:tilted}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:4}
Dessine sur du papier quadrill\'e~: un carr\'e de $3$ cases sur
$3$~; un rectangle de $5$ sur $2$~; un triangle en utilisant une
diagonale du quadrillage.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:5}
Place trois jouets en rang\'ee. D\'ecris la rang\'ee avec des mots de
position~: lequel est entre les autres~? Lequel est \`a gauche~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:6}
Dessine un cercle. Dessine une croix \emph{\`a l'int\'erieur}, une
\'etoile \emph{\`a l'ext\'erieur}, et un point \emph{sur} le cercle
lui-m\^eme.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:7}
Sur du papier quadrill\'e, marque une case de d\'epart. Suis~:
$\to\ \uparrow\ \to\ \uparrow\ \to$. Marque l'arriv\'ee. Puis
\'ecris le chemin qui revient tout droit au d\'epart.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:shapes:8}
Combien de sommets en tout~: deux triangles et un carr\'e~? Un
cercle et deux rectangles~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g1:shapes:9}
Prends $12$ allumettes. Peux-tu dessiner le contour d'un carr\'e en
utilisant \emph{toutes} les allumettes, avec des allumettes enti\`eres
et le m\^eme nombre sur chaque c\^ot\'e~? Combien d'allumettes par
c\^ot\'e~? Pourrait-on le faire avec $10$ allumettes~?
\end{exercise}
