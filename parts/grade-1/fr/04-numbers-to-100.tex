\chapter{Les nombres jusqu'\`a 100}\label{ch:g1:tens}

Compter un gros tas un par un prend une \'eternit\'e. La grande
astuce~: regrouper les objets par paquets de \emph{dix}, et compter
les paquets. C'est ainsi que les nombres jusqu'\`a $100$ se
construisent --- et pourquoi on les \'ecrit avec deux chiffres.

\section{Paquets de dix}

\begin{definition}[Dizaines et unit\'es]\label{def:g1:tens:def}
Une \emph{dizaine} est un paquet de dix unit\'es. Le nombre $34$
signifie $3$ dizaines et $4$ unit\'es~:
\[
34 = 10 + 10 + 10 + 4 .
\]
Le chiffre de gauche compte les paquets, celui de droite les unit\'es
isol\'ees.
\end{definition}

\begin{example}[Lire les nombres]\label{ex:g1:tens:reading}
$20$ c'est deux dizaines (vingt), $30$ trois dizaines (trente)\,\dots\
jusqu'\`a $90$ (quatre-vingt-dix) et $100$ (cent~: dix dizaines~!).
Entre les deux~: $47$ se lit \guillemotleft{} quarante-sept
\guillemotright{} --- quarante pour les $4$ dizaines, sept pour les
unit\'es.
\end{example}

\begin{example}[M\^emes chiffres, nombres diff\'erents]\label{ex:g1:tens:place}
$27$ et $72$ utilisent les m\^emes chiffres mais sont tr\`es
diff\'erents~: $27$ c'est $2$ dizaines et $7$ unit\'es~; $72$ c'est
$7$ dizaines et $2$ unit\'es --- bien plus grand~! La \emph{place}
d'un chiffre d\'ecide ce qu'il compte.
\end{example}

\section{La grille de 100}

\begin{omfigure}
\begin{tikzpicture}[scale=0.52]
  \foreach \r in {0,...,9} {
    \foreach \c in {0,...,9} {
      \pgfmathtruncatemacro{\n}{\r*10+\c+1}
      \draw[gray!50] (\c,-\r) +(-0.5,-0.5) rectangle +(0.5,0.5);
      \node[font=\footnotesize] at (\c,-\r) {\n};
    }
  }
  \draw[omThm, very thick] (3,-4) +(-0.5,-0.5) rectangle +(0.5,0.5);
  \draw[omDef, very thick] (3,-5) +(-0.5,-0.5) rectangle +(0.5,0.5);
\end{tikzpicture}

{\small La grille de $100$~: chaque ligne est une dizaine. Un pas \`a
droite c'est $+1$~; un pas \emph{vers le bas} c'est $+10$ (du $44$
rouge tout droit au $54$ bleu).}
\end{omfigure}

\begin{method}[Se d\'eplacer sur la grille]\label{met:g1:tens:grid}
\begin{enumerate}
  \item $+1$ / $-1$~: un pas \`a droite / \`a gauche~;
  \item $+10$ / $-10$~: un pas vers le bas / vers le haut --- seul le
        chiffre des dizaines change~: $37 + 10 = 47$~;
  \item compter de $10$ en $10$~: $10, 20, 30, \dots$ --- suivre la
        derni\`ere colonne de la grille tout droit vers le bas.
\end{enumerate}
\end{method}

\begin{example}[Compter de 2 en 2 et de 5 en 5]\label{ex:g1:tens:skip}
De $2$ en $2$~: $2, 4, 6, 8, 10, 12, \dots$ (les nombres pairs). De
$5$ en $5$~: $5, 10, 15, 20, 25, \dots$ --- ils se terminent par $5$
ou $0$ et dessinent un joli motif sur la grille. Compter par bonds
pr\'epare la multiplication (\cref{ch:g2:mult}).
\end{example}

\section{Comparer de plus grands nombres}

\begin{method}[Comparer avec les dizaines]\label{met:g1:tens:compare}
Plus de dizaines l'emporte~: $61 > 58$ parce que $6$ dizaines battent
$5$ dizaines --- m\^eme si $8$ est plus grand que $1$~! Seulement
quand les dizaines sont \'egales, les unit\'es d\'ecident~:
$43 < 47$.
\end{method}

\begin{example}\label{ex:g1:tens:compare}
Ranger $35$, $53$, $29$~: d'abord les dizaines ($2$, $3$, $5$)~:
\[
29 < 35 < 53 .
\]
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g1:tens:1}
Combien de dizaines et d'unit\'es dans~: $26$~; \ $40$~; \ $73$~;
\ $99$~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:2}
\'Ecris le nombre~: $5$ dizaines et $2$ unit\'es~; \ $8$ dizaines~;
\ $1$ dizaine et $9$ unit\'es~; \ $10$ dizaines.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:3}
\'Ecris en chiffres~: soixante-trois~; \ quarante~; \
quatre-vingt-cinq~; \ cent.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:4}
Calcule en te d\'epla\c{c}ant sur la grille de $100$~: $23 + 10$~;
\quad $67 + 10$~; \quad $45 - 10$~; \quad $80 - 10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:5}
Continue~: $10, 20, 30, ?, ?, ?$. \quad
$5, 10, 15, ?, ?, ?$. \quad $12, 14, 16, ?, ?, ?$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:6}
Recopie et compl\`ete avec $<$ ou $>$~:
\[
34 \;?\; 43, \qquad
58 \;?\; 61, \qquad
70 \;?\; 69, \qquad
99 \;?\; 100 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:7}
Range du plus petit au plus grand~: $46$~; \ $19$~; \ $64$~; \
$91$~; \ $41$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g1:tens:8}
Un jardinier regroupe $52$ fleurs en bouquets de dix. Combien de
bouquets complets, et combien de fleurs isol\'ees~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g1:tens:9}
Sur la grille de $100$, pars de $36$~; descends de deux cases et
d\'ecale d'une case \`a gauche. O\`u arrives-tu~? Quel calcul
viens-tu de faire~?
\end{exercise}
