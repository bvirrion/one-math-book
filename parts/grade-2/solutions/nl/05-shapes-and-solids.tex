\section*{Hoofdstuk \ref{ch:g2:shapes} --- Vormen en lichamen}

\begin{solution}{exo:g2:shapes:1}
$5$ zijden, $5$ hoeken: een vijfhoek. $3$ hoeken (en $3$ zijden):
een driehoek. Rond: een cirkel.
\end{solution}

\begin{solution}{exo:g2:shapes:2}
\omterm{def:g2:shapes:sort}{Veelhoeken}: vierkant, driehoek, zeshoek (rechte zijden). Geen
\omterm{def:g2:shapes:sort}{veelhoek}: de cirkel (gebogen rand).
\end{solution}

\begin{solution}{exo:g2:shapes:3}
Rechte hoeken bij bijvoorbeeld de hoeken van het bord, een boek, een
raam; een niet-rechte hoek bij het open deksel van een doos of de
wijzers van een klok om $2$ uur.
\end{solution}

\begin{solution}{exo:g2:shapes:4}
Vierkant: $4$ rechte hoeken. Rechthoek: $4$. Een driehoek kan
hoogstens $1$ rechte hoek hebben --- met twee zouden de andere twee
zijden elkaar nooit raken.
\end{solution}

\begin{solution}{exo:g2:shapes:5}
Rasterkopieën gemaakt door vakjes tussen hoeken te tellen;
vergelijken met het model controleert het werk.
\end{solution}

\begin{solution}{exo:g2:shapes:6}
Dobbelsteen: \omterm{def:g2:shapes:solids}{kubus}. Soepblikje: \omterm{def:g2:shapes:solids}{cilinder}. Voetbal: bol.
Schoenendoos: balk.
\end{solution}

\begin{solution}{exo:g2:shapes:7}
Een \omterm{def:g2:shapes:solids}{kubus} heeft $6$ \omterm{def:g2:shapes:solids}{vlakken}, elk een vierkant. Een balk heeft ook
$6$ \omterm{def:g2:shapes:solids}{vlakken} (rechthoeken).
\end{solution}

\begin{solution}{exo:g2:shapes:8}
De bol rolt (overal rond) en de \omterm{def:g2:shapes:solids}{cilinder} rolt (op de gebogen
zijkant). De \omterm{def:g2:shapes:solids}{kubus} en de balk niet: ze hebben alleen platte \omterm{def:g2:shapes:solids}{vlakken}
en blijven erop liggen.
\end{solution}

\begin{solution}{exo:g2:shapes:9}
$6$ zijden: een zeshoek, met $6$ hoeken.
\end{solution}

\begin{solution}{exo:g2:shapes:10}
De omtrek van het huis (vierkante muur met driehoekig dak, de
dakbasis op de bovenkant van het vierkant) heeft $5$ zijden en $5$
hoeken --- het is een vijfhoek (onderkant, twee muren, twee
dakhellingen; de bovenkant van de muur zit onder het dak).
\end{solution}
