\section*{Hoofdstuk \ref{ch:g2:measure} --- Geld en maten}

\begin{solution}{exo:g2:measure:1}
Potlood: cm. Lengte van de klas: m. Veer: g. Schooltas: kg.
Waterfles: L.
\end{solution}

\begin{solution}{exo:g2:measure:2}
$2$ m $= 200$ cm; \quad $300$ cm $= 3$ m; \quad $1$ kg $= 1000$ g.
\end{solution}

\begin{solution}{exo:g2:measure:3}
Bijvoorbeeld: $16 = 10 + 5 + 1$; \ $23 = 20 + 2 + 1$; \
$35 = 20 + 10 + 5$. (Andere manieren zijn mogelijk.)
\end{solution}

\begin{solution}{exo:g2:measure:4}
$10 - 6 = 4$; \quad $20 - 17 = 3$; \quad $50 - 32 = 18$.
\end{solution}

\begin{solution}{exo:g2:measure:5}
Lange wijzer op $6$, korte tussen $4$ en $5$: half vijf
($4{:}30$). Lange wijzer op $3$, korte net na $8$: kwart over acht
($8{:}15$).
\end{solution}

\begin{solution}{exo:g2:measure:6}
Half drie: de lange wijzer wijst omlaag naar de $6$, de korte
wijzer zit tussen de $2$ en de $3$.
\end{solution}

\begin{solution}{exo:g2:measure:7}
Na mei: juni. V\'{o}\'{o}r januari: december. (Verjaardagsmaand
varieert.)
\end{solution}

\begin{solution}{exo:g2:measure:8}
Uitgegeven: $3 + 4 = 7$. Wisselgeld: $10 - 7 = 3$.
\end{solution}

\begin{solution}{exo:g2:measure:9}
$1$ kg $= 1000$ g, wat meer is dan $800$ g: de suiker is zwaarder,
met $1000 - 800 = 200$ g.
\end{solution}

\begin{solution}{exo:g2:measure:10}
Bijvoorbeeld $20 + 10 + 5 + 5 = 40$ en $10 + 10 + 10 + 10 = 40$
(twee manieren met precies vier munten of biljetten).
\end{solution}
