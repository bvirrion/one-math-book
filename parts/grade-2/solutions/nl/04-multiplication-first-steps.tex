\section*{Hoofdstuk \ref{ch:g2:mult} --- Vermenigvuldigen: eerste stappen}

\begin{solution}{exo:g2:mult:1}
$2 + 2 + 2 = 3 \times 2 = 6$; \quad
$5 + 5 + 5 + 5 = 4 \times 5 = 20$; \quad
$10 + 10 + 10 = 3 \times 10 = 30$.
\end{solution}

\begin{solution}{exo:g2:mult:2}
$3$ rijen van $5$: $3 \times 5 = 15$. Op de zijkant gekeerd, $5$
rijen van $3$: $5 \times 3 = 15$ --- dezelfde $15$.
\end{solution}

\begin{solution}{exo:g2:mult:3}
Tafel van $2$: $2, 4, 6, 8, 10, 12$. Tafel van $5$:
$5, 10, 15, 20, 25, 30$.
\end{solution}

\begin{solution}{exo:g2:mult:4}
$2 \times 7 = 14$; \quad $5 \times 3 = 15$; \quad
$10 \times 4 = 40$; \quad $2 \times 9 = 18$; \quad
$5 \times 6 = 30$.
\end{solution}

\begin{solution}{exo:g2:mult:5}
Verdubbelingen: $12$; $16$; $20$. Helften: de helft van $12$ is
$6$; de helft van $20$ is $10$.
\end{solution}

\begin{solution}{exo:g2:mult:6}
$3, 6, 9, 12, 15, 18$. Dus $3 \times 5 = 15$.
\end{solution}

\begin{solution}{exo:g2:mult:7}
$4 \times 5 = 20$ stoelen.
\end{solution}

\begin{solution}{exo:g2:mult:8}
$3 \times 8 = 24$ poten ($8, 16, 24$).
\end{solution}

\begin{solution}{exo:g2:mult:9}
$2$ rijen van $6$: $2 \times 6 = 12$. $3$ rijen van $4$:
$3 \times 4 = 12$.
\end{solution}

\begin{solution}{exo:g2:mult:10}
Hopsgewijs met $5$: $5, 10, 15, 20$ --- drie dozen geven maar $15$
(te weinig), vier dozen geven $20$. Mia moet $4$ dozen kopen.
\end{solution}
