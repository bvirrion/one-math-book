\section*{Hoofdstuk \ref{ch:g2:problems} --- Problemen oplossen}

\begin{solution}{exo:g2:problems:1}
Bij elkaar: optelling. $50$ snoepjes, $12$ opgegeten: aftrekking.
$4$ dozen van $5$: vermenigvuldiging. Hoeveel meer: aftrekking.
\end{solution}

\begin{solution}{exo:g2:problems:2}
$156 - 48 = 108$. Er blijven $108$ auto's.
\end{solution}

\begin{solution}{exo:g2:problems:3}
$5 \times 4 = 20$ bloemen.
\end{solution}

\begin{solution}{exo:g2:problems:4}
Vergelijkingsstaaf: Tom $45$, Mia korter met $17$:
$45 - 17 = 28$ kaarten.
\end{solution}

\begin{solution}{exo:g2:problems:5}
Ontvangen flessen: $4 \times 6 = 24$; overblijvend:
$24 - 9 = 15$ flessen.
\end{solution}

\begin{solution}{exo:g2:problems:6}
Het nutteloze getal is de leeftijd ($8$). Knikkers:
$23 + 15 = 38$.
\end{solution}

\begin{solution}{exo:g2:problems:7}
Taartjes: $7 + 10 = 17$. Zaterdag, alle cakes:
$12 + 7 = 19$.
\end{solution}

\begin{solution}{exo:g2:problems:8}
Rood $8$, groen $4$, geel $6$: in totaal $8 + 4 + 6 = 18$ stuks
fruit. Rood heeft de meeste.
\end{solution}

\begin{solution}{exo:g2:problems:9}
$80 - 35 = 45$ cm blijft over.
\end{solution}

\begin{solution}{exo:g2:problems:10}
Ochtend: $5 \times 4 = 20$; middag: $3 \times 4 = 12$; totaal
$20 + 12 = 32$ cakes. (Of: $5 + 3 = 8$ bladen van $4$, en
$8 \times 4 = 32$.)
\end{solution}
