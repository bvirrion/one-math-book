\section*{Hoofdstuk \ref{ch:g2:addition} --- Optellen met onthouden}

\begin{solution}{exo:g2:addition:1}
$34 + 25 = 59$; \quad $126 + 43 = 169$; \quad $507 + 82 = 589$.
\end{solution}

\begin{solution}{exo:g2:addition:2}
$47 + 8 = 55$; \quad $56 + 27 = 83$; \quad $65 + 19 = 84$.
\end{solution}

\begin{solution}{exo:g2:addition:3}
$268 + 145 = 413$; \quad $374 + 158 = 532$; \quad
$429 + 293 = 722$.
\end{solution}

\begin{solution}{exo:g2:addition:4}
$38 + 5 = 38 + 2 + 3 = 43$; \quad $67 + 6 = 67 + 3 + 3 = 73$; \quad
$54 + 8 = 54 + 6 + 2 = 62$.
\end{solution}

\begin{solution}{exo:g2:addition:5}
$42 + 36 = 70 + 8 = 78$; \quad $53 + 24 = 70 + 7 = 77$; \quad
$61 + 28 = 80 + 9 = 89$.
\end{solution}

\begin{solution}{exo:g2:addition:6}
$58 + 34$: schatting $60 + 30 = 90$; precies $92$.
$296 + 187$: schatting $300 + 200 = 500$; precies $483$.
\end{solution}

\begin{solution}{exo:g2:addition:7}
$128 + 95 = 223$. Samen hebben ze $223$ knikkers.
\end{solution}

\begin{solution}{exo:g2:addition:8}
Het eerste getal plus $27$ is $64$, dus het is $64 - 27 = 37$: het
ontbrekende eenhedencijfer is $7$ (het getal is $37$).
\end{solution}

\begin{solution}{exo:g2:addition:9}
Kinderen: $27 + 28 = 55$; met de volwassenen, $55 + 6 = 61$
mensen.
\end{solution}

\begin{solution}{exo:g2:addition:10}
$27$ en $33$ hebben eenheden $7$ en $3$ (tientalvrienden):
$27 + 33 = 60$, dan $60 + 40 = 100$. De drie getallen zijn
$27$, $33$ en $40$.
\end{solution}
