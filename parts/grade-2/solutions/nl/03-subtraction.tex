\section*{Hoofdstuk \ref{ch:g2:subtraction} --- Aftrekken}

\begin{solution}{exo:g2:subtraction:1}
$58 - 23 = 35$; \quad $176 - 45 = 131$; \quad $389 - 164 = 225$.
\end{solution}

\begin{solution}{exo:g2:subtraction:2}
$52 - 17 = 35$; \quad $73 - 28 = 45$; \quad $60 - 24 = 36$.
\end{solution}

\begin{solution}{exo:g2:subtraction:3}
$324 - 167 = 157$ (controle $157 + 167 = 324$); \quad
$500 - 236 = 264$ (controle $264 + 236 = 500$); \quad
$703 - 158 = 545$ (controle $545 + 158 = 703$).
\end{solution}

\begin{solution}{exo:g2:subtraction:4}
$62 - 58 = 4$; \quad $91 - 85 = 6$; \quad $100 - 96 = 4$.
\end{solution}

\begin{solution}{exo:g2:subtraction:5}
$50 - 42 = 8$; \quad $20 - 13 = 7$; \quad $100 - 75 = 25$.
\end{solution}

\begin{solution}{exo:g2:subtraction:6}
Vorig jaar: $156 - 28 = 128$ cm.
\end{solution}

\begin{solution}{exo:g2:subtraction:7}
$85 - 39 = 46$: controle $46 + 39 = 85$ --- goed.
$214 - 68 = 156$: controle $156 + 68 = 224 \neq 214$ --- fout (het
juiste antwoord is $146$).
\end{solution}

\begin{solution}{exo:g2:subtraction:8}
``wegvliegen'': $47 - 19 = 28$. ``meer geplukt'': $47 + 19 = 66$.
``hoeveel meer'': $47 - 19 = 28$.
\end{solution}

\begin{solution}{exo:g2:subtraction:9}
Van $45$ naar $63$ is $18$, dus $63 - 18 = 45$: het gat is $18$.
\end{solution}

\begin{solution}{exo:g2:subtraction:10}
Werk achteruit vanaf $58$: v\'{o}\'{o}r het aftrekken van $18$ was het
getal $58 + 18 = 76$; v\'{o}\'{o}r het optellen van $30$ was het
$76 - 30 = 46$. Het startgetal was $46$ (controle:
$46 + 30 - 18 = 58$).
\end{solution}
