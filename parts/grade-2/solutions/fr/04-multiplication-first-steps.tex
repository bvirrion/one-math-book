\section*{Chapitre \ref{ch:g2:mult} --- Multiplication~: premiers pas}

\begin{solution}{exo:g2:mult:1}
$2 + 2 + 2 = 3 \times 2 = 6$~; \quad
$5 + 5 + 5 + 5 = 4 \times 5 = 20$~; \quad
$10 + 10 + 10 = 3 \times 10 = 30$.
\end{solution}

\begin{solution}{exo:g2:mult:2}
$3$ lignes de $5$~: $3 \times 5 = 15$. Tourn\'e sur le c\^ot\'e, $5$
lignes de $3$~: $5 \times 3 = 15$ --- le m\^eme $15$.
\end{solution}

\begin{solution}{exo:g2:mult:3}
Table de $2$~: $2, 4, 6, 8, 10, 12$. Table de $5$~:
$5, 10, 15, 20, 25, 30$.
\end{solution}

\begin{solution}{exo:g2:mult:4}
$2 \times 7 = 14$~; \quad $5 \times 3 = 15$~; \quad
$10 \times 4 = 40$~; \quad $2 \times 9 = 18$~; \quad
$5 \times 6 = 30$.
\end{solution}

\begin{solution}{exo:g2:mult:5}
Doubles~: $12$~; $16$~; $20$. Moiti\'es~: la moiti\'e de $12$ est
$6$~; la moiti\'e de $20$ est $10$.
\end{solution}

\begin{solution}{exo:g2:mult:6}
$3, 6, 9, 12, 15, 18$. Donc $3 \times 5 = 15$.
\end{solution}

\begin{solution}{exo:g2:mult:7}
$4 \times 5 = 20$ chaises.
\end{solution}

\begin{solution}{exo:g2:mult:8}
$3 \times 8 = 24$ pattes ($8, 16, 24$).
\end{solution}

\begin{solution}{exo:g2:mult:9}
$2$ lignes de $6$~: $2 \times 6 = 12$. $3$ lignes de $4$~:
$3 \times 4 = 12$.
\end{solution}

\begin{solution}{exo:g2:mult:10}
Compter de $5$ en $5$~: $5, 10, 15, 20$ --- trois bo\^ites ne donnent
que $15$ (trop peu), quatre bo\^ites donnent $20$. Mia doit acheter
$4$ bo\^ites.
\end{solution}
