\section*{Chapitre \ref{ch:g2:problems} --- R\'esoudre des probl\`emes}

\begin{solution}{exo:g2:problems:1}
Mettre ensemble~: addition. $50$ bonbons, $12$ mang\'es~:
soustraction. $4$ bo\^ites de $5$~: multiplication. Combien de
plus~: soustraction.
\end{solution}

\begin{solution}{exo:g2:problems:2}
$156 - 48 = 108$. Il reste $108$ voitures.
\end{solution}

\begin{solution}{exo:g2:problems:3}
$5 \times 4 = 20$ fleurs.
\end{solution}

\begin{solution}{exo:g2:problems:4}
Barre de comparaison~: Tom $45$, Mia plus courte de $17$~:
$45 - 17 = 28$ cartes.
\end{solution}

\begin{solution}{exo:g2:problems:5}
Bouteilles re\c{c}ues~: $4 \times 6 = 24$~; restantes~:
$24 - 9 = 15$ bouteilles.
\end{solution}

\begin{solution}{exo:g2:problems:6}
Le nombre inutile est l'\^age ($8$). Billes~:
$23 + 15 = 38$.
\end{solution}

\begin{solution}{exo:g2:problems:7}
Tartes~: $7 + 10 = 17$. Samedi, tous les g\^ateaux~:
$12 + 7 = 19$.
\end{solution}

\begin{solution}{exo:g2:problems:8}
Rouges $8$, vertes $4$, jaunes $6$~: en tout
$8 + 4 + 6 = 18$ fruits. Le rouge en a le plus.
\end{solution}

\begin{solution}{exo:g2:problems:9}
$80 - 35 = 45$~cm restent.
\end{solution}

\begin{solution}{exo:g2:problems:10}
Matin~: $5 \times 4 = 20$~; apr\`es-midi~: $3 \times 4 = 12$~; total
$20 + 12 = 32$ g\^ateaux. (Ou~: $5 + 3 = 8$ plateaux de $4$, et
$8 \times 4 = 32$.)
\end{solution}
