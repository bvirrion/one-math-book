\section*{Capítulo \ref{ch:g2:problems} --- Resolvendo Problemas}

\begin{solution}{exo:g2:problems:1}
Juntar: adição. $50$ doces, $12$ comido: subtração. $4$
caixas de $5$: multiplicação. Quantos mais: subtração.
\end{solution}

\begin{solution}{exo:g2:problems:2}
$156 - 48 = 108$. Lá fique $108$ carros.
\end{solution}

\begin{solution}{exo:g2:problems:3}
$5 \times 4 = 20$ flores.
\end{solution}

\begin{solution}{exo:g2:problems:4}
Barra de comparação: Tom $45$, Mia mais curta por $17$:
$45 - 17 = 28$ cartões.
\end{solution}

\begin{solution}{exo:g2:problems:5}
Garrafas recebidas: $4 \times 6 = 24$; restante:
$24 - 9 = 15$ garrafas.
\end{solution}

\begin{solution}{exo:g2:problems:6}
O número inútil é a idade ($8$). Mármores:
$23 + 15 = 38$.
\end{solution}

\begin{solution}{exo:g2:problems:7}
Tortas: $7 + 10 = 17$. Sábado, todos os bolos:
$12 + 7 = 19$.
\end{solution}

\begin{solution}{exo:g2:problems:8}
Vermelho $8$, verde $4$, amarelo $6$: em tudo $8 + 4 + 6 = 18$ frutas.
O vermelho tem mais.
\end{solution}

\begin{solution}{exo:g2:problems:9}
$80 - 35 = 45$ cm permanecem.
\end{solution}

\begin{solution}{exo:g2:problems:10}
Manhã: $5 \times 4 = 20$; tarde: $3 \times 4 = 12$; total
$20 + 12 = 32$ bolos. (Ou: $5 + 3 = 8$ bandejas de $4$, e
$8 \times 4 = 32$.)
\end{solution}
