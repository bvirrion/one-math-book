\section*{Capítulo \ref{ch:g2:subtraction} --- Subtração}

\begin{solution}{exo:g2:subtraction:1}
$58 - 23 = 35$; \quad $176 - 45 = 131$; \quad $389 - 164 = 225$.
\end{solution}

\begin{solution}{exo:g2:subtraction:2}
$52 - 17 = 35$; \quad $73 - 28 = 45$; \quad $60 - 24 = 36$.
\end{solution}

\begin{solution}{exo:g2:subtraction:3}
$324 - 167 = 157$ (verificar $157 + 167 = 324$); \quad
$500 - 236 = 264$ (verificar $264 + 236 = 500$); \quad
$703 - 158 = 545$ (verificar $545 + 158 = 703$).
\end{solution}

\begin{solution}{exo:g2:subtraction:4}
$62 - 58 = 4$; \quad $91 - 85 = 6$; \quad $100 - 96 = 4$.
\end{solution}

\begin{solution}{exo:g2:subtraction:5}
$50 - 42 = 8$; \quad $20 - 13 = 7$; \quad $100 - 75 = 25$.
\end{solution}

\begin{solution}{exo:g2:subtraction:6}
Ano passado: $156 - 28 = 128$ cm.
\end{solution}

\begin{solution}{exo:g2:subtraction:7}
$85 - 39 = 46$: verificar $46 + 39 = 85$ --- certo.
$214 - 68 = 156$: verificar $156 + 68 = 224 \neq 214$ --- errado (o
a resposta correta é $146$).
\end{solution}

\begin{solution}{exo:g2:subtraction:8}
``voar para longe'': $47 - 19 = 28$. ``mais escolhido'': $47 + 19 = 66$.
``quantos mais'': $47 - 19 = 28$.
\end{solution}

\begin{solution}{exo:g2:subtraction:9}
De $45$ até $63$ é $18$, então $63 - 18 = 45$: o buraco é $18$.
\end{solution}

\begin{solution}{exo:g2:subtraction:10}
Trabalhe de trás para frente a partir $58$: antes de tirar $18$, o número foi
$58 + 18 = 76$; antes de adicionar $30$, era $76 - 30 = 46$. O
número inicial era $46$ (verificar: $46 + 30 - 18 = 58$).
\end{solution}
