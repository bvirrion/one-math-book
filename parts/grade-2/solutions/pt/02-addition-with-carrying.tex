\section*{Capítulo \ref{ch:g2:addition} --- Adição com transporte}

\begin{solution}{exo:g2:addition:1}
$34 + 25 = 59$; \quad $126 + 43 = 169$; \quad $507 + 82 = 589$.
\end{solution}

\begin{solution}{exo:g2:addition:2}
$47 + 8 = 55$; \quad $56 + 27 = 83$; \quad $65 + 19 = 84$.
\end{solution}

\begin{solution}{exo:g2:addition:3}
$268 + 145 = 413$; \quad $374 + 158 = 532$; \quad
$429 + 293 = 722$.
\end{solution}

\begin{solution}{exo:g2:addition:4}
$38 + 5 = 38 + 2 + 3 = 43$; \quad $67 + 6 = 67 + 3 + 3 = 73$; \quad
$54 + 8 = 54 + 6 + 2 = 62$.
\end{solution}

\begin{solution}{exo:g2:addition:5}
$42 + 36 = 70 + 8 = 78$; \quad $53 + 24 = 70 + 7 = 77$; \quad
$61 + 28 = 80 + 9 = 89$.
\end{solution}

\begin{solution}{exo:g2:addition:6}
$58 + 34$: estimativa $60 + 30 = 90$; exatamente $92$.
$296 + 187$: estimativa $300 + 200 = 500$; exatamente $483$.
\end{solution}

\begin{solution}{exo:g2:addition:7}
$128 + 95 = 223$. Juntos eles têm $223$ bolinhas de gude.
\end{solution}

\begin{solution}{exo:g2:addition:8}
O primeiro número mais $27$ é $64$, então é $64 - 27 = 37$: o
dígito das unidades faltantes é $7$ (o número é $37$).
\end{solution}

\begin{solution}{exo:g2:addition:9}
Crianças: $27 + 28 = 55$; com os adultos, $55 + 6 = 61$
pessoas.
\end{solution}

\begin{solution}{exo:g2:addition:10}
$27$ e $33$ tem unidades $7$ e $3$ (dez amigos):
$27 + 33 = 60$, então $60 + 40 = 100$. Os três números são
$27$, $33$ e $40$.
\end{solution}
