\section*{Chapter \ref{ch:g2:addition} --- Addition with Carrying}

\begin{solution}{exo:g2:addition:1}
$34 + 25 = 59$; \quad $126 + 43 = 169$; \quad $507 + 82 = 589$.
\end{solution}

\begin{solution}{exo:g2:addition:2}
$47 + 8 = 55$; \quad $56 + 27 = 83$; \quad $65 + 19 = 84$.
\end{solution}

\begin{solution}{exo:g2:addition:3}
$268 + 145 = 413$; \quad $374 + 158 = 532$; \quad
$429 + 293 = 722$.
\end{solution}

\begin{solution}{exo:g2:addition:4}
$38 + 5 = 38 + 2 + 3 = 43$; \quad $67 + 6 = 67 + 3 + 3 = 73$; \quad
$54 + 8 = 54 + 6 + 2 = 62$.
\end{solution}

\begin{solution}{exo:g2:addition:5}
$42 + 36 = 70 + 8 = 78$; \quad $53 + 24 = 70 + 7 = 77$; \quad
$61 + 28 = 80 + 9 = 89$.
\end{solution}

\begin{solution}{exo:g2:addition:6}
$58 + 34$: estimate $60 + 30 = 90$; exactly $92$.
$296 + 187$: estimate $300 + 200 = 500$; exactly $483$.
\end{solution}

\begin{solution}{exo:g2:addition:7}
$128 + 95 = 223$. Together they have $223$ marbles.
\end{solution}

\begin{solution}{exo:g2:addition:8}
The first number plus $27$ is $64$, so it is $64 - 27 = 37$: the
missing units digit is $7$ (the number is $37$).
\end{solution}

\begin{solution}{exo:g2:addition:9}
Children: $27 + 28 = 55$; with the grown-ups, $55 + 6 = 61$
people.
\end{solution}

\begin{solution}{exo:g2:addition:10}
$27$ and $33$ have units $7$ and $3$ (ten friends):
$27 + 33 = 60$, then $60 + 40 = 100$. The three numbers are
$27$, $33$ and $40$.
\end{solution}
