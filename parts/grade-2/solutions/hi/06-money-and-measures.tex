\section*{अध्याय \ref{ch:g2:measure} --- धन और उपाय}

\begin{solution}{exo:g2:measure:1}
पेंसिल: सेमी. कक्षा की लंबाई: मी. पंख: जी. स्कूलबैग: किग्रा. पानी
बोतल: एल.
\end{solution}

\begin{solution}{exo:g2:measure:2}
$2$ एम $= 200$ सेमी; \quad $300$ सेमी $= 3$ एम; \quad $1$ किग्रा $= 1000$ जी।
\end{solution}

\begin{solution}{exo:g2:measure:3}
उदाहरण के लिए: $16 = 10 + 5 + 1$; \ $23 = 20 + 2 + 1$; \
$35 = 20 + 10 + 5$. (अन्य तरीके भी संभव हैं।)
\end{solution}

\begin{solution}{exo:g2:measure:4}
$10 - 6 = 4$; \quad $20 - 17 = 3$; \quad $50 - 32 = 18$.
\end{solution}

\begin{solution}{exo:g2:measure:5}
लंबा हाथ $6$, बीच में छोटा $4$ और $5$: साढ़े चार बजे
($4{:}30$). लंबा हाथ $3$, ठीक बाद में $8$: आठ बजकर पन्द्रह मिनट हुए ($8{:}15$).
\end{solution}

\begin{solution}{exo:g2:measure:6}
ढाई बजे: लंबा हाथ नीचे की ओर इशारा करता है $6$, छोटा हाथ
के बीच बैठता है $2$ और यह $3$.
\end{solution}

\begin{solution}{exo:g2:measure:7}
मई के बाद: जून. जनवरी से पहले: दिसंबर. (जन्मदिन का महीना अलग-अलग होता है।)
\end{solution}

\begin{solution}{exo:g2:measure:8}
खर्च किया गया: $3 + 4 = 7$. परिवर्तन: $10 - 7 = 3$.
\end{solution}

\begin{solution}{exo:g2:measure:9}
$1$ किग्रा $= 1000$ जी, जो कि इससे अधिक है $800$ जी: चीनी है
भारी, द्वारा $1000 - 800 = 200$ जी।
\end{solution}

\begin{solution}{exo:g2:measure:10}
उदाहरण के लिए $20 + 10 + 5 + 5 = 40$ और $10 + 10 + 10 + 10 = 40$
(बिल्कुल चार सिक्कों या बिलों के साथ दो तरीके)।
\end{solution}
