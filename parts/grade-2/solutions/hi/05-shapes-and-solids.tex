\section*{अध्याय \ref{ch:g2:shapes} --- आकार और ठोस}

\begin{solution}{exo:g2:shapes:1}
$5$ पक्ष, $5$ कोने: एक पंचकोण. $3$ कोने (और $3$ पक्ष): ए
\omterm{def:g1:shapes:def}{त्रिकोण}. गोल: एक वृत्त.
\end{solution}

\begin{solution}{exo:g2:shapes:2}
\omterm{def:g2:shapes:sort}{बहुभुज}: वर्ग, त्रिभुज, षट्भुज (सीधी भुजाएँ)। नहीं ए
\omterm{def:g2:shapes:sort}{बहुभुज}: वृत्त (घुमावदार सीमा)।
\end{solution}

\begin{solution}{exo:g2:shapes:3}
उदाहरण के लिए, बोर्ड, किताब, आदि के कोनों पर समकोण
खिड़की; किसी डिब्बे के खुले ढक्कन या हाथों पर एक सही कोण न होना
एक घड़ी पर $2$ बजे.
\end{solution}

\begin{solution}{exo:g2:shapes:4}
वर्ग: $4$ सही कोण। आयत: $4$. एक त्रिभुज में पर हो सकता है
सबसे $1$ समकोण --- दो अन्य दो भुजाएँ कभी नहीं बनेंगी
मिलना.
\end{solution}

\begin{solution}{exo:g2:shapes:5}
कोनों के बीच वर्गों की गिनती करके बनाई गई ग्रिड प्रतियां; तुलना करना
मॉडल के साथ काम की जाँच करता है।
\end{solution}

\begin{solution}{exo:g2:shapes:6}
मरना: \omterm{def:g2:shapes:solids}{घनक्षेत्र}. सूप का डिब्बा: \omterm{def:g2:shapes:solids}{सिलेंडर}. फ़ुटबॉल: गेंद. शूबॉक्स: डिब्बा.
\end{solution}

\begin{solution}{exo:g2:shapes:7}
ए \omterm{def:g2:shapes:solids}{घनक्षेत्र} है $6$ \omterm{def:g2:shapes:solids}{चेहरे}, प्रत्येक एक वर्ग। एक बॉक्स भी है $6$ \omterm{def:g2:shapes:solids}{चेहरे}
(आयत)।
\end{solution}

\begin{solution}{exo:g2:shapes:8}
गेंद लुढ़कती है (हर तरफ गोल) और \omterm{def:g2:shapes:solids}{सिलेंडर} रोल (उस पर)
घुमावदार पक्ष)। \omterm{def:g2:shapes:solids}{घनक्षेत्र} और बॉक्स नहीं है: उनके पास केवल फ्लैट है
\omterm{def:g2:shapes:solids}{चेहरे} और उन पर रुकें.
\end{solution}

\begin{solution}{exo:g2:shapes:9}
$6$ भुजाएँ: एक षट्भुज, के साथ $6$ कोने.
\end{solution}

\begin{solution}{exo:g2:shapes:10}
घर की रूपरेखा (\omterm{def:g1:shapes:def}{त्रिकोण} छत के साथ चौकोर दीवार, छत का आधार
वर्ग के ऊपरी भाग पर बैठा है) है $5$ पक्ष और $5$ कोने ---
यह एक पंचकोण (नीचे, दो दीवारें, दो छत ढलान; शीर्ष) है
दीवार छत के नीचे छिपी हुई है)।
\end{solution}
