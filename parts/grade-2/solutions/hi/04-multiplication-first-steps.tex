\section*{अध्याय \ref{ch:g2:mult} --- गुणन: पहला चरण}

\begin{solution}{exo:g2:mult:1}
$2 + 2 + 2 = 3 \times 2 = 6$; \quad
$5 + 5 + 5 + 5 = 4 \times 5 = 20$; \quad
$10 + 10 + 10 = 3 \times 10 = 30$.
\end{solution}

\begin{solution}{exo:g2:mult:2}
$3$ की पंक्तियाँ $5$: $3 \times 5 = 15$. अपनी तरफ कर दिया, $5$ की पंक्तियाँ
$3$: $5 \times 3 = 15$ --- जो उसी $15$.
\end{solution}

\begin{solution}{exo:g2:mult:3}
की तालिका $2$: $2, 4, 6, 8, 10, 12$. की तालिका $5$:
$5, 10, 15, 20, 25, 30$.
\end{solution}

\begin{solution}{exo:g2:mult:4}
$2 \times 7 = 14$; \quad $5 \times 3 = 15$; \quad
$10 \times 4 = 40$; \quad $2 \times 9 = 18$; \quad
$5 \times 6 = 30$.
\end{solution}

\begin{solution}{exo:g2:mult:5}
युगल: $12$; $16$; $20$. आधा भाग: आधा भाग $12$ है $6$; का आधा
$20$ है $10$.
\end{solution}

\begin{solution}{exo:g2:mult:6}
$3, 6, 9, 12, 15, 18$. इसलिए $3 \times 5 = 15$.
\end{solution}

\begin{solution}{exo:g2:mult:7}
$4 \times 5 = 20$ कुर्सियाँ.
\end{solution}

\begin{solution}{exo:g2:mult:8}
$3 \times 8 = 24$ पैर ($8, 16, 24$).
\end{solution}

\begin{solution}{exo:g2:mult:9}
$2$ की पंक्तियाँ $6$: $2 \times 6 = 12$. $3$ की पंक्तियाँ $4$:
$3 \times 4 = 12$.
\end{solution}

\begin{solution}{exo:g2:mult:10}
गिनती छोड़ें $5$: $5, 10, 15, 20$ --- तीन डिब्बे ही देते हैं $15$
(बहुत कम), चार डिब्बे देते हैं $20$. मिया को खरीदना होगा $4$ बक्से.
\end{solution}
