\end{solution}

\begin{solution}{exo:g2:subtraction:7}
$85 - 39 = 46$: जाँच करना $46 + 39 = 85$ --- सही।
$214 - 68 = 156$: जाँच करना $156 + 68 = 224 \neq 214$ --- गलत (द
सही उत्तर है $146$).
\end{solution}

\begin{solution}{exo:g2:subtraction:8}
``उड़ जाओ'': $47 - 19 = 28$. ``अधिक चुना गया'': $47 + 19 = 66$.
``और कितने'': $47 - 19 = 28$.
\end{solution}

\begin{solution}{exo:g2:subtraction:9}
से $45$ तक $63$ है $18$, इसलिए $63 - 18 = 45$: छेद है $18$.
\end{solution}

\begin{solution}{exo:g2:subtraction:10}
से पीछे की ओर कार्य करें $58$: ले जाने से पहले $18$, संख्या थी
$58 + 18 = 76$; जोड़ने से पहले $30$, वह था $76 - 30 = 46$. द
आरंभिक संख्या थी $46$ (जाँच करना: $46 + 30 - 18 = 58$).
\end{solution}
