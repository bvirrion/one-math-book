\section*{अध्याय \ref{ch:g2:addition} --- ले जाने के साथ जोड़}

\begin{solution}{exo:g2:addition:1}
$34 + 25 = 59$; \quad $126 + 43 = 169$; \quad $507 + 82 = 589$.
\end{solution}

\begin{solution}{exo:g2:addition:2}
$47 + 8 = 55$; \quad $56 + 27 = 83$; \quad $65 + 19 = 84$.
\end{solution}

\begin{solution}{exo:g2:addition:3}
$268 + 145 = 413$; \quad $374 + 158 = 532$; \quad
$429 + 293 = 722$.
\end{solution}

\begin{solution}{exo:g2:addition:4}
$38 + 5 = 38 + 2 + 3 = 43$; \quad $67 + 6 = 67 + 3 + 3 = 73$; \quad
$54 + 8 = 54 + 6 + 2 = 62$.
\end{solution}

\begin{solution}{exo:g2:addition:5}
$42 + 36 = 70 + 8 = 78$; \quad $53 + 24 = 70 + 7 = 77$; \quad
$61 + 28 = 80 + 9 = 89$.
\end{solution}

\begin{solution}{exo:g2:addition:6}
$58 + 34$: अनुमान लगाना $60 + 30 = 90$; बिल्कुल $92$.
$296 + 187$: अनुमान लगाना $300 + 200 = 500$; बिल्कुल $483$.
\end{solution}

\begin{solution}{exo:g2:addition:7}
$128 + 95 = 223$. उनके पास एक साथ है $223$ पत्थर.
\end{solution}

\begin{solution}{exo:g2:addition:8}
पहला नंबर प्लस $27$ है $64$, सो है $64 - 27 = 37$: द
लुप्त इकाई अंक है $7$ (संख्या है $37$).
\end{solution}

\begin{solution}{exo:g2:addition:9}
बच्चे: $27 + 28 = 55$; बड़ों के साथ, $55 + 6 = 61$
लोग।
\end{solution}

\begin{solution}{exo:g2:addition:10}
$27$ और $33$ इकाइयाँ हैं $7$ और $3$ (दस मित्र):
$27 + 33 = 60$, तब $60 + 40 = 100$. तीन नंबर हैं
$27$, $33$ और $40$.
\end{solution}
