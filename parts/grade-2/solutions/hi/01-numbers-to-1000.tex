\section*{अध्याय \ref{ch:g2:numbers} --- संख्या 1000 तक}

\begin{solution}{exo:g2:numbers:1}
$258$: $2$ सैकड़ों, $5$ दसियों, $8$ इकाइयाँ। $470$: $4$, $7$, $0$.
$600$: $6$, $0$, $0$. $909$: $9$, $0$, $9$.
\end{solution}
