\section*{अध्याय \ref{ch:g2:problems} --- समस्याओं को सुलझा रहा}

\begin{solution}{exo:g2:problems:1}
एक साथ रखें: \omterm{def:g1:addition:def}{जोड़}. $50$ मिठाई, $12$ खाया: घटाव. $4$
के बक्से $5$: गुणन. और कितने: घटाव.
\end{solution}

\begin{solution}{exo:g2:problems:2}
$156 - 48 = 108$. वहीं रहो $108$ गाड़ियाँ.
\end{solution}

\begin{solution}{exo:g2:problems:3}
$5 \times 4 = 20$ फूल.
\end{solution}

\begin{solution}{exo:g2:problems:4}
तुलना बार: टॉम $45$, मिया छोटा $17$:
$45 - 17 = 28$ कार्ड.
\end{solution}

\begin{solution}{exo:g2:problems:5}
बोतलें प्राप्त हुईं: $4 \times 6 = 24$; शेष:
$24 - 9 = 15$ बोतलें.
\end{solution}

\begin{solution}{exo:g2:problems:6}
बेकार संख्या आयु है ($8$). संगमरमर:
$23 + 15 = 38$.
\end{solution}

\begin{solution}{exo:g2:problems:7}
टार्ट्स: $7 + 10 = 17$. शनिवार, सभी केक:
$12 + 7 = 19$.
\end{solution}

\begin{solution}{exo:g2:problems:8}
लाल $8$, हरा $4$, पीला $6$: सभी में $8 + 4 + 6 = 18$ फल.
लाल के पास सबसे ज्यादा है.
\end{solution}

\begin{solution}{exo:g2:problems:9}
$80 - 35 = 45$ सेमी शेष.
\end{solution}

\begin{solution}{exo:g2:problems:10}
सुबह: $5 \times 4 = 20$; दोपहर: $3 \times 4 = 12$; कुल
$20 + 12 = 32$ केक. (या: $5 + 3 = 8$ की ट्रे $4$, और
$8 \times 4 = 32$.)
\end{solution}
