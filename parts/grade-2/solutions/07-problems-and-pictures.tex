\section*{Chapter \ref{ch:g2:problems} --- Solving Problems}

\begin{solution}{exo:g2:problems:1}
Put together: addition. $50$ sweets, $12$ eaten: subtraction. $4$
boxes of $5$: multiplication. How many more: subtraction.
\end{solution}

\begin{solution}{exo:g2:problems:2}
$156 - 48 = 108$. There stay $108$ cars.
\end{solution}

\begin{solution}{exo:g2:problems:3}
$5 \times 4 = 20$ flowers.
\end{solution}

\begin{solution}{exo:g2:problems:4}
Comparison bar: Tom $45$, Mia shorter by $17$:
$45 - 17 = 28$ cards.
\end{solution}

\begin{solution}{exo:g2:problems:5}
Bottles received: $4 \times 6 = 24$; remaining:
$24 - 9 = 15$ bottles.
\end{solution}

\begin{solution}{exo:g2:problems:6}
The useless number is the age ($8$). Marbles:
$23 + 15 = 38$.
\end{solution}

\begin{solution}{exo:g2:problems:7}
Tarts: $7 + 10 = 17$. Saturday, all cakes:
$12 + 7 = 19$.
\end{solution}

\begin{solution}{exo:g2:problems:8}
Red $8$, green $4$, yellow $6$: in all $8 + 4 + 6 = 18$ fruits.
Red has the most.
\end{solution}

\begin{solution}{exo:g2:problems:9}
$80 - 35 = 45$ cm remain.
\end{solution}

\begin{solution}{exo:g2:problems:10}
Morning: $5 \times 4 = 20$; afternoon: $3 \times 4 = 12$; total
$20 + 12 = 32$ cakes. (Or: $5 + 3 = 8$ trays of $4$, and
$8 \times 4 = 32$.)
\end{solution}
