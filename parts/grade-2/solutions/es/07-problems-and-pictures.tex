\section*{Capítulo \ref{ch:g2:problems} --- Solving Problems}

\begin{solution}{exo:g2:problems:1}
Juntar: suma. $50$ dulces, $12$ comidos: resta. $4$
casillas de $5$: multiplicación. Cuántos más: resta.
\end{solution}

\begin{solution}{exo:g2:problems:2}
$156 - 48 = 108$. Quedan $108$ autos.
\end{solution}

\begin{solution}{exo:g2:problems:3}
$5 \times 4 = 20$ flores.
\end{solution}

\begin{solution}{exo:g2:problems:4}
Barra de comparación: Tom $45$, Mia más corta por $17$:
$45 - 17 = 28$ tarjetas.
\end{solution}

\begin{solution}{exo:g2:problems:5}
Botellas recibidas: $4 \times 6 = 24$; restante:
$24 - 9 = 15$ botellas.
\end{solution}

\begin{solution}{exo:g2:problems:6}
El número inútil es la edad ($8$). Canicas:
$23 + 15 = 38$.
\end{solution}

\begin{solution}{exo:g2:problems:7}
Tartas: $7 + 10 = 17$. Sábado, todas las tartas:
$12 + 7 = 19$.
\end{solution}

\begin{solution}{exo:g2:problems:8}
Rojo $8$, verde $4$, amarillo $6$: en todos los frutos $8 + 4 + 6 = 18$.
El rojo tiene más.
\end{solution}

\begin{solution}{exo:g2:problems:10}
Mañana: $5 \times 4 = 20$; tarde: $3 \times 4 = 12$; total
$20 + 12 = 32$ pasteles. (O: $5 + 3 = 8$ bandejas de $4$, y
$8 \times 4 = 32$.)
\end{solution}
