\section*{Capítulo \ref{ch:g2:mult} --- Multiplication: First Steps}

\begin{solution}{exo:g2:mult:1}
$2 + 2 + 2 = 3 \times 2 = 6$; \quad
$5 + 5 + 5 + 5 = 4 \times 5 = 20$; \quad
$10 + 10 + 10 = 3 \times 10 = 30$.
\end{solution}

\begin{solution}{exo:g2:mult:2}
$3$ filas de $5$: $3 \times 5 = 15$. Girado de lado, $5$ filas de
$3$: $5 \times 3 = 15$ --- el mismo $15$.
\end{solution}

\begin{solution}{exo:g2:mult:3}
Tabla de $2$: $2, 4, 6, 8, 10, 12$. Tabla de $5$:
$5, 10, 15, 20, 25, 30$.
\end{solution}

\begin{solution}{exo:g2:mult:4}
$2 \times 7 = 14$; \quad $5 \times 3 = 15$; \quad
$10 \times 4 = 40$; \quad $2 \times 9 = 18$; \quad
$5 \times 6 = 30$.
\end{solution}

\begin{solution}{exo:g2:mult:5}
Dobles: $12$; $16$; $20$. Mitades: la mitad de $12$ es $6$; la mitad de
$20$ es $10$.
\end{solution}

\begin{solution}{exo:g2:mult:6}
$3, 6, 9, 12, 15, 18$. So $3 \times 5 = 15$.
\end{solution}

\begin{solution}{exo:g2:mult:7}
$4 \times 5 = 20$ sillas.
\end{solution}

\begin{solution}{exo:g2:mult:8}
$3 \times 8 = 24$ piernas ($8, 16, 24$).
\end{solution}

\begin{solution}{exo:g2:mult:9}
$2$ filas de $6$: $2 \times 6 = 12$. $3$ filas de $4$:
$3 \times 4 = 12$.
\end{solution}

\begin{solution}{exo:g2:mult:10}
Conteo de saltos por $5$: $5, 10, 15, 20$ --- tres casillas dan solo $15$
(muy pocos), cuatro casillas dan $20$. Mia debe comprar cajas $4$.
\end{solution}
