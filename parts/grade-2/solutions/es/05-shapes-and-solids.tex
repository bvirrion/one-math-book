\section*{Capítulo \ref{ch:g2:shapes} --- Shapes and Solids}

\begin{solution}{exo:g2:shapes:1}
$5$ lados, $5$ esquinas: un pentágono. Esquinas $3$ (y lados $3$): a
triángulo. Redondo: un círculo.
\end{solution}

\begin{solution}{exo:g2:shapes:2}
\omterm{def:g2:shapes:sort}{Polígonos}: cuadrado, triángulo, hexágono (lados rectos). No es un
\omterm{def:g2:shapes:sort}{polígono}: el círculo (borde curvo).
\end{solution}

\begin{solution}{exo:g2:shapes:3}
Ángulos rectos en, por ejemplo, las esquinas del tablero, un libro, un
ventana; un ángulo no recto en la tapa abierta de una caja o las manos de
un reloj a las $2$ en punto.
\end{solution}

\begin{solution}{exo:g2:shapes:4}
Cuadrado: $4$ ángulos rectos. Rectángulo: $4$. Un triángulo puede tener al
la mayoría de los ángulos rectos $1$ --- dos harían que los otros dos lados nunca
reunirse.
\end{solution}

\begin{solution}{exo:g2:shapes:5}
Copias de cuadrícula realizadas contando cuadrados entre esquinas; comparando
Con el modelo comprueba el trabajo.
\end{solution}

\begin{solution}{exo:g2:shapes:6}
Muere: \omterm{def:g2:shapes:solids}{cubo}. Lata de sopa: \omterm{def:g2:shapes:solids}{cilindro}. Fútbol: pelota. Caja de zapatos: caja.
\end{solution}

\begin{solution}{exo:g2:shapes:7}
Un \omterm{def:g2:shapes:solids}{cubo} tiene $6$ \omterm{def:g2:shapes:solids}{caras}, cada uno un cuadrado. Una caja también tiene $6$ \omterm{def:g2:shapes:solids}{caras}
(rectángulos).
\end{solution}

\begin{solution}{exo:g2:shapes:8}
La bola rueda (redondeada por todas partes) y el \omterm{def:g2:shapes:solids}{cilindro} rueda (sobre su
lado curvo). El \omterm{def:g2:shapes:solids}{cubo} y la caja no: solo tienen planos.
\omterm{def:g2:shapes:solids}{caras} y deténgase en ellos.
\end{solution}

\begin{solution}{exo:g2:shapes:9}
Lados $6$: un hexágono, con esquinas $6$.
\end{solution}

\begin{solution}{exo:g2:shapes:10}
El contorno de la casa (pared cuadrada con techo triangular, base del techo
sentado en el lado superior del cuadrado) tiene lados $5$ y esquinas $5$ ---
es un pentágono (abajo, dos paredes, dos vertientes del techo; la parte superior de
la pared está oculta bajo el techo).
\end{solution}
