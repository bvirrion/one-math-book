\section*{Capítulo \ref{ch:g2:measure} --- Money and Measures}

\begin{solution}{exo:g2:measure:1}
Lápiz: cm. Longitud del aula: m. Pluma: g. Mochila escolar: kg. agua
botella: l.
\end{solution}

\begin{solution}{exo:g2:measure:2}
$2$ m $= 200$ cm; \quad $300$ cm $= 3$ m; \quad $1$ kg $= 1000$ g.
\end{solution}

\begin{solution}{exo:g2:measure:3}
Por ejemplo: $16 = 10 + 5 + 1$; $23 = 20 + 2 + 1$; \
$35 = 20 + 10 + 5$. (Son posibles otras formas).
\end{solution}

\begin{solution}{exo:g2:measure:4}
$10 - 6 = 4$; \quad $20 - 17 = 3$; \quad $50 - 32 = 18$.
\end{solution}

\begin{solution}{exo:g2:measure:5}
Mano larga en $6$, corta entre $4$ y $5$: cuatro y media
($4{:}30$). Mano larga en $3$, corta justo después de $8$: y cuarto
ocho ($8{:}15$).
\end{solution}

\begin{solution}{exo:g2:measure:6}
Dos y media: la manecilla larga apunta hacia $6$, la manecilla corta
se encuentra entre $2$ y $3$.
\end{solution}

\begin{solution}{exo:g2:measure:7}
Después de mayo: junio. Antes de enero: diciembre. (El mes de cumpleaños varía).
\end{solution}

\begin{solution}{exo:g2:measure:8}
Gastado: $3 + 4 = 7$. Cambio: $10 - 7 = 3$.
\end{solution}

\begin{solution}{exo:g2:measure:9}
$1$ kg $= 1000$ g, que es más de $800$ g: el azúcar es
más pesado, en $1000 - 800 = 200$ g.
\end{solution}

\begin{solution}{exo:g2:measure:10}
Por ejemplo $20 + 10 + 5 + 5 = 40$ y $10 + 10 + 10 + 10 = 40$
(de dos maneras con exactamente cuatro monedas o billetes).
\end{solution}
