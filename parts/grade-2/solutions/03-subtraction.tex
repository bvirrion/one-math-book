\begin{solution}{exo:g2:subtraction:1}
$58 - 23 = 35$; \quad $176 - 45 = 131$; \quad $389 - 164 = 225$.
\end{solution}

\begin{solution}{exo:g2:subtraction:2}
$52 - 17 = 35$; \quad $73 - 28 = 45$; \quad $60 - 24 = 36$.
\end{solution}

\begin{solution}{exo:g2:subtraction:3}
$324 - 167 = 157$ (check $157 + 167 = 324$); \quad
$500 - 236 = 264$ (check $264 + 236 = 500$); \quad
$703 - 158 = 545$ (check $545 + 158 = 703$).
\end{solution}

\begin{solution}{exo:g2:subtraction:6}

\begin{solution}{exo:g2:subtraction:7}
$85 - 39 = 46$: check $46 + 39 = 85$ --- right.
$214 - 68 = 156$: check $156 + 68 = 224 \neq 214$ --- wrong (the
correct answer is $146$).
\end{solution}

\begin{solution}{exo:g2:subtraction:8}
``fly away'': $47 - 19 = 28$. ``more picked'': $47 + 19 = 66$.
``how many more'': $47 - 19 = 28$.
\end{solution}

\begin{solution}{exo:g2:subtraction:9}
From $45$ up to $63$ is $18$, so $63 - 18 = 45$: the hole is $18$.
\end{solution}

\begin{solution}{exo:g2:subtraction:10}
Work backward from $58$: before taking away $18$, the number was
$58 + 18 = 76$; before adding $30$, it was $76 - 30 = 46$. The
starting number was $46$ (check: $46 + 30 - 18 = 58$).
\end{solution}
