\section*{Chapter \ref{ch:g2:mult} --- Multiplication: First Steps}

\begin{solution}{exo:g2:mult:1}
$2 + 2 + 2 = 3 \times 2 = 6$; \quad
$5 + 5 + 5 + 5 = 4 \times 5 = 20$; \quad
$10 + 10 + 10 = 3 \times 10 = 30$.
\end{solution}

\begin{solution}{exo:g2:mult:2}
$3$ rows of $5$: $3 \times 5 = 15$. Turned on its side, $5$ rows of
$3$: $5 \times 3 = 15$ --- the same $15$.
\end{solution}

\begin{solution}{exo:g2:mult:3}
Table of $2$: $2, 4, 6, 8, 10, 12$. Table of $5$:
$5, 10, 15, 20, 25, 30$.
\end{solution}

\begin{solution}{exo:g2:mult:4}
$2 \times 7 = 14$; \quad $5 \times 3 = 15$; \quad
$10 \times 4 = 40$; \quad $2 \times 9 = 18$; \quad
$5 \times 6 = 30$.
\end{solution}

\begin{solution}{exo:g2:mult:5}
Doubles: $12$; $16$; $20$. Halves: half of $12$ is $6$; half of
$20$ is $10$.
\end{solution}

\begin{solution}{exo:g2:mult:6}
$3, 6, 9, 12, 15, 18$. So $3 \times 5 = 15$.
\end{solution}

\begin{solution}{exo:g2:mult:7}
$4 \times 5 = 20$ chairs.
\end{solution}

\begin{solution}{exo:g2:mult:8}
$3 \times 8 = 24$ legs ($8, 16, 24$).
\end{solution}

\begin{solution}{exo:g2:mult:9}
$2$ rows of $6$: $2 \times 6 = 12$. $3$ rows of $4$:
$3 \times 4 = 12$.
\end{solution}

\begin{solution}{exo:g2:mult:10}
Skip count by $5$: $5, 10, 15, 20$ --- three boxes give only $15$
(too few), four boxes give $20$. Mia must buy $4$ boxes.
\end{solution}
