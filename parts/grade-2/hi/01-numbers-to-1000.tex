\chapter{संख्या 1000 तक}\label{ch:g2:numbers}

ग्रेड 1 ने इकाइयों को दसियों में बाँटा (\cref{ch:g1:tens}); अब दस दहाई
एक में बंडल करें \emph{सौ}, और संख्याएँ सभी तरह से फैली हुई हैं
$1000$. तीन अंक, तीन कार्य: सैकड़ा, दहाई, इकाई।

\section{सैकड़ों, दसियों, इकाइयाँ}

\begin{definition}[तीन जगह]\label{def:g2:numbers:places}
ए \emph{सौ} दस दहाई है, यानी दस के दस बंडल। जो नंबर $346$ इसके तीन भाग हैं:
\[
346 = 3 \text{ hundreds} + 4 \text{ tens} + 6 \text{ units}
= 300 + 40 + 6 .
\]
\end{definition}

  \node[font=\large, omDef] at (3.45,0.425) {$4$};
  \node[font=\large, omDef] at (5.75,0.425) {$6$};
\end{tikzpicture}

{\small की स्थान तालिका $346$, पढ़ें ``तीन सौ छियालीस''।
(बड़ी संख्या की प्रतीक्षा है \cref{ch:g3:numbers}.)}
\end{omfigure}

\begin{example}[शून्य स्थान रखता है]\label{ex:g2:numbers:zero}
``तीन सौ पांच'' है $305$: तीन शतक, \emph{नहीं}
दसियों, पाँच इकाइयाँ। $0$ रखता है $5$ इकाइयों के स्थान पर. बिना
यह, $35$ बिल्कुल अलग संख्या होगी.
\end{example}

\begin{example}[का आदान प्रदान]\label{ex:g2:numbers:exchange}
दस इकाइयाँ दस बनाती हैं, दस दहाई सौ बनाती हैं। इसलिए $10$ दसियों
$= 100$, और $2$ सैकड़ों $= 20$ दसियों. जो नंबर $250$ रोकना
$25$ दहाई, केवल अंक नहीं $5$.
\end{example}

\section{तुलना करना और ऑर्डर करना}

\begin{method}[तीन अंकों की संख्याओं की तुलना करना]\label{met:g2:numbers:compare}
\begin{enumerate}
  \item अधिक अंकों से जीत: $103 > 98$;
  \item समान लंबाई: बाईं ओर से तुलना करें --- पहले सैकड़ों, फिर
        दहाई, फिर इकाई। पहला \omterm{ex:g1:subtraction:difference}{अंतर} तय करता है:
        $472 > 468$ क्योंकि दहाई में, $7 > 6$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g2:numbers:order}
आदेश $530$, $305$, $503$: सैकड़ा पहले ($5$, $3$, $5$), इसलिए
$305$ सुराग; बीच में $530$ और $503$, दहाई निर्णय लेते हैं ($3 > 0$):
\[
305 < 503 < 530 .
\]
\end{example}

\section{संख्या रेखा पर कूदता है}

\begin{method}[दहाई और सैकड़ा जोड़ना]\label{met:g2:numbers:jumps}
लाइन पर:
\begin{enumerate}
  \item $+1$ / $-1$: इकाइयाँ बदलें;
  \item $+10$ / $-10$: दहाई बदलें ($362 + 10 = 372$);
  \item $+100$ / $-100$: सैकड़ों बदलें ($362 + 100 = 462$).
\end{enumerate}
हर बार केवल एक अंक चलता है --- जब तक कि कोई स्थान ``भर न जाए'' और
ले जाता है, जैसे $195 + 10 = 205$.
\end{method}

\begin{omfigure}
\begin{tikzpicture}[scale=1.0]
  \draw[-Stealth] (-0.3,0) -- (10.6,0);
  \foreach \x/\l in {0.5/{362}, 5.5/{372}, 9.5/{472}}
    { \draw (\x,-0.1) -- (\x,0.1);
      \node[below, font=\small] at (\x,-0.12) {$\l $}; }
  \draw[-Stealth, omDef, thick] (0.5,0.28) .. controls (2.5,1.0) ..
    (5.5,0.28);
  \node[omDef, font=\small] at (3.0,1.05) {$+10$};
  \draw[-Stealth, omThm, thick] (5.5,0.28) .. controls (7.5,1.05) ..
    (9.5,0.28);
  \node[omThm, font=\small] at (7.5,1.1) {$+100$};
\end{tikzpicture}

{\small से कूदता है $362$: की एक छोटी सी छलांग $+10$ दहाई बदलता है, ए
की बड़ी आशा $+100$ सैकड़ों को बदल देता है।}
\end{omfigure}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g2:numbers:1}
इसमें कितने सैकड़े, दहाई और इकाई हैं: $258$; \ $470$; \ $600$; \
$909$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:2}
संख्या लिखें: $4$ सैकड़ों, $0$ दसियों, $7$ इकाइयाँ; \ $6$ सैकड़ों
और $3$ दसियों; \ दस सौ.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:3}
अंकों में लिखें: दो सौ सोलह; \ पाँच सौ चार; \
सात सौ नब्बे.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:4}
के रूप में विघटित करें \cref{def:g2:numbers:places}: $438$; \ $702$; \
$560$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:5}
कॉपी करें और पूरा करें $<$ या $>$:
\[
327 \;?\; 372, \qquad
580 \;?\; 508, \qquad
199 \;?\; 200, \qquad
640 \;?\; 604 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:6}
सबसे छोटे से सबसे बड़े तक ऑर्डर करें: $430$; \ $304$; \ $340$; \ $403$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:7}
लाइन पर कूदकर गणना करें: $253 + 10$; \quad $253 + 100$;
\quad $480 - 10$; \quad $716 - 100$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:8}
भरोसा रखें: $195, 205, 215, ?, ?$. \quad
$300, 400, 500, ?, ?$. \quad पहले वाले से सावधान!
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:9}
एक दुकान है $6$ के बक्से $100$ पेंसिल और $5$ ढीली पेंसिलें. कैसे
कुल मिलाकर बहुत सारी पेंसिलें?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g2:numbers:10}
अंकों का उपयोग करना $2$, $5$ और $8$ प्रत्येक एक बार, सबसे बड़ा लिखें
तीन अंकों की संख्या और सबसे छोटी संख्या। वे कितनी दूर हैं?
(आप उन्हें छलांग लगाकर लाइन में \omterm{def:g1:addition:def}{जोड़} सकते हैं।)
\end{exercise}
