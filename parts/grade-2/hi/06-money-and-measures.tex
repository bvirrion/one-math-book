\chapter{धन और उपाय}\label{ch:g2:measure}

मापने के शासक, रसोई के तराजू, जग, सिक्के, घड़ियाँ आदि
कैलेंडर: यह अध्याय रोजमर्रा की जिंदगी की संख्याओं के बारे में है ---
यह कितना लंबा है, कितना भारी है, यह कितना धारण कर सकता है, इसकी कीमत कितनी है और क्या
समय यह है.

\section{लंबाई, द्रव्यमान, क्षमता}

\begin{definition}[प्रतिदिन इकाइयाँ]\label{def:g2:measure:units}
\begin{itemize}
  \item \emph{लंबाई}: छोटी चीज़ों के लिए सेंटीमीटर (सेमी), मीटर (एम)
        बड़े लोगों के लिए, के साथ $1$ एम $= 100$ सेमी;
  \item \emph{द्रव्यमान}\index{द्रव्यमान}: हल्की चीजों के लिए ग्राम (जी), किलोग्राम (किग्रा) के लिए
        भारी वाले, के साथ $1$ किग्रा $= 1000$ जी;
  \item \emph{क्षमता}\index{क्षमता} (एक कंटेनर में कितना सामान है): लीटर (एल)।
\end{itemize}
छोटी इकाइयाँ और उनके सटीक नियम आते हैं \cref{ch:g3:measure}.
\end{definition}

\begin{example}[कौन सी इकाई?]\label{ex:g2:measure:choose}
एक क्रेयॉन: के बारे में $9$ सेमी। एक दरवाजा: के बारे में $2$ एम। एक सेब: के बारे में $150$
जी। आटे का एक बड़ा बैग: $1$ किग्रा. एक दूध की बोतल: $1$ एल. इकाई का अनुमान लगाएं
मापने से पहले --- यह ``दरवाजा है'' जैसे मूर्खतापूर्ण उत्तर देना बंद कर देता है $2$
सेमी''।
\end{example}

\section{धन}

\begin{method}[रकम बनाना]\label{met:g2:measure:money}
के सिक्कों और बिलों के साथ $1$, $2$, $5$, $10$, $20$:
\begin{enumerate}
  \item सबसे बड़े से शुरू करें जो फिट बैठता है, फिर बाकी भरें;
  \item बनाने के लिए $27$: ए $20$, ए $5$, और ए $2$;
  \item अक्सर कई तरीके होते हैं --- $27$ ई आल्सो $10 + 10 + 5 +
        2$.
\end{enumerate}
\end{method}

\begin{example}[परिवर्तन]\label{ex:g2:measure:change}
एक किताब की कीमत होती है $14$; आप एक के साथ भुगतान करते हैं $20$. परिवर्तन है $20 - 14 =
6$, गिनती करने पर पता चला: $14 \to 20$ है $6$
(\cref{met:g2:subtraction:countup}).
\end{example}

\section{समय}

\begin{method}[आधा पौना घंटा]\label{met:g2:measure:clock}
लंबी सुई घंटे के बाद मिनट बताती है:
\begin{enumerate}
  \item सीधे ऊपर ($12$): सटीक समय पर --- $3{:}00$;
  \item दाईं ओर चौथाई मोड़ ($3$): पिछली तिमाही --- $3{:}15$;
  \item सीधे नीचे ($6$): आधा बीतना --- $3{:}30$;
  \item तीन-चौथाई मोड़ ($9$): चौथाई से अगले घंटे ---
        $3{:}45$, ``पौने चार बजे''।
\end{enumerate}
\end{method}

\end{omfigure}

\begin{example}[द कैलेंडर]\label{ex:g2:measure:calendar}
एक साल हो गया है $12$ महीने, क्रम में: जनवरी, फरवरी, मार्च, अप्रैल,
मई, जून, जुलाई, अगस्त, सितंबर, अक्टूबर, नवंबर, दिसंबर।
अधिकांश महीनों में है $30$ या $31$ दिन; फरवरी है $28$ (या $29$).
एक तारीख दिन और महीने का नाम देती है: ``द $5$ मार्च''
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g2:measure:1}
इकाई चुनें (सेमी, मी, जी, किग्रा या एल): एक पेंसिल; कक्षा
लंबाई; एक पंख; स्कूल बैग; पानी की बोतल।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:2}
बदलना: $2$ मी में सेमी; $300$ सेमी में मी; $1$ किग्रा में जी.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:3}
सिक्कों और बिलों से ये रकम बनाएं: $16$; \quad $23$; \quad
$35$. प्रत्येक के लिए एक रास्ता बताएं.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:4}
परिवर्तन की गणना करें: भुगतान करें $10$ एक के लिए $6$ खिलौना; वेतन $20$ एक के लिए $17$
किताब; वेतन $50$ एक के लिए $32$ खेल।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:5}
समय पढ़ें: लंबा हाथ $6$, बीच में छोटा हाथ $4$ और $5$;
लंबा हाथ $3$, ठीक बाद में छोटा हाथ $8$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:6}
ढाई बजे दिखाने वाली एक घड़ी बनाएं। प्रत्येक हाथ कहाँ इंगित करता है?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:7}
महीने का नाम बताएं: मई के बाद; जनवरी से पहले; आपका महीना
जन्मदिन.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:8}
ली एक पेय खरीदती है $3$ और केक के लिए $4$. वह एक के साथ भुगतान करती है $10$
बिल। कितना बदलाव?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:9}
कौन सा भारी है: ए $1$ चीनी का किलो बैग या $800$ सेब का ग्राम? कैसे
कई ग्राम?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g2:measure:10}
बनाना $40$ ठीक चार सिक्कों या बिलों के साथ (इनमें से चुना गया)। $1$, $2$,
$5$, $10$, $20$). दो अलग-अलग तरीके खोजें.
\end{exercise}
