\chapter{Money and Measures}\label{ch:g2:measure}

Measuring rulers, kitchen scales, jugs, coins, clocks and
calendars: this chapter is about the numbers of everyday life ---
how long, how heavy, how much it holds, how much it costs, and what
time it is.

\section{Lengths, masses, capacities}

\begin{definition}[Everyday units]\label{def:g2:measure:units}
\begin{itemize}
  \item \emph{Length}: centimeter (cm) for small things, meter (m)
        for big ones, with $1$ m $= 100$ cm;
  \item \emph{mass}: gram (g) for light things, kilogram (kg) for
        heavy ones, with $1$ kg $= 1000$ g;
  \item \emph{capacity} (how much a container holds): liter (L).
\end{itemize}
Smaller units and their exact rules come in \cref{ch:g3:measure}.
\end{definition}

\begin{example}[Which unit?]\label{ex:g2:measure:choose}
A crayon: about $9$ cm. A door: about $2$ m. An apple: about $150$
g. A big bag of flour: $1$ kg. A milk bottle: $1$ L. Guess the unit
before measuring --- it stops silly answers like ``the door is $2$
cm''.
\end{example}

\section{Money}

\begin{method}[Making an amount]\label{met:g2:measure:money}
With coins and bills of $1$, $2$, $5$, $10$, $20$:
\begin{enumerate}
  \item start with the biggest that fits, then fill the rest;
  \item to make $27$: a $20$, a $5$, and a $2$;
  \item there are often several ways --- $27$ is also $10 + 10 + 5 +
        2$.
\end{enumerate}
\end{method}

\begin{example}[Change]\label{ex:g2:measure:change}
A book costs $14$; you pay with a $20$. The change is $20 - 14 =
6$, found by counting up: $14 \to 20$ is $6$
(\cref{met:g2:subtraction:countup}).
\end{example}

\section{Time}

\begin{method}[Half and quarter hours]\label{met:g2:measure:clock}
The long hand tells the minutes past the hour:
\begin{enumerate}
  \item straight up ($12$): on the hour --- $3{:}00$;
  \item quarter turn to the right ($3$): quarter past --- $3{:}15$;
  \item straight down ($6$): half past --- $3{:}30$;
  \item three-quarter turn ($9$): quarter to the next hour ---
        $3{:}45$, ``quarter to four''.
\end{enumerate}
\end{method}

\begin{example}[The calendar]\label{ex:g2:measure:calendar}
A year has $12$ months, in order: January, February, March, April,
May, June, July, August, September, October, November, December.
Most months have $30$ or $31$ days; February has $28$ (or $29$).
A date names the day and the month: ``the $5$th of March''.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g2:measure:1}
Choose the unit (cm, m, g, kg or L): a pencil; the classroom
length; a feather; a schoolbag; a water bottle.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:2}
Convert: $2$ m into cm; $300$ cm into m; $1$ kg into g.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:3}
Make these amounts with coins and bills: $16$; \quad $23$; \quad
$35$. Give one way for each.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:4}
Compute the change: pay $10$ for a $6$ toy; pay $20$ for a $17$
book; pay $50$ for a $32$ game.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:5}
Read the time: long hand on $6$, short hand between $4$ and $5$;
long hand on $3$, short hand just after $8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:6}
Draw a clock showing half past two. Where does each hand point?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:7}
Name the month: after May; before January; the month of your
birthday.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:8}
Lea buys a drink for $3$ and a cake for $4$. She pays with a $10$
bill. How much change?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:9}
Which is heavier: a $1$ kg bag of sugar or $800$ g of apples? By how
many grams?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:measure:10}
Make $40$ with exactly four coins or bills (chosen from $1$, $2$,
$5$, $10$, $20$). Find two different ways.
\end{exercise}
