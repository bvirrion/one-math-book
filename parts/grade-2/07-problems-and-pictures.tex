\chapter{Solving Problems}\label{ch:g2:problems}

A story with numbers hides a question: add? take away? multiply?
This chapter gives a way to decide, uses simple bar pictures to
\emph{see} the operation, and reads information from tables and
pictograms.

\section{Choosing the operation}

\begin{method}[Which operation?]\label{met:g2:problems:choose}
\begin{enumerate}
  \item \emph{putting together, in all}: add ($+$);
  \item \emph{taking away, how many left, how many more}: subtract
        ($-$);
  \item \emph{so many equal groups}: multiply ($\times$);
  \item draw a quick picture when unsure, and always answer with a
        sentence.
\end{enumerate}
\end{method}

\begin{example}[A two-step problem]\label{ex:g2:problems:twostep}
``Sam buys $3$ packs of $6$ stickers, then gives $5$ away. How many
does he keep?''
\begin{enumerate}
  \item Groups: $3 \times 6 = 18$ stickers;
  \item give away: $18 - 5 = 13$.
\end{enumerate}
Sam keeps $13$ stickers. Reread the question: is the story really
finished? Here yes.
\end{example}

\begin{example}[A hidden extra number]\label{ex:g2:problems:extra}
``A $210$-page book\,\dots'' --- careful, not every number in a
story is needed. ``Ben has $8$ red and $6$ blue marbles and is $7$
years old. How many marbles?'' The age is a decoy: $8 + 6 = 14$
marbles. Sort the numbers before computing.
\end{example}

\section{Tables and pictograms}

\begin{example}[A table]\label{ex:g2:problems:table}
Cakes sold at a stall:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|ccc}
 & Saturday & Sunday & total \\
\hline
muffins & $12$ & $9$ & $21$ \\
tarts & $7$ & $10$ & $17$ \\
\end{tabular}
\end{center}
Reading: Sunday tarts, $10$. Computing: muffins over the two days,
$12 + 9 = 21$; more tarts on Sunday than Saturday by $10 - 7 = 3$.
\end{example}

\begin{example}[A pictogram]\label{ex:g2:problems:pictogram}
Read the key first: here one apple stands for $2$ pieces of fruit.

\begin{omfigure}
\begin{tikzpicture}[scale=0.62]
  \node[left, font=\small] at (0,2) {red};
  \foreach \i in {0,1,2,3} \fill[omThm] (0.5+\i*0.75,2) circle (0.24);
  \node[left, font=\small] at (0,1) {green};
  \foreach \i in {0,1} \fill[omMeth] (0.5+\i*0.75,1) circle (0.24);
  \node[left, font=\small] at (0,0) {yellow};
  \foreach \i in {0,1,2} \fill[omProp] (0.5+\i*0.75,0) circle (0.24);
  \node[right, font=\small] at (4.0,1) {key: one dot $= 2$ fruits};
\end{tikzpicture}

{\small A pictogram: red shows $4$ dots, so $4 \times 2 = 8$ red
fruits; green $2$ dots, $4$ fruits; yellow $3$ dots, $6$ fruits.}
\end{omfigure}
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g2:problems:1}
Name the operation (do not compute): ``$34$ and $28$ marbles put
together''; ``$50$ sweets, $12$ eaten''; ``$4$ boxes of $5$
apples''; ``Ana has $30$, Leo has $18$: how many more has Ana?''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:2}
Solve: ``A car park has $156$ cars; $48$ leave. How many stay?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:3}
Solve: ``$5$ children each pick $4$ flowers. How many flowers in
all?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:4}
Draw a bar picture, then solve: ``Tom has $45$ cards, Mia has $17$
fewer. How many has Mia?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:5}
Two steps: ``A crate holds $6$ bottles. A shop gets $4$ crates and
sells $9$ bottles. How many bottles remain?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:6}
Find and cross out the useless number, then solve: ``Ben is $8$
years old and has $23$ marbles. He wins $15$ more. How many marbles
now?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:7}
Using the table of \cref{ex:g2:problems:table}: how many tarts sold
over the two days? How many cakes of all kinds on Saturday?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:8}
Using the pictogram of \cref{ex:g2:problems:pictogram}: how many
fruits in all? Which colour has the most?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:9}
Solve: ``A ribbon is $80$ cm long. Lea cuts off $35$ cm. How long is
the piece that remains?''
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:problems:10}
``A baker makes $5$ trays of $4$ cakes in the morning and $3$ trays
of $4$ cakes in the afternoon. How many cakes that day?'' (Two
multiplications, then a sum --- or count the trays first!)
\end{exercise}
