\chapter{Problemen oplossen}\label{ch:g2:problems}

Een verhaal met getallen verbergt een vraag: optellen? aftrekken?
vermenigvuldigen? Dit hoofdstuk geeft een manier om te kiezen,
gebruikt eenvoudige staafjesplaatjes om de bewerking te
\emph{zien}, en leest informatie uit tabellen en pictogrammen.

\section{De bewerking kiezen}

\begin{method}[Welke bewerking?]\label{met:g2:problems:choose}
\begin{enumerate}
  \item \emph{bij elkaar, in totaal}: optellen ($+$);
  \item \emph{wegnemen, hoeveel over, hoeveel meer}: aftrekken
        ($-$);
  \item \emph{zoveel gelijke groepen}: vermenigvuldigen ($\times$);
  \item teken een snelle schets als je twijfelt, en antwoord altijd
        met een zin.
\end{enumerate}
\end{method}

\begin{example}[Een probleem in twee stappen]\label{ex:g2:problems:twostep}
``Sam koopt $3$ pakjes van $6$ stickers, en geeft er dan $5$ weg.
Hoeveel houdt hij over?''
\begin{enumerate}
  \item Groepen: $3 \times 6 = 18$ stickers;
  \item weggeven: $18 - 5 = 13$.
\end{enumerate}
Sam houdt $13$ stickers over. Lees de vraag opnieuw: is het verhaal
echt klaar? Hier wel.
\end{example}

\begin{example}[Een verborgen extra getal]\label{ex:g2:problems:extra}
``Een boek van $210$ bladzijden\,\dots'' --- let op, niet elk getal
in een verhaal is nodig. ``Ben heeft $8$ rode en $6$ blauwe
knikkers en is $7$ jaar oud. Hoeveel knikkers?'' De leeftijd is een
lokmiddel: $8 + 6 = 14$ knikkers. Sorteer de getallen v\'{o}\'{o}r het
rekenen.
\end{example}

\section{Tabellen en pictogrammen}

\begin{example}[Een tabel]\label{ex:g2:problems:table}
Verkochte cakes bij een kraam:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|ccc}
 & zaterdag & zondag & totaal \\
\hline
muffins & $12$ & $9$ & $21$ \\
taartjes & $7$ & $10$ & $17$ \\
\end{tabular}
\end{center}
Lezen: zondag taartjes, $10$. Rekenen: muffins over de twee dagen,
$12 + 9 = 21$; meer taartjes op zondag dan zaterdag met
$10 - 7 = 3$.
\end{example}

\begin{example}[Een pictogram]\label{ex:g2:problems:pictogram}
Lees eerst de legenda: hier staat \'{e}\'{e}n appel voor $2$ stuks fruit.

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g2:problems:1}
Noem de bewerking (reken niet uit): ``$34$ en $28$ knikkers bij
elkaar''; ``$50$ snoepjes, er worden er $12$ opgegeten''; ``$4$
dozen van $5$ appels''; ``Ana heeft $30$, Leo heeft $18$:
hoeveel meer heeft Ana?''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:2}
Los op: ``Een parkeerplaats heeft $156$ auto's; er vertrekken er
$48$. Hoeveel blijven er?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:3}
Los op: ``$5$ kinderen plukken elk $4$ bloemen. Hoeveel bloemen in
totaal?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:4}
Teken een staafjesplaatje, los dan op: ``Tom heeft $45$ kaarten,
Mia heeft er $17$ minder. Hoeveel heeft Mia?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:5}
Twee stappen: ``Een krat bevat $6$ flessen. Een winkel krijgt $4$
kratten en verkoopt $9$ flessen. Hoeveel flessen blijven er over?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:6}
Zoek en streep het nutteloze getal door, los dan op: ``Ben is $8$
jaar oud en heeft $23$ knikkers. Hij wint er $15$ bij. Hoeveel
knikkers nu?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:7}
Met de tabel van \cref{ex:g2:problems:table}: hoeveel taartjes
verkocht over de twee dagen? Hoeveel cakes van alle soorten op
zaterdag?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:8}
Met het pictogram van \cref{ex:g2:problems:pictogram}: hoeveel
stuks fruit in totaal? Welke kleur heeft de meeste?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:problems:9}
Los op: ``Een lint is $80$ cm lang. Lea knipt er $35$ cm af. Hoe
lang is het stuk dat overblijft?''
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:problems:10}
``Een bakker maakt 's ochtends $5$ bladen van $4$ cakes en 's
middags $3$ bladen van $4$ cakes. Hoeveel cakes die dag?'' (Twee
vermenigvuldigingen, dan een som --- of tel eerst de bladen!)
\end{exercise}
