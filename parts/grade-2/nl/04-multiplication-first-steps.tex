\chapter{Vermenigvuldigen: eerste stappen}\label{ch:g2:mult}

Vier dozen van drie cakes: in plaats van $3 + 3 + 3 + 3$ schrijven
we $4 \times 3$. Vermenigvuldigen is de snelweg voor het optellen
van hetzelfde getal vele keren --- en het mooiste plaatje is een
net rooster van rijen en kolommen.

\section{Hetzelfde getal optellen}

\begin{definition}[Vermenigvuldiging]\label{def:g2:mult:def}
$4 \times 3$ betekent ``vier keer drie'':
\[
4 \times 3 = 3 + 3 + 3 + 3 = 12 .
\]
Het teken is $\times$, gelezen als ``keer''. Het antwoord is het
\emph{product}\index{product} (vermenigvuldigen groeit verder in
\cref{ch:g3:mult}).
\end{definition}

\begin{omfigure}
\begin{tikzpicture}[scale=0.5]
  \foreach \r in {0,1,2,3}
    \foreach \c in {0,1,2}
      \fill[omDef] (\c,-\r) circle (0.28);
  \node[right, font=\small] at (3.0,-1.5)
    {$4$ rijen van $3$ stippen: $4 \times 3 = 12$};
\end{tikzpicture}

{\small Een \emph{rooster}: $4$ rijen van $3$. Tellen per rij geeft
$4 \times 3$; dezelfde stippen per kolom geven $3 \times 4$ --- dus
de volgorde doet er niet toe.}
\end{omfigure}

\begin{example}[De volgorde doet er niet toe]\label{ex:g2:mult:order}
$4 \times 3$ en $3 \times 4$ tellen dezelfde $12$ stippen, de ene
per rij, de andere per kolom. Het rooster op zijn kant zetten
bewijst het. Dus \'{e}\'{e}n \omterm{def:g2:mult:def}{product} kennen geeft de tweeling gratis.
\end{example}

\section{Hopsgewijs tellen en de makkelijke tafels}

\begin{method}[Een tafel opbouwen door hopsgewijs te tellen]\label{met:g2:mult:skip}
Om $5 \times 4$ te vinden, tel vijf keer met vieren:
$4, 8, 12, 16, 20$. Hopsgewijs tellen op de getallenlijn
\emph{is} de tafel.
\begin{itemize}
  \item Tafel van $2$: $2, 4, 6, 8, 10, \dots$ (de verdubbelingen);
  \item tafel van $5$: $5, 10, 15, 20, 25, \dots$ (eindigt op $5$
        of $0$);
  \item tafel van $10$: $10, 20, 30, \dots$ (voeg gewoon een \omterm{def:g1:counting:zero}{nul}
        toe).
\end{itemize}
\end{method}

{\small De tafel van $2$ als sprongen op de lijn: $2, 4, 6, 8, 10$.
Vijf sprongen van $2$ bereiken $10$, dus $5 \times 2 = 10$.}
\end{omfigure}

\begin{example}[Verdubbelen en halveren]\label{ex:g2:mult:double}
Vermenigvuldigen met $2$ is verdubbelen: $2 \times 7 = 14$. Het
omgekeerde, halveren, maakt het ongedaan: de helft van $14$ is $7$.
Verdubbelingen uit het hoofd maken de hele tafel van $2$ direct ---
en bereiden delen voor (\cref{ch:g3:division}).
\end{example}

\section{Vermenigvuldigen in sommen}

\begin{example}\label{ex:g2:mult:problem}
``Een doos bevat $5$ eieren. Hoeveel eieren in $4$ dozen?'' Vier
groepen van vijf: $4 \times 5 = 20$ (hopsgewijs: $5, 10, 15, 20$).
Er zijn $20$ eieren.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g2:mult:1}
Schrijf als vermenigvuldiging, reken dan uit: $2 + 2 + 2$; \quad
$5 + 5 + 5 + 5$; \quad $10 + 10 + 10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:2}
Teken een rooster van $3$ rijen van $5$ stippen. Welke
vermenigvuldiging toont het? En op de zijkant gekeerd?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:3}
Tel hopsgewijs om de tafel van $2$ tot $2 \times 6$ te vullen; de
tafel van $5$ tot $5 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:4}
Reken uit: $2 \times 7$; \quad $5 \times 3$; \quad $10 \times 4$;
\quad $2 \times 9$; \quad $5 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:5}
Reken de verdubbelingen: dubbel van $6$; dubbel van $8$; dubbel van
$10$. Dan de helften: helft van $12$; helft van $20$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:6}
Reken de tafel van $3$ door hopsgewijs te tellen:
$3, 6, 9, ?, ?, ?$. Wat is $3 \times 5$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:7}
``Er staan $4$ tafels met elk $5$ stoelen. Hoeveel stoelen?''
Schrijf de vermenigvuldiging en het antwoord.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:8}
Een spin heeft $8$ poten. Hoeveel poten hebben $3$ spinnen? (Tel
hopsgewijs met $8$: $8, 16, \dots$)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:9}
Twee manieren om $12$ stippen in gelijke rijen te leggen: teken
$2$ rijen van $6$, en $3$ rijen van $4$. Schrijf de
vermenigvuldiging voor elk.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:mult:10}
Cakes komen in dozen van $5$. Mia heeft $18$ cakes nodig voor een
feest. Hoeveel dozen moet ze kopen? (Tel hopsgewijs met $5$ tot je
$18$ voorbij bent.)
\end{exercise}
