\chapter{Aftrekken}\label{ch:g2:subtraction}

Als de eenheden boven te weinig zijn om af te trekken, breken we een
tiental open --- het \emph{lenen}, de spiegel van het onthouden.
Dit hoofdstuk legt het uit met stokjes, oefent daarna de
schriftelijke aftrekking, met vooruit tellen als vriendelijke
snelweg.

\section{Waarom we lenen}

\begin{example}[Een tiental openmaken]\label{ex:g2:subtraction:unbundle}
$42 - 17$: we willen $7$ eenheden wegnemen van $2$ eenheden --- te
weinig! Dus maak \'{e}\'{e}n van de $4$ tientallen los tot tien losse
eenheden: nu heeft de bovenste rij $3$ tientallen en $12$ eenheden.
Wegnemen: $12 - 7 = 5$ eenheden, en $3 - 1 = 2$ tientallen. Dus
$42 - 17 = 25$.
\end{example}

\section{De schriftelijke methode}

\begin{method}[Kolomsgewijs aftrekken]\label{met:g2:subtraction:column}
\begin{enumerate}
  \item Schrijf het grotere getal boven, rechts uitgelijnd;
  \item trek de eenheden af; als het bovenste cijfer te klein is,
        leen een tiental (het wordt tien eenheden in de huidige
        kolom);
  \item trek de tientallen af, daarna de honderdtallen;
  \item \emph{controle}: het antwoord plus het weggenomen getal
        moet het bovenste getal teruggeven.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g2:subtraction:worked}
Reken uit $503 - 168$:
\[
\begin{array}{r}
5\,0\,3 \\
-\ 1\,6\,8 \\
\hline
3\,3\,5
\end{array}
\]
Eenheden: $3 - 8$ te klein; leen via de tientallen (die zijn er
niet, dus leen eerst van de honderdtallen): de bovenste rij wordt
$4$ honderdtallen, $9$ tientallen, $13$ eenheden. Dan $13 - 8 = 5$;
$9 - 6 = 3$; $4 - 1 = 3$. Controle: $335 + 168 = 503$. \checkmark
\end{example}

\section{Vooruit tellen}

\begin{method}[Aftrekken door vooruit te tellen]\label{met:g2:subtraction:countup}
Als de getallen dicht bij elkaar liggen, klim van het kleine naar
het grote en tel de sprongen op:
\[
83 - 78: \quad 78 \to 80 \text{ is } 2, \ 80 \to 83 \text{ is } 3,
\text{ dus } 83 - 78 = 5 .
\]
Vooruit tellen is sneller dan lenen als de twee getallen dicht bij
elkaar liggen --- en zo geeft een winkelier wisselgeld.
\end{method}

\begin{example}[Wisselgeld geven]\label{ex:g2:subtraction:change}
Een speelgoed kost $34$; je betaalt met een $50$. Wisselgeld: tel
vooruit, $34 \to 40$ is $6$, $40 \to 50$ is $10$, dus het
wisselgeld is $16$. De aftrekking $50 - 34 = 16$, op de manier van
de winkelier.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:1}
Reken uit (geen lenen nodig): $58 - 23$; \quad $176 - 45$; \quad
$389 - 164$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:2}
Reken uit met lenen: $52 - 17$; \quad $73 - 28$; \quad
$60 - 24$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:3}
Reken kolomsgewijs uit en controleer elk antwoord: $324 - 167$;
\quad $500 - 236$; \quad $703 - 158$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:4}
Reken uit door vooruit te tellen: $62 - 58$; \quad $91 - 85$;
\quad $100 - 96$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:5}
Bereken het wisselgeld (tel vooruit): betaal $50$ voor een boek van
$42$; betaal $20$ voor een speelgoed van $13$; betaal $100$ voor
een tas van $75$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:6}
``Een boom is $156$ cm hoog. Hij groeide $28$ cm dit jaar. Hoe hoog
was hij vorig jaar?'' Schrijf de aftrekking en het antwoord.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:7}
Vul aan met de tweelingoptelling om te controleren:
$85 - 39 = 46$; klopt het? $214 - 68 = 156$; klopt het?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:8}
Kies voor elk verhaal $+$ of $-$: ``$47$ vogels, er vliegen er $19$
weg''; ``$47$ appels, er worden er $19$ meer geplukt''; ``Sam
heeft $47$, Lea heeft $19$: hoeveel meer heeft Sam?''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:9}
Vul het gat: $63 - \;?\; = 45$. (Tel vooruit van $45$ naar $63$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:subtraction:10}
Een spel: bedenk een getal, tel er $30$ bij, trek er dan $18$ af.
Het resultaat is $58$. Wat was het getal? (Werk achteruit!)
\end{exercise}
