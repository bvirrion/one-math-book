\chapter{Optellen met onthouden}\label{ch:g2:addition}

Als de eenheden van een \omterm{def:g1:addition:def}{som} over de negen gaan, bundelen tien ervan
tot een nieuw tiental --- en die kleine bundel, verplaatst naar de
volgende kolom, is het beroemde \emph{onthouden}. Dit hoofdstuk legt
het onthouden uit met stokjes, en oefent daarna de schriftelijke
methode.

\section{Waarom we onthouden}

\begin{example}[Tien eenheden bundelen]\label{ex:g2:addition:bundle}
$27 + 15$: doe de eenheden bij elkaar, $7 + 5 = 12$ eenheden. Maar
tien eenheden maken een tiental! Bundel er tien: er blijven $2$
losse eenheden over en \'{e}\'{e}n nieuw tiental om te onthouden. Nu de
tientallen: $2 + 1 + 1 = 4$ tientallen. Dus $27 + 15 = 42$.
\end{example}

\section{De schriftelijke methode}

\begin{method}[Kolomsgewijs optellen]\label{met:g2:addition:column}
\begin{enumerate}
  \item Schrijf de getallen onder elkaar, eenheden onder eenheden,
        tientallen onder tientallen;
  \item tel de eenheden op; als het antwoord over $9$ gaat, schrijf
        het eenhedencijfer en onthoud een kleine $1$ boven de
        tientallen;
  \item tel de tientallen op (met het onthouden), daarna de
        honderdtallen;
  \item controleer met een schatting: rond elk getal af en tel
        grofweg op.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g2:addition:worked}
Reken uit $268 + 156$:
\[
\begin{array}{r}
{\scriptstyle 1}\ {\scriptstyle 1}\ \phantom{0} \\[-3pt]
2\,6\,8 \\
+\ 1\,5\,6 \\
\hline
4\,2\,4
\end{array}
\]
Eenheden: $8 + 6 = 14$: schrijf $4$, onthoud $1$. Tientallen:
$6 + 5 + 1 = 12$: schrijf $2$, onthoud $1$. Honderdtallen:
$2 + 1 + 1 = 4$. Schattingscontrole: ongeveer $270 + 160 = 430$,
dicht bij $424$. \checkmark
\end{example}

\section{Uit het hoofd optellen}

\begin{example}[Maak een tiental]\label{ex:g2:addition:mental}
Bij kleine sommen, spring eerst naar een rond getal:
\[
48 + 6 = 48 + 2 + 4 = 50 + 4 = 54 .
\]
Tel tientallen en eenheden apart:
$35 + 23 = (30 + 20) + (5 + 3) = 50 + 8 = 58$. En een truc met
bijna-honderd: $99 + 46 = 100 + 46 - 1 = 145$.
\end{example}

\begin{example}[Een klein probleem]\label{ex:g2:addition:problem}
``De bibliotheek heeft $185$ boeken; er komen $57$ nieuwe bij.
Hoeveel boeken nu?'' Optellen: $185 + 57 = 242$ (eenheden
$5 + 7 = 12$, onthouden\,\dots). De bibliotheek heeft $242$
boeken.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g2:addition:1}
Reken uit (geen onthouden nodig): $34 + 25$; \quad $126 + 43$;
\quad $507 + 82$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:2}
Reken uit met onthouden: $47 + 8$; \quad $56 + 27$; \quad
$65 + 19$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:3}
Reken kolomsgewijs uit: $268 + 145$; \quad $374 + 158$; \quad
$429 + 293$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:4}
Reken uit het hoofd door een tiental te maken: $38 + 5$; \quad
$67 + 6$; \quad $54 + 8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:5}
Reken uit door tientallen en eenheden apart op te tellen:
$42 + 36$; \quad $53 + 24$; \quad $61 + 28$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:6}
Schat eerst (rond af op het dichtstbijzijnde tiental of honderdtal),
reken dan precies: $58 + 34$; \quad $296 + 187$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:7}
``Ana heeft $128$ knikkers, Tom heeft er $95$. Hoeveel samen?''
Schrijf de optelling en het antwoordzinnetje.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:8}
Vul het eenhedengat: $3\,?\, + 27 = 64$ (vind het ontbrekende
eenhedencijfer van het eerste getal).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:9}
Twee klassen gaan naar de voorstelling: $27$ en $28$ kinderen, plus
$6$ volwassenen. Hoeveel mensen in totaal? (Tel eerst de twee
klassen op.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:addition:10}
Zoek drie getallen op de lijst $27$, $33$, $40$, $65$ die samen
$100$ maken. (Zoek eerst twee waarvan de eenheden tientalvrienden
zijn.)
\end{exercise}
