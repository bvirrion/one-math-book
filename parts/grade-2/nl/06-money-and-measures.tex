\chapter{Geld en maten}\label{ch:g2:measure}

Linialen, keukenweegschalen, kannen, munten, klokken en kalenders:
dit hoofdstuk gaat over de getallen van het dagelijks leven --- hoe
lang, hoe zwaar, hoeveel erin past, hoeveel het kost, en hoe laat
het is.

\section{Lengtes, massa's, inhouden}

\begin{definition}[Alledaagse eenheden]\label{def:g2:measure:units}
\begin{itemize}
  \item \emph{Lengte}: centimeter (cm) voor kleine dingen, meter
        (m) voor grote, met $1$ m $= 100$ cm;
  \item \emph{massa}: gram (g) voor lichte dingen, kilogram (kg)
        voor zware, met $1$ kg $= 1000$ g;
  \item \emph{inhoud} (hoeveel een vat houdt): liter (L).
\end{itemize}
Kleinere eenheden en hun exacte regels komen in
\cref{ch:g3:measure}.
\end{definition}

\begin{example}[Welke eenheid?]\label{ex:g2:measure:choose}
Een potlood: ongeveer $9$ cm. Een deur: ongeveer $2$ m. Een appel:
ongeveer $150$ g. Een grote zak meel: $1$ kg. Een fles melk:
$1$ L. Raadt de eenheid v\'{o}\'{o}r het meten --- dat voorkomt rare
antwoorden zoals ``de deur is $2$ cm''.
\end{example}

\section{Geld}

\begin{method}[Een bedrag maken]\label{met:g2:measure:money}
Met munten en biljetten van $1$, $2$, $5$, $10$, $20$:
\begin{enumerate}
  \item begin met de grootste die past, vul dan de rest aan;
  \item om $27$ te maken: een $20$, een $5$, en een $2$;
  \item er zijn vaak meerdere manieren --- $27$ is ook
        $10 + 10 + 5 + 2$.
\end{enumerate}
\end{method}

\begin{example}[Wisselgeld]\label{ex:g2:measure:change}
Een boek kost $14$; je betaalt met een $20$. Het wisselgeld is
$20 - 14 = 6$, gevonden door vooruit te tellen: $14 \to 20$ is $6$
(\cref{met:g2:subtraction:countup}).
\end{example}

\section{Tijd}

\begin{method}[Halve en kwartieren]\label{met:g2:measure:clock}
De lange wijzer vertelt de minuten na het uur:
\begin{enumerate}
  \item recht omhoog ($12$): op het uur --- $3{:}00$;
  \item een kwartslag naar rechts ($3$): kwart over --- $3{:}15$;
  \item recht omlaag ($6$): half --- $3{:}30$;
  \item driekwartslag ($9$): kwart voor het volgende uur ---
        $3{:}45$, ``kwart voor vier''.
\end{enumerate}
\end{method}

\begin{example}[De kalender]\label{ex:g2:measure:calendar}
Een jaar heeft $12$ maanden, op volgorde: januari, februari, maart,
april, mei, juni, juli, augustus, september, oktober, november,
december. De meeste maanden hebben $30$ of $31$ dagen; februari
heeft $28$ (of $29$). Een datum noemt de dag en de maand: ``$5$
maart''.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g2:measure:1}
Kies de eenheid (cm, m, g, kg of L): een potlood; de lengte van de
klas; een veer; een schooltas; een waterfles.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:2}
Zet om: $2$ m in cm; $300$ cm in m; $1$ kg in g.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:3}
Maak deze bedragen met munten en biljetten: $16$; \quad $23$;
\quad $35$. Geef \'{e}\'{e}n manier voor elk.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:4}
Bereken het wisselgeld: betaal $10$ voor een speelgoed van $6$;
betaal $20$ voor een boek van $17$; betaal $50$ voor een spel van
$32$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:5}
Lees de tijd: lange wijzer op $6$, korte wijzer tussen $4$ en $5$;
lange wijzer op $3$, korte wijzer net na $8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:6}
Teken een klok die half drie aangeeft. Waar wijst elke wijzer
naartoe?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:7}
Noem de maand: na mei; v\'{o}\'{o}r januari; de maand van je
verjaardag.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:8}
Lea koopt een drankje voor $3$ en een cake voor $4$. Ze betaalt met
een biljet van $10$. Hoeveel wisselgeld?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:9}
Wat is zwaarder: een zak suiker van $1$ kg of $800$ g appels? Met
hoeveel gram?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:measure:10}
Maak $40$ met precies vier munten of biljetten (gekozen uit $1$,
$2$, $5$, $10$, $20$). Zoek twee verschillende manieren.
\end{exercise}
