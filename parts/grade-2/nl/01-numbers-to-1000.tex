\chapter{Getallen tot 1000}\label{ch:g2:numbers}

In jaar~1 bundelden we eenheden tot tientallen (\cref{ch:g1:tens});
nu bundelen tien tientallen tot een \emph{honderdtal}, en reiken de
getallen helemaal tot $1000$. Drie cijfers, drie taken: honderdtallen,
tientallen, eenheden.

\section{Honderdtallen, tientallen, eenheden}

\begin{definition}[Drie plaatsen]\label{def:g2:numbers:places}
Een \emph{honderdtal} is tien tientallen, dus tien bundels van tien.
Het getal $346$ heeft drie delen:
\[
346 = 3 \text{ honderdtallen} + 4 \text{ tientallen} + 6 \text{ eenheden}
= 300 + 40 + 6 .
\]
\end{definition}

\begin{example}[De nul houdt de plaats]\label{ex:g2:numbers:zero}
``Driehonderdvijf'' is $305$: drie honderdtallen, \emph{geen}
tientallen, vijf eenheden. De $0$ houdt de $5$ op de plaats van de
eenheden. Zonder die $0$ zou $35$ een heel ander getal zijn.
\end{example}

\begin{example}[Omwisselen]\label{ex:g2:numbers:exchange}
Tien eenheden maken een tiental, tien tientallen maken een
honderdtal. Dus $10$ tientallen $= 100$, en $2$ honderdtallen $= 20$
tientallen. Het getal $250$ bevat $25$ tientallen, niet alleen het
cijfer $5$.
\end{example}

\section{Vergelijken en ordenen}

\begin{method}[Driedijferige getallen vergelijken]\label{met:g2:numbers:compare}
\begin{enumerate}
  \item Meer cijfers wint: $103 > 98$;
  \item zelfde lengte: vergelijk van links --- eerst
        honderdtallen, dan tientallen, dan eenheden. Het eerste
        verschil beslist: $472 > 468$ omdat bij de tientallen
        $7 > 6$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g2:numbers:order}
Zet $530$, $305$, $503$ op volgorde: eerst de honderdtallen
($5$, $3$, $5$), dus $305$ is het kleinst; tussen $530$ en $503$
beslissen de tientallen ($3 > 0$):
\[
305 < 503 < 530 .
\]
\end{example}

\section{Sprongen op de getallenlijn}

\begin{method}[Tientallen en honderdtallen optellen]\label{met:g2:numbers:jumps}
Op de lijn:
\begin{enumerate}
  \item $+1$ / $-1$: verander de eenheden;
  \item $+10$ / $-10$: verander de tientallen ($362 + 10 = 372$);
  \item $+100$ / $-100$: verander de honderdtallen
        ($362 + 100 = 462$).
\end{enumerate}
Slechts \'{e}\'{e}n cijfer beweegt telkens --- tenzij een plaats ``vol
loopt'' en overdraagt, zoals $195 + 10 = 205$.
\end{method}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g2:numbers:1}
Hoeveel honderdtallen, tientallen en eenheden in: $258$; \ $470$; \
$600$; \ $909$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:2}
Schrijf het getal: $4$ honderdtallen, $0$ tientallen, $7$
eenheden; \ $6$ honderdtallen en $3$ tientallen; \ tien
honderdtallen.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:3}
Schrijf in cijfers: tweehonderdzestien; \ vijfhonderdvier; \
zevenhonderdnegentig.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:4}
Ontbind zoals in \cref{def:g2:numbers:places}: $438$; \ $702$; \
$560$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:5}
Schrijf over en vul in met $<$ of $>$:
\[
327 \;?\; 372, \qquad
580 \;?\; 508, \qquad
199 \;?\; 200, \qquad
640 \;?\; 604 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:6}
Zet op volgorde van klein naar groot: $430$; \ $304$; \ $340$; \
$403$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:7}
Reken uit door te springen op de lijn: $253 + 10$; \quad
$253 + 100$; \quad $480 - 10$; \quad $716 - 100$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:8}
Ga door: $195, 205, 215, ?, ?$. \quad
$300, 400, 500, ?, ?$. \quad Let op de eerste!
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:9}
Een winkel heeft $6$ dozen van $100$ potloden en $5$ losse
potloden. Hoeveel potloden in totaal?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:numbers:10}
Met de cijfers $2$, $5$ en $8$ elk \'{e}\'{e}n keer, schrijf het
grootste driedijferige getal en het kleinste. Hoe ver liggen ze uit
elkaar? (Je mag ze optellen op de lijn met sprongen.)
\end{exercise}
