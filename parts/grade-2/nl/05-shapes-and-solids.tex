\chapter{Vormen en lichamen}\label{ch:g2:shapes}

Platte vormen leven op papier; lichamen vullen je handen. Dit
hoofdstuk sorteert platte vormen door zijden en hoeken te tellen,
test de rechte hoek met een gevouwen papieren hoek, en ontmoet de
eerste lichamen --- waarvan de vlakken de platte vormen zijn die we
al kennen.

\section{Platte vormen sorteren}

\begin{definition}[Rechte zijden en rond]\label{def:g2:shapes:sort}
Een vorm waarvan de rand alleen uit rechte zijden bestaat is een
\emph{veelhoek}: driehoek ($3$ zijden), vierhoek ($4$ zijden:
vierkanten en rechthoeken zijn vierhoeken), vijfhoek ($5$),
zeshoek ($6$). Een vorm met een gebogen rand, zoals de cirkel, is
\emph{geen} veelhoek.
\end{definition}

\begin{method}[Een rechte hoek testen]\label{met:g2:shapes:rightangle}
Vouw een vel papier twee keer om een scherpe hoek te maken --- die
hoek is een rechte hoek. Leg hem in de hoek van een vorm:
\begin{enumerate}
  \item als de twee zijden van de vorm precies langs de gevouwen
        hoek lopen, is de hoek recht;
  \item als de hoek breder of smaller is, is hij niet recht.
\end{enumerate}
Een vierkant en een rechthoek hebben vier rechte hoeken.
\end{method}

\section{Natekenen op een raster}

\begin{method}[Een figuur kopiëren]\label{met:g2:shapes:copy}
Op ruitjespapier, kopieer een figuur hoekpunt voor hoekpunt:
\begin{enumerate}
  \item kies een hoek en markeer hem op het lege raster;
  \item tel de vakjes naar de volgende hoek (``$3$ naar rechts, $2$
        omhoog'') en markeer hem;
  \item verbind de hoeken op volgorde; vergelijk met het model.
\end{enumerate}
\end{method}

\section{Lichamen}

\begin{definition}[Eerste lichamen]\label{def:g2:shapes:solids}
Lichamen nemen ruimte in. Hun platte kanten zijn \emph{vlakken}:
\begin{itemize}
  \item de \emph{kubus}: $6$ vierkante vlakken (een dobbelsteen);
  \item de \emph{balk} (rechthoekig prisma): $6$ rechthoekige
        vlakken (een pak ontbijtgranen);
  \item de \emph{cilinder}: twee ronde vlakken en een opgerolde
        zijkant (een blikje);
  \item de \emph{bol}: perfect rond, geen plat vlak.
\end{itemize}
Lichamen en hun vlakken komen terug in \cref{ch:g5:solids}.
\end{definition}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g2:shapes:1}
Tel voor elk de zijden en hoeken en noem hem: een vorm met $5$
zijden; een vorm met $3$ hoeken; een ronde vorm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:2}
Sorteer in ``veelhoek'' of ``geen veelhoek'': een vierkant; een
cirkel; een driehoek; een zeshoek.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:3}
Maak een recht-hoek-tester door papier te vouwen. Zoek drie rechte
hoeken in de kamer, en \'{e}\'{e}n hoek die niet recht is.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:4}
Hoeveel rechte hoeken heeft een vierkant? Een rechthoek? Een
driehoek kan er hoogstens hoeveel hebben? (Probeer er \'{e}\'{e}n met
twee te tekenen.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:5}
Kopieer op ruitjespapier een rechthoek van $4$ vakjes breed en $2$
hoog, dan een driehoek met een basis van $3$ vakjes.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:6}
Noem het lichaam: een dobbelsteen; een soepblikje; een voetbal; een
schoenendoos.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:7}
Hoeveel vlakken heeft een kubus? Welke vorm heeft elk vlak?
Hoeveel vlakken heeft een balk?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:8}
Welke lichamen kunnen rollen: de kubus, de balk, de bol, de
cilinder? Waarom?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:9}
Een vorm heeft $6$ zijden. Hoe heet die? Hoeveel hoeken heeft hij?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:shapes:10}
Teken op ruitjespapier een huis: een vierkant voor de muur
($4 \times 4$) met een driehoekig dak erbovenop. Hoeveel zijden en
hoeken heeft de hele omtrek van het huis?
\end{exercise}
