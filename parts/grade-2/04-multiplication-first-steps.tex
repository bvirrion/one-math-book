\chapter{Multiplication: First Steps}\label{ch:g2:mult}

Four boxes of three cakes: rather than $3 + 3 + 3 + 3$, we write
$4 \times 3$. Multiplication is the shortcut for adding the same
number many times --- and its best picture is a neat array of rows
and columns.

\section{Adding the same number}

\begin{definition}[Multiplication]\label{def:g2:mult:def}
$4 \times 3$ means ``four times three'':
\[
4 \times 3 = 3 + 3 + 3 + 3 = 12 .
\]
The sign is $\times$, read ``times''. The answer is the
\emph{product} (multiplication grows up in \cref{ch:g3:mult}).
\end{definition}

\begin{omfigure}
\begin{tikzpicture}[scale=0.5]
  \foreach \r in {0,1,2,3}
    \foreach \c in {0,1,2}
      \fill[omDef] (\c,-\r) circle (0.28);
  \node[right, font=\small] at (3.0,-1.5)
    {$4$ rows of $3$ dots: $4 \times 3 = 12$};
\end{tikzpicture}

{\small An \emph{array}: $4$ rows of $3$. Counting by rows gives
$4 \times 3$; the same dots by columns give $3 \times 4$ --- so the
order does not matter.}
\end{omfigure}

\begin{example}[Order does not matter]\label{ex:g2:mult:order}
$4 \times 3$ and $3 \times 4$ count the same $12$ dots, one by
rows, one by columns. Turning the array on its side proves it. So
learning one product gives its twin for free.
\end{example}

\section{Skip counting and the easy tables}

\begin{method}[Building a table by skip counting]\label{met:g2:mult:skip}
To find $5 \times 4$, count by fours five times:
$4, 8, 12, 16, 20$. Skip counting on the number line \emph{is} the
table.
\begin{itemize}
  \item Table of $2$: $2, 4, 6, 8, 10, \dots$ (the doubles);
  \item table of $5$: $5, 10, 15, 20, 25, \dots$ (ends in $5$ or
        $0$);
  \item table of $10$: $10, 20, 30, \dots$ (just add a zero).
\end{itemize}
\end{method}

  \foreach \i in {0,...,11} \draw (\i,-0.1) -- (\i,0.1);
  \foreach \i in {0,2,4,6,8,10}

    \draw[-Stealth, omDef, thick] (2*\i,0.28) .. controls

  \node[omDef, font=\small] at (5.5,1.1) {hops of $2$};
\end{tikzpicture}

{\small The table of $2$ as hops on the line: $2, 4, 6, 8, 10$. Five
hops of $2$ reach $10$, so $5 \times 2 = 10$.}
\end{omfigure}

\begin{example}[Doubles and halves]\label{ex:g2:mult:double}
Multiplying by $2$ is doubling: $2 \times 7 = 14$. The reverse,
halving, undoes it: half of $14$ is $7$. Doubles known by heart
make the whole table of $2$ instant --- and prepare division
(\cref{ch:g3:division}).
\end{example}

\section{Multiplying in problems}

\begin{example}\label{ex:g2:mult:problem}
``A box holds $5$ eggs. How many eggs in $4$ boxes?'' Four groups of
five: $4 \times 5 = 20$ (skip count: $5, 10, 15, 20$). There are
$20$ eggs.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g2:mult:1}
Write as a multiplication, then compute: $2 + 2 + 2$; \quad
$5 + 5 + 5 + 5$; \quad $10 + 10 + 10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:2}
Draw an array of $3$ rows of $5$ dots. What multiplication does it
show? And turned on its side?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:3}
Skip count to fill the table of $2$ up to $2 \times 6$; the table
of $5$ up to $5 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:4}
Compute: $2 \times 7$; \quad $5 \times 3$; \quad $10 \times 4$;
\quad $2 \times 9$; \quad $5 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:5}
Compute the doubles: double of $6$; double of $8$; double of $10$.
Then the halves: half of $12$; half of $20$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:6}
Compute the table of $3$ by skip counting: $3, 6, 9, ?, ?, ?$. What
is $3 \times 5$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:7}
``There are $4$ tables with $5$ chairs each. How many chairs?''
Write the multiplication and answer.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:8}
A spider has $8$ legs. How many legs have $3$ spiders? (Skip count
by $8$: $8, 16, \dots$)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:9}
Two ways to arrange $12$ dots in equal rows: draw $2$ rows of $6$,
and $3$ rows of $4$. Write the multiplication for each.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:mult:10}
Cakes come in boxes of $5$. Mia needs $18$ cakes for a party. How
many boxes must she buy? (Skip count by $5$ until you pass $18$.)
\end{exercise}
