\chapter{Números hasta 1000}\label{ch:g2:numbers}

Grado 1 agrupa unidades en decenas (\cref{ch:g1:tens}); ahora diez decenas
agruparse en un \emph{centenar}, y los números se extienden hasta
$1000$. Tres dígitos, tres trabajos: centenas, decenas, unidades.

\section{Centenas, decenas, unidades.}

\begin{definition}[tres lugares]\label{def:g2:numbers:places}
Un \emph{centenar} son diez decenas, es decir, diez paquetes de diez. el
El número $346$ consta de tres partes:
\[
346 = 3 \text{ hundreds} + 4 \text{ tens} + 6 \text{ units}
= 300 + 40 + 6 .
\]
\end{definition}

\begin{example}[El cero ocupa el lugar.]\label{ex:g2:numbers:zero}
``Trescientos cinco'' es $305$: trescientos, \emph{no}
decenas, cinco unidades. El $0$ mantiene el $5$ en el lugar de las unidades. sin
$35$ sería un número completamente diferente.
\end{example}

\begin{example}[Intercambio]\label{ex:g2:numbers:exchange}
Diez unidades forman una decena, diez decenas forman una centena. Entonces $10$ decenas
$= 100$ y $2$ centenas $= 20$ decenas. El número $250$ contiene
$25$ decenas, no solo el dígito $5$.
\end{example}

\section{Comparar y ordenar}

\begin{method}[Comparando números de tres dígitos]\label{met:g2:numbers:compare}
\begin{enumerate}
  \item Más dígitos gana: $103 > 98$;
  \item misma longitud: comparar desde la izquierda --- centenas primero, luego
        decenas, luego unidades. La primera diferencia decide:
        $472 > 468$ porque en las decenas, $7 > 6$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g2:numbers:order}
Ordene $530$, $305$, $503$: centenas primero ($5$, $3$, $5$), por lo que
$305$ clientes potenciales; entre $530$ y $503$, las decenas deciden ($3 > 0$):
\[
305 < 503 < 530 .
\]
\end{example}

\section{Salta sobre la recta numérica}

\begin{method}[Suma de decenas y centenas]\label{met:g2:numbers:jumps}
En la línea:
\begin{enumerate}
  \item $+1$ / $-1$: cambiar las unidades;
  \item $+10$ / $-10$: cambiar las decenas ($362 + 10 = 372$);
  \item $+100$ / $-100$: cambiar las centenas ($362 + 100 = 462$).
\end{enumerate}
Sólo se mueve un dígito cada vez, a menos que un lugar "se llene" y
se traslada, como $195 + 10 = 205$.
\end{method}

\section{Ceremonias}

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:2}
Escribe el número: $4$ centenas, $0$ decenas, $7$ unidades; \ $6$ cientos

\begin{exercise}[$\star$]\label{exo:g2:numbers:3}

\begin{exercise}[$\star$]\label{exo:g2:numbers:6}
