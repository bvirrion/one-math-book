\chapter{Dinero y medidas}\label{ch:g2:measure}

Reglas de medir, básculas de cocina, jarras, monedas, relojes y
calendarios: este capítulo trata sobre los números de la vida cotidiana ---
cuánto tiempo, qué peso, cuánto aguanta, cuánto cuesta y qué
hora es.

\section{Longitudes, masas, capacidades.}

\begin{definition}[unidades diarias]\label{def:g2:measure:units}
\begin{itemize}
  \item \emph{Longitud}: centímetro (cm) para cosas pequeñas, metro (m)
        para los grandes, con $1$ m $= 100$ cm;
  \item \emph{masa}\index{masa}: gramo (g) para cosas livianas, kilogramo (kg) para
        los pesados, con $1$ kg $= 1000$ g;
  \item \emph{capacidad}\index{capacidad} (cuánto cabe en un recipiente): litro (L).
\end{itemize}
Las unidades más pequeñas y sus reglas exactas vienen en \cref{ch:g3:measure}.
\end{definition}

\begin{example}[¿Qué unidad?]\label{ex:g2:measure:choose}
Un crayón: aproximadamente $9$ cm. Una puerta: aproximadamente $2$ m. Una manzana: aproximadamente $150$
g. Un saco grande de harina: $1$ kg. Una botella de leche: $1$ L. Adivina la unidad
antes de medir --- detiene respuestas tontas como ``la puerta es $2$
centímetros''.
\end{example}

\section{Dinero}

\begin{method}[Haciendo una cantidad]\label{met:g2:measure:money}
Con monedas y billetes de $1$, $2$, $5$, $10$, $20$:
\begin{enumerate}
  \item comienza con el más grande que quepa, luego llena el resto;
  \item para hacer $27$: un $20$, un $5$ y un $2$;
  \item suele haber varias formas: $27$ también es $10 + 10 + 5 +
        2$.
\end{enumerate}
\end{method}

\begin{example}[Cambiar]\label{ex:g2:measure:change}
Un libro cuesta $14$; Pagas con un $20$. El cambio es $20 - 14 =
6$, se encuentra contando hacia arriba: $14 \to 20$ es $6$
(\cref{met:g2:subtraction:countup}).
\end{example}

\section{Tiempo}

\begin{method}[Media hora y cuarto]\label{met:g2:measure:clock}
La manecilla larga indica los minutos después de la hora:
\begin{enumerate}
  \item directo hacia arriba ($12$): en punto --- $3{:}00$;
  \item cuarto de vuelta a la derecha ($3$): cuarto y cuarto --- $3{:}15$;
  \item hacia abajo ($6$): y media --- $3{:}30$;
  \item giro de tres cuartos ($9$): cuarto para la hora siguiente ---
        $3{:}45$, ``las cuatro menos cuarto''.
\end{enumerate}
\end{method}

\begin{example}[El calendario]\label{ex:g2:measure:calendar}
Un año tiene $12$ meses, en orden: enero, febrero, marzo, abril,
Mayo, Junio, Julio, Agosto, Septiembre, Octubre, Noviembre, Diciembre.
La mayoría de los meses tienen días $30$ o $31$; Febrero tiene $28$ (o $29$).
Una fecha nombra el día y el mes: ``el $5$ de marzo''.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g2:measure:1}
Elige la unidad (cm, m, g, kg o L): un lápiz; el aula
longitud; una pluma; una mochila; una botella de agua.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:2}
Convertir: $2$ m a cm; $300$ cm en m; $1$ kg en g.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:3}
Realiza estas cantidades con monedas y billetes: $16$; \quad $23$; \quad
$35$. Dé una forma para cada uno.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:4}
Calcula el cambio: paga $10$ por un juguete $6$; paga $20$ por un $17$
libro; paga $50$ por un juego $32$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:5}
Lea la hora: manecilla larga en $6$, manecilla corta entre $4$ y $5$;
mano larga en $3$, mano corta justo después de $8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:6}
Dibuja un reloj que muestre las dos y media. ¿Hacia dónde apunta cada mano?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:7}
Nombre el mes: después de mayo; antes de enero; el mes de tu
cumpleaños.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:8}
Lea compra una bebida por $3$ y un pastel por $4$. Ella paga con un $10$
factura. ¿Cuánto cambio?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:9}
¿Qué pesa más: un saco de $1$ kg de azúcar o $800$ g de manzanas? por como
cuantos gramos?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:measure:10}
Haga $40$ con exactamente cuatro monedas o billetes (elegidos entre $1$, $2$,
$5$, $10$, $20$). Encuentra dos formas diferentes.
\end{exercise}
