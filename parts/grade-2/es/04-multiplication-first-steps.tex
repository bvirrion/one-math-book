\chapter{Multiplicación: primeros pasos}\label{ch:g2:mult}

Cuatro cajas de tres pasteles: en lugar de $3 + 3 + 3 + 3$ escribimos
$4 \times 3$. La multiplicación es el atajo para sumar lo mismo.
número muchas veces --- y su mejor imagen es una ordenada serie de filas
y columnas.

\section{Sumando el mismo numero}

\begin{definition}[multiplicacion]\label{def:g2:mult:def}
$4 \times 3$ significa "cuatro por tres":
\[
4 \times 3 = 3 + 3 + 3 + 3 = 12 .
\]
El signo es $\times$, léase "veces". La respuesta es la
\emph{producto}\index{producto} (la multiplicación crece en \cref{ch:g3:mult}).
\end{definition}

      \fill[omDef] (\c,-\r) circle (0.28);

{\small Un \emph{formación}: $4$ filas de $3$. Contar por filas da

\end{omfigure}

\begin{example}[El orden no importa]\label{ex:g2:mult:order}
$4 \times 3$ y $3 \times 4$ cuentan los mismos puntos $12$, uno por
filas, una por columnas. Girar la matriz de lado lo demuestra. entonces
aprender uno \omterm{def:g2:mult:def}{producto} da su gemelo gratis.
\end{example}

\section{Contar salteado y las tablas fáciles}

\begin{method}[Construyendo una tabla contando salteado]\label{met:g2:mult:skip}
Para encontrar $5 \times 4$, cuente de cuatro en cuatro cinco veces:
$4, 8, 12, 16, 20$. Contando salteado en la recta numérica \emph{is} el
mesa.
\begin{itemize}
  \item Tabla de $2$: $2, 4, 6, 8, 10, \dots$ (los dobles);
  \item Tabla  de $5$: $5, 10, 15, 20, 25, \dots$ (termina en $5$ o
        $0$);
  \item Tabla  de $10$: $10, 20, 30, \dots$ (solo agrega un \omterm{def:g1:counting:zero}{cero}).
\end{itemize}
\end{method}

\begin{example}[Dobles y mitades]\label{ex:g2:mult:double}
Multiplicar por $2$ es duplicar: $2 \times 7 = 14$. Al revés,
reducir a la mitad, lo deshace: la mitad de $14$ es $7$. Dobles conocidos de memoria
haga que toda la tabla de $2$ sea instantánea --- y prepare la división
(\cref{ch:g3:division}).
\end{example}

\section{Multiplicando en problemas}

\begin{example}\label{ex:g2:mult:problem}
``Una caja contiene huevos $5$. ¿Cuántos huevos hay en las cajas $4$?'' Cuatro grupos de
cinco: $4 \times 5 = 20$ (cuenta de saltos: $5, 10, 15, 20$). hay
$20$ huevos.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g2:mult:1}
Escribe como una multiplicación, luego calcula: $2 + 2 + 2$; \quad
$5 + 5 + 5 + 5$; \quad$10 + 10 + 10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:2}
Dibuja una matriz de $3$ filas de puntos $5$. ¿Qué multiplicación hace?
mostrar? ¿Y volteado de lado?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:3}
Saltar cuenta para llenar la tabla de $2$ hasta $2 \times 6$; la mesa
de $5$ hasta $5 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:4}
Calcular: $2 \times 7$; \quad$5 \times 3$; \quad$10 \times 4$;
\quad$2 \times 9$; \quad$5 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:5}
Calcular los dobles: doble de $6$; doble de $8$; doble de $10$.
Luego las mitades: la mitad de $12$; la mitad de $20$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:6}
Calcule la tabla de $3$ contando salteado: $3, 6, 9, ?, ?, ?$. que
es $3 \times 5$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:7}
``Hay mesas $4$ con sillas $5$ cada una. ¿Cuántas sillas?
Escribe la multiplicación y responde.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:8}
Una araña tiene patas $8$. ¿Cuántas patas tienen las arañas $3$? (Cuenta salteada
por $8$: $8, 16, \dots$)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:mult:9}
Dos formas de organizar puntos $12$ en filas iguales: dibujar $2$ filas de $6$,
y $3$ filas de $4$. Escribe la multiplicación de cada uno.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:mult:10}
Los pasteles vienen en cajas de $5$. Mia necesita pasteles $18$ para una fiesta. como
¿Cuántas cajas debe comprar? (Salte la cuenta hasta $5$ hasta pasar $18$).
\end{exercise}
