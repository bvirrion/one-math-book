\chapter{Formas y sólidos}\label{ch:g2:shapes}

Las formas planas viven en el papel; Los sólidos llenan tus manos. este capitulo
clasifica formas planas contando lados y esquinas, prueba si son correctas
ángulo con una esquina de papel doblado y se encuentra con los primeros sólidos ---
cuyos \omterm{def:g2:shapes:solids}{caras} son las formas planas que ya conocemos.

\section{Clasificar formas planas}

\begin{definition}[Rectos y redondos]\label{def:g2:shapes:sort}
Una figura cuyo borde está hecho sólo de lados rectos es una
\emph{polígono}\index{polígono}: triángulo ($3$ lados), cuadrilátero ($4$ lados:
cuadrados y rectángulos son cuadriláteros), pentágono ($5$),
hexágono ($6$). Una figura con un borde curvo, como el círculo, es
\emph{no} un polígono.
\end{definition}

\begin{method}[Probando un ángulo recto]\label{met:g2:shapes:rightangle}
Dobla una hoja de papel dos veces para hacer una esquina afilada\dots esa esquina
es un ángulo recto. Colóquelo en la esquina de una forma:
\begin{enumerate}
  \item si los dos lados de la forma corren exactamente a lo largo del plegado
        esquina, el ángulo es recto;
  \item si la esquina es más ancha o más estrecha, no está bien.
\end{enumerate}
Un cuadrado y un rectángulo tienen cuatro ángulos rectos.
\end{method}

\section{Reproduciendo en una grilla}

\begin{method}[Copiando una figura]\label{met:g2:shapes:copy}
En papel cuadriculado, copia una figura vértice por vértice:
\begin{enumerate}
  \item elige una esquina y márcala en la cuadrícula vacía;
  \item cuenta los cuadrados hasta la siguiente esquina (``$3$ derecha, $2$
        arriba'') y márquelo;
  \item une las esquinas en orden; comparar con el modelo.
\end{enumerate}
\end{method}

\section{Sólidos}

\begin{definition}[Primeros solidos]\label{def:g2:shapes:solids}
Los sólidos ocupan espacio. Sus lados planos son \emph{caras}\index{rostro}:
\begin{itemize}
  \item o \emph{cubo}\index{cubo}: $6$ caras cuadradas (un dado);
  \item el \emph{caja} (prisma rectangular): $6$ caras rectangulares (un
        paquete de cereales);
  \item o \emph{cilindro}\index{cilindro}: dos caras redondas y un lado enrollado (un
        puede);
  \item o \emph{pelota}: perfectamente redondo, sin cara plana.
\end{itemize}
Los sólidos y sus caras regresan en \cref{ch:g5:solids}.
\end{definition}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g2:shapes:1}
Para cada uno, cuenta los lados y las esquinas y nómbralo: una forma con $5$
lados; una forma con esquinas $3$; una forma redonda.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:2}
Ordenar en ``\omterm{def:g2:shapes:sort}{polígono}'' o ``no \omterm{def:g2:shapes:sort}{polígono}'': un cuadrado; un círculo; un
triángulo; un hexágono.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:3}
Haga un probador de ángulo recto doblando papel. Encuentra tres ángulos rectos.
en la habitación y un ángulo que no es correcto.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:4}
¿Cuántos ángulos rectos tiene un cuadrado? ¿Un rectángulo? Un triángulo puede tener
como mucho cuantos? (Intenta dibujar uno con dos).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:5}
En papel cuadriculado, copie un rectángulo de $4$ cuadrados de ancho y $2$
de alto, luego un triángulo con una base cuadrada $3$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:6}
Nombra el sólido: un dado; una lata de sopa; una pelota de fútbol; una caja de zapatos.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:7}
¿Cuántos \omterm{def:g2:shapes:solids}{caras} tiene un \omterm{def:g2:shapes:solids}{cubo}? ¿Qué forma tiene cada \omterm{def:g2:shapes:solids}{rostro}? Cuantos \omterm{def:g2:shapes:solids}{caras}
tiene una caja?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:8}
¿Qué sólidos pueden rodar: el \omterm{def:g2:shapes:solids}{cubo}, la caja, la pelota, el \omterm{def:g2:shapes:solids}{cilindro}?
¿Por qué?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:9}
Una forma tiene lados $6$. ¿Cómo se llama? cuantas esquinas tiene?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:shapes:10}
En papel cuadriculado, dibuja una casa: un cuadrado para la pared.
($4 \times 4$) con techo triangular en la parte superior. cuantos lados y
rincones tiene todo el contorno de la casa?
\end{exercise}
