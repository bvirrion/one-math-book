\chapter{Sustracción}\label{ch:g2:subtraction}

Cuando las unidades de arriba son muy pocas para tomarlas, abrimos un
diez --- el \emph{pedir prestado}, el espejo del carry. este capitulo
lo explica con palos, luego entrena la resta escrita, con
contando como un atajo amistoso.

\section{¿Por qué tomamos prestado?}

\begin{example}[Desagregando un diez]\label{ex:g2:subtraction:unbundle}
$42 - 17$: queremos tomar unidades $7$ de unidades $2$ --- ¡muy pocas!
Así que desata una de las decenas $4$ en diez unidades sueltas: ahora la parte superior tiene
$3$ decenas y $12$ unidades. Para llevar: $12 - 7 = 5$ unidades, y
$3 - 1 = 2$ decenas. Entonces $42 - 17 = 25$.
\end{example}

\section{El método escrito}

\begin{method}[resto de columna]\label{met:g2:subtraction:column}
\begin{enumerate}
  \item Escribe el número mayor arriba, alineado a la derecha;
  \item restar las unidades; si el dígito superior es demasiado pequeño, pida prestado
        una decena (se convierte en diez unidades en la columna actual);
  \item resta las decenas, luego las centenas;
  \item \emph{controlar}: la respuesta más el número quitado debe
        Devuelve el número superior.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g2:subtraction:worked}
Calcular $503 - 168$:
\[
\begin{array}{r}
5\,0\,3 \\
-\ 1\,6\,8 \\
\hline
3\,3\,5
\end{array}
\]
Unidades: $3 - 8$ demasiado pequeño; pedir prestado a través de las decenas (no hay ninguno,
así que tome prestado primero las centenas): la parte superior se convierte en $4$ centenas,
$9$ decenas, $13$ unidades. Entonces $13 - 8 = 5$; $9 - 6 = 3$;
$4 - 1 = 3$. Verificar: $335 + 168 = 503$. \checkmark
\end{example}

\section{contando}

\begin{method}[Reiniciar contando hacia arriba]\label{met:g2:subtraction:countup}
Para números cercanos, suba del pequeño al grande.
uno y agrega el lúpulo:
\[
83 - 78: \quad 78 \to 80 \text{ is } 2, \ 80 \to 83 \text{ is } 3,
\text{ so } 83 - 78 = 5 .
\]
Contar es más rápido que pedir prestado cuando los dos números están cerca
entre sí --- y así es exactamente como un comerciante da cambio.
\end{method}

\begin{example}[Dando cambio]\label{ex:g2:subtraction:change}
Un juguete cuesta $34$; Pagas con un $50$. Cambio: contar hacia arriba,
$34 \to 40$ es $6$, $40 \to 50$ es $10$, por lo que el cambio es $16$.
La resta $50 - 34 = 16$, hecha a la manera del comerciante.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:1}
Calcular (no es necesario pedir prestado): $58 - 23$; \quad$176 - 45$; \quad
$389 - 164$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:2}
Calcular con préstamo: $52 - 17$; \quad$73 - 28$; \quad
$60 - 24$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:3}
Calcule en columnas y verifique cada una: $324 - 167$; \quad
$500 - 236$; \quad $703 - 158$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:4}
Calcular contando: $62 - 58$; \quad$91 - 85$; \quad
$100 - 96$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:5}
Calcule el cambio (cuente hacia arriba): pague $50$ por un libro $42$; pagar $20$
para un juguete $13$; paga $100$ por una bolsa $75$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:6}
``Un árbol mide $156$ cm de altura. Creció $28$ cm este año. ¿Qué altura tenía?
¿El año pasado?'' Escribe la resta y responde.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:7}
Complete con la adición de gemelos para verificar: $85 - 39 = 46$; es
¿verdad? $214 - 68 = 156$; ¿es correcto?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:8}
Para cada historia, elija $+$ o $-$: ``$47$ pájaros, $19$ se van volando'';
``$47$ manzanas, $19$ más recogidas''; ``Sam tiene $47$, Lea tiene $19$:
¿Cuántos más tiene Sam?''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:9}
Llene el agujero: $63 - \;?\; = 45$. (Cuente desde $45$ hasta $63$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:subtraction:10}

El resultado es $58$. ¿Cuál era el número? (¡Trabaja al revés!)
\end{exercise}
