\chapter{Suma con transporte}\label{ch:g2:addition}

Cuando las unidades de un \omterm{def:g1:addition:def}{suma} pasan de nueve, diez de ellas se agrupan en una nueva
diez --- y ese pequeño paquete, movido a la siguiente columna, es el
famoso \emph{llevar}. Este capítulo explica el transporte con palos,
luego entrena el método escrito.

\section{¿Por qué llevamos?}

\begin{example}[Agrupación de diez unidades]\label{ex:g2:addition:bundle}
$27 + 15$: juntar las unidades, $7 + 5 = 12$ unidades. pero diez
¡las unidades forman una decena! Agrupar diez de ellos: eso deja $2$ unidades sueltas
y un diez nuevo para llevar. Ahora las decenas: $2 + 1 + 1 = 4$ decenas. entonces
$27 + 15 = 42$.
\end{example}

\section{El método escrito}

\begin{method}[Adición de columna]\label{met:g2:addition:column}
\begin{enumerate}
  \item Escribe los números uno debajo del otro, unidades debajo de unidades,
        decenas bajo decenas;
  \item agregar las unidades; si la respuesta pasa $9$, escribe las unidades
        dígito y lleva un pequeño $1$ encima de las decenas;
  \item suma las decenas (con el acarreo), luego las centenas;
  \item verifique con una estimación: redondee cada número y sume aproximadamente.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g2:addition:worked}
Calcular $268 + 156$:
\[
\begin{array}{r}
{\scriptstyle 1}\ {\scriptstyle 1}\ \phantom{0} \\[-3pt]
2\,6\,8 \\
+\ 1\,5\,6 \\
\hline
4\,2\,4
\end{array}
\]
Unidades: $8 + 6 = 14$: escriba $4$, lleve $1$. Decenas:
$6 + 5 + 1 = 12$: escribe $2$, lleva $1$. Cientos:
$2 + 1 + 1 = 4$. Verificación de estimación: aproximadamente $270 + 160 = 430$, cerca de
$424$. \checkmark
\end{example}

\section{Agregando en tu cabeza}

\begin{example}[Haz un diez]\label{ex:g2:addition:mental}
Para \omterm{def:g1:addition:def}{sumas} pequeño, salte primero a un número redondo:
\[
48 + 6 = 48 + 2 + 4 = 50 + 4 = 54 .
\]
Suma decenas y unidades por separado:
$35 + 23 = (30 + 20) + (5 + 3) = 50 + 8 = 58$. Y casi cien
truco: $99 + 46 = 100 + 46 - 1 = 145$.
\end{example}

\begin{example}[Un pequeño problema]\label{ex:g2:addition:problem}
``La biblioteca tiene libros $185$; $57$ llegan nuevos. cuantos libros
¿ahora?" Añadiendo: $185 + 57 = 242$ (unidades $5 + 7 = 12$, carry\,\dots).
La biblioteca tiene libros $242$.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g2:addition:1}
Calcular (no se necesita transporte): $34 + 25$; \quad $126 + 43$; \quad
$507 + 82$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:2}
Calcular con acarreo: $47 + 8$; \quad$56 + 27$; \quad$65 + 19$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:3}
Calcular en columnas: $268 + 145$; \quad$374 + 158$; \quad
$429 + 293$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:4}
Calcula mentalmente formando una decena: $38 + 5$; \quad$67 + 6$;
\quad$54 + 8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:5}
Calcular sumando decenas y unidades por separado: $42 + 36$; \quad
$53 + 24$; \quad$61 + 28$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:6}
Primero estime (redondee a la decena o centena más cercana), luego calcule
exactamente: $58 + 34$; \quad $296 + 187$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:7}
``Ana tiene canicas $128$, Tom tiene $95$. ¿Cuantos juntos?'' Escribe
la suma y la oración de respuesta.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:8}
Complete el agujero de las unidades: $3\,?\, + 27 = 64$ (encuentre las unidades que faltan
dígito del primer número).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:9}
Dos clases van al espectáculo: $27$ y $28$ niños, más $6$
adultos. ¿Cuántas personas en total? (Agregue las dos clases primero).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:addition:10}
Encuentra tres números en la lista $27$, $33$, $40$, $65$ que sumen
a $100$. (Busque primero dos cuyas unidades sean diez amigos).
\end{exercise}
