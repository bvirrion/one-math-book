\chapter{Addition avec retenue}\label{ch:g2:addition}

Quand les unit\'es d'une somme d\'epassent neuf, dix d'entre elles se
regroupent en une nouvelle dizaine --- et ce petit paquet, d\'eplac\'e
dans la colonne suivante, c'est la c\'el\`ebre \emph{retenue}. Ce
chapitre explique la retenue avec des b\^atonnets, puis entra\^ine
la technique \'ecrite.

\section{Pourquoi on retient}

\begin{example}[Regrouper dix unit\'es]\label{ex:g2:addition:bundle}
$27 + 15$~: rassembler les unit\'es, $7 + 5 = 12$ unit\'es. Mais
dix unit\'es font une dizaine~! En regrouper dix~: il reste $2$
unit\'es isol\'ees et une nouvelle dizaine \`a retenir. Maintenant
les dizaines~: $2 + 1 + 1 = 4$ dizaines. Donc $27 + 15 = 42$.
\end{example}

\section{La technique \'ecrite}

\begin{method}[Addition en colonnes]\label{met:g2:addition:column}
\begin{enumerate}
  \item \'Ecrire les nombres l'un sous l'autre, unit\'es sous
        unit\'es, dizaines sous dizaines~;
  \item additionner les unit\'es~; si le r\'esultat d\'epasse $9$,
        \'ecrire le chiffre des unit\'es et retenir un petit $1$
        au-dessus des dizaines~;
  \item additionner les dizaines (avec la retenue), puis les
        centaines~;
  \item v\'erifier avec une estimation~: arrondir chaque nombre et
        additionner approximativement.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g2:addition:worked}
Calculer $268 + 156$~:
\[
\begin{array}{r}
{\scriptstyle 1}\ {\scriptstyle 1}\ \phantom{0} \\[-3pt]
2\,6\,8 \\
+\ 1\,5\,6 \\
\hline
4\,2\,4
\end{array}
\]
Unit\'es~: $8 + 6 = 14$~: \'ecrire $4$, retenir $1$. Dizaines~:
$6 + 5 + 1 = 12$~: \'ecrire $2$, retenir $1$. Centaines~:
$2 + 1 + 1 = 4$. Estimation de contr\^ole~: environ
$270 + 160 = 430$, proche de $424$. \checkmark
\end{example}

\section{Additionner de t\^ete}

\begin{example}[Faire une dizaine]\label{ex:g2:addition:mental}
Pour les petites sommes, sauter d'abord vers un nombre rond~:
\[
48 + 6 = 48 + 2 + 4 = 50 + 4 = 54 .
\]
Ajouter s\'epar\'ement dizaines et unit\'es~:
$35 + 23 = (30 + 20) + (5 + 3) = 50 + 8 = 58$. Et l'astuce du
presque-cent~: $99 + 46 = 100 + 46 - 1 = 145$.
\end{example}

\begin{example}[Un petit probl\`eme]\label{ex:g2:addition:problem}
\guillemotleft{} La biblioth\`eque a $185$ livres~; $57$ nouveaux
arrivent. Combien de livres maintenant~? \guillemotright{}
Addition~: $185 + 57 = 242$ (unit\'es $5 + 7 = 12$,
retenue\,\dots). La biblioth\`eque a $242$ livres.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g2:addition:1}
Calcule (sans retenue)~: $34 + 25$~; \quad $126 + 43$~; \quad
$507 + 82$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:2}
Calcule avec une retenue~: $47 + 8$~; \quad $56 + 27$~; \quad
$65 + 19$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:3}
Calcule en colonnes~: $268 + 145$~; \quad $374 + 158$~; \quad
$429 + 293$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:4}
Calcule de t\^ete en faisant une dizaine~: $38 + 5$~; \quad
$67 + 6$~; \quad $54 + 8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:5}
Calcule en ajoutant s\'epar\'ement dizaines et unit\'es~:
$42 + 36$~; \quad $53 + 24$~; \quad $61 + 28$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:6}
Estime d'abord (arrondis \`a la dizaine ou \`a la centaine la plus
proche), puis calcule exactement~: $58 + 34$~; \quad $296 + 187$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:7}
\guillemotleft{} Ana a $128$ billes, Tom en a $95$. Combien en
tout~? \guillemotright{} \'Ecris l'addition et la phrase r\'eponse.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:8}
Compl\`ete le trou des unit\'es~: $3\,?\, + 27 = 64$ (trouve le
chiffre des unit\'es manquant du premier nombre).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:addition:9}
Deux classes vont au spectacle~: $27$ et $28$ enfants, plus $6$
adultes. Combien de personnes en tout~? (Ajoute d'abord les deux
classes.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:addition:10}
Trouve trois nombres de la liste $27$, $33$, $40$, $65$ dont la
somme fait $100$. (Cherche d'abord deux dont les unit\'es sont des
amis de $10$.)
\end{exercise}
