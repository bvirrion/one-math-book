\chapter{Figures et solides}\label{ch:g2:shapes}

Les figures plates vivent sur le papier~; les solides remplissent les
mains. Ce chapitre trie les figures plates en comptant c\^ot\'es et
sommets, teste l'angle droit avec un coin de papier pli\'e, et
rencontre les premiers solides --- dont les faces sont les figures
plates d\'ej\`a connues.

\section{Trier les figures plates}

\begin{definition}[C\^ot\'es droits et ronds]\label{def:g2:shapes:sort}
Une figure dont le bord n'est fait que de c\^ot\'es droits est un
\emph{polygone}~: triangle ($3$ c\^ot\'es), quadrilat\`ere ($4$
c\^ot\'es~: les carr\'es et les rectangles sont des
quadrilat\`eres), pentagone ($5$), hexagone ($6$). Une figure au
bord courb\'e, comme le cercle, n'est \emph{pas} un polygone.
\end{definition}

\begin{method}[Tester un angle droit]\label{met:g2:shapes:rightangle}
Plier une feuille de papier deux fois pour faire un coin net --- ce
coin est un angle droit. Le poser dans le coin d'une figure~:
\begin{enumerate}
  \item si les deux c\^ot\'es de la figure suivent exactement le coin
        pli\'e, l'angle est droit~;
  \item si le coin est plus ouvert ou plus ferm\'e, ce n'est pas un
        angle droit.
\end{enumerate}
Un carr\'e et un rectangle ont quatre angles droits.
\end{method}

\section{Reproduire sur un quadrillage}

\begin{method}[Copier une figure]\label{met:g2:shapes:copy}
Sur du papier quadrill\'e, copier une figure sommet par sommet~:
\begin{enumerate}
  \item choisir un sommet et le marquer sur le quadrillage vide~;
  \item compter les cases jusqu'au sommet suivant
        (\guillemotleft{} $3$ \`a droite, $2$ vers le haut
        \guillemotright{}) et le marquer~;
  \item relier les sommets dans l'ordre~; comparer avec le mod\`ele.
\end{enumerate}
\end{method}

\section{Les solides}

\begin{definition}[Premiers solides]\label{def:g2:shapes:solids}
Les solides occupent de l'espace. Leurs c\^ot\'es plats sont des
\emph{faces}~:
\begin{itemize}
  \item le \emph{cube}~: $6$ faces carr\'ees (un d\'e)~;
  \item le \emph{pav\'e} (parall\'el\'epip\`ede rectangle)~: $6$ faces
        rectangulaires (une bo\^ite de c\'er\'eales)~;
  \item le \emph{cylindre}~: deux faces rondes et un c\^ot\'e enroul\'e
        (une bo\^ite de conserve)~;
  \item la \emph{boule}~: parfaitement ronde, pas de face plate.
\end{itemize}
Les solides et leurs faces reviennent dans \cref{ch:g5:solids}.
\end{definition}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g2:shapes:1}
Pour chacune, compte c\^ot\'es et sommets et nomme-la~: une figure
avec $5$ c\^ot\'es~; une figure avec $3$ sommets~; une figure ronde.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:2}
Classe en \guillemotleft{} polygone \guillemotright{} ou
\guillemotleft{} pas un polygone \guillemotright{}~: un carr\'e~; un
cercle~; un triangle~; un hexagone.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:3}
Fabrique un testeur d'angle droit en pliant du papier. Trouve trois
angles droits dans la pi\`ece, et un angle qui n'est pas droit.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:4}
Combien d'angles droits a un carr\'e~? Un rectangle~? Un triangle
peut en avoir au plus combien~? (Essaie d'en dessiner un avec
deux.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:5}
Sur du papier quadrill\'e, copie un rectangle de $4$ cases de large
et $2$ de haut, puis un triangle avec une base de $3$ cases.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:6}
Nomme le solide~: un d\'e~; une bo\^ite de conserve de soupe~; un
ballon de football~; une bo\^ite \`a chaussures.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:7}
Combien de faces a un cube~? Quelle figure est chaque face~? Combien
de faces a un pav\'e~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:8}
Quels solides peuvent rouler~: le cube, le pav\'e, la boule, le
cylindre~? Pourquoi~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:shapes:9}
Une figure a $6$ c\^ot\'es. Comment s'appelle-t-elle~? Combien de
sommets a-t-elle~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:shapes:10}
Sur du papier quadrill\'e, dessine une maison~: un carr\'e pour le
mur ($4 \times 4$) avec un toit triangulaire au-dessus. Combien de
c\^ot\'es et de sommets a le contour entier de la maison~?
\end{exercise}
