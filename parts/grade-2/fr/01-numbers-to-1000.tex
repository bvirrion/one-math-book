\chapter{Les nombres jusqu'\`a 1000}\label{ch:g2:numbers}

L'ann\'ee~1 a regroup\'e les unit\'es en dizaines
(\cref{ch:g1:tens})~; maintenant dix dizaines se regroupent en une
\emph{centaine}, et les nombres s'\'etendent jusqu'\`a $1000$. Trois
chiffres, trois r\^oles~: centaines, dizaines, unit\'es.

\section{Centaines, dizaines, unit\'es}

\begin{definition}[Trois places]\label{def:g2:numbers:places}
Une \emph{centaine} c'est dix dizaines, c'est-\`a-dire dix paquets de
dix. Le nombre $346$ a trois parties~:
\[
346 = 3 \text{ centaines} + 4 \text{ dizaines} + 6 \text{ unit\'es}
= 300 + 40 + 6 .
\]
\end{definition}

\begin{example}[Le z\'ero garde la place]\label{ex:g2:numbers:zero}
\guillemotleft{} Trois cent cinq \guillemotright{} s'\'ecrit $305$~:
trois centaines, \emph{pas} de dizaine, cinq unit\'es. Le $0$ garde
le $5$ \`a la place des unit\'es. Sans lui, $35$ serait un nombre
compl\`etement diff\'erent.
\end{example}

\begin{example}[\'Echanges]\label{ex:g2:numbers:exchange}
Dix unit\'es font une dizaine, dix dizaines font une centaine. Donc
$10$ dizaines $= 100$, et $2$ centaines $= 20$ dizaines. Le nombre
$250$ contient $25$ dizaines, pas seulement le chiffre $5$.
\end{example}

\section{Comparer et ranger}

\begin{method}[Comparer des nombres \`a trois chiffres]\label{met:g2:numbers:compare}
\begin{enumerate}
  \item Plus de chiffres l'emporte~: $103 > 98$~;
  \item m\^eme longueur~: comparer de gauche \`a droite --- d'abord
        les centaines, puis les dizaines, puis les unit\'es. La
        premi\`ere diff\'erence d\'ecide~: $472 > 468$ car aux
        dizaines, $7 > 6$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g2:numbers:order}
Ranger $530$, $305$, $503$~: d'abord les centaines ($5$, $3$, $5$),
donc $305$ est le plus petit~; entre $530$ et $503$, les dizaines
d\'ecident ($3 > 0$)~:
\[
305 < 503 < 530 .
\]
\end{example}

\section{Sauts sur la droite num\'erique}

\begin{method}[Ajouter des dizaines et des centaines]\label{met:g2:numbers:jumps}
Sur la droite~:
\begin{enumerate}
  \item $+1$ / $-1$~: changer les unit\'es~;
  \item $+10$ / $-10$~: changer les dizaines ($362 + 10 = 372$)~;
  \item $+100$ / $-100$~: changer les centaines
        ($362 + 100 = 462$).
\end{enumerate}
Un seul chiffre bouge \`a chaque fois --- sauf si une place
\guillemotleft{} se remplit \guillemotright{} et d\'eborde, comme
$195 + 10 = 205$.
\end{method}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g2:numbers:1}
Combien de centaines, de dizaines et d'unit\'es dans~: $258$~;
\ $470$~; \ $600$~; \ $909$~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:2}
\'Ecris le nombre~: $4$ centaines, $0$ dizaine, $7$ unit\'es~; \
$6$ centaines et $3$ dizaines~; \ dix centaines.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:3}
\'Ecris en chiffres~: deux cent seize~; \ cinq cent quatre~; \
sept cent quatre-vingt-dix.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:4}
D\'ecompose comme dans \cref{def:g2:numbers:places}~: $438$~; \
$702$~; \ $560$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:5}
Recopie et compl\`ete avec $<$ ou $>$~:
\[
327 \;?\; 372, \qquad
580 \;?\; 508, \qquad
199 \;?\; 200, \qquad
640 \;?\; 604 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:6}
Range du plus petit au plus grand~: $430$~; \ $304$~; \ $340$~; \
$403$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:7}
Calcule en sautant sur la droite~: $253 + 10$~; \quad
$253 + 100$~; \quad $480 - 10$~; \quad $716 - 100$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:8}
Continue~: $195, 205, 215, ?, ?$. \quad
$300, 400, 500, ?, ?$. \quad Attention au premier~!
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:9}
Un magasin a $6$ bo\^ites de $100$ crayons et $5$ crayons isol\'es.
Combien de crayons en tout~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:numbers:10}
Avec les chiffres $2$, $5$ et $8$ utilis\'es une seule fois chacun,
\'ecris le plus grand nombre \`a trois chiffres et le plus petit.
Quelle est la distance entre eux~? (Tu peux les soustraire sur la
droite par sauts.)
\end{exercise}
