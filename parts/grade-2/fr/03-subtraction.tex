\chapter{Soustraction}\label{ch:g2:subtraction}

Quand les unit\'es du haut sont trop peu nombreuses pour enlever,
on d\'efait une dizaine --- c'est l'\emph{emprunt}, le miroir de la
retenue. Ce chapitre l'explique avec des b\^atonnets, puis
entra\^ine la soustraction \'ecrite, avec le comptage en montant
comme raccourci amical.

\section{Pourquoi on emprunte}

\begin{example}[D\'efaire une dizaine]\label{ex:g2:subtraction:unbundle}
$42 - 17$~: on veut enlever $7$ unit\'es de $2$ unit\'es --- trop
peu~! Alors d\'elier une des $4$ dizaines en dix unit\'es isol\'ees~:
maintenant le haut a $3$ dizaines et $12$ unit\'es. Enlever~:
$12 - 7 = 5$ unit\'es, et $3 - 1 = 2$ dizaines. Donc
$42 - 17 = 25$.
\end{example}

\section{La technique \'ecrite}

\begin{method}[Soustraction en colonnes]\label{met:g2:subtraction:column}
\begin{enumerate}
  \item \'Ecrire le plus grand nombre en haut, align\'e \`a droite~;
  \item soustraire les unit\'es~; si le chiffre du haut est trop
        petit, emprunter une dizaine (elle devient dix unit\'es dans
        la colonne en cours)~;
  \item soustraire les dizaines, puis les centaines~;
  \item \emph{v\'erifier}~: le r\'esultat plus le nombre enlev\'e
        doit redonner le nombre du haut.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g2:subtraction:worked}
Calculer $503 - 168$~:
\[
\begin{array}{r}
5\,0\,3 \\
-\ 1\,6\,8 \\
\hline
3\,3\,5
\end{array}
\]
Unit\'es~: $3 - 8$ trop petit~; emprunter \`a travers les dizaines
(il n'y en a pas, donc emprunter d'abord aux centaines)~: le haut
devient $4$ centaines, $9$ dizaines, $13$ unit\'es. Puis
$13 - 8 = 5$~; $9 - 6 = 3$~; $4 - 1 = 3$. V\'erification~:
$335 + 168 = 503$. \checkmark
\end{example}

\section{Compter en montant}

\begin{method}[Soustraire en comptant en montant]\label{met:g2:subtraction:countup}
Quand les nombres sont proches, monter du plus petit au plus grand
et additionner les sauts~:
\[
83 - 78: \quad 78 \to 80 \text{ c'est } 2, \ 80 \to 83 \text{ c'est } 3,
\text{ donc } 83 - 78 = 5 .
\]
Compter en montant est plus rapide qu'emprunter quand les deux
nombres sont voisins --- et c'est exactement comment un commer\c{c}ant
rend la monnaie.
\end{method}

  \node[omDef, font=\small] at (79,1.0) {$+2$};

\end{tikzpicture}

{\small $83 - 78$ en comptant en montant~: deux sauts, $2 + 3 = 5$.}
\end{omfigure}

\begin{example}[Rendre la monnaie]\label{ex:g2:subtraction:change}
Un jouet co\^ute $34$~; on paie avec un billet de $50$. Monnaie~:
compter en montant, $34 \to 40$ c'est $6$, $40 \to 50$ c'est $10$,
donc la monnaie est $16$. La soustraction $50 - 34 = 16$, faite \`a
la mani\`ere du commer\c{c}ant.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:1}
Calcule (sans emprunt)~: $58 - 23$~; \quad $176 - 45$~; \quad
$389 - 164$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:2}
Calcule avec un emprunt~: $52 - 17$~; \quad $73 - 28$~; \quad
$60 - 24$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:3}
Calcule en colonnes et v\'erifie chaque r\'esultat~: $324 - 167$~;
\quad $500 - 236$~; \quad $703 - 158$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:4}
Calcule en comptant en montant~: $62 - 58$~; \quad $91 - 85$~; \quad
$100 - 96$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:5}
Calcule la monnaie (compte en montant)~: payer $50$ pour un livre \`a
$42$~; payer $20$ pour un jouet \`a $13$~; payer $100$ pour un sac \`a
$75$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:6}
\guillemotleft{} Un arbre mesure $156$~cm de haut. Il a grandi de
$28$~cm cette ann\'ee. Quelle \'etait sa hauteur l'an dernier~?
\guillemotright{} \'Ecris la soustraction et la r\'eponse.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:7}
Compl\`ete avec l'addition jumelle pour v\'erifier~:
$85 - 39 = 46$~; est-ce juste~? $214 - 68 = 156$~; est-ce juste~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:8}
Pour chaque histoire, choisis $+$ ou $-$~:
\guillemotleft{} $47$ oiseaux, $19$ s'envolent \guillemotright{}~;
\guillemotleft{} $47$ pommes, $19$ de plus cueillies
\guillemotright{}~; \guillemotleft{} Sam a $47$, L\'ea a $19$~:
combien de plus a Sam~? \guillemotright{}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:subtraction:9}
Remplis le trou~: $63 - \;?\; = 45$. (Compte de $45$ \`a $63$ en
montant.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:subtraction:10}
Un jeu~: pense \`a un nombre, ajoute $30$, puis enl\`eve $18$. Le
r\'esultat est $58$. Quel \'etait le nombre~? (Remonte en arri\`ere~!)
\end{exercise}
