\chapter{Argent et mesures}\label{ch:g2:measure}

R\`egles \`a mesurer, balances de cuisine, pichets, pi\`eces,
horloges et calendriers~: ce chapitre parle des nombres de la vie
quotidienne --- combien de long, combien de lourd, combien \c{c}a
contient, combien \c{c}a co\^ute, et quelle heure il est.

\section{Longueurs, masses, contenances}

\begin{definition}[Unit\'es du quotidien]\label{def:g2:measure:units}
\begin{itemize}
  \item \emph{Longueur}~: le centim\`etre (cm) pour les petites
        choses, le m\`etre (m) pour les grandes, avec
        $1$~m $= 100$~cm~;
  \item \emph{masse}\index{masse}~: le gramme (g) pour les choses l\'eg\`eres, le
        kilogramme (kg) pour les lourdes, avec
        $1$~kg $= 1000$~g~;
  \item \emph{contenance}\index{contenance} (ce qu'un r\'ecipient peut contenir)~: le
        litre (L).
\end{itemize}
Les unit\'es plus petites et leurs r\`egles exactes arrivent dans
\cref{ch:g3:measure}.
\end{definition}

\begin{example}[Quelle unit\'e~?]\label{ex:g2:measure:choose}
Un crayon~: environ $9$~cm. Une porte~: environ $2$~m. Une pomme~:
environ $150$~g. Un gros sac de farine~: $1$~kg. Une bouteille de
lait~: $1$~L. Deviner l'unit\'e avant de mesurer --- \c{c}a \'evite
des r\'eponses absurdes comme \guillemotleft{} la porte fait
$2$~cm \guillemotright{}.
\end{example}

\section{L'argent}

\begin{method}[Composer une somme]\label{met:g2:measure:money}
Avec des pi\`eces et des billets de $1$, $2$, $5$, $10$, $20$~:
\begin{enumerate}
  \item commencer par le plus grand qui convient, puis compl\'eter
        le reste~;
  \item pour faire $27$~: un $20$, un $5$ et un $2$~;
  \item il y a souvent plusieurs fa\c{c}ons --- $27$ c'est aussi
        $10 + 10 + 5 + 2$.
\end{enumerate}
\end{method}

\begin{example}[La monnaie]\label{ex:g2:measure:change}
Un livre co\^ute $14$~; on paie avec un billet de $20$. La monnaie
est $20 - 14 = 6$, trouv\'ee en comptant en montant~:
$14 \to 20$ c'est $6$ (\cref{met:g2:subtraction:countup}).
\end{example}

\section{Le temps}

\begin{method}[Demies et quarts d'heure]\label{met:g2:measure:clock}
La grande aiguille indique les minutes pass\'ees de l'heure~:
\begin{enumerate}
  \item tout droit en haut ($12$)~: pile --- $3{:}00$~;
  \item quart de tour \`a droite ($3$)~: et quart --- $3{:}15$~;
  \item tout droit en bas ($6$)~: et demie --- $3{:}30$~;
  \item trois quarts de tour ($9$)~: moins le quart de l'heure
        suivante --- $3{:}45$, \guillemotleft{} quatre heures moins
        le quart \guillemotright{}.
\end{enumerate}
\end{method}

\begin{example}[Le calendrier]\label{ex:g2:measure:calendar}
Une ann\'ee a $12$ mois, dans l'ordre~: janvier, f\'evrier, mars,
avril, mai, juin, juillet, ao\^ut, septembre, octobre, novembre,
d\'ecembre. La plupart des mois ont $30$ ou $31$ jours~; f\'evrier
en a $28$ (ou $29$). Une date nomme le jour et le mois~:
\guillemotleft{} le $5$ mars \guillemotright{}.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g2:measure:1}
Choisis l'unit\'e (cm, m, g, kg ou L)~: un crayon~; la longueur de
la classe~; une plume~; un cartable~; une bouteille d'eau.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:2}
Convertis~: $2$~m en cm~; $300$~cm en m~; $1$~kg en g.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:3}
Compose ces \omterm{def:g1:addition:def}{sommes} avec des pi\`eces et des billets~: $16$~; \quad
$23$~; \quad $35$. Donne une fa\c{c}on pour chacune.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:4}
Calcule la monnaie~: payer $10$ pour un jouet \`a $6$~; payer $20$
pour un livre \`a $17$~; payer $50$ pour un jeu \`a $32$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:5}
Lis l'heure~: grande aiguille sur $6$, petite entre $4$ et $5$~;
grande aiguille sur $3$, petite juste apr\`es $8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:6}
Dessine une horloge indiquant deux heures et demie. O\`u pointe
chaque aiguille~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:7}
Nomme le mois~: apr\`es mai~; avant janvier~; le mois de ton
anniversaire.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:8}
L\'ea ach\`ete une boisson \`a $3$ et un g\^ateau \`a $4$. Elle paie
avec un billet de $10$. Combien de monnaie~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:measure:9}
Lequel est plus lourd~: un sac de sucre de $1$~kg ou $800$~g de
pommes~? De combien de grammes~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:measure:10}
Fais $40$ avec exactement quatre pi\`eces ou billets (choisis parmi
$1$, $2$, $5$, $10$, $20$). Trouve deux fa\c{c}ons diff\'erentes.
\end{exercise}
