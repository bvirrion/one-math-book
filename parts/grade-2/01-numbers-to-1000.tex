\chapter{Numbers to 1000}\label{ch:g2:numbers}

Grade 1 bundled units into tens (\cref{ch:g1:tens}); now ten tens
bundle into a \emph{hundred}, and the numbers stretch all the way to
$1000$. Three digits, three jobs: hundreds, tens, units.

\section{Hundreds, tens, units}

\begin{definition}[Three places]\label{def:g2:numbers:places}
A \emph{hundred} is ten tens, that is ten bundles of ten. The
number $346$ has three parts:
\[
346 = 3 \text{ hundreds} + 4 \text{ tens} + 6 \text{ units}
= 300 + 40 + 6 .
\]
\end{definition}

\begin{example}[The zero holds the place]\label{ex:g2:numbers:zero}
``Three hundred and five'' is $305$: three hundreds, \emph{no}
tens, five units. The $0$ keeps the $5$ in the units place. Without
it, $35$ would be a completely different number.
\end{example}

\begin{example}[Exchanging]\label{ex:g2:numbers:exchange}
Ten units make a ten, ten tens make a hundred. So $10$ tens
$= 100$, and $2$ hundreds $= 20$ tens. The number $250$ contains
$25$ tens, not just the digit $5$.
\end{example}

\section{Comparing and ordering}

\begin{method}[Comparing three-digit numbers]\label{met:g2:numbers:compare}
\begin{enumerate}
  \item More digits wins: $103 > 98$;
  \item same length: compare from the left --- hundreds first, then
        tens, then units. The first difference decides:
        $472 > 468$ because at the tens, $7 > 6$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g2:numbers:order}
Order $530$, $305$, $503$: hundreds first ($5$, $3$, $5$), so
$305$ leads; between $530$ and $503$, the tens decide ($3 > 0$):
\[
305 < 503 < 530 .
\]
\end{example}

\section{Jumps on the number line}

\begin{method}[Adding tens and hundreds]\label{met:g2:numbers:jumps}
On the line:
\begin{enumerate}
  \item $+1$ / $-1$: change the units;
  \item $+10$ / $-10$: change the tens ($362 + 10 = 372$);
  \item $+100$ / $-100$: change the hundreds ($362 + 100 = 462$).
\end{enumerate}
Only one digit moves each time --- unless a place ``fills up'' and
carries over, like $195 + 10 = 205$.
\end{method}

\begin{tikzpicture}[scale=1.0]

\end{tikzpicture}

\end{omfigure}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g2:numbers:1}
How many hundreds, tens and units in: $258$; \ $470$; \ $600$; \
$909$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:2}
Write the number: $4$ hundreds, $0$ tens, $7$ units; \ $6$ hundreds
and $3$ tens; \ ten hundreds.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:3}
Write in digits: two hundred sixteen; \ five hundred and four; \
seven hundred ninety.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:4}
Decompose as in \cref{def:g2:numbers:places}: $438$; \ $702$; \
$560$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:5}
Copy and complete with $<$ or $>$:
\[
327 \;?\; 372, \qquad
580 \;?\; 508, \qquad
199 \;?\; 200, \qquad
640 \;?\; 604 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:6}
Order from smallest to biggest: $430$; \ $304$; \ $340$; \ $403$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:7}
Compute by jumping on the line: $253 + 10$; \quad $253 + 100$;
\quad $480 - 10$; \quad $716 - 100$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:8}
Count on: $195, 205, 215, ?, ?$. \quad
$300, 400, 500, ?, ?$. \quad Careful with the first one!
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g2:numbers:9}
A shop has $6$ boxes of $100$ pencils and $5$ loose pencils. How
many pencils in all?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g2:numbers:10}
Using the digits $2$, $5$ and $8$ once each, write the biggest
three-digit number and the smallest one. How far apart are they?
(You may add them up on the line by jumps.)
\end{exercise}
