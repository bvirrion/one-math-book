\chapter{Adição com transporte}\label{ch:g2:addition}

Quando as unidades de um \omterm{def:g1:addition:def}{soma} passe nove, dez deles agrupados em um novo
dez --- e esse pequeno pacote, movido para a próxima coluna, é o
famoso \emph{carregar}. Este capítulo explica o transporte com paus,
em seguida, treina o método escrito.

\section{Por que carregamos}

\begin{example}[Pacote de dez unidades]\label{ex:g2:addition:bundle}
$27 + 15$: juntar as unidades, $7 + 5 = 12$ unidades. Mas dez
unidades fazem dez! Agrupe dez deles: isso deixa $2$ unidades soltas
e um novo dez para carregar. Agora as dezenas: $2 + 1 + 1 = 4$ dezenas. Então
$27 + 15 = 42$.
\end{example}

\section{O método escrito}

\begin{method}[Adição de coluna]\label{met:g2:addition:column}
\begin{enumerate}
  \item Escreva os números um abaixo do outro, unidades sob unidades,
        dezenas sob dezenas;
  \item adicione as unidades; se a resposta passar $9$, escreva as unidades
        dígito e carregue um pequeno $1$ acima das dezenas;
  \item adicione as dezenas (com o carry) e depois as centenas;
  \item verifique com uma estimativa: arredonde cada número e some aproximadamente.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g2:addition:worked}
Calcular $268 + 156$:
\[
\begin{array}{r}
{\scriptstyle 1}\ {\scriptstyle 1}\ \phantom{0} \\[-3pt]
2\,6\,8 \\
+\ 1\,5\,6 \\
\hline
4\,2\,4
\end{array}
\]
Unidades: $8 + 6 = 14$: escrever $4$, carregar $1$. Dezenas:
$6 + 5 + 1 = 12$: escrever $2$, carregar $1$. Centenas:
$2 + 1 + 1 = 4$. Verificação de estimativa: sobre $270 + 160 = 430$, perto de
$424$. \checkmark
\end{example}

\section{Adicionando em sua cabeça}

\begin{example}[Faça um dez]\label{ex:g2:addition:mental}
Para pequenos \omterm{def:g1:addition:def}{somas}, pule primeiro para um número redondo:
\[
48 + 6 = 48 + 2 + 4 = 50 + 4 = 54 .
\]
Adicione dezenas e unidades separadamente:
$35 + 23 = (30 + 20) + (5 + 3) = 50 + 8 = 58$. E quase cem
truque: $99 + 46 = 100 + 46 - 1 = 145$.
\end{example}

\begin{example}[Um pequeno problema]\label{ex:g2:addition:problem}
``A biblioteca tem $185$ livros; $57$ chegam novos. Quantos livros
agora?'' Adicionando: $185 + 57 = 242$ (unidades $5 + 7 = 12$, carregar\,\dots).
A biblioteca tem $242$ livros.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g2:addition:1}
Calcular (sem necessidade de transporte): $34 + 25$; \quad $126 + 43$; \quad
$507 + 82$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:addition:2}
Calcule com um carry: $47 + 8$; \quad $56 + 27$; \quad $65 + 19$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:addition:3}
Calcule em colunas: $268 + 145$; \quad $374 + 158$; \quad
$429 + 293$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:addition:4}
Calcule mentalmente fazendo um dez: $38 + 5$; \quad $67 + 6$;
\quad $54 + 8$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:addition:5}
Calcule adicionando dezenas e unidades separadamente: $42 + 36$; \quad
$53 + 24$; \quad $61 + 28$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:addition:6}
Estime primeiro (arredondar para a dezena ou centena mais próxima) e depois calcule
exatamente: $58 + 34$; \quad $296 + 187$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:addition:7}
``Ana tem $128$ bolinhas de gude, Tom tem $95$. Quantos juntos?'' Escreva
a adição e a sentença de resposta.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:addition:8}
Complete o buraco das unidades: $3\,?\, + 27 = 64$ (encontre as unidades que faltam
algarismo do primeiro número).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:addition:9}
Duas turmas vão ao show: $27$ e $28$ crianças, mais $6$
adultos. Quantas pessoas ao todo? (Adicione as duas classes primeiro.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g2:addition:10}
Encontre três números na lista $27$, $33$, $40$, $65$ que somam
para $100$. (Procure dois cujas unidades sejam dez amigas primeiro.)
\end{exercise}
