\chapter{Formas e Sólidos}\label{ch:g2:shapes}

As formas planas vivem no papel; sólidos enchem suas mãos. Este capítulo
classifica formas planas contando lados e cantos, testa o correto
ângulo com um canto de papel dobrado e encontra os primeiros sólidos ---
cujo \omterm{def:g2:shapes:solids}{rostos} são as formas planas que já conhecemos.

\section{Classificando formas planas}

\begin{definition}[Reto e redondo]\label{def:g2:shapes:sort}
Uma forma cuja borda é feita apenas de lados retos é uma
\emph{polígono}\index{polígono}: triângulo ($3$ lados), quadrilátero ($4$ lados:
quadrados e retângulos são quadriláteros), pentágono ($5$),
hexágono ($6$). Uma forma com borda curva, como o círculo, é
\emph{não} um polígono.
\end{definition}

\begin{method}[Testando um ângulo reto]\label{met:g2:shapes:rightangle}
Dobre uma folha de papel duas vezes para fazer um canto pontiagudo --- aquele canto
é um ângulo reto. Coloque-o no canto de uma forma:
\begin{enumerate}
  \item se os dois lados da forma correrem exatamente ao longo da dobra
        canto, o ângulo está certo;
  \item se o canto for mais largo ou mais estreito, não está certo.
\end{enumerate}
Um quadrado e um retângulo têm quatro ângulos retos.
\end{method}

\section{Reproduzindo em uma grade}

\begin{method}[Copiando uma figura]\label{met:g2:shapes:copy}
Em papel quadriculado, copie uma figura vértice por vértice:
\begin{enumerate}
  \item escolha um canto e marque-o na grade vazia;
  \item conte os quadrados até o próximo canto (``$3$ certo, $2$
        up'') e marque-o;
  \item junte os cantos em ordem; compare com o modelo.
\end{enumerate}
\end{method}

\section{Sólidos}

\begin{definition}[Primeiros sólidos]\label{def:g2:shapes:solids}
Os sólidos ocupam espaço. Seus lados planos são \emph{rostos}\index{rostos}:
\begin{itemize}
  \item o \emph{cubo}\index{cubo}: $6$ faces quadradas (um dado);
  \item o \emph{caixa} (prisma retangular): $6$ faces retangulares (uma
        pacote de cereais);
  \item o \emph{cilindro}\index{cilindro}: duas faces redondas e um lado enrolado (um
        pode);
  \item o \emph{bola}: perfeitamente redondo, sem face plana.
\end{itemize}
Sólidos e suas faces retornam \cref{ch:g5:solids}.
\end{definition}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g2:shapes:1}
Para cada um, conte os lados e cantos e nomeie-os: uma forma com $5$
lados; uma forma com $3$ cantos; uma forma redonda.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:shapes:2}
Classifique em ``\omterm{def:g2:shapes:sort}{polígono}'' ou ``não é um \omterm{def:g2:shapes:sort}{polígono}'': um quadrado; um círculo; um
triângulo; um hexágono.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:shapes:3}
Faça um testador de ângulo reto dobrando papel. Encontre três ângulos retos
na sala e um ângulo que não está certo.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:shapes:4}
Quantos ângulos retos tem um quadrado? Um retângulo? Um triângulo pode ter
no máximo quantos? (Tente desenhar um com dois.)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:shapes:5}
Em papel quadriculado, copie um retângulo que seja $4$ quadrados de largura e $2$
alto, depois um triângulo com um $3$-base quadrada.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:shapes:6}
Nomeie o sólido: um dado; uma lata de sopa; uma bola de futebol; uma caixa de sapatos.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:shapes:7}
Quantos \omterm{def:g2:shapes:solids}{rostos} tem um \omterm{def:g2:shapes:solids}{cubo}? Qual é a forma de cada face? Quantos \omterm{def:g2:shapes:solids}{rostos}
tem uma caixa?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:shapes:8}
Quais sólidos podem rolar: o \omterm{def:g2:shapes:solids}{cubo}, a caixa, a bola, o \omterm{def:g2:shapes:solids}{cilindro}?
Por que?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:shapes:9}
Uma forma tem $6$ lados. Como é chamado? Quantos cantos tem?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g2:shapes:10}
Em papel quadriculado, desenhe uma casa: um quadrado para a parede
($4 \times 4$) com um telhado triangular no topo. Quantos lados e
cantos tem todo o contorno da casa?
\end{exercise}
