\chapter{Números até 1000}\label{ch:g2:numbers}

Unidades agrupadas de grau 1 em dezenas (\cref{ch:g1:tens}); agora dez dezenas
agrupar em um \emph{centenas}, e os números se estendem até
$1000$. Três dígitos, três empregos: centenas, dezenas, unidades.

\section{Centenas, dezenas, unidades}

\begin{definition}[Três lugares]\label{def:g2:numbers:places}
UM \emph{centenas} são dez dezenas, ou seja, dez maços de dez. O
número $346$ tem três partes:
\[
346 = 3 \text{ hundreds} + 4 \text{ tens} + 6 \text{ units}
= 300 + 40 + 6 .
\]
\end{definition}

\begin{example}[O zero ocupa o lugar]\label{ex:g2:numbers:zero}
`` Trezentos e cinco '' é $305$: trezentos, \emph{não}
dezenas, cinco unidades. O $0$ mantém o $5$ no lugar das unidades. Sem
isso, $35$ seria um número completamente diferente.
\end{example}

\begin{example}[Troca]\label{ex:g2:numbers:exchange}
Dez unidades equivalem a dez, dez dezenas equivalem a cem. Então $10$ dezenas
$= 100$, e $2$ centenas $= 20$ dezenas. O número $250$ contém
$25$ dezenas, não apenas o dígito $5$.
\end{example}

\section{Comparando e ordenando}

\begin{method}[Comparando números de três dígitos]\label{met:g2:numbers:compare}
\begin{enumerate}
  \item Mais dígitos ganhos: $103 > 98$;
  \item mesmo comprimento: compare a partir da esquerda --- centenas primeiro, depois
        dezenas e depois unidades. A primeira \omterm{ex:g1:subtraction:difference}{diferença} decide:
        $472 > 468$ porque nas dezenas, $7 > 6$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g2:numbers:order}
Ordem $530$, $305$, $503$: centenas primeiro ($5$, $3$, $5$), então
$305$ pistas; entre $530$ e $503$, as dezenas decidem ($3 > 0$):
\[
305 < 503 < 530 .
\]
\end{example}

\section{Salta na reta numérica}

\begin{method}[Adicionando dezenas e centenas]\label{met:g2:numbers:jumps}
Na linha:
\begin{enumerate}
  \item $+1$ / $-1$: alterar as unidades;
  \item $+10$ / $-10$: mude as dezenas ($362 + 10 = 372$);
  \item $+100$ / $-100$: mude as centenas ($362 + 100 = 462$).
\end{enumerate}
Apenas um dígito se move de cada vez --- a menos que uma casa seja "preenchida" e
transporta, como $195 + 10 = 205$.
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g2:numbers:1}
Quantas centenas, dezenas e unidades existem em: $258$; \ $470$; \ $600$; \
$909$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:2}
Escreva o número: $4$ centenas, $0$ dezenas, $7$ unidades; \ $6$ centenas
e $3$ dezenas; \ dez centenas.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:3}
Escreva em algarismos: duzentos e dezesseis; \ quinhentos e quatro; \
setecentos e noventa.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:4}
Decompor como em \cref{def:g2:numbers:places}: $438$; \ $702$; \
$560$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:5}
Copie e complete com $<$ ou $>$:
\[
327 \;?\; 372, \qquad
580 \;?\; 508, \qquad
199 \;?\; 200, \qquad
640 \;?\; 604 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:6}
Ordene do menor para o maior: $430$; \ $304$; \ $340$; \ $403$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:7}
Calcule pulando na linha: $253 + 10$; \quad $253 + 100$;
\quad $480 - 10$; \quad $716 - 100$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:8}
Contar com: $195, 205, 215, ?, ?$. \quad
$300, 400, 500, ?, ?$. \quad Cuidado com o primeiro!
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:numbers:9}
Uma loja tem $6$ caixas de $100$ lápis e $5$ lápis soltos. Como
muitos lápis ao todo?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g2:numbers:10}
Usando os dígitos $2$, $5$ e $8$ uma vez cada, escreva o maior
número de três dígitos e o menor. A que distância eles estão?
(Você pode adicioná-los na linha por saltos.)
\end{exercise}
