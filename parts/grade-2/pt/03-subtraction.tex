\chapter{Subtração}\label{ch:g2:subtraction}

Quando as unidades no topo são muito poucas para serem retiradas, abrimos um
dez --- o \emph{emprestar}, o espelho do transporte. Este capítulo
explica com paus, depois treina a subtração escrita, com
contando como um atalho amigável.

\section{Por que emprestamos}

\begin{example}[Desempacotando um dez]\label{ex:g2:subtraction:unbundle}
$42 - 17$: queremos levar $7$ unidades de $2$ unidades --- muito poucas!
Então desamarre um dos $4$ dezenas em dez unidades soltas: agora o topo tem
$3$ dezenas e $12$ unidades. Remover: $12 - 7 = 5$ unidades, e
$3 - 1 = 2$ dezenas. Então $42 - 17 = 25$.
\end{example}

\section{O método escrito}

\begin{method}[Subtração de coluna]\label{met:g2:subtraction:column}
\begin{enumerate}
  \item Escreva o número maior em cima, alinhado à direita;
  \item subtraia as unidades; se o dígito superior for muito pequeno, peça emprestado
        uma dezena (torna-se dez unidades na coluna atual);
  \item subtraia as dezenas e depois as centenas;
  \item \emph{verificar}: a resposta mais o número subtraído deve
        devolva o número superior.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g2:subtraction:worked}
Calcular $503 - 168$:
\[
\begin{array}{r}
5\,0\,3 \\
-\ 1\,6\,8 \\
\hline
3\,3\,5
\end{array}
\]
Unidades: $3 - 8$ muito pequeno; emprestar até as dezenas (não há nenhum,
então pegue emprestado das centenas primeiro): o topo se torna $4$ centenas,
$9$ dezenas, $13$ unidades. Então $13 - 8 = 5$; $9 - 6 = 3$;
$4 - 1 = 3$. Verificar: $335 + 168 = 503$. \checkmark
\end{example}

\section{Contando}

\begin{method}[Subtraindo contando]\label{met:g2:subtraction:countup}
Para números próximos, suba do menor para o grande
one e adicione o lúpulo:
\[
83 - 78: \quad 78 \to 80 \text{ is } 2, \ 80 \to 83 \text{ is } 3,
\text{ so } 83 - 78 = 5 .
\]
Contar é mais rápido do que pedir emprestado quando os dois números estão próximos
uns aos outros --- e é exatamente assim que um lojista dá troco.
\end{method}

\begin{example}[Dando troco]\label{ex:g2:subtraction:change}
Um brinquedo custa $34$; você paga com um $50$. Mudança: contagem crescente,
$34 \to 40$ é $6$, $40 \to 50$ é $10$, então a mudança é $16$.
A subtração $50 - 34 = 16$, feito do jeito do lojista.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g2:subtraction:1}
Calcular (sem necessidade de empréstimo): $58 - 23$; \quad $176 - 45$; \quad
$389 - 164$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:subtraction:2}
Calcule com um empréstimo: $52 - 17$; \quad $73 - 28$; \quad
$60 - 24$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:subtraction:3}
Calcule em colunas e verifique cada uma: $324 - 167$; \quad
$500 - 236$; \quad $703 - 158$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:subtraction:4}
Calcule contando: $62 - 58$; \quad $91 - 85$; \quad
$100 - 96$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:subtraction:5}
Calcule o troco (contagem): pague $50$ por um $42$ livro; pagar $20$
por um $13$ brinquedo; pagar $100$ por um $75$ bolsa.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:subtraction:6}
``Uma árvore é $156$ cm de altura. Cresceu $28$ cm este ano. Quão alto era
foi no ano passado?'' Escreva a subtração e responda.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:subtraction:7}
Complete com a adição dupla para verificar: $85 - 39 = 46$; é isso
certo? $214 - 68 = 156$; está certo?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:subtraction:8}
Para cada história, escolha $+$ ou $-$: ``$47$ pássaros, $19$ voar para longe'';
``$47$ maçãs, $19$ mais escolhido''; ``Sam tem $47$, Léa tem $19$:
quantos mais Sam tem?''.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:subtraction:9}
Preencha o buraco: $63 - \;?\; = 45$. (Contar a partir de $45$ para $63$.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g2:subtraction:10}
Um jogo: pense em um número, some $30$, então tire $18$. O
o resultado é $58$. Qual era o número? (Trabalhe para trás!)
\end{exercise}
