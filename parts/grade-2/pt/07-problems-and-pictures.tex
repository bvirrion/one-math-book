\chapter{Resolvendo Problemas}\label{ch:g2:problems}

Uma história com números esconde uma pergunta: adicionar? remover? multiplicar?
Este capítulo dá uma maneira de decidir, usa imagens simples de barras para
\emph{ver} a operação e lê informações de tabelas e
pictogramas.

\section{Escolhendo a operação}

\begin{method}[Qual operação?]\label{met:g2:problems:choose}
\begin{enumerate}
  \item \emph{reunindo, em todos}: adicionar ($+$);
  \item \emph{tirando, quantos sobraram, quantos mais}: subtrair
        ($-$);
  \item \emph{tantos grupos iguais}: multiplicar ($\times $);
  \item faça um desenho rápido quando não tiver certeza e sempre responda com um
        sentença.
\end{enumerate}
\end{method}

\begin{example}[Um problema em duas etapas]\label{ex:g2:problems:twostep}
``Sam compra $3$ pacotes de $6$ adesivos e depois dá $5$ ausente. Quantos
ele mantém?
\begin{enumerate}
  \item Grupos: $3 \times 6 = 18$ adesivos;
  \item doar: $18 - 5 = 13$.
\end{enumerate}
Sam mantém $13$ adesivos. Releia a pergunta: a história é realmente
terminou? Aqui sim.
\end{example}

\begin{example}[Um número extra oculto]\label{ex:g2:problems:extra}
``Um $210$ livro de páginas\,\dots'' --- cuidado, nem todos os números em um
é necessária uma história. ``Ben tem $8$ vermelho e $6$ bolinhas de gude azuis e é $7$
anos. Quantas bolinhas de gude?'' A idade é uma isca: $8 + 6 = 14$
bolinhas de gude. Classifique os números antes de calcular.
\end{example}

\section{Tabelas e pictogramas}

\begin{example}[Uma mesa]\label{ex:g2:problems:table}
Bolos vendidos em uma barraca:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|ccc}
 & Saturday & Sunday & total \\
\hline
muffins & $12$ & $9$ & $21$ \\
tarts & $7$ & $10$ & $17$ \\
\end{tabular}
\end{center}
Leitura: Tortas de domingo, $10$. Computação: muffins durante os dois dias,
$12 + 9 = 21$; mais tortas no domingo do que no sábado $10 - 7 = 3$.
\end{example}

\begin{example}[Um pictograma]\label{ex:g2:problems:pictogram}
Leia a chave primeiro: aqui uma maçã representa $2$ pedaços de fruta.

\begin{tikzpicture}[scale=0.62]

  \foreach \i in {0,1,2,3} \fill[omThm] (0.5+\i*0.75,2) circle (0.24);
  \node[left, font=\small] at (0,1) {green};
  \foreach \i in {0,1} \fill[omMeth] (0.5+\i*0.75,1) circle (0.24);

  \foreach \i in {0,1,2} \fill[omProp] (0.5+\i*0.75,0) circle (0.24);

\end{tikzpicture}

{\small Um pictograma: vermelho mostra $4$ pontos, então $4 \times 2 = 8$ vermelho
frutas; verde $2$ pontos, $4$ frutas; amarelo $3$ pontos, $6$ frutas.}
\end{omfigure}
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g2:problems:1}
Nomeie a operação (não calcule): ``$34$ e $28$ bolinhas de gude colocadas
juntos''; ``$50$ doces, $12$ comido''; ``$4$ caixas de $5$
maçãs''; ``Ana tem $30$, Leão tem $18$: quantos mais Ana tem?''.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:problems:2}
Resolva: ``Um estacionamento tem $156$ carros; $48$ deixar. Quantos ficam?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:problems:3}
Resolva: ``$5$ cada criança escolhe $4$ flores. Quantas flores tem
tudo?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:problems:4}
Desenhe uma barra e resolva: ``Tom tem $45$ cartas, Mia tem $17$
menos. Quantos tem Mia?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:problems:5}
Duas etapas: ``Uma caixa contém $6$ garrafas. Uma loja recebe $4$ caixas e
vende $9$ garrafas. Quantas garrafas restam?''
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:problems:6}
Encontre e risque o número inútil e resolva: ``Ben é $8$
anos e tem $23$ bolinhas de gude. Ele vence $15$ mais. Quantas bolinhas de gude
agora?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:problems:7}
Usando a tabela de \cref{ex:g2:problems:table}: quantas tortas foram vendidas
durante os dois dias? Quantos bolos de todos os tipos no sábado?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:problems:8}
Usando o pictograma de \cref{ex:g2:problems:pictogram}: quantos
frutas ao todo? Qual cor tem mais?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:problems:9}
Resolva: ``Uma fita é $80$ cm de comprimento. Lea interrompe $35$ cm. Quanto tempo é
a peça que resta?''
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g2:problems:10}
``Um padeiro faz $5$ bandejas de $4$ bolos pela manhã e $3$ bandejas
de $4$ bolos à tarde. Quantos bolos naquele dia?'' (Dois
multiplicações, então um \omterm{def:g1:addition:def}{soma} --- ou conte as bandejas primeiro!)
\end{exercise}
