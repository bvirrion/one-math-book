\chapter{Multiplicação: primeiros passos}\label{ch:g2:mult}

Quatro caixas de três bolos: em vez de $3 + 3 + 3 + 3$, escrevemos
$4 \times 3$. Multiplicação é o atalho para somar o mesmo
número muitas vezes --- e sua melhor imagem é um conjunto organizado de linhas
e colunas.

\section{Adicionando o mesmo número}

\begin{definition}[Multiplicação]\label{def:g2:mult:def}
$4 \times 3$ significa ``quatro vezes três'':
\[
4 \times 3 = 3 + 3 + 3 + 3 = 12 .
\]
O sinal é $\times $, leia ``vezes''. A resposta é a
\emph{produto}\index{produto} (a multiplicação cresce em \cref{ch:g3:mult}).
\end{definition}

\begin{omfigure}
\begin{tikzpicture}[scale=0.5]
  \foreach \r in {0,1,2,3}
    \foreach \c in {0,1,2}
      \fill[omDef] (\c,-\r) circle (0.28);
  \node[right, font=\small] at (3.0,-1.5)
    {$4$ rows of $3$ dots: $4 \times 3 = 12$};
\end{tikzpicture}

{\small Um \emph{variedade}: $4$ linhas de $3$. Contar por linhas dá
$4 \times 3$; os mesmos pontos por colunas dão $3 \times 4$ --- então o
a ordem não importa.}
\end{omfigure}

\begin{example}[A ordem não importa]\label{ex:g2:mult:order}
$4 \times 3$ e $3 \times 4$ conte o mesmo $12$ pontos, um por
linhas, uma por colunas. Virar o array de lado prova isso. Então
aprendendo um \omterm{def:g2:mult:def}{produto} dá seu gêmeo de graça.
\end{example}

\section{Pule a contagem e as tabelas fáceis}

\begin{method}[Construindo uma mesa contando em saltos]\label{met:g2:mult:skip}
Para encontrar $5 \times 4$, conte de quatro em cinco vezes:
$4, 8, 12, 16, 20$. Pule a contagem na reta numérica \emph{é} o
mesa.
\begin{itemize}
  \item Tabela de $2$: $2, 4, 6, 8, 10, \dots $ (as duplas);
  \item tabela de $5$: $5, 10, 15, 20, 25, \dots $ (termina em $5$ ou
        $0$);
  \item tabela de $10$: $10, 20, 30, \dots $ (basta adicionar um \omterm{def:g1:counting:zero}{zero}).
\end{itemize}
\end{method}

\end{tikzpicture}

{\small A mesa de $2$ como saltos na linha: $2, 4, 6, 8, 10$. Cinco
lúpulo de $2$ alcançar $10$, então $5 \times 2 = 10$.}
\end{omfigure}

\begin{example}[Duplas e metades]\label{ex:g2:mult:double}
Multiplicando por $2$ está duplicando: $2 \times 7 = 14$. O inverso,
reduzir pela metade, desfaz: metade de $14$ é $7$. Duplas conhecidas de cor
faça a mesa inteira $2$ instantâneo --- e prepare a divisão
(\cref{ch:g3:division}).
\end{example}

\section{Multiplicando em problemas}

\begin{example}\label{ex:g2:mult:problem}
``Uma caixa contém $5$ ovos. Quantos ovos tem $4$ caixas?'' Quatro grupos de
cinco: $4 \times 5 = 20$ (pular contagem: $5, 10, 15, 20$). Há
$20$ ovos.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g2:mult:1}
Escreva como uma multiplicação e calcule: $2 + 2 + 2$; \quad
$5 + 5 + 5 + 5$; \quad $10 + 10 + 10$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:mult:2}
Desenhe uma matriz de $3$ linhas de $5$ pontos. Que multiplicação faz
mostrar? E virou de lado?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:mult:3}
Pule a contagem para preencher a tabela de $2$ até $2 \times 6$; a mesa
de $5$ até $5 \times 6$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:mult:4}
Calcular: $2 \times 7$; \quad $5 \times 3$; \quad $10 \times 4$;
\quad $2 \times 9$; \quad $5 \times 6$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:mult:5}
Calcule os duplos: o dobro de $6$; o dobro de $8$; o dobro de $10$.
Depois as metades: metade de $12$; metade de $20$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:mult:6}
Calcule a tabela de $3$ por contagem saltitante: $3, 6, 9, ?, ?, ?$. O que é $3 \times 5$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:mult:7}
``Existem $4$ mesas com $5$ cadeiras cada. Quantas cadeiras?
Escreva a multiplicação e responda.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:mult:8}
Uma aranha tem $8$ pernas. Quantas pernas tem $3$ aranhas? (Pular contagem
por $8$: $8, 16, \dots $)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:mult:9}
Duas maneiras de organizar $12$ pontos em linhas iguais: desenhar $2$ linhas de $6$,
e $3$ linhas de $4$. Escreva a multiplicação de cada um.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g2:mult:10}
Os bolos vêm em caixas de $5$. Mia precisa $18$ bolos para uma festa. Como
quantas caixas ela deve comprar? (Pular contagem por $5$ até você passar $18$.)
\end{exercise}
