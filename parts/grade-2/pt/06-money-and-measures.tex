\chapter{Dinheiro e Medidas}\label{ch:g2:measure}

Réguas de medição, balanças de cozinha, jarras, moedas, relógios e
calendários: este capítulo é sobre os números da vida cotidiana ---
quanto tempo, quão pesado, quanto suporta, quanto custa e o que
é a hora.

\section{Comprimentos, massas, capacidades}

\begin{definition}[Unidades diárias]\label{def:g2:measure:units}
\begin{itemize}
  \item \emph{Comprimento}: centímetro (cm) para coisas pequenas, metro (m)
        para os grandes, com $1$ eu $= 100$ cm;
  \item \emph{massa}\index{massa}: grama (g) para coisas leves, quilograma (kg) para
        pesados, com $1$ kg $= 1000$ g;
  \item \emph{capacidade}\index{capacidade} (quantidade comporta um recipiente): litro (L).
\end{itemize}
Unidades menores e suas regras exatas entram \cref{ch:g3:measure}.
\end{definition}

\begin{example}[Qual unidade?]\label{ex:g2:measure:choose}
Um giz de cera: sobre $9$ cm. Uma porta: cerca de $2$ m. Uma maçã: cerca de $150$
g. Um grande saco de farinha: $1$ kg. Uma garrafa de leite: $1$ L. Adivinhe a unidade
antes de medir --- ele interrompe respostas bobas como "a porta está $2$
cm''.
\end{example}

\section{Dinheiro}

\begin{method}[Fazendo uma quantia]\label{met:g2:measure:money}
Com moedas e notas de $1$, $2$, $5$, $10$, $20$:
\begin{enumerate}
  \item comece com o maior que couber e depois preencha o restante;
  \item fazer $27$: um $20$, um $5$, e um $2$;
  \item muitas vezes há várias maneiras --- $27$ também é $10 + 10 + 5 +
        2$.
\end{enumerate}
\end{method}

\begin{example}[Mudar]\label{ex:g2:measure:change}
Um livro custa $14$; você paga com um $20$. A mudança é $20 - 14 =
6$, encontrado pela contagem crescente: $14 \to 20$ é $6$
(\cref{met:g2:subtraction:countup}).
\end{example}

\section{Tempo}

\begin{method}[Meia hora e quarto]\label{met:g2:measure:clock}
O ponteiro longo indica os minutos após a hora:
\begin{enumerate}
  \item direto para cima ($12$): na hora --- $3{:}00$;
  \item um quarto de volta para a direita ($3$): e um quarto --- $3{:}15$;
  \item direto para baixo ($6$): e meia --- $3{:}30$;
  \item três quartos de volta ($9$): um quarto para a próxima hora ---
        $3{:}45$, ``quatro para as quatro''.
\end{enumerate}
\end{method}

\begin{example}[O calendário]\label{ex:g2:measure:calendar}
Um ano tem $12$ meses, na ordem: janeiro, fevereiro, março, abril,
Maio, junho, julho, agosto, setembro, outubro, novembro, dezembro.
A maioria dos meses tem $30$ ou $31$ dias; Fevereiro tem $28$ (ou $29$).
Uma data nomeia o dia e o mês: ``o $5$ dia de março''.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g2:measure:1}
Escolha a unidade (cm, m, g, kg ou L): lápis; a sala de aula
comprimento; uma pena; uma mochila escolar; uma garrafa de água.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:2}
Converter: $2$ m em cm; $300$ cm em m; $1$ kg em g.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:3}
Faça estes valores com moedas e notas: $16$; \quad $23$; \quad
$35$. Dê uma maneira para cada um.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:4}
Calcule a mudança: pague $10$ por um $6$ brinquedo; pagar $20$ por um $17$
livro; pagar $50$ por um $32$ jogo.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:5}
Leia a hora: mão longa $6$, ponteiro abreviado entre $4$ e $5$;
mão longa $3$, abreviação logo após $8$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:6}
Desenhe um relógio marcando duas e meia. Para onde cada mão aponta?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:7}
Nomeie o mês: depois de maio; antes de janeiro; o mês do seu
aniversário.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:8}
Lea compra uma bebida para $3$ e um bolo para $4$. Ela paga com um $10$
conta. Quanto mudou?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g2:measure:9}
O que é mais pesado: um $1$ saco de kg de açúcar ou $800$ g de maçãs? Por como
muitos gramas?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g2:measure:10}
Fazer $40$ com exatamente quatro moedas ou notas (escolhidas entre $1$, $2$,
$5$, $10$, $20$). Encontre duas maneiras diferentes.
\end{exercise}
