\chapter{Organizing Data}\label{ch:g7:stats}

Before computing anything, data must be collected, counted and
displayed. This chapter is about turning a messy list of observations
into a clear table or picture: counts, frequencies, grouped data, bar
charts and pie charts. Summarizing data by a single number (mean,
median) is the subject of \cref{ch:g9:statproba}.

\section{Counts and frequencies}

\begin{definition}[Count and frequency]\label{def:g7:stats:frequency}
For each possible value of a survey, its \emph{count}\index{count} is
the number of times it appears; its \emph{frequency}\index{frequency}
is the count divided by the total number of observations:
\[
\text{frequency} = \frac{\text{count}}{\text{total}} .
\]
Frequencies can be written as \omterm{def:g6:fractions:def}{fractions}, decimals or percentages, and
they always add up to $1$ (i.e.\ $100\,\%$).
\end{definition}

\begin{example}\label{ex:g7:stats:frequency}
Favorite sport of the $25$ students of a class, from the raw list to
the table:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cccc|c}
sport & football & swimming & tennis & judo & total \\
\hline
count & $10$ & $6$ & $5$ & $4$ & $25$ \\
\omterm{def:g7:stats:frequency}{frequency} & $0.40$ & $0.24$ & $0.20$ & $0.16$ & $1$ \\
\end{tabular}
\end{center}
For football: $\frac{10}{25} = \frac{40}{100} = 0.40 = 40\,\%$. The
\omterm{def:g7:stats:frequency}{frequency} row must sum to $1$ --- a permanent, free check.
\end{example}

\begin{remark}[Why frequencies?]
Counts cannot be compared across groups of different sizes: $10$
football fans in a class of $25$ (that is $40\,\%$) show more
enthusiasm than $12$ in a school year of $120$ ($10\,\%$).
Frequencies put everything on the same \omterm{def:g7:prop:scale}{scale}.
\end{remark}

\section{Grouping data in classes}

\begin{example}[Classes]\label{ex:g7:stats:classes}
The heights (in cm) of $20$ students range from $148$ to $172$.
Listing every height separately would give a table with almost as many
columns as students; instead, group into \emph{classes}:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|ccc|c}
height (cm) & $\intco{145}{155}$ & $\intco{155}{165}$ &
$\intco{165}{175}$ & total \\
\hline
count & $5$ & $9$ & $6$ & $20$ \\
\omterm{def:g7:stats:frequency}{frequency} & $0.25$ & $0.45$ & $0.30$ & $1$ \\
\end{tabular}
\end{center}
Each height is counted in exactly one class ($155$ goes in
$\intco{155}{165}$, not in both --- that is why the intervals are
half-open). Grouping loses detail but gains readability.
\end{example}

\section{Charts}

\begin{method}[Choosing and drawing a chart]\label{met:g7:stats:charts}
\begin{enumerate}
  \item \emph{Bar chart}: one bar per value or class, height
        \omterm{def:g6:propdata:def}{proportional} to the count (or the \omterm{def:g7:stats:frequency}{frequency}). Good for
        comparing values.
  \item \emph{Pie chart}: one sector per value, angle \omterm{def:g6:propdata:def}{proportional} to
        the \omterm{def:g7:stats:frequency}{frequency} --- the full circle ($360^\circ$) represents the
        total, so
        \[
        \text{angle} = \text{frequency} \times 360^\circ .
        \]
        Good for showing shares of a whole.
  \item In both cases: title, labels, and a graduation or the
        percentages written on the chart.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:stats:pie}
Angles for the sports of \cref{ex:g7:stats:frequency}: football
$0.40 \times 360 = 144^\circ$; swimming $0.24 \times 360 =
86.4^\circ$; tennis $72^\circ$; judo $57.6^\circ$. Check:
$144 + 86.4 + 72 + 57.6 = 360$.
\end{example}

\begin{example}[Reading critically]\label{ex:g7:stats:critical}
A chart whose vertical axis starts at $9$ instead of $0$ makes a bar
of $10$ look ten times taller than a bar of $9.1$ --- visually
misleading, although the numbers are honest. First reflex in front of
any chart: check where the axis starts and what one graduation is
worth (\cref{met:g6:propdata:read}).
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g7:stats:1}
Here are the marks of $20$ students: $12$, $15$, $9$, $12$, $14$,
$9$, $12$, $15$, $11$, $12$, $9$, $14$, $12$, $11$, $15$, $9$, $12$,
$14$, $11$, $12$. Build the table of counts (one column per mark).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:stats:2}
Complete the table of \cref{exo:g7:stats:1} with a \omterm{def:g7:stats:frequency}{frequency} row
(as \omterm{def:g6:fractions:def}{fractions} of $20$, then as percentages). Check the \omterm{def:g1:addition:def}{sum}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:stats:3}
In a survey of $50$ families, $30$ have one car. What is the
\omterm{def:g7:stats:frequency}{frequency} of ``one car'', as a decimal and as a percentage? In another
survey, $45$ families out of $90$ have one car: which survey shows
the higher proportion?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:stats:4}
The masses (kg) of $15$ dogs: $8$, $23$, $31$, $12$, $9$, $27$, $35$,
$14$, $22$, $18$, $29$, $11$, $25$, $33$, $16$. Group them into the
classes $\intco{5}{15}$, $\intco{15}{25}$, $\intco{25}{35}$,
$\intco{35}{45}$ and give the counts.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:stats:5}
Draw the bar chart of the table:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|cccc}
pets & $0$ & $1$ & $2$ & $3$ \\
\hline
families & $8$ & $12$ & $6$ & $4$ \\
\end{tabular}
\end{center}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:stats:6}
For the table of \cref{exo:g7:stats:5}, compute the frequencies and
the pie-chart angles of each value (check: total $360^\circ$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:stats:7}
A pie chart about favorite seasons shows: summer $162^\circ$, spring
$90^\circ$, autumn $54^\circ$, winter $54^\circ$. What percentage
chose each season? If $60$ people answered, how many chose summer?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:stats:8}
In class A, $12$ students out of $30$ walk to school; in class B,
$14$ out of $40$. Which class has the higher proportion of walkers?
Justify with frequencies, not counts.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:stats:9}
A magazine prints a bar chart of monthly sales: $9\,800$, then
$10\,000$, then $10\,100$, with the vertical axis starting at
$9\,700$. Describe what the reader sees, and what an honest axis
starting at $0$ would show instead.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:stats:10}
A \omterm{def:g7:stats:frequency}{frequency} table has three values with frequencies $0.35$, $0.4$ and
$f$. Find $f$. If the total count is $80$, give the three counts.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:stats:11}
In a school, $55\,\%$ of the students are girls. Among the girls,
$40\,\%$ eat at the cafeteria; among the boys, $60\,\%$ do. Out of
$400$ students, how many eat at the cafeteria? (Compute the four
group sizes step by step.) What overall percentage is that?
\end{exercise}
