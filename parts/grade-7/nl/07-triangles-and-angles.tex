\chapter{Driehoeken en hoeken}\label{ch:g7:triangles}

Kun je een driehoek tekenen met zijden $10$, $3$ en $2$? Met hoeken
$80^\circ$, $70^\circ$ en $50^\circ$? Dit hoofdstuk beantwoordt beide
vragen met twee van de nuttigste feiten van de vlakke meetkunde: de
driehoeksongelijkheid en de hoekensom van $180^\circ$ --- plus de
hoekwoordenschat die bij snijdende en evenwijdige lijnen hoort.

\section{Hoeken rond snijdende lijnen}

\begin{definition}[Hoekparen]\label{def:g7:triangles:pairs}
\begin{itemize}
  \item Twee hoeken zijn \emph{complementair} wanneer hun maten
        optellen tot $90^\circ$, \emph{supplementair} wanneer ze
        optellen tot $180^\circ$.
  \item Wanneer twee lijnen snijden, heten de hoeken die elkaar over
        het snijpunt aankijken \emph{overstaande
        hoeken}\index{overstaande hoeken}.
  \item Wanneer een lijn twee andere lijnen snijdt, heten hoeken in
        dezelfde positie bij elke snijding \emph{overeenkomstige
        hoeken}; hoeken tussen de twee lijnen aan tegengestelde
        kanten van de snijdende lijn heten \emph{wisselhoeken}\index{wisselhoeken}.
\end{itemize}
\end{definition}

\begin{theorem}[Hoekgelijkheden]\label{thm:g7:triangles:equalities}
\begin{enumerate}
  \item Overstaande hoeken zijn gelijk.
  \item Als twee lijnen evenwijdig zijn, zijn overeenkomstige hoeken
        gelijk, en wisselhoeken gelijk. Omgekeerd dwingen gelijke
        overeenkomstige (of wissel)hoeken de lijnen evenwijdig te
        zijn.
\end{enumerate}
\end{theorem}

\begin{proof}[Bewijs van 1]
De twee hoeken $\widehat{1}$ en $\widehat{2}$ aan één kant van een
lijn vormen een gestrekte hoek: $\widehat 1 + \widehat 2 = 180^\circ$.
De overstaande hoek $\widehat 3$ van $\widehat 1$ voldoet ook aan
$\widehat 2 + \widehat 3 = 180^\circ$ (zelfde reden). Dus
$\widehat 1 = 180^\circ - \widehat 2 = \widehat 3$. Punt 2 is
toegegeven op dit niveau.
\end{proof}

\section{De som van de hoeken van een driehoek}

\begin{theorem}[Hoekensom]\label{thm:g7:triangles:sum}
In elke driehoek tellen de drie hoeken op tot $180^\circ$.
\end{theorem}

\begin{proof}
Laat $ABC$ een driehoek zijn. Teken de rechte door $A$ evenwijdig aan
$(BC)$. In het hoekpunt $A$ liggen drie hoeken langs deze evenwijdige
en vormen een gestrekte hoek, $180^\circ$: de middelste is
$\widehat{A}$, en de twee buitenste zijn wisselhoeken met
$\widehat B$ en $\widehat C$ (evenwijdige lijnen!), dus ze zijn gelijk
aan $\widehat B$ en $\widehat C$. Dus
$\widehat B + \widehat A + \widehat C = 180^\circ$.
\end{proof}

\begin{example}\label{ex:g7:triangles:sum}
Twee hoeken van een driehoek meten $67^\circ$ en $58^\circ$. De derde
meet
\[
180 - (67 + 58) = 180 - 125 = 55^\circ .
\]
Speciale gevallen om te onthouden: elke hoek van een gelijkzijdige
driehoek is $180 \div 3 = 60^\circ$; in een rechthoekige driehoek
zijn de twee scherpe hoeken complementair (ze tellen op tot
$180 - 90 = 90^\circ$); de basishoeken van een gelijkbenige driehoek
zijn gelijk, dus elk is $\frac{180 - \text{tophoek}}{2}$.
\end{example}

\begin{example}[Geen driehoek met twee rechte hoeken]\label{ex:g7:triangles:tworight}
Een driehoek met twee hoeken van $90^\circ$ zou al $180^\circ$
verbruiken, met $0^\circ$ over voor de derde --- onmogelijk. De
hoekensom verbiedt meteen veel driehoeken.
\end{example}

\section{De driehoeksongelijkheid}

\begin{theorem}[Driehoeksongelijkheid]\label{thm:g7:triangles:inequality}
In elke driehoek is elke zijde korter dan de som van de twee andere:
voor een driehoek met zijden $a$, $b$, $c$,
\[
a < b + c .
\]
Omgekeerd kunnen drie lengten waarvan de \emph{grootste} kleiner is
dan de som van de twee andere altijd een driehoek vormen. Als de
grootste \emph{gelijk} is aan de som van de andere, is de ``driehoek''
plat: de drie hoekpunten zijn gealigneerd.
\end{theorem}
\admitted

\begin{example}\label{ex:g7:triangles:inequality}
Kunnen de zijden $10$, $3$ en $2$ zijn? Test de grootste: is
$10 < 3 + 2 = 5$? Nee --- zo'n driehoek bestaat niet. De passer
toont waarom: bogen met stralen $3$ en $2$ getekend vanuit de twee
uiteinden van een lijnstuk van lengte $10$ ontmoeten elkaar nooit.

Zijden $7$, $5$ en $4$? Grootste: $7 < 5 + 4 = 9$: ja, de driehoek
bestaat (de grootste zijde controleren volstaat).
\end{example}

\begin{method}[Bestaan en constructie van een driehoek]\label{met:g7:triangles:construct}
Gegeven drie lengten:
\begin{enumerate}
  \item vergelijk de grootste met de som van de twee andere; als die
        niet strikt kleiner is, stop: geen driehoek;
  \item construeer anders zoals in \cref{met:g6:shapes:construct}
        (basislijnstuk, twee passerbogen);
  \item na elke constructie uit hoeken, controleer mentaal dat de
        gegeven hoeken minder dan $180^\circ$ sommeren --- met de
        derde hoek exact $180^\circ$ makend.
\end{enumerate}
\end{method}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g7:triangles:1}
Geef het complement (tot $90^\circ$) en het supplement (tot
$180^\circ$) van: $30^\circ$; \ $45^\circ$; \ $72^\circ$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:triangles:2}
Twee lijnen snijden; één van de vier hoeken meet $35^\circ$. Geef de
maten van de drie andere, met een reden voor elk.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:triangles:3}
Bereken de derde hoek van een driehoek waarvan de eerste twee hoeken
meten: (a) $40^\circ$ en $60^\circ$; \
(b) $90^\circ$ en $28^\circ$; \
(c) $75^\circ$ en $75^\circ$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:triangles:4}
Een gelijkbenige driehoek heeft tophoek $40^\circ$. Bereken zijn
twee basishoeken. Een andere gelijkbenige driehoek heeft een
basishoek van $70^\circ$: bereken zijn tophoek.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:triangles:5}
Welke van deze drietallen kunnen de zijden van een driehoek zijn?
Motiveer met de driehoeksongelijkheid (grootste zijde!):
\[
(6,\ 8,\ 12); \qquad
(5,\ 5,\ 11); \qquad
(4,\ 9,\ 13); \qquad
(7,\ 7,\ 7).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:triangles:6}
Een driehoek heeft hoeken $x$, $2x$ en $3x$. Vind $x$ en de drie
hoeken. Wat voor driehoek is het?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:triangles:7}
Construeer een driehoek $ABC$ met $BC = 6$ cm,
$\widehat{ABC} = 50^\circ$ en $\widehat{ACB} = 60^\circ$
(gradenboog in $B$ en in $C$). Voorspel vóór de constructie de maat
van $\widehat{BAC}$, en controleer op je figuur.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:triangles:8}
Twee evenwijdige lijnen worden gesneden door een derde lijn, die een
hoek van $54^\circ$ maakt bij de eerste snijding (tussen de
snijdende lijn en één evenwijdige). Teken de situatie en geef de
maten van alle acht gevormde hoeken.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:triangles:9}
Twee zijden van een driehoek meten $8$ cm en $3$ cm. Tussen welke
twee waarden moet de derde zijde liggen? Geef een gehele lengte die
werkt en één die niet werkt.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:triangles:10}
In een driehoek $ABC$ is $\widehat A = 2\widehat B$ en
$\widehat C = 90^\circ$. Bereken $\widehat A$ en $\widehat B$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:triangles:11}
De drie hoeken van elke vierhoek $ABCD$ kunnen bestudeerd worden door
hem langs een diagonaal in twee driehoeken te snijden. Gebruik dit om
te bewijzen dat de vier hoeken van elke vierhoek optellen tot
$360^\circ$, en leid de hoekensom van een vijfhoek af (vijf zijden).
\end{exercise}
