\chapter{Rekenvolgorde}\label{ch:g7:priorities}

Wat is $3 + 4 \times 5$? Als je van links naar rechts rekent krijg je
$35$; als je eerst vermenigvuldigt krijg je $23$. Wiskunde heeft één
enkel antwoord nodig, dus er zijn verkeersregels voor berekeningen ---
de rekenvolgorde. Dit hoofdstuk stelt ze vast en introduceert een
eerste algebraïsche identiteit, de distributiviteit.

\section{De voorrangsregels}

\begin{theorem}[Voorrangsregels]\label{thm:g7:priorities:rules}
In een berekening zonder haakjes:
\begin{enumerate}
  \item vermenigvuldigingen en delingen worden uitgevoerd \emph{vóór}
        optellingen en aftrekkingen;
  \item bewerkingen van hetzelfde niveau worden van links naar rechts
        uitgevoerd.
\end{enumerate}
Haakjes overruilen alles: wat tussen haakjes staat wordt eerst
berekend, te beginnen bij de binnenste.
\end{theorem}

\begin{example}\label{ex:g7:priorities:basic}
Eén stap per regel, (mentaal) onderstrepen wat berekend wordt:
\begin{align*}
3 + 4 \times 5 &= 3 + 20 = 23
&&\text{(eerst vermenigvuldiging)}\\
20 - 8 + 2 &= 12 + 2 = 14
&&\text{(zelfde niveau: links naar rechts)}\\
18 \div 3 \times 2 &= 6 \times 2 = 12
&&\text{(zelfde niveau: links naar rechts)}\\
(3 + 4) \times 5 &= 7 \times 5 = 35
&&\text{(eerst haakjes).}
\end{align*}
Let op de valkuilen: $20 - 8 + 2$ is \emph{niet} $20 - 10$, en
$18 \div 3 \times 2$ is \emph{niet} $18 \div 6$.
\end{example}

\begin{example}[Geneste haakjes]\label{ex:g7:priorities:nested}
\begin{align*}
50 - 2 \times \bigl(3 + (10 - 4)\bigr)
&= 50 - 2 \times (3 + 6) && \text{(binnenste haakjes)}\\
&= 50 - 2 \times 9 && \text{(resterende haakjes)}\\
&= 50 - 18 && \text{(vermenigvuldiging)}\\
&= 32 .
\end{align*}
\end{example}

\begin{remark}[Breukstrepen verbergen haakjes]
Een breukstreep groepeert \omterm{def:g4:fractions:def}{teller} en \omterm{def:g4:fractions:def}{noemer}, alsof elk onzichtbare
haakjes had:
\[
\frac{12 + 8}{4} = \frac{20}{4} = 5,
\qquad\text{niet}\quad 12 + \frac{8}{4} = 14 .
\]
\end{remark}

\begin{method}[Veilig rekenen]\label{met:g7:priorities:safe}
\begin{enumerate}
  \item Spot de bewerking die eerst moet (binnenste haakjes, daarna
        $\times$ en $\div$, daarna $+$ en $-$);
  \item bereken \emph{alleen die ene}, en herschrijf de hele
        uitdrukking met het resultaat op die plaats;
  \item herhaal tot er één getal overblijft --- één bewerking per
        regel, geen shortcuts.
\end{enumerate}
\end{method}

\section{Distributiviteit}

\begin{theorem}[Distributiviteit]\label{thm:g7:priorities:distributivity}
Voor alle getallen $k$, $a$, $b$:
\[
k \times (a + b) = k \times a + k \times b,
\qquad
k \times (a - b) = k \times a - k \times b .
\]
\end{theorem}

\begin{proof}[Bewijs met een plaatje]
Tel de \omterm{def:g6:measure:area}{oppervlakte} van een rechthoek met breedte $k$ gesplitst in twee
kleinere rechthoeken met lengten $a$ en $b$: als geheel is die
$k \times (a+b)$; in twee delen is die $k \times a + k \times b$.
Zelfde rechthoek, zelfde \omterm{def:g6:measure:area}{oppervlakte}.
\end{proof}

\end{omfigure}

\begin{example}[Beide richtingen]\label{ex:g7:priorities:distributivity}
\emph{Uitwerken} (van links naar rechts) maakt mentaal rekenen makkelijk:
\[
7 \times 103 = 7 \times (100 + 3) = 700 + 21 = 721,
\qquad
6 \times 98 = 6 \times (100 - 2) = 600 - 12 = 588 .
\]
\emph{Ontbinden} (van rechts naar links) verzamelt een gemeenschappelijke
factor:
\[
23 \times 12 + 23 \times 8 = 23 \times (12 + 8) = 23 \times 20 = 460 .
\]
\end{example}

\begin{example}[Woorden omzetten in een berekening]\label{ex:g7:priorities:words}
``De \omterm{def:g1:addition:def}{som} van $8$ en het \omterm{def:g2:mult:def}{product} van $5$ met $3$'' is $8 + 5 \times 3 =
23$. ``Het \omterm{def:g2:mult:def}{product} van $5$ met de \omterm{def:g1:addition:def}{som} van $8$ en $3$'' is
$5 \times (8 + 3) = 55$: de woorden ``het \omterm{def:g2:mult:def}{product} van \dots\ met de
\omterm{def:g1:addition:def}{som} van\dots'' zetten haakjes om de \omterm{def:g1:addition:def}{som}.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g7:priorities:1}
Bereken, één bewerking per regel:
\[
5 + 3 \times 6, \qquad
(5 + 3) \times 6, \qquad
30 - 12 \div 4, \qquad
(30 - 12) \div 4 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:2}
Bereken:
\[
24 - 6 + 2, \qquad
24 \div 6 \times 2, \qquad
24 - 6 \times 2, \qquad
24 \div (6 \times 2).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:3}
Bereken stap voor stap:
\[
7 + 3 \times (10 - 6), \qquad
40 - (2 + 3) \times 4, \qquad
60 \div \bigl(2 \times (7 - 4)\bigr).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:4}
Bereken de \omterm{def:g6:fractions:def}{breuken} (onzichtbare haakjes!):
\[
\frac{18 + 6}{8}, \qquad
\frac{30}{7 - 2}, \qquad
\frac{5 \times 8}{4 + 6} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:5}
Gebruik distributiviteit om uit het hoofd te rekenen:
\[
8 \times 102, \qquad
25 \times 99, \qquad
14 \times 12 + 14 \times 8, \qquad
37 \times 45 - 37 \times 35 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:6}
Schrijf de berekening in één regel, met haakjes waar nodig, en
bereken dan: ``het \omterm{ex:g1:subtraction:difference}{verschil} van $50$ en het \omterm{def:g2:mult:def}{product} van $6$ met $7$'';
``het \omterm{def:g2:mult:def}{product} van de \omterm{def:g1:addition:def}{som} van $4$ en $5$ met het \omterm{ex:g1:subtraction:difference}{verschil} van $10$ en
$8$''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:7}
Amir kocht $3$ schriften van $2.50$ elk en $2$ pennen van $1.20$ elk.
Schrijf zijn totaal in één uitdrukking, en bereken die.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:priorities:8}
Plaats haakjes in $4 + 2 \times 5 - 1$ om als resultaat te krijgen:
(a) $13$; (b) $29$; (c) $24$. (Voor één van de drie zijn helemaal geen
haakjes nodig --- welke?)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:priorities:9}
Zijn $12 + 4 \times k$ en $16 \times k$ hetzelfde voor elk getal $k$?
Test met $k = 1$ en $k = 2$, en concludeer. Welke voorrangsfout zou
van de ene naar de andere leiden?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:priorities:10}
Een bioscoopticket kost $9$ euro, maar een groep van $n$ personen
betaalt een vast bedrag van $12$ euro plus $6$ euro per persoon.
\begin{enumerate}
  \item Schrijf een uitdrukking voor de groepsprijs, en bereken die
        voor $n = 5$.
  \item Vanaf hoeveel personen is het groepstarief goedkoper dan $n$
        individuele tickets? (Probeer een paar waarden van $n$.)
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:priorities:11}
Gebruik elk van de cijfers $1$, $2$, $3$, $4$ precies één keer, de
vier bewerkingen en haakjes naar believen, en schrijf uitdrukkingen
gelijk aan $24$, aan $10$, en aan $1$.
\end{exercise}
