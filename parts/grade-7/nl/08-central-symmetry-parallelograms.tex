\chapter{Centrale spiegeling en parallellogrammen}\label{ch:g7:central}

Draai een figuur een halve slag om een punt: dat is
\emph{\omterm{def:g7:central:def}{centrale spiegeling}}, de tweede transformatie van het boek na de
spiegelingen van \cref{ch:g6:symmetry}. Zijn sterfiguur is het
\omterm{def:g7:central:parallelogram}{parallellogram} --- de \omterm{def:g4:geometry:polygon}{vierhoek} die symmetrisch is ten opzichte van het
snijpunt van zijn eigen diagonalen.

\section{Centrale spiegeling}

\begin{definition}[Spiegelbeeld ten opzichte van een punt]\label{def:g7:central:def}
Het \emph{spiegelbeeld}\index{centrale spiegeling} van een punt $M$ ten
opzichte van een punt $O$ is het punt $M'$ zo dat $O$ het
\emph{midden} is van $[MM']$. De spiegelbeeldfiguur is wat het
origineel wordt na een halve slag ($180^\circ$) om $O$.
\end{definition}

\begin{method}[Het spiegelbeeld van een punt construeren]\label{met:g7:central:construct}
Om het \omterm{def:g7:central:def}{spiegelbeeld} $M'$ van $M$ ten opzichte van $O$ te construeren:
\begin{enumerate}
  \item teken de \omterm{def:g6:lines:objects}{rechte} $(MO)$ en verleng die voorbij $O$;
  \item meet $MO$ (liniaal of passer);
  \item plaats $M'$ op de verlenging met $OM' = OM$.
\end{enumerate}
Op ruitjespapier: tel de horizontale en verticale stappen van $M$
naar $O$, en herhaal \emph{dezelfde} stappen vanuit $O$ om $M'$ te
bereiken.
\end{method}

\begin{example}[Op een rooster]\label{ex:g7:central:grid}
Als $M$ $3$ hokjes links en $1$ hokje omhoog van $O$ ligt, ligt zijn
\omterm{def:g7:central:def}{spiegelbeeld} $M'$ $3$ hokjes \emph{rechts} en $1$ hokje
\emph{omlaag} van $O$: \omterm{def:g7:central:def}{centrale spiegeling} keert beide richtingen
tegelijk om (een asspiegeling keert er maar één om).
\end{example}

\begin{proposition}[Wat een halve slag behoudt]\label{prop:g7:central:preserved}
\omterm{def:g7:central:def}{Centrale spiegeling} behoudt lengten, hoeken, omtrekken en
\omterm{def:g6:measure:area}{oppervlakten}; het beeld van een \omterm{def:g6:lines:objects}{rechte} is een \emph{evenwijdige}
\omterm{def:g6:lines:objects}{rechte}, het beeld van een \omterm{def:g6:lines:objects}{lijnstuk} is een \omterm{def:g6:lines:perp}{evenwijdig} \omterm{def:g6:lines:objects}{lijnstuk} van
dezelfde lengte, het beeld van een cirkel is een cirkel van dezelfde
\omterm{def:g3:shapes:shapes}{straal} (met middelpunt het \omterm{def:g7:central:def}{spiegelbeeld} van het middelpunt).
\end{proposition}
\admitted

\begin{remark}
Vergelijk met asspiegelingen (\cref{prop:g6:symmetry:preserved}):
beide behouden lengten en hoeken; maar de asspiegeling klapt figuren
om (een ``b'' wordt een ``d''), terwijl de halve slag ze leesbaar
houdt --- ondersteboven (een ``b'' wordt een ``q''). En anders dan
een asspiegeling stuurt een halve slag elke \omterm{def:g6:lines:objects}{rechte} naar een evenwijdige
\omterm{def:g6:lines:objects}{rechte}.
\end{remark}

\begin{definition}[Symmetriecentrum]\label{def:g7:central:center}
Een punt $O$ is een \emph{symmetriecentrum}\index{symmetriecentrum}
van een figuur wanneer de halve slag om $O$ de figuur exact op
zichzelf stuurt. Bijvoorbeeld hebben de letters S, N, Z een
symmetriecentrum; een cirkel heeft er één (zijn middelpunt); een
driehoek heeft er nooit één.
\end{definition}

\section{Parallellogrammen}

\begin{definition}[Parallellogram]\label{def:g7:central:parallelogram}
Een \emph{parallellogram}\index{parallellogram} is een \omterm{def:g4:geometry:polygon}{vierhoek}
waarvan de diagonalen snijden in hun gemeenschappelijk midden. Dat
snijpunt is dan een \omterm{def:g7:central:center}{symmetriecentrum} van de figuur: elk \omterm{def:g5:solids:def}{hoekpunt} is
het halfslagbeeld van het overstaande \omterm{def:g5:solids:def}{hoekpunt}.
\end{definition}

\begin{theorem}[Eigenschappen van een parallellogram]\label{thm:g7:central:properties}
In een \omterm{def:g7:central:parallelogram}{parallellogram} $ABCD$ met centrum $O$:
\begin{enumerate}
  \item overstaande zijden zijn \omterm{def:g6:lines:perp}{evenwijdig}: $(AB) \parallel (CD)$ en
        $(BC) \parallel (AD)$;
  \item overstaande zijden hebben dezelfde lengte: $AB = CD$ en
        $BC = AD$;
  \item \omterm{def:g7:triangles:pairs}{overstaande hoeken} zijn gelijk: $\widehat A = \widehat C$ en
        $\widehat B = \widehat D$.
\end{enumerate}
\end{theorem}

\begin{proof}
De halve slag om $O$ stuurt $A$ naar $C$ en $B$ naar $D$ (definitie:
$O$ is het midden van beide diagonalen). Dus stuurt hij het \omterm{def:g6:lines:objects}{lijnstuk}
$[AB]$ naar het \omterm{def:g6:lines:objects}{lijnstuk} $[CD]$; door
\cref{prop:g7:central:preserved} is het beeld \omterm{def:g6:lines:perp}{evenwijdig} aan $[AB]$
en van dezelfde lengte: punten 1 en 2. Hij stuurt ook de hoek in $A$
naar de hoek in $C$, en hoeken blijven behouden: punt 3.
\end{proof}

\begin{theorem}[Een parallellogram herkennen]\label{thm:g7:central:recognize}
Een \omterm{def:g4:geometry:polygon}{vierhoek} (zonder snijdende zijden) is een \omterm{def:g7:central:parallelogram}{parallellogram} zodra
\emph{één} van de volgende geldt:
\begin{enumerate}
  \item zijn diagonalen hebben hetzelfde midden (de definitie);
  \item zijn overstaande zijden zijn twee aan twee \omterm{def:g6:lines:perp}{evenwijdig};
  \item twee overstaande zijden zijn \omterm{def:g6:lines:perp}{evenwijdig} \emph{en} van dezelfde
        lengte.
\end{enumerate}
\end{theorem}
\admitted

\begin{example}\label{ex:g7:central:recognize}
$ABCD$ heeft $(AB) \parallel (CD)$ en $AB = CD = 5$ cm. Criterium 3
past: $ABCD$ is een \omterm{def:g7:central:parallelogram}{parallellogram} --- dus zonder verder te meten
weten we ook $AD = BC$, $(AD) \parallel (BC)$, en dat zijn diagonalen
elkaar in hun midden snijden.
\end{example}

\begin{remark}[Speciale parallellogrammen]
De rechthoek, \omterm{def:g6:shapes:quadrilaterals}{ruit} en vierkant van \cref{ch:g6:shapes} zijn precies
de parallellogrammen met een extra eigenschap: gelijke diagonalen
(rechthoek), loodrechte diagonalen (\omterm{def:g6:shapes:quadrilaterals}{ruit}), beide (vierkant). Het
hoekargument beloofd in \cref{ex:g6:shapes:recognize} werkt nu: in een
\omterm{def:g7:central:parallelogram}{parallellogram} zijn opeenvolgende hoeken supplementair
(wissel/overeenkomstige hoeken met de \omterm{def:g6:lines:perp}{evenwijdigen},
\cref{thm:g7:triangles:equalities}), dus één \omterm{def:g3:shapes:rightangle}{rechte hoek} dwingt alle
vier.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g7:central:1}
Plaats op ruitjespapier $O$, daarna $M$ vier hokjes rechts van $O$ en
één hokje omhoog. Construeer het \omterm{def:g7:central:def}{spiegelbeeld} $M'$ van $M$ ten
opzichte van $O$, en het \omterm{def:g7:central:def}{spiegelbeeld} van $M$ ten opzichte van de
\emph{verticale roosterlijn} door $O$. Vergelijk de twee beelden.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:central:2}
Teken een driehoek $ABC$ en een punt $O$ erbuiten. Construeer zijn
beeld $A'B'C'$ door de halve slag om $O$. Wat kun je zeggen over de
lengten $A'B'$ en $AB$? Over de \omterm{def:g6:lines:objects}{rechten} $(A'B')$ en $(AB)$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:central:3}
Welke cijfers, geschreven als op een rekenmachinedisplay ($0$ tot
$9$), hebben een \omterm{def:g7:central:center}{symmetriecentrum}? Welke hoofdletters onder H, A, N,
S, T, Z, O?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:central:4}
$KLMN$ is een \omterm{def:g7:central:parallelogram}{parallellogram} met centrum $O$ met $KL = 7$ cm,
$LM = 4$ cm en $\widehat K = 115^\circ$. Geef, met redenen: $MN$,
$NK$, $\widehat M$, en het midden van $[LN]$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:central:5}
Construeer een \omterm{def:g7:central:parallelogram}{parallellogram} $ABCD$ uit zijn diagonalen: $AC = 8$ cm
en $BD = 5$ cm, snijdend in hun middens met een hoek van $60^\circ$
ertussen. (Teken eerst de diagonalen!)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:central:6}
In een \omterm{def:g7:central:parallelogram}{parallellogram} meet één hoek $115^\circ$. Gebruik de opmerking
over supplementaire opeenvolgende hoeken en geef de drie andere
hoeken.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:central:7}
Teken $A(1, 1)$, $B(4, 2)$ en het punt $O(2.5, 2.5)$. Bereken de
coördinaten van de beelden $A'$ en $B'$ door de halve slag om $O$
(gebruik het roostertellen van \cref{ex:g7:central:grid}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:central:8}
$ABCD$ is een \omterm{def:g4:geometry:polygon}{vierhoek} waarin $AB = CD$. Is dat genoeg om er een
\omterm{def:g7:central:parallelogram}{parallellogram} van te maken? Zo niet, schets een tegenvoorbeeld (een
\omterm{def:g6:shapes:triangles}{gelijkbenig} trapezium helpt), en zeg wat aan de hypothese moet
worden toegevoegd.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:central:9}
Laat $ABC$ een driehoek zijn en $O$ het midden van $[BC]$. Construeer
het \omterm{def:g7:central:def}{spiegelbeeld} $A'$ van $A$ ten opzichte van $O$. Toon dat $ABA'C$
een \omterm{def:g7:central:parallelogram}{parallellogram} is. (Welk criterium van
\cref{thm:g7:central:recognize} is hier gratis?)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:central:10}
Waar of onwaar, met een reden: ``een \omterm{def:g4:geometry:polygon}{vierhoek} waarvan de \omterm{def:g7:triangles:pairs}{overstaande
hoeken} twee aan twee gelijk zijn is een \omterm{def:g7:central:parallelogram}{parallellogram}''; ``een
\omterm{def:g7:central:parallelogram}{parallellogram} met loodrechte diagonalen is een \omterm{def:g6:shapes:quadrilaterals}{ruit}''; ``een
\omterm{def:g7:central:parallelogram}{parallellogram} met één \omterm{def:g3:shapes:rightangle}{rechte hoek} is een rechthoek''.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:central:11}
Laat $ABCD$ een willekeurige \omterm{def:g4:geometry:polygon}{vierhoek} zijn en $I$, $J$, $K$, $L$ de
middens van zijn zijden $[AB]$, $[BC]$, $[CD]$, $[DA]$. Teken
verschillende voorbeelden. Wat lijkt $IJKL$ altijd te zijn? (Het
bewijs, met de middenparallellenstelling, komt in
\cref{ch:g8:midpoints} --- en opnieuw met vectoren, in het High
School-volume.)
\end{exercise}
