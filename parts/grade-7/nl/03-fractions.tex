\chapter{Breuken: vergelijken en optellen}\label{ch:g7:fractions}

Jaar~6 introduceerde \omterm{def:g6:fractions:def}{breuken} als delen en als exacte \omterm{def:g3:division:remainder}{quotiënten}
(\cref{ch:g6:fractions}). Dit hoofdstuk leert ermee te \emph{rekenen}:
gelijke \omterm{def:g6:fractions:def}{breuken} herkennen, vergelijken, optellen en aftrekken --- voorlopig
met vriendelijke \omterm{def:g4:fractions:def}{noemers}; het algemene geval komt in
\cref{ch:g9:fractions}.

\section{Gelijke breuken}

\begin{theorem}[Gelijke breuken]\label{thm:g7:fractions:equal}
Een \omterm{def:g6:fractions:def}{breuk} verandert niet wanneer \omterm{def:g4:fractions:def}{teller} en \omterm{def:g4:fractions:def}{noemer} vermenigvuldigd, of
gedeeld, worden met hetzelfde getal ongelijk aan \omterm{def:g1:counting:zero}{nul}:
\[
\frac{a}{b} = \frac{a \times k}{b \times k} .
\]
\end{theorem}
\admitted

\begin{example}\label{ex:g7:fractions:equal}
$\dfrac{3}{4} = \dfrac{3 \times 5}{4 \times 5} = \dfrac{15}{20}$, en
in de andere richting $\dfrac{42}{30} = \dfrac{42 \div 6}{30 \div 6}
= \dfrac{7}{5}$ (vereenvoudigd met $6$).

Om te testen of $\dfrac{8}{12}$ en $\dfrac{10}{15}$ gelijk zijn,
vereenvoudig beide: $\dfrac{8}{12} = \dfrac{2}{3}$ en
$\dfrac{10}{15} = \dfrac{2}{3}$ --- ja, gelijk.
\end{example}

\section{Breuken vergelijken}

\begin{method}[Twee breuken vergelijken]\label{met:g7:fractions:compare}
\begin{enumerate}
  \item \emph{Zelfde \omterm{def:g4:fractions:def}{noemer}}: vergelijk de \omterm{def:g4:fractions:def}{tellers}:
        $\frac{5}{9} < \frac{7}{9}$.
  \item \emph{Eén \omterm{def:g4:fractions:def}{noemer} is een \omterm{def:g5:division:multiple}{veelvoud} van de andere}: herschrijf de
        grovere \omterm{def:g6:fractions:def}{breuk}, en vergelijk daarna:
        $\frac{5}{6}$ vs $\frac{7}{12}$: $\frac56 = \frac{10}{12} >
        \frac{7}{12}$.
  \item \emph{Vergelijk met $1$ of $\frac12$} wanneer mogelijk:
        $\frac{9}{8} > 1 > \frac{7}{9}$ beslist
        $\frac98$ vs $\frac79$ meteen.
\end{enumerate}
\end{method}

\section{Optellen en aftrekken}

\begin{theorem}[Zelfde noemer]\label{thm:g7:fractions:add}
\omterm{def:g6:fractions:def}{Breuken} met dezelfde \omterm{def:g4:fractions:def}{noemer} worden opgeteld (of afgetrokken) door de
\omterm{def:g4:fractions:def}{tellers} op te tellen (of af te trekken):
\[
\frac{a}{d} + \frac{b}{d} = \frac{a + b}{d},
\qquad
\frac{a}{d} - \frac{b}{d} = \frac{a - b}{d} .
\]
\end{theorem}

\begin{proof}[Waarom]
$a$ delen van grootte $\frac1d$ plus $b$ delen van dezelfde grootte
maken $a + b$ delen van die grootte.
\end{proof}

\begin{method}[Verschillende noemers]\label{met:g7:fractions:add}
Wanneer één \omterm{def:g4:fractions:def}{noemer} een \omterm{def:g5:division:multiple}{veelvoud} is van de andere:
\begin{enumerate}
  \item herschrijf de \omterm{def:g6:fractions:def}{breuk} met de kleinere \omterm{def:g4:fractions:def}{noemer} zodat beide de
        grotere hebben (\cref{thm:g7:fractions:equal});
  \item tel of trek de \omterm{def:g4:fractions:def}{tellers} op/af;
  \item vereenvoudig het resultaat indien mogelijk.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:fractions:add}
Bereken $\dfrac{3}{4} + \dfrac{5}{8}$, stap voor stap:
\begin{enumerate}
  \item $8$ is een \omterm{def:g5:division:multiple}{veelvoud} van $4$: herschrijf
        $\dfrac34 = \dfrac{3 \times 2}{4 \times 2} = \dfrac{6}{8}$;
  \item tel op: $\dfrac{6}{8} + \dfrac{5}{8} = \dfrac{11}{8}$;
  \item niets te vereenvoudigen: het resultaat is $\dfrac{11}{8}$,
        d.w.z.\ $1 + \dfrac38$.
\end{enumerate}
Nog een: $2 - \dfrac{4}{5} = \dfrac{10}{5} - \dfrac{4}{5} =
\dfrac{6}{5}$ (schrijf eerst het gehele getal over de \omterm{def:g4:fractions:def}{noemer} $5$).
\end{example}

\begin{example}[Een klassieke valkuil]\label{ex:g7:fractions:trap}
$\dfrac12 + \dfrac13$ is \emph{niet} $\dfrac{2}{5}$! \omterm{def:g4:fractions:def}{Tellers} en
\omterm{def:g4:fractions:def}{noemers} apart optellen is fout --- het antwoord $\frac25$ zou kleiner
zijn dan $\frac12$, absurd wanneer je iets positiefs bij $\frac12$
\emph{optelt}. (De juiste \omterm{def:g1:addition:def}{som}, $\frac56$, heeft een gemeenschappelijke
\omterm{def:g4:fractions:def}{noemer} $6$ nodig; de algemene methode staat in
\cref{ch:g9:fractions}.)
\end{example}

\section{Een breuk met een getal vermenigvuldigen}

\begin{proposition}[Breuk maal een getal]\label{prop:g7:fractions:mult}
Voor een getal $k$:
\[
k \times \frac{a}{b} = \frac{k \times a}{b} .
\]
$\frac ab$ \emph{van} een hoeveelheid nemen betekent vermenigvuldigen
met $\frac ab$.
\end{proposition}
\admitted

\begin{example}\label{ex:g7:fractions:mult}
$6 \times \dfrac{2}{3} = \dfrac{12}{3} = 4$: zes keer twee derden is
vier helen. En ``$\frac34$ van $60$ euro'' is
$\frac34 \times 60 = \frac{180}{4} = 45$ euro --- hetzelfde antwoord
als delen door $4$ en daarna vermenigvuldigen met $3$
(\cref{met:g6:fractions:ofquantity}).
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g7:fractions:1}
Vul in: $\dfrac{2}{5} = \dfrac{?}{20}$; \quad
$\dfrac{18}{24} = \dfrac{3}{?}$; \quad
$\dfrac{7}{3} = \dfrac{28}{?}$; \quad
$\dfrac{?}{9} = \dfrac{20}{36}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:2}
Vereenvoudig zoveel mogelijk:
$\dfrac{12}{16}$; \quad $\dfrac{30}{45}$; \quad $\dfrac{27}{36}$;
\quad $\dfrac{48}{12}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:3}
Zijn $\dfrac{15}{35}$ en $\dfrac{21}{49}$ gelijk? En $\dfrac{16}{28}$
en $\dfrac{20}{36}$? Motiveer door te vereenvoudigen.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:4}
Vergelijk (schrijf de redenering, niet alleen het antwoord):
\[
\frac{7}{11} \text{ en } \frac{9}{11}; \qquad
\frac{5}{6} \text{ en } \frac{11}{18}; \qquad
\frac{13}{12} \text{ en } \frac{19}{20}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:5}
Bereken en vereenvoudig indien mogelijk:
\[
\frac{5}{9} + \frac{2}{9}, \qquad
\frac{11}{7} - \frac{4}{7}, \qquad
\frac{5}{8} + \frac{7}{8} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:6}
Bereken met \cref{met:g7:fractions:add}:
\[
\frac{2}{3} + \frac{5}{6}, \qquad
\frac{7}{10} - \frac{2}{5}, \qquad
\frac{3}{4} + \frac{5}{12} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:7}
Bereken:
\[
3 - \frac{5}{4}, \qquad
1 + \frac{3}{8}, \qquad
2 - \frac{7}{6} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:8}
Bereken:
\[
5 \times \frac{3}{10}, \qquad
\frac{7}{8} \times 4, \qquad
\frac{2}{3} \text{ van } 45 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:fractions:9}
Marc at $\frac14$ van een pizza en Julie $\frac38$ van dezelfde pizza.
Welk deel van de pizza aten ze samen? Welk deel blijft over?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:fractions:10}
Een fles bevat $\frac34$ L sap. Lea schenkt twee keer $\frac16$ L uit.
Hoeveel sap blijft er over? (Gemeenschappelijke \omterm{def:g4:fractions:def}{noemer}: $12$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:fractions:11}
Leg uit, met het argument van \cref{ex:g7:fractions:trap}, waarom
$\frac35 + \frac12$ niet $\frac{4}{7}$ kan zijn, zonder de juiste \omterm{def:g1:addition:def}{som}
te berekenen.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:fractions:12}
Bereken de \omterm{def:g1:addition:def}{som}
\[
\frac12 + \frac14 + \frac18 + \frac{1}{16}
\]
stap voor stap (van links naar rechts). Observeer het patroon van de
deelsommen: welk getal naderen ze wanneer steeds meer termen
$\frac{1}{32}, \frac{1}{64}, \dots$ worden toegevoegd?
\end{exercise}
