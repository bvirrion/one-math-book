\chapter{Negatieve getallen}\label{ch:g7:negatives}

Temperaturen onder \omterm{def:g1:counting:zero}{nul}, verdiepingen onder de grond, schulden: veel
grootheden gaan onder $0$. Negatieve getallen breiden de getallenlijn
uit naar links van \omterm{def:g1:counting:zero}{nul}. Dit hoofdstuk leert ze vergelijken en optellen
en aftrekken --- hun vermenigvuldiging wacht op
\cref{ch:g8:negprod}.

\section{Relatieve getallen}

\begin{definition}[Relatieve getallen]\label{def:g7:negatives:def}
Een \emph{relatief getal}\index{relatief getal} bestaat uit een teken
($+$ of $-$) en een \emph{afstand tot \omterm{def:g1:counting:zero}{nul}}: $+3$ en $-3$ liggen beide
op afstand $3$ van $0$, aan tegenovergestelde kanten. Positieve
getallen worden meestal zonder teken geschreven: $+3 = 3$. De getallen
$+3$ en $-3$ zijn \emph{tegengestelden}\index{tegengestelde} van
elkaar.
\end{definition}

  \foreach \x in {-5,...,5} \draw (\x,-0.08) -- (\x,0.08);
  \foreach \x in {-5,...,5} \node[below, font=\small] at (\x,-0.1) {$\x$};
  \fill[omThm] (-3,0) circle (2pt);
  \fill[omDef] (3,0) circle (2pt);
  \draw[Stealth-Stealth, omExo] (-3,0.35) -- (0,0.35)

  \draw[Stealth-Stealth, omExo] (0,0.35) -- (3,0.35)
    node[midway, above, font=\small] {$3$};
\end{tikzpicture}

{\small De getallenlijn uitgebreid naar links: $-3$ en $+3$ zijn
\omterm{def:g7:negatives:def}{tegengestelde} getallen, op dezelfde afstand van $0$.}
\end{omfigure}

\begin{method}[Relatieve getallen vergelijken]\label{met:g7:negatives:compare}
Op de getallenlijn betekent kleiner verder naar links. In de praktijk:
\begin{enumerate}
  \item een negatief getal is altijd kleiner dan een positief:
        $-7 < 2$;
  \item tussen twee positieven, de gewone volgorde;
  \item tussen twee negatieven is die met de \emph{grotere} afstand tot
        \omterm{def:g1:counting:zero}{nul} de \emph{kleinere}: $-7 < -2$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:negatives:compare}
Orden van klein naar groot: $3$; $-5$; $0$; $-1.5$; $2.8$.
Eerst de negatieven, de meest negatieve voorop:
\[
-5 < -1.5 < 0 < 2.8 < 3 .
\]
\end{example}

\section{Relatieve getallen optellen}

Denk aan een \omterm{def:g7:negatives:def}{relatief getal} als een beweging langs de lijn: $+5$
betekent ``$5$ stappen naar rechts'', $-3$ betekent ``$3$ stappen naar
links''. Getallen optellen is de wandelingen aaneenschakelen.

\begin{method}[Twee relatieve getallen optellen]\label{met:g7:negatives:add}
\begin{enumerate}
  \item \emph{Zelfde teken}: tel de afstanden op, houd het gemeenschappelijke
        teken: $(-3) + (-4) = -7$.
  \item \emph{\omterm{def:g7:negatives:def}{Tegengestelde} tekens}: trek de kleinere afstand af van de
        grotere, neem het teken van de grotere:
        $(-7) + (+2) = -5$ en $(+7) + (-2) = +5$.
  \item \omterm{def:g7:negatives:def}{Tegengestelden} heffen elkaar op: $(+4) + (-4) = 0$.
\end{enumerate}
\end{method}

\end{tikzpicture}

{\small Optellen als wandelen: startend in $0$, de wandeling $-3.5$
en daarna de wandeling $+2$ landt op $(-3.5) + 2 = -1.5$.}
\end{omfigure}

\begin{example}\label{ex:g7:negatives:add}
Stap voor stap, groeperend waar handig:
\begin{align*}
(-8) + (+3) &= -5 && \text{(tegengestelde tekens: } 8 - 3 = 5,
\text{ teken van } 8)\\
(-2.5) + (-1.5) &= -4 && \text{(zelfde teken: afstanden optellen)}\\
(+6) + (-9) + (+3) &= (-3) + (+3) = 0 .
\end{align*}
\end{example}

\section{Relatieve getallen aftrekken}

\begin{theorem}[Aftrekregel]\label{thm:g7:negatives:sub}
Een \omterm{def:g7:negatives:def}{relatief getal} aftrekken is hetzelfde als \emph{zijn \omterm{def:g7:negatives:def}{tegengestelde}
optellen}:
\[
a - b = a + (-b) .
\]
\end{theorem}

\begin{proof}[Waarom het werkt]
$a - b$ is het getal dat, opgeteld bij $b$, $a$ geeft. Controleer dat
$a + (-b)$ dat doet:
$\bigl(a + (-b)\bigr) + b = a + \bigl((-b) + b\bigr) = a + 0 = a$.
\end{proof}

\begin{example}\label{ex:g7:negatives:sub}
Elke aftrekking wordt een optelling, daarna past
\cref{met:g7:negatives:add}:
\begin{align*}
5 - 9 &= 5 + (-9) = -4, \\
(-3) - (+4) &= (-3) + (-4) = -7, \\
(-2) - (-6) &= (-2) + (+6) = +4 .
\end{align*}
De laatste regel is de beroemde: \emph{een negatief aftrekken betekent
optellen}. Een schuld wegnemen maakt je rijker!
\end{example}

\begin{example}[Temperatuurverschillen]\label{ex:g7:negatives:temperature}
Op één dag ging de temperatuur van $-6$ graden naar $+9$ graden. De
stijging is het \omterm{ex:g1:subtraction:difference}{verschil}
\[
9 - (-6) = 9 + 6 = 15 \text{ graden}.
\]
Op de getallenlijn: van $-6$ tot $0$ is $6$ graden, van $0$ tot $9$
nog $9$ meer.
\end{example}

\section{Coördinaten in het hele vlak}

\begin{definition}[Coördinaten met tekens]\label{def:g7:negatives:coords}
Met twee loodrechte gegradueerde assen die snijden in de oorsprong
$O$ krijgt elk punt van het vlak twee relatieve coördinaten $(x, y)$:
negatieve $x$ betekent links van de verticale as, negatieve $y$
betekent onder de horizontale as.
\end{definition}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g7:negatives:1}
Geef het \omterm{def:g7:negatives:def}{tegengestelde} van: $7$; \ $-3.5$; \ $0$; \ $+12$. Vul daarna
aan: ``twee getallen zijn tegengesteld wanneer hun \omterm{def:g1:addition:def}{som}~? is''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:2}
Kopieer en vul in met $<$ of $>$:
\[
-4 \;?\; 1, \qquad
-4 \;?\; -7, \qquad
-2.5 \;?\; -2.4, \qquad
0 \;?\; -0.1 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:3}
Orden van klein naar groot: $-3$; \ $2.5$; \ $-0.5$; \ $-3.1$;
\ $0$; \ $1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:4}
Bereken:
\[
(-5) + (-9), \qquad
(-12) + (+7), \qquad
(+3.5) + (-1.5), \qquad
(-6) + (+6).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:5}
Zet om in optellingen, en bereken daarna:
\[
4 - 11, \qquad
(-5) - (+3), \qquad
(-7) - (-10), \qquad
0 - (-8).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:6}
Bereken door slim te groeperen (eerst \omterm{def:g7:negatives:def}{tegengestelden}):
\[
(+7) + (-12) + (+3) + (-8), \qquad
(-5.5) + (+9) + (-4.5) + (+1).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:7}
De temperatuur was vanochtend $-3$ graden; ze steeg overdag met $8$
graden, en daalde 's nachts met $12$ graden. Schrijf één berekening
die de eindtemperatuur geeft, en reken die uit.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:8}
Teken in een assenstelsel: $A(2, 3)$; $B(-3, 1)$; $C(-2, -2)$;
$D(3, -1)$; $E(0, -3)$. Welk punt ligt op een as?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:negatives:9}
Bereken het \omterm{ex:g1:subtraction:difference}{verschil} tussen de hoogste en laagste temperatuur:
Moskou in januari, max $-4$ graden, min $-13$ graden; Helsinki, max
$2$, min $-9$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:negatives:10}
Een duiker bevindt zich op hoogte $-12$ m (twaalf meter onder het
oppervlak). Ze stijgt $7$ m, en daalt daarna $4$ m. Op welke hoogte
is ze? Hoe ver moet ze nog stijgen om het oppervlak te bereiken?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:negatives:11}
Teken $A(-2, 1)$, $B(2, 1)$, $C(2, -2)$. Vind de coördinaten van het
punt $D$ zodat $ABCD$ een rechthoek is, en bereken zijn \omterm{def:g6:measure:perimeter}{omtrek}
(tel de roostervierkanten voor de zijdelengten).
\end{exercise}

\end{exercise}
