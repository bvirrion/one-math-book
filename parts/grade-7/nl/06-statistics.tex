\chapter{Gegevens organiseren}\label{ch:g7:stats}

Voor je iets berekent, moeten gegevens verzameld, geteld en
weergegeven worden. Dit hoofdstuk gaat over het omzetten van een
rommelige lijst observaties in een duidelijke tabel of figuur:
tellingen, frequenties, gegroepeerde gegevens, staafdiagrammen en
cirkeldiagrammen. Gegevens samenvatten met één getal (gemiddelde,
mediaan) is het onderwerp van \cref{ch:g9:statproba}.

\section{Tellingen en frequenties}

\begin{definition}[Telling en frequentie]\label{def:g7:stats:frequency}
Voor elke mogelijke waarde van een enquête is de
\emph{telling}\index{telling} het aantal keren dat ze voorkomt; de
\emph{frequentie}\index{frequentie} is de telling gedeeld door het
totale aantal observaties:
\[
\text{frequentie} = \frac{\text{telling}}{\text{totaal}} .
\]
Frequenties kunnen als breuken, decimalen of percentages geschreven
worden, en ze tellen altijd op tot $1$ (d.w.z.\ $100\,\%$).
\end{definition}

\begin{example}\label{ex:g7:stats:frequency}
Favoriete sport van de $25$ leerlingen van een klas, van de ruwe
lijst naar de tabel:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cccc|c}
sport & voetbal & zwemmen & tennis & judo & totaal \\
\hline
telling & $10$ & $6$ & $5$ & $4$ & $25$ \\
frequentie & $0.40$ & $0.24$ & $0.20$ & $0.16$ & $1$ \\
\end{tabular}
\end{center}
Voor voetbal: $\frac{10}{25} = \frac{40}{100} = 0.40 = 40\,\%$. De
frequentierij moet tot $1$ sommeren --- een permanente, gratis
controle.
\end{example}

\begin{remark}[Waarom frequenties?]
Tellingen kunnen niet vergeleken worden over groepen van verschillende
grootte: $10$ voetbalfans in een klas van $25$ (dat is $40\,\%$) tonen
meer enthousiasme dan $12$ in een schooljaar van $120$ ($10\,\%$).
Frequenties zetten alles op dezelfde schaal.
\end{remark}

\section{Gegevens groeperen in klassen}

\begin{example}[Klassen]\label{ex:g7:stats:classes}
De lengten (in cm) van $20$ leerlingen lopen van $148$ tot $172$.
Elke lengte apart opsommen zou een tabel geven met bijna evenveel
kolommen als leerlingen; groepeer in plaats daarvan in \emph{klassen}:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|ccc|c}
lengte (cm) & $\intco{145}{155}$ & $\intco{155}{165}$ &
$\intco{165}{175}$ & totaal \\
\hline
telling & $5$ & $9$ & $6$ & $20$ \\
frequentie & $0.25$ & $0.45$ & $0.30$ & $1$ \\
\end{tabular}
\end{center}
Elke lengte wordt in precies één klas geteld ($155$ gaat in
$\intco{155}{165}$, niet in beide --- daarom zijn de intervallen
halfopen). Groeperen verliest detail maar wint leesbaarheid.
\end{example}

\section{Diagrammen}

\begin{method}[Een diagram kiezen en tekenen]\label{met:g7:stats:charts}
\begin{enumerate}
  \item \emph{Staafdiagram}: één staaf per waarde of klas, hoogte
        evenredig met de telling (of de frequentie). Goed om waarden
        te vergelijken.
  \item \emph{Cirkeldiagram}: één sector per waarde, hoek evenredig
        met de frequentie --- de volle cirkel ($360^\circ$) stelt het
        totaal voor, dus
        \[
        \text{hoek} = \text{frequentie} \times 360^\circ .
        \]
        Goed om delen van een geheel te tonen.
  \item In beide gevallen: titel, labels, en een schaalverdeling of de
        percentages op het diagram geschreven.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:stats:pie}
Hoeken voor de sporten van \cref{ex:g7:stats:frequency}: voetbal
$0.40 \times 360 = 144^\circ$; zwemmen $0.24 \times 360 =
86.4^\circ$; tennis $72^\circ$; judo $57.6^\circ$. Controle:
$144 + 86.4 + 72 + 57.6 = 360$.
\end{example}

\begin{example}[Kritisch lezen]\label{ex:g7:stats:critical}
Een diagram waarvan de verticale as bij $9$ begint in plaats van $0$
laat een staaf van $10$ er tien keer hoger uitzien dan een staaf van
$9.1$ --- visueel misleidend, hoewel de getallen eerlijk zijn. Eerste
reflex bij elk diagram: controleer waar de as begint en wat één
streepje waard is (\cref{met:g6:propdata:read}).
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g7:stats:1}
Hier zijn de cijfers van $20$ leerlingen: $12$, $15$, $9$, $12$,
$14$, $9$, $12$, $15$, $11$, $12$, $9$, $14$, $12$, $11$, $15$, $9$,
$12$, $14$, $11$, $12$. Bouw de tellingentabel (één kolom per
cijfer).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:stats:2}
Vul de tabel van \cref{exo:g7:stats:1} aan met een frequentierij
(als breuken van $20$, daarna als percentages). Controleer de som.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:stats:3}
In een enquête bij $50$ gezinnen hebben $30$ één auto. Wat is de
frequentie van ``één auto'', als decimaal en als percentage? In een
andere enquête hebben $45$ gezinnen van de $90$ één auto: welke
enquête toont het hogere aandeel?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:stats:4}
De massa's (kg) van $15$ honden: $8$, $23$, $31$, $12$, $9$, $27$,
$35$, $14$, $22$, $18$, $29$, $11$, $25$, $33$, $16$. Groepeer ze in
de klassen $\intco{5}{15}$, $\intco{15}{25}$, $\intco{25}{35}$,
$\intco{35}{45}$ en geef de tellingen.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:stats:5}
Teken het staafdiagram van de tabel:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|cccc}
huisdieren & $0$ & $1$ & $2$ & $3$ \\
\hline
gezinnen & $8$ & $12$ & $6$ & $4$ \\
\end{tabular}
\end{center}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:stats:6}
Voor de tabel van \cref{exo:g7:stats:5}, bereken de frequenties en
de cirkeldiagramhoeken van elke waarde (controle: totaal $360^\circ$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:stats:7}
Een cirkeldiagram over favoriete seizoenen toont: zomer $162^\circ$,
lente $90^\circ$, herfst $54^\circ$, winter $54^\circ$. Welk percentage
koos elk seizoen? Als $60$ personen antwoordden, hoeveel kozen de
zomer?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:stats:8}
In klas A lopen $12$ leerlingen van de $30$ naar school; in klas B,
$14$ van de $40$. Welke klas heeft het hogere aandeel lopenden?
Motiveer met frequenties, niet met tellingen.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:stats:9}
Een tijdschrift print een staafdiagram van maandelijkse verkopen:
$9\,800$, daarna $10\,000$, daarna $10\,100$, met de verticale as
beginnend bij $9\,700$. Beschrijf wat de lezer ziet, en wat een
eerlijke as vanaf $0$ in plaats daarvan zou tonen.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:stats:10}
Een frequentietabel heeft drie waarden met frequenties $0.35$, $0.4$
en $f$. Vind $f$. Als de totale telling $80$ is, geef de drie
tellingen.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:stats:11}
In een school is $55\,\%$ van de leerlingen meisje. Onder de meisjes
eet $40\,\%$ in de kantine; onder de jongens $60\,\%$. Van de $400$
leerlingen, hoeveel eten in de kantine? (Bereken de vier
groepsgroottes stap voor stap.) Welk overall percentage is dat?
\end{exercise}
