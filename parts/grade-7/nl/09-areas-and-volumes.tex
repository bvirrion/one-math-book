\chapter{Oppervlakten en volumes}\label{ch:g7:areas}

Jaar~6 mat rechthoeken, rechthoekige driehoeken en balken
(\cref{ch:g6:measure}). Met een schaar --- snijd hier een stuk af, plak
het daar --- geven dezelfde ideeën nu de oppervlakten van
parallellogrammen, van \emph{alle} driehoeken en van de schijf, en de
volumes van prisma's en cilinders.

\section{Oppervlakte van een parallellogram}

\begin{theorem}[Parallellogram]\label{thm:g7:areas:parallelogram}
Een parallellogram met een zijde van lengte $b$ (zijn \emph{basis}) en
afstand $h$ tussen die zijde en de overstaande (zijn \emph{hoogte})
heeft oppervlakte
\[
A = b \times h .
\]
\end{theorem}

\begin{proof}[Bewijs met de schaar]
Snijd de rechthoekige driehoek die aan één kant uitsteekt af en plak
hem aan de andere kant: het parallellogram wordt een rechthoek
$b \times h$ met dezelfde oppervlakte.
\end{proof}

\begin{remark}
De hoogte is de \emph{loodrechte} afstand tussen de twee basissen,
niet de lengte van de schuine zijde! Een zeer scheef parallellogram
kan lange zijden en een kleine oppervlakte hebben.
\end{remark}

\section{Oppervlakte van een driehoek}

\begin{theorem}[Driehoek]\label{thm:g7:areas:triangle}
Een driehoek met basis $b$ en bijbehorende hoogte $h$ (de loodrechte
afstand van het overstaande hoekpunt tot de basislijn) heeft
oppervlakte
\[
A = \frac{b \times h}{2} .
\]
\end{theorem}

\begin{proof}
Twee kopieën van de driehoek, één een halve slag gedraaid om het
midden van één zijde, passen samen in een parallellogram met basis
$b$ en hoogte $h$ (dat is centrale spiegeling aan het werk,
\cref{prop:g7:central:preserved}). De driehoek is de helft ervan:
$\frac{bh}{2}$.
\end{proof}

{\small Een driehoek en zijn halfslagkopie (gestippeld) maken een
parallellogram: de oppervlakte van de driehoek is
$\frac{b\times h}{2}$. Elk van de drie zijden kan als basis dienen ---
met \emph{zijn eigen} hoogte.}
\end{omfigure}

\begin{example}\label{ex:g7:areas:triangle}
Een driehoek heeft een basis van $9$ cm en de bijbehorende hoogte
meet $4$ cm:
\[
A = \frac{9 \times 4}{2} = \frac{36}{2} = 18 \text{ cm}^2 .
\]
Voor een \emph{rechthoekige} driehoek zijn de twee benen een basis en
haar hoogte: we herkrijgen de formule van
\cref{prop:g6:measure:formulas}.
\end{example}

\section{Oppervlakte van een schijf}

\begin{theorem}[Schijf]\label{thm:g7:areas:disk}
Een schijf met straal $r$ heeft oppervlakte
\[
A = \pi r^2 .
\]
\end{theorem}

\begin{proof}[Bewijs-idee]
Snijd de schijf in veel dunne gelijke plakjes en leg ze kop aan
staart: ze vormen bijna een parallellogram van hoogte $r$ waarvan de
basis de helft van de omtrek is, $\pi r$
(\cref{prop:g6:measure:circle}). Zijn oppervlakte is dicht bij
$\pi r \times r = \pi r^2$, en de benadering wordt perfect als de
plakjes dunner worden. Een volledig bewijs vraagt integraalrekening,
uit het High School-volume.
\end{proof}

\end{omfigure}

\begin{example}\label{ex:g7:areas:disk}
Een ronde tafelblad heeft straal $60$ cm. Zijn oppervlakte is
$\pi \times 60^2 = 3600\pi \approx 11\,310$ cm$^2$, iets meer dan
$1.1$ m$^2$. Let op het verschil: de \emph{omtrek} $2\pi r$ gebruikt
$r$, de \emph{oppervlakte} $\pi r^2$ gebruikt $r^2$.
\end{example}

\section{Prisma's en cilinders}

\begin{definition}[Prisma, cilinder]\label{def:g7:areas:prism}
Een \emph{prisma}\index{prisma} is een lichaam met twee identieke
evenwijdige veelhoekige vlakken (de \emph{grondvlakken}) verbonden
door rechthoeken; een \emph{cilinder}\index{cilinder} is hetzelfde
met schijven als grondvlakken. De \emph{hoogte} is de afstand tussen
de twee grondvlakken.
\end{definition}

\begin{theorem}[Volumes]\label{thm:g7:areas:volume}
Voor een prisma of een cilinder met grondvlakoppervlakte $B$ en
hoogte $h$:
\[
V = B \times h .
\]
\end{theorem}
\admitted

\begin{example}\label{ex:g7:areas:volume}
Een tent is een prisma op zijn zij: zijn grondvlakken zijn driehoeken
met basis $2$ m en hoogte $1.5$ m, en de tent is $3$ m lang.
\begin{enumerate}
  \item Grondvlakoppervlakte: $B = \frac{2 \times 1.5}{2} = 1.5$ m$^2$.
  \item Volume: $V = B \times h = 1.5 \times 3 = 4.5$ m$^3$.
\end{enumerate}
Een blikje met straal $4$ cm en hoogte $11$ cm:
$V = \pi \times 4^2 \times 11 = 176\pi \approx 553$ cm$^3$, ruwweg
een halve liter.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g7:areas:1}
Bereken de oppervlakte van een parallellogram met basis $8$ cm en
hoogte $5$ cm; van een ander met basis $6.5$ m en hoogte $4$ m.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:areas:2}
Een parallellogram heeft zijden $10$ cm en $6$ cm, en de hoogte ten
opzichte van de basis van $10$ cm meet $4$ cm. Bereken zijn
oppervlakte. Waarom is het antwoord niet $60$ cm$^2$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:areas:3}
Bereken de oppervlakte van een driehoek met basis $12$ cm en hoogte
$7$ cm; van een rechthoekige driehoek met benen $9$ en $6$; van een
driehoek met basis $5.5$ m en hoogte $2$ m.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:areas:4}
Bereken de oppervlakte van een schijf met straal $5$ cm, daarna van
een halfschijf met straal $10$ cm (exacte waarden met $\pi$, daarna
afgerond op de eenheid).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:areas:5}
Bereken de omtrek \emph{en} de oppervlakte van een schijf met straal
$3$ cm. Welk van de twee antwoorden is in cm en welk in cm$^2$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:areas:6}
Bereken het volume van een prisma met grondvlakoppervlakte $24$ cm$^2$
en hoogte $10$ cm; van een cilinder met straal $3$ cm en hoogte
$8$ cm (exact, daarna afgerond).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:areas:7}
Een driehoek heeft oppervlakte $28$ cm$^2$ en basis $8$ cm. Vind de
bijbehorende hoogte, en schrijf eerst de vergelijking.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:areas:8}
Teken een willekeurige driehoek en meet zorgvuldig de drie
basis--hoogte-paren. Bereken $\frac{b \times h}{2}$ voor elk paar: de
drie resultaten zouden moeten overeenkomen (op meetfout na). Waarom?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:areas:9}
Een trapeziumvormige tuin kan door één diagonaal gesplitst worden in
twee driehoeken die dezelfde hoogte $h = 20$ m delen, met basissen
$30$ m en $18$ m. Bereken de totale oppervlakte. Kun je een algemene
formule voor het trapezium raden?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:areas:10}
Een cilindrische vaas met straal $6$ cm bevat water tot een hoogte van
$15$ cm. Al het water wordt in een lege bak gegoten van
$18$ cm $\times$ $12$ cm aan de basis. Welke waterhoogte wordt in de
bak bereikt? (Zelfde volume, andere basis.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:areas:11}
Een pizza met diameter $30$ cm kost $9$ euro; één met diameter
$40$ cm kost $14$ euro. Bereken de twee oppervlakten en de prijs per
$100$ cm$^2$. Welke pizza is de betere deal?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:areas:12}
De twee benen van een rechthoekige driehoek meten $6$ cm en $8$ cm,
en zijn schuine zijde meet $10$ cm. Bereken zijn oppervlakte met de
benen, en gebruik die oppervlakte om de hoogte ten opzichte van de
\emph{schuine zijde} te vinden.
\end{exercise}
