\chapter{Oppervlakten en volumes}\label{ch:g7:areas}

Jaar~6 mat rechthoeken, \omterm{def:g6:shapes:triangles}{rechthoekige driehoeken} en \omterm{def:g5:solids:def}{balken}
(\cref{ch:g6:measure}). Met een schaar --- snijd hier een stuk af, plak
het daar --- geven dezelfde ideeën nu de \omterm{def:g6:measure:area}{oppervlakten} van
parallellogrammen, van \emph{alle} driehoeken en van de schijf, en de
\omterm{def:g6:measure:volume}{volumes} van \omterm{def:g7:areas:prism}{prisma}'s en \omterm{def:g7:areas:prism}{cilinders}.

\section{Oppervlakte van een parallellogram}

\begin{theorem}[Parallellogram]\label{thm:g7:areas:parallelogram}
Een \omterm{def:g7:central:parallelogram}{parallellogram} met een zijde van lengte $b$ (zijn \emph{basis}) en
afstand $h$ tussen die zijde en de overstaande (zijn \emph{hoogte})
heeft \omterm{def:g6:measure:area}{oppervlakte}
\[
A = b \times h .
\]
\end{theorem}

\begin{proof}[Bewijs met de schaar]
Snijd de \omterm{def:g6:shapes:triangles}{rechthoekige driehoek} die aan één kant uitsteekt af en plak
hem aan de andere kant: het \omterm{def:g7:central:parallelogram}{parallellogram} wordt een rechthoek
$b \times h$ met dezelfde \omterm{def:g6:measure:area}{oppervlakte}.
\end{proof}

  \fill[omDef!14] (0,0) -- (4,0) -- (5.2,1.8) -- (1.2,1.8) -- cycle;
  \draw[thick] (0,0) -- (4,0) -- (5.2,1.8) -- (1.2,1.8) -- cycle;
  \fill[omThm!25] (0,0) -- (1.2,1.8) -- (1.2,0) -- cycle;

  \draw[dashed] (1.2,0) -- (1.2,1.8);
  \node[below, font=\small] at (2.0,0) {$b$};
  \draw[Stealth-Stealth] (6.0,0) -- (6.0,1.8)
    node[midway, right, font=\small] {$h$};
  \begin{scope}[xshift=8.2cm]
  \fill[omDef!14] (0,0) rectangle (4,1.8);

  \fill[omThm!25] (2.8,0) -- (4,1.8) -- (4,0) -- cycle;
  \draw[omThm, thick] (2.8,0) -- (4,1.8) -- (4,0) -- cycle;
  \node[below, font=\small] at (2.0,0) {$b$};
  \end{scope}
\end{tikzpicture}

{\small Het schaarbewijs: de rode driehoek van het linkereinde naar
het rechtereinde schuiven maakt van het \omterm{def:g7:central:parallelogram}{parallellogram} een rechthoek
---zelfde \omterm{def:g6:measure:area}{oppervlakte}, $b \times h$.}
\end{omfigure}

\begin{remark}
De hoogte is de \emph{loodrechte} afstand tussen de twee basissen,
niet de lengte van de schuine zijde! Een zeer scheef \omterm{def:g7:central:parallelogram}{parallellogram}
kan lange zijden en een kleine \omterm{def:g6:measure:area}{oppervlakte} hebben.
\end{remark}

\section{Oppervlakte van een driehoek}

\begin{theorem}[Driehoek]\label{thm:g7:areas:triangle}
Een driehoek met basis $b$ en bijbehorende hoogte $h$ (de loodrechte
afstand van het overstaande \omterm{def:g5:solids:def}{hoekpunt} tot de basislijn) heeft
\omterm{def:g6:measure:area}{oppervlakte}
\[
A = \frac{b \times h}{2} .
\]
\end{theorem}

\begin{proof}
Twee kopieën van de driehoek, één een halve slag gedraaid om het
midden van één zijde, passen samen in een \omterm{def:g7:central:parallelogram}{parallellogram} met basis
$b$ en hoogte $h$ (dat is \omterm{def:g7:central:def}{centrale spiegeling} aan het werk,
\cref{prop:g7:central:preserved}). De driehoek is de \omterm{def:g3:division:half}{helft} ervan:
$\frac{bh}{2}$.
\end{proof}

\begin{tikzpicture}[scale=0.9]
  \fill[omDef!14] (0,0) -- (4,0) -- (1.4,2.1) -- cycle;
  \draw[thick] (0,0) -- (4,0) -- (1.4,2.1) -- cycle;
  \fill[omThm!12] (4,0) -- (1.4,2.1) -- (5.4,2.1) -- cycle;
  \draw[thick, omThm, dashed] (4,0) -- (5.4,2.1) -- (1.4,2.1);
  \draw[dashed] (1.4,0) -- (1.4,2.1);
  \draw[thin] (1.4,0) ++(-0.2,0) -- ++(0,0.2) -- ++(0.2,0);
  \node[below, font=\small] at (2,0) {$b$};
  \node[right, font=\small] at (1.42,1.0) {$h$};
\end{tikzpicture}

{\small Een driehoek en zijn halfslagkopie (gestippeld) maken een
\omterm{def:g7:central:parallelogram}{parallellogram}: de \omterm{def:g6:measure:area}{oppervlakte} van de driehoek is
$\frac{b\times h}{2}$. Elk van de drie zijden kan als basis dienen ---
met \emph{zijn eigen} hoogte.}
\end{omfigure}

\begin{example}\label{ex:g7:areas:triangle}
Een driehoek heeft een basis van $9$ cm en de bijbehorende hoogte
meet $4$ cm:
\[
A = \frac{9 \times 4}{2} = \frac{36}{2} = 18 \text{ cm}^2 .
\]
Voor een \emph{rechthoekige} driehoek zijn de twee benen een basis en
haar hoogte: we herkrijgen de formule van
\cref{prop:g6:measure:formulas}.
\end{example}

\section{Oppervlakte van een schijf}

\begin{theorem}[Schijf]\label{thm:g7:areas:disk}
Een schijf met \omterm{def:g3:shapes:shapes}{straal} $r$ heeft \omterm{def:g6:measure:area}{oppervlakte}
\[
A = \pi r^2 .
\]
\end{theorem}

\begin{proof}[Bewijs-idee]
Snijd de schijf in veel dunne gelijke plakjes en leg ze kop aan
staart: ze vormen bijna een \omterm{def:g7:central:parallelogram}{parallellogram} van hoogte $r$ waarvan de
basis de \omterm{def:g3:division:half}{helft} van de \omterm{def:g6:measure:perimeter}{omtrek} is, $\pi r$
(\cref{prop:g6:measure:circle}). Zijn \omterm{def:g6:measure:area}{oppervlakte} is dicht bij
$\pi r \times r = \pi r^2$, en de benadering wordt perfect als de
plakjes dunner worden. Een volledig bewijs vraagt integraalrekening,
uit het High School-volume.
\end{proof}

\begin{tikzpicture}[scale=0.85]
  \foreach \a in {0,45,...,315}
    \fill[omDef!20, draw=omDef] (0,0) -- (\a:1.4)
      arc (\a:\a+45:1.4) -- cycle;

  \foreach \i in {0,1,2,3} {
    \fill[omDef!20, draw=omDef]
      ({\i*1.1},0) -- ({\i*1.1+0.55},1.4) -- ({\i*1.1+1.1},0) -- cycle;
    \fill[omDef!20, draw=omDef]
      ({\i*1.1+0.55},1.4) -- ({\i*1.1+1.1},0) -- ({\i*1.1+1.65},1.4)
      -- cycle;
  }
  \draw[Stealth-Stealth] (5.2,0) -- (5.2,1.4)
    node[midway, right, font=\small] {$\approx r$};
  \node[below, font=\small] at (2.5,-0.1) {basis $\approx \pi r$};
  \end{scope}
\end{tikzpicture}

{\small De schijf snijden en de plakjes afrollen: bijna een
\omterm{def:g7:central:parallelogram}{parallellogram} met basis $\pi r$ (de \omterm{def:g3:division:half}{helft} van de rand) en hoogte
$r$.}
\end{omfigure}

\begin{example}\label{ex:g7:areas:disk}
Een ronde tafelblad heeft \omterm{def:g3:shapes:shapes}{straal} $60$ cm. Zijn \omterm{def:g6:measure:area}{oppervlakte} is
$\pi \times 60^2 = 3600\pi \approx 11\,310$ cm$^2$, iets meer dan
$1.1$ m$^2$. Let op het \omterm{ex:g1:subtraction:difference}{verschil}: de \emph{\omterm{def:g6:measure:perimeter}{omtrek}} $2\pi r$ gebruikt
$r$, de \emph{\omterm{def:g6:measure:area}{oppervlakte}} $\pi r^2$ gebruikt $r^2$.
\end{example}

\section{Prisma's en cilinders}

\begin{definition}[Prisma, cilinder]\label{def:g7:areas:prism}
Een \emph{prisma}\index{prisma} is een \omterm{def:g5:solids:def}{lichaam} met twee identieke
evenwijdige veelhoekige \omterm{def:g5:solids:def}{vlakken} (de \emph{grondvlakken}) verbonden
door rechthoeken; een \emph{cilinder}\index{cilinder} is hetzelfde
met schijven als grondvlakken. De \emph{hoogte} is de afstand tussen
de twee grondvlakken.
\end{definition}

\begin{theorem}[Volumes]\label{thm:g7:areas:volume}
Voor een \omterm{def:g7:areas:prism}{prisma} of een \omterm{def:g7:areas:prism}{cilinder} met grondvlakoppervlakte $B$ en
hoogte $h$:
\[
V = B \times h .
\]
\end{theorem}
\admitted

\begin{example}\label{ex:g7:areas:volume}
Een tent is een \omterm{def:g7:areas:prism}{prisma} op zijn zij: zijn grondvlakken zijn driehoeken
met basis $2$ m en hoogte $1.5$ m, en de tent is $3$ m lang.
\begin{enumerate}
  \item Grondvlakoppervlakte: $B = \frac{2 \times 1.5}{2} = 1.5$ m$^2$.
  \item \omterm{def:g6:measure:volume}{Volume}: $V = B \times h = 1.5 \times 3 = 4.5$ m$^3$.
\end{enumerate}
Een blikje met \omterm{def:g3:shapes:shapes}{straal} $4$ cm en hoogte $11$ cm:
$V = \pi \times 4^2 \times 11 = 176\pi \approx 553$ cm$^3$, ruwweg
een halve liter.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g7:areas:1}
Bereken de \omterm{def:g6:measure:area}{oppervlakte} van een \omterm{def:g7:central:parallelogram}{parallellogram} met basis $8$ cm en
hoogte $5$ cm; van een ander met basis $6.5$ m en hoogte $4$ m.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:areas:2}
Een \omterm{def:g7:central:parallelogram}{parallellogram} heeft zijden $10$ cm en $6$ cm, en de hoogte ten
opzichte van de basis van $10$ cm meet $4$ cm. Bereken zijn
\omterm{def:g6:measure:area}{oppervlakte}. Waarom is het antwoord niet $60$ cm$^2$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:areas:3}
Bereken de \omterm{def:g6:measure:area}{oppervlakte} van een driehoek met basis $12$ cm en hoogte
$7$ cm; van een \omterm{def:g6:shapes:triangles}{rechthoekige driehoek} met benen $9$ en $6$; van een
driehoek met basis $5.5$ m en hoogte $2$ m.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:areas:4}
Bereken de \omterm{def:g6:measure:area}{oppervlakte} van een schijf met \omterm{def:g3:shapes:shapes}{straal} $5$ cm, daarna van
een halfschijf met \omterm{def:g3:shapes:shapes}{straal} $10$ cm (exacte waarden met $\pi$, daarna
afgerond op de eenheid).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:areas:5}
Bereken de \omterm{def:g6:measure:perimeter}{omtrek} \emph{en} de \omterm{def:g6:measure:area}{oppervlakte} van een schijf met \omterm{def:g3:shapes:shapes}{straal}
$3$ cm. Welk van de twee antwoorden is in cm en welk in cm$^2$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:areas:6}
Bereken het \omterm{def:g6:measure:volume}{volume} van een \omterm{def:g7:areas:prism}{prisma} met grondvlakoppervlakte $24$ cm$^2$
en hoogte $10$ cm; van een \omterm{def:g7:areas:prism}{cilinder} met \omterm{def:g3:shapes:shapes}{straal} $3$ cm en hoogte
$8$ cm (exact, daarna afgerond).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:areas:7}
Een driehoek heeft \omterm{def:g6:measure:area}{oppervlakte} $28$ cm$^2$ en basis $8$ cm. Vind de
bijbehorende hoogte, en schrijf eerst de vergelijking.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:areas:8}
Teken een willekeurige driehoek en meet zorgvuldig de drie
basis--hoogte-paren. Bereken $\frac{b \times h}{2}$ voor elk paar: de
drie resultaten zouden moeten overeenkomen (op meetfout na). Waarom?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:areas:9}
Een trapeziumvormige tuin kan door één diagonaal gesplitst worden in
twee driehoeken die dezelfde hoogte $h = 20$ m delen, met basissen
$30$ m en $18$ m. Bereken de totale \omterm{def:g6:measure:area}{oppervlakte}. Kun je een algemene
formule voor het trapezium raden?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:areas:10}
Een cilindrische vaas met \omterm{def:g3:shapes:shapes}{straal} $6$ cm bevat water tot een hoogte van
$15$ cm. Al het water wordt in een lege bak gegoten van
$18$ cm $\times$ $12$ cm aan de basis. Welke waterhoogte wordt in de
bak bereikt? (Zelfde \omterm{def:g6:measure:volume}{volume}, andere basis.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:areas:11}
Een pizza met \omterm{def:g4:geometry:circle}{diameter} $30$ cm kost $9$ euro; één met \omterm{def:g4:geometry:circle}{diameter}
$40$ cm kost $14$ euro. Bereken de twee \omterm{def:g6:measure:area}{oppervlakten} en de prijs per
$100$ cm$^2$. Welke pizza is de betere deal?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:areas:12}
De twee benen van een \omterm{def:g6:shapes:triangles}{rechthoekige driehoek} meten $6$ cm en $8$ cm,
en zijn schuine zijde meet $10$ cm. Bereken zijn \omterm{def:g6:measure:area}{oppervlakte} met de
benen, en gebruik die \omterm{def:g6:measure:area}{oppervlakte} om de hoogte ten opzichte van de
\emph{schuine zijde} te vinden.
\end{exercise}
