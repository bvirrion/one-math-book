\chapter{Evenredigheid}\label{ch:g7:prop}

Jaar~6 leerde evenredigheidstabellen (\cref{ch:g6:propdata}); dit
hoofdstuk maakt van \omterm{def:g7:prop:table}{evenredigheid} een volwaardig gereedschap: de
kruisregel om een vierde waarde te vinden, percentages, en schalen van
kaarten en modellen. Snelheid, de beroemdste \omterm{def:g7:prop:table}{evenredigheid} van alle,
wordt bestudeerd in \cref{ch:g8:speed}.

\section{Tabellen herkennen en invullen}

\begin{definition}[Evenredigheidstabel]\label{def:g7:prop:table}
Een tabel met twee rijen is een \emph{evenredigheidstabel}\index{evenredigheid}
wanneer de getallen van de tweede rij die van de eerste zijn,
vermenigvuldigd met één vast getal, de \emph{coëfficiënt}. Equivalent:
alle kolomquotiënten $\frac{\text{onder}}{\text{boven}}$ zijn gelijk.
\end{definition}

\begin{example}\label{ex:g7:prop:recognize}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|ccc}
boven & $4$ & $10$ & $14$ \\
\hline
onder & $6$ & $15$ & $21$ \\
\end{tabular}
\end{center}
\omterm{def:g3:division:remainder}{Quotiënten}: $\frac64 = 1.5$, $\frac{15}{10} = 1.5$,
$\frac{21}{14} = 1.5$: \omterm{def:g6:propdata:def}{evenredig}, coëfficiënt $1.5$.
\end{example}

\begin{theorem}[Kruisregel]\label{thm:g7:prop:cross}
In een \omterm{def:g7:prop:table}{evenredigheidstabel} met kolommen $\begin{pmatrix} a \\ b
\end{pmatrix}$ en $\begin{pmatrix} c \\ d \end{pmatrix}$ zijn de
kruisproducten gelijk:
\[
a \times d = b \times c .
\]
Bijgevolg kan de vierde waarde uit de drie andere berekend worden:
$d = \dfrac{b \times c}{a}$.
\end{theorem}

\begin{proof}
Laat $k$ de coëfficiënt zijn: $b = ka$ en $d = kc$. Dan
\[
a \times d = a \times kc = k \times ac,
\qquad
b \times c = ka \times c = k \times ac :
\]
beide kruisproducten zijn $k \times ac$. Delen van $ad = bc$ door $a$
geeft de formule voor $d$.
\end{proof}

\begin{example}[Vierde evenredige]\label{ex:g7:prop:fourth}
Als $7$ identieke boeken $2.8$ kg wegen, hoeveel wegen $12$ zulke
boeken? Tabel en kruisregel:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cc}
boeken & $7$ & $12$ \\
\hline
kg & $2.8$ & $m$ \\
\end{tabular}
\end{center}
\[
7 \times m = 2.8 \times 12
\quad\Longrightarrow\quad
m = \frac{2.8 \times 12}{7} = \frac{33.6}{7} = 4.8 \text{ kg}.
\]
Controle via de eenheid: één boek weegt $0.4$ kg, twaalf wegen
$4.8$ kg.
\end{example}

\section{Percentages}

\begin{method}[Percentages toepassen en vinden]\label{met:g7:prop:percent}
\begin{enumerate}
  \item \emph{Toepassen} van $t\,\%$: vermenigvuldig met $\frac{t}{100}$
        ($12\,\%$ van $350$ is $0.12 \times 350 = 42$);
  \item \emph{vinden} welk percentage een deel voorstelt: bereken
        $\frac{\text{deel}}{\text{geheel}}$ en herschrijf over $100$
        ($21$ van de $60$: $\frac{21}{60} = \frac{35}{100} = 35\,\%$);
  \item vraag altijd: \emph{een percentage van wat?} Het
        referentiegeheel telt.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:prop:percent}
In een school zijn $180$ van de $400$ leerlingen meisjes. Percentage:
\[
\frac{180}{400} = \frac{180 \div 4}{400 \div 4} = \frac{45}{100}
= 45\,\% .
\]
Omgekeerd: hoeveel zijn de $55\,\%$ jongens?
$0.55 \times 400 = 220$. Controle: $180 + 220 = 400$.
\end{example}

\section{Schalen}

\begin{definition}[Schaal]\label{def:g7:prop:scale}
Op een kaart of model op \emph{schaal}\index{schaal}
$\frac{1}{n}$ (geschreven $1 : n$) zijn de afstanden op de kaart
\omterm{def:g6:propdata:def}{evenredig} met de echte afstanden, met coëfficiënt $\frac1n$:
\[
\text{kaartafstand} = \frac{1}{n} \times \text{echte afstand},
\]
beide gemeten in \emph{dezelfde eenheid}.
\end{definition}

\begin{example}\label{ex:g7:prop:scale}
Op een kaart $1 : 50\,000$ liggen twee steden $7$ cm uit elkaar.
Echte afstand:
\[
7 \times 50\,000 = 350\,000 \text{ cm} = 3\,500 \text{ m} = 3.5
\text{ km}.
\]
Omgekeerd meet een wandeling van $12$ km op de kaart
$12$ km $= 1\,200\,000$ cm, dus $1\,200\,000 \div 50\,000 = 24$ cm.
\end{example}

{\small Een \omterm{def:g7:prop:scale}{schaal} is een \omterm{def:g7:prop:table}{evenredigheid}: kaartafstanden tegen echte

$\leftrightarrow$ $28$ km).}
\end{omfigure}

\begin{remark}
Grafieken zijn de snelste test van \omterm{def:g7:prop:table}{evenredigheid}
(\cref{ch:g6:propdata}): punten op een \omterm{def:g6:lines:objects}{rechte} lijn \emph{door de
oorsprong} betekenen evenredige grootheden; een lijn die de oorsprong
mist (zoals een taxitarief met vast startbedrag) betekent \emph{niet}
\omterm{def:g6:propdata:def}{evenredig}, ook al is het een \omterm{def:g6:lines:objects}{rechte} lijn. Affiene functies, in
\cref{ch:g9:linfunc}, maken dit precies.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g7:prop:1}
Welke tabellen zijn evenredigheidstabellen? Geef de coëfficiënt als
die bestaat.
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|ccc}
 & $3$ & $7$ & $11$ \\
\hline
 & $7.5$ & $17.5$ & $27.5$ \\
\end{tabular}
\qquad
\begin{tabular}{l|ccc}
 & $2$ & $5$ & $9$ \\
\hline
 & $6$ & $15$ & $28$ \\
\end{tabular}
\end{center}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:2}
Vul in met de kruisregel, en schrijf de berekening:
$5$ kg aardappelen kosten $6$ euro; hoeveel kosten $8$ kg?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:3}
Een printer print $36$ pagina's in $3$ minuten. Met dezelfde
snelheid, hoeveel pagina's in $5$ minuten? Hoe lang voor $96$
pagina's?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:4}
Bereken: $30\,\%$ van $250$; \quad $15\,\%$ van $60$; \quad $t\,\%$
zodat $t\,\%$ van $80$ gelijk is aan $20$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:5}
Van $25$ schoten scoorde een basketbalspeler $18$. Wat is haar
succespercentage? (Herschrijf de \omterm{def:g6:fractions:def}{breuk} over $100$.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:6}
Op een kaart $1 : 25\,000$ meet een pad $9$ cm. Wat is zijn echte
lengte in km? Een meer is $2$ km lang: hoe lang is het op de kaart?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:7}
Een modelauto is gebouwd op \omterm{def:g7:prop:scale}{schaal} $1 : 43$. De echte auto is $4.3$ m
lang. Hoe lang is het model, in cm?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:prop:8}
Een recept gebruikt $240$ g chocolade voor $8$ porties. Aline heeft
$300$ g chocolade. Voor hoeveel porties is dat genoeg (geheel aantal
porties)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:prop:9}
De prijs van een jas daalt van $80$ naar $60$ euro.
\begin{enumerate}
  \item Wat is de korting in euro? In procent van de oorspronkelijke
        prijs?
  \item Later stijgt de prijs terug van $60$ naar $80$ euro. Leg uit
        waarom deze stijging \emph{niet} $25\,\%$ is --- bereken het
        juiste stijgingspercentage.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:prop:10}
Twee rijen van een tabel zijn \omterm{def:g6:propdata:def}{evenredig}:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|ccc}
 & $4$ & $x$ & $10$ \\
\hline
 & $y$ & $10.5$ & $17.5$ \\
\end{tabular}
\end{center}
Vind de coëfficiënt, daarna $x$ en $y$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:prop:11}
Een fotokopieerapparaat verkleint documenten tot $80\,\%$ van hun
grootte. Een \omterm{def:g6:lines:objects}{lijnstuk} van $10$ cm wordt gekopieerd, daarna wordt de
\emph{kopie} opnieuw gekopieerd met dezelfde instelling.
\begin{enumerate}
  \item Hoe lang is het \omterm{def:g6:lines:objects}{lijnstuk} na één kopie? Na twee?
  \item Waarom is het antwoord na twee kopieën niet $60\,\%$ van het
        origineel? Welk enkel percentage komt overeen met twee
        kopieën?
\end{enumerate}
\end{exercise}
