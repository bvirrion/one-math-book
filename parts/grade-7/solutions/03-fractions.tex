\begin{solution}{exo:g7:fractions:4}

\begin{solution}{exo:g7:fractions:5}

\begin{solution}{exo:g7:fractions:10}

\text{ L}.
\]
\end{solution}

\begin{solution}{exo:g7:fractions:11}
Adding the positive quantity $\frac12$ to $\frac35$ must give a result
\emph{larger} than $\frac35$. But $\frac47 < \frac35$ (indeed
$\frac47 = \frac{20}{35}$ and $\frac35 = \frac{21}{35}$): the claimed
\omterm{def:g1:addition:def}{sum} is smaller than one of its terms --- impossible.
\end{solution}

\begin{solution}{exo:g7:fractions:12}
Step by step:
\[
\frac12 + \frac14 = \frac34, \qquad
\frac34 + \frac18 = \frac68 + \frac18 = \frac78, \qquad
\frac78 + \frac1{16} = \frac{14}{16} + \frac1{16} = \frac{15}{16}.
\]
The partial \omterm{def:g1:addition:def}{sums} $\frac12, \frac34, \frac78, \frac{15}{16}, \dots$ are
each time half-way closer to $1$: at every step, the missing part is
cut in \omterm{def:g3:division:half}{half}. The \omterm{def:g1:addition:def}{sums} approach $1$ without ever reaching it.
\end{solution}
