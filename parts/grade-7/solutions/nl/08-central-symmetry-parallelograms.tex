\section*{Hoofdstuk \ref{ch:g7:central} --- Centrale spiegeling en parallellogrammen}

\begin{solution}{exo:g7:central:1}
Halfslagbeeld: $M'$ ligt vier hokjes \emph{links} van $O$ en ├®├®n
hokje \emph{omlaag} (beide richtingen omgekeerd). \omterm{def:g7:central:def}{Spiegelbeeld} ten
opzichte van de verticale lijn: vier hokjes links maar nog ├®├®n hokje
\emph{omhoog}. De twee beelden verschillen: ze zijn elkaars
verticale \omterm{def:g7:central:def}{spiegelbeelden}.
\end{solution}

\begin{solution}{exo:g7:central:2}
$A'B' = AB$ (halve slagen behouden lengten) en
$(A'B') \parallel (AB)$ (het beeld van een \omterm{def:g6:lines:objects}{rechte} is een evenwijdige
\omterm{def:g6:lines:objects}{rechte}, \cref{prop:g7:central:preserved}).
\end{solution}

\begin{solution}{exo:g7:central:3}
Rekenmachinecijfers met een \omterm{def:g7:central:center}{symmetriecentrum}: $0$, $2$, $5$, $8$ (en
$1$ als die als een kale streep getekend is). Letters: H, N, S, Z, O
hebben er ├®├®n; A en T niet (die hebben een as in plaats daarvan).
\end{solution}

\begin{solution}{exo:g7:central:4}
Overstaande zijden gelijk: $MN = KL = 7$ cm en $NK = LM = 4$ cm.
\omterm{def:g7:triangles:pairs}{Overstaande hoeken} gelijk: $\widehat M = \widehat K = 115^\circ$.
De diagonalen snijden elkaar in hun gemeenschappelijk midden, en dat
punt is $O$: het midden van $[LN]$ is $O$.
\end{solution}

\begin{solution}{exo:g7:central:5}
Teken een \omterm{def:g6:lines:objects}{lijnstuk} $[AC]$ van $8$ cm, markeer zijn midden $O$, teken
door $O$ een lijn die $60^\circ$ maakt met $(AC)$, en plaats $B$ en
$D$ erop op $2.5$ cm van $O$ aan weerszijden. Verbind $A$, $B$, $C$,
$D$: de diagonalen snijden in hun middens, dus $ABCD$ is een
\omterm{def:g7:central:parallelogram}{parallellogram} per definitie.
\end{solution}

\begin{solution}{exo:g7:central:6}
Opeenvolgende hoeken zijn supplementair: naast $115^\circ$ zit
$180 - 115 = 65^\circ$. \omterm{def:g7:triangles:pairs}{Overstaande hoeken} zijn gelijk, dus de vier
hoeken zijn $115^\circ$, $65^\circ$, $115^\circ$, $65^\circ$ (\omterm{def:g1:addition:def}{som}
$360^\circ$).
\end{solution}

\begin{solution}{exo:g7:central:7}
Van $A(1,1)$ naar $O(2.5, 2.5)$: $1.5$ rechts en $1.5$ omhoog;
doorgaan met hetzelfde: $A'(4, 4)$. Van $B(4,2)$ naar $O$: $1.5$
links en $0.5$ omhoog; doorgaan: $B'(1, 3)$.
\end{solution}

\begin{solution}{exo:g7:central:8}
Nee. Een \omterm{def:g6:shapes:triangles}{gelijkbenig} trapezium heeft zijn twee schuine zijden gelijk
zonder een \omterm{def:g7:central:parallelogram}{parallellogram} te zijn (de twee evenwijdige zijden hebben
verschillende lengten). Wat volstaat is criterium 3 van
\cref{thm:g7:central:recognize}: twee overstaande zijden gelijk
\emph{en \omterm{def:g6:lines:perp}{evenwijdig}}.
\end{solution}

\begin{solution}{exo:g7:central:9}
$O$ is het midden van $[BC]$ per keuze, en het is het midden van
$[AA']$ door constructie van het \omterm{def:g7:central:def}{spiegelbeeld}. De \omterm{def:g4:geometry:polygon}{vierhoek} $ABA'C$
heeft diagonalen $[AA']$ en $[BC]$ die het midden $O$ delen:
criterium 1 (de definitie) maakt het gratis een \omterm{def:g7:central:parallelogram}{parallellogram}.
\end{solution}

\begin{solution}{exo:g7:central:10}
``\omterm{def:g7:triangles:pairs}{Overstaande hoeken} gelijk $\Rightarrow$ \omterm{def:g7:central:parallelogram}{parallellogram}'':
\emph{waar} (dit is een ander herkenningscriterium; met de hoekensom
$360^\circ$ dwingen gelijke overstaande paren supplementaire
opeenvolgende hoeken, dus evenwijdige zijden door
\cref{thm:g7:triangles:equalities}).

``Loodrechte diagonalen $\Rightarrow$ \omterm{def:g6:shapes:quadrilaterals}{ruit}'': \emph{waar} voor een
\omterm{def:g7:central:parallelogram}{parallellogram} (de diagonalen snijden al in hun middens).

``E├®n \omterm{def:g3:shapes:rightangle}{rechte hoek} $\Rightarrow$ rechthoek'': \emph{waar} ---
opeenvolgende hoeken zijn supplementair, dus alle vier worden recht.
\end{solution}

\begin{solution}{exo:g7:central:11}
Welke \omterm{def:g4:geometry:polygon}{vierhoek} ook --- zelfs een zeer onregelmatige --- de middens
$I$, $J$, $K$, $L$ vormen altijd een \emph{\omterm{def:g7:central:parallelogram}{parallellogram}}. Het
bewijs met de middenparallellenstelling staat in
\cref{ch:g8:midpoints} (\cref{exo:g8:midpoints:9}).
\end{solution}
