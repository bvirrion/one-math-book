\begin{solution}{exo:g7:fractions:4}

\end{solution}

\begin{solution}{exo:g7:fractions:5}

\end{solution}

\begin{solution}{exo:g7:fractions:10}

\]
\end{solution}

\begin{solution}{exo:g7:fractions:11}
Iets positiefs $\frac12$ bij $\frac35$ optellen moet een resultaat
\emph{groter} dan $\frac35$ geven. Maar $\frac47 < \frac35$
(inderdaad $\frac47 = \frac{20}{35}$ en $\frac35 = \frac{21}{35}$):
de beweerde \omterm{def:g1:addition:def}{som} is kleiner dan één van haar termen --- onmogelijk.
\end{solution}

\begin{solution}{exo:g7:fractions:12}
Stap voor stap:
\[
\frac12 + \frac14 = \frac34, \qquad
\frac34 + \frac18 = \frac68 + \frac18 = \frac78, \qquad
\frac78 + \frac1{16} = \frac{14}{16} + \frac1{16} = \frac{15}{16}.
\]
De deelsommen $\frac12, \frac34, \frac78, \frac{15}{16}, \dots$ komen
elke keer half zo dicht bij $1$: bij elke stap wordt het ontbrekende
deel gehalveerd. De sommen naderen $1$ zonder hem ooit te bereiken.
\end{solution}
