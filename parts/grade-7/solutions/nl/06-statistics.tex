\section*{Hoofdstuk \ref{ch:g7:stats} --- Gegevens organiseren}

\begin{solution}{exo:g7:stats:1}
Elk cijfer tellen:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|ccccc|c}
cijfer & $9$ & $11$ & $12$ & $14$ & $15$ & totaal \\
\hline
\omterm{def:g7:stats:frequency}{telling} & $4$ & $3$ & $7$ & $3$ & $3$ & $20$ \\
\end{tabular}
\end{center}
\end{solution}

\begin{solution}{exo:g7:stats:2}
\omterm{def:g7:stats:frequency}{Frequenties}: $\frac{4}{20} = 20\,\%$, $\frac{3}{20} = 15\,\%$,
$\frac{7}{20} = 35\,\%$, $15\,\%$, $15\,\%$. \omterm{def:g1:addition:def}{Som}:
$20 + 15 + 35 + 15 + 15 = 100\,\%$. \checkmark
\end{solution}

\begin{solution}{exo:g7:stats:3}
Eerste enquête: $\frac{30}{50} = 0.6 = 60\,\%$. Tweede:
$\frac{45}{90} = 0.5 = 50\,\%$. De eerste enquête toont het hogere
aandeel, hoewel ze minder gezinnen met één auto in absolute \omterm{def:g7:stats:frequency}{telling}
heeft.
\end{solution}

\begin{solution}{exo:g7:stats:4}
$\intco{5}{15}$: $8, 12, 9, 14, 11$ --- \omterm{def:g7:stats:frequency}{telling} $5$.
$\intco{15}{25}$: $23, 22, 18, 16$ --- \omterm{def:g7:stats:frequency}{telling} $4$.
$\intco{25}{35}$: $31, 27, 29, 25, 33$ --- \omterm{def:g7:stats:frequency}{telling} $5$.
$\intco{35}{45}$: $35$ --- \omterm{def:g7:stats:frequency}{telling} $1$. Totaal $15$. \checkmark
\end{solution}

\begin{solution}{exo:g7:stats:5}
Vier staven van hoogten $8$, $12$, $6$, $4$ boven de waarden $0$,
$1$, $2$, $3$ (schaalverdeling om de $2$ werkt goed).
\end{solution}

\begin{solution}{exo:g7:stats:6}
Totaal $30$ gezinnen. \omterm{def:g7:stats:frequency}{Frequenties}: $\frac{8}{30}$, $\frac{12}{30}$,
$\frac{6}{30}$, $\frac{4}{30}$. Hoeken ($\times 360^\circ$):
$96^\circ$, $144^\circ$, $72^\circ$, $48^\circ$; \omterm{def:g1:addition:def}{som}
$96 + 144 + 72 + 48 = 360^\circ$. \checkmark
\end{solution}

\begin{solution}{exo:g7:stats:7}
Percentages: zomer $\frac{162}{360} = 45\,\%$; lente
$\frac{90}{360} = 25\,\%$; herfst en winter
$\frac{54}{360} = 15\,\%$ elk. Van de $60$ personen koos de zomer
$0.45 \times 60 = 27$ van hen.
\end{solution}

\begin{solution}{exo:g7:stats:8}
Klas A: $\frac{12}{30} = 40\,\%$. Klas B: $\frac{14}{40} = 35\,\%$.
Klas A heeft het hogere aandeel lopenden, hoewel klas B meer lopenden
in \omterm{def:g7:stats:frequency}{telling} heeft.
\end{solution}

\begin{solution}{exo:g7:stats:9}
Met de as beginnend bij $9\,700$ hebben de drie staven zichtbare
hoogten $100$, $300$ en $400$ eenheden: de laatste lijkt vier keer de
eerste, wat suggereert dat de verkopen verviervoudigden. Met een as
vanaf $0$ zijn de staven bijna gelijk ($9\,800$ tot $10\,100$ is een
stijging van ongeveer $3\,\%$): het eerlijke beeld toont bijna vlakke
verkopen.
\end{solution}

\begin{solution}{exo:g7:stats:10}
\omterm{def:g7:stats:frequency}{Frequenties} sommeren tot $1$: $f = 1 - 0.35 - 0.4 = 0.25$. \omterm{def:g7:stats:frequency}{Tellingen}
(totaal $80$): $0.35 \times 80 = 28$; $0.4 \times 80 = 32$;
$0.25 \times 80 = 20$. Controle: $28 + 32 + 20 = 80$.
\end{solution}

\begin{solution}{exo:g7:stats:11}
Meisjes: $55\,\%$ van $400 = 220$; jongens: $180$. Kantine-meisjes:
$0.4 \times 220 = 88$; jongens: $0.6 \times 180 = 108$. Totaal:
$88 + 108 = 196$ leerlingen, d.w.z.\ $\frac{196}{400} = 49\,\%$ van
de school --- tussen $40\,\%$ en $60\,\%$, dichter bij het meisjes-
percentage omdat meisjes talrijker zijn.
\end{solution}
