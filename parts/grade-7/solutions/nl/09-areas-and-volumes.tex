\section*{Hoofdstuk \ref{ch:g7:areas} --- Oppervlakten en volumes}

\begin{solution}{exo:g7:areas:1}
$8 \times 5 = 40$ cm$^2$; \qquad $6.5 \times 4 = 26$ m$^2$.
\end{solution}

\begin{solution}{exo:g7:areas:2}
$A = 10 \times 4 = 40$ cm$^2$. Het antwoord is niet
$10 \times 6 = 60$ omdat de zijde van $6$ cm scheef is: de formule
gebruikt de \emph{hoogte} (loodrechte afstand), die $4$ cm is.
\end{solution}

\begin{solution}{exo:g7:areas:3}
$\frac{12 \times 7}{2} = 42$ cm$^2$; \quad
$\frac{9 \times 6}{2} = 27$; \quad
$\frac{5.5 \times 2}{2} = 5.5$ m$^2$.
\end{solution}

\begin{solution}{exo:g7:areas:4}
Schijf: $\pi \times 5^2 = 25\pi \approx 79$ cm$^2$. Halfschijf met
\omterm{def:g3:shapes:shapes}{straal} $10$: $\frac{\pi \times 10^2}{2} = 50\pi \approx 157$ cm$^2$.
\end{solution}

\begin{solution}{exo:g7:areas:5}
\omterm{def:g6:measure:perimeter}{Omtrek}: $2\pi \times 3 = 6\pi \approx 18.8$ cm (een lengte, in cm).
\omterm{def:g6:measure:area}{Oppervlakte}: $\pi \times 3^2 = 9\pi \approx 28.3$ cm$^2$ (een
\omterm{def:g6:measure:area}{oppervlakte}, in cm$^2$).
\end{solution}

\begin{solution}{exo:g7:areas:6}
\omterm{def:g7:areas:prism}{Prisma}: $24 \times 10 = 240$ cm$^3$. \omterm{def:g7:areas:prism}{Cilinder}:
$\pi \times 3^2 \times 8 = 72\pi \approx 226$ cm$^3$.
\end{solution}

\begin{solution}{exo:g7:areas:7}
$\frac{8 \times h}{2} = 28$, dus $4h = 28$ en $h = 7$ cm.
\end{solution}

\begin{solution}{exo:g7:areas:8}
De drie \omterm{def:g2:mult:def}{producten} komen overeen omdat elk de \emph{zelfde} grootheid
berekent --- de \omterm{def:g6:measure:area}{oppervlakte} van de driehoek. Dit geeft een
praktische controle van constructies, en een manier om een hoogte te
berekenen uit een ander basis--hoogte-paar (gebruikt in
\cref{exo:g7:areas:12}).
\end{solution}

\begin{solution}{exo:g7:areas:9}
De twee driehoeken hebben \omterm{def:g6:measure:area}{oppervlakten} $\frac{30 \times 20}{2} =
300$ m$^2$ en $\frac{18 \times 20}{2} = 180$ m$^2$: totaal
$480$ m$^2$. Voorgestelde algemene formule: voor evenwijdige zijden
$a$ en $b$ op afstand $h$,
\[
A = \frac{(a + b) \times h}{2}
\]
--- de trapeziumformule (hier $\frac{(30+18)\times 20}{2} = 480$).
\end{solution}

\begin{solution}{exo:g7:areas:10}
Watervolume: $\pi \times 6^2 \times 15 = 540\pi \approx 1\,696$
cm$^3$. Bakbasis: $18 \times 12 = 216$ cm$^2$. Bereikte hoogte:
$\frac{540\pi}{216} = 2.5\pi \approx 7.9$ cm.
\end{solution}

\begin{solution}{exo:g7:areas:11}
Stralen $15$ en $20$ cm. \omterm{def:g6:measure:area}{Oppervlakten}: $225\pi \approx 707$ cm$^2$ en
$400\pi \approx 1257$ cm$^2$. Prijs per $100$ cm$^2$:
$\frac{9}{7.07} \approx 1.27$ euro en
$\frac{14}{12.57} \approx 1.11$ euro. De grote pizza is de betere
deal --- \omterm{def:g6:measure:area}{oppervlakten} groeien met het \emph{kwadraat} van de
\omterm{def:g4:geometry:circle}{diameter}.
\end{solution}

\begin{solution}{exo:g7:areas:12}
Met de benen: $A = \frac{6 \times 8}{2} = 24$ cm$^2$. Met de
schuine zijde als basis: $24 = \frac{10 \times h}{2} = 5h$, dus
$h = 4.8$ cm.
\end{solution}
