\section*{Hoofdstuk \ref{ch:g7:negatives} --- Negatieve getallen}

\begin{solution}{exo:g7:negatives:1}
\omterm{def:g7:negatives:def}{Tegengestelden}: $-7$; \ $+3.5$; \ $0$ (zijn eigen \omterm{def:g7:negatives:def}{tegengestelde});
\ $-12$. Twee getallen zijn tegengesteld wanneer hun \omterm{def:g1:addition:def}{som} $0$ is.
\end{solution}

\begin{solution}{exo:g7:negatives:2}
$-4 < 1$; \quad $-4 > -7$; \quad $-2.5 < -2.4$; \quad $0 > -0.1$.
\end{solution}

\begin{solution}{exo:g7:negatives:3}
$-3.1 < -3 < -0.5 < 0 < 1 < 2.5$.
\end{solution}

\begin{solution}{exo:g7:negatives:4}
$(-5) + (-9) = -14$; \quad
$(-12) + (+7) = -5$; \quad
$(+3.5) + (-1.5) = +2$; \quad
$(-6) + (+6) = 0$.
\end{solution}

\begin{solution}{exo:g7:negatives:5}
$4 - 11 = 4 + (-11) = -7$.

$(-5) - (+3) = (-5) + (-3) = -8$.

$(-7) - (-10) = (-7) + (+10) = +3$.

$0 - (-8) = 0 + (+8) = +8$.
\end{solution}

\begin{solution}{exo:g7:negatives:6}
$(+7) + (-12) + (+3) + (-8) = (+10) + (-20) = -10$ (positieven
samen, negatieven samen).

$(-5.5) + (+9) + (-4.5) + (+1) = (-10) + (+10) = 0$.
\end{solution}

\begin{solution}{exo:g7:negatives:7}
$(-3) + 8 - 12 = 5 - 12 = -7$: het is $-7$ graden 's nachts.
\end{solution}

\begin{solution}{exo:g7:negatives:8}
Vrije tekening; $E(0, -3)$ ligt op de verticale as (zijn eerste
coördinaat is $0$).
\end{solution}

\begin{solution}{exo:g7:negatives:9}
Moskou: $(-4) - (-13) = -4 + 13 = 9$ graden \omterm{ex:g1:subtraction:difference}{verschil}.
Helsinki: $2 - (-9) = 11$ graden.
\end{solution}

\begin{solution}{exo:g7:negatives:10}
$(-12) + 7 - 4 = -5 - 4 = -9$: ze is op $-9$ m. Om het oppervlak
(hoogte $0$) te bereiken moet ze $0 - (-9) = 9$ m stijgen.
\end{solution}

\begin{solution}{exo:g7:negatives:11}
$D(-2, -2)$: zelfde eerste coördinaat als $A$, zelfde tweede
coördinaat als $C$. Zijden: $AB = 4$ (van $-2$ tot $2$) en $BC = 3$
(van $1$ omlaag tot $-2$), dus de \omterm{def:g6:measure:perimeter}{omtrek} is $2 \times (4 + 3) = 14$.
\end{solution}

\begin{solution}{exo:g7:negatives:12}
De complete diagonaal $0, -2, -4$ geeft de magische \omterm{def:g1:addition:def}{som}
$0 + (-2) + (-4) = -6$. Dan, cel voor cel:
linksboven $= -6 - (-7) - 0 = 1$; midden-rechts $= -6 - (-3) - (-2) =
-1$; rechtsonder $= -6 - 0 - (-1) = -5$; midden-onder
$= -6 - (-7) - (-2) = 3$.
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|c|c|}
\hline
$1$ & $-7$ & $0$ \\
\hline
$-3$ & $-2$ & $-1$ \\
\hline
$-4$ & $3$ & $-5$ \\
\hline
\end{tabular}
\end{center}
Elke rij, kolom en diagonaal somt tot $-6$.
\end{solution}
