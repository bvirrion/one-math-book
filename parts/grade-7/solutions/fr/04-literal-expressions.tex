\begin{solution}{exo:g7:literal:1}
$7x$~; \quad $3y$~; \quad $5ab$~; \quad $x^2$~; \quad $2(a + 4)$.
\end{solution}

\begin{solution}{exo:g7:literal:2}
$5ab = 5 \times a \times b$~; \quad
$3(x+2) = 3 \times (x + 2)$~; \quad
$x^2 + 4x = x \times x + 4 \times x$~; \quad
$2a^3 = 2 \times a \times a \times a$.
\end{solution}

\begin{solution}{exo:g7:literal:3}
Pour $x = 3$~: \ $5 \times 3 + 1 = 16$~; \quad $3 \times 3 = 9$~; \quad
$2 \times 9 - 3 = 15$~; \quad $10 - 6 = 4$.
\end{solution}

\begin{solution}{exo:g7:literal:4}
$3a + 2b = 12 + 3 = 15$~; \qquad
$ab + 5 = 4 \times 1.5 + 5 = 6 + 5 = 11$.
\end{solution}

\begin{solution}{exo:g7:literal:5}

$9a - 4a + a = 6a$~; \quad

\begin{solution}{exo:g7:literal:7}
$2n$~: le double~; \ $n + 2$~: deux de plus que $n$~; \ $n^2$~: le
carré~; \ $\frac n2$~: la \omterm{def:g3:division:half}{moitié}.
\end{solution}

\begin{solution}{exo:g7:literal:8}
$3(x+4) = 3x + 12$~: \emph{vrai} pour tout $x$ (distributivité ---
c'est une preuve, pas besoin de test).

$2x + 5 = 7x$~: tester $x = 2$~: gauche $9$, droite $14$ ---
\emph{faux} en général (cela ne vaut que pour $x = 1$).

$x + x = x^2$~: tester $x = 3$~: gauche $6$, droite $9$ ---
\emph{faux} en général ($x + x = 2x$).
\end{solution}

\begin{solution}{exo:g7:literal:9}
\emph{1.} Longueur~: $w + 3$. \omterm{def:g6:measure:perimeter}{Périmètre}~:
\[
P = 2\bigl(w + (w + 3)\bigr) = 2(2w + 3) = 4w + 6 .
\]

\emph{2.} Pour $w = 5$~: directement, les côtés mesurent $5$ et $8$,
donc $P = 2 \times 13 = 26$ cm~; avec la formule,
$4 \times 5 + 6 = 26$ cm. Elles coïncident.
\end{solution}

\begin{solution}{exo:g7:literal:10}
En suivant les étapes avec la lettre $n$~:
\[
n \ \to\ n + 6 \ \to\ 3(n + 6) = 3n + 18 \ \to\ 3n + 18 - 18 = 3n
\ \to\ \frac{3n}{n} = 3 .
\]
Tout le monde obtient $3$.
\end{solution}

\begin{solution}{exo:g7:literal:11}
Comme \omterm{ex:g1:subtraction:difference}{différence}~: $A = a^2 - b^2$. En découpant le L en deux
rectangles~: une bande de largeur $a$ et de hauteur $a - b$, puis un
rectangle plus étroit au-dessus~:
\[
A = a(a - b) + b(a - b)
\]
(rectangle du bas large de $a$ et haut de $a - b$, rectangle du haut
large de $a - b$ et haut de $b$ donne la variante
$a(a-b) + (a-b)b$ --- la même). Pour $a = 5$, $b = 2$~:
$a^2 - b^2 = 25 - 4 = 21$, et $5 \times 3 + 2 \times 3 = 15 + 6 = 21$.
Les deux expressions coïncident --- elles le font toujours~: c'est
l'identité $a^2 - b^2 = (a + b)(a - b)$ du \cref{ch:g9:algebra} en
cachette.
\end{solution}
