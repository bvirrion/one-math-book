\section*{Chapitre \ref{ch:g7:negatives} --- Nombres relatifs}

\begin{solution}{exo:g7:negatives:1}
Opposés~: $-7$~; \ $+3.5$~; \ $0$ (son propre opposé)~; \ $-12$. Deux
nombres sont opposés lorsque leur \omterm{def:g1:addition:def}{somme} vaut $0$.
\end{solution}

\begin{solution}{exo:g7:negatives:2}
$-4 < 1$~; \quad $-4 > -7$~; \quad $-2.5 < -2.4$~; \quad $0 > -0.1$.
\end{solution}

\begin{solution}{exo:g7:negatives:3}
$-3.1 < -3 < -0.5 < 0 < 1 < 2.5$.
\end{solution}

\begin{solution}{exo:g7:negatives:4}
$(-5) + (-9) = -14$~; \quad
$(-12) + (+7) = -5$~; \quad
$(+3.5) + (-1.5) = +2$~; \quad
$(-6) + (+6) = 0$.
\end{solution}

\begin{solution}{exo:g7:negatives:5}
$4 - 11 = 4 + (-11) = -7$.

$(-5) - (+3) = (-5) + (-3) = -8$.

$(-7) - (-10) = (-7) + (+10) = +3$.

$0 - (-8) = 0 + (+8) = +8$.
\end{solution}

\begin{solution}{exo:g7:negatives:6}
$(+7) + (-12) + (+3) + (-8) = (+10) + (-20) = -10$ (positifs
ensemble, négatifs ensemble).

$(-5.5) + (+9) + (-4.5) + (+1) = (-10) + (+10) = 0$.
\end{solution}

\begin{solution}{exo:g7:negatives:7}
$(-3) + 8 - 12 = 5 - 12 = -7$~: il fait $-7$ degrés la nuit.
\end{solution}

\begin{solution}{exo:g7:negatives:8}
Tracé libre~; $E(0, -3)$ est sur l'axe vertical (sa première
coordonnée est $0$).
\end{solution}

\begin{solution}{exo:g7:negatives:9}
Moscou~: $(-4) - (-13) = -4 + 13 = 9$ degrés d'écart.
Helsinki~: $2 - (-9) = 11$ degrés.
\end{solution}

\begin{solution}{exo:g7:negatives:10}
$(-12) + 7 - 4 = -5 - 4 = -9$~: elle est à $-9$ m. Pour atteindre la
surface (altitude $0$) elle doit remonter $0 - (-9) = 9$ m.
\end{solution}

\begin{solution}{exo:g7:negatives:11}
$D(-2, -2)$~: même première coordonnée que $A$, même seconde
coordonnée que $C$. Côtés~: $AB = 4$ (de $-2$ à $2$) et $BC = 3$ (de
$1$ jusqu'à $-2$), donc le \omterm{def:g6:measure:perimeter}{périmètre} est $2 \times (4 + 3) = 14$.
\end{solution}

\begin{solution}{exo:g7:negatives:12}
La diagonale complète $0$, $-2$, $-4$ donne la \omterm{def:g1:addition:def}{somme} magique
$0 + (-2) + (-4) = -6$. Puis, case par case~:
en haut à gauche $= -6 - (-7) - 0 = 1$~; au milieu à droite
$= -6 - (-3) - (-2) = -1$~; en bas à droite $= -6 - 0 - (-1) = -5$~;
en bas au milieu $= -6 - (-7) - (-2) = 3$.
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|c|c|}
\hline
$1$ & $-7$ & $0$ \\
\hline
$-3$ & $-2$ & $-1$ \\
\hline
$-4$ & $3$ & $-5$ \\
\hline
\end{tabular}
\end{center}
Chaque ligne, colonne et diagonale a pour \omterm{def:g1:addition:def}{somme} $-6$.
\end{solution}
