\section*{Chapitre \ref{ch:g7:stats} --- Organiser des données}

\begin{solution}{exo:g7:stats:1}
En comptant chaque note~:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|ccccc|c}
note & $9$ & $11$ & $12$ & $14$ & $15$ & total \\
\hline
\omterm{def:g7:stats:frequency}{effectif} & $4$ & $3$ & $7$ & $3$ & $3$ & $20$ \\
\end{tabular}
\end{center}
\end{solution}

\begin{solution}{exo:g7:stats:2}
\omterm{def:g7:stats:frequency}{Fréquences}~: $\frac{4}{20} = 20\,\%$, $\frac{3}{20} = 15\,\%$,
$\frac{7}{20} = 35\,\%$, $15\,\%$, $15\,\%$. \omterm{def:g1:addition:def}{Somme}~:
$20 + 15 + 35 + 15 + 15 = 100\,\%$. \checkmark
\end{solution}

\begin{solution}{exo:g7:stats:3}
Première enquête~: $\frac{30}{50} = 0.6 = 60\,\%$. Seconde~:
$\frac{45}{90} = 0.5 = 50\,\%$. La première enquête montre la
proportion la plus élevée, bien qu'elle ait moins de familles avec
une voiture en \omterm{def:g7:stats:frequency}{effectif} absolu.
\end{solution}

\begin{solution}{exo:g7:stats:4}
$\intco{5}{15}$~: $8, 12, 9, 14, 11$ --- \omterm{def:g7:stats:frequency}{effectif} $5$.
$\intco{15}{25}$~: $23, 22, 18, 16$ --- \omterm{def:g7:stats:frequency}{effectif} $4$.
$\intco{25}{35}$~: $31, 27, 29, 25, 33$ --- \omterm{def:g7:stats:frequency}{effectif} $5$.
$\intco{35}{45}$~: $35$ --- \omterm{def:g7:stats:frequency}{effectif} $1$. Total $15$. \checkmark
\end{solution}

\begin{solution}{exo:g7:stats:5}
Quatre barres de hauteurs $8$, $12$, $6$, $4$ au-dessus des valeurs
$0$, $1$, $2$, $3$ (graduation de $2$ en $2$ convient bien).
\end{solution}

\begin{solution}{exo:g7:stats:6}
Total $30$ familles. \omterm{def:g7:stats:frequency}{Fréquences}~: $\frac{8}{30}$, $\frac{12}{30}$,
$\frac{6}{30}$, $\frac{4}{30}$. Angles ($\times 360^\circ$)~:
$96^\circ$, $144^\circ$, $72^\circ$, $48^\circ$~; \omterm{def:g1:addition:def}{somme}
$96 + 144 + 72 + 48 = 360^\circ$. \checkmark
\end{solution}

\begin{solution}{exo:g7:stats:7}
Pourcentages~: été $\frac{162}{360} = 45\,\%$~; printemps
$\frac{90}{360} = 25\,\%$~; automne et hiver
$\frac{54}{360} = 15\,\%$ chacun. Sur $60$ personnes, l'été a été
choisi par $0.45 \times 60 = 27$ d'entre elles.
\end{solution}

\begin{solution}{exo:g7:stats:8}
Classe A~: $\frac{12}{30} = 40\,\%$. Classe B~: $\frac{14}{40} =
35\,\%$. La classe A a la proportion de piétons la plus élevée, même
si la classe B a plus de piétons en \omterm{def:g7:stats:frequency}{effectif}.
\end{solution}

\begin{solution}{exo:g7:stats:9}
Avec l'axe commençant à $9\,700$, les trois barres ont des hauteurs
visibles de $100$, $300$ et $400$ unités~: la dernière paraît quatre
fois la première, suggérant que les ventes ont quadruplé. Avec un axe
partant de $0$, les barres sont presque égales ($9\,800$ à $10\,100$
est une hausse d'environ $3\,\%$)~: l'image honnête montre des ventes
presque plates.
\end{solution}

\begin{solution}{exo:g7:stats:10}
Les \omterm{def:g7:stats:frequency}{fréquences} somment à $1$~: $f = 1 - 0.35 - 0.4 = 0.25$.
\omterm{def:g7:stats:frequency}{Effectifs} (total $80$)~: $0.35 \times 80 = 28$~; $0.4 \times 80 =
32$~; $0.25 \times 80 = 20$. Vérification~: $28 + 32 + 20 = 80$.
\end{solution}

\begin{solution}{exo:g7:stats:11}
Filles~: $55\,\%$ de $400 = 220$~; garçons~: $180$. Filles à la
cantine~: $0.4 \times 220 = 88$~; garçons~: $0.6 \times 180 = 108$.
Total~: $88 + 108 = 196$ élèves, soit $\frac{196}{400} = 49\,\%$ de
l'établissement --- entre $40\,\%$ et $60\,\%$, plus proche du taux
des filles car elles sont plus nombreuses.
\end{solution}
