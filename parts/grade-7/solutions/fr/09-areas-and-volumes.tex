\section*{Chapitre \ref{ch:g7:areas} --- Aires et volumes}

\begin{solution}{exo:g7:areas:1}
$8 \times 5 = 40$ cm$^2$~; \qquad $6.5 \times 4 = 26$ m$^2$.
\end{solution}

\begin{solution}{exo:g7:areas:2}
$A = 10 \times 4 = 40$ cm$^2$. La réponse n'est pas
$10 \times 6 = 60$ car le côté de $6$ cm est oblique~: la formule
utilise la \emph{hauteur} (distance perpendiculaire), qui vaut $4$ cm.
\end{solution}

\begin{solution}{exo:g7:areas:3}
$\frac{12 \times 7}{2} = 42$ cm$^2$~; \quad
$\frac{9 \times 6}{2} = 27$~; \quad
$\frac{5.5 \times 2}{2} = 5.5$ m$^2$.
\end{solution}

\begin{solution}{exo:g7:areas:4}
Disque~: $\pi \times 5^2 = 25\pi \approx 79$ cm$^2$. Demi-disque de
rayon $10$~: $\frac{\pi \times 10^2}{2} = 50\pi \approx 157$ cm$^2$.
\end{solution}

\begin{solution}{exo:g7:areas:5}
Périmètre~: $2\pi \times 3 = 6\pi \approx 18.8$ cm (une longueur, en
cm). Aire~: $\pi \times 3^2 = 9\pi \approx 28.3$ cm$^2$ (une surface,
en cm$^2$).
\end{solution}

\begin{solution}{exo:g7:areas:6}
Prisme~: $24 \times 10 = 240$ cm$^3$. Cylindre~:
$\pi \times 3^2 \times 8 = 72\pi \approx 226$ cm$^3$.
\end{solution}

\begin{solution}{exo:g7:areas:7}
$\frac{8 \times h}{2} = 28$, donc $4h = 28$ et $h = 7$ cm.
\end{solution}

\begin{solution}{exo:g7:areas:8}
Les trois produits coïncident car chacun calcule la \emph{même}
quantité --- l'aire du triangle. Cela donne un contrôle pratique des
constructions, et un moyen de calculer une hauteur à partir d'une
autre paire base--hauteur (utilisé en \cref{exo:g7:areas:12}).
\end{solution}

\begin{solution}{exo:g7:areas:9}
Les deux triangles ont pour aires $\frac{30 \times 20}{2} = 300$
m$^2$ et $\frac{18 \times 20}{2} = 180$ m$^2$~: total $480$ m$^2$.
Formule générale suggérée~: pour des côtés parallèles $a$ et $b$ à
distance $h$,
\[
A = \frac{(a + b) \times h}{2}
\]
--- la formule du trapèze (ici $\frac{(30+18)\times 20}{2} = 480$).
\end{solution}

\begin{solution}{exo:g7:areas:10}
Volume d'eau~: $\pi \times 6^2 \times 15 = 540\pi \approx 1\,696$
cm$^3$. Base de la boîte~: $18 \times 12 = 216$ cm$^2$. Hauteur
atteinte~: $\frac{540\pi}{216} = 2.5\pi \approx 7.9$ cm.
\end{solution}

\begin{solution}{exo:g7:areas:11}
Rayons $15$ et $20$ cm. Aires~: $225\pi \approx 707$ cm$^2$ et
$400\pi \approx 1257$ cm$^2$. Prix pour $100$ cm$^2$~:
$\frac{9}{7.07} \approx 1.27$ euros et
$\frac{14}{12.57} \approx 1.11$ euros. La grande pizza est la
meilleure affaire --- les aires croissent avec le \emph{carré} du
diamètre.
\end{solution}

\begin{solution}{exo:g7:areas:12}
Avec les côtés de l'angle droit~: $A = \frac{6 \times 8}{2} = 24$
cm$^2$. Avec l'hypoténuse comme base~: $24 = \frac{10 \times h}{2} =
5h$, donc $h = 4.8$ cm.
\end{solution}
