\section*{Capítulo \ref{ch:g7:negatives} --- Números Negativos}

\begin{solution}{exo:g7:negatives:1}
Opostos: $-7$; \ $+3.5$; \ $0$ (seu próprio oposto); \ $-12$. Dois
números são opostos quando seu \omterm{def:g1:addition:def}{soma} é $0$.
\end{solution}

\begin{solution}{exo:g7:negatives:2}
$-4 < 1$; \quad $-4 > -7$; \quad $-2.5 < -2.4$; \quad $0 > -0.1$.
\end{solution}

\begin{solution}{exo:g7:negatives:3}
$-3.1 < -3 < -0.5 < 0 < 1 < 2.5$.
\end{solution}

\begin{solution}{exo:g7:negatives:4}
$(-5) + (-9) = -14$; \quad
$(-12) + (+7) = -5$; \quad
$(+3.5) + (-1.5) = +2$; \quad
$(-6) + (+6) = 0$.
\end{solution}

\begin{solution}{exo:g7:negatives:5}
$4 - 11 = 4 + (-11) = -7$.

$(-5) - (+3) = (-5) + (-3) = -8$.

$(-7) - (-10) = (-7) + (+10) = +3$.

$0 - (-8) = 0 + (+8) = +8$.
\end{solution}

\begin{solution}{exo:g7:negatives:6}
$(+7) + (-12) + (+3) + (-8) = (+10) + (-20) = -10$ (positivos
juntos, negativos juntos).

$(-5.5) + (+9) + (-4.5) + (+1) = (-10) + (+10) = 0$.
\end{solution}

\begin{solution}{exo:g7:negatives:7}
$(-3) + 8 - 12 = 5 - 12 = -7$: isso é $-7$ graus à noite.
\end{solution}

\begin{solution}{exo:g7:negatives:9}
Moscou: $(-4) - (-13) = -4 + 13 = 9$ graus de \omterm{ex:g1:subtraction:difference}{diferença}.

\begin{solution}{exo:g7:negatives:10}
$(-12) + 7 - 4 = -5 - 4 = -9$: ela está em $-9$ m. Para chegar à superfície
(altitude $0$) ela deve subir $0 - (-9) = 9$ m.
\end{solution}

\begin{solution}{exo:g7:negatives:11}
$ D(-2, -2)$: mesma primeira coordenada que $ A $, mesma segunda coordenada que
$ C $. Lados: $ AB = 4$ (de $-2$ para $2$) e $ BC = 3$ (de $1$ até
$-2$), então o \omterm{def:g6:measure:perimeter}{perímetro} é $2 \times (4 + 3) = 14$.
\end{solution}

\begin{solution}{exo:g7:negatives:12}
A diagonal completa $0, -2, -4$ dá a magia \omterm{def:g1:addition:def}{soma}
$0 + (-2) + (-4) = -6$. Então, célula por célula:
canto superior esquerdo $= -6 - (-7) - 0 = 1$; meio-direita $= -6 - (-3) - (-2) =
-1$; canto inferior direito $= -6 - 0 - (-1) = -5$; meio inferior
$= -6 - (-7) - (-2) = 3$.
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|c|c|}
\hline
$1$ & $-7$ & $0$ \\
\hline
$-3$ & $-2$ & $-1$ \\
\hline
$-4$ & $3$ & $-5$ \\
\hline
\end{tabular}
\end{center}
Cada \omterm{def:g6:lines:objects}{linha}, coluna e diagonal somam $-6$.
\end{solution}

\begin{solution}{pb:g7:negatives:1}
\textbf{1.} $4\,808 - (-430) = 4\,808 + 430 = 5\,238$ eu
(\cref{thm:g7:negatives:sub}: subtrair um meio negativo
adicionando).

\textbf{2.} Equilíbrio:
\[
(-45) + 120 + (-60) + 15 = 75 + (-60) + 15 = 15 + 15 = 30
\text{ euros}.
\]
Para retornar o inicial $-45$ para $0$, deposite seu oposto:
$45$ euros.

\textbf{3.} $(-44) - (-100) = (-44) + 100 = 56$: César viveu
$56$ anos. (Não há armadilha aqui: ambos os anos estão do mesmo lado
o limite da era.)

\textbf{4.} O ano $0$: contando a partir de $-63$ até $+14$ passes
através $0$, mas o calendário dos historiadores salta direto de
$1$ AC a DC $1$ --- o ano $0$ nunca existiu, então o ingênuo
a contagem é um ano a mais.

\textbf{5.} $63$ AC se torna $-(63 - 1) = -62$. Então
$14 - (-62) = 14 + 62 = 76$ anos: exatamente os historiadores
figura. Na \omterm{def:g6:lines:objects}{linha} renomeada dos astrônomos não há lacunas,
então a subtração comum fornece durações verdadeiras.

\textbf{6.} $1$ AC $\to 0$; \ $10$ AC $\to -9$; \ ANÚNCIO
$476 \to +476$; \ $753$ AC $\to -752$.

\textbf{7.} $476 - (-752) = 476 + 752 = 1\,228$ anos.

\textbf{8.} $12$ AC é $-11$. Próximas aparições:
$-11 + 76 = 65$, então $141$, então $217$: os anos DC $65$, ANÚNCIO
$141$ e ANÚNCIO $217$. Antes: $-11 - 76 = -87$, que é o
ano dos historiadores $87 + 1 = 88$ AC.

\textbf{9.} $5$ AC é $-4$, então $5 - (-4) = 9$ anos --- não
$10$. O ano que falta \omterm{def:g1:counting:zero}{zero} ataca sempre que uma duração cruza
o limite da época.

\textbf{10.} $3$ AC é $-2$; dez anos depois é $-2 + 10 = +8$:
a criança completou dez anos em AD $8$.

\textbf{11.} Entre $-7$ e $-2$: $(-2) - (-7) = 5$. Entre
$-3.5$ e $4.5$: $4.5 - (-3.5) = 8$.

\textbf{12.} Agrupamento opostos e amigos:
\[
3 + (-5) + 2 + (-1) + 4
= \bigl(3 + 2\bigr) + (-5) + (-1) + 4
= 0 + (-1) + 4 = +3 :
\]
o elevador termina no chão $3$. A ordem não importa: um \omterm{def:g1:addition:def}{soma} de
números relativos pode ser reorganizado livremente (cada movimento é percorrido
ao longo da mesma \omterm{def:g6:lines:objects}{linha}, e a caminhada total é a mesma).

\textbf{13.} \omterm{def:g5:solids:net}{Líquido} mudar:
$2 + (-4) + 1 + 0 + (-3) + 5 + (-2) = -1$ grau. Começando em
$-2$: temperatura final $-2 + (-1) = -3$ graus. Em geral,
o valor final é o valor inicial mais o \omterm{def:g1:addition:def}{soma} de todos os
mudanças --- não há necessidade de seguir os altos e baixos um por um.

\textbf{14.} Voltar ao início significa \omterm{def:g5:solids:net}{líquido} mudar $0$: os altos e baixos
os baixos devem ser cancelados exatamente. As cinco mudanças conhecidas somam
$4 + (-7) + 2 + 3 + (-5) = -3$, então a sexta mudança é a
oposto: $+3$ (trezentos metros de altura).

\textbf{15.} Astrônomos: $509$ AC $\to -508$, e
$476 - (-508) = 984$ anos. O historiador $509 + 476 = 985$
conta um ano a mais --- mais uma vez o ano fantasma \omterm{def:g1:counting:zero}{zero},
silenciosamente incluído na adição, mas ausente da história.
(A título de verificação, as duas épocas: República $-508$ para $-26$, $482$
anos; Império $-26$ para $+476$, $502$ anos;
$482 + 502 = 984$.)
\end{solution}
