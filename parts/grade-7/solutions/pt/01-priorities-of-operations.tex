\begin{solution}{exo:g7:priorities:1}
$5 + 3 \times 6 = 5 + 18 = 23$.

$(5 + 3) \times 6 = 8 \times 6 = 48$.

$30 - 12 \div 4 = 30 - 3 = 27$.

$(30 - 12) \div 4 = 18 \div 4 = 4.5$.
\end{solution}

\begin{solution}{exo:g7:priorities:2}
$24 - 6 + 2 = 18 + 2 = 20$ (da esquerda para a direita).

$24 \div 6 \times 2 = 4 \times 2 = 8$ (da esquerda para a direita).

$24 - 6 \times 2 = 24 - 12 = 12$ (multiplicação primeiro).

$24 \div (6 \times 2) = 24 \div 12 = 2$.
\end{solution}

\begin{solution}{exo:g7:priorities:3}
$7 + 3 \times (10 - 6) = 7 + 3 \times 4 = 7 + 12 = 19$.

$40 - (2 + 3) \times 4 = 40 - 5 \times 4 = 40 - 20 = 20$.

$60 \div \bigl(2 \times (7 - 4)\bigr) = 60 \div (2 \times 3)
= 60 \div 6 = 10$.
\end{solution}

\begin{solution}{exo:g7:priorities:4}
$\dfrac{18 + 6}{8} = \dfrac{24}{8} = 3$; \quad
$\dfrac{30}{7 - 2} = \dfrac{30}{5} = 6$; \quad
$\dfrac{5 \times 8}{4 + 6} = \dfrac{40}{10} = 4$.
\end{solution}

\begin{solution}{exo:g7:priorities:6}
``O \omterm{ex:g1:subtraction:difference}{diferença} de $50$ e o \omterm{def:g2:mult:def}{produto} de $6$ por $7$'':
$50 - 6 \times 7 = 50 - 42 = 8$ (não são necessários colchetes, a prioridade não
o trabalho).

``O \omterm{def:g2:mult:def}{produto} do \omterm{def:g1:addition:def}{soma} de $4$ e $5$ pelo \omterm{ex:g1:subtraction:difference}{diferença} de $10$ e
$8$'': $(4 + 5) \times (10 - 8) = 9 \times 2 = 18$.
\end{solution}

\begin{solution}{exo:g7:priorities:9}
$ k = 1$: $12 + 4 \times 1 = 16$ e $16 \times 1 = 16$ --- eles concordam
(é por isso $ k = 1$ é um teste ruim!). $ k = 2$:
$12 + 4 \times 2 = 20$ mas $16 \times 2 = 32$: diferente. Então os dois
expressões são \emph{não} iguais em geral. O erro
$12 + 4 \times k = 16 \times k $ vem da adição $12 + 4$ primeiro,
ignorando a prioridade $\times $.
\end{solution}

\begin{solution}{exo:g7:priorities:10}
\emph{1.} Preço do grupo: $12 + 6 \times n $. Para $ n = 5$:
$12 + 30 = 42$ euros.

\emph{2.} Preço individual: $9n $. Comparar: $ n = 3$: $30$ contra $27$
(individualmente mais barato); $ n = 4$: $36$ contra $36$ (igual); $ n = 5$: $42$ contra
$45$ (grupo mais barato). De $5$ pessoas, a tarifa do grupo ganha
(cada pessoa extra custa $6$ em vez de $9$, então a \omterm{ex:g1:subtraction:difference}{diferença} só aumenta).
\end{solution}

\begin{solution}{pb:g7:priorities:1}
\textbf{1.} $7 \times 102 = 7 \times (100 + 2) = 700 + 14 =
714$; \quad $9 \times 98 = 9 \times (100 - 2) = 900 - 18 = 882$.

\textbf{2.} $17 \times 13 + 17 \times 7 = 17 \times (13 + 7) =
17 \times 20 = 340$; \quad $45 \times 101 - 45 = 45 \times (101
- 1) = 45 \times 100 = 4\,500$.

\textbf{3.} $5 \times 87 = 870 \div 2 = 435$. Legitimidade:
$5 = 10 \div 2$, então multiplicando por $5$ está multiplicando por $10$
então dividindo por $2$ (a ordem de $\times $ e $\div $ no
mesmo nível, da esquerda para a direita, faz o resto).

\textbf{4.} $25 \times 32 = 3\,200 \div 4 = 800$, porque
$25 = 100 \div 4$. E
$99 \times 47 = (100 - 1) \times 47 = 4\,700 - 47 = 4\,653$:
distributividade ao longo do \omterm{ex:g1:subtraction:difference}{diferença}
(\cref{thm:g7:priorities:distributivity}).

\textbf{5.} $8 \times (5 + 3) = 64$ e $8 \times 5 + 8 \times 3
= 40 + 24 = 64$: eles combinam --- isso é distributividade. Mas
$8 \times (5 \times 3) = 8 \times 15 = 120$, enquanto
$(8 \times 5) \times (8 \times 3) = 40 \times 24 = 960$: não
combinar. O fator se espalha pelo \emph{termos de um \omterm{def:g1:addition:def}{soma}}, não
sobre os fatores de um \omterm{def:g2:mult:def}{produto}.

\textbf{6.}
\begin{center}
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{ccl}
$13$ & $24$ & kept ($13$ \omterm{def:g3:numbers:evenodd}{ímpar}) \\
$6$ & $48$ & crossed out ($6$ \omterm{def:g3:numbers:evenodd}{par}) \\
$3$ & $96$ & kept \\
$1$ & $192$ & kept \\
\end{tabular}
\end{center}
\omterm{def:g1:addition:def}{Soma} dos números certos sobreviventes: $24 + 96 + 192 = 312$ e
de fato $13 \times 24 = 312$.

\textbf{7.} Reduzir pela \omterm{def:g3:division:half}{metade} $21$: $21, 10, 5, 2, 1$; duplicação $17$:
$17, 34, 68, 136, 272$. \omterm{def:g3:numbers:evenodd}{Par} esquerdas $10$ e $2$ estão riscados;
sobreviventes $17 + 68 + 272 = 357$. Verificar:
$21 \times 17 = 357$.

\textbf{8.} Reduzir pela \omterm{def:g3:division:half}{metade} $16$: $16, 8, 4, 2, 1$; duplicação $25$:
$25, 50, 100, 200, 400$. Cada número à esquerda, exceto o final $1$
é \omterm{def:g3:numbers:evenodd}{par}: uma única \omterm{def:g6:lines:objects}{linha} sobrevive e a resposta é $400$. Uma esquerda
coluna feita de duplicações puras mantém apenas sua última \omterm{def:g6:lines:objects}{linha} --- a
método se reduz apenas à duplicação.

\textbf{9.} $13 = 12 + 1$, então por distributividade
$13 \times 24 = 12 \times 24 + 1 \times 24$; e
$12 \times 24 = 6 \times 48$ desde reduzir pela \omterm{def:g3:division:half}{metade} um fator enquanto
duplicar o outro deixa um \omterm{def:g2:mult:def}{produto} inalterado
($12 \times 24 = 6 \times 2 \times 24 = 6 \times 48$). O \omterm{def:g3:numbers:evenodd}{ímpar}
número esquerdo $13$ separe uma cópia do número certo: o
bancos de primeira \omterm{def:g6:lines:objects}{linha} $24$.

\textbf{10.} $6 \times 48 = 3 \times 96$ (\omterm{def:g3:division:half}{metade} - o dobro, não
bancário: $6$ é \omterm{def:g3:numbers:evenodd}{par}). $3 \times 96 = 2 \times 96 + 96 =
1 \times 192 + 96$: o \omterm{def:g3:numbers:evenodd}{ímpar} $3$ bancos $96$. Finalmente
$1 \times 192 = 192$: o \omterm{def:g3:numbers:evenodd}{ímpar} $1$ bancos $192$ e a mesa
termina. Total bancado: $24 + 96 + 192 = 312$. Uma fileira deposita seu
número certo exatamente quando seu número esquerdo é \omterm{def:g3:numbers:evenodd}{ímpar} --- então o
receita ``manter as \omterm{def:g6:lines:objects}{linhas} \omterm{def:g3:numbers:evenodd}{ímpares} à esquerda e adicionar'' é precisamente a
escrituração da distributividade.

\textbf{11.} $1 + 4 + 8 = 13$. Para $21$, as \omterm{def:g6:lines:objects}{linhas} sobreviventes foram
aqueles de $17$, $68 = 4 \times 17$ e $272 = 16 \times 17$: o
duplicações $1 + 4 + 16 = 21$.

\textbf{12.} $25 = 16 + 8 + 1$ e $42 = 32 + 8 + 2$.

\textbf{13.} Pegue o maior número de duplicação que não exceda o
mire e subtraia; o que resta é menor e menor
do que o número de duplicação que acabamos de usar (caso contrário, um número maior
caberia). Repetindo, o \omterm{def:g3:division:remainder}{restante} encolhe estritamente
e deve alcançar $0$: o processo para e as duplicações usadas
são todos diferentes. Para $100$: $64$ cabe ($36$ esquerda), $32$ cabe
($4$ esquerda), $4$ cabe ($0$ esquerda): $100 = 64 + 32 + 4$.

\textbf{14.} Tabela de duplicação de $35$: $1 \to 35$, $2 \to 70$,
$4 \to 140$, $8 \to 280$, $16 \to 560$. Desde
$19 = 16 + 2 + 1$, escolha essas \omterm{def:g6:lines:objects}{linhas}:
$560 + 70 + 35 = 665$. Verificar: $19 \times 35 = 665$.

\textbf{15.} Os mesmos três do escriba e do camponês:
$24 + 96 + 192$ --- as duplicações de $24$ selecionado por
$13 = 1 + 4 + 8$. Escrevendo números como \omterm{def:g1:addition:def}{somas} de duplicações é
exatamente escrevendo-os com $0$ areia $1$ é; a história completa é contada
em \cref{pb:g8:powers:1}.
\end{solution}
