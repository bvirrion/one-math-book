\section*{Chapter \ref{ch:g7:central} --- Central Symmetry and Parallelograms}

\begin{solution}{exo:g7:central:1}
Half-turn image: $M'$ is four squares \emph{left} of $O$ and one
square \emph{down} (both directions reversed). Reflection image
across the vertical line: four squares left but still one square
\emph{up}. The two images differ: they are vertical mirror images of
each other.
\end{solution}

\begin{solution}{exo:g7:central:2}
$A'B' = AB$ (half-turns preserve lengths) and
$(A'B') \parallel (AB)$ (the image of a line is a \omterm{def:g6:lines:perp}{parallel} line,
\cref{prop:g7:central:preserved}).
\end{solution}

\begin{solution}{exo:g7:central:3}
Calculator digits with a \omterm{def:g7:central:center}{center of symmetry}: $0$, $2$, $5$, $8$ (and
$1$ if drawn as a plain bar). Letters: H, N, S, Z, O have one; A and
T do not (they have an axis instead).
\end{solution}

\begin{solution}{exo:g7:central:4}
Opposite sides equal: $MN = KL = 7$ cm and $NK = LM = 4$ cm. Opposite
angles equal: $\widehat M = \widehat K = 115^\circ$. The diagonals
cut each other at their common midpoint, and that point is $O$: the
midpoint of $[LN]$ is $O$.
\end{solution}

\begin{solution}{exo:g7:central:5}
Draw a \omterm{def:g6:lines:objects}{segment} $[AC]$ of $8$ cm, mark its midpoint $O$, draw through
$O$ a line making $60^\circ$ with $(AC)$, and place $B$ and $D$ on it
at $2.5$ cm from $O$ on either side. Join $A$, $B$, $C$, $D$: the
diagonals cut at their midpoints, so $ABCD$ is a \omterm{def:g7:central:parallelogram}{parallelogram} by
definition.
\end{solution}

\begin{solution}{exo:g7:central:6}
Consecutive angles are supplementary: next to $115^\circ$ sits
$180 - 115 = 65^\circ$. Opposite angles are equal, so the four angles
are $115^\circ$, $65^\circ$, $115^\circ$, $65^\circ$ (\omterm{def:g1:addition:def}{sum}
$360^\circ$).
\end{solution}

\begin{solution}{exo:g7:central:7}
From $A(1,1)$ to $O(2.5, 2.5)$: $1.5$ right and $1.5$ up; continuing
the same: $A'(4, 4)$. From $B(4,2)$ to $O$: $1.5$ left and $0.5$ up;
continuing: $B'(1, 3)$.
\end{solution}

\begin{solution}{exo:g7:central:8}
No. An \omterm{def:g6:shapes:triangles}{isosceles} trapezoid has its two slanted sides equal without
being a \omterm{def:g7:central:parallelogram}{parallelogram} (the two \omterm{def:g6:lines:perp}{parallel} sides have different
lengths). What suffices is criterion 3 of
\cref{thm:g7:central:recognize}: two opposite sides equal
\emph{and \omterm{def:g6:lines:perp}{parallel}}.
\end{solution}

\begin{solution}{exo:g7:central:9}
$O$ is the midpoint of $[BC]$ by choice, and it is the midpoint of
$[AA']$ by construction of the \omterm{def:g7:central:def}{symmetric}. The \omterm{def:g4:geometry:polygon}{quadrilateral} $ABA'C$
has diagonals $[AA']$ and $[BC]$ sharing the midpoint $O$: criterion
1 (the definition) makes it a \omterm{def:g7:central:parallelogram}{parallelogram} at no extra cost.
\end{solution}

\begin{solution}{exo:g7:central:10}
``Opposite angles equal $\Rightarrow$ \omterm{def:g7:central:parallelogram}{parallelogram}'': \emph{true}
(this is another recognition criterion; with the angle \omterm{def:g1:addition:def}{sum}
$360^\circ$, equal opposite pairs force supplementary consecutive
angles, hence \omterm{def:g6:lines:perp}{parallel} sides by
\cref{thm:g7:triangles:equalities}).

``\omterm{def:g6:lines:perp}{Perpendicular} diagonals $\Rightarrow$ \omterm{def:g6:shapes:quadrilaterals}{rhombus}'': \emph{true} for a
\omterm{def:g7:central:parallelogram}{parallelogram} (the diagonals already cut at their midpoints).

``One \omterm{def:g3:shapes:rightangle}{right angle} $\Rightarrow$ rectangle'': \emph{true} ---
consecutive angles are supplementary, so all four become right.
\end{solution}

\begin{solution}{exo:g7:central:11}
Whatever the \omterm{def:g4:geometry:polygon}{quadrilateral} --- even a very irregular one --- the
midpoints $I$, $J$, $K$, $L$ always form a \emph{\omterm{def:g7:central:parallelogram}{parallelogram}}. The
proof with the midpoint theorem is in \cref{ch:g8:midpoints}
(\cref{exo:g8:midpoints:9}).
\end{solution}
