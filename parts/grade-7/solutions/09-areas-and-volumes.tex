\section*{Chapter \ref{ch:g7:areas} --- Areas and Volumes}

\begin{solution}{exo:g7:areas:1}
$8 \times 5 = 40$ cm$^2$; \qquad $6.5 \times 4 = 26$ m$^2$.
\end{solution}

\begin{solution}{exo:g7:areas:2}
$A = 10 \times 4 = 40$ cm$^2$. The answer is not
$10 \times 6 = 60$ because the $6$ cm side is slanted: the formula
uses the \emph{height} (perpendicular distance), which is $4$ cm.
\end{solution}

\begin{solution}{exo:g7:areas:3}
$\frac{12 \times 7}{2} = 42$ cm$^2$; \quad
$\frac{9 \times 6}{2} = 27$; \quad
$\frac{5.5 \times 2}{2} = 5.5$ m$^2$.
\end{solution}

\begin{solution}{exo:g7:areas:4}
Disk: $\pi \times 5^2 = 25\pi \approx 79$ cm$^2$. Half-disk of radius
$10$: $\frac{\pi \times 10^2}{2} = 50\pi \approx 157$ cm$^2$.
\end{solution}

\begin{solution}{exo:g7:areas:5}
Perimeter: $2\pi \times 3 = 6\pi \approx 18.8$ cm (a length, in cm).
Area: $\pi \times 3^2 = 9\pi \approx 28.3$ cm$^2$ (a surface, in
cm$^2$).
\end{solution}

\begin{solution}{exo:g7:areas:6}
Prism: $24 \times 10 = 240$ cm$^3$. Cylinder:
$\pi \times 3^2 \times 8 = 72\pi \approx 226$ cm$^3$.
\end{solution}

\begin{solution}{exo:g7:areas:7}
$\frac{8 \times h}{2} = 28$, so $4h = 28$ and $h = 7$ cm.
\end{solution}

\begin{solution}{exo:g7:areas:8}
The three products agree because each one computes the \emph{same}
quantity --- the area of the triangle. This gives a practical check
of constructions, and a way of computing a height from another
base--height pair (used in \cref{exo:g7:areas:12}).
\end{solution}

\begin{solution}{exo:g7:areas:9}
The two triangles have areas $\frac{30 \times 20}{2} = 300$ m$^2$
and $\frac{18 \times 20}{2} = 180$ m$^2$: total $480$ m$^2$. General
formula suggested: for parallel sides $a$ and $b$ at distance $h$,
\[
A = \frac{(a + b) \times h}{2}
\]
--- the trapezoid formula (here $\frac{(30+18)\times 20}{2} = 480$).
\end{solution}

\begin{solution}{exo:g7:areas:10}
Water volume: $\pi \times 6^2 \times 15 = 540\pi \approx 1\,696$
cm$^3$. Box base: $18 \times 12 = 216$ cm$^2$. Height reached:
$\frac{540\pi}{216} = 2.5\pi \approx 7.9$ cm.
\end{solution}

\begin{solution}{exo:g7:areas:11}
Radii $15$ and $20$ cm. Areas: $225\pi \approx 707$ cm$^2$ and
$400\pi \approx 1257$ cm$^2$. Price per $100$ cm$^2$:
$\frac{9}{7.07} \approx 1.27$ euros and
$\frac{14}{12.57} \approx 1.11$ euros. The large pizza is the better
deal --- areas grow with the \emph{square} of the diameter.
\end{solution}

\begin{solution}{exo:g7:areas:12}
With the legs: $A = \frac{6 \times 8}{2} = 24$ cm$^2$. With the
hypotenuse as base: $24 = \frac{10 \times h}{2} = 5h$, so
$h = 4.8$ cm.
\end{solution}
