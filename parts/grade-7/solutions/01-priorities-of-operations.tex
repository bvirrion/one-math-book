\section*{Chapter \ref{ch:g7:priorities} --- Priorities of Operations}

\begin{solution}{exo:g7:priorities:1}
$5 + 3 \times 6 = 5 + 18 = 23$.

$(5 + 3) \times 6 = 8 \times 6 = 48$.

$30 - 12 \div 4 = 30 - 3 = 27$.

$(30 - 12) \div 4 = 18 \div 4 = 4.5$.
\end{solution}

\begin{solution}{exo:g7:priorities:2}
$24 - 6 + 2 = 18 + 2 = 20$ (left to right).

$24 \div 6 \times 2 = 4 \times 2 = 8$ (left to right).

$24 - 6 \times 2 = 24 - 12 = 12$ (multiplication first).

$24 \div (6 \times 2) = 24 \div 12 = 2$.
\end{solution}

\begin{solution}{exo:g7:priorities:3}
$7 + 3 \times (10 - 6) = 7 + 3 \times 4 = 7 + 12 = 19$.

$40 - (2 + 3) \times 4 = 40 - 5 \times 4 = 40 - 20 = 20$.

$60 \div \bigl(2 \times (7 - 4)\bigr) = 60 \div (2 \times 3)
= 60 \div 6 = 10$.
\end{solution}

\begin{solution}{exo:g7:priorities:4}
$\dfrac{18 + 6}{8} = \dfrac{24}{8} = 3$; \quad
$\dfrac{30}{7 - 2} = \dfrac{30}{5} = 6$; \quad
$\dfrac{5 \times 8}{4 + 6} = \dfrac{40}{10} = 4$.
\end{solution}

\begin{solution}{exo:g7:priorities:5}

\begin{solution}{exo:g7:priorities:6}
``The \omterm{ex:g1:subtraction:difference}{difference} of $50$ and the \omterm{def:g2:mult:def}{product} of $6$ by $7$'':
$50 - 6 \times 7 = 50 - 42 = 8$ (no brackets needed, the priority does
the work).

``The \omterm{def:g2:mult:def}{product} of the \omterm{def:g1:addition:def}{sum} of $4$ and $5$ by the \omterm{ex:g1:subtraction:difference}{difference} of $10$ and
$8$'': $(4 + 5) \times (10 - 8) = 9 \times 2 = 18$.
\end{solution}

\begin{solution}{exo:g7:priorities:9}
$k = 1$: $12 + 4 \times 1 = 16$ and $16 \times 1 = 16$ --- they agree
(that is why $k = 1$ is a bad test!). $k = 2$:
$12 + 4 \times 2 = 20$ but $16 \times 2 = 32$: different. So the two
expressions are \emph{not} equal in general. The mistake
$12 + 4 \times k = 16 \times k$ comes from adding $12 + 4$ first,
ignoring the priority of $\times$.
\end{solution}

\begin{solution}{exo:g7:priorities:10}
\emph{1.} Group price: $12 + 6 \times n$. For $n = 5$:
$12 + 30 = 42$ euros.

\emph{2.} Individual price: $9n$. Compare: $n = 3$: $30$ vs $27$
(individual cheaper); $n = 4$: $36$ vs $36$ (equal); $n = 5$: $42$ vs
$45$ (group cheaper). From $5$ people on, the group rate wins
(each extra person costs $6$ instead of $9$, so the gap only grows).
\end{solution}

\begin{solution}{pb:g7:priorities:1}
\textbf{1.} $7 \times 102 = 7 \times (100 + 2) = 700 + 14 =
714$; \quad $9 \times 98 = 9 \times (100 - 2) = 900 - 18 = 882$.

\textbf{2.} $17 \times 13 + 17 \times 7 = 17 \times (13 + 7) =
17 \times 20 = 340$; \quad $45 \times 101 - 45 = 45 \times (101
- 1) = 45 \times 100 = 4\,500$.

\textbf{3.} $5 \times 87 = 870 \div 2 = 435$. Legitimacy:
$5 = 10 \div 2$, so multiplying by $5$ is multiplying by $10$
then dividing by $2$ (the order of $\times$ and $\div$ at the
same level, left to right, does the rest).

\textbf{4.} $25 \times 32 = 3\,200 \div 4 = 800$, because
$25 = 100 \div 4$. And
$99 \times 47 = (100 - 1) \times 47 = 4\,700 - 47 = 4\,653$:
distributivity over the \omterm{ex:g1:subtraction:difference}{difference}
(\cref{thm:g7:priorities:distributivity}).

\textbf{5.} $8 \times (5 + 3) = 64$ and $8 \times 5 + 8 \times 3
= 40 + 24 = 64$: they match --- that is distributivity. But
$8 \times (5 \times 3) = 8 \times 15 = 120$, while
$(8 \times 5) \times (8 \times 3) = 40 \times 24 = 960$: no
match. The factor spreads over the \emph{terms of a \omterm{def:g1:addition:def}{sum}}, not
over the factors of a \omterm{def:g2:mult:def}{product}.

\textbf{6.}
\begin{center}
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{ccl}
$13$ & $24$ & kept ($13$ \omterm{def:g3:numbers:evenodd}{odd}) \\
$6$ & $48$ & crossed out ($6$ \omterm{def:g3:numbers:evenodd}{even}) \\
$3$ & $96$ & kept \\
$1$ & $192$ & kept \\
\end{tabular}
\end{center}
\omterm{def:g1:addition:def}{Sum} of the surviving right numbers: $24 + 96 + 192 = 312$, and
indeed $13 \times 24 = 312$.

\textbf{7.} Halving $21$: $21, 10, 5, 2, 1$; doubling $17$:
$17, 34, 68, 136, 272$. \omterm{def:g3:numbers:evenodd}{Even} lefts $10$ and $2$ are crossed out;
survivors $17 + 68 + 272 = 357$. Check:
$21 \times 17 = 357$.

\textbf{8.} Halving $16$: $16, 8, 4, 2, 1$; doubling $25$:
$25, 50, 100, 200, 400$. Every left number except the final $1$
is \omterm{def:g3:numbers:evenodd}{even}: a single row survives, and the answer is $400$. A left
column made of pure doublings keeps only its last row --- the
method reduces to doubling alone.

\textbf{9.} $13 = 12 + 1$, so by distributivity
$13 \times 24 = 12 \times 24 + 1 \times 24$; and
$12 \times 24 = 6 \times 48$ since halving one factor while
doubling the other leaves a \omterm{def:g2:mult:def}{product} unchanged
($12 \times 24 = 6 \times 2 \times 24 = 6 \times 48$). The \omterm{def:g3:numbers:evenodd}{odd}
left number $13$ split off one copy of the right number: the
first row banks $24$.

\textbf{10.} $6 \times 48 = 3 \times 96$ (halve--double, no
banking: $6$ is \omterm{def:g3:numbers:evenodd}{even}). $3 \times 96 = 2 \times 96 + 96 =
1 \times 192 + 96$: the \omterm{def:g3:numbers:evenodd}{odd} $3$ banks $96$. Finally
$1 \times 192 = 192$: the \omterm{def:g3:numbers:evenodd}{odd} $1$ banks $192$ and the table
ends. Total banked: $24 + 96 + 192 = 312$. A row banks its
right number exactly when its left number is \omterm{def:g3:numbers:evenodd}{odd} --- so the
recipe ``keep the odd-left rows and add'' is precisely the
bookkeeping of distributivity.

\textbf{11.} $1 + 4 + 8 = 13$. For $21$, the surviving rows were
those of $17$, $68 = 4 \times 17$ and $272 = 16 \times 17$: the
doublings $1 + 4 + 16 = 21$.

\textbf{12.} $25 = 16 + 8 + 1$ and $42 = 32 + 8 + 2$.

\textbf{13.} Take the largest doubling number not exceeding the
target and subtract it; what remains is smaller, and smaller
than the doubling number just used (otherwise a bigger one
would have fitted). Repeating, the \omterm{def:g3:division:remainder}{remainder} strictly shrinks
and must reach $0$: the process stops, and the doublings used
are all different. For $100$: $64$ fits ($36$ left), $32$ fits
($4$ left), $4$ fits ($0$ left): $100 = 64 + 32 + 4$.

\textbf{14.} Doubling table of $35$: $1 \to 35$, $2 \to 70$,
$4 \to 140$, $8 \to 280$, $16 \to 560$. Since
$19 = 16 + 2 + 1$, pick those rows:
$560 + 70 + 35 = 665$. Check: $19 \times 35 = 665$.

\textbf{15.} The same three as the scribe and the peasant:
$24 + 96 + 192$ --- the doublings of $24$ selected by
$13 = 1 + 4 + 8$. Writing numbers as \omterm{def:g1:addition:def}{sums} of doublings is
exactly writing them with $0$s and $1$s; the full story is told
in \cref{pb:g8:powers:1}.
\end{solution}
