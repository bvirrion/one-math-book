\section*{अध्याय \ref{ch:g7:central} --- केंद्रीय समरूपता और समांतर चतुर्भुज}

\begin{solution}{exo:g7:central:1}
अर्ध-मोड़ छवि: $M'$ चार वर्ग है \emph{बाएं} का $ O $ और एक
वर्ग \emph{नीचे} (दोनों दिशाएं उलट गईं)। प्रतिबिम्ब छवि
ऊर्ध्वाधर \omterm{def:g6:lines:objects}{रेखा} के पार: चार वर्ग बचे हैं लेकिन फिर भी एक वर्ग है
\emph{ऊपर}. दोनों छवियाँ भिन्न हैं: वे ऊर्ध्वाधर दर्पण छवियाँ हैं
एक दूसरे.
\end{solution}

\begin{solution}{exo:g7:central:2}

$(A'B') \parallel (AB)$ (एक \omterm{def:g6:lines:objects}{रेखा} की छवि है a \omterm{def:g6:lines:perp}{समानांतर} \omterm{def:g6:lines:objects}{रेखा},
\cref{prop:g7:central:preserved}).
\end{solution}

\begin{solution}{exo:g7:central:3}

$1$ यदि एक सादे बार के रूप में खींचा गया हो)। अक्षर: H, N, S, Z, O में एक है; ए और

\begin{solution}{exo:g7:central:4}

कोण बराबर: $\widehat M = \widehat K = 115^\circ $. विकर्ण
एक दूसरे को उनके सामान्य मध्यबिंदु पर काटें, और वह बिंदु है $ O $: द
का मध्यबिंदु $[LN]$ है $ O $.
\end{solution}

\begin{solution}{exo:g7:central:5}
चित्रित करो \omterm{def:g6:lines:objects}{खंड} $[AC]$ का $8$ सेमी, इसके मध्यबिंदु को चिह्नित करें $ O $, के माध्यम से ड्रा
$ O $ एक लाइन बनाना $60^\circ $ साथ $(AC)$, और जगह $ B $ और $ D $ उस पर
पर $2.5$ सेमी से $ O $ दोनों तरफ़। जोड़ना $ A $, $ B $, $ C $, $ D $: द
विकर्ण अपने मध्यबिंदुओं पर कटते हैं, इसलिए $ ABCD $ एक है \omterm{def:g7:central:parallelogram}{चतुर्भुज} परिभाषा से।
\end{solution}

\begin{solution}{exo:g7:central:6}
क्रमागत कोण संपूरक हैं: के आगे $115^\circ $ बैठता
$180 - 115 = 65^\circ $. सम्मुख कोण बराबर होते हैं, अतः चारों कोण
हैं $115^\circ $, $65^\circ $, $115^\circ $, $65^\circ $ (\omterm{def:g1:addition:def}{जोड़}
$360^\circ $).
\end{solution}

\begin{solution}{exo:g7:central:7}

\end{solution}

\begin{solution}{exo:g7:central:8}

एक होने के नाते \omterm{def:g7:central:parallelogram}{चतुर्भुज} (दो \omterm{def:g6:lines:perp}{समानांतर} पक्ष अलग-अलग हैं

\end{solution}

\begin{solution}{exo:g7:central:9}
$ O $ का मध्यबिंदु है $[BC]$ पसंद से, और यह इसका मध्यबिंदु है
$[AA']$ के निर्माण द्वारा \omterm{def:g7:central:def}{सममित}. \omterm{def:g4:geometry:polygon}{चतुष्कोष} $ ABA'C $
विकर्ण हैं $[AA']$ और $[BC]$ मध्यबिंदु साझा करना $ O $: कसौटी
1 (परिभाषा) इसे बनाती है \omterm{def:g7:central:parallelogram}{चतुर्भुज} बिना किसी अतिरिक्त लागत के.
\end{solution}

\begin{solution}{exo:g7:central:10}
``विपरीत कोण बराबर होते हैं $\Rightarrow $ \omterm{def:g7:central:parallelogram}{चतुर्भुज}'': \emph{सत्य}
(यह एक और मान्यता मानदंड है; कोण के साथ \omterm{def:g1:addition:def}{जोड़}
$360^\circ $, समान विपरीत \omterm{def:g1:addition:def}{जोड़}े लगातार पूरक को बल देते हैं
कोण, इसलिए \omterm{def:g6:lines:perp}{समानांतर} पक्षों द्वारा
\cref{thm:g7:triangles:equalities}).

``\omterm{def:g6:lines:perp}{सीधा} विकर्णों $\Rightarrow $ \omterm{def:g6:shapes:quadrilaterals}{विषमकोण}'': \emph{सत्य} एक के लिए
\omterm{def:g7:central:parallelogram}{चतुर्भुज} (विकर्ण पहले से ही अपने मध्य बिंदुओं पर कट जाते हैं)।

``एक \omterm{def:g3:shapes:rightangle}{समकोण} $\Rightarrow $ आयत'': \emph{सत्य} ---
क्रमागत कोण संपूरक होते हैं, इसलिए चारों \omterm{def:g3:shapes:rightangle}{समकोण} बनते हैं।
\end{solution}

\begin{solution}{exo:g7:central:11}
जो भी \omterm{def:g4:geometry:polygon}{चतुष्कोष} --- यहां तक कि एक बहुत ही अनियमित --- भी
मध्यबिंदु $ I $, $ J $, $ K $, $ L $ हमेशा एक फार्म \emph{\omterm{def:g7:central:parallelogram}{चतुर्भुज}}. द
मध्यबिंदु प्रमेय के साथ प्रमाण अंदर है \cref{ch:g8:midpoints}
(\cref{exo:g8:midpoints:9}).
\end{solution}

\begin{solution}{pb:g7:central:1}
\textbf{1.} $ M(3, 1) \to (-3, -1)$; $ N(-2, 4) \to (2, -4)$;
$ P(0, -5) \to (0, 5)$. नियम: मूल के बारे में \omterm{def:g3:division:half}{आधा} मोड़
भेजता है $(x, y)$ को $(-x, -y)$ --- दोनों निर्देशांक चिन्ह बदलते हैं।

\textbf{2.} से $ M(5, 3)$: के दाईं ओर तीन वर्ग $ C $ बन जाता है
तीन वर्ग बचे, दो ऊपर दो नीचे हो गए: छवि
$(-1, -1)$. रेसिपी जांच: केंद्र से बिंदु घटाकर दो बार,
$(2 \times 2 - 5,\ 2 \times 1 - 3) = (-1, -1)$.

\textbf{3.} कार्ड के केंद्र के बारे में \omterm{def:g3:division:half}{आधा} मोड़ पिप्स को स्वैप करता है
\omterm{def:g1:addition:def}{जोड़}े में. एक साथ \omterm{def:g3:numbers:evenodd}{विषम} संख्या पिप्स में से, एक पिप का कोई साथी नहीं है: यह
उसकी अपनी छवि होनी चाहिए, और एकमात्र बिंदु उसकी अपनी छवि है
केंद्र ही है. इसलिए 3, 5, का मध्य पिप
7\,\dots\ मृत केंद्र में बैठता है (वास्तविक कार्ड को देखें: ऐसा होता है)।

\textbf{4.} के बारे में $ O_1(2,0)$: $ A(1,1) \to (3,-1)$; के बारे में
$ O_2(5,0)$: $(3,-1) \to (7,1)$. शुरू $ A(1,1)$, खत्म करना
$(7,1)$: फिसलना $6$ दाईं ओर, फ़्लिप नहीं किया गया। के लिए भी वैसा ही
$ B(0,3) \to (4,-3) \to (6,3)$: फिसलना $6$ सही। दो
आधे-मोड़ की राशि a होती है \emph{अनुवाद}.

\textbf{5.} स्लाइड है $6$, और $ O_1 O_2 = 3$: अनुवाद
कवर \emph{दो बार} केन्द्रों के बीच की दूरी (में)
से दिशा $ O_1$ को $ O_2$) --- बस दो के रूप में \omterm{def:g6:lines:perp}{समानांतर} दर्पण
उनके \omterm{ex:g1:subtraction:difference}{अंतर} से दोगुना अनुवाद किया गया \cref{pb:g6:symmetry:1}.

\textbf{6.} \omterm{def:g3:division:half}{आधा} चक्कर घूमना $ O $ भेजता है \omterm{def:g7:central:parallelogram}{चतुर्भुज} पर
स्वयं (वही है ``\omterm{def:g7:central:center}{समरूपता का केंद्र}'' इसका मतलब है,
\cref{thm:g7:central:properties}). के माध्यम से एक पंक्ति $ O $ भी है
स्वयं पर भेजा गया, इसलिए दो बिंदु जहां यह पार करता है
एक दूसरे के साथ सीमा की अदला-बदली: वे हैं \omterm{def:g7:central:def}{सममित} के बारे में $ O $. द
के दो टुकड़े \omterm{def:g7:central:parallelogram}{चतुर्भुज} लाइन के दोनों ओर स्वैप करें
भी, इसलिए वे सटीक प्रतियाँ हैं (\cref{prop:g7:central:preserved})
--- बराबर \omterm{def:g6:measure:area}{क्षेत्र}ों.

\textbf{7.} के माध्यम से लाइन के साथ काटें \emph{दो केंद्र}:
केक आयत का केंद्र और छेद का केंद्र।
प्रश्न 6 द्वारा केक आयत पर लागू किया गया, कट \omterm{def:g3:division:half}{आधा} हो गया
केक; छेद आयत पर लागू किया गया (\omterm{def:g6:lines:objects}{रेखा} इसके माध्यम से गुजरती है
केंद्र भी), यह छेद को \omterm{def:g3:division:half}{आधा} कर देता है। प्रत्येक भाग मिलता है \omterm{def:g3:division:half}{आधा} केक
माइनस \omterm{def:g3:division:half}{आधा} छेद: बिल्कुल निष्पक्ष.

\textbf{8.} $ D = (0 + 8 - 6,\ 0 + 5 - 1) = (2, 4)$ (द
विकर्ण अपना मध्यबिंदु साझा करते हैं, इसलिए $ D $ की अर्ध-मोड़ छवि है
$ B $ केंद्र के बारे में) केंद्र: का मध्यबिंदु $[AC]$,
$ O(4,\ 2.5)$.

\textbf{9.} में \omterm{def:g7:central:parallelogram}{चतुर्भुज} $ ABA'C $, विपरीत भुजाएँ हैं
बराबर: $ BA' = AC $. में त्रिभुज असमानता $ ABA'$ देता है
$ AA' < AB + BA' = AB + AC $. तब से $ O $ का मध्यबिंदु है
$[AA']$, $ AA' = 2\,AO $, इसलिए $2\,AO < AB + AC $, अर्थात\
$ AO < \frac{AB + AC}{2}$.

\textbf{10.} ऊपरी सीमा: $ AO < \frac{5 + 7}{2} = 6$ सेमी। निचला
बाध्य: में $ ABA'$, $ AA' > BA' - AB = 7 - 5 = 2$, इसलिए
$ AO > 1$ सेमी। से माध्यिका $ A $ सख्ती से बीच में स्थित है $1$ और
$6$ सेमी।

\textbf{11.} \omterm{def:g6:lines:objects}{खंड}: हाँ, यह मध्यबिंदु है। पंक्ति: हाँ --- हर एक
इसके बिंदुओं में से एक केंद्र है. समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण}: कोई नहीं। वर्ग:
हाँ, विकर्णों का प्रतिच्छेदन। वृत्त: हाँ, यह केंद्र है।
एस और जेड: हाँ, उनका मध्य बिंदु (पृष्ठ को उल्टा करें:
वे वही पढ़ते हैं)।

\textbf{12.} \omterm{def:g3:division:half}{आधा} मोड़ तीनों को \omterm{def:g1:addition:def}{जोड़} देगा \omterm{def:g5:solids:def}{कोने} आपस में; तीन होना \omterm{def:g3:numbers:evenodd}{विषम}, एक शिखर $ A $ अपना ही होगा
छवि, जबरदस्ती $ A $ आधे-मोड़ का केंद्र बनना। दो
अन्य \omterm{def:g5:solids:def}{कोने} $ B $ और $ C $ तब होगा \omterm{def:g7:central:def}{सममित} के बारे में $ A $ ---
इसलिए $ A $ का मध्यबिंदु होगा $[BC]$, तीन डालकर
\omterm{def:g5:solids:def}{कोने} एक पंक्ति पर. एक त्रिभुज में कोई भी तीन संरेखित नहीं है \omterm{def:g5:solids:def}{कोने}:
विरोधाभास। कोई केंद्र मौजूद नहीं है.

\textbf{13.} वही समता तर्क: आधा-मोड़ संरक्षण
द \omterm{def:g4:geometry:polygon}{बहुभुज} इसे \omterm{def:g1:addition:def}{जोड़}े \omterm{def:g5:solids:def}{कोने}; एक \omterm{def:g3:numbers:evenodd}{विषम} संख्या का \omterm{def:g5:solids:def}{कोने}
एक निश्चित बल देता है शिखर, जो केंद्र और मध्यबिंदु होना चाहिए
का \omterm{def:g6:lines:objects}{खंड} दो अन्य को जोड़ना \omterm{def:g5:solids:def}{कोने} --- तीन संरेखित
\omterm{def:g5:solids:def}{कोने}, एक में असंभव \omterm{def:g4:geometry:polygon}{बहुभुज}\,\dots\ तो नहीं \omterm{def:g4:geometry:polygon}{बहुभुज} एक साथ
\omterm{def:g3:numbers:evenodd}{विषम} संख्या का \omterm{def:g5:solids:def}{कोने} एक \omterm{def:g7:central:center}{समरूपता का केंद्र}.

\textbf{14.} अर्ध-मोड़ का प्रदर्शन करना $ O_1$, फिर के बारे में
$ O_2$, दोनों बार आंकड़ा अपने ऊपर भेजता है --- तो उनका
संयोजन भी करता है. प्रश्न 5 से, वह संयोजन एक है
दुगुनी दूरी से अनुवाद $ O_1 O_2$, जो नहीं है \omterm{def:g1:counting:zero}{शून्य}
चूंकि केंद्र अलग-अलग हैं। एक शीट पर कोई आकृति फिट नहीं हो सकती
बराबर अपने आप को एक निश्चित गै\omterm{def:g1:counting:zero}{र-शून्य} राशि से स्लाइड करें (इसे पर्याप्त रूप से स्लाइड करें)।
कई बार और यह शीट को पूरी तरह से छोड़ देता है)। तो एक परिबद्ध आकृति है
अधिक से अधिक एक \omterm{def:g7:central:center}{समरूपता का केंद्र}.

\textbf{15.} अंतहीन पट्टी पर, बिंदु के बारे में \omterm{def:g3:division:half}{आधा} मोड़
बाएँ पदचिह्न और अगले दाएँ पदचिह्न के बीच में
पैटर्न को स्वयं भेजता है; बात एक कदम आगे
भी करता है: दो अलग-अलग केंद्र। दो अर्ध-मोड़ों की रचना करना
आधे चरण से दोगुना अनुवाद देता है --- बिल्कुल
एक-चरणीय स्लाइड जो स्पष्ट रूप से अनंत फ्रिज़ को मैप करती है
स्वयं, जैसा कि प्रश्न 14 भविष्यवाणी करता है। असीमित पैटर्न जीवित रहते हैं
अलग-अलग नियम: यही है वॉलपेपर का गणित।
\end{solution}
