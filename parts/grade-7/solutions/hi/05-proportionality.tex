\section*{अध्याय \ref{ch:g7:prop} --- समानता}

\begin{solution}{exo:g7:prop:1}
पहली तालिका: \omterm{def:g3:division:remainder}{भागफल} $\frac{7.5}{3} = 2.5$,
$\frac{17.5}{7} = 2.5$, $\frac{27.5}{11} = 2.5$: \omterm{def:g6:propdata:def}{आनुपातिक},
गुणांक $2.5$.

दूसरी तालिका: $\frac62 = 3$ और $\frac{15}{5} = 3$, लेकिन
$\frac{28}{9} \approx 3.1$: नहीं \omterm{def:g6:propdata:def}{आनुपातिक}.
\end{solution}

\begin{solution}{exo:g7:prop:2}
क्रॉस नियम के साथ: $5 \times p = 6 \times 8$, इसलिए
$ p = \frac{48}{5} = 9.60$ यूरो.
\end{solution}

\begin{solution}{exo:g7:prop:3}
दर: $36 \div 3 = 12$ प्रति मिनट पेज. में $5$ मिनट:
$60$ पन्ने. के लिए $96$ पन्ने: $96 \div 12 = 8$ मिनट।
\end{solution}

\begin{solution}{exo:g7:prop:4}
$30\,\%$ का $250$: $0.3 \times 250 = 75$. \quad
$15\,\%$ का $60$: $0.15 \times 60 = 9$. \quad
$ t\,\%$ का $80$ है $20$: $\frac{20}{80} = \frac{25}{100}$, इसलिए
$ t = 25$.
\end{solution}

\begin{solution}{exo:g7:prop:5}
$\frac{18}{25} = \frac{18 \times 4}{25 \times 4} = \frac{72}{100} =
72\,\%$.
\end{solution}

\begin{solution}{exo:g7:prop:6}
पथ: $9 \times 25\,000 = 225\,000$ सेमी $= 2.25$ किमी.
झील: $2$ किमी $= 200\,000$ जमीन पर सेमी, तो
$200\,000 \div 25\,000 = 8$ मानचित्र पर सेमी.
\end{solution}

\begin{solution}{exo:g7:prop:7}
$4.3$ एम $= 430$ सेमी, और $430 \div 43 = 10$ सेमी।
\end{solution}

\begin{solution}{exo:g7:prop:8}
चॉकलेट प्रति सेवारत: $240 \div 8 = 30$ जी। साथ $300$ जी:
$300 \div 30 = 10$ बिलकुल परोसना।
\end{solution}

\begin{solution}{exo:g7:prop:9}
\emph{1.} छूट: $20$ यूरो, यानी\ $\frac{20}{80} = 25\,\%$ का
मूल कीमत.

\emph{2.} का उदय $20$ अब यूरो की तुलना की जाती है \emph{नया}
संदर्भ $60$: $\frac{20}{60} = \frac13 \approx 33\,\%$. एक प्रतिशत
हमेशा संपूर्ण को संदर्भित करता है; सब कुछ बदल गया, इसलिए प्रतिशत भी बदल गया
भी.
\end{solution}

\begin{solution}{exo:g7:prop:10}
संपूर्ण कॉलम से गुणांक: $\frac{17.5}{10} = 1.75$. तब
$ y = 4 \times 1.75 = 7$ और $ x = \frac{10.5}{1.75} = 6$.
\end{solution}

\begin{solution}{exo:g7:prop:11}
\emph{1.} एक प्रति के बाद: $10 \times 0.8 = 8$ सेमी। दो के बाद:
$8 \times 0.8 = 6.4$ सेमी।

\emph{2.} दूसरी कमी पहले से कम की गई प्रतिलिपि पर लागू होती है,
मूल से नहीं: कारक कई गुना बढ़ जाते हैं,
$0.8 \times 0.8 = 0.64$, तो दो प्रतियाँ कम हो जाती हैं $64\,\%$ का
मूल --- नहीं $60\,\%$.
\end{solution}

\begin{solution}{pb:g7:prop:1}
\textbf{1.} ऊंचाइयां और छायाएं हैं \omterm{def:g6:propdata:def}{आनुपातिक}, तो द्वारा
क्रॉस नियम (\cref{thm:g7:prop:cross}), छड़ी के स्तंभ के साथ
$(1,\ 0.4)$ और पेड़ का $(h,\ 3.2)$:
$0.4 \times h = 1 \times 3.2$, इस तरह
$ h = \frac{3.2}{0.4} = 8$ एम।

\textbf{2.} मीनार: $\frac{26}{0.4} = 65$ एम। बच्चे की छाया:
$1.5 \times 0.4 = 0.6$ मी (छाया $=$ ऊंचाई $\times $ द
पल का गुणांक $0.4$).

\textbf{3.} जैसे ही सूर्य आकाश में घूमता है, गुणांक
ऊंचाइयों को छाया परिवर्तनों से जोड़ना; केवल माप लिया गया
एक ही क्षण में एक ही गुणांक साझा करें।

\textbf{4.} उस क्षण गुणांक है $1$: हर ऊंचाई
उसकी छाया के बराबर है. पिरामिडका शीर्ष इसलिए है
$147$ मी ऊँचा --- महान की ऊँचाई \omterm{def:g5:solids:def}{पिरामिड} (इसकी छाया है
टिप, शीर्ष के नीचे के बिंदु से मापा जाता है, जिसकी सभी को आवश्यकता होती है)।

\textbf{5.} स्तम्भ \omterm{def:g3:division:remainder}{भागफल}
$\frac{\text{shadow}}{\text{height}}$ --- द समानता
क्षण का गुणांक --- प्रत्येक ऊर्ध्वाधर के लिए सामान्य है
वस्तु: यही वह चीज़ है जो ऊँचाई की तालिका बनाती है
छाया ए आनुपातिकता तालिका (\cref{def:g7:prop:table}).

\textbf{6.} सूरज की किरणें आती हैं \omterm{def:g6:lines:perp}{समानांतर}. एक पर \emph{समतल} पृथ्वी,
दो \omterm{def:g6:lines:perp}{समानांतर} किरणें दो ऊर्ध्वाधर छड़ियों पर एक साथ प्रहार करेंगी
कोण: दोनों डालेंगे \omterm{def:g6:propdata:def}{आनुपातिक} छाया --- या तो दोनों नहीं
छाया, या दोनों छाया। बिना छाया वाली एक छड़ी (साइने) और
एक छाया के साथ (अलेक्जेंड्रिया), एक ही पल में, है
समतल पृथ्वी पर असंभव: दोनों ऊर्ध्वाधरों को इंगित करना चाहिए
विभिन्न दिशाएँ---सतह वक्र।

\textbf{7.} ड्राइंग पर, साइने वर्टिकल सीधे इंगित करता है
सूरज पर; अलेक्जेंड्रिया ऊर्ध्वाधर कोण से झुका हुआ है
शहर की दो दिशाओं के बीच, केंद्र से देखा गया। के बाद से
किरणें हैं \omterm{def:g6:lines:perp}{समानांतर}, वह झुकाव बिल्कुल वैसा ही है $7.2^\circ $ मापा गया
अलेक्जेंड्रिया में किरण और छड़ी के बीच।

\textbf{8.} $\frac{7.2}{360} = \frac{72}{3600} = \frac{1}{50}$:
पूर्ण चक्कर का एक पचासवाँ भाग।

\textbf{9.} चाप की लंबाई हैं \omterm{def:g6:propdata:def}{आनुपातिक} कोणों के लिए: यदि
$7.2^\circ $ --- मोड़ का पचासवां हिस्सा --- से मेल खाता है
$800$ किमी, पूर्ण मोड़ से मेल खाती है
\[
50 \times 800 = 40\,000 \text{ km}.
\]

\textbf{10.} \omterm{def:g4:geometry:circle}{व्यास} $= \frac{\text{circumference}}{\pi}
\approx \frac{40\,000}{3.14} \approx 12\,700$ किमी --- विरुद्ध
आधुनिक $12\,742$ किमी: एक प्रतिशत के अंदर तक सही,
एक छड़ी के साथ.

\textbf{11.} $40\,000$ किमी $= 4\,000\,000\,000$ सेमी, और
$4\,000\,000\,000 \div 40\,000\,000 = 100$ सेमी $= 1$ का मी
\omterm{def:g6:measure:perimeter}{परिधि}. \omterm{def:g4:geometry:circle}{व्यास}: $100 \div 3.14 \approx 32$ सेमी --- ए
सुंदर डेस्क ग्लोब.

\textbf{12.} $4.8$ किमी $= 480\,000$ सेमी, और
$480\,000 \div 40\,000\,000 = 0.012$ सेमी $= 0.12$ मिमी. द
आल्प्स का सबसे ऊंचा पर्वत एक मिली\omterm{def:g4:fractions:def}{मीटर} का दसवां हिस्सा है
मी\omterm{def:g4:numbers:round}{टर-गोल} ग्लोब: धूल का एक कण। इस पर \omterm{def:g7:prop:scale}{पैमाना} पृथ्वी है
अधिकांश पॉलिश गेंदों की तुलना में अधिक चिकनी।

\textbf{13.} $40\,000 \div 40 = 1\,000$ दिन, और
$1\,000 \div 365 \approx 2.7$: लगभग तीन साल पैदल चलना।

\textbf{14.} $\frac{10}{6\,370} \approx 0.0016$, वह इसके बारे में है
$0.2\,\%$ (ज्यादा ठीक $0.16\,\%$): वातावरण एक है
ग्रह पर कानाफूसी-पतली त्वचा।

\textbf{15.} गलती: $42\,000 - 40\,000 = 2\,000$ किमी, और
$\frac{2\,000}{40\,000} = \frac{5}{100} = 5\,\%$. एक वाक्य:
एक छड़ी, एक कुआँ, एक मापी हुई सड़क और एक के साथ समानता,
एराटोस्थनीज़ ने एक ग्रह को कुछ प्रतिशत के भीतर मापा ---
गणित बहुत कम समय में बहुत दूर तक यात्रा करता है।
\end{solution}
