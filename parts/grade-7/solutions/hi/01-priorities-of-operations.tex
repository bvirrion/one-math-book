\section*{अध्याय \ref{ch:g7:priorities} --- संचालन की प्राथमिकताएँ}

\begin{solution}{exo:g7:priorities:1}
$5 + 3 \times 6 = 5 + 18 = 23$.

$(5 + 3) \times 6 = 8 \times 6 = 48$.

$30 - 12 \div 4 = 30 - 3 = 27$.

$(30 - 12) \div 4 = 18 \div 4 = 4.5$.
\end{solution}

\begin{solution}{exo:g7:priorities:2}
$24 - 6 + 2 = 18 + 2 = 20$ (बाएं से दायां)।

$24 \div 6 \times 2 = 4 \times 2 = 8$ (बाएं से दायां)।

$24 - 6 \times 2 = 24 - 12 = 12$ (पहले गुणा)।

$24 \div (6 \times 2) = 24 \div 12 = 2$.
\end{solution}

\begin{solution}{exo:g7:priorities:3}
$7 + 3 \times (10 - 6) = 7 + 3 \times 4 = 7 + 12 = 19$.

$40 - (2 + 3) \times 4 = 40 - 5 \times 4 = 40 - 20 = 20$.

$60 \div \bigl(2 \times (7 - 4)\bigr) = 60 \div (2 \times 3)
= 60 \div 6 = 10$.
\end{solution}

\begin{solution}{exo:g7:priorities:4}
$\dfrac{18 + 6}{8} = \dfrac{24}{8} = 3$; \quad
$\dfrac{30}{7 - 2} = \dfrac{30}{5} = 6$; \quad
$\dfrac{5 \times 8}{4 + 6} = \dfrac{40}{10} = 4$.
\end{solution}

\begin{solution}{exo:g7:priorities:5}
$8 \times 102 = 8 \times (100 + 2) = 800 + 16 = 816$.

$25 \times 99 = 25 \times (100 - 1) = 2500 - 25 = 2475$.

$14 \times 12 + 14 \times 8 = 14 \times (12 + 8) = 14 \times 20 = 280$.

$37 \times 45 - 37 \times 35 = 37 \times (45 - 35) = 37 \times 10 =
370$.
\end{solution}

\begin{solution}{exo:g7:priorities:6}
``द \omterm{ex:g1:subtraction:difference}{अंतर} का $50$ और यह \omterm{def:g2:mult:def}{उत्पाद} का $6$ द्वारा $7$'':
$50 - 6 \times 7 = 50 - 42 = 8$ (कोष्ठक की आवश्यकता नहीं है, प्राथमिकता है
काम)।

``द \omterm{def:g2:mult:def}{उत्पाद} की \omterm{def:g1:addition:def}{जोड़} का $4$ और $5$ से \omterm{ex:g1:subtraction:difference}{अंतर} का $10$ और
$8$'': $(4 + 5) \times (10 - 8) = 9 \times 2 = 18$.
\end{solution}

\begin{solution}{exo:g7:priorities:7}
$3 \times 2.50 + 2 \times 1.20 = 7.50 + 2.40 = 9.90$ यूरो.
\end{solution}

\begin{solution}{exo:g7:priorities:8}
(ए) कोई कोष्ठक नहीं: $4 + 2 \times 5 - 1 = 4 + 10 - 1 = 13$.

(बी) $(4 + 2) \times 5 - 1 = 30 - 1 = 29$.

(सी) $(4 + 2) \times (5 - 1) = 6 \times 4 = 24$.
\end{solution}

\begin{solution}{exo:g7:priorities:9}
$ k = 1$: $12 + 4 \times 1 = 16$ और $16 \times 1 = 16$ --- वे सहमत हैं
(इसीलिए $ k = 1$ एक ख़राब परीक्षा है!) $ k = 2$:
$12 + 4 \times 2 = 20$ लेकिन $16 \times 2 = 32$: अलग। तो दोनों
अभिव्यक्तियाँ हैं \emph{नहीं} सामान्य तौर पर बराबर. गलती
$12 + 4 \times k = 16 \times k $ जोड़ने से आता है $12 + 4$ प्रथम,
की प्राथमिकता को नजरअंदाज कर रहे हैं $\times $.
\end{solution}

\begin{solution}{exo:g7:priorities:10}
\emph{1.} समूह मूल्य: $12 + 6 \times n $. के लिए $ n = 5$:
$12 + 30 = 42$ यूरो.

\emph{2.} व्यक्तिगत कीमत: $9n $. तुलना करना: $ n = 3$: $30$ बनाम $27$
(व्यक्तिगत सस्ता); $ n = 4$: $36$ बनाम $36$ (बराबर); $ n = 5$: $42$ बनाम
$45$ (समूह सस्ता). से $5$ लोग चालू हैं, समूह दर जीतती है
(प्रत्येक अतिरिक्त व्यक्ति की लागत होती है $6$ के बजाय $9$, इसलिए \omterm{ex:g1:subtraction:difference}{अंतर} केवल बढ़ता है)।
\end{solution}

\begin{solution}{exo:g7:priorities:11}

24 = 1 \times 2 \times 3 \times 4, \qquad

\begin{solution}{pb:g7:priorities:1}
\textbf{1.} $7 \times 102 = 7 \times (100 + 2) = 700 + 14 =
714$; \quad $9 \times 98 = 9 \times (100 - 2) = 900 - 18 = 882$.

\textbf{2.} $17 \times 13 + 17 \times 7 = 17 \times (13 + 7) =
17 \times 20 = 340$; \quad $45 \times 101 - 45 = 45 \times (101
- 1) = 45 \times 100 = 4\,500$.

\textbf{3.} $5 \times 87 = 870 \div 2 = 435$. वैधता:
$5 = 10 \div 2$, इसलिए गुणा करना $5$ से गुणा हो रहा है $10$
फिर से विभाजित करना $2$ (के लिए $\times $ और $\div $ पर
समान स्तर, बाएँ से दाएँ, बाकी काम करता है)।

\textbf{4.} $25 \times 32 = 3\,200 \div 4 = 800$, क्योंकि
$25 = 100 \div 4$. और
$99 \times 47 = (100 - 1) \times 47 = 4\,700 - 47 = 4\,653$:
पर वितरण \omterm{ex:g1:subtraction:difference}{अंतर}
(\cref{thm:g7:priorities:distributivity}).

\textbf{5.} $8 \times (5 + 3) = 64$ और $8 \times 5 + 8 \times 3
= 40 + 24 = 64$: वे मेल खाते हैं --- यही वितरणात्मकता है। लेकिन
$8 \times (5 \times 3) = 8 \times 15 = 120$, जबकि
$(8 \times 5) \times (8 \times 3) = 40 \times 24 = 960$: नहीं
मिलान. कारक फैला हुआ है \emph{ए की शर्तें \omterm{def:g1:addition:def}{जोड़}}, नहीं
ए के कारकों पर \omterm{def:g2:mult:def}{उत्पाद}.

\textbf{6.}
\begin{center}
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{ccl}
$13$ & $24$ & kept ($13$ \omterm{def:g3:numbers:evenodd}{विषम}) \\
$6$ & $48$ & crossed out ($6$ \omterm{def:g3:numbers:evenodd}{यहां तक की}) \\
$3$ & $96$ & kept \\
$1$ & $192$ & kept \\
\end{tabular}
\end{center}
\omterm{def:g1:addition:def}{जोड़} जीवित सही संख्याओं में से: $24 + 96 + 192 = 312$, सचमुच $13 \times 24 = 312$.

\textbf{7.} संयोग $21$: $21, 10, 5, 2, 1$; दोहरीकरण $17$:
$17, 34, 68, 136, 272$. \omterm{def:g3:numbers:evenodd}{यहां तक की} बायां $10$ और $2$ काट दिए गए हैं;
जीवित बचे लोगों $17 + 68 + 272 = 357$. जाँच करना:
$21 \times 17 = 357$.

\textbf{8.} संयोग $16$: $16, 8, 4, 2, 1$; दोहरीकरण $25$:
$25, 50, 100, 200, 400$. अंतिम को छोड़कर प्रत्येक बायां नंबर $1$
है \omterm{def:g3:numbers:evenodd}{यहां तक की}: एक पंक्ति बची है, और उत्तर है $400$. एक बायां
शुद्ध दोहरीकरण से बना स्तंभ केवल अपनी अंतिम पंक्ति रखता है ---
विधि अकेले दोहरीकरण तक कम हो जाती है।

\textbf{9.} $13 = 12 + 1$, तो वितरण द्वारा
$13 \times 24 = 12 \times 24 + 1 \times 24$; और
$12 \times 24 = 6 \times 48$ एक कारक को \omterm{def:g3:division:half}{आधा} करने के बाद से
अन्य पत्तों को दोगुना करना a \omterm{def:g2:mult:def}{उत्पाद} अपरिवर्तित
($12 \times 24 = 6 \times 2 \times 24 = 6 \times 48$). \omterm{def:g3:numbers:evenodd}{विषम}
बायां नंबर $13$ सही संख्या की एक प्रति को विभाजित करें:
पहली पंक्ति के बैंक $24$.

\textbf{10.} $6 \times 48 = 3 \times 96$ (\omterm{def:g3:division:half}{आधा} - दोगुना, नहीं
बैंकिंग: $6$ है \omterm{def:g3:numbers:evenodd}{यहां तक की}). $3 \times 96 = 2 \times 96 + 96 =
1 \times 192 + 96$: द \omterm{def:g3:numbers:evenodd}{विषम} $3$ बैंकों $96$. अंत में
$1 \times 192 = 192$: द \omterm{def:g3:numbers:evenodd}{विषम} $1$ बैंकों $192$ और मेज
समाप्त होता है. कुल बैंकांकित: $24 + 96 + 192 = 312$. एक पंक्ति इसकी बैंक करती है
सही संख्या ठीक उसी समय है जब उसका बायां नंबर है \omterm{def:g3:numbers:evenodd}{विषम} ---तो
नुस्खा ``विषम-बाएँ पंक्तियाँ रखें और \omterm{def:g1:addition:def}{जोड़}ें'' ठीक यही है
वितरण का बहीखाता.

\textbf{11.} $1 + 4 + 8 = 13$. के लिए $21$, जीवित पंक्तियाँ थीं
के $17$, $68 = 4 \times 17$ और $272 = 16 \times 17$: द
दोगुना $1 + 4 + 16 = 21$.

\textbf{12.} $25 = 16 + 8 + 1$ और $42 = 32 + 8 + 2$.

\textbf{13.} सबसे बड़ी दोहरीकरण संख्या लें जो इससे अधिक न हो
इसे लक्षित करें और घटाएं; जो बचता है वह छोटा है, और छोटा है
अभी उपयोग की गई दोहरीकरण संख्या से (अन्यथा एक बड़ी संख्या)।
फिट हो गया होगा)। दोहराते हुए, \omterm{def:g3:division:remainder}{शेष} सख्ती से सिकुड़ता है
और पहुंचना ही चाहिए $0$: प्रक्रिया रुक जाती है, और दोहरीकरण का उपयोग किया जाता है
सभी अलग हैं. के लिए $100$: $64$ फिट बैठता है ($36$ बाएं), $32$ फिट बैठता है
($4$ बाएं), $4$ फिट बैठता है ($0$ बाएं): $100 = 64 + 32 + 4$.

\textbf{14.} की दोहरीकरण तालिका $35$: $1 \to 35$, $2 \to 70$,
$4 \to 140$, $8 \to 280$, $16 \to 560$. तब से
$19 = 16 + 2 + 1$, उन पंक्तियों को चुनें:
$560 + 70 + 35 = 665$. जाँच करना: $19 \times 35 = 665$.

\textbf{15.} मुंशी और किसान के समान तीन:
$24 + 96 + 192$ --- का दोगुना होना $24$ द्वारा चयनित
$13 = 1 + 4 + 8$. संख्याओं को इस प्रकार लिखना रकम दोहरीकरण का है
बिल्कुल उनके साथ लिखना $0$ रेत $1$ एस; पूरी कहानी बताई गई है
में \cref{pb:g8:powers:1}.
\end{solution}
