\section*{अध्याय \ref{ch:g7:areas} --- क्षेत्र और आयतन}

\begin{solution}{exo:g7:areas:1}
$8 \times 5 = 40$ सेमी $^2$; \qquad $6.5 \times 4 = 26$ एम $^2$.
\end{solution}

\begin{solution}{exo:g7:areas:2}
$ A = 10 \times 4 = 40$ सेमी $^2$. उत्तर नहीं है
$10 \times 6 = 60$ क्योंकि $6$ सेमी भुजा तिरछी है: सूत्र
का उपयोग करता है \emph{ऊंचाई} (\omterm{def:g6:lines:perp}{सीधा} दूरी), जो है $4$ सेमी।
\end{solution}

\begin{solution}{exo:g7:areas:3}
$\frac{12 \times 7}{2} = 42$ सेमी $^2$; \quad
$\frac{9 \times 6}{2} = 27$; \quad
$\frac{5.5 \times 2}{2} = 5.5$ एम $^2$.
\end{solution}

\begin{solution}{exo:g7:areas:4}
डिस्क: $\pi \times 5^2 = 25\pi \approx 79$ सेमी $^2$. की आधी डिस्क \omterm{def:g3:shapes:shapes}{RADIUS}
$10$: $\frac{\pi \times 10^2}{2} = 50\pi \approx 157$ सेमी $^2$.
\end{solution}

\begin{solution}{exo:g7:areas:5}
\omterm{def:g6:measure:perimeter}{परिधि}: $2\pi \times 3 = 6\pi \approx 18.8$ सेमी (एक लंबाई, सेमी में)।
\omterm{def:g6:measure:area}{क्षेत्र}: $\pi \times 3^2 = 9\pi \approx 28.3$ सेमी $^2$ (एक सतह, में
सेमी $^2$).
\end{solution}

\begin{solution}{exo:g7:areas:6}
\omterm{def:g7:areas:prism}{चश्मे}: $24 \times 10 = 240$ सेमी $^3$. \omterm{def:g7:areas:prism}{सिलेंडर}:
$\pi \times 3^2 \times 8 = 72\pi \approx 226$ सेमी $^3$.
\end{solution}

\begin{solution}{exo:g7:areas:7}
$\frac{8 \times h}{2} = 28$, इसलिए $4h = 28$ और $ h = 7$ सेमी।
\end{solution}

\begin{solution}{exo:g7:areas:8}
तीन \omterm{def:g2:mult:def}{उत्पाद}ों सहमत हूं क्योंकि प्रत्येक व्यक्ति गणना करता है \emph{वही}
मात्रा --- \omterm{def:g6:measure:area}{क्षेत्र} \omterm{def:g3:shapes:shapes}{त्रिकोण} का. इससे व्यवहारिक जांच होती है
निर्माणों का, और दूसरे से ऊंचाई की गणना करने का एक तरीका
आधार--ऊंचाई \omterm{def:g1:addition:def}{जोड़}ी (में प्रयुक्त) \cref{exo:g7:areas:12}).
\end{solution}

\begin{solution}{exo:g7:areas:9}
दो \omterm{def:g3:shapes:shapes}{त्रिकोण} हैं \omterm{def:g6:measure:area}{क्षेत्र}ों $\frac{30 \times 20}{2} = 300$ एम $^2$
और $\frac{18 \times 20}{2} = 180$ एम $^2$: कुल $480$ एम $^2$. सामान्य
सूत्र सुझाया गया: के लिए \omterm{def:g6:lines:perp}{समानांतर} दोनों पक्ष $ a $ और $ b $ दूरी पर $ h $,
\[
A = \frac{(a + b) \times h}{2}
\]
--- समलम्बाकार सूत्र (यहाँ $\frac{(30+18)\times 20}{2} = 480$).
\end{solution}

\begin{solution}{exo:g7:areas:10}
पानी \omterm{def:g6:measure:volume}{आयतन}: $\pi \times 6^2 \times 15 = 540\pi \approx 1\,696$
सेमी $^3$. \omterm{def:g5:solids:def}{डिब्बा} आधार: $18 \times 12 = 216$ सेमी $^2$. ऊँचाई तक पहुँच गया:
$\frac{540\pi}{216} = 2.5\pi \approx 7.9$ सेमी।
\end{solution}

\begin{solution}{exo:g7:areas:11}
त्रिज्या $15$ और $20$ सेमी। \omterm{def:g6:measure:area}{क्षेत्र}ों: $225\pi \approx 707$ सेमी $^2$ और
$400\pi \approx 1257$ सेमी $^2$. प्रति कीमत $100$ सेमी $^2$:
$\frac{9}{7.07} \approx 1.27$ यूरो और
$\frac{14}{12.57} \approx 1.11$ यूरो. बड़ा पिज़्ज़ा बेहतर है
सौदा --- \omterm{def:g6:measure:area}{क्षेत्र}ों के साथ बढ़ें \emph{वर्ग} की \omterm{def:g4:geometry:circle}{व्यास}.
\end{solution}

\begin{solution}{exo:g7:areas:12}
पैरों के साथ: $ A = \frac{6 \times 8}{2} = 24$ सेमी $^2$. के साथ
आधार के रूप में कर्ण: $24 = \frac{10 \times h}{2} = 5h $, इसलिए
$ h = 4.8$ सेमी।
\end{solution}

\begin{solution}{pb:g7:areas:1}
\textbf{1.} वर्ग: $8 \times 8 = 64$. आयत:
$5 \times 13 = 65$. वही चार टुकड़े ढंके हुए दिखाई देते हैं $64$
एक आकृति में वर्ग इकाइयाँ और $65$ दूसरे में: की एक इकाई
\omterm{def:g6:measure:area}{क्षेत्र} कुछ भी नहीं से बनाया गया है --- माना जाता है।

\textbf{2.} प्रत्येक त्रिभुज: $\frac{8 \times 3}{2} = 12$. प्रत्येक
समलम्बाकार: $\frac{(5 + 3) \times 5}{2} = 20$.

\textbf{3.} $12 + 12 + 20 + 20 = 64$: कुल टुकड़े $64$,
ताकि वे वर्ग को सटीक रूप से भर सकें --- और कर सकें \emph{नहीं} भरें
द $65$-इकाई आयत. आयत चित्र में एक होना चाहिए
छेद.

\textbf{4.} वही ढलान का मतलब होगा $3$ आर-पार $8$
\omterm{def:g6:propdata:def}{आनुपातिक} को $2$ आर-पार $5$; पार करना \omterm{def:g2:mult:def}{उत्पाद}ों:
$3 \times 5 = 15$ और $2 \times 8 = 16$. बराबर नहीं: द
कर्ण और तिरछा किनारा करते हैं \emph{नहीं} पंक्ति बनायें। द
आयत का ``विकर्ण'' एक झूठ है --- दो थोड़ा सा
विभिन्न ढलानें उथले कोण पर मिलती हैं।

\textbf{5.} झूठे विकर्ण के साथ चार टुकड़े एक छोड़ते हैं
लंबा, बेहद पतला गैप --- बहुत के आकार का एक ज़ुल्फ़
समतल \omterm{def:g7:central:parallelogram}{चतुर्भुज} --- किसका \omterm{def:g6:measure:area}{क्षेत्र} बिल्कुल गायब है
$65 - 64 = 1$ वर्ग इकाई. एक विकर्ण के साथ फैला हुआ
$13$ इकाई लंबी है, इसकी औसत चौड़ाई लगभग है
$1 \div 13 \approx 0.08$ इकाइयाँ: वास्तविक चित्र पर, उससे भी पतली
पेंसिल की \omterm{def:g6:lines:objects}{रेखा} जो इसे छुपाती है।

\textbf{6.} प्रत्येक त्रिभुज का आधार होता है $6$ सेमी और ऊंचाई $4$ सेमी ---
के बीच की दूरी समानताएं, जहां भी शीर्ष बैठता है ---
तो प्रत्येक \omterm{def:g6:measure:area}{क्षेत्र} है $\frac{6 \times 4}{2} = 12$ सेमी $^2$
(\cref{thm:g7:areas:triangle}). शीर्ष के साथ सरकना
\omterm{def:g6:lines:perp}{समानांतर} आकार बदलता है, आधार या ऊंचाई कभी नहीं:
बराबर \omterm{def:g6:measure:area}{क्षेत्र}ों हमेशा के लिए।

\textbf{7.} विकर्ण समलम्ब \omterm{def:g7:central:parallelogram}{चतुर्भुज} को एक त्रिभुज में काटता है
आधार के साथ $ B $ और एक आधार के साथ $ b $, दोनों की ऊंचाई $ h $ (द
के बीच की दूरी \omterm{def:g6:lines:perp}{समानांतर} पक्ष):
\[
A = \frac{B \times h}{2} + \frac{b \times h}{2}
= \frac{(B + b) \times h}{2} .
\]

\textbf{8.} $ A = \frac{(9 + 5) \times 4}{2} = \frac{56}{2} =
28$ सेमी $^2$.

\textbf{9.} का औसत \omterm{def:g6:lines:perp}{समानांतर} पक्ष है
$\frac{B + b}{2}$, तो (औसत) $\times h =
\frac{(B+b) \times h}{2}$: वही सूत्र, गुणनखंडित
अलग ढंग से. जांचें: औसत $= \frac{9+5}{2} = 7$, और
$7 \times 4 = 28$ सेमी $^2$.

\textbf{10.} तिरछे ढेर में बिल्कुल वही कार्ड होते हैं
--- वही परतें, प्रत्येक परत एक जैसी \omterm{def:g6:measure:area}{क्षेत्र} --- तो वैसा ही
\omterm{def:g6:measure:volume}{आयतन}. में $ V = B \times h $, ऊंचाई मापी जानी चाहिए
\emph{\omterm{def:g6:lines:perp}{सीधा}} आधार तक (ऊर्ध्वाधर ऊँचाई
स्टैक), तिरछा नहीं: बिल्कुल वैसा ही \omterm{def:g7:central:parallelogram}{चतुर्भुज}'एस
\omterm{def:g6:measure:area}{क्षेत्र} चाहता है \omterm{def:g6:lines:perp}{सीधा} ऊँचाई, तिरछी ओर नहीं
(\cref{exo:g7:areas:2}), एक आयाम ऊपर।

\textbf{11.} अँगूठी $= \pi \times 5^2 - \pi \times 3^2 = 25\pi -
9\pi = 16\pi \approx 50$ एम $^2$ (\cref{thm:g7:areas:disk}).

\textbf{12.} की एक डिस्क \omterm{def:g6:measure:area}{क्षेत्र} $16\pi $ है \omterm{def:g3:shapes:shapes}{RADIUS} $4$, तब से
$\pi \times 4^2 = 16\pi $. त्रिज्या $3$, $4$, $5$: सबसे ज्यादा
गणित में प्रसिद्ध ट्रिपल, अभिनीत
\cref{ch:g8:pythagoras}.

\textbf{13.} एक $40$ सेमी पिज्जा: \omterm{def:g3:shapes:shapes}{RADIUS} $20$,
\omterm{def:g6:measure:area}{क्षेत्र} $\pi \times 400 \approx 1\,257$ सेमी $^2$. दो $28$ सेमी
पिज़्ज़ा: \omterm{def:g3:shapes:shapes}{RADIUS} $14$, एक साथ
$2 \times \pi \times 196 = 392\pi \approx 1\,232$ सेमी $^2$. द
सिंगल लार्ज पिज़्ज़ा है \emph{अधिक} दो माध्यमों की तुलना में पिज़्ज़ा।

\textbf{14.} स्केलिंग \omterm{def:g3:shapes:shapes}{RADIUS} द्वारा $1.4$ तराजू  \omterm{def:g6:measure:area}{क्षेत्र} द्वारा
$1.4^2 = 1.96$ --- लगभग दोगुना. सटीक कारक यह है
वह संख्या जिसका वर्ग है $2$, के बारे में $1.414$: विकर्ण संख्या
का \cref{ex:g8:pythagoras:diagonal}. (अंगूठे का नियम: \omterm{def:g3:division:half}{आधा} पुनः
जितना चौड़ा पिज़्ज़ा से दोगुना है, लगभग।)

\textbf{15.} $13 \times 13 = 169$, लेकिन $8 \times 21 = 168$:
इस बार एक वर्ग इकाई \emph{गायब} --- टुकड़े
खाली जगह छोड़ने के बजाय एक पतले टुकड़े में ओवरलैप करें। साथ
लगातार शर्तें $3, 5, 8, 13, 21$ का
हेमचन्द्र--फाइबोनैचि अनुक्रम, द \omterm{def:g2:mult:def}{उत्पाद}ों एक के बीच पलटें
वर्ग से अधिक तथा एक कम ($5 \times 13 = 65 =
8^2 + 1$, तब $8 \times 21 = 168 = 13^2 - 1$): जादूगर
बारी-बारी से एक इकाई ``लाभ'' और ``खोती'' है। \omterm{def:g6:measure:area}{क्षेत्र}ों कभी झूठ मत बोलो:
टुकड़े कुल मिलाकर $64$ ढकना $64$, वे जहां भी स्लाइड करते हैं --- प्रत्येक
गुम या अतिरिक्त इकाई ढलान की प्रतीक्षा में एक स्लिवर में छिपी हुई है
इसे उजागर करने के लिए जाँच करें।
\end{solution}
