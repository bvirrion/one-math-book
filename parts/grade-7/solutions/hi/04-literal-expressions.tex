\end{solution}

\begin{solution}{exo:g7:literal:2}
$5ab = 5 \times a \times b $; \quad
$3(x+2) = 3 \times (x + 2)$; \quad

\end{solution}

\begin{solution}{exo:g7:literal:3}

$2 \times 9 - 3 = 15$; \quad $10 - 6 = 4$.
\end{solution}

\begin{solution}{exo:g7:literal:4}
$3a + 2b = 12 + 3 = 15$; \qquad
$ ab + 5 = 4 \times 1.5 + 5 = 6 + 5 = 11$.
\end{solution}

\begin{solution}{exo:g7:literal:5}
$8x + 5x = 13x $; \quad
$9a - 4a + a = 6a $; \quad
$6y + 2 + 3y + 7 = 9y + 9$; \quad

\begin{solution}{exo:g7:literal:6}

\begin{solution}{exo:g7:literal:7}
$2n $: जुड़वा; \ $ n + 2$: दो से अधिक $ n $; \ $ n^2$: वर्ग;
\ $\frac n2$: द \omterm{def:g3:division:half}{आधा}.
\end{solution}

\begin{solution}{exo:g7:literal:8}
$3(x+4) = 3x + 12$: \emph{सत्य} हरएक के लिए $ x $ (वितरणशीलता ---
यह एक प्रमाण है, किसी परीक्षण की आवश्यकता नहीं है)।

$2x + 5 = 7x $: परीक्षा $ x = 2$: बाएं $9$, सही $14$ --- \emph{असत्य} में
सामान्य (यह केवल के लिए है $ x = 1$).

$ x + x = x^2$: परीक्षा $ x = 3$: बाएं $6$, सही $9$ --- \emph{असत्य} में
सामान्य ($ x + x = 2x $).
\end{solution}

\begin{solution}{exo:g7:literal:9}
\emph{1.} लंबाई: $ w + 3$. \omterm{def:g6:measure:perimeter}{परिधि}:
\[
P = 2\bigl(w + (w + 3)\bigr) = 2(2w + 3) = 4w + 6 .
\]

\emph{2.} के लिए $ w = 5$: सीधे तौर पर, भुजाएँ हैं $5$ और $8$, इसलिए
$ P = 2 \times 13 = 26$ सेमी; सूत्र के साथ, $4 \times 5 + 6 = 26$ सेमी।
वे सहमत हैं.
\end{solution}

\begin{solution}{exo:g7:literal:10}
पत्र के साथ चरणों का पालन करें $ n $:
\[
n \ \to\ n + 6 \ \to\ 3(n + 6) = 3n + 18 \ \to\ 3n + 18 - 18 = 3n
\ \to\ \frac{3n}{n} = 3 .
\]
सबको मिलता है $3$.
\end{solution}

\begin{solution}{exo:g7:literal:11}
के तौर पर \omterm{ex:g1:subtraction:difference}{अंतर}: $ A = a^2 - b^2$. L को दो भागों में काटकर
आयतें: ऊँचाई की एक पूर्ण-चौड़ाई वाली पट्टी $ a - b $ और, इसके शीर्ष पर, ए
संकरा आयत:
\[
A = a(a - b) + b(a - b)
\]
(निचला आयत $ a $ चौड़ा और $ a - b $ लंबा, शीर्ष आयताकार $ a - b $
चौड़ा और $ b $ लंबा प्रकार देता है $ a(a-b) + (a-b)b $ --- जो उसी)।
के लिए $ a = 5$, $ b = 2$: $ a^2 - b^2 = 25 - 4 = 21$, और
$5 \times 3 + 2 \times 3 = 15 + 6 = 21$. दोनों अभिव्यक्तियाँ सहमत हैं ---
वे हमेशा यही करते हैं: यही पहचान है $ a^2 - b^2 = (a + b)(a - b)$ का
\cref{ch:g9:algebra} छद्मवेश में.
\end{solution}

\begin{solution}{pb:g7:literal:1}
\textbf{1.} आंकड़ों $1$ को $4$ ज़रूरत $4$, $7$, $10$, $13$
मेल खाता है.

\textbf{2.} एक नया वर्ग पिछले वर्ग के साथ एक भुजा साझा करता है
चित्रा, तो केवल $3$ उसके जैसा $4$ पक्ष नए मैच हैं. से
आंकड़ा $4$ ($13$ माचिस), छह और वर्गों की लागत
$6 \times 3 = 18$: आकृति $10$ आवश्यकताओं $13 + 18 = 31$ मेल खाता है.

\textbf{3.} बाईं ओर से पंक्ति को पढ़ना: यह एक से शुरू होती है
ऊर्ध्वाधर मिलान, और प्रत्येक $ n $ तब वर्ग एक योगदान देता है
ऊपर, एक नीचे और एक दाहिनी ओर --- $3$ प्रत्येक से मेल खाता है. कुल:
$3n + 1$; ``$+1$'' सबसे पहली ऊर्ध्वाधर छड़ी है। जाँच करना:
$ n = 1, 2, 3, 4$ देना $4, 7, 10, 13$.

\textbf{4.} $4 + 3(n - 1) = 4 + 3n - 3 = 3n + 1$: आमिर का
सूत्र बिल्कुल उसी घटी हुई अभिव्यक्ति तक विस्तारित होता है। पूर्ण
समझौता.

\textbf{5.} लीना: $2n + (n + 1) = 3n + 1$ --- फिर वही.
देखने के तीन तरीके, एक छोटा रूप। आकृति $100$:
$3 \times 100 + 1 = 301$ मेल खाता है.

\textbf{6.} गिनती: $3$, $5$, $7$ मेल खाता है. प्रत्येक नया \omterm{def:g3:shapes:shapes}{त्रिकोण}
मौजूदा पक्ष पर झुक जाता है और केवल जोड़ता है $2$ मैच, प्रारंभ
से $3$: आकृति $ n $ आवश्यकताओं $3 + 2(n - 1) = 2n + 1$ मेल खाता है.

\textbf{7.} पूर्ववत $2n + 1 = 49$: पहली माचिस की पत्तियाँ हटाना
$48$, और फिर प्रत्येक त्रिभुज का हिसाब होता है $2$:
$48 \div 2 = 24$ \omterm{def:g3:shapes:shapes}{त्रिकोण}.

\textbf{8.} की एक पंक्ति में $ n $ टेबल, प्रत्येक टेबल अपना शीर्ष प्रदान करती है
और नीचे की ओर ($2n $ सीटें), और दो अंतिम टेबल एक की पेशकश करते हैं
प्रत्येक अतिरिक्त पक्ष ($+2$): $2n + 2$ सीटें. जाँच करना: $4$, $6$, $8$
के लिए $ n = 1, 2, 3$.

\textbf{9.} एक पंक्ति: $2n + 2 \geq 30$ आवश्यकताओं $2n \geq 28$, इसलिए
$14$ टेबल. वही बंटवारा $14$ तालिकाओं को दो पंक्तियों में विभाजित करें
$7$: प्रत्येक पंक्ति में सीटें $2 \times 7 + 2 = 16$, एक साथ $32$ ---
एकल पंक्ति से दो अधिक, क्योंकि प्रत्येक नई पंक्ति दो लाती है
नया \emph{समाप्त होता है}. (प्रत्येक विभाजन से लाभ होता है $2$ सीटें; कैटरर्स को पता है।)

\textbf{10.} आकार का एल $ n $ है $ n $ बिंदु ऊपर जा रहे हैं और $ n $
पार जाकर, \omterm{def:g5:solids:def}{कोने} का बिंदु साझा किया गया: $ n + n - 1 = 2n - 1$ बिंदु.
आंकड़ों $1$, $2$, $3$: $1$, $3$, $5$ बिंदु --- \emph{अजीब
नंबर}. अगले के आसपास एक नेस्टेड, एलएस एक भरें
$ n \times n $ बिन्दुओं का वर्ग: की कहानी
\cref{pb:g8:equations:1}.

\textbf{11.} $3(n + 2) - 5 = 3n + 6 - 5 = 3n + 1$: दावा
सिद्ध फार्मूले को कम करता है, इसलिए यह हर किसी के लिए सही है $ n $ ---
बीजगणित द्वारा स्थापित, उदाहरणों द्वारा नहीं।

\textbf{12.} टेस्ट: $4 \geq 2$, $9 \geq 3$, $100 \geq 10$: सभी पास। लेकिन $ n = 0.5$ देता है $ n^2 = 0.25 < 0.5$: दावा विफल रहता है.
तीन परीक्षणों से कुछ भी साबित नहीं हुआ \emph{सभी} संख्याएँ (वे
केवल परेशानी ढूंढने में असफल); एकल प्रति उदाहरण
\emph{को सिद्ध करता} सामान्य दावा ग़लत है
(\cref{met:g7:literal:test}). प्रत्येक संख्या के लिए एक अक्षर होता है
--- दशमलव शामिल हैं.

\textbf{13.} \omterm{def:g6:measure:area}{क्षेत्र}: $2$ अंक $\to 2$; $3$ अंक $\to 4$;
$4$ अंक $\to 8$; $5$ अंक $\to 16$. पैटर्न फुसफुसाता है
``प्रत्येक बिंदु गिनती को दोगुना कर देता है'': $32$ \omterm{def:g6:measure:area}{क्षेत्र}ों के लिए अपेक्षित
$6$ अंक.

\textbf{14.} सावधानीपूर्वक गणना से पता चलता है $31$ \omterm{def:g6:measure:area}{क्षेत्र} --- नहीं $32$.
दोहरीकरण अनुमान ख़त्म हो गया है: यह चार जाँचों से बच गया
पांचवी फेल. (यह प्रतिउदाहरण काफी प्रसिद्ध है
एक नाम, ``सर्\omterm{def:g6:measure:area}{कल-क्षेत्र} \omterm{def:g5:solids:net}{जाल}''; कोई समान दूरी वाला शॉर्टकट नहीं
--- एक बिंदु पर मिलने वाले तीन तार \omterm{def:g6:measure:area}{क्षेत्र}ों का विलय करेंगे और
गिनती ख़राब करो.)

\textbf{15.} एक प्रमाण की आवश्यकता होगी \emph{संरचनात्मक}
कारण कि एक बिंदु जोड़ने से \omterm{def:g6:measure:area}{क्षेत्र} दोगुना हो जाता है --- और ऐसा है
कोई नहीं: दोहरीकरण छोटे मामलों का एक संयोग था। द
माचिस की तीली के फार्मूले सुरक्षित थे क्योंकि प्रत्येक को पढ़ा जाता था
निर्माण स्वयं (प्रत्येक वर्ग की लागत स्पष्ट रूप से दिखाई देती है $3$ मेल खाता है,
प्रत्येक तालिका स्पष्ट रूप से ऑफर करती है $2$ सीटें): संरचना, परीक्षण नहीं।
सिद्ध भविष्यवाणी: $3 \times 2\,026 + 1 = 6\,079$ मेल खाता है.
\end{solution}
