\section*{अध्याय \ref{ch:g7:stats} --- डेटा व्यवस्थित करना}

\begin{solution}{exo:g7:stats:1}
प्रत्येक चिह्न की गिनती:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|ccccc|c}
mark & $9$ & $11$ & $12$ & $14$ & $15$ & total \\
\hline
count & $4$ & $3$ & $7$ & $3$ & $3$ & $20$ \\
\end{tabular}
\end{center}
\end{solution}

\begin{solution}{exo:g7:stats:2}
आवृत्तियाँ: $\frac{4}{20} = 20\,\%$, $\frac{3}{20} = 15\,\%$,
$\frac{7}{20} = 35\,\%$, $15\,\%$, $15\,\%$. \omterm{def:g1:addition:def}{जोड़}:
$20 + 15 + 35 + 15 + 15 = 100\,\%$. \checkmark
\end{solution}

\begin{solution}{exo:g7:stats:3}
पहला सर्वेक्षण: $\frac{30}{50} = 0.6 = 60\,\%$. दूसरा:
$\frac{45}{90} = 0.5 = 50\,\%$. पहला सर्वेक्षण उच्चतर दिखाता है
अनुपात, हालांकि इसमें कुल मिलाकर एक कार वाले कम परिवार हैं
गिनती.
\end{solution}

\begin{solution}{exo:g7:stats:4}
$\intco{5}{15}$: $8, 12, 9, 14, 11$ --- \omterm{def:g7:stats:frequency}{गिनती करना} $5$.
$\intco{15}{25}$: $23, 22, 18, 16$ --- \omterm{def:g7:stats:frequency}{गिनती करना} $4$.
$\intco{25}{35}$: $31, 27, 29, 25, 33$ --- \omterm{def:g7:stats:frequency}{गिनती करना} $5$.
$\intco{35}{45}$: $35$ --- \omterm{def:g7:stats:frequency}{गिनती करना} $1$. कुल $15$. \checkmark
\end{solution}

\begin{solution}{exo:g7:stats:5}
ऊँचाई की चार पट्टियाँ $8$, $12$, $6$, $4$ मूल्यों के ऊपर $0$, $1$,
$2$, $3$ (स्नातक प्रत्येक $2$ अच्छी तरह से काम करता हुँ)।
\end{solution}

\begin{solution}{exo:g7:stats:6}
कुल $30$ परिवार. आवृत्तियाँ: $\frac{8}{30}$, $\frac{12}{30}$,
$\frac{6}{30}$, $\frac{4}{30}$. कोण ($\times 360^\circ $): $96^\circ $,
$144^\circ $, $72^\circ $, $48^\circ $; \omterm{def:g1:addition:def}{जोड़}
$96 + 144 + 72 + 48 = 360^\circ $. \checkmark
\end{solution}

\begin{solution}{exo:g7:stats:7}
प्रतिशत: ग्रीष्म $\frac{162}{360} = 45\,\%$; वसंत
$\frac{90}{360} = 25\,\%$; पतझड़ और शरद
$\frac{54}{360} = 15\,\%$ प्रत्येक। से बाहर $60$ लोग, गर्मी को चुना गया
द्वारा $0.45 \times 60 = 27$ उनमें से.
\end{solution}

\begin{solution}{exo:g7:stats:8}
एक कक्षा: $\frac{12}{30} = 40\,\%$. कक्षा बी: $\frac{14}{40} = 35\,\%$.
कक्षा ए में पैदल चलने वालों का अनुपात अधिक है, हालांकि कक्षा बी में पैदल चलने वालों का अनुपात अधिक है
गिनती में अधिक पैदल यात्री.
\end{solution}

\begin{solution}{exo:g7:stats:9}
अक्ष से प्रारंभ होने पर $9\,700$, तीन बार दिखाई दे रहे हैं
ऊंचाई $100$, $300$ और $400$ इकाइयाँ: अंतिम चार गुना दिखता है
सबसे पहले, बिक्री चौगुनी होने का सुझाव। से एक अक्ष के साथ $0$, सलाखें
लगभग बराबर हैं ($9\,800$ को $10\,100$ के बारे में वृद्धि है $3\,\%$):
ईमानदार तस्वीर लगभग सपाट बिक्री दिखाती है।
\end{solution}

\begin{solution}{exo:g7:stats:10}
आवृत्तियों का योग है $1$: $ f = 1 - 0.35 - 0.4 = 0.25$. गिनती (कुल
$80$): $0.35 \times 80 = 28$; $0.4 \times 80 = 32$;
$0.25 \times 80 = 20$. जाँच करना: $28 + 32 + 20 = 80$.
\end{solution}

\begin{solution}{exo:g7:stats:11}
लड़कियाँ: $55\,\%$ का $400 = 220$; लड़के: $180$. कैफेटेरिया जाने वाली लड़कियाँ:
$0.4 \times 220 = 88$; लड़के: $0.6 \times 180 = 108$. कुल:
$88 + 108 = 196$ छात्र, यानी\ $\frac{196}{400} = 49\,\%$ का
स्कूल---के बीच $40\,\%$ और $60\,\%$, लड़कियों की दर के करीब
क्योंकि लड़कियाँ अधिक हैं।
\end{solution}

\begin{solution}{pb:g7:stats:1}
\textbf{1.} गिनती: $120 > 90$, स्कूल ए जीतता है। आवृत्तियाँ:
$\frac{120}{400} = 0.30 = 30\,\%$ ख़िलाफ़
$\frac{90}{250} = 0.36 = 36\,\%$: स्कूल बी जीत गया। पुरस्कार
का पालन करना चाहिए \omterm{def:g7:stats:frequency}{आवृत्ति} --- स्कूल बी एक बड़े हिस्से का पुनर्चक्रण करता है
यह क्या उपयोग करता है; स्कूल A केवल अधिक जूस पीता है।

\textbf{2.} शिक्षक: $\frac{18}{30} = 0.60 = 60\,\%$. छात्र:
$\frac{210}{600} = 0.35 = 35\,\%$. एक साथ:
$\frac{18 + 210}{630} = \frac{228}{630} \approx 0.36 =
36\,\%$. विद्यार्थियों की संख्या बीस गुना अधिक है, इसलिए
संयुक्त \omterm{def:g7:stats:frequency}{आवृत्ति} लगभग पूरी तरह से उनकी तरफ खींच लिया गया है ---
बड़ा समूह बड़ा भार वहन करता है।

\textbf{3.} आवृत्तियों को जोड़ना होगा $100\,\%$
(\cref{def:g7:stats:frequency}), लेकिन
$45 + 30 + 20 + 10 = 105$: कम से कम एक सेक्टर गलत है।

\textbf{4.} के लिए गिनें \omterm{def:g7:stats:frequency}{आवृत्ति} $0.15$:
$0.15 \times 40 = 6$. अब तक की गिनती: $14 + 10 + 6 = 30$, तो
आखिरी गिनती है $40 - 30 = 10$, साथ \omterm{def:g7:stats:frequency}{आवृत्ति}
$\frac{10}{40} = 0.25$. (जाँच करना: $0.35 + 0.25 + 0.15 + 0.25 =
1$.)

\textbf{5.} कोण: $0.40 \times 360 = 144^\circ $,
$0.35 \times 360 = 126^\circ $, $0.25 \times 360 = 90^\circ $;
और $144 + 126 + 90 = 360^\circ $: पाई बंद हो जाती है.

\textbf{6.} $\frac{2.4}{10} = 0.24 = 24\,\%$ के मतपत्र आये
वापस.

\textbf{7.} टेलीफोन निर्देशिकाएँ और कार रजिस्ट्रियाँ सूचीबद्ध
ज्यादातर अमीर घराने, जिनके वोट अलग-अलग थे
देश का; $2.4$ लाखों उत्तर बड़ी संख्या में निकाले गए थे
गलत आबादी से. जैसा कि प्रश्न 1 में है: क्या मायने रखता है
यह नहीं कि आप कितने गिनते हैं, बल्कि किसे--- a \omterm{def:g7:stats:frequency}{आवृत्ति} ए पर गणना की गई
विकृत नमूना नमूने का वर्णन करता है, देश का नहीं।

\textbf{8.} सच्चा समर्थन:
$0.7 \times 1\,000 + 0.3 \times 9\,000 = 700 + 2\,700 = 3\,400$
समर्थक बाहर $10\,000$: $34\,\%$. सर्वेक्षण:
$0.7 \times 500 + 0.3 \times 500 = 350 + 150 = 500$ से बाहर
$1\,000$: $50\,\%$ --- सोलह अंक बहुत अधिक हैं, क्योंकि
अमीर हैं \omterm{def:g3:division:half}{आधा} नमूना लेकिन शहर का दसवां हिस्सा.

\textbf{9.} उत्तर: $0.4 \times 200 = 80$ दुखी और
$0.1 \times 800 = 80$ खुश, तो $160$ जिसके उत्तर $80$
दुखी: मापी गई नाखुशी $\frac{80}{160} = 50\,\%$. सच है
दुःख: $\frac{200}{1\,000} = 20\,\%$. किसी ने झूठ नहीं बोला ---
नाखुश बस अधिक उत्तर देते हैं, और सर्वेक्षण उनकी बात सुनता है
जोर से.

\textbf{10.} सही अनुपात: दसवां हिस्सा अमीर है, इसलिए $100$
धनी और $900$ मामूली साक्षात्कार:
$0.7 \times 100 + 0.3 \times 900 = 70 + 270 = 340$ से बाहर
$1\,000$: मरम्मत किए गए सर्वेक्षण की भविष्यवाणी की गई है $34\,\%$ ---की सच्चाई
प्रश्न 8. पचास हजार अच्छी तरह से चुने गए साक्षात्कारों ने दो को पछाड़ दिया
लाखों बुरी तरह चुने गए।

\textbf{11.} अस्पताल ए: हल्का $\frac{90}{100} = 90\,\%$; गंभीर
$\frac{280}{400} = 70\,\%$; कुल मिलाकर
$\frac{90 + 280}{500} = \frac{370}{500} = 74\,\%$.

\textbf{12.} अस्पताल बी: हल्का $\frac{340}{400} = 85\,\%$;
गंभीर $\frac{65}{100} = 65\,\%$; कुल मिलाकर
$\frac{340 + 65}{500} = \frac{405}{500} = 81\,\%$. हॉस्पिटल ए
हल्के मामलों पर जीत ($90 > 85$) \emph{और} गंभीर मामलों पर
($70 > 65$).

\textbf{13.} कुल मिलाकर, बी दिखाता है $81\,\%$ ए के खिलाफ $74\,\%$:
वह अस्पताल जो हार जाता है \emph{प्रत्येक} श्रेणी जीतती है
कुल. (इस उलटफेर का एक नाम है: सिम्पसन विरोधाभास।)

\textbf{14.} मिश्रण विपरीत हैं: ए के रोगी अधिकतर हैं
गंभीर ($400$ का $500$), बी का ज्यादातर हल्का ($400$ का $500$).
गंभीर मामलों का जहां भी इलाज किया जाता है, वे कम ही ठीक होते हैं, इसलिए ए
कुल मिलाकर यह आंकड़ा कठिन मामलों द्वारा नीचे खींचा गया है जिसे वह स्वीकार करता है ---
कुल मिलाकर एक भारित मिश्रण है, और वजन अलग-अलग होता है
अस्पताल. गंभीर रोगी को अस्पताल ए चुनना चाहिए: $70\,\%$
धड़कता है $65\,\%$ एकमात्र पंक्ति में जो उनसे संबंधित है।

\textbf{15.} उदाहरण के लिए: ``आपकी तुलना में हल्के और का मिश्रण है
गंभीर मरीज़, और दोनों अस्पताल विपरीत मिश्रण का इलाज करते हैं ---
गंभीरता से विभाजित, अस्पताल ए का अधिक हिस्सा ठीक होता है
\emph{दोनों} प्रकार. Overall figures may only be compared when the
groups behind them are alike; otherwise the weights, not the
quality, decide the winner.''
\end{solution}
