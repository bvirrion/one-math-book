\section*{अध्याय \ref{ch:g7:negatives} --- नकारात्मक संख्याएँ}

\begin{solution}{exo:g7:negatives:1}
विपरीत: $-7$; \ $+3.5$; \ $0$ (इसके अपने विपरीत); \ $-12$. दो
संख्याएँ हैं विपरीत जब उनका \omterm{def:g1:addition:def}{जोड़} है $0$.
\end{solution}

\begin{solution}{exo:g7:negatives:2}
$-4 < 1$; \quad $-4 > -7$; \quad $-2.5 < -2.4$; \quad $0 > -0.1$.
\end{solution}

\begin{solution}{exo:g7:negatives:3}
$-3.1 < -3 < -0.5 < 0 < 1 < 2.5$.
\end{solution}

\begin{solution}{exo:g7:negatives:4}
$(-5) + (-9) = -14$; \quad
$(-12) + (+7) = -5$; \quad
$(+3.5) + (-1.5) = +2$; \quad
$(-6) + (+6) = 0$.
\end{solution}

\begin{solution}{exo:g7:negatives:5}
$4 - 11 = 4 + (-11) = -7$.

$(-5) - (+3) = (-5) + (-3) = -8$.

$(-7) - (-10) = (-7) + (+10) = +3$.

$0 - (-8) = 0 + (+8) = +8$.
\end{solution}

\begin{solution}{exo:g7:negatives:6}
$(+7) + (-12) + (+3) + (-8) = (+10) + (-20) = -10$ (सकारात्मक
एक साथ, नकारात्मक एक साथ)।

$(-5.5) + (+9) + (-4.5) + (+1) = (-10) + (+10) = 0$.
\end{solution}

\begin{solution}{exo:g7:negatives:7}
$(-3) + 8 - 12 = 5 - 12 = -7$: यह है $-7$ रात में डिग्री.
\end{solution}

\begin{solution}{exo:g7:negatives:8}
मुफ़्त प्लॉट; $ E(0, -3)$ ऊर्ध्वाधर अक्ष पर स्थित है (इसका पहला निर्देशांक है
है $0$).
\end{solution}

\begin{solution}{exo:g7:negatives:9}
मास्को: $(-4) - (-13) = -4 + 13 = 9$ की डिग्री \omterm{ex:g1:subtraction:difference}{अंतर}.
हेलसिंकी: $2 - (-9) = 11$ डिग्री.
\end{solution}

\begin{solution}{exo:g7:negatives:10}
$(-12) + 7 - 4 = -5 - 4 = -9$: वह इस पर है $-9$ एम। सतह तक पहुँचने के लिए
(ऊंचाई $0$) उसे उठना ही होगा $0 - (-9) = 9$ एम।
\end{solution}

\begin{solution}{exo:g7:negatives:11}
$ D(-2, -2)$: वही पहला समन्वय $ A $, वही दूसरा समन्वय
$ C $. पक्ष: $ AB = 4$ (से $-2$ को $2$) और $ BC = 3$ (से $1$ नीचे तक
$-2$), इतना \omterm{def:g6:measure:perimeter}{परिधि} है $2 \times (4 + 3) = 14$.
\end{solution}

\begin{solution}{exo:g7:negatives:12}
पूर्ण विकर्ण $0, -2, -4$ जादू देता है \omterm{def:g1:addition:def}{जोड़}
$0 + (-2) + (-4) = -6$. फिर, सेल दर सेल:
शीर्ष-बाएँ $= -6 - (-7) - 0 = 1$; मध्य-सही $= -6 - (-3) - (-2) =
-1$; नीचे दाएं $= -6 - 0 - (-1) = -5$; नीचे-मध्य
$= -6 - (-7) - (-2) = 3$.
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|c|c|}
\hline
$1$ & $-7$ & $0$ \\
\hline
$-3$ & $-2$ & $-1$ \\
\hline
$-4$ & $3$ & $-5$ \\
\hline
\end{tabular}
\end{center}
प्रत्येक पंक्ति, स्तंभ और विकर्ण का योग है $-6$.
\end{solution}

\begin{solution}{pb:g7:negatives:1}
\textbf{1.} $4\,808 - (-430) = 4\,808 + 430 = 5\,238$ एम
(\cref{thm:g7:negatives:sub}: ऋणात्मक को घटाने का अर्थ है
जोड़ना)।

\textbf{2.} संतुलन:
\[
(-45) + 120 + (-60) + 15 = 75 + (-60) + 15 = 15 + 15 = 30
\text{ euros}.
\]
प्रारंभिक वापस करने के लिए $-45$ को $0$, इसके विपरीत जमा करें:
$45$ यूरो.

\textbf{3.} $(-44) - (-100) = (-44) + 100 = 56$: सीज़र रहता था
$56$ साल। (यहां कोई \omterm{def:g5:solids:net}{जाल} नहीं है: दोनों वर्ष एक ही तरफ हैं
युग सीमा.)

\textbf{4.} वर्ष $0$: से गिनती $-63$ तक $+14$ गुजरता है
के माध्यम से $0$, लेकिन इतिहासकारों का कैलेंडर सीधे कूद पड़ता है
$1$ ईसा पूर्व से ई.पू $1$ --- वर्ष $0$ कभी अस्तित्व में नहीं था, इसलिए भोला है
गिनती एक साल बहुत लंबी है.

\textbf{5.} $63$ बीसी बन जाता है $-(63 - 1) = -62$. तब
$14 - (-62) = 14 + 62 = 76$ वर्ष: बिल्कुल इतिहासकारों का
आंकड़ा. खगोलशास्त्रियों की पुनः लेबल वाली \omterm{def:g6:lines:objects}{रेखा} पर कोई \omterm{ex:g1:subtraction:difference}{अंतर}ाल नहीं है,
इसलिए साधारण घटाव वास्तविक अवधि देता है।

\textbf{6.} $1$ ईसा पूर्व $\to 0$; \ $10$ ईसा पूर्व $\to -9$; \ विज्ञापन
$476 \to +476$; \ $753$ ईसा पूर्व $\to -752$.

\textbf{7.} $476 - (-752) = 476 + 752 = 1\,228$ साल।

\textbf{8.} $12$ बीसी है $-11$. अगली प्रस्तुतियाँ:
$-11 + 76 = 65$, तब $141$, तब $217$: वर्ष ई.पू $65$, ई.पू
$141$ और ई.पू $217$. पहले: $-11 - 76 = -87$, जो है
इतिहासकारों का वर्ष $87 + 1 = 88$ ईसा पूर्व.

\textbf{9.} $5$ बीसी है $-4$, इसलिए $5 - (-4) = 9$ वर्ष --- नहीं
$10$. लुप्त वर्ष \omterm{def:g1:counting:zero}{शून्य} जब भी कोई अवधि पार हो जाती है तब हमला होता है
युग सीमा.

\textbf{10.} $3$ बीसी है $-2$; दस साल बाद है $-2 + 10 = +8$:
ईस्वी सन् में बच्चा दस वर्ष का हो गया $8$.

\textbf{11.} बीच में $-7$ और $-2$: $(-2) - (-7) = 5$. बीच में
$-3.5$ और $4.5$: $4.5 - (-3.5) = 8$.

\textbf{12.} समूहन विपरीत और मित्रों:
\[
3 + (-5) + 2 + (-1) + 4
= \bigl(3 + 2\bigr) + (-5) + (-1) + 4
= 0 + (-1) + 4 = +3 :
\]
लिफ्ट मंजिल पर समाप्त होती है $3$. आदेश मायने नहीं रखता: a \omterm{def:g1:addition:def}{जोड़} का
सापेक्ष संख्याएँ स्वतंत्र रूप से पुनर्गठित किया जा सकता है (प्रत्येक चाल चलती है
एक ही पंक्ति के साथ, और कुल चलना समान है)।

\textbf{13.} \omterm{def:g5:solids:net}{जाल} परिवर्तन:
$2 + (-4) + 1 + 0 + (-3) + 5 + (-2) = -1$ डिग्री। से शुरू हो रहा है
$-2$: अंतिम तापमान $-2 + (-1) = -3$ डिग्री. सामान्य तौर पर,
अंतिम मान आरंभिक मान प्लस है \omterm{def:g1:addition:def}{जोड़} सभी का
परिवर्तन --- एक-एक करके उतार-चढ़ाव का अनुसरण करने की आवश्यकता नहीं है।

\textbf{14.} प्रारंभ में वापस आने का मतलब है \omterm{def:g5:solids:net}{जाल} परिवर्तन $0$: अप और
डाउन बिल्कुल रद्द होना चाहिए। पाँच ज्ञात परिवर्तनों का योग है
$4 + (-7) + 2 + 3 + (-5) = -3$, तो छठा परिवर्तन है
विपरीत: $+3$ (तीन सौ \omterm{def:g4:fractions:def}{मीटर} ऊपर)।

\textbf{15.} खगोलशास्त्री: $509$ ईसा पूर्व $\to -508$, और
$476 - (-508) = 984$ साल। इतिहासकार का $509 + 476 = 985$
एक वर्ष को बहुत अधिक गिनता है --- एक बार फिर प्रेत वर्ष \omterm{def:g1:counting:zero}{शून्य},
चुपचाप \omterm{def:g1:addition:def}{जोड़} में शामिल कर लिया गया लेकिन इतिहास से गायब कर दिया गया।
(एक जाँच के रूप में, दो युग: गणतंत्र $-508$ को $-26$, $482$
साल; साम्राज्य $-26$ को $+476$, $502$ साल;
$482 + 502 = 984$.)
\end{solution}
