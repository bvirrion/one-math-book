\section*{Chapter \ref{ch:g7:negatives} --- Negative Numbers}

\begin{solution}{exo:g7:negatives:1}
\omterm{def:g7:negatives:def}{Opposites}: $-7$; \ $+3.5$; \ $0$ (its own opposite); \ $-12$. Two
numbers are \omterm{def:g7:negatives:def}{opposites} when their \omterm{def:g1:addition:def}{sum} is $0$.
\end{solution}

\begin{solution}{exo:g7:negatives:2}
$-4 < 1$; \quad $-4 > -7$; \quad $-2.5 < -2.4$; \quad $0 > -0.1$.
\end{solution}

\begin{solution}{exo:g7:negatives:3}
$-3.1 < -3 < -0.5 < 0 < 1 < 2.5$.
\end{solution}

\begin{solution}{exo:g7:negatives:4}
$(-5) + (-9) = -14$; \quad
$(-12) + (+7) = -5$; \quad
$(+3.5) + (-1.5) = +2$; \quad
$(-6) + (+6) = 0$.
\end{solution}

\begin{solution}{exo:g7:negatives:5}
$4 - 11 = 4 + (-11) = -7$.

$(-5) - (+3) = (-5) + (-3) = -8$.

$(-7) - (-10) = (-7) + (+10) = +3$.

$0 - (-8) = 0 + (+8) = +8$.
\end{solution}

\begin{solution}{exo:g7:negatives:6}
$(+7) + (-12) + (+3) + (-8) = (+10) + (-20) = -10$ (positives
together, negatives together).

$(-5.5) + (+9) + (-4.5) + (+1) = (-10) + (+10) = 0$.
\end{solution}

\begin{solution}{exo:g7:negatives:7}
$(-3) + 8 - 12 = 5 - 12 = -7$: it is $-7$ degrees at night.
\end{solution}

\begin{solution}{exo:g7:negatives:8}
Free plot; $E(0, -3)$ lies on the vertical axis (its first coordinate
is $0$).
\end{solution}

\begin{solution}{exo:g7:negatives:9}
Moscow: $(-4) - (-13) = -4 + 13 = 9$ degrees of \omterm{ex:g1:subtraction:difference}{difference}.
Helsinki: $2 - (-9) = 11$ degrees.
\end{solution}

\begin{solution}{exo:g7:negatives:10}
$(-12) + 7 - 4 = -5 - 4 = -9$: she is at $-9$ m. To reach the surface
(altitude $0$) she must rise $0 - (-9) = 9$ m.
\end{solution}

\begin{solution}{exo:g7:negatives:11}
$D(-2, -2)$: same first coordinate as $A$, same second coordinate as
$C$. Sides: $AB = 4$ (from $-2$ to $2$) and $BC = 3$ (from $1$ down to
$-2$), so the \omterm{def:g6:measure:perimeter}{perimeter} is $2 \times (4 + 3) = 14$.
\end{solution}

\begin{solution}{exo:g7:negatives:12}
The complete diagonal $0, -2, -4$ gives the magic \omterm{def:g1:addition:def}{sum}
$0 + (-2) + (-4) = -6$. Then, cell by cell:
top-left $= -6 - (-7) - 0 = 1$; middle-right $= -6 - (-3) - (-2) =
-1$; bottom-right $= -6 - 0 - (-1) = -5$; bottom-middle
$= -6 - (-7) - (-2) = 3$.
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|c|c|}
\hline
$1$ & $-7$ & $0$ \\
\hline
$-3$ & $-2$ & $-1$ \\
\hline
$-4$ & $3$ & $-5$ \\
\hline
\end{tabular}
\end{center}
Every row, column and diagonal sums to $-6$.
\end{solution}

\begin{solution}{pb:g7:negatives:1}
\textbf{1.} $4\,808 - (-430) = 4\,808 + 430 = 5\,238$ m
(\cref{thm:g7:negatives:sub}: subtracting a negative means
adding).

\textbf{2.} Balance:
\[
(-45) + 120 + (-60) + 15 = 75 + (-60) + 15 = 15 + 15 = 30
\text{ euros}.
\]
To return the initial $-45$ to $0$, deposit its opposite:
$45$ euros.

\textbf{3.} $(-44) - (-100) = (-44) + 100 = 56$: Caesar lived
$56$ years. (No trap here: both years are on the same side of
the era boundary.)

\textbf{4.} The year $0$: counting from $-63$ up to $+14$ passes
through $0$, but the historians' calendar jumps straight from
$1$ BC to AD $1$ --- the year $0$ never existed, so the naive
count is one year too long.

\textbf{5.} $63$ BC becomes $-(63 - 1) = -62$. Then
$14 - (-62) = 14 + 62 = 76$ years: exactly the historians'
figure. On the astronomers' relabelled line there are no gaps,
so ordinary subtraction gives true durations.

\textbf{6.} $1$ BC $\to 0$; \ $10$ BC $\to -9$; \ AD
$476 \to +476$; \ $753$ BC $\to -752$.

\textbf{7.} $476 - (-752) = 476 + 752 = 1\,228$ years.

\textbf{8.} $12$ BC is $-11$. Next appearances:
$-11 + 76 = 65$, then $141$, then $217$: the years AD $65$, AD
$141$ and AD $217$. Before: $-11 - 76 = -87$, which is the
historians' year $87 + 1 = 88$ BC.

\textbf{9.} $5$ BC is $-4$, so $5 - (-4) = 9$ years --- not
$10$. The missing year \omterm{def:g1:counting:zero}{zero} strikes whenever a duration crosses
the era boundary.

\textbf{10.} $3$ BC is $-2$; ten years later is $-2 + 10 = +8$:
the child turned ten in AD $8$.

\textbf{11.} Between $-7$ and $-2$: $(-2) - (-7) = 5$. Between
$-3.5$ and $4.5$: $4.5 - (-3.5) = 8$.

\textbf{12.} Grouping \omterm{def:g7:negatives:def}{opposites} and friends:
\[
3 + (-5) + 2 + (-1) + 4
= \bigl(3 + 2\bigr) + (-5) + (-1) + 4
= 0 + (-1) + 4 = +3 :
\]
the lift ends at floor $3$. The order does not matter: a \omterm{def:g1:addition:def}{sum} of
\omterm{def:g7:negatives:def}{relative numbers} can be reorganized freely (each move is walked
along the same line, and the total walk is the same).

\textbf{13.} \omterm{def:g5:solids:net}{Net} change:
$2 + (-4) + 1 + 0 + (-3) + 5 + (-2) = -1$ degree. Starting at
$-2$: final temperature $-2 + (-1) = -3$ degrees. In general,
the final value is the starting value plus the \omterm{def:g1:addition:def}{sum} of all the
changes --- no need to follow the ups and downs one by one.

\textbf{14.} Back to the start means \omterm{def:g5:solids:net}{net} change $0$: the ups and
downs must cancel exactly. The five known changes sum to
$4 + (-7) + 2 + 3 + (-5) = -3$, so the sixth change is the
opposite: $+3$ (three hundred meters up).

\textbf{15.} Astronomers: $509$ BC $\to -508$, and
$476 - (-508) = 984$ years. The historian's $509 + 476 = 985$
counts one year too many --- once more the phantom year \omterm{def:g1:counting:zero}{zero},
silently included in the addition but absent from history.
(As a check, the two eras: Republic $-508$ to $-26$, $482$
years; Empire $-26$ to $+476$, $502$ years;
$482 + 502 = 984$.)
\end{solution}
