\section*{Chapter \ref{ch:g7:negatives} --- Negative Numbers}

\begin{solution}{exo:g7:negatives:1}
\omterm{def:g7:negatives:def}{Opposites}: $-7$; \ $+3.5$; \ $0$ (its own opposite); \ $-12$. Two
numbers are \omterm{def:g7:negatives:def}{opposites} when their \omterm{def:g1:addition:def}{sum} is $0$.
\end{solution}

\begin{solution}{exo:g7:negatives:2}
$-4 < 1$; \quad $-4 > -7$; \quad $-2.5 < -2.4$; \quad $0 > -0.1$.
\end{solution}

\begin{solution}{exo:g7:negatives:3}
$-3.1 < -3 < -0.5 < 0 < 1 < 2.5$.
\end{solution}

\begin{solution}{exo:g7:negatives:4}
$(-5) + (-9) = -14$; \quad
$(-12) + (+7) = -5$; \quad
$(+3.5) + (-1.5) = +2$; \quad
$(-6) + (+6) = 0$.
\end{solution}

\begin{solution}{exo:g7:negatives:5}
$4 - 11 = 4 + (-11) = -7$.

$(-5) - (+3) = (-5) + (-3) = -8$.

$(-7) - (-10) = (-7) + (+10) = +3$.

$0 - (-8) = 0 + (+8) = +8$.
\end{solution}

\begin{solution}{exo:g7:negatives:6}
$(+7) + (-12) + (+3) + (-8) = (+10) + (-20) = -10$ (positives
together, negatives together).

$(-5.5) + (+9) + (-4.5) + (+1) = (-10) + (+10) = 0$.
\end{solution}

\begin{solution}{exo:g7:negatives:7}
$(-3) + 8 - 12 = 5 - 12 = -7$: it is $-7$ degrees at night.
\end{solution}

\begin{solution}{exo:g7:negatives:8}
Free plot; $E(0, -3)$ lies on the vertical axis (its first coordinate
is $0$).
\end{solution}

\begin{solution}{exo:g7:negatives:9}
Moscow: $(-4) - (-13) = -4 + 13 = 9$ degrees of \omterm{ex:g1:subtraction:difference}{difference}.
Helsinki: $2 - (-9) = 11$ degrees.
\end{solution}

\begin{solution}{exo:g7:negatives:10}
$(-12) + 7 - 4 = -5 - 4 = -9$: she is at $-9$ m. To reach the surface
(altitude $0$) she must rise $0 - (-9) = 9$ m.
\end{solution}

\begin{solution}{exo:g7:negatives:11}
$D(-2, -2)$: same first coordinate as $A$, same second coordinate as
$C$. Sides: $AB = 4$ (from $-2$ to $2$) and $BC = 3$ (from $1$ down to
$-2$), so the \omterm{def:g6:measure:perimeter}{perimeter} is $2 \times (4 + 3) = 14$.
\end{solution}

\begin{solution}{exo:g7:negatives:12}
The complete diagonal $0, -2, -4$ gives the magic \omterm{def:g1:addition:def}{sum}
$0 + (-2) + (-4) = -6$. Then, cell by cell:
top-left $= -6 - (-7) - 0 = 1$; middle-right $= -6 - (-3) - (-2) =
-1$; bottom-right $= -6 - 0 - (-1) = -5$; bottom-middle
$= -6 - (-7) - (-2) = 3$.
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|c|c|}
\hline
$1$ & $-7$ & $0$ \\
\hline
$-3$ & $-2$ & $-1$ \\
\hline
$-4$ & $3$ & $-5$ \\
\hline
\end{tabular}
\end{center}
Every row, column and diagonal sums to $-6$.
\end{solution}
