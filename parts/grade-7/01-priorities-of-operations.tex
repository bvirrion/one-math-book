\chapter{Priorities of Operations}\label{ch:g7:priorities}

What is $3 + 4 \times 5$? If you compute left to right you get $35$;
if you multiply first you get $23$. Mathematics needs one single
answer, so there are traffic rules for computations --- the priorities.
This chapter states them and introduces a first algebraic identity,
distributivity.

\section{The priority rules}

\begin{theorem}[Priority rules]\label{thm:g7:priorities:rules}
In a computation without brackets:
\begin{enumerate}
  \item multiplications and divisions are performed \emph{before}
        additions and subtractions;
  \item operations of the same level are performed from left to right.
\end{enumerate}
Brackets override everything: what is inside brackets is computed
first, starting with the innermost ones.
\end{theorem}

\begin{example}\label{ex:g7:priorities:basic}
One step per line, underlining (mentally) what is computed:
\begin{align*}
3 + 4 \times 5 &= 3 + 20 = 23
&&\text{(multiplication first)}\\
20 - 8 + 2 &= 12 + 2 = 14
&&\text{(same level: left to right)}\\
18 \div 3 \times 2 &= 6 \times 2 = 12
&&\text{(same level: left to right)}\\
(3 + 4) \times 5 &= 7 \times 5 = 35
&&\text{(brackets first).}
\end{align*}
Note the traps: $20 - 8 + 2$ is \emph{not} $20 - 10$, and
$18 \div 3 \times 2$ is \emph{not} $18 \div 6$.
\end{example}

\begin{example}[Nested brackets]\label{ex:g7:priorities:nested}
\begin{align*}
50 - 2 \times \bigl(3 + (10 - 4)\bigr)
&= 50 - 2 \times (3 + 6) && \text{(innermost bracket)}\\
&= 50 - 2 \times 9 && \text{(remaining bracket)}\\
&= 50 - 18 && \text{(multiplication)}\\
&= 32 .
\end{align*}
\end{example}

\begin{remark}[Fraction bars hide brackets]
A fraction bar groups its numerator and its denominator, as if each
had invisible brackets:
\[
\frac{12 + 8}{4} = \frac{20}{4} = 5,
\qquad\text{not}\quad 12 + \frac{8}{4} = 14 .
\]
\end{remark}

\begin{method}[Computing safely]\label{met:g7:priorities:safe}
\begin{enumerate}
  \item Spot the operation to do first (innermost brackets, then
        $\times$ and $\div$, then $+$ and $-$);
  \item compute \emph{only that one}, and rewrite the whole expression
        with the result in place;
  \item repeat until one number remains --- one operation per line, no
        shortcuts.
\end{enumerate}
\end{method}

\section{Distributivity}

\begin{theorem}[Distributivity]\label{thm:g7:priorities:distributivity}
For all numbers $k$, $a$, $b$:
\[
k \times (a + b) = k \times a + k \times b,
\qquad
k \times (a - b) = k \times a - k \times b .
\]
\end{theorem}

\begin{proof}[Proof by a picture]
Count the area of a rectangle of width $k$ split into two smaller
rectangles of lengths $a$ and $b$: as a whole it is $k \times (a+b)$;
in two parts it is $k \times a + k \times b$. Same rectangle, same
area.
\end{proof}

  \node[font=\small] at (2.1,0.9) {$k \times a$};

\end{omfigure}

\begin{example}[Both directions]\label{ex:g7:priorities:distributivity}
\emph{Expanding} (left to right) makes mental math easy:
\[
7 \times 103 = 7 \times (100 + 3) = 700 + 21 = 721,
\qquad
6 \times 98 = 6 \times (100 - 2) = 600 - 12 = 588 .
\]
\emph{Factoring} (right to left) collects a common factor:
\[
23 \times 12 + 23 \times 8 = 23 \times (12 + 8) = 23 \times 20 = 460 .
\]
\end{example}

\begin{example}[Translating words into a computation]\label{ex:g7:priorities:words}
``The sum of $8$ and the product of $5$ by $3$'' is $8 + 5 \times 3 =
23$. ``The product of $5$ by the sum of $8$ and $3$'' is
$5 \times (8 + 3) = 55$: the words ``the product of \dots\ by the sum
of\dots'' put brackets around the sum.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g7:priorities:1}
Compute, one operation per line:
\[
5 + 3 \times 6, \qquad
(5 + 3) \times 6, \qquad
30 - 12 \div 4, \qquad
(30 - 12) \div 4 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:2}
Compute:
\[
24 - 6 + 2, \qquad
24 \div 6 \times 2, \qquad
24 - 6 \times 2, \qquad
24 \div (6 \times 2).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:3}
Compute step by step:
\[
7 + 3 \times (10 - 6), \qquad
40 - (2 + 3) \times 4, \qquad
60 \div \bigl(2 \times (7 - 4)\bigr).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:4}
Compute the fractions (invisible brackets!):
\[
\frac{18 + 6}{8}, \qquad
\frac{30}{7 - 2}, \qquad
\frac{5 \times 8}{4 + 6} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:5}
Use distributivity to compute mentally:
\[
8 \times 102, \qquad
25 \times 99, \qquad
14 \times 12 + 14 \times 8, \qquad
37 \times 45 - 37 \times 35 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:6}
Write the computation in one line, with brackets where needed, then
compute: ``the difference of $50$ and the product of $6$ by $7$'';
``the product of the sum of $4$ and $5$ by the difference of $10$ and
$8$''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:7}
Amir bought $3$ notebooks at $2.50$ each and $2$ pens at $1.20$ each.
Write his total in a single expression, then compute it.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:priorities:8}
Place brackets in $4 + 2 \times 5 - 1$ to make the result: (a) $13$;
(b) $29$; (c) $24$. (One of the three needs no brackets at all ---
which one?)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:priorities:9}
Are $12 + 4 \times k$ and $16 \times k$ the same for every number $k$?
Test with $k = 1$ and $k = 2$, and conclude. Which priority mistake
would lead from one to the other?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:priorities:10}
A cinema ticket costs $9$ euros, but a group of $n$ people pays a flat
$12$ euros plus $6$ euros per person.
\begin{enumerate}
  \item Write an expression for the group price, and compute it for
        $n = 5$.
  \item From how many people is the group rate cheaper than $n$
        individual tickets? (Try a few values of $n$.)
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:priorities:11}
Using each of the digits $1$, $2$, $3$, $4$ exactly once, the four
operations and brackets at will, write expressions equal to $24$, to
$10$, and to $1$.
\end{exercise}
