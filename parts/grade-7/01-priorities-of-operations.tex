\chapter{Priorities of Operations}\label{ch:g7:priorities}

What is $3 + 4 \times 5$? If you compute left to right you get $35$;
if you multiply first you get $23$. Mathematics needs one single
answer, so there are traffic rules for computations --- the priorities.
This chapter states them and introduces a first algebraic identity,
distributivity.

\section{The priority rules}

\begin{theorem}[Priority rules]\label{thm:g7:priorities:rules}
In a computation without brackets:
\begin{enumerate}
  \item multiplications and divisions are performed \emph{before}
        additions and subtractions;
  \item operations of the same level are performed from left to right.
\end{enumerate}
Brackets override everything: what is inside brackets is computed
first, starting with the innermost ones.
\end{theorem}

\begin{example}\label{ex:g7:priorities:basic}
One step per line, underlining (mentally) what is computed:
\begin{align*}
3 + 4 \times 5 &= 3 + 20 = 23
&&\text{(multiplication first)}\\
20 - 8 + 2 &= 12 + 2 = 14
&&\text{(same level: left to right)}\\
18 \div 3 \times 2 &= 6 \times 2 = 12
&&\text{(same level: left to right)}\\
(3 + 4) \times 5 &= 7 \times 5 = 35
&&\text{(brackets first).}
\end{align*}
Note the traps: $20 - 8 + 2$ is \emph{not} $20 - 10$, and
$18 \div 3 \times 2$ is \emph{not} $18 \div 6$.
\end{example}

\begin{example}[Nested brackets]\label{ex:g7:priorities:nested}
\begin{align*}
50 - 2 \times \bigl(3 + (10 - 4)\bigr)
&= 50 - 2 \times (3 + 6) && \text{(innermost bracket)}\\
&= 50 - 2 \times 9 && \text{(remaining bracket)}\\
&= 50 - 18 && \text{(multiplication)}\\
&= 32 .
\end{align*}
\end{example}

\begin{remark}[Fraction bars hide brackets]
A \omterm{def:g6:fractions:def}{fraction} bar groups its \omterm{def:g4:fractions:def}{numerator} and its \omterm{def:g4:fractions:def}{denominator}, as if each
had invisible brackets:
\[
\frac{12 + 8}{4} = \frac{20}{4} = 5,
\qquad\text{not}\quad 12 + \frac{8}{4} = 14 .
\]
\end{remark}

\begin{method}[Computing safely]\label{met:g7:priorities:safe}
\begin{enumerate}
  \item Spot the operation to do first (innermost brackets, then
        $\times$ and $\div$, then $+$ and $-$);
  \item compute \emph{only that one}, and rewrite the whole expression
        with the result in place;
  \item repeat until one number remains --- one operation per line, no
        shortcuts.
\end{enumerate}
\end{method}

\section{Distributivity}

\begin{theorem}[Distributivity]\label{thm:g7:priorities:distributivity}
For all numbers $k$, $a$, $b$:
\[
k \times (a + b) = k \times a + k \times b,
\qquad
k \times (a - b) = k \times a - k \times b .
\]
\end{theorem}

\begin{proof}[Proof by a picture]
Count the \omterm{def:g6:measure:area}{area} of a rectangle of width $k$ split into two smaller
rectangles of lengths $a$ and $b$: as a whole it is $k \times (a+b)$;
in two parts it is $k \times a + k \times b$. Same rectangle, same
\omterm{def:g6:measure:area}{area}.
\end{proof}

\begin{omfigure}
\begin{tikzpicture}[scale=0.75]
  \fill[omDef!14] (0,0) rectangle (4.2,1.8);
  \fill[omThm!14] (4.2,0) rectangle (6.4,1.8);
  \draw[thick] (0,0) rectangle (6.4,1.8);
  \draw[thick] (4.2,0) -- (4.2,1.8);
  \node[below, font=\small] at (2.1,0) {$a$};
  \node[below, font=\small] at (5.3,0) {$b$};
  \node[left, font=\small] at (0,0.9) {$k$};
  \node[font=\small] at (2.1,0.9) {$k \times a$};
  \node[font=\small] at (5.3,0.9) {$k \times b$};
\end{tikzpicture}

{\small Distributivity as \omterm{def:g6:measure:area}{areas}: the big rectangle
$k \times (a + b)$ is the \omterm{def:g1:addition:def}{sum} of the two smaller ones.}
\end{omfigure}

\begin{example}[Both directions]\label{ex:g7:priorities:distributivity}
\emph{Expanding} (left to right) makes mental math easy:
\[
7 \times 103 = 7 \times (100 + 3) = 700 + 21 = 721,
\qquad
6 \times 98 = 6 \times (100 - 2) = 600 - 12 = 588 .
\]
\emph{Factoring} (right to left) collects a common factor:
\[
23 \times 12 + 23 \times 8 = 23 \times (12 + 8) = 23 \times 20 = 460 .
\]
\end{example}

\begin{example}[Translating words into a computation]\label{ex:g7:priorities:words}
``The \omterm{def:g1:addition:def}{sum} of $8$ and the \omterm{def:g2:mult:def}{product} of $5$ by $3$'' is $8 + 5 \times 3 =
23$. ``The \omterm{def:g2:mult:def}{product} of $5$ by the \omterm{def:g1:addition:def}{sum} of $8$ and $3$'' is
$5 \times (8 + 3) = 55$: the words ``the \omterm{def:g2:mult:def}{product} of \dots\ by the \omterm{def:g1:addition:def}{sum}
of\dots'' put brackets around the \omterm{def:g1:addition:def}{sum}.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g7:priorities:1}
Compute, one operation per line:
\[
5 + 3 \times 6, \qquad
(5 + 3) \times 6, \qquad
30 - 12 \div 4, \qquad
(30 - 12) \div 4 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:2}
Compute:
\[
24 - 6 + 2, \qquad
24 \div 6 \times 2, \qquad
24 - 6 \times 2, \qquad
24 \div (6 \times 2).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:3}
Compute step by step:
\[
7 + 3 \times (10 - 6), \qquad
40 - (2 + 3) \times 4, \qquad
60 \div \bigl(2 \times (7 - 4)\bigr).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:4}
Compute the \omterm{def:g6:fractions:def}{fractions} (invisible brackets!):
\[
\frac{18 + 6}{8}, \qquad
\frac{30}{7 - 2}, \qquad
\frac{5 \times 8}{4 + 6} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:5}
Use distributivity to compute mentally:
\[
8 \times 102, \qquad
25 \times 99, \qquad
14 \times 12 + 14 \times 8, \qquad
37 \times 45 - 37 \times 35 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:6}
Write the computation in one line, with brackets where needed, then
compute: ``the \omterm{ex:g1:subtraction:difference}{difference} of $50$ and the \omterm{def:g2:mult:def}{product} of $6$ by $7$'';
``the \omterm{def:g2:mult:def}{product} of the \omterm{def:g1:addition:def}{sum} of $4$ and $5$ by the \omterm{ex:g1:subtraction:difference}{difference} of $10$ and
$8$''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:7}
Amir bought $3$ notebooks at $2.50$ each and $2$ pens at $1.20$ each.
Write his total in a single expression, then compute it.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:priorities:8}
Place brackets in $4 + 2 \times 5 - 1$ to make the result: (a) $13$;
(b) $29$; (c) $24$. (One of the three needs no brackets at all ---
which one?)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:priorities:9}
Are $12 + 4 \times k$ and $16 \times k$ the same for every number $k$?
Test with $k = 1$ and $k = 2$, and conclude. Which priority mistake
would lead from one to the other?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:priorities:10}
A cinema ticket costs $9$ euros, but a group of $n$ people pays a flat
$12$ euros plus $6$ euros per person.
\begin{enumerate}
  \item Write an expression for the group price, and compute it for
        $n = 5$.
  \item From how many people is the group rate cheaper than $n$
        individual tickets? (Try a few values of $n$.)
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:priorities:11}
Using each of the digits $1$, $2$, $3$, $4$ exactly once, the four
operations and brackets at will, write expressions equal to $24$, to
$10$, and to $1$.
\end{exercise}

\section{Problem: Multiplying like an Egyptian scribe}

\begin{problem}[{Weekend problem --- mental-math tricks from
distributivity, and the doubling--halving multiplication used for
four thousand years}]
\label{pb:g7:priorities:1}
Long before calculators --- and before multiplication tables were
drilled at school --- Egyptian scribes and Russian peasants
multiplied any two numbers using only \emph{doubling, halving and
adding}. Their method looks like magic; it is in fact
distributivity (\cref{thm:g7:priorities:distributivity}) working
at full power. This problem sharpens your mental arithmetic, then
teaches you the ancient method, then opens the hood to see exactly
why it works.

\medskip
\noindent\textbf{Part I --- The art of not computing.}
\begin{enumerate}
  \item Compute mentally, expanding as in
        \cref{ex:g7:priorities:distributivity}:
        $7 \times 102$ and $9 \times 98$.
  \item Compute mentally, factoring:
        $17 \times 13 + 17 \times 7$ and
        $45 \times 101 - 45$.
  \item The ``times $5$'' trick: to multiply by $5$, multiply by
        $10$ and halve. Compute $5 \times 87$ this way, and
        explain why the trick is legitimate.
  \item Two more tricks to justify and use: multiply by $25$ by
        multiplying by $100$ and dividing by $4$ (compute
        $25 \times 32$); multiply by $99$ by multiplying by $100$
        and subtracting once (compute $99 \times 47$).
  \item A warning shot: distributivity spreads a factor over a
        \emph{\omterm{def:g1:addition:def}{sum}}, never over a \omterm{def:g2:mult:def}{product}. Compare
        $8 \times (5 + 3)$ with $8 \times 5 + 8 \times 3$, and
        then $8 \times (5 \times 3)$ with
        $(8 \times 5) \times (8 \times 3)$: which pair matches,
        which does not?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The doubling--halving method.}
Here is the recipe for $13 \times 24$. Write two columns. Left
column: start at $13$ and \emph{halve}, dropping any \omterm{def:g3:division:remainder}{remainder},
down to $1$: \ $13,\ 6,\ 3,\ 1$. Right column: start at $24$ and
\emph{double} as many times: \ $24,\ 48,\ 96,\ 192$. Cross out
every row whose \emph{left} number is \omterm{def:g3:numbers:evenodd}{even}, and add the surviving
right numbers.
\begin{enumerate}[resume]
  \item Carry out the recipe for $13 \times 24$: which row is
        crossed out, what \omterm{def:g1:addition:def}{sum} remains, and does it equal
        $13 \times 24$?
  \item Multiply $21 \times 17$ by the method (halve $21$, double
        $17$), and check the answer by ordinary multiplication.
  \item Multiply $16 \times 25$ by the method. How many rows
        survive, and what is special about a left column that
        starts at $1$, $2$, $4$, $8$, $16, \dots$?
  \item Now \emph{why} it works, one step at a time. Justify each
        equality with distributivity:
        \[
        13 \times 24 = (12 + 1) \times 24
        = 12 \times 24 + 24 = 6 \times 48 + 24 .
        \]
        Moral: halving the left and doubling the right keeps the
        \omterm{def:g2:mult:def}{product} --- but when the left number is \emph{\omterm{def:g3:numbers:evenodd}{odd}}, one
        copy of the right number splits off and must be banked.
        Which number does the first row of the table bank?
  \item Continue the bookkeeping down the table:
        $6 \times 48 = 3 \times 96$, then
        $3 \times 96 = 1 \times 192 + 96$, then
        $1 \times 192 = 192$. Collect everything that was banked
        along the way, and explain why the banked numbers are
        exactly the right-column numbers of the rows with an \omterm{def:g3:numbers:evenodd}{odd}
        left --- the rows the recipe keeps.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Under the hood: \omterm{def:g1:addition:def}{sums} of
doublings.}
\begin{enumerate}[resume]
  \item In question 6, the surviving rows were worth $1$, $4$
        and $8$ doublings of $24$: the method silently wrote
        $13 \times 24 = (1 + 4 + 8) \times 24$. Check that
        $1 + 4 + 8 = 13$, and find which doublings ($1$, $2$,
        $4$, $8$, $16$) the method kept for $21$ in question 7.
  \item Write $25$ and $42$ as \omterm{def:g1:addition:def}{sums} of numbers from the doubling
        list $1, 2, 4, 8, 16, 32$, using each at most once.
  \item Explain why \emph{every} whole number can be written
        this way: take the largest doubling number that fits,
        subtract, repeat --- why must this stop? Write $100$ as
        such a \omterm{def:g1:addition:def}{sum}.
  \item The scribes of Egypt did exactly this, four thousand
        years ago: to multiply $19 \times 35$, they made a
        doubling table of $35$ ($1 \to 35$, $2 \to 70$,
        $4 \to 140$, $8 \to 280$, $16 \to 560$), picked the rows
        adding up to $19$, and summed. Do the computation their
        way.
  \item Finale: modern computers, which count with $0$s and
        $1$s, multiply essentially like the scribes --- by
        doublings and additions. Which three additions would a
        computer make for $13 \times 24$? (Why every number
        splits into doublings, and what $0$s and $1$s have to do
        with it, is the weekend problem of the Powers chapter:
        \cref{pb:g8:powers:1}.)
\end{enumerate}
\end{problem}
