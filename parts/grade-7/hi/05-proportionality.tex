\section{तालिकाओं को पहचानना और पूरा करना}

\begin{definition}[आनुपातिकता तालिका]\label{def:g7:prop:table}
दो पंक्तियों की एक तालिका है \emph{आनुपातिकता
टेबल}\index{आनुपातिकता
टेबल} जब दूसरी पंक्ति के नंबर हों
पहले वाले को एक निश्चित संख्या से गुणा किया जाता है
\emph{गुणक}. समतुल्य: सभी स्तंभ \omterm{def:g3:division:remainder}{भागफल}
$\frac{\text{bottom}}{\text{top}}$ बराबर हैं.
\end{definition}

\begin{example}\label{ex:g7:prop:recognize}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|ccc}
top & $4$ & $10$ & $14$ \\
\hline
bottom & $6$ & $15$ & $21$ \\
\end{tabular}
\end{center}
\omterm{def:g3:division:remainder}{भागफल}: $\frac64 = 1.5$, $\frac{15}{10} = 1.5$,
$\frac{21}{14} = 1.5$: \omterm{def:g6:propdata:def}{आनुपातिक}, गुणांक $1.5$.
\end{example}

\begin{theorem}[क्रॉस नियम]\label{thm:g7:prop:cross}
में एक आनुपातिकता तालिका स्तंभों के साथ $\begin{pmatrix} a \\ b
\end{pmatrix}$ और $\begin{pmatrix} c \\ d \end{pmatrix}$, क्रौस
\omterm{def:g2:mult:def}{उत्पाद}ों बराबर हैं:
\[
a \times d = b \times c .
\]
परिणामस्वरूप चौथे मान की गणना तीन अन्य से की जा सकती है:
$ d = \dfrac{b \times c}{a}$.
\end{theorem}

\begin{proof}
होने देना $ k $ गुणांक हो: $ b = ka $ और $ d = kc $. तब
\[
a \times d = a \times kc = k \times ac,
\qquad
b \times c = ka \times c = k \times ac :
\]
दोनों पार \omterm{def:g2:mult:def}{उत्पाद}ों बराबर $ k \times ac $. डिवाइडिंग $ ad = bc $ द्वारा $ a $
का सूत्र देता है $ d $.
\end{proof}

\begin{example}[चतुर्थ आनुपातिक]\label{ex:g7:prop:fourth}
अगर $7$ समान पुस्तकों का वजन होता है $2.8$ किलो, कितना करते हैं $12$ ऐसी किताबें
वजन? टेबल और क्रॉस नियम:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cc}
books & $7$ & $12$ \\
\hline
kg & $2.8$ & $ m $ \\
\end{tabular}
\end{center}
\[
7 \times m = 2.8 \times 12
\quad\Longrightarrow\quad
m = \frac{2.8 \times 12}{7} = \frac{33.6}{7} = 4.8 \text{ kg}.
\]
इकाई द्वारा जांचें: एक किताब का वजन होता है $0.4$ किलो, बारह वजन $4.8$ किग्रा.
\end{example}

\section{प्रतिशत}

\begin{method}[प्रतिशत लागू करना और ज्ञात करना]\label{met:g7:prop:percent}
\begin{enumerate}
  \item \emph{आवेदन करना} $ t\,\%$: गुणा करके $\frac{t}{100}$
        ($12\,\%$ का $350$ है $0.12 \times 350 = 42$);
  \item \emph{खोजो} वह प्रतिशत जो एक भाग दर्शाता है: गणना करें
        $\frac{\text{part}}{\text{whole}}$ और फिर से लिखें $100$
        ($21$ से बाहर $60$: $\frac{21}{60} = \frac{35}{100} = 35\,\%$);
  \item हमेशा पूछिये: \emph{किसका प्रतिशत?} संदर्भ संपूर्ण
        मायने रखता है.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:prop:percent}
एक स्कूल में, $180$ की $400$ छात्र लड़कियां हैं. प्रतिशत:
\[
\frac{180}{400} = \frac{180 \div 4}{400 \div 4} = \frac{45}{100}
= 45\,\% .
\]
दूसरा तरीका: कितने हैं $55\,\%$ लड़के?
$0.55 \times 400 = 220$. जाँच करना: $180 + 220 = 400$.
\end{example}

\section{तराजू}

\begin{definition}[पैमाना]\label{def:g7:prop:scale}
किसी मानचित्र या मॉडल पर \emph{पैमाना}\index{पैमाना}
$\frac{1}{n}$ (लिखा हुआ $1 : n $), मानचित्र पर दूरियाँ हैं
\omterm{def:g6:propdata:def}{आनुपातिक} वास्तविक दूरियों तक, गुणांक के साथ $\frac1n $:
\[
\text{map distance} = \frac{1}{n} \times \text{real distance},
\]
दोनों में मापा गया \emph{एक ही इकाई}.
\end{definition}

\begin{example}\label{ex:g7:prop:scale}
एक पर $1 : 50\,000$ मानचित्र, दो कस्बे हैं $7$ सेमी अलग. वास्तविक दूरी:
\[
7 \times 50\,000 = 350\,000 \text{ cm} = 3\,500 \text{ m} = 3.5
\text{ km}.
\]
इसके विपरीत, ए $12$ मानचित्र पर किमी लंबी पैदल यात्रा पथ के उपाय
$12$ किमी $= 1\,200\,000$ सेमी, तो $1\,200\,000 \div 50\,000 = 24$ सेमी।
\end{example}

दूरियाँ मूल बिन्दु से होकर एक सीधी \omterm{def:g6:lines:objects}{रेखा} पर ऊपर उठती हैं (यहाँ)।
$1$ सेमी $\leftrightarrow $ $4$ किमी; डैश पढ़ते हैं $7$ सेमी
$\leftrightarrow $ $28$ किमी).}
\end{omfigure}

\begin{remark}
ग्राफ़ सबसे तेज़ परीक्षण हैं समानता
(\cref{ch:g6:propdata}): एक सीधी \omterm{def:g6:lines:objects}{रेखा} पर बिंदु \emph{के माध्यम से
उत्पत्ति} अर्थ \omterm{def:g6:propdata:def}{आनुपातिक} मात्राएँ; एक पंक्ति जिसका मूल बिंदु गायब है (जैसे
एक निश्चित शुल्क के साथ टैक्सी का किराया) का अर्थ है \emph{नहीं} \omterm{def:g6:propdata:def}{आनुपातिक}, यहां तक कि
हालाँकि यह एक सीधी \omterm{def:g6:lines:objects}{रेखा} है. एफ़िन फ़ंक्शंस, में
\cref{ch:g9:linfunc}, इसे सटीक बना देगा।
\end{remark}

\section{समस्या: छड़ी से पृथ्वी को मापना}

\begin{problem}[{सप्ताहांत समस्या---छायाएँ एक हैं आनुपातिकता, और एराटोस्थनीज़ ने परिधि की गणना कैसे की 240 ईसा पूर्व में पृथ्वी का}]
\label{pb:g7:prop:1}
बाईस शताब्दी पहले, न तो कोई दूरबीन थी, न कोई उपग्रह और न ही कोई
कैलकुलेटर, अलेक्जेंड्रिया के लाइब्रेरियन ने पृथ्वी को मापा ---
एक छड़ी, एक कुआँ, एक ऊँट कारवां और इस अध्याय का उपयोग करना
गणित. यह समस्या उसके कदमों को पीछे ले जाती है: सबसे पहले
समानता छाया की, फिर प्रसिद्ध गणना ही,
और अंततः उसका उत्तर मानव पर कैसा दिखता है \omterm{def:g7:prop:scale}{पैमाना}.

\medskip
\noindent\textbf{Part I --- The shadow of a stick.}
सूरज इतना दूर है कि उसकी किरणें हम तक पहुँचती हैं \omterm{def:g6:lines:perp}{समानांतर} एक दूसरे के लिए। एक निश्चित समय पर, सभी ऊर्ध्वाधर वस्तुएं और उनकी
छाया इसलिए हैं \omterm{def:g6:propdata:def}{आनुपातिक}.
\begin{enumerate}
  \item दोपहर के समय, एक खड़ी छड़ी $1$ मी की छाया पड़ती है
        $0.4$ मी, और एक पेड़ की छाया मापती है $3.2$ एम। कितना लंबा
        पेड़ है (\cref{thm:g7:prop:cross})?
  \item उसी क्षण, एक टावर गिरता है $26$ मी छाया: कैसे
        टावर कितना ऊंचा है? और बच्चे की परछाई कितनी लंबी होती है
        $1.5$ मैं लंबा हूँ?
  \item एक वाक्य में बताएं कि ये सभी गणनाएँ क्यों हैं
        पर ही मान्य है \emph{वही क्षण} दिन का।
  \item किंवदंती कहती है कि थेल्स ने महान को मापा था \omterm{def:g5:solids:def}{पिरामिड} इंतज़ार करके
        उस पल के लिए जब \emph{उसकी अपनी परछाई बिल्कुल वैसी ही थी
        जब तक वह लम्बा था}. क्या है समानता
        उस पल में गुणांक, और कितना लंबा है \omterm{def:g5:solids:def}{पिरामिड}
        यदि उसकी छाया का सिरा झूठ है $147$ के बिंदु से मी
        इसके शीर्ष के ठीक नीचे की भूमि?
  \item भाग I को सारांशित करें: एक निश्चित समय पर, कौन सी मात्रा है
        छड़ी, पेड़, मीनार और के लिए भी यही बात लागू है
        \omterm{def:g5:solids:def}{पिरामिड} (\cref{def:g7:prop:table})?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Eratosthenes' computation.}
एराटोस्थनीज दो तथ्य जानता था। सायन शहर में, दोपहर के समय
ग्रीष्म संक्रांति, सूर्य खड़ा था \emph{बिल्कुल उपरि}: सूरज की रोशनी
सबसे गहरे कुओं की तली तक पहुँच गया, और खड़ी छड़ियाँ डाली गईं
कोई छाया नहीं. अलेक्जेंड्रिया में, $800$ उत्तर की ओर किमी, उसी क्षण,
एक खड़ी छड़ी \emph{किया} छाया डालें, और सूरज की किरणें
का एक कोण बनाया $7.2^\circ $ ऊर्ध्वाधर के साथ.
\begin{enumerate}[resume]
  \item बताएं कि ये दोनों अवलोकन मिलकर ऐसा क्यों साबित करते हैं
        साइने की भूमि और अलेक्जेंड्रिया की भूमि नहीं हैं
        \omterm{def:g6:lines:perp}{समानांतर} ---अर्थात् पृथ्वी की सतह घुमावदार है।
        (क्या होगा \omterm{def:g6:lines:perp}{समानांतर} किरणें एक समतल पर दो छड़ियों तक पहुँचती हैं
        पृथ्वी?)
  \item के एक चित्र पर \omterm{def:g4:numbers:round}{गोल} पृथ्वी के साथ \omterm{def:g6:lines:perp}{समानांतर} सूरज की किरणें,
        द $7.2^\circ $ अलेक्जेंड्रिया के केंद्र में फिर से प्रकट होता है
        पृथ्वी, की दिशाओं के बीच के कोण के रूप में
        दो शहर. चित्र बनाएं और स्वयं को आश्वस्त करें (\dots
        स्वच्छ औचित्य समान-कोण \omterm{def:g1:addition:def}{जोड़}े का उपयोग करता है
        \cref{ch:g7:triangles}).
  \item क्या \omterm{def:g6:fractions:def}{अंश} का एक पूर्ण मोड़ है $7.2^\circ $?
  \item साइने से अलेक्जेंड्रिया तक चाप ($800$ किमी) से मेल खाता है
        को $7.2^\circ $; पूरी \omterm{def:g6:measure:perimeter}{परिधि} मेल खाती है

\noindent\textbf{Part III --- The Earth at human \omterm{def:g7:prop:scale}{पैमाना}.}
\begin{enumerate}[resume]
  \item पर एक ग्लोब बनाया गया है \omterm{def:g7:prop:scale}{पैमाना} $1 : 40\,000\,000$. का उपयोग करना
        प्रश्न 9 में पाई गई \omterm{def:g6:measure:perimeter}{परिधि} दर्शाइए कि
        ग्लोब की \omterm{def:g6:measure:perimeter}{परिधि} बिल्कुल है $1$ एम। यह क्या है
        \omterm{def:g4:geometry:circle}{व्यास}, निकटतम सेंटी\omterm{def:g4:fractions:def}{मीटर} तक?
  \item मोंट ब्लांक ऊपर उठता है $4.8$ किमी. बदलना $4.8$ किमी से
        सेंटी\omterm{def:g4:fractions:def}{मीटर}, से विभाजित करें $40\,000\,000$, और व्यक्त करें
        ग्लोब पर पर्वत की ऊँचाई मिली\omterm{def:g4:fractions:def}{मीटर} में। क्या करता है
        उत्तर बताता है कि पृथ्वी वास्तव में कितनी ``ऊबड़-खाबड़'' है?
  \item एक वॉकर कवर करता है $40$ प्रति दिन किमी. उस गति से, कितने
        एराटोस्थनीज़ की पूरी \omterm{def:g6:measure:perimeter}{परिधि} के लिए दिन --- और
        मोटे तौर पर वह कितने वर्ष है?
  \item सांस लेने योग्य वातावरण मोटे तौर पर केंद्रित है
        प्रथम $10$ जमीन से ऊपर किमी. का कितना प्रतिशत
        पृथ्वी का \omterm{def:g3:shapes:shapes}{RADIUS} (के बारे में $6\,370$ किमी) वह है
        (\cref{met:g7:prop:percent})? निकटतम तक \omterm{def:g4:numbers:round}{गोल} करें
        एक प्रतिशत का दसवां हिस्सा.
  \item इतिहासकारों का अनुमान है कि एराटोस्थनीज़ द्वारा घोषित मूल्य,
        आधुनिक इकाइयों में परिवर्तित, के बारे में हो सकता है
        $42\,000$ किमी. लेना $40\,000$ वास्तविक मान के रूप में किमी,
        उसकी त्रुटि प्रतिशत की गणना करें। एक में निष्कर्ष निकालें
        लाठी, कुएँ और के बारे में वाक्य समानता.
\end{enumerate}
