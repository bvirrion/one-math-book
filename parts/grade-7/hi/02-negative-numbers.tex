\chapter{नकारात्मक संख्याएँ}\label{ch:g7:negatives}

तापमान नीचे \omterm{def:g1:counting:zero}{शून्य}, फर्श ज़मीन से नीचे, पैसा बकाया: बहुत सारे
मात्राएँ नीचे जाती हैं $0$. ऋणात्मक संख्याएँ संख्या \omterm{def:g6:lines:objects}{रेखा} का विस्तार करती हैं
के बाएँ \omterm{def:g1:counting:zero}{शून्य}. यह अध्याय सिखाता है कि उनकी तुलना कैसे करें और कैसे \omterm{def:g1:addition:def}{जोड़}ें
और उन्हें घटाएं --- उनका गुणा इंतजार करता है
\cref{ch:g8:negprod}.

\section{सापेक्ष संख्याएँ}

\begin{definition}[सापेक्ष संख्याएँ]\label{def:g7:negatives:def}
ए \emph{सापेक्ष संख्या}\index{सापेक्ष संख्या} एक चिन्ह से बना है
($+$ या $-$) और ए \emph{की दूरी \omterm{def:g1:counting:zero}{शून्य}}: $+3$ और $-3$ दोनों पर हैं
दूरी $3$ से $0$, विपरीत पक्षों पर। सकारात्मक संख्याएँ आमतौर पर होती हैं
उनके हस्ताक्षर के बिना लिखा गया: $+3 = 3$. संख्या $+3$ और $-3$ हैं
\emph{विपरीत}\index{विपरीत} एक दूसरे के.
\end{definition}

\begin{tikzpicture}[scale=1.15]

  \foreach \x in {-5,...,5} \draw (\x,-0.08) -- (\x,0.08);

\end{tikzpicture}

{\small संख्या \omterm{def:g6:lines:objects}{रेखा} बाईं ओर विस्तारित: $-3$ और $+3$ हैं
विपरीत संख्याएँ, समान दूरी पर $0$.}
\end{omfigure}

\begin{method}[सापेक्ष संख्याओं की तुलना करना]\label{met:g7:negatives:compare}
संख्या \omterm{def:g6:lines:objects}{रेखा} पर छोटे का मतलब आगे बाईं ओर है। व्यवहार में:
\begin{enumerate}
  \item ऋणात्मक संख्या सदैव धनात्मक संख्या से छोटी होती है:
        $-7 < 2$;
  \item दो सकारात्मकताओं के बीच, सामान्य क्रम;
  \item दो नकारात्मकताओं के बीच, एक के साथ \emph{बड़ा} दूरी
        को \omterm{def:g1:counting:zero}{शून्य} है \emph{छोटे}: $-7 < -2$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:negatives:compare}
सबसे छोटे से सबसे बड़े तक ऑर्डर करें: $3$; $-5$; $0$; $-1.5$; $2.8$.
सबसे पहले नकारात्मक, सबसे नकारात्मक अग्रणी:
\[
-5 < -1.5 < 0 < 2.8 < 3 .
\]
\end{example}

\section{सापेक्ष संख्याएँ जोड़ना}

एक के बारे में सोचो \omterm{def:g7:negatives:def}{सापेक्ष संख्या} लाइन के साथ एक आंदोलन के रूप में: $+5$ मतलब
``चलना $5$ दाएँ कदम'', $-3$ का अर्थ है ``चलना $3$ चरण बाएँ''। जोड़ना
संख्याओं का अर्थ है चलने को जंजीर बनाना।

\begin{method}[दो सापेक्ष संख्याएँ जोड़ना]\label{met:g7:negatives:add}
\begin{enumerate}
  \item \emph{वही संकेत}: दूरियाँ \omterm{def:g1:addition:def}{जोड़}ें, उभयनिष्ठ चिह्न रखें:
        $(-3) + (-4) = -7$.
  \item \emph{विपरीत संकेत}: से छोटी दूरी घटाएं
        बड़ा, बड़े का चिह्न लें:
        $(-7) + (+2) = -5$ और $(+7) + (-2) = +5$.
  \item विपरीत रद्द करना: $(+4) + (-4) = 0$.
\end{enumerate}
\end{method}

\begin{tikzpicture}[scale=1.05]

  \foreach \x in {-5,...,4} \draw (\x,-0.07) -- (\x,0.07);

{\small चलने के रूप में जोड़ना: से शुरू करना $0$, टहलना $-3.5$ फिर
चलना $+2$ पर उतरता है $(-3.5) + 2 = -1.5$.}
\end{omfigure}

\begin{example}\label{ex:g7:negatives:add}
चरण दर चरण, सुविधाजनक के रूप में समूहीकरण:
\begin{align*}
(-8) + (+3) &= -5 && \text{(opposite signs: } 8 - 3 = 5,
\text{ sign of } 8)\\
(-2.5) + (-1.5) &= -4 && \text{(same sign: add distances)}\\
(+6) + (-9) + (+3) &= (-3) + (+3) = 0 .
\end{align*}
\end{example}

\section{सापेक्ष संख्याओं को घटाना}

\begin{theorem}[घटाव नियम]\label{thm:g7:negatives:sub}
ए घटाना \omterm{def:g7:negatives:def}{सापेक्ष संख्या} वैसा ही है जैसा कि \emph{इसे जोड़ना
विपरीत}:
\[
a - b = a + (-b) .
\]
\end{theorem}

\begin{proof}[यह क्यों काम करता है]
$ a - b $ वह संख्या है जिसे \omterm{def:g1:addition:def}{जोड़}ा गया है $ b $ देता है $ a $. जाँच करें कि
$ a + (-b)$ कार्य करता है:
$\bigl(a + (-b)\bigr) + b = a + \bigl((-b) + b\bigr) = a + 0 = a $.
\end{proof}

\begin{example}\label{ex:g7:negatives:sub}
तब प्रत्येक घटाव एक \omterm{def:g1:addition:def}{जोड़} बन जाता है
\cref{met:g7:negatives:add} लागू होता है:
\begin{align*}
5 - 9 &= 5 + (-9) = -4, \\
(-3) - (+4) &= (-3) + (-4) = -7, \\
(-2) - (-6) &= (-2) + (+6) = +4 .
\end{align*}
अंतिम पंक्ति प्रसिद्ध है: \emph{ऋणात्मक साधन घटाने पर
जोड़ना}. कर्ज उतारना आपको अमीर बनाता है!
\end{example}

\begin{example}[तापमान में अंतर]\label{ex:g7:negatives:temperature}
एक ही दिन में तापमान कहां से कहां चला गया $-6$ डिग्री तक $+9$ डिग्री. द
वृद्धि है \omterm{ex:g1:subtraction:difference}{अंतर}
\[
9 - (-6) = 9 + 6 = 15 \text{ degrees}.
\]
संख्या \omterm{def:g6:lines:objects}{रेखा} पर: से $-6$ तक $0$ है $6$ डिग्री, से $0$ को
$9$ है $9$ अधिक।
\end{example}

\section{पूरे विमान में निर्देशांक}

\begin{definition}[संकेतों के साथ समन्वय करता है]\label{def:g7:negatives:coords}
दो के साथ \omterm{def:g6:lines:perp}{सीधा} \omterm{def:g6:fractions:def}{अंश}ांकित कुल्हाड़ियाँ मूल बिंदु पर प्रतिच्छेद करती हैं $ O $,
समतल के प्रत्येक बिंदु को दो सापेक्ष निर्देशांक मिलते हैं $(x, y)$:
नकारात्मक $ x $ मतलब ऊर्ध्वाधर अक्ष के बाईं ओर, नकारात्मक $ y $ मतलब नीचे
क्षैतिज अक्ष.
\end{definition}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g7:negatives:1}
इसका विपरीत बताइये: $7$; \ $-3.5$; \ $0$; \ $+12$. फिर पूरा करें:
``दो संख्याएँ हैं विपरीत जब उनका \omterm{def:g1:addition:def}{जोड़} है~?''
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:negatives:2}
कॉपी करें और पूरा करें $<$ या $>$:
\[
-4 \;?\; 1, \qquad
-4 \;?\; -7, \qquad
-2.5 \;?\; -2.4, \qquad
0 \;?\; -0.1 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:negatives:3}
सबसे छोटे से सबसे बड़े तक ऑर्डर करें: $-3$; \ $2.5$; \ $-0.5$; \ $-3.1$;
\ $0$; \ $1$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:negatives:4}
गणना करें:
\[
(-5) + (-9), \qquad
(-12) + (+7), \qquad
(+3.5) + (-1.5), \qquad

\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:negatives:5}
परिवर्धन में रूपांतरित करें, फिर गणना करें:
\[
4 - 11, \qquad
(-5) - (+3), \qquad
(-7) - (-10), \qquad

\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:negatives:6}
चतुराई से समूह बनाकर गणना करें (विपरीत पहला):
\[
(+7) + (-12) + (+3) + (-8), \qquad
(-5.5) + (+9) + (-4.5) + (+1).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:negatives:7}
तापमान था $-3$ आज सुबह डिग्री; यह बढ़ गया $8$ डिग्री
दिन के दौरान, फिर गिरा दिया गया $12$ रात में डिग्री. एक एकल लिखें
अंतिम तापमान बताने वाली गणना, और इसकी गणना करें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:negatives:8}
एक समन्वय प्रणाली में प्लॉट करें: $ A(2, 3)$; $ B(-3, 1)$; $ C(-2, -2)$;
$ D(3, -1)$; $ E(0, -3)$. कौन सा बिंदु एक अक्ष पर स्थित है?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:negatives:9}
की गणना करें \omterm{ex:g1:subtraction:difference}{अंतर} उच्चतम और न्यूनतम तापमान के बीच:
जनवरी में मास्को, अधिकतम $-4$ डिग्री, न्यूनतम $-13$ डिग्री; हेलसिंकी, मैक्स
$2$, मि $-9$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:negatives:10}
एक गोताखोर ऊंचाई पर है $-12$ मी (सतह से बारह \omterm{def:g4:fractions:def}{मीटर} नीचे)। वह
उगता है $7$ मी, फिर उतरता है $4$ एम। वह किस ऊंचाई पर है? कितनी दूर
क्या उसे अभी भी सतह पर पहुँचने के लिए उठना होगा?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:negatives:11}
कथानक $ A(-2, 1)$, $ B(2, 1)$, $ C(2, -2)$. के निर्देशांक ज्ञात कीजिए
बिंदु $ D $ ताकि $ ABCD $ एक आयत है, और इसकी गणना करें \omterm{def:g6:measure:perimeter}{परिधि}
(साइड की लंबाई के लिए ग्रिड वर्गों की गिनती करें)।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g7:negatives:12}

पूर्ण.)
\end{exercise}

\section{समस्या: वह वर्ष शून्य जो कभी था ही नहीं}

\begin{problem}[{सप्ताहांत समस्या --- सापेक्ष संख्याएँ इतिहास की समयरेखा: युग सीमा के पार की अवधि खगोलविदों की चाल, और नेट-परिवर्तन सिद्धांत}]
\label{pb:g7:negatives:1}
इस अध्याय की संख्या \omterm{def:g6:lines:objects}{रेखा} में वास्तविक दुनिया का एक प्रसिद्ध जुड़वां है: द
इतिहास की समयरेखा, ईसा पूर्व के वर्षों को बाईं ओर खींचते हुए
ईस्वी सन् के वर्ष दाहिनी ओर फैले हुए हैं। लेकिन इतिहासकारों की पंक्ति में एक \omterm{def:g5:solids:net}{जाल} छिपा है
इससे कई गणनाएं खराब हो गई हैं: \emph{कोई वर्ष नहीं है \omterm{def:g1:counting:zero}{शून्य}}
--- वर्ष 1 ई.पू. के तुरंत बाद ई.पू. 1 आता है। यह समस्या
तापमान, लिफ्ट और बैंक खातों की गणना करता है, फिर मरम्मत करता है
इतिहास टूट गया \omterm{def:g1:counting:zero}{शून्य} जिस तरह से खगोलशास्त्री करते हैं: साथ रिश्तेदार
नंबर.

\medskip
\noindent\textbf{Part I --- Warm-ups, and a first trap.}
\begin{enumerate}
  \item मोंट ब्लांक का शिखर ऊंचाई पर है $+4\,808$ एम; द
        मृत सागर के तट पर $-430$ एम। की गणना करें \omterm{ex:g1:subtraction:difference}{अंतर}
        उनके बीच की ऊंचाई में.
  \item एक बैंक खाता रखता है $-45$ यूरो (एक ओवरड्राफ्ट)। यह
        मालिक जमा $120$ यूरो का बिल चुकाता है $60$ यूरो,
        फिर जमा $15$ यूरो. \omterm{def:g3:division:remainder}{शेष} को एकल के रूप में लिखें
        परिवर्धन की श्रृंखला और इसकी गणना करें। क्या एकल जमा
        ओरिजिनल लेकर आते $-45$ बिलकुल वापस
        $0$?
  \item जूलियस सीज़र का जन्म हुआ था $100$ ईसा पूर्व और में मृत्यु हो गई $44$ ईसा पूर्व.
        BC वर्षों को ऋणात्मक संख्याओं के रूप में गिनना ($100$ ईसा पूर्व के रूप में
        $-100$, $44$ ईसा पूर्व के रूप में $-44$), गणना करें $(-44) - (-100)$: कैसे
        सीज़र कितने वर्ष जीवित रहा?
  \item सम्राट ऑगस्टस का जन्म हुआ था $63$ ईसा पूर्व और मृत्यु ईसवी में हुई
        $14$. वही नुस्खा देता है $14 - (-63) = 77$ वर्ष ---
        लेकिन इतिहासकार इस बात पर ज़ोर देते हैं कि उनकी मृत्यु यहीं हुई $76$. उनकी आपत्ति:
        गिनती $77$ एक ऐसे वर्ष से गुजरता है जो कभी अस्तित्व में नहीं था।
        कौन सा?
  \item खगोलशास्त्री पुनः लेबल करके समयरेखा की मरम्मत करते हैं: AD वर्ष
        उनकी संख्या रखें ($+14$ रहता है $+14$), लेकिन $1$ ईसा पूर्व
        बन जाता है $0$, $2$ बीसी बन जाता है $-1$, और आम तौर पर वर्ष
        $ n $ बीसी बन जाता है $-(n - 1)$. बदलना $63$ बीसी, और फिर से करें
        खगोलशास्त्रियों की संख्या के साथ प्रश्न 4 की गणना।
        क्या अब उत्तर इतिहासकारों को संतुष्ट करता है?
\end{enumerate}

\noindent\textbf{Part III --- The net-change principle.}
\begin{enumerate}[resume]
  \item  \emph{दूरी} \omterm{def:g6:lines:objects}{रेखा} पर दो संख्याओं के बीच है
        बड़ा \omterm{def:g1:counting:zero}{शून्य} से छोटा --- हमेशा सकारात्मक।
        के बीच की दूरी की गणना करें $-7$ और $-2$, फिर बीच में
        $-3.5$ और $4.5$.
  \item एक लिफ्ट मंजिल से शुरू होती है $0$ और चाल चलता है $+3$,
        $-5$, $+2$, $-1$, $+4$. द्वारा अंतिम मंजिल की गणना करें
        चतुराई से समूह बनाना (\cref{met:g7:negatives:add},
        विपरीत पहले जहां संभव हो)। का प्रदर्शन करेंगे
        एक अलग क्रम में चलने से गंतव्य बदल जाता है? क्यों?
  \item दिन-ब-दिन तापमान में बदलाव होता जा रहा है $+2$, $-4$, $+1$,
        $0$, $-3$, $+5$, $-2$ एक सप्ताह में डिग्री. की गणना करें
        \emph{\omterm{def:g5:solids:net}{जाल} परिवर्तन}. यदि सप्ताह प्रारंभ हुआ $-2$ डिग्री,
        यह कहां समाप्त हुआ? सिद्धांत बताएं: अंतिम मूल्य $=$
        आरंभिक मूल्य $+$ (\omterm{def:g1:addition:def}{जोड़} सभी परिवर्तनों में से)।
  \item एक पैदल यात्री की पैदल यात्रा उसे ठीक उसकी शुरुआत पर वापस लाती है
        ऊंचाई. क्या चाहिए \omterm{def:g1:addition:def}{जोड़} उसकी सारी ऊंचाई बदल जाती है
        हो, और क्यों? उसके दर्ज किए गए छह परिवर्तनों में से पाँच हैं
        $+4$, $-7$, $+2$, $+3$, $-5$ (सैकड़ों \omterm{def:g4:fractions:def}{मीटर} में):
        छठा खोजें.
  \item समापन, रोम में वापस। एक इतिहासकार गणना करता है: ``द
        गणतंत्र से भाग गया $509$ बी.सी. को $27$ ईसा पूर्व, फिर साम्राज्य
        से ई.पू $476$; सबमें, से $509$ ईसा पूर्व से ई.पू $476$, वह है
        $509 + 476 = 985$ ''रोमन राज्य के वर्ष।'' फिर से करें
        खगोलशास्त्रियों की संख्या के साथ गणना। क्या है
        सही कुल --- और कौन सा वर्ष, एक बार फिर, इसका कारण बना
        त्रुटि?
\end{enumerate}
\end{problem}
