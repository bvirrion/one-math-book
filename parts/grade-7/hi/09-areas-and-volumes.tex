\chapter{क्षेत्र और आयतन}\label{ch:g7:areas}

ग्रेड 6 मापे गए आयत, \omterm{def:g3:shapes:rightangle}{समकोण} त्रिभुज और बक्से
(\cref{ch:g6:measure}). कैंची की एक \omterm{def:g1:addition:def}{जोड़}ी से --- यहाँ एक टुकड़ा काटें,
इसे वहां चिपका दें --- वही विचार अब देते हैं \omterm{def:g6:measure:area}{क्षेत्र}ों का \omterm{def:g6:lines:perp}{समानांतर} \omterm{def:g7:central:parallelogram}{चतुर्भुज},
का \emph{सभी} \omterm{def:g3:shapes:shapes}{त्रिकोण} और डिस्क के, और संस्करणों का प्रिज्म और
\omterm{def:g7:areas:prism}{सिलेंडर}.

\section{समांतर चतुर्भुज का क्षेत्रफल}

\begin{theorem}[चतुर्भुज]\label{thm:g7:areas:parallelogram}
ए \omterm{def:g7:central:parallelogram}{चतुर्भुज} लंबाई की एक भुजा के साथ $b $ (इसका \emph{आधार}) और
दूरी $ h $ उस पक्ष और विपरीत पक्ष के बीच (इसके)
\emph{ऊंचाई}) है \omterm{def:g6:measure:area}{क्षेत्र}
\[
A = b \times h .
\]
\end{theorem}

\begin{proof}[कैंची द्वारा प्रमाण]
काट दो \omterm{def:g6:shapes:triangles}{समकोण का} \omterm{def:g3:shapes:shapes}{त्रिकोण} को एक तरफ से चिपकाकर चिपका दें
दूसरे पर: \omterm{def:g7:central:parallelogram}{चतुर्भुज} एक बन जाता है $ b \times h $ के साथ आयत
वही \omterm{def:g6:measure:area}{क्षेत्र}.
\end{proof}

तिरछी तरफ की लंबाई नहीं! एक बहुत ही तिरछा \omterm{def:g7:central:parallelogram}{चतुर्भुज} कर सकते हैं
लंबी भुजाएँ और एक छोटी होती हैं \omterm{def:g6:measure:area}{क्षेत्र}.
\end{remark}

\section{एक त्रिभुज का क्षेत्रफल}

\begin{theorem}[त्रिकोण]\label{thm:g7:areas:triangle}
आधार वाला एक त्रिभुज $ b $ और संगत ऊँचाई $ h $ (द
\omterm{def:g6:lines:perp}{सीधा} विपरीत से दूरी शिखर आधार \omterm{def:g6:lines:objects}{रेखा} तक) है
\omterm{def:g6:measure:area}{क्षेत्र}
\[
A = \frac{b \times h}{2} .
\]
\end{theorem}

\begin{proof}
त्रिभुज की दो प्रतियाँ, एक को चारों ओर \omterm{def:g3:division:half}{आधा} मोड़कर घुमाया गया
एक तरफ का मध्यबिंदु, एक में एक साथ फिट \omterm{def:g7:central:parallelogram}{चतुर्भुज} आधार के साथ $ b $
और ऊंचाई $ h $ (वह है केंद्रीय समरूपता काम पर,
\cref{prop:g7:central:preserved}). \omterm{def:g3:shapes:shapes}{त्रिकोण} है \omterm{def:g3:division:half}{आधा} इसका:
$\frac{bh}{2}$.
\end{proof}

\omterm{def:g7:central:parallelogram}{चतुर्भुज}: \omterm{def:g3:shapes:shapes}{त्रिकोण} का \omterm{def:g6:measure:area}{क्षेत्र} है $\frac{b\times h}{2}$. इनमें से कोई भी
तीन भुजाएँ आधार के रूप में काम कर सकती हैं --- साथ \emph{अपना ही है} ऊंचाई।}
\end{omfigure}

\begin{example}\label{ex:g7:areas:triangle}
एक त्रिभुज का आधार होता है $9$ सेमी और संबंधित ऊंचाई माप
$4$ सेमी:
\[
A = \frac{9 \times 4}{2} = \frac{36}{2} = 18 \text{ cm}^2 .
\]
एक के लिए \emph{सही} त्रिभुज, दोनों पैर एक आधार हैं और इसकी ऊंचाई:
हम का सूत्र पुनर्प्राप्त करते हैं \cref{prop:g6:measure:formulas}.
\end{example}

\section{डिस्क का क्षेत्रफल}

\begin{theorem}[डिस्क]\label{thm:g7:areas:disk}
की एक डिस्क \omterm{def:g3:shapes:shapes}{RADIUS} $ r $ है \omterm{def:g6:measure:area}{क्षेत्र}
\[
A = \pi r^2 .
\]
\end{theorem}

\begin{proof}[प्रमाण का विचार]
डिस्क को कई पतली समान स्लाइस में काटें और उन्हें सिर से पूंछ तक रखें:
वे लगभग एक रूप बनाते हैं \omterm{def:g7:central:parallelogram}{चतुर्भुज} ऊंचाई का $ r $ जिसका आधार है \omterm{def:g3:division:half}{आधा} द
\omterm{def:g6:measure:perimeter}{परिधि}, $\pi r $ (\cref{prop:g6:measure:circle}). इसका \omterm{def:g6:measure:area}{क्षेत्र} है
के करीब $\pi r \times r = \pi r^2$, और सन्निकटन बन जाता है
जैसे-जैसे स्लाइस पतले होते जाते हैं, बिल्कुल सही। पूर्ण प्रमाण के लिए समग्रता की आवश्यकता होती है
कैलकुलस, हाई स्कूल वॉल्यूम से।
\end{proof}

\begin{example}\label{ex:g7:areas:disk}
एक \omterm{def:g4:numbers:round}{गोल} टेबल टॉप है \omterm{def:g3:shapes:shapes}{RADIUS} $60$ सेमी। इसका \omterm{def:g6:measure:area}{क्षेत्र} है
$\pi \times 60^2 = 3600\pi \approx 11\,310$ सेमी $^2$, से थोड़ा अधिक
$1.1$ एम $^2$. देखें \omterm{ex:g1:subtraction:difference}{अंतर}: द \emph{\omterm{def:g6:measure:perimeter}{परिधि}} $2\pi r $ उपयोग
$ r $, द \emph{\omterm{def:g6:measure:area}{क्षेत्र}} $\pi r^2$ उपयोग $ r^2$.
\end{example}

\section{प्रिज्म और सिलेंडर}

\begin{definition}[प्रिज्म, सिलेंडर]\label{def:g7:areas:prism}
ए \emph{चश्मे}\index{चश्मे} एक है \omterm{def:g5:solids:def}{ठोस} दो समान के साथ \omterm{def:g6:lines:perp}{समानांतर}
\omterm{def:g4:geometry:polygon}{बहुभुज} \omterm{def:g5:solids:def}{चेहरे} (द \emph{अड्डों}) आयतों से जुड़ा हुआ; ए
\emph{सिलेंडर}\index{सिलेंडर} यह डिस्क के साथ आधार के समान ही है।
\emph{ऊंचाई} दो आधारों के बीच की दूरी है.
\end{definition}

\begin{theorem}[संस्करणों]\label{thm:g7:areas:volume}
एक के लिए \omterm{def:g7:areas:prism}{चश्मे} या ए \omterm{def:g7:areas:prism}{सिलेंडर} आधार के साथ \omterm{def:g6:measure:area}{क्षेत्र} $ B $ और ऊंचाई $ h $:
\[
V = B \times h .
\]
\end{theorem}
\admitted

\begin{example}\label{ex:g7:areas:volume}
एक तंबू है \omterm{def:g7:areas:prism}{चश्मे} इसके किनारे पर स्थित: इसके आधार आधार के \omterm{def:g3:shapes:shapes}{त्रिकोण} हैं
$2$ मी और ऊंचाई $1.5$ मी, और तम्बू है $3$ मी लंबा.
\begin{enumerate}
  \item आधार \omterm{def:g6:measure:area}{क्षेत्र}: $ B = \frac{2 \times 1.5}{2} = 1.5$ एम $^2$.
  \item \omterm{def:g6:measure:volume}{आयतन}: $ V = B \times h = 1.5 \times 3 = 4.5$ एम $^3$.
\end{enumerate}
का एक टिन का \omterm{def:g5:solids:def}{डिब्बा} \omterm{def:g3:shapes:shapes}{RADIUS} $4$ सेमी और ऊंचाई $11$ सेमी:
$ V = \pi \times 4^2 \times 11 = 176\pi \approx 553$ सेमी $^3$, मोटे तौर पर
\omterm{def:g3:division:half}{आधा} एक लीटर।
\end{example}

\end{exercise}

\section{समस्या: लुप्त हो रहा वर्ग}

\begin{problem}[{सप्ताहांत समस्या --- एक प्रसिद्ध ``प्रमाण'' $64 = 65$, पिज्जा और अंगूठियों के साथ, क्षेत्रों और ढलानों से बेपर्दा मिठाई}]
\label{pb:g7:areas:1}
एक जादूगर एक काटता है $8 \times 8$ चार टुकड़ों में वर्गाकार, स्लाइड
उन्हें चारों ओर से घेरता है, और उन्हें पुनः एक में जोड़ता है $5 \times 13$ आयत।
श्रोतागण गिनते हैं: $8 \times 8 = 64$, लेकिन
$5 \times 13 = 65$ ---पतले से एक वर्गाकार इकाई प्रकट हुई है
वायु! इस अध्याय का \omterm{def:g6:measure:area}{क्षेत्र} सूत्र बिल्कुल वही उपकरण हैं जो
चाल पकड़ो. समस्या तब ईमानदार डालती है \omterm{def:g6:measure:area}{क्षेत्र} करने के लिए तर्क
कार्य: ट्रेपेज़ॉइड्स पर, झुके हुए ढेरों पर, पिज़्ज़ा अर्थशास्त्र पर --- और
अंततः एक क्षण के लिए जादूगर के पास लौट आता है, \omterm{def:g3:numbers:evenodd}{यहां तक की} अजनबी
प्रदर्शन.

\medskip
\noindent\textbf{Part I --- The trick, performed and unmasked.}
के चार टुकड़े $8 \times 8$ वर्ग हैं: दो ठीक है
\omterm{def:g3:shapes:shapes}{त्रिकोण} पैरों के साथ $8$ और $3$, और दो दाएँ समलम्ब \omterm{def:g7:central:parallelogram}{चतुर्भुज} के साथ
\omterm{def:g6:lines:perp}{समानांतर} दोनों पक्ष $5$ और $3$ और ऊंचाई $5$.
\begin{enumerate}
  \item की गणना करें \omterm{def:g6:measure:area}{क्षेत्र} मूल वर्ग का और \omterm{def:g6:measure:area}{क्षेत्र} का
        दावा किया गया आयत। इस घोटाले को एक पंक्ति में बताएं।
  \item की गणना करें \omterm{def:g6:measure:area}{क्षेत्र} चार टुकड़ों में से प्रत्येक का
        (\cref{thm:g7:areas:triangle}; ट्रेपेज़ॉइड के लिए, काटें
        या उपयोग करें \cref{exo:g7:areas:9}).
  \item चार \omterm{def:g1:addition:def}{जोड़}ें \omterm{def:g6:measure:area}{क्षेत्र}ों. दोनों आकृतियों में से कौन-सी --- वर्गाकार है
        या आयत --- क्या टुकड़े सचमुच भर सकते हैं?
  \item आयत व्यवस्था में, एक त्रिभुज का कर्ण
        (उठ रहा है $3$ आर-पार $8$) को जारी रखना चाहिए
        ट्रेपेज़ॉइड का झुका हुआ किनारा (उठता हुआ)। $2$ आर-पार $5$) एक में
        \omterm{def:g6:lines:perp}{सीधा} ``विकर्ण''। संरेखण का परीक्षण करें: हैं $3$ आर-पार
        $8$ और $2$ आर-पार $5$ वही ढलान
        (\cref{thm:g7:prop:cross})?
  \item बताएं कि पैंसठवाँ वर्ग कहाँ छिपा है: आकार क्या है
        झूठे विकर्ण के साथ \omterm{ex:g1:subtraction:difference}{अंतर}ाल, इसका सटीक क्या है \omterm{def:g6:measure:area}{क्षेत्र},
        और आंख इसे क्यों भूल जाती है? (औसतन कितना चौड़ा है
        उसका टुकड़ा \omterm{def:g6:measure:area}{क्षेत्र} आयत के ऊपर फैला हुआ
        विकर्ण, मोटे तौर पर $13$ इकाई लंबी?)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Honest \omterm{def:g6:measure:area}{क्षेत्र} reasoning.}
\begin{enumerate}[resume]
  \item दो ड्रा करें \omterm{def:g6:lines:perp}{समानांतर} रेखाएँ $4$ सेमी अलग, का एक आधार $6$ सेमी पर
        उनमें से एक, और उस आधार पर तीन अलग-अलग \omterm{def:g3:shapes:shapes}{त्रिकोण}
        दूसरी पंक्ति के विभिन्न बिंदुओं पर शीर्षों के साथ। गणना करें
        तीन \omterm{def:g6:measure:area}{क्षेत्र}ों. वे सभी समान क्यों हैं, फिर भी शीर्ष पर हैं
        इसकी \omterm{def:g6:lines:objects}{रेखा} के साथ स्लाइड?
  \item सामान्य रूप से समलम्ब \omterm{def:g7:central:parallelogram}{चतुर्भुज} सूत्र सिद्ध करें: एक समलम्ब \omterm{def:g7:central:parallelogram}{चतुर्भुज} के साथ
        \omterm{def:g6:lines:perp}{समानांतर} दोनों पक्ष $ B $ और $ b $ और ऊंचाई $ h $, साथ में काटें
        विकर्ण, समान ऊँचाई के दो त्रिभुज देता है $ h $.
        निष्कर्ष:
        \[
        A = \frac{(B + b) \times h}{2} .
        \]
  \item इसे एक ट्रेपेज़ॉइड पर लागू करें $ B = 9$ सेमी, $ b = 5$ सेमी,
        $ h = 4$ सेमी।
  \item बढ़ई भेष बदलने के लिए इसी सूत्र का उपयोग करते हैं: \omterm{def:g6:measure:area}{क्षेत्र} $=$
        (दोनों का औसत \omterm{def:g6:lines:perp}{समानांतर} पक्ष) $\times $ ऊंचाई।
        समझाएं कि यह वही नियम क्यों है, और प्रश्न की पुनः गणना करें
        8 उस तरह.
  \item ताश के पत्तों के एक साफ़ ढेर को तिरछे ढेर में झुकाएँ।
        स्पष्ट करें कि क्यों \omterm{def:g6:measure:volume}{आयतन} नहीं बदला है, और यह क्या
        में ऊंचाई के बारे में कहता है \omterm{def:g7:areas:prism}{चश्मे} FORMULA
        $ V = B \times h $ (\cref{thm:g7:areas:volume}): जो
        तिरछे ढेर के लिए ऊंचाई अवश्य लेनी चाहिए --- तिरछी
        लंबाई या ऊर्ध्वाधर? (तुलना करना
        \cref{exo:g7:areas:2}, एक आयाम ऊपर।)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Rings, pizzas, and the magician's
return.}
\begin{enumerate}[resume]
  \item का एक \omterm{def:g4:numbers:round}{गोल}ाकार तालाब \omterm{def:g3:shapes:shapes}{RADIUS} $3$ m, a के केंद्र पर बैठा है
        का \omterm{def:g4:numbers:round}{गोल}ाकार लॉन \omterm{def:g3:shapes:shapes}{RADIUS} $5$ एम। की गणना करें \omterm{def:g6:measure:area}{क्षेत्र} का
        घास की अंगूठी (सटीक के साथ $\pi $, फिर पूर्णांकित किया गया
        एम$^2$).
  \item खोजें \omterm{def:g3:shapes:shapes}{RADIUS} एकल डिस्क का जिसका \omterm{def:g6:measure:area}{क्षेत्र} बराबर
        वह अंगूठी \omterm{def:g6:measure:area}{क्षेत्र} बिल्कुल। (अब आपके पास तीन त्रिज्याएँ हैं
        एक प्रसिद्ध त्रिक का निर्माण करें, जिसकी कहानी इसमें बताई गई है
        \cref{ch:g8:pythagoras}.)
  \item कौन सा अधिक पिज़्ज़ा है: एक पिज़्ज़ा \omterm{def:g4:geometry:circle}{व्यास} $40$ सेमी, या
        के दो पिज्जा \omterm{def:g4:geometry:circle}{व्यास} $28$ सेमी? गणना करें और तुलना करें.
  \item को \emph{दोहरा} एक पिज़्ज़ा \omterm{def:g6:measure:area}{क्षेत्र}, मोटे तौर पर किस कारक से
        यह होना चाहिए \omterm{def:g3:shapes:shapes}{RADIUS} बढ़ना? उम्मीदवार की जांच करें $1.4$ (याद करें
        वह \omterm{def:g6:measure:area}{क्षेत्र}ों का पीछा करो \emph{वर्ग} स्केलिंग का
        कारक, \cref{pb:g6:propdata:1}), और कहो कहाँ
        सटीक गुणनखंड --- वह संख्या जिसका वर्ग है $2$ --- बनाता है
        इसका आधिकारिक प्रवेश द्वार
        (\cref{ex:g8:pythagoras:diagonal}).
  \item जादूगर लौटता है: वह काटता है $13 \times 13$ वर्ग
        संख्याओं के साथ $5$, $8$, $13$ वह भूमिकाएँ निभा रहे हैं
        $3$, $5$, $8$ पहले खेला, और पुन: संयोजन किया
        $8 \times 21$ आयत। दोनों की गणना करें \omterm{def:g6:measure:area}{क्षेत्र}ों: क्या
        इस बार गायब हो गया? संख्या $3, 5, 8, 13, 21$ हैं
        के अनुक्रम के लगातार पद
        \cref{pb:g7:fractions:1} --- का पैटर्न बताएं
        जादूगर के लाभ और हानि, और नैतिक: क्यों करते हैं \omterm{def:g6:measure:area}{क्षेत्र}ों
        कभी झूठ न बोलो?
\end{enumerate}
\end{problem}
