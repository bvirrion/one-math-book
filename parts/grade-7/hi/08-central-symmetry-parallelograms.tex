\chapter{केंद्रीय समरूपता और समांतर चतुर्भुज}\label{ch:g7:central}

एक आकृति को घुमाएँ \omterm{def:g3:division:half}{आधा} एक बिंदु के चारों ओर एक मोड़: यह है \emph{केंद्रीय
समरूपता}, चिंतन के बाद पुस्तक का दूसरा परिवर्तन
का \cref{ch:g6:symmetry}. इसका तारा चित्र है \omterm{def:g7:central:parallelogram}{चतुर्भुज} --- द
\omterm{def:g4:geometry:polygon}{चतुष्कोष} वह है \omterm{def:g7:central:def}{सममित} अपने स्वयं के क्रॉसिंग बिंदु के बारे में
विकर्ण.

\section{केंद्रीय समरूपता}

\begin{definition}[एक बिंदु के बारे में सममित]\label{def:g7:central:def}
 \emph{सममित}\index{सममित} एक बिंदु का $M $ ए के बारे में
बिंदु $ O $ मुद्दा यह है $ M'$ ऐसा है कि $ O $ है \emph{मध्य} का
$[MM']$. सममित आकृति वह है जो a के बाद मूल बन जाती है
\omterm{def:g3:division:half}{आधा} मोड़ ($180^\circ $) आस-पास $ O $.
\end{definition}

\end{omfigure}

\begin{method}[एक बिंदु की सममिति का निर्माण करना]\label{met:g7:central:construct}
का निर्माण करने के लिए \omterm{def:g7:central:def}{सममित} $ M'$ का $ M $ के बारे में $ O $:
\begin{enumerate}
  \item हदबंदी करना $(MO)$ और इसे आगे बढ़ाएँ $ O $;
  \item उपाय $ MO $ (शासक या कम्पास);
  \item जगह $ M'$ के साथ विस्तार पर $ OM' = OM $.
\end{enumerate}
ग्रिड पेपर पर: क्षैतिज और ऊर्ध्वाधर चरणों की गिनती करें $ M $ को
$ O $, और दोहराएँ \emph{वही} से कदम $ O $ तक पहुँचने $ M'$.
\end{method}

\begin{example}[एक ग्रिड पर]\label{ex:g7:central:grid}
अगर $ M $ है $3$ वर्ग बाएँ और $1$ से वर्गाकार $ O $, इसका \omterm{def:g7:central:def}{सममित}
$ M'$ है $3$ चौकों \emph{सही} और $1$ वर्ग \emph{नीचे} से $ O $:
केंद्रीय समरूपता दोनों दिशाओं को एक साथ उलट देता है (एक प्रतिबिंब
केवल एक को उलट देता है)।
\end{example}

\begin{proposition}[आधा मोड़ क्या सुरक्षित रखता है]\label{prop:g7:central:preserved}
केंद्रीय समरूपता लंबाई, कोण, को सुरक्षित रखता है \omterm{def:g6:measure:perimeter}{परिधि} और \omterm{def:g6:measure:area}{क्षेत्र}ों; द
एक \omterm{def:g6:lines:objects}{रेखा} की छवि एक है \emph{\omterm{def:g6:lines:perp}{समानांतर}} \omterm{def:g6:lines:objects}{रेखा}, ए की छवि \omterm{def:g6:lines:objects}{खंड} एक है
\omterm{def:g6:lines:perp}{समानांतर} \omterm{def:g6:lines:objects}{खंड} समान लंबाई का, एक वृत्त का प्रतिबिम्ब एक वृत्त होता है
उसी का \omterm{def:g3:shapes:shapes}{RADIUS} (पर केन्द्रित \omterm{def:g7:central:def}{सममित} केंद्र के)।
\end{proposition}
\admitted

\begin{remark}
प्रतिबिंबों के साथ तुलना करें (\cref{prop:g6:symmetry:preserved}): दोनों
लंबाई और कोण सुरक्षित रखें; लेकिन प्रतिबिंब आकृतियों को पलट देता है
(ए ``बी'' ``डी'' बन जाता है), जबकि \omterm{def:g3:division:half}{आधा} मोड़ उन्हें पढ़ने योग्य रखता है ---
उल्टा (a ``b'' ``q'' बन जाता है)। और एक प्रतिबिंब के विपरीत, ए
आधा-मोड़ प्रत्येक पंक्ति को एक पंक्ति में भेजता है \omterm{def:g6:lines:perp}{समानांतर} इसे.
\end{remark}

\begin{definition}[समरूपता का केंद्र]\label{def:g7:central:center}
एक बिंदु $ O $ एक है \emph{समरूपता का केंद्र}\index{समरूपता का केंद्र}
एक आकृति का जब \omterm{def:g3:division:half}{आधा} मोड़ घूम जाता है $ O $ बिल्कुल सही आंकड़ा भेजता है
खुद पर. उदाहरण के लिए, अक्षर S, N, Z का एक केंद्र होता है
समरूपता; एक वृत्त में एक (उसका केंद्र) होता है; त्रिभुज में कभी एक नहीं होता.
\end{definition}

\section{समानांतर चतुर्भुज}

\begin{definition}[चतुर्भुज]\label{def:g7:central:parallelogram}
ए \emph{चतुर्भुज}\index{चतुर्भुज} एक है \omterm{def:g4:geometry:polygon}{चतुष्कोष} किसका
विकर्ण अपने उभयनिष्ठ मध्यबिंदु पर प्रतिच्छेद करते हैं। वह क्रॉसिंग प्वाइंट तब है
ए \omterm{def:g7:central:center}{समरूपता का केंद्र} चित्र का: प्रत्येक शिखर अर्ध-मोड़ छवि है
विपरीत का शिखर.
\end{definition}

\begin{theorem}[समांतर चतुर्भुज के गुण]\label{thm:g7:central:properties}
में एक \omterm{def:g7:central:parallelogram}{चतुर्भुज} $ ABCD $ केंद्र के साथ $ O $:
\begin{enumerate}
  \item विपरीत भुजाएँ हैं \omterm{def:g6:lines:perp}{समानांतर}: $(AB) \parallel (CD)$ और
        $(BC) \parallel (AD)$;
  \item विपरीत भुजाओं की लंबाई समान होती है: $ AB = CD $ और
        $ BC = AD $;
  \item सम्मुख कोण बराबर होते हैं: $\widehat A = \widehat C $ और
        $\widehat B = \widehat D $.
\end{enumerate}
\end{theorem}

\begin{proof}
\omterm{def:g3:division:half}{आधा} चक्कर घूमना $ O $ भेजता है $ A $ को $ C $ और $ B $ को $ D $ (परिभाषा:
$ O $ दोनों विकर्णों का मध्यबिंदु है)। तो यह भेजता है \omterm{def:g6:lines:objects}{खंड}
$[AB]$ तक \omterm{def:g6:lines:objects}{खंड} $[CD]$; द्वारा
\cref{prop:g7:central:preserved} छवि है \omterm{def:g6:lines:perp}{समानांतर} को $[AB]$ और
समान लंबाई का: बिंदु 1 और 2। यह कोण भी भेजता है $ A $
कोण पर $ C $, और कोण संरक्षित हैं: बिंदु 3।
\end{proof}

\begin{theorem}[समांतर चतुर्भुज को पहचानना]\label{thm:g7:central:recognize}
ए \omterm{def:g4:geometry:polygon}{चतुष्कोष} (बिना किसी क्रॉसिंग साइड के) एक है \omterm{def:g7:central:parallelogram}{चतुर्भुज} जैसे ही
\emph{एक} निम्नलिखित में से धारण:
\begin{enumerate}
  \item इसके विकर्णों का मध्यबिंदु (परिभाषा) समान है;
  \item इसके विपरीत पक्ष हैं \omterm{def:g6:lines:perp}{समानांतर} दो बटा दो;
  \item दो विपरीत पक्ष हैं \omterm{def:g6:lines:perp}{समानांतर} \emph{और} समान लंबाई का.
\end{enumerate}
\end{theorem}
\admitted

\begin{example}\label{ex:g7:central:recognize}
$ ABCD $ है $(AB) \parallel (CD)$ और $ AB = CD = 5$ सेमी। कसौटी 3
लागू होता है: $ ABCD $ एक है \omterm{def:g7:central:parallelogram}{चतुर्भुज} --- तो बिना किसी और बात के
मापना तो हम भी जानते हैं $ AD = BC $, $(AD) \parallel (BC)$, और वह यह है
विकर्ण एक दूसरे को बीच में काटते हैं।
\end{example}

\begin{remark}[विशेष समांतर चतुर्भुज]
आयत, \omterm{def:g6:shapes:quadrilaterals}{विषमकोण} और का वर्ग \cref{ch:g6:shapes} बिलकुल हैं
द \omterm{def:g6:lines:perp}{समानांतर} \omterm{def:g7:central:parallelogram}{चतुर्भुज} एक अतिरिक्त गुण के साथ: समान विकर्ण
(आयत), \omterm{def:g6:lines:perp}{सीधा} विकर्ण (\omterm{def:g6:shapes:quadrilaterals}{विषमकोण}), दोनों (वर्ग)। द
कोण तर्क का वादा किया गया \cref{ex:g6:shapes:recognize} अब काम करता है:
एक में \omterm{def:g7:central:parallelogram}{चतुर्भुज}, क्रमागत कोण संपूरक होते हैं
(के साथ वैकल्पिक/संगत कोण समानताएं,
\cref{thm:g7:triangles:equalities}), तो एक \omterm{def:g3:shapes:rightangle}{समकोण} सभी को मजबूर करता है
चार.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g7:central:1}
ग्रिड पेपर पर रखें $ O $, तब $ M $ के दाईं ओर चार वर्ग $ O $ और एक
चौकोर होना. का निर्माण करें \omterm{def:g7:central:def}{सममित} $ M'$ का $ M $ के बारे में $ O $, और यह
\omterm{def:g7:central:def}{सममित} का $ M $ के आर - पार \emph{ऊर्ध्वाधर ग्रिड लाइन} के माध्यम से $ O $.
दोनों छवियों की तुलना करें.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:central:2}
एक \omterm{def:g3:shapes:shapes}{त्रिकोण} बनाएं $ ABC $ और एक बिंदु $ O $ इसके बाहर. इसकी छवि का निर्माण करें
$ A'B'C'$ आधे मोड़ से $ O $. आप इसके बारे में क्या कह सकते हैं
लंबाई $ A'B'$ और $ AB $? पंक्तियों के बारे में $(A'B')$ और $(AB)$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:central:3}
कैलकुलेटर पर कौन से अंक लिखे होते हैं ($0$ को $9$), लीजिये
\omterm{def:g7:central:center}{समरूपता का केंद्र}? H, A, N, S, T, Z, O में से कौन सा बड़ा अक्षर?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:central:4}
$ KLMN $ एक है \omterm{def:g7:central:parallelogram}{चतुर्भुज} केंद्र का $ O $ साथ $ KL = 7$ सेमी,
$ LM = 4$ सेमी और $\widehat K = 115^\circ $. कारण सहित दीजिए: $ MN $,
$ NK $, $\widehat M $, और का मध्यबिंदु $[LN]$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:central:5}
ए का निर्माण करें \omterm{def:g7:central:parallelogram}{चतुर्भुज} $ ABCD $ इसके विकर्णों से: $ AC = 8$ सेमी और
$ BD = 5$ सेमी, के कोण से उनके मध्यबिंदु पर प्रतिच्छेद करते हैं $60^\circ $
उन दोनों के बीच। (पहले विकर्ण खींचिए!)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:central:6}
में एक \omterm{def:g7:central:parallelogram}{चतुर्भुज}, एक कोण मापता है $115^\circ $. टिप्पणी का उपयोग करना
संपूरक क्रमागत कोणों पर, तीन अन्य कोण दीजिए।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:central:7}
कथानक $ A(1, 1)$, $ B(4, 2)$ और बात $ O(2.5, 2.5)$. की गणना करें
छवियों के निर्देशांक $ A'$ और $ B'$ आधे मोड़ से $ O $
(ग्रिड गिनती का उपयोग करें \cref{ex:g7:central:grid}).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:central:8}
$ ABCD $ एक है \omterm{def:g4:geometry:polygon}{चतुष्कोष} जिसमें $ AB = CD $. क्या इतना काफी है बनाने के लिए
यह ए \omterm{def:g7:central:parallelogram}{चतुर्भुज}? यदि नहीं, तो एक प्रति-उदाहरण रेखाचित्र बनाएं (a \omterm{def:g6:shapes:triangles}{समद्विबाहु}
ट्रैपेज़ॉइड मदद करता है), और बताएं कि परिकल्पना में क्या \omterm{def:g1:addition:def}{जोड़}ा जाना चाहिए।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:central:9}
होने देना $ ABC $ एक \omterm{def:g3:shapes:shapes}{त्रिकोण} हो और $ O $ का मध्यबिंदु $[BC]$. का निर्माण करें
\omterm{def:g7:central:def}{सममित} $ A'$ का $ A $ के बारे में $ O $. बताते हैं कि $ ABA'C $ एक है
\omterm{def:g7:central:parallelogram}{चतुर्भुज}. (कौन सा मानदंड \cref{thm:g7:central:recognize} है
यहाँ निःशुल्क?)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:central:10}
सही या गलत, एक कारण के साथ: ``ए \omterm{def:g4:geometry:polygon}{चतुष्कोष} जिनके विपरीत कोण हैं
दो बटा दो बराबर हैं ए \omterm{def:g7:central:parallelogram}{चतुर्भुज}''; ``ए \omterm{def:g7:central:parallelogram}{चतुर्भुज} साथ
\omterm{def:g6:lines:perp}{सीधा} विकर्ण एक है \omterm{def:g6:shapes:quadrilaterals}{विषमकोण}''; ``ए \omterm{def:g7:central:parallelogram}{चतुर्भुज} एक के साथ
\omterm{def:g3:shapes:rightangle}{समकोण} '' एक आयत है.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g7:central:11}
होने देना $ ABCD $ कोई भी हो \omterm{def:g4:geometry:polygon}{चतुष्कोष} और $ I $, $ J $, $ K $, $ L $ मध्यबिंदु
इसके पक्षों का $[AB]$, $[BC]$, $[CD]$, $[DA]$. अनेक उदाहरण बनाइये।
क्या करता है $ IJKL $ हमेशा लगता है? (प्रमाण, मध्यबिंदु के साथ
प्रमेय, अंदर आता है \cref{ch:g8:midpoints} --- और फिर से वैक्टर के साथ,
हाई स्कूल वॉल्यूम में।)
\end{exercise}

\section{समस्या: केक, माध्यिका और असंभव दूसरा
केंद्र}

\begin{problem}[{सप्ताहांत की समस्या --- काम पर आधा-अधूरा: उचित केक कट्स, माध्यिकाओं पर एक बंधन, और किसी भी आकृति के दो केंद्र क्यों नहीं हो सकते समरूपता}]
\label{pb:g7:central:1}
आधा-मोड़ अब तक के सबसे सरल परिवर्तनों जैसा दिखता है
यह केक को उचित रूप से आधे भागों में काटता है, माध्यिकाएँ मापता है
एक त्रिभुज, और शुद्ध गणित का एक छोटा सा मोती छिपा है: a
परिबद्ध आकृति में एक हो सकता है \omterm{def:g7:central:center}{समरूपता का केंद्र}, लेकिन \emph{कभी नहीं
दो}. प्रमाण उस खोज का उपयोग करता है जो आप भाग I: दो में करेंगे
एक के बाद एक किए गए आधे-मोड़ को एक स्लाइड में \omterm{def:g1:addition:def}{जोड़}ा जाता है।

\medskip
\noindent\textbf{Part I --- Half-turn gymnastics.}
\begin{enumerate}
  \item ग्रिड पेपर पर, मूल लें $ O(0, 0)$ केंद्र के रूप में.
        की ग्रिड गिनती का उपयोग करना \cref{ex:g7:central:grid},
        की छवियां दें $ M(3, 1)$, $ N(-2, 4)$ और
        $ P(0, -5)$ आधे मोड़ से $ O $. क्या है
        के लिए सामान्य नियम $(x, y)$?
  \item अब केंद्र है $ C(2, 1)$. की छवि की गणना करें
        $ M(5, 3)$, और सामान्य नुस्खा की जाँच करें: प्रत्येक समन्वय
        छवि का है \emph{दो बार केंद्र का \omterm{def:g1:counting:zero}{शून्य} से
        बिंदु का}.
  \item ताश के पत्तों को उल्टा पढ़ने के लिए डिज़ाइन किया गया है
        --- ए केंद्रीय समरूपता, इसलिए कोई भी खिलाड़ी कार्ड नहीं देखता है
        ``गलत तरीके से''। एक कार्ड पर स्पष्ट करें कि क्यों \emph{\omterm{def:g3:numbers:evenodd}{विषम}}
        पिप्स की संख्या, एक पिप बिल्कुल कार्ड के स्थान पर होनी चाहिए
        केंद्र.
  \item एक पंक्ति में दो आधे-मोड़: केंद्र लें $ O_1(2, 0)$
        और $ O_2(5, 0)$. बिंदु भेजो $ A(1, 1)$ के माध्यम से
        आधा-मोड़ घूमना $ O_1$, फिर छवि को इसके माध्यम से भेजें
        आधा-मोड़ घूमना $ O_2$. प्रारंभ और समाप्ति की तुलना करें. दोहराएँ
        के साथ $ B(0, 3)$. एकल, सरल परिवर्तन ने क्या किया
        दो आधे-मोड़ की राशि क्या है?
  \item प्रश्न 4 की स्लाइड को दूरी से मापें
        $ O_1 O_2$, और खोज बताएं। (तुलना करें
        दो \omterm{def:g6:lines:perp}{समानांतर} के दर्पण \cref{pb:g6:symmetry:1}: वही
        घटना, नए अभिनेता।)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Cakes and medians.}
\begin{enumerate}[resume]
  \item होने देना $ O $ का केंद्र हो \omterm{def:g7:central:parallelogram}{चतुर्भुज}. स्पष्ट करें क्यों
        किसी भी लाइन के माध्यम से $ O $ सीमा दो बिंदुओं पर मिलती है
        वह हैं \omterm{def:g7:central:def}{सममित} के बारे में $ O $, और \omterm{def:g6:lines:objects}{रेखा} क्यों काटती है
        \omterm{def:g7:central:parallelogram}{चतुर्भुज} दो टुकड़ों में जो बिल्कुल आधा-मोड़ें
        एक दूसरे की प्रतियाँ --- इसलिए बराबर की \omterm{def:g6:measure:area}{क्षेत्र}ों.
  \item बेकर का प्रमेय: एक आयताकार केक साझा किया जाता है
        बिल्कुल निष्पक्ष रूप से \emph{कोई} इसके माध्यम से सीधे काटें
        केंद्र. बेहतर: एक आयताकार केक में एक आयताकार केक होता है
        छेद (कोई भी आकार, कोई भी स्थिति, \omterm{def:g3:numbers:evenodd}{यहां तक की} झुका हुआ)। का वर्णन करें
        एक \omterm{def:g6:lines:perp}{सीधा} कट जो समान रूप से दो भाग देता है
        केक --- और इसे प्रश्न 6 से उचित ठहराएँ।
  \item तीन \omterm{def:g5:solids:def}{कोने} एक का \omterm{def:g7:central:parallelogram}{चतुर्भुज} $ ABCD $ हैं $ A(0, 0)$,
        $ B(6, 1)$ और $ C(8, 5)$. के निर्देशांक की गणना करें $ D $
        और केंद्र का $ O $.
  \item \Cref{exo:g7:central:9} निर्मित, एक \omterm{def:g3:shapes:shapes}{त्रिकोण} से $ ABC $
        और मध्यबिंदु $ O $ का $[BC]$, द \omterm{def:g7:central:parallelogram}{चतुर्भुज}
        $ ABA'C $ कहाँ $ A'$ है \omterm{def:g7:central:def}{सममित} का $ A $ के बारे में $ O $.
        त्रिभुज असमानता के साथ इसका प्रयोग करें
        (\cref{thm:g7:triangles:inequality}) \omterm{def:g3:shapes:shapes}{त्रिकोण} में
        $ ABA'$, सिद्ध करने के लिए \emph{मंझला बंधा हुआ}: मध्यस्थ
        $[AO]$ संतुष्ट
        \[
        AO < \frac{AB + AC}{2} .
        \]
        (ध्यान दें कि $ AA' = 2\,AO $ और $ BA' = AC $.)
  \item के साथ एक \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ AB = 5$ सेमी और $ AC = 7$ सेमी, बीच में
        माध्यिका की लंबाई किन दो मानों से होनी चाहिए? $ A $
        झूठ? (त्रिभुज असमानता की दोनों दिशाओं का उपयोग करें
        $ ABA'$.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- How many centers can a figure
have?}
\begin{enumerate}[resume]
  \item प्रत्येक आकृति के लिए कहें कि क्या इसमें कोई है का केंद्र
        समरूपता, और कहां: ए \omterm{def:g6:lines:objects}{खंड}; एक पूरी पंक्ति; एक
        समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण}; एक वर्ग; एक चक्र; अक्षर एस
        और ज़ेड.
  \item के लिए अपना दावा साबित करें समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण}: ए
        अर्ध-मोड़ में त्रिभुज को स्वयं भेजने से \omterm{def:g1:addition:def}{जोड़}ी बन जाएगी
        तीन \omterm{def:g5:solids:def}{कोने} --- लेकिन तीन है \omterm{def:g3:numbers:evenodd}{विषम}, इतना कुछ शिखर
        खुद को भेजा जाएगा. वो क्या मजबूर करेगा
        शिखर होना, और यह असंभव क्यों है?
  \item प्रश्न 12 को सामान्यीकृत करें: क्यों समझाएँ \emph{नहीं} \omterm{def:g4:geometry:polygon}{बहुभुज}
        एक साथ \omterm{def:g3:numbers:evenodd}{विषम} संख्या का \omterm{def:g5:solids:def}{कोने} एक का केंद्र
        समरूपता.
  \item मोती: मान लीजिए कि एक आकृति थी \emph{दो} विशिष्ट
        समरूपता के केंद्र $ O_1$ और $ O_2$. दो का पालन करें
        एक के बाद एक \omterm{def:g3:division:half}{आधा} मोड़ें और खोज का उपयोग करें
        प्रश्न 5: आकृति को सटीक रूप से किस गति से भेजना चाहिए
        अपने आप पर? बताएं कि एक आकृति एक शीट पर क्यों फिट बैठती है
        काग़ज़ उससे बच नहीं सकता, और निष्कर्ष निकाल सकता हूँ।
  \item प्रश्न 14 का निष्कर्ष एक खामी छोड़ता है:
        \emph{असीम} आंकड़े. अनंत फ्रिज़ पर सत्यापित करें
        --- मान लीजिए, पैरों के निशानों की एक अंतहीन पट्टी, बाएँ, दाएँ,
        बाएँ, दाएँ\,\dots\ --- कि इसमें (कम से कम) दो हैं
        समरूपता के विभिन्न केंद्र, और पहचानें
        अनुवाद उस प्रश्न 14 की भविष्यवाणी की गई है।
\end{enumerate}
\end{problem}
