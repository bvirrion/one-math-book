\chapter{शाब्दिक अभिव्यक्तियाँ}\label{ch:g7:literal}

गणना में एक अक्षर एक ऐसी संख्या को दर्शाता है जिसे हम अभी तक नहीं जानते हैं, या
के लिए \emph{सभी} एक साथ संख्याएँ: लिखना $P = 4c $ देता है \omterm{def:g6:measure:perimeter}{परिधि}
प्रत्येक वर्ग का एक सूत्र में। यह अध्याय लिखना सिखाता है
परंपराएं, मूल्यों को कैसे प्रतिस्थापित करें, और सरल को सरल कैसे बनाएं
भाव---बीजगणित की शुरुआत, जारी रही
\cref{ch:g8:equations}.

\section{अक्षरों से लिखना}

\begin{definition}[शाब्दिक अभिव्यक्ति]\label{def:g7:literal:def}
ए \emph{शाब्दिक अभिव्यक्ति}\index{शाब्दिक अभिव्यक्ति} एक गणना है
जिसमें एक या अधिक अक्षर हों, प्रत्येक अक्षर एक संख्या को दर्शाता हो। लेखन
सम्मेलन:
\begin{itemize}
  \item संकेत $\times $ किसी अक्षर या कोष्ठक से पहले डाला जाता है:
        $3 \times a = 3a $, \ $ a \times b = ab $, \
        $2 \times (x + 5) = 2(x + 5)$;
  \item लेकिन यह दो संख्याओं के बीच रहता है: $3 \times 5$ नहीं हो सकता
        लिखा हुआ $35$!
  \item $ a \times a $ लिखा है $ a^2$ (``$ a $ चुकता''),
        $ a \times a \times a = a^3$;
  \item $1a $ लिखा है $ a $, और $0a $ है $0$.
\end{itemize}
\end{definition}

\begin{example}\label{ex:g7:literal:write}
लेखन को सरल बनाएं:
\[
5 \times x + 3 \times y = 5x + 3y,
\qquad
a \times 7 = 7a \ \text{(number first)},
\qquad
x \times x \times 4 = 4x^2 .
\]
\end{example}

\begin{example}[सूत्र शाब्दिक अभिव्यक्ति हैं]\label{ex:g7:literal:formulas}
लंबाई के एक आयत के लिए $ L $ और चौड़ाई $ w $:
\omterm{def:g6:measure:perimeter}{परिधि} $ P = 2(L + w)$, \omterm{def:g6:measure:area}{क्षेत्र} $ A = Lw $. एक सूत्र इसका सारांश प्रस्तुत करता है
एक साथ अनंत रूप से अनेक आयतें --- यही संपूर्ण बिंदु है
पत्र.
\end{example}

\section{स्थानापन्न}

\begin{method}[किसी अभिव्यक्ति का मूल्यांकन करना]\label{met:g7:literal:substitute}
किसी अभिव्यक्ति के अक्षरों के दिए गए मानों का मूल्यांकन करने के लिए:
\begin{enumerate}
  \item छुपे हुए भाव को डालते हुए अभिव्यक्ति को फिर से लिखें $\times $ वापस संकेत:
        $3a + a^2$ है $3 \times a + a \times a $;
  \item प्रत्येक अक्षर को उसके मान से बदलें, उसके चारों ओर कोष्ठक रखें
        यदि आवश्यक हो;
  \item की प्राथमिकताओं का सम्मान करते हुए गणना करें
        \cref{ch:g7:priorities}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:literal:substitute}
मूल्यांकन करना $ E = 3x + x^2$ के लिए $ x = 5$:
\[
E = 3 \times 5 + 5 \times 5 = 15 + 25 = 40 .
\]
मूल्यांकन करना $ F = 2(a + 3b)$ के लिए $ a = 4$ और $ b = 0.5$:
\[
F = 2 \times (4 + 3 \times 0.5) = 2 \times (4 + 1.5) = 2 \times 5.5
= 11 .
\]
\end{example}

\section{भावों को कम करना}

\begin{proposition}[समान पद एकत्रित करना]\label{prop:g7:literal:reduce}
समान अक्षर भाग वाले शब्दों को जोड़कर एकत्रित किया जा सकता है
भागों की संख्या (यह वितरणशीलता है,
\cref{thm:g7:priorities:distributivity}, पीछे की ओर पढ़ें):
\[
5x + 3x = (5 + 3)x = 8x,
\qquad
7a - 2a = 5a .
\]
विभिन्न अक्षर भागों वाले शब्द (जैसे $3x $ और $5y $, या $ x $ और
$ x^2$) एकत्र नहीं किया जा सकता.
\end{proposition}

\begin{example}\label{ex:g7:literal:reduce}
कम करना $ E = 4x + 7 + 3x + 2$, शर्तों को क्रमबद्ध करना:
\[
E = (4x + 3x) + (7 + 2) = 7x + 9 .
\]
सावधान: $7x + 9$ है \emph{नहीं} $16x $! के साथ परीक्षण करें $ x = 1$:
$4 + 7 + 3 + 2 = 16$ और $7 \times 1 + 9 = 16$ सहमत हूँ, लेकिन $16x $ होगा
भी हो $16$ --- के साथ परीक्षण करें $ x = 2$: मूल देता है
$8 + 7 + 6 + 2 = 23$, और $7 \times 2 + 9 = 23$, जबकि
$16 \times 2 = 32$. कमी $7x + 9$ सही है, $16x $ ग़लत है.
\end{example}

\begin{method}[समानता का परीक्षण]\label{met:g7:literal:test}
यह जाँचने के लिए कि क्या प्रत्येक मान के लिए दो अभिव्यक्तियाँ समान हो सकती हैं:
\begin{enumerate}
  \item दोनों में समान मान को कम से कम दो बार प्रतिस्थापित करें (उदा.\
        $ x = 2$ और $ x = 10$; टालना $0$ और $1$, वे बहुत कुछ छिपाते हैं
        मतभेद);
  \item यदि परिणाम एक बार भी भिन्न हों, तो व्यंजक हैं
        \emph{नहीं} बराबर --- एक प्रतिउदाहरण एक सामान्य दावे को ख़त्म कर देता है;
  \item यदि परिणाम सहमत हों, तो समानता ही है \emph{प्रशंसनीय}: ए
        प्रमाण के लिए बीजगणित (विस्तार करना, घटाना) की आवश्यकता होती है।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:literal:test}
है $2(x + 3)$ के बराबर $2x + 6$ हरएक के लिए $ x $? विस्तार यह सिद्ध करता है:
$2(x+3) = 2x + 2 \times 3 = 2x + 6$. है $(x + 1)^2$ के बराबर
$ x^2 + 1$? परीक्षा $ x = 2$: $(2+1)^2 = 9$ लेकिन $2^2 + 1 = 5$. नहीं ---
और एक प्रतिउदाहरण ``नहीं'' का पूर्ण प्रमाण है।
\end{example}

\section{अभिव्यक्तियाँ उत्पन्न करना}

\begin{example}[स्थिति से अभिव्यक्ति तक]\label{ex:g7:literal:produce}
एक टैक्सी शुल्क लेती है $4$ यूरो प्लस $2$ यूरो प्रति किलो\omterm{def:g4:fractions:def}{मीटर}. की यात्रा के लिए
$ k $ किलो\omterm{def:g4:fractions:def}{मीटर} की कीमत है
\[
p = 4 + 2k \quad\text{euros}.
\]
के लिए $ k = 7$: $ p = 4 + 14 = 18$ यूरो. इजहार \emph{है} द
सामान्य उत्तर; प्रतिस्थापन से प्रत्येक वि\omterm{def:g3:division:remainder}{शेष} का निर्माण होता है।
\end{example}

\begin{example}[नंबर ट्रिक्स]\label{ex:g7:literal:trick}
``एक संख्या के बारे में सोचें, उसे दोगुना करें, \omterm{def:g1:addition:def}{जोड़}ें $10$, से विभाजित करें $2$, अपना घटाओ
नंबर.'' नंबर पर कॉल करें $ n $ और चरणों का पालन करें:
\[
n \ \to\ 2n \ \to\ 2n + 10 \ \to\ \frac{2n + 10}{2} = n + 5 \ \to\
n + 5 - n = 5 .
\]
सबको मिलता है $5$, जो कुछ भी $ n $ था --- पत्र समझाते हैं
जादू.
\end{example}

\section{समस्या: माचिस की तीली का दैवज्ञ}

\begin{problem}[{सप्ताहांत समस्या---बढ़ते पैटर्न और उनके $ n $-वाँ पद: एक पैटर्न के लिए तीन सूत्र, और प्रसिद्ध पैटर्न जो छठे चरण पर टूटता है}]
\label{pb:g7:literal:1}
माचिस की तीलियाँ वर्गों की एक पंक्ति में रखें: एक वर्ग, फिर दो, फिर
तीन\,\dots\ सौवें अंक के लिए कितने मिलानों की आवश्यकता होगी?
उन्हें एक-एक करके गिनने में पूरा सप्ताहांत लग जाएगा; ए \emph{सूत्र
में $ n $} एक ही बार में प्रत्येक आंकड़े के लिए तुरंत उत्तर --- अर्थात
वास्तव में पत्र किस लिए हैं
(\cref{ex:g7:literal:formulas}). लेकिन सावधान रहें: यह समस्या समाप्त हो जाती है
एक प्रसिद्ध पैटर्न के साथ जो पाँच चरणों और उसके बाद तक निष्पक्ष रहता है
उन सभी को धोखा देता है जिन्होंने उस पर भरोसा किया।

\medskip
\noindent\textbf{Part I --- A row of squares.}
आकृति $1$ एक माचिस की तीली का वर्ग है ($4$ माचिस); प्रत्येक निम्नलिखित
आकृति पिछले वर्ग के साथ एक भुजा साझा करते हुए एक और वर्ग जोड़ती है।
\begin{enumerate}
  \item आकृतियाँ बनाएं $1$ को $4$ और उनके मिलान गिनें।
  \item स्पष्ट करें कि प्रत्येक नए वर्ग की कीमत ठीक-ठीक क्यों है $3$ अतिरिक्त
        मिलान करें, और आंकड़े की गिनती की भविष्यवाणी करने के लिए इसका उपयोग करें
        $10$ बिना किसी फॉर्मूले के.
  \item सूत्र का औचित्य सिद्ध करें: चित्र $ n $ आवश्यकताओं $3n + 1$ मेल खाता है.
        (अकेला कहाँ है ``$+1$'' से आये?) इसकी जाँच करें
        प्रश्न 1 के विरुद्ध.
  \item आमिर अलग तर्क देते हैं: ``$4$ पहले के लिए मेल खाता है
        वर्ग, फिर $3$ प्रत्येक के लिए $ n - 1$ अन्य'', दे रहे हैं
        $4 + 3(n - 1)$. उसकी अभिव्यक्ति को बढ़ाएँ और घटाएँ
        (\cref{prop:g7:literal:reduce}): क्या वह इससे सहमत है
        प्रश्न 3?
  \item लीना तीसरा तरीका गिनाती है: ``$2n $ क्षैतिज मिलान, और
        $ n + 1$ ऊर्ध्वाधर वाले।'' उसका सूत्र लिखें, इसे छोटा करें,
        और निष्कर्ष निकालो. फिर प्रारंभिक प्रश्न का उत्तर दें: कितने
        आकृति के लिए मिलान $100$?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Triangles, tables and going
backwards.}
\begin{enumerate}[resume]
  \item माचिस की तीली के \omterm{def:g3:shapes:shapes}{त्रिकोण}ों की एक पंक्ति (प्रत्येक नया \omterm{def:g3:shapes:shapes}{त्रिकोण} झुकता है
        पिछला वाला): के लिए मिलान गिनें $1$, $2$, $3$
        \omterm{def:g3:shapes:shapes}{त्रिकोण}, फिर सूत्र का औचित्य सिद्ध करें $2n + 1$ के लिए $ n $
        \omterm{def:g3:shapes:shapes}{त्रिकोण}.
  \item पीछे की ओर जाना: बिल्कुल के साथ $49$ मिलान, कितने
        पंक्ति में त्रिभुज हो सकते हैं? (सूत्र को चरण दर चरण पूर्ववत करें
        कदम.)
  \item वर्गाकार भोज मेजों पर प्रति खाली तरफ एक अतिथि के बैठने की जगह है। कब
        $ n $ तालिकाओं को एक लंबी पंक्ति में धकेल दिया गया है, समझाएँ कि क्यों
        बैठने का फार्मूला है $2n + 2$, और इसकी जांच करें $1$, $2$,
        $3$ टेबल.
  \item एक पार्टी उम्मीद करती है $30$ मेहमान. एक पंक्ति में कितनी टेबल होती है
        जरुरत? इसके बजाय कैटरर उन्हीं तालिकाओं को विभाजित कर देता है
        \emph{दो} सात की पंक्तियाँ: अब कितने मेहमान आएँगे?
        लाभ को सूत्र सहित समझाइये।
  \item बिंदु पैटर्न: आकृति $ n $ बिंदुओं का L है, $ n $ लम्बे बिंदु
        और $ n $ बिंदु चौड़े (\omterm{def:g5:solids:def}{कोने} को एक बार गिना गया)। गिनें
        आकृतियों के बिंदु $1$, $2$, $3$, सूत्र और नाम ढूंढें
        यह जो संख्याएँ उत्पन्न करता है। (एक साथ खड़ी, ये एलएस टाइल
        एक वर्ग --- एक कहानी जारी रही
        \cref{pb:g8:equations:1}.)
\end{enumerate}

\noindent\textbf{Part III --- Trust, test, prove.}
\begin{enumerate}[resume]
  \item एक सहपाठी का दावा है आंकड़ा $ n $ भाग I की आवश्यकता है
        $3(n + 2) - 5$ मेल खाता है. और का विस्तार करके दावे का निपटान करें
        कम करना --- किसी परीक्षण की आवश्यकता नहीं।
  \item ``वर्ग करने से संख्याएँ बड़ी हो जाती हैं, इसलिए $ n^2 \geq n $ के लिए
        हर संख्या $ n $.'' दावे का परीक्षण करें $ n = 2$, $3$,
        $10$; फिर वह संख्या खोजें जिसके लिए यह विफल हो। क्या किया
        तीन सफल परीक्षण साबित करते हैं, और अकेला क्या साबित करता है
        विफलता साबित (\cref{met:g7:literal:test})?
  \item एक वृत्त पर बिंदु अंकित करें और \omterm{def:g6:lines:objects}{रेखा}ंकन करें \emph{प्रत्येक} \omterm{def:g6:lines:objects}{खंड}
        उनके बीच, फिर वृत्त के अंदर के \omterm{def:g6:measure:area}{क्षेत्र}ों को गिनें।
        के लिए यह करें $2$, $3$, $4$ और $5$ अंक (अंक लगाएं
        असमान रूप से, ताकि कोई तीन न हो \omterm{def:g6:lines:objects}{खंड}ों एक आंतरिक भाग पर क्रॉस करें
        बिंदु). गिनती रिकार्ड करें. पैटर्न किस सूत्र से बनता है
        कानाफूसी?
  \item अब $6$ अंक, बड़े और असमान रूप से, गिनती के साथ खींचे गए
        बहुत अच्छी देखभाल (कई छोटे \omterm{def:g6:measure:area}{क्षेत्र} हैं)। कितने
        क्या आपको \omterm{def:g6:measure:area}{क्षेत्र} मिलते हैं --- और उनका क्या हुआ

  \item नैतिक, दो वाक्यों में: क्या आवश्यक होता
        को \emph{सिद्ध करना} दोहरीकरण अनुमान, और क्यों थे
        भाग I और II के माचिस के फार्मूले कभी भी ऐसे नहीं होते
        ख़तरा? सिद्ध भविष्यवाणी के साथ जश्न मनाएं: कितने
        की पंक्ति से मेल खाता है $2\,026$ वर्ग?
