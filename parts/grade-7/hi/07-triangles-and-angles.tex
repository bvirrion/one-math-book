\chapter{त्रिकोण और कोण}\label{ch:g7:triangles}

क्या आप भुजाओं वाला एक त्रिभुज बना सकते हैं? $10$, $3$ और $2$? कोणों के साथ
$80^\circ $, $70^\circ $ और $50^\circ $? यह अध्याय दोनों का उत्तर देता है
समतल ज्यामिति के दो सबसे उपयोगी तथ्यों वाले प्रश्न:
\omterm{def:g3:shapes:shapes}{त्रिकोण} असमानता और $180^\circ $ कोण \omterm{def:g1:addition:def}{जोड़} --- प्लस कोण
शब्दावली जो क्रॉसिंग और के साथ आती है \omterm{def:g6:lines:perp}{समानांतर} रेखाएँ.

\section{क्रॉसिंग रेखाओं के चारों ओर के कोण}

\begin{definition}[कोण युग्म]\label{def:g7:triangles:pairs}
\begin{itemize}
  \item दो कोण हैं \emph{पूरक} जब उनके उपाय जुड़ जाते हैं
        को $90^\circ $, \emph{पूरक} जब वे जुड़ते हैं
        $180^\circ $.
  \item जब दो रेखाएं एक-दूसरे को काटती हैं, तो एक-दूसरे के आर-पार बनने वाले कोण एक-दूसरे के सम्मुख होते हैं
        क्रॉसिंग प्वाइंट हैं \emph{लंबवत
        विपरीत}\index{लंबवत
        विपरीत}.
  \item जब एक \omterm{def:g6:lines:objects}{रेखा} दो अन्य रेखाओं को काटती है, तो कोण स्थित होते हैं
        प्रत्येक क्रॉसिंग पर एक ही स्थिति है
        \emph{संगत कोण}; दो रेखाओं के बीच का कोण
        क्रॉसिंग लाइन की विपरीत भुजाएँ हैं \emph{वैकल्पिक
        कोण}\index{वैकल्पिक
        कोण}.
\end{itemize}
\end{definition}

\begin{theorem}[कोण समानताएँ]\label{thm:g7:triangles:equalities}
\begin{enumerate}
  \item लंबवत् विपरीत कोण बराबर हैं.
  \item यदि दो पंक्तियाँ हैं \omterm{def:g6:lines:perp}{समानांतर}, संगत कोण बराबर होते हैं, और
        \omterm{def:g7:triangles:pairs}{वैकल्पिक कोण} बराबर हैं. इसके विपरीत, समान संगत
        (या वैकल्पिक) कोण रेखाओं को होने के लिए बाध्य करते हैं \omterm{def:g6:lines:perp}{समानांतर}.
\end{enumerate}
\end{theorem}

\begin{proof}[1 का प्रमाण]
दो कोण $\widehat{1}$ और $\widehat{2}$ एक पंक्ति के एक तरफ
एक \omterm{def:g6:lines:perp}{सीधा} कोण बनाएं: $\widehat 1 + \widehat 2 = 180^\circ $.
लंबवत् विपरीत कोण $\widehat 3$ का $\widehat 1$ संतुष्ट भी करता है
$\widehat 2 + \widehat 3 = 180^\circ $ (वही कारण). इसलिए
$\widehat 1 = 180^\circ - \widehat 2 = \widehat 3$. प्वाइंट 2 है
इस स्तर पर स्वीकार किया गया।
\end{proof}

\section{एक त्रिभुज के कोणों का योग}

\begin{theorem}[कोण योग]\label{thm:g7:triangles:sum}
प्रत्येक त्रिभुज में तीनों कोणों का योग होता है $180^\circ $.
\end{theorem}

\begin{proof}
होने देना $ ABC $ एक \omterm{def:g3:shapes:shapes}{त्रिकोण} हो. के माध्यम से \omterm{def:g6:lines:objects}{रेखा} खींचें $ A $ \omterm{def:g6:lines:perp}{समानांतर} को
$(BC)$. पर शिखर $ A $, तीन कोण इसके अनुदिश \omterm{def:g6:lines:objects}{रेखा} बनाते हैं \omterm{def:g6:lines:perp}{समानांतर}
और एक \omterm{def:g6:lines:perp}{सीधा} कोण बनाओ, $180^\circ $: बीच वाला है
$\widehat{A}$, और दो बाहरी हैं \omterm{def:g7:triangles:pairs}{वैकल्पिक कोण} साथ
$\widehat B $ और $\widehat C $ (\omterm{def:g6:lines:perp}{समानांतर} रेखाएँ!), इसलिए वे बराबर हैं
$\widehat B $ और $\widehat C $. इस तरह
$\widehat B + \widehat A + \widehat C = 180^\circ $.
\end{proof}

\begin{example}\label{ex:g7:triangles:sum}
एक त्रिभुज के दो कोण मापे जाते हैं $67^\circ $ और $58^\circ $. तीसरा
उपाय
\[
180 - (67 + 58) = 180 - 125 = 55^\circ .
\]
याद रखने योग्य वि\omterm{def:g3:division:remainder}{शेष} मामले: प्रत्येक कोण समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण}
है $180 \div 3 = 60^\circ $; में एक सही \omterm{def:g3:shapes:shapes}{त्रिकोण} दो न्यूनकोण
पूरक हैं (वे जोड़ते हैं $180 - 90 = 90^\circ $); आधार
एक के कोण \omterm{def:g6:shapes:triangles}{समद्विबाहु} त्रिभुज समान हैं, इसलिए प्रत्येक है
$\frac{180 - \text{apex}}{2}$.
\end{example}

\begin{example}[दो समकोण वाला कोई त्रिभुज नहीं]\label{ex:g7:triangles:tworight}
के दो कोणों वाला एक त्रिभुज $90^\circ $ पहले से ही उपयोग होगा
$180^\circ $, जा रहा हूँ $0^\circ $ तीसरे के लिए --- असंभव. कोना
\omterm{def:g1:addition:def}{जोड़} एक साथ कई त्रिभुजों का निषेध करता है।
\end{example}

\section{त्रिकोण असमानता}

\begin{theorem}[त्रिभुज असमानता]\label{thm:g7:triangles:inequality}
प्रत्येक त्रिभुज में, प्रत्येक भुजा इससे छोटी होती है \omterm{def:g1:addition:def}{जोड़} दोनों में से
अन्य: भुजाओं वाले त्रिभुज के लिए $ a $, $ b $, $ c $,
\[
a < b + c .
\]
इसके विपरीत, तीन लंबाई के साथ \emph{सबसे बड़ा} से छोटा \omterm{def:g1:addition:def}{जोड़}
अन्य दो में से हमेशा एक त्रिभुज बनाया जा सकता है। यदि सबसे बड़ा
\emph{के बराबर होती है}  \omterm{def:g1:addition:def}{जोड़} अन्य में, ``\omterm{def:g3:shapes:shapes}{त्रिकोण}'' समतल है:
तीन \omterm{def:g5:solids:def}{कोने} संरेखित हैं.
\end{theorem}
\admitted

\begin{example}\label{ex:g7:triangles:inequality}
क्या पक्ष हो सकते हैं? $10$, $3$ और $2$? परीक्षण सबसे बड़ा: है $10 < 3 + 2 = 5$?
नहीं --- ऐसा कोई त्रिभुज मौजूद नहीं है। कम्पास दिखाता है क्यों: त्रिज्या के चाप
$3$ और $2$ ए के दोनों सिरों से खींचा गया \omterm{def:g6:lines:objects}{खंड} लम्बाई का $10$ कभी नहीं मिलना।

पक्षों $7$, $5$ और $4$? सबसे बड़ा: $7 < 5 + 4 = 9$: हाँ, \omterm{def:g3:shapes:shapes}{त्रिकोण}
मौजूद है (सबसे बड़े पक्ष की जाँच करना पर्याप्त है)।
\end{example}

\begin{method}[त्रिभुज का अस्तित्व एवं निर्माण]\label{met:g7:triangles:construct}
तीन लंबाई दी गई:
\begin{enumerate}
  \item सबसे बड़े की तुलना के साथ करें \omterm{def:g1:addition:def}{जोड़} दो अन्य में से; यदि यह है
        कड़ाई से छोटा नहीं, रुकें: कोई \omterm{def:g3:shapes:shapes}{त्रिकोण} नहीं;
  \item अन्यथा जैसा निर्माण करें \cref{met:g6:shapes:construct}
        (आधार \omterm{def:g6:lines:objects}{खंड}, दो कम्पास चाप);
  \item किसी भी निर्माण के बाद कोणों से मानसिक रूप से जांच लें कि
        दिए गए कोणों का योग इससे कम होता है $180^\circ $ --- तीसरे के साथ
        बिल्कुल कोण बनाना $180^\circ $.
\end{enumerate}
\end{method}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g7:triangles:1}
(को) पूरक दीजिए $90^\circ $) और पूरक (से.)
$180^\circ $) का: $30^\circ $; \ $45^\circ $; \ $72^\circ $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:triangles:2}
दो रेखाएँ प्रतिच्छेद करती हैं; चार कोण मापों में से एक $35^\circ $. दे दो
तीन अन्य के माप, प्रत्येक के लिए एक कारण सहित।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:triangles:3}
उस त्रिभुज के तीसरे कोण की गणना करें जिसके पहले दो कोण मापते हैं:
(ए) $40^\circ $ और $60^\circ $; \
(बी) $90^\circ $ और $28^\circ $; \
(सी) $75^\circ $ और $75^\circ $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:triangles:4}
एक \omterm{def:g6:shapes:triangles}{समद्विबाहु} त्रिभुज इसका शीर्ष कोण बराबर है $40^\circ $. गणना करें
इसके दो आधार कोण हैं। एक और \omterm{def:g6:shapes:triangles}{समद्विबाहु} त्रिभुज का आधार कोण है
$70^\circ $: इसके शीर्ष कोण की गणना करें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:triangles:5}
इनमें से कौन सा त्रिक त्रिभुज की भुजाएँ हो सकता है? द्वारा औचित्य सिद्ध करें
त्रिभुज असमानता (सबसे बड़ी भुजा!):
\[
(6,\ 8,\ 12); \qquad
(5,\ 5,\ 11); \qquad
(4,\ 9,\ 13); \qquad
(7,\ 7,\ 7).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:triangles:6}
एक त्रिभुज में कोण होते हैं $ x $, $2x $ और $3x $. खोजो $ x $ और तीन
कोण. यह किस प्रकार का त्रिभुज है?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:triangles:7}
एक त्रिभुज का निर्माण करें $ ABC $ साथ $ BC = 6$ सेमी,
$\widehat{ABC} = 50^\circ $ और $\widehat{ACB} = 60^\circ $ (चाँचक
पर $ B $ और कम से $ C $). निर्माण से पहले, माप की भविष्यवाणी करें
$\widehat{BAC}$, फिर अपने फिगर की जांच करें।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:triangles:8}
दो \omterm{def:g6:lines:perp}{समानांतर} रेखाएँ का कोण बनाते हुए तीसरी \omterm{def:g6:lines:objects}{रेखा} से काटे जाते हैं
$54^\circ $ पहले क्रॉसिंग पर (क्रॉसिंग लाइन और एक के बीच)।
\omterm{def:g6:lines:perp}{समानांतर}). स्थिति का चित्र बनाइये और आठों के माप बताइये
कोण बने.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:triangles:9}
एक त्रिभुज की दो भुजाएँ मापी जाती हैं $8$ सेमी और $3$ सेमी। किन दो के बीच
मूल्यों को तीसरा पक्ष झूठ बोलना चाहिए? एक पूर्णांक लंबाई दें जो काम करती हो
और एक जो नहीं करता.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:triangles:10}
एक \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ ABC $, $\widehat A = 2\widehat B $ और
$\widehat C = 90^\circ $. गणना करना $\widehat A $ और $\widehat B $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g7:triangles:11}
किसी के तीन कोण \omterm{def:g4:geometry:polygon}{चतुष्कोष} $ ABCD $ द्वारा अध्ययन किया जा सकता है
इसे एक विकर्ण के अनुदिश दो त्रिभुजों में काटना। इसे सिद्ध करने के लिए इसका उपयोग करें
प्रत्येक के चार कोण \omterm{def:g4:geometry:polygon}{चतुष्कोष} तक \omterm{def:g1:addition:def}{जोड़}ा गया $360^\circ $, और
कोण निकालें \omterm{def:g1:addition:def}{जोड़} एक का \omterm{def:g4:geometry:polygon}{पंचकोण} (पाँच भुजाएँ)।
\end{exercise}

\section{समस्या: ब्लॉक के चारों ओर घूमना}

\begin{problem}[{सप्ताहांत समस्या---किसी भी बहुभुज के कोण: $(n-2) \times 180^\circ $ अंदर, हमेशा $360^\circ $ मोड़ने का बाहर, और एक फर्श पर केवल तीन नियमित आकृतियाँ ही टाइलें क्यों लगाती हैं}]
\label{pb:g7:triangles:1}
\Cref{exo:g7:triangles:11} काटो ए \omterm{def:g4:geometry:polygon}{चतुष्कोष} दो \omterm{def:g3:shapes:shapes}{त्रिकोण}ों में
और पाया $360^\circ $. यह समस्या इस विचार को आगे बढ़ाती है \omterm{def:g4:geometry:polygon}{बहुभुज}
किसी भी संख्या में भुजाओं के साथ, फिर एक सेकंड को पूरी तरह से खोज लेता है
भिन्न प्रमाण---द्वारा \emph{चलना} आकृति के चारों ओर और जोड़ना
ऊपर की ओर --- और आपके बाथरूम के फर्श पर समाप्त होता है: सबके बीच
नियमित \omterm{def:g4:geometry:polygon}{बहुभुज}, बिल्कुल तीन बिना \omterm{ex:g1:subtraction:difference}{अंतर}ाल के एक विमान को टाइल कर सकते हैं या
ओवरलैप्स, और मधुमक्खियों ने अपना चयन कर लिया है।

\medskip
\noindent\textbf{Part I --- The angles inside.}
\begin{enumerate}
  \item एक से शिखर एक का \omterm{def:g4:geometry:polygon}{षट्भुज} (छह भुजाएँ), सभी को ड्रा करें
        विकर्ण उसे छोड़ रहे हैं शिखर. कितने त्रिभुजों में है
        द \omterm{def:g4:geometry:polygon}{षट्भुज} काटो, और क्या है \omterm{def:g1:addition:def}{जोड़} का \emph{सभी} इसका
        आंतरिक कोण (\cref{thm:g7:triangles:sum})?
  \item सामान्यीकरण: एक से शिखर एक का \omterm{def:g4:geometry:polygon}{बहुभुज} साथ $ n $ पक्ष,
        विकर्णों वाला पंखा कितने त्रिभुज उत्पन्न करता है?
        के लिए सूत्र निकालें \omterm{def:g1:addition:def}{जोड़} आंतरिक कोणों का,
        और इसकी जांच करें $ n = 3$, $4$, $5$, $6$.
  \item की गणना करें \omterm{def:g1:addition:def}{जोड़} एक दशमांश के आंतरिक कोणों का
        ($10$ भुजाएँ) और एक डोडेकागन ($12$ पक्ष)।
  \item में एक \emph{नियमित} \omterm{def:g4:geometry:polygon}{बहुभुज} सभी आंतरिक कोण हैं
        बराबर. के आंतरिक कोण की गणना करें समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण}, वर्ग, नियमित \omterm{def:g4:geometry:polygon}{पंचकोण}, \omterm{def:g4:geometry:polygon}{षट्भुज} और
        \omterm{def:g4:geometry:polygon}{अष्टकोना}.
  \item बिना किसी सूत्र के स्पष्ट करें कि कोण क्यों है \omterm{def:g1:addition:def}{जोड़} से बढ़ता है
        बिलकुल $180^\circ $ हर बार \omterm{def:g4:geometry:polygon}{बहुभुज} एक प्राप्त करता है
        शिखर.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The walk around the block.}
\omterm{def:g4:geometry:polygon}{बहुभुज} ब्लॉक की सीमा के साथ-साथ चलें, हमेशा आगे की ओर। पर
प्रत्येक \omterm{def:g5:solids:def}{कोने} में आप \emph{मोड़} किसी कोण से --- द \emph{बाहरी
कोण} उस \omterm{def:g5:solids:def}{कोने} का; जब तक आप वापस नहीं आ जाते तब तक मुड़ना जारी रहता है
आपका प्रारंभिक बिंदु, आपकी प्रारंभिक दिशा की ओर मुख करके।
\begin{enumerate}[resume]
  \item एक \omterm{def:g5:solids:def}{कोने} पर जिसका आंतरिक कोण है $\widehat A $, कैसे
        आप कितना घुमाते हैं? (आंतरिक और बाहरी कोण हैं
        पूरक।) गणना करें
        कोणों वाले त्रिभुज के लिए तीन मोड़ $40^\circ $,
        $60^\circ $, $80^\circ $.
  \item प्रश्न 6 के तीन मोड़ों को \omterm{def:g1:addition:def}{जोड़}ें। क्या पूर्ण-चक्र
        क्या आप वास्तव में निरीक्षण करते हैं?
  \item बताएं कि पूरी तरह से घूमने की बारी क्यों आती है
        \emph{कोई} \omterm{def:g4:geometry:polygon}{बहुभुज} --- तीन भुजाएँ या तीस --- सदैव
        कुल बिल्कुल $360^\circ $. (तुम्हारी नाक ने क्या किया है, सब
        संयुक्त हो जाता है, जब आप वापस आते हैं?)
  \item आंतरिक-कोण सूत्र दूसरी बार निकालें: प्रत्येक पर
        का $ n $ \omterm{def:g5:solids:def}{कोने}, आंतरिक भाग $+$ मोड़ $= 180^\circ $; \omterm{def:g1:addition:def}{जोड़}
        सभी कोनों पर और प्रश्न 8 का उपयोग करें।
  \item एक नियमित में \omterm{def:g4:geometry:polygon}{बहुभुज}, सभी मोड़ समान हैं, इसलिए प्रत्येक
        बाह्य कोण है $\frac{360^\circ}{n}$. इसका उपयोग करें
        तुरंत उत्तर दें: कौन सा नियमित \omterm{def:g4:geometry:polygon}{बहुभुज} आंतरिक भाग है
        के कोण $150^\circ $? और क्यों करता है \emph{नहीं} नियमित
        \omterm{def:g4:geometry:polygon}{बहुभुज} के आंतरिक कोण हैं $155^\circ $?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Tiling the floor.}
टाइल वाले फर्श के हर बिंदु के आसपास, बैठक के \omterm{def:g5:solids:def}{कोने}
टाइलें बिल्कुल पूरी होनी चाहिए $360^\circ $ --- कोई गैप नहीं, कोई ओवरलैप नहीं
(\cref{pb:g6:lines:1}).
\begin{enumerate}[resume]
  \item प्रत्येक के लिए समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण}, वर्ग और
        नियमित \omterm{def:g4:geometry:polygon}{षट्भुज}: एक \omterm{def:g5:solids:def}{कोने} के बिंदु पर कितनी प्रतियाँ मिलती हैं
        टाइलिंग का? सत्यापित करें $360^\circ $ हर बार.
  \item वह नियमित दिखाओ \omterm{def:g4:geometry:polygon}{पंचकोण} फर्श पर टाइल नहीं लगा सकते: क्या
        कुल तीन कोनों का योग क्या होगा, और कुल चार का योग क्या होगा?
  \item दिखाओ कि कोई नियमित नहीं \omterm{def:g4:geometry:polygon}{बहुभुज} साथ \emph{अधिक} छह से अधिक
        \omterm{def:g5:solids:def}{किनारों} पर टाइल लगाई जा सकती है: प्रत्येक पर कम से कम तीन टाइलें मिलनी चाहिए
        कोना (प्रत्येक कोण से छोटा है $180^\circ $), और क्या
        के तीन कोनों पर होता है $135^\circ $ या अधिक? निष्कर्ष
        नियमित टिलर की पूरी सूची।
  \item बाथरूम के फर्श अक्सर नियमित रूप से मिश्रित होते हैं octagons छोटे के साथ
        वर्ग। सत्यापित करें कि यह कोना काम करता है:
        $135^\circ + 135^\circ + 90^\circ $. एक और क्लासिक मिश्रण
        एक वर्ग, एक नियमित \omterm{def:g4:geometry:polygon}{षट्भुज} और एक नियमित डोडेकागोन
        प्रत्येक कोना: इसे भी सत्यापित करें (डोडेकागन का आंतरिक भाग)।
        कोण प्रश्न 10 की विधि से अनुसरण करता है)।
  \item मधुमक्खियों की पसंद: छत्ते की कोशिकाएँ नियमित होती हैं षट्कोण.
        उनके कोनों को सत्यापित करें ($3$ कोण), और एक में समझाएं
        वाक्य --- डिडो की खोज की सहायता से
        (\cref{pb:g6:measure:1}) --- क्यों, तीन में से संभव है
        नियमित टाइल्स, \omterm{def:g4:geometry:polygon}{षट्भुज} मोम के मामले में यह सबसे सस्ता है
        इसमें जो शहद है.
\end{enumerate}
\end{problem}
