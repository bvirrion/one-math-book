\chapter{डेटा व्यवस्थित करना}\label{ch:g7:stats}

किसी भी चीज़ की गणना करने से पहले डेटा को एकत्र करना, गिनना और करना आवश्यक है
प्रदर्शित. यह अध्याय टिप्पणियों की एक अव्यवस्थित सूची को बदलने के बारे में है
एक स्पष्ट तालिका या चित्र में: गिनती, आवृत्तियाँ, समूहीकृत डेटा, बार
चार्ट और पाई चार्ट. एकल संख्या द्वारा डेटा का सारांश (मतलब,
माध्यिका) का विषय है \cref{ch:g9:statproba}.

\section{गिनती और आवृत्तियाँ}

\begin{definition}[गिनती और आवृत्ति]\label{def:g7:stats:frequency}
किसी सर्वेक्षण के प्रत्येक संभावित मान के लिए, यह \emph{गिनती करना}\index{गिनती करना} है
यह जितनी बार प्रकट होता है; इसका \emph{आवृत्ति}\index{आवृत्ति}
गणना को अवलोकनों की कुल संख्या से विभाजित किया जाता है:
\[
\text{frequency} = \frac{\text{count}}{\text{total}} .
\]
आवृत्तियों को इस प्रकार लिखा जा सकता है \omterm{def:g6:fractions:def}{अंश}ों, दशमलव या प्रतिशत, और
वे हमेशा जोड़ते हैं $1$ (अर्थात्\ $100\,\%$).
\end{definition}

\begin{example}\label{ex:g7:stats:frequency}
का पसंदीदा खेल $25$ एक कक्षा के छात्र, कच्ची सूची से लेकर
टेबल:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cccc|c}
sport & football & swimming & tennis & judo & total \\
\hline
count & $10$ & $6$ & $5$ & $4$ & $25$ \\
\omterm{def:g7:stats:frequency}{आवृत्ति} & $0.40$ & $0.24$ & $0.20$ & $0.16$ & $1$ \\
\end{tabular}
\end{center}
फ़ुटबॉल के लिए: $\frac{10}{25} = \frac{40}{100} = 0.40 = 40\,\%$.
\omterm{def:g7:stats:frequency}{आवृत्ति} पंक्ति का योग होना चाहिए $1$ --- एक स्थायी, निःशुल्क जांच।
\end{example}

\begin{remark}[आवृत्तियाँ क्यों?]
विभिन्न आकारों के समूहों में गिनती की तुलना नहीं की जा सकती: $10$
की एक कक्षा में फुटबॉल प्रशंसक $25$ (वह है $40\,\%$) और दिखाओ
की तुलना में उत्साह $12$ के एक स्कूल वर्ष में $120$ ($10\,\%$).
फ़्रीक्वेंसी हर चीज़ को एक समान रखती है \omterm{def:g7:prop:scale}{पैमाना}.
\end{remark}

\section{डेटा को कक्षाओं में समूहित करना}

\begin{example}[कक्षाओं]\label{ex:g7:stats:classes}
की ऊंचाई (सेमी में) $20$ से लेकर छात्र हैं $148$ को $172$.
प्रत्येक ऊँचाई को अलग-अलग सूचीबद्ध करने से लगभग उतनी ही ऊँचाई वाली एक तालिका प्राप्त होगी
छात्रों के रूप में कॉलम; इसके बजाय, समूह बनाएं \emph{कक्षाओं}:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|ccc|c}
height (cm) & $\intco{145}{155}$ & $\intco{155}{165}$ &
$\intco{165}{175}$ & total \\
\hline
count & $5$ & $9$ & $6$ & $20$ \\
\omterm{def:g7:stats:frequency}{आवृत्ति} & $0.25$ & $0.45$ & $0.30$ & $1$ \\
\end{tabular}
\end{center}
प्रत्येक ऊँचाई को बिल्कुल एक वर्ग में गिना जाता है ($155$ मे जाता
$\intco{155}{165}$, दोनों में नहीं --- इसीलिए \omterm{ex:g1:subtraction:difference}{अंतर}ाल हैं
\omterm{def:g3:division:half}{आधा} खुला)। समूहीकरण विवरण खो देता है लेकिन पठनीयता प्राप्त कर लेता है।
\end{example}

\section{चार्ट}

\begin{method}[एक चार्ट चुनना और उसका रेखांकन करना]\label{met:g7:stats:charts}
\begin{enumerate}
  \item \emph{बार चार्ट}: प्रति मान या वर्ग, ऊंचाई पर एक बार
        \omterm{def:g6:propdata:def}{आनुपातिक} गिनती के लिए (या \omterm{def:g7:stats:frequency}{आवृत्ति}). के लिए अच्छा है
        मूल्यों की तुलना करना।
  \item \emph{पाई चार्ट}: प्रति मान एक सेक्टर, कोण \omterm{def:g6:propdata:def}{आनुपातिक} तक \omterm{def:g7:stats:frequency}{आवृत्ति} --- पूर्ण वृत्त ($360^\circ $) का प्रतिनिधित्व करता है
        कुल, तो
        \[
        \text{angle} = \text{frequency} \times 360^\circ .
        \]
        संपूर्ण शेयर दिखाने के लिए अच्छा है.
  \item दोनों मामलों में: शीर्षक, लेबल, और स्नातक या
        चार्ट पर प्रतिशत लिखा है.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:stats:pie}
के खेल के लिए कोण \cref{ex:g7:stats:frequency}: फ़ुटबॉल
$0.40 \times 360 = 144^\circ $; तैरना $0.24 \times 360 =
86.4^\circ $; टेनिस $72^\circ $; जूडो $57.6^\circ $. जाँच करना:
$144 + 86.4 + 72 + 57.6 = 360$.
\end{example}

\begin{example}[आलोचनात्मक ढंग से पढ़ना]\label{ex:g7:stats:critical}
एक चार्ट जिसका ऊर्ध्वाधर अक्ष प्रारंभ होता है $9$ के बजाय $0$ एक बार बनाता है
का $10$ की एक पट्टी से दस गुना लंबा दिखता है $9.1$ --- दृष्टिगत रूप से
भ्रामक, हालाँकि आँकड़े ईमानदार हैं। के सामने पहला प्रतिबिम्ब
कोई भी चार्ट: जांचें कि अक्ष कहां से शुरू होता है और एक स्नातक स्तर क्या है
लायक (\cref{met:g6:propdata:read}).
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g7:stats:1}
यहाँ के निशान हैं $20$ छात्र: $12$, $15$, $9$, $12$, $14$,
$9$, $12$, $15$, $11$, $12$, $9$, $14$, $12$, $11$, $15$, $9$, $12$,
$14$, $11$, $12$. गिनती की तालिका बनाएं (प्रति चिह्न एक कॉलम)।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:stats:2}
की तालिका पूरी करें \cref{exo:g7:stats:1} के साथ \omterm{def:g7:stats:frequency}{आवृत्ति} पंक्ति
(जैसे \omterm{def:g6:fractions:def}{अंश}ों का $20$, फिर प्रतिशत के रूप में)। जाँचें \omterm{def:g1:addition:def}{जोड़}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:stats:3}
के एक सर्वेक्षण में $50$ परिवार, $30$ एक कार है. क्या है
\omterm{def:g7:stats:frequency}{आवृत्ति} ``एक कार'' का, दशमलव के रूप में और प्रतिशत के रूप में? दूसरे में
सर्वेक्षण, $45$ परिवारों से बाहर $90$ एक कार है: जो सर्वेक्षण से पता चलता है
उच्चतर अनुपात?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:stats:4}
का \omterm{def:g2:measure:units}{द्रव्यमान} (कि.ग्रा.) $15$ कुत्ते: $8$, $23$, $31$, $12$, $9$, $27$, $35$,
$14$, $22$, $18$, $29$, $11$, $25$, $33$, $16$. उन्हें इसमें समूहित करें
कक्षाएं $\intco{5}{15}$, $\intco{15}{25}$, $\intco{25}{35}$,
$\intco{35}{45}$ और गिनती दीजिए.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:stats:5}
तालिका का बार चार्ट बनाएं:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|cccc}
pets & $0$ & $1$ & $2$ & $3$ \\
\hline
families & $8$ & $12$ & $6$ & $4$ \\
\end{tabular}
\end{center}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:stats:6}
की मेज के लिए \cref{exo:g7:stats:5}, आवृत्तियों की गणना करें और
प्रत्येक मान के पाई-चार्ट कोण (जांचें: कुल $360^\circ $).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:stats:7}
पसंदीदा मौसमों के बारे में एक पाई चार्ट दिखाता है: गर्मी $162^\circ $, वसंत
$90^\circ $, शरद ऋतु $54^\circ $, सर्दी $54^\circ $. कितना प्रतिशत
प्रत्येक सीज़न को चुना? अगर $60$ लोगों ने उत्तर दिया, कितनों ने गर्मी चुनी?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:stats:8}
कक्षा ए में, $12$ छात्र बाहर $30$ स्कूल जाओ; कक्षा बी में,
$14$ से बाहर $40$. किस वर्ग में पैदल चलने वालों का अनुपात अधिक है?
आवृत्तियों से औचित्य सिद्ध करें, गिनती से नहीं।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:stats:9}
एक पत्रिका मासिक बिक्री का बार चार्ट छापती है: $9\,800$, तब
$10\,000$, तब $10\,100$, ऊर्ध्वाधर अक्ष से शुरू होने के साथ
$9\,700$. वर्णन करें कि पाठक क्या देखता है, और कितनी ईमानदार धुरी है
से शुरू हो रहा है $0$ इसके बजाय दिखाएगा.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:stats:10}
ए \omterm{def:g7:stats:frequency}{आवृत्ति} तालिका में आवृत्तियों के साथ तीन मान हैं $0.35$, $0.4$ और
$ f $. खोजो $ f $. यदि कुल गिनती है $80$, तीन गिनती दीजिए।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g7:stats:11}
एक स्कूल में, $55\,\%$ छात्रों में लड़कियाँ हैं। लड़कियों के बीच,
$40\,\%$ कैफेटेरिया में खाओ; लड़कों के बीच, $60\,\%$ करना। से बाहर
$400$ छात्र, कैफेटेरिया में कितने लोग खाते हैं? (चारों की गणना करें
समूह का आकार चरण दर चरण।) वह कुल मिलाकर कितना प्रतिशत है?
\end{exercise}

\section{समस्या: डेटा कैसे झूठ बोलता है --- और इसे कैसे पकड़ा जाए}

\begin{problem}[{सप्ताहांत समस्या --- आवृत्तियों के विरुद्ध गणना, वह सर्वेक्षण जिसने एक देश को मूर्ख बनाया, और दोनों का विरोधाभास अस्पताल}]
\label{pb:g7:stats:1}
संख्याएँ झूठ नहीं बोलतीं, लेकिन उन्हें गुमराह करने के लिए बनाया जा सकता है: एक बड़ी संख्या
एक छोटा सा छिपा सकते हैं \omterm{def:g7:stats:frequency}{आवृत्ति}, एक विशाल सर्वेक्षण बेकार हो सकता है,
और --- सबसे अजीब बात --- एक अस्पताल अपने प्रतिद्वंद्वी को हरा सकता है
\emph{प्रत्येक} रोगियों की श्रेणी और कुल मिलाकर अभी भी नुकसान है
आंकड़े. इस समस्या में तीनों \omterm{def:g5:solids:net}{जाल} केवल हथियारों से लैस होकर ही फैले हैं
इस अध्याय की गिनती और आवृत्तियों के साथ
(\cref{def:g7:stats:frequency}).

\medskip
\noindent\textbf{Part I --- Counts are not frequencies.}
\begin{enumerate}
  \item स्कूल ए रीसायकल करता है $120$ उसके जैसा $400$ जूस के डिब्बे;
        स्कूल बी, $90$ उसके जैसा $250$. कौन सा स्कूल बड़ा है
        \emph{\omterm{def:g7:stats:frequency}{गिनती करना}} पुनर्चक्रित डिब्बों का? बड़ा
        \emph{\omterm{def:g7:stats:frequency}{आवृत्ति}}? कौन सा स्कूल रीसाइक्लिंग का हकदार है
        पुरस्कार?
  \item एक नई समय सारिणी का सर्वेक्षण किया गया है: $18$ की $30$ शिक्षक
        यह पसंद है, और $210$ की $600$ छात्र करते हैं. की गणना करें
        दो आवृत्तियाँ, फिर \omterm{def:g7:stats:frequency}{आवृत्ति} इन सब में $630$
        लोग एक साथ. संयुक्त आंकड़ा ऐसा क्यों बैठता है
        छात्रों के एक के करीब?
  \item एक प्रकाशित पाई चार्ट सेक्टरों को लेबल करके दिखाता है $45\,\%$,
        $30\,\%$, $20\,\%$ और $10\,\%$. बिना किसी और के
        जानकारी, आपको कैसे पता चलेगा कि गलती हुई है?
  \item ए \omterm{def:g7:stats:frequency}{आवृत्ति} टेबल खत्म $40$ अवलोकन आधे-मिटाए गए हैं:
        मायने रखता है $14$, $10$, $?$, $?$ और आवृत्तियाँ $0.35$,
        $0.25$, $0.15$, $?$. लुप्त प्रविष्टियों का पुनर्निर्माण करें.
  \item एक सर्वेक्षण आवृत्तियों के साथ समाप्त होता है $0.40$, $0.35$ और
        $0.25$. तीन पाई-चार्ट कोणों की गणना करें
        (\cref{met:g7:stats:charts}) और जांचें कि वे बंद कर देते हैं
        वृत्त.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The poll that fooled a country.}
1936 में, एक अमेरिकी पत्रिका ने दस मिलियन मतपत्र डाक से भेजे
टेलीफोन निर्देशिकाओं और कार पंजीकरण से लिए गए पते
सूचियाँ, प्राप्त $2.4$ लाखों उत्तर दिए, और विनाश की भविष्यवाणी की
उम्मीदवार लैंडन की जीत। एक युवा सांख्यिकीविद्, जॉर्ज
गैलप ने लगभग पचास हजार लोगों से ही पूछताछ की--- चुना गया
पूरी आबादी से मिलते-जुलते --- और इसके विपरीत भविष्यवाणी की।
रूज़वेल्ट भारी बहुमत से जीते।
\begin{enumerate}[resume]
  \item क्या \omterm{def:g7:stats:frequency}{आवृत्ति} डाक द्वारा भेजे गए मतपत्र वापस आ गए?
  \item 1936 में टेलीफोन और कारें विलासिता की वस्तुएँ थीं। में स्पष्ट करें
        हालाँकि, एक या दो वाक्य पत्रिका का नमूना क्यों
        प्रचंड, बर्बाद हो गया --- और प्रश्न 1 का कौन सा पाठ
        यह राष्ट्रीय स्तर पर दोहराया जाता है \omterm{def:g7:prop:scale}{पैमाना}.
  \item एक खिलौना मॉडल. एक शहर है $1\,000$ धनी मतदाता, जिनमें से
        $70\,\%$ समर्थन एल, और $9\,000$ मामूली मतदाता, जिनमें से
        $30\,\%$ एल का समर्थन करें। एल के सच्चे समर्थन की गणना करें
        शहर. एक ``टेलीफोन-बुक पोल'' पहुंचता है $500$ धनी
        और $500$ मामूली मतदाता: यह किस समर्थन की भविष्यवाणी करता है?
  \item मूक \omterm{def:g5:solids:net}{जाल}: एक कंपनी के पास है $200$ नाखुश ग्राहक
        और $800$ खुश लोग. यह सभी का सर्वेक्षण करता है; $40\,\%$ का
        नाखुश जवाब, लेकिन केवल $10\,\%$ खुशियों का करो.
        प्रत्येक प्रकार के उत्तरों की संख्या की गणना करें, और
        \omterm{def:g7:stats:frequency}{आवृत्ति} दुःख का \emph{उत्तरों के बीच}.
        सत्य से तुलना करें \omterm{def:g7:stats:frequency}{आवृत्ति}.
  \item प्रश्न 8 के पोल को सुधारें: रखें $1\,000$
        साक्षात्कार, लेकिन उन्हें शहर के सच में वितरित करें
        अनुपात. मरम्मत किया गया सर्वेक्षण क्या भविष्यवाणी करता है?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The paradox of the two
hospitals.}
दो अस्पतालों ने गंभीरता के अनुसार विभाजित होकर अपने इलाज की संख्या प्रकाशित की:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cc}
 & mild cases & severe cases \\
\hline
hospital A & $90$ cured of $100$ & $280$ cured of $400$ \\
hospital B & $340$ cured of $400$ & $65$ cured of $100$ \\
\end{tabular}
\end{center}
\begin{enumerate}[resume]
  \item अस्पताल ए के इलाज की गणना करें \omterm{def:g7:stats:frequency}{आवृत्ति} हल्के मामलों के लिए, के लिए
        गंभीर मामले, और कुल मिलाकर (सभी) $500$ मरीज़)।
  \item अस्पताल बी के लिए वही तीन गणनाएँ। कौन सा अस्पताल
        हल्के मामलों पर जीत? गंभीर मामलों पर?
  \item दो समग्र आवृत्तियों की तुलना करें। स्पष्ट रूप से बताएं
        अजीब बात हुई है.
  \item तरकीब समझाएं: दोनों अस्पतालों की तुलना करें'
        \emph{घोला जा सकता है} रोगियों की, और कहें कि कुल मिलाकर क्यों
        \omterm{def:g7:stats:frequency}{आवृत्ति} दोनों श्रेणी की आवृत्तियों को धोखा दे सकता है। यदि आप
        कोई गंभीर मामला था, आपको कौन सा अस्पताल चुनना चाहिए?
  \item एक अख़बार का शीर्षक है: ``अस्पताल बी बेहतर है
        इलाज दर: $81\,\%$ ख़िलाफ़ $74\,\%$.''लिखें
        संपादक को दो-वाक्य का पत्र कि यह पूरा
        समस्या ने तुम्हें --- एक वाक्य लिखना सिखाया है
        शीर्षक किस चीज़ को नज़रअंदाज़ करता है, एक सामान्य नैतिकता पर
        (समान की तुलना समान से करें और वज़न का ध्यान रखें)।
\end{enumerate}
\end{problem}
