\section{प्राथमिकता नियम}

\begin{theorem}[प्राथमिकता नियम]\label{thm:g7:priorities:rules}
कोष्ठक के बिना गणना में:
\begin{enumerate}
  \item गुणा और भाग किया जाता है \emph{पहले}
        \omterm{def:g1:addition:def}{जोड़} और घटाव;
  \item समान स्तर के ऑपरेशन बाएँ से दाएँ किए जाते हैं।
\end{enumerate}
ब्रैकेट हर चीज़ को ओवरराइड करते हैं: ब्रैकेट के अंदर क्या है इसकी गणना की जाती है
सबसे पहले, अंतरतम से शुरू करना।
\end{theorem}

\begin{example}\label{ex:g7:priorities:basic}
प्रति पंक्ति एक कदम, जो गणना की गई है उसे (मानसिक रूप से) \omterm{def:g6:lines:objects}{रेखा}ंकित करना:
\begin{align*}
3 + 4 \times 5 &= 3 + 20 = 23
&&\text{(multiplication first)}\\
20 - 8 + 2 &= 12 + 2 = 14
&&\text{(same level: left to right)}\\
18 \div 3 \times 2 &= 6 \times 2 = 12
&&\text{(same level: left to right)}\\
(3 + 4) \times 5 &= 7 \times 5 = 35
&&\text{(brackets first).}
\end{align*}
\omterm{def:g5:solids:net}{जाल}ों पर ध्यान दें: $20 - 8 + 2$ है \emph{नहीं} $20 - 10$, और
$18 \div 3 \times 2$ है \emph{नहीं} $18 \div 6$.
\end{example}

\begin{example}[नेस्टेड कोष्ठक]\label{ex:g7:priorities:nested}
\begin{align*}
50 - 2 \times \bigl(3 + (10 - 4)\bigr)
&= 50 - 2 \times (3 + 6) && \text{(innermost bracket)}\\
&= 50 - 2 \times 9 && \text{(remaining bracket)}\\
&= 50 - 18 && \text{(multiplication)}\\
&= 32 .
\end{align*}
\end{example}

\begin{remark}[अंश पट्टियाँ कोष्ठकों को छिपाती हैं]
ए \omterm{def:g6:fractions:def}{अंश} बार इसका समूह बनाता है \omterm{def:g4:fractions:def}{मीटर} और इसके \omterm{def:g5:division:multiple}{भाजक}, मानो प्रत्येक
अदृश्य कोष्ठक थे:
\[
\frac{12 + 8}{4} = \frac{20}{4} = 5,
\qquad\text{not}\quad 12 + \frac{8}{4} = 14 .
\]
\end{remark}

\begin{method}[सुरक्षित रूप से कंप्यूटिंग]\label{met:g7:priorities:safe}
\begin{enumerate}
  \item सबसे पहले करने के लिए ऑपरेशन का पता लगाएं (फिर सबसे अंदर वाले कोष्ठक)।
        $\times $ और $\div $, तब $+$ और $-$);
  \item गणना \emph{केवल वही}, और संपूर्ण अभिव्यक्ति को फिर से लिखें
        परिणाम के साथ;
  \item एक संख्या \omterm{def:g3:division:remainder}{शेष} रहने तक दोहराएँ --- प्रति पंक्ति एक ऑपरेशन, नहीं
        शॉर्टकट.
\end{enumerate}
\end{method}

\section{वितरणशीलता}

\begin{theorem}[वितरणशीलता]\label{thm:g7:priorities:distributivity}
सभी नंबरों के लिए $ k $, $ a $, $ b $:
\[
k \times (a + b) = k \times a + k \times b,
\qquad
k \times (a - b) = k \times a - k \times b .
\]
\end{theorem}

\begin{proof}[एक चित्र द्वारा प्रमाण]
गिनें \omterm{def:g6:measure:area}{क्षेत्र} चौड़ाई के एक आयत का $ k $ दो छोटे भागों में विभाजित
लंबाई के आयत $ a $ और $ b $: समग्र रूप से यह है $ k \times (a+b)$;
यह दो भागों में है $ k \times a + k \times b $. वही आयत, वही
\omterm{def:g6:measure:area}{क्षेत्र}.
\end{proof}

  \fill[omThm!14] (4.2,0) rectangle (6.4,1.8);

\end{tikzpicture}

{\small वितरण के रूप में \omterm{def:g6:measure:area}{क्षेत्र}ों: बड़ा आयत
$ k \times (a + b)$ है \omterm{def:g1:addition:def}{जोड़} दो छोटे लोगों में से।}
\end{omfigure}

\begin{example}[दोनों दिशाओं में]\label{ex:g7:priorities:distributivity}
\emph{विस्तार} (बाएं से दाएं) मानसिक गणित को आसान बनाता है:
\[
7 \times 103 = 7 \times (100 + 3) = 700 + 21 = 721,
\qquad
6 \times 98 = 6 \times (100 - 2) = 600 - 12 = 588 .
\]
\emph{फैक्टरिंग} (दाएं से बाएं) एक सामान्य कारक एकत्र करता है:
\[
23 \times 12 + 23 \times 8 = 23 \times (12 + 8) = 23 \times 20 = 460 .
\]
\end{example}

\begin{example}[शब्दों को गणना में अनुवाद करना]\label{ex:g7:priorities:words}
``द \omterm{def:g1:addition:def}{जोड़} का $8$ और यह \omterm{def:g2:mult:def}{उत्पाद} का $5$ द्वारा $3$'' है $8 + 5 \times 3 =
23$. ``द \omterm{def:g2:mult:def}{उत्पाद} का $5$ से \omterm{def:g1:addition:def}{जोड़} का $8$ और $3$'' है
$5 \times (8 + 3) = 55$: शब्द ``द \omterm{def:g2:mult:def}{उत्पाद} का \dots\ से \omterm{def:g1:addition:def}{जोड़}
का\dots'' चारों ओर कोष्ठक लगाएं \omterm{def:g1:addition:def}{जोड़}.
\end{example}

\section{समस्या: मिस्र के मुंशी की तरह गुणा करना}

\begin{problem}[{सप्ताहांत समस्या--- मानसिक-गणित युक्तियाँ वितरणशीलता, और दोहरीकरण-आधा गुणन के लिए उपयोग किया जाता है चार हजार वर्ष}]
\label{pb:g7:priorities:1}
कैलकुलेटर से बहुत पहले --- और गुणन सारणी से भी पहले
स्कूल में ड्रिल किया गया --- मिस्र के शास्त्री और रूसी किसान
केवल का उपयोग करके किन्हीं दो संख्याओं को गुणा करें \emph{दोगुना करना, \omterm{def:g3:division:half}{आधा} करना और
जोड़ना}. उनका तरीका जादू जैसा दिखता है; यह वास्तव में है
वितरणशीलता (\cref{thm:g7:priorities:distributivity}) काम करना
पूरी शक्ति से. तो यह समस्या आपके मानसिक अंकगणित को तेज़ कर देती है
आपको प्राचीन विधि सिखाता है, फिर हूबहू देखने के लिए हुड खोलता है
यह क्यों काम करता है.

\medskip
\noindent\textbf{Part I --- The art of not computing.}
\begin{enumerate}
  \item मानसिक रूप से गणना करें, जैसे कि विस्तार करें
        \cref{ex:g7:priorities:distributivity}:
        $7 \times 102$ और $9 \times 98$.
  \item मानसिक रूप से गणना करें, फैक्टरिंग:
        $17 \times 13 + 17 \times 7$ और
        $45 \times 101 - 45$.
  \item ``समय $5$'' युक्ति: से गुणा करना $5$, गुणा करके
        $10$ और \omterm{def:g3:division:half}{आधा}. गणना करना $5 \times 87$ इस तरह, और
        बताएं कि यह युक्ति वैध क्यों है।
  \item औचित्य सिद्ध करने और उपयोग करने के लिए दो और तरकीबें: से गुणा करें $25$ द्वारा
        से गुणा करना $100$ और से विभाजित करना $4$ (गणना करें
        $25 \times 32$); गुणा करके $99$ से गुणा करके $100$
        और एक बार घटाना (गणना करें)। $99 \times 47$).
  \item एक चेतावनी शॉट: वितरण एक कारक से अधिक फैलता है
        \emph{\omterm{def:g1:addition:def}{जोड़}}, कदापि नहीं \omterm{def:g2:mult:def}{उत्पाद}. तुलना करना
        $8 \times (5 + 3)$ साथ $8 \times 5 + 8 \times 3$, और तब $8 \times (5 \times 3)$ साथ
        $(8 \times 5) \times (8 \times 3)$: कौन सा \omterm{def:g1:addition:def}{जोड़}ा मेल खाता है,
        कौन सा नहीं?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The doubling--halving method.}
यहाँ के लिए नुस्खा है $13 \times 24$. दो कॉलम लिखें. वाम
कॉलम: प्रारंभ करें $13$ और \emph{\omterm{def:g3:division:half}{आधा} करना}, किसी को छोड़ना \omterm{def:g3:division:remainder}{शेष},
नीचे तक $1$: \ $13,\ 6,\ 3,\ 1$. दायां स्तंभ: प्रारंभ करें $24$ और
\emph{दोहरा} जितनी बार: \ $24,\ 48,\ 96,\ 192$. पार करना
हर पंक्ति जिसका \emph{बाएं} वह संख्या है \omterm{def:g3:numbers:evenodd}{यहां तक की}, और बचे हुए को \omterm{def:g1:addition:def}{जोड़}ें
सही संख्या.
\begin{enumerate}[resume]
  \item के लिए नुस्खा अपनाएं $13 \times 24$: कौन सी पंक्ति है
        काट दिया, क्या \omterm{def:g1:addition:def}{जोड़} रहता है, और बराबर होता है
        $13 \times 24$?
  \item गुणा $21 \times 17$ विधि द्वारा (\omterm{def:g3:division:half}{आधा} $21$, दोहरा
        $17$), और सामान्य गुणन द्वारा उत्तर की जाँच करें।
  \item गुणा $16 \times 25$ विधि द्वारा. कितनी पंक्तियाँ
        जीवित रहें, और बाएं कॉलम के बारे में क्या खास है
        से शुरू होता है $1$, $2$, $4$, $8$, $16, \dots $?
  \item अब \emph{क्यों} यह एक समय में एक कदम पर काम करता है। प्रत्येक को उचित ठहराएँ
        वितरण के साथ समानता:
        \[
        13 \times 24 = (12 + 1) \times 24
        = 12 \times 24 + 24 = 6 \times 48 + 24 .
        \]
        नैतिक: बाएँ को \omterm{def:g3:division:half}{आधा} करना और दाएँ को दोगुना करना बनाए रखता है
        \omterm{def:g2:mult:def}{उत्पाद} --- लेकिन जब बायां नंबर है \emph{\omterm{def:g3:numbers:evenodd}{विषम}}, एक
        सही संख्या की प्रतिलिपि अलग हो जाती है और उसे बैंक में रखा जाना चाहिए।
        टेबल की पहली पंक्ति किस नंबर को बैंक करती है?
  \item टेबल के नीचे बहीखाता जारी रखें:
        $6 \times 48 = 3 \times 96$, तब
        $3 \times 96 = 1 \times 192 + 96$, तब
        $1 \times 192 = 192$. वह सब कुछ एकत्र करें जो बैंक में रखा गया था
        रास्ते में, और बताएं कि बैंक नंबर क्यों हैं
        a के साथ पंक्तियों की बिल्कुल सही-स्तंभ संख्याएँ \omterm{def:g3:numbers:evenodd}{विषम}
        बाएँ --- रेसिपी द्वारा रखी गई पंक्तियाँ।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Under the hood: रकम of
doublings.}
\begin{enumerate}[resume]
  \item प्रश्न 6 में, बची हुई पंक्तियाँ सार्थक थीं $1$, $4$
        और $8$ का दोगुना $24$: विधि ने चुपचाप लिखा
        $13 \times 24 = (1 + 4 + 8) \times 24$. जाँच करें कि
        $1 + 4 + 8 = 13$, और पता लगाएं कि कौन सा दोगुना हो गया है ($1$, $2$,
        $4$, $8$, $16$) विधि के लिए रखा गया है $21$ प्रश्न 7 में.
  \item लिखना $25$ और $42$ जैसा रकम दोहरीकरण से संख्याओं का
        सूची $1, 2, 4, 8, 16, 32$, प्रत्येक का अधिकतम एक बार उपयोग करना।
  \item समझाइए क्यों \emph{प्रत्येक} पूर्ण संख्या लिखी जा सकती है
        इस प्रकार: सबसे बड़ी दोहरीकरण संख्या लें जो फिट बैठती हो,
        घटाना, दोहराना --- यह क्यों रुकना चाहिए? लिखना $100$ जैसे
        ऐसा \omterm{def:g1:addition:def}{जोड़}.
  \item मिस्र के चार हज़ार शास्त्रियों ने ठीक ऐसा ही किया
        वर्षों पहले: गुणा करना $19 \times 35$, उन्होंने एक बनाया
        दोहरीकरण तालिका $35$ ($1 \to 35$, $2 \to 70$,
        $4 \to 140$, $8 \to 280$, $16 \to 560$), पंक्तियों को चुना
        तक जोड़ना $19$, और सारांशित किया। उनकी गणना करें
        रास्ता.
  \item समापन: आधुनिक कंप्यूटर, जिनकी गिनती होती है $0$ रेत
        $1$ एस, मूलतः शास्त्रियों की तरह गुणा करें --- द्वारा
        दोहरीकरण और परिवर्धन। कौन से तीन \omterm{def:g1:addition:def}{जोड़} होंगे a
        कंप्यूटर के लिए बनाओ $13 \times 24$? (प्रत्येक संख्या क्यों
        दोहरीकरण में विभाजित होता है, और क्या $0$ रेत $1$ करना होगा
        इसके साथ, पॉवर्स अध्याय की सप्ताहांत समस्या है:
        \cref{pb:g8:powers:1}.)
\end{enumerate}
\end{problem}
