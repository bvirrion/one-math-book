\chapter{भिन्न: तुलना करना और जोड़ना}\label{ch:g7:fractions}

ग्रेड 6 का परिचय दिया गया \omterm{def:g6:fractions:def}{अंश}ों शेयरों के रूप में और सटीक के रूप में \omterm{def:g3:division:remainder}{भागफल}
(\cref{ch:g6:fractions}). यह अध्याय सीखता है \emph{गणना} के साथ
उन्हें: समान पहचानना \omterm{def:g6:fractions:def}{अंश}ों, तुलना करना, जोड़ना और घटाना
--- अभी के लिए मैत्रीपूर्ण हरों; सामान्य मामला सामने आता है
\cref{ch:g9:fractions}.

\section{समान भिन्न}

\begin{theorem}[समान भिन्न]\label{thm:g7:fractions:equal}
ए \omterm{def:g6:fractions:def}{अंश} जब यह अपरिवर्तित होता है \omterm{def:g4:fractions:def}{मीटर} और \omterm{def:g5:division:multiple}{भाजक} हैं
समान \omterm{def:g1:counting:zero}{शून्य}ेतर संख्या से गुणा या भाग करना:
\[
\frac{a}{b} = \frac{a \times k}{b \times k} .
\]
\end{theorem}
\admitted

\begin{example}\label{ex:g7:fractions:equal}
$\dfrac{3}{4} = \dfrac{3 \times 5}{4 \times 5} = \dfrac{15}{20}$, और
दूसरी दिशा में $\dfrac{42}{30} = \dfrac{42 \div 6}{30 \div 6}
= \dfrac{7}{5}$ (द्वारा सरलीकृत) $6$).

यह जांचने के लिए कि क्या $\dfrac{8}{12}$ और $\dfrac{10}{15}$ बराबर हैं,
दोनों को सरल बनाएं: $\dfrac{8}{12} = \dfrac{2}{3}$ और
$\dfrac{10}{15} = \dfrac{2}{3}$ --- हाँ, बराबर.
\end{example}

\section{भिन्नों की तुलना करना}

\begin{method}[दो भिन्नों की तुलना करना]\label{met:g7:fractions:compare}
\begin{enumerate}
  \item \emph{वही \omterm{def:g5:division:multiple}{भाजक}}: तुलना करें \omterm{def:g6:fractions:def}{अंश}:
        $\frac{5}{9} < \frac{7}{9}$.
  \item \emph{एक \omterm{def:g5:division:multiple}{भाजक} एक है एकाधिक दूसरे का}: पुनः लिखें
        मोटा \omterm{def:g6:fractions:def}{अंश}, फिर तुलना करें:
        $\frac{5}{6}$ बनाम $\frac{7}{12}$: $\frac56 = \frac{10}{12} >
        \frac{7}{12}$.
  \item \emph{साथ तुलना करें $1$ या $\frac12$} जब संभव हो:
        $\frac{9}{8} > 1 > \frac{7}{9}$ सुलझेगी
        $\frac98$ बनाम $\frac79$ तुरन्त।
\end{enumerate}
\end{method}

\section{जोड़ना और घटाना}

\begin{theorem}[वही भाजक]\label{thm:g7:fractions:add}
भिन्न उसी के साथ \omterm{def:g5:division:multiple}{भाजक} द्वारा \omterm{def:g1:addition:def}{जोड़}ा (या घटाया) जाता है
जोड़ना (या घटाना)। \omterm{def:g6:fractions:def}{अंश}:
\[
\frac{a}{d} + \frac{b}{d} = \frac{a + b}{d},
\qquad
\frac{a}{d} - \frac{b}{d} = \frac{a - b}{d} .
\]
\end{theorem}

\begin{proof}[क्यों]
$ a $ आकार के भाग $\frac1d $ प्लस $ b $ एक ही आकार के हिस्से बनाते हैं
$ a + b $ उस आकार के हिस्से.
\end{proof}

\begin{method}[विभिन्न भाजक]\label{met:g7:fractions:add}
जब एक \omterm{def:g5:division:multiple}{भाजक} एक है एकाधिक दूसरे का:
\begin{enumerate}
  \item फिर से लिखें \omterm{def:g6:fractions:def}{अंश} छोटे के साथ \omterm{def:g5:division:multiple}{भाजक} तो दोनों के पास है
        बड़ा वाला (\cref{thm:g7:fractions:equal});
  \item \omterm{def:g1:addition:def}{जोड़}ें या घटाएँ \omterm{def:g6:fractions:def}{अंश};
  \item यदि संभव हो तो परिणाम को सरल बनाएं।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:fractions:add}
गणना करना $\dfrac{3}{4} + \dfrac{5}{8}$, क्रमशः:
\begin{enumerate}
  \item $8$ एक है एकाधिक का $4$: पुनः लिखें
        $\dfrac34 = \dfrac{3 \times 2}{4 \times 2} = \dfrac{6}{8}$;
  \item जोड़ना: $\dfrac{6}{8} + \dfrac{5}{8} = \dfrac{11}{8}$;
  \item सरल बनाने के लिए कुछ भी नहीं: परिणाम है $\dfrac{11}{8}$, अर्थात\
        $1 + \dfrac38$.
\end{enumerate}
एक और: $2 - \dfrac{4}{5} = \dfrac{10}{5} - \dfrac{4}{5} =
\dfrac{6}{5}$ (के ऊपर पूरी संख्या लिखें \omterm{def:g5:division:multiple}{भाजक} $5$
पहला)।
\end{example}

\begin{example}[एक क्लासिक जाल]\label{ex:g7:fractions:trap}
$\dfrac12 + \dfrac13$ है \emph{नहीं} $\dfrac{2}{5}$! \omterm{def:g1:addition:def}{जोड़}ा जा रहा है \omterm{def:g6:fractions:def}{अंश}
और हरों अलग से गलत है --- उत्तर $\frac25$ होगा
से छोटा $\frac12$, बेतुका जब \emph{जोड़ना} कुछ सकारात्मक
को $\frac12$. (सही \omterm{def:g1:addition:def}{जोड़}, $\frac56$, एक आम की जरूरत है \omterm{def:g5:division:multiple}{भाजक}
$6$; सामान्य विधि अंदर है \cref{ch:g9:fractions}.)
\end{example}

\section{किसी भिन्न को किसी संख्या से गुणा करना}

\begin{proposition}[किसी संख्या का भिन्न गुना]\label{prop:g7:fractions:mult}
एक नंबर के लिए $ k $:
\[
k \times \frac{a}{b} = \frac{k \times a}{b} .
\]
लेना $\frac ab $ \emph{का} एक मात्रा का अर्थ है गुणा करना
$\frac ab $.
\end{proposition}
\admitted

\begin{example}\label{ex:g7:fractions:mult}
$6 \times \dfrac{2}{3} = \dfrac{12}{3} = 4$: छह गुना दो तिहाई है
चार पूर्ण. और ``$\frac34$ का $60$ यूरो'' है
$\frac34 \times 60 = \frac{180}{4} = 45$ यूरो --- वही उत्तर
द्वारा विभाजित करना $4$ फिर से गुणा करना $3$
(\cref{met:g6:fractions:ofquantity}).
\end{example}

\section{समस्या: संगीत की एक पट्टी में छिपे अंश}

\begin{problem}[{सप्ताहांत समस्या --- नोट मान भिन्न हैं इसका योग बिल्कुल सटीक होना चाहिए, लय गिनती एक प्रसिद्ध अनुक्रम को छुपाती है, और समता बाल्कन बीट पर शासन करती है}]
\label{pb:g7:fractions:1}
संगीत में लिखा गया है \omterm{def:g6:fractions:def}{अंश}ों. ए \emph{संपूर्ण नोट} रहता है $1$
माप (सामान्य ``चार-चार'' समय में); ए \emph{\omterm{def:g3:division:half}{आधा} टिप्पणी}
रहता है $\frac12$; फिर आओ \emph{\omterm{def:g3:division:half}{तिमाही} टिप्पणी} $\frac14$, द
\emph{आठवां नोट} $\frac18$ और यह \emph{सोलहवाँ नोट}
$\frac{1}{16}$. एक अनुल्लंघनीय कानून: \emph{ए के अंदर की अवधि
माप का योग बिल्कुल माप के कुल योग के बराबर होना चाहिए} --- $1$ में
चार-चार बार. नीचे दिया गया प्रत्येक प्रश्न इस अध्याय का अंकगणित है
(\cref{thm:g7:fractions:add}, \cref{met:g7:fractions:add}) पर सेट करें
संगीत; रास्ते में आपको एक गिनती क्रम का पता चलेगा
एक हजार वर्ष पूर्व भारतीय विद्वान।

\medskip
\noindent\textbf{Part I --- Notes that must add up.}
\begin{enumerate}
  \item जांचें कि इनमें से प्रत्येक उपाय चार-चार में कानूनी है
        समय: (ए) \omterm{def:g3:division:half}{आधा} $+$ \omterm{def:g3:division:half}{तिमाही} $+$ \omterm{def:g3:division:half}{तिमाही}; (बी) \omterm{def:g3:division:half}{तिमाही} $+$
        आठवाँ $+$ आठवाँ $+$ \omterm{def:g3:division:half}{आधा}.
  \item एक नकलची ने लिखा: \omterm{def:g3:division:half}{आधा} $+$ \omterm{def:g3:division:half}{तिमाही} $+$ आठवां. क्या
        \omterm{def:g6:fractions:def}{अंश} माप का लिखा है, और कौन सा एकल
        नोट इसे पूरा करता है?
  \item कितने सोलहवें नोट एक पूरा माप भरते हैं? कैसे
        कई सोलहवें एक आठवां नोट बनाते हैं?
  \item ए \emph{डॉट} एक नोट जोड़ने के बाद \omterm{def:g3:division:half}{आधा} नोट का मूल्य: ए
        बिंदीदार \omterm{def:g3:division:half}{आधा} है $\frac12 + \frac14$, एक बिंदीदार \omterm{def:g3:division:half}{तिमाही} है
        $\frac14 + \frac18$. दोनों मानों की गणना करें.
  \item एक वाल्ट्ज तीन-चार बार लिखा जाता है: प्रत्येक माप अवश्य होना चाहिए
        कुल $\frac34$. जांचें कि एक एकल बिंदीदार है \omterm{def:g3:division:half}{आधा} एक भरता है
        वाल्ट्ज मापें, और दो अन्य कानूनी वाल्ट्ज लिखें
        केवल उपयोग करने वाले उपाय क्वार्टरों और आठवें.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The copyist's workshop.}
\begin{enumerate}[resume]
  \item चार-चार माप में शामिल हैं: बिंदीदार \omterm{def:g3:division:half}{तिमाही} $+$ आठवाँ
        $+$ \omterm{def:g3:division:half}{तिमाही}. किसकी कमी है?
  \item जो अधिक समय तक चलता है, एक बिंदीदार आठवां या एक \omterm{def:g3:division:half}{तिमाही} टिप्पणी?
        सोलहवीं से तुलना करें.
  \item ए \emph{बाँधना} अवधियों को एक लंबे समय में एक साथ चिपका देता है
        ध्वनि: ए \omterm{def:g3:division:half}{आधा} आठवें से बंधा नोट कितने समय तक चलता है? कैसे
        कई सोलहवाँ वह है?
  \item माप के नोट्स का योग बनता है $\frac{11}{16}$;
        \omterm{def:g3:division:remainder}{शेष} मौन है (ए \emph{आराम}). कितना समय है
        आराम?
  \item एक ढोल वाला एक भरता है \omterm{def:g3:division:half}{तिमाही} टिप्पणी ($\frac{4}{16}$ का
        माप) केवल आठवें भाग का उपयोग करके ($\frac{2}{16}$) और
        सोलहवीं ($\frac{1}{16}$), आदेश मायने रखता है: के लिए
        उदाहरण आठवां--सोलहवां--सोलहवां, या
        सोलहवाँ - आठवाँ - सोलहवाँ। जांचें कि दोनों उदाहरण
        कानूनी हैं, फिर सूची \emph{सभी} संभव लय.
        कितने हैं?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- A thousand-year-old sequence, and
an \omterm{def:g3:numbers:evenodd}{विषम} measure.}
\begin{enumerate}[resume]
  \item आइए हम लंबे स्वरों के लिए ड्रमर की लय को गिनें।
        प्रत्येक लय या तो सोलहवें के साथ या एक के साथ समाप्त होती है
        आठवां; समझाएं कि यह नियम क्यों देता है: (लय
        भरना $ n $ सोलहवाँ भाग) $=$ (लय भरना $ n - 1$)
        $+$ (लय भरना $ n - 2$). से शुरू $1$ लय
        एक सोलहवें के लिए और $2$ दो के लिए, टेबल बनाएं
        $ n = 8$:कितनी लय भरते हैं ए \omterm{def:g3:division:half}{आधा} टिप्पणी?
  \item जो नंबर आपको मिले --- $1, 2, 3, 5, 8, 13, 21, 34$
        --- भारतीय विद्वान हेमचन्द्र द्वारा प्रकाशित किये गये थे
        वर्ष 1150 के आसपास, बिल्कुल ऐसी ही लय गिनती में, दो
        फाइबोनैचि द्वारा पीढ़ियों पहले इन्हें इटली में लिखा गया था।
        उनके परिभाषित पैटर्न को एक वाक्य में बताएं और जांचें
        यह आपकी तालिका की अंतिम तीन प्रविष्टियों पर है।
  \item ``स्विंग'' शैली दो समान आठवें को ए से प्रतिस्थापित करती है
        लंबी-छोटी \omterm{def:g1:addition:def}{जोड़}ी: $\frac16$ तब $\frac{1}{12}$. सत्यापित करें
        का उपयोग करते हुए यह स्वैप कानूनी (समान कुल अवधि) है
        सामान्य \omterm{def:g5:division:multiple}{भाजक} $12$.
  \item ए \emph{त्रिक} तीन समान नोटों को एक में निचोड़ता है
        \omterm{def:g3:division:half}{तिमाही} टिप्पणी। प्रत्येक नोट कितना लंबा है (अपना उत्तर जांचें)।
        से गुणा करके $3$,
        \cref{prop:g7:fractions:mult})? तीन के लिए एक ही सवाल
        समान नोट भरना a \omterm{def:g3:division:half}{आधा} टिप्पणी।
  \item बाल्कन नृत्य संगीत में सात-आठ समय का उपयोग होता है: प्रत्येक माप
        कुल $\frac78$. सत्यापित करें कि बिंदीदार \omterm{def:g3:division:half}{तिमाही} $+$ \omterm{def:g3:division:half}{तिमाही}
        $+$ \omterm{def:g3:division:half}{तिमाही} इसे भरता है. फिर समझाएं क्यों \emph{नहीं}
        की संयोजन \omterm{def:g3:division:half}{आधा} नोट्स और \omterm{def:g3:division:half}{तिमाही} अकेले नोट्स,
        कितने भी हों, कभी भी सात-आठ माप भर सकते हैं।
        (आठवें में गिनें और सम तथा के बारे में सोचें \omterm{def:g3:numbers:evenodd}{विषम}.)
