\chapter{Prioridades de Operações}\label{ch:g7:priorities}

O que é $3 + 4 \times 5$? Se você calcular da esquerda para a direita, você obtém $35$;
se você multiplicar primeiro você obtém $23$. A matemática precisa de um único
resposta, então existem regras de trânsito para cálculos --- as prioridades.
Este capítulo os declara e introduz uma primeira identidade algébrica,
distributividade.

\section{As regras de prioridade}

\begin{theorem}[Regras de prioridade]\label{thm:g7:priorities:rules}
Em um cálculo sem colchetes:
\begin{enumerate}
  \item multiplicações e divisões são realizadas \emph{antes}
        adições e subtrações;
  \item operações do mesmo nível são realizadas da esquerda para a direita.
\end{enumerate}
Os colchetes substituem tudo: o que está entre colchetes é computado
primeiro, começando pelos mais íntimos.
\end{theorem}

\begin{example}\label{ex:g7:priorities:basic}
Um passo por \omterm{def:g6:lines:objects}{linha}, sublinhando (mentalmente) o que é computado:
\begin{align*}
3 + 4 \times 5 &= 3 + 20 = 23
&&\text{(multiplication first)}\\
20 - 8 + 2 &= 12 + 2 = 14
&&\text{(same level: left to right)}\\
18 \div 3 \times 2 &= 6 \times 2 = 12
&&\text{(same level: left to right)}\\
(3 + 4) \times 5 &= 7 \times 5 = 35
&&\text{(brackets first).}
\end{align*}
Observe as armadilhas: $20 - 8 + 2$ é \emph{não} $20 - 10$, e
$18 \div 3 \times 2$ é \emph{não} $18 \div 6$.
\end{example}

\begin{example}[Colchetes aninhados]\label{ex:g7:priorities:nested}
\begin{align*}
50 - 2 \times \bigl(3 + (10 - 4)\bigr)
&= 50 - 2 \times (3 + 6) && \text{(innermost bracket)}\\
&= 50 - 2 \times 9 && \text{(remaining bracket)}\\
&= 50 - 18 && \text{(multiplication)}\\
&= 32 .
\end{align*}
\end{example}

\begin{remark}[Barras de fração ocultam colchetes]
UM \omterm{def:g6:fractions:def}{fração} bar agrupa seu \omterm{def:g4:fractions:def}{numerador} e seu \omterm{def:g4:fractions:def}{denominador}, como se cada
tinha colchetes invisíveis:
\[
\frac{12 + 8}{4} = \frac{20}{4} = 5,
\qquad\text{not}\quad 12 + \frac{8}{4} = 14 .
\]
\end{remark}

\begin{method}[Computando com segurança]\label{met:g7:priorities:safe}
\begin{enumerate}
  \item Identifique a operação a ser feita primeiro (colchetes mais internos e depois
        $\times $ e $\div $, então $+$ e $-$);
  \item calcular \emph{só aquele}, e reescreva toda a expressão
        com o resultado em vigor;
  \item repita até que reste um número --- uma operação por \omterm{def:g6:lines:objects}{linha}, não
        atalhos.
\end{enumerate}
\end{method}

\section{Distributividade}

\begin{theorem}[Distributividade]\label{thm:g7:priorities:distributivity}
Para todos os números $ k $, $ a $, $ b $:
\[
k \times (a + b) = k \times a + k \times b,
\qquad
k \times (a - b) = k \times a - k \times b .
\]
\end{theorem}

\begin{proof}[Prova por uma foto]
Conte o \omterm{def:g6:measure:area}{área} de um retângulo de largura $ k $ dividido em dois menores
retângulos de comprimentos $ a $ e $ b $: como um todo é $ k \times (a+b)$;
em duas partes é $ k \times a + k \times b $. Mesmo retângulo, mesmo
\omterm{def:g6:measure:area}{área}.
\end{proof}

\begin{omfigure}
\begin{tikzpicture}[scale=0.75]
  \fill[omDef!14] (0,0) rectangle (4.2,1.8);
  \fill[omThm!14] (4.2,0) rectangle (6.4,1.8);
  \draw[thick] (0,0) rectangle (6.4,1.8);
  \draw[thick] (4.2,0) -- (4.2,1.8);
  \node[below, font=\small] at (2.1,0) {$ a $};
  \node[below, font=\small] at (5.3,0) {$ b $};
  \node[left, font=\small] at (0,0.9) {$ k $};
  \node[font=\small] at (2.1,0.9) {$ k \times a $};
  \node[font=\small] at (5.3,0.9) {$ k \times b $};
\end{tikzpicture}

{\small Distributividade como \omterm{def:g6:measure:area}{áreas}: o grande retângulo
$ k \times (a + b)$ é o \omterm{def:g1:addition:def}{soma} dos dois menores.}
\end{omfigure}

\begin{example}[Ambas as direções]\label{ex:g7:priorities:distributivity}
\emph{Expandindo} (da esquerda para a direita) facilita a matemática mental:
\[
7 \times 103 = 7 \times (100 + 3) = 700 + 21 = 721,
\qquad
6 \times 98 = 6 \times (100 - 2) = 600 - 12 = 588 .
\]
\emph{Fatoração} (da direita para a esquerda) coleta um fator comum:
\[
23 \times 12 + 23 \times 8 = 23 \times (12 + 8) = 23 \times 20 = 460 .
\]
\end{example}

\begin{example}[Traduzindo palavras em um cálculo]\label{ex:g7:priorities:words}
``O \omterm{def:g1:addition:def}{soma} de $8$ e o \omterm{def:g2:mult:def}{produto} de $5$ por $3$'' é $8 + 5 \times 3 =
23$. ``O \omterm{def:g2:mult:def}{produto} de $5$ pelo \omterm{def:g1:addition:def}{soma} de $8$ e $3$'' é
$5 \times (8 + 3) = 55$: as palavras ``o \omterm{def:g2:mult:def}{produto} de \dots\ pelo \omterm{def:g1:addition:def}{soma}
de\dots'' coloque colchetes ao redor do \omterm{def:g1:addition:def}{soma}.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g7:priorities:1}
Calcule, uma operação por \omterm{def:g6:lines:objects}{linha}:
\[
5 + 3 \times 6, \qquad
(5 + 3) \times 6, \qquad
30 - 12 \div 4, \qquad
(30 - 12) \div 4 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:priorities:2}
Calcular:
\[
24 - 6 + 2, \qquad
24 \div 6 \times 2, \qquad
24 - 6 \times 2, \qquad
24 \div (6 \times 2).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:priorities:3}
Calcule passo a passo:
\[
7 + 3 \times (10 - 6), \qquad
40 - (2 + 3) \times 4, \qquad
60 \div \bigl(2 \times (7 - 4)\bigr).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:priorities:4}
Calcule o frações (colchetes invisíveis!):
\[
\frac{18 + 6}{8}, \qquad
\frac{30}{7 - 2}, \qquad
\frac{5 \times 8}{4 + 6} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:priorities:5}
Use a distributividade para calcular mentalmente:
\[
8 \times 102, \qquad
25 \times 99, \qquad
14 \times 12 + 14 \times 8, \qquad
37 \times 45 - 37 \times 35 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:priorities:6}
Escreva o cálculo em uma \omterm{def:g6:lines:objects}{linha}, com colchetes quando necessário, e então
computar: ``o \omterm{ex:g1:subtraction:difference}{diferença} de $50$ e o \omterm{def:g2:mult:def}{produto} de $6$ por $7$'';
``o \omterm{def:g2:mult:def}{produto} do \omterm{def:g1:addition:def}{soma} de $4$ e $5$ pelo \omterm{ex:g1:subtraction:difference}{diferença} de $10$ e
$8$''.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:priorities:7}
Amir comprou $3$ cadernos em $2.50$ cada um e $2$ canetas em $1.20$ cada.
Escreva seu total em uma única expressão e depois calcule-o.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:priorities:8}
Coloque colchetes em $4 + 2 \times 5 - 1$ para obter o resultado: (a) $13$;
(b) $29$; (c) $24$. (Um dos três não precisa de colchetes ---
qual?)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:priorities:9}
São $12 + 4 \times k $ e $16 \times k $ o mesmo para cada número $ k $?
Teste com $ k = 1$ e $ k = 2$ e conclua. Qual erro de prioridade
levaria de um para o outro?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:priorities:10}
Um ingresso de cinema custa $9$ euros, mas um grupo de $ n $ as pessoas pagam um apartamento
$12$ euros mais $6$ euros por pessoa.
\begin{enumerate}
  \item Escreva uma expressão para o preço do grupo e calcule-a para
        $ n = 5$.
  \item De quantas pessoas a tarifa de grupo é mais barata que $ n $
        ingressos individuais? (Tente alguns valores de $ n $.)
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g7:priorities:11}
Usando cada um dos dígitos $1$, $2$, $3$, $4$ exatamente uma vez, os quatro
operações e colchetes à vontade, escreva expressões iguais a $24$, para
$10$, e para $1$.
\end{exercise}

\section{Problema: Multiplicando como um escriba egípcio}

\begin{problem}[{Problema de fim de semana --- truques de matemática mental de
distributividade, e a multiplicação duplicada-reduzida pela metade usada para
quatro mil anos}]
\label{pb:g7:priorities:1}
Muito antes das calculadoras --- e antes das tabelas de multiplicação serem
treinado na escola --- escribas egípcios e camponeses russos
multiplicou quaisquer dois números usando apenas \emph{duplicar, reduzir pela \omterm{def:g3:division:half}{metade} e
adicionando}. O método deles parece mágico; é de fato
distributividade (\cref{thm:g7:priorities:distributivity}) trabalhando
em plena potência. Este problema aguça sua aritmética mental, então
ensina o método antigo e depois abre o capô para ver exatamente
por que funciona.

\medskip
\noindent\textbf{Part I --- The art of not computing.}
\begin{enumerate}
  \item Calcule mentalmente, expandindo como em
        \cref{ex:g7:priorities:distributivity}:
        $7 \times 102$ e $9 \times 98$.
  \item Calcule mentalmente, fatorando:
        $17 \times 13 + 17 \times 7$ e
        $45 \times 101 - 45$.
  \item Os ``tempos $5$'' truque: multiplicar por $5$, multiplique por
        $10$ e \omterm{def:g3:division:half}{metade}. Calcular $5 \times 87$ desta forma, e
        explique por que o truque é legítimo.
  \item Mais dois truques para justificar e usar: multiplicar por $25$ por
        multiplicando por $100$ e dividindo por $4$ (calcular
        $25 \times 32$); multiplique por $99$ multiplicando por $100$
        e subtraindo uma vez (calcular $99 \times 47$).
  \item Um alerta: a distributividade espalha um fator por um
        \emph{\omterm{def:g1:addition:def}{soma}}, nunca mais de um \omterm{def:g2:mult:def}{produto}. Comparar
        $8 \times (5 + 3)$ com $8 \times 5 + 8 \times 3$, e então $8 \times (5 \times 3)$ com
        $(8 \times 5) \times (8 \times 3)$: qual \omterm{def:g3:numbers:evenodd}{par} corresponde,
        o que não acontece?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The doubling--halving method.}
Aqui está a receita para $13 \times 24$. Escreva duas colunas. Esquerda
coluna: começar em $13$ e \emph{\omterm{def:g3:division:half}{metade}}, deixando cair qualquer \omterm{def:g3:division:remainder}{restante},
até $1$: \ $13,\ 6,\ 3,\ 1$. Coluna da direita: início em $24$ e
\emph{dobro} quantas vezes: \ $24,\ 48,\ 96,\ 192$. Riscar
cada \omterm{def:g6:lines:objects}{linha} cujo \emph{esquerda} número é pare adicione o sobrevivente
números certos.
\begin{enumerate}[resume]
  \item Execute a receita de $13 \times 24$: qual \omterm{def:g6:lines:objects}{linha} é
        riscado, o que \omterm{def:g1:addition:def}{soma} permanece, e é igual
        $13 \times 24$?
  \item Multiplicar $21 \times 17$ pelo método (\omterm{def:g3:division:half}{metade} $21$, dobro
        $17$) e verifique a resposta por multiplicação ordinária.
  \item Multiplicar $16 \times 25$ by the method. How many rows
        sobreviver, e o que há de especial em uma coluna da esquerda que
        começa às $1$, $2$, $4$, $8$, $16, \dots $?
  \item Agora \emph{por que} funciona, um passo de cada vez. Justifique cada um
        igualdade com distributividade:
        \[
        13 \times 24 = (12 + 1) \times 24
        = 12 \times 24 + 24 = 6 \times 48 + 24 .
        \]
        Moral: reduzir pela \omterm{def:g3:division:half}{metade} a esquerda e duplicar a direita mantém o
        \omterm{def:g2:mult:def}{produto} --- mas quando o número da esquerda é \emph{\omterm{def:g3:numbers:evenodd}{ímpar}}, um
        a cópia do número certo se divide e deve ser armazenada.
        Qual número ocupa a primeira \omterm{def:g6:lines:objects}{linha} da tabela?
  \item Continue a contabilidade na tabela:
        $6 \times 48 = 3 \times 96$, então
        $3 \times 96 = 1 \times 192 + 96$, então
        $1 \times 192 = 192$. Colete tudo o que foi depositado
        ao longo do caminho e explique por que os números bancários são
        exatamente os números da coluna direita das \omterm{def:g6:lines:objects}{linhas} com um \omterm{def:g3:numbers:evenodd}{ímpar}
        esquerda --- as \omterm{def:g6:lines:objects}{linhas} que a receita mantém.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Under the hood: \omterm{def:g1:addition:def}{somas} of
doublings.}
\begin{enumerate}[resume]
  \item Na questão 6, as \omterm{def:g6:lines:objects}{linhas} sobreviventes valeram $1$, $4$
        e $8$ duplicações de $24$: o método escreveu silenciosamente
        $13 \times 24 = (1 + 4 + 8) \times 24$. Verifique isso
        $1 + 4 + 8 = 13$ e descubra quais duplicações ($1$, $2$,
        $4$, $8$, $16$) o método mantido para $21$ na questão 7.
  \item Escrever $25$ e $42$ como \omterm{def:g1:addition:def}{somas} de números da duplicação
        lista $1, 2, 4, 8, 16, 32$, usando cada um no máximo uma vez.
  \item Explique por que \emph{todo} número inteiro pode ser escrito
        desta forma: pegue o maior número duplicado que couber,
        subtraia, repita --- por que isso deve parar? Escrever $100$ como
        tal \omterm{def:g1:addition:def}{soma}.
  \item Os escribas do Egito fizeram exatamente isso, quatro mil
        anos atrás: multiplicar $19 \times 35$, eles fizeram um
        tabela de duplicação de $35$ ($1 \to 35$, $2 \to 70$,
        $4 \to 140$, $8 \to 280$, $16 \to 560$), escolheu as \omterm{def:g6:lines:objects}{linhas}
        somando a $19$, e resumido. Faça o cálculo deles
        caminho.
  \item Final: computadores modernos, que contam com $0$ areia
        $1$ s, multiplique essencialmente como os escribas --- por
        duplicações e acréscimos. Quais três adições seriam
        computador fazer para $13 \times 24$? (Por que cada número
        se divide em duplicações, e o que $0$ areia $1$ o que tenho que fazer
        com ele, está o problema do fim de semana do capítulo dos Poderes:
        \cref{pb:g8:powers:1}.)
\end{enumerate}
\end{problem}
