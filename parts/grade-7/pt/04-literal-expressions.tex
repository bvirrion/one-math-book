\chapter{Expressões Literais}\label{ch:g7:literal}

Uma letra em um cálculo representa um número que ainda não conhecemos, ou
para \emph{todos} números de uma vez: escrevendo $P = 4c $ dá o \omterm{def:g6:measure:perimeter}{perímetro}
de cada quadrado em uma fórmula. Este capítulo ensina a escrita
convenções, como substituir valores e como simplificar
expressões --- o início da álgebra, continuado em
\cref{ch:g8:equations}.

\section{Escrevendo com letras}

\begin{definition}[Expressão literal]\label{def:g7:literal:def}
UM \emph{expressão literal}\index{expressão literal} é um cálculo
contendo uma ou mais letras, cada uma representando um número. Escrita
convenções:
\begin{itemize}
  \item o sinal $\times $ é descartado antes de uma letra ou colchete:
        $3 \times a = 3a $, \ $ a \times b = ab $, \
        $2 \times (x + 5) = 2(x + 5)$;
  \item mas fica entre dois números: $3 \times 5$ não pode ser
        escrito $35$!
  \item $ a \times a $ está escrito $ a^2$ (``$ a $ ao quadrado''),
        $ a \times a \times a = a^3$;
  \item $1a $ está escrito $ a $, e $0a $ é $0$.
\end{itemize}
\end{definition}

\begin{example}\label{ex:g7:literal:write}
Simplifique a escrita:
\[
5 \times x + 3 \times y = 5x + 3y,
\qquad
a \times 7 = 7a \ \text{(number first)},
\qquad
x \times x \times 4 = 4x^2 .
\]
\end{example}

\begin{example}[Fórmulas são expressões literais]\label{ex:g7:literal:formulas}
Para um retângulo de comprimento $ L $ e largura $ w $:
\omterm{def:g6:measure:perimeter}{perímetro} $ P = 2(L + w)$, \omterm{def:g6:measure:area}{área} $ A = Lw $. Uma fórmula resume o
infinitos retângulos ao mesmo tempo --- esse é o ponto principal
cartas.
\end{example}

\section{Substituindo}

\begin{method}[Avaliando uma expressão]\label{met:g7:literal:substitute}
Para avaliar uma expressão para determinados valores de suas letras:
\begin{enumerate}
  \item reescreva a expressão colocando o oculto $\times $ assina de volta:
        $3a + a^2$ é $3 \times a + a \times a $;
  \item substitua cada letra pelo seu valor, mantendo colchetes ao redor dela
        se necessário;
  \item computar, respeitando as prioridades de
        \cref{ch:g7:priorities}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:literal:substitute}
Avaliar $ E = 3x + x^2$ para $ x = 5$:
\[
E = 3 \times 5 + 5 \times 5 = 15 + 25 = 40 .
\]
Avaliar $ F = 2(a + 3b)$ para $ a = 4$ e $ b = 0.5$:
\[
F = 2 \times (4 + 3 \times 0.5) = 2 \times (4 + 1.5) = 2 \times 5.5
= 11 .
\]
\end{example}

\section{Reduzindo expressões}

\begin{proposition}[Coletando termos semelhantes]\label{prop:g7:literal:reduce}
Termos contendo a mesma parte da letra podem ser coletados adicionando seus
partes numéricas (isso é distributividade,
\cref{thm:g7:priorities:distributivity}, leia de trás para frente):
\[
5x + 3x = (5 + 3)x = 8x,
\qquad
7a - 2a = 5a .
\]
Termos com partes de letras diferentes (como $3x $ e $5y $, ou $ x $ e
$ x^2$) não pode ser coletado.
\end{proposition}

\begin{example}\label{ex:g7:literal:reduce}
Reduzir $ E = 4x + 7 + 3x + 2$, classificando os termos:
\[
E = (4x + 3x) + (7 + 2) = 7x + 9 .
\]
Cuidadoso: $7x + 9$ é \emph{não} $16x $! Teste com $ x = 1$:
$4 + 7 + 3 + 2 = 16$ e $7 \times 1 + 9 = 16$ concordo, mas $16x $ faria
também seja $16$ --- teste com $ x = 2$: o original dá
$8 + 7 + 6 + 2 = 23$, e $7 \times 2 + 9 = 23$, enquanto
$16 \times 2 = 32$. A redução $7x + 9$ está certo, $16x $ está errado.
\end{example}

\begin{method}[Testando uma igualdade]\label{met:g7:literal:test}
Para verificar se duas expressões podem ser iguais para cada valor:
\begin{enumerate}
  \item substitua o mesmo valor em ambos, pelo menos duas vezes (ex.:\
        $ x = 2$ e $ x = 10$; evitar $0$ e $1$, eles escondem muitos
        \omterm{ex:g1:subtraction:difference}{diferenças});
  \item se os resultados diferirem pelo menos uma vez, as expressões são
        \emph{não} igual --- um contra-exemplo mata uma afirmação geral;
  \item se os resultados concordarem, a igualdade só é \emph{plausível}: um
        a prova precisa de álgebra (expandir, reduzir).
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:literal:test}
É $2(x + 3)$ igual a $2x + 6$ para cada $ x $? A expansão prova isso:
$2(x+3) = 2x + 2 \times 3 = 2x + 6$. É $(x + 1)^2$ igual a
$ x^2 + 1$? Teste $ x = 2$: $(2+1)^2 = 9$ mas $2^2 + 1 = 5$. Não ---
e um contra-exemplo é uma prova completa do “não”.
\end{example}

\section{Produzindo expressões}

\begin{example}[Da situação à expressão]\label{ex:g7:literal:produce}
Um táxi cobra $4$ euros mais $2$ euros por quilómetro. Para uma viagem de
$ k $ quilômetros o preço é
\[
p = 4 + 2k \quad\text{euros}.
\]
Para $ k = 7$: $ p = 4 + 14 = 18$ euros. A expressão \emph{é} o
resposta geral; a substituição produz cada um em particular.
\end{example}

\begin{example}[Truques numéricos]\label{ex:g7:literal:trick}
``Pense em um número, duplique-o, some $10$, divida por $2$, subtraia seu
número. '' Ligue para o número $ n $ e siga os passos:
\[
n \ \to\ 2n \ \to\ 2n + 10 \ \to\ \frac{2n + 10}{2} = n + 5 \ \to\
n + 5 - n = 5 .
\]
Todo mundo recebe $5$, qualquer que seja $ n $ foi --- as letras explicam o
magia.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g7:literal:1}
Simplifique a escrita:
\[
7 \times x, \qquad
y \times 3, \qquad
a \times b \times 5, \qquad
x \times x, \qquad
2 \times (a + 4).

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:literal:2}

5ab, \qquad
3(x + 2), \qquad
x^2 + 4x, \qquad
2a^3 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:literal:3}
Avaliar para $ x = 3$: \ $5x + 1$; \quad $ x^2$; \quad $2x^2 - x $;
\quad $10 - 2x $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:literal:4}
Avaliar $3a + 2b $ e $ a b + 5$ para $ a = 4$, $ b = 1.5$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:literal:5}
Reduzir:
\[
8x + 5x, \qquad
9a - 4a + a, \qquad
6y + 2 + 3y + 7, \qquad
5t + 3 - 2t - 3 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:literal:6}
Dê uma fórmula para: o \omterm{def:g6:measure:perimeter}{perímetro} de um quadrado de lado $ c $; o
\omterm{def:g6:measure:perimeter}{perímetro} de um triângulo \omterm{def:g6:shapes:triangles}{isósceles} com base $ b $ e lados iguais $ s $;
o preço de $ n $ pães em $1.20$ euros cada.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:literal:7}
Combine cada expressão com uma descrição: $2n $; $ n + 2$; $ n^2$;
$\frac n2$. Descrições: ``o \omterm{def:g3:division:half}{metade} de $ n $''; ``o dobro de $ n $'';
``a praça de $ n $''; ``dois a mais que $ n $''.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:literal:8}
Usando \cref{met:g7:literal:test}, decidir (com prova ou
contraexemplo) se essas igualdades são válidas para todos $ x $:
\[
3(x + 4) = 3x + 12; \qquad
2x + 5 = 7x; \qquad
x + x = x^2 .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:literal:9}
Um retângulo tem largura $ w $ e seu comprimento é $3$ cm a mais do que o seu
largura.
\begin{enumerate}
  \item Expresse o comprimento e depois o \omterm{def:g6:measure:perimeter}{perímetro}, em termos de $ w $ e
        reduzir.
  \item Calcule o \omterm{def:g6:measure:perimeter}{perímetro} para $ w = 5$ cm, diretamente e com o seu
        fórmula reduzida. Mesma resposta?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:literal:10}
Siga o truque de \cref{ex:g7:literal:trick} com letras: ``pense

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g7:literal:11}
A figura abaixo é um quadrado de lado $ a $ com um canto quadrado do lado
$ b $ removido ($ b $ menor que $ a $). Escreva duas expressões diferentes

\end{exercise}

\section{Problema: O oráculo do palito de fósforo}

\begin{problem}[{Problema de fim de semana --- padrões de crescimento e seus
$ n $-º termo: três fórmulas para um padrão, e o famoso
padrão que quebra na sexta etapa}]
\label{pb:g7:literal:1}
Disponha os palitos de fósforo em uma fileira de quadrados: um quadrado, depois dois, depois
três\,\dots\ De quantas correspondências o centésimo algarismo precisará?
Contá-los um por um levaria todo o fim de semana; um \emph{fórmula
em $ n $} responde instantaneamente para cada número de uma vez --- isto é
exatamente para que servem as letras
(\cref{ex:g7:literal:formulas}). Mas atenção: esse problema acaba
com um padrão célebre que funciona limpo por cinco etapas e depois
trai todos que confiaram nele.

\medskip
\noindent\textbf{Part I --- A row of squares.}
Figura $1$ é um quadrado de palito de fósforo ($4$ partidas); cada um seguindo
A figura adiciona mais um quadrado compartilhando um lado com o anterior.
\begin{enumerate}
  \item Desenhe figuras $1$ para $4$ e conte suas partidas.
  \item Explique por que cada novo quadrado custa exatamente $3$ extra
        correspondências e use isso para prever a contagem da figura
        $10$ sem nenhuma fórmula.
  \item Justifique a fórmula: figura $ n $ precisa $3n + 1$ partidas.
        (Onde é que o solitário ``$+1$'' vem?) Confira
        contra a questão 1.
  \item Amir raciocina de forma diferente: ``$4$ partidas pela primeira
        quadrado, então $3$ para cada um dos $ n - 1$ outros'', dando
        $4 + 3(n - 1)$. Expanda e reduza sua expressão
        (\cref{prop:g7:literal:reduce}): ele concorda com
        pergunta 3?
  \item Lena conta uma terceira via: ``$2n $ correspondências horizontais, e
        $ n + 1$ verticais.'' Escreva a fórmula dela, reduza-a,
        e concluir. Em seguida, responda à pergunta inicial: quantos
        correspondências para figura $100$?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Triangles, tables and going
backwards.}
\begin{enumerate}[resume]
  \item Uma fileira de triângulos palito de fósforo (cada novo triângulo se apoia
        o anterior): conte as partidas para $1$, $2$, $3$
        triângulos e justifique a fórmula $2n + 1$ para $ n $
        triângulos.
  \item Retrocedendo: com exatamente $49$ partidas, quantas
        triângulos a \omterm{def:g6:lines:objects}{linha} pode ter? (Desfaça a fórmula passo a passo
        passo.)
  \item Mesas de banquete quadradas acomodam um convidado por lado livre. Quando
        $ n $ tabelas são colocadas em uma longa \omterm{def:g6:lines:objects}{linha}, explique por que o
        fórmula de assento é $2n + 2$ e verifique se $1$, $2$,
        $3$ tabelas.
  \item Uma festa espera $30$ convidados. Quantas tabelas tem uma \omterm{def:g6:lines:objects}{linha}
        precisa? O fornecedor, em vez disso, divide as mesmas mesas em
        \emph{dois} fileiras de sete: quantos convidados cabem agora?
        Explique o ganho com a fórmula.
  \item Padrão de pontos: figura $ n $ é um L de pontos, $ n $ pontos altos
        e $ n $ pontos de largura (o canto contado uma vez). Conte o
        pontos de figuras $1$, $2$, $3$, encontre a fórmula e nomeie
        os números que produz. (Empilhados juntos, esses blocos Ls
        um quadrado --- uma história continuada em
        \cref{pb:g8:equations:1}.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Trust, test, prove.}
\begin{enumerate}[resume]
  \item Um colega afirma figura $ n $ da Parte I precisa
        $3(n + 2) - 5$ partidas. Resolva a reclamação expandindo e
        reduzindo --- nenhum teste necessário.
  \item ``A quadratura torna os números maiores, então $ n^2 \geq n $ para
        cada número $ n $.'' Teste a reivindicação para $ n = 2$, $3$,
        $10$; em seguida, encontre um número para o qual ele falha. O que fez o
        três testes bem-sucedidos comprovam, e o que o único
        prova de falha (\cref{met:g7:literal:test})?
  \item Marque pontos em um círculo e desenhe \emph{todo} \omterm{def:g6:lines:objects}{segmento}
        entre eles e conte as regiões dentro do círculo.
        Faça isso por $2$, $3$, $4$ e $5$ pontos (coloque os pontos
        de forma desigual, para que não haja três \omterm{def:g6:lines:objects}{segmentos} cruzar em um interior
        ponto). Registre as contagens. Qual fórmula o padrão
        sussurrar?
  \item Agora $6$ pontos, desenhados grandes e desiguais, contando com
        muito cuidado (há muitas regiões pequenas). Quantos
        regiões você encontra --- e o que aconteceu com o
        conjectura?
  \item A moral, em duas frases: o que seria necessário
        para \emph{provar} a conjectura da duplicação e por que
        fórmulas palito de fósforo das Partes I e II nunca em tal
        perigo? Comemore com uma previsão comprovada: quantos
        corresponde à \omterm{def:g6:lines:objects}{linha} de $2\,026$ quadrados?
\end{enumerate}
\end{problem}
