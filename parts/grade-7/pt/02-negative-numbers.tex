\chapter{Números Negativos}\label{ch:g7:negatives}

Temperaturas abaixo \omterm{def:g1:counting:zero}{zero}, andares abaixo do solo, dinheiro devido: muitos
quantidades vão abaixo $0$. Os números negativos estendem a reta numérica até o
esquerda de \omterm{def:g1:counting:zero}{zero}. Este capítulo ensina como compará-los e como adicionar
e subtraí-los --- sua multiplicação espera por
\cref{ch:g8:negprod}.

\section{Números relativos}

\begin{definition}[Números relativos]\label{def:g7:negatives:def}
UM \emph{número relativo}\index{número relativo} é feito de um sinal
($+$ ou $-$) e um \emph{distância para \omterm{def:g1:counting:zero}{zero}}: $+3$ e $-3$ estão ambos em
distância $3$ de $0$, em lados opostos. Os números positivos geralmente são
escrito sem seu sinal: $+3 = 3$. Os números $+3$ e $-3$ são
\emph{opostos}\index{opostos} um do outro.
\end{definition}

\begin{tikzpicture}[scale=1.15]
  \draw[-Stealth] (-5.6,0) -- (5.6,0);
  \foreach \x in {-5,...,5} \draw (\x,-0.08) -- (\x,0.08);

    node[midway, above, font=\small] {$3$};

    node[midway, above, font=\small] {$3$};
\end{tikzpicture}

{\small A reta numérica estendida para a esquerda: $-3$ e $+3$ são
números opostos, à mesma distância de $0$.}
\end{omfigure}

\begin{method}[Comparando números relativos]\label{met:g7:negatives:compare}
Na reta numérica, menor significa mais à esquerda. Na prática:
\begin{enumerate}
  \item um número negativo é sempre menor que um positivo:
        $-7 < 2$;
  \item entre dois positivos, a ordem habitual;
  \item entre dois negativos, aquele com o \emph{maior} distância
        para \omterm{def:g1:counting:zero}{zero} é o \emph{menor}: $-7 < -2$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:negatives:compare}
Ordene do menor para o maior: $3$; $-5$; $0$; $-1.5$; $2.8$.
Negativos primeiro, os principais negativos:
\[
-5 < -1.5 < 0 < 2.8 < 3 .
\]
\end{example}

\section{Adicionando números relativos}

Pense em um \omterm{def:g7:negatives:def}{número relativo} como um movimento ao longo da \omterm{def:g6:lines:objects}{linha}: $+5$ significa
``caminhar $5$ passos certos'', $-3$ significa ``caminhar $3$ passos \omterm{def:g3:division:remainder}{restantes}''. Adicionando
números significa encadear as caminhadas.

\begin{method}[Adicionando dois números relativos]\label{met:g7:negatives:add}
\begin{enumerate}
  \item \emph{Mesmo sinal}: some as distâncias, mantenha o sinal comum:
        $(-3) + (-4) = -7$.
  \item \emph{Sinais opostos}: subtraia a menor distância do
        maior, pegue o sinal do maior:
        $(-7) + (+2) = -5$ e $(+7) + (-2) = +5$.
  \item Opostos cancelar: $(+4) + (-4) = 0$.
\end{enumerate}
\end{method}

  \foreach \x in {-5,...,4} \draw (\x,-0.07) -- (\x,0.07);

{\small Adicionando como caminhada: começando em $0$, a caminhada $-3.5$ então o
caminhar $+2$ pousa em $(-3.5) + 2 = -1.5$.}
\end{omfigure}

\begin{example}\label{ex:g7:negatives:add}
Passo a passo, agrupando conforme conveniente:
\begin{align*}
(-8) + (+3) &= -5 && \text{(opposite signs: } 8 - 3 = 5,
\text{ sign of } 8)\\
(-2.5) + (-1.5) &= -4 && \text{(same sign: add distances)}\\
(+6) + (-9) + (+3) &= (-3) + (+3) = 0 .
\end{align*}
\end{example}

\section{Subtraindo números relativos}

\begin{theorem}[Regra de subtração]\label{thm:g7:negatives:sub}
Subtraindo um \omterm{def:g7:negatives:def}{número relativo} é o mesmo que \emph{adicionando seu
oposto}:
\[
a - b = a + (-b) .
\]
\end{theorem}

\begin{proof}[Por que funciona]
$ a - b $ é o número que somado a $ b $ dá $ a $. Verifique isso
$ a + (-b)$ faz o trabalho:
$\bigl(a + (-b)\bigr) + b = a + \bigl((-b) + b\bigr) = a + 0 = a $.
\end{proof}

\begin{example}\label{ex:g7:negatives:sub}
Toda subtração se torna uma adição, então
\cref{met:g7:negatives:add} aplica-se:
\begin{align*}
5 - 9 &= 5 + (-9) = -4, \\
(-3) - (+4) &= (-3) + (-4) = -7, \\
(-2) - (-6) &= (-2) + (+6) = +4 .
\end{align*}
A última \omterm{def:g6:lines:objects}{linha} é a famosa: \emph{subtraindo um meio negativo
adicionando}. Eliminar uma dívida deixa você mais rico!
\end{example}

\begin{example}[Diferenças de temperatura]\label{ex:g7:negatives:temperature}
Em um dia a temperatura passou de $-6$ graus para $+9$ graus. O
ascensão é o \omterm{ex:g1:subtraction:difference}{diferença}
\[
9 - (-6) = 9 + 6 = 15 \text{ degrees}.
\]
Na reta numérica: de $-6$ até $0$ é $6$ graus, de $0$ para
$9$ é $9$ mais.
\end{example}

\section{Coordenadas em todo o plano}

\begin{definition}[Coordenadas com sinais]\label{def:g7:negatives:coords}
Com dois \omterm{def:g6:lines:perp}{perpendicular} eixos graduados cruzando na origem $ O $,
cada ponto do plano recebe duas coordenadas relativas $(x, y)$:
negativo $ x $ significa à esquerda do eixo vertical, negativo $ y $ significa abaixo
o eixo horizontal.
\end{definition}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g7:negatives:1}
Dê o oposto de: $7$; \ $-3.5$; \ $0$; \ $+12$. Depois complete:
``dois números são opostos quando seu \omterm{def:g1:addition:def}{soma} é~?''.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:negatives:2}
Copie e complete com $<$ ou $>$:
\[
-4 \;?\; 1, \qquad
-4 \;?\; -7, \qquad
-2.5 \;?\; -2.4, \qquad
0 \;?\; -0.1 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:negatives:3}
Ordene do menor para o maior: $-3$; \ $2.5$; \ $-0.5$; \ $-3.1$;
\ $0$; \ $1$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:negatives:4}
Calcular:
\[
(-5) + (-9), \qquad
(-12) + (+7), \qquad
(+3.5) + (-1.5), \qquad
(-6) + (+6).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:negatives:5}
Transforme em adições e calcule:
\[
4 - 11, \qquad
(-5) - (+3), \qquad
(-7) - (-10), \qquad
0 - (-8).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:negatives:6}
Calcule agrupando de forma inteligente (opostos primeiro):
\[
(+7) + (-12) + (+3) + (-8), \qquad
(-5.5) + (+9) + (-4.5) + (+1).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:negatives:7}
A temperatura estava $-3$ graus esta manhã; subiu $8$ graus
durante o dia, depois apareceu $12$ graus à noite. Escreva um único
cálculo fornecendo a temperatura final e calcule-a.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:negatives:8}
Plotar em um sistema de coordenadas: $ A(2, 3)$; $ B(-3, 1)$; $ C(-2, -2)$;
$ D(3, -1)$; $ E(0, -3)$. Qual ponto está em um eixo?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:negatives:9}
Calcule o \omterm{ex:g1:subtraction:difference}{diferença} entre a temperatura mais alta e a mais baixa:
Moscou em janeiro, máx. $-4$ graus, min $-13$ graus; Helsinque, máx.
$2$, mínimo $-9$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:negatives:10}
Um mergulhador está em altitude $-12$ m (doze metros abaixo da superfície). Ela
sobe $7$ m, então desce $4$ m. Em que altitude ela está? Quão longe
ela ainda deve subir para alcançar a superfície?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:negatives:11}
Trama $ A(-2, 1)$, $ B(2, 1)$, $ C(2, -2)$. Encontre as coordenadas do
ponto $ D $ para que $ ABCD $ é um retângulo e calcule seu \omterm{def:g6:measure:perimeter}{perímetro}
(conte os quadrados da grade para os comprimentos dos lados).
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g7:negatives:12}
No quadrado mágico abaixo, cada \omterm{def:g6:lines:objects}{linha}, coluna e diagonal deve ter
o mesmo \omterm{def:g1:addition:def}{soma}. Copie e complete:
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|c|c|}
\hline
$?$ & $-7$ & $0$ \\
\hline
$-3$ & $-2$ & $?$ \\
\hline
$-4$ & $?$ & $?$ \\
\hline
\end{tabular}
\end{center}
(Comece encontrando a magia \omterm{def:g1:addition:def}{soma} usando a diagonal que é quase
completo.)
\end{exercise}

\section{Problema: O ano zero que nunca existiu}

\begin{problem}[{Problema de fim de semana --- números relativos no
linha do tempo da história: durações além dos limites da era, o
truque dos astrônomos e o princípio da mudança líquida}]
\label{pb:g7:negatives:1}
A reta numérica deste capítulo tem um famoso gêmeo do mundo real: o
\omterm{def:g6:lines:objects}{linha} do tempo da história, com os anos AC se estendendo para a esquerda e o
anos DC estendendo-se para a direita. Mas a \omterm{def:g6:lines:objects}{linha} dos historiadores esconde uma armadilha
isso estragou muitos cálculos: \emph{não há ano \omterm{def:g1:counting:zero}{zero}}
--- o ano 1 AC é seguido imediatamente pelo 1 DC.
calcula com temperaturas, elevadores e contas bancárias, depois repara
a história está quebrada \omterm{def:g1:counting:zero}{zero} da mesma forma que os astrônomos fazem: com relativo
números.

\medskip
\noindent\textbf{Part I --- Warm-ups, and a first trap.}
\begin{enumerate}
  \item O cume do Mont Blanc está em altitude $+4\,808$ m; o
        costa do Mar Morto em $-430$ m. Calcule o \omterm{ex:g1:subtraction:difference}{diferença}
        de altitude entre eles.
  \item Uma conta bancária contém $-45$ euros (cheque especial). É
        depósitos do proprietário $120$ euros, paga uma conta de $60$ euros,
        então depósitos $15$ euros. Escreva o saldo como um único
        cadeia de adições e calcule-a. Qual depósito único
        teria trazido o original $-45$ de volta exatamente
        $0$?
  \item Júlio César nasceu em $100$ AC e morreu em $44$ AC.
        Contando os anos AC como números negativos ($100$ AC como
        $-100$, $44$ AC como $-44$), calcular $(-44) - (-100)$: como
        quantos anos César viveu?
  \item O imperador Augusto nasceu em $63$ AC e morreu em DC
        $14$. A mesma receita dá $14 - (-63) = 77$ anos ---
        mas os historiadores insistem que ele morreu em $76$. Sua objeção:
        a contagem $77$ caminha por um ano que nunca existiu.
        Qual deles?
  \item Astrônomos reparam a \omterm{def:g6:lines:objects}{linha} do tempo renomeando: anos d.C.
        mantenha o número deles ($+14$ fica $+14$), mas $1$ AC
        torna-se $0$, $2$ AC se torna $-1$ e geralmente o ano
        $ n $ AC se torna $-(n - 1)$. Converter $63$ AC, e refaça o
        cálculo da questão 4 com os números dos astrônomos.
        A resposta agora satisfaz os historiadores?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Computing across the eras.}
De agora em diante, use a reclassificação dos astrônomos para cada
cálculo, convertendo de volta para BC/AD no final (um ano rotulado
$0$ ou negativo $-m $ é o ano dos historiadores $ m + 1$ AC).
\begin{enumerate}[resume]
  \item Converta para números de astrônomos: $1$ AC; $10$ AC; ANÚNCIO
        $476$; e $753$ AC, a lendária fundação de Roma.
  \item Quantos anos se passaram desde a fundação de Roma
        ($753$ AC) até a queda do Império Ocidental (AD $476$)?
  \item Um cometa retorna a cada $76$ anos e apareceu em
        $12$ AC. Dê aos historiadores os anos dos próximos três
        aparências, e da aparência pouco antes $12$ AC.
  \item Quantos anos se passaram desde $5$ AC a DC $5$? (O
        resposta tentadora é $10$; a \omterm{def:g6:lines:objects}{linha} do tempo diz o contrário.)
  \item Uma criança nascida em $3$ BC completou dez anos em que
        ano (AC ou DC)?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The net-change principle.}
\begin{enumerate}[resume]
  \item O \emph{distância} entre dois números na \omterm{def:g6:lines:objects}{linha} é
        quanto maior menos o menor --- sempre positivo.
        Calcule a distância entre $-7$ e $-2$, então entre
        $-3.5$ e $4.5$.
  \item Um elevador começa no chão $0$ e faz os movimentos $+3$,
        $-5$, $+2$, $-1$, $+4$. Calcule o andar final por
        agrupando habilmente (\cref{met:g7:negatives:add},
        opostos primeiro sempre que possível). Realizaria o
        se move em uma ordem diferente altera o destino? Por que?
  \item Dia após dia, a temperatura muda $+2$, $-4$, $+1$,
        $0$, $-3$, $+5$, $-2$ graus durante uma semana. Calcule o
        \emph{\omterm{def:g5:solids:net}{líquido} mudar}. Se a semana começasse às $-2$ graus,
        onde isso terminou? Indique o princípio: valor final $=$
        valor inicial $+$ (\omterm{def:g1:addition:def}{soma} de todas as mudanças).
  \item A caminhada de um caminhante a traz de volta exatamente ao ponto de partida
        altitude. O que deve o \omterm{def:g1:addition:def}{soma} de todas as suas mudanças de altitude
        ser, e por quê? Das seis alterações gravadas, cinco são
        $+4$, $-7$, $+2$, $+3$, $-5$ (em centenas de metros):
        encontre o sexto.
  \item O final, de volta a Roma. Um historiador calcula: “o
        República fugiu de $509$ AC para $27$ AC, então o Império
        para AD $476$; em tudo, de $509$ AC a DC $476$, aquilo é
        $509 + 476 = 985$ anos de estado romano. '' Refaça o
        cálculo com números de astrônomos. Qual é o
        total correto --- e qual ano, mais uma vez, causou o
        erro?
\end{enumerate}
\end{problem}
