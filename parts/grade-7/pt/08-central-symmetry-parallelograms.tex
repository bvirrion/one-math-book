\chapter{Simetria Central e Paralelogramos}\label{ch:g7:central}

Vire uma figura por \omterm{def:g3:division:half}{metade} uma volta em torno de um ponto: isto é \emph{central
simetria}, a segunda transformação do livro após as reflexões
de \cref{ch:g6:symmetry}. Sua figura estelar é o \omterm{def:g7:central:parallelogram}{paralelogramo} --- o
\omterm{def:g4:geometry:polygon}{quadrilátero} aquilo é \omterm{def:g7:central:def}{simétrico} sobre o seu próprio ponto de passagem
diagonais.

\section{Simetria central}

\begin{definition}[Simétrico em relação a um ponto]\label{def:g7:central:def}
O \emph{simétrico}\index{simétrico} de um ponto $M $ sobre um
ponto $ O $ é o ponto $ M'$ tal que $ O $ é o \emph{ponto médio} de
$[MM']$. A figura simétrica é o que o original se torna depois de um
meia volta ($180^\circ $) em volta $ O $.
\end{definition}

\begin{method}[Construindo a simétrica de um ponto]\label{met:g7:central:construct}
Para construir o \omterm{def:g7:central:def}{simétrico} $ M'$ de $ M $ sobre $ O $:
\begin{enumerate}
  \item desenhe a \omterm{def:g6:lines:objects}{linha} $(MO)$ e estendê-lo além $ O $;
  \item medir $ MO $ (régua ou compasso);
  \item lugar $ M'$ na extensão com $ OM' = OM $.
\end{enumerate}
Em papel quadriculado: conte os passos horizontais e verticais a partir $ M $ para
$ O $ e repita o \emph{mesmo} passos de $ O $ alcançar $ M'$.
\end{method}

\begin{example}[Em uma grade]\label{ex:g7:central:grid}
Se $ M $ é $3$ quadrados à esquerda e $1$ quadrado de $ O $, isso é \omterm{def:g7:central:def}{simétrico}
$ M'$ é $3$ quadrados \emph{certo} e $1$ quadrado \emph{abaixo} de $ O $:
simetria central inverte ambas as direções ao mesmo tempo (um reflexo
inverte apenas um).
\end{example}

\begin{proposition}[O que meia volta preserva]\label{prop:g7:central:preserved}
Simetria central preserva comprimentos, ângulos, \omterm{def:g6:measure:perimeter}{perímetros} e \omterm{def:g6:measure:area}{áreas}; o
imagem de uma \omterm{def:g6:lines:objects}{linha} é uma \emph{\omterm{def:g6:lines:perp}{paralelo}} \omterm{def:g6:lines:objects}{linha}, a imagem de um \omterm{def:g6:lines:objects}{segmento} é um
\omterm{def:g6:lines:perp}{paralelo} \omterm{def:g6:lines:objects}{segmento} do mesmo comprimento, a imagem de um círculo é um círculo
do mesmo \omterm{def:g3:shapes:shapes}{raio} (centrado no \omterm{def:g7:central:def}{simétrico} do centro).
\end{proposition}
\admitted

\begin{remark}
Compare com reflexões (\cref{prop:g6:symmetry:preserved}): ambos
preservar comprimentos e ângulos; mas o reflexo vira os números
(um ``b'' torna-se um ``d''), enquanto a meia volta os mantém legíveis ---
de cabeça para baixo (um ``b'' torna-se um ``q''). E diferentemente de um reflexo, um
meia volta envia cada \omterm{def:g6:lines:objects}{linha} para uma \omterm{def:g6:lines:objects}{linha} \omterm{def:g6:lines:perp}{paralelo} para isso.
\end{remark}

\begin{definition}[Centro de simetria]\label{def:g7:central:center}
Um ponto $ O $ é um \emph{centro de simetria}\index{centro de simetria}
de uma figura quando a meia volta $ O $ envia a figura exatamente
sobre si mesmo. Por exemplo, as letras S, N, Z têm um centro de
simetria; um círculo tem um (seu centro); um triângulo nunca tem um.
\end{definition}

\section{Paralelogramos}

\begin{definition}[Paralelogramo]\label{def:g7:central:parallelogram}
UM \emph{paralelogramo}\index{paralelogramo} é um \omterm{def:g4:geometry:polygon}{quadrilátero} cujo
as diagonais se cruzam em seu ponto médio comum. Esse ponto de passagem é então
um \omterm{def:g7:central:center}{centro de simetria} da figura: cada vértice é a imagem de meia volta
do oposto vértice.
\end{definition}

\begin{theorem}[Propriedades de um paralelogramo]\label{thm:g7:central:properties}
Em um \omterm{def:g7:central:parallelogram}{paralelogramo} $ ABCD $ com centro $ O $:
\begin{enumerate}
  \item lados opostos são \omterm{def:g6:lines:perp}{paralelo}: $(AB) \parallel (CD)$ e
        $(BC) \parallel (AD)$;
  \item lados opostos têm o mesmo comprimento: $ AB = CD $ e
        $ BC = AD $;
  \item ângulos opostos são iguais: $\widehat A = \widehat C $ e
        $\widehat B = \widehat D $.
\end{enumerate}
\end{theorem}

\begin{proof}
A meia volta $ O $ envia $ A $ para $ C $ e $ B $ para $ D $ (definição:
$ O $ é o ponto médio de ambas as diagonais). Então ele envia o \omterm{def:g6:lines:objects}{segmento}
$[AB]$ para o \omterm{def:g6:lines:objects}{segmento} $[CD]$; por
\cref{prop:g7:central:preserved} a imagem é \omterm{def:g6:lines:perp}{paralelo} para $[AB]$ e
do mesmo comprimento: pontos 1 e 2. Também envia o ângulo em $ A $
para o ângulo em $ C $, e os ângulos são preservados: ponto 3.
\end{proof}

\begin{theorem}[Reconhecendo um paralelogramo]\label{thm:g7:central:recognize}
UM \omterm{def:g4:geometry:polygon}{quadrilátero} (sem lados cruzados) é um \omterm{def:g7:central:parallelogram}{paralelogramo} assim que
\emph{um} das seguintes posses:
\begin{enumerate}
  \item suas diagonais têm o mesmo ponto médio (a definição);
  \item seus lados opostos são \omterm{def:g6:lines:perp}{paralelo} dois por dois;
  \item dois lados opostos são \omterm{def:g6:lines:perp}{paralelo} \emph{e} do mesmo comprimento.
\end{enumerate}
\end{theorem}
\admitted

\begin{example}\label{ex:g7:central:recognize}
$ ABCD $ tem $(AB) \parallel (CD)$ e $ AB = CD = 5$ cm. Critério 3
aplica-se: $ ABCD $ é um \omterm{def:g7:central:parallelogram}{paralelogramo} --- então sem mais nada
medindo também sabemos $ AD = BC $, $(AD) \parallel (BC)$, e que é
diagonais se cortam no meio.
\end{example}

\begin{remark}[Paralelogramos especiais]
O retângulo, losango e quadrado de \cref{ch:g6:shapes} são exatamente
o \omterm{def:g7:central:parallelogram}{paralelogramos} com uma propriedade extra: diagonais iguais
(retângulo), \omterm{def:g6:lines:perp}{perpendicular} diagonais (losango), ambos (quadrado). O
argumento de ângulo prometido em \cref{ex:g6:shapes:recognize} agora funciona:
em um \omterm{def:g7:central:parallelogram}{paralelogramo}, ângulos consecutivos são suplementares
(ângulos alternados/correspondentes com o \omterm{def:g6:lines:perp}{paralelos},
\cref{thm:g7:triangles:equalities}), então um \omterm{def:g3:shapes:rightangle}{ângulo reto} força tudo
quatro.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g7:central:1}
Em papel quadriculado, coloque $ O $, então $ M $ quatro quadrados à direita de $ O $ e um
quadrado. Construa o \omterm{def:g7:central:def}{simétrico} $ M'$ de $ M $ sobre $ O $, e o
\omterm{def:g7:central:def}{simétrico} de $ M $ em todo o \emph{\omterm{def:g6:lines:objects}{linha} de grade vertical} através $ O $.
Compare as duas imagens.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:central:2}
Desenhe um triângulo $ ABC $ e um ponto $ O $ fora dele. Construa sua imagem
$ A'B'C'$ pela meia volta $ O $. O que você pode dizer sobre
comprimentos $ A'B'$ e $ AB $? Sobre as \omterm{def:g6:lines:objects}{linhas} $(A'B')$ e $(AB)$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:central:3}
Quais dígitos, escritos como no visor de uma calculadora ($0$ para $9$), tem um
\omterm{def:g7:central:center}{centro de simetria}? Quais letras maiúsculas entre H, A, N, S, T, Z, O?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:central:4}
$ KLMN $ é um \omterm{def:g7:central:parallelogram}{paralelogramo} do centro $ O $ com $ KL = 7$ cm,
$ LM = 4$ centímetros e $\widehat K = 115^\circ $. Dê, com motivos: $ MN $,
$ NK $, $\widehat M $, e o ponto médio de $[LN]$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:central:5}
Construa um \omterm{def:g7:central:parallelogram}{paralelogramo} $ ABCD $ de suas diagonais: $ AC = 8$ centímetros e
$ BD = 5$ cm, cruzando em seus pontos médios com um ângulo de $60^\circ $
entre eles. (Desenhe as diagonais primeiro!)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:central:6}
Em um \omterm{def:g7:central:parallelogram}{paralelogramo}, um ângulo mede $115^\circ $. Usando a observação
em ângulos consecutivos suplementares, forneça os outros três ângulos.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:central:7}
Trama $ A(1, 1)$, $ B(4, 2)$ e o ponto $ O(2.5, 2.5)$. Calcule o
coordenadas das imagens $ A'$ e $ B'$ pela meia volta $ O $
(use a contagem da grade de \cref{ex:g7:central:grid}).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:central:8}
$ ABCD $ é um \omterm{def:g4:geometry:polygon}{quadrilátero} em que $ AB = CD $. Isso é suficiente para fazer
é um \omterm{def:g7:central:parallelogram}{paralelogramo}? Caso contrário, esboce um contra-exemplo (um \omterm{def:g6:shapes:triangles}{isósceles}
trapézio ajuda) e indique o que deve ser adicionado à hipótese.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:central:9}
Deixar $ ABC $ seja um triângulo e $ O $ o ponto médio de $[BC]$. Construa o
\omterm{def:g7:central:def}{simétrico} $ A'$ de $ A $ sobre $ O $. Mostre isso $ ABA'C $ é um
\omterm{def:g7:central:parallelogram}{paralelogramo}. (Qual critério de \cref{thm:g7:central:recognize} é
gratuitamente aqui?)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:central:10}
Verdadeiro ou falso, com uma razão: ``a \omterm{def:g4:geometry:polygon}{quadrilátero} cujos ângulos opostos
são iguais dois por dois é um \omterm{def:g7:central:parallelogram}{paralelogramo}''; ``um \omterm{def:g7:central:parallelogram}{paralelogramo} com
\omterm{def:g6:lines:perp}{perpendicular} diagonais é um losango''; ``um \omterm{def:g7:central:parallelogram}{paralelogramo} com um
\omterm{def:g3:shapes:rightangle}{ângulo reto} é um retângulo''.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g7:central:11}
Deixar $ ABCD $ seja qualquer \omterm{def:g4:geometry:polygon}{quadrilátero} e $ I $, $ J $, $ K $, $ L $ os pontos médios
dos seus lados $[AB]$, $[BC]$, $[CD]$, $[DA]$. Desenhe vários exemplos.
O que faz $ IJKL $ sempre parece ser? (A prova, com o ponto médio
teorema, entra \cref{ch:g8:midpoints} --- e novamente com vetores,
no \omterm{def:g6:measure:volume}{volume} do Ensino Médio.)
\end{exercise}

\section{Problema: O bolo, a mediana e o segundo impossível
centro}

\begin{problem}[{Problema de fim de semana --- meia volta no trabalho: bolo justo
cortes, um limite nas medianas e por que nenhuma forma pode ter dois centros de
simetria}]
\label{pb:g7:central:1}
A meia volta parece a mais simples das transformações --- ainda
corta bolos em \omterm{def:g3:division:half}{metades} comprovadamente justas, mede as medianas de
um triângulo, e esconde uma pequena pérola de matemática pura: um
figura limitada pode ter um \omterm{def:g7:central:center}{centro de simetria}, mas \emph{nunca
dois}. A prova usa uma descoberta que você fará na Parte I: duas
meias voltas, realizadas uma após a outra, formam um slide.

\medskip
\noindent\textbf{Part I --- Half-turn gymnastics.}
\begin{enumerate}
  \item Em papel quadriculado, pegue a origem $ O(0, 0)$ como centro.
        Usando a contagem em grade de \cref{ex:g7:central:grid},
        dê as imagens de $ M(3, 1)$, $ N(-2, 4)$ e
        $ P(0, -5)$ pela meia volta $ O $. Qual é o
        regra geral para $(x, y)$?
  \item Agora o centro é $ C(2, 1)$. Calcule a imagem de
        $ M(5, 3)$, e verifique a receita geral: cada coordenada
        da imagem é \emph{duas vezes o centro menos o
        ponto}.
  \item As cartas de jogar são projetadas para serem lidas de cabeça para baixo
        --- um simetria central, então nenhum jogador vê a carta
        ``erroneamente''. Explique por que, em um cartão com um \emph{\omterm{def:g3:numbers:evenodd}{ímpar}}
        número de pips, um pip deve ficar exatamente no topo do cartão
        centro.
  \item Duas meias voltas seguidas: pegue os centros $ O_1(2, 0)$
        e $ O_2(5, 0)$. Envie o ponto $ A(1, 1)$ através do
        meia volta $ O_1$ e, em seguida, envie a imagem através do
        meia volta $ O_2$. Compare o início e o fim. Repetir
        com $ B(0, 3)$. Que transformação única e simples fez
        as duas meias voltas equivalem a?
  \item Meça o slide da questão 4 em relação à distância
        $ O_1 O_2$ e indique a descoberta. (Compare com o
        dois \omterm{def:g6:lines:perp}{paralelo} espelhos de \cref{pb:g6:symmetry:1}: mesmo
        fenômeno, novos atores.)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Cakes and medians.}
\begin{enumerate}[resume]
  \item Deixar $ O $ ser o centro de um \omterm{def:g7:central:parallelogram}{paralelogramo}. Explique por que
        qualquer \omterm{def:g6:lines:objects}{linha} através $ O $ encontra o limite em dois pontos
        que são \omterm{def:g7:central:def}{simétrico} sobre $ O $, e por que a \omterm{def:g6:lines:objects}{linha} corta o
        \omterm{def:g7:central:parallelogram}{paralelogramo} em duas peças que são exatamente meia volta
        cópias umas das outras --- portanto iguais \omterm{def:g6:measure:area}{áreas}.
  \item Teorema do padeiro: um bolo retangular é compartilhado
        perfeitamente justo por \emph{qualquer} corte reto através de seu
        centro. Melhor: um bolo retangular tem formato retangular
        buraco (qualquer tamanho, qualquer posição, \omterm{def:g3:numbers:evenodd}{par} inclinado). Descreva o
        um corte reto que dá duas partes com igual quantidade
        bolo --- e justifique com a questão 6.
  \item Três \omterm{def:g5:solids:def}{vértices} de um \omterm{def:g7:central:parallelogram}{paralelogramo} $ ABCD $ são $ A(0, 0)$,
        $ B(6, 1)$ e $ C(8, 5)$. Calcule as coordenadas de $ D $
        e do centro $ O $.
  \item \Cref{exo:g7:central:9} construído, a partir de um triângulo $ ABC $
        e o ponto médio $ O $ de $[BC]$, o \omterm{def:g7:central:parallelogram}{paralelogramo}
        $ ABA'C $ onde $ A'$ é o \omterm{def:g7:central:def}{simétrico} de $ A $ sobre $ O $.
        Use-o, juntamente com a desigualdade triangular
        (\cref{thm:g7:triangles:inequality}) no triângulo
        $ ABA'$, para provar \emph{limite mediano}: a mediana
        $[AO]$ satisfaz
        \[
        AO < \frac{AB + AC}{2} .
        \]
        (Observe que $ AA' = 2\,AO $ e $ BA' = AC $.)
  \item Em um triângulo com $ AB = 5$ centímetros e $ AC = 7$ cm, entre
        quais dois valores devem ser o comprimento da mediana de $ A $
        mentira? (Use ambas as direções da desigualdade triangular em
        $ ABA'$.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- How many centers can a figure
have?}
\begin{enumerate}[resume]
  \item Para cada figura, diga se ela tem um \omterm{def:g7:central:center}{centro de
        simetria}, e onde: a \omterm{def:g6:lines:objects}{segmento}; uma \omterm{def:g6:lines:objects}{linha} completa; um
        triângulo \omterm{def:g6:shapes:triangles}{equilátero}; um quadrado; um círculo; as letras S
        e Z.
  \item Prove sua reivindicação para o triângulo \omterm{def:g6:shapes:triangles}{equilátero}: um
        meia volta enviando o triângulo para si mesmo formaria um \omterm{def:g3:numbers:evenodd}{par}
        os três \omterm{def:g5:solids:def}{vértices} --- mas três é \omterm{def:g3:numbers:evenodd}{ímpar}, então alguns vértice
        seria enviado para si mesmo. O que isso forçaria o
        vértice ser, e por que é impossível?
  \item Generalize a questão 12: explique por que \emph{não} \omterm{def:g4:geometry:polygon}{polígono}
        com um número \omterm{def:g3:numbers:evenodd}{ímpar} de \omterm{def:g5:solids:def}{vértices} tem um \omterm{def:g7:central:center}{centro de
        simetria}.
  \item A pérola: suponha que uma figura tivesse \emph{dois} distinto
        centros de simetria $ O_1$ e $ O_2$. Execute os dois
        meias voltas uma após a outra e use a descoberta de
        questão 5: qual movimento deve enviar a figura exatamente
        sobre si mesmo? Explique por que uma figura que cabe em uma folha
        do papel não pode sobreviver a isso, e concluo.
  \item A conclusão da questão 14 deixa uma lacuna:
        \emph{ilimitado} figuras. Verifique em um friso infinito
        --- digamos, uma faixa interminável de pegadas, esquerda, direita,
        esquerda, direita\,\dots\ --- que tem (pelo menos) dois
        diferentes centros de simetria e identificar os
        tradução que a questão 14 previu.
\end{enumerate}
\end{problem}
