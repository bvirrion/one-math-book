\chapter{Áreas e Volumes}\label{ch:g7:areas}

Retângulos medidos de grau 6, triângulos retângulos e \omterm{def:g5:solids:def}{caixas}
(\cref{ch:g6:measure}). Com uma tesoura --- corte um pedaço aqui,
cole-o ali --- as mesmas idéias agora dão o \omterm{def:g6:measure:area}{áreas} de \omterm{def:g7:central:parallelogram}{paralelogramos},
de \emph{todos} triângulos e do disco, e o \omterm{def:g6:measure:volume}{volumes} de \omterm{def:g7:areas:prism}{prismas} e
\omterm{def:g7:areas:prism}{cilindros}.

\section{Área de um paralelogramo}

\begin{theorem}[Paralelogramo]\label{thm:g7:areas:parallelogram}
UM \omterm{def:g7:central:parallelogram}{paralelogramo} com um lado de comprimento $b $ (isso é \emph{base}) e
distância $ h $ entre esse lado e o oposto (seu
\emph{altura}) tem \omterm{def:g6:measure:area}{área}
\[
A = b \times h .
\]
\end{theorem}

\begin{proof}[Prova com tesoura]
Corte o \omterm{def:g6:shapes:triangles}{em ângulo reto} triângulo saindo de um lado e cole-o
por outro: o \omterm{def:g7:central:parallelogram}{paralelogramo} torna-se um $ b \times h $ retângulo com
o mesmo \omterm{def:g6:measure:area}{área}.
\end{proof}

\begin{remark}
A altura é a \emph{\omterm{def:g6:lines:perp}{perpendicular}} distância entre as duas bases,
não o comprimento do lado inclinado! Um muito inclinado \omterm{def:g7:central:parallelogram}{paralelogramo} pode
têm lados longos e um pequeno \omterm{def:g6:measure:area}{área}.
\end{remark}

\section{Área de um triângulo}

\begin{theorem}[Triângulo]\label{thm:g7:areas:triangle}
Um triângulo com base $ b $ e altura correspondente $ h $ (o
\omterm{def:g6:lines:perp}{perpendicular} distância do oposto vértice para a \omterm{def:g6:lines:objects}{linha} de base) tem
\omterm{def:g6:measure:area}{área}
\[
A = \frac{b \times h}{2} .
\]
\end{theorem}

\begin{proof}
Duas cópias do triângulo, uma delas girada meia volta em torno do
ponto médio de um lado, encaixem-se em um \omterm{def:g7:central:parallelogram}{paralelogramo} com base $ b $
e altura $ h $ (aquilo é simetria central No trabalho,
\cref{prop:g7:central:preserved}). O triângulo é \omterm{def:g3:division:half}{metade} disso:
$\frac{bh}{2}$.
\end{proof}

\begin{tikzpicture}[scale=0.9]
  \fill[omDef!14] (0,0) -- (4,0) -- (1.4,2.1) -- cycle;
  \draw[thick] (0,0) -- (4,0) -- (1.4,2.1) -- cycle;
  \fill[omThm!12] (4,0) -- (1.4,2.1) -- (5.4,2.1) -- cycle;
  \draw[thick, omThm, dashed] (4,0) -- (5.4,2.1) -- (1.4,2.1);
  \draw[dashed] (1.4,0) -- (1.4,2.1);
  \draw[thin] (1.4,0) ++(-0.2,0) -- ++(0,0.2) -- ++(0.2,0);
  \node[below, font=\small] at (2,0) {$ b $};
  \node[right, font=\small] at (1.42,1.0) {$ h $};
\end{tikzpicture}

{\small Um triângulo e sua cópia semi-voltada (tracejada) formam uma
\omterm{def:g7:central:parallelogram}{paralelogramo}: o triângulo \omterm{def:g6:measure:area}{área} é $\frac{b\times h}{2}$. Qualquer um dos
três lados podem servir de base --- com \emph{próprio} altura.}
\end{omfigure}

\begin{example}\label{ex:g7:areas:triangle}
Um triângulo tem uma base de $9$ cm e as medidas de altura correspondentes
$4$ cm:
\[
A = \frac{9 \times 4}{2} = \frac{36}{2} = 18 \text{ cm}^2 .
\]
Por um \emph{certo} triângulo, os dois catetos são uma base e sua altura:
recuperamos a fórmula de \cref{prop:g6:measure:formulas}.
\end{example}

\section{Área de um disco}

\begin{theorem}[Disco]\label{thm:g7:areas:disk}
Um disco de \omterm{def:g3:shapes:shapes}{raio} $ r $ tem \omterm{def:g6:measure:area}{área}
\[
A = \pi r^2 .
\]
\end{theorem}

\begin{proof}[Ideia de prova]
Corte o disco em muitas fatias finas e iguais e coloque-as da cabeça para a cauda:
eles quase formam um \omterm{def:g7:central:parallelogram}{paralelogramo} de altura $ r $ cuja base é \omterm{def:g3:division:half}{metade} o
circunferência, $\pi r $ (\cref{prop:g6:measure:circle}). Isso é \omterm{def:g6:measure:area}{área} é
perto de $\pi r \times r = \pi r^2$, e a aproximação torna-se
perfeito à medida que as fatias ficam mais finas. Uma prova completa precisa de integral
cálculo, do \omterm{def:g6:measure:volume}{volume} do Ensino Médio.
\end{proof}

    node[midway, right, font=\small] {$\approx r $};
  \node[below, font=\small] at (2.5,-0.1) {base $\approx \pi r $};
  \end{scope}
\end{tikzpicture}

{\small Fatiar o disco e desenrolar as fatias: quase um
\omterm{def:g7:central:parallelogram}{paralelogramo} de base $\pi r $ (\omterm{def:g3:division:half}{metade} a borda) e altura $ r $.}
\end{omfigure}

\begin{example}\label{ex:g7:areas:disk}
Um tampo de mesa redondo tem \omterm{def:g3:shapes:shapes}{raio} $60$ cm. Isso é \omterm{def:g6:measure:area}{área} é
$\pi \times 60^2 = 3600\pi \approx 11\,310$ cm $^2$, um pouco mais do que
$1.1$ eu $^2$. Assista o \omterm{ex:g1:subtraction:difference}{diferença}: o \emph{\omterm{def:g6:measure:perimeter}{perímetro}} $2\pi r $ usa
$ r $, o \emph{\omterm{def:g6:measure:area}{área}} $\pi r^2$ usa $ r^2$.
\end{example}

\section{Prismas e cilindros}

\begin{definition}[Prisma, cilindro]\label{def:g7:areas:prism}
UM \emph{prisma}\index{prisma} é um \omterm{def:g5:solids:def}{sólido} com dois idênticos \omterm{def:g6:lines:perp}{paralelo}
poligonal \omterm{def:g5:solids:def}{rostos} (o \emph{bases}) unidos por retângulos; um
\emph{cilindro}\index{cilindro} é o mesmo com discos como bases. O
\emph{altura} é a distância entre as duas bases.
\end{definition}

\begin{theorem}[Volumes]\label{thm:g7:areas:volume}
Por um \omterm{def:g7:areas:prism}{prisma} ou um \omterm{def:g7:areas:prism}{cilindro} com base \omterm{def:g6:measure:area}{área} $ B $ e altura $ h $:
\[
V = B \times h .
\]
\end{theorem}
\admitted

\begin{example}\label{ex:g7:areas:volume}
Uma tenda é um \omterm{def:g7:areas:prism}{prisma} deitado de lado: suas bases são triângulos de base
$2$ m e altura $1.5$ m, e a tenda é $3$ m longo.
\begin{enumerate}
  \item Base \omterm{def:g6:measure:area}{área}: $ B = \frac{2 \times 1.5}{2} = 1.5$ eu $^2$.
  \item \omterm{def:g6:measure:volume}{Volume}: $ V = B \times h = 1.5 \times 3 = 4.5$ eu $^3$.
\end{enumerate}
Uma lata de \omterm{def:g3:shapes:shapes}{raio} $4$ cm e altura $11$ cm:
$ V = \pi \times 4^2 \times 11 = 176\pi \approx 553$ cm $^3$, aproximadamente
\omterm{def:g3:division:half}{metade} um litro.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g7:areas:1}
Calcule o \omterm{def:g6:measure:area}{área} de um \omterm{def:g7:central:parallelogram}{paralelogramo} com base $8$ cm e altura
$5$ cm; de outro com base $6.5$ m e altura $4$ m.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:areas:2}
UM \omterm{def:g7:central:parallelogram}{paralelogramo} tem lados $10$ centímetros e $6$ cm, e a altura relativa
para o $10$ cm medidas básicas $4$ cm. Calcule seu \omterm{def:g6:measure:area}{área}. Por que é que
responda não $60$ cm $^2$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:areas:3}
Calcule o \omterm{def:g6:measure:area}{área} de um triângulo com base $12$ cm e altura $7$ cm;
de um triângulo retângulo com pernas $9$ e $6$; de um triângulo com base
$5.5$ m e altura $2$ m.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:areas:4}
Calcule o \omterm{def:g6:measure:area}{área} de um disco de \omterm{def:g3:shapes:shapes}{raio} $5$ cm, então de meio disco de
\omterm{def:g3:shapes:shapes}{raio} $10$ cm (valores exatos com $\pi $, depois \omterm{def:g4:numbers:round}{arredondado} para a unidade).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:areas:5}
Calcule o \omterm{def:g6:measure:perimeter}{perímetro} \emph{e} o \omterm{def:g6:measure:area}{área} de um disco de \omterm{def:g3:shapes:shapes}{raio} $3$ cm.
Qual das duas respostas está em cm e qual está em cm $^2$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:areas:6}
Calcule o \omterm{def:g6:measure:volume}{volume} de um \omterm{def:g7:areas:prism}{prisma} com base \omterm{def:g6:measure:area}{área} $24$ cm $^2$ e altura
$10$ cm; de um \omterm{def:g7:areas:prism}{cilindro} de \omterm{def:g3:shapes:shapes}{raio} $3$ cm e altura $8$ centímetros (exato,
então \omterm{def:g4:numbers:round}{arredondado}).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:areas:7}
Um triângulo tem \omterm{def:g6:measure:area}{área} $28$ cm $^2$ e base $8$ cm. Encontre o
altura correspondente, escrevendo a equação primeiro.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:areas:8}
Desenhe qualquer triângulo e meça cuidadosamente os três \omterm{def:g3:numbers:evenodd}{pares} base-altura.
Calcular $\frac{b \times h}{2}$ para cada \omterm{def:g3:numbers:evenodd}{par}: os três resultados devem
concordo (até erro de medição). Por que?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:areas:9}
Um jardim em forma de trapézio pode ser dividido, por uma diagonal, em duas
triângulos compartilhando a mesma altura $ h = 20$ m, com bases $30$ eu e
$18$ m. Calcule seu total \omterm{def:g6:measure:area}{área}. Você consegue adivinhar uma fórmula geral para
o trapézio?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:areas:10}
Um vaso cilíndrico de \omterm{def:g3:shapes:shapes}{raio} $6$ cm contém água a uma altura de
$15$ cm. Toda a água é despejada em um vazio \omterm{def:g5:solids:def}{caixa}
$18$ cm $\times $ $12$ cm na base. Que altura da água é alcançada em
o \omterm{def:g5:solids:def}{caixa}? (Mesmo \omterm{def:g6:measure:volume}{volume}, base diferente.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:areas:11}
Uma pizza de \omterm{def:g4:geometry:circle}{diâmetro} $30$ cm custos $9$ euros; um dos \omterm{def:g4:geometry:circle}{diâmetro} $40$ centímetros
custos $14$ euros. Calcule os dois \omterm{def:g6:measure:area}{áreas} e o preço por $100$
cm $^2$. Qual pizza é o melhor negócio?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g7:areas:12}
As duas pernas de um triângulo retângulo medir $6$ centímetros e $8$ cm, e seu
medidas de hipotenusa $10$ cm. Calcule seu \omterm{def:g6:measure:area}{área} com as pernas, então use
isso \omterm{def:g6:measure:area}{área} para encontrar a altura relativa ao \emph{hipotenusa}.
\end{exercise}

\section{Problema: O quadrado que desaparece}

\begin{problem}[{Problema de fim de semana --- uma famosa ``prova'' de que
$64 = 65$, desmascarados por áreas e encostas, com pizzas e anéis para
sobremesa}]
\label{pb:g7:areas:1}
Um mágico corta um $8 \times 8$ quadrado em quatro pedaços, slides
eles ao redor e os remonta em um $5 \times 13$ retângulo.
A audiência conta: $8 \times 8 = 64$, mas
$5 \times 13 = 65$ --- uma unidade quadrada apareceu do nada
ar! Este capítulo \omterm{def:g6:measure:area}{área} fórmulas são exatamente as ferramentas que
pegue o truque. O problema então coloca honesto \omterm{def:g6:measure:area}{área} raciocínio para
trabalho: em trapézios, em pilhas inclinadas, em economia de pizza --- e
finalmente retorna ao mágico por um segundo, \omterm{def:g3:numbers:evenodd}{par} estranho
desempenho.

\medskip
\noindent\textbf{Part I --- The trick, performed and unmasked.}
As quatro peças do $8 \times 8$ quadrados são: dois certo
triângulos com pernas $8$ e $3$, e dois trapézios retos com
\omterm{def:g6:lines:perp}{paralelo} lados $5$ e $3$ e altura $5$.
\begin{enumerate}
  \item Calcule o \omterm{def:g6:measure:area}{área} do quadrado original e do \omterm{def:g6:measure:area}{área} de
        o retângulo reivindicado. Declare o escândalo em uma \omterm{def:g6:lines:objects}{linha}.
  \item Calcule o \omterm{def:g6:measure:area}{área} de cada uma das quatro peças
        (\cref{thm:g7:areas:triangle}; para os trapézios, corte
        ou usar \cref{exo:g7:areas:9}).
  \item Adicione os quatro \omterm{def:g6:measure:area}{áreas}. Qual das duas figuras --- quadrado
        ou retângulo --- as peças podem realmente preencher?
  \item No arranjo retangular, a hipotenusa de um triângulo
        (aumentando $3$ entre $8$) deve continuar um
        borda inclinada do trapézio (ascendente $2$ entre $5$) em um
        reta ``diagonal''. Teste o alinhamento: são $3$ entre
        $8$ e $2$ entre $5$ a mesma inclinação
        (\cref{thm:g7:prop:cross})?
  \item Explique onde o sexagésimo quinto quadrado se esconde: qual é a forma
        a lacuna ao longo da falsa diagonal, qual é a sua exata \omterm{def:g6:measure:area}{área},
        e por que o olho não percebe? (Qual a largura, em média, de um
        pedaço disso \omterm{def:g6:measure:area}{área} esticado sobre o retângulo
        diagonal, aproximadamente $13$ unidades de comprimento?)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Honest \omterm{def:g6:measure:area}{área} reasoning.}
\begin{enumerate}[resume]
  \item Desenhe dois \omterm{def:g6:lines:objects}{linhas} paralelas $4$ cm de distância, uma base de $6$ cm em
        um deles, e três triângulos diferentes nessa base
        com \omterm{def:g5:solids:def}{vértices} em vários pontos da outra \omterm{def:g6:lines:objects}{linha}. Calcular
        os três \omterm{def:g6:measure:area}{áreas}. Por que são todos iguais, porém o ápice
        desliza ao longo de sua \omterm{def:g6:lines:objects}{linha}?
  \item Prove a fórmula do trapézio em geral: um trapézio com
        \omterm{def:g6:lines:perp}{paralelo} lados $ B $ e $ b $ e altura $ h $, corte ao longo de um
        diagonal, dá dois triângulos da mesma altura $ h $.
        Concluir:
        \[
        A = \frac{(B + b) \times h}{2} .
        \]
  \item Aplique-o a um trapézio com $ B = 9$ cm, $ b = 5$ cm,
        $ h = 4$ cm.
  \item Os carpinteiros usam a mesma fórmula disfarçada: \omterm{def:g6:measure:area}{área} $=$
        (média dos dois \omterm{def:g6:lines:perp}{paralelo} lados) $\times $ altura.
        Explique por que esta é a mesma regra e recalcule a questão
        8 dessa forma.
  \item Incline uma pilha organizada de cartas de baralho em uma pilha inclinada.
        Explique por que \omterm{def:g6:measure:volume}{volume} não mudou, e o que isso
        diz sobre a altura no \omterm{def:g7:areas:prism}{prisma} fórmula
        $ V = B \times h $ (\cref{thm:g7:areas:volume}): qual
        altura deve ser considerada para uma pilha inclinada --- a inclinada
        comprimento ou vertical? (Comparar
        \cref{exo:g7:areas:2}, uma dimensão acima.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Rings, pizzas, and the magician's
return.}
\begin{enumerate}[resume]
  \item Um lago circular de \omterm{def:g3:shapes:shapes}{raio} $3$ m fica no centro de um
        gramado circular de \omterm{def:g3:shapes:shapes}{raio} $5$ m. Calcule o \omterm{def:g6:measure:area}{área} do
        anel de grama (exato com $\pi $, depois \omterm{def:g4:numbers:round}{arredondado} para
        eu $^2$).
  \item Encontre o \omterm{def:g3:shapes:shapes}{raio} do único disco cujo \omterm{def:g6:measure:area}{área} é igual
        aquele anel é \omterm{def:g6:measure:area}{área} exatamente. (Os três \omterm{def:g3:shapes:shapes}{raios} que você tem agora
        formam um famoso triplo, cuja história é contada em
        \cref{ch:g8:pythagoras}.)
  \item O que é mais pizza: uma pizza de \omterm{def:g4:geometry:circle}{diâmetro} $40$ cm, ou
        duas pizzas de \omterm{def:g4:geometry:circle}{diâmetro} $28$ cm? Calcule e compare.
  \item Para \emph{dobro} uma pizza \omterm{def:g6:measure:area}{área}, aproximadamente por qual fator
        deve seu \omterm{def:g3:shapes:shapes}{raio} crescer? Verifique o candidato $1.4$ (lembre-se
        isso \omterm{def:g6:measure:area}{áreas} siga o \emph{quadrado} da escalação
        fator, \cref{pb:g6:propdata:1}) e diga onde
        fator exato --- o número cujo quadrado é $2$ --- faz
        sua entrada oficial
        (\cref{ex:g8:pythagoras:diagonal}).
  \item O mágico retorna: ele corta um $13 \times 13$ quadrado
        com os números $5$, $8$, $13$ desempenhando os papéis que
        $3$, $5$, $8$ jogado antes, e remonta um
        $8 \times 21$ retângulo. Calcule ambos \omterm{def:g6:measure:area}{áreas}: o que
        desaparece desta vez? Os números $3, 5, 8, 13, 21$ são
        termos consecutivos da sequência de
        \cref{pb:g7:fractions:1} --- indique o padrão do
        ganhos e perdas do mágico, e a moral: por que \omterm{def:g6:measure:area}{áreas}
        nunca minta?
\end{enumerate}
\end{problem}
