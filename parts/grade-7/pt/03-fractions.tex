\chapter{Frações: Comparando e Adicionando}\label{ch:g7:fractions}

6ª série introduzida frações como ações e como exato \omterm{def:g3:division:remainder}{quocientes}
(\cref{ch:g6:fractions}). Este capítulo aprende a \emph{calcular} com
eles: reconhecendo igual frações, comparando, somando e subtraindo
--- por enquanto com amigável \omterm{def:g4:fractions:def}{denominadores}; o caso geral entra
\cref{ch:g9:fractions}.

\section{Frações iguais}

\begin{theorem}[Frações iguais]\label{thm:g7:fractions:equal}
UM \omterm{def:g6:fractions:def}{fração} permanece inalterado quando \omterm{def:g4:fractions:def}{numerador} e \omterm{def:g4:fractions:def}{denominador} são
multiplicado ou dividido pelo mesmo número diferente de \omterm{def:g1:counting:zero}{zero}:
\[
\frac{a}{b} = \frac{a \times k}{b \times k} .
\]
\end{theorem}
\admitted

\begin{example}\label{ex:g7:fractions:equal}
$\dfrac{3}{4} = \dfrac{3 \times 5}{4 \times 5} = \dfrac{15}{20}$ e
na outra direção $\dfrac{42}{30} = \dfrac{42 \div 6}{30 \div 6}
= \dfrac{7}{5}$ (simplificado por $6$).

Para testar se $\dfrac{8}{12}$ e $\dfrac{10}{15}$ são iguais,
simplifique ambos: $\dfrac{8}{12} = \dfrac{2}{3}$ e
$\dfrac{10}{15} = \dfrac{2}{3}$ --- sim, igual.
\end{example}

\section{Comparando frações}

\begin{method}[Comparando duas frações]\label{met:g7:fractions:compare}
\begin{enumerate}
  \item \emph{Mesmo \omterm{def:g4:fractions:def}{denominador}}: compare o \omterm{def:g4:fractions:def}{numeradores}:
        $\frac{5}{9} < \frac{7}{9}$.
  \item \emph{Um \omterm{def:g4:fractions:def}{denominador} é um múltiplo do outro}: reescrever o
        mais grosseiro \omterm{def:g6:fractions:def}{fração}, então compare:
        $\frac{5}{6}$ contra $\frac{7}{12}$: $\frac56 = \frac{10}{12} >
        \frac{7}{12}$.
  \item \emph{Compare com $1$ ou $\frac12$} quando possível:
        $\frac{9}{8} > 1 > \frac{7}{9}$ resolve
        $\frac98$ contra $\frac79$ imediatamente.
\end{enumerate}
\end{method}

\section{Adição e subtração}

\begin{theorem}[Mesmo denominador]\label{thm:g7:fractions:add}
Frações com o mesmo \omterm{def:g4:fractions:def}{denominador} são adicionados (ou subtraídos) por
adicionando (ou subtraindo) o \omterm{def:g4:fractions:def}{numeradores}:
\[
\frac{a}{d} + \frac{b}{d} = \frac{a + b}{d},
\qquad
\frac{a}{d} - \frac{b}{d} = \frac{a - b}{d} .
\]
\end{theorem}

\begin{proof}[Por que]
$ a $ partes de tamanho $\frac1d $ mais $ b $ partes do mesmo tamanho fazem
$ a + b $ peças desse tamanho.
\end{proof}

\begin{method}[Denominadores diferentes]\label{met:g7:fractions:add}
Quando um \omterm{def:g4:fractions:def}{denominador} é um múltiplo do outro:
\begin{enumerate}
  \item reescrever o \omterm{def:g6:fractions:def}{fração} com o menor \omterm{def:g4:fractions:def}{denominador} então ambos têm
        o maior (\cref{thm:g7:fractions:equal});
  \item adicione ou subtraia o \omterm{def:g4:fractions:def}{numeradores};
  \item simplifique o resultado, se possível.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:fractions:add}
Calcular $\dfrac{3}{4} + \dfrac{5}{8}$, passo a passo:
\begin{enumerate}
  \item $8$ é um múltiplo de $4$: reescrever
        $\dfrac34 = \dfrac{3 \times 2}{4 \times 2} = \dfrac{6}{8}$;
  \item adicionar: $\dfrac{6}{8} + \dfrac{5}{8} = \dfrac{11}{8}$;
  \item nada para simplificar: o resultado é $\dfrac{11}{8}$, ou seja\
        $1 + \dfrac38$.
\end{enumerate}
Outro: $2 - \dfrac{4}{5} = \dfrac{10}{5} - \dfrac{4}{5} =
\dfrac{6}{5}$ (escreva o número inteiro sobre o \omterm{def:g4:fractions:def}{denominador} $5$
primeiro).
\end{example}

\begin{example}[Uma armadilha clássica]\label{ex:g7:fractions:trap}
$\dfrac12 + \dfrac13$ é \emph{não} $\dfrac{2}{5}$! Adicionando \omterm{def:g4:fractions:def}{numeradores}
e \omterm{def:g4:fractions:def}{denominadores} separadamente está errado --- a resposta $\frac25$ seria
menor que $\frac12$, absurdo quando \emph{adicionando} algo positivo
para $\frac12$. (O correto \omterm{def:g1:addition:def}{soma}, $\frac56$, precisa de um comum \omterm{def:g4:fractions:def}{denominador}
$6$; o método geral está em \cref{ch:g9:fractions}.)
\end{example}

\section{Multiplicando uma fração por um número}

\begin{proposition}[Fração vezes um número]\label{prop:g7:fractions:mult}
Para um número $ k $:
\[
k \times \frac{a}{b} = \frac{k \times a}{b} .
\]
Tirando $\frac ab $ \emph{de} uma quantidade significa multiplicar por
$\frac ab $.
\end{proposition}
\admitted

\begin{example}\label{ex:g7:fractions:mult}
$6 \times \dfrac{2}{3} = \dfrac{12}{3} = 4$: seis vezes dois terços é
quatro inteiros. E ``$\frac34$ de $60$ euros'' é
$\frac34 \times 60 = \frac{180}{4} = 45$ euros --- a mesma resposta que
dividindo por $4$ então multiplicando por $3$
(\cref{met:g6:fractions:ofquantity}).
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g7:fractions:1}
Completo: $\dfrac{2}{5} = \dfrac{?}{20}$; \quad
$\dfrac{18}{24} = \dfrac{3}{?}$; \quad
$\dfrac{7}{3} = \dfrac{28}{?}$; \quad
$\dfrac{?}{9} = \dfrac{20}{36}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:fractions:2}
Simplifique o máximo possível:
$\dfrac{12}{16}$; \quad $\dfrac{30}{45}$; \quad $\dfrac{27}{36}$;
\quad $\dfrac{48}{12}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:fractions:3}
São $\dfrac{15}{35}$ e $\dfrac{21}{49}$ igual? E $\dfrac{16}{28}$
e $\dfrac{20}{36}$? Justifique simplificando.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:fractions:4}
Compare (escreva o raciocínio, não apenas a resposta):
\[
\frac{7}{11} \text{ and } \frac{9}{11}; \qquad
\frac{5}{6} \text{ and } \frac{11}{18}; \qquad
\frac{13}{12} \text{ and } \frac{19}{20}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:fractions:5}
Calcule e simplifique se possível:
\[
\frac{5}{9} + \frac{2}{9}, \qquad
\frac{11}{7} - \frac{4}{7}, \qquad
\frac{5}{8} + \frac{7}{8} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:fractions:6}
Calcular usando \cref{met:g7:fractions:add}:
\[
\frac{2}{3} + \frac{5}{6}, \qquad
\frac{7}{10} - \frac{2}{5}, \qquad
\frac{3}{4} + \frac{5}{12} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:fractions:7}
Calcular:
\[
3 - \frac{5}{4}, \qquad
1 + \frac{3}{8}, \qquad
2 - \frac{7}{6} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:fractions:8}
Calcular:
\[
5 \times \frac{3}{10}, \qquad
\frac{7}{8} \times 4, \qquad
\frac{2}{3} \text{ of } 45 .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:fractions:9}
Marc comeu $\frac14$ de uma pizza e Julie $\frac38$ da mesma pizza.
O que \omterm{def:g6:fractions:def}{fração} da pizza eles comeram juntos? O que \omterm{def:g6:fractions:def}{fração} é
saiu?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:fractions:10}
Uma garrafa contém $\frac34$ L de suco. Lea derrama $\frac16$ eu
duas vezes. Quanto suco resta? (Comum \omterm{def:g4:fractions:def}{denominador}: $12$.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:fractions:11}
Explique, com o argumento de \cref{ex:g7:fractions:trap}, por que
$\frac35 + \frac12$ não pode ser $\frac{4}{7}$, sem calcular o
correto \omterm{def:g1:addition:def}{soma}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g7:fractions:12}
Calcule o \omterm{def:g1:addition:def}{soma}
\[
\frac12 + \frac14 + \frac18 + \frac{1}{16}
\]
passo a passo (da esquerda para a direita). Observe o padrão da parcial \omterm{def:g1:addition:def}{somas}:
que número eles se aproximam à medida que mais e mais termos
$\frac{1}{32}, \frac{1}{64}, \dots $ são adicionados?
\end{exercise}

\section{Problema: As frações escondidas em um compasso de música}

\begin{problem}[{Problema de fim de semana --- os valores das notas são frações
que devem somar exatamente, a contagem rítmica esconde uma sequência famosa,
e a paridade governa o ritmo dos Balcãs}]
\label{pb:g7:fractions:1}
A música é escrita em frações. UM \emph{nota inteira} dura $1$
medida (no tempo comum ``quatro-quatro''); um \emph{\omterm{def:g3:division:half}{metade} observação}
dura $\frac12$; então vem o \emph{\omterm{def:g3:division:half}{trimestre} observação} $\frac14$, o
\emph{colcheia} $\frac18$ e o \emph{semicolcheia}
$\frac{1}{16}$. Uma lei inviolável: \emph{as durações dentro de um
medida deve somar exatamente o total da medida} --- $1$ em
quatro-quatro vezes. Cada questão abaixo é a aritmética deste capítulo
(\cref{thm:g7:fractions:add}, \cref{met:g7:fractions:add}) definido como
música; no caminho você encontrará uma sequência de contagem descoberta por
Estudiosos indianos há mil anos.

\medskip
\noindent\textbf{Part I --- Notes that must add up.}
\begin{enumerate}
  \item Verifique se cada uma dessas medidas é legal em quatro ou quatro
        tempo: (a) \omterm{def:g3:division:half}{metade} $+$ \omterm{def:g3:division:half}{trimestre} $+$ \omterm{def:g3:division:half}{trimestre}; (b) \omterm{def:g3:division:half}{trimestre} $+$
        oitavo $+$ oitavo $+$ \omterm{def:g3:division:half}{metade}.
  \item Um copista escreveu: \omterm{def:g3:division:half}{metade} $+$ \omterm{def:g3:division:half}{trimestre} $+$ oitavo. O que
        \omterm{def:g6:fractions:def}{fração} da medida está escrito, e qual único
        nota completa isso?
  \item Quantas semicolcheias preenchem um compasso inteiro? Como
        muitas semicolcheias formam uma colcheia?
  \item UM \emph{ponto} depois de uma nota adicionar \omterm{def:g3:division:half}{metade} o valor da nota: a
        pontilhado \omterm{def:g3:division:half}{metade} é $\frac12 + \frac14$, um pontilhado \omterm{def:g3:division:half}{trimestre} é
        $\frac14 + \frac18$. Calcule ambos os valores.
  \item Uma valsa é escrita em três ou quatro tempos: cada compasso deve
        total $\frac34$. Verifique se há um único ponto \omterm{def:g3:division:half}{metade} preenche um
        medida de valsa e anote outras duas valsas legais
        medidas usando apenas \omterm{def:g3:division:half}{trimestres} e oitavos.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The copyist's workshop.}
\begin{enumerate}[resume]
  \item Uma medida quatro-quatro contém: pontilhado \omterm{def:g3:division:half}{trimestre} $+$ oitavo
        $+$ \omterm{def:g3:division:half}{trimestre}. O que está faltando?
  \item O que dura mais, um oitavo pontilhado ou um \omterm{def:g3:division:half}{trimestre} observação?
        Compare por meio de dezesseis décimos.
  \item UM \emph{gravata} cola durações juntas em um longo
        som: um \omterm{def:g3:division:half}{metade} nota amarrada a uma oitava dura quanto tempo? Como
        muitas semicolcheias é isso?
  \item As notas de um compasso somam $\frac{11}{16}$; o
        \omterm{def:g3:division:remainder}{restante} é o silêncio (um \emph{descansar}). Quanto tempo dura
        descansar?
  \item Um baterista preenche um \omterm{def:g3:division:half}{trimestre} observação ($\frac{4}{16}$ do
        compasso) usando apenas oitavos ($\frac{2}{16}$) e
        décimos sextos ($\frac{1}{16}$), a ordem importa: para
        instância oitavo - décimo sexto - décimo sexto, ou
        décimo sexto - oitavo - décimo sexto. Verifique se ambos os exemplos
        são legais, então liste \emph{todos} os ritmos possíveis.
        Quantos são?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- A thousand-year-old sequence, and
an \omterm{def:g3:numbers:evenodd}{ímpar} measure.}
\begin{enumerate}[resume]
  \item Vamos contar os ritmos do baterista para notas mais longas.
        Cada ritmo termina com uma semicolcheia ou com uma
        oitavo; explique por que isso dá a regra: (ritmos
        recheio $ n $ décimos sextos) $=$ (ritmos preenchendo $ n - 1$)
        $+$ (ritmos preenchendo $ n - 2$). A partir de $1$ ritmo
        por um décimo sexto e $2$ para dois, construa a mesa até
        $ n = 8$: quantos ritmos preenchem um \omterm{def:g3:division:half}{metade} observação?
  \item Os números que você encontrou --- $1, 2, 3, 5, 8, 13, 21, 34$
        --- foram publicados pelo estudioso indiano Hemachandra
        por volta do ano 1150, contando exatamente esses ritmos, dois
        gerações antes de Fibonacci escrevê-los na Itália.
        Declare seu padrão definidor em uma frase e verifique
        nas três últimas entradas da sua tabela.
  \item O estilo ``Swing'' substitui duas colcheias iguais por um
        \omterm{def:g3:numbers:evenodd}{par} longo-curto: $\frac16$ então $\frac{1}{12}$. Verifique
        que a troca é legal (mesma duração total), usando o
        comum \omterm{def:g4:fractions:def}{denominador} $12$.
  \item UM \emph{trigêmeo} espreme três notas iguais em uma
        \omterm{def:g3:division:half}{trimestre} observação. Quanto tempo dura cada nota (verifique sua resposta
        multiplicando-o por $3$,
        \cref{prop:g7:fractions:mult})? A mesma pergunta para três
        notas iguais preenchendo um \omterm{def:g3:division:half}{metade} observação.
  \item A música de dança dos Balcãs usa o tempo sete-oito: cada compasso
        totais $\frac78$. Verifique se o ponto é pontilhado \omterm{def:g3:division:half}{trimestre} $+$ \omterm{def:g3:division:half}{trimestre}
        $+$ \omterm{def:g3:division:half}{trimestre} preenche. Então explique por que \emph{não}
        combinação de \omterm{def:g3:division:half}{metade} notas e \omterm{def:g3:division:half}{trimestre} notas sozinho,
        não importa quantos, podem preencher um compasso sete-oito.
        (Conte em oitavos e pense em \omterm{def:g3:numbers:evenodd}{pares} e \omterm{def:g3:numbers:evenodd}{ímpar}.)
\end{enumerate}
\end{problem}
