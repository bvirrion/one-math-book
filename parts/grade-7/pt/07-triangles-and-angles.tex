\chapter{Triângulos e ângulos}\label{ch:g7:triangles}

Você pode desenhar um triângulo com lados $10$, $3$ e $2$? Com ângulos
$80^\circ $, $70^\circ $ e $50^\circ $? Este capítulo responde a ambos
questões com dois dos fatos mais úteis da geometria plana: o
desigualdade triangular e a $180^\circ $ ângulo \omterm{def:g1:addition:def}{soma} --- mais o ângulo
vocabulário que vem com o cruzamento e \omterm{def:g6:lines:objects}{linhas} paralelas.

\section{Ângulos em torno de linhas cruzadas}

\begin{definition}[Pares de ângulos]\label{def:g7:triangles:pairs}
\begin{itemize}
  \item Dois ângulos são \emph{complementar} quando suas medidas somam
        para $90^\circ $, \emph{suplementar} quando eles somarem
        $180^\circ $.
  \item Quando duas \omterm{def:g6:lines:objects}{linhas} se cruzam, os ângulos voltados um para o outro na
        ponto de passagem são \emph{verticalmente
        oposto}\index{verticalmente
        oposto}.
  \item Quando uma \omterm{def:g6:lines:objects}{linha} cruza duas outras \omterm{def:g6:lines:objects}{linhas}, os ângulos localizados no
        mesma posição em cada cruzamento são
        \emph{ângulos correspondentes}; ângulos entre as duas \omterm{def:g6:lines:objects}{linhas} em
        lados opostos da \omterm{def:g6:lines:objects}{linha} de cruzamento são \emph{alternativa
        ângulos}\index{alternativa
        ângulos}.
\end{itemize}
\end{definition}

\begin{theorem}[Igualdades de ângulo]\label{thm:g7:triangles:equalities}
\begin{enumerate}
  \item Ângulos \omterm{def:g7:triangles:pairs}{verticalmente opostos} são iguais.
  \item Se duas \omterm{def:g6:lines:objects}{linhas} forem \omterm{def:g6:lines:perp}{paralelo}, os ângulos correspondentes são iguais e
        ângulos alternativos são iguais. Por outro lado, correspondente igual
        ângulos (ou alternados) forçam as \omterm{def:g6:lines:objects}{linhas} a serem \omterm{def:g6:lines:perp}{paralelo}.
\end{enumerate}
\end{theorem}

\begin{proof}[Prova de 1]
Os dois ângulos $\widehat{1}$ e $\widehat{2}$ de um lado de uma \omterm{def:g6:lines:objects}{linha}
faça um \omterm{def:g3:shapes:rightangle}{ângulo reto}: $\widehat 1 + \widehat 2 = 180^\circ $. O
\omterm{def:g7:triangles:pairs}{verticalmente oposto} ângulo $\widehat 3$ de $\widehat 1$ também satisfaz
$\widehat 2 + \widehat 3 = 180^\circ $ (mesmo motivo). Então
$\widehat 1 = 180^\circ - \widehat 2 = \widehat 3$. O ponto 2 é
admitidos neste nível.
\end{proof}

\section{A soma dos ângulos de um triângulo}

\begin{theorem}[Soma do ângulo]\label{thm:g7:triangles:sum}
Em todo triângulo, a \omterm{def:g1:addition:def}{soma} dos três ângulos dá $180^\circ $.
\end{theorem}

\begin{proof}
Deixar $ ABC $ seja um triângulo. Desenhe a \omterm{def:g6:lines:objects}{linha} através $ A $ \omterm{def:g6:lines:perp}{paralelo} para
$(BC)$. No vértice $ A $, três ângulos se alinham ao longo deste \omterm{def:g6:lines:perp}{paralelo}
e faça um \omterm{def:g3:shapes:rightangle}{ângulo reto}, $180^\circ $: o do meio é
$\widehat{A}$, e os dois externos são ângulos alternativos com
$\widehat B $ e $\widehat C $ (\omterm{def:g6:lines:objects}{linhas} paralelas!), então eles são iguais
$\widehat B $ e $\widehat C $. Por isso
$\widehat B + \widehat A + \widehat C = 180^\circ $.
\end{proof}

\begin{example}\label{ex:g7:triangles:sum}
Medida de dois ângulos de um triângulo $67^\circ $ e $58^\circ $. O terceiro
medidas
\[
180 - (67 + 58) = 180 - 125 = 55^\circ .
\]
Casos especiais que valem a pena memorizar: cada ângulo de uma triângulo \omterm{def:g6:shapes:triangles}{equilátero}
é $180 \div 3 = 60^\circ $; em um triângulo retângulo os dois ângulos agudos
são complementares (eles somam $180 - 90 = 90^\circ $); a base
ângulos de um triângulo \omterm{def:g6:shapes:triangles}{isósceles} são iguais, então cada um é
$\frac{180 - \text{apex}}{2}$.
\end{example}

\begin{example}[Nenhum triângulo com dois ângulos retos]\label{ex:g7:triangles:tworight}
Um triângulo com dois ângulos de $90^\circ $ já iria consumir
$180^\circ $, saindo $0^\circ $ para o terceiro --- impossível. O ângulo
\omterm{def:g1:addition:def}{soma} proíbe muitos triângulos ao mesmo tempo.
\end{example}

\section{A desigualdade triangular}

\begin{theorem}[Desigualdade triangular]\label{thm:g7:triangles:inequality}
Em todo triângulo, cada lado é menor que o \omterm{def:g1:addition:def}{soma} dos dois
outros: para um triângulo com lados $ a $, $ b $, $ c $,
\[
a < b + c .
\]
Por outro lado, três comprimentos com o \emph{maior} menor que o \omterm{def:g1:addition:def}{soma}
dos outros dois sempre pode ser transformado em um triângulo. Se o maior
\emph{é igual} o \omterm{def:g1:addition:def}{soma} dos outros, o ``triângulo'' é plano: o
três \omterm{def:g5:solids:def}{vértices} estão alinhados.
\end{theorem}
\admitted

\begin{example}\label{ex:g7:triangles:inequality}
Os lados podem ser $10$, $3$ e $2$? Teste o maior: é $10 < 3 + 2 = 5$?
Não --- tal triângulo não existe. A bússola mostra o porquê: arcos de \omterm{def:g3:shapes:shapes}{raios}
$3$ e $2$ extraído das duas extremidades de um \omterm{def:g6:lines:objects}{segmento} de comprimento $10$ nunca
conhecer.

Lados $7$, $5$ e $4$? Maior: $7 < 5 + 4 = 9$: sim, o triângulo
existe (basta verificar o lado maior).
\end{example}

\begin{method}[Existência e construção de um triângulo]\label{met:g7:triangles:construct}
Dados três comprimentos:
\begin{enumerate}
  \item compare o maior com o \omterm{def:g1:addition:def}{soma} dos outros dois; se for
        não estritamente menor, pare: sem triângulo;
  \item caso contrário, construa como em \cref{met:g6:shapes:construct}
        (base \omterm{def:g6:lines:objects}{segmento}, dois arcos de bússola);
  \item após qualquer construção em ângulo, verifique mentalmente se o
        ângulos dados somam menos que $180^\circ $ --- com o terceiro
        ângulo fazendo exatamente $180^\circ $.
\end{enumerate}
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g7:triangles:1}
Dê o complemento (para $90^\circ $) e o suplemento (para
$180^\circ $) de: $30^\circ $; \ $45^\circ $; \ $72^\circ $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:triangles:2}
Duas \omterm{def:g6:lines:objects}{linhas} se cruzam; um dos quatro ângulos mede $35^\circ $. Dê o
medidas dos outros três, com uma razão para cada um.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:triangles:3}
Calcule o terceiro ângulo de um triângulo cujos dois primeiros ângulos medem:
(a) $40^\circ $ e $60^\circ $; \
(b) $90^\circ $ e $28^\circ $; \
(c) $75^\circ $ e $75^\circ $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:triangles:4}
Um triângulo \omterm{def:g6:shapes:triangles}{isósceles} tem seu ângulo de vértice igual a $40^\circ $. Calcular
seus dois ângulos de base. Outro triângulo \omterm{def:g6:shapes:triangles}{isósceles} tem um ângulo de base de
$70^\circ $: calcule seu ângulo de vértice.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:triangles:5}
Quais desses triplos podem ser os lados de um triângulo? Justifique pelo
desigualdade triangular (maior lado!):
\[
(6,\ 8,\ 12); \qquad
(5,\ 5,\ 11); \qquad
(4,\ 9,\ 13); \qquad
(7,\ 7,\ 7).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:triangles:6}
Um triângulo tem ângulos $ x $, $2x $ e $3x $. Encontrar $ x $ e os três
ângulos. Que tipo de triângulo é esse?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:triangles:7}
Construa um triângulo $ ABC $ com $ BC = 6$ cm,
$\widehat{ABC} = 50^\circ $ e $\widehat{ACB} = 60^\circ $ (transferidor
em $ B $ e em $ C $). Antes de construir, preveja a medida de
$\widehat{BAC}$ e verifique sua figura.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:triangles:8}
Dois \omterm{def:g6:lines:objects}{linhas} paralelas são cortados por uma terceira \omterm{def:g6:lines:objects}{linha}, formando um ângulo de
$54^\circ $ no primeiro cruzamento (entre a \omterm{def:g6:lines:objects}{linha} de cruzamento e um
\omterm{def:g6:lines:perp}{paralelo}). Desenhe a situação e dê as medidas de todos os oito
ângulos formados.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:triangles:9}
Medida de dois lados de um triângulo $8$ centímetros e $3$ cm. Entre quais dois
valores o terceiro lado deve mentir? Forneça um comprimento de número inteiro que funcione
e um que não.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:triangles:10}
Em um triângulo $ ABC $, $\widehat A = 2\widehat B $ e
$\widehat C = 90^\circ $. Calcular $\widehat A $ e $\widehat B $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g7:triangles:11}
Os três ângulos de qualquer \omterm{def:g4:geometry:polygon}{quadrilátero} $ ABCD $ pode ser estudado por
cortando-o ao longo de uma diagonal em dois triângulos. Use isso para provar que
os quatro ângulos de cada \omterm{def:g4:geometry:polygon}{quadrilátero} somar a $360^\circ $ e
deduzir o ângulo \omterm{def:g1:addition:def}{soma} de um \omterm{def:g4:geometry:polygon}{Pentágono} (cinco lados).
\end{exercise}

\section{Problema: Andando pelo quarteirão}

\begin{problem}[{Problema de fim de semana --- os ângulos de qualquer polígono:
$(n-2) \times 180^\circ $ por dentro, sempre $360^\circ $ de virar
do lado de fora, e por que apenas três formas regulares ladrilham o chão}]
\label{pb:g7:triangles:1}
\Cref{exo:g7:triangles:11} corte um \omterm{def:g4:geometry:polygon}{quadrilátero} em dois triângulos
e encontrei $360^\circ $. Este problema empurra a ideia para \omterm{def:g4:geometry:polygon}{polígonos}
com qualquer número de lados, então descobre um segundo, completamente
prova diferente --- por \emph{andando} ao redor da forma e adicionando
sobe as curvas --- e termina no chão do seu banheiro: entre todos
normal \omterm{def:g4:geometry:polygon}{polígonos}, exatamente três podem ladrilhar um plano sem lacunas ou
se sobrepõem e as abelhas escolheram o seu.

\medskip
\noindent\textbf{Part I --- The angles inside.}
\begin{enumerate}
  \item De um vértice de um \omterm{def:g4:geometry:polygon}{hexágono} (seis lados), desenhe todos os
        diagonais deixando isso vértice. Em quantos triângulos está
        o \omterm{def:g4:geometry:polygon}{hexágono} cortado e qual é o \omterm{def:g1:addition:def}{soma} de \emph{todos} é
        ângulos internos (\cref{thm:g7:triangles:sum})?
  \item Generalizar: de um vértice de um \omterm{def:g4:geometry:polygon}{polígono} com $ n $ lados,
        quantos triângulos o leque de diagonais produz?
        Deduza a fórmula para \omterm{def:g1:addition:def}{soma} dos ângulos internos,
        e verifique se $ n = 3$, $4$, $5$, $6$.
  \item Calcule o \omterm{def:g1:addition:def}{soma} dos ângulos internos de um decágono
        ($10$ lados) e de um dodecágono ($12$ lados).
  \item Em um \emph{regular} \omterm{def:g4:geometry:polygon}{polígono} todos os ângulos internos são
        igual. Calcule o ângulo interno do \omterm{def:g6:shapes:triangles}{equilátero}
        triângulo, o quadrado, o regular \omterm{def:g4:geometry:polygon}{Pentágono}, \omterm{def:g4:geometry:polygon}{hexágono} e
        \omterm{def:g4:geometry:polygon}{octógono}.
  \item Explique, sem qualquer fórmula, por que o ângulo \omterm{def:g1:addition:def}{soma} cresce por
        exatamente $180^\circ $ cada vez que \omterm{def:g4:geometry:polygon}{polígono} ganha um
        vértice.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The walk around the block.}
Caminhe ao longo do limite de um bloco poligonal, sempre para frente. Em
cada canto você \emph{vez} por algum ângulo --- o \emph{exterior
ângulo} daquele canto; a rotação continua até que você esteja de volta
seu ponto de partida, voltado para sua direção de partida.
\begin{enumerate}[resume]
  \item Em um canto cujo ângulo interno é $\widehat A $, por como
        quanto você gira? (Os ângulos interno e externo são
        suplementar.) Calcular
        as três voltas para um triângulo com ângulos $40^\circ $,
        $60^\circ $, $80^\circ $.
  \item Some as três voltas da questão 6. Que círculo completo
        fato você observa?
  \item Explique por que as curvas de uma volta completa
        \emph{qualquer} \omterm{def:g4:geometry:polygon}{polígono} --- três lados ou trinta --- sempre
        total exatamente $360^\circ $. (O que seu nariz fez, tudo
        turnos combinados, quando você voltar?)
  \item Deduza a fórmula do ângulo interno uma segunda vez: em cada
        do $ n $ cantos, interiores $+$ vez $= 180^\circ $; \omterm{def:g1:addition:def}{soma}
        em todos os cantos e use a pergunta 8.
  \item Em um regular \omterm{def:g4:geometry:polygon}{polígono}, todas as voltas são iguais, então cada
        ângulo externo é $\frac{360^\circ}{n}$. Use isso para
        responda instantaneamente: qual regular \omterm{def:g4:geometry:polygon}{polígono} tem interior
        ângulos de $150^\circ $? E por que \emph{não} regular
        \omterm{def:g4:geometry:polygon}{polígono} tem ângulos internos de $155^\circ $?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Tiling the floor.}
Em torno de cada ponto do piso de cerâmica, os cantos da reunião
as peças devem totalizar exatamente $360^\circ $ --- sem lacuna, sem sobreposição
(\cref{pb:g6:lines:1}).
\begin{enumerate}[resume]
  \item Para cada um dos triângulo \omterm{def:g6:shapes:triangles}{equilátero}, o quadrado e o
        normal \omterm{def:g4:geometry:polygon}{hexágono}: quantas cópias se encontram em um ponto de canto
        do azulejo? Verifique o $360^\circ $ cada vez.
  \item Mostre isso regularmente pentágonos não é possível revestir o chão: o que
        faça três cantos no total e quanto seriam quatro no total?
  \item Mostre que não é regular \omterm{def:g4:geometry:polygon}{polígono} com \emph{mais} do que seis
        os lados podem ser lado a lado: pelo menos três peças devem se encontrar em cada
        canto (cada ângulo é menor que $180^\circ $) e o que
        acontece em três cantos $135^\circ $ ou mais? Concluir
        a lista completa de ladrilhadores regulares.
  \item Os pisos dos banheiros costumam ser misturados regularmente \omterm{def:g4:geometry:polygon}{octógonos} com pequeno
        quadrados. Verifique se este canto funciona:
        $135^\circ + 135^\circ + 90^\circ $. Outra mixagem clássica
        um quadrado, um regular \omterm{def:g4:geometry:polygon}{hexágono} e um dodecágono regular em
        cada canto: verifique também (o interior do dodecágono
        ângulo segue do método da questão 10).
  \item A escolha das abelhas: as células do favo de mel são regulares \omterm{def:g4:geometry:polygon}{hexágonos}.
        Verifique seus cantos ($3$ ângulos) e explique em um
        frase --- com a ajuda da descoberta de Dido
        (\cref{pb:g6:measure:1}) --- por que, dos três possíveis
        ladrilhos normais, o \omterm{def:g4:geometry:polygon}{hexágono} é o mais barato em cera para
        o mel que contém.
\end{enumerate}
\end{problem}
