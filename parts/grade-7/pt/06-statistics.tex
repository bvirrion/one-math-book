\chapter{Organizando Dados}\label{ch:g7:stats}

Antes de calcular qualquer coisa, os dados devem ser coletados, contados e
exibido. Este capítulo é sobre como transformar uma lista confusa de observações
em uma tabela ou imagem clara: contagens, \omterm{def:g7:stats:frequency}{frequências}, dados agrupados, barra
gráficos e gráficos de pizza. Resumindo os dados por um único número (média,
mediana) é o assunto de \cref{ch:g9:statproba}.

\section{Contagens e frequências}

\begin{definition}[Contagem e frequência]\label{def:g7:stats:frequency}
Para cada valor possível de uma pesquisa, seu \emph{contar}\index{contar} é
o número de vezes que aparece; isso é \emph{frequência}\index{frequência}
é a contagem dividida pelo número total de observações:
\[
\text{frequency} = \frac{\text{count}}{\text{total}} .
\]
As frequências podem ser escritas como frações, decimais ou porcentagens, e
eles sempre somam $1$ (ou seja,\ $100\,\%$).
\end{definition}

\begin{example}\label{ex:g7:stats:frequency}
Esporte favorito do $25$ alunos de uma turma, da lista bruta até
a mesa:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cccc|c}
sport & football & swimming & tennis & judo & total \\
\hline
count & $10$ & $6$ & $5$ & $4$ & $25$ \\
freqüência & $0.40$ & $0.24$ & $0.20$ & $0.16$ & $1$ \\
\end{tabular}
\end{center}
Para futebol: $\frac{10}{25} = \frac{40}{100} = 0.40 = 40\,\%$. O
freqüência \omterm{def:g6:lines:objects}{linha} deve somar $1$ --- um cheque permanente e gratuito.
\end{example}

\begin{remark}[Por que frequências?]
As contagens não podem ser comparadas entre grupos de tamanhos diferentes: $10$
fãs de futebol em uma classe de $25$ (aquilo é $40\,\%$) mostrar mais
entusiasmo do que $12$ num ano letivo de $120$ ($10\,\%$).
\omterm{def:g7:stats:frequency}{Frequências} colocam tudo no mesmo escala.
\end{remark}

\section{Agrupando dados em classes}

\begin{example}[Aulas]\label{ex:g7:stats:classes}
As alturas (em cm) de $20$ os alunos variam de $148$ para $172$.
Listar cada altura separadamente daria uma mesa com quase o mesmo número
colunas como alunos; em vez disso, agrupe em \emph{aulas}:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|ccc|c}
height (cm) & $\intco{145}{155}$ & $\intco{155}{165}$ &
$\intco{165}{175}$ & total \\
\hline
count & $5$ & $9$ & $6$ & $20$ \\
freqüência & $0.25$ & $0.45$ & $0.30$ & $1$ \\
\end{tabular}
\end{center}
Cada altura é contada em exatamente uma classe ($155$ entra
$\intco{155}{165}$, não em ambos --- é por isso que os intervalos são
meio aberto). O agrupamento perde detalhes, mas ganha legibilidade.
\end{example}

\section{Gráficos}

\begin{method}[Escolhendo e desenhando um gráfico]\label{met:g7:stats:charts}
\begin{enumerate}
  \item \emph{Gráfico de barras}: uma barra por valor ou classe, altura
        proporcional para a contagem (ou o freqüência). Bom para
        comparando valores.
  \item \emph{Gráfico de pizza}: um setor por valor, ângulo proporcional para o freqüência --- o círculo completo ($360^\circ $) representa o
        total, então
        \[
        \text{angle} = \text{frequency} \times 360^\circ .
        \]
        Bom para mostrar partes de um todo.
  \item Em ambos os casos: título, rótulos e graduação ou o
        porcentagens escritas no gráfico.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:stats:pie}
Ângulos para os esportes de \cref{ex:g7:stats:frequency}: futebol
$0.40 \times 360 = 144^\circ $; natação $0.24 \times 360 =
86.4^\circ $; tênis $72^\circ $; judo $57.6^\circ $. Verificar:
$144 + 86.4 + 72 + 57.6 = 360$.
\end{example}

\begin{example}[Lendo criticamente]\label{ex:g7:stats:critical}
Um gráfico cujo eixo vertical começa em $9$ em vez de $0$ faz uma barra
de $10$ parecer dez vezes mais alto que uma barra de $9.1$ --- visualmente
enganosa, embora os números sejam honestos. Primeiro reflexo na frente de
qualquer gráfico: verifique onde começa o eixo e qual é a graduação
vale a pena (\cref{met:g6:propdata:read}).
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g7:stats:1}
Aqui estão as marcas de $20$ alunos: $12$, $15$, $9$, $12$, $14$,
$9$, $12$, $15$, $11$, $12$, $9$, $14$, $12$, $11$, $15$, $9$, $12$,
$14$, $11$, $12$. Construa a tabela de contagens (uma coluna por marca).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:stats:2}
Complete a tabela de \cref{exo:g7:stats:1} com um freqüência \omterm{def:g6:lines:objects}{linha}
(como frações de $20$, depois como porcentagens). Verifique o \omterm{def:g1:addition:def}{soma}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:stats:3}
Em uma pesquisa de $50$ famílias, $30$ tem um carro. Qual é o
freqüência de ``um carro'', como decimal e como porcentagem? Em outro
pesquisa, $45$ famílias fora $90$ ter um carro: o que a pesquisa mostra
a maior proporção?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:stats:4}
As \omterm{def:g2:measure:units}{massas} (kg) de $15$ cães: $8$, $23$, $31$, $12$, $9$, $27$, $35$,
$14$, $22$, $18$, $29$, $11$, $25$, $33$, $16$. Agrupe-os no
aulas $\intco{5}{15}$, $\intco{15}{25}$, $\intco{25}{35}$,
$\intco{35}{45}$ e dê as contas.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:stats:5}
Desenhe o gráfico de barras da tabela:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|cccc}
pets & $0$ & $1$ & $2$ & $3$ \\
\hline
families & $8$ & $12$ & $6$ & $4$ \\
\end{tabular}
\end{center}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:stats:6}
Para a mesa de \cref{exo:g7:stats:5}, calcule as \omterm{def:g7:stats:frequency}{frequências} e
os ângulos do gráfico de pizza de cada valor (verifique: total $360^\circ $).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:stats:7}
Um gráfico de pizza sobre as temporadas favoritas dos programas: verão $162^\circ $, primavera
$90^\circ $, outono $54^\circ $, inverno $54^\circ $. Qual porcentagem
escolheu cada temporada? Se $60$ as pessoas responderam, quantas escolheram o verão?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:stats:8}
Na classe A, $12$ estudantes fora $30$ caminhar para a escola; na classe B,
$14$ fora de $40$. Qual classe tem a maior proporção de caminhantes?
Justifique com \omterm{def:g7:stats:frequency}{frequências}, não com contagens.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:stats:9}
Uma revista imprime um gráfico de barras das vendas mensais: $9\,800$, então
$10\,000$, então $10\,100$, com o eixo vertical começando em
$9\,700$. Descreva o que o leitor vê e que eixo honesto
começando em $0$ mostraria em vez disso.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:stats:10}
UM freqüência tabela tem três valores com \omterm{def:g7:stats:frequency}{frequências} $0.35$, $0.4$ e
$ f $. Encontrar $ f $. Se a contagem total for $80$, dê as três contagens.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g7:stats:11}
Em uma escola, $55\,\%$ dos alunos são meninas. Entre as meninas,
$40\,\%$ comer no refeitório; entre os meninos, $60\,\%$ fazer. Fora de
$400$ alunos, quantos comem no refeitório? (Calcule os quatro
tamanhos de grupo passo a passo.) Qual é a porcentagem geral?
\end{exercise}

\section{Problema: como os dados estão --- e como capturá-los}

\begin{problem}[{Problema de fim de semana --- conta em relação às frequências,
a pesquisa que enganou um país e o paradoxo dos dois
hospitais}]
\label{pb:g7:stats:1}
Os números não mentem, mas podem ser enganados: uma contagem enorme
pode esconder um pequeno freqüência, uma pesquisa gigantesca pode ser inútil,
e --- o mais estranho de tudo --- um hospital pode vencer seu rival em
\emph{todo} categoria de pacientes e ainda perde no geral
figuras. Todas as três armadilhas são acionadas neste problema, armadas apenas
com as contagens e \omterm{def:g7:stats:frequency}{frequências} deste capítulo
(\cref{def:g7:stats:frequency}).

\medskip
\noindent\textbf{Part I --- Counts are not frequencies.}
\begin{enumerate}
  \item A escola A recicla $120$ do seu $400$ \omterm{def:g5:solids:def}{caixas} de suco;
        escola B, $90$ do seu $250$. Qual escola tem maior
        \emph{\omterm{def:g7:stats:frequency}{contar}} de \omterm{def:g5:solids:def}{caixas} recicladas? Quanto maior
        \emph{freqüência}? Qual escola merece a reciclagem
        prêmio?
  \item Um novo horário é levantado: $18$ do $30$ professores
        gosto, e $210$ do $600$ os alunos fazem. Calcule o
        duas \omterm{def:g7:stats:frequency}{frequências}, então o freqüência entre todos $630$
        pessoas juntas. Por que a figura combinada fica tão
        perto do dos alunos?
  \item Um gráfico de pizza publicado mostra setores rotulados $45\,\%$,
        $30\,\%$, $20\,\%$ e $10\,\%$. Sem mais
        informações, como você sabe que um erro foi cometido?
  \item UM freqüência mesa acima $40$ observações estão meio apagadas:
        conta $14$, $10$, $?$, $?$ e \omterm{def:g7:stats:frequency}{frequências} $0.35$,
        $0.25$, $0.15$, $?$. Reconstrua as entradas ausentes.
  \item Uma pesquisa termina com \omterm{def:g7:stats:frequency}{frequências} $0.40$, $0.35$ e
        $0.25$. Calcule os três ângulos do gráfico de pizza
        (\cref{met:g7:stats:charts}) e verifique se eles fecham o
        círculo.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The poll that fooled a country.}
Em 1936, uma revista americana enviou pelo correio dez milhões de cédulas para
endereços retirados de listas telefônicas e registros de automóveis
listas, recebidas $2.4$ milhões de respostas e previu um esmagamento
vitória do candidato Landon. Um jovem estatístico, George
Gallup, questionou apenas cerca de cinquenta mil pessoas --- escolhidas para
assemelham-se a toda a população --- e previram o oposto.
Roosevelt venceu por uma vitória esmagadora.
\begin{enumerate}[resume]
  \item O que freqüência das cédulas enviadas voltaram?
  \item Em 1936, telefones e carros eram um luxo. Explique em
        uma ou duas frases porque a amostra da revista, embora
        enorme, estava condenado --- e qual lição da pergunta 1
        repete-se a nível nacional escala.
  \item Um modelo de brinquedo. Uma cidade tem $1\,000$ eleitores ricos, dos quais
        $70\,\%$ suporte L, e $9\,000$ eleitores modestos, dos quais
        $30\,\%$ suporte L. Calcule o verdadeiro suporte de L no
        cidade. Uma “pesquisa de lista telefônica” chega $500$ rico
        e $500$ eleitores modestos: que apoio prevê?
  \item A armadilha silenciosa: uma empresa tem $200$ clientes insatisfeitos
        e $800$ felizes. Ele examina todos; $40\,\%$ de
        a resposta infeliz, mas apenas $10\,\%$ do feliz fazer.
        Calcule o número de respostas de cada tipo e o
        freqüência de infelicidade \emph{entre as respostas}.
        Compare com o verdadeiro freqüência.
  \item Repare a enquete da questão 8: mantenha $1\,000$
        entrevistas, mas distribuí-las no verdadeiro
        proporções. O que a enquete reparada prevê?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The paradox of the two
hospitals.}
Dois hospitais publicam suas contagens de curas, divididas por gravidade:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cc}
 & mild cases & severe cases \\
\hline
hospital A & $90$ cured of $100$ & $280$ cured of $400$ \\
hospital B & $340$ cured of $400$ & $65$ cured of $100$ \\
\end{tabular}
\end{center}
\begin{enumerate}[resume]
  \item Calcular a cura do hospital A freqüência para casos leves, para
        casos graves e em geral (todos $500$ pacientes).
  \item Os mesmos três cálculos para o hospital B. Qual hospital
        vence em casos leves? Em casos graves?
  \item Compare as duas \omterm{def:g7:stats:frequency}{frequências} gerais. Indique claramente o
        coisa estranha que aconteceu.
  \item Explique o truque: compare os dois hospitais
        \emph{mistura} dos pacientes e dizer por que o resultado geral
        freqüência pode trair as \omterm{def:g7:stats:frequency}{frequências} de ambas as categorias. Se você
        teve um caso grave, qual hospital você deve escolher?
  \item Uma manchete de jornal diz: “O Hospital B tem o melhor
        taxa de cura: $81\,\%$ contra $74\,\%$.'' Escreva o
        carta de duas frases ao editor de que tudo isso
        problema ensinou você a escrever --- uma frase em
        o que a manchete ignora, um sobre a moral geral
        (compare iguais com semelhantes e preste atenção aos pesos).
\end{enumerate}
\end{problem}
