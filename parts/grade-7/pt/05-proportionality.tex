\chapter{Proporcionalidade}\label{ch:g7:prop}

6ª série cumprida proporcional tabelas (\cref{ch:g6:propdata}); este capítulo
voltas \omterm{def:g7:prop:table}{proporcionalidade} numa ferramenta completa: a regra cruzada para
encontrar um quarto valor, porcentagens e \omterm{def:g7:prop:scale}{escalas} de mapas e modelos.
Velocidade, a mais famosa \omterm{def:g7:prop:table}{proporcionalidade} de tudo, é estudado em
\cref{ch:g8:speed}.

\section{Reconhecendo e completando tabelas}

\begin{definition}[Tabela de proporcionalidade]\label{def:g7:prop:table}
Uma tabela de duas \omterm{def:g6:lines:objects}{linhas} é uma \emph{proporcionalidade
mesa}\index{proporcionalidade
mesa} quando os números da segunda \omterm{def:g6:lines:objects}{linha} são
aqueles do primeiro multiplicados por um número fixo, o
\emph{coeficiente}. Equivalentemente: todas as colunas \omterm{def:g3:division:remainder}{quocientes}
$\frac{\text{bottom}}{\text{top}}$ são iguais.
\end{definition}

\begin{example}\label{ex:g7:prop:recognize}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|ccc}
top & $4$ & $10$ & $14$ \\
\hline
bottom & $6$ & $15$ & $21$ \\
\end{tabular}
\end{center}
\omterm{def:g3:division:remainder}{Quocientes}: $\frac64 = 1.5$, $\frac{15}{10} = 1.5$,
$\frac{21}{14} = 1.5$: proporcional, coeficiente $1.5$.
\end{example}

\begin{theorem}[Regra cruzada]\label{thm:g7:prop:cross}
Em um \omterm{def:g7:prop:table}{tabela de proporcionalidade} com colunas $\begin{pmatrix} a \\ b
\end{pmatrix}$ e $\begin{pmatrix} c \\ d \end{pmatrix}$, a cruz
\omterm{def:g2:mult:def}{produtos} são iguais:
\[
a \times d = b \times c .
\]
Consequentemente, o quarto valor pode ser calculado a partir dos outros três:
$ d = \dfrac{b \times c}{a}$.
\end{theorem}

\begin{proof}
Deixar $ k $ seja o coeficiente: $ b = ka $ e $ d = kc $. Então
\[
a \times d = a \times kc = k \times ac,
\qquad
b \times c = ka \times c = k \times ac :
\]
ambos se cruzam \omterm{def:g2:mult:def}{produtos} igual $ k \times ac $. Dividindo $ ad = bc $ por $ a $
dá a fórmula para $ d $.
\end{proof}

\begin{example}[Quarta proporcional]\label{ex:g7:prop:fourth}
Se $7$ livros idênticos pesam $2.8$ kg, quanto $12$ tais livros
pesar? Tabela e regra cruzada:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cc}
books & $7$ & $12$ \\
\hline
kg & $2.8$ & $ m $ \\
\end{tabular}
\end{center}
\[
7 \times m = 2.8 \times 12
\quad\Longrightarrow\quad
m = \frac{2.8 \times 12}{7} = \frac{33.6}{7} = 4.8 \text{ kg}.
\]
Confira pela unidade: pesa um livro $0.4$ kg, doze pesam $4.8$ kg.
\end{example}

\section{Porcentagens}

\begin{method}[Aplicando e encontrando porcentagens]\label{met:g7:prop:percent}
\begin{enumerate}
  \item \emph{Aplicar} $ t\,\%$: multiplicar por $\frac{t}{100}$
        ($12\,\%$ de $350$ é $0.12 \times 350 = 42$);
  \item \emph{encontrar} a porcentagem que uma parte representa: calcular
        $\frac{\text{part}}{\text{whole}}$ e reescrever $100$
        ($21$ fora de $60$: $\frac{21}{60} = \frac{35}{100} = 35\,\%$);
  \item pergunte sempre: \emph{uma porcentagem de quê?} A referência inteira
        importa.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:prop:percent}
Em uma escola, $180$ do $400$ os alunos são meninas. Percentagem:
\[
\frac{180}{400} = \frac{180 \div 4}{400 \div 4} = \frac{45}{100}
= 45\,\% .
\]
A outra maneira: quantos são os $55\,\%$ meninos?
$0.55 \times 400 = 220$. Verificar: $180 + 220 = 400$.
\end{example}

\section{Balanças}

\begin{definition}[Escala]\label{def:g7:prop:scale}
Em um mapa ou modelo em \emph{escala}\index{escala}
$\frac{1}{n}$ (escrito $1 : n $), as distâncias no mapa são
proporcional às distâncias reais, com coeficiente $\frac1n $:
\[
\text{map distance} = \frac{1}{n} \times \text{real distance},
\]
ambos medidos no \emph{mesma unidade}.
\end{definition}

\begin{example}\label{ex:g7:prop:scale}
Em um $1 : 50\,000$ mapa, duas cidades estão $7$ cm de distância. Distância real:
\[
7 \times 50\,000 = 350\,000 \text{ cm} = 3\,500 \text{ m} = 3.5
\text{ km}.
\]
Por outro lado, um $12$ km medidas de trilha de caminhada no mapa
$12$ quilômetros $= 1\,200\,000$ centímetros, então $1\,200\,000 \div 50\,000 = 24$ cm.
\end{example}

\begin{remark}
Os gráficos são o teste mais rápido de \omterm{def:g7:prop:table}{proporcionalidade}
(\cref{ch:g6:propdata}): pontos em uma \omterm{def:g6:lines:objects}{linha} reta \emph{através do
origem} significar proporcional quantidades; uma \omterm{def:g6:lines:objects}{linha} sem a origem (como
uma tarifa de táxi com tarifa fixa) significa \emph{não} proporcional, mesmo
embora seja uma \omterm{def:g6:lines:objects}{linha} reta. Funções afins, em
\cref{ch:g9:linfunc}, tornará isso preciso.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g7:prop:1}
Quais tabelas são \omterm{def:g7:prop:table}{tabelas de proporcionalidade}? Dê o coeficiente quando
existe.
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|ccc}
 & $3$ & $7$ & $11$ \\
\hline
 & $7.5$ & $17.5$ & $27.5$ \\
\end{tabular}
\qquad
\begin{tabular}{l|ccc}
 & $2$ & $5$ & $9$ \\
\hline
 & $6$ & $15$ & $28$ \\
\end{tabular}
\end{center}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:prop:2}
Complete usando a regra cruzada, escrevendo o cálculo:
$5$ custo de kg de batatas $6$ euros; quanto fazer $8$ custa kg?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:prop:3}
Uma impressora imprime $36$ páginas em $3$ minutos. Na mesma proporção, quantos
páginas em $5$ minutos? Quanto tempo $96$ páginas?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:prop:4}
Calcular: $30\,\%$ de $250$; \quad $15\,\%$ de $60$; \quad $ t\,\%$ tal
isso $ t\,\%$ de $80$ é $20$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:prop:5}
Fora de $25$ arremessos, um jogador de basquete marcou $18$. O que é ela
porcentagem de sucesso? (Reescreva o \omterm{def:g6:fractions:def}{fração} sobre $100$.)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:prop:6}
Em um $1 : 25\,000$ mapa, um caminho mede $9$ cm. Qual é o seu real
comprimento em km? Um lago é $2$ km de comprimento: quanto tempo está no mapa?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g7:prop:7}
Um modelo de carro é construído em \omterm{def:g7:prop:scale}{escala} $1 : 43$. O verdadeiro carro é $4.3$ m longo.
Quanto tempo tem o modelo, em cm?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:prop:8}
Uma receita usa $240$ g de chocolate para $8$ porções. Aline tem
$300$ g de chocolate. Para quantas porções isso é suficiente (inteiro
número de porções)?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:prop:9}
O preço de uma jaqueta cai de $80$ para $60$ euros.
\begin{enumerate}
  \item Qual é o desconto em euros? Em porcentagem do original
        preço?
  \item Mais tarde, o preço volta a subir de $60$ para $80$ euros. Explique
        por que esse aumento é \emph{não} $25\,\%$ --- calcule o correto
        percentual de aumento.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g7:prop:10}
Duas \omterm{def:g6:lines:objects}{linhas} de uma tabela são proporcional:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|ccc}
 & $4$ & $ x $ & $10$ \\
\hline
 & $ y $ & $10.5$ & $17.5$ \\
\end{tabular}
\end{center}
Encontre o coeficiente, então $ x $ e $ y $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g7:prop:11}
Uma fotocopiadora reduz documentos a $80\,\%$ do seu tamanho. UM \omterm{def:g6:lines:objects}{segmento}
de $10$ cm é copiado, então o \emph{cópia} é copiado novamente com o
mesma configuração.
\begin{enumerate}
  \item Quanto tempo dura \omterm{def:g6:lines:objects}{segmento} depois de uma cópia? Depois das duas?
  \item Por que a resposta depois de duas cópias não $60\,\%$ do
        originais? Que porcentagem única corresponde a duas cópias?
\end{enumerate}
\end{exercise}

\section{Problema: Medindo a Terra com uma vara}

\begin{problem}[{Problema de fim de semana --- sombras são um
proporcionalidade e como Eratóstenes calculou a circunferência
da Terra em 240 AC}]
\label{pb:g7:prop:1}
Há vinte e dois séculos, sem telescópio, sem satélite e sem
calculadora, o bibliotecário de Alexandria mediu a Terra ---
usando uma vara, um poço, uma caravana de camelos e este capítulo
matemática. Este problema refaz seus passos: primeiro o
\omterm{def:g7:prop:table}{proporcionalidade} de sombras, então a famosa computação em si,
e finalmente como é sua resposta para humanos \omterm{def:g7:prop:scale}{escala}.

\medskip
\noindent\textbf{Part I --- The shadow of a stick.}
O sol está tão longe que seus \omterm{def:g3:shapes:shapes}{raios} chegam até nós \omterm{def:g6:lines:perp}{paralelo} para um
outro. Num dado momento, todos os objetos verticais e suas
as sombras são, portanto, proporcional.
\begin{enumerate}
  \item Ao meio-dia, uma vara vertical de $1$ m lança uma sombra de
        $0.4$ m, e a sombra de uma árvore mede $3.2$ m. Quão alto
        é a árvore (\cref{thm:g7:prop:cross})?
  \item No mesmo momento, uma torre lança um $26$ m sombra: como
        alta é a torre? E quanto tempo dura a sombra de uma criança
        $1.5$ sou alto?
  \item Explique em uma frase por que todos esses cálculos são
        válido apenas no \emph{mesmo momento} do dia.
  \item A lenda diz que Tales mediu o Grande \omterm{def:g5:solids:def}{Pirâmide} esperando
        para o momento em que \emph{sua própria sombra era exatamente como
        contanto que ele fosse alto}. Qual é o \omterm{def:g7:prop:table}{proporcionalidade}
        coeficiente naquele momento, e qual é a altura do \omterm{def:g5:solids:def}{pirâmide}
        se a ponta de sua sombra estiver $147$ m do ponto de
        o solo diretamente abaixo de seu ápice?
  \item Resuma a Parte I: em um determinado momento, qual quantidade é
        o mesmo para o bastão, a árvore, a torre e o
        \omterm{def:g5:solids:def}{pirâmide} (\cref{def:g7:prop:table})?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Eratosthenes' computation.}
Eratóstenes conhecia dois fatos. Na cidade de Syene, ao meio-dia do
solstício de verão, o sol estava \emph{exatamente acima da cabeça}: luz solar
atingiu o fundo dos poços mais profundos, e varas verticais foram lançadas
sem sombra. Em Alexandria, $800$ km ao norte, no mesmo momento,
uma vara vertical \emph{fez} lança uma sombra e os \omterm{def:g3:shapes:shapes}{raios} do sol
made an angle of $7.2^\circ $ com a vertical.
\begin{enumerate}[resume]
  \item Explique por que essas duas observações juntas provam que
        o terreno de Syene e o terreno de Alexandria não são
        \omterm{def:g6:lines:perp}{paralelo} --- isto é, a superfície da Terra é curva.
        (O que seria \omterm{def:g6:lines:perp}{paralelo} \omterm{def:g3:shapes:shapes}{raios} fazem com duas varas em uma superfície plana
        Terra?)
  \item Num desenho do redondo Terra com \omterm{def:g6:lines:perp}{paralelo} \omterm{def:g3:shapes:shapes}{raios} solares,
        o $7.2^\circ $ em Alexandria reaparece no centro de
        da Terra, como o ângulo entre as direções do
        duas cidades. Faça o desenho e convença-se (o
        a justificação limpa usa os \omterm{def:g3:numbers:evenodd}{pares} de ângulos iguais de
        \cref{ch:g7:triangles}).
  \item O que \omterm{def:g6:fractions:def}{fração} de uma volta completa é $7.2^\circ $?
  \item O arco de Syene a Alexandria ($800$ km) corresponde
        para $7.2^\circ $; toda a circunferência corresponde a
        $360^\circ $. Configure o \omterm{def:g7:prop:table}{proporcionalidade} e calcular o
        circunferência da Terra.
  \item Deduza o \omterm{def:g4:geometry:circle}{diâmetro} da Terra, usando $\pi \approx
        3.14$ (\cref{prop:g6:measure:circle}) e \omterm{def:g6:decimals:rounding}{arredondamento} para
        os cem quilômetros mais próximos. (O valor moderno é
        $12\,742$ km --- quão perto estava um homem com uma vara
        240 a.C.?)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The Earth at human \omterm{def:g7:prop:scale}{escala}.}
\begin{enumerate}[resume]
  \item Um globo é construído em \omterm{def:g7:prop:scale}{escala} $1 : 40\,000\,000$. Usando o
        circunferência encontrada na questão 9, mostram que o
        a circunferência do globo é exatamente $1$ m. Qual é o seu
        \omterm{def:g4:geometry:circle}{diâmetro}, até o centímetro mais próximo?
  \item Mont Blanc se eleva sobre $4.8$ km. Converter $4.8$ km para
        centímetros, divida por $40\,000\,000$, e expressar o
        altura da montanha no globo em milímetros. O que faz
        a resposta diz sobre quão “acidentada” a Terra realmente é?
  \item Um andador cobre $40$ km por dia. Nesse ritmo, quantos
        dias para a circunferência total de Eratóstenes --- e
        aproximadamente quantos anos são isso?
  \item A atmosfera respirável está concentrada aproximadamente no
        primeiro $10$ km acima do solo. Que porcentagem do
        da Terra \omterm{def:g3:shapes:shapes}{raio} (sobre $6\,370$ km) é que
        (\cref{met:g7:prop:percent})? \omterm{def:g4:numbers:round}{Arredonde} para o mais próximo
        décimo de um por cento.
  \item Os historiadores estimam que o valor anunciado de Eratóstenes,
        convertido em unidades modernas, pode ter sido cerca de
        $42\,000$ km. Tirando $40\,000$ km como o valor verdadeiro,
        calcule sua porcentagem de erro. Conclua em um
        frase sobre paus, poços e \omterm{def:g7:prop:table}{proporcionalidade}.
\end{enumerate}
\end{problem}
