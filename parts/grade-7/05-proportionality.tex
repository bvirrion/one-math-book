\chapter{Proportionality}\label{ch:g7:prop}

Grade 6 met \omterm{def:g6:propdata:def}{proportional} tables (\cref{ch:g6:propdata}); this chapter
turns \omterm{def:g7:prop:table}{proportionality} into a fully-fledged tool: the cross rule for
finding a fourth value, percentages, and \omterm{def:g7:prop:scale}{scales} of maps and models.
Speed, the most famous \omterm{def:g7:prop:table}{proportionality} of all, is studied in
\cref{ch:g8:speed}.

\section{Recognizing and completing tables}

\begin{definition}[Proportionality table]\label{def:g7:prop:table}
A table of two rows is a \emph{proportionality
table}\index{proportionality} when the numbers of the second row are
those of the first multiplied by one fixed number, the
\emph{coefficient}. Equivalently: all column \omterm{def:g3:division:remainder}{quotients}
$\frac{\text{bottom}}{\text{top}}$ are equal.
\end{definition}

\begin{example}\label{ex:g7:prop:recognize}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|ccc}
top & $4$ & $10$ & $14$ \\
\hline
bottom & $6$ & $15$ & $21$ \\
\end{tabular}
\end{center}
\omterm{def:g3:division:remainder}{Quotients}: $\frac64 = 1.5$, $\frac{15}{10} = 1.5$,
$\frac{21}{14} = 1.5$: \omterm{def:g6:propdata:def}{proportional}, coefficient $1.5$.
\end{example}

\begin{theorem}[Cross rule]\label{thm:g7:prop:cross}
In a \omterm{def:g7:prop:table}{proportionality table} with columns $\begin{pmatrix} a \\ b
\end{pmatrix}$ and $\begin{pmatrix} c \\ d \end{pmatrix}$, the cross
\omterm{def:g2:mult:def}{products} are equal:
\[
a \times d = b \times c .
\]
Consequently the fourth value can be computed from the three others:
$d = \dfrac{b \times c}{a}$.
\end{theorem}

\begin{proof}
Let $k$ be the coefficient: $b = ka$ and $d = kc$. Then
\[
a \times d = a \times kc = k \times ac,
\qquad
b \times c = ka \times c = k \times ac :
\]
both cross \omterm{def:g2:mult:def}{products} equal $k \times ac$. Dividing $ad = bc$ by $a$
gives the formula for $d$.
\end{proof}

\begin{example}[Fourth proportional]\label{ex:g7:prop:fourth}
If $7$ identical books weigh $2.8$ kg, how much do $12$ such books
weigh? Table and cross rule:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cc}
books & $7$ & $12$ \\
\hline
kg & $2.8$ & $m$ \\
\end{tabular}
\end{center}
\[
7 \times m = 2.8 \times 12
\quad\Longrightarrow\quad
m = \frac{2.8 \times 12}{7} = \frac{33.6}{7} = 4.8 \text{ kg}.
\]
Check by the unit: one book weighs $0.4$ kg, twelve weigh $4.8$ kg.
\end{example}

\section{Percentages}

\begin{method}[Applying and finding percentages]\label{met:g7:prop:percent}
\begin{enumerate}
  \item \emph{Apply} $t\,\%$: multiply by $\frac{t}{100}$
        ($12\,\%$ of $350$ is $0.12 \times 350 = 42$);
  \item \emph{find} the percentage that a part represents: compute
        $\frac{\text{part}}{\text{whole}}$ and rewrite over $100$
        ($21$ out of $60$: $\frac{21}{60} = \frac{35}{100} = 35\,\%$);
  \item always ask: \emph{a percentage of what?} The reference whole
        matters.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:prop:percent}
In a school, $180$ of the $400$ students are girls. Percentage:
\[
\frac{180}{400} = \frac{180 \div 4}{400 \div 4} = \frac{45}{100}
= 45\,\% .
\]
The other way: how many are the $55\,\%$ boys?
$0.55 \times 400 = 220$. Check: $180 + 220 = 400$.
\end{example}

\section{Scales}

\begin{definition}[Scale]\label{def:g7:prop:scale}
On a map or model at \emph{scale}\index{scale}
$\frac{1}{n}$ (written $1 : n$), the distances on the map are
\omterm{def:g6:propdata:def}{proportional} to the real distances, with coefficient $\frac1n$:
\[
\text{map distance} = \frac{1}{n} \times \text{real distance},
\]
both measured in the \emph{same unit}.
\end{definition}

\begin{example}\label{ex:g7:prop:scale}
On a $1 : 50\,000$ map, two towns are $7$ cm apart. Real distance:
\[
7 \times 50\,000 = 350\,000 \text{ cm} = 3\,500 \text{ m} = 3.5
\text{ km}.
\]
Conversely, a $12$ km hiking trail measures on the map
$12$ km $= 1\,200\,000$ cm, so $1\,200\,000 \div 50\,000 = 24$ cm.
\end{example}

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[omaxis,
  width=8.4cm, height=5.6cm,
  xmin=0, xmax=8.6, ymin=0, ymax=34,
  xlabel={map (cm)}, ylabel={real (km)},
  xtick={2,4,6,8}, ytick={10,20,30}]
  \addplot[omDef, thick, domain=0:8.2] {4*x};
  \addplot[omDef, only marks, mark=*, mark size=1.5pt] coordinates
    {(2,8) (4,16) (6,24) (8,32)};
  \draw[dashed, omThm] (axis cs:7,0) -- (axis cs:7,28)
    -- (axis cs:0,28);
\end{axis}
\end{tikzpicture}

{\small A \omterm{def:g7:prop:scale}{scale} is a \omterm{def:g7:prop:table}{proportionality}: map distances against real
distances line up on a straight line through the origin (here
$1$ cm $\leftrightarrow$ $4$ km; the dashes read $7$ cm
$\leftrightarrow$ $28$ km).}
\end{omfigure}

\begin{remark}
Graphs are the quickest test of \omterm{def:g7:prop:table}{proportionality}
(\cref{ch:g6:propdata}): points on a straight line \emph{through the
origin} mean \omterm{def:g6:propdata:def}{proportional} quantities; a line missing the origin (like
a taxi fare with a fixed charge) means \emph{not} \omterm{def:g6:propdata:def}{proportional}, even
though it is a straight line. Affine functions, in
\cref{ch:g9:linfunc}, will make this precise.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g7:prop:1}
Which tables are \omterm{def:g7:prop:table}{proportionality tables}? Give the coefficient when it
exists.
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|ccc}
 & $3$ & $7$ & $11$ \\
\hline
 & $7.5$ & $17.5$ & $27.5$ \\
\end{tabular}
\qquad
\begin{tabular}{l|ccc}
 & $2$ & $5$ & $9$ \\
\hline
 & $6$ & $15$ & $28$ \\
\end{tabular}
\end{center}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:2}
Complete using the cross rule, writing the computation:
$5$ kg of potatoes cost $6$ euros; how much do $8$ kg cost?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:3}
A printer prints $36$ pages in $3$ minutes. At the same rate, how many
pages in $5$ minutes? How long for $96$ pages?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:4}
Compute: $30\,\%$ of $250$; \quad $15\,\%$ of $60$; \quad $t\,\%$ such
that $t\,\%$ of $80$ is $20$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:5}
Out of $25$ shots, a basketball player scored $18$. What is her
success percentage? (Rewrite the \omterm{def:g6:fractions:def}{fraction} over $100$.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:6}
On a $1 : 25\,000$ map, a path measures $9$ cm. What is its real
length in km? A lake is $2$ km long: how long is it on the map?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:7}
A model car is built at \omterm{def:g7:prop:scale}{scale} $1 : 43$. The real car is $4.3$ m long.
How long is the model, in cm?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:prop:8}
A recipe uses $240$ g of chocolate for $8$ servings. Aline has
$300$ g of chocolate. For how many servings is that enough (whole
number of servings)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:prop:9}
The price of a jacket drops from $80$ to $60$ euros.
\begin{enumerate}
  \item What is the discount in euros? In percent of the original
        price?
  \item Later the price goes back up from $60$ to $80$ euros. Explain
        why this rise is \emph{not} $25\,\%$ --- compute the correct
        percentage of increase.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:prop:10}
Two rows of a table are \omterm{def:g6:propdata:def}{proportional}:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|ccc}
 & $4$ & $x$ & $10$ \\
\hline
 & $y$ & $10.5$ & $17.5$ \\
\end{tabular}
\end{center}
Find the coefficient, then $x$ and $y$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:prop:11}
A photocopier reduces documents to $80\,\%$ of their size. A \omterm{def:g6:lines:objects}{segment}
of $10$ cm is copied, then the \emph{copy} is copied again with the
same setting.
\begin{enumerate}
  \item How long is the \omterm{def:g6:lines:objects}{segment} after one copy? After two?
  \item Why is the answer after two copies not $60\,\%$ of the
        original? What single percentage corresponds to two copies?
\end{enumerate}
\end{exercise}

\section{Problem: Measuring the Earth with a stick}

\begin{problem}[{Weekend problem --- shadows are a
proportionality, and how Eratosthenes computed the circumference
of the Earth in 240 BC}]
\label{pb:g7:prop:1}
Twenty-two centuries ago, with no telescope, no satellite and no
calculator, the librarian of Alexandria measured the Earth ---
using a stick, a well, a camel caravan and this chapter's
mathematics. This problem retraces his steps: first the
\omterm{def:g7:prop:table}{proportionality} of shadows, then the famous computation itself,
and finally what his answer looks like at human \omterm{def:g7:prop:scale}{scale}.

\medskip
\noindent\textbf{Part I --- The shadow of a stick.}
The sun is so far away that its rays reach us \omterm{def:g6:lines:perp}{parallel} to one
another. At a given moment, all vertical objects and their
shadows are therefore \omterm{def:g6:propdata:def}{proportional}.
\begin{enumerate}
  \item At noon, a vertical stick of $1$ m casts a shadow of
        $0.4$ m, and a tree's shadow measures $3.2$ m. How tall
        is the tree (\cref{thm:g7:prop:cross})?
  \item At the same moment, a tower casts a $26$ m shadow: how
        tall is the tower? And how long is the shadow of a child
        $1.5$ m tall?
  \item Explain in one sentence why all these computations are
        valid only at the \emph{same moment} of the day.
  \item Legend says Thales measured the Great \omterm{def:g5:solids:def}{Pyramid} by waiting
        for the moment when \emph{his own shadow was exactly as
        long as he was tall}. What is the \omterm{def:g7:prop:table}{proportionality}
        coefficient at that moment, and how tall is the \omterm{def:g5:solids:def}{pyramid}
        if the tip of its shadow lies $147$ m from the point of
        the ground directly below its apex?
  \item Summarize Part I: at one given moment, which quantity is
        the same for the stick, the tree, the tower and the
        \omterm{def:g5:solids:def}{pyramid} (\cref{def:g7:prop:table})?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Eratosthenes' computation.}
Eratosthenes knew two facts. In the town of Syene, at noon on the
summer solstice, the sun stood \emph{exactly overhead}: sunlight
reached the bottom of the deepest wells, and vertical sticks cast
no shadow. In Alexandria, $800$ km due north, at the same moment,
a vertical stick \emph{did} cast a shadow, and the sun's rays
made an angle of $7.2^\circ$ with the vertical.
\begin{enumerate}[resume]
  \item Explain why these two observations together prove that
        the ground of Syene and the ground of Alexandria are not
        \omterm{def:g6:lines:perp}{parallel} --- that is, the Earth's surface is curved.
        (What would \omterm{def:g6:lines:perp}{parallel} rays do to two sticks on a flat
        Earth?)
  \item On a drawing of the \omterm{def:g4:numbers:round}{round} Earth with \omterm{def:g6:lines:perp}{parallel} sun rays,
        the $7.2^\circ$ in Alexandria reappears at the center of
        the Earth, as the angle between the directions of the
        two cities. Make the drawing and convince yourself (the
        clean justification uses the equal-angle pairs of
        \cref{ch:g7:triangles}).
  \item What \omterm{def:g6:fractions:def}{fraction} of a full turn is $7.2^\circ$?
  \item The arc from Syene to Alexandria ($800$ km) corresponds
        to $7.2^\circ$; the whole circumference corresponds to
        $360^\circ$. Set up the \omterm{def:g7:prop:table}{proportionality} and compute the
        circumference of the Earth.
  \item Deduce the \omterm{def:g4:geometry:circle}{diameter} of the Earth, using $\pi \approx
        3.14$ (\cref{prop:g6:measure:circle}) and \omterm{def:g6:decimals:rounding}{rounding} to
        the nearest hundred kilometers. (The modern value is
        $12\,742$ km --- how close was a man with a stick in
        240 BC?)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The Earth at human \omterm{def:g7:prop:scale}{scale}.}
\begin{enumerate}[resume]
  \item A globe is built at \omterm{def:g7:prop:scale}{scale} $1 : 40\,000\,000$. Using the
        circumference found in question 9, show that the
        globe's circumference is exactly $1$ m. What is its
        \omterm{def:g4:geometry:circle}{diameter}, to the nearest centimeter?
  \item Mont Blanc rises about $4.8$ km. Convert $4.8$ km to
        centimeters, divide by $40\,000\,000$, and express the
        mountain's height on the globe in millimeters. What does
        the answer say about how ``bumpy'' the Earth really is?
  \item A walker covers $40$ km per day. At that pace, how many
        days for Eratosthenes' full circumference --- and
        roughly how many years is that?
  \item The breathable atmosphere is concentrated in roughly the
        first $10$ km above the ground. What percentage of the
        Earth's \omterm{def:g3:shapes:shapes}{radius} (about $6\,370$ km) is that
        (\cref{met:g7:prop:percent})? Round to the nearest
        tenth of a percent.
  \item Historians estimate that Eratosthenes' announced value,
        converted to modern units, may have been about
        $42\,000$ km. Taking $40\,000$ km as the true value,
        compute his percentage of error. Conclude in one
        sentence about sticks, wells and \omterm{def:g7:prop:table}{proportionality}.
\end{enumerate}
\end{problem}
