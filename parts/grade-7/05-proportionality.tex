\chapter{Proportionality}\label{ch:g7:prop}

Grade 6 met proportional tables (\cref{ch:g6:propdata}); this chapter
turns proportionality into a fully-fledged tool: the cross rule for
finding a fourth value, percentages, and scales of maps and models.
Speed, the most famous proportionality of all, is studied in
\cref{ch:g8:speed}.

\section{Recognizing and completing tables}

\begin{definition}[Proportionality table]\label{def:g7:prop:table}
A table of two rows is a \emph{proportionality
table}\index{proportionality} when the numbers of the second row are
those of the first multiplied by one fixed number, the
\emph{coefficient}. Equivalently: all column quotients
$\frac{\text{bottom}}{\text{top}}$ are equal.
\end{definition}

\begin{example}\label{ex:g7:prop:recognize}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|ccc}
top & $4$ & $10$ & $14$ \\
\hline
bottom & $6$ & $15$ & $21$ \\
\end{tabular}
\end{center}
Quotients: $\frac64 = 1.5$, $\frac{15}{10} = 1.5$,
$\frac{21}{14} = 1.5$: proportional, coefficient $1.5$.
\end{example}

\begin{theorem}[Cross rule]\label{thm:g7:prop:cross}
In a proportionality table with columns $\begin{pmatrix} a \\ b
\end{pmatrix}$ and $\begin{pmatrix} c \\ d \end{pmatrix}$, the cross
products are equal:
\[
a \times d = b \times c .
\]
Consequently the fourth value can be computed from the three others:
$d = \dfrac{b \times c}{a}$.
\end{theorem}

\begin{proof}
Let $k$ be the coefficient: $b = ka$ and $d = kc$. Then
\[
a \times d = a \times kc = k \times ac,
\qquad
b \times c = ka \times c = k \times ac :
\]
both cross products equal $k \times ac$. Dividing $ad = bc$ by $a$
gives the formula for $d$.
\end{proof}

\begin{example}[Fourth proportional]\label{ex:g7:prop:fourth}
If $7$ identical books weigh $2.8$ kg, how much do $12$ such books
weigh? Table and cross rule:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cc}
books & $7$ & $12$ \\
\hline
kg & $2.8$ & $m$ \\
\end{tabular}
\end{center}
\[
7 \times m = 2.8 \times 12
\quad\Longrightarrow\quad
m = \frac{2.8 \times 12}{7} = \frac{33.6}{7} = 4.8 \text{ kg}.
\]
Check by the unit: one book weighs $0.4$ kg, twelve weigh $4.8$ kg.
\end{example}

\section{Percentages}

\begin{method}[Applying and finding percentages]\label{met:g7:prop:percent}
\begin{enumerate}
  \item \emph{Apply} $t\,\%$: multiply by $\frac{t}{100}$
        ($12\,\%$ of $350$ is $0.12 \times 350 = 42$);
  \item \emph{find} the percentage that a part represents: compute
        $\frac{\text{part}}{\text{whole}}$ and rewrite over $100$
        ($21$ out of $60$: $\frac{21}{60} = \frac{35}{100} = 35\,\%$);
  \item always ask: \emph{a percentage of what?} The reference whole
        matters.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:prop:percent}
In a school, $180$ of the $400$ students are girls. Percentage:
\[
\frac{180}{400} = \frac{180 \div 4}{400 \div 4} = \frac{45}{100}
= 45\,\% .
\]
The other way: how many are the $55\,\%$ boys?
$0.55 \times 400 = 220$. Check: $180 + 220 = 400$.
\end{example}

\section{Scales}

\begin{definition}[Scale]\label{def:g7:prop:scale}
On a map or model at \emph{scale}\index{scale}
$\frac{1}{n}$ (written $1 : n$), the distances on the map are
proportional to the real distances, with coefficient $\frac1n$:
\[
\text{map distance} = \frac{1}{n} \times \text{real distance},
\]
both measured in the \emph{same unit}.
\end{definition}

\begin{example}\label{ex:g7:prop:scale}
On a $1 : 50\,000$ map, two towns are $7$ cm apart. Real distance:
\[
7 \times 50\,000 = 350\,000 \text{ cm} = 3\,500 \text{ m} = 3.5
\text{ km}.
\]
Conversely, a $12$ km hiking trail measures on the map
$12$ km $= 1\,200\,000$ cm, so $1\,200\,000 \div 50\,000 = 24$ cm.
\end{example}

\begin{tikzpicture}
\begin{axis}[omaxis,

  \draw[dashed, omThm] (axis cs:7,0) -- (axis cs:7,28)
    -- (axis cs:0,28);
\end{axis}
\end{tikzpicture}

{\small A scale is a proportionality: map distances against real
distances line up on a straight line through the origin (here
$1$ cm $\leftrightarrow$ $4$ km; the dashes read $7$ cm
$\leftrightarrow$ $28$ km).}
\end{omfigure}

\begin{remark}
Graphs are the quickest test of proportionality
(\cref{ch:g6:propdata}): points on a straight line \emph{through the
origin} mean proportional quantities; a line missing the origin (like
a taxi fare with a fixed charge) means \emph{not} proportional, even
though it is a straight line. Affine functions, in
\cref{ch:g9:linfunc}, will make this precise.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g7:prop:1}
Which tables are proportionality tables? Give the coefficient when it
exists.
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|ccc}
 & $3$ & $7$ & $11$ \\
\hline
 & $7.5$ & $17.5$ & $27.5$ \\
\end{tabular}
\qquad
\begin{tabular}{l|ccc}
 & $2$ & $5$ & $9$ \\
\hline
 & $6$ & $15$ & $28$ \\
\end{tabular}
\end{center}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:2}
Complete using the cross rule, writing the computation:
$5$ kg of potatoes cost $6$ euros; how much do $8$ kg cost?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:3}
A printer prints $36$ pages in $3$ minutes. At the same rate, how many
pages in $5$ minutes? How long for $96$ pages?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:4}
Compute: $30\,\%$ of $250$; \quad $15\,\%$ of $60$; \quad $t\,\%$ such
that $t\,\%$ of $80$ is $20$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:5}
Out of $25$ shots, a basketball player scored $18$. What is her
success percentage? (Rewrite the fraction over $100$.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:6}
On a $1 : 25\,000$ map, a path measures $9$ cm. What is its real
length in km? A lake is $2$ km long: how long is it on the map?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:prop:7}
A model car is built at scale $1 : 43$. The real car is $4.3$ m long.
How long is the model, in cm?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:prop:8}
A recipe uses $240$ g of chocolate for $8$ servings. Aline has
$300$ g of chocolate. For how many servings is that enough (whole
number of servings)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:prop:9}
The price of a jacket drops from $80$ to $60$ euros.
\begin{enumerate}
  \item What is the discount in euros? In percent of the original
        price?
  \item Later the price goes back up from $60$ to $80$ euros. Explain
        why this rise is \emph{not} $25\,\%$ --- compute the correct
        percentage of increase.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:prop:10}
Two rows of a table are proportional:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|ccc}
 & $4$ & $x$ & $10$ \\
\hline
 & $y$ & $10.5$ & $17.5$ \\
\end{tabular}
\end{center}
Find the coefficient, then $x$ and $y$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:prop:11}
A photocopier reduces documents to $80\,\%$ of their size. A segment
of $10$ cm is copied, then the \emph{copy} is copied again with the
same setting.
\begin{enumerate}
  \item How long is the segment after one copy? After two?
  \item Why is the answer after two copies not $60\,\%$ of the
        original? What single percentage corresponds to two copies?
\end{enumerate}
\end{exercise}
