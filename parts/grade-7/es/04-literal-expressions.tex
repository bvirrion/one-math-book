\section{Escribiendo con letras}

\begin{definition}[expresiĂłn literal]\label{def:g7:literal:def}
A \emph{expresiĂłn literal}\index{expresiĂłn literal} es un cĂĄlculo
que contiene una o mĂĄs letras, cada una de las cuales representa un nĂşmero. escribiendo
convenciones:
\begin{itemize}
  \item se elimina el signo $\times$ antes de una letra o un corchete:
        $3 \times a = 3a$, $a \times b = ab$,\
        $2 \times (x + 5) = 2(x + 5)$;
  \item pero queda entre dos nĂşmeros: $3 \times 5$ no se puede
        escrito $35$!
  \item $a \times a$ se escribe $a^2$ ("$a$ al cuadrado"),
        $a \times a \times a = a^3$;
  \item $1a$ se escribe $a$ y $0a$ es $0$.
\end{itemize}
\end{definition}

\begin{example}\label{ex:g7:literal:write}
Simplifica la escritura:
\[
5 \times x + 3 \times y = 5x + 3y,
\qquad
a \times 7 = 7a \ \text{(number first)},
\qquad
x \times x \times 4 = 4x^2 .
\]
\end{example}

\begin{example}[Las fĂłrmulas son expresiones literales.]\label{ex:g7:literal:formulas}
Para un rectĂĄngulo de largo $L$ y ancho $w$:
\omterm{def:g6:measure:perimeter}{perĂ­metro} $P = 2(L + w)$, \omterm{def:g6:measure:area}{ĂĄrea} $A = Lw$. Una fĂłrmula resume la
infinitos rectĂĄngulos a la vez --- ese es el objetivo de
letras.
\end{example}

\section{Sustituyendo}

\begin{method}[Evaluando una expresiĂłn]\label{met:g7:literal:substitute}
Para evaluar una expresiĂłn para valores dados de sus letras:
\begin{enumerate}
  \item reescribe la expresiĂłn poniendo de nuevo los signos ocultos $\times$:
        $3a + a^2$ es $3 \times a + a \times a$;
  \item reemplaza cada letra por su valor, manteniendo corchetes alrededor
        si es necesario;
  \item computa, respetando las prioridades de
        \cref{ch:g7:priorities}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:literal:substitute}
EvalĂşe $E = 3x + x^2$ para $x = 5$:
\[
E = 3 \times 5 + 5 \times 5 = 15 + 25 = 40 .
\]
EvalĂşe $F = 2(a + 3b)$ para $a = 4$ y $b = 0.5$:
\[
F = 2 \times (4 + 3 \times 0.5) = 2 \times (4 + 1.5) = 2 \times 5.5
= 11 .
\]
\end{example}

\section{Reducir expresiones}

\begin{proposition}[RecopilaciĂłn de tĂŠrminos similares]\label{prop:g7:literal:reduce}
Los tĂŠrminos que contienen la misma parte de letra se pueden recopilar agregando sus
partes numĂŠricas (esto es distributividad,
\cref{thm:g7:priorities:distributivity}, leĂ­do al revĂŠs):
\[
5x + 3x = (5 + 3)x = 8x,
\qquad
7a - 2a = 5a .
\]
TĂŠrminos con diferentes partes de letras (como $3x$ y $5y$, o $x$ y
$x^2$) no se puede recopilar.
\end{proposition}

\begin{example}\label{ex:g7:literal:reduce}
Reducir $E = 4x + 7 + 3x + 2$, ordenando los tĂŠrminos:
\[
E = (4x + 3x) + (7 + 2) = 7x + 9 .
\]
Cuidado: ÂĄ$7x + 9$ es \emph{no} $16x$! Prueba con $x = 1$:
$4 + 7 + 3 + 2 = 16$ y $7 \times 1 + 9 = 16$ estĂĄn de acuerdo, pero $16x$
tambiĂŠn ser $16$ --- prueba con $x = 2$: el original da
$8 + 7 + 6 + 2 = 23$ y $7 \times 2 + 9 = 23$, mientras
$16 \times 2 = 32$. La reducciĂłn $7x + 9$ es correcta, $16x$ es incorrecta.
\end{example}

\begin{method}[Probando una igualdad]\label{met:g7:literal:test}
Para comprobar si dos expresiones pueden ser iguales para cada valor:
\begin{enumerate}
  \item sustituye el mismo valor en ambos, al menos dos veces (p. ej.\
        $x = 2$ y $x = 10$; Evite $0$ y $1$, esconden demasiados.
        \omterm{ex:g1:subtraction:difference}{diferencias});
  \item si los resultados difieren aunque sea una vez, las expresiones son
        \emph{no} igual --- un contraejemplo anula una afirmaciĂłn general;
  \item si los resultados concuerdan, la igualdad es solo \emph{plausible}: a
        la prueba necesita ĂĄlgebra (expandir, reducir).
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:literal:test}
Âż$2(x + 3)$ es igual a $2x + 6$ por cada $x$? La ampliaciĂłn lo demuestra:
$2(x+3) = 2x + 2 \times 3 = 2x + 6$. Âż$(x + 1)^2$ es igual a
$x^2 + 1$? Prueba $x = 2$: $(2+1)^2 = 9$ pero $2^2 + 1 = 5$. No ---
y un contraejemplo es una prueba completa del "no".
\end{example}

\section{Producir expresiones}

\begin{example}[De la situaciĂłn a la expresiĂłn]\label{ex:g7:literal:produce}
Un taxi cuesta $4$ euros mĂĄs $2$ euros por kilĂłmetro. para un viaje de
$k$ kilĂłmetros el precio es
\[
p = 4 + 2k \quad\text{euros}.
\]
Para $k = 7$: $p = 4 + 14 = 18$ euros. La expresiĂłn \emph{is} la
respuesta general; la sustituciĂłn produce cada uno en particular.
\end{example}

\begin{example}[Trucos de nĂşmeros]\label{ex:g7:literal:trick}
``Piensa en un nĂşmero, duplĂ­calo, suma $10$, divide por $2$, resta tu
nĂşmero.'' Llama al nĂşmero $n$ y sigue los pasos:
\[
n \ \to\ 2n \ \to\ 2n + 10 \ \to\ \frac{2n + 10}{2} = n + 5 \ \to\
n + 5 - n = 5 .
\]
Todo el mundo recibe $5$, sea lo que sea que fuera $n$ --- las cartas explican el
magia.
\end{example}
