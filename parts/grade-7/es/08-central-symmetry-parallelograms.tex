\chapter{Simetría central y paralelogramos}\label{ch:g7:central}

Girar una figura por \omterm{def:g3:division:half}{medio} una vuelta alrededor de un punto: así es \emph{\omterm{def:g7:central:def}{céntrico
simetría}}, la segunda transformación del libro después de las reflexiones
de \cref{ch:g6:symmetry}. Su figura estrella es el \omterm{def:g7:central:parallelogram}{paralelogramo} --- el
\omterm{def:g4:geometry:polygon}{cuadrilátero} es decir \omterm{def:g7:central:def}{simétrico} sobre su propio punto de cruce
diagonales.

\section{simetría central}

\begin{definition}[Simétrica respecto a un punto]\label{def:g7:central:def}
El \emph{simétrico}\index{simetría central} de un punto $M$ sobre un
el punto $O$ es el punto $M'$ tal que $O$ es el \emph{punto medio} de
$[MM']$. La figura simétrica es en lo que se convierte el original después de un
media vuelta ($180^\circ$) alrededor de $O$.
\end{definition}

\begin{method}[Construyendo la simétrica de un punto]\label{met:g7:central:construct}
Para construir el \omterm{def:g7:central:def}{simétrico} $M'$ de $M$ sobre $O$:
\begin{enumerate}
  \item dibuje la línea $(MO)$ y extiéndala más allá de $O$;
  \item medida $MO$ (regla o compás);
  \item coloque $M'$ en la extensión con $OM' = OM$.
\end{enumerate}
En papel cuadriculado: cuente los pasos horizontales y verticales desde $M$ hasta
$O$ y repita los pasos \emph{mismo} desde $O$ hasta llegar a $M'$.
\end{method}

\begin{example}[En una cuadrícula]\label{ex:g7:central:grid}
Si $M$ es $3$ cuadrados a la izquierda y $1$ cuadra arriba de $O$, es \omterm{def:g7:central:def}{simétrico}
$M'$ es $3$ cuadrados \emph{bien} y $1$ cuadrado \emph{abajo} de $O$:
\omterm{def:g7:central:def}{simetría central} invierte ambas direcciones a la vez (un reflejo
invierte sólo uno).
\end{example}

\begin{proposition}[Que conserva media vuelta]\label{prop:g7:central:preserved}
\omterm{def:g7:central:def}{simetría central} conserva longitudes, ángulos, \omterm{def:g6:measure:perimeter}{perímetros} y \omterm{def:g6:measure:area}{áreas}; el
La imagen de una línea es una línea \emph{\omterm{def:g6:lines:perp}{paralelo}}, la imagen de una \omterm{def:g6:lines:objects}{segmento} es una
\omterm{def:g6:lines:perp}{paralelo} \omterm{def:g6:lines:objects}{segmento} de la misma longitud, la imagen de un círculo es un círculo
del mismo \omterm{def:g3:shapes:shapes}{radio} (centrado en el \omterm{def:g7:central:def}{simétrico} del centro).
\end{proposition}
\admitted

\begin{remark}
Comparar con reflexiones (\cref{prop:g6:symmetry:preserved}): ambas
preservar longitudes y ángulos; pero el reflejo voltea las figuras
(una ``b'' se convierte en una ``d''), mientras que la media vuelta los mantiene legibles ---
al revés (una ``b'' se convierte en una ``q''). Y a diferencia de un reflejo, un
media vuelta envía cada línea a una línea \omterm{def:g6:lines:perp}{paralelo}.
\end{remark}

\begin{definition}[Centro de simetria]\label{def:g7:central:center}
Un punto $O$ es un \emph{centro de simetría}\index{centro de simetría}
de una figura cuando la media vuelta $O$ envía la figura exactamente
sobre sí mismo. Por ejemplo, las letras S, N, Z tienen un centro de
simetría; un círculo tiene uno (su centro); un triángulo nunca tiene uno.
\end{definition}

\section{paralelogramos}

\begin{definition}[paralelogramo]\label{def:g7:central:parallelogram}
Un \emph{paralelogramo}\index{paralelogramo} es un \omterm{def:g4:geometry:polygon}{cuadrilátero} cuyo
las diagonales se cruzan en su punto medio común. Ese punto de cruce es entonces
a \omterm{def:g7:central:center}{centro de simetría} de la figura: cada \omterm{def:g5:solids:def}{vértice} es la imagen de media vuelta
del opuesto \omterm{def:g5:solids:def}{vértice}.
\end{definition}

\begin{theorem}[Propiedades de un paralelogramo]\label{thm:g7:central:properties}
En un \omterm{def:g7:central:parallelogram}{paralelogramo} $ABCD$ con centro $O$:
\begin{enumerate}
  \item los lados opuestos son \omterm{def:g6:lines:perp}{paralelo}: $(AB) \parallel (CD)$ y
        $(BC) \parallel (AD)$;
  \item los lados opuestos tienen la misma longitud: $AB = CD$ y
        $BC = AD$;
  \item los ángulos opuestos son iguales: $\widehat A = \widehat C$ y
        $\widehat B = \widehat D$.
\end{enumerate}
\end{theorem}

\begin{proof}
La media vuelta $O$ envía $A$ a $C$ y $B$ a $D$ (definición:
$O$ es el punto medio de ambas diagonales). Entonces envía el \omterm{def:g6:lines:objects}{segmento}
$[AB]$ al \omterm{def:g6:lines:objects}{segmento} $[CD]$; por
\cref{prop:g7:central:preserved} la imagen es \omterm{def:g6:lines:perp}{paralelo} a $[AB]$ y
de la misma longitud: puntos 1 y 2. También envía el ángulo en $A$
al ángulo en $C$, y se conservan los ángulos: punto 3.
\end{proof}

\begin{theorem}[Reconociendo un paralelogramo]\label{thm:g7:central:recognize}
Un \omterm{def:g4:geometry:polygon}{cuadrilátero} (sin lados cruzados) es un \omterm{def:g7:central:parallelogram}{paralelogramo} tan pronto como
\emph{uno} de las siguientes retenciones:
\begin{enumerate}
  \item sus diagonales tienen el mismo punto medio (la definición);
  \item sus lados opuestos son \omterm{def:g6:lines:perp}{paralelo} de dos en dos;
  \item dos lados opuestos son \omterm{def:g6:lines:perp}{paralelo} \emph{y} de la misma longitud.
\end{enumerate}
\end{theorem}
\admitted

\begin{example}\label{ex:g7:central:recognize}
$ABCD$ tiene $(AB) \parallel (CD)$ y $AB = CD = 5$ cm. Criterio 3
aplica: $ABCD$ es un \omterm{def:g7:central:parallelogram}{paralelogramo} --- así que sin más
midiendo también conocemos $AD = BC$, $(AD) \parallel (BC)$, y que su
las diagonales se cortan entre sí por la mitad.
\end{example}

\begin{remark}[Paralelogramos especiales]
El rectángulo \omterm{def:g6:shapes:quadrilaterals}{rombo} y el cuadrado de \cref{ch:g6:shapes} son exactamente
el \omterm{def:g7:central:parallelogram}{paralelogramos} con una propiedad extra: diagonales iguales
(rectángulo), \omterm{def:g6:lines:perp}{perpendicular} diagonales (\omterm{def:g6:shapes:quadrilaterals}{rombo}), ambos (cuadrado). el
El argumento de ángulo prometido en \cref{ex:g6:shapes:recognize} ahora funciona:
en un \omterm{def:g7:central:parallelogram}{paralelogramo}, los ángulos consecutivos son suplementarios
(ángulos alternos/correspondientes con el \omterm{def:g6:lines:perp}{paralelas},
\cref{thm:g7:triangles:equalities}), por lo que un \omterm{def:g3:shapes:rightangle}{ángulo recto} obliga a todos
cuatro.
\end{remark}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g7:central:1}
En papel cuadriculado, coloque $O$, luego $M$ cuatro cuadrados a la derecha de $O$ y uno
cuadrar. Construya el \omterm{def:g7:central:def}{simétrico} $M'$ de $M$ sobre $O$, y el
\omterm{def:g7:central:def}{simétrico} de $M$ a través de \emph{línea de cuadrícula vertical} a $O$.
Compara las dos imágenes.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:central:2}
Dibuja un triángulo $ABC$ y un punto $O$ fuera de él. Construir su imagen
$A'B'C'$ por la media vuelta $O$. ¿Qué puedes decir sobre el
longitudes $A'B'$ y $AB$? ¿Sobre las líneas $(A'B')$ y $(AB)$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:central:3}
¿Qué dígitos, escritos como en la pantalla de una calculadora ($0$ a $9$), tienen un
\omterm{def:g7:central:center}{centro de simetría}? ¿Qué letras mayúsculas entre H, A, N, S, T, Z, O?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:central:4}

$NK$, $\widehat M$ y el punto medio de $[LN]$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:central:5}
Construya un \omterm{def:g7:central:parallelogram}{paralelogramo} $ABCD$ a partir de sus diagonales: $AC = 8$ cm y

\begin{exercise}[$\star$]\label{exo:g7:central:6}

\end{exercise}

\section{Problema: el pastel, la mediana y el segundo imposible
centro}

\begin{problem}[{Problema de fin de semana --- medias vueltas en el trabajo: pastel justo
cortes, un límite en las medianas y por qué ninguna figura puede tener dos centros de
simetria}]
\label{pb:g7:central:1}
La media vuelta parece la más simple de las transformaciones, pero
corta pasteles en mitades demostrablemente justas, mide las medianas de
un triángulo, y esconde una pequeña perla de la matemática pura: un
La figura acotada puede tener un \omterm{def:g7:central:center}{centro de simetría}, pero un \emph{nunca
dos}. La prueba utiliza un descubrimiento que harás en la Parte I: dos
medias vueltas, realizadas una tras otra, suman un deslizamiento.

\medskip
\noindent\textbf{Parte I --- Half-turn gymnastics.}
\begin{enumerate}
  \item En papel cuadriculado, tome como centro el origen $O(0, 0)$.
        Usando el conteo de cuadrícula de \cref{ex:g7:central:grid},

  \item Ahora el centro es $C(2, 1)$. Calcular la imagen de
        $M(5, 3)$, y consulta la receta general: cada coordenada

        puntos}.
  \item Los naipes están diseñados para leerse igual al revés

        ``incorrectamente''. Explique por qué, en una tarjeta con \emph{\omterm{def:g3:numbers:evenodd}{extraño}}
        número de pips, un pip debe estar exactamente en el punto de la tarjeta
        centro.
  \item Dos medias vueltas seguidas: tomar los centros $O_1(2, 0)$
        y $O_2(5, 0)$. Enviar el punto $A(1, 1)$ a través del

\noindent\textbf{Parte II --- Cakes and medians.}
\begin{enumerate}[resume]
  \item Sea $O$ el centro de un \omterm{def:g7:central:parallelogram}{paralelogramo}. Explica por qué
        cualquier línea que pasa por $O$ encuentra el límite en dos puntos
        que son \omterm{def:g7:central:def}{simétrico} sobre $O$, y por qué la línea corta el
        \omterm{def:g7:central:parallelogram}{paralelogramo} en dos piezas que estén exactamente media vuelta
        copias entre sí --- por lo tanto de igual \omterm{def:g6:measure:area}{áreas}.
  \item El teorema del panadero: un pastel rectangular se comparte
        perfectamente y de manera justa por \emph{cualquier} corte recto a través de su
        centro. Mejor: un pastel rectangular tiene una forma rectangular.
        agujero (cualquier tamaño, cualquier posición, \omterm{def:g3:numbers:evenodd}{incluso} inclinado). Describe el
        un corte recto que da dos partes con la misma cantidad
        pastel --- y justificarlo con la pregunta 6.
  \item Tres \omterm{def:g5:solids:def}{vértices} de un \omterm{def:g7:central:parallelogram}{paralelogramo} $ABCD$ son $A(0, 0)$,
        $B(6, 1)$ y $C(8, 5)$. Calcular las coordenadas de $D$.
        y del centro $O$.
  \item \Cref{exo:g7:central:9} construido, a partir de un triángulo $ABC$
        y el punto medio $O$ de $[BC]$, el \omterm{def:g7:central:parallelogram}{paralelogramo}
        $ABA'C$ donde $A'$ es el \omterm{def:g7:central:def}{simétrico} de $A$ sobre $O$.
        Úselo, junto con la desigualdad del triángulo.
        (\cref{thm:g7:triangles:inequality}) en el triángulo
        $ABA'$, para demostrar el \emph{límite medio}: la mediana
        $[AO]$ satisface
        \[
        AO < \frac{AB + AC}{2} .
        \]
        (Tenga en cuenta que $AA' = 2\,AO$ y $BA' = AC$).
  \item En un triángulo con $AB = 5$ cm y $AC = 7$ cm, entre

\end{enumerate}

\noindent\textbf{Parte III --- How many centers can a figure
have?}
\begin{enumerate}[resume]
  \item Para cada figura, indique si tiene un \omterm{def:g7:central:center}{centro de
        simetría} y dónde: un \omterm{def:g6:lines:objects}{segmento}; una línea completa; un
        \omterm{def:g6:shapes:triangles}{triangulo equilatero}; un cuadrado; un círculo; las letras s
        y Z.
  \item Acredite su reclamo por el \omterm{def:g6:shapes:triangles}{triangulo equilatero}: a
        media vuelta enviando el triángulo hacia sí mismo se emparejaría
        los tres \omterm{def:g5:solids:def}{vértices} --- pero tres es \omterm{def:g3:numbers:evenodd}{extraño}, entonces algunos \omterm{def:g5:solids:def}{vértice}
        sería enviado a sí mismo. ¿Qué obligaría eso a
        \omterm{def:g5:solids:def}{vértice} ser, ¿y por qué es imposible?
  \item Generalice la pregunta 12: explique por qué \emph{no} \omterm{def:g4:geometry:polygon}{polígono}
        con un \omterm{def:g3:numbers:evenodd}{número impar} de \omterm{def:g5:solids:def}{vértices} tiene un \omterm{def:g7:central:center}{centro de
        simetría}.
  \item La perla: supongamos que una figura tuviera \emph{dos} distinta
        centros de simetría $O_1$ y $O_2$. realizar los dos
        medias vueltas una tras otra y utilizar el descubrimiento de
        pregunta 5: qué movimiento debe enviar la figura exactamente
        ¿Sobre sí mismo? Explica por qué una figura que cabe en una hoja.
        El papel no puede sobrevivir a eso y concluir.
  \item La conclusión de la pregunta 14 deja un vacío legal:
        \emph{ilimitado} cifras. Verificar en un friso infinito
        --- digamos, una franja interminable de huellas, izquierda, derecha,
        left, right\,\dots\ --- que tenga (al menos) dos
        diferentes centros de simetría e identificar los
        traducción que predijo la pregunta 14.
\end{enumerate}
\end{problem}
