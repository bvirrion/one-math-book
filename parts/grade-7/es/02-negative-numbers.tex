\chapter{Números negativos}\label{ch:g7:negatives}

Temperaturas inferiores a \omterm{def:g1:counting:zero}{cero}, pisos bajo rasante, dinero adeudado: muchos
las cantidades son inferiores a $0$. Los números negativos extienden la recta numérica hasta el
izquierda de \omterm{def:g1:counting:zero}{cero}. Este capítulo enseña cómo compararlos y cómo sumarlos.
y restarlos --- su multiplicación espera
\cref{ch:g8:negprod}.

\section{Números relativos}

\begin{definition}[numeros relativos]\label{def:g7:negatives:def}
Un \emph{número relativo}\index{número relativo} está hecho de un letrero.
($+$ o $-$) y \emph{distance to \omterm{def:g1:counting:zero}{cero}}: $+3$ y $-3$ están ambos en
distancia $3$ de $0$, en lados opuestos. Los números positivos suelen ser
escrito sin su signo: $+3 = 3$. Los números $+3$ y $-3$ son
\emph{opuestos}\index{opuesto} el uno del otro.
\end{definition}

\begin{method}[Comparando números relativos]\label{met:g7:negatives:compare}
En la recta numérica, más pequeño significa más a la izquierda. En la práctica:
\begin{enumerate}
  \item un número negativo siempre es menor que uno positivo:
        $-7 < 2$;
  \item entre dos positivos, el orden habitual;
  \item entre dos negativos, el que tiene la distancia \emph{más grande}
        a \omterm{def:g1:counting:zero}{cero} es el \emph{menor}: $-7 < -2$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:negatives:compare}
Ordenar de menor a mayor: $3$; $-5$; $0$; $-1.5$; $2.8$.
Lo negativo primero, lo más negativo:
\[
-5 < -1.5 < 0 < 2.8 < 3 .
\]
\end{example}

\section{Sumar números relativos}

Piense en \omterm{def:g7:negatives:def}{número relativo} como un movimiento a lo largo de la línea: $+5$ significa
``caminar $5$ pasos hacia la derecha'', $-3$ significa ``caminar $3$ pasos hacia la izquierda''. Añadiendo
números significa encadenar los paseos.

\begin{method}[Agregando dos números relativos]\label{met:g7:negatives:add}
\begin{enumerate}
  \item \emph{Mismo signo}: sumar las distancias, mantener el signo común:
        $(-3) + (-4) = -7$.
  \item \emph{signos opuestos}: restar la distancia menor del
        más grande, toma el signo del más grande:
        $(-7) + (+2) = -5$ y $(+7) + (-2) = +5$.
  \item \omterm{def:g7:negatives:def}{Opuestos} cancelar: $(+4) + (-4) = 0$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:negatives:add}
Paso a paso, agrupando como convenga:
\begin{align*}
(-8) + (+3) &= -5 && \text{(opposite signs: } 8 - 3 = 5,
\text{ sign of } 8)\\
(-2.5) + (-1.5) &= -4 && \text{(same sign: add distances)}\\
(+6) + (-9) + (+3) &= (-3) + (+3) = 0 .
\end{align*}
\end{example}

\section{Restar números relativos}

\begin{theorem}[Regla de resto]\label{thm:g7:negatives:sub}
Restar un \omterm{def:g7:negatives:def}{número relativo} es lo mismo que \emph{agregando su
opuesto}:
\[
a - b = a + (-b) .
\]
\end{theorem}

\begin{proof}[¿Por qué funciona?]
$a - b$ es el número que sumado a $b$ da $a$. comprueba eso
$a + (-b)$ hace el trabajo:
$\bigl(a + (-b)\bigr) + b = a + \bigl((-b) + b\bigr) = a + 0 = a$.
\end{proof}

\begin{example}\label{ex:g7:negatives:sub}
Toda resta se convierte en una suma, entonces
\cref{met:g7:negatives:add} se aplica:
\begin{align*}
5 - 9 &= 5 + (-9) = -4, \\
(-3) - (+4) &= (-3) + (-4) = -7, \\
(-2) - (-6) &= (-2) + (+6) = +4 .
\end{align*}
La última línea es la famosa: \emph{restar un negativo significa
añadiendo}. ¡Eliminar una deuda te hace más rico!
\end{example}

\begin{example}[Diferencias de temperatura]\label{ex:g7:negatives:temperature}
En un día la temperatura pasó de $-6$ grados a $+9$ grados. el
subida es el \omterm{ex:g1:subtraction:difference}{diferencia}
\[
9 - (-6) = 9 + 6 = 15 \text{ degrees}.
\]
En la recta numérica: desde $-6$ hasta $0$ son $6$ grados, desde $0$ hasta
$9$ es $9$ más.
\end{example}

\section{Coordenadas en todo el plano.}

\begin{definition}[Coordenadas con signos]\label{def:g7:negatives:coords}
Con dos ejes graduados \omterm{def:g6:lines:perp}{perpendicular} que se cruzan en el origen $O$,
cada punto del plano tiene dos coordenadas relativas $(x, y)$:
negativo $x$ significa a la izquierda del eje vertical, negativo $y$ significa debajo
el eje horizontal.
\end{definition}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g7:negatives:1}
Dar lo contrario de: $7$; $-3.5$; $0$; $+12$. Luego completa:
``¿dos números son \omterm{def:g7:negatives:def}{opuestos} cuando su \omterm{def:g1:addition:def}{suma} es~?''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:2}
Copiar y completar con $<$ o $>$:
\[
-4 \;?\; 1, \qquad
-4 \;?\; -7, \qquad
-2.5 \;?\; -2.4, \qquad
0 \;?\; -0.1 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:3}
Ordenar de menor a mayor: $-3$; $2.5$; $-0.5$; $-3.1$;
$0$; $1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:4}
Calcular:
\[
(-5) + (-9), \qquad
(-12) + (+7), \qquad
(+3.5) + (-1.5), \qquad
(-6) + (+6).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:5}
Transformar en adiciones, luego calcular:
\[
4 - 11, \qquad
(-5) - (+3), \qquad
(-7) - (-10), \qquad
0 - (-8).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:6}
Calcule agrupando inteligentemente (\omterm{def:g7:negatives:def}{opuestos} primero):
\[
(+7) + (-12) + (+3) + (-8), \qquad
(-5.5) + (+9) + (-4.5) + (+1).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:7}
La temperatura era de $-3$ grados esta mañana; subió $8$ grados
durante el día, y luego bajó $12$ grados durante la noche. escribe un solo
cálculo que dé la temperatura final y calcule.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:8}
Trazar en un sistema de coordenadas: $A(2, 3)$; $B(-3, 1)$; $C(-2, -2)$;
$D(3, -1)$; $E(0, -3)$. ¿Qué punto se encuentra sobre un eje?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:negatives:9}
Calcule el \omterm{ex:g1:subtraction:difference}{diferencia} entre la temperatura más alta y más baja:
Moscú en enero, máximo $-4$ grados, mínimo $-13$ grados; Helsinki, máx.
$2$, min $-9$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:negatives:10}
Un buzo se encuentra a una altitud $-12$ m (doce metros bajo la superficie). ella
sube $7$ m, luego desciende $4$ m. ¿A qué altura se encuentra? ¿Qué tan lejos?
¿Aún debe elevarse para llegar a la superficie?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:negatives:11}
Parcela $A(-2, 1)$, $B(2, 1)$, $C(2, -2)$. Encuentra las coordenadas del
apunte $D$ para que $ABCD$ sea un rectángulo y calcule su \omterm{def:g6:measure:perimeter}{perímetro}
(cuente los cuadrados de la cuadrícula para las longitudes de los lados).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:negatives:12}
En el cuadrado mágico de abajo, cada fila, columna y diagonal debe tener

\hline
$-3$ & $-2$ & $?$ \\

\end{center}

completo.)
\end{exercise}

\section{Problema: El año cero que nunca existió}

\begin{problem}[{Problema de fin de semana --- números relativos en el
línea de tiempo de la historia: duraciones a través del límite de la era, el
el truco de los astrónomos y el principio de cambio neto}]
\label{pb:g7:negatives:1}
La recta numérica de este capítulo tiene un gemelo famoso en el mundo real: el
línea de tiempo de la historia, con los años antes de Cristo extendiéndose hacia la izquierda y el
años d.C. extendiéndose hacia la derecha. Pero la línea de los historiadores esconde una trampa.
que ha estropeado muchos cálculos: \emph{there is no year \omterm{def:g1:counting:zero}{cero}}
--- el año 1 a.C. es seguido inmediatamente por el año 1 d.C. Este problema
calcula temperaturas, ascensores y cuentas bancarias, luego repara
La historia está rota \omterm{def:g1:counting:zero}{cero} como lo hacen los astrónomos: con \omterm{def:g7:negatives:def}{relativo
numeros}.

\medskip
\noindent\textbf{Parte I --- Warm-ups, and a first trap.}
\begin{enumerate}
  \item La cumbre del Mont Blanc está a la altitud $+4\,808$ m; el
        orilla del Mar Muerto a $-430$ m. Calcular el \omterm{ex:g1:subtraction:difference}{diferencia}
        de altitud entre ellos.
  \item Una cuenta bancaria tiene $-45$ euros (un descubierto). su
        el propietario deposita $120$ euros, paga una factura de $60$ euros,
        luego deposita $15$ euros. Escribe el saldo como un solo
        cadena de sumas y calcularla. ¿Qué depósito único?
        habría devuelto el $-45$ original a exactamente
        $0$?
  \item Julio César nació en $100$ a.C. y murió en $44$ a.C.
        Contando los años antes de Cristo como números negativos ($100$ BC como
        $-100$, $44$ BC como $-44$), calcular $(-44) - (-100)$: cómo
        ¿Cuántos años vivió César?
  \item El emperador Augusto nació en $63$ a.C. y murió en d.C.
        $14$. La misma receta da $14 - (-63) = 77$ años ---
        pero los historiadores insisten en que murió en $76$. Su objeción:
        el conteo $77$ recorre un año que nunca existió.
        ¿Cuál?
  \item Los astrónomos reparan la línea de tiempo reetiquetándola: años d.C.
        mantener su número ($+14$ permanece $+14$), pero $1$ BC
        se convierte en $0$, $2$ BC se convierte en $-1$ y, en general, el año
        $n$ BC se convierte en $-(n - 1)$. Convierta $63$ BC y rehaga el
        Cálculo de la pregunta 4 con los números de los astrónomos.
        ¿La respuesta satisface ahora a los historiadores?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Computing across the eras.}
De ahora en adelante, utilice el reetiquetado de los astrónomos para cada
cálculo, convirtiéndolo nuevamente a BC/AD al final (un año etiquetado
$0$ o negativo $-m$ es el año de los historiadores $m + 1$ a.C.).
\begin{enumerate}[resume]
  \item Convertir a números de astrónomos: $1$ BC; $10$ antes de Cristo; ANUNCIO
        $476$; y $753$ a.C., la legendaria fundación de Roma.
  \item Cuantos años pasaron desde la fundación de Roma
        ($753$ aC) hasta la caída del Imperio Occidental (AD $476$)?
  \item Un cometa regresa cada $76$ años y apareció en
        $12$ antes de Cristo. Dé los años de los historiadores de sus siguientes tres
        apariciones, y de la aparición justo antes del $12$ a.C.
  \item ¿Cuántos años pasaron de $5$ a.C. a $5$ d.C.? (El
        la respuesta tentadora es $10$; la línea de tiempo dice lo contrario).
  \item Un niño nacido en $3$ aC cumplió diez años en el cual
        año (a.C. o d.C.)?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The net-change principle.}
\begin{enumerate}[resume]
  \item El \emph{distancia} entre dos números en la línea es
        el más grande menos el más pequeño --- siempre positivo.
        Calcule la distancia entre $-7$ y $-2$, luego entre
        $-3.5$ y $4.5$.
  \item Un ascensor parte del piso $0$ y realiza los movimientos $+3$,
        $-5$, $+2$, $-1$, $+4$. Calcular el piso final por
        agrupando inteligentemente (\cref{met:g7:negatives:add},
        \omterm{def:g7:negatives:def}{opuestos} primero cuando sea posible). ¿Realizaría el
        se mueve en un orden diferente ¿cambia el destino? ¿Por qué?
  \item Día a día, la temperatura cambia en $+2$, $-4$, $+1$,
        $0$, $-3$, $+5$, $-2$ grados durante una semana. Calcular el
        \emph{\omterm{def:g5:solids:net}{neto} change}. Si la semana comenzó en $-2$ grados,
        donde termino? Indique el principio: valor final $=$
        valor inicial $+$ (\omterm{def:g1:addition:def}{suma} de todos los cambios).
  \item La caminata de una excursionista la devuelve exactamente al punto de partida.
        altitud. ¿Qué debe hacer el \omterm{def:g1:addition:def}{suma} de todos sus cambios de altitud?
        ser, y ¿por qué? De sus seis cambios registrados, cinco son
        $+4$, $-7$, $+2$, $+3$, $-5$ (en cientos de metros):
        Encuentra el sexto.
  \item El final, de vuelta en Roma. Un historiador calcula: "la
        La República corrió desde $509$ a.C. hasta $27$ a.C., luego el Imperio
        al AD $476$; en total, desde $509$ a.C. hasta $476$ d.C., es decir
        $509 + 476 = 985$ años de estado romano.'' Rehacer el
        cálculo con números de astrónomos. cual es el
        total correcto --- y qué año, una vez más, causó la
        error?
\end{enumerate}
\end{problem}
