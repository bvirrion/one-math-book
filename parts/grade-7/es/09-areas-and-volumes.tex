\chapter{Áreas y volúmenes}\label{ch:g7:areas}

Rectángulos medidos de grado 6, \omterm{def:g6:shapes:triangles}{triangulos rectángulos} y cajas
(\cref{ch:g6:measure}). Con unas tijeras --- corta un trozo aquí,

\section{Área de un paralelogramo}

\begin{theorem}[paralelogramo]\label{thm:g7:areas:parallelogram}
A \omterm{def:g7:central:parallelogram}{paralelogramo} con un lado de longitud $b$ (su \emph{base}) y
distancia $h$ entre ese lado y el opuesto (su
\emph{altura}) tiene \omterm{def:g6:measure:area}{área}
\[
A = b \times h .
\]
\end{theorem}

\begin{proof}[Prueba con tijeras]
Corta el triángulo \omterm{def:g6:shapes:triangles}{en ángulo recto} que sobresale de un lado y pégalo.
por el otro: el \omterm{def:g7:central:parallelogram}{paralelogramo} se convierte en un rectángulo $b \times h$ con
el mismo \omterm{def:g6:measure:area}{área}.
\end{proof}

\begin{remark}
La altura es la distancia \emph{\omterm{def:g6:lines:perp}{perpendicular}} entre las dos bases,
¡No la longitud del lado inclinado! Un \omterm{def:g7:central:parallelogram}{paralelogramo} muy inclinado puede
tienen lados largos y un pequeño \omterm{def:g6:measure:area}{área}.
\end{remark}

\section{Área de un triángulo}

\begin{theorem}[Triángulo]\label{thm:g7:areas:triangle}
Un triángulo con base $b$ y altura correspondiente $h$ (el
\omterm{def:g6:lines:perp}{perpendicular} distancia desde el \omterm{def:g5:solids:def}{vértice} opuesto a la línea base) tiene
\omterm{def:g6:measure:area}{área}
\[
A = \frac{b \times h}{2} .
\]
\end{theorem}

\begin{proof}
Dos copias del triángulo, una girada media vuelta alrededor del
punto medio de un lado, encajan en un \omterm{def:g7:central:parallelogram}{paralelogramo} con base $b$
y altura $h$ (es decir, \omterm{def:g7:central:def}{simetría central} en el trabajo,
\cref{prop:g7:central:preserved}). El triángulo es \omterm{def:g3:division:half}{medio} del mismo:
$\frac{bh}{2}$.
\end{proof}

  \draw[thick] (0,0) -- (4,0) -- (1.4,2.1) -- cycle;

  \node[below, font=\small] at (2,0) {$b$};
  \node[right, font=\small] at (1.42,1.0) {$h$};
\end{tikzpicture}

{\small Un triángulo y su copia medio vuelta (discontinua) forman una
\omterm{def:g7:central:parallelogram}{paralelogramo}: el \omterm{def:g6:measure:area}{área} del triángulo es $\frac{b\times h}{2}$. Cualquiera de los
tres lados pueden servir como base --- con altura \emph{su propio}.}
\end{omfigure}

\begin{example}\label{ex:g7:areas:triangle}
Un triángulo tiene una base de $9$ cm y la altura correspondiente mide
$4$cm:
\[
A = \frac{9 \times 4}{2} = \frac{36}{2} = 18 \text{ cm}^2 .
\]
Para un triángulo \emph{bien}, los dos catetos son una base y su altura:
recuperamos la fórmula de \cref{prop:g6:measure:formulas}.
\end{example}

\section{Área de un disco}

\begin{theorem}[Disco]\label{thm:g7:areas:disk}
Un disco de \omterm{def:g3:shapes:shapes}{radio} $r$ tiene \omterm{def:g6:measure:area}{área}
\[
A = \pi r^2 .
\]
\end{theorem}

\begin{proof}[Idea de prueba]
Cortar el disco en muchas rodajas finas iguales y colocarlas de la cabeza a la cola:
casi forman un \omterm{def:g7:central:parallelogram}{paralelogramo} de altura $r$ cuya base es \omterm{def:g3:division:half}{medio} el
circunferencia, $\pi r$ (\cref{prop:g6:measure:circle}). Su \omterm{def:g6:measure:area}{área} es
cerca de $\pi r \times r = \pi r^2$, y la aproximación se vuelve
Perfecto a medida que las rodajas se vuelven más finas. Una prueba completa necesita integral
Cálculo, del volumen High School.
\end{proof}

{\small Cortar el disco y desenrollar las rodajas: casi un
\omterm{def:g7:central:parallelogram}{paralelogramo} de base $\pi r$ (\omterm{def:g3:division:half}{medio} el borde) y altura $r$.}
\end{omfigure}

\begin{example}\label{ex:g7:areas:disk}
Un tablero de mesa redondo tiene \omterm{def:g3:shapes:shapes}{radio} $60$ cm. Su \omterm{def:g6:measure:area}{área} es
$\pi \times 60^2 = 3600\pi \approx 11\,310$ cm$^2$, un poco más que
$1.1$ m$^2$. Mira el \omterm{ex:g1:subtraction:difference}{diferencia}: usos del \emph{\omterm{def:g6:measure:perimeter}{perímetro}} $2\pi r$
$r$, el \emph{\omterm{def:g6:measure:area}{área}} $\pi r^2$ usa $r^2$.
\end{example}

\section{Prismas y cilindros}

\begin{definition}[prisma, cilindro]\label{def:g7:areas:prism}
Un \emph{prisma}\index{prisma} es un \omterm{def:g5:solids:def}{sólido} con dos \omterm{def:g6:lines:perp}{paralelo} idénticos.
poligonal \omterm{def:g5:solids:def}{caras} (el \emph{bases}) unido por rectángulos; un
\emph{cilindro}\index{cilindro} ocurre lo mismo con los discos como bases. el
\emph{altura} es la distancia entre las dos bases.
\end{definition}

\begin{theorem}[Volúmenes]\label{thm:g7:areas:volume}
Para un \omterm{def:g7:areas:prism}{prisma} o un \omterm{def:g7:areas:prism}{cilindro} con base \omterm{def:g6:measure:area}{área} $B$ y altura $h$:
\[
V = B \times h .
\]
\end{theorem}
\admitted

\begin{example}\label{ex:g7:areas:volume}
Una tienda de campaña es una \omterm{def:g7:areas:prism}{prisma} acostada de lado: sus bases son triángulos de base
$2$ m y altura $1.5$ m, y la tienda mide $3$ m de largo.
\begin{enumerate}
  \item Base \omterm{def:g6:measure:area}{área}: $B = \frac{2 \times 1.5}{2} = 1.5$ m$^2$.
  \item \omterm{def:g6:measure:volume}{Volumen}: $V = B \times h = 1.5 \times 3 = 4.5$ m$^3$.
\end{enumerate}
Una lata de \omterm{def:g3:shapes:shapes}{radio} $4$ cm y altura $11$ cm:
$V = \pi \times 4^2 \times 11 = 176\pi \approx 553$ cm$^3$, aproximadamente
\omterm{def:g3:division:half}{medio} un litro.
\end{example}

\section{Problema: el cuadrado que desaparece}

\begin{problem}[{Problema del fin de semana --- una famosa ``prueba'' que
$64 = 65$, desenmascarado por zonas y pistas, con pizzas y anillas para
postre}]
\label{pb:g7:areas:1}
Un mago corta un cuadrado $8 \times 8$ en cuatro pedazos, lo desliza
alrededor y los vuelve a ensamblar en un rectángulo $5 \times 13$.
La audiencia cuenta: $8 \times 8 = 64$, pero
$5 \times 13 = 65$ --- una unidad cuadrada ha aparecido de la nada
aire! Las fórmulas \omterm{def:g6:measure:area}{área} de este capítulo son exactamente las herramientas que
capta el truco. El problema entonces plantea un razonamiento honesto \omterm{def:g6:measure:area}{área}.
trabajo: sobre trapecios, sobre pilas inclinadas, sobre economía de la pizza --- y
finalmente regresa con el mago por un segundo, \omterm{def:g3:numbers:evenodd}{incluso} extraño
rendimiento.

\medskip
\noindent\textbf{Parte I --- The trick, performed and unmasked.}
Las cuatro piezas del cuadrado $8 \times 8$ son: dos \omterm{def:g6:shapes:triangles}{correcto
triangulos} con catetos $8$ y $3$, y dos trapecios rectos con
\omterm{def:g6:lines:perp}{paralelo} lados $5$ y $3$ y altura $5$.
\begin{enumerate}
  \item Calcular el \omterm{def:g6:measure:area}{área} del cuadrado original y el \omterm{def:g6:measure:area}{área} del
        el rectángulo reclamado. Exprese el escándalo en una línea.
  \item Calcular el \omterm{def:g6:measure:area}{área} de cada una de las cuatro piezas.
        (\cref{thm:g7:areas:triangle}; para los trapecios, corte
        o utilice \cref{exo:g7:areas:9}).
  \item Añade los cuatro \omterm{def:g6:measure:area}{áreas}. ¿Cuál de las dos figuras --- cuadrado
        o rectángulo --- ¿se pueden llenar realmente las piezas?
  \item En la disposición del rectángulo, la hipotenusa de un triángulo
        (aumentando $3$ a través de $8$) se supone que continúa un
        borde inclinado del trapezoide (que se eleva $2$ a través de $5$) en uno
        recta ``diagonal''. Pruebe la alineación: son $3$ de ancho
        $8$ y $2$ en $5$ con la misma inclinación
        (\cref{thm:g7:prop:cross})?
  \item Explica dónde se esconde el cuadrado sexagésimo quinto: qué forma tiene
        el espacio a lo largo de la falsa diagonal, ¿cuál es su exactitud \omterm{def:g6:measure:area}{área}?
        ¿Y por qué el ojo no lo ve? (¿Qué ancho, en promedio, tiene un
        astilla de ese \omterm{def:g6:measure:area}{área} estirada sobre el rectángulo
        diagonal, aproximadamente $13$ unidades de largo?)
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Honest \omterm{def:g6:measure:area}{área} reasoning.}
\begin{enumerate}[resume]
  \item Dibujar dos \omterm{def:g6:lines:perp}{lineas paralelas} $4$ cm de distancia, una base de $6$ cm en
        uno de ellos, y tres triángulos diferentes en esa base
        con vértices en varios puntos de la otra línea. calcular
        los tres \omterm{def:g6:measure:area}{áreas}. ¿Por qué son todos iguales, sin importar cuál sea el vértice?
        se desliza a lo largo de su línea?
  \item Demuestre la fórmula del trapezoide en general: un trapezoide con
        \omterm{def:g6:lines:perp}{paralelo} lados $B$ y $b$ y altura $h$, cortados a lo largo de
        diagonal, da dos triángulos de la misma altura $h$.
        Concluir:
        \[
        A = \frac{(B + b) \times h}{2} .
        \]
  \item Aplicarlo a un trapezoide de $B = 9$ cm, $b = 5$ cm,
        $h = 4$cm.
  \item Los carpinteros usan la misma fórmula disfrazada: \omterm{def:g6:measure:area}{área} $=$
        (promedio de los dos lados \omterm{def:g6:lines:perp}{paralelo}) Altura $\times$.
        Explique por qué esta es la misma regla y vuelva a calcular la pregunta.
        8 de esa manera.
  \item Incline una pila ordenada de naipes en una pila inclinada.
        Explique por qué el \omterm{def:g6:measure:volume}{volumen} no ha cambiado y qué
        dice sobre la altura en la fórmula \omterm{def:g7:areas:prism}{prisma}
        $V = B \times h$ (\cref{thm:g7:areas:volume}): cuál
        altura que se debe tomar para una pila inclinada --- la inclinación
        ¿Largo o el vertical? (Comparar
        \cref{exo:g7:areas:2}, una dimensión hacia arriba.)
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Rings, pizzas, and the magician's
return.}
\begin{enumerate}[resume]
  \item Un estanque circular de \omterm{def:g3:shapes:shapes}{radio} $3$ m se encuentra en el centro de un
        Césped circular de \omterm{def:g3:shapes:shapes}{radio} $5$ m. Calcule el \omterm{def:g6:measure:area}{área} del
        anillo de pasto (exactamente con $\pi$, luego redondeado a la
        m$^2$).
  \item Encuentre el \omterm{def:g3:shapes:shapes}{radio} del disco único cuyo \omterm{def:g6:measure:area}{área} es igual
        Ese anillo es \omterm{def:g6:measure:area}{área} exactamente. (Los tres radios que tienes ahora
        forman un famoso triple, cuya historia se cuenta en
        \cref{ch:g8:pythagoras}.)
  \item Cuál es más pizza: una pizza de \omterm{def:g4:geometry:circle}{diámetro} $40$ cm, o
        dos pizzas de \omterm{def:g4:geometry:circle}{diámetro} $28$ cm? Calcula y compara.
  \item A \emph{doble} \omterm{def:g6:measure:area}{área} de una pizza, aproximadamente en qué factor
        ¿Debe crecer su \omterm{def:g3:shapes:shapes}{radio}? Verifique el candidato $1.4$ (recuerde
        que \omterm{def:g6:measure:area}{áreas} siga el \emph{cuadrado} del escalamiento
        factor, \cref{pb:g6:propdata:1}), y diga dónde
        factor exacto --- el número cuyo cuadrado es $2$ --- hace
        su entrada oficial
        (\cref{ex:g8:pythagoras:diagonal}).
  \item El mago regresa: corta un cuadrado $13 \times 13$
        con los números $5$, $8$, $13$ desempeñando los roles que
        $3$, $5$, $8$ jugado antes y vuelve a ensamblar un
        $8 \times 21$ rectángulo. Calcule ambos \omterm{def:g6:measure:area}{áreas}: ¿qué
        desaparece esta vez? Los números $3, 5, 8, 13, 21$ son
        términos consecutivos de la secuencia de
        \cref{pb:g7:fractions:1} --- indica el patrón del
        Las ganancias y pérdidas del mago, y la moraleja: ¿por qué \omterm{def:g6:measure:area}{áreas}?
        ¿Nunca mientes?
\end{enumerate}
\end{problem}
