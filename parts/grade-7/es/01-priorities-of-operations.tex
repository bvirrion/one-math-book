\chapter{Prioridades de operaciones}\label{ch:g7:priorities}

¿Qué es $3 + 4 \times 5$? Si calcula de izquierda a derecha obtendrá $35$;
si multiplicas primero obtienes $23$. Las matemáticas necesitan una sola
respuesta, por lo que existen reglas de tráfico para los cálculos: las prioridades.
Este capítulo los establece e introduce una primera identidad algebraica,
distributividad.

\section{Las reglas de prioridad}

\begin{theorem}[Reglas de prioridad]\label{thm:g7:priorities:rules}
En un cálculo sin paréntesis:
\begin{enumerate}
  \item se realizan multiplicaciones y divisiones \emph{antes}
        sumas y restas;
  \item las operaciones del mismo nivel se realizan de izquierda a derecha.
\end{enumerate}
Los corchetes anulan todo: se calcula lo que está dentro de los corchetes
Primero, empezando por los más internos.
\end{theorem}

\begin{example}\label{ex:g7:priorities:basic}
Un paso por línea, subrayando (mentalmente) lo que se calcula:
\begin{align*}
3 + 4 \times 5 &= 3 + 20 = 23
&&\text{(multiplication first)}\\
20 - 8 + 2 &= 12 + 2 = 14
&&\text{(same level: left to right)}\\
18 \div 3 \times 2 &= 6 \times 2 = 12
&&\text{(same level: left to right)}\\
(3 + 4) \times 5 &= 7 \times 5 = 35
&&\text{(brackets first).}
\end{align*}
Tenga en cuenta las trampas: $20 - 8 + 2$ es \emph{no} $20 - 10$, y
$18 \div 3 \times 2$ es \emph{no} $18 \div 6$.
\end{example}

\begin{example}[corchetes anidados]\label{ex:g7:priorities:nested}
\begin{align*}
50 - 2 \times \bigl(3 + (10 - 4)\bigr)
&= 50 - 2 \times (3 + 6) && \text{(innermost bracket)}\\
&= 50 - 2 \times 9 && \text{(remaining bracket)}\\
&= 50 - 18 && \text{(multiplication)}\\
&= 32 .
\end{align*}
\end{example}

\begin{remark}[Las barras de fracciones ocultan corchetes]
Una barra \omterm{def:g6:fractions:def}{fracción} agrupa su \omterm{def:g4:fractions:def}{numerador} y su \omterm{def:g4:fractions:def}{denominador}, como si cada uno
tenía corchetes invisibles:
\[
\frac{12 + 8}{4} = \frac{20}{4} = 5,
\qquad\text{not}\quad 12 + \frac{8}{4} = 14 .
\]
\end{remark}

\begin{method}[Informática segura]\label{met:g7:priorities:safe}
\begin{enumerate}
  \item Localice la operación a realizar primero (paréntesis más internos, luego
        $\times$ y $\div$, luego $+$ y $-$);
  \item calcula \emph{solo ese} y reescribe la expresión completa.
        con el resultado en su lugar;
  \item repetir hasta que quede un número --- una operación por línea, no
        atajos.
\end{enumerate}
\end{method}

\section{Distributividad}

\begin{theorem}[distributividad]\label{thm:g7:priorities:distributivity}
Para todos los números $k$, $a$, $b$:
\[
k \times (a + b) = k \times a + k \times b,
\qquad
k \times (a - b) = k \times a - k \times b .
\]
\end{theorem}

\begin{proof}[Prueba mediante una imagen]
Cuente el \omterm{def:g6:measure:area}{área} de un rectángulo de ancho $k$ dividido en dos más pequeños
rectángulos de longitud $a$ y $b$: en su conjunto es $k \times (a+b)$;
en dos partes es $k \times a + k \times b$. Mismo rectángulo, mismo
\omterm{def:g6:measure:area}{área}.
\end{proof}

\begin{omfigure}
\begin{tikzpicture}[scale=0.75]
  \fill[omDef!14] (0,0) rectangle (4.2,1.8);
  \fill[omThm!14] (4.2,0) rectangle (6.4,1.8);
  \draw[thick] (0,0) rectangle (6.4,1.8);
  \draw[thick] (4.2,0) -- (4.2,1.8);
  \node[below, font=\small] at (2.1,0) {$a$};
  \node[below, font=\small] at (5.3,0) {$b$};
  \node[left, font=\small] at (0,0.9) {$k$};
  \node[font=\small] at (2.1,0.9) {$k \times a$};
  \node[font=\small] at (5.3,0.9) {$k \times b$};
\end{tikzpicture}

{\small Distributividad como \omterm{def:g6:measure:area}{áreas}: el rectángulo grande
$k \times (a + b)$ es el \omterm{def:g1:addition:def}{suma} de los dos más pequeños.}
\end{omfigure}

\begin{example}[Ambas direcciones]\label{ex:g7:priorities:distributivity}
\emph{En expansión} (de izquierda a derecha) facilita el cálculo mental:
\[
7 \times 103 = 7 \times (100 + 3) = 700 + 21 = 721,
\qquad
6 \times 98 = 6 \times (100 - 2) = 600 - 12 = 588 .
\]
\emph{Factorización} (de derecha a izquierda) recoge un factor común:
\[
23 \times 12 + 23 \times 8 = 23 \times (12 + 8) = 23 \times 20 = 460 .
\]
\end{example}

\begin{example}[Traducir palabras a un calculo]\label{ex:g7:priorities:words}
``El \omterm{def:g1:addition:def}{suma} de $8$ y el \omterm{def:g2:mult:def}{producto} de $5$ por $3$'' es $8 + 5 \times 3 =
23$. ``El \omterm{def:g2:mult:def}{producto} de $5$ por el \omterm{def:g1:addition:def}{suma} de $8$ y $3$'' es
$5 \times (8 + 3) = 55$: las palabras ``el \omterm{def:g2:mult:def}{producto} de \dots\ por el \omterm{def:g1:addition:def}{suma}
of\dots'' coloque corchetes alrededor del \omterm{def:g1:addition:def}{suma}.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g7:priorities:1}
Calcular, una operación por línea:
\[
5 + 3 \times 6, \qquad
(5 + 3) \times 6, \qquad
30 - 12 \div 4, \qquad
(30 - 12) \div 4 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:2}
Calcular:
\[
24 - 6 + 2, \qquad
24 \div 6 \times 2, \qquad
24 - 6 \times 2, \qquad
24 \div (6 \times 2).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:3}
Calcular paso a paso:
\[
7 + 3 \times (10 - 6), \qquad
40 - (2 + 3) \times 4, \qquad
60 \div \bigl(2 \times (7 - 4)\bigr).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:4}
Calcule el \omterm{def:g6:fractions:def}{fracciones} (¡paréntesis invisibles!):
\[
\frac{18 + 6}{8}, \qquad
\frac{30}{7 - 2}, \qquad
\frac{5 \times 8}{4 + 6} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:5}
Utilice la distributividad para calcular mentalmente:
\[
8 \times 102, \qquad
25 \times 99, \qquad
14 \times 12 + 14 \times 8, \qquad
37 \times 45 - 37 \times 35 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:6}
Escriba el cálculo en una línea, con paréntesis donde sea necesario, luego
calcular: ``el \omterm{ex:g1:subtraction:difference}{diferencia} de $50$ y el \omterm{def:g2:mult:def}{producto} de $6$ por $7$'';
``el \omterm{def:g2:mult:def}{producto} del \omterm{def:g1:addition:def}{suma} de $4$ y el $5$ por el \omterm{ex:g1:subtraction:difference}{diferencia} de $10$ y
$8$''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:priorities:7}
Amir compró cuadernos $3$ a $2.50$ cada uno y bolígrafos $2$ a $1.20$ cada uno.
Escribe su total en una sola expresión y luego calculalo.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:priorities:8}
Coloque corchetes en $4 + 2 \times 5 - 1$ para obtener el resultado: (a) $13$;
(b) $29$; (c)$24$. (Uno de los tres no necesita ningún paréntesis ---
¿cuál?)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:priorities:9}
¿$12 + 4 \times k$ y $16 \times k$ son iguales para cada número $k$?
Pruebe con $k = 1$ y $k = 2$ y concluya. ¿Qué error de prioridad?
llevaría de uno a otro?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:priorities:10}
Una entrada de cine cuesta $9$ euros, pero un grupo de $n$ personas paga un piso
$12$ euros más $6$ euros por persona.
\begin{enumerate}
  \item Escriba una expresión para el precio del grupo y calcule para
        $n = 5$.
  \item ¿A partir de cuántas personas la tarifa de grupo es más barata que $n$?
        entradas individuales? (Pruebe algunos valores de $n$).
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:priorities:11}
Usando cada uno de los dígitos $1$, $2$, $3$, $4$ exactamente una vez, los cuatro
operaciones y corchetes a voluntad, escriba expresiones iguales a $24$, para
$10$ y a $1$.
\end{exercise}

\section{Problema: multiplicar como un escriba egipcio}

\begin{problem}[{Problema de fin de semana --- trucos de cálculo mental de
distributividad, y la multiplicación duplicar-reducir a la mitad utilizada para
cuatro mil años}]
\label{pb:g7:priorities:1}
Mucho antes de que existieran las calculadoras y antes de que aparecieran las tablas de multiplicar.
perforado en la escuela --- escribas egipcios y campesinos rusos
multiplicó dos números cualesquiera usando solo \emph{duplicar, reducir a la mitad y
añadiendo}. Su método parece mágico; es de hecho
distributividad (\cref{thm:g7:priorities:distributivity}) trabajando
a máxima potencia. Este problema agudiza tu cálculo mental, entonces
te enseña el método antiguo, luego abre el capó para ver exactamente
por qué funciona.

\medskip
\noindent\textbf{Parte I --- The art of not computing.}
\begin{enumerate}
  \item Calcula mentalmente, expandiendo como en
        \cref{ex:g7:priorities:distributivity}:
        $7 \times 102$ y $9 \times 98$.
  \item Calcula mentalmente, factorizando:
        $17 \times 13 + 17 \times 7$ y
        $45 \times 101 - 45$.
  \item El truco ``por $5$'': para multiplicar por $5$, multiplica por
        $10$ y reducir a la mitad. Calcule $5 \times 87$ de esta manera y
        Explique por qué el truco es legítimo.
  \item Dos trucos más para justificar y utilizar: multiplicar por $25$ por
        multiplicando por $100$ y dividiendo por $4$ (calcular
        $25 \times 32$); multiplicar por $99$ multiplicando por $100$
        y restando una vez (calcular $99 \times 47$).
  \item Un disparo de advertencia: la distributividad extiende un factor sobre un
        \emph{\omterm{def:g1:addition:def}{suma}}, nunca sobre un \omterm{def:g2:mult:def}{producto}. Comparar
        $8 \times (5 + 3)$ con $8 \times 5 + 8 \times 3$, y
        entonces $8 \times (5 \times 3)$ con
        $(8 \times 5) \times (8 \times 3)$: qué par coincide,
        cual no?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The doubling--halving method.}
Aquí está la receta de $13 \times 24$. Escribe dos columnas. Izquierda
columna: comience en $13$ y \emph{reducir en la mitad}, eliminando cualquier \omterm{def:g3:division:remainder}{resto},
hasta $1$:$13,\ 6,\ 3,\ 1$. Columna derecha: comience en $24$ y
\emph{doble} tantas veces: $24,\ 48,\ 96,\ 192$. tachar
cada fila cuyo número \emph{izquierda} sea \omterm{def:g3:numbers:evenodd}{incluso} y agregue los supervivientes
números correctos.
\begin{enumerate}[resume]
  \item Realizar la receta de $13 \times 24$: ¿qué fila es
        tachado, lo que queda \omterm{def:g1:addition:def}{suma}, y es igual
        $13 \times 24$?
  \item Multiplica $21 \times 17$ por el método (reducir a la mitad $21$, duplicar
        $17$), y comprueba la respuesta mediante multiplicación ordinaria.
  \item Multiplica $16 \times 25$ por el método. cuantas filas
        sobrevivir, y qué tiene de especial una columna izquierda que
        comienza en $1$, $2$, $4$, $8$, $16, \dots$?
  \item Ahora \emph{por qué} funciona, paso a paso. Justifica cada uno
        igualdad con distributividad:
        \[
        13 \times 24 = (12 + 1) \times 24
        = 12 \times 24 + 24 = 6 \times 48 + 24 .
        \]
        Moraleja: dividir a la mitad la izquierda y duplicar la derecha mantiene la
        \omterm{def:g2:mult:def}{producto} --- pero cuando el número de la izquierda es \emph{\omterm{def:g3:numbers:evenodd}{extraño}}, uno
        copia del número correcto se separa y debe ser depositada.
        ¿Qué número corresponde a la primera fila de la tabla?
  \item Continúe la contabilidad en la tabla:
        $6 \times 48 = 3 \times 96$, entonces
        $3 \times 96 = 1 \times 192 + 96$, entonces
        $1 \times 192 = 192$. Recoger todo lo que estaba bancarizado
        en el camino, y explique por qué los números depositados son
        exactamente los números de la columna derecha de las filas con un \omterm{def:g3:numbers:evenodd}{extraño}
        izquierda --- las filas que mantiene la receta.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Under the hood: \omterm{def:g1:addition:def}{sumas} of
doublings.}
\begin{enumerate}[resume]
  \item En la pregunta 6, las filas supervivientes valían $1$, $4$
        y $8$ duplicaciones de $24$: el método escribió silenciosamente
        $13 \times 24 = (1 + 4 + 8) \times 24$. comprueba eso
        $1 + 4 + 8 = 13$, y encuentre qué duplicaciones ($1$, $2$,
        $4$, $8$, $16$) el método mantenido para $21$ en la pregunta 7.
  \item Escriba $25$ y $42$ como \omterm{def:g1:addition:def}{sumas} de los números de la duplicación.
        list $1, 2, 4, 8, 16, 32$, usando cada uno como máximo una vez.
  \item Explique por qué \emph{cada} se puede escribir un número entero.
        de esta manera: tome el número de duplicación más grande que encaje,
        restar, repetir --- ¿por qué esto debe detenerse? Escribe $100$ como
        tal \omterm{def:g1:addition:def}{suma}.
  \item Los escribas de Egipto hicieron exactamente esto, cuatro mil
        Hace años: para multiplicar $19 \times 35$, hicieron un
        tabla de duplicación de $35$ ($1 \to 35$, $2 \to 70$,
        $4 \to 140$, $8 \to 280$, $16 \to 560$), seleccionó las filas
        sumando $19$, y resumido. Haz el cálculo
        manera.
  \item Finale: computadoras modernas, que cuentan con $0$ y
        $1$s, multiplica esencialmente como los escribas --- por
        duplicaciones y adiciones. ¿Cuáles tres adiciones
        ¿Marca de computadora para $13 \times 24$? (¿Por qué cada número
        se divide en duplicaciones y qué tienen que hacer $0$ y $1$
        Con ello, está el problema del fin de semana del capítulo de Powers:
        \cref{pb:g8:powers:1}.)
\end{enumerate}
\end{problem}
