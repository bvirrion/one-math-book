\chapter{Fracciones: comparar y sumar}\label{ch:g7:fractions}

El grado 6 introdujo \omterm{def:g6:fractions:def}{fracciones} como acciones y como \omterm{def:g3:division:remainder}{cocientes} exacto
(\cref{ch:g6:fractions}). Este capítulo aprende a \emph{calcular} con
ellos: reconocer igual \omterm{def:g6:fractions:def}{fracciones}, comparar, sumar y restar
--- por ahora con el amigable \omterm{def:g4:fractions:def}{denominadores}; el caso general entra
\cref{ch:g9:fractions}.

\section{fracciones iguales}

\begin{theorem}[fracciones iguales]\label{thm:g7:fractions:equal}
Un \omterm{def:g6:fractions:def}{fracción} no cambia cuando sus \omterm{def:g4:fractions:def}{numerador} y \omterm{def:g4:fractions:def}{denominador} son
multiplicado o dividido por el mismo número distinto de cero:
\[
\frac{a}{b} = \frac{a \times k}{b \times k} .
\]
\end{theorem}
\admitted

\begin{example}\label{ex:g7:fractions:equal}
$\dfrac{3}{4} = \dfrac{3 \times 5}{4 \times 5} = \dfrac{15}{20}$, y
en el otro sentido $\dfrac{42}{30} = \dfrac{42 \div 6}{30 \div 6}
= \dfrac{7}{5}$ (simplificado por $6$).

Para probar si $\dfrac{8}{12}$ y $\dfrac{10}{15}$ son iguales,
simplificar ambos: $\dfrac{8}{12} = \dfrac{2}{3}$ y
$\dfrac{10}{15} = \dfrac{2}{3}$ --- si, igual.
\end{example}

\section{Comparar fracciones}

\begin{method}[Comparando dos fracciones]\label{met:g7:fractions:compare}
\begin{enumerate}
  \item \emph{Same \omterm{def:g4:fractions:def}{denominador}}: comparar el \omterm{def:g4:fractions:def}{numeradores}:
        $\frac{5}{9} < \frac{7}{9}$.
  \item \emph{One \omterm{def:g4:fractions:def}{denominador} is a \omterm{def:g5:division:multiple}{múltiple} of the other}: reescribe el
        \omterm{def:g6:fractions:def}{fracción} más grueso, luego compare:
        $\frac{5}{6}$ frente a $\frac{7}{12}$: $\frac56 = \frac{10}{12} >
        \frac{7}{12}$.
  \item \emph{Compare with $1$ or $\frac12$} cuando sea posible:
        $\frac{9}{8} > 1 > \frac{7}{9}$ se instala
        $\frac98$ frente a $\frac79$ al instante.
\end{enumerate}
\end{method}

\section{Sumar y restar}

\begin{theorem}[Mismo denominador]\label{thm:g7:fractions:add}
\omterm{def:g6:fractions:def}{fracciones} con el mismo \omterm{def:g4:fractions:def}{denominador} se suman (o restan) por
sumando (o restando) el \omterm{def:g4:fractions:def}{numeradores}:
\[
\frac{a}{d} + \frac{b}{d} = \frac{a + b}{d},
\qquad
\frac{a}{d} - \frac{b}{d} = \frac{a - b}{d} .
\]
\end{theorem}

\begin{proof}[¿Por qué?]
$a$ piezas del tamaño $\frac1d$ más $b$ piezas del mismo tamaño marca
$a + b$ piezas de ese tamaño.
\end{proof}

\begin{method}[Diferentes denominadores]\label{met:g7:fractions:add}
Cuando un \omterm{def:g4:fractions:def}{denominador} es un \omterm{def:g5:division:multiple}{múltiple} del otro:
\begin{enumerate}
  \item reescribe el \omterm{def:g6:fractions:def}{fracción} con el \omterm{def:g4:fractions:def}{denominador} más pequeño para que ambos tengan
        el más grande (\cref{thm:g7:fractions:equal});
  \item suma o resta el \omterm{def:g4:fractions:def}{numeradores};
  \item simplifique el resultado si es posible.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:fractions:add}
Calcula $\dfrac{3}{4} + \dfrac{5}{8}$, paso a paso:
\begin{enumerate}
  \item $8$ es un \omterm{def:g5:division:multiple}{múltiple} de $4$: reescribir
        $\dfrac34 = \dfrac{3 \times 2}{4 \times 2} = \dfrac{6}{8}$;
  \item agregar: $\dfrac{6}{8} + \dfrac{5}{8} = \dfrac{11}{8}$;
  \item nada que simplificar: el resultado es $\dfrac{11}{8}$, es decir\
        $1 + \dfrac38$.
\end{enumerate}
Otro: $2 - \dfrac{4}{5} = \dfrac{10}{5} - \dfrac{4}{5} =
\dfrac{6}{5}$ (escribe el número entero sobre \omterm{def:g4:fractions:def}{denominador} $5$
primero).
\end{example}

\begin{example}[Una trampa clásica]\label{ex:g7:fractions:trap}
$\dfrac12 + \dfrac13$ es \emph{no} $\dfrac{2}{5}$! Agregando \omterm{def:g4:fractions:def}{numeradores}
y \omterm{def:g4:fractions:def}{denominadores} por separado es incorrecto --- la respuesta $\frac25$ sería
menor que $\frac12$, absurdo cuando \emph{añadiendo} algo positivo
a $\frac12$. (El \omterm{def:g1:addition:def}{suma}, $\frac56$ correcto, necesita un \omterm{def:g4:fractions:def}{denominador} común
$6$; el método general está en \cref{ch:g9:fractions}.)
\end{example}

\section{Multiplicar una fracción por un número}

\begin{proposition}[Fracción multiplicada por un número]\label{prop:g7:fractions:mult}
Para un número $k$:
\[
k \times \frac{a}{b} = \frac{k \times a}{b} .
\]
Tomar $\frac ab$ \emph{of} una cantidad significa multiplicar por
$\frac ab$.
\end{proposition}
\admitted

\begin{example}\label{ex:g7:fractions:mult}
$6 \times \dfrac{2}{3} = \dfrac{12}{3} = 4$: seis por dos tercios es
cuatro enteros. Y ``$\frac34$ de $60$ euros'' es
$\frac34 \times 60 = \frac{180}{4} = 45$ euros --- la misma respuesta que
dividir por $4$ y luego multiplicar por $3$
(\cref{met:g6:fractions:ofquantity}).
\end{example}

\section{Problema: Las fracciones escondidas en un compás de música.}

\begin{problem}[{Problema de fin de semana --- los valores de las notas son fracciones
que deben sumar exactamente, el conteo rítmico esconde una secuencia famosa,
y la paridad gobierna el ritmo de los Balcanes}]
\label{pb:g7:fractions:1}
La música está escrita en \omterm{def:g6:fractions:def}{fracciones}. A \emph{nota completa} dura $1$
medida (en el tiempo común "cuatro-cuatro"); a \emph{\omterm{def:g3:division:half}{medio} note}
dura $\frac12$; luego viene el \emph{\omterm{def:g3:division:half}{cuarto} note} $\frac14$, el
\emph{octava nota} $\frac18$ y el \emph{semicorchea}
$\frac{1}{16}$. Una ley inviolable: \emph{las duraciones dentro de un
La medida debe sumar exactamente el total de la medida.} --- $1$ en
cuatro por cuatro veces. Cada pregunta a continuación es la aritmética de este capítulo.
(\cref{thm:g7:fractions:add}, \cref{met:g7:fractions:add}) configurado en
música; en el camino te encontrarás con una secuencia de conteo descubierta por
Eruditos indios hace mil años.

\medskip
\noindent\textbf{Parte I --- Notes that must add up.}
\begin{enumerate}
  \item Comprueba que cada una de estas medidas es legal en cuatro-cuatro
        hora: (a) \omterm{def:g3:division:half}{medio} $+$ \omterm{def:g3:division:half}{cuarto} $+$ \omterm{def:g3:division:half}{cuarto}; (b)\omterm{def:g3:division:half}{cuarto}$+$
        octavo $+$ octavo $+$ \omterm{def:g3:division:half}{medio}.
  \item Un copista escribió: \omterm{def:g3:division:half}{medio} $+$ \omterm{def:g3:division:half}{cuarto} $+$ octavo. que
        \omterm{def:g6:fractions:def}{fracción} de la medida está escrito, y qué único
        nota lo completa?
  \item ¿Cuántas semicorcheas ocupan un compás completo? como
        ¿Cuántas semicorcheas forman una corchea?
  \item A \emph{punto} después de una nota agrega \omterm{def:g3:division:half}{medio} el valor de la nota: a
        \omterm{def:g3:division:half}{medio} con puntos es $\frac12 + \frac14$, \omterm{def:g3:division:half}{cuarto} con puntos es
        $\frac14 + \frac18$. Calcule ambos valores.
  \item Un vals se escribe en compás de tres por cuatro: cada compás debe
        total $\frac34$. Compruebe que un único punto \omterm{def:g3:division:half}{medio} llene un
        medida de vals y anota otras dos medidas de vals legales.
        medidas utilizando sólo \omterm{def:g3:division:half}{cuarteles} y octavas.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The copyist's workshop.}
\begin{enumerate}[resume]
  \item Un compás cuatro por cuatro contiene: punteado \omterm{def:g3:division:half}{cuarto} $+$ octavo
        $+$\omterm{def:g3:division:half}{cuarto}. ¿Qué está faltando?
  \item ¿Qué dura más, una octava con puntillo o una nota \omterm{def:g3:division:half}{cuarto}?
        Comparar mediante dieciseisavos.
  \item A \emph{atar} une duraciones en una larga
        sonido: una nota \omterm{def:g3:division:half}{medio} atada a una octava dura ¿cuánto tiempo? como
        ¿Cuántos dieciseisavos es eso?
  \item Las notas de una medida suman $\frac{11}{16}$; el
        \omterm{def:g3:division:remainder}{resto} es silencio (un \emph{descansar}). ¿Cuánto dura el
        descansar?
  \item Un baterista llena una nota \omterm{def:g3:division:half}{cuarto} ($\frac{4}{16}$ de la
        compás) usando sólo octavos ($\frac{2}{16}$) y
        dieciseisavos ($\frac{1}{16}$), orden importante: para
        instancia octava--decimosexta--decimosexta, o
        decimosexto – octavo – decimosexto. Compruebe que ambos ejemplos
        son legales, luego enumere \emph{todo} los ritmos posibles.
        ¿Cuantos hay?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- A thousand-year-old sequence, and
an \omterm{def:g3:numbers:evenodd}{extraño} measure.}
\begin{enumerate}[resume]
  \item Contemos los ritmos del baterista para notas más largas.
        Cada ritmo termina con una semicorchea o con una
        octavo; explica por qué esto da la regla: (ritmos
        relleno $n$ semicorcheas) $=$ (relleno de ritmos $n - 1$)
        $+$ (ritmos llenando $n - 2$). A partir del ritmo $1$
        para un decimosexto y $2$ para dos, construya la tabla hasta
        $n = 8$: ¿cuántos ritmos llenan una nota \omterm{def:g3:division:half}{medio}?
  \item Los números que encontraste --- $1, 2, 3, 5, 8, 13, 21, 34$
        --- fueron publicados por el erudito indio Hemachandra
        alrededor del año 1150, contando exactamente esos ritmos, dos
        generaciones antes de que Fibonacci los escribiera en Italia.
        Indique su patrón definitorio en una oración y verifique
        en las últimas tres entradas de su tabla.
  \item El estilo ``Swing'' reemplaza dos octavos iguales por un
        par largo-corto: $\frac16$ y luego $\frac{1}{12}$. verificar
        que el swap es legal (misma duración total), utilizando el
        común \omterm{def:g4:fractions:def}{denominador} $12$.
  \item A \emph{trillizo} comprime tres notas iguales en una
        \omterm{def:g3:division:half}{cuarto} nota. ¿Cuánto dura cada nota? (comprueba tu respuesta).
        multiplicándolo por $3$,
        \cref{prop:g7:fractions:mult})? Misma pregunta para tres.
        notas iguales llenando una nota \omterm{def:g3:division:half}{medio}.
  \item La música de baile balcánica utiliza el compás siete-ocho: cada compás
        totaliza $\frac78$. Verifique que el punto \omterm{def:g3:division:half}{cuarto} $+$ \omterm{def:g3:division:half}{cuarto}
        $+$ \omterm{def:g3:division:half}{cuarto} lo llena. Luego explica por qué \emph{no}
        combinación de billetes \omterm{def:g3:division:half}{medio} y billetes \omterm{def:g3:division:half}{cuarto} solos,
        por muchos que sean, alguna vez podrán llenar un compás siete-ocho.
        (Cuente en octavos y piense en pares y \omterm{def:g3:numbers:evenodd}{extraño}).
\end{enumerate}
\end{problem}
