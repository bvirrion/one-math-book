\section{Reconocer y completar tablas.}

\begin{definition}[Tabla de proporcionalidad]\label{def:g7:prop:table}
Una tabla de dos filas es una \emph{proporcionalidad
mesa}\index{proporcionalidad} cuando los números de la segunda fila son
los del primero multiplicados por un número fijo, el
\emph{coeficiente}. Equivalentemente: toda la columna \omterm{def:g3:division:remainder}{cocientes}
$\frac{\text{bottom}}{\text{top}}$ son iguales.
\end{definition}

\begin{example}\label{ex:g7:prop:recognize}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|ccc}
top & $4$ & $10$ & $14$ \\
\hline
bottom & $6$ & $15$ & $21$ \\
\end{tabular}
\end{center}
\omterm{def:g3:division:remainder}{Cocientes}: $\frac64 = 1.5$, $\frac{15}{10} = 1.5$,
$\frac{21}{14} = 1.5$: \omterm{def:g6:propdata:def}{proporcional}, coeficiente $1.5$.
\end{example}

\begin{theorem}[regla cruzada]\label{thm:g7:prop:cross}
En un \omterm{def:g7:prop:table}{tabla de proporcionalidad} con las columnas $\begin{pmatrix} a \\ b
\end{pmatrix}$ y $\begin{pmatrix} c \\ d \end{pmatrix}$, la cruz
\omterm{def:g2:mult:def}{productos} son iguales:
\[
a \times d = b \times c .
\]
En consecuencia, el cuarto valor se puede calcular a partir de los otros tres:
$d = \dfrac{b \times c}{a}$.
\end{theorem}

\begin{proof}
Sea $k$ el coeficiente: $b = ka$ y $d = kc$. entonces
\[
a \times d = a \times kc = k \times ac,
\qquad
b \times c = ka \times c = k \times ac :
\]
ambos cruzan \omterm{def:g2:mult:def}{productos} igual a $k \times ac$. Dividiendo $ad = bc$ por $a$
da la fórmula para $d$.
\end{proof}

\begin{example}[Cuarta proporcional]\label{ex:g7:prop:fourth}
Si $7$ libros idénticos pesan $2.8$ kg, ¿cuánto pesan $12$ dichos libros?
pesar? Regla de tabla y cruz:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l|cc}
books & $7$ & $12$ \\
\hline
kg & $2.8$ & $m$ \\
\end{tabular}
\end{center}
\[
7 \times m = 2.8 \times 12
\quad\Longrightarrow\quad
m = \frac{2.8 \times 12}{7} = \frac{33.6}{7} = 4.8 \text{ kg}.
\]
Verificar por unidad: un libro pesa $0.4$ kg, doce pesan $4.8$ kg.
\end{example}

\section{Porcentajes}

\begin{method}[Aplicar y encontrar porcentajes]\label{met:g7:prop:percent}
\begin{enumerate}
  \item \emph{Aplicar} $t\,\%$: multiplicar por $\frac{t}{100}$
        ($12\,\%$ de $350$ es $0.12 \times 350 = 42$);
  \item \emph{encontrar} el porcentaje que representa una parte: calcular
        $\frac{\text{part}}{\text{whole}}$ y reescribir sobre $100$
        ($21$ de $60$: $\frac{21}{60} = \frac{35}{100} = 35\,\%$);
  \item siempre pregunta: \emph{¿un porcentaje de qué?} La referencia completa
        importa.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:prop:percent}
En una escuela, $180$ de los estudiantes $400$ son niñas. Porcentaje:
\[
\frac{180}{400} = \frac{180 \div 4}{400 \div 4} = \frac{45}{100}
= 45\,\% .
\]
Al revés: ¿cuántos son los chicos $55\,\%$?
$0.55 \times 400 = 220$. Verificar: $180 + 220 = 400$.
\end{example}

\section{Balanza}

\begin{definition}[Escala]\label{def:g7:prop:scale}
En un mapa o modelo en \emph{escala}\index{escala}
$\frac{1}{n}$ (escrito $1 : n$), las distancias en el mapa son
\omterm{def:g6:propdata:def}{proporcional} a las distancias reales, con coeficiente $\frac1n$:
\[
\text{map distance} = \frac{1}{n} \times \text{real distance},
\]
ambos medidos en el \emph{misma unidad}.
\end{definition}

\begin{example}\label{ex:g7:prop:scale}
En un mapa $1 : 50\,000$, dos ciudades están separadas por $7$ cm. Distancia real:
\[
7 \times 50\,000 = 350\,000 \text{ cm} = 3\,500 \text{ m} = 3.5
\text{ km}.
\]
Por el contrario, una ruta de senderismo de $12$ km mide en el mapa
$12$ km $= 1\,200\,000$ cm, entonces $1\,200\,000 \div 50\,000 = 24$ cm.
\end{example}

\begin{remark}
Los gráficos son la prueba más rápida de \omterm{def:g7:prop:table}{proporcionalidad}
(\cref{ch:g6:propdata}): los puntos de una recta \emph{a través del
origen} significan cantidades \omterm{def:g6:propdata:def}{proporcional}; una línea a la que le falta el origen (como
una tarifa de taxi con cargo fijo) significa \emph{no} \omterm{def:g6:propdata:def}{proporcional}, incluso
aunque es una línea recta. Funciones afines, en
\cref{ch:g9:linfunc}, aclarará esto.
\end{remark}

\section{Problema: medir la Tierra con un palo}

\begin{problem}[{Problema del fin de semana --- las sombras son un
proporcionalidad y cómo Eratóstenes calculó la circunferencia
de la Tierra en 240 aC}]
\label{pb:g7:prop:1}
Hace veintidós siglos, sin telescopio, sin satélite y sin
calculadora, el bibliotecario de Alejandría midió la Tierra ---
usando un palo, un pozo, una caravana de camellos y los de este capítulo.
matemáticas. Este problema vuelve sobre sus pasos: primero el
\omterm{def:g7:prop:table}{proporcionalidad} de sombras, luego el famoso cómputo mismo,
y finalmente cómo se ve su respuesta en el humano \omterm{def:g7:prop:scale}{escala}.

\medskip
\noindent\textbf{Parte I --- The shadow of a stick.}
El sol está tan lejos que sus rayos nos llegan \omterm{def:g6:lines:perp}{paralelo} a uno
otro. En un momento dado, todos los objetos verticales y sus
las sombras son por lo tanto \omterm{def:g6:propdata:def}{proporcional}.
\begin{enumerate}
  \item Al mediodía, un palo vertical de $1$ m proyecta una sombra de
        $0.4$ m, y la sombra de un árbol mide $3.2$ m. que altura
        Qué es el árbol (\cref{thm:g7:prop:cross})?
  \item En el mismo momento, una torre proyecta una sombra de $26$ m: ¿cómo
        ¿Qué altura tiene la torre? ¿Y cuánto mide la sombra de un niño?
        $1.5$ m de altura?
  \item Explique en una frase por qué todos estos cálculos son
        Válido únicamente el \emph{mismo momento} del día.
  \item La leyenda dice que Tales midió el Grande \omterm{def:g5:solids:def}{Pirámide} esperando
        por el momento \emph{su propia sombra era exactamente igual
        mientras fuera alto}. ¿Qué es el \omterm{def:g7:prop:table}{proporcionalidad}?
        coeficiente en ese momento, y que altura tiene el \omterm{def:g5:solids:def}{pirámide}
        si la punta de su sombra se encuentra a $147$ m desde el punto de
        el suelo directamente debajo de su vértice?
  \item Resuma la Parte I: en un momento dado, ¿qué cantidad es
        lo mismo para el palo, el árbol, la torre y el
        \omterm{def:g5:solids:def}{pirámide} (\cref{def:g7:prop:table})?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Eratosthenes' computation.}
Eratóstenes conocía dos hechos. En la ciudad de Syene, al mediodía del
solsticio de verano, se puso el sol \emph{exactamente por encima}: luz del sol
llegó al fondo de los pozos más profundos, y se arrojaron palos verticales
sin sombra. En Alejandría, $800$ km al norte, en el mismo momento,
un palo vertical \emph{hizo} proyecta una sombra y los rayos del sol
formó un ángulo de $7.2^\circ$ con la vertical.
\begin{enumerate}[resume]
  \item Explique por qué estas dos observaciones juntas prueban que
        La tierra de Syene y la tierra de Alejandría no son
        \omterm{def:g6:lines:perp}{paralelo} --- es decir, la superficie de la Tierra es curva.
        (¿Qué le harían los rayos \omterm{def:g6:lines:perp}{paralelo} a dos palos en una superficie plana?
        ¿Tierra?)
  \item En un dibujo de la Tierra \omterm{def:g4:numbers:round}{redondo} con rayos de sol \omterm{def:g6:lines:perp}{paralelo},
        el $7.2^\circ$ en Alejandría reaparece en el centro de
        la Tierra, como el ángulo entre las direcciones de la
        dos ciudades. Haz el dibujo y convéncete tú mismo (el
        La justificación limpia utiliza los pares de ángulos iguales.
        \cref{ch:g7:triangles}).
  \item ¿Qué \omterm{def:g6:fractions:def}{fracción} de un turno completo es $7.2^\circ$?
  \item El arco de Syene a Alejandría ($800$ km) corresponde
        a $7.2^\circ$; toda la circunferencia corresponde a
        $360^\circ$. Configure el \omterm{def:g7:prop:table}{proporcionalidad} y calcule el
        circunferencia de la Tierra.
  \item Deduce el \omterm{def:g4:geometry:circle}{diámetro} de la Tierra, usando $\pi \approx
        3.14$ (\cref{prop:g6:measure:circle}) y \omterm{def:g6:decimals:rounding}{redondeo} para
        la centena de kilómetros más cercana. (El valor moderno es
        $12\,742$ km --- que tan cerca estaba un hombre con un palo adentro
        240 aC?)
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The Earth at human \omterm{def:g7:prop:scale}{escala}.}
\begin{enumerate}[resume]
  \item Se construye un globo terráqueo en \omterm{def:g7:prop:scale}{escala} $1 : 40\,000\,000$. Usando el
        circunferencia encontrada en la pregunta 9, muestran que la
        La circunferencia del globo es exactamente $1$ m. cual es su
        \omterm{def:g4:geometry:circle}{diámetro}, al centímetro más cercano?
  \item El Mont Blanc se eleva a unos $4.8$ km. Convertir $4.8$ km a
        centímetros, dividir por $40\,000\,000$ y expresar el
        Altura de la montaña en el mundo en milímetros. ¿Qué hace?
        ¿Qué dice la respuesta sobre cuán "baches" es realmente la Tierra?
  \item Un andador recorre $40$ km por día. A ese ritmo, ¿cuántas
        días para la circunferencia completa de Eratóstenes --- y
        aproximadamente cuantos años son eso?
  \item La atmósfera respirable se concentra aproximadamente en la
        primeros $10$ km sobre el suelo. ¿Qué porcentaje de la

        (\cref{met:g7:prop:percent})? Redondear al más cercano
        décima parte de un por ciento.
  \item Los historiadores estiman que el valor anunciado por Eratóstenes,

        $42\,000$ kilómetros. Tomando $40\,000$ km como valor real,

\end{problem}
