\chapter{Central Symmetry and Parallelograms}\label{ch:g7:central}

Turn a figure by \omterm{def:g3:division:half}{half} a turn around a point: this is \emph{\omterm{def:g7:central:def}{central
symmetry}}, the second transformation of the book after the reflections
of \cref{ch:g6:symmetry}. Its star figure is the \omterm{def:g7:central:parallelogram}{parallelogram} --- the
\omterm{def:g4:geometry:polygon}{quadrilateral} that is \omterm{def:g7:central:def}{symmetric} about the crossing point of its own
diagonals.

\section{Central symmetry}

\begin{definition}[Symmetric about a point]\label{def:g7:central:def}
The \emph{symmetric}\index{central symmetry} of a point $M$ about a
point $O$ is the point $M'$ such that $O$ is the \emph{midpoint} of
$[MM']$. The symmetric figure is what the original becomes after a
half-turn ($180^\circ$) around $O$.
\end{definition}

\begin{method}[Constructing the symmetric of a point]\label{met:g7:central:construct}
To construct the \omterm{def:g7:central:def}{symmetric} $M'$ of $M$ about $O$:
\begin{enumerate}
  \item draw the line $(MO)$ and extend it beyond $O$;
  \item measure $MO$ (ruler or compass);
  \item place $M'$ on the extension with $OM' = OM$.
\end{enumerate}
On grid paper: count the horizontal and vertical steps from $M$ to
$O$, and repeat the \emph{same} steps from $O$ to reach $M'$.
\end{method}

\begin{example}[On a grid]\label{ex:g7:central:grid}
If $M$ is $3$ squares left and $1$ square up from $O$, its \omterm{def:g7:central:def}{symmetric}
$M'$ is $3$ squares \emph{right} and $1$ square \emph{down} from $O$:
\omterm{def:g7:central:def}{central symmetry} reverses both directions at once (a reflection
reverses only one).
\end{example}

\begin{proposition}[What a half-turn preserves]\label{prop:g7:central:preserved}
\omterm{def:g7:central:def}{Central symmetry} preserves lengths, angles, \omterm{def:g6:measure:perimeter}{perimeters} and \omterm{def:g6:measure:area}{areas}; the
image of a line is a \emph{\omterm{def:g6:lines:perp}{parallel}} line, the image of a \omterm{def:g6:lines:objects}{segment} is a
\omterm{def:g6:lines:perp}{parallel} \omterm{def:g6:lines:objects}{segment} of the same length, the image of a circle is a circle
of the same \omterm{def:g3:shapes:shapes}{radius} (centered at the \omterm{def:g7:central:def}{symmetric} of the center).
\end{proposition}
\admitted

\begin{remark}
Compare with reflections (\cref{prop:g6:symmetry:preserved}): both
preserve lengths and angles; but the reflection flips figures over
(a ``b'' becomes a ``d''), while the half-turn keeps them readable ---
upside down (a ``b'' becomes a ``q''). And unlike a reflection, a
half-turn sends every line to a line \omterm{def:g6:lines:perp}{parallel} to it.
\end{remark}

\begin{definition}[Center of symmetry]\label{def:g7:central:center}
A point $O$ is a \emph{center of symmetry}\index{center of symmetry}
of a figure when the half-turn around $O$ sends the figure exactly
onto itself. For instance, the letters S, N, Z have a center of
symmetry; a circle has one (its center); a triangle never has one.
\end{definition}

\section{Parallelograms}

\begin{definition}[Parallelogram]\label{def:g7:central:parallelogram}
A \emph{parallelogram}\index{parallelogram} is a \omterm{def:g4:geometry:polygon}{quadrilateral} whose
diagonals cross at their common midpoint. That crossing point is then
a \omterm{def:g7:central:center}{center of symmetry} of the figure: each \omterm{def:g5:solids:def}{vertex} is the half-turn image
of the opposite \omterm{def:g5:solids:def}{vertex}.
\end{definition}

\begin{theorem}[Properties of a parallelogram]\label{thm:g7:central:properties}
In a \omterm{def:g7:central:parallelogram}{parallelogram} $ABCD$ with center $O$:
\begin{enumerate}
  \item opposite sides are \omterm{def:g6:lines:perp}{parallel}: $(AB) \parallel (CD)$ and
        $(BC) \parallel (AD)$;
  \item opposite sides have the same length: $AB = CD$ and
        $BC = AD$;
  \item opposite angles are equal: $\widehat A = \widehat C$ and
        $\widehat B = \widehat D$.
\end{enumerate}
\end{theorem}

\begin{proof}
The half-turn around $O$ sends $A$ to $C$ and $B$ to $D$ (definition:
$O$ is the midpoint of both diagonals). So it sends the \omterm{def:g6:lines:objects}{segment}
$[AB]$ to the \omterm{def:g6:lines:objects}{segment} $[CD]$; by
\cref{prop:g7:central:preserved} the image is \omterm{def:g6:lines:perp}{parallel} to $[AB]$ and
of the same length: points 1 and 2. It also sends the angle at $A$
to the angle at $C$, and angles are preserved: point 3.
\end{proof}

\begin{theorem}[Recognizing a parallelogram]\label{thm:g7:central:recognize}
A \omterm{def:g4:geometry:polygon}{quadrilateral} (with no crossing sides) is a \omterm{def:g7:central:parallelogram}{parallelogram} as soon as
\emph{one} of the following holds:
\begin{enumerate}
  \item its diagonals have the same midpoint (the definition);
  \item its opposite sides are \omterm{def:g6:lines:perp}{parallel} two by two;
  \item two opposite sides are \omterm{def:g6:lines:perp}{parallel} \emph{and} of the same length.
\end{enumerate}
\end{theorem}
\admitted

\begin{example}\label{ex:g7:central:recognize}
$ABCD$ has $(AB) \parallel (CD)$ and $AB = CD = 5$ cm. Criterion 3
applies: $ABCD$ is a \omterm{def:g7:central:parallelogram}{parallelogram} --- so without any further
measuring we also know $AD = BC$, $(AD) \parallel (BC)$, and that its
diagonals cut each other in their middle.
\end{example}

\begin{remark}[Special parallelograms]
The rectangle, \omterm{def:g6:shapes:quadrilaterals}{rhombus} and square of \cref{ch:g6:shapes} are exactly
the \omterm{def:g7:central:parallelogram}{parallelograms} with an extra property: equal diagonals
(rectangle), \omterm{def:g6:lines:perp}{perpendicular} diagonals (\omterm{def:g6:shapes:quadrilaterals}{rhombus}), both (square). The
angle argument promised in \cref{ex:g6:shapes:recognize} now works:
in a \omterm{def:g7:central:parallelogram}{parallelogram}, consecutive angles are supplementary
(alternate/corresponding angles with the \omterm{def:g6:lines:perp}{parallels},
\cref{thm:g7:triangles:equalities}), so one \omterm{def:g3:shapes:rightangle}{right angle} forces all
four.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g7:central:1}
On grid paper, place $O$, then $M$ four squares right of $O$ and one
square up. Construct the \omterm{def:g7:central:def}{symmetric} $M'$ of $M$ about $O$, and the
\omterm{def:g7:central:def}{symmetric} of $M$ across the \emph{vertical grid line} through $O$.
Compare the two images.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:central:2}
Draw a triangle $ABC$ and a point $O$ outside it. Construct its image
$A'B'C'$ by the half-turn around $O$. What can you say about the
lengths $A'B'$ and $AB$? About the lines $(A'B')$ and $(AB)$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:central:3}
Which digits, written as on a calculator display ($0$ to $9$), have a
\omterm{def:g7:central:center}{center of symmetry}? Which capital letters among H, A, N, S, T, Z, O?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:central:4}
$KLMN$ is a \omterm{def:g7:central:parallelogram}{parallelogram} of center $O$ with $KL = 7$ cm,
$LM = 4$ cm and $\widehat K = 115^\circ$. Give, with reasons: $MN$,
$NK$, $\widehat M$, and the midpoint of $[LN]$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:central:5}
Construct a \omterm{def:g7:central:parallelogram}{parallelogram} $ABCD$ from its diagonals: $AC = 8$ cm and
$BD = 5$ cm, crossing at their midpoints with an angle of $60^\circ$
between them. (Draw the diagonals first!)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:central:6}
In a \omterm{def:g7:central:parallelogram}{parallelogram}, one angle measures $115^\circ$. Using the remark
on supplementary consecutive angles, give the three other angles.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:central:7}
Plot $A(1, 1)$, $B(4, 2)$ and the point $O(2.5, 2.5)$. Compute the
coordinates of the images $A'$ and $B'$ by the half-turn around $O$
(use the grid counting of \cref{ex:g7:central:grid}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:central:8}
$ABCD$ is a \omterm{def:g4:geometry:polygon}{quadrilateral} in which $AB = CD$. Is that enough to make
it a \omterm{def:g7:central:parallelogram}{parallelogram}? If not, sketch a counterexample (an \omterm{def:g6:shapes:triangles}{isosceles}
trapezoid helps), and state what must be added to the hypothesis.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:central:9}
Let $ABC$ be a triangle and $O$ the midpoint of $[BC]$. Construct the
\omterm{def:g7:central:def}{symmetric} $A'$ of $A$ about $O$. Show that $ABA'C$ is a
\omterm{def:g7:central:parallelogram}{parallelogram}. (Which criterion of \cref{thm:g7:central:recognize} is
free of charge here?)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:central:10}
True or false, with a reason: ``a \omterm{def:g4:geometry:polygon}{quadrilateral} whose opposite angles
are equal two by two is a \omterm{def:g7:central:parallelogram}{parallelogram}''; ``a \omterm{def:g7:central:parallelogram}{parallelogram} with
\omterm{def:g6:lines:perp}{perpendicular} diagonals is a \omterm{def:g6:shapes:quadrilaterals}{rhombus}''; ``a \omterm{def:g7:central:parallelogram}{parallelogram} with one
\omterm{def:g3:shapes:rightangle}{right angle} is a rectangle''.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:central:11}
Let $ABCD$ be any \omterm{def:g4:geometry:polygon}{quadrilateral} and $I$, $J$, $K$, $L$ the midpoints
of its sides $[AB]$, $[BC]$, $[CD]$, $[DA]$. Draw several examples.
What does $IJKL$ always seem to be? (The proof, with the midpoint
theorem, comes in \cref{ch:g8:midpoints} --- and again with vectors,
in the High School volume.)
\end{exercise}

\section{Problem: The cake, the median, and the impossible second
center}

\begin{problem}[{Weekend problem --- half-turns at work: fair cake
cuts, a bound on medians, and why no shape can have two centers of
symmetry}]
\label{pb:g7:central:1}
The half-turn looks like the simplest of transformations --- yet
it cuts cakes into provably fair halves, measures the medians of
a triangle, and hides a small pearl of pure mathematics: a
bounded figure can have one \omterm{def:g7:central:center}{center of symmetry}, but \emph{never
two}. The proof uses a discovery you will make in Part I: two
half-turns, performed one after the other, add up to a slide.

\medskip
\noindent\textbf{Part I --- Half-turn gymnastics.}
\begin{enumerate}
  \item On grid paper, take the origin $O(0, 0)$ as center.
        Using the grid counting of \cref{ex:g7:central:grid},
        give the images of $M(3, 1)$, $N(-2, 4)$ and
        $P(0, -5)$ by the half-turn about $O$. What is the
        general rule for $(x, y)$?
  \item Now the center is $C(2, 1)$. Compute the image of
        $M(5, 3)$, and check the general recipe: each coordinate
        of the image is \emph{twice the center's minus the
        point's}.
  \item Playing cards are designed to read the same upside down
        --- a \omterm{def:g7:central:def}{central symmetry}, so neither player sees the card
        ``wrongly''. Explain why, on a card with an \emph{\omterm{def:g3:numbers:evenodd}{odd}}
        number of pips, one pip must sit exactly at the card's
        center.
  \item Two half-turns in a row: take the centers $O_1(2, 0)$
        and $O_2(5, 0)$. Send the point $A(1, 1)$ through the
        half-turn about $O_1$, then send the image through the
        half-turn about $O_2$. Compare start and finish. Repeat
        with $B(0, 3)$. What single, simple transformation did
        the two half-turns amount to?
  \item Measure the slide of question 4 against the distance
        $O_1 O_2$, and state the discovery. (Compare with the
        two \omterm{def:g6:lines:perp}{parallel} mirrors of \cref{pb:g6:symmetry:1}: same
        phenomenon, new actors.)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Cakes and medians.}
\begin{enumerate}[resume]
  \item Let $O$ be the center of a \omterm{def:g7:central:parallelogram}{parallelogram}. Explain why
        any line through $O$ meets the boundary at two points
        that are \omterm{def:g7:central:def}{symmetric} about $O$, and why the line cuts the
        \omterm{def:g7:central:parallelogram}{parallelogram} into two pieces that are exact half-turn
        copies of each other --- hence of equal \omterm{def:g6:measure:area}{areas}.
  \item The baker's theorem: a rectangular cake is shared
        perfectly fairly by \emph{any} straight cut through its
        center. Better: a rectangular cake has a rectangular
        hole (any size, any position, \omterm{def:g3:numbers:evenodd}{even} tilted). Describe the
        one straight cut that gives two parts with equally much
        cake --- and justify it with question 6.
  \item Three \omterm{def:g5:solids:def}{vertices} of a \omterm{def:g7:central:parallelogram}{parallelogram} $ABCD$ are $A(0, 0)$,
        $B(6, 1)$ and $C(8, 5)$. Compute the coordinates of $D$
        and of the center $O$.
  \item \Cref{exo:g7:central:9} built, from a triangle $ABC$
        and the midpoint $O$ of $[BC]$, the \omterm{def:g7:central:parallelogram}{parallelogram}
        $ABA'C$ where $A'$ is the \omterm{def:g7:central:def}{symmetric} of $A$ about $O$.
        Use it, together with the triangle inequality
        (\cref{thm:g7:triangles:inequality}) in the triangle
        $ABA'$, to prove the \emph{median bound}: the median
        $[AO]$ satisfies
        \[
        AO < \frac{AB + AC}{2} .
        \]
        (Note that $AA' = 2\,AO$ and $BA' = AC$.)
  \item In a triangle with $AB = 5$ cm and $AC = 7$ cm, between
        which two values must the length of the median from $A$
        lie? (Use both directions of the triangle inequality in
        $ABA'$.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- How many centers can a figure
have?}
\begin{enumerate}[resume]
  \item For each figure, say whether it has a \omterm{def:g7:central:center}{center of
        symmetry}, and where: a \omterm{def:g6:lines:objects}{segment}; a full line; an
        \omterm{def:g6:shapes:triangles}{equilateral triangle}; a square; a circle; the letters S
        and Z.
  \item Prove your claim for the \omterm{def:g6:shapes:triangles}{equilateral triangle}: a
        half-turn sending the triangle to itself would pair up
        the three \omterm{def:g5:solids:def}{vertices} --- but three is \omterm{def:g3:numbers:evenodd}{odd}, so some \omterm{def:g5:solids:def}{vertex}
        would be sent to itself. What would that force the
        \omterm{def:g5:solids:def}{vertex} to be, and why is it impossible?
  \item Generalize question 12: explain why \emph{no} \omterm{def:g4:geometry:polygon}{polygon}
        with an \omterm{def:g3:numbers:evenodd}{odd number} of \omterm{def:g5:solids:def}{vertices} has a \omterm{def:g7:central:center}{center of
        symmetry}.
  \item The pearl: suppose a figure had \emph{two} distinct
        centers of symmetry $O_1$ and $O_2$. Perform the two
        half-turns one after the other and use the discovery of
        question 5: what motion must send the figure exactly
        onto itself? Explain why a figure that fits on a sheet
        of paper cannot survive that, and conclude.
  \item Question 14's conclusion leaves a loophole:
        \emph{unbounded} figures. Verify on an infinite frieze
        --- say, an endless strip of footprints, left, right,
        left, right\,\dots\ --- that it has (at least) two
        different centers of symmetry, and identify the
        translation that question 14 predicted.
\end{enumerate}
\end{problem}
