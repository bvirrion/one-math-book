\chapter{Negative Numbers}\label{ch:g7:negatives}

Temperatures below \omterm{def:g1:counting:zero}{zero}, floors below ground, money owed: many
quantities go below $0$. Negative numbers extend the number line to the
left of \omterm{def:g1:counting:zero}{zero}. This chapter teaches how to compare them and how to add
and subtract them --- their multiplication waits for
\cref{ch:g8:negprod}.

\section{Relative numbers}

\begin{definition}[Relative numbers]\label{def:g7:negatives:def}
A \emph{relative number}\index{relative number} is made of a sign
($+$ or $-$) and a \emph{distance to \omterm{def:g1:counting:zero}{zero}}: $+3$ and $-3$ are both at
distance $3$ from $0$, on opposite sides. Positive numbers are usually
written without their sign: $+3 = 3$. The numbers $+3$ and $-3$ are
\emph{opposites}\index{opposite} of each other.
\end{definition}

\begin{omfigure}
\begin{tikzpicture}[scale=1.15]
  \draw[-Stealth] (-5.6,0) -- (5.6,0);
  \foreach \x in {-5,...,5} \draw (\x,-0.08) -- (\x,0.08);
  \foreach \x in {-5,...,5} \node[below, font=\small] at (\x,-0.1) {$\x$};
  \fill[omThm] (-3,0) circle (2pt);
  \fill[omDef] (3,0) circle (2pt);
  \draw[Stealth-Stealth, omExo] (-3,0.35) -- (0,0.35)
    node[midway, above, font=\small] {$3$};
  \draw[Stealth-Stealth, omExo] (0,0.35) -- (3,0.35)
    node[midway, above, font=\small] {$3$};
\end{tikzpicture}

{\small The number line extended to the left: $-3$ and $+3$ are
opposite numbers, at the same distance from $0$.}
\end{omfigure}

\begin{method}[Comparing relative numbers]\label{met:g7:negatives:compare}
On the number line, smaller means further left. In practice:
\begin{enumerate}
  \item a negative number is always smaller than a positive one:
        $-7 < 2$;
  \item between two positives, the usual order;
  \item between two negatives, the one with the \emph{larger} distance
        to \omterm{def:g1:counting:zero}{zero} is the \emph{smaller}: $-7 < -2$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:negatives:compare}
Order from smallest to largest: $3$; $-5$; $0$; $-1.5$; $2.8$.
Negatives first, the most negative leading:
\[
-5 < -1.5 < 0 < 2.8 < 3 .
\]
\end{example}

\section{Adding relative numbers}

Think of a \omterm{def:g7:negatives:def}{relative number} as a movement along the line: $+5$ means
``walk $5$ steps right'', $-3$ means ``walk $3$ steps left''. Adding
numbers means chaining the walks.

\begin{method}[Adding two relative numbers]\label{met:g7:negatives:add}
\begin{enumerate}
  \item \emph{Same sign}: add the distances, keep the common sign:
        $(-3) + (-4) = -7$.
  \item \emph{Opposite signs}: subtract the smaller distance from the
        larger, take the sign of the larger:
        $(-7) + (+2) = -5$ and $(+7) + (-2) = +5$.
  \item \omterm{def:g7:negatives:def}{Opposites} cancel: $(+4) + (-4) = 0$.
\end{enumerate}
\end{method}

\begin{omfigure}
\begin{tikzpicture}[scale=1.05]
  \draw[-Stealth] (-5.4,0) -- (4.4,0);
  \foreach \x in {-5,...,4} \draw (\x,-0.07) -- (\x,0.07);
  \foreach \x in {-5,-3,-1,0,2,4}
    \node[below, font=\small] at (\x,-0.09) {$\x$};
  \draw[-Stealth, omThm, very thick] (0,0.35) -- (-3.5,0.35);
  \draw[-Stealth, omDef, very thick] (-3.5,0.75) -- (-1.5,0.75);
  \node[omThm, font=\small] at (-1.75,0.18) {$-3.5$};
  \node[omDef, font=\small] at (-2.5,1.0) {$+2$};
  \fill (-1.5,0) circle (2pt) node[below=6pt, font=\small] {$-1.5$};
\end{tikzpicture}

{\small Adding as walking: starting at $0$, the walk $-3.5$ then the
walk $+2$ lands at $(-3.5) + 2 = -1.5$.}
\end{omfigure}

\begin{example}\label{ex:g7:negatives:add}
Step by step, grouping as convenient:
\begin{align*}
(-8) + (+3) &= -5 && \text{(opposite signs: } 8 - 3 = 5,
\text{ sign of } 8)\\
(-2.5) + (-1.5) &= -4 && \text{(same sign: add distances)}\\
(+6) + (-9) + (+3) &= (-3) + (+3) = 0 .
\end{align*}
\end{example}

\section{Subtracting relative numbers}

\begin{theorem}[Subtraction rule]\label{thm:g7:negatives:sub}
Subtracting a \omterm{def:g7:negatives:def}{relative number} is the same as \emph{adding its
opposite}:
\[
a - b = a + (-b) .
\]
\end{theorem}

\begin{proof}[Why it works]
$a - b$ is the number that added to $b$ gives $a$. Check that
$a + (-b)$ does the job:
$\bigl(a + (-b)\bigr) + b = a + \bigl((-b) + b\bigr) = a + 0 = a$.
\end{proof}

\begin{example}\label{ex:g7:negatives:sub}
Every subtraction becomes an addition, then
\cref{met:g7:negatives:add} applies:
\begin{align*}
5 - 9 &= 5 + (-9) = -4, \\
(-3) - (+4) &= (-3) + (-4) = -7, \\
(-2) - (-6) &= (-2) + (+6) = +4 .
\end{align*}
The last line is the famous one: \emph{subtracting a negative means
adding}. Removing a debt makes you richer!
\end{example}

\begin{example}[Temperature differences]\label{ex:g7:negatives:temperature}
In one day the temperature went from $-6$ degrees to $+9$ degrees. The
rise is the \omterm{ex:g1:subtraction:difference}{difference}
\[
9 - (-6) = 9 + 6 = 15 \text{ degrees}.
\]
On the number line: from $-6$ up to $0$ is $6$ degrees, from $0$ to
$9$ is $9$ more.
\end{example}

\section{Coordinates in the whole plane}

\begin{definition}[Coordinates with signs]\label{def:g7:negatives:coords}
With two \omterm{def:g6:lines:perp}{perpendicular} graduated axes crossing at the origin $O$,
every point of the plane gets two relative coordinates $(x, y)$:
negative $x$ means left of the vertical axis, negative $y$ means below
the horizontal axis.
\end{definition}

\end{exercise}

-4 \;?\; 1, \qquad
-4 \;?\; -7, \qquad
-2.5 \;?\; -2.4, \qquad
0 \;?\; -0.1 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:3}
Order from smallest to largest: $-3$; \ $2.5$; \ $-0.5$; \ $-3.1$;

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:4}

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:7}
The temperature was $-3$ degrees this morning; it rose by $8$ degrees
during the day, then dropped by $12$ degrees at night. Write a single

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:8}
Plot in a coordinate system: $A(2, 3)$; $B(-3, 1)$; $C(-2, -2)$;

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:negatives:9}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:negatives:10}
A diver is at altitude $-12$ m (twelve meters below the surface). She

\end{exercise}

\section{Problem: The year zero that never was}

\begin{problem}[{Weekend problem --- relative numbers on the
timeline of history: durations across the era boundary, the
astronomers' trick, and the net-change principle}]
\label{pb:g7:negatives:1}
The number line of this chapter has a famous real-world twin: the
timeline of history, with the years BC stretching left and the
years AD stretching right. But the historians' line hides a trap
that has spoiled many a computation: \emph{there is no year \omterm{def:g1:counting:zero}{zero}}
--- the year 1 BC is followed immediately by AD 1. This problem
computes with temperatures, lifts and bank accounts, then repairs
history's broken \omterm{def:g1:counting:zero}{zero} the way astronomers do: with \omterm{def:g7:negatives:def}{relative
numbers}.

\medskip
\noindent\textbf{Part I --- Warm-ups, and a first trap.}
\begin{enumerate}
  \item The summit of Mont Blanc is at altitude $+4\,808$ m; the
        shore of the Dead Sea at $-430$ m. Compute the \omterm{ex:g1:subtraction:difference}{difference}
        in altitude between them.
  \item A bank account holds $-45$ euros (an overdraft). Its
        owner deposits $120$ euros, pays a bill of $60$ euros,
        then deposits $15$ euros. Write the balance as a single
        chain of additions and compute it. What single deposit
        would have brought the original $-45$ back to exactly
        $0$?
  \item Julius Caesar was born in $100$ BC and died in $44$ BC.
        Counting BC years as negative numbers ($100$ BC as
        $-100$, $44$ BC as $-44$), compute $(-44) - (-100)$: how
        many years did Caesar live?
  \item Emperor Augustus was born in $63$ BC and died in AD
        $14$. The same recipe gives $14 - (-63) = 77$ years ---
        but historians insist he died at $76$. Their objection:
        the count $77$ walks through a year that never existed.
        Which one?
  \item Astronomers repair the timeline by relabelling: AD years
        keep their number ($+14$ stays $+14$), but $1$ BC
        becomes $0$, $2$ BC becomes $-1$, and generally the year
        $n$ BC becomes $-(n - 1)$. Convert $63$ BC, and redo the
        computation of question 4 with the astronomers' numbers.
        Does the answer now satisfy the historians?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Computing across the eras.}
From now on, use the astronomers' relabelling for every
computation, converting back to BC/AD at the end (a year labelled
$0$ or negative $-m$ is the historians' year $m + 1$ BC).
\begin{enumerate}[resume]
  \item Convert to astronomers' numbers: $1$ BC; $10$ BC; AD
        $476$; and $753$ BC, the legendary founding of Rome.
  \item How many years passed from the founding of Rome
        ($753$ BC) to the fall of the Western Empire (AD $476$)?
  \item A comet returns every $76$ years and appeared in
        $12$ BC. Give the historians' years of its next three
        appearances, and of the appearance just before $12$ BC.
  \item How many years passed from $5$ BC to AD $5$? (The
        tempting answer is $10$; the timeline says otherwise.)
  \item A child born in $3$ BC turned ten years old in which
        year (BC or AD)?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The net-change principle.}
\begin{enumerate}[resume]
  \item The \emph{distance} between two numbers on the line is
        the larger minus the smaller --- always positive.
        Compute the distance between $-7$ and $-2$, then between
        $-3.5$ and $4.5$.
  \item A lift starts at floor $0$ and makes the moves $+3$,
        $-5$, $+2$, $-1$, $+4$. Compute the final floor by
        grouping cleverly (\cref{met:g7:negatives:add},
        \omterm{def:g7:negatives:def}{opposites} first where possible). Would performing the
        moves in a different order change the destination? Why?
  \item Day by day, the temperature changes by $+2$, $-4$, $+1$,
        $0$, $-3$, $+5$, $-2$ degrees over a week. Compute the
        \emph{\omterm{def:g5:solids:net}{net} change}. If the week started at $-2$ degrees,
        where did it end? State the principle: final value $=$
        starting value $+$ (\omterm{def:g1:addition:def}{sum} of all changes).
  \item A hiker's walk brings her back exactly to her starting
        altitude. What must the \omterm{def:g1:addition:def}{sum} of all her altitude changes
        be, and why? Of her six recorded changes, five are
        $+4$, $-7$, $+2$, $+3$, $-5$ (in hundreds of meters):
        find the sixth.
  \item The finale, back in Rome. A historian computes: ``the
        Republic ran from $509$ BC to $27$ BC, then the Empire
        to AD $476$; in all, from $509$ BC to AD $476$, that is
        $509 + 476 = 985$ years of Roman state.'' Redo the
        computation with astronomers' numbers. What is the
        correct total --- and which year, once again, caused the
        error?
\end{enumerate}
\end{problem}
