\chapter{Negative Numbers}\label{ch:g7:negatives}

Temperatures below \omterm{def:g1:counting:zero}{zero}, floors below ground, money owed: many
quantities go below $0$. Negative numbers extend the number line to the
left of \omterm{def:g1:counting:zero}{zero}. This chapter teaches how to compare them and how to add
and subtract them --- their multiplication waits for
\cref{ch:g8:negprod}.

\section{Relative numbers}

\begin{definition}[Relative numbers]\label{def:g7:negatives:def}
A \emph{relative number}\index{relative number} is made of a sign
($+$ or $-$) and a \emph{distance to \omterm{def:g1:counting:zero}{zero}}: $+3$ and $-3$ are both at
distance $3$ from $0$, on opposite sides. Positive numbers are usually
written without their sign: $+3 = 3$. The numbers $+3$ and $-3$ are
\emph{opposites}\index{opposite} of each other.
\end{definition}

\begin{omfigure}
\begin{tikzpicture}[scale=1.15]
  \draw[-Stealth] (-5.6,0) -- (5.6,0);
  \foreach \x in {-5,...,5} \draw (\x,-0.08) -- (\x,0.08);
  \foreach \x in {-5,...,5} \node[below, font=\small] at (\x,-0.1) {$\x$};
  \fill[omThm] (-3,0) circle (2pt);
  \fill[omDef] (3,0) circle (2pt);
  \draw[Stealth-Stealth, omExo] (-3,0.35) -- (0,0.35)
    node[midway, above, font=\small] {$3$};
  \draw[Stealth-Stealth, omExo] (0,0.35) -- (3,0.35)
    node[midway, above, font=\small] {$3$};
\end{tikzpicture}

{\small The number line extended to the left: $-3$ and $+3$ are
opposite numbers, at the same distance from $0$.}
\end{omfigure}

\begin{method}[Comparing relative numbers]\label{met:g7:negatives:compare}
On the number line, smaller means further left. In practice:
\begin{enumerate}
  \item a negative number is always smaller than a positive one:
        $-7 < 2$;
  \item between two positives, the usual order;
  \item between two negatives, the one with the \emph{larger} distance
        to \omterm{def:g1:counting:zero}{zero} is the \emph{smaller}: $-7 < -2$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:negatives:compare}
Order from smallest to largest: $3$; $-5$; $0$; $-1.5$; $2.8$.
Negatives first, the most negative leading:
\[
-5 < -1.5 < 0 < 2.8 < 3 .
\]
\end{example}

\section{Adding relative numbers}

Think of a \omterm{def:g7:negatives:def}{relative number} as a movement along the line: $+5$ means
``walk $5$ steps right'', $-3$ means ``walk $3$ steps left''. Adding
numbers means chaining the walks.

\begin{method}[Adding two relative numbers]\label{met:g7:negatives:add}
\begin{enumerate}
  \item \emph{Same sign}: add the distances, keep the common sign:
        $(-3) + (-4) = -7$.
  \item \emph{Opposite signs}: subtract the smaller distance from the
        larger, take the sign of the larger:
        $(-7) + (+2) = -5$ and $(+7) + (-2) = +5$.
  \item \omterm{def:g7:negatives:def}{Opposites} cancel: $(+4) + (-4) = 0$.
\end{enumerate}
\end{method}

\begin{omfigure}
\begin{tikzpicture}[scale=1.05]
  \draw[-Stealth] (-5.4,0) -- (4.4,0);
  \foreach \x in {-5,...,4} \draw (\x,-0.07) -- (\x,0.07);
  \foreach \x in {-5,-3,-1,0,2,4}
    \node[below, font=\small] at (\x,-0.09) {$\x$};
  \draw[-Stealth, omThm, very thick] (0,0.35) -- (-3.5,0.35);
  \draw[-Stealth, omDef, very thick] (-3.5,0.75) -- (-1.5,0.75);
  \node[omThm, font=\small] at (-1.75,0.18) {$-3.5$};
  \node[omDef, font=\small] at (-2.5,1.0) {$+2$};
  \fill (-1.5,0) circle (2pt) node[below=6pt, font=\small] {$-1.5$};
\end{tikzpicture}

{\small Adding as walking: starting at $0$, the walk $-3.5$ then the
walk $+2$ lands at $(-3.5) + 2 = -1.5$.}
\end{omfigure}

\begin{example}\label{ex:g7:negatives:add}
Step by step, grouping as convenient:
\begin{align*}
(-8) + (+3) &= -5 && \text{(opposite signs: } 8 - 3 = 5,
\text{ sign of } 8)\\
(-2.5) + (-1.5) &= -4 && \text{(same sign: add distances)}\\
(+6) + (-9) + (+3) &= (-3) + (+3) = 0 .
\end{align*}
\end{example}

\section{Subtracting relative numbers}

\begin{theorem}[Subtraction rule]\label{thm:g7:negatives:sub}
Subtracting a \omterm{def:g7:negatives:def}{relative number} is the same as \emph{adding its
opposite}:
\[
a - b = a + (-b) .
\]
\end{theorem}

\begin{proof}[Why it works]
$a - b$ is the number that added to $b$ gives $a$. Check that
$a + (-b)$ does the job:
$\bigl(a + (-b)\bigr) + b = a + \bigl((-b) + b\bigr) = a + 0 = a$.
\end{proof}

\begin{example}\label{ex:g7:negatives:sub}
Every subtraction becomes an addition, then
\cref{met:g7:negatives:add} applies:
\begin{align*}
5 - 9 &= 5 + (-9) = -4, \\
(-3) - (+4) &= (-3) + (-4) = -7, \\
(-2) - (-6) &= (-2) + (+6) = +4 .
\end{align*}
The last line is the famous one: \emph{subtracting a negative means
adding}. Removing a debt makes you richer!
\end{example}

\begin{example}[Temperature differences]\label{ex:g7:negatives:temperature}
In one day the temperature went from $-6$ degrees to $+9$ degrees. The
rise is the \omterm{ex:g1:subtraction:difference}{difference}
\[
9 - (-6) = 9 + 6 = 15 \text{ degrees}.
\]
On the number line: from $-6$ up to $0$ is $6$ degrees, from $0$ to
$9$ is $9$ more.
\end{example}

\section{Coordinates in the whole plane}

\begin{definition}[Coordinates with signs]\label{def:g7:negatives:coords}
With two \omterm{def:g6:lines:perp}{perpendicular} graduated axes crossing at the origin $O$,
every point of the plane gets two relative coordinates $(x, y)$:
negative $x$ means left of the vertical axis, negative $y$ means below
the horizontal axis.
\end{definition}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g7:negatives:1}
Give the opposite of: $7$; \ $-3.5$; \ $0$; \ $+12$. Then complete:
``two numbers are \omterm{def:g7:negatives:def}{opposites} when their \omterm{def:g1:addition:def}{sum} is~?''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:2}
Copy and complete with $<$ or $>$:
\[
-4 \;?\; 1, \qquad
-4 \;?\; -7, \qquad
-2.5 \;?\; -2.4, \qquad
0 \;?\; -0.1 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:3}
Order from smallest to largest: $-3$; \ $2.5$; \ $-0.5$; \ $-3.1$;
\ $0$; \ $1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:4}
Compute:
\[
(-5) + (-9), \qquad
(-12) + (+7), \qquad
(+3.5) + (-1.5), \qquad
(-6) + (+6).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:5}
Transform into additions, then compute:
\[
4 - 11, \qquad
(-5) - (+3), \qquad
(-7) - (-10), \qquad
0 - (-8).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:6}
Compute by grouping cleverly (\omterm{def:g7:negatives:def}{opposites} first):
\[
(+7) + (-12) + (+3) + (-8), \qquad
(-5.5) + (+9) + (-4.5) + (+1).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:7}
The temperature was $-3$ degrees this morning; it rose by $8$ degrees
during the day, then dropped by $12$ degrees at night. Write a single
computation giving the final temperature, and compute it.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:8}
Plot in a coordinate system: $A(2, 3)$; $B(-3, 1)$; $C(-2, -2)$;
$D(3, -1)$; $E(0, -3)$. Which point lies on an axis?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:negatives:9}
Compute the \omterm{ex:g1:subtraction:difference}{difference} between the highest and lowest temperature:
Moscow in January, max $-4$ degrees, min $-13$ degrees; Helsinki, max
$2$, min $-9$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:negatives:10}
A diver is at altitude $-12$ m (twelve meters below the surface). She
rises $7$ m, then descends $4$ m. At what altitude is she? How far
must she still rise to reach the surface?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:negatives:11}
Plot $A(-2, 1)$, $B(2, 1)$, $C(2, -2)$. Find the coordinates of the
point $D$ so that $ABCD$ is a rectangle, and compute its \omterm{def:g6:measure:perimeter}{perimeter}
(count the grid squares for the side lengths).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:negatives:12}
In the magic square below, every row, column and diagonal must have
the same \omterm{def:g1:addition:def}{sum}. Copy and complete it:
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|c|c|}
\hline
$?$ & $-7$ & $0$ \\
\hline
$-3$ & $-2$ & $?$ \\
\hline
$-4$ & $?$ & $?$ \\
\hline
\end{tabular}
\end{center}
(Start by finding the magic \omterm{def:g1:addition:def}{sum} using the diagonal that is almost
complete.)
\end{exercise}
