\chapter{Triangles et angles}\label{ch:g7:triangles}

Peut-on tracer un triangle de côtés $10$, $3$ et $2$~? D'angles
$80^\circ$, $70^\circ$ et $50^\circ$~? Ce chapitre répond aux deux
questions avec deux des faits les plus utiles de la géométrie plane~:
l'inégalité triangulaire et la \omterm{def:g1:addition:def}{somme} des angles $180^\circ$ --- plus
le vocabulaire des angles qui accompagne les droites sécantes et
\omterm{def:g6:lines:perp}{parallèles}.

\section{Angles autour de droites sécantes}

\begin{definition}[Paires d'angles]\label{def:g7:triangles:pairs}
\begin{itemize}
  \item Deux angles sont \emph{complémentaires} lorsque leurs mesures
        s'additionnent à $90^\circ$, \emph{supplémentaires} lorsqu'elles
        s'additionnent à $180^\circ$.
  \item Lorsque deux droites se croisent, les angles se faisant \omterm{def:g5:solids:def}{face}
        de part et d'autre du point d'intersection sont
        \emph{opposés par le
        sommet}\index{angles opposés par le sommet}.
  \item Lorsqu'une droite coupe deux autres droites, les angles
        situés dans la même position à chaque intersection sont des
        \emph{angles correspondants}~; les angles situés entre les
        deux droites de part et d'autre de la sécante sont des
        \emph{angles alternes-internes}\index{angles alternes-internes}.
\end{itemize}
\end{definition}

\begin{theorem}[Égalités d'angles]\label{thm:g7:triangles:equalities}
\begin{enumerate}
  \item Les \omterm{def:g7:triangles:pairs}{angles opposés par le sommet} sont égaux.
  \item Si deux droites sont \omterm{def:g6:lines:perp}{parallèles}, les angles correspondants
        sont égaux, et les \omterm{def:g7:triangles:pairs}{angles alternes-internes} sont égaux.
        Réciproquement, des angles correspondants (ou
        alternes-internes) égaux forcent les droites à être
        \omterm{def:g6:lines:perp}{parallèles}.
\end{enumerate}
\end{theorem}

\begin{proof}[Preuve du 1]
Les deux angles $\widehat{1}$ et $\widehat{2}$ d'un même côté d'une
droite forment un angle plat~: $\widehat 1 + \widehat 2 = 180^\circ$.
L'angle $\widehat 3$ opposé par le \omterm{def:g5:solids:def}{sommet} à $\widehat 1$ satisfait
aussi $\widehat 2 + \widehat 3 = 180^\circ$ (même raison). Donc
$\widehat 1 = 180^\circ - \widehat 2 = \widehat 3$. Le point 2 est
admis à ce niveau.
\end{proof}

\section{La somme des angles d'un triangle}

\begin{theorem}[Somme des angles]\label{thm:g7:triangles:sum}
Dans tout triangle, les trois angles s'additionnent à $180^\circ$.
\end{theorem}

\begin{proof}
Soit $ABC$ un triangle. Tracer la droite passant par $A$ \omterm{def:g6:lines:perp}{parallèle} à
$(BC)$. Au \omterm{def:g5:solids:def}{sommet} $A$, trois angles s'alignent le long de cette
\omterm{def:g6:lines:perp}{parallèle} et forment un angle plat, $180^\circ$~: celui du milieu est
$\widehat{A}$, et les deux extérieurs sont des \omterm{def:g7:triangles:pairs}{angles
alternes-internes} avec $\widehat B$ et $\widehat C$ (\omterm{def:g6:lines:perp}{droites
parallèles}~!), donc ils égalent $\widehat B$ et $\widehat C$. D'où
$\widehat B + \widehat A + \widehat C = 180^\circ$.
\end{proof}

\begin{example}\label{ex:g7:triangles:sum}
Deux angles d'un triangle mesurent $67^\circ$ et $58^\circ$. Le
troisième mesure
\[
180 - (67 + 58) = 180 - 125 = 55^\circ .
\]
Cas particuliers à mémoriser~: chaque angle d'un \omterm{def:g6:shapes:triangles}{triangle équilatéral}
vaut $180 \div 3 = 60^\circ$~; dans un \omterm{def:g6:shapes:triangles}{triangle rectangle} les deux
angles aigus sont complémentaires (ils s'additionnent à
$180 - 90 = 90^\circ$)~; les angles à la base d'un \omterm{def:g6:shapes:triangles}{triangle isocèle}
sont égaux, donc chacun vaut $\frac{180 - \text{angle au sommet}}{2}$.
\end{example}

\begin{example}[Pas de triangle à deux angles droits]\label{ex:g7:triangles:tworight}
Un triangle avec deux angles de $90^\circ$ utiliserait déjà
$180^\circ$, ne laissant $0^\circ$ pour le troisième --- impossible.
La \omterm{def:g1:addition:def}{somme} des angles interdit d'un coup bien des triangles.
\end{example}

\section{L'inégalité triangulaire}

\begin{theorem}[Inégalité triangulaire]\label{thm:g7:triangles:inequality}
Dans tout triangle, chaque côté est plus court que la \omterm{def:g1:addition:def}{somme} des deux
autres~: pour un triangle de côtés $a$, $b$, $c$,
\[
a < b + c .
\]
Réciproquement, trois longueurs dont la \emph{plus grande} est
strictement inférieure à la \omterm{def:g1:addition:def}{somme} des deux autres peuvent toujours
former un triangle. Si la plus grande \emph{égale} la \omterm{def:g1:addition:def}{somme} des
autres, le «~triangle~» est plat~: les trois \omterm{def:g5:solids:def}{sommets} sont alignés.
\end{theorem}
\admitted

\begin{example}\label{ex:g7:triangles:inequality}
Des côtés $10$, $3$ et $2$ sont-ils possibles~? Tester le plus
grand~: a-t-on $10 < 3 + 2 = 5$~? Non --- aucun tel triangle
n'existe. Le compas montre pourquoi~: les arcs de \omterm{def:g3:shapes:shapes}{rayons} $3$ et $2$
tracés des deux extrémités d'un \omterm{def:g6:lines:objects}{segment} de longueur $10$ ne se
rencontrent jamais.

Côtés $7$, $5$ et $4$~? Plus grand~: $7 < 5 + 4 = 9$~: oui, le
triangle existe (vérifier le plus grand côté suffit).
\end{example}

\begin{method}[Existence et construction d'un triangle]\label{met:g7:triangles:construct}
Étant donné trois longueurs~:
\begin{enumerate}
  \item comparer la plus grande à la \omterm{def:g1:addition:def}{somme} des deux autres~; si elle
        n'est pas strictement plus petite, s'arrêter~: pas de
        triangle~;
  \item sinon construire comme en \cref{met:g6:shapes:construct}
        (\omterm{def:g6:lines:objects}{segment} de base, deux arcs de compas)~;
  \item après toute construction à partir d'angles, vérifier
        mentalement que les angles donnés somment à moins de
        $180^\circ$ --- le troisième angle complétant exactement
        $180^\circ$.
\end{enumerate}
\end{method}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g7:triangles:1}
Donner le complément (à $90^\circ$) et le supplément (à $180^\circ$)
de~: $30^\circ$~; \ $45^\circ$~; \ $72^\circ$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:triangles:2}
Deux droites se croisent~; l'un des quatre angles mesure $35^\circ$.
Donner les mesures des trois autres, avec une raison pour chacune.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:triangles:3}
Calculer le troisième angle d'un triangle dont les deux premiers
mesurent~: (a)~$40^\circ$ et $60^\circ$~; \
(b)~$90^\circ$ et $28^\circ$~; \
(c)~$75^\circ$ et $75^\circ$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:triangles:4}
Un \omterm{def:g6:shapes:triangles}{triangle isocèle} a son angle au \omterm{def:g5:solids:def}{sommet} égal à $40^\circ$. Calculer
ses deux angles à la base. Un autre \omterm{def:g6:shapes:triangles}{triangle isocèle} a un angle à la
base de $70^\circ$~: calculer son angle au \omterm{def:g5:solids:def}{sommet}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:triangles:5}
Lesquels de ces triplets peuvent être les côtés d'un triangle~?
Justifier par l'inégalité triangulaire (le plus grand côté~!)~:
\[
(6,\ 8,\ 12); \qquad
(5,\ 5,\ 11); \qquad
(4,\ 9,\ 13); \qquad
(7,\ 7,\ 7).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:triangles:6}
Un triangle a pour angles $x$, $2x$ et $3x$. Trouver $x$ et les trois
angles. De quel type de triangle s'agit-il~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:triangles:7}
Construire un triangle $ABC$ avec $BC = 6$ cm,
$\widehat{ABC} = 50^\circ$ et $\widehat{ACB} = 60^\circ$ (rapporteur
en $B$ et en $C$). Avant de construire, prédire la mesure de
$\widehat{BAC}$, puis vérifier sur la figure.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:triangles:8}
Deux \omterm{def:g6:lines:perp}{droites parallèles} sont coupées par une troisième, formant un
angle de $54^\circ$ à la première intersection (entre la sécante et
une \omterm{def:g6:lines:perp}{parallèle}). Dessiner la situation et donner les mesures des huit
angles formés.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:triangles:9}
Deux côtés d'un triangle mesurent $8$ cm et $3$ cm. Entre quelles
deux valeurs le troisième côté doit-il se situer~? Donner une
longueur entière qui convient et une qui ne convient pas.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:triangles:10}
Dans un triangle $ABC$, $\widehat A = 2\widehat B$ et
$\widehat C = 90^\circ$. Calculer $\widehat A$ et $\widehat B$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:triangles:11}
Les trois angles de tout \omterm{def:g4:geometry:polygon}{quadrilatère} $ABCD$ peuvent s'étudier en le
découpant le long d'une diagonale en deux triangles. Utiliser cela
pour prouver que les quatre angles de tout \omterm{def:g4:geometry:polygon}{quadrilatère}
s'additionnent à $360^\circ$, et en déduire la \omterm{def:g1:addition:def}{somme} des angles d'un
\omterm{def:g4:geometry:polygon}{pentagone} (cinq côtés).
\end{exercise}
