\chapter{Fractions~: comparer et additionner}\label{ch:g7:fractions}

L'année 6 a présenté les \omterm{def:g6:fractions:def}{fractions} comme parts et comme \omterm{def:g3:division:remainder}{quotients}
exacts (\cref{ch:g6:fractions}). Ce chapitre apprend à
\emph{calculer} avec elles~: reconnaître des \omterm{def:g6:fractions:def}{fractions} égales,
comparer, additionner et soustraire --- pour l'instant avec des
\omterm{def:g4:fractions:def}{dénominateurs} amiables~; le cas général arrive au
\cref{ch:g9:fractions}.

\section{Fractions égales}

\begin{theorem}[Fractions égales]\label{thm:g7:fractions:equal}
Une \omterm{def:g6:fractions:def}{fraction} est inchangée lorsque son \omterm{def:g4:fractions:def}{numérateur} et son \omterm{def:g4:fractions:def}{dénominateur}
sont multipliés, ou divisés, par le même nombre non nul~:
\[
\frac{a}{b} = \frac{a \times k}{b \times k} .
\]
\end{theorem}
\admitted

\begin{example}\label{ex:g7:fractions:equal}
$\dfrac{3}{4} = \dfrac{3 \times 5}{4 \times 5} = \dfrac{15}{20}$, et
dans l'autre sens $\dfrac{42}{30} = \dfrac{42 \div 6}{30 \div 6}
= \dfrac{7}{5}$ (simplifiée par $6$).

Pour tester si $\dfrac{8}{12}$ et $\dfrac{10}{15}$ sont égales,
simplifier les deux~: $\dfrac{8}{12} = \dfrac{2}{3}$ et
$\dfrac{10}{15} = \dfrac{2}{3}$ --- oui, égales.
\end{example}

\section{Comparer des fractions}

\begin{method}[Comparer deux fractions]\label{met:g7:fractions:compare}
\begin{enumerate}
  \item \emph{Même \omterm{def:g4:fractions:def}{dénominateur}}~: comparer les \omterm{def:g4:fractions:def}{numérateurs}~:
        $\frac{5}{9} < \frac{7}{9}$.
  \item \emph{Un \omterm{def:g4:fractions:def}{dénominateur} est \omterm{def:g5:division:multiple}{multiple} de l'autre}~: réécrire la
        \omterm{def:g6:fractions:def}{fraction} la plus grossière, puis comparer~:
        $\frac{5}{6}$ contre $\frac{7}{12}$~: $\frac56 = \frac{10}{12}
        > \frac{7}{12}$.
  \item \emph{Comparer à $1$ ou $\frac12$} lorsque c'est possible~:
        $\frac{9}{8} > 1 > \frac{7}{9}$ tranche
        $\frac98$ contre $\frac79$ d'un coup.
\end{enumerate}
\end{method}

\section{Additionner et soustraire}

\begin{theorem}[Même dénominateur]\label{thm:g7:fractions:add}
Des \omterm{def:g6:fractions:def}{fractions} de même \omterm{def:g4:fractions:def}{dénominateur} s'additionnent (ou se soustraient)
en additionnant (ou soustrayant) les \omterm{def:g4:fractions:def}{numérateurs}~:
\[
\frac{a}{d} + \frac{b}{d} = \frac{a + b}{d},
\qquad
\frac{a}{d} - \frac{b}{d} = \frac{a - b}{d} .
\]
\end{theorem}

\begin{proof}[Pourquoi]
$a$ parts de taille $\frac1d$ plus $b$ parts de la même taille font
$a + b$ parts de cette taille.
\end{proof}

\begin{method}[Dénominateurs différents]\label{met:g7:fractions:add}
Lorsqu'un \omterm{def:g4:fractions:def}{dénominateur} est \omterm{def:g5:division:multiple}{multiple} de l'autre~:
\begin{enumerate}
  \item réécrire la \omterm{def:g6:fractions:def}{fraction} de plus petit \omterm{def:g4:fractions:def}{dénominateur} pour que les
        deux aient le plus grand (\cref{thm:g7:fractions:equal})~;
  \item additionner ou soustraire les \omterm{def:g4:fractions:def}{numérateurs}~;
  \item simplifier le résultat si possible.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:fractions:add}
Calculer $\dfrac{3}{4} + \dfrac{5}{8}$, étape par étape~:
\begin{enumerate}
  \item $8$ est \omterm{def:g5:division:multiple}{multiple} de $4$~: réécrire
        $\dfrac34 = \dfrac{3 \times 2}{4 \times 2} = \dfrac{6}{8}$~;
  \item additionner~: $\dfrac{6}{8} + \dfrac{5}{8} = \dfrac{11}{8}$~;
  \item rien à simplifier~: le résultat est $\dfrac{11}{8}$, c'est-à-dire
        $1 + \dfrac38$.
\end{enumerate}
Autre~: $2 - \dfrac{4}{5} = \dfrac{10}{5} - \dfrac{4}{5} =
\dfrac{6}{5}$ (écrire d'abord le nombre entier sur le \omterm{def:g4:fractions:def}{dénominateur}
$5$).
\end{example}

\begin{example}[Un piège classique]\label{ex:g7:fractions:trap}
$\dfrac12 + \dfrac13$ n'est \emph{pas} $\dfrac{2}{5}$~! Additionner
séparément \omterm{def:g4:fractions:def}{numérateurs} et \omterm{def:g4:fractions:def}{dénominateurs} est faux --- la réponse
$\frac25$ serait plus petite que $\frac12$, absurde quand on
\emph{ajoute} une quantité positive à $\frac12$. (La \omterm{def:g1:addition:def}{somme} correcte,
$\frac56$, demande un \omterm{def:g4:fractions:def}{dénominateur} commun $6$~; la méthode générale
est au \cref{ch:g9:fractions}.)
\end{example}

\section{Multiplier une fraction par un nombre}

\begin{proposition}[Fraction fois un nombre]\label{prop:g7:fractions:mult}
Pour un nombre $k$~:
\[
k \times \frac{a}{b} = \frac{k \times a}{b} .
\]
Prendre $\frac ab$ \emph{de} une quantité, c'est multiplier par
$\frac ab$.
\end{proposition}
\admitted

\begin{example}\label{ex:g7:fractions:mult}
$6 \times \dfrac{2}{3} = \dfrac{12}{3} = 4$~: six fois deux tiers font
quatre entiers. Et «~$\frac34$ de $60$ euros~» c'est
$\frac34 \times 60 = \frac{180}{4} = 45$ euros --- la même réponse
qu'en divisant par $4$ puis multipliant par $3$
(\cref{met:g6:fractions:ofquantity}).
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g7:fractions:1}
Compléter~: $\dfrac{2}{5} = \dfrac{?}{20}$~; \quad
$\dfrac{18}{24} = \dfrac{3}{?}$~; \quad
$\dfrac{7}{3} = \dfrac{28}{?}$~; \quad
$\dfrac{?}{9} = \dfrac{20}{36}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:2}
Simplifier autant que possible~:
$\dfrac{12}{16}$~; \quad $\dfrac{30}{45}$~; \quad $\dfrac{27}{36}$~;
\quad $\dfrac{48}{12}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:3}
$\dfrac{15}{35}$ et $\dfrac{21}{49}$ sont-elles égales~? Et
$\dfrac{16}{28}$ et $\dfrac{20}{36}$~? Justifier en simplifiant.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:4}
Comparer (écrire le raisonnement, pas seulement la réponse)~:
\[
\frac{7}{11} \text{ et } \frac{9}{11}; \qquad
\frac{5}{6} \text{ et } \frac{11}{18}; \qquad
\frac{13}{12} \text{ et } \frac{19}{20}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:5}
Calculer et simplifier si possible~:
\[
\frac{5}{9} + \frac{2}{9}, \qquad
\frac{11}{7} - \frac{4}{7}, \qquad
\frac{5}{8} + \frac{7}{8} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:6}
Calculer en utilisant \cref{met:g7:fractions:add}~:
\[
\frac{2}{3} + \frac{5}{6}, \qquad
\frac{7}{10} - \frac{2}{5}, \qquad
\frac{3}{4} + \frac{5}{12} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:7}
Calculer~:
\[
3 - \frac{5}{4}, \qquad
1 + \frac{3}{8}, \qquad
2 - \frac{7}{6} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:8}
Calculer~:
\[
5 \times \frac{3}{10}, \qquad
\frac{7}{8} \times 4, \qquad
\frac{2}{3} \text{ de } 45 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:fractions:9}
Marc a mangé $\frac14$ d'une pizza et Julie $\frac38$ de la même
pizza. Quelle \omterm{def:g6:fractions:def}{fraction} de la pizza ont-ils mangée ensemble~? Quelle
fraction reste-t-il~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:fractions:10}
Une bouteille contient $\frac34$ L de jus. Léa en verse $\frac16$ L
deux fois. Combien de jus reste-t-il~? (\omterm{def:g4:fractions:def}{Dénominateur} commun~: $12$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:fractions:11}
Expliquer, avec l'argument de \cref{ex:g7:fractions:trap}, pourquoi
$\frac35 + \frac12$ ne peut pas être $\frac{4}{7}$, sans calculer la
\omterm{def:g1:addition:def}{somme} correcte.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:fractions:12}
Calculer la \omterm{def:g1:addition:def}{somme}
\[
\frac12 + \frac14 + \frac18 + \frac{1}{16}
\]
étape par étape (de gauche à droite). Observer le motif des \omterm{def:g1:addition:def}{sommes}
partielles~: quel nombre approchent-elles lorsque l'on ajoute de plus
en plus de termes $\frac{1}{32}, \frac{1}{64}, \dots$~?
\end{exercise}
