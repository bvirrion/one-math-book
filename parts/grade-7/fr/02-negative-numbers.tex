\chapter{Nombres relatifs}\label{ch:g7:negatives}

Températures sous zéro, étages sous le sol, dettes~: bien des
grandeurs passent sous $0$. Les nombres négatifs prolongent la droite
numérique à gauche de zéro. Ce chapitre enseigne à les comparer et à
les additionner et soustraire --- leur multiplication attend le
\cref{ch:g8:negprod}.

\section{Nombres relatifs}

\begin{definition}[Nombres relatifs]\label{def:g7:negatives:def}
Un \emph{nombre relatif}\index{nombre relatif} est formé d'un signe
($+$ ou $-$) et d'une \emph{distance à zéro}~: $+3$ et $-3$ sont tous
deux à distance $3$ de $0$, de part et d'autre. Les nombres positifs
s'écrivent d'ordinaire sans leur signe~: $+3 = 3$. Les nombres $+3$
et $-3$ sont \emph{opposés}\index{opposé} l'un de l'autre.
\end{definition}

  \foreach \x in {-5,...,5} \draw (\x,-0.08) -- (\x,0.08);

\end{tikzpicture}

\end{omfigure}

\begin{method}[Comparer des nombres relatifs]\label{met:g7:negatives:compare}
Sur la droite numérique, plus petit signifie plus à gauche. En
pratique~:
\begin{enumerate}
  \item un nombre négatif est toujours plus petit qu'un positif~:
        $-7 < 2$~;
  \item entre deux positifs, l'ordre habituel~;
  \item entre deux négatifs, celui de plus \emph{grande} distance à
        zéro est le plus \emph{petit}~: $-7 < -2$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:negatives:compare}
Ranger du plus petit au plus grand~: $3$~; $-5$~; $0$~; $-1.5$~;
$2.8$. Les négatifs d'abord, le plus négatif en tête~:
\[
-5 < -1.5 < 0 < 2.8 < 3 .
\]
\end{example}

\section{Additionner des nombres relatifs}

Penser un nombre relatif comme un déplacement sur la droite~: $+5$
signifie «~avancer de $5$ pas à droite~», $-3$ signifie «~avancer de
$3$ pas à gauche~». Additionner des nombres, c'est enchaîner les
déplacements.

\begin{method}[Additionner deux nombres relatifs]\label{met:g7:negatives:add}
\begin{enumerate}
  \item \emph{Même signe}~: ajouter les distances, garder le signe
        commun~: $(-3) + (-4) = -7$.
  \item \emph{Signes contraires}~: soustraire la plus petite distance
        de la plus grande, prendre le signe de la plus grande~:
        $(-7) + (+2) = -5$ et $(+7) + (-2) = +5$.
  \item Les opposés s'annulent~: $(+4) + (-4) = 0$.
\end{enumerate}
\end{method}

\begin{tikzpicture}[scale=1.05]
  \draw[-Stealth] (-5.4,0) -- (4.4,0);
  \foreach \x in {-5,...,4} \draw (\x,-0.07) -- (\x,0.07);
  \foreach \x in {-5,-3,-1,0,2,4}
    \node[below, font=\small] at (\x,-0.09) {$\x$};
  \draw[-Stealth, omThm, very thick] (0,0.35) -- (-3.5,0.35);
  \draw[-Stealth, omDef, very thick] (-3.5,0.75) -- (-1.5,0.75);
  \node[omThm, font=\small] at (-1.75,0.18) {$-3.5$};
  \node[omDef, font=\small] at (-2.5,1.0) {$+2$};
  \fill (-1.5,0) circle (2pt) node[below=6pt, font=\small] {$-1.5$};
\end{tikzpicture}

{\small Additionner comme se déplacer~: partant de $0$, le
déplacement $-3.5$ puis le déplacement $+2$ aboutit à
$(-3.5) + 2 = -1.5$.}
\end{omfigure}

\begin{example}\label{ex:g7:negatives:add}
Étape par étape, en regroupant à convenance~:
\begin{align*}
(-8) + (+3) &= -5 && \text{(signes contraires~: } 8 - 3 = 5,
\text{ signe de } 8)\\
(-2.5) + (-1.5) &= -4 && \text{(même signe~: ajouter les distances)}\\
(+6) + (-9) + (+3) &= (-3) + (+3) = 0 .
\end{align*}
\end{example}

\section{Soustraire des nombres relatifs}

\begin{theorem}[Règle de soustraction]\label{thm:g7:negatives:sub}
Soustraire un nombre relatif revient à \emph{ajouter son opposé}~:
\[
a - b = a + (-b) .
\]
\end{theorem}

\begin{proof}[Pourquoi cela fonctionne]
$a - b$ est le nombre qui, ajouté à $b$, donne $a$. Vérifier que
$a + (-b)$ fait le travail~:
$\bigl(a + (-b)\bigr) + b = a + \bigl((-b) + b\bigr) = a + 0 = a$.
\end{proof}

\begin{example}\label{ex:g7:negatives:sub}
Toute soustraction devient une addition, puis
\cref{met:g7:negatives:add} s'applique~:
\begin{align*}
5 - 9 &= 5 + (-9) = -4, \\
(-3) - (+4) &= (-3) + (-4) = -7, \\
(-2) - (-6) &= (-2) + (+6) = +4 .
\end{align*}
La dernière ligne est la célèbre~: \emph{soustraire un négatif,
c'est ajouter}. Supprimer une dette vous enrichit~!
\end{example}

\begin{example}[Écarts de température]\label{ex:g7:negatives:temperature}
En une journée, la température est passée de $-6$ degrés à $+9$
degrés. La hausse est la différence
\[
9 - (-6) = 9 + 6 = 15 \text{ degrés}.
\]
Sur la droite numérique~: de $-6$ jusqu'à $0$ c'est $6$ degrés, de
$0$ à $9$ c'est $9$ de plus.
\end{example}

\section{Coordonnées dans tout le plan}

\begin{definition}[Coordonnées avec signes]\label{def:g7:negatives:coords}
Avec deux axes gradués perpendiculaires se croisant à l'origine $O$,
tout point du plan reçoit deux coordonnées relatives $(x, y)$~: un
$x$ négatif signifie à gauche de l'axe vertical, un $y$ négatif
signifie au-dessous de l'axe horizontal.
\end{definition}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g7:negatives:1}
Donner l'opposé de~: $7$~; \ $-3.5$~; \ $0$~; \ $+12$. Puis
compléter~: «~deux nombres sont opposés lorsque leur somme
vaut~?~».
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:2}
Recopier et compléter avec $<$ ou $>$~:
\[
-4 \;?\; 1, \qquad
-4 \;?\; -7, \qquad
-2.5 \;?\; -2.4, \qquad
0 \;?\; -0.1 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:3}
Ranger du plus petit au plus grand~: $-3$~; \ $2.5$~; \ $-0.5$~;
\ $-3.1$~; \ $0$~; \ $1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:4}
Calculer~:
\[
(-5) + (-9), \qquad
(-12) + (+7), \qquad
(+3.5) + (-1.5), \qquad
(-6) + (+6).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:5}
Transformer en additions, puis calculer~:
\[
4 - 11, \qquad
(-5) - (+3), \qquad
(-7) - (-10), \qquad
0 - (-8).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:6}
Calculer en regroupant astucieusement (les opposés d'abord)~:
\[
(+7) + (-12) + (+3) + (-8), \qquad
(-5.5) + (+9) + (-4.5) + (+1).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:7}
La température était de $-3$ degrés ce matin~; elle a monté de $8$
degrés dans la journée, puis baissé de $12$ degrés la nuit. Écrire un
seul calcul donnant la température finale, et le mener.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:negatives:8}
Placer dans un repère~: $A(2, 3)$~; $B(-3, 1)$~; $C(-2, -2)$~;
$D(3, -1)$~; $E(0, -3)$. Quel point est sur un axe~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:negatives:9}
Calculer l'écart entre la température la plus haute et la plus
basse~: Moscou en janvier, max $-4$ degrés, min $-13$ degrés~;
Helsinki, max $2$, min $-9$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:negatives:10}
Une plongeuse est à l'altitude $-12$ m (douze mètres sous la
surface). Elle remonte de $7$ m, puis descend de $4$ m. À quelle
altitude est-elle~? De combien doit-elle encore remonter pour
atteindre la surface~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:negatives:11}
Placer $A(-2, 1)$, $B(2, 1)$, $C(2, -2)$. Trouver les coordonnées du
point $D$ pour que $ABCD$ soit un rectangle, et calculer son
périmètre (compter les carreaux pour les longueurs des côtés).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:negatives:12}
Dans le carré magique ci-dessous, chaque ligne, colonne et diagonale
doit avoir la même somme. Recopier et le compléter~:
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|c|c|}
\hline
$?$ & $-7$ & $0$ \\
\hline
$-3$ & $-2$ & $?$ \\
\hline
$-4$ & $?$ & $?$ \\
\hline
\end{tabular}
\end{center}
(Commencer par trouver la somme magique à l'aide de la diagonale
presque complète.)
\end{exercise}
