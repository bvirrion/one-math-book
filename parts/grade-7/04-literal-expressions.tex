\chapter{Literal Expressions}\label{ch:g7:literal}

A letter in a computation stands for a number we do not know yet, or
for \emph{all} numbers at once: writing $P = 4c$ gives the perimeter
of every square in one formula. This chapter teaches the writing
conventions, how to substitute values, and how to simplify simple
expressions --- the start of algebra, continued in
\cref{ch:g8:equations}.

\section{Writing with letters}

\begin{definition}[Literal expression]\label{def:g7:literal:def}
A \emph{literal expression}\index{literal expression} is a computation
containing one or more letters, each standing for a number. Writing
conventions:
\begin{itemize}
  \item the sign $\times$ is dropped before a letter or a bracket:
        $3 \times a = 3a$, \ $a \times b = ab$, \
        $2 \times (x + 5) = 2(x + 5)$;
  \item but it stays between two numbers: $3 \times 5$ cannot be
        written $35$!
  \item $a \times a$ is written $a^2$ (``$a$ squared''),
        $a \times a \times a = a^3$;
  \item $1a$ is written $a$, and $0a$ is $0$.
\end{itemize}
\end{definition}

\begin{example}\label{ex:g7:literal:write}
Simplify the writing:
\[
5 \times x + 3 \times y = 5x + 3y,
\qquad
a \times 7 = 7a \ \text{(number first)},
\qquad
x \times x \times 4 = 4x^2 .
\]
\end{example}

\begin{example}[Formulas are literal expressions]\label{ex:g7:literal:formulas}
For a rectangle of length $L$ and width $w$:
perimeter $P = 2(L + w)$, area $A = Lw$. One formula summarizes the
infinitely many rectangles at once --- that is the whole point of
letters.
\end{example}

\section{Substituting}

\begin{method}[Evaluating an expression]\label{met:g7:literal:substitute}
To evaluate an expression for given values of its letters:
\begin{enumerate}
  \item rewrite the expression putting the hidden $\times$ signs back:
        $3a + a^2$ is $3 \times a + a \times a$;
  \item replace each letter by its value, keeping brackets around it
        if needed;
  \item compute, respecting the priorities of
        \cref{ch:g7:priorities}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:literal:substitute}
Evaluate $E = 3x + x^2$ for $x = 5$:
\[
E = 3 \times 5 + 5 \times 5 = 15 + 25 = 40 .
\]
Evaluate $F = 2(a + 3b)$ for $a = 4$ and $b = 0.5$:
\[
F = 2 \times (4 + 3 \times 0.5) = 2 \times (4 + 1.5) = 2 \times 5.5
= 11 .
\]
\end{example}

\section{Reducing expressions}

\begin{proposition}[Collecting like terms]\label{prop:g7:literal:reduce}
Terms containing the same letter part can be collected by adding their
number parts (this is distributivity,
\cref{thm:g7:priorities:distributivity}, read backwards):
\[
5x + 3x = (5 + 3)x = 8x,
\qquad
7a - 2a = 5a .
\]
Terms with different letter parts (like $3x$ and $5y$, or $x$ and
$x^2$) cannot be collected.
\end{proposition}

\begin{example}\label{ex:g7:literal:reduce}
Reduce $E = 4x + 7 + 3x + 2$, sorting the terms:
\[
E = (4x + 3x) + (7 + 2) = 7x + 9 .
\]
Careful: $7x + 9$ is \emph{not} $16x$! Test with $x = 1$:
$4 + 7 + 3 + 2 = 16$ and $7 \times 1 + 9 = 16$ agree, but $16x$ would
also be $16$ --- test with $x = 2$: the original gives
$8 + 7 + 6 + 2 = 23$, and $7 \times 2 + 9 = 23$, while
$16 \times 2 = 32$. The reduction $7x + 9$ is right, $16x$ is wrong.
\end{example}

\begin{method}[Testing an equality]\label{met:g7:literal:test}
To check whether two expressions might be equal for every value:
\begin{enumerate}
  \item substitute the same value in both, at least twice (e.g.\
        $x = 2$ and $x = 10$; avoid $0$ and $1$, they hide too many
        differences);
  \item if the results differ even once, the expressions are
        \emph{not} equal --- one counterexample kills a general claim;
  \item if the results agree, the equality is only \emph{plausible}: a
        proof needs algebra (expanding, reducing).
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:literal:test}
Is $2(x + 3)$ equal to $2x + 6$ for every $x$? Expanding proves it:
$2(x+3) = 2x + 2 \times 3 = 2x + 6$. Is $(x + 1)^2$ equal to
$x^2 + 1$? Test $x = 2$: $(2+1)^2 = 9$ but $2^2 + 1 = 5$. No ---
and one counterexample is a complete proof of ``no''.
\end{example}

\section{Producing expressions}

\begin{example}[From situation to expression]\label{ex:g7:literal:produce}
A taxi charges $4$ euros plus $2$ euros per kilometer. For a trip of
$k$ kilometers the price is
\[
p = 4 + 2k \quad\text{euros}.
\]
For $k = 7$: $p = 4 + 14 = 18$ euros. The expression \emph{is} the
general answer; substituting produces every particular one.
\end{example}

\begin{example}[Number tricks]\label{ex:g7:literal:trick}
``Think of a number, double it, add $10$, divide by $2$, subtract your
number.'' Call the number $n$ and follow the steps:
\[
n \ \to\ 2n \ \to\ 2n + 10 \ \to\ \frac{2n + 10}{2} = n + 5 \ \to\
n + 5 - n = 5 .
\]
Everyone gets $5$, whatever $n$ was --- the letters explain the
magic.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g7:literal:1}
Simplify the writing:
\[
7 \times x, \qquad
y \times 3, \qquad
a \times b \times 5, \qquad
x \times x, \qquad
2 \times (a + 4).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:literal:2}
Put the $\times$ signs back:
\[
5ab, \qquad
3(x + 2), \qquad
x^2 + 4x, \qquad
2a^3 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:literal:3}
Evaluate for $x = 3$: \ $5x + 1$; \quad $x^2$; \quad $2x^2 - x$;
\quad $10 - 2x$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:literal:4}
Evaluate $3a + 2b$ and $a b + 5$ for $a = 4$, $b = 1.5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:literal:5}
Reduce:
\[
8x + 5x, \qquad
9a - 4a + a, \qquad
6y + 2 + 3y + 7, \qquad
5t + 3 - 2t - 3 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:literal:6}
Give a formula for: the perimeter of a square of side $c$; the
perimeter of an isosceles triangle with base $b$ and equal sides $s$;
the price of $n$ breads at $1.20$ euros each.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:literal:7}
Match each expression to a description: $2n$; $n + 2$; $n^2$;
$\frac n2$. Descriptions: ``the half of $n$''; ``the double of $n$'';
``the square of $n$''; ``two more than $n$''.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:literal:8}
Using \cref{met:g7:literal:test}, decide (with proof or
counterexample) whether these equalities hold for every $x$:
\[
3(x + 4) = 3x + 12; \qquad
2x + 5 = 7x; \qquad
x + x = x^2 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:literal:9}
A rectangle has width $w$ and its length is $3$ cm more than its
width.
\begin{enumerate}
  \item Express the length, then the perimeter, in terms of $w$, and
        reduce.
  \item Compute the perimeter for $w = 5$ cm, directly and with your
        reduced formula. Same answer?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:literal:10}
Follow the trick of \cref{ex:g7:literal:trick} with letters: ``think
of a number, add $6$, multiply by $3$, subtract $18$, divide by your
starting number'' (assume it is not zero). What does everyone get?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:literal:11}
The figure below is a square of side $a$ with a square corner of side
$b$ removed ($b$ smaller than $a$). Write two different expressions
for its area (one as a difference, one by cutting the L-shape into two
rectangles), and check they agree for $a = 5$, $b = 2$.
\end{exercise}
