\chapter{Fractions: Comparing and Adding}\label{ch:g7:fractions}

Grade 6 introduced \omterm{def:g6:fractions:def}{fractions} as shares and as exact \omterm{def:g3:division:remainder}{quotients}
(\cref{ch:g6:fractions}). This chapter learns to \emph{compute} with
them: recognizing equal \omterm{def:g6:fractions:def}{fractions}, comparing, adding and subtracting
--- for now with friendly \omterm{def:g4:fractions:def}{denominators}; the general case comes in
\cref{ch:g9:fractions}.

\section{Equal fractions}

\begin{theorem}[Equal fractions]\label{thm:g7:fractions:equal}
A \omterm{def:g6:fractions:def}{fraction} is unchanged when its \omterm{def:g4:fractions:def}{numerator} and \omterm{def:g4:fractions:def}{denominator} are
multiplied, or divided, by the same nonzero number:
\[
\frac{a}{b} = \frac{a \times k}{b \times k} .
\]
\end{theorem}
\admitted

\begin{example}\label{ex:g7:fractions:equal}
$\dfrac{3}{4} = \dfrac{3 \times 5}{4 \times 5} = \dfrac{15}{20}$, and
in the other direction $\dfrac{42}{30} = \dfrac{42 \div 6}{30 \div 6}
= \dfrac{7}{5}$ (simplified by $6$).

To test whether $\dfrac{8}{12}$ and $\dfrac{10}{15}$ are equal,
simplify both: $\dfrac{8}{12} = \dfrac{2}{3}$ and
$\dfrac{10}{15} = \dfrac{2}{3}$ --- yes, equal.
\end{example}

\section{Comparing fractions}

\begin{method}[Comparing two fractions]\label{met:g7:fractions:compare}
\begin{enumerate}
  \item \emph{Same \omterm{def:g4:fractions:def}{denominator}}: compare the \omterm{def:g4:fractions:def}{numerators}:
        $\frac{5}{9} < \frac{7}{9}$.
  \item \emph{One \omterm{def:g4:fractions:def}{denominator} is a \omterm{def:g5:division:multiple}{multiple} of the other}: rewrite the
        coarser \omterm{def:g6:fractions:def}{fraction}, then compare:
        $\frac{5}{6}$ vs $\frac{7}{12}$: $\frac56 = \frac{10}{12} >
        \frac{7}{12}$.
  \item \emph{Compare with $1$ or $\frac12$} when possible:
        $\frac{9}{8} > 1 > \frac{7}{9}$ settles
        $\frac98$ vs $\frac79$ instantly.
\end{enumerate}
\end{method}

\section{Adding and subtracting}

\begin{theorem}[Same denominator]\label{thm:g7:fractions:add}
\omterm{def:g6:fractions:def}{Fractions} with the same \omterm{def:g4:fractions:def}{denominator} are added (or subtracted) by
adding (or subtracting) the \omterm{def:g4:fractions:def}{numerators}:
\[
\frac{a}{d} + \frac{b}{d} = \frac{a + b}{d},
\qquad
\frac{a}{d} - \frac{b}{d} = \frac{a - b}{d} .
\]
\end{theorem}

\begin{proof}[Why]
$a$ parts of size $\frac1d$ plus $b$ parts of the same size make
$a + b$ parts of that size.
\end{proof}

\begin{method}[Different denominators]\label{met:g7:fractions:add}
When one \omterm{def:g4:fractions:def}{denominator} is a \omterm{def:g5:division:multiple}{multiple} of the other:
\begin{enumerate}
  \item rewrite the \omterm{def:g6:fractions:def}{fraction} with the smaller \omterm{def:g4:fractions:def}{denominator} so both have
        the larger one (\cref{thm:g7:fractions:equal});
  \item add or subtract the \omterm{def:g4:fractions:def}{numerators};
  \item simplify the result if possible.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g7:fractions:add}
Compute $\dfrac{3}{4} + \dfrac{5}{8}$, step by step:
\begin{enumerate}
  \item $8$ is a \omterm{def:g5:division:multiple}{multiple} of $4$: rewrite
        $\dfrac34 = \dfrac{3 \times 2}{4 \times 2} = \dfrac{6}{8}$;
  \item add: $\dfrac{6}{8} + \dfrac{5}{8} = \dfrac{11}{8}$;
  \item nothing to simplify: the result is $\dfrac{11}{8}$, i.e.\
        $1 + \dfrac38$.
\end{enumerate}
Another: $2 - \dfrac{4}{5} = \dfrac{10}{5} - \dfrac{4}{5} =
\dfrac{6}{5}$ (write the whole number over the \omterm{def:g4:fractions:def}{denominator} $5$
first).
\end{example}

\begin{example}[A classic trap]\label{ex:g7:fractions:trap}
$\dfrac12 + \dfrac13$ is \emph{not} $\dfrac{2}{5}$! Adding \omterm{def:g4:fractions:def}{numerators}
and \omterm{def:g4:fractions:def}{denominators} separately is wrong --- the answer $\frac25$ would be
smaller than $\frac12$, absurd when \emph{adding} something positive
to $\frac12$. (The correct \omterm{def:g1:addition:def}{sum}, $\frac56$, needs a common \omterm{def:g4:fractions:def}{denominator}
$6$; the general method is in \cref{ch:g9:fractions}.)
\end{example}

\section{Multiplying a fraction by a number}

\begin{proposition}[Fraction times a number]\label{prop:g7:fractions:mult}
For a number $k$:
\[
k \times \frac{a}{b} = \frac{k \times a}{b} .
\]
Taking $\frac ab$ \emph{of} a quantity means multiplying by
$\frac ab$.
\end{proposition}
\admitted

\begin{example}\label{ex:g7:fractions:mult}
$6 \times \dfrac{2}{3} = \dfrac{12}{3} = 4$: six times two thirds is
four wholes. And ``$\frac34$ of $60$ euros'' is
$\frac34 \times 60 = \frac{180}{4} = 45$ euros --- the same answer as
dividing by $4$ then multiplying by $3$
(\cref{met:g6:fractions:ofquantity}).
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g7:fractions:1}
Complete: $\dfrac{2}{5} = \dfrac{?}{20}$; \quad
$\dfrac{18}{24} = \dfrac{3}{?}$; \quad
$\dfrac{7}{3} = \dfrac{28}{?}$; \quad
$\dfrac{?}{9} = \dfrac{20}{36}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:2}
Simplify as much as possible:
$\dfrac{12}{16}$; \quad $\dfrac{30}{45}$; \quad $\dfrac{27}{36}$;
\quad $\dfrac{48}{12}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:3}
Are $\dfrac{15}{35}$ and $\dfrac{21}{49}$ equal? And $\dfrac{16}{28}$
and $\dfrac{20}{36}$? Justify by simplifying.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:4}
Compare (write the reasoning, not just the answer):
\[
\frac{7}{11} \text{ and } \frac{9}{11}; \qquad
\frac{5}{6} \text{ and } \frac{11}{18}; \qquad
\frac{13}{12} \text{ and } \frac{19}{20}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:5}
Compute and simplify if possible:
\[
\frac{5}{9} + \frac{2}{9}, \qquad
\frac{11}{7} - \frac{4}{7}, \qquad
\frac{5}{8} + \frac{7}{8} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:6}
Compute using \cref{met:g7:fractions:add}:
\[
\frac{2}{3} + \frac{5}{6}, \qquad
\frac{7}{10} - \frac{2}{5}, \qquad
\frac{3}{4} + \frac{5}{12} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:7}
Compute:
\[
3 - \frac{5}{4}, \qquad
1 + \frac{3}{8}, \qquad
2 - \frac{7}{6} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g7:fractions:8}
Compute:
\[
5 \times \frac{3}{10}, \qquad
\frac{7}{8} \times 4, \qquad
\frac{2}{3} \text{ of } 45 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:fractions:9}
Marc ate $\frac14$ of a pizza and Julie $\frac38$ of the same pizza.
What \omterm{def:g6:fractions:def}{fraction} of the pizza did they eat together? What \omterm{def:g6:fractions:def}{fraction} is
left?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:fractions:10}
A bottle contains $\frac34$ L of juice. Lea pours out $\frac16$ L
twice. How much juice remains? (Common \omterm{def:g4:fractions:def}{denominator}: $12$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g7:fractions:11}
Explain, with the argument of \cref{ex:g7:fractions:trap}, why
$\frac35 + \frac12$ cannot be $\frac{4}{7}$, without computing the
correct \omterm{def:g1:addition:def}{sum}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g7:fractions:12}
Compute the \omterm{def:g1:addition:def}{sum}
\[
\frac12 + \frac14 + \frac18 + \frac{1}{16}
\]
step by step (left to right). Observe the pattern of the partial \omterm{def:g1:addition:def}{sums}:
what number do they approach as more and more terms
$\frac{1}{32}, \frac{1}{64}, \dots$ are added?
\end{exercise}

\section{Problem: The fractions hidden in a bar of music}

\begin{problem}[{Weekend problem --- note values are fractions
that must add up exactly, rhythm counting hides a famous sequence,
and parity rules the Balkan beat}]
\label{pb:g7:fractions:1}
Music is written in \omterm{def:g6:fractions:def}{fractions}. A \emph{whole note} lasts $1$
measure (in the common ``four-four'' time); a \emph{\omterm{def:g3:division:half}{half} note}
lasts $\frac12$; then come the \emph{\omterm{def:g3:division:half}{quarter} note} $\frac14$, the
\emph{eighth note} $\frac18$ and the \emph{sixteenth note}
$\frac{1}{16}$. One inviolable law: \emph{the durations inside a
measure must add up to exactly the measure's total} --- $1$ in
four-four time. Every question below is this chapter's arithmetic
(\cref{thm:g7:fractions:add}, \cref{met:g7:fractions:add}) set to
music; on the way you will meet a counting sequence discovered by
Indian scholars a thousand years ago.

\medskip
\noindent\textbf{Part I --- Notes that must add up.}
\begin{enumerate}
  \item Check that each of these measures is legal in four-four
        time: (a) \omterm{def:g3:division:half}{half} $+$ \omterm{def:g3:division:half}{quarter} $+$ \omterm{def:g3:division:half}{quarter}; (b) \omterm{def:g3:division:half}{quarter} $+$
        eighth $+$ eighth $+$ \omterm{def:g3:division:half}{half}.
  \item A copyist wrote: \omterm{def:g3:division:half}{half} $+$ \omterm{def:g3:division:half}{quarter} $+$ eighth. What
        \omterm{def:g6:fractions:def}{fraction} of the measure is written, and which single
        note completes it?
  \item How many sixteenth notes fill one whole measure? How
        many sixteenths make one eighth note?
  \item A \emph{dot} after a note adds \omterm{def:g3:division:half}{half} the note's value: a
        dotted \omterm{def:g3:division:half}{half} is $\frac12 + \frac14$, a dotted \omterm{def:g3:division:half}{quarter} is
        $\frac14 + \frac18$. Compute both values.
  \item A waltz is written in three-four time: each measure must
        total $\frac34$. Check that a single dotted \omterm{def:g3:division:half}{half} fills a
        waltz measure, and write down two other legal waltz
        measures using only \omterm{def:g3:division:half}{quarters} and eighths.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The copyist's workshop.}
\begin{enumerate}[resume]
  \item A four-four measure contains: dotted \omterm{def:g3:division:half}{quarter} $+$ eighth
        $+$ \omterm{def:g3:division:half}{quarter}. What is missing?
  \item Which lasts longer, a dotted eighth or a \omterm{def:g3:division:half}{quarter} note?
        Compare via sixteenths.
  \item A \emph{tie} glues durations together into one long
        sound: a \omterm{def:g3:division:half}{half} note tied to an eighth lasts how long? How
        many sixteenths is that?
  \item The notes of a measure add up to $\frac{11}{16}$; the
        \omterm{def:g3:division:remainder}{remainder} is silence (a \emph{rest}). How long is the
        rest?
  \item A drummer fills one \omterm{def:g3:division:half}{quarter} note ($\frac{4}{16}$ of the
        measure) using only eighths ($\frac{2}{16}$) and
        sixteenths ($\frac{1}{16}$), order mattering: for
        instance eighth--sixteenth--sixteenth, or
        sixteenth--eighth--sixteenth. Check that both examples
        are legal, then list \emph{all} the possible rhythms.
        How many are there?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- A thousand-year-old sequence, and
an \omterm{def:g3:numbers:evenodd}{odd} measure.}
\begin{enumerate}[resume]
  \item Let us count the drummer's rhythms for longer notes.
        Every rhythm ends either with a sixteenth or with an
        eighth; explain why this gives the rule: (rhythms
        filling $n$ sixteenths) $=$ (rhythms filling $n - 1$)
        $+$ (rhythms filling $n - 2$). Starting from $1$ rhythm
        for one sixteenth and $2$ for two, build the table up to
        $n = 8$: how many rhythms fill a \omterm{def:g3:division:half}{half} note?
  \item The numbers you found --- $1, 2, 3, 5, 8, 13, 21, 34$
        --- were published by the Indian scholar Hemachandra
        around the year 1150, counting exactly such rhythms, two
        generations before Fibonacci wrote them down in Italy.
        State their defining pattern in one sentence, and check
        it on the last three entries of your table.
  \item ``Swing'' style replaces two equal eighths by a
        long--short pair: $\frac16$ then $\frac{1}{12}$. Verify
        that the swap is legal (same total duration), using the
        common \omterm{def:g4:fractions:def}{denominator} $12$.
  \item A \emph{triplet} squeezes three equal notes into one
        \omterm{def:g3:division:half}{quarter} note. How long is each note (check your answer
        by multiplying it by $3$,
        \cref{prop:g7:fractions:mult})? Same question for three
        equal notes filling a \omterm{def:g3:division:half}{half} note.
  \item Balkan dance music uses seven-eight time: each measure
        totals $\frac78$. Verify that dotted \omterm{def:g3:division:half}{quarter} $+$ \omterm{def:g3:division:half}{quarter}
        $+$ \omterm{def:g3:division:half}{quarter} fills it. Then explain why \emph{no}
        combination of \omterm{def:g3:division:half}{half} notes and \omterm{def:g3:division:half}{quarter} notes alone,
        however many, can ever fill a seven-eight measure.
        (Count in eighths and think about even and \omterm{def:g3:numbers:evenodd}{odd}.)
\end{enumerate}
\end{problem}
