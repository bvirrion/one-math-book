\section{शास्त्रीय ठोस और उनके आयतन}

\begin{theorem}[आयतन सूत्र]\label{thm:g9:solids:volumes}
लिखना $ B $ के लिए \omterm{def:g6:measure:area}{क्षेत्र} आधार का और $ h $ ऊंचाई के लिए (दूरी)
आधार और विपरीत के बीच चेहरा या शीर्ष)।
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{l|l}
\omterm{def:g5:solids:def}{ठोस} & \omterm{def:g6:measure:volume}{आयतन} \\
\hline
\omterm{def:g7:areas:prism}{चश्मे}, \omterm{def:g7:areas:prism}{सिलेंडर} & $ V = B \times h $ \\[5pt]
\omterm{def:g8:solids:def}{पिरामिड}, \omterm{def:g8:solids:def}{कोन} & $ V = \dfrac13\, B \times h $ \\[5pt]
sphere of \omterm{def:g3:shapes:shapes}{RADIUS} $ r $ & $ V = \dfrac43\,\pi r^3$ \\
\end{tabular}
\end{center}
एक के लिए \omterm{def:g7:areas:prism}{सिलेंडर} और ए \omterm{def:g8:solids:def}{कोन} का \omterm{def:g3:shapes:shapes}{RADIUS} $ r $, आधार \omterm{def:g6:measure:area}{क्षेत्र} है $ B = \pi r^2$;
\omterm{def:g4:numbers:round}{गोल}े की सतह \omterm{def:g6:measure:area}{क्षेत्र} है $4\pi r^2$.
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:volumes}
एक बेलनाकार कैन है \omterm{def:g3:shapes:shapes}{RADIUS} $4$ सेमी और ऊंचाई $10$ सेमी:
\[
V = \pi r^2 h = \pi \times 16 \times 10 = 160\pi \approx 503 \text{ cm}^3
\approx 0.5 \text{ L}.
\]
ए \omterm{def:g8:solids:def}{कोन} समान आधार और ऊंचाई के साथ इसका एक तिहाई हिस्सा होता है:
$\frac{160\pi}{3} \approx 168$ सेमी $^3$. का एक \omterm{def:g4:numbers:round}{गोल}ा \omterm{def:g3:shapes:shapes}{RADIUS} $4$ सेमी:
$ V = \frac43 \pi \times 64 = \frac{256\pi}{3} \approx 268$ सेमी $^3$.
\end{example}

\begin{example}[एक पिरामिड कदम दर कदम]\label{ex:g9:solids:pyramid}
ए \omterm{def:g8:solids:def}{पिरामिड} भुजा का आधार वर्गाकार है $6$ मी और ऊंचाई $5$ एम।
\begin{enumerate}
  \item आधार \omterm{def:g6:measure:area}{क्षेत्र}: $ B = 6^2 = 36$ एम $^2$.
  \item \omterm{def:g6:measure:volume}{आयतन}: $ V = \frac13 \times 36 \times 5 = 60$ एम $^3$.
\end{enumerate}
इकाइयों पर ध्यान दें: यदि भुजा सेमी में और ऊंचाई मी में दी गई है, तो इनमें से एक
उन्हें पहले परिवर्तित करना होगा।
\end{example}

\section{विमानों द्वारा अनुभाग}

\begin{proposition}[शास्त्रीय ठोसों के अनुभाग]\label{prop:g9:solids:sections}
ए काटना \omterm{def:g5:solids:def}{ठोस} एक समतल द्वारा एक सपाट आकृति बनती है, इसका
\emph{क्रॉस सेक्शन}\index{क्रॉस सेक्शन}:
\begin{itemize}
  \item ए \omterm{def:g7:areas:prism}{चश्मे} या \omterm{def:g7:areas:prism}{सिलेंडर} काटना \emph{\omterm{def:g6:lines:perp}{समानांतर} इसके आधार पर} एक प्रति देता है
        आधार का, हर ऊंचाई पर;
  \item ए \omterm{def:g8:solids:def}{पिरामिड} या \omterm{def:g8:solids:def}{कोन} काटना \omterm{def:g6:lines:perp}{समानांतर} इसके आधार को एक देता है
        \emph{कमी} आधार का: दूरी पर $ d $ शीर्ष से,
        द \omterm{def:g9:thales:scaling}{पैमाने का कारक} है $ k = \frac{d}{h}$;
  \item का एक \omterm{def:g4:numbers:round}{गोल}ा \omterm{def:g3:shapes:shapes}{RADIUS} $ r $ की दूरी पर एक विमान द्वारा काटा गया $ d < r $ से
        केंद्र का एक वृत्त देता है \omterm{def:g3:shapes:shapes}{RADIUS} $\sqrt{r^2 - d^2}$ (द्वारा
        पाइथागोरस प्रमेय)।
\end{itemize}
\end{proposition}
\admitted

\begin{example}\label{ex:g9:solids:section}
ए \omterm{def:g8:solids:def}{कोन} आधार है \omterm{def:g3:shapes:shapes}{RADIUS} $6$ और ऊंचाई $9$. एक विमान द्वारा अनुभाग
\omterm{def:g6:lines:perp}{समानांतर} आधार तक, दूरी पर $3$ शीर्ष से, एक डिस्क स्केल्ड है
$ k = \frac39 = \frac13$: इसका \omterm{def:g3:shapes:shapes}{RADIUS} है $6 \times \frac13 = 2$.

का एक \omterm{def:g4:numbers:round}{गोल}ा \omterm{def:g3:shapes:shapes}{RADIUS} $5$ दूरी पर एक विमान द्वारा काटा जाता है $3$ इसके से
केंद्र: अनुभाग एक वृत्त है \omterm{def:g3:shapes:shapes}{RADIUS} $\sqrt{25 - 9} = 4$.
\end{example}

\section{ठोस पदार्थों को स्केल करना}

\begin{theorem}[लंबाई, क्षेत्रफल, आयतन पर स्केलिंग का प्रभाव]\label{thm:g9:solids:scaling}
जब ए \omterm{def:g5:solids:def}{ठोस} द्वारा बढ़ाया या घटाया जाता है \omterm{def:g9:thales:scaling}{पैमाने का कारक} $ k > 0$:
\begin{itemize}
  \item सभी लंबाई को गुणा किया जाता है $ k $;
  \item सभी \omterm{def:g6:measure:area}{क्षेत्र}ों से गुणा किया जाता है $ k^2$;
  \item सभी संस्करणों से गुणा किया जाता है $ k^3$.
\end{itemize}
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:scaling}
एक मॉडल कार \omterm{def:g7:prop:scale}{पैमाना} $\frac{1}{18}$: लंबाई असली कार जितनी है
से गुणा किया गया $ k = \frac{1}{18}$, चित्रित सतह को गुणा किया गया
$ k^2 = \frac{1}{324}$, और \omterm{def:g6:measure:volume}{आयतन} द्वारा $ k^3 = \frac{1}{5832}$.

दोगुना करना \omterm{def:g3:shapes:shapes}{RADIUS} एक \omterm{def:g4:numbers:round}{गोल}े का ($ k = 2$) इसे गुणा करता है \omterm{def:g6:measure:volume}{आयतन} द्वारा
$2^3 = 8$: सूत्र पर जाँच करें,
$\frac43\pi (2r)^3 = \frac43 \pi \times 8r^3$.
\end{example}

\begin{example}[कटा हुआ शंकु]\label{ex:g9:solids:truncated}
ए \omterm{def:g8:solids:def}{कोन} ऊंचाई का $9$ और आधार \omterm{def:g3:shapes:shapes}{RADIUS} $6$ (\omterm{def:g6:measure:volume}{आयतन}
$\frac13 \pi \times 36 \times 9 = 108\pi $) की दूरी पर काटा जाता है $3$ से
शीर्ष, और छोटा \omterm{def:g8:solids:def}{कोन} ऊपर से कट हटा दिया गया है. छोटा \omterm{def:g8:solids:def}{कोन} है
द्वारा कमी $ k = \frac13$, तो यह है \omterm{def:g6:measure:volume}{आयतन} है
$108\pi \times \left(\frac13\right)^3 = 4\pi $. \omterm{def:g3:division:remainder}{शेष} \omterm{def:g5:solids:def}{ठोस} (ए
\emph{छंटनी की गई \omterm{def:g8:solids:def}{कोन}}) है \omterm{def:g6:measure:volume}{आयतन} $108\pi - 4\pi = 104\pi \approx 327$.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g9:solids:1}
की गणना करें संस्करणों: ए \omterm{def:g5:solids:def}{डिब्बा} (आयताकार \omterm{def:g7:areas:prism}{चश्मे}) आयामों का $4 \times 5

\begin{exercise}[$\star $]\label{exo:g9:solids:2}

\begin{exercise}[$\star $]\label{exo:g9:solids:3}

\begin{exercise}[$\star $]\label{exo:g9:solids:4}

\begin{exercise}[$\star\star $]\label{exo:g9:solids:5}
ए \omterm{def:g8:solids:def}{कोन} आधार है \omterm{def:g3:shapes:shapes}{RADIUS} $8$ और ऊंचाई $12$. एक हवाई जहाज \omterm{def:g6:lines:perp}{समानांतर} आधार तक
की दूरी पर इसे काटता है $9$ शीर्ष से. की गणना करें \omterm{def:g3:shapes:shapes}{RADIUS} का
अनुभाग, फिर \omterm{def:g6:measure:volume}{आयतन} छोटे का \omterm{def:g8:solids:def}{कोन} कट के ऊपर.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:solids:6}
भीतरी भाग का एक बेलनाकार शीशा \omterm{def:g3:shapes:shapes}{RADIUS} $3$ सेमी की ऊंचाई तक पानी होता है

आधार \omterm{def:g6:measure:area}{क्षेत्र}; फिर, सटीक वृद्धि दें \omterm{def:g4:numbers:round}{गोल} मिली\omterm{def:g4:fractions:def}{मीटर} तक.)

\begin{exercise}[$\star\star $]\label{exo:g9:solids:7}

\begin{exercise}[$\star\star\star $]\label{exo:g9:solids:9}
का एक शंकु के आकार का कीप \omterm{def:g3:shapes:shapes}{RADIUS} $6$ सेमी और ऊंचाई $12$ सेमी, शीर्ष नीचे, है
तक पानी से भर गया \omterm{def:g3:division:half}{आधा} उसके जैसा \emph{ऊंचाई} (से मापा गया
शीर्ष)।
\begin{enumerate}
  \item क्या \omterm{def:g9:fractions:fraction}{अंश} फ़नल का \omterm{def:g6:measure:volume}{आयतन} भरा है?
  \item यदि इसके स्थान पर यह भरा हुआ है \omterm{def:g3:division:half}{आधा} उसके जैसा \emph{\omterm{def:g6:measure:volume}{आयतन}} पानी का,
        दिखाओ कि पानी की ऊंचाई $ d $ संतुष्ट
        $\left(\frac{d}{12}\right)^3 = \frac12$, और दें $ d $ को
        मिली\omterm{def:g4:fractions:def}{मीटर} ($\sqrt[3]{0.5} \approx 0.7937$).
\end{enumerate}
\end{exercise}

\section{समस्या: आर्किमिडीज़ की समाधि का पत्थर}

\begin{problem}[{सप्ताहांत समस्या---गोला दो तिहाई है इसका सिलेंडर (दो बार), 1:2:3 स्टैक, और वर्ग-घन कानून जो दिग्गजों को मना करता है}]
\label{pb:g9:solids:1}
आर्किमिडीज़ ने कई प्रमेय सिद्ध किये, लेकिन एक ने उन्हें इतना गौरवान्वित कर दिया
उसने इसके लिए पूछा \emph{आकृति} उसकी कब्र पर उकेरा जाने वाला: एक \omterm{def:g4:numbers:round}{गोल}ा
इसके सबसे सघनतम में बसा हुआ \omterm{def:g7:areas:prism}{सिलेंडर}. एक सदी बाद रोमन लेखक
सिसरो ने सिरैक्यूज़ के पास ब्रैम्बल्स की खोज करते हुए इसे पहचान लिया
कब्र ``\omterm{def:g4:numbers:round}{गोल}े द्वारा और \omterm{def:g7:areas:prism}{सिलेंडर}''. यह समस्या पुनः प्राप्त हो जाती है
नक्काशी क्या कूटबद्ध करती है --- एक दोहरा दो-तिहाई चमत्कार --- फिर
अनुसरण करता है संस्करणों और सतहें (\cref{thm:g9:solids:volumes},
\cref{thm:g9:solids:scaling}) एक ऐसे कानून के लिए जो दिग्गजों, चींटियों को नियंत्रित करता है
और ग्रहों को ठंडा करना।

\medskip
\noindent\textbf{Part I --- The carving decoded.}
का एक \omterm{def:g4:numbers:round}{गोल}ा \omterm{def:g3:shapes:shapes}{RADIUS} $ r $ ठीक a के अंदर बैठता है \omterm{def:g7:areas:prism}{सिलेंडर}: वही
\omterm{def:g3:shapes:shapes}{RADIUS}, ऊंचाई $2r $.
\begin{enumerate}
  \item की गणना करें \omterm{def:g7:areas:prism}{सिलेंडर}'एस \omterm{def:g6:measure:volume}{आयतन} के अनुसार $ r $.
  \item \omterm{def:g4:numbers:round}{गोल}े के अनुपात की गणना करें \omterm{def:g6:measure:volume}{आयतन} तक
        \omterm{def:g7:areas:prism}{सिलेंडर}'एस। आर्किमिडीज़ को यह क्यों पसंद आया होगा
        उत्तर में नहीं है $\pi $ और नहीं $ r $?
  \item अब सतहें: \omterm{def:g4:numbers:round}{गोल}े की सतह की तुलना करें \omterm{def:g6:measure:area}{क्षेत्र} साथ \omterm{def:g7:areas:prism}{सिलेंडर}'एस \emph{कुल} सतह (पार्श्व भाग प्लस
        दो ढक्कन)। कौन सा अनुपात पुनः प्रकट होता है ---?
  \item \omterm{def:g4:numbers:round}{गोल}े और के बीच की खाली जगह \omterm{def:g7:areas:prism}{सिलेंडर} है \omterm{def:g6:measure:volume}{आयतन}
        $2\pi r^3 - \frac43\pi r^3$. दिखाओ कि यह बचा हुआ है
        बिल्कुल बराबर है \omterm{def:g6:measure:volume}{आयतन} का \emph{दो \omterm{def:g8:solids:def}{कोन}} का \omterm{def:g3:shapes:shapes}{RADIUS}
        $ r $ और ऊंचाई $ r $.
  \item का एक बास्केटबॉल \omterm{def:g3:shapes:shapes}{RADIUS} $12$ सेमी सबसे सस्ते में बिकता है
        बेलनाकार \omterm{def:g5:solids:def}{डिब्बा}. दोनों की गणना करें संस्करणों (निकटतम तक
        $100$ सेमी $^3$) और का प्रतिशत \omterm{def:g5:solids:def}{डिब्बा} वह है
        खाली.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Bowls, molds and moons.}
\begin{enumerate}[resume]
  \item एक अर्धगोलाकार कटोरा है \omterm{def:g3:shapes:shapes}{RADIUS} $10$ सेमी। इसकी गणना करें
        \omterm{def:g2:measure:units}{क्षमता}, सेमी में $^3$ और लीटर में (डेसीलीटर तक)।
  \item 1:2:3 स्टैक: ए \omterm{def:g8:solids:def}{कोन}, एक \omterm{def:g4:numbers:round}{गोल}ार्ध और एक \omterm{def:g7:areas:prism}{सिलेंडर},
        के सभी \omterm{def:g3:shapes:shapes}{RADIUS} $10$ सेमी और ऊंचाई $10$ सेमी। की गणना करें
        तीन संस्करणों और पौराणिक अनुपात को सत्यापित करें
        $1 : 2 : 3$. (आर्किमिडीज़ ने इसकी सराहना की होगी
        एक भी.)
  \item का एक चॉकलेट \omterm{def:g4:numbers:round}{गोल}ा \omterm{def:g3:shapes:shapes}{RADIUS} $3$ सेमी को एक में पिघलाया जाता है
        का बेलनाकार साँचा \omterm{def:g3:shapes:shapes}{RADIUS} $3$ सेमी। कितनी ऊंचाई होती है
        चॉकलेट पहुँची? (एकदम सही \omterm{def:g9:fractions:fraction}{अंश}, फिर को
        मिली\omterm{def:g4:fractions:def}{मीटर}.)
  \item पृथ्वी का \omterm{def:g3:shapes:shapes}{RADIUS} के बारे में है $6\,371$ किमी, चंद्रमा का
        के बारे में $1\,737$ किमी. फिर, त्रि\omterm{def:g9:trig:ratios}{ज्या} के अनुपात की गणना करें
        ---साथ \cref{thm:g9:solids:scaling} --- का अनुपात
        द संस्करणों: कितने चंद्रमा एक खोखले में समा जाएंगे
        पृथ्वी, द्वारा \omterm{def:g6:measure:volume}{आयतन}?
  \item उसी प्रमेय से: पृथ्वी-चंद्रमा का अनुपात क्या है
        \emph{सतह \omterm{def:g6:measure:area}{क्षेत्र}ों}? और सामान्य तौर पर, जब एक गुब्बारा होता है
        \omterm{def:g3:shapes:shapes}{RADIUS} दोगुना हो जाता है, उसका क्या होता है \omterm{def:g6:measure:volume}{आयतन} और इसके लिए
        सतह? कहें \emph{वर्ग-घन नियम}: एक आकृति के रूप में
        तराजू ऊपर, संस्करणों सतहों से आगे निकल जाना।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The square--cube law rules the
world.}
\begin{enumerate}[resume]
  \item दिग्गजों के विरुद्ध गैलीलियो का तर्क: एक मानव की कल्पना करें
        एक कारक द्वारा बढ़ाया गया $10$, समान अनुपात। किसलिए
        कारक वजन बढ़ता है (वजन इस प्रकार है)। \omterm{def:g6:measure:volume}{आयतन})? द्वारा
        कौन सा कारक करता है \omterm{prop:g9:solids:sections}{क्रॉस सेक्शन} हड्डियाँ बढ़ती हैं
        (ए.एन \omterm{def:g6:measure:area}{क्षेत्र})? तो फिर, किस कारक से \emph{दबाव}
        प्रत्येक वर्ग सेंटी\omterm{def:g4:fractions:def}{मीटर} पर हड्डी बढ़ती है --- और क्या
        विशाल को होता है?
  \item वही नियम विपरीत रूप से चींटी की वीरता की व्याख्या करता है: \omterm{def:g9:fractions:power}{शक्ति}
        मांसपेशी का अनुसरण करता है \omterm{prop:g9:solids:sections}{क्रॉस सेक्शन} (एक \omterm{def:g6:measure:area}{क्षेत्र}), वजन अनुसरण करता है
        \omterm{def:g6:measure:volume}{आयतन}. एक जानवर के लिए $100$ प्रत्येक में कई गुना छोटा
        दिशा, वजन और ताकत किन कारकों से प्रभावित होती है
        सिकुड़ना, और ताकत-से-वजन किस कारक से होता है
        अनुपात में सुधार?
  \item शरीर में ऊष्मा उत्पन्न होती है \omterm{def:g6:measure:volume}{आयतन} और हार गया
        इसकी सतह. दिखाएँ कि एक \omterm{def:g4:numbers:round}{गोल}े के लिए
        सतह से \omterm{def:g6:measure:volume}{आयतन} अनुपात है $\frac{3}{r}$, और इसकी गणना करें
        के लिए $ r = 1$ और $ r = 2$. जो तेजी से ठंडा होता है, एक माउस या
        एक भालू --- और सर्दियों में बच्चों को टोपी की आवश्यकता क्यों होती है?
  \item त्रि\omterm{def:g9:trig:ratios}{ज्या} के दो \omterm{def:g4:numbers:round}{गोल}ाकार संतरे $4$ सेमी और $5$ सेमी हैं
        उसी कीमत पर बेचा गया। उनके अनुपात की गणना करें
        संस्करणों: बड़ा वाला कितना अधिक संतरा देता है
        वही पैसा?
  \item समापन: समाधि के पत्थर की आकृति का सटीक वर्णन करें
        प्रश्न 2 और 3 के दो ``दो तिहाई'' को एनकोड करता है
        --- और सिसरो को एक वाक्य में उत्तर दें: ऐसा क्यों होगा
        गणितज्ञ अपनी हर विजय पर चुनाव करता है
        इंजीनियरिंग प्रतिभा, एक \omterm{def:g6:measure:area}{क्षेत्र} में \omterm{def:g7:areas:prism}{सिलेंडर}?
\end{enumerate}
\end{problem}
