\chapter{थेल्स प्रमेय}\label{ch:g9:thales}

आप की ऊंचाई कैसे मापते हैं? \omterm{def:g8:solids:def}{पिरामिड} उस पर चढ़े बिना? थेल्स
मिलिटस ने इसकी छाया की तुलना छड़ी की छाया से की। उसका प्रमेय
--- \omterm{def:g6:lines:perp}{समानांतर} रेखाएँ काटना \omterm{def:g6:lines:objects}{खंड}ों में \omterm{def:g9:linfunc:linear}{आनुपातिक} टुकड़े --- है
का गणितीय हृदय \omterm{def:g7:prop:scale}{पैमाना} मॉडल, मानचित्र, और \omterm{def:g9:algebra:expand}{विस्तार}, और इनमें से एक
नाम के साथ सबसे पुराने प्रमेय.

\section{प्रमेय}

\begin{theorem}[थेल्स]\label{thm:g9:thales:direct}
मान लीजिए कि दो रेखाएँ एक बिंदु पर मिलती हैं $A $. लेना $ M $ पहली पंक्ति पर और $ N $
दूसरे पर, साथ $ B $ पहले और के साथ आगे $ C $ आगे भी
दूसरा. यदि पंक्तियाँ $(MN)$ और $(BC)$ हैं \emph{\omterm{def:g6:lines:perp}{समानांतर}}, फिर
\omterm{def:g3:shapes:shapes}{त्रिकोण} $ AMN $ और $ ABC $ पास होना \omterm{def:g9:linfunc:linear}{आनुपातिक} पक्ष:
\[
\frac{AM}{AB} = \frac{AN}{AC} = \frac{MN}{BC}.
\]
\end{theorem}
\admitted

\begin{method}[थेल्स के साथ लंबाई की गणना करना]\label{met:g9:thales:compute}
\begin{enumerate}
  \item बिंदु को पहचानें $ A $ जहां दो रेखाएं काटती हैं और दो
        \omterm{def:g6:lines:perp}{समानांतर} रेखाएँ; दो त्रिभुजों के नाम बताइए;
  \item तीन बराबर अनुपात लिखें, हमेशा ``बड़े पर छोटा \omterm{def:g3:shapes:shapes}{त्रिकोण}
        \omterm{def:g3:shapes:shapes}{त्रिकोण}'';
  \item ज्ञात तीन अनुपातों को शामिल करते हुए दो अनुपातों की समानता रखें
        लंबाई और अज्ञात एक;
  \item परिणामी अनुपात को क्रॉस-गुणा द्वारा हल करें।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:thales:compute}
उपरोक्त बाएँ कॉन्फ़िगरेशन में, मान लीजिए $ AM = 3$, $ AB = 5$, $ AN = 2.4$
और $ MN = 3.3$, और आइए गणना करें $ AC $ और $ BC $. थेल्स देता है
\[
\frac{AM}{AB} = \frac{AN}{AC} = \frac{MN}{BC},
\qquad\text{i.e.}\qquad
\frac35 = \frac{2.4}{AC} = \frac{3.3}{BC}.
\]
से $\frac35 = \frac{2.4}{AC}$, क्रॉस-गुणा करना:
$3 \times AC = 5 \times 2.4 = 12$, इसलिए $ AC = 4$. से
$\frac35 = \frac{3.3}{BC}$: $3 \times BC = 16.5$, इसलिए $ BC = 5.5$.
\end{example}

\begin{example}[वास्तविक जीवन में थेल्स]\label{ex:g9:thales:stick}
ऊँचाई की एक खड़ी छड़ी $1$ मी की छाया पड़ती है $1.5$ मी, जबकि एक पेड़
की छाया डालता है $12$ मैं उसी क्षण. सूर्य की किरणें हैं \omterm{def:g6:lines:perp}{समानांतर}, तो
छड़ी-और-छाया और पेड़-और-छाया \omterm{def:g3:shapes:shapes}{त्रिकोण} थेल्स में हैं
विन्यास:
\[
\frac{\text{height of tree}}{1} = \frac{12}{1.5},
\qquad\text{so the tree is } 8 \text{ m tall.}
\]
\end{example}

\section{संभाषण}

\begin{theorem}[थेल्स की बातचीत]\label{thm:g9:thales:converse}
के रूप में रखे गए बिंदुओं के साथ \cref{thm:g9:thales:direct} ($ M $ पर $(AB)$,
$ N $ पर $(AC)$, दोनों पंक्तियों पर समान क्रम में):
यदि
\[
\frac{AM}{AB} = \frac{AN}{AC},
\]
फिर पंक्तियाँ $(MN)$ और $(BC)$ हैं \omterm{def:g6:lines:perp}{समानांतर}.
\end{theorem}
\admitted

\begin{method}[समानता को सिद्ध या असिद्ध करना]\label{met:g9:thales:converse}
\begin{enumerate}
  \item दोनों अनुपातों की अलग-अलग गणना करें $\dfrac{AM}{AB}$ और
        $\dfrac{AN}{AC}$ (जैसा \omterm{def:g9:fractions:fraction}{अंश}ों, नहीं \omterm{def:g4:numbers:round}{गोल}ियाँ);
  \item यदि वे बराबर हैं \emph{और} बिंदु उसी क्रम में हैं
        दो पंक्तियाँ, पंक्तियाँ हैं \omterm{def:g6:lines:perp}{समानांतर} (थेल्स का वार्तालाप);
  \item यदि वे भिन्न हैं, तो रेखाएँ हैं \emph{नहीं} \omterm{def:g6:lines:perp}{समानांतर} --- क्योंकि अगर
        वे थे, थेल्स का प्रमेय अनुपातों को बराबर होने के लिए बाध्य करेगा।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:thales:converse}
लाइन 1 पर, $ AM = 4$ और $ AB = 10$; लाइन 2 पर, $ AN = 6$ और $ AC = 15$.
तब
\[
\frac{AM}{AB} = \frac{4}{10} = \frac25,
\qquad
\frac{AN}{AC} = \frac{6}{15} = \frac25 :
\]
समान अनुपात, समान क्रम: $(MN) \parallel (BC)$.

साथ $ AC = 14$ बजाय: $\frac{6}{14} = \frac37 \neq \frac25$, तो
पंक्तियाँ नहीं हैं \omterm{def:g6:lines:perp}{समानांतर}.
\end{example}

\section{इज़ाफ़ा और कमी}

\begin{definition}[एक आकृति स्केल करना]\label{def:g9:thales:scaling}
किसी आकृति को बड़ा या छोटा करना \emph{पैमाने का कारक}\index{पैमाने का कारक} $ k > 0$ मतलब गुणा करना \emph{सभी} इसका
लंबाई द्वारा $ k $: एक \emph{इज़ाफ़ा} कब $ k > 1$, ए \emph{कमी}
कब $ k < 1$. थेल्स के विन्यास में, त्रिभुज $ AMN $ है
की कमी $ ABC $ कारक द्वारा $ k = \frac{AM}{AB}$.
\end{definition}

\begin{proposition}[कोणों एवं क्षेत्रफलों पर प्रभाव]\label{prop:g9:thales:scaling}
कारक द्वारा स्केलिंग $ k $ कोणों और आकृतियों के आकार को सुरक्षित रखता है,
प्रत्येक लम्बाई को इससे गुणा करता है $ k $, और प्रत्येक को गुणा करता है \emph{\omterm{def:g6:measure:area}{क्षेत्र}} द्वारा
$ k^2$.
\end{proposition}
\admitted

\begin{omfigure}
\begin{tikzpicture}[scale=0.72]
  \draw[thick, omDef, fill=omDef!8] (0,0) rectangle (2,1.2);
  \draw[thick, omThm, fill=omThm!8] (3.4,0) rectangle (7.4,2.4);
  \node[font=\small, omDef] at (1,-0.45) {$2 \times 1.2$};
  \node[font=\small, omThm] at (5.4,-0.45) {$4 \times 2.4$};
  \node[font=\small] at (1,0.6) {क्षेत्र$2.4$};
  \node[font=\small] at (5.4,1.2) {क्षेत्र$9.6$};
\end{tikzpicture}

{\small लंबाई दोगुनी करना ($ k = 2$) को गुणा करता है \omterm{def:g6:measure:area}{क्षेत्र} द्वारा
$ k^2 = 4$: छोटे आयत की चार प्रतियाँ, बड़े वाले को टाइल करें।}
\end{omfigure}

\begin{example}
की एक तस्वीर $10 \times 15$ सेमी से बड़ा किया गया है \omterm{def:g9:thales:scaling}{पैमाने का कारक} $ k = 3$: द
उपाय प्रिंट करें $30 \times 45$ सेमी। इसका \omterm{def:g6:measure:area}{क्षेत्र} से चला जाता है $150$ सेमी $^2$ को
$150 \times 9 = 1350$ सेमी $^2$ --- $9$ गुना बड़ा, नहीं $3$.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g9:thales:1}
पंक्तियाँ $(MN)$ और $(BC)$ हैं \omterm{def:g6:lines:perp}{समानांतर}, साथ $ M $ पर $[AB]$ और $ N $ पर
$[AC]$; $ AM = 2$, $ AB = 6$, $ AN = 3$, $ BC = 9$. गणना करना $ AC $ और $ MN $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:thales:2}
समान विन्यास: $ AM = 5$, $ MB = 3$ (सावधान: $ MB $, नहीं $ AB $!),
$ AN = 4$. गणना करना $ AC $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:thales:3}
तितली विन्यास में, $ A $ के बीच है $ M $ और $ B $ और बीच में $ N $
और $ C $, साथ $(MN) \parallel (BC)$; $ AM = 3$, $ AB = 7.5$, $ AN = 2$,
$ MN = 2.6$. गणना करना $ AC $ और $ BC $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:thales:4}
$ M $ चालू है $[AB]$ साथ $ AM = 6$, $ AB = 8$; $ N $ चालू है $[AC]$ साथ $ AN = 9$,
$ AC = 12$. पंक्तियाँ हैं $(MN)$ और $(BC)$ \omterm{def:g6:lines:perp}{समानांतर}?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:thales:5}
उपरोक्त के समान ही $ AM = 4$, $ AB = 6$, $ AN = 5$, $ AC = 8$. हैं $(MN)$
और $(BC)$ \omterm{def:g6:lines:perp}{समानांतर}? सावधानी से औचित्य सिद्ध करें.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:thales:6}
ए $1.8$ मी लंबा व्यक्ति खड़ा है $2$ स्ट्रीट लैंप के बेस से मी दूर;
उनकी छाया उपाय $3$ एम। द्वारा निर्मित थेल्स विन्यास बनाइये
दीपक, व्यक्ति, और छाया की नोक, और की ऊंचाई की गणना करें
दीपक.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:thales:7}
एक नक्शा है \omterm{def:g7:prop:scale}{पैमाना} $1 : 25\,000$ (मानचित्र पर लंबाई वास्तविक लंबाई है
से गुणा किया गया $ k = \frac{1}{25000}$).
\begin{enumerate}
  \item दो गांव हैं $6.8$ मानचित्र पर सेमी की दूरी। असली बात क्या है
        दूरी, किमी में?
  \item एक जंगल में एक है \omterm{def:g6:measure:area}{क्षेत्र} का $8$ सेमी $^2$ मानचित्र पर. इसकी असलियत क्या है
        \omterm{def:g6:measure:area}{क्षेत्र}, किमी में $^2$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:thales:8}
\omterm{def:g3:shapes:shapes}{त्रिकोण} $ ABC $ है $ AB = 12$, $ AC = 15$, $ BC = 18$. बात $ M $ पर
$[AB]$ संतुष्ट $ AM = 8$, और लाइन के माध्यम से $ M $ \omterm{def:g6:lines:perp}{समानांतर} को $(BC)$
कटौती $[AC]$ पर $ N $.
\begin{enumerate}
  \item गणना करना $ AN $ और $ MN $.
  \item क्या है \omterm{def:g9:thales:scaling}{पैमाने का कारक} से $ ABC $ को $ AMN $? तुलना करें
        \omterm{def:g6:measure:perimeter}{परिधि} दो \omterm{def:g3:shapes:shapes}{त्रिकोण}ों में से.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g9:thales:9}
होने देना $ ABCD $ के साथ एक समलम्बाकार बनें $(AB) \parallel (CD)$, $ AB = 4$ और
$ CD = 6$, जिसके विकर्ण $[AC]$ और $[BD]$ यहां मिलना $ O $.
चारों ओर तितली विन्यास में थेल्स का उपयोग करना $ O $, गणना करें
अनुपात $\dfrac{OA}{OC}$, और दिखाओ $ O $ में दोनों विकर्णों को काटता है
समान अनुपात.
\end{exercise}

\section{समस्या: अचिह्नित रूलर, पिनहोल कैमरा, और
एक समलम्बाकार के तीन साधन}

\begin{problem}[{सप्ताहांत समस्या --- काम पर थेल्स: किसी को विभाजित करना जूते के डिब्बे से सूर्य को मापते हुए, बराबर भागों में विभाजित करें, और ए ट्रैपेज़ॉइड जहां तीनों प्रसिद्ध साधन मिलते हैं}]
\label{pb:g9:thales:1}
थेल्स का प्रमेय किसका गणित है? \emph{किरणों}: सूरज की किरणें,
पिनहोल से प्रकाश किरणें, पेंसिल से किरणें शिखर. यह
समस्या इसे तीन तरीकों से उपयोग करती है --- एक निर्माण उपकरण के रूप में (ए को \omterm{def:g9:arith:divisor}{विभाजित} करना)।
\omterm{def:g6:lines:objects}{खंड} का \emph{कोई} लंबाई, \omterm{def:g3:numbers:evenodd}{यहां तक की} $\sqrt2$, बिल्कुल बराबर में
भागों), एक मापने के उपकरण के रूप में (द \omterm{def:g4:geometry:circle}{व्यास} सूर्य का, साथ में
एक जूते का \omterm{def:g5:solids:def}{डिब्बा} और एक सिक्का), और समलम्ब \omterm{def:g7:central:parallelogram}{चतुर्भुज} पर एक आवर्धक कांच के रूप में
का \cref{exo:g9:thales:9}, जहां अंकगणित, ज्यामितीय और
इस शृंखला के सभी हार्मोनिक साधन एक आकृति में दिखाई देते हैं।

\medskip
\noindent\textbf{Part I --- Dividing with an unmarked ruler.}
\begin{enumerate}
  \item चित्रित करो \omterm{def:g6:lines:objects}{खंड} $[AB]$ का $7$ सेमी। इसे तीन टुकड़ों में काटने के लिए
        कम्पास और अचिह्नित शासक के साथ समान भागों में: कोई भी किरण खींचें
        से $ A $ (के माध्यम से नहीं $ B $); तीन समान कम्पास से कदम हटाएँ
        उस पर लंबाई, अंक दे रही है $ P_1$, $ P_2$, $ P_3$; जोड़ना
        $ P_3$ को $ B $; खींचना समानताएं को $(P_3 B)$ के माध्यम से
        $ P_1$ और $ P_2$. निर्माण करें और कहां चिह्नित करें
        द समानताएं काटना $[AB]$.
  \item के साथ औचित्य सिद्ध करें \cref{thm:g9:thales:direct} वह दो
        चिह्नित अंक काटे गए $[AB]$ बिल्कुल अपने तीसरे बिंदु पर।
  \item ए को \omterm{def:g9:arith:divisor}{विभाजित} करने की विधि अपनाएं \omterm{def:g6:lines:objects}{खंड} में $5$ बराबर
        भागों, फिर निर्माण के लिए $\frac35$ किसी दिए गए का \omterm{def:g6:lines:objects}{खंड}.
  \item बिंदु का निर्माण करें $ M $ का $[AB]$ साथ
        $\frac{AM}{MB} = \frac23$ (वह है, $ M $ पर $\frac25$ का
        से रास्ता $ A $). किरण पर कितने समान कदम हैं, और
        कौन सा बिंदु जुड़ता है $ B $?
  \item यह निर्माण ए से मापने से बेहतर क्यों है?
        स्नातक शासक? एक पर विचार करें \omterm{def:g6:lines:objects}{खंड} लम्बाई का $\sqrt2$
        (एक इकाई वर्ग का विकर्ण, द्वारा अपरिमेय
        \cref{pb:g9:sqrt:1}): माप क्या देगा, और
        थेल्स क्या देता है?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The pinhole camera.}
एक के माध्यम से एक पिन चुभोएं चेहरा एक बंद का \omterm{def:g5:solids:def}{डिब्बा}: इसके विपरीत
चेहरा, दुनिया की एक उलटी छवि दिखाई देती है। एक बिंदु
वस्तु, पिनहोल और छवि बिंदु संरेखित हैं --- प्रकाश
सीधी यात्रा करती है --- इसलिए वस्तु (ऊंचाई) $ H $, दूरी पर $ D $
छेद के सामने) और उसकी छवि (ऊंचाई)। $ h $, पीठ पर
गहराई पर दीवार $ d $ छेद के पीछे) तितली में बैठो
थेल्स का विन्यास.
\begin{enumerate}[resume]
  \item पिनहोल सूत्र व्युत्पन्न करें
        $ h = H \times \frac dD $ तितली में थेल्स से
        छेद के चारों ओर.
  \item ए $12$ मी पेड़ खड़ा है $30$ गहराई के जूते के डिब्बे से मी
        $20$ सेमी। इसकी छवि कितनी लंबी है और किस ओर ऊपर है?
  \item इंगित करें \omterm{def:g5:solids:def}{डिब्बा} सूर्य पर: पर $ d = 1$ के पीछे हूँ
        पिनहोल, सूर्य की छवि एक डिस्क है \omterm{def:g4:geometry:circle}{व्यास} के बारे में
        $9$ मिमी. सूर्य की दूरी को देखते हुए
        $ D = 1.5 \times 10^{11}$ मी, इसकी गणना करें \omterm{def:g4:geometry:circle}{व्यास} में
        \omterm{def:g9:fractions:scientific}{वैज्ञानिक संकेतन} (\cref{pb:g9:fractions:1}का
        काम पर संकेतन)। वास्तविक मूल्य से तुलना करें,
        $1.39 \times 10^9$ एम।
  \item चांद: \omterm{def:g4:geometry:circle}{व्यास} $3.5 \times 10^6$ मी, दूरी
        $3.8 \times 10^8$ एम। अनुपात की गणना करें
        $\frac{\text{diameter}}{\text{distance}}$ चंद्रमा के लिए
        और सूर्य के लिए. कितना भाग्यशाली संयोग है दोनों का
        संख्याओं से पता चलता है --- और यह क्या शानदार घटना है
        संभव बनाएं?
  \item एक या दो वाक्यों में: पिनहोल छवि क्यों है
        उल्टा, और पिनहोल को बड़ा करने से ऐसा क्यों होता है
        छवि उज्जवल लेकिन धुंधली?
\end{enumerate}

\noindent\textbf{Part III --- The trapezoid of the three
means.}
$ ABCD $ का समलंब है \cref{exo:g9:thales:9}:
$(AB) \parallel (CD)$, $ AB = 4$, $ CD = 6$, विकर्ण पर प्रतिच्छेद करते हैं
$ O $ साथ $\frac{OA}{OC} = \frac{OB}{OD} = \frac{4}{6} =
\frac23$.
\begin{enumerate}[resume]
  \item तितली अभ्यास: दो रेखाएं एक दूसरे को काटती हैं $ O $ साथ
        $(MN) \parallel (PQ)$ तितली की स्थिति में; $ OM = 3$,
        $ OP = 7.5$, $ MN = 4$ और $ OQ = 6$. गणना करना $ PQ $ और
        $ ON $.
  \item बताएं कि थेल्स ने अनुपात के साथ कैसे आवेदन किया $\frac12$,
        का मध्यबिंदु प्रमेय शामिल है
        \cref{thm:g8:midpoints:direct} इसके वि\omterm{def:g3:division:remainder}{शेष} मामले के रूप में.
  \item खींचना \omterm{def:g6:lines:perp}{समानांतर} के माध्यम से आधारों तक $ O $; यह मिलता है
        $[AD]$ पर $ E $. \omterm{def:g3:shapes:shapes}{त्रिकोण} में कार्य करना $ ACD $ (टिप्पणी
        $\frac{AO}{AC} = \frac25$), गणना करें $ EO $.
  \item त्रिभुज में दर्पण गणना द्वारा $ BDC $, द
        टुकड़ा $ OF $ (साथ $ F $ पर $[BC]$) की लंबाई समान है.
        समाप्त करें कि
        \[
        EF = 2 \times \frac{4 \times 6}{4 + 6} = 4.8 :
        \]
         \omterm{def:g6:lines:perp}{समानांतर} विकर्ण क्रॉसिंग के माध्यम से मापता है
        \emph{\omterm{pb:g8:speed:1}{अनुकूल माध्य}} आधारों की --- गोल-यात्रा
        का मतलब \cref{pb:g8:speed:1}, शुद्ध रूप में पुनः प्रकट होना
        ज्यामिति.
  \item भव्य समापन: तीन शास्त्रीय साधनों की गणना करें
        आधार $4$ और $6$ --- अंकगणित
        $\frac{4+6}{2}$, ज्यामितीय $\sqrt{4 \times 6}$, हार्मोनिक
        $\frac{2 \times 4 \times 6}{4 + 6}$ --- और उन्हें ऑर्डर करें.
        प्रत्येक समलम्ब \omterm{def:g7:central:parallelogram}{चतुर्भुज} में रहता है: अंकगणितीय माध्य है
        पैरों के मध्य बिंदुओं को जोड़ने वाली मध्य \omterm{def:g6:lines:objects}{रेखा}
        (\cref{pb:g7:areas:1}), द \omterm{pb:g8:speed:1}{अनुकूल माध्य} आपकी है
        \omterm{def:g6:lines:objects}{खंड} $ EF$, और यह ज्यामितीय माध्य है \omterm{def:g6:lines:perp}{समानांतर}
        जो समलंब को दो भागों में काट देता है \emph{समान}
        ट्रैपेज़ॉइड्स (कैलकुलेटर के साथ इसका मूल्य जांचें;
        सबूत एक सुंदर अतिरिक्त चुनौती है)। कहाँ है
        इस पुस्तक में तीनों के बीच असमानता सिद्ध की गई है
        (\cref{pb:g8:cosine:1}, \cref{pb:g8:speed:1})?
\end{enumerate}
\end{problem}
