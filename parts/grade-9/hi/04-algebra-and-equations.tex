\chapter{बीजगणित और समीकरण}\label{ch:g9:algebra}

अक्षर संख्याओं का प्रतिनिधित्व करते हैं, और अक्षरों से गणना करने पर तथ्य सिद्ध होते हैं
\emph{सभी} एक ही बार में नंबर. यह अध्याय अभ्यास करता है का \omterm{def:g9:algebra:expand}{विस्तार} और
फ़ैक्टरिंग --- जिसमें तीन पहचानें शामिल हैं जो आपका अनुसरण करेंगी
गणित के सभी --- और प्रथम-डिग्री हल करता है \omterm{def:g8:equations:def}{समीकरण}, \omterm{def:g2:mult:def}{उत्पाद}
\omterm{def:g8:equations:def}{समीकरण}, और सरल असमानताएँ, हमेशा एक समय में एक उचित कदम।

\section{विस्तार}

\begin{definition}[विस्तार]\label{def:g9:algebra:expand}
\emph{विस्तार}\index{विस्तार} का अर्थ है परिवर्तन करना \omterm{def:g2:mult:def}{उत्पाद} एक में
\omterm{def:g1:addition:def}{जोड़}, वितरणशीलता का उपयोग करते हुए:
\[
k(a + b) = ka + kb,
\qquad
(a + b)(c + d) = ac + ad + bc + bd .
\]
\end{definition}

\begin{example}\label{ex:g9:algebra:expand}
पहले कोष्ठक का प्रत्येक पद दूसरे कोष्ठक के प्रत्येक पद को गुणा करता है:
\begin{align*}
(x + 3)(2x + 5)
&= x \times 2x + x \times 5 + 3 \times 2x + 3 \times 5 \\
&= 2x^2 + 5x + 6x + 15 \\
&= 2x^2 + 11x + 15 .
\end{align*}
संकेत शर्तों के साथ यात्रा करते हैं:
\begin{align*}
(3x - 2)(x - 4)
&= 3x^2 - 12x - 2x + 8
= 3x^2 - 14x + 8 .
\end{align*}
\end{example}

\begin{theorem}[तीन पहचान]\label{thm:g9:algebra:identities}
सभी नंबरों के लिए $a $ और $ b $:
\[
(a+b)^2 = a^2 + 2ab + b^2,
\qquad
(a-b)^2 = a^2 - 2ab + b^2,
\qquad
(a+b)(a-b) = a^2 - b^2 .
\]
\end{theorem}

\begin{proof}
प्रत्येक बायीं ओर का \omterm{def:g9:algebra:expand}{विस्तार} करें. पहले के लिए:
$(a+b)(a+b) = a^2 + ab + ba + b^2 = a^2 + 2ab + b^2$. दूसरे के लिए,
बदलें $ b $ द्वारा $-b $ पहले में। तीसरे के लिए:
$(a+b)(a-b) = a^2 - ab + ba - b^2 = a^2 - b^2$ (क्रॉस शर्तें रद्द करें)।
\end{proof}

\begin{example}\label{ex:g9:algebra:identities}
\[
(x + 5)^2 = x^2 + 10x + 25, \qquad
(3x - 1)^2 = 9x^2 - 6x + 1, \qquad
(2x + 7)(2x - 7) = 4x^2 - 49 .
\]
मानसिक अंकगणित: $102^2 = (100 + 2)^2 = 10000 + 400 + 4 = 10404$, और
$98 \times 102 = (100-2)(100+2) = 10000 - 4 = 9996$.
\end{example}

\section{फैक्टरिंग}

\begin{method}[फैक्टरिंग]\label{met:g9:algebra:factor}
किसी अभिव्यक्ति को गुणनखंडित करने के लिए, क्रम में प्रयास करें:
\begin{enumerate}
  \item \emph{सामान्य कारक}: $ ka + kb = k(a + b)$, कहाँ $ k $ स्वयं हो सकता है
        एक कोष्ठक हो;
  \item \emph{\omterm{ex:g1:subtraction:difference}{अंतर} वर्गों का}: $ a^2 - b^2 = (a+b)(a-b)$;
  \item \emph{पूर्ण वर्ग}: $ a^2 + 2ab + b^2 = (a+b)^2$ (और साथ में
        $-2ab $, $(a-b)^2$).
\end{enumerate}
हमेशा जाँच करें का \omterm{def:g9:algebra:expand}{विस्तार} नतीजा.
\end{method}

\begin{example}\label{ex:g9:algebra:factor}
\emph{सामान्य कारक:}
$15x^2 + 10x = 5x \times 3x + 5x \times 2 = 5x(3x + 2)$.

\emph{सामान्य ब्रैकेट:}
\[
(x+2)(x-1) + (x+2)(3x+4) = (x+2)\bigl[(x-1) + (3x+4)\bigr]
= (x+2)(4x+3).
\]

\emph{\omterm{ex:g1:subtraction:difference}{अंतर} वर्गों का:}
$ x^2 - 81 = (x+9)(x-9)$, और
\[
(2x+1)^2 - 9 = (2x+1)^2 - 3^2 = (2x+1+3)(2x+1-3) = (2x+4)(2x-2).
\]
प्रत्येक कारक का अभी भी एक सामान्य कारक है: $(2x+4)(2x-2) = 2(x+2) \times
2(x-1) = 4(x+2)(x-1)$.
\end{example}

\section{समीकरण}

\begin{method}[प्रथम-डिग्री समीकरण]\label{met:g9:algebra:firstdegree}
एक को हल करने के लिए \omterm{def:g8:equations:def}{समीकरण} पसंद $5x - 3 = 2x + 9$:
\begin{enumerate}
  \item इकट्ठा करो $ x $-शब्द एक तरफ, संख्याएं दूसरी तरफ, द्वारा
        दोनों तरफ समान मात्रा जोड़ना;
  \item प्रत्येक पक्ष को कम करें;
  \item के गुणांक से दोनों पक्षों को \omterm{def:g9:arith:divisor}{विभाजित} करें $ x $;
  \item मूल में समाधान की जाँच करें \omterm{def:g8:equations:def}{समीकरण}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:algebra:firstdegree}
\begin{align*}
5x - 3 &= 2x + 9 \\
5x - 2x &= 9 + 3 && \text{(add $3 - 2x $ to both sides)}\\
3x &= 12 \\
x &= 4 && \text{(divide by 3).}
\end{align*}
जाँच करना: $5 \times 4 - 3 = 17$ और $2 \times 4 + 9 = 17$. समाधान है
$4$.

कोष्ठक के साथ, पहले \omterm{def:g9:algebra:expand}{विस्तार} करें: $3(x - 2) = x + 8$ बन जाता है $3x - 6 = x + 8$,
इसलिए $2x = 14$ और $ x = 7$.
\end{example}

\begin{theorem}[शून्य-उत्पाद नियम]\label{thm:g9:algebra:zeroproduct}
ए \omterm{def:g2:mult:def}{उत्पाद} है \omterm{def:g1:counting:zero}{शून्य} यदि और केवल यदि कम से कम एक कारक है \omterm{def:g1:counting:zero}{शून्य}:
\[
A \times B = 0
\quad\Longleftrightarrow\quad
A = 0 \ \text{ or } \ B = 0 .
\]
\end{theorem}

\begin{proof}
यदि कोई कारक है \omterm{def:g1:counting:zero}{शून्य}  \omterm{def:g2:mult:def}{उत्पाद} स्पष्ट रूप से है. इसके विपरीत यदि $ A B = 0$ साथ
$ A \neq 0$, दोनों पक्षों को \omterm{def:g9:arith:divisor}{विभाजित} करते हुए $ A $ देता है $ B = 0$.
\end{proof}

\begin{example}\label{ex:g9:algebra:zeroproduct}
हल करना $(x - 3)(2x + 10) = 0$: दोनों में से एक $ x - 3 = 0$ या $2x + 10 = 0$, तो
समाधान हैं $3$ और $-5$.

हल करना $ x^2 - 49 = 0$: कारक पहले, $(x-7)(x+7) = 0$, समाधान $7$ और
$-7$.

हल करना $ x^2 = 3x $: कभी भी \omterm{def:g9:arith:divisor}{विभाजित} न करें $ x $! चाल और कारक:
$ x^2 - 3x = x(x - 3) = 0$, समाधान $0$ और $3$.
\end{example}

\section{असमानता}

\begin{proposition}[असमानताओं के लिए नियम]\label{prop:g9:algebra:ineq}
दोनों पक्षों में समान संख्या जोड़ने पर एक असमानता संरक्षित रहती है, और
दोनों पक्षों को समान से गुणा करने पर \emph{सकारात्मक} संख्या।
दोनों पक्षों को a से गुणा करने पर \emph{नकारात्मक} संख्या उलट जाती है
असमानता का संकेत.
\end{proposition}

\begin{proof}
अगर $ a \leq b $ तब $ b - a \geq 0$. \omterm{def:g1:addition:def}{जोड़}ा जा रहा है $ c $: $(b+c) - (a+c) = b - a
\geq 0$, इसलिए $ a + c \leq b + c $. से गुणा करना $ c > 0$:
$ bc - ac = (b-a)c \geq 0$, इसलिए $ ac \leq bc $. से गुणा करना $ c < 0$:
$(b-a)c \leq 0$, इसलिए $ ac \geq bc $: क्रम उलट गया है.
\end{proof}

\begin{example}\label{ex:g9:algebra:ineq}
हल करना $7 - 2x < 15$:
\begin{align*}
7 - 2x &< 15 \\
-2x &< 8 && \text{(subtract 7)} \\
x &> -4 && \text{(divide by $-2$: reverse the sign).}
\end{align*}
समाधान सभी संख्याओं से बड़े हैं $-4$: एक संख्या \omterm{def:g6:lines:objects}{रेखा} पर, ए
खोखला बिंदु पर $-4$ और दाहिनी ओर छायांकन।
\end{example}

\section{समस्या: अंकों का बीजगणित}

\begin{problem}[{सप्ताहान्त समस्या --- विभाज्यता नियम क्यों काम, 1089 चाल का पर्दाफाश, और मानसिक अंकगणित जो दिखता है जादू की तरह}]
\label{pb:g9:algebra:1}
ग्रेड स्कूल ने आपको सिखाया \emph{नियम} (एक संख्या है \omterm{def:g6:wholes:divisible}{भाज्य} द्वारा
$9$ जब उसका अंक \omterm{def:g1:addition:def}{जोड़} है); जादूगर प्रदर्शन करते हैं \emph{चाल}
(हर कोई समाप्त होता है $1089$); बाज़ार जानते हैं \emph{शॉर्टकट}
($35^2 = 1225$, तुरन्त)। यह सब एक ही रहस्य है: ए
अंकों के साथ संख्या $ a $, $ b $, $ c $ अभिव्यक्ति है
$100a + 10b + c $, और इस अध्याय का बीजगणित
(\cref{thm:g9:algebra:identities}, का \omterm{def:g9:algebra:expand}{विस्तार}, फ़ैक्टरिंग) ले सकते हैं
यह अलग. इस समस्या के अंत तक आप यह सिद्ध कर चुके होंगे
नियम, चाल का भंडाफोड़, और जादू सीख लिया।

\medskip
\noindent\textbf{Part I --- Digits under the microscope.}
\begin{enumerate}
  \item दहाई अंक वाली दो अंकीय संख्या $ a $ और इकाई अंक
        $ b $ है $10a + b $. \omterm{def:g9:algebra:expand}{विस्तार} करें और गुणनखंड करें \omterm{ex:g1:subtraction:difference}{अंतर}
        संख्या और उसके उलटने के बीच,
        $(10a + b) - (10b + a)$, और खोज बताएं। जांचें
        यह पर $72 - 27$.
  \item तीन अंकीय संख्याओं के लिए नौ का नियम सिद्ध करें: सत्यापित करें
        पहचान
        \[
        100a + 10b + c = 9\,(11a + b) + (a + b + c),
        \]
        और समझाएं कि यह एक संख्या और उसका अंक क्यों दिखाता है \omterm{def:g1:addition:def}{जोड़}
        छोड़ दो \emph{वही \omterm{def:g3:division:remainder}{शेष}} द्वारा प्रभाग में $9$
        --- तो एक है \omterm{def:g6:wholes:divisible}{भाज्य} द्वारा $9$ बिल्कुल जब दूसरा है
        (\cref{prop:g9:arith:rules}, अब साबित हुआ)।
  \item एक ही पहचान से, एक में तीन का नियम निकालें
        वाक्य.
  \item ``नौ को बाहर निकालना'', लेखाकारों की जाँच: परीक्षण करने के लिए
        $47 \times 86 = 4\,042$, प्रत्येक संख्या को से बदलें
        अंक \omterm{def:g1:addition:def}{जोड़} इसके अंक का \omterm{def:g1:addition:def}{जोड़}: $47 \to 11 \to 2$,
        $86 \to 14 \to 5$, और $2 \times 5 = 10 \to 1$; तब
        $4\,042 \to 10 \to 1$: सुसंगत। स्पष्ट करें कि क्यों सही है
        \omterm{def:g2:mult:def}{उत्पाद} परीक्षा उत्तीर्ण करनी होगी (लिखें) $47 = 9k + 2$,
        $86 = 9l + 5$ और \omterm{def:g9:algebra:expand}{विस्तार} करें), और एक गलत परिणाम खोजें
        परीक्षण पकड़ने में विफल रहता है (अंकों की अदला-बदली क्यों की जाती है
        इसके लिए अदृश्य?)
  \item चार अंकीय संख्याओं के लिए ग्यारह का नियम सिद्ध करें: सत्यापित करें
        \[
        1000a + 100b + 10c + d
        = 11\,(91a + 9b + c) + (- a + b - c + d),
        \]
        और इसका उपयोग यह तय करने के लिए करें कि क्या $2\,926$ है \omterm{def:g6:wholes:divisible}{भाज्य} द्वारा
        $11$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The 1089 trick.}
तरकीब: एक तीन अंकों की संख्या के बारे में सोचें जिसका पहला और आखिरी
अंकों में कम से कम \omterm{ex:g1:subtraction:difference}{अंतर} है $2$; इसे उल्टा करो; छोटे को घटाओ
बड़े से; परिणाम को उल्टा करें; अंतिम दो संख्याएँ \omterm{def:g1:addition:def}{जोड़}ें.
जादूगर ने घोषणा की: \emph{1089}.
\begin{enumerate}[resume]
  \item चाल को चालू करें $742$ और आपके एक नंबर पर
        विकल्प.
  \item माना कि संख्या में अंक हैं $ a $, $ b $, $ c $ साथ $ a > c $.
        पहले घटाव का गुणनखंड करें,
        $(100a + 10b + c) - (100c + 10b + a)$, और सभी की सूची बनाएं
        मान यह कब ले सकता है $ a - c $ से चलती है $2$ को $9$.
  \item उनमें से प्रत्येक \omterm{def:g5:division:multiple}{गुणकों} का $99$ एक आकर्षक अंक है
        पैटर्न. लिखना $ m = a - c $, पहचान सत्यापित करें
        \[
        99m = 100\,(m - 1) + 90 + (10 - m),
        \]
        और के तीन अंक पढ़ें $99m $.
  \item जोड़ना $99m $ प्रश्न 8 का उपयोग करते हुए, अपने स्वयं के उलटफेर के लिए
        अंक, और साबित करें कि कुल है $1089$ --- जो कुछ भी
        $ m $ है। युक्ति मर चुकी है; बीजगणित लंबे समय तक जीवित रहे।
  \item जादूगर को पहले और आखिरी अंक की आवश्यकता क्यों पड़ी?
        कम से कम भिन्न होना $2$? मामलों की जांच करें $ a = c $ और
        $ a - c = 1$ (जहां घटाव देता है $99$, जो
        लिखा जाना चाहिए $099$ काम में उलटफेर के लिए), और
        वह छोटा प्रिंट बताएं जो चाल को ईमानदार बनाए रखता है।
\end{enumerate}

\noindent\textbf{Part III --- Sorcery for daily use.}
\begin{enumerate}[resume]
  \item जन्मदिन की युक्ति: ``अपने जन्म महीने को गुणा करें $5$,
        जोड़ना $7$, दोगुना करें, अपना जन्म दिन \omterm{def:g1:addition:def}{जोड़}ें, घटाएँ $14$.''
        दिखाएँ कि परिणाम हमेशा है $10m + d $ --- महीना और
        दिन, एक नज़र में पढ़ने योग्य। (इसे किसी पर आज़माएं।)
  \item आप ट्रिक को इस प्रकार पुनः डिज़ाइन करें: ``महीने को इससे गुणा करें $5$,
        जोड़ना $9$, दोगुना, दिन \omterm{def:g1:addition:def}{जोड़}ें।'' कौन सी संख्या होनी चाहिए
        दर्शक अब इसके लिए अंत में घटाएँ
        महीना-दिन रीडआउट?
  \item लुप्त अंक वाली युक्ति: दर्शक कोई भी संख्या चुनता है,
        उसका अंक घटा देता है \omterm{def:g1:addition:def}{जोड़} (प्रश्न 2 कहता है कि परिणाम है
        हमेशा ए \omterm{def:g9:arith:divisor}{एकाधिक} का $9$ --- स्पष्ट करें कि क्यों
        तीन अंकों की संख्या), फिर एक को काट देता है \emph{अशून्य}
        अंक और आपको \omterm{def:g3:division:remainder}{शेष} अंक पढ़ता है। बताएं कैसे
        आप पार किए गए अंक को पुनर्प्राप्त करते हैं, और क्यों $0$ होना ही था
        बहिष्कृत.
  \item बाजार का जादू: पहचान साबित करें
        $(10t + 5)^2 = 100\,t(t+1) + 25$, के लिए नियम बताएं
        से समाप्त होने वाली किसी भी संख्या का वर्ग करना $5$, और गणना करें $45^2$,
        $85^2$ और $105^2$ अपने सिर में।
  \item समापन, \omterm{def:g9:algebra:expand}{विस्तार} \cref{exo:g9:algebra:10}: सिद्ध करो
        द \omterm{def:g1:addition:def}{जोड़} पाँच क्रमागत पूर्णांकों का सदैव a होता है
        \omterm{def:g9:arith:divisor}{एकाधिक} का $5$, लेकिन वह \omterm{def:g1:addition:def}{जोड़} का \emph{चार}
        क्रमागत पूर्णांक है \emph{कभी नहीं} ए \omterm{def:g9:arith:divisor}{एकाधिक} का $4$.
        (दो छोटी गणनाएँ --- और किस बारे में एक नैतिक
        बीजगणित ``हमेशा'' और ``कभी नहीं'' शब्दों पर लागू होता है।)
\end{enumerate}
\end{problem}
