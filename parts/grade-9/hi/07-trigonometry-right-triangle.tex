\chapter{समकोण त्रिभुज में त्रिकोणमिति}\label{ch:g9:trig}

में एक सही \omterm{def:g3:shapes:shapes}{त्रिकोण}, एक बार जब आप एक न्यून कोण जान लेते हैं, तो \emph{आकार} का
त्रिभुज स्थिर है --- केवल इसका आकार भिन्न हो सकता है। इसकी भुजाओं का अनुपात
इसलिए केवल कोण पर निर्भर करते हैं: ये अनुपात हैं \omterm{def:g9:trig:ratios}{कोज्या}, \omterm{def:g9:trig:ratios}{ज्या} और
\omterm{def:g9:trig:ratios}{स्पर्शरेखा}. पाइथागोरस प्रमेय के साथ, वे आपको गणना करने देते हैं
प्रत्येक पक्ष और प्रत्येक कोण सही \omterm{def:g3:shapes:shapes}{त्रिकोण} बहुत कम से
जानकारी.

\section{पाइथागोरस प्रमेय, फिर से}

\begin{theorem}[पाइथागोरस]\label{thm:g9:trig:pythagoras}
एक \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ABC $ \omterm{def:g6:shapes:triangles}{समकोण का} पर $ A $, कर्ण का वर्ग है
द \omterm{def:g1:addition:def}{जोड़} दोनों पैरों के वर्गों का:
\[
BC^2 = AB^2 + AC^2 .
\]
इसके विपरीत, यदि $ BC^2 = AB^2 + AC^2$ एक \omterm{def:g3:shapes:shapes}{त्रिकोण} में, तो \omterm{def:g3:shapes:shapes}{त्रिकोण} है
\omterm{def:g6:shapes:triangles}{समकोण का} पर $ A $.
\end{theorem}
\admitted

\begin{example}\label{ex:g9:trig:pythagoras}
ए सही \omterm{def:g3:shapes:shapes}{त्रिकोण} पैर हैं $ AB = 5$ और $ AC = 12$. तब
\[
BC^2 = 5^2 + 12^2 = 25 + 144 = 169,
\qquad\text{so}\quad BC = \sqrt{169} = 13 .
\]
एक खोजने के लिए \emph{टांग}: अगर $ BC = 10$ और $ AB = 6$, तब
$ AC^2 = BC^2 - AB^2 = 100 - 36 = 64$, इसलिए $ AC = 8$.
\end{example}

\section{कोज्या, ज्या, स्पर्शज्या}

में एक सही \omterm{def:g3:shapes:shapes}{त्रिकोण}, एक न्यून कोण के सापेक्ष $\theta $, तीन पक्ष
नाम हैं: द \emph{कर्ण} (विपरीत) \omterm{def:g3:shapes:rightangle}{समकोण}, सबसे लंबा
ओर), ओर \emph{विलोम} को $\theta $, और पक्ष
\emph{नज़दीक} को $\theta $ (पैर छूना $\theta $).

\begin{definition}[त्रिकोणमितीय अनुपात]\label{def:g9:trig:ratios}
न्यून कोण के लिए $\theta $ एक का सही \omterm{def:g3:shapes:shapes}{त्रिकोण}:
\[
\cos\theta = \frac{\text{adjacent}}{\text{hypotenuse}},
\qquad
\sin\theta = \frac{\text{opposite}}{\text{hypotenuse}},
\qquad
\tan\theta = \frac{\text{opposite}}{\text{adjacent}} .
\]
ये अनुपात\index{कोज्या}\index{ज्या}\index{स्पर्शरेखा} पर ही निर्भर हैं
$\theta $, त्रिभुज के आकार पर नहीं (सभी) \omterm{def:g3:shapes:rightangle}{समकोण} त्रिभुज के साथ
थेल्स द्वारा समान न्यून कोण एक दूसरे की स्केलिंग हैं)।
\end{definition}

\begin{proposition}[प्रथम गुण]\label{prop:g9:trig:properties}
प्रत्येक न्यून कोण के लिए $\theta $:
\begin{enumerate}
  \item $0 < \cos\theta < 1$ और $0 < \sin\theta < 1$ (एक पैर छोटा है
        कर्ण से);
  \item $\cos^2\theta + \sin^2\theta = 1$;
  \item $\tan\theta = \dfrac{\sin\theta}{\cos\theta}$.
\end{enumerate}
\end{proposition}

\begin{proof}
लिखना $ a $ बगल वाले हिस्से के लिए, $ o $ विपरीत पक्ष के लिए, $ h $ के लिए
ए का कर्ण सही \omterm{def:g3:shapes:shapes}{त्रिकोण} कोण के साथ $\theta $.

1. $0 < a < h $ और $0 < o < h $, इसलिए दोनों अनुपात सख्ती से बीच में हैं
$0$ और $1$.

2. पाइथागोरस द्वारा, $ a^2 + o^2 = h^2$. हर चीज़ को \omterm{def:g9:arith:divisor}{विभाजित} करें $ h^2$:
\[
\cos^2\theta + \sin^2\theta
= \frac{a^2}{h^2} + \frac{o^2}{h^2}
= \frac{a^2 + o^2}{h^2} = 1 .
\]

3. $\dfrac{\sin\theta}{\cos\theta} = \dfrac{o/h}{a/h} = \dfrac oa
= \tan\theta $.
\end{proof}

\begin{method}[एक पक्ष ढूँढना]\label{met:g9:trig:side}
जब एक भुजा और एक न्यूनकोण ज्ञात हो तो अज्ञात भुजा की गणना करने के लिए:
\begin{enumerate}
  \item ज्ञात कोण को चिह्नित करें और तीनों भुजाओं (कर्ण,
        विपरीत, आसन्न) \emph{उस कोण से};
  \item वह अनुपात (कॉस, साइन या टैन) चुनें जिसमें ज्ञात पक्ष शामिल हो
        और वांछित पक्ष;
  \item लिखना \omterm{def:g8:equations:def}{समीकरण} और इसे हल करें;
  \item विवेक-जांच: कर्ण सबसे लंबे समय तक बाहर आना चाहिए।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:trig:side}
में एक सही \omterm{def:g3:shapes:shapes}{त्रिकोण}, कर्ण माप $8$ और एक कोण है
$35^\circ $; इसके विपरीत पक्ष की गणना करें। अनुपात शामिल है
विपरीत भुजा और कर्ण है \omterm{def:g9:trig:ratios}{ज्या}:
\[
\sin 35^\circ = \frac{\text{opposite}}{8}
\quad\Longrightarrow\quad
\text{opposite} = 8 \sin 35^\circ \approx 8 \times 0.574 \approx 4.6 .
\]
निकटवर्ती पक्ष के लिए, का उपयोग करें \omterm{def:g9:trig:ratios}{कोज्या}:
$8\cos 35^\circ \approx 8 \times 0.819 \approx 6.6$. के साथ जाँच करें
पाइथागोरस: $4.6^2 + 6.6^2 \approx 21 + 43 \approx 64 = 8^2$.
\end{example}

\begin{method}[एक कोण ढूँढना]\label{met:g9:trig:angle}
जब दो पक्ष ज्ञात हों, तो उनके द्वारा बनाए गए अनुपात की गणना करें, इसे इस रूप में पहचानें
\omterm{def:g9:trig:ratios}{कोज्या}, \omterm{def:g9:trig:ratios}{ज्या} या \omterm{def:g9:trig:ratios}{स्पर्शरेखा} अज्ञात कोण का, और कैलकुलेटर का उपयोग करें
\omterm{def:g8:fractions:inverse}{श्लोक में} समारोह ($\cos^{-1}$, $\sin^{-1}$ या $\tan^{-1}$) अनुपात के लिए.
\end{method}

\begin{example}\label{ex:g9:trig:angle}
एक सीढ़ी $5$ मी लंबे समय तक एक दीवार के सहारे टिका रहा, उसका पैर $1.4$ मी से
दीवार. \omterm{def:g8:solids:def}{कोन}ा $\theta $ सीढ़ी और जमीन के बीच संतुष्टि मिलती है
\[
\cos\theta = \frac{1.4}{5} = 0.28,
\qquad\text{so}\quad
\theta = \cos^{-1}(0.28) \approx 73.7^\circ .
\]
\end{example}

\begin{example}[विशेष कोण]\label{ex:g9:trig:special}
दो त्रिभुज सटीक मान देते हैं। \omterm{def:g3:division:half}{आधा} भुजा 1 का एक वर्ग (एक दाईं ओर
\omterm{def:g6:shapes:triangles}{समद्विबाहु} त्रिभुज) में कर्ण है $\sqrt2$ और के कोण $45^\circ $:
\[
\cos 45^\circ = \sin 45^\circ = \frac{1}{\sqrt2} = \frac{\sqrt2}{2},
\qquad \tan 45^\circ = 1 .
\]
\omterm{def:g3:division:half}{आधा} एक समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण} भुजा 1 का कोण देता है $30^\circ $ और
$60^\circ $:
\[
\cos 60^\circ = \sin 30^\circ = \frac12,
\qquad
\cos 30^\circ = \sin 60^\circ = \frac{\sqrt3}{2}.
\]
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g9:trig:1}
ए सही \omterm{def:g3:shapes:shapes}{त्रिकोण} पैर हैं $9$ और $12$. इसके कर्ण की गणना करें। दूसरा
कर्ण है $17$ और एक पैर $8$: दूसरे पैर की गणना करें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:trig:2}
एक त्रिभुज की भुजाएँ होती हैं $7$, $24$ और $25$. यह है \omterm{def:g6:shapes:triangles}{समकोण का}? वही
पक्षों के लिए प्रश्न $5$, $6$ और $8$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:trig:3}
एक \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ DEF $ \omterm{def:g6:shapes:triangles}{समकोण का} पर $ D $, पर कोण के साथ $ E $ विख्यात
$\theta $: कर्ण किस तरफ है? कौन सा पक्ष विपरीत है
$\theta $? लिखना $\cos\theta $, $\sin\theta $ और $\tan\theta $ के अनुपात के रूप में
पक्ष $ DE $, $ DF $, $ EF $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:trig:4}
में एक सही \omterm{def:g3:shapes:shapes}{त्रिकोण}, कर्ण माप $10$ और एक न्यून कोण है
$28^\circ $. दोनों पैरों की गणना करें (\omterm{def:g4:numbers:round}{गोल} दसवें तक;
$\cos 28^\circ \approx 0.883$, $\sin 28^\circ \approx 0.469$).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:trig:5}
में एक सही \omterm{def:g3:shapes:shapes}{त्रिकोण}, कोण से सटा हुआ पैर $\theta $ पैमाने $6$
और विपरीत पैर का माप $4.5$. गणना करना $\tan\theta $, तब $\theta $
(\omterm{def:g4:numbers:round}{गोल} डिग्री तक; $\tan^{-1}(0.75) \approx 36.9^\circ $).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:trig:6}
की ऊँचाई पर एक ड्रोन उड़ता है $120$ एम। ज़मीन पर मौजूद एक पर्यवेक्षक से,
के उन्नयन कोण पर देखा जाता है $32^\circ $. किस क्षैतिज पर
पर्यवेक्षक से दूरी ड्रोन है
($\tan 32^\circ \approx 0.625$)?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:trig:7}
एक पहुंच रैंप अवश्य बढ़ना चाहिए $1.2$ अधिकतम कोण के साथ मी $5^\circ $ को
क्षैतिज. के साथ न्यूनतम लंबाई क्या है \omterm{prop:g9:linfunc:affinegraph}{ढलान} रैंप होना चाहिए
($\sin 5^\circ \approx 0.0872$)? ए के निकटतम दसवें तक पूर्णांकित करें
\omterm{def:g4:fractions:def}{मीटर}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:trig:8}
का उपयोग करते हुए \cref{prop:g9:trig:properties}: एक न्यूनकोण $\theta $ संतुष्ट
$\cos\theta = 0.6$. गणना करना $\sin\theta $ बिना खोजे $\theta $, तब
$\tan\theta $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g9:trig:9}
के सटीक मानों का औचित्य सिद्ध करें \cref{ex:g9:trig:special}: दाईं ओर
\omterm{def:g6:shapes:triangles}{समद्विबाहु} त्रिभुज पैरों के साथ $1$, कर्ण की गणना करें; में
समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण} इस पक्ष की $1$, ऊंचाई की गणना करें, और निष्कर्ष निकालें
\omterm{def:g9:trig:ratios}{कोज्या} और \omterm{def:g9:trig:ratios}{ज्या} का $30^\circ $ और $60^\circ $.
\end{exercise}

\section{समस्या: जिस चीज़ तक आप नहीं पहुँच सकते उसे मापना}

\begin{problem}[{सप्ताहांत समस्या---सर्वेक्षक का दो-स्टेशन दुर्गम ऊंचाइयों, सड़क ग्रेड, की दूरी के लिए विधि क्षितिज, और स्पर्शरेखा की अनंत तक दौड़}]
\label{pb:g9:trig:1}
कोई भी टेप माप किसी पहाड़ की चोटी, शिखर तक नहीं पहुँचता
किसी नदी के पार टॉवर, या समुद्र में क्षितिज --- लेकिन एक कोण
गेज और इस अध्याय के तीन अनुपात करते हैं। इसका केंद्रबिंदु
समस्या सर्वेक्षकों की क्लासिक है \emph{दो-स्टेशन विधि},
जो किसी ऊंचाई की गणना उसके आधार तक पहुंचे बिना करता है;
इसके चारों ओर: सड़क के संकेत, सीढ़ियाँ, सीढ़ियाँ, की वक्रता
पृथ्वी, और अनंत तक विस्फोटित होने वाले किसी फ़ंक्शन की पहली झलक।

\medskip
\noindent\textbf{Part I --- One station.}
(मान: $\tan 35^\circ \approx 0.700$,
$\sin 35^\circ \approx 0.574$, $\cos 35^\circ \approx 0.819$,
$\tan 1^\circ \approx 0.0175$.)
\begin{enumerate}
  \item एक बिंदु से $60$ एक प्रकाशस्तंभ के तल से मी,
        की ऊँचाई पर शीर्ष दिखाई देता है $35^\circ $ (से मापा गया
        क्षैतिज, आँख के स्तर पर)। ऊपर की चोटी कितनी ऊँची है
        आँख का स्तर?
  \item एक सड़क चिन्ह एक की घोषणा करता है $12\,\%$ ग्रेड: सड़क ऊपर उठती है
        $12$ मी प्रति $100$ क्षैतिज रन का मी. कोण क्या करता है
        सड़क क्षैतिज के साथ बनाते हैं? और कौन सा ग्रेड
        प्रतिशत होगा ए $45^\circ $ सड़क है?
  \item ए के दो न्यून कोण सही \omterm{def:g3:shapes:shapes}{त्रिकोण} हैं
        पूरक, और प्रत्येक का विपरीत पक्ष है
        दूसरे का अगल-बगल. ``सह'' पहचान निकालें:
        \[
        \sin(90^\circ - \theta) = \cos\theta,
        \qquad
        \cos(90^\circ - \theta) = \sin\theta .
        \]
  \item एक $8$ मी सीढ़ी पर झुक जाता है $60^\circ $ भूमि पर।
        के सटीक मानों का उपयोग करना \cref{ex:g9:trig:special},
        पहुँची हुई ऊँचाई और पैदल दूरी बताइए
        दीवार से --- बिल्कुल, फिर सेंटी\omterm{def:g4:fractions:def}{मीटर} तक।
  \item सत्यापित करें $\cos^2\theta + \sin^2\theta = 1$ संख्यात्मक रूप से के लिए
        $\theta = 35^\circ $ और बिल्कुल के लिए $\theta = 60^\circ $.
        ये पहचान क्यों है
        (\cref{prop:g9:trig:properties}) के लिए एक अच्छी आदत
        कैलकुलेटर के काम की जाँच कर रहे हैं?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Two stations.}
एक पर्वत का शिखर दिखाई देता है, उसका आधार अप्राप्य है। ए से
बिंदु $ A $, शिखर की ऊंचाई है $30^\circ $; चलना $200$ एम
सीधे पहाड़ की ओर, को $ B $, उत्कर्ष हो जाता है
$40^\circ $. लिखना $ h $ आँख के स्तर से ऊपर शिखर की ऊँचाई के लिए
और $ x $ से क्षैतिज दूरी के लिए $ B $ के ऊर्ध्वाधर तक
शिखर. ($\tan 30^\circ \approx 0.577$,
$\tan 40^\circ \approx 0.839$.)
\begin{enumerate}[resume]
  \item अभिव्यक्त करना $\tan 40^\circ $ और $\tan 30^\circ $ का उपयोग करते हुए $ h $,
        $ x $ और $ x + 200$: दो \omterm{def:g8:equations:def}{समीकरण} दो अज्ञात के लिए.
  \item प्रत्येक से \omterm{def:g8:equations:def}{समीकरण} क्षैतिज दूरी को व्यक्त करें
        की शर्तें $ h $, घटाना, और दो-स्टेशन प्राप्त करना
        सूत्र:
        \[
        h = \frac{200}
        {\ \dfrac{1}{\tan 30^\circ} - \dfrac{1}{\tan 40^\circ}\ } .
        \]
        गणना करना $ h $ निकटतम दस \omterm{def:g4:fractions:def}{मीटर} तक.
  \item गणना करना $ x $ साथ ही, और अपने दो उत्तरों की विवेक-जांच करें
        के विरुद्ध $30^\circ $ से देखना $ A $.
  \item एक वाक्य में: दो-स्टेशन पद्धति क्यों है?
        अपरिहार्य --- क्या एकल माप, असंभव
        यहाँ, प्रश्न 1 की एक-स्टेशन विधि होगी
        आवश्यक?
  \item एक टावर नजर आता है $30^\circ $, और कम से $60^\circ $
        चलने के बाद $200$ मैं इसकी ओर हूं. का उपयोग करते हुए
        $\tan 30^\circ = \frac{1}{\sqrt3}$ और
        $\tan 60^\circ = \sqrt3$, ऊंचाई की गणना करें
        \emph{बिल्कुल} --- उत्तर है ए \omterm{def:g9:arith:divisor}{एकाधिक} का
        $\sqrt3$ --- फिर \omterm{def:g4:fractions:def}{मीटर} तक.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- To the horizon and up the wall.}
\begin{enumerate}[resume]
  \item क्षितिज कितनी दूर है? आँखें बिछाये खड़ा है $ h $ \omterm{def:g4:fractions:def}{मीटर}
        की एक \omterm{def:g4:numbers:round}{गोल}ाकार पृथ्वी के ऊपर \omterm{def:g3:shapes:shapes}{RADIUS} $ R $, आपकी पंक्ति
        दृष्टि \omterm{def:g4:numbers:round}{गोल}े को पकड़ती है: दृष्टि \omterm{def:g6:lines:objects}{रेखा}, \omterm{def:g3:shapes:shapes}{RADIUS} को
        चराई बिंदु और \omterm{def:g3:shapes:shapes}{RADIUS} आपके पैरों के लिए फॉर्म ए
        सही \omterm{def:g3:shapes:shapes}{त्रिकोण}. पाइथागोरस के साथ दिखाएँ
        (\cref{thm:g9:trig:pythagoras}) वह दूरी $ d $ को
        क्षितिज संतुष्ट करता है $ d^2 = 2Rh + h^2 \approx 2Rh $,
        और गणना करें $ d $ आँखों के लिए $1.80$ एम
        ($ R = 6.37 \times 10^6$ एम)।
  \item ए के शीर्ष से वही प्रश्न $324$ एम टावर. (नाविकों'
        अंगूठे का नियम: किलो\omterm{def:g4:fractions:def}{मीटर} में क्षितिज लगभग है
        $3.6\sqrt{h}$ साथ $ h $ \omterm{def:g4:fractions:def}{मीटर} में --- अपने दो की जांच करें
        इसके विरुद्ध उत्तर.)
  \item आपका अंगूठा, के बारे में $2$ सेमी चौड़ा, हाथ की लंबाई पर रखा हुआ,
        के बारे में $60$ आपकी आंख से सेमी: यह किस कोण को कवर करता है?
        पूर्णिमा का चंद्रमा चारों ओर छा जाता है $0.5^\circ $: आपका कितना
        क्या अंगूठा पूरे चंद्रमा को छिपा देता है?
  \item आरामदायक सीढ़ियों में राइजर लगे हैं $17$ सेमी और के treads
        $29$ सेमी। वे किस कोण पर चढ़ते हैं? से तुलना करें
        $30^\circ $ का \cref{ex:g9:trig:special}, और एक के साथ
        खड़ी अटारी सीढ़ी पर $60^\circ $: क्या राइजर होगा
        \emph{यह} एक पर जरूरत है $29$ सेमी चलना?
  \item  स्पर्शरेखाकी दौड़: गणना (कैलकुलेटर)
        $\tan 10^\circ $, $\tan 45^\circ $, $\tan 80^\circ $,
        $\tan 89^\circ $. जैसे-जैसे कोण निकट आता है, क्या होता है
        $90^\circ$, और क्यों (आसन्न पक्ष के बारे में सोचें)? दीवार है
        ऊर्ध्वाधर, अनुपात का कोई मूल्य नहीं है, और ग्राफ़ शूट हो जाता है
        अनंत तक --- आपका पहला \emph{अनंतस्पर्शी}, एक प्राणी
        हाई स्कूल वॉल्यूम में बारीकी से अध्ययन किया।
\end{enumerate}
\end{problem}
