\chapter{भिन्न और घात}\label{ch:g9:fractions}

भिन्न और पॉवर्स गणना के रोजमर्रा के उपकरण हैं: साझा करना
मात्राएँ, अनुपातों की तुलना करना, बहुत बड़ा या बहुत छोटा लिखना
नंबर. यह अध्याय उनके साथ कंप्यूटिंग के नियमों को समेकित करता है ---
प्रत्येक मध्यवर्ती चरण को लिखा जाता है --- और परिचय दिया जाता है \omterm{def:g9:fractions:scientific}{वैज्ञानिक संकेतन}.

\section{भिन्नों के साथ गणना}

\begin{definition}[अंश]\label{def:g9:fractions:fraction}
ए \emph{अंश}\index{अंश} $\dfrac ab $ (साथ $ b \neq 0$) है
\omterm{def:g3:division:remainder}{भागफल} का $ a $ द्वारा $ b $. जब एक प्राप्त किया जाता है तो दो भिन्न बराबर होती हैं
दूसरे को गुणा करके (या विभाजित करके) \omterm{def:g4:fractions:def}{मीटर} और \omterm{def:g5:division:multiple}{भाजक} द्वारा
वही अशून्य संख्या:
\[
\frac ab = \frac{a \times k}{b \times k} \qquad (k \neq 0).
\]
\end{definition}

\begin{example}\label{ex:g9:fractions:simplify}
सरल $\dfrac{42}{56}$ क्रमशः:
\[
\frac{42}{56} = \frac{42 \div 2}{56 \div 2} = \frac{21}{28}
= \frac{21 \div 7}{28 \div 7} = \frac34 .
\]
(में \cref{ch:g9:arith} सबसे बड़ा आम \omterm{def:g5:division:multiple}{भाजक} इसे एक में करेंगे
कदम.)
\end{example}

\begin{method}[भिन्नों को जोड़ना और घटाना]\label{met:g9:fractions:add}
\begin{enumerate}
  \item दोनो डालो \omterm{def:g9:fractions:fraction}{अंश}ों एक से अधिक \emph{सामान्य \omterm{def:g5:division:multiple}{भाजक}} (एक सामान्य
        एकाधिक दोनों में से हरों);
  \item \omterm{def:g1:addition:def}{जोड़}ें या घटाएँ \omterm{def:g9:fractions:fraction}{अंश}, रखते हुए \omterm{def:g5:division:multiple}{भाजक};
  \item यदि संभव हो तो परिणाम को सरल बनाएं।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:fractions:add}
गणना करना $\dfrac56 + \dfrac34$. एक आम \omterm{def:g5:division:multiple}{भाजक} का $6$ और $4$ है
$12$:
\[
\frac56 + \frac34
= \frac{5 \times 2}{6 \times 2} + \frac{3 \times 3}{4 \times 3}
= \frac{10}{12} + \frac{9}{12}
= \frac{19}{12}.
\]
गणना करना $2 - \dfrac37$. लिखना $2$ के तौर पर \omterm{def:g9:fractions:fraction}{अंश} ऊपर $7$:
\[
2 - \frac37 = \frac{14}{7} - \frac37 = \frac{11}{7}.
\]
\end{example}

\begin{method}[भिन्नों को गुणा करना और विभाजित करना]\label{met:g9:fractions:mult}
\begin{enumerate}
  \item गुणा करना, गुणा करना \omterm{def:g9:fractions:fraction}{अंश} एक साथ और हरों
        एक साथ: $\dfrac ab \times \dfrac cd = \dfrac{ac}{bd}$
        (सरल करें \emph{पहले} जब भी संभव हो गुणा करना);
  \item भाग देना, से गुणा करना \omterm{def:g8:fractions:inverse}{श्लोक में} की \omterm{def:g5:division:multiple}{भाजक}:
        $\dfrac ab \div \dfrac cd = \dfrac ab \times \dfrac dc $.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:fractions:mult}
\[
\frac{7}{15} \times \frac{25}{14}
= \frac{7 \times 25}{15 \times 14}
= \frac{7 \times 25}{15 \times 2 \times 7}
= \frac{25}{30} = \frac56,
\]
जहां हमने सामान्य कारक द्वारा सरलीकृत किया है $7$, फिर द्वारा $5$. और ए
प्रभाग:
\[
\frac{3}{4} \div \frac{9}{8}
= \frac34 \times \frac89
= \frac{3 \times 8}{4 \times 9}
= \frac{24}{36} = \frac23 .
\]
\end{example}

\section{पॉवर्स}

\begin{definition}[शक्ति]\label{def:g9:fractions:power}
वास्तविक संख्या के लिए $ a $ और एक सकारात्मक पूर्णांक $ n $, द
\emph{शक्ति}\index{शक्ति} $ a^n $ है \omterm{def:g2:mult:def}{उत्पाद} का $ n $ कारकों के बराबर
$ a $:
\[
a^n = \underbrace{a \times a \times \dots \times a}_{n \text{ factors}} .
\]
रिवाज के सन्दर्भ मे $ a^0 = 1$ (के लिए $ a \neq 0$), और नकारात्मक घातांक निरूपित
प्रतिलोम:
\[
a^{-n} = \frac{1}{a^n}.
\]
\end{definition}

\begin{example}
$2^4 = 2 \times 2 \times 2 \times 2 = 16$; \quad
$(-3)^2 = 9$ लेकिन $-3^2 = -(3^2) = -9$ (द \omterm{def:g8:powers:def}{प्रतिपादक} के पहले बांधता है
ऋण चिह्न); \quad $10^{-3} = \frac{1}{1000} = 0.001$; \quad
$5^1 = 5$ और $5^0 = 1$.
\end{example}

\begin{theorem}[घातांक के नियम]\label{thm:g9:fractions:rules}
सभी अशून्य के लिए $ a $, $ b $ और सभी पूर्णांक $ m $, $ n $:
\[
a^m \times a^n = a^{m+n},
\qquad
\frac{a^m}{a^n} = a^{m-n},
\qquad
\left(a^m\right)^n = a^{mn},
\qquad
(ab)^n = a^n b^n .
\]
\end{theorem}

\begin{proof}
सकारात्मक के लिए घातांक, कारकों को गिनें। पहले नियम के लिए:
$ a^m \times a^n $ है $ m $ कारकों $ a $ के बाद $ n $ कारकों $ a $, कुल मिलाकर
$ m + n $ कारक. तीसरे के लिए:
$\left(a^m\right)^n $ दोहराता $ n $ का एक ब्लॉक गुना $ m $ कारक, देना
$ mn $ कारक. अन्य दो नियम समान हैं; का विस्तार \omterm{def:g1:counting:zero}{शून्य} और
नकारात्मक घातांक से जाँच की जाती है $ a^{-n} = \frac{1}{a^n}$ (और बनाता है
सम्मेलन $ a^0 = 1$ तब से, एकमात्र सुसंगत विकल्प
$ a^n \times a^0$ होना ही चाहिए $ a^{n+0} = a^n $).
\end{proof}

\begin{example}\label{ex:g9:fractions:rules}
चरण दर चरण सरल बनाएं:
\[
\frac{3^7 \times 3^{-2}}{3^4}
= \frac{3^{7 + (-2)}}{3^4}
= \frac{3^5}{3^4}
= 3^{5-4} = 3^1 = 3 .
\]
और दो अक्षरों के साथ: $(2a^3)^2 \times a = 4 a^6 \times a = 4a^7$.
\end{example}

\section{दस की शक्तियाँ और वैज्ञानिक संकेतन}

\begin{proposition}[दस की शक्तियाँ]\label{prop:g9:fractions:ten}
एक सकारात्मक पूर्णांक के लिए $ n $: $10^n $ लिखा है ``$1$ के बाद $ n $
\omterm{def:g1:counting:zero}{शून्य}'', और $10^{-n} = 0.0\dots01$ साथ $1$ में $ n $-वां दशमलव
जगह. ए को गुणा करना \omterm{def:g6:decimals:places}{दशमलव संख्या} द्वारा $10^n $ (सम्मान.\ $10^{-n}$) बदलाव
इसका \omterm{def:g4:decimals:point}{दशमलव बिंदु} $ n $ दाहिनी ओर के स्थान (सम्मान)\ बाएं)।
\end{proposition}

\begin{definition}[वैज्ञानिक संकेतन]\label{def:g9:fractions:scientific}
 \emph{वैज्ञानिक संकेतन}\index{वैज्ञानिक संकेतन} एक सकारात्मक का
\omterm{def:g6:decimals:places}{दशमलव संख्या} इसे लिखने का अनोखा तरीका है
\[
a \times 10^n,
\qquad\text{with } 1 \leq a < 10 \text{ and } n \text{ an integer.}
\]
\end{definition}

\begin{example}\label{ex:g9:fractions:scientific}
पृथ्वी-सूर्य की दूरी लगभग है $149\,600\,000$ किमी $= 1.496 \times
10^8$ किमी. पानी के अणु का आकार लगभग होता है $0.000\,000\,000\,28$ एम
$= 2.8 \times 10^{-10}$ एम। ऐसी संख्याओं की गणना करने के लिए, समूह बनाएं
दशमलव भाग और पॉवर्स अक्सर:
\[
(3 \times 10^5) \times (2.5 \times 10^{-2})
= (3 \times 2.5) \times 10^{5 + (-2)}
= 7.5 \times 10^{3}.
\]
\end{example}

\begin{method}[किसी संख्या को वैज्ञानिक संकेतन में रखना]\label{met:g9:fractions:scientific}
\begin{enumerate}
  \item चलाएं \omterm{def:g4:decimals:point}{दशमलव बिंदु} पहले \omterm{def:g1:counting:zero}{शून्य}ेतर अंक के ठीक बाद; यह
        कारक देता है $ a $ साथ $1 \leq a < 10$;
  \item बिंदु जितने स्थानों पर चला गया, उसकी संख्या गिनें: वह गिनती है $ n $,
        सकारात्मक यदि मूल संख्या थी $\geq 10$, नकारात्मक यदि यह था
        $< 1$;
  \item जाँच करना: $ a \times 10^n $ मूल संख्या को पुन: प्रस्तुत करना होगा.
\end{enumerate}
\end{method}

\section{समस्या: दशमलव को दोहराने का गुप्त कोड}

\begin{problem}[{सप्ताहान्त समस्या--- प्रत्येक भिन्न का एक दशमलव होता है जो लिखना रुकता है या दोहराता है, प्रत्येक दोहराव वाला दशमलव होता है अंश, और $0.999\ldots $ बिल्कुल बराबर है $1$}]
\label{pb:g9:fractions:1}
विभाजित करना $3$ द्वारा $11$ और अंक एक मंत्र में आते हैं:
$0.272727\ldots $ हमेशा के लिए। विभाजित करना $1$ द्वारा $4$ और अंक रुक जाते हैं
मृत: $0.25$. यह समस्या सिद्ध करती है कि ये केवल दो ही हैं
ए के लिए संभावित व्यवहार \omterm{def:g9:fractions:fraction}{अंश} --- और फिर उलट देता है
मशीन: किसी भी दोहराए जाने वाले दशमलव को देखते हुए, यह उसका पुनर्निर्माण करती है \omterm{def:g9:fractions:fraction}{अंश}
इसके पीछे छुपे हुए हैं. रास्ते में यह एक बार और हमेशा के लिए सुलझ जाता है
स्कूली गणित में समानता के बारे में सबसे अधिक तर्क-वितर्क।

\medskip
\noindent\textbf{Part I --- From \omterm{def:g9:fractions:fraction}{अंश} to decimal.}
\begin{enumerate}
  \item के दशमलव लेखन को दीर्घ विभाजन द्वारा परिकलित करें
        $\frac14$, $\frac13$, $\frac56$ और $\frac{3}{11}$.
        कौन सा स्टॉप, कौन सा रिपीट, और कौन सा रिपीटिंग ब्लॉक?
  \item बताएं कि दशमलव लेखन क्यों होता है \emph{कोई} \omterm{def:g9:fractions:fraction}{अंश}
        $\frac ab $ या तो रुकना चाहिए या अंततः दोहराना चाहिए। (में
        द्वारा लंबा विभाजन $ b $, कौन से मान क्रमिक हो सकते हैं
        \omterm{def:g3:division:remainder}{शेष} लेना? उस क्षण क्या होता है a \omterm{def:g3:division:remainder}{शेष}
        दूसरी बार दिखाई देता है?)
  \item के विभाजन को धक्का दें $1$ द्वारा $7$ खोजने के लिए काफी दूर है
        दोहराव वाला ब्लॉक. यह कितनी लंबी है, और यह लंबाई कितनी है?
        प्रश्न 2 द्वारा वादा की गई सीमा से तुलना करें?
  \item कुछ \omterm{def:g9:fractions:fraction}{अंश}ों रुकें क्योंकि उनके \omterm{def:g5:division:multiple}{भाजक} हो सकता है
        तक उड़ा दिया गया \omterm{def:g9:fractions:power}{शक्ति} अक्सर: $\frac14 =
        \frac{25}{100}$, $\frac{3}{8} = \frac{375}{1000}$.
        मानदंड स्पष्ट करें, और कोई पूर्ण-संख्या गुणक क्यों नहीं
        कभी भी पलट सकता है $3$, $6$ या $11$ में $10$, $100$,
        $1000, \dots $ (कौन सा अंक होगा \omterm{def:g2:mult:def}{उत्पाद} अंत में?)।
  \item बिना विभाजित किये क्रमबद्ध करें $\frac{9}{40}$, $\frac{5}{12}$ और
        $\frac{33}{50}$ ``स्टॉप'' और ``रिपीट'' में, फिर दें
        दोनों का दशमलव लेखन रुक जाता है।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- From repeating decimal back to the
\omterm{def:g9:fractions:fraction}{अंश}.}
\begin{enumerate}[resume]
  \item बदलाव की चाल. होने देना $ x = 0.7777\ldots $ गणना करना $10x $,
        तब $10x - x $, और निष्कर्ष निकालें $ x $ के तौर पर \omterm{def:g9:fractions:fraction}{अंश}. द्वारा जांचें
        बाँटना.
  \item होने देना $ x = 0.727272\ldots $ कौन \omterm{def:g9:fractions:power}{शक्ति} दस शिफ्टों का $ x $ द्वारा
        बिल्कुल एक दोहराव वाला ब्लॉक? इसे लिखने के लिए उपयोग करें $ x $ के तौर पर
        \omterm{def:g9:fractions:fraction}{अंश} सबसे कम शर्तों में.
  \item एक मिश्रित मामला: $ x = 0.58333\ldots $ (द $3$ अकेले
        दोहराता है)। गणना करना $1000x - 100x $ और निष्कर्ष निकालें $ x $ के तौर पर
        \omterm{def:g9:fractions:fraction}{अंश} सबसे कम शर्तों में.
  \item प्रसिद्ध एक: शिफ्ट ट्रिक को लागू करें
        $ x = 0.9999\ldots $ क्या \omterm{def:g9:fractions:fraction}{अंश} --- क्या \emph{संख्या}
        ---क्या आप पाते हैं? अपने साथ परिणाम का मिलान करें
        \omterm{ex:g1:subtraction:difference}{अंतर}्ज्ञान, के प्रेत पड़ोसी को याद करते हुए
        \cref{pb:g6:decimals:1} और सिकुड़ रहा है \omterm{def:g3:division:remainder}{शेष} का
        \cref{pb:g6:fractions:1}: है $0.999\ldots $ ``बस नीचे''
        $1$, या इसके लिए कोई अन्य नाम $1$?
  \item बदलना $2.454545\ldots $ और $0.123123123\ldots $ में
        \omterm{def:g9:fractions:fraction}{अंश}ों सबसे कम शर्तों में.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The dictionary completed.}
\begin{enumerate}[resume]
  \item प्रश्न 6-10 एक शब्दकोष का सुझाव दें: का एक \omterm{def:g6:lines:objects}{खंड} $ k $
        से दोहराए जाने वाले अंक \omterm{def:g4:decimals:point}{दशमलव बिंदु} उस ब्लॉक के बराबर है
        खत्म $ k $ नौ ($0.727272\ldots = \frac{72}{99}$). उपयोग करें
        भविष्यवाणी करने के लिए शब्दकोश $0.037037\ldots $ के तौर पर \omterm{def:g9:fractions:fraction}{अंश},
        सरल बनाएं --- और आश्चर्यचकित करें: जो आश्चर्यजनक रूप से सरल है
        \omterm{def:g9:fractions:fraction}{अंश} मंत्र ``$037$''? इसे प्रभाग द्वारा जांचें।
  \item शब्दकोश को एक बदलाव के साथ संयोजित करें: लिखें
        $0.0272727\ldots $ के तौर पर \omterm{def:g9:fractions:fraction}{अंश} सबसे कम शर्तों में.
  \item जो नंबर $0.101001000100001\ldots $ (ए $1$, तब एक
        $0$, फिर ए $1$, फिर दो $0$ एस, फिर ए $1$, फिर तीन
        $0$ s, इत्यादि) कभी नहीं रुकता। क्या यह दोहराव वाला दशमलव है?
        फिर भाग I इसके बारे में क्या कहता है --- क्या यह हो सकता है?
        \omterm{def:g9:fractions:fraction}{अंश}? (आप अभी-अभी अपनी पहली प्रमाणित मुलाकात से मिले हैं
        \emph{तर्कहीन} संख्या; सबसे प्रसिद्ध पकड़ा गया है
        में \cref{ch:g9:sqrt}.)
  \item मोटे तौर पर कितने अंक होते हैं $2^{30}$ पास होना? उपयोग
        $2^{10} = 1\,024 \approx 1.024 \times 10^3$ और यह
        \omterm{def:g8:powers:def}{प्रतिपादक} नियम (\cref{thm:g9:fractions:rules}) लिखना
        $2^{30}$ में (अनुमानित) \omterm{def:g9:fractions:scientific}{वैज्ञानिक संकेतन}, और
        निष्कर्ष. (तुलना करना \cref{exo:g9:fractions:10}.)
  \item समापन: लिखो $0.20262026\ldots$ के तौर पर \omterm{def:g9:fractions:fraction}{अंश}, और
        पूरी समस्या का दोतरफा नैतिक पक्ष बताएं: कौन सा
        दशमलव लेखन हैं \omterm{def:g9:fractions:fraction}{अंश}ों, और क्या \omterm{def:g9:fractions:fraction}{अंश}ों है
        कौन सा दशमलव लेखन?
\end{enumerate}
\end{problem}
