\chapter{रैखिक और एफ़िन फ़ंक्शंस}\label{ch:g9:linfunc}

``तीन किलोग्राम की कीमत एक किलोग्राम से तीन गुना अधिक है'': यानी
समानता, दो मात्राओं को संबंधित करने का सबसे सरल तरीका। \omterm{def:g9:linfunc:linear}{रैखिक
कार्य} हैं समानता कार्यों की भाषा में; एफ़िन
कार्य एक निश्चित आरंभिक राशि \omterm{def:g1:addition:def}{जोड़}ें. उनके ग्राफ़ सीधी रेखाएँ हैं,
और उन पंक्तियों को पढ़ना ही वह कौशल है जो यह अध्याय विकसित करता है।

\section{आनुपातिकता और रैखिक कार्य}

\begin{definition}[रैखिक कार्य]\label{def:g9:linfunc:linear}
ए \emph{रैखिक कार्य}\index{रैखिक कार्य} प्रत्येक संख्या को गुणा करता है
एक निश्चित संख्या से $a $:
\[
f(x) = a x .
\]
जो नंबर $ a $ है \emph{गुणक} का $ f $. दो मात्राएँ हैं
\emph{आनुपातिक}\index{आनुपातिक} बिल्कुल जब कोई एक रैखिक होता है
दूसरे का कार्य.
\end{definition}

\begin{example}\label{ex:g9:linfunc:linear}
अगर सेब की कीमत $3$ प्रति किलोग्राम की कीमत $ x $ किलोग्राम है
$ f(x) = 3x $: दोगुनी कीमत पर खरीदने पर लागत दोगुनी और कीमत दोगुनी
$2.5$ किलो है $ f(2.5) = 7.5$.

इसके विपरीत, यदि ए \omterm{def:g9:linfunc:linear}{रैखिक कार्य} संतुष्ट $ f(4) = 10$, तब
$ a = \frac{10}{4} = 2.5$ और $ f(x) = 2.5x $: \emph{एक मान एक निर्धारित करता है
\omterm{def:g9:linfunc:linear}{रैखिक कार्य} पूरी तरह।}
\end{example}

\begin{proposition}[एक रैखिक फलन का ग्राफ़]\label{prop:g9:linfunc:lineargraph}
का ग्राफ $ f(x) = ax $ एक सीधी \omterm{def:g6:lines:objects}{रेखा} है \emph{उत्पत्ति के माध्यम से}.
इसके विपरीत, मूल बिंदु से होकर जाने वाली प्रत्येक गैर-ऊर्ध्वाधर \omterm{def:g6:lines:objects}{रेखा} a का ग्राफ़ होती है
\omterm{def:g9:linfunc:linear}{रैखिक कार्य}.
\end{proposition}
\admitted

\section{एफ़िन फ़ंक्शन}

\begin{definition}[एफ़िन फ़ंक्शन]\label{def:g9:linfunc:affine}
एक \emph{एफ़िन फ़ंक्शन}\index{एफ़िन फ़ंक्शन} रूप है
\[
f(x) = a x + b ,
\]
कहाँ $ a $ और $ b $ निश्चित संख्याएँ हैं. \omterm{def:g9:linfunc:linear}{रैखिक कार्य} वि\omterm{def:g3:division:remainder}{शेष} हैं
मामला $ b = 0$; मामला निरंतर कार्य करता है $ a = 0$.
\end{definition}

\begin{example}
एक जिम पंजीकरण शुल्क लेता है $20$, तब $5$ प्रति विज़िट: कुल
की लागत $ x $ दौरा है $ f(x) = 5x + 20$. लागत है \emph{नहीं}
\omterm{def:g9:linfunc:linear}{आनुपातिक} विज़िट की संख्या के अनुसार (10 विज़िट की लागत दोगुनी नहीं है
5 के रूप में), लेकिन प्रत्येक अतिरिक्त विज़िट में समान राशि जुड़ती है, $5$.
\end{example}

\begin{proposition}[एक एफ़िन फ़ंक्शन का ग्राफ़]\label{prop:g9:linfunc:affinegraph}
का ग्राफ $ f(x) = ax + b $ एक सीधी \omterm{def:g6:lines:objects}{रेखा} है:
\begin{itemize}
  \item $ b = f(0)$ है \emph{$ y $संवाद}: \omterm{def:g6:lines:objects}{रेखा} पार करती है
        ऊर्ध्वाधर अक्ष पर $(0, b)$;
  \item $ a $ है \emph{ढलान}\index{ढलान}: एक इकाई को आगे बढ़ाना
        सही चाल $ a $ इकाइयाँ ऊपर (यदि नीचे) $ a < 0$); फ़ंक्शन है
        जब बढ़ रहा है $ a > 0$, घट रहा है जब $ a < 0$;
  \item किन्हीं दो इनपुट के लिए $ u \neq v $,
        $ a = \dfrac{f(v) - f(u)}{v - u}$.
\end{itemize}
\end{proposition}

\begin{proof}[अंतिम बिंदु का प्रमाण]
$ f(v) - f(u) = (av + b) - (au + b) = a(v - u)$; द्वारा \omterm{def:g9:arith:divisor}{विभाजित} करें $ v - u $. द
बाकी को इस स्तर पर प्रवेश दिया जाता है (हाई स्कूल वॉल्यूम पूर्ण देता है)।
रेखाओं और उनके उपचार \omterm{def:g8:equations:def}{समीकरण}).
\end{proof}

$ a = 1$, $ b = 1$).}

\begin{method}[दो मानों से एक एफ़िन फ़ंक्शन ढूँढना]\label{met:g9:linfunc:findaffine}
दिया गया $ f(u)$ और $ f(v)$ साथ $ u \neq v $:
\begin{enumerate}
  \item की गणना करें \omterm{prop:g9:linfunc:affinegraph}{ढलान} $ a = \dfrac{f(v) - f(u)}{v - u}$;
  \item दो ज्ञात मानों में से एक को प्रतिस्थापित करें $ f(x) = ax + b $ ढूँढ़ने के लिए $ b $;
  \item अन्य मान से जांचें.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:linfunc:findaffine}
खोजें \omterm{def:g9:linfunc:affine}{एफ़िन फ़ंक्शन} साथ $ f(2) = 7$ और $ f(5) = 16$.
\[
a = \frac{16 - 7}{5 - 2} = \frac93 = 3 ,
\]
तब $ f(2) = 3 \times 2 + b = 7$ देता है $ b = 1$: $ f(x) = 3x + 1$. जाँच करना:
$ f(5) = 15 + 1 = 16$.
\end{example}

\section{प्रतिशत}

\begin{proposition}[प्रतिशत परिवर्तन]\label{prop:g9:linfunc:percent}
द्वारा एक मात्रा बढ़ाना $ t\,\%$ से गुणा करता है
$1 + \dfrac{t}{100}$; इसे कम करके $ t\,\%$ से गुणा करता है
$1 - \dfrac{t}{100}$. प्रतिशत परिवर्तन हैं \omterm{def:g9:linfunc:linear}{रैखिक कार्य}.
\end{proposition}

\begin{proof}
की बढ़ती $ x $ द्वारा $ t\,\%$ जोड़ने का मतलब है $\frac{t}{100}\,x $:
\[
x + \frac{t}{100}\,x = \left(1 + \frac{t}{100}\right) x .
\]
कमी के लिए ऋण चिह्न के साथ समान गणना।
\end{proof}

\begin{example}\label{ex:g9:linfunc:percent}
की एक कीमत $80$ से बढ़ जाता है $15\%$: नई कीमत
$80 \times 1.15 = 92$. की एक कीमत $60$ से घट जाती है $30\%$:
$60 \times 0.7 = 42$.

\emph{चेनिंग परिवर्तन कारकों को कई गुना बढ़ा देता है।} ए $20\%$ वृद्धि
इसके बाद ए $20\%$ कमी देता है
$1.2 \times 0.8 = 0.96$: ए $4\%$ \emph{घटाना} कुल मिलाकर, वापसी नहीं
शुरुआत!
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g9:linfunc:1}
इन कार्यों में से कौन रैखिक हैं? एफ़िन? कोई भी नहीं?
\[
f(x) = 4x, \qquad
g(x) = 3x - 5, \qquad
h(x) = x^2, \qquad
k(x) = -\tfrac x2, \qquad
m(x) = 7 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:linfunc:2}
होने देना $ f(x) = 2x - 3$. गणना करना $ f(0)$, $ f(4)$, $ f(-1)$, और संख्या ज्ञात करें
$ x $ ऐसा है कि $ f(x) = 9$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:linfunc:3}
ए \omterm{def:g9:linfunc:linear}{रैखिक कार्य} संतुष्ट $ f(6) = 15$. तो इसका गुणांक ज्ञात कीजिए
गणना करें $ f(10)$ और $ f(-2)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:linfunc:4}
उसी समन्वय प्रणाली पर ग्राफ़ बनाएं $ f(x) = 2x $,
$ g(x) = 2x + 3$ और $ h(x) = -x + 4$. के ग्राफ़ के बारे में क्या कहा जा सकता है
$ f $ और $ g $?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:linfunc:5}
खोजें \omterm{def:g9:linfunc:affine}{एफ़िन फ़ंक्शन} ऐसा है कि $ f(1) = 5$ और $ f(3) = 11$; फिर
एक ऐसा कि $ g(0) = 4$ और $ g(2) = 0$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:linfunc:6}
गणना करें: ए के बाद कीमत $25\%$ पर वृद्धि $120$; एक के बाद कीमत
$40\%$ पर छूट $65$; यदि किसी वस्तु की कीमत है तो मूल कीमत $69$
बाद एक $25\%$ छूट।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:linfunc:7}
एक फ़ोन प्लान A की लागत होती है $10$ प्रति माह प्लस $0.05$ प्रति मिनट कॉल; योजना
बी लागत $25$ प्रति माह असीमित कॉल के साथ।
\begin{enumerate}
  \item योजना ए की मासिक लागत को संख्या के फलन के रूप में व्यक्त करें
        $ x $ मिनटों का.
  \item प्लान बी प्रति माह कितने मिनट से सस्ता है?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:linfunc:8}
की एक आबादी $2000$ बैक्टीरिया बढ़ जाते हैं $10\%$ हर घंटे।
\begin{enumerate}
  \item एक घंटे के बाद कितने बैक्टीरिया होते हैं? दो घंटे बाद?
  \item बताएं कि दो घंटे बाद उत्तर क्यों नहीं है $2400$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:linfunc:9}
एक का ग्राफ \omterm{def:g9:linfunc:affine}{एफ़िन फ़ंक्शन} बिन्दुओं से होकर गुजरता है $(2, 3)$ और
$(6, 1)$.
\begin{enumerate}
  \item इसकी गणना करें \omterm{prop:g9:linfunc:affinegraph}{ढलान}. क्या फलन बढ़ रहा है या घट रहा है?
  \item इसकी अभिव्यक्ति दीजिए और उस इनपुट की गणना कीजिए जिसके लिए आउटपुट है
        है $0$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g9:linfunc:10}
एक दुकान सबसे पहले कीमत बढ़ाती है $ t\,\%$, फिर नई कीमत कम कर देता है
$ t\,\%$.
\begin{enumerate}
  \item दिखाएँ कि अंतिम कीमत मूल से गुणा के बराबर है
        $1 - \dfrac{t^2}{10000}$.
  \item निष्कर्ष निकालें कि अंतिम कीमत हमेशा होती है \emph{निचला} की तुलना में
        मूल (के लिए) $0 < t \leq 100$), और कुल प्रतिशत ज्ञात करें
        के लिए बदलें $ t = 10$.
\end{enumerate}
\end{exercise}

\section{समस्या: दो थर्मामीटर}

\begin{problem}[{सप्ताहांत समस्या---सेल्सियस, फारेनहाइट और दैनिक जीवन के एफ़िन कार्य: एक तापमान पर समान पढ़ता है दोनों तराजू, और निरंतर ढलान सीधेपन के हस्ताक्षर हैं}]
\label{pb:g9:linfunc:1}
अमेरिकी सुनते हैं ``$68$ डिग्री'' और हल्के कपड़े पहनें; यूरोपीय लोग सुनते हैं
यह और एक कोट ले लो. विश्व के दो थर्मा\omterm{def:g4:fractions:def}{मीटर} आपस में जुड़े हुए हैं
एक द्वारा \omterm{def:g9:linfunc:affine}{एफ़िन फ़ंक्शन} --- और इसे दो तथ्यों से निर्मित करना है
बिल्कुल का कौशल \cref{met:g9:linfunc:findaffine}. यह
समस्या रूपांतरण का निर्माण करती है, भयानक तापमान का पता लगाती है
जो दो तराजू सहमत हूँ, फिर टैक्सी, क्रिकेट और पढ़ता है
एक ही \omterm{def:g1:addition:def}{जोड़}ी \omterm{def:g7:areas:prism}{चश्मे} के साथ मोमबत्तियाँ, और परीक्षण के साथ समाप्त होता है
जो सीधेपन को ही पहचान लेता है।

\medskip
\noindent\textbf{Part I --- Building the conversion.}
फ़ारेनहाइट को दो तथ्य पिन करते हैं \omterm{def:g7:prop:scale}{पैमाना}: पानी जम जाता है
$32\,^\circ $ एफ और उबलता है $212\,^\circ $ मोटा $0\,^\circ $ सी और
$100\,^\circ $ सी)।
\begin{enumerate}
  \item खोजें \omterm{def:g9:linfunc:affine}{एफ़िन फ़ंक्शन} $F = f(C)$ सेल्सियस परिवर्तित करना
        फ़ारेनहाइट में (\cref{met:g9:linfunc:findaffine} के साथ
        मूल्य $ f(0) = 32$ और $ f(100) = 212$).
  \item फ़ारेनहाइट में कनवर्ट करें: एक हल्का $20\,^\circ $ सी दिन; द
        शरीर का $37\,^\circ $ सी; एक ठंड $-10\,^\circ $ सी।
  \item के लिए हल करें $C $ रिटर्न रूपांतरण बनाने और उसका उपयोग करने के लिए
        पर $50\,^\circ $ एफ और आगे $98.6\,^\circ $ एफ।
  \item है $f $ ए \emph{रेखीय} समारोह
        (\cref{def:g9:linfunc:linear})? बताए गए संकेतों का परीक्षण करें:
        क्या है $ f(0)$, और सेल्सियस रीडिंग को दोगुना कर देता है
        फ़ारेनहाइट से दोगुना? सही शब्द के साथ समापन करें
        (\cref{def:g9:linfunc:affine}).
  \item क्लासिक पहेली: क्या कोई तापमान है जो पढ़ता है
        \emph{वही संख्या} दोनों पर तराजू? हल करना
        $ f(x) = x $, और वर्णन करें कि आपके समाधान का क्या अर्थ है
        का ग्राफ $ f $ (वहां ग्राफ़ किस \omterm{def:g6:lines:objects}{रेखा} को काटता है?)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- \omterm{def:g9:linfunc:affine}{एफ़िन फ़ंक्शन} in the wild.}
\begin{enumerate}[resume]
  \item टैक्सी ए शुल्क $3$ यूरो प्लस $1.50$ प्रति किमी; टैक्सी बी
        एक फ्लैट चार्ज करता है $2$ यूरो प्रति किमी. दोनों की कीमत लिखो
        कार्य करता है और उनकी तुलना करता है $4$ किमी की सवारी और एक
        $8$ किमी की सवारी.
  \item खोजें ब्रेक - ईवन वह दूरी जिस पर दोनों टैक्सियों की लागत है
        वही, और एक ग्राफ़ पर स्थिति का वर्णन करें (दो)।
        पंक्तियाँ --- क्रॉसिंग पॉइंट पर क्या होता है, उससे पहले,
        इसके बाद?)
  \item प्रकृतिवादियों के अंगूठे का नियम: एक क्रिकेट चहचहाता है
        $ N = 7T - 30$ तापमान पर प्रति मिनट कई बार
        $ T\,^\circ $सी. ए पर कितनी चहचहाहट $20\,^\circ $ सी
        शाम? आप पर भरोसा $82$ एक मिनट में चहकती है: क्या
        क्रिकेट तापमान की घोषणा करता है?
  \item एक फोन खरीदा $600$ यूरो का मूल्य कम हो जाता है $15$ यूरो प्रति
        महीना: मान फ़ंक्शन लिखें $ v(t)$, पता लगाएं कि कब
        फोन लायक है $240$ यूरो, और जब मॉडल यह कहता है
        इसका कोई मूल्य नहीं है. क्या सूत्र का अब भी कोई मतलब है?
        उस तारीख के बाद?
  \item प्रतिशत चालें हैं \omterm{def:g9:linfunc:linear}{रैखिक कार्य}: $+25\,\%$ है
        से गुणा $1.25$, और $-20\,\%$ द्वारा $0.8$
        (\cref{prop:g9:linfunc:percent}). उन्हें लिखें: क्या
        करता है ए $+25\,\%$ वृद्धि के बाद ए $-20\,\%$ कट करो
        एक कीमत? छोटे चमत्कार की व्याख्या करें और तुलना करें
        \cref{exo:g9:linfunc:10}.
\end{enumerate}

\noindent\textbf{Part III --- The \omterm{prop:g9:linfunc:affinegraph}{ढलान} is everything.}
\begin{enumerate}[resume]
  \item एक मोमबत्ती मापती है $20$ सेमी जब जलाया जाता है और $17$ सेमी \omterm{def:g3:division:half}{आधा} एक
        घंटे बाद. यह मानते हुए कि एफाइन पिघल रहा है, खोजें $ h(t)$ (ऊंचाई
        सेमी में, $ t $ मिनटों में), और भविष्यवाणी करें कि मोमबत्ती कब बुझती है।
  \item दो साइकिल चालक एक ही सड़क पर चलते हैं: पहले की दूरी है
        $ d_1(t) = 24t $ (किमी के बाद $ t $ घंटे), दूसरे का है
        $ d_2(t) = 10 + 18t $. किसने बढ़त के साथ शुरुआत की, किसने
        तेजी से चलती है, और किस समय और किलो\omterm{def:g4:fractions:def}{मीटर} पर चलती है
        पहले पकड़ो दूसरा पकड़ो?
  \item होने देना $ f(x) = 2x + 3$ और $ g(x) = -0.5x + 7$. गणना करना
        $ f(x) + g(x)$ और दोबारा जांचें कि यह ठीक है या नहीं। क्या है
        \omterm{prop:g9:linfunc:affinegraph}{ढलान} और ए का अवरोधन \omterm{def:g1:addition:def}{जोड़}, सामान्य तौर पर?
  \item पूरी चर्चा करें \omterm{def:g8:equations:def}{समीकरण} $ mx + p = 0$: कितने
        समाधान जब $ m \neq 0$ (और कौन सा)? कब $ m = 0$
        और $ p \neq 0$? कब $ m = 0$ और $ p = 0$? (हर मामला
        एक ग्राफ़ कहानी है: की एक पंक्ति कहां है \omterm{prop:g9:linfunc:affinegraph}{ढलान} $ m $ पार
        क्षैतिज अक्ष?)
  \item संरेखण परीक्षण: बिंदु हैं $(1, 5)$, $(3, 9)$,
        $(7, 17)$ एक सीधी \omterm{def:g6:lines:objects}{रेखा} पर? हैं $(1, 5)$, $(3, 9)$,
        $(6, 16)$? \omterm{def:g1:addition:def}{जोड़}ियों और के बीच परिवर्तन की दरों की गणना करें
        निर्णय करें. नैतिक बताएं: ए \emph{की स्थिर दर
        परिवर्तन} का हस्ताक्षर है एफ़िन फ़ंक्शंस --- घुमावदार
        ग्राफ़ अपनी दर और माप बदलते हैं \emph{वह} एक है
        हाई स्कूल वॉल्यूम के लिए कहानी।
\end{enumerate}
\end{problem}
