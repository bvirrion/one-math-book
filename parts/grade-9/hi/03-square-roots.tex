\chapter{वर्गमूल}\label{ch:g9:sqrt}

भुजा वाले वर्ग का विकर्ण कितना लंबा होता है $1$? पाइथागोरस प्रमेय
उत्तर $\sqrt 2$, एक संख्या जिसका वर्ग है $2$ --- और जो निकला
एक नहीं होना \omterm{def:g9:fractions:fraction}{अंश}. यह अध्याय परिभाषित करता है \omterm{def:g9:sqrt:def}{वर्गमूल}, स्थापित करता है
उनके साथ कंप्यूटिंग के नियम, और अभिव्यक्तियों को सरल बनाना सिखाता है
जैसे $\sqrt{75}$.

\section{परिभाषा और प्रथम गुण}

\begin{definition}[वर्गमूल]\label{def:g9:sqrt:def}
होने देना $ a \geq 0$. \emph{वर्गमूल}\index{वर्गमूल} का $ a $, लिखा हुआ
$\sqrt a $, अद्वितीय है \emph{गैर नकारात्मक} वह संख्या जिसका वर्ग है $ a $:
\[
\sqrt a \geq 0
\qquad\text{and}\qquad
\left(\sqrt a\right)^2 = a .
\]
ऋणात्मक संख्याओं का कोई वर्गमूल नहीं होता, क्योंकि प्रत्येक वर्ग अऋणात्मक होता है।
\end{definition}

\begin{example}
$\sqrt{49} = 7$, $\sqrt 0 = 0$, $\sqrt 1 = 1$,
$\sqrt{2.25} = 1.5$. पहला \omterm{def:g9:sqrt:def}{वर्गमूल} दिल से जानने के लिए वे हैं
द \emph{पूर्ण वर्ग}: $1, 4, 9, 16, 25, 36, 49, 64, 81, 100, 121,
144$.
\end{example}

कभी न दोहराएँ.}

\begin{proposition}[एक वर्ग का वर्गमूल]\label{prop:g9:sqrt:square}
प्रत्येक वास्तविक संख्या के लिए $x $ (सकारात्मक या नहीं):
\[
\sqrt{x^2} = \abs{x} =
\begin{cases}
x & \text{if } x \geq 0,\\
-x & \text{if } x < 0 .
\end{cases}
\]
\end{proposition}

\begin{proof}
जो नंबर $\abs{x}$ अऋणात्मक है और इसका वर्ग है $ x^2$ (एक संख्या और
इसके विपरीत का वर्ग समान है)। अद्वितीय अऋणात्मक संख्या होना
वर्ग का $ x^2$, यह है $\sqrt{x^2}$.
\end{proof}

\begin{example}
$\sqrt{(-5)^2} = \sqrt{25} = 5 = \abs{-5}$: द \omterm{def:g9:sqrt:def}{वर्गमूल} ``भूल जाता है''
संकेत, यह इसे पुनर्स्थापित नहीं करता है। वि\omterm{def:g3:division:remainder}{शेष} रूप से $\sqrt{x^2} = x $ है
\emph{गलत} नकारात्मक के लिए $ x $.
\end{example}

\section{वर्गमूलों के गुणनफल और भागफल}

\begin{theorem}[गुणा और भाग के नियम]\label{thm:g9:sqrt:rules}
सभी के लिए $ a \geq 0$ और $ b \geq 0$:
\[
\sqrt{a b} = \sqrt a \times \sqrt b,
\qquad\text{and for } b > 0:\quad
\sqrt{\frac ab} = \frac{\sqrt a}{\sqrt b} .
\]
\end{theorem}

\begin{proof}
जो नंबर $\sqrt a \times \sqrt b $ गैरनकारात्मक है (\omterm{def:g2:mult:def}{उत्पाद} का
अऋणात्मक संख्याएँ), और इसका वर्ग है
\[
\left(\sqrt a \times \sqrt b\right)^2
= \left(\sqrt a\right)^2 \times \left(\sqrt b\right)^2
= a b .
\]
उस अऋणात्मक संख्या की विशिष्टता से जिसका वर्ग है $ ab $, यह बराबर है
$\sqrt{ab}$. \omterm{def:g3:division:remainder}{भागफल} नियम वैसे ही सिद्ध होता है.
\end{proof}

\begin{remark}[रकम के लिए ऐसा कोई नियम नहीं!]
$\sqrt{a + b}$ है \emph{नहीं} $\sqrt a + \sqrt b $ सामान्य तौर पर:
\[
\sqrt{9 + 16} = \sqrt{25} = 5,
\qquad\text{but}\qquad
\sqrt 9 + \sqrt{16} = 3 + 4 = 7 .
\]
\end{remark}

\begin{method}[सरल बनाना $\sqrt n $]\label{met:g9:sqrt:simplify}
को सरल बनाने के लिए \omterm{def:g9:sqrt:def}{वर्गमूल} एक पूर्णांक का:
\begin{enumerate}
  \item सबसे बड़ा पूर्ण वर्ग विभाजन ज्ञात कीजिए $ n $ (कारक $ n $ यदि ज़रूरत हो तो);
  \item के साथ \omterm{def:g9:arith:divisor}{विभाजित} \omterm{def:g2:mult:def}{उत्पाद} नियम बनाएं और वर्ग निकालें:
        $\sqrt{k^2 m} = k\sqrt m $;
  \item जाँच करें कि कोई पूर्ण वर्ग नहीं है \omterm{def:g9:arith:divisor}{विभाजित} के नीचे क्या बचा है
        जड़.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:sqrt:simplify}
सरल $\sqrt{75}$: तब से $75 = 25 \times 3$,
\[
\sqrt{75} = \sqrt{25 \times 3} = \sqrt{25} \times \sqrt 3 = 5\sqrt3 .
\]
सरल $\sqrt{72}$: तब से $72 = 36 \times 2$,
$\sqrt{72} = 6\sqrt2$. और ए \omterm{def:g3:division:remainder}{भागफल}:
$\sqrt{\dfrac{49}{16}} = \dfrac{\sqrt{49}}{\sqrt{16}} = \dfrac74$.
\end{example}

\begin{example}[वर्गमूल जोड़ना]\label{ex:g9:sqrt:add}
रकम का \omterm{def:g9:sqrt:def}{वर्गमूल} केवल तभी सरल करें जब जड़ें हों \emph{एक जैसे}.
गणना करना $\sqrt{12} + \sqrt{27} - \sqrt{48}$, पहले प्रत्येक पद को सरल बनाना:
\begin{align*}
\sqrt{12} &= \sqrt{4 \times 3} = 2\sqrt3, \\
\sqrt{27} &= \sqrt{9 \times 3} = 3\sqrt3, \\
\sqrt{48} &= \sqrt{16 \times 3} = 4\sqrt3,
\end{align*}
इतना \omterm{def:g1:addition:def}{जोड़} है $2\sqrt3 + 3\sqrt3 - 4\sqrt3 = (2 + 3 - 4)\sqrt3 = \sqrt3$.
\end{example}

\begin{example}[वर्गमूलों के साथ विस्तार]\label{ex:g9:sqrt:expand}
बीजगणित की सर्वसमिकाएँ लागू होती हैं \omterm{def:g9:sqrt:def}{वर्गमूल}. बढ़ाना
$\left(\sqrt5 + 2\right)^2$ साथ $(a+b)^2 = a^2 + 2ab + b^2$:
\[
\left(\sqrt5 + 2\right)^2 = 5 + 2 \times 2\sqrt5 + 4 = 9 + 4\sqrt5 .
\]
और तीसरी पहचान के साथ, $(a+b)(a-b) = a^2 - b^2$:
\[
\left(\sqrt7 + \sqrt3\right)\left(\sqrt7 - \sqrt3\right) = 7 - 3 = 4 :
\]
 \omterm{def:g2:mult:def}{उत्पाद} दो अपरिमेय संख्याओं का एक पूर्णांक हो सकता है।
\end{example}

\section{समीकरण \texorpdfstring{$ x^2 = a $}{x2 = a}}

\begin{theorem}[हल $ x^2 = a $]\label{thm:g9:sqrt:equation}
\begin{enumerate}
  \item अगर $ a > 0$, द \omterm{def:g8:equations:def}{समीकरण} $ x^2 = a $ इसके ठीक दो समाधान हैं:
        $\sqrt a $ और $-\sqrt a $;
  \item अगर $ a = 0$, एकमात्र समाधान है $0$;
  \item अगर $ a < 0$, कोई समाधान नहीं है.
\end{enumerate}
\end{theorem}

\begin{proof}
पुनर्लेखन $ x^2 = a $ जैसा $ x^2 - \left(\sqrt a\right)^2 = 0$ कब $ a \geq 0$,
और एक कारक के रूप में \omterm{ex:g1:subtraction:difference}{अंतर} वर्गों का:
\[
\left(x - \sqrt a\right)\left(x + \sqrt a\right) = 0 ,
\]
इसलिए $ x = \sqrt a $ या $ x = -\sqrt a $ (ये मेल खाते हैं जब $ a = 0$). कब
$ a < 0$ तब से कोई समाधान नहीं है $ x^2 \geq 0 > a $ हरएक के लिए $ x $.
\end{proof}

$ y = 3$ इसे दो भुजाओं पर काटता है $\pm\sqrt3$.}
\end{omfigure}

\begin{example}
$ x^2 = 16$: समाधान $4$ और $-4$. \quad
$ x^2 = 5$: समाधान $\sqrt5$ और $-\sqrt5$ (सटीक मान; $\sqrt5
\approx 2.236$). \quad
$ x^2 = -9$: कोई समाधान नहीं. \quad
$3x^2 = 21$: पहले 3 से भाग दें, $ x^2 = 7$, समाधान $\pm\sqrt7$.
\end{example}

\section{समस्या: वह संख्या जो भिन्न नहीं है}

\begin{problem}[{सप्ताहांत समस्या --- इसका पौराणिक प्रमाण $\sqrt2$ तर्कहीन है, इसके कई चचेरे भाई हैं, और बगुला अलौकिक है इसका अनुमान लगाने के लिए अच्छा नुस्खा}]
\label{pb:g9:sqrt:1}
एक इकाई वर्ग का विकर्ण बिल्कुल वास्तविक लंबाई है --- आप
इसे अंदर खींच लिया \cref{ex:g8:pythagoras:diagonal} --- और फिर भी, जैसा कि
पाइथागोरस ने लगभग पच्चीस की खोज की जिससे वे भयभीत हो गए
सदियों पहले, \emph{नहीं \omterm{def:g9:fractions:fraction}{अंश} जो भी} इसे मापता है. किंवदंती
दावा है कि इस खोज की सज़ा डूबने से दी गई थी। यह समस्या चलती है
आप अमर प्रमाण के माध्यम से (चार प्रश्न और यह आपका है
जीवन के लिए), पीड़ितों को गुणा करता है, और नुस्खा के साथ समाप्त होता है
इंजीनियरों ने दो हज़ार वर्षों तक अदम्य को वश में करने के लिए प्रयोग किया
संख्या.

\medskip
\noindent\textbf{Part I --- The proof.}
\begin{enumerate}
  \item पहले इसका पता लगाएं: इसकी जांच करें
        $1.4 < \sqrt2 < 1.5$, फिर उस $1.41 < \sqrt2 < 1.42$,
        सीमाएँ चुकता करके. एक और अंक: जिसके बीच में
        तीन-दशमलव संख्याएँ करती हैं $\sqrt2$ झूठ?
  \item अब मान लीजिए --- विरोधाभास के लिए --- वह
        $\sqrt2 = \frac ab $ कुछ के लिए \omterm{def:g9:fractions:fraction}{अंश} \emph{सबसे कम में
        शर्तें} (\cref{met:g9:arith:simplify}). दोनों तरफ चौकोर
        और दिखाओ $ a^2 = 2 b^2$. की समता क्या है
        $ a^2$?
  \item के समता तथ्य \cref{pb:g8:pythagoras:1} कहना: अजीब
        नंबर पास होना \omterm{def:g3:numbers:evenodd}{विषम} वर्ग। उसका निष्कर्ष निकालो $ a $ है \omterm{def:g3:numbers:evenodd}{यहां तक की}, लिखना
        $ a = 2k $, स्थानापन्न --- और वह दिखाओ $ b $ होना ही चाहिए \omterm{def:g3:numbers:evenodd}{यहां तक की}
        बहुत।
  \item विरोधाभास कहां है? निष्कर्ष निकालिए और बताइए
        प्रमेय पूर्णतः: \emph{$\sqrt2$ तर्कहीन है --- यह है
        नहीं \omterm{def:g3:division:remainder}{भागफल} पूर्ण संख्याओं का}.
  \item सबूत एक बार, विवेकपूर्वक, शब्दों पर झुक गया ``अंदर
        सबसे कम शर्तें ''। उस सटीक चरण को इंगित करें जो होगा
        उनके बिना पतन, और समझाएं कि निम्नतम क्यों मान रहे हैं
        शर्तें पहले स्थान पर वैध थीं।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The victims multiply.}
\begin{enumerate}[resume]
  \item अभाज्य गुणनखंडों के माध्यम से दूसरी प्रमाण शैली
        (\cref{thm:g9:arith:factorization}): में $ a^2 = 3b^2$,
        की समता की तुलना करें \emph{\omterm{def:g8:powers:def}{प्रतिपादक} का $3$} प्रत्येक पर
        पक्ष (\cref{pb:g9:arith:1}: वर्ग ले जाते हैं \omterm{def:g3:numbers:evenodd}{यहां तक की}
        घातांक). समाप्त करें कि $\sqrt3$ तर्कहीन है.
  \item वही चलाओ \omterm{def:g8:powers:def}{प्रतिपादक} पर तर्क $ a^2 = n b^2$ एक के लिए
        सामान्य पूर्णांक $ n $: जिसके लिए $ n $ क्या यह उत्पादन करता है?
        एक विरोधाभास, और यह किसके लिए विफल होता है? बताएं
        पूरा परिणाम: $\sqrt n $ बिल्कुल अतार्किक है
        कब\,\dots
  \item शील्ड लेम्मा साबित करें: एक गै\omterm{def:g1:counting:zero}{र-शून्य} तर्कसंगत समय
        अतार्किक अतार्किक है. (कल्पना करना $ r x = q $ साथ $ r, q $
        तर्कसंगत, $ r \neq 0$, और के लिए हल करें $ x $.) उसका निष्कर्ष निकालें
        $2\sqrt2$ और $\frac{\sqrt2}{2}$ तर्कहीन हैं.
  \item साबित करें कि $1 + \sqrt2$ तर्कहीन है. फिर सुंदर
        एक: मान लीजिए $ s = \sqrt2 + \sqrt3$ तर्कसंगत थे,
        गणना करें $ s^2$, अलग करना $\sqrt6$, और खोजें
        विरोधाभास.
  \item उत्साह को कम करें: दो अपरिमेय संख्याएँ दें जिनके
        \emph{\omterm{def:g1:addition:def}{जोड़}} तर्कसंगत है, और दो जिसका \emph{\omterm{def:g2:mult:def}{उत्पाद}} है
        तर्कसंगत. (अतार्किकता अंकगणित द्वारा संरक्षित नहीं है
        --- प्रत्येक मामले को अपने स्वयं के प्रमाण की आवश्यकता होती है।)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Heron's recipe.}
कैलकुलेटर से दो हजार साल पहले, अलेक्जेंड्रिया का बगुला
अनुमानित $\sqrt2$ इस कदर: \emph{अनुमान $ x $; प्रतिस्थापित करें
के औसत से अनुमान लगाएं $ x $ और $\frac2x $; दोहराना।}
\begin{enumerate}[resume]
  \item अनुमान से शुरू करें $ x = 1$ और अगले दो की गणना करें
        अनुमान बिल्कुल सटीक है \omterm{def:g9:fractions:fraction}{अंश}ों.
  \item तीसरे अनुमान की फिर से सटीक गणना करें \omterm{def:g9:fractions:fraction}{अंश}, और
        इसका दशमलव मान. प्रश्न 1 से तुलना करें: कितने
        का दशमलव $\sqrt2$ क्या पहले से ही सही हैं?
  \item रेसिपी के पीछे का विचार: एक आयत \omterm{def:g6:measure:area}{क्षेत्र} $2$ के साथ
        एक तरफ $ x $ दूसरा पक्ष है $\frac2x $. बताते हैं कि
        $\sqrt2$ हमेशा झूठ बोलता है \emph{बीच में} $ x $ और $\frac2x $
        (दो मामलों पर विचार करें $ x^2 > 2$ और $ x^2 < 2$), तो
        दोनों पक्षों का औसत आयत को एक ओर खींचता है
        वर्ग.
  \item अनुमान $\frac32$, $\frac{17}{12}$,
        $\frac{577}{408}$ एक रत्न छिपाएँ: गणना करें $3^2 - 2 \times
        2^2$, तब $17^2 - 2 \times 12^2$, तब
        $577^2 - 2 \times 408^2$. भाग I सिद्ध हुआ
        $ a^2 - 2b^2 = 0$ असंभव ---बगुले कितने करीब हैं
        \omterm{def:g9:fractions:fraction}{अंश}ों असंभव पर आएं?
  \item समापन, दो वाक्यों में: इकाई का विकर्ण
        वर्ग कागज पर मौजूद है, नहीं \omterm{def:g9:fractions:fraction}{अंश} इसे मापता है, और इसका
        दशमलव लेखन कभी भी दोहराया नहीं जा सकता
        (\cref{pb:g9:fractions:1}). किस तरह के ``नए नंबर''
        इसलिए संख्या \omterm{def:g6:lines:objects}{रेखा} में --- और कहाँ शामिल होना चाहिए
        क्या यह श्रृंखला उनकी पूरी कहानी बताती है?
\end{enumerate}
\end{problem}
