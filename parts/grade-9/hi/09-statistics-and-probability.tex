\chapter{सांख्यिकी और संभाव्यता}\label{ch:g9:statproba}

डेटा और मौका हर जगह हैं: खेल परिणाम, मौसम पूर्वानुमान,
खेल. यह अध्याय सिखाता है कि संख्याओं की श्रृंखला को उसके आधार पर कैसे सारांशित किया जाए
\omterm{def:g9:statproba:indicators}{अर्थ}, \omterm{def:g9:statproba:indicators}{MEDIAN} और \omterm{def:g9:statproba:indicators}{श्रेणी}, और इसकी गणना कैसे करें \omterm{def:g9:statproba:probability}{संभावना} सरल का
यादृच्छिक प्रयोग --- जिसमें पेड़ के साथ संभाले गए दो-चरणीय प्रयोग शामिल हैं
आरेख. हाई स्कूल में दोनों विषयों को और अधिक विकसित किया जाता है
इस श्रृंखला का वॉल्यूम.

\section{डेटा का सारांश}

\begin{definition}[माध्य, माध्य, सीमा]\label{def:g9:statproba:indicators}
की एक श्रृंखला के लिए $N $ संख्याएँ:
\begin{itemize}
  \item  \emph{अर्थ}\index{अर्थ} है \omterm{def:g1:addition:def}{जोड़} सभी मूल्यों का विभाजन
        द्वारा $ N $;
  \item  \emph{MEDIAN}\index{MEDIAN} क्रमबद्ध को \omterm{def:g9:arith:divisor}{विभाजित} करने वाला एक मान है
        समान आकार के दो हिस्सों में श्रृंखला: के लिए $ N $ \omterm{def:g3:numbers:evenodd}{विषम}, मध्य
        मूल्य; के लिए $ N $ \omterm{def:g3:numbers:evenodd}{यहां तक की}, दो मध्य मानों का मध्यबिंदु;
  \item  \emph{श्रेणी}\index{श्रेणी} को घटाकर सबसे बड़ा मान है
        सबसे छोटा.
\end{itemize}
\end{definition}

\begin{example}\label{ex:g9:statproba:indicators}
एक छात्र के अंक: $8,\ 12,\ 9,\ 15,\ 11$.
\begin{enumerate}
  \item \omterm{def:g9:statproba:indicators}{अर्थ}: $\dfrac{8 + 12 + 9 + 15 + 11}{5} = \dfrac{55}{5} = 11$.
  \item मंझला: श्रृंखला क्रमबद्ध करें: $8, 9, 11, 12, 15$; मध्य (तीसरा)
        मूल्य है $11$.
  \item \omterm{def:g9:statproba:indicators}{श्रेणी}: $15 - 8 = 7$.
\end{enumerate}
के छठे निशान के साथ $17$: क्रमबद्ध $8, 9, 11, 12, 15, 17$, द \omterm{def:g9:statproba:indicators}{MEDIAN}
का मध्यबिंदु बन जाता है $11$ और $12$, वह है $11.5$, और यह \omterm{def:g9:statproba:indicators}{अर्थ}
बन जाता है $\frac{72}{6} = 12$.
\end{example}

\begin{example}[भारित माध्य]\label{ex:g9:statproba:weighted}
एक दिन में बिकने वाले जूते के आकार:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccccc}
size & $38$ & $39$ & $40$ & $41$ & $42$ \\
\hline
count & $3$ & $7$ & $6$ & $3$ & $1$ \\
\end{tabular}
\end{center}
 \omterm{def:g9:statproba:indicators}{अर्थ} आकार है
\[
\frac{3 \times 38 + 7 \times 39 + 6 \times 40 + 3 \times 41 + 1 \times 42}
{3 + 7 + 6 + 3 + 1}
= \frac{114 + 273 + 240 + 123 + 42}{20}
= \frac{792}{20} = 39.6 .
\]
\end{example}

\begin{remark}
 \omterm{def:g9:statproba:indicators}{अर्थ} और यह \omterm{def:g9:statproba:indicators}{MEDIAN} बहुत भिन्न हो सकते हैं. श्रंखला में
$10, 10, 10, 10, 60$  \omterm{def:g9:statproba:indicators}{अर्थ} है $20$ (मूल्य से ऊपर खींच लिया गया $60$) जब \omterm{def:g9:statproba:indicators}{MEDIAN} है $10$. हमेशा पूछें कि कौन सा संकेतक स्थिति का प्रतिनिधित्व करता है
बेहतर.
\end{remark}

\section{संभावना}

\begin{definition}[संभावना]\label{def:g9:statproba:probability}
एक यादृच्छिक प्रयोग में कई संभावनाएं होती हैं \emph{परणाम}; एक
\emph{आयोजन} परिणामों का एक सेट है. प्रत्येक परिणाम को एक मिलता है
\emph{संभावना}\index{संभावना}: के बीच एक संख्या $0$ और $1$,
परिणामों की सभी संभावनाओं के योग के साथ $1$; द
संभाव्यता $\P(A)$ किसी घटना का $ A $ है \omterm{def:g1:addition:def}{जोड़} की संभावनाओं का
इसके परिणाम. जब सब $ n $ परिणाम समान रूप से संभावित हैं,
\[
\P(A) = \frac{\text{number of favorable outcomes}}{n}.
\]
\end{definition}

\begin{example}\label{ex:g9:statproba:spinner}
एक पहिया दो टुकड़ों में बंटा हुआ है $8$ समान \omterm{def:g6:measure:area}{क्षेत्र}: $3$ लाल, $3$ नीला, $2$ हरा।
प्रत्येक सेक्टर में है \omterm{def:g9:statproba:probability}{संभावना} $\frac18$, इसलिए
\[
\P(\text{red}) = \frac38, \qquad
\P(\text{green}) = \frac28 = \frac14, \qquad
\P(\text{not green}) = 1 - \frac14 = \frac34 .
\]
\end{example}

\begin{proposition}[पूरक]\label{prop:g9:statproba:complement}
हर घटना के लिए $ A $, द \emph{विपरीत घटना} $\bar A $ (``$ A $ नहीं करता
घटित'') संतुष्ट करता है
\[
\P(\bar A) = 1 - \P(A).
\]
\end{proposition}

\begin{proof}
प्रत्येक परिणाम बिल्कुल एक में है $ A $ और $\bar A $, इसलिए
$\P(A) + \P(\bar A)$ रकम सभी परिणामों की संभावनाएँ, जो है
$1$.
\end{proof}

\section{दो-चरणीय प्रयोग}

\begin{method}[वृक्ष रेखाचित्र]\label{met:g9:statproba:tree}
दो चरणों में एक प्रयोग के लिए:
\begin{enumerate}
  \item पहले चरण के प्रत्येक परिणाम के अनुसार एक शाखा बनाएं, फिर जारी रखें
        प्रत्येक शाखा को दूसरे चरण के परिणामों के साथ लिखते हुए
        \omterm{def:g9:statproba:probability}{संभावना} इसकी शाखा पर;
  \item प्राप्त करने के लिए पथ में संभावनाओं को गुणा करें \omterm{def:g9:statproba:probability}{संभावना}
        उस पथ का;
  \item किसी ईवेंट को बनाने वाले सभी पथों की संभावनाओं को \omterm{def:g1:addition:def}{जोड़}ें।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:statproba:tree}
एक बैग रखता है $2$ लाल और $3$ काले टोकन. एक टोकन बनाएं, उसे नोट करें
रंग, \emph{इसे वापस रखें}, फिर से ड्रा करें। प्रत्येक ड्रा के साथ लाल रंग मिलता है
\omterm{def:g9:statproba:probability}{संभावना} $\frac25$, काले के साथ \omterm{def:g9:statproba:probability}{संभावना} $\frac35$.

\begin{omfigure}
\begin{tikzpicture}[scale=0.95]

  \draw (R) -- (RR) node[midway, above, font=\small, sloped] {$\frac25$};

शाखाएँ, और जाँचें कि चारों पथों का योग है $1$.}

\omterm{def:g9:statproba:probability}{संभावना} एक ही रंग के दो टोकन:
$\frac{4}{25} + \frac{9}{25} = \frac{13}{25}$. \omterm{def:g9:statproba:probability}{संभावना} कम से कम
एक लाल: $1 - \P(\text{BB}) = 1 - \frac{9}{25} = \frac{16}{25}$.
\end{example}

\begin{remark}[आवृत्तियाँ संभावनाओं के करीब पहुँचती हैं]
एक निष्पक्ष पासा घुमाना $6000$ समय देता है \emph{के बारे में} $1000$ छक्के---नहीं
बिलकुल. जैसे-जैसे दोहराव की संख्या बढ़ती है, प्रेक्षित आवृत्तियाँ बढ़ती जाती हैं
संभावनाओं के करीब और करीब: यही बनाता है \omterm{def:g9:statproba:probability}{संभावना}
बार-बार प्रयोगों को मॉडल करने के लिए सही उपकरण (एक कहानी विकसित हुई)।
इस शृंखला का हाई स्कूल \omterm{def:g6:lines:objects}{खंड})।
\end{remark}

एक सिक्का उछाला जाता है, फिर एक निष्पक्ष पासा फेंका जाता है। पेड़ बनाएं (या गिनें)।
परिणाम) और गणना करें \omterm{def:g9:statproba:probability}{संभावना} सिर पाने का \emph{और} ए

\begin{exercise}[$\star\star $]\label{exo:g9:statproba:7}

\omterm{def:g9:statproba:probability}{संभावना} दो काले टोकन निकालने का, फिर दो अलग-अलग टोकन निकालने का

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:statproba:8}
बाद $9$ मैच, एक बास्केटबॉल खिलाड़ी का औसत $14$ प्रति मैच अंक.
\begin{enumerate}
  \item उसने कुल कितने अंक अर्जित किये हैं?
  \item उसे कितने अंक प्राप्त करने होंगे $10$ वां औसत से मेल खाता है

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g9:statproba:9}
एक खेल: दो निष्पक्ष पासे फेंको; यदि दोनों परिणाम बराबर हों तो आप जीत जाते हैं

\end{enumerate}
\end{exercise}

\section{समस्या: शेवेलियर के विनाशकारी दांव}

\begin{problem}[{सप्ताहांत समस्या---1654 जुआ विवाद जिसने संभाव्यता सिद्धांत और जन्मदिन संयोग का निर्माण किया जो हर कक्षा को मूर्ख बनाता है}]
\label{pb:g9:statproba:1}
1654 में एक कठिन जुआ खेलने वाला फ्रांसीसी रईस, शेवेलियर डी
एम\'एर\'ई, एक पासे के दांव पर अमीर हो गया और दूसरे पर हारने लगा
कि वह समकक्ष मानता था। चकित होकर उसने गणितज्ञ से पूछा
ब्लेज़ पास्कल; पास्कल ने पियरे डी फ़र्मेट को लिखा; और उनके
पत्रों के आदान-प्रदान के सिद्धांत की स्थापना की \omterm{def:g9:statproba:probability}{संभावना}. यह
समस्या इस अध्याय के उपकरणों के साथ पूरे मामले को दोहराती है
--- समान रूप से संभावित परिणाम, पेड़ और पूरक नियम
(\cref{prop:g9:statproba:complement}) --- और सबसे के साथ समाप्त होता है
उन सभी का प्रति-सहज संयोग।

\medskip
\noindent\textbf{Part I --- Dice, honestly counted.}
\begin{enumerate}
  \item एक निष्पक्ष पासे को दो बार घुमाएँ। एक पेड़ का उपयोग करना (या a $6 \times 6$
        तालिका), गणना करें \omterm{def:g9:statproba:probability}{संभावना} का \emph{नहीं} छह में
        दो रोल, और घटाएँ \omterm{def:g9:statproba:probability}{संभावना} कम से कम एक का
        छह.
  \item एक सहपाठी का तर्क है: ``प्रति रोल छह में से एक मौका, दो बार,
        तो $\frac16 + \frac16 = \frac13$.'' प्रश्न से तुलना करें
        1 और सटीक परिणामों पर इंगित करें कि उसका \omterm{def:g1:addition:def}{जोड़} मायने रखता है
        दो बार.
  \item दो पासे पलटना और जोड़ना: गणना करें \omterm{def:g9:statproba:probability}{संभावना} का
        ए \omterm{def:g1:addition:def}{जोड़} का $7$ और एक का \omterm{def:g1:addition:def}{जोड़} का $2$. क्यों $7$ द
        जुआरियों का पसंदीदा?
  \item पुराना विवाद: दो पासों के साथ, एक है \omterm{def:g1:addition:def}{जोड़} का $9$ या ए \omterm{def:g1:addition:def}{जोड़} का
        $10$ अधिक \omterm{def:g9:statproba:probability}{संभावना}? परिणामों को गिनें और उसका निपटारा करें।
  \item प्रश्न 1 को सामान्यीकृत करें: स्पष्ट करें कि ऐसा क्यों है \omterm{def:g9:statproba:probability}{संभावना} का
        कम से कम एक छह इंच $ n $ रोल्स है
        \[
        1 - \left(\frac56\right)^{\!n} ,
        \]
        पेड़ की नो-छह शाखाएँ स्तर से बढ़ रही हैं
        समतल करना।
\end{enumerate}

\noindent\textbf{Part II --- The two bets of the Chevalier.}
\begin{enumerate}[resume]
  \item पहला दांव, \omterm{def:g3:numbers:evenodd}{यहां तक की} धन: \emph{चार में से कम से कम एक छक्का
        पासे का रोल}. गणना करना
        $\left(\frac56\right)^4$ के तौर पर \omterm{def:g9:fractions:fraction}{अंश} और दशमलव के रूप में,
        और शेवेलियर की जीत \omterm{def:g9:statproba:probability}{संभावना}. शर्त थी
        अच्छा?
  \item शेवेलियर का अपना तर्क था: ``चार रोल पर
        $\frac16$ प्रत्येक: $\frac46$ एक मौक़ा।'' इसने लगभग दे दिया
        यहाँ सही उत्तर है --- लेकिन इसे आगे बढ़ाएँ $7$ रोल्स: क्या
        यह बेतुकापन पैदा करता है, और प्रश्न 2 की कौन सी त्रुटि है
        क्या यह दोहराता है?
  \item दूसरा दांव, द्वारा समकक्ष माना जाता है समानता
        (``$24$ पर रोल करता है $\frac1{36}$ प्रत्येक: फिर से
        $\frac{24}{36} = \frac46$''): \emph{कम से कम एक
        डबल-सिक्स इन $24$ दो पासों का रोल}. सत्य की गणना करें
        जीतना \omterm{def:g9:statproba:probability}{संभावना} $1 - \left(\frac{35}{36}\right)^{24}$
        (कैलकुलेटर)। शेवेलियर धीरे-धीरे क्यों बर्बाद हो गया?
  \item दो पासों को कितने बार घुमाने से उसकी स्थिति ठीक हो जाती
        फायदा? परीक्षा $ n = 25$ कैलकुलेटर के साथ और
        निष्कर्ष.
  \item नैतिकता को दो वाक्यों में बताएं: क्या गलत है, में
        सामान्य, गुणा करने के साथ \omterm{def:g9:statproba:probability}{संभावना} की संख्या से
        कोशिश करता है --- और कौन सा सही उपकरण उसे प्रतिस्थापित करता है
        प्रलोभन?
\end{enumerate}

\noindent\textbf{Part III --- मतलब, माध्यिकाओं, birthdays.}
\begin{enumerate}[resume]
  \item एक छोटी फर्म नौ कर्मचारियों को वेतन देती है $2\,000$ यूरो प्रति माह
        और निर्देशक $20\,000$. की गणना करें \omterm{def:g9:statproba:indicators}{अर्थ} और यह
        \omterm{def:g9:statproba:indicators}{MEDIAN} वेतन (\cref{def:g9:statproba:indicators}).
        ईमानदारी से नौकरी का विज्ञापन किस नंबर पर करना चाहिए
        बोली?
  \item पांच परीक्षण अंकों की एक सूची बनाएं \omterm{def:g9:statproba:indicators}{अर्थ} $12$ और
        \omterm{def:g9:statproba:indicators}{MEDIAN} $15$. \omterm{def:g1:addition:def}{जोड़}ी क्या करती है (\omterm{def:g9:statproba:indicators}{अर्थ} नीचे \omterm{def:g9:statproba:indicators}{MEDIAN}) कहो
        निशानों के आकार के बारे में?
  \item जन्मदिन की समस्या: की एक कक्षा में $23$ छात्र, क्या
        है \omterm{def:g9:statproba:probability}{संभावना} कि कम से कम दो लोग एक ही जन्मदिन साझा करें?
        पूरक स्थापित करें (सभी)। $23$ जन्मदिन अलग):
        \[
        \frac{365}{365} \times \frac{364}{365} \times
        \frac{363}{365} \times \dots \times \frac{343}{365} .
        \]
        की गणना करें \omterm{def:g2:mult:def}{उत्पाद} पहले तीन कारकों का वर्णन करें
        कारक कैसे विकसित होते हैं, और --- यह देखते हुए कि पूर्ण
        \omterm{def:g2:mult:def}{उत्पाद} के बारे में है $0.493$ --- सवाल का जवाब दें। अधिकांश
        लोग ``बहुत असंभावित'' की उम्मीद करते हैं: क्या करता है
        गणित कहता है?
  \item \omterm{ex:g1:subtraction:difference}{अंतर}्ज्ञान ने मरम्मत की: कितने \emph{\omterm{def:g1:addition:def}{जोड़}े} का
        छात्र अपना जन्मदिन एक कक्षा में साझा कर सकते हैं $23$
        (\cref{pb:g6:wholes:1}(हाथ मिलाने की गिनती)? एक में स्पष्ट करें
        वाक्य यह संख्या क्यों, नहीं $23$ स्वयं, चलाता है
        आश्चर्य.
  \item समापन: किसी भी परिणाम की जांच करने के लिए एक प्रयोग का वर्णन करें
        --- डी एम\'एर\'ई का दांव असली पासे या जन्मदिन के साथ है
        कई वर्गों में समस्या --- और ध्यान से बताएं
        संख्या के रूप में प्रेक्षित आवृत्तियों से आप क्या अपेक्षा करते हैं
        दोहराव बढ़ता है। (उस उम्मीद का एक नाम है, द
        बड़ी संख्या का नियम; इसका प्रमाण उच्च में प्रतीक्षारत है
        स्कूल वॉल्यूम.)
\end{enumerate}
\end{problem}
