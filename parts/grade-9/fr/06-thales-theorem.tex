\chapter{Le théorème de Thalès}\label{ch:g9:thales}

Comment mesurer la hauteur d'une pyramide sans y grimper~? Thalès de
Milet a comparé son ombre avec l'ombre d'un bâton. Son théorème --- des
droites parallèles découpent des segments en parts proportionnelles ---
est le cœur mathématique des maquettes, des cartes et des agrandissements,
et l'un des plus anciens théorèmes portant un nom.

\section{Le théorème}

\begin{theorem}[Thalès]\label{thm:g9:thales:direct}
Soient deux droites se rencontrant en un point $A$. Prendre $M$ sur la
première droite et $N$ sur la seconde, avec $B$ plus loin sur la
première et $C$ plus loin sur la seconde. Si les droites $(MN)$ et
$(BC)$ sont \emph{parallèles}, alors les triangles $AMN$ et $ABC$ ont
des côtés proportionnels~:
\[
\frac{AM}{AB} = \frac{AN}{AC} = \frac{MN}{BC}.
\]
\end{theorem}
\admitted

\begin{method}[Calculer une longueur avec Thalès]\label{met:g9:thales:compute}
\begin{enumerate}
  \item Identifier le point $A$ où les deux droites se croisent et les
        deux droites parallèles~; nommer les deux triangles~;
  \item écrire les trois rapports égaux, toujours «~petit triangle sur
        grand triangle~»~;
  \item garder l'égalité des deux rapports impliquant les trois
        longueurs connues et l'inconnue~;
  \item résoudre la proportion résultante par produits en croix.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:thales:compute}
Dans la configuration de gauche ci-dessus, supposons $AM = 3$, $AB = 5$,
$AN = 2.4$ et $MN = 3.3$, et calculons $AC$ et $BC$. Thalès donne
\[
\frac{AM}{AB} = \frac{AN}{AC} = \frac{MN}{BC},
\qquad\text{c'est-à-dire}\qquad
\frac35 = \frac{2.4}{AC} = \frac{3.3}{BC}.
\]
De $\frac35 = \frac{2.4}{AC}$, en croisant les produits~:
$3 \times AC = 5 \times 2.4 = 12$, donc $AC = 4$. De
$\frac35 = \frac{3.3}{BC}$~: $3 \times BC = 16.5$, donc $BC = 5.5$.
\end{example}

\begin{example}[Thalès dans la vie réelle]\label{ex:g9:thales:stick}
Un bâton vertical de hauteur $1$ m projette une ombre de $1.5$ m,
tandis qu'un arbre projette une ombre de $12$ m au même moment. Les
rayons du Soleil sont parallèles, donc les triangles bâton-ombre et
arbre-ombre sont en configuration de Thalès~:
\[
\frac{\text{hauteur de l'arbre}}{1} = \frac{12}{1.5},
\qquad\text{donc l'arbre mesure } 8 \text{ m de haut.}
\]
\end{example}

\section{La réciproque}

\begin{theorem}[Réciproque de Thalès]\label{thm:g9:thales:converse}
Avec les points placés comme en \cref{thm:g9:thales:direct} ($M$ sur
$(AB)$, $N$ sur $(AC)$, dans le même ordre sur les deux droites)~: si
\[
\frac{AM}{AB} = \frac{AN}{AC},
\]
alors les droites $(MN)$ et $(BC)$ sont parallèles.
\end{theorem}
\admitted

\begin{method}[Prouver ou infirmer le parallélisme]\label{met:g9:thales:converse}
\begin{enumerate}
  \item Calculer séparément les deux rapports $\dfrac{AM}{AB}$ et
        $\dfrac{AN}{AC}$ (en fractions, pas en arrondis)~;
  \item s'ils sont égaux \emph{et} que les points sont dans le même
        ordre sur les deux droites, les droites sont parallèles
        (réciproque de Thalès)~;
  \item s'ils diffèrent, les droites ne sont \emph{pas} parallèles ---
        car si elles l'étaient, le théorème de Thalès forcerait les
        rapports à être égaux.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:thales:converse}
Sur la droite 1, $AM = 4$ et $AB = 10$~; sur la droite 2, $AN = 6$ et
$AC = 15$. Alors
\[
\frac{AM}{AB} = \frac{4}{10} = \frac25,
\qquad
\frac{AN}{AC} = \frac{6}{15} = \frac25 :
\]
rapports égaux, même ordre~: $(MN) \parallel (BC)$.

Avec $AC = 14$ à la place~: $\frac{6}{14} = \frac37 \neq \frac25$,
donc les droites ne sont pas parallèles.
\end{example}

\section{Agrandissement et réduction}

\begin{definition}[Homothétie d'une figure]\label{def:g9:thales:scaling}
Agrandir ou réduire une figure par le \emph{facteur
d'échelle}\index{facteur d'échelle} $k > 0$ signifie multiplier
\emph{toutes} ses longueurs par $k$~: un \emph{agrandissement} lorsque
$k > 1$, une \emph{réduction} lorsque $k < 1$. Dans la configuration de
Thalès, le triangle $AMN$ est la réduction de $ABC$ par le facteur
$k = \frac{AM}{AB}$.
\end{definition}

\begin{proposition}[Effet sur les angles et les aires]\label{prop:g9:thales:scaling}
L'homothétie de facteur $k$ conserve les angles et la forme des figures,
multiplie chaque longueur par $k$, et multiplie chaque \emph{aire} par
$k^2$.
\end{proposition}
\admitted

\begin{omfigure}
\begin{tikzpicture}[scale=0.72]
  \draw[thick, omDef, fill=omDef!8] (0,0) rectangle (2,1.2);
  \draw[thick, omThm, fill=omThm!8] (3.4,0) rectangle (7.4,2.4);
  \node[font=\small, omDef] at (1,-0.45) {$2 \times 1.2$};
  \node[font=\small, omThm] at (5.4,-0.45) {$4 \times 2.4$};
  \node[font=\small] at (1,0.6) {aire $2.4$};
  \node[font=\small] at (5.4,1.2) {aire $9.6$};
\end{tikzpicture}

{\small Doubler les longueurs ($k = 2$) multiplie l'aire par
$k^2 = 4$~: quatre copies du petit rectangle pavent le grand.}
\end{omfigure}

\begin{example}
Une photo de $10 \times 15$ cm est agrandie avec le facteur d'échelle
$k = 3$~: le tirage mesure $30 \times 45$ cm. Son aire passe de
$150$ cm$^2$ à $150 \times 9 = 1350$ cm$^2$ --- $9$ fois plus grande,
pas $3$.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g9:thales:1}
Les droites $(MN)$ et $(BC)$ sont parallèles, avec $M$ sur $[AB]$ et $N$
sur $[AC]$~; $AM = 2$, $AB = 6$, $AN = 3$, $BC = 9$. Calculer $AC$ et
$MN$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:thales:2}
Même configuration~: $AM = 5$, $MB = 3$ (attention~: $MB$, pas $AB$~!),
$AN = 4$. Calculer $AC$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:thales:3}
En configuration papillon, $A$ est entre $M$ et $B$ et entre $N$ et $C$,
avec $(MN) \parallel (BC)$~; $AM = 3$, $AB = 7.5$, $AN = 2$, $MN = 2.6$.
Calculer $AC$ et $BC$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:thales:4}
$M$ est sur $[AB]$ avec $AM = 6$, $AB = 8$~; $N$ est sur $[AC]$ avec
$AN = 9$, $AC = 12$. Les droites $(MN)$ et $(BC)$ sont-elles
parallèles~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:5}
Comme ci-dessus avec $AM = 4$, $AB = 6$, $AN = 5$, $AC = 8$. Les droites
$(MN)$ et $(BC)$ sont-elles parallèles~? Justifier soigneusement.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:6}
Une personne de $1.8$ m de haut se tient à $2$ m de la base d'un lampadaire~;
son ombre mesure $3$ m. Dessiner la configuration de Thalès formée par le
lampadaire, la personne et le bout de l'ombre, et calculer la hauteur du
lampadaire.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:7}
Une carte est à l'échelle $1 : 25\,000$ (les longueurs sur la carte sont
les longueurs réelles multipliées par $k = \frac{1}{25000}$).
\begin{enumerate}
  \item Deux villages sont distants de $6.8$ cm sur la carte. Quelle est
        la distance réelle, en km~?
  \item Une forêt a une aire de $8$ cm$^2$ sur la carte. Quelle est son
        aire réelle, en km$^2$~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:8}
Le triangle $ABC$ a $AB = 12$, $AC = 15$, $BC = 18$. Le point $M$ sur
$[AB]$ satisfait $AM = 8$, et la droite passant par $M$ parallèle à
$(BC)$ coupe $[AC]$ en $N$.
\begin{enumerate}
  \item Calculer $AN$ et $MN$.
  \item Quel est le facteur d'échelle de $ABC$ vers $AMN$~? Comparer les
        périmètres des deux triangles.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:thales:9}
Soit $ABCD$ un trapèze avec $(AB) \parallel (CD)$, $AB = 4$ et $CD = 6$,
dont les diagonales $[AC]$ et $[BD]$ se rencontrent en $O$. En utilisant
Thalès dans la configuration papillon autour de $O$, calculer le rapport
$\dfrac{OA}{OC}$, et montrer que $O$ coupe les deux diagonales dans le
même rapport.
\end{exercise}
