\chapter{Fonctions linéaires et affines}\label{ch:g9:linfunc}

«~Trois kilogrammes coûtent trois fois plus cher qu'un~»~: c'est la
proportionnalité, la façon la plus simple dont deux quantités peuvent
être liées. Les fonctions linéaires sont la proportionnalité dans le
langage des fonctions~; les fonctions affines ajoutent un montant de
départ fixe. Leurs graphiques sont des droites, et lire ces droites est
la compétence que ce chapitre construit.

\section{Proportionnalité et fonctions linéaires}

\begin{definition}[Fonction linéaire]\label{def:g9:linfunc:linear}
Une \emph{fonction linéaire}\index{fonction linéaire} multiplie tout
nombre par un nombre fixe $a$~:
\[
f(x) = a x .
\]
Le nombre $a$ est le \emph{coefficient} de $f$. Deux quantités sont
\emph{proportionnelles}\index{proportionnalité} exactement lorsque l'une
est une fonction linéaire de l'autre.
\end{definition}

\begin{example}\label{ex:g9:linfunc:linear}
Si les pommes coûtent $3$ l'unité de kilogramme, le prix de $x$
kilogrammes est $f(x) = 3x$~: en acheter deux fois plus coûte deux fois
plus, et le prix de $2.5$ kg est $f(2.5) = 7.5$.

Réciproquement, si une fonction linéaire satisfait $f(4) = 10$, alors
$a = \frac{10}{4} = 2.5$ et $f(x) = 2.5x$~: \emph{une seule valeur
détermine complètement une fonction linéaire.}
\end{example}

\begin{proposition}[Graphique d'une fonction linéaire]\label{prop:g9:linfunc:lineargraph}
Le graphique de $f(x) = ax$ est une droite \emph{passant par l'origine}.
Réciproquement, toute droite non verticale passant par l'origine est le
graphique d'une fonction linéaire.
\end{proposition}
\admitted

\section{Fonctions affines}

\begin{definition}[Fonction affine]\label{def:g9:linfunc:affine}
Une \emph{fonction affine}\index{fonction affine} a la forme
\[
f(x) = a x + b ,
\]
où $a$ et $b$ sont des nombres fixes. Les fonctions linéaires sont le cas
particulier $b = 0$~; les fonctions constantes le cas $a = 0$.
\end{definition}

\begin{example}
Une salle de sport facture des frais d'inscription de $20$, puis $5$ par
visite~: le coût total de $x$ visites est $f(x) = 5x + 20$. Le coût
n'est \emph{pas} proportionnel au nombre de visites (10 visites ne
coûtent pas deux fois plus que 5), mais chaque visite supplémentaire
ajoute le même montant, $5$.
\end{example}

\begin{proposition}[Graphique d'une fonction affine]\label{prop:g9:linfunc:affinegraph}
Le graphique de $f(x) = ax + b$ est une droite~:
\begin{itemize}
  \item $b = f(0)$ est l'\emph{ordonnée à l'origine}~: la droite coupe
        l'axe vertical en $(0, b)$~;
  \item $a$ est le \emph{coefficient directeur}\index{coefficient
        directeur}~: avancer d'une unité vers la droite fait monter de
        $a$ unités (descendre si $a < 0$)~; la fonction est croissante
        lorsque $a > 0$, décroissante lorsque $a < 0$~;
  \item pour deux entrées $u \neq v$,
        $a = \dfrac{f(v) - f(u)}{v - u}$.
\end{itemize}
\end{proposition}

\begin{proof}[Preuve du dernier point]
$f(v) - f(u) = (av + b) - (au + b) = a(v - u)$~; diviser par $v - u$. Le
reste est admis à ce niveau (le volume de lycée donne un traitement
complet des droites et de leurs équations).
\end{proof}

\begin{method}[Trouver une fonction affine à partir de deux valeurs]\label{met:g9:linfunc:findaffine}
Étant donnés $f(u)$ et $f(v)$ avec $u \neq v$~:
\begin{enumerate}
  \item calculer le coefficient directeur
        $a = \dfrac{f(v) - f(u)}{v - u}$~;
  \item substituer l'une des deux valeurs connues dans $f(x) = ax + b$
        pour trouver $b$~;
  \item vérifier avec l'autre valeur.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:linfunc:findaffine}
Trouver la fonction affine avec $f(2) = 7$ et $f(5) = 16$.
\[
a = \frac{16 - 7}{5 - 2} = \frac93 = 3 ,
\]
puis $f(2) = 3 \times 2 + b = 7$ donne $b = 1$~: $f(x) = 3x + 1$.
Vérification~: $f(5) = 15 + 1 = 16$.
\end{example}

\section{Pourcentages}

\begin{proposition}[Variation en pourcentage]\label{prop:g9:linfunc:percent}
Augmenter une quantité de $t\,\%$ la multiplie par
$1 + \dfrac{t}{100}$~; la diminuer de $t\,\%$ la multiplie par
$1 - \dfrac{t}{100}$. Les variations en pourcentage sont des fonctions
linéaires.
\end{proposition}

\begin{proof}
Augmenter $x$ de $t\,\%$ signifie ajouter $\frac{t}{100}\,x$~:
\[
x + \frac{t}{100}\,x = \left(1 + \frac{t}{100}\right) x .
\]
Même calcul avec un signe moins pour une diminution.
\end{proof}

\begin{example}\label{ex:g9:linfunc:percent}
Un prix de $80$ augmente de $15\,\%$~: nouveau prix
$80 \times 1.15 = 92$. Un prix de $60$ diminue de $30\,\%$~:
$60 \times 0.7 = 42$.

\emph{Enchaîner des variations multiplie les facteurs.} Une hausse de
$20\,\%$ suivie d'une baisse de $20\,\%$ donne
$1.2 \times 0.8 = 0.96$~: une \emph{baisse} de $4\,\%$ au total, pas un
retour au départ~!
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g9:linfunc:1}
Parmi ces fonctions, lesquelles sont linéaires~? affines~? ni l'un ni
l'autre~?
\[
f(x) = 4x, \qquad
g(x) = 3x - 5, \qquad
h(x) = x^2, \qquad
k(x) = -\tfrac x2, \qquad
m(x) = 7 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:linfunc:2}
Soit $f(x) = 2x - 3$. Calculer $f(0)$, $f(4)$, $f(-1)$, et trouver le
nombre $x$ tel que $f(x) = 9$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:linfunc:3}
Une fonction linéaire satisfait $f(6) = 15$. Trouver son coefficient,
puis calculer $f(10)$ et $f(-2)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:linfunc:4}
Tracer sur le même repère les graphiques de $f(x) = 2x$,
$g(x) = 2x + 3$ et $h(x) = -x + 4$. Que peut-on dire des graphiques de
$f$ et $g$~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:linfunc:5}
Trouver la fonction affine telle que $f(1) = 5$ et $f(3) = 11$~; puis
celle telle que $g(0) = 4$ et $g(2) = 0$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:linfunc:6}
Calculer~: le prix après une hausse de $25\,\%$ sur $120$~; le prix après
une remise de $40\,\%$ sur $65$~; le prix d'origine si un article coûte
$69$ après une remise de $25\,\%$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:linfunc:7}
Un forfait téléphonique A coûte $10$ par mois plus $0.05$ par minute
d'appels~; le forfait B coûte $25$ par mois avec appels illimités.
\begin{enumerate}
  \item Exprimer le coût mensuel du forfait A en fonction du nombre $x$
        de minutes.
  \item À partir de combien de minutes par mois le forfait B est-il
        moins cher~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:linfunc:8}
Une population de $2000$ bactéries augmente de $10\,\%$ chaque heure.
\begin{enumerate}
  \item Combien de bactéries y a-t-il après une heure~? Après deux
        heures~?
  \item Expliquer pourquoi la réponse après deux heures n'est pas
        $2400$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:linfunc:9}
Le graphique d'une fonction affine passe par les points $(2, 3)$ et
$(6, 1)$.
\begin{enumerate}
  \item Calculer son coefficient directeur. La fonction est-elle
        croissante ou décroissante~?
  \item Donner son expression, et calculer l'entrée pour laquelle la
        sortie est $0$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:linfunc:10}
Un magasin augmente d'abord un prix de $t\,\%$, puis baisse le nouveau
prix de $t\,\%$.
\begin{enumerate}
  \item Montrer que le prix final égale l'original multiplié par
        $1 - \dfrac{t^2}{10000}$.
  \item En déduire que le prix final est toujours \emph{plus bas} que
        l'original (pour $0 < t \leq 100$), et trouver la variation en
        pourcentage globale pour $t = 10$.
\end{enumerate}
\end{exercise}
