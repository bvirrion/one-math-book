\chapter{Statistiques et probabilités}\label{ch:g9:statproba}

Données et hasard sont partout~: résultats sportifs, prévisions météo,
jeux. Ce chapitre enseigne à résumer une série de nombres par sa
\omterm{def:g9:statproba:indicators}{moyenne}, sa \omterm{def:g9:statproba:indicators}{médiane} et son \omterm{def:g9:statproba:indicators}{étendue}, et à calculer la \omterm{def:g9:statproba:probability}{probabilité}
d'expériences aléatoires simples --- y compris des expériences à deux
étapes, traitées avec des arbres. Les deux sujets sont développés bien
plus loin dans le volume de lycée de cette série.

\section{Résumer des données}

\begin{definition}[Moyenne, médiane, étendue]\label{def:g9:statproba:indicators}
Pour une série de $N$ nombres~:
\begin{itemize}
  \item la \emph{moyenne}\index{moyenne} est la \omterm{def:g1:addition:def}{somme} de toutes les
        valeurs divisée par $N$~;
  \item la \emph{médiane}\index{médiane} est une valeur partageant la
        série triée en deux \omterm{def:g3:division:half}{moitiés} de même taille~: pour $N$ \omterm{def:g3:numbers:evenodd}{impair}, la
        valeur du milieu~; pour $N$ \omterm{def:g3:numbers:evenodd}{pair}, le milieu des deux valeurs
        centrales~;
  \item l'\emph{étendue}\index{étendue} est la plus grande valeur moins
        la plus petite.
\end{itemize}
\end{definition}

\begin{example}\label{ex:g9:statproba:indicators}
Notes d'un élève~: $8,\ 12,\ 9,\ 15,\ 11$.
\begin{enumerate}
  \item \omterm{def:g9:statproba:indicators}{Moyenne}~: $\dfrac{8 + 12 + 9 + 15 + 11}{5} = \dfrac{55}{5} = 11$.
  \item \omterm{def:g9:statproba:indicators}{Médiane}~: trier la série~: $8, 9, 11, 12, 15$~; la valeur du
        milieu (3\textsuperscript{e}) est $11$.
  \item \omterm{def:g9:statproba:indicators}{Étendue}~: $15 - 8 = 7$.
\end{enumerate}
Avec une sixième note de $17$~: triée $8, 9, 11, 12, 15, 17$, la \omterm{def:g9:statproba:indicators}{médiane}
devient le milieu de $11$ et $12$, c'est-à-dire $11.5$, et la \omterm{def:g9:statproba:indicators}{moyenne}
devient $\frac{72}{6} = 12$.
\end{example}

\begin{example}[Moyenne pondérée]\label{ex:g9:statproba:weighted}
Les pointures vendues en une journée~:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccccc}
pointure & $38$ & $39$ & $40$ & $41$ & $42$ \\
\hline
\omterm{def:g7:stats:frequency}{effectif} & $3$ & $7$ & $6$ & $3$ & $1$ \\
\end{tabular}
\end{center}
La pointure \omterm{def:g9:statproba:indicators}{moyenne} est
\[
\frac{3 \times 38 + 7 \times 39 + 6 \times 40 + 3 \times 41 + 1 \times 42}
{3 + 7 + 6 + 3 + 1}
= \frac{114 + 273 + 240 + 123 + 42}{20}
= \frac{792}{20} = 39.6 .
\]
\end{example}

\begin{remark}
La \omterm{def:g9:statproba:indicators}{moyenne} et la \omterm{def:g9:statproba:indicators}{médiane} peuvent différer beaucoup. Dans la série
$10, 10, 10, 10, 60$ la \omterm{def:g9:statproba:indicators}{moyenne} est $20$ (tirée vers le haut par la
valeur $60$) tandis que la \omterm{def:g9:statproba:indicators}{médiane} est $10$. Toujours se demander quel
indicateur représente mieux la situation.
\end{remark}

\section{Probabilité}

\begin{definition}[Probabilité]\label{def:g9:statproba:probability}
Une expérience aléatoire a plusieurs \emph{issues} possibles~; un
\emph{événement} est un ensemble d'issues. Chaque issue reçoit une
\emph{probabilité}\index{probabilité}~: un nombre entre $0$ et $1$,
avec toutes les probabilités des issues sommant à $1$~; la probabilité
$\P(A)$ d'un événement $A$ est la \omterm{def:g1:addition:def}{somme} des probabilités de ses issues.
Lorsque les $n$ issues sont équiprobables,
\[
\P(A) = \frac{\text{nombre d'issues favorables}}{n}.
\]
\end{definition}

\begin{example}\label{ex:g9:statproba:spinner}
Une roue est partagée en $8$ secteurs égaux~: $3$ rouges, $3$ bleus,
$2$ verts. Chaque secteur a pour \omterm{def:g9:statproba:probability}{probabilité} $\frac18$, donc
\[
\P(\text{rouge}) = \frac38, \qquad
\P(\text{vert}) = \frac28 = \frac14, \qquad
\P(\text{pas vert}) = 1 - \frac14 = \frac34 .
\]
\end{example}

\begin{proposition}[Complément]\label{prop:g9:statproba:complement}
Pour tout événement $A$, l'\emph{événement contraire} $\bar A$ («~$A$ ne
se \omterm{def:g2:mult:def}{produit} pas~») satisfait
\[
\P(\bar A) = 1 - \P(A).
\]
\end{proposition}

\begin{proof}
Toute issue est dans exactement l'un de $A$ et $\bar A$, donc
$\P(A) + \P(\bar A)$ \omterm{def:g1:addition:def}{somme} les \omterm{def:g9:statproba:probability}{probabilités} de toutes les issues, qui
valent $1$.
\end{proof}

\section{Expériences à deux étapes}

\begin{method}[Arbres de probabilités]\label{met:g9:statproba:tree}
Pour une expérience en deux étapes~:
\begin{enumerate}
  \item tracer une branche par issue de la première étape, puis
        continuer chaque branche avec les issues de la seconde étape, en
        écrivant chaque \omterm{def:g9:statproba:probability}{probabilité} sur sa branche~;
  \item multiplier les \omterm{def:g9:statproba:probability}{probabilités} le long d'un chemin pour obtenir la
        \omterm{def:g9:statproba:probability}{probabilité} de ce chemin~;
  \item additionner les \omterm{def:g9:statproba:probability}{probabilités} de tous les chemins formant un
        événement.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:statproba:tree}
Un sac contient $2$ jetons rouges et $3$ noirs. Tirer un jeton, noter sa
couleur, le \emph{remettre}, tirer à nouveau. Chaque tirage donne rouge
avec \omterm{def:g9:statproba:probability}{probabilité} $\frac25$, noir avec \omterm{def:g9:statproba:probability}{probabilité} $\frac35$.

\begin{omfigure}
\begin{tikzpicture}[scale=0.95]
  \coordinate (root) at (0,0);
  \coordinate (R) at (2.2,1.0);
  \coordinate (B) at (2.2,-1.0);
  \coordinate (RR) at (4.7,1.55);
  \coordinate (RB) at (4.7,0.45);
  \coordinate (BR) at (4.7,-0.45);
  \coordinate (BB) at (4.7,-1.55);
  \draw (root) -- (R) node[midway, above, font=\small, sloped] {$\frac25$};
  \draw (root) -- (B) node[midway, below, font=\small, sloped] {$\frac35$};
  \draw (R) -- (RR) node[midway, above, font=\small, sloped] {$\frac25$};
  \draw (R) -- (RB) node[midway, below, font=\small, sloped] {$\frac35$};
  \draw (B) -- (BR) node[midway, above, font=\small, sloped] {$\frac25$};
  \draw (B) -- (BB) node[midway, below, font=\small, sloped] {$\frac35$};
  \node[omThm, font=\small, above left=-2pt] at (R) {R};
  \node[font=\small, below left=-2pt] at (B) {B};
  \node[right, font=\small] at (RR)
    {RR~: $\frac25 \times \frac25 = \frac{4}{25}$};
  \node[right, font=\small] at (RB)
    {RB~: $\frac25 \times \frac35 = \frac{6}{25}$};
  \node[right, font=\small] at (BR)
    {BR~: $\frac35 \times \frac25 = \frac{6}{25}$};
  \node[right, font=\small] at (BB)
    {BB~: $\frac35 \times \frac35 = \frac{9}{25}$};
\end{tikzpicture}

{\small L'arbre des deux tirages avec remise~: multiplier le long des
branches, et vérifier que les quatre chemins somment à $1$.}
\end{omfigure}

\omterm{def:g9:statproba:probability}{Probabilité} de deux jetons de la même couleur~:
$\frac{4}{25} + \frac{9}{25} = \frac{13}{25}$. \omterm{def:g9:statproba:probability}{Probabilité} d'au moins un
rouge~: $1 - \P(\text{BB}) = 1 - \frac{9}{25} = \frac{16}{25}$.
\end{example}

\begin{remark}[Les fréquences approchent les probabilités]
Lancer un dé équilibré $6000$ fois donne \emph{environ} $1000$ six ---
pas exactement. Lorsque le nombre de répétitions croît, les \omterm{def:g7:stats:frequency}{fréquences}
observées se rapprochent de plus en plus des \omterm{def:g9:statproba:probability}{probabilités}~: c'est ce qui
fait de la \omterm{def:g9:statproba:probability}{probabilité} l'outil juste pour modéliser des expériences
répétées (une histoire développée dans le volume de lycée de cette
série).
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g9:statproba:1}
Calculer la \omterm{def:g9:statproba:indicators}{moyenne}, la \omterm{def:g9:statproba:indicators}{médiane} et l'\omterm{def:g9:statproba:indicators}{étendue} de la série
$14,\ 9,\ 17,\ 9,\ 12,\ 11,\ 12$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:statproba:2}
Les températures à midi sur une semaine étaient $19,\ 21,\ 24,\ 18,\ 22,\
25,\ 19$ (degrés). Calculer la \omterm{def:g9:statproba:indicators}{moyenne} (au dixième) et la \omterm{def:g9:statproba:indicators}{médiane}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:statproba:3}
Le nombre de buts marqués par une équipe en $10$ matchs~:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccc}
buts & $0$ & $1$ & $2$ & $3$ \\
\hline
matchs & $2$ & $4$ & $3$ & $1$ \\
\end{tabular}
\end{center}
Calculer le nombre moyen de buts par match, et la \omterm{def:g9:statproba:indicators}{médiane}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:statproba:4}
On lance une fois un dé équilibré. Calculer la \omterm{def:g9:statproba:probability}{probabilité} de~: «~obtenir
un $3$~»~; «~obtenir un \omterm{def:g3:numbers:evenodd}{nombre pair}~»~; «~obtenir au moins un $3$~»~;
«~ne pas obtenir un $6$~».
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:statproba:5}
Une boîte contient $12$ boules~: $5$ blanches, $4$ rouges et $3$ vertes~;
on tire une boule au hasard. Calculer $\P(\text{blanche})$,
$\P(\text{rouge ou verte})$, et $\P(\text{pas blanche})$. Que remarque-t-on
sur les deux dernières~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:statproba:6}
On lance une pièce, puis un dé équilibré. Dessiner l'arbre (ou compter
les issues) et calculer la \omterm{def:g9:statproba:probability}{probabilité} d'obtenir pile \emph{et} un $6$~;
puis d'obtenir pile \emph{ou} un $6$ (ou les deux).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:statproba:7}
Dans le sac de \cref{ex:g9:statproba:tree} ($2$ rouges, $3$ noirs), les
deux tirages se font maintenant \emph{sans} remise. Dessiner le nouvel
arbre (attention~: les \omterm{def:g9:statproba:probability}{probabilités} du second niveau changent) et
calculer la \omterm{def:g9:statproba:probability}{probabilité} de tirer deux jetons noirs, puis deux jetons de
couleurs différentes.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:statproba:8}
Après $9$ matchs, une basketteuse \omterm{def:g9:statproba:indicators}{moyenne} $14$ points par match.
\begin{enumerate}
  \item Combien de points a-t-elle marqués au total~?
  \item Combien de points doit-elle marquer au $10$\textsuperscript{e} match pour
        \omterm{def:g9:statproba:indicators}{moyenne} $15$~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:statproba:9}
Un jeu~: lancer deux dés équilibrés~; on gagne si les deux résultats
sont égaux («~un double~»).
\begin{enumerate}
  \item Calculer la \omterm{def:g9:statproba:probability}{probabilité} de gagner.
  \item On joue deux fois (parties indépendantes). À l'aide d'un arbre,
        calculer la \omterm{def:g9:statproba:probability}{probabilité} de gagner au moins une fois.
\end{enumerate}
\end{exercise}
