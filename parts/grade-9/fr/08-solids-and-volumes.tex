\chapter{Solides et volumes}\label{ch:g9:solids}

La géométrie de l'espace commence par une petite famille de solides ---
prismes, cylindres, pyramides, cônes, sphères --- et deux questions~:
combien contiennent-ils (volume), et que voit-on lorsqu'on les coupe
(sections)~? Le chapitre se clôt sur un fait d'une grande importance
pratique~: homothétier un solide par $k$ multiplie son volume par $k^3$.

\section{Les solides classiques et leurs volumes}

\begin{theorem}[Formules de volume]\label{thm:g9:solids:volumes}
Noter $B$ l'aire de la base et $h$ la hauteur (la distance entre la base
et la face ou le sommet opposé).
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{l|l}
solide & volume \\
\hline
prisme, cylindre & $V = B \times h$ \\[5pt]
pyramide, cône & $V = \dfrac13\, B \times h$ \\[5pt]
sphère de rayon $r$ & $V = \dfrac43\,\pi r^3$ \\
\end{tabular}
\end{center}
Pour un cylindre et un cône de rayon $r$, l'aire de base est
$B = \pi r^2$~; l'aire de surface de la sphère est $4\pi r^2$.
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:volumes}
Une boîte de conserve cylindrique a pour rayon $4$ cm et pour hauteur
$10$ cm~:
\[
V = \pi r^2 h = \pi \times 16 \times 10 = 160\pi \approx 503 \text{ cm}^3
\approx 0.5 \text{ L}.
\]
Un cône de même base et même hauteur contient le tiers de cela~:
$\frac{160\pi}{3} \approx 168$ cm$^3$. Une sphère de rayon $4$ cm~:
$V = \frac43 \pi \times 64 = \frac{256\pi}{3} \approx 268$ cm$^3$.
\end{example}

\begin{example}[Une pyramide étape par étape]\label{ex:g9:solids:pyramid}
Une pyramide a une base carrée de côté $6$ m et une hauteur de $5$ m.
\begin{enumerate}
  \item Aire de base~: $B = 6^2 = 36$ m$^2$.
  \item Volume~: $V = \frac13 \times 36 \times 5 = 60$ m$^3$.
\end{enumerate}
Attention aux unités~: si le côté était donné en cm et la hauteur en m,
il faudrait convertir l'un d'eux d'abord.
\end{example}

\section{Sections par des plans}

\begin{proposition}[Sections des solides classiques]\label{prop:g9:solids:sections}
Couper un solide par un plan produit une figure plane, sa
\emph{section}\index{section}~:
\begin{itemize}
  \item un prisme ou un cylindre coupé \emph{parallèlement à sa base}
        donne une copie de la base, à toute hauteur~;
  \item une pyramide ou un cône coupé parallèlement à sa base donne une
        \emph{réduction} de la base~: à distance $d$ du sommet, le
        facteur d'échelle est $k = \frac{d}{h}$~;
  \item une sphère de rayon $r$ coupée par un plan à distance $d < r$
        du centre donne un cercle de rayon $\sqrt{r^2 - d^2}$ (par le
        théorème de Pythagore).
\end{itemize}
\end{proposition}
\admitted

\begin{example}\label{ex:g9:solids:section}
Un cône a pour rayon de base $6$ et pour hauteur $9$. La section par un
plan parallèle à la base, à distance $3$ du sommet, est un disque réduit
par $k = \frac39 = \frac13$~: son rayon est $6 \times \frac13 = 2$.

Une sphère de rayon $5$ est coupée par un plan à distance $3$ de son
centre~: la section est un cercle de rayon $\sqrt{25 - 9} = 4$.
\end{example}

\section{Homothétie des solides}

\begin{theorem}[Effet d'une homothétie sur longueurs, aires, volumes]\label{thm:g9:solids:scaling}
Lorsqu'un solide est agrandi ou réduit par le facteur d'échelle
$k > 0$~:
\begin{itemize}
  \item toutes les longueurs sont multipliées par $k$~;
  \item toutes les aires sont multipliées par $k^2$~;
  \item tous les volumes sont multipliés par $k^3$.
\end{itemize}
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:scaling}
Une voiture miniature à l'échelle $\frac{1}{18}$~: les longueurs sont
celles de la voiture réelle multipliées par $k = \frac{1}{18}$, la
surface peinte multipliée par $k^2 = \frac{1}{324}$, et le volume par
$k^3 = \frac{1}{5832}$.

Doubler le rayon d'une sphère ($k = 2$) multiplie son volume par
$2^3 = 8$~: vérifier sur la formule,
$\frac43\pi (2r)^3 = \frac43 \pi \times 8r^3$.
\end{example}

\begin{example}[Tronc de cône]\label{ex:g9:solids:truncated}
Un cône de hauteur $9$ et de rayon de base $6$ (volume
$\frac13 \pi \times 36 \times 9 = 108\pi$) est coupé à distance $3$ du
sommet, et le petit cône au-dessus de la coupe est enlevé. Le petit cône
est la réduction par $k = \frac13$, donc son volume est
$108\pi \times \left(\frac13\right)^3 = 4\pi$. Le solide restant (un
\emph{tronc de cône}) a pour volume $108\pi - 4\pi = 104\pi \approx 327$.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g9:solids:1}
Calculer les volumes~: un pavé (prisme rectangulaire) de dimensions
$4 \times 5 \times 12$~; un cylindre de rayon $3$ et de hauteur $7$
(valeur exacte avec $\pi$, puis arrondie à l'unité).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:solids:2}
Calculer le volume d'un cône de rayon $5$ et de hauteur $12$, et d'une
pyramide de base rectangulaire $8 \times 3$ et de hauteur $10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:solids:3}
Calculer le volume et l'aire de surface d'une sphère de rayon $6$
(valeurs exactes avec $\pi$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:solids:4}
Une sphère de rayon $10$ est coupée par un plan à distance $8$ de son
centre. Quel est le rayon du cercle de section~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:5}
Un cône a pour rayon de base $8$ et pour hauteur $12$. Un plan parallèle
à la base le coupe à distance $9$ du sommet. Calculer le rayon de la
section, puis le volume du petit cône au-dessus de la coupe.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:6}
Un verre cylindrique de rayon intérieur $3$ cm contient de l'eau jusqu'à
une hauteur de $10$ cm. Une bille de rayon $2$ cm y est entièrement
immergée. De combien le niveau d'eau monte-t-il~? (Le volume ajouté se
répartit sur l'aire de base du verre~; donner la montée exacte, puis
arrondir au millimètre.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:7}
Une recette remplit un moule sphérique de rayon $6$ cm. On ne dispose que
d'un moule sphérique de rayon $3$ cm. Combien de petits moules peut-on
remplir~? (Répondre sans calculer explicitement ni l'un ni l'autre
volume.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:8}
La Grande Pyramide de Gizeh a une base carrée de côté environ $230$ m et
une hauteur d'environ $147$ m. Estimer son volume, et l'exprimer en
millions de mètres cubes.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:solids:9}
Un entonnoir en forme de cône de rayon $6$ cm et de hauteur $12$ cm,
sommet en bas, est rempli d'eau jusqu'à la moitié de sa \emph{hauteur}
(mesurée depuis le sommet).
\begin{enumerate}
  \item Quelle fraction du volume de l'entonnoir est remplie~?
  \item Si au contraire on le remplit de la moitié de son \emph{volume}
        d'eau, montrer que la hauteur d'eau $d$ satisfait
        $\left(\frac{d}{12}\right)^3 = \frac12$, et donner $d$ au
        millimètre près ($\sqrt[3]{0.5} \approx 0.7937$).
\end{enumerate}
\end{exercise}
