\chapter{Trigonométrie dans le triangle rectangle}\label{ch:g9:trig}

Dans un triangle rectangle, une fois qu'on connaît un angle aigu, la
\emph{forme} du triangle est fixée --- seule sa taille peut varier. Les
rapports de ses côtés ne dépendent donc que de l'angle~: ces rapports
sont le cosinus, le sinus et la tangente. Avec le théorème de Pythagore,
ils permettent de calculer chaque côté et chaque angle d'un triangle
rectangle à partir de très peu d'informations.

\section{Le théorème de Pythagore, encore}

\begin{theorem}[Pythagore]\label{thm:g9:trig:pythagoras}
Dans un triangle $ABC$ rectangle en $A$, le carré de l'hypoténuse est la
somme des carrés des deux côtés de l'angle droit~:
\[
BC^2 = AB^2 + AC^2 .
\]
Réciproquement, si $BC^2 = AB^2 + AC^2$ dans un triangle, alors le
triangle est rectangle en $A$.
\end{theorem}
\admitted

\begin{example}\label{ex:g9:trig:pythagoras}
Un triangle rectangle a pour côtés de l'angle droit $AB = 5$ et
$AC = 12$. Alors
\[
BC^2 = 5^2 + 12^2 = 25 + 144 = 169,
\qquad\text{donc}\quad BC = \sqrt{169} = 13 .
\]
Pour trouver un \emph{côté de l'angle droit}~: si $BC = 10$ et
$AB = 6$, alors $AC^2 = BC^2 - AB^2 = 100 - 36 = 64$, donc $AC = 8$.
\end{example}

\section{Cosinus, sinus, tangente}

Dans un triangle rectangle, par rapport à un angle aigu $\theta$, les
trois côtés ont des noms~: l'\emph{hypoténuse} (opposée à l'angle droit,
le plus long côté), le côté \emph{opposé} à $\theta$, et le côté
\emph{adjacent} à $\theta$ (le côté de l'angle droit touchant $\theta$).

\begin{definition}[Rapports trigonométriques]\label{def:g9:trig:ratios}
Pour un angle aigu $\theta$ d'un triangle rectangle~:
\[
\cos\theta = \frac{\text{adjacent}}{\text{hypoténuse}},
\qquad
\sin\theta = \frac{\text{opposé}}{\text{hypoténuse}},
\qquad
\tan\theta = \frac{\text{opposé}}{\text{adjacent}} .
\]
Ces rapports\index{cosinus}\index{sinus}\index{tangente} ne dépendent
que de $\theta$, pas de la taille du triangle (tous les triangles
rectangles de même angle aigu sont des homothéties les uns des autres,
par Thalès).
\end{definition}

\begin{proposition}[Premières propriétés]\label{prop:g9:trig:properties}
Pour tout angle aigu $\theta$~:
\begin{enumerate}
  \item $0 < \cos\theta < 1$ et $0 < \sin\theta < 1$ (un côté de l'angle
        droit est plus court que l'hypoténuse)~;
  \item $\cos^2\theta + \sin^2\theta = 1$~;
  \item $\tan\theta = \dfrac{\sin\theta}{\cos\theta}$.
\end{enumerate}
\end{proposition}

\begin{proof}
Noter $a$ le côté adjacent, $o$ le côté opposé, $h$ l'hypoténuse d'un
triangle rectangle d'angle $\theta$.

1. $0 < a < h$ et $0 < o < h$, donc les deux rapports sont strictement
entre $0$ et $1$.

2. Par Pythagore, $a^2 + o^2 = h^2$. Diviser tout par $h^2$~:
\[
\cos^2\theta + \sin^2\theta
= \frac{a^2}{h^2} + \frac{o^2}{h^2}
= \frac{a^2 + o^2}{h^2} = 1 .
\]

3. $\dfrac{\sin\theta}{\cos\theta} = \dfrac{o/h}{a/h} = \dfrac oa
= \tan\theta$.
\end{proof}

\begin{method}[Trouver un côté]\label{met:g9:trig:side}
Pour calculer un côté inconnu lorsqu'un côté et un angle aigu sont
connus~:
\begin{enumerate}
  \item marquer l'angle connu et étiqueter les trois côtés (hypoténuse,
        opposé, adjacent) \emph{depuis cet angle}~;
  \item choisir le rapport (cos, sin ou tan) qui implique le côté connu
        et le côté cherché~;
  \item écrire l'équation et la résoudre~;
  \item contrôle de bon sens~: l'hypoténuse doit sortir la plus longue.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:trig:side}
Dans un triangle rectangle, l'hypoténuse mesure $8$ et un angle est
$35^\circ$~; calculer le côté qui lui est opposé. Le rapport impliquant
le côté opposé et l'hypoténuse est le sinus~:
\[
\sin 35^\circ = \frac{\text{opposé}}{8}
\quad\Longrightarrow\quad
\text{opposé} = 8 \sin 35^\circ \approx 8 \times 0.574 \approx 4.6 .
\]
Pour le côté adjacent, utiliser le cosinus~:
$8\cos 35^\circ \approx 8 \times 0.819 \approx 6.6$. Vérifier avec
Pythagore~: $4.6^2 + 6.6^2 \approx 21 + 43 \approx 64 = 8^2$.
\end{example}

\begin{method}[Trouver un angle]\label{met:g9:trig:angle}
Lorsque deux côtés sont connus, calculer le rapport qu'ils forment,
l'identifier comme cosinus, sinus ou tangente de l'angle inconnu, et
appliquer la fonction inverse de la calculatrice ($\cos^{-1}$,
$\sin^{-1}$ ou $\tan^{-1}$) au rapport.
\end{method}

\begin{example}\label{ex:g9:trig:angle}
Une échelle de $5$ m s'appuie contre un mur, son pied à $1.4$ m du mur.
L'angle $\theta$ entre l'échelle et le sol satisfait
\[
\cos\theta = \frac{1.4}{5} = 0.28,
\qquad\text{donc}\quad
\theta = \cos^{-1}(0.28) \approx 73.7^\circ .
\]
\end{example}

\begin{example}[Angles remarquables]\label{ex:g9:trig:special}
Deux triangles donnent des valeurs exactes. La moitié d'un carré de côté
$1$ (un triangle rectangle isocèle) a pour hypoténuse $\sqrt2$ et des
angles de $45^\circ$~:
\[
\cos 45^\circ = \sin 45^\circ = \frac{1}{\sqrt2} = \frac{\sqrt2}{2},
\qquad \tan 45^\circ = 1 .
\]
La moitié d'un triangle équilatéral de côté $1$ donne les angles
$30^\circ$ et $60^\circ$~:
\[
\cos 60^\circ = \sin 30^\circ = \frac12,
\qquad
\cos 30^\circ = \sin 60^\circ = \frac{\sqrt3}{2}.
\]
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g9:trig:1}
Un triangle rectangle a pour côtés de l'angle droit $9$ et $12$.
Calculer son hypoténuse. Un autre a pour hypoténuse $17$ et un côté de
l'angle droit $8$~: calculer l'autre côté.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:trig:2}
Un triangle a pour côtés $7$, $24$ et $25$. Est-il rectangle~? Même
question pour les côtés $5$, $6$ et $8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:trig:3}
Dans un triangle $DEF$ rectangle en $D$, avec l'angle en $E$ noté
$\theta$~: quel côté est l'hypoténuse~? Quel côté est opposé à
$\theta$~? Écrire $\cos\theta$, $\sin\theta$ et $\tan\theta$ comme
rapports des côtés $DE$, $DF$, $EF$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:trig:4}
Dans un triangle rectangle, l'hypoténuse mesure $10$ et un angle aigu est
$28^\circ$. Calculer les deux côtés de l'angle droit (arrondir au
dixième~; $\cos 28^\circ \approx 0.883$, $\sin 28^\circ \approx 0.469$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:trig:5}
Dans un triangle rectangle, le côté adjacent à l'angle $\theta$ mesure
$6$ et le côté opposé mesure $4.5$. Calculer $\tan\theta$, puis $\theta$
(arrondir au degré~; $\tan^{-1}(0.75) \approx 36.9^\circ$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:trig:6}
Un drone vole à une altitude de $120$ m. Depuis un observateur au sol, on
le voit sous un angle d'élévation de $32^\circ$. À quelle distance
horizontale de l'observateur se trouve le drone
($\tan 32^\circ \approx 0.625$)~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:trig:7}
Une rampe d'accès doit monter de $1.2$ m avec un angle d'au plus
$5^\circ$ par rapport à l'horizontale. Quelle longueur minimale le long
de la pente la rampe doit-elle avoir ($\sin 5^\circ \approx 0.0872$)~?
Arrondir au dixième de mètre supérieur.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:trig:8}
En utilisant \cref{prop:g9:trig:properties}~: un angle aigu $\theta$
satisfait $\cos\theta = 0.6$. Calculer $\sin\theta$ sans trouver
$\theta$, puis $\tan\theta$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:trig:9}
Justifier les valeurs exactes de \cref{ex:g9:trig:special}~: dans le
triangle rectangle isocèle de côtés de l'angle droit $1$, calculer
l'hypoténuse~; dans le triangle équilatéral de côté $1$, calculer la
hauteur, et en déduire le cosinus et le sinus de $30^\circ$ et
$60^\circ$.
\end{exercise}
