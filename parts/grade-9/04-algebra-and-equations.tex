\chapter{Algebra and Equations}\label{ch:g9:algebra}

Letters stand for numbers, and computing with letters proves facts about
\emph{all} numbers at once. This chapter practices \omterm{def:g9:algebra:expand}{expanding} and
factoring --- including the three identities that will follow you through
all of mathematics --- and solves first-degree \omterm{def:g8:equations:def}{equations}, \omterm{def:g2:mult:def}{product}
\omterm{def:g8:equations:def}{equations}, and simple inequalities, always one justified step at a time.

\section{Expanding}

\begin{definition}[Expanding]\label{def:g9:algebra:expand}
\emph{Expanding}\index{expanding} means transforming a \omterm{def:g2:mult:def}{product} into a
\omterm{def:g1:addition:def}{sum}, using distributivity:
\[
k(a + b) = ka + kb,
\qquad
(a + b)(c + d) = ac + ad + bc + bd .
\]
\end{definition}

\begin{example}\label{ex:g9:algebra:expand}
Every term of the first bracket multiplies every term of the second:
\begin{align*}
(x + 3)(2x + 5)
&= x \times 2x + x \times 5 + 3 \times 2x + 3 \times 5 \\
&= 2x^2 + 5x + 6x + 15 \\
&= 2x^2 + 11x + 15 .
\end{align*}
Signs travel with the terms:
\begin{align*}
(3x - 2)(x - 4)
&= 3x^2 - 12x - 2x + 8
= 3x^2 - 14x + 8 .
\end{align*}
\end{example}

\begin{theorem}[The three identities]\label{thm:g9:algebra:identities}
For all numbers $a$ and $b$:
\[
(a+b)^2 = a^2 + 2ab + b^2,
\qquad
(a-b)^2 = a^2 - 2ab + b^2,
\qquad
(a+b)(a-b) = a^2 - b^2 .
\]
\end{theorem}

\begin{proof}
Expand each left-hand side. For the first:
$(a+b)(a+b) = a^2 + ab + ba + b^2 = a^2 + 2ab + b^2$. For the second,
replace $b$ by $-b$ in the first. For the third:
$(a+b)(a-b) = a^2 - ab + ba - b^2 = a^2 - b^2$ (the cross terms cancel).
\end{proof}

\[
(x + 5)^2 = x^2 + 10x + 25, \qquad

\]
Mental arithmetic: $102^2 = (100 + 2)^2 = 10000 + 400 + 4 = 10404$, and
$98 \times 102 = (100-2)(100+2) = 10000 - 4 = 9996$.
\end{example}

\section{Factoring}

\begin{method}[Factoring]\label{met:g9:algebra:factor}
To factor an expression, try in order:
\begin{enumerate}
  \item \emph{common factor}: $ka + kb = k(a + b)$, where $k$ may itself
        be a bracket;
  \item \emph{\omterm{ex:g1:subtraction:difference}{difference} of squares}: $a^2 - b^2 = (a+b)(a-b)$;
  \item \emph{perfect square}: $a^2 + 2ab + b^2 = (a+b)^2$ (and with
        $-2ab$, $(a-b)^2$).
\end{enumerate}
Always check by \omterm{def:g9:algebra:expand}{expanding} the result.
\end{method}

\begin{example}\label{ex:g9:algebra:factor}
\emph{Common factor:}
$15x^2 + 10x = 5x \times 3x + 5x \times 2 = 5x(3x + 2)$.

\emph{Common bracket:}
\[
(x+2)(x-1) + (x+2)(3x+4) = (x+2)\bigl[(x-1) + (3x+4)\bigr]
= (x+2)(4x+3).
\]

\emph{\omterm{ex:g1:subtraction:difference}{Difference} of squares:}
$x^2 - 81 = (x+9)(x-9)$, and
\[
(2x+1)^2 - 9 = (2x+1)^2 - 3^2 = (2x+1+3)(2x+1-3) = (2x+4)(2x-2).
\]
Each factor still has a common factor: $(2x+4)(2x-2) = 2(x+2) \times
2(x-1) = 4(x+2)(x-1)$.
\end{example}

\section{Equations}

\begin{method}[First-degree equations]\label{met:g9:algebra:firstdegree}
To solve an \omterm{def:g8:equations:def}{equation} like $5x - 3 = 2x + 9$:
\begin{enumerate}
  \item gather the $x$-terms on one side, the numbers on the other, by
        adding the same quantity to both sides;
  \item reduce each side;
  \item divide both sides by the coefficient of $x$;
  \item check the solution in the original \omterm{def:g8:equations:def}{equation}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:algebra:firstdegree}
\begin{align*}
5x - 3 &= 2x + 9 \\
5x - 2x &= 9 + 3 && \text{(add $3 - 2x$ to both sides)}\\
3x &= 12 \\
x &= 4 && \text{(divide by 3).}
\end{align*}
Check: $5 \times 4 - 3 = 17$ and $2 \times 4 + 9 = 17$. The solution is
$4$.

With brackets, expand first: $3(x - 2) = x + 8$ becomes $3x - 6 = x + 8$,
so $2x = 14$ and $x = 7$.
\end{example}

\begin{theorem}[Zero-product rule]\label{thm:g9:algebra:zeroproduct}
A \omterm{def:g2:mult:def}{product} is \omterm{def:g1:counting:zero}{zero} if and only if at least one factor is \omterm{def:g1:counting:zero}{zero}:
\[
A \times B = 0
\quad\Longleftrightarrow\quad
A = 0 \ \text{ or } \ B = 0 .
\]
\end{theorem}

\begin{proof}
If a factor is \omterm{def:g1:counting:zero}{zero} the \omterm{def:g2:mult:def}{product} clearly is. Conversely if $A B = 0$ with
$A \neq 0$, dividing both sides by $A$ gives $B = 0$.
\end{proof}

\begin{example}\label{ex:g9:algebra:zeroproduct}
Solve $(x - 3)(2x + 10) = 0$: either $x - 3 = 0$ or $2x + 10 = 0$, so the
solutions are $3$ and $-5$.

Solve $x^2 - 49 = 0$: factor first, $(x-7)(x+7) = 0$, solutions $7$ and
$-7$.

Solve $x^2 = 3x$: never divide by $x$! Move and factor:
$x^2 - 3x = x(x - 3) = 0$, solutions $0$ and $3$.
\end{example}

\section{Inequalities}

\begin{proposition}[Rules for inequalities]\label{prop:g9:algebra:ineq}
An inequality is preserved when adding the same number to both sides, and
when multiplying both sides by the same \emph{positive} number.
Multiplying both sides by a \emph{negative} number reverses the
inequality sign.
\end{proposition}

\begin{proof}
If $a \leq b$ then $b - a \geq 0$. Adding $c$: $(b+c) - (a+c) = b - a
\geq 0$, so $a + c \leq b + c$. Multiplying by $c > 0$:
$bc - ac = (b-a)c \geq 0$, so $ac \leq bc$. Multiplying by $c < 0$:
$(b-a)c \leq 0$, so $ac \geq bc$: the order is reversed.
\end{proof}

\begin{example}\label{ex:g9:algebra:ineq}
Solve $7 - 2x < 15$:
\begin{align*}
7 - 2x &< 15 \\
-2x &< 8 && \text{(subtract 7)} \\
x &> -4 && \text{(divide by $-2$: reverse the sign).}
\end{align*}
The solutions are all numbers greater than $-4$: on a number line, a
hollow dot at $-4$ and shading to the right.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g9:algebra:1}
Expand and reduce:
\[
4(3x - 2), \qquad
(x + 6)(x + 2), \qquad
(2x - 3)(x + 5), \qquad
(x - 1)(x - 9).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:algebra:2}
Expand using the identities:
\[
(x + 8)^2, \qquad
(3x - 4)^2, \qquad
(x + 10)(x - 10), \qquad
(5x + 2)(5x - 2).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:algebra:3}
Compute mentally, using an identity: $101^2$; \quad $59^2$
(write $59 = 60 - 1$); \quad $53 \times 47$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:algebra:4}
Factor:
\[
9x + 12, \qquad
x^2 - 64, \qquad
x^2 + 14x + 49, \qquad
18x^2 - 6x .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:algebra:5}
Solve, writing every step:
\[
4x + 7 = 19, \qquad
6x - 5 = 2x + 11, \qquad
5(x - 3) = 3x + 1 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:algebra:6}
Solve using the \omterm{def:g2:mult:def}{zero-product} rule:
\[
(x + 4)(3x - 12) = 0, \qquad
x^2 - 121 = 0, \qquad
(2x - 1)^2 = 0, \qquad
x^2 = 8x .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:algebra:7}
Factor $E = (3x + 2)^2 - 25$, then solve $E = 0$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:algebra:8}
Solve the inequalities and represent the solutions on a number line:
\[
3x - 4 \geq 8, \qquad
5 - 2x > 1, \qquad
4(x + 1) \leq x - 5 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:algebra:9}
I think of a number, multiply it by $3$, add $7$, and get the same result
as if I had multiplied it by $5$ and subtracted $9$. What is my number?
(Set up an \omterm{def:g8:equations:def}{equation} and solve it.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:algebra:10}
Let $n$ be a whole number.
\begin{enumerate}
  \item Expand $(n + 1)(n - 1)$ and deduce that the \omterm{def:g2:mult:def}{product} of the two
        neighbors of $n$ is always $n^2 - 1$ (e.g.\ $24 \times 26 =
        25^2 - 1 = 624$).
  \item Show that the \omterm{def:g1:addition:def}{sum} of three consecutive integers is always a
        \omterm{def:g9:arith:divisor}{multiple} of $3$.
\end{enumerate}
\end{exercise}
