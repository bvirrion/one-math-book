\chapter{Algebra and Equations}\label{ch:g9:algebra}

Letters stand for numbers, and computing with letters proves facts about
\emph{all} numbers at once. This chapter practices \omterm{def:g9:algebra:expand}{expanding} and
factoring --- including the three identities that will follow you through
all of mathematics --- and solves first-degree \omterm{def:g8:equations:def}{equations}, \omterm{def:g2:mult:def}{product}
\omterm{def:g8:equations:def}{equations}, and simple inequalities, always one justified step at a time.

\section{Expanding}

\begin{definition}[Expanding]\label{def:g9:algebra:expand}
\emph{Expanding}\index{expanding} means transforming a \omterm{def:g2:mult:def}{product} into a
\omterm{def:g1:addition:def}{sum}, using distributivity:
\[
k(a + b) = ka + kb,
\qquad
(a + b)(c + d) = ac + ad + bc + bd .
\]
\end{definition}

\begin{example}\label{ex:g9:algebra:expand}
Every term of the first bracket multiplies every term of the second:
\begin{align*}
(x + 3)(2x + 5)
&= x \times 2x + x \times 5 + 3 \times 2x + 3 \times 5 \\
&= 2x^2 + 5x + 6x + 15 \\
&= 2x^2 + 11x + 15 .
\end{align*}
Signs travel with the terms:
\begin{align*}
(3x - 2)(x - 4)
&= 3x^2 - 12x - 2x + 8
= 3x^2 - 14x + 8 .
\end{align*}
\end{example}

\begin{theorem}[The three identities]\label{thm:g9:algebra:identities}
For all numbers $a$ and $b$:
\[
(a+b)^2 = a^2 + 2ab + b^2,
\qquad
(a-b)^2 = a^2 - 2ab + b^2,
\qquad
(a+b)(a-b) = a^2 - b^2 .
\]
\end{theorem}

\begin{proof}
Expand each left-hand side. For the first:
$(a+b)(a+b) = a^2 + ab + ba + b^2 = a^2 + 2ab + b^2$. For the second,
replace $b$ by $-b$ in the first. For the third:
$(a+b)(a-b) = a^2 - ab + ba - b^2 = a^2 - b^2$ (the cross terms cancel).
\end{proof}

\begin{example}\label{ex:g9:algebra:identities}
\[
(x + 5)^2 = x^2 + 10x + 25, \qquad
(3x - 1)^2 = 9x^2 - 6x + 1, \qquad
(2x + 7)(2x - 7) = 4x^2 - 49 .
\]
Mental arithmetic: $102^2 = (100 + 2)^2 = 10000 + 400 + 4 = 10404$, and
$98 \times 102 = (100-2)(100+2) = 10000 - 4 = 9996$.
\end{example}

\section{Factoring}

\begin{method}[Factoring]\label{met:g9:algebra:factor}
To factor an expression, try in order:
\begin{enumerate}
  \item \emph{common factor}: $ka + kb = k(a + b)$, where $k$ may itself
        be a bracket;
  \item \emph{\omterm{ex:g1:subtraction:difference}{difference} of squares}: $a^2 - b^2 = (a+b)(a-b)$;
  \item \emph{perfect square}: $a^2 + 2ab + b^2 = (a+b)^2$ (and with
        $-2ab$, $(a-b)^2$).
\end{enumerate}
Always check by \omterm{def:g9:algebra:expand}{expanding} the result.
\end{method}

\begin{example}\label{ex:g9:algebra:factor}
\emph{Common factor:}
$15x^2 + 10x = 5x \times 3x + 5x \times 2 = 5x(3x + 2)$.

\emph{Common bracket:}
\[
(x+2)(x-1) + (x+2)(3x+4) = (x+2)\bigl[(x-1) + (3x+4)\bigr]
= (x+2)(4x+3).
\]

\emph{\omterm{ex:g1:subtraction:difference}{Difference} of squares:}
$x^2 - 81 = (x+9)(x-9)$, and
\[
(2x+1)^2 - 9 = (2x+1)^2 - 3^2 = (2x+1+3)(2x+1-3) = (2x+4)(2x-2).
\]
Each factor still has a common factor: $(2x+4)(2x-2) = 2(x+2) \times
2(x-1) = 4(x+2)(x-1)$.
\end{example}

\section{Equations}

\begin{method}[First-degree equations]\label{met:g9:algebra:firstdegree}
To solve an \omterm{def:g8:equations:def}{equation} like $5x - 3 = 2x + 9$:
\begin{enumerate}
  \item gather the $x$-terms on one side, the numbers on the other, by
        adding the same quantity to both sides;
  \item reduce each side;
  \item divide both sides by the coefficient of $x$;
  \item check the solution in the original \omterm{def:g8:equations:def}{equation}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:algebra:firstdegree}
\begin{align*}
5x - 3 &= 2x + 9 \\
5x - 2x &= 9 + 3 && \text{(add $3 - 2x$ to both sides)}\\
3x &= 12 \\
x &= 4 && \text{(divide by 3).}
\end{align*}
Check: $5 \times 4 - 3 = 17$ and $2 \times 4 + 9 = 17$. The solution is
$4$.

With brackets, expand first: $3(x - 2) = x + 8$ becomes $3x - 6 = x + 8$,
so $2x = 14$ and $x = 7$.
\end{example}

\begin{theorem}[Zero-product rule]\label{thm:g9:algebra:zeroproduct}
A \omterm{def:g2:mult:def}{product} is \omterm{def:g1:counting:zero}{zero} if and only if at least one factor is \omterm{def:g1:counting:zero}{zero}:
\[
A \times B = 0
\quad\Longleftrightarrow\quad
A = 0 \ \text{ or } \ B = 0 .
\]
\end{theorem}

\begin{proof}
If a factor is \omterm{def:g1:counting:zero}{zero} the \omterm{def:g2:mult:def}{product} clearly is. Conversely if $A B = 0$ with
$A \neq 0$, dividing both sides by $A$ gives $B = 0$.
\end{proof}

\begin{example}\label{ex:g9:algebra:zeroproduct}
Solve $(x - 3)(2x + 10) = 0$: either $x - 3 = 0$ or $2x + 10 = 0$, so the
solutions are $3$ and $-5$.

Solve $x^2 - 49 = 0$: factor first, $(x-7)(x+7) = 0$, solutions $7$ and
$-7$.

Solve $x^2 = 3x$: never divide by $x$! Move and factor:
$x^2 - 3x = x(x - 3) = 0$, solutions $0$ and $3$.
\end{example}

\section{Inequalities}

\begin{proposition}[Rules for inequalities]\label{prop:g9:algebra:ineq}
An inequality is preserved when adding the same number to both sides, and
when multiplying both sides by the same \emph{positive} number.
Multiplying both sides by a \emph{negative} number reverses the
inequality sign.
\end{proposition}

\begin{proof}
If $a \leq b$ then $b - a \geq 0$. Adding $c$: $(b+c) - (a+c) = b - a
\geq 0$, so $a + c \leq b + c$. Multiplying by $c > 0$:
$bc - ac = (b-a)c \geq 0$, so $ac \leq bc$. Multiplying by $c < 0$:
$(b-a)c \leq 0$, so $ac \geq bc$: the order is reversed.
\end{proof}

\begin{example}\label{ex:g9:algebra:ineq}
Solve $7 - 2x < 15$:
\begin{align*}
7 - 2x &< 15 \\
-2x &< 8 && \text{(subtract 7)} \\
x &> -4 && \text{(divide by $-2$: reverse the sign).}
\end{align*}
The solutions are all numbers greater than $-4$: on a number line, a
hollow dot at $-4$ and shading to the right.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g9:algebra:1}
Expand and reduce:
\[
4(3x - 2), \qquad
(x + 6)(x + 2), \qquad
(2x - 3)(x + 5), \qquad
(x - 1)(x - 9).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:algebra:2}
Expand using the identities:
\[
(x + 8)^2, \qquad
(3x - 4)^2, \qquad
(x + 10)(x - 10), \qquad
(5x + 2)(5x - 2).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:algebra:3}
Compute mentally, using an identity: $101^2$; \quad $59^2$
(write $59 = 60 - 1$); \quad $53 \times 47$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:algebra:4}
Factor:
\[
9x + 12, \qquad
x^2 - 64, \qquad
x^2 + 14x + 49, \qquad
18x^2 - 6x .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:algebra:5}
Solve, writing every step:
\[
4x + 7 = 19, \qquad
6x - 5 = 2x + 11, \qquad
5(x - 3) = 3x + 1 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:algebra:6}
Solve using the \omterm{def:g2:mult:def}{zero-product} rule:
\[
(x + 4)(3x - 12) = 0, \qquad
x^2 - 121 = 0, \qquad
(2x - 1)^2 = 0, \qquad
x^2 = 8x .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:algebra:7}
Factor $E = (3x + 2)^2 - 25$, then solve $E = 0$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:algebra:8}
Solve the inequalities and represent the solutions on a number line:
\[
3x - 4 \geq 8, \qquad
5 - 2x > 1, \qquad
4(x + 1) \leq x - 5 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:algebra:9}
I think of a number, multiply it by $3$, add $7$, and get the same result
as if I had multiplied it by $5$ and subtracted $9$. What is my number?
(Set up an \omterm{def:g8:equations:def}{equation} and solve it.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:algebra:10}
Let $n$ be a whole number.
\begin{enumerate}
  \item Expand $(n + 1)(n - 1)$ and deduce that the \omterm{def:g2:mult:def}{product} of the two
        neighbors of $n$ is always $n^2 - 1$ (e.g.\ $24 \times 26 =
        25^2 - 1 = 624$).
  \item Show that the \omterm{def:g1:addition:def}{sum} of three consecutive integers is always a
        \omterm{def:g9:arith:divisor}{multiple} of $3$.
\end{enumerate}
\end{exercise}

\section{Problem: The algebra of digits}

\begin{problem}[{Weekend problem --- why the divisibility rules
work, the 1089 trick unmasked, and mental arithmetic that looks
like sorcery}]
\label{pb:g9:algebra:1}
Grade school taught you \emph{rules} (a number is \omterm{def:g6:wholes:divisible}{divisible} by
$9$ when its digit \omterm{def:g1:addition:def}{sum} is); magicians perform \emph{tricks}
(everyone ends on $1089$); markets know \emph{shortcuts}
($35^2 = 1225$, instantly). All of it is the same secret: a
number with digits $a$, $b$, $c$ is the expression
$100a + 10b + c$, and this chapter's algebra
(\cref{thm:g9:algebra:identities}, \omterm{def:g9:algebra:expand}{expanding}, factoring) can take
it apart. By the end of this problem you will have proved the
rules, unmasked the trick, and learned the sorcery.

\medskip
\noindent\textbf{Part I --- Digits under the microscope.}
\begin{enumerate}
  \item A two-digit number with tens digit $a$ and units digit
        $b$ is $10a + b$. Expand and factor the \omterm{ex:g1:subtraction:difference}{difference}
        between the number and its reversal,
        $(10a + b) - (10b + a)$, and state the discovery. Check
        it on $72 - 27$.
  \item Prove the rule of nine for three-digit numbers: verify
        the identity
        \[
        100a + 10b + c = 9\,(11a + b) + (a + b + c),
        \]
        and explain why it shows that a number and its digit \omterm{def:g1:addition:def}{sum}
        leave the \emph{same \omterm{def:g3:division:remainder}{remainder}} in the division by $9$
        --- so one is \omterm{def:g6:wholes:divisible}{divisible} by $9$ exactly when the other is
        (\cref{prop:g9:arith:rules}, now proved).
  \item Deduce the rule of three from the same identity, in one
        sentence.
  \item ``Casting out nines'', the accountants' check: to test
        $47 \times 86 = 4\,042$, replace each number by the
        digit \omterm{def:g1:addition:def}{sum} of its digit \omterm{def:g1:addition:def}{sum}: $47 \to 11 \to 2$,
        $86 \to 14 \to 5$, and $2 \times 5 = 10 \to 1$; then
        $4\,042 \to 10 \to 1$: consistent. Explain why a correct
        \omterm{def:g2:mult:def}{product} must pass the test (write $47 = 9k + 2$,
        $86 = 9l + 5$ and expand), and find a wrong result that
        the test fails to catch (why are swapped digits
        invisible to it?).
  \item Prove the rule of eleven for four-digit numbers: verify
        \[
        1000a + 100b + 10c + d
        = 11\,(91a + 9b + c) + (- a + b - c + d),
        \]
        and use it to decide whether $2\,926$ is \omterm{def:g6:wholes:divisible}{divisible} by
        $11$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The 1089 trick.}
The trick: think of a three-digit number whose first and last
digits differ by at least $2$; reverse it; subtract the smaller
from the larger; reverse the result; add the last two numbers.
The magician announces: \emph{1089}.
\begin{enumerate}[resume]
  \item Perform the trick on $742$ and on one number of your
        choice.
  \item Let the number have digits $a$, $b$, $c$ with $a > c$.
        Factor the first subtraction,
        $(100a + 10b + c) - (100c + 10b + a)$, and list all the
        values it can take when $a - c$ runs from $2$ to $9$.
  \item Each of those \omterm{def:g5:division:multiple}{multiples} of $99$ has a striking digit
        pattern. Writing $m = a - c$, verify the identity
        \[
        99m = 100\,(m - 1) + 90 + (10 - m),
        \]
        and read off the three digits of $99m$.
  \item Add $99m$ to its own reversal, using question 8's
        digits, and prove that the total is $1089$ --- whatever
        $m$ is. The trick is dead; long live the algebra.
  \item Why did the magician require the first and last digits
        to differ by at least $2$? Examine the cases $a = c$ and
        $a - c = 1$ (where the subtraction gives $99$, which
        must be written $099$ for the reversal to work), and
        state the small print that keeps the trick honest.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Sorcery for daily use.}
\begin{enumerate}[resume]
  \item The birthday trick: ``multiply your birth month by $5$,
        add $7$, double, add your birth day, subtract $14$.''
        Show that the result is always $10m + d$ --- month and
        day, readable at a glance. (Try it on someone.)
  \item You redesign the trick as: ``multiply the month by $5$,
        add $9$, double, add the day.'' What number must the
        spectator now subtract at the end for the same
        month--day readout?
  \item The missing-digit trick: the spectator picks any number,
        subtracts its digit \omterm{def:g1:addition:def}{sum} (question 2 says the result is
        always a \omterm{def:g9:arith:divisor}{multiple} of $9$ --- explain why for a
        three-digit number), then crosses out one \emph{nonzero}
        digit and reads you the remaining digits. Explain how
        you recover the crossed-out digit, and why $0$ had to be
        excluded.
  \item Market sorcery: prove the identity
        $(10t + 5)^2 = 100\,t(t+1) + 25$, state the rule for
        squaring any number ending in $5$, and compute $45^2$,
        $85^2$ and $105^2$ in your head.
  \item Finale, extending \cref{exo:g9:algebra:10}: prove that
        the \omterm{def:g1:addition:def}{sum} of five consecutive integers is always a
        \omterm{def:g9:arith:divisor}{multiple} of $5$, but that the \omterm{def:g1:addition:def}{sum} of \emph{four}
        consecutive integers is \emph{never} a \omterm{def:g9:arith:divisor}{multiple} of $4$.
        (Two short computations --- and one moral about what
        algebra does to the words ``always'' and ``never''.)
\end{enumerate}
\end{problem}
