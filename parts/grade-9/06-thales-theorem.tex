\chapter{Thales' Theorem}\label{ch:g9:thales}

How do you measure the height of a pyramid without climbing it? Thales
of Miletus compared its shadow with the shadow of a stick. His theorem
--- parallel lines cut segments in proportional pieces --- is the
mathematical heart of scale models, maps, and enlargements, and one of
the oldest theorems with a name.

\section{The theorem}

\begin{theorem}[Thales]\label{thm:g9:thales:direct}
Let two lines meet at a point $A$. Take $M$ on the first line and $N$
on the second, with $B$ further along the first and $C$ further along
the second. If the lines $(MN)$ and $(BC)$ are \emph{parallel}, then the
triangles $AMN$ and $ABC$ have proportional sides:
\[
\frac{AM}{AB} = \frac{AN}{AC} = \frac{MN}{BC}.
\]
\end{theorem}
\admitted

\begin{method}[Computing a length with Thales]\label{met:g9:thales:compute}
\begin{enumerate}
  \item Identify the point $A$ where the two lines cross and the two
        parallel lines; name the two triangles;
  \item write the three equal ratios, always ``small triangle over large
        triangle'';
  \item keep the equality of the two ratios involving the three known
        lengths and the unknown one;
  \item solve the resulting proportion by cross-multiplication.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:thales:compute}
In the left configuration above, suppose $AM = 3$, $AB = 5$, $AN = 2.4$
and $MN = 3.3$, and let us compute $AC$ and $BC$. Thales gives
\[
\frac{AM}{AB} = \frac{AN}{AC} = \frac{MN}{BC},
\qquad\text{i.e.}\qquad
\frac35 = \frac{2.4}{AC} = \frac{3.3}{BC}.
\]
From $\frac35 = \frac{2.4}{AC}$, cross-multiplying:
$3 \times AC = 5 \times 2.4 = 12$, so $AC = 4$. From
$\frac35 = \frac{3.3}{BC}$: $3 \times BC = 16.5$, so $BC = 5.5$.
\end{example}

\begin{example}[Thales in real life]\label{ex:g9:thales:stick}
A vertical stick of height $1$ m casts a shadow of $1.5$ m, while a tree
casts a shadow of $12$ m at the same moment. Sun rays are parallel, so
the stick-and-shadow and tree-and-shadow triangles are in Thales
configuration:
\[
\frac{\text{height of tree}}{1} = \frac{12}{1.5},
\qquad\text{so the tree is } 8 \text{ m tall.}
\]
\end{example}

\section{The converse}

\begin{theorem}[Converse of Thales]\label{thm:g9:thales:converse}
With the points placed as in \cref{thm:g9:thales:direct} ($M$ on $(AB)$,
$N$ on $(AC)$, in the same order on both lines):
if
\[
\frac{AM}{AB} = \frac{AN}{AC},
\]
then the lines $(MN)$ and $(BC)$ are parallel.
\end{theorem}
\admitted

\begin{method}[Proving or disproving parallelism]\label{met:g9:thales:converse}
\begin{enumerate}
  \item Compute separately the two ratios $\dfrac{AM}{AB}$ and
        $\dfrac{AN}{AC}$ (as fractions, not roundings);
  \item if they are equal \emph{and} the points are in the same order on
        the two lines, the lines are parallel (converse of Thales);
  \item if they differ, the lines are \emph{not} parallel --- because if
        they were, Thales' theorem would force the ratios to be equal.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:thales:converse}
On line 1, $AM = 4$ and $AB = 10$; on line 2, $AN = 6$ and $AC = 15$.
Then
\[
\frac{AM}{AB} = \frac{4}{10} = \frac25,
\qquad
\frac{AN}{AC} = \frac{6}{15} = \frac25 :
\]
equal ratios, same order: $(MN) \parallel (BC)$.

With $AC = 14$ instead: $\frac{6}{14} = \frac37 \neq \frac25$, so the
lines are not parallel.
\end{example}

\section{Enlargement and reduction}

\begin{definition}[Scaling a figure]\label{def:g9:thales:scaling}
Enlarging or reducing a figure by the \emph{scale
factor}\index{scale factor} $k > 0$ means multiplying \emph{all} its
lengths by $k$: an \emph{enlargement} when $k > 1$, a \emph{reduction}
when $k < 1$. In Thales' configuration, the triangle $AMN$ is the
reduction of $ABC$ by the factor $k = \frac{AM}{AB}$.
\end{definition}

\begin{proposition}[Effect on angles and areas]\label{prop:g9:thales:scaling}
Scaling by the factor $k$ preserves angles and the shape of figures,
multiplies every length by $k$, and multiplies every \emph{area} by
$k^2$.
\end{proposition}
\admitted

\begin{omfigure}
\begin{tikzpicture}[scale=0.72]
  \draw[thick, omDef, fill=omDef!8] (0,0) rectangle (2,1.2);
  \draw[thick, omThm, fill=omThm!8] (3.4,0) rectangle (7.4,2.4);
  \node[font=\small, omDef] at (1,-0.45) {$2 \times 1.2$};
  \node[font=\small, omThm] at (5.4,-0.45) {$4 \times 2.4$};
  \node[font=\small] at (1,0.6) {area $2.4$};
  \node[font=\small] at (5.4,1.2) {area $9.6$};
\end{tikzpicture}

{\small Doubling the lengths ($k = 2$) multiplies the area by
$k^2 = 4$: four copies of the small rectangle tile the large one.}
\end{omfigure}

\begin{example}
A photo of $10 \times 15$ cm is enlarged with scale factor $k = 3$: the
print measures $30 \times 45$ cm. Its area goes from $150$ cm$^2$ to
$150 \times 9 = 1350$ cm$^2$ --- $9$ times larger, not $3$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g9:thales:1}
The lines $(MN)$ and $(BC)$ are parallel, with $M$ on $[AB]$ and $N$ on
$[AC]$; $AM = 2$, $AB = 6$, $AN = 3$, $BC = 9$. Compute $AC$ and $MN$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:thales:2}
Same configuration: $AM = 5$, $MB = 3$ (careful: $MB$, not $AB$!),
$AN = 4$. Compute $AC$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:thales:3}
In a butterfly configuration, $A$ is between $M$ and $B$ and between $N$
and $C$, with $(MN) \parallel (BC)$; $AM = 3$, $AB = 7.5$, $AN = 2$,
$MN = 2.6$. Compute $AC$ and $BC$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:thales:4}
$M$ is on $[AB]$ with $AM = 6$, $AB = 8$; $N$ is on $[AC]$ with $AN = 9$,
$AC = 12$. Are the lines $(MN)$ and $(BC)$ parallel?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:5}
Same as above with $AM = 4$, $AB = 6$, $AN = 5$, $AC = 8$. Are $(MN)$
and $(BC)$ parallel? Justify carefully.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:6}
A $1.8$ m tall person stands $2$ m away from a street lamp's base;
their shadow measures $3$ m. Draw the Thales configuration formed by the
lamp, the person, and the tip of the shadow, and compute the height of
the lamp.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:7}
A map has scale $1 : 25\,000$ (lengths on the map are the real lengths
multiplied by $k = \frac{1}{25000}$).
\begin{enumerate}
  \item Two villages are $6.8$ cm apart on the map. What is the real
        distance, in km?
  \item A forest has an area of $8$ cm$^2$ on the map. What is its real
        area, in km$^2$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:8}
Triangle $ABC$ has $AB = 12$, $AC = 15$, $BC = 18$. The point $M$ on
$[AB]$ satisfies $AM = 8$, and the line through $M$ parallel to $(BC)$
cuts $[AC]$ at $N$.
\begin{enumerate}
  \item Compute $AN$ and $MN$.
  \item What is the scale factor from $ABC$ to $AMN$? Compare the
        perimeters of the two triangles.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:thales:9}
Let $ABCD$ be a trapezoid with $(AB) \parallel (CD)$, $AB = 4$ and
$CD = 6$, whose diagonals $[AC]$ and $[BD]$ meet at $O$.
Using Thales in the butterfly configuration around $O$, compute the
ratio $\dfrac{OA}{OC}$, and show that $O$ cuts both diagonals in the
same ratio.
\end{exercise}
