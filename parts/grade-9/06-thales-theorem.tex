\chapter{Thales' Theorem}\label{ch:g9:thales}

How do you measure the height of a \omterm{def:g8:solids:def}{pyramid} without climbing it? Thales
of Miletus compared its shadow with the shadow of a stick. His theorem
--- \omterm{def:g6:lines:perp}{parallel lines} cut \omterm{def:g6:lines:objects}{segments} in \omterm{def:g9:linfunc:linear}{proportional} pieces --- is the
mathematical heart of \omterm{def:g7:prop:scale}{scale} models, maps, and enlargements, and one of
the oldest theorems with a name.

\section{The theorem}

\begin{theorem}[Thales]\label{thm:g9:thales:direct}
Let two lines meet at a point $A$. Take $M$ on the first line and $N$
on the second, with $B$ further along the first and $C$ further along
the second. If the lines $(MN)$ and $(BC)$ are \emph{\omterm{def:g6:lines:perp}{parallel}}, then the
triangles $AMN$ and $ABC$ have \omterm{def:g9:linfunc:linear}{proportional} sides:
\[
\frac{AM}{AB} = \frac{AN}{AC} = \frac{MN}{BC}.
\]
\end{theorem}
\admitted

\begin{method}[Computing a length with Thales]\label{met:g9:thales:compute}
\begin{enumerate}
  \item Identify the point $A$ where the two lines cross and the two
        \omterm{def:g6:lines:perp}{parallel lines}; name the two triangles;
  \item write the three equal ratios, always ``small triangle over large
        triangle'';
  \item keep the equality of the two ratios involving the three known
        lengths and the unknown one;
  \item solve the resulting proportion by cross-multiplication.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:thales:compute}
In the left configuration above, suppose $AM = 3$, $AB = 5$, $AN = 2.4$
and $MN = 3.3$, and let us compute $AC$ and $BC$. Thales gives
\[
\frac{AM}{AB} = \frac{AN}{AC} = \frac{MN}{BC},
\qquad\text{i.e.}\qquad
\frac35 = \frac{2.4}{AC} = \frac{3.3}{BC}.
\]
From $\frac35 = \frac{2.4}{AC}$, cross-multiplying:
$3 \times AC = 5 \times 2.4 = 12$, so $AC = 4$. From
$\frac35 = \frac{3.3}{BC}$: $3 \times BC = 16.5$, so $BC = 5.5$.
\end{example}

\begin{example}[Thales in real life]\label{ex:g9:thales:stick}
A vertical stick of height $1$ m casts a shadow of $1.5$ m, while a tree
casts a shadow of $12$ m at the same moment. Sun rays are \omterm{def:g6:lines:perp}{parallel}, so
the stick-and-shadow and tree-and-shadow triangles are in Thales
configuration:
\[
\frac{\text{height of tree}}{1} = \frac{12}{1.5},
\qquad\text{so the tree is } 8 \text{ m tall.}
\]
\end{example}

\section{The converse}

\begin{theorem}[Converse of Thales]\label{thm:g9:thales:converse}
With the points placed as in \cref{thm:g9:thales:direct} ($M$ on $(AB)$,
$N$ on $(AC)$, in the same order on both lines):
if
\[
\frac{AM}{AB} = \frac{AN}{AC},
\]
then the lines $(MN)$ and $(BC)$ are \omterm{def:g6:lines:perp}{parallel}.
\end{theorem}
\admitted

\begin{method}[Proving or disproving parallelism]\label{met:g9:thales:converse}
\begin{enumerate}
  \item Compute separately the two ratios $\dfrac{AM}{AB}$ and
        $\dfrac{AN}{AC}$ (as \omterm{def:g9:fractions:fraction}{fractions}, not \omterm{def:g6:decimals:rounding}{roundings});
  \item if they are equal \emph{and} the points are in the same order on
        the two lines, the lines are \omterm{def:g6:lines:perp}{parallel} (converse of Thales);
  \item if they differ, the lines are \emph{not} \omterm{def:g6:lines:perp}{parallel} --- because if
        they were, Thales' theorem would force the ratios to be equal.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:thales:converse}
On line 1, $AM = 4$ and $AB = 10$; on line 2, $AN = 6$ and $AC = 15$.
Then
\[
\frac{AM}{AB} = \frac{4}{10} = \frac25,
\qquad
\frac{AN}{AC} = \frac{6}{15} = \frac25 :
\]
equal ratios, same order: $(MN) \parallel (BC)$.

With $AC = 14$ instead: $\frac{6}{14} = \frac37 \neq \frac25$, so the
lines are not \omterm{def:g6:lines:perp}{parallel}.
\end{example}

\section{Enlargement and reduction}

\begin{definition}[Scaling a figure]\label{def:g9:thales:scaling}
Enlarging or reducing a figure by the \emph{scale
factor}\index{scale factor} $k > 0$ means multiplying \emph{all} its
lengths by $k$: an \emph{enlargement} when $k > 1$, a \emph{reduction}
when $k < 1$. In Thales' configuration, the triangle $AMN$ is the
reduction of $ABC$ by the factor $k = \frac{AM}{AB}$.
\end{definition}

\begin{proposition}[Effect on angles and areas]\label{prop:g9:thales:scaling}
Scaling by the factor $k$ preserves angles and the shape of figures,
multiplies every length by $k$, and multiplies every \emph{\omterm{def:g6:measure:area}{area}} by
$k^2$.
\end{proposition}
\admitted

\begin{omfigure}
\begin{tikzpicture}[scale=0.72]
  \draw[thick, omDef, fill=omDef!8] (0,0) rectangle (2,1.2);
  \draw[thick, omThm, fill=omThm!8] (3.4,0) rectangle (7.4,2.4);
  \node[font=\small, omDef] at (1,-0.45) {$2 \times 1.2$};
  \node[font=\small, omThm] at (5.4,-0.45) {$4 \times 2.4$};
  \node[font=\small] at (1,0.6) {area $2.4$};
  \node[font=\small] at (5.4,1.2) {area $9.6$};
\end{tikzpicture}

{\small Doubling the lengths ($k = 2$) multiplies the \omterm{def:g6:measure:area}{area} by
$k^2 = 4$: four copies of the small rectangle tile the large one.}
\end{omfigure}

\begin{example}
A photo of $10 \times 15$ cm is enlarged with \omterm{def:g9:thales:scaling}{scale factor} $k = 3$: the
print measures $30 \times 45$ cm. Its \omterm{def:g6:measure:area}{area} goes from $150$ cm$^2$ to
$150 \times 9 = 1350$ cm$^2$ --- $9$ times larger, not $3$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g9:thales:1}
The lines $(MN)$ and $(BC)$ are \omterm{def:g6:lines:perp}{parallel}, with $M$ on $[AB]$ and $N$ on
$[AC]$; $AM = 2$, $AB = 6$, $AN = 3$, $BC = 9$. Compute $AC$ and $MN$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:thales:2}
Same configuration: $AM = 5$, $MB = 3$ (careful: $MB$, not $AB$!),
$AN = 4$. Compute $AC$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:thales:3}
In a butterfly configuration, $A$ is between $M$ and $B$ and between $N$
and $C$, with $(MN) \parallel (BC)$; $AM = 3$, $AB = 7.5$, $AN = 2$,
$MN = 2.6$. Compute $AC$ and $BC$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:thales:4}
$M$ is on $[AB]$ with $AM = 6$, $AB = 8$; $N$ is on $[AC]$ with $AN = 9$,
$AC = 12$. Are the lines $(MN)$ and $(BC)$ \omterm{def:g6:lines:perp}{parallel}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:5}
Same as above with $AM = 4$, $AB = 6$, $AN = 5$, $AC = 8$. Are $(MN)$
and $(BC)$ \omterm{def:g6:lines:perp}{parallel}? Justify carefully.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:6}
A $1.8$ m tall person stands $2$ m away from a street lamp's base;
their shadow measures $3$ m. Draw the Thales configuration formed by the
lamp, the person, and the tip of the shadow, and compute the height of
the lamp.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:7}
A map has \omterm{def:g7:prop:scale}{scale} $1 : 25\,000$ (lengths on the map are the real lengths
multiplied by $k = \frac{1}{25000}$).
\begin{enumerate}
  \item Two villages are $6.8$ cm apart on the map. What is the real
        distance, in km?
  \item A forest has an \omterm{def:g6:measure:area}{area} of $8$ cm$^2$ on the map. What is its real
        \omterm{def:g6:measure:area}{area}, in km$^2$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:8}
Triangle $ABC$ has $AB = 12$, $AC = 15$, $BC = 18$. The point $M$ on
$[AB]$ satisfies $AM = 8$, and the line through $M$ \omterm{def:g6:lines:perp}{parallel} to $(BC)$
cuts $[AC]$ at $N$.
\begin{enumerate}
  \item Compute $AN$ and $MN$.
  \item What is the \omterm{def:g9:thales:scaling}{scale factor} from $ABC$ to $AMN$? Compare the
        \omterm{def:g6:measure:perimeter}{perimeters} of the two triangles.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:thales:9}
Let $ABCD$ be a trapezoid with $(AB) \parallel (CD)$, $AB = 4$ and
$CD = 6$, whose diagonals $[AC]$ and $[BD]$ meet at $O$.
Using Thales in the butterfly configuration around $O$, compute the
ratio $\dfrac{OA}{OC}$, and show that $O$ cuts both diagonals in the
same ratio.
\end{exercise}

\section{Problem: The unmarked ruler, the pinhole camera, and the
three means of a trapezoid}

\begin{problem}[{Weekend problem --- Thales at work: dividing any
segment into equal parts, measuring the Sun with a shoebox, and a
trapezoid where all three famous means meet}]
\label{pb:g9:thales:1}
Thales' theorem is the mathematics of \emph{rays}: sun rays,
light rays through a pinhole, pencil rays from a \omterm{def:g5:solids:def}{vertex}. This
problem uses it three ways --- as a construction tool (dividing a
\omterm{def:g6:lines:objects}{segment} of \emph{any} length, \omterm{def:g3:numbers:evenodd}{even} $\sqrt2$, into perfectly equal
parts), as a measuring instrument (the \omterm{def:g4:geometry:circle}{diameter} of the Sun, with
a shoebox and a coin), and as a magnifying glass on the trapezoid
of \cref{exo:g9:thales:9}, where the arithmetic, geometric and
harmonic means of this series all turn up in one figure.

\medskip
\noindent\textbf{Part I --- Dividing with an unmarked ruler.}
\begin{enumerate}
  \item Draw a \omterm{def:g6:lines:objects}{segment} $[AB]$ of $7$ cm. To cut it into three
        equal parts with compass and unmarked ruler: draw any ray
        from $A$ (not through $B$); step off three equal compass
        lengths on it, giving points $P_1$, $P_2$, $P_3$; join
        $P_3$ to $B$; draw the \omterm{def:g6:lines:perp}{parallels} to $(P_3 B)$ through
        $P_1$ and $P_2$. Perform the construction and mark where
        the \omterm{def:g6:lines:perp}{parallels} cut $[AB]$.
  \item Justify with \cref{thm:g9:thales:direct} that the two
        marked points cut $[AB]$ exactly at its third points.
  \item Adapt the method to divide a \omterm{def:g6:lines:objects}{segment} into $5$ equal
        parts, then to construct $\frac35$ of a given \omterm{def:g6:lines:objects}{segment}.
  \item Construct the point $M$ of $[AB]$ with
        $\frac{AM}{MB} = \frac23$ (that is, $M$ at $\frac25$ of
        the way from $A$). How many equal steps on the ray, and
        which point joins $B$?
  \item Why is this construction better than measuring with a
        graduated ruler? Consider a \omterm{def:g6:lines:objects}{segment} of length $\sqrt2$
        (a unit square's diagonal, irrational by
        \cref{pb:g9:sqrt:1}): what would measuring give, and
        what does Thales give?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The pinhole camera.}
Poke a pin through one \omterm{def:g5:solids:def}{face} of a closed \omterm{def:g5:solids:def}{box}: on the opposite
\omterm{def:g5:solids:def}{face}, an upside-down image of the world appears. A point of an
object, the pinhole, and the image point are aligned --- light
travels straight --- so the object (height $H$, at distance $D$
in front of the hole) and its image (height $h$, on the back
wall at depth $d$ behind the hole) sit in the butterfly
configuration of Thales.
\begin{enumerate}[resume]
  \item Derive the pinhole formula
        $h = H \times \frac dD$ from Thales in the butterfly
        around the hole.
  \item A $12$ m tree stands $30$ m from a shoebox of depth
        $20$ cm. How tall is its image, and which way up?
  \item Point the \omterm{def:g5:solids:def}{box} at the Sun: at $d = 1$ m behind the
        pinhole, the Sun's image is a disk of \omterm{def:g4:geometry:circle}{diameter} about
        $9$ mm. Given the Sun's distance
        $D = 1.5 \times 10^{11}$ m, compute its \omterm{def:g4:geometry:circle}{diameter} in
        \omterm{def:g9:fractions:scientific}{scientific notation} (\cref{pb:g9:fractions:1}'s
        notation at work). Compare with the true value,
        $1.39 \times 10^9$ m.
  \item The Moon: \omterm{def:g4:geometry:circle}{diameter} $3.5 \times 10^6$ m, distance
        $3.8 \times 10^8$ m. Compute the ratio
        $\frac{\text{diameter}}{\text{distance}}$ for the Moon
        and for the Sun. What lucky coincidence do the two
        numbers reveal --- and what spectacular event does it
        make possible?
  \item In one or two sentences: why is the pinhole image
        upside down, and why does enlarging the pinhole make the
        image brighter but blurrier?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The trapezoid of the three
means.}
$ABCD$ is the trapezoid of \cref{exo:g9:thales:9}:
$(AB) \parallel (CD)$, $AB = 4$, $CD = 6$, diagonals crossing at
$O$ with $\frac{OA}{OC} = \frac{OB}{OD} = \frac{4}{6} =
\frac23$.
\begin{enumerate}[resume]
  \item Butterfly practice: two lines cross at $O$ with
        $(MN) \parallel (PQ)$ in butterfly position; $OM = 3$,
        $OP = 7.5$, $MN = 4$ and $OQ = 6$. Compute $PQ$ and
        $ON$.
  \item Explain how Thales, applied with ratio $\frac12$,
        contains the midpoint theorem of
        \cref{thm:g8:midpoints:direct} as its special case.
  \item Draw the \omterm{def:g6:lines:perp}{parallel} to the bases through $O$; it meets
        $[AD]$ at $E$. Working in the triangle $ACD$ (note
        $\frac{AO}{AC} = \frac25$), compute $EO$.
  \item By the mirror computation in the triangle $BDC$, the
        piece $OF$ (with $F$ on $[BC]$) has the same length.
        Conclude that
        \[
        EF = 2 \times \frac{4 \times 6}{4 + 6} = 4.8 :
        \]
        the \omterm{def:g6:lines:perp}{parallel} through the diagonal crossing measures the
        \emph{\omterm{pb:g8:speed:1}{harmonic mean}} of the bases --- the round-trip
        mean of \cref{pb:g8:speed:1}, reappearing in pure
        geometry.
  \item The grand finale: compute the three classical means of
        the bases $4$ and $6$ --- arithmetic
        $\frac{4+6}{2}$, geometric $\sqrt{4 \times 6}$, harmonic
        $\frac{2 \times 4 \times 6}{4 + 6}$ --- and order them.
        Each lives in the trapezoid: the arithmetic mean is the
        midline joining the legs' midpoints
        (\cref{pb:g7:areas:1}), the \omterm{pb:g8:speed:1}{harmonic mean} is your
        \omterm{def:g6:lines:objects}{segment} $EF$, and the \omterm{pb:g8:cosine:1}{geometric mean} is the \omterm{def:g6:lines:perp}{parallel}
        that cuts the trapezoid into two \emph{similar}
        trapezoids (check its value with a calculator; the
        proof is a lovely extra challenge). Where has the
        inequality between the three been proved in this book
        (\cref{pb:g8:cosine:1}, \cref{pb:g8:speed:1})?
\end{enumerate}
\end{problem}
