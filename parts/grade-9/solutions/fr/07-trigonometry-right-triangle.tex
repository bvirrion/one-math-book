\section*{Chapitre \ref{ch:g9:trig} --- Trigonométrie dans le triangle rectangle}

\begin{solution}{exo:g9:trig:1}
Hypoténuse~: $\sqrt{9^2 + 12^2} = \sqrt{81 + 144} = \sqrt{225} = 15$.

Autre côté~: $\sqrt{17^2 - 8^2} = \sqrt{289 - 64} = \sqrt{225} = 15$.
\end{solution}

\begin{solution}{exo:g9:trig:2}
$7^2 + 24^2 = 49 + 576 = 625 = 25^2$~: oui, rectangle (réciproque de
Pythagore), avec l'\omterm{def:g3:shapes:rightangle}{angle droit} opposé au côté $25$.

$5^2 + 6^2 = 61 \neq 64 = 8^2$~: non rectangle.
\end{solution}

\begin{solution}{exo:g9:trig:3}
L'hypoténuse est $[EF]$ (opposée à l'\omterm{def:g3:shapes:rightangle}{angle droit} en $D$). Le côté
opposé à $\theta = \widehat E$ est $[DF]$~; le côté adjacent est
$[DE]$. D'où
\[
\cos\theta = \frac{DE}{EF}, \qquad
\sin\theta = \frac{DF}{EF}, \qquad
\tan\theta = \frac{DF}{DE}.
\]
\end{solution}

\begin{solution}{exo:g9:trig:4}
Côté opposé~: $10 \sin 28^\circ \approx 10 \times 0.469 = 4.7$.
Côté adjacent~: $10 \cos 28^\circ \approx 10 \times 0.883 = 8.8$.
(Vérification~: $4.7^2 + 8.8^2 \approx 22 + 77 \approx 100$.)
\end{solution}

\begin{solution}{exo:g9:trig:5}
$\tan\theta = \dfrac{4.5}{6} = 0.75$, donc
$\theta = \tan^{-1}(0.75) \approx 37^\circ$.
\end{solution}

\begin{solution}{exo:g9:trig:6}
Dans le \omterm{def:g6:shapes:triangles}{triangle rectangle} formé par l'observateur, le point au sol sous
le drone, et le drone~: l'altitude $120$ m est le côté opposé à l'angle
d'élévation, la distance horizontale $d$ est adjacente. Donc
\[
\tan 32^\circ = \frac{120}{d},
\qquad
d = \frac{120}{\tan 32^\circ} \approx \frac{120}{0.625} = 192 \text{ m}.
\]
\end{solution}

\begin{solution}{exo:g9:trig:7}
La montée ($1.2$ m) est opposée à l'angle~; la longueur $L$ de la rampe
est l'hypoténuse~:
\[
\sin 5^\circ = \frac{1.2}{L},
\qquad
L = \frac{1.2}{\sin 5^\circ} \approx \frac{1.2}{0.0872} \approx 13.8
\text{ m}.
\]
La rampe doit mesurer au moins $13.8$ m.
\end{solution}

\begin{solution}{exo:g9:trig:8}
De $\cos^2\theta + \sin^2\theta = 1$~:
$\sin^2\theta = 1 - 0.36 = 0.64$, et $\sin\theta > 0$ pour un angle
aigu, donc $\sin\theta = 0.8$. Puis
$\tan\theta = \dfrac{\sin\theta}{\cos\theta} = \dfrac{0.8}{0.6} =
\dfrac43$.
\end{solution}

\begin{solution}{exo:g9:trig:9}
\emph{\omterm{def:g6:shapes:triangles}{Triangle rectangle} \omterm{def:g6:shapes:triangles}{isocèle}} de côtés de l'\omterm{def:g3:shapes:rightangle}{angle droit} $1$~:
hypoténuse $\sqrt{1 + 1} = \sqrt2$~; chaque angle aigu vaut $45^\circ$,
et
\[
\cos 45^\circ = \frac{1}{\sqrt2} = \frac{\sqrt2}{2} = \sin 45^\circ,
\qquad
\tan 45^\circ = \frac11 = 1 .
\]

\emph{\omterm{def:g6:shapes:triangles}{Triangle équilatéral}} de côté $1$~: sa hauteur le partage en deux
\omterm{def:g6:shapes:triangles}{triangles rectangles} d'hypoténuse $1$, de base $\frac12$, et de hauteur
$h = \sqrt{1 - \frac14} = \frac{\sqrt3}{2}$ (Pythagore). Les angles
sont $60^\circ$ (à la base) et $30^\circ$ (au \omterm{def:g5:solids:def}{sommet}). En lisant les
rapports~:
\[
\cos 60^\circ = \frac{1/2}{1} = \frac12, \quad
\sin 60^\circ = \frac{\sqrt3/2}{1} = \frac{\sqrt3}{2}, \quad
\cos 30^\circ = \frac{\sqrt3}{2}, \quad
\sin 30^\circ = \frac12 .
\]
\end{solution}
