\begin{solution}{exo:g9:statproba:1}
\omterm{def:g9:statproba:indicators}{Moyenne}~: $\dfrac{14 + 9 + 17 + 9 + 12 + 11 + 12}{7} = \dfrac{84}{7} = 12$.

\omterm{def:g9:statproba:indicators}{Médiane}~: triée, $9, 9, 11, 12, 12, 14, 17$~; la valeur du milieu
($4$\textsuperscript{e}) est $12$.

\omterm{def:g9:statproba:indicators}{Étendue}~: $17 - 9 = 8$.
\end{solution}

\begin{solution}{exo:g9:statproba:2}

\approx 21.1$ degrés.

\omterm{def:g9:statproba:indicators}{Médiane}~: triée, $18, 19, 19, 21, 22, 24, 25$~; la $4$\textsuperscript{e} valeur est

\begin{solution}{exo:g9:statproba:3}
\omterm{def:g9:statproba:indicators}{Moyenne}~:
\[
\frac{2 \times 0 + 4 \times 1 + 3 \times 2 + 1 \times 3}{10}
= \frac{0 + 4 + 6 + 3}{10} = \frac{13}{10} = 1.3
\text{ buts par match.}
\]
\omterm{def:g9:statproba:indicators}{Médiane}~: la série triée est $0,0,1,1,1,1,2,2,2,3$~; les $5$\textsuperscript{e} et
$6$\textsuperscript{e} valeurs sont toutes deux $1$~: \omterm{def:g9:statproba:indicators}{médiane} $1$.
\end{solution}

\begin{solution}{exo:g9:statproba:4}
$\P(3) = \frac16$. \quad
$\P(\text{pair}) = \frac36 = \frac12$. \quad
$\P(\text{au moins } 3) = \P(\{3,4,5,6\}) = \frac46 = \frac23$. \quad
$\P(\text{pas } 6) = 1 - \frac16 = \frac56$.
\end{solution}

\begin{solution}{exo:g9:statproba:5}
$\P(\text{blanche}) = \frac{5}{12}$.

$\P(\text{rouge ou verte}) = \frac{4 + 3}{12} = \frac{7}{12}$.

$\P(\text{pas blanche}) = 1 - \frac{5}{12} = \frac{7}{12}$.

Les deux dernières sont égales~: «~pas blanche~» \emph{est} l'événement
«~rouge ou verte~».
\end{solution}

\begin{solution}{exo:g9:statproba:6}
Les $12$ issues (\omterm{def:g5:solids:def}{face} de la pièce, valeur du dé) sont équiprobables.

Pile et un $6$~: une issue sur $12$~: $\frac{1}{12}$.

Pile ou un $6$~: les issues avec pile ($6$ d'entre elles) plus l'issue
(\omterm{def:g5:solids:def}{face}, $6$)~: $7$ issues, donc $\frac{7}{12}$. Compter chaque événement
et soustraire le chevauchement donne la même chose~:
$\frac{6}{12} + \frac{2}{12} - \frac{1}{12} = \frac{7}{12}$.
\end{solution}

\begin{solution}{exo:g9:statproba:7}
Premier tirage~: rouge $\frac25$, noir $\frac35$. Second tirage sans
remise~: après un rouge, $1$ rouge et $3$ noirs restent parmi $4$~;
après un noir, $2$ rouges et $2$ noirs parmi $4$.

Deux noirs~: $\frac35 \times \frac24 = \frac{6}{20} = \frac{3}{10}$.

Couleurs différentes (RN ou NR)~:
$\frac25 \times \frac34 + \frac35 \times \frac24
= \frac{6}{20} + \frac{6}{20} = \frac{12}{20} = \frac35$.
\end{solution}

\begin{solution}{exo:g9:statproba:8}
\emph{1.} $9 \times 14 = 126$ points.

\emph{2.} Une \omterm{def:g9:statproba:indicators}{moyenne} de $15$ sur $10$ matchs signifie $150$ points au
total~: elle doit marquer $150 - 126 = 24$ points.
\end{solution}

\begin{solution}{exo:g9:statproba:9}
\emph{1.} Parmi les $36$ paires équiprobables, les doubles sont
$(1,1), (2,2), \dots, (6,6)$~: six d'entre eux, donc
$\P(\text{double}) = \frac{6}{36} = \frac16$.

\emph{2.} Deux parties indépendantes~: arbre avec gain ($\frac16$) /
perte ($\frac56$) à chaque niveau. \omterm{def:g9:statproba:probability}{Probabilité} de ne jamais gagner~:
$\frac56 \times \frac56 = \frac{25}{36}$. Donc
\[
\P(\text{au moins un gain}) = 1 - \frac{25}{36} = \frac{11}{36}
\approx 0.31 .
\]
\end{solution}
