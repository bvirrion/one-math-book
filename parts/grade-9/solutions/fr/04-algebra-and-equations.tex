\section*{Chapitre \ref{ch:g9:algebra} --- Algèbre et équations}

\begin{solution}{exo:g9:algebra:1}
$4(3x - 2) = 12x - 8$.

$(x+6)(x+2) = x^2 + 2x + 6x + 12 = x^2 + 8x + 12$.

$(2x-3)(x+5) = 2x^2 + 10x - 3x - 15 = 2x^2 + 7x - 15$.

$(x-1)(x-9) = x^2 - 9x - x + 9 = x^2 - 10x + 9$.
\end{solution}

\begin{solution}{exo:g9:algebra:2}
$(x+8)^2 = x^2 + 16x + 64$.

$(3x-4)^2 = 9x^2 - 24x + 16$.

$(x+10)(x-10) = x^2 - 100$.

$(5x+2)(5x-2) = 25x^2 - 4$.
\end{solution}

\begin{solution}{exo:g9:algebra:3}
$101^2 = (100+1)^2 = 10000 + 200 + 1 = 10201$.

$59^2 = (60-1)^2 = 3600 - 120 + 1 = 3481$.

$53 \times 47 = (50+3)(50-3) = 2500 - 9 = 2491$.
\end{solution}

\begin{solution}{exo:g9:algebra:4}
$9x + 12 = 3(3x + 4)$.

$x^2 - 64 = (x+8)(x-8)$.

$x^2 + 14x + 49 = (x+7)^2$.

$18x^2 - 6x = 6x(3x - 1)$.
\end{solution}

\begin{solution}{exo:g9:algebra:5}
$4x + 7 = 19$~: $4x = 12$, donc $x = 3$.

$6x - 5 = 2x + 11$~: $4x = 16$, donc $x = 4$.

$5(x-3) = 3x + 1$~: $5x - 15 = 3x + 1$, donc $2x = 16$ et $x = 8$.
\end{solution}

\begin{solution}{exo:g9:algebra:6}
$(x+4)(3x-12) = 0$~: $x = -4$ ou $x = 4$.

$x^2 - 121 = (x-11)(x+11) = 0$~: $x = 11$ ou $x = -11$.

$(2x-1)^2 = 0$~: la solution unique $x = \frac12$.

$x^2 = 8x$~: $x(x - 8) = 0$, donc $x = 0$ ou $x = 8$.
\end{solution}

\begin{solution}{exo:g9:algebra:7}
Différence de carrés avec $a = 3x + 2$ et $b = 5$~:
\[
E = (3x + 2)^2 - 5^2 = (3x + 2 + 5)(3x + 2 - 5) = (3x + 7)(3x - 3).
\]
Règle du produit nul~: $3x + 7 = 0$ donne $x = -\frac73$, et
$3x - 3 = 0$ donne $x = 1$.
\end{solution}

\begin{solution}{exo:g9:algebra:8}
$3x - 4 \geq 8$~: $3x \geq 12$, donc $x \geq 4$ (point plein en $4$,
hachurer à droite).

$5 - 2x > 1$~: $-2x > -4$, diviser par $-2$ renverse le signe~: $x < 2$
(point creux en $2$, hachurer à gauche).

$4(x+1) \leq x - 5$~: $4x + 4 \leq x - 5$, donc $3x \leq -9$, donc
$x \leq -3$ (point plein en $-3$, hachurer à gauche).
\end{solution}

\begin{solution}{exo:g9:algebra:9}
Soit $x$ le nombre~: $3x + 7 = 5x - 9$. Alors $16 = 2x$, donc $x = 8$.
Vérification~: $3 \times 8 + 7 = 31$ et $5 \times 8 - 9 = 31$.
\end{solution}

\begin{solution}{exo:g9:algebra:10}
\emph{1.} $(n+1)(n-1) = n^2 - 1$ (troisième identité). Les voisins de
$n$ sont $n - 1$ et $n + 1$, donc leur produit est $n^2 - 1$~: par ex.\
$24 \times 26 = 625 - 1 = 624$.

\emph{2.} Trois entiers consécutifs s'écrivent $n - 1$, $n$, $n + 1$.
Leur somme est
\[
(n-1) + n + (n+1) = 3n ,
\]
un multiple de $3$ (trois fois le milieu).
\end{solution}
