\section*{Chapitre \ref{ch:g9:solids} --- Solides et volumes}

\begin{solution}{exo:g9:solids:1}
Pavé~: $V = 4 \times 5 \times 12 = 240$.

Cylindre~: $V = \pi r^2 h = \pi \times 9 \times 7 = 63\pi \approx 198$.
\end{solution}

\begin{solution}{exo:g9:solids:2}
Cône~: $V = \frac13 \pi r^2 h = \frac13 \pi \times 25 \times 12 =
100\pi \approx 314$.

Pyramide~: aire de base $B = 8 \times 3 = 24$, donc
$V = \frac13 \times 24 \times 10 = 80$.
\end{solution}

\begin{solution}{exo:g9:solids:3}
Volume~: $V = \frac43 \pi \times 6^3 = \frac43 \pi \times 216 = 288\pi$.

Aire de surface~: $4\pi \times 6^2 = 144\pi$.
\end{solution}

\begin{solution}{exo:g9:solids:4}
Rayon de la section~: $\sqrt{r^2 - d^2} = \sqrt{100 - 64} = \sqrt{36}
= 6$.
\end{solution}

\begin{solution}{exo:g9:solids:5}
Facteur d'échelle du cône entier vers le petit~:
$k = \frac{9}{12} = \frac34$. Rayon de section~: $8 \times \frac34 = 6$.

Volume du cône entier~: $\frac13\pi \times 64 \times 12 = 256\pi$.
Volume du petit cône~: $256\pi \times \left(\frac34\right)^3 =
256\pi \times \frac{27}{64} = 108\pi \approx 339$.
\end{solution}

\begin{solution}{exo:g9:solids:6}
Volume de la bille~: $\frac43\pi \times 2^3 = \frac{32\pi}{3}$ cm$^3$.
Ce volume se répartit sur l'aire de base $\pi \times 3^2 = 9\pi$ cm$^2$,
donc le niveau monte de
\[
\frac{32\pi/3}{9\pi} = \frac{32}{27} \approx 1.2 \text{ cm}
\]
(environ $12$ mm).
\end{solution}

\begin{solution}{exo:g9:solids:7}
Le petit moule est la réduction du grand par $k = \frac36 =
\frac12$, donc son volume est $\left(\frac12\right)^3 = \frac18$ du
grand~: la recette remplit $8$ petits moules.
\end{solution}

\begin{solution}{exo:g9:solids:8}
$V = \frac13 \times 230^2 \times 147 = \frac13 \times 52\,900 \times 147
= 52\,900 \times 49 = 2\,592\,100$ m$^3$ --- environ $2.6$ millions de
mètres cubes.
\end{solution}

\begin{solution}{exo:g9:solids:9}
\emph{1.} L'eau forme un cône réduit par $k = \frac12$ (sommet en bas,
moitié de la hauteur), donc son volume est $\left(\frac12\right)^3 =
\frac18$ de celui de l'entonnoir~: un huitième, bien moins que la
moitié~!

\emph{2.} L'eau jusqu'à la hauteur $d$ forme un cône réduit par
$k = \frac{d}{12}$, de volume $k^3$ fois celui de l'entonnoir. La moitié
du volume signifie $k^3 = \frac12$, c'est-à-dire
$\left(\frac{d}{12}\right)^3 = \frac12$. Puis
$\frac{d}{12} = \sqrt[3]{0.5} \approx 0.7937$, donc
$d \approx 9.5$ cm~: la seconde moitié du volume n'occupe que les $2.5$
cm du haut --- la partie large du cône contient la plupart de l'eau.
\end{solution}
