\section*{Chapter \ref{ch:g9:trig} --- Trigonometry in the Right Triangle}

\begin{solution}{exo:g9:trig:1}
Hypotenuse: $\sqrt{9^2 + 12^2} = \sqrt{81 + 144} = \sqrt{225} = 15$.

Other leg: $\sqrt{17^2 - 8^2} = \sqrt{289 - 64} = \sqrt{225} = 15$.
\end{solution}

\begin{solution}{exo:g9:trig:2}
$7^2 + 24^2 = 49 + 576 = 625 = 25^2$: yes, \omterm{def:g6:shapes:triangles}{right-angled} (converse of
Pythagoras), with the \omterm{def:g3:shapes:rightangle}{right angle} opposite the side $25$.

$5^2 + 6^2 = 61 \neq 64 = 8^2$: not \omterm{def:g6:shapes:triangles}{right-angled}.
\end{solution}

\begin{solution}{exo:g9:trig:3}
The hypotenuse is $[EF]$ (opposite the \omterm{def:g3:shapes:rightangle}{right angle} at $D$). The side
opposite to $\theta = \widehat E$ is $[DF]$; the adjacent side is
$[DE]$. Hence
\[
\cos\theta = \frac{DE}{EF}, \qquad
\sin\theta = \frac{DF}{EF}, \qquad
\tan\theta = \frac{DF}{DE}.
\]
\end{solution}

\begin{solution}{exo:g9:trig:4}
Opposite leg: $10 \sin 28^\circ \approx 10 \times 0.469 = 4.7$.
Adjacent leg: $10 \cos 28^\circ \approx 10 \times 0.883 = 8.8$.
(Check: $4.7^2 + 8.8^2 \approx 22 + 77 \approx 100$.)
\end{solution}

\end{solution}

\begin{solution}{exo:g9:trig:6}
In the \omterm{def:g6:shapes:triangles}{right triangle} formed by the observer, the point on the ground
below the drone, and the drone: the altitude $120$ m is the side
opposite the elevation angle, the horizontal distance $d$ is adjacent.
So
\[
\tan 32^\circ = \frac{120}{d},
\qquad
d = \frac{120}{\tan 32^\circ} \approx \frac{120}{0.625} = 192 \text{ m}.
\]
\end{solution}

\begin{solution}{exo:g9:trig:7}
The rise ($1.2$ m) is opposite the angle; the ramp length $L$ is the
hypotenuse:
\[
\sin 5^\circ = \frac{1.2}{L},
\qquad
L = \frac{1.2}{\sin 5^\circ} \approx \frac{1.2}{0.0872} \approx 13.8
\text{ m}.
\]
The ramp must be at least $13.8$ m long.
\end{solution}

\begin{solution}{exo:g9:trig:8}
From $\cos^2\theta + \sin^2\theta = 1$:
$\sin^2\theta = 1 - 0.36 = 0.64$, and $\sin\theta > 0$ for an acute
angle, so $\sin\theta = 0.8$. Then
$\tan\theta = \dfrac{\sin\theta}{\cos\theta} = \dfrac{0.8}{0.6} =
\dfrac43$.
\end{solution}

\begin{solution}{exo:g9:trig:9}
\emph{Right \omterm{def:g6:shapes:triangles}{isosceles triangle}} with legs $1$: hypotenuse
$\sqrt{1 + 1} = \sqrt2$; each acute angle is $45^\circ$, and
\[
\cos 45^\circ = \frac{1}{\sqrt2} = \frac{\sqrt2}{2} = \sin 45^\circ,
\qquad
\tan 45^\circ = \frac11 = 1 .
\]

\emph{\omterm{def:g6:shapes:triangles}{Equilateral triangle}} of side $1$: its height splits it into two
\omterm{def:g6:shapes:triangles}{right triangles} with hypotenuse $1$, base $\frac12$, and height
$h = \sqrt{1 - \frac14} = \frac{\sqrt3}{2}$ (Pythagoras). The angles are
$60^\circ$ (at the base) and $30^\circ$ (at the top). Reading the
ratios:
\[
\cos 60^\circ = \frac{1/2}{1} = \frac12, \quad
\sin 60^\circ = \frac{\sqrt3/2}{1} = \frac{\sqrt3}{2}, \quad
\cos 30^\circ = \frac{\sqrt3}{2}, \quad
\sin 30^\circ = \frac12 .
\]
\end{solution}

\begin{solution}{pb:g9:trig:1}
\textbf{1.} The height above eye level is the side opposite
$35^\circ$, with adjacent side $60$ m:
$h = 60 \tan 35^\circ \approx 60 \times 0.700 = 42$ m.

\textbf{2.} $\tan\theta = 0.12$ gives
$\theta = \tan^{-1}(0.12) \approx 6.8^\circ$ --- steep for a
road, yet a modest angle. A $45^\circ$ road has
$\tan 45^\circ = 1$: a $100\,\%$ grade. (``One hundred
percent'' is far from vertical --- a favorite ski-lift
misunderstanding.)

\textbf{3.} In a \omterm{def:g6:shapes:triangles}{right triangle} with acute angles $\theta$ and
$90^\circ - \theta$, the side opposite one is adjacent to the
other and the hypotenuse is shared. So
$\sin(90^\circ - \theta) = \frac{\text{its opposite}}{h} =
\frac{\text{adjacent of }\theta}{h} = \cos\theta$, and
symmetrically $\cos(90^\circ - \theta) = \sin\theta$: the
\emph{co}\omterm{def:g9:trig:ratios}{sine} is the \omterm{def:g9:trig:ratios}{sine} of the \emph{co}mplement.

\textbf{4.} Height: $8 \sin 60^\circ = 8 \times
\frac{\sqrt3}{2} = 4\sqrt3 \approx 6.93$ m. Foot:
$8 \cos 60^\circ = 4$ m exactly.

\textbf{5.} $0.574^2 + 0.819^2 = 0.329 + 0.671 = 1.000$ (to
\omterm{def:g6:decimals:rounding}{rounding}); and $\left(\frac12\right)^2 +
\left(\frac{\sqrt3}{2}\right)^2 = \frac14 + \frac34 = 1$. Any
pair of values failing the identity betrays a wrong reading or
a calculator in the wrong mode --- a free error detector.

\textbf{6.} $\tan 40^\circ = \dfrac hx$ and
$\tan 30^\circ = \dfrac{h}{x + 200}$.

\textbf{7.} $x = \dfrac{h}{\tan 40^\circ}$ and
$x + 200 = \dfrac{h}{\tan 30^\circ}$; subtracting,
$200 = h\left(\frac{1}{\tan 30^\circ} -
\frac{1}{\tan 40^\circ}\right)$, whence the formula. Numbers:
$\frac{1}{0.577} \approx 1.732$, $\frac{1}{0.839} \approx
1.192$, \omterm{ex:g1:subtraction:difference}{difference} $0.540$:
$h \approx \frac{200}{0.540} \approx 370$ m.

\textbf{8.} $x = \frac{h}{\tan 40^\circ} \approx
\frac{370}{0.839} \approx 441$ m. Check from $A$:
$\frac{h}{x + 200} \approx \frac{370}{641} \approx 0.577 =
\tan 30^\circ$: consistent.

\textbf{9.} The one-station method needs the horizontal
distance to the point \emph{directly below the summit} ---
unmeasurable across a gorge or inside a mountain. The
two-station method replaces it by a distance you can pace out
yourself: the $200$ m between your own two viewpoints.

\textbf{10.} $h = \dfrac{200}{\sqrt3 - \frac{1}{\sqrt3}}
= \dfrac{200}{\frac{2}{\sqrt3}} = 100\sqrt3 \approx 173$ m ---
the Eiffel Tower's second floor, measured with two sightings
and a stroll.

\textbf{11.} Sight line $d$, radii $R$ (to the grazing point,
\omterm{def:g6:lines:perp}{perpendicular} to the sight line) and $R + h$ (to the eye):
Pythagoras gives $d^2 + R^2 = (R + h)^2 = R^2 + 2Rh + h^2$, so
$d^2 = 2Rh + h^2 \approx 2Rh$ ($h$ is minuscule against $R$).
Eyes at $1.80$ m:
$d \approx \sqrt{2 \times 6.37 \times 10^6 \times 1.8}
\approx 4\,800$ m --- the sea horizon is barely five
kilometers away.

\textbf{12.} $d \approx \sqrt{2 \times 6.37 \times 10^6 \times
324} \approx 64$ km. Sailors' rule: $3.6\sqrt{1.8} \approx
4.8$ km and $3.6\sqrt{324} = 64.8$ km --- both match.

\textbf{13.} $\tan\theta = \frac{2}{60}$ gives $\theta \approx
1.9^\circ$. The Moon's $0.5^\circ$ is about a \emph{\omterm{def:g3:division:half}{quarter}} of
a thumb: the Moon looks enormous and is hidden by a fingernail
--- the eye is a poor protractor.

\textbf{14.} Stairs: $\tan^{-1}\frac{17}{29} \approx 30^\circ$
--- comfortable stairs climb at almost exactly the
$30^\circ$ of the special values. At $60^\circ$ on a $29$ cm
tread, the riser would be $29\tan 60^\circ = 29\sqrt3 \approx
50$ cm: a ladder, not a staircase.

\textbf{15.} $\tan 10^\circ \approx 0.18$;
$\tan 45^\circ = 1$; $\tan 80^\circ \approx 5.7$;
$\tan 89^\circ \approx 57.3$. As $\theta \to 90^\circ$ the
adjacent side shrinks to nothing while the opposite side holds:
the \omterm{def:g3:division:remainder}{quotient} grows beyond every bound. At $90^\circ$ exactly
there is no triangle and no value --- the graph of the \omterm{def:g9:trig:ratios}{tangent}
climbs a vertical asymptote, first of many in the High School
volume.
\end{solution}
