\section*{Hoofdstuk \ref{ch:g9:linfunc} --- Lineaire en affiene functies}

\begin{solution}{exo:g9:linfunc:1}
Lineair: $f$ en $k$ (vorm $ax$). Affien maar niet lineair: $g$
(\omterm{prop:g9:linfunc:affinegraph}{richtingscoëfficiënt} $3$, snijpunt $-5$) en $m$ (constant,
\omterm{prop:g9:linfunc:affinegraph}{richtingscoëfficiënt} $0$). Geen van beide: $h$, vanwege het kwadraat.
\end{solution}

\begin{solution}{exo:g9:linfunc:2}
$f(0) = -3$; $f(4) = 5$; $f(-1) = -5$. En $f(x) = 9$ betekent
$2x - 3 = 9$, dus $2x = 12$ en $x = 6$.
\end{solution}

\begin{solution}{exo:g9:linfunc:3}
$a = \frac{15}{6} = 2.5$, dus $f(x) = 2.5x$. Dan $f(10) = 25$ en
$f(-2) = -5$.
\end{solution}

\begin{solution}{exo:g9:linfunc:4}
$f$ gaat door de oorsprong met \omterm{prop:g9:linfunc:affinegraph}{richtingscoëfficiënt} $2$; $g$ heeft
dezelfde \omterm{prop:g9:linfunc:affinegraph}{richtingscoëfficiënt} maar snijdt de $y$-as in $3$; $h$ daalt
vanuit $(0, 4)$ met \omterm{prop:g9:linfunc:affinegraph}{richtingscoëfficiënt} $-1$. De grafieken van $f$ en
$g$ zijn \emph{\omterm{def:g6:lines:perp}{evenwijdige lijnen}} (zelfde \omterm{prop:g9:linfunc:affinegraph}{richtingscoëfficiënt},
verschillende snijpunten).
\end{solution}

\begin{solution}{exo:g9:linfunc:5}
Voor $f$: \omterm{prop:g9:linfunc:affinegraph}{richtingscoëfficiënt} $\frac{11 - 5}{3 - 1} = 3$, daarna
$5 = 3 \times 1 + b$ geeft $b = 2$: $f(x) = 3x + 2$.

Voor $g$: \omterm{prop:g9:linfunc:affinegraph}{richtingscoëfficiënt} $\frac{0 - 4}{2 - 0} = -2$, en
$g(0) = 4$ is al het snijpunt: $g(x) = -2x + 4$.
\end{solution}

\begin{solution}{exo:g9:linfunc:6}
Stijging van $25\%$: $120 \times 1.25 = 150$.

Korting van $40\%$: $65 \times 0.6 = 39$.

Oorspronkelijke prijs $p$ met $p \times 0.75 = 69$:
$p = \frac{69}{0.75} = 92$.
\end{solution}

\begin{solution}{exo:g9:linfunc:7}
\emph{1.} $f(x) = 0.05x + 10$.

\emph{2.} Plan B is goedkoper wanneer $25 < 0.05x + 10$, d.w.z.\
$15 < 0.05x$, d.w.z.\ $x > 300$. Vanaf $301$ minuten ($5$ uur) per
maand wint plan B.
\end{solution}

\begin{solution}{exo:g9:linfunc:8}
\emph{1.} Na één uur: $2000 \times 1.1 = 2200$. Na twee uur:
$2200 \times 1.1 = 2420$.

\emph{2.} De tweede stijging geldt voor $2200$, niet voor $2000$:
tweemaal met $10\%$ groeien vermenigvuldigt met $1.1^2 = 1.21$, een
stijging van $21\%$, niet $20\%$.
\end{solution}

\begin{solution}{exo:g9:linfunc:9}
\emph{1.} \omterm{prop:g9:linfunc:affinegraph}{Richtingscoëfficiënt}: $\frac{1 - 3}{6 - 2} = \frac{-2}{4} =
-0.5$: de functie is dalend.

\emph{2.} $f(x) = -0.5x + b$ met $f(2) = 3$: $-1 + b = 3$, dus $b = 4$
en $f(x) = -0.5x + 4$. Output $0$: $-0.5x + 4 = 0$ geeft $x = 8$.
\end{solution}

\begin{solution}{exo:g9:linfunc:10}
\emph{1.} De twee veranderingen vermenigvuldigen de prijs met
\[
\left(1 + \frac{t}{100}\right)\left(1 - \frac{t}{100}\right)
= 1 - \frac{t^2}{10000}
\]
(derde identiteit met $a = 1$, $b = \frac{t}{100}$).

\emph{2.} Voor $0 < t \leq 100$ is $\frac{t^2}{10000} > 0$, dus de
factor is kleiner dan $1$: de eindprijs is lager dan de
oorspronkelijke. Voor $t = 10$: factor $1 - \frac{100}{10000} = 0.99$,
d.w.z.\ een overall daling van $1\%$.
\end{solution}
