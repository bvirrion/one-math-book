\section*{Hoofdstuk \ref{ch:g9:arith} --- Rekenkunde: delers en priemgetallen}

\begin{solution}{exo:g9:arith:1}
Delers van $36$: $1, 2, 3, 4, 6, 9, 12, 18, 36$.
Delers van $45$: $1, 3, 5, 9, 15, 45$.
Delers van $17$: alleen $1$ en $17$ ($17$ is priem).
\end{solution}

\begin{solution}{exo:g9:arith:2}
$2\,346$ eindigt op $6$: deelbaar door $2$, niet door $5$. Cijfersom
$2 + 3 + 4 + 6 = 15$: deelbaar door $3$, niet door $9$. Laatste twee
cijfers $46$, en $46 = 4 \times 11 + 2$ is niet deelbaar door $4$:
$2\,346$ is niet deelbaar door $4$.
\end{solution}

\begin{solution}{exo:g9:arith:3}
$72 = 2^3 \times 3^2$; \quad
$150 = 2 \times 3 \times 5^2$; \quad
$210 = 2 \times 3 \times 5 \times 7$; \quad
$121 = 11^2$.
\end{solution}

\begin{solution}{exo:g9:arith:4}
$101$: test de priemgetallen $p$ met $p^2 \leq 101$, d.w.z.\ $2, 3, 5,
7$. Geen deelt $101$ (oneven, cijfersom $2$, eindigt niet op $0/5$,
$101 = 7 \times 14 + 3$): $101$ is priem.

$91 = 7 \times 13$: niet priem.

$143 = 11 \times 13$: niet priem.
\end{solution}

\begin{solution}{exo:g9:arith:5}
Gemeenschappelijke delers van $48$ en $60$: delers van 48 zijn $1, 2,
3, 4, 6, 8, 12, 16, 24, 48$; delers van $60$ zijn $1, 2, 3, 4, 5, 6,
10, 12, 15, 20, 30, 60$; de gemeenschappelijke zijn $1, 2, 3, 4, 6,
12$, dus $\gcd(48,60) = 12$.

Via factorisatie: $48 = 2^4 \times 3$ en $60 = 2^2 \times 3 \times 5$;
gemeenschappelijke priemgetallen met kleinere exponenten:
$2^2 \times 3 = 12$.
\end{solution}

\begin{solution}{exo:g9:arith:6}
$\gcd(255, 154)$:
\begin{align*}
255 &= 154 \times 1 + 101, \\
154 &= 101 \times 1 + 53, \\
101 &= 53 \times 1 + 48, \\
53 &= 48 \times 1 + 5, \\
48 &= 5 \times 9 + 3, \\
5 &= 3 \times 1 + 2, \\
3 &= 2 \times 1 + 1, \\
2 &= 1 \times 2 + 0 .
\end{align*}
Laatste van-nul-verschillende rest: $\gcd(255, 154) = 1$ (ze zijn
onderling ondeelbaar).

$\gcd(1053, 325)$:
\begin{align*}
1053 &= 325 \times 3 + 78, \\
325 &= 78 \times 4 + 13, \\
78 &= 13 \times 6 + 0 .
\end{align*}
$\gcd(1053, 325) = 13$.
\end{solution}

\begin{solution}{exo:g9:arith:7}
Euclidisch algoritme: $588 = 504 \times 1 + 84$; $504 = 84 \times 6 + 0$:
$\gcd(588, 504) = 84$. Dan
\[
\frac{588}{504} = \frac{588 \div 84}{504 \div 84} = \frac{7}{6},
\]
wat onherleidbaar is.
\end{solution}

\begin{solution}{exo:g9:arith:8}
Het aantal boeketten moet zowel $84$ als $126$ delen; het grootst
mogelijke is $\gcd(84, 126)$. Factorisaties: $84 = 2^2 \times 3 \times
7$, $126 = 2 \times 3^2 \times 7$, dus de ggd is $2 \times 3 \times 7 =
42$. Ze kan $42$ boeketten maken, elk met $\frac{84}{42} = 2$ rozen en
$\frac{126}{42} = 3$ tulpen.
\end{solution}

\begin{solution}{exo:g9:arith:9}
We zoeken het kleinste gemeenschappelijke veelvoud. $24 = 2^3 \times 3$
en $36 = 2^2 \times 3^2$; elk priemgetal met de \emph{grotere}
exponent: $\mathrm{kgv} = 2^3 \times 3^2 = 72$. De veerboten
vertrekken de volgende keer samen $72$ minuten na 8:00, om 9:12.
\end{solution}

\begin{solution}{exo:g9:arith:10}
\emph{1.} Elke gemeenschappelijke deler $d$ van $n$ en $n+1$ deelt ook
hun verschil $(n+1) - n = 1$, dus $d = 1$: $\gcd(n, n+1) = 1$.

\emph{2.} Een breuk is onherleidbaar precies wanneer teller en noemer
onderling ondeelbaar zijn, wat voor $n$ en $n + 1$ het geval is door
deel 1.
\end{solution}
