\section*{Hoofdstuk \ref{ch:g9:solids} --- Lichamen en volumes}

\begin{solution}{exo:g9:solids:1}
Balk: $V = 4 \times 5 \times 12 = 240$.

Cilinder: $V = \pi r^2 h = \pi \times 9 \times 7 = 63\pi \approx 198$.
\end{solution}

\begin{solution}{exo:g9:solids:2}
Kegel: $V = \frac13 \pi r^2 h = \frac13 \pi \times 25 \times 12 =
100\pi \approx 314$.

Piramide: grondvlakoppervlakte $B = 8 \times 3 = 24$, dus
$V = \frac13 \times 24 \times 10 = 80$.
\end{solution}

\begin{solution}{exo:g9:solids:3}
Volume: $V = \frac43 \pi \times 6^3 = \frac43 \pi \times 216 = 288\pi$.

Oppervlakte: $4\pi \times 6^2 = 144\pi$.
\end{solution}

\begin{solution}{exo:g9:solids:4}
Straal van de doorsnede: $\sqrt{r^2 - d^2} = \sqrt{100 - 64} =
\sqrt{36} = 6$.
\end{solution}

\begin{solution}{exo:g9:solids:5}
Schaalfactor van de hele kegel naar de kleine:
$k = \frac{9}{12} = \frac34$. Doorsnedestraal: $8 \times \frac34 = 6$.

Volume van de hele kegel: $\frac13\pi \times 64 \times 12 = 256\pi$.
Volume van de kleine kegel: $256\pi \times \left(\frac34\right)^3 =
256\pi \times \frac{27}{64} = 108\pi \approx 339$.
\end{solution}

\begin{solution}{exo:g9:solids:6}
Volume van de bal: $\frac43\pi \times 2^3 = \frac{32\pi}{3}$ cm$^3$.
Dit volume spreidt zich over de grondvlakoppervlakte
$\pi \times 3^2 = 9\pi$ cm$^2$, dus het niveau stijgt met
\[
\frac{32\pi/3}{9\pi} = \frac{32}{27} \approx 1.2 \text{ cm}
\]
(ongeveer $12$ mm).
\end{solution}

\begin{solution}{exo:g9:solids:7}
De kleine mal is de verkleining van de grote met $k = \frac36 =
\frac12$, dus haar volume is $\left(\frac12\right)^3 = \frac18$ van de
grote: het recept vult $8$ kleine mallen.
\end{solution}

\begin{solution}{exo:g9:solids:8}
$V = \frac13 \times 230^2 \times 147 = \frac13 \times 52\,900 \times 147
= 52\,900 \times 49 = 2\,592\,100$ m$^3$ --- ongeveer $2.6$ miljoen
kubieke meters.
\end{solution}

\begin{solution}{exo:g9:solids:9}
\emph{1.} Het water vormt een kegel geschaald met $k = \frac12$ (top
naar beneden, half de hoogte), dus zijn volume is
$\left(\frac12\right)^3 = \frac18$ van de trechter: één achtste, veel
minder dan de helft!

\emph{2.} Water tot hoogte $d$ vormt een kegel geschaald met
$k = \frac{d}{12}$, van volume $k^3$ maal dat van de trechter. De
helft van het volume betekent $k^3 = \frac12$, d.w.z.\
$\left(\frac{d}{12}\right)^3 = \frac12$. Dan
$\frac{d}{12} = \sqrt[3]{0.5} \approx 0.7937$, dus
$d \approx 9.5$ cm: de tweede helft van het volume neemt slechts de
bovenste $2.5$ cm in --- het brede deel van de kegel houdt het meeste
water.
\end{solution}
