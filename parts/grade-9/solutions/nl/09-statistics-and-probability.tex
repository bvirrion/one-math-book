\section*{Hoofdstuk \ref{ch:g9:statproba} --- Statistiek en kansrekening}

\begin{solution}{exo:g9:statproba:1}
Gemiddelde: $\dfrac{14 + 9 + 17 + 9 + 12 + 11 + 12}{7} = \dfrac{84}{7}
= 12$.

Mediaan: gesorteerd, $9, 9, 11, 12, 12, 14, 17$; de middelste ($4$e)
waarde is $12$.

Spreidingsbreedte: $17 - 9 = 8$.
\end{solution}

\begin{solution}{exo:g9:statproba:2}
Gemiddelde: $\dfrac{19 + 21 + 24 + 18 + 22 + 25 + 19}{7} =
\dfrac{148}{7} \approx 21.1$ graden.

Mediaan: gesorteerd, $18, 19, 19, 21, 22, 24, 25$; de $4$e waarde is
$21$ graden.
\end{solution}

\begin{solution}{exo:g9:statproba:3}
Gemiddelde:
\[
\frac{2 \times 0 + 4 \times 1 + 3 \times 2 + 1 \times 3}{10}
= \frac{0 + 4 + 6 + 3}{10} = \frac{13}{10} = 1.3
\text{ doelpunten per wedstrijd.}
\]
Mediaan: de gesorteerde reeks is $0,0,1,1,1,1,2,2,2,3$; de $5$e en
$6$e waarden zijn beide $1$: mediaan $1$.
\end{solution}

\begin{solution}{exo:g9:statproba:4}
$\P(3) = \frac16$. \quad
$\P(\text{even}) = \frac36 = \frac12$. \quad
$\P(\text{minstens } 3) = \P(\{3,4,5,6\}) = \frac46 = \frac23$. \quad
$\P(\text{geen } 6) = 1 - \frac16 = \frac56$.
\end{solution}

\begin{solution}{exo:g9:statproba:5}
$\P(\text{wit}) = \frac{5}{12}$.

$\P(\text{rood of groen}) = \frac{4 + 3}{12} = \frac{7}{12}$.

$\P(\text{niet wit}) = 1 - \frac{5}{12} = \frac{7}{12}$.

De laatste twee zijn gelijk: ``niet wit'' \emph{is} de gebeurtenis
``rood of groen''.
\end{solution}

\begin{solution}{exo:g9:statproba:6}
De $12$ uitkomsten (zijde van de munt, waarde van de dobbelsteen) zijn
even waarschijnlijk.

Munt en een $6$: één uitkomst van de $12$: $\frac{1}{12}$.

Munt of een $6$: de uitkomsten met munt ($6$ ervan) plus de uitkomst
(kruis, $6$): $7$ uitkomsten, dus $\frac{7}{12}$. Elke gebeurtenis
tellen en de overlap aftrekken geeft hetzelfde:
$\frac{6}{12} + \frac{2}{12} - \frac{1}{12} = \frac{7}{12}$.
\end{solution}

\begin{solution}{exo:g9:statproba:7}
Eerste trekking: rood $\frac25$, zwart $\frac35$. Tweede trekking
zonder teruglegging: na een rood blijven $1$ rood en $3$ zwart van de
$4$; na een zwart $2$ rood en $2$ zwart van de $4$.

Twee zwart: $\frac35 \times \frac24 = \frac{6}{20} = \frac{3}{10}$.

Verschillende kleuren (RZ of ZR):
$\frac25 \times \frac34 + \frac35 \times \frac24
= \frac{6}{20} + \frac{6}{20} = \frac{12}{20} = \frac35$.
\end{solution}

\begin{solution}{exo:g9:statproba:8}
\emph{1.} $9 \times 14 = 126$ punten.

\emph{2.} Een gemiddelde van $15$ over $10$ wedstrijden betekent
$150$ punten in totaal: ze moet $150 - 126 = 24$ punten scoren.
\end{solution}

\begin{solution}{exo:g9:statproba:9}
\emph{1.} Onder de $36$ even waarschijnlijke paren zijn de doubles
$(1,1), (2,2), \dots, (6,6)$: zes ervan, dus
$\P(\text{double}) = \frac{6}{36} = \frac16$.

\emph{2.} Twee onafhankelijke spellen: boom met winst ($\frac16$) /
verlies ($\frac56$) op elk niveau. Kans om nooit te winnen:
$\frac56 \times \frac56 = \frac{25}{36}$. Dus
\[
\P(\text{minstens één winst}) = 1 - \frac{25}{36} = \frac{11}{36}
\approx 0.31 .
\]
\end{solution}
