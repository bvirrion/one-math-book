\section*{Hoofdstuk \ref{ch:g9:algebra} --- Algebra en vergelijkingen}

\begin{solution}{exo:g9:algebra:1}
$4(3x - 2) = 12x - 8$.

$(x+6)(x+2) = x^2 + 2x + 6x + 12 = x^2 + 8x + 12$.

$(2x-3)(x+5) = 2x^2 + 10x - 3x - 15 = 2x^2 + 7x - 15$.

$(x-1)(x-9) = x^2 - 9x - x + 9 = x^2 - 10x + 9$.
\end{solution}

\begin{solution}{exo:g9:algebra:2}
$(x+8)^2 = x^2 + 16x + 64$.

$(3x-4)^2 = 9x^2 - 24x + 16$.

$(x+10)(x-10) = x^2 - 100$.

$(5x+2)(5x-2) = 25x^2 - 4$.
\end{solution}

\begin{solution}{exo:g9:algebra:3}
$101^2 = (100+1)^2 = 10000 + 200 + 1 = 10201$.

$59^2 = (60-1)^2 = 3600 - 120 + 1 = 3481$.

$53 \times 47 = (50+3)(50-3) = 2500 - 9 = 2491$.
\end{solution}

\begin{solution}{exo:g9:algebra:4}
$9x + 12 = 3(3x + 4)$.

$x^2 - 64 = (x+8)(x-8)$.

$x^2 + 14x + 49 = (x+7)^2$.

$18x^2 - 6x = 6x(3x - 1)$.
\end{solution}

\begin{solution}{exo:g9:algebra:5}
$4x + 7 = 19$: $4x = 12$, dus $x = 3$.

$6x - 5 = 2x + 11$: $4x = 16$, dus $x = 4$.

$5(x-3) = 3x + 1$: $5x - 15 = 3x + 1$, dus $2x = 16$ en $x = 8$.
\end{solution}

\begin{solution}{exo:g9:algebra:6}
$(x+4)(3x-12) = 0$: $x = -4$ of $x = 4$.

$x^2 - 121 = (x-11)(x+11) = 0$: $x = 11$ of $x = -11$.

$(2x-1)^2 = 0$: de enige oplossing $x = \frac12$.

$x^2 = 8x$: $x(x - 8) = 0$, dus $x = 0$ of $x = 8$.
\end{solution}

\begin{solution}{exo:g9:algebra:7}
\omterm{ex:g1:subtraction:difference}{Verschil} van kwadraten met $a = 3x + 2$ en $b = 5$:
\[
E = (3x + 2)^2 - 5^2 = (3x + 2 + 5)(3x + 2 - 5) = (3x + 7)(3x - 3).
\]
Nulproductregel: $3x + 7 = 0$ geeft $x = -\frac73$, en $3x - 3 = 0$
geeft $x = 1$.
\end{solution}

\begin{solution}{exo:g9:algebra:8}
$3x - 4 \geq 8$: $3x \geq 12$, dus $x \geq 4$ (gevulde stip bij $4$,
arcering naar rechts).

$5 - 2x > 1$: $-2x > -4$, delen door $-2$ keert het teken om: $x < 2$
(holle stip bij $2$, arcering naar links).

$4(x+1) \leq x - 5$: $4x + 4 \leq x - 5$, dus $3x \leq -9$, dus
$x \leq -3$ (gevulde stip bij $-3$, arcering naar links).
\end{solution}

\begin{solution}{exo:g9:algebra:9}
Laat $x$ het getal zijn: $3x + 7 = 5x - 9$. Dan $16 = 2x$, dus
$x = 8$. Controle: $3 \times 8 + 7 = 31$ en $5 \times 8 - 9 = 31$.
\end{solution}

\begin{solution}{exo:g9:algebra:10}
\emph{1.} $(n+1)(n-1) = n^2 - 1$ (derde identiteit). De buren van $n$
zijn $n - 1$ en $n + 1$, dus hun \omterm{def:g2:mult:def}{product} is $n^2 - 1$: bijv.\
$24 \times 26 = 625 - 1 = 624$.

\emph{2.} Drie opeenvolgende gehele getallen kunnen geschreven worden
als $n - 1$, $n$, $n + 1$. Hun \omterm{def:g1:addition:def}{som} is
\[
(n-1) + n + (n+1) = 3n ,
\]
een \omterm{def:g9:arith:divisor}{veelvoud} van $3$ (driemaal de middelste).
\end{solution}
