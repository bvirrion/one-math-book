\section*{Hoofdstuk \ref{ch:g9:trig} --- Goniometrie in de rechthoekige driehoek}

\begin{solution}{exo:g9:trig:1}
Schuine zijde: $\sqrt{9^2 + 12^2} = \sqrt{81 + 144} = \sqrt{225} = 15$.

Ander been: $\sqrt{17^2 - 8^2} = \sqrt{289 - 64} = \sqrt{225} = 15$.
\end{solution}

\begin{solution}{exo:g9:trig:2}
$7^2 + 24^2 = 49 + 576 = 625 = 25^2$: ja, \omterm{def:g6:shapes:triangles}{rechthoekig} (omgekeerde van
Pythagoras), met de \omterm{def:g3:shapes:rightangle}{rechte hoek} tegenover de zijde $25$.

$5^2 + 6^2 = 61 \neq 64 = 8^2$: niet \omterm{def:g6:shapes:triangles}{rechthoekig}.
\end{solution}

\begin{solution}{exo:g9:trig:3}
De schuine zijde is $[EF]$ (tegenover de \omterm{def:g3:shapes:rightangle}{rechte hoek} in $D$). De
zijde tegenover $\theta = \widehat E$ is $[DF]$; de aanliggende
zijde is $[DE]$. Dus
\[
\cos\theta = \frac{DE}{EF}, \qquad
\sin\theta = \frac{DF}{EF}, \qquad
\tan\theta = \frac{DF}{DE}.
\]
\end{solution}

\begin{solution}{exo:g9:trig:4}
Overstaand been: $10 \sin 28^\circ \approx 10 \times 0.469 = 4.7$.
Aanliggend been: $10 \cos 28^\circ \approx 10 \times 0.883 = 8.8$.
(Controle: $4.7^2 + 8.8^2 \approx 22 + 77 \approx 100$.)
\end{solution}

\begin{solution}{exo:g9:trig:5}
$\tan\theta = \dfrac{4.5}{6} = 0.75$, dus
$\theta = \tan^{-1}(0.75) \approx 37^\circ$.
\end{solution}

\begin{solution}{exo:g9:trig:6}
In de \omterm{def:g6:shapes:triangles}{rechthoekige driehoek} gevormd door de waarnemer, het punt op de
grond onder de drone, en de drone: de hoogte $120$ m is de zijde
tegenover de elevatiehoek, de horizontale afstand $d$ is aanliggend.
Dus
\[
\tan 32^\circ = \frac{120}{d},
\qquad
d = \frac{120}{\tan 32^\circ} \approx \frac{120}{0.625} = 192 \text{ m}.
\]
\end{solution}

\begin{solution}{exo:g9:trig:7}
De stijging ($1.2$ m) ligt tegenover de hoek; de hellinglengte $L$ is
de schuine zijde:
\[
\sin 5^\circ = \frac{1.2}{L},
\qquad
L = \frac{1.2}{\sin 5^\circ} \approx \frac{1.2}{0.0872} \approx 13.8
\text{ m}.
\]
De helling moet minstens $13.8$ m lang zijn.
\end{solution}

\begin{solution}{exo:g9:trig:8}
Uit $\cos^2\theta + \sin^2\theta = 1$:
$\sin^2\theta = 1 - 0.36 = 0.64$, en $\sin\theta > 0$ voor een
scherpe hoek, dus $\sin\theta = 0.8$. Dan
$\tan\theta = \dfrac{\sin\theta}{\cos\theta} = \dfrac{0.8}{0.6} =
\dfrac43$.
\end{solution}

\begin{solution}{exo:g9:trig:9}
\emph{Rechthoekige \omterm{def:g6:shapes:triangles}{gelijkbenige driehoek}} met benen $1$: schuine
zijde $\sqrt{1 + 1} = \sqrt2$; elke scherpe hoek is $45^\circ$, en
\[
\cos 45^\circ = \frac{1}{\sqrt2} = \frac{\sqrt2}{2} = \sin 45^\circ,
\qquad
\tan 45^\circ = \frac11 = 1 .
\]

\emph{\omterm{def:g6:shapes:triangles}{Gelijkzijdige driehoek}} met zijde $1$: zijn hoogte splitst hem
in twee \omterm{def:g6:shapes:triangles}{rechthoekige driehoeken} met schuine zijde $1$, basis
$\frac12$, en hoogte $h = \sqrt{1 - \frac14} = \frac{\sqrt3}{2}$
(Pythagoras). De hoeken zijn $60^\circ$ (aan de basis) en $30^\circ$
(bovenaan). Aflezen van de verhoudingen:
\[
\cos 60^\circ = \frac{1/2}{1} = \frac12, \quad
\sin 60^\circ = \frac{\sqrt3/2}{1} = \frac{\sqrt3}{2}, \quad
\cos 30^\circ = \frac{\sqrt3}{2}, \quad
\sin 30^\circ = \frac12 .
\]
\end{solution}
