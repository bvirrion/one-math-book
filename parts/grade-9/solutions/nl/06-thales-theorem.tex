\section*{Hoofdstuk \ref{ch:g9:thales} --- De stelling van Thales}

\begin{solution}{exo:g9:thales:1}
Thales: $\dfrac{AM}{AB} = \dfrac{AN}{AC} = \dfrac{MN}{BC}$, d.w.z.\
$\dfrac26 = \dfrac13$. Dus $\dfrac{3}{AC} = \dfrac13$ geeft $AC = 9$,
en $\dfrac{MN}{9} = \dfrac13$ geeft $MN = 3$.
\end{solution}

\begin{solution}{exo:g9:thales:2}
Vind eerst $AB = AM + MB = 5 + 3 = 8$. Dan
$\dfrac{AM}{AB} = \dfrac{AN}{AC}$ geeft $\dfrac58 = \dfrac{4}{AC}$, dus
$5 \times AC = 32$ en $AC = 6.4$.
\end{solution}

\begin{solution}{exo:g9:thales:3}
De vlinder werkt precies als de geneste configuratie:
$\dfrac{AM}{AB} = \dfrac{AN}{AC} = \dfrac{MN}{BC}$, d.w.z.\
$\dfrac{3}{7.5} = \dfrac25$. Dus $\dfrac{2}{AC} = \dfrac25$ geeft
$AC = 5$, en $\dfrac{2.6}{BC} = \dfrac25$ geeft
$BC = \dfrac{2.6 \times 5}{2} = 6.5$.
\end{solution}

\begin{solution}{exo:g9:thales:4}
$\dfrac{AM}{AB} = \dfrac68 = \dfrac34$ en
$\dfrac{AN}{AC} = \dfrac{9}{12} = \dfrac34$: gelijke verhoudingen,
punten in dezelfde volgorde, dus $(MN) \parallel (BC)$ door het
omgekeerde van Thales.
\end{solution}

\begin{solution}{exo:g9:thales:5}
$\dfrac{AM}{AB} = \dfrac46 = \dfrac23$ en
$\dfrac{AN}{AC} = \dfrac58$. Kruiscontrole: $\dfrac23 = \dfrac{16}{24}$
en $\dfrac58 = \dfrac{15}{24}$: de verhoudingen verschillen. Als de
lijnen \omterm{def:g6:lines:perp}{evenwijdig} waren, zou de stelling van Thales ze gelijk forceren
--- dus zijn $(MN)$ en $(BC)$ \emph{niet} \omterm{def:g6:lines:perp}{evenwijdig}.
\end{solution}

\begin{solution}{exo:g9:thales:6}
Laat $S$ de top van de schaduw zijn, $P$ de top van het hoofd van de
persoon, $L$ de top van de lamp. De persoon ($1.8$ m op afstand $3$ m
van $S$) en de lamp (hoogte $h$ op afstand $3 + 2 = 5$ m van $S$) zijn
twee evenwijdige verticale segmenten gesneden door de lichtstraal
$(SL)$: Thales vanuit $S$ geeft
\[
\frac{1.8}{h} = \frac{3}{5},
\qquad\text{dus}\quad
h = \frac{1.8 \times 5}{3} = 3 \text{ m}.
\]
\end{solution}

\begin{solution}{exo:g9:thales:7}
\emph{1.} Echte afstand: $6.8 \times 25\,000 = 170\,000$ cm $=
1.7$ km.

\emph{2.} \omterm{def:g6:measure:area}{Oppervlakten} schalen met $k^2$: echte \omterm{def:g6:measure:area}{oppervlakte} $=
8 \times 25\,000^2$ cm$^2$ $= 8 \times 6.25 \times 10^8$ cm$^2$ $=
5 \times 10^9$ cm$^2$. Omdat $1$ km$^2 = 10^{10}$ cm$^2$, is dat
$0.5$ km$^2$.
\end{solution}

\begin{solution}{exo:g9:thales:8}
\emph{1.} De verhouding is $\dfrac{AM}{AB} = \dfrac{8}{12} =
\dfrac23$. Thales: $AN = \frac23 \times 15 = 10$ en
$MN = \frac23 \times 18 = 12$.

\emph{2.} De \omterm{def:g9:thales:scaling}{schaalfactor} is $k = \frac23$. Omtrekken: $ABC$ heeft
$12 + 15 + 18 = 45$, en $AMN$ heeft $8 + 10 + 12 = 30 = \frac23 \times
45$: de \omterm{def:g6:measure:perimeter}{omtrek} schaalt ook met $k$, zoals elke lengte.
\end{solution}

\begin{solution}{exo:g9:thales:9}
Rond $O$ snijden de lijnen $(AC)$ en $(BD)$, en
$(AB) \parallel (CD)$: vlinderconfiguratie met de driehoeken $OAB$
en $OCD$. Thales geeft
\[
\frac{OA}{OC} = \frac{OB}{OD} = \frac{AB}{CD} = \frac46 = \frac23 .
\]
Dus $\dfrac{OA}{OC} = \dfrac23$, en de gelijkheid
$\dfrac{OA}{OC} = \dfrac{OB}{OD}$ zegt precies dat $O$ de twee
diagonalen in dezelfde verhouding snijdt.
\end{solution}
