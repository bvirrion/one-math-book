\section*{अध्याय \ref{ch:g9:sqrt} --- वर्गमूल}

\begin{solution}{exo:g9:sqrt:1}
$\sqrt{64} = 8$; \quad $\sqrt{100} = 10$; \quad
$\sqrt{0.09} = 0.3$ (तब से $0.3^2 = 0.09$); \quad
$\sqrt{\frac{1}{25}} = \frac15$; \quad
$\left(\sqrt{13}\right)^2 = 13$; \quad
$\sqrt{(-4)^2} = \sqrt{16} = 4 = \abs{-4}$.
\end{solution}

\begin{solution}{exo:g9:sqrt:2}
$\sqrt{50} = \sqrt{25 \times 2} = 5\sqrt2$; \quad
$\sqrt{45} = \sqrt{9 \times 5} = 3\sqrt5$; \quad
$\sqrt{98} = \sqrt{49 \times 2} = 7\sqrt2$; \quad
$\sqrt{300} = \sqrt{100 \times 3} = 10\sqrt3$.
\end{solution}

\begin{solution}{exo:g9:sqrt:3}
$\sqrt2 \times \sqrt{18} = \sqrt{36} = 6$.

$\dfrac{\sqrt{75}}{\sqrt3} = \sqrt{\dfrac{75}{3}} = \sqrt{25} = 5$.

$\sqrt5 \times \sqrt{20} = \sqrt{100} = 10$.

$\dfrac{\sqrt8}{\sqrt{50}} = \sqrt{\dfrac{8}{50}} = \sqrt{\dfrac{4}{25}}
= \dfrac25$.
\end{solution}

\begin{solution}{exo:g9:sqrt:4}
$ x^2 = 36$: $ x = 6$ या $ x = -6$.

$ x^2 = 11$: $ x = \sqrt{11}$ या $ x = -\sqrt{11}$.

$ x^2 + 4 = 0$ मतलब $ x^2 = -4 < 0$: कोई समाधान नहीं.

$2x^2 = 50$: $ x^2 = 25$, इसलिए $ x = 5$ या $ x = -5$.
\end{solution}

\begin{solution}{exo:g9:sqrt:5}
$\sqrt{20} + \sqrt{45} = 2\sqrt5 + 3\sqrt5 = 5\sqrt5$.

$3\sqrt8 - \sqrt{18} + \sqrt2 = 3 \times 2\sqrt2 - 3\sqrt2 + \sqrt2
= (6 - 3 + 1)\sqrt2 = 4\sqrt2$.
\end{solution}

\begin{solution}{exo:g9:sqrt:6}
$\left(\sqrt3 + 1\right)^2 = 3 + 2\sqrt3 + 1 = 4 + 2\sqrt3$.

$\left(2\sqrt5 - 3\right)^2 = 4 \times 5 - 2 \times 3 \times 2\sqrt5 + 9
= 29 - 12\sqrt5$.

$\left(\sqrt6 + \sqrt2\right)\left(\sqrt6 - \sqrt2\right) = 6 - 2 = 4$.
\end{solution}

\begin{solution}{exo:g9:sqrt:7}
ओर: $\sqrt{6400} = 80$ एम। विकर्ण (पाइथागोरस):
$\sqrt{80^2 + 80^2} = 80\sqrt2 \approx 113$ एम।
\end{solution}

\begin{solution}{exo:g9:sqrt:8}
$\dfrac{1}{\sqrt2} = \dfrac{1 \times \sqrt2}{\sqrt2 \times \sqrt2}
= \dfrac{\sqrt2}{2}$.

$\dfrac{6}{\sqrt3} = \dfrac{6\sqrt3}{3} = 2\sqrt3$.
\quad
$\dfrac{10}{\sqrt5} = \dfrac{10\sqrt5}{5} = 2\sqrt5$.
\end{solution}

\begin{solution}{exo:g9:sqrt:9}
\emph{1. सत्य}: यह है \omterm{def:g2:mult:def}{उत्पाद} नियम (\cref{thm:g9:sqrt:rules}).

\emph{2. मिथ्या}: $\sqrt{9 + 16} = 5$ लेकिन $\sqrt9 + \sqrt{16} = 7$.

\emph{3. मिथ्या} नकारात्मक के लिए $ x $: $\sqrt{(-4)^2} = 4 \neq -4$. द
सही कथन है $\sqrt{x^2} = \abs{x}$.

\emph{4. सत्य}: $\left(3\sqrt2\right)^2 = 9 \times 2 = 18$.
\end{solution}

\begin{solution}{exo:g9:sqrt:10}
$\left(2 + \sqrt3\right)^2 = 4 + 4\sqrt3 + 3 = 7 + 4\sqrt3$. इसलिए
$ x = \sqrt{7 + 4\sqrt3} = \sqrt{\left(2+\sqrt3\right)^2} = 2 + \sqrt3$
(एक गैर-ऋणात्मक संख्या, इसलिए मूल केवल वर्ग को हटा देता है)।

उसी प्रकार $\left(\sqrt5 - 2\right)^2 = 5 - 4\sqrt5 + 4 = 9 - 4\sqrt5$,
और $\sqrt5 - 2 > 0$, इसलिए
$\sqrt{9 - 4\sqrt5} = \sqrt5 - 2$ (नहीं $2 - \sqrt5$, जो है
नकारात्मक!)
\end{solution}

\begin{solution}{pb:g9:sqrt:1}
\textbf{1.} $1.4^2 = 1.96 < 2 < 2.25 = 1.5^2$;
$1.41^2 = 1.9881 < 2 < 2.0164 = 1.42^2$. तीन दशमलव:
$1.414^2 = 1.999396 < 2 < 2.002225 = 1.415^2$, इसलिए
$1.414 < \sqrt2 < 1.415$.

\textbf{2.} बराबरी $\sqrt2 = \frac ab $ देता है
$2 = \frac{a^2}{b^2}$, इस तरह $ a^2 = 2b^2$: जो नंबर $ a^2$ है
एक पूर्ण संख्या का दो बार --- \emph{\omterm{def:g3:numbers:evenodd}{यहां तक की}}.

\textbf{3.} अगर $ a $ थे \omterm{def:g3:numbers:evenodd}{विषम}, $ a^2$ होगा \omterm{def:g3:numbers:evenodd}{विषम}
(\cref{pb:g8:pythagoras:1}); तब से $ a^2$ है \omterm{def:g3:numbers:evenodd}{यहां तक की}, $ a $ है \omterm{def:g3:numbers:evenodd}{यहां तक की}:
$ a = 2k $. प्रतिस्थापन: $4k^2 = 2b^2$, इसलिए $ b^2 = 2k^2$ है \omterm{def:g3:numbers:evenodd}{यहां तक की},
और उसी समता तथ्य से $ b $ है \omterm{def:g3:numbers:evenodd}{यहां तक की}.

\textbf{4.} दोनों $ a $ और $ b $ \omterm{def:g3:numbers:evenodd}{यहां तक की} दोनों का मतलब है \omterm{def:g6:wholes:divisible}{भाज्य} द्वारा $2$
--- लेकिन $\frac ab $ सबसे कम शर्तों पर था, साझा करना कोई सामान्य नहीं था
\omterm{def:g5:division:multiple}{भाजक}. विरोधाभास: कल्पित \omterm{def:g9:fractions:fraction}{अंश} अस्तित्व में नहीं रह सकता.
\emph{प्रमेय: $\sqrt2$ तर्कहीन है.}

\textbf{5.} विरोधाभास पूरी तरह से ``निम्नतम'' में रहता है
शर्तें'': इसके बिना, ``$ a $ और $ b $ दोनों \omterm{def:g3:numbers:evenodd}{यहां तक की}'' विरोधाभास
कुछ नहीं. न्यूनतम शर्तों को मानना वैध है क्योंकि प्रत्येक
\omterm{def:g9:fractions:fraction}{अंश} \emph{है} सबसे कम शर्तों वाला फॉर्म
(\cref{met:g9:arith:simplify}: जीसीडी को \omterm{def:g9:arith:divisor}{विभाजित} करें) --- यदि
$\sqrt2$ कोई भी थे \omterm{def:g9:fractions:fraction}{अंश} बिल्कुल भी, यह न्यूनतम शर्तें होंगी
एक, और वह असंभव है।

\textbf{6.} अभाज्य गुणनखंडों में, वर्ग चलते हैं \omterm{def:g3:numbers:evenodd}{यहां तक की}
घातांक (\cref{pb:g9:arith:1}). में $ a^2 = 3b^2$, द \omterm{def:g8:powers:def}{प्रतिपादक}
का $3$ है \omterm{def:g3:numbers:evenodd}{यहां तक की} बाईं ओर, लेकिन \omterm{def:g3:numbers:evenodd}{विषम} दायीं तरफ (\omterm{def:g3:numbers:evenodd}{यहां तक की} से
$ b^2$, मैं भी सहमत हूं)। किसी भी संख्या के दो गुणनखंड नहीं होते
(\cref{thm:g9:arith:factorization}): विरोधाभास, और
$\sqrt3$ तर्कहीन है.

\textbf{7.} में $ a^2 = n b^2$, तर्क को एक अभाज्य मिलता है
विरोधाभासी \omterm{def:g8:powers:def}{प्रतिपादक} समता बिल्कुल तब जब कुछ अभाज्य प्रकट होता है
$ n $ एक साथ \emph{\omterm{def:g3:numbers:evenodd}{विषम}} \omterm{def:g8:powers:def}{प्रतिपादक}. यदि प्रत्येक \omterm{def:g8:powers:def}{प्रतिपादक} का $ n $ है
\omterm{def:g3:numbers:evenodd}{यहां तक की}, $ n $ एक पूर्ण वर्ग है ($ n = m^2$, और
$\sqrt n = m $ संपूर्ण है)। पूर्ण परिणाम: $\sqrt n $ है
प्रत्येक संपूर्ण के लिए तर्कहीन $ n $ वह पूर्ण वर्ग नहीं है ---
$\sqrt2, \sqrt3, \sqrt5, \sqrt6, \sqrt7, \sqrt8, \sqrt{10},
\dots $

\textbf{8.} से $ r x = q $ साथ $ r \neq 0$:
$ x = \frac qr $, ए \omterm{def:g3:division:remainder}{भागफल} तर्कसंगत, इसलिए तर्कसंगत। तो यदि
$ x $ तर्कहीन है, $ r x $ तर्कसंगत नहीं हो सकता: $2\sqrt2$ और
$\frac{\sqrt2}{2}$ अपरिमेय (गुणक) हैं $2$ और
$\frac12$).

\textbf{9.} अगर $1 + \sqrt2 = q $ तब तर्कसंगत थे
$\sqrt2 = q - 1$ तर्कसंगत होगा: विरोधाभास. अगर
$ s = \sqrt2 + \sqrt3$ तब तर्कसंगत थे
$ s^2 = 2 + 2\sqrt6 + 3 = 5 + 2\sqrt6$ भी होगा, देना
$\sqrt6 = \frac{s^2 - 5}{2}$ तर्कसंगत --- लेकिन $\sqrt6$ है
तर्कहीन (प्रश्न 7). इसलिए $\sqrt2 + \sqrt3$ तर्कहीन है.

\textbf{10.} रकम: $\sqrt2$ और $-\sqrt2$ (या $\sqrt2$ और
$1 - \sqrt2$) प्रत्येक तर्कहीन हैं, तर्कसंगत के साथ रकम $0$ और
$1$. \omterm{def:g2:mult:def}{उत्पाद}ों: $\sqrt2 \times \sqrt2 = 2$. अतार्किकता हो सकती है
अंकगणित में लुप्त हो जाना --- इसलिए केस-दर-केस प्रमाण।

\textbf{11.} से $ x = 1$: का औसत $1$ और $\frac21 = 2$:
$ x_1 = \frac32$. फिर का औसत $\frac32$ और
$\frac{2}{3/2} = \frac43$:
$ x_2 = \frac12\left(\frac32 + \frac43\right)
= \frac12 \cdot \frac{17}{6} = \frac{17}{12} \approx 1.4167$.

\textbf{12.} $ x_3 = \frac12\left(\frac{17}{12} +
\frac{24}{17}\right) = \frac12 \cdot
\frac{289 + 288}{204} = \frac{577}{408} = 1.4142156\ldots $
ख़िलाफ़ $\sqrt2 = 1.4142135\ldots $: पाँच दशमलव के बाद सही
तीन चरण --- यह नुस्खा लगभग सही दशमलव को दोगुना कर देता है
प्रत्येक \omterm{def:g4:numbers:round}{गोल}.

\textbf{13.} अगर $ x^2 > 2$ (अनुमान बहुत बड़ा है), फिर $ x > \sqrt2$,
और $\frac2x < \frac{2}{\sqrt2} = \sqrt2$: सहचर पक्ष है
बहुत छोटा. अगर $ x^2 < 2$, असमानताएँ उलट जाती हैं। किसी भी तरह से
$\sqrt2$ के बीच बैठता है $ x $ और $\frac2x $, तो उनका औसत ---
अगला अनुमान --- दोनों पक्षों की तुलना में अधिक निकट है:
का आयत \omterm{def:g6:measure:area}{क्षेत्र} $2$ हर कदम पर चौकन्ना हो जाता है.

\textbf{14.} $9 - 8 = 1$; \ $289 - 288 = 1$; \
$332\,929 - 332\,928 = 1$. हर बगुला अनुमान लगाता है $\frac ab $
संतुष्ट $ a^2 - 2b^2 = 1$: भाग I ने उस प्रहार को सिद्ध किया $0$ है
असंभव, और ये \omterm{def:g9:fractions:fraction}{अंश}ों असंभव को बिल्कुल चूक जाओ
एक इकाई --- निकटतम पूर्ण संख्याएँ कभी भी आ सकती हैं।

\textbf{15.} संख्या \omterm{def:g6:lines:objects}{रेखा} में इससे आगे की संख्याएँ होनी चाहिए
\omterm{def:g9:fractions:fraction}{अंश}ों --- लंबाई की तरह $\sqrt2$, जिसका दशमलव लेखन चलता है
बिना दोहराए हमेशा के लिए: द \emph{वास्तविक संख्या}. उनके
ईमानदार निर्माण हाई स्कूल वॉल्यूम के लिए एक कहानी है और,
पूरी कठोरता से, विश्वविद्यालय वाले।
\end{solution}
