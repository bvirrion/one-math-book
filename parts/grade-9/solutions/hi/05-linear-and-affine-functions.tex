\section*{अध्याय \ref{ch:g9:linfunc} --- रैखिक और एफ़िन फ़ंक्शंस}

\begin{solution}{exo:g9:linfunc:1}
रैखिक: $f $ और $ k $ (रूप $ ax $). एफ़िन लेकिन रैखिक नहीं: $ g $ (\omterm{prop:g9:linfunc:affinegraph}{ढलान} $3$,
अवरोधन $-5$) और $ m $ (स्थिर, \omterm{prop:g9:linfunc:affinegraph}{ढलान} $0$). कोई भी नहीं: $ h $, के कारण
वर्ग.
\end{solution}

\begin{solution}{exo:g9:linfunc:2}
$ f(0) = -3$; $ f(4) = 5$; $ f(-1) = -5$. और $ f(x) = 9$ मतलब
$2x - 3 = 9$, इसलिए $2x = 12$ और $ x = 6$.
\end{solution}

\begin{solution}{exo:g9:linfunc:3}
$ a = \frac{15}{6} = 2.5$, इसलिए $ f(x) = 2.5x $. तब $ f(10) = 25$ और
$ f(-2) = -5$.
\end{solution}

\begin{solution}{exo:g9:linfunc:4}
$ f $ के साथ मूल से होकर गुजरता है \omterm{prop:g9:linfunc:affinegraph}{ढलान} $2$; $ g $ वही है \omterm{prop:g9:linfunc:affinegraph}{ढलान} लेकिन
को पार कर जाता है $ y $-अक्ष पर $3$; $ h $ से घट जाती है $(0, 4)$ साथ \omterm{prop:g9:linfunc:affinegraph}{ढलान}
$-1$. के \omterm{def:g6:lines:objects}{रेखा}ंकन $ f $ और $ g $ हैं \emph{\omterm{def:g6:lines:perp}{समानांतर} रेखाएँ} (वही \omterm{prop:g9:linfunc:affinegraph}{ढलान},
अलग-अलग इंटरसेप्ट्स)।
\end{solution}

\begin{solution}{exo:g9:linfunc:5}
के लिए $ f $: \omterm{prop:g9:linfunc:affinegraph}{ढलान} $\frac{11 - 5}{3 - 1} = 3$, तब $5 = 3 \times 1 + b $
देता है $ b = 2$: $ f(x) = 3x + 2$.

के लिए $ g $: \omterm{prop:g9:linfunc:affinegraph}{ढलान} $\frac{0 - 4}{2 - 0} = -2$, और $ g(0) = 4$ पहले से ही है
अवरोधन: $ g(x) = -2x + 4$.
\end{solution}

\begin{solution}{exo:g9:linfunc:6}
की वृद्धि $25\%$: $120 \times 1.25 = 150$.

की छूट $40\%$: $65 \times 0.6 = 39$.

असली कीमत $ p $ साथ $ p \times 0.75 = 69$:
$ p = \frac{69}{0.75} = 92$.
\end{solution}

\begin{solution}{exo:g9:linfunc:7}
\emph{1.} $ f(x) = 0.05x + 10$.

\emph{2.} प्लान बी सस्ता है जब $25 < 0.05x + 10$, अर्थात\
$15 < 0.05x $, यानी\ $ x > 300$. से $301$ मिनट ($5$ घंटे) प्रति माह
आगे, प्लान बी जीतता है।
\end{solution}

\begin{solution}{exo:g9:linfunc:8}
\emph{1.} एक घंटे के बाद: $2000 \times 1.1 = 2200$. दो घंटे बाद:
$2200 \times 1.1 = 2420$.

\emph{2.} दूसरी वृद्धि लागू होती है $2200$, नहीं करने के लिए $2000$: बढ़ रहा है
द्वारा $10\%$ से दो बार गुणा किया जाता है $1.1^2 = 1.21$, ए $21\%$ वृद्धि, नहीं
$20\%$.
\end{solution}

\begin{solution}{exo:g9:linfunc:9}
\emph{1.} \omterm{prop:g9:linfunc:affinegraph}{ढलान}: $\frac{1 - 3}{6 - 2} = \frac{-2}{4} = -0.5$: द
कार्य कम हो रहा है.

\emph{2.} $ f(x) = -0.5x + b $ साथ $ f(2) = 3$: $-1 + b = 3$, इसलिए $ b = 4$
और $ f(x) = -0.5x + 4$. उत्पादन $0$: $-0.5x + 4 = 0$ देता है $ x = 8$.
\end{solution}

\begin{solution}{exo:g9:linfunc:10}
\emph{1.} दो परिवर्तन कीमत को कई गुना बढ़ा देते हैं
\[
\left(1 + \frac{t}{100}\right)\left(1 - \frac{t}{100}\right)
= 1 - \frac{t^2}{10000}
\]
(तीसरी पहचान के साथ $ a = 1$, $ b = \frac{t}{100}$).

\emph{2.} के लिए $0 < t \leq 100$, $\frac{t^2}{10000} > 0$, तो कारक
से कम है $1$: अंतिम कीमत मूल से कम है। के लिए
$ t = 10$: कारक $1 - \frac{100}{10000} = 0.99$, यानी\ एक समग्र
की कमी $1\%$.
\end{solution}

\begin{solution}{pb:g9:linfunc:1}
\textbf{1.} \omterm{prop:g9:linfunc:affinegraph}{ढलान}: $\frac{f(100) - f(0)}{100 - 0} =
\frac{212 - 32}{100} = 1.8$; अवरोधन $ f(0) = 32$. इस तरह
\[
f(C) = 1.8\,C + 32 .
\]

\textbf{2.} $ f(20) = 68\,^\circ $एफ (प्रसिद्ध हल्के दिन);
$ f(37) = 98.6\,^\circ $एफ; $ f(-10) = 14\,^\circ $एफ।

\textbf{3.} से $ F = 1.8C + 32$:
$ C = \frac{F - 32}{1.8}$. तब $50\,^\circ $ एफ
$\to \frac{18}{1.8} = 10\,^\circ $ सी, और $98.6\,^\circ $ एफ
$\to \frac{66.6}{1.8} = 37\,^\circ $ सी।

\textbf{4.} $f(0) = 32 \neq 0$, और $ f(20) = 68$ नहीं है
का दोगुना $ f(10) = 50$: रैखिक नहीं. ग्राफ एक \omterm{def:g6:lines:perp}{सीधा} है
पंक्ति में मूल बिन्दु गायब है: $ f $ है \emph{एफ़िन}, रैखिक नहीं.

\textbf{5.} $1.8x + 32 = x $ देता है $0.8x = -32$, इसलिए $ x = -40$:
चालीस से नीचे, दोनों थर्मा\omterm{def:g4:fractions:def}{मीटर} पढ़ते हैं $-40$. ग्राफ़ पर,
की पंक्ति $ f $ विकर्ण \omterm{def:g6:lines:objects}{रेखा} को पार करता है $ y = x $ बिल्कुल वहीं.

\textbf{6.} $ p_A(k) = 1.5k + 3$ और $ p_B(k) = 2k $. के लिए $4$ किमी:
$9$ यूरो के विरुद्ध $8$: बी सस्ता. के लिए $8$ किमी: $15$ ख़िलाफ़
$16$: एक सस्ता.

\textbf{7.} $1.5k + 3 = 2k $ देता है $ k = 6$: छह किलो\omterm{def:g4:fractions:def}{मीटर} पर
दोनों की लागत $12$ यूरो. ग्राफ़िक रूप से, B की \omterm{def:g6:lines:objects}{रेखा} (के माध्यम से)
मूल, तीव्र) A के नीचे से शुरू होता है और इसे पार करता है
$(6, 12)$: बी छोटी सवारी जीतता है, ए लंबी सवारी जीतता है।

\textbf{8.} $ N = 7 \times 20 - 30 = 110$ प्रति मिनट चहचहाता है। से
$82 = 7T - 30$: $ T = \frac{112}{7} = 16\,^\circ $सी।

\textbf{9.} $ v(t) = 600 - 15t $. लायक $240$:
$600 - 15t = 240$, $ t = 24$ महीने. लायक $0$: $ t = 40$ महीने.
आगे $40$ महीनों तक सूत्र नकारात्मक हो जाता है --- लेकिन एक फ़ोन का
मान पर रुक जाता है \omterm{def:g1:counting:zero}{शून्य}: प्रत्येक एफ़िन मॉडल में एक डोमेन होता है जहां यह होता है
समझ में आता है.

\textbf{10.} रचना: $\times 1.25$ तब $\times 0.8$
से गुणा करता है $1.25 \times 0.8 = 1$: कीमत बिल्कुल वापस आती है
इसकी शुरुआत के लिए. कोई चमत्कार नहीं: प्रतिशत अलग-अलग कार्य करते हैं
संदर्भ (कटौती बढ़ी हुई कीमत पर लागू होती है), और यहां
मल्टीप्लायर सामान्य कहानी को रद्द कर देते हैं, इसके साथ
\omterm{ex:g1:subtraction:difference}{अंतर}्निहित हानि $1 - \frac{t^2}{10\,000}$, है
\cref{exo:g9:linfunc:10}.

\textbf{11.} \omterm{prop:g9:linfunc:affinegraph}{ढलान}: $\frac{17 - 20}{30} = -0.1$ सेमी प्रति मिनट;
$ h(t) = 20 - 0.1t $. यह मर जाता है $ h(t) = 0$: $ t = 200$ मिनट
--- तीन घंटे और बीस मिनट की रोशनी।

\textbf{12.} दूसरा शुरू हुआ $10$ किमी आगे (अवरोधन); द
पहला तेज़ है ($24 > 18$ किमी/घंटा). पकड़ो:
$24t = 10 + 18t $ देता है $6t = 10$, $ t = \frac53$ एच $= 1$ एच $40$
मिनट, किलो\omterm{def:g4:fractions:def}{मीटर} पर $24 \times \frac53 = 40$.

\textbf{13.} $ f(x) + g(x) = (2 - 0.5)x + (3 + 7) = 1.5x + 10$:
एफ़िन, के साथ \omterm{prop:g9:linfunc:affinegraph}{ढलान}  \emph{\omterm{def:g1:addition:def}{जोड़} की \omterm{prop:g9:linfunc:affinegraph}{ढलान}ों} और अवरोधन
द \omterm{def:g1:addition:def}{जोड़} अंतः\omterm{def:g6:lines:objects}{खंड}ों का ---समान शब्दों को एकत्रित करने से स्पष्ट
में $(mx + p) + (m'x + p')$.

\textbf{14.} अगर $ m \neq 0$: बिल्कुल एक समाधान,
$ x = -\frac pm $ --- एक गैर-क्षैतिज \omterm{def:g6:lines:objects}{रेखा} अक्ष को पार करती है
एक बार. अगर $ m = 0$, $ p \neq 0$: कोई समाधान नहीं --- एक क्षैतिज \omterm{def:g6:lines:objects}{रेखा}
अक्ष के ऊपर या नीचे इसे कभी स्पर्श नहीं करता। अगर $ m = 0$, $ p = 0$:
प्रत्येक $ x $ एक समाधान है --- लाइन \emph{है} धुरी।

\textbf{15.} पहला त्रि: दरें $\frac{9 - 5}{3 - 1} = 2$ और
$\frac{17 - 9}{7 - 3} = 2$: बराबर, बिंदु संरेखित हैं (चालू)।
$ y = 2x + 3$). दूसरा त्रिक: $\frac{9-5}{2} = 2$ लेकिन
$\frac{16 - 9}{3} = \frac73 \neq 2$: संरेखित नहीं. एफ़िन
कार्य बिल्कुल वही हैं जिनमें परिवर्तन की निरंतर दर होती है ---
एक नंबर, \omterm{prop:g9:linfunc:affinegraph}{ढलान}, पूरी कहानी बताता है; घटता के लिए
दर स्वयं एक बिंदु से दूसरे बिंदु पर बदलती रहती है, और इसका पीछा करने से आगे बढ़ती है
हाई स्कूल वॉल्यूम में, व्युत्पन्न के लिए।
\end{solution}
