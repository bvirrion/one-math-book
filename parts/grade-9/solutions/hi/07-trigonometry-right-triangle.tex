\section*{अध्याय \ref{ch:g9:trig} --- समकोण त्रिभुज में त्रिकोणमिति}

\begin{solution}{exo:g9:trig:1}
कर्ण: $\sqrt{9^2 + 12^2} = \sqrt{81 + 144} = \sqrt{225} = 15$.

दूसरा पैर: $\sqrt{17^2 - 8^2} = \sqrt{289 - 64} = \sqrt{225} = 15$.
\end{solution}

\begin{solution}{exo:g9:trig:2}
$7^2 + 24^2 = 49 + 576 = 625 = 25^2$: हाँ, \omterm{def:g6:shapes:triangles}{समकोण का} (बातचीत
पाइथागोरस), के साथ \omterm{def:g3:shapes:rightangle}{समकोण} विपरीत पक्ष $25$.

$5^2 + 6^2 = 61 \neq 64 = 8^2$: नहीं \omterm{def:g6:shapes:triangles}{समकोण का}.
\end{solution}

\begin{solution}{exo:g9:trig:3}
कर्ण है $[EF]$ (विपरीत) \omterm{def:g3:shapes:rightangle}{समकोण} पर $ D $). ओर
के विपरीत $\theta = \widehat E $ है $[DF]$; बगल वाला पक्ष है
$[DE]$. इस तरह
\[
\cos\theta = \frac{DE}{EF}, \qquad
\sin\theta = \frac{DF}{EF}, \qquad
\tan\theta = \frac{DF}{DE}.
\]
\end{solution}

\begin{solution}{exo:g9:trig:4}
विपरीत पैर: $10 \sin 28^\circ \approx 10 \times 0.469 = 4.7$.
निकटवर्ती पैर: $10 \cos 28^\circ \approx 10 \times 0.883 = 8.8$.
(जाँच करना: $4.7^2 + 8.8^2 \approx 22 + 77 \approx 100$.)
\end{solution}

\begin{solution}{exo:g9:trig:5}
$\tan\theta = \dfrac{4.5}{6} = 0.75$, इसलिए
$\theta = \tan^{-1}(0.75) \approx 37^\circ $.
\end{solution}

\begin{solution}{exo:g9:trig:6}
में सही \omterm{def:g3:shapes:shapes}{त्रिकोण} पर्यवेक्षक द्वारा गठित, जमीन पर बिंदु
ड्रोन के नीचे, और ड्रोन: ऊंचाई $120$ मी पक्ष है
उन्नयन कोण के विपरीत, क्षैतिज दूरी $ d $ आसन्न है.
इसलिए
\[
\tan 32^\circ = \frac{120}{d},
\qquad
d = \frac{120}{\tan 32^\circ} \approx \frac{120}{0.625} = 192 \text{ m}.
\]
\end{solution}

\begin{solution}{exo:g9:trig:7}
चढ़ाव ($1.2$ m) कोण के विपरीत है; रैंप की लंबाई $ L $ है
कर्ण:
\[
\sin 5^\circ = \frac{1.2}{L},
\qquad
L = \frac{1.2}{\sin 5^\circ} \approx \frac{1.2}{0.0872} \approx 13.8
\text{ m}.
\]
रैम्प तो कम से कम होना ही चाहिए $13.8$ मी लंबा.
\end{solution}

\begin{solution}{exo:g9:trig:8}
से $\cos^2\theta + \sin^2\theta = 1$:
$\sin^2\theta = 1 - 0.36 = 0.64$, और $\sin\theta > 0$ एक तीव्र के लिए
कोण, तो $\sin\theta = 0.8$. तब
$\tan\theta = \dfrac{\sin\theta}{\cos\theta} = \dfrac{0.8}{0.6} =
\dfrac43$.
\end{solution}

\begin{solution}{exo:g9:trig:9}
\emph{सही \omterm{def:g6:shapes:triangles}{समद्विबाहु} त्रिभुज} पैरों के साथ $1$: कर्ण
$\sqrt{1 + 1} = \sqrt2$; प्रत्येक न्यून कोण है $45^\circ $, और
\[
\cos 45^\circ = \frac{1}{\sqrt2} = \frac{\sqrt2}{2} = \sin 45^\circ,
\qquad
\tan 45^\circ = \frac11 = 1 .
\]

\emph{समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण}} इस पक्ष की $1$: इसकी ऊंचाई इसे दो भागों में \omterm{def:g9:arith:divisor}{विभाजित} करती है
\omterm{def:g3:shapes:rightangle}{समकोण} त्रिभुज कर्ण के साथ $1$, आधार $\frac12$, और ऊंचाई
$ h = \sqrt{1 - \frac14} = \frac{\sqrt3}{2}$ (पाइथागोरस)। कोण हैं
$60^\circ $ (आधार पर) और $30^\circ $ (शीर्ष पर)। पढ़ रहा हूँ
अनुपात:
\[
\cos 60^\circ = \frac{1/2}{1} = \frac12, \quad
\sin 60^\circ = \frac{\sqrt3/2}{1} = \frac{\sqrt3}{2}, \quad
\cos 30^\circ = \frac{\sqrt3}{2}, \quad
\sin 30^\circ = \frac12 .
\]
\end{solution}

\begin{solution}{pb:g9:trig:1}
\textbf{1.} आँख के स्तर से ऊपर की ऊँचाई विपरीत दिशा है
$35^\circ $, आसन्न पक्ष के साथ $60$ एम:
$ h = 60 \tan 35^\circ \approx 60 \times 0.700 = 42$ एम।

\textbf{2.} $\tan\theta = 0.12$ देता है
$\theta = \tan^{-1}(0.12) \approx 6.8^\circ $ --- एक के लिए खड़ी
सड़क, फिर भी मामूली कोण। ए $45^\circ $ सड़क है
$\tan 45^\circ = 1$: ए $100\,\%$ श्रेणी। (``एक सौ
प्रतिशत'' ऊर्ध्वाधर से बहुत दूर है --- एक पसंदीदा स्की-लिफ्ट
ग़लतफ़हमी.)

\textbf{3.} में एक सही \omterm{def:g3:shapes:shapes}{त्रिकोण} तीव्र कोणों के साथ $\theta $ और
$90^\circ - \theta $, एक के विपरीत पक्ष आसन्न है
अन्य और कर्ण साझा है। इसलिए
$\sin(90^\circ - \theta) = \frac{\text{its opposite}}{h} =
\frac{\text{adjacent of }\theta}{h} = \cos\theta $, और
\omterm{def:g7:central:def}{सममित} रूप से $\cos(90^\circ - \theta) = \sin\theta $: द
\emph{सह}\omterm{def:g9:trig:ratios}{ज्या} है \omterm{def:g9:trig:ratios}{ज्या} की \emph{सह}कार्यान्वयन.

\textbf{4.} ऊंचाई: $8 \sin 60^\circ = 8 \times
\frac{\sqrt3}{2} = 4\sqrt3 \approx 6.93$ एम। पैर:
$8 \cos 60^\circ = 4$ बिल्कुल.

\textbf{5.} $0.574^2 + 0.819^2 = 0.329 + 0.671 = 1.000$ (को
\omterm{def:g6:decimals:rounding}{गोलाई}); और $\left(\frac12\right)^2 +
\left(\frac{\sqrt3}{2}\right)^2 = \frac14 + \frac34 = 1$. कोई भी
पहचान में विफल रहने वाले मूल्यों की \omterm{def:g1:addition:def}{जोड़}ी एक गलत रीडिंग को दर्शाती है या
गलत मोड में एक कैलकुलेटर --- एक निःशुल्क त्रुटि डिटेक्टर।

\textbf{6.} $\tan 40^\circ = \dfrac hx $ और
$\tan 30^\circ = \dfrac{h}{x + 200}$.

\textbf{7.} $ x = \dfrac{h}{\tan 40^\circ}$ और
$ x + 200 = \dfrac{h}{\tan 30^\circ}$; घटाना,
$200 = h\left(\frac{1}{\tan 30^\circ} -
\frac{1}{\tan 40^\circ}\right)$, सूत्र कहां से। संख्याएँ:
$\frac{1}{0.577} \approx 1.732$, $\frac{1}{0.839} \approx
1.192$, \omterm{ex:g1:subtraction:difference}{अंतर} $0.540$:
$ h \approx \frac{200}{0.540} \approx 370$ एम।

\textbf{8.} $ x = \frac{h}{\tan 40^\circ} \approx
\frac{370}{0.839} \approx 441$ एम। से जांचें $ A $:
$\frac{h}{x + 200} \approx \frac{370}{641} \approx 0.577 =
\tan 30^\circ $: सुसंगत।

\textbf{9.} एक-स्टेशन विधि के लिए क्षैतिज की आवश्यकता होती है
बिंदु से दूरी \emph{सीधे शिखर के नीचे} ---
किसी घाटी के पार या पहाड़ के अंदर मापे जाने योग्य नहीं। द
दो-स्टेशन विधि इसे उस दूरी से बदल देती है जिसे आप तेजी से तय कर सकते हैं
अपने आप: $200$ मैं आपके अपने दो दृष्टिकोणों के बीच हूं।

\textbf{10.} $ h = \dfrac{200}{\sqrt3 - \frac{1}{\sqrt3}}
= \dfrac{200}{\frac{2}{\sqrt3}} = 100\sqrt3 \approx 173$ मी ---
एफिल टॉवर की दूसरी मंजिल, दो बार देखी गई
और एक सैर.

\textbf{11.} दृष्टि \omterm{def:g6:lines:objects}{रेखा} $ d $, त्रि\omterm{def:g9:trig:ratios}{ज्या} $ R $ (चराई बिंदु तक,
\omterm{def:g6:lines:perp}{सीधा} दृष्टि \omterm{def:g6:lines:objects}{रेखा} तक) और $ R + h $ (आँख से):
पाइथागोरस देता है $ d^2 + R^2 = (R + h)^2 = R^2 + 2Rh + h^2$, इसलिए
$ d^2 = 2Rh + h^2 \approx 2Rh $ ($ h $ के विरूद्ध बहुत कम है $ R $).
निगाहें पर $1.80$ एम:
$ d \approx \sqrt{2 \times 6.37 \times 10^6 \times 1.8}
\approx 4\,800$ मी---समुद्र का क्षितिज बमुश्किल पाँच है
किलो\omterm{def:g4:fractions:def}{मीटर} दूर.

\textbf{12.} $ d \approx \sqrt{2 \times 6.37 \times 10^6 \times
324} \approx 64$ किमी. नाविकों का नियम: $3.6\sqrt{1.8} \approx
4.8$ किमी और $3.6\sqrt{324} = 64.8$ किमी --- दोनों मेल खाते हैं।

\textbf{13.} $\tan\theta = \frac{2}{60}$ देता है $\theta \approx
1.9^\circ $. चंद्रमा का $0.5^\circ $ एक के बारे में है \emph{\omterm{def:g3:division:half}{तिमाही}} का
अंगूठा: चंद्रमा विशाल दिखता है और नाखून से छिपा होता है
--- आँख एक ख़राब चांदा है।

\textbf{14.} सीढ़ियाँ: $\tan^{-1}\frac{17}{29} \approx 30^\circ $
--- आरामदायक सीढ़ियाँ लगभग बिल्कुल ठीक ऊपर चढ़ती हैं
$30^\circ $ वि\omterm{def:g3:division:remainder}{शेष} मूल्यों का. पर $60^\circ $ एक पर $29$ सेमी
चलना, ऊपर उठना होगा $29\tan 60^\circ = 29\sqrt3 \approx
50$ सेमी: एक सीढ़ी, सीढ़ी नहीं.

\textbf{15.} $\tan 10^\circ \approx 0.18$;
$\tan 45^\circ = 1$; $\tan 80^\circ \approx 5.7$;
$\tan 89^\circ \approx 57.3$. जैसा $\theta \to 90^\circ $ द
आसन्न पक्ष सिकुड़कर \omterm{def:g1:counting:zero}{शून्य} हो जाता है जबकि विपरीत पक्ष धारण करता है:
द \omterm{def:g3:division:remainder}{भागफल} हर सीमा से परे बढ़ता है. पर $90^\circ$ बिलकुल
कोई त्रिभुज नहीं है और कोई मान नहीं है --- का ग्राफ़ \omterm{def:g9:trig:ratios}{स्पर्शरेखा}
हाई स्कूल में सबसे पहले, एक ऊर्ध्वाधर अनंतस्पर्शी पर चढ़ता है
\omterm{def:g6:measure:volume}{आयतन}.
\end{solution}
