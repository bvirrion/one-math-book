\section*{अध्याय \ref{ch:g9:thales} --- थेल्स प्रमेय}

\begin{solution}{exo:g9:thales:1}
थेल्स: $\dfrac{AM}{AB} = \dfrac{AN}{AC} = \dfrac{MN}{BC}$, अर्थात\
$\dfrac26 = \dfrac13$. इसलिए $\dfrac{3}{AC} = \dfrac13$ देता है $ AC = 9$,
और $\dfrac{MN}{9} = \dfrac13$ देता है $ MN = 3$.
\end{solution}

\begin{solution}{exo:g9:thales:2}
सबसे पहले खोजें $ AB = AM + MB = 5 + 3 = 8$. तब
$\dfrac{AM}{AB} = \dfrac{AN}{AC}$ देता है $\dfrac58 = \dfrac{4}{AC}$, इसलिए
$5 \times AC = 32$ और $ AC = 6.4$.
\end{solution}

\begin{solution}{exo:g9:thales:3}
तितली बिल्कुल नेस्टेड कॉन्फ़िगरेशन की तरह काम करती है:
$\dfrac{AM}{AB} = \dfrac{AN}{AC} = \dfrac{MN}{BC}$, अर्थात\
$\dfrac{3}{7.5} = \dfrac25$. इसलिए $\dfrac{2}{AC} = \dfrac25$ देता है
$ AC = 5$, और $\dfrac{2.6}{BC} = \dfrac25$ देता है
$ BC = \dfrac{2.6 \times 5}{2} = 6.5$.
\end{solution}

\begin{solution}{exo:g9:thales:4}
$\dfrac{AM}{AB} = \dfrac68 = \dfrac34$ और
$\dfrac{AN}{AC} = \dfrac{9}{12} = \dfrac34$: समान अनुपात, अंक
वही क्रम, तो $(MN) \parallel (BC)$ थेल्स की बातचीत से.
\end{solution}

\begin{solution}{exo:g9:thales:5}
$\dfrac{AM}{AB} = \dfrac46 = \dfrac23$ और
$\dfrac{AN}{AC} = \dfrac58$. क्रॉस चेक: $\dfrac23 = \dfrac{16}{24}$
और $\dfrac58 = \dfrac{15}{24}$: अनुपात अलग-अलग हैं। यदि पंक्तियाँ होतीं
\omterm{def:g6:lines:perp}{समानांतर}, थेल्स का प्रमेय उन्हें बराबर होने के लिए मजबूर करेगा --- इसलिए $(MN)$
और $(BC)$ हैं \emph{नहीं} \omterm{def:g6:lines:perp}{समानांतर}.
\end{solution}

\begin{solution}{exo:g9:thales:6}
होने देना $ S $ छाया का सिरा बनो, $ P $ व्यक्ति के सिर के ऊपर, $ L $
दीपक का शीर्ष. व्यक्ति ($1.8$ की दूरी पर मी $3$ मैं वहां से हूँ $ S $) और
दीपक (ऊंचाई) $ h $ दूरी पर $3 + 2 = 5$ मैं वहां से हूँ $ S $) दो हैं
\omterm{def:g6:lines:perp}{समानांतर} खड़ा \omterm{def:g6:lines:objects}{खंड}ों प्रकाश किरण द्वारा काटा गया $(SL)$: थेल्स से
बिंदु $ S $ देता है
\[
\frac{1.8}{h} = \frac{3}{5},
\qquad\text{so}\quad
h = \frac{1.8 \times 5}{3} = 3 \text{ m}.
\]
\end{solution}

\begin{solution}{exo:g9:thales:7}
\emph{1.} वास्तविक दूरी: $6.8 \times 25\,000 = 170\,000$ सेमी $=
1.7$ किमी.

\emph{2.} \omterm{def:g6:measure:area}{क्षेत्र}ों द्वारा स्केल $ k^2$: वास्तविक \omterm{def:g6:measure:area}{क्षेत्र} $= 8 \times 25\,000^2$
सेमी $^2$ $= 8 \times 6.25 \times 10^8$ सेमी $^2$ $= 5 \times 10^9$ सेमी $^2$.
तब से $1$ किमी $^2 = 10^{10}$ सेमी $^2$, वह है $0.5$ किमी $^2$.
\end{solution}

\begin{solution}{exo:g9:thales:8}
\emph{1.} अनुपात है $\dfrac{AM}{AB} = \dfrac{8}{12} = \dfrac23$.
थेल्स: $ AN = \frac23 \times 15 = 10$ और $ MN = \frac23 \times 18 = 12$.

\emph{2.} \omterm{def:g7:prop:scale}{पैमाना} कारक है $ k = \frac23$. \omterm{def:g6:measure:perimeter}{परिधि}: $ ABC $ है
$12 + 15 + 18 = 45$, और $ AMN $ है $8 + 10 + 12 = 30 = \frac23 \times
45$:\omterm{def:g6:measure:perimeter}{परिधि} मापती है $ k $ भी, हर लम्बाई की तरह।
\end{solution}

\begin{solution}{exo:g9:thales:9}
आस-पास $ O $, पंक्तियाँ $(AC)$ और $(BD)$ क्रॉस, और
$(AB) \parallel (CD)$: \omterm{def:g3:shapes:shapes}{त्रिकोण}ों के साथ तितली विन्यास $ OAB $
और $ OCD $. थेल्स देता है
\[
\frac{OA}{OC} = \frac{OB}{OD} = \frac{AB}{CD} = \frac46 = \frac23 .
\]
इसलिए $\dfrac{OA}{OC} = \dfrac23$, और समानता
$\dfrac{OA}{OC} = \dfrac{OB}{OD}$ बिलकुल यही कहता है $ O $ दोनों को काट देता है
विकर्ण समान अनुपात में।
\end{solution}

\begin{solution}{pb:g9:thales:1}
\textbf{1.} निर्माण पर दो बिंदु बनते हैं $[AB]$;
उनको बुलाएं $ M_1$ (से $ P_1$) और $ M_2$ (से $ P_2$).

\textbf{2.} \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ A P_3 B $, के \omterm{def:g6:lines:perp}{समानांतर}
$(P_3 B)$ के माध्यम से $ P_1$ और $ P_2$ \omterm{def:g5:solids:def}{किनारों} को काटें
\omterm{def:g9:linfunc:linear}{आनुपातिक} रूप से (\cref{thm:g9:thales:direct}):
\[
\frac{A M_1}{AB} = \frac{A P_1}{A P_3} = \frac13,
\qquad
\frac{A M_2}{AB} = \frac{A P_2}{A P_3} = \frac23 .
\]
कम्पास बनाया $ A P_1$, $ P_1 P_2$, $ P_2 P_3$ बराबर, तो
अनुपात बिल्कुल तिहाई हैं --- किरण का कोण कुछ भी हो
कम्पास खोलने का चयन किया गया।

\textbf{3.} पांचवें के लिए: पांच समान लंबाई को हटाएं, जुड़ें
पाँचवाँ बिंदु $ B $, और चार समानताएं बनाएं। के लिए $\frac35$ एक का
\omterm{def:g6:lines:objects}{खंड}: समान आकृति, और \omterm{def:g6:lines:perp}{समानांतर} द्वारा काटे गए बिंदु को लें
के माध्यम से $ P_3$: यह पर बैठता है $\frac{3}{5}$ का $[AB]$ से $ A $.

\textbf{4.} $\frac{AM}{MB} = \frac23$ मतलब
$ AM = \frac25 AB $: किरण पर पांच समान चरण, जुड़ें $ P_5$ को
$ B $, और \omterm{def:g6:lines:perp}{समानांतर} के माध्यम से $ P_2$ निशान $ M $.

\textbf{5.} मापने $\sqrt2 = 1.41421\ldots $ सेमी केवल कभी उपयोग करता है
परिमित रूप से अनेक दशमलव: कोई भी मापा गया ``तीसरा'' एक है
सन्निकटन. थेल्स निर्माण कभी भी एक संख्या नहीं पढ़ता: यह
सटीक बिंदु बताता है $\frac{\sqrt2}{3}$ --- बराबर भाग
एक ऐसे \omterm{def:g6:lines:objects}{खंड} की जिसे कोई शासक व्यक्त भी नहीं कर सकता
(\cref{pb:g9:sqrt:1}).

\textbf{6.} वस्तु का शीर्ष, छेद और छवि का निचला भाग
संरेखित हैं, और इसी प्रकार वस्तु के निचले भाग और के लिए भी
छवि का शीर्ष: वस्तु के साथ छेद पर दो रेखाएँ प्रतिच्छेद करती हैं
और छवि दीवार \omterm{def:g6:lines:perp}{समानांतर}। तितली में थेल्स:
\[
\frac{h}{H} = \frac{d}{D},
\qquad\text{so}\qquad
h = H \times \frac dD .
\]

\textbf{7.} $ h = 12 \times \frac{0.20}{30} = 0.08$ एम $= 8$ सेमी
--- उल्टा (किरणें छेद पर पार करती हैं)।

\textbf{8.} $ H = h \times \frac Dd = 9 \times 10^{-3} \times
\frac{1.5 \times 10^{11}}{1} = 1.35 \times 10^{9}$ एम। सच है
मूल्य $1.39 \times 10^9$ मी: एक जूते का \omterm{def:g5:solids:def}{डिब्बा} सूर्य को मापता है
कुछ प्रतिशत के भीतर.

\textbf{9.} चंद्रमा: $\frac{3.5 \times 10^6}{3.8 \times 10^8}
\approx 0.0092$. सूरज: $\frac{1.39 \times 10^9}{1.5 \times
10^{11}} \approx 0.0093$. दो अनुपात --- स्पष्ट आकार
आकाश में --- लगभग समान हैं: चंद्रमा सूर्य को ढक सकता है
\emph{बिल्कुल}, रिम से रिम तक। वह ब्रह्मांडीय अस्थायी परिवर्तन संपूर्ण है
सूर्य ग्रहण.

\textbf{10.} प्रत्येक किरण को एक छेद से गुजरना होगा, इसलिए किरणें
वस्तु के शीर्ष से जारी रखें \emph{नीचे} और किरणें से
नीचे जारी है \emph{ऊपर}: छवि उलटी है. बड़ा करना
छेद थोड़ी सी स्थानांतरित प्रतियों के पूरे बंडल के माध्यम से गुजरता है
छवि का, जो ओवरलैप होता है: उज्जवल, लेकिन धुंधला।

\textbf{11.} $\frac{PQ}{MN} = \frac{OP}{OM} = \frac{7.5}{3}
= 2.5$, इसलिए $ PQ = 10$; और $\frac{ON}{OQ} = \frac{OM}{OP} =
\frac{1}{2.5}$, इसलिए $ ON = \frac{6}{2.5} = 2.4$.

\textbf{12.} यदि समांतर किसी एक के मध्यबिंदु से होकर गुजरता है
पक्ष, थेल्स अनुपात है $\frac12$, तो यह दूसरी तरफ काटता है
पर \emph{इसका} मध्यबिंदु, और \omterm{def:g6:lines:perp}{समानांतर} \omterm{def:g6:lines:objects}{खंड} \omterm{def:g3:division:half}{आधा} मापता है
आधार: बिल्कुल \cref{thm:g8:midpoints:direct}. थेल्स है
मध्यबिंदु प्रमेय अनुपात से मुक्त हो गया $\frac12$.

\textbf{13.} \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ ACD $, \omterm{def:g6:lines:objects}{रेखा} $(EO)$ \omterm{def:g6:lines:perp}{समानांतर} है
आधार तक $(CD)$, साथ $\frac{AO}{AC} = \frac{OA}{OA + OC} =
\frac{4}{4 + 6} = \frac25$. थेल्स:
$\frac{EO}{DC} = \frac{AO}{AC} = \frac25$, इसलिए
$ EO = \frac25 \times 6 = 2.4$.

\textbf{14.} \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ BDC $, वैसे ही
$\frac{BO}{BD} = \frac25$ और $ OF = \frac25 \times 6 = 2.4$.
इस तरह
\[
EF = EO + OF = 4.8
= \frac{2 \times 4 \times 6}{4 + 6} :
\]
आधारों का हार्मोनिक माध्य --- \omterm{def:g4:numbers:round}{गोल} यात्राओं का माध्य
(\cref{pb:g8:speed:1}), एक समलम्ब \omterm{def:g7:central:parallelogram}{चतुर्भुज} में खींचा गया।

\textbf{15.} अंकगणित: $5$; ज्यामितीय:
$\sqrt{24} \approx 4.90$; हार्मोनिक: $4.8$. आदेश देना:
$4.8 < 4.90\ldots < 5$, यानी\ लयबद्ध $<$ ज्यामितिक $<$
अंकगणित --- केवल समान आधारों के लिए समानता के साथ। में
चित्र: मध्य \omterm{def:g6:lines:objects}{रेखा} माप $5$, के माध्यम से \omterm{def:g6:lines:perp}{समानांतर}
विकर्ण क्रॉसिंग $4.8$, और समानता में कटौती
$\sqrt{24}$, सभी को उसी क्रम में आधारों के बीच रखा गया है। द
में असमानताएँ सिद्ध हुईं \cref{pb:g8:cosine:1} (ज्यामितीय
$\leq $ अंकगणित, वृत्त द्वारा और पहचान द्वारा) और
\cref{pb:g8:speed:1} (हार्मोनिक $\leq $ अंकगणित); लयबद्ध
$\leq $ ज्यामितीय उन्हें संयोजित करके अनुसरण करता है ($ H \times A = G^2$,
वहां भी साबित हुआ)।
\end{solution}
