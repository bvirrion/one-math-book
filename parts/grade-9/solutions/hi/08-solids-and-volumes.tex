\omterm{def:g7:areas:prism}{सिलेंडर}: $ V = \pi r^2 h = \pi \times 9 \times 7 = 63\pi \approx 198$.
\end{solution}

\begin{solution}{exo:g9:solids:2}
\omterm{def:g8:solids:def}{कोन}: $ V = \frac13 \pi r^2 h = \frac13 \pi \times 25 \times 12 =
100\pi \approx 314$.

\omterm{def:g8:solids:def}{पिरामिड}: आधार \omterm{def:g6:measure:area}{क्षेत्र} $ B = 8 \times 3 = 24$, इसलिए
$ V = \frac13 \times 24 \times 10 = 80$.
\end{solution}

\begin{solution}{exo:g9:solids:3}
\omterm{def:g6:measure:volume}{आयतन}: $ V = \frac43 \pi \times 6^3 = \frac43 \pi \times 216 = 288\pi $.

सतह \omterm{def:g6:measure:area}{क्षेत्र}: $4\pi \times 6^2 = 144\pi $.
\end{solution}

\begin{solution}{exo:g9:solids:4}
\omterm{def:g3:shapes:shapes}{RADIUS} अनुभाग का: $\sqrt{r^2 - d^2} = \sqrt{100 - 64} = \sqrt{36}
= 6$.
\end{solution}

\begin{solution}{exo:g9:solids:5}
\omterm{def:g9:thales:scaling}{पैमाने का कारक} संपूर्ण से \omterm{def:g8:solids:def}{कोन} छोटे को:
$ k = \frac{9}{12} = \frac34$. अनुभाग \omterm{def:g3:shapes:shapes}{RADIUS}: $8 \times \frac34 = 6$.

\omterm{def:g6:measure:volume}{आयतन} संपूर्ण का \omterm{def:g8:solids:def}{कोन}: $\frac13\pi \times 64 \times 12 = 256\pi $.
\omterm{def:g6:measure:volume}{आयतन} छोटे का \omterm{def:g8:solids:def}{कोन}: $256\pi \times \left(\frac34\right)^3 =
256\pi \times \frac{27}{64} = 108\pi \approx 339$.
\end{solution}

\begin{solution}{exo:g9:solids:6}
\omterm{def:g6:measure:volume}{आयतन} गेंद का: $\frac43\pi \times 2^3 = \frac{32\pi}{3}$ सेमी $^3$.
यह \omterm{def:g6:measure:volume}{आयतन} आधार पर फैलता है \omterm{def:g6:measure:area}{क्षेत्र} $\pi \times 3^2 = 9\pi $ सेमी$^2$,
तो स्तर बढ़ जाता है
\[
\frac{32\pi/3}{9\pi} = \frac{32}{27} \approx 1.2 \text{ cm}
\]
(के बारे में $12$ मिमी).
\end{solution}

\begin{solution}{exo:g9:solids:7}
छोटा साँचा बड़े साँचे की कमी है $ k = \frac36 =
\frac12$, तो यह है \omterm{def:g6:measure:volume}{आयतन} है $\left(\frac12\right)^3 = \frac18$ का
बड़ा वाला: नुस्खा भरता है $8$ छोटे साँचे.
\end{solution}

\begin{solution}{exo:g9:solids:8}
$ V = \frac13 \times 230^2 \times 147 = \frac13 \times 52\,900 \times 147
= 52\,900 \times 49 = 2\,592\,100$ एम $^3$ --- के बारे में $2.6$ मिलियन घन
\omterm{def:g4:fractions:def}{मीटर}.
\end{solution}

\begin{solution}{exo:g9:solids:9}
\emph{1.} पानी बनता है a \omterm{def:g8:solids:def}{कोन} द्वारा स्केल किया गया $ k = \frac12$ (शीर्ष नीचे,
\omterm{def:g3:division:half}{आधा} ऊँचाई), तो यह है \omterm{def:g6:measure:volume}{आयतन} है $\left(\frac12\right)^3 = \frac18$
फ़नल का: एक आठवाँ, से बहुत कम \omterm{def:g3:division:half}{आधा}!

\emph{2.} ऊंचाई तक पानी $ d $ फॉर्म ए \omterm{def:g8:solids:def}{कोन} द्वारा स्केल किया गया
$ k = \frac{d}{12}$, का \omterm{def:g6:measure:volume}{आयतन} $ k^3$ फ़नल का समय. \omterm{def:g3:division:half}{आधा}  \omterm{def:g6:measure:volume}{आयतन}
मतलब $ k^3 = \frac12$, अर्थात\
$\left(\frac{d}{12}\right)^3 = \frac12$. तब
$\frac{d}{12} = \sqrt[3]{0.5} \approx 0.7937$, इसलिए
$ d \approx 9.5$ सेमी: दूसरा \omterm{def:g3:division:half}{आधा} की \omterm{def:g6:measure:volume}{आयतन} केवल शीर्ष पर है
$2.5$ सेमी---का चौड़ा भाग \omterm{def:g8:solids:def}{कोन} अधिकांश जल धारण करता है।
\end{solution}

\begin{solution}{pb:g9:solids:1}
\textbf{1.} $ V_{\text{cyl}} = \pi r^2 \times 2r = 2\pi r^3$.

\textbf{2.} $\dfrac{V_{\text{sphere}}}{V_{\text{cyl}}}
= \dfrac{\frac43 \pi r^3}{2 \pi r^3} = \dfrac23$. $\pi $ और यह $ r^3$ रद्द करें: अनुपात है \emph{सार्वभौमिक} --- सच
एक संगमरमर के लिए और एक ग्रह के लिए। आकृतियों के बीच कोई संबंध, नहीं
संख्याओं के बीच: बिल्कुल उसी प्रकार का सत्य जो उकेरने लायक है
पत्थर.

\textbf{3.} \omterm{def:g4:numbers:round}{गोल}ा: $4\pi r^2$. \omterm{def:g7:areas:prism}{सिलेंडर}: पार्श्व
$2\pi r \times 2r = 4\pi r^2$, प्लस दो ढक्कन
$2 \times \pi r^2$: कुल $6\pi r^2$. अनुपात:
$\frac{4\pi r^2}{6\pi r^2} = \frac23$ --- वही दो तिहाई,
सतहों के लिए. (और एक बोनस: \omterm{def:g6:measure:area}{क्षेत्र} का \omterm{def:g6:measure:area}{क्षेत्र} बिल्कुल बराबर
द सिलेंडरका पार्श्व \omterm{def:g6:measure:area}{क्षेत्र}.)

\textbf{4.} बचा खुचा: $2\pi r^3 - \frac43\pi r^3 =
\frac23 \pi r^3$. दो \omterm{def:g8:solids:def}{कोन} का \omterm{def:g3:shapes:shapes}{RADIUS} $ r $ और ऊंचाई $ r $:
$2 \times \frac13 \pi r^2 \times r = \frac23 \pi r^3$. बराबर।

\textbf{5.} गेंद: $\frac43 \pi \times 12^3 \approx 7\,200$
सेमी $^3$; \omterm{def:g5:solids:def}{डिब्बा}: $2\pi \times 12^3 \approx 10\,900$ सेमी $^3$. खाली:
का एक तिहाई \omterm{def:g5:solids:def}{डिब्बा}, के बारे में $33\,\%$ --- द्वारा गारंटी दी गई
प्रश्न 2, गेंद का आकार कुछ भी हो।

\textbf{6.} \omterm{def:g3:division:half}{आधा} \omterm{def:g4:numbers:round}{गोल}ा:
$\frac12 \times \frac43 \pi \times 10^3 = \frac23 \pi \times
1\,000 \approx 2\,094$ सेमी $^3 \approx 2.1$ एल

\textbf{7.} \omterm{def:g8:solids:def}{कोन}: $\frac13 \pi \times 10^2 \times 10 =
\frac{1\,000\pi}{3} \approx 1\,047$ सेमी $^3$. \omterm{def:g4:numbers:round}{गोल}ार्ध:
$\frac{2\,000\pi}{3} \approx 2\,094$ सेमी $^3$. \omterm{def:g7:areas:prism}{सिलेंडर}:
$1\,000\pi \approx 3\,142$ सेमी $^3$. अनुपात:
$\frac{1000\pi}{3} : \frac{2000\pi}{3} : \frac{3000\pi}{3}
= 1 : 2 : 3$ बिल्कुल।

\textbf{8.} $\frac43 \pi \times 27 = \pi \times 9 \times h $
देता है $ h = \frac43 \times 3 = 4$ बिल्कुल सेमी.

\textbf{9.} त्रि\omterm{def:g9:trig:ratios}{ज्या}: $\frac{6\,371}{1\,737} \approx 3.67$.
संस्करणों \omterm{def:g7:prop:scale}{पैमाना} के रूप में \omterm{def:g5:solids:def}{घनक्षेत्र} (\cref{thm:g9:solids:scaling}):
$3.67^3 \approx 49$. लगभग पचास चंद्रमा पृथ्वी में समाते हैं।

\textbf{10.} सतह \omterm{def:g7:prop:scale}{पैमाना} वर्ग के रूप में: $3.67^2 \approx
13.5$. एक गुब्बारे को दोगुना करना \omterm{def:g3:shapes:shapes}{RADIUS} इसे गुणा करता है \omterm{def:g6:measure:volume}{आयतन} द्वारा
$2^3 = 8$ लेकिन इसकी सतह केवल द्वारा $2^2 = 4$: जैसे-जैसे चीजें बढ़ती हैं,
\omterm{def:g6:measure:volume}{आयतन} (वजन, सामग्री, उत्पन्न गर्मी) सतह (त्वचा,
सामग्री, ऊष्मा नष्ट) --- वर्ग-घन नियम।

\textbf{11.} वज़न: $\times 10^3 = 1\,000$. हड्डी
\omterm{prop:g9:solids:sections}{क्रॉस सेक्शन}: $\times 10^2 = 100$. दबाव---वजन प्रति
\omterm{def:g6:measure:area}{क्षेत्र} हड्डी का: $\times \frac{1000}{100} = 10$. हड्डियों के लिए बनाया गया
मानव दबाव दस गुना अधिक प्राप्त करता है: विशाल का कंकाल
अपने ही वजन के नीचे टूट जाता है। दिग्गज ज्यामितीय रूप से हैं
असंभव; बड़े जानवरों को अनुपातहीन रूप से मोटी हड्डियों की आवश्यकता होती है
(हाथी के पैरों की तुलना चिकारे के पैरों से करें)।

\textbf{12.} वज़न: $\div 100^3 = 10^6$. ताकत (मांसपेशी
\omterm{prop:g9:solids:sections}{क्रॉस सेक्शन}): $\div 100^2 = 10^4$. ताकत-से-वजन:
$\times \frac{10^6}{10^4} = 100$. चींटी पचास बार उठाती है
इसका वजन सुपर-एथलीट नहीं है --- यह केवल छोटा है; ए
चींटी के आकार का छोटा मनुष्य भी ऐसा ही कर सकता है।

\textbf{13.} $\dfrac{S}{V} = \dfrac{4\pi r^2}{\frac43\pi r^3}
= \dfrac{3}{r}$: के लिए $ r = 1$ अनुपात है $3$, के लिए $ r = 2$ यह है $1.5$ ---आकार \omterm{def:g3:division:half}{आधा} करने से उपलब्ध सतह दोगुनी हो जाती है
गर्मी पैदा करने वाली प्रति इकाई \omterm{def:g6:measure:volume}{आयतन}. छोटे शरीर गर्मी से खून बहाते हैं:
चूहे को लगातार खाना चाहिए, भालू हाइबरनेट कर सकता है, और
बच्चा---छोटा \omterm{def:g4:numbers:round}{गोल}ा, बड़ा $\frac Sr $ --- टोपी की जरूरत है.

\textbf{14.} $\left(\frac54\right)^3 = \frac{125}{64} \approx
1.95$: बड़ा नारंगी लगभग धारण करता है \emph{दो बार} के लिए फल
वही कीमत. खरीदना \omterm{def:g3:shapes:shapes}{RADIUS}.

\textbf{15.} नक्काशी कहती है: \omterm{def:g4:numbers:round}{गोल}ा $= \frac23$ की
\omterm{def:g7:areas:prism}{सिलेंडर} में \omterm{def:g6:measure:volume}{आयतन}, \emph{और} $\frac23$ इसका कुल मिलाकर
सतह --- जोड़ने वाले दो सटीक, सार्वभौमिक अनुपात
सबसे \omterm{def:g4:numbers:round}{गोल} \omterm{def:g5:solids:def}{ठोस} सबसे सरल को. सिसरो का उत्तर: क्योंकि ए
प्रमेय एकमात्र ऐसा स्मारक है जिसके पास न तो सेनाएं हैं और न ही सदियां
इरोड --- आर्किमिडीज़ की मशीनें सिरैक्यूज़ के साथ जल गईं, लेकिन
दो तिहाई अभी भी ठीक दो तिहाई ही हैं।
\end{solution}
