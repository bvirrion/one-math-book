\section*{अध्याय \ref{ch:g9:fractions} --- भिन्न और घात}

\begin{solution}{exo:g9:fractions:1}
$\dfrac13 + \dfrac15 = \dfrac{5}{15} + \dfrac{3}{15} = \dfrac{8}{15}$.

$\dfrac72 - \dfrac53 = \dfrac{21}{6} - \dfrac{10}{6} = \dfrac{11}{6}$.

$\dfrac{5}{12} + \dfrac38 = \dfrac{10}{24} + \dfrac{9}{24} =
\dfrac{19}{24}$.

$3 - \dfrac45 = \dfrac{15}{5} - \dfrac45 = \dfrac{11}{5}$.
\end{solution}

\begin{solution}{exo:g9:fractions:2}
$\dfrac59 \times \dfrac{27}{35} = \dfrac{5 \times 27}{9 \times 35}
= \dfrac{27}{9} \times \dfrac{5}{35} = 3 \times \dfrac17 = \dfrac37$
(द्वारा सरल करें $9$ और तक $5$).

$\dfrac{8}{15} \div \dfrac45 = \dfrac{8}{15} \times \dfrac54
= \dfrac{8 \times 5}{15 \times 4} = \dfrac{40}{60} = \dfrac23$.

$\dfrac23 \times \dfrac34 \times \dfrac45 = \dfrac{2 \times 3 \times 4}
{3 \times 4 \times 5} = \dfrac25$ (द $3$ रेत $4$ रद्द करें)।
\end{solution}

\begin{solution}{exo:g9:fractions:3}
$2^5 = 32$; \quad $(-2)^4 = 16$; \quad $-2^4 = -16$; \quad
$4^{-2} = \frac{1}{16}$; \quad
$\left(\frac23\right)^3 = \frac{8}{27}$; \quad $7^0 = 1$.
\end{solution}

\begin{solution}{exo:g9:fractions:4}
$5^3 \times 5^6 = 5^9$; \quad $\dfrac{7^9}{7^5} = 7^4$; \quad
$\left(2^3\right)^4 = 2^{12}$; \quad
$\dfrac{3^2 \times 3^7}{3^5} = 3^{9-5} = 3^4$; \quad
$2^5 \times 4 = 2^5 \times 2^2 = 2^7$.
\end{solution}

\begin{solution}{exo:g9:fractions:5}
$45\,000 = 4.5 \times 10^4$; \quad
$0.0072 = 7.2 \times 10^{-3}$; \quad
$302.5 = 3.025 \times 10^2$; \quad
$0.000\,000\,9 = 9 \times 10^{-7}$; \quad
बारह करोड़ $= 1.2 \times 10^7$.
\end{solution}

\begin{solution}{exo:g9:fractions:6}
$(2 \times 10^7) \times (4.5 \times 10^{-3}) = 9 \times 10^{4}$.

$\dfrac{9 \times 10^5}{3 \times 10^{-2}} = 3 \times 10^{5 - (-2)}
= 3 \times 10^7$.

$(5 \times 10^3)^2 = 25 \times 10^6 = 2.5 \times 10^7$.
\end{solution}

\begin{solution}{exo:g9:fractions:7}
\omterm{def:g1:addition:def}{जोड़}ा गया पानी दर्शाता है
$\dfrac34 - \dfrac35 = \dfrac{15}{20} - \dfrac{12}{20} = \dfrac{3}{20}$
टैंक का. इसलिए $\frac{3}{20}$ की \omterm{def:g2:measure:units}{क्षमता} है $30$ लीटर:
$\frac{1}{20}$ है $10$ लीटर, और \omterm{def:g2:measure:units}{क्षमता} है $200$ लीटर.
\end{solution}

\begin{solution}{exo:g9:fractions:8}
ऐलिस के बाद: $1 - \frac14 = \frac34$ केक का हिस्सा बचा हुआ है. बेन खाता है
$\frac13 \times \frac34 = \frac14$, जा रहा हूँ
$\frac34 - \frac14 = \frac12$. कार्ला खाती है \omterm{def:g3:division:half}{आधा} उसमें से, $\frac14$,
जा रहा हूँ $\frac14$ केक का.
\end{solution}

\begin{solution}{exo:g9:fractions:9}
समय $= \dfrac{\text{distance}}{\text{speed}} =
\dfrac{1.5 \times 10^{11}}{3 \times 10^8} = 0.5 \times 10^3 = 500$
सेकंड, यानी\ $8$ मिनट और $20$ सेकंड.
\end{solution}

\begin{solution}{exo:g9:fractions:10}
दोनों संख्याओं को इस प्रकार लिखें पॉवर्स साथ \omterm{def:g8:powers:def}{प्रतिपादक} $10$:
\[
2^{30} = \left(2^3\right)^{10} = 8^{10},
\qquad
3^{20} = \left(3^2\right)^{10} = 9^{10}.
\]
तब से $8 < 9$, प्रत्येक की दस प्रतियों को गुणा करना: $8^{10} < 9^{10}$, इसलिए
$2^{30} < 3^{20}$.
\end{solution}

\begin{solution}{pb:g9:fractions:1}
\textbf{1.} $\frac14 = 0.25$ (रुक जाता है); $\frac13 = 0.333\ldots $
(अवरोध पैदा करना $3$); $\frac56 = 0.8333\ldots $ (अवरोध पैदा करना $3$ के बाद
$8$); $\frac{3}{11} = 0.272727\ldots $ (अवरोध पैदा करना $27$).

\textbf{2.} विभाजन के प्रत्येक चरण पर $ b $, द \omterm{def:g3:division:remainder}{शेष}
में से एक है $ b $ मान $0, 1, \dots, b - 1$
(\cref{thm:g6:wholes:division}). यदि एक \omterm{def:g3:division:remainder}{शेष} $0$ प्रकट होता है,
विभाजन रुक जाता है. अन्यथा, अधिक से अधिक बाद में $ b $ कुछ कदम
\omterm{def:g3:division:remainder}{शेष} दूसरी बार प्रकट होना चाहिए --- और भी चरण हैं
संभव से अधिक \omterm{def:g3:division:remainder}{शेष} --- और उसी क्षण से विभाजन
के बाद उत्पन्न अंकों के अनुक्रम को बिल्कुल दोहराता है
पहली उपस्थिति: दशमलव लेखन चक्र हमेशा के लिए। रुकें या
दोहराएँ: कोई तीसरा व्यवहार मौजूद नहीं है।

\textbf{3.} $\frac17 = 0.142857\,142857\ldots $: द ब्लॉक
$142857$ लंबाई है $6$. प्रश्न 2 ने भीतर दोहराव का वादा किया
अधिक से अधिक $7$ चरण---यहाँ \omterm{def:g3:division:remainder}{शेष}
$1, 3, 2, 6, 4, 5$ सभी के समक्ष उपस्थित होते हैं $1$ रिटर्न: द
द्वारा एक प्रभाग के लिए सबसे लंबा संभव जप $7$.

\textbf{4.} $\frac ab $ बिलकुल रुक जाता है जब कुछ
$\frac{a \times k}{b \times k}$ है \omterm{def:g5:division:multiple}{भाजक} $10$, $100$,
$1000, \dots $ --- अर्थात, जब \omterm{def:g5:division:multiple}{भाजक} हो सकता है
ए तक गुणा किया गया \omterm{def:g9:fractions:power}{शक्ति} दस में से, जैसे $4 \times 25 = 100$ या
$8 \times 125 = 1\,000$. के लिए $3$ और $6$ यह निराशाजनक है: ए
\omterm{def:g9:fractions:power}{शक्ति} दस का अंक है \omterm{def:g1:addition:def}{जोड़} $1$, और हर एकाधिक का $3$ एक है
अंक \omterm{def:g1:addition:def}{जोड़} वह एकाधिक का $3$ --- तो नहीं एकाधिक का $3$
(इसलिए इनमें से कोई नहीं $6$) सदैव है $10$, $100$, $1\,000, \dots $ के लिए
$11$: विभाजन $10$, $100$, $1\,000$, $10\,000$ द्वारा $11$ हमेशा
एक छोड़ देता है \omterm{def:g3:division:remainder}{शेष} ($10$, $1$, $10$, $1, \dots $), कभी नहीं $0$.
हरों जो एक तक पहुंचता है \omterm{def:g9:fractions:power}{शक्ति} दस स्टॉप का; अन्य सभी दोहराते हैं।

\textbf{5.} $\frac{9}{40}$: $40 \times 25 = 1\,000$, रुकता है:
$\frac{225}{1000} = 0.225$. $\frac{33}{50}$:
$50 \times 2 = 100$, रुकता है: $\frac{66}{100} = 0.66$.
$\frac{5}{12}$: $12$ एक है एकाधिक का $3$, दोहराता है
($0.41666\ldots $).

\textbf{6.} $10x = 7.777\ldots $, इसलिए $10x - x = 7$ (पूंछ
बिल्कुल रद्द करें), $9x = 7$ और $ x = \frac79$. प्रभाग जांच:
$7 \div 9 = 0.777\ldots $

\textbf{7.} ब्लॉक में दो अंक हैं, इसलिए शिफ्ट करें $100$:
$100x - x = 72.7272\ldots - 0.7272\ldots = 72$, इस तरह
$99x = 72$ और $ x = \frac{72}{99} = \frac{8}{11}$.

\textbf{8.} $1000x = 583.333\ldots $ और $100x = 58.333\ldots $;
घटाना, $900x = 525$, इसलिए
$ x = \frac{525}{900} = \frac{7}{12}$ (से विभाजित करें $75$). वास्तव में
$\frac{7}{12} = 0.58333\ldots $

\textbf{9.} $10x - x = 9.999\ldots - 0.999\ldots = 9$, इसलिए
$9x = 9$ और $ x = 1$. ``बस नीचे'' नहीं $1$:
$0.999\ldots $ \emph{है} जो नंबर $1$, एक सेकंड में लिखा
पोशाक. कोई भी संख्या सख्ती से नीचे $1$ एक \omterm{ex:g1:subtraction:difference}{अंतर} छोड़ देता है, और
\cref{pb:g6:decimals:1} दिखाया गया है कि एक \omterm{ex:g1:subtraction:difference}{अंतर}ाल में हमेशा आगे शामिल होता है
संख्याएँ --- जबकि बीच में कुछ भी फिट नहीं बैठता $0.999\ldots $ और $1$.
अंतहीन नाइन चॉकलेट हैं \omterm{def:g3:division:remainder}{शेष} का
\cref{pb:g6:fractions:1} प्रत्येक सकारात्मक राशि से नीचे सिकुड़ गया:
कुछ भी नहीं बचा.

\textbf{10.} $2.4545\ldots = 2 + \frac{45}{99} = 2 +
\frac{5}{11} = \frac{27}{11}$. और
$0.123123\ldots = \frac{123}{999} = \frac{41}{333}$ (से विभाजित करें
$3$).

\textbf{11.} शब्दकोष: $0.037037\ldots = \frac{37}{999}$.
तब से $999 = 27 \times 37$, इससे यह सरल हो जाता है $\frac{1}{27}$:
द \omterm{def:g9:fractions:fraction}{अंश} $\frac{1}{27}$ मंत्र ``$037$''. प्रभाग जांच:
$1 \div 27 = 0.037037\ldots $

\textbf{12.} $0.0272727\ldots = \frac{1}{10} \times
0.272727\ldots = \frac{1}{10} \times \frac{27}{99}
= \frac{27}{990} = \frac{3}{110}$ (से विभाजित करें $9$).

\textbf{13.} के ब्लॉक \omterm{def:g1:counting:zero}{शून्य} अंतहीन रूप से बढ़ें, इसलिए कोई निश्चित नहीं है
ब्लॉक हमेशा के लिए दोहरा सकता है: यह है \emph{नहीं} एक दोहराव वाला दशमलव
(और यह कभी नहीं रुकता)। भाग I द्वारा, प्रत्येक \omterm{def:g9:fractions:fraction}{अंश} रुकता है या
दोहराता है --- तो यह संख्या नहीं है \omterm{def:g9:fractions:fraction}{अंश} बिल्कुल: यह है
तर्कहीन, ऑर्डर के अनुसार बनाया गया। अतार्किकता के सुपरस्टार,
$\sqrt2$, बेनकाब है \cref{ch:g9:sqrt}.

\textbf{14.} $2^{30} = \left(2^{10}\right)^3 \approx
(1.024)^3 \times 10^9 \approx 1.07 \times 10^9$: एक संख्या
थोड़ा ऊपर $10^9$, इसलिए साथ में $10$ अंक. (बिल्कुल:
$2^{30} = 1\,073\,741\,824$.)

\textbf{15.} $0.20262026\ldots = \frac{2026}{9999}$, पहले से ही अंदर
सबसे कम शर्तें ($9999 = 3 \times 3 \times 11 \times 101$ शेयर
के साथ कोई कारक नहीं $2026 = 2 \times 1013$). पूरा हुआ
शब्दकोश: \emph{ \omterm{def:g9:fractions:fraction}{अंश}ों बिल्कुल वही दशमलव हैं
रुकना या दोहराना} --- हरों दसियों दोस्तों में से सभी रुक जाते हैं
अन्य लोग एक \omterm{def:g6:lines:objects}{खंड} का जाप करते हैं; और इसके विपरीत प्रत्येक मंत्र, \omterm{def:g3:numbers:evenodd}{यहां तक की} आपका
स्नातक वर्ष, एक है \omterm{def:g9:fractions:fraction}{अंश} छद्मवेश में.
\end{solution}
