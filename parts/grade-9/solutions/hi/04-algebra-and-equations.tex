\section*{अध्याय \ref{ch:g9:algebra} --- बीजगणित और समीकरण}

\begin{solution}{exo:g9:algebra:1}
$4(3x - 2) = 12x - 8$.

$(x+6)(x+2) = x^2 + 2x + 6x + 12 = x^2 + 8x + 12$.

$(2x-3)(x+5) = 2x^2 + 10x - 3x - 15 = 2x^2 + 7x - 15$.

$(x-1)(x-9) = x^2 - 9x - x + 9 = x^2 - 10x + 9$.
\end{solution}

\begin{solution}{exo:g9:algebra:2}
$(x+8)^2 = x^2 + 16x + 64$.

$(3x-4)^2 = 9x^2 - 24x + 16$.

$(x+10)(x-10) = x^2 - 100$.

$(5x+2)(5x-2) = 25x^2 - 4$.
\end{solution}

\begin{solution}{exo:g9:algebra:3}
$101^2 = (100+1)^2 = 10000 + 200 + 1 = 10201$.

$59^2 = (60-1)^2 = 3600 - 120 + 1 = 3481$.

$53 \times 47 = (50+3)(50-3) = 2500 - 9 = 2491$.
\end{solution}

\begin{solution}{exo:g9:algebra:4}
$9x + 12 = 3(3x + 4)$.

$ x^2 - 64 = (x+8)(x-8)$.

$ x^2 + 14x + 49 = (x+7)^2$.

$18x^2 - 6x = 6x(3x - 1)$.
\end{solution}

\begin{solution}{exo:g9:algebra:5}
$4x + 7 = 19$: $4x = 12$, इसलिए $ x = 3$.

$6x - 5 = 2x + 11$: $4x = 16$, इसलिए $ x = 4$.

$5(x-3) = 3x + 1$: $5x - 15 = 3x + 1$, इसलिए $2x = 16$ और $ x = 8$.
\end{solution}

\begin{solution}{exo:g9:algebra:6}
$(x+4)(3x-12) = 0$: $ x = -4$ या $ x = 4$.

$ x^2 - 121 = (x-11)(x+11) = 0$: $ x = 11$ या $ x = -11$.

$(2x-1)^2 = 0$: एकल समाधान $ x = \frac12$.

$ x^2 = 8x $: $ x(x - 8) = 0$, इसलिए $ x = 0$ या $ x = 8$.
\end{solution}

\begin{solution}{exo:g9:algebra:7}
\omterm{ex:g1:subtraction:difference}{अंतर} वर्गों के साथ $ a = 3x + 2$ और $ b = 5$:
\[
E = (3x + 2)^2 - 5^2 = (3x + 2 + 5)(3x + 2 - 5) = (3x + 7)(3x - 3).
\]
शून्\omterm{def:g2:mult:def}{य-उत्पाद} नियम: $3x + 7 = 0$ देता है $ x = -\frac73$, और $3x - 3 = 0$
देता है $ x = 1$.
\end{solution}

\begin{solution}{exo:g9:algebra:8}
$3x - 4 \geq 8$: $3x \geq 12$, इसलिए $ x \geq 4$ (भरा हुआ बिंदु $4$, छाया
सही)।

$5 - 2x > 1$: $-2x > -4$, से \omterm{def:g9:arith:divisor}{विभाजित} करना $-2$ संकेत को उलट देता है: $ x < 2$
(खोखला बिंदु पर $2$, छाया बायीं ओर)।

$4(x+1) \leq x - 5$: $4x + 4 \leq x - 5$, इसलिए $3x \leq -9$, इसलिए
$ x \leq -3$ (भरा हुआ बिंदु $-3$, छाया बायीं ओर)।
\end{solution}

\begin{solution}{exo:g9:algebra:9}
होने देना $ x $ संख्या हो: $3x + 7 = 5x - 9$. तब $16 = 2x $, इसलिए $ x = 8$.
जाँच करना: $3 \times 8 + 7 = 31$ और $5 \times 8 - 9 = 31$.
\end{solution}

\begin{solution}{exo:g9:algebra:10}
\emph{1.} $(n+1)(n-1) = n^2 - 1$ (तीसरी पहचान). के पड़ोसी $ n $
हैं $ n - 1$ और $ n + 1$, तो उनके \omterm{def:g2:mult:def}{उत्पाद} है $ n^2 - 1$: उदा.\
$24 \times 26 = 625 - 1 = 624$.

\emph{2.} तीन क्रमागत पूर्णांक लिखे जा सकते हैं $ n - 1$, $ n $,
$ n + 1$. उनका \omterm{def:g1:addition:def}{जोड़} है
\[
(n-1) + n + (n+1) = 3n ,
\]
ए \omterm{def:g9:arith:divisor}{एकाधिक} का $3$ (बीच वाले का तीन गुना)।
\end{solution}

\begin{solution}{pb:g9:algebra:1}
\textbf{1.} $(10a + b) - (10b + a) = 9a - 9b = 9(a - b)$: द
\omterm{ex:g1:subtraction:difference}{अंतर} दो अंकों की संख्या और उसके बीच हमेशा उलटा होता है
ए \omterm{def:g9:arith:divisor}{एकाधिक} का $9$. जाँच करना: $72 - 27 = 45 = 9 \times 5$, और
$ a - b = 7 - 2 = 5$.

\textbf{2.} \omterm{def:g9:algebra:expand}{विस्तार} दाईं ओर:
$9(11a + b) + a + b + c = 99a + 9b + a + b + c
= 100a + 10b + c $. अतः संख्या a के बराबर है \omterm{def:g9:arith:divisor}{एकाधिक} का $9$ प्लस
इसका अंक \omterm{def:g1:addition:def}{जोड़}: से \omterm{def:g9:arith:divisor}{विभाजित} करना $9$, दोनों वही छोड़ देते हैं \omterm{def:g3:division:remainder}{शेष}.
वि\omterm{def:g3:division:remainder}{शेष} रूप से \omterm{def:g3:division:remainder}{शेष} $0$ एक साधन के लिए \omterm{def:g3:division:remainder}{शेष} $0$ के लिए
अन्य: नौ का नियम, सिद्ध।

\textbf{3.} $9(11a + b)$ एक भी है \omterm{def:g9:arith:divisor}{एकाधिक} का $3$, तो वही
पहचान संख्या और अंक दर्शाती है \omterm{def:g1:addition:def}{जोड़} वैसा ही रहने दो \omterm{def:g3:division:remainder}{शेष} में
द्वारा विभाजन $3$: तीन का नियम.

\textbf{4.} लिखना $47 = 9k + 2$ और $86 = 9l + 5$. तब
\[
47 \times 86 = (9k + 2)(9l + 5)
= 9\,(9kl + 5k + 2l) + 10 :
\]
 \omterm{def:g2:mult:def}{उत्पाद} वैसे ही छोड़ देता है \omterm{def:g3:division:remainder}{शेष} जैसा $2 \times 5 = 10$,
अर्थात् $1$. इसलिए एक सही परिणाम भी निकलना चाहिए \omterm{def:g3:division:remainder}{शेष}
$1$ --- कौन $4\,042$ करता है। लेकिन परीक्षण केवल देखता है \omterm{def:g3:division:remainder}{शेष}:
$4\,042$ और $4\,402$ (अंकों की अदला-बदली) समान रूप से पास होती है, इसलिए
उत्तीर्ण होना कुछ भी साबित नहीं करता --- असफल होना ही दोषी ठहराता है।

\textbf{5.} \omterm{def:g9:algebra:expand}{विस्तार}: $11(91a + 9b + c) - a + b - c + d
= 1001a + 99b + 11c - a + b - c + d
= 1000a + 100b + 10c + d $. तो संख्या है \omterm{def:g6:wholes:divisible}{भाज्य} द्वारा $11$
बिल्कुल जब यह वैकल्पिक हो \omterm{def:g1:addition:def}{जोड़} $-a + b - c + d $ है। के लिए
$2\,926$: $-2 + 9 - 2 + 6 = 11$, \omterm{def:g6:wholes:divisible}{भाज्य} द्वारा $11$: हाँ,
$2\,926 = 11 \times 266$.

\textbf{6.} $742 - 247 = 495$; $495 + 594 = 1\,089$. कोई भी वैध
उदाहरण भूमि पर है $1\,089$ बहुत।

\textbf{7.} $(100a + 10b + c) - (100c + 10b + a) = 99(a - c)$.
के लिए $ a - c = 2, 3, \dots, 9$:
\[
198,\ 297,\ 396,\ 495,\ 594,\ 693,\ 792,\ 891 .
\]

\textbf{8.} $100(m - 1) + 90 + (10 - m) = 100m - 100 + 90 + 10
- m = 99m $. तो के तीन अंक $99m $ हैं $ m - 1$, तब $9$,
तब $10 - m $ (जाँच करना $495$: $ m = 5$ देता है $4$, $9$, $5$).

\textbf{9.} का उलटा $99m $ अंक हैं $10 - m $, $9$,
$ m - 1$, तो यह बराबर है $100(10 - m) + 90 + (m - 1)$. जोड़ना:
\[
100\,(m - 1 + 10 - m) + 180 + (10 - m + m - 1)
= 100 \times 9 + 180 + 9 = 1\,089 ,
\]
से स्वतंत्र $ m $. शुरुआती संख्या जो भी हो, जादूगर
सुरक्षित है.

\textbf{10.} अगर $ a = c $, घटाव देता है $0$ और चाल
तुरंत मर जाता है. अगर $ a - c = 1$, देता है $99$, जो होना चाहिए
तीन अंकों की स्ट्रिंग के रूप में माना जाता है $099$ (उलट $990$,
$99 + 990 = 1\,089$: सहेजा गया!) छोटा प्रिंट: मतभेद का
$1$ केवल तभी काम करें जब दर्शक घटाव परिणाम लिखता है
तीन अंकों के साथ, अग्रणी \omterm{def:g1:counting:zero}{शून्य} शामिल --- के \omterm{ex:g1:subtraction:difference}{अंतर}ाल की आवश्यकता है
कम से कम $2$ बहस से बचता है.

\textbf{11.} $2(5m + 7) + d - 14 = 10m + 14 + d - 14 =
10m + d $: दहाई (और सैकड़ों, अक्टूबर-दिसंबर के लिए) शो
महीना, इकाई (और दहाई) दिन।

\textbf{12.} $2(5m + 9) + d = 10m + 18 + d $: घटाना $18$.

\textbf{13.} प्रश्न 2 से, $ n - (\text{digit sum of } n) =
9(11a + b)$ तीन अंकों के लिए $ n $: हमेशा ए \omterm{def:g9:arith:divisor}{एकाधिक} का $9$
(समान पहचान किसी भी लम्बाई के लिए काम करती हैं)। ए \omterm{def:g9:arith:divisor}{एकाधिक} का $9$ है
अंक \omterm{def:g1:addition:def}{जोड़} ए \omterm{def:g9:arith:divisor}{एकाधिक} का $9$ (प्रश्न 2 फिर से), तो अंक
दर्शक जो पढ़ता है उसे अगले तक पूरा किया जाना चाहिए \omterm{def:g9:arith:divisor}{एकाधिक}
का $9$: लुप्त अंक वह पूरक है। अस्पष्टता: यदि
अंकों का योग पहले से ही a पढ़ें \omterm{def:g9:arith:divisor}{एकाधिक} का $9$, पार किया गया अंक
हो सकता है $0$ या $9$ --- को छोड़कर $0$ संदेह को दूर करता है.

\textbf{14.} $(10t + 5)^2 = 100t^2 + 100t + 25 =
100\,t(t + 1) + 25$: लिखना $ t(t+1)$, फिर गोंद $25$ इसके पीछे।
$45^2$: $4 \times 5 = 20$, इसलिए $2\,025$. $85^2$:
$8 \times 9 = 72$, इसलिए $7\,225$. $105^2$: $10 \times 11 = 110$,
इसलिए $11\,025$.

\textbf{15.} लगातार पांच: $ n + (n+1) + (n+2) + (n+3) +
(n+4) = 5n + 10 = 5(n + 2)$: ए \omterm{def:g9:arith:divisor}{एकाधिक} का $5$, हमेशा। चार
लगातार: $ n + (n+1) + (n+2) + (n+3) = 4n + 6 =
4(n + 1) + 2$: \omterm{def:g3:division:remainder}{शेष} $2$ द्वारा प्रभाग में $4$, कभी नहीं ए
\omterm{def:g9:arith:divisor}{एकाधिक}. बीजगणित ``ऐसा लगता है कि यह काम करता है'' में परिवर्तित करता है
\emph{हमेशा}, और ``मुझे कोई उदाहरण नहीं मिला'' \emph{कभी नहीं} ---
दो शब्द जिन तक कोई भी परीक्षण नहीं पहुंच सकता
(\cref{met:g7:literal:test}).
\end{solution}
