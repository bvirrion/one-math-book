\begin{solution}{exo:g9:arith:2}
$2\,346$ में समाप्त होता है $6$: \omterm{def:g6:wholes:divisible}{भाज्य} द्वारा $2$, द्वारा नहीं $5$. अंक \omterm{def:g1:addition:def}{जोड़}
$2 + 3 + 4 + 6 = 15$: \omterm{def:g6:wholes:divisible}{भाज्य} द्वारा $3$, द्वारा नहीं $9$. अंतिम दो अंक
$46$, और $46 = 4 \times 11 + 2$ क्या नहीं है \omterm{def:g6:wholes:divisible}{भाज्य} द्वारा $4$: $2\,346$ क्या नहीं है \omterm{def:g6:wholes:divisible}{भाज्य} द्वारा $4$.
\end{solution}

\begin{solution}{exo:g9:arith:4}
$101$: अभाज्य संख्याओं का परीक्षण करें $ p $ साथ $ p^2 \leq 101$, यानी\ $2, 3, 5, 7$. कोई नहीं
\omterm{def:g9:arith:divisor}{विभाजित} $101$ (\omterm{def:g3:numbers:evenodd}{विषम}, अंक \omterm{def:g1:addition:def}{जोड़} $2$, में ख़त्म नहीं होता $0/5$,
$101 = 7 \times 14 + 3$): $101$ प्रधान है.

$91 = 7 \times 13$: प्रधान नहीं.

$143 = 11 \times 13$: प्रधान नहीं.
\end{solution}

\begin{solution}{exo:g9:arith:5}
सामान्य वि\omterm{def:g5:division:multiple}{भाजक} का $48$ और $60$: वि\omterm{def:g5:division:multiple}{भाजक} 48 में से हैं $1, 2, 3, 4, 6, 8,
12, 16, 24, 48$; वि\omterm{def:g5:division:multiple}{भाजक} का $60$ हैं $1, 2, 3, 4, 5, 6, 10, 12, 15, 20,
30, 60$; आम हैं $1, 2, 3, 4, 6, 12$, इसलिए $\gcd(48,60) = 12$.

गुणनखंडन द्वारा: $48 = 2^4 \times 3$ और $60 = 2^2 \times 3 \times 5$;
छोटे के साथ सामान्य अभाज्य घातांक: $2^2 \times 3 = 12$.
\end{solution}

\begin{solution}{exo:g9:arith:6}
$\gcd(255, 154)$:
\begin{align*}
255 &= 154 \times 1 + 101, \\
154 &= 101 \times 1 + 53, \\
101 &= 53 \times 1 + 48, \\
53 &= 48 \times 1 + 5, \\
48 &= 5 \times 9 + 3, \\
5 &= 3 \times 1 + 2, \\
3 &= 2 \times 1 + 1, \\
2 &= 1 \times 2 + 0 .
\end{align*}
अंतिम अशून्य \omterm{def:g3:division:remainder}{शेष}: $\gcd(255, 154) = 1$ (वे हैं \omterm{def:g9:arith:gcd}{सह अभाज्य}).

$\gcd(1053, 325)$:
\begin{align*}
1053 &= 325 \times 3 + 78, \\
325 &= 78 \times 4 + 13, \\
78 &= 13 \times 6 + 0 .
\end{align*}
$\gcd(1053, 325) = 13$.
\end{solution}

\begin{solution}{exo:g9:arith:7}
यूक्लिडियन एल्गोरिथ्म: $588 = 504 \times 1 + 84$; $504 = 84 \times 6 + 0$:
$\gcd(588, 504) = 84$. तब
\[
\frac{588}{504} = \frac{588 \div 84}{504 \div 84} = \frac{7}{6},
\]
जो अपरिवर्तनीय है.
\end{solution}

\begin{solution}{exo:g9:arith:8}
गुलदस्तों की संख्या से दोनों को \omterm{def:g9:arith:divisor}{विभाजित} करना होगा $84$ और $126$; सबसे बड़ा
संभव है $\gcd(84, 126)$. गुणनखंड: $84 = 2^2 \times 3 \times
7$, $126 = 2 \times 3^2 \times 7$, तो जीसीडी है $2 \times 3 \times 7 =
42$. वह बना सकती है $42$ गुलदस्ते, प्रत्येक में शामिल
$\frac{84}{42} = 2$ गुलाब और $\frac{126}{42} = 3$ ट्यूलिप.
\end{solution}

\begin{solution}{exo:g9:arith:9}
हमें कम से कम सामान्य की आवश्यकता है \omterm{def:g9:arith:divisor}{एकाधिक}. $24 = 2^3 \times 3$ और
$36 = 2^2 \times 3^2$; प्रत्येक अभाज्य को साथ लेकर \emph{बड़ा}
\omterm{def:g8:powers:def}{प्रतिपादक}: $\lcm = 2^3 \times 3^2 = 72$. अगली नौकाएँ एक साथ रवाना होती हैं
$72$ 8:00 के बाद 9:12 मिनट पर।
\end{solution}

\begin{solution}{exo:g9:arith:10}
\emph{1.} कोई भी सामान्य \omterm{def:g5:division:multiple}{भाजक} $ d $ का $ n $ और $ n+1$ भी \omterm{def:g9:arith:divisor}{विभाजित} उनका
\omterm{ex:g1:subtraction:difference}{अंतर} $(n+1) - n = 1$, इसलिए $ d = 1$: $\gcd(n, n+1) = 1$.

\emph{2.} ए \omterm{def:g9:fractions:fraction}{अंश} वास्तव में अपरिवर्तनीय है जब यह \omterm{def:g4:fractions:def}{मीटर} और
\omterm{def:g5:division:multiple}{भाजक} हैं \omterm{def:g9:arith:gcd}{सह अभाज्य}, जो मामला है $ n $ और $ n + 1$ भाग से
1.
\end{solution}

\begin{solution}{pb:g9:arith:1}
\textbf{1.} भरें $3$ और इसे इसमें डाल दें $5$ (सामग्री
$0/3 \to 3$ बड़े में)। भरें $3$ फिर से डालें और डालें
$5$ जब तक यह भर न जाए: बड़ा जग ही लेता है $2$ और अधिक, जा रहा हूँ
\[
3 - 2 = 1 \text{ L in the small jug.}
\]
चालें: भरें $3$; बहना $3 \to 5$; भरना $3$; बहना $3 \to 5$.

\textbf{2.} भरें $5$; में डालो $3$ (पत्तियों $2$ में
बड़ा); खाली करें $3$; डालो $2$ में $3$; भरें $5$;
में डालो $3$ पूर्ण होने तक---लगता है $1$, जा रहा हूँ
$\mathbf{4}$ बड़े जग में एल. छह चालें.

\textbf{3.} उन सभी को: $1$ (प्रश्न 1), $2$ (दो चालों के बाद
प्रश्न 2 का), $3$ और $5$ (एकल भराई), $4$ (प्रश्न
2), $6 = 3 + 3$ (एक पूरा छोटा जग प्लस $3$ में डाल दिया
बड़ा), $7 = 5 + 2$, $8 = 5 + 3$ (दोनों पूर्ण)। प्रत्येक पूर्ण राशि
से $1$ को $8$ एल के साथ मापने योग्य है $5$ और यह $3$.

\textbf{4.} प्रारंभ: दोनों जग पकड़ें $0$, ए \omterm{def:g9:arith:divisor}{एकाधिक} का $2$.
भरना एक सामग्री को सेट करता है $6$ या $4$: \omterm{def:g3:numbers:evenodd}{यहां तक की}. खाली करने से यह सेट हो जाता है
$0$: \omterm{def:g3:numbers:evenodd}{यहां तक की}. डालने से कुछ पानी जगों के बीच चला जाता है जिनकी सामग्री
थे \omterm{def:g3:numbers:evenodd}{यहां तक की}, और डाली गई मात्रा एक है \omterm{ex:g1:subtraction:difference}{अंतर} का सम संख्या (स्थान \omterm{def:g3:division:remainder}{शेष}, या उपलब्ध मात्रा): सभी सामग्री यथावत रहती है
\omterm{def:g3:numbers:evenodd}{यहां तक की} हमेशा के लिए। एक \omterm{def:g3:numbers:evenodd}{विषम} जैसे लक्ष्य $1$ एल पहुंच योग्य नहीं है.

\textbf{5.} $\gcd(6, 4) = 2$: केवल \omterm{def:g3:numbers:evenodd}{यहां तक की} रकम --- द्वारा पुष्टि की गई
प्रश्न 4. $\gcd(5, 3) = 1$: प्रत्येक संपूर्ण राशि की अनुमति है
कानून, और प्रश्न 3 ने उन सभी को साकार कर दिया। जीसीडी बिल्कुल सही है
गुड़ की माप की इकाई।

\textbf{6.} $91 = 1 \times 65 + 26$; $65 = 2 \times 26 + 13$;
$26 = 2 \times 13 + 0$: $\gcd(91, 65) = 13$. और
$2\,026 = 44 \times 46 + 2$; $46 = 23 \times 2 + 0$:
$\gcd(2\,026, 46) = 2$.

\textbf{7.} डालने का कार्य $5$ में $13$- बार-बार सुराही: दो के बाद
बड़े जग को भरना $10$; तीसरी फिलिंग ही फिट बैठती है
$3$, जा रहा हूँ $5 - 3 = 2$ छोटे से जग में --- \omterm{def:g3:division:remainder}{शेष} का
$13$ द्वारा $5$ था $3$, और राशियाँ $3$ (\omterm{ex:g1:subtraction:difference}{अंतर}िक्ष) और $2$
(बचे हुए) बिल्कुल यूक्लिड की संख्याएँ हैं ($13 = 2 \times 5 + 3$,
$5 = 1 \times 3 + 2$). जारी है, $3 - 2 = 1$ प्रकट होता है:
एल्गोरिदम का अगला \omterm{def:g3:division:remainder}{शेष}. फव्वारा यूक्लिड का प्रदर्शन करता है
पानी के साथ विभाजन.

\textbf{8.} $\gcd(13, 5) = 1$, इसलिए प्रश्न 5 का कानून अनुमति देता है
प्रत्येक पूर्ण राशि --- और वास्तव में प्रश्न 7 का झरना
\omterm{def:g2:mult:def}{उत्पाद}ित $1$ एल. हाँ.

\textbf{9.} एक आम \omterm{def:g5:division:multiple}{भाजक} का $ n $ और $2n + 1$ \omterm{def:g9:arith:divisor}{विभाजित}
$2n + 1 - 2 \times n = 1$: यह होना ही चाहिए $1$. इस तरह
$\gcd(n, 2n+1) = 1$ हमेशा।

\textbf{10.} बस ए: 7:12, 7:24, 7:36, 7:48, 8:00\,\dots; बस बी:
7:18, 7:36, 7:54\,\dots\ प्रथम सामान्य प्रस्थान: 7:36, पश्चात
$36$ मिनट --- पहला सामान्य \omterm{def:g9:arith:divisor}{एकाधिक} का $12$ और $18$.
कानून: $36 \times \gcd(12, 18) = 36 \times 6 = 216 = 12 \times
18$. के लिए $5$ और $3$: पहला सामान्य \omterm{def:g9:arith:divisor}{एकाधिक} $15$, और
$15 \times \gcd(5,3) = 15 \times 1 = 15 = 5 \times 3$.

\textbf{11.} के साथ $17$-वर्ष चक्र: अगला संयोग है
पहला आम \omterm{def:g9:arith:divisor}{एकाधिक} का $17$ और $4$; तब से
$\gcd(17, 4) = 1$, वह है $17 \times 4 = 68$ वर्ष ---
सिकाडस चार उद्भवों में एक बार शिखर पर पहुँचते हैं। के साथ
$16$-वर्ष चक्र: $16$ एक है \omterm{def:g9:arith:divisor}{एकाधिक} का $4$, इसलिए \emph{प्रत्येक}
उद्भव चरम पर है। एक अभाज्य चक्र की लंबाई कोई कारक साझा नहीं करती
किसी भी छोटे शिकारी चक्र के साथ, संयोगों को दूर तक धकेलता है
यथासंभव अलग: छलावरण के रूप में अंकगणित।

\textbf{12.} के चार विकल्प \omterm{def:g8:powers:def}{प्रतिपादक} के लिए $2$, तीन के लिए $3$,
दो के लिए $5$: $4 \times 3 \times 2 = 24$ वि\omterm{def:g5:division:multiple}{भाजक}.

\textbf{13.} अगर $ m = 2^{a} \times 3^{b} \times \cdots $, तब
$ m^2 = 2^{2a} \times 3^{2b} \times \cdots $: प्रत्येक \omterm{def:g8:powers:def}{प्रतिपादक} है
इसलिए दोगुना हो गया \omterm{def:g3:numbers:evenodd}{यहां तक की}. में $360 = 2^3 \times 3^2 \times 5$, द
घातांक का $2$ और का $5$ हैं \omterm{def:g3:numbers:evenodd}{विषम}: $360$ पूर्ण नहीं है
वर्ग.

\textbf{14.} ए \omterm{def:g5:division:multiple}{भाजक} ठीक उसी समय उसका अपना साथी होता है
$ d = \frac nd $, यानी\ $ n = d^2$: केवल वर्गों में ऐसा होता है
मध्य \omterm{def:g5:division:multiple}{भाजक}. अन्य सभी के लिए $ n $  वि\omterm{def:g5:division:multiple}{भाजक} में \omterm{def:g9:arith:divisor}{विभाजित}
\omterm{def:g1:addition:def}{जोड़}े, ए \omterm{def:g3:numbers:evenodd}{यहां तक की} \omterm{def:g7:stats:frequency}{गिनती करना}। इसलिए: \omterm{def:g3:numbers:evenodd}{विषम} संख्या का वि\omterm{def:g5:division:multiple}{भाजक}
$\Leftrightarrow $ पूर्ण वर्ग. जाँच करना: $36$ है वि\omterm{def:g5:division:multiple}{भाजक}
$1, 2, 3, 4, 6, 9, 12, 18, 36$ --- उनमें से नौ, \omterm{def:g3:numbers:evenodd}{विषम}, और
$36 = 6^2$; जबकि $360$ है $24$ (प्रश्न 12), \omterm{def:g3:numbers:evenodd}{यहां तक की}, और है
कोई वर्ग नहीं (प्रश्न 13)।

\textbf{15.} लॉकर $ n $ प्रत्येक छात्र द्वारा एक बार टॉगल किया जाता है $ k $
किसका नंबर \omterm{def:g9:arith:divisor}{विभाजित} $ n $: कुल मिलाकर, जितनी बार $ n $ है
वि\omterm{def:g5:division:multiple}{भाजक}. एक लॉकर समाप्त होता है \emph{खुला} जब इसे टॉगल किया जाता है अजीब
संख्या समय का --- प्रश्न 14 द्वारा, बिल्कुल कब $ n $ एक है
पूर्ण वर्ग. लॉकर खोलें: $1, 4, 9, 16, 25, 36, 49, 64,
81, 100$ --- उनमें से दस.
\end{solution}
