\begin{solution}{exo:g9:statproba:4}
$\P(3) = \frac16$. \quad
$\P(\text{even}) = \frac36 = \frac12$. \quad
$\P(\text{at least } 3) = \P(\{3,4,5,6\}) = \frac46 = \frac23$. \quad
$\P(\text{not } 6) = 1 - \frac16 = \frac56$.
\end{solution}

\begin{solution}{exo:g9:statproba:5}
$\P(\text{white}) = \frac{5}{12}$.

$\P(\text{red or green}) = \frac{4 + 3}{12} = \frac{7}{12}$.

$\P(\text{not white}) = 1 - \frac{5}{12} = \frac{7}{12}$.

अंतिम दो बराबर हैं: ``सफेद नहीं'' \emph{है} घटना ``लाल या
हरा''
\end{solution}

\begin{solution}{exo:g9:statproba:6}
 $12$ परिणाम (चेहरा सिक्के का मूल्य, पासे का मूल्य) समान हैं
\omterm{def:g9:statproba:probability}{संभावना}.

प्रमुख और ए $6$: एक परिणाम बाहर $12$: $\frac{1}{12}$.

प्रमुखों या ए $6$: शीर्षों के साथ परिणाम ($6$ उनमें से) प्लस परिणाम
(पूंछ, $6$): $7$ परिणाम, तो $\frac{7}{12}$. प्रत्येक घटना की गिनती और
ओवरलैप घटाने पर वही परिणाम मिलता है:
$\frac{6}{12} + \frac{2}{12} - \frac{1}{12} = \frac{7}{12}$.
\end{solution}

\begin{solution}{exo:g9:statproba:7}
पहला ड्रा: लाल $\frac25$, काला $\frac35$. बिना दूसरा ड्रा
प्रतिस्थापन: एक लाल रंग के बाद, $1$ लाल और $3$ काले बीच में रहते हैं $4$; के बाद
एक काला, $2$ लाल और $2$ बीच में काला $4$.

दो काले: $\frac35 \times \frac24 = \frac{6}{20} = \frac{3}{10}$.

विभिन्न रंग (आरबी या बीआर):
$\frac25 \times \frac34 + \frac35 \times \frac24
= \frac{6}{20} + \frac{6}{20} = \frac{12}{20} = \frac35$.
\end{solution}

\begin{solution}{exo:g9:statproba:8}
\emph{1.} $9 \times 14 = 126$ अंक.

\emph{2.} एक औसत वर्ष $15$ ऊपर $10$ मिलान का मतलब है $150$ में अंक
कुल: उसे स्कोर करना होगा $150 - 126 = 24$ अंक.
\end{solution}

\begin{solution}{exo:g9:statproba:9}
\emph{1.} बिच में $36$ समान रूप से संभावित \omterm{def:g1:addition:def}{जोड़}े, युगल हैं
$(1,1), (2,2), \dots, (6,6)$: उनमें से छह, तो
$\P(\text{double}) = \frac{6}{36} = \frac16$.

\emph{2.} दो स्वतंत्र खेल: जीत के साथ पेड़ ($\frac16$) / खोना
($\frac56$) प्रत्येक स्तर पर। \omterm{def:g9:statproba:probability}{संभावना} कभी न जीतने का:
$\frac56 \times \frac56 = \frac{25}{36}$. इसलिए
\[
\P(\text{at least one win}) = 1 - \frac{25}{36} = \frac{11}{36}
\approx 0.31 .
\]
\end{solution}

\begin{solution}{pb:g9:statproba:1}
\textbf{1.} प्रत्येक रोल में है $5$ गैर-छह परिणाम बाहर $6$;
$6 \times 6$ तालिका से पता चलता है $5 \times 5 = 25$ छह मुक्त परिणाम बाहर
का $36$: $ P(\text{no six}) = \frac{25}{36}$, इसलिए
$ P(\text{at least one}) = 1 - \frac{25}{36} = \frac{11}{36}
\approx 0.31$.

\textbf{2.} $\frac13 = \frac{12}{36} \neq \frac{11}{36}$. द
\omterm{def:g1:addition:def}{जोड़} प्रत्येक में एक बार परिणाम ``छह फिर छह'' को गिनता है
$\frac16$: एक बार भी बार-बार. घटनाओं की संभावनाएँ जो हो सकती हैं
घटित \emph{एक साथ} बस \omterm{def:g1:addition:def}{जोड़}ें नहीं.

\textbf{3.} \omterm{def:g1:addition:def}{जोड़} $7$: छह परिणाम $(1,6), (2,5), (3,4),
(4,3), (5,2), (6,1)$: $ P = \frac{6}{36} = \frac16$. \omterm{def:g1:addition:def}{जोड़} $2$:
केवल $(1,1)$: $\frac{1}{36}$. सात के पास सबसे अधिक तरीके हैं
होता है--- मोटा बीच का रकम मेज़।

\textbf{4.} \omterm{def:g1:addition:def}{जोड़} $9$: $(3,6), (4,5), (5,4), (6,3)$ ---चार
परिणाम, $\frac{4}{36}$. \omterm{def:g1:addition:def}{जोड़} $10$: $(4,6), (5,5), (6,4)$ ---
तीन, $\frac{3}{36}$. नौ की \omterm{def:g9:statproba:probability}{संभावना} अधिक है: गिनती निर्णय करती है,
``लिखने के तरीकों'' की संख्या नहीं \omterm{def:g1:addition:def}{जोड़} अव्यवस्थित के साथ
\omterm{def:g5:solids:def}{चेहरे}.

\textbf{5.} पेड़ का प्रत्येक स्तर नो-छह को गुणा करता है
\omterm{def:g9:statproba:probability}{संभावना} द्वारा $\frac56$, पहले जो कुछ भी हुआ: बाद में $ n $
रोल, $ P(\text{no six at all}) = \left(\frac56\right)^n $, और
पूरक नियम सूत्र देता है.

\textbf{6.} $\left(\frac56\right)^4 = \frac{625}{1296} \approx
0.482$, इसलिए शेवेलियर ने जीत हासिल की \omterm{def:g9:statproba:probability}{संभावना}
$1 - 0.482 = 0.518$ --- लगभग एक शांत, स्थिर किनारा
$2\,\%$ प्रति गेम: एक उत्कृष्ट दांव, और इसका भुगतान उसे करना पड़ा
गाड़ियाँ.

\textbf{7.} पर $7$ रोल उसका नियम देता है $\frac76$ --- ए
\omterm{def:g9:statproba:probability}{संभावना} से अधिक $1$, बकवास। यह प्रश्न 2 है
डबल-गिनती, मिश्रित: रोल की सफलताएँ ओवरलैप होती हैं, और
उनके अवसरों को जोड़ने पर बार-बार ओवरलैप गिना जाता है।

\textbf{8.} $1 - \left(\frac{35}{36}\right)^{24} \approx
1 - 0.5086 = 0.491$: एक के नीचे \omterm{def:g3:division:half}{आधा}. शर्त \omterm{def:g3:numbers:evenodd}{यहां तक की} एक पर पैसा
$49.1\,\%$ घटना, शेवेलियर खो गया $2$ हर में खेल
$100$ औसतन --- धीरे-धीरे, रहस्यमय तरीके से (उसे), और
अनिवार्य रूप से.

\textbf{9.} $1 - \left(\frac{35}{36}\right)^{25} \approx
1 - 0.494 = 0.506$: साथ $25$ रोल्स दांव अनुकूल हो जाता है।
एक रोल ने शेवेलियर को लाभप्रदता से अलग कर दिया।

\textbf{10.} ए को गुणा करना \omterm{def:g9:statproba:probability}{संभावना} प्रयासों की संख्या से
कई बार ओवरलैपिंग सफलताओं को गिनता है, और अंततः
उपरोक्त असंभव उत्तर उत्पन्न करता है $1$. सही मार्ग है
हमेशा पूरक के माध्यम से: विफलता की संभावनाओं को श्रृंखलाबद्ध करें
गुणा करके, फिर घटाना $1$.

\textbf{11.} \omterm{def:g9:statproba:indicators}{अर्थ}: $\frac{9 \times 2\,000 + 20\,000}{10} =
\frac{38\,000}{10} = 3\,800$ यूरो. मंझला: मध्य वेतन
दोनों हैं $2\,000$: \omterm{def:g9:statproba:indicators}{MEDIAN} $2\,000$ यूरो. विज्ञापन
को उद्धृत करना चाहिए \omterm{def:g9:statproba:indicators}{MEDIAN} --- दस में से नौ कर्मचारी कभी नहीं देखते
कुछ भी पसंद है $3\,800$; \omterm{def:g9:statproba:indicators}{अर्थ} एक के द्वारा ऊपर खींच लिया जाता है
बाह्य (\cref{ex:g9:statproba:weighted} वज़न के बारे में चेतावनी दी गई)

\textbf{12.} उदाहरण के लिए $2, 8, 15, 17, 18$: \omterm{def:g9:statproba:indicators}{MEDIAN} $15$
(मध्य मान), \omterm{def:g9:statproba:indicators}{अर्थ} $\frac{60}{5} = 12$. ए \omterm{def:g9:statproba:indicators}{अर्थ} के नीचे
\omterm{def:g9:statproba:indicators}{MEDIAN} की पूँछ को धोखा देता है \emph{कम} औसत खींचने के निशान
नीचे जबकि ऊपर \omterm{def:g3:division:half}{आधा} ऊँचा बैठता है.

\textbf{13.} पहले तीन कारक:
$1 \times \frac{364}{365} \times \frac{363}{365} \approx
0.992$. प्रत्येक नए छात्र को एक और ली गई तारीख से बचना होगा, इसलिए
कारक सिकुड़ते हैं: $\frac{343}{365} \approx 0.94$ के लिए
$23$ तीसरा. पूर्ण \omterm{def:g2:mult:def}{उत्पाद} के बारे में गिर जाता है $0.493$ --- इसलिए
\[
P(\text{at least one shared birthday}) \approx 0.507 :
\]
से बेहतर \omterm{def:g3:numbers:evenodd}{यहां तक की}, बस की एक कक्षा में $23$. \omterm{ex:g1:subtraction:difference}{अंतर}्ज्ञान कहता है
``$23$ लोग, $365$ दिन, कोई मौका नहीं''; गणित कहता है
``एक सिक्का उछालो''।

\textbf{14.} $\frac{23 \times 22}{2} = 253$ \omterm{def:g1:addition:def}{जोड़}े. प्रत्येक \omterm{def:g1:addition:def}{जोड़}ी है
संयोग का एक ताज़ा अवसर, और $253$ अवसर
पर $\frac{1}{365}$ प्रत्येक अब कोई छोटा मामला नहीं है:
जब कोई \omterm{def:g1:addition:def}{जोड़}े को गिनता है तो आश्चर्य समाप्त हो जाता है, लोगों को नहीं।

\textbf{15.} उदाहरण प्रोटोकॉल: एक पासे को चार के ब्लॉक में रोल करें,
यह रिकॉर्ड करना कि क्या प्रत्येक ब्लॉक एक छक्का दिखाता है $100$ ब्लॉक; या
के बारे में कई वर्गों की जन्मदिन सूचियाँ एकत्र करें $23$ और
संयोग के साथ अनुपात रिकॉर्ड करें। अवलोकन किया गया
\omterm{def:g7:stats:frequency}{आवृत्ति} डगमगा जाएगा, लेकिन जैसे-जैसे पुनरावृत्ति की संख्या बढ़ती जाएगी
इसे गणना की गई संभावनाओं के और भी करीब पहुंचना चाहिए
($0.518$; $0.507$) --- आवृत्तियाँ संभावनाओं में परिवर्तित होती हैं:
बड़ी संख्या का नियम, यहां एक अपेक्षा के रूप में बताया गया है
हाई स्कूल वॉल्यूम में साबित हुआ।
\end{solution}
