\section*{Chapter \ref{ch:g9:algebra} --- Algebra and Equations}

\begin{solution}{exo:g9:algebra:1}
$4(3x - 2) = 12x - 8$.

$(x+6)(x+2) = x^2 + 2x + 6x + 12 = x^2 + 8x + 12$.

$(2x-3)(x+5) = 2x^2 + 10x - 3x - 15 = 2x^2 + 7x - 15$.

$(x-1)(x-9) = x^2 - 9x - x + 9 = x^2 - 10x + 9$.
\end{solution}

\begin{solution}{exo:g9:algebra:2}
$(x+8)^2 = x^2 + 16x + 64$.

$(3x-4)^2 = 9x^2 - 24x + 16$.

$(x+10)(x-10) = x^2 - 100$.

$(5x+2)(5x-2) = 25x^2 - 4$.
\end{solution}

\begin{solution}{exo:g9:algebra:3}
$101^2 = (100+1)^2 = 10000 + 200 + 1 = 10201$.

$59^2 = (60-1)^2 = 3600 - 120 + 1 = 3481$.

$53 \times 47 = (50+3)(50-3) = 2500 - 9 = 2491$.
\end{solution}

$x^2 + 14x + 49 = (x+7)^2$.

$18x^2 - 6x = 6x(3x - 1)$.

\begin{solution}{exo:g9:algebra:5}
$4x + 7 = 19$: $4x = 12$, so $x = 3$.

$6x - 5 = 2x + 11$: $4x = 16$, so $x = 4$.

$5(x-3) = 3x + 1$: $5x - 15 = 3x + 1$, so $2x = 16$ and $x = 8$.
\end{solution}

\begin{solution}{exo:g9:algebra:6}
$(x+4)(3x-12) = 0$: $x = -4$ or $x = 4$.

$x^2 - 121 = (x-11)(x+11) = 0$: $x = 11$ or $x = -11$.

$(2x-1)^2 = 0$: the single solution $x = \frac12$.

$x^2 = 8x$: $x(x - 8) = 0$, so $x = 0$ or $x = 8$.
\end{solution}

\begin{solution}{exo:g9:algebra:7}
Difference of squares with $a = 3x + 2$ and $b = 5$:
\[
E = (3x + 2)^2 - 5^2 = (3x + 2 + 5)(3x + 2 - 5) = (3x + 7)(3x - 3).
\]
Zero-product rule: $3x + 7 = 0$ gives $x = -\frac73$, and $3x - 3 = 0$
gives $x = 1$.
\end{solution}

\begin{solution}{exo:g9:algebra:8}
$3x - 4 \geq 8$: $3x \geq 12$, so $x \geq 4$ (filled dot at $4$, shade
right).

$5 - 2x > 1$: $-2x > -4$, dividing by $-2$ reverses the sign: $x < 2$
(hollow dot at $2$, shade left).

$4(x+1) \leq x - 5$: $4x + 4 \leq x - 5$, so $3x \leq -9$, so
$x \leq -3$ (filled dot at $-3$, shade left).
\end{solution}

\begin{solution}{exo:g9:algebra:9}
Let $x$ be the number: $3x + 7 = 5x - 9$. Then $16 = 2x$, so $x = 8$.
Check: $3 \times 8 + 7 = 31$ and $5 \times 8 - 9 = 31$.
\end{solution}

\begin{solution}{exo:g9:algebra:10}
\emph{1.} $(n+1)(n-1) = n^2 - 1$ (third identity). The neighbors of $n$
are $n - 1$ and $n + 1$, so their product is $n^2 - 1$: e.g.\
$24 \times 26 = 625 - 1 = 624$.

\emph{2.} Three consecutive integers can be written $n - 1$, $n$,
$n + 1$. Their sum is
\[
(n-1) + n + (n+1) = 3n ,
\]
a multiple of $3$ (three times the middle one).
\end{solution}
