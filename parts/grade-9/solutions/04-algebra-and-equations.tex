\begin{solution}{exo:g9:algebra:1}
$4(3x - 2) = 12x - 8$.

$(x+6)(x+2) = x^2 + 2x + 6x + 12 = x^2 + 8x + 12$.

$(2x-3)(x+5) = 2x^2 + 10x - 3x - 15 = 2x^2 + 7x - 15$.

$(x-1)(x-9) = x^2 - 9x - x + 9 = x^2 - 10x + 9$.
\end{solution}

\begin{solution}{exo:g9:algebra:2}
$(x+8)^2 = x^2 + 16x + 64$.

$(3x-4)^2 = 9x^2 - 24x + 16$.

$(x+10)(x-10) = x^2 - 100$.

$(5x+2)(5x-2) = 25x^2 - 4$.
\end{solution}

\begin{solution}{exo:g9:algebra:3}
$101^2 = (100+1)^2 = 10000 + 200 + 1 = 10201$.

$59^2 = (60-1)^2 = 3600 - 120 + 1 = 3481$.

$53 \times 47 = (50+3)(50-3) = 2500 - 9 = 2491$.
\end{solution}

\begin{solution}{exo:g9:algebra:4}
$9x + 12 = 3(3x + 4)$.

$x^2 - 64 = (x+8)(x-8)$.

$x^2 + 14x + 49 = (x+7)^2$.

$18x^2 - 6x = 6x(3x - 1)$.

\begin{solution}{exo:g9:algebra:5}
$4x + 7 = 19$: $4x = 12$, so $x = 3$.

$6x - 5 = 2x + 11$: $4x = 16$, so $x = 4$.

$5(x-3) = 3x + 1$: $5x - 15 = 3x + 1$, so $2x = 16$ and $x = 8$.
\end{solution}

\begin{solution}{exo:g9:algebra:6}
$(x+4)(3x-12) = 0$: $x = -4$ or $x = 4$.

$x^2 - 121 = (x-11)(x+11) = 0$: $x = 11$ or $x = -11$.

$(2x-1)^2 = 0$: the single solution $x = \frac12$.

$x^2 = 8x$: $x(x - 8) = 0$, so $x = 0$ or $x = 8$.
\end{solution}

\begin{solution}{exo:g9:algebra:7}
\omterm{ex:g1:subtraction:difference}{Difference} of squares with $a = 3x + 2$ and $b = 5$:
\[
E = (3x + 2)^2 - 5^2 = (3x + 2 + 5)(3x + 2 - 5) = (3x + 7)(3x - 3).
\]
\omterm{def:g2:mult:def}{Zero-product} rule: $3x + 7 = 0$ gives $x = -\frac73$, and $3x - 3 = 0$
gives $x = 1$.
\end{solution}

\begin{solution}{exo:g9:algebra:8}
$3x - 4 \geq 8$: $3x \geq 12$, so $x \geq 4$ (filled dot at $4$, shade
right).

$5 - 2x > 1$: $-2x > -4$, dividing by $-2$ reverses the sign: $x < 2$
(hollow dot at $2$, shade left).

$4(x+1) \leq x - 5$: $4x + 4 \leq x - 5$, so $3x \leq -9$, so

\end{solution}

\begin{solution}{exo:g9:algebra:9}
Let $x$ be the number: $3x + 7 = 5x - 9$. Then $16 = 2x$, so $x = 8$.
Check: $3 \times 8 + 7 = 31$ and $5 \times 8 - 9 = 31$.
\end{solution}

\begin{solution}{exo:g9:algebra:10}
\emph{1.} $(n+1)(n-1) = n^2 - 1$ (third identity). The neighbors of $n$
are $n - 1$ and $n + 1$, so their \omterm{def:g2:mult:def}{product} is $n^2 - 1$: e.g.\
$24 \times 26 = 625 - 1 = 624$.

\emph{2.} Three consecutive integers can be written $n - 1$, $n$,
$n + 1$. Their \omterm{def:g1:addition:def}{sum} is
\[
(n-1) + n + (n+1) = 3n ,
\]
a \omterm{def:g9:arith:divisor}{multiple} of $3$ (three times the middle one).
\end{solution}

\begin{solution}{pb:g9:algebra:1}
\textbf{1.} $(10a + b) - (10b + a) = 9a - 9b = 9(a - b)$: the
\omterm{ex:g1:subtraction:difference}{difference} between a two-digit number and its reversal is always
a \omterm{def:g9:arith:divisor}{multiple} of $9$. Check: $72 - 27 = 45 = 9 \times 5$, and
$a - b = 7 - 2 = 5$.

\textbf{2.} \omterm{def:g9:algebra:expand}{Expanding} the right side:
$9(11a + b) + a + b + c = 99a + 9b + a + b + c
= 100a + 10b + c$. So the number equals a \omterm{def:g9:arith:divisor}{multiple} of $9$ plus
its digit \omterm{def:g1:addition:def}{sum}: dividing by $9$, both leave the same \omterm{def:g3:division:remainder}{remainder}.
In particular \omterm{def:g3:division:remainder}{remainder} $0$ for one means \omterm{def:g3:division:remainder}{remainder} $0$ for the
other: the rule of nine, proved.

\textbf{3.} $9(11a + b)$ is also a \omterm{def:g9:arith:divisor}{multiple} of $3$, so the same
identity shows number and digit \omterm{def:g1:addition:def}{sum} leave the same \omterm{def:g3:division:remainder}{remainder} in
the division by $3$: the rule of three.

\textbf{4.} Write $47 = 9k + 2$ and $86 = 9l + 5$. Then
\[
47 \times 86 = (9k + 2)(9l + 5)
= 9\,(9kl + 5k + 2l) + 10 :
\]
the \omterm{def:g2:mult:def}{product} leaves the same \omterm{def:g3:division:remainder}{remainder} as $2 \times 5 = 10$,
namely $1$. A correct result must therefore also leave \omterm{def:g3:division:remainder}{remainder}
$1$ --- which $4\,042$ does. But the test only sees \omterm{def:g3:division:remainder}{remainders}:
$4\,042$ and $4\,402$ (digits swapped) pass identically, so
passing proves nothing --- failing is what convicts.

\textbf{5.} \omterm{def:g9:algebra:expand}{Expanding}: $11(91a + 9b + c) - a + b - c + d
= 1001a + 99b + 11c - a + b - c + d
= 1000a + 100b + 10c + d$. So the number is \omterm{def:g6:wholes:divisible}{divisible} by $11$
exactly when its alternating \omterm{def:g1:addition:def}{sum} $-a + b - c + d$ is. For
$2\,926$: $-2 + 9 - 2 + 6 = 11$, \omterm{def:g6:wholes:divisible}{divisible} by $11$: yes,
$2\,926 = 11 \times 266$.

\textbf{6.} $742 - 247 = 495$; $495 + 594 = 1\,089$. Any valid
example lands on $1\,089$ too.

\textbf{7.} $(100a + 10b + c) - (100c + 10b + a) = 99(a - c)$.
For $a - c = 2, 3, \dots, 9$:
\[
198,\ 297,\ 396,\ 495,\ 594,\ 693,\ 792,\ 891 .
\]

\textbf{8.} $100(m - 1) + 90 + (10 - m) = 100m - 100 + 90 + 10
- m = 99m$. So the three digits of $99m$ are $m - 1$, then $9$,
then $10 - m$ (check $495$: $m = 5$ gives $4$, $9$, $5$).

\textbf{9.} The reversal of $99m$ has digits $10 - m$, $9$,
$m - 1$, so it equals $100(10 - m) + 90 + (m - 1)$. Adding:
\[
100\,(m - 1 + 10 - m) + 180 + (10 - m + m - 1)
= 100 \times 9 + 180 + 9 = 1\,089 ,
\]
independent of $m$. Whatever the starting number, the magician
is safe.

\textbf{10.} If $a = c$, the subtraction gives $0$ and the trick
dies at once. If $a - c = 1$, it gives $99$, which must be
treated as the three-digit string $099$ (reversal $990$,
$99 + 990 = 1\,089$: saved!). The small print: \omterm{ex:g1:subtraction:difference}{differences} of
$1$ work only if the spectator writes the subtraction result
with three digits, leading \omterm{def:g1:counting:zero}{zero} included --- requiring a gap of
at least $2$ avoids the argument.

\textbf{11.} $2(5m + 7) + d - 14 = 10m + 14 + d - 14 =
10m + d$: the tens (and hundreds, for October--December) show
the month, the units (and tens) the day.

\textbf{12.} $2(5m + 9) + d = 10m + 18 + d$: subtract $18$.

\textbf{13.} By question 2, $n - (\text{digit sum of } n) =
9(11a + b)$ for a three-digit $n$: always a \omterm{def:g9:arith:divisor}{multiple} of $9$
(similar identities work for any length). A \omterm{def:g9:arith:divisor}{multiple} of $9$ has
digit \omterm{def:g1:addition:def}{sum} a \omterm{def:g9:arith:divisor}{multiple} of $9$ (question 2 again), so the digits
the spectator reads out must be completed to the next \omterm{def:g9:arith:divisor}{multiple}
of $9$: the missing digit is that complement. Ambiguity: if the
read digits already sum to a \omterm{def:g9:arith:divisor}{multiple} of $9$, the crossed digit
could be $0$ or $9$ --- excluding $0$ removes the doubt.

\textbf{14.} $(10t + 5)^2 = 100t^2 + 100t + 25 =
100\,t(t + 1) + 25$: write $t(t+1)$, then glue $25$ behind it.
$45^2$: $4 \times 5 = 20$, so $2\,025$. $85^2$:
$8 \times 9 = 72$, so $7\,225$. $105^2$: $10 \times 11 = 110$,
so $11\,025$.

\textbf{15.} Five consecutive: $n + (n+1) + (n+2) + (n+3) +
(n+4) = 5n + 10 = 5(n + 2)$: a \omterm{def:g9:arith:divisor}{multiple} of $5$, always. Four
consecutive: $n + (n+1) + (n+2) + (n+3) = 4n + 6 =
4(n + 1) + 2$: \omterm{def:g3:division:remainder}{remainder} $2$ in the division by $4$, never a
\omterm{def:g9:arith:divisor}{multiple}. Algebra converts ``it seems to work'' into
\emph{always}, and ``I found no example'' into \emph{never} ---
two words no amount of testing can reach
(\cref{met:g7:literal:test}).
\end{solution}
