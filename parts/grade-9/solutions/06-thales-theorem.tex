\section*{Chapter \ref{ch:g9:thales} --- Thales' Theorem}

\begin{solution}{exo:g9:thales:1}
Thales: $\dfrac{AM}{AB} = \dfrac{AN}{AC} = \dfrac{MN}{BC}$, i.e.\
$\dfrac26 = \dfrac13$. So $\dfrac{3}{AC} = \dfrac13$ gives $AC = 9$,
and $\dfrac{MN}{9} = \dfrac13$ gives $MN = 3$.
\end{solution}

\begin{solution}{exo:g9:thales:2}
First find $AB = AM + MB = 5 + 3 = 8$. Then
$\dfrac{AM}{AB} = \dfrac{AN}{AC}$ gives $\dfrac58 = \dfrac{4}{AC}$, so
$5 \times AC = 32$ and $AC = 6.4$.
\end{solution}

\begin{solution}{exo:g9:thales:3}
The butterfly works exactly like the nested configuration:
$\dfrac{AM}{AB} = \dfrac{AN}{AC} = \dfrac{MN}{BC}$, i.e.\
$\dfrac{3}{7.5} = \dfrac25$. So $\dfrac{2}{AC} = \dfrac25$ gives
$AC = 5$, and $\dfrac{2.6}{BC} = \dfrac25$ gives
$BC = \dfrac{2.6 \times 5}{2} = 6.5$.
\end{solution}

\begin{solution}{exo:g9:thales:4}
$\dfrac{AM}{AB} = \dfrac68 = \dfrac34$ and
$\dfrac{AN}{AC} = \dfrac{9}{12} = \dfrac34$: equal ratios, points in the
same order, so $(MN) \parallel (BC)$ by the converse of Thales.
\end{solution}

\begin{solution}{exo:g9:thales:5}
$\dfrac{AM}{AB} = \dfrac46 = \dfrac23$ and
$\dfrac{AN}{AC} = \dfrac58$. Cross-check: $\dfrac23 = \dfrac{16}{24}$
and $\dfrac58 = \dfrac{15}{24}$: the ratios differ. If the lines were
\omterm{def:g6:lines:perp}{parallel}, Thales' theorem would force them to be equal --- so $(MN)$
and $(BC)$ are \emph{not} \omterm{def:g6:lines:perp}{parallel}.
\end{solution}

\begin{solution}{exo:g9:thales:6}
Let $S$ be the tip of the shadow, $P$ the top of the person's head, $L$
the top of the lamp. The person ($1.8$ m at distance $3$ m from $S$) and
the lamp (height $h$ at distance $3 + 2 = 5$ m from $S$) are two
\omterm{def:g6:lines:perp}{parallel} vertical \omterm{def:g6:lines:objects}{segments} cut by the light ray $(SL)$: Thales from the
point $S$ gives
\[
\frac{1.8}{h} = \frac{3}{5},
\qquad\text{so}\quad
h = \frac{1.8 \times 5}{3} = 3 \text{ m}.
\]
\end{solution}

\begin{solution}{exo:g9:thales:7}
\emph{1.} Real distance: $6.8 \times 25\,000 = 170\,000$ cm $=
1.7$ km.

\emph{2.} \omterm{def:g6:measure:area}{Areas} scale by $k^2$: real area $= 8 \times 25\,000^2$
cm$^2$ $= 8 \times 6.25 \times 10^8$ cm$^2$ $= 5 \times 10^9$ cm$^2$.
Since $1$ km$^2 = 10^{10}$ cm$^2$, that is $0.5$ km$^2$.
\end{solution}

\begin{solution}{exo:g9:thales:8}
\emph{1.} The ratio is $\dfrac{AM}{AB} = \dfrac{8}{12} = \dfrac23$.
Thales: $AN = \frac23 \times 15 = 10$ and $MN = \frac23 \times 18 = 12$.

\emph{2.} The scale factor is $k = \frac23$. Perimeters: $ABC$ has
$12 + 15 + 18 = 45$, and $AMN$ has $8 + 10 + 12 = 30 = \frac23 \times
45$: the perimeter scales by $k$ too, like every length.
\end{solution}

\begin{solution}{exo:g9:thales:9}
Around $O$, the lines $(AC)$ and $(BD)$ cross, and
$(AB) \parallel (CD)$: butterfly configuration with the triangles $OAB$
and $OCD$. Thales gives
\[
\frac{OA}{OC} = \frac{OB}{OD} = \frac{AB}{CD} = \frac46 = \frac23 .
\]
So $\dfrac{OA}{OC} = \dfrac23$, and the equality
$\dfrac{OA}{OC} = \dfrac{OB}{OD}$ says precisely that $O$ cuts the two
diagonals in the same ratio.
\end{solution}

\begin{solution}{pb:g9:thales:1}
\textbf{1.} The construction produces two points on $[AB]$;
call them $M_1$ (from $P_1$) and $M_2$ (from $P_2$).

\textbf{2.} In the triangle $A P_3 B$, the parallels to
$(P_3 B)$ through $P_1$ and $P_2$ cut the sides
proportionally (\cref{thm:g9:thales:direct}):
\[
\frac{A M_1}{AB} = \frac{A P_1}{A P_3} = \frac13,
\qquad
\frac{A M_2}{AB} = \frac{A P_2}{A P_3} = \frac23 .
\]
The compass made $A P_1$, $P_1 P_2$, $P_2 P_3$ equal, so the
ratios are exactly thirds --- whatever the angle of the ray and
the compass opening chosen.

\textbf{3.} For fifths: step off five equal lengths, join the
fifth point to $B$, and draw four parallels. For $\frac35$ of a
segment: same figure, and take the point cut by the parallel
through $P_3$: it sits at $\frac{3}{5}$ of $[AB]$ from $A$.

\textbf{4.} $\frac{AM}{MB} = \frac23$ means
$AM = \frac25 AB$: five equal steps on the ray, join $P_5$ to
$B$, and the parallel through $P_2$ marks $M$.

\textbf{5.} Measuring $\sqrt2 = 1.41421\ldots$ cm only ever uses
finitely many decimals: any measured ``third'' is an
approximation. The Thales construction never reads a number: it
delivers the exact point at $\frac{\sqrt2}{3}$ --- equal parts
of a segment no ruler can even express
(\cref{pb:g9:sqrt:1}).

\textbf{6.} The object's top, the hole and the image's bottom
are aligned, and likewise for the object's bottom and the
image's top: two lines crossing at the hole, with the object
and the image wall parallel. Thales in the butterfly:
\[
\frac{h}{H} = \frac{d}{D},
\qquad\text{so}\qquad
h = H \times \frac dD .
\]

\textbf{7.} $h = 12 \times \frac{0.20}{30} = 0.08$ m $= 8$ cm
--- upside down (the rays cross at the hole).

\textbf{8.} $H = h \times \frac Dd = 9 \times 10^{-3} \times
\frac{1.5 \times 10^{11}}{1} = 1.35 \times 10^{9}$ m. True
value $1.39 \times 10^9$ m: a shoebox measures the Sun to
within a few percent.

\textbf{9.} Moon: $\frac{3.5 \times 10^6}{3.8 \times 10^8}
\approx 0.0092$. Sun: $\frac{1.39 \times 10^9}{1.5 \times
10^{11}} \approx 0.0093$. The two ratios --- the apparent sizes
in the sky --- are almost identical: the Moon can cover the Sun
\emph{exactly}, rim to rim. That cosmic fluke is the total
solar eclipse.

\textbf{10.} Every ray must pass through the one hole, so rays
from the top of the object continue \emph{down} and rays from
the bottom continue \emph{up}: the image is inverted. Enlarging
the hole lets through a whole bundle of slightly shifted copies
of the image, which overlap: brighter, but smeared.

\textbf{11.} $\frac{PQ}{MN} = \frac{OP}{OM} = \frac{7.5}{3}
= 2.5$, so $PQ = 10$; and $\frac{ON}{OQ} = \frac{OM}{OP} =
\frac{1}{2.5}$, so $ON = \frac{6}{2.5} = 2.4$.

\textbf{12.} If the parallel passes through the midpoint of one
side, the Thales ratio is $\frac12$, so it cuts the second side
at \emph{its} midpoint, and the parallel segment measures half
the base: precisely \cref{thm:g8:midpoints:direct}. Thales is
the midpoint theorem freed from the ratio $\frac12$.

\textbf{13.} In the triangle $ACD$, the line $(EO)$ is parallel
to the base $(CD)$, with $\frac{AO}{AC} = \frac{OA}{OA + OC} =
\frac{4}{4 + 6} = \frac25$. Thales:
$\frac{EO}{DC} = \frac{AO}{AC} = \frac25$, so
$EO = \frac25 \times 6 = 2.4$.

\textbf{14.} In the triangle $BDC$, likewise
$\frac{BO}{BD} = \frac25$ and $OF = \frac25 \times 6 = 2.4$.
Hence
\[
EF = EO + OF = 4.8
= \frac{2 \times 4 \times 6}{4 + 6} :
\]
the harmonic mean of the bases --- the mean of round trips
(\cref{pb:g8:speed:1}), drawn in a trapezoid.

\textbf{15.} Arithmetic: $5$; geometric:
$\sqrt{24} \approx 4.90$; harmonic: $4.8$. Order:
$4.8 < 4.90\ldots < 5$, i.e.\ harmonic $<$ geometric $<$
arithmetic --- with equality only for equal bases. In the
figure: the midline measures $5$, the parallel through the
diagonal crossing $4.8$, and the similarity cut
$\sqrt{24}$, all stacked between the bases in that order. The
inequalities were proved in \cref{pb:g8:cosine:1} (geometric
$\leq$ arithmetic, by circle and by identity) and
\cref{pb:g8:speed:1} (harmonic $\leq$ arithmetic); harmonic
$\leq$ geometric follows by combining them ($H \times A = G^2$,
proved there too).
\end{solution}
