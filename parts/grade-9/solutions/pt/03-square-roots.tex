\section*{Capítulo \ref{ch:g9:sqrt} --- Raízes Quadradas}

\begin{solution}{exo:g9:sqrt:1}
$\sqrt{64} = 8$; \quad $\sqrt{100} = 10$; \quad
$\sqrt{0.09} = 0.3$ (desde $0.3^2 = 0.09$); \quad
$\sqrt{\frac{1}{25}} = \frac15$; \quad
$\left(\sqrt{13}\right)^2 = 13$; \quad
$\sqrt{(-4)^2} = \sqrt{16} = 4 = \abs{-4}$.
\end{solution}

\begin{solution}{exo:g9:sqrt:2}
$\sqrt{50} = \sqrt{25 \times 2} = 5\sqrt2$; \quad
$\sqrt{45} = \sqrt{9 \times 5} = 3\sqrt5$; \quad
$\sqrt{98} = \sqrt{49 \times 2} = 7\sqrt2$; \quad
$\sqrt{300} = \sqrt{100 \times 3} = 10\sqrt3$.
\end{solution}

\begin{solution}{exo:g9:sqrt:3}
$\sqrt2 \times \sqrt{18} = \sqrt{36} = 6$.

$\dfrac{\sqrt{75}}{\sqrt3} = \sqrt{\dfrac{75}{3}} = \sqrt{25} = 5$.

$\sqrt5 \times \sqrt{20} = \sqrt{100} = 10$.

$\dfrac{\sqrt8}{\sqrt{50}} = \sqrt{\dfrac{8}{50}} = \sqrt{\dfrac{4}{25}}
= \dfrac25$.
\end{solution}

\begin{solution}{exo:g9:sqrt:4}
$ x^2 = 36$: $ x = 6$ ou $ x = -6$.

$ x^2 = 11$: $ x = \sqrt{11}$ ou $ x = -\sqrt{11}$.

$ x^2 + 4 = 0$ significa $ x^2 = -4 < 0$: nenhuma solução.

$2x^2 = 50$: $ x^2 = 25$, então $ x = 5$ ou $ x = -5$.
\end{solution}

\begin{solution}{exo:g9:sqrt:5}
$\sqrt{20} + \sqrt{45} = 2\sqrt5 + 3\sqrt5 = 5\sqrt5$.

$3\sqrt8 - \sqrt{18} + \sqrt2 = 3 \times 2\sqrt2 - 3\sqrt2 + \sqrt2
= (6 - 3 + 1)\sqrt2 = 4\sqrt2$.
\end{solution}

\begin{solution}{exo:g9:sqrt:6}
$\left(\sqrt3 + 1\right)^2 = 3 + 2\sqrt3 + 1 = 4 + 2\sqrt3$.

$\left(2\sqrt5 - 3\right)^2 = 4 \times 5 - 2 \times 3 \times 2\sqrt5 + 9
= 29 - 12\sqrt5$.

$\left(\sqrt6 + \sqrt2\right)\left(\sqrt6 - \sqrt2\right) = 6 - 2 = 4$.
\end{solution}

\begin{solution}{exo:g9:sqrt:7}
Lado: $\sqrt{6400} = 80$ m. Diagonal (Pitágoras):
$\sqrt{80^2 + 80^2} = 80\sqrt2 \approx 113$ m.
\end{solution}

\begin{solution}{exo:g9:sqrt:8}
$\dfrac{1}{\sqrt2} = \dfrac{1 \times \sqrt2}{\sqrt2 \times \sqrt2}
= \dfrac{\sqrt2}{2}$.

$\dfrac{6}{\sqrt3} = \dfrac{6\sqrt3}{3} = 2\sqrt3$.
\quad
$\dfrac{10}{\sqrt5} = \dfrac{10\sqrt5}{5} = 2\sqrt5$.
\end{solution}

\begin{solution}{exo:g9:sqrt:9}
\emph{1. Verdadeiro}: este é o \omterm{def:g2:mult:def}{produto} regra (\cref{thm:g9:sqrt:rules}).

\emph{2. Falso}: $\sqrt{9 + 16} = 5$ mas $\sqrt9 + \sqrt{16} = 7$.

\emph{3. Falso} para negativo $ x $: $\sqrt{(-4)^2} = 4 \neq -4$. O
afirmação correta é $\sqrt{x^2} = \abs{x}$.

\emph{4. Verdadeiro}: $\left(3\sqrt2\right)^2 = 9 \times 2 = 18$.
\end{solution}

\begin{solution}{exo:g9:sqrt:10}
$\left(2 + \sqrt3\right)^2 = 4 + 4\sqrt3 + 3 = 7 + 4\sqrt3$. Então
$ x = \sqrt{7 + 4\sqrt3} = \sqrt{\left(2+\sqrt3\right)^2} = 2 + \sqrt3$
(um número não negativo, então a raiz apenas remove o quadrado).

De forma similar $\left(\sqrt5 - 2\right)^2 = 5 - 4\sqrt5 + 4 = 9 - 4\sqrt5$,
e $\sqrt5 - 2 > 0$, então
$\sqrt{9 - 4\sqrt5} = \sqrt5 - 2$ (não $2 - \sqrt5$, que é
negativo!).
\end{solution}

\begin{solution}{pb:g9:sqrt:1}
\textbf{1.} $1.4^2 = 1.96 < 2 < 2.25 = 1.5^2$;
$1.41^2 = 1.9881 < 2 < 2.0164 = 1.42^2$. Três decimais:
$1.414^2 = 1.999396 < 2 < 2.002225 = 1.415^2$, então
$1.414 < \sqrt2 < 1.415$.

\textbf{2.} Quadratura $\sqrt2 = \frac ab $ dá
$2 = \frac{a^2}{b^2}$, por isso $ a^2 = 2b^2$: o número $ a^2$ é
duas vezes um número inteiro --- \emph{\omterm{def:g3:numbers:evenodd}{par}}.

\textbf{3.} Se $ a $ eram \omterm{def:g3:numbers:evenodd}{ímpar}, $ a^2$ seria \omterm{def:g3:numbers:evenodd}{ímpar}
(\cref{pb:g8:pythagoras:1}); desde $ a^2$ é \omterm{def:g3:numbers:evenodd}{par}, $ a $ é \omterm{def:g3:numbers:evenodd}{par}:
$ a = 2k $. Substituindo: $4k^2 = 2b^2$, então $ b^2 = 2k^2$ é \omterm{def:g3:numbers:evenodd}{par},
e pelo mesmo fato de paridade $ b $ é \omterm{def:g3:numbers:evenodd}{par}.

\textbf{4.} Ambos $ a $ e $ b $ \omterm{def:g3:numbers:evenodd}{par} significa ambos \omterm{def:g6:wholes:divisible}{divisível} por $2$
--- mas $\frac ab $ estava em termos mais baixos, não compartilhando nenhum
\omterm{def:g5:division:multiple}{divisor}. Contradição: o suposto \omterm{def:g9:fractions:fraction}{fração} não pode existir.
\emph{Teorema: $\sqrt2$ é irracional.}

\textbf{5.} A contradição reside inteiramente nos “mais baixos
termos'': sem ele, ``$ a $ e $ b $ ambos \omterm{def:g3:numbers:evenodd}{par}''contradiz
nada. Assumir os termos mais baixos é legítimo porque cada
\omterm{def:g9:fractions:fraction}{fração} \emph{tem} uma forma de termos mais baixos
(\cref{met:g9:arith:simplify}: divida o GCD) --- se
$\sqrt2$ havia algum \omterm{def:g9:fractions:fraction}{fração} de forma alguma, seria uma situação de termos mais baixos
um, e esse é impossível.

\textbf{6.} Nas fatorações primárias, os quadrados carregam \omterm{def:g3:numbers:evenodd}{par}
\omterm{def:g8:powers:def}{expoentes} (\cref{pb:g9:arith:1}). Em $ a^2 = 3b^2$, o \omterm{def:g8:powers:def}{expoente}
de $3$ é \omterm{def:g3:numbers:evenodd}{par} à esquerda, mas \omterm{def:g3:numbers:evenodd}{ímpar} à direita (\omterm{def:g3:numbers:evenodd}{par} de
$ b^2$, mais um). Nenhum número tem duas fatorações
(\cref{thm:g9:arith:factorization}): contradição e
$\sqrt3$ é irracional.

\textbf{7.} Em $ a^2 = n b^2$, o argumento encontra um \omterm{def:g9:arith:prime}{primo} com
contraditório \omterm{def:g8:powers:def}{expoente} paridade exatamente quando algum \omterm{def:g9:arith:prime}{primo} aparece em
$ n $ com um \emph{\omterm{def:g3:numbers:evenodd}{ímpar}} \omterm{def:g8:powers:def}{expoente}. Se cada \omterm{def:g8:powers:def}{expoente} de $ n $ é
\omterm{def:g3:numbers:evenodd}{par}, $ n $ é um quadrado perfeito ($ n = m^2$, e
$\sqrt n = m $ está inteiro). Resultado completo: $\sqrt n $ é
irracional para todo todo $ n $ isso não é um quadrado perfeito ---
$\sqrt2, \sqrt3, \sqrt5, \sqrt6, \sqrt7, \sqrt8, \sqrt{10},
\dots $

\textbf{8.} De $ r x = q $ com $ r \neq 0$:
$ x = \frac qr $, um \omterm{def:g3:division:remainder}{quociente} de racionais, portanto racionais. Então se
$ x $ é irracional, $ r x $ não pode ser racional: $2\sqrt2$ e
$\frac{\sqrt2}{2}$ são irracionais (multiplicadores $2$ e
$\frac12$).

\textbf{9.} Se $1 + \sqrt2 = q $ eram racionais, então
$\sqrt2 = q - 1$ seria racional: contradição. Se
$ s = \sqrt2 + \sqrt3$ eram racionais, então
$ s^2 = 2 + 2\sqrt6 + 3 = 5 + 2\sqrt6$ seria também, dando
$\sqrt6 = \frac{s^2 - 5}{2}$ racional --- mas $\sqrt6$ é
irracional (questão 7). Então $\sqrt2 + \sqrt3$ é irracional.

\textbf{10.} \omterm{def:g1:addition:def}{Somas}: $\sqrt2$ e $-\sqrt2$ (ou $\sqrt2$ e
$1 - \sqrt2$) são cada um irracional, com racional \omterm{def:g1:addition:def}{somas} $0$ e
$1$. \omterm{def:g2:mult:def}{Produtos}: $\sqrt2 \times \sqrt2 = 2$. A irracionalidade pode
evaporar em aritmética --- daí as provas caso a caso.

\textbf{11.} De $ x = 1$: média de $1$ e $\frac21 = 2$:
$ x_1 = \frac32$. Então média de $\frac32$ e
$\frac{2}{3/2} = \frac43$:
$ x_2 = \frac12\left(\frac32 + \frac43\right)
= \frac12 \cdot \frac{17}{6} = \frac{17}{12} \approx 1.4167$.

\textbf{12.} $ x_3 = \frac12\left(\frac{17}{12} +
\frac{24}{17}\right) = \frac12 \cdot
\frac{289 + 288}{204} = \frac{577}{408} = 1.4142156\ldots $
Contra $\sqrt2 = 1.4142135\ldots $: cinco casas decimais corretas depois
três etapas --- a receita quase dobra os decimais corretos
cada redondo.

\textbf{13.} Se $ x^2 > 2$ (acho que muito grande), então $ x > \sqrt2$,
e $\frac2x < \frac{2}{\sqrt2} = \sqrt2$: o lado companheiro é
muito pequeno. Se $ x^2 < 2$, as desigualdades se invertem. De qualquer jeito
$\sqrt2$ senta-se entre $ x $ e $\frac2x $, então a média deles ---
o próximo palpite --- está mais próximo do que o pior dos dois lados:
o retângulo de \omterm{def:g6:measure:area}{área} $2$ fica mais quadrado a cada passo.

\textbf{14.} $9 - 8 = 1$; \ $289 - 288 = 1$; \
$332\,929 - 332\,928 = 1$. Cada garça adivinha $\frac ab $
satisfaz $ a^2 - 2b^2 = 1$: Parte I provou que bater $0$ é
impossível, e estes frações perder o impossível exatamente
uma unidade --- os números inteiros mais próximos que podem chegar.

\textbf{15.} A reta numérica deve conter números além do
frações --- comprimentos como $\sqrt2$, cujas escritas decimais são executadas
para sempre sem repetir: o \emph{números reais}. Seu
construção honesta é uma história para o \omterm{def:g6:measure:volume}{volume} do Ensino Médio e,
com todo rigor, os universitários.
\end{solution}
