\begin{solution}{exo:g9:fractions:1}
$\dfrac13 + \dfrac15 = \dfrac{5}{15} + \dfrac{3}{15} = \dfrac{8}{15}$.

$\dfrac72 - \dfrac53 = \dfrac{21}{6} - \dfrac{10}{6} = \dfrac{11}{6}$.

$\dfrac{5}{12} + \dfrac38 = \dfrac{10}{24} + \dfrac{9}{24} =

\begin{solution}{exo:g9:fractions:2}
$\dfrac59 \times \dfrac{27}{35} = \dfrac{5 \times 27}{9 \times 35}
= \dfrac{27}{9} \times \dfrac{5}{35} = 3 \times \dfrac17 = \dfrac37$
(simplificar por $9$ e por $5$).

$\dfrac{8}{15} \div \dfrac45 = \dfrac{8}{15} \times \dfrac54
= \dfrac{8 \times 5}{15 \times 4} = \dfrac{40}{60} = \dfrac23$.

$\dfrac23 \times \dfrac34 \times \dfrac45 = \dfrac{2 \times 3 \times 4}
{3 \times 4 \times 5} = \dfrac25$ (o $3$ areia $4$ é cancelar).
\end{solution}

\begin{solution}{exo:g9:fractions:10}
Escreva ambos os números como \omterm{def:g9:fractions:power}{poderes} com \omterm{def:g8:powers:def}{expoente} $10$:
\[
2^{30} = \left(2^3\right)^{10} = 8^{10},
\qquad
3^{20} = \left(3^2\right)^{10} = 9^{10}.
\]
Desde $8 < 9$, multiplicando dez cópias de cada: $8^{10} < 9^{10}$, então
$2^{30} < 3^{20}$.
\end{solution}

\begin{solution}{pb:g9:fractions:1}
\textbf{1.} $\frac14 = 0.25$ (para); $\frac13 = 0.333\ldots $
(bloquear $3$); $\frac56 = 0.8333\ldots $ (bloquear $3$ depois do
$8$); $\frac{3}{11} = 0.272727\ldots $ (bloquear $27$).

\textbf{2.} Em cada etapa da divisão por $ b $, o \omterm{def:g3:division:remainder}{restante}
é um dos $ b $ valores $0, 1, \dots, b - 1$
(\cref{thm:g6:wholes:division}). Se um \omterm{def:g3:division:remainder}{restante} $0$ aparece, o
a divisão pára. Caso contrário, depois de no máximo $ b $ passos alguns
\omterm{def:g3:division:remainder}{restante} deve aparecer uma segunda vez --- há mais etapas
do que possível restos --- e a partir desse momento a divisão
repete exatamente a sequência de dígitos que produziu após o
primeira aparição: a escrita decimal gira para sempre. Pare ou
repita: não existe um terceiro comportamento.

\textbf{3.} $\frac17 = 0.142857\,142857\ldots $: o bloco
$142857$ tem comprimento $6$. A pergunta 2 prometia uma repetição dentro
no máximo $7$ passos --- aqui o restos
$1, 3, 2, 6, 4, 5$ todos aparecem diante do $1$ retorna: o
canto mais longo possível para uma divisão por $7$.

\textbf{4.} $\frac ab $ pára exatamente quando algum
$\frac{a \times k}{b \times k}$ tem \omterm{def:g4:fractions:def}{denominador} $10$, $100$,
$1000, \dots $ --- isto é, quando o \omterm{def:g4:fractions:def}{denominador} pode ser
multiplicado até um \omterm{def:g9:fractions:power}{poder} de dez, como $4 \times 25 = 100$ ou
$8 \times 125 = 1\,000$. Para $3$ e $6$ isso é impossível: um
\omterm{def:g9:fractions:power}{poder} de dez tem dígito \omterm{def:g1:addition:def}{soma} $1$, e cada múltiplo de $3$ tem um
dígito \omterm{def:g1:addition:def}{soma} isso é um múltiplo de $3$ --- então não múltiplo de $3$
(portanto, nenhum $6$) é sempre $10$, $100$, $1\,000, \dots $ Para
$11$: dividindo $10$, $100$, $1\,000$, $10\,000$ por $11$ sempre
deixa um \omterm{def:g3:division:remainder}{restante} ($10$, $1$, $10$, $1, \dots $), nunca $0$.
\omterm{def:g4:fractions:def}{Denominadores} que atingem um \omterm{def:g9:fractions:power}{poder} de dez paradas; todos os outros repetem.

\textbf{5.} $\frac{9}{40}$: $40 \times 25 = 1\,000$, para:
$\frac{225}{1000} = 0.225$. $\frac{33}{50}$:
$50 \times 2 = 100$, para: $\frac{66}{100} = 0.66$.
$\frac{5}{12}$: $12$ é um múltiplo de $3$, repete
($0.41666\ldots $).

\textbf{6.} $10x = 7.777\ldots $, então $10x - x = 7$ (as caudas
cancelar perfeitamente), $9x = 7$ e $ x = \frac79$. Verificação de divisão:
$7 \div 9 = 0.777\ldots $

\textbf{7.} O bloco tem dois dígitos, então mude por $100$:
$100x - x = 72.7272\ldots - 0.7272\ldots = 72$, por isso
$99x = 72$ e $ x = \frac{72}{99} = \frac{8}{11}$.

\textbf{8.} $1000x = 583.333\ldots $ e $100x = 58.333\ldots $;
subtraindo, $900x = 525$, então
$ x = \frac{525}{900} = \frac{7}{12}$ (dividir por $75$). De fato
$\frac{7}{12} = 0.58333\ldots $

\textbf{9.} $10x - x = 9.999\ldots - 0.999\ldots = 9$, então
$9x = 9$ e $ x = 1$. Não ``logo abaixo'' $1$:
$0.999\ldots $ \emph{é} o número $1$, escrito em um segundo
traje. Qualquer número estritamente abaixo $1$ deixa uma lacuna e
\cref{pb:g6:decimals:1} mostrou que uma lacuna sempre contém mais
números --- enquanto nada cabe entre $0.999\ldots $ e $1$.
Os intermináveis noves são o chocolate \omterm{def:g3:division:remainder}{restante} de
\cref{pb:g6:fractions:1} reduzido abaixo de cada valor positivo:
nada resta.

\textbf{10.} $2.4545\ldots = 2 + \frac{45}{99} = 2 +
\frac{5}{11} = \frac{27}{11}$. E
$0.123123\ldots = \frac{123}{999} = \frac{41}{333}$ (dividir por
$3$).

\textbf{11.} Dicionário: $0.037037\ldots = \frac{37}{999}$.
Desde $999 = 27 \times 37$, isso simplifica para $\frac{1}{27}$:
o \omterm{def:g9:fractions:fraction}{fração} $\frac{1}{27}$ canta ``$037$''. Verificação de divisão:
$1 \div 27 = 0.037037\ldots $

\textbf{12.} $0.0272727\ldots = \frac{1}{10} \times
0.272727\ldots = \frac{1}{10} \times \frac{27}{99}
= \frac{27}{990} = \frac{3}{110}$ (dividir por $9$).

\textbf{13.} Os blocos de \omterm{def:g1:counting:zero}{zeros} crescer sem fim, então não há fixo
bloco pode repetir para sempre: é \emph{não} uma dízima repetida
(e nunca para). Na Parte I, cada \omterm{def:g9:fractions:fraction}{fração} pára ou
repete --- então esse número não é \omterm{def:g9:fractions:fraction}{fração} de jeito nenhum: é
irracional, construído sob encomenda. A estrela da irracionalidade,
$\sqrt2$, é desmascarado em \cref{ch:g9:sqrt}.

\textbf{14.} $2^{30} = \left(2^{10}\right)^3 \approx
(1.024)^3 \times 10^9 \approx 1.07 \times 10^9$: um número
um pouco acima $10^9$, portanto com $10$ dígitos. (Exatamente:
$2^{30} = 1\,073\,741\,824$.)

\textbf{15.} $0.20262026\ldots = \frac{2026}{9999}$, já em
termos mais baixos ($9999 = 3 \times 3 \times 11 \times 101$ ações
nenhum fator com $2026 = 2 \times 1013$). O concluído
dicionário: \emph{o frações são exatamente os decimais que
pare ou repita} --- \omterm{def:g4:fractions:def}{denominadores} de dezenas de amigos param, todos
outros cantam um bloco; e inversamente cada canto, \omterm{def:g3:numbers:evenodd}{par} seu
ano de formatura, é um \omterm{def:g9:fractions:fraction}{fração} disfarçado.
\end{solution}
