\section*{Capítulo \ref{ch:g9:linfunc} --- Funções Lineares e Afins}

\begin{solution}{exo:g9:linfunc:1}
Linear: $f $ e $ k $ (forma $ ax $). Afim, mas não linear: $ g $ (\omterm{prop:g9:linfunc:affinegraph}{declive} $3$,
interceptar $-5$) e $ m $ (constante, \omterm{prop:g9:linfunc:affinegraph}{declive} $0$). Nenhum: $ h $, por causa
a praça.
\end{solution}

\begin{solution}{exo:g9:linfunc:2}
$ f(0) = -3$; $ f(4) = 5$; $ f(-1) = -5$. E $ f(x) = 9$ significa
$2x - 3 = 9$, então $2x = 12$ e $ x = 6$.
\end{solution}

\begin{solution}{exo:g9:linfunc:3}
$ a = \frac{15}{6} = 2.5$, então $ f(x) = 2.5x $. Então $ f(10) = 25$ e
$ f(-2) = -5$.
\end{solution}

\begin{solution}{exo:g9:linfunc:4}
$ f $ passa pela origem com \omterm{prop:g9:linfunc:affinegraph}{declive} $2$; $ g $ tem o mesmo \omterm{prop:g9:linfunc:affinegraph}{declive} mas
atravessa o $ y $-eixo em $3$; $ h $ diminui de $(0, 4)$ com \omterm{prop:g9:linfunc:affinegraph}{declive}
$-1$. Os gráficos de $ f $ e $ g $ são \emph{\omterm{def:g6:lines:objects}{linhas} paralelas} (mesmo \omterm{prop:g9:linfunc:affinegraph}{declive},
interceptações diferentes).
\end{solution}

\begin{solution}{exo:g9:linfunc:5}
Para $ f $: \omterm{prop:g9:linfunc:affinegraph}{declive} $\frac{11 - 5}{3 - 1} = 3$, então $5 = 3 \times 1 + b $
dá $ b = 2$: $ f(x) = 3x + 2$.

Para $ g $: \omterm{prop:g9:linfunc:affinegraph}{declive} $\frac{0 - 4}{2 - 0} = -2$, e $ g(0) = 4$ já é o
interceptar: $ g(x) = -2x + 4$.
\end{solution}

\begin{solution}{exo:g9:linfunc:6}
Aumento de $25\%$: $120 \times 1.25 = 150$.

Desconto de $40\%$: $65 \times 0.6 = 39$.

Preço original $ p $ com $ p \times 0.75 = 69$:
$ p = \frac{69}{0.75} = 92$.
\end{solution}

\begin{solution}{exo:g9:linfunc:7}
\emph{1.} $ f(x) = 0.05x + 10$.

\emph{2.} O Plano B é mais barato quando $25 < 0.05x + 10$, ou seja\
$15 < 0.05x $, ou seja\ $ x > 300$. De $301$ minutos ($5$ horas) por mês
em diante, o plano B vence.
\end{solution}

\begin{solution}{exo:g9:linfunc:8}
\emph{1.} Depois de uma hora: $2000 \times 1.1 = 2200$. Depois de duas horas:
$2200 \times 1.1 = 2420$.

\emph{2.} O segundo aumento aplica-se a $2200$, não para $2000$: crescendo
por $10\%$ duas vezes multiplica por $1.1^2 = 1.21$, um $21\%$ aumentar, não
$20\%$.
\end{solution}

\begin{solution}{exo:g9:linfunc:9}
\emph{1.} \omterm{prop:g9:linfunc:affinegraph}{Declive}: $\frac{1 - 3}{6 - 2} = \frac{-2}{4} = -0.5$: o
função está diminuindo.

\emph{2.} $ f(x) = -0.5x + b $ com $ f(2) = 3$: $-1 + b = 3$, então $ b = 4$
e $ f(x) = -0.5x + 4$. Saída $0$: $-0.5x + 4 = 0$ dá $ x = 8$.
\end{solution}

\begin{solution}{exo:g9:linfunc:10}
\emph{1.} As duas mudanças multiplicam o preço por
\[
\left(1 + \frac{t}{100}\right)\left(1 - \frac{t}{100}\right)
= 1 - \frac{t^2}{10000}
\]
(terceira identidade com $ a = 1$, $ b = \frac{t}{100}$).

\emph{2.} Para $0 < t \leq 100$, $\frac{t^2}{10000} > 0$, então o fator
é menor que $1$: o preço final é inferior ao original. Para
$ t = 10$: fator $1 - \frac{100}{10000} = 0.99$, ou seja,\ um total
diminuição de $1\%$.
\end{solution}

\begin{solution}{pb:g9:linfunc:1}
\textbf{1.} \omterm{prop:g9:linfunc:affinegraph}{Declive}: $\frac{f(100) - f(0)}{100 - 0} =
\frac{212 - 32}{100} = 1.8$; interceptar $ f(0) = 32$. Por isso
\[
f(C) = 1.8\,C + 32 .
\]

\textbf{2.} $ f(20) = 68\,^\circ $ F (o famoso dia ameno);
$ f(37) = 98.6\,^\circ $ F; $ f(-10) = 14\,^\circ $ F.

\textbf{3.} De $ F = 1.8C + 32$:
$ C = \frac{F - 32}{1.8}$. Então $50\,^\circ $ F
$\to \frac{18}{1.8} = 10\,^\circ $ C, e $98.6\,^\circ $ F
$\to \frac{66.6}{1.8} = 37\,^\circ $ C.

\textbf{4.} $ f(0) = 32 \neq 0$, e $ f(20) = 68$ não é o
o dobro de $ f(10) = 50$: não linear. O gráfico é uma reta
\omterm{def:g6:lines:objects}{linha} faltando a origem: $ f $ é \emph{afim}, não linear.

\textbf{5.} $1.8x + 32 = x $ dá $0.8x = -32$, então $ x = -40$:
a quarenta abaixo, ambos os termômetros marcam $-40$. No gráfico, o
\omterm{def:g6:lines:objects}{linha} de $ f $ cruza a \omterm{def:g6:lines:objects}{linha} diagonal $ y = x $ exatamente lá.

\textbf{6.} $ p_A(k) = 1.5k + 3$ e $ p_B(k) = 2k $. Para $4$ quilômetros:
$9$ euros contra $8$: B mais barato. Para $8$ quilômetros: $15$ contra
$16$: Um mais barato.

\textbf{7.} $1.5k + 3 = 2k $ dá $ k = 6$: a seis quilômetros
ambos custam $12$ euros. Graficamente, a \omterm{def:g6:lines:objects}{linha} de B (através do
origem, mais íngreme) começa abaixo de A e cruza-o em
$(6, 12)$: B ganha corridas curtas, A ganha corridas longas.

\textbf{8.} $ N = 7 \times 20 - 30 = 110$ chilreios por minuto. De
$82 = 7T - 30$: $ T = \frac{112}{7} = 16\,^\circ $ C.

\textbf{9.} $ v(t) = 600 - 15t $. Valor $240$:
$600 - 15t = 240$, $ t = 24$ meses. Valor $0$: $ t = 40$ meses.
Além $40$ meses a fórmula dá negativo --- mas um telefone
o valor pára em \omterm{def:g1:counting:zero}{zero}: todo modelo afim possui um domínio onde
faz sentido.

\textbf{10.} Composição: $\times 1.25$ então $\times 0.8$
multiplica por $1.25 \times 0.8 = 1$: o preço retorna exatamente
para o seu início. Não é milagre: as porcentagens atuam de forma diferente
referências (o corte aplica-se ao preço aumentado), e aqui o
multiplicadores cancelam --- a história geral, com seu
perda embutida $1 - \frac{t^2}{10\,000}$, é
\cref{exo:g9:linfunc:10}.

\textbf{11.} \omterm{prop:g9:linfunc:affinegraph}{Declive}: $\frac{17 - 20}{30} = -0.1$ cm por minuto;
$ h(t) = 20 - 0.1t $. Ele morre em $ h(t) = 0$: $ t = 200$ minutos
--- três horas e vinte minutos de luz.

\textbf{12.} O segundo começou $10$ km à frente (interceptação); o
primeiro é mais rápido ($24 > 18$ km/h). Alcançar:
$24t = 10 + 18t $ dá $6t = 10$, $ t = \frac53$ h $= 1$ h $40$
min, em quilômetro $24 \times \frac53 = 40$.

\textbf{13.} $ f(x) + g(x) = (2 - 0.5)x + (3 + 7) = 1.5x + 10$:
afim, com \omterm{prop:g9:linfunc:affinegraph}{declive} o \emph{\omterm{def:g1:addition:def}{soma} do encostas} e interceptar
o \omterm{def:g1:addition:def}{soma} das interceptações --- livre da coleta de termos semelhantes
em $(mx + p) + (m'x + p')$.

\textbf{14.} Se $ m \neq 0$: exatamente uma solução,
$ x = -\frac pm $ --- uma \omterm{def:g6:lines:objects}{linha} não horizontal cruza o eixo
uma vez. Se $ m = 0$, $ p \neq 0$: nenhuma solução --- uma \omterm{def:g6:lines:objects}{linha} horizontal
acima ou abaixo do eixo nunca o toca. Se $ m = 0$, $ p = 0$:
todo $ x $ é uma solução --- a \omterm{def:g6:lines:objects}{linha} \emph{é} o eixo.

\textbf{15.} Primeiro triplo: taxas $\frac{9 - 5}{3 - 1} = 2$ e
$\frac{17 - 9}{7 - 3} = 2$: iguais, os pontos estão alinhados (em
$ y = 2x + 3$). Segundo triplo: $\frac{9-5}{2} = 2$ mas
$\frac{16 - 9}{3} = \frac73 \neq 2$: não alinhado. Afim
funções são exatamente aqueles com uma taxa de mudança constante ---
um número, o \omterm{prop:g9:linfunc:affinegraph}{declive}, conta toda a história; para curvas o
a própria taxa muda de ponto a ponto, e persegui-la leva
à derivada, no \omterm{def:g6:measure:volume}{volume} Ensino Médio.
\end{solution}
