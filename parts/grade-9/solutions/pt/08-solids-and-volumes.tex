\section*{Capítulo \ref{ch:g9:solids} --- Sólidos e Volumes}

\begin{solution}{exo:g9:solids:1}
\omterm{def:g5:solids:def}{Caixa}: $V = 4 \times 5 \times 12 = 240$.

\omterm{def:g7:areas:prism}{Cilindro}: $ V = \pi r^2 h = \pi \times 9 \times 7 = 63\pi \approx 198$.
\end{solution}

\begin{solution}{exo:g9:solids:2}
\omterm{def:g8:solids:def}{Cone}: $ V = \frac13 \pi r^2 h = \frac13 \pi \times 25 \times 12 =
100\pi \approx 314$.

\omterm{def:g8:solids:def}{Pirâmide}: básico \omterm{def:g6:measure:area}{área} $ B = 8 \times 3 = 24$, então
$ V = \frac13 \times 24 \times 10 = 80$.
\end{solution}

\begin{solution}{exo:g9:solids:3}
\omterm{def:g6:measure:volume}{Volume}: $ V = \frac43 \pi \times 6^3 = \frac43 \pi \times 216 = 288\pi $.

Superfície \omterm{def:g6:measure:area}{área}: $4\pi \times 6^2 = 144\pi $.
\end{solution}

\begin{solution}{exo:g9:solids:4}
\omterm{def:g3:shapes:shapes}{Raio} da seção: $\sqrt{r^2 - d^2} = \sqrt{100 - 64} = \sqrt{36}
= 6$.
\end{solution}

\begin{solution}{exo:g9:solids:5}
\omterm{def:g9:thales:scaling}{Fator de escala} do todo \omterm{def:g8:solids:def}{cone} para o pequeno:
$ k = \frac{9}{12} = \frac34$. Seção \omterm{def:g3:shapes:shapes}{raio}: $8 \times \frac34 = 6$.

\omterm{def:g6:measure:volume}{Volume} do todo \omterm{def:g8:solids:def}{cone}: $\frac13\pi \times 64 \times 12 = 256\pi $.
\omterm{def:g6:measure:volume}{Volume} do pequeno \omterm{def:g8:solids:def}{cone}: $256\pi \times \left(\frac34\right)^3 =
256\pi \times \frac{27}{64} = 108\pi \approx 339$.
\end{solution}

\begin{solution}{exo:g9:solids:6}
\omterm{def:g6:measure:volume}{Volume} da bola: $\frac43\pi \times 2^3 = \frac{32\pi}{3}$ cm $^3$.
Esse \omterm{def:g6:measure:volume}{volume} se espalha pela \omterm{def:g8:powers:def}{base} \omterm{def:g6:measure:area}{área} $\pi \times 3^2 = 9\pi $ cm $^2$,
então o nível sobe em
\[
\frac{32\pi/3}{9\pi} = \frac{32}{27} \approx 1.2 \text{ cm}
\]
(sobre $12$ milímetros).
\end{solution}

\begin{solution}{exo:g9:solids:7}
O molde pequeno é a redução do grande por $ k = \frac36 =
\frac12$, então é \omterm{def:g6:measure:volume}{volume} é $\left(\frac12\right)^3 = \frac18$ do
grande: a receita preenche $8$ pequenos moldes.
\end{solution}

\begin{solution}{exo:g9:solids:8}
$ V = \frac13 \times 230^2 \times 147 = \frac13 \times 52\,900 \times 147
= 52\,900 \times 49 = 2\,592\,100$ eu $^3$ --- sobre $2.6$ milhão cúbico
metros.
\end{solution}

\begin{solution}{exo:g9:solids:9}
\emph{1.} A água forma um \omterm{def:g8:solids:def}{cone} dimensionado por $ k = \frac12$ (ápice para baixo,
\omterm{def:g3:division:half}{metade} a altura), então é \omterm{def:g6:measure:volume}{volume} é $\left(\frac12\right)^3 = \frac18$
do funil: um oitavo, muito menos que \omterm{def:g3:division:half}{metade}!

\emph{2.} Água até a altura $ d $ forma um \omterm{def:g8:solids:def}{cone} dimensionado por
$ k = \frac{d}{12}$, de \omterm{def:g6:measure:volume}{volume} $ k^3$ vezes o do funil. \omterm{def:g3:division:half}{Metade} o \omterm{def:g6:measure:volume}{volume}
significa $ k^3 = \frac12$, ou seja\
$\left(\frac{d}{12}\right)^3 = \frac12$. Então
$\frac{d}{12} = \sqrt[3]{0.5} \approx 0.7937$, então
$ d \approx 9.5$ cm: o segundo \omterm{def:g3:division:half}{metade} do \omterm{def:g6:measure:volume}{volume} ocupa apenas o topo
$2.5$ cm --- a parte larga do \omterm{def:g8:solids:def}{cone} retém a maior parte da água.
\end{solution}

\begin{solution}{pb:g9:solids:1}
\textbf{1.} $ V_{\text{cyl}} = \pi r^2 \times 2r = 2\pi r^3$.

\textbf{2.} $\dfrac{V_{\text{sphere}}}{V_{\text{cyl}}}
= \dfrac{\frac43 \pi r^3}{2 \pi r^3} = \dfrac23$. O $\pi $ e
o $ r^3$ cancelar: a proporção é \emph{universal} --- verdade
para uma bola de gude e para um planeta. Uma relação entre formas, não
entre números: exatamente o tipo de verdade que vale a pena esculpir
pedra.

\textbf{3.} Esfera: $4\pi r^2$. \omterm{def:g7:areas:prism}{Cilindro}: laterais
$2\pi r \times 2r = 4\pi r^2$, além de duas tampas
$2 \times \pi r^2$: total $6\pi r^2$. Razão:
$\frac{4\pi r^2}{6\pi r^2} = \frac23$ --- os mesmos dois terços,
para superfícies. (E um bônus: a esfera \omterm{def:g6:measure:area}{área} exatamente igual
o cilindrolateral \omterm{def:g6:measure:area}{área}.)

\textbf{4.} Sobras: $2\pi r^3 - \frac43\pi r^3 =
\frac23 \pi r^3$. Dois \omterm{def:g8:solids:def}{cones} de \omterm{def:g3:shapes:shapes}{raio} $ r $ e altura $ r $:
$2 \times \frac13 \pi r^2 \times r = \frac23 \pi r^3$. Igual.

\textbf{5.} Bola: $\frac43 \pi \times 12^3 \approx 7\,200$
cm $^3$; \omterm{def:g5:solids:def}{caixa}: $2\pi \times 12^3 \approx 10\,900$ cm $^3$. Vazio:
um terço do \omterm{def:g5:solids:def}{caixa}, sobre $33\,\%$ --- garantido por
questão 2, qualquer que seja o tamanho da bola.

\textbf{6.} \omterm{def:g3:division:half}{Metade} uma esfera:
$\frac12 \times \frac43 \pi \times 10^3 = \frac23 \pi \times
1\,000 \approx 2\,094$ cm $^3 \approx 2.1$ eu.

\textbf{7.} \omterm{def:g8:solids:def}{Cone}: $\frac13 \pi \times 10^2 \times 10 =
\frac{1\,000\pi}{3} \approx 1\,047$ cm $^3$. Hemisfério:
$\frac{2\,000\pi}{3} \approx 2\,094$ cm $^3$. \omterm{def:g7:areas:prism}{Cilindro}:
$1\,000\pi \approx 3\,142$ cm $^3$. Proporções:
$\frac{1000\pi}{3} : \frac{2000\pi}{3} : \frac{3000\pi}{3}
= 1 : 2 : 3$ exatamente.

\textbf{8.} $\frac43 \pi \times 27 = \pi \times 9 \times h $
dá $ h = \frac43 \times 3 = 4$ cm exatamente.

\textbf{9.} \omterm{def:g3:shapes:shapes}{Raios}: $\frac{6\,371}{1\,737} \approx 3.67$.
\omterm{def:g6:measure:volume}{Volumes} escala como o \omterm{def:g5:solids:def}{cubo} (\cref{thm:g9:solids:scaling}):
$3.67^3 \approx 49$. Cerca de cinquenta luas cabem na Terra.

\textbf{10.} Superfícies escala como o quadrado: $3.67^2 \approx
13.5$. Dobrando um balão \omterm{def:g3:shapes:shapes}{raio} multiplica seu \omterm{def:g6:measure:volume}{volume} por
$2^3 = 8$ mas sua superfície por apenas $2^2 = 4$: à medida que as coisas crescem,
\omterm{def:g6:measure:volume}{volume} (peso, conteúdo, calor produzido) ultrapassa a superfície (pele,
material, calor perdido) --- a lei do \omterm{def:g5:solids:def}{quadrado-cubo}.

\textbf{11.} Peso: $\times 10^3 = 1\,000$. Osso
\omterm{prop:g9:solids:sections}{corte transversal}: $\times 10^2 = 100$. Pressão --- peso por
\omterm{def:g6:measure:area}{área} de osso: $\times \frac{1000}{100} = 10$. Ossos construídos para
a pressão humana recebe dez vezes mais: o esqueleto do gigante
se encaixa sob seu próprio peso. Gigantes são geometricamente
impossível; animais grandes precisam de ossos desproporcionalmente grossos
(compare as pernas de um elefante com as de uma gazela).

\textbf{12.} Peso: $\div 100^3 = 10^6$. Força (músculo
\omterm{prop:g9:solids:sections}{corte transversal}): $\div 100^2 = 10^4$. Resistência ao peso:
$\times \frac{10^6}{10^4} = 100$. A formiga levantando cinquenta vezes
seu peso não é o de um superatleta – é apenas pequeno; um
um humano encolhido ao tamanho de uma formiga poderia fazer o mesmo.

\textbf{13.} $\dfrac{S}{V} = \dfrac{4\pi r^2}{\frac43\pi r^3}
= \dfrac{3}{r}$: para $ r = 1$ a proporção é $3$, para $ r = 2$ isso é $1.5$ --- reduzir pela \omterm{def:g3:division:half}{metade} o tamanho dobra a superfície disponível
por unidade de produção de calor \omterm{def:g6:measure:volume}{volume}. Corpos pequenos sangram calor:
o rato deve comer constantemente, o urso pode hibernar e o
bebê --- esfera pequena, grande $\frac Sr $ --- precisa do chapéu.

\textbf{14.} $\left(\frac54\right)^3 = \frac{125}{64} \approx
1.95$: a laranja grande segura quase \emph{duas vezes} a fruta para
o mesmo preço. Comprar \omterm{def:g3:shapes:shapes}{raio}.

\textbf{15.} A escultura diz: esfera $= \frac23$ do
\omterm{def:g7:areas:prism}{cilindro} em \omterm{def:g6:measure:volume}{volume}, \emph{e} $\frac23$ disso no total
superfície --- duas proporções exatas e universais ligando o
mais redondo \omterm{def:g5:solids:def}{sólido} para o mais simples. A resposta de Cícero: porque um
teorema é o único monumento que nem exércitos nem séculos
erodir --- As máquinas de Arquimedes queimaram com Siracusa, mas o
dois terços ainda são exatamente dois terços.
\end{solution}
