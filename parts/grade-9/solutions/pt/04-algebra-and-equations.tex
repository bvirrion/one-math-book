\section*{Capítulo \ref{ch:g9:algebra} --- Álgebra e Equações}

\begin{solution}{exo:g9:algebra:1}
$4(3x - 2) = 12x - 8$.

$(x+6)(x+2) = x^2 + 2x + 6x + 12 = x^2 + 8x + 12$.

$(2x-3)(x+5) = 2x^2 + 10x - 3x - 15 = 2x^2 + 7x - 15$.

$(x-1)(x-9) = x^2 - 9x - x + 9 = x^2 - 10x + 9$.
\end{solution}

\begin{solution}{exo:g9:algebra:2}
$(x+8)^2 = x^2 + 16x + 64$.

$(3x-4)^2 = 9x^2 - 24x + 16$.

$(x+10)(x-10) = x^2 - 100$.

$(5x+2)(5x-2) = 25x^2 - 4$.
\end{solution}

\begin{solution}{exo:g9:algebra:3}
$101^2 = (100+1)^2 = 10000 + 200 + 1 = 10201$.

$59^2 = (60-1)^2 = 3600 - 120 + 1 = 3481$.

$53 \times 47 = (50+3)(50-3) = 2500 - 9 = 2491$.
\end{solution}

\begin{solution}{exo:g9:algebra:4}
$9x + 12 = 3(3x + 4)$.

$ x^2 - 64 = (x+8)(x-8)$.

$ x^2 + 14x + 49 = (x+7)^2$.

$18x^2 - 6x = 6x(3x - 1)$.
\end{solution}

\begin{solution}{exo:g9:algebra:5}
$4x + 7 = 19$: $4x = 12$, então $ x = 3$.

$6x - 5 = 2x + 11$: $4x = 16$, então $ x = 4$.

$5(x-3) = 3x + 1$: $5x - 15 = 3x + 1$, então $2x = 16$ e $ x = 8$.
\end{solution}

\begin{solution}{exo:g9:algebra:6}
$(x+4)(3x-12) = 0$: $ x = -4$ ou $ x = 4$.

$ x^2 - 121 = (x-11)(x+11) = 0$: $ x = 11$ ou $ x = -11$.

$(2x-1)^2 = 0$: a única solução $ x = \frac12$.

$ x^2 = 8x $: $ x(x - 8) = 0$, então $ x = 0$ ou $ x = 8$.
\end{solution}

\begin{solution}{exo:g9:algebra:7}
\omterm{ex:g1:subtraction:difference}{Diferença} de quadrados com $ a = 3x + 2$ e $ b = 5$:
\[
E = (3x + 2)^2 - 5^2 = (3x + 2 + 5)(3x + 2 - 5) = (3x + 7)(3x - 3).
\]
\omterm{def:g2:mult:def}{Produto} \omterm{def:g1:counting:zero}{zero} regra: $3x + 7 = 0$ dá $ x = -\frac73$, e $3x - 3 = 0$
dá $ x = 1$.
\end{solution}

\begin{solution}{exo:g9:algebra:8}
$3x - 4 \geq 8$: $3x \geq 12$, então $ x \geq 4$ (ponto preenchido em $4$, sombra
certo).

$5 - 2x > 1$: $-2x > -4$, dividindo por $-2$ inverte o sinal: $ x < 2$
(ponto vazio em $2$, sombra à esquerda).

$4(x+1) \leq x - 5$: $4x + 4 \leq x - 5$, então $3x \leq -9$, então
$ x \leq -3$ (ponto preenchido em $-3$, sombra à esquerda).
\end{solution}

\begin{solution}{exo:g9:algebra:9}
Deixar $ x $ seja o número: $3x + 7 = 5x - 9$. Então $16 = 2x $, então $ x = 8$.
Verificar: $3 \times 8 + 7 = 31$ e $5 \times 8 - 9 = 31$.
\end{solution}

\begin{solution}{exo:g9:algebra:10}
\emph{1.} $(n+1)(n-1) = n^2 - 1$ (terceira identidade). Os vizinhos de $ n $
são $ n - 1$ e $ n + 1$, então seu \omterm{def:g2:mult:def}{produto} é $ n^2 - 1$: por exemplo\
$24 \times 26 = 625 - 1 = 624$.

\emph{2.} Três inteiros consecutivos podem ser escritos $ n - 1$, $ n $,
$ n + 1$. Deles \omterm{def:g1:addition:def}{soma} é
\[
(n-1) + n + (n+1) = 3n ,
\]
um \omterm{def:g9:arith:divisor}{múltiplo} de $3$ (três vezes o do meio).
\end{solution}

\begin{solution}{pb:g9:algebra:1}
\textbf{1.} $(10a + b) - (10b + a) = 9a - 9b = 9(a - b)$: o
\omterm{ex:g1:subtraction:difference}{diferença} entre um número de dois dígitos e sua reversão é sempre
um \omterm{def:g9:arith:divisor}{múltiplo} de $9$. Verificar: $72 - 27 = 45 = 9 \times 5$, e
$ a - b = 7 - 2 = 5$.

\textbf{2.} \omterm{def:g9:algebra:expand}{Expandindo} o lado direito:
$9(11a + b) + a + b + c = 99a + 9b + a + b + c
= 100a + 10b + c $. Então o número é igual a \omterm{def:g9:arith:divisor}{múltiplo} de $9$ mais
seu dígito \omterm{def:g1:addition:def}{soma}: dividindo por $9$, ambos saem do mesmo \omterm{def:g3:division:remainder}{restante}.
Em particular \omterm{def:g3:division:remainder}{restante} $0$ por um meio \omterm{def:g3:division:remainder}{restante} $0$ para o
outro: a regra dos nove, comprovada.

\textbf{3.} $9(11a + b)$ também é um \omterm{def:g9:arith:divisor}{múltiplo} de $3$, então o mesmo
identidade mostra número e dígito \omterm{def:g1:addition:def}{soma} deixe o mesmo \omterm{def:g3:division:remainder}{restante} em
a divisão por $3$: a regra de três.

\textbf{4.} Escrever $47 = 9k + 2$ e $86 = 9l + 5$. Então
\[
47 \times 86 = (9k + 2)(9l + 5)
= 9\,(9kl + 5k + 2l) + 10 :
\]
o \omterm{def:g2:mult:def}{produto} deixa o mesmo \omterm{def:g3:division:remainder}{restante} como $2 \times 5 = 10$,
a saber $1$. Um resultado correto deve, portanto, também deixar \omterm{def:g3:division:remainder}{restante}
$1$ --- qual $4\,042$ faz. Mas o teste só vê restos:
$4\,042$ e $4\,402$ (dígitos trocados) passam de forma idêntica, então
passar não prova nada --- falhar é o que condena.

\textbf{5.} \omterm{def:g9:algebra:expand}{Expandindo}: $11(91a + 9b + c) - a + b - c + d
= 1001a + 99b + 11c - a + b - c + d
= 1000a + 100b + 10c + d $. Então o número é \omterm{def:g6:wholes:divisible}{divisível} por $11$
exatamente quando está alternando \omterm{def:g1:addition:def}{soma} $-a + b - c + d $ é. Para
$2\,926$: $-2 + 9 - 2 + 6 = 11$, \omterm{def:g6:wholes:divisible}{divisível} por $11$: sim,
$2\,926 = 11 \times 266$.

\textbf{6.} $742 - 247 = 495$; $495 + 594 = 1\,089$. Qualquer válido
exemplo pousa em $1\,089$ também.

\textbf{7.} $(100a + 10b + c) - (100c + 10b + a) = 99(a - c)$.
Para $ a - c = 2, 3, \dots, 9$:
\[
198,\ 297,\ 396,\ 495,\ 594,\ 693,\ 792,\ 891 .
\]

\textbf{8.} $100(m - 1) + 90 + (10 - m) = 100m - 100 + 90 + 10
- m = 99m $. Então os três dígitos de $99m $ são $ m - 1$, então $9$,
então $10 - m $ (verificar $495$: $ m = 5$ dá $4$, $9$, $5$).

\textbf{9.} A reversão de $99m $ tem dígitos $10 - m $, $9$,
$ m - 1$, então é igual $100(10 - m) + 90 + (m - 1)$. Adicionando:
\[
100\,(m - 1 + 10 - m) + 180 + (10 - m + m - 1)
= 100 \times 9 + 180 + 9 = 1\,089 ,
\]
independente de $ m $. Qualquer que seja o número inicial, o mágico
é seguro.

\textbf{10.} Se $ a = c $, a subtração dá $0$ e o truque
morre de uma vez. Se $ a - c = 1$, dá $99$, que deve ser
tratado como a string de três dígitos $099$ (reversão $990$,
$99 + 990 = 1\,089$: salvo!). As letras pequenas: \omterm{ex:g1:subtraction:difference}{diferenças} de
$1$ funciona apenas se o espectador escrever o resultado da subtração
com três dígitos, começando \omterm{def:g1:counting:zero}{zero} incluído --- exigindo uma lacuna de
pelo menos $2$ evita a discussão.

\textbf{11.} $2(5m + 7) + d - 14 = 10m + 14 + d - 14 =
10m + d $: as dezenas (e centenas, para outubro-dezembro) mostram
o mês, as unidades (e dezenas) o dia.

\textbf{12.} $2(5m + 9) + d = 10m + 18 + d $: subtrair $18$.

\textbf{13.} Pela pergunta 2, $ n - (\text{digit sum of } n) =
9(11a + b)$ para um valor de três dígitos $ n $: sempre um \omterm{def:g9:arith:divisor}{múltiplo} de $9$
(identidades semelhantes funcionam para qualquer comprimento). UM \omterm{def:g9:arith:divisor}{múltiplo} de $9$ tem
dígito \omterm{def:g1:addition:def}{soma} um \omterm{def:g9:arith:divisor}{múltiplo} de $9$ (pergunta 2 novamente), então os dígitos
a leitura do espectador deve ser concluída até o próximo \omterm{def:g9:arith:divisor}{múltiplo}
de $9$: o dígito que falta é esse complemento. Ambiguidade: se o
os dígitos lidos já somam um \omterm{def:g9:arith:divisor}{múltiplo} de $9$, o dígito cruzado
poderia ser $0$ ou $9$ --- excluindo $0$ tira a dúvida.

\textbf{14.} $(10t + 5)^2 = 100t^2 + 100t + 25 =
100\,t(t + 1) + 25$: escrever $ t(t+1)$, depois cole $25$ por trás disso.
$45^2$: $4 \times 5 = 20$, então $2\,025$. $85^2$:
$8 \times 9 = 72$, então $7\,225$. $105^2$: $10 \times 11 = 110$,
então $11\,025$.

\textbf{15.} Cinco consecutivos: $ n + (n+1) + (n+2) + (n+3) +
(n+4) = 5n + 10 = 5(n + 2)$: um \omterm{def:g9:arith:divisor}{múltiplo} de $5$, sempre. Quatro
consecutivo: $ n + (n+1) + (n+2) + (n+3) = 4n + 6 =
4(n + 1) + 2$: \omterm{def:g3:division:remainder}{restante} $2$ na divisão por $4$, nunca um
\omterm{def:g9:arith:divisor}{múltiplo}. A álgebra converte “parece funcionar” em
\emph{sempre}, e ``Não encontrei nenhum exemplo'' em \emph{nunca} ---
duas palavras que nenhum teste pode alcançar
(\cref{met:g7:literal:test}).
\end{solution}
