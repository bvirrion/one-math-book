\section*{Capítulo \ref{ch:g9:trig} --- Trigonometria no Triângulo Retângulo}

\begin{solution}{exo:g9:trig:1}
Hipotenusa: $\sqrt{9^2 + 12^2} = \sqrt{81 + 144} = \sqrt{225} = 15$.

Outra perna: $\sqrt{17^2 - 8^2} = \sqrt{289 - 64} = \sqrt{225} = 15$.
\end{solution}

\begin{solution}{exo:g9:trig:2}
$7^2 + 24^2 = 49 + 576 = 625 = 25^2$: sim, \omterm{def:g6:shapes:triangles}{em ângulo reto} (conversa de
Pitágoras), com o \omterm{def:g3:shapes:rightangle}{ângulo reto} oposto ao lado $25$.

$5^2 + 6^2 = 61 \neq 64 = 8^2$: não \omterm{def:g6:shapes:triangles}{em ângulo reto}.
\end{solution}

\begin{solution}{exo:g9:trig:3}
A hipotenusa é $[EF]$ (em frente ao \omterm{def:g3:shapes:rightangle}{ângulo reto} no $ D $). O lado
oposto a $\theta = \widehat E $ é $[DF]$; o lado adjacente é
$[DE]$. Por isso
\[
\cos\theta = \frac{DE}{EF}, \qquad
\sin\theta = \frac{DF}{EF}, \qquad
\tan\theta = \frac{DF}{DE}.
\]
\end{solution}

\begin{solution}{exo:g9:trig:4}
Perna oposta: $10 \sin 28^\circ \approx 10 \times 0.469 = 4.7$.
Perna adjacente: $10 \cos 28^\circ \approx 10 \times 0.883 = 8.8$.
(Verificar: $4.7^2 + 8.8^2 \approx 22 + 77 \approx 100$.)
\end{solution}

\begin{solution}{exo:g9:trig:5}
$\tan\theta = \dfrac{4.5}{6} = 0.75$, então
$\theta = \tan^{-1}(0.75) \approx 37^\circ $.
\end{solution}

\begin{solution}{exo:g9:trig:6}
No triângulo retângulo formado pelo observador, o ponto no chão
abaixo do drone, e o drone: a altitude $120$ m é o lado
oposto ao ângulo de elevação, a distância horizontal $ d $ é adjacente.
Então
\[
\tan 32^\circ = \frac{120}{d},
\qquad
d = \frac{120}{\tan 32^\circ} \approx \frac{120}{0.625} = 192 \text{ m}.
\]
\end{solution}

\begin{solution}{exo:g9:trig:7}
A ascensão ($1.2$ m) é oposto ao ângulo; o comprimento da rampa $ L $ é o
hipotenusa:
\[
\sin 5^\circ = \frac{1.2}{L},
\qquad
L = \frac{1.2}{\sin 5^\circ} \approx \frac{1.2}{0.0872} \approx 13.8
\text{ m}.
\]
A rampa deve ter pelo menos $13.8$ m longo.
\end{solution}

\begin{solution}{exo:g9:trig:8}
De $\cos^2\theta + \sin^2\theta = 1$:
$\sin^2\theta = 1 - 0.36 = 0.64$, e $\sin\theta > 0$ para um agudo
ângulo, então $\sin\theta = 0.8$. Então
$\tan\theta = \dfrac{\sin\theta}{\cos\theta} = \dfrac{0.8}{0.6} =
\dfrac43$.
\end{solution}

\begin{solution}{exo:g9:trig:9}
\emph{Certo triângulo \omterm{def:g6:shapes:triangles}{isósceles}} com pernas $1$: hipotenusa
$\sqrt{1 + 1} = \sqrt2$; cada ângulo agudo é $45^\circ $, e
\[
\cos 45^\circ = \frac{1}{\sqrt2} = \frac{\sqrt2}{2} = \sin 45^\circ,
\qquad
\tan 45^\circ = \frac11 = 1 .
\]

\emph{Triângulo \omterm{def:g6:shapes:triangles}{Equilátero}} de lado $1$: sua altura o divide em dois
triângulos retângulos com hipotenusa $1$, básico $\frac12$ e altura
$ h = \sqrt{1 - \frac14} = \frac{\sqrt3}{2}$ (Pitágoras). Os ângulos são
$60^\circ $ (na \omterm{def:g8:powers:def}{base}) e $30^\circ $ (no topo). Lendo o
proporções:
\[
\cos 60^\circ = \frac{1/2}{1} = \frac12, \quad
\sin 60^\circ = \frac{\sqrt3/2}{1} = \frac{\sqrt3}{2}, \quad
\cos 30^\circ = \frac{\sqrt3}{2}, \quad
\sin 30^\circ = \frac12 .
\]
\end{solution}

\begin{solution}{pb:g9:trig:1}
\textbf{1.} A altura acima do nível dos olhos é o lado oposto
$35^\circ $, com lado adjacente $60$ eu:
$ h = 60 \tan 35^\circ \approx 60 \times 0.700 = 42$ m.

\textbf{2.} $\tan\theta = 0.12$ dá
$\theta = \tan^{-1}(0.12) \approx 6.8^\circ $ --- íngreme por um
estrada, mas um ângulo modesto. UM $45^\circ $ estrada tem
$\tan 45^\circ = 1$: um $100\,\%$ nota. (``Cem
por cento'' está longe de ser vertical --- um teleférico favorito
mal-entendido.)

\textbf{3.} Em um triângulo retângulo com ângulos agudos $\theta $ e
$90^\circ - \theta $, o lado oposto é adjacente ao
outro e a hipotenusa é compartilhada. Então
$\sin(90^\circ - \theta) = \frac{\text{its opposite}}{h} =
\frac{\text{adjacent of }\theta}{h} = \cos\theta $ e
simetricamente $\cos(90^\circ - \theta) = \sin\theta $: o
\emph{co}\omterm{def:g9:trig:ratios}{seno} é o \omterm{def:g9:trig:ratios}{seno} do \emph{co}implementar.

\textbf{4.} Altura: $8 \sin 60^\circ = 8 \times
\frac{\sqrt3}{2} = 4\sqrt3 \approx 6.93$ m. Pé:
$8 \cos 60^\circ = 4$ estou exatamente.

\textbf{5.} $0.574^2 + 0.819^2 = 0.329 + 0.671 = 1.000$ (para
\omterm{def:g6:decimals:rounding}{arredondamento}); e $\left(\frac12\right)^2 +
\left(\frac{\sqrt3}{2}\right)^2 = \frac14 + \frac34 = 1$. Qualquer
\omterm{def:g3:numbers:evenodd}{par} de valores que falha na identidade trai uma leitura errada ou
uma calculadora no modo errado --- um detector de erros gratuito.

\textbf{6.} $\tan 40^\circ = \dfrac hx $ e
$\tan 30^\circ = \dfrac{h}{x + 200}$.

\textbf{7.} $ x = \dfrac{h}{\tan 40^\circ}$ e
$ x + 200 = \dfrac{h}{\tan 30^\circ}$; subtraindo,
$200 = h\left(\frac{1}{\tan 30^\circ} -
\frac{1}{\tan 40^\circ}\right)$, de onde vem a fórmula. Números:
$\frac{1}{0.577} \approx 1.732$, $\frac{1}{0.839} \approx
1.192$, \omterm{ex:g1:subtraction:difference}{diferença} $0.540$:
$ h \approx \frac{200}{0.540} \approx 370$ m.

\textbf{8.} $ x = \frac{h}{\tan 40^\circ} \approx
\frac{370}{0.839} \approx 441$ m. Verifique de $ A $:
$\frac{h}{x + 200} \approx \frac{370}{641} \approx 0.577 =
\tan 30^\circ $: consistente.

\textbf{9.} O método de estação única precisa do método horizontal
distância até o ponto \emph{diretamente abaixo do cume} ---
incomensurável através de um desfiladeiro ou dentro de uma montanha. O
o método de duas estações o substitui por uma distância que você pode percorrer
você mesmo: o $200$ m entre seus próprios dois pontos de vista.

\textbf{10.} $ h = \dfrac{200}{\sqrt3 - \frac{1}{\sqrt3}}
= \dfrac{200}{\frac{2}{\sqrt3}} = 100\sqrt3 \approx 173$ m ---
segundo andar da Torre Eiffel, medido com dois avistamentos
e um passeio.

\textbf{11.} \omterm{def:g6:lines:objects}{Linha} de visão $ d $, \omterm{def:g3:shapes:shapes}{raios} $ R $ (até o ponto de pastagem,
\omterm{def:g6:lines:perp}{perpendicular} para a \omterm{def:g6:lines:objects}{linha} de visão) e $ R + h $ (ao olho):
Pitágoras dá $ d^2 + R^2 = (R + h)^2 = R^2 + 2Rh + h^2$, então
$ d^2 = 2Rh + h^2 \approx 2Rh $ ($ h $ é minúsculo contra $ R $).
Olhos em $1.80$ eu:
$ d \approx \sqrt{2 \times 6.37 \times 10^6 \times 1.8}
\approx 4\,800$ m --- o horizonte do mar mal chega a cinco
quilômetros de distância.

\textbf{12.} $ d \approx \sqrt{2 \times 6.37 \times 10^6 \times
324} \approx 64$ km. Regra dos marinheiros: $3.6\sqrt{1.8} \approx
4.8$ km e $3.6\sqrt{324} = 64.8$ km --- ambos correspondem.

\textbf{13.} $\tan\theta = \frac{2}{60}$ dá $\theta \approx
1.9^\circ $. A Lua $0.5^\circ $ é sobre um \emph{\omterm{def:g3:division:half}{trimestre}} de
um polegar: a Lua parece enorme e está escondida por uma unha
--- o olho é um péssimo transferidor.

\textbf{14.} Escadaria: $\tan^{-1}\frac{17}{29} \approx 30^\circ $
--- escadas confortáveis sobem quase exatamente
$30^\circ $ dos valores especiais. No $60^\circ $ em um $29$ centímetros
piso, o riser seria $29\tan 60^\circ = 29\sqrt3 \approx
50$ cm: uma escada, não uma escada.

\textbf{15.} $\tan 10^\circ \approx 0.18$;
$\tan 45^\circ = 1$; $\tan 80^\circ \approx 5.7$;
$\tan 89^\circ \approx 57.3$. Como $\theta \to 90^\circ $ o
lado adjacente encolhe a nada enquanto o lado oposto se mantém:
o \omterm{def:g3:division:remainder}{quociente} cresce além de todos os limites. No $90^\circ$ exatamente
não há triângulo nem valor --- o gráfico do \omterm{def:g9:trig:ratios}{tangente}
sobe uma assíntota vertical, a primeira de muitas no Ensino Médio
\omterm{def:g6:measure:volume}{volume}.
\end{solution}
