\section*{Capítulo \ref{ch:g9:thales} --- Teorema de Tales}

\begin{solution}{exo:g9:thales:1}
Tales: $\dfrac{AM}{AB} = \dfrac{AN}{AC} = \dfrac{MN}{BC}$, ou seja\
$\dfrac26 = \dfrac13$. Então $\dfrac{3}{AC} = \dfrac13$ dá $ AC = 9$,
e $\dfrac{MN}{9} = \dfrac13$ dá $ MN = 3$.
\end{solution}

\begin{solution}{exo:g9:thales:2}
Primeira descoberta $ AB = AM + MB = 5 + 3 = 8$. Então
$\dfrac{AM}{AB} = \dfrac{AN}{AC}$ dá $\dfrac58 = \dfrac{4}{AC}$, então
$5 \times AC = 32$ e $ AC = 6.4$.
\end{solution}

\begin{solution}{exo:g9:thales:3}
A borboleta funciona exatamente como a configuração aninhada:
$\dfrac{AM}{AB} = \dfrac{AN}{AC} = \dfrac{MN}{BC}$, ou seja\
$\dfrac{3}{7.5} = \dfrac25$. Então $\dfrac{2}{AC} = \dfrac25$ dá
$ AC = 5$, e $\dfrac{2.6}{BC} = \dfrac25$ dá
$ BC = \dfrac{2.6 \times 5}{2} = 6.5$.
\end{solution}

\begin{solution}{exo:g9:thales:4}
$\dfrac{AM}{AB} = \dfrac68 = \dfrac34$ e
$\dfrac{AN}{AC} = \dfrac{9}{12} = \dfrac34$: proporções iguais, pontos no
mesma ordem, então $(MN) \parallel (BC)$ pelo \omterm{def:g8:fractions:inverse}{inverso} de Tales.
\end{solution}

\begin{solution}{exo:g9:thales:5}
$\dfrac{AM}{AB} = \dfrac46 = \dfrac23$ e
$\dfrac{AN}{AC} = \dfrac58$. Verificação cruzada: $\dfrac23 = \dfrac{16}{24}$
e $\dfrac58 = \dfrac{15}{24}$: as proporções diferem. Se as \omterm{def:g6:lines:objects}{linhas} fossem
\omterm{def:g6:lines:perp}{paralelo}, o teorema de Tales os forçaria a serem iguais --- então $(MN)$
e $(BC)$ são \emph{não} \omterm{def:g6:lines:perp}{paralelo}.
\end{solution}

\begin{solution}{exo:g9:thales:6}
Deixar $ S $ seja a ponta da sombra, $ P $ o topo da cabeça da pessoa, $ L $
a parte superior da lâmpada. A pessoa ($1.8$ m à distância $3$ estou de $ S $) e
a lâmpada (altura $ h $ à distância $3 + 2 = 5$ estou de $ S $) são dois
\omterm{def:g6:lines:perp}{paralelo} vertical \omterm{def:g6:lines:objects}{segmentos} cortado pelo \omterm{def:g3:shapes:shapes}{raio} de luz $(SL)$: Tales do
ponto $ S $ dá
\[
\frac{1.8}{h} = \frac{3}{5},
\qquad\text{so}\quad
h = \frac{1.8 \times 5}{3} = 3 \text{ m}.
\]
\end{solution}

\begin{solution}{exo:g9:thales:7}
\emph{1.} Distância real: $6.8 \times 25\,000 = 170\,000$ cm $=
1.7$ km.

\emph{2.} \omterm{def:g6:measure:area}{Áreas} dimensionar por $ k^2$: \omterm{def:g6:measure:area}{área} real $= 8 \times 25\,000^2$
cm $^2$ $= 8 \times 6.25 \times 10^8$ cm $^2$ $= 5 \times 10^9$ cm $^2$.
Desde $1$ quilômetros $^2 = 10^{10}$ cm $^2$, aquilo é $0.5$ quilômetros $^2$.
\end{solution}

\begin{solution}{exo:g9:thales:8}
\emph{1.} A proporção é $\dfrac{AM}{AB} = \dfrac{8}{12} = \dfrac23$.
Tales: $ AN = \frac23 \times 15 = 10$ e $ MN = \frac23 \times 18 = 12$.

\emph{2.} O \omterm{def:g9:thales:scaling}{fator de escala} é $ k = \frac23$. \omterm{def:g6:measure:perimeter}{Perímetros}: $ ABC $ tem
$12 + 15 + 18 = 45$, e $ AMN $ tem $8 + 10 + 12 = 30 = \frac23 \times
45$: o \omterm{def:g6:measure:perimeter}{perímetro} é dimensionado por $ k $ também, como todo comprimento.
\end{solution}

\begin{solution}{exo:g9:thales:9}
Em volta $ O $, as \omterm{def:g6:lines:objects}{linhas} $(AC)$ e $(BD)$ cruz, e
$(AB) \parallel (CD)$: configuração de borboleta com os triângulos $ OAB $
e $ OCD $. Tales dá
\[
\frac{OA}{OC} = \frac{OB}{OD} = \frac{AB}{CD} = \frac46 = \frac23 .
\]
Então $\dfrac{OA}{OC} = \dfrac23$, e a igualdade
$\dfrac{OA}{OC} = \dfrac{OB}{OD}$ diz exatamente isso $ O $ corta os dois
diagonais na mesma proporção.
\end{solution}

\begin{solution}{pb:g9:thales:1}
\textbf{1.} A construção produz dois pontos em $[AB]$;
ligue para eles $ M_1$ (de $ P_1$) e $ M_2$ (de $ P_2$).

\textbf{2.} No triângulo $ A P_3 B $, os \omterm{def:g6:lines:perp}{paralelos} com
$(P_3 B)$ através $ P_1$ e $ P_2$ corte as laterais
\omterm{def:g9:linfunc:linear}{proporcionalmente} (\cref{thm:g9:thales:direct}):
\[
\frac{A M_1}{AB} = \frac{A P_1}{A P_3} = \frac13,
\qquad
\frac{A M_2}{AB} = \frac{A P_2}{A P_3} = \frac23 .
\]
A bússola feita $ A P_1$, $ P_1 P_2$, $ P_2 P_3$ iguais, então o
as proporções são exatamente terços --- qualquer que seja o ângulo do \omterm{def:g3:shapes:shapes}{raio} e
a abertura da bússola escolhida.

\textbf{3.} Para quintas: desça cinco comprimentos iguais, junte-se ao
quinto ponto para $ B $ e desenhe quatro \omterm{def:g6:lines:perp}{paralelos}. Para $\frac35$ de um
\omterm{def:g6:lines:objects}{segmento}: mesma figura, e tome o ponto cortado pelo \omterm{def:g6:lines:perp}{paralelo}
através $ P_3$: fica em $\frac{3}{5}$ de $[AB]$ de $ A $.

\textbf{4.} $\frac{AM}{MB} = \frac23$ significa
$ AM = \frac25 AB $: cinco passos iguais no \omterm{def:g3:shapes:shapes}{raio}, junte $ P_5$ para
$ B $, e o \omterm{def:g6:lines:perp}{paralelo} através $ P_2$ marcas $ M $.

\textbf{5.} Medindo $\sqrt2 = 1.41421\ldots $ cm só usa
um número finito de decimais: qualquer ``terço'' medido é um
aproximação. A construção de Tales nunca lê um número: ela
fornece o ponto exato em $\frac{\sqrt2}{3}$ --- partes iguais
de um \omterm{def:g6:lines:objects}{segmento} nenhum governante pode sequer expressar
(\cref{pb:g9:sqrt:1}).

\textbf{6.} A parte superior do objeto, o buraco e a parte inferior da imagem
estão alinhados, e da mesma forma para a parte inferior do objeto e o
topo da imagem: duas \omterm{def:g6:lines:objects}{linhas} se cruzando no buraco, com o objeto
e a parede da imagem paralela. Tales na borboleta:
\[
\frac{h}{H} = \frac{d}{D},
\qquad\text{so}\qquad
h = H \times \frac dD .
\]

\textbf{7.} $ h = 12 \times \frac{0.20}{30} = 0.08$ eu $= 8$ centímetros
--- de cabeça para baixo (os \omterm{def:g3:shapes:shapes}{raios} se cruzam no buraco).

\textbf{8.} $ H = h \times \frac Dd = 9 \times 10^{-3} \times
\frac{1.5 \times 10^{11}}{1} = 1.35 \times 10^{9}$ m. Verdadeiro
valor $1.39 \times 10^9$ m: uma \omterm{def:g5:solids:def}{caixa} de sapatos mede o Sol até
dentro de alguns por cento.

\textbf{9.} Lua: $\frac{3.5 \times 10^6}{3.8 \times 10^8}
\approx 0.0092$. Sol: $\frac{1.39 \times 10^9}{1.5 \times
10^{11}} \approx 0.0093$. As duas proporções --- os tamanhos aparentes
no céu --- são quase idênticos: a Lua pode cobrir o Sol
\emph{exatamente}, borda a borda. Esse acaso cósmico é o total
eclipse solar.

\textbf{10.} Cada \omterm{def:g3:shapes:shapes}{raio} deve passar por um buraco, então os \omterm{def:g3:shapes:shapes}{raios}
do topo do objeto continue \emph{abaixo} e \omterm{def:g3:shapes:shapes}{raios} de
o fundo continua \emph{acima}: a imagem é invertida. Ampliando
o buraco deixa passar um monte de cópias ligeiramente deslocadas
da imagem, que se sobrepõem: mais brilhantes, mas borradas.

\textbf{11.} $\frac{PQ}{MN} = \frac{OP}{OM} = \frac{7.5}{3}
= 2.5$, então $ PQ = 10$; e $\frac{ON}{OQ} = \frac{OM}{OP} =
\frac{1}{2.5}$, então $ ON = \frac{6}{2.5} = 2.4$.

\textbf{12.} Se o \omterm{def:g6:lines:perp}{paralelo} passa pelo ponto médio de um
lado, a razão de Tales é $\frac12$, então corta o segundo lado
em \emph{isso é} ponto médio, e o \omterm{def:g6:lines:objects}{segmento} \omterm{def:g6:lines:perp}{paralelo} mede \omterm{def:g3:division:half}{metade}
a \omterm{def:g8:powers:def}{base}: precisamente \cref{thm:g8:midpoints:direct}. Tales é
o teorema do ponto médio livre da razão $\frac12$.

\textbf{13.} No triângulo $ ACD $, a \omterm{def:g6:lines:objects}{linha} $(EO)$ é \omterm{def:g6:lines:perp}{paralelo}
para a \omterm{def:g8:powers:def}{base} $(CD)$, com $\frac{AO}{AC} = \frac{OA}{OA + OC} =
\frac{4}{4 + 6} = \frac25$. Tales:
$\frac{EO}{DC} = \frac{AO}{AC} = \frac25$, então
$ EO = \frac25 \times 6 = 2.4$.

\textbf{14.} No triângulo $ BDC $, da mesma maneira
$\frac{BO}{BD} = \frac25$ e $ OF = \frac25 \times 6 = 2.4$.
Por isso
\[
EF = EO + OF = 4.8
= \frac{2 \times 4 \times 6}{4 + 6} :
\]
a \omterm{pb:g8:speed:1}{média harmônica} das \omterm{def:g8:powers:def}{bases} --- a média das viagens de ida e volta
(\cref{pb:g8:speed:1}), desenhado em um trapézio.

\textbf{15.} Aritmética: $5$; geométrico:
$\sqrt{24} \approx 4.90$; harmônico: $4.8$. Ordem:
$4.8 < 4.90\ldots < 5$, ou seja,\ harmônico $<$ geométrico $<$
aritmética --- com igualdade apenas para \omterm{def:g8:powers:def}{bases} iguais. No
figura: as medidas da \omterm{def:g6:lines:objects}{linha} média $5$, o \omterm{def:g6:lines:perp}{paralelo} através do
cruzamento diagonal $4.8$, e o corte de similaridade
$\sqrt{24}$, todos empilhados entre as \omterm{def:g8:powers:def}{bases} nessa ordem. O
foram comprovadas desigualdades \cref{pb:g8:cosine:1} (geométrico
$\leq $ aritmética, por círculo e por identidade) e
\cref{pb:g8:speed:1} (harmônico $\leq $ aritmética); harmônico
$\leq $ geométrico segue combinando-os ($ H \times A = G^2$,
provou lá também).
\end{solution}
