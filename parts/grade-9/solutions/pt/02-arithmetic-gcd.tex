\section*{Capítulo \ref{ch:g9:arith} --- Aritmética: Divisores e Números Primos}

\begin{solution}{exo:g9:arith:1}

\omterm{def:g5:division:multiple}{Divisores} de $45$: $1, 3, 5, 9, 15, 45$.

\end{solution}

\begin{solution}{exo:g9:arith:2}
$2\,346$ termina em $6$: \omterm{def:g6:wholes:divisible}{divisível} por $2$, não por $5$. Dígito \omterm{def:g1:addition:def}{soma}
$2 + 3 + 4 + 6 = 15$: \omterm{def:g6:wholes:divisible}{divisível} por $3$, não por $9$. Últimos dois dígitos
$46$, e $46 = 4 \times 11 + 2$ não é \omterm{def:g6:wholes:divisible}{divisível} por $4$: $2\,346$ é
não \omterm{def:g6:wholes:divisible}{divisível} por $4$.
\end{solution}

\begin{solution}{exo:g9:arith:4}
$101$: teste os \omterm{def:g9:arith:prime}{primos} $ p $ com $ p^2 \leq 101$, ou seja\ $2, 3, 5, 7$. Nenhum
divide $101$ (\omterm{def:g3:numbers:evenodd}{ímpar}, dígito \omterm{def:g1:addition:def}{soma} $2$, não termina em $0/5$,
$101 = 7 \times 14 + 3$): $101$ é \omterm{def:g9:arith:prime}{primo}.

$91 = 7 \times 13$: não \omterm{def:g9:arith:prime}{primo}.

$143 = 11 \times 13$: não \omterm{def:g9:arith:prime}{primo}.
\end{solution}

\begin{solution}{exo:g9:arith:5}
Comum \omterm{def:g5:division:multiple}{divisores} de $48$ e $60$: \omterm{def:g5:division:multiple}{divisores} de 48 são $1, 2, 3, 4, 6, 8,
12, 16, 24, 48$; \omterm{def:g5:division:multiple}{divisores} de $60$ são $1, 2, 3, 4, 5, 6, 10, 12, 15, 20,
30, 60$; os comuns são $1, 2, 3, 4, 6, 12$, então $\gcd(48,60) = 12$.

Por fatoração: $48 = 2^4 \times 3$ e $60 = 2^2 \times 3 \times 5$;
\omterm{def:g9:arith:prime}{primos} comuns com menor \omterm{def:g8:powers:def}{expoentes}: $2^2 \times 3 = 12$.
\end{solution}

\begin{solution}{exo:g9:arith:6}
$\gcd(255, 154)$:
\begin{align*}
255 &= 154 \times 1 + 101, \\
154 &= 101 \times 1 + 53, \\
101 &= 53 \times 1 + 48, \\
53 &= 48 \times 1 + 5, \\
48 &= 5 \times 9 + 3, \\
5 &= 3 \times 1 + 2, \\
3 &= 2 \times 1 + 1, \\
2 &= 1 \times 2 + 0 .
\end{align*}
Último diferente de \omterm{def:g1:counting:zero}{zero} \omterm{def:g3:division:remainder}{restante}: $\gcd(255, 154) = 1$ (eles são \omterm{def:g9:arith:gcd}{coprime}).

$\gcd(1053, 325)$:
\begin{align*}
1053 &= 325 \times 3 + 78, \\
325 &= 78 \times 4 + 13, \\
78 &= 13 \times 6 + 0 .
\end{align*}
$\gcd(1053, 325) = 13$.
\end{solution}

\begin{solution}{exo:g9:arith:7}
Algoritmo euclidiano: $588 = 504 \times 1 + 84$; $504 = 84 \times 6 + 0$:
$\gcd(588, 504) = 84$. Então
\[
\frac{588}{504} = \frac{588 \div 84}{504 \div 84} = \frac{7}{6},
\]
que é irredutível.
\end{solution}

\begin{solution}{exo:g9:arith:8}
O número de buquês deve \omterm{def:g9:arith:divisor}{dividir} ambos $84$ e $126$; o maior
possível é $\gcd(84, 126)$. Fatorações: $84 = 2^2 \times 3 \times
7$, $126 = 2 \times 3^2 \times 7$, então o GCD é $2 \times 3 \times 7 =
42$. Ela pode fazer $42$ buquês, cada um contendo
$\frac{84}{42} = 2$ rosas e $\frac{126}{42} = 3$ tulipas.
\end{solution}

\begin{solution}{exo:g9:arith:9}
Precisamos do menos comum \omterm{def:g9:arith:divisor}{múltiplo}. $24 = 2^3 \times 3$ e
$36 = 2^2 \times 3^2$; tomando cada \omterm{def:g9:arith:prime}{primo} com o \emph{maior}
\omterm{def:g8:powers:def}{expoente}: $\lcm = 2^3 \times 3^2 = 72$. As próximas balsas partem juntas
$72$ minutos depois das 8h, às 9h12.
\end{solution}

\begin{solution}{exo:g9:arith:10}
\emph{1.} Qualquer comum \omterm{def:g5:division:multiple}{divisor} $ d $ de $ n $ e $ n+1$ também divide deles
\omterm{ex:g1:subtraction:difference}{diferença} $(n+1) - n = 1$, então $ d = 1$: $\gcd(n, n+1) = 1$.

\emph{2.} UM \omterm{def:g9:fractions:fraction}{fração} é irredutível exatamente quando \omterm{def:g4:fractions:def}{numerador} e
\omterm{def:g4:fractions:def}{denominador} são \omterm{def:g9:arith:gcd}{coprime}, que é o caso $ n $ e $ n + 1$ por parte
1.
\end{solution}

\begin{solution}{pb:g9:arith:1}
\textbf{1.} Preencha o $3$ e despeje-o no $5$ (conteúdo
$0/3 \to 3$ no grande). Preencha o $3$ novamente e despeje no
$5$ até ficar cheio: o jarro grande leva apenas $2$ mais, saindo
\[
3 - 2 = 1 \text{ L in the small jug.}
\]
Movimentos: preencher $3$; derramar $3 \to 5$; preencher $3$; derramar $3 \to 5$.

\textbf{2.} Preencha o $5$; despeje no $3$ (folhas $2$ no
grande); esvazie o $3$; despeje o $2$ para dentro do $3$; preencha o $5$;
despeje no $3$ até ficar cheio --- é preciso $1$, saindo
$\mathbf{4}$ L no jarro grande. Seis movimentos.

\textbf{3.} Todos eles: $1$ (pergunta 1), $2$ (depois de dois movimentos
da questão 2), $3$ e $5$ (recheios individuais), $4$ (pergunta
2), $6 = 3 + 3$ (um pequeno jarro cheio mais $3$ derramado no
grande), $7 = 5 + 2$, $8 = 5 + 3$ (ambos cheios). Cada valor total
de $1$ para $8$ L é mensurável com o $5$ e o $3$.

\textbf{4.} Início: ambos os jarros seguram $0$, um \omterm{def:g9:arith:divisor}{múltiplo} de $2$.
O preenchimento define um conteúdo para $6$ ou $4$: \omterm{def:g3:numbers:evenodd}{par}. Esvaziar define-o para
$0$: \omterm{def:g3:numbers:evenodd}{par}. O derramamento move um pouco de água entre jarros cujo conteúdo
eram \omterm{def:g3:numbers:evenodd}{par}, e a quantidade derramada é um \omterm{ex:g1:subtraction:difference}{diferença} de números \omterm{def:g3:numbers:evenodd}{pares} (espaço \omterm{def:g3:division:remainder}{restante} ou quantidade disponível): todos os conteúdos ficam
\omterm{def:g3:numbers:evenodd}{par} para sempre. Um \omterm{def:g3:numbers:evenodd}{ímpar} alvo como $1$ L está inacessível.

\textbf{5.} $\gcd(6, 4) = 2$: apenas \omterm{def:g3:numbers:evenodd}{par} valores --- confirmados por
questão 4. $\gcd(5, 3) = 1$: todo valor inteiro é permitido por
a lei, e a questão 3 realizou todas elas. O GCD é exatamente
a unidade de medida dos jarros.

\textbf{6.} $91 = 1 \times 65 + 26$; $65 = 2 \times 26 + 13$;
$26 = 2 \times 13 + 0$: $\gcd(91, 65) = 13$. E
$2\,026 = 44 \times 46 + 2$; $46 = 23 \times 2 + 0$:
$\gcd(2\,026, 46) = 2$.

\textbf{7.} Derramando $5$ para dentro do $13$-jug repetidamente: depois das duas
enche o grande jarro $10$; o terceiro recheio cabe apenas
$3$, saindo $5 - 3 = 2$ no pequeno jarro --- o \omterm{def:g3:division:remainder}{restante} de
$13$ por $5$ era $3$, e os montantes $3$ (espaço) e $2$
(sobras) são exatamente os números de Euclides ($13 = 2 \times 5 + 3$,
$5 = 1 \times 3 + 2$). Continuando, $3 - 2 = 1$ aparece: o
próximo algoritmo \omterm{def:g3:division:remainder}{restante}. A fonte executa Euclides
divisões com água.

\textbf{8.} $\gcd(13, 5) = 1$, então a lei da questão 5 permite
cada valor total --- e a cascata da pergunta 7, na verdade
produzido $1$ L. Sim.

\textbf{9.} Um comum \omterm{def:g5:division:multiple}{divisor} de $ n $ e $2n + 1$ divide
$2n + 1 - 2 \times n = 1$: deve ser $1$. Por isso
$\gcd(n, 2n+1) = 1$ sempre.

\textbf{10.} Ônibus A: 7h12, 7h24, 7h36, 7h48, 8h\,\dots; ônibus B:
7:18, 7:36, 7:54\,\dots\ Primeira saída comum: 7h36, após
$36$ minutos --- o primeiro comum \omterm{def:g9:arith:divisor}{múltiplo} de $12$ e $18$.
Lei: $36 \times \gcd(12, 18) = 36 \times 6 = 216 = 12 \times
18$. Para $5$ e $3$: primeiro comum \omterm{def:g9:arith:divisor}{múltiplo} $15$, e
$15 \times \gcd(5,3) = 15 \times 1 = 15 = 5 \times 3$.

\textbf{11.} Com um $17$ ciclo de anos: a próxima coincidência é
o primeiro comum \omterm{def:g9:arith:divisor}{múltiplo} de $17$ e $4$; desde
$\gcd(17, 4) = 1$, aquilo é $17 \times 4 = 68$ anos --- o
as cigarras atingem o pico uma vez em quatro emergências. Com um
$16$ ciclo de anos: $16$ é um \omterm{def:g9:arith:divisor}{múltiplo} de $4$, então \emph{todo}
emergência atinge um pico. A duração do ciclo principal não compartilha nenhum fator
com qualquer ciclo predador mais curto, levando as coincidências tão longe
tão distante quanto possível: aritmética como camuflagem.

\textbf{12.} Quatro opções de \omterm{def:g8:powers:def}{expoente} para $2$, três para $3$,
dois para $5$: $4 \times 3 \times 2 = 24$ \omterm{def:g5:division:multiple}{divisores}.

\textbf{13.} Se $ m = 2^{a} \times 3^{b} \times \cdots $, então
$ m^2 = 2^{2a} \times 3^{2b} \times \cdots $: todo \omterm{def:g8:powers:def}{expoente} é
dobrou, portanto \omterm{def:g3:numbers:evenodd}{par}. Em $360 = 2^3 \times 3^2 \times 5$, o
\omterm{def:g8:powers:def}{expoentes} de $2$ e de $5$ são \omterm{def:g3:numbers:evenodd}{ímpar}: $360$ não é perfeito
quadrado.

\textbf{14.} UM \omterm{def:g5:division:multiple}{divisor} é seu próprio parceiro exatamente quando
$ d = \frac nd $, ou seja\ $ n = d^2$: apenas os quadrados têm tal
meio \omterm{def:g5:division:multiple}{divisor}. Para todos os outros $ n $ o \omterm{def:g5:division:multiple}{divisores} dividido em
\omterm{def:g3:numbers:evenodd}{pares}, um \omterm{def:g3:numbers:evenodd}{par} \omterm{def:g7:stats:frequency}{contar}. Então: número \omterm{def:g3:numbers:evenodd}{ímpar} de \omterm{def:g5:division:multiple}{divisores}
$\Leftrightarrow $ quadrado perfeito. Verificar: $36$ tem \omterm{def:g5:division:multiple}{divisores}
$1, 2, 3, 4, 6, 9, 12, 18, 36$ --- nove deles, \omterm{def:g3:numbers:evenodd}{ímpar}, e
$36 = 6^2$; enquanto $360$ tem $24$ (questão 12), \omterm{def:g3:numbers:evenodd}{par}, e é
nenhum quadrado (questão 13).

\textbf{15.} Armário $ n $ é alternado uma vez por cada aluno $ k $
cujo número divide $ n $: ao todo, quantas vezes $ n $ tem
\omterm{def:g5:division:multiple}{divisores}. Um armário termina \emph{abrir} quando ele é alternado número \omterm{def:g3:numbers:evenodd}{ímpar} de vezes --- pela pergunta 14, exatamente quando $ n $ é um
quadrado perfeito. Armários abertos: $1, 4, 9, 16, 25, 36, 49, 64,
81, 100$ --- dez deles.
\end{solution}
