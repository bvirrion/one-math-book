\section*{Chapter \ref{ch:g9:linfunc} --- Linear and Affine Functions}

\begin{solution}{exo:g9:linfunc:1}
Linear: $f$ and $k$ (form $ax$). Affine but not linear: $g$ (\omterm{prop:g9:linfunc:affinegraph}{slope} $3$,
intercept $-5$) and $m$ (constant, \omterm{prop:g9:linfunc:affinegraph}{slope} $0$). Neither: $h$, because of
the square.
\end{solution}

\begin{solution}{exo:g9:linfunc:2}
$f(0) = -3$; $f(4) = 5$; $f(-1) = -5$. And $f(x) = 9$ means
$2x - 3 = 9$, so $2x = 12$ and $x = 6$.
\end{solution}

\begin{solution}{exo:g9:linfunc:3}
$a = \frac{15}{6} = 2.5$, so $f(x) = 2.5x$. Then $f(10) = 25$ and
$f(-2) = -5$.
\end{solution}

\begin{solution}{exo:g9:linfunc:4}
$f$ passes through the origin with \omterm{prop:g9:linfunc:affinegraph}{slope} $2$; $g$ has the same \omterm{prop:g9:linfunc:affinegraph}{slope} but
crosses the $y$-axis at $3$; $h$ decreases from $(0, 4)$ with \omterm{prop:g9:linfunc:affinegraph}{slope}
$-1$. The graphs of $f$ and $g$ are \emph{\omterm{def:g6:lines:perp}{parallel lines}} (same \omterm{prop:g9:linfunc:affinegraph}{slope},
different intercepts).
\end{solution}

\begin{solution}{exo:g9:linfunc:5}
For $f$: \omterm{prop:g9:linfunc:affinegraph}{slope} $\frac{11 - 5}{3 - 1} = 3$, then $5 = 3 \times 1 + b$
gives $b = 2$: $f(x) = 3x + 2$.

For $g$: \omterm{prop:g9:linfunc:affinegraph}{slope} $\frac{0 - 4}{2 - 0} = -2$, and $g(0) = 4$ is already the
intercept: $g(x) = -2x + 4$.
\end{solution}

\begin{solution}{exo:g9:linfunc:6}
Increase of $25\%$: $120 \times 1.25 = 150$.

Discount of $40\%$: $65 \times 0.6 = 39$.

Original price $p$ with $p \times 0.75 = 69$:
$p = \frac{69}{0.75} = 92$.
\end{solution}

\begin{solution}{exo:g9:linfunc:7}
\emph{1.} $f(x) = 0.05x + 10$.

\emph{2.} Plan B is cheaper when $25 < 0.05x + 10$, i.e.\
$15 < 0.05x$, i.e.\ $x > 300$. From $301$ minutes ($5$ hours) per month
onward, plan B wins.
\end{solution}

\begin{solution}{exo:g9:linfunc:8}
\emph{1.} After one hour: $2000 \times 1.1 = 2200$. After two hours:
$2200 \times 1.1 = 2420$.

\emph{2.} The second increase applies to $2200$, not to $2000$: growing
by $10\%$ twice multiplies by $1.1^2 = 1.21$, a $21\%$ increase, not
$20\%$.
\end{solution}

\begin{solution}{exo:g9:linfunc:9}
\emph{1.} \omterm{prop:g9:linfunc:affinegraph}{Slope}: $\frac{1 - 3}{6 - 2} = \frac{-2}{4} = -0.5$: the
function is decreasing.

\emph{2.} $f(x) = -0.5x + b$ with $f(2) = 3$: $-1 + b = 3$, so $b = 4$
and $f(x) = -0.5x + 4$. Output $0$: $-0.5x + 4 = 0$ gives $x = 8$.
\end{solution}

\begin{solution}{exo:g9:linfunc:10}
\emph{1.} The two changes multiply the price by
\[
\left(1 + \frac{t}{100}\right)\left(1 - \frac{t}{100}\right)
= 1 - \frac{t^2}{10000}
\]
(third identity with $a = 1$, $b = \frac{t}{100}$).

\emph{2.} For $0 < t \leq 100$, $\frac{t^2}{10000} > 0$, so the factor
is less than $1$: the final price is lower than the original. For
$t = 10$: factor $1 - \frac{100}{10000} = 0.99$, i.e.\ an overall
decrease of $1\%$.
\end{solution}
