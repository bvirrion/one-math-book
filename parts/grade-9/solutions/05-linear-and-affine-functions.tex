\section*{Chapter \ref{ch:g9:linfunc} --- Linear and Affine Functions}

\begin{solution}{exo:g9:linfunc:1}
Linear: $f$ and $k$ (form $ax$). Affine but not linear: $g$ (\omterm{prop:g9:linfunc:affinegraph}{slope} $3$,
intercept $-5$) and $m$ (constant, \omterm{prop:g9:linfunc:affinegraph}{slope} $0$). Neither: $h$, because of
the square.
\end{solution}

\begin{solution}{exo:g9:linfunc:2}
$f(0) = -3$; $f(4) = 5$; $f(-1) = -5$. And $f(x) = 9$ means
$2x - 3 = 9$, so $2x = 12$ and $x = 6$.
\end{solution}

\begin{solution}{exo:g9:linfunc:3}
$a = \frac{15}{6} = 2.5$, so $f(x) = 2.5x$. Then $f(10) = 25$ and
$f(-2) = -5$.
\end{solution}

\begin{solution}{exo:g9:linfunc:4}
$f$ passes through the origin with \omterm{prop:g9:linfunc:affinegraph}{slope} $2$; $g$ has the same \omterm{prop:g9:linfunc:affinegraph}{slope} but
crosses the $y$-axis at $3$; $h$ decreases from $(0, 4)$ with \omterm{prop:g9:linfunc:affinegraph}{slope}
$-1$. The graphs of $f$ and $g$ are \emph{\omterm{def:g6:lines:perp}{parallel lines}} (same \omterm{prop:g9:linfunc:affinegraph}{slope},
different intercepts).
\end{solution}

\begin{solution}{exo:g9:linfunc:5}
For $f$: \omterm{prop:g9:linfunc:affinegraph}{slope} $\frac{11 - 5}{3 - 1} = 3$, then $5 = 3 \times 1 + b$
gives $b = 2$: $f(x) = 3x + 2$.

For $g$: \omterm{prop:g9:linfunc:affinegraph}{slope} $\frac{0 - 4}{2 - 0} = -2$, and $g(0) = 4$ is already the
intercept: $g(x) = -2x + 4$.
\end{solution}

\begin{solution}{exo:g9:linfunc:6}
Increase of $25\%$: $120 \times 1.25 = 150$.

Discount of $40\%$: $65 \times 0.6 = 39$.

Original price $p$ with $p \times 0.75 = 69$:
$p = \frac{69}{0.75} = 92$.
\end{solution}

\begin{solution}{exo:g9:linfunc:7}
\emph{1.} $f(x) = 0.05x + 10$.

\emph{2.} Plan B is cheaper when $25 < 0.05x + 10$, i.e.\
$15 < 0.05x$, i.e.\ $x > 300$. From $301$ minutes ($5$ hours) per month
onward, plan B wins.
\end{solution}

\begin{solution}{exo:g9:linfunc:8}
\emph{1.} After one hour: $2000 \times 1.1 = 2200$. After two hours:
$2200 \times 1.1 = 2420$.

\emph{2.} The second increase applies to $2200$, not to $2000$: growing
by $10\%$ twice multiplies by $1.1^2 = 1.21$, a $21\%$ increase, not
$20\%$.
\end{solution}

\begin{solution}{exo:g9:linfunc:9}
\emph{1.} \omterm{prop:g9:linfunc:affinegraph}{Slope}: $\frac{1 - 3}{6 - 2} = \frac{-2}{4} = -0.5$: the
function is decreasing.

\emph{2.} $f(x) = -0.5x + b$ with $f(2) = 3$: $-1 + b = 3$, so $b = 4$
and $f(x) = -0.5x + 4$. Output $0$: $-0.5x + 4 = 0$ gives $x = 8$.
\end{solution}

\begin{solution}{exo:g9:linfunc:10}
\emph{1.} The two changes multiply the price by
\[
\left(1 + \frac{t}{100}\right)\left(1 - \frac{t}{100}\right)
= 1 - \frac{t^2}{10000}
\]
(third identity with $a = 1$, $b = \frac{t}{100}$).

\emph{2.} For $0 < t \leq 100$, $\frac{t^2}{10000} > 0$, so the factor
is less than $1$: the final price is lower than the original. For
$t = 10$: factor $1 - \frac{100}{10000} = 0.99$, i.e.\ an overall
decrease of $1\%$.
\end{solution}

\begin{solution}{pb:g9:linfunc:1}
\textbf{1.} \omterm{prop:g9:linfunc:affinegraph}{Slope}: $\frac{f(100) - f(0)}{100 - 0} =
\frac{212 - 32}{100} = 1.8$; intercept $f(0) = 32$. Hence
\[
f(C) = 1.8\,C + 32 .
\]

\textbf{2.} $f(20) = 68\,^\circ$F (the famous mild day);
$f(37) = 98.6\,^\circ$F; $f(-10) = 14\,^\circ$F.

\textbf{3.} From $F = 1.8C + 32$:
$C = \frac{F - 32}{1.8}$. Then $50\,^\circ$F
$\to \frac{18}{1.8} = 10\,^\circ$C, and $98.6\,^\circ$F
$\to \frac{66.6}{1.8} = 37\,^\circ$C.

\textbf{4.} $f(0) = 32 \neq 0$, and $f(20) = 68$ is not the
double of $f(10) = 50$: not linear. The graph is a straight
line missing the origin: $f$ is \emph{affine}, not linear.

\textbf{5.} $1.8x + 32 = x$ gives $0.8x = -32$, so $x = -40$:
at forty below, both thermometers read $-40$. On the graph, the
line of $f$ crosses the diagonal line $y = x$ exactly there.

\textbf{6.} $p_A(k) = 1.5k + 3$ and $p_B(k) = 2k$. For $4$ km:
$9$ euros against $8$: B cheaper. For $8$ km: $15$ against
$16$: A cheaper.

\textbf{7.} $1.5k + 3 = 2k$ gives $k = 6$: at six kilometers
both cost $12$ euros. Graphically, B's line (through the
origin, steeper) starts below A's and crosses it at
$(6, 12)$: B wins short rides, A wins long ones.

\textbf{8.} $N = 7 \times 20 - 30 = 110$ chirps per minute. From
$82 = 7T - 30$: $T = \frac{112}{7} = 16\,^\circ$C.

\textbf{9.} $v(t) = 600 - 15t$. Worth $240$:
$600 - 15t = 240$, $t = 24$ months. Worth $0$: $t = 40$ months.
Beyond $40$ months the formula goes negative --- but a phone's
value stops at \omterm{def:g1:counting:zero}{zero}: every affine model has a domain where it
makes sense.

\textbf{10.} Composing: $\times 1.25$ then $\times 0.8$
multiplies by $1.25 \times 0.8 = 1$: the price returns exactly
to its start. No miracle: the percentages act on different
references (the cut applies to the raised price), and here the
multipliers happen to cancel --- the general story, with its
built-in loss $1 - \frac{t^2}{10\,000}$, is
\cref{exo:g9:linfunc:10}.

\textbf{11.} \omterm{prop:g9:linfunc:affinegraph}{Slope}: $\frac{17 - 20}{30} = -0.1$ cm per minute;
$h(t) = 20 - 0.1t$. It dies at $h(t) = 0$: $t = 200$ minutes
--- three hours and twenty minutes of light.

\textbf{12.} The second started $10$ km ahead (intercept); the
first is faster ($24 > 18$ km/h). Catch-up:
$24t = 10 + 18t$ gives $6t = 10$, $t = \frac53$ h $= 1$ h $40$
min, at kilometer $24 \times \frac53 = 40$.

\textbf{13.} $f(x) + g(x) = (2 - 0.5)x + (3 + 7) = 1.5x + 10$:
affine, with \omterm{prop:g9:linfunc:affinegraph}{slope} the \emph{\omterm{def:g1:addition:def}{sum} of the \omterm{prop:g9:linfunc:affinegraph}{slopes}} and intercept
the \omterm{def:g1:addition:def}{sum} of the intercepts --- clear from collecting like terms
in $(mx + p) + (m'x + p')$.

\textbf{14.} If $m \neq 0$: exactly one solution,
$x = -\frac pm$ --- a non-horizontal line crosses the axis
once. If $m = 0$, $p \neq 0$: no solution --- a horizontal line
above or below the axis never touches it. If $m = 0$, $p = 0$:
every $x$ is a solution --- the line \emph{is} the axis.

\textbf{15.} First triple: rates $\frac{9 - 5}{3 - 1} = 2$ and
$\frac{17 - 9}{7 - 3} = 2$: equal, the points are aligned (on
$y = 2x + 3$). Second triple: $\frac{9-5}{2} = 2$ but
$\frac{16 - 9}{3} = \frac73 \neq 2$: not aligned. \omterm{def:g9:linfunc:affine}{Affine
functions} are exactly those with a constant rate of change ---
one number, the \omterm{prop:g9:linfunc:affinegraph}{slope}, tells the whole story; for curves the
rate itself changes from point to point, and chasing it leads
to the derivative, in the High School volume.
\end{solution}
