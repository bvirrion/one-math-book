\section*{Chapter \ref{ch:g9:solids} --- Solids and Volumes}

\begin{solution}{exo:g9:solids:1}
\omterm{def:g5:solids:def}{Box}: $V = 4 \times 5 \times 12 = 240$.

\omterm{def:g7:areas:prism}{Cylinder}: $V = \pi r^2 h = \pi \times 9 \times 7 = 63\pi \approx 198$.
\end{solution}

\begin{solution}{exo:g9:solids:2}
\omterm{def:g8:solids:def}{Cone}: $V = \frac13 \pi r^2 h = \frac13 \pi \times 25 \times 12 =
100\pi \approx 314$.

\omterm{def:g8:solids:def}{Pyramid}: base \omterm{def:g6:measure:area}{area} $B = 8 \times 3 = 24$, so
$V = \frac13 \times 24 \times 10 = 80$.
\end{solution}

\begin{solution}{exo:g9:solids:3}
\omterm{def:g6:measure:volume}{Volume}: $V = \frac43 \pi \times 6^3 = \frac43 \pi \times 216 = 288\pi$.

Surface \omterm{def:g6:measure:area}{area}: $4\pi \times 6^2 = 144\pi$.
\end{solution}

\begin{solution}{exo:g9:solids:4}
\omterm{def:g3:shapes:shapes}{Radius} of the section: $\sqrt{r^2 - d^2} = \sqrt{100 - 64} = \sqrt{36}
= 6$.
\end{solution}

\begin{solution}{exo:g9:solids:5}
\omterm{def:g9:thales:scaling}{Scale factor} from the whole \omterm{def:g8:solids:def}{cone} to the small one:
$k = \frac{9}{12} = \frac34$. Section \omterm{def:g3:shapes:shapes}{radius}: $8 \times \frac34 = 6$.

\omterm{def:g6:measure:volume}{Volume} of the whole \omterm{def:g8:solids:def}{cone}: $\frac13\pi \times 64 \times 12 = 256\pi$.
\omterm{def:g6:measure:volume}{Volume} of the small \omterm{def:g8:solids:def}{cone}: $256\pi \times \left(\frac34\right)^3 =
256\pi \times \frac{27}{64} = 108\pi \approx 339$.
\end{solution}

\begin{solution}{exo:g9:solids:6}
\omterm{def:g6:measure:volume}{Volume} of the ball: $\frac43\pi \times 2^3 = \frac{32\pi}{3}$ cm$^3$.
This \omterm{def:g6:measure:volume}{volume} spreads over the base \omterm{def:g6:measure:area}{area} $\pi \times 3^2 = 9\pi$ cm$^2$,
so the level rises by
\[
\frac{32\pi/3}{9\pi} = \frac{32}{27} \approx 1.2 \text{ cm}
\]
(about $12$ mm).
\end{solution}

\begin{solution}{exo:g9:solids:7}
The small mold is the reduction of the large one by $k = \frac36 =
\frac12$, so its \omterm{def:g6:measure:volume}{volume} is $\left(\frac12\right)^3 = \frac18$ of the
large one: the recipe fills $8$ small molds.
\end{solution}

\begin{solution}{exo:g9:solids:8}
$V = \frac13 \times 230^2 \times 147 = \frac13 \times 52\,900 \times 147
= 52\,900 \times 49 = 2\,592\,100$ m$^3$ --- about $2.6$ million cubic
meters.
\end{solution}

\begin{solution}{exo:g9:solids:9}
\emph{1.} The water forms a \omterm{def:g8:solids:def}{cone} scaled by $k = \frac12$ (apex down,
\omterm{def:g3:division:half}{half} the height), so its \omterm{def:g6:measure:volume}{volume} is $\left(\frac12\right)^3 = \frac18$
of the funnel's: one eighth, much less than \omterm{def:g3:division:half}{half}!

\emph{2.} Water up to height $d$ forms a \omterm{def:g8:solids:def}{cone} scaled by
$k = \frac{d}{12}$, of \omterm{def:g6:measure:volume}{volume} $k^3$ times the funnel's. \omterm{def:g3:division:half}{Half} the \omterm{def:g6:measure:volume}{volume}
means $k^3 = \frac12$, i.e.\
$\left(\frac{d}{12}\right)^3 = \frac12$. Then
$\frac{d}{12} = \sqrt[3]{0.5} \approx 0.7937$, so
$d \approx 9.5$ cm: the second \omterm{def:g3:division:half}{half} of the \omterm{def:g6:measure:volume}{volume} occupies only the top
$2.5$ cm --- the wide part of the \omterm{def:g8:solids:def}{cone} holds most of the water.
\end{solution}

\begin{solution}{pb:g9:solids:1}
\textbf{1.} $V_{\text{cyl}} = \pi r^2 \times 2r = 2\pi r^3$.

\textbf{2.} $\dfrac{V_{\text{sphere}}}{V_{\text{cyl}}}
= \dfrac{\frac43 \pi r^3}{2 \pi r^3} = \dfrac23$. The $\pi$ and
the $r^3$ cancel: the proportion is \emph{universal} --- true
for a marble and for a planet. A relation between shapes, not
between numbers: exactly the kind of truth worth carving in
stone.

\textbf{3.} Sphere: $4\pi r^2$. \omterm{def:g7:areas:prism}{Cylinder}: lateral
$2\pi r \times 2r = 4\pi r^2$, plus two lids
$2 \times \pi r^2$: total $6\pi r^2$. Ratio:
$\frac{4\pi r^2}{6\pi r^2} = \frac23$ --- the same two thirds,
for surfaces. (And a bonus: the sphere's \omterm{def:g6:measure:area}{area} exactly equals
the \omterm{def:g7:areas:prism}{cylinder}'s lateral \omterm{def:g6:measure:area}{area}.)

\textbf{4.} Leftover: $2\pi r^3 - \frac43\pi r^3 =
\frac23 \pi r^3$. Two \omterm{def:g8:solids:def}{cones} of \omterm{def:g3:shapes:shapes}{radius} $r$ and height $r$:
$2 \times \frac13 \pi r^2 \times r = \frac23 \pi r^3$. Equal.

\textbf{5.} Ball: $\frac43 \pi \times 12^3 \approx 7\,200$
cm$^3$; \omterm{def:g5:solids:def}{box}: $2\pi \times 12^3 \approx 10\,900$ cm$^3$. Empty:
one third of the \omterm{def:g5:solids:def}{box}, about $33\,\%$ --- guaranteed by
question 2, whatever the ball's size.

\textbf{6.} \omterm{def:g3:division:half}{Half} a sphere:
$\frac12 \times \frac43 \pi \times 10^3 = \frac23 \pi \times
1\,000 \approx 2\,094$ cm$^3 \approx 2.1$ L.

\textbf{7.} \omterm{def:g8:solids:def}{Cone}: $\frac13 \pi \times 10^2 \times 10 =
\frac{1\,000\pi}{3} \approx 1\,047$ cm$^3$. Hemisphere:
$\frac{2\,000\pi}{3} \approx 2\,094$ cm$^3$. \omterm{def:g7:areas:prism}{Cylinder}:
$1\,000\pi \approx 3\,142$ cm$^3$. Ratios:
$\frac{1000\pi}{3} : \frac{2000\pi}{3} : \frac{3000\pi}{3}
= 1 : 2 : 3$ exactly.

\textbf{8.} $\frac43 \pi \times 27 = \pi \times 9 \times h$
gives $h = \frac43 \times 3 = 4$ cm exactly.

\textbf{9.} Radii: $\frac{6\,371}{1\,737} \approx 3.67$.
\omterm{def:g6:measure:volume}{Volumes} \omterm{def:g7:prop:scale}{scale} as the \omterm{def:g5:solids:def}{cube} (\cref{thm:g9:solids:scaling}):
$3.67^3 \approx 49$. About fifty Moons fit in the Earth.

\textbf{10.} Surfaces \omterm{def:g7:prop:scale}{scale} as the square: $3.67^2 \approx
13.5$. Doubling a balloon's \omterm{def:g3:shapes:shapes}{radius} multiplies its \omterm{def:g6:measure:volume}{volume} by
$2^3 = 8$ but its surface by only $2^2 = 4$: as things grow,
\omterm{def:g6:measure:volume}{volume} (weight, content, heat produced) outruns surface (skin,
material, heat lost) --- the square--cube law.

\textbf{11.} Weight: $\times 10^3 = 1\,000$. Bone
\omterm{prop:g9:solids:sections}{cross-section}: $\times 10^2 = 100$. Pressure --- weight per
\omterm{def:g6:measure:area}{area} of bone: $\times \frac{1000}{100} = 10$. Bones built for
human pressure receive ten times more: the giant's skeleton
snaps under its own weight. Giants are geometrically
impossible; large animals need disproportionately thick bones
(compare an elephant's legs with a gazelle's).

\textbf{12.} Weight: $\div 100^3 = 10^6$. Strength (muscle
\omterm{prop:g9:solids:sections}{cross-section}): $\div 100^2 = 10^4$. Strength-to-weight:
$\times \frac{10^6}{10^4} = 100$. The ant lifting fifty times
its weight is not a super-athlete --- it is merely small; a
human shrunk to ant size could do likewise.

\textbf{13.} $\dfrac{S}{V} = \dfrac{4\pi r^2}{\frac43\pi r^3}
= \dfrac{3}{r}$: for $r = 1$ the ratio is $3$, for $r = 2$ it
is $1.5$ --- halving the size doubles the surface available
per unit of heat-producing \omterm{def:g6:measure:volume}{volume}. Small bodies bleed heat:
the mouse must eat constantly, the bear can hibernate, and the
baby --- small sphere, large $\frac Sr$ --- needs the hat.

\textbf{14.} $\left(\frac54\right)^3 = \frac{125}{64} \approx
1.95$: the big orange holds nearly \emph{twice} the fruit for
the same price. Buy \omterm{def:g3:shapes:shapes}{radius}.

\textbf{15.} The carving says: sphere $= \frac23$ of the
\omterm{def:g7:areas:prism}{cylinder} in \omterm{def:g6:measure:volume}{volume}, \emph{and} $\frac23$ of it in total
surface --- two exact, universal proportions linking the
roundest \omterm{def:g5:solids:def}{solid} to the simplest one. Cicero's answer: because a
theorem is the only monument that neither armies nor centuries
erode --- Archimedes' machines burned with Syracuse, but the
two thirds are still exactly two thirds.
\end{solution}
