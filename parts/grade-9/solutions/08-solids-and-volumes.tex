\section*{Chapter \ref{ch:g9:solids} --- Solids and Volumes}

\begin{solution}{exo:g9:solids:1}
\omterm{def:g5:solids:def}{Box}: $V = 4 \times 5 \times 12 = 240$.

\omterm{def:g7:areas:prism}{Cylinder}: $V = \pi r^2 h = \pi \times 9 \times 7 = 63\pi \approx 198$.
\end{solution}

\begin{solution}{exo:g9:solids:2}
\omterm{def:g8:solids:def}{Cone}: $V = \frac13 \pi r^2 h = \frac13 \pi \times 25 \times 12 =
100\pi \approx 314$.

\omterm{def:g8:solids:def}{Pyramid}: base \omterm{def:g6:measure:area}{area} $B = 8 \times 3 = 24$, so
$V = \frac13 \times 24 \times 10 = 80$.
\end{solution}

\begin{solution}{exo:g9:solids:3}
\omterm{def:g6:measure:volume}{Volume}: $V = \frac43 \pi \times 6^3 = \frac43 \pi \times 216 = 288\pi$.

Surface \omterm{def:g6:measure:area}{area}: $4\pi \times 6^2 = 144\pi$.
\end{solution}

\begin{solution}{exo:g9:solids:4}
\omterm{def:g3:shapes:shapes}{Radius} of the section: $\sqrt{r^2 - d^2} = \sqrt{100 - 64} = \sqrt{36}
= 6$.
\end{solution}

\begin{solution}{exo:g9:solids:5}
\omterm{def:g9:thales:scaling}{Scale factor} from the whole \omterm{def:g8:solids:def}{cone} to the small one:
$k = \frac{9}{12} = \frac34$. Section \omterm{def:g3:shapes:shapes}{radius}: $8 \times \frac34 = 6$.

\omterm{def:g6:measure:volume}{Volume} of the whole \omterm{def:g8:solids:def}{cone}: $\frac13\pi \times 64 \times 12 = 256\pi$.
\omterm{def:g6:measure:volume}{Volume} of the small \omterm{def:g8:solids:def}{cone}: $256\pi \times \left(\frac34\right)^3 =
256\pi \times \frac{27}{64} = 108\pi \approx 339$.
\end{solution}

\begin{solution}{exo:g9:solids:6}
\omterm{def:g6:measure:volume}{Volume} of the ball: $\frac43\pi \times 2^3 = \frac{32\pi}{3}$ cm$^3$.
This \omterm{def:g6:measure:volume}{volume} spreads over the base \omterm{def:g6:measure:area}{area} $\pi \times 3^2 = 9\pi$ cm$^2$,
so the level rises by
\[
\frac{32\pi/3}{9\pi} = \frac{32}{27} \approx 1.2 \text{ cm}
\]
(about $12$ mm).
\end{solution}

\begin{solution}{exo:g9:solids:7}
The small mold is the reduction of the large one by $k = \frac36 =
\frac12$, so its \omterm{def:g6:measure:volume}{volume} is $\left(\frac12\right)^3 = \frac18$ of the
large one: the recipe fills $8$ small molds.
\end{solution}

\begin{solution}{exo:g9:solids:8}
$V = \frac13 \times 230^2 \times 147 = \frac13 \times 52\,900 \times 147
= 52\,900 \times 49 = 2\,592\,100$ m$^3$ --- about $2.6$ million cubic
meters.
\end{solution}

\begin{solution}{exo:g9:solids:9}
\emph{1.} The water forms a \omterm{def:g8:solids:def}{cone} scaled by $k = \frac12$ (apex down,
\omterm{def:g3:division:half}{half} the height), so its \omterm{def:g6:measure:volume}{volume} is $\left(\frac12\right)^3 = \frac18$
of the funnel's: one eighth, much less than \omterm{def:g3:division:half}{half}!

\emph{2.} Water up to height $d$ forms a \omterm{def:g8:solids:def}{cone} scaled by
$k = \frac{d}{12}$, of \omterm{def:g6:measure:volume}{volume} $k^3$ times the funnel's. \omterm{def:g3:division:half}{Half} the \omterm{def:g6:measure:volume}{volume}
means $k^3 = \frac12$, i.e.\
$\left(\frac{d}{12}\right)^3 = \frac12$. Then
$\frac{d}{12} = \sqrt[3]{0.5} \approx 0.7937$, so
$d \approx 9.5$ cm: the second \omterm{def:g3:division:half}{half} of the \omterm{def:g6:measure:volume}{volume} occupies only the top
$2.5$ cm --- the wide part of the \omterm{def:g8:solids:def}{cone} holds most of the water.
\end{solution}
