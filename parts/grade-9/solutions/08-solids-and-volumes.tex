\section*{Chapter \ref{ch:g9:solids} --- Solids and Volumes}

\begin{solution}{exo:g9:solids:1}
Box: $V = 4 \times 5 \times 12 = 240$.

Cylinder: $V = \pi r^2 h = \pi \times 9 \times 7 = 63\pi \approx 198$.
\end{solution}

\begin{solution}{exo:g9:solids:2}
Cone: $V = \frac13 \pi r^2 h = \frac13 \pi \times 25 \times 12 =
100\pi \approx 314$.

Pyramid: base area $B = 8 \times 3 = 24$, so
$V = \frac13 \times 24 \times 10 = 80$.
\end{solution}

\begin{solution}{exo:g9:solids:3}
Volume: $V = \frac43 \pi \times 6^3 = \frac43 \pi \times 216 = 288\pi$.

Surface area: $4\pi \times 6^2 = 144\pi$.
\end{solution}

\begin{solution}{exo:g9:solids:4}
Radius of the section: $\sqrt{r^2 - d^2} = \sqrt{100 - 64} = \sqrt{36}
= 6$.
\end{solution}

\begin{solution}{exo:g9:solids:5}
Scale factor from the whole cone to the small one:
$k = \frac{9}{12} = \frac34$. Section radius: $8 \times \frac34 = 6$.

Volume of the whole cone: $\frac13\pi \times 64 \times 12 = 256\pi$.
Volume of the small cone: $256\pi \times \left(\frac34\right)^3 =
256\pi \times \frac{27}{64} = 108\pi \approx 339$.
\end{solution}

\begin{solution}{exo:g9:solids:6}
Volume of the ball: $\frac43\pi \times 2^3 = \frac{32\pi}{3}$ cm$^3$.
This volume spreads over the base area $\pi \times 3^2 = 9\pi$ cm$^2$,
so the level rises by
\[
\frac{32\pi/3}{9\pi} = \frac{32}{27} \approx 1.2 \text{ cm}
\]
(about $12$ mm).
\end{solution}

\begin{solution}{exo:g9:solids:7}
The small mold is the reduction of the large one by $k = \frac36 =
\frac12$, so its volume is $\left(\frac12\right)^3 = \frac18$ of the
large one: the recipe fills $8$ small molds.
\end{solution}

\begin{solution}{exo:g9:solids:8}
$V = \frac13 \times 230^2 \times 147 = \frac13 \times 52\,900 \times 147
= 52\,900 \times 49 = 2\,592\,100$ m$^3$ --- about $2.6$ million cubic
meters.
\end{solution}

\begin{solution}{exo:g9:solids:9}
\emph{1.} The water forms a cone scaled by $k = \frac12$ (apex down,
half the height), so its volume is $\left(\frac12\right)^3 = \frac18$
of the funnel's: one eighth, much less than half!

\emph{2.} Water up to height $d$ forms a cone scaled by
$k = \frac{d}{12}$, of volume $k^3$ times the funnel's. Half the volume
means $k^3 = \frac12$, i.e.\
$\left(\frac{d}{12}\right)^3 = \frac12$. Then
$\frac{d}{12} = \sqrt[3]{0.5} \approx 0.7937$, so
$d \approx 9.5$ cm: the second half of the volume occupies only the top
$2.5$ cm --- the wide part of the cone holds most of the water.
\end{solution}
