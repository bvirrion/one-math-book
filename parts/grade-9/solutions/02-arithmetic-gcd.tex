\begin{solution}{exo:g9:arith:2}

\end{solution}

\begin{solution}{exo:g9:arith:4}
$101$: test the primes $p$ with $p^2 \leq 101$, i.e.\ $2, 3, 5, 7$. None
\omterm{def:g9:arith:divisor}{divides} $101$ (\omterm{def:g3:numbers:evenodd}{odd}, digit \omterm{def:g1:addition:def}{sum} $2$, does not end in $0/5$,
$101 = 7 \times 14 + 3$): $101$ is prime.

$91 = 7 \times 13$: not prime.

$143 = 11 \times 13$: not prime.
\end{solution}

\begin{solution}{exo:g9:arith:5}
Common \omterm{def:g9:arith:divisor}{divisors} of $48$ and $60$: \omterm{def:g9:arith:divisor}{divisors} of 48 are $1, 2, 3, 4, 6, 8,
12, 16, 24, 48$; \omterm{def:g9:arith:divisor}{divisors} of $60$ are $1, 2, 3, 4, 5, 6, 10, 12, 15, 20,
30, 60$; the common ones are $1, 2, 3, 4, 6, 12$, so $\gcd(48,60) = 12$.

By factorization: $48 = 2^4 \times 3$ and $60 = 2^2 \times 3 \times 5$;
common primes with smaller \omterm{def:g8:powers:def}{exponents}: $2^2 \times 3 = 12$.
\end{solution}

\begin{solution}{exo:g9:arith:6}
$\gcd(255, 154)$:
\begin{align*}
255 &= 154 \times 1 + 101, \\
154 &= 101 \times 1 + 53, \\
101 &= 53 \times 1 + 48, \\
53 &= 48 \times 1 + 5, \\
48 &= 5 \times 9 + 3, \\
5 &= 3 \times 1 + 2, \\
3 &= 2 \times 1 + 1, \\
2 &= 1 \times 2 + 0 .
\end{align*}
Last nonzero \omterm{def:g3:division:remainder}{remainder}: $\gcd(255, 154) = 1$ (they are \omterm{def:g9:arith:gcd}{coprime}).

$\gcd(1053, 325)$:
\begin{align*}
1053 &= 325 \times 3 + 78, \\
325 &= 78 \times 4 + 13, \\
78 &= 13 \times 6 + 0 .
\end{align*}
$\gcd(1053, 325) = 13$.
\end{solution}

\begin{solution}{exo:g9:arith:7}
Euclidean algorithm: $588 = 504 \times 1 + 84$; $504 = 84 \times 6 + 0$:
$\gcd(588, 504) = 84$. Then
\[
\frac{588}{504} = \frac{588 \div 84}{504 \div 84} = \frac{7}{6},
\]
which is irreducible.
\end{solution}

\begin{solution}{exo:g9:arith:8}
The number of bouquets must divide both $84$ and $126$; the largest
possible is $\gcd(84, 126)$. Factorizations: $84 = 2^2 \times 3 \times
7$, $126 = 2 \times 3^2 \times 7$, so the GCD is $2 \times 3 \times 7 =
42$. She can make $42$ bouquets, each containing
$\frac{84}{42} = 2$ roses and $\frac{126}{42} = 3$ tulips.
\end{solution}

\begin{solution}{exo:g9:arith:9}
We need the least common \omterm{def:g9:arith:divisor}{multiple}. $24 = 2^3 \times 3$ and
$36 = 2^2 \times 3^2$; taking each prime with the \emph{larger}
\omterm{def:g8:powers:def}{exponent}: $\lcm = 2^3 \times 3^2 = 72$. The ferries next leave together
$72$ minutes after 8:00, at 9:12.
\end{solution}

\begin{solution}{exo:g9:arith:10}
\emph{1.} Any common \omterm{def:g9:arith:divisor}{divisor} $d$ of $n$ and $n+1$ also \omterm{def:g9:arith:divisor}{divides} their
\omterm{ex:g1:subtraction:difference}{difference} $(n+1) - n = 1$, so $d = 1$: $\gcd(n, n+1) = 1$.

\emph{2.} A \omterm{def:g9:fractions:fraction}{fraction} is irreducible exactly when its \omterm{def:g4:fractions:def}{numerator} and
\omterm{def:g4:fractions:def}{denominator} are \omterm{def:g9:arith:gcd}{coprime}, which is the case for $n$ and $n + 1$ by part
1.
\end{solution}

\begin{solution}{pb:g9:arith:1}
\textbf{1.} Fill the $3$ and pour it into the $5$ (contents
$0/3 \to 3$ in the big). Fill the $3$ again and pour into the
$5$ until it is full: the big jug takes only $2$ more, leaving
\[
3 - 2 = 1 \text{ L in the small jug.}
\]
Moves: fill $3$; pour $3 \to 5$; fill $3$; pour $3 \to 5$.

\textbf{2.} Fill the $5$; pour into the $3$ (leaves $2$ in the
big); empty the $3$; pour the $2$ into the $3$; fill the $5$;
pour into the $3$ until full --- it takes $1$, leaving
$\mathbf{4}$ L in the big jug. Six moves.

\textbf{3.} All of them: $1$ (question 1), $2$ (after two moves
of question 2), $3$ and $5$ (single fillings), $4$ (question
2), $6 = 3 + 3$ (a full small jug plus $3$ poured into the
big), $7 = 5 + 2$, $8 = 5 + 3$ (both full). Every whole amount
from $1$ to $8$ L is measurable with the $5$ and the $3$.

\textbf{4.} Start: both jugs hold $0$, a \omterm{def:g9:arith:divisor}{multiple} of $2$.
Filling sets a content to $6$ or $4$: \omterm{def:g3:numbers:evenodd}{even}. Emptying sets it to
$0$: \omterm{def:g3:numbers:evenodd}{even}. Pouring moves some water between jugs whose contents
were \omterm{def:g3:numbers:evenodd}{even}, and the poured quantity is a \omterm{ex:g1:subtraction:difference}{difference} of \omterm{def:g3:numbers:evenodd}{even
numbers} (space left, or amount available): all contents stay
\omterm{def:g3:numbers:evenodd}{even} forever. An \omterm{def:g3:numbers:evenodd}{odd} target like $1$ L is unreachable.

\textbf{5.} $\gcd(6, 4) = 2$: only \omterm{def:g3:numbers:evenodd}{even} amounts --- confirmed by
question 4. $\gcd(5, 3) = 1$: every whole amount is allowed by
the law, and question 3 realized them all. The GCD is exactly
the jugs' unit of measurement.

\textbf{6.} $91 = 1 \times 65 + 26$; $65 = 2 \times 26 + 13$;
$26 = 2 \times 13 + 0$: $\gcd(91, 65) = 13$. And
$2\,026 = 44 \times 46 + 2$; $46 = 23 \times 2 + 0$:
$\gcd(2\,026, 46) = 2$.

\textbf{7.} Pouring $5$ into the $13$-jug repeatedly: after two
fillings the big jug holds $10$; the third filling fits only
$3$, leaving $5 - 3 = 2$ in the small jug --- the \omterm{def:g3:division:remainder}{remainder} of
$13$ by $5$ was $3$, and the amounts $3$ (space) and $2$
(leftover) are exactly Euclid's numbers ($13 = 2 \times 5 + 3$,
$5 = 1 \times 3 + 2$). Continuing, $3 - 2 = 1$ appears: the
algorithm's next \omterm{def:g3:division:remainder}{remainder}. The fountain performs Euclid's
divisions with water.

\textbf{8.} $\gcd(13, 5) = 1$, so the law of question 5 allows
every whole amount --- and question 7's cascade actually
produced $1$ L. Yes.

\textbf{9.} A common \omterm{def:g9:arith:divisor}{divisor} of $n$ and $2n + 1$ \omterm{def:g9:arith:divisor}{divides}
$2n + 1 - 2 \times n = 1$: it must be $1$. Hence
$\gcd(n, 2n+1) = 1$ always.

\textbf{10.} Bus A: 7:12, 7:24, 7:36, 7:48, 8:00\,\dots; bus B:
7:18, 7:36, 7:54\,\dots\ First common departure: 7:36, after
$36$ minutes --- the first common \omterm{def:g9:arith:divisor}{multiple} of $12$ and $18$.
Law: $36 \times \gcd(12, 18) = 36 \times 6 = 216 = 12 \times
18$. For $5$ and $3$: first common \omterm{def:g9:arith:divisor}{multiple} $15$, and
$15 \times \gcd(5,3) = 15 \times 1 = 15 = 5 \times 3$.

\textbf{11.} With a $17$-year cycle: the next coincidence is
the first common \omterm{def:g9:arith:divisor}{multiple} of $17$ and $4$; since
$\gcd(17, 4) = 1$, that is $17 \times 4 = 68$ years --- the
cicadas meet the peak once in four emergences. With a
$16$-year cycle: $16$ is a \omterm{def:g9:arith:divisor}{multiple} of $4$, so \emph{every}
emergence hits a peak. A prime cycle length shares no factor
with any shorter predator cycle, pushing coincidences as far
apart as possible: arithmetic as camouflage.

\textbf{12.} Four choices of \omterm{def:g8:powers:def}{exponent} for $2$, three for $3$,
two for $5$: $4 \times 3 \times 2 = 24$ \omterm{def:g9:arith:divisor}{divisors}.

\textbf{13.} If $m = 2^{a} \times 3^{b} \times \cdots$, then
$m^2 = 2^{2a} \times 3^{2b} \times \cdots$: every \omterm{def:g8:powers:def}{exponent} is
doubled, hence \omterm{def:g3:numbers:evenodd}{even}. In $360 = 2^3 \times 3^2 \times 5$, the
\omterm{def:g8:powers:def}{exponents} of $2$ and of $5$ are \omterm{def:g3:numbers:evenodd}{odd}: $360$ is not a perfect
square.

\textbf{14.} A \omterm{def:g9:arith:divisor}{divisor} is its own partner exactly when
$d = \frac nd$, i.e.\ $n = d^2$: only squares have such a
middle \omterm{def:g9:arith:divisor}{divisor}. For all other $n$ the \omterm{def:g9:arith:divisor}{divisors} split into
pairs, an \omterm{def:g3:numbers:evenodd}{even} count. So: \omterm{def:g3:numbers:evenodd}{odd number} of \omterm{def:g9:arith:divisor}{divisors}
$\Leftrightarrow$ perfect square. Check: $36$ has \omterm{def:g9:arith:divisor}{divisors}
$1, 2, 3, 4, 6, 9, 12, 18, 36$ --- nine of them, \omterm{def:g3:numbers:evenodd}{odd}, and
$36 = 6^2$; while $360$ has $24$ (question 12), \omterm{def:g3:numbers:evenodd}{even}, and is
no square (question 13).

\textbf{15.} Locker $n$ is toggled once by each student $k$
whose number \omterm{def:g9:arith:divisor}{divides} $n$: in all, as many times as $n$ has
\omterm{def:g9:arith:divisor}{divisors}. A locker ends \emph{open} when it is toggled an \omterm{def:g3:numbers:evenodd}{odd
number} of times --- by question 14, exactly when $n$ is a
perfect square. Open lockers: $1, 4, 9, 16, 25, 36, 49, 64,
81, 100$ --- ten of them.
\end{solution}
