\begin{solution}{exo:g9:statproba:3}

\frac{2 \times 0 + 4 \times 1 + 3 \times 2 + 1 \times 3}{10}

\begin{solution}{exo:g9:statproba:4}
$\P(3) = \frac16$. \quad
$\P(\text{even}) = \frac36 = \frac12$. \quad
$\P(\text{at least } 3) = \P(\{3,4,5,6\}) = \frac46 = \frac23$. \quad
$\P(\text{not } 6) = 1 - \frac16 = \frac56$.
\end{solution}

\begin{solution}{exo:g9:statproba:5}
$\P(\text{white}) = \frac{5}{12}$.

$\P(\text{red or green}) = \frac{4 + 3}{12} = \frac{7}{12}$.

$\P(\text{not white}) = 1 - \frac{5}{12} = \frac{7}{12}$.

The last two are equal: ``not white'' \emph{is} the event ``red or
green''.
\end{solution}

\begin{solution}{exo:g9:statproba:6}
The $12$ outcomes (\omterm{def:g5:solids:def}{face} of the coin, value of the die) are equally
likely.

Heads and a $6$: one outcome out of $12$: $\frac{1}{12}$.

Heads or a $6$: the outcomes with heads ($6$ of them) plus the outcome
(tails, $6$): $7$ outcomes, so $\frac{7}{12}$. Counting each event and
subtracting the overlap gives the same:
$\frac{6}{12} + \frac{2}{12} - \frac{1}{12} = \frac{7}{12}$.
\end{solution}

\begin{solution}{exo:g9:statproba:7}
First draw: red $\frac25$, black $\frac35$. Second draw without
replacement: after a red, $1$ red and $3$ black remain among $4$; after
a black, $2$ red and $2$ black among $4$.

Two black: $\frac35 \times \frac24 = \frac{6}{20} = \frac{3}{10}$.

Different colors (RB or BR):
$\frac25 \times \frac34 + \frac35 \times \frac24
= \frac{6}{20} + \frac{6}{20} = \frac{12}{20} = \frac35$.
\end{solution}

\begin{solution}{exo:g9:statproba:9}
\emph{1.} Among the $36$ equally likely pairs, the doubles are
$(1,1), (2,2), \dots, (6,6)$: six of them, so
$\P(\text{double}) = \frac{6}{36} = \frac16$.

\emph{2.} Two independent games: tree with win ($\frac16$) / lose
($\frac56$) at each level. \omterm{def:g9:statproba:probability}{Probability} of never winning:
$\frac56 \times \frac56 = \frac{25}{36}$. So
\[
\P(\text{at least one win}) = 1 - \frac{25}{36} = \frac{11}{36}
\approx 0.31 .
\]
\end{solution}

\begin{solution}{pb:g9:statproba:1}
\textbf{1.} Each roll has $5$ non-six outcomes out of $6$; the
$6 \times 6$ table shows $5 \times 5 = 25$ six-free outcomes out
of $36$: $P(\text{no six}) = \frac{25}{36}$, so
$P(\text{at least one}) = 1 - \frac{25}{36} = \frac{11}{36}
\approx 0.31$.

\textbf{2.} $\frac13 = \frac{12}{36} \neq \frac{11}{36}$. The
addition counts the outcome ``six then six'' once in each
$\frac16$: once too often. Probabilities of events that can
happen \emph{together} do not simply add.

\textbf{3.} \omterm{def:g1:addition:def}{Sum} $7$: the six outcomes $(1,6), (2,5), (3,4),
(4,3), (5,2), (6,1)$: $P = \frac{6}{36} = \frac16$. \omterm{def:g1:addition:def}{Sum} $2$:
only $(1,1)$: $\frac{1}{36}$. Seven has the most ways to
happen --- the fat middle of the \omterm{def:g1:addition:def}{sums} table.

\textbf{4.} \omterm{def:g1:addition:def}{Sum} $9$: $(3,6), (4,5), (5,4), (6,3)$ --- four
outcomes, $\frac{4}{36}$. \omterm{def:g1:addition:def}{Sum} $10$: $(4,6), (5,5), (6,4)$ ---
three, $\frac{3}{36}$. Nine is more likely: the count decides,
not the number of ``ways to write'' the \omterm{def:g1:addition:def}{sum} with unordered
\omterm{def:g5:solids:def}{faces}.

\textbf{5.} Each level of the tree multiplies the no-six
\omterm{def:g9:statproba:probability}{probability} by $\frac56$, whatever happened before: after $n$
rolls, $P(\text{no six at all}) = \left(\frac56\right)^n$, and
the complement rule gives the formula.

\textbf{6.} $\left(\frac56\right)^4 = \frac{625}{1296} \approx
0.482$, so the Chevalier won with \omterm{def:g9:statproba:probability}{probability}
$1 - 0.482 = 0.518$ --- a quiet, steady edge of almost
$2\,\%$ per game: an excellent bet, and it paid for his
carriages.

\textbf{7.} At $7$ rolls his rule gives $\frac76$ --- a
\omterm{def:g9:statproba:probability}{probability} greater than $1$, nonsense. It is question 2's
double-counting, compounded: the rolls' successes overlap, and
adding their chances counts the overlaps again and again.

\textbf{8.} $1 - \left(\frac{35}{36}\right)^{24} \approx
1 - 0.5086 = 0.491$: below one \omterm{def:g3:division:half}{half}. Betting \omterm{def:g3:numbers:evenodd}{even} money on a
$49.1\,\%$ event, the Chevalier lost about $2$ games in every
$100$ on average --- slowly, mysteriously (to him), and
inevitably.

\textbf{9.} $1 - \left(\frac{35}{36}\right)^{25} \approx
1 - 0.494 = 0.506$: with $25$ rolls the bet turns favorable.
One roll separated the Chevalier from profitability.

\textbf{10.} Multiplying a \omterm{def:g9:statproba:probability}{probability} by the number of tries
counts overlapping successes several times, and eventually
produces impossible answers above $1$. The correct route is
always through the complement: chain the failure probabilities
by multiplication, then subtract from $1$.

\textbf{11.} \omterm{def:g9:statproba:indicators}{Mean}: $\frac{9 \times 2\,000 + 20\,000}{10} =
\frac{38\,000}{10} = 3\,800$ euros. \omterm{def:g9:statproba:indicators}{Median}: the middle salaries
are both $2\,000$: \omterm{def:g9:statproba:indicators}{median} $2\,000$ euros. The advertisement
should quote the \omterm{def:g9:statproba:indicators}{median} --- nine of the ten workers never see
anything like $3\,800$; the \omterm{def:g9:statproba:indicators}{mean} is dragged up by one
outlier (\cref{ex:g9:statproba:weighted} warned of weights).

\textbf{12.} For instance $2, 8, 15, 17, 18$: \omterm{def:g9:statproba:indicators}{median} $15$
(middle value), \omterm{def:g9:statproba:indicators}{mean} $\frac{60}{5} = 12$. A \omterm{def:g9:statproba:indicators}{mean} below the
\omterm{def:g9:statproba:indicators}{median} betrays a tail of \emph{low} marks pulling the average
down while the top \omterm{def:g3:division:half}{half} sits high.

\textbf{13.} First three factors:
$1 \times \frac{364}{365} \times \frac{363}{365} \approx
0.992$. Each new student must dodge one more taken date, so the
factors shrink: $\frac{343}{365} \approx 0.94$ for the
$23$rd. The full \omterm{def:g2:mult:def}{product} falls to about $0.493$ --- so
\[
P(\text{at least one shared birthday}) \approx 0.507 :
\]
better than \omterm{def:g3:numbers:evenodd}{even}, in a class of just $23$. Intuition says
``$23$ people, $365$ days, no chance''; the mathematics says
``flip a coin''.

\textbf{14.} $\frac{23 \times 22}{2} = 253$ pairs. Each pair is
a fresh opportunity for a coincidence, and $253$ opportunities
at $\frac{1}{365}$ each is no longer a small affair: the
surprise dissolves once one counts pairs, not people.

\textbf{15.} Example protocol: roll a die in blocks of four,
recording whether each block shows a six, for $100$ blocks; or
collect the birthday lists of many classes of about $23$ and
record the proportion with a coincidence. The observed
\omterm{def:g7:stats:frequency}{frequency} will wobble, but as the number of repetitions grows
it should settle ever closer to the computed probabilities
($0.518$; $0.507$) --- frequencies converge to probabilities:
the law of large numbers, stated here as an expectation and
proved in the High School volume.
\end{solution}
