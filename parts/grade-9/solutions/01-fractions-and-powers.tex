\begin{solution}{exo:g9:fractions:1}
$\dfrac13 + \dfrac15 = \dfrac{5}{15} + \dfrac{3}{15} = \dfrac{8}{15}$.

$\dfrac72 - \dfrac53 = \dfrac{21}{6} - \dfrac{10}{6} = \dfrac{11}{6}$.

$\dfrac{5}{12} + \dfrac38 = \dfrac{10}{24} + \dfrac{9}{24} =
\dfrac{19}{24}$.

$3 - \dfrac45 = \dfrac{15}{5} - \dfrac45 = \dfrac{11}{5}$.
\end{solution}

\begin{solution}{exo:g9:fractions:2}
$\dfrac59 \times \dfrac{27}{35} = \dfrac{5 \times 27}{9 \times 35}
= \dfrac{27}{9} \times \dfrac{5}{35} = 3 \times \dfrac17 = \dfrac37$
(simplify by $9$ and by $5$).

$\dfrac{8}{15} \div \dfrac45 = \dfrac{8}{15} \times \dfrac54
= \dfrac{8 \times 5}{15 \times 4} = \dfrac{40}{60} = \dfrac23$.

$\dfrac23 \times \dfrac34 \times \dfrac45 = \dfrac{2 \times 3 \times 4}
{3 \times 4 \times 5} = \dfrac25$ (the $3$s and $4$s cancel).
\end{solution}

\begin{solution}{exo:g9:fractions:3}
$2^5 = 32$; \quad $(-2)^4 = 16$; \quad $-2^4 = -16$; \quad
$4^{-2} = \frac{1}{16}$; \quad
$\left(\frac23\right)^3 = \frac{8}{27}$; \quad $7^0 = 1$.
\end{solution}

\begin{solution}{exo:g9:fractions:4}
$5^3 \times 5^6 = 5^9$; \quad $\dfrac{7^9}{7^5} = 7^4$; \quad
$\left(2^3\right)^4 = 2^{12}$; \quad
$\dfrac{3^2 \times 3^7}{3^5} = 3^{9-5} = 3^4$; \quad
$2^5 \times 4 = 2^5 \times 2^2 = 2^7$.
\end{solution}

\begin{solution}{exo:g9:fractions:6}

\begin{solution}{exo:g9:fractions:7}

\begin{solution}{exo:g9:fractions:8}

$\frac13 \times \frac34 = \frac14$, leaving

\begin{solution}{exo:g9:fractions:9}
Time $= \dfrac{\text{distance}}{\text{speed}} =
\dfrac{1.5 \times 10^{11}}{3 \times 10^8} = 0.5 \times 10^3 = 500$
seconds, i.e.\ $8$ minutes and $20$ seconds.
\end{solution}

\begin{solution}{exo:g9:fractions:10}
Write both numbers as \omterm{def:g9:fractions:power}{powers} with \omterm{def:g8:powers:def}{exponent} $10$:
\[
2^{30} = \left(2^3\right)^{10} = 8^{10},
\qquad
3^{20} = \left(3^2\right)^{10} = 9^{10}.
\]
Since $8 < 9$, multiplying ten copies of each: $8^{10} < 9^{10}$, so
$2^{30} < 3^{20}$.
\end{solution}

\begin{solution}{pb:g9:fractions:1}
\textbf{1.} $\frac14 = 0.25$ (stops); $\frac13 = 0.333\ldots$
(block $3$); $\frac56 = 0.8333\ldots$ (block $3$ after the
$8$); $\frac{3}{11} = 0.272727\ldots$ (block $27$).

\textbf{2.} At each step of the division by $b$, the \omterm{def:g3:division:remainder}{remainder}
is one of the $b$ values $0, 1, \dots, b - 1$
(\cref{thm:g6:wholes:division}). If a \omterm{def:g3:division:remainder}{remainder} $0$ appears, the
division stops. Otherwise, after at most $b$ steps some
\omterm{def:g3:division:remainder}{remainder} must appear a second time --- there are more steps
than possible \omterm{def:g3:division:remainder}{remainders} --- and from that moment the division
repeats exactly the sequence of digits it produced after the
first appearance: the decimal writing cycles forever. Stop or
repeat: no third behavior exists.

\textbf{3.} $\frac17 = 0.142857\,142857\ldots$: the block
$142857$ has length $6$. Question 2 promised a repetition within
at most $7$ steps --- here the \omterm{def:g3:division:remainder}{remainders}
$1, 3, 2, 6, 4, 5$ all appear before the $1$ returns: the
longest possible chant for a division by $7$.

\textbf{4.} $\frac ab$ stops exactly when some
$\frac{a \times k}{b \times k}$ has \omterm{def:g4:fractions:def}{denominator} $10$, $100$,
$1000, \dots$ --- that is, when the \omterm{def:g4:fractions:def}{denominator} can be
multiplied up to a \omterm{def:g9:fractions:power}{power} of ten, as $4 \times 25 = 100$ or
$8 \times 125 = 1\,000$. For $3$ and $6$ this is hopeless: a
\omterm{def:g9:fractions:power}{power} of ten has digit \omterm{def:g1:addition:def}{sum} $1$, and every \omterm{def:g5:division:multiple}{multiple} of $3$ has a
digit \omterm{def:g1:addition:def}{sum} that is a \omterm{def:g5:division:multiple}{multiple} of $3$ --- so no \omterm{def:g5:division:multiple}{multiple} of $3$
(hence none of $6$) is ever $10$, $100$, $1\,000, \dots$ For
$11$: dividing $10$, $100$, $1\,000$, $10\,000$ by $11$ always
leaves a \omterm{def:g3:division:remainder}{remainder} ($10$, $1$, $10$, $1, \dots$), never $0$.
\omterm{def:g4:fractions:def}{Denominators} that reach a \omterm{def:g9:fractions:power}{power} of ten stop; all others repeat.

\textbf{5.} $\frac{9}{40}$: $40 \times 25 = 1\,000$, stops:
$\frac{225}{1000} = 0.225$. $\frac{33}{50}$:
$50 \times 2 = 100$, stops: $\frac{66}{100} = 0.66$.
$\frac{5}{12}$: $12$ is a \omterm{def:g5:division:multiple}{multiple} of $3$, repeats
($0.41666\ldots$).

\textbf{6.} $10x = 7.777\ldots$, so $10x - x = 7$ (the tails
cancel perfectly), $9x = 7$ and $x = \frac79$. Division check:
$7 \div 9 = 0.777\ldots$

\textbf{7.} The block has two digits, so shift by $100$:
$100x - x = 72.7272\ldots - 0.7272\ldots = 72$, hence
$99x = 72$ and $x = \frac{72}{99} = \frac{8}{11}$.

\textbf{8.} $1000x = 583.333\ldots$ and $100x = 58.333\ldots$;
subtracting, $900x = 525$, so
$x = \frac{525}{900} = \frac{7}{12}$ (divide by $75$). Indeed
$\frac{7}{12} = 0.58333\ldots$

\textbf{9.} $10x - x = 9.999\ldots - 0.999\ldots = 9$, so
$9x = 9$ and $x = 1$. Not ``just below'' $1$:
$0.999\ldots$ \emph{is} the number $1$, written in a second
costume. Any number strictly below $1$ leaves a gap, and
\cref{pb:g6:decimals:1} showed a gap always contains further
numbers --- while nothing fits between $0.999\ldots$ and $1$.
The endless nines are the chocolate \omterm{def:g3:division:remainder}{remainder} of
\cref{pb:g6:fractions:1} shrunk below every positive amount:
nothing is left.

\textbf{10.} $2.4545\ldots = 2 + \frac{45}{99} = 2 +
\frac{5}{11} = \frac{27}{11}$. And
$0.123123\ldots = \frac{123}{999} = \frac{41}{333}$ (divide by
$3$).

\textbf{11.} Dictionary: $0.037037\ldots = \frac{37}{999}$.
Since $999 = 27 \times 37$, this simplifies to $\frac{1}{27}$:
the \omterm{def:g9:fractions:fraction}{fraction} $\frac{1}{27}$ chants ``$037$''. Division check:
$1 \div 27 = 0.037037\ldots$

\textbf{12.} $0.0272727\ldots = \frac{1}{10} \times
0.272727\ldots = \frac{1}{10} \times \frac{27}{99}
= \frac{27}{990} = \frac{3}{110}$ (divide by $9$).

\textbf{13.} The blocks of \omterm{def:g1:counting:zero}{zeros} grow without end, so no fixed
block can repeat forever: it is \emph{not} a repeating decimal
(and it never stops). By Part I, every \omterm{def:g9:fractions:fraction}{fraction} stops or
repeats --- so this number is no \omterm{def:g9:fractions:fraction}{fraction} at all: it is
irrational, built to order. The superstar of irrationality,
$\sqrt2$, is unmasked in \cref{ch:g9:sqrt}.

\textbf{14.} $2^{30} = \left(2^{10}\right)^3 \approx
(1.024)^3 \times 10^9 \approx 1.07 \times 10^9$: a number
slightly above $10^9$, hence with $10$ digits. (Exactly:
$2^{30} = 1\,073\,741\,824$.)

\textbf{15.} $0.20262026\ldots = \frac{2026}{9999}$, already in
lowest terms ($9999 = 3 \times 3 \times 11 \times 101$ shares
no factor with $2026 = 2 \times 1013$). The completed
dictionary: \emph{the \omterm{def:g9:fractions:fraction}{fractions} are exactly the decimals that
stop or repeat} --- \omterm{def:g4:fractions:def}{denominators} of tens' friends stop, all
others chant a block; and conversely every chant, \omterm{def:g3:numbers:evenodd}{even} your
graduation year, is a \omterm{def:g9:fractions:fraction}{fraction} in disguise.
\end{solution}
