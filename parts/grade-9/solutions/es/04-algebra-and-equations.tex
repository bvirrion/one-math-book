\begin{solution}{exo:g9:algebra:1}
$4(3x - 2) = 12x - 8$.

$(x+6)(x+2) = x^2 + 2x + 6x + 12 = x^2 + 8x + 12$.

$(2x-3)(x+5) = 2x^2 + 10x - 3x - 15 = 2x^2 + 7x - 15$.

$(x-1)(x-9) = x^2 - 9x - x + 9 = x^2 - 10x + 9$.
\end{solution}

\begin{solution}{exo:g9:algebra:2}

$(3x-4)^2 = 9x^2 - 24x + 16$.

$(x+10)(x-10) = x^2 - 100$.

$(5x+2)(5x-2) = 25x^2 - 4$.
\end{solution}

\begin{solution}{exo:g9:algebra:3}

$53 \times 47 = (50+3)(50-3) = 2500 - 9 = 2491$.

\begin{solution}{exo:g9:algebra:5}
$4x + 7 = 19$: $4x = 12$, so $x = 3$.

$6x - 5 = 2x + 11$: $4x = 16$, so $x = 4$.

$5(x-3) = 3x + 1$: $5x - 15 = 3x + 1$, entonces $2x = 16$ y $x = 8$.
\end{solution}

\begin{solution}{exo:g9:algebra:6}
$(x+4)(3x-12) = 0$: $x = -4$ or $x = 4$.

$x^2 - 121 = (x-11)(x+11) = 0$: $x = 11$ or $x = -11$.

$(2x-1)^2 = 0$: la solución única $x = \frac12$.

$x^2 = 8x$: $x(x - 8) = 0$, so $x = 0$ or $x = 8$.
\end{solution}

\begin{solution}{exo:g9:algebra:7}
\omterm{ex:g1:subtraction:difference}{Diferencia} de cuadrados con $a = 3x + 2$ y $b = 5$:
\[
E = (3x + 2)^2 - 5^2 = (3x + 2 + 5)(3x + 2 - 5) = (3x + 7)(3x - 3).
\]
Regla \omterm{def:g2:mult:def}{Producto cero}: $3x + 7 = 0$ da $x = -\frac73$ y $3x - 3 = 0$
da $x = 1$.
\end{solution}

\begin{solution}{exo:g9:algebra:8}
$3x - 4 \geq 8$: $3x \geq 12$, entonces $x \geq 4$ (punto relleno en $4$, sombreado
derecha).

$5 - 2x > 1$: $-2x > -4$, al dividir por $-2$ se invierte el signo: $x < 2$
(punto hueco en $2$, sombra a la izquierda).

$4(x+1) \leq x - 5$: $4x + 4 \leq x - 5$, entonces $3x \leq -9$, entonces
$x \leq -3$ (punto relleno en $-3$, sombreado a la izquierda).
\end{solution}

\begin{solution}{exo:g9:algebra:9}
Sea $x$ el número: $3x + 7 = 5x - 9$. Luego $16 = 2x$, entonces $x = 8$.
Verifique: $3 \times 8 + 7 = 31$ y $5 \times 8 - 9 = 31$.
\end{solution}

\begin{solution}{exo:g9:algebra:10}
\emph{1.} $(n+1)(n-1) = n^2 - 1$ (tercera identidad). Los vecinos de $n$
son $n - 1$ y $n + 1$, por lo que su \omterm{def:g2:mult:def}{producto} es $n^2 - 1$: por ejemplo\
$24 \times 26 = 625 - 1 = 624$.

\emph{2.} Se pueden escribir tres números enteros consecutivos $n - 1$, $n$,
$n + 1$. Su \omterm{def:g1:addition:def}{suma} es
\[
(n-1) + n + (n+1) = 3n ,
\]
a \omterm{def:g9:arith:divisor}{múltiple} de $3$ (tres veces el del medio).
\end{solution}

\begin{solution}{pb:g9:algebra:1}
\textbf{1.} $(10a + b) - (10b + a) = 9a - 9b = 9(a - b)$: el
\omterm{ex:g1:subtraction:difference}{diferencia} entre un número de dos cifras y su inversión siempre es
un \omterm{def:g9:arith:divisor}{múltiple} de $9$. Verificar: $72 - 27 = 45 = 9 \times 5$, y
$a - b = 7 - 2 = 5$.

\textbf{2.} \omterm{def:g9:algebra:expand}{En expansión} el lado derecho:
$9(11a + b) + a + b + c = 99a + 9b + a + b + c
= 100a + 10b + c$. Entonces el número es igual a \omterm{def:g9:arith:divisor}{múltiple} de $9$ más
su dígito \omterm{def:g1:addition:def}{suma}: dividiendo por $9$, ambos quedan igual \omterm{def:g3:division:remainder}{resto}.
En particular, \omterm{def:g3:division:remainder}{resto} $0$ para uno significa \omterm{def:g3:division:remainder}{resto} $0$ para el
otro: la regla del nueve, demostrada.

\textbf{3.} $9(11a + b)$ también es \omterm{def:g9:arith:divisor}{múltiple} de $3$, por lo que lo mismo
La identidad muestra el número y dígito \omterm{def:g1:addition:def}{suma} y deja el mismo \omterm{def:g3:division:remainder}{resto} en
la división por $3$: la regla de tres.

\textbf{4.} Escriba $47 = 9k + 2$ y $86 = 9l + 5$. entonces
\[
47 \times 86 = (9k + 2)(9l + 5)
= 9\,(9kl + 5k + 2l) + 10 :
\]
el \omterm{def:g2:mult:def}{producto} deja el mismo \omterm{def:g3:division:remainder}{resto} que $2 \times 5 = 10$,
a saber, $1$. Por lo tanto, un resultado correcto también debe salir de \omterm{def:g3:division:remainder}{resto}.
$1$ --- lo que hace $4\,042$. Pero la prueba sólo ve \omterm{def:g3:division:remainder}{restos}:
$4\,042$ y $4\,402$ (dígitos intercambiados) pasan de manera idéntica, por lo que
Pasar no prueba nada; fallar es lo que condena.

\textbf{5.} \omterm{def:g9:algebra:expand}{En expansión}: $11(91a + 9b + c) - a + b - c + d
= 1001a + 99b + 11c - a + b - c + d
= 1000a + 100b + 10c + d$. Entonces el número es \omterm{def:g6:wholes:divisible}{divisible} por $11$
exactamente cuando está alternando \omterm{def:g1:addition:def}{suma} $-a + b - c + d$. Para
$2\,926$: $-2 + 9 - 2 + 6 = 11$, \omterm{def:g6:wholes:divisible}{divisible} por $11$: sí,
$2\,926 = 11 \times 266$.

\textbf{6.} $742 - 247 = 495$; $495 + 594 = 1\,089$. Cualquiera válida
El ejemplo también llega a $1\,089$.

\textbf{7.} $(100a + 10b + c) - (100c + 10b + a) = 99(a - c)$.
Para $a - c = 2, 3, \dots, 9$:
\[
198,\ 297,\ 396,\ 495,\ 594,\ 693,\ 792,\ 891 .
\]

\textbf{8.} $100(m - 1) + 90 + (10 - m) = 100m - 100 + 90 + 10
- m = 99m$. Entonces los tres dígitos de $99m$ son $m - 1$, luego $9$,
luego $10 - m$ (verifique $495$: $m = 5$ da $4$, $9$, $5$).

\textbf{9.} La reversión de $99m$ tiene los dígitos $10 - m$, $9$,
$m - 1$, por lo que es igual a $100(10 - m) + 90 + (m - 1)$. Añadiendo:
\[
100\,(m - 1 + 10 - m) + 180 + (10 - m + m - 1)
= 100 \times 9 + 180 + 9 = 1\,089 ,
\]
independiente de $m$. Cualquiera que sea el número inicial, el mago
es seguro.

\textbf{10.} Si $a = c$, la resta da $0$ y el truco
muere de una vez. Si $a - c = 1$, da $99$, que debe ser
tratado como la cadena de tres dígitos $099$ (inversión $990$,
$99 + 990 = 1\,089$: ¡salvado!). La letra pequeña: \omterm{ex:g1:subtraction:difference}{diferencias} de
$1$ funciona solo si el espectador escribe el resultado de la resta
con tres dígitos, incluyendo \omterm{def:g1:counting:zero}{cero} inicial --- requiriendo un espacio de
al menos $2$ evita la discusión.

\textbf{11.} $2(5m + 7) + d - 14 = 10m + 14 + d - 14 =
10m + d$: se muestran las decenas (y las centenas, de octubre a diciembre)
el mes, las unidades (y decenas) el día.

\textbf{12.} $2(5m + 9) + d = 10m + 18 + d$: resta $18$.

\textbf{13.} Por pregunta 2, $n - (\text{digit sum of } n) =
9(11a + b)$ para un $n$ de tres dígitos: siempre un \omterm{def:g9:arith:divisor}{múltiple} de $9$
(Las identidades similares funcionan para cualquier duración). Un \omterm{def:g9:arith:divisor}{múltiple} de $9$ tiene
dígito \omterm{def:g1:addition:def}{suma} a \omterm{def:g9:arith:divisor}{múltiple} de $9$ (pregunta 2 nuevamente), por lo que los dígitos
el espectador lee en voz alta debe completarse hasta el siguiente \omterm{def:g9:arith:divisor}{múltiple}
de $9$: el dígito que falta es ese complemento. Ambigüedad: si el
Los dígitos leídos ya suman \omterm{def:g9:arith:divisor}{múltiple} de $9$, el dígito cruzado
podría ser $0$ o $9$ --- excluyendo $0$ elimina la duda.

\textbf{14.} $(10t + 5)^2 = 100t^2 + 100t + 25 =
100\,t(t + 1) + 25$: escribe $t(t+1)$, luego pega $25$ detrás.
$45^2$: $4 \times 5 = 20$, entonces $2\,025$. $85^2$:
$8 \times 9 = 72$, entonces $7\,225$. $105^2$: $10 \times 11 = 110$,
entonces $11\,025$.

\textbf{15.} Cinco consecutivos: $n + (n+1) + (n+2) + (n+3) +
(n+4) = 5n + 10 = 5(n + 2)$: un \omterm{def:g9:arith:divisor}{múltiple} de $5$, siempre. cuatro
consecutivos: $n + (n+1) + (n+2) + (n+3) = 4n + 6 =
4(n + 1) + 2$: \omterm{def:g3:division:remainder}{resto} $2$ en la división por $4$, nunca un
\omterm{def:g9:arith:divisor}{múltiple}. El álgebra convierte "parece funcionar" en
\emph{siempre} y ``No encontré ningún ejemplo'' en \emph{nunca} ---
dos palabras que ninguna cantidad de pruebas puede alcanzar
(\cref{met:g7:literal:test}).
\end{solution}
