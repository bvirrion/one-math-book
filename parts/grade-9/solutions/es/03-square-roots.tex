\begin{solution}{exo:g9:sqrt:3}
$\sqrt2 \times \sqrt{18} = \sqrt{36} = 6$.

$\dfrac{\sqrt{75}}{\sqrt3} = \sqrt{\dfrac{75}{3}} = \sqrt{25} = 5$.

$\sqrt5 \times \sqrt{20} = \sqrt{100} = 10$.

$\dfrac{\sqrt8}{\sqrt{50}} = \sqrt{\dfrac{8}{50}} = \sqrt{\dfrac{4}{25}}
= \dfrac25$.
\end{solution}

\begin{solution}{exo:g9:sqrt:4}
$x^2 = 36$: $x = 6$ or $x = -6$.

$x^2 = 11$: $x = \sqrt{11}$ or $x = -\sqrt{11}$.

$x^2 + 4 = 0$ significa $x^2 = -4 < 0$: sin solución.

$2x^2 = 50$: $x^2 = 25$, so $x = 5$ or $x = -5$.
\end{solution}

\begin{solution}{exo:g9:sqrt:5}
$\sqrt{20} + \sqrt{45} = 2\sqrt5 + 3\sqrt5 = 5\sqrt5$.

$3\sqrt8 - \sqrt{18} + \sqrt2 = 3 \times 2\sqrt2 - 3\sqrt2 + \sqrt2
= (6 - 3 + 1)\sqrt2 = 4\sqrt2$.
\end{solution}

\begin{solution}{exo:g9:sqrt:6}
$\left(\sqrt3 + 1\right)^2 = 3 + 2\sqrt3 + 1 = 4 + 2\sqrt3$.

$\left(2\sqrt5 - 3\right)^2 = 4 \times 5 - 2 \times 3 \times 2\sqrt5 + 9
= 29 - 12\sqrt5$.

$\left(\sqrt6 + \sqrt2\right)\left(\sqrt6 - \sqrt2\right) = 6 - 2 = 4$.
\end{solution}

\begin{solution}{exo:g9:sqrt:10}
$\left(2 + \sqrt3\right)^2 = 4 + 4\sqrt3 + 3 = 7 + 4\sqrt3$. entonces
$x = \sqrt{7 + 4\sqrt3} = \sqrt{\left(2+\sqrt3\right)^2} = 2 + \sqrt3$
(un número no negativo, por lo que la raíz simplemente elimina el cuadrado).

Del mismo modo $\left(\sqrt5 - 2\right)^2 = 5 - 4\sqrt5 + 4 = 9 - 4\sqrt5$,
y $\sqrt5 - 2 > 0$, entonces
$\sqrt{9 - 4\sqrt5} = \sqrt5 - 2$ (no $2 - \sqrt5$, que es
negativo!).
\end{solution}

\begin{solution}{pb:g9:sqrt:1}
\textbf{1.}$1.4^2 = 1.96 < 2 < 2.25 = 1.5^2$;
$1.41^2 = 1.9881 < 2 < 2.0164 = 1.42^2$. Tres decimales:
$1.414^2 = 1.999396 < 2 < 2.002225 = 1.415^2$, entonces
$1.414 < \sqrt2 < 1.415$.

\textbf{2.} Cuadratura $\sqrt2 = \frac ab$ da
$2 = \frac{a^2}{b^2}$, por lo tanto $a^2 = 2b^2$: el número $a^2$ es
dos veces un número entero --- \emph{\omterm{def:g3:numbers:evenodd}{incluso}}.

\textbf{3.} Si $a$ fuera \omterm{def:g3:numbers:evenodd}{extraño}, $a^2$ sería \omterm{def:g3:numbers:evenodd}{extraño}
(\cref{pb:g8:pythagoras:1}); ya que $a^2$ es \omterm{def:g3:numbers:evenodd}{incluso}, $a$ es \omterm{def:g3:numbers:evenodd}{incluso}:
$a = 2k$. Sustituyendo: $4k^2 = 2b^2$, entonces $b^2 = 2k^2$ es \omterm{def:g3:numbers:evenodd}{incluso},
y por el mismo hecho de paridad $b$ es \omterm{def:g3:numbers:evenodd}{incluso}.

\textbf{4.} Tanto $a$ como $b$ \omterm{def:g3:numbers:evenodd}{incluso} significa tanto \omterm{def:g6:wholes:divisible}{divisible} como $2$
--- pero $\frac ab$ estaba en los términos más bajos, sin compartir nada común
\omterm{def:g9:arith:divisor}{divisor}. Contradicción: el supuesto \omterm{def:g9:fractions:fraction}{fracción} no puede existir.
\emph{Theorem: $\sqrt2$ is irrational.}

\textbf{5.} La contradicción vive enteramente en ``más bajo
términos'': sin él, ``$a$ y $b$, ambos \omterm{def:g3:numbers:evenodd}{incluso}'' contradicen
nada. Asumir las condiciones más bajas es legítimo porque cada
\omterm{def:g9:fractions:fraction}{fracción} \emph{tiene} un formulario con los términos más bajos
(\cref{met:g9:arith:simplify}: dividir el MCD) --- si
$\sqrt2$ si hubiera algún \omterm{def:g9:fractions:fraction}{fracción}, serían los términos más bajos
uno, y ese es imposible.

\textbf{6.} En factorizaciones primas, los cuadrados llevan \omterm{def:g3:numbers:evenodd}{incluso}
\omterm{def:g8:powers:def}{exponentes} (\cref{pb:g9:arith:1}). En $a^2 = 3b^2$, el \omterm{def:g8:powers:def}{exponente}
de $3$ es \omterm{def:g3:numbers:evenodd}{incluso} a la izquierda, pero \omterm{def:g3:numbers:evenodd}{extraño} a la derecha (\omterm{def:g3:numbers:evenodd}{incluso} de
$b^2$, más uno). Ningún número tiene dos factorizaciones.
(\cref{thm:g9:arith:factorization}): contradicción, y
$\sqrt3$ es irracional.

\textbf{7.} En $a^2 = n b^2$, el argumento encuentra un primo con
paridad \omterm{def:g8:powers:def}{exponente} contradictoria exactamente cuando aparece algún primo en
$n$ con un \emph{\omterm{def:g3:numbers:evenodd}{extraño}} \omterm{def:g8:powers:def}{exponente}. Si cada \omterm{def:g8:powers:def}{exponente} de $n$ es
\omterm{def:g3:numbers:evenodd}{incluso}, $n$ es un cuadrado perfecto ($n = m^2$, y
$\sqrt n = m$ está completo). Resultado completo: $\sqrt n$ es
irracional para todo $n$ que no sea un cuadrado perfecto ---
$\sqrt2, \sqrt3, \sqrt5, \sqrt6, \sqrt7, \sqrt8, \sqrt{10},
\dots$

\textbf{8.} De $r x = q$ con $r \neq 0$:
$x = \frac qr$, un \omterm{def:g3:division:remainder}{cociente} de racionales, por tanto racional. Entonces si
$x$ es irracional, $r x$ no puede ser racional: $2\sqrt2$ y
$\frac{\sqrt2}{2}$ son irracionales (multiplicadores $2$ y
$\frac12$).

\textbf{9.} Si $1 + \sqrt2 = q$ fuera racional, entonces
$\sqrt2 = q - 1$ sería racional: contradicción. si
$s = \sqrt2 + \sqrt3$ eran racionales, entonces
$s^2 = 2 + 2\sqrt6 + 3 = 5 + 2\sqrt6$ también lo sería, dando
$\sqrt6 = \frac{s^2 - 5}{2}$ racional --- pero $\sqrt6$ es
irracional (pregunta 7). Entonces $\sqrt2 + \sqrt3$ es irracional.

\textbf{10.} \omterm{def:g1:addition:def}{sumas}: $\sqrt2$ y $-\sqrt2$ (o $\sqrt2$ y
$1 - \sqrt2$) son irracionales, siendo racionales \omterm{def:g1:addition:def}{sumas} $0$ y
$1$. \omterm{def:g2:mult:def}{Productos}: $\sqrt2 \times \sqrt2 = 2$. La irracionalidad puede
evaporarse en aritmética --- de ahí las pruebas caso por caso.

\textbf{11.} De $x = 1$: promedio de $1$ y $\frac21 = 2$:
$x_1 = \frac32$. Luego promedio de $\frac32$ y
$\frac{2}{3/2} = \frac43$:
$x_2 = \frac12\left(\frac32 + \frac43\right)
= \frac12 \cdot \frac{17}{6} = \frac{17}{12} \approx 1.4167$.

\textbf{12.}$x_3 = \frac12\left(\frac{17}{12} +
\frac{24}{17}\right) = \frac12 \cdot
\frac{289 + 288}{204} = \frac{577}{408} = 1.4142156\ldots$
Contra $\sqrt2 = 1.4142135\ldots$: cinco decimales correctos después
tres pasos --- la receta duplica aproximadamente los decimales correctos
cada \omterm{def:g4:numbers:round}{redondo}.

\textbf{13.} Si $x^2 > 2$ (adivina demasiado grande), entonces $x > \sqrt2$,
y $\frac2x < \frac{2}{\sqrt2} = \sqrt2$: el lado complementario es
demasiado pequeño. Si $x^2 < 2$, las desigualdades se invierten. De cualquier manera
$\sqrt2$ se encuentra entre $x$ y $\frac2x$, por lo que su promedio ---
la siguiente suposición --- está más cerca que el peor de los dos lados:
el rectángulo de \omterm{def:g6:measure:area}{área} $2$ se vuelve más cuadrado a cada paso.

\textbf{14.}$9 - 8 = 1$; $289 - 288 = 1$; \
$332\,929 - 332\,928 = 1$. Cada garza adivina $\frac ab$
satisface $a^2 - 2b^2 = 1$: La parte I demostró que golpear $0$ es
imposible, y estos \omterm{def:g9:fractions:fraction}{fracciones} pierden lo imposible exactamente
una unidad: los números enteros más cercanos que jamás se puedan obtener.

\textbf{15.} La recta numérica debe contener números más allá del
\omterm{def:g9:fractions:fraction}{fracciones} --- longitudes como $\sqrt2$, cuyas escrituras decimales se ejecutan
para siempre sin repetir: el \emph{números reales}. Su
La construcción honesta es una historia para el volumen de High School y,
en pleno rigor, los universitarios.
\end{solution}
