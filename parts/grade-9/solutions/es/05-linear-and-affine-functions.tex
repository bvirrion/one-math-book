\begin{solution}{exo:g9:linfunc:5}

\begin{solution}{exo:g9:linfunc:10}
\emph{1.} Los dos cambios multiplican el precio por
\[
\left(1 + \frac{t}{100}\right)\left(1 - \frac{t}{100}\right)
= 1 - \frac{t^2}{10000}
\]
(tercera identidad con $a = 1$, $b = \frac{t}{100}$).

\emph{2.} Para $0 < t \leq 100$, $\frac{t^2}{10000} > 0$, entonces el factor
es inferior a $1$: el precio final es inferior al original. Para
$t = 10$: factor $1 - \frac{100}{10000} = 0.99$, es decir\ un total
disminución de $1\%$.
\end{solution}

\begin{solution}{pb:g9:linfunc:1}
\textbf{1.} \omterm{prop:g9:linfunc:affinegraph}{Pendiente}: $\frac{f(100) - f(0)}{100 - 0} =

\[
f(C) = 1.8\,C + 32 .
\]

\textbf{2.} $f(20) = 68\,^\circ$F (el famoso día templado);
$f(37) = 98.6\,^\circ$F; $f(-10) = 14\,^\circ$F.

\textbf{3.} De $F = 1.8C + 32$:
$C = \frac{F - 32}{1.8}$. Entonces $50\,^\circ$F
$\to \frac{18}{1.8} = 10\,^\circ$C y $98.6\,^\circ$F
$\to \frac{66.6}{1.8} = 37\,^\circ$C.

\textbf{4.} $f(0) = 32 \neq 0$ y $f(20) = 68$ no es el
doble de $f(10) = 50$: no lineal. La gráfica es una recta.
A la línea le falta el origen: $f$ es \emph{afín}, no lineal.

\textbf{5.} $1.8x + 32 = x$ da $0.8x = -32$, por lo que $x = -40$:
a cuarenta bajo cero, ambos termómetros marcaban $-40$. En el gráfico, el
La línea de $f$ cruza la línea diagonal $y = x$ exactamente allí.

\textbf{6.} $p_A(k) = 1.5k + 3$ y $p_B(k) = 2k$. Para $4$ kilómetros:
$9$ euros frente a $8$: B más barato. Para $8$ km: $15$ contra

\textbf{7.} $1.5k + 3 = 2k$ da $k = 6$: a los seis kilómetros
ambos cuestan $12$ euros. Gráficamente, la línea de B (que pasa por el
origen, más empinado) comienza debajo de A y lo cruza en

$600 - 15t = 240$, $t = 24$ meses. Valor $0$: $t = 40$ meses.

\textbf{10.} Componiendo: $\times 1.25$ luego $\times 0.8$
multiplica por $1.25 \times 0.8 = 1$: el precio vuelve exactamente
a su inicio. Ningún milagro: los porcentajes actúan sobre diferentes

\textbf{12.} El segundo partió $10$ km adelante (intercepción); el

\textbf{13.} $f(x) + g(x) = (2 - 0.5)x + (3 + 7) = 1.5x + 10$:
afín, con \omterm{prop:g9:linfunc:affinegraph}{pendiente} el \emph{\omterm{def:g1:addition:def}{suma} of the \omterm{prop:g9:linfunc:affinegraph}{pendientes}} e interceptar
el \omterm{def:g1:addition:def}{suma} de las intersecciones --- claro al recopilar términos similares
en $(mx + p) + (m'x + p')$.

\textbf{14.} Si $m \neq 0$: exactamente una solución,
$x = -\frac pm$ --- una línea no horizontal cruza el eje
una vez. Si $m = 0$, $p \neq 0$: no hay solución --- una línea horizontal
por encima o por debajo del eje nunca lo toca. Si $m = 0$, $p = 0$:
cada $x$ es una solución --- la línea \emph{is} el eje.

\textbf{15.} Primer triple: tarifas $\frac{9 - 5}{3 - 1} = 2$ y
$\frac{17 - 9}{7 - 3} = 2$: igual, los puntos están alineados (en
$y = 2x + 3$). Segundo triple: $\frac{9-5}{2} = 2$ pero
$\frac{16 - 9}{3} = \frac73 \neq 2$: no alineado. \omterm{def:g9:linfunc:affine}{afín
funciones} son exactamente aquellos con una tasa de cambio constante ---
un número, el \omterm{prop:g9:linfunc:affinegraph}{pendiente}, cuenta toda la historia; para curvas el
La tasa misma cambia de un punto a otro, y perseguirla conduce
a la derivada, en el volumen High School.
\end{solution}
