\section{Proporcionalidad y funciones lineales.}

\begin{definition}[Función lineal]\label{def:g9:linfunc:linear}
A \emph{función lineal}\index{función lineal} multiplica cada número
mediante un número fijo $a$:
\[
f(x) = a x .
\]
El número $a$ es el \emph{coeficiente} de $f$. Dos cantidades son
\emph{proporcional}\index{proporcionalidad} exactamente cuando uno es lineal
función del otro.
\end{definition}

\begin{example}\label{ex:g9:linfunc:linear}
Si las manzanas cuestan $3$ el kilogramo, el precio de $x$ kilogramos es
$f(x) = 3x$: comprar el doble cuesta el doble, y el precio de
$2.5$ kg es $f(2.5) = 7.5$.

Por el contrario, si \omterm{def:g9:linfunc:linear}{función lineal} satisface $f(4) = 10$, entonces
$a = \frac{10}{4} = 2.5$ y $f(x) = 2.5x$: \emph{one value determines a
\omterm{def:g9:linfunc:linear}{función lineal} completely.}
\end{example}

\begin{proposition}[Gráfica de una función lineal]\label{prop:g9:linfunc:lineargraph}
La gráfica de $f(x) = ax$ es una línea recta \emph{a través del origen}.
Por el contrario, cada línea no vertical que pasa por el origen es la gráfica de un
\omterm{def:g9:linfunc:linear}{función lineal}.
\end{proposition}
\admitted

\section{Funciones afines}

\begin{definition}[Función afín]\label{def:g9:linfunc:affine}
Un \emph{función afín}\index{función afín} tiene la forma
\[
f(x) = a x + b ,
\]
donde $a$ y $b$ son números fijos. \omterm{def:g9:linfunc:linear}{Funciones lineales} son los especiales
caso $b = 0$; funciones constantes el caso $a = 0$.
\end{definition}

\begin{example}
Un gimnasio cobra una tarifa de inscripción de $20$, luego $5$ por visita: el total
El costo de las visitas $x$ es $f(x) = 5x + 20$. El costo es \emph{no}
\omterm{def:g9:linfunc:linear}{proporcional} al número de visitas (10 visitas no cuestan el doble
como 5), pero cada visita adicional suma la misma cantidad, $5$.
\end{example}

\begin{proposition}[Gráfica de una función afín]\label{prop:g9:linfunc:affinegraph}
La gráfica de $f(x) = ax + b$ es una línea recta:
\begin{itemize}
  \item $b = f(0)$ es el \emph{$y$-intercepción}: la línea cruza el
        eje vertical en $(0, b)$;
  \item $a$ es el \emph{pendiente}\index{pendiente}: mover una unidad a la
        hacia la derecha mueve $a$ unidades hacia arriba (hacia abajo si es $a < 0$); la función es
        aumentando cuando $a > 0$, disminuyendo cuando $a < 0$;
  \item para dos entradas cualesquiera $u \neq v$,
        $a = \dfrac{f(v) - f(u)}{v - u}$.
\end{itemize}
\end{proposition}

\begin{method}[Encontrar una función afín a partir de dos valores]\label{met:g9:linfunc:findaffine}
Dado $f(u)$ y $f(v)$ con $u \neq v$:
\begin{enumerate}
  \item calcula el \omterm{prop:g9:linfunc:affinegraph}{pendiente} $a = \dfrac{f(v) - f(u)}{v - u}$;
  \item sustituya uno de los dos valores conocidos en $f(x) = ax + b$ para
        encontrar $b$;
  \item consultar con el otro valor.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:linfunc:findaffine}
Encuentra el \omterm{def:g9:linfunc:affine}{función afín} con $f(2) = 7$ y $f(5) = 16$.
\[
a = \frac{16 - 7}{5 - 2} = \frac93 = 3 ,
\]
luego $f(2) = 3 \times 2 + b = 7$ da $b = 1$: $f(x) = 3x + 1$. Comprobar:
$f(5) = 15 + 1 = 16$.
\end{example}

\section{Porcentajes}

\begin{proposition}[cambio porcentual]\label{prop:g9:linfunc:percent}
Aumentar una cantidad en $t\,\%$ la multiplica por
$1 + \dfrac{t}{100}$; disminuyéndolo en $t\,\%$ se multiplica por
$1 - \dfrac{t}{100}$. Los cambios porcentuales son \omterm{def:g9:linfunc:linear}{funciones lineales}.
\end{proposition}

\begin{proof}
Aumentar $x$ en $t\,\%$ significa agregar $\frac{t}{100}\,x$:
\[
x + \frac{t}{100}\,x = \left(1 + \frac{t}{100}\right) x .
\]
Mismo cálculo con un signo menos para una disminución.
\end{proof}

\begin{example}\label{ex:g9:linfunc:percent}
Un precio de $80$ aumenta en $15\%$: nuevo precio
$80 \times 1.15 = 92$. Un precio de $60$ disminuye en $30\%$:
$60 \times 0.7 = 42$.

\emph{Encadenar cambios multiplica los factores.} Un aumento $20\%$
seguido de una disminución $20\%$ da
$1.2 \times 0.8 = 0.96$: un $4\%$ \emph{disminuir} en general, no un retorno a
¡el comienzo!
\end{example}
