\section{En expansión}

\begin{definition}[Ampliable]\label{def:g9:algebra:expand}
\emph{En expansión}\index{en expansión} significa transformar un \omterm{def:g2:mult:def}{producto} en un
\omterm{def:g1:addition:def}{suma}, usando distributividad:
\[
k(a + b) = ka + kb,
\qquad
(a + b)(c + d) = ac + ad + bc + bd .
\]
\end{definition}

\begin{example}\label{ex:g9:algebra:expand}
Cada término del primer paréntesis multiplica cada término del segundo:
\begin{align*}
(x + 3)(2x + 5)
&= x \times 2x + x \times 5 + 3 \times 2x + 3 \times 5 \\
&= 2x^2 + 5x + 6x + 15 \\
&= 2x^2 + 11x + 15 .
\end{align*}
Los signos viajan con los términos:
\begin{align*}
(3x - 2)(x - 4)
&= 3x^2 - 12x - 2x + 8
= 3x^2 - 14x + 8 .
\end{align*}
\end{example}

\begin{theorem}[las tres identidades]\label{thm:g9:algebra:identities}
Para todos los números $a$ y $b$:
\[
(a+b)^2 = a^2 + 2ab + b^2,
\qquad
(a-b)^2 = a^2 - 2ab + b^2,
\qquad
(a+b)(a-b) = a^2 - b^2 .
\]
\end{theorem}

\begin{proof}
Expande cada lado izquierdo. Para el primero:
$(a+b)(a+b) = a^2 + ab + ba + b^2 = a^2 + 2ab + b^2$. Para el segundo,
reemplace $b$ por $-b$ en el primero. Para el tercero:
$(a+b)(a-b) = a^2 - ab + ba - b^2 = a^2 - b^2$ (los términos cruzados se cancelan).
\end{proof}

\[

\section{Factorización}

\begin{method}[Factoraje]\label{met:g9:algebra:factor}
Para factorizar una expresión, intente en orden:
\begin{enumerate}
  \item \emph{factor común}: $ka + kb = k(a + b)$, donde $k$ puede
        ser un soporte;
  \item \emph{\omterm{ex:g1:subtraction:difference}{diferencia} of squares}: $a^2 - b^2 = (a+b)(a-b)$;
  \item \emph{cuadrado perfecto}: $a^2 + 2ab + b^2 = (a+b)^2$ (y con
        $-2ab$, $(a-b)^2$).
\end{enumerate}
Verifique siempre por \omterm{def:g9:algebra:expand}{en expansión} el resultado.
\end{method}

\begin{example}\label{ex:g9:algebra:factor}
\emph{Factor común:}
$15x^2 + 10x = 5x \times 3x + 5x \times 2 = 5x(3x + 2)$.

\emph{Soporte común:}
\[
(x+2)(x-1) + (x+2)(3x+4) = (x+2)\bigl[(x-1) + (3x+4)\bigr]
= (x+2)(4x+3).
\]

\emph{\omterm{ex:g1:subtraction:difference}{Diferencia} of squares:}
$x^2 - 81 = (x+9)(x-9)$, y
\[
(2x+1)^2 - 9 = (2x+1)^2 - 3^2 = (2x+1+3)(2x+1-3) = (2x+4)(2x-2).
\]
Cada factor todavía tiene un factor común: $(2x+4)(2x-2) = 2(x+2) \times
2(x-1) = 4(x+2)(x-1)$.
\end{example}

\section{Ecuaciones}

\begin{method}[Ecuaciones de primer grado]\label{met:g9:algebra:firstdegree}
Para resolver un \omterm{def:g8:equations:def}{ecuación} como $5x - 3 = 2x + 9$:
\begin{enumerate}
  \item reúne los términos $x$ en un lado y los números en el otro, por
        añadiendo la misma cantidad a ambos lados;
  \item reducir cada lado;
  \item divide ambos lados por el coeficiente de $x$;
  \item verifique la solución en el \omterm{def:g8:equations:def}{ecuación} original.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:algebra:firstdegree}
\begin{align*}
5x - 3 &= 2x + 9 \\
5x - 2x &= 9 + 3 && \text{(add $3 - 2x$ to both sides)}\\
3x &= 12 \\
x &= 4 && \text{(divide by 3).}
\end{align*}
Verifique: $5 \times 4 - 3 = 17$ y $2 \times 4 + 9 = 17$. La solución es
$4$.

Entre paréntesis, expanda primero: $3(x - 2) = x + 8$ se convierte en $3x - 6 = x + 8$,
entonces $2x = 14$ y $x = 7$.
\end{example}

\begin{theorem}[Regla de producto cero]\label{thm:g9:algebra:zeroproduct}
Un \omterm{def:g2:mult:def}{producto} es \omterm{def:g1:counting:zero}{cero} si y sólo si al menos un factor es \omterm{def:g1:counting:zero}{cero}:
\[
A \times B = 0
\quad\Longleftrightarrow\quad
A = 0 \ \text{ or } \ B = 0 .
\]
\end{theorem}

\begin{proof}
Si un factor es \omterm{def:g1:counting:zero}{cero}, el \omterm{def:g2:mult:def}{producto} claramente lo es. Por el contrario, si $A B = 0$ con
$A \neq 0$, al dividir ambos lados entre $A$ se obtiene $B = 0$.
\end{proof}

\begin{example}\label{ex:g9:algebra:zeroproduct}
Resuelva $(x - 3)(2x + 10) = 0$: ya sea $x - 3 = 0$ o $2x + 10 = 0$, por lo que
Las soluciones son $3$ y $-5$.

Resuelva $x^2 - 49 = 0$: factor primero, $(x-7)(x+7) = 0$, soluciones $7$ y
$-7$.

Resuelva $x^2 = 3x$: ¡nunca divida entre $x$! Mover y factorizar:
$x^2 - 3x = x(x - 3) = 0$, soluciones $0$ y $3$.
\end{example}

\section{Desigualdades}

\begin{proposition}[Reglas para desigualdades]\label{prop:g9:algebra:ineq}
Se conserva una desigualdad al sumar el mismo número a ambos lados, y
al multiplicar ambos lados por el mismo número \emph{positivo}.
Multiplicar ambos lados por un número \emph{negativo} invierte el
signo de desigualdad.
\end{proposition}

\begin{proof}
Si $a \leq b$ entonces $b - a \geq 0$. Agregando $c$: $(b+c) - (a+c) = b - a
\geq 0$, entonces $a + c \leq b + c$. Multiplicando por $c > 0$:
$bc - ac = (b-a)c \geq 0$, entonces $ac \leq bc$. Multiplicando por $c < 0$:
$(b-a)c \leq 0$, entonces $ac \geq bc$: el orden se invierte.
\end{proof}

\begin{example}\label{ex:g9:algebra:ineq}
Resuelva $7 - 2x < 15$:
\begin{align*}
7 - 2x &< 15 \\
-2x &< 8 && \text{(subtract 7)} \\
x &> -4 && \text{(divide by $-2$: reverse the sign).}
\end{align*}
Las soluciones son todos números mayores que $-4$: en una recta numérica, un
punto hueco en $-4$ y sombreado a la derecha.
\end{example}

\section{Problema: El álgebra de dígitos}

\begin{problem}[{Problema de fin de semana --- por qué manda la divisibilidad
trabajo, el truco 1089 desenmascarado y la aritmética mental que parece
como brujería}]
\label{pb:g9:algebra:1}
La escuela primaria te enseñó \emph{normas} (un número es \omterm{def:g6:wholes:divisible}{divisible} por
$9$ cuando su dígito sea \omterm{def:g1:addition:def}{suma}); magos realizan \emph{engaños}
(todos terminan en $1089$); los mercados saben \emph{atajos}
($35^2 = 1225$, al instante). Todo ello es el mismo secreto: un
número con dígitos $a$, $b$, $c$ es la expresión
$100a + 10b + c$ y el álgebra de este capítulo.
(\cref{thm:g9:algebra:identities}, \omterm{def:g9:algebra:expand}{en expansión}, factorización) puede tomar
aparte. Al final de este problema habrás demostrado la
reglas, desenmascaró el truco y aprendió la hechicería.

\medskip
\noindent\textbf{Parte I --- Digits under the microscope.}
\begin{enumerate}
  \item Un número de dos dígitos con el dígito de las decenas $a$ y el dígito de las unidades
        $b$ es $10a + b$. Expandir y factorizar el \omterm{ex:g1:subtraction:difference}{diferencia}
        entre el número y su inversión,
        $(10a + b) - (10b + a)$ e indique el descubrimiento. comprobar
        en $72 - 27$.
  \item Demostrar la regla del nueve para números de tres cifras: verificar
        la identidad
        \[
        100a + 10b + c = 9\,(11a + b) + (a + b + c),
        \]
        y explica por qué muestra que un número y su dígito \omterm{def:g1:addition:def}{suma}
        dejar el \emph{same \omterm{def:g3:division:remainder}{resto}} en la división por $9$
        --- entonces uno es \omterm{def:g6:wholes:divisible}{divisible} por $9$ exactamente cuando el otro es
        (\cref{prop:g9:arith:rules}, ahora probado).
  \item Deducir la regla de tres a partir de una misma identidad, en uno
        frase.
  \item ``Echando nueves'', el cheque de los contables: a prueba
        $47 \times 86 = 4\,042$, reemplace cada número por el
        dígito \omterm{def:g1:addition:def}{suma} de su dígito \omterm{def:g1:addition:def}{suma}: $47 \to 11 \to 2$,
        $86 \to 14 \to 5$ y $2 \times 5 = 10 \to 1$; entonces
        $4\,042 \to 10 \to 1$: consistente. Explique por qué una correcta
        \omterm{def:g2:mult:def}{producto} debe pasar la prueba (escriba $47 = 9k + 2$,
        $86 = 9l + 5$ y expandir), y encontrar un resultado incorrecto que
        la prueba no logra detectar (¿por qué se intercambian dígitos?
        ¿Invisible para él?).
  \item Demostrar la regla de once para números de cuatro dígitos: verificar
        \[
        1000a + 100b + 10c + d
        = 11\,(91a + 9b + c) + (- a + b - c + d),
        \]
        y utilícelo para decidir si $2\,926$ es \omterm{def:g6:wholes:divisible}{divisible} por
        $11$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The 1089 trick.}
El truco: piensa en un número de tres dígitos cuyo primer y último
los dígitos difieren en al menos $2$; invertirlo; restar el menor
del más grande; revertir el resultado; suma los dos últimos números.
El mago anuncia: \emph{1089}.
\begin{enumerate}[resume]
  \item Realiza el truco en $742$ y en un número de tu
        elección.
  \item Deje que el número tenga dígitos $a$, $b$, $c$ con $a > c$.
        Factoriza la primera resta,
        $(100a + 10b + c) - (100c + 10b + a)$ y enumere todos los
        valores que puede tomar cuando $a - c$ se ejecuta desde $2$ a $9$.
  \item Cada uno de esos \omterm{def:g5:division:multiple}{múltiplos} de $99$ tiene un dígito llamativo
        patrón. Escribiendo $m = a - c$, verificar la identidad
        \[
        99m = 100\,(m - 1) + 90 + (10 - m),
        \]
        y lea los tres dígitos de $99m$.
  \item Agregue $99m$ a su propia inversión, usando las preguntas 8
        dígitos, y demostrar que el total es $1089$ --- lo que sea
        $m$ es. El truco está muerto; ¡Viva el álgebra!
  \item ¿Por qué el mago requirió el primer y el último dígito?
        diferir en al menos $2$? Examinar los casos $a = c$ y
        $a - c = 1$ (donde la resta da $99$, que
        debe escribirse $099$ para que la inversión funcione), y
        Indique la letra pequeña que mantiene el truco honesto.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Sorcery for daily use.}
\begin{enumerate}[resume]
  \item El truco del cumpleaños: ``multiplica tu mes de nacimiento por $5$,
        suma $7$, duplica, suma tu día de nacimiento, resta $14$.''
        Demuestre que el resultado es siempre $10m + d$ --- mes y
        día, legible de un vistazo. (Pruébelo con alguien).
  \item Rediseñas el truco como: ``multiplica el mes por $5$,
        agregue $9$, doble, agregue el día". ¿Qué número debe
        espectador ahora resta al final por lo mismo
        lectura mes-día?
  \item El truco de los dígitos faltantes: el espectador elige cualquier número,
        resta su dígito \omterm{def:g1:addition:def}{suma} (la pregunta 2 dice que el resultado es
        siempre un \omterm{def:g9:arith:divisor}{múltiple} de $9$ --- explica por qué para un
        número de tres dígitos), luego tacha un \emph{distinto de cero}
        dígito y le lee los dígitos restantes. Explica como
        recuperas el dígito tachado, y por qué $0$ tuvo que ser
        excluido.
  \item Brujería en el mercado: probar la identidad
        $(10t + 5)^2 = 100\,t(t+1) + 25$, indique la regla para
        elevar al cuadrado cualquier número que termine en $5$ y calcular $45^2$,
        $85^2$ y $105^2$ en tu cabeza.
  \item Final, prolongando \cref{exo:g9:algebra:10}: demostrar que
        el \omterm{def:g1:addition:def}{suma} de cinco enteros consecutivos es siempre un
        \omterm{def:g9:arith:divisor}{múltiple} de $5$, pero que el \omterm{def:g1:addition:def}{suma} de \emph{cuatro}
        enteros consecutivos es \emph{nunca} a \omterm{def:g9:arith:divisor}{múltiple} de $4$.
        (Dos cálculos breves --- y una moraleja sobre lo que
        el álgebra hace con las palabras "siempre" y "nunca".)
\end{enumerate}
\end{problem}
