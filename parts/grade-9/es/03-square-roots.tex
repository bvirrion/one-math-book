\chapter{Raíces cuadradas}\label{ch:g9:sqrt}

¿Cuánto mide la diagonal de un cuadrado de lado $1$? El teorema de Pitágoras
responde $\sqrt 2$, un número cuyo cuadrado es $2$ --- y que resulta
no ser un \omterm{def:g9:fractions:fraction}{fracción}. Este capítulo define \omterm{def:g9:sqrt:def}{raíces cuadradas}, establece el
reglas para calcular con ellos y enseña cómo simplificar expresiones
como $\sqrt{75}$.

\section{Definición y primeras propiedades.}

\begin{definition}[raíz cuadrada]\label{def:g9:sqrt:def}
Deje $a \geq 0$. El \emph{raíz cuadrada}\index{raíz cuadrada} de $a$, escrito
$\sqrt a$, es el número único \emph{no negativo} cuyo cuadrado es $a$:
\[
\sqrt a \geq 0
\qquad\text{and}\qquad
\left(\sqrt a\right)^2 = a .
\]
Los números negativos no tienen raíz cuadrada, ya que todo cuadrado no es negativo.
\end{definition}

\begin{example}
$\sqrt{49} = 7$, $\sqrt 0 = 0$, $\sqrt 1 = 1$,
$\sqrt{2.25} = 1.5$. Los primeros \omterm{def:g9:sqrt:def}{raíces cuadradas} que debemos saber de memoria son los de
el \emph{cuadrados perfectos}: $1, 4, 9, 16, 25, 36, 49, 64, 81, 100, 121,
144$.
\end{example}

\begin{proposition}[Raíz cuadrada de un cuadrado]\label{prop:g9:sqrt:square}
Para cada número real $x$ (positivo o no):
\[
\sqrt{x^2} = \abs{x} =
\begin{cases}
x & \text{if } x \geq 0,\\
-x & \text{if } x < 0 .
\end{cases}
\]
\end{proposition}

\begin{proof}
El número $\abs{x}$ no es negativo y su cuadrado es $x^2$ (un número y
su opuesto tiene el mismo cuadrado). Siendo el número único no negativo
del cuadrado $x^2$, es $\sqrt{x^2}$.
\end{proof}

\begin{example}
$\sqrt{(-5)^2} = \sqrt{25} = 5 = \abs{-5}$: el \omterm{def:g9:sqrt:def}{raíz cuadrada} ``se olvida''
la señal, no la restaura. En particular $\sqrt{x^2} = x$ es
\emph{equivocado} para $x$ negativo.
\end{example}

\section{Productos y cocientes de raíces cuadradas.}

\begin{theorem}[Reglas de multiplicación y división.]\label{thm:g9:sqrt:rules}
Para todos los $a \geq 0$ y $b \geq 0$:
\[
\sqrt{a b} = \sqrt a \times \sqrt b,
\qquad\text{and for } b > 0:\quad
\sqrt{\frac ab} = \frac{\sqrt a}{\sqrt b} .
\]
\end{theorem}

\begin{proof}
El número $\sqrt a \times \sqrt b$ no es negativo (\omterm{def:g2:mult:def}{producto} de
números no negativos), y su cuadrado es
\[
\left(\sqrt a \times \sqrt b\right)^2
= \left(\sqrt a\right)^2 \times \left(\sqrt b\right)^2
= a b .
\]
Por unicidad del número no negativo cuyo cuadrado es $ab$, es igual
$\sqrt{ab}$. La regla \omterm{def:g3:division:remainder}{cociente} se demuestra de la misma manera.
\end{proof}

\begin{remark}[¡No existe tal regla para las sumas!]
$\sqrt{a + b}$ es \emph{no} $\sqrt a + \sqrt b$ en general:
\[
\sqrt{9 + 16} = \sqrt{25} = 5,
\qquad\text{but}\qquad
\sqrt 9 + \sqrt{16} = 3 + 4 = 7 .
\]
\end{remark}

\begin{method}[Simplificando $\sqrt n$]\label{met:g9:sqrt:simplify}
Para simplificar el \omterm{def:g9:sqrt:def}{raíz cuadrada} de un número entero:
\begin{enumerate}
  \item encuentre el cuadrado perfecto más grande dividiendo $n$ (factor $n$ si
        necesario);
  \item divide con la regla \omterm{def:g2:mult:def}{producto} y extrae el cuadrado:
        $\sqrt{k^2 m} = k\sqrt m$;
  \item comprueba que no hay ningún cuadrado perfecto \omterm{def:g9:arith:divisor}{divide} lo que queda debajo del
        raíz.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:sqrt:simplify}
Simplifique $\sqrt{75}$: desde $75 = 25 \times 3$,
\[
\sqrt{75} = \sqrt{25 \times 3} = \sqrt{25} \times \sqrt 3 = 5\sqrt3 .
\]
Simplifique $\sqrt{72}$: desde $72 = 36 \times 2$,
$\sqrt{72} = 6\sqrt2$. Y un \omterm{def:g3:division:remainder}{cociente}:
$\sqrt{\dfrac{49}{16}} = \dfrac{\sqrt{49}}{\sqrt{16}} = \dfrac74$.
\end{example}

\begin{example}[Añadiendo raíces cuadradas]\label{ex:g9:sqrt:add}
\omterm{def:g1:addition:def}{sumas} de \omterm{def:g9:sqrt:def}{raíces cuadradas} se simplifica solo cuando las raíces son \emph{similar}.
Calcule $\sqrt{12} + \sqrt{27} - \sqrt{48}$, simplificando primero cada término:
\begin{align*}
\sqrt{12} &= \sqrt{4 \times 3} = 2\sqrt3, \\
\sqrt{27} &= \sqrt{9 \times 3} = 3\sqrt3, \\
\sqrt{48} &= \sqrt{16 \times 3} = 4\sqrt3,
\end{align*}
entonces el \omterm{def:g1:addition:def}{suma} es $2\sqrt3 + 3\sqrt3 - 4\sqrt3 = (2 + 3 - 4)\sqrt3 = \sqrt3$.
\end{example}

\begin{example}[Expandiendo con raíces cuadradas]\label{ex:g9:sqrt:expand}
Las identidades del álgebra se aplican a \omterm{def:g9:sqrt:def}{raíces cuadradas}. Expandir
$\left(\sqrt5 + 2\right)^2$ con $(a+b)^2 = a^2 + 2ab + b^2$:
\[
\left(\sqrt5 + 2\right)^2 = 5 + 2 \times 2\sqrt5 + 4 = 9 + 4\sqrt5 .
\]
Y con la tercera identidad, $(a+b)(a-b) = a^2 - b^2$:
\[
\left(\sqrt7 + \sqrt3\right)\left(\sqrt7 - \sqrt3\right) = 7 - 3 = 4 :
\]
el \omterm{def:g2:mult:def}{producto} de dos números irracionales puede ser un número entero.
\end{example}

\section{La ecuación \texorpdfstring{$x^2 = a$}{x2 = a}}

\begin{theorem}[Resolviendo $x^2 = a$]\label{thm:g9:sqrt:equation}
\begin{enumerate}
  \item Si $a > 0$, \omterm{def:g8:equations:def}{ecuación} $x^2 = a$ tiene exactamente dos soluciones:
        $\sqrt a$ y $-\sqrt a$;
  \item si $a = 0$, la única solución es $0$;
  \item si $a < 0$, no hay solución.
\end{enumerate}
\end{theorem}

\begin{proof}
Reescribe $x^2 = a$ como $x^2 - \left(\sqrt a\right)^2 = 0$ cuando $a \geq 0$,
y factorizar como \omterm{ex:g1:subtraction:difference}{diferencia} de cuadrados:
\[
\left(x - \sqrt a\right)\left(x + \sqrt a\right) = 0 ,
\]
entonces $x = \sqrt a$ o $x = -\sqrt a$ (estos coinciden cuando $a = 0$). cuando
$a < 0$ no hay solución, ya que $x^2 \geq 0 > a$ por cada $x$.
\end{proof}

\begin{example}
$x^2 = 16$: soluciones $4$ y $-4$. \quad
$x^2 = 5$: soluciones $\sqrt5$ y $-\sqrt5$ (valores exactos; $\sqrt5
\approx 2.236$). \quad
$x^2 = -9$: sin solución. \quad
$3x^2 = 21$: dividir primero entre 3, $x^2 = 7$, soluciones $\pm\sqrt7$.
\end{example}

\section{Problema: El número que no es una fracción.}

\begin{problem}[{Problema del fin de semana --- la prueba legendaria de que
$\sqrt2$ es irracional, tiene muchos primos y el de Heron, asombrosamente
buena receta para aproximarlo}]
\label{pb:g9:sqrt:1}
La diagonal de un cuadrado unitario es una longitud perfectamente real.
lo dibujó en \cref{ex:g8:pythagoras:diagonal} --- y sin embargo, como el
Los pitagóricos descubrieron con horror unos veinticinco
Hace siglos, \emph{no \omterm{def:g9:fractions:fraction}{fracción} whatsoever} lo mide. Leyenda
Afirma que el descubrimiento fue castigado con ahogamiento. Este problema camina
a través de la prueba inmortal (cuatro preguntas y es tuya)
de por vida), multiplica las víctimas, y termina con la receta
Los ingenieros utilizaron durante dos mil años para domar lo indomable.

\medskip
\noindent\textbf{Parte I --- The proof.}
\begin{enumerate}
  \item Primero cazarlo: comprueba que
        $1.4 < \sqrt2 < 1.5$, entonces ese $1.41 < \sqrt2 < 1.42$,
        elevando al cuadrado los límites. Un dígito más: entre los cuales
        ¿Miente $\sqrt2$ los números tres decimales?
  \item Ahora supongamos --- en aras de la contradicción --- que
        $\sqrt2 = \frac ab$ para algunos \omterm{def:g9:fractions:fraction}{fracción} \emph{en lo más bajo
        términos} (\cref{met:g9:arith:simplify}). Cuadrar ambos lados
        y mostrar que $a^2 = 2 b^2$. ¿Cuál es la paridad de
        $a^2$?
  \item Los hechos de paridad de \cref{pb:g8:pythagoras:1} dicen: \omterm{def:g3:numbers:evenodd}{impar
        numeros} tiene cuadrados \omterm{def:g3:numbers:evenodd}{extraño}. Deduzca que $a$ es \omterm{def:g3:numbers:evenodd}{incluso}, escriba
        $a = 2k$, sustituir --- y mostrar que $b$ debe ser \omterm{def:g3:numbers:evenodd}{incluso}
        también.
  \item ¿Dónde está la contradicción? Concluya y exponga la
        teorema completo: \emph{$\sqrt2$ is irrational --- it is
        no \omterm{def:g3:division:remainder}{cociente} of whole numbers}.
  \item La prueba se apoyó una vez, discretamente, en las palabras ``en
        términos más bajos". Señale el paso exacto que
        colapsar sin ellos, y explicar por qué asumir el nivel más bajo
        términos eran legítimos en primer lugar.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The victims multiply.}
\begin{enumerate}[resume]
  \item Un segundo estilo de prueba, mediante factorizaciones primas.
        (\cref{thm:g9:arith:factorization}): en $a^2 = 3b^2$,
        compare la paridad del \emph{\omterm{def:g8:powers:def}{exponente} of $3$} en cada uno
        lado (\cref{pb:g9:arith:1}: los cuadrados llevan \omterm{def:g3:numbers:evenodd}{incluso}
        \omterm{def:g8:powers:def}{exponentes}). Concluya que $\sqrt3$ es irracional.
  \item Ejecute el mismo argumento \omterm{def:g8:powers:def}{exponente} en $a^2 = n b^2$ para un
        número entero general $n$: para el cual $n$ produce
        una contradicción, y ¿en qué falla? Indique el
        resultado completo: $\sqrt n$ es irracional exactamente
        cuando\,\dots
  \item Demuestre el lema del escudo: un racional distinto de cero multiplicado por un
        irracional es irracional. (Supongamos $r x = q$ con $r, q$
        racional, $r \neq 0$, y resuelva para $x$). Deduzca que
        $2\sqrt2$ y $\frac{\sqrt2}{2}$ son irracionales.
  \item Demuestre que $1 + \sqrt2$ es irracional. entonces la linda
        uno: suponiendo que $s = \sqrt2 + \sqrt3$ fueran racionales,
        calcule $s^2$, aísle $\sqrt6$ y encuentre el
        contradicción.
  \item Templar el entusiasmo: dar dos números irracionales cuyo
        \emph{\omterm{def:g1:addition:def}{suma}} es racional, y dos cuyo \emph{\omterm{def:g2:mult:def}{producto}} es
        racional. (La irracionalidad no se preserva mediante la aritmética.
        --- cada caso necesita su propia prueba.)
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Heron's recipe.}
Dos mil años antes de los calculadores, Garza de Alejandría
aproximado $\sqrt2$ así: \emph{guess $x$; replace the
guess by the average of $x$ and $\frac2x$; repeat.}
\begin{enumerate}[resume]
  \item Comience desde la suposición $x = 1$ y calcule los dos siguientes
        conjeturas como \omterm{def:g9:fractions:fraction}{fracciones} exacto.
  \item Calcule la tercera estimación, nuevamente como un \omterm{def:g9:fractions:fraction}{fracción} exacto, y
        su valor decimal. Comparar con la pregunta 1: ¿cuántos
        ¿Los decimales de $\sqrt2$ ya son correctos?
  \item La idea detrás de la receta: un rectángulo de \omterm{def:g6:measure:area}{área} $2$ con
        un lado $x$ tiene el otro lado $\frac2x$. mostrar eso
        $\sqrt2$ siempre miente \emph{entre} $x$ y $\frac2x$
        (considere los dos casos $x^2 > 2$ y $x^2 < 2$), entonces
        promediar los dos lados aprieta el rectángulo hacia
        la plaza.
  \item Las conjeturas $\frac32$, $\frac{17}{12}$,
        $\frac{577}{408}$ esconde una gema: calcula $3^2 - 2 \times
        2^2$, luego $17^2 - 2 \times 12^2$, luego

        \omterm{def:g9:fractions:fraction}{fracciones} ¿llegar a lo imposible?

\end{enumerate}
\end{problem}
