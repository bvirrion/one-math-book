\chapter{Fracciones y Potencias}\label{ch:g9:fractions}

\omterm{def:g9:fractions:fraction}{fracciones} y \omterm{def:g9:fractions:power}{potestades} son las herramientas cotidianas de computación: compartir
cantidades, comparar proporciones, escribir muy grande o muy pequeño
números. Este capítulo consolida las reglas para calcular con ellos ---
con cada paso intermedio escrito --- e introduce \omterm{def:g9:fractions:scientific}{científico
notación}.

\section{Calcular con fracciones}

\begin{definition}[fracción]\label{def:g9:fractions:fraction}
A \emph{fracción}\index{fracción} $\dfrac ab$ (con $b \neq 0$) es el
\omterm{def:g3:division:remainder}{cociente} de $a$ por $b$. Dos fracciones son iguales cuando una se obtiene de
el otro multiplicando (o dividiendo) \omterm{def:g4:fractions:def}{numerador} y \omterm{def:g4:fractions:def}{denominador} por el
mismo número distinto de cero:
\[
\frac ab = \frac{a \times k}{b \times k} \qquad (k \neq 0).
\]
\end{definition}

\begin{example}\label{ex:g9:fractions:simplify}
Simplifica $\dfrac{42}{56}$ paso a paso:
\[
\frac{42}{56} = \frac{42 \div 2}{56 \div 2} = \frac{21}{28}
= \frac{21 \div 7}{28 \div 7} = \frac34 .
\]
(En \cref{ch:g9:arith} el mayor común \omterm{def:g5:division:multiple}{divisor} hará esto de una vez
paso.)
\end{example}

\begin{method}[Suma y resto de fracciones]\label{met:g9:fractions:add}
\begin{enumerate}
  \item Coloque ambos \omterm{def:g9:fractions:fraction}{fracciones} sobre un \emph{common \omterm{def:g4:fractions:def}{denominador}} (un
        \omterm{def:g5:division:multiple}{múltiple} de los dos \omterm{def:g4:fractions:def}{denominadores});
  \item suma o resta el \omterm{def:g4:fractions:def}{numeradores}, manteniendo el \omterm{def:g4:fractions:def}{denominador};
  \item simplifique el resultado si es posible.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:fractions:add}
Calcule $\dfrac56 + \dfrac34$. Un \omterm{def:g4:fractions:def}{denominador} común de $6$ y $4$ es
$12$:
\[
\frac56 + \frac34
= \frac{5 \times 2}{6 \times 2} + \frac{3 \times 3}{4 \times 3}
= \frac{10}{12} + \frac{9}{12}
= \frac{19}{12}.
\]
Calcule $2 - \dfrac37$. Escriba $2$ como \omterm{def:g9:fractions:fraction}{fracción} sobre $7$:
\[
2 - \frac37 = \frac{14}{7} - \frac37 = \frac{11}{7}.
\]
\end{example}

\begin{method}[Multiplicar y dividir fracciones]\label{met:g9:fractions:mult}
\begin{enumerate}
  \item Para multiplicar, multiplica \omterm{def:g4:fractions:def}{numeradores} juntos y \omterm{def:g4:fractions:def}{denominadores}
        juntos: $\dfrac ab \times \dfrac cd = \dfrac{ac}{bd}$
        (simplifique \emph{antes} multiplicando siempre que sea posible);
  \item para dividir, multiplicar por el \omterm{def:g8:fractions:inverse}{inverso} del \omterm{def:g5:division:multiple}{divisor}:
        $\dfrac ab \div \dfrac cd = \dfrac ab \times \dfrac dc$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:fractions:mult}
\[
\frac{7}{15} \times \frac{25}{14}
= \frac{7 \times 25}{15 \times 14}
= \frac{7 \times 25}{15 \times 2 \times 7}
= \frac{25}{30} = \frac56,
\]
donde simplificamos por el factor común $7$, luego por $5$. y un
división:
\[
\frac{3}{4} \div \frac{9}{8}
= \frac34 \times \frac89
= \frac{3 \times 8}{4 \times 9}
= \frac{24}{36} = \frac23 .
\]
\end{example}

\section{Potestades}

\begin{definition}[poder]\label{def:g9:fractions:power}
Para un número real $a$ y un entero positivo $n$, el
\emph{fuerza}\index{fuerza} $a^n$ es el \omterm{def:g2:mult:def}{producto} de $n$ factores iguales a
$a$:
\[
a^n = \underbrace{a \times a \times \dots \times a}_{n \text{ factors}} .
\]
Por convención $a^0 = 1$ (para $a \neq 0$), y \omterm{def:g8:powers:def}{exponentes} negativo denotan
\omterm{def:g8:fractions:inverse}{inversas}:
\[
a^{-n} = \frac{1}{a^n}.
\]
\end{definition}

\begin{example}
$2^4 = 2 \times 2 \times 2 \times 2 = 16$; \quad
$(-3)^2 = 9$ pero $-3^2 = -(3^2) = -9$ (el \omterm{def:g8:powers:def}{exponente} se une antes que el
signo menos); \quad$10^{-3} = \frac{1}{1000} = 0.001$; \quad
$5^1 = 5$ y $5^0 = 1$.
\end{example}

\begin{theorem}[Reglas de exponentes]\label{thm:g9:fractions:rules}
Para todos los $a$, $b$ distintos de cero y todos los números enteros $m$, $n$:
\[
a^m \times a^n = a^{m+n},
\qquad
\frac{a^m}{a^n} = a^{m-n},
\qquad
\left(a^m\right)^n = a^{mn},
\qquad
(ab)^n = a^n b^n .
\]
\end{theorem}

\begin{proof}
Para \omterm{def:g8:powers:def}{exponentes} positivo, cuente los factores. Para la primera regla:
$a^m \times a^n$ es $m$ factores $a$ seguidos de $n$ factores $a$, en total
$m + n$ factores. Para el tercero:
$\left(a^m\right)^n$ repite $n$ veces un bloque de factores $m$, dando
$mn$ factores. Las otras dos reglas son similares; la extensión a \omterm{def:g1:counting:zero}{cero} y
\omterm{def:g8:powers:def}{exponentes} negativo se verifica desde $a^{-n} = \frac{1}{a^n}$ (y hace
la convención $a^0 = 1$ es la única opción consistente, ya que
$a^n \times a^0$ debe ser $a^{n+0} = a^n$).
\end{proof}

\begin{example}\label{ex:g9:fractions:rules}
Simplifica paso a paso:
\[
\frac{3^7 \times 3^{-2}}{3^4}
= \frac{3^{7 + (-2)}}{3^4}
= \frac{3^5}{3^4}
= 3^{5-4} = 3^1 = 3 .
\]
Y con dos letras: $(2a^3)^2 \times a = 4 a^6 \times a = 4a^7$.
\end{example}

\section{Potencias de diez y notación científica.}

\begin{proposition}[Potencias de diez]\label{prop:g9:fractions:ten}
Para un número entero positivo $n$: $10^n$ se escribe ``$1$ seguido de $n$
\omterm{def:g1:counting:zero}{ceros}'' y $10^{-n} = 0.0\dots01$ con $1$ en el decimal $n$
lugar. Multiplicar un \omterm{def:g6:decimals:places}{numero decimal} por $10^n$ (resp.\ $10^{-n}$) cambia
sus \omterm{def:g4:decimals:point}{coma decimal} $n$ se colocan a la derecha (resp.\ izquierda).
\end{proposition}

\begin{definition}[Notación científica]\label{def:g9:fractions:scientific}
El \emph{notación científica}\index{notación científica} de un positivo
\omterm{def:g6:decimals:places}{numero decimal} es la forma única de escribirlo como
\[
a \times 10^n,
\qquad\text{with } 1 \leq a < 10 \text{ and } n \text{ an integer.}
\]
\end{definition}

\begin{example}\label{ex:g9:fractions:scientific}
La distancia Tierra-Sol es de aproximadamente $149\,600\,000$ km $= 1.496 \times
10^8$ km. El tamaño de una molécula de agua es de aproximadamente $0.000\,000\,000\,28$ m.
$= 2.8 \times 10^{-10}$m. Para calcular con tales números, agrupe los
partes decimales y el \omterm{def:g9:fractions:power}{potestades} de diez:
\[
(3 \times 10^5) \times (2.5 \times 10^{-2})
= (3 \times 2.5) \times 10^{5 + (-2)}
= 7.5 \times 10^{3}.
\]
\end{example}

\begin{method}[Poner un número en notación científica]\label{met:g9:fractions:scientific}
\begin{enumerate}
  \item Mueva el \omterm{def:g4:decimals:point}{coma decimal} justo después del primer dígito distinto de cero; esto
        da el factor $a$ con $1 \leq a < 10$;
  \item cuenta el número de lugares que se movió el punto: ese recuento es $n$,
        positivo si el número original era $\geq 10$, negativo si era
        $< 1$;
  \item Comprobación : $a \times 10^n$ debe reproducir el número original.
\end{enumerate}
\end{method}

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:fractions:10}

\end{exercise}

\section{Problema: el código secreto de los decimales periódicos}

\begin{problem}[{Problema de fin de semana --- cada fracción tiene un decimal
escritura que se detiene o se repite, cada periódico decimal es un
fracción, y $0.999\ldots$ es igual exactamente a $1$}]
\label{pb:g9:fractions:1}
Divide $3$ por $11$ y los dígitos caen en un canto:
$0.272727\ldots$ para siempre. Divide $1$ por $4$ y los dígitos se detienen
muerto: $0.25$. Este problema demuestra que estos son los únicos dos
posibles comportamientos para un \omterm{def:g9:fractions:fraction}{fracción} --- y luego invierte el
máquina: dado cualquier decimal periódico, reconstruye el \omterm{def:g9:fractions:fraction}{fracción}
escondiéndose detrás de él. En el camino se asienta, de una vez por todas, la
El tema más discutido es el de la igualdad en las matemáticas escolares.

\medskip
\noindent\textbf{Parte I --- From \omterm{def:g9:fractions:fraction}{fracción} to decimal.}
\begin{enumerate}
  \item Calcular por división larga las escrituras decimales de
        $\frac14$, $\frac13$, $\frac56$ y $\frac{3}{11}$.
        ¿Qué parada, qué repetición y con qué bloque repetitivo?
  \item Explique por qué la escritura decimal de \emph{cualquier} \omterm{def:g9:fractions:fraction}{fracción}
        $\frac ab$ debe detenerse o eventualmente repetirse. (En el
        división larga por $b$, cuyos valores pueden los sucesivos
        \omterm{def:g3:division:remainder}{restos} tomar? ¿Qué pasa en el momento en que \omterm{def:g3:division:remainder}{resto}?
        aparece por segunda vez?)
  \item Empuje la división de $1$ por $7$ lo suficiente para encontrar el
        bloque repetitivo. ¿Cuánto mide y cómo mide esa longitud?
        comparar con el límite prometido en la pregunta 2?
  \item Algunos \omterm{def:g9:fractions:fraction}{fracciones} se detienen porque su \omterm{def:g4:fractions:def}{denominador} puede ser
        ampliado a un \omterm{def:g9:fractions:power}{fuerza} de diez: $\frac14 =
        \frac{25}{100}$, $\frac{3}{8} = \frac{375}{1000}$.
        Explique el criterio y por qué no hay multiplicador de números enteros.
        alguna vez puede convertir $3$, $6$ o $11$ en $10$, $100$,
        $1000, \dots$ (¿en qué dígito terminaría el \omterm{def:g2:mult:def}{producto}?).
  \item Sin dividir, ordenar $\frac{9}{40}$, $\frac{5}{12}$ y
        $\frac{33}{50}$ en ``paradas'' y ``repeticiones'', luego dé
        la escritura decimal de los dos que paran.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- From repeating decimal back to the
\omterm{def:g9:fractions:fraction}{fracción}.}
\begin{enumerate}[resume]
  \item El truco del cambio. Deje que $x = 0.7777\ldots$ calcule $10x$,
        luego $10x - x$ y deduzca $x$ como \omterm{def:g9:fractions:fraction}{fracción}. Consultar por
        dividiendo.
  \item Deje $x = 0.727272\ldots$ Cual \omterm{def:g9:fractions:power}{fuerza} de diez turnos $x$ por
        ¿Exactamente un bloque repetido? Úselo para escribir $x$ como
        \omterm{def:g9:fractions:fraction}{fracción} en términos más bajos.
  \item Un caso mixto: $x = 0.58333\ldots$ (el $3$ solo
        se repite). Calcule $1000x - 100x$ y deduzca $x$ como
        \omterm{def:g9:fractions:fraction}{fracción} en términos más bajos.
  \item El famoso: aplica el truco del cambio a
        $x = 0.9999\ldots$ Qué \omterm{def:g9:fractions:fraction}{fracción} --- qué \emph{número}
        --- ¿lo encuentras? Concilia el resultado con tu
        intuición, recordando al vecino fantasma de
        \cref{pb:g6:decimals:1} y el cada vez más reducido \omterm{def:g3:division:remainder}{resto} de
        \cref{pb:g6:fractions:1}: es $0.999\ldots$ ``justo debajo''
        ¿$1$ u otro nombre para $1$?
  \item Convertir $2.454545\ldots$ y $0.123123123\ldots$ en
        \omterm{def:g9:fractions:fraction}{fracciones} en términos más bajos.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The dictionary completed.}
\begin{enumerate}[resume]
  \item Las preguntas 6 a 10 sugieren un diccionario: un bloque de $k$
        Los dígitos que se repiten desde \omterm{def:g4:decimals:point}{coma decimal} son iguales a ese bloque.
        over $k$ nines ($0.727272\ldots = \frac{72}{99}$). uso
        el diccionario para predecir $0.037037\ldots$ como \omterm{def:g9:fractions:fraction}{fracción},
        simplificar --- y maravillarse: qué sorprendentemente simple
        \omterm{def:g9:fractions:fraction}{fracción} chants ``$037$''? Compruébalo por división.
  \item Combina el diccionario con un turno: escribe
        $0.0272727\ldots$ as a \omterm{def:g9:fractions:fraction}{fracción} in lowest terms.
  \item El número $0.101001000100001\ldots$ (un $1$, luego uno
        $0$, luego un $1$, luego dos $0$, luego un $1$, luego tres
        $0$s, and so on) never stops. Is it a repeating decimal?
        ¿Qué dice entonces la Parte I al respecto? ¿Puede ser una
        \omterm{def:g9:fractions:fraction}{fracción}? (You have just met your first provably
        \emph{irracional} número; the most famous one is caught
        en \cref{ch:g9:sqrt}.)
  \item ¿Aproximadamente cuántos dígitos tiene $2^{30}$? uso
        $2^{10} = 1\,024 \approx 1.024 \times 10^3$ y el
        \omterm{def:g8:powers:def}{exponente} rules (\cref{thm:g9:fractions:rules}) to write
        $2^{30}$ in (approximate) \omterm{def:g9:fractions:scientific}{notación científica}, and
        concluir. (Compárese \cref{exo:g9:fractions:10}.)
  \item Final: escriba $0.20262026\ldots$ como \omterm{def:g9:fractions:fraction}{fracción}, y
        enunciar la moraleja bidireccional de todo el problema: ¿cuál
        las escrituras decimales son \omterm{def:g9:fractions:fraction}{fracciones}, y que \omterm{def:g9:fractions:fraction}{fracciones} tienen
        ¿Qué escrituras decimales?
\end{enumerate}
\end{problem}
