\chapter{Trigonometría en el triángulo rectángulo}\label{ch:g9:trig}

En un \omterm{def:g6:shapes:triangles}{triangulo rectángulo}, una vez que conoces un ángulo agudo, el \emph{forma} de
el triángulo es fijo --- sólo su tamaño puede variar. Las proporciones de sus lados.
Por lo tanto dependen sólo del ángulo: estas relaciones son \omterm{def:g9:trig:ratios}{coseno}, \omterm{def:g9:trig:ratios}{seno} y
\omterm{def:g9:trig:ratios}{tangente}. Junto con el teorema de Pitágoras, te permiten calcular
cada lado y cada ángulo de un \omterm{def:g6:shapes:triangles}{triangulo rectángulo} desde muy poco
información.

\section{El teorema de Pitágoras, de nuevo}

\begin{theorem}[Pitágoras]\label{thm:g9:trig:pythagoras}
En un triángulo $ABC$ \omterm{def:g6:shapes:triangles}{en ángulo recto} en $A$, el cuadrado de la hipotenusa es
el \omterm{def:g1:addition:def}{suma} de los cuadrados de los dos catetos:
\[
BC^2 = AB^2 + AC^2 .
\]
Por el contrario, si $BC^2 = AB^2 + AC^2$ está en un triángulo, entonces el triángulo es
\omterm{def:g6:shapes:triangles}{en ángulo recto} en $A$.
\end{theorem}
\admitted

\begin{example}\label{ex:g9:trig:pythagoras}
Un \omterm{def:g6:shapes:triangles}{triangulo rectángulo} tiene patas $AB = 5$ y $AC = 12$. entonces
\[
BC^2 = 5^2 + 12^2 = 25 + 144 = 169,
\qquad\text{so}\quad BC = \sqrt{169} = 13 .
\]
Para encontrar un \emph{pierna}: si $BC = 10$ y $AB = 6$, entonces
$AC^2 = BC^2 - AB^2 = 100 - 36 = 64$, entonces $AC = 8$.
\end{example}

\section{Coseno, seno, tangente}

En un \omterm{def:g6:shapes:triangles}{triangulo rectángulo}, respecto a un ángulo agudo $\theta$, los tres lados
tienen nombres: el \emph{hipotenusa} (frente al \omterm{def:g3:shapes:rightangle}{ángulo recto}, el más largo
lado), el lado \emph{opuesto} a $\theta$, y el lado
\emph{adyacente} a $\theta$ (la pierna tocando $\theta$).

\begin{definition}[Razones trigonométricas]\label{def:g9:trig:ratios}
Para un ángulo agudo $\theta$ de un \omterm{def:g6:shapes:triangles}{triangulo rectángulo}:
\[
\cos\theta = \frac{\text{adjacent}}{\text{hypotenuse}},
\qquad
\sin\theta = \frac{\text{opposite}}{\text{hypotenuse}},
\qquad
\tan\theta = \frac{\text{opposite}}{\text{adjacent}} .
\]
Estas relaciones\index{coseno}\index{seno}\index{tangente} dependen únicamente de
$\theta$, no en el tamaño del triángulo (todos \omterm{def:g6:shapes:triangles}{triangulos rectángulos} con el
mismo ángulo agudo son escalas entre sí, por Tales).
\end{definition}

\begin{proposition}[Primeras propiedades]\label{prop:g9:trig:properties}
Para cada ángulo agudo $\theta$:
\begin{enumerate}
  \item $0 < \cos\theta < 1$ y $0 < \sin\theta < 1$ (una pierna es más corta
        que la hipotenusa);
  \item $\cos^2\theta + \sin^2\theta = 1$;
  \item $\tan\theta = \dfrac{\sin\theta}{\cos\theta}$.
\end{enumerate}
\end{proposition}

\begin{proof}
Escriba $a$ para el lado adyacente, $o$ para el lado opuesto, $h$ para el
hipotenusa de un \omterm{def:g6:shapes:triangles}{triangulo rectángulo} con ángulo $\theta$.

1. $0 < a < h$ y $0 < o < h$, por lo que las dos proporciones están estrictamente entre
$0$ y $1$.

2. Por Pitágoras, $a^2 + o^2 = h^2$. Dividir todo por $h^2$:
\[
\cos^2\theta + \sin^2\theta
= \frac{a^2}{h^2} + \frac{o^2}{h^2}
= \frac{a^2 + o^2}{h^2} = 1 .
\]

3. $\dfrac{\sin\theta}{\cos\theta} = \dfrac{o/h}{a/h} = \dfrac oa
= \tan\theta$.
\end{proof}

\begin{method}[Encontrando un lado]\label{met:g9:trig:side}
Para calcular un lado desconocido cuando se conocen un lado y un ángulo agudo:
\begin{enumerate}
  \item marque el ángulo conocido y etiquete los tres lados (hipotenusa,
        opuesto, adyacente) \emph{desde ese ángulo};
  \item elige la relación (cos, sen o tan) que involucra al lado conocido
        y el lado buscado;
  \item escribe el \omterm{def:g8:equations:def}{ecuación} y resuélvelo;
  \item control de cordura: la hipotenusa debe ser la más larga.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:trig:side}
En un \omterm{def:g6:shapes:triangles}{triangulo rectángulo}, la hipotenusa mide $8$ y un ángulo es
$35^\circ$; Calcula el lado opuesto a él. La relación que involucra la
El lado opuesto y la hipotenusa es el \omterm{def:g9:trig:ratios}{seno}:
\[
\sin 35^\circ = \frac{\text{opposite}}{8}
\quad\Longrightarrow\quad
\text{opposite} = 8 \sin 35^\circ \approx 8 \times 0.574 \approx 4.6 .
\]
Para el lado adyacente, utilice el \omterm{def:g9:trig:ratios}{coseno}:
$8\cos 35^\circ \approx 8 \times 0.819 \approx 6.6$. Consulte con
Pitágoras: $4.6^2 + 6.6^2 \approx 21 + 43 \approx 64 = 8^2$.
\end{example}

\begin{method}[Encontrar un ángulo]\label{met:g9:trig:angle}
Cuando se conocen dos lados, calcule la razón que forman, identifíquela como
\omterm{def:g9:trig:ratios}{coseno}, \omterm{def:g9:trig:ratios}{seno} o \omterm{def:g9:trig:ratios}{tangente} del ángulo desconocido y aplique la ecuación de la calculadora.
Función \omterm{def:g8:fractions:inverse}{inverso} ($\cos^{-1}$, $\sin^{-1}$ o $\tan^{-1}$) a la relación.
\end{method}

\begin{example}\label{ex:g9:trig:angle}
Una escalera de $5$ m de largo está apoyada contra una pared, su pie $1.4$ m de la
pared. El ángulo $\theta$ entre la escalera y el suelo satisface
\[
\cos\theta = \frac{1.4}{5} = 0.28,
\qquad\text{so}\quad
\theta = \cos^{-1}(0.28) \approx 73.7^\circ .
\]
\end{example}

\begin{example}[Ángulos especiales]\label{ex:g9:trig:special}
Dos triángulos dan valores exactos. \omterm{def:g3:division:half}{Medio} un cuadrado de lado 1 (un cuadrado
\omterm{def:g6:shapes:triangles}{triangulo isósceles}) tiene hipotenusa $\sqrt2$ y ángulos de $45^\circ$:
\[
\cos 45^\circ = \sin 45^\circ = \frac{1}{\sqrt2} = \frac{\sqrt2}{2},
\qquad \tan 45^\circ = 1 .
\]
\omterm{def:g3:division:half}{Medio} y \omterm{def:g6:shapes:triangles}{triangulo equilatero} del lado 1 dan los ángulos $30^\circ$ y
$60^\circ$:
\[
\cos 60^\circ = \sin 30^\circ = \frac12,
\qquad
\cos 30^\circ = \sin 60^\circ = \frac{\sqrt3}{2}.
\]
\end{example}
