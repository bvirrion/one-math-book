\chapter{Teorema de Tales}\label{ch:g9:thales}

¿Cómo se mide la altura de un \omterm{def:g8:solids:def}{pirámide} sin subirlo? Tales
Mileto comparó su sombra con la sombra de un palo. Su teorema
--- \omterm{def:g6:lines:perp}{lineas paralelas} corta \omterm{def:g6:lines:objects}{segmentos} en pedazos \omterm{def:g9:linfunc:linear}{proporcional} --- es el
corazón matemático de modelos, mapas y ampliaciones \omterm{def:g7:prop:scale}{escala}, y uno de
Los teoremas más antiguos con nombre.

\section{el teorema}

\begin{theorem}[Cuentos]\label{thm:g9:thales:direct}
Dejemos que dos líneas se crucen en un punto $A$. Tome $M$ en la primera línea y $N$
en el segundo, con $B$ más adelante en el primero y $C$ más adelante
el segundo. Si las líneas $(MN)$ y $(BC)$ son \emph{\omterm{def:g6:lines:perp}{paralelo}}, entonces el
Los triángulos $AMN$ y $ABC$ tienen lados \omterm{def:g9:linfunc:linear}{proporcional}:
\[
\frac{AM}{AB} = \frac{AN}{AC} = \frac{MN}{BC}.
\]
\end{theorem}
\admitted

\begin{method}[Calculando una longitud con Tales]\label{met:g9:thales:compute}
\begin{enumerate}
  \item Identifica el punto $A$ donde se cruzan las dos líneas y las dos
        \omterm{def:g6:lines:perp}{lineas paralelas}; nombra los dos triángulos;
  \item escribe las tres razones iguales, siempre ``triángulo pequeño sobre grande
        triángulo'';
  \item mantener la igualdad de las dos razones que involucran a los tres conocidos
        longitudes y la desconocida;
  \item resuelve la proporción resultante mediante multiplicación cruzada.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:thales:compute}
En la configuración izquierda arriba, supongamos $AM = 3$, $AB = 5$, $AN = 2.4$.
y $MN = 3.3$, y calculemos $AC$ y $BC$. Tales da
\[
\frac{AM}{AB} = \frac{AN}{AC} = \frac{MN}{BC},
\qquad\text{i.e.}\qquad
\frac35 = \frac{2.4}{AC} = \frac{3.3}{BC}.
\]
De $\frac35 = \frac{2.4}{AC}$, multiplicación cruzada:
$3 \times AC = 5 \times 2.4 = 12$, entonces $AC = 4$. De
$\frac35 = \frac{3.3}{BC}$: $3 \times BC = 16.5$, entonces $BC = 5.5$.
\end{example}

\begin{example}[Cuentos de la vida real]\label{ex:g9:thales:stick}
Un palo vertical de altura $1$ m proyecta una sombra de $1.5$ m, mientras que un árbol
proyecta una sombra de $12$ m al mismo tiempo. Los rayos del sol son \omterm{def:g6:lines:perp}{paralelo}, entonces
los triángulos del palo y la sombra y del árbol y la sombra están en Tales
configuración:
\[
\frac{\text{height of tree}}{1} = \frac{12}{1.5},
\qquad\text{so the tree is } 8 \text{ m tall.}
\]
\end{example}

\section{lo contrario}

\begin{theorem}[Conversación de cuentos]\label{thm:g9:thales:converse}
Con los puntos colocados como en \cref{thm:g9:thales:direct} ($M$ en $(AB)$,
$N$ en $(AC)$, en el mismo orden en ambas líneas):
si
\[
\frac{AM}{AB} = \frac{AN}{AC},
\]
entonces las líneas $(MN)$ y $(BC)$ son \omterm{def:g6:lines:perp}{paralelo}.
\end{theorem}
\admitted

\begin{method}[Probar o refutar el paralelismo]\label{met:g9:thales:converse}
\begin{enumerate}
  \item Calcule por separado las dos relaciones $\dfrac{AM}{AB}$ y
        $\dfrac{AN}{AC}$ (como \omterm{def:g9:fractions:fraction}{fracciones}, no \omterm{def:g6:decimals:rounding}{redondeos});
  \item si son iguales \emph{y} los puntos están en el mismo orden en
        las dos líneas, las líneas son \omterm{def:g6:lines:perp}{paralelo} (conversa de Tales);
  \item si difieren, las líneas son \emph{no} \omterm{def:g6:lines:perp}{paralelo} --- porque si
        Si lo fueran, el teorema de Tales obligaría a que las proporciones fueran iguales.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:thales:converse}
En la línea 1, $AM = 4$ y $AB = 10$; en la línea 2, $AN = 6$ y $AC = 15$.
entonces
\[
\frac{AM}{AB} = \frac{4}{10} = \frac25,
\qquad
\frac{AN}{AC} = \frac{6}{15} = \frac25 :
\]
proporciones iguales, mismo orden: $(MN) \parallel (BC)$.

Con $AC = 14$ en su lugar: $\frac{6}{14} = \frac37 \neq \frac25$, entonces el
Las líneas no son \omterm{def:g6:lines:perp}{paralelo}.
\end{example}

\section{Ampliación y reducción}

\begin{definition}[escalando una figura]\label{def:g9:thales:scaling}
Ampliar o reducir una cifra en \emph{escala
factor}\index{factor de escala} $k > 0$ significa multiplicar \emph{todo} por su
longitudes por $k$: un \emph{ampliación} cuando $k > 1$, un \emph{reducción}
cuando $k < 1$. En la configuración de Tales, el triángulo $AMN$ es el
reducción de $ABC$ por el factor $k = \frac{AM}{AB}$.
\end{definition}

\begin{proposition}[Efecto en ángulos y áreas]\label{prop:g9:thales:scaling}
La escala por el factor $k$ preserva los ángulos y la forma de las figuras,
multiplica cada longitud por $k$ y multiplica cada \emph{\omterm{def:g6:measure:area}{área}} por
$k^2$.
\end{proposition}
\admitted

\begin{omfigure}
\begin{tikzpicture}[scale=0.72]
  \draw[thick, omDef, fill=omDef!8] (0,0) rectangle (2,1.2);
  \draw[thick, omThm, fill=omThm!8] (3.4,0) rectangle (7.4,2.4);
  \node[font=\small, omDef] at (1,-0.45) {$2 \times 1.2$};
  \node[font=\small, omThm] at (5.4,-0.45) {$4 \times 2.4$};
  \node[font=\small] at (1,0.6) {area $2.4$};
  \node[font=\small] at (5.4,1.2) {area $9.6$};
\end{tikzpicture}

{\small Duplicar las longitudes ($k = 2$) multiplica el \omterm{def:g6:measure:area}{área} por
$k^2 = 4$: cuatro copias del rectángulo pequeño en mosaico del grande.}
\end{omfigure}

\begin{example}
Una foto de $10 \times 15$ cm está ampliada con \omterm{def:g9:thales:scaling}{factor de escala} $k = 3$: la
medidas de impresión $30 \times 45$ cm. Su \omterm{def:g6:measure:area}{área} va desde $150$ cm$^2$ a
$150 \times 9 = 1350$ cm$^2$ --- $9$ veces más grande, no $3$.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g9:thales:1}
Las líneas $(MN)$ y $(BC)$ son \omterm{def:g6:lines:perp}{paralelo}, con $M$ en $[AB]$ y $N$ en
$[AC]$; $AM = 2$, $AB = 6$, $AN = 3$, $BC = 9$. Calcule $AC$ y $MN$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:thales:2}
Misma configuración: $AM = 5$, $MB = 3$ (cuidado: $MB$, no $AB$!),
$AN = 4$. Calcule $AC$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:thales:3}
En una configuración de mariposa, $A$ está entre $M$ y $B$ y entre $N$
y $C$, con $(MN) \parallel (BC)$; $AM = 3$, $AB = 7.5$, $AN = 2$,
$MN = 2.6$. Calcule $AC$ y $BC$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:thales:4}
$M$ está en $[AB]$ con $AM = 6$, $AB = 8$; $N$ está en $[AC]$ con $AN = 9$,
$AC = 12$. ¿Las líneas $(MN)$ y $(BC)$ son \omterm{def:g6:lines:perp}{paralelo}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:5}
Igual que arriba con $AM = 4$, $AB = 6$, $AN = 5$, $AC = 8$. Son $(MN)$
y $(BC)$ \omterm{def:g6:lines:perp}{paralelo}? Justifique cuidadosamente.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:6}
Una persona de $1.8$ m de altura se encuentra a $2$ m de distancia de la base de una farola;
su sombra mide $3$ m. Dibuja la configuración de Tales formada por
lámpara, la persona y la punta de la sombra, y calcular la altura de
la lámpara.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:7}
Un mapa tiene \omterm{def:g7:prop:scale}{escala} $1 : 25\,000$ (las longitudes en el mapa son las longitudes reales
multiplicado por $k = \frac{1}{25000}$).
\begin{enumerate}
  \item Dos pueblos están separados por $6.8$ cm en el mapa. ¿Qué es lo real?
        distancia, en km?
  \item Un bosque tiene un \omterm{def:g6:measure:area}{área} de $8$ cm$^2$ en el mapa. cual es su verdadera
        \omterm{def:g6:measure:area}{área}, en km$^2$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:8}
El triángulo $ABC$ tiene $AB = 12$, $AC = 15$, $BC = 18$. El punto $M$ en
$[AB]$ satisface $AM = 8$, y la línea que pasa por $M$ de \omterm{def:g6:lines:perp}{paralelo} a $(BC)$
corta $[AC]$ en $N$.
\begin{enumerate}
  \item Calcula $AN$ y $MN$.
  \item ¿Qué es el \omterm{def:g9:thales:scaling}{factor de escala} de $ABC$ a $AMN$? comparar el
        \omterm{def:g6:measure:perimeter}{perímetros} de los dos triángulos.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:thales:9}
Sea $ABCD$ un trapezoide con $(AB) \parallel (CD)$, $AB = 4$ y
$CD = 6$, cuyas diagonales $[AC]$ y $[BD]$ se encuentran en $O$.
Utilizando Thales en la configuración mariposa alrededor de $O$, calcule el
relación $\dfrac{OA}{OC}$, y demuestre que $O$ corta ambas diagonales en la
misma proporción.
\end{exercise}

\section{Problema: La regla sin marcar, la cámara estenopeica y el
tres medios de un trapezoide}

\begin{problem}[{Problema de fin de semana --- Tales en el trabajo: dividir cualquier
segmento en partes iguales, midiendo el Sol con una caja de zapatos y un
trapezoide donde se encuentran los tres medios famosos}]
\label{pb:g9:thales:1}
El teorema de Tales es la matemática de \emph{rayos}: rayos de sol,
rayos de luz a través de un orificio, rayos de lápiz de un \omterm{def:g5:solids:def}{vértice}. esto
El problema lo utiliza de tres maneras: como herramienta de construcción (dividiendo un
\omterm{def:g6:lines:objects}{segmento} de longitud \emph{cualquier}, \omterm{def:g3:numbers:evenodd}{incluso} $\sqrt2$, en perfectamente iguales
partes), como instrumento de medida (el \omterm{def:g4:geometry:circle}{diámetro} del Sol, con
una caja de zapatos y una moneda), y como lupa en el trapezoide
de \cref{exo:g9:thales:9}, donde las operaciones aritméticas, geométricas y
Los medios armónicos de esta serie aparecen todos en una sola figura.

\medskip
\noindent\textbf{Parte I --- Dividing with an unmarked ruler.}
\begin{enumerate}
  \item Dibujar un \omterm{def:g6:lines:objects}{segmento} $[AB]$ de $7$ cm. Para cortarlo en tres
        partes iguales con compás y regla sin marcar: dibuja cualquier rayo
        desde $A$ (no hasta $B$); bajarse de tres compases iguales
        longitudes en él, dando puntos $P_1$, $P_2$, $P_3$; unirse
        $P_3$ a $B$; dibuje el \omterm{def:g6:lines:perp}{paralelas} al $(P_3 B)$ a través
        $P_1$ y $P_2$. Realizar la construcción y marcar donde
        el corte \omterm{def:g6:lines:perp}{paralelas} $[AB]$.
  \item Justifica con \cref{thm:g9:thales:direct} que los dos
        Los puntos marcados cortan $[AB]$ exactamente en su tercer punto.
  \item Adapte el método para dividir un \omterm{def:g6:lines:objects}{segmento} en $5$ igual
        partes, luego construir $\frac35$ de un \omterm{def:g6:lines:objects}{segmento} determinado.
  \item Construya el punto $M$ de $[AB]$ con
        $\frac{AM}{MB} = \frac23$ (es decir, $M$ en $\frac25$ de
        el camino desde $A$). ¿Cuántos pasos iguales en el rayo, y
        ¿Qué punto se une a $B$?
  \item ¿Por qué esta construcción es mejor que medir con un
        regla graduada? Considere un \omterm{def:g6:lines:objects}{segmento} de longitud $\sqrt2$
        (la diagonal de un cuadrado unitario, irracional por
        \cref{pb:g9:sqrt:1}): qué daría la medición, y
        ¿Qué da Tales?
\end{enumerate}

\medskip

Introduzca un alfiler a través de un \omterm{def:g5:solids:def}{rostro} de un \omterm{def:g5:solids:def}{caja} cerrado: en el lado opuesto
\omterm{def:g5:solids:def}{rostro}, aparece una imagen del mundo al revés. un punto de un
objeto, el orificio y el punto de la imagen están alineados --- luz
viaja recto --- entonces el objeto (altura $H$, a distancia $D$
delante del agujero) y su imagen (altura $h$, en la parte posterior
pared en profundidad $d$ detrás del agujero) sentarse en la mariposa
configuración de Tales.
\begin{enumerate}[resume]
  \item Deducir la fórmula estenopeica
        $h = H \times \frac dD$ de Tales en la mariposa
        alrededor del agujero.
  \item Un árbol de $12$ m se alza $30$ m desde una caja de zapatos de profundidad
        $20$cm. ¿Qué altura tiene su imagen y hacia dónde se eleva?
  \item Apunte el \omterm{def:g5:solids:def}{caja} al Sol: a $d = 1$ m detrás del
        agujero estenopeico, la imagen del Sol es un disco de \omterm{def:g4:geometry:circle}{diámetro} de aproximadamente
        $9$ mm. Dada la distancia del Sol
        $D = 1.5 \times 10^{11}$ m, calcule su \omterm{def:g4:geometry:circle}{diámetro} en
        \omterm{def:g9:fractions:scientific}{notación científica} (\cref{pb:g9:fractions:1}'s
        notación en el trabajo). Comparar con el valor real,
        $1.39 \times 10^9$m.
  \item La Luna: \omterm{def:g4:geometry:circle}{diámetro} $3.5 \times 10^6$ m, distancia
        $3.8 \times 10^8$m. Calcular la razón
        $\frac{\text{diameter}}{\text{distance}}$ para la Luna
        y para el sol. ¿Qué afortunada coincidencia tienen los dos?
        Los números revelan --- ¿y qué evento espectacular lo revela?
        hacer posible?
  \item En una o dos frases: ¿por qué aparece la imagen estenopeica?
        boca abajo, y ¿por qué al agrandar el orificio se hace que el
        ¿Imagen más brillante pero más borrosa?
\end{enumerate}

\noindent\textbf{Parte III --- The trapezoid of the three
means.}

$(AB) \parallel (CD)$, $AB = 4$, $CD = 6$, diagonales que se cruzan en

\begin{enumerate}[resume]
  \item Práctica de mariposa: dos líneas se cruzan en $O$ con
        $(MN) \parallel (PQ)$ en posición mariposa; $OM = 3$,

  \item Explique cómo Tales, aplicado con la proporción $\frac12$,

  \item Dibuja el \omterm{def:g6:lines:perp}{paralelo} a las bases hasta $O$; se encuentra

  \item Por el cálculo del espejo en el triángulo $BDC$, el
        la pieza $OF$ (con $F$ en $[BC]$) tiene la misma longitud.

        \[

        el \omterm{def:g6:lines:perp}{paralelo} a través del cruce diagonal mide el

        La desigualdad entre los tres ha sido demostrada en este libro.

\end{enumerate}
\end{problem}
