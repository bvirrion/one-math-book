\chapter{Sólidos y Volúmenes}\label{ch:g9:solids}

La geometría espacial comienza con una pequeña familia de \omterm{def:g5:solids:def}{sólidos} --- \omterm{def:g7:areas:prism}{prismas},
\omterm{def:g7:areas:prism}{cilindros}, \omterm{def:g8:solids:def}{pirámides}, \omterm{def:g8:solids:def}{conos}, esferas --- y dos preguntas: ¿cuánto
contienen (\omterm{def:g6:measure:volume}{volumen}) y ¿qué ves cuando las cortas (secciones)?
El capítulo cierra con un hecho de gran importancia práctica: ampliar una
\omterm{def:g5:solids:def}{sólido} por $k$ multiplica su \omterm{def:g6:measure:volume}{volumen} por $k^3$.

\section{Los sólidos clásicos y sus volúmenes.}

\begin{theorem}[Fórmulas de volumen]\label{thm:g9:solids:volumes}
Escribe $B$ para el \omterm{def:g6:measure:area}{área} de la base y $h$ para la altura (la distancia
entre la base y el opuesto \omterm{def:g5:solids:def}{rostro} o ápice).
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{l|l}
\omterm{def:g5:solids:def}{sólido} & \omterm{def:g6:measure:volume}{volumen} \\
\hline
\omterm{def:g7:areas:prism}{prisma}, \omterm{def:g7:areas:prism}{cilindro} & $V = B \times h$ \\[5pt]
\omterm{def:g8:solids:def}{pirámide}, \omterm{def:g8:solids:def}{cono} & $V = \dfrac13\, B \times h$ \\[5pt]
sphere of \omterm{def:g3:shapes:shapes}{radio} $r$ & $V = \dfrac43\,\pi r^3$ \\
\end{tabular}
\end{center}
Para un \omterm{def:g7:areas:prism}{cilindro} y un \omterm{def:g8:solids:def}{cono} de \omterm{def:g3:shapes:shapes}{radio} $r$, la base \omterm{def:g6:measure:area}{área} es $B = \pi r^2$;
la superficie de la esfera \omterm{def:g6:measure:area}{área} es $4\pi r^2$.
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:volumes}
Una lata cilíndrica tiene \omterm{def:g3:shapes:shapes}{radio} $4$ cm y altura $10$ cm:
\[
V = \pi r^2 h = \pi \times 16 \times 10 = 160\pi \approx 503 \text{ cm}^3
\approx 0.5 \text{ L}.
\]
Un \omterm{def:g8:solids:def}{cono} con la misma base y altura contiene un tercio de eso:
$\frac{160\pi}{3} \approx 168$ cm$^3$. Una esfera de \omterm{def:g3:shapes:shapes}{radio} $4$ cm:
$V = \frac43 \pi \times 64 = \frac{256\pi}{3} \approx 268$ cm$^3$.
\end{example}

\begin{example}[Una pirámide paso a paso]\label{ex:g9:solids:pyramid}
Un \omterm{def:g8:solids:def}{pirámide} tiene una base cuadrada de lado $6$ m y altura $5$ m.
\begin{enumerate}
  \item Base \omterm{def:g6:measure:area}{área}: $B = 6^2 = 36$ m$^2$.
  \item \omterm{def:g6:measure:volume}{Volumen}: $V = \frac13 \times 36 \times 5 = 60$ m$^3$.
\end{enumerate}
Cuidado con las unidades: si el lado estuviera en cm y la altura en m, uno de
primero habría que convertirlos.
\end{example}

\section{Secciones por planos}

\begin{proposition}[Secciones de los sólidos clásicos]\label{prop:g9:solids:sections}
Cortar un \omterm{def:g5:solids:def}{sólido} con un plano produce una figura plana, su
\emph{sección transversal}\index{sección transversal}:
\begin{itemize}
  \item un corte \omterm{def:g7:areas:prism}{prisma} o \omterm{def:g7:areas:prism}{cilindro} \emph{\omterm{def:g6:lines:perp}{paralelo} to its base} da una copia
        de la base, en cada altura;
  \item un \omterm{def:g8:solids:def}{pirámide} o \omterm{def:g8:solids:def}{cono} cortado \omterm{def:g6:lines:perp}{paralelo} a su base da un
        \emph{reducción} de la base: a la distancia $d$ del ápice,
        el \omterm{def:g9:thales:scaling}{factor de escala} es $k = \frac{d}{h}$;
  \item una esfera de \omterm{def:g3:shapes:shapes}{radio} $r$ cortada por un plano a la distancia $d < r$ de
        el centro da un círculo de \omterm{def:g3:shapes:shapes}{radio} $\sqrt{r^2 - d^2}$ (por el
        Teorema de Pitágoras).
\end{itemize}
\end{proposition}
\admitted

\begin{example}\label{ex:g9:solids:section}
Un \omterm{def:g8:solids:def}{cono} tiene base \omterm{def:g3:shapes:shapes}{radio} $6$ y altura $9$. La sección por un avión.
\omterm{def:g6:lines:perp}{paralelo} a la base, a la distancia $3$ del ápice, hay un disco escalado por
$k = \frac39 = \frac13$: su \omterm{def:g3:shapes:shapes}{radio} es $6 \times \frac13 = 2$.

Una esfera de \omterm{def:g3:shapes:shapes}{radio} $5$ es cortada por un plano a la distancia $3$ de su

\section{Sólidos escalados}

\begin{theorem}[Efecto de un escalamiento sobre longitudes, áreas, volúmenes.]\label{thm:g9:solids:scaling}
Cuando un \omterm{def:g5:solids:def}{sólido} se amplía o reduce con el \omterm{def:g9:thales:scaling}{factor de escala} $k > 0$:
\begin{itemize}
  \item todas las longitudes se multiplican por $k$;
  \item todos los \omterm{def:g6:measure:area}{áreas} se multiplican por $k^2$;
  \item todos los \omterm{def:g6:measure:volume}{volúmenes} se multiplican por $k^3$.
\end{itemize}
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:scaling}
Un modelo de coche en \omterm{def:g7:prop:scale}{escala} $\frac{1}{18}$: las longitudes son las del coche real
multiplicado por $k = \frac{1}{18}$, superficie pintada multiplicada por
$k^2 = \frac{1}{324}$ y \omterm{def:g6:measure:volume}{volumen} por $k^3 = \frac{1}{5832}$.

Duplicar el \omterm{def:g3:shapes:shapes}{radio} de una esfera ($k = 2$) multiplica su \omterm{def:g6:measure:volume}{volumen} por
$2^3 = 8$: verifique la fórmula,
$\frac43\pi (2r)^3 = \frac43 \pi \times 8r^3$.
\end{example}

\begin{example}[cono truncado]\label{ex:g9:solids:truncated}
A \omterm{def:g8:solids:def}{cono} de altura $9$ y base \omterm{def:g3:shapes:shapes}{radio} $6$ (\omterm{def:g6:measure:volume}{volumen}
$\frac13 \pi \times 36 \times 9 = 108\pi$) se corta a la distancia $3$ de
el ápice y se elimina el pequeño \omterm{def:g8:solids:def}{cono} encima del corte. El pequeño \omterm{def:g8:solids:def}{cono} es
la reducción en $k = \frac13$, por lo que su \omterm{def:g6:measure:volume}{volumen} es
$108\pi \times \left(\frac13\right)^3 = 4\pi$. El restante \omterm{def:g5:solids:def}{sólido} (un
\emph{truncated \omterm{def:g8:solids:def}{cono}}) tiene \omterm{def:g6:measure:volume}{volumen} $108\pi - 4\pi = 104\pi \approx 327$.
\end{example}

\section{Problema: la lápida de Arquímedes}

\begin{problem}[{Problema de fin de semana --- la esfera está a dos tercios de
su cilindro (dos veces), la pila 1:2:3 y el cubo cuadrado
ley que prohíbe los gigantes}]
\label{pb:g9:solids:1}
Arquímedes demostró muchos teoremas, pero uno lo enorgulleció tanto que
pidió que le grabaran en su tumba su \emph{cifra}: una esfera
anidado en su \omterm{def:g7:areas:prism}{cilindro} más apretado. Un siglo después el escritor romano
Cicerón, mientras buscaba entre las zarzas cerca de Siracusa, reconoció el
tumba ``por la esfera y el \omterm{def:g7:areas:prism}{cilindro}''. Este problema recupera
lo que codifica la talla --- un doble milagro de dos tercios --- entonces
sigue \omterm{def:g6:measure:volume}{volúmenes} y superficies (\cref{thm:g9:solids:volumes},
\cref{thm:g9:solids:scaling}) a una ley que rige a los gigantes, las hormigas
y planetas enfriados.
