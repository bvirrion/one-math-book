\section{Resumiendo datos}

\begin{definition}[Media, mediana, rango]\label{def:g9:statproba:indicators}
Para una serie de números $N$:
\begin{itemize}
  \item el \emph{significar}\index{significar} es el \omterm{def:g1:addition:def}{suma} de todos los valores divididos
        por $N$;
  \item el \emph{mediana}\index{mediana} es un valor que divide lo ordenado
        serie en dos mitades de igual tamaño: para $N$ \omterm{def:g3:numbers:evenodd}{extraño}, la mitad
        valor; para $N$ \omterm{def:g3:numbers:evenodd}{incluso}, el punto medio de los dos valores medios;
  \item el \emph{rango}\index{rango} es el valor más grande menos el
        más pequeño.
\end{itemize}
\end{definition}

\begin{example}\label{ex:g9:statproba:indicators}
Notas de un alumno: $8,\ 12,\ 9,\ 15,\ 11$.
\begin{enumerate}
  \item \omterm{def:g9:statproba:indicators}{Significar}: $\dfrac{8 + 12 + 9 + 15 + 11}{5} = \dfrac{55}{5} = 11$.
  \item \omterm{def:g9:statproba:indicators}{Mediana}: ordenar la serie: $8, 9, 11, 12, 15$; el medio (3er)
        El valor es $11$.
  \item \omterm{def:g9:statproba:indicators}{Rango}: $15 - 8 = 7$.
\end{enumerate}
Con una sexta marca de $17$: ordenado $8, 9, 11, 12, 15, 17$, el \omterm{def:g9:statproba:indicators}{mediana}
se convierte en el punto medio de $11$ y $12$, es decir, $11.5$, y \omterm{def:g9:statproba:indicators}{significar}
se convierte en $\frac{72}{6} = 12$.
\end{example}

\begin{example}[Media ponderada]\label{ex:g9:statproba:weighted}
Las tallas de zapatos vendidas en un día:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccccc}
size & $38$ & $39$ & $40$ & $41$ & $42$ \\
\hline
count & $3$ & $7$ & $6$ & $3$ & $1$ \\
\end{tabular}
\end{center}
El tamaño \omterm{def:g9:statproba:indicators}{significar} es
\[
\frac{3 \times 38 + 7 \times 39 + 6 \times 40 + 3 \times 41 + 1 \times 42}
{3 + 7 + 6 + 3 + 1}
= \frac{114 + 273 + 240 + 123 + 42}{20}
= \frac{792}{20} = 39.6 .
\]
\end{example}

\begin{remark}
El \omterm{def:g9:statproba:indicators}{significar} y el \omterm{def:g9:statproba:indicators}{mediana} pueden diferir mucho. en la serie
$10, 10, 10, 10, 60$ el \omterm{def:g9:statproba:indicators}{significar} es $20$ (levantado por el valor $60$) mientras
el \omterm{def:g9:statproba:indicators}{mediana} es $10$. Pregunte siempre qué indicador representa la situación.
mejor.
\end{remark}

\section{Probabilidad}

\begin{definition}[Probabilidad]\label{def:g9:statproba:probability}
Un experimento aleatorio tiene varios \emph{resultados} posibles; un
\emph{evento} es un conjunto de resultados. Cada resultado obtiene un
\emph{probabilidad}\index{probabilidad}: un número entre $0$ y $1$,
con todas las probabilidades de que los resultados sumen $1$; el
La probabilidad $\P(A)$ de un evento $A$ es la \omterm{def:g1:addition:def}{suma} de las probabilidades de
sus resultados. Cuando todos los resultados $n$ son igualmente probables,
\[
\P(A) = \frac{\text{number of favorable outcomes}}{n}.
\]
\end{definition}

\begin{example}\label{ex:g9:statproba:spinner}
Una rueda se divide en $8$ sectores iguales: $3$ rojo, $3$ azul, $2$ verde.
Cada sector tiene \omterm{def:g9:statproba:probability}{probabilidad} $\frac18$, por lo que
\[
\P(\text{red}) = \frac38, \qquad
\P(\text{green}) = \frac28 = \frac14, \qquad
\P(\text{not green}) = 1 - \frac14 = \frac34 .
\]
\end{example}

\begin{proposition}[complemento]\label{prop:g9:statproba:complement}
Para cada evento $A$, el \emph{evento contrario} $\bar A$ (``$A$ no
ocurrir'') satisface
\[
\P(\bar A) = 1 - \P(A).
\]
\end{proposition}

\begin{proof}
Cada resultado está exactamente en uno de $A$ y $\bar A$, por lo que
$\P(A) + \P(\bar A)$ \omterm{def:g1:addition:def}{sumas} las probabilidades de todos los resultados, que es
$1$.
\end{proof}

\section{Experimentos de dos pasos}

\begin{method}[diagramas de arbol]\label{met:g9:statproba:tree}
Para un experimento en dos pasos:
\begin{enumerate}
  \item dibuja una rama por resultado del primer paso, luego continúa
        cada rama con los resultados del segundo paso, escribiendo cada
        \omterm{def:g9:statproba:probability}{probabilidad} en su rama;
  \item multiplica las probabilidades a lo largo de un camino para obtener el \omterm{def:g9:statproba:probability}{probabilidad}
        de ese camino;
  \item suma las probabilidades de que todos los caminos formen un evento.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:statproba:tree}
Una bolsa contiene fichas rojas $2$ y negras $3$. Saca una ficha, observa su
color, \emph{ponlo de nuevo}, dibuja de nuevo. Cada sorteo da rojo con
\omterm{def:g9:statproba:probability}{probabilidad} $\frac25$, negro con \omterm{def:g9:statproba:probability}{probabilidad} $\frac35$.

$\frac{4}{25} + \frac{9}{25} = \frac{13}{25}$. \omterm{def:g9:statproba:probability}{Probabilidad} de al menos
uno rojo: $1 - \P(\text{BB}) = 1 - \frac{9}{25} = \frac{16}{25}$.
\end{example}

\begin{remark}[Probabilidades de aproximación de frecuencias]
Lanzar un dado justo $6000$ veces da \emph{acerca de} $1000$ seises --- no
exactamente. A medida que aumenta el número de repeticiones, las frecuencias observadas aumentan.
Cada vez más cerca de las probabilidades: esto es lo que hace que \omterm{def:g9:statproba:probability}{probabilidad}
la herramienta adecuada para modelar experimentos repetidos (una historia desarrollada en
el volumen de High School de esta serie).
\end{remark}

\section{El problema: las ruinosas apuestas del Chevalier}

\begin{problem}[{Problema del fin de semana --- la disputa de juego de 1654
que creó la teoría de la probabilidad y la coincidencia del cumpleaños
eso engaña a todas las aulas}]
\label{pb:g9:statproba:1}
En 1654, un noble francés aficionado al juego, el Chevalier de
Méré, se hizo rico con una apuesta de dados y empezó a perder con otra
que él creía equivalente. Desconcertado, le preguntó al matemático.
Blaise Pascal; Pascal escribió a Pierre de Fermat; y su
El intercambio de cartas fundó la teoría de \omterm{def:g9:statproba:probability}{probabilidad}. esto
El problema reproduce todo el asunto con las herramientas de este capítulo.
--- resultados igualmente probables, árboles y la regla del complemento
(\cref{prop:g9:statproba:complement}) --- y termina con lo más
coincidencia contraintuitiva de todos ellos.

\medskip
\noindent\textbf{Parte I --- Dice, honestly counted.}
\begin{enumerate}
  \item Tira un dado justo dos veces. Usando un árbol (o un $6 \times 6$
        tabla), calcule el \omterm{def:g9:statproba:probability}{probabilidad} de \emph{no} seis en el
        dos tiradas, y deducir el \omterm{def:g9:statproba:probability}{probabilidad} de al menos una
        seis.
  \item Un compañero argumenta: ``una probabilidad entre seis por tirada, dos veces,
        entonces $\frac16 + \frac16 = \frac13$." Comparar con la pregunta
        1 y señale los resultados exactos que cuenta su suma
        dos veces.
  \item Tirar dos dados y sumar: calcular el \omterm{def:g9:statproba:probability}{probabilidad} de
        un \omterm{def:g1:addition:def}{suma} de $7$ y de un \omterm{def:g1:addition:def}{suma} de $2$. ¿Por qué $7$ es el
        ¿El favorito de los jugadores?
  \item Antigua disputa: con dos dados, es un \omterm{def:g1:addition:def}{suma} de $9$ o un \omterm{def:g1:addition:def}{suma} de
        $10$ ¿Es más probable? Cuente los resultados y resuelva.
  \item Generalice la pregunta 1: explique por qué el \omterm{def:g9:statproba:probability}{probabilidad} de al
        al menos un seis en los rollos $n$ es
        \[
        1 - \left(\frac56\right)^{\!n} ,
        \]
        las seis ramas del árbol multiplicándose desde el nivel
        para nivelar.
\end{enumerate}

  \item Segunda apuesta, considerada equivalente por \omterm{def:g9:linfunc:linear}{proporcionalidad}
