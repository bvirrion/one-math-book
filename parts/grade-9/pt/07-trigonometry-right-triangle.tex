\chapter{Trigonometria no Triângulo Retângulo}\label{ch:g9:trig}

Em um triângulo retângulo, uma vez que você conhece um ângulo agudo, o \emph{forma} de
o triângulo é fixo --- apenas seu tamanho pode variar. As proporções de seus lados
dependem, portanto, apenas do ângulo: essas relações são as \omterm{def:g9:trig:ratios}{cosseno}, \omterm{def:g9:trig:ratios}{seno} e
\omterm{def:g9:trig:ratios}{tangente}. Juntamente com o teorema de Pitágoras, eles permitem calcular
todos os lados e todos os ângulos de um triângulo retângulo desde muito pouco
informação.

\section{O teorema de Pitágoras, novamente}

\begin{theorem}[Pitágoras]\label{thm:g9:trig:pythagoras}
Em um triângulo $ABC $ \omterm{def:g6:shapes:triangles}{em ângulo reto} no $ A $, o quadrado da hipotenusa é
o \omterm{def:g1:addition:def}{soma} dos quadrados dos dois catetos:
\[
BC^2 = AB^2 + AC^2 .
\]
Por outro lado, se $ BC^2 = AB^2 + AC^2$ em um triângulo, então o triângulo é
\omterm{def:g6:shapes:triangles}{em ângulo reto} no $ A $.
\end{theorem}
\admitted

\begin{example}\label{ex:g9:trig:pythagoras}
UM triângulo retângulo tem pernas $ AB = 5$ e $ AC = 12$. Então
\[
BC^2 = 5^2 + 12^2 = 25 + 144 = 169,
\qquad\text{so}\quad BC = \sqrt{169} = 13 .
\]
Para encontrar um \emph{perna}: se $ BC = 10$ e $ AB = 6$, então
$ AC^2 = BC^2 - AB^2 = 100 - 36 = 64$, então $ AC = 8$.
\end{example}

\section{Cosseno, seno, tangente}

Em um triângulo retângulo, em relação a um ângulo agudo $\theta $, os três lados
têm nomes: o \emph{hipotenusa} (em frente ao \omterm{def:g3:shapes:rightangle}{ângulo reto}, o mais longo
lado), o lado \emph{oposto} para $\theta $, e o lado
\emph{adjacente} para $\theta $ (a perna tocando $\theta $).

\begin{definition}[Razões trigonométricas]\label{def:g9:trig:ratios}
Para um ângulo agudo $\theta $ de um triângulo retângulo:
\[
\cos\theta = \frac{\text{adjacent}}{\text{hypotenuse}},
\qquad
\sin\theta = \frac{\text{opposite}}{\text{hypotenuse}},
\qquad
\tan\theta = \frac{\text{opposite}}{\text{adjacent}} .
\]
Essas proporções\index{cosseno}\index{seno}\index{tangente} depender apenas de
$\theta $, não no tamanho do triângulo (todos triângulos retângulos com o
mesmo ângulo agudo são escalas um do outro, de Tales).
\end{definition}

\begin{proposition}[Primeiras propriedades]\label{prop:g9:trig:properties}
Para cada ângulo agudo $\theta $:
\begin{enumerate}
  \item $0 < \cos\theta < 1$ e $0 < \sin\theta < 1$ (uma perna é mais curta
        do que a hipotenusa);
  \item $\cos^2\theta + \sin^2\theta = 1$;
  \item $\tan\theta = \dfrac{\sin\theta}{\cos\theta}$.
\end{enumerate}
\end{proposition}

\begin{proof}
Escrever $ a $ para o lado adjacente, $ o $ para o lado oposto, $ h $ para o
hipotenusa de um triângulo retângulo com ângulo $\theta $.

1. $0 < a < h $ e $0 < o < h $, então as duas proporções estão estritamente entre
$0$ e $1$.

2. Por Pitágoras, $ a^2 + o^2 = h^2$. Divida tudo por $ h^2$:
\[
\cos^2\theta + \sin^2\theta
= \frac{a^2}{h^2} + \frac{o^2}{h^2}
= \frac{a^2 + o^2}{h^2} = 1 .
\]

3. $\dfrac{\sin\theta}{\cos\theta} = \dfrac{o/h}{a/h} = \dfrac oa
= \tan\theta $.
\end{proof}

\begin{method}[Encontrando um lado]\label{met:g9:trig:side}
Para calcular um lado desconhecido quando um lado e um ângulo agudo são conhecidos:
\begin{enumerate}
  \item marque o ângulo conhecido e rotule os três lados (hipotenusa,
        oposto, adjacente) \emph{desse ângulo};
  \item escolha a razão (cos, sin ou tan) que envolve o lado conhecido
        e o lado procurado;
  \item escreva o \omterm{def:g8:equations:def}{equação} e resolvê-lo;
  \item verificação de sanidade: a hipotenusa deve sair por mais tempo.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:trig:side}
Em um triângulo retângulo, as medidas da hipotenusa $8$ e um ângulo é
$35^\circ $; calcule o lado oposto a ele. A proporção que envolve
lado oposto e a hipotenusa é o \omterm{def:g9:trig:ratios}{seno}:
\[
\sin 35^\circ = \frac{\text{opposite}}{8}
\quad\Longrightarrow\quad
\text{opposite} = 8 \sin 35^\circ \approx 8 \times 0.574 \approx 4.6 .
\]
Para o lado adjacente, use o \omterm{def:g9:trig:ratios}{cosseno}:
$8\cos 35^\circ \approx 8 \times 0.819 \approx 6.6$. Verifique com
Pitágoras: $4.6^2 + 6.6^2 \approx 21 + 43 \approx 64 = 8^2$.
\end{example}

\begin{method}[Encontrando um ângulo]\label{met:g9:trig:angle}
Quando dois lados são conhecidos, calcule a razão que eles formam, identifique-a como
\omterm{def:g9:trig:ratios}{cosseno}, \omterm{def:g9:trig:ratios}{seno} ou \omterm{def:g9:trig:ratios}{tangente} do ângulo desconhecido e aplique o cálculo da calculadora
\omterm{def:g8:fractions:inverse}{inverso} função ($\cos^{-1}$, $\sin^{-1}$ ou $\tan^{-1}$) à proporção.
\end{method}

\begin{example}\label{ex:g9:trig:angle}
Uma escada $5$ m longo encostado em uma parede, seu pé $1.4$ estou do
parede. O ângulo $\theta $ entre a escada e o solo satisfaz
\[
\cos\theta = \frac{1.4}{5} = 0.28,
\qquad\text{so}\quad
\theta = \cos^{-1}(0.28) \approx 73.7^\circ .
\]
\end{example}

\begin{example}[Ângulos especiais]\label{ex:g9:trig:special}
Dois triângulos fornecem valores exatos. \omterm{def:g3:division:half}{Metade} um quadrado de lado 1 (à direita
triângulo \omterm{def:g6:shapes:triangles}{isósceles}) tem hipotenusa $\sqrt2$ e ângulos de $45^\circ $:
\[
\cos 45^\circ = \sin 45^\circ = \frac{1}{\sqrt2} = \frac{\sqrt2}{2},
\qquad \tan 45^\circ = 1 .
\]
\omterm{def:g3:division:half}{Metade} um triângulo \omterm{def:g6:shapes:triangles}{equilátero} do lado 1 fornece os ângulos $30^\circ $ e
$60^\circ $:
\[
\cos 60^\circ = \sin 30^\circ = \frac12,
\qquad
\cos 30^\circ = \sin 60^\circ = \frac{\sqrt3}{2}.
\]
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g9:trig:1}
UM triângulo retângulo tem pernas $9$ e $12$. Calcule sua hipotenusa. Outro
tem hipotenusa $17$ e uma perna $8$: calcule a outra perna.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:trig:2}
Um triângulo tem lados $7$, $24$ e $25$. É isso \omterm{def:g6:shapes:triangles}{em ângulo reto}? Igual
pergunta para os lados $5$, $6$ e $8$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:trig:3}
Em um triângulo $ DEF $ \omterm{def:g6:shapes:triangles}{em ângulo reto} no $ D $, com o ângulo em $ E $ observado
$\theta $: de que lado está a hipotenusa? Qual lado é oposto
$\theta $? Escrever $\cos\theta $, $\sin\theta $ e $\tan\theta $ como proporções de
os lados $ DE $, $ DF $, $ EF $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:trig:4}
Em um triângulo retângulo, as medidas da hipotenusa $10$ e um ângulo agudo é
$28^\circ $. Calcule as duas pernas (redondo ao décimo;
$\cos 28^\circ \approx 0.883$, $\sin 28^\circ \approx 0.469$).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:trig:5}
Em um triângulo retângulo, a perna adjacente ao ângulo $\theta $ medidas $6$
e a perna oposta mede $4.5$. Calcular $\tan\theta $, então $\theta $
(redondo até o grau; $\tan^{-1}(0.75) \approx 36.9^\circ $).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:trig:6}
Um drone voa a uma altitude de $120$ m. De um observador no terreno,
é visto em um ângulo de elevação de $32^\circ $. Em que horizontal
distância do observador é o drone
($\tan 32^\circ \approx 0.625$)?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:trig:7}
Uma rampa de acesso deve subir $1.2$ m com um ângulo de no máximo $5^\circ $ para
a horizontal. Qual o comprimento mínimo ao longo do \omterm{prop:g9:linfunc:affinegraph}{declive} a rampa deve ter
($\sin 5^\circ \approx 0.0872$)? \omterm{def:g4:numbers:round}{Arredonde} para o décimo mais próximo
metro.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:trig:8}
Usando \cref{prop:g9:trig:properties}: um ângulo agudo $\theta $ satisfaz
$\cos\theta = 0.6$. Calcular $\sin\theta $ sem encontrar $\theta $, então
$\tan\theta $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g9:trig:9}
Justifique os valores exatos de \cref{ex:g9:trig:special}: à direita
triângulo \omterm{def:g6:shapes:triangles}{isósceles} com pernas $1$, calcule a hipotenusa; no
triângulo \omterm{def:g6:shapes:triangles}{equilátero} de lado $1$, calcule a altura e deduza a
\omterm{def:g9:trig:ratios}{cosseno} e \omterm{def:g9:trig:ratios}{seno} de $30^\circ $ e $60^\circ $.
\end{exercise}

\section{Problema: Medir o que você não pode alcançar}

\begin{problem}[{Problema de fim de semana --- as duas estações do topógrafo
método para alturas inacessíveis, inclinações da estrada, a distância até o
horizonte e a corrida da tangente ao infinito}]
\label{pb:g9:trig:1}
Nenhuma fita métrica chega ao topo de uma montanha, ao cume de um
torre através de um rio, ou o horizonte no mar --- mas um ângulo
medidor e as três proporções deste capítulo sim. A peça central deste
problema é o clássico dos topógrafos \emph{método de duas estações},
que calcula uma altura sem nunca se aproximar da sua \omterm{def:g8:powers:def}{base};
em torno dele: sinais de trânsito, escadas, escadas, a curvatura do
Terra, e um primeiro vislumbre de uma função explodindo até o infinito.

\medskip
\noindent\textbf{Part I --- One station.}
(Valores: $\tan 35^\circ \approx 0.700$,
$\sin 35^\circ \approx 0.574$, $\cos 35^\circ \approx 0.819$,
$\tan 1^\circ \approx 0.0175$.)
\begin{enumerate}
  \item De um ponto $60$ m da \omterm{def:g8:powers:def}{base} de um farol, o
        o topo é visto a uma altitude de $35^\circ $ (medido a partir
        horizontal, ao nível dos olhos). Quão alto é o topo acima
        nível dos olhos?
  \item Uma placa de trânsito anuncia $12\,\%$ grau: a estrada sobe
        $12$ m por $100$ m de percurso horizontal. Que ângulo faz
        a estrada faz com a horizontal? E que nota
        porcentagem seria um $45^\circ $ estrada tem?
  \item Os dois ângulos agudos de um triângulo retângulo são
        complementares, e o lado oposto de cada um é o
        lado adjacente do outro. Deduza as identidades ``co'':
        \[
        \sin(90^\circ - \theta) = \cos\theta,
        \qquad
        \cos(90^\circ - \theta) = \sin\theta .
        \]
  \item Um $8$ m escada se inclina para $60^\circ $ para o chão.
        Usando os valores exatos de \cref{ex:g9:trig:special},
        dê a altura alcançada e a distância do pé
        da parede --- exatamente, depois ao centímetro.
  \item Verificar $\cos^2\theta + \sin^2\theta = 1$ numericamente para
        $\theta = 35^\circ $ e exatamente para $\theta = 60^\circ $.
        Por que essa identidade
        (\cref{prop:g9:trig:properties}) um bom hábito para
        verificando o funcionamento da calculadora?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Two stations.}
O cume de uma montanha é visível, a sua \omterm{def:g8:powers:def}{base} inacessível. De um
ponto $ A $, a elevação do cume é $30^\circ $; andando $200$ eu
direto para a montanha, para $ B $, a elevação torna-se
$40^\circ $. Escrever $ h $ para a altura do cume acima do nível dos olhos
e $ x $ para a distância horizontal de $ B $ para a vertical de
o cume. ($\tan 30^\circ \approx 0.577$,
$\tan 40^\circ \approx 0.839$.)
\begin{enumerate}[resume]
  \item Expressar $\tan 40^\circ $ e $\tan 30^\circ $ usando $ h $,
        $ x $ e $ x + 200$: dois \omterm{def:g8:equations:def}{equações} para duas incógnitas.
  \item De cada \omterm{def:g8:equations:def}{equação} expressar a distância horizontal em
        termos de $ h $, subtraia e derive o valor de duas estações
        fórmula:
        \[
        h = \frac{200}
        {\ \dfrac{1}{\tan 30^\circ} - \dfrac{1}{\tan 40^\circ}\ } .
        \]
        Calcular $ h $ para os dez metros mais próximos.
  \item Calcular $ x $ também, e verifique a sanidade de suas duas respostas
        contra o $30^\circ $ avistamento de $ A $.
  \item Em uma frase: por que o método de duas estações
        indispensável --- que medição única, impossível
        aqui, o método de estação única da questão 1 teria
        necessário?
  \item Uma torre é avistada em $30^\circ $, e em $60^\circ $
        depois de caminhar $200$ estou em direção a ele. Usando
        $\tan 30^\circ = \frac{1}{\sqrt3}$ e
        $\tan 60^\circ = \sqrt3$, calcule a altura
        \emph{exatamente} --- a resposta é uma \omterm{def:g9:arith:divisor}{múltiplo} de
        $\sqrt3$ --- então para o medidor.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- To the horizon and up the wall.}
\begin{enumerate}[resume]
  \item A que distância está o horizonte? De pé com os olhos $ h $ metros
        acima de uma Terra esférica de \omterm{def:g3:shapes:shapes}{raio} $ R $, sua \omterm{def:g6:lines:objects}{linha} de
        visão roça a esfera: a \omterm{def:g6:lines:objects}{linha} de visão, o \omterm{def:g3:shapes:shapes}{raio} para
        o ponto de pastagem e o \omterm{def:g3:shapes:shapes}{raio} aos seus pés formam um
        triângulo retângulo. Show com Pitágoras
        (\cref{thm:g9:trig:pythagoras}) que a distância $ d $ para
        o horizonte satisfaz $ d^2 = 2Rh + h^2 \approx 2Rh $,
        e calcular $ d $ para olhos em $1.80$ eu
        ($ R = 6.37 \times 10^6$ m).
  \item A mesma pergunta do topo de uma $324$ m torre. (Marinheiros
        regra prática: o horizonte em quilômetros é cerca de
        $3.6\sqrt{h}$ com $ h $ em metros --- verifique seus dois
        responde contra isso.)
  \item Seu polegar, cerca $2$ cm de largura, segurado com o braço estendido,
        sobre $60$ cm do seu olho: que ângulo cobre?
        A Lua Cheia cobre cerca de $0.5^\circ $: quanto do seu
        polegar esconde a Lua inteira?
  \item Escadas confortáveis têm degraus de $17$ cm e passos de
        $29$ cm. Que ângulo eles sobem? Compare com o
        $30^\circ $ de \cref{ex:g9:trig:special}, e com um
        escada íngreme do sótão em $60^\circ $: qual riser seria
        \emph{isto} preciso em um $29$ centímetros de piso?
  \item O tangentecorrida: computar (calculadora)
        $\tan 10^\circ $, $\tan 45^\circ $, $\tan 80^\circ $,
        $\tan 89^\circ $. O que acontece quando o ângulo se aproxima
        $90^\circ$ e por que (pense no lado adjacente)? A parede é
        vertical, a proporção não tem valor e o gráfico dispara
        ao infinito --- seu primeiro \emph{assíntota}, uma criatura
        estudado de perto no \omterm{def:g6:measure:volume}{volume} do Ensino Médio.
\end{enumerate}
\end{problem}
