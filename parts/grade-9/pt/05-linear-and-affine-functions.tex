\chapter{Funções Lineares e Afins}\label{ch:g9:linfunc}

“Três quilogramas custam três vezes mais que um”: isto é
\omterm{def:g7:prop:table}{proporcionalidade}, a maneira mais simples de relacionar duas quantidades. linear
funções são \omterm{def:g7:prop:table}{proporcionalidade} na linguagem das funções; afim
funções adicione um valor inicial fixo. Seus gráficos são \omterm{def:g6:lines:objects}{linhas} retas,
e ler essas \omterm{def:g6:lines:objects}{linhas} é a habilidade que este capítulo desenvolve.

\section{Proporcionalidade e funções lineares}

\begin{definition}[Função linear]\label{def:g9:linfunc:linear}
UM \emph{função linear}\index{função linear} multiplica cada número
por um número fixo $a $:
\[
f(x) = a x .
\]
O número $ a $ é o \emph{coeficiente} de $ f $. Duas quantidades são
\emph{proporcionalmente}\index{proporcionalmente} exatamente quando alguém é linear
função do outro.
\end{definition}

\begin{example}\label{ex:g9:linfunc:linear}
Se as maçãs custarem $3$ por quilograma, o preço $ x $ quilogramas é
$ f(x) = 3x $: comprar o dobro custa o dobro, e o preço do
$2.5$ kg é $ f(2.5) = 7.5$.

Inversamente, se um \omterm{def:g9:linfunc:linear}{função linear} satisfaz $ f(4) = 10$, então
$ a = \frac{10}{4} = 2.5$ e $ f(x) = 2.5x $: \emph{um valor determina um
\omterm{def:g9:linfunc:linear}{função linear} completamente.}
\end{example}

\begin{proposition}[Gráfico de uma função linear]\label{prop:g9:linfunc:lineargraph}
O gráfico de $ f(x) = ax $ é uma \omterm{def:g6:lines:objects}{linha} reta \emph{através da origem}.
Por outro lado, toda \omterm{def:g6:lines:objects}{linha} não vertical que passa pela origem é o gráfico de um
\omterm{def:g9:linfunc:linear}{função linear}.
\end{proposition}
\admitted

\section{Funções afins}

\begin{definition}[Função afim]\label{def:g9:linfunc:affine}
Um \emph{função afim}\index{função afim} tem a forma
\[
f(x) = a x + b ,
\]
onde $ a $ e $ b $ são números fixos. Funções lineares são os especiais
caso $ b = 0$; funções constantes o caso $ a = 0$.
\end{definition}

\begin{example}
Uma academia cobra uma taxa de inscrição de $20$, então $5$ por visita: o total
custo de $ x $ visitas é $ f(x) = 5x + 20$. O custo é \emph{não}
proporcional ao número de visitas (10 visitas não custam o dobro
como 5), mas cada visita extra adiciona a mesma quantia, $5$.
\end{example}

\begin{proposition}[Gráfico de uma função afim]\label{prop:g9:linfunc:affinegraph}
O gráfico de $ f(x) = ax + b $ é uma \omterm{def:g6:lines:objects}{linha} reta:
\begin{itemize}
  \item $ b = f(0)$ é o \emph{$ y $-interceptar}: a \omterm{def:g6:lines:objects}{linha} cruza o
        eixo vertical em $(0, b)$;
  \item $ a $ é a \emph{inclinação}\index{inclinação}: mover uma unidade para o
        movimentos certos $ a $ unidades para cima (para baixo se $ a < 0$); a função é
        aumentando quando $ a > 0$, diminuindo quando $ a < 0$;
  \item para quaisquer duas entradas $ u \neq v $,
        $ a = \dfrac{f(v) - f(u)}{v - u}$.
\end{itemize}
\end{proposition}

\begin{proof}[Prova do último ponto]
$ f(v) - f(u) = (av + b) - (au + b) = a(v - u)$; \omterm{def:g9:arith:divisor}{dividir} por $ v - u $. O
o descanso é admitido neste nível (o \omterm{def:g6:measure:volume}{volume} do Ensino Médio dá uma visão completa
tratamento de \omterm{def:g6:lines:objects}{linhas} e seus \omterm{def:g8:equations:def}{equações}).
\end{proof}

\begin{method}[Encontrando uma função afim de dois valores]\label{met:g9:linfunc:findaffine}
Dado $ f(u)$ e $ f(v)$ com $ u \neq v $:
\begin{enumerate}
  \item calcular o \omterm{prop:g9:linfunc:affinegraph}{declive} $ a = \dfrac{f(v) - f(u)}{v - u}$;
  \item substitua um dos dois valores conhecidos em $ f(x) = ax + b $ para
        encontrar $ b $;
  \item verifique com o outro valor.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:linfunc:findaffine}
Encontre o \omterm{def:g9:linfunc:affine}{função afim} com $ f(2) = 7$ e $ f(5) = 16$.
\[
a = \frac{16 - 7}{5 - 2} = \frac93 = 3 ,
\]
então $ f(2) = 3 \times 2 + b = 7$ dá $ b = 1$: $ f(x) = 3x + 1$. Verificar:
$ f(5) = 15 + 1 = 16$.
\end{example}

\section{Porcentagens}

\begin{proposition}[Alteração percentual]\label{prop:g9:linfunc:percent}
Aumentando uma quantidade em $ t\,\%$ multiplica por
$1 + \dfrac{t}{100}$; diminuindo-o em $ t\,\%$ multiplica por
$1 - \dfrac{t}{100}$. As alterações percentuais são funções lineares.
\end{proposition}

\begin{proof}
Aumentando $ x $ por $ t\,\%$ significa adicionar $\frac{t}{100}\,x $:
\[
x + \frac{t}{100}\,x = \left(1 + \frac{t}{100}\right) x .
\]
O mesmo cálculo com um sinal de menos para uma diminuição.
\end{proof}

\begin{example}\label{ex:g9:linfunc:percent}
Um preço de $80$ aumenta em $15\%$: novo preço
$80 \times 1.15 = 92$. Um preço de $60$ diminui em $30\%$:
$60 \times 0.7 = 42$.

\emph{Encadear mudanças multiplica os fatores.} UM $20\%$ aumentar
seguido por um $20\%$ diminuição dá
$1.2 \times 0.8 = 0.96$: um $4\%$ \emph{diminuir} no geral, não é um retorno
o começo!
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g9:linfunc:1}
Dentre essas funções, quais são lineares? afim? nenhum?
\[
f(x) = 4x, \qquad
g(x) = 3x - 5, \qquad
h(x) = x^2, \qquad
k(x) = -\tfrac x2, \qquad
m(x) = 7 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:linfunc:2}
Deixar $ f(x) = 2x - 3$. Calcular $ f(0)$, $ f(4)$, $ f(-1)$, e encontre o número
$ x $ tal que $ f(x) = 9$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:linfunc:3}
UM \omterm{def:g9:linfunc:linear}{função linear} satisfaz $ f(6) = 15$. Encontre seu coeficiente, então
calcular $ f(10)$ e $ f(-2)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:linfunc:4}
Desenhe no mesmo sistema de coordenadas os gráficos de $ f(x) = 2x $,
$ g(x) = 2x + 3$ e $ h(x) = -x + 4$. O que pode ser dito sobre os gráficos de
$ f $ e $ g $?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:linfunc:5}
Encontre o \omterm{def:g9:linfunc:affine}{função afim} tal que $ f(1) = 5$ e $ f(3) = 11$; então o
um desses que $ g(0) = 4$ e $ g(2) = 0$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:linfunc:6}
Calcular: o preço após um $25\%$ aumentar em $120$; o preço depois de um
$40\%$ desconto em $65$; o preço original se um artigo custar $69$
depois de um $25\%$ desconto.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:linfunc:7}
Um plano telefônico A custa $10$ por mês mais $0.05$ por minuto de chamadas; planejar
Custos B $25$ por mês com chamadas ilimitadas.
\begin{enumerate}
  \item Expresse o custo mensal do plano A em função do número
        $ x $ de minutos.
  \item A partir de quantos minutos por mês o plano B é mais barato?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:linfunc:8}
Uma população de $2000$ bactérias aumenta em $10\%$ a cada hora.
\begin{enumerate}
  \item Quantas bactérias existem depois de uma hora? Depois de duas horas?
  \item Explique por que a resposta depois de duas horas não é $2400$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:linfunc:9}
O gráfico de um \omterm{def:g9:linfunc:affine}{função afim} passa pelos pontos $(2, 3)$ e
$(6, 1)$.
\begin{enumerate}
  \item Calcule seu \omterm{prop:g9:linfunc:affinegraph}{declive}. A função está aumentando ou diminuindo?
  \item Dê sua expressão e calcule a entrada para a qual a saída
        é $0$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g9:linfunc:10}
Uma loja primeiro aumenta um preço em $ t\,\%$, então reduz o novo preço em
$ t\,\%$.
\begin{enumerate}
  \item Mostre que o preço final é igual ao original multiplicado por
        $1 - \dfrac{t^2}{10000}$.
  \item Deduza que o preço final é sempre \emph{mais baixo} do que o
        originais (para $0 < t \leq 100$) e encontre a porcentagem geral
        mudar para $ t = 10$.
\end{enumerate}
\end{exercise}

\section{Problema: Os dois termômetros}

\begin{problem}[{Problema de fim de semana --- Celsius, Fahrenheit e o
funções afins da vida diária: uma temperatura é igual em
ambas as escalas, e a inclinação constante é a assinatura da retidão}]
\label{pb:g9:linfunc:1}
Os americanos ouvem ``$68$ graus '' e vestir-se com roupas leves; Os europeus ouvem
e pegue um casaco. Os dois termômetros do mundo estão ligados
por um \omterm{def:g9:linfunc:affine}{função afim} --- e construí-lo a partir de dois fatos é
exatamente a habilidade de \cref{met:g9:linfunc:findaffine}. Isto
problema constrói a conversão, encontra a temperatura estranha em
qual os dois escalas concordo, então lê táxis, grilos e
velas com o mesmo \omterm{def:g3:numbers:evenodd}{par} de óculos, e termina com o teste
que reconhece a própria retidão.

\medskip
\noindent\textbf{Part I --- Building the conversion.}
Dois fatos fixam o Fahrenheit escala: a água congela em
$32\,^\circ $ F e ferve em $212\,^\circ $ Gordo $0\,^\circ $ C e
$100\,^\circ $ C).
\begin{enumerate}
  \item Encontre o \omterm{def:g9:linfunc:affine}{função afim} $ F = f(C)$ convertendo Celsius
        em Fahrenheit (\cref{met:g9:linfunc:findaffine} com
        os valores $ f(0) = 32$ e $ f(100) = 212$).
  \item Converter para Fahrenheit: um suave $20\,^\circ $ dia C; o
        corpo $37\,^\circ $ C; um congelamento $-10\,^\circ $ C.
  \item Resolva para $ C $ para construir a conversão de retorno e usá-la
        ligado $50\,^\circ $ F e assim por diante $98.6\,^\circ $ F.
  \item É $ f $ um \emph{linear} função
        (\cref{def:g9:linfunc:linear})? Teste os sinais reveladores:
        o que é $ f(0)$, e dobra a leitura Celsius
        o dobro do Fahrenheit? Conclua com a palavra certa
        (\cref{def:g9:linfunc:affine}).
  \item O enigma clássico: existe uma temperatura que diz
        \emph{o mesmo número} em ambos escalas? Resolver
        $ f(x) = x $ e descreva o que sua solução significa no
        gráfico de $ f $ (qual \omterm{def:g6:lines:objects}{linha} o gráfico cruza ali?).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Funções afins in the wild.}
\begin{enumerate}[resume]
  \item Taxas de táxi A $3$ euros mais $1.50$ por km; táxi B
        cobra um apartamento $2$ euros por km. Escreva os dois preços
        funções e compará-las para um $4$ passeio de km e um
        $8$ passeio de km.
  \item Encontre o ponto de equilíbrio distância a que custam os dois táxis
        o mesmo, e descreva a situação em um gráfico (dois
        \omterm{def:g6:lines:objects}{linhas} --- o que acontece no ponto de cruzamento, antes dele,
        depois disso?).
  \item Regra prática dos naturalistas: um grilo canta
        $ N = 7T - 30$ vezes por minuto à temperatura
        $ T\,^\circ $ C. Quantos chilreios em um $20\,^\circ $ C
        noite? Você conta $82$ gorjeia em um minuto: o que
        temperatura o grilo anuncia?
  \item Um telefone comprado $600$ euros perdem valor por $15$ euros por
        mês: escreva a função de valor $ v(t)$, descubra quando o
        telefone vale a pena $240$ euros, e quando o modelo o diz
        não vale nada. A fórmula ainda significa alguma coisa
        depois dessa data?
  \item Os movimentos percentuais são funções lineares: $+25\,\%$ é
        multiplicação por $1.25$, e $-20\,\%$ por $0.8$
        (\cref{prop:g9:linfunc:percent}). Componha-os: o que
        faz um $+25\,\%$ ascensão seguida de $-20\,\%$ cortar fazer para
        um preço? Explique o pequeno milagre e compare com
        \cref{exo:g9:linfunc:10}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The \omterm{prop:g9:linfunc:affinegraph}{declive} is everything.}
\begin{enumerate}[resume]
  \item Uma vela mede $20$ cm quando aceso e $17$ cm \omterm{def:g3:division:half}{metade} um
        hora depois. Supondo fusão afim, encontre $ h(t)$ (altura
        em cm, $ t $ em minutos) e prever quando a vela morrerá.
  \item Dois ciclistas percorrem a mesma estrada: a distância do primeiro é
        $ d_1(t) = 24t $ (km depois $ t $ horas), o segundo é
        $ d_2(t) = 10 + 18t $. Quem começou com vantagem, quem
        anda mais rápido e a que horas e quilômetros o
        primeiro pegar o segundo?
  \item Deixar $ f(x) = 2x + 3$ e $ g(x) = -0.5x + 7$. Calcular
        $ f(x) + g(x)$ e verifique se está afim novamente. Quais são os
        \omterm{prop:g9:linfunc:affinegraph}{declive} e interceptação de um \omterm{def:g1:addition:def}{soma}, em geral?
  \item Discuta completamente o \omterm{def:g8:equations:def}{equação} $ mx + p = 0$: quantos
        soluções quando $ m \neq 0$ (e qual)? Quando $ m = 0$
        e $ p \neq 0$? Quando $ m = 0$ e $ p = 0$? (Cada caso
        tem uma história gráfica: onde fica uma \omterm{def:g6:lines:objects}{linha} de \omterm{prop:g9:linfunc:affinegraph}{declive} $ m $ cruz
        o eixo horizontal?)
  \item O teste de alinhamento: são os pontos $(1, 5)$, $(3, 9)$,
        $(7, 17)$ em uma \omterm{def:g6:lines:objects}{linha} reta? São $(1, 5)$, $(3, 9)$,
        $(6, 16)$? Calcular taxas de variação entre \omterm{def:g3:numbers:evenodd}{pares} e
        decidir. Enuncie a moral: uma \emph{taxa constante de
        mudar} é a assinatura de funções afins --- curvado
        gráficos mudam sua taxa e medindo \emph{que} é um
        história para o \omterm{def:g6:measure:volume}{volume} High School.
\end{enumerate}
\end{problem}
