\chapter{Álgebra e Equações}\label{ch:g9:algebra}

Letras representam números, e a computação com letras prova fatos sobre
\emph{todos} números de uma vez. Este capítulo pratica expandindo e
factoring --- incluindo as três identidades que irão acompanhá-lo através
toda a matemática --- e resolve o primeiro grau \omterm{def:g8:equations:def}{equações}, \omterm{def:g2:mult:def}{produto}
equaçõese desigualdades simples, sempre um passo justificado de cada vez.

\section{Expandindo}

\begin{definition}[Expandindo]\label{def:g9:algebra:expand}
\emph{Expandindo}\index{Expandindo} significa transformar um \omterm{def:g2:mult:def}{produto} em um
\omterm{def:g1:addition:def}{soma}, usando distributividade:
\[
k(a + b) = ka + kb,
\qquad
(a + b)(c + d) = ac + ad + bc + bd .
\]
\end{definition}

\begin{example}\label{ex:g9:algebra:expand}
Cada termo do primeiro colchete multiplica cada termo do segundo:
\begin{align*}
(x + 3)(2x + 5)
&= x \times 2x + x \times 5 + 3 \times 2x + 3 \times 5 \\
&= 2x^2 + 5x + 6x + 15 \\
&= 2x^2 + 11x + 15 .
\end{align*}
Os signos viajam com os termos:
\begin{align*}
(3x - 2)(x - 4)
&= 3x^2 - 12x - 2x + 8
= 3x^2 - 14x + 8 .
\end{align*}
\end{example}

\begin{theorem}[As três identidades]\label{thm:g9:algebra:identities}
Para todos os números $a $ e $ b $:
\[
(a+b)^2 = a^2 + 2ab + b^2,
\qquad
(a-b)^2 = a^2 - 2ab + b^2,
\qquad
(a+b)(a-b) = a^2 - b^2 .
\]
\end{theorem}

\begin{proof}
Expanda cada lado esquerdo. Para o primeiro:
$(a+b)(a+b) = a^2 + ab + ba + b^2 = a^2 + 2ab + b^2$. Para o segundo,
substituir $ b $ por $-b $ no primeiro. Para o terceiro:
$(a+b)(a-b) = a^2 - ab + ba - b^2 = a^2 - b^2$ (os termos cruzados são cancelados).
\end{proof}

\begin{example}\label{ex:g9:algebra:identities}
\[
(x + 5)^2 = x^2 + 10x + 25, \qquad
(3x - 1)^2 = 9x^2 - 6x + 1, \qquad
(2x + 7)(2x - 7) = 4x^2 - 49 .
\]
Aritmética mental: $102^2 = (100 + 2)^2 = 10000 + 400 + 4 = 10404$, e
$98 \times 102 = (100-2)(100+2) = 10000 - 4 = 9996$.
\end{example}

\section{Fatoração}

\begin{method}[Fatoração]\label{met:g9:algebra:factor}
Para fatorar uma expressão, tente na ordem:
\begin{enumerate}
  \item \emph{fator comum}: $ ka + kb = k(a + b)$, onde $ k $ pode em si
        seja um colchete;
  \item \emph{\omterm{ex:g1:subtraction:difference}{diferença} de quadrados}: $ a^2 - b^2 = (a+b)(a-b)$;
  \item \emph{quadrado perfeito}: $ a^2 + 2ab + b^2 = (a+b)^2$ (e com
        $-2ab $, $(a-b)^2$).
\end{enumerate}
Verifique sempre por expandindo o resultado.
\end{method}

\begin{example}\label{ex:g9:algebra:factor}
\emph{Fator comum:}
$15x^2 + 10x = 5x \times 3x + 5x \times 2 = 5x(3x + 2)$.

\emph{Colchete comum:}
\[
(x+2)(x-1) + (x+2)(3x+4) = (x+2)\bigl[(x-1) + (3x+4)\bigr]
= (x+2)(4x+3).
\]

\emph{\omterm{ex:g1:subtraction:difference}{Diferença} de quadrados:}
$ x^2 - 81 = (x+9)(x-9)$, e
\[
(2x+1)^2 - 9 = (2x+1)^2 - 3^2 = (2x+1+3)(2x+1-3) = (2x+4)(2x-2).
\]
Cada fator ainda tem um fator comum: $(2x+4)(2x-2) = 2(x+2) \times
2(x-1) = 4(x+2)(x-1)$.
\end{example}

\section{Equações}

\begin{method}[Equações de primeiro grau]\label{met:g9:algebra:firstdegree}
Para resolver um \omterm{def:g8:equations:def}{equação} como $5x - 3 = 2x + 9$:
\begin{enumerate}
  \item reunir o $ x $-termos de um lado, os números do outro, por
        adicionando a mesma quantidade em ambos os lados;
  \item reduza cada lado;
  \item divida ambos os lados pelo coeficiente de $ x $;
  \item verifique a solução no original \omterm{def:g8:equations:def}{equação}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:algebra:firstdegree}
\begin{align*}
5x - 3 &= 2x + 9 \\
5x - 2x &= 9 + 3 && \text{(add $3 - 2x $ to both sides)}\\
3x &= 12 \\
x &= 4 && \text{(divide by 3).}
\end{align*}
Verificar: $5 \times 4 - 3 = 17$ e $2 \times 4 + 9 = 17$. A solução é
$4$.

Com colchetes, expanda primeiro: $3(x - 2) = x + 8$ torna-se $3x - 6 = x + 8$,
então $2x = 14$ e $ x = 7$.
\end{example}

\begin{theorem}[Regra de produto zero]\label{thm:g9:algebra:zeroproduct}
UM \omterm{def:g2:mult:def}{produto} é \omterm{def:g1:counting:zero}{zero} se e somente se pelo menos um fator for \omterm{def:g1:counting:zero}{zero}:
\[
A \times B = 0
\quad\Longleftrightarrow\quad
A = 0 \ \text{ or } \ B = 0 .
\]
\end{theorem}

\begin{proof}
Se um fator for \omterm{def:g1:counting:zero}{zero} o \omterm{def:g2:mult:def}{produto} claramente é. Inversamente se $ A B = 0$ com
$ A \neq 0$, dividindo ambos os lados por $ A $ dá $ B = 0$.
\end{proof}

\begin{example}\label{ex:g9:algebra:zeroproduct}
Resolver $(x - 3)(2x + 10) = 0$: qualquer $ x - 3 = 0$ ou $2x + 10 = 0$, então o
soluções são $3$ e $-5$.

Resolver $ x^2 - 49 = 0$: fatorar primeiro, $(x-7)(x+7) = 0$, soluções $7$ e
$-7$.

Resolver $ x^2 = 3x $: nunca divida por $ x $! Mover e fatorar:
$ x^2 - 3x = x(x - 3) = 0$, soluções $0$ e $3$.
\end{example}

\section{Desigualdades}

\begin{proposition}[Regras para desigualdades]\label{prop:g9:algebra:ineq}
Uma desigualdade é preservada ao adicionar o mesmo número a ambos os lados, e
ao multiplicar ambos os lados pelo mesmo \emph{positivo} número.
Multiplicando ambos os lados por um \emph{negativo} número inverte o
sinal de desigualdade.
\end{proposition}

\begin{proof}
Se $ a \leq b $ então $ b - a \geq 0$. Adicionando $ c $: $(b+c) - (a+c) = b - a
\geq 0$, então $ a + c \leq b + c $. Multiplicando por $ c > 0$:
$ bc - ac = (b-a)c \geq 0$, então $ ac \leq bc $. Multiplicando por $ c < 0$:
$(b-a)c \leq 0$, então $ ac \geq bc $: a ordem é invertida.
\end{proof}

\begin{example}\label{ex:g9:algebra:ineq}
Resolver $7 - 2x < 15$:
\begin{align*}
7 - 2x &< 15 \\
-2x &< 8 && \text{(subtract 7)} \\
x &> -4 && \text{(divide by $-2$: reverse the sign).}
\end{align*}
As soluções são todos números maiores que $-4$: em uma reta numérica, um
ponto oco em $-4$ e sombreamento à direita.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g9:algebra:1}
Expandir e reduzir:
\[
4(3x - 2), \qquad
(x + 6)(x + 2), \qquad
(2x - 3)(x + 5), \qquad
(x - 1)(x - 9).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:algebra:2}
Expanda usando as identidades:
\[
(x + 8)^2, \qquad
(3x - 4)^2, \qquad
(x + 10)(x - 10), \qquad
(5x + 2)(5x - 2).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:algebra:3}
Calcule mentalmente, usando uma identidade: $101^2$; \quad $59^2$
(escrever $59 = 60 - 1$); \quad $53 \times 47$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:algebra:4}
Fator:
\[
9x + 12, \qquad
x^2 - 64, \qquad
x^2 + 14x + 49, \qquad
18x^2 - 6x .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:algebra:5}
Resolva, escrevendo cada passo:
\[
4x + 7 = 19, \qquad
6x - 5 = 2x + 11, \qquad
5(x - 3) = 3x + 1 .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:algebra:6}
Resolva usando o \omterm{def:g2:mult:def}{produto} \omterm{def:g1:counting:zero}{zero} regra:
\[
(x + 4)(3x - 12) = 0, \qquad
x^2 - 121 = 0, \qquad
(2x - 1)^2 = 0, \qquad
x^2 = 8x .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:algebra:7}
Fator $ E = (3x + 2)^2 - 25$, então resolva $ E = 0$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:algebra:8}
Resolva as inequações e represente as soluções em uma reta numérica:
\[
3x - 4 \geq 8, \qquad
5 - 2x > 1, \qquad
4(x + 1) \leq x - 5 .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:algebra:9}
Eu penso em um número, multiplico por $3$, adicionar $7$, e obtenha o mesmo resultado
como se eu tivesse multiplicado por $5$ e subtraído $9$. Qual é o meu número?
(Configure um \omterm{def:g8:equations:def}{equação} e resolvê-lo.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g9:algebra:10}
Deixar $ n $ seja um número inteiro.
\begin{enumerate}
  \item Expandir $(n + 1)(n - 1)$ e deduza que \omterm{def:g2:mult:def}{produto} dos dois
        vizinhos de $ n $ é sempre $ n^2 - 1$ (por exemplo\ $24 \times 26 =
        25^2 - 1 = 624$).
  \item Mostre que o \omterm{def:g1:addition:def}{soma} de três inteiros consecutivos é sempre um
        \omterm{def:g9:arith:divisor}{múltiplo} de $3$.
\end{enumerate}
\end{exercise}

\section{Problema: A álgebra dos dígitos}

\begin{problem}[{Problema de fim de semana --- por que as regras de divisibilidade
trabalho, o truque 1089 desmascarado e aritmética mental que parece
como feitiçaria}]
\label{pb:g9:algebra:1}
A escola primária te ensinou \emph{regras} (um número é \omterm{def:g6:wholes:divisible}{divisível} por
$9$ quando seu dígito \omterm{def:g1:addition:def}{soma} é); mágicos atuam \emph{truques}
(todo mundo termina em $1089$); os mercados sabem \emph{atalhos}
($35^2 = 1225$, instantaneamente). Tudo isso é o mesmo segredo: um
número com dígitos $ a $, $ b $, $ c $ é a expressão
$100a + 10b + c $, e a álgebra deste capítulo
(\cref{thm:g9:algebra:identities}, expandindo, fatoração) pode levar
isso à parte. Ao final deste problema você terá provado a
regras, desmascarou o truque e aprendeu a feitiçaria.

\medskip
\noindent\textbf{Part I --- Digits under the microscope.}
\begin{enumerate}
  \item Um número de dois algarismos com dezenas $ a $ e dígito das unidades
        $ b $ é $10a + b $. Expanda e fatore o \omterm{ex:g1:subtraction:difference}{diferença}
        entre o número e sua reversão,
        $(10a + b) - (10b + a)$ e indique a descoberta. Verifique
        está ligado $72 - 27$.
  \item Prove a regra dos nove para números de três algarismos: verifique
        a identidade
        \[
        100a + 10b + c = 9\,(11a + b) + (a + b + c),
        \]
        e explique por que mostra que um número e seu algarismo \omterm{def:g1:addition:def}{soma}
        deixe o \emph{mesmo \omterm{def:g3:division:remainder}{restante}} na divisão por $9$
        --- então um é \omterm{def:g6:wholes:divisible}{divisível} por $9$ exatamente quando o outro está
        (\cref{prop:g9:arith:rules}, agora comprovado).
  \item Deduza a regra de três da mesma identidade, em um
        sentença.
  \item ``Expulsar noves'', o cheque dos contadores: para testar
        $47 \times 86 = 4\,042$, substitua cada número pelo
        dígito \omterm{def:g1:addition:def}{soma} do seu dígito \omterm{def:g1:addition:def}{soma}: $47 \to 11 \to 2$,
        $86 \to 14 \to 5$, e $2 \times 5 = 10 \to 1$; então
        $4\,042 \to 10 \to 1$: consistente. Explique por que um correto
        \omterm{def:g2:mult:def}{produto} deve passar no teste (escrever $47 = 9k + 2$,
        $86 = 9l + 5$ e expandir) e encontrar um resultado errado que
        o teste não consegue capturar (por que os dígitos são trocados
        invisível para ele?).
  \item Prove a regra de onze para números de quatro dígitos: verifique
        \[
        1000a + 100b + 10c + d
        = 11\,(91a + 9b + c) + (- a + b - c + d),
        \]
        e usá-lo para decidir se $2\,926$ é \omterm{def:g6:wholes:divisible}{divisível} por
        $11$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The 1089 trick.}
O truque: pense em um número de três dígitos cujo primeiro e último dígito
dígitos diferem em pelo menos $2$; inverta-o; subtraia o menor
do maior; reverter o resultado; adicione os dois últimos números.
O mágico anuncia: \emph{1089}.
\begin{enumerate}[resume]
  \item Execute o truque em $742$ e em um número do seu
        escolha.
  \item Deixe o número ter dígitos $ a $, $ b $, $ c $ com $ a > c $.
        Fatore a primeira subtração,
        $(100a + 10b + c) - (100c + 10b + a)$ e liste todos os
        valores que pode assumir quando $ a - c $ corre de $2$ para $9$.
  \item Cada um desses \omterm{def:g5:division:multiple}{múltiplos} de $99$ tem um dígito marcante
        padrão. Escrita $ m = a - c $, verifique a identidade
        \[
        99m = 100\,(m - 1) + 90 + (10 - m),
        \]
        e leia os três dígitos de $99m $.
  \item Adicionar $99m $ para sua própria reversão, usando a questão 8
        algarismos e prove que o total é $1089$ --- qualquer que seja
        $ m $ é. O truque morreu; viva a álgebra.
  \item Por que o mágico exigiu o primeiro e o último dígito
        diferir em pelo menos $2$? Examine os casos $ a = c $ e
        $ a - c = 1$ (onde a subtração dá $99$, que
        deve ser escrito $099$ para que a reversão funcione), e
        indique as letras pequenas que mantêm o truque honesto.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Sorcery for daily use.}
\begin{enumerate}[resume]
  \item O truque do aniversário: ``multiplique o mês do seu nascimento por $5$,
        adicionar $7$, dobre, adicione seu dia de nascimento, subtraia $14$.''
        Mostre que o resultado é sempre $10m + d $ --- mês e
        dia, legível à primeira vista. (Experimente com alguém.)
  \item Você redesenha o truque como: ``multiplique o mês por $5$,
        adicionar $9$, duplo, adicione o dia.'' Qual número deve o
        espectador agora subtrai no final para o mesmo
        leitura mês-dia?
  \item O truque dos dígitos faltantes: o espectador escolhe qualquer número,
        subtrai seu dígito \omterm{def:g1:addition:def}{soma} (a questão 2 diz que o resultado é
        sempre um \omterm{def:g9:arith:divisor}{múltiplo} de $9$ --- explique por que para um
        número de três dígitos), em seguida, risca um \emph{diferente de \omterm{def:g1:counting:zero}{zero}}
        dígito e lê os dígitos \omterm{def:g3:division:remainder}{restantes}. Explique como
        você recupera o dígito riscado e por que $0$ tinha que ser
        excluído.
  \item Feitiçaria de mercado: prove a identidade
        $(10t + 5)^2 = 100\,t(t+1) + 25$, indique a regra para
        elevar ao quadrado qualquer número terminado em $5$ e calcular $45^2$,
        $85^2$ e $105^2$ na sua cabeça.
  \item Final, estendendo \cref{exo:g9:algebra:10}: provar isso
        o \omterm{def:g1:addition:def}{soma} de cinco inteiros consecutivos é sempre um
        \omterm{def:g9:arith:divisor}{múltiplo} de $5$, mas que o \omterm{def:g1:addition:def}{soma} de \emph{quatro}
        inteiros consecutivos é \emph{nunca} um \omterm{def:g9:arith:divisor}{múltiplo} de $4$.
        (Dois cálculos curtos --- e uma moral sobre o que
        a álgebra faz com as palavras “sempre” e “nunca”.)
\end{enumerate}
\end{problem}
