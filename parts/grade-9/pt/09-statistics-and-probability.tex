\chapter{Estatística e Probabilidade}\label{ch:g9:statproba}

Os dados e o acaso estão por toda parte: resultados esportivos, previsões meteorológicas,
jogos. Este capítulo ensina como resumir uma série de números por seus
significar, \omterm{def:g9:statproba:indicators}{mediana} e \omterm{def:g9:statproba:indicators}{faixa}, e como calcular o \omterm{def:g9:statproba:probability}{probabilidade} de simples
experimentos aleatórios --- incluindo experimentos em duas etapas, tratados com árvore
diagramas. Ambas as disciplinas são desenvolvidas muito mais no Ensino Médio
\omterm{def:g6:measure:volume}{volume} desta série.

\section{Resumindo dados}

\begin{definition}[Média, mediana, intervalo]\label{def:g9:statproba:indicators}
Para uma série de $N $ números:
\begin{itemize}
  \item o \emph{significativo}\index{significativo} é o \omterm{def:g1:addition:def}{soma} de todos os valores divididos
        por $ N $;
  \item o \emph{mediana}\index{mediana} é um valor que divide o classificado
        série em duas \omterm{def:g3:division:half}{metades} de tamanho igual: para $ N $ \omterm{def:g3:numbers:evenodd}{ímpar}, o meio
        valor; para $ N $ \omterm{def:g3:numbers:evenodd}{par}, o ponto médio dos dois valores intermediários;
  \item o \emph{faixa}\index{faixa} é o maior valor menos o
        menor.
\end{itemize}
\end{definition}

\begin{example}\label{ex:g9:statproba:indicators}
Notas de um aluno: $8,\ 12,\ 9,\ 15,\ 11$.
\begin{enumerate}
  \item Significar: $\dfrac{8 + 12 + 9 + 15 + 11}{5} = \dfrac{55}{5} = 11$.
  \item \omterm{def:g9:statproba:indicators}{Mediana}: classifique a série: $8, 9, 11, 12, 15$; o meio (3º)
        o valor é $11$.
  \item \omterm{def:g9:statproba:indicators}{Faixa}: $15 - 8 = 7$.
\end{enumerate}
Com uma sexta marca de $17$: classificado $8, 9, 11, 12, 15, 17$, o \omterm{def:g9:statproba:indicators}{mediana}
torna-se o ponto médio de $11$ e $12$, aquilo é $11.5$, e o significar
torna-se $\frac{72}{6} = 12$.
\end{example}

\begin{example}[Média ponderada]\label{ex:g9:statproba:weighted}
Os tamanhos de calçado vendidos em um dia:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccccc}
size & $38$ & $39$ & $40$ & $41$ & $42$ \\
\hline
count & $3$ & $7$ & $6$ & $3$ & $1$ \\
\end{tabular}
\end{center}
O significar o tamanho é
\[
\frac{3 \times 38 + 7 \times 39 + 6 \times 40 + 3 \times 41 + 1 \times 42}
{3 + 7 + 6 + 3 + 1}
= \frac{114 + 273 + 240 + 123 + 42}{20}
= \frac{792}{20} = 39.6 .
\]
\end{example}

\begin{remark}
O significar e o \omterm{def:g9:statproba:indicators}{mediana} pode diferir muito. Na série
$10, 10, 10, 10, 60$ o significar é $20$ (puxado pelo valor $60$) enquanto
o \omterm{def:g9:statproba:indicators}{mediana} é $10$. Pergunte sempre qual indicador representa a situação
melhor.
\end{remark}

\section{Probabilidade}

\begin{definition}[Probabilidade]\label{def:g9:statproba:probability}
Um experimento aleatório tem várias possibilidades \emph{resultados}; um
\emph{evento} é um conjunto de resultados. Cada resultado recebe um
\emph{probabilidade}\index{probabilidade}: um número entre $0$ e $1$,
com todas as probabilidades dos resultados somando $1$; o
probabilidade $\P(A)$ de um evento $ A $ é o \omterm{def:g1:addition:def}{soma} das probabilidades de
seus resultados. Quando tudo $ n $ os resultados são igualmente prováveis,
\[
\P(A) = \frac{\text{number of favorable outcomes}}{n}.
\]
\end{definition}

\begin{example}\label{ex:g9:statproba:spinner}
Uma roda é dividida em $8$ setores iguais: $3$ vermelho, $3$ azul, $2$ verde.
Cada setor tem \omterm{def:g9:statproba:probability}{probabilidade} $\frac18$, então
\[
\P(\text{red}) = \frac38, \qquad
\P(\text{green}) = \frac28 = \frac14, \qquad
\P(\text{not green}) = 1 - \frac14 = \frac34 .
\]
\end{example}

\begin{proposition}[Complemento]\label{prop:g9:statproba:complement}
Para cada evento $ A $, o \emph{evento contrário} $\bar A $ (``$ A $ não
ocorrer'') satisfaz
\[
\P(\bar A) = 1 - \P(A).
\]
\end{proposition}

\begin{proof}
Cada resultado está exatamente em um dos $ A $ e $\bar A $, então
$\P(A) + \P(\bar A)$ \omterm{def:g1:addition:def}{somas} as \omterm{def:g9:statproba:probability}{probabilidades} de todos os resultados, o que é
$1$.
\end{proof}

\section{Experimentos em duas etapas}

\begin{method}[Diagramas de árvore]\label{met:g9:statproba:tree}
Para um experimento em duas etapas:
\begin{enumerate}
  \item desenhe um ramo por resultado da primeira etapa e continue
        cada ramo com os resultados da segunda etapa, escrevendo cada
        \omterm{def:g9:statproba:probability}{probabilidade} em seu galho;
  \item multiplique as \omterm{def:g9:statproba:probability}{probabilidades} ao longo de um caminho para obter o \omterm{def:g9:statproba:probability}{probabilidade}
        desse caminho;
  \item adicione as \omterm{def:g9:statproba:probability}{probabilidades} de todos os caminhos que formam um evento.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:statproba:tree}
Uma bolsa contém $2$ vermelho e $3$ fichas pretas. Desenhe uma ficha, observe seu
cor, \emph{coloque de volta}, desenhe novamente. Cada sorteio dá vermelho com
\omterm{def:g9:statproba:probability}{probabilidade} $\frac25$, preto com \omterm{def:g9:statproba:probability}{probabilidade} $\frac35$.

\begin{omfigure}
\begin{tikzpicture}[scale=0.95]
  \coordinate (root) at (0,0);
  \coordinate (R) at (2.2,1.0);
  \coordinate (B) at (2.2,-1.0);
  \coordinate (RR) at (4.7,1.55);
  \coordinate (RB) at (4.7,0.45);
  \coordinate (BR) at (4.7,-0.45);
  \coordinate (BB) at (4.7,-1.55);
  \draw (root) -- (R) node[midway, above, font=\small, sloped] {$\frac25$};
  \draw (root) -- (B) node[midway, below, font=\small, sloped] {$\frac35$};
  \draw (R) -- (RR) node[midway, above, font=\small, sloped] {$\frac25$};
  \draw (R) -- (RB) node[midway, below, font=\small, sloped] {$\frac35$};
  \draw (B) -- (BR) node[midway, above, font=\small, sloped] {$\frac25$};
  \draw (B) -- (BB) node[midway, below, font=\small, sloped] {$\frac35$};
  \node[omThm, font=\small, above left=-2pt] at (R) {R};
  \node[font=\small, below left=-2pt] at (B) {B};
  \node[right, font=\small] at (RR)
    {RR:\ $\frac25 \times \frac25 = \frac{4}{25}$};
  \node[right, font=\small] at (RB)
    {RB:\ $\frac25 \times \frac35 = \frac{6}{25}$};
  \node[right, font=\small] at (BR)
    {BR:\ $\frac35 \times \frac25 = \frac{6}{25}$};
  \node[right, font=\small] at (BB)
    {BB:\ $\frac35 \times \frac35 = \frac{9}{25}$};
\end{tikzpicture}

{\small A árvore dos dois desenhos com reposição: multiplique ao longo do
ramificações e verifique se a \omterm{def:g1:addition:def}{soma} dos quatro caminhos é $1$.}
\end{omfigure}

\omterm{def:g9:statproba:probability}{Probabilidade} de duas fichas da mesma cor:
$\frac{4}{25} + \frac{9}{25} = \frac{13}{25}$. \omterm{def:g9:statproba:probability}{Probabilidade} de pelo menos
um vermelho: $1 - \P(\text{BB}) = 1 - \frac{9}{25} = \frac{16}{25}$.
\end{example}

\begin{remark}[Probabilidades de abordagem de frequências]
Jogando um dado justo $6000$ vezes dá \emph{sobre} $1000$ seis --- não
exatamente. À medida que o número de repetições aumenta, as \omterm{def:g7:stats:frequency}{frequências} observadas ficam
cada vez mais perto das \omterm{def:g9:statproba:probability}{probabilidades}: é isso que faz \omterm{def:g9:statproba:probability}{probabilidade}
a ferramenta certa para modelar experimentos repetidos (uma história desenvolvida em
o \omterm{def:g6:measure:volume}{volume} High School desta série).
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g9:statproba:1}
Calcule o significar, \omterm{def:g9:statproba:indicators}{mediana} e \omterm{def:g9:statproba:indicators}{faixa} da série
$14,\ 9,\ 17,\ 9,\ 12,\ 11,\ 12$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:statproba:2}
As temperaturas ao meio-dia durante uma semana foram $19,\ 21,\ 24,\ 18,\ 22,\ 25,\
19$ (graus). Calcule o significar (até um décimo) e o \omterm{def:g9:statproba:indicators}{mediana}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:statproba:3}
O número de gols marcados por uma equipe em $10$ partidas:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccc}
goals & $0$ & $1$ & $2$ & $3$ \\
\hline
matches & $2$ & $4$ & $3$ & $1$ \\
\end{tabular}
\end{center}
Calcule o significar número de gols por partida e o \omterm{def:g9:statproba:indicators}{mediana}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:statproba:4}
Um dado justo é lançado uma vez. Calcule o \omterm{def:g9:statproba:probability}{probabilidade} de: ``rolar um
$3$''; ``rolando um número \omterm{def:g3:numbers:evenodd}{par}''; ``rolando pelo menos um $3$''; ``não
rolando um $6$''.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:statproba:5}
Uma \omterm{def:g5:solids:def}{caixa} contém $12$ bolas: $5$ branco, $4$ vermelho e $3$ verde; uma bola é
sorteado aleatoriamente. Calcular $\P(\text{white})$, $\P(\text{red or green})$,
e $\P(\text{not white})$. O que você nota nos dois últimos?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:statproba:6}
Uma moeda é lançada e, em seguida, um dado justo é lançado. Desenhe a árvore (ou conte as
resultados) e calcular o \omterm{def:g9:statproba:probability}{probabilidade} de conseguir cabeças \emph{e} um
$6$; então de conseguir cabeças \emph{ou} um $6$ (ou ambos).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:statproba:7}
Na bolsa de \cref{ex:g9:statproba:tree} ($2$ vermelho, $3$ preto), os dois
os sorteios já foram feitos \emph{sem} substituição. Desenhe a nova árvore
(cuidado: as \omterm{def:g9:statproba:probability}{probabilidades} de segundo nível mudam) e calcule o
\omterm{def:g9:statproba:probability}{probabilidade} de tirar duas fichas pretas, depois duas fichas de cores diferentes
cores.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:statproba:8}
Depois $9$ partidas, um jogador de basquete calcula a média $14$ pontos por partida.
\begin{enumerate}
  \item Quantos pontos ela marcou no total?
  \item Quantos pontos ela deve marcar no $10$ª correspondência para a média
        $15$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g9:statproba:9}
Um jogo: jogue dois dados justos; você ganha se os dois resultados forem iguais
(``um duplo'').
\begin{enumerate}
  \item Calcule o \omterm{def:g9:statproba:probability}{probabilidade} de vencer.
  \item Você joga duas vezes (jogos independentes). Usando uma árvore, calcule o
        \omterm{def:g9:statproba:probability}{probabilidade} de ganhar pelo menos uma vez.
\end{enumerate}
\end{exercise}

\section{Problema: as apostas ruinosas do Chevalier}

\begin{problem}[{Problema de fim de semana --- a disputa de jogo de 1654
que criou a teoria da probabilidade, e a coincidência do aniversário
isso engana todas as salas de aula}]
\label{pb:g9:statproba:1}
Em 1654, um nobre francês apostador, o Chevalier de
Méré, ficou rico em uma aposta de dados e começou a perder em outra
que ele acreditava equivalente. Perplexo, ele perguntou ao matemático
Blaise Pascal; Pascal escreveu a Pierre de Fermat; e seus
troca de cartas fundou a teoria da \omterm{def:g9:statproba:probability}{probabilidade}. Isto
O problema repete todo o caso com as ferramentas deste capítulo
--- resultados igualmente prováveis, árvores e a regra do complemento
(\cref{prop:g9:statproba:complement}) --- e termina com o mais
coincidência contra-intuitiva de todos eles.

\medskip
\noindent\textbf{Part I --- Dice, honestly counted.}
\begin{enumerate}
  \item Jogue um dado justo duas vezes. Usando uma árvore (ou um $6 \times 6$
        tabela), calcule o \omterm{def:g9:statproba:probability}{probabilidade} de \emph{não} seis no
        dois lançamentos e deduza o \omterm{def:g9:statproba:probability}{probabilidade} de pelo menos um
        seis.
  \item Um colega argumenta: ``uma chance em seis por lançamento, duas vezes,
        então $\frac16 + \frac16 = \frac13$.'' Compare com a pergunta
        1 e aponte para os resultados exatos que sua adição conta
        duas vezes.
  \item Jogando dois dados e somando: calcule o \omterm{def:g9:statproba:probability}{probabilidade} de
        um \omterm{def:g1:addition:def}{soma} de $7$ e de um \omterm{def:g1:addition:def}{soma} de $2$. Por que é $7$ o
        favorito dos jogadores?
  \item Disputa antiga: com dois dados, é um \omterm{def:g1:addition:def}{soma} de $9$ ou um \omterm{def:g1:addition:def}{soma} de
        $10$ mais provável? Conte os resultados e resolva-os.
  \item Generalize a questão 1: explique por que \omterm{def:g9:statproba:probability}{probabilidade} de em
        pelo menos um seis em $ n $ rolos é
        \[
        1 - \left(\frac56\right)^{\!n} ,
        \]
        os não-seis ramos da árvore multiplicando-se a partir do nível
        para nivelar.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The two bets of the Chevalier.}
\begin{enumerate}[resume]
  \item Primeira aposta, \omterm{def:g3:numbers:evenodd}{par} dinheiro: \emph{pelo menos um seis em cada quatro
        rolos de um dado}. Calcular
        $\left(\frac56\right)^4$ como um \omterm{def:g9:fractions:fraction}{fração} e como decimal,
        e a vitória do Chevalier \omterm{def:g9:statproba:probability}{probabilidade}. Foi a aposta
        bom?
  \item O raciocínio do próprio Chevalier foi: “quatro rolos em
        $\frac16$ cada: $\frac46$ de uma chance. '' Isso deu quase
        a resposta certa aqui --- mas empurre para $7$ rolos: o que
        absurdo isso produz, e qual erro da questão 2
        isso se repete?
  \item Segunda aposta, considerada equivalente por \omterm{def:g7:prop:table}{proporcionalidade}
        (``$24$ rola em $\frac1{36}$ cada um: novamente
        $\frac{24}{36} = \frac46$''): \emph{pelo menos um
        duplo seis em $24$ lançamentos de dois dados}. Calcule o verdadeiro
        vencendo \omterm{def:g9:statproba:probability}{probabilidade} $1 - \left(\frac{35}{36}\right)^{24}$
        (calculadora). Por que o Chevalier foi lentamente arruinado?
  \item Quantos lançamentos de dois dados teriam restaurado seu
        vantagem? Teste $ n = 25$ com a calculadora e
        concluir.
  \item Enuncie a moral em duas frases: o que está errado, em
        geral, com a multiplicação de um \omterm{def:g9:statproba:probability}{probabilidade} pelo número de
        tenta --- e qual ferramenta correta substitui isso
        tentação?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Significa, \omterm{def:g9:statproba:indicators}{medianas}, birthdays.}
\begin{enumerate}[resume]
  \item Uma pequena empresa paga nove funcionários $2\,000$ euros por mês
        e o diretor $20\,000$. Calcule o significar e o
        \omterm{def:g9:statproba:indicators}{mediana} salário (\cref{def:g9:statproba:indicators}).
        Qual número deve o anúncio de emprego honestamente
        citação?
  \item Construa uma lista de cinco marcas de teste com significar $12$ e
        \omterm{def:g9:statproba:indicators}{mediana} $15$. O que o \omterm{def:g3:numbers:evenodd}{par} (significar abaixo \omterm{def:g9:statproba:indicators}{mediana}) dizer
        sobre o formato das marcas?
  \item O problema do aniversário: em uma turma de $23$ estudantes, o que
        é o \omterm{def:g9:statproba:probability}{probabilidade} que pelo menos dois fazem aniversário no mesmo dia?
        Configure o complemento (todos $23$ aniversários diferentes):
        \[
        \frac{365}{365} \times \frac{364}{365} \times
        \frac{363}{365} \times \dots \times \frac{343}{365} .
        \]
        Calcule o \omterm{def:g2:mult:def}{produto} dos três primeiros fatores, descreva
        como os fatores evoluem, e --- dado que o total
        \omterm{def:g2:mult:def}{produto} é sobre $0.493$ --- responda à pergunta. A maioria
        as pessoas esperam “muito improvável”: o que o
        matemática diz?
  \item A intuição reparada: quantos \emph{\omterm{def:g3:numbers:evenodd}{pares}} de
        os alunos podem compartilhar um aniversário em uma classe de $23$
        (\cref{pb:g6:wholes:1}contagem de apertos de mão)? Explique em um
        sentença por que esse número, não $23$ em si, impulsiona o
        surpresa.
  \item Final: descreva um experimento para verificar qualquer resultado
        --- de Mérápostamos com um dado real, ou o aniversário
        problema em várias \omterm{def:g4:numbers:classes}{classes} --- e declare cuidadosamente
        o que você espera das \omterm{def:g7:stats:frequency}{frequências} observadas como o número
        de repetições aumenta. (Essa expectativa tem um nome, o
        lei dos grandes números; sua prova aguarda no Alto
        \omterm{def:g6:measure:volume}{Volume} escolar.)
\end{enumerate}
\end{problem}
