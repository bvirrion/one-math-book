\chapter{Teorema de Tales}\label{ch:g9:thales}

Como você mede a altura de um \omterm{def:g8:solids:def}{pirâmide} sem subir? Tales
de Mileto comparou sua sombra com a sombra de uma vara. Seu teorema
--- \omterm{def:g6:lines:objects}{linhas} paralelas corte \omterm{def:g6:lines:objects}{segmentos} em proporcional peças --- é o
coração matemático de escala modelos, mapas e ampliações, e um dos
os teoremas mais antigos com um nome.

\section{O teorema}

\begin{theorem}[Tales]\label{thm:g9:thales:direct}
Deixe duas \omterm{def:g6:lines:objects}{linhas} se encontrarem em um ponto $A $. Pegar $ M $ na primeira \omterm{def:g6:lines:objects}{linha} e $ N $
no segundo, com $ B $ mais adiante no primeiro e $ C $ mais adiante
o segundo. Se as \omterm{def:g6:lines:objects}{linhas} $(MN)$ e $(BC)$ são \emph{\omterm{def:g6:lines:perp}{paralelo}}, então o
triângulos $ AMN $ e $ ABC $ ter proporcional lados:
\[
\frac{AM}{AB} = \frac{AN}{AC} = \frac{MN}{BC}.
\]
\end{theorem}
\admitted

\begin{method}[Calculando um comprimento com Thales]\label{met:g9:thales:compute}
\begin{enumerate}
  \item Identifique o ponto $ A $ onde as duas \omterm{def:g6:lines:objects}{linhas} se cruzam e os dois
        \omterm{def:g6:lines:objects}{linhas} paralelas; nomeie os dois triângulos;
  \item escreva as três proporções iguais, sempre ``triângulo pequeno sobre grande
        triângulo'';
  \item manter a igualdade das duas razões envolvendo os três conhecidos
        comprimentos e o desconhecido;
  \item resolva a proporção resultante por multiplicação cruzada.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:thales:compute}
Na configuração esquerda acima, suponha $ AM = 3$, $ AB = 5$, $ AN = 2.4$
e $ MN = 3.3$, e vamos calcular $ AC $ e $ BC $. Tales dá
\[
\frac{AM}{AB} = \frac{AN}{AC} = \frac{MN}{BC},
\qquad\text{i.e.}\qquad
\frac35 = \frac{2.4}{AC} = \frac{3.3}{BC}.
\]
De $\frac35 = \frac{2.4}{AC}$, multiplicação cruzada:
$3 \times AC = 5 \times 2.4 = 12$, então $ AC = 4$. De
$\frac35 = \frac{3.3}{BC}$: $3 \times BC = 16.5$, então $ BC = 5.5$.
\end{example}

\begin{example}[Tales na vida real]\label{ex:g9:thales:stick}
Uma vara vertical de altura $1$ m lança uma sombra de $1.5$ m, enquanto uma árvore
lança uma sombra de $12$ m no mesmo momento. Os \omterm{def:g3:shapes:shapes}{raios} solares são \omterm{def:g6:lines:perp}{paralelo}, então
os triângulos bastão e sombra e árvore e sombra estão em Tales
configuração:
\[
\frac{\text{height of tree}}{1} = \frac{12}{1.5},
\qquad\text{so the tree is } 8 \text{ m tall.}
\]
\end{example}

\section{O inverso}

\begin{theorem}[Conversa de Tales]\label{thm:g9:thales:converse}
Com os pontos colocados como em \cref{thm:g9:thales:direct} ($ M $ sobre $(AB)$,
$ N $ sobre $(AC)$, in the same order on both lines):
se
\[
\frac{AM}{AB} = \frac{AN}{AC},
\]
então as \omterm{def:g6:lines:objects}{linhas} $(MN)$ e $(BC)$ são \omterm{def:g6:lines:perp}{paralelo}.
\end{theorem}
\admitted

\begin{method}[Provando ou refutando o paralelismo]\label{met:g9:thales:converse}
\begin{enumerate}
  \item Calcule separadamente as duas proporções $\dfrac{AM}{AB}$ e
        $\dfrac{AN}{AC}$ (como frações, não \omterm{def:g6:decimals:rounding}{arredondamentos});
  \item se eles são iguais \emph{e} os pontos estão na mesma ordem em
        as duas \omterm{def:g6:lines:objects}{linhas}, as \omterm{def:g6:lines:objects}{linhas} são \omterm{def:g6:lines:perp}{paralelo} (conversa de Tales);
  \item se forem diferentes, as \omterm{def:g6:lines:objects}{linhas} são \emph{não} \omterm{def:g6:lines:perp}{paralelo} --- porque se
        fossem, o teorema de Tales forçaria as proporções a serem iguais.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:thales:converse}
Na \omterm{def:g6:lines:objects}{linha} 1, $ AM = 4$ e $ AB = 10$; na \omterm{def:g6:lines:objects}{linha} 2, $ AN = 6$ e $ AC = 15$.
Então
\[
\frac{AM}{AB} = \frac{4}{10} = \frac25,
\qquad
\frac{AN}{AC} = \frac{6}{15} = \frac25 :
\]
proporções iguais, mesma ordem: $(MN) \parallel (BC)$.

Com $ AC = 14$ em vez de: $\frac{6}{14} = \frac37 \neq \frac25$, então o
\omterm{def:g6:lines:objects}{linhas} não são \omterm{def:g6:lines:perp}{paralelo}.
\end{example}

\section{Ampliação e redução}

\begin{definition}[Homotetia de uma figura]\label{def:g9:thales:scaling}
Ampliar ou reduzir uma figura pelo \emph{escala
fator}\index{escala
fator} $ k > 0$ significa multiplicar \emph{todos} é
comprimentos por $ k $: um \emph{alargamento} quando $ k > 1$, um \emph{redução}
quando $ k < 1$. Na configuração de Tales, o triângulo $ AMN $ é o
redução de $ ABC $ pelo fator $ k = \frac{AM}{AB}$.
\end{definition}

\begin{proposition}[Efeito em ângulos e áreas]\label{prop:g9:thales:scaling}
Dimensionamento por fator $ k $ preserva ângulos e a forma das figuras,
multiplica cada comprimento por $ k $, e multiplica cada \emph{\omterm{def:g6:measure:area}{área}} por
$ k^2$.
\end{proposition}
\admitted

\begin{omfigure}
\begin{tikzpicture}[scale=0.72]
  \draw[thick, omDef, fill=omDef!8] (0,0) rectangle (2,1.2);
  \draw[thick, omThm, fill=omThm!8] (3.4,0) rectangle (7.4,2.4);
  \node[font=\small, omDef] at (1,-0.45) {$2 \times 1.2$};
  \node[font=\small, omThm] at (5.4,-0.45) {$4 \times 2.4$};
  \node[font=\small] at (1,0.6) {area $2.4$};
  \node[font=\small] at (5.4,1.2) {area $9.6$};
\end{tikzpicture}

{\small Dobrando os comprimentos ($ k = 2$) multiplica o \omterm{def:g6:measure:area}{área} por
$ k^2 = 4$: quatro cópias do retângulo pequeno lado a lado com o grande.}
\end{omfigure}

\begin{example}
Uma foto de $10 \times 15$ cm é ampliado com \omterm{def:g9:thales:scaling}{fator de escala} $ k = 3$: o
imprimir medidas $30 \times 45$ cm. Isso é \omterm{def:g6:measure:area}{área} vai de $150$ cm $^2$ para
$150 \times 9 = 1350$ cm $^2$ --- $9$ vezes maior, não $3$.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g9:thales:1}
As \omterm{def:g6:lines:objects}{linhas} $(MN)$ e $(BC)$ são \omterm{def:g6:lines:perp}{paralelo}, com $ M $ sobre $[AB]$ e $ N $ sobre
$[AC]$; $ AM = 2$, $ AB = 6$, $ AN = 3$, $ BC = 9$. Calcular $ AC $ e $ MN $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:thales:2}
Mesma configuração: $ AM = 5$, $ MB = 3$ (cuidadoso: $ MB $, não $ AB $!),
$ AN = 4$. Calcular $ AC $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:thales:3}
Na configuração borboleta, $ A $ está entre $ M $ e $ B $ e entre $ N $
e $ C $, com $(MN) \parallel (BC)$; $ AM = 3$, $ AB = 7.5$, $ AN = 2$,
$ MN = 2.6$. Calcular $ AC $ e $ BC $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:thales:4}
$ M $ está ligado $[AB]$ com $ AM = 6$, $ AB = 8$; $ N $ está ligado $[AC]$ com $ AN = 9$,
$ AC = 12$. São as \omterm{def:g6:lines:objects}{linhas} $(MN)$ e $(BC)$ \omterm{def:g6:lines:perp}{paralelo}?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:thales:5}
O mesmo que acima com $ AM = 4$, $ AB = 6$, $ AN = 5$, $ AC = 8$. São $(MN)$
e $(BC)$ \omterm{def:g6:lines:perp}{paralelo}? Justifique com cuidado.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:thales:6}
UM $1.8$ m pessoa alta fica de pé $2$ m de distância da \omterm{def:g8:powers:def}{base} de um poste de luz;
suas medidas de sombra $3$ m. Desenhe a configuração de Tales formada pelo
lâmpada, a pessoa e a ponta da sombra, e calcule a altura da
a lâmpada.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:thales:7}
Um mapa tem escala $1 : 25\,000$ (os comprimentos no mapa são os comprimentos reais
multiplicado por $ k = \frac{1}{25000}$).
\begin{enumerate}
  \item Duas aldeias são $6.8$ cm de distância no mapa. Qual é o verdadeiro
        distância, em km?
  \item Uma floresta tem \omterm{def:g6:measure:area}{área} de $8$ cm $^2$ no mapa. Qual é o seu real
        \omterm{def:g6:measure:area}{área}, em km $^2$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:thales:8}
Triângulo $ ABC $ tem $ AB = 12$, $ AC = 15$, $ BC = 18$. O ponto $ M $ sobre
$[AB]$ satisfaz $ AM = 8$, e a \omterm{def:g6:lines:objects}{linha} que passa $ M $ \omterm{def:g6:lines:perp}{paralelo} para $(BC)$
cortes $[AC]$ no $ N $.
\begin{enumerate}
  \item Calcular $ AN $ e $ MN $.
  \item Qual é o \omterm{def:g9:thales:scaling}{fator de escala} de $ ABC $ para $ AMN $? Compare o
        \omterm{def:g6:measure:perimeter}{perímetros} dos dois triângulos.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g9:thales:9}
Deixar $ ABCD $ seja um trapézio com $(AB) \parallel (CD)$, $ AB = 4$ e
$ CD = 6$, cujas diagonais $[AC]$ e $[BD]$ encontrar-se em $ O $.
Usando Thales na configuração borboleta ao redor $ O $, calcule o
proporção $\dfrac{OA}{OC}$, e mostre que $ O $ corta ambas as diagonais no
mesma proporção.
\end{exercise}

\section{Problema: A régua não marcada, a câmera pinhole e o
três meios de um trapézio}

\begin{problem}[{Problema de fim de semana --- Thales no trabalho: dividindo qualquer
segmento em partes iguais, medindo o Sol com uma caixa de sapatos, e um
trapézio onde todos os três meios famosos se encontram}]
\label{pb:g9:thales:1}
O teorema de Tales é a matemática de \emph{\omterm{def:g3:shapes:shapes}{raios}}: \omterm{def:g3:shapes:shapes}{raios} solares,
\omterm{def:g3:shapes:shapes}{raios} de luz através de um orifício, \omterm{def:g3:shapes:shapes}{raios} de lápis de um vértice. Isto
problema usa-o de três maneiras --- como uma ferramenta de construção (dividindo um
\omterm{def:g6:lines:objects}{segmento} de \emph{qualquer} comprimento, \omterm{def:g3:numbers:evenodd}{par} $\sqrt2$, em perfeitamente igual
partes), como um instrumento de medição (o \omterm{def:g4:geometry:circle}{diâmetro} do Sol, com
uma \omterm{def:g5:solids:def}{caixa} de sapatos e uma moeda), e como uma lupa no trapézio
de \cref{exo:g9:thales:9}, onde a aritmética, a geometria e
as médias harmônicas desta série aparecem todas em uma figura.

\medskip
\noindent\textbf{Part I --- Dividing with an unmarked ruler.}
\begin{enumerate}
  \item Desenhe um \omterm{def:g6:lines:objects}{segmento} $[AB]$ de $7$ cm. Para cortá-lo em três
        partes iguais com compasso e régua não marcada: desenhe qualquer \omterm{def:g3:shapes:shapes}{raio}
        de $ A $ (não através $ B $); sair de três bússolas iguais
        comprimentos nele, dando pontos $ P_1$, $ P_2$, $ P_3$; juntar
        $ P_3$ para $ B $; desenhe o \omterm{def:g6:lines:perp}{paralelos} para $(P_3 B)$ através
        $ P_1$ e $ P_2$. Execute a construção e marque onde
        o \omterm{def:g6:lines:perp}{paralelos} corte $[AB]$.
  \item Justifique com \cref{thm:g9:thales:direct} que os dois
        pontos marcados cortados $[AB]$ exatamente em seus terceiros pontos.
  \item Adapte o método para \omterm{def:g9:arith:divisor}{dividir} um \omterm{def:g6:lines:objects}{segmento} em $5$ igual
        partes, então para construir $\frac35$ de um dado \omterm{def:g6:lines:objects}{segmento}.
  \item Construa o ponto $ M $ de $[AB]$ com
        $\frac{AM}{MB} = \frac23$ (aquilo é, $ M $ no $\frac25$ de
        o caminho de $ A $). Quantos passos iguais no \omterm{def:g3:shapes:shapes}{raio}, e
        qual ponto se junta $ B $?
  \item Por que esta construção é melhor do que medir com um
        régua graduada? Considere um \omterm{def:g6:lines:objects}{segmento} de comprimento $\sqrt2$
        (diagonal de um quadrado unitário, irracional por
        \cref{pb:g9:sqrt:1}): o que a medição daria, e
        o que Tales dá?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The pinhole camera.}
Enfie um alfinete em um face de um fechado \omterm{def:g5:solids:def}{caixa}: pelo contrário
face, aparece uma imagem invertida do mundo. Um ponto de
objeto, o furo e o ponto da imagem estão alinhados --- luz
viaja em \omterm{def:g6:lines:objects}{linha} reta --- então o objeto (altura $ H $, à distância $ D $
na frente do buraco) e sua imagem (altura $ h $, nas costas
parede em profundidade $ d $ atrás do buraco) sente-se na borboleta
configuração de Tales.
\begin{enumerate}[resume]
  \item Derive a fórmula pinhole
        $ h = H \times \frac dD $ de Tales na borboleta
        ao redor do buraco.
  \item UM $12$ m suportes de árvores $30$ m de uma \omterm{def:g5:solids:def}{caixa} de sapatos de profundidade
        $20$ cm. Qual é a altura da sua imagem e qual é a altura?
  \item Aponte o \omterm{def:g5:solids:def}{caixa} ao Sol: às $ d = 1$ estou atrás do
        pinhole, a imagem do Sol é um disco de \omterm{def:g4:geometry:circle}{diâmetro} sobre
        $9$ mm. Dada a distância do Sol
        $ D = 1.5 \times 10^{11}$ m, calcule seu \omterm{def:g4:geometry:circle}{diâmetro} em
        \omterm{def:g9:fractions:scientific}{notação científica} (\cref{pb:g9:fractions:1}de
        notação no trabalho). Compare com o valor verdadeiro,
        $1.39 \times 10^9$ m.
  \item A Lua: \omterm{def:g4:geometry:circle}{diâmetro} $3.5 \times 10^6$ m, distância
        $3.8 \times 10^8$ m. Calcule a proporção
        $\frac{\text{diameter}}{\text{distance}}$ para a Lua
        e para o Sol. Que feliz coincidência os dois
        os números revelam --- e que evento espetacular isso causa
        tornar possível?
  \item Em uma ou duas frases: por que a imagem pinhole está
        de cabeça para baixo, e por que aumentar o orifício faz com que o
        imagem mais brilhante, mas mais desfocada?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The trapezoid of the three
means.}
$ ABCD $ é o trapézio de \cref{exo:g9:thales:9}:
$(AB) \parallel (CD)$, $ AB = 4$, $ CD = 6$, diagonais cruzando em
$ O $ com $\frac{OA}{OC} = \frac{OB}{OD} = \frac{4}{6} =
\frac23$.
\begin{enumerate}[resume]
  \item Prática borboleta: duas \omterm{def:g6:lines:objects}{linhas} se cruzam em $ O $ com
        $(MN) \parallel (PQ)$ em posição borboleta; $ OM = 3$,
        $ OP = 7.5$, $ MN = 4$ e $ OQ = 6$. Calcular $ PQ $ e
        $ ON $.
  \item Explique como Tales, aplicado com proporção $\frac12$,
        contém o teorema do ponto médio de
        \cref{thm:g8:midpoints:direct} como seu caso especial.
  \item Desenhe o \omterm{def:g6:lines:perp}{paralelo} para as \omterm{def:g8:powers:def}{bases} através $ O $; ele atende
        $[AD]$ no $ E $. Trabalhando no triângulo $ ACD $ (observação
        $\frac{AO}{AC} = \frac25$), calcular $ EO $.
  \item Pelo cálculo do espelho no triângulo $ BDC $, o
        pedaço $ OF $ (com $ F $ sobre $[BC]$) tem o mesmo comprimento.
        Conclua isso
        \[
        EF = 2 \times \frac{4 \times 6}{4 + 6} = 4.8 :
        \]
        o \omterm{def:g6:lines:perp}{paralelo} através do cruzamento diagonal mede o
        \emph{\omterm{pb:g8:speed:1}{média harmônica}} das \omterm{def:g8:powers:def}{bases} --- ida e volta
        média de \cref{pb:g8:speed:1}, reaparecendo em puro
        geometria.
  \item O grand finale: calcular as três médias clássicas de
        as \omterm{def:g8:powers:def}{bases} $4$ e $6$ --- aritmética
        $\frac{4+6}{2}$, geométrico $\sqrt{4 \times 6}$, harmônico
        $\frac{2 \times 4 \times 6}{4 + 6}$ --- e encomende-os.
        Cada um vive no trapézio: a média aritmética é o
        \omterm{def:g6:lines:objects}{linha} média unindo os pontos médios das pernas
        (\cref{pb:g7:areas:1}), o \omterm{pb:g8:speed:1}{média harmônica} é o seu
        \omterm{def:g6:lines:objects}{segmento} $ EF$, e o \omterm{pb:g8:cosine:1}{média geométrica} é o \omterm{def:g6:lines:perp}{paralelo}
        que corta o trapézio em dois \emph{semelhante}
        trapézios (verifique seu valor com uma calculadora; o
        a prova é um lindo desafio extra). Onde tem o
        a desigualdade entre os três foi provada neste livro
        (\cref{pb:g8:cosine:1}, \cref{pb:g8:speed:1})?
\end{enumerate}
\end{problem}
