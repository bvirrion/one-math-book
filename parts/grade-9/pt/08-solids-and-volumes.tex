\chapter{Sólidos e Volumes}\label{ch:g9:solids}

A geometria espacial começa com uma pequena família de \omterm{def:g5:solids:def}{sólidos} --- \omterm{def:g7:areas:prism}{prismas},
\omterm{def:g7:areas:prism}{cilindros}, \omterm{def:g8:solids:def}{pirâmides}, \omterm{def:g8:solids:def}{cones}, esferas --- e duas perguntas: quanto
eles seguram (\omterm{def:g6:measure:volume}{volume}) e o que você vê quando as corta (seções)?
O capítulo termina com um fato de grande importância prática: dimensionar um
\omterm{def:g5:solids:def}{sólido} por $k $ multiplica seu \omterm{def:g6:measure:volume}{volume} por $ k^3$.

\section{Os sólidos clássicos e seus volumes}

\begin{theorem}[Fórmulas de volume]\label{thm:g9:solids:volumes}
Escrever $ B $ para o \omterm{def:g6:measure:area}{área} da \omterm{def:g8:powers:def}{base} e $ h $ para a altura (a distância
entre a \omterm{def:g8:powers:def}{base} e o oposto face ou ápice).
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{l|l}
\omterm{def:g5:solids:def}{sólido} & \omterm{def:g6:measure:volume}{volume} \\
\hline
\omterm{def:g7:areas:prism}{prisma}, \omterm{def:g7:areas:prism}{cilindro} & $ V = B \times h $ \\[5pt]
\omterm{def:g8:solids:def}{pirâmide}, \omterm{def:g8:solids:def}{cone} & $ V = \dfrac13\, B \times h $ \\[5pt]
sphere of \omterm{def:g3:shapes:shapes}{raio} $ r $ & $ V = \dfrac43\,\pi r^3$ \\
\end{tabular}
\end{center}
Por um \omterm{def:g7:areas:prism}{cilindro} e um \omterm{def:g8:solids:def}{cone} de \omterm{def:g3:shapes:shapes}{raio} $ r $, a \omterm{def:g8:powers:def}{base} \omterm{def:g6:measure:area}{área} é $ B = \pi r^2$;
a superfície da esfera \omterm{def:g6:measure:area}{área} é $4\pi r^2$.
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:volumes}
Uma lata cilíndrica tem \omterm{def:g3:shapes:shapes}{raio} $4$ cm e altura $10$ cm:
\[
V = \pi r^2 h = \pi \times 16 \times 10 = 160\pi \approx 503 \text{ cm}^3
\approx 0.5 \text{ L}.
\]
UM \omterm{def:g8:solids:def}{cone} com a mesma \omterm{def:g8:powers:def}{base} e altura detém um terço disso:
$\frac{160\pi}{3} \approx 168$ cm $^3$. Uma esfera de \omterm{def:g3:shapes:shapes}{raio} $4$ cm:
$ V = \frac43 \pi \times 64 = \frac{256\pi}{3} \approx 268$ cm $^3$.
\end{example}

\begin{example}[Uma pirâmide passo a passo]\label{ex:g9:solids:pyramid}
UM \omterm{def:g8:solids:def}{pirâmide} tem uma \omterm{def:g8:powers:def}{base} quadrada de lado $6$ m e altura $5$ m.
\begin{enumerate}
  \item \omterm{def:g8:powers:def}{Base} \omterm{def:g6:measure:area}{área}: $ B = 6^2 = 36$ eu $^2$.
  \item \omterm{def:g6:measure:volume}{Volume}: $ V = \frac13 \times 36 \times 5 = 60$ eu $^3$.
\end{enumerate}
Cuidado com as unidades: se o lado fosse dado em cm e a altura em m, um dos
eles teriam que ser convertidos primeiro.
\end{example}

\section{Seções por aviões}

\begin{proposition}[Seções dos sólidos clássicos]\label{prop:g9:solids:sections}
Cortando um \omterm{def:g5:solids:def}{sólido} por um plano produz uma figura plana, seu
\emph{corte transversal}\index{corte transversal}:
\begin{itemize}
  \item um \omterm{def:g7:areas:prism}{prisma} ou \omterm{def:g7:areas:prism}{cilindro} corte \emph{\omterm{def:g6:lines:perp}{paralelo} para sua \omterm{def:g8:powers:def}{base}} dá uma cópia
        da \omterm{def:g8:powers:def}{base}, em todas as alturas;
  \item um \omterm{def:g8:solids:def}{pirâmide} ou \omterm{def:g8:solids:def}{cone} corte \omterm{def:g6:lines:perp}{paralelo} à sua \omterm{def:g8:powers:def}{base} dá uma
        \emph{redução} da \omterm{def:g8:powers:def}{base}: à distância $ d $ do ápice,
        o \omterm{def:g9:thales:scaling}{fator de escala} é $ k = \frac{d}{h}$;
  \item uma esfera de \omterm{def:g3:shapes:shapes}{raio} $ r $ cortado por um avião à distância $ d < r $ de
        o centro dá um círculo de \omterm{def:g3:shapes:shapes}{raio} $\sqrt{r^2 - d^2}$ (pelo
        teorema de Pitágoras).
\end{itemize}
\end{proposition}
\admitted

\begin{example}\label{ex:g9:solids:section}
UM \omterm{def:g8:solids:def}{cone} tem \omterm{def:g8:powers:def}{base} \omterm{def:g3:shapes:shapes}{raio} $6$ e altura $9$. A seção por um avião
\omterm{def:g6:lines:perp}{paralelo} para a \omterm{def:g8:powers:def}{base}, à distância $3$ do ápice, é um disco dimensionado por
$ k = \frac39 = \frac13$: isso é \omterm{def:g3:shapes:shapes}{raio} é $6 \times \frac13 = 2$.

Uma esfera de \omterm{def:g3:shapes:shapes}{raio} $5$ é cortado por um avião à distância $3$ de seu
centro: a seção é um círculo de \omterm{def:g3:shapes:shapes}{raio} $\sqrt{25 - 9} = 4$.
\end{example}

\section{Dimensionando sólidos}

\begin{theorem}[Efeito de uma escala em comprimentos, áreas, volumes]\label{thm:g9:solids:scaling}
Quando um \omterm{def:g5:solids:def}{sólido} é ampliado ou reduzido pelo \omterm{def:g9:thales:scaling}{fator de escala} $ k > 0$:
\begin{itemize}
  \item todos os comprimentos são multiplicados por $ k $;
  \item todos \omterm{def:g6:measure:area}{áreas} são multiplicados por $ k^2$;
  \item todos \omterm{def:g6:measure:volume}{volumes} são multiplicados por $ k^3$.
\end{itemize}
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:scaling}
Um modelo de carro em escala $\frac{1}{18}$: os comprimentos são os do carro real
multiplicado por $ k = \frac{1}{18}$, superfície pintada multiplicada por
$ k^2 = \frac{1}{324}$, e \omterm{def:g6:measure:volume}{volume} por $ k^3 = \frac{1}{5832}$.

Dobrando o \omterm{def:g3:shapes:shapes}{raio} de uma esfera ($ k = 2$) multiplica seu \omterm{def:g6:measure:volume}{volume} por
$2^3 = 8$: verifique a fórmula,
$\frac43\pi (2r)^3 = \frac43 \pi \times 8r^3$.
\end{example}

\begin{example}[Cone truncado]\label{ex:g9:solids:truncated}
UM \omterm{def:g8:solids:def}{cone} de altura $9$ e \omterm{def:g8:powers:def}{base} \omterm{def:g3:shapes:shapes}{raio} $6$ (\omterm{def:g6:measure:volume}{volume}
$\frac13 \pi \times 36 \times 9 = 108\pi $) é cortado à distância $3$ de
o ápice e o pequeno \omterm{def:g8:solids:def}{cone} acima do corte é removido. O pequeno \omterm{def:g8:solids:def}{cone} é
a redução por $ k = \frac13$, então é \omterm{def:g6:measure:volume}{volume} é
$108\pi \times \left(\frac13\right)^3 = 4\pi $. O \omterm{def:g3:division:remainder}{restante} \omterm{def:g5:solids:def}{sólido} (um
\emph{truncado \omterm{def:g8:solids:def}{cone}}) tem \omterm{def:g6:measure:volume}{volume} $108\pi - 4\pi = 104\pi \approx 327$.
\end{example}

\section{Problema: lápide de Arquimedes}

\begin{problem}[{Problema de fim de semana --- a esfera tem dois terços de
seu cilindro (duas vezes), a pilha 1:2:3 e o cubo quadrado
lei que proíbe gigantes}]
\label{pb:g9:solids:1}
Arquimedes provou muitos teoremas, mas um o deixou tão orgulhoso que
ele pediu seu \emph{figura} para ser esculpido em seu túmulo: uma esfera
aninhado em seu mais apertado \omterm{def:g7:areas:prism}{cilindro}. Um século depois, o escritor romano
Cícero, examinando os arbustos perto de Siracusa, reconheceu o
sepultura ``pela esfera e pelo \omterm{def:g7:areas:prism}{cilindro}''. Este problema recupera
o que a escultura codifica --- um milagre duplo de dois terços --- então
segue \omterm{def:g6:measure:volume}{volumes} e superfícies (\cref{thm:g9:solids:volumes},
\cref{thm:g9:solids:scaling}) a uma lei que rege gigantes, formigas
e resfriamento de planetas.

\medskip
\noindent\textbf{Part I --- The carving decoded.}
Uma esfera de \omterm{def:g3:shapes:shapes}{raio} $ r $ fica exatamente dentro de um \omterm{def:g7:areas:prism}{cilindro}: mesmo
\omterm{def:g3:shapes:shapes}{raio}, altura $2r $.
\begin{enumerate}
  \item Calcule o cilindrode \omterm{def:g6:measure:volume}{volume} em termos de $ r $.
  \item Calcule a razão entre as esferas \omterm{def:g6:measure:volume}{volume} para o
        \omterm{def:g7:areas:prism}{cilindro}'s. Por que Arquimedes poderia ter gostado que o
        a resposta não contém $\pi $ e não $ r $?
  \item Agora as superfícies: compare a superfície da esfera \omterm{def:g6:measure:area}{área} com
        o cilindrode \emph{total} superfície (parte lateral mais
        as duas tampas). Que proporção aparece --- de novo?
  \item O espaço vazio entre a esfera e \omterm{def:g7:areas:prism}{cilindro} tem \omterm{def:g6:measure:volume}{volume}
        $2\pi r^3 - \frac43\pi r^3$. Mostre que essa sobra
        é exatamente igual ao \omterm{def:g6:measure:volume}{volume} de \emph{dois \omterm{def:g8:solids:def}{cones}} de \omterm{def:g3:shapes:shapes}{raio}
        $ r $ e altura $ r $.
  \item Uma bola de basquete de \omterm{def:g3:shapes:shapes}{raio} $12$ cm é vendido no mais apertado
        cilíndrico \omterm{def:g5:solids:def}{caixa}. Calcule ambos \omterm{def:g6:measure:volume}{volumes} (para o mais próximo
        $100$ cm $^3$) e a porcentagem do \omterm{def:g5:solids:def}{caixa} isso é
        vazio.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Bowls, molds and moons.}
\begin{enumerate}[resume]
  \item Uma tigela hemisférica tem \omterm{def:g3:shapes:shapes}{raio} $10$ cm. Calcule seu
        \omterm{def:g2:measure:units}{capacidade}, em cm $^3$ e em litros (para o decilitro).
  \item A pilha 1:2:3: a \omterm{def:g8:solids:def}{cone}, um hemisfério e um \omterm{def:g7:areas:prism}{cilindro},
        todos \omterm{def:g3:shapes:shapes}{raio} $10$ cm e altura $10$ cm. Calcule o
        três \omterm{def:g6:measure:volume}{volumes} e verifique a proporção lendária
        $1 : 2 : 3$. (Arquimedes teria apreciado isso
        um também.)
  \item Uma esfera de chocolate de \omterm{def:g3:shapes:shapes}{raio} $3$ cm é derretido em um
        molde cilíndrico de \omterm{def:g3:shapes:shapes}{raio} $3$ cm. Qual a altura do
        alcance do chocolate? (Exato \omterm{def:g9:fractions:fraction}{fração}, então para o
        milímetro.)
  \item A Terra \omterm{def:g3:shapes:shapes}{raio} é sobre $6\,371$ km, a Lua
        sobre $1\,737$ km. Calcule a razão dos \omterm{def:g3:shapes:shapes}{raios} e então
        --- com \cref{thm:g9:solids:scaling} --- a proporção de
        o \omterm{def:g6:measure:volume}{volumes}: quantas luas caberiam em um buraco
        Terra, por \omterm{def:g6:measure:volume}{volume}?
  \item Do mesmo teorema: qual é a razão Terra-Lua de
        \emph{superfície \omterm{def:g6:measure:area}{áreas}}? E em geral, quando um balão
        \omterm{def:g3:shapes:shapes}{raio} dobra, o que acontece com seu \omterm{def:g6:measure:volume}{volume} e ao seu
        superfície? Indique o \emph{lei do \omterm{def:g5:solids:def}{quadrado-cubo}}: como uma forma
        escalas acima, \omterm{def:g6:measure:volume}{volumes} ultrapassar as superfícies.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The square--cube law rules the
world.}
\begin{enumerate}[resume]
  \item O argumento de Galileu contra os gigantes: imagine um humano
        ampliado por um fator $10$, mesmas proporções. Pelo que
        fator o peso aumenta (o peso segue \omterm{def:g6:measure:volume}{volume})? Por
        que fator o \omterm{prop:g9:solids:sections}{corte transversal} dos ossos crescem
        (um \omterm{def:g6:measure:area}{área})? Por qual fator, então, o \emph{pressão}
        em cada centímetro quadrado de osso cresce --- e o que
        acontece com o gigante?
  \item A mesma lei ao contrário explica o heroísmo das formigas: força
        segue o músculo \omterm{prop:g9:solids:sections}{corte transversal} (um \omterm{def:g6:measure:area}{área}), o peso segue
        \omterm{def:g6:measure:volume}{volume}. Para um animal $100$ vezes menor em cada
        direção, por quais fatores o peso e a força
        encolhe, e por qual fator a relação resistência-peso
        proporção melhora?
  \item O calor é produzido pelo corpo \omterm{def:g6:measure:volume}{volume} e perdido através
        sua superfície. Mostre que para uma esfera o
        a relação \omterm{def:g6:measure:volume}{superfície-volume} é $\frac{3}{r}$, e calcule-o
        para $ r = 1$ e $ r = 2$. O que esfria mais rápido, um mouse ou
        um urso --- e por que os bebês precisam de chapéus no inverno?
  \item Duas laranjas esféricas de \omterm{def:g3:shapes:shapes}{raios} $4$ centímetros e $5$ centímetros são
        vendido pelo mesmo preço. Calcule a proporção de seus
        \omterm{def:g6:measure:volume}{volumes}: quanto mais laranja o grande dá por
        o mesmo dinheiro?
  \item Finale: descreva precisamente o que a figura da lápide
        codifica --- os dois ``dois terços'' das questões 2 e 3
        --- e responda a Cícero em uma frase: por que um
        matemático escolhe, em cada conquista de sua
        gênio da engenharia, uma esfera em um \omterm{def:g7:areas:prism}{cilindro}?
\end{enumerate}
\end{problem}
