\chapter{Frações e Potências}\label{ch:g9:fractions}

Frações e \omterm{def:g9:fractions:power}{poderes} são as ferramentas diárias de computação: compartilhar
quantidades, comparar proporções, escrever muito grande ou muito pequeno
números. Este capítulo consolida as regras para computar com eles ---
com cada etapa intermediária escrita --- e apresenta científico
notação.

\section{Computando com frações}

\begin{definition}[Fração]\label{def:g9:fractions:fraction}
UM \emph{fração}\index{fração} $\dfrac ab $ (com $ b \neq 0$) é o
\omterm{def:g3:division:remainder}{quociente} de $ a $ por $ b $. Duas frações são iguais quando uma é obtida de
o outro multiplicando (ou dividindo) \omterm{def:g4:fractions:def}{numerador} e \omterm{def:g4:fractions:def}{denominador} pelo
mesmo número diferente de \omterm{def:g1:counting:zero}{zero}:
\[
\frac ab = \frac{a \times k}{b \times k} \qquad (k \neq 0).
\]
\end{definition}

\begin{example}\label{ex:g9:fractions:simplify}
Simplificar $\dfrac{42}{56}$ passo a passo:
\[
\frac{42}{56} = \frac{42 \div 2}{56 \div 2} = \frac{21}{28}
= \frac{21 \div 7}{28 \div 7} = \frac34 .
\]
(Em \cref{ch:g9:arith} o maior comum \omterm{def:g5:division:multiple}{divisor} farei isso em um
passo.)
\end{example}

\begin{method}[Adição e subtração de frações]\label{met:g9:fractions:add}
\begin{enumerate}
  \item Coloque os dois frações mais de um \emph{comum \omterm{def:g4:fractions:def}{denominador}} (um comum
        múltiplo dos dois \omterm{def:g4:fractions:def}{denominadores});
  \item adicione ou subtraia o \omterm{def:g4:fractions:def}{numeradores}, mantendo o \omterm{def:g4:fractions:def}{denominador};
  \item simplifique o resultado, se possível.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:fractions:add}
Calcular $\dfrac56 + \dfrac34$. Um comum \omterm{def:g4:fractions:def}{denominador} de $6$ e $4$ é
$12$:
\[
\frac56 + \frac34
= \frac{5 \times 2}{6 \times 2} + \frac{3 \times 3}{4 \times 3}
= \frac{10}{12} + \frac{9}{12}
= \frac{19}{12}.
\]
Calcular $2 - \dfrac37$. Escrever $2$ como um \omterm{def:g9:fractions:fraction}{fração} sobre $7$:
\[
2 - \frac37 = \frac{14}{7} - \frac37 = \frac{11}{7}.
\]
\end{example}

\begin{method}[Multiplicando e dividindo frações]\label{met:g9:fractions:mult}
\begin{enumerate}
  \item Para multiplicar, multiplique \omterm{def:g4:fractions:def}{numeradores} juntos e \omterm{def:g4:fractions:def}{denominadores}
        junto: $\dfrac ab \times \dfrac cd = \dfrac{ac}{bd}$
        (simplificar \emph{antes} multiplicando sempre que possível);
  \item dividir, multiplicar pelo \omterm{def:g8:fractions:inverse}{inverso} do \omterm{def:g5:division:multiple}{divisor}:
        $\dfrac ab \div \dfrac cd = \dfrac ab \times \dfrac dc $.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:fractions:mult}
\[
\frac{7}{15} \times \frac{25}{14}
= \frac{7 \times 25}{15 \times 14}
= \frac{7 \times 25}{15 \times 2 \times 7}
= \frac{25}{30} = \frac56,
\]
onde simplificamos pelo fator comum $7$, então por $5$. E um
divisão:
\[
\frac{3}{4} \div \frac{9}{8}
= \frac34 \times \frac89
= \frac{3 \times 8}{4 \times 9}
= \frac{24}{36} = \frac23 .
\]
\end{example}

\section{Poderes}

\begin{definition}[Poder]\label{def:g9:fractions:power}
Para um número real $ a $ e um número inteiro positivo $ n $, o
\emph{poder}\index{poder} $ a^n $ é o \omterm{def:g2:mult:def}{produto} de $ n $ fatores iguais a
$ a $:
\[
a^n = \underbrace{a \times a \times \dots \times a}_{n \text{ factors}} .
\]
Por convenção $ a^0 = 1$ (para $ a \neq 0$) e negativo \omterm{def:g8:powers:def}{expoentes} denotar
\omterm{def:g8:fractions:inverse}{inverso}:
\[
a^{-n} = \frac{1}{a^n}.
\]
\end{definition}

\begin{example}
$2^4 = 2 \times 2 \times 2 \times 2 = 16$; \quad
$(-3)^2 = 9$ mas $-3^2 = -(3^2) = -9$ (o \omterm{def:g8:powers:def}{expoente} liga antes do
sinal de menos); \quad $10^{-3} = \frac{1}{1000} = 0.001$; \quad
$5^1 = 5$ e $5^0 = 1$.
\end{example}

\begin{theorem}[Regras de expoentes]\label{thm:g9:fractions:rules}
Para todos diferentes de \omterm{def:g1:counting:zero}{zero} $ a $, $ b $ e todos os números inteiros $ m $, $ n $:
\[
a^m \times a^n = a^{m+n},
\qquad
\frac{a^m}{a^n} = a^{m-n},
\qquad
\left(a^m\right)^n = a^{mn},
\qquad
(ab)^n = a^n b^n .
\]
\end{theorem}

\begin{proof}
Para positivo \omterm{def:g8:powers:def}{expoentes}, conte os fatores. Para a primeira regra:
$ a^m \times a^n $ é $ m $ fatores $ a $ seguido pela $ n $ fatores $ a $, no total
$ m + n $ fatores. Para o terceiro:
$\left(a^m\right)^n $ repete $ n $ vezes um bloco de $ m $ fatores, dando
$ mn $ fatores. As outras duas regras são semelhantes; a extensão para \omterm{def:g1:counting:zero}{zero} e
negativo \omterm{def:g8:powers:def}{expoentes} é verificado a partir de $ a^{-n} = \frac{1}{a^n}$ (e faz
a convenção $ a^0 = 1$ a única escolha consistente, uma vez que
$ a^n \times a^0$ deve ser $ a^{n+0} = a^n $).
\end{proof}

\begin{example}\label{ex:g9:fractions:rules}
Simplifique passo a passo:
\[
\frac{3^7 \times 3^{-2}}{3^4}
= \frac{3^{7 + (-2)}}{3^4}
= \frac{3^5}{3^4}
= 3^{5-4} = 3^1 = 3 .
\]
E com duas letras: $(2a^3)^2 \times a = 4 a^6 \times a = 4a^7$.
\end{example}

\section{Potências de dez e notação científica}

\begin{proposition}[Potências de dez]\label{prop:g9:fractions:ten}
Para um número inteiro positivo $ n $: $10^n $ está escrito ``$1$ seguido pela $ n $
\omterm{def:g1:counting:zero}{zeros}'', e $10^{-n} = 0.0\dots01$ com o $1$ no $ n $-ésima casa decimal
lugar. Multiplicando um \omterm{def:g6:decimals:places}{número decimal} por $10^n $ (resp.\ $10^{-n}$) mudanças
é \omterm{def:g4:decimals:point}{ponto decimal} $ n $ lugares à direita (resp.\ esquerda).
\end{proposition}

\begin{definition}[Notação científica]\label{def:g9:fractions:scientific}
O \emph{notação científica}\index{notação científica} de um positivo
\omterm{def:g6:decimals:places}{número decimal} é a maneira única de escrevê-lo como
\[
a \times 10^n,
\qquad\text{with } 1 \leq a < 10 \text{ and } n \text{ an integer.}
\]
\end{definition}

\begin{example}\label{ex:g9:fractions:scientific}
A distância Terra-Sol é de cerca de $149\,600\,000$ quilômetros $= 1.496 \times
10^8$ km. O tamanho de uma molécula de água é cerca de $0.000\,000\,000\,28$ eu
$= 2.8 \times 10^{-10}$ m. Para calcular com esses números, agrupe os
partes decimais e o \omterm{def:g9:fractions:power}{poderes} muitas vezes:
\[
(3 \times 10^5) \times (2.5 \times 10^{-2})
= (3 \times 2.5) \times 10^{5 + (-2)}
= 7.5 \times 10^{3}.
\]
\end{example}

\begin{method}[Colocando um número em notação científica]\label{met:g9:fractions:scientific}
\begin{enumerate}
  \item Mova o \omterm{def:g4:decimals:point}{ponto decimal} logo após o primeiro dígito diferente de \omterm{def:g1:counting:zero}{zero}; isso
        dá o fator $ a $ com $1 \leq a < 10$;
  \item conte o número de lugares que o ponto se moveu: essa contagem é $ n $,
        positivo se o número original fosse $\geq 10$, negativo se fosse
        $< 1$;
  \item verificar: $ a \times 10^n $ deve reproduzir o número original.
\end{enumerate}
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g9:fractions:1}
Calcule e dê o resultado de forma totalmente simplificada \omterm{def:g9:fractions:fraction}{fração}:
\[
\frac13 + \frac15, \qquad
\frac72 - \frac53, \qquad
\frac{5}{12} + \frac38, \qquad
3 - \frac45 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:fractions:2}
Calcule e simplifique:
\[
\frac59 \times \frac{27}{35}, \qquad
\frac{8}{15} \div \frac{4}{5}, \qquad
\frac23 \times \frac34 \times \frac45 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:fractions:3}
Calcular:
\[
2^5, \qquad (-2)^4, \qquad -2^4, \qquad 4^{-2}, \qquad
\left(\tfrac23\right)^3, \qquad 7^0 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:fractions:4}
Escreva como um único \omterm{def:g9:fractions:power}{poder}:
\[
5^3 \times 5^6, \qquad
\frac{7^9}{7^5}, \qquad
\left(2^3\right)^4, \qquad
\frac{3^2 \times 3^7}{3^5}, \qquad
2^5 \times 4 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:fractions:5}
Escreva em \omterm{def:g9:fractions:scientific}{notação científica}: $45\,000$; $0.0072$; $302.5$;
$0.000\,000\,9$; doze milhões.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:fractions:6}
Calcule, dando o resultado em \omterm{def:g9:fractions:scientific}{notação científica}:
\[
(2 \times 10^7) \times (4.5 \times 10^{-3}),
\qquad
\frac{9 \times 10^5}{3 \times 10^{-2}},
\qquad
(5 \times 10^3)^2 .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:fractions:7}
Um tanque é $\frac35$ completo. Depois de adicionar $30$ litros é isso $\frac34$ completo.
Qual é o \omterm{def:g2:measure:units}{capacidade} do tanque? (Expresse o adicionado \omterm{def:g9:fractions:fraction}{fração} do tanque
primeiro.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:fractions:8}
Alice come $\frac14$ de um bolo, então Ben come $\frac13$ do que resta,
então Carla come $\frac12$ do que ainda resta. O que \omterm{def:g9:fractions:fraction}{fração} do
sobra bolo no final? (Calcule o \omterm{def:g3:division:remainder}{restante} após cada etapa.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:fractions:9}
A luz viaja $3 \times 10^8$ metros por segundo. O Sol está prestes
$1.5 \times 10^{11}$ estou longe. Quanto tempo leva a luz solar para chegar até nós?
Dê a resposta em segundos (\omterm{def:g9:fractions:scientific}{notação científica} não é necessário), então em
minutos e segundos.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g9:fractions:10}
Qual é maior, $2^{30}$ ou $3^{20}$? (Dica: escreva ambos como \omterm{def:g9:fractions:power}{poderes} com
\omterm{def:g8:powers:def}{expoente} $10$ usando $\left(a^m\right)^n = a^{mn}$ e compare $2^3$ com
$3^2$.)
\end{exercise}

\section{Problema: O código secreto de repetição de decimais}

\begin{problem}[{Problema de fim de semana --- cada fração tem um decimal
escrita que para ou se repete, cada decimal repetido é um
fração, e $0.999\ldots $ é exatamente igual $1$}]
\label{pb:g9:fractions:1}
Dividir $3$ por $11$ e os dígitos cantam:
$0.272727\ldots $ para sempre. Dividir $1$ por $4$ e os dígitos param
morto: $0.25$. Este problema prova que estes são os únicos dois
comportamentos possíveis para um \omterm{def:g9:fractions:fraction}{fração} --- e depois inverte o
máquina: dada qualquer dízima repetida, ela reconstrói o \omterm{def:g9:fractions:fraction}{fração}
escondendo-se atrás dele. No caminho resolve-se, de uma vez por todas, o
mais discutido sobre a igualdade na matemática escolar.

\medskip
\noindent\textbf{Part I --- From \omterm{def:g9:fractions:fraction}{fração} to decimal.}
\begin{enumerate}
  \item Calcule por divisão longa as escritas decimais de
        $\frac14$, $\frac13$, $\frac56$ e $\frac{3}{11}$.
        Qual parada, qual repetição e com qual bloco de repetição?
  \item Explique por que a escrita decimal de \emph{qualquer} \omterm{def:g9:fractions:fraction}{fração}
        $\frac ab $ deve parar ou eventualmente repetir. (No
        divisão longa por $ b $, quais valores podem os sucessivos
        restos pegar? O que acontece no momento em que um \omterm{def:g3:division:remainder}{restante}
        aparece pela segunda vez?)
  \item Empurre a divisão de $1$ por $7$ longe o suficiente para encontrar o
        bloco de repetição. Quanto tempo é e como esse comprimento
        comparar com o limite prometido pela pergunta 2?
  \item Alguns frações pare porque eles \omterm{def:g4:fractions:def}{denominador} pode ser
        explodido para um \omterm{def:g9:fractions:power}{poder} muitas vezes: $\frac14 =
        \frac{25}{100}$, $\frac{3}{8} = \frac{375}{1000}$.
        Explique o critério e por que não há multiplicador de número inteiro
        pode virar $3$, $6$ ou $11$ em $10$, $100$,
        $1000, \dots $ (qual dígito seria o \omterm{def:g2:mult:def}{produto} terminar em?).
  \item Sem dividir, classifique $\frac{9}{40}$, $\frac{5}{12}$ e
        $\frac{33}{50}$ em ``paradas'' e ``repetições'', então dê
        a escrita decimal dos dois que param.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- From repeating decimal back to the
\omterm{def:g9:fractions:fraction}{fração}.}
\begin{enumerate}[resume]
  \item O truque da mudança. Deixar $ x = 0.7777\ldots $ Calcular $10x $,
        então $10x - x $, e deduza $ x $ como um \omterm{def:g9:fractions:fraction}{fração}. Verifique por
        dividindo.
  \item Deixar $ x = 0.727272\ldots $ Qual \omterm{def:g9:fractions:power}{poder} de dez turnos $ x $ por
        exatamente um bloco de repetição? Use-o para escrever $ x $ como um
        \omterm{def:g9:fractions:fraction}{fração} em termos mais baixos.
  \item Um caso misto: $ x = 0.58333\ldots $ (o $3$ sozinho
        repete). Calcular $1000x - 100x $ e deduzir $ x $ como um
        \omterm{def:g9:fractions:fraction}{fração} em termos mais baixos.
  \item O famoso: aplique o truque do shift para
        $ x = 0.9999\ldots $ O que \omterm{def:g9:fractions:fraction}{fração} --- o que \emph{número}
        --- você acha? Reconcilie o resultado com o seu
        intuição, lembrando-se do vizinho fantasma de
        \cref{pb:g6:decimals:1} e o encolhimento \omterm{def:g3:division:remainder}{restante} de
        \cref{pb:g6:fractions:1}: é $0.999\ldots $ ``logo abaixo''
        $1$, ou outro nome para $1$?
  \item Converter $2.454545\ldots $ e $0.123123123\ldots $ em
        frações em termos mais baixos.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The dictionary completed.}
\begin{enumerate}[resume]
  \item As questões 6 a 10 sugerem um dicionário: um bloco de $ k $
        dígitos repetidos a partir do \omterm{def:g4:decimals:point}{ponto decimal} é igual a esse bloco
        acabou $ k $ noves ($0.727272\ldots = \frac{72}{99}$). Usar
        o dicionário para prever $0.037037\ldots $ como um \omterm{def:g9:fractions:fraction}{fração},
        simplificar --- e maravilhar-se: o que é surpreendentemente simples
        \omterm{def:g9:fractions:fraction}{fração} canta ``$037$''? Verifique por divisão.
  \item Combine o dicionário com uma mudança: escreva
        $0.0272727\ldots $ como um \omterm{def:g9:fractions:fraction}{fração} em termos mais baixos.
  \item O número $0.101001000100001\ldots $ (um $1$, então um
        $0$, então um $1$, então dois $0$ s, então um $1$, então três
        $0$ s, e assim por diante) nunca para. É uma dízima repetida?
        O que a Parte I diz sobre isso --- pode ser um
        \omterm{def:g9:fractions:fraction}{fração}? (Você acabou de conhecer seu primeiro, provavelmente
        \emph{irracional} número; o mais famoso é pego
        em \cref{ch:g9:sqrt}.)
  \item Aproximadamente quantos dígitos tem $2^{30}$ ter? Usar
        $2^{10} = 1\,024 \approx 1.024 \times 10^3$ e o
        \omterm{def:g8:powers:def}{expoente} regras (\cref{thm:g9:fractions:rules}) para escrever
        $2^{30}$ em (aproximadamente) notação científicae
        concluir. (Comparar \cref{exo:g9:fractions:10}.)
  \item Final: escrever $0.20262026\ldots$ como um fraçãoe
        declarar a moral dupla de todo o problema: qual
        escritas decimais são frações, e qual frações tem
        quais escritas decimais?
\end{enumerate}
\end{problem}
