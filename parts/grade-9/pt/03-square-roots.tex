\chapter{Raízes Quadradas}\label{ch:g9:sqrt}

Quanto mede a diagonal de um quadrado de lado $1$? O teorema de Pitágoras
respostas $\sqrt 2$, um número cujo quadrado é $2$ --- e o que acontece
não ser um \omterm{def:g9:fractions:fraction}{fração}. Este capítulo define raízes quadradas, estabelece o
regras para calcular com eles e ensina como simplificar expressões
gosto $\sqrt{75}$.

\section{Definição e primeiras propriedades}

\begin{definition}[Raiz quadrada]\label{def:g9:sqrt:def}
Deixar $ a \geq 0$. O \emph{raiz quadrada}\index{raiz quadrada} de $ a $, escrito
$\sqrt a $, é o único \emph{não negativo} número cujo quadrado é $ a $:
\[
\sqrt a \geq 0
\qquad\text{and}\qquad
\left(\sqrt a\right)^2 = a .
\]
Os números negativos não têm raiz quadrada, pois todo quadrado é não negativo.
\end{definition}

\begin{example}
$\sqrt{49} = 7$, $\sqrt 0 = 0$, $\sqrt 1 = 1$,
$\sqrt{2.25} = 1.5$. O primeiro raízes quadradas saber de cor são aqueles de
o \emph{quadrados perfeitos}: $1, 4, 9, 16, 25, 36, 49, 64, 81, 100, 121,
144$.
\end{example}

\begin{proposition}[Raiz quadrada de um quadrado]\label{prop:g9:sqrt:square}
Para cada número real $ x $ (positivo ou não):
\[
\sqrt{x^2} = \abs{x} =
\begin{cases}
x & \text{if } x \geq 0,\\
-x & \text{if } x < 0 .
\end{cases}
\]
\end{proposition}

\begin{proof}
O número $\abs{x}$ é não negativo e seu quadrado é $ x^2$ (um número e
seu oposto tem o mesmo quadrado). Sendo o único número não negativo
de quadrado $ x^2$, isso é $\sqrt{x^2}$.
\end{proof}

\begin{example}
$\sqrt{(-5)^2} = \sqrt{25} = 5 = \abs{-5}$: o \omterm{def:g9:sqrt:def}{raiz quadrada} ``esquece''
o sinal, não o restaura. Em particular $\sqrt{x^2} = x $ é
\emph{errado} para negativo $ x $.
\end{example}

\section{Produtos e quocientes de raízes quadradas}

\begin{theorem}[Regras de multiplicação e divisão]\label{thm:g9:sqrt:rules}
Para todos $ a \geq 0$ e $ b \geq 0$:
\[
\sqrt{a b} = \sqrt a \times \sqrt b,
\qquad\text{and for } b > 0:\quad
\sqrt{\frac ab} = \frac{\sqrt a}{\sqrt b} .
\]
\end{theorem}

\begin{proof}
O número $\sqrt a \times \sqrt b $ é não negativo (\omterm{def:g2:mult:def}{produto} de
números não negativos), e seu quadrado é
\[
\left(\sqrt a \times \sqrt b\right)^2
= \left(\sqrt a\right)^2 \times \left(\sqrt b\right)^2
= a b .
\]
Pela unicidade do número não negativo cujo quadrado é $ ab $, é igual
$\sqrt{ab}$. O \omterm{def:g3:division:remainder}{quociente} regra é provada da mesma maneira.
\end{proof}

\begin{remark}[Não existe tal regra para somas!]
$\sqrt{a + b}$ é \emph{não} $\sqrt a + \sqrt b $ em geral:
\[
\sqrt{9 + 16} = \sqrt{25} = 5,
\qquad\text{but}\qquad
\sqrt 9 + \sqrt{16} = 3 + 4 = 7 .
\]
\end{remark}

\begin{method}[Simplificando $\sqrt n $]\label{met:g9:sqrt:simplify}
Para simplificar o \omterm{def:g9:sqrt:def}{raiz quadrada} de um número inteiro:
\begin{enumerate}
  \item encontre o maior quadrado perfeito dividindo $ n $ (fator $ n $ se
        necessário);
  \item \omterm{def:g9:arith:divisor}{dividir} com o \omterm{def:g2:mult:def}{produto} regra e extraia o quadrado:
        $\sqrt{k^2 m} = k\sqrt m $;
  \item verifique se não há quadrado perfeito divide o que resta sob o
        raiz.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:sqrt:simplify}
Simplificar $\sqrt{75}$: desde $75 = 25 \times 3$,
\[
\sqrt{75} = \sqrt{25 \times 3} = \sqrt{25} \times \sqrt 3 = 5\sqrt3 .
\]
Simplificar $\sqrt{72}$: desde $72 = 36 \times 2$,
$\sqrt{72} = 6\sqrt2$. E um \omterm{def:g3:division:remainder}{quociente}:
$\sqrt{\dfrac{49}{16}} = \dfrac{\sqrt{49}}{\sqrt{16}} = \dfrac74$.
\end{example}

\begin{example}[Adicionando raízes quadradas]\label{ex:g9:sqrt:add}
\omterm{def:g1:addition:def}{Somas} de raízes quadradas simplifique apenas quando as raízes forem \emph{parecido}.
Calcular $\sqrt{12} + \sqrt{27} - \sqrt{48}$, simplificando cada termo primeiro:
\begin{align*}
\sqrt{12} &= \sqrt{4 \times 3} = 2\sqrt3, \\
\sqrt{27} &= \sqrt{9 \times 3} = 3\sqrt3, \\
\sqrt{48} &= \sqrt{16 \times 3} = 4\sqrt3,
\end{align*}
então o \omterm{def:g1:addition:def}{soma} é $2\sqrt3 + 3\sqrt3 - 4\sqrt3 = (2 + 3 - 4)\sqrt3 = \sqrt3$.
\end{example}

\begin{example}[Expandindo com raízes quadradas]\label{ex:g9:sqrt:expand}
As identidades da álgebra se aplicam a raízes quadradas. Expandir
$\left(\sqrt5 + 2\right)^2$ com $(a+b)^2 = a^2 + 2ab + b^2$:
\[
\left(\sqrt5 + 2\right)^2 = 5 + 2 \times 2\sqrt5 + 4 = 9 + 4\sqrt5 .
\]
E com a terceira identidade, $(a+b)(a-b) = a^2 - b^2$:
\[
\left(\sqrt7 + \sqrt3\right)\left(\sqrt7 - \sqrt3\right) = 7 - 3 = 4 :
\]
o \omterm{def:g2:mult:def}{produto} de dois números irracionais pode ser um número inteiro.
\end{example}

\section{A equação \texorpdfstring{$ x^2 = a $}{x2 = a}}

\begin{theorem}[Resolvendo $ x^2 = a $]\label{thm:g9:sqrt:equation}
\begin{enumerate}
  \item Se $ a > 0$, o \omterm{def:g8:equations:def}{equação} $ x^2 = a $ tem exatamente duas soluções:
        $\sqrt a $ e $-\sqrt a $;
  \item se $ a = 0$, a única solução é $0$;
  \item se $ a < 0$, não há solução.
\end{enumerate}
\end{theorem}

\begin{proof}
Reescrever $ x^2 = a $ como $ x^2 - \left(\sqrt a\right)^2 = 0$ quando $ a \geq 0$,
e fatorar como \omterm{ex:g1:subtraction:difference}{diferença} de quadrados:
\[
\left(x - \sqrt a\right)\left(x + \sqrt a\right) = 0 ,
\]
então $ x = \sqrt a $ ou $ x = -\sqrt a $ (estes coincidem quando $ a = 0$). Quando
$ a < 0$ não há solução, pois $ x^2 \geq 0 > a $ para cada $ x $.
\end{proof}

\begin{example}
$ x^2 = 16$: soluções $4$ e $-4$. \quad
$ x^2 = 5$: soluções $\sqrt5$ e $-\sqrt5$ (valores exatos; $\sqrt5
\approx 2.236$). \quad
$ x^2 = -9$: nenhuma solução. \quad
$3x^2 = 21$: divida por 3 primeiro, $ x^2 = 7$, soluções $\pm\sqrt7$.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g9:sqrt:1}
Calcule sem calculadora:
\[
\sqrt{64}, \qquad \sqrt{100}, \qquad \sqrt{0.09}, \qquad
\sqrt{\tfrac{1}{25}}, \qquad \left(\sqrt{13}\right)^2, \qquad
\sqrt{(-4)^2} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:sqrt:2}
Simplificar:
\[
\sqrt{50}, \qquad \sqrt{45}, \qquad \sqrt{98}, \qquad \sqrt{300} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:sqrt:3}
Calcule e simplifique:
\[
\sqrt2 \times \sqrt{18}, \qquad
\frac{\sqrt{75}}{\sqrt3}, \qquad
\sqrt5 \times \sqrt{20}, \qquad
\frac{\sqrt{8}}{\sqrt{50}} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:sqrt:4}
Resolver: $ x^2 = 36$; \quad $ x^2 = 11$; \quad $ x^2 + 4 = 0$; \quad
$2x^2 = 50$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g9:sqrt:5}
Reduza para um único termo: $\sqrt{20} + \sqrt{45}$, então
$3\sqrt8 - \sqrt{18} + \sqrt2$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:sqrt:6}
Expanda e simplifique:
\[
\left(\sqrt3 + 1\right)^2, \qquad
\left(2\sqrt5 - 3\right)^2, \qquad
\left(\sqrt6 + \sqrt2\right)\left(\sqrt6 - \sqrt2\right).
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:sqrt:7}
Um campo quadrado tem \omterm{def:g6:measure:area}{área} $6400$ eu $^2$. Qual é o comprimento do seu lado? De
sua diagonal (valor exato, depois \omterm{def:g4:numbers:round}{arredondado} para o metro mais próximo)?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:sqrt:8}
Mostre isso $\dfrac{1}{\sqrt2} = \dfrac{\sqrt2}{2}$ (multiplicar \omterm{def:g4:fractions:def}{numerador} e
\omterm{def:g4:fractions:def}{denominador} por $\sqrt2$). Use o mesmo truque para escrever $\dfrac{6}{\sqrt3}$
e $\dfrac{10}{\sqrt5}$ sem um \omterm{def:g9:sqrt:def}{raiz quadrada} no \omterm{def:g4:fractions:def}{denominador}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g9:sqrt:9}
Verdadeiro ou falso? Justifique com uma prova ou contra-exemplo.
\begin{enumerate}
  \item Para todos $ a, b \geq 0$: $\sqrt{ab} = \sqrt a \sqrt b $.
  \item Para todos $ a, b \geq 0$: $\sqrt{a+b} = \sqrt a + \sqrt b $.
  \item Para todos $ x $: $\sqrt{x^2} = x $.
  \item $\left(3\sqrt2\right)^2 = 18$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g9:sqrt:10}
Deixar $ x = \sqrt{7 + 4\sqrt3}$. Calcular
$\left(2 + \sqrt3\right)^2$, e deduza uma expressão mais simples de $ x $.
A mesma pergunta para $\sqrt{9 - 4\sqrt5}$ (aponte para um quadrado da forma
$\left(\sqrt5 - 2\right)^2$ e preste atenção ao sinal).
\end{exercise}

\section{Problema: O número que não é uma fração}

\begin{problem}[{Problema de fim de semana --- a prova lendária de que
$\sqrt2$ é irracional, seus muitos primos e o estranhamente
boa receita para aproximar isso}]
\label{pb:g9:sqrt:1}
A diagonal de um quadrado unitário tem um comprimento perfeitamente real --- você
desenhou \cref{ex:g8:pythagoras:diagonal} --- e ainda assim, como o
Os pitagóricos descobriram, para seu horror, cerca de vinte e cinco
séculos atrás, \emph{não \omterm{def:g9:fractions:fraction}{fração} de jeito nenhum} mede isso. Lenda
afirma que a descoberta foi punida com afogamento. Este problema caminha
você através da prova imortal (quatro perguntas e é sua
para o resto da vida), multiplica as vítimas e termina com a receita
engenheiros usaram por dois mil anos para domar o indomável
número.

\medskip
\noindent\textbf{Part I --- The proof.}
\begin{enumerate}
  \item Primeiro procure-o: verifique isso
        $1.4 < \sqrt2 < 1.5$, então isso $1.41 < \sqrt2 < 1.42$,
        quadrando os limites. Mais um dígito: entre os quais
        números de três decimais fazem $\sqrt2$ mentira?
  \item Agora suponhamos --- por uma questão de contradição --- que
        $\sqrt2 = \frac ab $ para alguns \omterm{def:g9:fractions:fraction}{fração} \emph{no mais baixo
        termos} (\cref{met:g9:arith:simplify}). Quadrar ambos os lados
        e mostre isso $ a^2 = 2 b^2$. Qual é a paridade de
        $ a^2$?
  \item Os fatos de paridade de \cref{pb:g8:pythagoras:1} dizer: números \omterm{def:g3:numbers:evenodd}{ímpares} ter \omterm{def:g3:numbers:evenodd}{ímpar} quadrados. Deduza isso $ a $ é \omterm{def:g3:numbers:evenodd}{par}, escrever
        $ a = 2k $, substitua --- e mostre que $ b $ deve ser \omterm{def:g3:numbers:evenodd}{par}
        também.
  \item Onde está a contradição? Conclua e indique o
        teorema completo: \emph{$\sqrt2$ é irracional --- é
        não \omterm{def:g3:division:remainder}{quociente} de números inteiros}.
  \item A prova apoiou-se uma vez, discretamente, nas palavras “em
        termos mais baixos''. Aponte para a etapa exata que
        entrar em colapso sem eles e explicar por que assumir o menor
        termos era legítimo em primeiro lugar.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The victims multiply.}
\begin{enumerate}[resume]
  \item Um segundo estilo de prova, por meio de fatorações primárias
        (\cref{thm:g9:arith:factorization}): em $ a^2 = 3b^2$,
        compare a paridade do \emph{\omterm{def:g8:powers:def}{expoente} de $3$} em cada
        lado (\cref{pb:g9:arith:1}: quadrados carregam \omterm{def:g3:numbers:evenodd}{par}
        \omterm{def:g8:powers:def}{expoentes}). Conclua isso $\sqrt3$ é irracional.
  \item Execute o mesmo \omterm{def:g8:powers:def}{expoente} argumento sobre $ a^2 = n b^2$ por um
        número inteiro geral $ n $: para o qual $ n $ isso produz
        uma contradição, e em que falha? Indique o
        resultado completo: $\sqrt n $ é irracional exatamente
        quando\,\dots
  \item Prove o lema do escudo: um racional diferente de \omterm{def:g1:counting:zero}{zero} vezes um
        irracional é irracional. (Suponha $ r x = q $ com $ r, q $
        racional, $ r \neq 0$, e resolva para $ x $.) Deduza isso
        $2\sqrt2$ e $\frac{\sqrt2}{2}$ são irracionais.
  \item Prove isso $1 + \sqrt2$ é irracional. Então a linda
        um: supondo $ s = \sqrt2 + \sqrt3$ eram racionais,
        calcular $ s^2$, isolar $\sqrt6$, e encontre o
        contradição.
  \item Modere o entusiasmo: dê dois números irracionais cujos
        \emph{\omterm{def:g1:addition:def}{soma}} é racional, e dois cujos \emph{\omterm{def:g2:mult:def}{produto}} é
        racional. (A irracionalidade não é preservada pela aritmética
        --- cada caso precisa de sua própria prova.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Heron's recipe.}
Dois mil anos antes das calculadoras, Garça de Alexandria
aproximado $\sqrt2$ assim: \emph{adivinhar $ x $; substitua o
adivinhe pela média de $ x $ e $\frac2x $; repita.}
\begin{enumerate}[resume]
  \item Comece pelo palpite $ x = 1$ e calcule os próximos dois
        adivinha tão exato frações.
  \item Calcule a terceira estimativa, novamente como uma estimativa exata fraçãoe
        seu valor decimal. Compare com a pergunta 1: quantos
        decimais de $\sqrt2$ já estão corretos?
  \item A ideia por trás da receita: um retângulo de \omterm{def:g6:measure:area}{área} $2$ com
        um lado $ x $ tem o outro lado $\frac2x $. Mostre isso
        $\sqrt2$ sempre mente \emph{entre} $ x $ e $\frac2x $
        (considere os dois casos $ x^2 > 2$ e $ x^2 < 2$), então
        calcular a média dos dois lados comprime o retângulo em direção
        a praça.
  \item Os palpites $\frac32$, $\frac{17}{12}$,
        $\frac{577}{408}$ esconder uma joia: computar $3^2 - 2 \times
        2^2$, então $17^2 - 2 \times 12^2$, então
        $577^2 - 2 \times 408^2$. Parte que provei
        $ a^2 - 2b^2 = 0$ impossível --- quão perto está o Heron
        frações chegar ao impossível?
  \item Finale, em duas frases: a diagonal da unidade
        quadrado existe no papel, não \omterm{def:g9:fractions:fraction}{fração} mede-o, e seu
        escrita decimal nunca pode repetir
        (\cref{pb:g9:fractions:1}). Que tipo de “novos números”
        a reta numérica deve, portanto, conter --- e onde em
        esta série tem a história completa contada?
\end{enumerate}
\end{problem}
