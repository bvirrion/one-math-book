\chapter{Solids and Volumes}\label{ch:g9:solids}

Space geometry begins with a small family of \omterm{def:g5:solids:def}{solids} --- \omterm{def:g7:areas:prism}{prisms},
\omterm{def:g7:areas:prism}{cylinders}, \omterm{def:g8:solids:def}{pyramids}, \omterm{def:g8:solids:def}{cones}, spheres --- and two questions: how much do
they hold (\omterm{def:g6:measure:volume}{volume}), and what do you see when you slice them (sections)?
The chapter closes with a fact of great practical importance: scaling a
\omterm{def:g5:solids:def}{solid} by $k$ multiplies its \omterm{def:g6:measure:volume}{volume} by $k^3$.

\section{The classical solids and their volumes}

\begin{theorem}[Volume formulas]\label{thm:g9:solids:volumes}
Write $B$ for the \omterm{def:g6:measure:area}{area} of the base and $h$ for the height (the distance
between the base and the opposite \omterm{def:g5:solids:def}{face} or apex).
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{l|l}
\omterm{def:g5:solids:def}{solid} & \omterm{def:g6:measure:volume}{volume} \\
\hline
\omterm{def:g7:areas:prism}{prism}, \omterm{def:g7:areas:prism}{cylinder} & $V = B \times h$ \\[5pt]
\omterm{def:g8:solids:def}{pyramid}, \omterm{def:g8:solids:def}{cone} & $V = \dfrac13\, B \times h$ \\[5pt]
sphere of \omterm{def:g3:shapes:shapes}{radius} $r$ & $V = \dfrac43\,\pi r^3$ \\
\end{tabular}
\end{center}
For a \omterm{def:g7:areas:prism}{cylinder} and a \omterm{def:g8:solids:def}{cone} of \omterm{def:g3:shapes:shapes}{radius} $r$, the base \omterm{def:g6:measure:area}{area} is $B = \pi r^2$;
the sphere's surface \omterm{def:g6:measure:area}{area} is $4\pi r^2$.
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:volumes}
A cylindrical can has \omterm{def:g3:shapes:shapes}{radius} $4$ cm and height $10$ cm:
\[
V = \pi r^2 h = \pi \times 16 \times 10 = 160\pi \approx 503 \text{ cm}^3
\approx 0.5 \text{ L}.
\]
A \omterm{def:g8:solids:def}{cone} with the same base and height holds one third of that:
$\frac{160\pi}{3} \approx 168$ cm$^3$. A sphere of \omterm{def:g3:shapes:shapes}{radius} $4$ cm:
$V = \frac43 \pi \times 64 = \frac{256\pi}{3} \approx 268$ cm$^3$.
\end{example}

\begin{example}[A pyramid step by step]\label{ex:g9:solids:pyramid}
A \omterm{def:g8:solids:def}{pyramid} has a square base of side $6$ m and height $5$ m.
\begin{enumerate}
  \item Base \omterm{def:g6:measure:area}{area}: $B = 6^2 = 36$ m$^2$.
  \item \omterm{def:g6:measure:volume}{Volume}: $V = \frac13 \times 36 \times 5 = 60$ m$^3$.
\end{enumerate}
Mind the units: if the side were given in cm and the height in m, one of
them would have to be converted first.
\end{example}

\section{Sections by planes}

\begin{proposition}[Sections of the classical solids]\label{prop:g9:solids:sections}
Cutting a \omterm{def:g5:solids:def}{solid} by a plane produces a flat figure, its
\emph{cross-section}\index{cross-section}:
\begin{itemize}
  \item a \omterm{def:g7:areas:prism}{prism} or \omterm{def:g7:areas:prism}{cylinder} cut \emph{\omterm{def:g6:lines:perp}{parallel} to its base} gives a copy
        of the base, at every height;
  \item a \omterm{def:g8:solids:def}{pyramid} or \omterm{def:g8:solids:def}{cone} cut \omterm{def:g6:lines:perp}{parallel} to its base gives a
        \emph{reduction} of the base: at distance $d$ from the apex,
        the \omterm{def:g9:thales:scaling}{scale factor} is $k = \frac{d}{h}$;
  \item a sphere of \omterm{def:g3:shapes:shapes}{radius} $r$ cut by a plane at distance $d < r$ from
        the center gives a circle of \omterm{def:g3:shapes:shapes}{radius} $\sqrt{r^2 - d^2}$ (by the
        Pythagorean theorem).
\end{itemize}
\end{proposition}
\admitted

\begin{example}\label{ex:g9:solids:section}
A \omterm{def:g8:solids:def}{cone} has base \omterm{def:g3:shapes:shapes}{radius} $6$ and height $9$. The section by a plane
\omterm{def:g6:lines:perp}{parallel} to the base, at distance $3$ from the apex, is a disk scaled by
$k = \frac39 = \frac13$: its \omterm{def:g3:shapes:shapes}{radius} is $6 \times \frac13 = 2$.

A sphere of \omterm{def:g3:shapes:shapes}{radius} $5$ is cut by a plane at distance $3$ from its
center: the section is a circle of \omterm{def:g3:shapes:shapes}{radius} $\sqrt{25 - 9} = 4$.
\end{example}

\section{Scaling solids}

\begin{theorem}[Effect of a scaling on lengths, areas, volumes]\label{thm:g9:solids:scaling}
When a \omterm{def:g5:solids:def}{solid} is enlarged or reduced by the \omterm{def:g9:thales:scaling}{scale factor} $k > 0$:
\begin{itemize}
  \item all lengths are multiplied by $k$;
  \item all \omterm{def:g6:measure:area}{areas} are multiplied by $k^2$;
  \item all \omterm{def:g6:measure:volume}{volumes} are multiplied by $k^3$.
\end{itemize}
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:scaling}
A model car at \omterm{def:g7:prop:scale}{scale} $\frac{1}{18}$: lengths are those of the real car
multiplied by $k = \frac{1}{18}$, painted surface multiplied by
$k^2 = \frac{1}{324}$, and \omterm{def:g6:measure:volume}{volume} by $k^3 = \frac{1}{5832}$.

Doubling the \omterm{def:g3:shapes:shapes}{radius} of a sphere ($k = 2$) multiplies its \omterm{def:g6:measure:volume}{volume} by
$2^3 = 8$: check on the formula,
$\frac43\pi (2r)^3 = \frac43 \pi \times 8r^3$.
\end{example}

\begin{example}[Truncated cone]\label{ex:g9:solids:truncated}
A \omterm{def:g8:solids:def}{cone} of height $9$ and base \omterm{def:g3:shapes:shapes}{radius} $6$ (\omterm{def:g6:measure:volume}{volume}
$\frac13 \pi \times 36 \times 9 = 108\pi$) is cut at distance $3$ from
the apex, and the small \omterm{def:g8:solids:def}{cone} above the cut is removed. The small \omterm{def:g8:solids:def}{cone} is
the reduction by $k = \frac13$, so its \omterm{def:g6:measure:volume}{volume} is
$108\pi \times \left(\frac13\right)^3 = 4\pi$. The remaining \omterm{def:g5:solids:def}{solid} (a
\emph{truncated \omterm{def:g8:solids:def}{cone}}) has \omterm{def:g6:measure:volume}{volume} $108\pi - 4\pi = 104\pi \approx 327$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g9:solids:1}
Compute the \omterm{def:g6:measure:volume}{volumes}: a \omterm{def:g5:solids:def}{box} (rectangular \omterm{def:g7:areas:prism}{prism}) of dimensions $4 \times 5
\times 12$; a \omterm{def:g7:areas:prism}{cylinder} of \omterm{def:g3:shapes:shapes}{radius} $3$ and height $7$ (exact value with
$\pi$, then rounded to the unit).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:solids:2}
Compute the \omterm{def:g6:measure:volume}{volume} of a \omterm{def:g8:solids:def}{cone} of \omterm{def:g3:shapes:shapes}{radius} $5$ and height $12$, and of a
\omterm{def:g8:solids:def}{pyramid} with rectangular base $8 \times 3$ and height $10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:solids:3}
Compute the \omterm{def:g6:measure:volume}{volume} and the surface \omterm{def:g6:measure:area}{area} of a sphere of \omterm{def:g3:shapes:shapes}{radius} $6$ (exact
values with $\pi$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:solids:4}
A sphere of \omterm{def:g3:shapes:shapes}{radius} $10$ is cut by a plane at distance $8$ from its
center. What is the \omterm{def:g3:shapes:shapes}{radius} of the section circle?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:5}
A \omterm{def:g8:solids:def}{cone} has base \omterm{def:g3:shapes:shapes}{radius} $8$ and height $12$. A plane \omterm{def:g6:lines:perp}{parallel} to the base
cuts it at distance $9$ from the apex. Compute the \omterm{def:g3:shapes:shapes}{radius} of the
section, then the \omterm{def:g6:measure:volume}{volume} of the small \omterm{def:g8:solids:def}{cone} above the cut.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:6}
A cylindrical glass of inner \omterm{def:g3:shapes:shapes}{radius} $3$ cm contains water to a height of
$10$ cm. A ball of \omterm{def:g3:shapes:shapes}{radius} $2$ cm is fully submerged in it. By how much
does the water level rise? (The added \omterm{def:g6:measure:volume}{volume} spreads over the glass's
base \omterm{def:g6:measure:area}{area}; give the exact rise, then \omterm{def:g4:numbers:round}{round} to the millimeter.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:7}
A recipe fills a spherical mold of \omterm{def:g3:shapes:shapes}{radius} $6$ cm. You only have a
spherical mold of \omterm{def:g3:shapes:shapes}{radius} $3$ cm. How many small molds can you fill?
(Answer without computing either \omterm{def:g6:measure:volume}{volume} explicitly.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:8}
The Great \omterm{def:g8:solids:def}{Pyramid} of Giza has a square base of side about $230$ m and a
height of about $147$ m. Estimate its \omterm{def:g6:measure:volume}{volume}, and express it in millions
of cubic meters.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:solids:9}
A cone-shaped funnel of \omterm{def:g3:shapes:shapes}{radius} $6$ cm and height $12$ cm, apex down, is
filled with water up to \omterm{def:g3:division:half}{half} of its \emph{height} (measured from the
apex).
\begin{enumerate}
  \item What \omterm{def:g9:fractions:fraction}{fraction} of the funnel's \omterm{def:g6:measure:volume}{volume} is filled?
  \item If instead it is filled with \omterm{def:g3:division:half}{half} of its \emph{\omterm{def:g6:measure:volume}{volume}} of water,
        show that the water height $d$ satisfies
        $\left(\frac{d}{12}\right)^3 = \frac12$, and give $d$ to the
        millimeter ($\sqrt[3]{0.5} \approx 0.7937$).
\end{enumerate}
\end{exercise}

\section{Problem: Archimedes' tombstone}

\begin{problem}[{Weekend problem --- the sphere is two thirds of
its cylinder (twice over), the 1:2:3 stack, and the square--cube
law that forbids giants}]
\label{pb:g9:solids:1}
Archimedes proved many theorems, but one made him so proud that
he asked for its \emph{figure} to be carved on his tomb: a sphere
nested in its tightest \omterm{def:g7:areas:prism}{cylinder}. A century later the Roman writer
Cicero, searching the brambles near Syracuse, recognized the
grave ``by the sphere and the \omterm{def:g7:areas:prism}{cylinder}''. This problem retrieves
what the carving encodes --- a double two-thirds miracle --- then
follows \omterm{def:g6:measure:volume}{volumes} and surfaces (\cref{thm:g9:solids:volumes},
\cref{thm:g9:solids:scaling}) to a law that governs giants, ants
and cooling planets.

\medskip
\noindent\textbf{Part I --- The carving decoded.}
A sphere of \omterm{def:g3:shapes:shapes}{radius} $r$ sits exactly inside a \omterm{def:g7:areas:prism}{cylinder}: same
\omterm{def:g3:shapes:shapes}{radius}, height $2r$.
\begin{enumerate}
  \item Compute the \omterm{def:g7:areas:prism}{cylinder}'s \omterm{def:g6:measure:volume}{volume} in terms of $r$.
  \item Compute the ratio of the sphere's \omterm{def:g6:measure:volume}{volume} to the
        \omterm{def:g7:areas:prism}{cylinder}'s. Why might Archimedes have liked that the
        answer contains no $\pi$ and no $r$?
  \item Now the surfaces: compare the sphere's surface \omterm{def:g6:measure:area}{area} with
        the \omterm{def:g7:areas:prism}{cylinder}'s \emph{total} surface (lateral part plus
        the two lids). What ratio appears --- again?
  \item The empty space between sphere and \omterm{def:g7:areas:prism}{cylinder} has \omterm{def:g6:measure:volume}{volume}
        $2\pi r^3 - \frac43\pi r^3$. Show that this leftover
        exactly equals the \omterm{def:g6:measure:volume}{volume} of \emph{two \omterm{def:g8:solids:def}{cones}} of \omterm{def:g3:shapes:shapes}{radius}
        $r$ and height $r$.
  \item A basketball of \omterm{def:g3:shapes:shapes}{radius} $12$ cm is sold in the tightest
        cylindrical \omterm{def:g5:solids:def}{box}. Compute both \omterm{def:g6:measure:volume}{volumes} (to the nearest
        $100$ cm$^3$) and the percentage of the \omterm{def:g5:solids:def}{box} that is
        empty.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Bowls, molds and moons.}
\begin{enumerate}[resume]
  \item A hemispherical bowl has \omterm{def:g3:shapes:shapes}{radius} $10$ cm. Compute its
        \omterm{def:g2:measure:units}{capacity}, in cm$^3$ and in liters (to the deciliter).
  \item The 1:2:3 stack: a \omterm{def:g8:solids:def}{cone}, a hemisphere and a \omterm{def:g7:areas:prism}{cylinder},
        all of \omterm{def:g3:shapes:shapes}{radius} $10$ cm and height $10$ cm. Compute the
        three \omterm{def:g6:measure:volume}{volumes} and verify the legendary proportion
        $1 : 2 : 3$. (Archimedes would have appreciated this
        one, too.)
  \item A chocolate sphere of \omterm{def:g3:shapes:shapes}{radius} $3$ cm is melted into a
        cylindrical mold of \omterm{def:g3:shapes:shapes}{radius} $3$ cm. What height does the
        chocolate reach? (Exact \omterm{def:g9:fractions:fraction}{fraction}, then to the
        millimeter.)
  \item The Earth's \omterm{def:g3:shapes:shapes}{radius} is about $6\,371$ km, the Moon's
        about $1\,737$ km. Compute the ratio of the radii, then
        --- with \cref{thm:g9:solids:scaling} --- the ratio of
        the \omterm{def:g6:measure:volume}{volumes}: how many Moons would fit into a hollow
        Earth, by \omterm{def:g6:measure:volume}{volume}?
  \item From the same theorem: what is the Earth--Moon ratio of
        \emph{surface \omterm{def:g6:measure:area}{areas}}? And in general, when a balloon's
        \omterm{def:g3:shapes:shapes}{radius} doubles, what happens to its \omterm{def:g6:measure:volume}{volume} and to its
        surface? State the \emph{square--cube law}: as a shape
        \omterm{def:g7:prop:scale}{scales} up, \omterm{def:g6:measure:volume}{volumes} outrun surfaces.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The square--cube law rules the
world.}
\begin{enumerate}[resume]
  \item Galileo's argument against giants: imagine a human
        scaled up by a factor $10$, same proportions. By what
        factor does the weight grow (weight follows \omterm{def:g6:measure:volume}{volume})? By
        what factor does the \omterm{prop:g9:solids:sections}{cross-section} of the bones grow
        (an \omterm{def:g6:measure:area}{area})? By what factor, then, does the \emph{pressure}
        on each square centimeter of bone grow --- and what
        happens to the giant?
  \item The same law in reverse explains ant heroics: strength
        follows muscle \omterm{prop:g9:solids:sections}{cross-section} (an \omterm{def:g6:measure:area}{area}), weight follows
        \omterm{def:g6:measure:volume}{volume}. For an animal $100$ times smaller in every
        direction, by what factors do weight and strength
        shrink, and by what factor does the strength-to-weight
        ratio improve?
  \item Heat is produced by the body's \omterm{def:g6:measure:volume}{volume} and lost through
        its surface. Show that for a sphere the
        surface-to-volume ratio is $\frac{3}{r}$, and compute it
        for $r = 1$ and $r = 2$. Which cools faster, a mouse or
        a bear --- and why do babies need hats in winter?
  \item Two spherical oranges of radii $4$ cm and $5$ cm are
        sold at the same price. Compute the ratio of their
        \omterm{def:g6:measure:volume}{volumes}: how much more orange does the big one give for
        the same money?
  \item Finale: describe precisely what the tombstone figure
        encodes --- the two ``two thirds'' of questions 2 and 3
        --- and answer Cicero in one sentence: why would a
        mathematician choose, over every conquest of his
        engineering genius, a sphere in a \omterm{def:g7:areas:prism}{cylinder}?
\end{enumerate}
\end{problem}
