\chapter{Solids and Volumes}\label{ch:g9:solids}

Space geometry begins with a small family of solids --- prisms,
cylinders, pyramids, cones, spheres --- and two questions: how much do
they hold (volume), and what do you see when you slice them (sections)?
The chapter closes with a fact of great practical importance: scaling a
solid by $k$ multiplies its volume by $k^3$.

\section{The classical solids and their volumes}

\begin{theorem}[Volume formulas]\label{thm:g9:solids:volumes}
Write $B$ for the area of the base and $h$ for the height (the distance
between the base and the opposite face or apex).
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{l|l}
solid & volume \\
\hline
prism, cylinder & $V = B \times h$ \\
pyramid, cone & $V = \dfrac13\, B \times h$ \\
sphere of radius $r$ & $V = \dfrac43\,\pi r^3$ \\
\end{tabular}
\end{center}
For a cylinder and a cone of radius $r$, the base area is $B = \pi r^2$;
the sphere's surface area is $4\pi r^2$.
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:volumes}
A cylindrical can has radius $4$ cm and height $10$ cm:
\[
V = \pi r^2 h = \pi \times 16 \times 10 = 160\pi \approx 503 \text{ cm}^3
\approx 0.5 \text{ L}.
\]
A cone with the same base and height holds one third of that:
$\frac{160\pi}{3} \approx 168$ cm$^3$. A sphere of radius $4$ cm:
$V = \frac43 \pi \times 64 = \frac{256\pi}{3} \approx 268$ cm$^3$.
\end{example}

\begin{example}[A pyramid step by step]\label{ex:g9:solids:pyramid}
A pyramid has a square base of side $6$ m and height $5$ m.
\begin{enumerate}
  \item Base area: $B = 6^2 = 36$ m$^2$.
  \item Volume: $V = \frac13 \times 36 \times 5 = 60$ m$^3$.
\end{enumerate}
Mind the units: if the side were given in cm and the height in m, one of
them would have to be converted first.
\end{example}

\section{Sections by planes}

\begin{proposition}[Sections of the classical solids]\label{prop:g9:solids:sections}
Cutting a solid by a plane produces a flat figure, its
\emph{cross-section}\index{cross-section}:
\begin{itemize}
  \item a prism or cylinder cut \emph{parallel to its base} gives a copy
        of the base, at every height;
  \item a pyramid or cone cut parallel to its base gives a
        \emph{reduction} of the base: at distance $d$ from the apex,
        the scale factor is $k = \frac{d}{h}$;
  \item a sphere of radius $r$ cut by a plane at distance $d < r$ from
        the center gives a circle of radius $\sqrt{r^2 - d^2}$ (by the
        Pythagorean theorem).
\end{itemize}
\end{proposition}
\admitted

\begin{example}\label{ex:g9:solids:section}
A cone has base radius $6$ and height $9$. The section by a plane
parallel to the base, at distance $3$ from the apex, is a disk scaled by
$k = \frac39 = \frac13$: its radius is $6 \times \frac13 = 2$.

A sphere of radius $5$ is cut by a plane at distance $3$ from its
center: the section is a circle of radius $\sqrt{25 - 9} = 4$.
\end{example}

\section{Scaling solids}

\begin{theorem}[Effect of a scaling on lengths, areas, volumes]\label{thm:g9:solids:scaling}
When a solid is enlarged or reduced by the scale factor $k > 0$:
\begin{itemize}
  \item all lengths are multiplied by $k$;
  \item all areas are multiplied by $k^2$;
  \item all volumes are multiplied by $k^3$.
\end{itemize}
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:scaling}
A model car at scale $\frac{1}{18}$: lengths are those of the real car
multiplied by $k = \frac{1}{18}$, painted surface multiplied by
$k^2 = \frac{1}{324}$, and volume by $k^3 = \frac{1}{5832}$.

Doubling the radius of a sphere ($k = 2$) multiplies its volume by
$2^3 = 8$: check on the formula,
$\frac43\pi (2r)^3 = \frac43 \pi \times 8r^3$.
\end{example}

\begin{example}[Truncated cone]\label{ex:g9:solids:truncated}
A cone of height $9$ and base radius $6$ (volume
$\frac13 \pi \times 36 \times 9 = 108\pi$) is cut at distance $3$ from
the apex, and the small cone above the cut is removed. The small cone is
the reduction by $k = \frac13$, so its volume is
$108\pi \times \left(\frac13\right)^3 = 4\pi$. The remaining solid (a
\emph{truncated cone}) has volume $108\pi - 4\pi = 104\pi \approx 327$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g9:solids:1}
Compute the volumes: a box (rectangular prism) of dimensions $4 \times 5
\times 12$; a cylinder of radius $3$ and height $7$ (exact value with
$\pi$, then rounded to the unit).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:solids:2}
Compute the volume of a cone of radius $5$ and height $12$, and of a
pyramid with rectangular base $8 \times 3$ and height $10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:solids:3}
Compute the volume and the surface area of a sphere of radius $6$ (exact
values with $\pi$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:solids:4}
A sphere of radius $10$ is cut by a plane at distance $8$ from its
center. What is the radius of the section circle?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:5}
A cone has base radius $8$ and height $12$. A plane parallel to the base
cuts it at distance $9$ from the apex. Compute the radius of the
section, then the volume of the small cone above the cut.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:6}
A cylindrical glass of inner radius $3$ cm contains water to a height of
$10$ cm. A ball of radius $2$ cm is fully submerged in it. By how much
does the water level rise? (The added volume spreads over the glass's
base area; give the exact rise, then round to the millimeter.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:7}
A recipe fills a spherical mold of radius $6$ cm. You only have a
spherical mold of radius $3$ cm. How many small molds can you fill?
(Answer without computing either volume explicitly.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:8}
The Great Pyramid of Giza has a square base of side about $230$ m and a
height of about $147$ m. Estimate its volume, and express it in millions
of cubic meters.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:solids:9}
A cone-shaped funnel of radius $6$ cm and height $12$ cm, apex down, is
filled with water up to half of its \emph{height} (measured from the
apex).
\begin{enumerate}
  \item What fraction of the funnel's volume is filled?
  \item If instead it is filled with half of its \emph{volume} of water,
        show that the water height $d$ satisfies
        $\left(\frac{d}{12}\right)^3 = \frac12$, and give $d$ to the
        millimeter ($\sqrt[3]{0.5} \approx 0.7937$).
\end{enumerate}
\end{exercise}
