\chapter{Statistics and Probability}\label{ch:g9:statproba}

Data and chance are everywhere: sports results, weather forecasts,
games. This chapter teaches how to summarize a series of numbers by its
mean, median and range, and how to compute the probability of simple
random experiments --- including two-step experiments, handled with tree
diagrams. Both subjects are developed much further in the High School
volume of this series.

\section{Summarizing data}

\begin{definition}[Mean, median, range]\label{def:g9:statproba:indicators}
For a series of $N$ numbers:
\begin{itemize}
  \item the \emph{mean}\index{mean} is the sum of all the values divided
        by $N$;
  \item the \emph{median}\index{median} is a value splitting the sorted
        series in two halves of equal size: for $N$ odd, the middle
        value; for $N$ even, the midpoint of the two middle values;
  \item the \emph{range}\index{range} is the largest value minus the
        smallest.
\end{itemize}
\end{definition}

\begin{example}\label{ex:g9:statproba:indicators}
Marks of a student: $8,\ 12,\ 9,\ 15,\ 11$.
\begin{enumerate}
  \item Mean: $\dfrac{8 + 12 + 9 + 15 + 11}{5} = \dfrac{55}{5} = 11$.
  \item Median: sort the series: $8, 9, 11, 12, 15$; the middle (3rd)
        value is $11$.
  \item Range: $15 - 8 = 7$.
\end{enumerate}
With a sixth mark of $17$: sorted $8, 9, 11, 12, 15, 17$, the median
becomes the midpoint of $11$ and $12$, that is $11.5$, and the mean
becomes $\frac{72}{6} = 12$.
\end{example}

\begin{example}[Weighted mean]\label{ex:g9:statproba:weighted}
The shoe sizes sold in a day:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccccc}
size & $38$ & $39$ & $40$ & $41$ & $42$ \\
\hline
count & $3$ & $7$ & $6$ & $3$ & $1$ \\
\end{tabular}
\end{center}
The mean size is
\[
\frac{3 \times 38 + 7 \times 39 + 6 \times 40 + 3 \times 41 + 1 \times 42}
{3 + 7 + 6 + 3 + 1}
= \frac{114 + 273 + 240 + 123 + 42}{20}
= \frac{792}{20} = 39.6 .
\]
\end{example}

\begin{remark}
The mean and the median can differ a lot. In the series
$10, 10, 10, 10, 60$ the mean is $20$ (pulled up by the value $60$) while
the median is $10$. Always ask which indicator represents the situation
better.
\end{remark}

\section{Probability}

\begin{definition}[Probability]\label{def:g9:statproba:probability}
A random experiment has several possible \emph{outcomes}; an
\emph{event} is a set of outcomes. Each outcome gets a
\emph{probability}\index{probability}: a number between $0$ and $1$,
with all the probabilities of the outcomes summing to $1$; the
probability $\P(A)$ of an event $A$ is the sum of the probabilities of
its outcomes. When all $n$ outcomes are equally likely,
\[
\P(A) = \frac{\text{number of favorable outcomes}}{n}.
\]
\end{definition}

\begin{example}\label{ex:g9:statproba:spinner}
A wheel is split into $8$ equal sectors: $3$ red, $3$ blue, $2$ green.
Each sector has probability $\frac18$, so
\[
\P(\text{red}) = \frac38, \qquad
\P(\text{green}) = \frac28 = \frac14, \qquad
\P(\text{not green}) = 1 - \frac14 = \frac34 .
\]
\end{example}

\begin{proposition}[Complement]\label{prop:g9:statproba:complement}
For every event $A$, the \emph{contrary event} $\bar A$ (``$A$ does not
occur'') satisfies
\[
\P(\bar A) = 1 - \P(A).
\]
\end{proposition}

\begin{proof}
Every outcome is in exactly one of $A$ and $\bar A$, so
$\P(A) + \P(\bar A)$ sums the probabilities of all outcomes, which is
$1$.
\end{proof}

\section{Two-step experiments}

\begin{method}[Tree diagrams]\label{met:g9:statproba:tree}
For an experiment in two steps:
\begin{enumerate}
  \item draw one branch per outcome of the first step, then continue
        each branch with the outcomes of the second step, writing each
        probability on its branch;
  \item multiply the probabilities along a path to get the probability
        of that path;
  \item add the probabilities of all the paths forming an event.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:statproba:tree}
A bag holds $2$ red and $3$ black tokens. Draw one token, note its
color, \emph{put it back}, draw again. Each draw gives red with
probability $\frac25$, black with probability $\frac35$.

\begin{omfigure}
\begin{tikzpicture}[scale=0.95]
  \coordinate (root) at (0,0);
  \coordinate (R) at (2.2,1.0);
  \coordinate (B) at (2.2,-1.0);
  \coordinate (RR) at (4.7,1.55);
  \coordinate (RB) at (4.7,0.45);
  \coordinate (BR) at (4.7,-0.45);
  \coordinate (BB) at (4.7,-1.55);
  \draw (root) -- (R) node[midway, above, font=\small, sloped] {$\frac25$};
  \draw (root) -- (B) node[midway, below, font=\small, sloped] {$\frac35$};
  \draw (R) -- (RR) node[midway, above, font=\small, sloped] {$\frac25$};
  \draw (R) -- (RB) node[midway, below, font=\small, sloped] {$\frac35$};
  \draw (B) -- (BR) node[midway, above, font=\small, sloped] {$\frac25$};
  \draw (B) -- (BB) node[midway, below, font=\small, sloped] {$\frac35$};
  \node[omThm, font=\small, above left=-2pt] at (R) {R};
  \node[font=\small, below left=-2pt] at (B) {B};
  \node[right, font=\small] at (RR)
    {RR:\ $\frac25 \times \frac25 = \frac{4}{25}$};
  \node[right, font=\small] at (RB)
    {RB:\ $\frac25 \times \frac35 = \frac{6}{25}$};
  \node[right, font=\small] at (BR)
    {BR:\ $\frac35 \times \frac25 = \frac{6}{25}$};
  \node[right, font=\small] at (BB)
    {BB:\ $\frac35 \times \frac35 = \frac{9}{25}$};
\end{tikzpicture}

{\small The tree of the two draws with replacement: multiply along the
branches, and check that the four paths sum to $1$.}
\end{omfigure}

Probability of two tokens of the same color:
$\frac{4}{25} + \frac{9}{25} = \frac{13}{25}$. Probability of at least
one red: $1 - \P(\text{BB}) = 1 - \frac{9}{25} = \frac{16}{25}$.
\end{example}

\begin{remark}[Frequencies approach probabilities]
Rolling a fair die $6000$ times gives \emph{about} $1000$ sixes --- not
exactly. As the number of repetitions grows, observed frequencies get
closer and closer to the probabilities: this is what makes probability
the right tool to model repeated experiments (a story developed in
the High School volume of this series).
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g9:statproba:1}
Compute the mean, median and range of the series
$14,\ 9,\ 17,\ 9,\ 12,\ 11,\ 12$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:statproba:2}
The temperatures at noon over a week were $19,\ 21,\ 24,\ 18,\ 22,\ 25,\
19$ (degrees). Compute the mean (to a tenth) and the median.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:statproba:3}
The number of goals scored by a team in $10$ matches:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccc}
goals & $0$ & $1$ & $2$ & $3$ \\
\hline
matches & $2$ & $4$ & $3$ & $1$ \\
\end{tabular}
\end{center}
Compute the mean number of goals per match, and the median.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:statproba:4}
A fair die is rolled once. Compute the probability of: ``rolling a
$3$''; ``rolling an even number''; ``rolling at least a $3$''; ``not
rolling a $6$''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:statproba:5}
A box contains $12$ balls: $5$ white, $4$ red and $3$ green; one ball is
drawn at random. Compute $\P(\text{white})$, $\P(\text{red or green})$,
and $\P(\text{not white})$. What do you notice about the last two?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:statproba:6}
A coin is tossed, then a fair die is rolled. Draw the tree (or count the
outcomes) and compute the probability of getting heads \emph{and} a
$6$; then of getting heads \emph{or} a $6$ (or both).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:statproba:7}
In the bag of \cref{ex:g9:statproba:tree} ($2$ red, $3$ black), the two
draws are now made \emph{without} replacement. Draw the new tree
(careful: the second-level probabilities change) and compute the
probability of drawing two black tokens, then two tokens of different
colors.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:statproba:8}
After $9$ matches, a basketball player averages $14$ points per match.
\begin{enumerate}
  \item How many points has she scored in total?
  \item How many points must she score in the $10$th match to average
        $15$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:statproba:9}
A game: roll two fair dice; you win if the two results are equal
(``a double'').
\begin{enumerate}
  \item Compute the probability of winning.
  \item You play twice (independent games). Using a tree, compute the
        probability of winning at least once.
\end{enumerate}
\end{exercise}
