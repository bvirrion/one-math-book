\chapter{Algebra en vergelijkingen}\label{ch:g9:algebra}

Letters staan voor getallen, en rekenen met letters bewijst feiten over
\emph{alle} getallen tegelijk. Dit hoofdstuk oefent \omterm{def:g9:algebra:expand}{uitwerken} en
ontbinden --- inclusief de drie identiteiten die je door de hele
wiskunde volgen --- en lost eerstegraadsvergelijkingen,
productvergelijkingen en eenvoudige ongelijkheden op, altijd één
gemotiveerde stap per keer.

\section{Uitwerken}

\begin{definition}[Uitwerken]\label{def:g9:algebra:expand}
\emph{Uitwerken}\index{uitwerken} betekent een \omterm{def:g2:mult:def}{product} omzetten in een
\omterm{def:g1:addition:def}{som}, met distributiviteit:
\[
k(a + b) = ka + kb,
\qquad
(a + b)(c + d) = ac + ad + bc + bd .
\]
\end{definition}

\begin{example}\label{ex:g9:algebra:expand}
Elke term van de eerste haak vermenigvuldigt elke term van de tweede:
\begin{align*}
(x + 3)(2x + 5)
&= x \times 2x + x \times 5 + 3 \times 2x + 3 \times 5 \\
&= 2x^2 + 5x + 6x + 15 \\
&= 2x^2 + 11x + 15 .
\end{align*}
Tekens reizen met de termen mee:
\begin{align*}
(3x - 2)(x - 4)
&= 3x^2 - 12x - 2x + 8
= 3x^2 - 14x + 8 .
\end{align*}
\end{example}

\begin{theorem}[De drie identiteiten]\label{thm:g9:algebra:identities}
Voor alle getallen $a$ en $b$:
\[
(a+b)^2 = a^2 + 2ab + b^2,
\qquad
(a-b)^2 = a^2 - 2ab + b^2,
\qquad
(a+b)(a-b) = a^2 - b^2 .
\]
\end{theorem}

\begin{proof}
Werk elke linkerkant uit. Voor de eerste:
$(a+b)(a+b) = a^2 + ab + ba + b^2 = a^2 + 2ab + b^2$. Voor de
tweede, vervang $b$ door $-b$ in de eerste. Voor de derde:
$(a+b)(a-b) = a^2 - ab + ba - b^2 = a^2 - b^2$ (de kruistermen vallen
weg).
\end{proof}

\begin{example}\label{ex:g9:algebra:identities}
\[
(x + 5)^2 = x^2 + 10x + 25, \qquad
(3x - 1)^2 = 9x^2 - 6x + 1, \qquad
(2x + 7)(2x - 7) = 4x^2 - 49 .
\]
Mentaal rekenen: $102^2 = (100 + 2)^2 = 10000 + 400 + 4 = 10404$, en
$98 \times 102 = (100-2)(100+2) = 10000 - 4 = 9996$.
\end{example}

\section{Ontbinden}

\begin{method}[Ontbinden]\label{met:g9:algebra:factor}
Om een uitdrukking te ontbinden, probeer in volgorde:
\begin{enumerate}
  \item \emph{gemeenschappelijke factor}: $ka + kb = k(a + b)$, waar
        $k$ zelf een haak mag zijn;
  \item \emph{\omterm{ex:g1:subtraction:difference}{verschil} van kwadraten}: $a^2 - b^2 = (a+b)(a-b)$;
  \item \emph{volmaakt kwadraat}: $a^2 + 2ab + b^2 = (a+b)^2$ (en met
        $-2ab$, $(a-b)^2$).
\end{enumerate}
Controleer altijd door het resultaat uit te werken.
\end{method}

\begin{example}\label{ex:g9:algebra:factor}
\emph{Gemeenschappelijke factor:}
$15x^2 + 10x = 5x \times 3x + 5x \times 2 = 5x(3x + 2)$.

\emph{Gemeenschappelijke haak:}
\[
(x+2)(x-1) + (x+2)(3x+4) = (x+2)\bigl[(x-1) + (3x+4)\bigr]
= (x+2)(4x+3).
\]

\emph{\omterm{ex:g1:subtraction:difference}{Verschil} van kwadraten:}
$x^2 - 81 = (x+9)(x-9)$, en
\[
(2x+1)^2 - 9 = (2x+1)^2 - 3^2 = (2x+1+3)(2x+1-3) = (2x+4)(2x-2).
\]
Elke factor heeft nog een gemeenschappelijke factor:
$(2x+4)(2x-2) = 2(x+2) \times 2(x-1) = 4(x+2)(x-1)$.
\end{example}

\section{Vergelijkingen}

\begin{method}[Eerstegraadsvergelijkingen]\label{met:g9:algebra:firstdegree}
Om een \omterm{def:g8:equations:def}{vergelijking} als $5x - 3 = 2x + 9$ op te lossen:
\begin{enumerate}
  \item verzamel de $x$-termen aan één kant, de getallen aan de
        andere, door dezelfde hoeveelheid aan beide kanten op te
        tellen;
  \item herleid elke kant;
  \item deel beide kanten door de coëfficiënt van $x$;
  \item controleer de oplossing in de oorspronkelijke \omterm{def:g8:equations:def}{vergelijking}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:algebra:firstdegree}
\begin{align*}
5x - 3 &= 2x + 9 \\
5x - 2x &= 9 + 3 && \text{(tel $3 - 2x$ op aan beide kanten)}\\
3x &= 12 \\
x &= 4 && \text{(deel door 3).}
\end{align*}
Controle: $5 \times 4 - 3 = 17$ en $2 \times 4 + 9 = 17$. De
oplossing is $4$.

Met haakjes, werk eerst uit: $3(x - 2) = x + 8$ wordt $3x - 6 = x + 8$,
dus $2x = 14$ en $x = 7$.
\end{example}

\begin{theorem}[Nulproductregel]\label{thm:g9:algebra:zeroproduct}
Een \omterm{def:g2:mult:def}{product} is \omterm{def:g1:counting:zero}{nul} precies wanneer minstens één factor \omterm{def:g1:counting:zero}{nul} is:
\[
A \times B = 0
\quad\Longleftrightarrow\quad
A = 0 \ \text{ of } \ B = 0 .
\]
\end{theorem}

\begin{proof}
Als een factor \omterm{def:g1:counting:zero}{nul} is, is het \omterm{def:g2:mult:def}{product} duidelijk \omterm{def:g1:counting:zero}{nul}. Omgekeerd, als
$A B = 0$ met $A \neq 0$, geeft delen van beide kanten door $A$ dat
$B = 0$.
\end{proof}

\begin{example}\label{ex:g9:algebra:zeroproduct}
Los op $(x - 3)(2x + 10) = 0$: ofwel $x - 3 = 0$ ofwel $2x + 10 = 0$,
dus de oplossingen zijn $3$ en $-5$.

Los op $x^2 - 49 = 0$: ontbind eerst, $(x-7)(x+7) = 0$, oplossingen
$7$ en $-7$.

Los op $x^2 = 3x$: deel nooit door $x$! Verplaats en ontbind:
$x^2 - 3x = x(x - 3) = 0$, oplossingen $0$ en $3$.
\end{example}

\section{Ongelijkheden}

\begin{proposition}[Regels voor ongelijkheden]\label{prop:g9:algebra:ineq}
Een ongelijkheid blijft behouden bij optellen van hetzelfde getal aan
beide kanten, en bij vermenigvuldigen van beide kanten met hetzelfde
\emph{positieve} getal. Vermenigvuldigen van beide kanten met een
\emph{negatief} getal keert het ongelijkheidsteken om.
\end{proposition}

\begin{proof}
Als $a \leq b$ dan $b - a \geq 0$. Optellen van $c$:
$(b+c) - (a+c) = b - a \geq 0$, dus $a + c \leq b + c$.
Vermenigvuldigen met $c > 0$: $bc - ac = (b-a)c \geq 0$, dus
$ac \leq bc$. Vermenigvuldigen met $c < 0$: $(b-a)c \leq 0$, dus
$ac \geq bc$: de orde keert om.
\end{proof}

\begin{example}\label{ex:g9:algebra:ineq}
Los op $7 - 2x < 15$:
\begin{align*}
7 - 2x &< 15 \\
-2x &< 8 && \text{(trek 7 af)} \\
x &> -4 && \text{(deel door $-2$: keer het teken om).}
\end{align*}
De oplossingen zijn alle getallen groter dan $-4$: op een getallenlijn
een holle stip bij $-4$ en arcering naar rechts.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g9:algebra:1}
Werk uit en herleid:
\[
4(3x - 2), \qquad
(x + 6)(x + 2), \qquad
(2x - 3)(x + 5), \qquad
(x - 1)(x - 9).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:algebra:2}
Werk uit met de identiteiten:
\[
(x + 8)^2, \qquad
(3x - 4)^2, \qquad
(x + 10)(x - 10), \qquad
(5x + 2)(5x - 2).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:algebra:3}
Reken uit het hoofd, met een identiteit: $101^2$; \quad $59^2$
(schrijf $59 = 60 - 1$); \quad $53 \times 47$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:algebra:4}
Ontbind:
\[
9x + 12, \qquad
x^2 - 64, \qquad
x^2 + 14x + 49, \qquad
18x^2 - 6x .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:algebra:5}
Los op, en schrijf elke stap:
\[
4x + 7 = 19, \qquad
6x - 5 = 2x + 11, \qquad
5(x - 3) = 3x + 1 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:algebra:6}
Los op met de nulproductregel:
\[
(x + 4)(3x - 12) = 0, \qquad
x^2 - 121 = 0, \qquad
(2x - 1)^2 = 0, \qquad
x^2 = 8x .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:algebra:7}
Ontbind $E = (3x + 2)^2 - 25$, en los daarna $E = 0$ op.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:algebra:8}
Los de ongelijkheden op en stel de oplossingen voor op een
getallenlijn:
\[
3x - 4 \geq 8, \qquad
5 - 2x > 1, \qquad
4(x + 1) \leq x - 5 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:algebra:9}
Ik denk aan een getal, vermenigvuldig het met $3$, tel $7$ op, en krijg
hetzelfde resultaat als wanneer ik het met $5$ vermenigvuldigd en $9$
afgetrokken had. Wat is mijn getal? (Stel een \omterm{def:g8:equations:def}{vergelijking} op en los
die op.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:algebra:10}
Laat $n$ een geheel getal zijn.
\begin{enumerate}
  \item Werk $(n + 1)(n - 1)$ uit en leid af dat het \omterm{def:g2:mult:def}{product} van de
        twee buren van $n$ altijd $n^2 - 1$ is (bijv.\ $24 \times 26 =
        25^2 - 1 = 624$).
  \item Toon dat de \omterm{def:g1:addition:def}{som} van drie opeenvolgende gehele getallen altijd
        een \omterm{def:g9:arith:divisor}{veelvoud} van $3$ is.
\end{enumerate}
\end{exercise}
