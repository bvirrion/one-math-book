\chapter{Lichamen en volumes}\label{ch:g9:solids}

Ruimtemeetkunde begint met een kleine familie van lichamen ---
\omterm{def:g7:areas:prism}{prisma}'s, \omterm{def:g7:areas:prism}{cilinders}, \omterm{def:g8:solids:def}{piramides}, \omterm{def:g8:solids:def}{kegels}, bollen --- en twee vragen:
hoeveel houden ze (\omterm{def:g6:measure:volume}{volume}), en wat zie je als je ze snijdt
(\omterm{prop:g9:solids:sections}{doorsneden})? Het hoofdstuk sluit af met een feit van groot praktisch
belang: een \omterm{def:g5:solids:def}{lichaam} schalen met $k$ vermenigvuldigt zijn \omterm{def:g6:measure:volume}{volume} met
$k^3$.

\section{De klassieke lichamen en hun volumes}

\begin{theorem}[Volumeformules]\label{thm:g9:solids:volumes}
Schrijf $B$ voor de \omterm{def:g6:measure:area}{oppervlakte} van het grondvlak en $h$ voor de
hoogte (de afstand tussen het grondvlak en het overstaande vlak of de
top).
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{l|l}
\omterm{def:g5:solids:def}{lichaam} & \omterm{def:g6:measure:volume}{volume} \\
\hline
\omterm{def:g7:areas:prism}{prisma}, \omterm{def:g7:areas:prism}{cilinder} & $V = B \times h$ \\[5pt]
\omterm{def:g8:solids:def}{piramide}, \omterm{def:g8:solids:def}{kegel} & $V = \dfrac13\, B \times h$ \\[5pt]
bol met \omterm{def:g3:shapes:shapes}{straal} $r$ & $V = \dfrac43\,\pi r^3$ \\
\end{tabular}
\end{center}
Voor een \omterm{def:g7:areas:prism}{cilinder} en een \omterm{def:g8:solids:def}{kegel} met \omterm{def:g3:shapes:shapes}{straal} $r$ is de
grondvlakoppervlakte $B = \pi r^2$; de \omterm{def:g6:measure:area}{oppervlakte} van de bol is
$4\pi r^2$.
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:volumes}
Een cilindrisch blik heeft \omterm{def:g3:shapes:shapes}{straal} $4$ cm en hoogte $10$ cm:
\[
V = \pi r^2 h = \pi \times 16 \times 10 = 160\pi \approx 503 \text{ cm}^3
\approx 0.5 \text{ L}.
\]
Een \omterm{def:g8:solids:def}{kegel} met hetzelfde grondvlak en dezelfde hoogte houdt één derde
daarvan: $\frac{160\pi}{3} \approx 168$ cm$^3$. Een bol met \omterm{def:g3:shapes:shapes}{straal}
$4$ cm: $V = \frac43 \pi \times 64 = \frac{256\pi}{3} \approx 268$
cm$^3$.
\end{example}

\begin{example}[Een piramide stap voor stap]\label{ex:g9:solids:pyramid}
Een \omterm{def:g8:solids:def}{piramide} heeft een vierkant grondvlak met zijde $6$ m en hoogte
$5$ m.
\begin{enumerate}
  \item Grondvlakoppervlakte: $B = 6^2 = 36$ m$^2$.
  \item \omterm{def:g6:measure:volume}{Volume}: $V = \frac13 \times 36 \times 5 = 60$ m$^3$.
\end{enumerate}
Let op de eenheden: als de zijde in cm gegeven was en de hoogte in m,
zou één van beide eerst omgerekend moeten worden.
\end{example}

\section{Doorsneden met vlakken}

\begin{proposition}[Doorsneden van de klassieke lichamen]\label{prop:g9:solids:sections}
Een \omterm{def:g5:solids:def}{lichaam} snijden met een vlak produceert een platte figuur, zijn
\emph{doorsnede}\index{doorsnede}:
\begin{itemize}
  \item een \omterm{def:g7:areas:prism}{prisma} of \omterm{def:g7:areas:prism}{cilinder} \emph{\omterm{def:g6:lines:perp}{evenwijdig} aan zijn grondvlak}
        gesneden geeft een kopie van het grondvlak, op elke hoogte;
  \item een \omterm{def:g8:solids:def}{piramide} of \omterm{def:g8:solids:def}{kegel} \omterm{def:g6:lines:perp}{evenwijdig} aan zijn grondvlak gesneden
        geeft een \emph{verkleining} van het grondvlak: op afstand $d$
        van de top is de \omterm{def:g9:thales:scaling}{schaalfactor} $k = \frac{d}{h}$;
  \item een bol met \omterm{def:g3:shapes:shapes}{straal} $r$ gesneden door een vlak op afstand
        $d < r$ van het middelpunt geeft een cirkel met \omterm{def:g3:shapes:shapes}{straal}
        $\sqrt{r^2 - d^2}$ (door de stelling van Pythagoras).
\end{itemize}
\end{proposition}
\admitted

\begin{example}\label{ex:g9:solids:section}
Een \omterm{def:g8:solids:def}{kegel} heeft grondstraal $6$ en hoogte $9$. De \omterm{prop:g9:solids:sections}{doorsnede} door een
vlak \omterm{def:g6:lines:perp}{evenwijdig} aan het grondvlak, op afstand $3$ van de top, is een
schijf geschaald met $k = \frac39 = \frac13$: zijn \omterm{def:g3:shapes:shapes}{straal} is
$6 \times \frac13 = 2$.

Een bol met \omterm{def:g3:shapes:shapes}{straal} $5$ wordt gesneden door een vlak op afstand $3$ van
zijn middelpunt: de \omterm{prop:g9:solids:sections}{doorsnede} is een cirkel met \omterm{def:g3:shapes:shapes}{straal}
$\sqrt{25 - 9} = 4$.
\end{example}

\section{Lichamen schalen}

\begin{theorem}[Effect van schalen op lengten, oppervlakten, volumes]\label{thm:g9:solids:scaling}
Wanneer een \omterm{def:g5:solids:def}{lichaam} vergroot of verkleind wordt met de \omterm{def:g9:thales:scaling}{schaalfactor}
$k > 0$:
\begin{itemize}
  \item alle lengten worden met $k$ vermenigvuldigd;
  \item alle \omterm{def:g6:measure:area}{oppervlakten} worden met $k^2$ vermenigvuldigd;
  \item alle \omterm{def:g6:measure:volume}{volumes} worden met $k^3$ vermenigvuldigd.
\end{itemize}
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:scaling}
Een modelauto op schaal $\frac{1}{18}$: lengten zijn die van de echte
auto vermenigvuldigd met $k = \frac{1}{18}$, beschilderd oppervlak
vermenigvuldigd met $k^2 = \frac{1}{324}$, en \omterm{def:g6:measure:volume}{volume} met
$k^3 = \frac{1}{5832}$.

De \omterm{def:g3:shapes:shapes}{straal} van een bol verdubbelen ($k = 2$) vermenigvuldigt zijn
\omterm{def:g6:measure:volume}{volume} met $2^3 = 8$: controle op de formule,
$\frac43\pi (2r)^3 = \frac43 \pi \times 8r^3$.
\end{example}

\begin{example}[Afgeknotte kegel]\label{ex:g9:solids:truncated}
Een \omterm{def:g8:solids:def}{kegel} met hoogte $9$ en grondstraal $6$ (\omterm{def:g6:measure:volume}{volume}
$\frac13 \pi \times 36 \times 9 = 108\pi$) wordt gesneden op afstand
$3$ van de top, en de kleine \omterm{def:g8:solids:def}{kegel} boven de snede wordt weggenomen. De
kleine \omterm{def:g8:solids:def}{kegel} is de verkleining met $k = \frac13$, dus zijn \omterm{def:g6:measure:volume}{volume} is
$108\pi \times \left(\frac13\right)^3 = 4\pi$. Het resterende \omterm{def:g5:solids:def}{lichaam}
(een \emph{afgeknotte \omterm{def:g8:solids:def}{kegel}}) heeft \omterm{def:g6:measure:volume}{volume} $108\pi - 4\pi = 104\pi
\approx 327$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g9:solids:1}
Bereken de \omterm{def:g6:measure:volume}{volumes}: een \omterm{def:g5:solids:def}{balk} (\omterm{def:g6:shapes:triangles}{rechthoekig} \omterm{def:g7:areas:prism}{prisma}) met afmetingen
$4 \times 5 \times 12$; een \omterm{def:g7:areas:prism}{cilinder} met \omterm{def:g3:shapes:shapes}{straal} $3$ en hoogte $7$
(exacte waarde met $\pi$, daarna afgerond op de eenheid).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:solids:2}
Bereken het \omterm{def:g6:measure:volume}{volume} van een \omterm{def:g8:solids:def}{kegel} met \omterm{def:g3:shapes:shapes}{straal} $5$ en hoogte $12$, en van
een \omterm{def:g8:solids:def}{piramide} met \omterm{def:g6:shapes:triangles}{rechthoekig} grondvlak $8 \times 3$ en hoogte $10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:solids:3}
Bereken het \omterm{def:g6:measure:volume}{volume} en de \omterm{def:g6:measure:area}{oppervlakte} van een bol met \omterm{def:g3:shapes:shapes}{straal} $6$
(exacte waarden met $\pi$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:solids:4}
Een bol met \omterm{def:g3:shapes:shapes}{straal} $10$ wordt gesneden door een vlak op afstand $8$ van
zijn middelpunt. Wat is de \omterm{def:g3:shapes:shapes}{straal} van de doorsnedecirkel?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:5}
Een \omterm{def:g8:solids:def}{kegel} heeft grondstraal $8$ en hoogte $12$. Een vlak \omterm{def:g6:lines:perp}{evenwijdig}
aan het grondvlak snijdt hem op afstand $9$ van de top. Bereken de
\omterm{def:g3:shapes:shapes}{straal} van de \omterm{prop:g9:solids:sections}{doorsnede}, daarna het \omterm{def:g6:measure:volume}{volume} van de kleine \omterm{def:g8:solids:def}{kegel} boven
de snede.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:6}
Een cilindrisch glas met binnenstraal $3$ cm bevat water tot een
hoogte van $10$ cm. Een bal met \omterm{def:g3:shapes:shapes}{straal} $2$ cm wordt er volledig in
ondergedompeld. Met hoeveel stijgt het waterniveau? (Het toegevoegde
\omterm{def:g6:measure:volume}{volume} spreidt zich over de grondvlakoppervlakte van het glas; geef de
exacte stijging, daarna afgerond op de millimeter.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:7}
Een recept vult een bolvormige mal met \omterm{def:g3:shapes:shapes}{straal} $6$ cm. Je hebt alleen
een bolvormige mal met \omterm{def:g3:shapes:shapes}{straal} $3$ cm. Hoeveel kleine mallen kun je
vullen? (Antwoord zonder een van beide \omterm{def:g6:measure:volume}{volumes} expliciet te
berekenen.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:8}
De Grote \omterm{def:g8:solids:def}{Piramide} van Gizeh heeft een vierkant grondvlak met zijde
ongeveer $230$ m en een hoogte van ongeveer $147$ m. Schat haar
\omterm{def:g6:measure:volume}{volume}, en druk het uit in miljoenen kubieke meters.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:solids:9}
Een kegelvormige trechter met \omterm{def:g3:shapes:shapes}{straal} $6$ cm en hoogte $12$ cm, top
naar beneden, wordt met water gevuld tot de \omterm{def:g3:division:half}{helft} van zijn
\emph{hoogte} (gemeten vanaf de top).
\begin{enumerate}
  \item Welk deel van het \omterm{def:g6:measure:volume}{volume} van de trechter is gevuld?
  \item Als hij in plaats daarvan met de \omterm{def:g3:division:half}{helft} van zijn \emph{\omterm{def:g6:measure:volume}{volume}}
        water gevuld wordt, toon dat de waterhoogte $d$ voldoet aan
        $\left(\frac{d}{12}\right)^3 = \frac12$, en geef $d$ op de
        millimeter ($\sqrt[3]{0.5} \approx 0.7937$).
\end{enumerate}
\end{exercise}
