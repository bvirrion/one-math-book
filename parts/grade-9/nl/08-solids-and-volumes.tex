\chapter{Lichamen en volumes}\label{ch:g9:solids}

Ruimtemeetkunde begint met een kleine familie van lichamen ---
prisma's, cilinders, piramides, kegels, bollen --- en twee vragen:
hoeveel houden ze (volume), en wat zie je als je ze snijdt
(doorsneden)? Het hoofdstuk sluit af met een feit van groot praktisch
belang: een lichaam schalen met $k$ vermenigvuldigt zijn volume met
$k^3$.

\section{De klassieke lichamen en hun volumes}

\begin{theorem}[Volumeformules]\label{thm:g9:solids:volumes}
Schrijf $B$ voor de oppervlakte van het grondvlak en $h$ voor de
hoogte (de afstand tussen het grondvlak en het overstaande vlak of de
top).
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{l|l}
lichaam & volume \\
\hline
prisma, cilinder & $V = B \times h$ \\[5pt]
piramide, kegel & $V = \dfrac13\, B \times h$ \\[5pt]
bol met straal $r$ & $V = \dfrac43\,\pi r^3$ \\
\end{tabular}
\end{center}
Voor een cilinder en een kegel met straal $r$ is de
grondvlakoppervlakte $B = \pi r^2$; de oppervlakte van de bol is
$4\pi r^2$.
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:volumes}
Een cilindrisch blik heeft straal $4$ cm en hoogte $10$ cm:
\[
V = \pi r^2 h = \pi \times 16 \times 10 = 160\pi \approx 503 \text{ cm}^3
\approx 0.5 \text{ L}.
\]
Een kegel met hetzelfde grondvlak en dezelfde hoogte houdt één derde
daarvan: $\frac{160\pi}{3} \approx 168$ cm$^3$. Een bol met straal
$4$ cm: $V = \frac43 \pi \times 64 = \frac{256\pi}{3} \approx 268$
cm$^3$.
\end{example}

\begin{example}[Een piramide stap voor stap]\label{ex:g9:solids:pyramid}
Een piramide heeft een vierkant grondvlak met zijde $6$ m en hoogte
$5$ m.
\begin{enumerate}
  \item Grondvlakoppervlakte: $B = 6^2 = 36$ m$^2$.
  \item Volume: $V = \frac13 \times 36 \times 5 = 60$ m$^3$.
\end{enumerate}
Let op de eenheden: als de zijde in cm gegeven was en de hoogte in m,
zou één van beide eerst omgerekend moeten worden.
\end{example}

\section{Doorsneden met vlakken}

\begin{proposition}[Doorsneden van de klassieke lichamen]\label{prop:g9:solids:sections}
Een lichaam snijden met een vlak produceert een platte figuur, zijn
\emph{doorsnede}\index{doorsnede}:
\begin{itemize}
  \item een prisma of cilinder \emph{evenwijdig aan zijn grondvlak}
        gesneden geeft een kopie van het grondvlak, op elke hoogte;
  \item een piramide of kegel evenwijdig aan zijn grondvlak gesneden
        geeft een \emph{verkleining} van het grondvlak: op afstand $d$
        van de top is de schaalfactor $k = \frac{d}{h}$;
  \item een bol met straal $r$ gesneden door een vlak op afstand
        $d < r$ van het middelpunt geeft een cirkel met straal
        $\sqrt{r^2 - d^2}$ (door de stelling van Pythagoras).
\end{itemize}
\end{proposition}
\admitted

\begin{example}\label{ex:g9:solids:section}
Een kegel heeft grondstraal $6$ en hoogte $9$. De doorsnede door een
vlak evenwijdig aan het grondvlak, op afstand $3$ van de top, is een
schijf geschaald met $k = \frac39 = \frac13$: zijn straal is
$6 \times \frac13 = 2$.

Een bol met straal $5$ wordt gesneden door een vlak op afstand $3$ van
zijn middelpunt: de doorsnede is een cirkel met straal
$\sqrt{25 - 9} = 4$.
\end{example}

\section{Lichamen schalen}

\begin{theorem}[Effect van schalen op lengten, oppervlakten, volumes]\label{thm:g9:solids:scaling}
Wanneer een lichaam vergroot of verkleind wordt met de schaalfactor
$k > 0$:
\begin{itemize}
  \item alle lengten worden met $k$ vermenigvuldigd;
  \item alle oppervlakten worden met $k^2$ vermenigvuldigd;
  \item alle volumes worden met $k^3$ vermenigvuldigd.
\end{itemize}
\end{theorem}
\admitted

\begin{example}\label{ex:g9:solids:scaling}
Een modelauto op schaal $\frac{1}{18}$: lengten zijn die van de echte
auto vermenigvuldigd met $k = \frac{1}{18}$, beschilderd oppervlak
vermenigvuldigd met $k^2 = \frac{1}{324}$, en volume met
$k^3 = \frac{1}{5832}$.

De straal van een bol verdubbelen ($k = 2$) vermenigvuldigt zijn
volume met $2^3 = 8$: controle op de formule,
$\frac43\pi (2r)^3 = \frac43 \pi \times 8r^3$.
\end{example}

\begin{example}[Afgeknotte kegel]\label{ex:g9:solids:truncated}
Een kegel met hoogte $9$ en grondstraal $6$ (volume
$\frac13 \pi \times 36 \times 9 = 108\pi$) wordt gesneden op afstand
$3$ van de top, en de kleine kegel boven de snede wordt weggenomen. De
kleine kegel is de verkleining met $k = \frac13$, dus zijn volume is
$108\pi \times \left(\frac13\right)^3 = 4\pi$. Het resterende lichaam
(een \emph{afgeknotte kegel}) heeft volume $108\pi - 4\pi = 104\pi
\approx 327$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g9:solids:1}
Bereken de volumes: een balk (rechthoekig prisma) met afmetingen
$4 \times 5 \times 12$; een cilinder met straal $3$ en hoogte $7$
(exacte waarde met $\pi$, daarna afgerond op de eenheid).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:solids:2}
Bereken het volume van een kegel met straal $5$ en hoogte $12$, en van
een piramide met rechthoekig grondvlak $8 \times 3$ en hoogte $10$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:solids:3}
Bereken het volume en de oppervlakte van een bol met straal $6$
(exacte waarden met $\pi$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:solids:4}
Een bol met straal $10$ wordt gesneden door een vlak op afstand $8$ van
zijn middelpunt. Wat is de straal van de doorsnedecirkel?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:5}
Een kegel heeft grondstraal $8$ en hoogte $12$. Een vlak evenwijdig
aan het grondvlak snijdt hem op afstand $9$ van de top. Bereken de
straal van de doorsnede, daarna het volume van de kleine kegel boven
de snede.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:6}
Een cilindrisch glas met binnenstraal $3$ cm bevat water tot een
hoogte van $10$ cm. Een bal met straal $2$ cm wordt er volledig in
ondergedompeld. Met hoeveel stijgt het waterniveau? (Het toegevoegde
volume spreidt zich over de grondvlakoppervlakte van het glas; geef de
exacte stijging, daarna afgerond op de millimeter.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:7}
Een recept vult een bolvormige mal met straal $6$ cm. Je hebt alleen
een bolvormige mal met straal $3$ cm. Hoeveel kleine mallen kun je
vullen? (Antwoord zonder een van beide volumes expliciet te
berekenen.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:solids:8}
De Grote Piramide van Gizeh heeft een vierkant grondvlak met zijde
ongeveer $230$ m en een hoogte van ongeveer $147$ m. Schat haar
volume, en druk het uit in miljoenen kubieke meters.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:solids:9}
Een kegelvormige trechter met straal $6$ cm en hoogte $12$ cm, top
naar beneden, wordt met water gevuld tot de helft van zijn
\emph{hoogte} (gemeten vanaf de top).
\begin{enumerate}
  \item Welk deel van het volume van de trechter is gevuld?
  \item Als hij in plaats daarvan met de helft van zijn \emph{volume}
        water gevuld wordt, toon dat de waterhoogte $d$ voldoet aan
        $\left(\frac{d}{12}\right)^3 = \frac12$, en geef $d$ op de
        millimeter ($\sqrt[3]{0.5} \approx 0.7937$).
\end{enumerate}
\end{exercise}
