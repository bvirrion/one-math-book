\chapter{Statistiek en kansrekening}\label{ch:g9:statproba}

Gegevens en toeval zijn overal: sportresultaten, weersvoorspellingen,
spellen. Dit hoofdstuk leert hoe je een reeks getallen samenvat met
gemiddelde, mediaan en spreidingsbreedte, en hoe je de kans van
eenvoudige toevalsexperimenten berekent --- inclusief
tweestapsexperimenten, behandeld met boomschema's. Beide onderwerpen
worden veel verder uitgewerkt in het High School-volume van deze
reeks.

\section{Gegevens samenvatten}

\begin{definition}[Gemiddelde, mediaan, spreidingsbreedte]\label{def:g9:statproba:indicators}
Voor een reeks van $N$ getallen:
\begin{itemize}
  \item het \emph{gemiddelde}\index{gemiddelde} is de som van alle
        waarden gedeeld door $N$;
  \item de \emph{mediaan}\index{mediaan} is een waarde die de
        gesorteerde reeks in twee helften van gelijke grootte splitst:
        voor $N$ oneven, de middelste waarde; voor $N$ even, het
        midden van de twee middelste waarden;
  \item de \emph{spreidingsbreedte}\index{spreidingsbreedte} is de
        grootste waarde min de kleinste.
\end{itemize}
\end{definition}

\begin{example}\label{ex:g9:statproba:indicators}
Cijfers van een leerling: $8,\ 12,\ 9,\ 15,\ 11$.
\begin{enumerate}
  \item Gemiddelde: $\dfrac{8 + 12 + 9 + 15 + 11}{5} = \dfrac{55}{5}
        = 11$.
  \item Mediaan: sorteer de reeks: $8, 9, 11, 12, 15$; de middelste
        (3e) waarde is $11$.
  \item Spreidingsbreedte: $15 - 8 = 7$.
\end{enumerate}
Met een zesde cijfer van $17$: gesorteerd $8, 9, 11, 12, 15, 17$,
wordt de mediaan het midden van $11$ en $12$, dat is $11.5$, en het
gemiddelde wordt $\frac{72}{6} = 12$.
\end{example}

\begin{example}[Gewogen gemiddelde]\label{ex:g9:statproba:weighted}
De schoenmaten verkocht op één dag:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccccc}
maat & $38$ & $39$ & $40$ & $41$ & $42$ \\
\hline
aantal & $3$ & $7$ & $6$ & $3$ & $1$ \\
\end{tabular}
\end{center}
De gemiddelde maat is
\[
\frac{3 \times 38 + 7 \times 39 + 6 \times 40 + 3 \times 41 + 1 \times 42}
{3 + 7 + 6 + 3 + 1}
= \frac{114 + 273 + 240 + 123 + 42}{20}
= \frac{792}{20} = 39.6 .
\]
\end{example}

\begin{remark}
Het gemiddelde en de mediaan kunnen sterk verschillen. In de reeks
$10, 10, 10, 10, 60$ is het gemiddelde $20$ (omhooggetrokken door de
waarde $60$) terwijl de mediaan $10$ is. Vraag altijd welk indicator
de situatie beter weergeeft.
\end{remark}

\section{Kansrekening}

\begin{definition}[Kans]\label{def:g9:statproba:probability}
Een toevalsexperiment heeft verschillende mogelijke
\emph{uitkomsten}; een \emph{gebeurtenis} is een verzameling
uitkomsten. Elke uitkomst krijgt een \emph{kans}\index{kans}: een
getal tussen $0$ en $1$, waarbij alle kansen van de uitkomsten optellen
tot $1$; de kans $\P(A)$ van een gebeurtenis $A$ is de som van de
kansen van haar uitkomsten. Wanneer alle $n$ uitkomsten even
waarschijnlijk zijn,
\[
\P(A) = \frac{\text{aantal gunstige uitkomsten}}{n}.
\]
\end{definition}

\begin{example}\label{ex:g9:statproba:spinner}
Een wiel is verdeeld in $8$ gelijke sectoren: $3$ rood, $3$ blauw,
$2$ groen. Elke sector heeft kans $\frac18$, dus
\[
\P(\text{rood}) = \frac38, \qquad
\P(\text{groen}) = \frac28 = \frac14, \qquad
\P(\text{niet groen}) = 1 - \frac14 = \frac34 .
\]
\end{example}

\begin{proposition}[Complement]\label{prop:g9:statproba:complement}
Voor elke gebeurtenis $A$ voldoet de \emph{tegengestelde
gebeurtenis} $\bar A$ (``$A$ treedt niet op'') aan
\[
\P(\bar A) = 1 - \P(A).
\]
\end{proposition}

\begin{proof}
Elke uitkomst zit in precies één van $A$ en $\bar A$, dus
$\P(A) + \P(\bar A)$ sommeert de kansen van alle uitkomsten, wat $1$
is.
\end{proof}

\section{Tweestapsexperimenten}

\begin{method}[Boomschema's]\label{met:g9:statproba:tree}
Voor een experiment in twee stappen:
\begin{enumerate}
  \item teken één tak per uitkomst van de eerste stap, en zet daarna
        elke tak voort met de uitkomsten van de tweede stap, met elke
        kans op haar tak;
  \item vermenigvuldig de kansen langs een pad om de kans van dat pad
        te krijgen;
  \item tel de kansen van alle paden die een gebeurtenis vormen op.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:statproba:tree}
Een zak houdt $2$ rode en $3$ zwarte fiches. Trek één fiche, noteer
zijn kleur, \emph{leg hem terug}, trek opnieuw. Elke trekking geeft
rood met kans $\frac25$, zwart met kans $\frac35$.

\begin{omfigure}
\begin{tikzpicture}[scale=0.95]
  \coordinate (root) at (0,0);
  \coordinate (R) at (2.2,1.0);
  \coordinate (B) at (2.2,-1.0);
  \coordinate (RR) at (4.7,1.55);
  \coordinate (RB) at (4.7,0.45);
  \coordinate (BR) at (4.7,-0.45);
  \coordinate (BB) at (4.7,-1.55);
  \draw (root) -- (R) node[midway, above, font=\small, sloped] {$\frac25$};
  \draw (root) -- (B) node[midway, below, font=\small, sloped] {$\frac35$};
  \draw (R) -- (RR) node[midway, above, font=\small, sloped] {$\frac25$};
  \draw (R) -- (RB) node[midway, below, font=\small, sloped] {$\frac35$};
  \draw (B) -- (BR) node[midway, above, font=\small, sloped] {$\frac25$};
  \draw (B) -- (BB) node[midway, below, font=\small, sloped] {$\frac35$};
  \node[omThm, font=\small, above left=-2pt] at (R) {R};
  \node[font=\small, below left=-2pt] at (B) {B};
  \node[right, font=\small] at (RR)
    {RR:\ $\frac25 \times \frac25 = \frac{4}{25}$};
  \node[right, font=\small] at (RB)
    {RB:\ $\frac25 \times \frac35 = \frac{6}{25}$};
  \node[right, font=\small] at (BR)
    {BR:\ $\frac35 \times \frac25 = \frac{6}{25}$};
  \node[right, font=\small] at (BB)
    {BB:\ $\frac35 \times \frac35 = \frac{9}{25}$};
\end{tikzpicture}

{\small De boom van de twee trekkingen met teruglegging:
vermenigvuldig langs de takken, en controleer dat de vier paden
optellen tot $1$.}
\end{omfigure}

Kans op twee fiches van dezelfde kleur:
$\frac{4}{25} + \frac{9}{25} = \frac{13}{25}$. Kans op minstens één
rood: $1 - \P(\text{BB}) = 1 - \frac{9}{25} = \frac{16}{25}$.
\end{example}

\begin{remark}[Frequenties naderen kansen]
Een eerlijke dobbelsteen $6000$ keer gooien geeft \emph{ongeveer}
$1000$ zessen --- niet exact. Als het aantal herhalingen groeit,
komen waargenomen frequenties dichter en dichter bij de kansen: dat
maakt kansrekening het juiste gereedschap om herhaalde experimenten te
modelleren (een verhaal uitgewerkt in het High School-volume van deze
reeks).
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g9:statproba:1}
Bereken het gemiddelde, de mediaan en de spreidingsbreedte van de
reeks $14,\ 9,\ 17,\ 9,\ 12,\ 11,\ 12$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:statproba:2}
De temperaturen om twaalf uur over een week waren $19,\ 21,\ 24,\ 18,\
22,\ 25,\ 19$ (graden). Bereken het gemiddelde (op een tiende) en de
mediaan.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:statproba:3}
Het aantal doelpunten gescoord door een team in $10$ wedstrijden:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccc}
doelpunten & $0$ & $1$ & $2$ & $3$ \\
\hline
wedstrijden & $2$ & $4$ & $3$ & $1$ \\
\end{tabular}
\end{center}
Bereken het gemiddeld aantal doelpunten per wedstrijd, en de mediaan.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:statproba:4}
Een eerlijke dobbelsteen wordt één keer gegooid. Bereken de kans op:
``een $3$ gooien''; ``een even getal gooien''; ``minstens een $3$
gooien''; ``geen $6$ gooien''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:statproba:5}
Een doos bevat $12$ ballen: $5$ wit, $4$ rood en $3$ groen; één bal
wordt willekeurig getrokken. Bereken $\P(\text{wit})$,
$\P(\text{rood of groen})$, en $\P(\text{niet wit})$. Wat merk je op
over de laatste twee?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:statproba:6}
Een munt wordt opgegooid, daarna een eerlijke dobbelsteen gegooid.
Teken de boom (of tel de uitkomsten) en bereken de kans op munt
\emph{en} een $6$; daarna op munt \emph{of} een $6$ (of beide).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:statproba:7}
In de zak van \cref{ex:g9:statproba:tree} ($2$ rood, $3$ zwart) worden
de twee trekkingen nu \emph{zonder} teruglegging gedaan. Teken de
nieuwe boom (let op: de kansen op het tweede niveau veranderen) en
bereken de kans op twee zwarte fiches, daarna twee fiches van
verschillende kleuren.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:statproba:8}
Na $9$ wedstrijden scoort een basketbalspeler gemiddeld $14$ punten
per wedstrijd.
\begin{enumerate}
  \item Hoeveel punten heeft ze in totaal gescoord?
  \item Hoeveel punten moet ze scoren in de $10$e wedstrijd om
        gemiddeld $15$ te scoren?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:statproba:9}
Een spel: gooi twee eerlijke dobbelstenen; je wint als de twee
resultaten gelijk zijn (``een double'').
\begin{enumerate}
  \item Bereken de kans op winnen.
  \item Je speelt twee keer (onafhankelijke spellen). Bereken met een
        boom de kans om minstens één keer te winnen.
\end{enumerate}
\end{exercise}
