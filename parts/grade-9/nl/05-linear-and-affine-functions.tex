\chapter{Lineaire en affiene functies}\label{ch:g9:linfunc}

``Drie kilogram kost driemaal zoveel als één'': dat is evenredigheid, de
eenvoudigste manier waarop twee grootheden gerelateerd kunnen zijn.
Lineaire functies zijn evenredigheid in de taal van functies; affiene
functies voegen een vast startbedrag toe. Hun grafieken zijn rechte
lijnen, en die lijnen lezen is de vaardigheid die dit hoofdstuk bouwt.

\section{Evenredigheid en lineaire functies}

\begin{definition}[Lineaire functie]\label{def:g9:linfunc:linear}
Een \emph{lineaire functie}\index{lineaire functie} vermenigvuldigt elk
getal met een vast getal $a$:
\[
f(x) = a x .
\]
Het getal $a$ is de \emph{coëfficiënt} van $f$. Twee grootheden zijn
\emph{evenredig}\index{evenredigheid} precies wanneer de ene een
lineaire functie van de andere is.
\end{definition}

\begin{example}\label{ex:g9:linfunc:linear}
Als appels $3$ per kilogram kosten, is de prijs van $x$ kilogram
$f(x) = 3x$: tweemaal zoveel kopen kost tweemaal zoveel, en de prijs
van $2.5$ kg is $f(2.5) = 7.5$.

Omgekeerd, als een lineaire functie voldoet aan $f(4) = 10$, dan
$a = \frac{10}{4} = 2.5$ en $f(x) = 2.5x$: \emph{één waarde bepaalt een
lineaire functie volledig.}
\end{example}

\begin{proposition}[Grafiek van een lineaire functie]\label{prop:g9:linfunc:lineargraph}
De grafiek van $f(x) = ax$ is een rechte lijn \emph{door de
oorsprong}. Omgekeerd is elke niet-verticale lijn door de oorsprong de
grafiek van een lineaire functie.
\end{proposition}
\admitted

\section{Affiene functies}

\begin{definition}[Affiene functie]\label{def:g9:linfunc:affine}
Een \emph{affiene functie}\index{affiene functie} heeft de vorm
\[
f(x) = a x + b ,
\]
waarbij $a$ en $b$ vaste getallen zijn. Lineaire functies zijn het
speciale geval $b = 0$; constante functies het geval $a = 0$.
\end{definition}

\begin{example}
Een sportschool rekent een inschrijfgeld van $20$, daarna $5$ per
bezoek: de totale kost van $x$ bezoeken is $f(x) = 5x + 20$. De kost
is \emph{niet} evenredig met het aantal bezoeken (10 bezoeken kosten
niet tweemaal zoveel als 5), maar elk extra bezoek voegt hetzelfde
bedrag toe, $5$.
\end{example}

\begin{proposition}[Grafiek van een affiene functie]\label{prop:g9:linfunc:affinegraph}
De grafiek van $f(x) = ax + b$ is een rechte lijn:
\begin{itemize}
  \item $b = f(0)$ is het \emph{$y$-snijpunt}: de lijn snijdt de
        verticale as in $(0, b)$;
  \item $a$ is de \emph{richtingscoëfficiënt}\index{richtingsco\"effici\"ent}:
        één eenheid naar rechts bewegen verplaatst $a$ eenheden omhoog
        (omlaag als $a < 0$); de functie is stijgend wanneer $a > 0$,
        dalend wanneer $a < 0$;
  \item voor elke twee inputs $u \neq v$,
        $a = \dfrac{f(v) - f(u)}{v - u}$.
\end{itemize}
\end{proposition}

\begin{proof}[Bewijs van het laatste punt]
$f(v) - f(u) = (av + b) - (au + b) = a(v - u)$; deel door $v - u$. De
rest is toegegeven op dit niveau (het High School-volume geeft een
volledige behandeling van lijnen en hun vergelijkingen).
\end{proof}

\begin{method}[Een affiene functie vinden uit twee waarden]\label{met:g9:linfunc:findaffine}
Gegeven $f(u)$ en $f(v)$ met $u \neq v$:
\begin{enumerate}
  \item bereken de richtingscoëfficiënt $a = \dfrac{f(v) - f(u)}{v - u}$;
  \item vul één van de twee bekende waarden in in $f(x) = ax + b$ om
        $b$ te vinden;
  \item controleer met de andere waarde.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:linfunc:findaffine}
Vind de affiene functie met $f(2) = 7$ en $f(5) = 16$.
\[
a = \frac{16 - 7}{5 - 2} = \frac93 = 3 ,
\]
dan geeft $f(2) = 3 \times 2 + b = 7$ dat $b = 1$: $f(x) = 3x + 1$.
Controle: $f(5) = 15 + 1 = 16$.
\end{example}

\section{Percentages}

\begin{proposition}[Procentuele verandering]\label{prop:g9:linfunc:percent}
Een hoeveelheid met $t\,\%$ verhogen vermenigvuldigt haar met
$1 + \dfrac{t}{100}$; met $t\,\%$ verlagen vermenigvuldigt haar met
$1 - \dfrac{t}{100}$. Procentuele veranderingen zijn lineaire
functies.
\end{proposition}

\begin{proof}
$x$ met $t\,\%$ verhogen betekent $\frac{t}{100}\,x$ optellen:
\[
x + \frac{t}{100}\,x = \left(1 + \frac{t}{100}\right) x .
\]
Zelfde berekening met een minteken voor een verlaging.
\end{proof}

\begin{example}\label{ex:g9:linfunc:percent}
Een prijs van $80$ stijgt met $15\%$: nieuwe prijs
$80 \times 1.15 = 92$. Een prijs van $60$ daalt met $30\%$:
$60 \times 0.7 = 42$.

\emph{Veranderingen aaneenschakelen vermenigvuldigt de factoren.} Een
stijging van $20\%$ gevolgd door een daling van $20\%$ geeft
$1.2 \times 0.8 = 0.96$: overall een daling van $4\%$, geen terugkeer
naar het begin!
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g9:linfunc:1}
Welke van deze functies zijn lineair? affien? geen van beide?
\[
f(x) = 4x, \qquad
g(x) = 3x - 5, \qquad
h(x) = x^2, \qquad
k(x) = -\tfrac x2, \qquad
m(x) = 7 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:linfunc:2}
Laat $f(x) = 2x - 3$. Bereken $f(0)$, $f(4)$, $f(-1)$, en vind het
getal $x$ zodat $f(x) = 9$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:linfunc:3}
Een lineaire functie voldoet aan $f(6) = 15$. Vind haar coëfficiënt,
en bereken daarna $f(10)$ en $f(-2)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:linfunc:4}
Teken in hetzelfde assenstelsel de grafieken van $f(x) = 2x$,
$g(x) = 2x + 3$ en $h(x) = -x + 4$. Wat kan gezegd worden over de
grafieken van $f$ en $g$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:linfunc:5}
Vind de affiene functie zodat $f(1) = 5$ en $f(3) = 11$; daarna die
zodat $g(0) = 4$ en $g(2) = 0$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:linfunc:6}
Bereken: de prijs na een stijging van $25\%$ op $120$; de prijs na
een korting van $40\%$ op $65$; de oorspronkelijke prijs als een
artikel $69$ kost na een korting van $25\%$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:linfunc:7}
Telefoonplan A kost $10$ per maand plus $0.05$ per belminuut; plan B
kost $25$ per maand met onbeperkt bellen.
\begin{enumerate}
  \item Druk de maandelijkse kost van plan A uit als functie van het
        aantal $x$ minuten.
  \item Vanaf hoeveel minuten per maand is plan B goedkoper?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:linfunc:8}
Een populatie van $2000$ bacteriën groeit elk uur met $10\%$.
\begin{enumerate}
  \item Hoeveel bacteriën zijn er na één uur? Na twee uur?
  \item Leg uit waarom het antwoord na twee uur niet $2400$ is.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:linfunc:9}
De grafiek van een affiene functie gaat door de punten $(2, 3)$ en
$(6, 1)$.
\begin{enumerate}
  \item Bereken haar richtingscoëfficiënt. Is de functie stijgend of
        dalend?
  \item Geef haar uitdrukking, en bereken de input waarvoor de output
        $0$ is.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:linfunc:10}
Een winkel verhoogt eerst een prijs met $t\,\%$, en verlaagt daarna de
nieuwe prijs met $t\,\%$.
\begin{enumerate}
  \item Toon dat de eindprijs gelijk is aan de oorspronkelijke
        vermenigvuldigd met $1 - \dfrac{t^2}{10000}$.
  \item Leid af dat de eindprijs altijd \emph{lager} is dan de
        oorspronkelijke (voor $0 < t \leq 100$), en vind de overall
        procentuele verandering voor $t = 10$.
\end{enumerate}
\end{exercise}
