\chapter{Vierkantswortels}\label{ch:g9:sqrt}

Hoe lang is de diagonaal van een vierkant met zijde $1$? De stelling
van Pythagoras antwoordt $\sqrt 2$, een getal waarvan het kwadraat
$2$ is --- en dat geen \omterm{def:g9:fractions:fraction}{breuk} blijkt te zijn. Dit hoofdstuk definieert
\omterm{def:g9:sqrt:def}{vierkantswortels}, stelt de regels vast om ermee te rekenen, en leert
uitdrukkingen als $\sqrt{75}$ te vereenvoudigen.

\section{Definitie en eerste eigenschappen}

\begin{definition}[Vierkantswortel]\label{def:g9:sqrt:def}
Laat $a \geq 0$. De \emph{vierkantswortel}\index{vierkantswortel} van
$a$, genoteerd $\sqrt a$, is het unieke \emph{niet-negatieve} getal
waarvan het kwadraat $a$ is:
\[
\sqrt a \geq 0
\qquad\text{en}\qquad
\left(\sqrt a\right)^2 = a .
\]
Negatieve getallen hebben geen vierkantswortel, want elk kwadraat is
niet-negatief.
\end{definition}

\begin{example}
$\sqrt{49} = 7$, $\sqrt 0 = 0$, $\sqrt 1 = 1$,
$\sqrt{2.25} = 1.5$. De eerste \omterm{def:g9:sqrt:def}{vierkantswortels} om uit het hoofd te
kennen zijn die van de \emph{volmaakte kwadraten}: $1, 4, 9, 16, 25,
36, 49, 64, 81, 100, 121, 144$.
\end{example}

\begin{proposition}[Vierkantswortel van een kwadraat]\label{prop:g9:sqrt:square}
Voor elk reëel getal $x$ (positief of niet):
\[
\sqrt{x^2} = \abs{x} =
\begin{cases}
x & \text{als } x \geq 0,\\
-x & \text{als } x < 0 .
\end{cases}
\]
\end{proposition}

\begin{proof}
Het getal $\abs{x}$ is niet-negatief en zijn kwadraat is $x^2$ (een
getal en zijn \omterm{def:g7:negatives:def}{tegengestelde} hebben hetzelfde kwadraat). Als uniek
niet-negatief getal met kwadraat $x^2$ is het $\sqrt{x^2}$.
\end{proof}

\begin{example}
$\sqrt{(-5)^2} = \sqrt{25} = 5 = \abs{-5}$: de \omterm{def:g9:sqrt:def}{vierkantswortel}
``vergeet'' het teken, hij herstelt het niet. In het bijzonder is
$\sqrt{x^2} = x$ \emph{fout} voor negatieve $x$.
\end{example}

\section{Producten en quotiënten van vierkantswortels}

\begin{theorem}[Vermenigvuldigings- en delingsregels]\label{thm:g9:sqrt:rules}
Voor alle $a \geq 0$ en $b \geq 0$:
\[
\sqrt{a b} = \sqrt a \times \sqrt b,
\qquad\text{en voor } b > 0:\quad
\sqrt{\frac ab} = \frac{\sqrt a}{\sqrt b} .
\]
\end{theorem}

\begin{proof}
Het getal $\sqrt a \times \sqrt b$ is niet-negatief (\omterm{def:g2:mult:def}{product} van
niet-negatieve getallen), en zijn kwadraat is
\[
\left(\sqrt a \times \sqrt b\right)^2
= \left(\sqrt a\right)^2 \times \left(\sqrt b\right)^2
= a b .
\]
Door uniciteit van het niet-negatieve getal met kwadraat $ab$ is het
$\sqrt{ab}$. De quotiëntrregel wordt op dezelfde manier bewezen.
\end{proof}

\begin{remark}[Geen zulke regel voor sommen!]
$\sqrt{a + b}$ is \emph{niet} $\sqrt a + \sqrt b$ in het algemeen:
\[
\sqrt{9 + 16} = \sqrt{25} = 5,
\qquad\text{maar}\qquad
\sqrt 9 + \sqrt{16} = 3 + 4 = 7 .
\]
\end{remark}

\begin{method}[$\sqrt n$ vereenvoudigen]\label{met:g9:sqrt:simplify}
Om de \omterm{def:g9:sqrt:def}{vierkantswortel} van een geheel getal te vereenvoudigen:
\begin{enumerate}
  \item vind het grootste volmaakte kwadraat dat $n$ \omterm{def:g9:arith:divisor}{deelt} (ontbind
        $n$ indien nodig);
  \item splits met de productregel en haal het kwadraat eruit:
        $\sqrt{k^2 m} = k\sqrt m$;
  \item controleer dat geen volmaakt kwadraat meer \omterm{def:g9:arith:divisor}{deelt} wat onder
        de wortel overblijft.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:sqrt:simplify}
Vereenvoudig $\sqrt{75}$: omdat $75 = 25 \times 3$,
\[
\sqrt{75} = \sqrt{25 \times 3} = \sqrt{25} \times \sqrt 3 = 5\sqrt3 .
\]
Vereenvoudig $\sqrt{72}$: omdat $72 = 36 \times 2$,
$\sqrt{72} = 6\sqrt2$. En een \omterm{def:g3:division:remainder}{quotiënt}:
$\sqrt{\dfrac{49}{16}} = \dfrac{\sqrt{49}}{\sqrt{16}} = \dfrac74$.
\end{example}

\begin{example}[Vierkantswortels optellen]\label{ex:g9:sqrt:add}
Sommen van \omterm{def:g9:sqrt:def}{vierkantswortels} vereenvoudigen alleen wanneer de wortels
\emph{gelijksoortig} zijn. Bereken $\sqrt{12} + \sqrt{27} - \sqrt{48}$,
eerst elke term vereenvoudigend:
\begin{align*}
\sqrt{12} &= \sqrt{4 \times 3} = 2\sqrt3, \\
\sqrt{27} &= \sqrt{9 \times 3} = 3\sqrt3, \\
\sqrt{48} &= \sqrt{16 \times 3} = 4\sqrt3,
\end{align*}
dus de \omterm{def:g1:addition:def}{som} is $2\sqrt3 + 3\sqrt3 - 4\sqrt3 = (2 + 3 - 4)\sqrt3 =
\sqrt3$.
\end{example}

\begin{example}[Uitwerken met vierkantswortels]\label{ex:g9:sqrt:expand}
De algebraïsche identiteiten gelden voor \omterm{def:g9:sqrt:def}{vierkantswortels}. Werk
$\left(\sqrt5 + 2\right)^2$ uit met $(a+b)^2 = a^2 + 2ab + b^2$:
\[
\left(\sqrt5 + 2\right)^2 = 5 + 2 \times 2\sqrt5 + 4 = 9 + 4\sqrt5 .
\]
En met de derde identiteit, $(a+b)(a-b) = a^2 - b^2$:
\[
\left(\sqrt7 + \sqrt3\right)\left(\sqrt7 - \sqrt3\right) = 7 - 3 = 4 :
\]
het \omterm{def:g2:mult:def}{product} van twee irrationale getallen kan een geheel getal zijn.
\end{example}

\section{De vergelijking \texorpdfstring{$x^2 = a$}{x2 = a}}

\begin{theorem}[Oplossen van $x^2 = a$]\label{thm:g9:sqrt:equation}
\begin{enumerate}
  \item Als $a > 0$, heeft de \omterm{def:g8:equations:def}{vergelijking} $x^2 = a$ precies twee
        oplossingen: $\sqrt a$ en $-\sqrt a$;
  \item als $a = 0$, is de enige oplossing $0$;
  \item als $a < 0$, is er geen oplossing.
\end{enumerate}
\end{theorem}

\begin{proof}
Herschrijf $x^2 = a$ als $x^2 - \left(\sqrt a\right)^2 = 0$ wanneer
$a \geq 0$, en ontbind als \omterm{ex:g1:subtraction:difference}{verschil} van kwadraten:
\[
\left(x - \sqrt a\right)\left(x + \sqrt a\right) = 0 ,
\]
dus $x = \sqrt a$ of $x = -\sqrt a$ (die samenvallen wanneer
$a = 0$). Wanneer $a < 0$ is er geen oplossing, want $x^2 \geq 0 > a$
voor elke $x$.
\end{proof}

\begin{example}
$x^2 = 16$: oplossingen $4$ en $-4$. \quad
$x^2 = 5$: oplossingen $\sqrt5$ en $-\sqrt5$ (exacte waarden;
$\sqrt5 \approx 2.236$). \quad
$x^2 = -9$: geen oplossing. \quad
$3x^2 = 21$: deel eerst door 3, $x^2 = 7$, oplossingen $\pm\sqrt7$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g9:sqrt:1}
Bereken zonder rekenmachine:
\[
\sqrt{64}, \qquad \sqrt{100}, \qquad \sqrt{0.09}, \qquad
\sqrt{\tfrac{1}{25}}, \qquad \left(\sqrt{13}\right)^2, \qquad
\sqrt{(-4)^2} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:sqrt:2}
Vereenvoudig:
\[
\sqrt{50}, \qquad \sqrt{45}, \qquad \sqrt{98}, \qquad \sqrt{300} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:sqrt:3}
Bereken en vereenvoudig:
\[
\sqrt2 \times \sqrt{18}, \qquad
\frac{\sqrt{75}}{\sqrt3}, \qquad
\sqrt5 \times \sqrt{20}, \qquad
\frac{\sqrt{8}}{\sqrt{50}} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:sqrt:4}
Los op: $x^2 = 36$; \quad $x^2 = 11$; \quad $x^2 + 4 = 0$; \quad
$2x^2 = 50$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:sqrt:5}
Herleid tot één term: $\sqrt{20} + \sqrt{45}$, daarna
$3\sqrt8 - \sqrt{18} + \sqrt2$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:sqrt:6}
Werk uit en vereenvoudig:
\[
\left(\sqrt3 + 1\right)^2, \qquad
\left(2\sqrt5 - 3\right)^2, \qquad
\left(\sqrt6 + \sqrt2\right)\left(\sqrt6 - \sqrt2\right).
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:sqrt:7}
Een vierkant veld heeft \omterm{def:g6:measure:area}{oppervlakte} $6400$ m$^2$. Wat is de lengte
van zijn zijde? Van zijn diagonaal (exacte waarde, daarna afgerond
op de dichtstbijzijnde meter)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:sqrt:8}
Toon dat $\dfrac{1}{\sqrt2} = \dfrac{\sqrt2}{2}$ (vermenigvuldig
\omterm{def:g4:fractions:def}{teller} en \omterm{def:g4:fractions:def}{noemer} met $\sqrt2$). Gebruik dezelfde truc om
$\dfrac{6}{\sqrt3}$ en $\dfrac{10}{\sqrt5}$ te schrijven zonder
\omterm{def:g9:sqrt:def}{vierkantswortel} in de \omterm{def:g4:fractions:def}{noemer}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:sqrt:9}
Waar of onwaar? Motiveer met een bewijs of een tegenvoorbeeld.
\begin{enumerate}
  \item Voor alle $a, b \geq 0$: $\sqrt{ab} = \sqrt a \sqrt b$.
  \item Voor alle $a, b \geq 0$: $\sqrt{a+b} = \sqrt a + \sqrt b$.
  \item Voor alle $x$: $\sqrt{x^2} = x$.
  \item $\left(3\sqrt2\right)^2 = 18$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:sqrt:10}
Laat $x = \sqrt{7 + 4\sqrt3}$. Bereken
$\left(2 + \sqrt3\right)^2$, en leid een eenvoudigere uitdrukking
van $x$ af. Zelfde vraag voor $\sqrt{9 - 4\sqrt5}$ (mik op een
kwadraat van de vorm $\left(\sqrt5 - 2\right)^2$, en let op het
teken).
\end{exercise}
