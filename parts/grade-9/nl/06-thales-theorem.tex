\chapter{De stelling van Thales}\label{ch:g9:thales}

Hoe meet je de hoogte van een piramide zonder erop te klimmen? Thales
van Milete vergeleek haar schaduw met de schaduw van een stok. Zijn
stelling --- evenwijdige lijnen snijden segmenten in evenredige stukken
--- is het wiskundige hart van schaalmodellen, kaarten en vergrotingen,
en een van de oudste stellingen met een naam.

\section{De stelling}

\begin{theorem}[Thales]\label{thm:g9:thales:direct}
Laat twee lijnen snijden in een punt $A$. Neem $M$ op de eerste lijn
en $N$ op de tweede, met $B$ verderop op de eerste en $C$ verderop op
de tweede. Als de lijnen $(MN)$ en $(BC)$ \emph{evenwijdig} zijn, dan
hebben de driehoeken $AMN$ en $ABC$ evenredige zijden:
\[
\frac{AM}{AB} = \frac{AN}{AC} = \frac{MN}{BC}.
\]
\end{theorem}
\admitted

\begin{method}[Een lengte berekenen met Thales]\label{met:g9:thales:compute}
\begin{enumerate}
  \item Identificeer het punt $A$ waar de twee lijnen snijden en de
        twee evenwijdige lijnen; noem de twee driehoeken;
  \item schrijf de drie gelijke verhoudingen, altijd ``kleine
        driehoek over grote driehoek'';
  \item houd de gelijkheid van de twee verhoudingen met de drie
        bekende lengten en de onbekende;
  \item los de resulterende evenredigheid op door kruisproducten.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:thales:compute}
In de linkerconfiguratie hierboven, stel $AM = 3$, $AB = 5$,
$AN = 2.4$ en $MN = 3.3$, en bereken $AC$ en $BC$. Thales geeft
\[
\frac{AM}{AB} = \frac{AN}{AC} = \frac{MN}{BC},
\qquad\text{d.w.z.}\qquad
\frac35 = \frac{2.4}{AC} = \frac{3.3}{BC}.
\]
Uit $\frac35 = \frac{2.4}{AC}$, kruisproduct:
$3 \times AC = 5 \times 2.4 = 12$, dus $AC = 4$. Uit
$\frac35 = \frac{3.3}{BC}$: $3 \times BC = 16.5$, dus $BC = 5.5$.
\end{example}

\begin{example}[Thales in het echt]\label{ex:g9:thales:stick}
Een verticale stok van hoogte $1$ m werpt een schaduw van $1.5$ m,
terwijl een boom op hetzelfde moment een schaduw van $12$ m werpt.
Zonnestralen zijn evenwijdig, dus de stok-en-schaduw- en
boom-en-schaduw-driehoeken staan in Thales-configuratie:
\[
\frac{\text{hoogte van de boom}}{1} = \frac{12}{1.5},
\qquad\text{dus de boom is } 8 \text{ m hoog.}
\]
\end{example}

\section{Het omgekeerde}

\begin{theorem}[Omgekeerde van Thales]\label{thm:g9:thales:converse}
Met de punten geplaatst zoals in \cref{thm:g9:thales:direct} ($M$ op
$(AB)$, $N$ op $(AC)$, in dezelfde volgorde op beide lijnen):
als
\[
\frac{AM}{AB} = \frac{AN}{AC},
\]
dan zijn de lijnen $(MN)$ en $(BC)$ evenwijdig.
\end{theorem}
\admitted

\begin{method}[Evenwijdigheid bewijzen of weerleggen]\label{met:g9:thales:converse}
\begin{enumerate}
  \item Bereken apart de twee verhoudingen $\dfrac{AM}{AB}$ en
        $\dfrac{AN}{AC}$ (als breuken, niet afrondingen);
  \item als ze gelijk zijn \emph{en} de punten in dezelfde volgorde op
        de twee lijnen staan, zijn de lijnen evenwijdig (omgekeerde
        van Thales);
  \item als ze verschillen, zijn de lijnen \emph{niet} evenwijdig ---
        want als ze dat waren, zou de stelling van Thales de
        verhoudingen gelijk forceren.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:thales:converse}
Op lijn 1: $AM = 4$ en $AB = 10$; op lijn 2: $AN = 6$ en $AC = 15$.
Dan
\[
\frac{AM}{AB} = \frac{4}{10} = \frac25,
\qquad
\frac{AN}{AC} = \frac{6}{15} = \frac25 :
\]
gelijke verhoudingen, zelfde volgorde: $(MN) \parallel (BC)$.

Met $AC = 14$ in plaats daarvan: $\frac{6}{14} = \frac37 \neq
\frac25$, dus de lijnen zijn niet evenwijdig.
\end{example}

\section{Vergroting en verkleining}

\begin{definition}[Een figuur schalen]\label{def:g9:thales:scaling}
Een figuur vergroten of verkleinen met de
\emph{schaalfactor}\index{schaalfactor} $k > 0$ betekent \emph{alle}
haar lengten met $k$ vermenigvuldigen: een \emph{vergroting} wanneer
$k > 1$, een \emph{verkleining} wanneer $k < 1$. In de
Thales-configuratie is de driehoek $AMN$ de verkleining van $ABC$ met
de factor $k = \frac{AM}{AB}$.
\end{definition}

\begin{proposition}[Effect op hoeken en oppervlakten]\label{prop:g9:thales:scaling}
Schalen met de factor $k$ behoudt hoeken en de vorm van figuren,
vermenigvuldigt elke lengte met $k$, en vermenigvuldigt elke
\emph{oppervlakte} met $k^2$.
\end{proposition}
\admitted

  \node[font=\small, omDef] at (1,-0.45) {$2 \times 1.2$};

  \node[font=\small] at (5.4,1.2) {opp. $9.6$};
\end{tikzpicture}

{\small Lengten verdubbelen ($k = 2$) vermenigvuldigt de oppervlakte
met $k^2 = 4$: vier kopieĂ«n van de kleine rechthoek betegelen de
grote.}
\end{omfigure}

\begin{example}
Een foto van $10 \times 15$ cm wordt vergroot met schaalfactor $k = 3$:
de afdruk meet $30 \times 45$ cm. Haar oppervlakte gaat van
$150$ cm$^2$ naar $150 \times 9 = 1350$ cm$^2$ --- $9$ keer groter,
niet $3$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g9:thales:1}
De lijnen $(MN)$ en $(BC)$ zijn evenwijdig, met $M$ op $[AB]$ en $N$
op $[AC]$; $AM = 2$, $AB = 6$, $AN = 3$, $BC = 9$. Bereken $AC$ en
$MN$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:thales:2}
Zelfde configuratie: $AM = 5$, $MB = 3$ (let op: $MB$, niet $AB$!),
$AN = 4$. Bereken $AC$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:thales:3}
In een vlinderconfiguratie ligt $A$ tussen $M$ en $B$ en tussen $N$ en
$C$, met $(MN) \parallel (BC)$; $AM = 3$, $AB = 7.5$, $AN = 2$,
$MN = 2.6$. Bereken $AC$ en $BC$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:thales:4}
$M$ ligt op $[AB]$ met $AM = 6$, $AB = 8$; $N$ ligt op $[AC]$ met
$AN = 9$, $AC = 12$. Zijn de lijnen $(MN)$ en $(BC)$ evenwijdig?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:5}
Zelfde als hierboven met $AM = 4$, $AB = 6$, $AN = 5$, $AC = 8$. Zijn
$(MN)$ en $(BC)$ evenwijdig? Motiveer zorgvuldig.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:6}
Een persoon van $1.8$ m staat $2$ m van de voet van een straatlamp; hun
schaduw meet $3$ m. Teken de Thales-configuratie gevormd door de lamp,
de persoon en de top van de schaduw, en bereken de hoogte van de lamp.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:7}
Een kaart heeft schaal $1 : 25\,000$ (lengten op de kaart zijn de
echte lengten vermenigvuldigd met $k = \frac{1}{25000}$).
\begin{enumerate}
  \item Twee dorpen liggen $6.8$ cm uit elkaar op de kaart. Wat is de
        echte afstand, in km?
  \item Een bos heeft een oppervlakte van $8$ cm$^2$ op de kaart. Wat
        is zijn echte oppervlakte, in km$^2$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:thales:8}
Driehoek $ABC$ heeft $AB = 12$, $AC = 15$, $BC = 18$. Het punt $M$ op
$[AB]$ voldoet aan $AM = 8$, en de lijn door $M$ evenwijdig aan
$(BC)$ snijdt $[AC]$ in $N$.
\begin{enumerate}
  \item Bereken $AN$ en $MN$.
  \item Wat is de schaalfactor van $ABC$ naar $AMN$? Vergelijk de
        omtrekken van de twee driehoeken.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:thales:9}
Laat $ABCD$ een trapezium zijn met $(AB) \parallel (CD)$, $AB = 4$ en
$CD = 6$, waarvan de diagonalen $[AC]$ en $[BD]$ snijden in $O$.
Gebruik Thales in de vlinderconfiguratie rond $O$ om de verhouding
$\dfrac{OA}{OC}$ te berekenen, en toon dat $O$ beide diagonalen in
dezelfde verhouding snijdt.
\end{exercise}
