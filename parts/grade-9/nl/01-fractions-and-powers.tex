\chapter{Breuken en machten}\label{ch:g9:fractions}

Breuken en machten zijn de dagelijkse gereedschappen van het rekenen:
hoeveelheden delen, verhoudingen vergelijken, zeer grote of zeer
kleine getallen schrijven. Dit hoofdstuk consolideert de regels om
ermee te rekenen --- met elke tussentijdse stap uitgeschreven --- en
introduceert wetenschappelijke notatie.

\section{Rekenen met breuken}

\begin{definition}[Breuk]\label{def:g9:fractions:fraction}
Een \emph{breuk}\index{breuk} $\dfrac ab$ (met $b \neq 0$) is het
quotiënt van $a$ door $b$. Twee breuken zijn gelijk wanneer de ene
uit de andere ontstaat door teller en noemer met hetzelfde getal
ongelijk aan nul te vermenigvuldigen (of te delen):
\[
\frac ab = \frac{a \times k}{b \times k} \qquad (k \neq 0).
\]
\end{definition}

\begin{example}\label{ex:g9:fractions:simplify}
Vereenvoudig $\dfrac{42}{56}$ stap voor stap:
\[
\frac{42}{56} = \frac{42 \div 2}{56 \div 2} = \frac{21}{28}
= \frac{21 \div 7}{28 \div 7} = \frac34 .
\]
(In \cref{ch:g9:arith} doet de grootste gemene deler dit in één
stap.)
\end{example}

\begin{method}[Breuken optellen en aftrekken]\label{met:g9:fractions:add}
\begin{enumerate}
  \item Zet beide breuken op een \emph{gemeenschappelijke noemer}
        (een gemeenschappelijk veelvoud van de twee noemers);
  \item tel of trek de tellers op/af, met behoud van de noemer;
  \item vereenvoudig het resultaat indien mogelijk.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:fractions:add}
Bereken $\dfrac56 + \dfrac34$. Een gemeenschappelijke noemer van $6$
en $4$ is $12$:
\[
\frac56 + \frac34
= \frac{5 \times 2}{6 \times 2} + \frac{3 \times 3}{4 \times 3}
= \frac{10}{12} + \frac{9}{12}
= \frac{19}{12}.
\]
Bereken $2 - \dfrac37$. Schrijf $2$ als breuk over $7$:
\[
2 - \frac37 = \frac{14}{7} - \frac37 = \frac{11}{7}.
\]
\end{example}

\begin{method}[Breuken vermenigvuldigen en delen]\label{met:g9:fractions:mult}
\begin{enumerate}
  \item Om te vermenigvuldigen, vermenigvuldig tellers onderling en
        noemers onderling: $\dfrac ab \times \dfrac cd = \dfrac{ac}{bd}$
        (vereenvoudig \emph{vóór} het vermenigvuldigen wanneer
        mogelijk);
  \item om te delen, vermenigvuldig met het omgekeerde van de deler:
        $\dfrac ab \div \dfrac cd = \dfrac ab \times \dfrac dc$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:fractions:mult}
\[
\frac{7}{15} \times \frac{25}{14}
= \frac{7 \times 25}{15 \times 14}
= \frac{7 \times 25}{15 \times 2 \times 7}
= \frac{25}{30} = \frac56,
\]
waarbij we met de gemeenschappelijke factor $7$ vereenvoudigden,
daarna met $5$. En een deling:
\[
\frac{3}{4} \div \frac{9}{8}
= \frac34 \times \frac89
= \frac{3 \times 8}{4 \times 9}
= \frac{24}{36} = \frac23 .
\]
\end{example}

\section{Machten}

\begin{definition}[Macht]\label{def:g9:fractions:power}
Voor een reëel getal $a$ en een positief geheel getal $n$ is de
\emph{macht}\index{macht} $a^n$ het product van $n$ factoren gelijk
aan $a$:
\[
a^n = \underbrace{a \times a \times \dots \times a}_{n \text{ factoren}} .
\]
Per conventie $a^0 = 1$ (voor $a \neq 0$), en negatieve exponenten
noteren omgekeerden:
\[
a^{-n} = \frac{1}{a^n}.
\]
\end{definition}

\begin{example}
$2^4 = 2 \times 2 \times 2 \times 2 = 16$; \quad
$(-3)^2 = 9$ maar $-3^2 = -(3^2) = -9$ (de exponent bindt vóór het
minteken); \quad $10^{-3} = \frac{1}{1000} = 0.001$; \quad
$5^1 = 5$ en $5^0 = 1$.
\end{example}

\begin{theorem}[Regels voor exponenten]\label{thm:g9:fractions:rules}
Voor alle $a$, $b$ ongelijk aan nul en alle gehele $m$, $n$:
\[
a^m \times a^n = a^{m+n},
\qquad
\frac{a^m}{a^n} = a^{m-n},
\qquad
\left(a^m\right)^n = a^{mn},
\qquad
(ab)^n = a^n b^n .
\]
\end{theorem}

\begin{proof}
Voor positieve exponenten, tel de factoren. Voor de eerste regel:
$a^m \times a^n$ is $m$ factoren $a$ gevolgd door $n$ factoren $a$,
in totaal $m + n$ factoren. Voor de derde:
$\left(a^m\right)^n$ herhaalt $n$ keer een blok van $m$ factoren,
wat $mn$ factoren geeft. De andere twee regels zijn gelijkaardig; de
uitbreiding tot nul en negatieve exponenten wordt gecontroleerd vanuit
$a^{-n} = \frac{1}{a^n}$ (en maakt de conventie $a^0 = 1$ de enige
consistente keuze, want $a^n \times a^0$ moet $a^{n+0} = a^n$ zijn).
\end{proof}

\begin{example}\label{ex:g9:fractions:rules}
Vereenvoudig stap voor stap:
\[
\frac{3^7 \times 3^{-2}}{3^4}
= \frac{3^{7 + (-2)}}{3^4}
= \frac{3^5}{3^4}
= 3^{5-4} = 3^1 = 3 .
\]
En met twee letters: $(2a^3)^2 \times a = 4 a^6 \times a = 4a^7$.
\end{example}

\section{Machten van tien en wetenschappelijke notatie}

\begin{proposition}[Machten van tien]\label{prop:g9:fractions:ten}
Voor een positief geheel getal $n$: $10^n$ wordt geschreven als ``$1$
gevolgd door $n$ nullen'', en $10^{-n} = 0.0\dots01$ met de $1$ op de
$n$-de decimale plaats. Een decimaal getal vermenigvuldigen met
$10^n$ (resp.\ $10^{-n}$) verschuift de decimale punt $n$ plaatsen
naar rechts (resp.\ links).
\end{proposition}

\begin{definition}[Wetenschappelijke notatie]\label{def:g9:fractions:scientific}
De \emph{wetenschappelijke notatie}\index{wetenschappelijke notatie}
van een positief decimaal getal is de unieke manier om het te schrijven
als
\[
a \times 10^n,
\qquad\text{met } 1 \leq a < 10 \text{ en } n \text{ een geheel getal.}
\]
\end{definition}

\begin{example}\label{ex:g9:fractions:scientific}
De Aarde--Zon-afstand is ongeveer $149\,600\,000$ km $= 1.496 \times
10^8$ km. De grootte van een watermolecule is ongeveer
$0.000\,000\,000\,28$ m $= 2.8 \times 10^{-10}$ m. Om met zulke
getallen te rekenen, groepeer de decimale delen en de machten van
tien:
\[
(3 \times 10^5) \times (2.5 \times 10^{-2})
= (3 \times 2.5) \times 10^{5 + (-2)}
= 7.5 \times 10^{3}.
\]
\end{example}

\begin{method}[Een getal in wetenschappelijke notatie zetten]\label{met:g9:fractions:scientific}
\begin{enumerate}
  \item Verplaats de decimale punt net na het eerste van-nul-verschillende
        cijfer; dit geeft de factor $a$ met $1 \leq a < 10$;
  \item tel het aantal plaatsen dat de punt verplaatste: die telling
        is $n$, positief als het originele getal $\geq 10$ was,
        negatief als het $< 1$ was;
  \item controleer: $a \times 10^n$ moet het originele getal
        reproduceren.
\end{enumerate}
\end{method}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g9:fractions:1}
Bereken en geef het resultaat als volledig vereenvoudigde breuk:
\[
\frac13 + \frac15, \qquad
\frac72 - \frac53, \qquad
\frac{5}{12} + \frac38, \qquad
3 - \frac45 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:fractions:2}
Bereken en vereenvoudig:
\[
\frac59 \times \frac{27}{35}, \qquad
\frac{8}{15} \div \frac{4}{5}, \qquad
\frac23 \times \frac34 \times \frac45 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:fractions:3}
Bereken:
\[
2^5, \qquad (-2)^4, \qquad -2^4, \qquad 4^{-2}, \qquad
\left(\tfrac23\right)^3, \qquad 7^0 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:fractions:4}
Schrijf als één macht:
\[
5^3 \times 5^6, \qquad
\frac{7^9}{7^5}, \qquad
\left(2^3\right)^4, \qquad
\frac{3^2 \times 3^7}{3^5}, \qquad
2^5 \times 4 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:fractions:5}
Schrijf in wetenschappelijke notatie: $45\,000$; $0.0072$; $302.5$;
$0.000\,000\,9$; twaalf miljoen.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:fractions:6}
Bereken, en geef het resultaat in wetenschappelijke notatie:
\[
(2 \times 10^7) \times (4.5 \times 10^{-3}),
\qquad
\frac{9 \times 10^5}{3 \times 10^{-2}},
\qquad
(5 \times 10^3)^2 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:fractions:7}
Een tank is $\frac35$ vol. Na toevoegen van $30$ liter is hij
$\frac34$ vol. Wat is de capaciteit van de tank? (Druk eerst de
toegevoegde breuk van de tank uit.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:fractions:8}
Alice eet $\frac14$ van een taart, daarna eet Ben $\frac13$ van wat
overblijft, daarna eet Carla $\frac12$ van wat nog overblijft. Welk
deel van de taart blijft er aan het eind over? (Bereken de rest na
elke stap.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:fractions:9}
Licht reist $3 \times 10^8$ meters per seconde. De Zon is ongeveer
$1.5 \times 10^{11}$ m ver. Hoe lang doet zonlicht erover om ons te
bereiken? Geef het antwoord in seconden (wetenschappelijke notatie
niet nodig), daarna in minuten en seconden.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:fractions:10}
Welke is groter, $2^{30}$ of $3^{20}$? (Hint: schrijf beide als
machten met exponent $10$ met $\left(a^m\right)^n = a^{mn}$, en
vergelijk $2^3$ met $3^2$.)
\end{exercise}
