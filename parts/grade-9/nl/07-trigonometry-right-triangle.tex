\chapter{Goniometrie in de rechthoekige driehoek}\label{ch:g9:trig}

In een rechthoekige driehoek is, zodra je één scherpe hoek kent, de
\emph{vorm} van de driehoek vast --- alleen zijn grootte kan variëren.
De verhoudingen van zijn zijden hangen daarom alleen van de hoek af:
deze verhoudingen zijn cosinus, sinus en tangens. Samen met de stelling
van Pythagoras laten ze je elke zijde en elke hoek van een
rechthoekige driehoek berekenen uit zeer weinig informatie.

\section{De stelling van Pythagoras, nogmaals}

\begin{theorem}[Pythagoras]\label{thm:g9:trig:pythagoras}
In een driehoek $ABC$ rechthoekig in $A$ is het kwadraat van de
schuine zijde de som van de kwadraten van de twee benen:
\[
BC^2 = AB^2 + AC^2 .
\]
Omgekeerd, als $BC^2 = AB^2 + AC^2$ in een driehoek, dan is de
driehoek rechthoekig in $A$.
\end{theorem}
\admitted

\begin{example}\label{ex:g9:trig:pythagoras}
Een rechthoekige driehoek heeft benen $AB = 5$ en $AC = 12$. Dan
\[
BC^2 = 5^2 + 12^2 = 25 + 144 = 169,
\qquad\text{dus}\quad BC = \sqrt{169} = 13 .
\]
Om een \emph{been} te vinden: als $BC = 10$ en $AB = 6$, dan
$AC^2 = BC^2 - AB^2 = 100 - 36 = 64$, dus $AC = 8$.
\end{example}

\section{Cosinus, sinus, tangens}

In een rechthoekige driehoek, ten opzichte van een scherpe hoek
$\theta$, hebben de drie zijden namen: de \emph{schuine zijde}
(tegenover de rechte hoek, de langste zijde), de zijde
\emph{tegenover} $\theta$, en de zijde \emph{aanliggend} aan
$\theta$ (het been dat $\theta$ raakt).

\begin{definition}[Goniometrische verhoudingen]\label{def:g9:trig:ratios}
Voor een scherpe hoek $\theta$ van een rechthoekige driehoek:
\[
\cos\theta = \frac{\text{aanliggend}}{\text{schuine zijde}},
\qquad
\sin\theta = \frac{\text{tegenover}}{\text{schuine zijde}},
\qquad
\tan\theta = \frac{\text{tegenover}}{\text{aanliggend}} .
\]
Deze verhoudingen\index{cosinus}\index{sinus}\index{tangens} hangen
alleen van $\theta$ af, niet van de grootte van de driehoek (alle
rechthoekige driehoeken met dezelfde scherpe hoek zijn schalingen van
elkaar, door Thales).
\end{definition}

\begin{proposition}[Eerste eigenschappen]\label{prop:g9:trig:properties}
Voor elke scherpe hoek $\theta$:
\begin{enumerate}
  \item $0 < \cos\theta < 1$ en $0 < \sin\theta < 1$ (een been is
        korter dan de schuine zijde);
  \item $\cos^2\theta + \sin^2\theta = 1$;
  \item $\tan\theta = \dfrac{\sin\theta}{\cos\theta}$.
\end{enumerate}
\end{proposition}

\begin{proof}
Schrijf $a$ voor de aanliggende zijde, $o$ voor de overstaande zijde,
$h$ voor de schuine zijde van een rechthoekige driehoek met hoek
$\theta$.

1. $0 < a < h$ en $0 < o < h$, dus de twee verhoudingen liggen strikt
tussen $0$ en $1$.

2. Door Pythagoras, $a^2 + o^2 = h^2$. Deel alles door $h^2$:
\[
\cos^2\theta + \sin^2\theta
= \frac{a^2}{h^2} + \frac{o^2}{h^2}
= \frac{a^2 + o^2}{h^2} = 1 .
\]

3. $\dfrac{\sin\theta}{\cos\theta} = \dfrac{o/h}{a/h} = \dfrac oa
= \tan\theta$.
\end{proof}

\begin{method}[Een zijde vinden]\label{met:g9:trig:side}
Om een onbekende zijde te berekenen wanneer één zijde en één scherpe
hoek bekend zijn:
\begin{enumerate}
  \item markeer de bekende hoek en label de drie zijden (schuine
        zijde, overstaand, aanliggend) \emph{vanuit die hoek};
  \item kies de verhouding (cos, sin of tan) die de bekende zijde en
        de gevraagde zijde betrekt;
  \item schrijf de vergelijking en los die op;
  \item saneiteitscontrole: de schuine zijde moet het langst
        uitkomen.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:trig:side}
In een rechthoekige driehoek meet de schuine zijde $8$ en is één
hoek $35^\circ$; bereken de zijde ertegenover. De verhouding met de
overstaande zijde en de schuine zijde is de sinus:
\[
\sin 35^\circ = \frac{\text{overstaand}}{8}
\quad\Longrightarrow\quad
\text{overstaand} = 8 \sin 35^\circ \approx 8 \times 0.574 \approx 4.6 .
\]
Voor de aanliggende zijde, gebruik de cosinus:
$8\cos 35^\circ \approx 8 \times 0.819 \approx 6.6$. Controle met
Pythagoras: $4.6^2 + 6.6^2 \approx 21 + 43 \approx 64 = 8^2$.
\end{example}

\begin{method}[Een hoek vinden]\label{met:g9:trig:angle}
Wanneer twee zijden bekend zijn, bereken de verhouding die ze vormen,
identificeer die als cosinus, sinus of tangens van de onbekende hoek,
en pas de inverse functie van de rekenmachine toe ($\cos^{-1}$,
$\sin^{-1}$ of $\tan^{-1}$) op de verhouding.
\end{method}

\begin{example}\label{ex:g9:trig:angle}
Een ladder van $5$ m leunt tegen een muur, met de voet $1.4$ m van de
muur. De hoek $\theta$ tussen de ladder en de grond voldoet aan
\[
\cos\theta = \frac{1.4}{5} = 0.28,
\qquad\text{dus}\quad
\theta = \cos^{-1}(0.28) \approx 73.7^\circ .
\]
\end{example}

\begin{example}[Speciale hoeken]\label{ex:g9:trig:special}
Twee driehoeken geven exacte waarden. De helft van een vierkant met
zijde 1 (een rechthoekige gelijkbenige driehoek) heeft schuine zijde
$\sqrt2$ en hoeken van $45^\circ$:
\[
\cos 45^\circ = \sin 45^\circ = \frac{1}{\sqrt2} = \frac{\sqrt2}{2},
\qquad \tan 45^\circ = 1 .
\]
De helft van een gelijkzijdige driehoek met zijde 1 geeft de hoeken
$30^\circ$ en $60^\circ$:
\[
\cos 60^\circ = \sin 30^\circ = \frac12,
\qquad
\cos 30^\circ = \sin 60^\circ = \frac{\sqrt3}{2}.
\]
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g9:trig:1}
Een rechthoekige driehoek heeft benen $9$ en $12$. Bereken zijn
schuine zijde. Een andere heeft schuine zijde $17$ en één been $8$:
bereken het andere been.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:trig:2}
Een driehoek heeft zijden $7$, $24$ en $25$. Is hij rechthoekig?
Zelfde vraag voor zijden $5$, $6$ en $8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:trig:3}
In een driehoek $DEF$ rechthoekig in $D$, met de hoek in $E$ genoteerd
$\theta$: welke zijde is de schuine zijde? Welke zijde ligt tegenover
$\theta$? Schrijf $\cos\theta$, $\sin\theta$ en $\tan\theta$ als
verhoudingen van de zijden $DE$, $DF$, $EF$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:trig:4}
In een rechthoekige driehoek meet de schuine zijde $10$ en is één
scherpe hoek $28^\circ$. Bereken de twee benen (rond af op de tiende;
$\cos 28^\circ \approx 0.883$, $\sin 28^\circ \approx 0.469$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:trig:5}
In een rechthoekige driehoek meet het been aanliggend aan de hoek
$\theta$ $6$ en het overstaande been $4.5$. Bereken $\tan\theta$,
daarna $\theta$ (rond af op de graad; $\tan^{-1}(0.75) \approx
36.9^\circ$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:trig:6}
Een drone vliegt op een hoogte van $120$ m. Vanaf een waarnemer op de
grond wordt hij gezien onder een elevatiehoek van $32^\circ$. Op
welke horizontale afstand van de waarnemer is de drone
($\tan 32^\circ \approx 0.625$)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:trig:7}
Een toegangshelling moet $1.2$ m stijgen met een hoek van hoogstens
$5^\circ$ met de horizontaal. Welke minimale lengte langs de helling
moet de helling hebben ($\sin 5^\circ \approx 0.0872$)? Rond naar boven
af op de dichtstbijzijnde tiende van een meter.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:trig:8}
Gebruik \cref{prop:g9:trig:properties}: een scherpe hoek $\theta$
voldoet aan $\cos\theta = 0.6$. Bereken $\sin\theta$ zonder $\theta$
te vinden, daarna $\tan\theta$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:trig:9}
Motiveer de exacte waarden van \cref{ex:g9:trig:special}: in de
rechthoekige gelijkbenige driehoek met benen $1$, bereken de schuine
zijde; in de gelijkzijdige driehoek met zijde $1$, bereken de hoogte,
en leid cosinus en sinus van $30^\circ$ en $60^\circ$ af.
\end{exercise}
