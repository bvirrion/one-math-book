\chapter{Fractions and Powers}\label{ch:g9:fractions}

\omterm{def:g9:fractions:fraction}{Fractions} and \omterm{def:g9:fractions:power}{powers} are the everyday tools of computation: sharing
quantities, comparing proportions, writing very large or very small
numbers. This chapter consolidates the rules for computing with them ---
with every intermediate step written out --- and introduces \omterm{def:g9:fractions:scientific}{scientific
notation}.

\section{Computing with fractions}

\begin{definition}[Fraction]\label{def:g9:fractions:fraction}
A \emph{fraction}\index{fraction} $\dfrac ab$ (with $b \neq 0$) is the
\omterm{def:g3:division:remainder}{quotient} of $a$ by $b$. Two fractions are equal when one is obtained from
the other by multiplying (or dividing) \omterm{def:g4:fractions:def}{numerator} and \omterm{def:g4:fractions:def}{denominator} by the
same nonzero number:
\[
\frac ab = \frac{a \times k}{b \times k} \qquad (k \neq 0).
\]
\end{definition}

\begin{example}\label{ex:g9:fractions:simplify}
Simplify $\dfrac{42}{56}$ step by step:
\[
\frac{42}{56} = \frac{42 \div 2}{56 \div 2} = \frac{21}{28}
= \frac{21 \div 7}{28 \div 7} = \frac34 .
\]
(In \cref{ch:g9:arith} the greatest common \omterm{def:g5:division:multiple}{divisor} will do this in one
step.)
\end{example}

\begin{method}[Adding and subtracting fractions]\label{met:g9:fractions:add}
\begin{enumerate}
  \item Put both \omterm{def:g9:fractions:fraction}{fractions} over a \emph{common \omterm{def:g4:fractions:def}{denominator}} (a common
        \omterm{def:g5:division:multiple}{multiple} of the two \omterm{def:g4:fractions:def}{denominators});
  \item add or subtract the \omterm{def:g4:fractions:def}{numerators}, keeping the \omterm{def:g4:fractions:def}{denominator};
  \item simplify the result if possible.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:fractions:add}
Compute $\dfrac56 + \dfrac34$. A common \omterm{def:g4:fractions:def}{denominator} of $6$ and $4$ is
$12$:
\[
\frac56 + \frac34
= \frac{5 \times 2}{6 \times 2} + \frac{3 \times 3}{4 \times 3}
= \frac{10}{12} + \frac{9}{12}
= \frac{19}{12}.
\]
Compute $2 - \dfrac37$. Write $2$ as a \omterm{def:g9:fractions:fraction}{fraction} over $7$:
\[
2 - \frac37 = \frac{14}{7} - \frac37 = \frac{11}{7}.
\]
\end{example}

\begin{method}[Multiplying and dividing fractions]\label{met:g9:fractions:mult}
\begin{enumerate}
  \item To multiply, multiply \omterm{def:g4:fractions:def}{numerators} together and \omterm{def:g4:fractions:def}{denominators}
        together: $\dfrac ab \times \dfrac cd = \dfrac{ac}{bd}$
        (simplify \emph{before} multiplying whenever possible);
  \item to divide, multiply by the \omterm{def:g8:fractions:inverse}{inverse} of the \omterm{def:g5:division:multiple}{divisor}:
        $\dfrac ab \div \dfrac cd = \dfrac ab \times \dfrac dc$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g9:fractions:mult}
\[
\frac{7}{15} \times \frac{25}{14}
= \frac{7 \times 25}{15 \times 14}
= \frac{7 \times 25}{15 \times 2 \times 7}
= \frac{25}{30} = \frac56,
\]
where we simplified by the common factor $7$, then by $5$. And a
division:
\[
\frac{3}{4} \div \frac{9}{8}
= \frac34 \times \frac89
= \frac{3 \times 8}{4 \times 9}
= \frac{24}{36} = \frac23 .
\]
\end{example}

\section{Powers}

\begin{definition}[Power]\label{def:g9:fractions:power}
For a real number $a$ and a positive integer $n$, the
\emph{power}\index{power} $a^n$ is the \omterm{def:g2:mult:def}{product} of $n$ factors equal to
$a$:
\[
a^n = \underbrace{a \times a \times \dots \times a}_{n \text{ factors}} .
\]
By convention $a^0 = 1$ (for $a \neq 0$), and negative \omterm{def:g8:powers:def}{exponents} denote
\omterm{def:g8:fractions:inverse}{inverses}:
\[
a^{-n} = \frac{1}{a^n}.
\]
\end{definition}

\begin{example}
$2^4 = 2 \times 2 \times 2 \times 2 = 16$; \quad
$(-3)^2 = 9$ but $-3^2 = -(3^2) = -9$ (the \omterm{def:g8:powers:def}{exponent} binds before the
minus sign); \quad $10^{-3} = \frac{1}{1000} = 0.001$; \quad
$5^1 = 5$ and $5^0 = 1$.
\end{example}

\begin{theorem}[Rules of exponents]\label{thm:g9:fractions:rules}
For all nonzero $a$, $b$ and all integers $m$, $n$:
\[
a^m \times a^n = a^{m+n},
\qquad
\frac{a^m}{a^n} = a^{m-n},
\qquad
\left(a^m\right)^n = a^{mn},
\qquad
(ab)^n = a^n b^n .
\]
\end{theorem}

\begin{proof}
For positive \omterm{def:g8:powers:def}{exponents}, count the factors. For the first rule:
$a^m \times a^n$ is $m$ factors $a$ followed by $n$ factors $a$, in total
$m + n$ factors. For the third:
$\left(a^m\right)^n$ repeats $n$ times a block of $m$ factors, giving
$mn$ factors. The other two rules are similar; the extension to \omterm{def:g1:counting:zero}{zero} and
negative \omterm{def:g8:powers:def}{exponents} is checked from $a^{-n} = \frac{1}{a^n}$ (and makes
the convention $a^0 = 1$ the only consistent choice, since
$a^n \times a^0$ must be $a^{n+0} = a^n$).
\end{proof}

\begin{example}\label{ex:g9:fractions:rules}
Simplify step by step:
\[
\frac{3^7 \times 3^{-2}}{3^4}
= \frac{3^{7 + (-2)}}{3^4}
= \frac{3^5}{3^4}
= 3^{5-4} = 3^1 = 3 .
\]
And with two letters: $(2a^3)^2 \times a = 4 a^6 \times a = 4a^7$.
\end{example}

\section{Powers of ten and scientific notation}

\begin{proposition}[Powers of ten]\label{prop:g9:fractions:ten}
For a positive integer $n$: $10^n$ is written ``$1$ followed by $n$
\omterm{def:g1:counting:zero}{zeros}'', and $10^{-n} = 0.0\dots01$ with the $1$ in the $n$-th decimal
place. Multiplying a \omterm{def:g6:decimals:places}{decimal number} by $10^n$ (resp.\ $10^{-n}$) shifts
its \omterm{def:g4:decimals:point}{decimal point} $n$ places to the right (resp.\ left).
\end{proposition}

\begin{definition}[Scientific notation]\label{def:g9:fractions:scientific}
The \emph{scientific notation}\index{scientific notation} of a positive
\omterm{def:g6:decimals:places}{decimal number} is the unique way of writing it as
\[
a \times 10^n,
\qquad\text{with } 1 \leq a < 10 \text{ and } n \text{ an integer.}
\]
\end{definition}

\begin{example}\label{ex:g9:fractions:scientific}
The Earth--Sun distance is about $149\,600\,000$ km $= 1.496 \times
10^8$ km. The size of a water molecule is about $0.000\,000\,000\,28$ m
$= 2.8 \times 10^{-10}$ m. To compute with such numbers, group the
decimal parts and the \omterm{def:g9:fractions:power}{powers} of ten:
\[
(3 \times 10^5) \times (2.5 \times 10^{-2})
= (3 \times 2.5) \times 10^{5 + (-2)}
= 7.5 \times 10^{3}.
\]
\end{example}

\begin{method}[Putting a number in scientific notation]\label{met:g9:fractions:scientific}
\begin{enumerate}
  \item Move the \omterm{def:g4:decimals:point}{decimal point} just after the first nonzero digit; this
        gives the factor $a$ with $1 \leq a < 10$;
  \item count the number of places the point moved: that count is $n$,
        positive if the original number was $\geq 10$, negative if it was
        $< 1$;
  \item check: $a \times 10^n$ must reproduce the original number.
\end{enumerate}
\end{method}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g9:fractions:1}
Compute and give the result as a fully simplified \omterm{def:g9:fractions:fraction}{fraction}:
\[
\frac13 + \frac15, \qquad
\frac72 - \frac53, \qquad
\frac{5}{12} + \frac38, \qquad
3 - \frac45 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:fractions:2}
Compute and simplify:
\[
\frac59 \times \frac{27}{35}, \qquad
\frac{8}{15} \div \frac{4}{5}, \qquad
\frac23 \times \frac34 \times \frac45 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:fractions:3}
Compute:
\[
2^5, \qquad (-2)^4, \qquad -2^4, \qquad 4^{-2}, \qquad
\left(\tfrac23\right)^3, \qquad 7^0 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:fractions:4}
Write as a single \omterm{def:g9:fractions:power}{power}:
\[
5^3 \times 5^6, \qquad
\frac{7^9}{7^5}, \qquad
\left(2^3\right)^4, \qquad
\frac{3^2 \times 3^7}{3^5}, \qquad
2^5 \times 4 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g9:fractions:5}
Write in \omterm{def:g9:fractions:scientific}{scientific notation}: $45\,000$; $0.0072$; $302.5$;
$0.000\,000\,9$; twelve million.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:fractions:6}
Compute, giving the result in \omterm{def:g9:fractions:scientific}{scientific notation}:
\[
(2 \times 10^7) \times (4.5 \times 10^{-3}),
\qquad
\frac{9 \times 10^5}{3 \times 10^{-2}},
\qquad
(5 \times 10^3)^2 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:fractions:7}
A tank is $\frac35$ full. After adding $30$ liters it is $\frac34$ full.
What is the \omterm{def:g2:measure:units}{capacity} of the tank? (Express the added \omterm{def:g9:fractions:fraction}{fraction} of the tank
first.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:fractions:8}
Alice eats $\frac14$ of a cake, then Ben eats $\frac13$ of what remains,
then Carla eats $\frac12$ of what still remains. What \omterm{def:g9:fractions:fraction}{fraction} of the
cake is left at the end? (Compute the \omterm{def:g3:division:remainder}{remainder} after each step.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g9:fractions:9}
Light travels $3 \times 10^8$ meters per second. The Sun is about
$1.5 \times 10^{11}$ m away. How long does sunlight take to reach us?
Give the answer in seconds (\omterm{def:g9:fractions:scientific}{scientific notation} not needed), then in
minutes and seconds.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g9:fractions:10}
Which is larger, $2^{30}$ or $3^{20}$? (Hint: write both as \omterm{def:g9:fractions:power}{powers} with
\omterm{def:g8:powers:def}{exponent} $10$ using $\left(a^m\right)^n = a^{mn}$, and compare $2^3$ with
$3^2$.)
\end{exercise}
