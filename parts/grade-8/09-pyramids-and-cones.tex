\chapter{Pyramids and Cones}\label{ch:g8:solids}

After the \omterm{def:g7:areas:prism}{prisms} and \omterm{def:g7:areas:prism}{cylinders} of \cref{ch:g7:areas} --- \omterm{def:g5:solids:def}{solids} with
constant cross-section --- come the \omterm{def:g5:solids:def}{solids} with a point: \omterm{def:g8:solids:def}{pyramids} and
\omterm{def:g8:solids:def}{cones}. Their \omterm{def:g6:measure:volume}{volume} formula carries a famous factor $\frac13$, which
this chapter makes plausible and uses; the study of \omterm{def:g5:solids:def}{solids} continues
with spheres and sections in \cref{ch:g9:solids}.

\section{Describing pyramids and cones}

\begin{definition}[Pyramid, cone]\label{def:g8:solids:def}
\begin{itemize}
  \item A \emph{pyramid}\index{pyramid} has a \omterm{def:g4:geometry:polygon}{polygon} as its
        \emph{base} and a point, the \emph{apex}, joined to every
        \omterm{def:g5:solids:def}{vertex} of the base by triangular \omterm{def:g5:solids:def}{faces}. Its \emph{height} is
        the distance from the apex to the plane of the base.
  \item A \emph{cone}\index{cone} is the same construction over a
        disk: a base disk, an apex, and a curved lateral surface.
\end{itemize}
A pyramid is \emph{regular} when its base is a regular \omterm{def:g4:geometry:polygon}{polygon}
(square, \omterm{def:g6:shapes:triangles}{equilateral triangle}, \dots) and its apex sits vertically
above the center of the base.
\end{definition}

\begin{example}[Counting faces, edges, vertices]\label{ex:g8:solids:count}
A \omterm{def:g8:solids:def}{pyramid} with a square base has $5$ \omterm{def:g5:solids:def}{faces} ($1$ square $+ 4$
triangles), $8$ edges and $5$ \omterm{def:g5:solids:def}{vertices}. With an $n$-sided base:
$n + 1$ \omterm{def:g5:solids:def}{faces}, $2n$ edges, $n + 1$ \omterm{def:g5:solids:def}{vertices}. Check Euler's little
pattern: \omterm{def:g5:solids:def}{faces} $+$ \omterm{def:g5:solids:def}{vertices} $=$ edges $+ 2$.
\end{example}

\section{Volume}

\begin{theorem}[Volume of a pyramid or cone]\label{thm:g8:solids:volume}
For a \omterm{def:g8:solids:def}{pyramid} or a \omterm{def:g8:solids:def}{cone} with base \omterm{def:g6:measure:area}{area} $B$ and height $h$:
\[
V = \frac{1}{3}\, B \times h
\]
--- one third of the \omterm{def:g7:areas:prism}{prism} or \omterm{def:g7:areas:prism}{cylinder} with the same base and height.
For a \omterm{def:g8:solids:def}{cone} of base \omterm{def:g3:shapes:shapes}{radius} $r$: $V = \frac13 \pi r^2 h$.
\end{theorem}
\admitted

\begin{remark}[Why one third?]
Fill a hollow \omterm{def:g8:solids:def}{pyramid} with water and pour it into the \omterm{def:g7:areas:prism}{prism} of same
base and height: it takes exactly three fills. Even better: a \omterm{def:g5:solids:def}{cube}
can be cut into three identical \omterm{def:g8:solids:def}{pyramids}, each having a \omterm{def:g5:solids:def}{face} of the
\omterm{def:g5:solids:def}{cube} as base and one \omterm{def:g5:solids:def}{vertex} of the \omterm{def:g5:solids:def}{cube} as apex (try to picture it!).
A general proof uses integration, a tool of the High School volume.
\end{remark}

\begin{example}\label{ex:g8:solids:volume}
A \omterm{def:g8:solids:def}{pyramid} has a square base of side $5$ cm and height $9$ cm:
\begin{enumerate}
  \item base \omterm{def:g6:measure:area}{area}: $B = 5^2 = 25$ cm$^2$;
  \item \omterm{def:g6:measure:volume}{volume}: $V = \frac13 \times 25 \times 9 = 25 \times 3 = 75$
        cm$^3$.
\end{enumerate}
An ice-cream \omterm{def:g8:solids:def}{cone} of \omterm{def:g3:shapes:shapes}{radius} $3$ cm and height $10$ cm:
\[
V = \frac13 \times \pi \times 3^2 \times 10 = 30\pi \approx 94
\text{ cm}^3 .
\]
\end{example}

\begin{example}[Careful with the height]\label{ex:g8:solids:slant}
For a \omterm{def:g8:solids:def}{cone}, do not confuse the height $h$ (apex to center of the
base, \omterm{def:g6:lines:perp}{perpendicular}) with the \emph{slant height} $s$ (apex to the
rim). They are related by Pythagoras
(\cref{thm:g8:pythagoras:direct}) in the \omterm{def:g6:shapes:triangles}{right triangle}
apex--center--rim:
\[
s^2 = h^2 + r^2 .
\]
A \omterm{def:g8:solids:def}{cone} with $r = 3$ and $s = 5$ therefore has height
$h = \sqrt{25 - 9} = 4$, and \omterm{def:g6:measure:volume}{volume}
$\frac13 \pi \times 9 \times 4 = 12\pi$.
\end{example}

\begin{method}[Volume problems]\label{met:g8:solids:method}
\begin{enumerate}
  \item Identify the \omterm{def:g5:solids:def}{solid} (\omterm{def:g7:areas:prism}{prism}/\omterm{def:g7:areas:prism}{cylinder}: $V = Bh$; \omterm{def:g8:solids:def}{pyramid}/\omterm{def:g8:solids:def}{cone}:
        $V = \frac13 Bh$);
  \item compute the base \omterm{def:g6:measure:area}{area} $B$ first, as a separate step;
  \item check that the height is \omterm{def:g6:lines:perp}{perpendicular} to the base (use
        Pythagoras if the slant height is given);
  \item multiply, keep units consistent, convert at the end if needed
        ($1$ L $= 1000$ cm$^3$).
\end{enumerate}
\end{method}

\section{Nets}

\begin{example}[Net of a pyramid]\label{ex:g8:solids:net}
Unfolding a regular square-based \omterm{def:g8:solids:def}{pyramid} flattens it into its
\emph{\omterm{def:g5:solids:net}{net}}: the square base with four identical \omterm{def:g6:shapes:triangles}{isosceles triangles}
attached to its sides. The triangles' equal sides are the
\emph{lateral edges} of the \omterm{def:g8:solids:def}{pyramid} --- their length is neither $h$
nor the slant height of a \omterm{def:g5:solids:def}{face}, so label carefully before cutting
cardboard.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g8:solids:1}
How many \omterm{def:g5:solids:def}{faces}, edges and \omterm{def:g5:solids:def}{vertices} has a \omterm{def:g8:solids:def}{pyramid} with a triangular
base (a \emph{tetrahedron})? With a hexagonal base?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:2}
Compute the \omterm{def:g6:measure:volume}{volume} of a \omterm{def:g8:solids:def}{pyramid} with base \omterm{def:g6:measure:area}{area} $36$ cm$^2$ and
height $10$ cm; of a \omterm{def:g8:solids:def}{pyramid} with rectangular base
$6 \times 4$ cm and height $7.5$ cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:3}
Compute the \omterm{def:g6:measure:volume}{volume} of a \omterm{def:g8:solids:def}{cone} of \omterm{def:g3:shapes:shapes}{radius} $5$ cm and height $9$ cm
(exact with $\pi$, then rounded to the cm$^3$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:4}
A \omterm{def:g7:areas:prism}{cylinder} and a \omterm{def:g8:solids:def}{cone} have the same \omterm{def:g3:shapes:shapes}{radius} $4$ cm and the same height
$12$ cm. Compute both \omterm{def:g6:measure:volume}{volumes}. What is their ratio?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:5}
A \omterm{def:g8:solids:def}{pyramid} has \omterm{def:g6:measure:volume}{volume} $56$ cm$^3$ and height $8$ cm. What is its base
\omterm{def:g6:measure:area}{area}? (Write the \omterm{def:g6:measure:volume}{volume} formula and solve.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:6}
Sketch the \omterm{def:g5:solids:net}{net} of a \omterm{def:g8:solids:def}{cone} (a disk for the base, and a sector of a
larger disk for the lateral surface). Which measurement of the \omterm{def:g8:solids:def}{cone}
is the \omterm{def:g3:shapes:shapes}{radius} of that sector: the height $h$ or the slant height
$s$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:7}
The Louvre \omterm{def:g8:solids:def}{pyramid} has a square base of side about $35$ m and a
height of about $22$ m. Estimate its \omterm{def:g6:measure:volume}{volume}, to the nearest hundred
cubic meters.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:8}
A \omterm{def:g8:solids:def}{cone} has slant height $13$ cm and base \omterm{def:g3:shapes:shapes}{radius} $5$ cm. Compute its
height, then its \omterm{def:g6:measure:volume}{volume} (exact with $\pi$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:9}
A conical glass of \omterm{def:g3:shapes:shapes}{radius} $4$ cm and height $9$ cm is filled to the
brim, then poured into a cylindrical glass of \omterm{def:g3:shapes:shapes}{radius} $4$ cm. What
height does the liquid reach in the \omterm{def:g7:areas:prism}{cylinder}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:10}
An hourglass is made of two identical \omterm{def:g8:solids:def}{cones} (\omterm{def:g3:shapes:shapes}{radius} $2$ cm, height
$4.5$ cm) joined at their apexes. All the sand fills exactly one
\omterm{def:g8:solids:def}{cone}. Compute the \omterm{def:g6:measure:volume}{volume} of sand (exact, then in cm$^3$ rounded to
the tenth), and the \omterm{def:g6:fractions:def}{fraction} of the hourglass's total inner \omterm{def:g6:measure:volume}{volume}
occupied by the sand.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:solids:11}
A square-based \omterm{def:g8:solids:def}{pyramid} with base side $6$ cm has its apex directly
above a corner of the base, at height $8$ cm.
\begin{enumerate}
  \item Does the formula $V = \frac13 Bh$ still apply? (It does ---
        the apex need not be centered. Compute $V$.)
  \item Compute the length of the longest lateral edge, from the apex
        to the opposite corner of the base (two Pythagoras steps: the
        base diagonal first).
\end{enumerate}
\end{exercise}

\section{Problem: A cube cut into three pyramids}

\begin{problem}[{Weekend problem --- where the factor $\frac13$
comes from, and what happens to a pyramid cut at half height}]
\label{pb:g8:solids:1}
The remark after \cref{thm:g8:solids:volume} claims that a \omterm{def:g5:solids:def}{cube}
can be cut into three identical \omterm{def:g8:solids:def}{pyramids} --- ``try to picture
it!''. This problem pictures it precisely, gets the famous factor
$\frac13$ out of it honestly, then cuts the \omterm{def:g5:solids:def}{cube} a second way into
\emph{six} \omterm{def:g8:solids:def}{pyramids}, and finishes by slicing a \omterm{def:g8:solids:def}{pyramid} at \omterm{def:g3:division:half}{half} its
height --- with a result few people guess right on the first try.

Throughout, $ABCDEFGH$ is a \omterm{def:g5:solids:def}{cube} of side $a$: base $ABCD$, top
\omterm{def:g5:solids:def}{face} $EFGH$, with $E$ above $A$, $F$ above $B$, $G$ above $C$ and
$H$ above $D$.

\medskip
\noindent\textbf{Part I --- Three \omterm{def:g8:solids:def}{pyramids} sharing an apex.}
\begin{enumerate}
  \item The \omterm{def:g5:solids:def}{vertex} $G$ belongs to three \omterm{def:g5:solids:def}{faces} of the \omterm{def:g5:solids:def}{cube}. List
        the three \omterm{def:g5:solids:def}{faces} that do \emph{not} contain $G$, and
        describe the three \omterm{def:g8:solids:def}{pyramids} obtained by taking each of
        them as a base, with apex $G$ every time. How many \omterm{def:g5:solids:def}{faces},
        edges and \omterm{def:g5:solids:def}{vertices} has each \omterm{def:g8:solids:def}{pyramid}
        (\cref{ex:g8:solids:count})?
  \item Explain why turning the \omterm{def:g5:solids:def}{cube} by a third of a turn around
        its long diagonal $(AG)$ leaves the \omterm{def:g5:solids:def}{cube} unchanged but
        shuffles the three bases of question 1 in a cycle ---
        and why the three \omterm{def:g8:solids:def}{pyramids} are therefore identical
        copies of one another.
  \item Admitting that the three \omterm{def:g8:solids:def}{pyramids} fill the \omterm{def:g5:solids:def}{cube} without
        overlapping (a cardboard model is quite convincing),
        deduce the \omterm{def:g6:measure:volume}{volume} of each one.
  \item Check this against the formula of
        \cref{thm:g8:solids:volume}: the \omterm{def:g8:solids:def}{pyramid} with base $ABCD$
        and apex $G$ has its apex directly above a \emph{corner}
        of its base, exactly as in \cref{exo:g8:solids:11}.
        Identify its height, apply the formula, and compare.
  \item For a \omterm{def:g5:solids:def}{cube} of side $6$ cm, give the \omterm{def:g6:measure:volume}{volume} of each of the
        three \omterm{def:g8:solids:def}{pyramids} --- and explain what this decomposition
        has to do with the water-pouring experiment of the
        chapter (three fills of the \omterm{def:g8:solids:def}{pyramid} for one \omterm{def:g7:areas:prism}{prism}).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Six \omterm{def:g8:solids:def}{pyramids} sharing the center.}
Now let $O$ be the center of the \omterm{def:g5:solids:def}{cube}, and join $O$ to the four
\omterm{def:g5:solids:def}{vertices} of each of the six \omterm{def:g5:solids:def}{faces}.
\begin{enumerate}[resume]
  \item Describe the six \omterm{def:g8:solids:def}{pyramids} obtained, and explain why they
        are identical copies of one another. What is the height
        of each one?
  \item Deduce, by sharing the \omterm{def:g5:solids:def}{cube}'s \omterm{def:g6:measure:volume}{volume}, the \omterm{def:g6:measure:volume}{volume} of each
        of the six \omterm{def:g8:solids:def}{pyramids}.
  \item Recompute that \omterm{def:g6:measure:volume}{volume} with the formula
        $V = \frac13 B h$, and check the two answers agree.
  \item The two decompositions confirm the factor $\frac13$ for
        some very special \omterm{def:g8:solids:def}{pyramids}. Explain why they do not yet
        \emph{prove} \cref{thm:g8:solids:volume} for every
        \omterm{def:g8:solids:def}{pyramid} and \omterm{def:g8:solids:def}{cone} --- and say how the chapter deals with
        that gap: which statement is admitted, which experiment
        supports it, and in which \omterm{def:g6:measure:volume}{volume} of the series is it
        honestly proved.
  \item A \omterm{def:g5:solids:def}{cube} of side $6$ cm is cut into its six center
        \omterm{def:g8:solids:def}{pyramids}. Compute the \omterm{def:g6:measure:volume}{volume} of each, both ways.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Cutting a \omterm{def:g8:solids:def}{pyramid} at \omterm{def:g3:division:half}{half} height.}
$SABCD$ is a regular square-based \omterm{def:g8:solids:def}{pyramid}: base side $c$, height
$h$, apex $S$ above the center of the base. Cut it by the plane
through the \emph{midpoints} of the four lateral edges $[SA]$,
$[SB]$, $[SC]$, $[SD]$.
\begin{enumerate}[resume]
  \item Apply the midpoint theorem
        (\cref{thm:g8:midpoints:direct}) in the triangle $SAB$:
        what are the direction and length of the \omterm{def:g6:lines:objects}{segment} joining
        the midpoints of $[SA]$ and $[SB]$? Deduce that the cut
        is a square of side $\frac{c}{2}$, \omterm{def:g6:lines:perp}{parallel} to the base,
        at height $\frac{h}{2}$.
  \item The piece above the cut is itself a square-based \omterm{def:g8:solids:def}{pyramid}.
        Give its base side and height, and show that its \omterm{def:g6:measure:volume}{volume}
        is exactly \emph{one eighth} of the original \omterm{def:g6:measure:volume}{volume} $V$.
  \item Deduce the \omterm{def:g6:measure:volume}{volume} of the lower piece (the \emph{frustum},
        the shape of an unfinished \omterm{def:g8:solids:def}{pyramid}). In what ratio does
        the half-height cut share the \omterm{def:g6:measure:volume}{volume}?
  \item The Louvre \omterm{def:g8:solids:def}{pyramid} (\cref{exo:g8:solids:7}: base side
        $35$ m, height $22$ m) is cleaned in two campaigns: the
        glass above \omterm{def:g3:division:half}{half} height, then the glass below. What
        \omterm{def:g6:fractions:def}{fraction} of the \emph{\omterm{def:g6:measure:volume}{volume}} sits below \omterm{def:g3:division:half}{half} height?
        Estimate it in cubic meters, to the nearest hundred.
  \item The general moral: if \emph{all} the dimensions of a
        \omterm{def:g8:solids:def}{pyramid} are multiplied by $2$, what happens to its
        \omterm{def:g6:measure:volume}{volume}? By what factor does the \omterm{def:g6:measure:volume}{volume} grow when every
        dimension is multiplied by $k$ --- and how does this
        explain, in one line, the ``one eighth'' of question 12?
\end{enumerate}
\end{problem}
