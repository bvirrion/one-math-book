\chapter{The Pythagorean Theorem}\label{ch:g8:pythagoras}

The most famous theorem of all mathematics relates the three sides of
a right triangle. With it, lengths become computable that no ruler
could measure directly: diagonals, distances, heights. It is also our
first big encounter with the difference between a theorem and its
converse.

\section{The theorem}

\begin{theorem}[Pythagoras]\label{thm:g8:pythagoras:direct}
If a triangle $ABC$ is right-angled at $A$, then
\[
BC^2 = AB^2 + AC^2 :
\]
the square of the hypotenuse (the side opposite the right angle)
equals the sum of the squares of the other two sides.
\end{theorem}

\begin{proof}[Idea of proof]
Four copies of the triangle, arranged inside a big square of side
$AB + AC$, leave uncovered a tilted square built on the hypotenuse:
comparing areas gives the equality. The detailed computation is
proposed as \cref{exo:g8:pythagoras:11}; the theorem also drops out
of the scalar product, studied in the High School volume.
\end{proof}

\begin{method}[Computing a length]\label{met:g8:pythagoras:compute}
In a right triangle where two sides are known:
\begin{enumerate}
  \item name the hypotenuse (opposite the right angle --- always the
        longest side);
  \item write the equality of \cref{thm:g8:pythagoras:direct} with
        the known values;
  \item solve for the missing square: \emph{add} the squares if the
        hypotenuse is unknown, \emph{subtract} if a leg is unknown;
  \item take the square root, exactly or with a calculator, and check
        that the hypotenuse came out longest.
\end{enumerate}
\end{method}

\begin{example}[Finding the hypotenuse]\label{ex:g8:pythagoras:hypotenuse}
A right triangle has legs $6$ and $8$. Hypotenuse $c$:
\[
c^2 = 6^2 + 8^2 = 36 + 64 = 100,
\qquad
c = \sqrt{100} = 10 .
\]
(The number $\sqrt{100}$, the \emph{square root} of $100$, is the
positive number whose square is $100$; square roots get a full
chapter in \cref{ch:g9:sqrt}.)
\end{example}

\begin{example}[Finding a leg]\label{ex:g8:pythagoras:leg}
A $5$ m ladder leans against a wall, its foot $1.4$ m from the wall.
Height reached $h$: the ladder is the hypotenuse, so
\[
h^2 = 5^2 - 1.4^2 = 25 - 1.96 = 23.04,
\qquad
h = \sqrt{23.04} = 4.8 \text{ m}.
\]
Subtract, not add: the unknown here is a \emph{leg}.
\end{example}

\section{The converse}

\begin{theorem}[Converse of Pythagoras]\label{thm:g8:pythagoras:converse}
If, in a triangle $ABC$, the sides satisfy $BC^2 = AB^2 + AC^2$, then
the triangle is right-angled at $A$.
\end{theorem}
\admitted

\begin{method}[Testing a right angle]\label{met:g8:pythagoras:test}
Given the three sides of a triangle:
\begin{enumerate}
  \item compute separately the square of the longest side, and the sum
        of the squares of the two others;
  \item if the two results are equal, the triangle is right-angled
        (converse), at the vertex opposite the longest side;
  \item if they differ, it is not right-angled --- because Pythagoras'
        theorem would force equality.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:pythagoras:test}
Sides $5$, $12$, $13$: longest side squared, $13^2 = 169$; sum of the
other squares, $5^2 + 12^2 = 25 + 144 = 169$. Equal: right-angled
(opposite the side $13$).

Sides $6$, $7$, $9$: $9^2 = 81$ but $6^2 + 7^2 = 85 \neq 81$: not a
right triangle (it is ``almost'' one --- the numbers decide, not the
eye).
\end{example}

\begin{remark}[Theorem vs converse]
The theorem and its converse say different things: the theorem
\emph{uses} a right angle to compute lengths; the converse uses
lengths to \emph{prove} a right angle. Builders have used the
converse for millennia: a rope with knots at $3$, $4$, $5$ units
pulled taut gives a perfect right angle.
\end{remark}

\begin{example}[Diagonal of a square]\label{ex:g8:pythagoras:diagonal}
The diagonal of a square of side $1$ is the hypotenuse of a right
triangle with legs $1$ and $1$:
\[
d^2 = 1^2 + 1^2 = 2 ,
\]
so $d = \sqrt2 \approx 1.414$ --- a number that is not a fraction, as
the High School volume will prove. For a square of side $c$, the
diagonal is $c\sqrt2$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:1}
In each right triangle, compute the hypotenuse:
legs $9$ and $12$; \quad legs $5$ and $12$; \quad legs $8$ and $15$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:2}
A right triangle has hypotenuse $25$ and one leg $7$. Compute the
other leg.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:3}
Compute the diagonal of a rectangle of sides $12$ cm and $9$ cm; the
diagonal of a square of side $5$ cm (exact value with a square root,
then rounded to the mm).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:4}
Which triangles are right-angled? Justify with
\cref{met:g8:pythagoras:test}:
\[
(20,\ 21,\ 29); \qquad
(7,\ 8,\ 11); \qquad
(1.5,\ 2,\ 2.5).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:5}
A gate is braced by a diagonal plank. The gate is $3$ m wide and
$1.6$ m tall: how long is the plank?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:6}
An isosceles triangle has two sides of $10$ cm and a base of $12$ cm.
Its height splits it into two right triangles: compute that height,
then the area of the triangle.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:pythagoras:7}
A television screen is a $16{:}9$ rectangle measuring $88.5$ cm by
$49.8$ cm. Screens are sold by the diagonal, in inches
($1$ inch $= 2.54$ cm). Compute the diagonal in cm, then in inches
(round to the nearest inch).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:pythagoras:8}
A boat sails $24$ km east, then $10$ km north. How far is it from its
starting point (as the crow flies)? Draw the situation first.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:pythagoras:9}
A $2.5$ m ladder must reach a window sill $2.4$ m above the ground.
How far from the wall must its foot be placed? Is the answer
compatible with safety advice (foot at about a quarter of the ladder
length from the wall)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:pythagoras:10}
On a coordinate grid, plot $A(1, 2)$ and $B(5, 5)$, and draw the
right triangle with legs parallel to the axes whose hypotenuse is
$[AB]$. Compute $AB$. (This construction, for any two points, becomes
the distance formula of coordinate geometry, in the High School
volume.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:pythagoras:11}
Four copies of a right triangle with legs $a$, $b$ and hypotenuse $c$
are placed in the corners of a square of side $a + b$, leaving a
tilted square of side $c$ uncovered in the middle.
\begin{enumerate}
  \item Express the area of the big square in two ways: directly, and
        as (four triangles) $+$ (tilted square).
  \item Expand $(a + b)^2$ (see \cref{thm:g8:equations:expand}) and
        deduce $c^2 = a^2 + b^2$: you have proved the Pythagorean
        theorem.
\end{enumerate}
\end{exercise}
