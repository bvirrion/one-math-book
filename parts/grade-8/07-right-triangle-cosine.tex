\chapter{The Right Triangle and the Cosine}\label{ch:g8:cosine}

Two beautiful facts tie the \omterm{def:g6:shapes:triangles}{right triangle} to the circle: a \omterm{def:g6:shapes:triangles}{right
triangle} fits exactly in a half-circle, its hypotenuse being a
\omterm{def:g4:geometry:circle}{diameter}. And one number, the \emph{cosine} of an angle, encodes the
shape of every \omterm{def:g6:shapes:triangles}{right triangle} with that angle --- the first
trigonometric ratio, ahead of its siblings sine and tangent
(\cref{ch:g9:trig}).

\section{The right triangle and its circle}

\begin{theorem}[Circle theorem]\label{thm:g8:cosine:circle}
\begin{enumerate}
  \item If a triangle $ABC$ is \omterm{def:g6:shapes:triangles}{right-angled} at $A$, then $A$ lies on
        the circle whose \omterm{def:g4:geometry:circle}{diameter} is the hypotenuse $[BC]$.
  \item Conversely, if $A$ lies on a circle of \omterm{def:g4:geometry:circle}{diameter} $[BC]$ (with
        $A \neq B, C$), then the triangle $ABC$ is \omterm{def:g6:shapes:triangles}{right-angled} at
        $A$.
\end{enumerate}
Equivalently: in a \omterm{def:g6:shapes:triangles}{right triangle}, the midpoint of the hypotenuse is
at equal distance from the three \omterm{def:g5:solids:def}{vertices} --- the median from the
\omterm{def:g3:shapes:rightangle}{right angle} measures \omterm{def:g3:division:half}{half} the hypotenuse.
\end{theorem}

\begin{proof}[Proof of 1]
Let $O$ be the midpoint of $[BC]$ and $D$ the \omterm{def:g7:central:def}{symmetric} of $A$ about
$O$. The diagonals of $ABDC$ cut at their common midpoint $O$, so
$ABDC$ is a \omterm{def:g7:central:parallelogram}{parallelogram} (\cref{def:g7:central:parallelogram}) ---
with a \omterm{def:g3:shapes:rightangle}{right angle} at $A$: it is a rectangle. The diagonals of a
rectangle are equal, so $OA = \frac{AD}{2} = \frac{BC}{2}$: the
point $A$ is at distance $\frac{BC}{2}$ from $O$, i.e.\ on the circle
of \omterm{def:g4:geometry:circle}{diameter} $[BC]$. (Point 2 is proved by running the argument
backwards.)
\end{proof}

\begin{example}\label{ex:g8:cosine:circle}
A triangle has a hypotenuse of $10$ cm. Without knowing anything
else, the median from the \omterm{def:g3:shapes:rightangle}{right angle} measures $5$ cm, and the
\omterm{pb:g6:symmetry:1}{circumscribed circle} of the triangle has \omterm{def:g3:shapes:shapes}{radius} $5$ cm, centered at
the midpoint of the hypotenuse.
\end{example}

\section{The cosine of an acute angle}

\begin{definition}[Cosine]\label{def:g8:cosine:def}
In a \omterm{def:g6:shapes:triangles}{right triangle}, for an acute angle $\theta$:
\[
\cos\theta = \frac{\text{side adjacent to } \theta}{\text{hypotenuse}}
\]
--- the leg touching $\theta$, divided by the hypotenuse. This ratio
depends only on the angle, not on the size of the triangle: all \omterm{def:g6:shapes:triangles}{right
triangles} with the same acute angle are enlargements of one another,
and enlargements preserve ratios of lengths
(\cref{ch:g8:midpoints} began this story; \cref{ch:g9:thales}
finishes it).
\end{definition}

that common value is $\cos\theta$.}
\end{omfigure}

\begin{proposition}[First values and bounds]\label{prop:g8:cosine:bounds}
For every acute angle $\theta$: $0 < \cos\theta < 1$, and the cosine
\emph{decreases} as the angle opens: a wider angle has a smaller
cosine. Landmarks: $\cos 0^\circ = 1$, $\cos 60^\circ = \frac12$,
$\cos 90^\circ = 0$ (the extreme values corresponding to flattened
triangles).
\end{proposition}
\admitted

\begin{method}[Using the cosine]\label{met:g8:cosine:use}
In a \omterm{def:g6:shapes:triangles}{right triangle}, when the known and wanted quantities are an
acute angle, its adjacent side, and the hypotenuse:
\begin{enumerate}
  \item write the definition:
        $\cos\theta = \frac{\text{adjacent}}{\text{hypotenuse}}$ with
        the known values in place;
  \item solve the small \omterm{def:g8:equations:def}{equation} for the unknown (multiply or divide);
  \item for an unknown \emph{angle}, apply the calculator's
        $\cos^{-1}$ to the computed ratio;
  \item sanity checks: a length must come out shorter than the
        hypotenuse; an angle strictly between $0^\circ$ and
        $90^\circ$.
\end{enumerate}
\end{method}

\begin{example}[Finding a side]\label{ex:g8:cosine:side}
A $6$ m ramp makes an angle of $20^\circ$ with the horizontal ground.
Horizontal distance covered (adjacent to $20^\circ$, hypotenuse
$6$ m):
\[
\cos 20^\circ = \frac{d}{6}
\quad\Longrightarrow\quad
d = 6 \cos 20^\circ \approx 6 \times 0.940 \approx 5.6 \text{ m}.
\]
\end{example}

\begin{example}[Finding an angle]\label{ex:g8:cosine:angle}
In a \omterm{def:g6:shapes:triangles}{right triangle}, the side adjacent to the angle $\theta$ measures
$3.5$ and the hypotenuse $5$:
\[
\cos\theta = \frac{3.5}{5} = 0.7,
\qquad
\theta = \cos^{-1}(0.7) \approx 45.6^\circ .
\]
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g8:cosine:1}
A \omterm{def:g6:shapes:triangles}{right triangle} has a hypotenuse of $13$ cm. What is the length of
the median from the right-angle \omterm{def:g5:solids:def}{vertex}? What is the \omterm{def:g3:shapes:shapes}{radius} of the
circle through the three \omterm{def:g5:solids:def}{vertices}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:2}
Draw a circle of \omterm{def:g4:geometry:circle}{diameter} $8$ cm with a \omterm{def:g4:geometry:circle}{diameter} $[BC]$, choose any
point $A$ on the circle and draw $ABC$. Which angle is right? Cite
the theorem. Measure $AB$ and $AC$ and check Pythagoras.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:3}
In a triangle $DEF$ \omterm{def:g6:shapes:triangles}{right-angled} at $D$, name the side adjacent to
the angle $\widehat E$, the side opposite it, and the hypotenuse.
Write $\cos \widehat E$ as a ratio.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:4}
Compute the missing quantity ($\cos 35^\circ \approx 0.819$,
$\cos 50^\circ \approx 0.643$):
\begin{enumerate}
  \item hypotenuse $10$, angle $35^\circ$: adjacent side?
  \item adjacent side $6$, angle $50^\circ$: hypotenuse?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:5}
In a \omterm{def:g6:shapes:triangles}{right triangle}, the adjacent side to $\theta$ measures $8$ and
the hypotenuse $10$. Compute $\cos\theta$, then $\theta$
($\cos^{-1}(0.8) \approx 36.9^\circ$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:6}
Explain why $\cos\theta$ can never equal $1.2$ for an acute angle of
a \omterm{def:g6:shapes:triangles}{right triangle}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:7}
A $25$ m zip line descends from a platform to the ground, making an
angle of $12^\circ$ with the horizontal ($\cos 12^\circ \approx
0.978$). What horizontal distance does it span?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:8}
Order without a calculator: $\cos 20^\circ$, $\cos 70^\circ$,
$\cos 45^\circ$ (use \cref{prop:g8:cosine:bounds}). Then check with a
calculator.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:9}
A ladder of length $L$ leans against a wall with an angle $\theta$
between ladder and \emph{ground}. Its foot is $1.2$ m from the wall
and $\theta = 68^\circ$ ($\cos 68^\circ \approx 0.375$). Compute $L$,
then the height reached (Pythagoras).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:10}
$[BC]$ is a \omterm{def:g4:geometry:circle}{diameter} of a circle of center $O$ and \omterm{def:g3:shapes:shapes}{radius} $4.5$ cm,
and $A$ is a point of the circle with $AB = 5.4$ cm.
\begin{enumerate}
  \item Why is $ABC$ \omterm{def:g6:shapes:triangles}{right-angled} at $A$?
  \item Compute $AC$, then $\cos \widehat B$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:cosine:11}
Using \omterm{def:g3:division:half}{half} an \omterm{def:g6:shapes:triangles}{equilateral triangle} of side $1$ (cut along a height),
justify the landmark $\cos 60^\circ = \frac12$ of
\cref{prop:g8:cosine:bounds}. (Where does the foot of the height
fall on the base? The full trigonometric version is
\cref{ex:g9:trig:special}.)
\end{exercise}

\section{Problem: Euclid's relations, and Pythagoras all over
again}

\begin{problem}[{Weekend problem --- the altitude of a right
triangle: $AB^2 = BH \times BC$, $AH^2 = BH \times HC$, a second
proof of Pythagoras, and a machine for constructing square roots}]
\label{pb:g8:cosine:1}
Drop, from the \omterm{def:g3:shapes:rightangle}{right angle} of a \omterm{def:g6:shapes:triangles}{right triangle}, the \omterm{def:g6:lines:perp}{perpendicular}
to the hypotenuse: this short \omterm{def:g6:lines:objects}{segment} --- the \emph{altitude} ---
satisfies relations so useful that Euclid put them at the heart of
his \emph{Elements}. In this problem the cosine's defining
property, ``the ratio depends only on the angle''
(\cref{def:g8:cosine:def}), proves all of them; on the way you
will re-prove Pythagoras' theorem by a completely different route,
and end with a ruler-and-compass machine that constructs
$\sqrt2$, $\sqrt5$, $\sqrt{n}$ for every whole number $n$.

Throughout, $ABC$ is a triangle \omterm{def:g6:shapes:triangles}{right-angled} at $A$, and $H$ is
the foot of the \omterm{def:g6:lines:perp}{perpendicular} from $A$ to the hypotenuse $[BC]$,
so that $H$ lies between $B$ and $C$ and $AH \perp BC$.

\medskip
\noindent\textbf{Part I --- One angle, two triangles.}
\begin{enumerate}
  \item Using the angle \omterm{def:g1:addition:def}{sum} of a triangle
        (\cref{thm:g7:triangles:sum}) in $ABH$ and in $ABC$, show
        that $\widehat{BAH} = \widehat{ACB}$: the altitude cuts
        the \omterm{def:g6:shapes:triangles}{right triangle} into two smaller triangles carrying
        the \emph{same angles} as the original.
  \item The triangles $ABC$ (\omterm{def:g6:shapes:triangles}{right-angled} at $A$) and $ABH$
        (\omterm{def:g6:shapes:triangles}{right-angled} at $H$) share the angle $\widehat B$. Write
        $\cos\widehat B$ in each of them, and deduce from
        \cref{def:g8:cosine:def} that
        \[
        AB^2 = BH \times BC .
        \]
        (To pass from equal ratios to equal \omterm{def:g2:mult:def}{products}, multiply
        both sides by both \omterm{def:g4:fractions:def}{denominators},
        \cref{thm:g8:equations:balance}.)
  \item State and prove the twin relation at the \omterm{def:g5:solids:def}{vertex} $C$:
        \[
        AC^2 = CH \times CB .
        \]
  \item Take the $3$--$4$--$5$ triangle: $AB = 3$, $AC = 4$,
        $BC = 5$. Compute $BH$ and $CH$ from questions 2 and 3,
        and check that $BH + HC = BC$.
  \item In general, express $BH$ and $CH$ as \omterm{def:g6:fractions:def}{fractions} involving
        only the three sides. In what sense does the foot $H$
        split the hypotenuse ``proportionally to the squares of
        the legs''?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Pythagoras again, and the altitude.}
\begin{enumerate}[resume]
  \item Add the relations of questions 2 and 3 and use
        $BH + HC = BC$ to obtain
        \[
        AB^2 + AC^2 = BC^2 :
        \]
        Pythagoras' theorem (\cref{thm:g8:pythagoras:direct}),
        re-proved --- with no \omterm{def:g6:measure:area}{area} puzzle in sight (compare
        \cref{exo:g8:pythagoras:11}).
  \item Apply Pythagoras in the small triangle $ABH$ to write
        $AH^2 = AB^2 - BH^2$, replace $AB^2$ by
        $BH \times BC$, and factor out $BH$ to prove
        \emph{Euclid's altitude relation}:
        \[
        AH^2 = BH \times HC .
        \]
  \item Compute the \omterm{def:g6:measure:area}{area} of $ABC$ in two ways --- legs as base
        and height, then hypotenuse as base
        (\cref{thm:g7:areas:triangle}) --- and deduce the third
        relation:
        \[
        AB \times AC = BC \times AH .
        \]
  \item Back to the $3$--$4$--$5$ triangle: compute $AH$ with
        question 8, then verify the altitude relation of question
        7 numerically, using the values of $BH$ and $CH$ found in
        question 4.
  \item The altitude's foot splits the hypotenuse of some \omterm{def:g6:shapes:triangles}{right
        triangle} into \omterm{def:g6:lines:objects}{segments} $BH = 2$ cm and $HC = 8$ cm.
        Compute the altitude $AH$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- A machine for square roots.}
Draw a \omterm{def:g6:lines:objects}{segment} $[BC]$ made of two pieces laid end to end:
$BH = p$ and $HC = q$. Draw the half-circle of \omterm{def:g4:geometry:circle}{diameter} $[BC]$,
and let $A$ be the point where the \omterm{def:g6:lines:perp}{perpendicular} to $(BC)$ at $H$
meets it.
\begin{enumerate}[resume]
  \item Using \cref{thm:g8:cosine:circle}, explain why the
        triangle $ABC$ is \omterm{def:g6:shapes:triangles}{right-angled} at $A$ --- so question 7
        applies and
        \[
        AH^2 = p \times q .
        \]
        (The length $AH$ is called the \emph{geometric
        mean}\index{geometric mean} of $p$ and $q$.)
  \item Take $p = 1$ and $q = 2$. What is $AH^2$? Which famous
        length of \cref{ex:g8:pythagoras:diagonal} has your
        ruler-and-compass figure just constructed?
  \item Describe the recipe that constructs a \omterm{def:g6:lines:objects}{segment} of length
        $\sqrt n$ for any whole number $n \geq 1$, and say which
        choice of $p$ and $q$ constructs $\sqrt 5$.
  \item Let $O$ be the center of the half-circle. Explain why
        $AH$ can never exceed the \omterm{def:g3:shapes:shapes}{radius} $\frac{p + q}{2}$, and
        for which position of $H$ the two are equal.
  \item Deduce the inequality: for all positive numbers $p$ and
        $q$,
        \[
        p \times q \leq \left(\frac{p + q}{2}\right)^2,
        \]
        with equality exactly when $p = q$. Then reprove it with
        no geometry at all: expand
        $\left(\frac{p+q}{2}\right)^2 - p q$ and recognize a
        remarkable identity (\cref{pb:g8:equations:1}).
\end{enumerate}
\end{problem}
