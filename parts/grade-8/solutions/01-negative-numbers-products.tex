\begin{solution}{exo:g8:negprod:1}
$(-8) \times (+7) = -56$; \quad
$(-9) \times (-6) = +54$; \quad
$(+12) \times (-0.5) = -6$; \quad
$(-1) \times (-1) \times (-1) = -1$ (three negative factors: \omterm{def:g3:numbers:evenodd}{odd}).
\end{solution}

\begin{solution}{exo:g8:negprod:3}
Three negative factors: \emph{negative}. \quad
$(-1)^{10}$: ten negative factors, \omterm{def:g3:numbers:evenodd}{even}: \emph{positive} (it equals
$1$). \quad
$(-2)^7$: seven factors, \omterm{def:g3:numbers:evenodd}{odd}: \emph{negative}. \quad
$(-4) \times 0 \times (-6) = 0$: neither positive nor negative.
\end{solution}

\begin{solution}{exo:g8:negprod:4}
$(-2)^4 = +16$; \quad $-2^4 = -16$ (the minus is not squared); \quad
$(-3)^3 = -27$; \quad $(-1)^{2026} = +1$ (\omterm{def:g3:numbers:evenodd}{even} exponent).
\end{solution}

\begin{solution}{exo:g8:negprod:5}
$7 + 2 \times (-6) = 7 + (-12) = -5$.

$(-4) \times (5 - 9) = (-4) \times (-4) = +16$.

$18 \div (-3) - (-2) = -6 + 2 = -4$.
\end{solution}

\begin{solution}{exo:g8:negprod:6}

\begin{solution}{exo:g8:negprod:7}
For $x = -3$: \ $5x = -15$; \quad $x^2 = (-3)^2 = 9$; \quad
$-x^2 = -9$; \quad $2x^2 + 4x = 18 - 12 = 6$; \quad
$(2x)^2 = (-6)^2 = 36$.
\end{solution}

\begin{solution}{exo:g8:negprod:9}
\emph{1. False}: $0.5 \times 0.5 = 0.25$ is smaller than
$0.5 + 0.5 = 1$ (or: $2 \times 3 = 6 > 5$, but the claim must hold
always).

\emph{2. True}: a square is a \omterm{def:g2:mult:def}{product} of two numbers of the same
sign, hence positive or \omterm{def:g1:counting:zero}{zero}.

\emph{3. False}: $(-2) \times (-3) \times 4 = +24$ is positive with
two negative factors.
\end{solution}

\begin{solution}{exo:g8:negprod:10}
Let $r$ be the number of right answers; then $20 - r$ are wrong, and
\[
5r - 3(20 - r) = 36
\ \Longrightarrow\
5r - 60 + 3r = 36
\ \Longrightarrow\
8r = 96
\ \Longrightarrow\
r = 12 .
\]
Twelve right answers (and eight wrong: check,
$60 - 24 = 36$).
\end{solution}

\begin{solution}{exo:g8:negprod:11}
\omterm{def:g1:counting:zero}{Zero} times anything is \omterm{def:g1:counting:zero}{zero}, so $(-1) \times 0 = 0$. Writing
$0 = 1 + (-1)$ and distributing:
\[
0 = (-1) \times \bigl(1 + (-1)\bigr)
= (-1) \times 1 + (-1) \times (-1)
= -1 + (-1) \times (-1).
\]
The number $(-1) \times (-1)$ added to $-1$ gives $0$: it must be the
opposite of $-1$, that is $+1$. Distributivity leaves no other
choice.
\end{solution}

\begin{solution}{pb:g8:negprod:1}
\textbf{1.} Since $0 = 0 + 0$, distributivity
(\cref{thm:g7:priorities:distributivity}) gives
\[
0 \times a = (0 + 0) \times a = 0 \times a + 0 \times a .
\]
So adding $0 \times a$ to itself changes nothing --- and the only
number one can add without changing anything is $0$. (Directly:
subtract $0 \times a$ from both sides.) Hence $0 \times a = 0$.

\textbf{2.} Expanding, then using $1 + (-1) = 0$ and question 1:
\[
a + (-1) \times a
= 1 \times a + (-1) \times a
= \bigl(1 + (-1)\bigr) \times a
= 0 \times a = 0 .
\]
So $(-1) \times a$ is the number which, added to $a$, gives $0$:
that is exactly the opposite of $a$, and $(-1) \times a = -a$.
\Cref{exo:g8:negprod:11} is the special case $a = -1$: there,
$(-1) \times (-1) = -(-1) = +1$.

\textbf{3.} Write $-a = (-1) \times a$ (question 2) and regroup the
factors:
\[
(-a) \times b
= \bigl((-1) \times a\bigr) \times b
= (-1) \times (a \times b)
= -(a \times b) .
\]

\textbf{4.} Apply question 3 twice (once on each factor):
\[
(-a) \times (-b)
= -\bigl(a \times (-b)\bigr)
= -\bigl(-(a \times b)\bigr)
= a \times b ,
\]
since the opposite of the opposite of a number is the number itself.

\textbf{5.} Write $a = +d$ or $a = -d$ and $b = +e$ or $b = -e$,
where $d$ and $e$ are the distances to \omterm{def:g1:counting:zero}{zero}. The four cases:
\[
(+d) \times (+e) = de, \quad
(+d) \times (-e) = -de, \quad
(-d) \times (+e) = -de, \quad
(-d) \times (-e) = +de,
\]
the middle two by question 3, the last by question 4. In every case
the distance of the result is $d \times e$, the \omterm{def:g2:mult:def}{product} of the
distances, and the sign is the one announced by the table of
\cref{thm:g8:negprod:rules}: same signs positive, opposite signs
negative.

\textbf{6.} $(-1)^2 = +1$, $(-1)^3 = -1$, $(-1)^4 = +1$,
$(-1)^5 = -1$. In general $(-1)^n$ is a \omterm{def:g2:mult:def}{product} of $n$ negative
factors, each of distance $1$: by \cref{prop:g8:negprod:many} it is
$+1$ when $n$ is \omterm{def:g3:numbers:evenodd}{even} and $-1$ when $n$ is \omterm{def:g3:numbers:evenodd}{odd}.

\textbf{7.} There are ten negative factors --- an \omterm{def:g3:numbers:evenodd}{even number} --- so
the \omterm{def:g2:mult:def}{product} is \emph{positive} (\cref{prop:g8:negprod:many}). Its
distance is $1 \times 2 \times 3 \times \dots \times 10$.

\textbf{8.} By question 4, $(-a)^2 = (-a) \times (-a) = a \times a =
a^2$. Then by question 3,
\[
(-a)^3 = (-a)^2 \times (-a) = a^2 \times (-a) = -(a^2 \times a)
= -a^3 .
\]
The minus sign disappears for \omterm{def:g3:numbers:evenodd}{even} exponents and survives for \omterm{def:g3:numbers:evenodd}{odd}
ones, in line with \cref{ex:g8:negprod:powers}.

\textbf{9.} If $a = 0$ then $a^2 = 0$. Otherwise $a$ and $a$ have
the \emph{same} sign, so $a^2 = a \times a$ is positive
(\cref{thm:g8:negprod:rules}). A square is therefore never
negative; since $-4$ is negative, no \omterm{def:g7:negatives:def}{relative number} $x$ satisfies
$x^2 = -4$.

\textbf{10.} Zoe is right. The exponents $2026$ and $2027$ have
opposite parities, so $(-1)^{2026} = +1$ and $(-1)^{2027} = -1$,
which add to $0$. In general, of two consecutive exponents one is
\omterm{def:g3:numbers:evenodd}{even} and one is \omterm{def:g3:numbers:evenodd}{odd}, so the two powers are $+1$ and $-1$ in some
order: opposite numbers, whose \omterm{def:g1:addition:def}{sum} is always $0$.

\textbf{11.} $A_3 = 1 - 2 + 3 = 2$, then
$A_4 = A_3 - 4 = -2$, $A_5 = A_4 + 5 = 3$, $A_6 = A_5 - 6 = -3$,
$A_7 = A_6 + 7 = 4$. Conjecture: $A_n = -\frac{n}{2}$ for \omterm{def:g3:numbers:evenodd}{even} $n$,
and $A_n = \frac{n+1}{2}$ for \omterm{def:g3:numbers:evenodd}{odd} $n$.

\textbf{12.} For \omterm{def:g3:numbers:evenodd}{even} $n$ the terms pair up completely:
\[
A_n = (1 - 2) + (3 - 4) + \dots + \bigl((n-1) - n\bigr),
\]
and each pair equals $-1$. There are $\frac{n}{2}$ pairs (two terms
per pair, $n$ terms in all), so
$A_n = \frac{n}{2} \times (-1) = -\frac{n}{2}$.

\textbf{13.} For \omterm{def:g3:numbers:evenodd}{odd} $n$, the \omterm{def:g1:addition:def}{sum} $A_n$ is the \omterm{def:g1:addition:def}{sum} $A_{n-1}$ of the
first $n - 1$ terms, plus the last term, which is $+n$ since $n$ is
\omterm{def:g3:numbers:evenodd}{odd}. As $n - 1$ is \omterm{def:g3:numbers:evenodd}{even}, question 12 gives
\[
A_n = A_{n-1} + n = -\frac{n-1}{2} + n
= \frac{-(n-1) + 2n}{2} = \frac{n+1}{2} .
\]

\textbf{14.} $A_{100} = -\frac{100}{2} = -50$, \quad
$A_{101} = \frac{102}{2} = 51$, \quad
$A_{2026} = -\frac{2026}{2} = -1013$.

\textbf{15.} No surprise: passing from $A_{100}$ to $A_{101}$ adds
exactly one term, namely $+101$, so the \omterm{ex:g1:subtraction:difference}{difference} had to be $101$.
For the same reason, $A_{2027} - A_{2026} = +2027$ --- without
computing $A_{2027}$ at all.
\end{solution}
