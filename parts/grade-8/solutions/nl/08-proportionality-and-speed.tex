\section*{Hoofdstuk \ref{ch:g8:speed} --- Evenredigheid, snelheid en gemiddelden}

\begin{solution}{exo:g8:speed:1}
$\frac{150}{2} = 75$ km/h. \quad
$45$ min $= 0.75$ u: $\frac{27}{0.75} = 36$ km/h. \quad
$\frac{100}{12.5} = 8$ m/s.
\end{solution}

\begin{solution}{exo:g8:speed:2}
$2 \times 85 = 170$ km. \quad
$40$ min $= \frac23$ u: $90 \times \frac23 = 60$ km. \quad
$\frac{315}{90} = 3.5$ u $= 3$ u $30$ min.
\end{solution}

\begin{solution}{exo:g8:speed:3}
$72 \div 3.6 = 20$ m/s. \quad
$5 \times 3.6 = 18$ km/h. \quad
$108 \div 3.6 = 30$ m/s.
\end{solution}

\begin{solution}{exo:g8:speed:4}
$600 \div 15 = 40$ min. \quad
$2$ u $30 = 150$ min: $15 \times 150 = 2250$ L.
\end{solution}

\begin{solution}{exo:g8:speed:5}
$\frac{3.30}{1.2} = 2.75$ euro/kg tegen
$\frac{2.16}{0.8} = 2.70$ euro/kg: het tweede aanbod is (iets)
de betere koop.
\end{solution}

\begin{solution}{exo:g8:speed:6}
\[
\bar v = \frac{25 \times 12.4 + 35 \times 10}{60}
= \frac{310 + 350}{60} = \frac{660}{60} = 11 .
\]
\end{solution}

\begin{solution}{exo:g8:speed:7}
$d = 340 \times 6 = 2040$ m $\approx 2$ km.
\end{solution}

\begin{solution}{exo:g8:speed:8}
Afstanden: $2 \times 5 = 10$ km, daarna $1.5 \times 4 = 6$ km: totaal
$16$ km. Verstreken tijd: $2 + 0.5 + 1.5 = 4$ u. Gemiddelde snelheid:
$\frac{16}{4} = 4$ km/h.
\end{solution}

\begin{solution}{exo:g8:speed:9}
Gewogen:
$\frac{3 \times 15 + 2 \times 9 + 1 \times 12}{3 + 2 + 1}
= \frac{45 + 18 + 12}{6} = \frac{75}{6} = 12.5$.
Ongewogen: $\frac{15 + 9 + 12}{3} = 12$. Ze verschillen omdat de
coëfficiënten het cijfer $15$ driemaal het gewicht van het cijfer
$12$ geven.
\end{solution}

\begin{solution}{exo:g8:speed:10}
\emph{1.} Tijden: $\frac{120}{80} = 1.5$ u en $\frac{120}{120} = 1$
u. Gemiddelde snelheid: $\frac{240}{2.5} = 96$ km/h.

\emph{2.} De auto spendeert \emph{meer tijd} bij $80$ km/h ($1.5$ u)
dan bij $120$ km/h ($1$ u): de trage snelheid weegt zwaarder in het
tijdgewogen gemiddelde, en trekt het onder het midden $100$.
\end{solution}

\begin{solution}{exo:g8:speed:11}
De kloof van $18$ km sluit met $5 + 4 = 9$ km/h: ze ontmoeten elkaar
na $\frac{18}{9} = 2$ u, om $12{:}00$. Anna heeft dan
$2 \times 5 = 10$ km gelopen: ze ontmoeten elkaar $10$ km van Anna's
dorp (en $8$ km van Boris' dorp --- controle: $10 + 8 = 18$).
\end{solution}
