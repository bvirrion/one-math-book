\section*{Hoofdstuk \ref{ch:g8:pythagoras} --- De stelling van Pythagoras}

\begin{solution}{exo:g8:pythagoras:1}
$\sqrt{81 + 144} = \sqrt{225} = 15$; \quad
$\sqrt{25 + 144} = \sqrt{169} = 13$; \quad
$\sqrt{64 + 225} = \sqrt{289} = 17$.
\end{solution}

\begin{solution}{exo:g8:pythagoras:2}
Been$^2 = 25^2 - 7^2 = 625 - 49 = 576$, dus het been is
$\sqrt{576} = 24$.
\end{solution}

\begin{solution}{exo:g8:pythagoras:3}
Rechthoek: $\sqrt{12^2 + 9^2} = \sqrt{144 + 81} = \sqrt{225} = 15$
cm.

Vierkant: $d = 5\sqrt2 \approx 7.1$ cm
(\cref{ex:g8:pythagoras:diagonal}).
\end{solution}

\begin{solution}{exo:g8:pythagoras:4}
$(20, 21, 29)$: $29^2 = 841$ en $20^2 + 21^2 = 400 + 441 = 841$:
\omterm{def:g6:shapes:triangles}{rechthoekig}.

$(7, 8, 11)$: $121 \neq 49 + 64 = 113$: niet \omterm{def:g6:shapes:triangles}{rechthoekig}.

$(1.5, 2, 2.5)$: $6.25 = 2.25 + 4$: \omterm{def:g6:shapes:triangles}{rechthoekig} (het is
$(3,4,5)$ gehalveerd).
\end{solution}

\begin{solution}{exo:g8:pythagoras:5}
Plank $= \sqrt{3^2 + 1.6^2} = \sqrt{9 + 2.56} = \sqrt{11.56} = 3.4$
m.
\end{solution}

\begin{solution}{exo:g8:pythagoras:6}
De hoogte valt op het midden van de basis (\omterm{def:g6:shapes:triangles}{gelijkbenige driehoek}),
dus elke halfdriehoek heeft schuine zijde $10$ en één been $6$:
$h = \sqrt{100 - 36} = \sqrt{64} = 8$ cm. \omterm{def:g6:measure:area}{Oppervlakte}:
$\frac{12 \times 8}{2} = 48$ cm$^2$.
\end{solution}

\begin{solution}{exo:g8:pythagoras:7}
Diagonaal: $\sqrt{88.5^2 + 49.8^2} = \sqrt{7832.25 + 2480.04}
= \sqrt{10312.29} \approx 101.5$ cm. In inches:
$101.5 \div 2.54 \approx 40$ inches.
\end{solution}

\begin{solution}{exo:g8:pythagoras:8}
De twee benen zijn $24$ km en $10$ km:
$\sqrt{24^2 + 10^2} = \sqrt{576 + 100} = \sqrt{676} = 26$ km.
\end{solution}

\begin{solution}{exo:g8:pythagoras:9}
Voetafstand: $\sqrt{2.5^2 - 2.4^2} = \sqrt{6.25 - 5.76} =
\sqrt{0.49} = 0.7$ m. Een \omterm{def:g3:division:half}{kwart} van de ladder is
$2.5 \div 4 \approx 0.6$ m: de gevonden $0.7$ m ligt dicht bij de
geadviseiseerde plaatsing --- aanvaardbaar.
\end{solution}

\begin{solution}{exo:g8:pythagoras:10}
De benen meten $5 - 1 = 4$ (horizontaal) en $5 - 2 = 3$
(verticaal), dus
\[
AB = \sqrt{4^2 + 3^2} = \sqrt{25} = 5 .
\]
\end{solution}

\begin{solution}{exo:g8:pythagoras:11}
\emph{1.} Rechtstreeks: $(a + b)^2$. Als stukken: vier driehoeken van
\omterm{def:g6:measure:area}{oppervlakte} $\frac{ab}{2}$ elk, plus het scheve vierkant $c^2$:
\[
(a + b)^2 = 4 \times \frac{ab}{2} + c^2 = 2ab + c^2 .
\]

\emph{2.} Linkerkant uitwerken:
$(a+b)^2 = a^2 + 2ab + b^2$. Dus
\[
a^2 + 2ab + b^2 = 2ab + c^2
\quad\Longrightarrow\quad
a^2 + b^2 = c^2 . \qed
\]
\end{solution}
