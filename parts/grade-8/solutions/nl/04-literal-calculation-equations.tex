\section*{Hoofdstuk \ref{ch:g8:equations} --- Letterrekenen en vergelijkingen}

\begin{solution}{exo:g8:equations:1}
$5(2x + 3) = 10x + 15$.

$-3(4x - 1) = -12x + 3$.

$2(3x - 2) + 4(x + 5) = 6x - 4 + 4x + 20 = 10x + 16$.
\end{solution}

\begin{solution}{exo:g8:equations:2}
$(x+2)(x+7) = x^2 + 7x + 2x + 14 = x^2 + 9x + 14$.

$(3x+1)(2x-5) = 6x^2 - 15x + 2x - 5 = 6x^2 - 13x - 5$.

$(x-4)(x-6) = x^2 - 6x - 4x + 24 = x^2 - 10x + 24$.
\end{solution}

\begin{solution}{exo:g8:equations:3}
$4 \times 3 - 5 = 7$: ja. \quad
$2 \times 3 + 1 = 7$ en $3 + 4 = 7$: ja. \quad
$3^2 = 9$ en $6 \times 3 - 9 = 9$: ja --- $x = 3$ lost alle drie op.
\end{solution}

\begin{solution}{exo:g8:equations:4}
$x + 9 = 4$: $x = -5$ (controle: $-5 + 9 = 4$).

$5x = -35$: $x = -7$.

$\frac x3 = 8$: $x = 24$.

$2x - 7 = 13$: $2x = 20$, $x = 10$ (controle: $20 - 7 = 13$).
\end{solution}

\begin{solution}{exo:g8:equations:5}
$6x + 5 = 2x + 25$: $4x = 20$, dus $x = 5$.

$9 - 4x = x - 11$: $20 = 5x$, dus $x = 4$.

$3(x + 2) = 2x + 11$: $3x + 6 = 2x + 11$, dus $x = 5$.
\end{solution}

\begin{solution}{exo:g8:equations:6}
$8x - 12 = 5x + 30$, dus $3x = 42$ en $x = 14$. Controle:
$4(28 - 3) = 100$ en $5(14 + 6) = 100$.
\end{solution}

\begin{solution}{exo:g8:equations:7}

``$2x + 5 = x + 12$'': $x = 7$.
\end{solution}

\begin{solution}{exo:g8:equations:8}
Breedte $x$, lengte $x + 5$; \omterm{def:g6:measure:perimeter}{omtrek}
$2\bigl(x + (x + 5)\bigr) = 4x + 10 = 58$, dus $4x = 48$ en $x = 12$:
breedte $12$ cm, lengte $17$ cm (controle:
$2 \times (12 + 17) = 58$).
\end{solution}

\begin{solution}{exo:g8:equations:9}
De getallen zijn $n - 1$, $n$, $n + 1$ met \omterm{def:g1:addition:def}{som} $3n = 141$, dus
$n = 47$: de getallen zijn $46$, $47$, $48$.
\end{solution}

\begin{solution}{exo:g8:equations:10}
$3x + 4 > 19$: $3x > 15$, dus $x > 5$.

$7 - 2x \geq 1$: $-2x \geq -6$, delen door $-2$ keert om:
$x \leq 3$.

$5(x-1) < 2x + 7$: $5x - 5 < 2x + 7$, dus $3x < 12$ en $x < 4$.
\end{solution}

\begin{solution}{exo:g8:equations:11}
Na $n$ maanden heeft Mia $140 + 15n$ en Tom $200 + 10n$. Mia leidt
wanneer
\[
140 + 15n > 200 + 10n
\ \Longleftrightarrow\
5n > 60
\ \Longleftrightarrow\
n > 12 :
\]
vanaf de $13$e maand.
\end{solution}

\begin{solution}{exo:g8:equations:12}
Vermenigvuldig beide kanten met $20$:
\[
5(x + 3) = 4(2x - 1)
\ \Longrightarrow\
5x + 15 = 8x - 4
\ \Longrightarrow\
19 = 3x
\ \Longrightarrow\
x = \frac{19}{3}.
\]
Controle: $\frac{19/3 + 3}{4} = \frac{28/3}{4} = \frac{28}{12} =
\frac73$, en $\frac{2 \times 19/3 - 1}{5} = \frac{35/3}{5} =
\frac{35}{15} = \frac73$. Gelijk: de oplossing is $\frac{19}{3}$.
\end{solution}
