\section*{Hoofdstuk \ref{ch:g8:midpoints} --- Middens en evenwijdigen}

\begin{solution}{exo:g8:midpoints:1}
$M$ en $N$ zijn de middens van twee zijden van de driehoek $RST$: door
de middenparallellenstelling (\cref{thm:g8:midpoints:direct}),
$(MN) \parallel (ST)$ en $MN = \frac{ST}{2} = 3.5$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:2}
$I$ is een midden en $(IJ) \parallel (BC)$: door het \emph{omgekeerde}
(\cref{thm:g8:midpoints:converse}) is $J$ het midden van $[AC]$, dus
$AJ = \frac{11}{2} = 5.5$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:3}
De meting geeft $IJ = 3.5$ cm $= \frac{BC}{2}$. \omterm{def:g6:measure:perimeter}{Omtrek} van $AIJ$:
$AI = 4$, $AJ = 3$, $IJ = 3.5$: totaal $10.5$ cm --- de \omterm{def:g3:division:half}{helft} van de
\omterm{def:g6:measure:perimeter}{omtrek} van $ABC$ ($8 + 6 + 7 = 21$ cm).
\end{solution}

\begin{solution}{exo:g8:midpoints:4}
Elke zijde van $IJK$ is de \omterm{def:g3:division:half}{helft} van een zijde van $ABC$: $5$, $6$ en
$8$ cm. \omterm{def:g6:measure:perimeter}{Omtrek}: $19$ cm, de \omterm{def:g3:division:half}{helft} van $10 + 12 + 16 = 38$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:5}
De middendriehoek is één van de vier gelijke driehoeken van
\cref{prop:g8:midpoints:medial}: \omterm{def:g6:measure:area}{oppervlakte} $\frac{36}{4} = 9$
cm$^2$ --- en elk van de vier kleine driehoeken heeft \omterm{def:g6:measure:area}{oppervlakte}
$9$ cm$^2$.
\end{solution}

\begin{solution}{exo:g8:midpoints:6}
Door de middenparallellenstelling in $ABC$: $(IJ) \parallel (BC)$,
d.w.z.\ $(IJ) \parallel (BK)$, en $IJ = \frac{BC}{2} = BK$ (want $K$
is het midden van $[BC]$). Twee overstaande zijden van $IJKB$ zijn
\omterm{def:g6:lines:perp}{evenwijdig} en gelijk: het is een \omterm{def:g7:central:parallelogram}{parallellogram}
(\cref{thm:g7:central:recognize}, criterium 3).
\end{solution}

\begin{solution}{exo:g8:midpoints:7}
De naad verbindt twee middens: $\frac{6.4}{2} = 3.2$ m. Op één
\omterm{def:g3:division:half}{kwart} vanaf het tophoekpunt suggereert de tekening een naad van
$\frac{6.4}{4} = 1.6$ m, \omterm{def:g6:lines:perp}{evenwijdig} aan de basis --- bevestigd door
Thales in \cref{ch:g9:thales}.
\end{solution}

\begin{solution}{exo:g8:midpoints:8}
\emph{1.} $M$ is het midden van $[BC]$ en $(MN) \parallel (AB)$: door
het omgekeerde van de middenparallellenstelling (in de driehoek
$ABC$, zijde $[AC]$) is $N$ het midden van $[AC]$.

\emph{2.} $(MN) \parallel (AB)$ en $(AB) \perp (AC)$ (\omterm{def:g3:shapes:rightangle}{rechte hoek} in
$A$): een lijn \omterm{def:g6:lines:perp}{evenwijdig} aan $(AB)$ staat ook \omterm{def:g6:lines:perp}{loodrecht} op $(AC)$
(\cref{prop:g6:lines:facts}). Dus $(MN) \perp (AC)$ --- het
\omterm{def:g6:lines:objects}{lijnstuk} $[MN]$ is het middelloodstuk dat opnieuw toont dat
$MA = MC$ (\cref{thm:g8:cosine:circle}).
\end{solution}

\begin{solution}{exo:g8:midpoints:9}
\emph{1.} In driehoek $ABC$ zijn $I$ en $J$ middens van $[AB]$ en
$[BC]$: $(IJ) \parallel (AC)$ en $IJ = \frac{AC}{2}$. In driehoek
$ACD$ zijn $L$ en $K$ middens van $[DA]$ en $[CD]$:
$(LK) \parallel (AC)$ en $LK = \frac{AC}{2}$.

\emph{2.} Dus zijn $[IJ]$ en $[LK]$ \omterm{def:g6:lines:perp}{evenwijdig} (beide \omterm{def:g6:lines:perp}{evenwijdig} aan
de diagonaal $(AC)$) en gelijk (beide de \omterm{def:g3:division:half}{helft} van $AC$): $IJKL$ is
een \omterm{def:g7:central:parallelogram}{parallellogram} (\cref{thm:g7:central:recognize}, criterium 3) ---
voor \emph{elke} \omterm{def:g4:geometry:polygon}{vierhoek} $ABCD$, zoals vermoed in
\cref{exo:g7:central:11}.
\end{solution}
