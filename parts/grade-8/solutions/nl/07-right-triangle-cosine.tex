\section*{Hoofdstuk \ref{ch:g8:cosine} --- De rechthoekige driehoek en de cosinus}

\begin{solution}{exo:g8:cosine:1}
Mediaan vanuit de \omterm{def:g3:shapes:rightangle}{rechte hoek}: de \omterm{def:g3:division:half}{helft} van de schuine zijde, $6.5$ cm
(\cref{thm:g8:cosine:circle}). De omgeschreven cirkel heeft
middelpunt in het midden van de schuine zijde met \omterm{def:g3:shapes:shapes}{straal} $6.5$ cm.
\end{solution}

\begin{solution}{exo:g8:cosine:2}
De hoek in $A$ is recht ($[BC]$ is een middellijn: punt 2 van
\cref{thm:g8:cosine:circle}). De maten voldoen aan
$AB^2 + AC^2 \approx 64 = BC^2$, op meetnauwkeurigheid na.
\end{solution}

\begin{solution}{exo:g8:cosine:3}
Schuine zijde: $[EF]$ (tegenover de \omterm{def:g3:shapes:rightangle}{rechte hoek} $D$). Aanliggend aan
$\widehat E$: $[ED]$. Tegenover: $[DF]$. Dus
$\cos\widehat E = \dfrac{ED}{EF}$.
\end{solution}

\begin{solution}{exo:g8:cosine:4}
\emph{1.} aanliggend $= 10 \cos 35^\circ \approx 8.2$.

\emph{2.} schuine zijde $= \dfrac{6}{\cos 50^\circ} \approx
\dfrac{6}{0.643} \approx 9.3$.
\end{solution}

\begin{solution}{exo:g8:cosine:5}
$\cos\theta = \frac{8}{10} = 0.8$, dus
$\theta = \cos^{-1}(0.8) \approx 37^\circ$.
\end{solution}

\begin{solution}{exo:g8:cosine:6}
$\cos\theta$ is een been gedeeld door de schuine zijde, en de schuine
zijde is de \emph{langste} zijde van een \omterm{def:g6:shapes:triangles}{rechthoekige driehoek}: het
\omterm{def:g3:division:remainder}{quotiënt} is altijd kleiner dan $1$. Een waarde $1.2$ zou een been
langer dan de schuine zijde betekenen --- onmogelijk.
\end{solution}

\begin{solution}{exo:g8:cosine:7}
Horizontale overspanning $= 25 \cos 12^\circ \approx 25 \times 0.978
\approx 24.5$ m.
\end{solution}

\begin{solution}{exo:g8:cosine:8}
De cosinus daalt naarmate de hoek groeit
(\cref{prop:g8:cosine:bounds}):
\[
\cos 70^\circ < \cos 45^\circ < \cos 20^\circ .
\]
(Rekenmachine: $0.342 < 0.707 < 0.940$.)
\end{solution}

\begin{solution}{exo:g8:cosine:9}
De grondafstand is aanliggend aan $\theta$:
$\cos 68^\circ = \frac{1.2}{L}$, dus
$L = \frac{1.2}{0.375} = 3.2$ m. Hoogte (Pythagoras):
$\sqrt{3.2^2 - 1.2^2} = \sqrt{10.24 - 1.44} = \sqrt{8.8} \approx 3.0$
m.
\end{solution}

\begin{solution}{exo:g8:cosine:10}
\emph{1.} $A$ ligt op een cirkel met middellijn $[BC]$: \omterm{def:g3:shapes:rightangle}{rechte hoek}
in $A$ (\cref{thm:g8:cosine:circle}).

\emph{2.} $BC = 9$ cm (tweemaal de \omterm{def:g3:shapes:shapes}{straal}). Pythagoras:
$AC = \sqrt{9^2 - 5.4^2} = \sqrt{81 - 29.16} = \sqrt{51.84} = 7.2$
cm. En $\cos\widehat B = \dfrac{AB}{BC} = \dfrac{5.4}{9} = 0.6$.
\end{solution}

\begin{solution}{exo:g8:cosine:11}
In de \omterm{def:g6:shapes:triangles}{gelijkzijdige driehoek} met zijde $1$ landt de hoogte vanuit één
\omterm{def:g5:solids:def}{hoekpunt} op het \emph{midden} van de overstaande zijde
(\omterm{def:g6:symmetry:axis}{symmetrieas}). De \omterm{def:g3:division:half}{helft} van de driehoek is \omterm{def:g6:shapes:triangles}{rechthoekig} met schuine
zijde $1$ (een volle zijde), een hoek van $60^\circ$ in het
basishoekpunt, en aanliggende zijde $\frac12$ (de \omterm{def:g3:division:half}{helft} van de
basis). Dus
\[
\cos 60^\circ = \frac{1/2}{1} = \frac12 .
\]
\end{solution}
