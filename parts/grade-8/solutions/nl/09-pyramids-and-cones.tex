\section*{Hoofdstuk \ref{ch:g8:solids} --- Piramide en kegel}

\begin{solution}{exo:g8:solids:1}
Tetraëder: $4$ \omterm{def:g5:solids:def}{vlakken}, $6$ \omterm{def:g5:solids:def}{ribben}, $4$ \omterm{def:g5:solids:def}{hoekpunten}
($4 + 4 = 6 + 2$). Zeshoekig grondvlak: $7$ \omterm{def:g5:solids:def}{vlakken}, $12$ \omterm{def:g5:solids:def}{ribben},
$7$ \omterm{def:g5:solids:def}{hoekpunten} ($7 + 7 = 12 + 2$).
\end{solution}

\begin{solution}{exo:g8:solids:2}
$V = \frac13 \times 36 \times 10 = 120$ cm$^3$.

\omterm{def:g6:shapes:triangles}{Rechthoekig} grondvlak: $B = 24$ cm$^2$, dus
$V = \frac13 \times 24 \times 7.5 = 60$ cm$^3$.
\end{solution}

\begin{solution}{exo:g8:solids:3}
$V = \frac13 \pi \times 25 \times 9 = 75\pi \approx 236$ cm$^3$.
\end{solution}

\begin{solution}{exo:g8:solids:4}
\omterm{def:g7:areas:prism}{Cilinder}: $\pi \times 16 \times 12 = 192\pi \approx 603$ cm$^3$.
\omterm{def:g8:solids:def}{Kegel}: $\frac{192\pi}{3} = 64\pi \approx 201$ cm$^3$. Verhouding: de
\omterm{def:g8:solids:def}{kegel} is één derde van de \omterm{def:g7:areas:prism}{cilinder}.
\end{solution}

\begin{solution}{exo:g8:solids:5}
$56 = \frac13 \times B \times 8$, dus $B = \frac{56 \times 3}{8} = 21$
cm$^2$.
\end{solution}

\begin{solution}{exo:g8:solids:6}
Het mantelvak rolt af tot een schijfsector waarvan de \omterm{def:g3:shapes:shapes}{straal} de
\emph{mantellengte} $s$ is (de afstand van de top tot de rand --- dat
\omterm{def:g6:lines:objects}{lijnstuk} ligt op het oppervlak), niet de hoogte $h$.
\end{solution}

\begin{solution}{exo:g8:solids:7}
$V = \frac13 \times 35^2 \times 22 = \frac13 \times 1225 \times 22
= \frac{26\,950}{3} \approx 8\,983$ m$^3$ --- ongeveer $9\,000$
kubieke meters.
\end{solution}

\begin{solution}{exo:g8:solids:8}
Hoogte: $h = \sqrt{13^2 - 5^2} = \sqrt{169 - 25} = \sqrt{144} = 12$
cm. \omterm{def:g6:measure:volume}{Volume}: $V = \frac13 \pi \times 25 \times 12 = 100\pi$ cm$^3$.
\end{solution}

\begin{solution}{exo:g8:solids:9}
Zelfde \omterm{def:g3:shapes:shapes}{straal}, dus het \omterm{def:g6:measure:volume}{volume} van de \omterm{def:g8:solids:def}{kegel} is één derde van het
\omterm{def:g6:measure:volume}{volume} van de \omterm{def:g7:areas:prism}{cilinder} \emph{voor dezelfde hoogte}: de vloeistof
bereikt $\frac{9}{3} = 3$ cm in de \omterm{def:g7:areas:prism}{cilinder}.
\end{solution}

\begin{solution}{exo:g8:solids:10}
Zand $=$ één \omterm{def:g8:solids:def}{kegel}: $V = \frac13 \pi \times 4 \times 4.5 = 6\pi
\approx 18.8$ cm$^3$. De zandloper houdt twee zulke \omterm{def:g8:solids:def}{kegels}, dus het
zand vult exact de \omterm{def:g3:division:half}{helft} van het totale binnenvolume.
\end{solution}

\begin{solution}{exo:g8:solids:11}
\emph{1.} Ja: de formule $V = \frac13 Bh$ geldt voor elke positie van
de top, zolang $h$ de loodrechte afstand tot het grondvlak is.
$V = \frac13 \times 36 \times 8 = 96$ cm$^3$.

\emph{2.} Gronddiagonaal: $\sqrt{6^2 + 6^2} = 6\sqrt2$ cm. De
langste zijribbe is de schuine zijde van een \omterm{def:g6:shapes:triangles}{rechthoekige driehoek}
met benen $6\sqrt2$ (in het grondvlak) en $8$ (verticaal):
\[
\sqrt{(6\sqrt2)^2 + 8^2} = \sqrt{72 + 64} = \sqrt{136} \approx 11.7
\text{ cm}.
\]
\end{solution}
