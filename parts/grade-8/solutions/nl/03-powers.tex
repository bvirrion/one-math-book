\section*{Hoofdstuk \ref{ch:g8:powers} --- Machten}

\begin{solution}{exo:g8:powers:1}
$2^6 = 64$; \quad $3^3 = 27$; \quad $10^5 = 100\,000$; \quad
$1^{100} = 1$; \quad $0.1^2 = 0.01$; \quad $6^0 = 1$.
\end{solution}

\begin{solution}{exo:g8:powers:2}
$(-3)^2 = 9$; \quad $-3^2 = -9$; \quad $(-1)^{15} = -1$; \quad
$(-5)^3 = -125$; \quad $\left(\frac23\right)^2 = \frac49$.
\end{solution}

\begin{solution}{exo:g8:powers:3}
$7^4 \times 7^5 = 7^9$; \quad
$\dfrac{2^9}{2^3} = 2^6$; \quad
$\left(10^3\right)^4 = 10^{12}$; \quad
$5^6 \times 5^{-2} = 5^4$; \quad
$3^4 \times 7^4 = 21^4$ (zelfde exponent: vermenigvuldig de
grondtallen).
\end{solution}

\begin{solution}{exo:g8:powers:4}
$10^{-2} = 0.01$; \quad
$4 \times 10^3 = 4000$; \quad
$2.5 \times 10^{-4} = 0.00025$; \quad
$10^0 = 1$.
\end{solution}

\begin{solution}{exo:g8:powers:5}
$10^5$; \quad $10^{-1}$; \quad $10^{10}$; \quad $10^{-6}$.
\end{solution}

\begin{solution}{exo:g8:powers:6}
$\dfrac{10^7 \times 10^{-3}}{10^2} = \dfrac{10^4}{10^2} = 10^2 =
100$.

$\dfrac{2^5 \times 2^4}{2^6} = \dfrac{2^9}{2^6} = 2^3 = 8$.

$\left(2^2\right)^3 \times 2^{-4} = 2^6 \times 2^{-4} = 2^2 = 4$.
\end{solution}

\begin{solution}{exo:g8:powers:7}
$2^3 \times 2^4 = 2^{12}$: \emph{onwaar} --- exponenten tellen op:
$2^7$.

$5^2 + 5^3 = 5^5$: \emph{onwaar} --- geen regel voor sommen:
$25 + 125 = 150$, terwijl $5^5 = 3125$.

$(3^2)^4 = 3^8$: \emph{waar}.

$10^3 \times 10^3 = 100^3$: \emph{waar} --- beide gelijk aan
$10^6$ (links: $10^{3+3}$; rechts: $(10^2)^3$).
\end{solution}

\begin{solution}{exo:g8:powers:8}
Nieuwe personen op dag $4$: elk van de $3^3 = 27$ personen
geïnformeerd op dag 3 vertelt het aan drie anderen: $3^4 = 81$.
Weten aan het einde van dag 4: $3 + 9 + 27 + 81 = 120$ personen.
\end{solution}

\begin{solution}{exo:g8:powers:9}
\emph{1.} Dikte: $2^{10} \times 10^{-4}$ m
$= 1024 \times 10^{-4}$ m $\approx 0.1$ m $= 10$ cm.

\emph{2.} $2^{42} \times 10^{-4} \approx 4.4 \times 10^{12} \times
10^{-4} = 4.4 \times 10^8$ m, groter dan $3.8 \times 10^8$ m: na
$42$ (theoretische!) vouwen zou de papieren stapel de Maan
passeren.
\end{solution}

\begin{solution}{exo:g8:powers:10}
$2^{10} = 1024$; $10^3 = 1000$; $3^6 = 729$; $5^4 = 625$. Volgorde:
\[
5^4 < 3^6 < 10^3 < 2^{10} .
\]
\end{solution}

\begin{solution}{exo:g8:powers:11}
$2^{100} = \left(2^{10}\right)^{10} = 1024^{10}$ en
$10^{30} = \left(10^3\right)^{10} = 1000^{10}$. Omdat $1024 > 1000$
houdt het vermenigvuldigen van tien kopieën van elk de ongelijkheid:
$2^{100} > 10^{30}$.
\end{solution}
