\begin{solution}{exo:g8:equations:10}

\begin{solution}{exo:g8:equations:11}

quando

\begin{solution}{exo:g8:equations:12}
Multiplique ambos os lados por $20$:
\[
5(x + 3) = 4(2x - 1)
\ \Longrightarrow\
5x + 15 = 8x - 4
\ \Longrightarrow\
19 = 3x
\ \Longrightarrow\
x = \frac{19}{3}.
\]
Verificar: $\frac{19/3 + 3}{4} = \frac{28/3}{4} = \frac{28}{12} =
\frac73$, e $\frac{2 \times 19/3 - 1}{5} = \frac{35/3}{5} =
\frac{35}{15} = \frac73$. Igual: a solução é $\frac{19}{3}$.
\end{solution}

\begin{solution}{pb:g8:equations:1}
\textbf{1.} Dupla distributividade
(\cref{thm:g8:equations:expand}) com $ c = a $, $ d = b $:
\[
(a + b)(a + b) = a^2 + ab + ba + b^2 = a^2 + 2ab + b^2 ,
\]
desde $ ab $ e $ ba $ são iguais \omterm{def:g2:mult:def}{produto}.

\textbf{2.} Mesma expansão, atentos aos sinais:
\[
(a - b)(a - b) = a^2 - ab - ba + b^2 = a^2 - 2ab + b^2 ,
\]
\[
(a + b)(a - b) = a^2 - ab + ba - b^2 = a^2 - b^2 :
\]
desta vez os termos cruzados $-ab $ e $+ba $ cancelar em vez de
duplicando.

\textbf{3.} O quadrado do lado $ a + b $ corta em quatro pedaços: um
quadrado $ a \times a $ (o $ a^2$), um quadrado $ b \times b $ (o
$ b^2$), e \emph{dois} retângulos $ a \times b $ (juntos, o
$2ab $). A identidade $(a+b)^2 = a^2 + 2ab + b^2$ é a declaração
que os quatro \omterm{def:g6:measure:area}{áreas} preencha o grande quadrado --- é a imagem de
distributividade dupla com ambos os lados divididos da mesma maneira.

\textbf{4.} Com as identidades:
\begin{align*}
101^2 &= (100 + 1)^2 = 10\,000 + 2 \times 100 + 1 = 10\,201, \\
99^2 &= (100 - 1)^2 = 10\,000 - 200 + 1 = 9\,801, \\
49 \times 51 &= (50 - 1)(50 + 1) = 50^2 - 1^2 = 2\,499 .
\end{align*}

\textbf{5.} Para $ a = 3$, $ b = 4$: $(3 + 4)^2 = 49$ mas
$3^2 + 4^2 = 9 + 16 = 25$. A fórmula de Zoe esquece os dois
$ a \times b $ retângulos da imagem --- os que faltam
$2ab = 24$, e de fato $25 + 24 = 49$.

\textbf{6.} $1$; $1 + 3 = 4$; $1 + 3 + 5 = 9$;
$1 + 3 + 5 + 7 = 16$; $1 + 3 + 5 + 7 + 9 = 25$. Os resultados são
os quadrados perfeitos $1^2, 2^2, 3^2, 4^2, 5^2$.

\textbf{7.} O números \omterm{def:g3:numbers:evenodd}{ímpares} começar em $1$ e suba em passos de
$2$: o $ n $-th é alcançado depois $ n - 1$ etapas de $2$ de
$1$, então é igual $1 + 2(n - 1) = 2n - 1$. Verificar: $ n = 1, 2, 3$
dar $1, 3, 5$.

\textbf{8.} Pela primeira identidade,
\[
(n + 1)^2 - n^2 = n^2 + 2n + 1 - n^2 = 2n + 1 :
\]
o \omterm{ex:g1:subtraction:difference}{diferença} entre dois quadrados consecutivos é exatamente o
$(n+1)$-º número \omterm{def:g3:numbers:evenodd}{ímpar} (pergunta 7 com $ n + 1$ no lugar de $ n $:
$2(n+1) - 1 = 2n + 1$).

\textbf{9.} Construa os quadrados um após o outro. Comece com um
célula única: $1 = 1^2$. Para passar do quadrado do lado $1$ para
o quadrado do lado $2$, adicionar $2 \times 1 + 1 = 3$ células (pergunta
8): $1 + 3 = 2^2$. Para passar para o lado $3$, adicionar $2 \times 2 + 1 = 5$
células: $1 + 3 + 5 = 3^2$. Cada novo número \omterm{def:g3:numbers:evenodd}{ímpar} é exatamente o
Camada em forma de L completando o próximo quadrado, então depois de adicionar o
primeiro $ n $ números \omterm{def:g3:numbers:evenodd}{ímpares} o quadrado do lado $ n $ está completo:
$1 + 3 + \dots + (2n - 1) = n^2$.

\textbf{10.} O números \omterm{def:g3:numbers:evenodd}{ímpares} de $1$ para $99$ são os primeiros $50$
deles ($2n - 1 = 99$ dá $ n = 50$), então
\[
1 + 3 + 5 + \dots + 99 = 50^2 = 2\,500 .
\]
E $1 + 3 + \dots + (2n - 1) = 400 = 20^2$ requer $ n = 20$:
o primeiro \emph{vinte} números \omterm{def:g3:numbers:evenodd}{ímpares} ($1$ até $39$) somam
$400$.

\textbf{11.} $ x^2 + 6x + 9 = x^2 + 2 \times 3 \times x + 3^2 =
(x + 3)^2$: a primeira identidade com $ a = x $, $ b = 3$. Então o
\omterm{def:g8:equations:def}{equação} lê $(x + 3)^2 = 0$. Um quadrado é \omterm{def:g1:counting:zero}{zero} somente quando o
o próprio número é \omterm{def:g1:counting:zero}{zero} (\cref{thm:g8:negprod:rules}: um diferente de \omterm{def:g1:counting:zero}{zero}
número tem distância diferente de \omterm{def:g1:counting:zero}{zero}, então seu quadrado tem distância diferente de \omterm{def:g1:counting:zero}{zero}
também). Por isso $ x + 3 = 0$: a única solução é $ x = -3$.

\textbf{12.} Por \cref{thm:g8:negprod:rules}, a distância de um
\omterm{def:g2:mult:def}{produto} é o \omterm{def:g2:mult:def}{produto} das distâncias. Se ambos os fatores forem
diferente de \omterm{def:g1:counting:zero}{zero}, ambas as distâncias são diferentes de \omterm{def:g1:counting:zero}{zero}, então o produtoa distância é
diferente de \omterm{def:g1:counting:zero}{zero}: o \omterm{def:g2:mult:def}{produto} não pode ser $0$. UM \omterm{def:g1:counting:zero}{zero} \omterm{def:g2:mult:def}{produto} portanto
força pelo menos um \omterm{def:g1:counting:zero}{zero} fator. Para $(x - 2)(x + 5) = 0$: qualquer
$ x - 2 = 0$ ou $ x + 5 = 0$, fornecendo as duas soluções $ x = 2$ e
$ x = -5$.

\textbf{13.} $ x^2 = 49$ torna-se $ x^2 - 49 = 0$, e o terceiro
fatores de identidade:
\[
x^2 - 49 = x^2 - 7^2 = (x - 7)(x + 7) = 0 .
\]
Pela pergunta 12, $ x = 7$ ou $ x = -7$: \emph{dois} soluções. Todo
\omterm{def:g8:equations:def}{equação} resolvido até agora neste capítulo foi de primeiro grau e
tinha exatamente uma solução; uma praça no \omterm{def:g8:equations:def}{equação} pode produzir
dois.

\textbf{14.} Traga tudo para o lado esquerdo:
$ x^2 - 6x + 9 = 0$ e reconheça a segunda identidade:
$ x^2 - 2 \times 3 \times x + 3^2 = (x - 3)^2$. Então o \omterm{def:g8:equations:def}{equação} é
$(x - 3)^2 = 0$, que força $ x - 3 = 0$ (como na questão 11):
$ x = 3$ é a única solução --- a verificação de
\cref{exo:g8:equations:3} encontrei, e a fatoração prova
não há outros.

\textbf{15.} Pela terceira identidade,
\[
2026^2 - 2025^2 = (2026 - 2025)(2026 + 2025)
= 1 \times 4051 = 4\,051 .
\]
Em geral $(n+1)^2 - n^2 = (n + 1 - n)(n + 1 + n) = 2n + 1$: o
\omterm{ex:g1:subtraction:difference}{diferença} de dois quadrados consecutivos é o \omterm{def:g1:addition:def}{soma} dos dois
números --- o mesmo $2n + 1$ como na questão 8, agora obtida de
a terceira identidade em vez da primeira.
\end{solution}
