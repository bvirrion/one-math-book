\section*{Capítulo \ref{ch:g8:powers} --- Poderes}

\begin{solution}{exo:g8:powers:1}
$2^6 = 64$; \quad $3^3 = 27$; \quad $10^5 = 100\,000$; \quad
$1^{100} = 1$; \quad $0.1^2 = 0.01$; \quad $6^0 = 1$.
\end{solution}

\begin{solution}{exo:g8:powers:2}
$(-3)^2 = 9$; \quad $-3^2 = -9$; \quad $(-1)^{15} = -1$; \quad
$(-5)^3 = -125$; \quad $\left(\frac23\right)^2 = \frac49$.
\end{solution}

\begin{solution}{exo:g8:powers:3}
$7^4 \times 7^5 = 7^9$; \quad
$\dfrac{2^9}{2^3} = 2^6$; \quad
$\left(10^3\right)^4 = 10^{12}$; \quad
$5^6 \times 5^{-2} = 5^4$; \quad
$3^4 \times 7^4 = 21^4$ (mesmo \omterm{def:g8:powers:def}{expoente}: multiplique as \omterm{def:g8:powers:def}{bases}).
\end{solution}

\begin{solution}{exo:g8:powers:4}
$10^{-2} = 0.01$; \quad
$4 \times 10^3 = 4000$; \quad
$2.5 \times 10^{-4} = 0.00025$; \quad
$10^0 = 1$.
\end{solution}

\begin{solution}{exo:g8:powers:5}
$10^5$; \quad $10^{-1}$; \quad $10^{10}$; \quad $10^{-6}$.
\end{solution}

\begin{solution}{exo:g8:powers:6}
$\dfrac{10^7 \times 10^{-3}}{10^2} = \dfrac{10^4}{10^2} = 10^2 =
100$.

$\dfrac{2^5 \times 2^4}{2^6} = \dfrac{2^9}{2^6} = 2^3 = 8$.

$\left(2^2\right)^3 \times 2^{-4} = 2^6 \times 2^{-4} = 2^2 = 4$.
\end{solution}

\begin{solution}{exo:g8:powers:7}
$2^3 \times 2^4 = 2^{12}$: \emph{falso} --- \omterm{def:g8:powers:def}{expoentes} adicionar:
$2^7$.

$5^2 + 5^3 = 5^5$: \emph{falso} --- nenhuma regra para \omterm{def:g1:addition:def}{somas}:
$25 + 125 = 150$, enquanto $5^5 = 3125$.

$(3^2)^4 = 3^8$: \emph{verdadeiro}.

$10^3 \times 10^3 = 100^3$: \emph{verdadeiro} --- ambos iguais $10^6$
(esquerda: $10^{3+3}$; certo: $(10^2)^3$).
\end{solution}

\begin{solution}{exo:g8:powers:8}
Novas pessoas no dia $4$: cada um dos $3^3 = 27$ pessoas informadas no dia
3 conta três outros: $3^4 = 81$. Sabendo no final do dia 4:
$3 + 9 + 27 + 81 = 120$ pessoas.
\end{solution}

\begin{solution}{exo:g8:powers:9}
\emph{1.} Grossura: $2^{10} \times 10^{-4}$ eu
$= 1024 \times 10^{-4}$ eu $\approx 0.1$ eu $= 10$ cm.

\emph{2.} $2^{42} \times 10^{-4} \approx 4.4 \times 10^{12} \times
10^{-4} = 4.4 \times 10^8$ m, maior que $3.8 \times 10^8$ m: depois
$42$ (teóricas!), o maço de papel passaria pela Lua.
\end{solution}

\begin{solution}{exo:g8:powers:10}
$2^{10} = 1024$; $10^3 = 1000$; $3^6 = 729$; $5^4 = 625$. Ordem:
\[
5^4 < 3^6 < 10^3 < 2^{10} .
\]
\end{solution}

\begin{solution}{exo:g8:powers:11}
$2^{100} = \left(2^{10}\right)^{10} = 1024^{10}$ e
$10^{30} = \left(10^3\right)^{10} = 1000^{10}$. Desde $1024 > 1000$,
multiplicar dez cópias de cada mantém a desigualdade:
$2^{100} > 10^{30}$.
\end{solution}

\begin{solution}{pb:g8:powers:1}
\textbf{1.} Os grãos dobram de quadrado para quadrado começando em
$1 = 2^0$: quadrado $ k $ detém $2^{k-1}$ grãos
(\cref{def:g8:powers:def}). Quadrado $64$ detém $2^{63}$.

\textbf{2.} $ S_1 = 1$, $ S_2 = 1 + 2 = 3$, $ S_3 = 3 + 4 = 7$,
$ S_4 = 7 + 8 = 15$, $ S_5 = 15 + 16 = 31$: sempre um a menos que
o próximo poder de $2$ ($2$, $4$, $8$, $16$, $32$). Conjectura:
$ S_n = 2^n - 1$.

\textbf{3.} Dobrando cada termo de $ S_n $ muda cada \omterm{def:g8:powers:def}{potência} em
um ($2 \times 2^{k} = 2^{k+1}$):
\begin{align*}
S_n &= 1 + 2 + 4 + \dots + 2^{n-1}, \\
2 \times S_n &= \phantom{1 + {}} 2 + 4 + \dots + 2^{n-1} + 2^n .
\end{align*}
Subtraindo a primeira \omterm{def:g6:lines:objects}{linha} da segunda, cada termo de $2$
para $2^{n-1}$ aparece em ambos e cancela:
\[
2 \times S_n - S_n = 2^n - 1,
\qquad\text{that is}\qquad
S_n = 2^n - 1 .
\]

\textbf{4.} O total é $ S_{64} = 2^{64} - 1$ grãos. Um
sexagésimo quinto quadrado conteria $2^{64}$ grãos: o tabuleiro inteiro
carrega exatamente um grão a menos do que aquele único quadrado carregaria.

\textbf{5.} O primeiro \omterm{def:g3:division:half}{metade} detém $ S_{32} = 2^{32} - 1$ grãos.
Todo o conselho mantém $2^{64} - 1$, então o segundo \omterm{def:g3:division:half}{metade} detém
\[
\left(2^{64} - 1\right) - \left(2^{32} - 1\right)
= 2^{64} - 2^{32}
= 2^{32} \times \left(2^{32} - 1\right)
\]
(fatorar $2^{32}$, usando $2^{32} \times 2^{32} = 2^{64}$,
\cref{thm:g8:powers:rules}): exatamente $2^{32}$ vezes o primeiro
\omterm{def:g3:division:half}{metade}.

\textbf{6.} Pelo \omterm{def:g8:powers:def}{expoente} regras, $2^{64} = 2^{4 + 60} =
2^4 \times \left(2^{10}\right)^6$. Desde $2^{10} = 1024 > 10^3$,
\[
2^{64} > 16 \times \left(10^3\right)^6 = 16 \times 10^{18}
= 1.6 \times 10^{19} .
\]

\textbf{7.} Mais do que $1.6 \times 10^{19}$ grãos em
$5 \times 10^{-2}$ g cada:
\[
1.6 \times 10^{19} \times 5 \times 10^{-2}
= 8 \times 10^{17} \text{ g} .
\]
Dividindo por $10^6$ g por tonelada: mais de $8 \times 10^{11}$
toneladas --- oitocentos bilhões de toneladas.

\textbf{8.} $\dfrac{8 \times 10^{11}}{8 \times 10^{8}} = 10^3$:
o rei prometeu pelo menos um \emph{mil anos} de hoje
colheita mundial inteira. (A lenda diz que seus conselheiros lhe contaram como
muito.)

\textbf{9.} $2^{19} = 2^9 \times 2^{10} = 512 \times 1024 =
524\,288 < 10^6$, enquanto $2^{20} = \left(2^{10}\right)^2 =
1024^2 = 1\,048\,576 > 10^6$. Então o menor \omterm{def:g8:powers:def}{expoente} é
$ n = 20$: vinte duplicações ultrapassam o milhão.

\textbf{10.} O pagamento em dia $ n $ é $2^{n-1}$ centavos (dia 1:
$2^0 = 1$). Agora
\[
2^{26} = 2^6 \times \left(2^{10}\right)^2
= 64 \times 1\,048\,576 = 67\,108\,864 < 10^8,
\]
\[
2^{27} = 2 \times 2^{26} = 134\,217\,728 > 10^8 .
\]
Portanto, o pagamento diário excede primeiro $10^8$ centavos quando $ n - 1 = 27$:
no dia $28$.

\textbf{11.} $21 = 16 + 4 + 1$: pesos $16$, $4$ e $1$ g.
$27 = 16 + 8 + 2 + 1$: pesos $16$, $8$, $2$ e $1$ g.

\textbf{12.} Pela questão 3, os pesos $1, 2, 4, 8$ juntos
pesar $ S_4 = 2^4 - 1 = 15$ g. Então sem o $16$ g pesar o
comerciante não pode passar $15$ g: qualquer alvo de $16$ g ou mais deve
use-o. E um alvo de $15$ g ou menos não deve usá-lo, pois o
$16$ g de peso por si só já excede a meta. A escolha do
maior peso é portanto \emph{forçado}. O que resta é um
meta de no máximo $15$ g a ser formado com $1, 2, 4, 8$ --- e
o mesmo argumento se repete: $8$ é forçado (usado se o \omterm{def:g3:division:remainder}{restante}
o alvo é $\geq 8$, não utilizado de outra forma, porque $1 + 2 + 4 = 7$),
então $4$ (porque $1 + 2 = 3$), então $2$, então $1$.

\textbf{13.} Após as escolhas forçadas, o alvo \omterm{def:g3:division:remainder}{restante}
após cada etapa é no máximo o total dos pesos \omterm{def:g3:division:remainder}{restantes},
então o processo termina com \omterm{def:g3:division:remainder}{restante} $0$: cada alvo de $1$ para
$31$ é alcançado. E como todas as escolhas ao longo do caminho foram forçadas,
nenhuma outra seleção de pesos pode atingir o mesmo objetivo: o
escrita de cada número de $1$ para $31$ como um \omterm{def:g1:addition:def}{soma} de distinto
poderes de $2$ existe e é \emph{exclusivo}.

\textbf{14.} Os seis pesos no total $ S_6 = 2^6 - 1 = 63$ g, e
o mesmo argumento da escolha forçada cobre todos os alvos, desde $1$ para
$63$ g. Para $45$: o alvo é $\geq 32$, então use $32$; restos
$13 < 16$, então pule $16$; usar $8$ (restos $5$), usar $4$ (restos
$1$), pular $2$, usar $1$:
\[
45 = 32 + 8 + 4 + 1 = 2^5 + 2^3 + 2^2 + 2^0 .
\]

\textbf{15.} Escolhendo um $0$ ou um $1$ em cada um dos $64$ quadrados
equivale a escolher quais poderes $2^0, 2^1, \dots, 2^{63}$ para
incluir em um \omterm{def:g1:addition:def}{soma} --- exatamente a pesagem do comerciante $64$
pesos. O menor número armazenável é $0$ (todos os quadrados em
$0$); o maior é o \omterm{def:g1:addition:def}{soma} de \emph{todos} os poderes, o que é
o total do rei: $ S_{64} = 2^{64} - 1$ (Parte I). Pelo
argumento de escolha forçada, todos os números inteiros intermediários são alcançados
exatamente uma vez: uma $64$ A máquina de bits armazena precisamente o todo
números de $0$ para $2^{64} - 1$.
\end{solution}
