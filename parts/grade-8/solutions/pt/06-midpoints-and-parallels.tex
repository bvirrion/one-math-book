\section*{Capítulo \ref{ch:g8:midpoints} --- Pontos Médios e Paralelos}

\begin{solution}{exo:g8:midpoints:1}
$M $ e $ N $ são os pontos médios de dois lados do triângulo $ RST $: por
o teorema do ponto médio (\cref{thm:g8:midpoints:direct}),
$(MN) \parallel (ST)$ e $ MN = \frac{ST}{2} = 3.5$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:2}
$ I $ é um ponto médio e $(IJ) \parallel (BC)$: pelo \emph{conversar}
(\cref{thm:g8:midpoints:converse}), $ J $ é o ponto médio de $[AC]$,
então $ AJ = \frac{11}{2} = 5.5$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:3}
A medição dá $ IJ = 3.5$ cm $= \frac{BC}{2}$. \omterm{def:g6:measure:perimeter}{Perímetro} de
$ AIJ $: $ AI = 4$, $ AJ = 3$, $ IJ = 3.5$: total $10.5$ cm --- \omterm{def:g3:division:half}{metade} o
\omterm{def:g6:measure:perimeter}{perímetro} de $ ABC $ ($8 + 6 + 7 = 21$ cm).
\end{solution}

\begin{solution}{exo:g8:midpoints:4}
Cada lado de $ IJK $ é \omterm{def:g3:division:half}{metade} um lado de $ ABC $: $5$, $6$ e $8$ cm.
\omterm{def:g6:measure:perimeter}{Perímetro}: $19$ cm, \omterm{def:g3:division:half}{metade} de $10 + 12 + 16 = 38$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:5}
O triângulo do ponto médio é um dos quatro triângulos iguais de
\cref{prop:g8:midpoints:medial}: \omterm{def:g6:measure:area}{área} $\frac{36}{4} = 9$ cm $^2$ ---
e cada um dos quatro pequenos triângulos tem \omterm{def:g6:measure:area}{área} $9$ cm $^2$.
\end{solution}

\begin{solution}{exo:g8:midpoints:6}
Pelo teorema do ponto médio em $ ABC $: $(IJ) \parallel (BC)$, ou seja\
$(IJ) \parallel (BK)$, e $ IJ = \frac{BC}{2} = BK $ (como $ K $ é o
ponto médio de $[BC]$). Dois lados opostos de $ IJKB $ são \omterm{def:g6:lines:perp}{paralelo} e
igual: é um \omterm{def:g7:central:parallelogram}{paralelogramo}
(\cref{thm:g7:central:recognize}, critério 3).
\end{solution}

\begin{solution}{exo:g8:midpoints:7}
A costura une dois pontos médios: $\frac{6.4}{2} = 3.2$ m. Em um
\omterm{def:g3:division:half}{trimestre} do topo vértice, o desenho sugere uma costura de
$\frac{6.4}{4} = 1.6$ m, \omterm{def:g6:lines:perp}{paralelo} para a \omterm{def:g8:powers:def}{base} --- confirmado por Thales
em \cref{ch:g9:thales}.
\end{solution}

\begin{solution}{exo:g8:midpoints:8}
\emph{1.} $ M $ é o ponto médio de $[BC]$ e $(MN) \parallel (AB)$:
pelo \omterm{def:g8:fractions:inverse}{inverso} do teorema do ponto médio (no triângulo $ ABC $,
lado $[AC]$), $ N $ é o ponto médio de $[AC]$.

\emph{2.} $(MN) \parallel (AB)$ e $(AB) \perp (AC)$ (\omterm{def:g3:shapes:rightangle}{ângulo reto}
no $ A $): uma \omterm{def:g6:lines:objects}{linha} \omterm{def:g6:lines:perp}{paralelo} para $(AB)$ também é \omterm{def:g6:lines:perp}{perpendicular} para $(AC)$
(\cref{prop:g6:lines:facts}). Então $(MN) \perp (AC)$ --- o \omterm{def:g6:lines:objects}{segmento}
$[MN]$ é o \omterm{def:g6:symmetry:bisector}{bissetriz perpendicular} peça mostrando novamente que
$ MA = MC $ (\cref{thm:g8:cosine:circle}).
\end{solution}

\begin{solution}{exo:g8:midpoints:9}
\emph{1.} Em triângulo $ ABC $, $ I $ e $ J $ são pontos médios de $[AB]$ e
$[BC]$: $(IJ) \parallel (AC)$ e $ IJ = \frac{AC}{2}$. Em triângulo
$ ACD $, $ L $ e $ K $ são pontos médios de $[DA]$ e $[CD]$:
$(LK) \parallel (AC)$ e $ LK = \frac{AC}{2}$.

\emph{2.} Então $[IJ]$ e $[LK]$ são \omterm{def:g6:lines:perp}{paralelo} (ambos \omterm{def:g6:lines:perp}{paralelo} para o
diagonal $(AC)$) e iguais (ambos \omterm{def:g3:division:half}{metade} de $ AC $): $ IJKL $ é um
\omterm{def:g7:central:parallelogram}{paralelogramo} (\cref{thm:g7:central:recognize}, critério 3) ---
para \emph{todo} \omterm{def:g4:geometry:polygon}{quadrilátero} $ ABCD $, como conjecturado em
\cref{exo:g7:central:11}.
\end{solution}

\begin{solution}{pb:g8:midpoints:1}
\textbf{1.} Qualquer que seja o triângulo desenhado, as três medianas aparecem
passar por um único ponto, situado dentro do triângulo,
visivelmente mais próximo do ponto médio de cada lado do que do oposto
vértice. O resto do problema prova isso.

\textbf{2.} Pegue a mediana $[AK]$ do triângulo $ ABC $: o
triângulos $ ABK $ e $ AKC $ tem \omterm{def:g8:powers:def}{bases} $ BK $ e $ KC $ do mesmo
comprimento ($ K $ é o ponto médio de $[BC]$) transportado pela mesma \omterm{def:g6:lines:objects}{linha}
$(BC)$, e a mesma altura: a distância de $ A $ para $(BC)$.
Pelo \omterm{def:g6:measure:area}{área} fórmula, cada um tem \omterm{def:g6:measure:area}{área}
\omterm{def:g8:powers:def}{base} $\times $ altura $\div\ 2$, com \omterm{def:g8:powers:def}{bases} iguais e iguais
alturas: o \omterm{def:g6:measure:area}{áreas} são iguais. Para um triângulo de \omterm{def:g6:measure:area}{área}
$36$ cm $^2$: dois pedaços de $18$ cm $^2$ cada.

\textbf{3.} $ I $ e $ J $ são os pontos médios dos lados $[AB]$ e
$[AC]$ do triângulo $ ABC $, então por
\cref{thm:g8:midpoints:direct}: $(IJ) \parallel (BC)$ e
$ IJ = \frac{BC}{2}$.

\textbf{4.} $ M $ e $ N $ são os pontos médios dos lados $[GB]$ e
$[GC]$ do triângulo $ GBC $, então pelo mesmo teorema:
$(MN) \parallel (BC)$ e $ MN = \frac{BC}{2}$.

\textbf{5.} Ambos $(IJ)$ e $(MN)$ são \omterm{def:g6:lines:perp}{paralelo} para $(BC)$, por isso
\omterm{def:g6:lines:perp}{paralelo} um para o outro; e $ IJ = \frac{BC}{2} = MN $. O
\omterm{def:g4:geometry:polygon}{quadrilátero} $ IJNM $ portanto tem dois lados opostos, $[IJ]$ e
$[NM]$, \omterm{def:g6:lines:perp}{paralelo} e do mesmo comprimento: pelo critério 3 de
\cref{thm:g7:central:recognize}, $ IJNM $ é um \omterm{def:g7:central:parallelogram}{paralelogramo}.

\textbf{6.} $ G $ encontra-se na mediana $[CI]$, e $ N $ é o
ponto médio de $[GC]$, um \omterm{def:g6:lines:objects}{segmento} transportado por essa mesma \omterm{def:g6:lines:objects}{linha} mediana:
então $ I $, $ N $ (e $ G $, $ C $) todos mentem $(CI)$. Da mesma maneira $ J $, $ M $
e $ G $ deite-se na \omterm{def:g6:lines:objects}{linha} mediana $(BJ)$. No \omterm{def:g7:central:parallelogram}{paralelogramo}
$ IJNM $, as diagonais são $[IN]$ e $[JM]$; eles são levados por
as duas \omterm{def:g6:lines:objects}{linhas} medianas, que se cruzam em $ G $ --- então as diagonais
cruzar em $ G $. Pelo \emph{definição} de um \omterm{def:g7:central:parallelogram}{paralelogramo}
(critério 1 de \cref{thm:g7:central:recognize}: as diagonais
têm o mesmo ponto médio), $ G $ é o ponto médio de $[IN]$ \emph{e}
de $[JM]$: $ GI = GN $ e $ GJ = GM $.

\textbf{7.} $ M $ é o ponto médio de $[GB]$, então $ BM = MG $; e
$ MG = GJ $ pela pergunta 6. Portanto $ BM = MG = GJ $: a peça mediana
$[BJ]$ é cortado em três partes iguais, das quais $ BG $ leva dois:
\[
BG = 2 \times GJ .
\]
Simetricamente, $ N $ é o ponto médio de $[GC]$ e $ GN = GI $, então
$ CN = NG = GI $ e $ CG = 2 \times GI $: na mediana $[CI]$ também,
o ponto de passagem fica a dois terços do caminho do vértice.

\textbf{8.} O argumento não usou nada de especial sobre o \omterm{def:g3:numbers:evenodd}{par}
$[BJ]$, $[CI]$: correr no \omterm{def:g3:numbers:evenodd}{par} $[BJ]$, $[AK]$, isso mostra a sua
ponto de passagem também fica em $[BJ]$ a dois terços do caminho
$ B $. Mas há apenas \emph{um} ponto do \omterm{def:g6:lines:objects}{segmento} $[BJ]$ em
distância $\frac23 BJ $ de $ B $ --- então o ponto de cruzamento de
$[BJ]$ e $[AK]$ é o mesmo $ G $. Todas as três medianas
portanto, passe por $ G $: eles são \emph{simultâneo}.

\textbf{9.} $ BG = \frac23 \times 9 = 6$ centímetros e
$ GJ = \frac13 \times 9 = 3$ cm.

\textbf{10.} Na questão 8, o mesmo cálculo de dois terços é válido
na mediana $[AK]$ (execute o argumento das questões 4 a 7 com
o \omterm{def:g3:numbers:evenodd}{par} $[AK]$, $[BJ]$): $ AG = 2 \times GK $.

\textbf{11.} No triângulo $ GBC $, o ponto $ K $ é o ponto médio
do lado $[BC]$, então $[GK]$ é uma mediana de $ GBC $; por pergunta
2, os triângulos $ GBK $ e $ GKC $ tem igual \omterm{def:g6:measure:area}{áreas}.

\textbf{12.} A mediana $[AK]$ cortes $ ABC $ em $ ABK $ e $ AKC $ de
igual \omterm{def:g6:measure:area}{áreas} (questão 2). Removendo de cada um seu pequeno triângulo
da questão 11:
\[
\text{area}(ABG) = \text{area}(ABK) - \text{area}(GBK),
\qquad
\text{area}(ACG) = \text{area}(AKC) - \text{area}(GKC),
\]
e o removido \omterm{def:g6:measure:area}{áreas} são iguais, então
$\text{area}(ABG) = \text{area}(ACG)$. O mesmo cálculo
ao longo da mediana $[BJ]$ --- quais \omterm{def:g3:division:half}{metades} $ ABC $ em $ ABJ $ e
$ CBJ $, enquanto $[GJ]$, uma mediana do triângulo $ GAC $, \omterm{def:g3:division:half}{metade}
em $ GAJ $ e $ GCJ $ --- dá
$\text{area}(ABG) = \text{area}(CBG)$. Então os três triângulos
$ ABG $, $ BCG $, $ CAG $ tem igual \omterm{def:g6:measure:area}{áreas}, e já que juntos eles
azulejo $ ABC $, cada um tem \omterm{def:g6:measure:area}{área} um terço do total.

\textbf{13.} No triângulo $ ABG $, o ponto $ I $ é o ponto médio
do lado $[AB]$, então $[GI]$ é uma mediana de $ ABG $ \emph{do
vértice $ G $}: corta $ ABG $ em dois triângulos iguais \omterm{def:g6:measure:area}{áreas}
(questão 2), nomeadamente $ AIG $ e $ IBG $. O mesmo acontece em $ BCG $
(mediana $[GK]$) e em $ CAG $ (mediana $[GJ]$). Cada terço de
$ ABC $ é cortado \omterm{def:g3:division:half}{metade}: as seis fatias ao redor $ G $ cada um tem \omterm{def:g6:measure:area}{área}
$\frac13 \times \frac12 = \frac16$ do triângulo.

\textbf{14.} Na mediana $[AK]$, de $ A(0,0)$ para $ K(3,3)$: $ G $
fica a dois terços do caminho, então suas coordenadas são
$\left(\frac23 \times 3,\ \frac23 \times 3\right) = (2, 2)$. Ligado
a mediana $[BJ]$, de $ B(6, 0)$ para $ J(0, 3)$: dois terços do
maneira significa adicionar dois terços do deslocamento ($-6$
horizontalmente, $+3$ verticalmente):
\[
\left(6 + \tfrac23 \times (-6),\ 0 + \tfrac23 \times 3\right)
= (2, 2) .
\]
O mesmo ponto --- como a pergunta 8 prometeu. E
$\frac{0 + 6 + 0}{3} = 2$, $\frac{0 + 0 + 6}{3} = 2$: o centro
da gravidadeAs coordenadas de são as \emph{médias} dos três
\omterm{def:g5:solids:def}{vértices}'coordenadas.

\textbf{15.} Pela questão 2, um fio de faca colocado ao longo de uma mediana tem
a mesma quantidade de papelão de cada lado (igual \omterm{def:g6:measure:area}{áreas}), então o
o triângulo se equilibra ao longo de cada uma das três \omterm{def:g6:lines:objects}{linhas} medianas. Um ponto
do equilíbrio deve estar em todas as três \omterm{def:g6:lines:objects}{linhas} de equilíbrio ao mesmo tempo, e por
questão 8 as três medianas compartilham exatamente um ponto: $ G $. Isso
é para onde vai a ponta do lápis. (Igual \omterm{def:g6:measure:area}{áreas} em ambos os lados está o
versão crível do argumento; o honesto, pesando
\emph{onde} o papelão fica e não apenas quanto há,
precisa do cálculo integral da universidade \omterm{def:g6:measure:volume}{volumes} --- qual
confirma $ G$.)
\end{solution}
