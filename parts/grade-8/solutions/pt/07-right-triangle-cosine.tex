\section*{Capítulo \ref{ch:g8:cosine} --- O triângulo retângulo e o cosseno}

\begin{solution}{exo:g8:cosine:1}
Mediana do \omterm{def:g3:shapes:rightangle}{ângulo reto}: \omterm{def:g3:division:half}{metade} a hipotenusa, $6.5$ cm
(\cref{thm:g8:cosine:circle}). O \omterm{pb:g6:symmetry:1}{círculo circunscrito} está centrado
no ponto médio da hipotenusa com \omterm{def:g3:shapes:shapes}{raio} $6.5$ cm.
\end{solution}

\begin{solution}{exo:g8:cosine:2}
O ângulo em $ A $ está certo ($[BC]$ é um \omterm{def:g4:geometry:circle}{diâmetro}: ponto 2 de
\cref{thm:g8:cosine:circle}). As medidas satisfazem
$ AB^2 + AC^2 \approx 64 = BC^2$, até medir a precisão.
\end{solution}

\begin{solution}{exo:g8:cosine:3}
Hipotenusa: $[EF]$ (em frente ao \omterm{def:g3:shapes:rightangle}{ângulo reto} $ D $). Adjacente a
$\widehat E $: $[ED]$. Oposto: $[DF]$. Então
$\cos\widehat E = \dfrac{ED}{EF}$.
\end{solution}

\begin{solution}{exo:g8:cosine:4}
\emph{1.} adjacente $= 10 \cos 35^\circ \approx 8.2$.

\emph{2.} hipotenusa $= \dfrac{6}{\cos 50^\circ} \approx
\dfrac{6}{0.643} \approx 9.3$.
\end{solution}

\begin{solution}{exo:g8:cosine:5}
$\cos\theta = \frac{8}{10} = 0.8$, então
$\theta = \cos^{-1}(0.8) \approx 37^\circ $.
\end{solution}

\begin{solution}{exo:g8:cosine:6}
$\cos\theta $ é uma perna dividida pela hipotenusa, e a hipotenusa
é o \emph{mais longo} lado de um triângulo retângulo: o \omterm{def:g3:division:remainder}{quociente} é
sempre menor que $1$. Um valor de $1.2$ significaria uma perna mais longa
do que a hipotenusa --- impossível.
\end{solution}

\begin{solution}{exo:g8:cosine:7}
Vão horizontal $= 25 \cos 12^\circ \approx 25 \times 0.978 \approx
24.5$ m.
\end{solution}

\begin{solution}{exo:g8:cosine:8}
O cosseno diminui à medida que o ângulo aumenta
(\cref{prop:g8:cosine:bounds}):
\[
\cos 70^\circ < \cos 45^\circ < \cos 20^\circ .
\]
(Calculadora: $0.342 < 0.707 < 0.940$.)
\end{solution}

\begin{solution}{exo:g8:cosine:9}
A distância do solo é adjacente a $\theta $:
$\cos 68^\circ = \frac{1.2}{L}$, então
$ L = \frac{1.2}{0.375} = 3.2$ m. Altura (Pitágoras):
$\sqrt{3.2^2 - 1.2^2} = \sqrt{10.24 - 1.44} = \sqrt{8.8} \approx 3.0$
m.
\end{solution}

\begin{solution}{exo:g8:cosine:10}
\emph{1.} $ A $ está em um círculo de \omterm{def:g4:geometry:circle}{diâmetro} $[BC]$: \omterm{def:g3:shapes:rightangle}{ângulo reto} no $ A $
(\cref{thm:g8:cosine:circle}).

\emph{2.} $ BC = 9$ cm (duas vezes o \omterm{def:g3:shapes:shapes}{raio}). Pitágoras:
$ AC = \sqrt{9^2 - 5.4^2} = \sqrt{81 - 29.16} = \sqrt{51.84} = 7.2$
cm. E $\cos\widehat B = \dfrac{AB}{BC} = \dfrac{5.4}{9} = 0.6$.
\end{solution}

\begin{solution}{exo:g8:cosine:11}
No triângulo \omterm{def:g6:shapes:triangles}{equilátero} de lado $1$, a altura de um vértice
pousa no \emph{ponto médio} do lado oposto (\omterm{def:g6:symmetry:axis}{eixo de
simetria}). \omterm{def:g3:division:half}{Metade} o triângulo é \omterm{def:g6:shapes:triangles}{em ângulo reto} com hipotenusa $1$
(um lado completo), um ângulo de $60^\circ $ na \omterm{def:g8:powers:def}{base} vérticee
lado adjacente $\frac12$ (\omterm{def:g3:division:half}{metade} a \omterm{def:g8:powers:def}{base}). Por isso
\[
\cos 60^\circ = \frac{1/2}{1} = \frac12 .
\]
\end{solution}

\begin{solution}{pb:g8:cosine:1}
\textbf{1.} No triângulo $ ABH $, \omterm{def:g6:shapes:triangles}{em ângulo reto} no $ H $, o ângulo
\omterm{def:g1:addition:def}{soma} (\cref{thm:g7:triangles:sum}) dá
$\widehat B + 90^\circ + \widehat{BAH} = 180^\circ $, então
$\widehat{BAH} = 90^\circ - \widehat B $. No triângulo $ ABC $,
\omterm{def:g6:shapes:triangles}{em ângulo reto} no $ A $:
$\widehat B + \widehat C + 90^\circ = 180^\circ $, então
$\widehat C = 90^\circ - \widehat B $. Por isso
$\widehat{BAH} = \widehat{ACB}$: ambos os pequenos triângulos repetem o
ângulos $\widehat B $, $\widehat C $, $90^\circ $ do original.

\textbf{2.} Em $ ABC $ (\omterm{def:g6:shapes:triangles}{em ângulo reto} no $ A $), o lado adjacente a
$\widehat B $ é $ AB $ e a hipotenusa é $ BC $:
$\cos\widehat B = \frac{AB}{BC}$. Em $ ABH $ (\omterm{def:g6:shapes:triangles}{em ângulo reto} no $ H $),
o lado adjacente a $\widehat B $ é $ BH $ e a hipotenusa é
$ AB $: $\cos\widehat B = \frac{BH}{AB}$. A proporção depende apenas de
o ângulo (\cref{def:g8:cosine:def}), então
\[
\frac{AB}{BC} = \frac{BH}{AB} ;
\]
multiplicando ambos os lados por $ BC \times AB $
(\cref{thm:g8:equations:balance}):
$ AB^2 = BH \times BC $.

\textbf{3.} Os triângulos $ ABC $ (\omterm{def:g6:shapes:triangles}{em ângulo reto} no $ A $) e $ ACH $
(\omterm{def:g6:shapes:triangles}{em ângulo reto} no $ H $) compartilhe o ângulo $\widehat C $. Lado adjacente
sobre a hipotenusa em cada um:
\[
\cos\widehat C = \frac{AC}{CB} = \frac{CH}{AC}
\qquad\Longrightarrow\qquad
AC^2 = CH \times CB .
\]

\textbf{4.} $ AB^2 = BH \times BC $ lê $9 = BH \times 5$, então
$ BH = 1.8$; e $16 = CH \times 5$ dá $ CH = 3.2$. Verificar:
$ BH + HC = 1.8 + 3.2 = 5 = BC $ --- como deveria, já que $ H $ mente
a hipotenusa entre $ B $ e $ C $.

\textbf{5.} Dividindo cada relação por $ BC $:
\[
BH = \frac{AB^2}{BC},
\qquad
CH = \frac{AC^2}{BC} .
\]
Então $ BH $ e $ CH $ são proporcional para $ AB^2$ e $ AC^2$: o pé
da altitude divide a hipotenusa na proporção dos quadrados
das pernas ($1.8 : 3.2 = 9 : 16$ na questão 4).

\textbf{6.} Somando as duas relações e fatorando $ BC $
(distributividade):
\[
AB^2 + AC^2 = BH \times BC + CH \times BC
= (BH + CH) \times BC = BC \times BC = BC^2 .
\]
Isso é \cref{thm:g8:pythagoras:direct}, obtido do cosseno
sozinho --- uma prova genuinamente diferente dos quatro triângulos \omterm{def:g6:measure:area}{área}
quebra-cabeça de \cref{exo:g8:pythagoras:11}.

\textbf{7.} Pitágoras em $ ABH $ (\omterm{def:g6:shapes:triangles}{em ângulo reto} no $ H $, hipotenusa
$[AB]$): $ AH^2 = AB^2 - BH^2$. Substituindo $ AB^2$ por
$ BH \times BC $ (questão 2) e fatorando $ BH $:
\[
AH^2 = BH \times BC - BH^2
= BH \times (BC - BH)
= BH \times HC .
\]

\textbf{8.} Tendo as pernas como \omterm{def:g8:powers:def}{base} e altura, o \omterm{def:g6:measure:area}{área} de $ ABC $
é $\frac{AB \times AC}{2}$; com a hipotenusa $[BC]$ como \omterm{def:g8:powers:def}{base},
a altura é exatamente $ AH $, então o \omterm{def:g6:measure:area}{área} também é
$\frac{BC \times AH}{2}$ (\cref{thm:g7:areas:triangle}). Equacionando
e duplicando: $ AB \times AC = BC \times AH $.

\textbf{9.} $ AH = \frac{AB \times AC}{BC} = \frac{3 \times 4}{5}
= 2.4$. Verificação da questão 7:
$ AH^2 = 2.4^2 = 5.76$ e
$ BH \times HC = 1.8 \times 3.2 = 5.76$. Igual.

\textbf{10.} $ AH^2 = BH \times HC = 2 \times 8 = 16$, então
$ AH = \sqrt{16} = 4$ cm.

\textbf{11.} O ponto $ A $ encontra-se no círculo de \omterm{def:g4:geometry:circle}{diâmetro} $[BC]$
(e difere de $ B $ e $ C $), então pelo ponto 2 de
\cref{thm:g8:cosine:circle} o triângulo $ ABC $ é \omterm{def:g6:shapes:triangles}{em ângulo reto} no
$ A $. Por construção $(AH) \perp (BC)$ com $ H $ entre $ B $ e
$ C $: o \omterm{def:g6:lines:objects}{segmento} $[AH]$ é precisamente a altitude da questão 7,
e $ AH^2 = BH \times HC = p \times q $.

\textbf{12.} $ AH^2 = 1 \times 2 = 2$: a figura constrói, com
régua e compasso sozinhos, um \omterm{def:g6:lines:objects}{segmento} cujo quadrado é $2$ --- o
diagonal do quadrado unitário, $\sqrt2 \approx 1.414$, de
\cref{ex:g8:pythagoras:diagonal}.

\textbf{13.} Receita: coloque de ponta a ponta dois \omterm{def:g6:lines:objects}{segmentos} de comprimentos
$ p = 1$ e $ q = n $; desenhe o semicírculo cujo \omterm{def:g4:geometry:circle}{diâmetro} é o
inteiro \omterm{def:g6:lines:objects}{segmento} (comprimento $ n + 1$); levantar o \omterm{def:g6:lines:perp}{perpendicular} no
ponto de junção; encontra o semicírculo em um ponto cujo
a distância até o cruzamento é $\sqrt{1 \times n} = \sqrt n $. Para
$\sqrt5$: pegar $ p = 1$ e $ q = 5$ (um semicírculo de \omterm{def:g4:geometry:circle}{diâmetro}
$6$).

\textbf{14.} $ A $ está no círculo do centro $ O $, então
$ OA = \frac{p+q}{2}$, o \omterm{def:g3:shapes:shapes}{raio}. Se $ H \neq O $, o triângulo
$ AHO $ é \omterm{def:g6:shapes:triangles}{em ângulo reto} no $ H $ com hipotenusa $[OA]$, e uma perna é
mais curto que a hipotenusa: $ AH < OA $. Se $ H = O $, então
$ AH = OA $ exatamente. Em todos os casos $ AH \leq \frac{p+q}{2}$, com
igualdade precisamente quando $ H $ é o centro --- isto é, quando
$ p = q $.

\textbf{15.} Quadratura $ AH \leq \frac{p+q}{2}$ (ambos os lados
positivo) e usando $ AH^2 = pq $:
\[
p \times q \leq \left(\frac{p+q}{2}\right)^2 ,
\]
com igualdade exatamente para $ p = q $. Reprovação algébrica, com o
identidades notáveis (\cref{pb:g8:equations:1}):
\[
\left(\frac{p+q}{2}\right)^2 - pq
= \frac{p^2 + 2pq + q^2 - 4pq}{4}
= \frac{p^2 - 2pq + q^2}{4}
= \left(\frac{p-q}{2}\right)^2 ,
\]
um quadrado, portanto nunca negativo --- e \omterm{def:g1:counting:zero}{zero} exatamente quando
$ p = q$. O \omterm{pb:g8:cosine:1}{média geométrica} de dois números nunca excede seu
meia \omterm{def:g1:addition:def}{soma}.
\end{solution}
