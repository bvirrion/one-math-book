\section*{Chapitre \ref{ch:g8:pythagoras} --- Le théorème de Pythagore}

\begin{solution}{exo:g8:pythagoras:1}
$\sqrt{81 + 144} = \sqrt{225} = 15$~; \quad
$\sqrt{25 + 144} = \sqrt{169} = 13$~; \quad
$\sqrt{64 + 225} = \sqrt{289} = 17$.
\end{solution}

\begin{solution}{exo:g8:pythagoras:2}
Côté$^2 = 25^2 - 7^2 = 625 - 49 = 576$, donc le côté mesure
$\sqrt{576} = 24$.
\end{solution}

\begin{solution}{exo:g8:pythagoras:3}
Rectangle~: $\sqrt{12^2 + 9^2} = \sqrt{144 + 81} = \sqrt{225} = 15$
cm.

Carré~: $d = 5\sqrt2 \approx 7.1$ cm
(\cref{ex:g8:pythagoras:diagonal}).
\end{solution}

\begin{solution}{exo:g8:pythagoras:4}
$(20, 21, 29)$~: $29^2 = 841$ et $20^2 + 21^2 = 400 + 441 = 841$~:
rectangle.

$(7, 8, 11)$~: $121 \neq 49 + 64 = 113$~: non rectangle.

$(1.5, 2, 2.5)$~: $6.25 = 2.25 + 4$~: rectangle (c'est
$(3,4,5)$ réduit de \omterm{def:g3:division:half}{moitié}).
\end{solution}

\begin{solution}{exo:g8:pythagoras:5}
Planche $= \sqrt{3^2 + 1.6^2} = \sqrt{9 + 2.56} = \sqrt{11.56} = 3.4$
m.
\end{solution}

\begin{solution}{exo:g8:pythagoras:6}
La hauteur tombe au milieu de la base (\omterm{def:g6:shapes:triangles}{triangle isocèle}), donc chaque
demi-triangle a pour hypoténuse $10$ et un côté de l'\omterm{def:g3:shapes:rightangle}{angle droit}
$6$~: $h = \sqrt{100 - 36} = \sqrt{64} = 8$ cm. \omterm{def:g6:measure:area}{Aire}~:
$\frac{12 \times 8}{2} = 48$ cm$^2$.
\end{solution}

\begin{solution}{exo:g8:pythagoras:7}
Diagonale~: $\sqrt{88.5^2 + 49.8^2} = \sqrt{7832.25 + 2480.04}
= \sqrt{10312.29} \approx 101.5$ cm. En pouces~:
$101.5 \div 2.54 \approx 40$ pouces.
\end{solution}

\begin{solution}{exo:g8:pythagoras:8}
Les deux côtés de l'\omterm{def:g3:shapes:rightangle}{angle droit} mesurent $24$ km et $10$ km~:
$\sqrt{24^2 + 10^2} = \sqrt{576 + 100} = \sqrt{676} = 26$ km.
\end{solution}

\begin{solution}{exo:g8:pythagoras:9}
Distance du pied~: $\sqrt{2.5^2 - 2.4^2} = \sqrt{6.25 - 5.76} =
\sqrt{0.49} = 0.7$ m. Un \omterm{def:g3:division:half}{quart} de l'échelle vaut
$2.5 \div 4 \approx 0.6$ m~: les $0.7$ m trouvés sont proches du
placement conseillé --- acceptable.
\end{solution}

\begin{solution}{exo:g8:pythagoras:10}
Les côtés de l'\omterm{def:g3:shapes:rightangle}{angle droit} mesurent $5 - 1 = 4$ (horizontal) et
$5 - 2 = 3$ (vertical), donc
\[
AB = \sqrt{4^2 + 3^2} = \sqrt{25} = 5 .
\]
\end{solution}

\begin{solution}{exo:g8:pythagoras:11}
\emph{1.} Directement~: $(a + b)^2$. En morceaux~: quatre triangles
d'\omterm{def:g6:measure:area}{aire} $\frac{ab}{2}$ chacun, plus le carré incliné $c^2$~:
\[
(a + b)^2 = 4 \times \frac{ab}{2} + c^2 = 2ab + c^2 .
\]

\emph{2.} En développant le membre de gauche~:
$(a+b)^2 = a^2 + 2ab + b^2$. Donc
\[
a^2 + 2ab + b^2 = 2ab + c^2
\quad\Longrightarrow\quad
a^2 + b^2 = c^2 . \qed
\]
\end{solution}
