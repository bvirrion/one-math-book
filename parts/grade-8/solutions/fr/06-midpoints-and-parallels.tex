\section*{Chapitre \ref{ch:g8:midpoints} --- Milieux et parallèles}

\begin{solution}{exo:g8:midpoints:1}
$M$ et $N$ sont les milieux de deux côtés du triangle $RST$~: par le
théorème des milieux (\cref{thm:g8:midpoints:direct}),
$(MN) \parallel (ST)$ et $MN = \frac{ST}{2} = 3.5$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:2}
$I$ est un milieu et $(IJ) \parallel (BC)$~: par la
\emph{réciproque} (\cref{thm:g8:midpoints:converse}), $J$ est le
milieu de $[AC]$, donc $AJ = \frac{11}{2} = 5.5$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:3}
La mesure donne $IJ = 3.5$ cm $= \frac{BC}{2}$. \omterm{def:g6:measure:perimeter}{Périmètre} de $AIJ$~:
$AI = 4$, $AJ = 3$, $IJ = 3.5$~: total $10.5$ cm --- la \omterm{def:g3:division:half}{moitié} du
\omterm{def:g6:measure:perimeter}{périmètre} de $ABC$ ($8 + 6 + 7 = 21$ cm).
\end{solution}

\begin{solution}{exo:g8:midpoints:4}
Chaque côté de $IJK$ est la \omterm{def:g3:division:half}{moitié} d'un côté de $ABC$~: $5$, $6$ et
$8$ cm. \omterm{def:g6:measure:perimeter}{Périmètre}~: $19$ cm, la \omterm{def:g3:division:half}{moitié} de $10 + 12 + 16 = 38$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:5}
Le triangle des milieux est l'un des quatre triangles égaux de
\cref{prop:g8:midpoints:medial}~: \omterm{def:g6:measure:area}{aire} $\frac{36}{4} = 9$ cm$^2$ ---
et chacun des quatre petits triangles a pour \omterm{def:g6:measure:area}{aire} $9$ cm$^2$.
\end{solution}

\begin{solution}{exo:g8:midpoints:6}
Par le théorème des milieux dans $ABC$~: $(IJ) \parallel (BC)$,
c'est-à-dire $(IJ) \parallel (BK)$, et $IJ = \frac{BC}{2} = BK$
(puisque $K$ est le milieu de $[BC]$). Deux côtés opposés de $IJKB$
sont \omterm{def:g6:lines:perp}{parallèles} et égaux~: c'est un \omterm{def:g7:central:parallelogram}{parallélogramme}
(\cref{thm:g7:central:recognize}, critère 3).
\end{solution}

\begin{solution}{exo:g8:midpoints:7}
La couture joint deux milieux~: $\frac{6.4}{2} = 3.2$ m. Au \omterm{def:g3:division:half}{quart}
depuis le \omterm{def:g5:solids:def}{sommet} du haut, le dessin suggère une couture de
$\frac{6.4}{4} = 1.6$ m, \omterm{def:g6:lines:perp}{parallèle} à la base --- confirmé par Thalès
au \cref{ch:g9:thales}.
\end{solution}

\begin{solution}{exo:g8:midpoints:8}
\emph{1.} $M$ est le milieu de $[BC]$ et $(MN) \parallel (AB)$~: par
la réciproque du théorème des milieux (dans le triangle $ABC$, côté
$[AC]$), $N$ est le milieu de $[AC]$.

\emph{2.} $(MN) \parallel (AB)$ et $(AB) \perp (AC)$ (\omterm{def:g3:shapes:rightangle}{angle droit} en
$A$)~: une droite \omterm{def:g6:lines:perp}{parallèle} à $(AB)$ est aussi \omterm{def:g6:lines:perp}{perpendiculaire} à
$(AC)$ (\cref{prop:g6:lines:facts}). Donc $(MN) \perp (AC)$ --- le
\omterm{def:g6:lines:objects}{segment} $[MN]$ est le morceau de \omterm{def:g6:symmetry:bisector}{médiatrice} montrant à nouveau que
$MA = MC$ (\cref{thm:g8:cosine:circle}).
\end{solution}

\begin{solution}{exo:g8:midpoints:9}
\emph{1.} Dans le triangle $ABC$, $I$ et $J$ sont milieux de $[AB]$
et $[BC]$~: $(IJ) \parallel (AC)$ et $IJ = \frac{AC}{2}$. Dans le
triangle $ACD$, $L$ et $K$ sont milieux de $[DA]$ et $[CD]$~:
$(LK) \parallel (AC)$ et $LK = \frac{AC}{2}$.

\emph{2.} Donc $[IJ]$ et $[LK]$ sont \omterm{def:g6:lines:perp}{parallèles} (tous deux \omterm{def:g6:lines:perp}{parallèles}
à la diagonale $(AC)$) et égaux (tous deux la \omterm{def:g3:division:half}{moitié} de $AC$)~:
$IJKL$ est un \omterm{def:g7:central:parallelogram}{parallélogramme} (\cref{thm:g7:central:recognize},
critère 3) --- pour \emph{tout} \omterm{def:g4:geometry:polygon}{quadrilatère} $ABCD$, comme conjecturé
en \cref{exo:g7:central:11}.
\end{solution}
