\section*{Chapitre \ref{ch:g8:negprod} --- Multiplier des nombres relatifs}

\begin{solution}{exo:g8:negprod:1}
$(-8) \times (+7) = -56$~; \quad
$(-9) \times (-6) = +54$~; \quad
$(+12) \times (-0.5) = -6$~; \quad
$(-1) \times (-1) \times (-1) = -1$ (trois facteurs négatifs~: \omterm{def:g3:numbers:evenodd}{impair}).
\end{solution}

\begin{solution}{exo:g8:negprod:2}
$(-63) \div (+9) = -7$~; \quad
$(-8) \div (-16) = +0.5$~; \quad
$\dfrac{45}{-5} = -9$~; \quad
$\dfrac{-7.2}{-8} = +0.9$.
\end{solution}

\begin{solution}{exo:g8:negprod:3}
Trois facteurs négatifs~: \emph{négatif}. \quad
$(-1)^{10}$~: dix facteurs négatifs, \omterm{def:g3:numbers:evenodd}{pair}~: \emph{positif} (cela
vaut $1$). \quad
$(-2)^7$~: sept facteurs, \omterm{def:g3:numbers:evenodd}{impair}~: \emph{négatif}. \quad
$(-4) \times 0 \times (-6) = 0$~: ni positif ni négatif.
\end{solution}

\begin{solution}{exo:g8:negprod:4}
$(-2)^4 = +16$~; \quad $-2^4 = -16$ (le moins n'est pas élevé au
carré)~; \quad
$(-3)^3 = -27$~; \quad $(-1)^{2026} = +1$ (exposant \omterm{def:g3:numbers:evenodd}{pair}).
\end{solution}

\begin{solution}{exo:g8:negprod:5}
$7 + 2 \times (-6) = 7 + (-12) = -5$.

$(-4) \times (5 - 9) = (-4) \times (-4) = +16$.

$18 \div (-3) - (-2) = -6 + 2 = -4$.
\end{solution}

\begin{solution}{exo:g8:negprod:6}
$\dfrac{(-5) \times (+8)}{-4} = \dfrac{-40}{-4} = +10$.

$\dfrac{(-3) \times (-6)}{(-2) \times (+9)} = \dfrac{+18}{-18} = -1$.
\end{solution}

\begin{solution}{exo:g8:negprod:7}
Pour $x = -3$~: \ $5x = -15$~; \quad $x^2 = (-3)^2 = 9$~; \quad
$-x^2 = -9$~; \quad $2x^2 + 4x = 18 - 12 = 6$~; \quad
$(2x)^2 = (-6)^2 = 36$.
\end{solution}

\begin{solution}{exo:g8:negprod:8}
$(-6) \times (-7) = 42$~; \quad
$10 \div (-4) = -2.5$~; \quad
$(-5) \times \frac15 = -1$, donc le nombre manquant est
$+\frac15 = 0.2$.
\end{solution}

\begin{solution}{exo:g8:negprod:9}
\emph{1. Faux}~: $0.5 \times 0.5 = 0.25$ est plus petit que
$0.5 + 0.5 = 1$ (ou~: $2 \times 3 = 6 > 5$, mais l'affirmation doit
valoir toujours).

\emph{2. Vrai}~: un carré est un \omterm{def:g2:mult:def}{produit} de deux nombres de même
signe, donc positif ou nul.

\emph{3. Faux}~: $(-2) \times (-3) \times 4 = +24$ est positif avec
deux facteurs négatifs.
\end{solution}

\begin{solution}{exo:g8:negprod:10}
Soit $r$ le nombre de bonnes réponses~; alors $20 - r$ sont fausses,
et
\[
5r - 3(20 - r) = 36
\ \Longrightarrow\
5r - 60 + 3r = 36
\ \Longrightarrow\
8r = 96
\ \Longrightarrow\
r = 12 .
\]
Douze bonnes réponses (et huit fausses~: vérification,
$60 - 24 = 36$).
\end{solution}

\begin{solution}{exo:g8:negprod:11}
\omterm{def:g1:counting:zero}{Zéro} multiplié par quoi que ce soit donne \omterm{def:g1:counting:zero}{zéro}, donc
$(-1) \times 0 = 0$. En écrivant $0 = 1 + (-1)$ et en distribuant~:
\[
0 = (-1) \times \bigl(1 + (-1)\bigr)
= (-1) \times 1 + (-1) \times (-1)
= -1 + (-1) \times (-1).
\]
Le nombre $(-1) \times (-1)$ ajouté à $-1$ donne $0$~: ce doit être
l'opposé de $-1$, c'est-à-dire $+1$. La distributivité ne laisse pas
d'autre choix.
\end{solution}
