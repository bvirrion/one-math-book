\section*{Chapitre \ref{ch:g8:cosine} --- Le triangle rectangle et le cosinus}

\begin{solution}{exo:g8:cosine:1}
Médiane issue de l'\omterm{def:g3:shapes:rightangle}{angle droit}~: la \omterm{def:g3:division:half}{moitié} de l'hypoténuse, $6.5$ cm
(\cref{thm:g8:cosine:circle}). Le cercle circonscrit est centré au
milieu de l'hypoténuse avec un \omterm{def:g3:shapes:shapes}{rayon} de $6.5$ cm.
\end{solution}

\begin{solution}{exo:g8:cosine:2}
L'angle en $A$ est droit ($[BC]$ est un \omterm{def:g4:geometry:circle}{diamètre}~: point 2 de
\cref{thm:g8:cosine:circle}). Les mesures satisfont
$AB^2 + AC^2 \approx 64 = BC^2$, à la précision de mesure près.
\end{solution}

\begin{solution}{exo:g8:cosine:3}
Hypoténuse~: $[EF]$ (opposée à l'\omterm{def:g3:shapes:rightangle}{angle droit} $D$). Adjacent à
$\widehat E$~: $[ED]$. Opposé~: $[DF]$. Donc
$\cos\widehat E = \dfrac{ED}{EF}$.
\end{solution}

\begin{solution}{exo:g8:cosine:4}
\emph{1.} adjacent $= 10 \cos 35^\circ \approx 8.2$.

\emph{2.} hypoténuse $= \dfrac{6}{\cos 50^\circ} \approx
\dfrac{6}{0.643} \approx 9.3$.
\end{solution}

\begin{solution}{exo:g8:cosine:5}
$\cos\theta = \frac{8}{10} = 0.8$, donc
$\theta = \cos^{-1}(0.8) \approx 37^\circ$.
\end{solution}

\begin{solution}{exo:g8:cosine:6}
$\cos\theta$ est un côté de l'\omterm{def:g3:shapes:rightangle}{angle droit} divisé par l'hypoténuse, et
l'hypoténuse est le \emph{plus long} côté d'un \omterm{def:g6:shapes:triangles}{triangle rectangle}~: le
\omterm{def:g3:division:remainder}{quotient} est toujours plus petit que $1$. Une valeur de $1.2$
signifierait un côté plus long que l'hypoténuse --- impossible.
\end{solution}

\begin{solution}{exo:g8:cosine:7}
Portée horizontale $= 25 \cos 12^\circ \approx 25 \times 0.978 \approx
24.5$ m.
\end{solution}

\begin{solution}{exo:g8:cosine:8}
Le cosinus diminue lorsque l'angle croît
(\cref{prop:g8:cosine:bounds})~:
\[
\cos 70^\circ < \cos 45^\circ < \cos 20^\circ .
\]
(Calculatrice~: $0.342 < 0.707 < 0.940$.)
\end{solution}

\begin{solution}{exo:g8:cosine:9}
La distance au sol est adjacente à $\theta$~:
$\cos 68^\circ = \frac{1.2}{L}$, donc
$L = \frac{1.2}{0.375} = 3.2$ m. Hauteur (Pythagore)~:
$\sqrt{3.2^2 - 1.2^2} = \sqrt{10.24 - 1.44} = \sqrt{8.8} \approx 3.0$
m.
\end{solution}

\begin{solution}{exo:g8:cosine:10}
\emph{1.} $A$ est sur un cercle de \omterm{def:g4:geometry:circle}{diamètre} $[BC]$~: \omterm{def:g3:shapes:rightangle}{angle droit} en
$A$ (\cref{thm:g8:cosine:circle}).

\emph{2.} $BC = 9$ cm (deux fois le \omterm{def:g3:shapes:shapes}{rayon}). Pythagore~:
$AC = \sqrt{9^2 - 5.4^2} = \sqrt{81 - 29.16} = \sqrt{51.84} = 7.2$
cm. Et $\cos\widehat B = \dfrac{AB}{BC} = \dfrac{5.4}{9} = 0.6$.
\end{solution}

\begin{solution}{exo:g8:cosine:11}
Dans le \omterm{def:g6:shapes:triangles}{triangle équilatéral} de côté $1$, la hauteur issue d'un
\omterm{def:g5:solids:def}{sommet} tombe au \emph{milieu} du côté opposé (\omterm{def:g6:symmetry:axis}{axe de symétrie}). La
\omterm{def:g3:division:half}{moitié} du triangle est rectangle avec hypoténuse $1$ (un côté
entier), un angle de $60^\circ$ au \omterm{def:g5:solids:def}{sommet} de la base, et côté
adjacent $\frac12$ (la \omterm{def:g3:division:half}{moitié} de la base). D'où
\[
\cos 60^\circ = \frac{1/2}{1} = \frac12 .
\]
\end{solution}
