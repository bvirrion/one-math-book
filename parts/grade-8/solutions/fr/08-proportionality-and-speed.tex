\section*{Chapitre \ref{ch:g8:speed} --- Proportionnalité, vitesse et moyennes}

\begin{solution}{exo:g8:speed:1}
$\frac{150}{2} = 75$ km/h. \quad
$45$ min $= 0.75$ h~: $\frac{27}{0.75} = 36$ km/h. \quad
$\frac{100}{12.5} = 8$ m/s.
\end{solution}

\begin{solution}{exo:g8:speed:2}
$2 \times 85 = 170$ km. \quad
$40$ min $= \frac23$ h~: $90 \times \frac23 = 60$ km. \quad
$\frac{315}{90} = 3.5$ h $= 3$ h $30$ min.
\end{solution}

\begin{solution}{exo:g8:speed:3}
$72 \div 3.6 = 20$ m/s. \quad
$5 \times 3.6 = 18$ km/h. \quad
$108 \div 3.6 = 30$ m/s.
\end{solution}

\begin{solution}{exo:g8:speed:4}
$600 \div 15 = 40$ min. \quad
$2$ h $30 = 150$ min~: $15 \times 150 = 2250$ L.
\end{solution}

\begin{solution}{exo:g8:speed:5}
$\frac{3.30}{1.2} = 2.75$ euros/kg contre
$\frac{2.16}{0.8} = 2.70$ euros/kg~: la seconde offre est (légèrement)
la meilleure affaire.
\end{solution}

\begin{solution}{exo:g8:speed:6}
\[
\bar v = \frac{25 \times 12.4 + 35 \times 10}{60}
= \frac{310 + 350}{60} = \frac{660}{60} = 11 .
\]
\end{solution}

\begin{solution}{exo:g8:speed:7}
$d = 340 \times 6 = 2040$ m $\approx 2$ km.
\end{solution}

\begin{solution}{exo:g8:speed:8}
Distances~: $2 \times 5 = 10$ km, puis $1.5 \times 4 = 6$ km~: total
$16$ km. Temps écoulé~: $2 + 0.5 + 1.5 = 4$ h. \omterm{def:g8:speed:def}{Vitesse moyenne}~:
$\frac{16}{4} = 4$ km/h.
\end{solution}

\begin{solution}{exo:g8:speed:9}
Pondérée~:
$\frac{3 \times 15 + 2 \times 9 + 1 \times 12}{3 + 2 + 1}
= \frac{45 + 18 + 12}{6} = \frac{75}{6} = 12.5$.
Simple~: $\frac{15 + 9 + 12}{3} = 12$. Elles diffèrent car les
coefficients donnent à la note $15$ trois fois le poids de la note
$12$.
\end{solution}

\begin{solution}{exo:g8:speed:10}
\emph{1.} Temps~: $\frac{120}{80} = 1.5$ h et $\frac{120}{120} = 1$
h. \omterm{def:g8:speed:def}{Vitesse moyenne}~: $\frac{240}{2.5} = 96$ km/h.

\emph{2.} La voiture passe \emph{plus de temps} à $80$ km/h ($1.5$ h)
qu'à $120$ km/h ($1$ h)~: la \omterm{def:g8:speed:def}{vitesse} lente pèse plus dans la \omterm{def:g8:speed:average}{moyenne
pondérée} par le temps, la tirant sous le milieu $100$.
\end{solution}

\begin{solution}{exo:g8:speed:11}
L'écart de $18$ km se referme à $5 + 4 = 9$ km/h~: ils se rencontrent
après $\frac{18}{9} = 2$ h, à $12{:}00$. Anna a alors marché
$2 \times 5 = 10$ km~: ils se rencontrent à $10$ km du village
d'Anna (et $8$ km de celui de Boris --- vérification~: $10 + 8 =
18$).
\end{solution}
