\section*{Chapitre \ref{ch:g8:solids} --- Pyramides et cônes}

\begin{solution}{exo:g8:solids:1}
Tétraèdre~: $4$ \omterm{def:g5:solids:def}{faces}, $6$ \omterm{def:g5:solids:def}{arêtes}, $4$ \omterm{def:g5:solids:def}{sommets}
($4 + 4 = 6 + 2$). Base hexagonale~: $7$ \omterm{def:g5:solids:def}{faces}, $12$ \omterm{def:g5:solids:def}{arêtes}, $7$
\omterm{def:g5:solids:def}{sommets} ($7 + 7 = 12 + 2$).
\end{solution}

\begin{solution}{exo:g8:solids:2}
$V = \frac13 \times 36 \times 10 = 120$ cm$^3$.

Base rectangulaire~: $B = 24$ cm$^2$, donc
$V = \frac13 \times 24 \times 7.5 = 60$ cm$^3$.
\end{solution}

\begin{solution}{exo:g8:solids:3}
$V = \frac13 \pi \times 25 \times 9 = 75\pi \approx 236$ cm$^3$.
\end{solution}

\begin{solution}{exo:g8:solids:4}
\omterm{def:g7:areas:prism}{Cylindre}~: $\pi \times 16 \times 12 = 192\pi \approx 603$ cm$^3$.
\omterm{def:g8:solids:def}{Cône}~: $\frac{192\pi}{3} = 64\pi \approx 201$ cm$^3$. Rapport~: le
\omterm{def:g8:solids:def}{cône} est le tiers du \omterm{def:g7:areas:prism}{cylindre}.
\end{solution}

\begin{solution}{exo:g8:solids:5}
$56 = \frac13 \times B \times 8$, donc $B = \frac{56 \times 3}{8} = 21$
cm$^2$.
\end{solution}

\begin{solution}{exo:g8:solids:6}
La surface latérale se déroule en un secteur de disque dont le \omterm{def:g3:shapes:shapes}{rayon}
est la \emph{génératrice} $s$ (la distance du \omterm{def:g5:solids:def}{sommet} au bord --- ce
\omterm{def:g6:lines:objects}{segment} est sur la surface), et non la hauteur $h$.
\end{solution}

\begin{solution}{exo:g8:solids:7}
$V = \frac13 \times 35^2 \times 22 = \frac13 \times 1225 \times 22
= \frac{26\,950}{3} \approx 8\,983$ m$^3$ --- environ $9\,000$ mètres
cubes.
\end{solution}

\begin{solution}{exo:g8:solids:8}
Hauteur~: $h = \sqrt{13^2 - 5^2} = \sqrt{169 - 25} = \sqrt{144} = 12$
cm. \omterm{def:g6:measure:volume}{Volume}~: $V = \frac13 \pi \times 25 \times 12 = 100\pi$ cm$^3$.
\end{solution}

\begin{solution}{exo:g8:solids:9}
Même \omterm{def:g3:shapes:shapes}{rayon}, donc le \omterm{def:g6:measure:volume}{volume} du \omterm{def:g8:solids:def}{cône} est le tiers du \omterm{def:g6:measure:volume}{volume} du \omterm{def:g7:areas:prism}{cylindre}
\emph{pour la même hauteur}~: le liquide atteint $\frac{9}{3} = 3$ cm
dans le \omterm{def:g7:areas:prism}{cylindre}.
\end{solution}

\begin{solution}{exo:g8:solids:10}
Sable $=$ un \omterm{def:g8:solids:def}{cône}~: $V = \frac13 \pi \times 4 \times 4.5 = 6\pi
\approx 18.8$ cm$^3$. Le sablier contient deux tels \omterm{def:g8:solids:def}{cônes}, donc le
sable remplit exactement la \omterm{def:g3:division:half}{moitié} du \omterm{def:g6:measure:volume}{volume} intérieur total.
\end{solution}

\begin{solution}{exo:g8:solids:11}
\emph{1.} Oui~: la formule $V = \frac13 Bh$ vaut pour toute position
du \omterm{def:g5:solids:def}{sommet}, tant que $h$ est la distance \omterm{def:g6:lines:perp}{perpendiculaire} au plan de la
base. $V = \frac13 \times 36 \times 8 = 96$ cm$^3$.

\emph{2.} Diagonale de la base~: $\sqrt{6^2 + 6^2} = 6\sqrt2$ cm. La
plus longue \omterm{def:g5:solids:def}{arête} latérale est l'hypoténuse d'un \omterm{def:g6:shapes:triangles}{triangle rectangle} de
côtés de l'\omterm{def:g3:shapes:rightangle}{angle droit} $6\sqrt2$ (dans le plan de la base) et $8$
(vertical)~:
\[
\sqrt{(6\sqrt2)^2 + 8^2} = \sqrt{72 + 64} = \sqrt{136} \approx 11.7
\text{ cm}.
\]
\end{solution}
