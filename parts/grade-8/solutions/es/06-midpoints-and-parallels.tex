\begin{solution}{exo:g8:midpoints:1}
$M$ y $N$ son los puntos medios de dos lados del triángulo $RST$: por
el teorema del punto medio (\cref{thm:g8:midpoints:direct}),
$(MN) \parallel (ST)$ y $MN = \frac{ST}{2} = 3.5$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:2}
$I$ es un punto medio y $(IJ) \parallel (BC)$: por el \emph{conversar}
(\cref{thm:g8:midpoints:converse}), $J$ es el punto medio de $[AC]$,
entonces $AJ = \frac{11}{2} = 5.5$ cm.
\end{solution}

\end{solution}

\begin{solution}{exo:g8:midpoints:6}
Según el teorema del punto medio en $ABC$: $(IJ) \parallel (BC)$, es decir\
$(IJ) \parallel (BK)$ y $IJ = \frac{BC}{2} = BK$ (ya que $K$ es el
punto medio de $[BC]$). Dos lados opuestos de $IJKB$ son \omterm{def:g6:lines:perp}{paralelo} y
igual: es un \omterm{def:g7:central:parallelogram}{paralelogramo}
(\cref{thm:g7:central:recognize}, criterio 3).
\end{solution}

\begin{solution}{exo:g8:midpoints:8}
\emph{1.} $M$ es el punto medio de $[BC]$ y $(MN) \parallel (AB)$:
por el inverso del teorema del punto medio (en el triángulo $ABC$,
lado $[AC]$), $N$ es el punto medio de $[AC]$.

\emph{2.} $(MN) \parallel (AB)$ y $(AB) \perp (AC)$ (\omterm{def:g3:shapes:rightangle}{ángulo recto}
en $A$): una línea \omterm{def:g6:lines:perp}{paralelo} a $(AB)$ también es \omterm{def:g6:lines:perp}{perpendicular} a $(AC)$
(\cref{prop:g6:lines:facts}). Entonces $(MN) \perp (AC)$ --- el \omterm{def:g6:lines:objects}{segmento}
$[MN]$ es la pieza \omterm{def:g6:symmetry:bisector}{bisectriz perpendicular} que muestra nuevamente que
$MA = MC$ (\cref{thm:g8:cosine:circle}).
\end{solution}

\begin{solution}{exo:g8:midpoints:9}
\emph{1.} En el triángulo $ABC$, $I$ y $J$ son puntos medios de $[AB]$ y
$[BC]$: $(IJ) \parallel (AC)$ y $IJ = \frac{AC}{2}$. en triangulo
$ACD$, $L$ y $K$ son puntos medios de $[DA]$ y $[CD]$:
$(LK) \parallel (AC)$ y $LK = \frac{AC}{2}$.

\emph{2.} Entonces $[IJ]$ y $[LK]$ son \omterm{def:g6:lines:perp}{paralelo} (ambos \omterm{def:g6:lines:perp}{paralelo} al
diagonal $(AC)$) e igual (ambos \omterm{def:g3:division:half}{medio} de $AC$): $IJKL$ es un
\omterm{def:g7:central:parallelogram}{paralelogramo} (\cref{thm:g7:central:recognize}, criterio 3) ---
para \emph{cada} \omterm{def:g4:geometry:polygon}{cuadrilátero} $ABCD$, como se conjetura en
\cref{exo:g7:central:11}.
\end{solution}

\begin{solution}{pb:g8:midpoints:1}
\textbf{1.} Cualquiera que sea el triángulo que se dibuje, aparecen las tres medianas
pasar por un solo punto, situado dentro del triángulo,
visiblemente más cerca del punto medio de cada lado que del opuesto
\omterm{def:g5:solids:def}{vértice}. El resto del problema lo demuestra.

\textbf{2.} Tome la mediana $[AK]$ del triángulo $ABC$: la
los triángulos $ABK$ y $AKC$ tienen bases $BK$ y $KC$ de la misma
longitud ($K$ es el punto medio de $[BC]$) transportada por la misma línea
$(BC)$, y la misma altura: la distancia de $A$ a $(BC)$.
Según la fórmula \omterm{def:g6:measure:area}{área}, cada uno tiene \omterm{def:g6:measure:area}{área}
base $\times$ altura $\div\ 2$, con bases iguales e iguales
alturas: las \omterm{def:g6:measure:area}{áreas} son iguales. Para un triángulo de \omterm{def:g6:measure:area}{área}
$36$ cm$^2$: dos piezas de $18$ cm$^2$ cada una.

\textbf{3.} $I$ y $J$ son los puntos medios de los lados $[AB]$ y
$[AC]$ del triángulo $ABC$, así por
\cref{thm:g8:midpoints:direct}: $(IJ) \parallel (BC)$ y
$IJ = \frac{BC}{2}$.

\textbf{4.} $M$ y $N$ son los puntos medios de los lados $[GB]$ y
$[GC]$ del triángulo $GBC$, entonces por el mismo teorema:
$(MN) \parallel (BC)$ y $MN = \frac{BC}{2}$.

\textbf{5.} Tanto $(IJ)$ como $(MN)$ son \omterm{def:g6:lines:perp}{paralelo} a $(BC)$, por lo tanto
\omterm{def:g6:lines:perp}{paralelo} entre sí; y $IJ = \frac{BC}{2} = MN$. el
\omterm{def:g4:geometry:polygon}{cuadrilátero} $IJNM$ tiene por lo tanto dos lados opuestos, $[IJ]$ y
$[NM]$, \omterm{def:g6:lines:perp}{paralelo} y de la misma longitud: por el criterio 3 de
\cref{thm:g7:central:recognize}, $IJNM$ es un \omterm{def:g7:central:parallelogram}{paralelogramo}.

\textbf{6.} $G$ se encuentra en la mediana $[CI]$ y $N$ es la
punto medio de $[GC]$, un \omterm{def:g6:lines:objects}{segmento} llevado por esa misma línea mediana:
entonces $I$, $N$ (y $G$, $C$) se encuentran en $(CI)$. Asimismo $J$, $M$
y $G$ se encuentran en la línea mediana $(BJ)$. En el \omterm{def:g7:central:parallelogram}{paralelogramo}
$IJNM$, las diagonales son $[IN]$ y $[JM]$; son llevados por
las dos líneas medianas, que se cruzan en $G$ --- entonces las diagonales
cruce en $G$. Por el \emph{definición} de un \omterm{def:g7:central:parallelogram}{paralelogramo}
(criterio 1 de \cref{thm:g7:central:recognize}: las diagonales
tienen el mismo punto medio), $G$ es el punto medio de $[IN]$ \emph{y}
de $[JM]$: $GI = GN$ y $GJ = GM$.

\textbf{7.} $M$ es el punto medio de $[GB]$, por lo que $BM = MG$; y
$MG = GJ$ por la pregunta 6. Por lo tanto $BM = MG = GJ$: la pieza mediana
$[BJ]$ se corta en tres partes iguales, de las cuales $BG$ toma dos:
\[
BG = 2 \times GJ .
\]
Simétricamente, $N$ es el punto medio de $[GC]$ y $GN = GI$, por lo que
$CN = NG = GI$ y $CG = 2 \times GI$: en la mediana $[CI]$ también,
El punto de cruce se encuentra a dos tercios del camino desde \omterm{def:g5:solids:def}{vértice}.

\textbf{8.} El argumento no usó nada especial sobre la pareja.
$[BJ]$, $[CI]$: ejecuta el par $[BJ]$, $[AK]$, muestra sus
El punto de cruce también se encuentra en $[BJ]$ a dos tercios del camino desde
$B$. Pero solo hay un punto \emph{uno} del \omterm{def:g6:lines:objects}{segmento} $[BJ]$ en
distancia $\frac23 BJ$ desde $B$ --- por lo que el punto de cruce de
$[BJ]$ y $[AK]$ es el mismo $G$. Las tres medianas
por lo tanto pasa por $G$: son \emph{concurrente}.

\textbf{9.} $BG = \frac23 \times 9 = 6$ cm y
$GJ = \frac13 \times 9 = 3$cm.

\textbf{10.} Para la pregunta 8 se cumple el mismo cálculo de dos tercios
en la mediana $[AK]$ (ejecute el argumento de las preguntas 4--7 con
el par $[AK]$, $[BJ]$): $AG = 2 \times GK$.

\textbf{11.} En el triángulo $GBC$, el punto $K$ es el punto medio
del lado $[BC]$, por lo que $[GK]$ es una mediana de $GBC$; por pregunta
2, los triángulos $GBK$ y $GKC$ tienen igual \omterm{def:g6:measure:area}{áreas}.

\textbf{12.} La mediana $[AK]$ corta a $ABC$ en $ABK$ y $AKC$ de
igual \omterm{def:g6:measure:area}{áreas} (pregunta 2). Quitando de cada uno su pequeño triángulo.
de la pregunta 11:
\[
\text{area}(ABG) = \text{area}(ABK) - \text{area}(GBK),
\qquad
\text{area}(ACG) = \text{area}(AKC) - \text{area}(GKC),
\]
y el \omterm{def:g6:measure:area}{áreas} eliminado son iguales, por lo que
$\text{area}(ABG) = \text{area}(ACG)$. El mismo cálculo
a lo largo de la mediana $[BJ]$ --- que divide a la mitad $ABC$ en $ABJ$ y
$CBJ$, mientras que $[GJ]$, una mediana del triángulo $GAC$, lo divide por la mitad
en $GAJ$ y $GCJ$ --- da
$\text{area}(ABG) = \text{area}(CBG)$. Entonces los tres triángulos
$ABG$, $BCG$, $CAG$ tienen igual \omterm{def:g6:measure:area}{áreas}, y como juntos
baldosa $ABC$, cada una tiene \omterm{def:g6:measure:area}{área} un tercio del total.

\textbf{13.} En el triángulo $ABG$, el punto $I$ es el punto medio
del lado $[AB]$, por lo que $[GI]$ es una mediana de $ABG$ \emph{from the
\omterm{def:g5:solids:def}{vértice} $G$}: corta a $ABG$ en dos triángulos de igual \omterm{def:g6:measure:area}{áreas}
(pregunta 2), a saber, $AIG$ y $IBG$. Lo mismo sucede en $BCG$
(mediana $[GK]$) y en $CAG$ (mediana $[GJ]$). Cada tercio de
$ABC$ se corta en \omterm{def:g3:division:half}{medio}: las seis rebanadas alrededor de $G$ tienen cada una \omterm{def:g6:measure:area}{área}
$\frac13 \times \frac12 = \frac16$ del triángulo.

\textbf{14.} En la mediana $[AK]$, de $A(0,0)$ a $K(3,3)$: $G$
se encuentra a dos tercios del camino, por lo que sus coordenadas son
$\left(\frac23 \times 3,\ \frac23 \times 3\right) = (2, 2)$. encendido
la mediana $[BJ]$, de $B(6, 0)$ a $J(0, 3)$: dos tercios de la
manera significa sumar dos tercios del desplazamiento ($-6$
horizontalmente, $+3$ verticalmente):
\[
\left(6 + \tfrac23 \times (-6),\ 0 + \tfrac23 \times 3\right)
= (2, 2) .
\]
El mismo punto, como prometía la pregunta 8. y
$\frac{0 + 6 + 0}{3} = 2$, $\frac{0 + 0 + 6}{3} = 2$: las coordenadas del \omterm{pb:g8:midpoints:1}{centro
de gravedad} son las \emph{promedios} de los tres
Coordenadas \omterm{def:g5:solids:def}{vértices}'.

\textbf{15.} En la pregunta 2, se ha colocado un filo de cuchillo a lo largo de una mediana.
la misma cantidad de cartón en cada lado (igual \omterm{def:g6:measure:area}{áreas}), por lo que el
El triángulo se equilibra a lo largo de cada una de las tres líneas medianas. un punto
del equilibrio debe recaer en las tres líneas de equilibrio a la vez, y por

\end{solution}
