\section*{Chapter \ref{ch:g8:cosine} --- The Right Triangle and the Cosine}

\begin{solution}{exo:g8:cosine:1}
Median from the \omterm{def:g3:shapes:rightangle}{right angle}: \omterm{def:g3:division:half}{half} the hypotenuse, $6.5$ cm
(\cref{thm:g8:cosine:circle}). The \omterm{pb:g6:symmetry:1}{circumscribed circle} is centered
at the midpoint of the hypotenuse with \omterm{def:g3:shapes:shapes}{radius} $6.5$ cm.
\end{solution}

\begin{solution}{exo:g8:cosine:2}
The angle at $A$ is right ($[BC]$ is a \omterm{def:g4:geometry:circle}{diameter}: point 2 of
\cref{thm:g8:cosine:circle}). The measures satisfy
$AB^2 + AC^2 \approx 64 = BC^2$, up to measuring precision.
\end{solution}

\begin{solution}{exo:g8:cosine:3}
Hypotenuse: $[EF]$ (opposite the \omterm{def:g3:shapes:rightangle}{right angle} $D$). Adjacent to
$\widehat E$: $[ED]$. Opposite: $[DF]$. So
$\cos\widehat E = \dfrac{ED}{EF}$.
\end{solution}

\begin{solution}{exo:g8:cosine:4}
\emph{1.} adjacent $= 10 \cos 35^\circ \approx 8.2$.

\emph{2.} hypotenuse $= \dfrac{6}{\cos 50^\circ} \approx
\dfrac{6}{0.643} \approx 9.3$.
\end{solution}

\begin{solution}{exo:g8:cosine:5}
$\cos\theta = \frac{8}{10} = 0.8$, so
$\theta = \cos^{-1}(0.8) \approx 37^\circ$.
\end{solution}

\begin{solution}{exo:g8:cosine:6}
$\cos\theta$ is a leg divided by the hypotenuse, and the hypotenuse
is the \emph{longest} side of a \omterm{def:g6:shapes:triangles}{right triangle}: the \omterm{def:g3:division:remainder}{quotient} is
always smaller than $1$. A value of $1.2$ would mean a leg longer
than the hypotenuse --- impossible.
\end{solution}

\begin{solution}{exo:g8:cosine:7}
Horizontal span $= 25 \cos 12^\circ \approx 25 \times 0.978 \approx
24.5$ m.
\end{solution}

\begin{solution}{exo:g8:cosine:8}
The cosine decreases as the angle grows
(\cref{prop:g8:cosine:bounds}):
\[
\cos 70^\circ < \cos 45^\circ < \cos 20^\circ .
\]
(Calculator: $0.342 < 0.707 < 0.940$.)
\end{solution}

\begin{solution}{exo:g8:cosine:9}
The ground distance is adjacent to $\theta$:
$\cos 68^\circ = \frac{1.2}{L}$, so
$L = \frac{1.2}{0.375} = 3.2$ m. Height (Pythagoras):
$\sqrt{3.2^2 - 1.2^2} = \sqrt{10.24 - 1.44} = \sqrt{8.8} \approx 3.0$
m.
\end{solution}

\begin{solution}{exo:g8:cosine:10}
\emph{1.} $A$ is on a circle of \omterm{def:g4:geometry:circle}{diameter} $[BC]$: \omterm{def:g3:shapes:rightangle}{right angle} at $A$
(\cref{thm:g8:cosine:circle}).

\emph{2.} $BC = 9$ cm (twice the \omterm{def:g3:shapes:shapes}{radius}). Pythagoras:
$AC = \sqrt{9^2 - 5.4^2} = \sqrt{81 - 29.16} = \sqrt{51.84} = 7.2$
cm. And $\cos\widehat B = \dfrac{AB}{BC} = \dfrac{5.4}{9} = 0.6$.
\end{solution}

\begin{solution}{exo:g8:cosine:11}
In the \omterm{def:g6:shapes:triangles}{equilateral triangle} of side $1$, the height from one \omterm{def:g5:solids:def}{vertex}
lands on the \emph{midpoint} of the opposite side (\omterm{def:g6:symmetry:axis}{axis of
symmetry}). \omterm{def:g3:division:half}{Half} the triangle is \omterm{def:g6:shapes:triangles}{right-angled} with hypotenuse $1$
(a full side), an angle of $60^\circ$ at the base \omterm{def:g5:solids:def}{vertex}, and
adjacent side $\frac12$ (\omterm{def:g3:division:half}{half} the base). Hence
\[
\cos 60^\circ = \frac{1/2}{1} = \frac12 .
\]
\end{solution}

\begin{solution}{pb:g8:cosine:1}
\textbf{1.} In the triangle $ABH$, \omterm{def:g6:shapes:triangles}{right-angled} at $H$, the angle
\omterm{def:g1:addition:def}{sum} (\cref{thm:g7:triangles:sum}) gives
$\widehat B + 90^\circ + \widehat{BAH} = 180^\circ$, so
$\widehat{BAH} = 90^\circ - \widehat B$. In the triangle $ABC$,
\omterm{def:g6:shapes:triangles}{right-angled} at $A$:
$\widehat B + \widehat C + 90^\circ = 180^\circ$, so
$\widehat C = 90^\circ - \widehat B$. Hence
$\widehat{BAH} = \widehat{ACB}$: both small triangles repeat the
angles $\widehat B$, $\widehat C$, $90^\circ$ of the original.

\textbf{2.} In $ABC$ (\omterm{def:g6:shapes:triangles}{right-angled} at $A$), the side adjacent to
$\widehat B$ is $AB$ and the hypotenuse is $BC$:
$\cos\widehat B = \frac{AB}{BC}$. In $ABH$ (\omterm{def:g6:shapes:triangles}{right-angled} at $H$),
the side adjacent to $\widehat B$ is $BH$ and the hypotenuse is
$AB$: $\cos\widehat B = \frac{BH}{AB}$. The ratio depends only on
the angle (\cref{def:g8:cosine:def}), so
\[
\frac{AB}{BC} = \frac{BH}{AB} ;
\]
multiplying both sides by $BC \times AB$
(\cref{thm:g8:equations:balance}):
$AB^2 = BH \times BC$.

\textbf{3.} The triangles $ABC$ (\omterm{def:g6:shapes:triangles}{right-angled} at $A$) and $ACH$
(\omterm{def:g6:shapes:triangles}{right-angled} at $H$) share the angle $\widehat C$. Adjacent side
over hypotenuse in each:
\[
\cos\widehat C = \frac{AC}{CB} = \frac{CH}{AC}
\qquad\Longrightarrow\qquad
AC^2 = CH \times CB .
\]

\textbf{4.} $AB^2 = BH \times BC$ reads $9 = BH \times 5$, so
$BH = 1.8$; and $16 = CH \times 5$ gives $CH = 3.2$. Check:
$BH + HC = 1.8 + 3.2 = 5 = BC$ --- as it must, since $H$ lies on
the hypotenuse between $B$ and $C$.

\textbf{5.} Dividing each relation by $BC$:
\[
BH = \frac{AB^2}{BC},
\qquad
CH = \frac{AC^2}{BC} .
\]
So $BH$ and $CH$ are \omterm{def:g6:propdata:def}{proportional} to $AB^2$ and $AC^2$: the foot
of the altitude splits the hypotenuse in the ratio of the squares
of the legs ($1.8 : 3.2 = 9 : 16$ in question 4).

\textbf{6.} Adding the two relations and factoring out $BC$
(distributivity):
\[
AB^2 + AC^2 = BH \times BC + CH \times BC
= (BH + CH) \times BC = BC \times BC = BC^2 .
\]
This is \cref{thm:g8:pythagoras:direct}, obtained from the cosine
alone --- a genuinely different proof from the four-triangle \omterm{def:g6:measure:area}{area}
puzzle of \cref{exo:g8:pythagoras:11}.

\textbf{7.} Pythagoras in $ABH$ (\omterm{def:g6:shapes:triangles}{right-angled} at $H$, hypotenuse
$[AB]$): $AH^2 = AB^2 - BH^2$. Replacing $AB^2$ by
$BH \times BC$ (question 2) and factoring out $BH$:
\[
AH^2 = BH \times BC - BH^2
= BH \times (BC - BH)
= BH \times HC .
\]

\textbf{8.} With the legs as base and height, the \omterm{def:g6:measure:area}{area} of $ABC$
is $\frac{AB \times AC}{2}$; with the hypotenuse $[BC]$ as base,
the height is exactly $AH$, so the \omterm{def:g6:measure:area}{area} is also
$\frac{BC \times AH}{2}$ (\cref{thm:g7:areas:triangle}). Equating
and doubling: $AB \times AC = BC \times AH$.

\textbf{9.} $AH = \frac{AB \times AC}{BC} = \frac{3 \times 4}{5}
= 2.4$. Verification of question 7:
$AH^2 = 2.4^2 = 5.76$ and
$BH \times HC = 1.8 \times 3.2 = 5.76$. Equal.

\textbf{10.} $AH^2 = BH \times HC = 2 \times 8 = 16$, so
$AH = \sqrt{16} = 4$ cm.

\textbf{11.} The point $A$ lies on the circle of \omterm{def:g4:geometry:circle}{diameter} $[BC]$
(and differs from $B$ and $C$), so by point 2 of
\cref{thm:g8:cosine:circle} the triangle $ABC$ is \omterm{def:g6:shapes:triangles}{right-angled} at
$A$. By construction $(AH) \perp (BC)$ with $H$ between $B$ and
$C$: the \omterm{def:g6:lines:objects}{segment} $[AH]$ is precisely the altitude of question 7,
and $AH^2 = BH \times HC = p \times q$.

\textbf{12.} $AH^2 = 1 \times 2 = 2$: the figure constructs, with
ruler and compass alone, a \omterm{def:g6:lines:objects}{segment} whose square is $2$ --- the
diagonal of the unit square, $\sqrt2 \approx 1.414$, of
\cref{ex:g8:pythagoras:diagonal}.

\textbf{13.} Recipe: lay end to end two \omterm{def:g6:lines:objects}{segments} of lengths
$p = 1$ and $q = n$; draw the half-circle whose \omterm{def:g4:geometry:circle}{diameter} is the
whole \omterm{def:g6:lines:objects}{segment} (length $n + 1$); raise the \omterm{def:g6:lines:perp}{perpendicular} at the
junction point; it meets the half-circle at a point whose
distance to the junction is $\sqrt{1 \times n} = \sqrt n$. For
$\sqrt5$: take $p = 1$ and $q = 5$ (a half-circle of \omterm{def:g4:geometry:circle}{diameter}
$6$).

\textbf{14.} $A$ is on the circle of center $O$, so
$OA = \frac{p+q}{2}$, the \omterm{def:g3:shapes:shapes}{radius}. If $H \neq O$, the triangle
$AHO$ is \omterm{def:g6:shapes:triangles}{right-angled} at $H$ with hypotenuse $[OA]$, and a leg is
shorter than the hypotenuse: $AH < OA$. If $H = O$, then
$AH = OA$ exactly. In all cases $AH \leq \frac{p+q}{2}$, with
equality precisely when $H$ is the center --- that is, when
$p = q$.

\textbf{15.} Squaring $AH \leq \frac{p+q}{2}$ (both sides
positive) and using $AH^2 = pq$:
\[
p \times q \leq \left(\frac{p+q}{2}\right)^2 ,
\]
with equality exactly for $p = q$. Algebraic re-proof, with the
remarkable identities (\cref{pb:g8:equations:1}):
\[
\left(\frac{p+q}{2}\right)^2 - pq
= \frac{p^2 + 2pq + q^2 - 4pq}{4}
= \frac{p^2 - 2pq + q^2}{4}
= \left(\frac{p-q}{2}\right)^2 ,
\]
a square, hence never negative --- and \omterm{def:g1:counting:zero}{zero} exactly when
$p = q$. The \omterm{pb:g8:cosine:1}{geometric mean} of two numbers never exceeds their
\omterm{def:g1:addition:def}{half-sum}.
\end{solution}
