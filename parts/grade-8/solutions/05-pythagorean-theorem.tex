\begin{solution}{exo:g8:pythagoras:11}
\emph{1.} Directly: $(a + b)^2$. As pieces: four triangles of \omterm{def:g6:measure:area}{area}
$\frac{ab}{2}$ each, plus the tilted square $c^2$:
\[
(a + b)^2 = 4 \times \frac{ab}{2} + c^2 = 2ab + c^2 .
\]

\emph{2.} Expanding the left side:
$(a+b)^2 = a^2 + 2ab + b^2$. So
\[
a^2 + 2ab + b^2 = 2ab + c^2
\quad\Longrightarrow\quad
a^2 + b^2 = c^2 . \qed
\]
\end{solution}

\begin{solution}{pb:g8:pythagoras:1}
\textbf{1.} $5^2 + 12^2 = 25 + 144 = 169 = 13^2$; \quad
$8^2 + 15^2 = 64 + 225 = 289 = 17^2$; \quad
$20^2 + 21^2 = 400 + 441 = 841 = 29^2$. All three are triples,
and by \cref{thm:g8:pythagoras:converse} a triangle with these
side lengths is \omterm{def:g6:shapes:triangles}{right-angled}, the \omterm{def:g3:shapes:rightangle}{right angle} facing the longest
side.

\textbf{2.} The rope forms a triangle with sides $3$, $4$, $5$
(in \omterm{def:g6:lines:objects}{segments}), and $3^2 + 4^2 = 25 = 5^2$: by the converse
(\cref{thm:g8:pythagoras:converse}), the angle between the sides
of $3$ and $4$ \omterm{def:g6:lines:objects}{segments} is exactly right --- no protractor
needed, only knots.

\textbf{3.} If $a^2 + b^2 = c^2$, then
\[
(ka)^2 + (kb)^2 = k^2 a^2 + k^2 b^2
= k^2\left(a^2 + b^2\right) = k^2 c^2 = (kc)^2 .
\]
From $(3, 4, 5)$: for instance $(6, 8, 10)$, $(9, 12, 15)$ and
$(30, 40, 50)$.

\textbf{4.} A \omterm{def:g5:division:multiple}{multiple} $(3k, 4k, 5k)$ containing $5$ as its
smallest number would need $3k = 5$, and no whole number $k$
satisfies that ($k = 1$ gives $3$, $k = 2$ gives $6$). So
$(5, 12, 13)$ is not a \omterm{def:g5:division:multiple}{multiple} of $(3, 4, 5)$.

\textbf{5.} Question 3 shows scaling produces only the triples
$(3k, 4k, 5k)$; question 4 exhibits a genuine triple that is not
of this form. So the family of all \omterm{pb:g8:pythagoras:1}{Pythagorean triples} is
strictly larger than the \omterm{def:g5:division:multiple}{multiples} of $(3, 4, 5)$: producing them
requires something new.

\textbf{6.} Write $A = m^2$ and $B = n^2$. By the remarkable
identities (\cref{pb:g8:equations:1}):
\[
a^2 = (A - B)^2 = A^2 - 2AB + B^2,
\qquad
b^2 = (2mn)^2 = 4m^2n^2 = 4AB .
\]
Adding:
\[
a^2 + b^2 = A^2 - 2AB + B^2 + 4AB
= A^2 + 2AB + B^2 = (A + B)^2
= \left(m^2 + n^2\right)^2 = c^2 .
\]

\textbf{7.}
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccc}
$(m, n)$ & $m^2 - n^2$ & $2mn$ & $m^2 + n^2$ \\
\hline
$(2, 1)$ & $3$ & $4$ & $5$ \\
$(3, 1)$ & $8$ & $6$ & $10$ \\
$(3, 2)$ & $5$ & $12$ & $13$ \\
$(4, 1)$ & $15$ & $8$ & $17$ \\
$(4, 3)$ & $7$ & $24$ & $25$ \\
\end{tabular}
\end{center}

\textbf{8.} $(8, 6, 10)$ is the double of $(4, 3, 5)$ --- that
is, of the builders' $(3, 4, 5)$. The machine sometimes
rediscovers an old triple in disguise.

\textbf{9.} We need $m^2 - n^2 = 21$ and $2mn = 20$, so
$mn = 10$: trying $m = 5$, $n = 2$ gives $25 - 4 = 21$ and
$2 \times 10 = 20$, with hypotenuse $25 + 4 = 29$. The machine
outputs $(21, 20, 29)$ for $(m, n) = (5, 2)$.

\textbf{10.} $m = 8$, $n = 6$ gives $m^2 + n^2 = 64 + 36 = 100$.
The triple is
\[
\left(64 - 36,\ 2 \times 48,\ 100\right) = (28,\ 96,\ 100),
\]
and indeed $28^2 + 96^2 = 784 + 9216 = 10\,000 = 100^2$.

\textbf{11.} With $m = n + 1$:
\[
m^2 - n^2 = (n + 1)^2 - n^2 = 2n + 1
\]
(\omterm{ex:g1:subtraction:difference}{difference} of consecutive squares, \cref{pb:g8:equations:1});
$2mn = 2n(n + 1) = 2n^2 + 2n$; and
$m^2 + n^2 = n^2 + n^2 + 2n + 1 = 2n^2 + 2n + 1$. The hypotenuse
exceeds the \omterm{def:g3:numbers:evenodd}{even} leg by exactly $1$. For $n = 1, 2, 3, 4$:
\[
(3, 4, 5), \quad (5, 12, 13), \quad (7, 24, 25), \quad
(9, 40, 41) .
\]

\textbf{12.} Every \omterm{def:g3:numbers:evenodd}{odd number} at least $3$ is $2n + 1$ for some
$n \geq 1$, and question 11 provides a triple whose \omterm{def:g3:numbers:evenodd}{odd} leg is
exactly $2n + 1$. For $11 = 2 \times 5 + 1$, take $n = 5$:
\[
(11,\ 60,\ 61),
\qquad
11^2 + 60^2 = 121 + 3600 = 3721 = 61^2 .
\]

\textbf{13.} With $n = 1$ the machine gives
$\left(m^2 - 1,\ 2m,\ m^2 + 1\right)$, whose \omterm{def:g3:numbers:evenodd}{even} leg $2m$ can be
any \omterm{def:g3:numbers:evenodd}{even number} from $4$ upward. For $14 = 2 \times 7$, take
$m = 7$: the triple $(48, 14, 50)$. The same triple appears by
doubling $(7, 24, 25)$ from the table of question 7:
$(14, 48, 50)$ --- the same three numbers.

\textbf{14.} An \omterm{def:g3:numbers:evenodd}{even number} is $2k$, and $(2k)^2 = 4k^2$ is \omterm{def:g3:numbers:evenodd}{even}
(a \omterm{def:g5:division:multiple}{multiple} of $4$, \omterm{def:g3:numbers:evenodd}{even}). An \omterm{def:g3:numbers:evenodd}{odd number} is $2k + 1$, and the
first remarkable identity gives
\[
(2k + 1)^2 = 4k^2 + 4k + 1 = 2\left(2k^2 + 2k\right) + 1 ,
\]
an \omterm{def:g3:numbers:evenodd}{even number} plus $1$: \omterm{def:g3:numbers:evenodd}{odd}.

\textbf{15.} Suppose all of $a$, $b$, $c$ were \omterm{def:g3:numbers:evenodd}{odd}. By question
14, $a^2$ and $b^2$ are \omterm{def:g3:numbers:evenodd}{odd}, so $a^2 + b^2$ is a \omterm{def:g1:addition:def}{sum} of two \omterm{def:g3:numbers:evenodd}{odd
numbers}: \emph{\omterm{def:g3:numbers:evenodd}{even}}. But $c^2$ would be \emph{\omterm{def:g3:numbers:evenodd}{odd}}. The equality
$a^2 + b^2 = c^2$ is then impossible. So in every \omterm{pb:g8:pythagoras:1}{Pythagorean
triple} at least one of the three numbers is \omterm{def:g3:numbers:evenodd}{even} --- as every
line of the table of question 7 confirms.
\end{solution}
