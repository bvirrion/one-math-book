\section*{Chapter \ref{ch:g8:midpoints} --- Midpoints and Parallels}

\begin{solution}{exo:g8:midpoints:1}
$M$ and $N$ are the midpoints of two sides of the triangle $RST$: by
the midpoint theorem (\cref{thm:g8:midpoints:direct}),
$(MN) \parallel (ST)$ and $MN = \frac{ST}{2} = 3.5$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:2}
$I$ is a midpoint and $(IJ) \parallel (BC)$: by the \emph{converse}
(\cref{thm:g8:midpoints:converse}), $J$ is the midpoint of $[AC]$,
so $AJ = \frac{11}{2} = 5.5$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:3}
The measurement gives $IJ = 3.5$ cm $= \frac{BC}{2}$. \omterm{def:g6:measure:perimeter}{Perimeter} of
$AIJ$: $AI = 4$, $AJ = 3$, $IJ = 3.5$: total $10.5$ cm --- \omterm{def:g3:division:half}{half} the
\omterm{def:g6:measure:perimeter}{perimeter} of $ABC$ ($8 + 6 + 7 = 21$ cm).
\end{solution}

\begin{solution}{exo:g8:midpoints:4}
Each side of $IJK$ is \omterm{def:g3:division:half}{half} a side of $ABC$: $5$, $6$ and $8$ cm.
\omterm{def:g6:measure:perimeter}{Perimeter}: $19$ cm, \omterm{def:g3:division:half}{half} of $10 + 12 + 16 = 38$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:5}
The midpoint triangle is one of the four equal triangles of
\cref{prop:g8:midpoints:medial}: \omterm{def:g6:measure:area}{area} $\frac{36}{4} = 9$ cm$^2$ ---
and each of the four small triangles has \omterm{def:g6:measure:area}{area} $9$ cm$^2$.
\end{solution}

\begin{solution}{exo:g8:midpoints:6}
By the midpoint theorem in $ABC$: $(IJ) \parallel (BC)$, i.e.\
$(IJ) \parallel (BK)$, and $IJ = \frac{BC}{2} = BK$ (as $K$ is the
midpoint of $[BC]$). Two opposite sides of $IJKB$ are \omterm{def:g6:lines:perp}{parallel} and
equal: it is a \omterm{def:g7:central:parallelogram}{parallelogram}
(\cref{thm:g7:central:recognize}, criterion 3).
\end{solution}

\begin{solution}{exo:g8:midpoints:7}
The seam joins two midpoints: $\frac{6.4}{2} = 3.2$ m. At one
\omterm{def:g3:division:half}{quarter} from the top \omterm{def:g5:solids:def}{vertex}, the drawing suggests a seam of
$\frac{6.4}{4} = 1.6$ m, \omterm{def:g6:lines:perp}{parallel} to the base --- confirmed by Thales
in \cref{ch:g9:thales}.
\end{solution}

\begin{solution}{exo:g8:midpoints:8}
\emph{1.} $M$ is the midpoint of $[BC]$ and $(MN) \parallel (AB)$:
by the converse of the midpoint theorem (in the triangle $ABC$,
side $[AC]$), $N$ is the midpoint of $[AC]$.

\emph{2.} $(MN) \parallel (AB)$ and $(AB) \perp (AC)$ (\omterm{def:g3:shapes:rightangle}{right angle}
at $A$): a line \omterm{def:g6:lines:perp}{parallel} to $(AB)$ is also \omterm{def:g6:lines:perp}{perpendicular} to $(AC)$
(\cref{prop:g6:lines:facts}). So $(MN) \perp (AC)$ --- the \omterm{def:g6:lines:objects}{segment}
$[MN]$ is the \omterm{def:g6:symmetry:bisector}{perpendicular bisector} piece showing again that
$MA = MC$ (\cref{thm:g8:cosine:circle}).
\end{solution}

\begin{solution}{exo:g8:midpoints:9}
\emph{1.} In triangle $ABC$, $I$ and $J$ are midpoints of $[AB]$ and
$[BC]$: $(IJ) \parallel (AC)$ and $IJ = \frac{AC}{2}$. In triangle
$ACD$, $L$ and $K$ are midpoints of $[DA]$ and $[CD]$:
$(LK) \parallel (AC)$ and $LK = \frac{AC}{2}$.

\emph{2.} So $[IJ]$ and $[LK]$ are \omterm{def:g6:lines:perp}{parallel} (both \omterm{def:g6:lines:perp}{parallel} to the
diagonal $(AC)$) and equal (both \omterm{def:g3:division:half}{half} of $AC$): $IJKL$ is a
\omterm{def:g7:central:parallelogram}{parallelogram} (\cref{thm:g7:central:recognize}, criterion 3) ---
for \emph{every} \omterm{def:g4:geometry:polygon}{quadrilateral} $ABCD$, as conjectured in
\cref{exo:g7:central:11}.
\end{solution}
