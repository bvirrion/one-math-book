\section*{Chapter \ref{ch:g8:midpoints} --- Midpoints and Parallels}

\begin{solution}{exo:g8:midpoints:1}
$M$ and $N$ are the midpoints of two sides of the triangle $RST$: by
the midpoint theorem (\cref{thm:g8:midpoints:direct}),
$(MN) \parallel (ST)$ and $MN = \frac{ST}{2} = 3.5$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:2}
$I$ is a midpoint and $(IJ) \parallel (BC)$: by the \emph{converse}
(\cref{thm:g8:midpoints:converse}), $J$ is the midpoint of $[AC]$,
so $AJ = \frac{11}{2} = 5.5$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:3}
The measurement gives $IJ = 3.5$ cm $= \frac{BC}{2}$. \omterm{def:g6:measure:perimeter}{Perimeter} of
$AIJ$: $AI = 4$, $AJ = 3$, $IJ = 3.5$: total $10.5$ cm --- \omterm{def:g3:division:half}{half} the
\omterm{def:g6:measure:perimeter}{perimeter} of $ABC$ ($8 + 6 + 7 = 21$ cm).
\end{solution}

\begin{solution}{exo:g8:midpoints:4}
Each side of $IJK$ is \omterm{def:g3:division:half}{half} a side of $ABC$: $5$, $6$ and $8$ cm.
\omterm{def:g6:measure:perimeter}{Perimeter}: $19$ cm, \omterm{def:g3:division:half}{half} of $10 + 12 + 16 = 38$ cm.
\end{solution}

\begin{solution}{exo:g8:midpoints:5}
The midpoint triangle is one of the four equal triangles of
\cref{prop:g8:midpoints:medial}: \omterm{def:g6:measure:area}{area} $\frac{36}{4} = 9$ cm$^2$ ---
and each of the four small triangles has \omterm{def:g6:measure:area}{area} $9$ cm$^2$.
\end{solution}

\begin{solution}{exo:g8:midpoints:6}
By the midpoint theorem in $ABC$: $(IJ) \parallel (BC)$, i.e.\
$(IJ) \parallel (BK)$, and $IJ = \frac{BC}{2} = BK$ (as $K$ is the
midpoint of $[BC]$). Two opposite sides of $IJKB$ are \omterm{def:g6:lines:perp}{parallel} and
equal: it is a \omterm{def:g7:central:parallelogram}{parallelogram}
(\cref{thm:g7:central:recognize}, criterion 3).
\end{solution}

\begin{solution}{exo:g8:midpoints:7}
The seam joins two midpoints: $\frac{6.4}{2} = 3.2$ m. At one
\omterm{def:g3:division:half}{quarter} from the top \omterm{def:g5:solids:def}{vertex}, the drawing suggests a seam of
$\frac{6.4}{4} = 1.6$ m, \omterm{def:g6:lines:perp}{parallel} to the base --- confirmed by Thales
in \cref{ch:g9:thales}.
\end{solution}

\begin{solution}{exo:g8:midpoints:8}
\emph{1.} $M$ is the midpoint of $[BC]$ and $(MN) \parallel (AB)$:
by the converse of the midpoint theorem (in the triangle $ABC$,
side $[AC]$), $N$ is the midpoint of $[AC]$.

\emph{2.} $(MN) \parallel (AB)$ and $(AB) \perp (AC)$ (\omterm{def:g3:shapes:rightangle}{right angle}
at $A$): a line \omterm{def:g6:lines:perp}{parallel} to $(AB)$ is also \omterm{def:g6:lines:perp}{perpendicular} to $(AC)$
(\cref{prop:g6:lines:facts}). So $(MN) \perp (AC)$ --- the \omterm{def:g6:lines:objects}{segment}
$[MN]$ is the \omterm{def:g6:symmetry:bisector}{perpendicular bisector} piece showing again that
$MA = MC$ (\cref{thm:g8:cosine:circle}).
\end{solution}

\begin{solution}{exo:g8:midpoints:9}
\emph{1.} In triangle $ABC$, $I$ and $J$ are midpoints of $[AB]$ and
$[BC]$: $(IJ) \parallel (AC)$ and $IJ = \frac{AC}{2}$. In triangle
$ACD$, $L$ and $K$ are midpoints of $[DA]$ and $[CD]$:
$(LK) \parallel (AC)$ and $LK = \frac{AC}{2}$.

\emph{2.} So $[IJ]$ and $[LK]$ are \omterm{def:g6:lines:perp}{parallel} (both \omterm{def:g6:lines:perp}{parallel} to the
diagonal $(AC)$) and equal (both \omterm{def:g3:division:half}{half} of $AC$): $IJKL$ is a
\omterm{def:g7:central:parallelogram}{parallelogram} (\cref{thm:g7:central:recognize}, criterion 3) ---
for \emph{every} \omterm{def:g4:geometry:polygon}{quadrilateral} $ABCD$, as conjectured in
\cref{exo:g7:central:11}.
\end{solution}

\begin{solution}{pb:g8:midpoints:1}
\textbf{1.} Whatever triangle is drawn, the three medians appear
to pass through a single point, situated inside the triangle,
visibly closer to each side's midpoint than to the opposite
\omterm{def:g5:solids:def}{vertex}. The rest of the problem proves it.

\textbf{2.} Take the median $[AK]$ of the triangle $ABC$: the
triangles $ABK$ and $AKC$ have bases $BK$ and $KC$ of the same
length ($K$ is the midpoint of $[BC]$) carried by the same line
$(BC)$, and the same height: the distance from $A$ to $(BC)$.
By the \omterm{def:g6:measure:area}{area} formula, each has \omterm{def:g6:measure:area}{area}
base $\times$ height $\div\ 2$, with equal bases and equal
heights: the \omterm{def:g6:measure:area}{areas} are equal. For a triangle of \omterm{def:g6:measure:area}{area}
$36$ cm$^2$: two pieces of $18$ cm$^2$ each.

\textbf{3.} $I$ and $J$ are the midpoints of the sides $[AB]$ and
$[AC]$ of the triangle $ABC$, so by
\cref{thm:g8:midpoints:direct}: $(IJ) \parallel (BC)$ and
$IJ = \frac{BC}{2}$.

\textbf{4.} $M$ and $N$ are the midpoints of the sides $[GB]$ and
$[GC]$ of the triangle $GBC$, so by the same theorem:
$(MN) \parallel (BC)$ and $MN = \frac{BC}{2}$.

\textbf{5.} Both $(IJ)$ and $(MN)$ are \omterm{def:g6:lines:perp}{parallel} to $(BC)$, hence
\omterm{def:g6:lines:perp}{parallel} to each other; and $IJ = \frac{BC}{2} = MN$. The
\omterm{def:g4:geometry:polygon}{quadrilateral} $IJNM$ thus has two opposite sides, $[IJ]$ and
$[NM]$, \omterm{def:g6:lines:perp}{parallel} and of the same length: by criterion 3 of
\cref{thm:g7:central:recognize}, $IJNM$ is a \omterm{def:g7:central:parallelogram}{parallelogram}.

\textbf{6.} $G$ lies on the median $[CI]$, and $N$ is the
midpoint of $[GC]$, a \omterm{def:g6:lines:objects}{segment} carried by that same median line:
so $I$, $N$ (and $G$, $C$) all lie on $(CI)$. Likewise $J$, $M$
and $G$ lie on the median line $(BJ)$. In the \omterm{def:g7:central:parallelogram}{parallelogram}
$IJNM$, the diagonals are $[IN]$ and $[JM]$; they are carried by
the two median lines, which cross at $G$ --- so the diagonals
cross at $G$. By the \emph{definition} of a \omterm{def:g7:central:parallelogram}{parallelogram}
(criterion 1 of \cref{thm:g7:central:recognize}: the diagonals
have the same midpoint), $G$ is the midpoint of $[IN]$ \emph{and}
of $[JM]$: $GI = GN$ and $GJ = GM$.

\textbf{7.} $M$ is the midpoint of $[GB]$, so $BM = MG$; and
$MG = GJ$ by question 6. Hence $BM = MG = GJ$: the median piece
$[BJ]$ is cut into three equal parts, of which $BG$ takes two:
\[
BG = 2 \times GJ .
\]
Symmetrically, $N$ is the midpoint of $[GC]$ and $GN = GI$, so
$CN = NG = GI$ and $CG = 2 \times GI$: on the median $[CI]$ too,
the crossing point sits two thirds of the way from the \omterm{def:g5:solids:def}{vertex}.

\textbf{8.} The argument used nothing special about the pair
$[BJ]$, $[CI]$: run on the pair $[BJ]$, $[AK]$, it shows their
crossing point also lies on $[BJ]$ at two thirds of the way from
$B$. But there is only \emph{one} point of the \omterm{def:g6:lines:objects}{segment} $[BJ]$ at
distance $\frac23 BJ$ from $B$ --- so the crossing point of
$[BJ]$ and $[AK]$ is the very same $G$. All three medians
therefore pass through $G$: they are \emph{concurrent}.

\textbf{9.} $BG = \frac23 \times 9 = 6$ cm and
$GJ = \frac13 \times 9 = 3$ cm.

\textbf{10.} By question 8 the same two-thirds computation holds
on the median $[AK]$ (run the argument of questions 4--7 with
the pair $[AK]$, $[BJ]$): $AG = 2 \times GK$.

\textbf{11.} In the triangle $GBC$, the point $K$ is the midpoint
of the side $[BC]$, so $[GK]$ is a median of $GBC$; by question
2, the triangles $GBK$ and $GKC$ have equal \omterm{def:g6:measure:area}{areas}.

\textbf{12.} The median $[AK]$ cuts $ABC$ into $ABK$ and $AKC$ of
equal \omterm{def:g6:measure:area}{areas} (question 2). Removing from each its small triangle
of question 11:
\[
\text{area}(ABG) = \text{area}(ABK) - \text{area}(GBK),
\qquad
\text{area}(ACG) = \text{area}(AKC) - \text{area}(GKC),
\]
and the removed \omterm{def:g6:measure:area}{areas} are equal, so
$\text{area}(ABG) = \text{area}(ACG)$. The same computation
along the median $[BJ]$ --- which halves $ABC$ into $ABJ$ and
$CBJ$, while $[GJ]$, a median of the triangle $GAC$, halves it
into $GAJ$ and $GCJ$ --- gives
$\text{area}(ABG) = \text{area}(CBG)$. So the three triangles
$ABG$, $BCG$, $CAG$ have equal \omterm{def:g6:measure:area}{areas}, and since together they
tile $ABC$, each has \omterm{def:g6:measure:area}{area} one third of the whole.

\textbf{13.} In the triangle $ABG$, the point $I$ is the midpoint
of the side $[AB]$, so $[GI]$ is a median of $ABG$ \emph{from the
\omterm{def:g5:solids:def}{vertex} $G$}: it cuts $ABG$ into two triangles of equal \omterm{def:g6:measure:area}{areas}
(question 2), namely $AIG$ and $IBG$. The same happens in $BCG$
(median $[GK]$) and in $CAG$ (median $[GJ]$). Each third of
$ABC$ is cut in \omterm{def:g3:division:half}{half}: the six slices around $G$ each have \omterm{def:g6:measure:area}{area}
$\frac13 \times \frac12 = \frac16$ of the triangle.

\textbf{14.} On the median $[AK]$, from $A(0,0)$ to $K(3,3)$: $G$
sits two thirds of the way, so its coordinates are
$\left(\frac23 \times 3,\ \frac23 \times 3\right) = (2, 2)$. On
the median $[BJ]$, from $B(6, 0)$ to $J(0, 3)$: two thirds of the
way means adding two thirds of the displacement ($-6$
horizontally, $+3$ vertically):
\[
\left(6 + \tfrac23 \times (-6),\ 0 + \tfrac23 \times 3\right)
= (2, 2) .
\]
The same point --- as question 8 promised. And
$\frac{0 + 6 + 0}{3} = 2$, $\frac{0 + 0 + 6}{3} = 2$: the \omterm{pb:g8:midpoints:1}{center
of gravity}'s coordinates are the \emph{averages} of the three
\omterm{def:g5:solids:def}{vertices}' coordinates.

\textbf{15.} By question 2, a knife edge laid along a median has
the same amount of cardboard on each side (equal \omterm{def:g6:measure:area}{areas}), so the
triangle balances along each of the three median lines. A point
of balance must lie on all three balance lines at once, and by
question 8 the three medians share exactly one point: $G$. That
is where the pencil tip goes. (Equal \omterm{def:g6:measure:area}{areas} on both sides is the
believable version of the argument; the honest one, weighing
\emph{where} the cardboard sits and not just how much there is,
needs the integral calculus of the university \omterm{def:g6:measure:volume}{volumes} --- which
confirms $G$.)
\end{solution}
