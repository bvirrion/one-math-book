\section*{अध्याय \ref{ch:g8:midpoints} --- मध्यबिंदु और समानताएं}

\begin{solution}{exo:g8:midpoints:1}
$M $ और $ N $ त्रिभुज की दो भुजाओं के मध्यबिंदु हैं $ RST $: द्वारा
मध्यबिंदु प्रमेय (\cref{thm:g8:midpoints:direct}),
$(MN) \parallel (ST)$ और $ MN = \frac{ST}{2} = 3.5$ सेमी।
\end{solution}

\begin{solution}{exo:g8:midpoints:2}
$ I $ एक मध्यबिंदु है और $(IJ) \parallel (BC)$: से \emph{CONVERSE}
(\cref{thm:g8:midpoints:converse}), $ J $ का मध्यबिंदु है $[AC]$,
इसलिए $ AJ = \frac{11}{2} = 5.5$ सेमी।
\end{solution}

\begin{solution}{exo:g8:midpoints:3}
माप देता है $ IJ = 3.5$ सेमी $= \frac{BC}{2}$. \omterm{def:g6:measure:perimeter}{परिधि} का
$ AIJ $: $ AI = 4$, $ AJ = 3$, $ IJ = 3.5$: कुल $10.5$ सेमी --- \omterm{def:g3:division:half}{आधा} 
\omterm{def:g6:measure:perimeter}{परिधि} का $ ABC $ ($8 + 6 + 7 = 21$ सेमी)।
\end{solution}

\begin{solution}{exo:g8:midpoints:4}
के प्रत्येक पक्ष $ IJK $ है \omterm{def:g3:division:half}{आधा} इसकी तरफ $ ABC $: $5$, $6$ और $8$ सेमी।
\omterm{def:g6:measure:perimeter}{परिधि}: $19$ सेमी, \omterm{def:g3:division:half}{आधा} का $10 + 12 + 16 = 38$ सेमी।
\end{solution}

\begin{solution}{exo:g8:midpoints:5}
मध्यबिंदु त्रिभुज चार समान त्रिभुजों में से एक है
\cref{prop:g8:midpoints:medial}: \omterm{def:g6:measure:area}{क्षेत्र} $\frac{36}{4} = 9$ सेमी $^2$ ---
और चार छोटे त्रिभुजों में से प्रत्येक में है \omterm{def:g6:measure:area}{क्षेत्र} $9$ सेमी $^2$.
\end{solution}

\begin{solution}{exo:g8:midpoints:6}
मध्यबिंदु प्रमेय द्वारा $ ABC $: $(IJ) \parallel (BC)$, अर्थात\
$(IJ) \parallel (BK)$, और $ IJ = \frac{BC}{2} = BK $ (जैसा $ K $ है
का मध्यबिंदु $[BC]$). के दो विपरीत पक्ष $ IJKB $ हैं \omterm{def:g6:lines:perp}{समानांतर} और
बराबर: यह एक है \omterm{def:g7:central:parallelogram}{चतुर्भुज}
(\cref{thm:g7:central:recognize}, मानदंड 3).
\end{solution}

\begin{solution}{exo:g8:midpoints:7}
सीवन दो मध्यबिंदुओं को जोड़ता है: $\frac{6.4}{2} = 3.2$ एम। एक पर
\omterm{def:g3:division:half}{तिमाही} ऊपर से शिखर, चित्र एक सीवन का सुझाव देता है
$\frac{6.4}{4} = 1.6$ एम, \omterm{def:g6:lines:perp}{समानांतर} आधार तक --- थेल्स द्वारा पुष्टि की गई
में \cref{ch:g9:thales}.
\end{solution}

\begin{solution}{exo:g8:midpoints:8}
\emph{1.} $ M $ का मध्यबिंदु है $[BC]$ और $(MN) \parallel (AB)$:
मध्यबिंदु प्रमेय के व्युत्क्रम द्वारा (त्रिभुज में $ ABC $,
ओर $[AC]$), $ N $ का मध्यबिंदु है $[AC]$.

\emph{2.} $(MN) \parallel (AB)$ और $(AB) \perp (AC)$ (\omterm{def:g3:shapes:rightangle}{समकोण}
पर $ A $): एक पंक्ति \omterm{def:g6:lines:perp}{समानांतर} को $(AB)$ ई आल्सो \omterm{def:g6:lines:perp}{सीधा} को $(AC)$
(\cref{prop:g6:lines:facts}). इसलिए $(MN) \perp (AC)$ --- द \omterm{def:g6:lines:objects}{खंड}
$[MN]$ है \omterm{def:g6:symmetry:bisector}{लंब समद्विभाजक} वह टुकड़ा फिर से दिखा रहा है
$ MA = MC $ (\cref{thm:g8:cosine:circle}).
\end{solution}

\begin{solution}{exo:g8:midpoints:9}
\emph{1.} \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ ABC $, $ I $ और $ J $ के मध्यबिंदु हैं $[AB]$ और
$[BC]$: $(IJ) \parallel (AC)$ और $ IJ = \frac{AC}{2}$. \omterm{def:g3:shapes:shapes}{त्रिकोण} में
$ ACD $, $ L $ और $ K $ के मध्यबिंदु हैं $[DA]$ और $[CD]$:
$(LK) \parallel (AC)$ और $ LK = \frac{AC}{2}$.

\emph{2.} इसलिए $[IJ]$ और $[LK]$ हैं \omterm{def:g6:lines:perp}{समानांतर} (दोनों \omterm{def:g6:lines:perp}{समानांतर} को
विकर्ण $(AC)$) और बराबर (दोनों)। \omterm{def:g3:division:half}{आधा} का $ AC $): $ IJKL $ एक है
\omterm{def:g7:central:parallelogram}{चतुर्भुज} (\cref{thm:g7:central:recognize}, मानदंड 3) ---
के लिए \emph{प्रत्येक} \omterm{def:g4:geometry:polygon}{चतुष्कोष} $ ABCD $, जैसा कि अनुमान लगाया गया है
\cref{exo:g7:central:11}.
\end{solution}

\begin{solution}{pb:g8:midpoints:1}
\textbf{1.} जो भी त्रिभुज बनाया जाता है, उसमें तीन माध्यिकाएँ दिखाई देती हैं
त्रिभुज के अंदर स्थित एक बिंदु से गुजरना,
विपरीत की तुलना में प्रत्येक पक्ष के मध्यबिंदु के अधिक निकट दिखाई देता है
शिखर. बाकी समस्या यह साबित करती है।

\textbf{2.} माध्यिका लें $[AK]$ \omterm{def:g3:shapes:shapes}{त्रिकोण} का $ ABC $: द
\omterm{def:g3:shapes:shapes}{त्रिकोण} $ ABK $ और $ AKC $ आधार हैं $ BK $ और $ KC $ उसी का
लंबाई ($ K $ का मध्यबिंदु है $[BC]$) उसी पंक्ति द्वारा किया गया
$(BC)$, और समान ऊँचाई: से दूरी $ A $ को $(BC)$.
से \omterm{def:g6:measure:area}{क्षेत्र} सूत्र, प्रत्येक के पास है \omterm{def:g6:measure:area}{क्षेत्र}
आधार $\times $ ऊंचाई $\div\ 2$, समान आधार और समान के साथ
ऊंचाई: \omterm{def:g6:measure:area}{क्षेत्र}ों बराबर हैं. के एक \omterm{def:g3:shapes:shapes}{त्रिकोण} के लिए \omterm{def:g6:measure:area}{क्षेत्र}
$36$ सेमी $^2$: के दो टुकड़े $18$ सेमी $^2$ प्रत्येक।

\textbf{3.} $ I $ और $ J $ भुजाओं के मध्यबिंदु हैं $[AB]$ और
$[AC]$ \omterm{def:g3:shapes:shapes}{त्रिकोण} का $ ABC $, तो द्वारा
\cref{thm:g8:midpoints:direct}: $(IJ) \parallel (BC)$ और
$ IJ = \frac{BC}{2}$.

\textbf{4.} $ M $ और $ N $ भुजाओं के मध्यबिंदु हैं $[GB]$ और
$[GC]$ \omterm{def:g3:shapes:shapes}{त्रिकोण} का $ GBC $, तो उसी प्रमेय द्वारा:
$(MN) \parallel (BC)$ और $ MN = \frac{BC}{2}$.

\textbf{5.} दोनों $(IJ)$ और $(MN)$ हैं \omterm{def:g6:lines:perp}{समानांतर} को $(BC)$, इस तरह
\omterm{def:g6:lines:perp}{समानांतर} एक दूसरे से; और $ IJ = \frac{BC}{2} = MN $.
\omterm{def:g4:geometry:polygon}{चतुष्कोष} $ IJNM $ इस प्रकार इसके दो विपरीत पक्ष हैं, $[IJ]$ और
$[NM]$, \omterm{def:g6:lines:perp}{समानांतर} और समान लंबाई का: मानदंड 3 के अनुसार
\cref{thm:g7:central:recognize}, $ IJNM $ एक है \omterm{def:g7:central:parallelogram}{चतुर्भुज}.

\textbf{6.} $ G $ माध्यिका पर स्थित है $[CI]$, और $ N $ है
का मध्यबिंदु $[GC]$, ए \omterm{def:g6:lines:objects}{खंड} उसी मध्य \omterm{def:g6:lines:objects}{रेखा} द्वारा ले जाया गया:
तो $ I $, $ N $ (और $ G $, $ C $) सभी झूठ बोलते हैं $(CI)$. वैसे ही $ J $, $ M $
और $ G $ मध्य \omterm{def:g6:lines:objects}{रेखा} पर लेटें $(BJ)$. में \omterm{def:g7:central:parallelogram}{चतुर्भुज}
$ IJNM $, विकर्ण हैं $[IN]$ और $[JM]$; उन्हें ले जाया जाता है
दो मध्य रेखाएँ, जो एक दूसरे को काटती हैं $ G $ --- तो विकर्ण
पर पार करें $ G $. से \emph{परिभाषा} एक का \omterm{def:g7:central:parallelogram}{चतुर्भुज}
(मानदंड 1 \cref{thm:g7:central:recognize}: विकर्ण
समान मध्यबिंदु हो), $ G $ का मध्यबिंदु है $[IN]$ \emph{और}
का $[JM]$: $ GI = GN $ और $ GJ = GM $.

\textbf{7.} $ M $ का मध्यबिंदु है $[GB]$, इसलिए $ BM = MG $; और
$ MG = GJ $ प्रश्न 6 से। अत: $ BM = MG = GJ $: मध्य भाग
$[BJ]$ को तीन बराबर भागों में काटा जाता है $ BG $ दो लेता है:
\[
BG = 2 \times GJ .
\]
\omterm{def:g7:central:def}{सममित} रूप से, $ N $ का मध्यबिंदु है $[GC]$ और $ GN = GI $, इसलिए
$ CN = NG = GI $ और $ CG = 2 \times GI $: माध्यिका पर $[CI]$ भी,
क्रॉसिंग पॉइंट रास्ते के दो तिहाई हिस्से पर है शिखर.

\textbf{8.} तर्क में \omterm{def:g1:addition:def}{जोड़}ी के बारे में कुछ खास इस्तेमाल नहीं किया गया
$[BJ]$, $[CI]$: \omterm{def:g1:addition:def}{जोड़}ी पर चलाएँ $[BJ]$, $[AK]$, यह उनका पता चलता है
क्रॉसिंग पॉइंट भी स्थित है $[BJ]$ से दो तिहाई रास्ते पर
$ B $. लेकिन वहाँ केवल है \emph{एक} का बिंदु \omterm{def:g6:lines:objects}{खंड} $[BJ]$ पर
दूरी $\frac23 BJ $ से $ B $ --- तो का पार बिंदु
$[BJ]$ और $[AK]$ बिलकुल वैसा ही है $ G $. तीनों माध्यिकाएँ
इसलिए गुजर जाओ $ G $: वे हैं \emph{समवर्ती}.

\textbf{9.} $ BG = \frac23 \times 9 = 6$ सेमी और
$ GJ = \frac13 \times 9 = 3$ सेमी।

\textbf{10.} प्रश्न 8 तक वही दो-तिहाई गणना लागू होती है
माध्यिका पर $[AK]$ (प्रश्न 4-7 का तर्क चलाएँ
\omterm{def:g1:addition:def}{जोड़}ी $[AK]$, $[BJ]$): $ AG = 2 \times GK $.

\textbf{11.} \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ GBC $, बात $ K $ मध्यबिंदु है
पक्ष का $[BC]$, इसलिए $[GK]$ का माध्यिका है $ GBC $; प्रश्न द्वारा
2, \omterm{def:g3:shapes:shapes}{त्रिकोण} $ GBK $ और $ GKC $ बराबर है \omterm{def:g6:measure:area}{क्षेत्र}ों.

\textbf{12.} मध्यस्थ $[AK]$ कटौती $ ABC $ में $ ABK $ और $ AKC $ का
बराबर \omterm{def:g6:measure:area}{क्षेत्र}ों (प्रश्न 2). प्रत्येक से उसका छोटा त्रिभुज हटाना
प्रश्न 11 का:
\[
\text{area}(ABG) = \text{area}(ABK) - \text{area}(GBK),
\qquad
\text{area}(ACG) = \text{area}(AKC) - \text{area}(GKC),
\]
और हटा दिया गया \omterm{def:g6:measure:area}{क्षेत्र}ों बराबर हैं, इसलिए
$\text{area}(ABG) = \text{area}(ACG)$. वही गणना
मध्यिका के साथ $[BJ]$ --- कौन सा \omterm{def:g3:division:half}{आधा} $ ABC $ में $ ABJ $ और
$ CBJ $, जबकि $[GJ]$, त्रिभुज की एक माध्यिका $ GAC $, इसे \omterm{def:g3:division:half}{आधा} कर देता है
में $ GAJ $ और $ GCJ $ --- देता है
$\text{area}(ABG) = \text{area}(CBG)$. तो तीन \omterm{def:g3:shapes:shapes}{त्रिकोण}
$ ABG $, $ BCG $, $ CAG $ बराबर है \omterm{def:g6:measure:area}{क्षेत्र}ों, और चूंकि वे एक साथ हैं
टाइल $ ABC $, प्रत्येक के पास है \omterm{def:g6:measure:area}{क्षेत्र} कुल का एक तिहाई.

\textbf{13.} \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ ABG $, बात $ I $ मध्यबिंदु है
पक्ष का $[AB]$, इसलिए $[GI]$ का माध्यिका है $ ABG $ \emph{से
शिखर $ G $}: यह काटता है $ ABG $ बराबर के दो \omterm{def:g3:shapes:shapes}{त्रिकोण}ों में \omterm{def:g6:measure:area}{क्षेत्र}ों
(प्रश्न 2), अर्थात् $ AIG $ और $ IBG $. में भी ऐसा ही होता है $ BCG $
(माध्यिका $[GK]$) और में $ CAG $ (माध्यिका $[GJ]$). प्रत्येक तिहाई
$ ABC $ में काटा जाता है \omterm{def:g3:division:half}{आधा}: चारों ओर छह स्लाइसें $ G $ प्रत्येक के पास है \omterm{def:g6:measure:area}{क्षेत्र}
$\frac13 \times \frac12 = \frac16$ \omterm{def:g3:shapes:shapes}{त्रिकोण} का.

\textbf{14.} माध्यिका पर $[AK]$, से $ A(0,0)$ को $ K(3,3)$: $ G $
रास्ते के दो-तिहाई भाग पर बैठता है, इसलिए इसके निर्देशांक हैं
$\left(\frac23 \times 3,\ \frac23 \times 3\right) = (2, 2)$. पर
माध्यिका $[BJ]$, से $ B(6, 0)$ को $ J(0, 3)$: दो तिहाई
रास्ते का अर्थ है विस्थापन का दो तिहाई जोड़ना ($-6$
क्षैतिज रूप से, $+3$ लंबवत):
\[
\left(6 + \tfrac23 \times (-6),\ 0 + \tfrac23 \times 3\right)
= (2, 2) .
\]
वही बिंदु --- जैसा कि प्रश्न 8 में वादा किया गया था। और
$\frac{0 + 6 + 0}{3} = 2$, $\frac{0 + 0 + 6}{3} = 2$: द ग्रैविटी केंद्रके निर्देशांक हैं \emph{औसत} तीन में से
\omterm{def:g5:solids:def}{कोने}'निर्देशांक।

\textbf{15.} प्रश्न 2 के अनुसार, एक चाकू की धार एक मध्यिका के साथ रखी गई है
प्रत्येक तरफ कार्डबोर्ड की समान मात्रा (बराबर)। \omterm{def:g6:measure:area}{क्षेत्र}ों), तो
त्रिभुज तीनों मध्य रेखाओं में से प्रत्येक के साथ संतुलन बनाता है। एक बिंदु
संतुलन का संतुलन सभी तीन संतुलन रेखाओं पर एक साथ और द्वारा होना चाहिए
प्रश्न 8 तीनों माध्यिकाएँ बिल्कुल एक बिंदु साझा करती हैं: $ G $. वह
वह स्थान है जहाँ पेंसिल की नोक जाती है। (बराबर \omterm{def:g6:measure:area}{क्षेत्र}ों दोनों तरफ है
तर्क का विश्वसनीय संस्करण; ईमानदार एक, वजन
\emph{कहाँ} कार्डबोर्ड बैठता है और न केवल कितना है,
विश्वविद्यालय के अभिन्न कलन की आवश्यकता है संस्करणों --- जो
पुष्टि करता है $ G$.)
\end{solution}
