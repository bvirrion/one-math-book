\begin{solution}{exo:g8:speed:9}
भारित:
$\frac{3 \times 15 + 2 \times 9 + 1 \times 12}{3 + 2 + 1}
= \frac{45 + 18 + 12}{6} = \frac{75}{6} = 12.5$.
अभारित: $\frac{15 + 9 + 12}{3} = 12$. वे भिन्न हैं क्योंकि
गुणांक चिह्न देते हैं $15$ निशान के वजन का तीन गुना
$12$.
\end{solution}

\begin{solution}{exo:g8:speed:10}
\emph{1.} टाइम्स: $\frac{120}{80} = 1.5$ हाथ $\frac{120}{120} = 1$
एच। \omterm{def:g8:speed:def}{औसत गति}: $\frac{240}{2.5} = 96$ किमी/घंटा.

\emph{2.} कार खर्च करती है \emph{अधिक समय} पर $80$ किमी/घंटा ($1.5$ ज)
की तुलना में $120$ किमी/घंटा ($1$ ज): धीमा रफ़्तार में अधिक वजन होता है
समय-भारित औसत, इसे मध्यबिंदु से नीचे खींच रहा है $100$.
\end{solution}

\begin{solution}{exo:g8:speed:11}
का \omterm{ex:g1:subtraction:difference}{अंतर} $18$ किमी पर बंद हो जाता है $5 + 4 = 9$ किमी/घंटा: वे बाद में मिलते हैं
$\frac{18}{9} = 2$ एच, पर $12{:}00$. फिर अन्ना चल पड़े
$2 \times 5 = 10$ किमी: वे मिलते हैं $10$ अन्ना के गांव से किमी (और)
$8$ बोरिस से किमी --- जाँच करें: $10 + 8 = 18$).
\end{solution}

\begin{solution}{pb:g8:speed:1}
\textbf{1.} द्वारा \cref{def:g8:speed:def}, $ t_1 = \frac{d}{v_1}$ और
$ t_2 = \frac{d}{v_2}$; पूरी यात्रा चलती है
$\frac{d}{v_1} + \frac{d}{v_2}$ की दूरी के लिए $2d $.

\textbf{2.} कारक $ d $ से बाहर \omterm{def:g5:division:multiple}{भाजक} और इसे सरल बनाएं
दूर:
\[
\bar v
= \frac{2d}{d\left(\frac{1}{v_1} + \frac{1}{v_2}\right)}
= \frac{2}{\ \frac{1}{v_1} + \frac{1}{v_2}\ } .
\]
सामान्य \omterm{def:g5:division:multiple}{भाजक} अंदर:
$\frac{1}{v_1} + \frac{1}{v_2} = \frac{v_2 + v_1}{v_1 v_2}$, और
भाग देने का अर्थ है गुणा करना \omterm{def:g8:fractions:inverse}{श्लोक में}
(\cref{thm:g8:fractions:div}):
\[
\bar v = 2 \times \frac{v_1 v_2}{v_1 + v_2}
= \frac{2\, v_1 v_2}{v_1 + v_2} .
\]

\textbf{3.} $\dfrac{2 \times 30 \times 15}{30 + 15}
= \dfrac{900}{45} = 20$ किमी/घंटा, जैसा कि
\cref{ex:g8:speed:twolegs}; और
$\dfrac{2 \times 80 \times 120}{80 + 120} = \dfrac{19\,200}{200}
= 96$ किमी/घंटा, जैसा कि पाया गया \cref{exo:g8:speed:10}.

\textbf{4.} औसत इन दोनों पर ही निर्भर करता है स्पीड, पर नहीं
$ d $: ए $1$ किमी राउंड ट्रिप और ए $100$ एक ही समय में किमी राउंड ट्रिप
दो स्पीड बिल्कुल वैसा ही है \omterm{def:g8:speed:def}{औसत गति}. (यही कारण है कि
विभिन्न लंबाई की यात्राओं पर सूत्र की जाँच की जा सकती है।)

\textbf{5.} समय के भीतर $ T $ हरेक पर रफ़्तार, दूरियाँ हैं $ v_1 T $
और $ v_2 T $, इसलिए
\[
\bar v = \frac{v_1 T + v_2 T}{2T}
= \frac{(v_1 + v_2)\,T}{2T}
= \frac{v_1 + v_2}{2} .
\]
एक औसत वर्ष स्पीड द्वारा सदैव भारित किया जाता है \emph{समय}
(\cref{def:g8:speed:average}). दोनों बराबर समय के साथ स्पीड
समान वजन रखें: सादा मध्यबिंदु। समान दूरी के साथ
धीमा रफ़्तार पर \emph{अधिक} समय, इसलिए अधिक भार ---
और औसत उसकी ओर सरकता है।

\textbf{6.} के लिए $(30, 15)$: $ H = 20$ और $ A = 22.5$. के लिए
$(80, 120)$: $ H = 96$ और $ A = 100$. अंकगणितीय माध्य जीतता है
दोनों बार.

\textbf{7.} सामान्य से अधिक \omterm{def:g5:division:multiple}{भाजक} $2(v_1 + v_2)$:
\[
A - H
= \frac{(v_1 + v_2)^2 - 4\, v_1 v_2}{2\,(v_1 + v_2)} ,
\]
और यह \omterm{def:g4:fractions:def}{मीटर} की पहचान के साथ विस्तारित होता है
\cref{pb:g8:equations:1}:
\[
(v_1 + v_2)^2 - 4 v_1 v_2
= v_1^2 + 2 v_1 v_2 + v_2^2 - 4 v_1 v_2
= v_1^2 - 2 v_1 v_2 + v_2^2
= (v_1 - v_2)^2 .
\]

\textbf{8.} एक वर्ग कभी भी ऋणात्मक नहीं होता, और $2(v_1 + v_2)$ है
सकारात्मक: तो $ A - H \geq 0$, वह है $ H \leq A $, हर \omterm{def:g1:addition:def}{जोड़}ी के लिए
सकारात्मक का स्पीड. समानता की आवश्यकता है $(v_1 - v_2)^2 = 0$, अर्थात\
$ v_1 = v_2$: जिस क्षण दो स्पीड भिन्न, \omterm{def:g4:numbers:round}{गोल} यात्रा की
औसत बिल्कुल मध्यबिंदु से नीचे चला जाता है --- आश्चर्य की बात है
प्रमेय.

\textbf{9.} गुणा करें:
\[
H \times A
= \frac{2\, v_1 v_2}{v_1 + v_2} \times \frac{v_1 + v_2}{2}
= v_1 v_2 ,
\]
कारक को रद्द करने के बाद $2$ और कारक $ v_1 + v_2$.

\textbf{10.} सूत्र: $\bar v = \dfrac{2 \times 40 \times 60}
{40 + 60} = \dfrac{4800}{100} = 48$ किमी/घंटा. लंबा रास्ता:
$\frac{60}{40} = 1.5$ हाथ $\frac{60}{60} = 1$ ज, तो $120$ किमी में
$2.5$ एच: $\frac{120}{2.5} = 48$ किमी/घंटा. वे सहमत हैं.

\textbf{11.} औसत $30$ किमी/घंटा से अधिक $30$ किमी का मतलब कुल समय है
अधिक से अधिक $\frac{30}{30} = 1$ एच। पहला $15$ किमी पर $15$ किमी/घंटा
पहले से ही ले लो $\frac{15}{15} = 1$ एच: द \emph{पूरा} समय बजट
के साथ बिताया जाता है \omterm{def:g3:division:half}{आधा} जाने का कोर्स.

\textbf{12.} पर $90$ किमी/घंटा दूसरा \omterm{def:g3:division:half}{आधा} लेता है
$\frac{15}{90} = \frac16$ ज, तो दौड़ लगती है $\frac76$ हाथ
\[
\bar v = \frac{30}{\ 7/6\ } = \frac{180}{7} \approx 25.7
\text{ km/h} < 30 .
\]
जो भी दूसरी छमाही रफ़्तार, इसकी अवधि सकारात्मक है
संख्या, अत: कुल समय अधिक हो गया है $1$ h और औसत रहता है
सख्ती से नीचे $30$ किमी/घंटा. (सूत्र के साथ: $\frac{2 \times 15
\times v_2}{15 + v_2} = 30$ जबरदस्ती करेगा $30 v_2 = 450 + 30 v_2$,
यानी\ $0 = 450$ --- नहीं रफ़्तार $ v_2$ काम करता है.)

\textbf{13.} ए $20$ किमी/घंटा औसत अनुमति देता है $\frac{30}{20} = 1.5$ एच;
पहले घंटे के बाद, $15$ किमी तक \omterm{def:g3:division:remainder}{शेष} है $0.5$ ज: उसे चाहिए
$\frac{15}{0.5} = 30$ किमी/घंटा. जाँच करना:
$ H(15, 30) = \frac{2 \times 15 \times 30}{45} = \frac{900}{45}
= 20$ किमी/घंटा.

\textbf{14.} प्रति मिनट नल भर जाते हैं $\frac{1}{20}$ और
$\frac{1}{30}$ टब का; एक साथ
\[
\frac{1}{20} + \frac{1}{30}
= \frac{3}{60} + \frac{2}{60}
= \frac{5}{60} = \frac{1}{12}
\]
टब का प्रति मिनट: टब भर जाता है $12$ मि. और
$ H(20, 30) = \frac{2 \times 600}{50} = 24 = 2 \times 12$: दो
नल एक साथ बिल्कुल लेते हैं \emph{\omterm{def:g3:division:half}{आधा}  \omterm{pb:g8:speed:1}{अनुकूल माध्य}} उनके
एकल समय --- दरें जुड़ती हैं, इसलिए यह फिर से हार्मोनिक-माध्य \omterm{def:g6:measure:area}{क्षेत्र} है।

\textbf{15.} हर तरफ समान दूरी, इसलिए
\[
\bar v = \frac{2 \times 900 \times 700}{900 + 700}
= \frac{1\,260\,000}{1\,600} = 787.5 \text{ km/h},
\]
शांत हवा के नीचे $800$ किमी/घंटा. एक वाक्य में: विमान खर्च करता है
\emph{अब} हवा की तुलना में उसके विपरीत उड़ना, इसलिए धीमा
समय-भारित औसत में पैर को बड़ा भार मिलता है --- ए
हवा केवल एक चक्कर को लंबा कर सकती है।
\end{solution}
