\section*{अध्याय \ref{ch:g8:negprod} --- ऋणात्मक संख्याओं को गुणा करना}

\begin{solution}{exo:g8:negprod:1}
$(-8) \times (+7) = -56$; \quad
$(-9) \times (-6) = +54$; \quad
$(+12) \times (-0.5) = -6$; \quad
$(-1) \times (-1) \times (-1) = -1$ (तीन नकारात्मक कारक: \omterm{def:g3:numbers:evenodd}{विषम}).
\end{solution}

\begin{solution}{exo:g8:negprod:2}
$(-63) \div (+9) = -7$; \quad
$(-8) \div (-16) = +0.5$; \quad
$\dfrac{45}{-5} = -9$; \quad
$\dfrac{-7.2}{-8} = +0.9$.
\end{solution}

\begin{solution}{exo:g8:negprod:3}
तीन नकारात्मक कारक: \emph{नकारात्मक}. \quad
$(-1)^{10}$: दस नकारात्मक कारक, \omterm{def:g3:numbers:evenodd}{यहां तक की}: \emph{सकारात्मक} (यह बराबर है
$1$). \quad
$(-2)^7$: सात कारक, \omterm{def:g3:numbers:evenodd}{विषम}: \emph{नकारात्मक}. \quad
$(-4) \times 0 \times (-6) = 0$: न तो सकारात्मक और न ही नकारात्मक.
\end{solution}

\begin{solution}{exo:g8:negprod:4}
$(-2)^4 = +16$; \quad $-2^4 = -16$ (\omterm{def:g1:counting:zero}{शून्य} का वर्ग नहीं किया गया है); \quad
$(-3)^3 = -27$; \quad $(-1)^{2026} = +1$ (\omterm{def:g3:numbers:evenodd}{यहां तक की} प्रतिपादक)।
\end{solution}

\begin{solution}{exo:g8:negprod:5}
$7 + 2 \times (-6) = 7 + (-12) = -5$.

$(-4) \times (5 - 9) = (-4) \times (-4) = +16$.

$18 \div (-3) - (-2) = -6 + 2 = -4$.
\end{solution}

\begin{solution}{exo:g8:negprod:6}
$\dfrac{(-5) \times (+8)}{-4} = \dfrac{-40}{-4} = +10$.

$\dfrac{(-3) \times (-6)}{(-2) \times (+9)} = \dfrac{+18}{-18} = -1$.
\end{solution}

\begin{solution}{exo:g8:negprod:7}
के लिए $ x = -3$: \ $5x = -15$; \quad $ x^2 = (-3)^2 = 9$; \quad
$-x^2 = -9$; \quad $2x^2 + 4x = 18 - 12 = 6$; \quad
$(2x)^2 = (-6)^2 = 36$.
\end{solution}

\begin{solution}{exo:g8:negprod:8}
$(-6) \times (-7) = 42$; \quad
$10 \div (-4) = -2.5$; \quad
$(-5) \times \frac15 = -1$, तो लुप्त संख्या है
$+\frac15 = 0.2$.
\end{solution}

\begin{solution}{exo:g8:negprod:9}
\emph{1. मिथ्या}: $0.5 \times 0.5 = 0.25$ की तुलना में छोटा है
$0.5 + 0.5 = 1$ (या: $2 \times 3 = 6 > 5$, लेकिन दावा कायम रहना चाहिए
हमेशा)।

\emph{2. सत्य}: एक वर्ग एक है \omterm{def:g2:mult:def}{उत्पाद} एक जैसी दो संख्याओं का
संकेत, इसलिए सकारात्मक या \omterm{def:g1:counting:zero}{शून्य}.

\emph{3. मिथ्या}: $(-2) \times (-3) \times 4 = +24$ के साथ सकारात्मक है
दो नकारात्मक कारक.
\end{solution}

\begin{solution}{exo:g8:negprod:10}
होने देना $ r $ सही उत्तरों की संख्या हो; तब $20 - r $ ग़लत हैं, और
\[
5r - 3(20 - r) = 36
\ \Longrightarrow\
5r - 60 + 3r = 36
\ \Longrightarrow\
8r = 96
\ \Longrightarrow\
r = 12 .
\]
बारह सही उत्तर (और आठ ग़लत: जाँचें,
$60 - 24 = 36$).
\end{solution}

\begin{solution}{exo:g8:negprod:11}
\omterm{def:g1:counting:zero}{शून्य} कई बार कुछ भी होता है \omterm{def:g1:counting:zero}{शून्य}, इसलिए $(-1) \times 0 = 0$. लिखना
$0 = 1 + (-1)$ और वितरण:
\[
0 = (-1) \times \bigl(1 + (-1)\bigr)
= (-1) \times 1 + (-1) \times (-1)
= -1 + (-1) \times (-1).
\]
जो नंबर $(-1) \times (-1)$ में \omterm{def:g1:addition:def}{जोड़}ा गया $-1$ देता है $0$: यह होना ही चाहिए
के विपरीत $-1$, वह है $+1$. वितरण किसी अन्य को नहीं छोड़ता
विकल्प.
\end{solution}

\begin{solution}{pb:g8:negprod:1}
\textbf{1.} तब से $0 = 0 + 0$, वितरणशीलता
(\cref{thm:g7:priorities:distributivity}) देता है
\[
0 \times a = (0 + 0) \times a = 0 \times a + 0 \times a .
\]
तो \omterm{def:g1:addition:def}{जोड़} रहा हूँ $0 \times a $ अपने आप में कुछ भी नहीं बदलता --- और केवल
नंबर एक कुछ भी बदले बिना \omterm{def:g1:addition:def}{जोड़}ा जा सकता है $0$. (सीधे:
घटाना $0 \times a $ दोनों तरफ से।) इसलिए $0 \times a = 0$.

\textbf{2.} विस्तार करना, फिर उपयोग करना $1 + (-1) = 0$ और प्रश्न 1:
\[
a + (-1) \times a
= 1 \times a + (-1) \times a
= \bigl(1 + (-1)\bigr) \times a
= 0 \times a = 0 .
\]
इसलिए $(-1) \times a $ वह संख्या है, जिसमें \omterm{def:g1:addition:def}{जोड़}ा जाता है $ a $, देता है $0$:
यह बिल्कुल विपरीत है $ a $, और $(-1) \times a = -a $.
\Cref{exo:g8:negprod:11} वि\omterm{def:g3:division:remainder}{शेष} मामला है $ a = -1$: वहाँ,
$(-1) \times (-1) = -(-1) = +1$.

\textbf{3.} लिखना $-a = (-1) \times a $ (प्रश्न 2) और पुनः समूहित करें
कारक:
\[
(-a) \times b
= \bigl((-1) \times a\bigr) \times b
= (-1) \times (a \times b)
= -(a \times b) .
\]

\textbf{4.} प्रश्न 3 को दो बार लागू करें (प्रत्येक कारक पर एक बार):
\[
(-a) \times (-b)
= -\bigl(a \times (-b)\bigr)
= -\bigl(-(a \times b)\bigr)
= a \times b ,
\]
चूँकि किसी संख्या के विपरीत का विपरीत संख्या ही वह संख्या होती है।

\textbf{5.} लिखना $ a = +d $ या $ a = -d $ और $ b = +e $ या $ b = -e $,
कहाँ $ d $ और $ e $ की दूरियाँ हैं \omterm{def:g1:counting:zero}{शून्य}. चार मामले:
\[
(+d) \times (+e) = de, \quad
(+d) \times (-e) = -de, \quad
(-d) \times (+e) = -de, \quad
(-d) \times (-e) = +de,
\]
बीच वाले दो को प्रश्न 3 से, अंतिम को प्रश्न 4 से। हर मामले में
परिणाम की दूरी है $ d \times e $, द \omterm{def:g2:mult:def}{उत्पाद} का
दूरियाँ, और संकेत वह है जिसकी तालिका द्वारा घोषणा की गई है
\cref{thm:g8:negprod:rules}: समान चिह्न सकारात्मक, विपरीत चिह्न
नकारात्मक.

\textbf{6.} $(-1)^2 = +1$, $(-1)^3 = -1$, $(-1)^4 = +1$,
$(-1)^5 = -1$. सामान्य तौर पर $(-1)^n $ एक है \omterm{def:g2:mult:def}{उत्पाद} का $ n $ नकारात्मक
कारक, प्रत्येक दूरी $1$: द्वारा \cref{prop:g8:negprod:many} यह है
$+1$ कब $ n $ है \omterm{def:g3:numbers:evenodd}{यहां तक की} और $-1$ कब $ n $ है \omterm{def:g3:numbers:evenodd}{विषम}.

\textbf{7.} दस नकारात्मक कारक हैं --- a सम संख्या --- इतना \omterm{def:g2:mult:def}{उत्पाद} है \emph{सकारात्मक} (\cref{prop:g8:negprod:many}). यह
दूरी है $1 \times 2 \times 3 \times \dots \times 10$.

\textbf{8.} प्रश्न 4 से, $(-a)^2 = (-a) \times (-a) = a \times a =
a^2$. फिर प्रश्न 3 से,
\[
(-a)^3 = (-a)^2 \times (-a) = a^2 \times (-a) = -(a^2 \times a)
= -a^3 .
\]
ऋण चिह्न गायब हो जाता है \omterm{def:g3:numbers:evenodd}{यहां तक की} प्रतिपादक और के लिए जीवित रहता है \omterm{def:g3:numbers:evenodd}{विषम}
वाले, के अनुरूप \cref{ex:g8:negprod:powers}.

\textbf{9.} अगर $ a = 0$ तब $ a^2 = 0$. अन्यथा $ a $ और $ a $ लीजिए \emph{वही} साइन, तो $ a^2 = a \times a $ सकारात्मक है
(\cref{thm:g8:negprod:rules}). अत: वर्ग कभी नहीं होता
नकारात्मक; तब से $-4$ नकारात्मक है, नहीं \omterm{def:g7:negatives:def}{सापेक्ष संख्या} $ x $ संतुष्ट
$ x^2 = -4$.

\textbf{10.} ज़ो सही है. प्रतिपादक $2026$ और $2027$ है
विपरीत समानताएं, इसलिए $(-1)^{2026} = +1$ और $(-1)^{2027} = -1$,
जो जोड़ता है $0$. सामान्यतः, दो क्रमागत घातांकों में से एक होता है
\omterm{def:g3:numbers:evenodd}{यहां तक की} और एक है \omterm{def:g3:numbers:evenodd}{विषम}, तो दो शक्तियाँ हैं $+1$ और $-1$ कुछ में
क्रम: विपरीत संख्याएँ, किसकी \omterm{def:g1:addition:def}{जोड़} हमेशा से रहा है $0$.

\textbf{11.} $ A_3 = 1 - 2 + 3 = 2$, तब
$ A_4 = A_3 - 4 = -2$, $ A_5 = A_4 + 5 = 3$, $ A_6 = A_5 - 6 = -3$,
$ A_7 = A_6 + 7 = 4$. अनुमान: $ A_n = -\frac{n}{2}$ के लिए \omterm{def:g3:numbers:evenodd}{यहां तक की} $ n $,
और $ A_n = \frac{n+1}{2}$ के लिए \omterm{def:g3:numbers:evenodd}{विषम} $ n $.

\textbf{12.} के लिए \omterm{def:g3:numbers:evenodd}{यहां तक की} $ n $ शर्तें पूरी तरह से \omterm{def:g1:addition:def}{जोड़}ी गई हैं:
\[
A_n = (1 - 2) + (3 - 4) + \dots + \bigl((n-1) - n\bigr),
\]
और प्रत्येक \omterm{def:g1:addition:def}{जोड़}ी बराबर है $-1$. वहाँ हैं $\frac{n}{2}$ \omterm{def:g1:addition:def}{जोड़}े (दो पद
प्रति \omterm{def:g1:addition:def}{जोड़}ा, $ n $ सभी शर्तें), तो
$ A_n = \frac{n}{2} \times (-1) = -\frac{n}{2}$.

\textbf{13.} के लिए \omterm{def:g3:numbers:evenodd}{विषम} $ n $, द \omterm{def:g1:addition:def}{जोड़} $ A_n $ है \omterm{def:g1:addition:def}{जोड़} $ A_{n-1}$ का
प्रथम $ n - 1$ पद, साथ ही अंतिम पद, जो है $+n $ तब से $ n $ है
\omterm{def:g3:numbers:evenodd}{विषम}. जैसा $ n - 1$ है \omterm{def:g3:numbers:evenodd}{यहां तक की}, प्रश्न 12 देता है
\[
A_n = A_{n-1} + n = -\frac{n-1}{2} + n
= \frac{-(n-1) + 2n}{2} = \frac{n+1}{2} .
\]

\textbf{14.} $ A_{100} = -\frac{100}{2} = -50$, \quad
$ A_{101} = \frac{102}{2} = 51$, \quad
$ A_{2026} = -\frac{2026}{2} = -1013$.

\textbf{15.} कोई आश्चर्य नहीं: से गुजर रहा है $ A_{100}$ को $ A_{101}$ जोड़ता है
बिल्कुल एक शब्द, अर्थात् $+101$, इतना \omterm{ex:g1:subtraction:difference}{अंतर} होना ही था $101$.
इसी कारण से, $ A_{2027} - A_{2026} = +2027$ --- बिना
कंप्यूटिंग $ A_{2027}$ बिल्कुल भी।
\end{solution}
