\begin{solution}{exo:g8:pythagoras:11}
\emph{1.} सीधे: $(a + b)^2$. टुकड़ों के रूप में: के चार \omterm{def:g3:shapes:shapes}{त्रिकोण} \omterm{def:g6:measure:area}{क्षेत्र}
$\frac{ab}{2}$ प्रत्येक, साथ ही झुका हुआ वर्ग $ c^2$:
\[
(a + b)^2 = 4 \times \frac{ab}{2} + c^2 = 2ab + c^2 .
\]

\emph{2.} बाईं ओर का विस्तार:
$(a+b)^2 = a^2 + 2ab + b^2$. इसलिए
\[
a^2 + 2ab + b^2 = 2ab + c^2
\quad\Longrightarrow\quad
a^2 + b^2 = c^2 . \qed
\]
\end{solution}

\begin{solution}{pb:g8:pythagoras:1}
\textbf{1.} $5^2 + 12^2 = 25 + 144 = 169 = 13^2$; \quad
$8^2 + 15^2 = 64 + 225 = 289 = 17^2$; \quad
$20^2 + 21^2 = 400 + 441 = 841 = 29^2$. तीनों त्रिगुण हैं,
और द्वारा \cref{thm:g8:pythagoras:converse} इनके साथ एक \omterm{def:g3:shapes:shapes}{त्रिकोण}
साइड की लंबाई है \omterm{def:g6:shapes:triangles}{समकोण का}, द \omterm{def:g3:shapes:rightangle}{समकोण} सबसे लंबे समय तक सामना करना
ओर.

\textbf{2.} रस्सी भुजाओं वाला एक त्रिभुज बनाती है $3$, $4$, $5$
(में \omterm{def:g6:lines:objects}{खंड}ों), और $3^2 + 4^2 = 25 = 5^2$: बातचीत से
(\cref{thm:g8:pythagoras:converse}), भुजाओं के बीच का कोण
का $3$ और $4$ \omterm{def:g6:lines:objects}{खंड}ों बिल्कुल सही है --- कोई चांदा नहीं
जरूरत है, केवल गांठें।

\textbf{3.} अगर $ a^2 + b^2 = c^2$, तब
\[
(ka)^2 + (kb)^2 = k^2 a^2 + k^2 b^2
= k^2\left(a^2 + b^2\right) = k^2 c^2 = (kc)^2 .
\]
से $(3, 4, 5)$: उदाहरण के लिए $(6, 8, 10)$, $(9, 12, 15)$ और
$(30, 40, 50)$.

\textbf{4.} ए एकाधिक $(3k, 4k, 5k)$ युक्त $5$ इसके रूप में
सबसे छोटी संख्या की आवश्यकता होगी $3k = 5$, और कोई पूर्ण संख्या नहीं $ k $
संतुष्ट करता है कि ($ k = 1$ देता है $3$, $ k = 2$ देता है $6$). इसलिए
$(5, 12, 13)$ नहीं है कोई एकाधिक का $(3, 4, 5)$.

\textbf{5.} प्रश्न 3 से पता चलता है कि स्केलिंग से केवल त्रिगुण उत्पन्न होते हैं
$(3k, 4k, 5k)$; प्रश्न 4 एक वास्तविक त्रिगुण प्रदर्शित करता है जो कि नहीं है
इस रूप का. तो सबका परिवार पायथागॉरियन त्रिगुण है
से सख्ती से बड़ा \omterm{def:g5:division:multiple}{गुणकों} का $(3, 4, 5)$: उनका उत्पादन करना
कुछ नया चाहिए.

\textbf{6.} लिखना $ A = m^2$ और $ B = n^2$. उल्लेखनीय द्वारा
पहचान (\cref{pb:g8:equations:1}):
\[
a^2 = (A - B)^2 = A^2 - 2AB + B^2,
\qquad
b^2 = (2mn)^2 = 4m^2n^2 = 4AB .
\]
जोड़ना:
\[
a^2 + b^2 = A^2 - 2AB + B^2 + 4AB
= A^2 + 2AB + B^2 = (A + B)^2
= \left(m^2 + n^2\right)^2 = c^2 .
\]

\textbf{7.}
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccc}
$(m, n)$ & $ m^2 - n^2$ & $2mn $ & $ m^2 + n^2$ \\
\hline
$(2, 1)$ & $3$ & $4$ & $5$ \\
$(3, 1)$ & $8$ & $6$ & $10$ \\
$(3, 2)$ & $5$ & $12$ & $13$ \\
$(4, 1)$ & $15$ & $8$ & $17$ \\
$(4, 3)$ & $7$ & $24$ & $25$ \\
\end{tabular}
\end{center}

\textbf{8.} $(8, 6, 10)$ का दोगुना है $(4, 3, 5)$ --- वह
है, बिल्डरों का $(3, 4, 5)$. मशीन कभी-कभी
भेस में एक पुराने त्रिक को फिर से खोजता है।

\textbf{9.} ज़रुरत है $ m^2 - n^2 = 21$ और $2mn = 20$, इसलिए
$ mn = 10$: कोशिश कर रहा हूँ $ m = 5$, $ n = 2$ देता है $25 - 4 = 21$ और
$2 \times 10 = 20$, कर्ण के साथ $25 + 4 = 29$. मशीन
आउटपुट $(21, 20, 29)$ के लिए $(m, n) = (5, 2)$.

\textbf{10.} $ m = 8$, $ n = 6$ देता है $ m^2 + n^2 = 64 + 36 = 100$.
त्रिगुण है
\[
\left(64 - 36,\ 2 \times 48,\ 100\right) = (28,\ 96,\ 100),
\]
सचमुच $28^2 + 96^2 = 784 + 9216 = 10\,000 = 100^2$.

\textbf{11.} साथ $ m = n + 1$:
\[
m^2 - n^2 = (n + 1)^2 - n^2 = 2n + 1
\]
(\omterm{ex:g1:subtraction:difference}{अंतर} क्रमागत वर्गों का, \cref{pb:g8:equations:1});
$2mn = 2n(n + 1) = 2n^2 + 2n $; और
$ m^2 + n^2 = n^2 + n^2 + 2n + 1 = 2n^2 + 2n + 1$. कर्ण
से अधिक है \omterm{def:g3:numbers:evenodd}{यहां तक की} बिल्कुल पैर से $1$. के लिए $ n = 1, 2, 3, 4$:
\[
(3, 4, 5), \quad (5, 12, 13), \quad (7, 24, 25), \quad
(9, 40, 41) .
\]

\textbf{12.} प्रत्येक \omterm{def:g3:numbers:evenodd}{विषम} संख्या कम से कम $3$ है $2n + 1$ कुछ के लिए
$ n \geq 1$, और प्रश्न 11 एक त्रिगुण प्रदान करता है जिसका \omterm{def:g3:numbers:evenodd}{विषम} पैर है
बिलकुल $2n + 1$. के लिए $11 = 2 \times 5 + 1$, लेना $ n = 5$:
\[
(11,\ 60,\ 61),
\qquad
11^2 + 60^2 = 121 + 3600 = 3721 = 61^2 .
\]

\textbf{13.} साथ $ n = 1$ मशीन देती है
$\left(m^2 - 1,\ 2m,\ m^2 + 1\right)$, किसका \omterm{def:g3:numbers:evenodd}{यहां तक की} टांग $2m $ हो सकता है
कोई भी सम संख्या से $4$ ऊपर की ओर. के लिए $14 = 2 \times 7$, लेना
$ m = 7$: त्रिगुण $(48, 14, 50)$. वही त्रिगुण प्रकट होता है
दोहरीकरण $(7, 24, 25)$ प्रश्न 7 की तालिका से:
$(14, 48, 50)$ --- वही तीन नंबर.

\textbf{14.} एक सम संख्या है $2k $, और $(2k)^2 = 4k^2$ है \omterm{def:g3:numbers:evenodd}{यहां तक की}
(ए एकाधिक का $4$, \omterm{def:g3:numbers:evenodd}{यहां तक की}). एक \omterm{def:g3:numbers:evenodd}{विषम} संख्या है $2k + 1$, और
पहली उल्लेखनीय पहचान देता है
\[
(2k + 1)^2 = 4k^2 + 4k + 1 = 2\left(2k^2 + 2k\right) + 1 ,
\]
एक सम संख्या प्लस $1$: \omterm{def:g3:numbers:evenodd}{विषम}.

\textbf{15.} मान लीजिए कि सभी $ a $, $ b $, $ c $ थे \omterm{def:g3:numbers:evenodd}{विषम}. सवाल से
14, $ a^2$ और $ b^2$ हैं \omterm{def:g3:numbers:evenodd}{विषम}, इसलिए $ a^2 + b^2$ एक है \omterm{def:g1:addition:def}{जोड़} दोनों में से अजीब
नंबर: \emph{\omterm{def:g3:numbers:evenodd}{यहां तक की}}. लेकिन $ c^2$ होगा \emph{\omterm{def:g3:numbers:evenodd}{विषम}}. समानता
$ a^2 + b^2 = c^2$ फिर असंभव है. तो हर में पायथागॉरियन
तिगुना तीन संख्याओं में से कम से कम एक है \omterm{def:g3:numbers:evenodd}{यहां तक की} --- हर एक की तरह
प्रश्न 7 की तालिका की पंक्ति पुष्टि करती है।
\end{solution}
