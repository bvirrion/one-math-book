\begin{solution}{exo:g8:cosine:10}
\emph{1.} $ A $ के घेरे पर है \omterm{def:g4:geometry:circle}{व्यास} $[BC]$: \omterm{def:g3:shapes:rightangle}{समकोण} पर $ A $
(\cref{thm:g8:cosine:circle}).

\emph{2.} $ BC = 9$ सेमी (दोगुना) \omterm{def:g3:shapes:shapes}{RADIUS}). पाइथागोरस:
$ AC = \sqrt{9^2 - 5.4^2} = \sqrt{81 - 29.16} = \sqrt{51.84} = 7.2$
सेमी। और $\cos\widehat B = \dfrac{AB}{BC} = \dfrac{5.4}{9} = 0.6$.
\end{solution}

\begin{solution}{exo:g8:cosine:11}
में समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण} इस पक्ष की $1$, एक से ऊंचाई शिखर
पर उतरता है \emph{मध्य} विपरीत दिशा का (की धुरी
समरूपता). \omterm{def:g3:division:half}{आधा} \omterm{def:g3:shapes:shapes}{त्रिकोण} है \omterm{def:g6:shapes:triangles}{समकोण का} कर्ण के साथ $1$
(एक पूर्ण पक्ष), का एक कोण $60^\circ $ आधार पर शिखर, और
बगल की ओर $\frac12$ (\omterm{def:g3:division:half}{आधा} आधार). इस तरह
\[
\cos 60^\circ = \frac{1/2}{1} = \frac12 .
\]
\end{solution}

\begin{solution}{pb:g8:cosine:1}
\textbf{1.} \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ ABH $, \omterm{def:g6:shapes:triangles}{समकोण का} पर $ H $, कोना
\omterm{def:g1:addition:def}{जोड़} (\cref{thm:g7:triangles:sum}) देता है
$\widehat B + 90^\circ + \widehat{BAH} = 180^\circ $, इसलिए
$\widehat{BAH} = 90^\circ - \widehat B $. \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ ABC $,
\omterm{def:g6:shapes:triangles}{समकोण का} पर $ A $:
$\widehat B + \widehat C + 90^\circ = 180^\circ $, इसलिए
$\widehat C = 90^\circ - \widehat B $. इस तरह
$\widehat{BAH} = \widehat{ACB}$: दोनों छोटे त्रिभुज दोहराते हैं
कोण $\widehat B $, $\widehat C $, $90^\circ $ मूल का.

\textbf{2.} में $ ABC $ (\omterm{def:g6:shapes:triangles}{समकोण का} पर $ A $), बगल वाला पक्ष
$\widehat B $ है $ AB $ और कर्ण है $ BC $:
$\cos\widehat B = \frac{AB}{BC}$. में $ ABH $ (\omterm{def:g6:shapes:triangles}{समकोण का} पर $ H $),
बगल की ओर $\widehat B $ है $ BH $ और कर्ण है
$ AB $: $\cos\widehat B = \frac{BH}{AB}$. अनुपात पर ही निर्भर करता है
कोण (\cref{def:g8:cosine:def}), इसलिए
\[
\frac{AB}{BC} = \frac{BH}{AB} ;
\]
दोनों पक्षों को गुणा करने पर $ BC \times AB $
(\cref{thm:g8:equations:balance}):
$ AB^2 = BH \times BC $.

\textbf{3.} \omterm{def:g3:shapes:shapes}{त्रिकोण} $ ABC $ (\omterm{def:g6:shapes:triangles}{समकोण का} पर $ A $) और $ ACH $
(\omterm{def:g6:shapes:triangles}{समकोण का} पर $ H $) कोण साझा करें $\widehat C $. निकटवर्ती पक्ष
प्रत्येक में कर्ण से अधिक:
\[
\cos\widehat C = \frac{AC}{CB} = \frac{CH}{AC}
\qquad\Longrightarrow\qquad
AC^2 = CH \times CB .
\]

\textbf{4.} $ AB^2 = BH \times BC $ पढ़ता $9 = BH \times 5$, इसलिए
$ BH = 1.8$; और $16 = CH \times 5$ देता है $ CH = 3.2$. जाँच करना:
$ BH + HC = 1.8 + 3.2 = 5 = BC $ --- जैसा कि होना चाहिए, चूँकि $ H $ पर झूठ बोलता है
बीच का कर्ण $ B $ और $ C $.

\textbf{5.} प्रत्येक संबंध को विभाजित करना $ BC $:
\[
BH = \frac{AB^2}{BC},
\qquad
CH = \frac{AC^2}{BC} .
\]
इसलिए $ BH $ और $ CH $ हैं \omterm{def:g6:propdata:def}{आनुपातिक} को $ AB^2$ और $ AC^2$: पैर
ऊँचाई का कर्ण कर्ण को वर्गों के अनुपात में विभाजित करता है
पैरों की ($1.8 : 3.2 = 9 : 16$ प्रश्न 4 में)।

\textbf{6.} दो संबंधों को जोड़ना और गुणनखंड बनाना $ BC $
(वितरणशीलता):
\[
AB^2 + AC^2 = BH \times BC + CH \times BC
= (BH + CH) \times BC = BC \times BC = BC^2 .
\]
यह है \cref{thm:g8:pythagoras:direct}, कोसाइन से प्राप्त किया गया
अकेला --- चा\omterm{def:g3:shapes:shapes}{र-त्रिकोण} से वास्तव में भिन्न प्रमाण \omterm{def:g6:measure:area}{क्षेत्र}
की पहेली \cref{exo:g8:pythagoras:11}.

\textbf{7.} पाइथागोरस में $ ABH $ (\omterm{def:g6:shapes:triangles}{समकोण का} पर $ H $, कर्ण
$[AB]$): $ AH^2 = AB^2 - BH^2$. की जगह $ AB^2$ द्वारा
$ BH \times BC $ (प्रश्न 2) और फैक्टरिंग आउट $ BH $:
\[
AH^2 = BH \times BC - BH^2
= BH \times (BC - BH)
= BH \times HC .
\]

\textbf{8.} पैरों को आधार और ऊंचाई के रूप में रखते हुए, \omterm{def:g6:measure:area}{क्षेत्र} का $ ABC $
है $\frac{AB \times AC}{2}$; कर्ण के साथ $[BC]$ आधार के रूप में,
ऊंचाई बिलकुल है $ AH $, इतना \omterm{def:g6:measure:area}{क्षेत्र} ई आल्सो
$\frac{BC \times AH}{2}$ (\cref{thm:g7:areas:triangle}). बराबरी करना
और दोहरीकरण: $ AB \times AC = BC \times AH $.

\textbf{9.} $ AH = \frac{AB \times AC}{BC} = \frac{3 \times 4}{5}
= 2.4$. प्रश्न 7 का सत्यापन:
$ AH^2 = 2.4^2 = 5.76$ और
$ BH \times HC = 1.8 \times 3.2 = 5.76$. बराबर।

\textbf{10.} $ AH^2 = BH \times HC = 2 \times 8 = 16$, इसलिए
$ AH = \sqrt{16} = 4$ सेमी।

\textbf{11.} बात $ A $ के वृत्त पर स्थित है \omterm{def:g4:geometry:circle}{व्यास} $[BC]$
(और इससे भिन्न है $ B $ और $ C $), तो बिंदु 2 से
\cref{thm:g8:cosine:circle} \omterm{def:g3:shapes:shapes}{त्रिकोण} $ ABC $ है \omterm{def:g6:shapes:triangles}{समकोण का} पर
$ A $. निर्माण द्वारा $(AH) \perp (BC)$ साथ $ H $ बीच में $ B $ और
$ C $: द \omterm{def:g6:lines:objects}{खंड} $[AH]$ ठीक प्रश्न 7 की ऊंचाई है,
और $ AH^2 = BH \times HC = p \times q $.

\textbf{12.} $ AH^2 = 1 \times 2 = 2$: आकृति निर्माण करती है, साथ
शासक और कम्पास अकेले, ए \omterm{def:g6:lines:objects}{खंड} जिसका वर्ग है $2$ --- द
इकाई वर्ग का विकर्ण, $\sqrt2 \approx 1.414$, का
\cref{ex:g8:pythagoras:diagonal}.

\textbf{13.} विधि: एक सिरे से दूसरे सिरे तक रखें \omterm{def:g6:lines:objects}{खंड}ों लंबाई का
$ p = 1$ और $ q = n $; अर्ध-वृत्त खींचिए जिसका \omterm{def:g4:geometry:circle}{व्यास} है
संपूर्ण \omterm{def:g6:lines:objects}{खंड} (लंबाई $ n + 1$); बढ़ाएँ \omterm{def:g6:lines:perp}{सीधा} पर
जंक्शन बिंदु; यह आधे वृत्त से एक बिंदु पर मिलता है जिसका
जंक्शन की दूरी है $\sqrt{1 \times n} = \sqrt n $. के लिए
$\sqrt5$: लेना $ p = 1$ और $ q = 5$ (का \omterm{def:g3:division:half}{आधा} \omterm{def:g6:lines:circle}{घेरा} \omterm{def:g4:geometry:circle}{व्यास}
$6$).

\textbf{14.} $ A $ केंद्र के वृत्त पर है $ O $, इसलिए
$ OA = \frac{p+q}{2}$, द \omterm{def:g3:shapes:shapes}{RADIUS}. अगर $ H \neq O $, \omterm{def:g3:shapes:shapes}{त्रिकोण}
$ AHO $ है \omterm{def:g6:shapes:triangles}{समकोण का} पर $ H $ कर्ण के साथ $[OA]$, और एक पैर है
कर्ण से छोटा: $ AH < OA $. अगर $ H = O $, तब
$ AH = OA $ बिल्कुल। सभी मामलों में $ AH \leq \frac{p+q}{2}$, के साथ
समानता ठीक जब $ H $ केंद्र है --- अर्थात, कब
$ p = q $.

\textbf{15.} बराबरी $ AH \leq \frac{p+q}{2}$ (दोनों तरफ
सकारात्मक) और उपयोग कर रहे हैं $ AH^2 = pq $:
\[
p \times q \leq \left(\frac{p+q}{2}\right)^2 ,
\]
बिल्कुल समानता के साथ $ p = q $. बीजगणित पुनः प्रमाण, के साथ
उल्लेखनीय पहचान (\cref{pb:g8:equations:1}):
\[
\left(\frac{p+q}{2}\right)^2 - pq
= \frac{p^2 + 2pq + q^2 - 4pq}{4}
= \frac{p^2 - 2pq + q^2}{4}
= \left(\frac{p-q}{2}\right)^2 ,
\]
एक वर्ग, इसलिए कभी भी ऋणात्मक नहीं --- और \omterm{def:g1:counting:zero}{शून्य} बिल्कुल कब
$ p = q$. ज्यामितीय माध्य दो संख्याओं की संख्या कभी भी उनकी संख्या से अधिक नहीं होती
\omterm{def:g3:division:half}{आधा} योग.
\end{solution}
