\section*{अध्याय \ref{ch:g8:powers} --- पॉवर्स}

\begin{solution}{exo:g8:powers:1}
$2^6 = 64$; \quad $3^3 = 27$; \quad $10^5 = 100\,000$; \quad
$1^{100} = 1$; \quad $0.1^2 = 0.01$; \quad $6^0 = 1$.
\end{solution}

\begin{solution}{exo:g8:powers:2}
$(-3)^2 = 9$; \quad $-3^2 = -9$; \quad $(-1)^{15} = -1$; \quad
$(-5)^3 = -125$; \quad $\left(\frac23\right)^2 = \frac49$.
\end{solution}

\begin{solution}{exo:g8:powers:3}
$7^4 \times 7^5 = 7^9$; \quad
$\dfrac{2^9}{2^3} = 2^6$; \quad
$\left(10^3\right)^4 = 10^{12}$; \quad
$5^6 \times 5^{-2} = 5^4$; \quad
$3^4 \times 7^4 = 21^4$ (वही \omterm{def:g8:powers:def}{प्रतिपादक}: आधारों को गुणा करें)।
\end{solution}

\begin{solution}{exo:g8:powers:4}
$10^{-2} = 0.01$; \quad
$4 \times 10^3 = 4000$; \quad
$2.5 \times 10^{-4} = 0.00025$; \quad
$10^0 = 1$.
\end{solution}

\begin{solution}{exo:g8:powers:5}
$10^5$; \quad $10^{-1}$; \quad $10^{10}$; \quad $10^{-6}$.
\end{solution}

\begin{solution}{exo:g8:powers:6}
$\dfrac{10^7 \times 10^{-3}}{10^2} = \dfrac{10^4}{10^2} = 10^2 =
100$.

$\dfrac{2^5 \times 2^4}{2^6} = \dfrac{2^9}{2^6} = 2^3 = 8$.

$\left(2^2\right)^3 \times 2^{-4} = 2^6 \times 2^{-4} = 2^2 = 4$.
\end{solution}

\begin{solution}{exo:g8:powers:7}
$2^3 \times 2^4 = 2^{12}$: \emph{असत्य} --- घातांक जोड़ना:
$2^7$.

$5^2 + 5^3 = 5^5$: \emph{असत्य} --- के लिए कोई नियम नहीं रकम:
$25 + 125 = 150$, जबकि $5^5 = 3125$.

$(3^2)^4 = 3^8$: \emph{सत्य}.

$10^3 \times 10^3 = 100^3$: \emph{सत्य} --- दोनों बराबर $10^6$
(बाएं: $10^{3+3}$; सही: $(10^2)^3$).
\end{solution}

\begin{solution}{exo:g8:powers:8}
दिन में नए लोग $4$: हरेक $3^3 = 27$ दिन में लोगों को सूचना दी गयी
3 अन्य तीन को बताता है: $3^4 = 81$. दिन 4 के अंत में जानना:
$3 + 9 + 27 + 81 = 120$ लोग।
\end{solution}

\begin{solution}{exo:g8:powers:9}
\emph{1.} मोटाई: $2^{10} \times 10^{-4}$ एम
$= 1024 \times 10^{-4}$ एम $\approx 0.1$ एम $= 10$ सेमी।

\emph{2.} $2^{42} \times 10^{-4} \approx 4.4 \times 10^{12} \times
10^{-4} = 4.4 \times 10^8$ मी, से बड़ा $3.8 \times 10^8$ एम: के बाद
$42$ (सैद्धांतिक!) मुड़ा हुआ, कागज का बंडल चंद्रमा को पार कर जाएगा।
\end{solution}

\begin{solution}{exo:g8:powers:10}
$2^{10} = 1024$; $10^3 = 1000$; $3^6 = 729$; $5^4 = 625$. आदेश देना:
\[
5^4 < 3^6 < 10^3 < 2^{10} .
\]
\end{solution}

\begin{solution}{exo:g8:powers:11}
$2^{100} = \left(2^{10}\right)^{10} = 1024^{10}$ और
$10^{30} = \left(10^3\right)^{10} = 1000^{10}$. तब से $1024 > 1000$,
प्रत्येक की दस प्रतियों को गुणा करने से असमानता बनी रहती है:
$2^{100} > 10^{30}$.
\end{solution}

\begin{solution}{pb:g8:powers:1}
\textbf{1.} शुरू में वर्गाकार से वर्गाकार होने पर दाने दोगुने हो जाते हैं
$1 = 2^0$: वर्ग $ k $ रखती है $2^{k-1}$ अनाज
(\cref{def:g8:powers:def}). वर्ग $64$ रखती है $2^{63}$.

\textbf{2.} $ S_1 = 1$, $ S_2 = 1 + 2 = 3$, $ S_3 = 3 + 4 = 7$,
$ S_4 = 7 + 8 = 15$, $ S_5 = 15 + 16 = 31$: हमेशा से एक कम
की अगली शक्ति $2$ ($2$, $4$, $8$, $16$, $32$). अनुमान:
$ S_n = 2^n - 1$.

\textbf{3.} के प्रत्येक पद को दोगुना करना $ S_n $ प्रत्येक शक्ति को ऊपर की ओर स्थानांतरित करता है
एक ($2 \times 2^{k} = 2^{k+1}$):
\begin{align*}
S_n &= 1 + 2 + 4 + \dots + 2^{n-1}, \\
2 \times S_n &= \phantom{1 + {}} 2 + 4 + \dots + 2^{n-1} + 2^n .
\end{align*}
पहली पंक्ति को दूसरी से घटाकर, प्रत्येक पद को $2$
को $2^{n-1}$ दोनों में प्रकट होता है और रद्द हो जाता है:
\[
2 \times S_n - S_n = 2^n - 1,
\qquad\text{that is}\qquad
S_n = 2^n - 1 .
\]

\textbf{4.} कुल है $ S_{64} = 2^{64} - 1$ अनाज. ए
पैंसठवाँ वर्ग धारण करेगा $2^{64}$ अनाज: पूरा बोर्ड
उस एक वर्ग से ठीक एक दाना कम होता है।

\textbf{5.} पहला \omterm{def:g3:division:half}{आधा} रखती है $ S_{32} = 2^{32} - 1$ अनाज.
पूरा बोर्ड रखता है $2^{64} - 1$, तो दूसरा \omterm{def:g3:division:half}{आधा} रखती है
\[
\left(2^{64} - 1\right) - \left(2^{32} - 1\right)
= 2^{64} - 2^{32}
= 2^{32} \times \left(2^{32} - 1\right)
\]
(फैक्टर आउट $2^{32}$, का उपयोग करना $2^{32} \times 2^{32} = 2^{64}$,
\cref{thm:g8:powers:rules}): बिल्कुल $2^{32}$ पहले बार
\omterm{def:g3:division:half}{आधा}.

\textbf{6.} से \omterm{def:g8:powers:def}{प्रतिपादक} नियम, $2^{64} = 2^{4 + 60} =
2^4 \times \left(2^{10}\right)^6$. तब से $2^{10} = 1024 > 10^3$,
\[
2^{64} > 16 \times \left(10^3\right)^6 = 16 \times 10^{18}
= 1.6 \times 10^{19} .
\]

\textbf{7.} इससे अधिक $1.6 \times 10^{19}$ अनाज पर
$5 \times 10^{-2}$ जी प्रत्येक:
\[
1.6 \times 10^{19} \times 5 \times 10^{-2}
= 8 \times 10^{17} \text{ g} .
\]
द्वारा विभाजित करना $10^6$ जी प्रति टन: से अधिक $8 \times 10^{11}$
टन --- आठ सौ अरब टन।

\textbf{8.} $\dfrac{8 \times 10^{11}}{8 \times 10^{8}} = 10^3$:
राजा ने कम से कम एक का वादा किया \emph{हज़ार वर्ष} आज का
पूरी दुनिया की फसल. (किंवदंती कहती है कि उनके सलाहकारों ने उन्हें ऐसा बताया था
बहुत।)

\textbf{9.} $2^{19} = 2^9 \times 2^{10} = 512 \times 1024 =
524\,288 < 10^6$, जबकि $2^{20} = \left(2^{10}\right)^2 =
1024^2 = 1\,048\,576 > 10^6$. तो सबसे छोटा \omterm{def:g8:powers:def}{प्रतिपादक} है
$ n = 20$: बीस दोहरीकरण मिलियन को पार करते हैं।

\textbf{10.} दिन पर भुगतान $ n $ है $2^{n-1}$ सेंट (दिन 1:
$2^0 = 1$). अब
\[
2^{26} = 2^6 \times \left(2^{10}\right)^2
= 64 \times 1\,048\,576 = 67\,108\,864 < 10^8,
\]
\[
2^{27} = 2 \times 2^{26} = 134\,217\,728 > 10^8 .
\]
तो दैनिक वेतन पहले से अधिक हो जाता है $10^8$ सेंट जब $ n - 1 = 27$:
दिन पर $28$.

\textbf{11.} $21 = 16 + 4 + 1$: वजन $16$, $4$ और $1$ जी।
$27 = 16 + 8 + 2 + 1$: वजन $16$, $8$, $2$ और $1$ जी।

\textbf{12.} प्रश्न 3 से, वज़न $1, 2, 4, 8$ एक साथ
तौलना $ S_4 = 2^4 - 1 = 15$ जी। तो बिना $16$ जी वजन
व्यापारी पास नहीं हो सकता $15$ जी: का कोई लक्ष्य $16$ जी या अधिक होना चाहिए
इसका उपयोग करें. और का एक लक्ष्य $15$ g या उससे कम का इसका उपयोग नहीं करना चाहिए, क्योंकि
$16$ अकेले जी वजन पहले ही लक्ष्य से अधिक है। का चुनाव
इसलिए सबसे बड़ा वजन है \emph{मजबूर}. जो बचता है वह एक है
अधिकतम का लक्ष्य $15$ जी के साथ बनना है $1, 2, 4, 8$ --- और
वही तर्क दोहराता है: $8$ मजबूर किया जाता है (यदि \omterm{def:g3:division:remainder}{शेष} है तो उपयोग किया जाता है)।
लक्ष्य है $\geq 8$, अन्यथा अप्रयुक्त, क्योंकि $1 + 2 + 4 = 7$),
तब $4$ (क्योंकि $1 + 2 = 3$), तब $2$, तब $1$.

\textbf{13.} मजबूर विकल्प के बाद, \omterm{def:g3:division:remainder}{शेष} लक्ष्य
प्रत्येक चरण के बाद अधिकतम \omterm{def:g3:division:remainder}{शेष} भार का योग होता है,
तो यह प्रक्रिया समाप्त हो जाती है \omterm{def:g3:division:remainder}{शेष} $0$: से हर लक्ष्य $1$ को
$31$ पहुंच गया है. और चूँकि रास्ते में हर विकल्प मजबूर था,
वज़न का कोई अन्य चयन समान लक्ष्य तक नहीं पहुंच सकता:
से प्रत्येक संख्या का लेखन $1$ को $31$ के तौर पर \omterm{def:g1:addition:def}{जोड़} विशिष्ट का
की शक्तियां $2$ मौजूद है और है \emph{अद्वितीय}.

\textbf{14.} कुल छह वज़न $ S_6 = 2^6 - 1 = 63$ जी, और
वही जबरन-पसंद तर्क प्रत्येक लक्ष्य को कवर करता है $1$ को
$63$ जी। के लिए $45$: लक्ष्य है $\geq 32$, तो उपयोग करें $32$; अवशेष
$13 < 16$, तो छोड़ें $16$; उपयोग $8$ (अवशेष $5$), उपयोग $4$ (अवशेष
$1$), छोडना $2$, उपयोग $1$:
\[
45 = 32 + 8 + 4 + 1 = 2^5 + 2^3 + 2^2 + 2^0 .
\]

\textbf{15.} ए चुनना $0$ या ए $1$ प्रत्येक पर $64$ वर्ग
यह किन शक्तियों को चुनने के बराबर है $2^0, 2^1, \dots, 2^{63}$ को
ए में शामिल करें \omterm{def:g1:addition:def}{जोड़} ---बिल्कुल व्यापारी का वजन के साथ $64$
वजन. सबसे छोटी संग्रहणीय संख्या है $0$ (सभी चौकों पर
$0$); सबसे बड़ा है \omterm{def:g1:addition:def}{जोड़} का \emph{सभी} शक्तियाँ, जो है
राजा का कुल: $ S_{64} = 2^{64} - 1$ (भाग 1). द्वारा
जबरन-चयन तर्क, बीच की प्रत्येक पूर्ण संख्या तक पहुँच जाता है
बिल्कुल एक बार: ए $64$-बिट मशीन पूरी तरह से स्टोर करती है
से संख्याएँ $0$ को $2^{64} - 1$.
\end{solution}
