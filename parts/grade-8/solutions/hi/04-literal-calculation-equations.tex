\begin{solution}{exo:g8:equations:1}
$5(2x + 3) = 10x + 15$.

$-3(4x - 1) = -12x + 3$.

$2(3x - 2) + 4(x + 5) = 6x - 4 + 4x + 20 = 10x + 16$.
\end{solution}

\begin{solution}{exo:g8:equations:2}
$(x+2)(x+7) = x^2 + 7x + 2x + 14 = x^2 + 9x + 14$.

$(3x+1)(2x-5) = 6x^2 - 15x + 2x - 5 = 6x^2 - 13x - 5$.

$(x-4)(x-6) = x^2 - 6x - 4x + 24 = x^2 - 10x + 24$.
\end{solution}

\begin{solution}{exo:g8:equations:3}

$2 \times 3 + 1 = 7$ और $3 + 4 = 7$: हाँ। \quad
$3^2 = 9$ और $6 \times 3 - 9 = 9$: हाँ --- $ x = 3$ सब हल करता है

\end{solution}

\begin{solution}{exo:g8:equations:4}
$ x + 9 = 4$: $ x = -5$ (जाँच करना: $-5 + 9 = 4$).

$5x = -35$: $ x = -7$.

$\frac x3 = 8$: $ x = 24$.

$2x - 7 = 13$: $2x = 20$, $ x = 10$ (जाँच करना: $20 - 7 = 13$).

\begin{solution}{exo:g8:equations:5}
$6x + 5 = 2x + 25$: $4x = 20$, इसलिए $ x = 5$.

$9 - 4x = x - 11$: $20 = 5x $, इसलिए $ x = 4$.

$3(x + 2) = 2x + 11$: $3x + 6 = 2x + 11$, इसलिए $ x = 5$.
\end{solution}

\begin{solution}{exo:g8:equations:6}

$4(28 - 3) = 100$ और $5(14 + 6) = 100$.
\end{solution}

\begin{solution}{exo:g8:equations:8}
चौड़ाई $ x $, लंबाई $ x + 5$; \omterm{def:g6:measure:perimeter}{परिधि}
$2\bigl(x + (x + 5)\bigr) = 4x + 10 = 58$, इसलिए $4x = 48$ और $ x = 12$:
चौड़ाई $12$ सेमी, लंबाई $17$ सेमी (जाँचें:
$2 \times (12 + 17) = 58$).
\end{solution}

\begin{solution}{exo:g8:equations:9}
संख्याएँ हैं $ n - 1$, $ n $, $ n + 1$ साथ \omterm{def:g1:addition:def}{जोड़} $3n = 141$, इसलिए
$ n = 47$: संख्याएं हैं $46$, $47$, $48$.
\end{solution}

\begin{solution}{exo:g8:equations:10}
$3x + 4 > 19$: $3x > 15$, इसलिए $ x > 5$.

$7 - 2x \geq 1$: $-2x \geq -6$, से विभाजित करना $-2$ उलटा:
$ x \leq 3$.

$5(x-1) < 2x + 7$: $5x - 5 < 2x + 7$, इसलिए $3x < 12$ और $ x < 4$.
\end{solution}

\begin{solution}{exo:g8:equations:11}
बाद $ n $ महीनों, मिया के पास है $140 + 15n $ और टॉम $200 + 10n $. मिया नेतृत्व करती है
कब
\[
140 + 15n > 200 + 10n
\ \Longleftrightarrow\
5n > 60
\ \Longleftrightarrow\
n > 12 :
\]
से $13$ वें महीने पर.
\end{solution}

\begin{solution}{exo:g8:equations:12}
दोनों पक्षों को इससे गुणा करें $20$:
\[
5(x + 3) = 4(2x - 1)
\ \Longrightarrow\
5x + 15 = 8x - 4
\ \Longrightarrow\
19 = 3x
\ \Longrightarrow\
x = \frac{19}{3}.
\]
जाँच करना: $\frac{19/3 + 3}{4} = \frac{28/3}{4} = \frac{28}{12} =
\frac73$, और $\frac{2 \times 19/3 - 1}{5} = \frac{35/3}{5} =
\frac{35}{15} = \frac73$. समान: समाधान है $\frac{19}{3}$.
\end{solution}

\begin{solution}{pb:g8:equations:1}
\textbf{1.} दोहरा वितरण
(\cref{thm:g8:equations:expand}) साथ $ c = a $, $ d = b $:
\[
(a + b)(a + b) = a^2 + ab + ba + b^2 = a^2 + 2ab + b^2 ,
\]
तब से $ ab $ और $ ba $ वही हैं \omterm{def:g2:mult:def}{उत्पाद}.

\textbf{2.} वही विस्तार, संकेतों को ध्यान में रखते हुए:
\[
(a - b)(a - b) = a^2 - ab - ba + b^2 = a^2 - 2ab + b^2 ,
\]
\[
(a + b)(a - b) = a^2 - ab + ba - b^2 = a^2 - b^2 :
\]
इस बार क्रॉस-शर्तें $-ab $ और $+ba $ इसके बजाय रद्द करें
दोहरीकरण.

\textbf{3.} भुजा का वर्ग $ a + b $ चार टुकड़ों में काटता है: a
वर्ग $ a \times a $ (द $ a^2$), एक वर्ग $ b \times b $ (द
$ b^2$), और \emph{दो} आयत $ a \times b $ (एक साथ,
$2ab $). पहचान $(a+b)^2 = a^2 + 2ab + b^2$ कथन है
वह चार \omterm{def:g6:measure:area}{क्षेत्र}ों बड़े वर्ग को भरें---यह का चित्र है
दोनों पक्षों के साथ दोहरा वितरण एक ही तरह से विभाजित होता है।

\textbf{4.} पहचान के साथ:
\begin{align*}
101^2 &= (100 + 1)^2 = 10\,000 + 2 \times 100 + 1 = 10\,201, \\
99^2 &= (100 - 1)^2 = 10\,000 - 200 + 1 = 9\,801, \\
49 \times 51 &= (50 - 1)(50 + 1) = 50^2 - 1^2 = 2\,499 .
\end{align*}

\textbf{5.} के लिए $ a = 3$, $ b = 4$: $(3 + 4)^2 = 49$ लेकिन
$3^2 + 4^2 = 9 + 16 = 25$. ज़ो का सूत्र दोनों को भूल जाता है
$ a \times b $ चित्र के आयत --- लुप्त
$2ab = 24$, सचमुच $25 + 24 = 49$.

\textbf{6.} $1$; $1 + 3 = 4$; $1 + 3 + 5 = 9$;
$1 + 3 + 5 + 7 = 16$; $1 + 3 + 5 + 7 + 9 = 25$. परिणाम हैं
पूर्ण वर्ग $1^2, 2^2, 3^2, 4^2, 5^2$.

\textbf{7.}  \omterm{def:g3:numbers:evenodd}{विषम} संख्याएँ पर शुरू करें $1$ और के चरणों में ऊपर जाएँ
$2$: द $ n $-वें के बाद पहुंचा जाता है $ n - 1$ के चरण $2$ से
$1$, तो यह बराबर है $1 + 2(n - 1) = 2n - 1$. जाँच करना: $ n = 1, 2, 3$
देना $1, 3, 5$.

\textbf{8.} पहली पहचान से,
\[
(n + 1)^2 - n^2 = n^2 + 2n + 1 - n^2 = 2n + 1 :
\]
 \omterm{ex:g1:subtraction:difference}{अंतर} दो लगातार वर्गों के बीच बिल्कुल वही है
$(n+1)$-वां \omterm{def:g3:numbers:evenodd}{विषम} संख्या (प्रश्न 7 के साथ $ n + 1$ की जगह $ n $:
$2(n+1) - 1 = 2n + 1$).

\textbf{9.} एक के बाद एक चौकोर बनाएँ। ए से शुरू करें
एकल कोशिका: $1 = 1^2$. पक्ष के वर्ग से गुजरना $1$ को
भुजा का वर्ग $2$, जोड़ना $2 \times 1 + 1 = 3$ कोशिकाएं (प्रश्न
8): $1 + 3 = 2^2$. बगल में जाना $3$, जोड़ना $2 \times 2 + 1 = 5$
कोशिकाएँ: $1 + 3 + 5 = 3^2$. प्रत्येक नया \omterm{def:g3:numbers:evenodd}{विषम} संख्या बिलकुल यही है
एल-आकार की परत अगले वर्ग को पूरा करती है, इसलिए जोड़ने के बाद
प्रथम $ n $ \omterm{def:g3:numbers:evenodd}{विषम} संख्याएँ भुजा का वर्ग $ n $ तैयार है:
$1 + 3 + \dots + (2n - 1) = n^2$.

\textbf{10.}  \omterm{def:g3:numbers:evenodd}{विषम} संख्याएँ से $1$ को $99$ प्रथम हैं $50$
उनमें से ($2n - 1 = 99$ देता है $ n = 50$), इसलिए
\[
1 + 3 + 5 + \dots + 99 = 50^2 = 2\,500 .
\]
और $1 + 3 + \dots + (2n - 1) = 400 = 20^2$ आवश्यक है $ n = 20$:
पहला \emph{बीस} \omterm{def:g3:numbers:evenodd}{विषम} संख्याएँ ($1$ तक $39$) तक \omterm{def:g1:addition:def}{जोड़}ा गया
$400$.

\textbf{11.} $ x^2 + 6x + 9 = x^2 + 2 \times 3 \times x + 3^2 =
(x + 3)^2$: से पहली पहचान $ a = x $, $ b = 3$. इतना
\omterm{def:g8:equations:def}{समीकरण} पढ़ता $(x + 3)^2 = 0$. एक वर्ग है \omterm{def:g1:counting:zero}{शून्य} केवल जब
संख्या ही है \omterm{def:g1:counting:zero}{शून्य} (\cref{thm:g8:negprod:rules}: एक \omterm{def:g1:counting:zero}{शून्य}ेतर
संख्या में \omterm{def:g1:counting:zero}{शून्य}ेतर दूरी होती है, इसलिए इसके वर्ग में \omterm{def:g1:counting:zero}{शून्य}ेतर दूरी होती है
भी)। इस तरह $ x + 3 = 0$: एकमात्र समाधान है $ x = -3$.

\textbf{12.} द्वारा \cref{thm:g8:negprod:rules}, ए की दूरी
\omterm{def:g2:mult:def}{उत्पाद} है \omterm{def:g2:mult:def}{उत्पाद} दूरियों का. यदि दोनों कारक हैं
\omterm{def:g1:counting:zero}{शून्य}ेतर, दोनों दूरियाँ \omterm{def:g1:counting:zero}{शून्य}ेतर हैं, इसलिए उत्पादकी दूरी है
\omterm{def:g1:counting:zero}{शून्य}ेतर: \omterm{def:g2:mult:def}{उत्पाद} हो नहीं सकता $0$. ए \omterm{def:g1:counting:zero}{शून्य} \omterm{def:g2:mult:def}{उत्पाद} इसलिए
कम से कम एक को मजबूर करता है \omterm{def:g1:counting:zero}{शून्य} कारक। के लिए $(x - 2)(x + 5) = 0$: दोनों में से एक
$ x - 2 = 0$ या $ x + 5 = 0$, दो समाधान दे रहा हूँ $ x = 2$ और
$ x = -5$.

\textbf{13.} $ x^2 = 49$ बन जाता है $ x^2 - 49 = 0$, और तीसरा
पहचान कारक यह:
\[
x^2 - 49 = x^2 - 7^2 = (x - 7)(x + 7) = 0 .
\]
प्रश्न 12 द्वारा, $ x = 7$ या $ x = -7$: \emph{दो} समाधान. प्रत्येक
\omterm{def:g8:equations:def}{समीकरण} इस अध्याय में अब तक हल किया गया प्रथम श्रेणी का था और
बिल्कुल एक ही समाधान था; में एक वर्ग \omterm{def:g8:equations:def}{समीकरण} उत्पादन कर सकते हैं
दो.

\textbf{14.} सब कुछ बायीं ओर लाएँ:
$ x^2 - 6x + 9 = 0$, और दूसरी पहचान पहचानें:
$ x^2 - 2 \times 3 \times x + 3^2 = (x - 3)^2$. इतना \omterm{def:g8:equations:def}{समीकरण} है
$(x - 3)^2 = 0$, जो मजबूर करता है $ x - 3 = 0$ (जैसा कि प्रश्न 11 में है):
$ x = 3$ एक और एकमात्र समाधान है --- की जाँच
\cref{exo:g8:equations:3} इसे पाया, और फैक्टरिंग साबित होती है
कोई अन्य नहीं हैं.

\textbf{15.} तीसरी पहचान से,
\[
2026^2 - 2025^2 = (2026 - 2025)(2026 + 2025)
= 1 \times 4051 = 4\,051 .
\]
सामान्य तौर पर $(n+1)^2 - n^2 = (n + 1 - n)(n + 1 + n) = 2n + 1$: द
\omterm{ex:g1:subtraction:difference}{अंतर} दो लगातार वर्गों का है \omterm{def:g1:addition:def}{जोड़} दोनों में से
संख्याएँ --- वही $2n + 1$ जैसा कि प्रश्न 8 में है, अब से प्राप्त किया गया है
पहली की जगह तीसरी पहचान.
\end{solution}
