\begin{solution}{exo:g8:equations:1}
$5(2x + 3) = 10x + 15$.

$-3(4x - 1) = -12x + 3$.

$2(3x - 2) + 4(x + 5) = 6x - 4 + 4x + 20 = 10x + 16$.
\end{solution}

\begin{solution}{exo:g8:equations:2}
$(x+2)(x+7) = x^2 + 7x + 2x + 14 = x^2 + 9x + 14$.

$(3x+1)(2x-5) = 6x^2 - 15x + 2x - 5 = 6x^2 - 13x - 5$.

$(x-4)(x-6) = x^2 - 6x - 4x + 24 = x^2 - 10x + 24$.
\end{solution}

\begin{solution}{exo:g8:equations:10}
$3x + 4 > 19$: $3x > 15$, so $x > 5$.

$7 - 2x \geq 1$: $-2x \geq -6$, dividing by $-2$ reverses:
$x \leq 3$.

$5(x-1) < 2x + 7$: $5x - 5 < 2x + 7$, so $3x < 12$ and $x < 4$.
\end{solution}

\begin{solution}{exo:g8:equations:12}
Multiply both sides by $20$:
\[
5(x + 3) = 4(2x - 1)
\ \Longrightarrow\
5x + 15 = 8x - 4
\ \Longrightarrow\
19 = 3x
\ \Longrightarrow\
x = \frac{19}{3}.
\]
Check: $\frac{19/3 + 3}{4} = \frac{28/3}{4} = \frac{28}{12} =
\frac73$, and $\frac{2 \times 19/3 - 1}{5} = \frac{35/3}{5} =
\frac{35}{15} = \frac73$. Equal: the solution is $\frac{19}{3}$.
\end{solution}
