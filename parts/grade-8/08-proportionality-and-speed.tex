\chapter{Proportionality, Speed and Averages}\label{ch:g8:speed}

Proportionality (\cref{ch:g7:prop}) now meets the physical world:
speed, flow rates, unit conversions --- quantities that are
\emph{quotients} of two others. The chapter closes with weighted
averages, where proportional thinking prevents a classic mistake.

\section{Average speed}

\begin{definition}[Average speed]\label{def:g8:speed:def}
The \emph{average speed}\index{speed} of a journey is the quotient
\[
v = \frac{d}{t}
\qquad
\left(\text{distance divided by duration}\right),
\]
in kilometers per hour (km/h), meters per second (m/s), \dots\ The
formula turns around: $d = v \times t$ and $t = \frac{d}{v}$.
\end{definition}

\begin{example}\label{ex:g8:speed:basic}
A train covers $270$ km in $1$ h $30$ min. First convert the time:
$1$ h $30$ min $= 1.5$ h. Then
\[
v = \frac{270}{1.5} = 180 \text{ km/h}.
\]
Distance at $90$ km/h during $2$ h $20$ min $= \frac73$ h:
$d = 90 \times \frac73 = 210$ km. Time for $35$ km at $14$ km/h:
$t = \frac{35}{14} = 2.5$ h $= 2$ h $30$ min.
\end{example}

\begin{remark}[Minutes are not decimals]
$1$ h $30$ min is $1.5$ h, but $1$ h $20$ min is \emph{not} $1.2$ h:
it is $1 + \frac{20}{60} = \frac43 \approx 1.33$ h. Always convert
minutes to a fraction of an hour ($60$ min $= 1$ h) before dividing.
\end{remark}

\begin{method}[Converting speeds]\label{met:g8:speed:convert}
To convert km/h into m/s: $1$ km $= 1000$ m and $1$ h $= 3600$ s, so
\[
v \text{ km/h} = \frac{v \times 1000}{3600} \text{ m/s}
= \frac{v}{3.6} \text{ m/s}.
\]
To convert m/s into km/h, multiply by $3.6$.
\end{method}

\begin{example}\label{ex:g8:speed:convert}
$90$ km/h $= \frac{90}{3.6} = 25$ m/s. A sprinter running $100$ m in
$10$ s moves at $10$ m/s $= 36$ km/h.
\end{example}

\section{Quotient quantities}

\begin{example}[Flow, density, price per kilo]\label{ex:g8:speed:quotients}
Speed has many cousins, all treated the same way:
\begin{itemize}
  \item a tap fills $48$ L in $4$ min: flow rate
        $\frac{48}{4} = 12$ L/min, so filling a $150$ L tub takes
        $\frac{150}{12} = 12.5$ min;
  \item $0.6$ kg of cheese costs $9$ euros: price
        $\frac{9}{0.6} = 15$ euros per kg;
  \item a car uses $6.3$ L for $90$ km: consumption
        $\frac{6.3}{90} \times 100 = 7$ L per $100$ km.
\end{itemize}
Identify the quotient, and the three-way formula
($q = \frac ab$, $a = qb$, $b = \frac aq$) does the rest.
\end{example}

\section{Weighted averages}

\begin{definition}[Weighted average]\label{def:g8:speed:average}
When values $v_1, v_2, \dots$ come with counts (or weights)
$n_1, n_2, \dots$, their \emph{weighted average}\index{weighted mean}
is
\[
\bar v = \frac{n_1 v_1 + n_2 v_2 + \dots}{n_1 + n_2 + \dots} :
\]
total of the values, divided by total of the weights.
\end{definition}

\begin{example}\label{ex:g8:speed:average}
A test was taken by two groups: group 1, $12$ students, average
$11$; group 2, $18$ students, average $14$. Average of the whole
class:
\[
\bar v = \frac{12 \times 11 + 18 \times 14}{12 + 18}
= \frac{132 + 252}{30} = \frac{384}{30} = 12.8 .
\]
\emph{Not} $\frac{11 + 14}{2} = 12.5$: the larger group pulls the
average towards its own — the weights matter.
\end{example}

\begin{example}[Average speed over two legs]\label{ex:g8:speed:twolegs}
A cyclist rides $30$ km at $30$ km/h, then $30$ km at $15$ km/h. The
average speed over the whole trip is \emph{not}
$\frac{30 + 15}{2} = 22.5$ km/h. Compute with the definition:
\begin{enumerate}
  \item times: $\frac{30}{30} = 1$ h, then $\frac{30}{15} = 2$ h;
        total $3$ h;
  \item total distance $60$ km, so
        $v = \frac{60}{3} = 20$ km/h.
\end{enumerate}
The slow leg lasts longer, so it weighs more --- an average of speeds
must be weighted by \emph{time}.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g8:speed:1}
Compute the average speed: $150$ km in $2$ h; $27$ km in $45$ min;
$100$ m in $12.5$ s.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:2}
Compute the distance: $2$ h at $85$ km/h; $40$ min at $90$ km/h.
Compute the time: $315$ km at $90$ km/h (in h and min).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:3}
Convert: $72$ km/h into m/s; $5$ m/s into km/h; $108$ km/h into m/s.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:4}
A tap delivers $15$ L/min. How long to fill a $600$ L tank? How many
liters in $2$ h $30$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:5}
Which is the better buy: $1.2$ kg of apples for $3.30$ euros, or
$0.8$ kg for $2.16$ euros? Compare prices per kilogram.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:6}
A class of $25$ students has an average of $12.4$ on a test; another
class of $35$ students has an average of $10$. Compute the average of
the two classes together.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:7}
Sound travels at about $340$ m/s. You see a lightning flash and hear
the thunder $6$ seconds later. How far away did the lightning strike
(to the nearest $100$ m)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:speed:8}
A hiker walks $2$ h at $5$ km/h, rests $30$ min, then walks $1$ h $30$
at $4$ km/h. Compute the total distance and the average speed
\emph{including the rest} (total distance over total elapsed time).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:speed:9}
Marks with coefficients: a student got $15$ (coefficient $3$), $9$
(coefficient $2$) and $12$ (coefficient $1$). Compute the weighted
average. What plain (unweighted) average would she have, and why do
they differ?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:speed:10}
A car drives $120$ km at $80$ km/h and then $120$ km at $120$ km/h.
\begin{enumerate}
  \item Compute the duration of each leg, then the average speed over
        the whole trip.
  \item Explain why the answer is less than $100$ km/h, the midpoint
        of the two speeds.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:speed:11}
Two villages are $18$ km apart. Anna leaves the first at $10{:}00$
walking at $5$ km/h towards the second; Boris leaves the second at
the same time walking at $4$ km/h towards the first. At what time do
they meet, and at what distance from Anna's village? (Together they
close the gap at $5 + 4$ km/h.)
\end{exercise}
