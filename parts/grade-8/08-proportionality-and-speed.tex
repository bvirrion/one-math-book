\chapter{Proportionality, Speed and Averages}\label{ch:g8:speed}

\omterm{def:g7:prop:table}{Proportionality} (\cref{ch:g7:prop}) now meets the physical world:
\omterm{def:g8:speed:def}{speed}, flow rates, unit conversions --- quantities that are
\emph{\omterm{def:g3:division:remainder}{quotients}} of two others. The chapter closes with \omterm{def:g8:speed:average}{weighted
averages}, where \omterm{def:g6:propdata:def}{proportional} thinking prevents a classic mistake.

\section{Average speed}

\begin{definition}[Average speed]\label{def:g8:speed:def}
The \emph{average speed}\index{speed} of a journey is the \omterm{def:g3:division:remainder}{quotient}
\[
v = \frac{d}{t}
\qquad
\left(\text{distance divided by duration}\right),
\]
in kilometers per hour (km/h), meters per second (m/s), \dots\ The
formula turns around: $d = v \times t$ and $t = \frac{d}{v}$.
\end{definition}

\begin{example}\label{ex:g8:speed:basic}
A train covers $270$ km in $1$ h $30$ min. First convert the time:
$1$ h $30$ min $= 1.5$ h. Then
\[
v = \frac{270}{1.5} = 180 \text{ km/h}.
\]
Distance at $90$ km/h during $2$ h $20$ min $= \frac73$ h:
$d = 90 \times \frac73 = 210$ km. Time for $35$ km at $14$ km/h:
$t = \frac{35}{14} = 2.5$ h $= 2$ h $30$ min.
\end{example}

\begin{remark}[Minutes are not decimals]
$1$ h $30$ min is $1.5$ h, but $1$ h $20$ min is \emph{not} $1.2$ h:
it is $1 + \frac{20}{60} = \frac43 \approx 1.33$ h. Always convert
minutes to a \omterm{def:g6:fractions:def}{fraction} of an hour ($60$ min $= 1$ h) before dividing.
\end{remark}

\begin{method}[Converting speeds]\label{met:g8:speed:convert}
To convert km/h into m/s: $1$ km $= 1000$ m and $1$ h $= 3600$ s, so
\[
v \text{ km/h} = \frac{v \times 1000}{3600} \text{ m/s}
= \frac{v}{3.6} \text{ m/s}.
\]
To convert m/s into km/h, multiply by $3.6$.
\end{method}

\begin{example}\label{ex:g8:speed:convert}
$90$ km/h $= \frac{90}{3.6} = 25$ m/s. A sprinter running $100$ m in
$10$ s moves at $10$ m/s $= 36$ km/h.
\end{example}

\section{Quotient quantities}

\begin{example}[Flow, density, price per kilo]\label{ex:g8:speed:quotients}
\omterm{def:g8:speed:def}{Speed} has many cousins, all treated the same way:
\begin{itemize}
  \item a tap fills $48$ L in $4$ min: flow rate
        $\frac{48}{4} = 12$ L/min, so filling a $150$ L tub takes
        $\frac{150}{12} = 12.5$ min;
  \item $0.6$ kg of cheese costs $9$ euros: price
        $\frac{9}{0.6} = 15$ euros per kg;
  \item a car uses $6.3$ L for $90$ km: consumption
        $\frac{6.3}{90} \times 100 = 7$ L per $100$ km.
\end{itemize}
Identify the \omterm{def:g3:division:remainder}{quotient}, and the three-way formula
($q = \frac ab$, $a = qb$, $b = \frac aq$) does the rest.
\end{example}

\section{Weighted averages}

\begin{definition}[Weighted average]\label{def:g8:speed:average}
When values $v_1, v_2, \dots$ come with counts (or weights)
$n_1, n_2, \dots$, their \emph{weighted average}\index{weighted mean}
is
\[
\bar v = \frac{n_1 v_1 + n_2 v_2 + \dots}{n_1 + n_2 + \dots} :
\]
total of the values, divided by total of the weights.
\end{definition}

\begin{example}\label{ex:g8:speed:average}
A test was taken by two groups: group 1, $12$ students, average
$11$; group 2, $18$ students, average $14$. Average of the whole
class:
\[
\bar v = \frac{12 \times 11 + 18 \times 14}{12 + 18}
= \frac{132 + 252}{30} = \frac{384}{30} = 12.8 .
\]
\emph{Not} $\frac{11 + 14}{2} = 12.5$: the larger group pulls the
average towards its own — the weights matter.
\end{example}

\begin{example}[Average speed over two legs]\label{ex:g8:speed:twolegs}
A cyclist rides $30$ km at $30$ km/h, then $30$ km at $15$ km/h. The
\omterm{def:g8:speed:def}{average speed} over the whole trip is \emph{not}
$\frac{30 + 15}{2} = 22.5$ km/h. Compute with the definition:
\begin{enumerate}
  \item times: $\frac{30}{30} = 1$ h, then $\frac{30}{15} = 2$ h;
        total $3$ h;
  \item total distance $60$ km, so
        $v = \frac{60}{3} = 20$ km/h.
\end{enumerate}
The slow leg lasts longer, so it weighs more --- an average of \omterm{def:g8:speed:def}{speeds}
must be weighted by \emph{time}.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g8:speed:1}
Compute the \omterm{def:g8:speed:def}{average speed}: $150$ km in $2$ h; $27$ km in $45$ min;
$100$ m in $12.5$ s.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:2}
Compute the distance: $2$ h at $85$ km/h; $40$ min at $90$ km/h.
Compute the time: $315$ km at $90$ km/h (in h and min).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:3}
Convert: $72$ km/h into m/s; $5$ m/s into km/h; $108$ km/h into m/s.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:4}
A tap delivers $15$ L/min. How long to fill a $600$ L tank? How many
liters in $2$ h $30$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:5}
Which is the better buy: $1.2$ kg of apples for $3.30$ euros, or
$0.8$ kg for $2.16$ euros? Compare prices per kilogram.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:6}
A class of $25$ students has an average of $12.4$ on a test; another
class of $35$ students has an average of $10$. Compute the average of
the two classes together.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:7}
Sound travels at about $340$ m/s. You see a lightning flash and hear
the thunder $6$ seconds later. How far away did the lightning strike
(to the nearest $100$ m)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:speed:8}
A hiker walks $2$ h at $5$ km/h, rests $30$ min, then walks $1$ h $30$
at $4$ km/h. Compute the total distance and the \omterm{def:g8:speed:def}{average speed}
\emph{including the rest} (total distance over total elapsed time).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:speed:9}
Marks with coefficients: a student got $15$ (coefficient $3$), $9$
(coefficient $2$) and $12$ (coefficient $1$). Compute the \omterm{def:g8:speed:average}{weighted
average}. What plain (unweighted) average would she have, and why do
they differ?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:speed:10}
A car drives $120$ km at $80$ km/h and then $120$ km at $120$ km/h.
\begin{enumerate}
  \item Compute the duration of each leg, then the \omterm{def:g8:speed:def}{average speed} over
        the whole trip.
  \item Explain why the answer is less than $100$ km/h, the midpoint
        of the two \omterm{def:g8:speed:def}{speeds}.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:speed:11}
Two villages are $18$ km apart. Anna leaves the first at $10{:}00$
walking at $5$ km/h towards the second; Boris leaves the second at
the same time walking at $4$ km/h towards the first. At what time do
they meet, and at what distance from Anna's village? (Together they
close the gap at $5 + 4$ km/h.)
\end{exercise}

\section{Problem: The harmonic mean, or why the return trip ruins
the average}

\begin{problem}[{Weekend problem --- the \omterm{def:g8:speed:def}{average speed} over equal
distances is $\frac{2 v_1 v_2}{v_1 + v_2}$, and it never beats the
midpoint of the \omterm{def:g8:speed:def}{speeds}}]
\label{pb:g8:speed:1}
\Cref{ex:g8:speed:twolegs} and \cref{exo:g8:speed:10} both ended
with the same surprise: ride out at one \omterm{def:g8:speed:def}{speed}, ride back at
another, and the \omterm{def:g8:speed:def}{average speed} lands \emph{below} the midpoint of
the two. This problem finds the exact formula behind the surprise
--- a new kind of average, the \emph{harmonic
mean}\index{harmonic mean} --- proves that the surprise is a
theorem, and pushes it to its logical extreme: a catch-up that no
\omterm{def:g8:speed:def}{speed} in the world can achieve.

\medskip
\noindent\textbf{Part I --- The formula for a round trip.}
A journey covers a distance $d$ at \omterm{def:g8:speed:def}{speed} $v_1$, then the same
distance $d$ again at \omterm{def:g8:speed:def}{speed} $v_2$ (all positive numbers).
\begin{enumerate}
  \item Express the two durations, then the total duration, as
        \omterm{def:g6:fractions:def}{fractions} involving $d$, $v_1$, $v_2$.
  \item The \omterm{def:g8:speed:def}{average speed} of the whole journey is
        $\bar v = \dfrac{2d}{\ \dfrac{d}{v_1} + \dfrac{d}{v_2}\ }$
        --- a double-decker \omterm{def:g6:fractions:def}{fraction}
        (\cref{ex:g8:fractions:complex}). Simplify it and prove:
        \[
        \bar v = \frac{2\, v_1 v_2}{v_1 + v_2} .
        \]
  \item Check the formula against both computations quoted above:
        $v_1 = 30$, $v_2 = 15$ km/h
        (\cref{ex:g8:speed:twolegs}), and $v_1 = 80$,
        $v_2 = 120$ km/h (\cref{exo:g8:speed:10}).
  \item The distance $d$ has disappeared from the formula. What
        does that mean, concretely, for the cyclist?
  \item Now suppose the journey instead spends the same
        \emph{time} $T$ at each \omterm{def:g8:speed:def}{speed}. Show that the \omterm{def:g8:speed:def}{average
        speed} is then $\frac{v_1 + v_2}{2}$, the plain midpoint.
        In the language of \cref{def:g8:speed:average}: by what
        quantity must an average of \omterm{def:g8:speed:def}{speeds} always be weighted,
        and what changes between the two scenarios?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Harmonic against arithmetic.}
For two positive numbers $v_1$, $v_2$, set
\[
H = \frac{2\, v_1 v_2}{v_1 + v_2}
\quad\text{(their \emph{harmonic mean}),}
\qquad
A = \frac{v_1 + v_2}{2}
\quad\text{(their \emph{arithmetic mean}).}
\]
\begin{enumerate}[resume]
  \item Compute $H$ and $A$ for the pairs $(30, 15)$ and
        $(80, 120)$. Which of the two means wins each time?
  \item Put $A - H$ over the common \omterm{def:g4:fractions:def}{denominator} $2(v_1 + v_2)$,
        expand the \omterm{def:g4:fractions:def}{numerator} with the remarkable identities
        (\cref{pb:g8:equations:1}), and prove:
        \[
        A - H = \frac{(v_1 - v_2)^2}{2\,(v_1 + v_2)} .
        \]
  \item Deduce the theorem behind all the surprises: for positive
        \omterm{def:g8:speed:def}{speeds}, $H \leq A$ always, with equality exactly when
        $v_1 = v_2$. Why does a square in the \omterm{def:g4:fractions:def}{numerator} settle
        the matter?
  \item Prove the elegant identity $H \times A = v_1 \times v_2$:
        the two means multiply back to the original \omterm{def:g2:mult:def}{product}.
  \item A car covers $60$ km at $40$ km/h and $60$ km at
        $60$ km/h. Predict the \omterm{def:g8:speed:def}{average speed} with the formula,
        then confirm it the long way, with times and total
        distance.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The impossible catch-up.}
\begin{enumerate}[resume]
  \item A cyclist enters a $30$ km race hoping to average
        $30$ km/h. What total time may the race take her, at
        most? She rides the first $15$ km at $15$ km/h: how much
        of that time budget is left?
  \item Show that the hoped-for average has become literally
        impossible: compute her \omterm{def:g8:speed:def}{average speed} if she rides the
        second \omterm{def:g3:division:half}{half} at $90$ km/h, and explain why \omterm{def:g3:numbers:evenodd}{even} an
        arbitrarily enormous \omterm{def:g8:speed:def}{speed} leaves her short of
        $30$ km/h.
  \item She lowers her goal to $20$ km/h. What \omterm{def:g8:speed:def}{speed} on the
        second $15$ km achieves it? Check your answer with the
        harmonic-mean formula.
  \item A cold tap alone fills a tub in $20$ min; a hot tap alone
        fills it in $30$ min. What \omterm{def:g6:fractions:def}{fraction} of the tub do both
        taps together fill per minute, and how long do they take?
        Compare your answer with $H(20, 30)$ --- what do you
        notice?
  \item A plane flies the same route out and back: $900$ km/h
        with the wind, $700$ km/h against it. Compute the
        round-trip \omterm{def:g8:speed:def}{average speed}, and explain in one sentence why
        a wind, however it blows, always \emph{lengthens} a round
        trip.
\end{enumerate}
\end{problem}
