\chapter{आनुपातिकता, गति और औसत}\label{ch:g8:speed}

समानता (\cref{ch:g7:prop}) अब भौतिक संसार से मिलता है:
रफ़्तार, प्रवाह दर, इकाई रूपांतरण --- मात्राएँ जो हैं

\section{औसत गति}

\begin{definition}[औसत गति]\label{def:g8:speed:def}
 \emph{औसत गति}\index{औसत गति} एक यात्रा का है \omterm{def:g3:division:remainder}{भागफल}
\[
v = \frac{d}{t}
\qquad
\left(\text{distance divided by duration}\right),
\]
किलो\omterm{def:g4:fractions:def}{मीटर} प्रति घंटा (किमी/घंटा), \omterm{def:g4:fractions:def}{मीटर} प्रति सेकंड (एम/एस) में, \dots\ द
सूत्र घूमता है: $d = v \times t $ और $ t = \frac{d}{v}$.
\end{definition}

\begin{example}\label{ex:g8:speed:basic}
एक ट्रेन कवर करती है $270$ किमी में $1$ एच $30$ मि. सबसे पहले समय परिवर्तित करें:
$1$ एच $30$ मिन $= 1.5$ एच। तब
\[
v = \frac{270}{1.5} = 180 \text{ km/h}.
\]
दूरी पर $90$ किमी/घंटा के दौरान $2$ एच $20$ मिन $= \frac73$ एच:
$ d = 90 \times \frac73 = 210$ किमी. के लिए समय $35$ किमी पर $14$ किमी/घंटा:
$ t = \frac{35}{14} = 2.5$ एच $= 2$ एच $30$ मि.
\end{example}

\begin{remark}[मिनट दशमलव नहीं हैं]
$1$ एच $30$ मिनट है $1.5$ ज, लेकिन $1$ एच $20$ मिनट है \emph{नहीं} $1.2$ ज:
यह है $1 + \frac{20}{60} = \frac43 \approx 1.33$ एच। हमेशा परिवर्तित करें
मिनट से ए \omterm{def:g6:fractions:def}{अंश} एक घंटे का ($60$ मिन $= 1$ ज) विभाजित करने से पहले।
\end{remark}

\begin{method}[परिवर्तित करने की गति]\label{met:g8:speed:convert}
किमी/घंटा को मी/सेकेंड में बदलने के लिए: $1$ किमी $= 1000$ मी और $1$ एच $= 3600$ एस, तो
\[
v \text{ km/h} = \frac{v \times 1000}{3600} \text{ m/s}
= \frac{v}{3.6} \text{ m/s}.
\]
m/s को किमी/घंटा में बदलने के लिए, से गुणा करें $3.6$.
\end{method}

\begin{example}\label{ex:g8:speed:convert}
$90$ किमी/घंटा $= \frac{90}{3.6} = 25$ एमएस। एक धावक दौड़ रहा है $100$ में
$10$ पर चलता है $10$ एमएस $= 36$ किमी/घंटा.
\end{example}

\section{भागफल मात्राएँ}

\begin{example}[प्रवाह, घनत्व, कीमत प्रति किलो]\label{ex:g8:speed:quotients}
रफ़्तार उसके कई चचेरे भाई-बहन हैं, सभी के साथ एक जैसा व्यवहार किया जाता है:
\begin{itemize}
  \item एक नल भर जाता है $48$ एल में $4$ न्यूनतम: प्रवाह दर
        $\frac{48}{4} = 12$ एल/मिनट, इसलिए ए भरना $150$ एल टब लेता है
        $\frac{150}{12} = 12.5$ मिनट;
  \item $0.6$ किलो पनीर की कीमत $9$ यूरो: कीमत
        $\frac{9}{0.6} = 15$ यूरो प्रति किलो;
  \item एक कार का उपयोग करता है $6.3$ एल के लिए $90$ किमी: खपत
        $\frac{6.3}{90} \times 100 = 7$ एल प्रति $100$ किमी.
\end{itemize}
की पहचान करें \omterm{def:g3:division:remainder}{भागफल}, और तीन-तरफ़ा सूत्र
($ q = \frac ab $, $ a = qb $, $ b = \frac aq $) बाकी करता है.
\end{example}

\section{भारित औसत}

\begin{definition}[भारित औसत]\label{def:g8:speed:average}
जब मान $ v_1, v_2, \dots $ गिनती (या वज़न) के साथ आएं
$ n_1, n_2, \dots $, उनका \emph{भारित औसत}\index{भारित औसत}
है
\[
\bar v = \frac{n_1 v_1 + n_2 v_2 + \dots}{n_1 + n_2 + \dots} :
\]
कुल मानों को भारों के योग से विभाजित किया जाता है।
\end{definition}

\begin{example}\label{ex:g8:speed:average}
एक परीक्षण दो समूहों द्वारा लिया गया: समूह 1, $12$ छात्र, औसत
$11$; समूह 2, $18$ छात्र, औसत $14$. कुल का औसत
कक्षा:
\[
\bar v = \frac{12 \times 11 + 18 \times 14}{12 + 18}
= \frac{132 + 252}{30} = \frac{384}{30} = 12.8 .
\]
\emph{नहीं} $\frac{11 + 14}{2} = 12.5$: बड़ा समूह खींचता है
अपने आप में औसत - वजन मायने रखता है।
\end{example}

\begin{example}[दो पैरों पर औसत गति]\label{ex:g8:speed:twolegs}
एक साइकिल चालक सवारी करता है $30$ किमी पर $30$ किमी/घंटा, तो $30$ किमी पर $15$ किमी/घंटा.
\omterm{def:g8:speed:def}{औसत गति} पूरी यात्रा पर है \emph{नहीं}
$\frac{30 + 15}{2} = 22.5$ किमी/घंटा. परिभाषा के साथ गणना करें:
\begin{enumerate}
  \item समय: $\frac{30}{30} = 1$ ज, फिर $\frac{30}{15} = 2$ एच;
        कुल $3$ एच;
  \item कुल दूरी $60$ किमी, तो
        $ v = \frac{60}{3} = 20$ किमी/घंटा.
\end{enumerate}
धीमा पैर लंबे समय तक चलता है, इसलिए इसका वजन अधिक होता है --- औसत स्पीड
द्वारा भारित किया जाना चाहिए \emph{समय}.
\end{example}

\begin{exercise}[$\star\star $]\label{exo:g8:speed:8}

पर $4$ किमी/घंटा. कुल दूरी की गणना करें और \omterm{def:g8:speed:def}{औसत गति}
\emph{बाकियों सहित} (कुल बीते समय में कुल दूरी)।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:speed:9}
गुणांक के साथ अंक: एक छात्र को मिला $15$ (गुणांक $3$), $9$

वे भिन्न हैं?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:speed:10}
एक कार चलती है $120$ किमी पर $80$ किमी/घंटा और फिर $120$ किमी पर $120$ किमी/घंटा.
\begin{enumerate}
  \item प्रत्येक चरण की अवधि की गणना करें, फिर \omterm{def:g8:speed:def}{औसत गति} खत्म
        पूरी यात्रा.
  \item बताएं कि उत्तर इससे कम क्यों है $100$ किमी/घंटा, मध्य बिंदु
        दोनों में से स्पीड.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g8:speed:11}
दो गांव हैं $18$ किमी की दूरी. अन्ना सबसे पहले निकलते हैं $10{:}00$

\end{exercise}

\section{समस्या: हार्मोनिक माध्य, या वापसी यात्रा क्यों बर्बाद हो जाती है
औसत}

\begin{problem}[{सप्ताहांत समस्या --- \omterm{def:g8:speed:def}{औसत गति} बराबर से अधिक दूरियाँ है $\frac{2 v_1 v_2}{v_1 + v_2}$, और यह कभी भी मात नहीं देता का मध्यबिंदु स्पीड}]
\label{pb:g8:speed:1}
\Cref{ex:g8:speed:twolegs} और \cref{exo:g8:speed:10} दोनों ख़त्म हो गए
उसी आश्चर्य के साथ: एक ही बार में बाहर निकलें रफ़्तार, वापस सवारी करें
दूसरा, और \omterm{def:g8:speed:def}{औसत गति} भूमि \emph{नीचे} का मध्यबिंदु
दो. यह समस्या आश्चर्य के पीछे का सटीक सूत्र ढूंढती है
--- एक नए प्रकार का औसत, द \emph{अनुकूल माध्य}\index{अनुकूल माध्य} --- सिद्ध करता है कि आश्चर्य एक है
प्रमेय, और इसे इसके तार्किक चरम तक पहुंचाता है: एक कैच-अप कि नहीं
रफ़्तार दुनिया में हासिल कर सकते हैं.

\medskip
\noindent\textbf{Part I --- The formula for a round trip.}
एक यात्रा एक दूरी तय करती है $ d $ पर रफ़्तार $ v_1$, फिर वही
दूरी $ d $ फिर से रफ़्तार $ v_2$ (सभी सकारात्मक संख्याएँ)।
\begin{enumerate}
  \item दो अवधियों को, फिर कुल अवधि को, इस प्रकार व्यक्त करें
        \omterm{def:g6:fractions:def}{अंश}ों को शामिल $ d $, $ v_1$, $ v_2$.
  \item  \omterm{def:g8:speed:def}{औसत गति} पूरी यात्रा का है
        $\bar v = \dfrac{2d}{\ \dfrac{d}{v_1} + \dfrac{d}{v_2}\ }$
        --- एक डबल डेकर \omterm{def:g6:fractions:def}{अंश}
        (\cref{ex:g8:fractions:complex}). इसे सरल कीजिए और सिद्ध कीजिए:
        \[
        \bar v = \frac{2\, v_1 v_2}{v_1 + v_2} .
        \]
  \item ऊपर उद्धृत दोनों गणनाओं के विरुद्ध सूत्र की जाँच करें:
        $ v_1 = 30$, $ v_2 = 15$ किमी/घंटा
        (\cref{ex:g8:speed:twolegs}), और $ v_1 = 80$,
        $ v_2 = 120$ किमी/घंटा (\cref{exo:g8:speed:10}).
  \item दूरी $ d $ सूत्र से गायब हो गया है. क्या
        क्या इसका मतलब, \omterm{def:g5:solids:def}{ठोस} रूप से, साइकिल चालक के लिए है?
  \item अब मान लीजिए कि यात्रा के बदले उतना ही खर्च होता है
        \emph{समय} $ T $ हरेक पर रफ़्तार. दिखाओ कि \omterm{def:g8:speed:def}{औसत गति} तब है $\frac{v_1 + v_2}{2}$, सादा मध्यबिंदु।
        की भाषा में \cref{def:g8:speed:average}: किससे
        मात्रा का औसत होना चाहिए स्पीड हमेशा वज़नदार रहो,
        और दोनों परिदृश्यों के बीच क्या परिवर्तन होता है?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Harmonic against arithmetic.}
दो सकारात्मक संख्याओं के लिए $ v_1$, $ v_2$, तय करना
\[
H = \frac{2\, v_1 v_2}{v_1 + v_2}
\quad\text{(their \emph{अनुकूल माध्य}),}
\qquad
A = \frac{v_1 + v_2}{2}
\quad\text{(their \emph{अंकगणित औसत}).}
\]
\begin{enumerate}[resume]
  \item गणना करना $ H $ और $ A $ \omterm{def:g1:addition:def}{जोड़}ियों के लिए $(30, 15)$ और
        $(80, 120)$. दोनों में से कौन सा साधन हर बार जीतता है?
  \item रखना $ A - H $ आम से ऊपर \omterm{def:g5:division:multiple}{भाजक} $2(v_1 + v_2)$,
        इसका विस्तार करें \omterm{def:g4:fractions:def}{मीटर} उल्लेखनीय पहचान के साथ
        (\cref{pb:g8:equations:1}), और साबित करें:
        \[
        A - H = \frac{(v_1 - v_2)^2}{2\,(v_1 + v_2)} .
        \]
  \item सभी आश्चर्यों के पीछे का प्रमेय निकालें: सकारात्मक के लिए
        स्पीड, $ H \leq A $ सदैव, समानता के साथ बिल्कुल जब भी
        $ v_1 = v_2$. में एक वर्ग क्यों होता है \omterm{def:g4:fractions:def}{मीटर} निपटान
        बात?
  \item सुरुचिपूर्ण पहचान सिद्ध करें $ H \times A = v_1 \times v_2$:
        दोनों का मतलब मूल से गुणा करना है \omterm{def:g2:mult:def}{उत्पाद}.
  \item एक कार कवर करती है $60$ किमी पर $40$ किमी/घंटा और $60$ किमी पर
        $60$ किमी/घंटा. भविष्यवाणी करें \omterm{def:g8:speed:def}{औसत गति} सूत्र के साथ,
        फिर समय और कुल के साथ इसकी पुष्टि करें
        दूरी.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The impossible catch-up.}
\begin{enumerate}[resume]
  \item एक साइकिल चालक प्रवेश करता है $30$ किमी दौड़ औसत की उम्मीद है
        $30$ किमी/घंटा. दौड़ में उसे कुल कितना समय लगेगा?
        सबसे ज्यादा? वह सबसे पहले सवारी करती है $15$ किमी पर $15$ किमी/घंटा: कितना
        उस समय का बजट बचा है?
  \item दिखाएँ कि अपेक्षित औसत वस्तुतः बन गया है
        असंभव: उसकी गणना करें \omterm{def:g8:speed:def}{औसत गति} अगर वह सवारी करती है
        दूसरा \omterm{def:g3:division:half}{आधा} पर $90$ किमी/घंटा, और समझाएं क्यों \omterm{def:g3:numbers:evenodd}{यहां तक की} एक
        मनमाने ढंग से विशाल रफ़्तार उसे कम छोड़ देता है
        $30$ किमी/घंटा.
  \item वह अपना लक्ष्य कम कर देती है $20$ किमी/घंटा. क्या रफ़्तार दूसरे पर $15$ किमी इसे प्राप्त करता है? अपने उत्तर की जाँच करें
        हार्मोनिक-माध्य सूत्र.
  \item केवल एक ठंडा नल ही एक टब को भर देता है $20$ मिनट; अकेले एक गर्म नल
        इसे भरता है $30$ मि. क्या \omterm{def:g6:fractions:def}{अंश} टब के दोनों करो
        नल एक साथ प्रति मिनट भरते हैं, और उन्हें कितना समय लगता है?
        अपने उत्तर की तुलना करें $ H(20, 30)$ --- आपको क्या लगा?
  \item एक विमान बाहर और वापस एक ही मार्ग से उड़ान भरता है: $900$ किमी/घंटा
        हवा के साथ, $700$ इसके विपरीत किमी/घंटा. की गणना करें
        \omterm{def:g4:numbers:round}{गोल} यात्रा \omterm{def:g8:speed:def}{औसत गति}, और एक वाक्य में समझाएं कि क्यों
        हवा, चाहे जैसी भी चले, हमेशा \emph{लंबा} एक दौर
        यात्रा.
\end{enumerate}
\end{problem}
