\chapter{ऋणात्मक संख्याओं को गुणा करना}\label{ch:g8:negprod}

ग्रेड 7 \omterm{def:g1:addition:def}{जोड़}ा और घटाया गया सापेक्ष संख्याएँ
(\cref{ch:g7:negatives}); उन्हें गुणा करना और विभाजित करना बाकी है। एक
प्रश्न अध्याय पर हावी है: ``माइनस टाइम्स माइनस'' क्यों होना चाहिए
प्लस? हम सिर्फ एक नियम नहीं बल्कि एक कारण बताते हैं।

\section{संकेत नियम}

\begin{theorem}[किसी उत्पाद का चिह्न]\label{thm:g8:negprod:rules}
 \omterm{def:g2:mult:def}{उत्पाद} दोनों में से सापेक्ष संख्याएँ की दूरी है \omterm{def:g1:counting:zero}{शून्य} के बराबर
\omterm{def:g2:mult:def}{उत्पाद} दूरियों का, और इसका चिह्न इस प्रकार दिया गया है:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|cc}
$\times $ & $+$ & $-$ \\
\hline
$+$ & $+$ & $-$ \\
$-$ & $-$ & $+$ \\
\end{tabular}
\end{center}
\emph{वही संकेत: सकारात्मक \omterm{def:g2:mult:def}{उत्पाद}. विपरीत संकेत: नकारात्मक \omterm{def:g2:mult:def}{उत्पाद}.}
वही टेबल शासन करती है \omterm{def:g3:division:remainder}{भागफल}.
\end{theorem}

\begin{proof}[क्यों $(-1) \times (-1) = +1$]
पहला, $3 \times (-4)$ मतलब $(-4) + (-4) + (-4) = -12$: एक सकारात्मक
कई बार नकारात्मक, नकारात्मक होता है। अब पैटर्न देखें:
\[
3 \times (-4) = -12, \quad
2 \times (-4) = -8, \quad
1 \times (-4) = -4, \quad
0 \times (-4) = 0 .
\]
प्रत्येक चरण में, पहला कारक छूट जाता है $1$ और परिणाम \emph{उगता है}
द्वारा $4$. पैटर्न को एक और कदम जारी रखते हुए:
\[
(-1) \times (-4) = 0 + 4 = +4 .
\]
कोई अन्य विकल्प अंकगणित की नियमितता को तोड़ देगा
(वितरणशीलता)। इसलिए माइनस गुणा माइनस प्लस होना चाहिए।
\end{proof}

{\small किसी चित्र में पैटर्न तर्क: बिंदु
$(k,\ k \times (-4))$ लाइन अप करें, और लाइन को आगे बढ़ाते रहें $ k = 0$
बल $(-1)\times(-4) = +4$ और $(-2)\times(-4) = +8$.}
\end{omfigure}

\begin{example}\label{ex:g8:negprod:products}
\begin{align*}
(-7) \times (+6) &= -42 && \text{(opposite signs)} \\
(-5) \times (-8) &= +40 && \text{(same signs)} \\
(+2.5) \times (-4) &= -10 \\
(-36) \div (-9) &= +4 \\
(+15) \div (-2) &= -7.5 .
\end{align*}
\end{example}

\section{कई कारकों के उत्पाद}

\begin{proposition}[एक लंबे उत्पाद का संकेत]\label{prop:g8:negprod:many}
ए का चिन्ह \omterm{def:g2:mult:def}{उत्पाद} अनेक अशून्य कारकों पर ही निर्भर करता है
नकारात्मक कारकों की संख्या:
\begin{itemize}
  \item एक \emph{\omterm{def:g3:numbers:evenodd}{यहां तक की}} नकारात्मक कारकों की संख्या सकारात्मक परिणाम देती है
        \omterm{def:g2:mult:def}{उत्पाद};
  \item एक \emph{\omterm{def:g3:numbers:evenodd}{विषम}} संख्या ऋणात्मक देती है \omterm{def:g2:mult:def}{उत्पाद}.
\end{itemize}
\end{proposition}

\begin{proof}
नकारात्मक कारक जुड़ते हैं, और प्रत्येक \omterm{def:g1:addition:def}{जोड़}ा एक सकारात्मक संकेत देता है
(\cref{thm:g8:negprod:rules}); एक \omterm{def:g3:numbers:evenodd}{विषम} गिनती एक को अयुग्मित छोड़ देती है
नकारात्मक कारक, जो संपूर्ण बनाता है \omterm{def:g2:mult:def}{उत्पाद} नकारात्मक।
\end{proof}

\begin{example}\label{ex:g8:negprod:many}
$(-2) \times (+3) \times (-5) \times (-1)$: तीन नकारात्मक कारक
(\omterm{def:g3:numbers:evenodd}{विषम}), इतना \omterm{def:g2:mult:def}{उत्पाद} नकारात्मक है; दूरियाँ: $2 \times 3 \times 5
\times 1 = 30$. परिणाम: $-30$. चिन्ह तय करें \emph{पहला}, फिर
दूरियाँ गुणा करें --- एक त्रुटि-प्रवण के बजाय दो आसान कार्य
एक.
\end{example}

\begin{example}[नकारात्मक की शक्तियाँ]\label{ex:g8:negprod:powers}
$(-2)^2 = (-2) \times (-2) = +4$, और
$(-2)^3 = (-2)^2 \times (-2) = -8$: \omterm{def:g3:numbers:evenodd}{यहां तक की} प्रतिपादक सकारात्मक देते हैं
मूल्य, \omterm{def:g3:numbers:evenodd}{विषम} प्रतिपादक चिन्ह रखते हैं। अंकन में सावधानी:
$(-3)^2 = 9$ लेकिन $-3^2 = -(3 \times 3) = -9$.
\end{example}

\section{सभी चार ऑपरेशनों के साथ कंप्यूटिंग}

\begin{method}[सापेक्ष संख्याओं के साथ सुरक्षित गणना]\label{met:g8:negprod:compute}
\begin{enumerate}
  \item की प्राथमिकताओं का सम्मान करें \cref{ch:g7:priorities}: कोष्ठक,
        फिर $\times $ और $\div $, तब $+$ और $-$;
  \item प्रत्येक गुणा या भाग पर, ज्ञात कीजिए \emph{संकेत
        प्रथम}, फिर दूरी;
  \item प्रति पंक्ति एक चरण पुनः लिखें.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:negprod:compute}
\begin{align*}
5 - 3 \times (-4)
&= 5 - (-12) && \text{(product first: opposite signs)}\\
&= 5 + 12 = 17 .
\end{align*}
\begin{align*}
\frac{(-6) \times (-10)}{-4}
&= \frac{+60}{-4} && \text{(numerator: same signs)}\\
&= -15 && \text{(quotient: opposite signs).}
\end{align*}
\end{example}

\begin{example}[नकारात्मक मानों को प्रतिस्थापित करना]\label{ex:g8:negprod:substitute}
मूल्यांकन करना $ E = 3x^2 - 2x $ के लिए $ x = -5$, मान के चारों ओर कोष्ठक:
\[
E = 3 \times (-5)^2 - 2 \times (-5)
= 3 \times 25 - (-10)
= 75 + 10 = 85 .
\]
कोष्ठक में $(-5)^2$ आवश्यक हैं: उनके बिना वर्ग होगा
पर लागू करें $5$ केवल।
\end{example}

\begin{exercise}[$\star\star $]\label{exo:g8:negprod:10}

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g8:negprod:11}
वितरणशीलता का उपयोग करना (जैसा कि प्रमाण में है
\cref{thm:g8:negprod:rules}), प्रत्येक चरण का विस्तार करें और उसका औचित्य सिद्ध करें
\[
0 = (-1) \times 0 = (-1) \times \bigl(1 + (-1)\bigr)
= (-1) \times 1 + (-1) \times (-1),
\]
और निष्कर्ष निकालो $(-1) \times (-1)$ बराबर होना चाहिए $+1$.
\end{exercise}

\section{समस्या: माइनस गुणा माइनस प्लस क्यों है?}

\begin{problem}[{सप्ताहान्त समस्या---संकेत नियम से व्युत्पन्न वितरणशीलता, और वैकल्पिक योग $1 - 2 + 3 - 4 + \dots $}]
\label{pb:g8:negprod:1}
का प्रमाण \cref{thm:g8:negprod:rules} एक पैटर्न जारी रखा और
दावा किया कि ``कोई भी अन्य विकल्प वितरणवाद को तोड़ देगा''। यह
समस्या दावे को सटीक बनाती है: अकेले वितरण से शुरू
(\cref{thm:g7:priorities:distributivity}), आप करेंगे \emph{सिद्ध करना} द
संकेत नियम --- कोई चित्र नहीं, कोई पैटर्न नहीं, बीजगणित जिस तरह से करता है ---
फिर उन्हें शक्तियों पर काम पर लगाओ $-1$ और एक प्रसिद्ध पर \omterm{def:g1:addition:def}{जोड़} के साथ
वैकल्पिक संकेत. लगातार, $ a $ और $ b $ हैं सापेक्ष संख्याएँ; हम
के अतिरिक्त नियमों को हल्के में लें \cref{ch:g7:negatives}, वह
$1 \times a = a $, और वह ए \omterm{def:g2:mult:def}{उत्पाद} किसी भी क्रम में गणना की जा सकती है
(इसलिए वितरणता ए के दोनों तरफ लागू होती है \omterm{def:g2:mult:def}{उत्पाद}).

\medskip
\noindent\textbf{Part I --- The sign rules are theorems.}
\begin{enumerate}
  \item साबित करें कि $0 \times a = 0$ हरएक के लिए \omterm{def:g7:negatives:def}{सापेक्ष संख्या} $ a $.
        (लिखना $0 = 0 + 0$, बढ़ाना $(0 + 0) \times a $, और पूछें कौन सा
        संख्या, में \omterm{def:g1:addition:def}{जोड़}ा गया $0 \times a $, देता है $0 \times a $ पीछे।)
  \item बढ़ाना $\bigl(1 + (-1)\bigr) \times a $ यह दिखाने के लिए
        $(-1) \times a + a = 0$, और निष्कर्ष निकालो
        \[
        (-1) \times a = -a :
        \]
        से गुणा करना $-1$ विपरीत देता है. साथ तुलना करें
        \cref{exo:g8:negprod:11}, जो मामला है $ a = -1$.
  \item उसका निष्कर्ष निकालो $(-a) \times b = -(a \times b)$: एक ऋण चिह्न
        ए से निकलता है \omterm{def:g2:mult:def}{उत्पाद} अपरिवर्तित.
  \item उसका निष्कर्ष निकालो $(-a) \times (-b) = a \times b $: दो ऋण चिह्न
        रद्द करें.
  \item प्रत्येक \omterm{def:g7:negatives:def}{सापेक्ष संख्या} की दूरी है \omterm{def:g1:counting:zero}{शून्य} साइन इन के साथ
        सामने: $ a = +d $ या $ a = -d $. बताएं कि प्रश्न 3 और ~4 कैसे हैं
        की साइन टेबल के सभी चार मामलों को साबित करें
        \cref{thm:g8:negprod:rules}, दूरियाँ शामिल हैं।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Powers of $-1$.}
\begin{enumerate}[resume]
  \item गणना करना $(-1)^2$, $(-1)^3$, $(-1)^4$, $(-1)^5$, तो दे दो,
        से औचित्य के साथ \cref{prop:g8:negprod:many}, का मान है $(-1)^n $ प्रत्येक पूर्ण संख्या के लिए $ n \geq 1$.
  \item (किसी भी दूरी की गणना किए बिना) का चिन्ह दीजिए
        \[
        (-1) \times (-2) \times (-3) \times \dots \times (-10),
        \]
        और इसकी दूरी लिखिए \omterm{def:g2:mult:def}{उत्पाद} के तौर पर \omterm{def:g2:mult:def}{उत्पाद} संपूर्ण का
        संख्याएँ, इस पर काम किए बिना।
  \item बताते हैं कि $(-a)^2 = a^2$ ओर वो $(-a)^3 = -a^3$. जिसके लिए
        घातांक क्या ऋण चिन्ह जीवित रहता है?
  \item किसी रिश्तेदार का वर्ग दर्शाने के लिए चिह्न नियमों का उपयोग करें
        संख्या कभी भी ऋणात्मक नहीं होती, और यह निष्कर्ष निकालो कि नहीं \omterm{def:g7:negatives:def}{सापेक्ष संख्या}
        $ x $ संतुष्ट $ x^2 = -4$.
  \item ज़ो का दावा है: ``$(-1)^{2026} + (-1)^{2027} = 0$, और भी बहुत कुछ
        आम तौर पर की दो लगातार शक्तियाँ $-1$ हमेशा रद्द करें.'' है
        वह सही है? औचित्य।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The alternating \omterm{def:g1:addition:def}{जोड़}.}
साइन नियमों को सुरक्षित करके, हम गणना कर सकते हैं रकम जो दोनों को मिला दे
संकेत. एक पूर्ण संख्या के लिए $ n \geq 1$, होने देना
\[
A_n = 1 - 2 + 3 - 4 + \dots
\]
हो \omterm{def:g1:addition:def}{जोड़} से पूर्ण संख्याओं का $1$ को $ n $ बारी-बारी से
संकेत: अंतिम पद है $+n $ कब $ n $ है \omterm{def:g3:numbers:evenodd}{विषम}, $-n $ कब $ n $ है
\omterm{def:g3:numbers:evenodd}{यहां तक की}. इसलिए $ A_1 = 1$ और $ A_2 = 1 - 2 = -1$.
\begin{enumerate}[resume]
  \item गणना करना $ A_3$, $ A_4$, $ A_5$, $ A_6$ और $ A_7$. आप क्या करते हैं?
        अनुमान?
  \item कल्पना करना $ n $ है \omterm{def:g3:numbers:evenodd}{यहां तक की}. पदों को दो बटा दो समूहित करें,
        $(1 - 2) + (3 - 4) + \dots $, \omterm{def:g1:addition:def}{जोड़}ियों को गिनें और सिद्ध करें
        $ A_n = -\frac{n}{2}$.
  \item कल्पना करना $ n $ है \omterm{def:g3:numbers:evenodd}{विषम}. समझाइए क्यों $ A_n = A_{n-1} + n $, और
        पिछले प्रश्न से यह निष्कर्ष निकालें
        $ A_n = \frac{n + 1}{2}$.
  \item गणना करना $ A_{100}$, $ A_{101}$ और $ A_{2026}$.
  \item अल्मा ने गणना की $ A_{101} - A_{100} = 51 - (-50) = 101$ और था
        बिलकुल खोज कर आश्चर्यचकित हो गया $101$. क्या उसे होना चाहिए था?
        एक वाक्य में समझाइये और इसका मूल्य बताइये
        $ A_{2027} - A_{2026}$ बिना किसी अतिरिक्त गणना के.
\end{enumerate}
\end{problem}
