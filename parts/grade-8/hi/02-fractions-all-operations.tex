\chapter{भिन्न: सभी चार संक्रियाएँ}\label{ch:g8:fractions}

यह अध्याय का अंकगणित पूरा करता है \omterm{def:g6:fractions:def}{अंश}ों: \omterm{def:g1:addition:def}{जोड़} और
के साथ घटाव \emph{कोई} हरों, गुणा, और ---
नवागंतुक --- विभाजन, जो गुणा में बदल जाता है
भेष बदलना। से संकेत \cref{ch:g8:negprod} सवारी के लिए साथ आ जाओ।

\section{किसी भी हर के साथ जोड़ना}

\begin{method}[आम विभाजक]\label{met:g8:fractions:add}
जोड़ना या घटाना $\frac ab $ और $\frac cd $:
\begin{enumerate}
  \item एक सामान्य खोजें एकाधिक का $ b $ और $ d $ (द \omterm{def:g2:mult:def}{उत्पाद}
        $ b \times d $ हमेशा काम करता है; छोटा वाला प्रयास बचाता है);
  \item दोनों को फिर से लिखें \omterm{def:g6:fractions:def}{अंश}ों उस के साथ \omterm{def:g5:division:multiple}{भाजक};
  \item \omterm{def:g1:addition:def}{जोड़}ें या घटाएँ \omterm{def:g6:fractions:def}{अंश}; सरल करें
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:fractions:add}
गणना करना $\dfrac{5}{6} + \dfrac{3}{4}$. एक आम एकाधिक का $6$ और
$4$ है $12$:
\[
\frac{5}{6} = \frac{10}{12},
\qquad
\frac{3}{4} = \frac{9}{12},
\qquad
\frac{5}{6} + \frac{3}{4} = \frac{10 + 9}{12} = \frac{19}{12}.
\]
संकेतों के साथ: $\dfrac{2}{3} - \dfrac{7}{5} = \dfrac{10}{15} -
\dfrac{21}{15} = -\dfrac{11}{15}$.
\end{example}

\section{भिन्नों को गुणा करना}

\begin{theorem}[भिन्नों का गुणनफल]\label{thm:g8:fractions:mult}
\[
\frac{a}{b} \times \frac{c}{d} = \frac{a \times c}{b \times d} .
\]
\end{theorem}

\begin{proof}[क्यों, एक तस्वीर पर]
लो $\frac34$ का $\frac25$ एक वर्ग का: वर्ग को अंदर काटें $5$
ऊर्ध्वाधर पट्टियाँ और रखें $2$; क्षैतिज रूप से काटें $4$ और रखो $3$
जो कुछ बचा उसकी पंक्तियाँ। रखा हुआ भाग एक ग्रिड है $3 \times 2$ छोटा
कोशिकाओं से बाहर $4 \times 5$: \omterm{def:g6:fractions:def}{अंश} $\frac{6}{20}$.
\end{proof}

  \fill[omDef!18] (0,0) rectangle (2,4);

{\small $\frac34 \times \frac25$: दोहरी छायांकित \omterm{def:g6:measure:area}{क्षेत्र} है
$3 \times 2 = 6$ कोशिकाओं से बाहर $4 \times 5 = 20$, अर्थात\
$\frac{6}{20} = \frac{3}{10}$ --- \omterm{def:g6:fractions:def}{अंश} गुणा किया हुआ, हरों
गुणा किया हुआ।}
\end{omfigure}

\begin{example}\label{ex:g8:fractions:mult}
सरल \emph{पहले} गुणा करना, सामान्य कारकों को पार करना:
\[
\frac{7}{12} \times \frac{9}{14}
= \frac{7 \times 9}{12 \times 14}
= \frac{7 \times 3 \times 3}{3 \times 4 \times 2 \times 7}
= \frac{3}{8}
\]
(कारक $7$ और एक कारक $3$ ऊपर और नीचे के बीच रद्द करें)।
संकेतों के साथ, के नियम \cref{thm:g8:negprod:rules} पहले आवेदन करें:
$\left(-\frac23\right) \times \left(-\frac{5}{4}\right) =
+\frac{10}{12} = \frac56$.
\end{example}

\section{भिन्नों को विभाजित करना}

\begin{definition}[श्लोक में]\label{def:g8:fractions:inverse}
 \emph{श्लोक में}\index{श्लोक में} एक \omterm{def:g1:counting:zero}{शून्य}ेतर संख्या का $ x $ है
वह संख्या जिससे गुणा किया गया $ x $ देता है $1$. का उलटा
$\frac ab $ है $\frac ba $, तब से
$\frac ab \times \frac ba = \frac{ab}{ab} = 1$; एक का उलटा
पूर्णांक $ n $ है $\frac1n $.
\end{definition}

\begin{theorem}[भाग देने का अर्थ है व्युत्क्रम से गुणा करना]\label{thm:g8:fractions:div}
के लिए $ c \neq 0$:
\[
\frac{a}{b} \div \frac{c}{d} = \frac{a}{b} \times \frac{d}{c} .
\]
\end{theorem}

\begin{proof}
परिभाषा के अनुसार, \omterm{def:g3:division:remainder}{भागफल} $ q $ का $\frac ab $ द्वारा $\frac cd $ है
संख्या के साथ $ q \times \frac cd = \frac ab $. उम्मीदवार की जांच करें
$ q = \frac ab \times \frac dc $:
\[
\left(\frac ab \times \frac dc\right) \times \frac cd
= \frac ab \times \left(\frac dc \times \frac cd\right)
= \frac ab \times 1 = \frac ab . \qedhere
\]
\end{proof}

\begin{example}\label{ex:g8:fractions:div}
\[
\frac{3}{5} \div \frac{7}{10}
= \frac{3}{5} \times \frac{10}{7}
= \frac{3 \times 10}{5 \times 7}
= \frac{30}{35} = \frac{6}{7}.
\]
एक विवेक जाँच: द्वारा विभाजित करना $\frac{7}{10}$, से छोटी संख्या
$1$, एक परिणाम देना होगा \emph{बड़ा} बजाय $\frac35$ --- और
$\frac67 > \frac35$ वास्तव में ($\frac{30}{35} > \frac{21}{35}$).
\end{example}

\begin{example}[भिन्न पट्टियों के भीतर भिन्न पट्टियाँ]\label{ex:g8:fractions:complex}
एक ``डबल-डेकर'' केवल लंबवत रूप से लिखा गया एक प्रभाग है:
\[
\frac{\ \frac{2}{3}\ }{\ \frac{5}{6}\ }
= \frac{2}{3} \div \frac{5}{6}
= \frac{2}{3} \times \frac{6}{5}
= \frac{12}{15} = \frac{4}{5}.
\]
का पता लगाएं \emph{मुख्य} पहले बार (सबसे लंबा वाला), फिर लागू करें
\cref{thm:g8:fractions:div}.
\end{example}

\begin{method}[मिश्रित गणना]\label{met:g8:fractions:mixed}
एक अभिव्यक्ति मिश्रण में \omterm{def:g6:fractions:def}{अंश}ों और चार ऑपरेशन:
\begin{enumerate}
  \item सामान्य प्राथमिकताएँ लागू करें (\cref{ch:g7:priorities});
  \item प्रत्येक भाग को गुणा में परिवर्तित करें \omterm{def:g8:fractions:inverse}{श्लोक में};
  \item पहले संकेत तय करें (\cref{met:g8:negprod:compute}), फिर
        गुणा करें या सामान्य खोजें हरों;
  \item यथाशीघ्र सरलीकरण करें।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:fractions:mixed}
\begin{align*}
\frac12 + \frac{2}{3} \times \frac{9}{4}
&= \frac12 + \frac{2 \times 9}{3 \times 4}
&& \text{(product first!)} \\
&= \frac12 + \frac{18}{12} = \frac12 + \frac32
= \frac{4}{2} = 2 .
\end{align*}
\end{example}

\section{समस्या: मिस्र के अंश}

\begin{problem}[{सप्ताहांत समस्या---प्रत्येक भिन्न का योग है विशिष्ट इकाई भिन्न}]
\label{pb:g8:fractions:1}
ए \emph{इकाई \omterm{def:g6:fractions:def}{अंश}} एक है \omterm{def:g6:fractions:def}{अंश} साथ \omterm{def:g4:fractions:def}{मीटर} $1$. द
प्राचीन मिस्र के शास्त्रियों --- रिहंद पपीरस की नकल की गई थी
$1550$~बीसी---प्रत्येक ने लिखा \omterm{def:g6:fractions:def}{अंश} के तौर पर \omterm{def:g1:addition:def}{जोड़} का \emph{विशिष्ट}
इकाई \omterm{def:g6:fractions:def}{अंश}ों: कभी नहीं $\frac27 = \frac17 + \frac17$, लेकिन
$\frac27 = \frac14 + \frac{1}{28}$. ऐसा \omterm{def:g1:addition:def}{जोड़} एक कहा जाता है
\emph{मिस्री लेखन} की \omterm{def:g6:fractions:def}{अंश}. यह समस्या विकसित होती है
इस अध्याय की \omterm{def:g1:addition:def}{जोड़} और तुलना तकनीकें
(\cref{met:g8:fractions:add}) एक विधि में --- द्वारा प्रकाशित
फाइबोनैचि में $1202$ --- जो किसी का मिस्री लेखन उत्पन्न करता है
\omterm{def:g6:fractions:def}{अंश} बीच में $0$ और $1$, और एक उत्सव के साथ समाप्त होता है
विरासत पहेली.

\medskip
\noindent\textbf{Part I --- Unit \omterm{def:g6:fractions:def}{अंश}ों and the splitting
identity.}
\begin{enumerate}
  \item गणना करना $\frac12 + \frac13$ और $\frac12 + \frac14$, और
        मिस्र के लेखन का निष्कर्ष निकालें $\frac56$ और $\frac34$.
  \item के लिए शास्त्रियों की प्रविष्टि सत्यापित करें $\frac27$: बताते हैं कि
        $\frac14 + \frac{1}{28} = \frac27$.
  \item जाँच करें कि $\frac13 + \frac16 = \frac12$ ओर वो
        $\frac14 + \frac{1}{12} = \frac13$. फिर सिद्ध करें
        \emph{पहचान बांटना}: प्रत्येक पूर्ण संख्या के लिए
        $ n \geq 1$,
        \[
        \frac{1}{n} = \frac{1}{n+1} + \frac{1}{n(n+1)} .
        \]
  \item एक मिस्री लेखन का निष्कर्ष निकालें $1$ स्वयं, एक के रूप में \omterm{def:g1:addition:def}{जोड़} का
        तीन अलग इकाई \omterm{def:g6:fractions:def}{अंश}ों.
  \item के अंतिम पद को विभाजित करें $\frac56 = \frac12 + \frac13$ को
        एक प्राप्त करें \emph{दूसरा} मिस्र का लेखन $\frac56$, के साथ
        तीन इकाई \omterm{def:g6:fractions:def}{अंश}ों. बताएं क्यों नहीं \omterm{def:g6:fractions:def}{अंश} एक अनोखा है
        मिस्री लेखन.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Fibonacci's greedy method.}
विधि है: घटाएँ \emph{सबसे बड़ा} इकाई \omterm{def:g6:fractions:def}{अंश} वह
फिट बैठता है, और जो बचता है उसके साथ दोहराएँ।
\begin{enumerate}[resume]
  \item सामान्य का उपयोग करना हरों, बताते हैं कि
        $\frac14 < \frac27 < \frac13$. इससे यह क्यों सिद्ध होता है
        $\frac14$ सबसे बड़ी इकाई है \omterm{def:g6:fractions:def}{अंश} तुलना में छोटा
        $\frac27$?
  \item गणना करना $\frac27 - \frac14$. किसका लेखन $\frac27$
        क्या लालची विधि से उत्पादन होता है?
  \item अब लीजिए $\frac57$. दिखाओ कि सबसे बड़ी इकाई \omterm{def:g6:fractions:def}{अंश}
        तुलना में छोटा $\frac57$ है $\frac12$, और गणना करें
        $\frac57 - \frac12$.
  \item दिखाओ कि सबसे बड़ी इकाई \omterm{def:g6:fractions:def}{अंश} तुलना में छोटा
        $\frac{3}{14}$ है $\frac15$ (तुलना करना $\frac{3}{14}$ साथ
        $\frac14$ और साथ में $\frac15$), गणना करें
        $\frac{3}{14} - \frac15$, और निष्कर्ष निकालें:
        \[
        \frac57 = \frac12 + \frac15 + \frac{1}{70} .
        \]
        इस लेखन को सीधे आम से सत्यापित करें \omterm{def:g5:division:multiple}{भाजक}
        $70$.
  \item लालची पद्धति चालू रखें $\frac45$, हर कदम पर जाँच
        वह आपकी इकाई \omterm{def:g6:fractions:def}{अंश} सबसे बड़ा है जो फिट बैठता है, और
        अंतिम लेखन को सत्यापित करें.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Why the method always stops, and the
eighteenth camel.}
\begin{enumerate}[resume]
  \item प्रश्न 7-10 में विधि लागू की गई थी $\frac27$,
        $\frac57$ और $\frac45$. प्रत्येक के लिए, सूची बनाएं \omterm{def:g6:fractions:def}{अंश} का
        क्रमिक \omterm{def:g3:division:remainder}{शेष} (किसी भी सरलीकरण से पहले).
        आप क्या निरीक्षण करते हैं?
  \item एक के लिए \omterm{def:g6:fractions:def}{अंश} $\frac ab $ और एक पूर्ण संख्या $ n $, बताते हैं कि
        \[
        \frac{a}{b} - \frac{1}{n}
        = \frac{a \times n - b}{b \times n} .
        \]
  \item मान लीजिए लालची विधि घटा देती है $\frac1n $ से
        $\frac ab $: तब $\frac1n $ लेकिन फिट बैठता है $\frac{1}{n-1}$ है
        पहले से ही बहुत बड़ा, $\frac{1}{n-1} > \frac ab $. दोनों को लगाना
        आम से ऊपर \omterm{def:g5:division:multiple}{भाजक} $ b \times (n - 1)$, निष्कर्ष निकालना
        वह $ b > a \times n - a $, इसलिए वह नया है \omterm{def:g4:fractions:def}{मीटर}
        संतुष्ट
        \[
        a \times n - b < a .
        \]
        बताएं कि इससे यह क्यों साबित होता है कि विधि हमेशा रुक जाती है।
  \item एक पुरानी कहानी: एक पिता को छोड़कर मर जाता है $17$ ऊँट, इच्छुक
        $\frac12$ अपने सबसे बड़े बच्चे को झुण्ड का, $\frac13$ को
        दूसरा, $\frac19$ तीसरे को --- और इनमें से कोई नहीं
        शेयर ऊंटों की एक पूरी संख्या है। एक पड़ोसी उधार देता है
        परिवार एक अठारहवाँ ऊँट; बच्चे लेते हैं $9$, $6$ और
        $2$ ऊँट, और उधार लिया हुआ ऊँट लौटा दिया जाता है। गणना करना
        $\frac12 + \frac13 + \frac19$ और युक्ति समझाएँ: क्यों
        क्या झुंड को पूरे ऊँटों में बाँटा जा सकता था, और ऐसा क्यों नहीं किया गया
        क्या बच्चे को वसीयत के वादे से कम मिलता है?
  \item वसीयत की मरम्मत करें: तीन का प्रस्ताव रखें \emph{विशिष्ट} इकाई
        \omterm{def:g6:fractions:def}{अंश}ों जो बिल्कुल जुड़ता है $1$, और दे दो
        सबसे छोटे झुंड का आकार जिसके लिए सभी तीन शेयर पूरे हैं
        ऊँटों की संख्या.
\end{enumerate}
\end{problem}
