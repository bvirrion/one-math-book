\chapter{शाब्दिक गणना और समीकरण}\label{ch:g8:equations}

ग्रेड 7 में अक्षरों का परिचय दिया गया और सरल भावों को कम किया गया
(\cref{ch:g7:literal}). यह अध्याय उन दो कौशलों को जोड़ता है जो बनाते हैं
बीजगणित शक्तिशाली: \emph{का विस्तार} \omterm{def:g2:mult:def}{उत्पाद}ों कोष्ठक का, और
\emph{सुलझाने \omterm{def:g8:equations:def}{समीकरण}} --- के पीछे छुपे नंबर को ढूंढना
पत्र.

\section{विस्तार}

\begin{theorem}[सरल और दोहरी वितरणशीलता]\label{thm:g8:equations:expand}
सभी नंबरों के लिए:
\[
k(a + b) = ka + kb,
\qquad
(a + b)(c + d) = ac + ad + bc + bd .
\]
\end{theorem}

\begin{proof}
पहला नियम है \cref{thm:g7:priorities:distributivity} (संकेत
सम्मिलित: $k $ और शर्तें नकारात्मक हो सकती हैं,
\cref{ch:g8:negprod}). दूसरे के लिए, पहले नियम को दो बार लागू करें,
इलाज $(c + d)$ एकल संख्या के रूप में:
\[
(a + b)(c + d) = a(c + d) + b(c + d) = ac + ad + bc + bd .
\qedhere
\]
\end{proof}

\begin{example}\label{ex:g8:equations:expand}
प्रत्येक पद पर संकेतों पर ध्यान दें:
\begin{align*}
3(2x - 5) &= 6x - 15, \\
-2(4x - 7) &= -8x + 14
&& \text{(a minus in front flips both signs)}, \\
(x + 3)(x + 5) &= x^2 + 5x + 3x + 15 = x^2 + 8x + 15, \\
(2x - 1)(x - 4) &= 2x^2 - 8x - x + 4 = 2x^2 - 9x + 4 .
\end{align*}
\end{example}

\begin{method}[बढ़ाओ और घटाओ]\label{met:g8:equations:reduce}
\begin{enumerate}
  \item प्रत्येक का विस्तार करें \omterm{def:g2:mult:def}{उत्पाद}, प्रत्येक पद को उसके चिह्न सहित लिखना;
  \item समान पद एकत्रित करें ($ x^2$ साथ $ x^2$, $ x $ साथ $ x $, संख्याएँ
        संख्याओं के साथ);
  \item एक मान के साथ जांचें: स्थानापन्न उदा.\ $ x = 2$ मूल में
        और परिणाम में --- समान परिणाम अधिकांश गलतियाँ पकड़ते हैं।
\end{enumerate}
\end{method}

\section{समीकरण हल करना}

\begin{definition}[समीकरण]\label{def:g8:equations:def}
एक \emph{समीकरण}\index{समीकरण} अज्ञात के साथ समानता है
संख्या, लिखा हुआ $ x $ (या कोई पत्र). \emph{हल} इसका मतलब है खोजना
के सभी मान $ x $ समानता को सत्य बनाना---
\emph{समाधान}.
\end{definition}

\begin{theorem}[संतुलन नियम]\label{thm:g8:equations:balance}
एक \omterm{def:g8:equations:def}{समीकरण} बिल्कुल वही समाधान रखता है जब:
\begin{enumerate}
  \item दोनों पक्षों में समान संख्या \omterm{def:g1:addition:def}{जोड़}ी जाती है (या घटाई जाती है);
  \item दोनों पक्षों को एक ही से गुणा (या विभाजित) किया जाता है
        \emph{अशून्य} संख्या।
\end{enumerate}
\end{theorem}
\admitted

दूसरे के प्रति किया जाना चाहिए, और संतुलन---समानता---

\begin{method}[प्रथम-डिग्री समीकरण को हल करना]\label{met:g8:equations:solve}
\begin{enumerate}
  \item यदि आवश्यक हो तो प्रत्येक पक्ष को विस्तृत और छोटा करें;
  \item इकट्ठा करने के लिए दोनों तरफ \omterm{def:g1:addition:def}{जोड़}ें या घटाएं $ x $-शर्तें एक पर
        एक तरफ, दूसरी तरफ सादे नंबर;
  \item घटाएं, फिर दोनों पक्षों को के गुणांक से विभाजित करें $ x $;
  \item \emph{जाँच करना} मूल में पाए गए मान को प्रतिस्थापित करके
        \omterm{def:g8:equations:def}{समीकरण}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:equations:solve}
हल करना $7x - 5 = 4x + 16$:
\begin{align*}
7x - 5 &= 4x + 16 \\
7x - 4x &= 16 + 5
&& \text{(add $5$, subtract $4x $, on both sides)} \\
3x &= 21 \\
x &= 7 && \text{(divide both sides by 3).}
\end{align*}
जांचें: बाईं ओर $7 \times 7 - 5 = 44$; दाहिनी ओर
$4 \times 7 + 16 = 44$. समाधान है $7$.
\end{example}

\begin{example}[कोष्ठक और नकारात्मक के साथ]\label{ex:g8:equations:brackets}
हल करना $3(x - 4) = 5 - 2(x + 1)$:
\begin{align*}
3x - 12 &= 5 - 2x - 2 && \text{(expand both sides)} \\
3x - 12 &= 3 - 2x && \text{(reduce)} \\
5x &= 15 && \text{(add $2x + 12$ to both sides)} \\
x &= 3 .
\end{align*}
जाँच करना: $3(3 - 4) = -3$ और $5 - 2(3+1) = 5 - 8 = -3$. \quad समाधान:
$3$.
\end{example}

\begin{method}[किसी समस्या को समीकरण से हल करना]\label{met:g8:equations:problem}
\begin{enumerate}
  \item अज्ञात चुनें: ``चलो $ x $ होना \dots'' (मात्रा पूछी गई
        के लिए, या एक सरल संबंधित);
  \item कहानी का अनुवाद एक में करें \omterm{def:g8:equations:def}{समीकरण};
  \item इसे हल करो;
  \item व्याख्या: मूल प्रश्न का उत्तर एक वाक्य में दें
        इकाइयाँ, और कहानी के सामने उत्तर की जाँच करें।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:equations:problem}
लुकास है $4$ अपनी बहन से कई वर्ष बड़े; कुल मिलाकर उनकी उम्र का योग है
$26$. होने देना $ x $ बहन की उम्र हो; फिर लुकास है $ x + 4$, और
\[
x + (x + 4) = 26
\ \Longrightarrow\
2x + 4 = 26
\ \Longrightarrow\
2x = 22
\ \Longrightarrow\
x = 11 .
\]
बहन है $11$, लुकास है $15$. जाँच करना: $11 + 15 = 26$ और
$15 - 11 = 4$.
\end{example}

\section{व्यवस्था और असमानताएँ}

\begin{proposition}[संचालन एवं व्यवस्था]\label{prop:g8:equations:order}
किसी असमानता के दोनों पक्षों में समान संख्या जोड़ने से वह सुरक्षित रहती है;
दोनों पक्षों को a से गुणा करना \emph{सकारात्मक} संख्या इसे सुरक्षित रखती है;
ए से गुणा करना \emph{नकारात्मक} संख्या \emph{पराजय} यह:
\[
2 < 5 \ \text{ but }\ (-1) \times 2 > (-1) \times 5 .
\]
\end{proposition}
\admitted

\begin{example}\label{ex:g8:equations:inequality}
हल करना $5 - 2x \leq 11$: घटाना $5$ दोनों तरफ, $-2x \leq 6$;
द्वारा विभाजित करें $-2$ \emph{और उलटा}: $ x \geq -3$. सभी नंबर
से अधिक या उसके बराबर $-3$ समाधान हैं --- एक \omterm{def:g8:equations:def}{समीकरण} आमतौर पर
एक समाधान है, एक असमानता में उनकी पूरी आधी-रेखा है।
\end{example}

\section{समस्या: उल्लेखनीय पहचान और विषम की सीढ़ी
नंबर}

\begin{problem}[{सप्ताहांत समस्या --- $(a+b)^2$, $(a-b)^2$, $(a+b)(a-b)$, और योग $1 + 3 + 5 + \dots + (2n-1) = n^2$}]
\label{pb:g8:equations:1}
तीन विस्तार इतनी बार सामने आते हैं कि बीजगणित उन्हें पहचान लेता है
हृदय: इन्हें कहा जाता है \emph{उल्लेखनीय पहचान}. यह
समस्या उन्हें दोहरे वितरण से उत्पन्न करती है
(\cref{thm:g8:equations:expand}), उन्हें मानसिक-अंकगणित में बदल देता है
महाशक्ति, एक सुंदर तथ्य को साबित करने के लिए उनका उपयोग करती है \omterm{def:g3:numbers:evenodd}{विषम} संख्याएँ,
और समाधान करके समाप्त होता है \omterm{def:g8:equations:def}{समीकरण} बिल्कुल नए प्रकार का: \omterm{def:g8:equations:def}{समीकरण} एक के साथ
उनमें वर्ग.

\medskip
\noindent\textbf{Part I --- The three identities.}
\begin{enumerate}
  \item बढ़ाना $(a + b)^2 = (a + b)(a + b)$ और कम करना, सिद्ध करना
        \[
        (a + b)^2 = a^2 + 2ab + b^2 .
        \]
  \item इसी प्रकार दोनों सहचरों को भी सिद्ध करें:
        \[
        (a - b)^2 = a^2 - 2ab + b^2,
        \qquad
        (a + b)(a - b) = a^2 - b^2 .
        \]
  \item भुजा का एक वर्ग बनाएं $ a + b $, प्रत्येक पक्ष को विभाजित करें $ a $
        और $ b $, और चौकोर को चार टुकड़ों में काट लें। कौन सा टुकड़ा
        चित्र के प्रत्येक भाग की पहली पहचान होती है
        प्रतिनिधित्व करते हैं?
  \item मानसिक अंकगणित: गणना करने के लिए पहचानों का उपयोग करें, बिना
        कोई भी गुणन प्रस्तुत करना,
        \[
        101^2, \qquad 99^2, \qquad 49 \times 51 .
        \]
  \item एक क्लासिक \omterm{def:g5:solids:net}{जाल}: ज़ो लिखती है ``$(a + b)^2 = a^2 + b^2$''.
        उसके सूत्र का परीक्षण करें $ a = 3$, $ b = 4$, और पर समझाएं
        प्रश्न 3 का चित्र बिल्कुल वही है जो उसके सूत्र का \omterm{def:g6:lines:objects}{खंड} है
        भूल जाता है.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The staircase of \omterm{def:g3:numbers:evenodd}{विषम} संख्याएँ.}
\begin{enumerate}[resume]
  \item गणना करना $1$, तब $1 + 3$, तब $1 + 3 + 5$, तब
        $1 + 3 + 5 + 7$, तब $1 + 3 + 5 + 7 + 9$. आप क्या करते हैं?
        परिणामों के बारे में सूचना?
  \item स्पष्ट करें कि क्यों $ n $-वां \omterm{def:g3:numbers:evenodd}{विषम} संख्या है $2n - 1$ (जाँच करना:
        $ n = 1, 2, 3$ देना $1, 3, 5$).
  \item विस्तार और घटाने के लिए भाग I की एक पहचान का उपयोग करें
        $(n + 1)^2 - n^2$. परिणाम का इससे क्या लेना-देना है
        प्रश्न 7?
  \item प्रश्न 8 कहता है: भुजा के एक वर्ग से गुजरना $ n $ एक को
        भुजा का वर्ग $ n + 1$, एक बिल्कुल अगला जोड़ता है अजीब
        संख्या इकाई कोशिकाओं की --- दो के साथ एक एल-आकार की सीमा
        किनारे और \omterm{def:g5:solids:def}{कोने}. से शुरू $1 = 1^2$ और बढ़ रहा है
        एक समय में एक परत को चौकोर करें, समझाएँ क्यों
        \[
        1 + 3 + 5 + \dots + (2n - 1) = n^2 :
        \]
         \omterm{def:g1:addition:def}{जोड़} पहले का $ n $ \omterm{def:g3:numbers:evenodd}{विषम} संख्याएँ है $ n $-वां वर्ग.
  \item उन्हें एक-एक करके \omterm{def:g1:addition:def}{जोड़}े बिना, निष्कर्ष निकालें: का मूल्य
        $1 + 3 + 5 + \dots + 99$; और कितने \omterm{def:g3:numbers:evenodd}{विषम} संख्याएँ, से शुरू $1$, बिल्कुल \omterm{def:g1:addition:def}{जोड़}ें $400$.
\end{enumerate}

\noindent\textbf{Part III --- \omterm{def:g8:equations:def}{समीकरण} with a square.}
\begin{enumerate}[resume]
  \item में एक उल्लेखनीय पहचान पहचानें $ x^2 + 6x + 9$, और
        हल करें \omterm{def:g8:equations:def}{समीकरण} $ x^2 + 6x + 9 = 0$.
  \item के संकेत-और-दूरी नियम का प्रयोग करते हुए समझाइए
        \cref{thm:g8:negprod:rules}, क्यों एक \omterm{def:g2:mult:def}{उत्पाद} दो संख्याओं का
        हो सकता है \omterm{def:g1:counting:zero}{शून्य} \emph{केवल} दो कारकों में से एक है \omterm{def:g1:counting:zero}{शून्य}.
        इसे हल करने के लिए उपयोग करें $(x - 2)(x + 5) = 0$.
  \item हल करना $ x^2 = 49$ सब कुछ एक तरफ ले जाकर और
        फ़ैक्टरिंग $ x^2 - 49$ तीसरी पहचान के साथ. कितने
        समाधान यह करता है \omterm{def:g8:equations:def}{समीकरण} है --- और वह कैसे होता है
        हर एक से भिन्न \omterm{def:g8:equations:def}{समीकरण} इस अध्याय में अब तक हल हुआ?
  \item में \cref{exo:g8:equations:3} आपने उसे चेक किया $ x = 3$ है
        का एक समाधान $ x^2 = 6x - 9$. दिखाएँ कि यह है
        \emph{केवल} एक: सब कुछ एक तरफ ले आओ, पहचानो
        एक पहचान, और निष्कर्ष।
  \item समापन: गणना करें $2026^2 - 2025^2$ आपके दिमाग में, और
        एक पहचान के साथ समझाएं कि क्यों \omterm{ex:g1:subtraction:difference}{अंतर} दो का
        क्रमागत वर्ग हमेशा होता है \omterm{def:g1:addition:def}{जोड़} दो संख्याओं में से
        --- जो एक बार फिर प्रश्न 8 है।
\end{enumerate}
\end{problem}
