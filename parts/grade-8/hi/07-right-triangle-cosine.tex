\chapter{समकोण त्रिभुज और कोज्या}\label{ch:g8:cosine}

दो खूबसूरत तथ्य आपस में जुड़े हुए हैं सही \omterm{def:g3:shapes:shapes}{त्रिकोण} वृत्त के लिए: ए सही \omterm{def:g3:shapes:shapes}{त्रिकोण} अर्ध-वृत्त में बिल्कुल फिट बैठता है, इसका कर्ण है
\omterm{def:g4:geometry:circle}{व्यास}. और एक नंबर, \emph{कोज्या} एक कोण का, एन्कोड करता है
प्रत्येक का आकार सही \omterm{def:g3:shapes:shapes}{त्रिकोण} उस कोण से --- पहला
त्रिकोणमितीय अनुपात, अपने सहोदर साइन और स्पर्शरेखा से आगे
(\cref{ch:g9:trig}).

\section{समकोण त्रिभुज और उसका वृत्त}

\begin{theorem}[वृत्त प्रमेय]\label{thm:g8:cosine:circle}
\begin{enumerate}
  \item यदि एक त्रिभुज $ABC $ है \omterm{def:g6:shapes:triangles}{समकोण का} पर $ A $, तब $ A $ पर झूठ बोलता है
        वह वृत्त जिसका \omterm{def:g4:geometry:circle}{व्यास} कर्ण है $[BC]$.
  \item इसके विपरीत, यदि $ A $ के वृत्त पर स्थित है \omterm{def:g4:geometry:circle}{व्यास} $[BC]$ (साथ
        $ A \neq B, C $), फिर त्रिभुज $ ABC $ है \omterm{def:g6:shapes:triangles}{समकोण का} पर
        $ A $.
\end{enumerate}
समतुल्य: ए में सही \omterm{def:g3:shapes:shapes}{त्रिकोण}, कर्ण का मध्यबिंदु है
तीनों से समान दूरी पर \omterm{def:g5:solids:def}{कोने} --- से माध्यिका
\omterm{def:g3:shapes:rightangle}{समकोण} पैमाने \omterm{def:g3:division:half}{आधा} कर्ण.
\end{theorem}

\begin{proof}[1 का प्रमाण]
चलो $ O $ का मध्यबिंदु हो $[BC]$ और $ D $  \omterm{def:g7:central:def}{सममित} का $ A $ के बारे में
$ O $. के विकर्ण $ ABDC $ उनके सामान्य मध्यबिंदु पर काटें $ O $, इसलिए
$ ABDC $ एक है \omterm{def:g7:central:parallelogram}{चतुर्भुज} (\cref{def:g7:central:parallelogram}) ---
के साथ \omterm{def:g3:shapes:rightangle}{समकोण} पर $ A $: यह एक आयत है. ए के विकर्ण
आयत बराबर हैं, इसलिए $ OA = \frac{AD}{2} = \frac{BC}{2}$: द
बिंदु $ A $ की दूरी पर है $\frac{BC}{2}$ से $ O $, यानी\ वृत्त पर
का \omterm{def:g4:geometry:circle}{व्यास} $[BC]$. (बिंदु 2 तर्क चलाने से सिद्ध होता है
पीछे की ओर.)
\end{proof}

\end{example}

\section{न्यून कोण की कोज्या}

\begin{definition}[कोज्या]\label{def:g8:cosine:def}
में एक सही \omterm{def:g3:shapes:shapes}{त्रिकोण}, एक न्यून कोण के लिए $\theta $:
\[
\cos\theta = \frac{\text{side adjacent to } \theta}{\text{hypotenuse}}
\]
---पैर छूना $\theta $, कर्ण से विभाजित। यह अनुपात
केवल कोण पर निर्भर करता है, त्रिभुज के आकार पर नहीं: सब कुछ ठीक है
\omterm{def:g3:shapes:shapes}{त्रिकोण} समान न्यून कोण से एक दूसरे का विस्तार होता है,
और इज़ाफ़ा लंबाई के अनुपात को संरक्षित करते हैं
(\cref{ch:g8:midpoints} यह कहानी शुरू हुई; \cref{ch:g9:thales}
इसे समाप्त करता है)।
\end{definition}

\begin{proposition}[प्रथम मान और सीमाएँ]\label{prop:g8:cosine:bounds}
प्रत्येक न्यून कोण के लिए $\theta $: $0 < \cos\theta < 1$, और कोसाइन
\emph{कम हो जाती है} जैसे ही कोण खुलता है: एक चौड़े कोण का एक छोटा कोण होता है
कोज्या स्थलचिह्न: $\cos 0^\circ = 1$, $\cos 60^\circ = \frac12$,
$\cos 90^\circ = 0$ (चरम मान चपटे के अनुरूप हैं
\omterm{def:g3:shapes:shapes}{त्रिकोण})।
\end{proposition}
\admitted

\begin{method}[कोसाइन का उपयोग करना]\label{met:g8:cosine:use}
में एक सही \omterm{def:g3:shapes:shapes}{त्रिकोण}, जब ज्ञात और वांछित मात्राएँ हों
न्यूनकोण, उसकी आसन्न भुजा और कर्ण:
\begin{enumerate}
  \item परिभाषा लिखें:
        $\cos\theta = \frac{\text{adjacent}}{\text{hypotenuse}}$ के साथ
        स्थान पर ज्ञात मान;
  \item छोटे को हल करें \omterm{def:g8:equations:def}{समीकरण} अज्ञात के लिए (गुणा या भाग);
  \item किसी अज्ञात के लिए \emph{कोण}, कैलकुलेटर का प्रयोग करें
        $\cos^{-1}$ परिकलित अनुपात के लिए;
  \item विवेक जांच: लंबाई इससे छोटी निकलनी चाहिए
        कर्ण; एक कोण बिल्कुल बीच में $0^\circ $ और
        $90^\circ $.
\end{enumerate}
\end{method}

\begin{example}[एक पक्ष ढूँढना]\label{ex:g8:cosine:side}
ए $6$ मी रैम्प का एक कोण बनाता है $20^\circ $ क्षैतिज भूमि के साथ.
तय की गई क्षैतिज दूरी (समीपवर्ती)। $20^\circ $, कर्ण
$6$ एम):
\[
\cos 20^\circ = \frac{d}{6}
\quad\Longrightarrow\quad
d = 6 \cos 20^\circ \approx 6 \times 0.940 \approx 5.6 \text{ m}.
\]
\end{example}

\begin{example}[एक कोण ढूँढना]\label{ex:g8:cosine:angle}
में एक सही \omterm{def:g3:shapes:shapes}{त्रिकोण}, कोण से सटी भुजा $\theta $ पैमाने
$3.5$ और कर्ण $5$:
\[
\cos\theta = \frac{3.5}{5} = 0.7,
\qquad
\theta = \cos^{-1}(0.7) \approx 45.6^\circ .
\]
\end{example}

\omterm{def:g3:shapes:rightangle}{समकोण} से माध्यिका शिखर? क्या है \omterm{def:g3:shapes:shapes}{RADIUS} का

\begin{exercise}[$\star $]\label{exo:g8:cosine:2}
का एक वृत्त बनाएं \omterm{def:g4:geometry:circle}{व्यास} $8$ ए के साथ सेमी \omterm{def:g4:geometry:circle}{व्यास} $[BC]$, कोई भी चुनें

\begin{exercise}[$\star $]\label{exo:g8:cosine:3}
एक \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ DEF $ \omterm{def:g6:shapes:triangles}{समकोण का} पर $ D $, बगल वाली भुजा का नाम बताएं
कोण $\widehat E $, इसके विपरीत पक्ष और कर्ण।
लिखना $\cos \widehat E $ अनुपात के रूप में.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:cosine:4}
लुप्त मात्रा की गणना करें ($\cos 35^\circ \approx 0.819$,
$\cos 50^\circ \approx 0.643$):
\begin{enumerate}
  \item कर्ण $10$, कोण $35^\circ $: बगल की ओर?
  \item बगल की ओर $6$, कोण $50^\circ $: कर्ण?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:cosine:5}
में एक सही \omterm{def:g3:shapes:shapes}{त्रिकोण}, के समीपवर्ती पक्ष $\theta $ पैमाने $8$ और
कर्ण $10$. गणना करना $\cos\theta $, तब $\theta $
($\cos^{-1}(0.8) \approx 36.9^\circ $).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:cosine:6}
समझाइए क्यों $\cos\theta $ कभी बराबरी नहीं कर सकता $1.2$ के न्यून कोण के लिए
ए सही \omterm{def:g3:shapes:shapes}{त्रिकोण}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:cosine:7}
ए $25$ m zip line descends from a platform to the ground, making an
का कोण $12^\circ $ क्षैतिज के साथ ($\cos 12^\circ \approx

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:cosine:8}
कैलकुलेटर के बिना ऑर्डर करें: $\cos 20^\circ $, $\cos 70^\circ $,

कैलकुलेटर.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:cosine:9}
लम्बाई की एक सीढ़ी $ L $ एक कोण से दीवार के सहारे झुक जाता है $\theta $
सीढ़ी और के बीच \emph{मैदान}. इसका पैर है $1.2$ दीवार से मी
और $\theta = 68^\circ $ ($\cos 68^\circ \approx 0.375$). गणना करना $ L $,
फिर ऊंचाई (पाइथागोरस) तक पहुंच गई।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:cosine:10}
$[BC]$ एक है \omterm{def:g4:geometry:circle}{व्यास} केंद्र के एक वृत्त का $ O $ और \omterm{def:g3:shapes:shapes}{RADIUS} $4.5$ सेमी,
और $ A $ वृत्त का एक बिंदु है $ AB = 5.4$ सेमी।
\begin{enumerate}
  \item क्यों $ ABC $ \omterm{def:g6:shapes:triangles}{समकोण का} पर $ A $?
  \item गणना करना $ AC $, तब $\cos \widehat B $.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g8:cosine:11}
का उपयोग करते हुए \omterm{def:g3:division:half}{आधा} एक समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण} इस पक्ष की $1$ (ऊंचाई के साथ काटें),
मील के पत्थर को उचित ठहराएँ $\cos 60^\circ = \frac12$ का
\cref{prop:g8:cosine:bounds}. (ऊंचाई का पैर कहां है
आधार पर गिरना? पूर्ण त्रिकोणमितीय संस्करण है

\end{exercise}

\section{समस्या: यूक्लिड के संबंध, और पाइथागोरस सभी जगह
पुनः}

\begin{problem}[{सप्ताहांत समस्या---दाहिनी ओर की ऊंचाई त्रिकोण: $ AB^2 = BH \times BC $, $ AH^2 = BH \times HC $, एक सेकंड पाइथागोरस का प्रमाण, और वर्गमूल बनाने की एक मशीन}]
\label{pb:g8:cosine:1}
गिराओ, से \omterm{def:g3:shapes:rightangle}{समकोण} एक का सही \omterm{def:g3:shapes:shapes}{त्रिकोण}, द \omterm{def:g6:lines:perp}{सीधा}
कर्ण तक: यह लघु \omterm{def:g6:lines:objects}{खंड} --- द \emph{ऊंचाई} ---
संबंधों को इतना उपयोगी संतुष्ट करता है कि यूक्लिड ने उन्हें हृदय में स्थान दिया
उसका \emph{तत्वों}. इस समस्या में कोसाइन की परिभाषा
संपत्ति, ``अनुपात केवल कोण पर निर्भर करता है''
(\cref{def:g8:cosine:def}), उन सभी को सिद्ध करता है; रास्ते में तुम
पाइथागोरस के प्रमेय को बिल्कुल अलग मार्ग से पुनः सिद्ध करेगा,
और एक रूलर-और-कम्पास मशीन के साथ समाप्त होता है जो निर्माण करती है
$\sqrt2$, $\sqrt5$, $\sqrt{n}$ प्रत्येक पूर्ण संख्या के लिए $ n $.

लगातार, $ ABC $ एक \omterm{def:g3:shapes:shapes}{त्रिकोण} है \omterm{def:g6:shapes:triangles}{समकोण का} पर $ A $, और $ H $ है
का पैर \omterm{def:g6:lines:perp}{सीधा} से $ A $ कर्ण को $[BC]$,
ताकि $ H $ के बीच स्थित है $ B $ और $ C $ और $ AH \perp BC $.

\medskip
\noindent\textbf{Part I --- One angle, two triangles.}
\begin{enumerate}
  \item कोण का उपयोग करना \omterm{def:g1:addition:def}{जोड़} एक \omterm{def:g3:shapes:shapes}{त्रिकोण} का
        (\cref{thm:g7:triangles:sum}) में $ ABH $ और में $ ABC $, बताते हैं कि $\widehat{BAH} = \widehat{ACB}$: ऊंचाई में कटौती
        द सही \omterm{def:g3:shapes:shapes}{त्रिकोण} दो छोटे \omterm{def:g3:shapes:shapes}{त्रिकोण}ों में ले जाना
        द \emph{समान कोण} मूल के रूप में.
  \item \omterm{def:g3:shapes:shapes}{त्रिकोण} $ ABC $ (\omterm{def:g6:shapes:triangles}{समकोण का} पर $ A $) और $ ABH $
        (\omterm{def:g6:shapes:triangles}{समकोण का} पर $ H $) कोण साझा करें $\widehat B $. लिखना
        $\cos\widehat B $ उनमें से प्रत्येक में, और से निष्कर्ष निकालें
        \cref{def:g8:cosine:def} वह
        \[
        AB^2 = BH \times BC .
        \]
        (समान अनुपात से समान अनुपात में जाने के लिए \omterm{def:g2:mult:def}{उत्पाद}ों, गुणा करें
        दोनों तरफ से दोनों हरों,
        \cref{thm:g8:equations:balance}.)
  \item जुड़वाँ संबंध बताएं और सिद्ध करें शिखर $ C $:
        \[
        AC^2 = CH \times CB .
        \]
  \item ले लो $3$--$4$--$5$ \omterm{def:g3:shapes:shapes}{त्रिकोण}: $ AB = 3$, $ AC = 4$,
        $ BC = 5$. गणना करना $ BH $ और $ CH $ प्रश्न 2 और 3 से,
        और उसे जांचें $ BH + HC = BC $.
  \item सामान्य तौर पर, व्यक्त करें $ BH $ और $ CH $ जैसा \omterm{def:g6:fractions:def}{अंश}ों सम्मिलित
        केवल तीन पक्ष. पैर किस अर्थ में है $ H $
        कर्ण को `` के वर्गों के \omterm{def:g6:propdata:def}{आनुपातिक} रूप से विभाजित करें
        पैर''?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Pythagoras again, and the altitude.}
\begin{enumerate}[resume]
  \item प्रश्न 2 और 3 के संबंध \omterm{def:g1:addition:def}{जोड़}ें और उपयोग करें
        $ BH + HC = BC $ प्राप्त करने के लिए
        \[
        AB^2 + AC^2 = BC^2 :
        \]
        पाइथागोरस प्रमेय (\cref{thm:g8:pythagoras:direct}),
        पुनः सिद्ध --- नहीं के साथ \omterm{def:g6:measure:area}{क्षेत्र} दृष्टि में पहेली (तुलना करें
        \cref{exo:g8:pythagoras:11}).
  \item छोटे \omterm{def:g3:shapes:shapes}{त्रिकोण} में पाइथागोरस लगाएं $ ABH $ लिखना
        $ AH^2 = AB^2 - BH^2$, प्रतिस्थापित करें $ AB^2$ द्वारा
        $ BH \times BC $, और कारक निकालो $ BH $ साबित करना
        \emph{यूक्लिड का ऊंचाई संबंध}:
        \[
        AH^2 = BH \times HC .
        \]
  \item की गणना करें \omterm{def:g6:measure:area}{क्षेत्र} का $ ABC $ दो तरह से --- आधार के रूप में पैर
        और ऊंचाई, फिर कर्ण को आधार मानें
        (\cref{thm:g7:areas:triangle}) --- और तीसरा निष्कर्ष निकालें
        संबंध:
        \[
        AB \times AC = BC \times AH .
        \]
  \item वापस जाएँ $3$--$4$--$5$ \omterm{def:g3:shapes:shapes}{त्रिकोण}: गणना करें $ AH $ के साथ
        प्रश्न 8, फिर प्रश्न के ऊंचाई संबंध को सत्यापित करें
        7 संख्यात्मक रूप से, के मानों का उपयोग करते हुए $ BH $ और $ CH $ में पाया गया
        प्रश्न 4.
  \item ऊंचाई का पाद कुछ के कर्ण को विभाजित करता है सही \omterm{def:g3:shapes:shapes}{त्रिकोण} में \omterm{def:g6:lines:objects}{खंड}ों $ BH = 2$ सेमी और $ HC = 8$ सेमी।
        ऊंचाई की गणना करें $ AH $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- A machine for square roots.}
चित्रित करो \omterm{def:g6:lines:objects}{खंड} $[BC]$ एक सिरे से दूसरे सिरे तक रखे गए दो टुकड़ों से बना:
$ BH = p $ और $ HC = q $. का \omterm{def:g3:division:half}{आधा} वृत्त खींचिए \omterm{def:g4:geometry:circle}{व्यास} $[BC]$,
और चलो $ A $ वह बिंदु हो जहां \omterm{def:g6:lines:perp}{सीधा} को $(BC)$ पर $ H $
उससे मिलता है.
\begin{enumerate}[resume]
  \item का उपयोग करते हुए \cref{thm:g8:cosine:circle}, समझाएं क्यों
        \omterm{def:g3:shapes:shapes}{त्रिकोण} $ ABC $ है \omterm{def:g6:shapes:triangles}{समकोण का} पर $ A $ ---तो प्रश्न 7
        लागू होता है और
        \[
        AH^2 = p \times q .
        \]
        (लंबाई $ AH $ कहा जाता है \emph{ज्यामितीय
        मतलब}\index{ज्यामितीय
        मतलब} का $ p $ और $ q $.)
  \item लेना $ p = 1$ और $ q = 2$. क्या है $ AH^2$? जो प्रसिद्ध है
        की लंबाई \cref{ex:g8:pythagoras:diagonal} आपके पास है
        शासक-और-दिशासूचक आकृति अभी-अभी निर्मित हुई?
  \item उस नुस्खे का वर्णन करें जो इसका निर्माण करता है \omterm{def:g6:lines:objects}{खंड} लम्बाई का
        $\sqrt n $ किसी भी पूर्ण संख्या के लिए $ n \geq 1$, और कहो कौन सा
        का चुनाव $ p $ और $ q $ निर्माणों $\sqrt 5$.
  \item होने देना $ O $ अर्धवृत्त का केंद्र बनें. समझाइए क्यों
        $ AH $ से अधिक कभी नहीं हो सकता \omterm{def:g3:shapes:shapes}{RADIUS} $\frac{p + q}{2}$, और
        किस पद के लिए $ H $ दोनों बराबर हैं.
  \item असमानता निकालें: सभी सकारात्मक संख्याओं के लिए $ p $ और
        $ q $,
        \[
        p \times q \leq \left(\frac{p + q}{2}\right)^2,
        \]
        बिल्कुल समानता के साथ जब $ p = q $. फिर इसे धिक्कारें
        बिल्कुल कोई ज्यामिति नहीं: विस्तार करें
        $\left(\frac{p+q}{2}\right)^2 - p q$ और पहचानें ए
        उल्लेखनीय पहचान (\cref{pb:g8:equations:1}).
\end{enumerate}
\end{problem}
