\chapter{मध्यबिंदु और समानताएं}\label{ch:g8:midpoints}

त्रिभुज की दो भुजाओं के मध्यबिंदुओं को मिलाएँ: \omterm{def:g6:lines:objects}{खंड} तुम्हें मिल गया है
हमेशा \omterm{def:g6:lines:perp}{समानांतर} तीसरी तरफ, और बिल्कुल \omterm{def:g3:division:half}{आधा} जब तक. यह
``मध्यबिंदु प्रमेय'' और इसका व्युत्क्रम किसी महान का पहला स्वाद है
विचार --- समानताएं लंबाई को \omterm{def:g6:propdata:def}{आनुपातिक} रूप से काटें --- जो फूलती है
थेल्स प्रमेय में \cref{ch:g9:thales}.

\section{मध्यबिंदु प्रमेय}

\begin{theorem}[मध्यबिंदु प्रमेय]\label{thm:g8:midpoints:direct}
एक \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ABC $, होने देना $ I $ का मध्यबिंदु हो $[AB]$ और $ J $ द
का मध्यबिंदु $[AC]$. फिर लाइन $(IJ)$ है \omterm{def:g6:lines:perp}{समानांतर} को $(BC)$, और
\[
IJ = \frac{BC}{2} .
\]
\end{theorem}

\begin{proof}[प्रमाण का विचार]
चलो $ K $ हो \omterm{def:g7:central:def}{सममित} का $ I $ के बारे में $ J $ (\omterm{def:g3:division:half}{आधा} मोड़,
\cref{ch:g7:central}). \omterm{def:g4:geometry:polygon}{चतुष्कोष} $ AICK $ विकर्ण हैं $[IK]$
और $[AC]$ अपने सामान्य मध्यबिंदु पर पार करना $ J $: यह एक है
\omterm{def:g7:central:parallelogram}{चतुर्भुज}, इसलिए $ KC $ है \omterm{def:g6:lines:perp}{समानांतर} को $ AI $, यानी\ को $(AB)$, और
$ KC = AI = IB $. तब $ IBCK $ इसकी दो विपरीत भुजाएँ हैं ($[IB]$ और
$[KC]$) \omterm{def:g6:lines:perp}{समानांतर} और बराबर: यह एक है \omterm{def:g7:central:parallelogram}{चतुर्भुज} बहुत
(\cref{thm:g7:central:recognize}), इसलिए $(IK) \parallel (BC)$ ---
वह है $(IJ) \parallel (BC)$ --- और $ BC = IK = 2\,IJ $.
\end{proof}

\begin{example}\label{ex:g8:midpoints:direct}
एक \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ ABC $ साथ $ BC = 9$ सेमी, मध्यबिंदु $ I $ का $[AB]$ और
$ J $ का $[AC]$ जुड़े हुए हैं. बिना किसी माप के:
$(IJ) \parallel (BC)$ और $ IJ = \frac92 = 4.5$ सेमी। छोटा \omterm{def:g3:shapes:shapes}{त्रिकोण}
$ AIJ $ की आधे आकार की प्रति है $ ABC $ --- इसका \omterm{def:g6:measure:perimeter}{परिधि} है \omterm{def:g3:division:half}{आधा} 
\omterm{def:g6:measure:perimeter}{परिधि} का $ ABC $ भी।
\end{example}

\begin{theorem}[उलटा]\label{thm:g8:midpoints:converse}
एक \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ ABC $, मध्यबिंदु से होकर जाने वाली \omterm{def:g6:lines:objects}{रेखा} $ I $ का $[AB]$ और
\emph{\omterm{def:g6:lines:perp}{समानांतर}} को $(BC)$ ओर पार कर जाता है $[AC]$ इसके मध्यबिंदु पर.
\end{theorem}
\admitted

\begin{example}\label{ex:g8:midpoints:converse}
एक पथ एक \omterm{def:g3:shapes:shapes}{त्रिकोण}ीय \omterm{def:g6:measure:area}{क्षेत्र} को पार करता है, जो एक किनारे के मध्य से शुरू होता है
और चल रहा है \omterm{def:g6:lines:perp}{समानांतर} विपरीत किनारे पर. बातचीत गारंटी देती है
कि रास्ता ठीक दूसरे किनारे के मध्य से निकलता है --- नहीं
दूर की ओर मापने की आवश्यकता है।
\end{example}

\begin{method}[प्रमेय और व्युत्क्रम के बीच चयन करना]\label{met:g8:midpoints:choose}
\begin{enumerate}
  \item \emph{दो मध्यबिंदु ज्ञात} $\Rightarrow $ प्रत्यक्ष का उपयोग करें
        प्रमेय: समांतरता और अर्ध-लंबाई का निष्कर्ष निकालें।
  \item \emph{एक मध्यबिंदु और एक \omterm{def:g6:lines:perp}{समानांतर} ज्ञात} $\Rightarrow $ उपयोग करें
        व्युत्क्रम: निष्कर्ष निकालें कि क्रॉसिंग बिंदु एक मध्यबिंदु है।
  \item आप कौन सा त्रिभुज, कौन सा मध्यबिंदु और कौन सा प्रमेय लिखें
        --- प्रत्येक एक वाक्य का प्रयोग करें।
\end{enumerate}
\end{method}

\section{आधे आकार के त्रिकोण}

\begin{proposition}[मध्यबिंदु त्रिभुज]\label{prop:g8:midpoints:medial}
एक त्रिभुज की भुजाओं के तीन मध्यबिंदुओं को मिलाने से यह कट जाता है
बराबर के चार \omterm{def:g3:shapes:shapes}{त्रिकोण} \omterm{def:g6:measure:area}{क्षेत्र}ों, प्रत्येक मूल की आधे आकार की प्रति।
\end{proposition}

\begin{proof}
मध्यबिंदु त्रिभुज की प्रत्येक भुजा है \omterm{def:g6:lines:perp}{समानांतर} के एक तरफ $ ABC $ और
\omterm{def:g3:division:half}{आधा} जब तक (\cref{thm:g8:midpoints:direct}, तीन बार लागू किया गया)।
इसलिए, चार छोटे त्रिभुजों की भुजाएँ समान तीन लंबाई की होती हैं
वे समान प्रतियाँ हैं; वे एक साथ टाइल लगाते हैं $ ABC $, इसलिए प्रत्येक के पास एक है
\omterm{def:g3:division:half}{तिमाही} उसके जैसा \omterm{def:g6:measure:area}{क्षेत्र}.
\end{proof}

\begin{remark}[थेल्स की ओर]
मध्यबिंदु प्रमेय मामला ``एक'' है \omterm{def:g3:division:half}{आधा}'' एक अधिक सामान्य तथ्य का:
एक पंक्ति \omterm{def:g6:lines:perp}{समानांतर} त्रिभुज की एक भुजा से अन्य दो भुजाओं को काट दिया जाता है
\emph{समान अनुपात} --- एक तिहाई और एक तिहाई, दो पांचवां और दो
पाँचवाँ\,\dots\ वह सामान्य कथन थेल्स का प्रमेय है
(\cref{ch:g9:thales}); केवल मध्यबिंदु मामला ही इसकी आवश्यकता है
वर्ष.
\end{remark}

\begin{exercise}[$\star\star $]\label{exo:g8:midpoints:8}
\omterm{def:g3:shapes:shapes}{त्रिकोण} में $ ABC $, \omterm{def:g6:shapes:triangles}{समकोण का} पर $ A $, $ M $ का मध्यबिंदु है

\end{exercise}

\section{समस्या: तीन माध्यिकाएँ और गुरुत्वाकर्षण का केंद्र}

\begin{problem}[{सप्ताहान्त समस्या---एक त्रिभुज की माध्यिकाएँ मिलती हैं एक ही बिंदु पर, उनमें से प्रत्येक के नीचे दो तिहाई रास्ता}]
\label{pb:g8:midpoints:1}
कड़े कार्डबोर्ड से एक \omterm{def:g3:shapes:shapes}{त्रिकोण} काटें और यह संतुलन बनाएगा,
बिल्कुल सपाट, एक पेंसिल की नोक पर एक वि\omterm{def:g3:division:remainder}{शेष} बिंदु पर रखा गया:
\emph{ग्रैविटी केंद्र}\index{ग्रैविटी केंद्र} \omterm{def:g3:shapes:shapes}{त्रिकोण} का.
यह समस्या उस बिंदु को ढूंढती है। ए \emph{MEDIAN}\index{MEDIAN} एक का
\omterm{def:g3:shapes:shapes}{त्रिकोण} है \omterm{def:g6:lines:objects}{खंड} एक में शामिल होना शिखर तक \emph{मध्य} का
विपरीत पक्ष. आप मध्यबिंदु प्रमेय से सिद्ध करेंगे
दो बार प्रयोग किया गया, अप्रत्याशित तरीके से --- कि सभी तीन माध्यिकाएँ
एक बिंदु से गुजरें, और उसका सटीक पता लगाएं।

\medskip
\noindent\textbf{Part I --- Warm-up.}
\begin{enumerate}
  \item एक बड़ा त्रिभुज बनाएं $ ABC $ (कोई वि\omterm{def:g3:division:remainder}{शेष} आकार नहीं), निर्माण
        मध्यबिंदु $ I $ का $[AB]$, $ J $ का $[AC]$ और $ K $ का
        $[BC]$, और तीन माध्यिकाएं बनाएं $[AK]$, $[BJ]$, $[CI]$.
        आप क्या निरीक्षण करते हैं?
  \item सिद्ध कीजिए कि एक माध्यिका त्रिभुज को दो त्रिभुजों में काटती है
        बराबर का \omterm{def:g6:measure:area}{क्षेत्र}ों. (आधार और ऊंचाई की तुलना करें, और इसका उपयोग करें
        \omterm{def:g6:measure:area}{क्षेत्र} सूत्र: आधार $\times $ ऊंचाई $\div\ 2$.) एक के लिए
        का \omterm{def:g3:shapes:shapes}{त्रिकोण} \omterm{def:g6:measure:area}{क्षेत्र} $36$ सेमी $^2$, दो टुकड़े कौन से हैं'
        \omterm{def:g6:measure:area}{क्षेत्र}ों?
  \item \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ ABC $, क्या करता है
        \cref{thm:g8:midpoints:direct} के बारे में कहो \omterm{def:g6:lines:objects}{खंड}
        $[IJ]$?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Two medians meet two thirds of the
way.}
माध्यिकाएँ $[BJ]$ और $[CI]$ एक बिंदु पर पार करें; इसे कहते हैं $ G $. होने देना
$ M $ का मध्यबिंदु हो $[GB]$ और $ N $ का मध्यबिंदु $[GC]$.
\begin{enumerate}[resume]
  \item मध्यबिंदु प्रमेय लागू करें \emph{\omterm{def:g3:shapes:shapes}{त्रिकोण} में $ GBC $}:
        इसके बारे में क्या कहता है \omterm{def:g6:lines:objects}{खंड} $[MN]$?
  \item उसका निष्कर्ष निकालो $(IJ) \parallel (MN)$ और $ IJ = MN $, और
        के साथ समाप्त करें \cref{thm:g7:central:recognize} वह $ IJNM $
        एक है \omterm{def:g7:central:parallelogram}{चतुर्भुज}.
  \item समझाइए क्यों $ I $ और $ N $ दोनों माध्यिका पर स्थित हैं $(CI)$,
        और $ J $ और $ M $ दोनों मध्यिका पर $(BJ)$ ---तो
        के विकर्ण \omterm{def:g7:central:parallelogram}{चतुर्भुज} $ IJNM $ हैं $[IN]$ और
        $[JM]$, और वे बिल्कुल पार करते हैं $ G $. क्या करता है
        \emph{परिभाषा} एक का \omterm{def:g7:central:parallelogram}{चतुर्भुज} फिर इसके बारे में कहें $ G $?
  \item उसका निष्कर्ष निकालो $ BM = MG = GJ $, और निष्कर्ष निकालें:
        \[
        BG = 2 \times GJ
        \]
        --- मुद्दा $ G $ माध्यिका पर बैठता है $[BJ]$ दो तिहाई पर
        रास्ते से शिखर $ B $. बताएं और औचित्य सिद्ध करें
        माध्यिका पर समान परिणाम $[CI]$.
  \item अब भूल जाओ $[CI]$ और वही तर्क चलाएँ
        मध्यस्थों की \omterm{def:g1:addition:def}{जोड़}ी $[BJ]$ और $[AK]$: उनका क्रॉसिंग प्वाइंट
        का भी दो-तिहाई बैठता है $[BJ]$ से $ B $. स्पष्ट करें क्यों
        यह इसे होने के लिए मजबूर करता है \emph{वही} बिंदु $ G $ --- और
        निष्कर्ष निकालें कि सभी तीन माध्यिकाएं गुजरती हैं $ G $.
  \item एक त्रिभुज में जहाँ माध्यिका है $[BJ]$ पैमाने $9$ सेमी,
        कितनी दूर है $ G $ से $ B $, और से $ J $?
  \item उसे (एक वाक्य में) उचित ठहराइए $ AG = 2 \times GK $ भी।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Six equal slices, and coordinates.}
तीन माध्यिकाएँ त्रिभुज को छह छोटे त्रिभुजों में काटती हैं
चारों ओर $ G $.
\begin{enumerate}[resume]
  \item \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ GBC $, द \omterm{def:g6:lines:objects}{खंड} $[GK]$ एक माध्यिका है.
        यह निष्कर्ष निकालें कि दो छोटे त्रिभुज हैं $ GBK $ और $ GKC $ है
        बराबर \omterm{def:g6:measure:area}{क्षेत्र}ों.
  \item त्रिभुजों में प्रश्न 2 का उपयोग करना $ ABK $ और $ AKC $ (दोनों
        मध्यिका द्वारा काटा गया $[AK]$ का $ ABC $), दिखाएँ कि
        \omterm{def:g3:shapes:shapes}{त्रिकोण} $ ABG $ और $ ACG $ बराबर है \omterm{def:g6:measure:area}{क्षेत्र}ों. दोहराना
        एक अन्य माध्यिका के साथ, निष्कर्ष निकालें कि तीन त्रिभुज
        $ ABG $, $ BCG $, $ CAG $ सब के पास है \omterm{def:g6:measure:area}{क्षेत्र} का एक तिहाई $ ABC $.
  \item इन तीनों त्रिभुजों में से प्रत्येक को काटा गया है \omterm{def:g3:division:half}{आधा} एक टुकड़े से
        माध्यिका का (उदाहरण के लिए $[GI]$ का माध्यिका है
        \omterm{def:g3:shapes:shapes}{त्रिकोण} $ ABG $ ---जिससे देखा गया शिखर?). निष्कर्ष: छह
        चारों ओर छोटे टुकड़े $ G $ सभी के पास समान है \omterm{def:g6:measure:area}{क्षेत्र}, एक छठा
        \omterm{def:g3:shapes:shapes}{त्रिकोण} का.
  \item एक समन्वय ग्रिड पर, प्लॉट करें $ A(0, 0)$, $ B(6, 0)$ और
        $ C(0, 6)$, साथ $ K(3, 3)$ का मध्यबिंदु $[BC]$. का प्रयोग करें
        निर्देशांक की गणना करने के लिए भाग II का दो-तिहाई प्रमेय
        का $ G $ माध्यिका पर $[AK]$; फिर वही जांचें
        बिंदु माध्यिका के दो तिहाई पर बैठता है $[BJ]$, कहाँ
        $ J(0, 3)$. तुलना करना $ G $का औसत के साथ समन्वय है
        \[
        \frac{0 + 6 + 0}{3},
        \qquad
        \frac{0 + 0 + 6}{3} .
        \]
  \item \omterm{def:g3:division:remainder}{शेष} समापन: प्रश्न 2 और 8 का उपयोग करते हुए समझाएं कि क्यों
        कार्डबोर्ड \omterm{def:g3:shapes:shapes}{त्रिकोण} को चाकू की धार पर संतुलित किया जाता है
        किसी भी माध्यिका के अनुदिश --- प्रत्येक पर समान मात्रा में कार्डबोर्ड
        पक्ष --- और इसलिए पेंसिल की नोक क्यों रखी जानी चाहिए
        पर $ G$, वह बिंदु जिसे भौतिक विज्ञानी कहते हैं \omterm{pb:g8:midpoints:1}{ग्रैविटी केंद्र}. (पूरी तरह से कठोर संतुलन प्रमाण की आवश्यकता है
        विश्वविद्यालय का अभिन्न कलन संस्करणों; बराबर
        \omterm{def:g6:measure:area}{क्षेत्र}ों आज ही इसे विश्वसनीय बनाएं।)
\end{enumerate}
\end{problem}
