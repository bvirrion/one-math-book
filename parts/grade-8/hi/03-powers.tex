\chapter{पॉवर्स}\label{ch:g8:powers}

कागज की एक शीट को मोड़ो \omterm{def:g3:division:half}{आधा} $10$ समय (यदि आप कर सकते हैं!): इसकी मोटाई
हर बार दोगुना हो जाता है, और $10$ इसे दोगुना करके गुणा करें
$2^{10} = 1024$. घातें बार-बार गुणन का आशुलिपि हैं;
यह अध्याय एक वि\omterm{def:g3:division:remainder}{शेष} भूमिका के साथ अंकन और उसके नियमों को स्थापित करता है
की शक्तियों के लिए $10$.

\section{परिभाषा}

\begin{definition}[शक्ति]\label{def:g8:powers:def}
एक नंबर के लिए $ a $ और एक पूर्ण संख्या $ n \geq 1$:
\[
a^n = \underbrace{a \times a \times \dots \times a}_{n
\text{ factors}},
\]
पढ़ें ``$ a $ (शक्ति) को $ n $''; $ a $ है \emph{आधार}, $ n $ 
\emph{प्रतिपादक}\index{प्रतिपादक}. वि\omterm{def:g3:division:remainder}{शेष} नाम: $ a^2$ `` है $ a$
वर्ग'', $ a^3$ ``$ a $ घनाकार''। रिवाज के सन्दर्भ मे:
\[
a^1 = a,
\qquad
a^0 = 1 \ (a \neq 0),
\qquad
a^{-n} = \frac{1}{a^n}.
\]
\end{definition}

\begin{example}\label{ex:g8:powers:def}
$3^4 = 3 \times 3 \times 3 \times 3 = 81$; \quad
$10^3 = 1000$; \quad $5^{-2} = \frac{1}{25} = 0.04$; \quad
$(-2)^3 = -8$ और $(-2)^4 = +16$
(\cref{ex:g8:negprod:powers}). ध्यान रहें: $ a^n $ है \emph{नहीं}
$ a \times n $: \ $2^5 = 32$, नहीं $10$.
\end{example}

बार-बार गुणा करने से वृद्धि की तुलना में बार-बार होने वाली वृद्धि बहुत तेजी से खत्म हो जाती है
इसके अलावा.}
\end{omfigure}

\section{घातांक के नियम}

\begin{theorem}[घातांक के नियम]\label{thm:g8:powers:rules}
\omterm{def:g1:counting:zero}{शून्य}ेतर आधार के लिए $ a $ और पूर्णांक घातांक $ m $, $ n $:
\[
a^m \times a^n = a^{m+n},
\qquad
\frac{a^m}{a^n} = a^{m-n},
\qquad
\left(a^m\right)^n = a^{m \times n},
\qquad
(ab)^n = a^n b^n .
\]
\end{theorem}

\begin{proof}[कारकों की गिनती द्वारा प्रमाण]
$ a^m \times a^n $ \omterm{def:g6:lines:objects}{रेखा} के ऊपर $ m $ कारकों $ a $ के बाद $ n $ अधिक:
$ m + n $ कुल मिलाकर। $\left(a^m\right)^n $ का एक ब्लॉक दोहराता है $ m $ कारकों
$ n $ समय: $ mn $ कारक. $(ab)^n $ रोकना $ n $ पत्र $ a $ और $ n $
पत्र $ b $, जिसे पुनः समूहीकृत किया जा सकता है। \omterm{def:g3:division:remainder}{भागफल} नियम से आता है
रद्द करना $ n $ की $ m $ कारक; सम्मेलनों के साथ
$ a^0 = 1$ और $ a^{-n} = \frac{1}{a^n}$, यह तब भी सत्य रहता है
$ n \geq m $.
\end{proof}

\begin{example}\label{ex:g8:powers:rules}
\[
2^3 \times 2^5 = 2^8 = 256,
\qquad
\frac{7^6}{7^4} = 7^2 = 49,
\qquad
\left(5^2\right)^3 = 5^6,
\qquad
2^4 \times 5^4 = 10^4 .
\]
एक \omterm{def:g5:solids:net}{जाल}: नियम एक पर लागू होते हैं \emph{सामान्य आधार} (या एक आम
\omterm{def:g8:powers:def}{प्रतिपादक}, आखिरी वाले के लिए)। कोई भी नियम सरल नहीं होता $2^3 \times 5^2$ ---
बस गणना करें: $8 \times 25 = 200$.
\end{example}

\section{दस की शक्तियाँ}

\begin{proposition}[दस की शक्तियाँ]\label{prop:g8:powers:ten}
के लिए $ n \geq 1$: $10^n = 1\underbrace{0\dots0}_{n}$, और
$10^{-n} = 0.\underbrace{0\dots0}_{n-1}1$. के नियम घातांक
पढ़ें: दस की घातों को गुणा करने पर यह जुड़ जाता है घातांक.
\end{proposition}

\begin{example}\label{ex:g8:powers:ten}
$10^4 \times 10^3 = 10^7$; \quad
$\dfrac{10^2}{10^5} = 10^{-3} = 0.001$. बड़ी और छोटी मात्रा
पठनीय बनें:
\[
\text{one billion} = 10^9,
\qquad
\text{one millionth} = 10^{-6}.
\]
दशमलव के साथ संयुक्त: $3.2 \times 10^5 = 320\,000$ और
$4.7 \times 10^{-3} = 0.0047$ (शिफ्ट करें \omterm{def:g4:decimals:point}{दशमलव बिंदु},
\cref{prop:g6:decimals:shift}). इस लेखन का व्यवस्थित उपयोग ---
\emph{वैज्ञानिक संकेतन} --- में विकसित किया गया है
\cref{ch:g9:fractions}.
\end{example}

\begin{example}[परिमाण के आदेश]\label{ex:g8:powers:magnitude}
प्रकाश भ्रमण करता है $3 \times 10^8$ एमएस; एक वर्ष के बारे में है
$3.2 \times 10^7$ सेकंड. अत: एक प्रकाश-वर्ष लगभग है
\[
3 \times 3.2 \times 10^{8+7} \approx 10 \times 10^{15} = 10^{16}
\text{ m}
\]
--- दस करोड़ अरब \omterm{def:g4:fractions:def}{मीटर}. दस की घातें खगोलीय बनाती हैं
गणनाएँ एक पंक्ति में फिट होती हैं।
\end{example}

\begin{method}[शक्तियों के साथ अभिव्यक्ति को सरल बनाना]\label{met:g8:powers:simplify}
\begin{enumerate}
  \item कारकों को आधार दर आधार समूहित करें;
  \item लागू करें \omterm{def:g8:powers:def}{प्रतिपादक} प्रत्येक आधार के भीतर नियम;
  \item \omterm{def:g3:division:remainder}{शेष} छोटी शक्तियों की गणना करें, या उत्तर को छोड़ दें
        यदि यह बड़ा है तो शक्ति।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:powers:simplify}
\[
\frac{3^5 \times 3^{-2} \times 4^2}{3^2}
= \frac{3^{5 + (-2)}}{3^2} \times 16
= 3^{3 - 2} \times 16
= 3 \times 16 = 48 .
\]
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g8:powers:1}
गणना करें:
\[
2^6, \qquad 3^3, \qquad 10^5, \qquad 1^{100}, \qquad 0.1^2, \qquad
6^0 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:powers:2}
गणना करें:
\[
(-3)^2, \qquad -3^2, \qquad (-1)^{15}, \qquad (-5)^3, \qquad
\left(\tfrac{2}{3}\right)^2 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:powers:3}
एकल शक्ति के रूप में लिखें:
\[
7^4 \times 7^5, \qquad
\frac{2^9}{2^3}, \qquad
\left(10^3\right)^4, \qquad
5^6 \times 5^{-2}, \qquad
3^4 \times 7^4 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:powers:4}
के रूप में लिखें \omterm{def:g6:decimals:places}{दशमलव संख्या}: $10^{-2}$; \quad $4 \times 10^3$; \quad
$2.5 \times 10^{-4}$; \quad $10^0$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:powers:5}
दस की घात के साथ लिखें: एक लाख; एक दसवाँ हिस्सा; दस अरब; $0.000\,001$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:powers:6}
सरल बनाएं, फिर गणना करें:
\[
\frac{10^7 \times 10^{-3}}{10^2},
\qquad
\frac{2^5 \times 2^4}{2^6},
\qquad
\left(2^2\right)^3 \times 2^{-4} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:powers:7}
सच या झूठ? झूठे को सुधारो.
\[
2^3 \times 2^4 = 2^{12}; \qquad
5^2 + 5^3 = 5^5; \qquad
(3^2)^4 = 3^8; \qquad
10^3 \times 10^3 = 100^3 .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:powers:8}
एक अफवाह फैलती है: पहले दिन, तीन लोग इसे जानते हैं; प्रत्येक दिन, प्रत्येक
जो व्यक्ति जानता है वह तीन नए लोगों को बताता है। संख्या को घात लगाकर लिखें
का \emph{नया} दिन में लोगों को सूचना दी गयी $4$, और गणना करें कि कितने लोग हैं
दिन 4 के अंत में अफवाह जानें (मूल तीन सहित)।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:powers:9}
कागज की एक शीट है $0.1$ मिमी मोटी, यानी\ $10^{-4}$ एम। इसे मोड़ना
हर बार इसकी मोटाई दोगुनी हो जाती है।
\begin{enumerate}
  \item बाद में मोटाई व्यक्त करें $10$ के रूप में मुड़ता है \omterm{def:g2:mult:def}{उत्पाद}, और
        इसे सेंटी\omterm{def:g4:fractions:def}{मीटर} में गिनें ($2^{10} = 1024$).
  \item बाद $42$ सिलवटों की मोटाई होगी
        $2^{42} \times 10^{-4}$ मी, साथ $2^{42} \approx 4.4 \times
        10^{12}$. दिखाएँ कि यह पृथ्वी-चन्द्रमा की दूरी से अधिक है,
        के बारे में $3.8 \times 10^8$ एम।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:powers:10}
कैलकुलेटर के बिना, सबसे छोटे से सबसे बड़े तक ऑर्डर करें:
\[
2^{10}, \qquad 10^3, \qquad 3^6, \qquad 5^4 .
\]
(प्रत्येक की गणना करें; $2^{10}$ और $10^3$ प्रसिद्ध पड़ोसी हैं।)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g8:powers:11}
जो बड़ा है, $2^{100}$ या $10^{30}$? उपयोग
$2^{10} = 1024 > 10^3$ तुलना करने के लिए
$2^{100} = \left(2^{10}\right)^{10}$ साथ
$\left(10^3\right)^{10}$.
\end{exercise}

\section{समस्या: शतरंज की बिसात और दो की शक्तियाँ}

\begin{problem}[{सप्ताहांत समस्या --- ज्यामितीय \omterm{def:g1:addition:def}{जोड़} $1 + 2 + 4 + \dots + 2^{n-1} = 2^n - 1$, एक प्रसिद्ध किंवदंती से लेकर बाइनरी नंबर}]
\label{pb:g8:powers:1}
किंवदंती: शतरंज का आविष्कार करने के लिए पुरस्कार के रूप में, ऋषि सिस्सा ने पूछा
बोर्ड के पहले वर्ग पर गेहूँ के एक दाने के लिए उसका राजा,
दूसरे पर दो, तीसरे पर चार --- वर्ग से दोगुना
वर्ग, चौसठवें तक। राजा को ऐसी विनम्रता पर हँसी आई।
यह समस्या यह गणना करती है कि राजा ने क्या वादा किया था, इसका उपयोग करते हुए \omterm{def:g8:powers:def}{प्रतिपादक}
के नियमों \cref{thm:g8:powers:rules}, और कहानी वहीं समाप्त होती है
गुप्त रूप से नेतृत्व करता है: प्रत्येक कंप्यूटर के अंदर बाइनरी नंबर।

\medskip
\noindent\textbf{Part I --- The doubling trick.}
के लिए $ n \geq 1$, होने देना $ S_n $ पर अनाज की कुल संख्या हो
प्रथम $ n $ वर्ग।
\begin{enumerate}
  \item वर्ग पर दानों की संख्या व्यक्त करें $ k $ की शक्ति के रूप में
        $2$. कौन सी शक्ति वर्ग पर विराजमान है $64$?
  \item गणना करना $ S_1$, $ S_2$, $ S_3$, $ S_4$ और $ S_5$, और तुलना करें
        प्रत्येक के पास पास की शक्ति है $2$. के लिए एक सूत्र का अनुमान लगाएं
        $ S_n $.
  \item  \emph{दोहरीकरण युक्ति}: लिखना रकम $ S_n $ और
        $2 \times S_n $ एक के नीचे एक, घटाना और सिद्ध करना
        आपका अनुमान:
        \[
        S_n = 1 + 2 + 4 + \dots + 2^{n-1} = 2^n - 1 .
        \]
  \item राजा ने कुल कितने अनाज देने का वादा किया था? व्यक्त करें
        की शक्ति से उत्तर दें $2$, और क्लासिक को पूरा करें
        टिप्पणी: ``पूरे बोर्ड में एक दाना कम होता है
        ''एक पैंसठवाँ वर्ग होगा।''
  \item वह दूसरा दिखाओ \omterm{def:g3:division:half}{आधा} बोर्ड का (वर्ग) $33$ को
        $64$) धारण करता है \emph{बिल्कुल} $2^{32}$ कई गुना अधिक अनाज
        पहले के रूप में \omterm{def:g3:division:half}{आधा}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- How big is $2^{64}$?}
तुलना $2^{10} = 1024 > 10^3$ का
\cref{exo:g8:powers:11} नीचे दिए गए सभी अनुमानों की कुंजी है।
\begin{enumerate}[resume]
  \item बताते हैं कि
        $2^{64} = 2^4 \times \left(2^{10}\right)^6 > 1.6 \times
        10^{19}$.
  \item गेहूँ के एक दाने का वजन लगभग होता है $0.05$ जी, यानी\
        $5 \times 10^{-2}$ जी। दिखाएँ कि वादा किया गया गेहूँ तौला गया है
        से भी अधिक $8 \times 10^{17}$ जी, और इसे इसमें परिवर्तित करें
        टन ($1$ टन $= 10^6$ जी)।
  \item पूरी दुनिया इस समय लगभग फसल काट रही है $8 \times 10^8$
        प्रति वर्ष टन गेहूँ. कम से कम कितने साल की दुनिया
        क्या राजा ने फसल काटने का वादा किया था?
  \item सबसे छोटी पूर्ण संख्या ज्ञात कीजिये $ n $ ऐसा है कि
        $2^n > 10^6$ ---अर्थात् कितने दोगुने होने में समय लगता है
        दस लाख पार करो. (गणना करें $2^{19}$ और $2^{20}$ बिलकुल,
        का उपयोग कर रहे हैं $2^{10} = 1024$.)
  \item एक चालबाज आपको एक महीने का वेतन प्रदान करता है: $1$ दिन पर शत प्रतिशत
        $1$, फिर हर दिन पिछले दिन का वेतन दोगुना करें। जिस पर
        दिन करता है \emph{दैनिक} पहले अकेले भुगतान करें एक से अधिक
        मिलियन यूरो ($10^8$ सेंट)? (गणना करें $2^{26}$ और
        $2^{27}$ बिल्कुल।)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Binary weights.}
एक व्यापारी के पास पाँच बाट होते हैं: $1$, $2$, $4$, $8$ और $16$ ग्राम,
प्रत्येक में से एक. वह उनमें से कुछ को तराजू के एक पलड़े पर रखती है
दूसरे पलड़े पर सामान तोलें.
\begin{enumerate}[resume]
  \item वह कौन सा बाट तौलने के लिए रखती है $21$ जी? वजन करना
        $27$ जी?
  \item बताएं कि कोई लक्ष्य क्यों $16$ जी या अधिक \emph{अवश्य} उपयोग $16$ जी वजन, और क्यों का कोई लक्ष्य $15$ जी या उससे कम
        अवश्य \emph{नहीं} इसका इस्तेमाल करें। (प्रश्न 3 आपको बताता है कि क्या है
        वजन $1, 2, 4, 8$ अधिक से अधिक पहुँच सकते हैं।) समझाएँ क्यों
        अगले सबसे बड़े वजन के साथ भी यही तर्क दोहराया जाता है
        हर चरण.
  \item प्रत्येक संपूर्ण लक्ष्य से यह निष्कर्ष निकालें $1$ को $31$ जी हो सकता है
        तौला, और \emph{बिल्कुल एक तरह से}: प्रत्येक संख्या
        के बीच $1$ और $31$ एक है \omterm{def:g1:addition:def}{जोड़} की विशिष्ट शक्तियों का $2$ में
        एक ही ढंग.
  \item व्यापारी छठा बाट खरीदता है $32$ जी। किस तक
        लक्ष्य क्या वह अब वजन कर सकती है? लिखना $45$ जी एक के रूप में \omterm{def:g1:addition:def}{जोड़} का
        की विशिष्ट शक्तियाँ $2$.
  \item कंप्यूटर में, एक ``64-बिट'' नंबर संग्रहीत होता है $64$
        वर्ग, प्रत्येक में एक है $0$ या ए $1$ --- वर्ग $ k $
        योगदान $2^{k-1}$ जब यह धारण करता है $1$, जैसे
        किंवदंती के दाने. भाग I का उपयोग करते हुए, समझाएँ कि संपूर्ण क्यों
        ऐसी मशीन जिन नंबरों को स्टोर कर सकती है वे बिल्कुल उसी से चल सकते हैं $0$ को
        $2^{64} - 1$.
\end{enumerate}
\end{problem}
