\chapter{पिरामिड और शंकु}\label{ch:g8:solids}

के बाद प्रिज्म और \omterm{def:g7:areas:prism}{सिलेंडर} का \cref{ch:g7:areas} --- एसएनएफ के साथ
निरंतर क्रॉस-सेक्शन --- आओ एसएनएफ एक बिंदु के साथ: \omterm{def:g8:solids:def}{पिरामिड} और
\omterm{def:g8:solids:def}{कोन}. उनका \omterm{def:g6:measure:volume}{आयतन} सूत्र एक प्रसिद्ध कारक रखता है $\frac13$, जो
यह अध्याय प्रशंसनीय बनाता है और उपयोग करता है; का अध्ययन एसएनएफ जारी है
\omterm{def:g4:numbers:round}{गोल}े और अनुभागों के साथ \cref{ch:g9:solids}.

\section{पिरामिड और शंकु का वर्णन}

\begin{definition}[पिरामिड, शंकु]\label{def:g8:solids:def}
\begin{itemize}
  \item ए \emph{पिरामिड}\index{पिरामिड} एक \omterm{def:g4:geometry:polygon}{बहुभुज} अपने रूप में
        \emph{आधार} और एक बिंदु, \emph{शीर्ष}, हर एक से जुड़ गया
        शिखर \omterm{def:g3:shapes:shapes}{त्रिकोण}ीय द्वारा आधार का \omterm{def:g5:solids:def}{चेहरे}. इसका \emph{ऊंचाई} है
        शीर्ष से आधार के तल तक की दूरी.
  \item ए \emph{कोन}\index{कोन} एक के ऊपर एक ही निर्माण है
        डिस्क: एक आधार डिस्क, एक शीर्ष और एक घुमावदार पार्श्व सतह।
\end{itemize}
एक पिरामिड है \emph{नियमित} जब इसका आधार नियमित हो \omterm{def:g4:geometry:polygon}{बहुभुज}
(वर्ग, समान भुजाओं वाला \omterm{def:g3:shapes:shapes}{त्रिकोण}, \dots) और इसका शीर्ष लंबवत बैठता है
आधार के केंद्र के ऊपर.
\end{definition}

\begin{example}[फलकों, किनारों, शीर्षों को गिनना]\label{ex:g8:solids:count}
ए \omterm{def:g8:solids:def}{पिरामिड} एक वर्गाकार आधार के साथ है $5$ \omterm{def:g5:solids:def}{चेहरे} ($1$ वर्ग $+ 4$
\omterm{def:g3:shapes:shapes}{त्रिकोण}), $8$ \omterm{def:g5:solids:def}{किनारों} और $5$ \omterm{def:g5:solids:def}{कोने}. एक साथ $ n $-पक्षीय आधार:
$ n + 1$ \omterm{def:g5:solids:def}{चेहरे}, $2n $ किनारे, $ n + 1$ \omterm{def:g5:solids:def}{कोने}. यूलर की थोड़ी जांच करें
पैटर्न: \omterm{def:g5:solids:def}{चेहरे} $+$ \omterm{def:g5:solids:def}{कोने} $=$ \omterm{def:g5:solids:def}{किनारों} $+ 2$.
\end{example}

\section{आयतन}

\begin{theorem}[पिरामिड या शंकु का आयतन]\label{thm:g8:solids:volume}
एक के लिए \omterm{def:g8:solids:def}{पिरामिड} या ए \omterm{def:g8:solids:def}{कोन} आधार के साथ \omterm{def:g6:measure:area}{क्षेत्र} $ B $ और ऊंचाई $ h $:
\[
V = \frac{1}{3}\, B \times h
\]
--- का एक तिहाई \omterm{def:g7:areas:prism}{चश्मे} या \omterm{def:g7:areas:prism}{सिलेंडर} समान आधार और ऊँचाई के साथ।
एक के लिए \omterm{def:g8:solids:def}{कोन} आधार का \omterm{def:g3:shapes:shapes}{RADIUS} $ r $: $ V = \frac13 \pi r^2 h $.
\end{theorem}
\admitted

\begin{remark}[एक तिहाई क्यों?]
एक खोखलापन भरें \omterm{def:g8:solids:def}{पिरामिड} पानी के साथ और इसे इसमें डालें \omterm{def:g7:areas:prism}{चश्मे} उसी का
आधार और ऊँचाई: इसमें ठीक तीन बार भरने की आवश्यकता होती है। और भी बेहतर: ए \omterm{def:g5:solids:def}{घनक्षेत्र}
तीन समान भागों में काटा जा सकता है \omterm{def:g8:solids:def}{पिरामिड}, प्रत्येक के पास एक है चेहरा की
\omterm{def:g5:solids:def}{घनक्षेत्र} आधार और एक के रूप में शिखर की \omterm{def:g5:solids:def}{घनक्षेत्र} शीर्ष के रूप में (इसे चित्रित करने का प्रयास करें!)
एक सामान्य प्रमाण एकीकरण का उपयोग करता है, जो हाई स्कूल वॉल्यूम का एक उपकरण है।
\end{remark}

\begin{example}\label{ex:g8:solids:volume}
ए \omterm{def:g8:solids:def}{पिरामिड} भुजा का आधार वर्गाकार है $5$ सेमी और ऊंचाई $9$ सेमी:
\begin{enumerate}
  \item आधार \omterm{def:g6:measure:area}{क्षेत्र}: $ B = 5^2 = 25$ सेमी $^2$;
  \item \omterm{def:g6:measure:volume}{आयतन}: $ V = \frac13 \times 25 \times 9 = 25 \times 3 = 75$
        सेमी $^3$.
\end{enumerate}
एक आइसक्रीम \omterm{def:g8:solids:def}{कोन} का \omterm{def:g3:shapes:shapes}{RADIUS} $3$ सेमी और ऊंचाई $10$ सेमी:
\[
V = \frac13 \times \pi \times 3^2 \times 10 = 30\pi \approx 94
\text{ cm}^3 .
\]
\end{example}

\begin{example}[ऊंचाई को लेकर सावधान रहें]\label{ex:g8:solids:slant}
एक के लिए \omterm{def:g8:solids:def}{कोन}, ऊंचाई को लेकर भ्रमित न हों $ h $ (के केंद्र के शीर्ष पर
आधार, \omterm{def:g6:lines:perp}{सीधा}) साथ \emph{तिरछी ऊंचाई} $ s $ (शीर्ष से
रिम). वे पाइथागोरस से संबंधित हैं
(\cref{thm:g8:pythagoras:direct}) में सही \omterm{def:g3:shapes:shapes}{त्रिकोण}
शीर्ष--केंद्र--रिम:
\[
s^2 = h^2 + r^2 .
\]
ए \omterm{def:g8:solids:def}{कोन} साथ $ r = 3$ और $ s = 5$ इसलिए ऊंचाई है
$ h = \sqrt{25 - 9} = 4$, और \omterm{def:g6:measure:volume}{आयतन}
$\frac13 \pi \times 9 \times 4 = 12\pi $.
\end{example}

\begin{method}[वॉल्यूम की समस्या]\label{met:g8:solids:method}
\begin{enumerate}
  \item की पहचान करें \omterm{def:g5:solids:def}{ठोस} (\omterm{def:g7:areas:prism}{चश्मे}/\omterm{def:g7:areas:prism}{सिलेंडर}: $ V = Bh $; \omterm{def:g8:solids:def}{पिरामिड}/\omterm{def:g8:solids:def}{कोन}:
        $ V = \frac13 Bh $);
  \item आधार की गणना करें \omterm{def:g6:measure:area}{क्षेत्र} $ B $ सबसे पहले, एक अलग कदम के रूप में;
  \item जांचें कि ऊंचाई कितनी है \omterm{def:g6:lines:perp}{सीधा} आधार के लिए (उपयोग करें)।
        पाइथागोरस यदि तिरछी ऊंचाई दी गई है);
  \item गुणा करें, इकाइयों को सुसंगत रखें, यदि आवश्यक हो तो अंत में परिवर्तित करें
        ($1$ एल $= 1000$ सेमी $^3$).
\end{enumerate}
\end{method}

\section{जाल}

\begin{example}[पिरामिड का जाल]\label{ex:g8:solids:net}
एक नियमित वर्गाकार आधार को खोलना \omterm{def:g8:solids:def}{पिरामिड} इसे अपने में समतल कर लेता है
\emph{\omterm{def:g5:solids:net}{जाल}}: चार समान के साथ वर्गाकार आधार \omterm{def:g6:shapes:triangles}{समद्विबाहु} त्रिभुज
इसके \omterm{def:g5:solids:def}{किनारों} से जुड़ा हुआ। त्रिभुजों की बराबर भुजाएँ हैं

गत्ता.
\end{example}

  \item शीर्ष से सबसे लंबे पार्श्व किनारे की लंबाई की गणना करें
        आधार के विपरीत \omterm{def:g5:solids:def}{कोने} में (पाइथागोरस के दो चरण:)।
        आधार विकर्ण पहले)।
\end{enumerate}
\end{exercise}

\section{समस्या: एक घन को तीन पिरामिडों में काटा गया}

\begin{problem}[{सप्ताहांत समस्या---जहां कारक $\frac13$ से आता है, और आधी ऊंचाई पर काटे गए पिरामिड का क्या होता है}]
\label{pb:g8:solids:1}
इसके बाद की टिप्पणी \cref{thm:g8:solids:volume} दावा है कि ए \omterm{def:g5:solids:def}{घनक्षेत्र}
तीन समान भागों में काटा जा सकता है \omterm{def:g8:solids:def}{पिरामिड} --- ``चित्र खींचने का प्रयास करें
यह!'' यह समस्या इसे सटीक रूप से चित्रित करती है, प्रसिद्ध कारक प्राप्त करती है
$\frac13$ इसमें से ईमानदारी से बाहर निकलें, फिर काट दें \omterm{def:g5:solids:def}{घनक्षेत्र} में दूसरा रास्ता
\emph{छह} \omterm{def:g8:solids:def}{पिरामिड}, और a को काटकर समाप्त करता है \omterm{def:g8:solids:def}{पिरामिड} पर \omterm{def:g3:division:half}{आधा} इसका
ऊँचाई --- परिणाम के साथ कुछ लोग पहली कोशिश में ही सही अनुमान लगा लेते हैं।

लगातार, $ ABCDEFGH $ एक है \omterm{def:g5:solids:def}{घनक्षेत्र} इस पक्ष की $ a $: आधार $ ABCD $, शीर्ष
चेहरा $ EFGH $, साथ $ E $ ऊपर $ A $, $ F $ ऊपर $ B $, $ G $ ऊपर $ C $ और
$ H $ ऊपर $ D $.

\medskip
\noindent\textbf{Part I --- Three \omterm{def:g8:solids:def}{पिरामिड} sharing an apex.}
\begin{enumerate}
  \item  शिखर $ G $ तीन का है \omterm{def:g5:solids:def}{चेहरे} की \omterm{def:g5:solids:def}{घनक्षेत्र}. सूची
        तीन \omterm{def:g5:solids:def}{चेहरे} वह करो \emph{नहीं} रोकना $ G $, और
        तीनों का वर्णन करें \omterm{def:g8:solids:def}{पिरामिड} प्रत्येक को लेकर प्राप्त किया गया
        उन्हें आधार के रूप में, शीर्ष के साथ $ G $ हर बार. कितने \omterm{def:g5:solids:def}{चेहरे},
        \omterm{def:g5:solids:def}{किनारों} और \omterm{def:g5:solids:def}{कोने} प्रत्येक के पास है \omterm{def:g8:solids:def}{पिरामिड}
        (\cref{ex:g8:solids:count})?
  \item समझाएं कि क्यों मुड़ रहे हैं \omterm{def:g5:solids:def}{घनक्षेत्र} एक तिहाई घुमाव से
        इसका विकर्ण लंबा है $(AG)$ छोड़ देता है \omterm{def:g5:solids:def}{घनक्षेत्र} अपरिवर्तित लेकिन
        प्रश्न 1 के तीन आधारों को एक चक्र में बदलता है ---
        और तीन क्यों \omterm{def:g8:solids:def}{पिरामिड} इसलिए समान हैं
        एक दूसरे की प्रतिलिपियाँ.
  \item यह स्वीकार करते हुए कि तीन \omterm{def:g8:solids:def}{पिरामिड} भरें \omterm{def:g5:solids:def}{घनक्षेत्र} बिना
        ओवरलैपिंग (एक कार्डबोर्ड मॉडल काफी \omterm{def:g5:solids:def}{ठोस} है),
        का निष्कर्ष निकालें \omterm{def:g6:measure:volume}{आयतन} हर एक का.
  \item के सूत्र से इसकी जाँच करें
        \cref{thm:g8:solids:volume}: द \omterm{def:g8:solids:def}{पिरामिड} आधार के साथ $ ABCD $
        और शीर्ष $ G $ इसका शीर्ष सीधे a के ऊपर है \emph{\omterm{def:g8:solids:def}{कोन}ा}
        इसके आधार का, बिल्कुल वैसा ही जैसा कि अंदर है \cref{exo:g8:solids:11}.
        इसकी ऊंचाई पहचानें, सूत्र लागू करें और तुलना करें।
  \item एक के लिए \omterm{def:g5:solids:def}{घनक्षेत्र} इस पक्ष की $6$ सेमी, दे दो \omterm{def:g6:measure:volume}{आयतन} प्रत्येक का
        तीन \omterm{def:g8:solids:def}{पिरामिड} --- और बताएं कि यह अपघटन क्या है
        के जल-उडेलने के प्रयोग से संबंधित है
        अध्याय (के तीन भरता है \omterm{def:g8:solids:def}{पिरामिड} एक के लिए \omterm{def:g7:areas:prism}{चश्मे}).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Six \omterm{def:g8:solids:def}{पिरामिड} sharing the center.}
अब चलो $ O $ का केंद्र हो \omterm{def:g5:solids:def}{घनक्षेत्र}, और शामिल हों $ O $ चार को
\omterm{def:g5:solids:def}{कोने} छह में से प्रत्येक का \omterm{def:g5:solids:def}{चेहरे}.
\begin{enumerate}[resume]
  \item छह का वर्णन करें \omterm{def:g8:solids:def}{पिरामिड} प्राप्त किया, और स्पष्ट किया कि वे क्यों हैं
        एक दूसरे की समरूप प्रतियाँ हैं। ऊंचाई कितनी है
        हर एक का?
  \item साझा करके, निष्कर्ष निकालें \omterm{def:g5:solids:def}{घनक्षेत्र}'एस \omterm{def:g6:measure:volume}{आयतन}, द \omterm{def:g6:measure:volume}{आयतन} प्रत्येक का
        छह में से \omterm{def:g8:solids:def}{पिरामिड}.
  \item उसकी पुनः गणना करें \omterm{def:g6:measure:volume}{आयतन} सूत्र के साथ
        $ V = \frac13 B h $, और जांचें कि दोनों उत्तर सहमत हैं।
  \item दो विघटन कारक की पुष्टि करते हैं $\frac13$ के लिए
        कुछ बहुत खास \omterm{def:g8:solids:def}{पिरामिड}. बताएं कि उन्होंने अभी तक ऐसा क्यों नहीं किया
        \emph{सिद्ध करना} \cref{thm:g8:solids:volume} हरएक के लिए
        \omterm{def:g8:solids:def}{पिरामिड} और \omterm{def:g8:solids:def}{कोन} --- और बताएं कि अध्याय किस प्रकार निपटता है
        वह \omterm{ex:g1:subtraction:difference}{अंतर}: कौन सा कथन स्वीकृत है, कौन सा प्रयोग
        इसका समर्थन करता है, और जिसमें \omterm{def:g6:measure:volume}{आयतन} श्रृंखला का यह है
        ईमानदारी से साबित हुआ.
  \item ए \omterm{def:g5:solids:def}{घनक्षेत्र} इस पक्ष की $6$ सेमी को इसके छह केंद्र में काटा जाता है
        \omterm{def:g8:solids:def}{पिरामिड}. की गणना करें \omterm{def:g6:measure:volume}{आयतन} प्रत्येक का, दोनों तरीकों से।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Cutting a \omterm{def:g8:solids:def}{पिरामिड} at \omterm{def:g3:division:half}{आधा} height.}
$ SABCD $ एक नियमित वर्ग-आधारित है \omterm{def:g8:solids:def}{पिरामिड}: आधार पक्ष $ c $, ऊंचाई
$ h $, शीर्ष $ S $ आधार के केंद्र के ऊपर. इसे हवाई जहाज़ से काटें
के माध्यम से \emph{मध्य बिन्दुओं} चार पार्श्व \omterm{def:g5:solids:def}{किनारों} में से $[SA]$,
$[SB]$, $[SC]$, $[SD]$.
\begin{enumerate}[resume]
  \item मध्यबिंदु प्रमेय लागू करें
        (\cref{thm:g8:midpoints:direct}) \omterm{def:g3:shapes:shapes}{त्रिकोण} में $ SAB $:
        की दिशा और लंबाई क्या है \omterm{def:g6:lines:objects}{खंड} शामिल होना
        के मध्यबिंदु $[SA]$ और $[SB]$? कटौती का अनुमान लगाएं
        भुजा का एक वर्ग है $\frac{c}{2}$, \omterm{def:g6:lines:perp}{समानांतर} आधार तक,
        ऊंचाई पर $\frac{h}{2}$.
  \item कट के ऊपर का टुकड़ा स्वयं वर्गाकार है \omterm{def:g8:solids:def}{पिरामिड}.
        इसकी आधार भुजा और ऊँचाई बताइए और दिखाइए कि यह \omterm{def:g6:measure:volume}{आयतन}
        बिलकुल है \emph{एक आठवां} मूल का \omterm{def:g6:measure:volume}{आयतन} $ V $.
  \item निष्कर्ष निकालें \omterm{def:g6:measure:volume}{आयतन} निचले टुकड़े का ( \emph{छिन्नक},
        एक अधूरे का आकार \omterm{def:g8:solids:def}{पिरामिड}). किस अनुपात में करता है
        आधी ऊंचाई की कटौती साझा करें \omterm{def:g6:measure:volume}{आयतन}?
  \item झिलमिली \omterm{def:g8:solids:def}{पिरामिड} (\cref{exo:g8:solids:7}: आधार पक्ष
        $35$ मी, ऊंचाई $22$ एम) को दो अभियानों में साफ किया गया है: द
        ऊपर का गिलास \omterm{def:g3:division:half}{आधा} ऊँचाई, फिर नीचे का शीशा। क्या
        \omterm{def:g6:fractions:def}{अंश} की \emph{\omterm{def:g6:measure:volume}{आयतन}} नीचे बैठता है \omterm{def:g3:division:half}{आधा} ऊंचाई?
        इसका अनुमान घन \omterm{def:g4:fractions:def}{मीटर} में, निकटतम सौ तक लगाएं।
  \item सामान्य नैतिक: यदि \emph{सभी} ए के आयाम
        \omterm{def:g8:solids:def}{पिरामिड} से गुणा किया जाता है $2$, इसका क्या होता है
        \omterm{def:g6:measure:volume}{आयतन}? किस कारक से होता है \omterm{def:g6:measure:volume}{आयतन} बढ़ो जब हर
        आयाम से गुणा किया जाता है $ k$ --- और यह कैसे होता है
        प्रश्न 12 के ``एक आठवें'' को एक पंक्ति में समझाएँ?
\end{enumerate}
\end{problem}
