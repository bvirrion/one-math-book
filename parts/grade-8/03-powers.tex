\chapter{Powers}\label{ch:g8:powers}

Fold a sheet of paper in \omterm{def:g3:division:half}{half} $10$ times (if you can!): its thickness
doubles each time, and $10$ doublings multiply it by
$2^{10} = 1024$. Powers are the shorthand of repeated multiplication;
this chapter sets up the notation and its rules, with a special role
for the powers of $10$.

\section{Definition}

\begin{definition}[Power]\label{def:g8:powers:def}
For a number $a$ and a whole number $n \geq 1$:
\[
a^n = \underbrace{a \times a \times \dots \times a}_{n
\text{ factors}},
\]
read ``$a$ to the (power) $n$''; $a$ is the \emph{base}, $n$ the
\emph{exponent}\index{exponent}. Special names: $a^2$ is ``$a$
squared'', $a^3$ ``$a$ cubed''. By convention:
\[
a^1 = a,
\qquad
a^0 = 1 \ (a \neq 0),
\qquad
a^{-n} = \frac{1}{a^n}.
\]
\end{definition}

\begin{example}\label{ex:g8:powers:def}
$3^4 = 3 \times 3 \times 3 \times 3 = 81$; \quad
$10^3 = 1000$; \quad $5^{-2} = \frac{1}{25} = 0.04$; \quad
$(-2)^3 = -8$ and $(-2)^4 = +16$
(\cref{ex:g8:negprod:powers}). Watch out: $a^n$ is \emph{not}
$a \times n$: \ $2^5 = 32$, not $10$.
\end{example}

\section{The rules of exponents}

\begin{theorem}[Rules of exponents]\label{thm:g8:powers:rules}
For a nonzero base $a$ and whole-number \omterm{def:g8:powers:def}{exponents} $m$, $n$:
\[
a^m \times a^n = a^{m+n},
\qquad
\frac{a^m}{a^n} = a^{m-n},
\qquad
\left(a^m\right)^n = a^{m \times n},
\qquad
(ab)^n = a^n b^n .
\]
\end{theorem}

\begin{proof}[Proof by counting factors]
$a^m \times a^n$ lines up $m$ factors $a$ followed by $n$ more:
$m + n$ in total. $\left(a^m\right)^n$ repeats a block of $m$ factors
$n$ times: $mn$ factors. $(ab)^n$ contains $n$ letters $a$ and $n$
letters $b$, which can be regrouped. The \omterm{def:g3:division:remainder}{quotient} rule comes from
canceling $n$ of the $m$ factors; with the conventions
$a^0 = 1$ and $a^{-n} = \frac{1}{a^n}$, it remains true even when
$n \geq m$.
\end{proof}

\begin{example}\label{ex:g8:powers:rules}
\[
2^3 \times 2^5 = 2^8 = 256,
\qquad
\frac{7^6}{7^4} = 7^2 = 49,
\qquad
\left(5^2\right)^3 = 5^6,
\qquad
2^4 \times 5^4 = 10^4 .
\]
A trap: the rules apply to a \emph{common base} (or a common
\omterm{def:g8:powers:def}{exponent}, for the last one). No rule simplifies $2^3 \times 5^2$ ---
just compute: $8 \times 25 = 200$.
\end{example}

\section{Powers of ten}

\begin{proposition}[Powers of ten]\label{prop:g8:powers:ten}
For $n \geq 1$: $10^n = 1\underbrace{0\dots0}_{n}$, and
$10^{-n} = 0.\underbrace{0\dots0}_{n-1}1$. The rules of \omterm{def:g8:powers:def}{exponents}
read: multiplying powers of ten adds the \omterm{def:g8:powers:def}{exponents}.
\end{proposition}

\begin{example}\label{ex:g8:powers:ten}
$10^4 \times 10^3 = 10^7$; \quad
$\dfrac{10^2}{10^5} = 10^{-3} = 0.001$. Large and small quantities
become readable:
\[
\text{one billion} = 10^9,
\qquad
\text{one millionth} = 10^{-6}.
\]
Combined with decimals: $3.2 \times 10^5 = 320\,000$ and
$4.7 \times 10^{-3} = 0.0047$ (shift the \omterm{def:g4:decimals:point}{decimal point},
\cref{prop:g6:decimals:shift}). The systematic use of this writing ---
\emph{scientific notation} --- is developed in
\cref{ch:g9:fractions}.
\end{example}

\begin{example}[Orders of magnitude]\label{ex:g8:powers:magnitude}
Light travels about $3 \times 10^8$ m/s; a year has about
$3.2 \times 10^7$ seconds. A light-year is therefore about
\[
3 \times 3.2 \times 10^{8+7} \approx 10 \times 10^{15} = 10^{16}
\text{ m}
\]
--- ten million billion meters. Powers of ten make astronomical
computations fit on one line.
\end{example}

\begin{method}[Simplifying an expression with powers]\label{met:g8:powers:simplify}
\begin{enumerate}
  \item Group the factors base by base;
  \item apply the \omterm{def:g8:powers:def}{exponent} rules within each base;
  \item compute the remaining small powers, or leave the answer as a
        power if it is large.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:powers:simplify}
\[
\frac{3^5 \times 3^{-2} \times 4^2}{3^2}
= \frac{3^{5 + (-2)}}{3^2} \times 16
= 3^{3 - 2} \times 16
= 3 \times 16 = 48 .
\]
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g8:powers:1}
Compute:
\[
2^6, \qquad 3^3, \qquad 10^5, \qquad 1^{100}, \qquad 0.1^2, \qquad
6^0 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:2}
Compute:
\[
(-3)^2, \qquad -3^2, \qquad (-1)^{15}, \qquad (-5)^3, \qquad
\left(\tfrac{2}{3}\right)^2 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:3}
Write as a single power:
\[
7^4 \times 7^5, \qquad
\frac{2^9}{2^3}, \qquad
\left(10^3\right)^4, \qquad
5^6 \times 5^{-2}, \qquad
3^4 \times 7^4 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:4}
Write as a \omterm{def:g6:decimals:places}{decimal number}: $10^{-2}$; \quad $4 \times 10^3$; \quad
$2.5 \times 10^{-4}$; \quad $10^0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:5}
Write with a power of ten: one hundred thousand; one tenth; ten
billion; $0.000\,001$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:6}
Simplify, then compute:
\[
\frac{10^7 \times 10^{-3}}{10^2},
\qquad
\frac{2^5 \times 2^4}{2^6},
\qquad
\left(2^2\right)^3 \times 2^{-4} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:7}
True or false? Correct the false ones.
\[
2^3 \times 2^4 = 2^{12}; \qquad
5^2 + 5^3 = 5^5; \qquad
(3^2)^4 = 3^8; \qquad
10^3 \times 10^3 = 100^3 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:powers:8}
A rumor spreads: on day 1, three people know it; each day, every
person who knows tells three new people. Write with a power the number
of \emph{new} people informed on day $4$, and compute how many people
know the rumor at the end of day 4 (including the original three).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:powers:9}
A sheet of paper is $0.1$ mm thick, i.e.\ $10^{-4}$ m. Folding it
doubles its thickness each time.
\begin{enumerate}
  \item Express the thickness after $10$ folds as a \omterm{def:g2:mult:def}{product}, and
        compute it in centimeters ($2^{10} = 1024$).
  \item After $42$ folds the thickness would be
        $2^{42} \times 10^{-4}$ m, with $2^{42} \approx 4.4 \times
        10^{12}$. Show that this exceeds the Earth--Moon distance,
        about $3.8 \times 10^8$ m.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:powers:10}
Order from smallest to largest, without a calculator:
\[
2^{10}, \qquad 10^3, \qquad 3^6, \qquad 5^4 .
\]
(Compute each one; $2^{10}$ and $10^3$ are famous neighbors.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:powers:11}
Which is larger, $2^{100}$ or $10^{30}$? Use
$2^{10} = 1024 > 10^3$ to compare
$2^{100} = \left(2^{10}\right)^{10}$ with
$\left(10^3\right)^{10}$.
\end{exercise}
