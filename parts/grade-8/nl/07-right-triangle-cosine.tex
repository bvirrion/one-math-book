\chapter{De rechthoekige driehoek en de cosinus}\label{ch:g8:cosine}

Twee mooie feiten verbinden de rechthoekige driehoek met de cirkel:
een rechthoekige driehoek past exact in een halve cirkel, waarbij de
schuine zijde een middellijn is. En één getal, de \emph{cosinus} van
een hoek, codeert de vorm van elke rechthoekige driehoek met die hoek
--- de eerste goniometrische verhouding, vooruitlopend op haar
broers sinus en tangens (\cref{ch:g9:trig}).

\section{De rechthoekige driehoek en zijn cirkel}

\begin{theorem}[Cirkelstelling]\label{thm:g8:cosine:circle}
\begin{enumerate}
  \item Als een driehoek $ABC$ rechthoekig is in $A$, dan ligt $A$ op
        de cirkel waarvan de middellijn de schuine zijde $[BC]$ is.
  \item Omgekeerd, als $A$ op een cirkel met middellijn $[BC]$ ligt
        (met $A \neq B, C$), dan is de driehoek $ABC$ rechthoekig in
        $A$.
\end{enumerate}
Equivalent: in een rechthoekige driehoek ligt het midden van de
schuine zijde even ver van de drie hoekpunten --- de mediaan vanuit
de rechte hoek meet de helft van de schuine zijde.
\end{theorem}

\begin{proof}[Bewijs van 1]
Laat $O$ het midden van $[BC]$ zijn en $D$ het spiegelbeeld van $A$
ten opzichte van $O$. De diagonalen van $ABDC$ snijden in hun
gemeenschappelijk midden $O$, dus $ABDC$ is een parallellogram
(\cref{def:g7:central:parallelogram}) --- met een rechte hoek in
$A$: het is een rechthoek. De diagonalen van een rechthoek zijn
gelijk, dus $OA = \frac{AD}{2} = \frac{BC}{2}$: het punt $A$ ligt op
afstand $\frac{BC}{2}$ van $O$, d.w.z.\ op de cirkel met middellijn
$[BC]$. (Punt 2 wordt bewezen door het argument achterstevoren te
lopen.)
\end{proof}

\begin{example}\label{ex:g8:cosine:circle}
Een driehoek heeft een schuine zijde van $10$ cm. Zonder iets anders
te weten meet de mediaan vanuit de rechte hoek $5$ cm, en de
omgeschreven cirkel van de driehoek heeft straal $5$ cm, met
middelpunt in het midden van de schuine zijde.
\end{example}

\section{De cosinus van een scherpe hoek}

\begin{definition}[Cosinus]\label{def:g8:cosine:def}
In een rechthoekige driehoek, voor een scherpe hoek $\theta$:
\[
\cos\theta = \frac{\text{aanliggende zijde bij } \theta}{\text{schuine zijde}}
\]
--- het been dat $\theta$ raakt, gedeeld door de schuine zijde.
Deze verhouding hangt alleen van de hoek af, niet van de grootte van
de driehoek: alle rechthoekige driehoeken met dezelfde scherpe hoek
zijn vergrotingen van elkaar, en vergrotingen behouden
lengteverhoudingen (\cref{ch:g8:midpoints} begon dit verhaal;
\cref{ch:g9:thales} maakt het af).
\end{definition}

\begin{proposition}[Eerste waarden en grenzen]\label{prop:g8:cosine:bounds}
Voor elke scherpe hoek $\theta$: $0 < \cos\theta < 1$, en de cosinus
\emph{daalt} naarmate de hoek opent: een wijdere hoek heeft een
kleinere cosinus. Ankerpunten: $\cos 0^\circ = 1$,
$\cos 60^\circ = \frac12$, $\cos 90^\circ = 0$ (de extreme waarden
die bij platte driehoeken horen).
\end{proposition}
\admitted

\begin{method}[De cosinus gebruiken]\label{met:g8:cosine:use}
In een rechthoekige driehoek, wanneer de bekende en gevraagde
grootheden een scherpe hoek, haar aanliggende zijde en de schuine
zijde zijn:
\begin{enumerate}
  \item schrijf de definitie:
        $\cos\theta = \frac{\text{aanliggend}}{\text{schuine zijde}}$
        met de bekende waarden ingevuld;
  \item los de kleine vergelijking op voor de onbekende
        (vermenigvuldig of deel);
  \item voor een onbekende \emph{hoek}, pas de $\cos^{-1}$ van de
        rekenmachine toe op de berekende verhouding;
  \item saneiteitscontroles: een lengte moet korter uitkomen dan de
        schuine zijde; een hoek strikt tussen $0^\circ$ en
        $90^\circ$.
\end{enumerate}
\end{method}

\begin{example}[Een zijde vinden]\label{ex:g8:cosine:side}
Een helling van $6$ m maakt een hoek van $20^\circ$ met de
horizontale grond. Overbrugde horizontale afstand (aanliggend aan
$20^\circ$, schuine zijde $6$ m):
\[
\cos 20^\circ = \frac{d}{6}
\quad\Longrightarrow\quad
d = 6 \cos 20^\circ \approx 6 \times 0.940 \approx 5.6 \text{ m}.
\]
\end{example}

\begin{example}[Een hoek vinden]\label{ex:g8:cosine:angle}
In een rechthoekige driehoek meet de aanliggende zijde bij de hoek
$\theta$ $3.5$ en de schuine zijde $5$:
\[
\cos\theta = \frac{3.5}{5} = 0.7,
\qquad
\theta = \cos^{-1}(0.7) \approx 45.6^\circ .
\]
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g8:cosine:1}
Een rechthoekige driehoek heeft een schuine zijde van $13$ cm. Wat is
de lengte van de mediaan vanuit het hoekpunt van de rechte hoek? Wat
is de straal van de cirkel door de drie hoekpunten?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:2}
Teken een cirkel met middellijn $8$ cm met een middellijn $[BC]$,
kies een willekeurig punt $A$ op de cirkel en teken $ABC$. Welke
hoek is recht? Citeer de stelling. Meet $AB$ en $AC$ en controleer
Pythagoras.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:3}
In een driehoek $DEF$ rechthoekig in $D$, noem de zijde aanliggend
aan de hoek $\widehat E$, de zijde ertegenover, en de schuine
zijde. Schrijf $\cos \widehat E$ als een verhouding.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:4}
Bereken de ontbrekende grootheid ($\cos 35^\circ \approx 0.819$,
$\cos 50^\circ \approx 0.643$):
\begin{enumerate}
  \item schuine zijde $10$, hoek $35^\circ$: aanliggende zijde?
  \item aanliggende zijde $6$, hoek $50^\circ$: schuine zijde?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:5}
In een rechthoekige driehoek meet de aanliggende zijde bij $\theta$
$8$ en de schuine zijde $10$. Bereken $\cos\theta$, daarna $\theta$
($\cos^{-1}(0.8) \approx 36.9^\circ$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:6}
Leg uit waarom $\cos\theta$ nooit gelijk kan zijn aan $1.2$ voor een
scherpe hoek van een rechthoekige driehoek.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:7}
Een zipline van $25$ m daalt van een platform naar de grond, met een
hoek van $12^\circ$ met de horizontaal ($\cos 12^\circ \approx
0.978$). Welke horizontale afstand overspant hij?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:8}
Orden zonder rekenmachine: $\cos 20^\circ$, $\cos 70^\circ$,
$\cos 45^\circ$ (gebruik \cref{prop:g8:cosine:bounds}). Controleer
daarna met een rekenmachine.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:9}
Een ladder van lengte $L$ leunt tegen een muur met een hoek $\theta$
tussen ladder en \emph{grond}. Zijn voet staat $1.2$ m van de muur
en $\theta = 68^\circ$ ($\cos 68^\circ \approx 0.375$). Bereken $L$,
daarna de bereikte hoogte (Pythagoras).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:10}
$[BC]$ is een middellijn van een cirkel met middelpunt $O$ en straal
$4.5$ cm, en $A$ is een punt van de cirkel met $AB = 5.4$ cm.
\begin{enumerate}
  \item Waarom is $ABC$ rechthoekig in $A$?
  \item Bereken $AC$, daarna $\cos \widehat B$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:cosine:11}
Gebruik de helft van een gelijkzijdige driehoek met zijde $1$
(gesneden langs een hoogte) om het ankerpunt
$\cos 60^\circ = \frac12$ van \cref{prop:g8:cosine:bounds} te
motiveren. (Waar valt de voet van de hoogte op de basis? De volledige
goniometrische versie is \cref{ex:g9:trig:special}.)
\end{exercise}
