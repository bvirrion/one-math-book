\chapter{Evenredigheid, snelheid en gemiddelden}\label{ch:g8:speed}

\omterm{def:g7:prop:table}{Evenredigheid} (\cref{ch:g7:prop}) ontmoet nu de fysieke wereld:
\omterm{def:g8:speed:def}{snelheid}, debieten, eenheidsomzettingen --- grootheden die
\emph{\omterm{def:g3:division:remainder}{quotiënten}} van twee andere zijn. Het hoofdstuk sluit af met
\omterm{def:g8:speed:average}{gewogen gemiddelden}, waar \omterm{def:g6:propdata:def}{evenredig} denken een klassieke fout
voorkomt.

\section{Gemiddelde snelheid}

\begin{definition}[Gemiddelde snelheid]\label{def:g8:speed:def}
De \emph{gemiddelde snelheid}\index{snelheid} van een reis is het
\omterm{def:g3:division:remainder}{quotiënt}
\[
v = \frac{d}{t}
\qquad
\left(\text{afstand gedeeld door duur}\right),
\]
in kilometers per uur (km/h), meters per seconde (m/s), \dots\ De
formule keert om: $d = v \times t$ en $t = \frac{d}{v}$.
\end{definition}

\begin{example}\label{ex:g8:speed:basic}
Een trein legt $270$ km af in $1$ u $30$ min. Zet eerst de tijd om:
$1$ u $30$ min $= 1.5$ u. Dan
\[
v = \frac{270}{1.5} = 180 \text{ km/h}.
\]
Afstand bij $90$ km/h gedurende $2$ u $20$ min $= \frac73$ u:
$d = 90 \times \frac73 = 210$ km. Tijd voor $35$ km bij $14$ km/h:
$t = \frac{35}{14} = 2.5$ u $= 2$ u $30$ min.
\end{example}

\begin{remark}[Minuten zijn geen decimalen]
$1$ u $30$ min is $1.5$ u, maar $1$ u $20$ min is \emph{niet} $1.2$
u: het is $1 + \frac{20}{60} = \frac43 \approx 1.33$ u. Zet altijd
minuten om naar een \omterm{def:g6:fractions:def}{breuk} van een uur ($60$ min $= 1$ u) vóór je
deelt.
\end{remark}

\begin{method}[Snelheden omrekenen]\label{met:g8:speed:convert}
Om km/h naar m/s om te rekenen: $1$ km $= 1000$ m en $1$ u $= 3600$
s, dus
\[
v \text{ km/h} = \frac{v \times 1000}{3600} \text{ m/s}
= \frac{v}{3.6} \text{ m/s}.
\]
Om m/s naar km/h om te rekenen, vermenigvuldig met $3.6$.
\end{method}

\begin{example}\label{ex:g8:speed:convert}
$90$ km/h $= \frac{90}{3.6} = 25$ m/s. Een sprinter die $100$ m in
$10$ s loopt beweegt met $10$ m/s $= 36$ km/h.
\end{example}

\section{Quotiëntgrootheden}

\begin{example}[Debiet, dichtheid, prijs per kilo]\label{ex:g8:speed:quotients}
\omterm{def:g8:speed:def}{Snelheid} heeft veel neven, die allemaal op dezelfde manier behandeld
worden:
\begin{itemize}
  \item een kraan vult $48$ L in $4$ min: debiet
        $\frac{48}{4} = 12$ L/min, dus een bad van $150$ L vullen
        duurt $\frac{150}{12} = 12.5$ min;
  \item $0.6$ kg kaas kost $9$ euro: prijs
        $\frac{9}{0.6} = 15$ euro per kg;
  \item een auto verbruikt $6.3$ L voor $90$ km: verbruik
        $\frac{6.3}{90} \times 100 = 7$ L per $100$ km.
\end{itemize}
Identificeer het \omterm{def:g3:division:remainder}{quotiënt}, en de drie-weg-formule
($q = \frac ab$, $a = qb$, $b = \frac aq$) doet de \omterm{def:g3:division:remainder}{rest}.
\end{example}

\section{Gewogen gemiddelden}

\begin{definition}[Gewogen gemiddelde]\label{def:g8:speed:average}
Wanneer waarden $v_1, v_2, \dots$ met \omterm{def:g7:stats:frequency}{tellingen} (of gewichten)
$n_1, n_2, \dots$ komen, is hun \emph{gewogen
gemiddelde}\index{gewogen gemiddelde}
\[
\bar v = \frac{n_1 v_1 + n_2 v_2 + \dots}{n_1 + n_2 + \dots} :
\]
totaal van de waarden, gedeeld door totaal van de gewichten.
\end{definition}

\begin{example}\label{ex:g8:speed:average}
Een toets werd afgelegd door twee groepen: groep 1, $12$ leerlingen,
gemiddelde $11$; groep 2, $18$ leerlingen, gemiddelde $14$.
Gemiddelde van de hele klas:
\[
\bar v = \frac{12 \times 11 + 18 \times 14}{12 + 18}
= \frac{132 + 252}{30} = \frac{384}{30} = 12.8 .
\]
\emph{Niet} $\frac{11 + 14}{2} = 12.5$: de grotere groep trekt het
gemiddelde naar zich toe — de gewichten tellen.
\end{example}

\begin{example}[Gemiddelde snelheid over twee etappes]\label{ex:g8:speed:twolegs}
Een fietser rijdt $30$ km bij $30$ km/h, daarna $30$ km bij $15$ km/h.
De \omterm{def:g8:speed:def}{gemiddelde snelheid} over de hele rit is \emph{niet}
$\frac{30 + 15}{2} = 22.5$ km/h. Bereken met de definitie:
\begin{enumerate}
  \item tijden: $\frac{30}{30} = 1$ u, daarna $\frac{30}{15} = 2$ u;
        totaal $3$ u;
  \item totale afstand $60$ km, dus
        $v = \frac{60}{3} = 20$ km/h.
\end{enumerate}
De trage etappe duurt langer, dus weegt ze zwaarder --- een
gemiddelde van snelheden moet gewogen worden met \emph{tijd}.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g8:speed:1}
Bereken de \omterm{def:g8:speed:def}{gemiddelde snelheid}: $150$ km in $2$ u; $27$ km in
$45$ min; $100$ m in $12.5$ s.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:2}
Bereken de afstand: $2$ u bij $85$ km/h; $40$ min bij $90$ km/h.
Bereken de tijd: $315$ km bij $90$ km/h (in u en min).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:3}
Reken om: $72$ km/h naar m/s; $5$ m/s naar km/h; $108$ km/h naar m/s.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:4}
Een kraan levert $15$ L/min. Hoe lang om een tank van $600$ L te
vullen? Hoeveel liters in $2$ u $30$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:5}
Welke is de betere koop: $1.2$ kg appels voor $3.30$ euro, of
$0.8$ kg voor $2.16$ euro? Vergelijk prijzen per kilogram.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:6}
Een klas van $25$ leerlingen heeft gemiddelde $12.4$ op een toets;
een andere klas van $35$ leerlingen heeft gemiddelde $10$. Bereken
het gemiddelde van de twee klassen samen.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:7}
Geluid reist met ongeveer $340$ m/s. Je ziet een bliksemflits en
hoort de donder $6$ seconden later. Hoe ver vandaan sloeg de bliksem
in (op de dichtstbijzijnde $100$ m)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:speed:8}
Een wandelaar loopt $2$ u bij $5$ km/h, rust $30$ min, en loopt
daarna $1$ u $30$ bij $4$ km/h. Bereken de totale afstand en de
\omterm{def:g8:speed:def}{gemiddelde snelheid} \emph{inclusief de rust} (totale afstand over
totale verstreken tijd).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:speed:9}
Cijfers met coëfficiënten: een leerling haalde $15$ (coëfficiënt
$3$), $9$ (coëfficiënt $2$) en $12$ (coëfficiënt $1$). Bereken het
\omterm{def:g8:speed:average}{gewogen gemiddelde}. Welk gewoon (ongewogen) gemiddelde zou ze hebben,
en waarom verschillen ze?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:speed:10}
Een auto rijdt $120$ km bij $80$ km/h en daarna $120$ km bij
$120$ km/h.
\begin{enumerate}
  \item Bereken de duur van elke etappe, daarna de \omterm{def:g8:speed:def}{gemiddelde
        snelheid} over de hele rit.
  \item Leg uit waarom het antwoord minder is dan $100$ km/h, het
        midden van de twee snelheden.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:speed:11}
Twee dorpen liggen $18$ km uit elkaar. Anna vertrekt om $10{:}00$
vanuit het eerste en loopt met $5$ km/h naar het tweede; Boris
vertrekt tegelijk vanuit het tweede met $4$ km/h naar het eerste. Op
welk tijdstip ontmoeten ze elkaar, en op welke afstand van Anna's
dorp? (Samen dichten ze de kloof met $5 + 4$ km/h.)
\end{exercise}
