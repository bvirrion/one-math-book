\chapter{Breuken: alle vier de bewerkingen}\label{ch:g8:fractions}

Dit hoofdstuk voltooit de rekenkunde van breuken: optellen en
aftrekken met \emph{elke} noemers, vermenigvuldigen, en --- de
nieuwkomer --- delen, wat een vermenigvuldiging in vermomming blijkt te
zijn. Tekens uit \cref{ch:g8:negprod} reizen mee.

\section{Optellen met willekeurige noemers}

\begin{method}[Gemeenschappelijke noemer]\label{met:g8:fractions:add}
Om $\frac ab$ en $\frac cd$ op te tellen of af te trekken:
\begin{enumerate}
  \item vind een gemeenschappelijk veelvoud van $b$ en $d$ (het
        product $b \times d$ werkt altijd; een kleiner spaart
        werk);
  \item herschrijf beide breuken met die noemer;
  \item tel of trek de tellers op/af; vereenvoudig.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:fractions:add}
Bereken $\dfrac{5}{6} + \dfrac{3}{4}$. Een gemeenschappelijk veelvoud
van $6$ en $4$ is $12$:
\[
\frac{5}{6} = \frac{10}{12},
\qquad
\frac{3}{4} = \frac{9}{12},
\qquad
\frac{5}{6} + \frac{3}{4} = \frac{10 + 9}{12} = \frac{19}{12}.
\]
Met tekens: $\dfrac{2}{3} - \dfrac{7}{5} = \dfrac{10}{15} -
\dfrac{21}{15} = -\dfrac{11}{15}$.
\end{example}

\section{Breuken vermenigvuldigen}

\begin{theorem}[Product van breuken]\label{thm:g8:fractions:mult}
\[
\frac{a}{b} \times \frac{c}{d} = \frac{a \times c}{b \times d} .
\]
\end{theorem}

\begin{proof}[Waarom, op een plaatje]
Neem $\frac34$ van $\frac25$ van een vierkant: snijd het vierkant in
$5$ verticale stroken en houd $2$; snijd horizontaal in $4$ en houd
$3$ rijen van wat overbleef. Het gehouden deel is een rooster van
$3 \times 2$ kleine cellen van de $4 \times 5$: breuk $\frac{6}{20}$.
\end{proof}

  \draw[gray!50, thin] (0,0) grid[xstep=1, ystep=1] (5,4);
  \fill[omDef!18] (0,0) rectangle (2,4);
  \fill[omThm!35] (0,0) rectangle (2,3);
  \draw[thick] (0,0) rectangle (5,4);
  \draw[omDef, thick] (2,0) -- (2,4);
  \draw[omThm, thick] (0,3) -- (5,3);
  \node[below, font=\small, omDef] at (1,0) {$\frac25$ gehouden};
  \node[right, font=\small, omThm] at (5.05,1.5) {$\frac34$ ervan};
\end{tikzpicture}

{\small $\frac34 \times \frac25$: het dubbel gearceerde gebied is
$3 \times 2 = 6$ cellen van de $4 \times 5 = 20$, d.w.z.\
$\frac{6}{20} = \frac{3}{10}$ --- tellers vermenigvuldigd, noemers
vermenigvuldigd.}
\end{omfigure}

\begin{example}\label{ex:g8:fractions:mult}
Vereenvoudig \emph{vóór} het vermenigvuldigen, door gemeenschappelijke
factoren weg te strepen:
\[
\frac{7}{12} \times \frac{9}{14}
= \frac{7 \times 9}{12 \times 14}
= \frac{7 \times 3 \times 3}{3 \times 4 \times 2 \times 7}
= \frac{3}{8}
\]
(de factor $7$ en één factor $3$ vallen weg tussen boven en onder).
Met tekens gelden eerst de regels van \cref{thm:g8:negprod:rules}:
$\left(-\frac23\right) \times \left(-\frac{5}{4}\right) =
+\frac{10}{12} = \frac56$.
\end{example}

\section{Breuken delen}

\begin{definition}[Omgekeerde]\label{def:g8:fractions:inverse}
Het \emph{omgekeerde}\index{omgekeerde} van een getal $x$ ongelijk aan
nul is het getal dat, vermenigvuldigd met $x$, $1$ geeft. Het
omgekeerde van $\frac ab$ is $\frac ba$, want
$\frac ab \times \frac ba = \frac{ab}{ab} = 1$; het omgekeerde van
een geheel getal $n$ is $\frac1n$.
\end{definition}

\begin{theorem}[Delen is vermenigvuldigen met het omgekeerde]\label{thm:g8:fractions:div}
Voor $c \neq 0$:
\[
\frac{a}{b} \div \frac{c}{d} = \frac{a}{b} \times \frac{d}{c} .
\]
\end{theorem}

\begin{proof}
Per definitie is het quotiënt $q$ van $\frac ab$ door $\frac cd$ het
getal met $q \times \frac cd = \frac ab$. Controleer de kandidaat
$q = \frac ab \times \frac dc$:
\[
\left(\frac ab \times \frac dc\right) \times \frac cd
= \frac ab \times \left(\frac dc \times \frac cd\right)
= \frac ab \times 1 = \frac ab . \qedhere
\]
\end{proof}

\begin{example}\label{ex:g8:fractions:div}
\[
\frac{3}{5} \div \frac{7}{10}
= \frac{3}{5} \times \frac{10}{7}
= \frac{3 \times 10}{5 \times 7}
= \frac{30}{35} = \frac{6}{7}.
\]
Een saneiteitscontrole: delen door $\frac{7}{10}$, een getal kleiner
dan $1$, moet een resultaat \emph{groter} dan $\frac35$ geven --- en
$\frac67 > \frac35$ inderdaad ($\frac{30}{35} > \frac{21}{35}$).
\end{example}

\begin{example}[Breukstrepen in breukstrepen]\label{ex:g8:fractions:complex}
Een ``dubbeldekker'' is gewoon een deling verticaal geschreven:
\[
\frac{\ \frac{2}{3}\ }{\ \frac{5}{6}\ }
= \frac{2}{3} \div \frac{5}{6}
= \frac{2}{3} \times \frac{6}{5}
= \frac{12}{15} = \frac{4}{5}.
\]
Zoek eerst de \emph{hoofd}streep (de langste), en pas daarna
\cref{thm:g8:fractions:div} toe.
\end{example}

\begin{method}[Gemengde berekeningen]\label{met:g8:fractions:mixed}
In een uitdrukking die breuken en de vier bewerkingen mengt:
\begin{enumerate}
  \item pas de gewone rekenvolgorde toe (\cref{ch:g7:priorities});
  \item zet elke deling om in een vermenigvuldiging met het
        omgekeerde;
  \item bepaal eerst de tekens (\cref{met:g8:negprod:compute}), en
        vermenigvuldig daarna of zoek gemeenschappelijke noemers;
  \item vereenvoudig bij de vroegste gelegenheid.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:fractions:mixed}
\begin{align*}
\frac12 + \frac{2}{3} \times \frac{9}{4}
&= \frac12 + \frac{2 \times 9}{3 \times 4}
&& \text{(eerst product!)} \\
&= \frac12 + \frac{18}{12} = \frac12 + \frac32
= \frac{4}{2} = 2 .
\end{align*}
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g8:fractions:1}
Bereken en vereenvoudig:
\[
\frac{3}{4} + \frac{2}{5}, \qquad
\frac{5}{6} - \frac{3}{8}, \qquad
\frac{7}{10} + \frac{8}{15} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:fractions:2}
Bereken (eerst de tekens):
\[
-\frac{2}{7} + \frac{3}{14}, \qquad
\frac{1}{6} - \frac{5}{4}, \qquad
-\frac{3}{5} - \frac{1}{2} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:fractions:3}
Bereken, en vereenvoudig vóór het vermenigvuldigen:
\[
\frac{5}{8} \times \frac{4}{15}, \qquad
\frac{9}{14} \times \frac{7}{3}, \qquad
\left(-\frac{6}{5}\right) \times \frac{10}{9} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:fractions:4}
Geef het omgekeerde van: $\dfrac{3}{7}$; \quad $5$; \quad
$\dfrac{1}{4}$; \quad $-\dfrac{2}{9}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:fractions:5}
Bereken:
\[
\frac{4}{9} \div \frac{2}{3}, \qquad
\frac{7}{5} \div 14, \qquad
6 \div \frac{3}{4} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:fractions:6}
Bereken de dubbeldekkers:
\[
\frac{\ \frac{3}{4}\ }{\ \frac{9}{8}\ },
\qquad
\frac{\ \frac{5}{6}\ }{\ 10\ },
\qquad
\frac{\ 2\ }{\ \frac{4}{7}\ } .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:fractions:7}
Bereken, met respect voor de rekenvolgorde:
\[
\frac{1}{3} + \frac{1}{2} \times \frac{4}{5},
\qquad
\left(\frac{1}{3} + \frac{1}{2}\right) \times \frac{4}{5} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:fractions:8}
Een tank is $\frac{2}{5}$ vol. Men voegt $\frac{1}{3}$ \emph{van de
capaciteit van de tank} toe. Welk deel van de tank is nu vol? Welk
deel is nog leeg?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:fractions:9}
Driekwart van de leerlingen van een klas slaagde voor een toets; onder
hen haalde twee derden meer dan $14$ op $20$. Welk deel van de klas
haalde meer dan $14$? Als dat $12$ leerlingen voorstelt, hoe groot is
de klas?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:fractions:10}
Een touw van $\frac{15}{2}$ m moet in stukken van $\frac{3}{4}$ m
worden geknipt. Hoeveel stukken krijgt men? (Een deling van breuken
--- controleer dat het antwoord een geheel getal is.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:fractions:11}
Bereken
\[
E = \frac{2}{3} - \frac{5}{6} \div \frac{10}{9},
\qquad
F = \left(-\frac{1}{2}\right)^2 \times \frac{8}{3} .
\]
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:fractions:12}
Vereenvoudig de uitdrukking
\[
1 + \cfrac{1}{1 + \cfrac{1}{1 + 1}}
\]
(begin bij de binnenste breuk en werk naar buiten, één streep per
keer).
\end{exercise}
