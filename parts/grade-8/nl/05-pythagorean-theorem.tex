\chapter{De stelling van Pythagoras}\label{ch:g8:pythagoras}

De beroemdste stelling van de hele wiskunde verbindt de drie zijden
van een rechthoekige driehoek. Daarmee worden lengten berekenbaar die
geen liniaal rechtstreeks kan meten: diagonalen, afstanden, hoogten.
Het is ook onze eerste grote ontmoeting met het verschil tussen een
stelling en haar omgekeerde.

\section{De stelling}

\begin{theorem}[Pythagoras]\label{thm:g8:pythagoras:direct}
Als een driehoek $ABC$ rechthoekig is in $A$, dan
\[
BC^2 = AB^2 + AC^2 :
\]
het kwadraat van de schuine zijde (de zijde tegenover de rechte
hoek) is gelijk aan de som van de kwadraten van de twee andere
zijden.
\end{theorem}

\begin{proof}[Bewijs-idee]
Vier kopieën van de driehoek, gerangschikt in een groot vierkant met
zijde $AB + AC$, laten een scheef vierkant op de schuine zijde
onbedekt: oppervlakten vergelijken geeft de gelijkheid. De
gedetailleerde berekening wordt voorgesteld als
\cref{exo:g8:pythagoras:11}; de stelling volgt ook uit het
scalair product, bestudeerd in het High School-volume.
\end{proof}

\begin{method}[Een lengte berekenen]\label{met:g8:pythagoras:compute}
In een rechthoekige driehoek waarvan twee zijden bekend zijn:
\begin{enumerate}
  \item noem de schuine zijde (tegenover de rechte hoek --- altijd de
        langste zijde);
  \item schrijf de gelijkheid van \cref{thm:g8:pythagoras:direct} met
        de bekende waarden;
  \item los op voor het ontbrekende kwadraat: \emph{tel} de kwadraten
        op als de schuine zijde onbekend is, \emph{trek af} als een
        been onbekend is;
  \item neem de vierkantswortel, exact of met een rekenmachine, en
        controleer dat de schuine zijde het langst uitkomt.
\end{enumerate}
\end{method}

\begin{example}[De schuine zijde vinden]\label{ex:g8:pythagoras:hypotenuse}
Een rechthoekige driehoek heeft benen $6$ en $8$. Schuine zijde $c$:
\[
c^2 = 6^2 + 8^2 = 36 + 64 = 100,
\qquad
c = \sqrt{100} = 10 .
\]
(Het getal $\sqrt{100}$, de \emph{vierkantswortel} van $100$, is het
positieve getal waarvan het kwadraat $100$ is; wortels krijgen een
vol hoofdstuk in \cref{ch:g9:sqrt}.)
\end{example}

\begin{example}[Een been vinden]\label{ex:g8:pythagoras:leg}
Een ladder van $5$ m leunt tegen een muur, met de voet $1.4$ m van de
muur. Bereikte hoogte $h$: de ladder is de schuine zijde, dus
\[
h^2 = 5^2 - 1.4^2 = 25 - 1.96 = 23.04,
\qquad
h = \sqrt{23.04} = 4.8 \text{ m}.
\]
Aftrekken, niet optellen: de onbekende is hier een \emph{been}.
\end{example}

\section{Het omgekeerde}

\begin{theorem}[Omgekeerde van Pythagoras]\label{thm:g8:pythagoras:converse}
Als in een driehoek $ABC$ de zijden voldoen aan $BC^2 = AB^2 + AC^2$,
dan is de driehoek rechthoekig in $A$.
\end{theorem}
\admitted

\begin{method}[Een rechte hoek testen]\label{met:g8:pythagoras:test}
Gegeven de drie zijden van een driehoek:
\begin{enumerate}
  \item bereken apart het kwadraat van de langste zijde, en de som
        van de kwadraten van de twee andere;
  \item als de twee resultaten gelijk zijn, is de driehoek
        rechthoekig (omgekeerde), in het hoekpunt tegenover de
        langste zijde;
  \item als ze verschillen, is hij niet rechthoekig --- want de
        stelling van Pythagoras zou gelijkheid forceren.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:pythagoras:test}
Zijden $5$, $12$, $13$: langste zijde gekwadrateerd, $13^2 = 169$;
som van de andere kwadraten, $5^2 + 12^2 = 25 + 144 = 169$. Gelijk:
rechthoekig (tegenover de zijde $13$).

Zijden $6$, $7$, $9$: $9^2 = 81$ maar $6^2 + 7^2 = 85 \neq 81$: geen
rechthoekige driehoek (hij is ``bijna'' één --- de getallen
beslissen, niet het oog).
\end{example}

\begin{remark}[Stelling vs omgekeerde]
De stelling en haar omgekeerde zeggen verschillende dingen: de
stelling \emph{gebruikt} een rechte hoek om lengten te berekenen; het
omgekeerde gebruikt lengten om een rechte hoek te \emph{bewijzen}.
Bouwers gebruiken het omgekeerde al millennia: een touw met knopen op
$3$, $4$, $5$ eenheden strak getrokken geeft een perfecte rechte
hoek.
\end{remark}

\begin{example}[Diagonaal van een vierkant]\label{ex:g8:pythagoras:diagonal}
De diagonaal van een vierkant met zijde $1$ is de schuine zijde van
een rechthoekige driehoek met benen $1$ en $1$:
\[
d^2 = 1^2 + 1^2 = 2 ,
\]
dus $d = \sqrt2 \approx 1.414$ --- een getal dat geen breuk is, zoals
het High School-volume zal bewijzen. Voor een vierkant met zijde $c$
is de diagonaal $c\sqrt2$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:1}
Bereken in elke rechthoekige driehoek de schuine zijde:
benen $9$ en $12$; \quad benen $5$ en $12$; \quad benen $8$ en $15$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:2}
Een rechthoekige driehoek heeft schuine zijde $25$ en één been $7$.
Bereken het andere been.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:3}
Bereken de diagonaal van een rechthoek met zijden $12$ cm en $9$ cm;
de diagonaal van een vierkant met zijde $5$ cm (exacte waarde met een
wortel, daarna afgerond op de mm).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:4}
Welke driehoeken zijn rechthoekig? Motiveer met
\cref{met:g8:pythagoras:test}:
\[
(20,\ 21,\ 29); \qquad
(7,\ 8,\ 11); \qquad
(1.5,\ 2,\ 2.5).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:5}
Een poort is verstevigd met een diagonale plank. De poort is $3$ m
breed en $1.6$ m hoog: hoe lang is de plank?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:6}
Een gelijkbenige driehoek heeft twee zijden van $10$ cm en een basis
van $12$ cm. Zijn hoogte splitst hem in twee rechthoekige driehoeken:
bereken die hoogte, daarna de oppervlakte van de driehoek.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:pythagoras:7}
Een televisiescherm is een $16{:}9$-rechthoek van $88.5$ cm bij
$49.8$ cm. Schermen worden verkocht op de diagonaal, in inches
($1$ inch $= 2.54$ cm). Bereken de diagonaal in cm, daarna in inches
(rond af op de dichtstbijzijnde inch).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:pythagoras:8}
Een boot vaart $24$ km oost, daarna $10$ km noord. Hoe ver is hij van
zijn startpunt (hemelsbreed)? Teken eerst de situatie.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:pythagoras:9}
Een ladder van $2.5$ m moet een vensterbank op $2.4$ m boven de
grond bereiken. Hoe ver van de muur moet de voet geplaatst worden? Is
het antwoord verenigbaar met veiligheidsadvies (voet op ongeveer een
kwart van de ladderlengte van de muur)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:pythagoras:10}
Teken op een coördinatenrooster $A(1, 2)$ en $B(5, 5)$, en teken de
rechthoekige driehoek met benen evenwijdig aan de assen waarvan de
schuine zijde $[AB]$ is. Bereken $AB$. (Deze constructie, voor
willekeurige twee punten, wordt de afstandsformule van de
coördinatenmeetkunde, in het High School-volume.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:pythagoras:11}
Vier kopieën van een rechthoekige driehoek met benen $a$, $b$ en
schuine zijde $c$ worden in de hoeken van een vierkant met zijde
$a + b$ geplaatst, waarbij een scheef vierkant met zijde $c$ in het
midden onbedekt blijft.
\begin{enumerate}
  \item Druk de oppervlakte van het grote vierkant op twee manieren
        uit: rechtstreeks, en als (vier driehoeken) $+$ (scheef
        vierkant).
  \item Werk $(a + b)^2$ uit (zie \cref{thm:g8:equations:expand}) en
        leid $c^2 = a^2 + b^2$ af: je hebt de stelling van Pythagoras
        bewezen.
\end{enumerate}
\end{exercise}
