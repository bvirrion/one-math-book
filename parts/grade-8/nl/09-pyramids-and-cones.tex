\chapter{Piramide en kegel}\label{ch:g8:solids}

Na de prisma's en cilinders van \cref{ch:g7:areas} --- lichamen met
constante doorsnede --- komen de lichamen met een punt: piramides en
kegels. Hun volumeformule draagt een beroemde factor $\frac13$, die
dit hoofdstuk aannemelijk maakt en gebruikt; de studie van lichamen
gaat verder met bollen en doorsneden in \cref{ch:g9:solids}.

\section{Piramides en kegels beschrijven}

\begin{definition}[Piramide, kegel]\label{def:g8:solids:def}
\begin{itemize}
  \item Een \emph{piramide}\index{piramide} heeft een veelhoek als
        \emph{grondvlak} en een punt, de \emph{top}, verbonden met
        elk hoekpunt van het grondvlak door driehoekige zijvlakken.
        Haar \emph{hoogte} is de afstand van de top tot het vlak van
        het grondvlak.
  \item Een \emph{kegel}\index{kegel} is dezelfde constructie over
        een schijf: een grondschijf, een top, en een gekromd
        mantelvak.
\end{itemize}
Een piramide is \emph{regelmatig} wanneer haar grondvlak een
regelmatige veelhoek is (vierkant, gelijkzijdige driehoek, \dots) en
haar top loodrecht boven het middelpunt van het grondvlak staat.
\end{definition}

\begin{example}[Vlakken, ribben, hoekpunten tellen]\label{ex:g8:solids:count}
Een piramide met vierkant grondvlak heeft $5$ vlakken ($1$ vierkant
$+ 4$ driehoeken), $8$ ribben en $5$ hoekpunten. Met een $n$-hoekig
grondvlak: $n + 1$ vlakken, $2n$ ribben, $n + 1$ hoekpunten.
Controleer Eulers kleine patroon: vlakken $+$ hoekpunten $=$ ribben
$+ 2$.
\end{example}

\section{Volume}

\begin{theorem}[Volume van een piramide of kegel]\label{thm:g8:solids:volume}
Voor een piramide of een kegel met grondvlakoppervlakte $B$ en
hoogte $h$:
\[
V = \frac{1}{3}\, B \times h
\]
--- één derde van het prisma of de cilinder met hetzelfde grondvlak
en dezelfde hoogte. Voor een kegel met grondstraal $r$:
$V = \frac13 \pi r^2 h$.
\end{theorem}
\admitted

\begin{remark}[Waarom één derde?]
Vul een holle piramide met water en giet die in het prisma met
hetzelfde grondvlak en dezelfde hoogte: het kost exact drie vullingen.
Nog beter: een kubus kan in drie identieke piramides gesneden worden,
elk met een vlak van de kubus als grondvlak en één hoekpunt van de
kubus als top (probeer het je voor te stellen!). Een algemeen bewijs
gebruikt integratie, een gereedschap van het High School-volume.
\end{remark}

\begin{example}\label{ex:g8:solids:volume}
Een piramide heeft een vierkant grondvlak met zijde $5$ cm en hoogte
$9$ cm:
\begin{enumerate}
  \item grondvlakoppervlakte: $B = 5^2 = 25$ cm$^2$;
  \item volume: $V = \frac13 \times 25 \times 9 = 25 \times 3 = 75$
        cm$^3$.
\end{enumerate}
Een ijsjeskegel met straal $3$ cm en hoogte $10$ cm:
\[
V = \frac13 \times \pi \times 3^2 \times 10 = 30\pi \approx 94
\text{ cm}^3 .
\]
\end{example}

\begin{example}[Let op de hoogte]\label{ex:g8:solids:slant}
Voor een kegel, verwar de hoogte $h$ (top tot middelpunt van het
grondvlak, loodrecht) niet met de \emph{mantellengte} $s$ (top tot
de rand). Ze hangen samen door Pythagoras
(\cref{thm:g8:pythagoras:direct}) in de rechthoekige driehoek
top--middelpunt--rand:
\[
s^2 = h^2 + r^2 .
\]
Een kegel met $r = 3$ en $s = 5$ heeft dus hoogte
$h = \sqrt{25 - 9} = 4$, en volume
$\frac13 \pi \times 9 \times 4 = 12\pi$.
\end{example}

\begin{method}[Volumeproblemen]\label{met:g8:solids:method}
\begin{enumerate}
  \item Identificeer het lichaam (prisma/cilinder: $V = Bh$;
        piramide/kegel: $V = \frac13 Bh$);
  \item bereken eerst de grondvlakoppervlakte $B$, als aparte stap;
  \item controleer dat de hoogte loodrecht op het grondvlak staat
        (gebruik Pythagoras als de mantellengte gegeven is);
  \item vermenigvuldig, houd eenheden consistent, reken op het
        einde om indien nodig ($1$ L $= 1000$ cm$^3$).
\end{enumerate}
\end{method}

\section{Uitslagen}

\begin{example}[Uitslag van een piramide]\label{ex:g8:solids:net}
Het openvouwen van een regelmatige vierkante piramide maakt haar
\emph{uitslag}: het vierkante grondvlak met vier identieke
gelijkbenige driehoeken vast aan de zijden. De gelijke zijden van de
driehoeken zijn de \emph{zijribben} van de piramide --- hun lengte is
noch $h$ noch de mantellengte van een vlak, dus label zorgvuldig
vóór je karton knipt.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g8:solids:1}
Hoeveel vlakken, ribben en hoekpunten heeft een piramide met
driehoekig grondvlak (een \emph{tetraëder})? Met zeshoekig
grondvlak?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:2}
Bereken het volume van een piramide met grondvlakoppervlakte
$36$ cm$^2$ en hoogte $10$ cm; van een piramide met rechthoekig
grondvlak $6 \times 4$ cm en hoogte $7.5$ cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:3}
Bereken het volume van een kegel met straal $5$ cm en hoogte $9$ cm
(exact met $\pi$, daarna afgerond op de cm$^3$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:4}
Een cilinder en een kegel hebben dezelfde straal $4$ cm en dezelfde
hoogte $12$ cm. Bereken beide volumes. Wat is hun verhouding?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:5}
Een piramide heeft volume $56$ cm$^3$ en hoogte $8$ cm. Wat is haar
grondvlakoppervlakte? (Schrijf de volumeformule en los op.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:6}
Schets de uitslag van een kegel (een schijf voor het grondvlak, en
een sector van een grotere schijf voor het mantelvak). Welke maat van
de kegel is de straal van die sector: de hoogte $h$ of de
mantellengte $s$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:7}
De Louvre-piramide heeft een vierkant grondvlak met zijde ongeveer
$35$ m en een hoogte van ongeveer $22$ m. Schat haar volume, op de
dichtstbijzijnde honderd kubieke meters.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:8}
Een kegel heeft mantellengte $13$ cm en grondstraal $5$ cm. Bereken
zijn hoogte, daarna zijn volume (exact met $\pi$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:9}
Een kegelvormig glas met straal $4$ cm en hoogte $9$ cm wordt tot de
rand gevuld, en daarna uitgegoten in een cilindrisch glas met straal
$4$ cm. Welke hoogte bereikt de vloeistof in de cilinder?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:10}
Een zandloper bestaat uit twee identieke kegels (straal $2$ cm,
hoogte $4.5$ cm) verbonden in hun toppen. Al het zand vult exact één
kegel. Bereken het volume zand (exact, daarna in cm$^3$ afgerond op
de tiende), en het deel van het totale binnenvolume van de zandloper
dat door het zand ingenomen wordt.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:solids:11}
Een vierkante piramide met grondzijde $6$ cm heeft haar top
rechtstreeks boven een hoek van het grondvlak, op hoogte $8$ cm.
\begin{enumerate}
  \item Geldt de formule $V = \frac13 Bh$ nog? (Ja --- de top hoeft
        niet gecentreerd te zijn. Bereken $V$.)
  \item Bereken de lengte van de langste zijribbe, van de top naar de
        overstaande hoek van het grondvlak (twee Pythagoras-stappen:
        eerst de gronddiagonaal).
\end{enumerate}
\end{exercise}
