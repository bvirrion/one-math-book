\chapter{Piramide en kegel}\label{ch:g8:solids}

Na de \omterm{def:g7:areas:prism}{prisma}'s en \omterm{def:g7:areas:prism}{cilinders} van \cref{ch:g7:areas} --- lichamen met
constante doorsnede --- komen de lichamen met een punt: \omterm{def:g8:solids:def}{piramides} en
\omterm{def:g8:solids:def}{kegels}. Hun volumeformule draagt een beroemde factor $\frac13$, die
dit hoofdstuk aannemelijk maakt en gebruikt; de studie van lichamen
gaat verder met bollen en doorsneden in \cref{ch:g9:solids}.

\section{Piramides en kegels beschrijven}

\begin{definition}[Piramide, kegel]\label{def:g8:solids:def}
\begin{itemize}
  \item Een \emph{piramide}\index{piramide} heeft een \omterm{def:g4:geometry:polygon}{veelhoek} als
        \emph{grondvlak} en een punt, de \emph{top}, verbonden met
        elk \omterm{def:g5:solids:def}{hoekpunt} van het grondvlak door driehoekige zijvlakken.
        Haar \emph{hoogte} is de afstand van de top tot het vlak van
        het grondvlak.
  \item Een \emph{kegel}\index{kegel} is dezelfde constructie over
        een schijf: een grondschijf, een top, en een gekromd
        mantelvak.
\end{itemize}
Een piramide is \emph{regelmatig} wanneer haar grondvlak een
regelmatige \omterm{def:g4:geometry:polygon}{veelhoek} is (vierkant, \omterm{def:g6:shapes:triangles}{gelijkzijdige driehoek}, \dots) en
haar top \omterm{def:g6:lines:perp}{loodrecht} boven het middelpunt van het grondvlak staat.
\end{definition}

\begin{example}[Vlakken, ribben, hoekpunten tellen]\label{ex:g8:solids:count}
Een \omterm{def:g8:solids:def}{piramide} met vierkant grondvlak heeft $5$ \omterm{def:g5:solids:def}{vlakken} ($1$ vierkant
$+ 4$ driehoeken), $8$ \omterm{def:g5:solids:def}{ribben} en $5$ \omterm{def:g5:solids:def}{hoekpunten}. Met een $n$-hoekig
grondvlak: $n + 1$ \omterm{def:g5:solids:def}{vlakken}, $2n$ \omterm{def:g5:solids:def}{ribben}, $n + 1$ \omterm{def:g5:solids:def}{hoekpunten}.
Controleer Eulers kleine patroon: \omterm{def:g5:solids:def}{vlakken} $+$ \omterm{def:g5:solids:def}{hoekpunten} $=$ \omterm{def:g5:solids:def}{ribben}
$+ 2$.
\end{example}

\section{Volume}

\begin{theorem}[Volume van een piramide of kegel]\label{thm:g8:solids:volume}
Voor een \omterm{def:g8:solids:def}{piramide} of een \omterm{def:g8:solids:def}{kegel} met grondvlakoppervlakte $B$ en
hoogte $h$:
\[
V = \frac{1}{3}\, B \times h
\]
--- één derde van het \omterm{def:g7:areas:prism}{prisma} of de \omterm{def:g7:areas:prism}{cilinder} met hetzelfde grondvlak
en dezelfde hoogte. Voor een \omterm{def:g8:solids:def}{kegel} met grondstraal $r$:
$V = \frac13 \pi r^2 h$.
\end{theorem}
\admitted

\begin{remark}[Waarom één derde?]
Vul een holle \omterm{def:g8:solids:def}{piramide} met water en giet die in het \omterm{def:g7:areas:prism}{prisma} met
hetzelfde grondvlak en dezelfde hoogte: het kost exact drie vullingen.
Nog beter: een \omterm{def:g5:solids:def}{kubus} kan in drie identieke \omterm{def:g8:solids:def}{piramides} gesneden worden,
elk met een vlak van de \omterm{def:g5:solids:def}{kubus} als grondvlak en één \omterm{def:g5:solids:def}{hoekpunt} van de
\omterm{def:g5:solids:def}{kubus} als top (probeer het je voor te stellen!). Een algemeen bewijs
gebruikt integratie, een gereedschap van het High School-volume.
\end{remark}

\begin{example}\label{ex:g8:solids:volume}
Een \omterm{def:g8:solids:def}{piramide} heeft een vierkant grondvlak met zijde $5$ cm en hoogte
$9$ cm:
\begin{enumerate}
  \item grondvlakoppervlakte: $B = 5^2 = 25$ cm$^2$;
  \item \omterm{def:g6:measure:volume}{volume}: $V = \frac13 \times 25 \times 9 = 25 \times 3 = 75$
        cm$^3$.
\end{enumerate}
Een ijsjeskegel met \omterm{def:g3:shapes:shapes}{straal} $3$ cm en hoogte $10$ cm:
\[
V = \frac13 \times \pi \times 3^2 \times 10 = 30\pi \approx 94
\text{ cm}^3 .
\]
\end{example}

\begin{example}[Let op de hoogte]\label{ex:g8:solids:slant}
Voor een \omterm{def:g8:solids:def}{kegel}, verwar de hoogte $h$ (top tot middelpunt van het
grondvlak, \omterm{def:g6:lines:perp}{loodrecht}) niet met de \emph{mantellengte} $s$ (top tot
de rand). Ze hangen samen door Pythagoras
(\cref{thm:g8:pythagoras:direct}) in de \omterm{def:g6:shapes:triangles}{rechthoekige driehoek}
top--middelpunt--rand:
\[
s^2 = h^2 + r^2 .
\]
Een \omterm{def:g8:solids:def}{kegel} met $r = 3$ en $s = 5$ heeft dus hoogte
$h = \sqrt{25 - 9} = 4$, en \omterm{def:g6:measure:volume}{volume}
$\frac13 \pi \times 9 \times 4 = 12\pi$.
\end{example}

\begin{method}[Volumeproblemen]\label{met:g8:solids:method}
\begin{enumerate}
  \item Identificeer het \omterm{def:g5:solids:def}{lichaam} (\omterm{def:g7:areas:prism}{prisma}/\omterm{def:g7:areas:prism}{cilinder}: $V = Bh$;
        \omterm{def:g8:solids:def}{piramide}/\omterm{def:g8:solids:def}{kegel}: $V = \frac13 Bh$);
  \item bereken eerst de grondvlakoppervlakte $B$, als aparte stap;
  \item controleer dat de hoogte \omterm{def:g6:lines:perp}{loodrecht} op het grondvlak staat
        (gebruik Pythagoras als de mantellengte gegeven is);
  \item vermenigvuldig, houd eenheden consistent, reken op het
        einde om indien nodig ($1$ L $= 1000$ cm$^3$).
\end{enumerate}
\end{method}

\section{Uitslagen}

\begin{example}[Uitslag van een piramide]\label{ex:g8:solids:net}
Het openvouwen van een regelmatige vierkante \omterm{def:g8:solids:def}{piramide} maakt haar
\emph{uitslag}: het vierkante grondvlak met vier identieke
\omterm{def:g6:shapes:triangles}{gelijkbenige driehoeken} vast aan de zijden. De gelijke zijden van de
driehoeken zijn de \emph{zijribben} van de \omterm{def:g8:solids:def}{piramide} --- hun lengte is
noch $h$ noch de mantellengte van een vlak, dus label zorgvuldig
vóór je karton knipt.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g8:solids:1}
Hoeveel \omterm{def:g5:solids:def}{vlakken}, \omterm{def:g5:solids:def}{ribben} en \omterm{def:g5:solids:def}{hoekpunten} heeft een \omterm{def:g8:solids:def}{piramide} met
driehoekig grondvlak (een \emph{tetraëder})? Met zeshoekig
grondvlak?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:2}
Bereken het \omterm{def:g6:measure:volume}{volume} van een \omterm{def:g8:solids:def}{piramide} met grondvlakoppervlakte
$36$ cm$^2$ en hoogte $10$ cm; van een \omterm{def:g8:solids:def}{piramide} met \omterm{def:g6:shapes:triangles}{rechthoekig}
grondvlak $6 \times 4$ cm en hoogte $7.5$ cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:3}
Bereken het \omterm{def:g6:measure:volume}{volume} van een \omterm{def:g8:solids:def}{kegel} met \omterm{def:g3:shapes:shapes}{straal} $5$ cm en hoogte $9$ cm
(exact met $\pi$, daarna afgerond op de cm$^3$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:4}
Een \omterm{def:g7:areas:prism}{cilinder} en een \omterm{def:g8:solids:def}{kegel} hebben dezelfde \omterm{def:g3:shapes:shapes}{straal} $4$ cm en dezelfde
hoogte $12$ cm. Bereken beide \omterm{def:g6:measure:volume}{volumes}. Wat is hun verhouding?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:5}
Een \omterm{def:g8:solids:def}{piramide} heeft \omterm{def:g6:measure:volume}{volume} $56$ cm$^3$ en hoogte $8$ cm. Wat is haar
grondvlakoppervlakte? (Schrijf de volumeformule en los op.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:6}
Schets de uitslag van een \omterm{def:g8:solids:def}{kegel} (een schijf voor het grondvlak, en
een sector van een grotere schijf voor het mantelvak). Welke maat van
de \omterm{def:g8:solids:def}{kegel} is de \omterm{def:g3:shapes:shapes}{straal} van die sector: de hoogte $h$ of de
mantellengte $s$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:7}
De \omterm{def:g8:solids:def}{Louvre-piramide} heeft een vierkant grondvlak met zijde ongeveer
$35$ m en een hoogte van ongeveer $22$ m. Schat haar \omterm{def:g6:measure:volume}{volume}, op de
dichtstbijzijnde honderd kubieke meters.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:8}
Een \omterm{def:g8:solids:def}{kegel} heeft mantellengte $13$ cm en grondstraal $5$ cm. Bereken
zijn hoogte, daarna zijn \omterm{def:g6:measure:volume}{volume} (exact met $\pi$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:9}
Een kegelvormig glas met \omterm{def:g3:shapes:shapes}{straal} $4$ cm en hoogte $9$ cm wordt tot de
rand gevuld, en daarna uitgegoten in een cilindrisch glas met \omterm{def:g3:shapes:shapes}{straal}
$4$ cm. Welke hoogte bereikt de vloeistof in de \omterm{def:g7:areas:prism}{cilinder}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:10}
Een zandloper bestaat uit twee identieke \omterm{def:g8:solids:def}{kegels} (\omterm{def:g3:shapes:shapes}{straal} $2$ cm,
hoogte $4.5$ cm) verbonden in hun toppen. Al het zand vult exact één
\omterm{def:g8:solids:def}{kegel}. Bereken het \omterm{def:g6:measure:volume}{volume} zand (exact, daarna in cm$^3$ afgerond op
de tiende), en het deel van het totale binnenvolume van de zandloper
dat door het zand ingenomen wordt.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:solids:11}
Een vierkante \omterm{def:g8:solids:def}{piramide} met grondzijde $6$ cm heeft haar top
rechtstreeks boven een hoek van het grondvlak, op hoogte $8$ cm.
\begin{enumerate}
  \item Geldt de formule $V = \frac13 Bh$ nog? (Ja --- de top hoeft
        niet gecentreerd te zijn. Bereken $V$.)
  \item Bereken de lengte van de langste zijribbe, van de top naar de
        overstaande hoek van het grondvlak (twee Pythagoras-stappen:
        eerst de gronddiagonaal).
\end{enumerate}
\end{exercise}
