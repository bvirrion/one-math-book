\chapter{Machten}\label{ch:g8:powers}

Vouw een vel papier $10$ keer dubbel (als je kunt!): de dikte
verdubbelt elke keer, en $10$ verdubbelingen vermenigvuldigen hem met
$2^{10} = 1024$. Machten zijn de verkorte notatie van herhaalde
vermenigvuldiging; dit hoofdstuk zet de notatie en haar regels op, met
een speciale plek voor de machten van $10$.

\section{Definitie}

\begin{definition}[Macht]\label{def:g8:powers:def}
Voor een getal $a$ en een geheel getal $n \geq 1$:
\[
a^n = \underbrace{a \times a \times \dots \times a}_{n
\text{ factoren}},
\]
gelezen ``$a$ tot de (macht) $n$''; $a$ is het \emph{grondtal}, $n$
de \emph{exponent}\index{exponent}. Speciale namen: $a^2$ is ``$a$
kwadraat'', $a^3$ ``$a$ derdemacht''. Per conventie:
\[
a^1 = a,
\qquad
a^0 = 1 \ (a \neq 0),
\qquad
a^{-n} = \frac{1}{a^n}.
\]
\end{definition}

\begin{example}\label{ex:g8:powers:def}
$3^4 = 3 \times 3 \times 3 \times 3 = 81$; \quad
$10^3 = 1000$; \quad $5^{-2} = \frac{1}{25} = 0.04$; \quad
$(-2)^3 = -8$ en $(-2)^4 = +16$
(\cref{ex:g8:negprod:powers}). Let op: $a^n$ is \emph{niet}
$a \times n$: \ $2^5 = 32$, niet $10$.
\end{example}

\section{De rekenregels voor exponenten}

\begin{theorem}[Regels voor exponenten]\label{thm:g8:powers:rules}
Voor een grondtal $a$ ongelijk aan nul en gehele exponenten $m$, $n$:
\[
a^m \times a^n = a^{m+n},
\qquad
\frac{a^m}{a^n} = a^{m-n},
\qquad
\left(a^m\right)^n = a^{m \times n},
\qquad
(ab)^n = a^n b^n .
\]
\end{theorem}

\begin{proof}[Bewijs door factoren te tellen]
$a^m \times a^n$ zet $m$ factoren $a$ gevolgd door $n$ meer op een
rij: $m + n$ in totaal. $\left(a^m\right)^n$ herhaalt een blok van
$m$ factoren $n$ keer: $mn$ factoren. $(ab)^n$ bevat $n$ letters $a$
en $n$ letters $b$, die hergroepeerd kunnen worden. De quotiëntrregel
komt van het wegstrepen van $n$ van de $m$ factoren; met de
conventies $a^0 = 1$ en $a^{-n} = \frac{1}{a^n}$ blijft hij waar ook
als $n \geq m$.
\end{proof}

\begin{example}\label{ex:g8:powers:rules}
\[
2^3 \times 2^5 = 2^8 = 256,
\qquad
\frac{7^6}{7^4} = 7^2 = 49,
\qquad
\left(5^2\right)^3 = 5^6,
\qquad
2^4 \times 5^4 = 10^4 .
\]
Een valkuil: de regels gelden voor een \emph{gemeenschappelijk
grondtal} (of een gemeenschappelijke exponent, voor de laatste). Geen
regel vereenvoudigt $2^3 \times 5^2$ --- reken gewoon uit:
$8 \times 25 = 200$.
\end{example}

\section{Machten van tien}

\begin{proposition}[Machten van tien]\label{prop:g8:powers:ten}
Voor $n \geq 1$: $10^n = 1\underbrace{0\dots0}_{n}$, en
$10^{-n} = 0.\underbrace{0\dots0}_{n-1}1$. De exponentenregels
lezen: machten van tien vermenigvuldigen is de exponenten optellen.
\end{proposition}

\begin{example}\label{ex:g8:powers:ten}
$10^4 \times 10^3 = 10^7$; \quad
$\dfrac{10^2}{10^5} = 10^{-3} = 0.001$. Grote en kleine hoeveelheden
worden leesbaar:
\[
\text{één miljard} = 10^9,
\qquad
\text{één miljoenste} = 10^{-6}.
\]
Gecombineerd met decimalen: $3.2 \times 10^5 = 320\,000$ en
$4.7 \times 10^{-3} = 0.0047$ (verschuif de decimale punt,
\cref{prop:g6:decimals:shift}). Het systematische gebruik van deze
schrijfwijze --- \emph{wetenschappelijke notatie} --- wordt
uitgewerkt in \cref{ch:g9:fractions}.
\end{example}

\begin{example}[Ordes van grootte]\label{ex:g8:powers:magnitude}
Licht reist ongeveer $3 \times 10^8$ m/s; een jaar heeft ongeveer
$3.2 \times 10^7$ seconden. Een lichtjaar is dus ongeveer
\[
3 \times 3.2 \times 10^{8+7} \approx 10 \times 10^{15} = 10^{16}
\text{ m}
\]
--- tien miljoen miljard meters. Machten van tien laten
astronomische berekeningen op één regel passen.
\end{example}

\begin{method}[Een uitdrukking met machten vereenvoudigen]\label{met:g8:powers:simplify}
\begin{enumerate}
  \item Groepeer de factoren grondtal voor grondtal;
  \item pas de exponentenregels binnen elk grondtal toe;
  \item bereken de overblijvende kleine machten, of laat het antwoord
        als macht staan als het groot is.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:powers:simplify}
\[
\frac{3^5 \times 3^{-2} \times 4^2}{3^2}
= \frac{3^{5 + (-2)}}{3^2} \times 16
= 3^{3 - 2} \times 16
= 3 \times 16 = 48 .
\]
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g8:powers:1}
Bereken:
\[
2^6, \qquad 3^3, \qquad 10^5, \qquad 1^{100}, \qquad 0.1^2, \qquad
6^0 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:2}
Bereken:
\[
(-3)^2, \qquad -3^2, \qquad (-1)^{15}, \qquad (-5)^3, \qquad
\left(\tfrac{2}{3}\right)^2 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:3}
Schrijf als één macht:
\[
7^4 \times 7^5, \qquad
\frac{2^9}{2^3}, \qquad
\left(10^3\right)^4, \qquad
5^6 \times 5^{-2}, \qquad
3^4 \times 7^4 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:4}
Schrijf als decimaal getal: $10^{-2}$; \quad $4 \times 10^3$; \quad
$2.5 \times 10^{-4}$; \quad $10^0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:5}
Schrijf met een macht van tien: honderdduizend; één tiende; tien
miljard; $0.000\,001$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:6}
Vereenvoudig, en bereken daarna:
\[
\frac{10^7 \times 10^{-3}}{10^2},
\qquad
\frac{2^5 \times 2^4}{2^6},
\qquad
\left(2^2\right)^3 \times 2^{-4} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:7}
Waar of onwaar? Corrigeer de foute.
\[
2^3 \times 2^4 = 2^{12}; \qquad
5^2 + 5^3 = 5^5; \qquad
(3^2)^4 = 3^8; \qquad
10^3 \times 10^3 = 100^3 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:powers:8}
Een gerucht verspreidt zich: op dag 1 weten drie personen het; elke
dag vertelt elke persoon die het weet het aan drie nieuwe personen.
Schrijf met een macht het aantal \emph{nieuwe} personen geïnformeerd
op dag $4$, en bereken hoeveel personen het gerucht kennen aan het
einde van dag 4 (inclusief de oorspronkelijke drie).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:powers:9}
Een vel papier is $0.1$ mm dik, d.w.z.\ $10^{-4}$ m. Vouwen verdubbelt
de dikte elke keer.
\begin{enumerate}
  \item Druk de dikte na $10$ vouwen uit als een product, en bereken
        die in centimeters ($2^{10} = 1024$).
  \item Na $42$ vouwen zou de dikte $2^{42} \times 10^{-4}$ m zijn,
        met $2^{42} \approx 4.4 \times 10^{12}$. Toon dat dit de
        Aarde--Maan-afstand overschrijdt, ongeveer $3.8 \times 10^8$ m.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:powers:10}
Orden van klein naar groot, zonder rekenmachine:
\[
2^{10}, \qquad 10^3, \qquad 3^6, \qquad 5^4 .
\]
(Bereken elk; $2^{10}$ en $10^3$ zijn beroemde buren.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:powers:11}
Welke is groter, $2^{100}$ of $10^{30}$? Gebruik
$2^{10} = 1024 > 10^3$ om $2^{100} = \left(2^{10}\right)^{10}$ te
vergelijken met $\left(10^3\right)^{10}$.
\end{exercise}
