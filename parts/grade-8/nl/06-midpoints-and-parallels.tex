\chapter{Middens en evenwijdigen}\label{ch:g8:midpoints}

Verbind de middens van twee zijden van een driehoek: het lijnstuk dat
je krijgt is altijd evenwijdig aan de derde zijde, en precies half zo
lang. Deze ``middenparallellenstelling'' en haar omgekeerde zijn de
eerste proef van een groot idee --- evenwijdigen snijden lengten
evenredig --- dat uitbloeit tot de stelling van Thales in
\cref{ch:g9:thales}.

\section{De middenparallellenstelling}

\begin{theorem}[Middenparallellenstelling]\label{thm:g8:midpoints:direct}
In een driehoek $ABC$, laat $I$ het midden van $[AB]$ zijn en $J$ het
midden van $[AC]$. Dan is de rechte $(IJ)$ evenwijdig aan $(BC)$, en
\[
IJ = \frac{BC}{2} .
\]
\end{theorem}

\begin{proof}[Bewijs-idee]
Laat $K$ het spiegelbeeld van $I$ ten opzichte van $J$ zijn (een
halve slag, \cref{ch:g7:central}). De vierhoek $AICK$ heeft
diagonalen $[IK]$ en $[AC]$ die snijden in hun gemeenschappelijk
midden $J$: het is een parallellogram, dus $KC$ is evenwijdig aan
$AI$, d.w.z.\ aan $(AB)$, en $KC = AI = IB$. Dan heeft $IBCK$ twee
overstaande zijden ($[IB]$ en $[KC]$) evenwijdig en gelijk: het is
ook een parallellogram (\cref{thm:g7:central:recognize}), dus
$(IK) \parallel (BC)$ --- dat is $(IJ) \parallel (BC)$ --- en
$BC = IK = 2\,IJ$.
\end{proof}

\begin{example}\label{ex:g8:midpoints:direct}
In een driehoek $ABC$ met $BC = 9$ cm worden de middens $I$ van
$[AB]$ en $J$ van $[AC]$ verbonden. Zonder enige meting:
$(IJ) \parallel (BC)$ en $IJ = \frac92 = 4.5$ cm. De kleine
driehoek $AIJ$ is een halfschaal-kopie van $ABC$ --- zijn omtrek is
ook de helft van de omtrek van $ABC$.
\end{example}

\begin{theorem}[Omgekeerde]\label{thm:g8:midpoints:converse}
In een driehoek $ABC$ snijdt de rechte door het midden $I$ van $[AB]$
en \emph{evenwijdig} aan $(BC)$ de zijde $[AC]$ in haar midden.
\end{theorem}
\admitted

\begin{example}\label{ex:g8:midpoints:converse}
Een pad doorkruist een driehoekig veld, startend in het midden van één
rand en evenwijdig lopend aan de overstaande rand. Het omgekeerde
garandeert dat het pad precies in het midden van de andere rand
uitkomt --- geen meting nodig aan de verre kant.
\end{example}

\begin{method}[Kiezen tussen stelling en omgekeerde]\label{met:g8:midpoints:choose}
\begin{enumerate}
  \item \emph{Twee middens bekend} $\Rightarrow$ gebruik de directe
        stelling: concludeer evenwijdigheid en halve lengte.
  \item \emph{Eén midden en een evenwijdige bekend} $\Rightarrow$
        gebruik het omgekeerde: concludeer dat het snijpunt een
        midden is.
  \item Schrijf welke driehoek, welke middens en welke stelling je
        gebruikt --- één zin elk.
\end{enumerate}
\end{method}

\section{Halfschaal-driehoeken}

\begin{proposition}[De middendriehoek]\label{prop:g8:midpoints:medial}
Het verbinden van de drie middens van de zijden van een driehoek
snijdt hem in vier driehoeken van gelijke oppervlakten, elk een
halfschaal-kopie van het origineel.
\end{proposition}

\begin{proof}
Elke zijde van de middendriehoek is evenwijdig aan een zijde van $ABC$
en half zo lang (\cref{thm:g8:midpoints:direct}, driemaal
toegepast). De vier kleine driehoeken hebben zijden van dezelfde drie
lengten, dus ze zijn identieke kopieën; samen betegelen ze $ABC$, dus
elk heeft een kwart van zijn oppervlakte.
\end{proof}

\begin{remark}[Op weg naar Thales]
De middenparallellenstelling is het geval ``één half'' van een
algemener feit: een lijn evenwijdig aan één zijde van een driehoek
snijdt de twee andere zijden in \emph{gelijke verhoudingen} --- één
derde en één derde, twee vijfden en twee vijfden\,\dots\ Die
algemene uitspraak is de stelling van Thales (\cref{ch:g9:thales});
het middencasus is het enige dat dit jaar nodig is.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g8:midpoints:1}
In een driehoek $RST$ is $M$ het midden van $[RS]$ en $N$ het midden
van $[RT]$, met $ST = 7$ cm. Wat kan gezegd worden over de rechte
$(MN)$ en de lengte $MN$? Citeer de gebruikte stelling.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:midpoints:2}
In een driehoek $ABC$ is $I$ het midden van $[AB]$ en snijdt de
rechte door $I$ evenwijdig aan $(BC)$ de zijde $[AC]$ in $J$, met
$AC = 11$ cm. Bereken $AJ$, met citeer van de gebruikte stelling.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:midpoints:3}
Teken een driehoek $ABC$ met $AB = 8$ cm, $AC = 6$ cm, $BC = 7$ cm,
plaats de middens $I$ van $[AB]$ en $J$ van $[AC]$, en meet $IJ$ om
de stelling te controleren. Bereken de omtrek van $AIJ$ zonder iets
anders te meten.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:midpoints:4}
$IJK$ is de middendriehoek van $ABC$, waarvan de zijden $10$, $12$ en
$16$ cm meten. Geef de drie zijden van $IJK$ en zijn omtrek. Hoe
verhouden de twee omtrekken zich?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:midpoints:5}
De oppervlakte van een driehoek is $36$ cm$^2$. Wat is de
oppervlakte van zijn middendriehoek? Van elk van de vier kleine
driehoeken?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:midpoints:6}
In een driehoek $ABC$ is $I$ het midden van $[AB]$, $J$ dat van
$[AC]$ en $K$ dat van $[BC]$. Toon dat $IJKB$ een parallellogram is.
(Vergelijk $[IJ]$ en $[BK]$: evenwijdig? gelijk?)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:midpoints:7}
Een driehoekig zeil heeft zijn laagste rand $6.4$ m lang. Een
versterkingsnaad verbindt de middens van de twee andere randen. Welke
naadlengte is nodig? Wat als de naad in plaats daarvan de punten
verbindt op één kwart van elke rand, vanaf het tophoekpunt?
(Vermod met een tekening; het bewijs is Thales',
\cref{ch:g9:thales}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:midpoints:8}
In driehoek $ABC$, rechthoekig in $A$, is $M$ het midden van de
schuine zijde $[BC]$, en snijdt de evenwijdige aan $(AB)$ door $M$ de
zijde $[AC]$ in $N$.
\begin{enumerate}
  \item Toon dat $N$ het midden van $[AC]$ is.
  \item Leg uit waarom $(MN)$ loodrecht staat op $(AC)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:midpoints:9}
Laat $ABCD$ een \emph{willekeurige} vierhoek zijn, en $I$, $J$, $K$,
$L$ de middens van $[AB]$, $[BC]$, $[CD]$, $[DA]$ (het vermoeden van
\cref{exo:g7:central:11}).
\begin{enumerate}
  \item Pas de middenparallellenstelling toe in de driehoek $ABC$ op
        het lijnstuk $[IJ]$, en in de driehoek $ACD$ op het lijnstuk
        $[LK]$: vergelijk elk met de diagonaal $[AC]$.
  \item Concludeer dat $IJKL$ altijd een parallellogram is.
\end{enumerate}
\end{exercise}
