\chapter{Letterrekenen en vergelijkingen}\label{ch:g8:equations}

Jaar~7 introduceerde letters en herleidde eenvoudige uitdrukkingen
(\cref{ch:g7:literal}). Dit hoofdstuk voegt de twee vaardigheden toe
die algebra krachtig maken: \emph{uitwerken} van producten van
haakjes, en \emph{vergelijkingen oplossen} --- het getal vinden dat
achter de letter schuilgaat.

\section{Uitwerken}

\begin{theorem}[Eenvoudige en dubbele distributiviteit]\label{thm:g8:equations:expand}
Voor alle getallen:
\[
k(a + b) = ka + kb,
\qquad
(a + b)(c + d) = ac + ad + bc + bd .
\]
\end{theorem}

\begin{proof}
De eerste regel is \cref{thm:g7:priorities:distributivity} (met tekens
inbegrepen: $k$ en de termen mogen negatief zijn,
\cref{ch:g8:negprod}). Voor de tweede, pas de eerste regel twee keer
toe, waarbij $(c + d)$ als één getal wordt behandeld:
\[
(a + b)(c + d) = a(c + d) + b(c + d) = ac + ad + bc + bd .
\qedhere
\]
\end{proof}

\begin{example}\label{ex:g8:equations:expand}
Let op de tekens bij elke term:
\begin{align*}
3(2x - 5) &= 6x - 15, \\
-2(4x - 7) &= -8x + 14
&& \text{(een min ervoor keert beide tekens om)}, \\
(x + 3)(x + 5) &= x^2 + 5x + 3x + 15 = x^2 + 8x + 15, \\
(2x - 1)(x - 4) &= 2x^2 - 8x - x + 4 = 2x^2 - 9x + 4 .
\end{align*}
\end{example}

\begin{method}[Uitwerken en herleiden]\label{met:g8:equations:reduce}
\begin{enumerate}
  \item Werk elk product uit, en schrijf elke term met zijn teken;
  \item verzamel gelijksoortige termen ($x^2$ met $x^2$, $x$ met $x$,
        getallen met getallen);
  \item controleer met een waarde: vul bijv.\ $x = 2$ in in het
        origineel en in het resultaat --- gelijke uitkomsten vangen
        de meeste fouten.
\end{enumerate}
\end{method}

\section{Vergelijkingen oplossen}

\begin{definition}[Vergelijking]\label{def:g8:equations:def}
Een \emph{vergelijking}\index{vergelijking} is een gelijkheid met een
onbekend getal, genoteerd $x$ (of een andere letter).
\emph{Oplossen} betekent alle waarden van $x$ vinden die de
gelijkheid waar maken --- de \emph{oplossingen}.
\end{definition}

\begin{theorem}[Balansregels]\label{thm:g8:equations:balance}
Een vergelijking houdt precies dezelfde oplossingen wanneer:
\begin{enumerate}
  \item hetzelfde getal aan beide kanten wordt opgeteld (of
        afgetrokken);
  \item beide kanten met hetzelfde getal \emph{ongelijk aan nul}
        worden vermenigvuldigd (of gedeeld).
\end{enumerate}
\end{theorem}
\admitted

\begin{method}[Een eerstegraadsvergelijking oplossen]\label{met:g8:equations:solve}
\begin{enumerate}
  \item Werk uit en herleid elke kant indien nodig;
  \item tel of trek aan beide kanten op/af om de $x$-termen aan één
        kant te verzamelen, de gewone getallen aan de andere;
  \item herleid, en deel daarna beide kanten door de coëfficiënt van
        $x$;
  \item \emph{controleer} door de gevonden waarde in de oorspronkelijke
        vergelijking in te vullen.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:equations:solve}
Los op $7x - 5 = 4x + 16$:
\begin{align*}
7x - 5 &= 4x + 16 \\
7x - 4x &= 16 + 5
&& \text{(tel $5$ op, trek $4x$ af, aan beide kanten)} \\
3x &= 21 \\
x &= 7 && \text{(deel beide kanten door 3).}
\end{align*}
Controle: linkerkant $7 \times 7 - 5 = 44$; rechterkant
$4 \times 7 + 16 = 44$. De oplossing is $7$.
\end{example}

\begin{example}[Met haakjes en negatieven]\label{ex:g8:equations:brackets}
Los op $3(x - 4) = 5 - 2(x + 1)$:
\begin{align*}
3x - 12 &= 5 - 2x - 2 && \text{(werk beide kanten uit)} \\
3x - 12 &= 3 - 2x && \text{(herleid)} \\
5x &= 15 && \text{(tel $2x + 12$ op aan beide kanten)} \\
x &= 3 .
\end{align*}
Controle: $3(3 - 4) = -3$ en $5 - 2(3+1) = 5 - 8 = -3$. \quad
Oplossing: $3$.
\end{example}

\begin{method}[Een probleem met een vergelijking oplossen]\label{met:g8:equations:problem}
\begin{enumerate}
  \item Kies de onbekende: ``laat $x$ \dots\ zijn'' (de gevraagde
        grootheid, of een eenvoudigere verwante);
  \item vertaal het verhaal in een vergelijking;
  \item los die op;
  \item interpreteer: beantwoord de oorspronkelijke vraag in een zin,
        met eenheden, en controleer het antwoord tegen het verhaal.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:equations:problem}
Lucas is $4$ jaar ouder dan zijn zus; samen is hun leeftijd $26$.
Laat $x$ de leeftijd van de zus zijn; dan is Lucas $x + 4$, en
\[
x + (x + 4) = 26
\ \Longrightarrow\
2x + 4 = 26
\ \Longrightarrow\
2x = 22
\ \Longrightarrow\
x = 11 .
\]
De zus is $11$, Lucas is $15$. Controle: $11 + 15 = 26$ en
$15 - 11 = 4$.
\end{example}

\section{Orde en ongelijkheden}

\begin{proposition}[Bewerkingen en orde]\label{prop:g8:equations:order}
Hetzelfde getal aan beide kanten van een ongelijkheid optellen houdt
haar in stand; beide kanten met een \emph{positief} getal
vermenigvuldigen houdt haar in stand; vermenigvuldigen met een
\emph{negatief} getal \emph{keert} haar om:
\[
2 < 5 \ \text{ maar }\ (-1) \times 2 > (-1) \times 5 .
\]
\end{proposition}
\admitted

\begin{example}\label{ex:g8:equations:inequality}
Los op $5 - 2x \leq 11$: trek $5$ af aan beide kanten,
$-2x \leq 6$; deel door $-2$ \emph{en keer om}: $x \geq -3$. Alle
getallen groter dan of gelijk aan $-3$ zijn oplossingen --- een
vergelijking heeft meestal één oplossing, een ongelijkheid heeft een
hele halfrechte ervan.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g8:equations:1}
Werk uit en herleid:
\[
5(2x + 3), \qquad
-3(4x - 1), \qquad
2(3x - 2) + 4(x + 5).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:equations:2}
Werk uit en herleid:
\[
(x + 2)(x + 7), \qquad
(3x + 1)(2x - 5), \qquad
(x - 4)(x - 6).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:equations:3}
Is $x = 3$ een oplossing van $4x - 5 = 7$? Van $2x + 1 = x + 4$? Van
$x^2 = 6x - 9$? (Vul in en vergelijk beide kanten.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:equations:4}
Los op, met controle:
\[
x + 9 = 4, \qquad
5x = -35, \qquad
\frac{x}{3} = 8, \qquad
2x - 7 = 13 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:equations:5}
Los op:
\[
6x + 5 = 2x + 25, \qquad
9 - 4x = x - 11, \qquad
3(x + 2) = 2x + 11 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:equations:6}
Los op $4(2x - 3) = 5(x + 6)$, eerst uitwerken, en controleer je
oplossing.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:equations:7}
Vertaal en los op: ``driemaal een getal, verminderd met $8$, is gelijk
aan $19$''; ``het dubbele van een getal vermeerderd met $5$ is
gelijk aan het getal vermeerderd met $12$''.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:equations:8}
De lengte van een rechthoek is $5$ cm meer dan zijn breedte, en zijn
omtrek is $58$ cm. Laat $x$ de breedte zijn: schrijf een vergelijking,
los die op en geef de twee afmetingen.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:equations:9}
Drie opeenvolgende gehele getallen hebben som $141$. Noem de
middelste $n$ en vind alle drie.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:equations:10}
Los de ongelijkheden op en beschrijf de oplossingen:
\[
3x + 4 > 19, \qquad
7 - 2x \geq 1, \qquad
5(x - 1) < 2x + 7 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:equations:11}
Mia heeft $140$ euro spaargeld en voegt elke maand $15$ euro toe; Tom
heeft $200$ euro en voegt elke maand $10$ euro toe. Na hoeveel
maanden heeft Mia strikt meer dan Tom? (Stel een ongelijkheid op.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:equations:12}
Los de vergelijking $\dfrac{x + 3}{4} = \dfrac{2x - 1}{5}$ op.
(Vermenigvuldig beide kanten met $20$, de gemeenschappelijke noemer,
en ga daarna verder zoals gewoonlijk; controleer je oplossing in de
oorspronkelijke vergelijking.)
\end{exercise}
