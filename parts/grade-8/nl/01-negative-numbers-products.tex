\chapter{Negatieve getallen vermenigvuldigen}\label{ch:g8:negprod}

Jaar~7 telde en trok relatieve getallen op/af
(\cref{ch:g7:negatives}); het rest om ze te vermenigvuldigen en te
delen. Eén vraag domineert het hoofdstuk: waarom zou ``min maal min''
plus moeten zijn? We geven een reden, niet alleen een regel.

\section{De tekenregels}

\begin{theorem}[Teken van een product]\label{thm:g8:negprod:rules}
Het product van twee relatieve getallen heeft afstand tot nul gelijk
aan het product van de afstanden, en zijn teken wordt gegeven door:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|cc}
$\times$ & $+$ & $-$ \\
\hline
$+$ & $+$ & $-$ \\
$-$ & $-$ & $+$ \\
\end{tabular}
\end{center}
\emph{Zelfde tekens: positief product. Tegengestelde tekens: negatief
product.} Dezelfde tabel geldt voor quotiënten.
\end{theorem}

\begin{proof}[Waarom $(-1) \times (-1) = +1$]
Eerst betekent $3 \times (-4)$: $(-4) + (-4) + (-4) = -12$: een
positief maal een negatief is negatief. Kijk nu naar het patroon:
\[
3 \times (-4) = -12, \quad
2 \times (-4) = -8, \quad
1 \times (-4) = -4, \quad
0 \times (-4) = 0 .
\]
Bij elke stap daalt de eerste factor met $1$ en \emph{stijgt} het
resultaat met $4$. Het patroon één stap verder zetten:
\[
(-1) \times (-4) = 0 + 4 = +4 .
\]
Elke andere keuze zou de regelmaat van de rekenkunde breken
(distributiviteit). Dus min maal min moet plus zijn.
\end{proof}

\begin{example}\label{ex:g8:negprod:products}
\begin{align*}
(-7) \times (+6) &= -42 && \text{(tegengestelde tekens)} \\
(-5) \times (-8) &= +40 && \text{(zelfde tekens)} \\
(+2.5) \times (-4) &= -10 \\
(-36) \div (-9) &= +4 \\
(+15) \div (-2) &= -7.5 .
\end{align*}
\end{example}

\section{Producten van meerdere factoren}

\begin{proposition}[Teken van een lang product]\label{prop:g8:negprod:many}
Het teken van een product van meerdere factoren ongelijk aan nul hangt
alleen af van het aantal negatieve factoren:
\begin{itemize}
  \item een \emph{even} aantal negatieve factoren geeft een positief
        product;
  \item een \emph{oneven} aantal geeft een negatief product.
\end{itemize}
\end{proposition}

\begin{proof}
Negatieve factoren paren op, en elk paar draagt een positief teken bij
(\cref{thm:g8:negprod:rules}); een oneven telling laat één
ongepaarde negatieve factor over, die het hele product negatief
maakt.
\end{proof}

\begin{example}\label{ex:g8:negprod:many}
$(-2) \times (+3) \times (-5) \times (-1)$: drie negatieve factoren
(oneven), dus het product is negatief; afstanden: $2 \times 3 \times 5
\times 1 = 30$. Resultaat: $-30$. Bepaal het teken \emph{eerst},
vermenigvuldig daarna de afstanden --- twee makkelijke taken in plaats
van één foutgevoelige.
\end{example}

\begin{example}[Machten van negatieven]\label{ex:g8:negprod:powers}
$(-2)^2 = (-2) \times (-2) = +4$, en
$(-2)^3 = (-2)^2 \times (-2) = -8$: even exponenten geven positieve
waarden, oneven exponenten behouden het teken. Let op de notatie:
$(-3)^2 = 9$ maar $-3^2 = -(3 \times 3) = -9$.
\end{example}

\section{Rekenen met alle vier de bewerkingen}

\begin{method}[Veilig rekenen met relatieve getallen]\label{met:g8:negprod:compute}
\begin{enumerate}
  \item Respecteer de rekenvolgorde van \cref{ch:g7:priorities}:
        haakjes, daarna $\times$ en $\div$, daarna $+$ en $-$;
  \item bij elke vermenigvuldiging of deling, vind het \emph{teken
        eerst}, daarna de afstand;
  \item herschrijf één stap per regel.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:negprod:compute}
\begin{align*}
5 - 3 \times (-4)
&= 5 - (-12) && \text{(eerst product: tegengestelde tekens)}\\
&= 5 + 12 = 17 .
\end{align*}
\begin{align*}
\frac{(-6) \times (-10)}{-4}
&= \frac{+60}{-4} && \text{(teller: zelfde tekens)}\\
&= -15 && \text{(quotiënt: tegengestelde tekens).}
\end{align*}
\end{example}

\begin{example}[Negatieve waarden invullen]\label{ex:g8:negprod:substitute}
Bereken $E = 3x^2 - 2x$ voor $x = -5$, met haakjes om de waarde:
\[
E = 3 \times (-5)^2 - 2 \times (-5)
= 3 \times 25 - (-10)
= 75 + 10 = 85 .
\]
De haakjes in $(-5)^2$ zijn essentieel: zonder zouden ze alleen op
$5$ slaan.
\end{example}

(-1)^{10}; \qquad

(-4) \times (5 - 9), \qquad

\end{exercise}

\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:negprod:8}

(-6) \times \;?\; = 42, \qquad
\;?\; \div (-4) = -2.5, \qquad
(-5) \times \;?\; = -1 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:negprod:9}
Waar of onwaar? Motiveer of geef een tegenvoorbeeld.
\begin{enumerate}
  \item Het product van twee relatieve getallen is altijd minstens zo
        groot als hun som.
  \item Het kwadraat van een relatief getal is nooit negatief.
  \item Als een product van drie factoren positief is, zijn alle drie
        de factoren positief.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:negprod:10}
Elk fout antwoord bij een quiz telt $-3$ punten, elk goed antwoord
$+5$. Zoe beantwoordde $20$ vragen en scoorde $36$ punten. Hoeveel
antwoorden waren goed? (Probeer waarden, of stel de berekening
$5r - 3(20 - r) = 36$ op.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:negprod:11}
Gebruik distributiviteit (zoals in het bewijs van
\cref{thm:g8:negprod:rules}), werk uit en motiveer elke stap van
\[
0 = (-1) \times 0 = (-1) \times \bigl(1 + (-1)\bigr)
= (-1) \times 1 + (-1) \times (-1),
\]
en concludeer dat $(-1) \times (-1)$ gelijk moet zijn aan $+1$.
\end{exercise}
