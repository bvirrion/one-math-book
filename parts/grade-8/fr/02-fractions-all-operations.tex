\chapter{Fractions~: les quatre opérations}\label{ch:g8:fractions}

Ce chapitre achève l'arithmétique des \omterm{def:g6:fractions:def}{fractions}~: addition et
soustraction avec \emph{n'importe quels} \omterm{def:g4:fractions:def}{dénominateurs}, multiplication,
et --- la nouveauté --- la division, qui se révèle être une
multiplication déguisée. Les signes du \cref{ch:g8:negprod}
accompagnent le voyage.

\section{Additionner avec n'importe quels dénominateurs}

\begin{method}[Dénominateur commun]\label{met:g8:fractions:add}
Pour additionner ou soustraire $\frac ab$ et $\frac cd$~:
\begin{enumerate}
  \item trouver un \omterm{def:g5:division:multiple}{multiple} commun de $b$ et $d$ (le \omterm{def:g2:mult:def}{produit}
        $b \times d$ fonctionne toujours~; un plus petit économise
        des efforts)~;
  \item réécrire les deux \omterm{def:g6:fractions:def}{fractions} avec ce \omterm{def:g4:fractions:def}{dénominateur}~;
  \item additionner ou soustraire les \omterm{def:g4:fractions:def}{numérateurs}~; simplifier.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:fractions:add}
Calculer $\dfrac{5}{6} + \dfrac{3}{4}$. Un \omterm{def:g5:division:multiple}{multiple} commun de $6$ et
$4$ est $12$~:
\[
\frac{5}{6} = \frac{10}{12},
\qquad
\frac{3}{4} = \frac{9}{12},
\qquad
\frac{5}{6} + \frac{3}{4} = \frac{10 + 9}{12} = \frac{19}{12}.
\]
Avec des signes~: $\dfrac{2}{3} - \dfrac{7}{5} = \dfrac{10}{15} -
\dfrac{21}{15} = -\dfrac{11}{15}$.
\end{example}

\section{Multiplier des fractions}

\begin{theorem}[Produit de fractions]\label{thm:g8:fractions:mult}
\[
\frac{a}{b} \times \frac{c}{d} = \frac{a \times c}{b \times d} .
\]
\end{theorem}

\begin{proof}[Pourquoi, sur une figure]
Prendre $\frac34$ de $\frac25$ d'un carré~: couper le carré en $5$
bandes verticales et en garder $2$~; couper horizontalement en $4$ et
garder $3$ rangées de ce qui reste. La partie gardée est une grille
de $3 \times 2$ petites cellules sur $4 \times 5$~: \omterm{def:g6:fractions:def}{fraction}
$\frac{6}{20}$.
\end{proof}

\begin{omfigure}
\begin{tikzpicture}[scale=0.68]
  \draw[gray!50, thin] (0,0) grid[xstep=1, ystep=1] (5,4);
  \fill[omDef!18] (0,0) rectangle (2,4);
  \fill[omThm!35] (0,0) rectangle (2,3);
  \draw[thick] (0,0) rectangle (5,4);
  \draw[omDef, thick] (2,0) -- (2,4);
  \draw[omThm, thick] (0,3) -- (5,3);
  \node[below, font=\small, omDef] at (1,0) {$\frac25$ gardé};
  \node[right, font=\small, omThm] at (5.05,1.5) {$\frac34$ de cela};
\end{tikzpicture}

{\small $\frac34 \times \frac25$~: la région doublement ombrée
compte $3 \times 2 = 6$ cellules sur $4 \times 5 = 20$, c'est-à-dire
$\frac{6}{20} = \frac{3}{10}$ --- \omterm{def:g4:fractions:def}{numérateurs} multipliés, \omterm{def:g4:fractions:def}{dénominateurs}
multipliés.}
\end{omfigure}

\begin{example}\label{ex:g8:fractions:mult}
Simplifier \emph{avant} de multiplier, en barrant les facteurs
communs~:
\[
\frac{7}{12} \times \frac{9}{14}
= \frac{7 \times 9}{12 \times 14}
= \frac{7 \times 3 \times 3}{3 \times 4 \times 2 \times 7}
= \frac{3}{8}
\]
(le facteur $7$ et un facteur $3$ s'annulent entre haut et bas).
Avec des signes, les règles de \cref{thm:g8:negprod:rules}
s'appliquent d'abord~:
$\left(-\frac23\right) \times \left(-\frac{5}{4}\right) =
+\frac{10}{12} = \frac56$.
\end{example}

\section{Diviser des fractions}

\begin{definition}[Inverse]\label{def:g8:fractions:inverse}
L'\emph{inverse}\index{inverse} d'un nombre non nul $x$ est le
nombre qui, multiplié par $x$, donne $1$. L'inverse de $\frac ab$ est
$\frac ba$, puisque $\frac ab \times \frac ba = \frac{ab}{ab} = 1$~;
l'inverse d'un entier $n$ est $\frac1n$.
\end{definition}

\begin{theorem}[Diviser, c'est multiplier par l'inverse]\label{thm:g8:fractions:div}
Pour $c \neq 0$~:
\[
\frac{a}{b} \div \frac{c}{d} = \frac{a}{b} \times \frac{d}{c} .
\]
\end{theorem}

\begin{proof}
Par définition, le \omterm{def:g3:division:remainder}{quotient} $q$ de $\frac ab$ par $\frac cd$ est le
nombre tel que $q \times \frac cd = \frac ab$. Vérifier le candidat
$q = \frac ab \times \frac dc$~:
\[
\left(\frac ab \times \frac dc\right) \times \frac cd
= \frac ab \times \left(\frac dc \times \frac cd\right)
= \frac ab \times 1 = \frac ab . \qedhere
\]
\end{proof}

\begin{example}\label{ex:g8:fractions:div}
\[
\frac{3}{5} \div \frac{7}{10}
= \frac{3}{5} \times \frac{10}{7}
= \frac{3 \times 10}{5 \times 7}
= \frac{30}{35} = \frac{6}{7}.
\]
Contrôle de bon sens~: diviser par $\frac{7}{10}$, un nombre plus
petit que $1$, doit donner un résultat \emph{plus grand} que
$\frac35$ --- et $\frac67 > \frac35$ en effet
($\frac{30}{35} > \frac{21}{35}$).
\end{example}

\begin{example}[Barres de fraction dans des barres de fraction]\label{ex:g8:fractions:complex}
Un «~double étage~» n'est qu'une division écrite verticalement~:
\[
\frac{\ \frac{2}{3}\ }{\ \frac{5}{6}\ }
= \frac{2}{3} \div \frac{5}{6}
= \frac{2}{3} \times \frac{6}{5}
= \frac{12}{15} = \frac{4}{5}.
\]
Repérer d'abord la barre \emph{principale} (la plus longue), puis
appliquer \cref{thm:g8:fractions:div}.
\end{example}

\begin{method}[Calculs mixtes]\label{met:g8:fractions:mixed}
Dans une expression mélangeant \omterm{def:g6:fractions:def}{fractions} et les quatre
opérations~:
\begin{enumerate}
  \item appliquer les priorités habituelles
        (\cref{ch:g7:priorities})~;
  \item convertir chaque division en multiplication par l'\omterm{def:g8:fractions:inverse}{inverse}~;
  \item décider les signes d'abord (\cref{met:g8:negprod:compute}),
        puis multiplier ou trouver des \omterm{def:g4:fractions:def}{dénominateurs} communs~;
  \item simplifier à la première occasion.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:fractions:mixed}
\begin{align*}
\frac12 + \frac{2}{3} \times \frac{9}{4}
&= \frac12 + \frac{2 \times 9}{3 \times 4}
&& \text{(produit d'abord~!)} \\
&= \frac12 + \frac{18}{12} = \frac12 + \frac32
= \frac{4}{2} = 2 .
\end{align*}
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g8:fractions:1}
Calculer et simplifier~:
\[
\frac{3}{4} + \frac{2}{5}, \qquad
\frac{5}{6} - \frac{3}{8}, \qquad
\frac{7}{10} + \frac{8}{15} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:fractions:2}
Calculer (signes d'abord)~:
\[
-\frac{2}{7} + \frac{3}{14}, \qquad
\frac{1}{6} - \frac{5}{4}, \qquad
-\frac{3}{5} - \frac{1}{2} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:fractions:3}
Calculer, en simplifiant avant de multiplier~:
\[
\frac{5}{8} \times \frac{4}{15}, \qquad
\frac{9}{14} \times \frac{7}{3}, \qquad
\left(-\frac{6}{5}\right) \times \frac{10}{9} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:fractions:4}
Donner l'\omterm{def:g8:fractions:inverse}{inverse} de~: $\dfrac{3}{7}$~; \quad $5$~; \quad
$\dfrac{1}{4}$~; \quad $-\dfrac{2}{9}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:fractions:5}
Calculer~:
\[
\frac{4}{9} \div \frac{2}{3}, \qquad
\frac{7}{5} \div 14, \qquad
6 \div \frac{3}{4} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:fractions:6}
Calculer les doubles étages~:
\[
\frac{\ \frac{3}{4}\ }{\ \frac{9}{8}\ },
\qquad
\frac{\ \frac{5}{6}\ }{\ 10\ },
\qquad
\frac{\ 2\ }{\ \frac{4}{7}\ } .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:fractions:7}
Calculer, en respectant les priorités~:
\[
\frac{1}{3} + \frac{1}{2} \times \frac{4}{5},
\qquad
\left(\frac{1}{3} + \frac{1}{2}\right) \times \frac{4}{5} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:fractions:8}
Un réservoir est plein aux $\frac{2}{5}$. On ajoute $\frac{1}{3}$
\emph{de la capacité du réservoir}. Quelle \omterm{def:g6:fractions:def}{fraction} du réservoir est
maintenant pleine~? Quelle \omterm{def:g6:fractions:def}{fraction} est encore vide~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:fractions:9}
Les trois \omterm{def:g3:division:half}{quarts} des élèves d'une classe ont réussi un contrôle~;
parmi ceux-ci, les deux tiers ont obtenu plus de $14$ sur $20$.
Quelle \omterm{def:g6:fractions:def}{fraction} de la classe a obtenu plus de $14$~? Si cela
représente $12$ élèves, quelle est la taille de la classe~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:fractions:10}
Une corde de $\frac{15}{2}$ m doit être coupée en morceaux de
$\frac{3}{4}$ m chacun. Combien de morceaux obtient-on~? (Une
division de \omterm{def:g6:fractions:def}{fractions} --- vérifier que la réponse est un nombre
entier.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:fractions:11}
Calculer
\[
E = \frac{2}{3} - \frac{5}{6} \div \frac{10}{9},
\qquad
F = \left(-\frac{1}{2}\right)^2 \times \frac{8}{3} .
\]
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:fractions:12}
Simplifier l'expression
\[
1 + \cfrac{1}{1 + \cfrac{1}{1 + 1}}
\]
(commencer par la \omterm{def:g6:fractions:def}{fraction} la plus intérieure et remonter, une barre
à la fois).
\end{exercise}
