\chapter{Le théorème de Pythagore}\label{ch:g8:pythagoras}

Le théorème le plus célèbre de toutes les mathématiques relie les
trois côtés d'un \omterm{def:g6:shapes:triangles}{triangle rectangle}. Avec lui, des longueurs deviennent
calculables qu'aucune règle ne mesurerait directement~: diagonales,
distances, hauteurs. C'est aussi notre première grande rencontre avec
la \omterm{ex:g1:subtraction:difference}{différence} entre un théorème et sa réciproque.

\section{Le théorème}

\begin{theorem}[Pythagore]\label{thm:g8:pythagoras:direct}
Si un triangle $ABC$ est rectangle en $A$, alors
\[
BC^2 = AB^2 + AC^2 :
\]
le carré de l'hypoténuse (le côté opposé à l'\omterm{def:g3:shapes:rightangle}{angle droit}) égale la
\omterm{def:g1:addition:def}{somme} des carrés des deux autres côtés.
\end{theorem}

\begin{proof}[Idée de preuve]
Quatre copies du triangle, disposées à l'intérieur d'un grand carré
de côté $AB + AC$, laissent découvert un carré incliné construit sur
l'hypoténuse~: comparer les \omterm{def:g6:measure:area}{aires} donne l'égalité. Le calcul détaillé
est proposé en \cref{exo:g8:pythagoras:11}~; le théorème découle
aussi du \omterm{def:g2:mult:def}{produit} scalaire, étudié dans le volume de lycée.
\end{proof}

\begin{method}[Calculer une longueur]\label{met:g8:pythagoras:compute}
Dans un \omterm{def:g6:shapes:triangles}{triangle rectangle} où deux côtés sont connus~:
\begin{enumerate}
  \item nommer l'hypoténuse (opposée à l'\omterm{def:g3:shapes:rightangle}{angle droit} --- toujours le
        plus long côté)~;
  \item écrire l'égalité de \cref{thm:g8:pythagoras:direct} avec les
        valeurs connues~;
  \item résoudre pour le carré manquant~: \emph{additionner} les
        carrés si l'hypoténuse est inconnue, \emph{soustraire} si un
        côté de l'\omterm{def:g3:shapes:rightangle}{angle droit} est inconnu~;
  \item prendre la racine carrée, exacte ou à la calculatrice, et
        vérifier que l'hypoténuse est bien sortie la plus longue.
\end{enumerate}
\end{method}

\begin{example}[Trouver l'hypoténuse]\label{ex:g8:pythagoras:hypotenuse}
Un \omterm{def:g6:shapes:triangles}{triangle rectangle} a pour côtés de l'\omterm{def:g3:shapes:rightangle}{angle droit} $6$ et $8$.
Hypoténuse $c$~:
\[
c^2 = 6^2 + 8^2 = 36 + 64 = 100,
\qquad
c = \sqrt{100} = 10 .
\]
(Le nombre $\sqrt{100}$, la \emph{racine carrée} de $100$, est le
nombre positif dont le carré est $100$~; les racines carrées ont un
chapitre entier au \cref{ch:g9:sqrt}.)
\end{example}

\begin{example}[Trouver un côté de l'angle droit]\label{ex:g8:pythagoras:leg}
Une échelle de $5$ m s'appuie contre un mur, son pied à $1.4$ m du
mur. Hauteur atteinte $h$~: l'échelle est l'hypoténuse, donc
\[
h^2 = 5^2 - 1.4^2 = 25 - 1.96 = 23.04,
\qquad
h = \sqrt{23.04} = 4.8 \text{ m}.
\]
Soustraire, ne pas additionner~: l'inconnue ici est un \emph{côté de
l'\omterm{def:g3:shapes:rightangle}{angle droit}}.
\end{example}

\section{La réciproque}

\begin{theorem}[Réciproque de Pythagore]\label{thm:g8:pythagoras:converse}
Si, dans un triangle $ABC$, les côtés satisfont
$BC^2 = AB^2 + AC^2$, alors le triangle est rectangle en $A$.
\end{theorem}
\admitted

\begin{method}[Tester un angle droit]\label{met:g8:pythagoras:test}
Étant donné les trois côtés d'un triangle~:
\begin{enumerate}
  \item calculer séparément le carré du plus long côté, et la \omterm{def:g1:addition:def}{somme}
        des carrés des deux autres~;
  \item si les deux résultats sont égaux, le triangle est rectangle
        (réciproque), au \omterm{def:g5:solids:def}{sommet} opposé au plus long côté~;
  \item s'ils diffèrent, il n'est pas rectangle --- car le théorème de
        Pythagore forcerait l'égalité.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:pythagoras:test}
Côtés $5$, $12$, $13$~: carré du plus long côté, $13^2 = 169$~;
\omterm{def:g1:addition:def}{somme} des autres carrés, $5^2 + 12^2 = 25 + 144 = 169$. Égaux~:
rectangle (opposé au côté $13$).

Côtés $6$, $7$, $9$~: $9^2 = 81$ mais $6^2 + 7^2 = 85 \neq 81$~:
pas un \omterm{def:g6:shapes:triangles}{triangle rectangle} (il l'est «~presque~» --- les nombres
décident, pas l'œil).
\end{example}

\begin{remark}[Théorème contre réciproque]
Le théorème et sa réciproque disent des choses différentes~: le
théorème \emph{utilise} un \omterm{def:g3:shapes:rightangle}{angle droit} pour calculer des longueurs~;
la réciproque utilise des longueurs pour \emph{prouver} un \omterm{def:g3:shapes:rightangle}{angle
droit}. Les bâtisseurs ont utilisé la réciproque depuis des
millénaires~: une corde nouée à $3$, $4$, $5$ unités, tendue, donne
un \omterm{def:g3:shapes:rightangle}{angle droit} parfait.
\end{remark}

\begin{example}[Diagonale d'un carré]\label{ex:g8:pythagoras:diagonal}
La diagonale d'un carré de côté $1$ est l'hypoténuse d'un \omterm{def:g6:shapes:triangles}{triangle
rectangle} de côtés de l'\omterm{def:g3:shapes:rightangle}{angle droit} $1$ et $1$~:
\[
d^2 = 1^2 + 1^2 = 2 ,
\]
donc $d = \sqrt2 \approx 1.414$ --- un nombre qui n'est pas une
\omterm{def:g6:fractions:def}{fraction}, comme le prouvera le volume de lycée. Pour un carré de côté
$c$, la diagonale est $c\sqrt2$.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:1}
Dans chaque \omterm{def:g6:shapes:triangles}{triangle rectangle}, calculer l'hypoténuse~:
côtés de l'\omterm{def:g3:shapes:rightangle}{angle droit} $9$ et $12$~; \quad $5$ et $12$~; \quad $8$
et $15$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:2}
Un \omterm{def:g6:shapes:triangles}{triangle rectangle} a pour hypoténuse $25$ et un côté de l'\omterm{def:g3:shapes:rightangle}{angle
droit} $7$. Calculer l'autre côté de l'\omterm{def:g3:shapes:rightangle}{angle droit}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:3}
Calculer la diagonale d'un rectangle de côtés $12$ cm et $9$ cm~; la
diagonale d'un carré de côté $5$ cm (valeur exacte avec une racine
carrée, puis arrondie au mm).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:4}
Quels triangles sont rectangles~? Justifier avec
\cref{met:g8:pythagoras:test}~:
\[
(20,\ 21,\ 29); \qquad
(7,\ 8,\ 11); \qquad
(1.5,\ 2,\ 2.5).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:5}
Un portail est contreventé par une planche diagonale. Le portail
mesure $3$ m de large et $1.6$ m de haut~: quelle longueur a la
planche~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:pythagoras:6}
Un \omterm{def:g6:shapes:triangles}{triangle isocèle} a deux côtés de $10$ cm et une base de $12$ cm.
Sa hauteur le partage en deux \omterm{def:g6:shapes:triangles}{triangles rectangles}~: calculer cette
hauteur, puis l'\omterm{def:g6:measure:area}{aire} du triangle.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:pythagoras:7}
Un écran de télévision est un rectangle $16{:}9$ mesurant $88.5$ cm
sur $49.8$ cm. Les écrans se vendent par la diagonale, en pouces
($1$ pouce $= 2.54$ cm). Calculer la diagonale en cm, puis en pouces
(\omterm{def:g4:numbers:round}{arrondir} au pouce le plus proche).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:pythagoras:8}
Un bateau navigue $24$ km vers l'est, puis $10$ km vers le nord. À
quelle distance est-il de son point de départ (à vol d'oiseau)~?
Dessiner d'abord la situation.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:pythagoras:9}
Une échelle de $2.5$ m doit atteindre un rebord de fenêtre à $2.4$ m
du sol. À quelle distance du mur faut-il placer son pied~? La réponse
est-elle compatible avec les conseils de sécurité (pied à environ un
\omterm{def:g3:division:half}{quart} de la longueur de l'échelle du mur)~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:pythagoras:10}
Sur un quadrillage, placer $A(1, 2)$ et $B(5, 5)$, et tracer le
\omterm{def:g6:shapes:triangles}{triangle rectangle} de côtés de l'\omterm{def:g3:shapes:rightangle}{angle droit} \omterm{def:g6:lines:perp}{parallèles} aux axes dont
l'hypoténuse est $[AB]$. Calculer $AB$. (Cette construction, pour
deux points quelconques, devient la formule de distance de la
géométrie analytique, dans le volume de lycée.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:pythagoras:11}
Quatre copies d'un \omterm{def:g6:shapes:triangles}{triangle rectangle} de côtés de l'\omterm{def:g3:shapes:rightangle}{angle droit} $a$,
$b$ et d'hypoténuse $c$ sont placées aux coins d'un carré de côté
$a + b$, laissant un carré incliné de côté $c$ découvert au milieu.
\begin{enumerate}
  \item Exprimer l'\omterm{def:g6:measure:area}{aire} du grand carré de deux façons~: directement,
        et comme (quatre triangles)~$+$~(carré incliné).
  \item Développer $(a + b)^2$ (voir \cref{thm:g8:equations:expand})
        et en déduire $c^2 = a^2 + b^2$~: vous avez prouvé le
        théorème de Pythagore.
\end{enumerate}
\end{exercise}
