\chapter{Milieux et parallèles}\label{ch:g8:midpoints}

Joindre les milieux de deux côtés d'un triangle~: le segment obtenu
est toujours parallèle au troisième côté, et exactement deux fois
plus court. Ce «~théorème des milieux~» et sa réciproque sont le
premier avant-goût d'une grande idée --- les parallèles découpent des
longueurs proportionnellement --- qui s'épanouit en le théorème de
Thalès au \cref{ch:g9:thales}.

\section{Le théorème des milieux}

\begin{theorem}[Théorème des milieux]\label{thm:g8:midpoints:direct}
Dans un triangle $ABC$, soit $I$ le milieu de $[AB]$ et $J$ le milieu
de $[AC]$. Alors la droite $(IJ)$ est parallèle à $(BC)$, et
\[
IJ = \frac{BC}{2} .
\]
\end{theorem}

\begin{proof}[Idée de preuve]
Soit $K$ le symétrique de $I$ par rapport à $J$ (un demi-tour,
\cref{ch:g7:central}). Le quadrilatère $AICK$ a des diagonales
$[IK]$ et $[AC]$ se croisant en leur milieu commun $J$~: c'est un
parallélogramme, donc $KC$ est parallèle à $AI$, c'est-à-dire à
$(AB)$, et $KC = AI = IB$. Alors $IBCK$ a deux côtés opposés
($[IB]$ et $[KC]$) parallèles et égaux~: c'est un parallélogramme
aussi (\cref{thm:g7:central:recognize}), donc
$(IK) \parallel (BC)$ --- c'est-à-dire $(IJ) \parallel (BC)$ --- et
$BC = IK = 2\,IJ$.
\end{proof}

\begin{example}\label{ex:g8:midpoints:direct}
Dans un triangle $ABC$ avec $BC = 9$ cm, les milieux $I$ de $[AB]$ et
$J$ de $[AC]$ sont joints. Sans aucune mesure~:
$(IJ) \parallel (BC)$ et $IJ = \frac92 = 4.5$ cm. Le petit triangle
$AIJ$ est une copie à demi-échelle de $ABC$ --- son périmètre est
aussi la moitié du périmètre de $ABC$.
\end{example}

\begin{theorem}[Réciproque]\label{thm:g8:midpoints:converse}
Dans un triangle $ABC$, la droite passant par le milieu $I$ de $[AB]$
et \emph{parallèle} à $(BC)$ coupe le côté $[AC]$ en son milieu.
\end{theorem}
\admitted

\begin{example}\label{ex:g8:midpoints:converse}
Un chemin traverse un champ triangulaire, partant du milieu d'un côté
et courant parallèle au côté opposé. La réciproque garantit que le
chemin sort exactement au milieu de l'autre côté --- sans mesure
nécessaire du côté lointain.
\end{example}

\begin{method}[Choisir entre théorème et réciproque]\label{met:g8:midpoints:choose}
\begin{enumerate}
  \item \emph{Deux milieux connus} $\Rightarrow$ utiliser le théorème
        direct~: conclure parallélisme et demi-longueur.
  \item \emph{Un milieu et une parallèle connus} $\Rightarrow$
        utiliser la réciproque~: conclure que le point d'intersection
        est un milieu.
  \item Écrire quel triangle, quels milieux et quel théorème on
        utilise --- une phrase chacun.
\end{enumerate}
\end{method}

\section{Triangles à demi-échelle}

\begin{proposition}[Le triangle des milieux]\label{prop:g8:midpoints:medial}
Joindre les trois milieux des côtés d'un triangle le découpe en
quatre triangles d'aires égales, chacun une copie à demi-échelle de
l'original.
\end{proposition}

\begin{proof}
Chaque côté du triangle des milieux est parallèle à un côté de $ABC$
et deux fois plus court (\cref{thm:g8:midpoints:direct}, appliqué
trois fois). Les quatre petits triangles ont des côtés des trois
mêmes longueurs, donc ce sont des copies identiques~; ensemble ils
pavent $ABC$, donc chacun a le quart de son aire.
\end{proof}

\begin{remark}[Vers Thalès]
Le théorème des milieux est le cas «~une moitié~» d'un fait plus
général~: une droite parallèle à un côté d'un triangle coupe les deux
autres côtés dans des \emph{rapports égaux} --- un tiers et un tiers,
deux cinquièmes et deux cinquièmes\,\dots\ Cet énoncé général est le
théorème de Thalès (\cref{ch:g9:thales})~; le cas des milieux est le
seul nécessaire cette année.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g8:midpoints:1}
Dans un triangle $RST$, $M$ est le milieu de $[RS]$ et $N$ le milieu
de $[RT]$, avec $ST = 7$ cm. Que peut-on dire de la droite $(MN)$ et
de la longueur $MN$~? Citer le théorème utilisé.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:midpoints:2}
Dans un triangle $ABC$, $I$ est le milieu de $[AB]$ et la droite
passant par $I$ parallèle à $(BC)$ coupe $[AC]$ en $J$, avec
$AC = 11$ cm. Calculer $AJ$, en citant le théorème utilisé.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:midpoints:3}
Tracer un triangle $ABC$ avec $AB = 8$ cm, $AC = 6$ cm, $BC = 7$ cm,
placer les milieux $I$ de $[AB]$ et $J$ de $[AC]$, et mesurer $IJ$
pour vérifier le théorème. Calculer le périmètre de $AIJ$ sans rien
mesurer d'autre.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:midpoints:4}
$IJK$ est le triangle des milieux de $ABC$, dont les côtés mesurent
$10$, $12$ et $16$ cm. Donner les trois côtés de $IJK$ et son
périmètre. Comment se comparent les deux périmètres~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:midpoints:5}
L'aire d'un triangle est $36$ cm$^2$. Quelle est l'aire de son
triangle des milieux~? De chacun des quatre petits triangles~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:midpoints:6}
Dans un triangle $ABC$, $I$ est le milieu de $[AB]$, $J$ celui de
$[AC]$ et $K$ celui de $[BC]$. Montrer que $IJKB$ est un
parallélogramme. (Comparer $[IJ]$ et $[BK]$~: parallèles~? égaux~?)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:midpoints:7}
Une voile triangulaire a son bord le plus bas long de $6.4$ m. Une
couture de renfort joint les milieux des deux autres bords. Quelle
longueur de couture faut-il~? Et si la couture joignait au contraire
les points situés au quart de chaque bord, en partant du sommet du
haut~? (Conjecture avec un dessin~; la preuve est celle de Thalès,
\cref{ch:g9:thales}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:midpoints:8}
Dans le triangle $ABC$, rectangle en $A$, $M$ est le milieu de
l'hypoténuse $[BC]$, et la parallèle à $(AB)$ passant par $M$
rencontre $[AC]$ en $N$.
\begin{enumerate}
  \item Montrer que $N$ est le milieu de $[AC]$.
  \item Expliquer pourquoi $(MN)$ est perpendiculaire à $(AC)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:midpoints:9}
Soit $ABCD$ un quadrilatère \emph{quelconque}, et $I$, $J$, $K$, $L$
les milieux de $[AB]$, $[BC]$, $[CD]$, $[DA]$ (la conjecture de
\cref{exo:g7:central:11}).
\begin{enumerate}
  \item Appliquer le théorème des milieux dans le triangle $ABC$ au
        segment $[IJ]$, et dans le triangle $ACD$ au segment $[LK]$~:
        comparer chacun à la diagonale $[AC]$.
  \item Conclure que $IJKL$ est toujours un parallélogramme.
\end{enumerate}
\end{exercise}
