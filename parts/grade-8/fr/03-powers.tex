\section{Les rÃĻgles des exposants}

\begin{theorem}[RÃĻgles des exposants]\label{thm:g8:powers:rules}
Pour une base non nulle $a$ et des \omterm{def:g8:powers:def}{exposants} entiers $m$, $n$~:
\[
a^m \times a^n = a^{m+n},
\qquad
\frac{a^m}{a^n} = a^{m-n},
\qquad
\left(a^m\right)^n = a^{m \times n},
\qquad
(ab)^n = a^n b^n .
\]
\end{theorem}

\begin{proof}[Preuve en comptant les facteurs]
$a^m \times a^n$ aligne $m$ facteurs $a$ suivis de $n$ autres~:
$m + n$ au total. $\left(a^m\right)^n$ rÃĐpÃĻte un bloc de $m$ facteurs

d'annuler $n$ des $m$ facteurs~; avec les conventions $a^0 = 1$ et
$a^{-n} = \frac{1}{a^n}$, elle reste vraie mÃŠme lorsque $n \geq m$.
\end{proof}

\begin{example}\label{ex:g8:powers:rules}
\[

\qquad

\qquad
