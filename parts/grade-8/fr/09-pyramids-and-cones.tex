\chapter{Pyramides et cônes}\label{ch:g8:solids}

Après les \omterm{def:g7:areas:prism}{prismes} et \omterm{def:g7:areas:prism}{cylindres} du \cref{ch:g7:areas} --- \omterm{def:g5:solids:def}{solides} à
section constante --- viennent les \omterm{def:g5:solids:def}{solides} à pointe~: \omterm{def:g8:solids:def}{pyramides} et
\omterm{def:g8:solids:def}{cônes}. Leur formule de \omterm{def:g6:measure:volume}{volume} porte un célèbre facteur $\frac13$, que
ce chapitre rend plausible et utilise~; l'étude des \omterm{def:g5:solids:def}{solides} se
poursuit avec les sphères et les sections au \cref{ch:g9:solids}.

\section{Décrire pyramides et cônes}

\begin{definition}[Pyramide, cône]\label{def:g8:solids:def}
\begin{itemize}
  \item Une \emph{pyramide}\index{pyramide} a un \omterm{def:g4:geometry:polygon}{polygone} pour
        \emph{base} et un point, le \emph{\omterm{def:g5:solids:def}{sommet}}, joint à chaque
        \omterm{def:g5:solids:def}{sommet} de la base par des \omterm{def:g5:solids:def}{faces} triangulaires. Sa
        \emph{hauteur} est la distance du \omterm{def:g5:solids:def}{sommet} au plan de la base.
  \item Un \emph{cône}\index{cône} est la même construction sur un
        disque~: un disque de base, un \omterm{def:g5:solids:def}{sommet}, et une surface latérale
        courbe.
\end{itemize}
Une pyramide est \emph{régulière} lorsque sa base est un \omterm{def:g4:geometry:polygon}{polygone}
régulier (carré, \omterm{def:g6:shapes:triangles}{triangle équilatéral}, \dots) et que son \omterm{def:g5:solids:def}{sommet} se
trouve à la verticale du centre de la base.
\end{definition}

\begin{example}[Compter faces, arêtes, sommets]\label{ex:g8:solids:count}
Une \omterm{def:g8:solids:def}{pyramide} à base carrée a $5$ \omterm{def:g5:solids:def}{faces} ($1$ carré $+ 4$ triangles),
$8$ \omterm{def:g5:solids:def}{arêtes} et $5$ \omterm{def:g5:solids:def}{sommets}. Avec une base à $n$ côtés~: $n + 1$
\omterm{def:g5:solids:def}{faces}, $2n$ \omterm{def:g5:solids:def}{arêtes}, $n + 1$ \omterm{def:g5:solids:def}{sommets}. Vérifier le petit motif
d'Euler~: \omterm{def:g5:solids:def}{faces} $+$ \omterm{def:g5:solids:def}{sommets} $=$ \omterm{def:g5:solids:def}{arêtes} $+ 2$.
\end{example}

\section{Volume}

\begin{theorem}[Volume d'une pyramide ou d'un cône]\label{thm:g8:solids:volume}
Pour une \omterm{def:g8:solids:def}{pyramide} ou un \omterm{def:g8:solids:def}{cône} d'\omterm{def:g6:measure:area}{aire} de base $B$ et de hauteur $h$~:
\[
V = \frac{1}{3}\, B \times h
\]
--- un tiers du \omterm{def:g7:areas:prism}{prisme} ou du \omterm{def:g7:areas:prism}{cylindre} de même base et même hauteur.
Pour un \omterm{def:g8:solids:def}{cône} de \omterm{def:g3:shapes:shapes}{rayon} de base $r$~: $V = \frac13 \pi r^2 h$.
\end{theorem}
\admitted

\begin{remark}[Pourquoi un tiers~?]
Remplir une \omterm{def:g8:solids:def}{pyramide} creuse d'eau et la verser dans le \omterm{def:g7:areas:prism}{prisme} de même
base et même hauteur~: il faut exactement trois remplissages. Mieux
encore~: un \omterm{def:g5:solids:def}{cube} peut se découper en trois \omterm{def:g8:solids:def}{pyramides} identiques,
chacune ayant une \omterm{def:g5:solids:def}{face} du \omterm{def:g5:solids:def}{cube} pour base et un \omterm{def:g5:solids:def}{sommet} du \omterm{def:g5:solids:def}{cube} pour
apex (essayer de se le représenter~!). Une preuve générale utilise
l'intégration, un outil du volume de lycée.
\end{remark}

\begin{example}\label{ex:g8:solids:volume}
Une \omterm{def:g8:solids:def}{pyramide} a une base carrée de côté $5$ cm et une hauteur de
$9$ cm~:
\begin{enumerate}
  \item \omterm{def:g6:measure:area}{aire} de base~: $B = 5^2 = 25$ cm$^2$~;
  \item \omterm{def:g6:measure:volume}{volume}~: $V = \frac13 \times 25 \times 9 = 25 \times 3 = 75$
        cm$^3$.
\end{enumerate}
Un cornet de glace de \omterm{def:g3:shapes:shapes}{rayon} $3$ cm et de hauteur $10$ cm~:
\[
V = \frac13 \times \pi \times 3^2 \times 10 = 30\pi \approx 94
\text{ cm}^3 .
\]
\end{example}

\begin{example}[Attention à la hauteur]\label{ex:g8:solids:slant}
Pour un \omterm{def:g8:solids:def}{cône}, ne pas confondre la hauteur $h$ (\omterm{def:g5:solids:def}{sommet} au centre de la
base, \omterm{def:g6:lines:perp}{perpendiculaire}) avec la \emph{génératrice} $s$ (\omterm{def:g5:solids:def}{sommet} au
bord). Elles sont liées par Pythagore
(\cref{thm:g8:pythagoras:direct}) dans le \omterm{def:g6:shapes:triangles}{triangle rectangle}
sommet--centre--bord~:
\[
s^2 = h^2 + r^2 .
\]
Un \omterm{def:g8:solids:def}{cône} avec $r = 3$ et $s = 5$ a donc pour hauteur
$h = \sqrt{25 - 9} = 4$, et pour \omterm{def:g6:measure:volume}{volume}
$\frac13 \pi \times 9 \times 4 = 12\pi$.
\end{example}

\begin{method}[Problèmes de volume]\label{met:g8:solids:method}
\begin{enumerate}
  \item Identifier le \omterm{def:g5:solids:def}{solide} (\omterm{def:g7:areas:prism}{prisme}/\omterm{def:g7:areas:prism}{cylindre}~: $V = Bh$~;
        \omterm{def:g8:solids:def}{pyramide}/\omterm{def:g8:solids:def}{cône}~: $V = \frac13 Bh$)~;
  \item calculer d'abord l'\omterm{def:g6:measure:area}{aire} de base $B$, comme une étape
        séparée~;
  \item vérifier que la hauteur est \omterm{def:g6:lines:perp}{perpendiculaire} à la base
        (utiliser Pythagore si la génératrice est donnée)~;
  \item multiplier, garder des unités cohérentes, convertir à la fin
        si besoin ($1$ L $= 1000$ cm$^3$).
\end{enumerate}
\end{method}

\section{Patrons}

\begin{example}[Patron d'une pyramide]\label{ex:g8:solids:net}
Déplier une \omterm{def:g8:solids:def}{pyramide} régulière à base carrée la met à plat en son
\emph{\omterm{def:g5:solids:net}{patron}}~: la base carrée avec quatre \omterm{def:g6:shapes:triangles}{triangles isocèles}
identiques attachés à ses côtés. Les côtés égaux des triangles sont
les \emph{\omterm{def:g5:solids:def}{arêtes} latérales} de la \omterm{def:g8:solids:def}{pyramide} --- leur longueur n'est ni
$h$ ni la hauteur d'une \omterm{def:g5:solids:def}{face}, donc étiqueter soigneusement avant de
découper le carton.
\end{example}

\end{tikzpicture}

{\small Le \omterm{def:g5:solids:net}{patron} d'une \omterm{def:g8:solids:def}{pyramide} à base carrée~: replier les quatre
triangles jusqu'à ce que leurs \omterm{def:g5:solids:def}{sommets} se rejoignent.}
\end{omfigure}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g8:solids:1}
Combien de \omterm{def:g5:solids:def}{faces}, d'\omterm{def:g5:solids:def}{arêtes} et de \omterm{def:g5:solids:def}{sommets} a une \omterm{def:g8:solids:def}{pyramide} à base
triangulaire (un \emph{tétraèdre})~? À base hexagonale~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:2}
Calculer le \omterm{def:g6:measure:volume}{volume} d'une \omterm{def:g8:solids:def}{pyramide} d'\omterm{def:g6:measure:area}{aire} de base $36$ cm$^2$ et de
hauteur $10$ cm~; d'une \omterm{def:g8:solids:def}{pyramide} de base rectangulaire
$6 \times 4$ cm et de hauteur $7.5$ cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:3}
Calculer le \omterm{def:g6:measure:volume}{volume} d'un \omterm{def:g8:solids:def}{cône} de \omterm{def:g3:shapes:shapes}{rayon} $5$ cm et de hauteur $9$ cm
(exact avec $\pi$, puis \omterm{def:g6:decimals:rounding}{arrondi} au cm$^3$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:4}
Un \omterm{def:g7:areas:prism}{cylindre} et un \omterm{def:g8:solids:def}{cône} ont le même \omterm{def:g3:shapes:shapes}{rayon} $4$ cm et la même hauteur
$12$ cm. Calculer les deux \omterm{def:g6:measure:volume}{volumes}. Quel est leur rapport~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:5}
Une \omterm{def:g8:solids:def}{pyramide} a pour \omterm{def:g6:measure:volume}{volume} $56$ cm$^3$ et pour hauteur $8$ cm. Quelle
est son \omterm{def:g6:measure:area}{aire} de base~? (Écrire la formule du \omterm{def:g6:measure:volume}{volume} et résoudre.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:6}
Esquisser le \omterm{def:g5:solids:net}{patron} d'un \omterm{def:g8:solids:def}{cône} (un disque pour la base, et un secteur
d'un disque plus grand pour la surface latérale). Quelle mesure du
\omterm{def:g8:solids:def}{cône} est le \omterm{def:g3:shapes:shapes}{rayon} de ce secteur~: la hauteur $h$ ou la génératrice
$s$~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:7}
La \omterm{def:g8:solids:def}{pyramide} du Louvre a une base carrée de côté environ $35$ m et une
hauteur d'environ $22$ m. Estimer son \omterm{def:g6:measure:volume}{volume}, à la centaine de mètres
cubes près.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:8}
Un \omterm{def:g8:solids:def}{cône} a pour génératrice $13$ cm et pour \omterm{def:g3:shapes:shapes}{rayon} de base $5$ cm.
Calculer sa hauteur, puis son \omterm{def:g6:measure:volume}{volume} (exact avec $\pi$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:9}
Un verre conique de \omterm{def:g3:shapes:shapes}{rayon} $4$ cm et de hauteur $9$ cm est rempli à
ras bord, puis versé dans un verre cylindrique de \omterm{def:g3:shapes:shapes}{rayon} $4$ cm.
Quelle hauteur le liquide atteint-il dans le \omterm{def:g7:areas:prism}{cylindre}~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:10}
Un sablier est formé de deux \omterm{def:g8:solids:def}{cônes} identiques (\omterm{def:g3:shapes:shapes}{rayon} $2$ cm, hauteur
$4.5$ cm) joints par leurs \omterm{def:g5:solids:def}{sommets}. Tout le sable remplit exactement
un \omterm{def:g8:solids:def}{cône}. Calculer le \omterm{def:g6:measure:volume}{volume} de sable (exact, puis en cm$^3$ \omterm{def:g6:decimals:rounding}{arrondi}
au dixième), et la \omterm{def:g6:fractions:def}{fraction} du \omterm{def:g6:measure:volume}{volume} intérieur total du sablier
occupée par le sable.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:solids:11}
Une \omterm{def:g8:solids:def}{pyramide} à base carrée de côté $6$ cm a son \omterm{def:g5:solids:def}{sommet} à la verticale
d'un coin de la base, à la hauteur $8$ cm.
\begin{enumerate}
  \item La formule $V = \frac13 Bh$ s'applique-t-elle encore~?
        (Oui --- le \omterm{def:g5:solids:def}{sommet} n'a pas besoin d'être centré. Calculer
        $V$.)
  \item Calculer la longueur de la plus longue \omterm{def:g5:solids:def}{arête} latérale, du
        \omterm{def:g5:solids:def}{sommet} au coin opposé de la base (deux étapes de Pythagore~:
        la diagonale de la base d'abord).
\end{enumerate}
\end{exercise}
