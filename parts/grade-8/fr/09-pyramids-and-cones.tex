\chapter{Pyramides et cônes}\label{ch:g8:solids}

Après les prismes et cylindres du \cref{ch:g7:areas} --- solides à
section constante --- viennent les solides à pointe~: pyramides et
cônes. Leur formule de volume porte un célèbre facteur $\frac13$, que
ce chapitre rend plausible et utilise~; l'étude des solides se
poursuit avec les sphères et les sections au \cref{ch:g9:solids}.

\section{Décrire pyramides et cônes}

\begin{definition}[Pyramide, cône]\label{def:g8:solids:def}
\begin{itemize}
  \item Une \emph{pyramide}\index{pyramide} a un polygone pour
        \emph{base} et un point, le \emph{sommet}, joint à chaque
        sommet de la base par des faces triangulaires. Sa
        \emph{hauteur} est la distance du sommet au plan de la base.
  \item Un \emph{cône}\index{cône} est la même construction sur un
        disque~: un disque de base, un sommet, et une surface latérale
        courbe.
\end{itemize}
Une pyramide est \emph{régulière} lorsque sa base est un polygone
régulier (carré, triangle équilatéral, \dots) et que son sommet se
trouve à la verticale du centre de la base.
\end{definition}

\begin{example}[Compter faces, arêtes, sommets]\label{ex:g8:solids:count}
Une pyramide à base carrée a $5$ faces ($1$ carré $+ 4$ triangles),
$8$ arêtes et $5$ sommets. Avec une base à $n$ côtés~: $n + 1$
faces, $2n$ arêtes, $n + 1$ sommets. Vérifier le petit motif
d'Euler~: faces $+$ sommets $=$ arêtes $+ 2$.
\end{example}

\section{Volume}

\begin{theorem}[Volume d'une pyramide ou d'un cône]\label{thm:g8:solids:volume}
Pour une pyramide ou un cône d'aire de base $B$ et de hauteur $h$~:
\[
V = \frac{1}{3}\, B \times h
\]
--- un tiers du prisme ou du cylindre de même base et même hauteur.
Pour un cône de rayon de base $r$~: $V = \frac13 \pi r^2 h$.
\end{theorem}
\admitted

\begin{remark}[Pourquoi un tiers~?]
Remplir une pyramide creuse d'eau et la verser dans le prisme de même
base et même hauteur~: il faut exactement trois remplissages. Mieux
encore~: un cube peut se découper en trois pyramides identiques,
chacune ayant une face du cube pour base et un sommet du cube pour
apex (essayer de se le représenter~!). Une preuve générale utilise
l'intégration, un outil du volume de lycée.
\end{remark}

\begin{example}\label{ex:g8:solids:volume}
Une pyramide a une base carrée de côté $5$ cm et une hauteur de
$9$ cm~:
\begin{enumerate}
  \item aire de base~: $B = 5^2 = 25$ cm$^2$~;
  \item volume~: $V = \frac13 \times 25 \times 9 = 25 \times 3 = 75$
        cm$^3$.
\end{enumerate}
Un cornet de glace de rayon $3$ cm et de hauteur $10$ cm~:
\[
V = \frac13 \times \pi \times 3^2 \times 10 = 30\pi \approx 94
\text{ cm}^3 .
\]
\end{example}

\begin{example}[Attention à la hauteur]\label{ex:g8:solids:slant}
Pour un cône, ne pas confondre la hauteur $h$ (sommet au centre de la
base, perpendiculaire) avec la \emph{génératrice} $s$ (sommet au
bord). Elles sont liées par Pythagore
(\cref{thm:g8:pythagoras:direct}) dans le triangle rectangle
sommet--centre--bord~:
\[
s^2 = h^2 + r^2 .
\]
Un cône avec $r = 3$ et $s = 5$ a donc pour hauteur
$h = \sqrt{25 - 9} = 4$, et pour volume
$\frac13 \pi \times 9 \times 4 = 12\pi$.
\end{example}

\begin{method}[Problèmes de volume]\label{met:g8:solids:method}
\begin{enumerate}
  \item Identifier le solide (prisme/cylindre~: $V = Bh$~;
        pyramide/cône~: $V = \frac13 Bh$)~;
  \item calculer d'abord l'aire de base $B$, comme une étape
        séparée~;
  \item vérifier que la hauteur est perpendiculaire à la base
        (utiliser Pythagore si la génératrice est donnée)~;
  \item multiplier, garder des unités cohérentes, convertir à la fin
        si besoin ($1$ L $= 1000$ cm$^3$).
\end{enumerate}
\end{method}

\section{Patrons}

\begin{example}[Patron d'une pyramide]\label{ex:g8:solids:net}
Déplier une pyramide régulière à base carrée la met à plat en son
\emph{patron}~: la base carrée avec quatre triangles isocèles
identiques attachés à ses côtés. Les côtés égaux des triangles sont
les \emph{arêtes latérales} de la pyramide --- leur longueur n'est ni
$h$ ni la hauteur d'une face, donc étiqueter soigneusement avant de
découper le carton.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g8:solids:1}
Combien de faces, d'arêtes et de sommets a une pyramide à base
triangulaire (un \emph{tétraèdre})~? À base hexagonale~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:2}
Calculer le volume d'une pyramide d'aire de base $36$ cm$^2$ et de
hauteur $10$ cm~; d'une pyramide de base rectangulaire
$6 \times 4$ cm et de hauteur $7.5$ cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:3}
Calculer le volume d'un cône de rayon $5$ cm et de hauteur $9$ cm
(exact avec $\pi$, puis arrondi au cm$^3$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:4}
Un cylindre et un cône ont le même rayon $4$ cm et la même hauteur
$12$ cm. Calculer les deux volumes. Quel est leur rapport~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:5}
Une pyramide a pour volume $56$ cm$^3$ et pour hauteur $8$ cm. Quelle
est son aire de base~? (Écrire la formule du volume et résoudre.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:solids:6}
Esquisser le patron d'un cône (un disque pour la base, et un secteur
d'un disque plus grand pour la surface latérale). Quelle mesure du
cône est le rayon de ce secteur~: la hauteur $h$ ou la génératrice
$s$~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:7}
La pyramide du Louvre a une base carrée de côté environ $35$ m et une
hauteur d'environ $22$ m. Estimer son volume, à la centaine de mètres
cubes près.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:8}
Un cône a pour génératrice $13$ cm et pour rayon de base $5$ cm.
Calculer sa hauteur, puis son volume (exact avec $\pi$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:9}
Un verre conique de rayon $4$ cm et de hauteur $9$ cm est rempli à
ras bord, puis versé dans un verre cylindrique de rayon $4$ cm.
Quelle hauteur le liquide atteint-il dans le cylindre~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:solids:10}
Un sablier est formé de deux cônes identiques (rayon $2$ cm, hauteur
$4.5$ cm) joints par leurs sommets. Tout le sable remplit exactement
un cône. Calculer le volume de sable (exact, puis en cm$^3$ arrondi
au dixième), et la fraction du volume intérieur total du sablier
occupée par le sable.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:solids:11}
Une pyramide à base carrée de côté $6$ cm a son sommet à la verticale
d'un coin de la base, à la hauteur $8$ cm.
\begin{enumerate}
  \item La formule $V = \frac13 Bh$ s'applique-t-elle encore~?
        (Oui --- le sommet n'a pas besoin d'être centré. Calculer
        $V$.)
  \item Calculer la longueur de la plus longue arête latérale, du
        sommet au coin opposé de la base (deux étapes de Pythagore~:
        la diagonale de la base d'abord).
\end{enumerate}
\end{exercise}
