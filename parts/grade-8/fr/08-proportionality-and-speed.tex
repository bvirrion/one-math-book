\chapter{Proportionnalité, vitesse et moyennes}\label{ch:g8:speed}

La proportionnalité (\cref{ch:g7:prop}) rencontre maintenant le monde
physique~: vitesse, débits, conversions d'unités --- des quantités qui
sont des \emph{quotients} de deux autres. Le chapitre se clôt sur les
moyennes pondérées, où la pensée proportionnelle évite une erreur
classique.

\section{Vitesse moyenne}

\begin{definition}[Vitesse moyenne]\label{def:g8:speed:def}
La \emph{vitesse moyenne}\index{vitesse} d'un trajet est le quotient
\[
v = \frac{d}{t}
\qquad
\left(\text{distance divisée par durée}\right),
\]
en kilomètres par heure (km/h), mètres par seconde (m/s), \dots\ La
formule s'inverse~: $d = v \times t$ et $t = \frac{d}{v}$.
\end{definition}

\begin{example}\label{ex:g8:speed:basic}
Un train parcourt $270$ km en $1$ h $30$ min. Convertir d'abord le
temps~: $1$ h $30$ min $= 1.5$ h. Puis
\[
v = \frac{270}{1.5} = 180 \text{ km/h}.
\]
Distance à $90$ km/h pendant $2$ h $20$ min $= \frac73$ h~:
$d = 90 \times \frac73 = 210$ km. Temps pour $35$ km à $14$ km/h~:
$t = \frac{35}{14} = 2.5$ h $= 2$ h $30$ min.
\end{example}

\begin{remark}[Les minutes ne sont pas des décimaux]
$1$ h $30$ min vaut $1.5$ h, mais $1$ h $20$ min n'est \emph{pas}
$1.2$ h~: c'est $1 + \frac{20}{60} = \frac43 \approx 1.33$ h. Toujours
convertir les minutes en fraction d'heure ($60$ min $= 1$ h) avant de
diviser.
\end{remark}

\begin{method}[Convertir des vitesses]\label{met:g8:speed:convert}
Pour convertir des km/h en m/s~: $1$ km $= 1000$ m et $1$ h $= 3600$ s,
donc
\[
v \text{ km/h} = \frac{v \times 1000}{3600} \text{ m/s}
= \frac{v}{3.6} \text{ m/s}.
\]
Pour convertir des m/s en km/h, multiplier par $3.6$.
\end{method}

\begin{example}\label{ex:g8:speed:convert}
$90$ km/h $= \frac{90}{3.6} = 25$ m/s. Un sprinter parcourant $100$ m
en $10$ s se déplace à $10$ m/s $= 36$ km/h.
\end{example}

\section{Grandeurs quotients}

\begin{example}[Débit, densité, prix au kilo]\label{ex:g8:speed:quotients}
La vitesse a de nombreux cousins, tous traités de la même façon~:
\begin{itemize}
  \item un robinet remplit $48$ L en $4$ min~: débit
        $\frac{48}{4} = 12$ L/min, donc remplir une baignoire de
        $150$ L prend $\frac{150}{12} = 12.5$ min~;
  \item $0.6$ kg de fromage coûtent $9$ euros~: prix
        $\frac{9}{0.6} = 15$ euros par kg~;
  \item une voiture consomme $6.3$ L pour $90$ km~: consommation
        $\frac{6.3}{90} \times 100 = 7$ L pour $100$ km.
\end{itemize}
Identifier le quotient, et la formule à trois voies
($q = \frac ab$, $a = qb$, $b = \frac aq$) fait le reste.
\end{example}

\section{Moyennes pondérées}

\begin{definition}[Moyenne pondérée]\label{def:g8:speed:average}
Lorsque des valeurs $v_1, v_2, \dots$ s'accompagnent d'effectifs (ou
de poids) $n_1, n_2, \dots$, leur \emph{moyenne
pondérée}\index{moyenne pondérée} est
\[
\bar v = \frac{n_1 v_1 + n_2 v_2 + \dots}{n_1 + n_2 + \dots} :
\]
total des valeurs, divisé par le total des poids.
\end{definition}

\begin{example}\label{ex:g8:speed:average}
Un contrôle a été passé par deux groupes~: groupe 1, $12$ élèves,
moyenne $11$~; groupe 2, $18$ élèves, moyenne $14$. Moyenne de toute
la classe~:
\[
\bar v = \frac{12 \times 11 + 18 \times 14}{12 + 18}
= \frac{132 + 252}{30} = \frac{384}{30} = 12.8 .
\]
\emph{Pas} $\frac{11 + 14}{2} = 12.5$~: le plus grand groupe tire la
moyenne vers la sienne --- les poids comptent.
\end{example}

\begin{example}[Vitesse moyenne sur deux étapes]\label{ex:g8:speed:twolegs}
Un cycliste parcourt $30$ km à $30$ km/h, puis $30$ km à $15$ km/h. La
vitesse moyenne sur tout le trajet n'est \emph{pas}
$\frac{30 + 15}{2} = 22.5$ km/h. Calculer avec la définition~:
\begin{enumerate}
  \item temps~: $\frac{30}{30} = 1$ h, puis $\frac{30}{15} = 2$ h~;
        total $3$ h~;
  \item distance totale $60$ km, donc
        $v = \frac{60}{3} = 20$ km/h.
\end{enumerate}
L'étape lente dure plus longtemps, donc elle pèse plus --- une moyenne
de vitesses doit être pondérée par le \emph{temps}.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g8:speed:1}
Calculer la vitesse moyenne~: $150$ km en $2$ h~; $27$ km en $45$
min~; $100$ m en $12.5$ s.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:2}
Calculer la distance~: $2$ h à $85$ km/h~; $40$ min à $90$ km/h.
Calculer le temps~: $315$ km à $90$ km/h (en h et min).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:3}
Convertir~: $72$ km/h en m/s~; $5$ m/s en km/h~; $108$ km/h en m/s.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:4}
Un robinet débite $15$ L/min. Combien de temps pour remplir un
réservoir de $600$ L~? Combien de litres en $2$ h $30$~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:5}
Quelle est la meilleure affaire~: $1.2$ kg de pommes pour $3.30$
euros, ou $0.8$ kg pour $2.16$ euros~? Comparer les prix au kilogramme.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:6}
Une classe de $25$ élèves a une moyenne de $12.4$ à un contrôle~;
une autre classe de $35$ élèves a une moyenne de $10$. Calculer la
moyenne des deux classes réunies.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:speed:7}
Le son se propage à environ $340$ m/s. On voit un éclair et on entend
le tonnerre $6$ secondes plus tard. À quelle distance a frappé
l'éclair (à $100$ m près)~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:speed:8}
Un randonneur marche $2$ h à $5$ km/h, se repose $30$ min, puis marche
$1$ h $30$ à $4$ km/h. Calculer la distance totale et la vitesse
moyenne \emph{y compris le repos} (distance totale sur le temps
écoulé total).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:speed:9}
Notes avec coefficients~: une élève a obtenu $15$ (coefficient $3$),
$9$ (coefficient $2$) et $12$ (coefficient $1$). Calculer la moyenne
pondérée. Quelle moyenne simple (non pondérée) aurait-elle, et
pourquoi diffèrent-elles~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:speed:10}
Une voiture parcourt $120$ km à $80$ km/h puis $120$ km à $120$ km/h.
\begin{enumerate}
  \item Calculer la durée de chaque étape, puis la vitesse moyenne sur
        tout le trajet.
  \item Expliquer pourquoi la réponse est inférieure à $100$ km/h, le
        milieu des deux vitesses.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:speed:11}
Deux villages sont distants de $18$ km. Anna part du premier à
$10{:}00$ en marchant à $5$ km/h vers le second~; Boris part du second
au même moment en marchant à $4$ km/h vers le premier. À quelle heure
se rencontrent-ils, et à quelle distance du village d'Anna~?
(Ensemble ils referment l'écart à $5 + 4$ km/h.)
\end{exercise}
