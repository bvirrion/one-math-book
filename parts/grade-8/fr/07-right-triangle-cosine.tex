\chapter{Le triangle rectangle et le cosinus}\label{ch:g8:cosine}

Deux beaux faits lient le \omterm{def:g6:shapes:triangles}{triangle rectangle} au cercle~: un \omterm{def:g6:shapes:triangles}{triangle
rectangle} s'inscrit exactement dans un demi-cercle, son hypoténuse
étant un \omterm{def:g4:geometry:circle}{diamètre}. Et un nombre, le \emph{cosinus} d'un angle, code
la forme de tout \omterm{def:g6:shapes:triangles}{triangle rectangle} ayant cet angle --- le premier
rapport trigonométrique, avant ses frères sinus et tangente
(\cref{ch:g9:trig}).

\section{Le triangle rectangle et son cercle}

\begin{theorem}[Théorème du cercle]\label{thm:g8:cosine:circle}
\begin{enumerate}
  \item Si un triangle $ABC$ est rectangle en $A$, alors $A$ se
        trouve sur le cercle de \omterm{def:g4:geometry:circle}{diamètre} l'hypoténuse $[BC]$.
  \item Réciproquement, si $A$ se trouve sur un cercle de \omterm{def:g4:geometry:circle}{diamètre}
        $[BC]$ (avec $A \neq B, C$), alors le triangle $ABC$ est
        rectangle en $A$.
\end{enumerate}
De façon équivalente~: dans un \omterm{def:g6:shapes:triangles}{triangle rectangle}, le milieu de
l'hypoténuse est à égale distance des trois \omterm{def:g5:solids:def}{sommets} --- la médiane
issue de l'\omterm{def:g3:shapes:rightangle}{angle droit} mesure la \omterm{def:g3:division:half}{moitié} de l'hypoténuse.
\end{theorem}

\begin{proof}[Preuve du 1]
Soit $O$ le milieu de $[BC]$ et $D$ le \omterm{def:g7:central:def}{symétrique} de $A$ par rapport
à $O$. Les diagonales de $ABDC$ se coupent en leur milieu commun $O$,
donc $ABDC$ est un \omterm{def:g7:central:parallelogram}{parallélogramme}
(\cref{def:g7:central:parallelogram}) --- avec un \omterm{def:g3:shapes:rightangle}{angle droit} en
$A$~: c'est un rectangle. Les diagonales d'un rectangle sont égales,
donc $OA = \frac{AD}{2} = \frac{BC}{2}$~: le point $A$ est à
distance $\frac{BC}{2}$ de $O$, c'est-à-dire sur le cercle de
\omterm{def:g4:geometry:circle}{diamètre} $[BC]$. (Le point 2 se prouve en parcourant l'argument à
l'envers.)
\end{proof}

Un triangle a une hypoténuse de $10$ cm. Sans rien savoir d'autre, la

circonscrit au triangle a pour \omterm{def:g3:shapes:shapes}{rayon} $5$ cm, centré au milieu de
l'hypoténuse.
\end{example}

\section{Le cosinus d'un angle aigu}

\begin{definition}[Cosinus]\label{def:g8:cosine:def}
Dans un \omterm{def:g6:shapes:triangles}{triangle rectangle}, pour un angle aigu $\theta$~:
\[
\cos\theta = \frac{\text{côté adjacent à } \theta}{\text{hypoténuse}}
\]
--- le côté de l'\omterm{def:g3:shapes:rightangle}{angle droit} touchant $\theta$, divisé par
l'hypoténuse. Ce rapport ne dépend que de l'angle, pas de la taille
du triangle~: tous les \omterm{def:g6:shapes:triangles}{triangles rectangles} de même angle aigu sont
des agrandissements les uns des autres, et les agrandissements
conservent les rapports de longueurs (\cref{ch:g8:midpoints} a
commencé cette histoire~; \cref{ch:g9:thales} l'achève).
\end{definition}

petit est une réduction du grand, donc
$\frac{\text{adjacent}}{\text{hypoténuse}}$ est le même pour les
deux --- cette valeur commune est $\cos\theta$.}
\end{omfigure}

\begin{proposition}[Premières valeurs et bornes]\label{prop:g8:cosine:bounds}
Pour tout angle aigu $\theta$~: $0 < \cos\theta < 1$, et le cosinus
\emph{diminue} lorsque l'angle s'ouvre~: un angle plus large a un
cosinus plus petit. Repères~: $\cos 0^\circ = 1$,
$\cos 60^\circ = \frac12$, $\cos 90^\circ = 0$ (les valeurs extrêmes
correspondant aux triangles aplatis).
\end{proposition}
\admitted

\begin{method}[Utiliser le cosinus]\label{met:g8:cosine:use}
Dans un \omterm{def:g6:shapes:triangles}{triangle rectangle}, lorsque les quantités connues et
cherchées sont un angle aigu, son côté adjacent et l'hypoténuse~:
\begin{enumerate}
  \item écrire la définition~:
        $\cos\theta = \frac{\text{adjacent}}{\text{hypoténuse}}$ avec
        les valeurs connues à leur place~;
  \item résoudre la petite \omterm{def:g8:equations:def}{équation} pour l'inconnue (multiplier ou
        diviser)~;
  \item pour un \emph{angle} inconnu, appliquer le $\cos^{-1}$ de la
        calculatrice au rapport calculé~;
  \item contrôles de bon sens~: une longueur doit sortir plus courte
        que l'hypoténuse~; un angle strictement entre $0^\circ$ et
        $90^\circ$.
\end{enumerate}
\end{method}

\begin{example}[Trouver un côté]\label{ex:g8:cosine:side}
Une rampe de $6$ m fait un angle de $20^\circ$ avec le sol horizontal.
Distance horizontale parcourue (adjacente à $20^\circ$, hypoténuse
$6$ m)~:
\[
\cos 20^\circ = \frac{d}{6}
\quad\Longrightarrow\quad
d = 6 \cos 20^\circ \approx 6 \times 0.940 \approx 5.6 \text{ m}.
\]
\end{example}

\begin{example}[Trouver un angle]\label{ex:g8:cosine:angle}
Dans un \omterm{def:g6:shapes:triangles}{triangle rectangle}, le côté adjacent à l'angle $\theta$
mesure $3.5$ et l'hypoténuse $5$~:
\[
\cos\theta = \frac{3.5}{5} = 0.7,
\qquad
\theta = \cos^{-1}(0.7) \approx 45.6^\circ .
\]
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g8:cosine:1}
Un \omterm{def:g6:shapes:triangles}{triangle rectangle} a une hypoténuse de $13$ cm. Quelle est la
longueur de la médiane issue du \omterm{def:g5:solids:def}{sommet} de l'\omterm{def:g3:shapes:rightangle}{angle droit}~? Quel est le
\omterm{def:g3:shapes:shapes}{rayon} du cercle passant par les trois \omterm{def:g5:solids:def}{sommets}~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:2}
Tracer un cercle de \omterm{def:g4:geometry:circle}{diamètre} $8$ cm avec un \omterm{def:g4:geometry:circle}{diamètre} $[BC]$, choisir
un point $A$ quelconque sur le cercle et tracer $ABC$. Quel angle est
droit~? Citer le théorème. Mesurer $AB$ et $AC$ et vérifier
Pythagore.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:3}
Dans un triangle $DEF$ rectangle en $D$, nommer le côté adjacent à
l'angle $\widehat E$, le côté opposé, et l'hypoténuse. Écrire
$\cos \widehat E$ comme un rapport.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:4}
Calculer la quantité manquante ($\cos 35^\circ \approx 0.819$,
$\cos 50^\circ \approx 0.643$)~:
\begin{enumerate}
  \item hypoténuse $10$, angle $35^\circ$~: côté adjacent~?
  \item côté adjacent $6$, angle $50^\circ$~: hypoténuse~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:5}
Dans un \omterm{def:g6:shapes:triangles}{triangle rectangle}, le côté adjacent à $\theta$ mesure $8$ et
l'hypoténuse $10$. Calculer $\cos\theta$, puis $\theta$
($\cos^{-1}(0.8) \approx 36.9^\circ$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:6}
Expliquer pourquoi $\cos\theta$ ne peut jamais valoir $1.2$ pour un
angle aigu d'un \omterm{def:g6:shapes:triangles}{triangle rectangle}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:7}
Une tyrolienne de $25$ m descend d'une plateforme jusqu'au sol, faisant
un angle de $12^\circ$ avec l'horizontale ($\cos 12^\circ \approx
0.978$). Quelle distance horizontale couvre-t-elle~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:8}
Ranger sans calculatrice~: $\cos 20^\circ$, $\cos 70^\circ$,
$\cos 45^\circ$ (utiliser \cref{prop:g8:cosine:bounds}). Puis
vérifier à la calculatrice.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:9}
Une échelle de longueur $L$ s'appuie contre un mur avec un angle
$\theta$ entre l'échelle et le \emph{sol}. Son pied est à $1.2$ m du
mur et $\theta = 68^\circ$ ($\cos 68^\circ \approx 0.375$). Calculer
$L$, puis la hauteur atteinte (Pythagore).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:10}
$[BC]$ est un \omterm{def:g4:geometry:circle}{diamètre} d'un cercle de centre $O$ et de \omterm{def:g3:shapes:shapes}{rayon}
$4.5$ cm, et $A$ est un point du cercle avec $AB = 5.4$ cm.
\begin{enumerate}
  \item Pourquoi $ABC$ est-il rectangle en $A$~?
  \item Calculer $AC$, puis $\cos \widehat B$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:cosine:11}
En utilisant la \omterm{def:g3:division:half}{moitié} d'un \omterm{def:g6:shapes:triangles}{triangle équilatéral} de côté $1$ (coupé
le long d'une hauteur), justifier le repère $\cos 60^\circ = \frac12$
de \cref{prop:g8:cosine:bounds}. (Où tombe le pied de la hauteur sur
la base~? La version trigonométrique complète est
\cref{ex:g9:trig:special}.)
\end{exercise}
