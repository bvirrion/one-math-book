\chapter{El triángulo rectángulo y el coseno}\label{ch:g8:cosine}

Dos hermosos hechos vinculan el \omterm{def:g6:shapes:triangles}{triangulo rectángulo} al círculo: un \omterm{def:g6:shapes:triangles}{correcto
triangulo} encaja exactamente en un semicírculo, siendo su hipotenusa una
\omterm{def:g4:geometry:circle}{diámetro}. Y un número, el \emph{coseno} de un ángulo, codifica el
forma de cada \omterm{def:g6:shapes:triangles}{triangulo rectángulo} con ese ángulo --- el primero
razón trigonométrica, por delante de sus hermanos seno y tangente
(\cref{ch:g9:trig}).

\section{El triángulo rectángulo y su círculo.}

\begin{theorem}[Teorema del círculo]\label{thm:g8:cosine:circle}
\begin{enumerate}
  \item Si un triángulo $ABC$ es \omterm{def:g6:shapes:triangles}{en ángulo recto} en $A$, entonces $A$ se encuentra en
        el círculo cuyo \omterm{def:g4:geometry:circle}{diámetro} es la hipotenusa $[BC]$.
  \item Por el contrario, si $A$ se encuentra en un círculo de \omterm{def:g4:geometry:circle}{diámetro} $[BC]$ (con
        $A \neq B, C$), entonces el triángulo $ABC$ es \omterm{def:g6:shapes:triangles}{en ángulo recto} en
        $A$.
\end{enumerate}
De manera equivalente: en un \omterm{def:g6:shapes:triangles}{triangulo rectángulo}, el punto medio de la hipotenusa es
a igual distancia de los tres \omterm{def:g5:solids:def}{vértices} --- la mediana del
\omterm{def:g3:shapes:rightangle}{ángulo recto} mide \omterm{def:g3:division:half}{medio} la hipotenusa.
\end{theorem}

\begin{proof}[Prueba de 1]
Sea $O$ el punto medio de $[BC]$ y $D$ el \omterm{def:g7:central:def}{simétrico} de $A$ aproximadamente
$O$. Las diagonales de $ABDC$ se cortan en su punto medio común $O$, por lo que
$ABDC$ es un \omterm{def:g7:central:parallelogram}{paralelogramo} (\cref{def:g7:central:parallelogram}) ---
con un \omterm{def:g3:shapes:rightangle}{ángulo recto} en $A$: es un rectángulo. Las diagonales de un
rectángulo son iguales, entonces $OA = \frac{AD}{2} = \frac{BC}{2}$: el
el punto $A$ está a la distancia $\frac{BC}{2}$ de $O$, es decir,\ en el círculo
de \omterm{def:g4:geometry:circle}{diámetro} $[BC]$. (El punto 2 se prueba ejecutando el argumento
hacia atrás.)
\end{proof}

Un triángulo tiene una hipotenusa de $10$ cm. sin saber nada
de lo contrario, la mediana del \omterm{def:g3:shapes:rightangle}{ángulo recto} mide $5$ cm, y el

el punto medio de la hipotenusa.
\end{example}

\section{El coseno de un ángulo agudo.}

\begin{definition}[Coseño]\label{def:g8:cosine:def}
En un \omterm{def:g6:shapes:triangles}{triangulo rectángulo}, para un ángulo agudo $\theta$:
\[
\cos\theta = \frac{\text{side adjacent to } \theta}{\text{hypotenuse}}
\]
--- el cateto que toca $\theta$, dividido por la hipotenusa. Esta relación
depende sólo del ángulo, no del tamaño del triángulo: todos los \omterm{def:g6:shapes:triangles}{correcto
triangulos} con el mismo ángulo agudo son ampliaciones entre sí,
y las ampliaciones preservan las proporciones de longitudes
(\cref{ch:g8:midpoints} comenzó esta historia; \cref{ch:g9:thales}
lo termina).
\end{definition}

$\frac{\text{adjacent}}{\text{hypotenuse}}$ es igual para ambos ---
ese valor común es $\cos\theta$.}
\end{omfigure}

\begin{proposition}[Primeros valores y límites]\label{prop:g8:cosine:bounds}
Para cada ángulo agudo $\theta$: $0 < \cos\theta < 1$, y el coseno
\emph{disminuye} a medida que se abre el ángulo: un ángulo más amplio tiene un ángulo más pequeño
coseno. Puntos de referencia: $\cos 0^\circ = 1$, $\cos 60^\circ = \frac12$,
$\cos 90^\circ = 0$ (los valores extremos correspondientes a aplanados
triángulos).
\end{proposition}
\admitted

\begin{method}[Usando el coseno]\label{met:g8:cosine:use}
En un \omterm{def:g6:shapes:triangles}{triangulo rectángulo}, cuando las cantidades conocidas y deseadas son una
ángulo agudo, su lado adyacente y la hipotenusa:
\begin{enumerate}
  \item escribe la definición:
        $\cos\theta = \frac{\text{adjacent}}{\text{hypotenuse}}$ con
        los valores conocidos vigentes;
  \item resuelve el pequeño \omterm{def:g8:equations:def}{ecuación} para la incógnita (multiplicar o dividir);
  \item para un \emph{ángulo} desconocido, aplique la
        $\cos^{-1}$ a la relación calculada;
  \item controles de cordura: un largo debe salir más corto que el
        hipotenusa; un ángulo estrictamente entre $0^\circ$ y
        $90^\circ$.
\end{enumerate}
\end{method}

\begin{example}[Encontrando un lado]\label{ex:g8:cosine:side}
Una rampa de $6$ m forma un ángulo de $20^\circ$ con el suelo horizontal.
Distancia horizontal recorrida (adyacente a $20^\circ$, hipotenusa
$6$m):
\[
\cos 20^\circ = \frac{d}{6}
\quad\Longrightarrow\quad
d = 6 \cos 20^\circ \approx 6 \times 0.940 \approx 5.6 \text{ m}.
\]
\end{example}

\begin{example}[Encontrar un ángulo]\label{ex:g8:cosine:angle}
En un \omterm{def:g6:shapes:triangles}{triangulo rectángulo}, el lado adyacente al ángulo $\theta$ mide
$3.5$ y la hipotenusa $5$:
\[
\cos\theta = \frac{3.5}{5} = 0.7,
\qquad
\theta = \cos^{-1}(0.7) \approx 45.6^\circ .
\]
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g8:cosine:1}
A \omterm{def:g6:shapes:triangles}{triangulo rectángulo} tiene una hipotenusa de $13$ cm. ¿Cuál es la longitud de
la mediana desde el ángulo recto \omterm{def:g5:solids:def}{vértice}? ¿Qué es el \omterm{def:g3:shapes:shapes}{radio} del
circula por los tres \omterm{def:g5:solids:def}{vértices}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:2}
Dibuja un círculo de \omterm{def:g4:geometry:circle}{diámetro} $8$ cm con un \omterm{def:g4:geometry:circle}{diámetro} $[BC]$, elige cualquiera
apunte $A$ en el círculo y dibuje $ABC$. ¿Qué ángulo es el correcto? citar
el teorema. Mida $AB$ y $AC$ y verifique a Pitágoras.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:3}
En un triángulo $DEF$ \omterm{def:g6:shapes:triangles}{en ángulo recto} en $D$, nombre el lado adyacente a
el ángulo $\widehat E$, el lado opuesto a él y la hipotenusa.
Escribe $\cos \widehat E$ como una proporción.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:4}
Calcule la cantidad que falta ($\cos 35^\circ \approx 0.819$,
$\cos 50^\circ \approx 0.643$):
\begin{enumerate}
  \item hipotenusa $10$, ángulo $35^\circ$: ¿lado adyacente?
  \item lado adyacente $6$, ángulo $50^\circ$: ¿hipotenusa?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:5}
En un \omterm{def:g6:shapes:triangles}{triangulo rectángulo}, el lado adyacente a $\theta$ mide $8$ y
la hipotenusa $10$. Calcule $\cos\theta$, luego $\theta$
($\cos^{-1}(0.8) \approx 36.9^\circ$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:cosine:6}
Explique por qué $\cos\theta$ nunca puede ser igual a $1.2$ para un ángulo agudo de
a \omterm{def:g6:shapes:triangles}{triangulo rectángulo}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:7}
Una tirolesa de $25$ m desciende desde una plataforma hasta el suelo, realizando un
ángulo de $12^\circ$ con la horizontal ($\cos 12^\circ \approx
0.978$). ¿Qué distancia horizontal abarca?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:8}
Pedido sin calculadora: $\cos 20^\circ$, $\cos 70^\circ$,
$\cos 45^\circ$ (use \cref{prop:g8:cosine:bounds}). Luego consulte con un
calculadora.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:9}
Una escalera de longitud $L$ se apoya contra una pared con un ángulo $\theta$
entre la escalera y \emph{suelo}. Su pie está a $1.2$ m de la pared.
y $\theta = 68^\circ$ ($\cos 68^\circ \approx 0.375$). Calcular $L$,
luego alcanzó la altura (Pitágoras).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:cosine:10}
$[BC]$ es un \omterm{def:g4:geometry:circle}{diámetro} de un círculo de centro $O$ y \omterm{def:g3:shapes:shapes}{radio} $4.5$ cm,
y $A$ es un punto del círculo con $AB = 5.4$ cm.
\begin{enumerate}
  \item ¿Por qué $ABC$ \omterm{def:g6:shapes:triangles}{en ángulo recto} está en $A$?
  \item Calcule $AC$ y luego $\cos \widehat B$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:cosine:11}
Usando \omterm{def:g3:division:half}{medio} y \omterm{def:g6:shapes:triangles}{triangulo equilatero} del lado $1$ (cortado a lo largo de una altura),
justificar el hito $\cos 60^\circ = \frac12$ de
\cref{prop:g8:cosine:bounds}. (¿De dónde viene el pie de la altura?
caer sobre la base? La versión trigonométrica completa es
\cref{ex:g9:trig:special}.)
\end{exercise}

\section{Problema: las relaciones de Euclides y Pitágoras por todos lados
otra vez}

\begin{problem}[{Problema de fin de semana --- la altitud de una derecha
triangulo: $AB^2 = BH \times BC$, $AH^2 = BH \times HC$, un segundo
prueba de Pitágoras y una máquina para construir raíces cuadradas}]
\label{pb:g8:cosine:1}
Soltar, del \omterm{def:g3:shapes:rightangle}{ángulo recto} de un \omterm{def:g6:shapes:triangles}{triangulo rectángulo}, el \omterm{def:g6:lines:perp}{perpendicular}
a la hipotenusa: este corto \omterm{def:g6:lines:objects}{segmento} --- el \emph{altitud} ---
satisface relaciones tan útiles que Euclides las puso en el centro de
su \emph{Elementos}. En este problema el coseno define
propiedad, ``la relación depende sólo del ángulo''
(\cref{def:g8:cosine:def}), prueba todas ellas; en el camino tu
volverá a demostrar el teorema de Pitágoras por una ruta completamente diferente,
y terminar con una máquina de regla y compás que construye
$\sqrt2$, $\sqrt5$, $\sqrt{n}$ para cada número entero $n$.

En todo momento, $ABC$ es un triángulo \omterm{def:g6:shapes:triangles}{en ángulo recto} en $A$ y $H$ es
el pie del \omterm{def:g6:lines:perp}{perpendicular} desde $A$ hasta la hipotenusa $[BC]$,
de modo que $H$ se encuentra entre $B$, $C$ y $AH \perp BC$.

\medskip
\noindent\textbf{Parte I --- One angle, two triangles.}
\begin{enumerate}
  \item Usando el ángulo \omterm{def:g1:addition:def}{suma} de un triángulo
        (\cref{thm:g7:triangles:sum}) en $ABH$ y en $ABC$, mostrar
        que $\widehat{BAH} = \widehat{ACB}$: la altitud corta
        el \omterm{def:g6:shapes:triangles}{triangulo rectángulo} en dos triángulos más pequeños que llevan
        El \emph{mismos ángulos} como el original.
  \item Los triángulos $ABC$ (\omterm{def:g6:shapes:triangles}{en ángulo recto} en $A$) y $ABH$
        (\omterm{def:g6:shapes:triangles}{en ángulo recto} en $H$) comparten el ángulo $\widehat B$. escribir
        $\cos\widehat B$ en cada uno de ellos, y deducir de
        \cref{def:g8:cosine:def} que
        \[
        AB^2 = BH \times BC .
        \]
        (Para pasar de razones iguales a igual \omterm{def:g2:mult:def}{productos}, multiplicar
        ambos lados por ambos \omterm{def:g4:fractions:def}{denominadores},
        \cref{thm:g8:equations:balance}.)
  \item Indique y pruebe la relación de gemela en el \omterm{def:g5:solids:def}{vértice} $C$:
        \[
        AC^2 = CH \times CB .
        \]
  \item Tome el triángulo $3$--$4$--$5$: $AB = 3$, $AC = 4$,
        $BC = 5$. Calcule $BH$ y $CH$ a partir de las preguntas 2 y 3,
        y verifique que $BH + HC = BC$.
  \item En general, exprese $BH$ y $CH$ como \omterm{def:g6:fractions:def}{fracciones} que involucran
        sólo los tres lados. ¿En qué sentido el pie $H$
        dividir la hipotenusa "proporcionalmente a los cuadrados de
        las piernas''?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Pythagoras again, and the altitude.}
\begin{enumerate}[resume]
  \item Agregue las relaciones de las preguntas 2 y 3 y use
        $BH + HC = BC$ para obtener
        \[
        AB^2 + AC^2 = BC^2 :
        \]
        Teorema de Pitágoras (\cref{thm:g8:pythagoras:direct}),
        re-probado --- sin ningún rompecabezas \omterm{def:g6:measure:area}{área} a la vista (comparar
        \cref{exo:g8:pythagoras:11}).
  \item Aplica Pitágoras en el pequeño triángulo $ABH$ para escribir
        $AH^2 = AB^2 - BH^2$, sustituir $AB^2$ por
        $BH \times BC$ y factorice $BH$ para demostrar
        \emph{Relación de altitud de Euclides}:
        \[
        AH^2 = BH \times HC .
        \]
  \item Calcule el \omterm{def:g6:measure:area}{área} de $ABC$ de dos maneras: patas como base
        y altura, luego hipotenusa como base
        (\cref{thm:g7:areas:triangle}) --- y deducir el tercero
        relación:
        \[
        AB \times AC = BC \times AH .
        \]
  \item Regresar al triángulo $3$--$4$--$5$: calcular $AH$ con
        pregunta 8, luego verifique la relación de altitud de la pregunta
        7 numéricamente, usando los valores de $BH$ y $CH$ que se encuentran en
        pregunta 4.
  \item El pie de altitud divide la hipotenusa de algunos \omterm{def:g6:shapes:triangles}{correcto
        triangulo} en \omterm{def:g6:lines:objects}{segmentos} $BH = 2$ cm y $HC = 8$ cm.
        Calcule la altitud $AH$.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- A machine for square roots.}
Dibuja un \omterm{def:g6:lines:objects}{segmento} $[BC]$ formado por dos piezas colocadas una al lado de la otra:
$BH = p$ y $HC = q$. Dibuja el semicírculo de \omterm{def:g4:geometry:circle}{diámetro} $[BC]$,
y sea $A$ el punto donde \omterm{def:g6:lines:perp}{perpendicular} a $(BC)$ en $H$
lo cumple.
\begin{enumerate}[resume]
  \item Utilizando \cref{thm:g8:cosine:circle}, explique por qué
        el triángulo $ABC$ es \omterm{def:g6:shapes:triangles}{en ángulo recto} en $A$ --- entonces la pregunta 7
        aplica y
        \[
        AH^2 = p \times q .
        \]
        (La longitud $AH$ se denomina \emph{geométrico
        significar}\index{media geométrica} de $p$ y $q$).
  \item Tome $p = 1$ y $q = 2$. ¿Qué es $AH^2$? cual famoso
        longitud de \cref{ex:g8:pythagoras:diagonal} tiene su
        ¿Se acaba de construir la figura de regla y compás?
  \item Describe la receta que construye un \omterm{def:g6:lines:objects}{segmento} de longitud.
        $\sqrt n$ para cualquier número entero $n \geq 1$ y diga cuál
        elección de las construcciones $p$ y $q$ $\sqrt 5$.
  \item Sea $O$ el centro del semicírculo. Explica por qué
        $AH$ nunca puede exceder el \omterm{def:g3:shapes:shapes}{radio} $\frac{p + q}{2}$, y
        para cuya posición de $H$ los dos son iguales.
  \item Deducir la desigualdad: para todos los números positivos $p$ y
        $q$,
        \[
        p \times q \leq \left(\frac{p + q}{2}\right)^2,
        \]
        con igualdad exactamente cuando $p = q$. Luego repréndelo con
        sin geometría en absoluto: expandir
        $\left(\frac{p+q}{2}\right)^2 - p q$ y reconocer un
        identidad notable (\cref{pb:g8:equations:1}).
\end{enumerate}
\end{problem}
