\chapter{Cálculo literal y ecuaciones.}\label{ch:g8:equations}

El séptimo grado introdujo letras y expresiones simples reducidas.
(\cref{ch:g7:literal}). Este capítulo añade las dos habilidades que hacen
álgebra poderosa: \emph{en expansión} \omterm{def:g2:mult:def}{productos} de corchetes, y
\emph{solving \omterm{def:g8:equations:def}{ecuaciones}} --- encontrando el número escondido detrás del
carta.

\section{En expansión}

\begin{theorem}[Distributividad simple y doble]\label{thm:g8:equations:expand}
Para todos los números:
\[
k(a + b) = ka + kb,
\qquad
(a + b)(c + d) = ac + ad + bc + bd .
\]
\end{theorem}

\begin{proof}
La primera regla es \cref{thm:g7:priorities:distributivity} (signos
incluido: $k$ y los términos pueden ser negativos,
\cref{ch:g8:negprod}). Para el segundo, aplica la primera regla dos veces,
tratando $(c + d)$ como un solo número:
\[
(a + b)(c + d) = a(c + d) + b(c + d) = ac + ad + bc + bd .
\qedhere
\]
\end{proof}

\begin{example}\label{ex:g8:equations:expand}
Tenga en cuenta las señales en cada término:
\begin{align*}
3(2x - 5) &= 6x - 15, \\
-2(4x - 7) &= -8x + 14
&& \text{(a minus in front flips both signs)}, \\
(x + 3)(x + 5) &= x^2 + 5x + 3x + 15 = x^2 + 8x + 15, \\
(2x - 1)(x - 4) &= 2x^2 - 8x - x + 4 = 2x^2 - 9x + 4 .
\end{align*}
\end{example}

\begin{method}[Ampliar y reducir]\label{met:g8:equations:reduce}
\begin{enumerate}
  \item Ampliar cada \omterm{def:g2:mult:def}{producto}, escribiendo cada término con su signo;
  \item recopila términos semejantes ($x^2$ con $x^2$, $x$ con $x$, números
        con números);
  \item verifique con un valor: sustituya, por ejemplo,\ $x = 2$ en el original
        y en el resultado, resultados iguales detectan la mayoría de los errores.
\end{enumerate}
\end{method}

\section{Resolver ecuaciones}

\begin{definition}[Ecuación]\label{def:g8:equations:def}
Un \emph{ecuación}\index{ecuación} es una igualdad con una incógnita
número, escrito $x$ (o cualquier letra). \emph{Resolviendo} significa encontrar
todos los valores de $x$ haciendo que la igualdad sea verdadera --- el
\emph{soluciones}.
\end{definition}

\begin{theorem}[Reglas de equilibrio]\label{thm:g8:equations:balance}
Un \omterm{def:g8:equations:def}{ecuación} mantiene exactamente las mismas soluciones cuando:
\begin{enumerate}
  \item se suma (o se resta) el mismo número en ambos lados;
  \item ambos lados se multiplican (o dividen) por el mismo
        \emph{distinto de cero} número.
\end{enumerate}
\end{theorem}
\admitted

\begin{method}[Resolver una ecuación de primer grado]\label{met:g8:equations:solve}
\begin{enumerate}
  \item Expanda y reduzca cada lado si es necesario;
  \item suma o resta en ambos lados para reunir los términos $x$ en uno
        lado, los números simples en el otro;
  \item reduce, luego divide ambos lados por el coeficiente de $x$;
  \item \emph{controlar} sustituyendo el valor encontrado en el original
        \omterm{def:g8:equations:def}{ecuación}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:equations:solve}
Resuelva $7x - 5 = 4x + 16$:
\begin{align*}
7x - 5 &= 4x + 16 \\
7x - 4x &= 16 + 5
&& \text{(add $5$, subtract $4x$, on both sides)} \\
3x &= 21 \\
x &= 7 && \text{(divide both sides by 3).}
\end{align*}
Verificar: lado izquierdo $7 \times 7 - 5 = 44$; lado derecho
$4 \times 7 + 16 = 44$. La solución es $7$.
\end{example}

\begin{example}[Con corchetes y negativos]\label{ex:g8:equations:brackets}
Resuelva $3(x - 4) = 5 - 2(x + 1)$:
\begin{align*}
3x - 12 &= 5 - 2x - 2 && \text{(expand both sides)} \\
3x - 12 &= 3 - 2x && \text{(reduce)} \\
5x &= 15 && \text{(add $2x + 12$ to both sides)} \\
x &= 3 .
\end{align*}
Verifique: $3(3 - 4) = -3$ y $5 - 2(3+1) = 5 - 8 = -3$. \quad Solución:
$3$.
\end{example}

\begin{method}[Resolver un problema con una ecuación]\label{met:g8:equations:problem}
\begin{enumerate}
  \item Elija la incógnita: ``que $x$ sea \dots'' (la cantidad solicitada
        para, o uno relacionado más simple);
  \item traduce la historia a \omterm{def:g8:equations:def}{ecuación};
  \item solucionarlo;
  \item interpretar: responder la pregunta original en una oración, con
        unidades y compare la respuesta con la historia.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:equations:problem}
Lucas es $4$ años mayor que su hermana; juntas sus edades suman
$26$. Sea $x$ la edad de la hermana; entonces Lucas es $x + 4$, y
\[
x + (x + 4) = 26
\ \Longrightarrow\
2x + 4 = 26
\ \Longrightarrow\
2x = 22
\ \Longrightarrow\
x = 11 .
\]
La hermana es $11$, Lucas es $15$. Verificar: $11 + 15 = 26$ y
$15 - 11 = 4$.
\end{example}

\section{Orden y desigualdades}

\begin{proposition}[Operaciones y orden]\label{prop:g8:equations:order}
Sumar el mismo número a ambos lados de una desigualdad la preserva;
multiplicar ambos lados por un número \emph{positivo} lo conserva;
multiplicando por un número \emph{negativo} \emph{invierte} se:
\[
2 < 5 \ \text{ but }\ (-1) \times 2 > (-1) \times 5 .
\]
\end{proposition}
\admitted

\begin{example}\label{ex:g8:equations:inequality}
Resuelva $5 - 2x \leq 11$: reste $5$ en ambos lados, $-2x \leq 6$;
dividir por $-2$ \emph{y revertir}: $x \geq -3$. todos los numeros
mayor o igual a $-3$ son soluciones --- un \omterm{def:g8:equations:def}{ecuación} generalmente
tiene una solución, una desigualdad tiene una media línea completa de ellas.
\end{example}

\section{Problema: identidades notables y la escalera de lo extraño
numeros}

\begin{problem}[{Problema de fin de semana --- $(a+b)^2$, $(a-b)^2$,
$(a+b)(a-b)$, y la suma $1 + 3 + 5 + \dots + (2n-1) = n^2$}]
\label{pb:g8:equations:1}
Tres expansiones surgen con tanta frecuencia que el álgebra las reconoce por
corazón: se llaman \emph{identidades notables}. esto
El problema los deriva de la doble distributividad.
(\cref{thm:g8:equations:expand}), los convierte en un cálculo-mental
superpotencia, los usa para demostrar un hermoso hecho sobre \omterm{def:g3:numbers:evenodd}{números impares},
y termina resolviendo \omterm{def:g8:equations:def}{ecuaciones} de un nuevo tipo: \omterm{def:g8:equations:def}{ecuaciones} con un
cuadrado en ellos.

\medskip
\noindent\textbf{Parte I --- The three identities.}
\begin{enumerate}
  \item Expandir $(a + b)^2 = (a + b)(a + b)$ y reducir, para probar
        \[
        (a + b)^2 = a^2 + 2ab + b^2 .
        \]
  \item Demuestren igualmente los dos compañeros:
        \[
        (a - b)^2 = a^2 - 2ab + b^2,
        \qquad
        (a + b)(a - b) = a^2 - b^2 .
        \]
  \item Dibuja un cuadrado de lado $a + b$, divide cada lado en $a$
        y $b$, y corta el cuadrado en cuatro pedazos. cual pieza
        de la primera identidad hace cada parte de la imagen
        representar?
  \item Aritmética mental: utilizar las identidades para calcular, sin
        planteando cualquier multiplicación,
        \[
        101^2, \qquad 99^2, \qquad 49 \times 51 .
        \]
  \item Una trampa clásica: Zoe escribe ``$(a + b)^2 = a^2 + b^2$''.
        Pruebe su fórmula con $a = 3$, $b = 4$ y explique en el
        Imagen de la pregunta 3 exactamente en qué piezas está su fórmula.
        olvida.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The staircase of \omterm{def:g3:numbers:evenodd}{números impares}.}
\begin{enumerate}[resume]
  \item Calcular $1$, luego $1 + 3$, luego $1 + 3 + 5$, luego
        $1 + 3 + 5 + 7$, luego $1 + 3 + 5 + 7 + 9$. ¿Qué haces?
        notar sobre los resultados?
  \item Explique por qué el $n$-ésimo \omterm{def:g3:numbers:evenodd}{número impar} es $2n - 1$ (verifique:
        $n = 1, 2, 3$ da $1, 3, 5$).
  \item Utilice una identidad de la Parte I para expandir y reducir
        $(n + 1)^2 - n^2$. ¿Qué tiene que ver el resultado?
        pregunta 7?
  \item La pregunta 8 dice: pasar de un cuadrado de lado $n$ a un
        cuadrado de lado $n + 1$, se agrega exactamente el siguiente \omterm{def:g3:numbers:evenodd}{impar
        numero} de celdas unitarias --- un borde en forma de L a lo largo de dos
        lados y la esquina. A partir de $1 = 1^2$ y creciendo
        el cuadrado una capa a la vez, explica por qué
        \[
        1 + 3 + 5 + \dots + (2n - 1) = n^2 :
        \]
        el \omterm{def:g1:addition:def}{suma} del primer $n$ \omterm{def:g3:numbers:evenodd}{números impares} es el $n$-ésimo cuadrado.
  \item Deducir, sin sumar uno por uno: el valor de
        $1 + 3 + 5 + \dots + 99$; y cuantos \omterm{def:g3:numbers:evenodd}{números impares}, empezando
        de $1$, sume exactamente $400$.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- \omterm{def:g8:equations:def}{Ecuaciones} with a square.}
\begin{enumerate}[resume]
  \item Reconocer una identidad notable en $x^2 + 6x + 9$, y
        resuelve el \omterm{def:g8:equations:def}{ecuación} $x^2 + 6x + 9 = 0$.
  \item Explique, utilizando la regla de signo y distancia de
        \cref{thm:g8:negprod:rules}, por qué un \omterm{def:g2:mult:def}{producto} de dos números
        puede ser \omterm{def:g1:counting:zero}{cero} \emph{solo si} uno de los dos factores es \omterm{def:g1:counting:zero}{cero}.
        Utilice esto para resolver $(x - 2)(x + 5) = 0$.
  \item Resuelve $x^2 = 49$ moviendo todo hacia un lado y
        factorizando $x^2 - 49$ con la tercera identidad. cuantos
        Soluciones tiene este \omterm{def:g8:equations:def}{ecuación} --- y ¿cómo funciona?
        ¿Difieren de todos los \omterm{def:g8:equations:def}{ecuación} resueltos en este capítulo hasta ahora?
  \item En \cref{exo:g8:equations:3} verificaste que $x = 3$ es
        una solución de $x^2 = 6x - 9$. Demuestre que es el
        \emph{solo} uno: dejar todo a un lado, reconocer
        una identidad y concluir.
  \item Final: calcula $2026^2 - 2025^2$ en tu cabeza y
        explica con una identidad por qué el \omterm{ex:g1:subtraction:difference}{diferencia} de dos
        cuadrados consecutivos es siempre el \omterm{def:g1:addition:def}{suma} de los dos números
        --- que es la pregunta 8 de nuevo.
\end{enumerate}
\end{problem}
