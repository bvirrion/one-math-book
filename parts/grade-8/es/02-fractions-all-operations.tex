\chapter{Fracciones: las cuatro operaciones}\label{ch:g8:fractions}

Este capítulo completa la aritmética de \omterm{def:g6:fractions:def}{fracciones}: suma y
resta con \emph{cualquier} \omterm{def:g4:fractions:def}{denominadores}, multiplicación y ---
el recién llegado --- división, que resulta ser una multiplicación en
disfrazarse. Las señales de \cref{ch:g8:negprod} te acompañan.

\section{Sumar con cualquier denominador}

\begin{method}[denominador común]\label{met:g8:fractions:add}
Para sumar o restar $\frac ab$ y $\frac cd$:
\begin{enumerate}
  \item encuentra un \omterm{def:g5:division:multiple}{múltiple} común de $b$ y $d$ (el \omterm{def:g2:mult:def}{producto}
        $b \times d$ siempre funciona; uno más pequeño ahorra esfuerzo);
  \item reescribe ambos \omterm{def:g6:fractions:def}{fracciones} con ese \omterm{def:g4:fractions:def}{denominador};
  \item suma o resta el \omterm{def:g4:fractions:def}{numeradores}; simplificar.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:fractions:add}
Calcule $\dfrac{5}{6} + \dfrac{3}{4}$. Un \omterm{def:g5:division:multiple}{múltiple} común de $6$ y
$4$ es $12$:
\[
\frac{5}{6} = \frac{10}{12},
\qquad
\frac{3}{4} = \frac{9}{12},
\qquad
\frac{5}{6} + \frac{3}{4} = \frac{10 + 9}{12} = \frac{19}{12}.
\]
Con carteles: $\dfrac{2}{3} - \dfrac{7}{5} = \dfrac{10}{15} -
\dfrac{21}{15} = -\dfrac{11}{15}$.
\end{example}

\section{Multiplicar fracciones}

\begin{theorem}[Producto de fracciones]\label{thm:g8:fractions:mult}
\[
\frac{a}{b} \times \frac{c}{d} = \frac{a \times c}{b \times d} .
\]
\end{theorem}

\begin{proof}[Por qué, en una imagen]
Tome $\frac34$ de $\frac25$ de un cuadrado: corte el cuadrado en $5$
tiras verticales y mantener $2$; corte horizontalmente en $4$ y mantenga $3$
filas de lo que quedaba. La parte conservada es una cuadrícula de $3 \times 2$ pequeña.
celdas de $4 \times 5$: \omterm{def:g6:fractions:def}{fracción} $\frac{6}{20}$.
\end{proof}

  \fill[omDef!18] (0,0) rectangle (2,4);
  \fill[omThm!35] (0,0) rectangle (2,3);
  \draw[thick] (0,0) rectangle (5,4);
  \draw[omDef, thick] (2,0) -- (2,4);
  \draw[omThm, thick] (0,3) -- (5,3);
  \node[below, font=\small, omDef] at (1,0) {$\frac25$ kept};
  \node[right, font=\small, omThm] at (5.05,1.5) {$\frac34$ of it};
\end{tikzpicture}

{\small $\frac34 \times \frac25$: la región doblemente sombreada es
$3 \times 2 = 6$ celdas de $4 \times 5 = 20$, es decir\
$\frac{6}{20} = \frac{3}{10}$ --- \omterm{def:g4:fractions:def}{numeradores} multiplicado, \omterm{def:g4:fractions:def}{denominadores}
multiplicado.}
\end{omfigure}

\begin{example}\label{ex:g8:fractions:mult}
Simplifica \emph{antes} multiplicando, tachando factores comunes:
\[
\frac{7}{12} \times \frac{9}{14}
= \frac{7 \times 9}{12 \times 14}
= \frac{7 \times 3 \times 3}{3 \times 4 \times 2 \times 7}
= \frac{3}{8}
\]
(el factor $7$ y un factor $3$ se cancelan entre arriba y abajo).
Con señales, se aplican primero las reglas de \cref{thm:g8:negprod:rules}:
$\left(-\frac23\right) \times \left(-\frac{5}{4}\right) =
+\frac{10}{12} = \frac56$.
\end{example}

\section{Dividir fracciones}

\begin{definition}[inversa]\label{def:g8:fractions:inverse}
El \emph{inverso}\index{inverso} de un número distinto de cero $x$ es el
número que multiplicado por $x$ da $1$. lo inverso de
$\frac ab$ es $\frac ba$, ya que
$\frac ab \times \frac ba = \frac{ab}{ab} = 1$; la inversa de un
El número entero $n$ es $\frac1n$.
\end{definition}

\begin{theorem}[Dividir es multiplicar por el inverso]\label{thm:g8:fractions:div}
Para $c \neq 0$:
\[
\frac{a}{b} \div \frac{c}{d} = \frac{a}{b} \times \frac{d}{c} .
\]
\end{theorem}

\begin{proof}
Por definición, el \omterm{def:g3:division:remainder}{cociente} $q$ de $\frac ab$ por $\frac cd$ es el
número con $q \times \frac cd = \frac ab$. comprobar el candidato
$q = \frac ab \times \frac dc$:
\[
\left(\frac ab \times \frac dc\right) \times \frac cd
= \frac ab \times \left(\frac dc \times \frac cd\right)
= \frac ab \times 1 = \frac ab . \qedhere
\]
\end{proof}

\begin{example}\label{ex:g8:fractions:div}
\[
\frac{3}{5} \div \frac{7}{10}
= \frac{3}{5} \times \frac{10}{7}
= \frac{3 \times 10}{5 \times 7}
= \frac{30}{35} = \frac{6}{7}.
\]
Una prueba de cordura: dividir por $\frac{7}{10}$, un número menor que
$1$, debe dar un resultado \emph{más grande} que $\frac35$ --- y
$\frac67 > \frac35$ efectivamente ($\frac{30}{35} > \frac{21}{35}$).
\end{example}

\begin{example}[Barras de fracciones dentro de barras de fracciones]\label{ex:g8:fractions:complex}
Un "dos pisos" es simplemente una división escrita verticalmente:
\[
\frac{\ \frac{2}{3}\ }{\ \frac{5}{6}\ }
= \frac{2}{3} \div \frac{5}{6}
= \frac{2}{3} \times \frac{6}{5}
= \frac{12}{15} = \frac{4}{5}.
\]
Localice primero la barra \emph{principal} (la más larga), luego aplique
\cref{thm:g8:fractions:div}.
\end{example}

\begin{method}[Cálculos mixtos]\label{met:g8:fractions:mixed}
En una expresión que mezcla \omterm{def:g6:fractions:def}{fracciones} y las cuatro operaciones:
\begin{enumerate}
  \item aplica las prioridades habituales (\cref{ch:g7:priorities});
  \item convierte cada división en una multiplicación por \omterm{def:g8:fractions:inverse}{inverso};
  \item decide los signos primero (\cref{met:g8:negprod:compute}), luego
        multiplicar o encontrar \omterm{def:g4:fractions:def}{denominadores} común;
  \item simplifique lo antes posible.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:fractions:mixed}
\begin{align*}
\frac12 + \frac{2}{3} \times \frac{9}{4}
&= \frac12 + \frac{2 \times 9}{3 \times 4}
&& \text{(product first!)} \\
&= \frac12 + \frac{18}{12} = \frac12 + \frac32
= \frac{4}{2} = 2 .
\end{align*}
\end{example}

\section{Problema: fracciones egipcias}

\begin{problem}[{Problema de fin de semana --- cada fracción es una suma de
fracciones unitarias distintas}]
\label{pb:g8:fractions:1}
Un \emph{unit \omterm{def:g6:fractions:def}{fracción}} es un \omterm{def:g6:fractions:def}{fracción} con \omterm{def:g4:fractions:def}{numerador} $1$. el
escribas del antiguo Egipto --- el papiro de Rhind fue copiado alrededor
$1550$~BC --- escribió cada \omterm{def:g6:fractions:def}{fracción} como \omterm{def:g1:addition:def}{suma} de \emph{distinto}
unidad \omterm{def:g6:fractions:def}{fracciones}: nunca $\frac27 = \frac17 + \frac17$, pero
$\frac27 = \frac14 + \frac{1}{28}$. Un \omterm{def:g1:addition:def}{suma} de este tipo se denomina
\emph{escritura egipcia} del \omterm{def:g6:fractions:def}{fracción}. Este problema desarrolla la
Técnicas de suma y comparación de este capítulo.
(\cref{met:g8:fractions:add}) en un método --- publicado por
Fibonacci en $1202$ --- que produce una escritura egipcia de cualquier
\omterm{def:g6:fractions:def}{fracción} entre $0$ y $1$, y termina con un celebrado
rompecabezas de herencia.

\medskip
\noindent\textbf{Parte I --- Unit \omterm{def:g6:fractions:def}{fracciones} and the splitting
identity.}
\begin{enumerate}
  \item Calcular $\frac12 + \frac13$ y $\frac12 + \frac14$, y
        deducir escritos egipcios de $\frac56$ y $\frac34$.
  \item Verifique la entrada de los escribas para $\frac27$: muestre que
        $\frac14 + \frac{1}{28} = \frac27$.
  \item Compruebe que $\frac13 + \frac16 = \frac12$ y que
        $\frac14 + \frac{1}{12} = \frac13$. Luego prueba la
        \emph{identidad dividida}: para cada número entero
        $n \geq 1$,
        \[
        \frac{1}{n} = \frac{1}{n+1} + \frac{1}{n(n+1)} .
        \]
  \item Deduce una escritura egipcia del propio $1$, como \omterm{def:g1:addition:def}{suma} de
        tres unidades distintas \omterm{def:g6:fractions:def}{fracciones}.
  \item Divida el último término de $\frac56 = \frac12 + \frac13$ en
        obtener una escritura egipcia \emph{segundo} de $\frac56$, con
        tres unidades \omterm{def:g6:fractions:def}{fracciones}. Explique por qué ningún \omterm{def:g6:fractions:def}{fracción} tiene un nombre único.
        Escritura egipcia.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Fibonacci's greedy method.}
El método es: restar la unidad \emph{más grande} \omterm{def:g6:fractions:def}{fracción} que
encaja y repite con lo que queda.
\begin{enumerate}[resume]
  \item Usando \omterm{def:g4:fractions:def}{denominadores} común, demuestre que
        $\frac14 < \frac27 < \frac13$. ¿Por qué esto prueba que
        $\frac14$ es la unidad más grande \omterm{def:g6:fractions:def}{fracción} más pequeña que
        $\frac27$?
  \item Calcular $\frac27 - \frac14$. Cual escrito de $\frac27$
        Qué produce el método codicioso?
  \item Ahora toma $\frac57$. Demuestre que la unidad más grande \omterm{def:g6:fractions:def}{fracción}
        menor que $\frac57$ es $\frac12$ y calcula
        $\frac57 - \frac12$.
  \item Demuestre que la unidad más grande \omterm{def:g6:fractions:def}{fracción} más pequeña que
        $\frac{3}{14}$ es $\frac15$ (compárese $\frac{3}{14}$ con
        $\frac14$ y con $\frac15$), calcular
        $\frac{3}{14} - \frac15$, y concluye:
        \[
        \frac57 = \frac12 + \frac15 + \frac{1}{70} .
        \]
        Verifique este escrito directamente con el común \omterm{def:g4:fractions:def}{denominador}
        $70$.
  \item Ejecute el método codicioso en $\frac45$, verificando en cada paso
        que su unidad \omterm{def:g6:fractions:def}{fracción} es la más grande que cabe, y
        verificar el escrito final.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Why the method always stops, and the
eighteenth camel.}
\begin{enumerate}[resume]
  \item En las preguntas 7--10 se aplicó el método a $\frac27$,
        $\frac57$ y $\frac45$. Para cada uno, indique el \omterm{def:g4:fractions:def}{numeradores} de
        el sucesivo \omterm{def:g3:division:remainder}{restos} (antes de cualquier simplificación).
        ¿Qué observas?
  \item Para un \omterm{def:g6:fractions:def}{fracción} $\frac ab$ y un número entero $n$, mostrar
        eso
        \[
        \frac{a}{b} - \frac{1}{n}
        = \frac{a \times n - b}{b \times n} .
        \]
  \item Supongamos que el método codicioso resta $\frac1n$ de
        $\frac ab$: entonces $\frac1n$ encaja pero $\frac{1}{n-1}$ es
        ya demasiado grande, $\frac{1}{n-1} > \frac ab$. poniendo ambos
        sobre el común \omterm{def:g4:fractions:def}{denominador} $b \times (n - 1)$, deducir
        que $b > a \times n - a$, de ahí que el nuevo \omterm{def:g4:fractions:def}{numerador}
        satisface
        \[
        a \times n - b < a .
        \]
        Explique por qué esto prueba que el método siempre se detiene.
  \item Un viejo cuento: un padre muere dejando $17$ camellos, dispuestos
        $\frac12$ del rebaño a su hijo mayor, $\frac13$ a
        el segundo, $\frac19$ al tercero --- y ninguno de los
        Las acciones son un número entero de camellos. Un vecino presta el
        familia un decimoctavo camello; los niños toman $9$, $6$ y
        $2$ camellos y se devuelve el camello prestado. calcular
        $\frac12 + \frac13 + \frac19$ y explica el truco: por qué
        ¿Podría dividirse la manada en camellos enteros y por qué no
        ¿El niño recibe menos de lo prometido en el testamento?
  \item Reparar el testamento: proponer tres unidades \emph{distinto}
        \omterm{def:g6:fractions:def}{fracciones} que suman exactamente $1$, y dan el
        tamaño de rebaño más pequeño para el cual las tres partes están enteras
        cantidad de camellos.
\end{enumerate}
\end{problem}
