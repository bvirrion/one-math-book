\chapter{Potestades}\label{ch:g8:powers}

Dobla una hoja de papel en \omterm{def:g3:division:half}{medio} $10$ veces (¡si puedes!): su grosor
se duplica cada vez, y $10$ se duplica y lo multiplica por
$2^{10} = 1024$. Las potencias son la abreviatura de la multiplicación repetida;
este capítulo establece la notación y sus reglas, con un papel especial
para los poderes de $10$.

\section{Definición}

\begin{definition}[poder]\label{def:g8:powers:def}
Para un número $a$ y un número entero $n \geq 1$:
\[
a^n = \underbrace{a \times a \times \dots \times a}_{n
\text{ factors}},
\]
lea ``$a$ al (poder) $n$''; $a$ es el \emph{base}, $n$ el
\emph{exponente}\index{exponente}. Nombres especiales: $a^2$ es ``$a$
al cuadrado'', $a^3$ ``$a$ al cubo''. Por convención:
\[
a^1 = a,
\qquad
a^0 = 1 \ (a \neq 0),
\qquad
a^{-n} = \frac{1}{a^n}.
\]
\end{definition}

\begin{example}\label{ex:g8:powers:def}
$3^4 = 3 \times 3 \times 3 \times 3 = 81$; \quad
$10^3 = 1000$; \quad$5^{-2} = \frac{1}{25} = 0.04$; \quad
$(-2)^3 = -8$ y $(-2)^4 = +16$
(\cref{ex:g8:negprod:powers}). Ojo: $a^n$ es \emph{no}
$a \times n$: $2^5 = 32$, no $10$.
\end{example}

\section{Las reglas de los exponentes.}

\begin{theorem}[Reglas de exponentes]\label{thm:g8:powers:rules}
Para una base distinta de cero $a$ y un número entero \omterm{def:g8:powers:def}{exponentes} $m$, $n$:
\[
a^m \times a^n = a^{m+n},
\qquad
\frac{a^m}{a^n} = a^{m-n},
\qquad
\left(a^m\right)^n = a^{m \times n},
\qquad
(ab)^n = a^n b^n .
\]
\end{theorem}

\begin{proof}[Prueba por conteo de factores]
$a^m \times a^n$ alinea los factores $m$ $a$ seguidos de $n$ más:
$m + n$ en total. $\left(a^m\right)^n$ repite un bloque de factores $m$
$n$ veces: $mn$ factores. $(ab)^n$ contiene las letras $n$ $a$ y $n$
letras $b$, que se pueden reagrupar. La regla \omterm{def:g3:division:remainder}{cociente} proviene de
cancelando $n$ de los factores $m$; con las convenciones
$a^0 = 1$ y $a^{-n} = \frac{1}{a^n}$, sigue siendo cierto incluso cuando
$n \geq m$.
\end{proof}

\begin{example}\label{ex:g8:powers:rules}
\[
2^3 \times 2^5 = 2^8 = 256,
\qquad
\frac{7^6}{7^4} = 7^2 = 49,
\qquad
\left(5^2\right)^3 = 5^6,
\qquad
2^4 \times 5^4 = 10^4 .
\]
Una trampa: las reglas se aplican a un \emph{base común} (o un común
\omterm{def:g8:powers:def}{exponente}, para el último). Ninguna regla simplifica $2^3 \times 5^2$ ---
simplemente calcule: $8 \times 25 = 200$.
\end{example}

\section{potencias de diez}

\begin{proposition}[Potencias de diez]\label{prop:g8:powers:ten}
Para $n \geq 1$: $10^n = 1\underbrace{0\dots0}_{n}$ y
$10^{-n} = 0.\underbrace{0\dots0}_{n-1}1$. Las reglas de \omterm{def:g8:powers:def}{exponentes}
léase: multiplicar potencias de diez suma el \omterm{def:g8:powers:def}{exponentes}.
\end{proposition}

\begin{example}\label{ex:g8:powers:ten}
$10^4 \times 10^3 = 10^7$; \quad
$\dfrac{10^2}{10^5} = 10^{-3} = 0.001$. Grandes y pequeñas cantidades
volverse legible:
\[
\text{one billion} = 10^9,
\qquad
\text{one millionth} = 10^{-6}.
\]
Combinado con decimales: $3.2 \times 10^5 = 320\,000$ y
$4.7 \times 10^{-3} = 0.0047$ (cambie el \omterm{def:g4:decimals:point}{coma decimal},
\cref{prop:g6:decimals:shift}). El uso sistemático de este escrito ---
\emph{notación científica} --- está desarrollado en
\cref{ch:g9:fractions}.
\end{example}

\begin{example}[Órdenes de magnitud]\label{ex:g8:powers:magnitude}
La luz viaja aproximadamente $3 \times 10^8$ m/s; un año tiene aproximadamente
$3.2 \times 10^7$ segundos. Por lo tanto, un año luz es aproximadamente
\[
3 \times 3.2 \times 10^{8+7} \approx 10 \times 10^{15} = 10^{16}
\text{ m}
\]
--- diez millones de billones de metros. Las potencias de diez hacen astronómico
los cálculos caben en una línea.
\end{example}

\begin{method}[Simplificando una expresión con potencias]\label{met:g8:powers:simplify}
\begin{enumerate}
  \item Agrupa los factores base por base;
  \item aplica las reglas \omterm{def:g8:powers:def}{exponente} dentro de cada base;
  \item calcula las potencias pequeñas restantes o deja la respuesta como
        potencia si es grande.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:powers:simplify}
\[
\frac{3^5 \times 3^{-2} \times 4^2}{3^2}
= \frac{3^{5 + (-2)}}{3^2} \times 16
= 3^{3 - 2} \times 16
= 3 \times 16 = 48 .
\]
\end{example}

\section{Ceremonias}

\[
2^6, \qquad 3^3, \qquad 10^5, \qquad 1^{100}, \qquad 0.1^2, \qquad
6^0 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:2}
Calcular:
\[
(-3)^2, \qquad -3^2, \qquad (-1)^{15}, \qquad (-5)^3, \qquad
\left(\tfrac{2}{3}\right)^2 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:3}
Escribe como una sola potencia:
\[
7^4 \times 7^5, \qquad
\frac{2^9}{2^3}, \qquad
\left(10^3\right)^4, \qquad
5^6 \times 5^{-2}, \qquad
3^4 \times 7^4 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:4}
Escriba como \omterm{def:g6:decimals:places}{numero decimal}: $10^{-2}$; \quad$4 \times 10^3$; \quad
$2.5 \times 10^{-4}$; \quad$10^0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:5}
Escribe con una potencia de diez: cien mil; un décimo; diez
mil millones; $0.000\,001$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:6}
Simplifica y luego calcula:
\[
\frac{10^7 \times 10^{-3}}{10^2},
\qquad
\frac{2^5 \times 2^4}{2^6},
\qquad
\left(2^2\right)^3 \times 2^{-4} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:powers:7}
¿Verdadero o falso? Corrija los falsos.
\[
2^3 \times 2^4 = 2^{12}; \qquad
5^2 + 5^3 = 5^5; \qquad
(3^2)^4 = 3^8; \qquad
10^3 \times 10^3 = 100^3 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:powers:8}
Se extiende un rumor: el día 1, tres personas lo saben; cada día, cada
La persona que sabe se lo cuenta a tres personas nuevas. Escribe con una potencia el número.
de \emph{nuevo} personas informadas el día $4$, y calcular cuántas personas
Conozca el rumor al final del día 4 (incluidos los tres originales).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:powers:9}
Una hoja de papel tiene un espesor de $0.1$ mm, es decir\ $10^{-4}$ m. Doblándolo
duplica su espesor cada vez.
\begin{enumerate}
  \item Exprese el espesor después de que $10$ se pliegue como \omterm{def:g2:mult:def}{producto}, y
        calcularlo en centímetros ($2^{10} = 1024$).
  \item Después de los pliegues $42$, el grosor sería
        $2^{42} \times 10^{-4}$ m, con $2^{42} \approx 4.4 \times
        10^{12}$. Demuestre que esto excede la distancia Tierra-Luna,
        aproximadamente $3.8 \times 10^8$ m.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:powers:10}
Ordene de menor a mayor, sin calculadora:
\[
2^{10}, \qquad 10^3, \qquad 3^6, \qquad 5^4 .
\]
(Calcule cada uno; $2^{10}$ y $10^3$ son vecinos famosos).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:powers:11}
¿Cuál es más grande, $2^{100}$ o $10^{30}$? uso
$2^{10} = 1024 > 10^3$ para comparar
$2^{100} = \left(2^{10}\right)^{10}$ con
$\left(10^3\right)^{10}$.
\end{exercise}

\section{Problema: El tablero de ajedrez y las potencias de dos}

\begin{problem}[{Problema de fin de semana --- el \omterm{def:g1:addition:def}{suma} geométrico
$1 + 2 + 4 + \dots + 2^{n-1} = 2^n - 1$, de una famosa leyenda a
números binarios}]
\label{pb:g8:powers:1}
La leyenda: como recompensa por inventar el ajedrez, el sabio Sissa pidió
su rey por un grano de trigo en la primera casilla del tablero,
dos en el segundo, cuatro en el tercero --- duplicando de cuadrado a
cuadrado, hasta el sexagésimo cuarto. El rey se rió de tanta modestia.
Este problema calcula lo que prometió el rey, usando el \omterm{def:g8:powers:def}{exponente}
reglas de \cref{thm:g8:powers:rules}, y termina donde la historia
lleva en secreto: los números binarios dentro de cada computadora.

\medskip
\noindent\textbf{Parte I --- The doubling trick.}
Para $n \geq 1$, sea $S_n$ el número total de granos en el
primeros $n$ cuadrados.
\begin{enumerate}
  \item Expresar el número de granos del cuadrado $k$ como potencia de
        $2$. ¿Qué potencia se encuentra en el cuadrado $64$?
  \item Calcule $S_1$, $S_2$, $S_3$, $S_4$ y $S_5$ y compare
        cada uno con una potencia cercana de $2$. Conjetura una fórmula para
        $S_n$.
  \item El \emph{truco de duplicación}: escriba el \omterm{def:g1:addition:def}{sumas} $S_n$ y
        $2 \times S_n$ uno debajo del otro, resta y prueba
        tu conjetura:
        \[
        S_n = 1 + 2 + 4 + \dots + 2^{n-1} = 2^n - 1 .
        \]
  \item ¿Cuántos granos prometió el rey en total? expresar el
        responde con una potencia de $2$, y completa el clásico
        observación: ``todo el tablero contiene un grano menos que
        un solo cuadrado sexagésimo quinto lo haría''.
  \item Demuestre que el segundo \omterm{def:g3:division:half}{medio} del tablero (cuadrados $33$ a
        $64$) contiene \emph{exactamente} $2^{32}$ veces más granos
        como el primer \omterm{def:g3:division:half}{medio}.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- How big is $2^{64}$?}
La comparación $2^{10} = 1024 > 10^3$ de
\cref{exo:g8:powers:11} es la clave de todas las estimaciones siguientes.
\begin{enumerate}[resume]
  \item Demuestra que
        $2^{64} = 2^4 \times \left(2^{10}\right)^6 > 1.6 \times
        10^{19}$.
  \item Un grano de trigo pesa aproximadamente $0.05$ g, es decir\
        $5 \times 10^{-2}$ g. Demuestre que el trigo prometido pesa
        más de $8 \times 10^{17}$ g, y convertir esto en
        toneladas ($1$ tonelada $= 10^6$ g).
  \item El mundo entero cosecha actualmente alrededor de $8 \times 10^8$
        toneladas de trigo al año. Al menos cuantos años de mundo
        cosecha prometió el rey?
  \item Encuentre el número entero más pequeño $n$ tal que
        $2^n > 10^6$ --- es decir, ¿cuántas duplicaciones se necesitan para
        pasar el millón. (Calcule $2^{19}$ y $2^{20}$ exactamente,
        utilizando $2^{10} = 1024$.)
  \item Un embaucador te ofrece el salario de un mes: $1$ centavo al día
        $1$, luego duplica el salario del día anterior cada día. en cual
        día que el \emph{a diario} paga solo por primera vez excede uno
        millones de euros ($10^8$ céntimos)? (Calcule $2^{26}$ y
        $2^{27}$ exactamente.)
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Binary weights.}
Un comerciante posee cinco pesos: $1$, $2$, $4$, $8$ y $16$ gramos,
uno de cada uno. Coloca algunos de ellos en el platillo de una balanza para
pesar los productos en el otro plato.
\begin{enumerate}[resume]
  \item ¿Qué pesas coloca para pesar $21$ g? pesar
        $27$g?
  \item Explique por qué se utiliza cualquier objetivo de $16$ go más \emph{debe}
        el peso $16$ g y por qué cualquier objetivo de $15$ g o menos
        debe \emph{no} usarlo. (La pregunta 3 le dice cuál es el
        pesos $1, 2, 4, 8$ puede alcanzar como máximo.) Explique por qué el
        El mismo razonamiento se repite con el siguiente peso más grande, en
        cada etapa.
  \item Deduzca que cada objetivo completo desde $1$ hasta $31$ g puede ser
        pesado, y \emph{exactamente de una manera}: cada número
        entre $1$ y $31$ hay un \omterm{def:g1:addition:def}{suma} de potencias distintas de $2$ en
        una sola manera.
  \item El comerciante compra un sexto peso, de $32$ g. hasta que
        objetivo ¿puede pesar ahora? Escribe $45$ g como \omterm{def:g1:addition:def}{suma} de
        poderes distintos de $2$.
  \item En una computadora, un número de "64 bits" se almacena en $64$
        cuadrados, cada uno con un $0$ o un $1$ --- cuadrado $k$
        contribuyendo $2^{k-1}$ cuando tiene un $1$, como el
        granos de la leyenda. Utilizando la Parte I, explique por qué todo
        Los números que una máquina de este tipo puede almacenar se ejecutan exactamente desde $0$ hasta
        $2^{64} - 1$.
\end{enumerate}
\end{problem}
