\chapter{Puntos medios y paralelos}\label{ch:g8:midpoints}

Une los puntos medios de dos lados de un triángulo: el \omterm{def:g6:lines:objects}{segmento} que obtienes es
siempre \omterm{def:g6:lines:perp}{paralelo} hacia el tercer lado, y exactamente \omterm{def:g3:division:half}{medio} igual de largo. esto
El "teorema del punto medio" y su recíproco son la primera muestra de un gran
idea --- \omterm{def:g6:lines:perp}{paralelas} cortar longitudes proporcionalmente --- que florece en
Teorema de Tales en \cref{ch:g9:thales}.

\section{El teorema del punto medio}

\begin{theorem}[Teorema del punto medio]\label{thm:g8:midpoints:direct}
En un triángulo $ABC$, sea $I$ el punto medio de $[AB]$ y $J$ el
punto medio de $[AC]$. Entonces la línea $(IJ)$ es \omterm{def:g6:lines:perp}{paralelo} a $(BC)$, y
\[
IJ = \frac{BC}{2} .
\]
\end{theorem}

\begin{proof}[Idea de prueba]
Sea $K$ el \omterm{def:g7:central:def}{simétrico} de $I$ aproximadamente $J$ (media vuelta,
\cref{ch:g7:central}). El \omterm{def:g4:geometry:polygon}{cuadrilátero} $AICK$ tiene diagonales $[IK]$
y $[AC]$ cruce en su punto medio común $J$: es un
\omterm{def:g7:central:parallelogram}{paralelogramo}, por lo que $KC$ es \omterm{def:g6:lines:perp}{paralelo} a $AI$, es decir, \ a $(AB)$, y
$KC = AI = IB$. Entonces $IBCK$ tiene dos lados opuestos ($[IB]$ y
$[KC]$) \omterm{def:g6:lines:perp}{paralelo} e igual: es un \omterm{def:g7:central:parallelogram}{paralelogramo} también
(\cref{thm:g7:central:recognize}), entonces $(IK) \parallel (BC)$ ---
es decir $(IJ) \parallel (BC)$ --- y $BC = IK = 2\,IJ$.
\end{proof}

\begin{example}\label{ex:g8:midpoints:direct}
En un triángulo $ABC$ con $BC = 9$ cm, los puntos medios $I$ de $[AB]$ y
Se unen $J$ de $[AC]$. Sin ninguna medida:
$(IJ) \parallel (BC)$ y $IJ = \frac92 = 4.5$ cm. El pequeño triángulo
$AIJ$ es una copia de tamaño medio de $ABC$ --- su \omterm{def:g6:measure:perimeter}{perímetro} es \omterm{def:g3:division:half}{medio} el
\omterm{def:g6:measure:perimeter}{perímetro} de $ABC$ también.
\end{example}

\begin{theorem}[Conversar]\label{thm:g8:midpoints:converse}
En un triángulo $ABC$, la línea que pasa por el punto medio $I$ de $[AB]$ y
\emph{\omterm{def:g6:lines:perp}{paralelo}} a $(BC)$ cruza el lado $[AC]$ en su punto medio.
\end{theorem}
\admitted

\begin{example}\label{ex:g8:midpoints:converse}
Un camino cruza un campo triangular, comenzando en el medio de un borde.
y ejecutando \omterm{def:g6:lines:perp}{paralelo} hasta el borde opuesto. Lo contrario garantiza
que el camino sale exactamente en el medio del otro borde --- no
Es necesario medir en el otro lado.
\end{example}

\begin{method}[Elección entre teorema y recíproco]\label{met:g8:midpoints:choose}
\begin{enumerate}
  \item \emph{Dos puntos medios conocidos} $\Rightarrow$ utilizar el directo
        teorema: concluir paralelismo y media longitud.
  \item \emph{One midpoint and a \omterm{def:g6:lines:perp}{paralelo} known} $\Rightarrow$ uso
        lo contrario: concluir que el punto de cruce es un punto medio.
  \item Escribe qué triángulo, qué puntos medios y qué teorema
        use --- una oración cada uno.
\end{enumerate}
\end{method}

\section{Triángulos de tamaño medio}

\begin{proposition}[El triangulo del punto medio]\label{prop:g8:midpoints:medial}
Unir los tres puntos medios de los lados de un triángulo lo corta en
cuatro triángulos de igual tamaño \omterm{def:g6:measure:area}{áreas}, cada uno de los cuales es una copia de tamaño medio del original.
\end{proposition}

\begin{proof}
Cada lado del punto medio del triángulo es \omterm{def:g6:lines:perp}{paralelo} a un lado de $ABC$ y
\omterm{def:g3:division:half}{medio} siempre y cuando (\cref{thm:g8:midpoints:direct}, aplicado tres veces).
Los cuatro triángulos pequeños tienen lados de las mismas tres longitudes, por lo que
son copias idénticas; juntos forman un mosaico $ABC$, por lo que cada uno tiene un
\omterm{def:g3:division:half}{cuarto} de su \omterm{def:g6:measure:area}{área}.
\end{proof}

\begin{remark}[Hacia Cuentos]
El teorema del punto medio es el caso "un \omterm{def:g3:division:half}{medio}" de un hecho más general:
una línea \omterm{def:g6:lines:perp}{paralelo} a un lado de un triángulo corta los otros dos lados en
\emph{proporciones iguales} --- un tercio y un tercio, dos quintos y dos
quintos\,\dots\ Ese enunciado general es el teorema de Tales
(\cref{ch:g9:thales}); el caso del punto medio es el único necesario
año.
\end{remark}

\section{Problema: Las tres medianas y el centro de gravedad.}

\begin{problem}[{Problema de fin de semana --- las medianas de un triángulo se encuentran
en un solo punto, dos tercios del camino hacia abajo de cada uno de ellos}]
\label{pb:g8:midpoints:1}
Corta un triángulo de cartón rígido y se equilibrará.
perfectamente plano, sobre la punta de un lápiz colocada en un punto especial: el
\emph{centro de gravedad}\index{centro de gravedad} del triángulo.
Este problema encuentra ese punto. A \emph{mediana}\index{mediana} de un
triángulo es el \omterm{def:g6:lines:objects}{segmento} que une un \omterm{def:g5:solids:def}{vértice} al \emph{punto medio} de
el lado opuesto. Demostrarás --- con el teorema del punto medio
usado dos veces, de una manera inesperada --- que las tres medianas todas
pasar por un punto y localizarlo con precisión.

\medskip
\noindent\textbf{Parte I --- Warm-up.}
\begin{enumerate}
  \item Dibuja un triángulo grande $ABC$ (sin forma especial), construye
        los puntos medios $I$ de $[AB]$, $J$ de $[AC]$ y $K$ de
        $[BC]$ y dibuja las tres medianas $[AK]$, $[BJ]$, $[CI]$.
        ¿Qué observas?
  \item Demuestra que una mediana corta el triángulo en dos triángulos
        de igual \omterm{def:g6:measure:area}{áreas}. (Compare bases y alturas, y use el
        \omterm{def:g6:measure:area}{área} fórmula: base $\times$ altura $\div\ 2$.) Para un
        triangulo de \omterm{def:g6:measure:area}{área} $36$ cm$^2$, cuales son las dos piezas?
        \omterm{def:g6:measure:area}{áreas}?
  \item En el triángulo $ABC$, ¿qué significa
        \cref{thm:g8:midpoints:direct} dice sobre el \omterm{def:g6:lines:objects}{segmento}
        $[IJ]$?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Two medians meet two thirds of the
way.}

$M$ será el punto medio de $[GB]$ y $N$ el punto medio de $[GC]$.

  \item Aplicar el teorema del punto medio \emph{en el triángulo $GBC$}:
        ¿Qué dice sobre el \omterm{def:g6:lines:objects}{segmento} $[MN]$?
  \item Deduzca que $(IJ) \parallel (MN)$ y $IJ = MN$, y
        concluir con \cref{thm:g7:central:recognize} que $IJNM$
        es un \omterm{def:g7:central:parallelogram}{paralelogramo}.
  \item Explique por qué $I$ y $N$ se encuentran en la mediana $(CI)$,

        \[
        BG = 2 \times GJ
        \]
