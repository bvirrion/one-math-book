\chapter{Pirámides y Conos}\label{ch:g8:solids}

Después del \omterm{def:g7:areas:prism}{prismas} y \omterm{def:g7:areas:prism}{cilindros} de \cref{ch:g7:areas} --- \omterm{def:g5:solids:def}{sólidos} con
sección constante --- vienen los \omterm{def:g5:solids:def}{sólidos} con un punto: \omterm{def:g8:solids:def}{pirámides} y
\omterm{def:g8:solids:def}{conos}. Su fórmula \omterm{def:g6:measure:volume}{volumen} lleva un famoso factor $\frac13$, que
este capítulo hace plausibles y utiliza; el estudio de \omterm{def:g5:solids:def}{sólidos} continúa
con esferas y secciones en \cref{ch:g9:solids}.

\section{Describiendo pirámides y conos.}

\begin{definition}[pirámide, cono]\label{def:g8:solids:def}
\begin{itemize}
  \item Un \emph{pirámide}\index{pirámide} tiene un \omterm{def:g4:geometry:polygon}{polígono} como
        \emph{base} y un punto, el \emph{ápex}, unido a cada
        \omterm{def:g5:solids:def}{vértice} de la base por triangular \omterm{def:g5:solids:def}{caras}. Su \emph{altura} es
        la distancia desde el vértice al plano de la base.
  \item A \emph{cono}\index{cono} es la misma construcción sobre un
        disco: un disco base, un ápice y una superficie lateral curva.
\end{itemize}
Una pirámide es \emph{regular} cuando su base es una \omterm{def:g4:geometry:polygon}{polígono} regular
(cuadrado, \omterm{def:g6:shapes:triangles}{triangulo equilatero}, \dots) y su ápice se encuentra verticalmente
encima del centro de la base.
\end{definition}

\begin{example}[Contando caras, aristas, vértices]\label{ex:g8:solids:count}
Un \omterm{def:g8:solids:def}{pirámide} con base cuadrada tiene $5$ \omterm{def:g5:solids:def}{caras} ($1$ cuadrado $+ 4$
triángulos), aristas $8$ y $5$ \omterm{def:g5:solids:def}{vértices}. Con base de lados $n$:
$n + 1$ \omterm{def:g5:solids:def}{caras}, $2n$ bordes, $n + 1$ \omterm{def:g5:solids:def}{vértices}. Mira el pequeño de Euler
patrón: \omterm{def:g5:solids:def}{caras} $+$ \omterm{def:g5:solids:def}{vértices} $=$ bordes $+ 2$.
\end{example}

\section{Volumen}

\begin{theorem}[Volumen de una pirámide o cono]\label{thm:g8:solids:volume}
Para un \omterm{def:g8:solids:def}{pirámide} o un \omterm{def:g8:solids:def}{cono} con base \omterm{def:g6:measure:area}{área} $B$ y altura $h$:
\[
V = \frac{1}{3}\, B \times h
\]
--- un tercio del \omterm{def:g7:areas:prism}{prisma} o \omterm{def:g7:areas:prism}{cilindro} con la misma base y altura.
Para un \omterm{def:g8:solids:def}{cono} de base \omterm{def:g3:shapes:shapes}{radio} $r$: $V = \frac13 \pi r^2 h$.
\end{theorem}
\admitted

\begin{remark}[¿Por qué un tercio?]
Rellenar un hueco \omterm{def:g8:solids:def}{pirámide} con agua y verterlo en el \omterm{def:g7:areas:prism}{prisma} del mismo.
base y altura: se necesitan exactamente tres rellenos. Aún mejor: un \omterm{def:g5:solids:def}{cubo}
se puede cortar en tres \omterm{def:g8:solids:def}{pirámides} idénticos, cada uno con un \omterm{def:g5:solids:def}{rostro} del
\omterm{def:g5:solids:def}{cubo} como base y un \omterm{def:g5:solids:def}{vértice} del \omterm{def:g5:solids:def}{cubo} como ápice (¡intenta imaginártelo!).
Una prueba general utiliza la integración, una herramienta del volumen de High School.
\end{remark}

\begin{example}\label{ex:g8:solids:volume}
Un \omterm{def:g8:solids:def}{pirámide} tiene una base cuadrada de lado $5$ cm y altura $9$ cm:
\begin{enumerate}
  \item base \omterm{def:g6:measure:area}{área}: $B = 5^2 = 25$ cm$^2$;
  \item \omterm{def:g6:measure:volume}{volumen}: $V = \frac13 \times 25 \times 9 = 25 \times 3 = 75$
        cm$^3$.
\end{enumerate}
Un helado \omterm{def:g8:solids:def}{cono} de \omterm{def:g3:shapes:shapes}{radio} $3$ cm y altura $10$ cm:
\[
V = \frac13 \times \pi \times 3^2 \times 10 = 30\pi \approx 94
\text{ cm}^3 .
\]
\end{example}

\begin{example}[Cuidado con la altura]\label{ex:g8:solids:slant}
Para un \omterm{def:g8:solids:def}{cono}, no confunda la altura $h$ (ápice al centro del
base, \omterm{def:g6:lines:perp}{perpendicular}) con el \emph{altura inclinada} $s$ (ápice del
llanta). Están relacionados por Pitágoras.
(\cref{thm:g8:pythagoras:direct}) en el \omterm{def:g6:shapes:triangles}{triangulo rectángulo}
ápice--centro--borde:
\[
s^2 = h^2 + r^2 .
\]
Un \omterm{def:g8:solids:def}{cono} con $r = 3$ y $s = 5$ tiene por tanto altura
$h = \sqrt{25 - 9} = 4$ y \omterm{def:g6:measure:volume}{volumen}
$\frac13 \pi \times 9 \times 4 = 12\pi$.
\end{example}

\begin{method}[Problemas de volumen]\label{met:g8:solids:method}
\begin{enumerate}
  \item Identifique el \omterm{def:g5:solids:def}{sólido} (\omterm{def:g7:areas:prism}{prisma}/\omterm{def:g7:areas:prism}{cilindro}: $V = Bh$; \omterm{def:g8:solids:def}{pirámide}/\omterm{def:g8:solids:def}{cono}:
        $V = \frac13 Bh$);
  \item calcula primero la base \omterm{def:g6:measure:area}{área} $B$, como un paso separado;
  \item verifique que la altura sea \omterm{def:g6:lines:perp}{perpendicular} hasta la base (use
        Pitágoras si se da la altura inclinada);
  \item multiplica, mantén las unidades consistentes, convierte al final si es necesario
        ($1$ L $= 1000$ cm$^3$).
\end{enumerate}
\end{method}

\section{Redes}

\begin{example}[Rojo de una pirámide]\label{ex:g8:solids:net}
Al desplegar un \omterm{def:g8:solids:def}{pirámide} regular de base cuadrada, se aplana en su
\emph{\omterm{def:g5:solids:net}{neto}}: la base cuadrada con cuatro \omterm{def:g6:shapes:triangles}{triángulos isósceles} idénticos
pegado a sus costados. Los lados iguales de los triángulos son los
\emph{bordes laterales} del \omterm{def:g8:solids:def}{pirámide} --- su longitud no es ni $h$
ni la altura inclinada de un \omterm{def:g5:solids:def}{rostro}, así que etiquételo cuidadosamente antes de cortar
cartón.
\end{example}

\section{Problema: un cubo cortado en tres pirámides}

\begin{problem}[{Problema de fin de semana --- donde el factor $\frac13$
viene, y que pasa con una pirámide cortada a media altura}]
\label{pb:g8:solids:1}
El comentario después de \cref{thm:g8:solids:volume} afirma que un \omterm{def:g5:solids:def}{cubo}
se puede cortar en tres \omterm{def:g8:solids:def}{pirámides} idénticos --- ``intenta imaginar
¡Eso! ''. Este problema lo representa con precisión, obtiene el famoso factor
$\frac13$ honestamente, luego corta el \omterm{def:g5:solids:def}{cubo} por una segunda vía
\emph{seis} \omterm{def:g8:solids:def}{pirámides}, y termina cortando un \omterm{def:g8:solids:def}{pirámide} en \omterm{def:g3:division:half}{medio} su
altura --- con el resultado de que pocas personas aciertan en el primer intento.

En todo momento, $ABCDEFGH$ es un \omterm{def:g5:solids:def}{cubo} del lado $a$: base $ABCD$, arriba
\omterm{def:g5:solids:def}{rostro} $EFGH$, con $E$ arriba de $A$, $F$ arriba de $B$, $G$ arriba de $C$ y
$H$ encima de $D$.

\medskip
\noindent\textbf{Parte I --- Three \omterm{def:g8:solids:def}{pirámides} sharing an apex.}
\begin{enumerate}
  \item El \omterm{def:g5:solids:def}{vértice} $G$ pertenece a tres \omterm{def:g5:solids:def}{caras} del \omterm{def:g5:solids:def}{cubo}. Lista
        los tres \omterm{def:g5:solids:def}{caras} que hacen \emph{no} contienen $G$, y
        describir los tres \omterm{def:g8:solids:def}{pirámides} obtenidos tomando cada uno de
        ellos como base, con el ápice $G$ cada vez. ¿Cuántos \omterm{def:g5:solids:def}{caras},
        bordes y \omterm{def:g5:solids:def}{vértices} tiene cada \omterm{def:g8:solids:def}{pirámide}
        (\cref{ex:g8:solids:count})?
  \item Explique por qué girar el \omterm{def:g5:solids:def}{cubo} un tercio de vuelta
        su larga diagonal $(AG)$ deja al \omterm{def:g5:solids:def}{cubo} sin cambios pero
        baraja las tres bases de la pregunta 1 en un ciclo ---
        y por qué los tres \omterm{def:g8:solids:def}{pirámides} son idénticos
        copias unas de otras.
  \item Admitiendo que los tres \omterm{def:g8:solids:def}{pirámides} llenan el \omterm{def:g5:solids:def}{cubo} sin
        superposición (un modelo de cartón es bastante convincente),
        deducir el \omterm{def:g6:measure:volume}{volumen} de cada uno.
  \item Verifique esto con la fórmula de
        \cref{thm:g8:solids:volume}: el \omterm{def:g8:solids:def}{pirámide} con base $ABCD$
        y el ápice $G$ tiene su ápice directamente encima de un \emph{esquina}
        de su base, exactamente como en \cref{exo:g8:solids:11}.
        Identifica su altura, aplica la fórmula y compara.
  \item Para un \omterm{def:g5:solids:def}{cubo} de lado $6$ cm, dé el \omterm{def:g6:measure:volume}{volumen} de cada uno de los
        tres \omterm{def:g8:solids:def}{pirámides} --- y explica en qué consiste esta descomposición
        tiene que ver con el experimento de verter agua del
        capítulo (tres rellenos del \omterm{def:g8:solids:def}{pirámide} para un \omterm{def:g7:areas:prism}{prisma}).
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Six \omterm{def:g8:solids:def}{pirámides} sharing the center.}
Ahora dejemos que $O$ sea el centro de \omterm{def:g5:solids:def}{cubo} y unamos $O$ a los cuatro.
\omterm{def:g5:solids:def}{vértices} de cada uno de los seis \omterm{def:g5:solids:def}{caras}.
\begin{enumerate}[resume]
  \item Describe los seis \omterm{def:g8:solids:def}{pirámides} obtenidos y explica por qué
        son copias idénticas unas de otras. cual es la altura
        de cada uno?
  \item Deducir, compartiendo el \omterm{def:g6:measure:volume}{volumen} del \omterm{def:g5:solids:def}{cubo}, el \omterm{def:g6:measure:volume}{volumen} de cada uno
        de los seis \omterm{def:g8:solids:def}{pirámides}.
  \item Vuelva a calcular ese \omterm{def:g6:measure:volume}{volumen} con la fórmula
        $V = \frac13 B h$ y verifique que las dos respuestas coincidan.
  \item Las dos descomposiciones confirman el factor $\frac13$ para
        unos \omterm{def:g8:solids:def}{pirámides} muy especiales. Explique por qué todavía no lo hacen.
        \emph{probar} \cref{thm:g8:solids:volume} para cada
        \omterm{def:g8:solids:def}{pirámide} y \omterm{def:g8:solids:def}{cono} --- y diga cómo trata el capítulo
        esa brecha: qué afirmación se admite, qué experimento
        lo admite, y en qué \omterm{def:g6:measure:volume}{volumen} de la serie se encuentra
        honestamente demostrado.
  \item Se corta un \omterm{def:g5:solids:def}{cubo} de lado $6$ cm en sus seis centros
        \omterm{def:g8:solids:def}{pirámides}. Calcule el \omterm{def:g6:measure:volume}{volumen} de cada uno, en ambos sentidos.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Cutting a \omterm{def:g8:solids:def}{pirámide} at \omterm{def:g3:division:half}{medio} height.}
$SABCD$ es un \omterm{def:g8:solids:def}{pirámide} regular de base cuadrada: lado base $c$, altura
$h$, ápice $S$ por encima del centro de la base. Cortalo por el avión
a través del \emph{puntos medios} de los cuatro bordes laterales $[SA]$,
$[SB]$, $[SC]$, $[SD]$.
\begin{enumerate}[resume]
  \item Aplicar el teorema del punto medio
        (\cref{thm:g8:midpoints:direct}) en el triángulo $SAB$:
        ¿Cuáles son la dirección y la longitud de la unión \omterm{def:g6:lines:objects}{segmento}?
        los puntos medios de $[SA]$ y $[SB]$? Deducir que el corte
        es un cuadrado de lado $\frac{c}{2}$, \omterm{def:g6:lines:perp}{paralelo} a la base,
        en altura $\frac{h}{2}$.
  \item La pieza encima del corte es en sí misma una \omterm{def:g8:solids:def}{pirámide} de base cuadrada.
        Indique el lado y la altura de su base y demuestre que es \omterm{def:g6:measure:volume}{volumen}.
        es exactamente \emph{un octavo} del \omterm{def:g6:measure:volume}{volumen} $V$ original.
  \item Deducir el \omterm{def:g6:measure:volume}{volumen} de la pieza inferior (el \emph{tronco},
        la forma de un \omterm{def:g8:solids:def}{pirámide} sin terminar). ¿En qué proporción
        ¿El corte de media altura comparte el \omterm{def:g6:measure:volume}{volumen}?
  \item El Louvre \omterm{def:g8:solids:def}{pirámide} (\cref{exo:g8:solids:7}: lado de la base
        $35$ m, altura $22$ m) se limpia en dos campañas: la
        vidrio por encima de la altura \omterm{def:g3:division:half}{medio}, luego el vidrio por debajo. que
        \omterm{def:g6:fractions:def}{fracción} del \emph{\omterm{def:g6:measure:volume}{volumen}} se encuentra por debajo de la altura de \omterm{def:g3:division:half}{medio}?
        Estímalo en metros cúbicos, a la centena más cercana.
  \item La moraleja general: si \emph{todo} las dimensiones de un
        \omterm{def:g8:solids:def}{pirámide} se multiplican por $2$, ¿qué pasa con su
        \omterm{def:g6:measure:volume}{volumen}? ¿En qué factor crece el \omterm{def:g6:measure:volume}{volumen} cuando cada
        dimensión se multiplica por $k$ --- y ¿cómo funciona esto?
        ¿Explicar, en una línea, el ``un octavo'' de la pregunta 12?
\end{enumerate}
\end{problem}
