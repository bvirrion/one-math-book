\chapter{Proporcionalidad, Velocidad y Promedios}\label{ch:g8:speed}

\omterm{def:g7:prop:table}{Proporcionalidad} (\cref{ch:g7:prop}) ahora se encuentra con el mundo físico:
\omterm{def:g8:speed:def}{velocidad}, caudales, conversiones de unidades --- cantidades que son
\emph{\omterm{def:g3:division:remainder}{cocientes}} de otros dos. El capítulo cierra con \omterm{def:g8:speed:average}{ponderado
promedios}, donde el pensamiento \omterm{def:g6:propdata:def}{proporcional} evita un error clásico.

\section{Velocidad media}

\begin{definition}[medios de velocidad]\label{def:g8:speed:def}
El \emph{velocidad promedio}\index{velocidad} de un viaje es el \omterm{def:g3:division:remainder}{cociente}
\[
v = \frac{d}{t}
\qquad
\left(\text{distance divided by duration}\right),
\]
en kilómetros por hora (km/h), metros por segundo (m/s), \dots\ El
la fórmula cambia: $d = v \times t$ y $t = \frac{d}{v}$.
\end{definition}

\begin{example}\label{ex:g8:speed:basic}
Un tren recorre $270$ km en $1$ h $30$ min. Primero convierta la hora:
$1$ h $30$ min $= 1.5$ h. entonces
\[
v = \frac{270}{1.5} = 180 \text{ km/h}.
\]
Distancia a $90$ km/h durante $2$ h $20$ min $= \frac73$ h:
$d = 90 \times \frac73 = 210$ kilómetros. Tiempo para $35$ km a $14$ km/h:
$t = \frac{35}{14} = 2.5$ h $= 2$ h $30$ mín.
\end{example}

\begin{remark}[Los minutos no son decimales]
$1$ h $30$ min es $1.5$ h, pero $1$ h $20$ min es \emph{no} $1.2$ h:
son $1 + \frac{20}{60} = \frac43 \approx 1.33$h. Convertir siempre
minutos a \omterm{def:g6:fractions:def}{fracción} de una hora ($60$ min $= 1$ h) antes de dividir.
\end{remark}

\begin{method}[Velocidades de conversión]\label{met:g8:speed:convert}
Para convertir km/h a m/s: $1$ km $= 1000$ m y $1$ h $= 3600$ s, entonces
\[
v \text{ km/h} = \frac{v \times 1000}{3600} \text{ m/s}
= \frac{v}{3.6} \text{ m/s}.
\]
Para convertir m/s a km/h, multiplique por $3.6$.
\end{method}

\begin{example}\label{ex:g8:speed:convert}
$90$ km/h $= \frac{90}{3.6} = 25$ m/s. Un velocista que corre $100$ m en
$10$ s se mueve a $10$ m/s $= 36$ km/h.
\end{example}

\section{Cantidades cocientes}

\begin{example}[Caudal, densidad, precio por kilo]\label{ex:g8:speed:quotients}
\omterm{def:g8:speed:def}{Velocidad} tiene muchos primos, todos tratados de la misma manera:
\begin{itemize}
  \item un grifo llena $48$ L en $4$ min: caudal
        $\frac{48}{4} = 12$ L/min, por lo que llenar una tina $150$ L requiere
        $\frac{150}{12} = 12.5$ mín;
  \item $0.6$ kg de queso cuesta $9$ euros: precio
        $\frac{9}{0.6} = 15$ euros por kg;
  \item un coche utiliza $6.3$ L para $90$ km: consumo
        $\frac{6.3}{90} \times 100 = 7$ L por $100$ km.
\end{itemize}
Identificar el \omterm{def:g3:division:remainder}{cociente} y la fórmula de tres vías.
($q = \frac ab$, $a = qb$, $b = \frac aq$) hace el resto.
\end{example}

\section{Promedios ponderados}

\begin{definition}[Promedio ponderado]\label{def:g8:speed:average}
Cuando los valores $v_1, v_2, \dots$ vienen con recuentos (o pesos)
$n_1, n_2, \dots$, su \emph{promedio ponderado}\index{media ponderada}
es
\[
\bar v = \frac{n_1 v_1 + n_2 v_2 + \dots}{n_1 + n_2 + \dots} :
\]
total de los valores, dividido por el total de los pesos.
\end{definition}

\begin{example}\label{ex:g8:speed:average}
Se realizó una prueba entre dos grupos: grupo 1, $12$ estudiantes, promedio
$11$; grupo 2, $18$ estudiantes, promedio $14$. promedio del conjunto
clase:
\[
\bar v = \frac{12 \times 11 + 18 \times 14}{12 + 18}
= \frac{132 + 252}{30} = \frac{384}{30} = 12.8 .
\]
\emph{No} $\frac{11 + 14}{2} = 12.5$: el grupo mayor tira del
promedio hacia el suyo propio: los pesos importan.
\end{example}

\begin{example}[Velocidad media en dos piernas]\label{ex:g8:speed:twolegs}
Un ciclista recorre $30$ km a $30$ km/h, luego $30$ km a $15$ km/h. el
\omterm{def:g8:speed:def}{velocidad promedio} durante todo el viaje es \emph{no}
$\frac{30 + 15}{2} = 22.5$ kilómetros por hora. Calcular con la definición:
\begin{enumerate}
  \item veces: $\frac{30}{30} = 1$ h, luego $\frac{30}{15} = 2$ h;
        total $3$ h;
  \item distancia total $60$ km, entonces
        $v = \frac{60}{3} = 20$ kilómetros por hora.
\end{enumerate}
La pierna lenta dura más, por lo que pesa más --- un promedio de \omterm{def:g8:speed:def}{velocidades}
debe ponderarse por \emph{tiempo}.
\end{example}

\section{Problema: La media armónica, o por qué se arruina el viaje de regreso
el promedio}

\begin{problem}[{Problema de fin de semana --- el \omterm{def:g8:speed:def}{velocidad promedio} sobre igual
distancias es $\frac{2 v_1 v_2}{v_1 + v_2}$, y nunca supera el
punto medio del \omterm{def:g8:speed:def}{velocidades}}]
\label{pb:g8:speed:1}
\Cref{ex:g8:speed:twolegs} y \cref{exo:g8:speed:10} terminaron ambos
con la misma sorpresa: salimos a la una \omterm{def:g8:speed:def}{velocidad}, regresamos a las
otro, y el \omterm{def:g8:speed:def}{velocidad promedio} aterriza \emph{abajo} en el punto medio de
los dos. Este problema encuentra la fórmula exacta detrás de la sorpresa.
--- un nuevo tipo de promedio, el \emph{armónico
significar}\index{media armónica} --- demuestra que la sorpresa es una
teorema, y lo lleva a su extremo lógico: una puesta al día que ningún
\omterm{def:g8:speed:def}{velocidad} en el mundo puede lograr.

\medskip
\noindent\textbf{Parte I --- The formula for a round trip.}
Un viaje recorre una distancia $d$ a \omterm{def:g8:speed:def}{velocidad} $v_1$, luego lo mismo
distancia $d$ nuevamente en \omterm{def:g8:speed:def}{velocidad} $v_2$ (todos números positivos).
\begin{enumerate}
  \item Exprese las dos duraciones, luego la duración total, como
        \omterm{def:g6:fractions:def}{fracciones} que involucra a $d$, $v_1$, $v_2$.
  \item El \omterm{def:g8:speed:def}{velocidad promedio} de todo el recorrido es
        $\bar v = \dfrac{2d}{\ \dfrac{d}{v_1} + \dfrac{d}{v_2}\ }$
        --- un dos pisos \omterm{def:g6:fractions:def}{fracción}
        (\cref{ex:g8:fractions:complex}). Simplificalo y demuestra:
        \[
        \bar v = \frac{2\, v_1 v_2}{v_1 + v_2} .
        \]
  \item Verifique la fórmula con los dos cálculos citados anteriormente:
        $v_1 = 30$, $v_2 = 15$ km/h
        (\cref{ex:g8:speed:twolegs}) y $v_1 = 80$,
        $v_2 = 120$ km/h (\cref{exo:g8:speed:10}).
  \item La distancia $d$ ha desaparecido de la fórmula. que
        ¿Qué significa eso, concretamente, para el ciclista?
  \item Ahora supongamos que el viaje pasa lo mismo
        \emph{tiempo} $T$ en cada \omterm{def:g8:speed:def}{velocidad}. Demuestre que \omterm{def:g8:speed:def}{promedio
        velocidad} es entonces $\frac{v_1 + v_2}{2}$, el punto medio simple.
        En el idioma de \cref{def:g8:speed:average}: por qué
        cantidad siempre se debe ponderar un promedio de \omterm{def:g8:speed:def}{velocidades},
        ¿Y qué cambia entre los dos escenarios?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Harmonic against arithmetic.}
Para dos números positivos $v_1$, $v_2$, establezca
\[
H = \frac{2\, v_1 v_2}{v_1 + v_2}
\quad\text{(their \emph{media armónica}),}
\qquad
A = \frac{v_1 + v_2}{2}
\quad\text{(their \emph{media aritmética}).}
\]
\begin{enumerate}[resume]
  \item Calcule $H$ y $A$ para los pares $(30, 15)$ y
        $(80, 120)$. ¿Cuál de los dos medios gana cada vez?
  \item Coloque $A - H$ sobre el \omterm{def:g4:fractions:def}{denominador} $2(v_1 + v_2)$ común,
        ampliar el \omterm{def:g4:fractions:def}{numerador} con las identidades notables
        (\cref{pb:g8:equations:1}), y demostrar:
        \[
        A - H = \frac{(v_1 - v_2)^2}{2\,(v_1 + v_2)} .
        \]
  \item Deduce el teorema detrás de todas las sorpresas: por positivo
        \omterm{def:g8:speed:def}{velocidades}, $H \leq A$ siempre, con igualdad exactamente cuando
        $v_1 = v_2$. ¿Por qué se asienta un cuadrado en el \omterm{def:g4:fractions:def}{numerador}?
        el asunto?
  \item Demuestre la identidad elegante $H \times A = v_1 \times v_2$:
        los dos medios se multiplican hasta obtener el \omterm{def:g2:mult:def}{producto} original.
  \item Un automóvil recorre $60$ km a $40$ km/h y $60$ km a
        $60$ kilómetros por hora. Predice el \omterm{def:g8:speed:def}{velocidad promedio} con la fórmula,
        Luego confírmalo a lo largo, con tiempos y total.
        distancia.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The impossible catch-up.}
\begin{enumerate}[resume]
  \item Un ciclista participa en una carrera de $30$ km con la esperanza de promediar
        $30$ kilómetros por hora. ¿Qué tiempo total puede tomarle la carrera, en
        ¿la mayoría? Recorre los primeros $15$ km a $15$ km/h: ¿cuánto
        ¿De ese tiempo queda presupuesto?
  \item Demuestra que el promedio esperado se ha convertido literalmente
        imposible: calcule su \omterm{def:g8:speed:def}{velocidad promedio} si monta la
        segundo \omterm{def:g3:division:half}{medio} a $90$ km/h, y explique por qué \omterm{def:g3:numbers:evenodd}{incluso}
        arbitrariamente enorme \omterm{def:g8:speed:def}{velocidad} la deja sin
        $30$ kilómetros por hora.
  \item Baja su meta a $20$ km/h. ¿Qué \omterm{def:g8:speed:def}{velocidad} en el
        segundo $15$ km lo logra? Comprueba tu respuesta con el
        fórmula de media armónica.
  \item Un solo grifo de agua fría llena una tina en $20$ min; un grifo caliente solo
        lo llena en $30$ min. Que \omterm{def:g6:fractions:def}{fracción} de la tina hacen ambos
        Los grifos juntos se llenan por minuto y ¿cuánto tardan?
        Compara tu respuesta con $H(20, 30)$ --- ¿qué
        aviso?
  \item Un avión recorre la misma ruta de ida y vuelta: $900$ km/h
        con viento de $700$ km/h en contra. Calcular el
        ida y vuelta \omterm{def:g8:speed:def}{velocidad promedio}, y explica en una frase por qué
        un viento, como sople, siempre \emph{alarga} una vuelta
        viaje.
\end{enumerate}
\end{problem}
