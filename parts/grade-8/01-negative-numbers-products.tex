\chapter{Multiplying Negative Numbers}\label{ch:g8:negprod}

Grade 7 added and subtracted \omterm{def:g7:negatives:def}{relative numbers}
(\cref{ch:g7:negatives}); it remains to multiply and divide them. One
question dominates the chapter: why should ``minus times minus'' be
plus? We give a reason, not just a rule.

\section{The sign rules}

\begin{theorem}[Sign of a product]\label{thm:g8:negprod:rules}
The \omterm{def:g2:mult:def}{product} of two \omterm{def:g7:negatives:def}{relative numbers} has distance to \omterm{def:g1:counting:zero}{zero} equal to the
\omterm{def:g2:mult:def}{product} of the distances, and its sign is given by:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|cc}
$\times$ & $+$ & $-$ \\
\hline
$+$ & $+$ & $-$ \\
$-$ & $-$ & $+$ \\
\end{tabular}
\end{center}
\emph{Same signs: positive \omterm{def:g2:mult:def}{product}. Opposite signs: negative \omterm{def:g2:mult:def}{product}.}
The same table governs \omterm{def:g3:division:remainder}{quotients}.
\end{theorem}

\begin{proof}[Why $(-1) \times (-1) = +1$]
First, $3 \times (-4)$ means $(-4) + (-4) + (-4) = -12$: a positive
times a negative is negative. Now watch the pattern:
\[
3 \times (-4) = -12, \quad
2 \times (-4) = -8, \quad
1 \times (-4) = -4, \quad
0 \times (-4) = 0 .
\]
Each step, the first factor drops by $1$ and the result \emph{rises}
by $4$. Continuing the pattern one more step:
\[
(-1) \times (-4) = 0 + 4 = +4 .
\]
Any other choice would break the regularity of arithmetic
(distributivity). So minus times minus must be plus.
\end{proof}

\begin{example}\label{ex:g8:negprod:products}
\begin{align*}
(-7) \times (+6) &= -42 && \text{(opposite signs)} \\
(-5) \times (-8) &= +40 && \text{(same signs)} \\
(+2.5) \times (-4) &= -10 \\
(-36) \div (-9) &= +4 \\
(+15) \div (-2) &= -7.5 .
\end{align*}
\end{example}

\section{Products of several factors}

\begin{proposition}[Sign of a long product]\label{prop:g8:negprod:many}
The sign of a \omterm{def:g2:mult:def}{product} of several nonzero factors depends only on the
number of negative factors:
\begin{itemize}
  \item an \emph{\omterm{def:g3:numbers:evenodd}{even}} number of negative factors gives a positive
        \omterm{def:g2:mult:def}{product};
  \item an \emph{\omterm{def:g3:numbers:evenodd}{odd}} number gives a negative \omterm{def:g2:mult:def}{product}.
\end{itemize}
\end{proposition}

\begin{proof}
Negative factors pair up, and each pair contributes a positive sign
(\cref{thm:g8:negprod:rules}); an \omterm{def:g3:numbers:evenodd}{odd} count leaves one unpaired
negative factor, which makes the whole \omterm{def:g2:mult:def}{product} negative.
\end{proof}

\begin{example}\label{ex:g8:negprod:many}
$(-2) \times (+3) \times (-5) \times (-1)$: three negative factors
(\omterm{def:g3:numbers:evenodd}{odd}), so the \omterm{def:g2:mult:def}{product} is negative; distances: $2 \times 3 \times 5
\times 1 = 30$. Result: $-30$. Decide the sign \emph{first}, then
multiply the distances --- two easy tasks instead of one error-prone
one.
\end{example}

\begin{example}[Powers of negatives]\label{ex:g8:negprod:powers}
$(-2)^2 = (-2) \times (-2) = +4$, and
$(-2)^3 = (-2)^2 \times (-2) = -8$: \omterm{def:g3:numbers:evenodd}{even} exponents give positive
values, \omterm{def:g3:numbers:evenodd}{odd} exponents keep the sign. Careful with notation:
$(-3)^2 = 9$ but $-3^2 = -(3 \times 3) = -9$.
\end{example}

\section{Computing with all four operations}

\begin{method}[Safe computation with relative numbers]\label{met:g8:negprod:compute}
\begin{enumerate}
  \item Respect the priorities of \cref{ch:g7:priorities}: brackets,
        then $\times$ and $\div$, then $+$ and $-$;
  \item at each multiplication or division, find the \emph{sign
        first}, then the distance;
  \item rewrite one step per line.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:negprod:compute}
\begin{align*}
5 - 3 \times (-4)
&= 5 - (-12) && \text{(product first: opposite signs)}\\
&= 5 + 12 = 17 .
\end{align*}
\begin{align*}
\frac{(-6) \times (-10)}{-4}
&= \frac{+60}{-4} && \text{(numerator: same signs)}\\
&= -15 && \text{(quotient: opposite signs).}
\end{align*}
\end{example}

\begin{example}[Substituting negative values]\label{ex:g8:negprod:substitute}
Evaluate $E = 3x^2 - 2x$ for $x = -5$, brackets around the value:
\[
E = 3 \times (-5)^2 - 2 \times (-5)
= 3 \times 25 - (-10)
= 75 + 10 = 85 .
\]
The brackets in $(-5)^2$ are essential: without them the square would
apply to $5$ only.
\end{example}

\begin{exercise}[$\star\star$]\label{exo:g8:negprod:8}

(-6) \times \;?\; = 42, \qquad

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:negprod:9}
True or false? Justify or give a counterexample.
\begin{enumerate}
  \item The \omterm{def:g2:mult:def}{product} of two \omterm{def:g7:negatives:def}{relative numbers} is always at least as
        large as their \omterm{def:g1:addition:def}{sum}.
  \item The square of a relative number is never negative.
  \item If a \omterm{def:g2:mult:def}{product} of three factors is positive, all three factors
        are positive.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:negprod:10}
Each wrong answer at a quiz counts $-3$ points, each right answer
$+5$. Zoe answered $20$ questions and scored $36$ points. How many
answers were right? (Try values, or set up the computation
$5r - 3(20 - r) = 36$.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g8:negprod:11}
Using distributivity (as in the proof of
\cref{thm:g8:negprod:rules}), expand and justify each step of
\[
0 = (-1) \times 0 = (-1) \times \bigl(1 + (-1)\bigr)
= (-1) \times 1 + (-1) \times (-1),
\]
and conclude that $(-1) \times (-1)$ must equal $+1$.
\end{exercise}

\section{Problem: Why minus times minus is plus}

\begin{problem}[{Weekend problem --- the sign rules deduced from
distributivity, and the alternating sum $1 - 2 + 3 - 4 + \dots$}]
\label{pb:g8:negprod:1}
The proof of \cref{thm:g8:negprod:rules} continued a pattern and
claimed that ``any other choice would break distributivity''. This
problem makes the claim exact: starting from distributivity alone
(\cref{thm:g7:priorities:distributivity}), you will \emph{prove} the
sign rules --- no picture, no pattern, the way algebra does it ---
then put them to work on powers of $-1$ and on a famous \omterm{def:g1:addition:def}{sum} with
alternating signs. Throughout, $a$ and $b$ are \omterm{def:g7:negatives:def}{relative numbers}; we
take for granted the addition rules of \cref{ch:g7:negatives}, that
$1 \times a = a$, and that a \omterm{def:g2:mult:def}{product} may be computed in any order
(so distributivity applies on either side of a \omterm{def:g2:mult:def}{product}).

\medskip
\noindent\textbf{Part I --- The sign rules are theorems.}
\begin{enumerate}
  \item Prove that $0 \times a = 0$ for every \omterm{def:g7:negatives:def}{relative number} $a$.
        (Write $0 = 0 + 0$, expand $(0 + 0) \times a$, and ask which
        number, added to $0 \times a$, gives $0 \times a$ back.)
  \item Expand $\bigl(1 + (-1)\bigr) \times a$ to show that
        $(-1) \times a + a = 0$, and conclude that
        \[
        (-1) \times a = -a :
        \]
        multiplying by $-1$ gives the opposite. Compare with
        \cref{exo:g8:negprod:11}, which is the case $a = -1$.
  \item Deduce that $(-a) \times b = -(a \times b)$: one minus sign
        comes out of a \omterm{def:g2:mult:def}{product} unchanged.
  \item Deduce that $(-a) \times (-b) = a \times b$: two minus signs
        cancel.
  \item Every \omterm{def:g7:negatives:def}{relative number} is its distance to \omterm{def:g1:counting:zero}{zero} with a sign in
        front: $a = +d$ or $a = -d$. Explain how questions 3 and~4
        prove all four cases of the sign table of
        \cref{thm:g8:negprod:rules}, distances included.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Powers of $-1$.}
\begin{enumerate}[resume]
  \item Compute $(-1)^2$, $(-1)^3$, $(-1)^4$, $(-1)^5$, then give,
        with justification from \cref{prop:g8:negprod:many}, the
        value of $(-1)^n$ for every whole number $n \geq 1$.
  \item Give (without computing any distance) the sign of
        \[
        (-1) \times (-2) \times (-3) \times \dots \times (-10),
        \]
        and write the distance of this \omterm{def:g2:mult:def}{product} as a \omterm{def:g2:mult:def}{product} of whole
        numbers, without working it out.
  \item Show that $(-a)^2 = a^2$ and that $(-a)^3 = -a^3$. For which
        exponents does the minus sign survive?
  \item Use the sign rules to show that the square of a relative
        number is never negative, and deduce that no \omterm{def:g7:negatives:def}{relative number}
        $x$ satisfies $x^2 = -4$.
  \item Zoe claims: ``$(-1)^{2026} + (-1)^{2027} = 0$, and more
        generally two consecutive powers of $-1$ always cancel.'' Is
        she right? Justify.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The alternating \omterm{def:g1:addition:def}{sum}.}
With the sign rules secured, we can compute \omterm{def:g1:addition:def}{sums} that mix both
signs. For a whole number $n \geq 1$, let
\[
A_n = 1 - 2 + 3 - 4 + \dots
\]
be the \omterm{def:g1:addition:def}{sum} of the whole numbers from $1$ to $n$ with alternating
signs: the last term is $+n$ when $n$ is \omterm{def:g3:numbers:evenodd}{odd}, $-n$ when $n$ is
\omterm{def:g3:numbers:evenodd}{even}. So $A_1 = 1$ and $A_2 = 1 - 2 = -1$.
\begin{enumerate}[resume]
  \item Compute $A_3$, $A_4$, $A_5$, $A_6$ and $A_7$. What do you
        conjecture?
  \item Suppose $n$ is \omterm{def:g3:numbers:evenodd}{even}. Group the terms two by two,
        $(1 - 2) + (3 - 4) + \dots$, count the pairs, and prove that
        $A_n = -\frac{n}{2}$.
  \item Suppose $n$ is \omterm{def:g3:numbers:evenodd}{odd}. Explain why $A_n = A_{n-1} + n$, and
        deduce from the previous question that
        $A_n = \frac{n + 1}{2}$.
  \item Compute $A_{100}$, $A_{101}$ and $A_{2026}$.
  \item Alma computed $A_{101} - A_{100} = 51 - (-50) = 101$ and was
        surprised to find exactly $101$. Should she have been?
        Explain in one sentence, and state the value of
        $A_{2027} - A_{2026}$ without any further computation.
\end{enumerate}
\end{problem}
