\chapter{Fractions: All Four Operations}\label{ch:g8:fractions}

This chapter completes the arithmetic of \omterm{def:g6:fractions:def}{fractions}: addition and
subtraction with \emph{any} \omterm{def:g4:fractions:def}{denominators}, multiplication, and ---
the newcomer --- division, which turns out to be a multiplication in
disguise. Signs from \cref{ch:g8:negprod} come along for the ride.

\section{Adding with any denominators}

\begin{method}[Common denominator]\label{met:g8:fractions:add}
To add or subtract $\frac ab$ and $\frac cd$:
\begin{enumerate}
  \item find a common \omterm{def:g5:division:multiple}{multiple} of $b$ and $d$ (the \omterm{def:g2:mult:def}{product}
        $b \times d$ always works; a smaller one saves effort);
  \item rewrite both \omterm{def:g6:fractions:def}{fractions} with that \omterm{def:g4:fractions:def}{denominator};
  \item add or subtract the \omterm{def:g4:fractions:def}{numerators}; simplify.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:fractions:add}
Compute $\dfrac{5}{6} + \dfrac{3}{4}$. A common \omterm{def:g5:division:multiple}{multiple} of $6$ and
$4$ is $12$:
\[
\frac{5}{6} = \frac{10}{12},
\qquad
\frac{3}{4} = \frac{9}{12},
\qquad
\frac{5}{6} + \frac{3}{4} = \frac{10 + 9}{12} = \frac{19}{12}.
\]
With signs: $\dfrac{2}{3} - \dfrac{7}{5} = \dfrac{10}{15} -
\dfrac{21}{15} = -\dfrac{11}{15}$.
\end{example}

\section{Multiplying fractions}

\begin{theorem}[Product of fractions]\label{thm:g8:fractions:mult}
\[
\frac{a}{b} \times \frac{c}{d} = \frac{a \times c}{b \times d} .
\]
\end{theorem}

\begin{proof}[Why, on a picture]
Take $\frac34$ of $\frac25$ of a square: cut the square in $5$
vertical strips and keep $2$; cut horizontally in $4$ and keep $3$
rows of what remained. The kept part is a grid of $3 \times 2$ small
cells out of $4 \times 5$: \omterm{def:g6:fractions:def}{fraction} $\frac{6}{20}$.
\end{proof}

\begin{omfigure}
\begin{tikzpicture}[scale=0.68]
  \draw[gray!50, thin] (0,0) grid[xstep=1, ystep=1] (5,4);
  \fill[omDef!18] (0,0) rectangle (2,4);
  \fill[omThm!35] (0,0) rectangle (2,3);
  \draw[thick] (0,0) rectangle (5,4);
  \draw[omDef, thick] (2,0) -- (2,4);
  \draw[omThm, thick] (0,3) -- (5,3);
  \node[below, font=\small, omDef] at (1,0) {$\frac25$ kept};
  \node[right, font=\small, omThm] at (5.05,1.5) {$\frac34$ of it};
\end{tikzpicture}

{\small $\frac34 \times \frac25$: the doubly-shaded region is
$3 \times 2 = 6$ cells out of $4 \times 5 = 20$, i.e.\
$\frac{6}{20} = \frac{3}{10}$ --- \omterm{def:g4:fractions:def}{numerators} multiplied, \omterm{def:g4:fractions:def}{denominators}
multiplied.}
\end{omfigure}

\begin{example}\label{ex:g8:fractions:mult}
Simplify \emph{before} multiplying, crossing out common factors:
\[
\frac{7}{12} \times \frac{9}{14}
= \frac{7 \times 9}{12 \times 14}
= \frac{7 \times 3 \times 3}{3 \times 4 \times 2 \times 7}
= \frac{3}{8}
\]
(the factor $7$ and one factor $3$ cancel between top and bottom).
With signs, the rules of \cref{thm:g8:negprod:rules} apply first:
$\left(-\frac23\right) \times \left(-\frac{5}{4}\right) =
+\frac{10}{12} = \frac56$.
\end{example}

\section{Dividing fractions}

\begin{definition}[Inverse]\label{def:g8:fractions:inverse}
The \emph{inverse}\index{inverse} of a nonzero number $x$ is the
number which multiplied by $x$ gives $1$. The inverse of
$\frac ab$ is $\frac ba$, since
$\frac ab \times \frac ba = \frac{ab}{ab} = 1$; the inverse of an
integer $n$ is $\frac1n$.
\end{definition}

\begin{theorem}[Dividing means multiplying by the inverse]\label{thm:g8:fractions:div}
For $c \neq 0$:
\[
\frac{a}{b} \div \frac{c}{d} = \frac{a}{b} \times \frac{d}{c} .
\]
\end{theorem}

\begin{proof}
By definition, the \omterm{def:g3:division:remainder}{quotient} $q$ of $\frac ab$ by $\frac cd$ is the
number with $q \times \frac cd = \frac ab$. Check the candidate
$q = \frac ab \times \frac dc$:
\[
\left(\frac ab \times \frac dc\right) \times \frac cd
= \frac ab \times \left(\frac dc \times \frac cd\right)
= \frac ab \times 1 = \frac ab . \qedhere
\]
\end{proof}

\begin{example}\label{ex:g8:fractions:div}
\[
\frac{3}{5} \div \frac{7}{10}
= \frac{3}{5} \times \frac{10}{7}
= \frac{3 \times 10}{5 \times 7}
= \frac{30}{35} = \frac{6}{7}.
\]
A sanity check: dividing by $\frac{7}{10}$, a number smaller than
$1$, must give a result \emph{larger} than $\frac35$ --- and
$\frac67 > \frac35$ indeed ($\frac{30}{35} > \frac{21}{35}$).
\end{example}

\begin{example}[Fraction bars within fraction bars]\label{ex:g8:fractions:complex}
A ``double-decker'' is just a division written vertically:
\[
\frac{\ \frac{2}{3}\ }{\ \frac{5}{6}\ }
= \frac{2}{3} \div \frac{5}{6}
= \frac{2}{3} \times \frac{6}{5}
= \frac{12}{15} = \frac{4}{5}.
\]
Locate the \emph{main} bar first (the longest one), then apply
\cref{thm:g8:fractions:div}.
\end{example}

\begin{method}[Mixed computations]\label{met:g8:fractions:mixed}
In an expression mixing \omterm{def:g6:fractions:def}{fractions} and the four operations:
\begin{enumerate}
  \item apply the usual priorities (\cref{ch:g7:priorities});
  \item convert every division into a multiplication by the \omterm{def:g8:fractions:inverse}{inverse};
  \item decide signs first (\cref{met:g8:negprod:compute}), then
        multiply or find common \omterm{def:g4:fractions:def}{denominators};
  \item simplify at the earliest opportunity.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:fractions:mixed}
\begin{align*}
\frac12 + \frac{2}{3} \times \frac{9}{4}
&= \frac12 + \frac{2 \times 9}{3 \times 4}
&& \text{(product first!)} \\
&= \frac12 + \frac{18}{12} = \frac12 + \frac32
= \frac{4}{2} = 2 .
\end{align*}
\end{example}

Compute and simplify:

\frac{5}{6} - \frac{3}{8}, \qquad
\frac{7}{10} + \frac{8}{15} .

-\frac{2}{7} + \frac{3}{14}, \qquad
\frac{1}{6} - \frac{5}{4}, \qquad
-\frac{3}{5} - \frac{1}{2} .

\begin{exercise}[$\star$]\label{exo:g8:fractions:3}

\frac{5}{8} \times \frac{4}{15}, \qquad

\left(-\frac{6}{5}\right) \times \frac{10}{9} .

\end{exercise}

Give the \omterm{def:g8:fractions:inverse}{inverse} of: $\dfrac{3}{7}$; \quad $5$; \quad

\begin{exercise}[$\star$]\label{exo:g8:fractions:5}

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:fractions:6}

\frac{\ \frac{3}{4}\ }{\ \frac{9}{8}\ },
\qquad
\frac{\ \frac{5}{6}\ }{\ 10\ },
\qquad
\frac{\ 2\ }{\ \frac{4}{7}\ } .

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g8:fractions:7}
Compute, respecting priorities:
\[
\frac{1}{3} + \frac{1}{2} \times \frac{4}{5},
\qquad
\left(\frac{1}{3} + \frac{1}{2}\right) \times \frac{4}{5} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:fractions:8}
A tank is $\frac{2}{5}$ full. One adds $\frac{1}{3}$ \emph{of the

\omterm{def:g6:fractions:def}{fraction} is still empty?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:fractions:9}
Three \omterm{def:g3:division:half}{quarters} of the students of a class passed a test; among those,
two thirds got more than $14$ out of $20$. What \omterm{def:g6:fractions:def}{fraction} of the class
got more than $14$? If that represents $12$ students, how large is
the class?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:fractions:10}
A rope of $\frac{15}{2}$ m must be cut into pieces of $\frac{3}{4}$ m
each. How many pieces does one get? (A division of \omterm{def:g6:fractions:def}{fractions} ---
check that the answer is a whole number.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g8:fractions:11}

\begin{exercise}[$\star\star\star$]\label{exo:g8:fractions:12}

\section{Problem: Egyptian fractions}

\begin{problem}[{Weekend problem --- every fraction is a sum of
distinct unit fractions}]
\label{pb:g8:fractions:1}
A \emph{unit \omterm{def:g6:fractions:def}{fraction}} is a \omterm{def:g6:fractions:def}{fraction} with \omterm{def:g4:fractions:def}{numerator} $1$. The
scribes of ancient Egypt --- the Rhind papyrus was copied around
$1550$~BC --- wrote every \omterm{def:g6:fractions:def}{fraction} as a \omterm{def:g1:addition:def}{sum} of \emph{distinct}
unit \omterm{def:g6:fractions:def}{fractions}: never $\frac27 = \frac17 + \frac17$, but
$\frac27 = \frac14 + \frac{1}{28}$. Such a \omterm{def:g1:addition:def}{sum} is called an
\emph{Egyptian writing} of the \omterm{def:g6:fractions:def}{fraction}. This problem develops the
addition and comparison techniques of this chapter
(\cref{met:g8:fractions:add}) into a method --- published by
Fibonacci in $1202$ --- that produces an Egyptian writing of any
\omterm{def:g6:fractions:def}{fraction} between $0$ and $1$, and ends with a celebrated
inheritance puzzle.

\medskip
\noindent\textbf{Part I --- Unit \omterm{def:g6:fractions:def}{fractions} and the splitting
identity.}
\begin{enumerate}
  \item Compute $\frac12 + \frac13$ and $\frac12 + \frac14$, and
        deduce Egyptian writings of $\frac56$ and $\frac34$.
  \item Verify the scribes' entry for $\frac27$: show that
        $\frac14 + \frac{1}{28} = \frac27$.
  \item Check that $\frac13 + \frac16 = \frac12$ and that
        $\frac14 + \frac{1}{12} = \frac13$. Then prove the
        \emph{splitting identity}: for every whole number
        $n \geq 1$,
        \[
        \frac{1}{n} = \frac{1}{n+1} + \frac{1}{n(n+1)} .
        \]
  \item Deduce an Egyptian writing of $1$ itself, as a \omterm{def:g1:addition:def}{sum} of
        three distinct unit \omterm{def:g6:fractions:def}{fractions}.
  \item Split the last term of $\frac56 = \frac12 + \frac13$ to
        obtain a \emph{second} Egyptian writing of $\frac56$, with
        three unit \omterm{def:g6:fractions:def}{fractions}. Explain why no \omterm{def:g6:fractions:def}{fraction} has a unique
        Egyptian writing.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Fibonacci's greedy method.}
The method is: subtract the \emph{largest} unit \omterm{def:g6:fractions:def}{fraction} that
fits, and repeat with what remains.
\begin{enumerate}[resume]
  \item Using common \omterm{def:g4:fractions:def}{denominators}, show that
        $\frac14 < \frac27 < \frac13$. Why does this prove that
        $\frac14$ is the largest unit \omterm{def:g6:fractions:def}{fraction} smaller than
        $\frac27$?
  \item Compute $\frac27 - \frac14$. Which writing of $\frac27$
        does the greedy method produce?
  \item Now take $\frac57$. Show that the largest unit \omterm{def:g6:fractions:def}{fraction}
        smaller than $\frac57$ is $\frac12$, and compute
        $\frac57 - \frac12$.
  \item Show that the largest unit \omterm{def:g6:fractions:def}{fraction} smaller than
        $\frac{3}{14}$ is $\frac15$ (compare $\frac{3}{14}$ with
        $\frac14$ and with $\frac15$), compute
        $\frac{3}{14} - \frac15$, and conclude:
        \[
        \frac57 = \frac12 + \frac15 + \frac{1}{70} .
        \]
        Verify this writing directly with the common \omterm{def:g4:fractions:def}{denominator}
        $70$.
  \item Run the greedy method on $\frac45$, checking at each step
        that your unit \omterm{def:g6:fractions:def}{fraction} is the largest one that fits, and
        verify the final writing.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Why the method always stops, and the
eighteenth camel.}
\begin{enumerate}[resume]
  \item In questions 7--10 the method was applied to $\frac27$,
        $\frac57$ and $\frac45$. For each, list the \omterm{def:g4:fractions:def}{numerators} of
        the successive \omterm{def:g3:division:remainder}{remainders} (before any simplification).
        What do you observe?
  \item For a \omterm{def:g6:fractions:def}{fraction} $\frac ab$ and a whole number $n$, show
        that
        \[
        \frac{a}{b} - \frac{1}{n}
        = \frac{a \times n - b}{b \times n} .
        \]
  \item Suppose the greedy method subtracts $\frac1n$ from
        $\frac ab$: then $\frac1n$ fits but $\frac{1}{n-1}$ is
        already too big, $\frac{1}{n-1} > \frac ab$. Putting both
        over the common \omterm{def:g4:fractions:def}{denominator} $b \times (n - 1)$, deduce
        that $b > a \times n - a$, hence that the new \omterm{def:g4:fractions:def}{numerator}
        satisfies
        \[
        a \times n - b < a .
        \]
        Explain why this proves that the method always stops.
  \item An old tale: a father dies leaving $17$ camels, willing
        $\frac12$ of the herd to his eldest child, $\frac13$ to
        the second, $\frac19$ to the third --- and none of the
        shares is a whole number of camels. A neighbour lends the
        family an eighteenth camel; the children take $9$, $6$ and
        $2$ camels, and the borrowed camel is returned. Compute
        $\frac12 + \frac13 + \frac19$ and explain the trick: why
        could the herd be shared in whole camels, and why did no
        child receive less than the will promised?
  \item Repair the will: propose three \emph{distinct} unit
        \omterm{def:g6:fractions:def}{fractions} that add up to exactly $1$, and give the
        smallest herd size for which all three shares are whole
        numbers of camels.
\end{enumerate}
\end{problem}
