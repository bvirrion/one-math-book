\chapter{Multiplicando números negativos}\label{ch:g8:negprod}

Grau 7 adicionado e subtraído números relativos
(\cref{ch:g7:negatives}); resta multiplicá-los e dividi-los. Um
questão domina o capítulo: por que “menos vezes menos” deveria ser
mais? Damos uma razão, não apenas uma regra.

\section{As regras do sinal}

\begin{theorem}[Sinal de um produto]\label{thm:g8:negprod:rules}
O \omterm{def:g2:mult:def}{produto} de dois números relativos tem distância para \omterm{def:g1:counting:zero}{zero} igual ao
\omterm{def:g2:mult:def}{produto} das distâncias, e seu sinal é dado por:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|cc}
$\times $ & $+$ & $-$ \\
\hline
$+$ & $+$ & $-$ \\
$-$ & $-$ & $+$ \\
\end{tabular}
\end{center}
\emph{Mesmos sinais: positivo \omterm{def:g2:mult:def}{produto}. Sinais opostos: negativo \omterm{def:g2:mult:def}{produto}.}
A mesma tabela rege \omterm{def:g3:division:remainder}{quocientes}.
\end{theorem}

\begin{proof}[Por que $(-1) \times (-1) = +1$]
Primeiro, $3 \times (-4)$ significa $(-4) + (-4) + (-4) = -12$: um positivo
vezes um negativo é negativo. Agora observe o padrão:
\[
3 \times (-4) = -12, \quad
2 \times (-4) = -8, \quad
1 \times (-4) = -4, \quad
0 \times (-4) = 0 .
\]
A cada passo, o primeiro fator cai $1$ e o resultado \emph{sobe}
por $4$. Continuando o padrão mais um passo:
\[
(-1) \times (-4) = 0 + 4 = +4 .
\]
Qualquer outra escolha quebraria a regularidade da aritmética
(distributividade). Então menos vezes menos deve ser mais.
\end{proof}

\begin{example}\label{ex:g8:negprod:products}
\begin{align*}
(-7) \times (+6) &= -42 && \text{(opposite signs)} \\
(-5) \times (-8) &= +40 && \text{(same signs)} \\
(+2.5) \times (-4) &= -10 \\
(-36) \div (-9) &= +4 \\
(+15) \div (-2) &= -7.5 .
\end{align*}
\end{example}

\section{Produtos de vários fatores}

\begin{proposition}[Sinal de um produto longo]\label{prop:g8:negprod:many}
O sinal de um \omterm{def:g2:mult:def}{produto} de vários fatores diferentes de \omterm{def:g1:counting:zero}{zero} depende apenas da
número de fatores negativos:
\begin{itemize}
  \item um \emph{\omterm{def:g3:numbers:evenodd}{par}} número de fatores negativos dá um resultado positivo
        \omterm{def:g2:mult:def}{produto};
  \item um \emph{\omterm{def:g3:numbers:evenodd}{ímpar}} número dá um negativo \omterm{def:g2:mult:def}{produto}.
\end{itemize}
\end{proposition}

\begin{proof}
Os fatores negativos formam \omterm{def:g3:numbers:evenodd}{pares} e cada \omterm{def:g3:numbers:evenodd}{par} contribui com um sinal positivo
(\cref{thm:g8:negprod:rules}); um \omterm{def:g3:numbers:evenodd}{ímpar} contagem deixa um desemparelhado
fator negativo, o que torna o todo \omterm{def:g2:mult:def}{produto} negativo.
\end{proof}

\begin{example}\label{ex:g8:negprod:many}
$(-2) \times (+3) \times (-5) \times (-1)$: três fatores negativos
(\omterm{def:g3:numbers:evenodd}{ímpar}), então o \omterm{def:g2:mult:def}{produto} é negativo; distâncias: $2 \times 3 \times 5
\times 1 = 30$. Resultado: $-30$. Decida o sinal \emph{primeiro}, então
multiplique as distâncias --- duas tarefas fáceis em vez de uma propensa a erros
um.
\end{example}

\begin{example}[Poderes dos negativos]\label{ex:g8:negprod:powers}
$(-2)^2 = (-2) \times (-2) = +4$, e
$(-2)^3 = (-2)^2 \times (-2) = -8$: \omterm{def:g3:numbers:evenodd}{par} expoentes dão positivo
valores, \omterm{def:g3:numbers:evenodd}{ímpar} expoentes mantêm o sinal. Cuidado com a notação:
$(-3)^2 = 9$ mas $-3^2 = -(3 \times 3) = -9$.
\end{example}

\section{Computando com todas as quatro operações}

\begin{method}[Cálculo seguro com números relativos]\label{met:g8:negprod:compute}
\begin{enumerate}
  \item Respeite as prioridades de \cref{ch:g7:priorities}: colchetes,
        então $\times $ e $\div $, então $+$ e $-$;
  \item em cada multiplicação ou divisão, encontre o \emph{assinar
        primeiro}, então a distância;
  \item reescreva um passo por \omterm{def:g6:lines:objects}{linha}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:negprod:compute}
\begin{align*}
5 - 3 \times (-4)
&= 5 - (-12) && \text{(product first: opposite signs)}\\
&= 5 + 12 = 17 .
\end{align*}
\begin{align*}
\frac{(-6) \times (-10)}{-4}
&= \frac{+60}{-4} && \text{(numerator: same signs)}\\
&= -15 && \text{(quotient: opposite signs).}
\end{align*}
\end{example}

\begin{example}[Substituindo valores negativos]\label{ex:g8:negprod:substitute}
Avaliar $ E = 3x^2 - 2x $ para $ x = -5$, colchetes ao redor do valor:
\[
E = 3 \times (-5)^2 - 2 \times (-5)
= 3 \times 25 - (-10)
= 75 + 10 = 85 .
\]
Os colchetes em $(-5)^2$ são essenciais: sem eles a praça seria
aplicar para $5$ apenas.
\end{example}

\end{exercise}

\end{exercise}

\section{Problema: Por que menos vezes menos é mais}

\begin{problem}[{Problema de fim de semana --- as regras de sinalização deduzidas de
distributividade e a soma alternada $1 - 2 + 3 - 4 + \dots $}]
\label{pb:g8:negprod:1}
A prova de \cref{thm:g8:negprod:rules} continuou um padrão e
afirmou que “qualquer outra escolha quebraria a distributividade”. Isto
O problema torna a afirmação exata: começando apenas pela distributividade
(\cref{thm:g7:priorities:distributivity}), você vai \emph{provar} o
regras de sinais --- sem imagem, sem padrão, do jeito que a álgebra faz ---
em seguida, colocá-los para trabalhar em poderes de $-1$ e em um famoso \omterm{def:g1:addition:def}{soma} com
sinais alternados. Por todo, $ a $ e $ b $ são números relativos; nós
tome como certo as regras de adição de \cref{ch:g7:negatives}, que
$1 \times a = a $, e que um \omterm{def:g2:mult:def}{produto} pode ser calculado em qualquer ordem
(então a distributividade se aplica em ambos os lados de um \omterm{def:g2:mult:def}{produto}).

\medskip
\noindent\textbf{Part I --- The sign rules are theorems.}
\begin{enumerate}
  \item Prove isso $0 \times a = 0$ para cada \omterm{def:g7:negatives:def}{número relativo} $ a $.
        (Escrever $0 = 0 + 0$, expandir $(0 + 0) \times a $ e pergunte qual
        número, adicionado a $0 \times a $, dá $0 \times a $ voltar.)
  \item Expandir $\bigl(1 + (-1)\bigr) \times a $ para mostrar isso
        $(-1) \times a + a = 0$, e concluir que
        \[
        (-1) \times a = -a :
        \]
        multiplicando por $-1$ dá o oposto. Compare com
        \cref{exo:g8:negprod:11}, que é o caso $ a = -1$.
  \item Deduza isso $(-a) \times b = -(a \times b)$: um sinal de menos
        sai de um \omterm{def:g2:mult:def}{produto} inalterado.
  \item Deduza isso $(-a) \times (-b) = a \times b $: dois sinais de menos
        cancelar.
  \item Todo \omterm{def:g7:negatives:def}{número relativo} é a sua distância até \omterm{def:g1:counting:zero}{zero} com um login
        frente: $ a = +d $ ou $ a = -d $. Explique como as questões 3 e ~ 4
        prove todos os quatro casos da tabela de sinais de
        \cref{thm:g8:negprod:rules}, distâncias incluídas.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Powers of $-1$.}
\begin{enumerate}[resume]
  \item Calcular $(-1)^2$, $(-1)^3$, $(-1)^4$, $(-1)^5$, então dê,
        com justificativa de \cref{prop:g8:negprod:many}, o
        valor de $(-1)^n $ para cada número inteiro $ n \geq 1$.
  \item Dê (sem calcular qualquer distância) o sinal de
        \[
        (-1) \times (-2) \times (-3) \times \dots \times (-10),
        \]
        e escreva a distância disso \omterm{def:g2:mult:def}{produto} como um \omterm{def:g2:mult:def}{produto} do todo
        números, sem resolver.
  \item Mostre isso $(-a)^2 = a^2$ e isso $(-a)^3 = -a^3$. Para qual
        expoentes o sinal de menos sobrevive?
  \item Use as regras de sinais para mostrar que o quadrado de um parente
        número nunca é negativo, e deduza que não \omterm{def:g7:negatives:def}{número relativo}
        $ x $ satisfaz $ x^2 = -4$.
  \item Zoe afirma: ``$(-1)^{2026} + (-1)^{2027} = 0$ e mais
        geralmente dois poderes consecutivos de $-1$ sempre cancele.''É
        ela está certa? Justificar.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The alternating \omterm{def:g1:addition:def}{soma}.}
Com as regras de sinais garantidas, podemos calcular \omterm{def:g1:addition:def}{somas} que mistura os dois
sinais. Para um número inteiro $ n \geq 1$, deixar
\[
A_n = 1 - 2 + 3 - 4 + \dots
\]
seja o \omterm{def:g1:addition:def}{soma} dos números inteiros de $1$ para $ n $ com alternância
sinais: o último termo é $+n $ quando $ n $ é \omterm{def:g3:numbers:evenodd}{ímpar}, $-n $ quando $ n $ é
\omterm{def:g3:numbers:evenodd}{par}. Então $ A_1 = 1$ e $ A_2 = 1 - 2 = -1$.
\begin{enumerate}[resume]
  \item Calcular $ A_3$, $ A_4$, $ A_5$, $ A_6$ e $ A_7$. O que você
        conjectura?
  \item Suponha $ n $ é \omterm{def:g3:numbers:evenodd}{par}. Agrupe os termos dois a dois,
        $(1 - 2) + (3 - 4) + \dots $, conte os \omterm{def:g3:numbers:evenodd}{pares} e prove que
        $ A_n = -\frac{n}{2}$.
  \item Suponha $ n $ é \omterm{def:g3:numbers:evenodd}{ímpar}. Explique por que $ A_n = A_{n-1} + n $ e
        deduza da questão anterior que
        $ A_n = \frac{n + 1}{2}$.
  \item Calcular $ A_{100}$, $ A_{101}$ e $ A_{2026}$.
  \item Alma computada $ A_{101} - A_{100} = 51 - (-50) = 101$ e foi
        surpreso ao descobrir exatamente $101$. Ela deveria estar?
        Explique em uma frase e indique o valor de
        $ A_{2027} - A_{2026}$ sem qualquer cálculo adicional.
\end{enumerate}
\end{problem}
