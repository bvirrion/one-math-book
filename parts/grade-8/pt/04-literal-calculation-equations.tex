\chapter{Cálculo Literal e Equações}\label{ch:g8:equations}

A 7ª série introduziu letras e reduziu expressões simples
(\cref{ch:g7:literal}). Este capítulo adiciona as duas habilidades que tornam
álgebra poderosa: \emph{expandindo} \omterm{def:g2:mult:def}{produtos} entre colchetes e
\emph{resolvendo \omterm{def:g8:equations:def}{equações}} --- encontrar o número escondido atrás do
carta.

\section{Expandindo}

\begin{theorem}[Distributividade simples e dupla]\label{thm:g8:equations:expand}
Para todos os números:
\[
k(a + b) = ka + kb,
\qquad
(a + b)(c + d) = ac + ad + bc + bd .
\]
\end{theorem}

\begin{proof}
A primeira regra é \cref{thm:g7:priorities:distributivity} (sinais
incluído: $k $ e os termos podem ser negativos,
\cref{ch:g8:negprod}). Para a segunda, aplique a primeira regra duas vezes,
tratando $(c + d)$ como um único número:
\[
(a + b)(c + d) = a(c + d) + b(c + d) = ac + ad + bc + bd .
\qedhere
\]
\end{proof}

\begin{example}\label{ex:g8:equations:expand}
Cuidado com os sinais em cada termo:
\begin{align*}
3(2x - 5) &= 6x - 15, \\
-2(4x - 7) &= -8x + 14
&& \text{(a minus in front flips both signs)}, \\
(x + 3)(x + 5) &= x^2 + 5x + 3x + 15 = x^2 + 8x + 15, \\
(2x - 1)(x - 4) &= 2x^2 - 8x - x + 4 = 2x^2 - 9x + 4 .
\end{align*}
\end{example}

\begin{method}[Expandir e reduzir]\label{met:g8:equations:reduce}
\begin{enumerate}
  \item Expandir cada \omterm{def:g2:mult:def}{produto}, escrevendo cada termo com seu sinal;
  \item coletar termos semelhantes ($ x^2$ com $ x^2$, $ x $ com $ x $, números
        com números);
  \item verifique com um valor: substitua, por exemplo\ $ x = 2$ no original
        e no resultado --- resultados iguais detectam a maioria dos erros.
\end{enumerate}
\end{method}

\section{Resolvendo equações}

\begin{definition}[Equação]\label{def:g8:equations:def}
Uma \emph{equação}\index{equação} é uma igualdade com uma incógnita
número, escrito $ x $ (ou qualquer carta). \emph{Resolvendo} isso significa encontrar
todos os valores de $ x $ tornando a igualdade verdadeira --- o
\emph{soluções}.
\end{definition}

\begin{theorem}[Regras de equilíbrio]\label{thm:g8:equations:balance}
Um \omterm{def:g8:equations:def}{equação} mantém exatamente as mesmas soluções quando:
\begin{enumerate}
  \item o mesmo número é adicionado (ou subtraído) de ambos os lados;
  \item ambos os lados são multiplicados (ou divididos) pelo mesmo
        \emph{diferente de \omterm{def:g1:counting:zero}{zero}} número.
\end{enumerate}
\end{theorem}
\admitted

\begin{method}[Resolvendo uma equação de primeiro grau]\label{met:g8:equations:solve}
\begin{enumerate}
  \item Expanda e reduza cada lado, se necessário;
  \item adicione ou subtraia em ambos os lados para reunir o $ x $-termos em um
        lado, os números simples do outro;
  \item reduza e divida ambos os lados pelo coeficiente de $ x $;
  \item \emph{verificar} substituindo o valor encontrado no original
        \omterm{def:g8:equations:def}{equação}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:equations:solve}
Resolver $7x - 5 = 4x + 16$:
\begin{align*}
7x - 5 &= 4x + 16 \\
7x - 4x &= 16 + 5
&& \text{(add $5$, subtract $4x $, on both sides)} \\
3x &= 21 \\
x &= 7 && \text{(divide both sides by 3).}
\end{align*}
Verifique: lado esquerdo $7 \times 7 - 5 = 44$; lado direito
$4 \times 7 + 16 = 44$. A solução é $7$.
\end{example}

\begin{example}[Com colchetes e negativos]\label{ex:g8:equations:brackets}
Resolver $3(x - 4) = 5 - 2(x + 1)$:
\begin{align*}
3x - 12 &= 5 - 2x - 2 && \text{(expand both sides)} \\
3x - 12 &= 3 - 2x && \text{(reduce)} \\
5x &= 15 && \text{(add $2x + 12$ to both sides)} \\
x &= 3 .
\end{align*}
Verificar: $3(3 - 4) = -3$ e $5 - 2(3+1) = 5 - 8 = -3$. \quad Solução:
$3$.
\end{example}

\begin{method}[Resolvendo um problema com uma equação]\label{met:g8:equations:problem}
\begin{enumerate}
  \item Escolha o desconhecido: ``deixe $ x $ ser \dots'' (a quantidade pedida
        para, ou um relacionado mais simples);
  \item traduzir a história em um \omterm{def:g8:equations:def}{equação};
  \item resolva;
  \item interpretar: responda à pergunta original em uma frase, com
        unidades e compare a resposta com a história.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:equations:problem}
Lucas é $4$ anos mais velho que sua irmã; juntos, suas idades somam
$26$. Deixar $ x $ ter a idade da irmã; então Lucas é $ x + 4$, e
\[
x + (x + 4) = 26
\ \Longrightarrow\
2x + 4 = 26
\ \Longrightarrow\
2x = 22
\ \Longrightarrow\
x = 11 .
\]
A irmã é $11$, Lucas é $15$. Verificar: $11 + 15 = 26$ e
$15 - 11 = 4$.
\end{example}

\section{Ordem e desigualdades}

\begin{proposition}[Operações e ordem]\label{prop:g8:equations:order}
Adicionar o mesmo número a ambos os lados de uma desigualdade preserva-a;
multiplicando ambos os lados por um \emph{positivo} o número o preserva;
multiplicando por um \emph{negativo} número \emph{inverte} isto:
\[
2 < 5 \ \text{ but }\ (-1) \times 2 > (-1) \times 5 .
\]
\end{proposition}
\admitted

\begin{example}\label{ex:g8:equations:inequality}
Resolver $5 - 2x \leq 11$: subtrair $5$ em ambos os lados, $-2x \leq 6$;
dividir por $-2$ \emph{e reverter}: $ x \geq -3$. Todos os números
maior ou igual a $-3$ são soluções --- um \omterm{def:g8:equations:def}{equação} normalmente
tem uma solução, uma desigualdade tem uma meia-reta inteira delas.
\end{example}

\section{Problema: Identidades notáveis e a escada dos
ímpares}

\begin{problem}[{Problema de fim de semana --- $(a+b)^2$, $(a-b)^2$,
$(a+b)(a-b)$, e a soma $1 + 3 + 5 + \dots + (2n-1) = n^2$}]
\label{pb:g8:equations:1}
Três expansões surgem com tanta \omterm{def:g7:stats:frequency}{frequência} que a álgebra as conhece por
coração: eles são chamados de \emph{identidades notáveis}. Isto
problema os deriva da dupla distributividade
(\cref{thm:g8:equations:expand}), transforma-os em uma aritmética mental
superpotência, usa-os para provar um belo fato sobre números \omterm{def:g3:numbers:evenodd}{ímpares},
e termina resolvendo \omterm{def:g8:equations:def}{equações} de um tipo totalmente novo: \omterm{def:g8:equations:def}{equações} com um
quadrado neles.

\medskip
\noindent\textbf{Part I --- The three identities.}
\begin{enumerate}
  \item Expandir $(a + b)^2 = (a + b)(a + b)$ e reduzir, para provar
        \[
        (a + b)^2 = a^2 + 2ab + b^2 .
        \]
  \item Prove da mesma forma os dois companheiros:
        \[
        (a - b)^2 = a^2 - 2ab + b^2,
        \qquad
        (a + b)(a - b) = a^2 - b^2 .
        \]
  \item Desenhe um quadrado de lado $ a + b $, divida cada lado em $ a $
        e $ b $ e corte o quadrado em quatro pedaços. Qual peça
        da primeira identidade cada parte da imagem
        representa?
  \item Aritmética mental: use as identidades para calcular, sem
        colocando qualquer multiplicação,
        \[
        101^2, \qquad 99^2, \qquad 49 \times 51 .
        \]
  \item Uma armadilha clássica: Zoe escreve ``$(a + b)^2 = a^2 + b^2$''.
        Teste sua fórmula com $ a = 3$, $ b = 4$ e explique no
        imagem da pergunta 3 exatamente quais peças sua fórmula
        esquece.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The staircase of números \omterm{def:g3:numbers:evenodd}{ímpares}.}
\begin{enumerate}[resume]
  \item Calcular $1$, então $1 + 3$, então $1 + 3 + 5$, então
        $1 + 3 + 5 + 7$, então $1 + 3 + 5 + 7 + 9$. O que você
        aviso sobre os resultados?
  \item Explique por que $ n $-º número \omterm{def:g3:numbers:evenodd}{ímpar} é $2n - 1$ (verificar:
        $ n = 1, 2, 3$ dar $1, 3, 5$).
  \item Use uma identidade da Parte I para expandir e reduzir
        $(n + 1)^2 - n^2$. O que o resultado tem a ver com
        questão 7?
  \item A questão 8 diz: passar de um quadrado de lado $ n $ para um
        quadrado de lado $ n + 1$, adiciona-se exatamente o próximo número \omterm{def:g3:numbers:evenodd}{ímpar} de células unitárias --- uma borda em forma de L ao longo de duas
        lados e o canto. A partir de $1 = 1^2$ e crescendo
        o quadrado, uma camada de cada vez, explique por que
        \[
        1 + 3 + 5 + \dots + (2n - 1) = n^2 :
        \]
        o \omterm{def:g1:addition:def}{soma} do primeiro $ n $ números \omterm{def:g3:numbers:evenodd}{ímpares} é o $ n $-ésimo quadrado.
  \item Deduza, sem somar um por um: o valor de
        $1 + 3 + 5 + \dots + 99$; e quantos números \omterm{def:g3:numbers:evenodd}{ímpares}, começando
        de $1$, some exatamente $400$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- \omterm{def:g8:equations:def}{Equações} with a square.}
\begin{enumerate}[resume]
  \item Reconhecer uma identidade notável em $ x^2 + 6x + 9$ e
        resolver o \omterm{def:g8:equations:def}{equação} $ x^2 + 6x + 9 = 0$.
  \item Explique, usando a regra de sinal e distância de
        \cref{thm:g8:negprod:rules}, por que um \omterm{def:g2:mult:def}{produto} de dois números
        pode ser \omterm{def:g1:counting:zero}{zero} \emph{somente se} um dos dois fatores é \omterm{def:g1:counting:zero}{zero}.
        Use isso para resolver $(x - 2)(x + 5) = 0$.
  \item Resolver $ x^2 = 49$ movendo tudo para um lado e
        factoring $ x^2 - 49$ com a terceira identidade. Quantos
        soluções faz isso \omterm{def:g8:equations:def}{equação} tem --- e como isso
        diferem de cada \omterm{def:g8:equations:def}{equação} resolvido neste capítulo até agora?
  \item Em \cref{exo:g8:equations:3} você verificou isso $ x = 3$ é
        uma solução de $ x^2 = 6x - 9$. Mostre que é o
        \emph{apenas} um: colocar tudo de lado, reconhecer
        uma identidade e concluir.
  \item Final: calcular $2026^2 - 2025^2$ na sua cabeça e
        explicar com uma identidade por que \omterm{ex:g1:subtraction:difference}{diferença} de dois
        quadrados consecutivos é sempre o \omterm{def:g1:addition:def}{soma} dos dois números
        --- que é a questão 8 novamente.
\end{enumerate}
\end{problem}
