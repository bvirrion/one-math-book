\chapter{O triângulo retângulo e o cosseno}\label{ch:g8:cosine}

Dois belos fatos unem o triângulo retângulo para o círculo: a certo
triângulo se ajusta exatamente a um semicírculo, sendo sua hipotenusa um
\omterm{def:g4:geometry:circle}{diâmetro}. E um número, o \emph{cosseno} de um ângulo, codifica o
forma de cada triângulo retângulo com esse ângulo --- o primeiro
razão trigonométrica, à frente de seus irmãos seno e tangente
(\cref{ch:g9:trig}).

\section{O triângulo retângulo e seu círculo}

\begin{theorem}[Teorema do círculo]\label{thm:g8:cosine:circle}
\begin{enumerate}
  \item Se um triângulo $ABC $ é \omterm{def:g6:shapes:triangles}{em ângulo reto} no $ A $, então $ A $ mente
        o círculo cujo \omterm{def:g4:geometry:circle}{diâmetro} é a hipotenusa $[BC]$.
  \item Por outro lado, se $ A $ encontra-se em um círculo de \omterm{def:g4:geometry:circle}{diâmetro} $[BC]$ (com
        $ A \neq B, C $), então o triângulo $ ABC $ é \omterm{def:g6:shapes:triangles}{em ângulo reto} no
        $ A $.
\end{enumerate}
Equivalentemente: em um triângulo retângulo, o ponto médio da hipotenusa é
à mesma distância dos três \omterm{def:g5:solids:def}{vértices} --- a mediana do
\omterm{def:g3:shapes:rightangle}{ângulo reto} medidas \omterm{def:g3:division:half}{metade} a hipotenusa.
\end{theorem}

\begin{proof}[Prova de 1]
Deixe $ O $ ser o ponto médio de $[BC]$ e $ D $ o \omterm{def:g7:central:def}{simétrico} de $ A $ sobre
$ O $. As diagonais de $ ABDC $ cortado em seu ponto médio comum $ O $, então
$ ABDC $ é um \omterm{def:g7:central:parallelogram}{paralelogramo} (\cref{def:g7:central:parallelogram}) ---
com um \omterm{def:g3:shapes:rightangle}{ângulo reto} no $ A $: é um retângulo. As diagonais de um
retângulo são iguais, então $ OA = \frac{AD}{2} = \frac{BC}{2}$: o
ponto $ A $ está à distância $\frac{BC}{2}$ de $ O $, ou seja,\ no círculo
de \omterm{def:g4:geometry:circle}{diâmetro} $[BC]$. (O ponto 2 é provado executando o argumento
para trás.)
\end{proof}

o ponto médio da hipotenusa.
\end{example}

\section{O cosseno de um ângulo agudo}

\begin{definition}[Cosseno]\label{def:g8:cosine:def}
Em um triângulo retângulo, para um ângulo agudo $\theta $:
\[
\cos\theta = \frac{\text{side adjacent to } \theta}{\text{hypotenuse}}
\]
--- a perna tocando $\theta $, dividido pela hipotenusa. Esta proporção
depende apenas do ângulo, não do tamanho do triângulo: todos certo
triângulos com o mesmo ângulo agudo são ampliações um do outro,
e ampliações preservam proporções de comprimentos
(\cref{ch:g8:midpoints} começou esta história; \cref{ch:g9:thales}
termina).
\end{definition}

{\small Dois triângulos retângulos compartilhando o ângulo $\theta $: o pequeno
é uma redução do grande, então
$\frac{\text{adjacent}}{\text{hypotenuse}}$ é o mesmo para ambos ---
esse valor comum é $\cos\theta $.}
\end{omfigure}

\begin{proposition}[Primeiros valores e limites]\label{prop:g8:cosine:bounds}
Para cada ângulo agudo $\theta $: $0 < \cos\theta < 1$, e o cosseno
\emph{diminui} à medida que o ângulo se abre: um ângulo mais amplo tem um menor
cosseno. Marcos: $\cos 0^\circ = 1$, $\cos 60^\circ = \frac12$,
$\cos 90^\circ = 0$ (os valores extremos correspondentes ao achatado
triângulos).
\end{proposition}
\admitted

\begin{method}[Usando o cosseno]\label{met:g8:cosine:use}
Em um triângulo retângulo, quando as quantidades conhecidas e desejadas são um
ângulo agudo, seu lado adjacente e a hipotenusa:
\begin{enumerate}
  \item escreva a definição:
        $\cos\theta = \frac{\text{adjacent}}{\text{hypotenuse}}$ com
        os valores conhecidos em vigor;
  \item resolver o pequeno \omterm{def:g8:equations:def}{equação} para o desconhecido (multiplicar ou dividir);
  \item para um desconhecido \emph{ângulo}, aplique a calculadora
        $\cos^{-1}$ à proporção calculada;
  \item verificações de sanidade: um comprimento deve ser menor que o
        hipotenusa; um ângulo estritamente entre $0^\circ $ e
        $90^\circ $.
\end{enumerate}
\end{method}

\begin{example}[Encontrando um lado]\label{ex:g8:cosine:side}
UM $6$ m rampa faz um ângulo de $20^\circ $ com o terreno horizontal.
Distância horizontal percorrida (adjacente a $20^\circ $, hipotenusa
$6$ m):
\[
\cos 20^\circ = \frac{d}{6}
\quad\Longrightarrow\quad
d = 6 \cos 20^\circ \approx 6 \times 0.940 \approx 5.6 \text{ m}.
\]
\end{example}

\begin{example}[Encontrando um ângulo]\label{ex:g8:cosine:angle}
Em um triângulo retângulo, o lado adjacente ao ângulo $\theta $ medidas
$3.5$ e a hipotenusa $5$:
\[
\cos\theta = \frac{3.5}{5} = 0.7,
\qquad
\theta = \cos^{-1}(0.7) \approx 45.6^\circ .
\]
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g8:cosine:1}
UM triângulo retângulo tem hipotenusa de $13$ cm. Qual é o comprimento
a mediana do \omterm{def:g3:shapes:rightangle}{ângulo reto} vértice? Qual é o \omterm{def:g3:shapes:shapes}{raio} do
circule pelos três \omterm{def:g5:solids:def}{vértices}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:cosine:2}
Desenhe um círculo de \omterm{def:g4:geometry:circle}{diâmetro} $8$ cm com um \omterm{def:g4:geometry:circle}{diâmetro} $[BC]$, escolha qualquer
ponto $ A $ no círculo e desenhe $ ABC $. Qual ângulo está certo? Citar
o teorema. Medir $ AB $ e $ AC $ e verifique Pitágoras.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:cosine:3}
Em um triângulo $ DEF $ \omterm{def:g6:shapes:triangles}{em ângulo reto} no $ D $, nomeie o lado adjacente a
o ângulo $\widehat E $, o lado oposto e a hipotenusa.
Escrever $\cos \widehat E $ como uma proporção.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:cosine:4}
Calcule a quantidade que falta ($\cos 35^\circ \approx 0.819$,
$\cos 50^\circ \approx 0.643$):
\begin{enumerate}
  \item hipotenusa $10$, ângulo $35^\circ $: lado adjacente?
  \item lado adjacente $6$, ângulo $50^\circ $: hipotenusa?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:cosine:5}
Em um triângulo retângulo, o lado adjacente a $\theta $ medidas $8$ e
a hipotenusa $10$. Calcular $\cos\theta $, então $\theta $
($\cos^{-1}(0.8) \approx 36.9^\circ $).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:cosine:6}
Explique por que $\cos\theta $ nunca pode igualar $1.2$ para um ângulo agudo de
um triângulo retângulo.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:cosine:7}
UM $25$ m tirolesa desce de uma plataforma até o chão, fazendo uma
ângulo de $12^\circ $ com a horizontal ($\cos 12^\circ \approx
0.978$). Que distância horizontal ela abrange?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:cosine:8}
Encomende sem calculadora: $\cos 20^\circ $, $\cos 70^\circ $,
$\cos 45^\circ $ (usar \cref{prop:g8:cosine:bounds}). Então verifique com um
calculadora.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:cosine:9}
Uma escada de comprimento $ L $ encosta-se a uma parede com um ângulo $\theta $
entre escada e \emph{chão}. Seu pé é $1.2$ estou da parede
e $\theta = 68^\circ $ ($\cos 68^\circ \approx 0.375$). Calcular $ L $,
então a altura alcançada (Pitágoras).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:cosine:10}
$[BC]$ é um \omterm{def:g4:geometry:circle}{diâmetro} de um círculo de centro $ O $ e \omterm{def:g3:shapes:shapes}{raio} $4.5$ cm,
e $ A $ é um ponto do círculo com $ AB = 5.4$ cm.
\begin{enumerate}
  \item Por que é $ ABC $ \omterm{def:g6:shapes:triangles}{em ângulo reto} no $ A $?
  \item Calcular $ AC $, então $\cos \widehat B $.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g8:cosine:11}
Usando \omterm{def:g3:division:half}{metade} um triângulo \omterm{def:g6:shapes:triangles}{equilátero} de lado $1$ (corte ao longo de uma altura),
justifique o marco $\cos 60^\circ = \frac12$ de
\cref{prop:g8:cosine:bounds}. (Onde fica o pé da altura
cair na \omterm{def:g8:powers:def}{base}? A versão trigonométrica completa é
\cref{ex:g9:trig:special}.)
\end{exercise}

\section{Problema: as relações de Euclides e Pitágoras por toda parte
de novo}

\begin{problem}[{Problema de fim de semana --- a altitude da direita
triângulo: $ AB^2 = BH \times BC $, $ AH^2 = BH \times HC $, um segundo
prova de Pitágoras e uma máquina para construir raízes quadradas}]
\label{pb:g8:cosine:1}
Solte, do \omterm{def:g3:shapes:rightangle}{ângulo reto} de um triângulo retângulo, o \omterm{def:g6:lines:perp}{perpendicular}
para a hipotenusa: este curto \omterm{def:g6:lines:objects}{segmento} --- o \emph{altitude} ---
satisfaz relações tão úteis que Euclides as colocou no centro de
dele \emph{Elementos}. Neste problema a definição do cosseno
propriedade, ``a proporção depende apenas do ângulo''
(\cref{def:g8:cosine:def}), comprova todas elas; no caminho você
irá reprovar o teorema de Pitágoras por um caminho completamente diferente,
e termina com uma máquina de régua e compasso que constrói
$\sqrt2$, $\sqrt5$, $\sqrt{n}$ para cada número inteiro $ n $.

Por todo, $ ABC $ é um triângulo \omterm{def:g6:shapes:triangles}{em ângulo reto} no $ A $, e $ H $ é
o pé do \omterm{def:g6:lines:perp}{perpendicular} de $ A $ para a hipotenusa $[BC]$,
para que $ H $ fica entre $ B $ e $ C $ e $ AH \perp BC $.

\medskip
\noindent\textbf{Part I --- One angle, two triangles.}
\begin{enumerate}
  \item Usando o ângulo \omterm{def:g1:addition:def}{soma} de um triângulo
        (\cref{thm:g7:triangles:sum}) em $ ABH $ e em $ ABC $, mostrar
        isso $\widehat{BAH} = \widehat{ACB}$: os cortes de altitude
        o triângulo retângulo em dois triângulos menores carregando
        o \emph{mesmos ângulos} como o original.
  \item Os triângulos $ ABC $ (\omterm{def:g6:shapes:triangles}{em ângulo reto} no $ A $) e $ ABH $
        (\omterm{def:g6:shapes:triangles}{em ângulo reto} no $ H $) compartilhe o ângulo $\widehat B $. Escrever
        $\cos\widehat B $ em cada um deles e deduzir
        \cref{def:g8:cosine:def} que
        \[
        AB^2 = BH \times BC .
        \]
        (Para passar de proporções iguais para proporções iguais \omterm{def:g2:mult:def}{produtos}, multiplique
        ambos os lados por ambos \omterm{def:g4:fractions:def}{denominadores},
        \cref{thm:g8:equations:balance}.)
  \item Enuncie e prove a relação gêmea no vértice $ C $:
        \[
        AC^2 = CH \times CB .
        \]
  \item Pegue o $3$--$4$--$5$ triângulo: $ AB = 3$, $ AC = 4$,
        $ BC = 5$. Calcular $ BH $ e $ CH $ das questões 2 e 3,
        e verifique isso $ BH + HC = BC $.
  \item Em geral, expresso $ BH $ e $ CH $ como frações envolvendo
        apenas os três lados. Em que sentido o pé $ H $
        dividir a hipotenusa ``\omterm{def:g6:propdata:def}{proporcionalmente} aos quadrados de
        as pernas''?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Pythagoras again, and the altitude.}
\begin{enumerate}[resume]
  \item Adicione as relações das questões 2 e 3 e use
        $ BH + HC = BC $ para obter
        \[
        AB^2 + AC^2 = BC^2 :
        \]
        Teorema de Pitágoras (\cref{thm:g8:pythagoras:direct}),
        re-provado --- sem \omterm{def:g6:measure:area}{área} quebra-cabeça à vista (compare
        \cref{exo:g8:pythagoras:11}).
  \item Aplique Pitágoras no pequeno triângulo $ ABH $ escrever
        $ AH^2 = AB^2 - BH^2$, substituir $ AB^2$ por
        $ BH \times BC $ e fatorar $ BH $ provar
        \emph{Relação de altitude de Euclides}:
        \[
        AH^2 = BH \times HC .
        \]
  \item Calcule o \omterm{def:g6:measure:area}{área} de $ ABC $ de duas maneiras --- pernas como \omterm{def:g8:powers:def}{base}
        e altura, então a hipotenusa como \omterm{def:g8:powers:def}{base}
        (\cref{thm:g7:areas:triangle}) --- e deduzir o terceiro
        relação:
        \[
        AB \times AC = BC \times AH .
        \]
  \item De volta ao $3$--$4$--$5$ triângulo: calcular $ AH $ com
        questão 8, então verifique a relação de altitude da questão
        7 numericamente, usando os valores de $ BH $ e $ CH $ encontrado em
        questão 4.
  \item O pé da altitude divide a hipotenusa de alguns certo
        triângulo em \omterm{def:g6:lines:objects}{segmentos} $ BH = 2$ centímetros e $ HC = 8$ cm.
        Calcule a altitude $ AH $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- A machine for square roots.}
Desenhe um \omterm{def:g6:lines:objects}{segmento} $[BC]$ feito de duas peças colocadas de ponta a ponta:
$ BH = p $ e $ HC = q $. Desenhe o semicírculo de \omterm{def:g4:geometry:circle}{diâmetro} $[BC]$,
e deixe $ A $ ser o ponto onde \omterm{def:g6:lines:perp}{perpendicular} para $(BC)$ no $ H $
atende.
\begin{enumerate}[resume]
  \item Usando \cref{thm:g8:cosine:circle}, explique por que
        triângulo $ ABC $ é \omterm{def:g6:shapes:triangles}{em ângulo reto} no $ A $ --- então pergunta 7
        se aplica e
        \[
        AH^2 = p \times q .
        \]
        (O comprimento $ AH $ é chamado de \emph{geométrico
        significa}\index{geométrico
        significa} de $ p $ e $ q $.)
  \item Pegar $ p = 1$ e $ q = 2$. O que é $ AH^2$? Qual famoso
        comprimento de \cref{ex:g8:pythagoras:diagonal} tem o seu
        figura de régua e compasso recém construída?
  \item Descreva a receita que constrói um \omterm{def:g6:lines:objects}{segmento} de comprimento
        $\sqrt n $ para qualquer número inteiro $ n \geq 1$ e diga qual
        escolha de $ p $ e $ q $ construções $\sqrt 5$.
  \item Deixar $ O $ seja o centro do semicírculo. Explique por que
        $ AH $ nunca poderá exceder o \omterm{def:g3:shapes:shapes}{raio} $\frac{p + q}{2}$ e
        para qual posição de $ H $ os dois são iguais.
  \item Deduza a desigualdade: para todos os números positivos $ p $ e
        $ q $,
        \[
        p \times q \leq \left(\frac{p + q}{2}\right)^2,
        \]
        com igualdade exatamente quando $ p = q $. Então repreenda-o com
        nenhuma geometria: expandir
        $\left(\frac{p+q}{2}\right)^2 - p q$ e reconhecer um
        identidade notável (\cref{pb:g8:equations:1}).
\end{enumerate}
\end{problem}
