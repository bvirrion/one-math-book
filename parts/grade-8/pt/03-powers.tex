\chapter{Poderes}\label{ch:g8:powers}

Dobre uma folha de papel \omterm{def:g3:division:half}{metade} $10$ vezes (se você puder!): sua espessura
dobra a cada vez, e $10$ duplicações multiplique por
$2^{10} = 1024$. As \omterm{def:g8:powers:def}{potências} são a abreviatura da multiplicação repetida;
este capítulo estabelece a notação e suas regras, com um papel especial
para os poderes de $10$.

\section{Definição}

\begin{definition}[Potência]\label{def:g8:powers:def}
Para um número $ a $ e um número inteiro $ n \geq 1$:
\[
a^n = \underbrace{a \times a \times \dots \times a}_{n
\text{ factors}},
\]
leia ``$ a $ elevado a $ n $''; $ a $ é a \emph{base}\index{base}, $ n $ o
\emph{expoente}\index{expoente}. Nomes especiais: $ a^2$ é ``$ a $
ao quadrado'', $ a^3$ ``$ a $ ao \omterm{def:g5:solids:def}{cubo}''. Por convenção:
\[
a^1 = a,
\qquad
a^0 = 1 \ (a \neq 0),
\qquad
a^{-n} = \frac{1}{a^n}.
\]
\end{definition}

\begin{example}\label{ex:g8:powers:def}
$3^4 = 3 \times 3 \times 3 \times 3 = 81$; \quad
$10^3 = 1000$; \quad $5^{-2} = \frac{1}{25} = 0.04$; \quad
$(-2)^3 = -8$ e $(-2)^4 = +16$
(\cref{ex:g8:negprod:powers}). Atenção: $ a^n $ é \emph{não}
$ a \times n $: \ $2^5 = 32$, não $10$.
\end{example}

\section{As regras dos expoentes}

\begin{theorem}[Regras de expoentes]\label{thm:g8:powers:rules}
Para uma \omterm{def:g8:powers:def}{base} diferente de \omterm{def:g1:counting:zero}{zero} $ a $ e número inteiro \omterm{def:g8:powers:def}{expoentes} $ m $, $ n $:
\[
a^m \times a^n = a^{m+n},
\qquad
\frac{a^m}{a^n} = a^{m-n},
\qquad
\left(a^m\right)^n = a^{m \times n},
\qquad
(ab)^n = a^n b^n .
\]
\end{theorem}

\begin{proof}[Prova por contagem de fatores]
$ a^m \times a^n $ alinha-se $ m $ fatores $ a $ seguido pela $ n $ mais:
$ m + n $ no total. $\left(a^m\right)^n $ repete um bloco de $ m $ fatores
$ n $ vezes: $ mn $ fatores. $(ab)^n $ contém $ n $ cartas $ a $ e $ n $
cartas $ b $, que pode ser reagrupado. O \omterm{def:g3:division:remainder}{quociente} regra vem de
cancelando $ n $ do $ m $ fatores; com as convenções
$ a^0 = 1$ e $ a^{-n} = \frac{1}{a^n}$, permanece verdadeiro mesmo quando
$ n \geq m $.
\end{proof}

\begin{example}\label{ex:g8:powers:rules}
\[
2^3 \times 2^5 = 2^8 = 256,
\qquad
\frac{7^6}{7^4} = 7^2 = 49,
\qquad
\left(5^2\right)^3 = 5^6,
\qquad
2^4 \times 5^4 = 10^4 .
\]
Uma armadilha: as regras se aplicam a um \emph{\omterm{def:g8:powers:def}{base} comum} (ou um comum
\omterm{def:g8:powers:def}{expoente}, para o último). Nenhuma regra simplifica $2^3 \times 5^2$ ---
basta calcular: $8 \times 25 = 200$.
\end{example}

\section{Potências de dez}

\begin{proposition}[Potências de dez]\label{prop:g8:powers:ten}
Para $ n \geq 1$: $10^n = 1\underbrace{0\dots0}_{n}$, e
$10^{-n} = 0.\underbrace{0\dots0}_{n-1}1$. As regras de \omterm{def:g8:powers:def}{expoentes}
leia-se: multiplicar \omterm{def:g8:powers:def}{potências} de dez adiciona o \omterm{def:g8:powers:def}{expoentes}.
\end{proposition}

\begin{example}\label{ex:g8:powers:ten}
$10^4 \times 10^3 = 10^7$; \quad
$\dfrac{10^2}{10^5} = 10^{-3} = 0.001$. Grandes e pequenas quantidades
tornar-se legível:
\[
\text{one billion} = 10^9,
\qquad
\text{one millionth} = 10^{-6}.
\]
Combinado com decimais: $3.2 \times 10^5 = 320\,000$ e
$4.7 \times 10^{-3} = 0.0047$ (mude o \omterm{def:g4:decimals:point}{ponto decimal},
\cref{prop:g6:decimals:shift}). O uso sistemático desta escrita ---
\emph{notação científica} --- é desenvolvido em
\cref{ch:g9:fractions}.
\end{example}

\begin{example}[Ordens de magnitude]\label{ex:g8:powers:magnitude}
A luz viaja $3 \times 10^8$ EM; um ano tem cerca
$3.2 \times 10^7$ segundos. Um ano-luz é, portanto, cerca de
\[
3 \times 3.2 \times 10^{8+7} \approx 10 \times 10^{15} = 10^{16}
\text{ m}
\]
--- dez milhões de bilhões de metros. \omterm{def:g8:powers:def}{Potências} de dez tornam astronômicas
os cálculos cabem em uma \omterm{def:g6:lines:objects}{linha}.
\end{example}

\begin{method}[Simplificando uma expressão com potências]\label{met:g8:powers:simplify}
\begin{enumerate}
  \item Agrupar os fatores \omterm{def:g8:powers:def}{base} por \omterm{def:g8:powers:def}{base};
  \item aplique o \omterm{def:g8:powers:def}{expoente} regras dentro de cada \omterm{def:g8:powers:def}{base};
  \item calcule as pequenas \omterm{def:g8:powers:def}{potências} \omterm{def:g3:division:remainder}{restantes} ou deixe a resposta como
        \omterm{def:g8:powers:def}{potência} se for grande.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:powers:simplify}
\[
\frac{3^5 \times 3^{-2} \times 4^2}{3^2}
= \frac{3^{5 + (-2)}}{3^2} \times 16
= 3^{3 - 2} \times 16
= 3 \times 16 = 48 .
\]
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g8:powers:1}
Calcular:
\[
2^6, \qquad 3^3, \qquad 10^5, \qquad 1^{100}, \qquad 0.1^2, \qquad
6^0 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:powers:2}
Calcular:
\[
(-3)^2, \qquad -3^2, \qquad (-1)^{15}, \qquad (-5)^3, \qquad
\left(\tfrac{2}{3}\right)^2 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:powers:3}
Escreva como uma única \omterm{def:g8:powers:def}{potência}:
\[
7^4 \times 7^5, \qquad
\frac{2^9}{2^3}, \qquad
\left(10^3\right)^4, \qquad
5^6 \times 5^{-2}, \qquad
3^4 \times 7^4 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:powers:4}
Escreva como um \omterm{def:g6:decimals:places}{número decimal}: $10^{-2}$; \quad $4 \times 10^3$; \quad
$2.5 \times 10^{-4}$; \quad $10^0$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:powers:5}
Escreva com \omterm{def:g8:powers:def}{potência} de dez: cem mil; um décimo; dez
bilhão; $0.000\,001$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:powers:6}
Simplifique e depois calcule:
\[
\frac{10^7 \times 10^{-3}}{10^2},
\qquad
\frac{2^5 \times 2^4}{2^6},
\qquad
\left(2^2\right)^3 \times 2^{-4} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:powers:7}
Verdadeiro ou falso? Corrija os falsos.
\[
2^3 \times 2^4 = 2^{12}; \qquad
5^2 + 5^3 = 5^5; \qquad
(3^2)^4 = 3^8; \qquad
10^3 \times 10^3 = 100^3 .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:powers:8}
Um boato se espalha: no primeiro dia, três pessoas sabem disso; cada dia, cada
pessoa que sabe conta para três novas pessoas. Escreva com uma \omterm{def:g8:powers:def}{potência} o número
de \emph{novo} pessoas informadas no dia $4$ e calcule quantas pessoas
conheça o boato no final do dia 4 (incluindo os três originais).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:powers:9}
Uma folha de papel é $0.1$ mm de espessura, ou seja,\ $10^{-4}$ m. Dobrando
dobra sua espessura a cada vez.
\begin{enumerate}
  \item Expresse a espessura depois $10$ dobra como um produtoe
        calcule em centímetros ($2^{10} = 1024$).
  \item Depois $42$ dobras a espessura seria
        $2^{42} \times 10^{-4}$ m, com $2^{42} \approx 4.4 \times
        10^{12}$. Mostre que isso excede a distância Terra-Lua,
        sobre $3.8 \times 10^8$ m.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:powers:10}
Ordene do menor para o maior, sem calculadora:
\[
2^{10}, \qquad 10^3, \qquad 3^6, \qquad 5^4 .
\]
(Calcule cada um; $2^{10}$ e $10^3$ são vizinhos famosos.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g8:powers:11}
Qual é maior, $2^{100}$ ou $10^{30}$? Usar
$2^{10} = 1024 > 10^3$ comparar
$2^{100} = \left(2^{10}\right)^{10}$ com
$\left(10^3\right)^{10}$.
\end{exercise}

\section{Problema: O tabuleiro de xadrez e as potências de dois}

\begin{problem}[{Problema de fim de semana --- o geométrico \omterm{def:g1:addition:def}{soma}
$1 + 2 + 4 + \dots + 2^{n-1} = 2^n - 1$, de uma lenda famosa
números binários}]
\label{pb:g8:powers:1}
A lenda: como recompensa pela invenção do xadrez, o sábio Sissa perguntou
seu rei por um grão de trigo na primeira casa do tabuleiro,
dois no segundo, quatro no terceiro --- dobrando de quadrado para
quadrado, até o sexagésimo quarto. O rei riu de tal modéstia.
Este problema calcula o que o rei prometeu, usando o \omterm{def:g8:powers:def}{expoente}
regras de \cref{thm:g8:powers:rules}, e termina onde a história
leva secretamente: os números binários dentro de cada computador.

\medskip
\noindent\textbf{Part I --- The doubling trick.}
Para $ n \geq 1$, deixar $ S_n $ seja o número total de grãos no
primeiro $ n $ quadrados.
\begin{enumerate}
  \item Expresse o número de grãos ao quadrado $ k $ como um poder de
        $2$. Qual poder fica no quadrado $64$?
  \item Calcular $ S_1$, $ S_2$, $ S_3$, $ S_4$ e $ S_5$ e compare
        cada um com uma \omterm{def:g8:powers:def}{potência} próxima de $2$. Conjecture uma fórmula para
        $ S_n $.
  \item O \emph{truque de duplicação}: escreva o \omterm{def:g1:addition:def}{somas} $ S_n $ e
        $2 \times S_n $ um abaixo do outro, subtraia e prove
        sua conjectura:
        \[
        S_n = 1 + 2 + 4 + \dots + 2^{n-1} = 2^n - 1 .
        \]
  \item Quantos grãos o rei prometeu no total? Expresse o
        responda com uma \omterm{def:g8:powers:def}{potência} de $2$ e complete o clássico
        observação: ``todo o tabuleiro contém um grão a menos do que
        um único sexagésimo quinto quadrado faria isso.
  \item Mostre que o segundo \omterm{def:g3:division:half}{metade} do tabuleiro (quadrados $33$ para
        $64$) detém \emph{exatamente} $2^{32}$ vezes mais grãos
        como o primeiro \omterm{def:g3:division:half}{metade}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- How big is $2^{64}$?}
A comparação $2^{10} = 1024 > 10^3$ de
\cref{exo:g8:powers:11} é a chave para todas as estimativas abaixo.
\begin{enumerate}[resume]
  \item Mostre isso
        $2^{64} = 2^4 \times \left(2^{10}\right)^6 > 1.6 \times
        10^{19}$.
  \item Um grão de trigo pesa cerca de $0.05$ g, ou seja\
        $5 \times 10^{-2}$ g. Mostre que o trigo prometido pesa
        mais do que $8 \times 10^{17}$ g, e converta isso em
        toneladas ($1$ tonelada $= 10^6$ g).
  \item O mundo inteiro atualmente colhe cerca de $8 \times 10^8$
        toneladas de trigo por ano. Pelo menos quantos anos de mundo
        colheita o rei prometeu?
  \item Encontre o menor número inteiro $ n $ tal que
        $2^n > 10^6$ --- isto é, quantas duplicações são necessárias para
        passar de um milhão. (Calcular $2^{19}$ e $2^{20}$ exatamente,
        usando $2^{10} = 1024$.)
  \item Um malandro oferece um mês de salário: $1$ centavo no dia
        $1$ e, em seguida, dobre o pagamento do dia anterior a cada dia. Em que
        dia faz o \emph{diário} pague sozinho primeiro exceda um
        milhões de euros ($10^8$ centavos)? (Calcular $2^{26}$ e
        $2^{27}$ exatamente.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Binary weights.}
Um comerciante possui cinco pesos: $1$, $2$, $4$, $8$ e $16$ gramas,
um de cada. Ela coloca alguns deles em um prato de balança para
pese as mercadorias no outro prato.
\begin{enumerate}[resume]
  \item Quais pesos ela coloca para pesar $21$ g? Pesar
        $27$ g?
  \item Explique por que qualquer alvo de $16$ g ou mais \emph{deve} usar
        o $16$ peso g, e por que qualquer alvo de $15$ gramas ou menos
        deve \emph{não} use-o. (A pergunta 3 diz qual é o
        pesos $1, 2, 4, 8$ pode alcançar no máximo.) Explique por que o
        o mesmo raciocínio se repete com o próximo maior peso, em
        cada etapa.
  \item Deduza que todo alvo inteiro de $1$ para $31$ g pode ser
        pesado, e \emph{exatamente de uma maneira}: cada número
        entre $1$ e $31$ é um \omterm{def:g1:addition:def}{soma} de poderes distintos de $2$ em
        uma única maneira.
  \item O comerciante compra um sexto peso, de $32$ g. Até o que
        alvo ela pode agora pesar? Escrever $45$ g como um \omterm{def:g1:addition:def}{soma} de
        poderes distintos de $2$.
  \item Em um computador, um número de ``64 bits'' é armazenado em $64$
        quadrados, cada um segurando um $0$ ou um $1$ --- quadrado $ k $
        contribuindo $2^{k-1}$ quando detém um $1$, como o
        grãos da lenda. Usando a Parte I, explique por que o todo
        números que tal máquina pode armazenar são executados exatamente a partir de $0$ para
        $2^{64} - 1$.
\end{enumerate}
\end{problem}
