\chapter{Proporcionalidade, Velocidade e Médias}\label{ch:g8:speed}

\omterm{def:g7:prop:table}{Proporcionalidade} (\cref{ch:g7:prop}) agora encontra o mundo físico:
velocidade, taxas de fluxo, conversões de unidades --- quantidades que são
\emph{\omterm{def:g3:division:remainder}{quocientes}} de outros dois. O capítulo termina com ponderado
médias, onde proporcional pensar evita um erro clássico.

\section{Velocidade média}

\begin{definition}[Velocidade média]\label{def:g8:speed:def}
O \emph{velocidade média}\index{velocidade média} de uma viagem é o \omterm{def:g3:division:remainder}{quociente}
\[
v = \frac{d}{t}
\qquad
\left(\text{distance divided by duration}\right),
\]
em quilômetros por hora (km/h), metros por segundo (m/s), \dots\ O
a fórmula gira: $d = v \times t $ e $ t = \frac{d}{v}$.
\end{definition}

\begin{example}\label{ex:g8:speed:basic}
Um trem cobre $270$ km em $1$ h $30$ min. Primeiro converta o tempo:
$1$ h $30$ min $= 1.5$ h. Então
\[
v = \frac{270}{1.5} = 180 \text{ km/h}.
\]
Distância em $90$ km/h durante $2$ h $20$ min $= \frac73$ h:
$ d = 90 \times \frac73 = 210$ km. Hora de $35$ km em $14$ km/h:
$ t = \frac{35}{14} = 2.5$ h $= 2$ h $30$ min.
\end{example}

\begin{remark}[Minutos não são decimais]
$1$ h $30$ min é $1.5$ h, mas $1$ h $20$ min é \emph{não} $1.2$ h:
é $1 + \frac{20}{60} = \frac43 \approx 1.33$ h. Sempre converta
minutos para um \omterm{def:g6:fractions:def}{fração} de uma hora ($60$ min $= 1$ h) antes da divisão.
\end{remark}

\begin{method}[Conversão de velocidades]\label{met:g8:speed:convert}
Para converter km/h em m/s: $1$ quilômetros $= 1000$ eu e $1$ h $= 3600$ é, então
\[
v \text{ km/h} = \frac{v \times 1000}{3600} \text{ m/s}
= \frac{v}{3.6} \text{ m/s}.
\]
Para converter m/s em km/h, multiplique por $3.6$.
\end{method}

\begin{example}\label{ex:g8:speed:convert}
$90$ km/h $= \frac{90}{3.6} = 25$ EM. Um velocista correndo $100$ estou dentro
$10$ s se move em $10$ EM $= 36$ km/h.
\end{example}

\section{Quantidades quocientes}

\begin{example}[Fluxo, densidade, preço por quilo]\label{ex:g8:speed:quotients}
Velocidade tem muitos primos, todos tratados da mesma forma:
\begin{itemize}
  \item uma torneira enche $48$ eu em $4$ min: taxa de fluxo
        $\frac{48}{4} = 12$ L/min, então encher um $150$ L banheira leva
        $\frac{150}{12} = 12.5$ min;
  \item $0.6$ custos de kg de queijo $9$ euros: preço
        $\frac{9}{0.6} = 15$ euros por kg;
  \item um carro usa $6.3$ eu para $90$ km: consumo
        $\frac{6.3}{90} \times 100 = 7$ L por $100$ km.
\end{itemize}
Identifique o \omterm{def:g3:division:remainder}{quociente}, e a fórmula de três vias
($ q = \frac ab $, $ a = qb $, $ b = \frac aq $) faz o resto.
\end{example}

\section{Médias ponderadas}

\begin{definition}[Média ponderada]\label{def:g8:speed:average}
Quando os valores $ v_1, v_2, \dots $ vem com contagens (ou pesos)
$ n_1, n_2, \dots $, deles \emph{média ponderada}\index{média ponderada}
é
\[
\bar v = \frac{n_1 v_1 + n_2 v_2 + \dots}{n_1 + n_2 + \dots} :
\]
total dos valores, dividido pelo total dos pesos.
\end{definition}

\begin{example}\label{ex:g8:speed:average}
Um teste foi realizado por dois grupos: grupo 1, $12$ estudantes, média
$11$; grupo 2, $18$ estudantes, média $14$. Média do todo
aula:
\[
\bar v = \frac{12 \times 11 + 18 \times 14}{12 + 18}
= \frac{132 + 252}{30} = \frac{384}{30} = 12.8 .
\]
\emph{Não} $\frac{11 + 14}{2} = 12.5$: o grupo maior puxa o
média em direção à sua própria - os pesos são importantes.
\end{example}

\begin{example}[Velocidade média em duas pernas]\label{ex:g8:speed:twolegs}
Um ciclista anda $30$ km em $30$ km/h, então $30$ km em $15$ km/h. O
\omterm{def:g8:speed:def}{velocidade média} durante toda a viagem é \emph{não}
$\frac{30 + 15}{2} = 22.5$ km/h. Calcule com a definição:
\begin{enumerate}
  \item vezes: $\frac{30}{30} = 1$ h, então $\frac{30}{15} = 2$ h;
        total $3$ h;
  \item distância total $60$ km, então
        $ v = \frac{60}{3} = 20$ km/h.
\end{enumerate}
A perna lenta dura mais, por isso pesa mais --- uma média de velocidades
deve ser ponderado por \emph{tempo}.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g8:speed:1}
Calcule o \omterm{def:g8:speed:def}{velocidade média}: $150$ km em $2$ h; $27$ km em $45$ min;
$100$ estou dentro $12.5$ S.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:speed:2}
Calcule a distância: $2$ h em $85$ km/h; $40$ min às $90$ km/h.
Calcule o tempo: $315$ km em $90$ km/h (em horas e minutos).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:speed:3}
Converter: $72$ km/h em m/s; $5$ m/s em km/h; $108$ km/h em m/s.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:speed:4}
Uma torneira entrega $15$ L/min. Quanto tempo para preencher um $600$ L tanque? Quantos
litros em $2$ h $30$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:speed:5}
Qual é a melhor compra: $1.2$ kg de maçãs para $3.30$ euros, ou
$0.8$ kg para $2.16$ euros? Compare preços por quilograma.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:speed:6}
Uma aula de $25$ os alunos têm uma média de $12.4$ em um teste; outro
classe de $35$ os alunos têm uma média de $10$. Calcule a média de
as duas turmas juntas.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:speed:7}
O som viaja cerca de $340$ EM. Você vê um relâmpago e ouve
o trovão $6$ segundos depois. A que distância o \omterm{def:g3:shapes:shapes}{raio} caiu
(para o mais próximo $100$ m)?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:speed:8}
Um caminhante caminha $2$ h em $5$ km/h, descansa $30$ min, depois caminha $1$ h $30$
no $4$ km/h. Calcule a distância total e o \omterm{def:g8:speed:def}{velocidade média}
\emph{incluindo o resto} (distância total sobre o tempo total decorrido).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:speed:9}
Notas com coeficientes: um aluno obteve $15$ (coeficiente $3$), $9$
(coeficiente $2$) e $12$ (coeficiente $1$). Calcule o ponderado
média. Que média simples (não ponderada) ela teria e por que
eles diferem?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:speed:10}
Um carro dirige $120$ km em $80$ km/h e depois $120$ km em $120$ km/h.
\begin{enumerate}
  \item Calcule a duração de cada perna e, em seguida, o \omterm{def:g8:speed:def}{velocidade média} acabou
        toda a viagem.
  \item Explique por que a resposta é menor que $100$ km/h, o ponto médio
        dos dois velocidades.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g8:speed:11}
Duas aldeias são $18$ quilômetros de distância. Anna sai primeiro às $10{:}00$
andando em $5$ km/h em direção ao segundo; Boris sai do segundo às
ao mesmo tempo andando em $4$ km/h em direção ao primeiro. A que horas fazer
eles se encontram, e a que distância da aldeia de Anna? (Juntos eles
feche a lacuna em $5 + 4$ km/h.)
\end{exercise}

\section{Problema: A média harmônica ou por que a viagem de volta estraga
a média}

\begin{problem}[{Problema de fim de semana --- o \omterm{def:g8:speed:def}{velocidade média} mais iguais
distâncias é $\frac{2 v_1 v_2}{v_1 + v_2}$, e nunca supera o
ponto médio do velocidades}]
\label{pb:g8:speed:1}
\Cref{ex:g8:speed:twolegs} e \cref{exo:g8:speed:10} ambos terminaram
com a mesma surpresa: saia à uma velocidade, volte às
outro, e o \omterm{def:g8:speed:def}{velocidade média} terras \emph{abaixo} o ponto médio de
os dois. Este problema encontra a fórmula exata por trás da surpresa
--- um novo tipo de média, o \emph{harmônico
significa}\index{harmônico
significa} --- prova que a surpresa é uma
teorema, e o leva ao seu extremo lógico: uma recuperação que não
velocidade no mundo pode alcançar.

\medskip
\noindent\textbf{Part I --- The formula for a round trip.}
Uma viagem cobre uma distância $ d $ no velocidade $ v_1$, então o mesmo
distância $ d $ novamente em velocidade $ v_2$ (todos os números positivos).
\begin{enumerate}
  \item Expresse as duas durações e depois a duração total, como
        frações envolvendo $ d $, $ v_1$, $ v_2$.
  \item O \omterm{def:g8:speed:def}{velocidade média} de toda a jornada é
        $\bar v = \dfrac{2d}{\ \dfrac{d}{v_1} + \dfrac{d}{v_2}\ }$
        --- um de dois andares \omterm{def:g6:fractions:def}{fração}
        (\cref{ex:g8:fractions:complex}). Simplifique e prove:
        \[
        \bar v = \frac{2\, v_1 v_2}{v_1 + v_2} .
        \]
  \item Verifique a fórmula em relação aos dois cálculos citados acima:
        $ v_1 = 30$, $ v_2 = 15$ km/h
        (\cref{ex:g8:speed:twolegs}), e $ v_1 = 80$,
        $ v_2 = 120$ km/h (\cref{exo:g8:speed:10}).
  \item A distância $ d $ desapareceu da fórmula. O que
        isso significa, concretamente, para o ciclista?
  \item Agora suponha que a viagem passe o mesmo tempo
        \emph{tempo} $ T $ em cada velocidade. Mostre que o média
        velocidade é então $\frac{v_1 + v_2}{2}$, o ponto médio simples.
        Na linguagem de \cref{def:g8:speed:average}: pelo que
        quantidade deve ser uma média de velocidades sempre ser ponderado,
        e o que muda entre os dois cenários?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Harmonic against arithmetic.}
Para dois números positivos $ v_1$, $ v_2$, definir
\[
H = \frac{2\, v_1 v_2}{v_1 + v_2}
\quad\text{(their \emph{média harmônica}),}
\qquad
A = \frac{v_1 + v_2}{2}
\quad\text{(their \emph{média aritmética}).}
\]
\begin{enumerate}[resume]
  \item Calcular $ H $ e $ A $ para os \omterm{def:g3:numbers:evenodd}{pares} $(30, 15)$ e
        $(80, 120)$. Qual dos dois meios vence todas as vezes?
  \item Colocar $ A - H $ acima do comum \omterm{def:g4:fractions:def}{denominador} $2(v_1 + v_2)$,
        expandir o \omterm{def:g4:fractions:def}{numerador} com as identidades notáveis
        (\cref{pb:g8:equations:1}) e prove:
        \[
        A - H = \frac{(v_1 - v_2)^2}{2\,(v_1 + v_2)} .
        \]
  \item Deduza o teorema por trás de todas as surpresas: para positivo
        velocidades, $ H \leq A $ sempre, com igualdade exatamente quando
        $ v_1 = v_2$. Por que um quadrado no \omterm{def:g4:fractions:def}{numerador} resolver
        o assunto?
  \item Prove a identidade elegante $ H \times A = v_1 \times v_2$:
        os dois significam multiplicar de volta ao original \omterm{def:g2:mult:def}{produto}.
  \item Um carro cobre $60$ km em $40$ km/h e $60$ km em
        $60$ km/h. Preveja o \omterm{def:g8:speed:def}{velocidade média} com a fórmula,
        então confirme no longo caminho, com tempos e total
        distância.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The impossible catch-up.}
\begin{enumerate}[resume]
  \item Um ciclista entra em um $30$ corrida de km esperando a média
        $30$ km/h. Que tempo total a corrida pode levar para ela, em
        mais? Ela monta o primeiro $15$ km em $15$ km/h: quanto
        desse orçamento de tempo resta?
  \item Mostre que a média esperada tornou-se literalmente
        impossível: calcule-a \omterm{def:g8:speed:def}{velocidade média} se ela cavalgar
        segundo \omterm{def:g3:division:half}{metade} no $90$ km/h e explique por que \omterm{def:g3:numbers:evenodd}{par} um
        arbitrariamente enorme velocidade deixa ela com falta de
        $30$ km/h.
  \item Ela reduz seu objetivo para $20$ km/h. O que velocidade no
        segundo $15$ km consegue isso? Verifique sua resposta com o
        fórmula da \omterm{pb:g8:speed:1}{média harmônica}.
  \item Só uma torneira fria enche uma banheira $20$ min; uma torneira quente sozinha
        preenche-o $30$ min. O que \omterm{def:g6:fractions:def}{fração} da banheira faça as duas coisas
        torneiras juntas enchem por minuto e quanto tempo demoram?
        Compare sua resposta com $ H(20, 30)$ --- o que você
        percebeu?
  \item Um avião voa na mesma rota de ida e volta: $900$ km/h
        com o vento, $700$ km/h contra ele. Calcule o
        viagem de ida e volta velocidade médiae explique em uma frase por que
        um vento, por mais que sopre, sempre \emph{alonga} uma rodada
        viagem.
\end{enumerate}
\end{problem}
