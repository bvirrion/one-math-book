\chapter{Frações: todas as quatro operações}\label{ch:g8:fractions}

Este capítulo completa a aritmética de frações: adição e
subtração com \emph{qualquer} \omterm{def:g4:fractions:def}{denominadores}, multiplicação e ---
o recém-chegado --- divisão, que acaba sendo uma multiplicação em
disfarce. Sinais de \cref{ch:g8:negprod} venha junto para o passeio.

\section{Adicionando com qualquer denominador}

\begin{method}[Denominador comum]\label{met:g8:fractions:add}
Para adicionar ou subtrair $\frac ab $ e $\frac cd $:
\begin{enumerate}
  \item encontrar um comum múltiplo de $ b $ e $ d $ (o \omterm{def:g2:mult:def}{produto}
        $ b \times d $ sempre funciona; um menor economiza esforço);
  \item reescrever ambos frações com isso \omterm{def:g4:fractions:def}{denominador};
  \item adicione ou subtraia o \omterm{def:g4:fractions:def}{numeradores}; simplificar.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:fractions:add}
Calcular $\dfrac{5}{6} + \dfrac{3}{4}$. Um comum múltiplo de $6$ e
$4$ é $12$:
\[
\frac{5}{6} = \frac{10}{12},
\qquad
\frac{3}{4} = \frac{9}{12},
\qquad
\frac{5}{6} + \frac{3}{4} = \frac{10 + 9}{12} = \frac{19}{12}.
\]
Com sinais: $\dfrac{2}{3} - \dfrac{7}{5} = \dfrac{10}{15} -
\dfrac{21}{15} = -\dfrac{11}{15}$.
\end{example}

\section{Multiplicando frações}

\begin{theorem}[Produto de frações]\label{thm:g8:fractions:mult}
\[
\frac{a}{b} \times \frac{c}{d} = \frac{a \times c}{b \times d} .
\]
\end{theorem}

\begin{proof}[Por que, em uma foto]
Pegue $\frac34$ de $\frac25$ de um quadrado: corte o quadrado em $5$
tiras verticais e mantenha $2$; corte horizontalmente em $4$ e manter $3$
fileiras do que restou. A parte mantida é uma grade de $3 \times 2$ pequeno
células fora de $4 \times 5$: \omterm{def:g6:fractions:def}{fração} $\frac{6}{20}$.
\end{proof}

\begin{omfigure}
\begin{tikzpicture}[scale=0.68]
  \draw[gray!50, thin] (0,0) grid[xstep=1, ystep=1] (5,4);
  \fill[omDef!18] (0,0) rectangle (2,4);
  \fill[omThm!35] (0,0) rectangle (2,3);
  \draw[thick] (0,0) rectangle (5,4);
  \draw[omDef, thick] (2,0) -- (2,4);
  \draw[omThm, thick] (0,3) -- (5,3);
  \node[below, font=\small, omDef] at (1,0) {$\frac25$ kept};
  \node[right, font=\small, omThm] at (5.05,1.5) {$\frac34$ of it};
\end{tikzpicture}

{\small $\frac34 \times \frac25$: a região duplamente sombreada é
$3 \times 2 = 6$ células fora de $4 \times 5 = 20$, ou seja\
$\frac{6}{20} = \frac{3}{10}$ --- \omterm{def:g4:fractions:def}{numeradores} multiplicado, \omterm{def:g4:fractions:def}{denominadores}
multiplicado.}
\end{omfigure}

\begin{example}\label{ex:g8:fractions:mult}
Simplificar \emph{antes} multiplicando, riscando fatores comuns:
\[
\frac{7}{12} \times \frac{9}{14}
= \frac{7 \times 9}{12 \times 14}
= \frac{7 \times 3 \times 3}{3 \times 4 \times 2 \times 7}
= \frac{3}{8}
\]
(o fator $7$ e um fator $3$ cancelar entre superior e inferior).
Com sinais, as regras de \cref{thm:g8:negprod:rules} aplique primeiro:
$\left(-\frac23\right) \times \left(-\frac{5}{4}\right) =
+\frac{10}{12} = \frac56$.
\end{example}

\section{Dividindo frações}

\begin{definition}[Inverso]\label{def:g8:fractions:inverse}
O \emph{inverso}\index{inverso} de um número diferente de \omterm{def:g1:counting:zero}{zero} $ x $ é o
número que multiplicado por $ x $ dá $1$. O inverso de
$\frac ab $ é $\frac ba $, desde
$\frac ab \times \frac ba = \frac{ab}{ab} = 1$; o inverso de um
inteiro $ n $ é $\frac1n $.
\end{definition}

\begin{theorem}[Dividir significa multiplicar pelo inverso]\label{thm:g8:fractions:div}
Para $ c \neq 0$:
\[
\frac{a}{b} \div \frac{c}{d} = \frac{a}{b} \times \frac{d}{c} .
\]
\end{theorem}

\begin{proof}
Por definição, o \omterm{def:g3:division:remainder}{quociente} $ q $ de $\frac ab $ por $\frac cd $ é o
número com $ q \times \frac cd = \frac ab $. Verifique o candidato
$ q = \frac ab \times \frac dc $:
\[
\left(\frac ab \times \frac dc\right) \times \frac cd
= \frac ab \times \left(\frac dc \times \frac cd\right)
= \frac ab \times 1 = \frac ab . \qedhere
\]
\end{proof}

\begin{example}\label{ex:g8:fractions:div}
\[
\frac{3}{5} \div \frac{7}{10}
= \frac{3}{5} \times \frac{10}{7}
= \frac{3 \times 10}{5 \times 7}
= \frac{30}{35} = \frac{6}{7}.
\]
Uma verificação de sanidade: dividindo por $\frac{7}{10}$, um número menor que
$1$, deve dar um resultado \emph{maior} que $\frac35$ --- e
$\frac67 > \frac35$ de fato ($\frac{30}{35} > \frac{21}{35}$).
\end{example}

\begin{example}[Barras de fração dentro de barras de fração]\label{ex:g8:fractions:complex}
Um ``de dois andares'' é apenas uma divisão escrita verticalmente:
\[
\frac{\ \frac{2}{3}\ }{\ \frac{5}{6}\ }
= \frac{2}{3} \div \frac{5}{6}
= \frac{2}{3} \times \frac{6}{5}
= \frac{12}{15} = \frac{4}{5}.
\]
Localize o \emph{principal} barra primeiro (a mais longa) e depois aplique
\cref{thm:g8:fractions:div}.
\end{example}

\begin{method}[Cálculos mistos]\label{met:g8:fractions:mixed}
Numa mistura de expressões frações e as quatro operações:
\begin{enumerate}
  \item aplicar as prioridades habituais (\cref{ch:g7:priorities});
  \item converta cada divisão em uma multiplicação pelo \omterm{def:g8:fractions:inverse}{inverso};
  \item decida os sinais primeiro (\cref{met:g8:negprod:compute}), então
        multiplique ou encontre comum \omterm{def:g4:fractions:def}{denominadores};
  \item simplificar na primeira oportunidade.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g8:fractions:mixed}
\begin{align*}
\frac12 + \frac{2}{3} \times \frac{9}{4}
&= \frac12 + \frac{2 \times 9}{3 \times 4}
&& \text{(product first!)} \\
&= \frac12 + \frac{18}{12} = \frac12 + \frac32
= \frac{4}{2} = 2 .
\end{align*}
\end{example}

\section{Problema: frações egípcias}

\begin{problem}[{Problema de fim de semana --- cada fração é uma soma de
frações unitárias distintas}]
\label{pb:g8:fractions:1}
UM \emph{unidade \omterm{def:g6:fractions:def}{fração}} é um \omterm{def:g6:fractions:def}{fração} com \omterm{def:g4:fractions:def}{numerador} $1$. O
escribas do antigo Egito --- o papiro Rhind foi copiado por volta
$1550$~BC --- escreveu todos \omterm{def:g6:fractions:def}{fração} como um \omterm{def:g1:addition:def}{soma} de \emph{distinto}
unidade frações: nunca $\frac27 = \frac17 + \frac17$, mas
$\frac27 = \frac14 + \frac{1}{28}$. Tal \omterm{def:g1:addition:def}{soma} é chamado de
\emph{Escrita egípcia} do \omterm{def:g6:fractions:def}{fração}. Este problema desenvolve o
técnicas de adição e comparação deste capítulo
(\cref{met:g8:fractions:add}) em um método --- publicado por
Fibonacci em $1202$ --- que produz uma escrita egípcia de qualquer
\omterm{def:g6:fractions:def}{fração} entre $0$ e $1$ e termina com um célebre
quebra-cabeça de herança.

\medskip
\noindent\textbf{Part I --- Unit frações and the splitting
identity.}
\begin{enumerate}
  \item Calcular $\frac12 + \frac13$ e $\frac12 + \frac14$ e
        deduzir escritos egípcios de $\frac56$ e $\frac34$.
  \item Verifique a entrada dos escribas para $\frac27$: mostre isso
        $\frac14 + \frac{1}{28} = \frac27$.
  \item Verifique isso $\frac13 + \frac16 = \frac12$ e isso
        $\frac14 + \frac{1}{12} = \frac13$. Então prove o
        \emph{divisão de identidade}: para cada número inteiro
        $ n \geq 1$,
        \[
        \frac{1}{n} = \frac{1}{n+1} + \frac{1}{n(n+1)} .
        \]
  \item Deduza uma escrita egípcia de $1$ em si, como um \omterm{def:g1:addition:def}{soma} de
        três unidades distintas frações.
  \item Divida o último termo de $\frac56 = \frac12 + \frac13$ para
        obter um \emph{segundo} escrita egípcia de $\frac56$, com
        três unidades frações. Explique por que não \omterm{def:g6:fractions:def}{fração} tem um único
        escrita egípcia.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Fibonacci's greedy method.}
O método é: subtrair o \emph{maior} unidade \omterm{def:g6:fractions:def}{fração} isso
cabe e repita com o que resta.
\begin{enumerate}[resume]
  \item Usando comum \omterm{def:g4:fractions:def}{denominadores}, mostre que
        $\frac14 < \frac27 < \frac13$. Por que isso prova que
        $\frac14$ é a maior unidade \omterm{def:g6:fractions:def}{fração} menor que
        $\frac27$?
  \item Calcular $\frac27 - \frac14$. Qual escrita de $\frac27$
        o método ganancioso produz?
  \item Agora pegue $\frac57$. Mostre que a maior unidade \omterm{def:g6:fractions:def}{fração}
        menor que $\frac57$ é $\frac12$ e calcular
        $\frac57 - \frac12$.
  \item Mostre que a maior unidade \omterm{def:g6:fractions:def}{fração} menor que
        $\frac{3}{14}$ é $\frac15$ (comparar $\frac{3}{14}$ com
        $\frac14$ e com $\frac15$), calcular
        $\frac{3}{14} - \frac15$ e conclua:
        \[
        \frac57 = \frac12 + \frac15 + \frac{1}{70} .
        \]
        Verifique esta escrita diretamente com o comum \omterm{def:g4:fractions:def}{denominador}
        $70$.
  \item Execute o método ganancioso em $\frac45$, verificando a cada passo
        que sua unidade \omterm{def:g6:fractions:def}{fração} é o maior que cabe, e
        verificar a redação final.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Why the method always stops, and the
eighteenth camel.}
\begin{enumerate}[resume]
  \item Nas questões 7 a 10 o método foi aplicado para $\frac27$,
        $\frac57$ e $\frac45$. Para cada um, liste os \omterm{def:g4:fractions:def}{numeradores} de
        o sucessivo restos (antes de qualquer simplificação).
        O que você observa?
  \item Por um \omterm{def:g6:fractions:def}{fração} $\frac ab $ e um número inteiro $ n $, mostrar
        isso
        \[
        \frac{a}{b} - \frac{1}{n}
        = \frac{a \times n - b}{b \times n} .
        \]
  \item Suponha que o método ganancioso subtraia $\frac1n $ de
        $\frac ab $: então $\frac1n $ cabe, mas $\frac{1}{n-1}$ é
        já é muito grande, $\frac{1}{n-1} > \frac ab $. Colocando ambos
        acima do comum \omterm{def:g4:fractions:def}{denominador} $ b \times (n - 1)$, deduzir
        isso $ b > a \times n - a $, daí que o novo \omterm{def:g4:fractions:def}{numerador}
        satisfaz
        \[
        a \times n - b < a .
        \]
        Explique por que isso prova que o método sempre para.
  \item Uma velha história: um pai morre indo embora $17$ camelos, dispostos
        $\frac12$ do rebanho para seu filho mais velho, $\frac13$ para
        o segundo, $\frac19$ para o terceiro --- e nenhum dos
        ações é um número inteiro de camelos. Um vizinho empresta o
        família, um décimo oitavo camelo; as crianças levam $9$, $6$ e
        $2$ camelos, e o camelo emprestado é devolvido. Calcular
        $\frac12 + \frac13 + \frac19$ e explique o truque: por que
        o rebanho poderia ser dividido em camelos inteiros, e por que não
        criança recebe menos do que o testamento prometido?
  \item Reparar o testamento: propor três \emph{distinto} unidade
        frações que somam exatamente $1$, e dê o
        menor tamanho de rebanho para o qual todas as três partes são inteiras
        número de camelos.
\end{enumerate}
\end{problem}
