\chapter{Pirâmides e Cones}\label{ch:g8:solids}

Depois do \omterm{def:g7:areas:prism}{prismas} e \omterm{def:g7:areas:prism}{cilindros} de \cref{ch:g7:areas} --- \omterm{def:g5:solids:def}{sólidos} com
seção transversal constante --- vem o \omterm{def:g5:solids:def}{sólidos} com um ponto: \omterm{def:g8:solids:def}{pirâmides} e
\omterm{def:g8:solids:def}{cones}. Deles \omterm{def:g6:measure:volume}{volume} fórmula carrega um fator famoso $\frac13$, que
este capítulo torna plausível e usa; o estudo de \omterm{def:g5:solids:def}{sólidos} continua
com esferas e seções em \cref{ch:g9:solids}.

\section{Descrevendo pirâmides e cones}

\begin{definition}[Pirâmide, cone]\label{def:g8:solids:def}
\begin{itemize}
  \item UM \emph{pirâmide}\index{pirâmide} tem um \omterm{def:g4:geometry:polygon}{polígono} como seu
        \emph{\omterm{def:g8:powers:def}{base}} e um ponto, o \emph{ápice}, unido a cada
        vértice da \omterm{def:g8:powers:def}{base} por triangular \omterm{def:g5:solids:def}{rostos}. Isso é \emph{altura} é
        a distância do ápice ao plano da \omterm{def:g8:powers:def}{base}.
  \item UM \emph{cone}\index{cone} é a mesma construção sobre um
        disco: um disco \omterm{def:g8:powers:def}{base}, um ápice e uma superfície lateral curva.
\end{itemize}
Uma pirâmide é \emph{regular} quando sua \omterm{def:g8:powers:def}{base} é regular \omterm{def:g4:geometry:polygon}{polígono}
(quadrado, triângulo \omterm{def:g6:shapes:triangles}{equilátero}, \dots) e seu ápice fica verticalmente
acima do centro da \omterm{def:g8:powers:def}{base}.
\end{definition}

\begin{example}[Contando faces, arestas, vértices]\label{ex:g8:solids:count}
UM \omterm{def:g8:solids:def}{pirâmide} com \omterm{def:g8:powers:def}{base} quadrada tem $5$ \omterm{def:g5:solids:def}{rostos} ($1$ quadrado $+ 4$
triângulos), $8$ \omterm{def:g5:solids:def}{bordas} e $5$ \omterm{def:g5:solids:def}{vértices}. Com um $ n $ \omterm{def:g8:powers:def}{base} lateral:
$ n + 1$ \omterm{def:g5:solids:def}{rostos}, $2n $ \omterm{def:g5:solids:def}{bordas}, $ n + 1$ \omterm{def:g5:solids:def}{vértices}. Verifique o pequeno
padrão: \omterm{def:g5:solids:def}{rostos} $+$ \omterm{def:g5:solids:def}{vértices} $=$ \omterm{def:g5:solids:def}{bordas} $+ 2$.
\end{example}

\section{Volume}

\begin{theorem}[Volume de uma pirâmide ou cone]\label{thm:g8:solids:volume}
Por um \omterm{def:g8:solids:def}{pirâmide} ou um \omterm{def:g8:solids:def}{cone} com \omterm{def:g8:powers:def}{base} \omterm{def:g6:measure:area}{área} $ B $ e altura $ h $:
\[
V = \frac{1}{3}\, B \times h
\]
--- um terço do \omterm{def:g7:areas:prism}{prisma} ou \omterm{def:g7:areas:prism}{cilindro} com a mesma \omterm{def:g8:powers:def}{base} e altura.
Por um \omterm{def:g8:solids:def}{cone} de \omterm{def:g8:powers:def}{base} \omterm{def:g3:shapes:shapes}{raio} $ r $: $ V = \frac13 \pi r^2 h $.
\end{theorem}
\admitted

\begin{remark}[Por que um terço?]
Preencha um buraco \omterm{def:g8:solids:def}{pirâmide} com água e despeje no \omterm{def:g7:areas:prism}{prisma} do mesmo
\omterm{def:g8:powers:def}{base} e altura: são necessários exatamente três preenchimentos. Melhor ainda: um \omterm{def:g5:solids:def}{cubo}
pode ser cortado em três idênticos \omterm{def:g8:solids:def}{pirâmides}, cada um tendo um face do
\omterm{def:g5:solids:def}{cubo} como \omterm{def:g8:powers:def}{base} e um vértice do \omterm{def:g5:solids:def}{cubo} como ápice (tente imaginar!).
Uma prova geral utiliza integração, ferramenta do \omterm{def:g6:measure:volume}{volume} do Ensino Médio.
\end{remark}

\begin{example}\label{ex:g8:solids:volume}
UM \omterm{def:g8:solids:def}{pirâmide} tem uma \omterm{def:g8:powers:def}{base} quadrada de lado $5$ cm e altura $9$ cm:
\begin{enumerate}
  \item \omterm{def:g8:powers:def}{base} \omterm{def:g6:measure:area}{área}: $ B = 5^2 = 25$ cm $^2$;
  \item \omterm{def:g6:measure:volume}{volume}: $ V = \frac13 \times 25 \times 9 = 25 \times 3 = 75$
        cm $^3$.
\end{enumerate}
Um sorvete \omterm{def:g8:solids:def}{cone} de \omterm{def:g3:shapes:shapes}{raio} $3$ cm e altura $10$ cm:
\[
V = \frac13 \times \pi \times 3^2 \times 10 = 30\pi \approx 94
\text{ cm}^3 .
\]
\end{example}

\begin{example}[Cuidado com a altura]\label{ex:g8:solids:slant}
Por um \omterm{def:g8:solids:def}{cone}, não confunda a altura $ h $ (ápice ao centro do
básico, \omterm{def:g6:lines:perp}{perpendicular}) com o \emph{altura inclinada} $ s $ (ápice para o
borda). Eles estão relacionados por Pitágoras
(\cref{thm:g8:pythagoras:direct}) no triângulo retângulo
ápice - centro - borda:
\[
s^2 = h^2 + r^2 .
\]
UM \omterm{def:g8:solids:def}{cone} com $ r = 3$ e $ s = 5$ portanto tem altura
$ h = \sqrt{25 - 9} = 4$, e \omterm{def:g6:measure:volume}{volume}
$\frac13 \pi \times 9 \times 4 = 12\pi $.
\end{example}

\begin{method}[Problemas de volume]\label{met:g8:solids:method}
\begin{enumerate}
  \item Identifique o \omterm{def:g5:solids:def}{sólido} (\omterm{def:g7:areas:prism}{prisma}/\omterm{def:g7:areas:prism}{cilindro}: $ V = Bh $; \omterm{def:g8:solids:def}{pirâmide}/\omterm{def:g8:solids:def}{cone}:
        $ V = \frac13 Bh $);
  \item calcular a \omterm{def:g8:powers:def}{base} \omterm{def:g6:measure:area}{área} $ B $ primeiro, como uma etapa separada;
  \item verifique se a altura é \omterm{def:g6:lines:perp}{perpendicular} para a \omterm{def:g8:powers:def}{base} (use
        Pitágoras se a altura inclinada for dada);
  \item multiplique, mantenha as unidades consistentes, converta no final, se necessário
        ($1$ eu $= 1000$ cm $^3$).
\end{enumerate}
\end{method}

\section{Redes}

\begin{example}[Rede de uma pirâmide]\label{ex:g8:solids:net}
Desdobrando uma \omterm{def:g8:powers:def}{base} quadrada regular \omterm{def:g8:solids:def}{pirâmide} achata-lo em seu
\emph{\omterm{def:g5:solids:net}{líquido}}: a \omterm{def:g8:powers:def}{base} quadrada com quatro idênticos triângulos \omterm{def:g6:shapes:triangles}{isósceles}
preso às suas laterais. Os lados iguais dos triângulos são os
\emph{\omterm{def:g5:solids:def}{bordas} laterais} do \omterm{def:g8:solids:def}{pirâmide} --- seu comprimento não é nenhum dos dois $ h $
nem a altura inclinada de um face, então etiquete com cuidado antes de cortar
papelão.
\end{example}

\section{Problema: Um cubo cortado em três pirâmides}

\begin{problem}[{Problema de fim de semana --- onde o fator $\frac13$
de onde vem, e o que acontece com uma pirâmide cortada na meia altura}]
\label{pb:g8:solids:1}
A observação depois \cref{thm:g8:solids:volume} afirma que um \omterm{def:g5:solids:def}{cubo}
pode ser cortado em três idênticos \omterm{def:g8:solids:def}{pirâmides} --- ``tente imaginar
isso!''. Este problema retrata com precisão, obtém o famoso fator
$\frac13$ fora disso honestamente, então corta o \omterm{def:g5:solids:def}{cubo} um segundo caminho para
\emph{seis} \omterm{def:g8:solids:def}{pirâmides}, e termina fatiando um \omterm{def:g8:solids:def}{pirâmide} no \omterm{def:g3:division:half}{metade} é
altura --- com um resultado que poucas pessoas acertam na primeira tentativa.

Por todo, $ ABCDEFGH $ é um \omterm{def:g5:solids:def}{cubo} de lado $ a $: básico $ ABCD $, principal
face $ EFGH $, com $ E $ acima $ A $, $ F $ acima $ B $, $ G $ acima $ C $ e
$ H $ acima $ D $.

\medskip
\noindent\textbf{Part I --- Three \omterm{def:g8:solids:def}{pirâmides} sharing an apex.}
\begin{enumerate}
  \item O vértice $ G $ pertence a três \omterm{def:g5:solids:def}{rostos} do \omterm{def:g5:solids:def}{cubo}. Lista
        os três \omterm{def:g5:solids:def}{rostos} isso faz \emph{não} conter $ G $ e
        descreva os três \omterm{def:g8:solids:def}{pirâmides} obtido tomando cada um
        eles como \omterm{def:g8:powers:def}{base}, com ápice $ G $ toda vez. Quantos \omterm{def:g5:solids:def}{rostos},
        \omterm{def:g5:solids:def}{bordas} e \omterm{def:g5:solids:def}{vértices} tem cada \omterm{def:g8:solids:def}{pirâmide}
        (\cref{ex:g8:solids:count})?
  \item Explique por que virar o \omterm{def:g5:solids:def}{cubo} por um terço de volta
        sua longa diagonal $(AG)$ deixa o \omterm{def:g5:solids:def}{cubo} inalterado, mas
        embaralha as três \omterm{def:g8:powers:def}{bases} da questão 1 em um ciclo ---
        e por que os três \omterm{def:g8:solids:def}{pirâmides} são, portanto, idênticos
        cópias um do outro.
  \item Admitindo que os três \omterm{def:g8:solids:def}{pirâmides} preencha o \omterm{def:g5:solids:def}{cubo} sem
        sobreposição (um modelo de papelão é bastante convincente),
        deduzir o \omterm{def:g6:measure:volume}{volume} de cada um.
  \item Compare isso com a fórmula de
        \cref{thm:g8:solids:volume}: o \omterm{def:g8:solids:def}{pirâmide} com \omterm{def:g8:powers:def}{base} $ ABCD $
        e ápice $ G $ tem seu ápice diretamente acima de um \emph{canto}
        de sua \omterm{def:g8:powers:def}{base}, exatamente como em \cref{exo:g8:solids:11}.
        Identifique sua altura, aplique a fórmula e compare.
  \item Por um \omterm{def:g5:solids:def}{cubo} de lado $6$ cm, dê o \omterm{def:g6:measure:volume}{volume} de cada um dos
        três \omterm{def:g8:solids:def}{pirâmides} --- e explique o que é essa decomposição
        tem a ver com a experiência de derramar água do
        capítulo (três preenchimentos do \omterm{def:g8:solids:def}{pirâmide} por um \omterm{def:g7:areas:prism}{prisma}).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Six \omterm{def:g8:solids:def}{pirâmides} sharing the center.}
Agora deixe $ O $ ser o centro do cuboe junte-se $ O $ para os quatro
\omterm{def:g5:solids:def}{vértices} de cada um dos seis \omterm{def:g5:solids:def}{rostos}.
\begin{enumerate}[resume]
  \item Descreva os seis \omterm{def:g8:solids:def}{pirâmides} obtidos e explique por que eles
        são cópias idênticas um do outro. Qual é a altura
        de cada um?
  \item Deduza, compartilhando o cubode \omterm{def:g6:measure:volume}{volume}, o \omterm{def:g6:measure:volume}{volume} de cada
        dos seis \omterm{def:g8:solids:def}{pirâmides}.
  \item Recalcular isso \omterm{def:g6:measure:volume}{volume} com a fórmula
        $ V = \frac13 B h $ e verifique se as duas respostas concordam.
  \item As duas decomposições confirmam o fator $\frac13$ para
        alguns muito especiais \omterm{def:g8:solids:def}{pirâmides}. Explique por que eles ainda não
        \emph{provar} \cref{thm:g8:solids:volume} para cada
        \omterm{def:g8:solids:def}{pirâmide} e \omterm{def:g8:solids:def}{cone} --- e diga como o capítulo trata
        essa lacuna: qual afirmação é admitida, qual experimento
        apoia, e em que \omterm{def:g6:measure:volume}{volume} da série é isso
        honestamente provado.
  \item UM \omterm{def:g5:solids:def}{cubo} de lado $6$ cm é cortado em seu centro seis
        \omterm{def:g8:solids:def}{pirâmides}. Calcule o \omterm{def:g6:measure:volume}{volume} de cada um, nos dois sentidos.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Cutting a \omterm{def:g8:solids:def}{pirâmide} at \omterm{def:g3:division:half}{metade} height.}
$ SABCD $ é um quadrado regular baseado \omterm{def:g8:solids:def}{pirâmide}: lado da \omterm{def:g8:powers:def}{base} $ c $, altura
$ h $, ápice $ S $ acima do centro da \omterm{def:g8:powers:def}{base}. Corte pelo avião
através do \emph{pontos médios} das quatro \omterm{def:g5:solids:def}{bordas} laterais $[SA]$,
$[SB]$, $[SC]$, $[SD]$.
\begin{enumerate}[resume]
  \item Aplique o teorema do ponto médio
        (\cref{thm:g8:midpoints:direct}) no triângulo $ SAB $:
        qual é a direção e comprimento do \omterm{def:g6:lines:objects}{segmento} juntando-se
        os pontos médios de $[SA]$ e $[SB]$? Deduza que o corte
        é um quadrado de lado $\frac{c}{2}$, \omterm{def:g6:lines:perp}{paralelo} para a \omterm{def:g8:powers:def}{base},
        em altura $\frac{h}{2}$.
  \item A peça acima do corte é em si uma \omterm{def:g8:powers:def}{base} quadrada \omterm{def:g8:solids:def}{pirâmide}.
        Dê o lado e a altura da \omterm{def:g8:powers:def}{base} e mostre que \omterm{def:g6:measure:volume}{volume}
        é exatamente \emph{um oitavo} do original \omterm{def:g6:measure:volume}{volume} $ V $.
  \item Deduza o \omterm{def:g6:measure:volume}{volume} da peça inferior (o \emph{tronco},
        a forma de um inacabado \omterm{def:g8:solids:def}{pirâmide}). Em que proporção
        o corte de meia altura compartilha o \omterm{def:g6:measure:volume}{volume}?
  \item O Louvre \omterm{def:g8:solids:def}{pirâmide} (\cref{exo:g8:solids:7}: lado da \omterm{def:g8:powers:def}{base}
        $35$ m, altura $22$ m) é limpo em duas campanhas: a
        vidro acima \omterm{def:g3:division:half}{metade} altura, depois o vidro abaixo. O que
        \omterm{def:g6:fractions:def}{fração} do \emph{\omterm{def:g6:measure:volume}{volume}} senta abaixo \omterm{def:g3:division:half}{metade} altura?
        Estime-o em metros cúbicos, com aproximação à centena mais próxima.
  \item A moral geral: se \emph{todos} as dimensões de um
        \omterm{def:g8:solids:def}{pirâmide} são multiplicados por $2$, o que acontece com seu
        \omterm{def:g6:measure:volume}{volume}? Por qual fator o \omterm{def:g6:measure:volume}{volume} crescer quando cada
        dimensão é multiplicada por $ k$ --- e como isso
        explicar, em uma \omterm{def:g6:lines:objects}{linha}, o ``um oitavo'' da questão 12?
