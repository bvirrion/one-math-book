\chapter{Pontos Médios e Paralelos}\label{ch:g8:midpoints}

Una os pontos médios de dois lados de um triângulo: o \omterm{def:g6:lines:objects}{segmento} você consegue é
sempre \omterm{def:g6:lines:perp}{paralelo} para o terceiro lado, e exatamente \omterm{def:g3:division:half}{metade} tanto tempo. Isto
O “teorema do ponto médio” e seu \omterm{def:g8:fractions:inverse}{inverso} são a primeira amostra de uma grande
ideia --- \omterm{def:g6:lines:perp}{paralelos} corte comprimentos \omterm{def:g6:propdata:def}{proporcionalmente} --- que floresce em
Teorema de Tales em \cref{ch:g9:thales}.

\section{O teorema do ponto médio}

\begin{theorem}[Teorema do ponto médio]\label{thm:g8:midpoints:direct}
Em um triângulo $ABC $, deixar $ I $ ser o ponto médio de $[AB]$ e $ J $ o
ponto médio de $[AC]$. Então a \omterm{def:g6:lines:objects}{linha} $(IJ)$ é \omterm{def:g6:lines:perp}{paralelo} para $(BC)$, e
\[
IJ = \frac{BC}{2} .
\]
\end{theorem}

\begin{proof}[Ideia de prova]
Deixe $ K $ seja o \omterm{def:g7:central:def}{simétrico} de $ I $ sobre $ J $ (meia volta,
\cref{ch:g7:central}). O \omterm{def:g4:geometry:polygon}{quadrilátero} $ AICK $ tem diagonais $[IK]$
e $[AC]$ cruzando em seu ponto médio comum $ J $: é um
\omterm{def:g7:central:parallelogram}{paralelogramo}, então $ KC $ é \omterm{def:g6:lines:perp}{paralelo} para $ AI $, ou seja,\ para $(AB)$, e
$ KC = AI = IB $. Então $ IBCK $ tem dois lados opostos ($[IB]$ e
$[KC]$) \omterm{def:g6:lines:perp}{paralelo} e igual: é um \omterm{def:g7:central:parallelogram}{paralelogramo} também
(\cref{thm:g7:central:recognize}), então $(IK) \parallel (BC)$ ---
aquilo é $(IJ) \parallel (BC)$ --- e $ BC = IK = 2\,IJ $.
\end{proof}

\begin{example}\label{ex:g8:midpoints:direct}
Em um triângulo $ ABC $ com $ BC = 9$ cm, os pontos médios $ I $ de $[AB]$ e
$ J $ de $[AC]$ estão unidos. Sem qualquer medição:
$(IJ) \parallel (BC)$ e $ IJ = \frac92 = 4.5$ cm. O pequeno triângulo
$ AIJ $ é uma cópia em tamanho médio de $ ABC $ --- isso é \omterm{def:g6:measure:perimeter}{perímetro} é \omterm{def:g3:division:half}{metade} o
\omterm{def:g6:measure:perimeter}{perímetro} de $ ABC $ também.
\end{example}

\begin{theorem}[Conversar]\label{thm:g8:midpoints:converse}
Em um triângulo $ ABC $, a \omterm{def:g6:lines:objects}{linha} que passa pelo ponto médio $ I $ de $[AB]$ e
\emph{\omterm{def:g6:lines:perp}{paralelo}} para $(BC)$ atravessa o lado $[AC]$ em seu ponto médio.
\end{theorem}
\admitted

\begin{example}\label{ex:g8:midpoints:converse}
Um caminho atravessa um campo triangular, começando no meio de uma aresta
e correndo \omterm{def:g6:lines:perp}{paralelo} para a borda oposta. O \omterm{def:g8:fractions:inverse}{inverso} garante
que o caminho sai exatamente no meio da outra aresta --- não
medição necessária no outro lado.
\end{example}

\begin{method}[Escolhendo entre teorema e inverso]\label{met:g8:midpoints:choose}
\begin{enumerate}
  \item \emph{Dois pontos médios conhecidos} $\Rightarrow $ use o direto
        teorema: conclua paralelismo e meio comprimento.
  \item \emph{Um ponto médio e um \omterm{def:g6:lines:perp}{paralelo} conhecido} $\Rightarrow $ usar
        o \omterm{def:g8:fractions:inverse}{inverso}: conclua que o ponto de cruzamento é um ponto médio.
  \item Escreva qual triângulo, quais pontos médios e qual teorema você
        use --- uma frase cada.
\end{enumerate}
\end{method}

\section{Triângulos de meio tamanho}

\begin{proposition}[O triângulo do ponto médio]\label{prop:g8:midpoints:medial}
Unir os três pontos médios dos lados de um triângulo corta-o em
quatro triângulos iguais \omterm{def:g6:measure:area}{áreas}, cada uma uma cópia com \omterm{def:g3:division:half}{metade} do tamanho do original.
\end{proposition}

\begin{proof}
Cada lado do triângulo do ponto médio é \omterm{def:g6:lines:perp}{paralelo} para um lado de $ ABC $ e
\omterm{def:g3:division:half}{metade} desde que (\cref{thm:g8:midpoints:direct}, aplicado três vezes).
Os quatro pequenos triângulos têm lados com os mesmos três comprimentos, então
são cópias idênticas; juntos eles ladrilham $ ABC $, então cada um tem um
\omterm{def:g3:division:half}{trimestre} do seu \omterm{def:g6:measure:area}{área}.
\end{proof}

\begin{remark}[Em direção a Tales]
O teorema do ponto médio é o caso ``um \omterm{def:g3:division:half}{metade}'' de um fato mais geral:
uma \omterm{def:g6:lines:objects}{linha} \omterm{def:g6:lines:perp}{paralelo} a um lado de um triângulo corta os outros dois lados
\emph{proporções iguais} --- um terço e um terço, dois quintos e dois
quintos\,\dots\ Essa afirmação geral é o teorema de Tales
(\cref{ch:g9:thales}); o caso intermediário é o único necessário para isso
ano.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g8:midpoints:1}
Em um triângulo $ RST $, $ M $ é o ponto médio de $[RS]$ e $ N $ o
ponto médio de $[RT]$, com $ ST = 7$ cm. O que pode ser dito da \omterm{def:g6:lines:objects}{linha}
$(MN)$ e o comprimento $ MN $? Cite o teorema utilizado.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:midpoints:2}
Em um triângulo $ ABC $, $ I $ é o ponto médio de $[AB]$ e a \omterm{def:g6:lines:objects}{linha}
através $ I $ \omterm{def:g6:lines:perp}{paralelo} para $(BC)$ cortes $[AC]$ no $ J $, com
$ AC = 11$ cm. Calcular $ AJ $, citando o teorema utilizado.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:midpoints:3}
Desenhe um triângulo $ ABC $ com $ AB = 8$ cm, $ AC = 6$ cm, $ BC = 7$ cm,
coloque os pontos médios $ I $ de $[AB]$ e $ J $ de $[AC]$ e medir $ IJ $
para verificar o teorema. Calcule o \omterm{def:g6:measure:perimeter}{perímetro} de $ AIJ $ sem
medindo qualquer outra coisa.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:midpoints:4}
$ IJK $ é o triângulo do ponto médio de $ ABC $, cujos lados medem $10$,
$12$ e $16$ cm. Dê os três lados de $ IJK $ e seu \omterm{def:g6:measure:perimeter}{perímetro}.
Como é que os dois \omterm{def:g6:measure:perimeter}{perímetros} comparar?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g8:midpoints:5}
O \omterm{def:g6:measure:area}{área} de um triângulo é $36$ cm $^2$. Qual é o \omterm{def:g6:measure:area}{área} do seu
triângulo de ponto médio? De cada um dos quatro pequenos triângulos?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:midpoints:6}
Em um triângulo $ ABC $, $ I $ é o ponto médio de $[AB]$, $ J $ o de
$[AC]$ e $ K $ o de $[BC]$. Mostre isso $ IJKB $ é um \omterm{def:g7:central:parallelogram}{paralelogramo}.
(Comparar $[IJ]$ e $[BK]$: \omterm{def:g6:lines:perp}{paralelo}? igual?)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:midpoints:7}
Uma vela triangular tem sua borda mais baixa $6.4$ m longo. Um reforço
costura une os pontos médios das outras duas \omterm{def:g5:solids:def}{bordas}. Qual comprimento da costura
é necessário? E se a costura unir os pontos localizados em um
\omterm{def:g3:division:half}{trimestre} de cada aresta, começando do topo vértice? (Conjectura com
um desenho; a prova é a de Tales, \cref{ch:g9:thales}.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g8:midpoints:8}
Em triângulo $ ABC $, \omterm{def:g6:shapes:triangles}{em ângulo reto} no $ A $, $ M $ é o ponto médio do
hipotenusa $[BC]$, e o \omterm{def:g6:lines:perp}{paralelo} para $(AB)$ através $ M $ conhece
$[AC]$ no $ N $.
\begin{enumerate}
  \item Mostre isso $ N $ é o ponto médio de $[AC]$.
  \item Explique por que $(MN)$ é \omterm{def:g6:lines:perp}{perpendicular} para $(AC)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g8:midpoints:9}
Deixar $ ABCD $ ser \emph{qualquer} \omterm{def:g4:geometry:polygon}{quadrilátero}, e $ I $, $ J $, $ K $, $ L $ o
pontos médios de $[AB]$, $[BC]$, $[CD]$, $[DA]$ (a conjectura de
\cref{exo:g7:central:11}).
\begin{enumerate}
  \item Aplique o teorema do ponto médio no triângulo $ ABC $ para o
        \omterm{def:g6:lines:objects}{segmento} $[IJ]$, e no triângulo $ ACD $ para o \omterm{def:g6:lines:objects}{segmento}
        $[LK]$: compare cada um com a diagonal $[AC]$.
  \item Conclua isso $ IJKL $ é sempre um \omterm{def:g7:central:parallelogram}{paralelogramo}.
\end{enumerate}
\end{exercise}

\section{Problema: As três medianas e o centro de gravidade}

\begin{problem}[{Problema de fim de semana --- as medianas de um triângulo se encontram
em um único ponto, dois terços do caminho de cada um deles}]
\label{pb:g8:midpoints:1}
Corte um triângulo de papelão rígido e ele ficará equilibrado,
perfeitamente plana, na ponta de um lápis colocada em um ponto especial: o
\emph{centro de gravidade}\index{centro de gravidade} do triângulo.
Este problema encontra esse ponto. UM \emph{mediana}\index{mediana} de um
triângulo é o \omterm{def:g6:lines:objects}{segmento} juntando-se a um vértice para o \emph{ponto médio} de
o lado oposto. Você provará --- com o teorema do ponto médio
usado duas vezes, de uma forma inesperada --- que todas as três medianas
passe por um ponto e localize-o com precisão.

\medskip
\noindent\textbf{Part I --- Warm-up.}
\begin{enumerate}
  \item Desenhe um grande triângulo $ ABC $ (sem forma especial), construir
        os pontos médios $ I $ de $[AB]$, $ J $ de $[AC]$ e $ K $ de
        $[BC]$, e desenhe as três medianas $[AK]$, $[BJ]$, $[CI]$.
        O que você observa?
  \item Prove que uma mediana corta o triângulo em dois triângulos
        de igual \omterm{def:g6:measure:area}{áreas}. (Compare \omterm{def:g8:powers:def}{bases} e alturas e use o
        \omterm{def:g6:measure:area}{área} fórmula: \omterm{def:g8:powers:def}{base} $\times $ altura $\div\ 2$.) Por um
        triângulo de \omterm{def:g6:measure:area}{área} $36$ cm $^2$, quais são as duas peças?
        \omterm{def:g6:measure:area}{áreas}?
  \item No triângulo $ ABC $, o que faz
        \cref{thm:g8:midpoints:direct} dizer sobre o \omterm{def:g6:lines:objects}{segmento}
        $[IJ]$?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Two medians meet two thirds of the
way.}
As medianas $[BJ]$ e $[CI]$ cruzar em um ponto; chame isso $ G $. Deixar
$ M $ ser o ponto médio de $[GB]$ e $ N $ o ponto médio de $[GC]$.
\begin{enumerate}[resume]
  \item Aplique o teorema do ponto médio \emph{no triângulo $ GBC $}:
        o que diz sobre o \omterm{def:g6:lines:objects}{segmento} $[MN]$?
  \item Deduza isso $(IJ) \parallel (MN)$ e $ IJ = MN $ e
        concluir com \cref{thm:g7:central:recognize} que $ IJNM $
        é um \omterm{def:g7:central:parallelogram}{paralelogramo}.
  \item Explique por que $ I $ e $ N $ ambos estão na mediana $(CI)$,
        e $ J $ e $ M $ ambos na mediana $(BJ)$ --- então o
        diagonais do \omterm{def:g7:central:parallelogram}{paralelogramo} $ IJNM $ são $[IN]$ e
        $[JM]$, e eles se cruzam exatamente em $ G $. O que o
        \emph{definição} de um \omterm{def:g7:central:parallelogram}{paralelogramo} então diga sobre $ G $?
  \item Deduza isso $ BM = MG = GJ $ e conclua:
        \[
        BG = 2 \times GJ
        \]
        --- o ponto $ G $ senta-se na mediana $[BJ]$ em dois terços
        do caminho do vértice $ B $. Indique e justifique a
        resultado análogo na mediana $[CI]$.
  \item Agora esqueça $[CI]$ e execute o mesmo argumento no
        \omterm{def:g3:numbers:evenodd}{par} de medianas $[BJ]$ e $[AK]$: seu ponto de passagem
        também fica em dois terços do $[BJ]$ de $ B $. Explique por que
        isso obriga a ser o \emph{mesmo} apontar $ G $ --- e
        concluir que todas as três medianas passam por $ G $.
  \item Em um triângulo onde a mediana $[BJ]$ medidas $9$ cm,
        quão longe é $ G $ de $ B $, e de $ J $?
  \item Justifique (uma frase) que $ AG = 2 \times GK $ também.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Six equal slices, and coordinates.}
As três medianas cortam o triângulo em seis pequenos triângulos
ao redor $ G $.
\begin{enumerate}[resume]
  \item No triângulo $ GBC $, o \omterm{def:g6:lines:objects}{segmento} $[GK]$ é uma mediana.
        Deduza que os dois pequenos triângulos $ GBK $ e $ GKC $ tem
        igual \omterm{def:g6:measure:area}{áreas}.
  \item Usando a questão 2 nos triângulos $ ABK $ e $ AKC $ (ambos
        cortado pela mediana $[AK]$ de $ ABC $), mostre que
        triângulos $ ABG $ e $ ACG $ tem igual \omterm{def:g6:measure:area}{áreas}. Repetindo
        com outra mediana, conclua que os três triângulos
        $ ABG $, $ BCG $, $ CAG $ todos têm \omterm{def:g6:measure:area}{área} um terço de $ ABC $.
  \item Cada um desses três triângulos é cortado em \omterm{def:g3:division:half}{metade} por um pedaço
        da mediana (por exemplo $[GI]$ é uma mediana do
        triângulo $ ABG $ --- visto de onde vértice?). Conclua: os seis
        pequenas fatias ao redor $ G $ todos têm o mesmo \omterm{def:g6:measure:area}{área}, um sexto
        do triângulo.
  \item Em uma grade de coordenadas, plote $ A(0, 0)$, $ B(6, 0)$ e
        $ C(0, 6)$, com $ K(3, 3)$ o ponto médio de $[BC]$. Use o
        teorema dos dois terços da Parte II para calcular as coordenadas
        de $ G $ na mediana $[AK]$; então verifique se é o mesmo
        ponto fica em dois terços da mediana $[BJ]$, onde
        $ J(0, 3)$. Comparar $ G $ coordenadas com as médias
        \[
        \frac{0 + 6 + 0}{3},
        \qquad
        \frac{0 + 0 + 6}{3} .
        \]
  \item O balanço final: usando as questões 2 e 8, explique por quê
        o triângulo de papelão se equilibra em um fio de faca colocado
        ao longo de qualquer mediana --- quantidades iguais de papelão em cada
        lado --- e por que a ponta do lápis deve ser colocada
        em $ G$, o ponto que os físicos chamam de \omterm{pb:g8:midpoints:1}{centro de
        gravidade}. (Uma prova de equilíbrio totalmente rigorosa precisa do
        cálculo integral da universidade \omterm{def:g6:measure:volume}{volumes}; o igual
        \omterm{def:g6:measure:area}{áreas} torná-lo crível hoje.)
\end{enumerate}
\end{problem}
